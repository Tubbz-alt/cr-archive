 25 package test.com.sun.javafx.iio;
 26 
 27 import javafx.application.Application;
 28 import javafx.application.Platform;
 29 import javafx.scene.image.Image;
 30 import javafx.scene.image.ImageView;
 31 import javafx.scene.layout.HBox;
 32 import javafx.scene.Scene;
 33 import javafx.stage.Stage;
 34 
 35 import java.io.FileInputStream;
 36 import java.io.FileNotFoundException;
 37 import java.net.URL;
 38 import java.util.concurrent.CountDownLatch;
 39 import java.util.concurrent.TimeUnit;
 40 
 41 import org.junit.After;
 42 import org.junit.AfterClass;
 43 import org.junit.BeforeClass;
 44 import org.junit.Test;
 45 import static org.junit.Assert.fail;
 46 import static org.junit.Assert.assertTrue;
 47 
 48 import test.util.Util;
 49 
 50 public class LoadCorruptJPEGTest {
 51     static CountDownLatch startupLatch;
 52     static HBox root;
 53     static volatile Scene scene;
 54     static volatile Stage stage;
 55     static final int SCENE_WIDTH = 200;
 56     static final int SCENE_HEIGHT = 200;
 57 
 58     @Test
 59     public void testCorruptJPEGImage() {
 60         Util.runAndWait(() -> {
 61             URL resource = LoadCorruptJPEGTest.class.getResource("corrupt.jpg");
 62             FileInputStream input = null;
 63             try {
 64                 input = new FileInputStream(resource.getFile());
 65             } catch (FileNotFoundException e) {
 66                 fail("File not found: corrupt.jpg");
 67             }
 68             Image image = new Image(input);
 69             ImageView iv = new ImageView(image);
 70             root.getChildren().add(iv);
 71         });
 72     }
 73 
 74     public static class TestApp extends Application {
 75         @Override
 76         public void start(Stage primaryStage) {
 77             stage = primaryStage;
 78             root = new HBox();
 79             scene = new Scene(root, SCENE_WIDTH, SCENE_HEIGHT);
 80             stage.setScene(scene);
 81             stage.setOnShown(l -> {
 82                 Platform.runLater(() -> startupLatch.countDown());
 83             });
 84             stage.show();
 85         }
 86     }
 87 
 88     @BeforeClass
 89     public static void initFX() throws Exception {
 90         startupLatch = new CountDownLatch(1);
 91         new Thread(() -> Application.launch(TestApp.class, (String[])null)).start();
 92         assertTrue("Timeout waiting for FX runtime to start",
 93                 startupLatch.await(15, TimeUnit.SECONDS));
 94     }
 95 
 96     @AfterClass
 97     public static void exit() {
 98         Util.runAndWait(() -> {
 99             stage.hide();
100         });
101         Platform.exit();
102     }
103 
104     @After
105     public void resetTest() {
106         Util.runAndWait(() -> {
107             root.getChildren().clear();
108         });
109     }
110 }
