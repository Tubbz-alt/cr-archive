26 package com.sun.prism.sw;
27 
28 import com.sun.openpisces.AlphaConsumer;
29 import com.sun.openpisces.Renderer;
30 import com.sun.pisces.PiscesRenderer;
31 
32 final class DirectRTPiscesAlphaConsumer implements AlphaConsumer {
33 
34     private byte alpha_map[];
35     private int outpix_xmin;
36     private int outpix_ymin;
37     private int w;
38     private int h;
39     private int rowNum;
40 
41     private PiscesRenderer pr;
42 
43     void initConsumer(Renderer renderer, PiscesRenderer pr) {
44         outpix_xmin = renderer.getOutpixMinX();
45         outpix_ymin = renderer.getOutpixMinY();
46         w = renderer.getOutpixMaxX() - outpix_xmin;
47         if (w < 0) { w = 0; }
48         h = renderer.getOutpixMaxY() - outpix_ymin;
49         if (h < 0) { h = 0; }
50         rowNum = 0;
51         this.pr = pr;
52     }
53 
54     @Override
55     public int getOriginX() {
56         return outpix_xmin;
57     }
58 
59     @Override
60     public int getOriginY() {
61         return outpix_ymin;
62     }
63 
64     @Override
65     public int getWidth() {
66         return w;
67     }
68 
69     @Override
70     public int getHeight() {
71         return h;
72     }
73 
74     @Override
75     public void setMaxAlpha(int maxalpha) {
76         if ((alpha_map == null) || (alpha_map.length != maxalpha+1)) {
77             alpha_map = new byte[maxalpha+1];
78             for (int i = 0; i <= maxalpha; i++) {
79                 alpha_map[i] = (byte) ((i*255 + maxalpha/2)/maxalpha);
80             }
81         }
82     }
83 
84     @Override
85     public void setAndClearRelativeAlphas(int[] alphaDeltas, int pix_y, int firstdelta, int lastdelta) {
86         pr.emitAndClearAlphaRow(alpha_map, alphaDeltas, pix_y, firstdelta, lastdelta, rowNum);
87         rowNum++;
88     }
89 }
