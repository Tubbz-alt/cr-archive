 136 void destroy_and_delete_ctx(WindowContext *ctx) {
 137     if (ctx) {
 138         ctx->process_destroy();
 139 
 140         if (!ctx->get_events_count()) {
 141             delete ctx;
 142         }
 143         // else: ctx will be deleted in EventsCounterHelper after completing
 144         // an event processing
 145     }
 146 }

 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 174 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 175                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 176         screen(_screen),
 177         frame_type(_frame_type),
 178         window_type(type),
 179         owner(_owner),
 180         jview(NULL),
 181         map_received(false),
 182         visible_received(false),
 183         on_top(false),
 184         is_fullscreen(false),
 185         is_iconified(false),
 186         is_maximized(false),
 187         is_mouse_entered(false),
 188         can_be_deleted(false),
 189         events_processing_cnt(0),
 190         grab_pointer(NULL) {
 191 
 192     jwindow = mainEnv->NewGlobalRef(_jwindow);
 193 
 194     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 195 
 196     if (gchar * app_name = get_application_name()) {
 197         gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 198         g_free(app_name);
 199     }
 200 
 201     if (owner) {
 202         owner->add_child(this);
 203         if (on_top_inherited()) {
 204             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 205         }
 206     }
 207 
 208     if (type == UTILITY) {
 209         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 210     }
 211 
 212     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 213 
 214     if (xvisualID != 0) {
 215         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 216         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 217     }
 218 
 219     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 220     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 221 
 222     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 223     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 224 
 225     gdk_window = gtk_widget_get_window(gtk_widget);
 226     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 227 
 228     glass_dnd_attach_context(this);
 229 
 230     gdk_windowManagerFunctions = wmf;
 231     if (wmf) {
 232         gdk_window_set_functions(gdk_window, wmf);
 233     }
 234 
 235     if (frame_type != TITLED) {
 236         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 237     }
 238 
 239     connect_signals(gtk_widget, this);
 240 }

 242 void WindowContext::paint(void *data, jint width, jint height) {
 243 #if GTK_CHECK_VERSION(3, 0, 0)
 244     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 245 #if GTK_CHECK_VERSION(3, 22, 0)
 246     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 247     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 248 #else
 249     gdk_window_begin_paint_region(gdk_window, region);
 250     cairo_t* context = gdk_cairo_create(gdk_window);
 251 #endif
 252 #else
 253     cairo_t *context = gdk_cairo_create(gdk_window);
 254 #endif
 255 
 256     if (bg_color.is_set) {
 257         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 258                                 (frame_type == TRANSPARENT) ? 0 : 1);
 259         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 260         cairo_paint(context);
 261     }
 262 
 263     cairo_surface_t *cairo_surface;
 264     cairo_surface = cairo_image_surface_create_for_data(
 265             (unsigned char *) data,
 266             CAIRO_FORMAT_ARGB32,
 267             width, height, width * 4);
 268 
 269     cairo_set_source_surface(context, cairo_surface, 0, 0);
 270 
 271     applyShapeMask(data, width, height);
 272     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 273     cairo_paint(context);
 274 
 275 #if GTK_CHECK_VERSION(3, 0, 0)
 276 #if GTK_CHECK_VERSION(3, 22, 0)
 277     gdk_window_end_draw_frame(gdk_window, dcontext);
 278     cairo_region_destroy(region);
 279 #else
 280     gdk_window_end_paint(gdk_window);
 281     cairo_region_destroy(region);
 282     cairo_destroy(context);
 283 #endif
 284 #else
 285     cairo_destroy(context);
 286 #endif
 287 
 288     cairo_surface_destroy(cairo_surface);
 289 }

 331 void WindowContext::process_focus(GdkEventFocus *event) {
 332     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 333         ungrab_mouse_drag_focus();
 334     }
 335 
 336     if (!event->in && WindowContext::sm_grab_window == this) {
 337         ungrab_focus();
 338     }
 339 
 340     if (xim.enabled && xim.ic) {
 341         if (event->in) {
 342             XSetICFocus(xim.ic);
 343         } else {
 344             XUnsetICFocus(xim.ic);
 345         }
 346     }
 347 
 348     if (jwindow) {
 349         if (!event->in || isEnabled()) {
 350             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 351                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 352                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 353             CHECK_JNI_EXCEPTION(mainEnv)
 354         } else {
 355             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 356             CHECK_JNI_EXCEPTION(mainEnv)
 357         }
 358     }
 359 }

 361 void WindowContext::process_property_notify(GdkEventProperty *event) {
 362     if (event->window == gdk_window) {
 363         if (event->atom == atom_net_wm_state) {
 364             process_net_wm_property();
 365         } else if (event->atom == atom_net_wm_frame_extents) {
 366             if (frame_type != TITLED) {
 367                 return;
 368             }
 369 
 370             int top, left, bottom, right;
 371 
 372             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 373                 if (top + left + bottom + right > 0) {
 374                     geometry.frame_extents_received = true;
 375                     geometry.adjust_w = left + right;
 376                     geometry.adjust_h = top + bottom;
 377                     geometry.view_x = left;
 378                     geometry.view_y = top;
 379 
 380                     // set bounds again to set to correct window size that must
 381                     // be the total width and height accounting extents
 382                     // this is ignored if size is "content size" instead of "window size"
 383                     if (geometry.needs_ajustment) {
 384                         set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
 385                     }
 386 
 387                     // force position notify so java will know about view_y and view_x
 388                     size_position_notify(false, true);
 389                 }
 390             }
 391         }
 392     }
 393 }

 395 void WindowContext::process_configure() {
 396     gint x, y, w, h, gtk_w, gtk_h;
 397 
 398     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 399     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 400 
 401     w = gtk_w + geometry.adjust_w;
 402     h = gtk_h + geometry.adjust_h;
 403 
 404     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 405     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 406                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 407 
 408     geometry.current_x = x;
 409     geometry.current_y = y;
 410     geometry.current_w = w;
 411     geometry.current_h = h;
 412     geometry.current_cw = gtk_w;
 413     geometry.current_ch = gtk_h;
 414 
 415     size_position_notify(size_changed, pos_changed);
 416 }

 418 void WindowContext::process_destroy() {
 419     if (owner) {
 420         owner->remove_child(this);
 421     }
 422 
 423     if (WindowContext::sm_mouse_drag_window == this) {
 424         ungrab_mouse_drag_focus();
 425     }
 426 
 427     if (WindowContext::sm_grab_window == this) {
 428         ungrab_focus();
 429     }
 430 
 431     std::set<WindowContext *>::iterator it;
 432     for (it = children.begin(); it != children.end(); ++it) {
 433         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 434         // WindowContext::process_destroy() to call remove_child() (because children
 435         // is being iterated here) but also prevents gtk_window_set_transient_for from
 436         // being called - this causes the crash on gnome.
 437         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 438         (*it)->set_owner(NULL);
 439         destroy_and_delete_ctx(*it);
 440     }
 441     children.clear();
 442 
 443     if (jwindow) {
 444         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 445         EXCEPTION_OCCURED(mainEnv);
 446     }
 447 
 448     if (jview) {
 449         mainEnv->DeleteGlobalRef(jview);
 450         jview = NULL;
 451     }
 452 
 453     if (jwindow) {
 454         mainEnv->DeleteGlobalRef(jwindow);
 455         jwindow = NULL;
 456     }
 457 
 458     can_be_deleted = true;
 459 }

 461 void WindowContext::process_delete() {
 462     if (jwindow && isEnabled()) {
 463         gtk_widget_hide_on_delete(gtk_widget);
 464         mainEnv->CallVoidMethod(jwindow, jWindowNotifyClose);
 465         CHECK_JNI_EXCEPTION(mainEnv)
 466     }

 477 void WindowContext::process_mouse_button(GdkEventButton *event) {
 478     bool press = event->type == GDK_BUTTON_PRESS;
 479     bool release = event->type == GDK_BUTTON_RELEASE;
 480 
 481     guint state = event->state;
 482     guint mask = 0;
 483 
 484     // We need to add/remove current mouse button from the modifier flags
 485     // as X lib state represents the state just prior to the event and
 486     // glass needs the state just after the event
 487     switch (event->button) {
 488         case 1:
 489             mask = GDK_BUTTON1_MASK;
 490             break;
 491         case 2:
 492             mask = GDK_BUTTON2_MASK;
 493             break;
 494         case 3:
 495             mask = GDK_BUTTON3_MASK;
 496             break;
 497         case MOUSE_BACK_BTN:
 498             mask = GDK_BUTTON4_MASK;
 499             break;
 500         case MOUSE_FORWARD_BTN:
 501             mask = GDK_BUTTON5_MASK;
 502             break;
 503     }
 504 
 505     if (press) {
 506         state |= mask;
 507     } else {
 508         state &= ~mask;
 509     }
 510 
 511     // Upper layers expects from us Windows behavior:
 512     // all mouse events should be delivered to window where drag begins
 513     // and no exit/enter event should be reported during this drag.
 514     // We can grab mouse pointer for these needs.
 515     if (press) {
 516         grab_mouse_drag_focus(gdk_window, (GdkEvent *) event, NULL, true);
 517     } else {
 518         if ((event->state & MOUSE_BUTTONS_MASK)
 519             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 520             ungrab_mouse_drag_focus();
 521         } else if (event->button == 8 || event->button == 9) {
 522             // GDK X backend interprets button press events for buttons 4-7 as
 523             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 524             // be set on the event->state from GDK. Thus we cannot check if all
 525             // buttons have been released in the usual way (as above).
 526             ungrab_mouse_drag_focus();
 527         }
 528     }
 529 
 530     jint button = gtk_button_number_to_mouse_button(event->button);
 531 
 532     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 533         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 534                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 535                                 button,
 536                                 (jint) event->x, (jint) event->y,
 537                                 (jint) event->x_root, (jint) event->y_root,
 538                                 gdk_modifier_mask_to_glass(state),
 539                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 540                                 JNI_FALSE);
 541         CHECK_JNI_EXCEPTION(mainEnv)
 542 
 543         if (jview && event->button == 3 && press) {
 544             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 545                                     (jint) event->x, (jint) event->y,
 546                                     (jint) event->x_root, (jint) event->y_root,
 547                                     JNI_FALSE);
 548             CHECK_JNI_EXCEPTION(mainEnv)
 549         }
 550     }
 551 }

 553 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 554     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 555     jint isDrag = glass_modifier & (
 556             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 557             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 558             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 559             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 560             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 561     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 562 
 563     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 564         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 565     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 566         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 567     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 568         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 569     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 570         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 571     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 572         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 573     }
 574 
 575     if (jview) {
 576         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 577                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 578                                 button,
 579                                 (jint) event->x, (jint) event->y,
 580                                 (jint) event->x_root, (jint) event->y_root,
 581                                 glass_modifier,
 582                                 JNI_FALSE,
 583                                 JNI_FALSE);
 584         CHECK_JNI_EXCEPTION(mainEnv)
 585     }
 586 }

 588 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 589     jdouble dx = 0;
 590     jdouble dy = 0;
 591 
 592     // converting direction to change in pixels
 593     switch (event->direction) {
 594 #if GTK_CHECK_VERSION(3, 4, 0)
 595         case GDK_SCROLL_SMOOTH:

 655 void WindowContext::process_key(GdkEventKey *event) {
 656     bool press = event->type == GDK_KEY_PRESS;
 657     jint glassKey = get_glass_key(event);
 658     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 659     if (press) {
 660         glassModifier |= glass_key_to_modifier(glassKey);
 661     } else {
 662         glassModifier &= ~glass_key_to_modifier(glassKey);
 663     }
 664     jcharArray jChars = NULL;
 665     jchar key = gdk_keyval_to_unicode(event->keyval);
 666     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 667         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 668     } else {
 669 #ifdef GLASS_GTK2
 670         if (key == 0) {
 671             // Work around "bug" fixed in gtk-3.0:
 672             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 673             switch (event->keyval) {
 674             case 0xFF08 /* Backspace */: key =  '\b';
 675             case 0xFF09 /* Tab       */: key =  '\t';
 676             case 0xFF0A /* Linefeed  */: key =  '\n';
 677             case 0xFF0B /* Vert. Tab */: key =  '\v';
 678             case 0xFF0D /* Return    */: key =  '\r';
 679             case 0xFF1B /* Escape    */: key =  '\033';
 680             case 0xFFFF /* Delete    */: key =  '\177';
 681             }
 682         }
 683 #endif
 684     }
 685 
 686     if (key > 0) {
 687         jChars = mainEnv->NewCharArray(1);
 688         if (jChars) {
 689             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 690             CHECK_JNI_EXCEPTION(mainEnv)
 691         }
 692     } else {
 693         jChars = mainEnv->NewCharArray(0);
 694     }
 695     if (jview) {
 696         if (press) {
 697             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 698                                     com_sun_glass_events_KeyEvent_PRESS,
 699                                     glassKey,
 700                                     jChars,
 701                                     glassModifier);
 702             CHECK_JNI_EXCEPTION(mainEnv)
 703 
 704             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 705                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 706                                         com_sun_glass_events_KeyEvent_TYPED,
 707                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 708                                         jChars,
 709                                         glassModifier);
 710                 CHECK_JNI_EXCEPTION(mainEnv)
 711             }
 712         } else {
 713             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 714                                     com_sun_glass_events_KeyEvent_RELEASE,
 715                                     glassKey,
 716                                     jChars,
 717                                     glassModifier);
 718             CHECK_JNI_EXCEPTION(mainEnv)
 719         }
 720     }
 721 }

 723 void WindowContext::process_state(GdkEventWindowState *event) {
 724     if (event->changed_mask &
 725         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 726 
 727         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 728             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 729         }
 730         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 731             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 732         }
 733 
 734         jint stateChangeEvent;
 735 
 736         if (is_iconified) {
 737             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 738         } else if (is_maximized) {
 739             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 740         } else {
 741             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 742             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 743                 // in this case - the window manager will not support the programatic
 744                 // request to iconify - so we need to restore it now.
 745                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 746             }
 747         }
 748 
 749         notify_state(stateChangeEvent);
 750     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 751         notify_on_top(event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 752     }
 753 }

 755 void WindowContext::process_net_wm_property() {
 756     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 757 
 758     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 759     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 760     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 761     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 762 
 763     gint length;
 764     glong *atoms = NULL;
 765 
 766     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 767                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 768 
 769         bool is_hidden = false;
 770         bool is_above = false;
 771         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 772             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 773                 is_hidden = true;
 774             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 775                 is_above = true;
 776             }
 777         }
 778 
 779         g_free(atoms);
 780 
 781         if (is_iconified != is_hidden) {
 782             is_iconified = is_hidden;
 783 
 784             notify_state((is_hidden)
 785                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 786                          : com_sun_glass_events_WindowEvent_RESTORE);
 787         }
 788 
 789         notify_on_top(is_above);
 790     }
 791 }

 976 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 977     // this will tell if adjustments are needed - that's because GTK does not have full window size
 978     // values, just content values. Frame extents (window decorations) are handled by the window manager.
 979     geometry.needs_ajustment = (w > 0 || h > 0) || geometry.needs_ajustment;
 980 
 981     // newW / newH always content sizes compatible with GTK+
 982     // if window has no decoration, adjustments will be ZERO
 983     // this will always be content size
 984     int newW, newH;
 985     if (w > 0) {
 986         newW = w - geometry.adjust_w;
 987         geometry.current_w = newW;
 988     } else {
 989         newW = cw;
 990     }
 991 
 992     if (h > 0) {
 993         newH = h - geometry.adjust_h;
 994         geometry.current_h = newH;
 995     } else {
 996         newH = ch;
 997     }
 998 
 999     gboolean size_changed = FALSE;
1000     gboolean pos_changed = FALSE;
1001 
1002     if (newW > 0 && newH > 0) {
1003         size_changed = TRUE;
1004 
1005         // content size
1006         geometry.current_cw = newW;
1007         geometry.current_ch = newH;
1008 
1009         if (visible_received) {
1010             // call apply_geometry() to let gtk_window_resize succeed, because it's bound to
1011             // geometry constraints
1012             apply_geometry();
1013             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
1014         } else {
1015             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
1016         }
1017     }
1018 
1019     if (xSet || ySet) {
1020         int newX = (xSet) ? x : geometry.current_x;
1021         int newY = (ySet) ? y : geometry.current_y;
1022 
1023         if (newX != geometry.current_x || newY != geometry.current_y) {
1024             pos_changed = TRUE;
1025             geometry.current_x = newX;
1026             geometry.current_y = newY;
1027             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1028         }
1029     }
1030 
1031     size_position_notify(size_changed, pos_changed);
1032 }

1219 // It's used when the window manager does not support the _NET_FRAME_EXTENTS extension or when
1220 // it's not received on time.
1221 void WindowContext::calculate_adjustments() {
1222     if (frame_type != TITLED || geometry.frame_extents_received) {
1223         return;
1224     }
1225 
1226     gint x, y, rx, ry;
1227     gdk_window_get_origin(gdk_window, &x, &y);
1228     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1229 
1230     if (rx != x || ry != y) {
1231         // the left extends are correct - the right one is guessed to be the same
1232         geometry.adjust_w = (x - rx) * 2;
1233         // guess that bottom size is the same as left and right
1234         geometry.adjust_h = (y - ry) + (x - rx);
1235 
1236         // those will be correct
1237         geometry.view_x = (x - rx);
1238         geometry.view_y = (y - ry);
1239 
1240         if (geometry.needs_ajustment) {
1241             set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
1242         }
1243 
1244         // force position notify so java will know about view_y and view_x
1245         size_position_notify(false, true);
1246     }
1247 }

1249 void WindowContext::apply_geometry() {
1250     if (!map_received) {
1251         return;
1252     }
1253 
1254     GdkGeometry gdk_geometry;
1255     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
1256 
1257     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1258         // not resizeable
1259         int w = geometry.current_cw > 0
1260                 ? geometry.current_cw
1261                 : geometry.current_w - geometry.adjust_w;
1262 
1263         int h = geometry.current_ch > 0
1264                 ? geometry.current_ch
1265                 : geometry.current_h - geometry.adjust_h;
1266 
1267         gdk_geometry.min_width = gdk_geometry.max_width = w;
1268         gdk_geometry.min_height = gdk_geometry.max_height = h;
1269     } else {
1270         //min/max width/height always whole window size (with decors)
1271         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1272                                  ? geometry.minw - geometry.adjust_w : 1;
1273         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1274                                   ? geometry.minh - geometry.adjust_h : 1;
1275 
1276         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
1277                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1278         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1279                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1280     }
1281 
1282     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1283                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
1284 }

1312 void WindowContext::activate_window() {
1313     Display *display = GDK_DISPLAY_XDISPLAY(gdk_window_get_display(gdk_window));
1314     Atom navAtom = XInternAtom(display, "_NET_ACTIVE_WINDOW", True);
1315     if (navAtom != None) {
1316         XClientMessageEvent clientMessage;
1317         memset(&clientMessage, 0, sizeof(clientMessage));
1318 
1319         clientMessage.type = ClientMessage;
1320         clientMessage.window = GDK_WINDOW_XID(gdk_window);
1321         clientMessage.message_type = navAtom;
1322         clientMessage.format = 32;
1323         clientMessage.data.l[0] = 1;
1324         clientMessage.data.l[1] = gdk_x11_get_server_time(gdk_window);
1325         clientMessage.data.l[2] = 0;
1326

1328                    SubstructureRedirectMask | SubstructureNotifyMask,
1329                    (XEvent * ) & clientMessage);

1336     if (jview) {

1373 bool WindowContext::on_top_inherited() {
1374     WindowContext *o = owner;
1375     while (o) {
1376         WindowContext *topO = dynamic_cast<WindowContext *>(o);
1377         if (!topO) break;
1378         if (topO->on_top) {
1379             return true;
1380         }
1381         o = topO->owner;
1382     }
1383     return false;
