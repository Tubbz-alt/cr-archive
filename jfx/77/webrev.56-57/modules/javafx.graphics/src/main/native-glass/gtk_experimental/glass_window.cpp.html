 135 void destroy_and_delete_ctx(WindowContext *ctx) {
 136     if (ctx) {
 137         ctx->process_destroy();
 138 
 139         if (!ctx->get_events_count()) {
 140             delete ctx;
 141         }
 142         // else: ctx will be deleted in EventsCounterHelper after completing
 143         // an event processing
 144     }
 145 }

 147 static inline jint gtk_button_number_to_mouse_button(guint button) {
 148     switch (button) {
 149         case 1:
 150             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 151         case 2:
 152             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 153         case 3:
 154             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 155         case MOUSE_BACK_BTN:
 156             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 157         case MOUSE_FORWARD_BTN:
 158             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 159         default:
 160             // Other buttons are not supported by quantum and are not reported by other platforms
 161             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 162     }
 163 }

 173 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 174                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 175         screen(_screen),
 176         frame_type(_frame_type),
 177         window_type(type),
 178         owner(_owner),
 179         jview(NULL),
 180         map_received(false),
 181         visible_received(false),
 182         on_top(false),
 183         is_fullscreen(false),
 184         is_iconified(false),
 185         is_maximized(false),
 186         is_mouse_entered(false),
 187         can_be_deleted(false),
 188         events_processing_cnt(0),
 189         grab_pointer(NULL) {
 190 
 191     jwindow = mainEnv->NewGlobalRef(_jwindow);
 192 
 193     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 194 
 195 // Not useful, see: https://developer.gnome.org/gtk3/stable/GtkWindow.html#gtk-window-set-wmclass
 196 //    if (gchar * app_name = get_application_name()) {
 197 //        gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 198 //        g_free(app_name);
 199 //    }
 200 
 201     if (owner) {
 202         owner->add_child(this);
 203         if (on_top_inherited()) {
 204             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 205         }
 206     }
 207 
 208     if (type == UTILITY) {
 209         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 210     }
 211 
 212     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 213 
 214     if (xvisualID != 0) {
 215         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 216         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 217     }
 218 
 219     gtk_widget_set_events(gtk_widget, (GDK_EXPOSURE_MASK | GDK_POINTER_MOTION_MASK | GDK_BUTTON_MOTION_MASK
 220                                         | GDK_BUTTON1_MOTION_MASK | GDK_BUTTON2_MOTION_MASK
 221                                         | GDK_BUTTON3_MOTION_MASK | GDK_BUTTON_PRESS_MASK
 222                                         | GDK_BUTTON_RELEASE_MASK | GDK_KEY_PRESS_MASK
 223                                         | GDK_KEY_RELEASE_MASK | GDK_ENTER_NOTIFY_MASK
 224                                         | GDK_LEAVE_NOTIFY_MASK | GDK_FOCUS_CHANGE_MASK
 225                                         | GDK_STRUCTURE_MASK | GDK_PROPERTY_CHANGE_MASK
 226                                         | GDK_VISIBILITY_NOTIFY_MASK | GDK_PROXIMITY_IN_MASK
 227                                         | GDK_PROXIMITY_OUT_MASK | GDK_SUBSTRUCTURE_MASK
 228                                         | GDK_SCROLL_MASK));
 229     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 230 
 231     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 232     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 233 
 234     gdk_window = gtk_widget_get_window(gtk_widget);
 235     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 236 
 237     glass_dnd_attach_context(this);
 238 
 239     gdk_windowManagerFunctions = wmf;
 240     if (wmf) {
 241         gdk_window_set_functions(gdk_window, wmf);
 242     }
 243 
 244     if (frame_type != TITLED) {
 245         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 246     }
 247 
 248     connect_signals(gtk_widget, this);
 249 }

 251 void WindowContext::paint(void *data, jint width, jint height) {
 252 #ifdef GLASS_GTK3
 253     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 254     gdk_window_begin_paint_region(gdk_window, region);
 255     cairo_t* context = gdk_cairo_create(gdk_window);
 256 #else
 257     cairo_t *context = gdk_cairo_create(gdk_window);
 258 #endif
 259 
 260     if (bg_color.is_set) {
 261         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 262                                 (frame_type == TRANSPARENT) ? 0 : 1);
 263         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 264         cairo_paint(context);
 265     }
 266 
 267     cairo_surface_t *cairo_surface;
 268     cairo_surface = cairo_image_surface_create_for_data(
 269             (unsigned char *) data,
 270             CAIRO_FORMAT_ARGB32,
 271             width, height, width * 4);
 272 
 273     cairo_set_source_surface(context, cairo_surface, 0, 0);
 274 
 275     applyShapeMask(data, width, height);
 276     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 277     cairo_paint(context);
 278 
 279 #ifdef GLASS_GTK3
 280     gdk_window_end_paint(gdk_window);
 281     cairo_region_destroy(region);
 282     cairo_destroy(context);
 283 #else
 284     cairo_destroy(context);
 285 #endif
 286 
 287     cairo_surface_destroy(cairo_surface);
 288 }

 330 void WindowContext::process_focus(GdkEventFocus *event) {
 331     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 332         ungrab_mouse_drag_focus();
 333     }
 334 
 335     if (!event->in && WindowContext::sm_grab_window == this) {
 336         ungrab_focus();
 337     }
 338 
 339     if (xim.enabled && xim.ic) {
 340         if (event->in) {
 341             XSetICFocus(xim.ic);
 342         } else {
 343             XUnsetICFocus(xim.ic);
 344         }
 345     }
 346 
 347     if (jwindow) {
 348         if (!event->in || isEnabled()) {
 349             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 350                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 351                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 352             CHECK_JNI_EXCEPTION(mainEnv)
 353         } else {
 354             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 355             CHECK_JNI_EXCEPTION(mainEnv)
 356         }
 357     }
 358 }

 360 void WindowContext::process_property_notify(GdkEventProperty *event) {
 361     if (event->window == gdk_window) {
 362         // This work-around is only necessary for Unity
 363         if (event->atom == atom_net_wm_state) {
 364             process_net_wm_property();
 365         } else if (event->atom == atom_net_wm_frame_extents) {
 366             if (frame_type != TITLED) {
 367                 return;
 368             }
 369 
 370             int top, left, bottom, right;
 371 
 372             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 373                 if (top + left + bottom + right > 0) {
 374                     geometry.frame_extents_received = true;
 375                     geometry.adjust_w = left + right;
 376                     geometry.adjust_h = top + bottom;
 377                     geometry.view_x = left;
 378                     geometry.view_y = top;
 379 
 380                     // set bounds again to set to correct window size that must
 381                     // be the total width and height accounting extents
 382                     // this is ignored if size is "content size" instead of "window size"
 383                     if (geometry.needs_ajustment) {
 384                         set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
 385                     }
 386 
 387                     // force position notify so java will know about view_y and view_x
 388                     size_position_notify(false, true);
 389                 }
 390             }
 391         }
 392     }
 393 }

 395 void WindowContext::process_configure() {
 396     gint x, y, w, h, gtk_w, gtk_h;
 397 
 398     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 399     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 400 
 401     w = gtk_w + geometry.adjust_w;
 402     h = gtk_h + geometry.adjust_h;
 403 
 404     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 405     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 406                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 407 
 408     geometry.current_x = x;
 409     geometry.current_y = y;
 410     geometry.current_w = w;
 411     geometry.current_h = h;
 412     geometry.current_cw = gtk_w;
 413     geometry.current_ch = gtk_h;
 414 
 415     if (!is_fullscreen && !is_maximized) {
 416         geometry.last_cw = gtk_w;
 417         geometry.last_ch = gtk_h;
 418     }
 419 
 420     size_position_notify(size_changed, pos_changed);
 421 }

 423 void WindowContext::process_destroy() {
 424     if (owner) {
 425         owner->remove_child(this);
 426     }
 427 
 428     if (WindowContext::sm_mouse_drag_window == this) {
 429         ungrab_mouse_drag_focus();
 430     }
 431 
 432     if (WindowContext::sm_grab_window == this) {
 433         ungrab_focus();
 434     }
 435 
 436     std::set<WindowContext *>::iterator it;
 437     for (it = children.begin(); it != children.end(); ++it) {
 438         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 439         // WindowContext::process_destroy() to call remove_child() (because children
 440         // is being iterated here) but also prevents gtk_window_set_transient_for from
 441         // being called - this causes the crash on gnome.
 442         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 443         (*it)->set_owner(NULL);
 444         destroy_and_delete_ctx(*it);
 445     }
 446     children.clear();
 447 
 448     if (jwindow) {
 449         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 450         EXCEPTION_OCCURED(mainEnv);
 451     }
 452 
 453     if (jview) {
 454         mainEnv->DeleteGlobalRef(jview);
 455         jview = NULL;
 456     }
 457 
 458     if (jwindow) {
 459         mainEnv->DeleteGlobalRef(jwindow);
 460         jwindow = NULL;
 461     }
 462 
 463     can_be_deleted = true;
 464 }

 482 void WindowContext::process_mouse_button(GdkEventButton *event) {
 483     bool press = event->type == GDK_BUTTON_PRESS;
 484 
 485     guint state = event->state;
 486     guint mask = 0;
 487 
 488     // We need to add/remove current mouse button from the modifier flags
 489     // as X lib state represents the state just prior to the event and
 490     // glass needs the state just after the event
 491     switch (event->button) {
 492         case 1:
 493             mask = GDK_BUTTON1_MASK;
 494             break;
 495         case 2:
 496             mask = GDK_BUTTON2_MASK;
 497             break;
 498         case 3:
 499             mask = GDK_BUTTON3_MASK;
 500             break;
 501         case MOUSE_BACK_BTN:
 502             mask = GDK_BUTTON4_MASK;
 503             break;
 504         case MOUSE_FORWARD_BTN:
 505             mask = GDK_BUTTON5_MASK;
 506             break;
 507     }
 508 
 509     if (press) {
 510         state |= mask;
 511     } else {
 512         state &= ~mask;
 513     }
 514 
 515     if (press) {
 516         GdkDevice* device = event->device;
 517 
 518         if (glass_gdk_device_is_grabbed(device)
 519                 && (glass_gdk_device_get_window_at_position(device, NULL, NULL)
 520                 == NULL)) {
 521             ungrab_focus();
 522             return;
 523         }
 524     }
 525 
 526     // Upper layers expects from us Windows behavior:
 527     // all mouse events should be delivered to window where drag begins
 528     // and no exit/enter event should be reported during this drag.
 529     // We can grab mouse pointer for these needs.
 530     if (press) {
 531         grab_mouse_drag_focus(gdk_window, (GdkEvent *) event, NULL, true);
 532     } else {
 533         if ((event->state & MOUSE_BUTTONS_MASK)
 534             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 535             ungrab_mouse_drag_focus();
 536         } else if (event->button == 8 || event->button == 9) {
 537             // GDK X backend interprets button press events for buttons 4-7 as
 538             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 539             // be set on the event->state from GDK. Thus we cannot check if all
 540             // buttons have been released in the usual way (as above).
 541             ungrab_mouse_drag_focus();
 542         }
 543     }
 544 
 545     jint button = gtk_button_number_to_mouse_button(event->button);
 546 
 547     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 548         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 549                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 550                                 button,
 551                                 (jint) event->x, (jint) event->y,
 552                                 (jint) event->x_root, (jint) event->y_root,
 553                                 gdk_modifier_mask_to_glass(state),
 554                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 555                                 JNI_FALSE);
 556         CHECK_JNI_EXCEPTION(mainEnv)
 557 
 558         if (jview && event->button == 3 && press) {
 559             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 560                                     (jint) event->x, (jint) event->y,
 561                                     (jint) event->x_root, (jint) event->y_root,
 562                                     JNI_FALSE);
 563             CHECK_JNI_EXCEPTION(mainEnv)
 564         }
 565     }
 566 }

 568 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 569     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 570     jint isDrag = glass_modifier & (
 571             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 572             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 573             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 574             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 575             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 576     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 577 
 578     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 579         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 580     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 581         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 582     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 583         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 584     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 585         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 586     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 587         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 588     }
 589

 668 void WindowContext::process_key(GdkEventKey *event) {
 669     bool press = event->type == GDK_KEY_PRESS;
 670     jint glassKey = get_glass_key(event);
 671     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 672     if (press) {
 673         glassModifier |= glass_key_to_modifier(glassKey);
 674     } else {
 675         glassModifier &= ~glass_key_to_modifier(glassKey);
 676     }
 677     jcharArray jChars = NULL;
 678     jchar key = gdk_keyval_to_unicode(event->keyval);
 679     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 680         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 681     } else {
 682 #ifdef GLASS_GTK2
 683         if (key == 0) {
 684             // Work around "bug" fixed in gtk-3.0:
 685             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 686             switch (event->keyval) {
 687             case 0xFF08 /* Backspace */: key =  '\b';
 688             case 0xFF09 /* Tab       */: key =  '\t';
 689             case 0xFF0A /* Linefeed  */: key =  '\n';
 690             case 0xFF0B /* Vert. Tab */: key =  '\v';
 691             case 0xFF0D /* Return    */: key =  '\r';
 692             case 0xFF1B /* Escape    */: key =  '\033';
 693             case 0xFFFF /* Delete    */: key =  '\177';
 694             }
 695         }
 696 #endif
 697     }
 698 
 699     if (key > 0) {
 700         jChars = mainEnv->NewCharArray(1);
 701         if (jChars) {
 702             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 703             CHECK_JNI_EXCEPTION(mainEnv)
 704         }
 705     } else {
 706         jChars = mainEnv->NewCharArray(0);
 707     }
 708     if (jview) {
 709         if (press) {
 710             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 711                                     com_sun_glass_events_KeyEvent_PRESS,
 712                                     glassKey,
 713                                     jChars,
 714                                     glassModifier);
 715             CHECK_JNI_EXCEPTION(mainEnv)
 716 
 717             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 718                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 719                                         com_sun_glass_events_KeyEvent_TYPED,
 720                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 721                                         jChars,
 722                                         glassModifier);
 723                 CHECK_JNI_EXCEPTION(mainEnv)
 724             }
 725         } else {
 726             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 727                                     com_sun_glass_events_KeyEvent_RELEASE,
 728                                     glassKey,
 729                                     jChars,
 730                                     glassModifier);
 731             CHECK_JNI_EXCEPTION(mainEnv)
 732         }
 733     }
 734 }

 736 void WindowContext::process_state(GdkEventWindowState *event) {
 737     if (event->changed_mask &
 738         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 739 
 740         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 741             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 742         }
 743         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 744             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 745         }
 746 
 747         jint stateChangeEvent;
 748 
 749         if (is_iconified) {
 750             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 751         } else if (is_maximized) {
 752             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 753         } else {
 754             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 755             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 756                 // in this case - the window manager will not support the programatic
 757                 // request to iconify - so we need to restore it now.
 758                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 759             }
 760         }
 761 
 762         notify_state(stateChangeEvent);
 763     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 764         notify_on_top(event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 765     }
 766 }

 768 void WindowContext::process_net_wm_property() {
 769     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 770 
 771     // This is a Unity bug
 772     if (!g_strcmp0("Unity", gdk_x11_screen_get_window_manager_name(gdk_screen_get_default()))) {
 773         return;
 774     }
 775 
 776     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 777     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 778     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 779     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 780 
 781     gint length;
 782     glong *atoms = NULL;
 783 
 784     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 785                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 786 
 787         bool is_hidden = false;
 788         bool is_above = false;
 789         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {

 793                 is_above = true;
 794             }
 795         }
 796 
 797         g_free(atoms);
 798

 800             is_iconified = is_hidden;

 803                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 804                          : com_sun_glass_events_WindowEvent_RESTORE);

 994 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 995     // this will tell if adjustments are needed - that's because GTK does not have full window size
 996     // values, just content values. Frame extents (window decorations) are handled by the window manager.
 997     geometry.needs_ajustment = (w > 0 || h > 0) || geometry.needs_ajustment;
 998 
 999     // newW / newH always content sizes compatible with GTK+
1000     // if window has no decoration, adjustments will be ZERO
1001     int newW, newH;
1002     newW = (w > 0) ? w - geometry.adjust_w : cw;
1003     newH = (h > 0) ? h - geometry.adjust_h : ch;
1004 
1005     geometry.current_w = newW;
1006     geometry.current_h = newH;
1007 
1008     gboolean size_changed = FALSE;
1009     gboolean pos_changed = FALSE;
1010 
1011     if (newW > 0 && newH > 0) {
1012         size_changed = TRUE;
1013 
1014         // content size
1015         geometry.current_cw = newW;
1016         geometry.current_ch = newH;
1017         geometry.last_cw = newW;
1018         geometry.last_ch = newH;
1019 
1020         if (visible_received) {
1021             // call apply_geometry() to let gtk_window_resize succeed, because it's bound to
1022             // geometry constraints
1023             apply_geometry();
1024             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
1025         } else {
1026             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
1027         }
1028     }
1029 
1030     if (xSet || ySet) {
1031         int newX = (xSet) ? x : geometry.current_x;
1032         int newY = (ySet) ? y : geometry.current_y;
1033 
1034         if (newX != geometry.current_x || newY != geometry.current_y) {
1035             pos_changed = TRUE;
1036             geometry.current_x = newX;
1037             geometry.current_y = newY;
1038

1250 // This function calculate the deltas between window and window + decoration (titlebar, borders).
1251 // It's used when the window manager does not support the _NET_FRAME_EXTENTS extension or when
1252 // it's not received on time.
1253 void WindowContext::calculate_adjustments() {
1254     if (frame_type != TITLED || geometry.frame_extents_received) {
1255         return;
1256     }
1257 
1258     gint x, y, rx, ry;
1259     gdk_window_get_origin(gdk_window, &x, &y);
1260     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1261 
1262     if (rx != x || ry != y) {
1263         // the left extends are correct - the right one is guessed to be the same
1264         geometry.adjust_w = (x - rx) * 2;
1265         // guess that bottom size is the same as left and right
1266         geometry.adjust_h = (y - ry) + (x - rx);
1267 
1268         // those will be correct
1269         geometry.view_x = (x - rx);
1270         geometry.view_y = (y - ry);
1271 
1272         if (geometry.needs_ajustment) {
1273             set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
1274         }
1275 
1276         // force position notify so java will know about view_y and view_x
1277         size_position_notify(false, true);
1278     }

1282     if (!map_received) {
1283         return;
1284     }
1285 
1286     GdkGeometry gdk_geometry;
1287     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
1288 
1289     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1290         // not resizeable
1291         int w = geometry.current_cw > 0
1292                 ? geometry.current_cw
1293                 : geometry.current_w - geometry.adjust_w;
1294 
1295         int h = geometry.current_ch > 0
1296                 ? geometry.current_ch
1297                 : geometry.current_h - geometry.adjust_h;
1298

1300         gdk_geometry.min_height = gdk_geometry.max_height = h;
1301     } else {
1302         //min/max width/height always whole window size (with decors)
1303         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1304                                  ? geometry.minw - geometry.adjust_w : 1;
1305         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1306                                   ? geometry.minh - geometry.adjust_h : 1;

1309                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1310         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1311                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1312     }
1313 
1314     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1315                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
