 136 void destroy_and_delete_ctx(WindowContext *ctx) {
 137     if (ctx) {
 138         ctx->process_destroy();
 139 
 140         if (!ctx->get_events_count()) {
 141             delete ctx;
 142         }
 143         // else: ctx will be deleted in EventsCounterHelper after completing
 144         // an event processing
 145     }
 146 }

 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 174 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 175                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 176         screen(_screen),
 177         frame_type(_frame_type),
 178         window_type(type),
 179         owner(_owner),
 180         jview(NULL),
 181         map_received(false),
 182         visible_received(false),
 183         on_top(false),
 184         is_fullscreen(false),
 185         is_iconified(false),
 186         is_maximized(false),
 187         is_mouse_entered(false),
 188         can_be_deleted(false),
 189         events_processing_cnt(0),
 190         grab_pointer(NULL) {
 191 
 192     jwindow = mainEnv->NewGlobalRef(_jwindow);
 193 
 194     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 195 
 196 // Not useful, see: https://developer.gnome.org/gtk3/stable/GtkWindow.html#gtk-window-set-wmclass
 197 //    if (gchar * app_name = get_application_name()) {
 198 //        gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 199 //        g_free(app_name);
 200 //    }
 201 
 202     if (owner) {
 203         owner->add_child(this);
 204         if (on_top_inherited()) {
 205             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 206         }
 207     }
 208 
 209     if (type == UTILITY) {
 210         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 211     }
 212 
 213     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 214 
 215     if (xvisualID != 0) {
 216         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 217         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 218     }
 219 
 220     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 221     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 222 
 223     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 224     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 225 
 226     gdk_window = gtk_widget_get_window(gtk_widget);
 227     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 228 
 229     glass_dnd_attach_context(this);
 230 
 231     gdk_windowManagerFunctions = wmf;
 232     if (wmf) {
 233         gdk_window_set_functions(gdk_window, wmf);
 234     }
 235 
 236     if (frame_type != TITLED) {
 237         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 238     }
 239 
 240     connect_signals(gtk_widget, this);
 241 }

 243 void WindowContext::paint(void *data, jint width, jint height) {
 244 #ifdef GLASS_GTK3
 245     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 246     gdk_window_begin_paint_region(gdk_window, region);
 247     cairo_t* context = gdk_cairo_create(gdk_window);
 248 #else
 249     cairo_t *context = gdk_cairo_create(gdk_window);
 250 #endif
 251 
 252     if (bg_color.is_set) {
 253         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 254                                 (frame_type == TRANSPARENT) ? 0 : 1);
 255         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 256         cairo_paint(context);
 257     }
 258 
 259     cairo_surface_t *cairo_surface;
 260     cairo_surface = cairo_image_surface_create_for_data(
 261             (unsigned char *) data,
 262             CAIRO_FORMAT_ARGB32,
 263             width, height, width * 4);
 264 
 265     cairo_set_source_surface(context, cairo_surface, 0, 0);
 266 
 267     applyShapeMask(data, width, height);
 268     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 269     cairo_paint(context);
 270 
 271 #ifdef GLASS_GTK3
 272     gdk_window_end_paint(gdk_window);
 273     cairo_region_destroy(region);
 274     cairo_destroy(context);
 275 #else
 276     cairo_destroy(context);
 277 #endif
 278 
 279     cairo_surface_destroy(cairo_surface);
 280 }

 322 void WindowContext::process_focus(GdkEventFocus *event) {
 323     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 324         ungrab_mouse_drag_focus();
 325     }
 326 
 327     if (!event->in && WindowContext::sm_grab_window == this) {
 328         ungrab_focus();
 329     }
 330 
 331     if (xim.enabled && xim.ic) {
 332         if (event->in) {
 333             XSetICFocus(xim.ic);
 334         } else {
 335             XUnsetICFocus(xim.ic);
 336         }
 337     }
 338 
 339     if (jwindow) {
 340         if (!event->in || isEnabled()) {
 341             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 342                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 343                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 344             CHECK_JNI_EXCEPTION(mainEnv)
 345         } else {
 346             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 347             CHECK_JNI_EXCEPTION(mainEnv)
 348         }
 349     }
 350 }

 352 void WindowContext::process_property_notify(GdkEventProperty *event) {
 353     if (event->window == gdk_window) {
 354         // This work-around is only necessary for Unity
 355         if (event->atom == atom_net_wm_state) {
 356             process_net_wm_property();
 357         } else if (event->atom == atom_net_wm_frame_extents) {
 358             if (frame_type != TITLED) {
 359                 return;
 360             }
 361 
 362             int top, left, bottom, right;
 363 
 364             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 365                 if (top + left + bottom + right > 0) {
 366                     geometry.frame_extents_received = true;
 367                     geometry.adjust_w = left + right;
 368                     geometry.adjust_h = top + bottom;
 369                     geometry.view_x = left;
 370                     geometry.view_y = top;
 371 
 372                     // set bounds again to set to correct window size that must
 373                     // be the total width and height accounting extents
 374                     // this is ignored if size is "content size" instead of "window size"
 375                     if (geometry.needs_ajustment) {
 376                         set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
 377                     }
 378 
 379                     // force position notify so java will know about view_y and view_x
 380                     size_position_notify(false, true);
 381                 }
 382             }
 383         }
 384     }
 385 }

 387 void WindowContext::process_configure() {
 388     gint x, y, w, h, gtk_w, gtk_h;
 389 
 390     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 391     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 392 
 393     w = gtk_w + geometry.adjust_w;
 394     h = gtk_h + geometry.adjust_h;
 395 
 396     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 397     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 398                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 399 
 400     geometry.current_x = x;
 401     geometry.current_y = y;
 402     geometry.current_w = w;
 403     geometry.current_h = h;
 404     geometry.current_cw = gtk_w;
 405     geometry.current_ch = gtk_h;
 406 
 407     if (!is_fullscreen && !is_maximized) {
 408         geometry.last_cw = gtk_w;
 409         geometry.last_ch = gtk_h;
 410     }
 411 
 412     size_position_notify(size_changed, pos_changed);
 413 }

 415 void WindowContext::process_destroy() {
 416     if (owner) {
 417         owner->remove_child(this);
 418     }
 419 
 420     if (WindowContext::sm_mouse_drag_window == this) {
 421         ungrab_mouse_drag_focus();
 422     }
 423 
 424     if (WindowContext::sm_grab_window == this) {
 425         ungrab_focus();
 426     }
 427 
 428     std::set<WindowContext *>::iterator it;
 429     for (it = children.begin(); it != children.end(); ++it) {
 430         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 431         // WindowContext::process_destroy() to call remove_child() (because children
 432         // is being iterated here) but also prevents gtk_window_set_transient_for from
 433         // being called - this causes the crash on gnome.
 434         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 435         (*it)->set_owner(NULL);
 436         destroy_and_delete_ctx(*it);
 437     }
 438     children.clear();
 439 
 440     if (jwindow) {
 441         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 442         EXCEPTION_OCCURED(mainEnv);
 443     }
 444 
 445     if (jview) {
 446         mainEnv->DeleteGlobalRef(jview);
 447         jview = NULL;
 448     }
 449 
 450     if (jwindow) {
 451         mainEnv->DeleteGlobalRef(jwindow);
 452         jwindow = NULL;
 453     }
 454 
 455     can_be_deleted = true;
 456 }

 458 void WindowContext::process_delete() {
 459     if (jwindow && isEnabled()) {
 460         gtk_widget_hide_on_delete(gtk_widget);
 461         mainEnv->CallVoidMethod(jwindow, jWindowNotifyClose);
 462         CHECK_JNI_EXCEPTION(mainEnv)
 463     }

 474 void WindowContext::process_mouse_button(GdkEventButton *event) {
 475     bool press = event->type == GDK_BUTTON_PRESS;
 476     bool release = event->type == GDK_BUTTON_RELEASE;
 477 
 478     guint state = event->state;
 479     guint mask = 0;
 480 
 481     // We need to add/remove current mouse button from the modifier flags
 482     // as X lib state represents the state just prior to the event and
 483     // glass needs the state just after the event
 484     switch (event->button) {
 485         case 1:
 486             mask = GDK_BUTTON1_MASK;
 487             break;
 488         case 2:
 489             mask = GDK_BUTTON2_MASK;
 490             break;
 491         case 3:
 492             mask = GDK_BUTTON3_MASK;
 493             break;
 494         case MOUSE_BACK_BTN:
 495             mask = GDK_BUTTON4_MASK;
 496             break;
 497         case MOUSE_FORWARD_BTN:
 498             mask = GDK_BUTTON5_MASK;
 499             break;
 500     }
 501 
 502     if (press) {
 503         state |= mask;
 504     } else {
 505         state &= ~mask;
 506     }
 507 
 508     // Upper layers expects from us Windows behavior:
 509     // all mouse events should be delivered to window where drag begins
 510     // and no exit/enter event should be reported during this drag.
 511     // We can grab mouse pointer for these needs.
 512     if (press) {
 513         grab_mouse_drag_focus(gdk_window, (GdkEvent *) event, NULL, true);
 514     } else {
 515         if ((event->state & MOUSE_BUTTONS_MASK)
 516             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 517             ungrab_mouse_drag_focus();
 518         } else if (event->button == 8 || event->button == 9) {
 519             // GDK X backend interprets button press events for buttons 4-7 as
 520             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 521             // be set on the event->state from GDK. Thus we cannot check if all
 522             // buttons have been released in the usual way (as above).
 523             ungrab_mouse_drag_focus();
 524         }
 525     }
 526 
 527     jint button = gtk_button_number_to_mouse_button(event->button);
 528 
 529     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 530         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 531                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 532                                 button,
 533                                 (jint) event->x, (jint) event->y,
 534                                 (jint) event->x_root, (jint) event->y_root,
 535                                 gdk_modifier_mask_to_glass(state),
 536                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 537                                 JNI_FALSE);
 538         CHECK_JNI_EXCEPTION(mainEnv)
 539 
 540         if (jview && event->button == 3 && press) {
 541             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 542                                     (jint) event->x, (jint) event->y,
 543                                     (jint) event->x_root, (jint) event->y_root,
 544                                     JNI_FALSE);
 545             CHECK_JNI_EXCEPTION(mainEnv)
 546         }
 547     }
 548 }

 550 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 551     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 552     jint isDrag = glass_modifier & (
 553             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 554             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 555             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 556             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 557             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 558     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 559 
 560     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 561         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 562     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 563         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 564     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 565         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 566     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 567         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 568     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 569         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 570     }
 571 
 572     if (jview) {
 573         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 574                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 575                                 button,
 576                                 (jint) event->x, (jint) event->y,
 577                                 (jint) event->x_root, (jint) event->y_root,
 578                                 glass_modifier,
 579                                 JNI_FALSE,
 580                                 JNI_FALSE);
 581         CHECK_JNI_EXCEPTION(mainEnv)
 582     }
 583 }

 585 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 586     jdouble dx = 0, dy = 0;
 587 
 588     // converting direction to change in pixels
 589     switch (event->direction) {
 590         case GDK_SCROLL_UP:
 591             dy = 1;
 592             break;
 593         case GDK_SCROLL_DOWN:
 594             dy = -1;
 595             break;
 596         case GDK_SCROLL_LEFT:
 597             dx = 1;
 598             break;
 599         case GDK_SCROLL_RIGHT:
 600             dx = -1;
 601             break;
 602         default:
 603             break;
 604     }
 605     if (event->state & GDK_SHIFT_MASK) {
 606         jdouble t = dy;
 607         dy = dx;
 608         dx = t;
 609     }
 610     if (jview) {
 611         mainEnv->CallVoidMethod(jview, jViewNotifyScroll,
 612                                 (jint) event->x, (jint) event->y,
 613                                 (jint) event->x_root, (jint) event->y_root,
 614                                 dx, dy,
 615                                 gdk_modifier_mask_to_glass(event->state),
 616                                 (jint) 0, (jint) 0,
 617                                 (jint) 0, (jint) 0,
 618                                 (jdouble) 40.0, (jdouble) 40.0);
 619         CHECK_JNI_EXCEPTION(mainEnv)
 620     }
 621 }

 648 void WindowContext::process_key(GdkEventKey *event) {
 649     bool press = event->type == GDK_KEY_PRESS;
 650     jint glassKey = get_glass_key(event);
 651     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 652     if (press) {
 653         glassModifier |= glass_key_to_modifier(glassKey);
 654     } else {
 655         glassModifier &= ~glass_key_to_modifier(glassKey);
 656     }
 657     jcharArray jChars = NULL;
 658     jchar key = gdk_keyval_to_unicode(event->keyval);
 659     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 660         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 661     } else {
 662 #ifdef GLASS_GTK2
 663         if (key == 0) {
 664             // Work around "bug" fixed in gtk-3.0:
 665             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 666             switch (event->keyval) {
 667             case 0xFF08 /* Backspace */: key =  '\b';
 668             case 0xFF09 /* Tab       */: key =  '\t';
 669             case 0xFF0A /* Linefeed  */: key =  '\n';
 670             case 0xFF0B /* Vert. Tab */: key =  '\v';
 671             case 0xFF0D /* Return    */: key =  '\r';
 672             case 0xFF1B /* Escape    */: key =  '\033';
 673             case 0xFFFF /* Delete    */: key =  '\177';
 674             }
 675         }
 676 #endif
 677     }
 678 
 679     if (key > 0) {
 680         jChars = mainEnv->NewCharArray(1);
 681         if (jChars) {
 682             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 683             CHECK_JNI_EXCEPTION(mainEnv)
 684         }
 685     } else {
 686         jChars = mainEnv->NewCharArray(0);
 687     }
 688     if (jview) {
 689         if (press) {
 690             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 691                                     com_sun_glass_events_KeyEvent_PRESS,
 692                                     glassKey,
 693                                     jChars,
 694                                     glassModifier);
 695             CHECK_JNI_EXCEPTION(mainEnv)
 696 
 697             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 698                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 699                                         com_sun_glass_events_KeyEvent_TYPED,
 700                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 701                                         jChars,
 702                                         glassModifier);
 703                 CHECK_JNI_EXCEPTION(mainEnv)
 704             }
 705         } else {
 706             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 707                                     com_sun_glass_events_KeyEvent_RELEASE,
 708                                     glassKey,
 709                                     jChars,
 710                                     glassModifier);
 711             CHECK_JNI_EXCEPTION(mainEnv)
 712         }
 713     }
 714 }

 716 void WindowContext::process_state(GdkEventWindowState *event) {
 717     if (event->changed_mask &
 718         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 719 
 720         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 721             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 722         }
 723         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 724             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 725         }
 726 
 727         jint stateChangeEvent;
 728 
 729         if (is_iconified) {
 730             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 731         } else if (is_maximized) {
 732             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 733         } else {
 734             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 735             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 736                 // in this case - the window manager will not support the programatic
 737                 // request to iconify - so we need to restore it now.
 738                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 739             }
 740         }
 741 
 742         notify_state(stateChangeEvent);
 743     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 744         notify_on_top(event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 745     }
 746 }

 748 void WindowContext::process_net_wm_property() {
 749     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 750 
 751     // This is a Unity bug
 752     if (!g_strcmp0("Unity", gdk_x11_screen_get_window_manager_name(gdk_screen_get_default()))) {
 753         return;
 754     }
 755 
 756     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 757     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 758     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 759     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 760 
 761     gint length;
 762     glong *atoms = NULL;
 763 
 764     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 765                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 766 
 767         bool is_hidden = false;
 768         bool is_above = false;
 769         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 770             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 771                 is_hidden = true;
 772             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 773                 is_above = true;
 774             }
 775         }
 776 
 777         g_free(atoms);
 778 
 779         if (is_iconified != is_hidden) {
 780             is_iconified = is_hidden;
 781 
 782             notify_state((is_hidden)
 783                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 784                          : com_sun_glass_events_WindowEvent_RESTORE);
 785         }
 786 
 787         notify_on_top(is_above);
 788     }
 789 }

 975 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 976     // this will tell if adjustments are needed - that's because GTK does not have full window size
 977     // values, just content values. Frame extents (window decorations) are handled by the window manager.
 978     geometry.needs_ajustment = (w > 0 || h > 0) || geometry.needs_ajustment;
 979 
 980     // newW / newH always content sizes compatible with GTK+
 981     // if window has no decoration, adjustments will be ZERO
 982     int newW, newH;
 983     newW = (w > 0) ? w - geometry.adjust_w : cw;
 984     newH = (h > 0) ? h - geometry.adjust_h : ch;
 985 
 986     geometry.current_w = newW;
 987     geometry.current_h = newH;
 988 
 989     gboolean size_changed = FALSE;
 990     gboolean pos_changed = FALSE;
 991 
 992     if (newW > 0 && newH > 0) {
 993         size_changed = TRUE;
 994 
 995         // content size
 996         geometry.current_cw = newW;
 997         geometry.current_ch = newH;
 998         geometry.last_cw = newW;
 999         geometry.last_ch = newH;
1000 
1001         if (visible_received) {
1002             // call apply_geometry() to let gtk_window_resize succeed, because it's bound to
1003             // geometry constraints
1004             apply_geometry();
1005             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
1006         } else {
1007             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
1008         }
1009     }
1010 
1011     if (xSet || ySet) {
1012         int newX = (xSet) ? x : geometry.current_x;
1013         int newY = (ySet) ? y : geometry.current_y;
1014 
1015         if (newX != geometry.current_x || newY != geometry.current_y) {
1016             pos_changed = TRUE;
1017             geometry.current_x = newX;
1018             geometry.current_y = newY;
1019 
1020             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1021         }
1022     }
1023 
1024     size_position_notify(size_changed, pos_changed);
1025 }

1231 // This function calculate the deltas between window and window + decoration (titlebar, borders).
1232 // It's used when the window manager does not support the _NET_FRAME_EXTENTS extension or when
1233 // it's not received on time.
1234 void WindowContext::calculate_adjustments() {
1235     if (frame_type != TITLED || geometry.frame_extents_received) {
1236         return;
1237     }
1238 
1239     gint x, y, rx, ry;
1240     gdk_window_get_origin(gdk_window, &x, &y);
1241     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1242 
1243     if (rx != x || ry != y) {
1244         // the left extends are correct - the right one is guessed to be the same
1245         geometry.adjust_w = (x - rx) * 2;
1246         // guess that bottom size is the same as left and right
1247         geometry.adjust_h = (y - ry) + (x - rx);
1248 
1249         // those will be correct
1250         geometry.view_x = (x - rx);
1251         geometry.view_y = (y - ry);
1252 
1253         if (geometry.needs_ajustment) {
1254             set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
1255         }
1256 
1257         // force position notify so java will know about view_y and view_x
1258         size_position_notify(false, true);
1259     }
1260 }

1262 void WindowContext::apply_geometry() {
1263     if (!map_received) {
1264         return;
1265     }
1266 
1267     GdkGeometry gdk_geometry;
1268     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
1269 
1270     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1271         // not resizeable
1272         int w = geometry.current_cw > 0
1273                 ? geometry.current_cw
1274                 : geometry.current_w - geometry.adjust_w;
1275 
1276         int h = geometry.current_ch > 0
1277                 ? geometry.current_ch
1278                 : geometry.current_h - geometry.adjust_h;
1279 
1280         gdk_geometry.min_width = gdk_geometry.max_width = w;
1281         gdk_geometry.min_height = gdk_geometry.max_height = h;
1282     } else {
1283         //min/max width/height always whole window size (with decors)
1284         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1285                                  ? geometry.minw - geometry.adjust_w : 1;
1286         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1287                                   ? geometry.minh - geometry.adjust_h : 1;
1288 
1289         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
1290                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1291         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1292                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1293     }
1294 
1295     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1296                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
1297 }

1325 void WindowContext::activate_window() {
1326     Display *display = GDK_DISPLAY_XDISPLAY(gdk_window_get_display(gdk_window));
1327     Atom navAtom = XInternAtom(display, "_NET_ACTIVE_WINDOW", True);
1328     if (navAtom != None) {
1329         XClientMessageEvent clientMessage;
1330         memset(&clientMessage, 0, sizeof(clientMessage));
1331 
1332         clientMessage.type = ClientMessage;
1333         clientMessage.window = GDK_WINDOW_XID(gdk_window);
1334         clientMessage.message_type = navAtom;
1335         clientMessage.format = 32;
1336         clientMessage.data.l[0] = 1;
1337         clientMessage.data.l[1] = gdk_x11_get_server_time(gdk_window);
1338         clientMessage.data.l[2] = 0;
1339 
1340         XSendEvent(display, XDefaultRootWindow(display), False,
1341                    SubstructureRedirectMask | SubstructureNotifyMask,
1342                    (XEvent * ) & clientMessage);
1343         XFlush(display);
1344     }
1345 }

1347 void WindowContext::size_position_notify(bool size_changed, bool pos_changed) {
1348 
1349     if (jview) {
1350         if (size_changed) {
1351             mainEnv->CallVoidMethod(jview, jViewNotifyResize, geometry.current_cw, geometry.current_ch);
1352             CHECK_JNI_EXCEPTION(mainEnv);
1353         }
1354 
1355         if (pos_changed) {
1356             mainEnv->CallVoidMethod(jview, jViewNotifyView, com_sun_glass_events_ViewEvent_MOVE);
1357             CHECK_JNI_EXCEPTION(mainEnv)
1358         }
1359     }
1360 
1361     if (jwindow) {
1362         if (size_changed || is_maximized) {
1363             mainEnv->CallVoidMethod(jwindow, jWindowNotifyResize,
1364                                     (is_maximized)
1365                                     ? com_sun_glass_events_WindowEvent_MAXIMIZE
1366                                     : com_sun_glass_events_WindowEvent_RESIZE,
1367                                     geometry.current_w, geometry.current_h);
1368             CHECK_JNI_EXCEPTION(mainEnv)
1369         }
1370 
1371         if (pos_changed) {
1372             mainEnv->CallVoidMethod(jwindow, jWindowNotifyMove, geometry.current_x, geometry.current_y);
1373             CHECK_JNI_EXCEPTION(mainEnv)
1374         }
1375     }
1376 }

1386 bool WindowContext::on_top_inherited() {
1387     WindowContext *o = owner;
1388     while (o) {
1389         WindowContext *topO = dynamic_cast<WindowContext *>(o);
1390         if (!topO) break;
1391         if (topO->on_top) {
1392             return true;
1393         }
1394         o = topO->owner;
1395     }
1396     return false;
1397 }

1400     if (owner) {
1401         WindowContext *topO = dynamic_cast<WindowContext *>(owner);
1402         return (topO && topO->effective_on_top()) || on_top;
1403     }
1404     return on_top;

1408     if (is_grab_disabled()) {
1409         return true;

1412     ungrab_mouse_drag_focus();

1414 #ifdef GLASS_GTK3

1419     }
1420

1422                                                 (GdkEventMask)

1429                                                           | GDK_BUTTON_PRESS_MASK
1430                                                           | GDK_BUTTON_RELEASE_MASK), cursor, GDK_CURRENT_TIME);
1431 #else
1432     GdkGrabStatus status = gdk_pointer_grab(gdk_w, owner_events,
1433                                                 (GdkEventMask)
1434                                                       (GDK_POINTER_MOTION_MASK
