 136 void destroy_and_delete_ctx(WindowContext *ctx) {
 137     if (ctx) {
 138         ctx->process_destroy();
 139 
 140         if (!ctx->get_events_count()) {
 141             delete ctx;
 142         }
 143         // else: ctx will be deleted in EventsCounterHelper after completing
 144         // an event processing
 145     }
 146 }

 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 174 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 175                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 176         screen(_screen),
 177         frame_type(_frame_type),
 178         window_type(type),
 179         owner(_owner),
 180         jview(NULL),
 181         map_received(false),
 182         visible_received(false),
 183         on_top(false),
 184         is_fullscreen(false),
 185         is_iconified(false),
 186         is_maximized(false),
 187         is_mouse_entered(false),
 188         can_be_deleted(false),
 189         events_processing_cnt(0),
 190         grab_pointer(NULL) {
 191 
 192     jwindow = mainEnv->NewGlobalRef(_jwindow);
 193 
 194     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 195 
 196 // Not useful, see: https://developer.gnome.org/gtk3/stable/GtkWindow.html#gtk-window-set-wmclass
 197 //    if (gchar * app_name = get_application_name()) {
 198 //        gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 199 //        g_free(app_name);
 200 //    }
 201 
 202     if (owner) {
 203         owner->add_child(this);
 204         if (on_top_inherited()) {
 205             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 206         }
 207     }
 208 
 209     if (type == UTILITY) {
 210         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 211     }
 212 
 213     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 214 
 215     if (xvisualID != 0) {
 216         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 217         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 218     }
 219 
 220     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 221     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 222 
 223     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 224     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 225 
 226     gdk_window = gtk_widget_get_window(gtk_widget);
 227     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 228 
 229     glass_dnd_attach_context(this);
 230 
 231     gdk_windowManagerFunctions = wmf;
 232     if (wmf) {
 233         gdk_window_set_functions(gdk_window, wmf);
 234     }
 235 
 236     if (frame_type != TITLED) {
 237         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 238     }
 239 
 240     connect_signals(gtk_widget, this);
 241 }

 243 void WindowContext::paint(void *data, jint width, jint height) {
 244 #if GTK_CHECK_VERSION(3, 0, 0)
 245     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 246 #if GTK_CHECK_VERSION(3, 22, 0)
 247     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 248     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 249 #else
 250     gdk_window_begin_paint_region(gdk_window, region);
 251     cairo_t* context = gdk_cairo_create(gdk_window);
 252 #endif
 253 #else
 254     cairo_t *context = gdk_cairo_create(gdk_window);
 255 #endif
 256 
 257     if (bg_color.is_set) {
 258         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 259                                 (frame_type == TRANSPARENT) ? 0 : 1);
 260         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 261         cairo_paint(context);
 262     }
 263 
 264     cairo_surface_t *cairo_surface;
 265     cairo_surface = cairo_image_surface_create_for_data(
 266             (unsigned char *) data,
 267             CAIRO_FORMAT_ARGB32,
 268             width, height, width * 4);
 269 
 270     cairo_set_source_surface(context, cairo_surface, 0, 0);
 271 
 272     applyShapeMask(data, width, height);
 273     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 274     cairo_paint(context);
 275 
 276 #if GTK_CHECK_VERSION(3, 0, 0)
 277 #if GTK_CHECK_VERSION(3, 22, 0)
 278     gdk_window_end_draw_frame(gdk_window, dcontext);
 279     cairo_region_destroy(region);
 280 #else
 281     gdk_window_end_paint(gdk_window);
 282     cairo_region_destroy(region);
 283     cairo_destroy(context);
 284 #endif
 285 #else
 286     cairo_destroy(context);
 287 #endif
 288 
 289     cairo_surface_destroy(cairo_surface);
 290 }

 332 void WindowContext::process_focus(GdkEventFocus *event) {
 333     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 334         ungrab_mouse_drag_focus();
 335     }
 336 
 337     if (!event->in && WindowContext::sm_grab_window == this) {
 338         ungrab_focus();
 339     }
 340 
 341     if (xim.enabled && xim.ic) {
 342         if (event->in) {
 343             XSetICFocus(xim.ic);
 344         } else {
 345             XUnsetICFocus(xim.ic);
 346         }
 347     }
 348 
 349     if (jwindow) {
 350         if (!event->in || isEnabled()) {
 351             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 352                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 353                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 354             CHECK_JNI_EXCEPTION(mainEnv)
 355         } else {
 356             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 357             CHECK_JNI_EXCEPTION(mainEnv)
 358         }
 359     }
 360 }

 362 void WindowContext::process_property_notify(GdkEventProperty *event) {
 363     if (event->window == gdk_window) {
 364         // This work-around is only necessary for Unity
 365         if (event->atom == atom_net_wm_state) {
 366             process_net_wm_property();
 367         } else if (event->atom == atom_net_wm_frame_extents) {
 368             if (frame_type != TITLED) {
 369                 return;
 370             }
 371 
 372             int top, left, bottom, right;
 373 
 374             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 375                 if (top + left + bottom + right > 0) {
 376                     geometry.frame_extents_received = true;
 377                     geometry.adjust_w = left + right;
 378                     geometry.adjust_h = top + bottom;
 379                     geometry.view_x = left;
 380                     geometry.view_y = top;
 381 
 382                     // set bounds again to set to correct window size that must
 383                     // be the total width and height accounting extents
 384                     // this is ignored if size is "content size" instead of "window size"
 385                     if (geometry.needs_ajustment) {
 386                         set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
 387                     }
 388 
 389                     // force position notify so java will know about view_y and view_x
 390                     size_position_notify(false, true);
 391                 }
 392             }
 393         }
 394     }
 395 }

 397 void WindowContext::process_configure() {
 398     gint x, y, w, h, gtk_w, gtk_h;
 399 
 400     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 401     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 402 
 403     w = gtk_w + geometry.adjust_w;
 404     h = gtk_h + geometry.adjust_h;
 405 
 406     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 407     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 408                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 409 
 410     geometry.current_x = x;
 411     geometry.current_y = y;
 412     geometry.current_w = w;
 413     geometry.current_h = h;
 414     geometry.current_cw = gtk_w;
 415     geometry.current_ch = gtk_h;
 416 
 417     if (!is_fullscreen && !is_maximized) {
 418         geometry.last_cw = gtk_w;
 419         geometry.last_ch = gtk_h;
 420     }
 421 
 422     size_position_notify(size_changed, pos_changed);
 423 }

 425 void WindowContext::process_destroy() {
 426     if (owner) {
 427         owner->remove_child(this);
 428     }
 429 
 430     if (WindowContext::sm_mouse_drag_window == this) {
 431         ungrab_mouse_drag_focus();
 432     }
 433 
 434     if (WindowContext::sm_grab_window == this) {
 435         ungrab_focus();
 436     }
 437 
 438     std::set<WindowContext *>::iterator it;
 439     for (it = children.begin(); it != children.end(); ++it) {
 440         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 441         // WindowContext::process_destroy() to call remove_child() (because children
 442         // is being iterated here) but also prevents gtk_window_set_transient_for from
 443         // being called - this causes the crash on gnome.
 444         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 445         (*it)->set_owner(NULL);
 446         destroy_and_delete_ctx(*it);
 447     }
 448     children.clear();
 449 
 450     if (jwindow) {
 451         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 452         EXCEPTION_OCCURED(mainEnv);
 453     }
 454 
 455     if (jview) {
 456         mainEnv->DeleteGlobalRef(jview);
 457         jview = NULL;
 458     }
 459 
 460     if (jwindow) {
 461         mainEnv->DeleteGlobalRef(jwindow);
 462         jwindow = NULL;
 463     }
 464 
 465     can_be_deleted = true;
 466 }

 484 void WindowContext::process_mouse_button(GdkEventButton *event) {
 485     bool press = event->type == GDK_BUTTON_PRESS;
 486     bool release = event->type == GDK_BUTTON_RELEASE;
 487 
 488     guint state = event->state;
 489     guint mask = 0;
 490 
 491     // We need to add/remove current mouse button from the modifier flags
 492     // as X lib state represents the state just prior to the event and
 493     // glass needs the state just after the event
 494     switch (event->button) {
 495         case 1:
 496             mask = GDK_BUTTON1_MASK;
 497             break;
 498         case 2:
 499             mask = GDK_BUTTON2_MASK;
 500             break;
 501         case 3:
 502             mask = GDK_BUTTON3_MASK;
 503             break;
 504         case MOUSE_BACK_BTN:
 505             mask = GDK_BUTTON4_MASK;
 506             break;
 507         case MOUSE_FORWARD_BTN:
 508             mask = GDK_BUTTON5_MASK;
 509             break;
 510     }
 511 
 512     if (press) {
 513         state |= mask;
 514     } else {
 515         state &= ~mask;
 516     }
 517 
 518     // Upper layers expects from us Windows behavior:
 519     // all mouse events should be delivered to window where drag begins
 520     // and no exit/enter event should be reported during this drag.
 521     // We can grab mouse pointer for these needs.
 522     if (press) {
 523         grab_mouse_drag_focus(gdk_window, (GdkEvent *) event, NULL, true);
 524     } else {
 525         if ((event->state & MOUSE_BUTTONS_MASK)
 526             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 527             ungrab_mouse_drag_focus();
 528         } else if (event->button == 8 || event->button == 9) {
 529             // GDK X backend interprets button press events for buttons 4-7 as
 530             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 531             // be set on the event->state from GDK. Thus we cannot check if all
 532             // buttons have been released in the usual way (as above).
 533             ungrab_mouse_drag_focus();
 534         }
 535     }
 536 
 537     jint button = gtk_button_number_to_mouse_button(event->button);
 538 
 539     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 540         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 541                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 542                                 button,
 543                                 (jint) event->x, (jint) event->y,
 544                                 (jint) event->x_root, (jint) event->y_root,
 545                                 gdk_modifier_mask_to_glass(state),
 546                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 547                                 JNI_FALSE);
 548         CHECK_JNI_EXCEPTION(mainEnv)
 549 
 550         if (jview && event->button == 3 && press) {
 551             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 552                                     (jint) event->x, (jint) event->y,
 553                                     (jint) event->x_root, (jint) event->y_root,
 554                                     JNI_FALSE);
 555             CHECK_JNI_EXCEPTION(mainEnv)
 556         }
 557     }
 558 }

 560 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 561     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 562     jint isDrag = glass_modifier & (
 563             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 564             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 565             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 566             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 567             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 568     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 569 
 570     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 571         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 572     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 573         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 574     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 575         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 576     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 577         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 578     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 579         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 580     }
 581 
 582     if (jview) {
 583         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 584                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 585                                 button,
 586                                 (jint) event->x, (jint) event->y,
 587                                 (jint) event->x_root, (jint) event->y_root,
 588                                 glass_modifier,
 589                                 JNI_FALSE,
 590                                 JNI_FALSE);
 591         CHECK_JNI_EXCEPTION(mainEnv)

 662 void WindowContext::process_key(GdkEventKey *event) {
 663     bool press = event->type == GDK_KEY_PRESS;
 664     jint glassKey = get_glass_key(event);
 665     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 666     if (press) {
 667         glassModifier |= glass_key_to_modifier(glassKey);
 668     } else {
 669         glassModifier &= ~glass_key_to_modifier(glassKey);
 670     }
 671     jcharArray jChars = NULL;
 672     jchar key = gdk_keyval_to_unicode(event->keyval);
 673     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 674         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 675     } else {
 676 #ifdef GLASS_GTK2
 677         if (key == 0) {
 678             // Work around "bug" fixed in gtk-3.0:
 679             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 680             switch (event->keyval) {
 681             case 0xFF08 /* Backspace */: key =  '\b';
 682             case 0xFF09 /* Tab       */: key =  '\t';
 683             case 0xFF0A /* Linefeed  */: key =  '\n';
 684             case 0xFF0B /* Vert. Tab */: key =  '\v';
 685             case 0xFF0D /* Return    */: key =  '\r';
 686             case 0xFF1B /* Escape    */: key =  '\033';
 687             case 0xFFFF /* Delete    */: key =  '\177';
 688             }
 689         }
 690 #endif
 691     }
 692 
 693     if (key > 0) {
 694         jChars = mainEnv->NewCharArray(1);
 695         if (jChars) {
 696             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 697             CHECK_JNI_EXCEPTION(mainEnv)
 698         }
 699     } else {
 700         jChars = mainEnv->NewCharArray(0);
 701     }
 702     if (jview) {
 703         if (press) {
 704             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 705                                     com_sun_glass_events_KeyEvent_PRESS,
 706                                     glassKey,
 707                                     jChars,
 708                                     glassModifier);
 709             CHECK_JNI_EXCEPTION(mainEnv)
 710 
 711             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 712                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 713                                         com_sun_glass_events_KeyEvent_TYPED,
 714                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 715                                         jChars,
 716                                         glassModifier);
 717                 CHECK_JNI_EXCEPTION(mainEnv)
 718             }
 719         } else {
 720             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 721                                     com_sun_glass_events_KeyEvent_RELEASE,
 722                                     glassKey,
 723                                     jChars,
 724                                     glassModifier);
 725             CHECK_JNI_EXCEPTION(mainEnv)
 726         }
 727     }
 728 }

 730 void WindowContext::process_state(GdkEventWindowState *event) {
 731     if (event->changed_mask &
 732         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 733 
 734         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 735             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 736         }
 737         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 738             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 739         }
 740 
 741         jint stateChangeEvent;
 742 
 743         if (is_iconified) {
 744             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 745         } else if (is_maximized) {
 746             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 747         } else {
 748             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 749             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 750                 // in this case - the window manager will not support the programatic
 751                 // request to iconify - so we need to restore it now.
 752                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 753             }
 754         }
 755 
 756         notify_state(stateChangeEvent);
 757     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 758         notify_on_top(event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 759     }
 760 }

 762 void WindowContext::process_net_wm_property() {
 763     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 764 
 765     // This is a Unity bug
 766     if (!g_strcmp0("Unity", gdk_x11_screen_get_window_manager_name(gdk_screen_get_default()))) {
 767         return;
 768     }
 769 
 770     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 771     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 772     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 773     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 774 
 775     gint length;
 776     glong *atoms = NULL;
 777 
 778     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 779                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 780 
 781         bool is_hidden = false;
 782         bool is_above = false;
 783         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 784             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 785                 is_hidden = true;
 786             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 787                 is_above = true;
 788             }
 789         }
 790 
 791         g_free(atoms);
 792 
 793         if (is_iconified != is_hidden) {
 794             is_iconified = is_hidden;
 795 
 796             notify_state((is_hidden)
 797                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 798                          : com_sun_glass_events_WindowEvent_RESTORE);
 799         }
 800 
 801         notify_on_top(is_above);
 802     }

 989 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 990     // this will tell if adjustments are needed - that's because GTK does not have full window size
 991     // values, just content values. Frame extents (window decorations) are handled by the window manager.
 992     geometry.needs_ajustment = (w > 0 || h > 0) || geometry.needs_ajustment;
 993 
 994     // newW / newH always content sizes compatible with GTK+
 995     // if window has no decoration, adjustments will be ZERO
 996     // this will always be content size
 997     int newW, newH;
 998     if (w > 0) {
 999         newW = w - geometry.adjust_w;
1000         geometry.current_w = newW;
1001     } else {
1002         newW = cw;
1003     }
1004 
1005     if (h > 0) {
1006         newH = h - geometry.adjust_h;
1007         geometry.current_h = newH;
1008     } else {
1009         newH = ch;
1010     }
1011 
1012     gboolean size_changed = FALSE;
1013     gboolean pos_changed = FALSE;
1014 
1015     if (newW > 0 && newH > 0) {
1016         size_changed = TRUE;
1017 
1018         // content size
1019         geometry.current_cw = newW;
1020         geometry.current_ch = newH;
1021         geometry.last_cw = newW;
1022         geometry.last_ch = newH;
1023 
1024         if (visible_received) {
1025             // call apply_geometry() to let gtk_window_resize succeed, because it's bound to
1026             // geometry constraints
1027             apply_geometry();
1028             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
1029         } else {
1030             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
1031         }
1032     }
1033 
1034     if (xSet || ySet) {
1035         int newX = (xSet) ? x : geometry.current_x;
1036         int newY = (ySet) ? y : geometry.current_y;
1037 
1038         if (newX != geometry.current_x || newY != geometry.current_y) {
1039             pos_changed = TRUE;
1040             geometry.current_x = newX;
1041             geometry.current_y = newY;
1042             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1043         }
1044     }
1045 
1046     size_position_notify(size_changed, pos_changed);
1047 }

1255 // This function calculate the deltas between window and window + decoration (titlebar, borders).
1256 // It's used when the window manager does not support the _NET_FRAME_EXTENTS extension or when
1257 // it's not received on time.
1258 void WindowContext::calculate_adjustments() {
1259     if (frame_type != TITLED || geometry.frame_extents_received) {
1260         return;
1261     }
1262 
1263     gint x, y, rx, ry;
1264     gdk_window_get_origin(gdk_window, &x, &y);
1265     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1266 
1267     if (rx != x || ry != y) {
1268         // the left extends are correct - the right one is guessed to be the same
1269         geometry.adjust_w = (x - rx) * 2;
1270         // guess that bottom size is the same as left and right
1271         geometry.adjust_h = (y - ry) + (x - rx);
1272 
1273         // those will be correct
1274         geometry.view_x = (x - rx);
1275         geometry.view_y = (y - ry);
1276 
1277         if (geometry.needs_ajustment) {
1278             set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
1279         }
1280 
1281         // force position notify so java will know about view_y and view_x
1282         size_position_notify(false, true);
1283     }
1284 }

1286 void WindowContext::apply_geometry() {
1287     if (!map_received) {
1288         return;
1289     }
1290 
1291     GdkGeometry gdk_geometry;
1292     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
1293 
1294     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1295         // not resizeable
1296         int w = geometry.current_cw > 0
1297                 ? geometry.current_cw
1298                 : geometry.current_w - geometry.adjust_w;
1299 
1300         int h = geometry.current_ch > 0
1301                 ? geometry.current_ch
1302                 : geometry.current_h - geometry.adjust_h;
1303 
1304         gdk_geometry.min_width = gdk_geometry.max_width = w;
1305         gdk_geometry.min_height = gdk_geometry.max_height = h;
1306     } else {
1307         //min/max width/height always whole window size (with decors)
1308         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1309                                  ? geometry.minw - geometry.adjust_w : 1;
1310         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1311                                   ? geometry.minh - geometry.adjust_h : 1;
1312 
1313         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
1314                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1315         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1316                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1317     }
1318 
1319     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1320                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
1321 }
