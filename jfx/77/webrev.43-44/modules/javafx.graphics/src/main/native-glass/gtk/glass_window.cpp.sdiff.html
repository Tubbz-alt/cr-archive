 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 174 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 175                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 176         screen(_screen),
 177         frame_type(_frame_type),
 178         window_type(type),
 179         owner(_owner),
 180         jview(NULL),
 181         map_received(false),
 182         visible_received(false),
 183         on_top(false),
 184         is_fullscreen(false),
 185         is_iconified(false),
 186         is_maximized(false),
 187         is_mouse_entered(false),
 188         can_be_deleted(false),
 189         events_processing_cnt(0),
 190         grab_pointer(NULL) {
 191 
 192     jwindow = mainEnv->NewGlobalRef(_jwindow);
 193 
 194     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 195 
 196     if (gchar * app_name = get_application_name()) {
 197         gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 198         g_free(app_name);
 199     }
 200 
 201     if (owner) {
 202         owner->add_child(this);
 203         if (on_top_inherited()) {
 204             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 205         }
 206     }
 207 
 208     if (type == UTILITY) {
 209         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 210     }
 211 
 212     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 213 
 214     if (xvisualID != 0) {
 215         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 216         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 217     }
 218 
 219     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 220     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 221 
 222     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 223     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 224 
 225     gdk_window = gtk_widget_get_window(gtk_widget);
 226     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 227 
 228     glass_dnd_attach_context(this);
 229 
 230     gdk_windowManagerFunctions = wmf;
 231     if (wmf) {
 232         gdk_window_set_functions(gdk_window, wmf);
 233     }
 234 
 235     if (frame_type != TITLED) {
 236         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 237     }
 238 
 239     connect_signals(gtk_widget, this);
 240 }

 242 void WindowContext::paint(void *data, jint width, jint height) {
 243 #if GTK_CHECK_VERSION(3, 0, 0)
 244     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 245 #if GTK_CHECK_VERSION(3, 22, 0)
 246     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 247     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 248 #else
 249     gdk_window_begin_paint_region(gdk_window, region);
 250     cairo_t* context = gdk_cairo_create(gdk_window);
 251 #endif
 252 #else
 253     cairo_t *context = gdk_cairo_create(gdk_window);
 254 #endif
 255 
 256     if (bg_color.is_set) {
 257         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 258                                 (frame_type == TRANSPARENT) ? 0 : 1);
 259         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 260         cairo_paint(context);
 261     }
 262 
 263     cairo_surface_t *cairo_surface;
 264     cairo_surface = cairo_image_surface_create_for_data(
 265             (unsigned char *) data,
 266             CAIRO_FORMAT_ARGB32,
 267             width, height, width * 4);
 268 
 269     cairo_set_source_surface(context, cairo_surface, 0, 0);
 270 
 271     applyShapeMask(data, width, height);
 272     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 273     cairo_paint(context);
 274 
 275 #if GTK_CHECK_VERSION(3, 0, 0)
 276 #if GTK_CHECK_VERSION(3, 22, 0)
 277     gdk_window_end_draw_frame(gdk_window, dcontext);
 278     cairo_region_destroy(region);
 279 #else
 280     gdk_window_end_paint(gdk_window);
 281     cairo_region_destroy(region);
 282     cairo_destroy(context);
 283 #endif
 284 #else
 285     cairo_destroy(context);
 286 #endif
 287 
 288     cairo_surface_destroy(cairo_surface);
 289 }

 331 void WindowContext::process_focus(GdkEventFocus *event) {
 332     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 333         ungrab_mouse_drag_focus();
 334     }
 335 
 336     if (!event->in && WindowContext::sm_grab_window == this) {
 337         ungrab_focus();
 338     }
 339 
 340     if (xim.enabled && xim.ic) {
 341         if (event->in) {
 342             XSetICFocus(xim.ic);
 343         } else {
 344             XUnsetICFocus(xim.ic);
 345         }
 346     }
 347 
 348     if (jwindow) {
 349         if (!event->in || isEnabled()) {
 350             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 351                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 352                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 353             CHECK_JNI_EXCEPTION(mainEnv)
 354         } else {
 355             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 356             CHECK_JNI_EXCEPTION(mainEnv)
 357         }
 358     }
 359 }

 361 void WindowContext::process_property_notify(GdkEventProperty *event) {
 362     if (event->window == gdk_window) {
 363         // This work-around is only necessary for Unity
 364         if (event->atom == atom_net_wm_state) {
 365             process_net_wm_property();
 366         } else if (event->atom == atom_net_wm_frame_extents) {
 367             if (frame_type != TITLED) {
 368                 return;
 369             }
 370 
 371             int top, left, bottom, right;
 372 
 373             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 374                 if (top + left + bottom + right > 0) {
 375                     geometry.frame_extents_received = true;
 376                     geometry.adjust_w = left + right;
 377                     geometry.adjust_h = top + bottom;
 378                     geometry.view_x = left;
 379                     geometry.view_y = top;
 380 
 381                     // set bounds again to set to correct window size that must
 382                     // be the total width and height accounting extents
 383                     // this is ignored if size is "content size" instead of "window size"
 384                     if (geometry.needs_ajustment) {
 385                         set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
 386                     }
 387 
 388                     // force position notify so java will know about view_y and view_x
 389                     size_position_notify(false, true);
 390                 }
 391             }
 392         }
 393     }
 394 }

 396 void WindowContext::process_configure() {
 397     gint x, y, w, h, gtk_w, gtk_h;
 398 
 399     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 400     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 401 
 402     w = gtk_w + geometry.adjust_w;
 403     h = gtk_h + geometry.adjust_h;

 404 
 405     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 406     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 407                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 408 
 409     geometry.current_x = x;
 410     geometry.current_y = y;
 411     geometry.current_w = w;
 412     geometry.current_h = h;
 413     geometry.current_cw = gtk_w;
 414     geometry.current_ch = gtk_h;
 415 
 416     if (!is_fullscreen && !is_maximized) {
 417         geometry.last_cw = gtk_w;
 418         geometry.last_ch = gtk_h;


 419     }
 420 
 421     size_position_notify(size_changed, pos_changed);
 422 }

 424 void WindowContext::process_destroy() {
 425     if (owner) {
 426         owner->remove_child(this);
 427     }
 428 
 429     if (WindowContext::sm_mouse_drag_window == this) {
 430         ungrab_mouse_drag_focus();
 431     }
 432 
 433     if (WindowContext::sm_grab_window == this) {
 434         ungrab_focus();
 435     }
 436 
 437     std::set<WindowContext *>::iterator it;
 438     for (it = children.begin(); it != children.end(); ++it) {
 439         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 440         // WindowContext::process_destroy() to call remove_child() (because children
 441         // is being iterated here) but also prevents gtk_window_set_transient_for from
 442         // being called - this causes the crash on gnome.
 443         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 444         (*it)->set_owner(NULL);
 445         destroy_and_delete_ctx(*it);
 446     }
 447     children.clear();
 448 
 449     if (jwindow) {
 450         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 451         EXCEPTION_OCCURED(mainEnv);
 452     }
 453 
 454     if (jview) {
 455         mainEnv->DeleteGlobalRef(jview);
 456         jview = NULL;
 457     }
 458 
 459     if (jwindow) {
 460         mainEnv->DeleteGlobalRef(jwindow);
 461         jwindow = NULL;
 462     }
 463 
 464     can_be_deleted = true;
 465 }

 483 void WindowContext::process_mouse_button(GdkEventButton *event) {
 484     bool press = event->type == GDK_BUTTON_PRESS;
 485     bool release = event->type == GDK_BUTTON_RELEASE;














 486 


 487     guint state = event->state;
 488     guint mask = 0;
 489 
 490     // We need to add/remove current mouse button from the modifier flags
 491     // as X lib state represents the state just prior to the event and
 492     // glass needs the state just after the event
 493     switch (event->button) {
 494         case 1:
 495             mask = GDK_BUTTON1_MASK;
 496             break;
 497         case 2:
 498             mask = GDK_BUTTON2_MASK;
 499             break;
 500         case 3:
 501             mask = GDK_BUTTON3_MASK;
 502             break;
 503         case MOUSE_BACK_BTN:
 504             mask = GDK_BUTTON4_MASK;
 505             break;
 506         case MOUSE_FORWARD_BTN:
 507             mask = GDK_BUTTON5_MASK;
 508             break;
 509     }
 510 
 511     if (press) {
 512         state |= mask;
 513     } else {
 514         state &= ~mask;
 515     }
 516 











 517     // Upper layers expects from us Windows behavior:
 518     // all mouse events should be delivered to window where drag begins
 519     // and no exit/enter event should be reported during this drag.
 520     // We can grab mouse pointer for these needs.
 521     if (press) {
 522         grab_mouse_drag_focus(gdk_window, (GdkEvent *) event, NULL, true);
 523     } else {
 524         if ((event->state & MOUSE_BUTTONS_MASK)
 525             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 526             ungrab_mouse_drag_focus();
 527         } else if (event->button == 8 || event->button == 9) {
 528             // GDK X backend interprets button press events for buttons 4-7 as
 529             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 530             // be set on the event->state from GDK. Thus we cannot check if all
 531             // buttons have been released in the usual way (as above).
 532             ungrab_mouse_drag_focus();
 533         }
 534     }
 535 
 536     jint button = gtk_button_number_to_mouse_button(event->button);
 537 
 538     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 539         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 540                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 541                                 button,
 542                                 (jint) event->x, (jint) event->y,
 543                                 (jint) event->x_root, (jint) event->y_root,
 544                                 gdk_modifier_mask_to_glass(state),
 545                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 546                                 JNI_FALSE);
 547         CHECK_JNI_EXCEPTION(mainEnv)
 548 
 549         if (jview && event->button == 3 && press) {
 550             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 551                                     (jint) event->x, (jint) event->y,
 552                                     (jint) event->x_root, (jint) event->y_root,
 553                                     JNI_FALSE);
 554             CHECK_JNI_EXCEPTION(mainEnv)
 555         }
 556     }
 557 }

 559 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 560     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 561     jint isDrag = glass_modifier & (
 562             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 563             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 564             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 565             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 566             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 567     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 568 
 569     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 570         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 571     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 572         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 573     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 574         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 575     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 576         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 577     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 578         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 579     }
 580 
 581     if (jview) {
 582         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 583                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 584                                 button,
 585                                 (jint) event->x, (jint) event->y,
 586                                 (jint) event->x_root, (jint) event->y_root,
 587                                 glass_modifier,
 588                                 JNI_FALSE,
 589                                 JNI_FALSE);
 590         CHECK_JNI_EXCEPTION(mainEnv)
 591     }
 592 }

 594 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 595     jdouble dx = 0;
 596     jdouble dy = 0;

 661 void WindowContext::process_key(GdkEventKey *event) {
 662     bool press = event->type == GDK_KEY_PRESS;
 663     jint glassKey = get_glass_key(event);
 664     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 665     if (press) {
 666         glassModifier |= glass_key_to_modifier(glassKey);
 667     } else {
 668         glassModifier &= ~glass_key_to_modifier(glassKey);
 669     }
 670     jcharArray jChars = NULL;
 671     jchar key = gdk_keyval_to_unicode(event->keyval);
 672     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 673         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 674     } else {
 675 #ifdef GLASS_GTK2
 676         if (key == 0) {
 677             // Work around "bug" fixed in gtk-3.0:
 678             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 679             switch (event->keyval) {
 680             case 0xFF08 /* Backspace */: key =  '\b';
 681             case 0xFF09 /* Tab       */: key =  '\t';

 684             case 0xFF0D /* Return    */: key =  '\r';
 685             case 0xFF1B /* Escape    */: key =  '\033';
 686             case 0xFFFF /* Delete    */: key =  '\177';
 687             }
 688         }
 689 #endif
 690     }
 691 
 692     if (key > 0) {
 693         jChars = mainEnv->NewCharArray(1);
 694         if (jChars) {
 695             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 696             CHECK_JNI_EXCEPTION(mainEnv)
 697         }
 698     } else {
 699         jChars = mainEnv->NewCharArray(0);
 700     }
 701     if (jview) {
 702         if (press) {
 703             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 704                                     com_sun_glass_events_KeyEvent_PRESS,
 705                                     glassKey,
 706                                     jChars,
 707                                     glassModifier);
 708             CHECK_JNI_EXCEPTION(mainEnv)
 709 
 710             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 711                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 712                                         com_sun_glass_events_KeyEvent_TYPED,
 713                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 714                                         jChars,
 715                                         glassModifier);
 716                 CHECK_JNI_EXCEPTION(mainEnv)
 717             }
 718         } else {
 719             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 720                                     com_sun_glass_events_KeyEvent_RELEASE,
 721                                     glassKey,
 722                                     jChars,
 723                                     glassModifier);
 724             CHECK_JNI_EXCEPTION(mainEnv)
 725         }
 726     }

 254         case MOUSE_FORWARD_BTN:
 255             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;

 262 void WindowContextBase::process_mouse_button(GdkEventButton* event) {
 263     bool press = event->type == GDK_BUTTON_PRESS;
 264     guint state = event->state;
 265     guint mask = 0;
 266 
 267     // We need to add/remove current mouse button from the modifier flags
 268     // as X lib state represents the state just prior to the event and
 269     // glass needs the state just after the event
 270     switch (event->button) {
 271         case 1:
 272             mask = GDK_BUTTON1_MASK;
 273             break;
 274         case 2:
 275             mask = GDK_BUTTON2_MASK;
 276             break;
 277         case 3:
 278             mask = GDK_BUTTON3_MASK;
 279             break;
 280         case MOUSE_BACK_BTN:
 281             mask = GDK_BUTTON4_MASK;
 282             break;
 283         case MOUSE_FORWARD_BTN:
 284             mask = GDK_BUTTON5_MASK;
 285             break;
 286     }
 287 
 288     if (press) {
 289         state |= mask;
 290     } else {
 291         state &= ~mask;
 292     }
 293 
 294     if (press) {
 295         GdkDevice* device = event->device;
 296 
 297         if (glass_gdk_device_is_grabbed(device)
 298                 && (glass_gdk_device_get_window_at_position(device, NULL, NULL)
 299                 == NULL)) {
 300             ungrab_focus();
 301             return;
 302         }
 303     }
 304 
 305     // Upper layers expects from us Windows behavior:
 306     // all mouse events should be delivered to window where drag begins
 307     // and no exit/enter event should be reported during this drag.
 308     // We can grab mouse pointer for these needs.
 309     if (press) {
 310         grab_mouse_drag_focus();
 311     } else {
 312         if ((event->state & MOUSE_BUTTONS_MASK)
 313             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 314             ungrab_mouse_drag_focus();
 315         } else if (event->button == 8 || event->button == 9) {
 316             // GDK X backend interprets button press events for buttons 4-7 as
 317             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 318             // be set on the event->state from GDK. Thus we cannot check if all
 319             // buttons have been released in the usual way (as above).
 320             ungrab_mouse_drag_focus();
 321         }
 322     }
 323 
 324     jint button = gtk_button_number_to_mouse_button(event->button);
 325 
 326     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 327         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 328                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 329                 button,

 347 void WindowContextBase::process_mouse_motion(GdkEventMotion* event) {
 348     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 349     jint isDrag = glass_modifier & (
 350             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 351             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 352             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 353             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 354             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 355     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 356 
 357     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 358         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 359     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 360         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 361     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 362         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 363     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 364         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 365     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 366         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 367     }
 368 
 369     if (jview) {
 370         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 371                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 372                 button,
 373                 (jint) event->x, (jint) event->y,
 374                 (jint) event->x_root, (jint) event->y_root,
 375                 glass_modifier,
 376                 JNI_FALSE,
 377                 JNI_FALSE);
 378         CHECK_JNI_EXCEPTION(mainEnv)
 379     }

 421     }

 448 void WindowContextBase::process_key(GdkEventKey* event) {
 449     bool press = event->type == GDK_KEY_PRESS;
 450     jint glassKey = get_glass_key(event);
 451     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 452     if (press) {
 453         glassModifier |= glass_key_to_modifier(glassKey);
 454     } else {
 455         glassModifier &= ~glass_key_to_modifier(glassKey);
 456     }
 457     jcharArray jChars = NULL;
 458     jchar key = gdk_keyval_to_unicode(event->keyval);
 459     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 460         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 461     } else {
 462 #ifdef GLASS_GTK2
 463         if (key == 0) {
 464             // Work around "bug" fixed in gtk-3.0:
 465             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 466             switch (event->keyval) {
 467             case 0xFF08 /* Backspace */: key =  '\b';
 468             case 0xFF09 /* Tab       */: key =  '\t';

 471             case 0xFF0D /* Return    */: key =  '\r';
 472             case 0xFF1B /* Escape    */: key =  '\033';
 473             case 0xFFFF /* Delete    */: key =  '\177';
 474             }
 475         }
 476 #endif
 477     }
 478 
 479     if (key > 0) {
 480         jChars = mainEnv->NewCharArray(1);
 481         if (jChars) {
 482             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 483             CHECK_JNI_EXCEPTION(mainEnv)
 484         }
 485     } else {
 486         jChars = mainEnv->NewCharArray(0);
 487     }
 488     if (jview) {
 489         if (press) {
 490             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 491                     com_sun_glass_events_KeyEvent_PRESS,
 492                     glassKey,
 493                     jChars,
 494                     glassModifier);
 495             CHECK_JNI_EXCEPTION(mainEnv)
 496 
 497             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 498                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 499                         com_sun_glass_events_KeyEvent_TYPED,
 500                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 501                         jChars,
 502                         glassModifier);
 503                 CHECK_JNI_EXCEPTION(mainEnv)
 504             }
 505         } else {
 506             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 507                     com_sun_glass_events_KeyEvent_RELEASE,
 508                     glassKey,
 509                     jChars,
 510                     glassModifier);
 511             CHECK_JNI_EXCEPTION(mainEnv)
 512         }
 513     }
 514 }
