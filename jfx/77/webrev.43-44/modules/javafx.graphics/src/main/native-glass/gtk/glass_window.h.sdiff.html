261 class EventsCounterHelper {
262 private:
263     WindowContext *ctx;
264 public:
265     explicit EventsCounterHelper(WindowContext *context) {
266         ctx = context;
267         ctx->increment_events_counter();
268     }
269 
270     ~EventsCounterHelper() {
271         ctx->decrement_events_counter();

273             delete ctx;
274         }
