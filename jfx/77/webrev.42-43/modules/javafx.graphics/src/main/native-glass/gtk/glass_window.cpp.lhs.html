 136 void destroy_and_delete_ctx(WindowContext *ctx) {
 137     if (ctx) {
 138         ctx->process_destroy();
 139 
 140         if (!ctx->get_events_count()) {
 141             delete ctx;
 142         }
 143         // else: ctx will be deleted in EventsCounterHelper after completing
 144         // an event processing
 145     }

 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 174 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 175                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 176         screen(_screen),
 177         frame_type(_frame_type),
 178         window_type(type),
 179         owner(_owner),
 180         jview(NULL),
 181         map_received(false),
 182         visible_received(false),
 183         on_top(false),
 184         is_fullscreen(false),
 185         is_iconified(false),
 186         is_maximized(false),
 187         is_mouse_entered(false),
 188         can_be_deleted(false),
 189         events_processing_cnt(0),
 190         grab_pointer(NULL) {
 191 
 192     jwindow = mainEnv->NewGlobalRef(_jwindow);
 193 
 194     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 195 
 196     if (gchar * app_name = get_application_name()) {
 197         gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 198         g_free(app_name);
 199     }
 200 
 201     if (owner) {
 202         owner->add_child(this);
 203         if (on_top_inherited()) {
 204             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 205         }
 206     }
 207 
 208     if (type == UTILITY) {
 209         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 210     }
 211 
 212     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 213 
 214     if (xvisualID != 0) {
 215         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 216         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 217     }
 218 
 219     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 220     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 221 
 222     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 223     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 224 
 225     gdk_window = gtk_widget_get_window(gtk_widget);
 226     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 227 
 228     glass_dnd_attach_context(this);
 229 
 230     gdk_windowManagerFunctions = wmf;
 231     if (wmf) {
 232         gdk_window_set_functions(gdk_window, wmf);
 233     }
 234 
 235     if (frame_type != TITLED) {
 236         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 237     }
 238 
 239     connect_signals(gtk_widget, this);
 240 }

 242 void WindowContext::paint(void *data, jint width, jint height) {
 243 #if GTK_CHECK_VERSION(3, 0, 0)
 244     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 245 #if GTK_CHECK_VERSION(3, 22, 0)
 246     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 247     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 248 #else
 249     gdk_window_begin_paint_region(gdk_window, region);
 250     cairo_t* context = gdk_cairo_create(gdk_window);
 251 #endif
 252 #else
 253     cairo_t *context = gdk_cairo_create(gdk_window);
 254 #endif
 255 
 256     if (bg_color.is_set) {
 257         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 258                                 (frame_type == TRANSPARENT) ? 0 : 1);
 259         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 260         cairo_paint(context);
 261     }
 262 
 263     cairo_surface_t *cairo_surface;
 264     cairo_surface = cairo_image_surface_create_for_data(
 265             (unsigned char *) data,
 266             CAIRO_FORMAT_ARGB32,
 267             width, height, width * 4);
 268 
 269     cairo_set_source_surface(context, cairo_surface, 0, 0);
 270 
 271     applyShapeMask(data, width, height);
 272     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 273     cairo_paint(context);
 274 
 275 #if GTK_CHECK_VERSION(3, 0, 0)
 276 #if GTK_CHECK_VERSION(3, 22, 0)
 277     gdk_window_end_draw_frame(gdk_window, dcontext);
 278     cairo_region_destroy(region);
 279 #else
 280     gdk_window_end_paint(gdk_window);
 281     cairo_region_destroy(region);
 282     cairo_destroy(context);
 283 #endif
 284 #else
 285     cairo_destroy(context);
 286 #endif
 287 
 288     cairo_surface_destroy(cairo_surface);
 289 }

 331 void WindowContext::process_focus(GdkEventFocus *event) {
 332     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 333         ungrab_mouse_drag_focus();
 334     }
 335 
 336     if (!event->in && WindowContext::sm_grab_window == this) {
 337         ungrab_focus();
 338     }
 339 
 340     if (xim.enabled && xim.ic) {
 341         if (event->in) {
 342             XSetICFocus(xim.ic);
 343         } else {
 344             XUnsetICFocus(xim.ic);
 345         }
 346     }
 347 
 348     if (jwindow) {
 349         if (!event->in || isEnabled()) {
 350             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 351                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 352                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 353             CHECK_JNI_EXCEPTION(mainEnv)
 354         } else {
 355             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 356             CHECK_JNI_EXCEPTION(mainEnv)
 357         }
 358     }
 359 }

 361 void WindowContext::process_property_notify(GdkEventProperty *event) {
 362     if (event->window == gdk_window) {

 363         if (event->atom == atom_net_wm_state) {
 364             process_net_wm_property();
 365         } else if (event->atom == atom_net_wm_frame_extents) {
 366             if (frame_type != TITLED) {
 367                 return;
 368             }
 369 
 370             int top, left, bottom, right;
 371 
 372             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 373                 if (top + left + bottom + right > 0) {
 374                     geometry.frame_extents_received = true;
 375                     geometry.adjust_w = left + right;
 376                     geometry.adjust_h = top + bottom;
 377                     geometry.view_x = left;
 378                     geometry.view_y = top;
 379 
 380                     // set bounds again to set to correct window size that must
 381                     // be the total width and height accounting extents
 382                     // this is ignored if size is "content size" instead of "window size"
 383                     if (geometry.needs_ajustment) {
 384                         set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
 385                     }
 386 
 387                     // force position notify so java will know about view_y and view_x
 388                     size_position_notify(false, true);
 389                 }
 390             }
 391         }
 392     }
 393 }

 395 void WindowContext::process_configure() {
 396     gint x, y, w, h, gtk_w, gtk_h;
 397 
 398     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 399     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 400 
 401     w = gtk_w + geometry.adjust_w;
 402     h = gtk_h + geometry.adjust_h;
 403 
 404     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 405     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 406                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 407 
 408     geometry.current_x = x;
 409     geometry.current_y = y;
 410     geometry.current_w = w;
 411     geometry.current_h = h;
 412     geometry.current_cw = gtk_w;
 413     geometry.current_ch = gtk_h;
 414 
 415     if (!is_fullscreen && !is_maximized) {
 416         geometry.last_cw = gtk_w;
 417         geometry.last_ch = gtk_h;
 418     }
 419 
 420     size_position_notify(size_changed, pos_changed);
 421 }

 423 void WindowContext::process_destroy() {
 424     if (owner) {
 425         owner->remove_child(this);
 426     }
 427 
 428     if (WindowContext::sm_mouse_drag_window == this) {
 429         ungrab_mouse_drag_focus();
 430     }
 431 
 432     if (WindowContext::sm_grab_window == this) {
 433         ungrab_focus();
 434     }
 435 
 436     std::set<WindowContext *>::iterator it;
 437     for (it = children.begin(); it != children.end(); ++it) {
 438         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 439         // WindowContext::process_destroy() to call remove_child() (because children
 440         // is being iterated here) but also prevents gtk_window_set_transient_for from
 441         // being called - this causes the crash on gnome.
 442         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 443         (*it)->set_owner(NULL);
 444         destroy_and_delete_ctx(*it);
 445     }
 446     children.clear();
 447 
 448     if (jwindow) {
 449         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 450         EXCEPTION_OCCURED(mainEnv);
 451     }
 452 
 453     if (jview) {
 454         mainEnv->DeleteGlobalRef(jview);
 455         jview = NULL;
 456     }
 457 
 458     if (jwindow) {
 459         mainEnv->DeleteGlobalRef(jwindow);
 460         jwindow = NULL;
 461     }
 462 
 463     can_be_deleted = true;
 464 }

 482 void WindowContext::process_mouse_button(GdkEventButton *event) {
 483     bool press = event->type == GDK_BUTTON_PRESS;
 484     bool release = event->type == GDK_BUTTON_RELEASE;
 485 
 486     guint state = event->state;
 487     guint mask = 0;
 488 
 489     // We need to add/remove current mouse button from the modifier flags
 490     // as X lib state represents the state just prior to the event and
 491     // glass needs the state just after the event
 492     switch (event->button) {
 493         case 1:
 494             mask = GDK_BUTTON1_MASK;
 495             break;
 496         case 2:
 497             mask = GDK_BUTTON2_MASK;
 498             break;
 499         case 3:
 500             mask = GDK_BUTTON3_MASK;
 501             break;
 502         case MOUSE_BACK_BTN:
 503             mask = GDK_BUTTON4_MASK;
 504             break;
 505         case MOUSE_FORWARD_BTN:
 506             mask = GDK_BUTTON5_MASK;
 507             break;
 508     }
 509 
 510     if (press) {
 511         state |= mask;
 512     } else {
 513         state &= ~mask;
 514     }
 515 
 516     // Upper layers expects from us Windows behavior:
 517     // all mouse events should be delivered to window where drag begins
 518     // and no exit/enter event should be reported during this drag.
 519     // We can grab mouse pointer for these needs.
 520     if (press) {
 521         grab_mouse_drag_focus(gdk_window, (GdkEvent *) event, NULL, true);
 522     } else {
 523         if ((event->state & MOUSE_BUTTONS_MASK)
 524             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 525             ungrab_mouse_drag_focus();
 526         } else if (event->button == 8 || event->button == 9) {
 527             // GDK X backend interprets button press events for buttons 4-7 as
 528             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 529             // be set on the event->state from GDK. Thus we cannot check if all
 530             // buttons have been released in the usual way (as above).
 531             ungrab_mouse_drag_focus();
 532         }
 533     }
 534 
 535     jint button = gtk_button_number_to_mouse_button(event->button);
 536 
 537     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 538         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 539                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 540                                 button,
 541                                 (jint) event->x, (jint) event->y,
 542                                 (jint) event->x_root, (jint) event->y_root,
 543                                 gdk_modifier_mask_to_glass(state),
 544                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 545                                 JNI_FALSE);
 546         CHECK_JNI_EXCEPTION(mainEnv)
 547 
 548         if (jview && event->button == 3 && press) {
 549             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 550                                     (jint) event->x, (jint) event->y,
 551                                     (jint) event->x_root, (jint) event->y_root,
 552                                     JNI_FALSE);
 553             CHECK_JNI_EXCEPTION(mainEnv)
 554         }
 555     }
 556 }

 558 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 559     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 560     jint isDrag = glass_modifier & (
 561             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 562             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 563             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 564             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 565             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 566     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 567 
 568     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 569         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 570     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 571         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 572     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 573         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 574     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 575         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 576     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 577         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 578     }
 579 
 580     if (jview) {
 581         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 582                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 583                                 button,
 584                                 (jint) event->x, (jint) event->y,
 585                                 (jint) event->x_root, (jint) event->y_root,
 586                                 glass_modifier,
 587                                 JNI_FALSE,
 588                                 JNI_FALSE);
 589         CHECK_JNI_EXCEPTION(mainEnv)
 590     }
 591 }

 593 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {

 660 void WindowContext::process_key(GdkEventKey *event) {
 661     bool press = event->type == GDK_KEY_PRESS;
 662     jint glassKey = get_glass_key(event);
 663     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 664     if (press) {
 665         glassModifier |= glass_key_to_modifier(glassKey);
 666     } else {
 667         glassModifier &= ~glass_key_to_modifier(glassKey);
 668     }
 669     jcharArray jChars = NULL;
 670     jchar key = gdk_keyval_to_unicode(event->keyval);
 671     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 672         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 673     } else {
 674 #ifdef GLASS_GTK2
 675         if (key == 0) {
 676             // Work around "bug" fixed in gtk-3.0:
 677             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 678             switch (event->keyval) {
 679             case 0xFF08 /* Backspace */: key =  '\b';
 680             case 0xFF09 /* Tab       */: key =  '\t';
 681             case 0xFF0A /* Linefeed  */: key =  '\n';
 682             case 0xFF0B /* Vert. Tab */: key =  '\v';
 683             case 0xFF0D /* Return    */: key =  '\r';
 684             case 0xFF1B /* Escape    */: key =  '\033';
 685             case 0xFFFF /* Delete    */: key =  '\177';
 686             }
 687         }
 688 #endif
 689     }
 690 
 691     if (key > 0) {
 692         jChars = mainEnv->NewCharArray(1);
 693         if (jChars) {
 694             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 695             CHECK_JNI_EXCEPTION(mainEnv)
 696         }
 697     } else {
 698         jChars = mainEnv->NewCharArray(0);
 699     }
 700     if (jview) {
 701         if (press) {
 702             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 703                                     com_sun_glass_events_KeyEvent_PRESS,
 704                                     glassKey,
 705                                     jChars,
 706                                     glassModifier);
 707             CHECK_JNI_EXCEPTION(mainEnv)
 708 
 709             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 710                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 711                                         com_sun_glass_events_KeyEvent_TYPED,
 712                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 713                                         jChars,
 714                                         glassModifier);
 715                 CHECK_JNI_EXCEPTION(mainEnv)
 716             }
 717         } else {
 718             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 719                                     com_sun_glass_events_KeyEvent_RELEASE,
 720                                     glassKey,
 721                                     jChars,
 722                                     glassModifier);
 723             CHECK_JNI_EXCEPTION(mainEnv)
 724         }
 725     }

 728 void WindowContext::process_state(GdkEventWindowState *event) {
 729     if (event->changed_mask &
 730         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 731 
 732         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 733             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 734         }
 735         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 736             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 737         }
 738 
 739         jint stateChangeEvent;
 740 
 741         if (is_iconified) {
 742             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 743         } else if (is_maximized) {
 744             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 745         } else {

 750                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 751             }
 752         }
 753 
 754         notify_state(stateChangeEvent);
 755     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 756         notify_on_top(event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 757     }

 982 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 983     // this will tell if adjustments are needed - that's because GTK does not have full window size

 985     geometry.needs_ajustment = (w > 0 || h > 0) || geometry.needs_ajustment;
 986

 999         newH = h - geometry.adjust_h;
1000         geometry.current_h = newH;

1020             apply_geometry();

1244 // This function calculate the deltas between window and window + decoration (titlebar, borders).
1245 // It's used when the window manager does not support the _NET_FRAME_EXTENTS extension or when
1246 // it's not received on time.
1247 void WindowContext::calculate_adjustments() {
1248     if (frame_type != TITLED || geometry.frame_extents_received) {
1249         return;
1250     }
1251 
1252     gint x, y, rx, ry;
1253     gdk_window_get_origin(gdk_window, &x, &y);
1254     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1255 
1256     if (rx != x || ry != y) {
1257         // the left extends are correct - the right one is guessed to be the same
1258         geometry.adjust_w = (x - rx) * 2;
1259         // guess that bottom size is the same as left and right
1260         geometry.adjust_h = (y - ry) + (x - rx);
1261 
1262         // those will be correct
1263         geometry.view_x = (x - rx);
1264         geometry.view_y = (y - ry);
1265 
1266         if (geometry.needs_ajustment) {
1267             set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
1268         }
1269 
1270         // force position notify so java will know about view_y and view_x
1271         size_position_notify(false, true);
1272     }

1275 void WindowContext::apply_geometry() {
1276     if (!map_received) {
1277         return;
1278     }
1279 
1280     GdkGeometry gdk_geometry;
1281     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
1282 
1283     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1284         // not resizeable
1285         int w = geometry.current_cw > 0
1286                 ? geometry.current_cw
1287                 : geometry.current_w - geometry.adjust_w;
1288 
1289         int h = geometry.current_ch > 0
1290                 ? geometry.current_ch
1291                 : geometry.current_h - geometry.adjust_h;
1292 
1293         gdk_geometry.min_width = gdk_geometry.max_width = w;
1294         gdk_geometry.min_height = gdk_geometry.max_height = h;
1295     } else {
1296         //min/max width/height always whole window size (with decors)
1297         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1298                                  ? geometry.minw - geometry.adjust_w : 1;
1299         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1300                                   ? geometry.minh - geometry.adjust_h : 1;

1303                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1304         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1305                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1306     }
1307 
1308     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
