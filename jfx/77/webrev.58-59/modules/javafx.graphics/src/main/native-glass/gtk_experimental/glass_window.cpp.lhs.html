 135 void destroy_and_delete_ctx(WindowContext *ctx) {
 136     if (ctx) {
 137         ctx->process_destroy();
 138 
 139         if (!ctx->get_events_count()) {
 140             delete ctx;
 141         }
 142         // else: ctx will be deleted in EventsCounterHelper after completing
 143         // an event processing
 144     }
 145 }

 147 static inline jint gtk_button_number_to_mouse_button(guint button) {
 148     switch (button) {
 149         case 1:
 150             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 151         case 2:
 152             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 153         case 3:
 154             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 155         case MOUSE_BACK_BTN:
 156             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 157         case MOUSE_FORWARD_BTN:
 158             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 159         default:
 160             // Other buttons are not supported by quantum and are not reported by other platforms
 161             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 162     }
 163 }

 173 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 174                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 175         screen(_screen),
 176         frame_type(_frame_type),
 177         window_type(type),
 178         owner(_owner),
 179         jview(NULL),
 180         map_received(false),
 181         visible_received(false),
 182         on_top(false),
 183         is_fullscreen(false),
 184         is_iconified(false),
 185         is_maximized(false),
 186         is_mouse_entered(false),
 187         can_be_deleted(false),
 188         events_processing_cnt(0),
 189         grab_pointer(NULL) {
 190 
 191     jwindow = mainEnv->NewGlobalRef(_jwindow);
 192 
 193     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 194 
 195 // Not useful, see: https://developer.gnome.org/gtk3/stable/GtkWindow.html#gtk-window-set-wmclass
 196 //    if (gchar * app_name = get_application_name()) {
 197 //        gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 198 //        g_free(app_name);
 199 //    }
 200 
 201     if (owner) {
 202         owner->add_child(this);
 203         if (on_top_inherited()) {
 204             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 205         }
 206     }
 207 
 208     if (type == UTILITY) {
 209         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 210     }
 211 
 212     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 213 
 214     if (xvisualID != 0) {
 215         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 216         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 217     }
 218 
 219     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 220     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 221 
 222     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 223     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 224 
 225     gdk_window = gtk_widget_get_window(gtk_widget);
 226     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 227 
 228     glass_dnd_attach_context(this);
 229 
 230     gdk_windowManagerFunctions = wmf;
 231     if (wmf) {
 232         gdk_window_set_functions(gdk_window, wmf);
 233     }
 234 
 235     if (frame_type != TITLED) {
 236         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 237     }
 238 
 239     connect_signals(gtk_widget, this);
 240 }

 242 void WindowContext::paint(void *data, jint width, jint height) {
 243 #ifdef GLASS_GTK3
 244     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 245     gdk_window_begin_paint_region(gdk_window, region);
 246     cairo_t* context = gdk_cairo_create(gdk_window);
 247 #else
 248     cairo_t *context = gdk_cairo_create(gdk_window);
 249 #endif
 250 
 251     if (bg_color.is_set) {
 252         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 253                                 (frame_type == TRANSPARENT) ? 0 : 1);
 254         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 255         cairo_paint(context);
 256     }
 257 
 258     cairo_surface_t *cairo_surface;
 259     cairo_surface = cairo_image_surface_create_for_data(
 260             (unsigned char *) data,
 261             CAIRO_FORMAT_ARGB32,
 262             width, height, width * 4);
 263 
 264     cairo_set_source_surface(context, cairo_surface, 0, 0);
 265 
 266     applyShapeMask(data, width, height);
 267     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 268     cairo_paint(context);
 269 
 270 #ifdef GLASS_GTK3
 271     gdk_window_end_paint(gdk_window);
 272     cairo_region_destroy(region);
 273     cairo_destroy(context);
 274 #else
 275     cairo_destroy(context);
 276 #endif
 277 
 278     cairo_surface_destroy(cairo_surface);
 279 }

 321 void WindowContext::process_focus(GdkEventFocus *event) {
 322     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 323         ungrab_mouse_drag_focus();
 324     }
 325 
 326     if (!event->in && WindowContext::sm_grab_window == this) {
 327         ungrab_focus();
 328     }
 329 
 330     if (xim.enabled && xim.ic) {
 331         if (event->in) {
 332             XSetICFocus(xim.ic);
 333         } else {
 334             XUnsetICFocus(xim.ic);
 335         }
 336     }
 337 
 338     if (jwindow) {
 339         if (!event->in || isEnabled()) {
 340             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 341                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 342                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 343             CHECK_JNI_EXCEPTION(mainEnv)
 344         } else {
 345             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 346             CHECK_JNI_EXCEPTION(mainEnv)
 347         }
 348     }
 349 }

 351 void WindowContext::process_property_notify(GdkEventProperty *event) {
 352     if (event->window == gdk_window) {
 353         // This work-around is only necessary for Unity
 354         if (event->atom == atom_net_wm_state) {
 355             process_net_wm_property();
 356         } else if (event->atom == atom_net_wm_frame_extents) {
 357             if (frame_type != TITLED) {
 358                 return;
 359             }
 360 
 361             int top, left, bottom, right;
 362 
 363             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 364                 if (top + left + bottom + right > 0) {
 365                     geometry.frame_extents_received = true;
 366                     geometry.adjust_w = left + right;
 367                     geometry.adjust_h = top + bottom;
 368                     geometry.view_x = left;
 369                     geometry.view_y = top;
 370 
 371                     // set bounds again to set to correct window size that must
 372                     // be the total width and height accounting extents
 373                     // this is ignored if size is "content size" instead of "window size"
 374                     if (geometry.needs_ajustment) {
 375                         set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
 376                     }
 377 
 378                     // force position notify so java will know about view_y and view_x
 379                     size_position_notify(false, true);
 380                 }
 381             }
 382         }
 383     }
 384 }

 386 void WindowContext::process_configure() {
 387     gint x, y, w, h, gtk_w, gtk_h;
 388 
 389     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 390     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 391 
 392     w = gtk_w + geometry.adjust_w;
 393     h = gtk_h + geometry.adjust_h;
 394 
 395     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 396     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 397                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 398 
 399     geometry.current_x = x;
 400     geometry.current_y = y;
 401     geometry.current_w = w;
 402     geometry.current_h = h;
 403     geometry.current_cw = gtk_w;
 404     geometry.current_ch = gtk_h;
 405 
 406     if (!is_fullscreen && !is_maximized) {
 407         geometry.last_cw = gtk_w;
 408         geometry.last_ch = gtk_h;
 409     }
 410 
 411     size_position_notify(size_changed, pos_changed);
 412 }

 414 void WindowContext::process_destroy() {
 415     if (owner) {
 416         owner->remove_child(this);
 417     }
 418 
 419     if (WindowContext::sm_mouse_drag_window == this) {
 420         ungrab_mouse_drag_focus();
 421     }
 422 
 423     if (WindowContext::sm_grab_window == this) {
 424         ungrab_focus();
 425     }
 426 
 427     std::set<WindowContext *>::iterator it;
 428     for (it = children.begin(); it != children.end(); ++it) {
 429         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 430         // WindowContext::process_destroy() to call remove_child() (because children
 431         // is being iterated here) but also prevents gtk_window_set_transient_for from
 432         // being called - this causes the crash on gnome.
 433         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 434         (*it)->set_owner(NULL);
 435         destroy_and_delete_ctx(*it);
 436     }
 437     children.clear();
 438 
 439     if (jwindow) {
 440         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 441         EXCEPTION_OCCURED(mainEnv);
 442     }
 443 
 444     if (jview) {
 445         mainEnv->DeleteGlobalRef(jview);
 446         jview = NULL;
 447     }
 448 
 449     if (jwindow) {
 450         mainEnv->DeleteGlobalRef(jwindow);
 451         jwindow = NULL;
 452     }
 453 
 454     can_be_deleted = true;
 455 }

 457 void WindowContext::process_delete() {
 458     if (jwindow && isEnabled()) {
 459         gtk_widget_hide_on_delete(gtk_widget);
 460         mainEnv->CallVoidMethod(jwindow, jWindowNotifyClose);
 461         CHECK_JNI_EXCEPTION(mainEnv)
 462     }

 473 void WindowContext::process_mouse_button(GdkEventButton *event) {





 474     bool press = event->type == GDK_BUTTON_PRESS;
 475 
 476     guint state = event->state;
 477     guint mask = 0;
 478 
 479     // We need to add/remove current mouse button from the modifier flags
 480     // as X lib state represents the state just prior to the event and
 481     // glass needs the state just after the event
 482     switch (event->button) {
 483         case 1:
 484             mask = GDK_BUTTON1_MASK;
 485             break;
 486         case 2:
 487             mask = GDK_BUTTON2_MASK;
 488             break;
 489         case 3:
 490             mask = GDK_BUTTON3_MASK;
 491             break;
 492         case MOUSE_BACK_BTN:
 493             mask = GDK_BUTTON4_MASK;
 494             break;
 495         case MOUSE_FORWARD_BTN:
 496             mask = GDK_BUTTON5_MASK;
 497             break;
 498     }
 499 
 500     if (press) {
 501         state |= mask;
 502     } else {
 503         state &= ~mask;
 504     }
 505 
 506     if (press) {
 507         GdkDevice* device = event->device;
 508 
 509         if (glass_gdk_device_is_grabbed(device)
 510                 && (glass_gdk_device_get_window_at_position(device, NULL, NULL)
 511                 == NULL)) {
 512             ungrab_focus();
 513             return;
 514         }
 515     }
 516 
 517     // Upper layers expects from us Windows behavior:
 518     // all mouse events should be delivered to window where drag begins
 519     // and no exit/enter event should be reported during this drag.
 520     // We can grab mouse pointer for these needs.
 521     if (press) {
 522         grab_mouse_drag_focus(gdk_window, (GdkEvent *) event, NULL, true);
 523     } else {
 524         if ((event->state & MOUSE_BUTTONS_MASK)
 525             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 526             ungrab_mouse_drag_focus();
 527         } else if (event->button == 8 || event->button == 9) {
 528             // GDK X backend interprets button press events for buttons 4-7 as
 529             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 530             // be set on the event->state from GDK. Thus we cannot check if all
 531             // buttons have been released in the usual way (as above).
 532             ungrab_mouse_drag_focus();
 533         }
 534     }
 535 

 536     jint button = gtk_button_number_to_mouse_button(event->button);
 537 
 538     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 539         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 540                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 541                                 button,
 542                                 (jint) event->x, (jint) event->y,
 543                                 (jint) event->x_root, (jint) event->y_root,
 544                                 gdk_modifier_mask_to_glass(state),
 545                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 546                                 JNI_FALSE);
 547         CHECK_JNI_EXCEPTION(mainEnv)
 548 
 549         if (jview && event->button == 3 && press) {
 550             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 551                                     (jint) event->x, (jint) event->y,
 552                                     (jint) event->x_root, (jint) event->y_root,
 553                                     JNI_FALSE);
 554             CHECK_JNI_EXCEPTION(mainEnv)
 555         }
 556     }
 557 }

 559 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 560     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 561     jint isDrag = glass_modifier & (
 562             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 563             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 564             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 565             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 566             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 567     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 568 
 569     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 570         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 571     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 572         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 573     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 574         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 575     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 576         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 577     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 578         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 579     }
 580 
 581     if (jview) {
 582         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 583                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 584                                 button,
 585                                 (jint) event->x, (jint) event->y,
 586                                 (jint) event->x_root, (jint) event->y_root,
 587                                 glass_modifier,
 588                                 JNI_FALSE,
 589                                 JNI_FALSE);
 590         CHECK_JNI_EXCEPTION(mainEnv)
 591     }
 592 
 593     gdk_event_request_motions(event);
 594 }

 596 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 597     jdouble dx = 0, dy = 0;
 598 
 599     // converting direction to change in pixels
 600     switch (event->direction) {
 601         case GDK_SCROLL_UP:
 602             dy = 1;
 603             break;
 604         case GDK_SCROLL_DOWN:
 605             dy = -1;
 606             break;
 607         case GDK_SCROLL_LEFT:
 608             dx = 1;
 609             break;
 610         case GDK_SCROLL_RIGHT:
 611             dx = -1;
 612             break;
 613         default:
 614             break;
 615     }
 616     if (event->state & GDK_SHIFT_MASK) {
 617         jdouble t = dy;
 618         dy = dx;
 619         dx = t;
 620     }
 621     if (jview) {
 622         mainEnv->CallVoidMethod(jview, jViewNotifyScroll,
 623                                 (jint) event->x, (jint) event->y,
 624                                 (jint) event->x_root, (jint) event->y_root,
 625                                 dx, dy,
 626                                 gdk_modifier_mask_to_glass(event->state),
 627                                 (jint) 0, (jint) 0,
 628                                 (jint) 0, (jint) 0,
 629                                 (jdouble) 40.0, (jdouble) 40.0);
 630         CHECK_JNI_EXCEPTION(mainEnv)
 631     }
 632 }

 659 void WindowContext::process_key(GdkEventKey *event) {
 660     bool press = event->type == GDK_KEY_PRESS;
 661     jint glassKey = get_glass_key(event);
 662     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 663     if (press) {
 664         glassModifier |= glass_key_to_modifier(glassKey);
 665     } else {
 666         glassModifier &= ~glass_key_to_modifier(glassKey);
 667     }
 668     jcharArray jChars = NULL;
 669     jchar key = gdk_keyval_to_unicode(event->keyval);
 670     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 671         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 672     } else {
 673 #ifdef GLASS_GTK2
 674         if (key == 0) {
 675             // Work around "bug" fixed in gtk-3.0:
 676             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 677             switch (event->keyval) {
 678             case 0xFF08 /* Backspace */: key =  '\b';
 679             case 0xFF09 /* Tab       */: key =  '\t';
 680             case 0xFF0A /* Linefeed  */: key =  '\n';
 681             case 0xFF0B /* Vert. Tab */: key =  '\v';
 682             case 0xFF0D /* Return    */: key =  '\r';
 683             case 0xFF1B /* Escape    */: key =  '\033';
 684             case 0xFFFF /* Delete    */: key =  '\177';
 685             }
 686         }
 687 #endif
 688     }
 689 
 690     if (key > 0) {
 691         jChars = mainEnv->NewCharArray(1);
 692         if (jChars) {
 693             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 694             CHECK_JNI_EXCEPTION(mainEnv)
 695         }
 696     } else {
 697         jChars = mainEnv->NewCharArray(0);
 698     }
 699     if (jview) {
 700         if (press) {
 701             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 702                                     com_sun_glass_events_KeyEvent_PRESS,
 703                                     glassKey,
 704                                     jChars,
 705                                     glassModifier);
 706             CHECK_JNI_EXCEPTION(mainEnv)
 707 
 708             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 709                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 710                                         com_sun_glass_events_KeyEvent_TYPED,
 711                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 712                                         jChars,
 713                                         glassModifier);
 714                 CHECK_JNI_EXCEPTION(mainEnv)
 715             }
 716         } else {
 717             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 718                                     com_sun_glass_events_KeyEvent_RELEASE,
 719                                     glassKey,
 720                                     jChars,
 721                                     glassModifier);
 722             CHECK_JNI_EXCEPTION(mainEnv)
 723         }
 724     }
 725 }

 727 void WindowContext::process_state(GdkEventWindowState *event) {
 728     if (event->changed_mask &
 729         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 730 
 731         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 732             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 733         }
 734         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 735             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 736         }
 737 
 738         jint stateChangeEvent;
 739 
 740         if (is_iconified) {
 741             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 742         } else if (is_maximized) {
 743             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 744         } else {
 745             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 746             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 747                 // in this case - the window manager will not support the programatic
 748                 // request to iconify - so we need to restore it now.
 749                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 750             }
 751         }
 752 
 753         notify_state(stateChangeEvent);
 754     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 755         notify_on_top(event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 756     }
 757 }

 759 void WindowContext::process_net_wm_property() {
 760     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 761 
 762     // This is a Unity bug
 763     if (!g_strcmp0("Unity", gdk_x11_screen_get_window_manager_name(gdk_screen_get_default()))) {
 764         return;
 765     }
 766 
 767     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 768     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 769     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 770     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 771 
 772     gint length;
 773     glong *atoms = NULL;
 774 
 775     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 776                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 777 
 778         bool is_hidden = false;
 779         bool is_above = false;
 780         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 781             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 782                 is_hidden = true;
 783             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 784                 is_above = true;
 785             }
 786         }
 787 
 788         g_free(atoms);
 789 
 790         if (is_iconified != is_hidden) {
 791             is_iconified = is_hidden;
 792 
 793             notify_state((is_hidden)
 794                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 795                          : com_sun_glass_events_WindowEvent_RESTORE);
 796         }
 797 
 798         notify_on_top(is_above);
 799     }

 985 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 986     // this will tell if adjustments are needed - that's because GTK does not have full window size
 987     // values, just content values. Frame extents (window decorations) are handled by the window manager.
 988     geometry.needs_ajustment = (w > 0 || h > 0) || geometry.needs_ajustment;
 989 
 990     // newW / newH always content sizes compatible with GTK+
 991     // if window has no decoration, adjustments will be ZERO
 992     int newW, newH;
 993     newW = (w > 0) ? w - geometry.adjust_w : cw;
 994     newH = (h > 0) ? h - geometry.adjust_h : ch;
 995 
 996     geometry.current_w = newW;
 997     geometry.current_h = newH;
 998 
 999     gboolean size_changed = FALSE;
1000     gboolean pos_changed = FALSE;
1001 
1002     if (newW > 0 && newH > 0) {
1003         size_changed = TRUE;
1004 
1005         // content size
1006         geometry.current_cw = newW;
1007         geometry.current_ch = newH;
1008         geometry.last_cw = newW;
1009         geometry.last_ch = newH;
1010 
1011         if (visible_received) {
1012             // call apply_geometry() to let gtk_window_resize succeed, because it's bound to
1013             // geometry constraints
1014             apply_geometry();
1015             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
1016         } else {
1017             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
1018         }
1019     }
1020 
1021     if (xSet || ySet) {
1022         int newX = (xSet) ? x : geometry.current_x;
1023         int newY = (ySet) ? y : geometry.current_y;
1024 
1025         if (newX != geometry.current_x || newY != geometry.current_y) {
1026             pos_changed = TRUE;
1027             geometry.current_x = newX;
1028             geometry.current_y = newY;
1029 
1030             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1031         }
1032     }
1033 
1034     size_position_notify(size_changed, pos_changed);
1035 }

1241 // This function calculate the deltas between window and window + decoration (titlebar, borders).
1242 // It's used when the window manager does not support the _NET_FRAME_EXTENTS extension or when
1243 // it's not received on time.
1244 void WindowContext::calculate_adjustments() {
1245     if (frame_type != TITLED || geometry.frame_extents_received) {
1246         return;
1247     }
1248 
1249     gint x, y, rx, ry;
1250     gdk_window_get_origin(gdk_window, &x, &y);
1251     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1252 
1253     if (rx != x || ry != y) {
1254         // the left extends are correct - the right one is guessed to be the same
1255         geometry.adjust_w = (x - rx) * 2;
1256         // guess that bottom size is the same as left and right
1257         geometry.adjust_h = (y - ry) + (x - rx);
1258 
1259         // those will be correct
1260         geometry.view_x = (x - rx);
1261         geometry.view_y = (y - ry);
1262 
1263         if (geometry.needs_ajustment) {
1264             set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
1265         }
1266 
1267         // force position notify so java will know about view_y and view_x
1268         size_position_notify(false, true);
1269     }
1270 }

1272 void WindowContext::apply_geometry() {
1273     if (!map_received) {
1274         return;
1275     }
1276 
1277     GdkGeometry gdk_geometry;
1278     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
1279 
1280     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1281         // not resizeable
1282         int w = geometry.current_cw > 0
1283                 ? geometry.current_cw
1284                 : geometry.current_w - geometry.adjust_w;
1285 
1286         int h = geometry.current_ch > 0
1287                 ? geometry.current_ch
1288                 : geometry.current_h - geometry.adjust_h;
1289 
1290         gdk_geometry.min_width = gdk_geometry.max_width = w;
1291         gdk_geometry.min_height = gdk_geometry.max_height = h;
1292     } else {
1293         //min/max width/height always whole window size (with decors)
1294         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1295                                  ? geometry.minw - geometry.adjust_w : 1;
1296         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1297                                   ? geometry.minh - geometry.adjust_h : 1;
1298 
1299         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
1300                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1301         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1302                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1303     }
1304 
1305     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1306                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
1307 }

1335 void WindowContext::activate_window() {
1336     Display *display = GDK_DISPLAY_XDISPLAY(gdk_window_get_display(gdk_window));
1337     Atom navAtom = XInternAtom(display, "_NET_ACTIVE_WINDOW", True);
1338     if (navAtom != None) {
1339         XClientMessageEvent clientMessage;
1340         memset(&clientMessage, 0, sizeof(clientMessage));
1341 
1342         clientMessage.type = ClientMessage;
1343         clientMessage.window = GDK_WINDOW_XID(gdk_window);
1344         clientMessage.message_type = navAtom;
1345         clientMessage.format = 32;

1351                    SubstructureRedirectMask | SubstructureNotifyMask,

1357 void WindowContext::size_position_notify(bool size_changed, bool pos_changed) {
1358 
1359     if (jview) {
1360         if (size_changed) {
1361             mainEnv->CallVoidMethod(jview, jViewNotifyResize, geometry.current_cw, geometry.current_ch);
1362             CHECK_JNI_EXCEPTION(mainEnv);
1363         }
1364 
1365         if (pos_changed) {
1366             mainEnv->CallVoidMethod(jview, jViewNotifyView, com_sun_glass_events_ViewEvent_MOVE);
1367             CHECK_JNI_EXCEPTION(mainEnv)
1368         }
1369     }
1370 
1371     if (jwindow) {
1372         if (size_changed || is_maximized) {
1373             mainEnv->CallVoidMethod(jwindow, jWindowNotifyResize,
1374                                     (is_maximized)
1375                                     ? com_sun_glass_events_WindowEvent_MAXIMIZE
1376                                     : com_sun_glass_events_WindowEvent_RESIZE,
1377                                     geometry.current_w, geometry.current_h);
1378             CHECK_JNI_EXCEPTION(mainEnv)
1379         }
1380 
1381         if (pos_changed) {
1382             mainEnv->CallVoidMethod(jwindow, jWindowNotifyMove, geometry.current_x, geometry.current_y);
1383             CHECK_JNI_EXCEPTION(mainEnv)
1384         }
1385     }
