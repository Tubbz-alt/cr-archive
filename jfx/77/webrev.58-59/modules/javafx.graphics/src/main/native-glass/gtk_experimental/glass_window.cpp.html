 135 void destroy_and_delete_ctx(WindowContext *ctx) {
 136     if (ctx) {
 137         ctx->process_destroy();
 138 
 139         if (!ctx->get_events_count()) {
 140             delete ctx;
 141         }
 142         // else: ctx will be deleted in EventsCounterHelper after completing
 143         // an event processing
 144     }
 145 }

 147 static inline jint gtk_button_number_to_mouse_button(guint button) {
 148     switch (button) {
 149         case 1:
 150             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 151         case 2:
 152             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 153         case 3:
 154             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 155         case MOUSE_BACK_BTN:
 156             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 157         case MOUSE_FORWARD_BTN:
 158             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 159         default:
 160             // Other buttons are not supported by quantum and are not reported by other platforms
 161             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 162     }
 163 }

 173 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 174                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 175         screen(_screen),
 176         frame_type(_frame_type),
 177         window_type(type),
 178         owner(_owner),
 179         jview(NULL),
 180         map_received(false),
 181         visible_received(false),
 182         on_top(false),
 183         is_fullscreen(false),
 184         is_iconified(false),
 185         is_maximized(false),
 186         is_mouse_entered(false),
 187         can_be_deleted(false),
 188         events_processing_cnt(0),
 189         grab_pointer(NULL) {
 190 
 191     jwindow = mainEnv->NewGlobalRef(_jwindow);
 192 
 193     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 194 
 195 // Not useful, see: https://developer.gnome.org/gtk3/stable/GtkWindow.html#gtk-window-set-wmclass
 196 //    if (gchar * app_name = get_application_name()) {
 197 //        gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 198 //        g_free(app_name);
 199 //    }
 200 
 201     if (owner) {
 202         owner->add_child(this);
 203         if (on_top_inherited()) {
 204             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 205         }
 206     }
 207 
 208     if (type == UTILITY) {
 209         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 210     }
 211 
 212     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 213 
 214     if (xvisualID != 0) {
 215         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 216         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 217     }
 218 
 219     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 220     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 221 
 222     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 223     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 224 
 225     gdk_window = gtk_widget_get_window(gtk_widget);
 226     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 227 
 228     glass_dnd_attach_context(this);
 229 
 230     gdk_windowManagerFunctions = wmf;
 231     if (wmf) {
 232         gdk_window_set_functions(gdk_window, wmf);
 233     }
 234 
 235     if (frame_type != TITLED) {
 236         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 237     }
 238 
 239     connect_signals(gtk_widget, this);
 240 }

 242 void WindowContext::paint(void *data, jint width, jint height) {
 243 #ifdef GLASS_GTK3
 244     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 245     gdk_window_begin_paint_region(gdk_window, region);
 246     cairo_t* context = gdk_cairo_create(gdk_window);
 247 #else
 248     cairo_t *context = gdk_cairo_create(gdk_window);
 249 #endif
 250 
 251     if (bg_color.is_set) {
 252         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 253                                 (frame_type == TRANSPARENT) ? 0 : 1);
 254         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 255         cairo_paint(context);
 256     }
 257 
 258     cairo_surface_t *cairo_surface;
 259     cairo_surface = cairo_image_surface_create_for_data(
 260             (unsigned char *) data,
 261             CAIRO_FORMAT_ARGB32,
 262             width, height, width * 4);
 263 
 264     cairo_set_source_surface(context, cairo_surface, 0, 0);
 265 
 266     applyShapeMask(data, width, height);
 267     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 268     cairo_paint(context);
 269 
 270 #ifdef GLASS_GTK3
 271     gdk_window_end_paint(gdk_window);
 272     cairo_region_destroy(region);
 273     cairo_destroy(context);
 274 #else
 275     cairo_destroy(context);
 276 #endif
 277 
 278     cairo_surface_destroy(cairo_surface);
 279 }

 321 void WindowContext::process_focus(GdkEventFocus *event) {
 322     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 323         ungrab_mouse_drag_focus();
 324     }
 325 
 326     if (!event->in && WindowContext::sm_grab_window == this) {
 327         ungrab_focus();
 328     }
 329 
 330     if (xim.enabled && xim.ic) {
 331         if (event->in) {
 332             XSetICFocus(xim.ic);
 333         } else {
 334             XUnsetICFocus(xim.ic);
 335         }
 336     }
 337 
 338     if (jwindow) {
 339         if (!event->in || isEnabled()) {
 340             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 341                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 342                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 343             CHECK_JNI_EXCEPTION(mainEnv)
 344         } else {
 345             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 346             CHECK_JNI_EXCEPTION(mainEnv)
 347         }
 348     }
 349 }

 351 void WindowContext::process_property_notify(GdkEventProperty *event) {
 352     if (event->window == gdk_window) {
 353         // This work-around is only necessary for Unity
 354         if (event->atom == atom_net_wm_state) {
 355             process_net_wm_property();
 356         } else if (event->atom == atom_net_wm_frame_extents) {
 357             if (frame_type != TITLED) {
 358                 return;
 359             }
 360 
 361             int top, left, bottom, right;
 362 
 363             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 364                 if (top + left + bottom + right > 0) {
 365                     geometry.frame_extents_received = true;
 366                     geometry.adjust_w = left + right;
 367                     geometry.adjust_h = top + bottom;
 368                     geometry.view_x = left;
 369                     geometry.view_y = top;
 370 
 371                     // set bounds again to set to correct window size that must
 372                     // be the total width and height accounting extents
 373                     // this is ignored if size is "content size" instead of "window size"
 374                     if (geometry.needs_ajustment) {
 375                         set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
 376                     }
 377 
 378                     // force position notify so java will know about view_y and view_x
 379                     size_position_notify(false, true);
 380                 }
 381             }
 382         }
 383     }
 384 }

 386 void WindowContext::process_configure() {
 387     gint x, y, w, h, gtk_w, gtk_h;
 388 
 389     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 390     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 391 
 392     w = gtk_w + geometry.adjust_w;
 393     h = gtk_h + geometry.adjust_h;
 394 
 395     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 396     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 397                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 398 
 399     geometry.current_x = x;
 400     geometry.current_y = y;
 401     geometry.current_w = w;
 402     geometry.current_h = h;
 403     geometry.current_cw = gtk_w;
 404     geometry.current_ch = gtk_h;
 405 
 406     if (!is_fullscreen && !is_maximized) {
 407         geometry.last_cw = gtk_w;
 408         geometry.last_ch = gtk_h;
 409     }
 410 
 411     size_position_notify(size_changed, pos_changed);
 412 }

 414 void WindowContext::process_destroy() {
 415     if (owner) {
 416         owner->remove_child(this);
 417     }
 418 
 419     if (WindowContext::sm_mouse_drag_window == this) {
 420         ungrab_mouse_drag_focus();
 421     }
 422 
 423     if (WindowContext::sm_grab_window == this) {
 424         ungrab_focus();
 425     }
 426 
 427     std::set<WindowContext *>::iterator it;
 428     for (it = children.begin(); it != children.end(); ++it) {
 429         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 430         // WindowContext::process_destroy() to call remove_child() (because children
 431         // is being iterated here) but also prevents gtk_window_set_transient_for from
 432         // being called - this causes the crash on gnome.
 433         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 434         (*it)->set_owner(NULL);
 435         destroy_and_delete_ctx(*it);
 436     }
 437     children.clear();
 438 
 439     if (jwindow) {
 440         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 441         EXCEPTION_OCCURED(mainEnv);
 442     }
 443 
 444     if (jview) {
 445         mainEnv->DeleteGlobalRef(jview);
 446         jview = NULL;
 447     }
 448 
 449     if (jwindow) {
 450         mainEnv->DeleteGlobalRef(jwindow);
 451         jwindow = NULL;
 452     }
 453 
 454     can_be_deleted = true;
 455 }

 457 void WindowContext::process_delete() {
 458     if (jwindow && isEnabled()) {
 459         gtk_widget_hide_on_delete(gtk_widget);
 460         mainEnv->CallVoidMethod(jwindow, jWindowNotifyClose);
 461         CHECK_JNI_EXCEPTION(mainEnv)
 462     }

 473 void WindowContext::process_mouse_button(GdkEventButton *event) {
 474     // there are other events like GDK_2BUTTON_PRESS
 475     if (event->type != GDK_BUTTON_PRESS && event->type != GDK_BUTTON_RELEASE) {
 476         return;
 477     }
 478 
 479     bool press = event->type == GDK_BUTTON_PRESS;
 480 
 481     guint state = event->state;
 482     guint mask = 0;
 483 
 484     // We need to add/remove current mouse button from the modifier flags
 485     // as X lib state represents the state just prior to the event and
 486     // glass needs the state just after the event
 487     switch (event->button) {
 488         case 1:
 489             mask = GDK_BUTTON1_MASK;
 490             break;
 491         case 2:
 492             mask = GDK_BUTTON2_MASK;
 493             break;
 494         case 3:
 495             mask = GDK_BUTTON3_MASK;
 496             break;
 497         case MOUSE_BACK_BTN:
 498             mask = GDK_BUTTON4_MASK;
 499             break;
 500         case MOUSE_FORWARD_BTN:
 501             mask = GDK_BUTTON5_MASK;
 502             break;
 503     }
 504 
 505     if (press) {
 506         state |= mask;
 507     } else {
 508         state &= ~mask;
 509     }
 510 
 511     if (press) {
 512         GdkDevice* device = event->device;
 513 
 514         if (glass_gdk_device_is_grabbed(device)
 515                 && (glass_gdk_device_get_window_at_position(device, NULL, NULL)
 516                 == NULL)) {
 517             ungrab_focus();
 518             return;
 519         }
 520 
 521         // Upper layers expects from us Windows behavior:
 522         // all mouse events should be delivered to window where drag begins
 523         // and no exit/enter event should be reported during this drag.
 524         // We can grab mouse pointer for these needs.
 525         grab_mouse_drag_focus(gdk_window, (GdkEvent *) event, NULL, true);
 526     } else {
 527         if ((event->state & MOUSE_BUTTONS_MASK)
 528             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 529             ungrab_mouse_drag_focus();
 530         } else if (event->button == 8 || event->button == 9) {
 531             // GDK X backend interprets button press events for buttons 4-7 as
 532             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 533             // be set on the event->state from GDK. Thus we cannot check if all
 534             // buttons have been released in the usual way (as above).
 535             ungrab_mouse_drag_focus();
 536         }
 537     }
 538 
 539     bool is_popup_trigger = (event->button == 3);
 540     jint button = gtk_button_number_to_mouse_button(event->button);
 541 
 542     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 543         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 544                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 545                                 button,
 546                                 (jint) event->x, (jint) event->y,
 547                                 (jint) event->x_root, (jint) event->y_root,
 548                                 gdk_modifier_mask_to_glass(state),
 549                                 (is_popup_trigger) ? JNI_TRUE : JNI_FALSE,
 550                                 JNI_FALSE);
 551         CHECK_JNI_EXCEPTION(mainEnv)
 552 
 553         if (jview && is_popup_trigger) {
 554             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 555                                     (jint) event->x, (jint) event->y,
 556                                     (jint) event->x_root, (jint) event->y_root,
 557                                     JNI_FALSE);
 558             CHECK_JNI_EXCEPTION(mainEnv)
 559         }
 560     }
 561 }

 563 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 564     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 565     jint isDrag = glass_modifier & (
 566             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 567             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 568             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 569             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 570             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 571     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 572 
 573     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 574         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 575     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 576         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 577     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 578         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 579     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 580         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 581     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 582         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 583     }
 584 
 585     if (jview) {
 586         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 587                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 588                                 button,
 589                                 (jint) event->x, (jint) event->y,
 590                                 (jint) event->x_root, (jint) event->y_root,
 591                                 glass_modifier,
 592                                 JNI_FALSE,
 593                                 JNI_FALSE);
 594         CHECK_JNI_EXCEPTION(mainEnv)
 595     }
 596 
 597     gdk_event_request_motions(event);

 600 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 601     jdouble dx = 0, dy = 0;
 602 
 603     // converting direction to change in pixels
 604     switch (event->direction) {
 605         case GDK_SCROLL_UP:
 606             dy = 1;
 607             break;
 608         case GDK_SCROLL_DOWN:
 609             dy = -1;
 610             break;
 611         case GDK_SCROLL_LEFT:
 612             dx = 1;
 613             break;
 614         case GDK_SCROLL_RIGHT:
 615             dx = -1;
 616             break;
 617         default:
 618             break;
 619     }
 620     if (event->state & GDK_SHIFT_MASK) {
 621         jdouble t = dy;
 622         dy = dx;
 623         dx = t;
 624     }
 625     if (jview) {
 626         mainEnv->CallVoidMethod(jview, jViewNotifyScroll,
 627                                 (jint) event->x, (jint) event->y,
 628                                 (jint) event->x_root, (jint) event->y_root,
 629                                 dx, dy,
 630                                 gdk_modifier_mask_to_glass(event->state),
 631                                 (jint) 0, (jint) 0,
 632                                 (jint) 0, (jint) 0,
 633                                 (jdouble) 40.0, (jdouble) 40.0);
 634         CHECK_JNI_EXCEPTION(mainEnv)
 635     }
 636 }

 638 void WindowContext::process_mouse_cross(GdkEventCrossing *event) {

 663 void WindowContext::process_key(GdkEventKey *event) {
 664     bool press = event->type == GDK_KEY_PRESS;
 665     jint glassKey = get_glass_key(event);
 666     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 667     if (press) {
 668         glassModifier |= glass_key_to_modifier(glassKey);
 669     } else {
 670         glassModifier &= ~glass_key_to_modifier(glassKey);
 671     }
 672     jcharArray jChars = NULL;
 673     jchar key = gdk_keyval_to_unicode(event->keyval);
 674     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 675         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 676     } else {
 677 #ifdef GLASS_GTK2
 678         if (key == 0) {
 679             // Work around "bug" fixed in gtk-3.0:
 680             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 681             switch (event->keyval) {
 682             case 0xFF08 /* Backspace */: key =  '\b';
 683             case 0xFF09 /* Tab       */: key =  '\t';
 684             case 0xFF0A /* Linefeed  */: key =  '\n';
 685             case 0xFF0B /* Vert. Tab */: key =  '\v';
 686             case 0xFF0D /* Return    */: key =  '\r';
 687             case 0xFF1B /* Escape    */: key =  '\033';
 688             case 0xFFFF /* Delete    */: key =  '\177';
 689             }
 690         }
 691 #endif
 692     }
 693 
 694     if (key > 0) {
 695         jChars = mainEnv->NewCharArray(1);
 696         if (jChars) {
 697             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 698             CHECK_JNI_EXCEPTION(mainEnv)
 699         }
 700     } else {
 701         jChars = mainEnv->NewCharArray(0);
 702     }
 703     if (jview) {
 704         if (press) {
 705             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 706                                     com_sun_glass_events_KeyEvent_PRESS,
 707                                     glassKey,
 708                                     jChars,
 709                                     glassModifier);
 710             CHECK_JNI_EXCEPTION(mainEnv)
 711 
 712             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 713                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 714                                         com_sun_glass_events_KeyEvent_TYPED,
 715                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 716                                         jChars,
 717                                         glassModifier);
 718                 CHECK_JNI_EXCEPTION(mainEnv)
 719             }
 720         } else {
 721             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 722                                     com_sun_glass_events_KeyEvent_RELEASE,
 723                                     glassKey,
 724                                     jChars,
 725                                     glassModifier);
 726             CHECK_JNI_EXCEPTION(mainEnv)
 727         }
 728     }
 729 }

 731 void WindowContext::process_state(GdkEventWindowState *event) {
 732     if (event->changed_mask &
 733         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 734 
 735         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 736             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 737         }
 738         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 739             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 740         }
 741 
 742         jint stateChangeEvent;
 743 
 744         if (is_iconified) {
 745             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 746         } else if (is_maximized) {
 747             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 748         } else {
 749             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 750             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 751                 // in this case - the window manager will not support the programatic
 752                 // request to iconify - so we need to restore it now.
 753                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 754             }
 755         }
 756 
 757         notify_state(stateChangeEvent);
 758     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 759         notify_on_top(event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 760     }
 761 }

 763 void WindowContext::process_net_wm_property() {
 764     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 765 
 766     // This is a Unity bug
 767     if (!g_strcmp0("Unity", gdk_x11_screen_get_window_manager_name(gdk_screen_get_default()))) {
 768         return;
 769     }
 770 
 771     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 772     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 773     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 774     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 775 
 776     gint length;
 777     glong *atoms = NULL;
 778 
 779     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 780                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 781 
 782         bool is_hidden = false;
 783         bool is_above = false;
 784         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 785             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 786                 is_hidden = true;
 787             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 788                 is_above = true;
 789             }
 790         }
 791 
 792         g_free(atoms);
 793 
 794         if (is_iconified != is_hidden) {
 795             is_iconified = is_hidden;
 796 
 797             notify_state((is_hidden)
 798                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 799                          : com_sun_glass_events_WindowEvent_RESTORE);
 800         }
 801 
 802         notify_on_top(is_above);
 803     }
 804 }

 989 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 990     // this will tell if adjustments are needed - that's because GTK does not have full window size
 991     // values, just content values. Frame extents (window decorations) are handled by the window manager.
 992     geometry.needs_ajustment = (w > 0 || h > 0) || geometry.needs_ajustment;
 993 
 994     // newW / newH always content sizes compatible with GTK+
 995     // if window has no decoration, adjustments will be ZERO
 996     int newW, newH;
 997     newW = (w > 0) ? w - geometry.adjust_w : cw;
 998     newH = (h > 0) ? h - geometry.adjust_h : ch;
 999 
1000     geometry.current_w = newW;
1001     geometry.current_h = newH;
1002 
1003     gboolean size_changed = FALSE;
1004     gboolean pos_changed = FALSE;
1005 
1006     if (newW > 0 && newH > 0) {
1007         size_changed = TRUE;
1008 
1009         // content size
1010         geometry.current_cw = newW;
1011         geometry.current_ch = newH;
1012         geometry.last_cw = newW;
1013         geometry.last_ch = newH;
1014 
1015         if (visible_received) {
1016             // call apply_geometry() to let gtk_window_resize succeed, because it's bound to
1017             // geometry constraints
1018             apply_geometry();
1019             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
1020         } else {
1021             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
1022         }
1023     }
1024 
1025     if (xSet || ySet) {
1026         int newX = (xSet) ? x : geometry.current_x;
1027         int newY = (ySet) ? y : geometry.current_y;
1028 
1029         if (newX != geometry.current_x || newY != geometry.current_y) {
1030             pos_changed = TRUE;
1031             geometry.current_x = newX;
1032             geometry.current_y = newY;
1033 
1034             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1035         }
1036     }
1037 
1038     size_position_notify(size_changed, pos_changed);
1039 }

1245 // This function calculate the deltas between window and window + decoration (titlebar, borders).
1246 // It's used when the window manager does not support the _NET_FRAME_EXTENTS extension or when
1247 // it's not received on time.
1248 void WindowContext::calculate_adjustments() {
1249     if (frame_type != TITLED || geometry.frame_extents_received) {
1250         return;
1251     }
1252 
1253     gint x, y, rx, ry;
1254     gdk_window_get_origin(gdk_window, &x, &y);
1255     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1256 
1257     if (rx != x || ry != y) {
1258         // the left extends are correct - the right one is guessed to be the same
1259         geometry.adjust_w = (x - rx) * 2;
1260         // guess that bottom size is the same as left and right
1261         geometry.adjust_h = (y - ry) + (x - rx);
1262 
1263         // those will be correct
1264         geometry.view_x = (x - rx);
1265         geometry.view_y = (y - ry);
1266 
1267         if (geometry.needs_ajustment) {
1268             set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
1269         }
1270 
1271         // force position notify so java will know about view_y and view_x
1272         size_position_notify(false, true);
1273     }
1274 }

1276 void WindowContext::apply_geometry() {
1277     if (!map_received) {
1278         return;
1279     }
1280 
1281     GdkGeometry gdk_geometry;
1282     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
1283 
1284     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1285         // not resizeable
1286         int w = geometry.current_cw > 0
1287                 ? geometry.current_cw
1288                 : geometry.current_w - geometry.adjust_w;
1289 
1290         int h = geometry.current_ch > 0
1291                 ? geometry.current_ch
1292                 : geometry.current_h - geometry.adjust_h;
1293 
1294         gdk_geometry.min_width = gdk_geometry.max_width = w;
1295         gdk_geometry.min_height = gdk_geometry.max_height = h;
1296     } else {
1297         //min/max width/height always whole window size (with decors)
1298         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1299                                  ? geometry.minw - geometry.adjust_w : 1;
1300         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1301                                   ? geometry.minh - geometry.adjust_h : 1;
1302 
1303         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
1304                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1305         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1306                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1307     }
1308 
1309     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1310                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
1311 }

1339 void WindowContext::activate_window() {
1340     Display *display = GDK_DISPLAY_XDISPLAY(gdk_window_get_display(gdk_window));
1341     Atom navAtom = XInternAtom(display, "_NET_ACTIVE_WINDOW", True);
1342     if (navAtom != None) {
1343         XClientMessageEvent clientMessage;
1344         memset(&clientMessage, 0, sizeof(clientMessage));
1345 
1346         clientMessage.type = ClientMessage;
1347         clientMessage.window = GDK_WINDOW_XID(gdk_window);
1348         clientMessage.message_type = navAtom;
1349         clientMessage.format = 32;
1350         clientMessage.data.l[0] = 1;
1351         clientMessage.data.l[1] = gdk_x11_get_server_time(gdk_window);
1352         clientMessage.data.l[2] = 0;
1353 
1354         XSendEvent(display, XDefaultRootWindow(display), False,
1355                    SubstructureRedirectMask | SubstructureNotifyMask,
1356                    (XEvent * ) & clientMessage);
1357         XFlush(display);
1358     }
1359 }

1361 void WindowContext::size_position_notify(bool size_changed, bool pos_changed) {
1362 
1363     if (jview) {
1364         if (size_changed) {
1365             mainEnv->CallVoidMethod(jview, jViewNotifyResize, geometry.current_cw, geometry.current_ch);
1366             CHECK_JNI_EXCEPTION(mainEnv);
1367         }
1368 
1369         if (pos_changed) {
1370             mainEnv->CallVoidMethod(jview, jViewNotifyView, com_sun_glass_events_ViewEvent_MOVE);
1371             CHECK_JNI_EXCEPTION(mainEnv)
1372         }
1373     }
1374 
1375     if (jwindow) {
1376         if (size_changed || is_maximized) {
1377             mainEnv->CallVoidMethod(jwindow, jWindowNotifyResize,
1378                                     (is_maximized)
1379                                     ? com_sun_glass_events_WindowEvent_MAXIMIZE
1380                                     : com_sun_glass_events_WindowEvent_RESIZE,
1381                                     geometry.current_w, geometry.current_h);
1382             CHECK_JNI_EXCEPTION(mainEnv)
1383         }
1384 
1385         if (pos_changed) {
1386             mainEnv->CallVoidMethod(jwindow, jWindowNotifyMove, geometry.current_x, geometry.current_y);
1387             CHECK_JNI_EXCEPTION(mainEnv)
1388         }
1389     }
1390 }

1400 bool WindowContext::on_top_inherited() {
1401     WindowContext *o = owner;
1402     while (o) {
1403         WindowContext *topO = dynamic_cast<WindowContext *>(o);
1404         if (!topO) break;
1405         if (topO->on_top) {
1406             return true;
1407         }
1408         o = topO->owner;
1409     }
1410     return false;
1411 }

1418     return on_top;

1422     if (is_grab_disabled()) {

1426     ungrab_mouse_drag_focus();

1434

1443                                                           | GDK_BUTTON_PRESS_MASK
1444                                                           | GDK_BUTTON_RELEASE_MASK), cursor, GDK_CURRENT_TIME);
1445 #else
1446     GdkGrabStatus status = gdk_pointer_grab(gdk_w, owner_events,
1447                                                 (GdkEventMask)
1448                                                       (GDK_POINTER_MOTION_MASK
1449                                                           | GDK_POINTER_MOTION_HINT_MASK

1458
