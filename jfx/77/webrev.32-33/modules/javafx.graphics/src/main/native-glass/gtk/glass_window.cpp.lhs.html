 136 void destroy_and_delete_ctx(WindowContext *ctx) {
 137     if (ctx) {
 138         ctx->process_destroy();
 139 
 140         if (!ctx->get_events_count()) {
 141             delete ctx;
 142         }
 143         // else: ctx will be deleted in EventsCounterHelper after completing
 144         // an event processing
 145     }
 146 }

 148 static inline jint gtk_button_number_to_mouse_button(guint button) {
 149     switch (button) {
 150         case 1:
 151             return com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 152         case 2:
 153             return com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 154         case 3:
 155             return com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 156         case MOUSE_BACK_BTN:
 157             return com_sun_glass_events_MouseEvent_BUTTON_BACK;
 158         case MOUSE_FORWARD_BTN:
 159             return com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 160         default:
 161             // Other buttons are not supported by quantum and are not reported by other platforms
 162             return com_sun_glass_events_MouseEvent_BUTTON_NONE;
 163     }
 164 }

 174 WindowContext::WindowContext(jobject _jwindow, WindowContext *_owner, long _screen,
 175                                    WindowFrameType _frame_type, WindowType type, GdkWMFunction wmf) :
 176         screen(_screen),
 177         frame_type(_frame_type),
 178         window_type(type),
 179         owner(_owner),
 180         jview(NULL),
 181         map_received(false),
 182         visible_received(false),
 183         on_top(false),
 184         is_fullscreen(false),
 185         is_iconified(false),
 186         is_maximized(false),
 187         is_mouse_entered(false),
 188         can_be_deleted(false),
 189         events_processing_cnt(0),
 190         pointer_device(NULL) {
 191 
 192     jwindow = mainEnv->NewGlobalRef(_jwindow);
 193 
 194     gtk_widget = gtk_window_new(type == POPUP ? GTK_WINDOW_POPUP : GTK_WINDOW_TOPLEVEL);
 195 
 196     if (gchar * app_name = get_application_name()) {
 197         gtk_window_set_wmclass(GTK_WINDOW(gtk_widget), app_name, app_name);
 198         g_free(app_name);
 199     }
 200 
 201     if (owner) {
 202         owner->add_child(this);
 203         if (on_top_inherited()) {
 204             gtk_window_set_keep_above(GTK_WINDOW(gtk_widget), TRUE);
 205         }
 206     }
 207 
 208     if (type == UTILITY) {
 209         gtk_window_set_type_hint(GTK_WINDOW(gtk_widget), GDK_WINDOW_TYPE_HINT_UTILITY);
 210     }
 211 
 212     glong xvisualID = (glong) mainEnv->GetStaticLongField(jApplicationCls, jApplicationVisualID);
 213 
 214     if (xvisualID != 0) {
 215         GdkVisual *visual = gdk_x11_screen_lookup_visual(gdk_screen_get_default(), xvisualID);
 216         glass_gtk_window_configure_from_visual(gtk_widget, visual);
 217     }
 218 
 219     gtk_widget_set_events(gtk_widget, GDK_ALL_EVENTS_MASK);
 220     gtk_widget_set_app_paintable(gtk_widget, TRUE);
 221 
 222     glass_gtk_configure_transparency_and_realize(gtk_widget, frame_type == TRANSPARENT);
 223     gtk_window_set_title(GTK_WINDOW(gtk_widget), "");
 224 
 225     gdk_window = gtk_widget_get_window(gtk_widget);
 226     g_object_set_data_full(G_OBJECT(gdk_window), GDK_WINDOW_DATA_CONTEXT, this, NULL);
 227 
 228     glass_dnd_attach_context(this);
 229 
 230     gdk_windowManagerFunctions = wmf;
 231     if (wmf) {
 232         gdk_window_set_functions(gdk_window, wmf);
 233     }
 234 
 235     if (frame_type != TITLED) {
 236         gtk_window_set_decorated(GTK_WINDOW(gtk_widget), FALSE);
 237     }
 238 
 239     geometry.frame_extents_supported = gdk_x11_screen_supports_net_wm_hint(gdk_screen_get_default(),
 240                                                                            atom_net_wm_frame_extents);
 241 
 242     connect_signals(gtk_widget, this);
 243 }

 245 void WindowContext::paint(void *data, jint width, jint height) {
 246 #if GTK_CHECK_VERSION(3, 0, 0)
 247     cairo_region_t *region = gdk_window_get_clip_region(gdk_window);
 248 #if GTK_CHECK_VERSION(3, 22, 0)
 249     GdkDrawingContext *dcontext = gdk_window_begin_draw_frame(gdk_window, region);
 250     cairo_t *context = gdk_drawing_context_get_cairo_context(dcontext);
 251 #else
 252     gdk_window_begin_paint_region(gdk_window, region);
 253     cairo_t* context = gdk_cairo_create(gdk_window);
 254 #endif
 255 #else
 256     cairo_t *context = gdk_cairo_create(gdk_window);
 257 #endif
 258 
 259     if (bg_color.is_set) {
 260         cairo_set_source_rgba(context, bg_color.red, bg_color.green, bg_color.blue,
 261                                 (frame_type == TRANSPARENT) ? 0 : 1);
 262         cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 263         cairo_paint(context);
 264     }
 265 
 266     cairo_surface_t *cairo_surface;
 267     cairo_surface = cairo_image_surface_create_for_data(
 268             (unsigned char *) data,
 269             CAIRO_FORMAT_ARGB32,
 270             width, height, width * 4);
 271 
 272     cairo_set_source_surface(context, cairo_surface, 0, 0);
 273 
 274     applyShapeMask(data, width, height);
 275     cairo_set_operator(context, CAIRO_OPERATOR_SOURCE);
 276     cairo_paint(context);
 277 
 278 #if GTK_CHECK_VERSION(3, 0, 0)
 279 #if GTK_CHECK_VERSION(3, 22, 0)
 280     gdk_window_end_draw_frame(gdk_window, dcontext);
 281     cairo_region_destroy(region);
 282 #else
 283     gdk_window_end_paint(gdk_window);
 284     cairo_region_destroy(region);
 285     cairo_destroy(context);
 286 #endif
 287 #else
 288     cairo_destroy(context);
 289 #endif
 290 
 291     cairo_surface_destroy(cairo_surface);
 292 }

 334 void WindowContext::process_focus(GdkEventFocus *event) {
 335     if (!event->in && WindowContext::sm_mouse_drag_window == this) {
 336         ungrab_mouse_drag_focus();
 337     }
 338 
 339     if (!event->in && WindowContext::sm_grab_window == this) {
 340         ungrab_focus();
 341     }
 342 
 343     if (xim.enabled && xim.ic) {
 344         if (event->in) {
 345             XSetICFocus(xim.ic);
 346         } else {
 347             XUnsetICFocus(xim.ic);
 348         }
 349     }
 350 
 351     if (jwindow) {
 352         if (!event->in || isEnabled()) {
 353             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocus,
 354                                     event->in ? com_sun_glass_events_WindowEvent_FOCUS_GAINED
 355                                               : com_sun_glass_events_WindowEvent_FOCUS_LOST);
 356             CHECK_JNI_EXCEPTION(mainEnv)
 357         } else {
 358             mainEnv->CallVoidMethod(jwindow, jWindowNotifyFocusDisabled);
 359             CHECK_JNI_EXCEPTION(mainEnv)
 360         }
 361     }
 362 }

 364 void WindowContext::process_property_notify(GdkEventProperty *event) {
 365     if (event->window == gdk_window) {
 366         if (event->atom == atom_net_wm_state) {
 367             process_net_wm_property();
 368         } else if (event->atom == atom_net_wm_frame_extents) {
 369             if (frame_type != TITLED) {
 370                 return;
 371             }
 372 
 373             int top, left, bottom, right;
 374 
 375             if (get_frame_extents_property(&top, &left, &bottom, &right)) {
 376                 if (top + left + bottom + right > 0) {
 377                     geometry.frame_extents_received = true;
 378                     geometry.adjust_w = left + right;
 379                     geometry.adjust_h = top + bottom;
 380                     geometry.view_x = left;
 381                     geometry.view_y = top;
 382 
 383                     // set bounds again to set to correct window size that must
 384                     // be the total width and height accounting extents
 385                     // this is ignored if size is "content size" instead of "window size"
 386                     if (geometry.window_size_set) {
 387                         set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
 388                     }
 389 
 390                     // force position notify so java will know about view_y and view_x
 391                     size_position_notify(false, true);
 392                 }
 393             }
 394         }
 395     }
 396 }

 398 void WindowContext::process_configure() {
 399     calculate_adjustments();
 400 
 401     gint x, y, w, h, gtk_w, gtk_h;
 402 
 403     gtk_window_get_position(GTK_WINDOW(gtk_widget), &x, &y);
 404     gtk_window_get_size(GTK_WINDOW(gtk_widget), &gtk_w, &gtk_h);
 405 
 406     w = gtk_w + geometry.adjust_w;
 407     h = gtk_h + geometry.adjust_h;
 408 
 409     gboolean pos_changed = geometry.current_x != x || geometry.current_y != y;
 410     gboolean size_changed = geometry.current_w != w || geometry.current_h != h
 411                             || geometry.current_cw != gtk_w || geometry.current_ch != gtk_h;
 412 
 413     geometry.current_x = x;
 414     geometry.current_y = y;
 415     geometry.current_w = w;
 416     geometry.current_h = h;
 417     geometry.current_cw = gtk_w;
 418     geometry.current_ch = gtk_h;
 419 
 420     size_position_notify(size_changed, pos_changed);
 421 }

 423 void WindowContext::process_destroy() {
 424     if (owner) {
 425         owner->remove_child(this);
 426     }
 427 
 428     if (WindowContext::sm_mouse_drag_window == this) {
 429         ungrab_mouse_drag_focus();
 430     }
 431 
 432     if (WindowContext::sm_grab_window == this) {
 433         ungrab_focus();
 434     }
 435 
 436     std::set<WindowContext *>::iterator it;
 437     for (it = children.begin(); it != children.end(); ++it) {
 438         // FIX JDK-8226537: this method calls set_owner(NULL) which prevents
 439         // WindowContext::process_destroy() to call remove_child() (because children
 440         // is being iterated here) but also prevents gtk_window_set_transient_for from
 441         // being called - this causes the crash on gnome.
 442         gtk_window_set_transient_for((*it)->get_gtk_window(), NULL);
 443         (*it)->set_owner(NULL);
 444         destroy_and_delete_ctx(*it);
 445     }
 446     children.clear();
 447 
 448     if (jwindow) {
 449         mainEnv->CallVoidMethod(jwindow, jWindowNotifyDestroy);
 450         EXCEPTION_OCCURED(mainEnv);
 451     }
 452 
 453     if (jview) {
 454         mainEnv->DeleteGlobalRef(jview);
 455         jview = NULL;
 456     }
 457 
 458     if (jwindow) {
 459         mainEnv->DeleteGlobalRef(jwindow);
 460         jwindow = NULL;
 461     }
 462 
 463     can_be_deleted = true;
 464 }

 466 void WindowContext::process_delete() {

 468         gtk_widget_hide_on_delete(gtk_widget);
 469         mainEnv->CallVoidMethod(jwindow, jWindowNotifyClose);
 470         CHECK_JNI_EXCEPTION(mainEnv)
 471     }

 482 void WindowContext::process_mouse_button(GdkEventButton *event) {
 483     bool press = event->type == GDK_BUTTON_PRESS;
 484     bool release = event->type == GDK_BUTTON_RELEASE;
 485 
 486     guint state = event->state;
 487     guint mask = 0;
 488 
 489     // We need to add/remove current mouse button from the modifier flags
 490     // as X lib state represents the state just prior to the event and
 491     // glass needs the state just after the event
 492     switch (event->button) {
 493         case 1:
 494             mask = GDK_BUTTON1_MASK;
 495             break;
 496         case 2:
 497             mask = GDK_BUTTON2_MASK;
 498             break;
 499         case 3:
 500             mask = GDK_BUTTON3_MASK;
 501             break;
 502         case MOUSE_BACK_BTN:
 503             mask = GDK_BUTTON4_MASK;
 504             break;
 505         case MOUSE_FORWARD_BTN:
 506             mask = GDK_BUTTON5_MASK;
 507             break;
 508     }
 509 
 510     if (press) {
 511         state |= mask;
 512     } else {
 513         state &= ~mask;
 514     }
 515 
 516     // Upper layers expects from us Windows behavior:
 517     // all mouse events should be delivered to window where drag begins
 518     // and no exit/enter event should be reported during this drag.
 519     // We can grab mouse pointer for these needs.
 520     if (press) {
 521         grab_mouse_drag_focus(gdk_window, NULL, true);
 522     } else {
 523         if ((event->state & MOUSE_BUTTONS_MASK)
 524             && !(state & MOUSE_BUTTONS_MASK)) { // all buttons released
 525             ungrab_mouse_drag_focus();
 526         } else if (event->button == 8 || event->button == 9) {
 527             // GDK X backend interprets button press events for buttons 4-7 as
 528             // scroll events so GDK_BUTTON4_MASK and GDK_BUTTON5_MASK will never
 529             // be set on the event->state from GDK. Thus we cannot check if all
 530             // buttons have been released in the usual way (as above).
 531             ungrab_mouse_drag_focus();
 532         }
 533     }
 534 
 535     jint button = gtk_button_number_to_mouse_button(event->button);
 536 
 537     if (jview && button != com_sun_glass_events_MouseEvent_BUTTON_NONE) {
 538         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 539                                 press ? com_sun_glass_events_MouseEvent_DOWN : com_sun_glass_events_MouseEvent_UP,
 540                                 button,
 541                                 (jint) event->x, (jint) event->y,
 542                                 (jint) event->x_root, (jint) event->y_root,
 543                                 gdk_modifier_mask_to_glass(state),
 544                                 (event->button == 3 && press) ? JNI_TRUE : JNI_FALSE,
 545                                 JNI_FALSE);
 546         CHECK_JNI_EXCEPTION(mainEnv)
 547 
 548         if (jview && event->button == 3 && press) {
 549             mainEnv->CallVoidMethod(jview, jViewNotifyMenu,
 550                                     (jint) event->x, (jint) event->y,
 551                                     (jint) event->x_root, (jint) event->y_root,
 552                                     JNI_FALSE);
 553             CHECK_JNI_EXCEPTION(mainEnv)
 554         }
 555     }
 556 }

 558 void WindowContext::process_mouse_motion(GdkEventMotion *event) {
 559     jint glass_modifier = gdk_modifier_mask_to_glass(event->state);
 560     jint isDrag = glass_modifier & (
 561             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY |
 562             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE |
 563             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY |
 564             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK |
 565             com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD);
 566     jint button = com_sun_glass_events_MouseEvent_BUTTON_NONE;
 567 
 568     if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_PRIMARY) {
 569         button = com_sun_glass_events_MouseEvent_BUTTON_LEFT;
 570     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_MIDDLE) {
 571         button = com_sun_glass_events_MouseEvent_BUTTON_OTHER;
 572     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_SECONDARY) {
 573         button = com_sun_glass_events_MouseEvent_BUTTON_RIGHT;
 574     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_BACK) {
 575         button = com_sun_glass_events_MouseEvent_BUTTON_BACK;
 576     } else if (glass_modifier & com_sun_glass_events_KeyEvent_MODIFIER_BUTTON_FORWARD) {
 577         button = com_sun_glass_events_MouseEvent_BUTTON_FORWARD;
 578     }
 579 
 580     if (jview) {
 581         mainEnv->CallVoidMethod(jview, jViewNotifyMouse,
 582                                 isDrag ? com_sun_glass_events_MouseEvent_DRAG : com_sun_glass_events_MouseEvent_MOVE,
 583                                 button,
 584                                 (jint) event->x, (jint) event->y,
 585                                 (jint) event->x_root, (jint) event->y_root,
 586                                 glass_modifier,
 587                                 JNI_FALSE,
 588                                 JNI_FALSE);
 589         CHECK_JNI_EXCEPTION(mainEnv)
 590     }
 591 }

 593 void WindowContext::process_mouse_scroll(GdkEventScroll *event) {
 594     jdouble dx = 0;
 595     jdouble dy = 0;
 596 
 597     // converting direction to change in pixels
 598     switch (event->direction) {
 599 #if GTK_CHECK_VERSION(3, 4, 0)
 600         case GDK_SCROLL_SMOOTH:
 601             //FIXME 3.4 ???
 602             break;
 603 #endif
 604         case GDK_SCROLL_UP:
 605             dy = 1;
 606             break;
 607         case GDK_SCROLL_DOWN:
 608             dy = -1;
 609             break;
 610         case GDK_SCROLL_LEFT:
 611             dx = 1;
 612             break;
 613         case GDK_SCROLL_RIGHT:
 614             dx = -1;
 615             break;
 616     }
 617     if (event->state & GDK_SHIFT_MASK) {
 618         jdouble t = dy;
 619         dy = dx;
 620         dx = t;
 621     }
 622     if (jview) {
 623         mainEnv->CallVoidMethod(jview, jViewNotifyScroll,
 624                                 (jint) event->x, (jint) event->y,
 625                                 (jint) event->x_root, (jint) event->y_root,
 626                                 dx, dy,
 627                                 gdk_modifier_mask_to_glass(event->state),
 628                                 (jint) 0, (jint) 0,
 629                                 (jint) 0, (jint) 0,
 630                                 (jdouble) 40.0, (jdouble) 40.0);
 631         CHECK_JNI_EXCEPTION(mainEnv)
 632     }
 633 }

 660 void WindowContext::process_key(GdkEventKey *event) {
 661     bool press = event->type == GDK_KEY_PRESS;
 662     jint glassKey = get_glass_key(event);
 663     jint glassModifier = gdk_modifier_mask_to_glass(event->state);
 664     if (press) {
 665         glassModifier |= glass_key_to_modifier(glassKey);
 666     } else {
 667         glassModifier &= ~glass_key_to_modifier(glassKey);
 668     }
 669     jcharArray jChars = NULL;
 670     jchar key = gdk_keyval_to_unicode(event->keyval);
 671     if (key >= 'a' && key <= 'z' && (event->state & GDK_CONTROL_MASK)) {
 672         key = key - 'a' + 1; // map 'a' to ctrl-a, and so on.
 673     } else {
 674 #ifdef GLASS_GTK2
 675         if (key == 0) {
 676             // Work around "bug" fixed in gtk-3.0:
 677             // http://mail.gnome.org/archives/commits-list/2011-March/msg06832.html
 678             switch (event->keyval) {
 679             case 0xFF08 /* Backspace */: key =  '\b';
 680             case 0xFF09 /* Tab       */: key =  '\t';
 681             case 0xFF0A /* Linefeed  */: key =  '\n';
 682             case 0xFF0B /* Vert. Tab */: key =  '\v';
 683             case 0xFF0D /* Return    */: key =  '\r';
 684             case 0xFF1B /* Escape    */: key =  '\033';
 685             case 0xFFFF /* Delete    */: key =  '\177';
 686             }
 687         }
 688 #endif
 689     }
 690 
 691     if (key > 0) {
 692         jChars = mainEnv->NewCharArray(1);
 693         if (jChars) {
 694             mainEnv->SetCharArrayRegion(jChars, 0, 1, &key);
 695             CHECK_JNI_EXCEPTION(mainEnv)
 696         }
 697     } else {
 698         jChars = mainEnv->NewCharArray(0);
 699     }
 700     if (jview) {
 701         if (press) {
 702             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 703                                     com_sun_glass_events_KeyEvent_PRESS,
 704                                     glassKey,
 705                                     jChars,
 706                                     glassModifier);
 707             CHECK_JNI_EXCEPTION(mainEnv)
 708 
 709             if (jview && key > 0) { // TYPED events should only be sent for printable characters.
 710                 mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 711                                         com_sun_glass_events_KeyEvent_TYPED,
 712                                         com_sun_glass_events_KeyEvent_VK_UNDEFINED,
 713                                         jChars,
 714                                         glassModifier);
 715                 CHECK_JNI_EXCEPTION(mainEnv)
 716             }
 717         } else {
 718             mainEnv->CallVoidMethod(jview, jViewNotifyKey,
 719                                     com_sun_glass_events_KeyEvent_RELEASE,
 720                                     glassKey,
 721                                     jChars,
 722                                     glassModifier);
 723             CHECK_JNI_EXCEPTION(mainEnv)
 724         }
 725     }
 726 }

 728 void WindowContext::process_state(GdkEventWindowState *event) {
 729     if (event->changed_mask &
 730         (GDK_WINDOW_STATE_ICONIFIED | GDK_WINDOW_STATE_MAXIMIZED)) {
 731 
 732         if (event->changed_mask & GDK_WINDOW_STATE_ICONIFIED) {
 733             is_iconified = event->new_window_state & GDK_WINDOW_STATE_ICONIFIED;
 734         }
 735         if (event->changed_mask & GDK_WINDOW_STATE_MAXIMIZED) {
 736             is_maximized = event->new_window_state & GDK_WINDOW_STATE_MAXIMIZED;
 737         }
 738 
 739         jint stateChangeEvent;
 740 
 741         if (is_iconified) {
 742             stateChangeEvent = com_sun_glass_events_WindowEvent_MINIMIZE;
 743         } else if (is_maximized) {
 744             stateChangeEvent = com_sun_glass_events_WindowEvent_MAXIMIZE;
 745         } else {
 746             stateChangeEvent = com_sun_glass_events_WindowEvent_RESTORE;
 747             if ((gdk_windowManagerFunctions & GDK_FUNC_MINIMIZE) == 0) {
 748                 // in this case - the window manager will not support the programatic
 749                 // request to iconify - so we need to restore it now.
 750                 gdk_window_set_functions(gdk_window, gdk_windowManagerFunctions);
 751             }
 752         }
 753 
 754         notify_state(stateChangeEvent);
 755     } else if (event->changed_mask & GDK_WINDOW_STATE_ABOVE) {
 756         notify_on_top(event->new_window_state & GDK_WINDOW_STATE_ABOVE);
 757     }
 758 }

 760 void WindowContext::process_net_wm_property() {
 761     // Workaround for https://bugs.launchpad.net/unity/+bug/998073
 762 
 763     static GdkAtom atom_atom = gdk_atom_intern_static_string("ATOM");
 764     static GdkAtom atom_net_wm_state = gdk_atom_intern_static_string("_NET_WM_STATE");
 765     static GdkAtom atom_net_wm_state_hidden = gdk_atom_intern_static_string("_NET_WM_STATE_HIDDEN");
 766     static GdkAtom atom_net_wm_state_above = gdk_atom_intern_static_string("_NET_WM_STATE_ABOVE");
 767 
 768     gint length;
 769     glong *atoms = NULL;
 770 
 771     if (gdk_property_get(gdk_window, atom_net_wm_state, atom_atom,
 772                          0, G_MAXLONG, FALSE, NULL, NULL, &length, (guchar * *) & atoms)) {
 773 
 774         bool is_hidden = false;
 775         bool is_above = false;
 776         for (gint i = 0; i < (gint)(length / sizeof(glong)); i++) {
 777             if (atom_net_wm_state_hidden == (GdkAtom) atoms[i]) {
 778                 is_hidden = true;
 779             } else if (atom_net_wm_state_above == (GdkAtom) atoms[i]) {
 780                 is_above = true;
 781             }
 782         }
 783 
 784         g_free(atoms);
 785 
 786         if (is_iconified != is_hidden) {
 787             is_iconified = is_hidden;
 788 
 789             notify_state((is_hidden)
 790                          ? com_sun_glass_events_WindowEvent_MINIMIZE
 791                          : com_sun_glass_events_WindowEvent_RESTORE);
 792         }
 793 
 794         notify_on_top(is_above);
 795     }
 796 }

 979 void WindowContext::set_bounds(int x, int y, bool xSet, bool ySet, int w, int h, int cw, int ch) {
 980     calculate_adjustments();
 981 
 982     // newW / newH always content sizes compatible with GTK+
 983     // if window has no decoration, adjustments will be ZERO
 984     int newW = w > 0 ? w - geometry.adjust_w : cw;
 985     int newH = h > 0 ? h - geometry.adjust_h : ch;
 986 
 987     gboolean size_changed = FALSE;
 988     gboolean pos_changed = FALSE;
 989 
 990     geometry.window_size_set = (w > 0 || h > 0);


 991 
 992     if (newW > 0 && newH > 0) {
 993         size_changed = TRUE;
 994 
 995         geometry.current_cw = newW;
 996         geometry.current_ch = newH;
 997         geometry.current_w = newW + geometry.adjust_w;
 998         geometry.current_h = newH + geometry.adjust_h;
 999 
1000         if (visible_received) {
1001             // update constraints if not resized by the user interface so it will
1002             // let gtk_window_resize succeed, because it's bound to geometry constraints
1003             apply_geometry();
1004             gtk_window_resize(GTK_WINDOW(gtk_widget), newW, newH);
1005         } else {
1006             gtk_window_set_default_size(GTK_WINDOW(gtk_widget), newW, newH);
1007         }
1008     }
1009 
1010     if (xSet || ySet) {
1011         int newX = (xSet) ? x : geometry.current_x;
1012         int newY = (ySet) ? y : geometry.current_y;
1013 
1014         if (newX != geometry.current_x || newY != geometry.current_y) {
1015             pos_changed = TRUE;
1016             geometry.current_x = newX;
1017             geometry.current_y = newY;
1018             gtk_window_move(GTK_WINDOW(gtk_widget), newX, newY);
1019         }
1020     }
1021 
1022     size_position_notify(size_changed, pos_changed);
1023 }

1209 // This function calculate the deltas between window and window + decoration (titleblar, borders).
1210 // It's used when the window manager does not support the _NET_FRAME_EXTENTS extension.

1211 void WindowContext::calculate_adjustments() {
1212     // if WM supports _NET_FRAME_EXTENTS, wait for it
1213     if (frame_type != TITLED || geometry.frame_extents_supported) {
1214         return;
1215     }
1216 
1217     gint x, y, rx, ry;
1218     gdk_window_get_origin(gdk_window, &x, &y);
1219     gdk_window_get_root_origin(gdk_window, &rx, &ry);
1220 
1221     if (rx != x || ry != y) {
1222         // the left extends are correct - the right one is guessed to be the same
1223         geometry.adjust_w = (x - rx) * 2;
1224         // guess that bottom size is the same as left and right
1225         geometry.adjust_h = (y - ry) + (x - rx);
1226 
1227         // those will be correct
1228         geometry.view_x = (x - rx);
1229         geometry.view_y = (y - ry);
1230 
1231         if (geometry.window_size_set) {
1232             set_bounds(0, 0, false, false, geometry.current_w, geometry.current_h, -1, -1);
1233         }
1234 
1235         // force position notify so java will know about view_y and view_x
1236         size_position_notify(false, true);
1237     }
1238 }

1240 void WindowContext::apply_geometry() {
1241     if (!map_received) {
1242         return;
1243     }
1244 
1245     GdkGeometry gdk_geometry;
1246     gdk_geometry.win_gravity = GDK_GRAVITY_NORTH_WEST;
1247 




1248     if ((!geometry.resizable || !geometry.enabled) && !(is_maximized || is_fullscreen)) {
1249         // not resizeable
1250         int w = (geometry.current_w - geometry.adjust_w) > 0
1251                 ? geometry.current_w - geometry.adjust_w
1252                 : geometry.current_cw;
1253 
1254         int h = (geometry.current_h - geometry.adjust_h) > 0
1255                 ? geometry.current_h - geometry.adjust_h
1256                 : geometry.current_ch;
1257 
1258         gdk_geometry.min_width = gdk_geometry.max_width = w;
1259         gdk_geometry.min_height = gdk_geometry.max_height = h;
1260     } else {
1261         gdk_geometry.min_width = (geometry.minw - geometry.adjust_w) > 0
1262                                  ? geometry.minw - geometry.adjust_w : 1;
1263         gdk_geometry.min_height = (geometry.minh - geometry.adjust_h) > 0
1264                                   ? geometry.minh - geometry.adjust_h : 1;
1265 
1266         gdk_geometry.max_width = (geometry.maxw - geometry.adjust_w > 0)
1267                                  ? geometry.maxw - geometry.adjust_w : G_MAXINT;
1268         gdk_geometry.max_height = (geometry.maxh - geometry.adjust_h > 0)
1269                                   ? geometry.maxh - geometry.adjust_h : G_MAXINT;
1270     }
1271 
1272     gtk_window_set_geometry_hints(GTK_WINDOW(gtk_widget), NULL, &gdk_geometry,
1273                                   (GdkWindowHints)(GDK_HINT_MIN_SIZE | GDK_HINT_MAX_SIZE | GDK_HINT_WIN_GRAVITY));
1274 }

1302 void WindowContext::activate_window() {
1303     Display *display = GDK_DISPLAY_XDISPLAY(gdk_window_get_display(gdk_window));
1304     Atom navAtom = XInternAtom(display, "_NET_ACTIVE_WINDOW", True);
1305     if (navAtom != None) {
1306         XClientMessageEvent clientMessage;

1308

1315         clientMessage.data.l[2] = 0;
1316

1318                    SubstructureRedirectMask | SubstructureNotifyMask,
1319                    (XEvent * ) & clientMessage);

1324 void WindowContext::size_position_notify(bool size_changed, bool pos_changed) {
1325 
1326     if (jview) {

1329             CHECK_JNI_EXCEPTION(mainEnv);

1331

1333             mainEnv->CallVoidMethod(jview, jViewNotifyView, com_sun_glass_events_ViewEvent_MOVE);
1334             CHECK_JNI_EXCEPTION(mainEnv)

1344                                     geometry.current_w, geometry.current_h);
1345             CHECK_JNI_EXCEPTION(mainEnv)
1346         }
1347 
1348         if (pos_changed) {
1349             mainEnv->CallVoidMethod(jwindow, jWindowNotifyMove, geometry.current_x, geometry.current_y);

1363 bool WindowContext::on_top_inherited() {
1364     WindowContext *o = owner;
1365     while (o) {
1366         WindowContext *topO = dynamic_cast<WindowContext *>(o);
1367         if (!topO) break;
1368         if (topO->on_top) {
1369             return true;
1370         }
1371         o = topO->owner;
1372     }
1373     return false;
1374 }
