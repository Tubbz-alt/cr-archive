25 package javafx.scene.control;
26 
27 import javafx.beans.property.StringProperty;
28 
29 public class ControlShim extends Control {
30 
31     /**
32      * Installs the default skin for the given control.
33      *
34      * Note that this has no noticeable effect if the control's
35      * skin already is set to the default skin (see skinProperty for
36      * implementations details).
37      *
38      * @param control the control to set the default skin on
39      */
40     public static void installDefaultSkin(Control control) {
41         control.setSkin(control.createDefaultSkin());
42     }
43 
44     public static StringProperty skinClassNameProperty(Control c) {
45         return c.skinClassNameProperty();
46     }
47 
48     public static void layoutChildren(Control c) {
49         c.layoutChildren();
50     }
51 
52     public static  double computePrefWidth(Control c, double height) {
53         return c.computePrefWidth(height);
54     }
55 
56     public static  double computePrefHeight(Control c, double width) {
57         return c.computePrefHeight(width);
58     }
59 
60 }
