25 package javafx.scene.control;
26 
27 import javafx.beans.property.StringProperty;
28 
29 public class ControlShim extends Control {
30 
31     public static StringProperty skinClassNameProperty(Control c) {
32         return c.skinClassNameProperty();
33     }
34 
35     public static void layoutChildren(Control c) {
36         c.layoutChildren();
37     }
38 
39     public static  double computePrefWidth(Control c, double height) {
40         return c.computePrefWidth(height);
41     }
42 
43     public static  double computePrefHeight(Control c, double width) {
44         return c.computePrefHeight(width);
45     }
46 
47 }
