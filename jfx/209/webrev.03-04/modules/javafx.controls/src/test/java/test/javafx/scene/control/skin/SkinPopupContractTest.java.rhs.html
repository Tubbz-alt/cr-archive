26 package test.javafx.scene.control.skin;
27 
28 import org.junit.Test;
29 
30 import javafx.scene.control.ContextMenu;
31 import javafx.scene.control.Tooltip;
32 import javafx.scene.control.skin.ContextMenuSkin;
33 import javafx.scene.control.skin.TooltipSkin;
34 
35 /**
36  * Tests for Skinnables that are not Controls.
37  */
38 public class SkinPopupContractTest {
39 
40     @Test
41     public void testTooltipSkinDispose() {
42         Tooltip tooltip = new Tooltip();
43         tooltip.setSkin(new TooltipSkin(tooltip));
44         tooltip.getSkin().dispose();
45         tooltip.getSkin().dispose();
46     }
47 
48     @Test
49     public void testContextMenuSkinDispose() {
50         ContextMenu contextMenu = new ContextMenu();
51         contextMenu.setSkin(new ContextMenuSkin(contextMenu));
52         contextMenu.getSkin().dispose();
53         contextMenu.getSkin().dispose();
54     }
55 
56 }
