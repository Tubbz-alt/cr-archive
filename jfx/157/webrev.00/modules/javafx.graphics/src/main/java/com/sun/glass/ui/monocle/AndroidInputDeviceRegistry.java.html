 25 package com.sun.glass.ui.monocle;
 26 
 27 import com.sun.glass.ui.monocle.TouchState.Point;
 28 import java.security.AllPermission;
 29 import javafx.application.Platform;
 30 
 31 
 32 public class AndroidInputDeviceRegistry extends InputDeviceRegistry {
 33 
 34     private static AndroidInputDeviceRegistry instance = new AndroidInputDeviceRegistry();
 35     private AndroidInputDevice androidDevice;
 36     private AndroidInputProcessor processor;
 37     private final KeyState keyState = new KeyState();
 38 
 39     static AndroidInputDeviceRegistry getInstance() {
 40         return instance;
 41     }
 42 
 43     public static void registerDevice() {
 44         Platform.runLater(() -> instance.createDevice());
 45     }
 46 
 47     public static void gotTouchEventFromNative(int count, int[] actions, int[] ids, int[] x, int[] y, int primary) {
 48         TouchState touchState = new TouchState();
 49 
 50         if (primary == -1) {
 51            System.out.println("don't add points, primary = -1");
 52         }
 53         else {
 54             for (int i = 0; i < count; i++) {
 55                 Point p = new Point();
 56                 p.id = ids[i];
 57                 p.x = x[i];
 58                 p.y = y[i];
 59                 touchState.addPoint(p);
 60             }
 61         }
 62         Platform.runLater(() -> instance.gotTouchEvent(touchState));
 63     }
 64 
 65     private void gotTouchEvent(TouchState touchState) {
 66         if (androidDevice == null) {
 67             System.out.println("[MON] got touch event, but no registered device yet");
 68             Thread.dumpStack();
 69             return;
 70         }
 71         if (processor == null) {
 72             System.out.println("[MON] got touch event, but processor not yet initialized");
 73             Thread.dumpStack();
 74             return;
 75         }
 76         processor.pushEvent(touchState);
 77     }
 78 
 79 
 80     public static void dispatchKeyEventFromNative(int type, int key, char[] chars, int modifiers) {
 81         instance.processor.dispatchKeyEvent(type, key, chars, modifiers);
 82     }
 83 
 84     public static void gotKeyEventFromNative(int action, int linuxKey) {
 85         instance.gotKeyEvent (action, linuxKey);
 86     }
 87 
 88     private void gotKeyEvent(int action, int lk) {
 89         int vk = LinuxKeyProcessor.getVirtualKeyCode(lk);
 90         if (action == 0) {
 91             keyState.pressKey(vk);
 92         }
 93         else if (action ==1) {
 94             keyState.releaseKey(vk);
 95         }
 96         else {
 97             System.out.println("[JVDBG] ERROR, what action is this? "+action);
 98         }
 99         instance.gotKeyEvent(keyState);
100     }
101 
102     private void gotKeyEvent(KeyState keyState) {
103         processor.pushKeyEvent(keyState);
104     }
105 
106     private AndroidInputDeviceRegistry() {
107     }
108 
109     private void createDevice() {
110         System.out.println("[MON] Create device");
111         AndroidInputDevice device = new AndroidInputDevice();
112         androidDevice = addDeviceInternal(device, "Android Touch Input");
113         System.out.println("[MON] Create device done, add done");
114     }
115 
116     private AndroidInputDevice addDeviceInternal(AndroidInputDevice device, String name) {
117         processor = createInputProcessor(device);
118 
119         device.setInputProcessor(processor);
120         Thread thread = new Thread(device);
121         thread.setName(name);
122         thread.setDaemon(true);
123         thread.start();
124         devices.add(device);
125 
126         return device;
127 
128     }
129 
130     void removeDevice(AndroidInputDevice device) {
131         SecurityManager security = System.getSecurityManager();
132         if (security != null) {
133             security.checkPermission(new AllPermission());
134         }
135         devices.remove(device);
136     }
137 
138     private AndroidInputProcessor createInputProcessor(AndroidInputDevice device) {
139         return new AndroidInputProcessor(device);
140     }
141 
142 }
