 34 // Bit mask for tracking gesture begin / end
 35 typedef enum GestureMaskType {
 36     GESTURE_MASK_SCROLL  = 1 << 0,
 37     GESTURE_MASK_SWIPE   = 1 << 1,
 38     GESTURE_MASK_ROTATE  = 1 << 2,
 39     GESTURE_MASK_MAGNIFY = 1 << 3,
 40 } GestureMaskType;
