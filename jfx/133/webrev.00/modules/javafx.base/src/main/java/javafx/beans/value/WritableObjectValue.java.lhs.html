26 package javafx.beans.value;
27 
28 /**
29  * A writable typed value.
30  *
31  * @param T
32  *            The type of the wrapped value
33  *
34  * @see WritableValue
35  *
36  *
37  * @since JavaFX 2.0
38  */
39 public interface WritableObjectValue<T> extends WritableValue<T> {
40 
41     /**
42      * Get the wrapped value. This must be identical to
43      * the value returned from {@link #getValue()}.
44      * <p>
45      * This method exists only to align WritableObjectValue API with
46      * {@link WritableBooleanValue} and subclasses of {@link WritableNumberValue}
47      * @return The current value
48      */
49     T get();
50 
51     /**
52      * Set the wrapped value.
53      * Should be equivalent to {@link #setValue(java.lang.Object) }
54      * @see #get()
55      *
56      * @param value
57      *            The new value
58      */
59     void set(T value);
60 
61 }
