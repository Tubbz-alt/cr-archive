 26 package test.com.sun.javafx.scene.control.infrastructure;
 27 
 28 import java.lang.ref.WeakReference;
 29 import java.util.Collection;
 30 
 31 import org.junit.After;
 32 import org.junit.Before;
 33 import org.junit.Test;
 34 import org.junit.runner.RunWith;
 35 import org.junit.runners.Parameterized;
 36 
 37 import com.sun.javafx.scene.control.behavior.BehaviorBase;
 38 
 39 import static org.junit.Assert.*;
 40 import static test.com.sun.javafx.scene.control.infrastructure.ControlSkinFactory.*;
 41 
 42 import javafx.scene.control.Control;
 43 
 44 /**
 45  * Example of writing a test for a streak of similar issues,
 46  * here f.i. memory leak in behavior.
 47  *
 48  * Note: doesn't have postfix _Test_ to not be included in
 49  * default testing.
 50  */
 51 @RunWith(Parameterized.class)
 52 public class ControlBehaviorTestExample {
 53 
 54     private Control control;
 55     private Class<Control> controlClass;
 56 
 57     /**
 58      * Create behavior -> dispose behavior -> gc
 59      */
 60     @Test
 61     public void testMemoryLeakDisposeBehavior() {
 62         WeakReference<BehaviorBase<?>> weakRef = new WeakReference<>(createBehavior(control));
 63         assertNotNull(weakRef.get());
 64         weakRef.get().dispose();
 65         attemptGC(weakRef);
 66         assertNull("behavior must be gc'ed", weakRef.get());
 67     }
 68 
 69 //------------ parameters
 70 
 71     // Note: name property not supported before junit 4.11
 72     @Parameterized.Parameters //(name = "{index}: {0} ")
 73     public static Collection<Object[]> data() {
 74         return asArrays(getControlClassesWithBehavior());
 75     }
 76 
 77     public ControlBehaviorTestExample(Class<Control> controlClass) {
 78         this.controlClass = controlClass;
 79     }
 80 
 81 //------------ setup
 82 
 83     @Before
 84     public void setup() {
 85         assertNotNull(controlClass);
 86 
 87         Thread.currentThread().setUncaughtExceptionHandler((thread, throwable) -> {
 88             if (throwable instanceof RuntimeException) {
 89                 throw (RuntimeException)throwable;
 90             } else {
 91                 Thread.currentThread().getThreadGroup().uncaughtException(thread, throwable);
 92             }
 93         });
 94 
 95         control = createControl(controlClass);
 96     }
 97 
 98     @After
 99     public void cleanup() {
100         Thread.currentThread().setUncaughtExceptionHandler(null);
101     }
102 
103 }
