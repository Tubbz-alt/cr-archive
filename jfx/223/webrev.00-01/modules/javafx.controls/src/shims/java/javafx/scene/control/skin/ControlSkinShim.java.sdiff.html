37 public class ControlSkinShim {

39     /**
40      * Reflectively accesses and returns the value of the skin's behavior field.
41      *
42      * @param skin the skin to get the behavior from
43      * @return the value of the skin's behavior field
44      * @throws RuntimeException with the exception thrown by the reflective access
45      */
46     public static BehaviorBase<?> getBehavior(Skin<?> skin) {
47         try {
48             Field field = skin.getClass().getDeclaredField("behavior");
49             field.setAccessible(true);
50             return (BehaviorBase<?>) field.get(skin);
51         } catch (NoSuchFieldException | SecurityException | IllegalArgumentException | IllegalAccessException e) {
52             throw new RuntimeException("failed access to behavior in " + skin.getClass(), e);
53         }
54     }

58 }
