28 import javafx.beans.value.ChangeListener;
29 import javafx.beans.value.ObservableValue;
30 import javafx.scene.control.TextArea;
31 import javafx.scene.control.skin.TextAreaSkin;
32 
33 public class TextAreaSkinAndroid extends TextAreaSkin {
34 
35     public TextAreaSkinAndroid(final TextArea textArea) {
36         super(textArea);
37 
38         textArea.focusedProperty().addListener(new ChangeListener<Boolean>() {
39             public void changed(ObservableValue<? extends Boolean> observable,
40                     Boolean wasFocused, Boolean isFocused) {
41                 if (textArea.isEditable()) {
42                     if (isFocused) {
43                         showSoftwareKeyboard();
44                     } else {
45                         hideSoftwareKeyboard();
46                     }



47                 }
48             }
49         });
50     }
51 
52     native void showSoftwareKeyboard();
53     native void hideSoftwareKeyboard();
54 
55 }

26 package javafx.scene.control.skin;

28 import javafx.scene.control.TextArea;
29 import javafx.scene.input.MouseEvent;
30 
31 public class TextAreaSkinAndroid extends TextAreaSkin {
32 
33     public TextAreaSkinAndroid(final TextArea textArea) {
34         super(textArea);
35 
36         textArea.addEventHandler(MouseEvent.MOUSE_CLICKED, e -> {
37             if (textArea.isEditable() && textArea.isFocused()) {
38                 showSoftwareKeyboard();
39             }
40         });
41 
42         textArea.focusedProperty().addListener((observable, wasFocused, isFocused) -> {
43             if (textArea.isEditable()) {
44                 if (isFocused) {
45                     showSoftwareKeyboard();
46                 } else {
47                     hideSoftwareKeyboard();
48                 }
49             }
50         });
51     }
52 
53     native void showSoftwareKeyboard();
54     native void hideSoftwareKeyboard();
55 
56 }
