26 package javafx.scene.control.skin;
27 
28 import javafx.scene.control.TextField;
29 import javafx.scene.input.MouseEvent;
30 
31 public class TextFieldSkinAndroid extends TextFieldSkin {
32 
33     public TextFieldSkinAndroid(final TextField textField) {
34         super(textField);
35 
36         textField.addEventHandler(MouseEvent.MOUSE_CLICKED, e -> {
37             if (textField.isEditable() && textField.isFocused()) {
38                 showSoftwareKeyboard();
39             }
40         });
41 
42         textField.focusedProperty().addListener((observable, wasFocused, isFocused) -> {
43             if (textField.isEditable()) {
44                 if (isFocused) {
45                     showSoftwareKeyboard();
46                 } else {
47                     hideSoftwareKeyboard();
48                 }
49             }
50         });
51     }
52 
53     native void showSoftwareKeyboard();
54     native void hideSoftwareKeyboard();
55 
56 }
