115         Util.runAndWait(() -> {
116             webView.getEngine().getLoadWorker().stateProperty().
117                 addListener((observable, oldValue, newValue) -> {
118                 if (newValue == SUCCEEDED) {

123             assertNotNull(webView);
124             webView.getEngine().loadContent(htmlSVGContent);
