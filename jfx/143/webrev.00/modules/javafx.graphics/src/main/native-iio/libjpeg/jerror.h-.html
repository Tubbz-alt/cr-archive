207 /* Macros to simplify using the error and trace message stuff */
208 /* The first parameter is either type of cinfo pointer */
209 
210 /* Fatal errors (print message and exit) */
211 #define ERREXIT(cinfo,code)  \
212   ((cinfo)->err->msg_code = (code), \
213    (*(cinfo)->err->error_exit) ((j_common_ptr) (cinfo)))
214 #define ERREXIT1(cinfo,code,p1)  \
215   ((cinfo)->err->msg_code = (code), \
216    (cinfo)->err->msg_parm.i[0] = (p1), \
217    (*(cinfo)->err->error_exit) ((j_common_ptr) (cinfo)))
218 #define ERREXIT2(cinfo,code,p1,p2)  \
219   ((cinfo)->err->msg_code = (code), \
220    (cinfo)->err->msg_parm.i[0] = (p1), \
221    (cinfo)->err->msg_parm.i[1] = (p2), \
222    (*(cinfo)->err->error_exit) ((j_common_ptr) (cinfo)))
223 #define ERREXIT3(cinfo,code,p1,p2,p3)  \
224   ((cinfo)->err->msg_code = (code), \
225    (cinfo)->err->msg_parm.i[0] = (p1), \
226    (cinfo)->err->msg_parm.i[1] = (p2), \
227    (cinfo)->err->msg_parm.i[2] = (p3), \
228    (*(cinfo)->err->error_exit) ((j_common_ptr) (cinfo)))
229 #define ERREXIT4(cinfo,code,p1,p2,p3,p4)  \
230   ((cinfo)->err->msg_code = (code), \
231    (cinfo)->err->msg_parm.i[0] = (p1), \
232    (cinfo)->err->msg_parm.i[1] = (p2), \
233    (cinfo)->err->msg_parm.i[2] = (p3), \
234    (cinfo)->err->msg_parm.i[3] = (p4), \
235    (*(cinfo)->err->error_exit) ((j_common_ptr) (cinfo)))
236 #define ERREXIT6(cinfo,code,p1,p2,p3,p4,p5,p6)  \
237   ((cinfo)->err->msg_code = (code), \
238    (cinfo)->err->msg_parm.i[0] = (p1), \
239    (cinfo)->err->msg_parm.i[1] = (p2), \
240    (cinfo)->err->msg_parm.i[2] = (p3), \
241    (cinfo)->err->msg_parm.i[3] = (p4), \
242    (cinfo)->err->msg_parm.i[4] = (p5), \
243    (cinfo)->err->msg_parm.i[5] = (p6), \
244    (*(cinfo)->err->error_exit) ((j_common_ptr) (cinfo)))
245 #define ERREXITS(cinfo,code,str)  \
246   ((cinfo)->err->msg_code = (code), \
247    strncpy((cinfo)->err->msg_parm.s, (str), JMSG_STR_PARM_MAX), \
248    (*(cinfo)->err->error_exit) ((j_common_ptr) (cinfo)))
249 
250 #define MAKESTMT(stuff)        do { stuff } while (0)

252 /* Nonfatal errors (we can keep going, but the data is probably corrupt) */
253 #define WARNMS(cinfo,code)  \
254   ((cinfo)->err->msg_code = (code), \
255    (*(cinfo)->err->emit_message) ((j_common_ptr) (cinfo), -1))
256 #define WARNMS1(cinfo,code,p1)  \
257   ((cinfo)->err->msg_code = (code), \
258    (cinfo)->err->msg_parm.i[0] = (p1), \
259    (*(cinfo)->err->emit_message) ((j_common_ptr) (cinfo), -1))
260 #define WARNMS2(cinfo,code,p1,p2)  \
261   ((cinfo)->err->msg_code = (code), \
262    (cinfo)->err->msg_parm.i[0] = (p1), \
263    (cinfo)->err->msg_parm.i[1] = (p2), \
264    (*(cinfo)->err->emit_message) ((j_common_ptr) (cinfo), -1))

266 /* Informational/debugging messages */
267 #define TRACEMS(cinfo,lvl,code)  \
268   ((cinfo)->err->msg_code = (code), \
269    (*(cinfo)->err->emit_message) ((j_common_ptr) (cinfo), (lvl)))
270 #define TRACEMS1(cinfo,lvl,code,p1)  \
271   ((cinfo)->err->msg_code = (code), \
272    (cinfo)->err->msg_parm.i[0] = (p1), \
273    (*(cinfo)->err->emit_message) ((j_common_ptr) (cinfo), (lvl)))
274 #define TRACEMS2(cinfo,lvl,code,p1,p2)  \
275   ((cinfo)->err->msg_code = (code), \
276    (cinfo)->err->msg_parm.i[0] = (p1), \
277    (cinfo)->err->msg_parm.i[1] = (p2), \
278    (*(cinfo)->err->emit_message) ((j_common_ptr) (cinfo), (lvl)))
279 #define TRACEMS3(cinfo,lvl,code,p1,p2,p3)  \
280   MAKESTMT(int * _mp = (cinfo)->err->msg_parm.i; \
281        _mp[0] = (p1); _mp[1] = (p2); _mp[2] = (p3); \
282        (cinfo)->err->msg_code = (code); \
283        (*(cinfo)->err->emit_message) ((j_common_ptr) (cinfo), (lvl)); )
284 #define TRACEMS4(cinfo,lvl,code,p1,p2,p3,p4)  \
285   MAKESTMT(int * _mp = (cinfo)->err->msg_parm.i; \
286        _mp[0] = (p1); _mp[1] = (p2); _mp[2] = (p3); _mp[3] = (p4); \
287        (cinfo)->err->msg_code = (code); \
288        (*(cinfo)->err->emit_message) ((j_common_ptr) (cinfo), (lvl)); )
289 #define TRACEMS5(cinfo,lvl,code,p1,p2,p3,p4,p5)  \
290   MAKESTMT(int * _mp = (cinfo)->err->msg_parm.i; \
291        _mp[0] = (p1); _mp[1] = (p2); _mp[2] = (p3); _mp[3] = (p4); \
292        _mp[4] = (p5); \
293        (cinfo)->err->msg_code = (code); \
294        (*(cinfo)->err->emit_message) ((j_common_ptr) (cinfo), (lvl)); )
295 #define TRACEMS8(cinfo,lvl,code,p1,p2,p3,p4,p5,p6,p7,p8)  \
296   MAKESTMT(int * _mp = (cinfo)->err->msg_parm.i; \
297        _mp[0] = (p1); _mp[1] = (p2); _mp[2] = (p3); _mp[3] = (p4); \
298        _mp[4] = (p5); _mp[5] = (p6); _mp[6] = (p7); _mp[7] = (p8); \
299        (cinfo)->err->msg_code = (code); \
300        (*(cinfo)->err->emit_message) ((j_common_ptr) (cinfo), (lvl)); )
301 #define TRACEMSS(cinfo,lvl,code,str)  \
302   ((cinfo)->err->msg_code = (code), \
303    strncpy((cinfo)->err->msg_parm.s, (str), JMSG_STR_PARM_MAX), \
304    (*(cinfo)->err->emit_message) ((j_common_ptr) (cinfo), (lvl)))
