123     private MapChangeListener<Object, Object> propertiesMapListener = c -> {
124         if (! c.wasAdded()) return;
125         if (Properties.RECREATE.equals(c.getKey())) {
126             needCellsRebuilt = true;
127             getSkinnable().requestLayout();
128             getSkinnable().getProperties().remove(Properties.RECREATE);
129         }
130     };

135     private final ListChangeListener<T> listViewItemsListener = new ListChangeListener<T>() {
136         @Override public void onChanged(Change<? extends T> c) {
137             while (c.next()) {
138                 if (c.wasReplaced()) {
139                     // RT-28397: Support for when an item is replaced with itself (but
140                     // updated internal values that should be shown visually).
141                     // This code was updated for RT-36714 to not update all cells,
142                     // just those affected by the change
143                     for (int i = c.getFrom(); i < c.getTo(); i++) {
144                         flow.setCellDirty(i);
145                     }
146 
147                     break;
148                 } else if (c.getRemovedSize() == itemCount) {
149                     // RT-22463: If the user clears out an items list then we
150                     // should reset all cells (in particular their contained
151                     // items) such that a subsequent addition to the list of
152                     // an item which equals the old item (but is rendered
153                     // differently) still displays as expected (i.e. with the
154                     // updated display, not the old display).
155                     itemCount = 0;
156                     break;
157                 }
158             }
159 
160             // fix for RT-37853
161             getSkinnable().edit(-1);
162 
163             markItemCountDirty();
164             getSkinnable().requestLayout();
165         }
166     };

185     /**
186      * Creates a new ListViewSkin instance, installing the necessary child
187      * nodes into the Control {@link Control#getChildren() children} list, as
188      * well as the necessary input mappings for handling key, mouse, etc events.
189      *
190      * @param control The control that this skin should be installed onto.
191      */
192     public ListViewSkin(final ListView<T> control) {
193         super(control);
194 
195         // install default input map for the ListView control
196         behavior = new ListViewBehavior<>(control);
197 //        control.setInputMap(behavior.getInputMap());
198 
199         // init the behavior 'closures'
200         behavior.setOnFocusPreviousRow(() -> onFocusPreviousCell());
201         behavior.setOnFocusNextRow(() -> onFocusNextCell());
202         behavior.setOnMoveToFirstCell(() -> onMoveToFirstCell());
203         behavior.setOnMoveToLastCell(() -> onMoveToLastCell());
204         behavior.setOnSelectPreviousRow(() -> onSelectPreviousCell());
205         behavior.setOnSelectNextRow(() -> onSelectNextCell());
206         behavior.setOnScrollPageDown(this::onScrollPageDown);
207         behavior.setOnScrollPageUp(this::onScrollPageUp);
208         Boolean isRemoveKeyMappings = (Boolean) control.getProperties().get("removeKeyMappingsForComboBoxEditor");
209         if (isRemoveKeyMappings != null && isRemoveKeyMappings) {
210             behavior.removeKeyMappingsForComboBoxEditor();
211         }
212 
213         updateListViewItems();
214 
215         // init the VirtualFlow
216         flow = getVirtualFlow();
217         flow.setId("virtual-flow");
218         flow.setPannable(IS_PANNABLE);
219         flow.setVertical(control.getOrientation() == Orientation.VERTICAL);
220         flow.setCellFactory(flow -> createCell());
221         flow.setFixedCellSize(control.getFixedCellSize());
222         getChildren().add(flow);
223 
224         ml = event -> {
225             // RT-15127: cancel editing on scroll. This is a bit extreme
226             // (we are cancelling editing on touching the scrollbars).
227             // This can be improved at a later date.
228             if (control.getEditingIndex() > -1) {
229                 control.edit(-1);
230             }
231 
232             // This ensures that the list maintains the focus, even when the vbar
233             // and hbar controls inside the flow are clicked. Without this, the
234             // focus border will not be shown when the user interacts with the
235             // scrollbars, and more importantly, keyboard navigation won't be
236             // available to the user.
237             if (control.isFocusTraversable()) {
238                 control.requestFocus();
239             }
240         };
241         flow.getVbar().addEventFilter(MouseEvent.MOUSE_PRESSED, ml);
242         flow.getHbar().addEventFilter(MouseEvent.MOUSE_PRESSED, ml);
243 
244         updateItemCount();
245 
246         control.itemsProperty().addListener(weakItemsChangeListener);
247 
248         final ObservableMap<Object, Object> properties = control.getProperties();
249         properties.remove(Properties.RECREATE);
250         properties.addListener(weakPropertiesMapListener);
251 
252         // Register listeners
253         registerChangeListener(control.itemsProperty(), o -> updateListViewItems());
254         registerChangeListener(control.orientationProperty(), o ->
255             flow.setVertical(control.getOrientation() == Orientation.VERTICAL)
256         );
257         registerChangeListener(control.cellFactoryProperty(), o -> flow.recreateCells());
258         registerChangeListener(control.parentProperty(), o -> {
259             if (control.getParent() != null && control.isVisible()) {
260                 control.requestLayout();
261             }
262         });
263         registerChangeListener(control.placeholderProperty(), o -> updatePlaceholderRegionVisibility());
264         registerChangeListener(control.fixedCellSizeProperty(), o ->
265             flow.setFixedCellSize(control.getFixedCellSize())
266         );
267     }

275     /** {@inheritDoc} */
276     @Override public void dispose() {
277         if (getSkinnable() == null) return;
278         // listener cleanup fixes side-effects (NPE on refresh, setItems, modifyItems)
279         getSkinnable().getProperties().removeListener(weakPropertiesMapListener);
280         getSkinnable().itemsProperty().removeListener(weakItemsChangeListener);
281         if (listViewItems != null) {
282             listViewItems.removeListener(weakListViewItemsListener);
283             listViewItems = null;
284         }
285         // flow related cleanup
286         // leaking without nulling factory
287         flow.setCellFactory(null);
288         // for completeness - but no effect with/out?
289         flow.getVbar().removeEventFilter(MouseEvent.MOUSE_PRESSED, ml);
290         flow.getHbar().removeEventFilter(MouseEvent.MOUSE_PRESSED, ml);
291         super.dispose();
292 
293         if (behavior != null) {
294             behavior.dispose();
295         }
296     }

298     /** {@inheritDoc} */
299     @Override protected void layoutChildren(final double x, final double y,
300                                             final double w, final double h) {
301         super.layoutChildren(x, y, w, h);
302 
303         if (needCellsRebuilt) {
304             flow.rebuildCells();
305         } else if (needCellsReconfigured) {
306             flow.reconfigureCells();
307         }
308 
309         needCellsRebuilt = false;
310         needCellsReconfigured = false;
311 
312         if (getItemCount() == 0) {
313             // show message overlay instead of empty listview
314             if (placeholderRegion != null) {
315                 placeholderRegion.setVisible(w > 0 && h > 0);
316                 placeholderRegion.resizeRelocate(x, y, w, h);
317             }
318         } else {
319             flow.resizeRelocate(x, y, w, h);
320         }
321     }

323     /** {@inheritDoc} */
324     @Override protected double computePrefWidth(double height, double topInset, double rightInset, double bottomInset, double leftInset) {
325         checkState();
326 
327         if (getItemCount() == 0) {
328             if (placeholderRegion == null) {
329                 updatePlaceholderRegionVisibility();
330             }
331             if (placeholderRegion != null) {
332                 return placeholderRegion.prefWidth(height) + leftInset + rightInset;
333             }
334         }
335 
336         return computePrefHeight(-1, topInset, rightInset, bottomInset, leftInset) * 0.618033987;
337     }

341         return 400;

349     /** {@inheritDoc} */
350     @Override protected void updateItemCount() {
351         if (flow == null) return;
352 
353         int oldCount = itemCount;
354         int newCount = listViewItems == null ? 0 : listViewItems.size();
355 
356         itemCount = newCount;
357 
358         flow.setCellCount(newCount);
359 
360         updatePlaceholderRegionVisibility();
361         if (newCount != oldCount) {
362             requestRebuildCells();
363         } else {
364             needCellsReconfigured = true;
365         }
366     }

368     /** {@inheritDoc} */
369     @Override protected Object queryAccessibleAttribute(AccessibleAttribute attribute, Object... parameters) {
370         switch (attribute) {
371             case FOCUS_ITEM: {
372                 FocusModel<?> fm = getSkinnable().getFocusModel();
373                 int focusedIndex = fm.getFocusedIndex();
374                 if (focusedIndex == -1) {
375                     if (placeholderRegion != null && placeholderRegion.isVisible()) {
376                         return placeholderRegion.getChildren().get(0);
377                     }
378                     if (getItemCount() > 0) {
379                         focusedIndex = 0;
380                     } else {
381                         return null;
382                     }
383                 }
384                 return flow.getPrivateCell(focusedIndex);
385             }
386             case ITEM_COUNT: return getItemCount();
387             case ITEM_AT_INDEX: {
388                 Integer rowIndex = (Integer)parameters[0];
389                 if (rowIndex == null) return null;
390                 if (0 <= rowIndex && rowIndex < getItemCount()) {
391                     return flow.getPrivateCell(rowIndex);
392                 }
393                 return null;
394             }
395             case SELECTED_ITEMS: {
396                 MultipleSelectionModel<T> sm = getSkinnable().getSelectionModel();
397                 ObservableList<Integer> indices = sm.getSelectedIndices();
398                 List<Node> selection = new ArrayList<>(indices.size());
399                 for (int i : indices) {
400                     ListCell<T> row = flow.getPrivateCell(i);
401                     if (row != null) selection.add(row);
402                 }
403                 return FXCollections.observableArrayList(selection);
404             }
405             case VERTICAL_SCROLLBAR: return flow.getVbar();
406             case HORIZONTAL_SCROLLBAR: return flow.getHbar();
407             default: return super.queryAccessibleAttribute(attribute, parameters);
408         }
409     }

411     /** {@inheritDoc} */
412     @Override protected void executeAccessibleAction(AccessibleAction action, Object... parameters) {
413         switch (action) {
414             case SHOW_ITEM: {
415                 Node item = (Node)parameters[0];
416                 if (item instanceof ListCell) {
417                     @SuppressWarnings("unchecked")
418                     ListCell<T> cell = (ListCell<T>)item;
419                     flow.scrollTo(cell.getIndex());
420                 }
421                 break;
422             }
423             case SET_SELECTED_ITEMS: {
424                 @SuppressWarnings("unchecked")
425                 ObservableList<Node> items = (ObservableList<Node>)parameters[0];
426                 if (items != null) {
427                     MultipleSelectionModel<T> sm = getSkinnable().getSelectionModel();
428                     if (sm != null) {
429                         sm.clearSelection();
430                         for (Node item : items) {
431                             if (item instanceof ListCell) {
432                                 @SuppressWarnings("unchecked")
433                                 ListCell<T> cell = (ListCell<T>)item;
434                                 sm.select(cell.getIndex());
435                             }
436                         }
437                     }
438                 }
439                 break;
440             }
441             default: super.executeAccessibleAction(action, parameters);
442         }
443     }

453     /** {@inheritDoc} */
454     private ListCell<T> createCell() {
455         ListCell<T> cell;
456         if (getSkinnable().getCellFactory() != null) {
457             cell = getSkinnable().getCellFactory().call(getSkinnable());
458         } else {
459             cell = createDefaultCellImpl();
460         }
461 
462         cell.updateListView(getSkinnable());
463 
464         return cell;
465     }
466 
467     private void updateListViewItems() {
468         if (listViewItems != null) {
469             listViewItems.removeListener(weakListViewItemsListener);
470         }
471 
472         this.listViewItems = getSkinnable().getItems();
473 
474         if (listViewItems != null) {
475             listViewItems.addListener(weakListViewItemsListener);
476         }
477 
478         markItemCountDirty();
479         getSkinnable().requestLayout();
480     }

482     private final void updatePlaceholderRegionVisibility() {
483         boolean visible = getItemCount() == 0;
484 
485         if (visible) {
486             placeholderNode = getSkinnable().getPlaceholder();
487             if (placeholderNode == null && (EMPTY_LIST_TEXT != null && ! EMPTY_LIST_TEXT.isEmpty())) {
488                 placeholderNode = new Label();
489                 ((Label)placeholderNode).setText(EMPTY_LIST_TEXT);
490             }
491 
492             if (placeholderNode != null) {
493                 if (placeholderRegion == null) {
494                     placeholderRegion = new StackPane();
495                     placeholderRegion.getStyleClass().setAll("placeholder");
496                     getChildren().add(placeholderRegion);
497                 }
498 
499                 placeholderRegion.getChildren().setAll(placeholderNode);
500             }
501         }
502 
503         flow.setVisible(!visible);
504         if (placeholderRegion != null) {
505             placeholderRegion.setVisible(visible);
506         }
507     }

509     private static <T> ListCell<T> createDefaultCellImpl() {
510         return new ListCell<T>() {
511             @Override public void updateItem(T item, boolean empty) {
512                 super.updateItem(item, empty);
513 
514                 if (empty) {
515                     setText(null);
516                     setGraphic(null);
517                 } else if (item instanceof Node) {
518                     setText(null);
519                     Node currentNode = getGraphic();
520                     Node newNode = (Node) item;
521                     if (currentNode == null || ! currentNode.equals(newNode)) {
522                         setGraphic(newNode);
523                     }
524                 } else {
525                     /**
526                      * This label is used if the item associated with this cell is to be
527                      * represented as a String. While we will lazily instantiate it
528                      * we never clear it, being more afraid of object churn than a minor
529                      * "leak" (which will not become a "major" leak).
530                      */
531                     setText(item == null ? "null" : item.toString());
532                     setGraphic(null);
533                 }
534             }
535         };
536     }

544     private void onFocusNextCell() {

550     private void onSelectPreviousCell() {
551         SelectionModel<T> sm = getSkinnable().getSelectionModel();
552         if (sm == null) return;
553 
554         int pos = sm.getSelectedIndex();
555         flow.scrollTo(pos);
556 
557         // Fix for RT-11299
558         IndexedCell<T> cell = flow.getFirstVisibleCell();
559         if (cell == null || pos < cell.getIndex()) {
560             flow.setPosition(pos / (double) getItemCount());
561         }
562     }
563 
564     private void onSelectNextCell() {
565         SelectionModel<T> sm = getSkinnable().getSelectionModel();
566         if (sm == null) return;
567 
568         int pos = sm.getSelectedIndex();
569         flow.scrollTo(pos);
570 
571         // Fix for RT-11299
572         ListCell<T> cell = flow.getLastVisibleCell();
573         if (cell == null || cell.getIndex() < pos) {
574             flow.setPosition(pos / (double) getItemCount());
575         }
576     }

593     /**
594      * Function used to scroll the container down by one 'page', although
595      * if this is a horizontal container, then the scrolling will be to the right.
596      */
597     private int onScrollPageDown(boolean isFocusDriven) {
598         ListCell<T> lastVisibleCell = flow.getLastVisibleCellWithinViewport();
599         if (lastVisibleCell == null) return -1;
600 
601         final SelectionModel<T> sm = getSkinnable().getSelectionModel();
602         final FocusModel<T> fm = getSkinnable().getFocusModel();
603         if (sm == null || fm == null) return -1;
604 
605         int lastVisibleCellIndex = lastVisibleCell.getIndex();
606 
607 //        boolean isSelected = sm.isSelected(lastVisibleCellIndex) || fm.isFocused(lastVisibleCellIndex) || lastVisibleCellIndex == anchor;
608         // isSelected represents focus OR selection
609         boolean isSelected = false;
610         if (isFocusDriven) {
611             isSelected = lastVisibleCell.isFocused() || fm.isFocused(lastVisibleCellIndex);
612         } else {
613             isSelected = lastVisibleCell.isSelected() || sm.isSelected(lastVisibleCellIndex);
614         }
615 
616         if (isSelected) {
617             boolean isLeadIndex = (isFocusDriven && fm.getFocusedIndex() == lastVisibleCellIndex)
618                                || (! isFocusDriven && sm.getSelectedIndex() == lastVisibleCellIndex);
619 
620             if (isLeadIndex) {
621                 // if the last visible cell is selected, we want to shift that cell up
622                 // to be the top-most cell, or at least as far to the top as we can go.
623                 flow.scrollToTop(lastVisibleCell);
624 
625                 ListCell<T> newLastVisibleCell = flow.getLastVisibleCellWithinViewport();
626                 lastVisibleCell = newLastVisibleCell == null ? lastVisibleCell : newLastVisibleCell;
627             }
628         } else {
629             // if the selection is not on the 'bottom' most cell, we firstly move
630             // the selection down to that, without scrolling the contents, so
631             // this is a no-op
632         }
633 
634         int newSelectionIndex = lastVisibleCell.getIndex();
635         flow.scrollTo(lastVisibleCell);
636         return newSelectionIndex;
637     }

639     /**
640      * Function used to scroll the container up by one 'page', although
641      * if this is a horizontal container, then the scrolling will be to the left.
642      */
643     private int onScrollPageUp(boolean isFocusDriven) {
644         ListCell<T> firstVisibleCell = flow.getFirstVisibleCellWithinViewport();
645         if (firstVisibleCell == null) return -1;
646 
647         final SelectionModel<T> sm = getSkinnable().getSelectionModel();
648         final FocusModel<T> fm = getSkinnable().getFocusModel();
649         if (sm == null || fm == null) return -1;
650 
651         int firstVisibleCellIndex = firstVisibleCell.getIndex();
652 
653         // isSelected represents focus OR selection
654         boolean isSelected = false;
655         if (isFocusDriven) {
656             isSelected = firstVisibleCell.isFocused() || fm.isFocused(firstVisibleCellIndex);
657         } else {
658             isSelected = firstVisibleCell.isSelected() || sm.isSelected(firstVisibleCellIndex);
659         }
660 
661         if (isSelected) {
662             boolean isLeadIndex = (isFocusDriven && fm.getFocusedIndex() == firstVisibleCellIndex)
663                                || (! isFocusDriven && sm.getSelectedIndex() == firstVisibleCellIndex);
664 
665             if (isLeadIndex) {
666                 // if the first visible cell is selected, we want to shift that cell down
667                 // to be the bottom-most cell, or at least as far to the bottom as we can go.
668                 flow.scrollToBottom(firstVisibleCell);
669 
670                 ListCell<T> newFirstVisibleCell = flow.getFirstVisibleCellWithinViewport();
671                 firstVisibleCell = newFirstVisibleCell == null ? firstVisibleCell : newFirstVisibleCell;
672             }
673         } else {
674             // if the selection is not on the 'top' most cell, we firstly move
675             // the selection up to that, without scrolling the contents, so
676             // this is a no-op
677         }
678 
679         int newSelectionIndex = firstVisibleCell.getIndex();
680         flow.scrollTo(firstVisibleCell);
681         return newSelectionIndex;
