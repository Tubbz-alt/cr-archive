26 package pseudoScriptEngineCompilable;
27 
28 import javax.script.Bindings;
29 import javax.script.CompiledScript;
30 import javax.script.ScriptContext;
31 import javax.script.ScriptEngine;
32 import javax.script.ScriptException;
33 
34 public class RgfPseudoCompiledScript extends CompiledScript {

35     String code = null;
36     ScriptEngine scriptEngine = null;
37 
38     RgfPseudoCompiledScript(String code, ScriptEngine scriptEngine) {
39         this.code = code;
40         this.scriptEngine = scriptEngine;
41     }
42 
43     public Object eval(Bindings bindings) throws ScriptException {
44         return scriptEngine.eval("RgfPseudoCompiledScript.eval(Bindings bindings): " + code, bindings);
45     }
46 
47     public Object eval(ScriptContext context) throws ScriptException {
48         return scriptEngine.eval("RgfPseudoCompiledScript.eval(ScriptContext context): " + code, context);
49     }
50 
51     public Object eval() throws ScriptException {
52         return scriptEngine.eval("RgfPseudoCompiledScript.eval(): " + code );
53     }
54 
55     public ScriptEngine getEngine() {
56         return scriptEngine;
57     }
58 }
