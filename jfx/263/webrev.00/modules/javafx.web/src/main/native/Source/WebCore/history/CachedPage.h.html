26 #pragma once
27 
28 #include "CachedFrame.h"
29 #include <wtf/MonotonicTime.h>
30 
31 namespace WebCore {
32 
33 class Document;
34 class DocumentLoader;
35 class Page;
36 
37 class CachedPage {
38     WTF_MAKE_FAST_ALLOCATED;
39 public:
40     explicit CachedPage(Page&);
41     ~CachedPage();
42 
43     void restore(Page&);
44     void clear();
45 
46     Page& page() const { return m_page; }
47     Document* document() const { return m_cachedMainFrame->document(); }
48     DocumentLoader* documentLoader() const { return m_cachedMainFrame->documentLoader(); }
49 
50     bool hasExpired() const;
51 
52     CachedFrame* cachedMainFrame() { return m_cachedMainFrame.get(); }
53 
54 #if ENABLE(VIDEO_TRACK)
55     void markForCaptionPreferencesChanged() { m_needsCaptionPreferencesChanged = true; }
56 #endif
57 
58     void markForDeviceOrPageScaleChanged() { m_needsDeviceOrPageScaleChanged = true; }
59 
60     void markForContentsSizeChanged() { m_needsUpdateContentsSize = true; }
61 
62 private:
63     Page& m_page;
64     MonotonicTime m_expirationTime;
65     std::unique_ptr<CachedFrame> m_cachedMainFrame;
66 #if ENABLE(VIDEO_TRACK)
67     bool m_needsCaptionPreferencesChanged { false };
68 #endif
69     bool m_needsDeviceOrPageScaleChanged { false };
70     bool m_needsUpdateContentsSize { false };
71 };
72 
73 } // namespace WebCore
