26 // Base class for all constructor objects in the JSC bindings.
27 class JSDOMConstructorBase : public JSDOMObject {
28 public:
29     using Base = JSDOMObject;
30 
31     static constexpr unsigned StructureFlags = Base::StructureFlags | JSC::ImplementsHasInstance | JSC::ImplementsDefaultHasInstance | JSC::OverridesGetCallData;
32     static constexpr bool needsDestruction = false;
33     static JSC::Structure* createStructure(JSC::VM&, JSC::JSGlobalObject*, JSC::JSValue);
34 
35     template<typename CellType, JSC::SubspaceAccess>
36     static JSC::IsoSubspace* subspaceFor(JSC::VM& vm)
37     {
38         static_assert(sizeof(CellType) == sizeof(JSDOMConstructorBase));
39         STATIC_ASSERT_ISO_SUBSPACE_SHARABLE(CellType, JSDOMConstructorBase);
40         static_assert(CellType::destroy == JSC::JSCell::destroy, "JSDOMConstructor<JSClass> is not destructible actually");
41         return subspaceForImpl(vm);
42     }
43 
44     static JSC::IsoSubspace* subspaceForImpl(JSC::VM&);
45 
46 protected:
47     JSDOMConstructorBase(JSC::Structure* structure, JSDOMGlobalObject& globalObject)
48         : JSDOMObject(structure, globalObject)
49     {
50     }
51 
52     static String className(const JSObject*, JSC::VM&);
53     static String toStringName(const JSObject*, JSC::JSGlobalObject*);
54     static JSC::CallType getCallData(JSCell*, JSC::CallData&);
55 };
56 
57 inline JSC::Structure* JSDOMConstructorBase::createStructure(JSC::VM& vm, JSC::JSGlobalObject* globalObject, JSC::JSValue prototype)
58 {
59     return JSC::Structure::create(vm, globalObject, prototype, JSC::TypeInfo(JSC::ObjectType, StructureFlags), info());
60 }
