26 #pragma once
27 
28 #include "JSCJSValue.h"
29 #include <wtf/PrintStream.h>
30 #include <wtf/WallTime.h>
31 #include <wtf/text/CString.h>
32 
33 namespace JSC { namespace Profiler {
34 
35 class Bytecodes;
36 class Compilation;
37 
38 class Event {
39 public:
40     Event()
41     {
42     }
43 
44     Event(WallTime time, Bytecodes* bytecodes, Compilation* compilation, const char* summary, const CString& detail)
45         : m_time(time)
46         , m_bytecodes(bytecodes)
47         , m_compilation(compilation)
48         , m_summary(summary)
49         , m_detail(detail)
50     {
51     }
52 
53     explicit operator bool() const
54     {
55         return m_bytecodes;
56     }
57 
58     WallTime time() const { return m_time; }
59     Bytecodes* bytecodes() const { return m_bytecodes; }
60     Compilation* compilation() const { return m_compilation; }
61     const char* summary() const { return m_summary; }
62     const CString& detail() const { return m_detail; }
63 
64     void dump(PrintStream&) const;
65     JSValue toJS(JSGlobalObject*) const;
66 
67 private:
68     WallTime m_time { };
69     Bytecodes* m_bytecodes { nullptr };
70     Compilation* m_compilation { nullptr };
71     const char* m_summary { nullptr };
72     CString m_detail;
73 };
74 
75 } } // namespace JSC::Profiler
