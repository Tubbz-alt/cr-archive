 38 void ArrayBufferViewWatchpointAdaptor::add(
 39     CodeBlock* codeBlock, JSArrayBufferView* view, CommonData& common)
 40 {
 41     VM& vm = codeBlock->vm();
 42     Watchpoint* watchpoint = common.watchpoints.add(codeBlock);
 43     ArrayBufferNeuteringWatchpointSet* neuteringWatchpoint =
 44         ArrayBufferNeuteringWatchpointSet::create(vm);
 45     neuteringWatchpoint->set().add(watchpoint);
 46     codeBlock->addConstant(neuteringWatchpoint);


 47     // FIXME: We don't need to set this watchpoint at all for shared buffers.
 48     // https://bugs.webkit.org/show_bug.cgi?id=164108
 49     vm.heap.addReference(neuteringWatchpoint, view->possiblySharedBuffer());
 50 }

 67     CodeBlock* codeBlock, const ObjectPropertyCondition& key, CommonData& common)
 68 {
 69     VM& vm = codeBlock->vm();
 70     switch (key.kind()) {
 71     case PropertyCondition::Equivalence:
 72         common.adaptiveInferredPropertyValueWatchpoints.add(key, codeBlock)->install(vm);





 73         break;
 74     default:
 75         common.adaptiveStructureWatchpoints.add(key, codeBlock)->install(vm);






 76         break;
 77     }

 78 }

 37 void ArrayBufferViewWatchpointAdaptor::add(
 38     CodeBlock* codeBlock, JSArrayBufferView* view, CommonData& common)
 39 {
 40     // view is already frozen. If it is deallocated, jettisoning happens.
 41     CodeBlockJettisoningWatchpoint* watchpoint = nullptr;
 42     {
 43         ConcurrentJSLocker locker(codeBlock->m_lock);
 44         watchpoint = common.watchpoints.add(codeBlock);
 45     }
 46     ArrayBuffer* arrayBuffer = view->possiblySharedBuffer();
 47     RELEASE_ASSERT(arrayBuffer);
 48     // FIXME: We don't need to set this watchpoint at all for shared buffers.
 49     // https://bugs.webkit.org/show_bug.cgi?id=164108
 50     arrayBuffer->neuteringWatchpointSet().add(WTFMove(watchpoint));
 51 }

 53 void SymbolTableAdaptor::add(
 54     CodeBlock* codeBlock, SymbolTable* symbolTable, CommonData& common)
 55 {
 56     codeBlock->addConstant(ConcurrentJSLocker(codeBlock->m_lock), symbolTable); // For common users, it doesn't really matter if it's weak or not. If references to it go away, we go away, too.
 57     CodeBlockJettisoningWatchpoint* watchpoint = nullptr;
 58     {
 59         ConcurrentJSLocker locker(codeBlock->m_lock);
 60         watchpoint = common.watchpoints.add(codeBlock);
 61     }
 62     symbolTable->singleton().add(WTFMove(watchpoint));
 63 }

 77 void AdaptiveStructureWatchpointAdaptor::add(
 78     CodeBlock* codeBlock, const ObjectPropertyCondition& key, CommonData& common)
 79 {
 80     VM& vm = codeBlock->vm();
 81     switch (key.kind()) {
 82     case PropertyCondition::Equivalence: {
 83         AdaptiveInferredPropertyValueWatchpoint* watchpoint = nullptr;
 84         {
 85             ConcurrentJSLocker locker(codeBlock->m_lock);
 86             watchpoint = common.adaptiveInferredPropertyValueWatchpoints.add(key, codeBlock);
 87         }
 88         watchpoint->install(vm);
 89         break;
 90     }
 91     default: {
 92         AdaptiveStructureWatchpoint* watchpoint = nullptr;
 93         {
 94             ConcurrentJSLocker locker(codeBlock->m_lock);
 95             watchpoint = common.adaptiveStructureWatchpoints.add(key, codeBlock);
 96         }
 97         watchpoint->install(vm);
 98         break;
 99     }
100     }
101 }
