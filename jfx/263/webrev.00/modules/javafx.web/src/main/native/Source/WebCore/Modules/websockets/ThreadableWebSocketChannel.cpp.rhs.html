 52 Ref<ThreadableWebSocketChannel> ThreadableWebSocketChannel::create(Document& document, WebSocketChannelClient& client, SocketProvider& provider)
 53 {
 54 #if USE(SOUP)
 55     auto channel = provider.createWebSocketChannel(document, client);
 56     ASSERT(channel);
 57     return channel.releaseNonNull();
 58 #else
 59 
 60 #if HAVE(NSURLSESSION_WEBSOCKET)
 61     if (RuntimeEnabledFeatures::sharedFeatures().isNSURLSessionWebSocketEnabled()) {
 62         if (auto channel = provider.createWebSocketChannel(document, client))
 63             return channel.releaseNonNull();
 64     }
 65 #endif
 66 
 67     return WebSocketChannel::create(document, client, provider);
 68 #endif
 69 }
 70 
 71 Ref<ThreadableWebSocketChannel> ThreadableWebSocketChannel::create(ScriptExecutionContext& context, WebSocketChannelClient& client, SocketProvider& provider)
 72 {
 73     if (is<WorkerGlobalScope>(context)) {
 74         WorkerGlobalScope& workerGlobalScope = downcast<WorkerGlobalScope>(context);
 75         WorkerRunLoop& runLoop = workerGlobalScope.thread().runLoop();
 76         return WorkerThreadableWebSocketChannel::create(workerGlobalScope, client, makeString("webSocketChannelMode", runLoop.createUniqueId()), provider);
 77     }
 78 
 79     return create(downcast<Document>(context), client, provider);
 80 }

 82 Optional<ThreadableWebSocketChannel::ValidatedURL> ThreadableWebSocketChannel::validateURL(Document& document, const URL& requestedURL)
 83 {
 84     ValidatedURL validatedURL { requestedURL, true };
 85 #if ENABLE(CONTENT_EXTENSIONS)
 86     if (auto* page = document.page()) {
 87         if (auto* documentLoader = document.loader()) {
 88             auto results = page->userContentProvider().processContentRuleListsForLoad(validatedURL.url, ContentExtensions::ResourceType::Raw, *documentLoader);
 89             if (results.summary.blockedLoad)
 90                 return { };
 91             if (results.summary.madeHTTPS) {
 92                 ASSERT(validatedURL.url.protocolIs("ws"));
 93                 validatedURL.url.setProtocol("wss");
 94             }
 95             validatedURL.areCookiesAllowed = !results.summary.blockedCookies;
 96         }
 97     }
 98 #else
 99     UNUSED_PARAM(document);
100 #endif
101     return validatedURL;
102 }
103 
104 Optional<ResourceRequest> ThreadableWebSocketChannel::webSocketConnectRequest(Document& document, const URL& url)
105 {
106     auto validatedURL = validateURL(document, url);
107     if (!validatedURL)
108         return { };
109 
110     ResourceRequest request { validatedURL->url };
111     request.setHTTPUserAgent(document.userAgent(validatedURL->url));
112     request.setDomainForCachePartition(document.domainForCachePartition());
113     request.setAllowCookies(validatedURL->areCookiesAllowed);
114     request.setFirstPartyForCookies(document.firstPartyForCookies());
115     request.setHTTPHeaderField(HTTPHeaderName::Origin, document.securityOrigin().toString());
116 
117     // Add no-cache headers to avoid compatibility issue.
118     // There are some proxies that rewrite "Connection: upgrade"
119     // to "Connection: close" in the response if a request doesn't contain
120     // these headers.
121     request.addHTTPHeaderField(HTTPHeaderName::Pragma, HTTPHeaderValues::noCache());
122     request.addHTTPHeaderField(HTTPHeaderName::CacheControl, HTTPHeaderValues::noCache());
123 
124     return request;
125 }
