38 ProxyRevoke* ProxyRevoke::create(VM& vm, Structure* structure, ProxyObject* proxy)
39 {
40     ProxyRevoke* revoke = new (NotNull, allocateCell<ProxyRevoke>(vm.heap)) ProxyRevoke(vm, structure);
41     revoke->finishCreation(vm, proxy);
42     return revoke;
43 }
44 
45 static EncodedJSValue JSC_HOST_CALL performProxyRevoke(JSGlobalObject*, CallFrame*);
46 
47 ProxyRevoke::ProxyRevoke(VM& vm, Structure* structure)
48     : Base(vm, structure, performProxyRevoke, nullptr)
49 {
50 }
51 
52 void ProxyRevoke::finishCreation(VM& vm, ProxyObject* proxy)
53 {
54     Base::finishCreation(vm, emptyString());
55     m_proxy.set(vm, this, proxy);
56 
57     putDirect(vm, vm.propertyNames->length, jsNumber(0), PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum);
58 }
59 
60 static EncodedJSValue JSC_HOST_CALL performProxyRevoke(JSGlobalObject* globalObject, CallFrame* callFrame)
61 {
62     ProxyRevoke* proxyRevoke = jsCast<ProxyRevoke*>(callFrame->jsCallee());
63     JSValue proxyValue = proxyRevoke->proxy();
64     if (proxyValue.isNull())
65         return JSValue::encode(jsUndefined());
66 
67     ProxyObject* proxy = jsCast<ProxyObject*>(proxyValue);
68     VM& vm = globalObject->vm();
69     proxy->revoke(vm);
70     proxyRevoke->setProxyToNull(vm);
71     return JSValue::encode(jsUndefined());
72 }
73 
74 void ProxyRevoke::visitChildren(JSCell* cell, SlotVisitor& visitor)
75 {
76     ProxyRevoke* thisObject = jsCast<ProxyRevoke*>(cell);
77     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
78     Base::visitChildren(thisObject, visitor);
79 
80     visitor.append(thisObject->m_proxy);
81 }
