 54 DocumentStorageAccess* DocumentStorageAccess::from(Document& document)
 55 {
 56     auto* supplement = static_cast<DocumentStorageAccess*>(Supplement<Document>::from(&document, supplementName()));
 57     if (!supplement) {
 58         auto newSupplement = makeUnique<DocumentStorageAccess>(document);
 59         supplement = newSupplement.get();
 60         provideTo(&document, supplementName(), WTFMove(newSupplement));
 61     }
 62     return supplement;
 63 }
 64 
 65 const char* DocumentStorageAccess::supplementName()
 66 {
 67     return "DocumentStorageAccess";
 68 }

 80 void DocumentStorageAccess::hasStorageAccess(Ref<DeferredPromise>&& promise)
 81 {
 82     ASSERT(m_document.settings().storageAccessAPIEnabled());
 83 
 84     auto* frame = m_document.frame();
 85     if (frame && hasFrameSpecificStorageAccess()) {
 86         promise->resolve<IDLBoolean>(true);
 87         return;
 88     }
 89 
 90     if (!frame || m_document.securityOrigin().isUnique()) {
 91         promise->resolve<IDLBoolean>(false);
 92         return;
 93     }
 94 
 95     if (frame->isMainFrame()) {
 96         promise->resolve<IDLBoolean>(true);
 97         return;
 98     }
 99 
100     auto& securityOrigin = m_document.securityOrigin();
101     auto& topSecurityOrigin = m_document.topDocument().securityOrigin();
102     if (securityOrigin.equal(&topSecurityOrigin)) {
103         promise->resolve<IDLBoolean>(true);
104         return;
105     }
106 
107     auto* page = frame->page();
108     if (!page) {
109         promise->reject();
110         return;
111     }
112 
113     auto subFrameDomain = RegistrableDomain::uncheckedCreateFromHost(securityOrigin.host());
114     auto topFrameDomain = RegistrableDomain::uncheckedCreateFromHost(topSecurityOrigin.host());
115     page->chrome().client().hasStorageAccess(WTFMove(subFrameDomain), WTFMove(topFrameDomain), *frame, [weakThis = makeWeakPtr(*this), promise = WTFMove(promise)] (bool hasAccess) {
116         if (!weakThis)
117             return;
118 
119         promise->resolve<IDLBoolean>(hasAccess);
120     });
121 }

123 void DocumentStorageAccess::requestStorageAccess(Ref<DeferredPromise>&& promise)
124 {
125     ASSERT(m_document.settings().storageAccessAPIEnabled());
126 
127     auto* frame = m_document.frame();
128     if (frame && hasFrameSpecificStorageAccess()) {
129         promise->resolve();
130         return;
131     }
132 
133     if (!frame || m_document.securityOrigin().isUnique() || !isAllowedToRequestFrameSpecificStorageAccess()) {
134         promise->reject();
135         return;
136     }
137 
138     if (frame->isMainFrame()) {
139         promise->resolve();
140         return;
141     }
142 
143     auto& topDocument = m_document.topDocument();
144     auto& topSecurityOrigin = topDocument.securityOrigin();
145     auto& securityOrigin = m_document.securityOrigin();
146     if (securityOrigin.equal(&topSecurityOrigin)) {
147         promise->resolve();
148         return;
149     }
150 
151     // If there is a sandbox, it has to allow the storage access API to be called.
152     if (m_document.sandboxFlags() != SandboxNone && m_document.isSandboxed(SandboxStorageAccessByUserActivation)) {
153         promise->reject();
154         return;
155     }
156 
157     // The iframe has to be a direct child of the top document.
158     if (&topDocument != m_document.parentDocument()) {
159         promise->reject();
160         return;
161     }
162 
163     if (!UserGestureIndicator::processingUserGesture()) {
164         promise->reject();
165         return;
166     }
167 
168     auto* page = frame->page();
169     if (!page) {
170         promise->reject();
171         return;
172     }
173 
174     auto subFrameDomain = RegistrableDomain::uncheckedCreateFromHost(securityOrigin.host());
175     auto topFrameDomain = RegistrableDomain::uncheckedCreateFromHost(topSecurityOrigin.host());
176 
177     page->chrome().client().requestStorageAccess(WTFMove(subFrameDomain), WTFMove(topFrameDomain), *frame, [this, weakThis = makeWeakPtr(*this), promise = WTFMove(promise)] (StorageAccessWasGranted wasGranted, StorageAccessPromptWasShown promptWasShown) mutable {
178         if (!weakThis)
179             return;
180 
181         // Consume the user gesture only if the user explicitly denied access.
182         bool shouldPreserveUserGesture = wasGranted == StorageAccessWasGranted::Yes || promptWasShown == StorageAccessPromptWasShown::No;
183 
184         if (shouldPreserveUserGesture) {
185             m_document.eventLoop().queueMicrotask([this, weakThis = makeWeakPtr(*this)] {
186                 if (weakThis)
187                     enableTemporaryTimeUserGesture();
188             });
189         }
190 
191         if (wasGranted == StorageAccessWasGranted::Yes)
192             promise->resolve();
193         else {
194             if (promptWasShown == StorageAccessPromptWasShown::Yes)
195                 setWasExplicitlyDeniedFrameSpecificStorageAccess();
196             promise->reject();
197         }
198 
199         if (shouldPreserveUserGesture) {
200             m_document.eventLoop().queueMicrotask([this, weakThis = makeWeakPtr(*this)] {
201                 if (weakThis)
202                     consumeTemporaryTimeUserGesture();
203             });
204         }
205     });
206 }
207 
208 void DocumentStorageAccess::enableTemporaryTimeUserGesture()
209 {
210     m_temporaryUserGesture = makeUnique<UserGestureIndicator>(ProcessingUserGesture, &m_document);
211 }
212 
213 void DocumentStorageAccess::consumeTemporaryTimeUserGesture()
214 {
215     m_temporaryUserGesture = nullptr;
216 }
217 
218 bool DocumentStorageAccess::hasFrameSpecificStorageAccess() const
219 {
220     auto* frame = m_document.frame();
221     return frame && frame->loader().client().hasFrameSpecificStorageAccess();
222 }
