 62 void ScrollingTreeStickyNode::commitStateBeforeChildren(const ScrollingStateNode& stateNode)
 63 {
 64     auto& stickyStateNode = downcast<ScrollingStateStickyNode>(stateNode);
 65 
 66     if (stickyStateNode.hasChangedProperty(ScrollingStateNode::Layer)) {
 67         auto* layer = static_cast<Nicosia::PlatformLayer*>(stickyStateNode.layer());
 68         m_layer = downcast<Nicosia::CompositionLayer>(layer);
 69     }
 70 
 71     if (stickyStateNode.hasChangedProperty(ScrollingStateStickyNode::ViewportConstraints))
 72         m_constraints = stickyStateNode.viewportConstraints();
 73 }

 75 void ScrollingTreeStickyNode::applyLayerPositions()
 76 {
 77     auto layerPosition = computeLayerPosition();
 78 
 79     LOG_WITH_STREAM(Scrolling, stream << "ScrollingTreeStickyNode " << scrollingNodeID() << " constrainingRectAtLastLayout " << m_constraints.constrainingRectAtLastLayout() << " last layer pos " << m_constraints.layerPositionAtLastLayout() << " layerPosition " << layerPosition);
 80 
 81     layerPosition -= m_constraints.alignmentOffset();
 82 
 83     ASSERT(m_layer);
 84     m_layer->updateState(
 85         [&layerPosition](Nicosia::CompositionLayer::LayerState& state)
 86         {
 87             state.position = layerPosition;
 88             state.delta.positionChanged = true;
 89         });
 90 }

 92 void ScrollingTreeStickyNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
 93 {
 94     ts << "sticky node";
 95 
 96     ScrollingTreeNode::dumpProperties(ts, behavior);
 97     ts.dumpProperty("sticky constraints", m_constraints);
 98 
 99     if (behavior & ScrollingStateTreeAsTextBehaviorIncludeLayerPositions) {
100         FloatPoint layerTopLeft;
101         ASSERT(m_layer);
102         m_layer->accessCommitted(
103             [this, &layerTopLeft](Nicosia::CompositionLayer::LayerState& state)
104             {
105                 layerTopLeft = state.position - toFloatSize(state.anchorPoint.xy()) * state.size + m_constraints.alignmentOffset();
106             });
107 
108         ts.dumpProperty("layer top left", layerTopLeft);
109     }
110 }

112 FloatPoint ScrollingTreeStickyNode::computeLayerPosition() const
113 {
114     auto computeLayerPositionForScrollingNode = [&](ScrollingTreeNode& scrollingNode) {
115         FloatRect constrainingRect;
116         if (is<ScrollingTreeFrameScrollingNode>(scrollingNode)) {
117             auto& frameScrollingNode = downcast<ScrollingTreeFrameScrollingNode>(scrollingNode);
118             constrainingRect = frameScrollingNode.layoutViewport();
119         } else {
120             auto& overflowScrollingNode = downcast<ScrollingTreeOverflowScrollingNode>(scrollingNode);
121             constrainingRect = FloatRect(overflowScrollingNode.currentScrollPosition(), m_constraints.constrainingRectAtLastLayout().size());
122         }
123         return m_constraints.layerPositionForConstrainingRect(constrainingRect);
124     };
125 
126     for (auto* ancestor = parent(); ancestor; ancestor = ancestor->parent()) {
127         if (is<ScrollingTreeOverflowScrollProxyNode>(*ancestor)) {
128             auto& overflowProxyNode = downcast<ScrollingTreeOverflowScrollProxyNode>(*ancestor);
129             auto overflowNode = scrollingTree().nodeForID(overflowProxyNode.overflowScrollingNodeID());
130             if (!overflowNode)
131                 break;
132 
133             return computeLayerPositionForScrollingNode(*overflowNode);
134         }
135 
136         if (is<ScrollingTreeScrollingNode>(*ancestor))
137             return computeLayerPositionForScrollingNode(*ancestor);
138 
139         if (is<ScrollingTreeFixedNode>(*ancestor) || is<ScrollingTreeStickyNode>(*ancestor)) {
140             // FIXME: Do we need scrolling tree nodes at all for nested cases?
141             return m_constraints.layerPositionAtLastLayout();
142         }
143     }
144     ASSERT_NOT_REACHED();
145     return m_constraints.layerPositionAtLastLayout();
146 }
