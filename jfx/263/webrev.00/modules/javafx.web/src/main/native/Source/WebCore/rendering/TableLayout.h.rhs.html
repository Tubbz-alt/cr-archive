21 #pragma once
22 
23 #include "LayoutUnit.h"
24 #include <wtf/FastMalloc.h>
25 #include <wtf/Noncopyable.h>
26 
27 namespace WebCore {
28 
29 class RenderTable;
30 
31 DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(TableLayout);
32 class TableLayout {
33     WTF_MAKE_NONCOPYABLE(TableLayout);
34     WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(TableLayout);
35 public:
36     explicit TableLayout(RenderTable* table)
37         : m_table(table)
38     {
39     }
40 
41     virtual ~TableLayout() = default;
42 
43     virtual void computeIntrinsicLogicalWidths(LayoutUnit& minWidth, LayoutUnit& maxWidth) = 0;
44     virtual LayoutUnit scaledWidthFromPercentColumns() const { return 0_lu; }
45     virtual void applyPreferredLogicalWidthQuirks(LayoutUnit& minWidth, LayoutUnit& maxWidth) const = 0;
46     virtual void layout() = 0;
47 
48 protected:
49     // FIXME: Once we enable SATURATED_LAYOUT_ARITHMETHIC, this should just be LayoutUnit::nearlyMax().
50     // Until then though, using nearlyMax causes overflow in some tests, so we just pick a large number.
51     static constexpr int tableMaxWidth = 1000000;
52 
53     RenderTable* m_table;
54 };
55 
56 } // namespace WebCore
