21 #pragma once
22 
23 #include "InternalFunction.h"
24 
25 namespace JSC {
26 
27 class StringPrototype;
28 class GetterSetter;
29 
30 class StringConstructor final : public InternalFunction {
31 public:
32     typedef InternalFunction Base;
33     static const unsigned StructureFlags = Base::StructureFlags | HasStaticPropertyTable;
34 
35     static StringConstructor* create(VM& vm, Structure* structure, StringPrototype* stringPrototype, GetterSetter*)
36     {
37         StringConstructor* constructor = new (NotNull, allocateCell<StringConstructor>(vm.heap)) StringConstructor(vm, structure);
38         constructor->finishCreation(vm, stringPrototype);
39         return constructor;
40     }
41 
42     DECLARE_INFO;
43 
44     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
45     {
46         return Structure::create(vm, globalObject, prototype, TypeInfo(InternalFunctionType, StructureFlags), info());
47     }
48 
49 private:
50     StringConstructor(VM&, Structure*);
51     void finishCreation(VM&, StringPrototype*);
52 };
53 
54 JSString* JSC_HOST_CALL stringFromCharCode(ExecState*, int32_t);
55 JSString* stringConstructor(ExecState*, JSValue);
