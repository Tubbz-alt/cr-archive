27 
28 #if ENABLE(WEBGPU)
29 
30 #include "GPUPipeline.h"
31 #include "GPUProgrammableStageDescriptor.h"
32 #include "GPURenderPipelineDescriptor.h"
33 #include <wtf/Optional.h>
34 #include <wtf/RefCounted.h>
35 #include <wtf/RefPtr.h>
36 #include <wtf/RetainPtr.h>
37 
38 #if USE(METAL)
39 OBJC_PROTOCOL(MTLDepthStencilState);
40 OBJC_PROTOCOL(MTLRenderPipelineState);
41 #endif // USE(METAL)
42 
43 namespace WebCore {
44 
45 class GPUDevice;
46 class GPUErrorScopes;
47 
48 using PlatformRenderPipeline = MTLRenderPipelineState;
49 using PlatformRenderPipelineSmartPtr = RetainPtr<MTLRenderPipelineState>;
50 
51 class GPURenderPipeline final : public GPUPipeline {
52 public:
53     virtual ~GPURenderPipeline();
54 
55     static RefPtr<GPURenderPipeline> tryCreate(const GPUDevice&, const GPURenderPipelineDescriptor&, GPUErrorScopes&);
56 
57     bool isRenderPipeline() const { return true; }
58 
59     bool recompile(const GPUDevice&, GPUProgrammableStageDescriptor&& vertexStage, Optional<GPUProgrammableStageDescriptor>&& fragmentStage);
60 
61 #if USE(METAL)
62     MTLDepthStencilState *depthStencilState() const { return m_depthStencilState.get(); }
63 #endif
64     PlatformRenderPipeline* platformRenderPipeline() const { return m_platformRenderPipeline.get(); }
65     GPUPrimitiveTopology primitiveTopology() const { return m_primitiveTopology; }
66     Optional<GPUIndexFormat> indexFormat() const { return m_indexFormat; }
67 
68 private:
69 #if USE(METAL)
70     GPURenderPipeline(RetainPtr<MTLDepthStencilState>&&, PlatformRenderPipelineSmartPtr&&, GPUPrimitiveTopology, Optional<GPUIndexFormat>, const RefPtr<GPUPipelineLayout>&, const GPURenderPipelineDescriptorBase&);
71 
72     RetainPtr<MTLDepthStencilState> m_depthStencilState;
73 #endif // USE(METAL)
74     PlatformRenderPipelineSmartPtr m_platformRenderPipeline;
75     GPUPrimitiveTopology m_primitiveTopology;
76     Optional<GPUIndexFormat> m_indexFormat;
77 
78     // Preserved for Web Inspector recompilation.
79     RefPtr<GPUPipelineLayout> m_layout;
80     GPURenderPipelineDescriptorBase m_renderDescriptorBase;
81 };
82 
83 } // namespace WebCore
84 
85 SPECIALIZE_TYPE_TRAITS_GPUPIPELINE(WebCore::GPURenderPipeline, isRenderPipeline())

87 #endif // ENABLE(WEBGPU)
