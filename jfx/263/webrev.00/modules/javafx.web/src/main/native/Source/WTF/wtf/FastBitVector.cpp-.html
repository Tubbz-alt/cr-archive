31 void FastBitVectorWordOwner::setEqualsSlow(const FastBitVectorWordOwner& other)
32 {
33     uint32_t* newArray = static_cast<uint32_t*>(
34         fastCalloc(other.arrayLength(), sizeof(uint32_t)));
35     memcpy(newArray, other.m_words, other.arrayLength() * sizeof(uint32_t));
36     if (m_words)
37         fastFree(m_words);
38     m_words = newArray;
39     m_numBits = other.m_numBits;
40 }
41 
42 void FastBitVectorWordOwner::resizeSlow(size_t numBits)
43 {
44     size_t newLength = fastBitVectorArrayLength(numBits);
45 
46     RELEASE_ASSERT(newLength >= arrayLength());
47 
48     // Use fastCalloc instead of fastRealloc because we expect the common
49     // use case for this method to be initializing the size of the bitvector.
50 
51     uint32_t* newArray = static_cast<uint32_t*>(fastCalloc(newLength, sizeof(uint32_t)));
52     memcpy(newArray, m_words, arrayLength() * sizeof(uint32_t));
53     if (m_words)
54         fastFree(m_words);
55     m_words = newArray;
56 }
57 
58 void FastBitVector::clearRange(size_t begin, size_t end)
59 {
60     if (end - begin < 32) {
61         for (size_t i = begin; i < end; ++i)
62             at(i) = false;
63         return;
64     }
65 
66     size_t endBeginSlop = (begin + 31) & ~31;
67     size_t beginEndSlop = end & ~31;
68 
69     for (size_t i = begin; i < endBeginSlop; ++i)
70         at(i) = false;
71     for (size_t i = beginEndSlop; i < end; ++i)
72         at(i) = false;
73     for (size_t i = endBeginSlop / 32; i < beginEndSlop / 32; ++i)
74         m_words.word(i) = 0;
75 }
