26 #pragma once
27 
28 #include "BAssert.h"
29 #include "DeferredTrigger.h"
30 
31 namespace bmalloc {
32 
33 template<IsoPageTrigger trigger>
34 template<typename Config>
35 void DeferredTrigger<trigger>::didBecome(const LockHolder& locker, IsoPage<Config>& page)
36 {
37     if (page.isInUseForAllocation())
38         m_hasBeenDeferred = true;
39     else
40         page.directory().didBecome(locker, &page, trigger);
41 }
42 
43 template<IsoPageTrigger trigger>
44 template<typename Config>
45 void DeferredTrigger<trigger>::handleDeferral(const LockHolder& locker, IsoPage<Config>& page)
46 {
47     RELEASE_BASSERT(!page.isInUseForAllocation());
48 
49     if (m_hasBeenDeferred) {
50         page.directory().didBecome(locker, &page, trigger);
51         m_hasBeenDeferred = false;
52     }
53 }

55 } // namespace bmalloc
