25 #pragma once
26 
27 #include "VRPlatformDisplay.h"
28 
29 #include <wtf/Ref.h>
30 #include <wtf/RefCounted.h>
31 
32 namespace WebCore {
33 
34 class VRFieldOfView : public RefCounted<VRFieldOfView> {
35 public:
36     static Ref<VRFieldOfView> create(VRPlatformDisplayInfo::FieldOfView fieldOfView)
37     {
38         return adoptRef(*new VRFieldOfView(fieldOfView));
39     }
40 
41     double upDegrees() const { return m_upDegrees; };
42     double rightDegrees() const { return m_rightDegrees; };
43     double downDegrees() const { return m_downDegrees; };
44     double leftDegrees() const { return m_leftDegrees; };
45 
46 private:
47     VRFieldOfView(VRPlatformDisplayInfo::FieldOfView fieldOfView)
48         : m_upDegrees(fieldOfView.upDegrees)
49         , m_rightDegrees(fieldOfView.rightDegrees)
50         , m_downDegrees(fieldOfView.downDegrees)
51         , m_leftDegrees(fieldOfView.leftDegrees)
52     {
53     }
54 
55     double m_upDegrees;
56     double m_rightDegrees;
57     double m_downDegrees;
58     double m_leftDegrees;
59 };
60 
61 } // namespace WebCore
