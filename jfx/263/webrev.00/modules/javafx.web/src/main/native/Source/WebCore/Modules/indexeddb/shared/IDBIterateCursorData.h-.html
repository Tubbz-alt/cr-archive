26 #pragma once
27 
28 #if ENABLE(INDEXED_DATABASE)
29 
30 #include "IDBKeyData.h"
31 
32 namespace WebCore {
33 
34 struct IDBIterateCursorData {
35     IDBKeyData keyData;
36     IDBKeyData primaryKeyData;
37     unsigned count;
38 
39     IDBIterateCursorData isolatedCopy() const;
40 
41 #if !LOG_DISABLED
42     String loggingString() const;
43 #endif
44 
45     template<class Encoder> void encode(Encoder&) const;
46     template<class Decoder> static bool decode(Decoder&, IDBIterateCursorData&);
47 };
48 
49 template<class Encoder>
50 void IDBIterateCursorData::encode(Encoder& encoder) const
51 {
52     encoder << keyData << primaryKeyData << static_cast<uint64_t>(count);
53 }
54 
55 template<class Decoder>
56 bool IDBIterateCursorData::decode(Decoder& decoder, IDBIterateCursorData& iteratorCursorData)
57 {
58     Optional<IDBKeyData> keyData;
59     decoder >> keyData;
60     if (!keyData)
61         return false;
62     iteratorCursorData.keyData = WTFMove(*keyData);
63 
64     Optional<IDBKeyData> primaryKeyData;
65     decoder >> primaryKeyData;
66     if (!primaryKeyData)
67         return false;
68     iteratorCursorData.primaryKeyData = WTFMove(*primaryKeyData);
69 
70     uint64_t count;
71     if (!decoder.decode(count))
72         return false;
73 
74     if (count > std::numeric_limits<unsigned>::max())
75         return false;
76 
77     iteratorCursorData.count = static_cast<unsigned>(count);
78 
79     return true;
80 }
81 
82 } // namespace WebCore
83 
84 #endif // ENABLE(INDEXED_DATABASE)
