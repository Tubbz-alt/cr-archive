 33 Ref<StorageMap> StorageMap::create(unsigned quota)
 34 {
 35     return adoptRef(*new StorageMap(quota));
 36 }
 37 
 38 StorageMap::StorageMap(unsigned quota)
 39     : m_iterator(m_map.end())
 40     , m_quotaSize(quota)  // quota measured in bytes
 41 {
 42 }
 43 
 44 Ref<StorageMap> StorageMap::copy()
 45 {
 46     Ref<StorageMap> newMap = create(m_quotaSize);
 47     newMap->m_map = m_map;
 48     newMap->m_currentLength = m_currentLength;
 49     return newMap;
 50 }
 51 
 52 void StorageMap::invalidateIterator()
 53 {
 54     m_iterator = m_map.end();
 55     m_iteratorIndex = UINT_MAX;
 56 }
 57 
 58 void StorageMap::setIteratorToIndex(unsigned index)
 59 {
 60     // FIXME: Once we have bidirectional iterators for HashMap we can be more intelligent about this.
 61     // The requested index will be closest to begin(), our current iterator, or end(), and we
 62     // can take the shortest route.
 63     // Until that mechanism is available, we'll always increment our iterator from begin() or current.
 64 
 65     if (m_iteratorIndex == index)
 66         return;
 67 
 68     if (index < m_iteratorIndex) {
 69         m_iteratorIndex = 0;
 70         m_iterator = m_map.begin();
 71         ASSERT(m_iterator != m_map.end());
 72     }
 73 
 74     while (m_iteratorIndex < index) {
 75         ++m_iteratorIndex;
 76         ++m_iterator;
 77         ASSERT(m_iterator != m_map.end());
 78     }
 79 }

 81 unsigned StorageMap::length() const
 82 {
 83     return m_map.size();
 84 }
 85 
 86 String StorageMap::key(unsigned index)
 87 {
 88     if (index >= length())
 89         return String();
 90 
 91     setIteratorToIndex(index);
 92     return m_iterator->key;
 93 }

100 RefPtr<StorageMap> StorageMap::setItem(const String& key, const String& value, String& oldValue, bool& quotaException)
101 {
102     ASSERT(!value.isNull());
103     quotaException = false;
104 
105     // Implement copy-on-write semantics here.  We're guaranteed that the only refs of StorageMaps belong to Storage objects
106     // so if more than one Storage object refs this map, copy it before mutating it.
107     if (refCount() > 1) {
108         RefPtr<StorageMap> newStorageMap = copy();
109         newStorageMap->setItem(key, value, oldValue, quotaException);
110         return newStorageMap;
111     }
112 
113     // Quota tracking.  This is done in a couple of steps to keep the overflow tracking simple.
114     unsigned newLength = m_currentLength;
115     bool overflow = newLength + value.length() < newLength;
116     newLength += value.length();
117 
118     oldValue = m_map.get(key);
119     overflow |= newLength - oldValue.length() > newLength;
120     newLength -= oldValue.length();
121 
122     unsigned adjustedKeyLength = oldValue.isNull() ? key.length() : 0;
123     overflow |= newLength + adjustedKeyLength < newLength;
124     newLength += adjustedKeyLength;
125 
126     ASSERT(!overflow);  // Overflow is bad...even if quotas are off.
127     bool overQuota = newLength > m_quotaSize / sizeof(UChar);
128     if (m_quotaSize != noQuota && (overflow || overQuota)) {
129         quotaException = true;
130         return nullptr;
131     }
132     m_currentLength = newLength;
133 
134     HashMap<String, String>::AddResult addResult = m_map.add(key, value);
135     if (!addResult.isNewEntry)
136         addResult.iterator->value = value;
137 
138     invalidateIterator();
139 
140     return nullptr;
141 }

143 RefPtr<StorageMap> StorageMap::setItemIgnoringQuota(const String& key, const String& value)
144 {
145     SetForScope<unsigned> quotaSizeChange(m_quotaSize, static_cast<unsigned>(noQuota));
146 
147     String oldValue;
148     bool quotaException;
149 
150     RefPtr<StorageMap> map = setItem(key, value, oldValue, quotaException);
151     ASSERT(!quotaException);
152 
153     return map;
154 }

156 RefPtr<StorageMap> StorageMap::removeItem(const String& key, String& oldValue)
157 {
158     // Implement copy-on-write semantics here.  We're guaranteed that the only refs of StorageMaps belong to Storage objects
159     // so if more than one Storage object refs this map, copy it before mutating it.
160     if (refCount() > 1) {
161         RefPtr<StorageMap> newStorage = copy();
162         newStorage->removeItem(key, oldValue);
163         return newStorage;
164     }
165 
166     oldValue = m_map.take(key);
167     if (!oldValue.isNull()) {
168         invalidateIterator();
169         ASSERT(m_currentLength - key.length() <= m_currentLength);
170         m_currentLength -= key.length();
171     }
172     ASSERT(m_currentLength - oldValue.length() <= m_currentLength);
173     m_currentLength -= oldValue.length();
174 
175     return nullptr;
176 }

183 void StorageMap::importItems(HashMap<String, String>&& items)
184 {
185     if (m_map.isEmpty()) {
186         // Fast path.
187         m_map = WTFMove(items);
188         for (auto& pair : m_map) {
189             ASSERT(m_currentLength + pair.key.length() + pair.value.length() >= m_currentLength);
190             m_currentLength += (pair.key.length() + pair.value.length());
191         }
192         return;
193     }
194 
195     for (auto& item : items) {
196         auto& key = item.key;
197         auto& value = item.value;
198 
199         ASSERT(m_currentLength + key.length() + value.length() >= m_currentLength);
200         m_currentLength += (key.length() + value.length());
201 
202         auto result = m_map.add(WTFMove(key), WTFMove(value));
203         ASSERT_UNUSED(result, result.isNewEntry); // True if the key didn't exist previously.
204     }
205 }
