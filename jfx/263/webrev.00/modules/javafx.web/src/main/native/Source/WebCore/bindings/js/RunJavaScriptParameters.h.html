26 #pragma once
27 
28 #include <wtf/HashMap.h>
29 #include <wtf/Vector.h>
30 #include <wtf/text/WTFString.h>
31 
32 namespace WebCore {
33 
34 enum class RunAsAsyncFunction : bool { No, Yes };
35 enum class ForceUserGesture : bool { No, Yes };
36 
37 using ArgumentWireBytesMap = HashMap<String, Vector<uint8_t>>;
38 
39 struct RunJavaScriptParameters {
40     RunJavaScriptParameters(String&& source, RunAsAsyncFunction runAsAsyncFunction, Optional<ArgumentWireBytesMap>&& arguments, ForceUserGesture forceUserGesture)
41         : source(WTFMove(source))
42         , runAsAsyncFunction(runAsAsyncFunction)
43         , arguments(WTFMove(arguments))
44         , forceUserGesture(forceUserGesture)
45     {
46     }
47 
48     RunJavaScriptParameters(const String& source, bool runAsAsyncFunction, Optional<ArgumentWireBytesMap>&& arguments, bool forceUserGesture)
49         : source(source)
50         , runAsAsyncFunction(runAsAsyncFunction ? RunAsAsyncFunction::Yes : RunAsAsyncFunction::No)
51         , arguments(WTFMove(arguments))
52         , forceUserGesture(forceUserGesture ? ForceUserGesture::Yes : ForceUserGesture::No)
53     {
54     }
55 
56     RunJavaScriptParameters(String&& source, bool runAsAsyncFunction, Optional<ArgumentWireBytesMap>&& arguments, bool forceUserGesture)
57         : source(WTFMove(source))
58         , runAsAsyncFunction(runAsAsyncFunction ? RunAsAsyncFunction::Yes : RunAsAsyncFunction::No)
59         , arguments(WTFMove(arguments))
60         , forceUserGesture(forceUserGesture ? ForceUserGesture::Yes : ForceUserGesture::No)
61     {
62     }
63 
64     String source;
65     RunAsAsyncFunction runAsAsyncFunction;
66     Optional<ArgumentWireBytesMap> arguments;
67     ForceUserGesture forceUserGesture;
68 
69     template<typename Encoder> void encode(Encoder& encoder) const
70     {
71         encoder << source << runAsAsyncFunction << arguments << forceUserGesture;
72     }
73 
74     template<typename Decoder> static Optional<RunJavaScriptParameters> decode(Decoder& decoder)
75     {
76         String source;
77         if (!decoder.decode(source))
78             return WTF::nullopt;
79 
80         RunAsAsyncFunction runAsAsyncFunction;
81         if (!decoder.decode(runAsAsyncFunction))
82             return WTF::nullopt;
83 
84         Optional<ArgumentWireBytesMap> arguments;
85         if (!decoder.decode(arguments))
86             return WTF::nullopt;
87 
88         ForceUserGesture forceUserGesture;
89         if (!decoder.decode(forceUserGesture))
90             return WTF::nullopt;
91 
92         return { RunJavaScriptParameters { WTFMove(source), runAsAsyncFunction, WTFMove(arguments), forceUserGesture } };
93     }
94 };
95 
96 } // namespace WebCore
