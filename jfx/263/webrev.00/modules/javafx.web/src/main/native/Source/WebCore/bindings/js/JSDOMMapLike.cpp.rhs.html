 35 std::pair<bool, std::reference_wrapper<JSC::JSObject>> getBackingMap(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& mapLike)
 36 {
 37     auto& vm = lexicalGlobalObject.vm();
 38     auto backingMap = mapLike.get(&lexicalGlobalObject, static_cast<JSVMClientData*>(vm.clientData)->builtinNames().backingMapPrivateName());
 39     if (!backingMap.isUndefined())
 40         return { false, *JSC::asObject(backingMap) };
 41 
 42     auto scope = DECLARE_CATCH_SCOPE(vm);
 43 
 44     backingMap = JSC::JSMap::create(&lexicalGlobalObject, vm, lexicalGlobalObject.mapStructure());
 45     scope.releaseAssertNoException();
 46 
 47     mapLike.putDirect(vm, static_cast<JSVMClientData*>(vm.clientData)->builtinNames().backingMapPrivateName(), backingMap, static_cast<unsigned>(JSC::PropertyAttribute::DontEnum));
 48     return { true, *JSC::asObject(backingMap) };
 49 }

 51 void clearBackingMap(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& backingMap)
 52 {
 53     auto& vm = JSC::getVM(&lexicalGlobalObject);
 54     auto function = backingMap.get(&lexicalGlobalObject, vm.propertyNames->clear);
 55 
 56     JSC::CallData callData;
 57     auto callType = JSC::getCallData(vm, function, callData);
 58     if (callType == JSC::CallType::None)
 59         return;
 60 
 61     JSC::MarkedArgumentBuffer arguments;
 62     JSC::call(&lexicalGlobalObject, function, callType, callData, &backingMap, arguments);
 63 }

 65 void setToBackingMap(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& backingMap, JSC::JSValue key, JSC::JSValue value)
 66 {
 67     auto& vm = JSC::getVM(&lexicalGlobalObject);
 68     auto function = backingMap.get(&lexicalGlobalObject, vm.propertyNames->set);
 69 
 70     JSC::CallData callData;
 71     auto callType = JSC::getCallData(vm, function, callData);
 72     if (callType == JSC::CallType::None)
 73         return;
 74 
 75     JSC::MarkedArgumentBuffer arguments;
 76     arguments.append(key);
 77     arguments.append(value);
 78     JSC::call(&lexicalGlobalObject, function, callType, callData, &backingMap, arguments);
 79 }

 81 JSC::JSValue forwardAttributeGetterToBackingMap(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& backingMap, const JSC::Identifier& attributeName)
 82 {
 83     return backingMap.get(&lexicalGlobalObject, attributeName);
 84 }

 86 JSC::JSValue forwardFunctionCallToBackingMap(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, JSC::JSObject& backingMap, const JSC::Identifier& functionName)
 87 {
 88     auto function = backingMap.get(&lexicalGlobalObject, functionName);



 89 
 90     JSC::CallData callData;
 91     auto callType = JSC::getCallData(lexicalGlobalObject.vm(), function, callData);
 92     if (callType == JSC::CallType::None)
 93         return JSC::jsUndefined();
 94 
 95     JSC::MarkedArgumentBuffer arguments;
 96     for (size_t cptr = 0; cptr < callFrame.argumentCount(); ++cptr)
 97         arguments.append(callFrame.uncheckedArgument(cptr));
 98     ASSERT(!arguments.hasOverflowed());
 99     return JSC::call(&lexicalGlobalObject, function, callType, callData, &backingMap, arguments);
100 }

102 JSC::JSValue forwardForEachCallToBackingMap(JSDOMGlobalObject& globalObject, JSC::CallFrame& callFrame, JSC::JSObject& mapLike)
103 {
104     auto result = getBackingMap(globalObject, mapLike);
105     ASSERT(!result.first);
106 
107     auto* function = globalObject.builtinInternalFunctions().jsDOMBindingInternals().m_forEachWrapperFunction.get();
108     ASSERT(function);
109 
110     JSC::CallData callData;
111     auto callType = JSC::getCallData(globalObject.vm(), function, callData);
112     ASSERT(callType != JSC::CallType::None);
113 
114     JSC::MarkedArgumentBuffer arguments;
115     arguments.append(&result.second.get());
116     for (size_t cptr = 0; cptr < callFrame.argumentCount(); ++cptr)
117         arguments.append(callFrame.uncheckedArgument(cptr));
118     ASSERT(!arguments.hasOverflowed());
119     return JSC::call(&globalObject, function, callType, callData, &mapLike, arguments);
120 }
