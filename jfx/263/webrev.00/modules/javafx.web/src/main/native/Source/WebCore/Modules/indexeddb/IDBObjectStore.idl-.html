26 // NOTE: This type is specified as 'any' in the IndexedDB specification, but is always
27 // constrained to this union.
28 typedef (DOMString or sequence<DOMString>) IDBKeyPath;
29 
30 [
31     Conditional=INDEXED_DATABASE,
32     GenerateIsReachable=Impl,
33     JSCustomMarkFunction,
34     SkipVTableValidation,
35 ] interface IDBObjectStore {
36     attribute DOMString name;
37     readonly attribute IDBKeyPath? keyPath;
38     readonly attribute DOMStringList indexNames;
39     readonly attribute IDBTransaction transaction;
40     readonly attribute boolean autoIncrement;
41 
42     [CallWith=ExecState, MayThrowException] IDBRequest put(any value, optional any key);
43     [CallWith=ExecState, MayThrowException] IDBRequest add(any value, optional any key);
44     [CallWith=ExecState, ImplementedAs=deleteFunction, MayThrowException] IDBRequest delete(IDBKeyRange? keyRange);
45     [CallWith=ExecState, ImplementedAs=deleteFunction, MayThrowException] IDBRequest delete(any key);
46     [CallWith=ExecState, MayThrowException] IDBRequest get(IDBKeyRange? key);
47     [CallWith=ExecState, MayThrowException] IDBRequest get(any key);
48     [CallWith=ExecState, MayThrowException] IDBRequest getKey(IDBKeyRange? key);
49     [CallWith=ExecState, MayThrowException] IDBRequest getKey(any key);
50     [CallWith=ExecState, MayThrowException] IDBRequest clear();
51     [CallWith=ExecState, MayThrowException] IDBRequest openCursor(optional IDBKeyRange? range = null, optional IDBCursorDirection direction = "next");
52     [CallWith=ExecState, MayThrowException] IDBRequest openCursor(any key, optional IDBCursorDirection direction = "next");
53     [CallWith=ExecState, MayThrowException] IDBRequest openKeyCursor(optional IDBKeyRange? range = null, optional IDBCursorDirection direction = "next");
54     [CallWith=ExecState, MayThrowException] IDBRequest openKeyCursor(any key, optional IDBCursorDirection direction = "next");
55     [CallWith=ExecState, MayThrowException] IDBRequest getAll(optional IDBKeyRange? range = null, optional [EnforceRange] unsigned long count);
56     [CallWith=ExecState, MayThrowException] IDBRequest getAll(any key, optional [EnforceRange] unsigned long count);
57     [CallWith=ExecState, MayThrowException] IDBRequest getAllKeys(optional IDBKeyRange? range = null, optional [EnforceRange] unsigned long count);
58     [CallWith=ExecState, MayThrowException] IDBRequest getAllKeys(any key, optional [EnforceRange] unsigned long count);
59     [CallWith=ExecState, MayThrowException] IDBIndex createIndex(DOMString name, (DOMString or sequence<DOMString>) keyPath, optional IDBIndexParameters options);
60     [MayThrowException] IDBIndex index(DOMString name);
61     [MayThrowException] void deleteIndex(DOMString name);
62     [CallWith=ExecState, MayThrowException] IDBRequest count(optional IDBKeyRange? range = null);
63     [CallWith=ExecState, MayThrowException] IDBRequest count(any key);
64 };
65 
66 dictionary IDBIndexParameters {
67     boolean unique = false;
68     boolean multiEntry = false;
69 };
