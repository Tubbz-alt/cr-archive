26 #pragma once
27 
28 #if ENABLE(JAVA_BRIDGE)
29 
30 #include "BridgeJSC.h"
31 #include "JNIUtility.h"
32 #include "JobjectWrapper.h"
33 #include "runtime_root.h"
34 
35 namespace JSC {
36 
37 namespace Bindings {
38 
39 class JavaClass;
40 
41 class JavaInstance : public Instance {
42 public:
43     static RefPtr<JavaInstance> create(jobject instance, RefPtr<RootObject>&& rootObject, jobject accessControlContext)
44     {
45         return adoptRef(new JavaInstance(instance, WTFMove(rootObject), accessControlContext));
46     }
47 
48     ~JavaInstance();
49 
50     virtual Class* getClass() const;
51 
52     virtual JSValue valueOf(ExecState*) const;
53     virtual JSValue defaultValue(ExecState*, PreferredPrimitiveType) const;
54 
55     virtual JSValue getMethod(ExecState*, PropertyName);
56     virtual JSValue invokeMethod(ExecState*, RuntimeMethod*);
57 
58     jobject javaInstance() const { return m_instance->instance(); }
59 
60     JSValue stringValue(ExecState*) const;
61     JSValue numberValue(ExecState*) const;
62     JSValue booleanValue() const;
63     jobject accessControlContext() const { return m_accessControlContext->instance(); }
64 
65 protected:
66     JavaInstance(jobject instance, RefPtr<RootObject>&&, jobject accessControlContext);
67 
68     virtual RuntimeObject* newRuntimeObject(ExecState*);
69 
70     virtual void virtualBegin();
71     virtual void virtualEnd();
72 
73     RefPtr<JobjectWrapper> m_instance;
74     mutable JavaClass* m_class;
75     RefPtr<JobjectWrapper> m_accessControlContext;
76 };
77 
78 } // namespace Bindings
79 
80 } // namespace JSC
81 
82 #endif // ENABLE(JAVA_BRIDGE)
