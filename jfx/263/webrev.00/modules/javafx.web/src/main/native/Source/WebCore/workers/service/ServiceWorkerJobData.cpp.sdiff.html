66 // https://w3c.github.io/ServiceWorker/#dfn-job-equivalent
67 bool ServiceWorkerJobData::isEquivalent(const ServiceWorkerJobData& job) const
68 {
69     if (type != job.type)
70         return false;
71 
72     switch (type) {
73     case ServiceWorkerJobType::Register:
74     case ServiceWorkerJobType::Update:
75         return scopeURL == job.scopeURL
76             && scriptURL == job.scriptURL
77             && registrationOptions.type == job.registrationOptions.type
78             && registrationOptions.updateViaCache == job.registrationOptions.updateViaCache;
79     case ServiceWorkerJobType::Unregister:
80         return scopeURL == job.scopeURL;
81     }
82     return false;
83 }
