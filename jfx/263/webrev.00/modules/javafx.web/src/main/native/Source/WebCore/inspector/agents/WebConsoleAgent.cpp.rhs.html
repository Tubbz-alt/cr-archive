52 
53 void WebConsoleAgent::frameWindowDiscarded(DOMWindow* window)
54 {



55     for (auto& message : m_consoleMessages) {
56         JSC::JSGlobalObject* lexicalGlobalObject = message->globalObject();
57         if (!lexicalGlobalObject)
58             continue;
59         if (domWindowFromExecState(lexicalGlobalObject) != window)
60             continue;
61         message->clear();
62     }
63 
64     static_cast<WebInjectedScriptManager&>(m_injectedScriptManager).discardInjectedScriptsFor(window);
65 }
66 
67 void WebConsoleAgent::didReceiveResponse(unsigned long requestIdentifier, const ResourceResponse& response)
68 {



69     if (response.httpStatusCode() >= 400) {
70         String message = makeString("Failed to load resource: the server responded with a status of ", response.httpStatusCode(), " (", response.httpStatusText(), ')');
71         addMessageToConsole(makeUnique<ConsoleMessage>(MessageSource::Network, MessageType::Log, MessageLevel::Error, message, response.url().string(), 0, 0, nullptr, requestIdentifier));
72     }
73 }
74 
75 void WebConsoleAgent::didFailLoading(unsigned long requestIdentifier, const ResourceError& error)
76 {



77     // Report failures only.
78     if (error.isCancellation())
79         return;
80 
81     StringBuilder message;
82     message.appendLiteral("Failed to load resource");
83     if (!error.localizedDescription().isEmpty()) {
84         message.appendLiteral(": ");
85         message.append(error.localizedDescription());
86     }
87 
88     addMessageToConsole(makeUnique<ConsoleMessage>(MessageSource::Network, MessageType::Log, MessageLevel::Error, message.toString(), error.failingURL(), 0, 0, nullptr, requestIdentifier));
89 }
