34 namespace WebCore {
35 namespace Layout {
36 
37 class InvalidationState;
38 class TableFormattingState;
39 // This class implements the layout logic for table formatting contexts.
40 // https://www.w3.org/TR/CSS22/tables.html
41 class TableFormattingContext final : public FormattingContext {
42     WTF_MAKE_ISO_ALLOCATED(TableFormattingContext);
43 public:
44     TableFormattingContext(const Container& formattingContextRoot, TableFormattingState&);
45     void layoutInFlowContent(InvalidationState&, const HorizontalConstraints&, const VerticalConstraints&) override;
46 
47 private:
48     class Geometry : public FormattingContext::Geometry {
49     public:
50         ContentHeightAndMargin tableCellHeightAndMargin(const Box&) const;
51         Optional<LayoutUnit> computedColumnWidth(const Box& columnBox) const;
52 
53     private:
54         friend class TableFormattingContext;
55         Geometry(const TableFormattingContext&);
56 
57         const TableFormattingContext& formattingContext() const { return downcast<TableFormattingContext>(FormattingContext::Geometry::formattingContext()); }
58     };
59     TableFormattingContext::Geometry geometry() const { return Geometry(*this); }
60 
61     IntrinsicWidthConstraints computedIntrinsicWidthConstraints() override;
62     void layoutTableCellBox(const Box& cellLayoutBox, const TableGrid::Column&, InvalidationState&, const HorizontalConstraints&);
63     void positionTableCells();
64     void setComputedGeometryForRows();
65     void setComputedGeometryForSections();
66 
67     void ensureTableGrid();
68     void computePreferredWidthForColumns();
69     void computeAndDistributeExtraHorizontalSpace();
70     enum class WidthConstraintsType { Minimum, Maximum };
71     void useAsContentLogicalWidth(WidthConstraintsType);
72 
73     void initializeDisplayBoxToBlank(Display::Box&) const;
74 
75     const TableFormattingState& formattingState() const { return downcast<TableFormattingState>(FormattingContext::formattingState()); }
76     TableFormattingState& formattingState() { return downcast<TableFormattingState>(FormattingContext::formattingState()); }
77 };
78 
79 inline TableFormattingContext::Geometry::Geometry(const TableFormattingContext& tableFormattingContext)
80     : FormattingContext::Geometry(tableFormattingContext)
81 {
82 }
83 
84 }
85 }
86
