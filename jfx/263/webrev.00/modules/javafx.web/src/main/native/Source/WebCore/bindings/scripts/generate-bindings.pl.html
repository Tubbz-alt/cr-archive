 48 my $outputDirectory;
 49 my $outputHeadersDirectory;
 50 my $generator;
 51 my $defines;

 53 my $prefix;

 62 GetOptions('includeDirsList=s' => \$includeDirsList,
 63            'outputDir=s' => \$outputDirectory,
 64            'outputHeadersDir=s' => \$outputHeadersDirectory,
 65            'generator=s' => \$generator,

 76 open(my $dh, '<', $includeDirsList) or die "Cannot open $includeDirsList";
 77 @idlDirectories = map { (my $path = $_) =~ s/\r?\n?$//; CygwinPathIfNeeded($path) } <$dh>;
 78 close($dh) or die;
 79 
 80 die('Must specify input file.') unless @ARGV;
 81 die('Must specify generator') unless defined($generator);
 82 die('Must specify output directory.') unless defined($outputDirectory);
 83 die('Must specify IDL attributes file.') unless defined($idlAttributesFile);
 84 
 85 if (!$outputHeadersDirectory) {
 86     $outputHeadersDirectory = $outputDirectory;
 87 }

 89 generateBindings($_) for (@ARGV);

 91 sub generateBindings
 92 {
 93     my ($targetIdlFile) = @_;
 94 
 95     $targetIdlFile = Cwd::realpath($targetIdlFile);
 96     if ($verbose) {
 97         print "$generator: $targetIdlFile\n";
 98     }
 99     my $targetInterfaceName = fileparse($targetIdlFile, ".idl");
100 
101     my $idlFound = 0;
102     my %supplementalDependencies;
103     if ($supplementalDependencyFile) {
104         # The format of a supplemental dependency file:
105         #
106         # DOMWindow.idl P.idl Q.idl R.idl
107         # Document.idl S.idl
108         # Event.idl
109         # ...
110         #
111         # The above indicates that DOMWindow.idl is supplemented by P.idl, Q.idl and R.idl,
112         # Document.idl is supplemented by S.idl, and Event.idl is supplemented by no IDLs.
113         # The IDL that supplements another IDL (e.g. P.idl) never appears in the dependency file.
114         open FH, "< $supplementalDependencyFile" or die "Cannot open $supplementalDependencyFile\n";
115         while (my $line = <FH>) {
116             my ($idlFile, @followingIdlFiles) = split(/\s+/, $line);
117             $supplementalDependencies{fileparse($idlFile)} = [sort @followingIdlFiles] if $idlFile;
118         }
119         close FH;
120 
121         if (exists $supplementalDependencies{fileparse($targetIdlFile)}) {
122             $idlFound = 1;
123         }
124 
125         # $additionalIdlFiles is list of IDL files which should not be included in
126         # DerivedSources*.cpp (i.e. they are not described in the supplemental
127         # dependency file) but should generate .h and .cpp files.
128         if (!$idlFound and $additionalIdlFiles) {
129             my @idlFiles = shellwords($additionalIdlFiles);
130             $idlFound = grep { $_ and fileparse($_) eq fileparse($targetIdlFile) } @idlFiles;
131         }
132 
133         if (!$idlFound) {
134             my $codeGen = CodeGenerator->new(\@idlDirectories, $generator, $outputDirectory, $outputHeadersDirectory, $preprocessor, $writeDependencies, $verbose);
135 
136             # We generate empty .h and .cpp files just to tell build scripts that .h and .cpp files are created.
137             generateEmptyHeaderAndCpp($codeGen->FileNamePrefix(), $targetInterfaceName, $outputHeadersDirectory, $outputDirectory);
138             return;
139         }
140     }
141 
142     my $input;
143     {
144         local $INPUT_RECORD_SEPARATOR;
145         open(JSON, "<", $idlAttributesFile) or die "Couldn't open $idlAttributesFile: $!";
146         $input = <JSON>;
147         close(JSON);
148     }
149 
150     my $jsonDecoder = JSON::PP->new->utf8;
151     my $jsonHashRef = $jsonDecoder->decode($input);
152     my $idlAttributes = $jsonHashRef->{attributes};
153 
154     # Parse the target IDL file.
155     my $targetParser = IDLParser->new(!$verbose);
156     my $targetDocument = $targetParser->Parse($targetIdlFile, $defines, $preprocessor, $idlAttributes);
157 
158     # Generate desired output for the target IDL file.
159     my $codeGen = CodeGenerator->new(\@idlDirectories, $generator, $outputDirectory, $outputHeadersDirectory, $preprocessor, $writeDependencies, $verbose, $targetIdlFile, $idlAttributes, \%supplementalDependencies);
160     $codeGen->ProcessDocument($targetDocument, $defines);
161 }
162 
163 sub generateEmptyHeaderAndCpp
164 {
165     my ($prefix, $targetInterfaceName, $outputHeadersDirectory, $outputDirectory) = @_;
166 
167     my $headerName = "${prefix}${targetInterfaceName}.h";
168     my $cppName = "${prefix}${targetInterfaceName}.cpp";
169     my $contents = "/*
170     This file is generated just to tell build scripts that $headerName and
171     $cppName are created for ${targetInterfaceName}.idl, and thus
172     prevent the build scripts from trying to generate $headerName and
173     $cppName at every build. This file must not be tried to compile.
174 */
175 ";
176     open FH, "> ${outputHeadersDirectory}/${headerName}" or die "Cannot open $headerName\n";
177     print FH $contents;
178     close FH;
179 
180     open FH, "> ${outputDirectory}/${cppName}" or die "Cannot open $cppName\n";
181     print FH $contents;
182     close FH;
183 }
184 
185 sub CygwinPathIfNeeded
186 {
187     my $path = shift;
188     return Cygwin::win_to_posix_path($path) if ($^O eq 'cygwin');
189     return $path;
190 }
