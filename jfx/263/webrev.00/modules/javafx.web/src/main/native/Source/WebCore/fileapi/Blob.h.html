 34 #include "BlobPropertyBag.h"
 35 #include "ScriptExecutionContext.h"
 36 #include "ScriptWrappable.h"
 37 #include <wtf/IsoMalloc.h>
 38 #include <wtf/URL.h>

 40 #include <wtf/Variant.h>

 42 namespace JSC {
 43 class ArrayBufferView;
 44 class ArrayBuffer;
 45 }
 46 
 47 namespace WebCore {
 48 
 49 class Blob;
 50 class ScriptExecutionContext;
 51 class SharedBuffer;
 52 
 53 using BlobPartVariant = Variant<RefPtr<JSC::ArrayBufferView>, RefPtr<JSC::ArrayBuffer>, RefPtr<Blob>, String>;
 54 
 55 class Blob : public ScriptWrappable, public URLRegistrable, public RefCounted<Blob> {
 56     WTF_MAKE_ISO_ALLOCATED_EXPORT(Blob, WEBCORE_EXPORT);
 57 public:
 58     static Ref<Blob> create()
 59     {
 60         return adoptRef(*new Blob);
 61     }
 62 
 63     static Ref<Blob> create(Vector<BlobPartVariant>&& blobPartVariants, const BlobPropertyBag& propertyBag)
 64     {
 65         return adoptRef(*new Blob(WTFMove(blobPartVariants), propertyBag));
 66     }
 67 
 68     static Ref<Blob> create(const SharedBuffer& buffer, const String& contentType)
 69     {
 70         return adoptRef(*new Blob(buffer, contentType));
 71     }
 72 
 73     static Ref<Blob> create(Vector<uint8_t>&& data, const String& contentType)
 74     {
 75         return adoptRef(*new Blob(WTFMove(data), contentType));
 76     }
 77 
 78     static Ref<Blob> deserialize(const URL& srcURL, const String& type, long long size, const String& fileBackedPath)
 79     {
 80         ASSERT(Blob::isNormalizedContentType(type));
 81         return adoptRef(*new Blob(deserializationContructor, srcURL, type, size, fileBackedPath));
 82     }
 83 
 84     virtual ~Blob();
 85 
 86     const URL& url() const { return m_internalURL; }
 87     const String& type() const { return m_type; }
 88 
 89     WEBCORE_EXPORT unsigned long long size() const;
 90     virtual bool isFile() const { return false; }
 91 
 92     // The checks described in the File API spec.
 93     static bool isValidContentType(const String&);
 94     // The normalization procedure described in the File API spec.
 95     static String normalizedContentType(const String&);
 96 #if ASSERT_ENABLED
 97     static bool isNormalizedContentType(const String&);
 98     static bool isNormalizedContentType(const CString&);
 99 #endif
100 
101     // URLRegistrable
102     URLRegistry& registry() const override;
103 
104     Ref<Blob> slice(long long start = 0, long long end = std::numeric_limits<long long>::max(), const String& contentType = String()) const
105     {
106         return adoptRef(*new Blob(m_internalURL, start, end, contentType));
107     }
108 
109 protected:
110     WEBCORE_EXPORT Blob();
111     Blob(Vector<BlobPartVariant>&&, const BlobPropertyBag&);
112     Blob(const SharedBuffer&, const String& contentType);
113     Blob(Vector<uint8_t>&&, const String& contentType);
114 
115     enum ReferencingExistingBlobConstructor { referencingExistingBlobConstructor };
116     Blob(ReferencingExistingBlobConstructor, const Blob&);
117 
118     enum UninitializedContructor { uninitializedContructor };
119     Blob(UninitializedContructor, URL&&, String&& type);
120 
121     enum DeserializationContructor { deserializationContructor };
122     Blob(DeserializationContructor, const URL& srcURL, const String& type, Optional<unsigned long long> size, const String& fileBackedPath);
123 
124     // For slicing.
125     Blob(const URL& srcURL, long long start, long long end, const String& contentType);
126 
127 private:
128     // This is an internal URL referring to the blob data associated with this object. It serves
129     // as an identifier for this blob. The internal URL is never used to source the blob's content
130     // into an HTML or for FileRead'ing, public blob URLs must be used for those purposes.
131     URL m_internalURL;
132 
133     String m_type;
134 
135     mutable Optional<unsigned long long> m_size;
136 };
137 
138 } // namespace WebCore
