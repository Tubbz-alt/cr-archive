 52 Ref<Notification> Notification::create(Document& context, const String& title, const Options& options)
 53 {
 54     auto notification = adoptRef(*new Notification(context, title, options));
 55     notification->suspendIfNeeded();
 56     return notification;
 57 }
 58 
 59 Notification::Notification(Document& document, const String& title, const Options& options)
 60     : ActiveDOMObject(document)
 61     , m_title(title)
 62     , m_direction(options.dir)
 63     , m_lang(options.lang)
 64     , m_body(options.body)
 65     , m_tag(options.tag)
 66     , m_state(Idle)
 67     , m_showNotificationTimer(&document, *this, &Notification::show)
 68 {
 69     if (!options.icon.isEmpty()) {
 70         auto iconURL = document.completeURL(options.icon);
 71         if (iconURL.isValid())
 72             m_icon = iconURL;
 73     }
 74 
 75     m_showNotificationTimer.startOneShot(0_s);
 76     m_showNotificationTimer.suspendIfNeeded();
 77 }
 78 
 79 Notification::~Notification() = default;
 80 
 81 void Notification::show()
 82 {
 83     // prevent double-showing
 84     if (m_state != Idle)
 85         return;
 86 
 87     auto* page = document()->page();
 88     if (!page)
 89         return;
 90 
 91     auto& client = NotificationController::from(page)->client();
 92 
 93     if (client.checkPermission(scriptExecutionContext()) != Permission::Granted) {
 94         dispatchErrorEvent();
 95         return;
 96     }
 97     if (client.show(this)) {
 98         m_state = Showing;
 99         setPendingActivity(*this);
100     }
101 }
102 
103 void Notification::close()
104 {
105     switch (m_state) {
106     case Idle:
107         break;
108     case Showing: {
109         if (auto* page = document()->page())
110             NotificationController::from(page)->client().cancel(this);
111         break;
112     }
113     case Closed:
114         break;
115     }
116 }
117 
118 Document* Notification::document() const
119 {
120     return downcast<Document>(scriptExecutionContext());
121 }
122 
123 const char* Notification::activeDOMObjectName() const
124 {
125     return "Notification";
126 }
127 
128 void Notification::stop()
129 {
130     ActiveDOMObject::stop();
131 
132     if (auto* page = document()->page())
133         NotificationController::from(page)->client().notificationObjectDestroyed(this);
134 }
135 
136 void Notification::suspend(ReasonForSuspension)
137 {
138     close();
139 }
140 
141 void Notification::finalize()
142 {
143     if (m_state == Closed)
144         return;
145     m_state = Closed;
146     unsetPendingActivity(*this);
147 }
148 
149 void Notification::queueTask(Function<void()>&& task)
150 {
151     auto* document = this->document();
152     if (!document)
153         return;
154 
155     document->eventLoop().queueTask(TaskSource::UserInteraction, WTFMove(task));
156 }
157 
158 void Notification::dispatchShowEvent()
159 {
160     queueTask([this, pendingActivity = makePendingActivity(*this)] {
161         dispatchEvent(Event::create(eventNames().showEvent, Event::CanBubble::No, Event::IsCancelable::No));
162     });
163 }
164 
165 void Notification::dispatchClickEvent()
166 {
167     queueTask([this, pendingActivity = makePendingActivity(*this)] {
168         WindowFocusAllowedIndicator windowFocusAllowed;
169         dispatchEvent(Event::create(eventNames().clickEvent, Event::CanBubble::No, Event::IsCancelable::No));
170     });
171 }
172 
173 void Notification::dispatchCloseEvent()
174 {
175     queueTask([this, pendingActivity = makePendingActivity(*this)] {
176         dispatchEvent(Event::create(eventNames().closeEvent, Event::CanBubble::No, Event::IsCancelable::No));
177     });
178     finalize();
179 }
180 
181 void Notification::dispatchErrorEvent()
182 {
183     queueTask([this, pendingActivity = makePendingActivity(*this)] {
184         dispatchEvent(Event::create(eventNames().errorEvent, Event::CanBubble::No, Event::IsCancelable::No));
185     });
186 }

188 auto Notification::permission(Document& document) -> Permission
189 {
190     auto* page = document.page();
191     if (!page)
192         return Permission::Default;
193 
194     if (!document.isSecureContext())
195         return Permission::Denied;
196 
197     return NotificationController::from(document.page())->client().checkPermission(&document);
198 }
199 
200 void Notification::requestPermission(Document& document, RefPtr<NotificationPermissionCallback>&& callback)
201 {
202     auto* page = document.page();
203     if (!page)
204         return;
205 
206     if (!document.isSecureContext()) {
207         if (callback) {
208             document.eventLoop().queueTask(TaskSource::DOMManipulation, [callback = WTFMove(callback)]() mutable {
209                 callback->handleEvent(Permission::Denied);
210             });
211         }
212         return;
213     }
214 
215     NotificationController::from(page)->client().requestPermission(&document, WTFMove(callback));
216 }
