34 // The default controller used for managing the relationship between
35 // array buffers and their wrappers in JavaScriptCore. This isn't what
36 // WebCore uses, but it is what JSC uses when running standalone. This
37 // is pretty simple:
38 //
39 // - If the JSArrayBuffer is live, then the ArrayBuffer stays alive.
40 //
41 // - If there is a JSArrayBufferView that is holding an ArrayBuffer
42 //   then any existing wrapper for that ArrayBuffer will be kept
43 //   alive.
44 //
45 // - If you ask an ArrayBuffer for a JSArrayBuffer after one had
46 //   already been created and it didn't die, then you get the same
47 //   one.
48 
49 class SimpleTypedArrayController : public TypedArrayController {
50 public:
51     SimpleTypedArrayController();
52     virtual ~SimpleTypedArrayController();
53 
54     JSArrayBuffer* toJS(ExecState*, JSGlobalObject*, ArrayBuffer*) override;
55     void registerWrapper(JSGlobalObject*, ArrayBuffer*, JSArrayBuffer*) override;
56     bool isAtomicsWaitAllowedOnCurrentThread() override;
57 
58 private:
59     class JSArrayBufferOwner : public WeakHandleOwner {
60     public:
61         bool isReachableFromOpaqueRoots(JSC::Handle<JSC::Unknown>, void* context, SlotVisitor&, const char** reason) override;
62         void finalize(JSC::Handle<JSC::Unknown>, void* context) override;
63     };
64 
65     JSArrayBufferOwner m_owner;
66 };
67 
68 } // namespace JSC
