36 enum class ErrorCode : uint8_t {
37     NoError = 0,
38     PatternTooLarge,
39     QuantifierOutOfOrder,
40     QuantifierWithoutAtom,
41     QuantifierTooLarge,
42     QuantifierIncomplete,
43     MissingParentheses,
44     BracketUnmatched,
45     ParenthesesUnmatched,
46     ParenthesesTypeInvalid,
47     InvalidGroupName,
48     DuplicateGroupName,
49     CharacterClassUnmatched,
50     CharacterClassRangeOutOfOrder,
51     CharacterClassRangeInvalid,
52     EscapeUnterminated,
53     InvalidUnicodeEscape,
54     InvalidBackreference,
55     InvalidIdentityEscape,
56     InvalidControlLetterEscape,
57     InvalidUnicodePropertyExpression,
58     TooManyDisjunctions,
59     OffsetTooLarge,
60     InvalidRegularExpressionFlags,
61 };
62 
63 JS_EXPORT_PRIVATE const char* errorMessage(ErrorCode);
64 inline bool hasError(ErrorCode errorCode)
65 {
66     return errorCode != ErrorCode::NoError;
67 }
68 
69 inline bool hasHardError(ErrorCode errorCode)
70 {
71     // TooManyDisjunctions means that we ran out stack compiling.
72     // All other errors are due to problems in the expression.
73     return hasError(errorCode) && errorCode != ErrorCode::TooManyDisjunctions;
74 }
75 JS_EXPORT_PRIVATE JSObject* errorToThrow(JSGlobalObject*, ErrorCode);
