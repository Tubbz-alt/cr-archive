 39 namespace WebCore {
 40 
 41 class IntPoint;
 42 class ScrollingStateTree;
 43 class ScrollingStateNode;
 44 class ScrollingTreeFrameScrollingNode;
 45 class ScrollingTreeNode;
 46 class ScrollingTreeOverflowScrollProxyNode;
 47 class ScrollingTreePositionedNode;
 48 class ScrollingTreeScrollingNode;
 49 
 50 class ScrollingTree : public ThreadSafeRefCounted<ScrollingTree> {
 51 public:
 52     WEBCORE_EXPORT ScrollingTree();
 53     WEBCORE_EXPORT virtual ~ScrollingTree();
 54 
 55     virtual bool isThreadedScrollingTree() const { return false; }
 56     virtual bool isRemoteScrollingTree() const { return false; }
 57     virtual bool isScrollingTreeIOS() const { return false; }
 58 
 59     // This implies that we'll do hit-testing in the scrolling tree.
 60     bool asyncFrameOrOverflowScrollingEnabled() const { return m_asyncFrameOrOverflowScrollingEnabled; }
 61     void setAsyncFrameOrOverflowScrollingEnabled(bool);
 62 
 63     virtual ScrollingEventResult tryToHandleWheelEvent(const PlatformWheelEvent&) = 0;
 64     WEBCORE_EXPORT bool shouldHandleWheelEventSynchronously(const PlatformWheelEvent&);
 65 
 66     void setMainFrameIsRubberBanding(bool);
 67     bool isRubberBandInProgress();
 68     void setMainFrameIsScrollSnapping(bool);
 69     bool isScrollSnapInProgress();
 70 
 71     virtual void invalidate() { }
 72     WEBCORE_EXPORT virtual void commitTreeState(std::unique_ptr<ScrollingStateTree>);
 73 
 74     WEBCORE_EXPORT virtual void applyLayerPositions();
 75     WEBCORE_EXPORT void applyLayerPositionsAfterCommit();
 76 
 77     virtual Ref<ScrollingTreeNode> createScrollingTreeNode(ScrollingNodeType, ScrollingNodeID) = 0;
 78 
 79     WEBCORE_EXPORT ScrollingTreeNode* nodeForID(ScrollingNodeID) const;
 80 
 81     // Called after a scrolling tree node has handled a scroll and updated its layers.
 82     // Updates FrameView/RenderLayer scrolling state and GraphicsLayers.
 83     virtual void scrollingTreeNodeDidScroll(ScrollingTreeScrollingNode&, ScrollingLayerPositionAction = ScrollingLayerPositionAction::Sync) = 0;
 84 
 85     // Called for requested scroll position updates.
 86     virtual void scrollingTreeNodeRequestsScroll(ScrollingNodeID, const FloatPoint& /*scrollPosition*/, ScrollType, ScrollClamping) { }
 87 
 88     // Delegated scrolling/zooming has caused the viewport to change, so update viewport-constrained layers
 89     WEBCORE_EXPORT void mainFrameViewportChangedViaDelegatedScrolling(const FloatPoint& scrollPosition, const WebCore::FloatRect& layoutViewport, double scale);
 90 
 91     void didScrollByDelegatedScrolling() { m_wasScrolledByDelegatedScrollingSincePreviousCommit = true; }
 92 
 93     void notifyRelatedNodesAfterScrollPositionChange(ScrollingTreeScrollingNode& changedNode);
 94 
 95     virtual void reportSynchronousScrollingReasonsChanged(MonotonicTime, SynchronousScrollingReasons) { }
 96     virtual void reportExposedUnfilledArea(MonotonicTime, unsigned /* unfilledArea */) { }
 97 
 98 #if PLATFORM(IOS_FAMILY)
 99     virtual void scrollingTreeNodeWillStartPanGesture() { }
100     virtual void scrollingTreeNodeWillStartScroll() { }
101     virtual void scrollingTreeNodeDidEndScroll() { }
102 #endif
103 
104     WEBCORE_EXPORT TrackingType eventTrackingTypeForPoint(const AtomString& eventName, IntPoint);
105 
106 #if PLATFORM(MAC)
107     virtual void handleWheelEventPhase(PlatformWheelEventPhase) = 0;
108     virtual void setActiveScrollSnapIndices(ScrollingNodeID, unsigned /*horizontalIndex*/, unsigned /*verticalIndex*/) { }
109 
110     virtual void deferWheelEventTestCompletionForReason(WheelEventTestMonitor::ScrollableAreaIdentifier, WheelEventTestMonitor::DeferReason) { }
111     virtual void removeWheelEventTestCompletionDeferralForReason(WheelEventTestMonitor::ScrollableAreaIdentifier, WheelEventTestMonitor::DeferReason) { }
112 #endif
113 
114 #if PLATFORM(COCOA)
115     WEBCORE_EXPORT virtual void currentSnapPointIndicesDidChange(ScrollingNodeID, unsigned horizontal, unsigned vertical) = 0;
116 #endif
117 
118     void setMainFramePinState(bool pinnedToTheLeft, bool pinnedToTheRight, bool pinnedToTheTop, bool pinnedToTheBottom);
119 
120     // Can be called from any thread. Will update what edges allow rubber-banding.
121     WEBCORE_EXPORT void setCanRubberBandState(bool canRubberBandAtLeft, bool canRubberBandAtRight, bool canRubberBandAtTop, bool canRubberBandAtBottom);
122 
123     bool isHandlingProgrammaticScroll() const { return m_isHandlingProgrammaticScroll; }
124     void setIsHandlingProgrammaticScroll(bool isHandlingProgrammaticScroll) { m_isHandlingProgrammaticScroll = isHandlingProgrammaticScroll; }
125 
126     void setScrollPinningBehavior(ScrollPinningBehavior);
127     WEBCORE_EXPORT ScrollPinningBehavior scrollPinningBehavior();
128 
129     WEBCORE_EXPORT bool willWheelEventStartSwipeGesture(const PlatformWheelEvent&);
130 
131     WEBCORE_EXPORT void setScrollingPerformanceLoggingEnabled(bool flag);
132     bool scrollingPerformanceLoggingEnabled();
133 
134     ScrollingTreeFrameScrollingNode* rootNode() const { return m_rootNode.get(); }
135 
136     ScrollingNodeID latchedNode();
137     void setLatchedNode(ScrollingNodeID);
138     void clearLatchedNode();
139 
140     bool hasLatchedNode() const { return m_treeState.latchedNodeID; }
141     void setOrClearLatchedNode(const PlatformWheelEvent&, ScrollingNodeID);
142 
143     bool hasFixedOrSticky() const { return !!m_fixedOrStickyNodeCount; }
144     void fixedOrStickyNodeAdded() { ++m_fixedOrStickyNodeCount; }
145     void fixedOrStickyNodeRemoved()
146     {
147         ASSERT(m_fixedOrStickyNodeCount);
148         --m_fixedOrStickyNodeCount;
149     }
150 
151     // A map of overflow scrolling nodes to positioned nodes which need to be updated
152     // when the scroller changes, but are not descendants.
153     using RelatedNodesMap = HashMap<ScrollingNodeID, Vector<ScrollingNodeID>>;
154     RelatedNodesMap& overflowRelatedNodes() { return m_overflowRelatedNodesMap; }
155 
156     HashSet<Ref<ScrollingTreeOverflowScrollProxyNode>>& activeOverflowScrollProxyNodes() { return m_activeOverflowScrollProxyNodes; }
157     HashSet<Ref<ScrollingTreePositionedNode>>& activePositionedNodes() { return m_activePositionedNodes; }
158 
159     WEBCORE_EXPORT String scrollingTreeAsText(ScrollingStateTreeAsTextBehavior = ScrollingStateTreeAsTextBehaviorNormal);
160 
161     bool isMonitoringWheelEvents() const { return m_isMonitoringWheelEvents; }
162     bool inCommitTreeState() const { return m_inCommitTreeState; }
163 
164 protected:
165     void setMainFrameScrollPosition(FloatPoint);
166 
167     WEBCORE_EXPORT virtual ScrollingEventResult handleWheelEvent(const PlatformWheelEvent&);
168 
169 private:
170     using OrphanScrollingNodeMap = HashMap<ScrollingNodeID, RefPtr<ScrollingTreeNode>>;
171     void updateTreeFromStateNode(const ScrollingStateNode*, OrphanScrollingNodeMap&, HashSet<ScrollingNodeID>& unvisitedNodes);
172 
173     void applyLayerPositionsRecursive(ScrollingTreeNode&);
174 
175     void notifyRelatedNodesRecursive(ScrollingTreeNode&);
176 
177     Lock m_treeMutex; // Protects the scrolling tree.
178 
179     RefPtr<ScrollingTreeFrameScrollingNode> m_rootNode;
180 
181     using ScrollingTreeNodeMap = HashMap<ScrollingNodeID, RefPtr<ScrollingTreeNode>>;
182     ScrollingTreeNodeMap m_nodeMap;
183 
184     RelatedNodesMap m_overflowRelatedNodesMap;
185 
186     HashSet<Ref<ScrollingTreeOverflowScrollProxyNode>> m_activeOverflowScrollProxyNodes;
187     HashSet<Ref<ScrollingTreePositionedNode>> m_activePositionedNodes;
188 
189     struct TreeState {
190         ScrollingNodeID latchedNodeID { 0 };
191         EventTrackingRegions eventTrackingRegions;
192         FloatPoint mainFrameScrollPosition;
193         bool mainFrameIsRubberBanding { false };
194         bool mainFrameIsScrollSnapping { false };
195     };
196 
197     Lock m_treeStateMutex;
198     TreeState m_treeState;
199 
200     struct SwipeState {
201         ScrollPinningBehavior scrollPinningBehavior { DoNotPin };
202         bool rubberBandsAtLeft { true };
203         bool rubberBandsAtRight { true };
204         bool rubberBandsAtTop { true };
205         bool rubberBandsAtBottom { true };
206         bool mainFramePinnedToTheLeft { true };
207         bool mainFramePinnedToTheRight { true };
208         bool mainFramePinnedToTheTop { true };
209         bool mainFramePinnedToTheBottom { true };
210     };
211 
212     Lock m_swipeStateMutex;
213     SwipeState m_swipeState;
214 
215     unsigned m_fixedOrStickyNodeCount { 0 };
216     bool m_isHandlingProgrammaticScroll { false };
217     bool m_isMonitoringWheelEvents { false };
218     bool m_scrollingPerformanceLoggingEnabled { false };
219     bool m_asyncFrameOrOverflowScrollingEnabled { false };
220     bool m_wasScrolledByDelegatedScrollingSincePreviousCommit { false };
221     bool m_inCommitTreeState { false };
222 };
223 
224 } // namespace WebCore
