26 #pragma once
27 
28 #include <wtf/Forward.h>
29 #include <wtf/Optional.h>
30 
31 namespace WebCore {
32 
33 class Geolocation;
34 class GeolocationPositionData;
35 class Page;
36 
37 class GeolocationClient {
38 public:
39     virtual void geolocationDestroyed() = 0;
40 
41     virtual void startUpdating(const String& authorizationToken) = 0;
42     virtual void stopUpdating() = 0;
43     virtual void revokeAuthorizationToken(const String&) { }
44 
45     // FIXME: The V2 Geolocation specification proposes that this property is
46     // renamed. See http://www.w3.org/2008/geolocation/track/issues/6
47     // We should update WebKit to reflect this if and when the V2 specification
48     // is published.
49     virtual void setEnableHighAccuracy(bool) = 0;
50     virtual Optional<GeolocationPositionData> lastPosition() = 0;
51 
52     virtual void requestPermission(Geolocation&) = 0;
53     virtual void cancelPermissionRequest(Geolocation&) = 0;
54 
55     void provideGeolocationTo(Page*, GeolocationClient&);
56 
57 protected:
58     virtual ~GeolocationClient() = default;
59 };
60 
61 WEBCORE_EXPORT void provideGeolocationTo(Page*, GeolocationClient&);
62 
63 } // namespace WebCore
