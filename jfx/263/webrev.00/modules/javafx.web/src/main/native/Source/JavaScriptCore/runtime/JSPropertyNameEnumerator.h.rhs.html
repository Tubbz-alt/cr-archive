 35 class JSPropertyNameEnumerator final : public JSCell {
 36 public:
 37     using Base = JSCell;
 38     static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
 39 
 40     template<typename CellType, SubspaceAccess mode>
 41     static IsoSubspace* subspaceFor(VM& vm)
 42     {
 43         return &vm.propertyNameEnumeratorSpace;
 44     }
 45 
 46     static JSPropertyNameEnumerator* create(VM&, Structure*, uint32_t, uint32_t, PropertyNameArray&&);
 47 
 48     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
 49     {
 50         return Structure::create(vm, globalObject, prototype, TypeInfo(CellType, StructureFlags), info());
 51     }
 52 
 53     DECLARE_EXPORT_INFO;
 54 
 55     JSString* propertyNameAtIndex(uint32_t index)
 56     {
 57         if (index >= sizeOfPropertyNames())
 58             return nullptr;
 59         return m_propertyNames.get()[index].get();
 60     }
 61 
 62     StructureChain* cachedPrototypeChain() const { return m_prototypeChain.get(); }
 63     void setCachedPrototypeChain(VM& vm, StructureChain* prototypeChain) { return m_prototypeChain.set(vm, this, prototypeChain); }
 64 
 65     Structure* cachedStructure(VM& vm) const
 66     {
 67         if (!m_cachedStructureID)
 68             return nullptr;
 69         return vm.heap.structureIDTable().get(m_cachedStructureID);
 70     }
 71     StructureID cachedStructureID() const { return m_cachedStructureID; }
 72     uint32_t indexedLength() const { return m_indexedLength; }
 73     uint32_t endStructurePropertyIndex() const { return m_endStructurePropertyIndex; }
 74     uint32_t endGenericPropertyIndex() const { return m_endGenericPropertyIndex; }
 75     uint32_t cachedInlineCapacity() const { return m_cachedInlineCapacity; }
 76     uint32_t sizeOfPropertyNames() const { return endGenericPropertyIndex(); }
 77     static ptrdiff_t cachedStructureIDOffset() { return OBJECT_OFFSETOF(JSPropertyNameEnumerator, m_cachedStructureID); }
 78     static ptrdiff_t indexedLengthOffset() { return OBJECT_OFFSETOF(JSPropertyNameEnumerator, m_indexedLength); }
 79     static ptrdiff_t endStructurePropertyIndexOffset() { return OBJECT_OFFSETOF(JSPropertyNameEnumerator, m_endStructurePropertyIndex); }
 80     static ptrdiff_t endGenericPropertyIndexOffset() { return OBJECT_OFFSETOF(JSPropertyNameEnumerator, m_endGenericPropertyIndex); }
 81     static ptrdiff_t cachedInlineCapacityOffset() { return OBJECT_OFFSETOF(JSPropertyNameEnumerator, m_cachedInlineCapacity); }
 82     static ptrdiff_t cachedPropertyNamesVectorOffset()
 83     {
 84         return OBJECT_OFFSETOF(JSPropertyNameEnumerator, m_propertyNames);
 85     }
 86 
 87     static void visitChildren(JSCell*, SlotVisitor&);
 88 
 89 private:
 90     JSPropertyNameEnumerator(VM&, Structure*, uint32_t, uint32_t, WriteBarrier<JSString>*, unsigned);
 91     void finishCreation(VM&, RefPtr<PropertyNameArrayData>&&);
 92 
 93     AuxiliaryBarrier<WriteBarrier<JSString>*> m_propertyNames;
 94     WriteBarrier<StructureChain> m_prototypeChain;
 95     StructureID m_cachedStructureID;
 96     uint32_t m_indexedLength;
 97     uint32_t m_endStructurePropertyIndex;
 98     uint32_t m_endGenericPropertyIndex;
 99     uint32_t m_cachedInlineCapacity;
100 };
101 
102 inline JSPropertyNameEnumerator* propertyNameEnumerator(JSGlobalObject* globalObject, JSObject* base)
103 {
104     VM& vm = getVM(globalObject);
105     auto scope = DECLARE_THROW_SCOPE(vm);
106 
107     uint32_t indexedLength = base->methodTable(vm)->getEnumerableLength(globalObject, base);
108 
109     JSPropertyNameEnumerator* enumerator = nullptr;
110 
111     Structure* structure = base->structure(vm);
112     if (!indexedLength
113         && (enumerator = structure->cachedPropertyNameEnumerator())
114         && enumerator->cachedPrototypeChain() == structure->prototypeChain(globalObject, base))
115         return enumerator;
116 
117     uint32_t numberStructureProperties = 0;
118 
119     PropertyNameArray propertyNames(vm, PropertyNameMode::Strings, PrivateSymbolMode::Exclude);
120 
121     if (structure->canAccessPropertiesQuicklyForEnumeration() && indexedLength == base->getArrayLength()) {
122         base->methodTable(vm)->getStructurePropertyNames(base, globalObject, propertyNames, EnumerationMode());
123         scope.assertNoException();
124 
125         numberStructureProperties = propertyNames.size();
126 
127         base->methodTable(vm)->getGenericPropertyNames(base, globalObject, propertyNames, EnumerationMode());
128     } else {
129         // Generic property names vector contains all indexed property names.
130         // So disable indexed property enumeration phase by setting |indexedLength| to 0.
131         indexedLength = 0;
132         base->methodTable(vm)->getPropertyNames(base, globalObject, propertyNames, EnumerationMode());
133     }
134     RETURN_IF_EXCEPTION(scope, nullptr);
135 
136     ASSERT(propertyNames.size() < UINT32_MAX);
137 
138     bool sawPolyProto;
139     bool successfullyNormalizedChain = normalizePrototypeChain(globalObject, base, sawPolyProto) != InvalidPrototypeChain;
140 
141     Structure* structureAfterGettingPropertyNames = base->structure(vm);
142     enumerator = JSPropertyNameEnumerator::create(vm, structureAfterGettingPropertyNames, indexedLength, numberStructureProperties, WTFMove(propertyNames));
143     if (!indexedLength && successfullyNormalizedChain && structureAfterGettingPropertyNames == structure) {
144         enumerator->setCachedPrototypeChain(vm, structure->prototypeChain(globalObject, base));
145         if (structure->canCachePropertyNameEnumerator(vm))
146             structure->setCachedPropertyNameEnumerator(vm, enumerator);
147     }
148     return enumerator;
149 }
