 49 JSWebAssemblyCodeBlock::JSWebAssemblyCodeBlock(VM& vm, Ref<Wasm::CodeBlock>&& codeBlock, const Wasm::ModuleInformation& moduleInformation)
 50     : Base(vm, vm.webAssemblyCodeBlockStructure.get())
 51     , m_codeBlock(WTFMove(codeBlock))
 52 {
 53     // FIXME: We should not need to do this synchronously.
 54     // https://bugs.webkit.org/show_bug.cgi?id=170567
 55     m_wasmToJSExitStubs.reserveCapacity(m_codeBlock->functionImportCount());
 56     for (unsigned importIndex = 0; importIndex < m_codeBlock->functionImportCount(); ++importIndex) {
 57         Wasm::SignatureIndex signatureIndex = moduleInformation.importFunctionSignatureIndices.at(importIndex);
 58         auto binding = Wasm::wasmToJS(&vm, m_callLinkInfos, signatureIndex, importIndex);
 59         if (UNLIKELY(!binding)) {
 60             switch (binding.error()) {
 61             case Wasm::BindingFailure::OutOfMemory:
 62                 m_errorMessage = "Out of executable memory"_s;
 63                 return;
 64             }
 65             RELEASE_ASSERT_NOT_REACHED();
 66         }
 67         m_wasmToJSExitStubs.uncheckedAppend(binding.value());
 68     }
 69 }
