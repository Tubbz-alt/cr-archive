 42 class PrintStream;
 43 
 44 class WTF_EXPORT_PRIVATE MediaTime final {
 45     WTF_MAKE_FAST_ALLOCATED;
 46 public:
 47     enum {
 48         Valid = 1 << 0,
 49         HasBeenRounded = 1 << 1,
 50         PositiveInfinite = 1 << 2,
 51         NegativeInfinite = 1 << 3,
 52         Indefinite = 1 << 4,
 53         DoubleValue = 1 << 5,
 54     };
 55 
 56     constexpr MediaTime();
 57     constexpr MediaTime(int64_t value, uint32_t scale, uint8_t flags = Valid);
 58     MediaTime(const MediaTime& rhs);
 59 
 60     static MediaTime createWithFloat(float floatTime);
 61     static MediaTime createWithFloat(float floatTime, uint32_t timeScale);
 62     static MediaTime createWithDouble(double doubleTime);
 63     static MediaTime createWithDouble(double doubleTime, uint32_t timeScale);
 64 
 65     float toFloat() const;
 66     double toDouble() const;
 67 
 68     MediaTime& operator=(const MediaTime& rhs);
 69     MediaTime& operator+=(const MediaTime& rhs) { return *this = *this + rhs; }
 70     MediaTime& operator-=(const MediaTime& rhs) { return *this = *this - rhs; }
 71     MediaTime operator+(const MediaTime& rhs) const;
 72     MediaTime operator-(const MediaTime& rhs) const;
 73     MediaTime operator-() const;
 74     MediaTime operator*(int32_t) const;
 75     bool operator<(const MediaTime& rhs) const { return compare(rhs) == LessThan; }
 76     bool operator>(const MediaTime& rhs) const { return compare(rhs) == GreaterThan; }
 77     bool operator!=(const MediaTime& rhs) const { return compare(rhs) != EqualTo; }
 78     bool operator==(const MediaTime& rhs) const { return compare(rhs) == EqualTo; }
 79     bool operator>=(const MediaTime& rhs) const { return compare(rhs) >= EqualTo; }
 80     bool operator<=(const MediaTime& rhs) const { return compare(rhs) <= EqualTo; }
 81     bool operator!() const;
 82     explicit operator bool() const;
 83 
 84     typedef enum {
 85         LessThan = -1,
 86         EqualTo = 0,
 87         GreaterThan = 1,
 88     } ComparisonFlags;
 89 
 90     ComparisonFlags compare(const MediaTime& rhs) const;
 91     bool isBetween(const MediaTime&, const MediaTime&) const;
 92 
 93     bool isValid() const { return m_timeFlags & Valid; }
 94     bool isInvalid() const { return !isValid(); }
 95     bool hasBeenRounded() const { return m_timeFlags & HasBeenRounded; }
 96     bool isPositiveInfinite() const { return m_timeFlags & PositiveInfinite; }
 97     bool isNegativeInfinite() const { return m_timeFlags & NegativeInfinite; }
 98     bool isIndefinite() const { return m_timeFlags & Indefinite; }
 99     bool hasDoubleValue() const { return m_timeFlags & DoubleValue; }
100     uint8_t timeFlags() const { return m_timeFlags; }
101 
102     static const MediaTime& zeroTime();
103     static const MediaTime& invalidTime();
104     static const MediaTime& positiveInfiniteTime();
105     static const MediaTime& negativeInfiniteTime();
106     static const MediaTime& indefiniteTime();
107 
108     const int64_t& timeValue() const { return m_timeValue; }
109     const uint32_t& timeScale() const { return m_timeScale; }
110 
111     void dump(PrintStream& out) const;
112     String toString() const;
113     String toJSONString() const;
114     Ref<JSON::Object> toJSONObject() const;
115 
116     // Make the following casts errors:
117     operator double() const = delete;
118     MediaTime(double) = delete;
119     operator int() const = delete;
120     MediaTime(int) = delete;
121 
122     friend WTF_EXPORT_PRIVATE MediaTime abs(const MediaTime& rhs);
123 
124     static const uint32_t DefaultTimeScale = 10000000;
125     static const uint32_t MaximumTimeScale;
126 
127     enum class RoundingFlags {
128         HalfAwayFromZero = 0,
129         TowardZero,
130         AwayFromZero,
131         TowardPositiveInfinity,
132         TowardNegativeInfinity,
133     };
134 
135     MediaTime toTimeScale(uint32_t, RoundingFlags = RoundingFlags::HalfAwayFromZero) const;
136 
137     template<class Encoder> void encode(Encoder&) const;
138     template<class Decoder> static bool decode(Decoder&, MediaTime&);
139 
140 private:
141     void setTimeScale(uint32_t, RoundingFlags = RoundingFlags::HalfAwayFromZero);
142 
143     union {
144         int64_t m_timeValue;
145         double m_timeValueAsDouble;
146     };
147     uint32_t m_timeScale;
148     uint8_t m_timeFlags;
149 };
150 
151 constexpr MediaTime::MediaTime()
152     : m_timeValue(0)
153     , m_timeScale(DefaultTimeScale)
154     , m_timeFlags(Valid)
155 {
156 }
157 
158 constexpr MediaTime::MediaTime(int64_t value, uint32_t scale, uint8_t flags)
159     : m_timeValue(value)
160     , m_timeScale(scale)
161     , m_timeFlags(flags)
162 {
163     if (scale || !(flags & Valid))
164         return;
165 
166     *this = value < 0 ? negativeInfiniteTime() : positiveInfiniteTime();
167 }
168 
169 inline MediaTime operator*(int32_t lhs, const MediaTime& rhs) { return rhs.operator*(lhs); }
170 
171 WTF_EXPORT_PRIVATE extern MediaTime abs(const MediaTime& rhs);
172 
173 struct WTF_EXPORT_PRIVATE MediaTimeRange {
174     WTF_MAKE_STRUCT_FAST_ALLOCATED;
175 
176     String toJSONString() const;
177 
178     const MediaTime start;
179     const MediaTime end;
180 };
181 
182 template<class Encoder>
183 void MediaTime::encode(Encoder& encoder) const
184 {
185     encoder << m_timeValue << m_timeScale << m_timeFlags;
186 }
187 
188 template<class Decoder>
189 bool MediaTime::decode(Decoder& decoder, MediaTime& time)
190 {
191     return decoder.decode(time.m_timeValue)
192         && decoder.decode(time.m_timeScale)
193         && decoder.decode(time.m_timeFlags);
194 }
195 
196 template<typename> struct LogArgument;
197 
198 template<> struct LogArgument<MediaTime> {
199     static String toString(const MediaTime& time) { return time.toJSONString(); }
200 };
201 template<> struct LogArgument<MediaTimeRange> {
202     static String toString(const MediaTimeRange& range) { return range.toJSONString(); }
203 };

205 #ifndef NDEBUG
206 WTF_EXPORT_PRIVATE TextStream& operator<<(TextStream&, const MediaTime&);
207 #endif
208 
209 }
