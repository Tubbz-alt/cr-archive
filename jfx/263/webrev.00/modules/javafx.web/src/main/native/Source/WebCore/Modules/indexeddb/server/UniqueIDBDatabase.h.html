 66 typedef Function<void(const IDBError&)> ErrorCallback;
 67 typedef Function<void(const IDBError&, const IDBKeyData&)> KeyDataCallback;
 68 typedef Function<void(const IDBError&, const IDBGetResult&)> GetResultCallback;
 69 typedef Function<void(const IDBError&, const IDBGetAllResult&)> GetAllResultsCallback;
 70 typedef Function<void(const IDBError&, uint64_t)> CountCallback;
 71 
 72 class UniqueIDBDatabase : public CanMakeWeakPtr<UniqueIDBDatabase> {
 73     WTF_MAKE_FAST_ALLOCATED;
 74 public:
 75     UniqueIDBDatabase(IDBServer&, const IDBDatabaseIdentifier&);
 76     UniqueIDBDatabase(UniqueIDBDatabase&) = delete;
 77     WEBCORE_EXPORT ~UniqueIDBDatabase();
 78 
 79     void openDatabaseConnection(IDBConnectionToClient&, const IDBRequestData&);
 80 
 81     const IDBDatabaseInfo& info() const;
 82     IDBServer& server() { return m_server; }
 83     const IDBDatabaseIdentifier& identifier() const { return m_identifier; }
 84 
 85     void createObjectStore(UniqueIDBDatabaseTransaction&, const IDBObjectStoreInfo&, ErrorCallback);
 86     void deleteObjectStore(UniqueIDBDatabaseTransaction&, const String& objectStoreName, ErrorCallback);
 87     void renameObjectStore(UniqueIDBDatabaseTransaction&, uint64_t objectStoreIdentifier, const String& newName, ErrorCallback);
 88     void clearObjectStore(UniqueIDBDatabaseTransaction&, uint64_t objectStoreIdentifier, ErrorCallback);
 89     void createIndex(UniqueIDBDatabaseTransaction&, const IDBIndexInfo&, ErrorCallback);
 90     void deleteIndex(UniqueIDBDatabaseTransaction&, uint64_t objectStoreIdentifier, const String& indexName, ErrorCallback);
 91     void renameIndex(UniqueIDBDatabaseTransaction&, uint64_t objectStoreIdentifier, uint64_t indexIdentifier, const String& newName, ErrorCallback);
 92     void putOrAdd(const IDBRequestData&, const IDBKeyData&, const IDBValue&, IndexedDB::ObjectStoreOverwriteMode, KeyDataCallback);
 93     void getRecord(const IDBRequestData&, const IDBGetRecordData&, GetResultCallback);
 94     void getAllRecords(const IDBRequestData&, const IDBGetAllRecordsData&, GetAllResultsCallback);
 95     void getCount(const IDBRequestData&, const IDBKeyRangeData&, CountCallback);
 96     void deleteRecord(const IDBRequestData&, const IDBKeyRangeData&, ErrorCallback);
 97     void openCursor(const IDBRequestData&, const IDBCursorInfo&, GetResultCallback);
 98     void iterateCursor(const IDBRequestData&, const IDBIterateCursorData&, GetResultCallback);
 99     void commitTransaction(UniqueIDBDatabaseTransaction&, ErrorCallback);
100     void abortTransaction(UniqueIDBDatabaseTransaction&, ErrorCallback);
101 
102     void didFinishHandlingVersionChange(UniqueIDBDatabaseConnection&, const IDBResourceIdentifier& transactionIdentifier);
103     void connectionClosedFromClient(UniqueIDBDatabaseConnection&);
104     void didFireVersionChangeEvent(UniqueIDBDatabaseConnection&, const IDBResourceIdentifier& requestIdentifier, IndexedDB::ConnectionClosedOnBehalfOfServer);
105     void openDBRequestCancelled(const IDBResourceIdentifier& requestIdentifier);
106 
107     void enqueueTransaction(Ref<UniqueIDBDatabaseTransaction>&&);
108 
109     void handleDelete(IDBConnectionToClient&, const IDBRequestData&);
110     void immediateCloseForUserDelete();
111 
112     void abortActiveTransactions();
113 
114 private:
115     void handleDatabaseOperations();
116     void handleCurrentOperation();
117     void performCurrentOpenOperation();
118     void performCurrentDeleteOperation();
119     enum class RequestType { Delete, Any };
120     RefPtr<ServerOpenDBRequest> takeNextRunnableRequest(RequestType = RequestType::Any);
121     void addOpenDatabaseConnection(Ref<UniqueIDBDatabaseConnection>&&);
122     bool hasAnyOpenConnections() const;
123     bool allConnectionsAreClosedOrClosing() const;
124 
125     void startVersionChangeTransaction();
126     void maybeNotifyConnectionsOfVersionChange();
127     void notifyCurrentRequestConnectionClosedOrFiredVersionChangeEvent(uint64_t connectionIdentifier);
128 
129     void handleTransactions();
130     RefPtr<UniqueIDBDatabaseTransaction> takeNextRunnableTransaction(bool& hadDeferredTransactions);
131 
132     void activateTransactionInBackingStore(UniqueIDBDatabaseTransaction&);
133     void transactionCompleted(RefPtr<UniqueIDBDatabaseTransaction>&&);
134 
135     void connectionClosedFromServer(UniqueIDBDatabaseConnection&);
136     void deleteBackingStore();
137     void didDeleteBackingStore(uint64_t deletedVersion);
138     void close();
139 
140     bool isCurrentlyInUse() const;
141     void clearStalePendingOpenDBRequests();
142 
143     void clearTransactionsOnConnection(UniqueIDBDatabaseConnection&);
144 
145     IDBServer& m_server;
146     IDBDatabaseIdentifier m_identifier;
147 
148     ListHashSet<RefPtr<ServerOpenDBRequest>> m_pendingOpenDBRequests;
149     RefPtr<ServerOpenDBRequest> m_currentOpenDBRequest;
150 
151     ListHashSet<RefPtr<UniqueIDBDatabaseConnection>> m_openDatabaseConnections;
152 
153     RefPtr<UniqueIDBDatabaseConnection> m_versionChangeDatabaseConnection;
154     RefPtr<UniqueIDBDatabaseTransaction> m_versionChangeTransaction;
155 
156     std::unique_ptr<IDBBackingStore> m_backingStore;
157     std::unique_ptr<IDBDatabaseInfo> m_databaseInfo;
158     std::unique_ptr<IDBDatabaseInfo> m_mostRecentDeletedDatabaseInfo;
159 
160     Deque<RefPtr<UniqueIDBDatabaseTransaction>> m_pendingTransactions;
161     HashMap<IDBResourceIdentifier, RefPtr<UniqueIDBDatabaseTransaction>> m_inProgressTransactions;
162 
163     // The keys into these sets are the object store ID.
164     // These sets help to decide which transactions can be started and which must be deferred.
165     HashCountedSet<uint64_t> m_objectStoreTransactionCounts;
166     HashSet<uint64_t> m_objectStoreWriteTransactions;
167 
168     HashSet<IDBResourceIdentifier> m_cursorPrefetches;
169 };
