 32 #pragma once
 33 
 34 #include "ExceptionOr.h"
 35 #include "GCReachableRef.h"
 36 #include <wtf/Forward.h>
 37 #include <wtf/HashSet.h>
 38 #include <wtf/IsoMalloc.h>
 39 #include <wtf/Vector.h>
 40 
 41 namespace WebCore {
 42 
 43 class Document;
 44 class HTMLSlotElement;
 45 class MutationCallback;
 46 class MutationObserverRegistration;
 47 class MutationRecord;
 48 class Node;
 49 class WindowEventLoop;
 50 
 51 using MutationObserverOptions = unsigned char;
 52 using MutationRecordDeliveryOptions = unsigned char;
 53 
 54 class MutationObserver final : public RefCounted<MutationObserver> {
 55     WTF_MAKE_ISO_ALLOCATED(MutationObserver);

 56 public:
 57     enum MutationType {
 58         ChildList = 1 << 0,
 59         Attributes = 1 << 1,
 60         CharacterData = 1 << 2,
 61 
 62         AllMutationTypes = ChildList | Attributes | CharacterData
 63     };
 64 
 65     enum ObservationFlags  {
 66         Subtree = 1 << 3,
 67         AttributeFilter = 1 << 4
 68     };
 69 
 70     enum DeliveryFlags {
 71         AttributeOldValue = 1 << 5,
 72         CharacterDataOldValue = 1 << 6,
 73     };
 74 
 75     static Ref<MutationObserver> create(Ref<MutationCallback>&&);
 76 
 77     ~MutationObserver();
 78 
 79     struct Init {
 80         bool childList;
 81         Optional<bool> attributes;
 82         Optional<bool> characterData;
 83         bool subtree;
 84         Optional<bool> attributeOldValue;
 85         Optional<bool> characterDataOldValue;
 86         Optional<Vector<String>> attributeFilter;
 87     };
 88 
 89     ExceptionOr<void> observe(Node&, const Init&);
 90 
 91     struct TakenRecords {
 92         Vector<Ref<MutationRecord>> records;
 93         HashSet<GCReachableRef<Node>> pendingTargets;
 94     };
 95     TakenRecords takeRecords();
 96     void disconnect();
 97 
 98     void observationStarted(MutationObserverRegistration&);
 99     void observationEnded(MutationObserverRegistration&);
100     void enqueueMutationRecord(Ref<MutationRecord>&&);
101     void setHasTransientRegistration(Document&);
102     bool canDeliver();
103 
104     HashSet<Node*> observedNodes() const;
105 
106     MutationCallback& callback() const { return m_callback.get(); }
107 
108     static void enqueueSlotChangeEvent(HTMLSlotElement&);
109 
110     static void notifyMutationObservers(WindowEventLoop&);
111 
112 private:
113     explicit MutationObserver(Ref<MutationCallback>&&);
114     void deliver();
115 

116     static bool validateOptions(MutationObserverOptions);
117 
118     Ref<MutationCallback> m_callback;
119     Vector<Ref<MutationRecord>> m_records;
120     HashSet<GCReachableRef<Node>> m_pendingTargets;
121     HashSet<MutationObserverRegistration*> m_registrations;
122     unsigned m_priority;
123 };
124 
125 } // namespace WebCore
