 45 void ServiceWorkerInternals::setOnline(bool isOnline)
 46 {
 47     callOnMainThread([identifier = m_identifier, isOnline] () {
 48         if (auto* proxy = SWContextManager::singleton().workerByID(identifier))
 49             proxy->notifyNetworkStateChange(isOnline);
 50     });
 51 }
 52 
 53 void ServiceWorkerInternals::waitForFetchEventToFinish(FetchEvent& event, DOMPromiseDeferred<IDLInterface<FetchResponse>>&& promise)
 54 {
 55     event.onResponse([promise = WTFMove(promise), event = makeRef(event)] (auto&& result) mutable {
 56         if (result.has_value())
 57             promise.resolve(WTFMove(result.value()));
 58         else
 59             promise.reject(TypeError, result.error().localizedDescription());
 60     });
 61 }

 63 Ref<FetchEvent> ServiceWorkerInternals::createBeingDispatchedFetchEvent(ScriptExecutionContext& context)
 64 {
 65     auto event = FetchEvent::createForTesting(context);
 66     event->setEventPhase(Event::CAPTURING_PHASE);
 67     return event;
 68 }
 69 
 70 Ref<FetchResponse> ServiceWorkerInternals::createOpaqueWithBlobBodyResponse(ScriptExecutionContext& context)
 71 {
 72     auto blob = Blob::create();
 73     auto formData = FormData::create();
 74     formData->appendBlob(blob->url());
 75 
 76     ResourceResponse response;
 77     response.setType(ResourceResponse::Type::Cors);
 78     response.setTainting(ResourceResponse::Tainting::Opaque);
 79     auto fetchResponse = FetchResponse::create(context, FetchBody::fromFormData(formData), FetchHeaders::Guard::Response, WTFMove(response));
 80     fetchResponse->initializeOpaqueLoadIdentifierForTesting();
 81     return fetchResponse;
 82 }
 83 
 84 Vector<String> ServiceWorkerInternals::fetchResponseHeaderList(FetchResponse& response)
 85 {
 86     Vector<String> headerNames;
 87     headerNames.reserveInitialCapacity(response.internalResponseHeaders().size());
 88     for (auto keyValue : response.internalResponseHeaders())
 89         headerNames.uncheckedAppend(keyValue.key);
 90     return headerNames;
 91 }
