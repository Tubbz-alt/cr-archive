 46 namespace WebCore {
 47 
 48 class ScriptExecutionContext;
 49 class WorkerGlobalScopeProxy;
 50 class WorkerScriptLoader;
 51 
 52 class Worker final : public AbstractWorker, public ActiveDOMObject, private WorkerScriptLoaderClient {
 53     WTF_MAKE_ISO_ALLOCATED(Worker);
 54 public:
 55     struct Options {
 56         String name;
 57     };
 58     static ExceptionOr<Ref<Worker>> create(ScriptExecutionContext&, JSC::RuntimeFlags, const String& url, const Options&);
 59     virtual ~Worker();
 60 
 61     ExceptionOr<void> postMessage(JSC::JSGlobalObject&, JSC::JSValue message, PostMessageOptions&&);
 62 
 63     void terminate();
 64     bool wasTerminated() const { return m_wasTerminated; }
 65 
 66     bool hasPendingActivity() const final;
 67 
 68     String identifier() const { return m_identifier; }
 69 
 70     ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
 71 
 72     void dispatchEvent(Event&) final;
 73 
 74 private:
 75     explicit Worker(ScriptExecutionContext&, JSC::RuntimeFlags, const Options&);
 76 
 77     EventTargetInterface eventTargetInterface() const final { return WorkerEventTargetInterfaceType; }
 78 
 79     void notifyNetworkStateChange(bool isOnline);
 80 
 81     void didReceiveResponse(unsigned long identifier, const ResourceResponse&) final;
 82     void notifyFinished() final;
 83 
 84     void stop() final;
 85     void suspend(ReasonForSuspension) final;
 86     void resume() final;
 87     const char* activeDOMObjectName() const final;
 88 
 89     static void networkStateChanged(bool isOnLine);
 90 
 91     RefPtr<WorkerScriptLoader> m_scriptLoader;
 92     String m_name;
 93     String m_identifier;
 94     WorkerGlobalScopeProxy& m_contextProxy; // The proxy outlives the worker to perform thread shutdown.
 95     Optional<ContentSecurityPolicyResponseHeaders> m_contentSecurityPolicyResponseHeaders;
 96     MonotonicTime m_workerCreationTime;
 97     bool m_shouldBypassMainWorldContentSecurityPolicy { false };
 98     bool m_isSuspendedForBackForwardCache { false };
 99     JSC::RuntimeFlags m_runtimeFlags;
100     Deque<RefPtr<Event>> m_pendingEvents;
101     bool m_wasTerminated { false };
102 };
