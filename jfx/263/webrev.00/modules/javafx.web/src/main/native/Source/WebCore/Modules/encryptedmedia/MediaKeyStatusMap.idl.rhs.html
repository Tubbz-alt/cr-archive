29 enum MediaKeyStatus {

32     "released",
33     "output-restricted",
34     "output-downscaled",
35     "status-pending",
36     "internal-error"
37 };

43 ] interface MediaKeyStatusMap {
44     iterable<BufferSource, MediaKeyStatus>;
45     readonly attribute unsigned long size;
46     boolean has(BufferSource keyId);
47     [CallWith=GlobalObject] any get(BufferSource keyId);
48 };
