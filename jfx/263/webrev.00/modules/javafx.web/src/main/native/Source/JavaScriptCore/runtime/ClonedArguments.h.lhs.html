33 // This is an Arguments-class object that we create when you do function.arguments, or you say
34 // "arguments" inside a function in strict mode. It behaves almpst entirely like an ordinary
35 // JavaScript object. All of the arguments values are simply copied from the stack (possibly via
36 // some sophisticated ValueRecovery's if an optimizing compiler is in play) and the appropriate
37 // properties of the object are populated. The only reason why we need a special class is to make
38 // the object claim to be "Arguments" from a toString standpoint, and to avoid materializing the
39 // caller/callee/@@iterator properties unless someone asks for them.
40 class ClonedArguments final : public JSNonFinalObject {
41 public:
42     typedef JSNonFinalObject Base;
43     static const unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetPropertyNames;







44 
45 private:
46     ClonedArguments(VM&, Structure*, Butterfly*);
47 
48 public:
49     static ClonedArguments* createEmpty(VM&, Structure*, JSFunction* callee, unsigned length);
50     static ClonedArguments* createEmpty(ExecState*, JSFunction* callee, unsigned length);
51     static ClonedArguments* createWithInlineFrame(ExecState* myFrame, ExecState* targetFrame, InlineCallFrame*, ArgumentsMode);
52     static ClonedArguments* createWithMachineFrame(ExecState* myFrame, ExecState* targetFrame, ArgumentsMode);
53     static ClonedArguments* createByCopyingFrom(ExecState*, Structure*, Register* argumentsStart, unsigned length, JSFunction* callee);
54 
55     static Structure* createStructure(VM&, JSGlobalObject*, JSValue prototype);
56     static Structure* createSlowPutStructure(VM&, JSGlobalObject*, JSValue prototype);
57 
58     static void visitChildren(JSCell*, SlotVisitor&);
59 
60     DECLARE_INFO;
61 
62 private:
63     static Structure* createStructure(VM&, JSGlobalObject*, JSValue prototype, IndexingType);
64 
65     static bool getOwnPropertySlot(JSObject*, ExecState*, PropertyName, PropertySlot&);
66     static void getOwnPropertyNames(JSObject*, ExecState*, PropertyNameArray&, EnumerationMode);
67     static bool put(JSCell*, ExecState*, PropertyName, JSValue, PutPropertySlot&);
68     static bool deleteProperty(JSCell*, ExecState*, PropertyName);
69     static bool defineOwnProperty(JSObject*, ExecState*, PropertyName, const PropertyDescriptor&, bool shouldThrow);
70 
71     bool specialsMaterialized() const { return !m_callee; }
72     void materializeSpecials(ExecState*);
73     void materializeSpecialsIfNecessary(ExecState*);
74 
75     WriteBarrier<JSFunction> m_callee; // Set to nullptr when we materialize all of our special properties.
76 };
77 
78 static const PropertyOffset clonedArgumentsLengthPropertyOffset = 100;
79
