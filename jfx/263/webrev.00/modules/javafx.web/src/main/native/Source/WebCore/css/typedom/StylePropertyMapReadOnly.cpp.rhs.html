48 RefPtr<TypedOMCSSStyleValue> StylePropertyMapReadOnly::reifyValue(CSSValue* value, Document& document, Element*)
49 {
50     if (!value)
51         return nullptr;
52 
53     // FIXME: Properly reify all length values.
54     if (is<CSSPrimitiveValue>(*value) && downcast<CSSPrimitiveValue>(*value).primitiveType() == CSSUnitType::CSS_PX)
55         return TypedOMCSSUnitValue::create(downcast<CSSPrimitiveValue>(*value).doubleValue(), "px");
56 
57     if (is<CSSImageValue>(*value))
58         return TypedOMCSSImageValue::create(downcast<CSSImageValue>(*value), document);
59 
60     // FIXME: should use raw TypedOMCSSStyleValue
61     return TypedOMCSSUnparsedValue::create(value->cssText());
62 }
63 
64 RefPtr<TypedOMCSSStyleValue> StylePropertyMapReadOnly::customPropertyValueOrDefault(const String& name, Document& document, CSSValue* inputValue, Element* element)
65 {
66     if (!inputValue) {
67         auto* registered = document.getCSSRegisteredCustomPropertySet().get(name);
68 
69         if (registered && registered->initialValue()) {
70             auto value = registered->initialValueCopy();
71             return StylePropertyMapReadOnly::reifyValue(value.get(), document, element);
72         }
73 
74         return TypedOMCSSUnparsedValue::create(emptyString());
75     }
76 
77     return StylePropertyMapReadOnly::reifyValue(inputValue, document, element);
78 }
