41 Ref<PublicKeyCredential> PublicKeyCredential::create(Ref<AuthenticatorResponse>&& response)
42 {
43     return adoptRef(*new PublicKeyCredential(WTFMove(response)));
44 }
45 
46 ArrayBuffer* PublicKeyCredential::rawId() const
47 {
48     return m_response->rawId();
49 }
50 
51 AuthenticationExtensionsClientOutputs PublicKeyCredential::getClientExtensionResults() const
52 {
53     return m_response->extensions();
54 }
55 
56 PublicKeyCredential::PublicKeyCredential(Ref<AuthenticatorResponse>&& response)
57     : BasicCredential(WTF::base64URLEncode(response->rawId()->data(), response->rawId()->byteLength()), Type::PublicKey, Discovery::Remote)
58     , m_response(WTFMove(response))
59 {
60 }
61 
62 void PublicKeyCredential::isUserVerifyingPlatformAuthenticatorAvailable(Document& document, DOMPromiseDeferred<IDLBoolean>&& promise)
63 {
64     if (!RuntimeEnabledFeatures::sharedFeatures().webAuthenticationLocalAuthenticatorEnabled()) {
65         promise.resolve(false);
66         return;
67     }
68     document.page()->authenticatorCoordinator().isUserVerifyingPlatformAuthenticatorAvailable(WTFMove(promise));
69 }
