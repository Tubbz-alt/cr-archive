33 namespace JSC {

35 class GetterSetterAccessCase : public ProxyableAccessCase {
36 public:
37     typedef ProxyableAccessCase Base;
38     friend class AccessCase;
39 
40     // This can return null if it hasn't been generated yet. That's
41     // actually somewhat likely because of how we do buffering of new cases.
42     // CallLinkInfo's ownership is held both by generated code via GCAwareJITStubRoutine and PolymorphicAccess.
43     // The ownership relation is PolymorphicAccess -> GCAwareJITStubRoutine -> CallLinkInfo.
44     // PolymorphicAccess can be destroyed while GCAwareJITStubRoutine is alive if we are destroying PolymorphicAccess
45     // while we are executing GCAwareJITStubRoutine. It is not possible that GetterSetterAccessCase is alive while
46     // GCAwareJITStubRoutine is destroyed.
47     CallLinkInfo* callLinkInfo() const { return m_callLinkInfo; }
48     JSObject* customSlotBase() const { return m_customSlotBase.get(); }
49     Optional<DOMAttributeAnnotation> domAttribute() const { return m_domAttribute; }
50 
51     bool hasAlternateBase() const override;
52     JSObject* alternateBase() const override;
53 
54     void emitDOMJITGetter(AccessGenerationState&, const DOMJIT::GetterSetter*, GPRReg baseForGetGPR);
55 
56     static std::unique_ptr<AccessCase> create(
57         VM&, JSCell* owner, AccessType, CacheableIdentifier, PropertyOffset, Structure*,
58         const ObjectPropertyConditionSet&, bool viaProxy, WatchpointSet* additionalSet, FunctionPtr<OperationPtrTag> customGetter,
59         JSObject* customSlotBase, Optional<DOMAttributeAnnotation>, std::unique_ptr<PolyProtoAccessChain>);
60 
61     static std::unique_ptr<AccessCase> create(VM&, JSCell* owner, AccessType, Structure*, CacheableIdentifier, PropertyOffset,
62         const ObjectPropertyConditionSet&, std::unique_ptr<PolyProtoAccessChain>,
63         FunctionPtr<OperationPtrTag> customSetter = nullptr, JSObject* customSlotBase = nullptr);
64 
65     void dumpImpl(PrintStream&, CommaPrinter&) const override;
66     std::unique_ptr<AccessCase> clone() const override;
67 
68     ~GetterSetterAccessCase();
69 
70     FunctionPtr<OperationPtrTag> customAccessor() const { return m_customAccessor; }
71 
72 private:
73     GetterSetterAccessCase(VM&, JSCell*, AccessType, CacheableIdentifier, PropertyOffset, Structure*, const ObjectPropertyConditionSet&, bool viaProxy, WatchpointSet* additionalSet, JSObject* customSlotBase, std::unique_ptr<PolyProtoAccessChain>);
74 
75     GetterSetterAccessCase(const GetterSetterAccessCase&);
76 
77     WriteBarrier<JSObject> m_customSlotBase;
78     CallLinkInfo* m_callLinkInfo { nullptr };
79     FunctionPtr<OperationPtrTag> m_customAccessor;
80     Optional<DOMAttributeAnnotation> m_domAttribute;
81 };
