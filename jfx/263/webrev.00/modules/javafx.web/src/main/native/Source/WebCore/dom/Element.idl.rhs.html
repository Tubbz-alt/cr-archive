 27 ] interface Element : Node {
 28     readonly attribute DOMString? namespaceURI;
 29     readonly attribute DOMString? prefix;
 30     readonly attribute DOMString localName;
 31     readonly attribute DOMString tagName;
 32 
 33     [CEReactions, Reflect] attribute DOMString id;
 34     [CEReactions, Reflect=class] attribute DOMString className;

 36     [CEReactions, EnabledAtRuntime=ShadowDOM, Reflect, Unscopable] attribute DOMString slot;
 37     [EnabledAtRuntime=CSSShadowParts, SameObject, PutForwards=value] readonly attribute DOMTokenList part;
 38 
 39     [DOMJIT=ReadDOM] boolean hasAttributes();
 40     readonly attribute NamedNodeMap attributes; // FIXME: Should be [SameObject].
 41     sequence<DOMString> getAttributeNames();
 42     [DOMJIT=ReadDOM] DOMString? getAttribute(DOMString qualifiedName);
 43     DOMString? getAttributeNS(DOMString? namespaceURI, DOMString localName);
 44     [CEReactions, MayThrowException] void setAttribute(DOMString qualifiedName, DOMString value);
 45     [CEReactions, MayThrowException] void setAttributeNS(DOMString? namespaceURI, DOMString qualifiedName, DOMString value);
 46     [CEReactions] void removeAttribute(DOMString qualifiedName);
 47     [CEReactions] void removeAttributeNS(DOMString? namespaceURI, DOMString localName);
 48     [CEReactions, MayThrowException] boolean toggleAttribute(DOMString qualifiedName, optional boolean force);
 49     boolean hasAttribute(DOMString qualifiedName);
 50     boolean hasAttributeNS(DOMString? namespaceURI, DOMString localName);
 51 
 52     [DOMJIT=ReadDOM] Attr? getAttributeNode(DOMString qualifiedName);
 53     Attr? getAttributeNodeNS(DOMString? namespaceURI, DOMString localName);
 54     [CEReactions, MayThrowException] Attr? setAttributeNode(Attr attr);
 55     [CEReactions, MayThrowException] Attr? setAttributeNodeNS(Attr attr);
 56     [CEReactions, MayThrowException] Attr removeAttributeNode(Attr attr);
 57 
 58     [EnabledAtRuntime=ShadowDOM, MayThrowException] ShadowRoot attachShadow(ShadowRootInit init);
 59     [EnabledAtRuntime=ShadowDOM, ImplementedAs=shadowRootForBindings, CallWith=GlobalObject] readonly attribute ShadowRoot? shadowRoot;
 60 
 61     [MayThrowException] Element? closest(DOMString selectors);
 62     [MayThrowException] boolean matches(DOMString selectors);
 63     [ImplementedAs=matches, MayThrowException] boolean webkitMatchesSelector(DOMString selectors); // Historical alias of .matches.

 65     [DOMJIT=ReadDOM] HTMLCollection getElementsByTagName(DOMString qualifiedName);
 66     HTMLCollection getElementsByTagNameNS(DOMString? namespaceURI, DOMString localName);
 67     HTMLCollection getElementsByClassName(DOMString name);
 68 
 69     [CEReactions, MayThrowException] Element? insertAdjacentElement(DOMString where, Element element); // Historical.
 70     [MayThrowException] void insertAdjacentText(DOMString where, DOMString data); // Historical.
 71 
 72     // CSSOM-view extensions (https://drafts.csswg.org/cssom-view/#extension-to-the-element-interface).
 73     DOMRectList getClientRects();
 74     [NewObject] DOMRect getBoundingClientRect();
 75     void scrollIntoView(optional (boolean or ScrollIntoViewOptions) arg);
 76     [ImplementedAs=scrollTo] void scroll(optional ScrollToOptions options);
 77     [ImplementedAs=scrollTo] void scroll(unrestricted double x, unrestricted double y);

 79     void scrollTo(unrestricted double x, unrestricted double y);
 80     void scrollBy(optional ScrollToOptions option);
 81     void scrollBy(unrestricted double x, unrestricted double y);

 85     readonly attribute long scrollHeight;
 86     readonly attribute double clientTop; // FIXME: Should be of type long.
 87     readonly attribute double clientLeft; // FIXME: Should be of type long.
 88     readonly attribute double clientWidth; // FIXME: Should be of type long.
 89     readonly attribute double clientHeight; // FIXME: Should be of type long.

 91     // Extensions to Element from DOM Parsing and Serialization specification.

 94     [CEReactions] attribute [TreatNullAs=EmptyString] DOMString outerHTML;
 95     [CEReactions, MayThrowException] void insertAdjacentHTML(DOMString position, DOMString text);

 97     // Extensions from Full Screen API (https://fullscreen.spec.whatwg.org/#api).

105     [Conditional=POINTER_EVENTS, EnabledAtRuntime=PointerEvents] boolean hasPointerCapture(long pointerId);
106 
107     // Extensions from Pointer Lock API (https://w3c.github.io/pointerlock/#extensions-to-the-element-interface).

111     void scrollIntoViewIfNeeded(optional boolean centerIfNeeded = true);

115     [NotEnumerable] attribute EventHandler onselectstart; // FIXME: Should be enumerable.

117     // Non-standard WebKit API (https://developer.apple.com/reference/webkitjs/gestureevent).

119     [NotEnumerable, Conditional=IOS_GESTURE_EVENTS] attribute EventHandler ongestureend;
120     [NotEnumerable, Conditional=IOS_GESTURE_EVENTS] attribute EventHandler ongesturestart;
121 
122     // Non standard.
123     [NotEnumerable, ImplementedAs=onwebkitAnimationEnd] attribute EventHandler onwebkitanimationend;
124     [NotEnumerable, ImplementedAs=onwebkitAnimationIteration] attribute EventHandler onwebkitanimationiteration;
125     [NotEnumerable, ImplementedAs=onwebkitAnimationStart] attribute EventHandler onwebkitanimationstart;
126     [NotEnumerable, ImplementedAs=onwebkitTransitionEnd] attribute EventHandler onwebkittransitionend;
127 
128     // FIXME: These event handlers should only be on Document (https://fullscreen.spec.whatwg.org/#api).
129     [NotEnumerable, Conditional=FULLSCREEN_API, EnabledBySetting=FullScreen, DisabledByQuirk=shouldDisableElementFullscreen] attribute EventHandler onwebkitfullscreenchange;
130     [NotEnumerable, Conditional=FULLSCREEN_API, EnabledBySetting=FullScreen, DisabledByQuirk=shouldDisableElementFullscreen] attribute EventHandler onwebkitfullscreenerror;
131 
132     // FIXME: Cannot find those EventHandlers in the latest specification (https://w3c.github.io/uievents/#events-focus-types).
133     [NotEnumerable] attribute EventHandler onfocusin;
134     [NotEnumerable] attribute EventHandler onfocusout;
135 
136     // Non standard event handler (https://developer.apple.com/reference/webkitjs/element/1633146-onbeforeload).
137     [NotEnumerable] attribute EventHandler onbeforeload;
138 
139     // Non standard event handler (https://developer.apple.com/reference/webkitjs/element/1634205-onwebkitneedkey).
140     [NotEnumerable, Conditional=LEGACY_ENCRYPTED_MEDIA, EnabledAtRuntime=LegacyEncryptedMediaAPI] attribute EventHandler onwebkitneedkey;
141 
142     // Non standard event handler (https://developer.apple.com/reference/webkitjs/element/1633943-onwebkitpresentationmodechanged).
143     [NotEnumerable, Conditional=VIDEO_PRESENTATION_MODE] attribute EventHandler onwebkitpresentationmodechanged;
144 
145     // Non standard event handler (https://developer.apple.com/reference/webkitjs/element/1630042-onwebkitcurrentplaybacktargetisw).
146     [NotEnumerable, Conditional=WIRELESS_PLAYBACK_TARGET] attribute EventHandler onwebkitcurrentplaybacktargetiswirelesschanged;
147 
148     // Non standard event handler (https://developer.apple.com/reference/webkitjs/element/1629580-onwebkitplaybacktargetavailabili).
149     [NotEnumerable, Conditional=WIRELESS_PLAYBACK_TARGET] attribute EventHandler onwebkitplaybacktargetavailabilitychanged;
150 };
