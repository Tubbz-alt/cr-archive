21 #include "config.h"
22 #include "VRPlatformDisplay.h"
23 
24 namespace WebCore {
25 
26 void VRPlatformDisplay::setClient(VRPlatformDisplayClient* client)
27 {
28     ASSERT(!client || !m_client);
29     m_client = client;
30 }
31 
32 void VRPlatformDisplay::notifyVRPlatformDisplayEvent(Event event)
33 {
34     if (!m_client)
35         return;
36 
37     switch (event) {
38     case Event::Connected:
39         m_client->platformDisplayConnected();
40         break;
41     case Event::Disconnected:
42         m_client->platformDisplayDisconnected();
43         break;
44     case Event::Mounted:
45         m_client->platformDisplayMounted();
46         break;
47     case Event::Unmounted:
48         m_client->platformDisplayUnmounted();
49         break;
50     default:
51         ASSERT_NOT_REACHED();
52     };
53 }
54 
55 } // namespace WebCore
