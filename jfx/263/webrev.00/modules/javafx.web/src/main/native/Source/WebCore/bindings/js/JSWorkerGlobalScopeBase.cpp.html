 73 JSWorkerGlobalScopeBase::JSWorkerGlobalScopeBase(JSC::VM& vm, JSC::Structure* structure, RefPtr<WorkerGlobalScope>&& impl)
 74     : JSDOMGlobalObject(vm, structure, normalWorld(vm), &s_globalObjectMethodTable)
 75     , m_wrapped(WTFMove(impl))
 76 {
 77 }

 79 void JSWorkerGlobalScopeBase::finishCreation(VM& vm, JSProxy* proxy)
 80 {
 81     m_proxy.set(vm, this, proxy);
 82 
 83     Base::finishCreation(vm, m_proxy.get());
 84     ASSERT(inherits(vm, info()));
 85 }

 87 void JSWorkerGlobalScopeBase::clearDOMGuardedObjects()
 88 {
 89     auto guardedObjects = m_guardedObjects;
 90     for (auto& guarded : guardedObjects)
 91         guarded->clear();
 92 }

 94 void JSWorkerGlobalScopeBase::visitChildren(JSCell* cell, SlotVisitor& visitor)
 95 {
 96     JSWorkerGlobalScopeBase* thisObject = jsCast<JSWorkerGlobalScopeBase*>(cell);
 97     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
 98     Base::visitChildren(thisObject, visitor);
 99     visitor.append(thisObject->m_proxy);
100 }

133 void JSWorkerGlobalScopeBase::queueMicrotaskToEventLoop(JSGlobalObject& object, Ref<JSC::Microtask>&& task)
134 {
135     JSWorkerGlobalScopeBase& thisObject = static_cast<JSWorkerGlobalScopeBase&>(object);
136 
137     auto callback = JSMicrotaskCallback::create(thisObject, WTFMove(task));
138     auto& context = thisObject.wrapped();
139     context.eventLoop().queueMicrotask([callback = WTFMove(callback)]() mutable {
140         callback->call();
141     });
142 }

145 {
146     return toJS(lexicalGlobalObject, workerGlobalScope);

148 
149 JSValue toJS(JSGlobalObject*, WorkerGlobalScope& workerGlobalScope)
150 {
151     WorkerScriptController* script = workerGlobalScope.script();
152     if (!script)
153         return jsNull();
154     JSWorkerGlobalScope* contextWrapper = script->workerGlobalScopeWrapper();
155     ASSERT(contextWrapper);
156     return contextWrapper->proxy();
157 }

159 JSDedicatedWorkerGlobalScope* toJSDedicatedWorkerGlobalScope(VM& vm, JSValue value)
160 {
161     if (!value.isObject())
162         return nullptr;
163     const ClassInfo* classInfo = asObject(value)->classInfo(vm);
164     if (classInfo == JSDedicatedWorkerGlobalScope::info())
165         return jsCast<JSDedicatedWorkerGlobalScope*>(asObject(value));
166     if (classInfo == JSProxy::info())
167         return jsDynamicCast<JSDedicatedWorkerGlobalScope*>(vm, jsCast<JSProxy*>(asObject(value))->target());
168     return nullptr;
169 }
170 
171 JSWorkerGlobalScope* toJSWorkerGlobalScope(VM& vm, JSValue value)
172 {
173     if (!value.isObject())
174         return nullptr;
175     const ClassInfo* classInfo = asObject(value)->classInfo(vm);
176     if (classInfo == JSDedicatedWorkerGlobalScope::info())
177         return jsCast<JSDedicatedWorkerGlobalScope*>(asObject(value));
178 
179 #if ENABLE(SERVICE_WORKER)
180     if (classInfo == JSServiceWorkerGlobalScope::info())
181         return jsCast<JSServiceWorkerGlobalScope*>(asObject(value));
182 #endif
183 
184     if (classInfo == JSProxy::info())
185         return jsDynamicCast<JSWorkerGlobalScope*>(vm, jsCast<JSProxy*>(asObject(value))->target());
186 
187     return nullptr;
188 }
