37 // This calls operationCreateDirectArguments but then restores the value of lengthGPR.
38 class CallCreateDirectArgumentsSlowPathGenerator : public JumpingSlowPathGenerator<MacroAssembler::JumpList> {
39 public:
40     CallCreateDirectArgumentsSlowPathGenerator(
41         MacroAssembler::JumpList from, SpeculativeJIT* jit, GPRReg resultGPR, RegisteredStructure structure,
42         GPRReg lengthGPR, unsigned minCapacity)
43         : JumpingSlowPathGenerator<MacroAssembler::JumpList>(from, jit)
44         , m_resultGPR(resultGPR)
45         , m_structure(structure)
46         , m_lengthGPR(lengthGPR)
47         , m_minCapacity(minCapacity)
48     {
49         jit->silentSpillAllRegistersImpl(false, m_plans, resultGPR);
50     }
51 
52 protected:
53     void generateInternal(SpeculativeJIT* jit) override
54     {
55         linkFrom(jit);
56         for (unsigned i = 0; i < m_plans.size(); ++i)
57             jit->silentSpill(m_plans[i]);
58         jit->callOperation(
59             operationCreateDirectArguments, m_resultGPR, &jit->vm(), m_structure, m_lengthGPR, m_minCapacity);
60         for (unsigned i = m_plans.size(); i--;)
61             jit->silentFill(m_plans[i]);
62         jit->m_jit.exceptionCheck();
63         jit->m_jit.loadPtr(
64             MacroAssembler::Address(m_resultGPR, DirectArguments::offsetOfLength()), m_lengthGPR);
65         jumpTo(jit);
66     }
67 
68 private:
69     GPRReg m_resultGPR;
70     RegisteredStructure m_structure;
71     GPRReg m_lengthGPR;
72     unsigned m_minCapacity;
73     Vector<SilentRegisterSavePlan, 2> m_plans;
74 };
