 71 JSObject* Instance::createRuntimeObject(ExecState* exec)
 72 {
 73     ASSERT(m_rootObject);
 74     ASSERT(m_rootObject->isValid());
 75     if (RuntimeObject* existingObject = m_runtimeObject.get())
 76         return existingObject;
 77 
 78     JSLockHolder lock(exec);
 79     RuntimeObject* newObject = newRuntimeObject(exec);
 80     m_runtimeObject = JSC::Weak<RuntimeObject>(newObject);
 81     m_rootObject->addRuntimeObject(exec->vm(), newObject);
 82     return newObject;
 83 }
 84 
 85 RuntimeObject* Instance::newRuntimeObject(ExecState* exec)
 86 {
 87     JSLockHolder lock(exec);
 88 
 89     // FIXME: deprecatedGetDOMStructure uses the prototype off of the wrong global object.
 90     return RuntimeObject::create(exec->vm(), WebCore::deprecatedGetDOMStructure<RuntimeObject>(exec), this);
 91 }
 92 
 93 void Instance::willInvalidateRuntimeObject()
 94 {
 95     m_runtimeObject.clear();
 96 }
 97 
 98 RootObject* Instance::rootObject() const
 99 {
100     return m_rootObject && m_rootObject->isValid() ? m_rootObject.get() : 0;
101 }
