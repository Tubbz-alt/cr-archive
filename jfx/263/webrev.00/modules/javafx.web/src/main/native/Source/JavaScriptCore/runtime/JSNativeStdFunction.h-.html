26 #pragma once
27 
28 #include "JSFunction.h"
29 
30 namespace JSC {
31 
32 class JSGlobalObject;
33 class NativeStdFunctionCell;
34 
35 using NativeStdFunction = WTF::Function<EncodedJSValue(ExecState*)>;
36 
37 class JSNativeStdFunction final : public JSFunction {
38 public:
39     typedef JSFunction Base;
40 
41     const static unsigned StructureFlags = Base::StructureFlags;
42 
43     template<typename CellType, SubspaceAccess mode>
44     static IsoSubspace* subspaceFor(VM& vm)
45     {
46         return vm.nativeStdFunctionSpace<mode>();
47     }
48 
49     DECLARE_EXPORT_INFO;
50 
51     JS_EXPORT_PRIVATE static JSNativeStdFunction* create(VM&, JSGlobalObject*, int length, const String& name, NativeStdFunction&&, Intrinsic = NoIntrinsic, NativeFunction nativeConstructor = callHostFunctionAsConstructor);
52 
53     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
54     {
55         ASSERT(globalObject);
56         return Structure::create(vm, globalObject, prototype, TypeInfo(JSFunctionType, StructureFlags), info());
57     }
58 
59     NativeStdFunctionCell* nativeStdFunctionCell() { return m_functionCell.get(); }
60 
61 protected:
62     static void visitChildren(JSCell*, SlotVisitor&);
63 
64     void finishCreation(VM&, NativeExecutable*, int length, const String& name, NativeStdFunctionCell*);
65 
66 private:
67     JSNativeStdFunction(VM&, JSGlobalObject*, Structure*);
68 
69     WriteBarrier<NativeStdFunctionCell> m_functionCell;
70 };
71 
72 } // namespace JSC
