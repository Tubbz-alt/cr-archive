34 inline SVGRectElement::SVGRectElement(const QualifiedName& tagName, Document& document)
35     : SVGGeometryElement(tagName, document)

36 {
37     ASSERT(hasTagName(SVGNames::rectTag));
38 
39     static std::once_flag onceFlag;
40     std::call_once(onceFlag, [] {
41         PropertyRegistry::registerProperty<SVGNames::xAttr, &SVGRectElement::m_x>();
42         PropertyRegistry::registerProperty<SVGNames::yAttr, &SVGRectElement::m_y>();
43         PropertyRegistry::registerProperty<SVGNames::widthAttr, &SVGRectElement::m_width>();
44         PropertyRegistry::registerProperty<SVGNames::heightAttr, &SVGRectElement::m_height>();
45         PropertyRegistry::registerProperty<SVGNames::rxAttr, &SVGRectElement::m_rx>();
46         PropertyRegistry::registerProperty<SVGNames::ryAttr, &SVGRectElement::m_ry>();
47     });
48 }

50 Ref<SVGRectElement> SVGRectElement::create(const QualifiedName& tagName, Document& document)
51 {
52     return adoptRef(*new SVGRectElement(tagName, document));
53 }
54 
55 void SVGRectElement::parseAttribute(const QualifiedName& name, const AtomString& value)
56 {
57     SVGParsingError parseError = NoError;
58 
59     if (name == SVGNames::xAttr)
60         m_x->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
61     else if (name == SVGNames::yAttr)
62         m_y->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
63     else if (name == SVGNames::rxAttr)
64         m_rx->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError, SVGLengthNegativeValuesMode::Forbid));
65     else if (name == SVGNames::ryAttr)
66         m_ry->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError, SVGLengthNegativeValuesMode::Forbid));
67     else if (name == SVGNames::widthAttr)
68         m_width->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError, SVGLengthNegativeValuesMode::Forbid));
69     else if (name == SVGNames::heightAttr)
70         m_height->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError, SVGLengthNegativeValuesMode::Forbid));
71 
72     reportAttributeParsingError(parseError, name, value);
73 
74     SVGGeometryElement::parseAttribute(name, value);

75 }
76 
77 void SVGRectElement::svgAttributeChanged(const QualifiedName& attrName)
78 {
79     if (PropertyRegistry::isKnownAttribute(attrName)) {
80         InstanceInvalidationGuard guard(*this);
81         invalidateSVGPresentationAttributeStyle();
82         return;
83     }
84 
85     SVGGeometryElement::svgAttributeChanged(attrName);

86 }

88 RenderPtr<RenderElement> SVGRectElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
89 {
90     return createRenderer<RenderSVGRect>(*this, WTFMove(style));
91 }
