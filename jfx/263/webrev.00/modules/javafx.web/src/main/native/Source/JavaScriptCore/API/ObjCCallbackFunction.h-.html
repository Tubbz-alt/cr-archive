42 namespace JSC {
43 
44 class ObjCCallbackFunctionImpl;
45 
46 class ObjCCallbackFunction : public InternalFunction {
47     friend struct APICallbackFunction;
48 public:
49     typedef InternalFunction Base;
50 
51     template<typename CellType, SubspaceAccess mode>
52     static IsoSubspace* subspaceFor(VM& vm)
53     {
54         return vm.objCCallbackFunctionSpace<mode>();
55     }
56 
57     static ObjCCallbackFunction* create(VM&, JSGlobalObject*, const String& name, std::unique_ptr<ObjCCallbackFunctionImpl>);
58     static void destroy(JSCell*);
59 
60     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
61     {
62         ASSERT(globalObject);
63         return Structure::create(vm, globalObject, prototype, TypeInfo(InternalFunctionType, StructureFlags), info());
64     }
65 
66     DECLARE_EXPORT_INFO;
67 
68     ObjCCallbackFunctionImpl* impl() const { return m_impl.get(); }
69 
70 protected:
71     ObjCCallbackFunction(VM&, Structure*, JSObjectCallAsFunctionCallback, JSObjectCallAsConstructorCallback, std::unique_ptr<ObjCCallbackFunctionImpl>);
72 
73 private:
74     JSObjectCallAsFunctionCallback functionCallback() { return m_functionCallback; }
75     JSObjectCallAsConstructorCallback constructCallback() { return m_constructCallback; }
76 
77     JSObjectCallAsFunctionCallback m_functionCallback;
78     JSObjectCallAsConstructorCallback m_constructCallback;
79     std::unique_ptr<ObjCCallbackFunctionImpl> m_impl;
80 };
81 
82 } // namespace JSC
