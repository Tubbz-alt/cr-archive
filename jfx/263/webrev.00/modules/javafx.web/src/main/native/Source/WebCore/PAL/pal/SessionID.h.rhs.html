 26 #pragma once
 27 
 28 #include <wtf/HashFunctions.h>
 29 #include <wtf/HashTraits.h>
 30 
 31 namespace PAL {
 32 
 33 class SessionID {
 34 public:
 35     SessionID() = delete;
 36 
 37     enum SessionConstants : uint64_t {
 38         EphemeralSessionMask    = 0x8000000000000000,
 39         DefaultSessionID        = 1,
 40         LegacyPrivateSessionID  = DefaultSessionID | EphemeralSessionMask,
 41         HashTableEmptyValueID   = 0,
 42         HashTableDeletedValueID = std::numeric_limits<uint64_t>::max(),
 43     };
 44 
 45     static SessionID defaultSessionID() { return SessionID(DefaultSessionID); }
 46     static SessionID legacyPrivateSessionID() { return SessionID(LegacyPrivateSessionID); }
 47 
 48     explicit SessionID(WTF::HashTableDeletedValueType)
 49         : m_identifier(HashTableDeletedValueID)
 50     {
 51     }
 52 
 53     explicit SessionID(WTF::HashTableEmptyValueType)
 54         : m_identifier(HashTableEmptyValueID)
 55     {
 56     }
 57 
 58     explicit SessionID(uint64_t identifier)
 59         : m_identifier(identifier)
 60     {
 61     }
 62 
 63     PAL_EXPORT static SessionID generateEphemeralSessionID();
 64     PAL_EXPORT static SessionID generatePersistentSessionID();
 65     PAL_EXPORT static void enableGenerationProtection();
 66 
 67     bool isValid() const { return isValidSessionIDValue(m_identifier); }
 68     bool isEphemeral() const { return m_identifier & EphemeralSessionMask && m_identifier != HashTableDeletedValueID; }
 69     bool isHashTableDeletedValue() const { return m_identifier == HashTableDeletedValueID; }
 70 
 71     uint64_t toUInt64() const { return m_identifier; }
 72     bool operator==(SessionID sessionID) const { return m_identifier == sessionID.m_identifier; }
 73     bool operator!=(SessionID sessionID) const { return m_identifier != sessionID.m_identifier; }
 74     bool isAlwaysOnLoggingAllowed() const { return !isEphemeral(); }
 75 
 76     template<class Encoder> void encode(Encoder&) const;
 77     template<class Decoder> static Optional<SessionID> decode(Decoder&);
 78 
 79     SessionID isolatedCopy() const { return *this; }
 80 
 81     explicit operator bool() const { return m_identifier; }
 82 
 83 private:





 84     static bool isValidSessionIDValue(uint64_t sessionID) { return sessionID != HashTableEmptyValueID && sessionID != HashTableDeletedValueID; }
 85 
 86     uint64_t m_identifier;
 87 };
 88 
 89 template<class Encoder>
 90 void SessionID::encode(Encoder& encoder) const
 91 {
 92     ASSERT(isValid());
 93     encoder << m_identifier;
 94 }
 95 
 96 template<class Decoder>
 97 Optional<SessionID> SessionID::decode(Decoder& decoder)
 98 {
 99     Optional<uint64_t> sessionID;
100     decoder >> sessionID;
101     if (!sessionID || !isValidSessionIDValue(*sessionID))
102         return WTF::nullopt;

103     return SessionID { *sessionID };
104 }
105 
106 } // namespace PAL
107 
108 namespace WTF {
109 
110 struct SessionIDHash {
111     static unsigned hash(const PAL::SessionID& p) { return intHash(p.toUInt64()); }
112     static bool equal(const PAL::SessionID& a, const PAL::SessionID& b) { return a == b; }
113     static const bool safeToCompareToEmptyOrDeleted = true;
114 };
115 
116 template<> struct HashTraits<PAL::SessionID> : GenericHashTraits<PAL::SessionID> {
117     static PAL::SessionID emptyValue() { return PAL::SessionID(HashTableEmptyValue); }
118     static void constructDeletedValue(PAL::SessionID& slot) { slot = PAL::SessionID(HashTableDeletedValue); }
119     static bool isDeletedValue(const PAL::SessionID& slot) { return slot.isHashTableDeletedValue(); }
120 };
121 
122 template<> struct DefaultHash<PAL::SessionID> {
123     typedef SessionIDHash Hash;
124 };
