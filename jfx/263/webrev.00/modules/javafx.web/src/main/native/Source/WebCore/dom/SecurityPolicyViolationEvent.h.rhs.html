 28 #include "Event.h"

 30 namespace WebCore {
 31 
 32 class SecurityPolicyViolationEvent final : public Event {
 33     WTF_MAKE_ISO_ALLOCATED(SecurityPolicyViolationEvent);
 34 public:
 35     static Ref<SecurityPolicyViolationEvent> create(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, const String& documentURI, const String& referrer, const String& blockedURI, const String& violatedDirective, const String& effectiveDirective, const String& originalPolicy, const String& sourceFile, unsigned short statusCode, int lineNumber, int columnNumber)
 36     {
 37         return adoptRef(*new SecurityPolicyViolationEvent(type, canBubble, cancelable, documentURI, referrer, blockedURI, violatedDirective, effectiveDirective, originalPolicy, sourceFile, statusCode, lineNumber, columnNumber));
 38     }
 39 
 40     static Ref<SecurityPolicyViolationEvent> createForBindings()
 41     {
 42         return adoptRef(*new SecurityPolicyViolationEvent());
 43     }
 44 
 45     struct Init : EventInit {
 46         String documentURI;
 47         String referrer;
 48         String blockedURI;
 49         String violatedDirective;
 50         String effectiveDirective;
 51         String originalPolicy;
 52         String sourceFile;
 53         unsigned short statusCode { 0 };
 54         int lineNumber { 0 };
 55         int columnNumber { 0 };
 56 
 57         template<class Encoder> void encode(Encoder&) const;
 58         template<class Decoder> static bool decode(Decoder&, Init&);
 59     };
 60 
 61     static Ref<SecurityPolicyViolationEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
 62     {
 63         return adoptRef(*new SecurityPolicyViolationEvent(type, initializer, isTrusted));
 64     }
 65 
 66     const String& documentURI() const { return m_documentURI; }
 67     const String& referrer() const { return m_referrer; }
 68     const String& blockedURI() const { return m_blockedURI; }
 69     const String& violatedDirective() const { return m_violatedDirective; }
 70     const String& effectiveDirective() const { return m_effectiveDirective; }
 71     const String& originalPolicy() const { return m_originalPolicy; }
 72     const String& sourceFile() const { return m_sourceFile; }
 73     unsigned short statusCode() const { return m_statusCode; }
 74     int lineNumber() const { return m_lineNumber; }
 75     int columnNumber() const { return m_columnNumber; }
 76 
 77     EventInterface eventInterface() const final { return SecurityPolicyViolationEventInterfaceType; }
 78 
 79 private:
 80     SecurityPolicyViolationEvent()
 81     {
 82     }
 83 
 84     SecurityPolicyViolationEvent(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, const String& documentURI, const String& referrer, const String& blockedURI, const String& violatedDirective, const String& effectiveDirective, const String& originalPolicy, const String& sourceFile, unsigned short statusCode, int lineNumber, int columnNumber)
 85         : Event(type, canBubble, cancelable)
 86         , m_documentURI(documentURI)
 87         , m_referrer(referrer)
 88         , m_blockedURI(blockedURI)
 89         , m_violatedDirective(violatedDirective)
 90         , m_effectiveDirective(effectiveDirective)
 91         , m_originalPolicy(originalPolicy)
 92         , m_sourceFile(sourceFile)
 93         , m_statusCode(statusCode)
 94         , m_lineNumber(lineNumber)
 95         , m_columnNumber(columnNumber)
 96     {
 97     }
 98 
 99     SecurityPolicyViolationEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
100         : Event(type, initializer, isTrusted)
101         , m_documentURI(initializer.documentURI)
102         , m_referrer(initializer.referrer)
103         , m_blockedURI(initializer.blockedURI)
104         , m_violatedDirective(initializer.violatedDirective)
105         , m_effectiveDirective(initializer.effectiveDirective)
106         , m_originalPolicy(initializer.originalPolicy)
107         , m_sourceFile(initializer.sourceFile)
108         , m_statusCode(initializer.statusCode)
109         , m_lineNumber(initializer.lineNumber)
110         , m_columnNumber(initializer.columnNumber)
111     {
112     }
113 
114     String m_documentURI;
115     String m_referrer;
116     String m_blockedURI;
117     String m_violatedDirective;
118     String m_effectiveDirective;
119     String m_originalPolicy;
120     String m_sourceFile;
121     unsigned short m_statusCode;
122     int m_lineNumber;
123     int m_columnNumber;
124 };
125 
126 template<class Encoder>
127 void SecurityPolicyViolationEvent::Init::encode(Encoder& encoder) const
128 {
129     encoder << static_cast<const EventInit&>(*this);
130     encoder << documentURI;
131     encoder << referrer;
132     encoder << blockedURI;
133     encoder << violatedDirective;
134     encoder << effectiveDirective;
135     encoder << originalPolicy;
136     encoder << sourceFile;
137     encoder << statusCode;
138     encoder << lineNumber;
139     encoder << columnNumber;
140 }
141 
142 template<class Decoder>
143 bool SecurityPolicyViolationEvent::Init::decode(Decoder& decoder, SecurityPolicyViolationEvent::Init& eventInit)
144 {
145     if (!decoder.decode(static_cast<EventInit&>(eventInit)))
146         return false;
147     if (!decoder.decode(eventInit.documentURI))
148         return false;
149     if (!decoder.decode(eventInit.referrer))
150         return false;
151     if (!decoder.decode(eventInit.blockedURI))
152         return false;
153     if (!decoder.decode(eventInit.violatedDirective))
154         return false;
155     if (!decoder.decode(eventInit.effectiveDirective))
156         return false;
157     if (!decoder.decode(eventInit.originalPolicy))
158         return false;
159     if (!decoder.decode(eventInit.sourceFile))
160         return false;
161     if (!decoder.decode(eventInit.statusCode))
162         return false;
163     if (!decoder.decode(eventInit.lineNumber))
164         return false;
165     if (!decoder.decode(eventInit.columnNumber))
166         return false;
167     return true;
168 }
