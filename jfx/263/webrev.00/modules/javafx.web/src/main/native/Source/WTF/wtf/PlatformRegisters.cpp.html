33 #if USE(PLATFORM_REGISTERS_WITH_PROFILE) && CPU(ARM64E)

35 #define USE_UNTAGGED_THREAD_STATE_PTR 1
36 
37 void* threadStateLRInternal(PlatformRegisters& regs)
38 {
39     void* candidateLR = arm_thread_state64_get_lr_fptr(regs);
40 
41 #if USE(UNTAGGED_THREAD_STATE_PTR)
42     if (candidateLR && isTaggedWith(candidateLR, CFunctionPtrTag))
43         return retagCodePtr<CFunctionPtrTag, PlatformRegistersLRPtrTag>(candidateLR);
44     candidateLR = bitwise_cast<void*>(arm_thread_state64_get_lr(regs));
45     if (!candidateLR)
46         return candidateLR;
47     return tagCodePtr<PlatformRegistersLRPtrTag>(candidateLR);
48 
49 #else
50     return retagCodePtr<CFunctionPtrTag, PlatformRegistersLRPtrTag>(candidateLR);
51 #endif
52 }
53 
54 void* threadStatePCInternal(PlatformRegisters& regs)
55 {
56     void* candidatePC = arm_thread_state64_get_pc_fptr(regs);
57 
58 #if USE(UNTAGGED_THREAD_STATE_PTR)
59     if (candidatePC && isTaggedWith(candidatePC, CFunctionPtrTag))
60         return retagCodePtr<CFunctionPtrTag, PlatformRegistersPCPtrTag>(candidatePC);
61     candidatePC = bitwise_cast<void*>(arm_thread_state64_get_pc(regs));
62     if (!candidatePC)
63         return candidatePC;
64     return tagCodePtr<PlatformRegistersPCPtrTag>(candidatePC);
65 
66 #else
67     return retagCodePtr<CFunctionPtrTag, PlatformRegistersPCPtrTag>(candidatePC);
68 #endif
69 }

71 #endif // USE(PLATFORM_REGISTERS_WITH_PROFILE) && CPU(ARM64E)
