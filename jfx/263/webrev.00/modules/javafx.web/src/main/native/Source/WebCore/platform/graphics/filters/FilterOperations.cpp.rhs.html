 37 bool FilterOperations::operator==(const FilterOperations& other) const
 38 {
 39     size_t size = m_operations.size();
 40     if (size != other.m_operations.size())
 41         return false;
 42     for (size_t i = 0; i < size; i++) {
 43         if (*m_operations[i] != *other.m_operations[i])
 44             return false;
 45     }
 46     return true;
 47 }
 48 
 49 bool FilterOperations::operationsMatch(const FilterOperations& other) const
 50 {
 51     size_t size = operations().size();
 52     if (size != other.operations().size())
 53         return false;
 54     for (size_t i = 0; i < size; ++i) {
 55         if (!operations()[i]->isSameType(*other.operations()[i]))
 56             return false;
 57     }
 58     return true;
 59 }

 61 bool FilterOperations::hasReferenceFilter() const
 62 {
 63     for (auto& operation : m_operations) {
 64         if (operation->type() == FilterOperation::REFERENCE)
 65             return true;
 66     }
 67     return false;
 68 }

 70 IntOutsets FilterOperations::outsets() const










 71 {
 72     IntOutsets totalOutsets;
 73     for (auto& operation : m_operations) {
 74         switch (operation->type()) {
 75         case FilterOperation::BLUR: {
 76             auto& blurOperation = downcast<BlurFilterOperation>(*operation);
 77             float stdDeviation = floatValueForLength(blurOperation.stdDeviation(), 0);
 78             IntSize outsetSize = FEGaussianBlur::calculateOutsetSize({ stdDeviation, stdDeviation });
 79             IntOutsets outsets(outsetSize.height(), outsetSize.width(), outsetSize.height(), outsetSize.width());
 80             totalOutsets += outsets;
 81             break;
 82         }
 83         case FilterOperation::DROP_SHADOW: {
 84             auto& dropShadowOperation = downcast<DropShadowFilterOperation>(*operation);
 85             float stdDeviation = dropShadowOperation.stdDeviation();
 86             IntSize outsetSize = FEGaussianBlur::calculateOutsetSize({ stdDeviation, stdDeviation });
 87             IntOutsets outsets {
 88                 std::max(0, outsetSize.height() - dropShadowOperation.y()),
 89                 std::max(0, outsetSize.width() + dropShadowOperation.x()),
 90                 std::max(0, outsetSize.height() + dropShadowOperation.y()),
 91                 std::max(0, outsetSize.width() - dropShadowOperation.x())
 92             };
 93             totalOutsets += outsets;
 94             break;
 95         }
 96         case FilterOperation::REFERENCE:
 97             ASSERT_NOT_REACHED();
 98             break;
 99         default:
100             break;
101         }
102     }
103     return totalOutsets;
104 }

106 bool FilterOperations::transformColor(Color& color) const
107 {
108     if (isEmpty() || !color.isValid())
109         return false;
110     // Color filter does not apply to semantic CSS colors (like "Windowframe").
111     if (color.isSemantic())
112         return false;
113 
114     FloatComponents components;
115     color.getRGBA(components.components[0], components.components[1], components.components[2], components.components[3]);
116 
117     for (auto& operation : m_operations) {
118         if (!operation->transformColor(components))
119             return false;
120     }
121 
122     color = Color(components.components[0], components.components[1], components.components[2], components.components[3]);
123     return true;
124 }

126 bool FilterOperations::inverseTransformColor(Color& color) const
127 {
128     if (isEmpty() || !color.isValid())
129         return false;
130     // Color filter does not apply to semantic CSS colors (like "Windowframe").
131     if (color.isSemantic())
132         return false;
133 
134     FloatComponents components;
135     color.getRGBA(components.components[0], components.components[1], components.components[2], components.components[3]);
136 
137     for (auto& operation : m_operations) {
138         if (!operation->inverseTransformColor(components))
139             return false;
140     }
141 
142     color = Color(components.components[0], components.components[1], components.components[2], components.components[3]);
143     return true;
144 }

173 TextStream& operator<<(TextStream& ts, const FilterOperations& filters)
174 {
175     for (size_t i = 0; i < filters.size(); ++i) {
176         auto filter = filters.at(i);
177         if (filter)
178             ts << *filter;
179         else
180             ts << "(null)";
181         if (i < filters.size() - 1)
182             ts << " ";
183     }
184     return ts;
185 }
