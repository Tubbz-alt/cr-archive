26 #pragma once
27 
28 #include "JSGlobalObject.h"
29 #include "JSObject.h"
30 #include "SourceCode.h"
31 
32 namespace JSC {
33 
34 class JSSourceCode final : public JSCell {
35 public:
36     using Base = JSCell;
37 
38     static const unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
39     static const bool needsDestruction = true;
40 
41     DECLARE_EXPORT_INFO;
42 
43     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
44     {
45         return Structure::create(vm, globalObject, prototype, TypeInfo(JSSourceCodeType, StructureFlags), info());
46     }
47 
48     static JSSourceCode* create(VM& vm, Structure* structure, SourceCode&& sourceCode)
49     {
50         auto* result = new (NotNull, allocateCell<JSSourceCode>(vm.heap)) JSSourceCode(vm, structure, WTFMove(sourceCode));
51         result->finishCreation(vm);
52         return result;
53     }
54 
55     static JSSourceCode* create(VM& vm, SourceCode&& sourceCode)
56     {
57         return create(vm, vm.sourceCodeStructure.get(), WTFMove(sourceCode));
58     }
59 
60     const SourceCode& sourceCode() const
61     {
62         return m_sourceCode;
63     }
64 
65     static void destroy(JSCell*);
66 
67 private:
68     JSSourceCode(VM& vm, Structure* structure, SourceCode&& sourceCode)
69         : Base(vm, structure)
70         , m_sourceCode(WTFMove(sourceCode))
71     {
72     }
73 
74     SourceCode m_sourceCode;
75 };
76 
77 } // namespace JSC
