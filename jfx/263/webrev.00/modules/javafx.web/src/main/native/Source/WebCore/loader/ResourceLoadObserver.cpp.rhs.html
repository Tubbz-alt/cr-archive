26 #include "config.h"
27 #include "ResourceLoadObserver.h"
28 #include <wtf/NeverDestroyed.h>

















29 
30 namespace WebCore {
31 
32 static ResourceLoadObserver*& sharedObserver()

























33 {
34     static ResourceLoadObserver* observer = nullptr;
35     return observer;
36 }
37 
38 void ResourceLoadObserver::setShared(ResourceLoadObserver& observer)
39 {
40     RELEASE_ASSERT(!sharedObserver());
41     sharedObserver() = &observer;
42 }
43 
44 ResourceLoadObserver& ResourceLoadObserver::shared()







































































































































































































































































































































45 {
46     static NeverDestroyed<ResourceLoadObserver> emptyObserver;
47     if (!sharedObserver())
48         return emptyObserver;
49     return *sharedObserver();
50 }
51 
52 ResourceLoadObserver* ResourceLoadObserver::sharedIfExists()
53 {
54     return sharedObserver();















55 }
