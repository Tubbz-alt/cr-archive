26 #ifndef ResourceLoadPriority_h
27 #define ResourceLoadPriority_h
28 
29 namespace WebCore {
30 
31 enum class ResourceLoadPriority : uint8_t {
32     VeryLow,
33     Low,
34     Medium,
35     High,
36     VeryHigh,
37     Lowest = VeryLow,
38     Highest = VeryHigh,
39 };


40 
41 static const unsigned resourceLoadPriorityCount { static_cast<unsigned>(ResourceLoadPriority::Highest) + 1 };
42 
43 inline ResourceLoadPriority& operator++(ResourceLoadPriority& priority)
44 {
45     ASSERT(priority != ResourceLoadPriority::Highest);
46     return priority = static_cast<ResourceLoadPriority>(static_cast<int>(priority) + 1);
47 }
48 
49 inline ResourceLoadPriority& operator--(ResourceLoadPriority& priority)
50 {
51     ASSERT(priority != ResourceLoadPriority::Lowest);
52     return priority = static_cast<ResourceLoadPriority>(static_cast<int>(priority) - 1);
53 }
54 
55 }
56 
57 #endif
