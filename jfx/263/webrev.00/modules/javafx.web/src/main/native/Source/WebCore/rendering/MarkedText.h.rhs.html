28 #include <wtf/Vector.h>
29 #include <wtf/text/WTFString.h>
30 
31 namespace WebCore {
32 
33 class RenderedDocumentMarker;
34 
35 struct MarkedText {
36     // Sorted by paint order
37     enum Type {
38         Unmarked,
39         GrammarError,
40         Correction,
41         SpellingError,
42         TextMatch,
43         DictationAlternatives,
44         Highlight,
45 #if PLATFORM(IOS_FAMILY)
46         // FIXME: See <rdar://problem/8933352>. Also, remove the PLATFORM(IOS_FAMILY)-guard.
47         DictationPhraseWithAlternatives,
48 #endif
49         Selection,
50         DraggedContent,
51     };
52     unsigned startOffset;
53     unsigned endOffset;
54     Type type;
55     const RenderedDocumentMarker* marker { nullptr };
56     String highlightName { };
57 
58     bool isEmpty() const { return endOffset <= startOffset; }
59     bool operator!=(const MarkedText& other) const { return !(*this == other); }
60     bool operator==(const MarkedText& other) const
61     {
62         return startOffset == other.startOffset && endOffset == other.endOffset && type == other.type && marker == other.marker && highlightName == other.highlightName;
63     }
64 };
65 
66 enum class OverlapStrategy { None, Frontmost };
67 WEBCORE_EXPORT Vector<MarkedText> subdivide(const Vector<MarkedText>&, OverlapStrategy = OverlapStrategy::None);
68 
69 }
