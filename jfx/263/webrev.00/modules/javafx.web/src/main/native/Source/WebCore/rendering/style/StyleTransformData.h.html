27 #include "Length.h"
28 #include "RenderStyleConstants.h"
29 #include "TransformOperations.h"
30 #include <wtf/Ref.h>
31 #include <wtf/RefCounted.h>

33 namespace WebCore {
34 
35 DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(StyleTransformData);
36 class StyleTransformData : public RefCounted<StyleTransformData> {
37     WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(StyleTransformData);
38 public:
39     static Ref<StyleTransformData> create() { return adoptRef(*new StyleTransformData); }
40     Ref<StyleTransformData> copy() const;
41 
42     bool operator==(const StyleTransformData&) const;
43     bool operator!=(const StyleTransformData& other) const
44     {
45         return !(*this == other);
46     }
47 
48     bool hasTransform() const { return operations.size(); }
49 
50     TransformOperations operations;
51     Length x;
52     Length y;
53     float z;
54     TransformBox transformBox;
55 
56 private:
57     StyleTransformData();
58     StyleTransformData(const StyleTransformData&);
59 };
60 
61 } // namespace WebCore
