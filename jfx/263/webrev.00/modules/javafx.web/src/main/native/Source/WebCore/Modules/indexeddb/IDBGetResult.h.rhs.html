 37 namespace WebCore {
 38 
 39 class IDBGetResult {
 40     WTF_MAKE_FAST_ALLOCATED;
 41 public:
 42     IDBGetResult()
 43         : m_isDefined(false)
 44     {
 45     }
 46 
 47     IDBGetResult(const IDBKeyData& keyData)
 48         : m_keyData(keyData)
 49     {
 50     }
 51 
 52     IDBGetResult(const IDBKeyData& keyData, const IDBKeyData& primaryKeyData)
 53         : m_keyData(keyData)
 54         , m_primaryKeyData(primaryKeyData)
 55     {
 56     }
 57 
 58     IDBGetResult(const IDBKeyData& keyData, const ThreadSafeDataBuffer& buffer, const Optional<IDBKeyPath>& keyPath)
 59         : m_value(buffer)
 60         , m_keyData(keyData)
 61         , m_keyPath(keyPath)
 62     {
 63     }
 64 
 65     IDBGetResult(const IDBKeyData& keyData, IDBValue&& value, const Optional<IDBKeyPath>& keyPath)
 66         : m_value(WTFMove(value))
 67         , m_keyData(keyData)
 68         , m_keyPath(keyPath)
 69     {
 70     }
 71 
 72     IDBGetResult(const IDBKeyData& keyData, const IDBKeyData& primaryKeyData, IDBValue&& value, const Optional<IDBKeyPath>& keyPath)
 73         : m_value(WTFMove(value))
 74         , m_keyData(keyData)
 75         , m_primaryKeyData(primaryKeyData)
 76         , m_keyPath(keyPath)
 77     {
 78     }
 79 
 80     IDBGetResult(const IDBKeyData& keyData, const IDBKeyData& primaryKeyData, IDBValue&& value, const Optional<IDBKeyPath>& keyPath, Vector<IDBCursorRecord>&& prefetechedRecords)
 81         : m_value(WTFMove(value))
 82         , m_keyData(keyData)
 83         , m_primaryKeyData(primaryKeyData)
 84         , m_keyPath(keyPath)
 85         , m_prefetchedRecords(WTFMove(prefetechedRecords))
 86     {
 87     }
 88 
 89     enum IsolatedCopyTag { IsolatedCopy };
 90     IDBGetResult(const IDBGetResult&, IsolatedCopyTag);
 91 
 92     IDBGetResult isolatedCopy() const;
 93 
 94     void setValue(IDBValue&&);
 95 
 96     const IDBValue& value() const { return m_value; }
 97     const IDBKeyData& keyData() const { return m_keyData; }
 98     const IDBKeyData& primaryKeyData() const { return m_primaryKeyData; }
 99     const Optional<IDBKeyPath>& keyPath() const { return m_keyPath; }
100     const Vector<IDBCursorRecord>& prefetchedRecords() const { return m_prefetchedRecords; }
101     bool isDefined() const { return m_isDefined; }
102 
103     template<class Encoder> void encode(Encoder&) const;
104     template<class Decoder> static bool decode(Decoder&, IDBGetResult&);
105 
106 private:
107     void dataFromBuffer(SharedBuffer&);
108 
109     static void isolatedCopy(const IDBGetResult& source, IDBGetResult& destination);
110 
111     IDBValue m_value;
112     IDBKeyData m_keyData;
113     IDBKeyData m_primaryKeyData;
114     Optional<IDBKeyPath> m_keyPath;
115     Vector<IDBCursorRecord> m_prefetchedRecords;
116     bool m_isDefined { true };
117 };
118 
119 template<class Encoder>
120 void IDBGetResult::encode(Encoder& encoder) const
121 {
122     encoder << m_keyData << m_primaryKeyData << m_keyPath << m_isDefined << m_value << m_prefetchedRecords;
123 }
124 
125 template<class Decoder>
126 bool IDBGetResult::decode(Decoder& decoder, IDBGetResult& result)
127 {
128     Optional<IDBKeyData> keyData;
129     decoder >> keyData;
130     if (!keyData)
131         return false;
132     result.m_keyData = WTFMove(*keyData);
133 
134     Optional<IDBKeyData> primaryKeyData;
135     decoder >> primaryKeyData;
136     if (!primaryKeyData)
137         return false;
138     result.m_primaryKeyData = WTFMove(*primaryKeyData);
139 
140     if (!decoder.decode(result.m_keyPath))
141         return false;
142 
143     if (!decoder.decode(result.m_isDefined))
144         return false;
145 
146     Optional<IDBValue> value;
147     decoder >> value;
148     if (!value)
149         return false;
150     result.m_value = WTFMove(*value);
151 
152     Optional<Vector<IDBCursorRecord>> prefetchedRecords;
153     decoder >> prefetchedRecords;
154     if (!prefetchedRecords)
155         return false;
156     result.m_prefetchedRecords = WTFMove(*prefetchedRecords);
157 
158     return true;
159 }
