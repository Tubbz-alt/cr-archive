44 ExceptionOr<void> CanvasGradient::addColorStop(float value, const String& colorString)
45 {
46     if (!(value >= 0 && value <= 1))
47         return Exception { IndexSizeError };
48 
49     // Passing null for canvas ensures that currentColor is treated as black,
50     // as required by the standard.
51     Color color = parseColorOrCurrentColor(colorString, nullptr /*canvas*/);
52     if (!color.isValid())
53         return Exception { SyntaxError };
54

56     return { };

47 ExceptionOr<void> CanvasGradient::addColorStop(float value, const String& colorString)
48 {
49     if (!(value >= 0 && value <= 1))
50         return Exception { IndexSizeError };
51 
52     // Treat currentColor as black, as required by the standard.
53     Color color = isCurrentColorString(colorString) ? Color::black : parseColor(colorString, m_canvas);

54     if (!color.isValid())
55         return Exception { SyntaxError };
56 
57     m_gradient->addColorStop(value, color);
58     return { };
59 }
