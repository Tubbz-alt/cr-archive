21 #pragma once
22 
23 #include "JSObject.h"
24 
25 namespace JSC {
26 
27 class DateInstance final : public JSNonFinalObject {
28 public:
29     using Base = JSNonFinalObject;
30 
31     static constexpr bool needsDestruction = true;
32     static void destroy(JSCell* cell)
33     {
34         static_cast<DateInstance*>(cell)->DateInstance::~DateInstance();
35     }
36 
37     template<typename CellType, SubspaceAccess mode>
38     static IsoSubspace* subspaceFor(VM& vm)
39     {
40         return &vm.dateInstanceSpace;
41     }
42 
43     static DateInstance* create(VM& vm, Structure* structure, double date)
44     {
45         DateInstance* instance = new (NotNull, allocateCell<DateInstance>(vm.heap)) DateInstance(vm, structure);
46         instance->finishCreation(vm, date);
47         return instance;
48     }
49 
50     static DateInstance* create(VM& vm, Structure* structure)
51     {
52         DateInstance* instance = new (NotNull, allocateCell<DateInstance>(vm.heap)) DateInstance(vm, structure);
53         instance->finishCreation(vm);
54         return instance;
55     }
56 
57     double internalNumber() const { return m_internalNumber; }
58     void setInternalNumber(double value) { m_internalNumber = value; }
59 
60     DECLARE_EXPORT_INFO;
61 
62     const GregorianDateTime* gregorianDateTime(VM& vm) const
63     {
64         if (m_data && m_data->m_gregorianDateTimeCachedForMS == internalNumber())
65             return &m_data->m_cachedGregorianDateTime;
66         return calculateGregorianDateTime(vm);
67     }
68 
69     const GregorianDateTime* gregorianDateTimeUTC(VM& vm) const
70     {
71         if (m_data && m_data->m_gregorianDateTimeUTCCachedForMS == internalNumber())
72             return &m_data->m_cachedGregorianDateTimeUTC;
73         return calculateGregorianDateTimeUTC(vm);
74     }
75 
76     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
77     {
78         return Structure::create(vm, globalObject, prototype, TypeInfo(JSDateType, StructureFlags), info());
79     }
80 
81     static ptrdiff_t offsetOfInternalNumber() { return OBJECT_OFFSETOF(DateInstance, m_internalNumber); }
82     static ptrdiff_t offsetOfData() { return OBJECT_OFFSETOF(DateInstance, m_data); }
83 
84 private:
85     JS_EXPORT_PRIVATE DateInstance(VM&, Structure*);
86     void finishCreation(VM&);
87     JS_EXPORT_PRIVATE void finishCreation(VM&, double);
88     JS_EXPORT_PRIVATE const GregorianDateTime* calculateGregorianDateTime(VM&) const;
89     JS_EXPORT_PRIVATE const GregorianDateTime* calculateGregorianDateTimeUTC(VM&) const;
90 
91     double m_internalNumber { PNaN };
92     mutable RefPtr<DateInstanceData> m_data;
93 };
94 
95 } // namespace JSC
