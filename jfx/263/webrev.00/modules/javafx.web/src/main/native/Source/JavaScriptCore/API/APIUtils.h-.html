26 #ifndef APIUtils_h
27 #define APIUtils_h
28 
29 #include "CatchScope.h"
30 #include "Exception.h"
31 #include "JSCJSValue.h"
32 #include "JSGlobalObjectInspectorController.h"
33 #include "JSValueRef.h"
34 
35 enum class ExceptionStatus {
36     DidThrow,
37     DidNotThrow
38 };
39 
40 inline ExceptionStatus handleExceptionIfNeeded(JSC::CatchScope& scope, JSC::ExecState* exec, JSValueRef* returnedExceptionRef)
41 {
42     if (UNLIKELY(scope.exception())) {
43         JSC::Exception* exception = scope.exception();
44         if (returnedExceptionRef)
45             *returnedExceptionRef = toRef(exec, exception->value());
46         scope.clearException();
47 #if ENABLE(REMOTE_INSPECTOR)
48         scope.vm().vmEntryGlobalObject(exec)->inspectorController().reportAPIException(exec, exception);
49 #endif
50         return ExceptionStatus::DidThrow;
51     }
52     return ExceptionStatus::DidNotThrow;
53 }
54 
55 inline void setException(JSC::ExecState* exec, JSValueRef* returnedExceptionRef, JSC::JSValue exception)
56 {
57     if (returnedExceptionRef)
58         *returnedExceptionRef = toRef(exec, exception);
59 #if ENABLE(REMOTE_INSPECTOR)
60     JSC::VM& vm = exec->vm();
61     vm.vmEntryGlobalObject(exec)->inspectorController().reportAPIException(exec, JSC::Exception::create(vm, exception));
62 #endif
63 }

65 #endif /* APIUtils_h */
