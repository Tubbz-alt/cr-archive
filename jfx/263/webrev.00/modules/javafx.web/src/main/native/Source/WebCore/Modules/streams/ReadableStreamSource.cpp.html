27 #include "config.h"

30 #include "JSDOMPromiseDeferred.h"

32 #if ENABLE(STREAMS_API)
33 
34 namespace WebCore {
35 
36 ReadableStreamSource::~ReadableStreamSource() = default;
37 
38 void ReadableStreamSource::start(ReadableStreamDefaultController&& controller, DOMPromiseDeferred<void>&& promise)
39 {
40     ASSERT(!m_promise);
41     m_promise = WTF::makeUnique<DOMPromiseDeferred<void>>(WTFMove(promise));
42     m_controller = WTFMove(controller);
43 
44     setActive();
45     doStart();
46 }
47 
48 void ReadableStreamSource::pull(DOMPromiseDeferred<void>&& promise)
49 {
50     ASSERT(!m_promise);
51     ASSERT(m_controller);
52 
53     m_promise = WTF::makeUnique<DOMPromiseDeferred<void>>(WTFMove(promise));
54 
55     setActive();
56     doPull();
57 }
58 
59 void ReadableStreamSource::startFinished()
60 {
61     ASSERT(m_promise);
62     m_promise->resolve();
63     m_promise = nullptr;
64     setInactive();
65 }
66 
67 void ReadableStreamSource::pullFinished()
68 {
69     ASSERT(m_promise);
70     m_promise->resolve();
71     m_promise = nullptr;
72     setInactive();
73 }
74 
75 void ReadableStreamSource::cancel(JSC::JSValue)
76 {
77     clean();
78     doCancel();
79 }
80 
81 void ReadableStreamSource::clean()
82 {
83     if (m_promise) {
84         m_promise = nullptr;
85         setInactive();
86     }
87 }
88 
89 } // namespace WebCore

91 #endif // ENABLE(STREAMS_API)
