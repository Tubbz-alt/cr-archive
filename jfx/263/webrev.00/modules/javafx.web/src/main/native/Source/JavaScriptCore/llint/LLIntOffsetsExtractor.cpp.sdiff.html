 85 const int64_t* LLIntOffsetsExtractor::dummy()
 86 {
 87 // This is a file generated by offlineasm/generate_offsets_extractor.rb, and contains code
 88 // to create a table of offsets, sizes, and a header identifying what combination of
 89 // Platform.h macros we have set. We include it inside of a method on LLIntOffsetsExtractor
 90 // because the fields whose offsets we're extracting are mostly private. So we make their
 91 // classes friends with LLIntOffsetsExtractor, and include the header here, to get the C++
