 28 
 29 #include "CSSImageGeneratorValue.h"
 30 #include "CachedImageClient.h"
 31 #include "CachedResourceHandle.h"
 32 #include "FilterOperations.h"
 33 #include "Image.h"
 34 #include <wtf/Function.h>
 35 
 36 namespace WebCore {
 37 
 38 class CachedImage;
 39 class FilterSubimageObserverProxy;
 40 class RenderElement;
 41 class Document;
 42 
 43 namespace Style {
 44 class BuilderState;
 45 }
 46 
 47 class CSSFilterImageValue final : public CSSImageGeneratorValue {
 48     friend class FilterSubimageObserverProxy;
 49 public:
 50     static Ref<CSSFilterImageValue> create(Ref<CSSValue>&& imageValue, Ref<CSSValue>&& filterValue)
 51     {
 52         return adoptRef(*new CSSFilterImageValue(WTFMove(imageValue), WTFMove(filterValue)));
 53     }
 54 
 55     ~CSSFilterImageValue();
 56 
 57     String customCSSText() const;
 58 
 59     RefPtr<Image> image(RenderElement&, const FloatSize&);
 60     bool isFixedSize() const { return true; }
 61     FloatSize fixedSize(const RenderElement&);
 62 
 63     bool isPending() const;
 64     bool knownToBeOpaque(const RenderElement&) const;
 65 
 66     void loadSubimages(CachedResourceLoader&, const ResourceLoaderOptions&);
 67 
 68     bool traverseSubresources(const WTF::Function<bool (const CachedResource&)>& handler) const;
 69 
 70     bool equals(const CSSFilterImageValue&) const;
 71 
 72     bool equalInputImages(const CSSFilterImageValue&) const;
 73 
 74     void createFilterOperations(Style::BuilderState&);
 75 
 76     const FilterOperations& filterOperations() const { return m_filterOperations; }
 77     void setFilterOperations(const FilterOperations& filterOperations)
 78     {
 79         m_filterOperations = filterOperations;
 80     }
 81     CachedImage* cachedImage() const { return m_cachedImage.get(); }
 82 
 83 private:
 84     CSSFilterImageValue(Ref<CSSValue>&& imageValue, Ref<CSSValue>&& filterValue)
 85         : CSSImageGeneratorValue(FilterImageClass)
 86         , m_imageValue(WTFMove(imageValue))
 87         , m_filterValue(WTFMove(filterValue))
 88         , m_filterSubimageObserver(this)
 89     {
 90     }
 91 
 92     class FilterSubimageObserverProxy final : public CachedImageClient {
 93     public:
 94         FilterSubimageObserverProxy(CSSFilterImageValue* ownerValue)
 95             : m_ownerValue(ownerValue)
 96             , m_ready(false)
 97         {
 98         }
 99 
100         virtual ~FilterSubimageObserverProxy() = default;
101         void imageChanged(CachedImage*, const IntRect* = nullptr) final;
102         void setReady(bool ready) { m_ready = ready; }
103     private:
104         CSSFilterImageValue* m_ownerValue;
105         bool m_ready;
106     };
107 
108     void filterImageChanged(const IntRect&);
109 
110     Ref<CSSValue> m_imageValue;
111     Ref<CSSValue> m_filterValue;
112 
113     FilterOperations m_filterOperations;
114 
115     CachedResourceHandle<CachedImage> m_cachedImage;
116 
117     FilterSubimageObserverProxy m_filterSubimageObserver;
118 };
119 
120 } // namespace WebCore
121 
122 SPECIALIZE_TYPE_TRAITS_CSS_VALUE(CSSFilterImageValue, isFilterImageValue())
