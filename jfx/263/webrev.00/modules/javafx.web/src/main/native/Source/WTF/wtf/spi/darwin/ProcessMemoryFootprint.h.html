35              && (PLATFORM(IOS_FAMILY) \
36                  || (PLATFORM(MAC) && __MAC_OS_X_VERSION_MIN_REQUIRED >= 101400))
37 #            define HAS_RESET_FOOTPRINT_INTERVAL
38 #            define MAX_FOOTPRINT_FIELD ri_interval_max_phys_footprint

48 struct ProcessMemoryFootprint {
49 public:
50     uint64_t current;
51     uint64_t peak;
52 
53     static ProcessMemoryFootprint now()
54     {
55 #ifdef HAS_MAX_FOOTPRINT
56         rusage_info_v4 rusage;
57         if (proc_pid_rusage(getpid(), RUSAGE_INFO_V4, (rusage_info_t *)&rusage))
58             return { 0L, 0L };
59 
60         return { rusage.ri_phys_footprint, rusage.MAX_FOOTPRINT_FIELD };
61 #elif defined(HAS_ONLY_PHYS_FOOTPRINT)
62         rusage_info_v0 rusage;
63         if (proc_pid_rusage(getpid(), RUSAGE_INFO_V0, (rusage_info_t *)&rusage))
64             return { 0L, 0L };
65 
66         return { rusage.ri_phys_footprint, 0L };
67 #else
68         return { 0L, 0L };
69 #endif
70     }
71 
72     static void resetPeak()
73     {
74 #ifdef HAS_RESET_FOOTPRINT_INTERVAL
75         proc_reset_footprint_interval(getpid());
76 #endif
77     }
78 };

80 #endif
