 50 inline JSErrorHandler::JSErrorHandler(JSObject& listener, JSObject& wrapper, bool isAttribute, DOMWrapperWorld& world)
 51     : JSEventListener(&listener, &wrapper, isAttribute, world)
 52 {

 55 Ref<JSErrorHandler> JSErrorHandler::create(JSC::JSObject& listener, JSC::JSObject& wrapper, bool isAttribute, DOMWrapperWorld& world)
 56 {
 57     return adoptRef(*new JSErrorHandler(listener, wrapper, isAttribute, world));
 58 }

 60 JSErrorHandler::~JSErrorHandler() = default;

 62 void JSErrorHandler::handleEvent(ScriptExecutionContext& scriptExecutionContext, Event& event)
 63 {
 64     if (!is<ErrorEvent>(event))
 65         return JSEventListener::handleEvent(scriptExecutionContext, event);
 66 
 67     VM& vm = scriptExecutionContext.vm();
 68     JSLockHolder lock(vm);
 69 
 70     JSObject* jsFunction = this->jsFunction(scriptExecutionContext);
 71     if (!jsFunction)
 72         return;
 73 
 74     auto* globalObject = toJSDOMGlobalObject(scriptExecutionContext, isolatedWorld());
 75     if (!globalObject)
 76         return;
 77 
 78     ExecState* exec = globalObject->globalExec();
 79 
 80     CallData callData;
 81     CallType callType = jsFunction->methodTable(vm)->getCallData(jsFunction, callData);
 82 
 83     if (callType != CallType::None) {
 84         Ref<JSErrorHandler> protectedThis(*this);
 85 
 86         Event* savedEvent = globalObject->currentEvent();
 87         globalObject->setCurrentEvent(&event);
 88 
 89         auto& errorEvent = downcast<ErrorEvent>(event);
 90 
 91         MarkedArgumentBuffer args;
 92         args.append(toJS<IDLDOMString>(*exec, errorEvent.message()));
 93         args.append(toJS<IDLUSVString>(*exec, errorEvent.filename()));
 94         args.append(toJS<IDLUnsignedLong>(errorEvent.lineno()));
 95         args.append(toJS<IDLUnsignedLong>(errorEvent.colno()));
 96         args.append(errorEvent.error(*exec, *globalObject));
 97         ASSERT(!args.hasOverflowed());
 98 
 99         VM& vm = globalObject->vm();
100         VMEntryScope entryScope(vm, vm.entryScope ? vm.entryScope->globalObject() : globalObject);
101 
102         InspectorInstrumentationCookie cookie = JSExecState::instrumentFunctionCall(&scriptExecutionContext, callType, callData);
103 
104         NakedPtr<JSC::Exception> exception;
105         JSValue returnValue = JSExecState::profiledCall(exec, JSC::ProfilingReason::Other, jsFunction, callType, callData, globalObject, args, exception);
106 
107         InspectorInstrumentation::didCallFunction(cookie, &scriptExecutionContext);
108 
109         globalObject->setCurrentEvent(savedEvent);
110 
111         if (exception)
112             reportException(exec, exception);
113         else {
114             if (returnValue.isTrue())
115                 event.preventDefault();
116         }
117     }
118 }
