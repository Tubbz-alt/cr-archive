 44 Ref<TextEvent> TextEvent::create(RefPtr<WindowProxy>&& view, const String& data, TextEventInputType inputType)
 45 {
 46     return adoptRef(*new TextEvent(WTFMove(view), data, inputType));

 49 Ref<TextEvent> TextEvent::createForPlainTextPaste(RefPtr<WindowProxy>&& view, const String& data, bool shouldSmartReplace)
 50 {
 51     return adoptRef(*new TextEvent(WTFMove(view), data, 0, shouldSmartReplace, false, MailBlockquoteHandling::RespectBlockquote));
 52 }

 55 {
 56     return adoptRef(*new TextEvent(WTFMove(view), emptyString(), WTFMove(data), shouldSmartReplace, shouldMatchStyle, mailBlockquoteHandling));
 57 }

 64 Ref<TextEvent> TextEvent::createForDictation(RefPtr<WindowProxy>&& view, const String& data, const Vector<DictationAlternative>& dictationAlternatives)

 77 TextEvent::TextEvent(RefPtr<WindowProxy>&& view, const String& data, TextEventInputType inputType)
 78     : UIEvent(eventNames().textInputEvent, CanBubble::Yes, IsCancelable::Yes, IsComposed::Yes, WTFMove(view), 0)

 80     , m_data(data)
 81     , m_shouldSmartReplace(false)
 82     , m_shouldMatchStyle(false)
 83     , m_mailBlockquoteHandling(MailBlockquoteHandling::RespectBlockquote)
 84 {
 85 }

 87 TextEvent::TextEvent(RefPtr<WindowProxy>&& view, const String& data, RefPtr<DocumentFragment>&& pastingFragment, bool shouldSmartReplace, bool shouldMatchStyle, MailBlockquoteHandling mailBlockquoteHandling)
 88     : UIEvent(eventNames().textInputEvent, CanBubble::Yes, IsCancelable::Yes, IsComposed::Yes, WTFMove(view), 0)
 89     , m_inputType(TextEventInputPaste)
 90     , m_data(data)
 91     , m_pastingFragment(WTFMove(pastingFragment))
 92     , m_shouldSmartReplace(shouldSmartReplace)
 93     , m_shouldMatchStyle(shouldMatchStyle)

 95 {
 96 }

 98 TextEvent::TextEvent(RefPtr<WindowProxy>&& view, const String& data, const Vector<DictationAlternative>& dictationAlternatives)

107 }

109 TextEvent::~TextEvent() = default;
110 
111 void TextEvent::initTextEvent(const AtomString& type, bool canBubble, bool cancelable, RefPtr<WindowProxy>&& view, const String& data)
112 {
113     if (isBeingDispatched())
114         return;
115 
116     initUIEvent(type, canBubble, cancelable, WTFMove(view), 0);
117 
118     m_inputType = TextEventInputKeyboard;
119 
120     m_data = data;
121 
122     m_pastingFragment = nullptr;
123     m_shouldSmartReplace = false;
124     m_shouldMatchStyle = false;
125     m_mailBlockquoteHandling = MailBlockquoteHandling::RespectBlockquote;
126     m_dictationAlternatives = { };
127 }

131     return TextEventInterfaceType;
