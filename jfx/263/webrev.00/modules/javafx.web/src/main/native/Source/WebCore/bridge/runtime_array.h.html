 26 #ifndef RUNTIME_ARRAY_H_
 27 #define RUNTIME_ARRAY_H_
 28 
 29 #include "BridgeJSC.h"
 30 #include "JSDOMBinding.h"
 31 #include <JavaScriptCore/ArrayPrototype.h>
 32 
 33 namespace JSC {
 34 
 35 class RuntimeArray final : public JSArray {
 36 public:
 37     using Base = JSArray;
 38     static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | InterceptsGetOwnPropertySlotByIndexEvenWhenLengthIsNotZero | OverridesGetPropertyNames;
 39     static constexpr bool needsDestruction = true;
 40 
 41     template<typename CellType, JSC::SubspaceAccess>
 42     static JSC::IsoSubspace* subspaceFor(JSC::VM& vm)
 43     {
 44         return subspaceForImpl(vm);
 45     }
 46 
 47     static RuntimeArray* create(JSGlobalObject* lexicalGlobalObject, Bindings::Array* array)
 48     {
 49         VM& vm = lexicalGlobalObject->vm();
 50         // FIXME: deprecatedGetDOMStructure uses the prototype off of the wrong global object
 51         // We need to pass in the right global object for "array".
 52         Structure* domStructure = WebCore::deprecatedGetDOMStructure<RuntimeArray>(lexicalGlobalObject);
 53         RuntimeArray* runtimeArray = new (NotNull, allocateCell<RuntimeArray>(vm.heap)) RuntimeArray(vm, domStructure);
 54         runtimeArray->finishCreation(vm, array);
 55         return runtimeArray;
 56     }
 57 
 58     typedef Bindings::Array BindingsArray;
 59     ~RuntimeArray();
 60     static void destroy(JSCell*);
 61 
 62     static void getOwnPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, EnumerationMode);
 63     static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
 64     static bool getOwnPropertySlotByIndex(JSObject*, JSGlobalObject*, unsigned, PropertySlot&);
 65     static bool put(JSCell*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&);
 66     static bool putByIndex(JSCell*, JSGlobalObject*, unsigned propertyName, JSValue, bool shouldThrow);
 67 
 68     static bool deleteProperty(JSCell*, JSGlobalObject*, PropertyName);
 69     static bool deletePropertyByIndex(JSCell*, JSGlobalObject*, unsigned propertyName);
 70 
 71     unsigned getLength() const { return m_array->getLength(); }
 72 
 73     Bindings::Array* getConcreteArray() const { return m_array; }
 74 
 75     DECLARE_INFO;
 76 
 77     static ArrayPrototype* createPrototype(VM&, JSGlobalObject& globalObject)
 78     {
 79         return globalObject.arrayPrototype();
 80     }
 81 
 82     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
 83     {
 84         return Structure::create(vm, globalObject, prototype, TypeInfo(DerivedArrayType, StructureFlags), info(), ArrayClass);
 85     }
 86 
 87 protected:
 88     void finishCreation(VM&, Bindings::Array*);
 89 
 90 private:
 91     RuntimeArray(VM&, Structure*);
 92     static EncodedJSValue lengthGetter(JSGlobalObject*, EncodedJSValue, PropertyName);
 93     static JSC::IsoSubspace* subspaceForImpl(JSC::VM&);
 94 
 95     BindingsArray* m_array;
 96 };
 97 
 98 } // namespace JSC
 99 
100 #endif // RUNTIME_ARRAY_H_
