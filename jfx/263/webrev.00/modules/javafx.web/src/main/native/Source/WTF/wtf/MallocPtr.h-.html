 26 #pragma once
 27 
 28 #include <wtf/FastMalloc.h>
 29 
 30 // MallocPtr is a smart pointer class that calls fastFree in its destructor.
 31 // It is intended to be used for pointers where the C++ lifetime semantics
 32 // (calling constructors and destructors) is not desired.
 33 
 34 namespace WTF {
 35 
 36 template<typename T, typename Malloc = FastMalloc> class MallocPtr {
 37 public:
 38     MallocPtr()
 39         : m_ptr(nullptr)
 40     {
 41     }
 42 
 43     MallocPtr(std::nullptr_t)
 44         : m_ptr(nullptr)
 45     {
 46     }
 47 
 48     MallocPtr(MallocPtr&& other)
 49         : m_ptr(other.leakPtr())
 50     {
 51     }
 52 
 53     ~MallocPtr()
 54     {
 55         Malloc::free(m_ptr);
 56     }
 57 
 58     T* get() const
 59     {
 60         return m_ptr;
 61     }
 62 
 63     T *leakPtr() WARN_UNUSED_RETURN
 64     {
 65         return std::exchange(m_ptr, nullptr);
 66     }
 67 
 68     explicit operator bool() const
 69     {
 70         return m_ptr;
 71     }
 72 
 73     bool operator!() const
 74     {
 75         return !m_ptr;
 76     }
 77 
 78     T& operator*() const
 79     {
 80         ASSERT(m_ptr);
 81         return *m_ptr;
 82     }
 83 
 84     T* operator->() const
 85     {
 86         return m_ptr;
 87     }
 88 
 89     MallocPtr& operator=(MallocPtr&& other)
 90     {
 91         MallocPtr ptr = WTFMove(other);
 92         swap(ptr);
 93 
 94         return *this;
 95     }
 96 
 97     void swap(MallocPtr& other)
 98     {
 99         std::swap(m_ptr, other.m_ptr);
100     }
101 
102     template<typename U> friend MallocPtr<U> adoptMallocPtr(U*);
103 
104     static MallocPtr malloc(size_t size)
105     {
106         return MallocPtr { static_cast<T*>(Malloc::malloc(size)) };
107     }
108 
109     static MallocPtr tryMalloc(size_t size)
110     {
111         return MallocPtr { static_cast<T*>(Malloc::tryMalloc(size)) };
112     }
113 
114     void realloc(size_t newSize)
115     {
116         m_ptr = static_cast<T*>(Malloc::realloc(m_ptr, newSize));
117     }
118 
119 private:
120     explicit MallocPtr(T* ptr)
121         : m_ptr(ptr)
122     {
123     }
124 
125     T* m_ptr;
126 };
127 
128 static_assert(sizeof(MallocPtr<int>) == sizeof(int*), "");
129 
130 template<typename U> MallocPtr<U> adoptMallocPtr(U* ptr)
131 {
132     return MallocPtr<U>(ptr);
133 }
134 
135 } // namespace WTF
136 
137 using WTF::MallocPtr;
138 using WTF::adoptMallocPtr;
