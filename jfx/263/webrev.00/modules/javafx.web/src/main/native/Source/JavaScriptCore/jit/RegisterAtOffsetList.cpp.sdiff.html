37 RegisterAtOffsetList::RegisterAtOffsetList(RegisterSet registerSet, OffsetBaseType offsetBaseType)
38 {
39     size_t numberOfRegisters = registerSet.numberOfSetRegisters();
40     ptrdiff_t offset = 0;
41 
42     if (offsetBaseType == FramePointerBased)
43         offset = -(static_cast<ptrdiff_t>(numberOfRegisters) * sizeof(CPURegister));
44 
45     m_registers.reserveInitialCapacity(numberOfRegisters);
46     registerSet.forEach([&] (Reg reg) {
47         m_registers.append(RegisterAtOffset(reg, offset));
48         offset += sizeof(CPURegister);
49     });
50 }
