33 inline WeakImpl* WeakSet::allocate(JSValue jsValue, WeakHandleOwner* weakHandleOwner, void* context)
34 {
35     WeakSet& weakSet = jsValue.asCell()->cellContainer().weakSet();

36     WeakBlock::FreeCell* allocator = weakSet.m_allocator;
37     if (UNLIKELY(!allocator))
38         allocator = weakSet.findAllocator();
39     weakSet.m_allocator = allocator->next;
40 
41     WeakImpl* weakImpl = WeakBlock::asWeakImpl(allocator);
42     return new (NotNull, weakImpl) WeakImpl(jsValue, weakHandleOwner, context);
43 }

33 inline WeakImpl* WeakSet::allocate(JSValue jsValue, WeakHandleOwner* weakHandleOwner, void* context)
34 {
35     CellContainer container = jsValue.asCell()->cellContainer();
36     WeakSet& weakSet = container.weakSet();
37     WeakBlock::FreeCell* allocator = weakSet.m_allocator;
38     if (UNLIKELY(!allocator))
39         allocator = weakSet.findAllocator(container);
40     weakSet.m_allocator = allocator->next;
41 
42     WeakImpl* weakImpl = WeakBlock::asWeakImpl(allocator);
43     return new (NotNull, weakImpl) WeakImpl(jsValue, weakHandleOwner, context);
44 }
