39 const unsigned MaxContexts = 16;
40 
41 class GraphicsContextGLOpenGL;

44 using PlatformDisplayID = uint32_t;

46 #if HAVE(APPLE_GRAPHICS_CONTROL)
47 WEBCORE_EXPORT bool hasLowAndHighPowerGPUs();
48 #endif

50 class GraphicsContextGLOpenGLManager {
51     friend NeverDestroyed<GraphicsContextGLOpenGLManager>;
52 public:
53     static GraphicsContextGLOpenGLManager& sharedManager();
54 
55     void addContext(GraphicsContextGLOpenGL*, HostWindow*);
56     void removeContext(GraphicsContextGLOpenGL*);
57 
58     HostWindow* hostWindowForContext(GraphicsContextGLOpenGL*) const;
59 
60     void addContextRequiringHighPerformance(GraphicsContextGLOpenGL*);
61     void removeContextRequiringHighPerformance(GraphicsContextGLOpenGL*);
62 
63     void recycleContextIfNecessary();
64     bool hasTooManyContexts() const { return m_contexts.size() >= MaxContexts; }
65 
66     void updateAllContexts();
67 
68 #if PLATFORM(MAC)
69     void screenDidChange(PlatformDisplayID, const HostWindow*);
70     WEBCORE_EXPORT static void displayWasReconfigured(CGDirectDisplayID, CGDisplayChangeSummaryFlags, void*);
71 #endif
72 
73 private:
74     GraphicsContextGLOpenGLManager()
75         : m_disableHighPerformanceGPUTimer(*this, &GraphicsContextGLOpenGLManager::disableHighPerformanceGPUTimerFired)
76     {
77     }
78 
79     void updateHighPerformanceState();
80     void disableHighPerformanceGPUTimerFired();
81 
82     Vector<GraphicsContextGLOpenGL*> m_contexts;
83     HashMap<GraphicsContextGLOpenGL*, HostWindow*> m_contextWindowMap;
84     HashSet<GraphicsContextGLOpenGL*> m_contextsRequiringHighPerformance;
85 
86     Timer m_disableHighPerformanceGPUTimer;
87     bool m_requestingHighPerformance { false };
88 };
