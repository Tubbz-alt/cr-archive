33 ScrollLatchingState::ScrollLatchingState() = default;
34 
35 ScrollLatchingState::~ScrollLatchingState() = default;

37 void ScrollLatchingState::clear()
38 {
39     m_wheelEventElement = nullptr;
40     m_frame = nullptr;
41     m_scrollableContainer = nullptr;
42     m_widgetIsLatched = false;
43     m_previousWheelScrolledElement = nullptr;
44 }
45 
46 void ScrollLatchingState::setWheelEventElement(Element* element)
47 {
48     m_wheelEventElement = element;
49 }
50 
51 void ScrollLatchingState::setWidgetIsLatched(bool isOverWidget)
52 {
53     m_widgetIsLatched = isOverWidget;
54 }
55 
56 void ScrollLatchingState::setPreviousWheelScrolledElement(Element* element)
57 {
58     m_previousWheelScrolledElement = element;
59 }
60 
61 void ScrollLatchingState::setScrollableContainer(ContainerNode* container)
62 {
63     m_scrollableContainer = container;
64 }
