 42 JSWebAssemblyCodeBlock* JSWebAssemblyCodeBlock::create(VM& vm, Ref<Wasm::CodeBlock> codeBlock, const Wasm::ModuleInformation& moduleInformation)
 43 {
 44     auto* result = new (NotNull, allocateCell<JSWebAssemblyCodeBlock>(vm.heap)) JSWebAssemblyCodeBlock(vm, WTFMove(codeBlock), moduleInformation);
 45     result->finishCreation(vm);
 46     return result;
 47 }
 48 
 49 JSWebAssemblyCodeBlock::JSWebAssemblyCodeBlock(VM& vm, Ref<Wasm::CodeBlock>&& codeBlock, const Wasm::ModuleInformation& moduleInformation)
 50     : Base(vm, vm.webAssemblyCodeBlockStructure.get())
 51     , m_codeBlock(WTFMove(codeBlock))
 52 {
 53     // FIXME: We should not need to do this synchronously.
 54     // https://bugs.webkit.org/show_bug.cgi?id=170567
 55     m_wasmToJSExitStubs.reserveCapacity(m_codeBlock->functionImportCount());
 56     for (unsigned importIndex = 0; importIndex < m_codeBlock->functionImportCount(); ++importIndex) {
 57         Wasm::SignatureIndex signatureIndex = moduleInformation.importFunctionSignatureIndices.at(importIndex);
 58         auto binding = Wasm::wasmToJS(vm, m_callLinkInfos, signatureIndex, importIndex);
 59         if (UNLIKELY(!binding)) {
 60             switch (binding.error()) {
 61             case Wasm::BindingFailure::OutOfMemory:
 62                 m_errorMessage = "Out of executable memory"_s;
 63                 return;
 64             }
 65             RELEASE_ASSERT_NOT_REACHED();
 66         }
 67         m_wasmToJSExitStubs.uncheckedAppend(binding.value());
 68     }
 69 }
 70 
 71 void JSWebAssemblyCodeBlock::finishCreation(VM& vm)
 72 {
 73     Base::finishCreation(vm);
 74 }
 75 
 76 void JSWebAssemblyCodeBlock::destroy(JSCell* cell)
 77 {
 78     static_cast<JSWebAssemblyCodeBlock*>(cell)->JSWebAssemblyCodeBlock::~JSWebAssemblyCodeBlock();
 79 }
 80 
 81 void JSWebAssemblyCodeBlock::clearJSCallICs(VM& vm)
 82 {
 83     for (auto iter = m_callLinkInfos.begin(); !!iter; ++iter)
 84         (*iter)->unlink(vm);
 85 }
 86 
 87 void JSWebAssemblyCodeBlock::visitChildren(JSCell* cell, SlotVisitor& visitor)
 88 {
 89     JSWebAssemblyCodeBlock* thisObject = jsCast<JSWebAssemblyCodeBlock*>(cell);
 90     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
 91 
 92     Base::visitChildren(thisObject, visitor);
 93 }
 94 
 95 void JSWebAssemblyCodeBlock::finalizeUnconditionally(VM& vm)
 96 {
 97     for (auto iter = m_callLinkInfos.begin(); !!iter; ++iter)
 98         (*iter)->visitWeak(vm);
 99 }
