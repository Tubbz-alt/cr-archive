 52 DOMWindow* domWindowFromExecState(JSC::ExecState* scriptState)
 53 {
 54     JSC::JSGlobalObject* globalObject = scriptState->lexicalGlobalObject();
 55     JSC::VM& vm = globalObject->vm();
 56     if (!globalObject->inherits<JSDOMWindowBase>(vm))
 57         return nullptr;
 58     return &JSC::jsCast<JSDOMWindowBase*>(globalObject)->wrapped();
 59 }
 60 
 61 Frame* frameFromExecState(JSC::ExecState* scriptState)
 62 {
 63     ScriptExecutionContext* context = scriptExecutionContextFromExecState(scriptState);
 64     Document* document = is<Document>(context) ? downcast<Document>(context) : nullptr;
 65     return document ? document->frame() : nullptr;
 66 }
 67 
 68 ScriptExecutionContext* scriptExecutionContextFromExecState(JSC::ExecState* scriptState)
 69 {
 70     JSC::JSGlobalObject* globalObject = scriptState->lexicalGlobalObject();
 71     JSC::VM& vm = globalObject->vm();
 72     if (!globalObject->inherits<JSDOMGlobalObject>(vm))
 73         return nullptr;
 74     return JSC::jsCast<JSDOMGlobalObject*>(globalObject)->scriptExecutionContext();
 75 }
 76 
 77 JSC::ExecState* mainWorldExecState(Frame* frame)
 78 {
 79     if (!frame)
 80         return nullptr;
 81     return frame->windowProxy().jsWindowProxy(mainThreadNormalWorld())->window()->globalExec();
 82 }
 83 
 84 JSC::ExecState* execStateFromNode(DOMWrapperWorld& world, Node* node)
 85 {
 86     if (!node)
 87         return nullptr;
 88     Frame* frame = node->document().frame();
 89     if (!frame)
 90         return nullptr;
 91     if (!frame->script().canExecuteScripts(NotAboutToExecuteScript))
 92         return nullptr;
 93     return frame->script().globalObject(world)->globalExec();
 94 }
 95 
 96 JSC::ExecState* execStateFromPage(DOMWrapperWorld& world, Page* page)
 97 {
 98     return page ? page->mainFrame().script().globalObject(world)->globalExec() : nullptr;
 99 }
100 
101 JSC::ExecState* execStateFromWorkerGlobalScope(WorkerGlobalScope& workerGlobalScope)
102 {
103     return workerGlobalScope.script()->workerGlobalScopeWrapper()->globalExec();
104 }
