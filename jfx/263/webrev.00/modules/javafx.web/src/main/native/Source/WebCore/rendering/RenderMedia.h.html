27 
28 #if ENABLE(VIDEO)
29 
30 #include "HTMLMediaElement.h"
31 #include "RenderImage.h"
32 
33 namespace WebCore {
34 
35 class RenderMedia : public RenderImage {
36     WTF_MAKE_ISO_ALLOCATED(RenderMedia);
37 public:
38     RenderMedia(HTMLMediaElement&, RenderStyle&&);
39     RenderMedia(HTMLMediaElement&, RenderStyle&&, const IntSize& intrinsicSize);
40     virtual ~RenderMedia();
41 
42     HTMLMediaElement& mediaElement() const { return downcast<HTMLMediaElement>(nodeForNonAnonymous()); }
43 
44     bool shouldDisplayBrokenImageIcon() const final { return false; }
45 
46 protected:
47     void layout() override;
48 
49 private:
50     void element() const = delete;
51 
52     bool canHaveChildren() const final { return true; }
53 
54     const char* renderName() const override { return "RenderMedia"; }
55     bool isMedia() const final { return true; }
56     bool isImage() const final { return false; }
57     void paintReplaced(PaintInfo&, const LayoutPoint&) override;
58 
59     bool shadowControlsNeedCustomLayoutMetrics() const override { return true; }
60     void styleDidChange(StyleDifference, const RenderStyle* oldStyle) final;
61 };
62 
63 inline RenderMedia* HTMLMediaElement::renderer() const
64 {
65     return downcast<RenderMedia>(HTMLElement::renderer());
66 }
67 
68 } // namespace WebCore
69 
70 SPECIALIZE_TYPE_TRAITS_RENDER_OBJECT(RenderMedia, isMedia())
