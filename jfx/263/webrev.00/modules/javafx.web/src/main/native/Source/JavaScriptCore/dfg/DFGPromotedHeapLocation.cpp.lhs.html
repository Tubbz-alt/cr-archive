 36 void PromotedLocationDescriptor::dump(PrintStream& out) const
 37 {
 38     out.print(m_kind, "(", m_info, ")");
 39 }
 40 
 41 Node* PromotedHeapLocation::createHint(Graph& graph, NodeOrigin origin, Node* value)
 42 {
 43     return graph.addNode(
 44         SpecNone, PutHint, origin, OpInfo(descriptor().imm1()), OpInfo(descriptor().imm2()),
 45         base()->defaultEdge(), value->defaultEdge());
 46 }

 59 void printInternal(PrintStream& out, PromotedLocationKind kind)
 60 {
 61     switch (kind) {
 62     case InvalidPromotedLocationKind:
 63         out.print("InvalidPromotedLocationKind");
 64         return;
 65 
 66     case StructurePLoc:
 67         out.print("StructurePLoc");
 68         return;
 69 
 70     case ActivationSymbolTablePLoc:
 71         out.print("ActivationSymbolTablePLoc");
 72         return;
 73 
 74     case NamedPropertyPLoc:
 75         out.print("NamedPropertyPLoc");
 76         return;
 77 
 78     case IndexedPropertyPLoc:
 79         out.print("IndexedPropertyPLoc");
 80         return;
 81 
 82     case ArgumentPLoc:
 83         out.print("ArgumentPLoc");
 84         return;
 85 
 86     case ArgumentCountPLoc:
 87         out.print("ArgumentCountPLoc");
 88         return;
 89 
 90     case ArgumentsCalleePLoc:
 91         out.print("ArgumentsCalleePLoc");
 92         return;
 93 
 94     case FunctionExecutablePLoc:
 95         out.print("FunctionExecutablePLoc");
 96         return;
 97 
 98     case FunctionActivationPLoc:
 99         out.print("FunctionActivationPLoc");
100         return;
101 
102     case ActivationScopePLoc:
103         out.print("ActivationScopePLoc");
104         return;
105 
106     case ClosureVarPLoc:
107         out.print("ClosureVarPLoc");
108         return;
109 
110     case PublicLengthPLoc:
111         out.print("PublicLengthPLoc");
112         return;
113 
114     case VectorLengthPLoc:
115         out.print("VectorLengthPLoc");
116         return;
117 
118     case SpreadPLoc:
119         out.print("SpreadPLoc");
120         return;
121 
122     case NewArrayWithSpreadArgumentPLoc:
123         out.print("NewArrayWithSpreadArgumentPLoc");
124         return;
125 
126     case NewArrayBufferPLoc:
127         out.print("NewArrayBufferPLoc");
128         return;
129 
130     case RegExpObjectRegExpPLoc:
131         out.print("RegExpObjectRegExpPLoc");
132         return;
133 
134     case RegExpObjectLastIndexPLoc:
135         out.print("RegExpObjectLastIndexPLoc");
136         return;




137     }
138 
139     RELEASE_ASSERT_NOT_REACHED();
140 }
