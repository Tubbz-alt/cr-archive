 68 Rect Box::marginBox() const
 69 {
 70     auto borderBox = this->borderBox();
 71 
 72     Rect marginBox;
 73     marginBox.setTop(borderBox.top() - marginBefore());
 74     marginBox.setLeft(borderBox.left() - marginStart());
 75     marginBox.setHeight(borderBox.height() + marginBefore() + marginAfter());
 76     marginBox.setWidth(borderBox.width() + marginStart() + marginEnd());
 77     return marginBox;
 78 }
 79 
 80 Rect Box::nonCollapsedMarginBox() const
 81 {
 82     auto borderBox = this->borderBox();
 83 
 84     Rect marginBox;
 85     marginBox.setTop(borderBox.top() - nonCollapsedMarginBefore());
 86     marginBox.setLeft(borderBox.left() - marginStart());
 87     marginBox.setHeight(borderBox.height() + nonCollapsedMarginBefore() + nonCollapsedMarginAfter());
 88     marginBox.setWidth(borderBox.width() + marginStart() + marginEnd());
 89     return marginBox;
 90 }
 91 
 92 Rect Box::borderBox() const
 93 {
 94     Rect borderBox;
 95     borderBox.setTopLeft({ });
 96     borderBox.setSize({ width(), height() });
 97     return borderBox;
 98 }
 99 
100 Rect Box::paddingBox() const
101 {
102     auto borderBox = this->borderBox();
103 
104     Rect paddingBox;
105     paddingBox.setTop(borderBox.top() + borderTop());
106     paddingBox.setLeft(borderBox.left() + borderLeft());
107     paddingBox.setHeight(borderBox.bottom() - borderTop() - borderBottom());
108     paddingBox.setWidth(borderBox.width() - borderLeft() - borderRight());
109     return paddingBox;
110 }
111 
112 Rect Box::contentBox() const
113 {
114     Rect contentBox;
115     contentBox.setTop(contentBoxTop());
116     contentBox.setLeft(contentBoxLeft());
117     contentBox.setWidth(contentBoxWidth());
118     contentBox.setHeight(contentBoxHeight());
119     return contentBox;
120 }
