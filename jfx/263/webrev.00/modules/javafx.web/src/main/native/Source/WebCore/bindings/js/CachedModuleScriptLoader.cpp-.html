42 Ref<CachedModuleScriptLoader> CachedModuleScriptLoader::create(CachedModuleScriptLoaderClient& client, DeferredPromise& promise, CachedScriptFetcher& scriptFetcher, RefPtr<ModuleFetchParameters>&& parameters)
43 {
44     return adoptRef(*new CachedModuleScriptLoader(client, promise, scriptFetcher, WTFMove(parameters)));
45 }
46 
47 CachedModuleScriptLoader::CachedModuleScriptLoader(CachedModuleScriptLoaderClient& client, DeferredPromise& promise, CachedScriptFetcher& scriptFetcher, RefPtr<ModuleFetchParameters>&& parameters)
48     : m_client(&client)
49     , m_promise(&promise)
50     , m_scriptFetcher(scriptFetcher)
51     , m_parameters(WTFMove(parameters))
52 {
53 }
54 
55 CachedModuleScriptLoader::~CachedModuleScriptLoader()
56 {
57     if (m_cachedScript) {
58         m_cachedScript->removeClient(*this);
59         m_cachedScript = nullptr;
60     }
61 }
62 
63 bool CachedModuleScriptLoader::load(Document& document, const URL& sourceURL)
64 {
65     ASSERT(!m_cachedScript);
66     String integrity = m_parameters ? m_parameters->integrity() : String { };
67     m_cachedScript = m_scriptFetcher->requestModuleScript(document, sourceURL, WTFMove(integrity));
68     if (!m_cachedScript)
69         return false;
70 
71     // If the content is already cached, this immediately calls notifyFinished.
72     m_cachedScript->addClient(*this);
73     return true;
74 }
75 
76 void CachedModuleScriptLoader::notifyFinished(CachedResource& resource)
77 {
78     ASSERT_UNUSED(resource, &resource == m_cachedScript);
79     ASSERT(m_cachedScript);
80     ASSERT(m_promise);
81 
82     Ref<CachedModuleScriptLoader> protectedThis(*this);
83     if (m_client)
84         m_client->notifyFinished(*this, WTFMove(m_promise));
85 
86     // Remove the client after calling notifyFinished to keep the data buffer in
87     // CachedResource alive while notifyFinished processes the resource.
88     m_cachedScript->removeClient(*this);
89     m_cachedScript = nullptr;
90 }
