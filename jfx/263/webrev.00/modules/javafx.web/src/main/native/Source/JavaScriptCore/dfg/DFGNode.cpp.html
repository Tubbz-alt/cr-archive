 71 bool Node::hasVariableAccessData(Graph& graph)
 72 {
 73     switch (op()) {
 74     case Phi:
 75         return graph.m_form != SSA;
 76     case GetLocal:
 77     case SetLocal:
 78     case SetArgumentDefinitely:
 79     case SetArgumentMaybe:
 80     case Flush:
 81     case PhantomLocal:
 82         return true;
 83     default:
 84         return false;
 85     }
 86 }

 88 void Node::remove(Graph& graph)
 89 {
 90     switch (op()) {
 91     case MultiGetByOffset: {
 92         MultiGetByOffsetData& data = multiGetByOffsetData();
 93         StructureSet set;
 94         for (MultiGetByOffsetCase& getCase : data.cases) {
 95             getCase.set().forEach(
 96                 [&] (RegisteredStructure structure) {
 97                     set.add(structure.get());
 98                 });
 99         }
100         convertToCheckStructure(graph.addStructureSet(set));
101         return;
102     }
103 
104     case MatchStructure: {
105         MatchStructureData& data = matchStructureData();
106         RegisteredStructureSet set;
107         for (MatchStructureVariant& variant : data.variants)
108             set.add(variant.structure);
109         convertToCheckStructure(graph.addStructureSet(set));
110         return;
111     }
112 
113     default:
114         if (flags() & NodeHasVarArgs) {
115             unsigned targetIndex = 0;
116             for (unsigned i = 0; i < numChildren(); ++i) {
117                 Edge& edge = graph.varArgChild(this, i);
118                 if (!edge)
119                     continue;
120                 if (edge.willHaveCheck()) {
121                     Edge& dst = graph.varArgChild(this, targetIndex++);
122                     std::swap(dst, edge);
123                     continue;
124                 }
125                 edge = Edge();
126             }
127             setOpAndDefaultFlags(CheckVarargs);
128             children.setNumChildren(targetIndex);
129         } else {
130             children = children.justChecks();
131             setOpAndDefaultFlags(Check);
132         }
133         return;
134     }
135 }

155 void Node::convertToIdentity()
156 {
157     RELEASE_ASSERT(child1());
158     RELEASE_ASSERT(!child2());
159     NodeFlags result = canonicalResultRepresentation(this->result());
160     setOpAndDefaultFlags(Identity);
161     setResult(result);
162 }

164 void Node::convertToIdentityOn(Node* child)
165 {
166     children.reset();
167     clearFlags(NodeHasVarArgs);
168     child1() = child->defaultEdge();
169     NodeFlags output = canonicalResultRepresentation(this->result());
170     NodeFlags input = canonicalResultRepresentation(child->result());
171     if (output == input) {
172         setOpAndDefaultFlags(Identity);
173         setResult(output);
174         return;
175     }
176     switch (output) {
177     case NodeResultDouble:
178         setOpAndDefaultFlags(DoubleRep);
179         switch (input) {
180         case NodeResultInt52:
181             child1().setUseKind(Int52RepUse);
182             return;
183         case NodeResultJS:
184             child1().setUseKind(NumberUse);
185             return;
186         default:
187             RELEASE_ASSERT_NOT_REACHED();
188             return;
189         }
190     case NodeResultInt52:
191         setOpAndDefaultFlags(Int52Rep);
192         switch (input) {
193         case NodeResultDouble:
194             child1().setUseKind(DoubleRepAnyIntUse);
195             return;
196         case NodeResultJS:
197             child1().setUseKind(AnyIntUse);
198             return;
199         default:
200             RELEASE_ASSERT_NOT_REACHED();
201             return;
202         }
203     case NodeResultJS:
204         setOpAndDefaultFlags(ValueRep);
205         switch (input) {
206         case NodeResultDouble:
207             child1().setUseKind(DoubleRepUse);
208             return;
209         case NodeResultInt52:
210             child1().setUseKind(Int52RepUse);
211             return;
212         default:
213             RELEASE_ASSERT_NOT_REACHED();
214             return;
215         }
216     default:
217         RELEASE_ASSERT_NOT_REACHED();
218         return;
219     }
220 }

241 void Node::convertToDirectCall(FrozenValue* executable)
242 {
243     NodeType newOp = LastNodeType;
244     switch (op()) {
245     case Call:
246         newOp = DirectCall;
247         break;
248     case Construct:
249         newOp = DirectConstruct;
250         break;
251     case TailCallInlinedCaller:
252         newOp = DirectTailCallInlinedCaller;
253         break;
254     case TailCall:
255         newOp = DirectTailCall;
256         break;
257     default:
258         RELEASE_ASSERT_NOT_REACHED();
259         break;
260     }
261 
262     m_op = newOp;
263     m_opInfo = executable;
264 }

266 void Node::convertToCallDOM(Graph& graph)
267 {
268     ASSERT(op() == Call);
269     ASSERT(signature());
270 
271     Edge edges[3];
272     // Skip the first one. This is callee.
273     RELEASE_ASSERT(numChildren() <= 4);
274     for (unsigned i = 1; i < numChildren(); ++i)
275         edges[i - 1] = graph.varArgChild(this, i);
276 
277     setOpAndDefaultFlags(CallDOM);
278     children.setChild1(edges[0]);
279     children.setChild2(edges[1]);
280     children.setChild3(edges[2]);
281 
282     if (!signature()->effect.mustGenerate())
283         clearFlags(NodeMustGenerate);
284 }

327 void printInternal(PrintStream& out, SwitchKind kind)
328 {
329     switch (kind) {
330     case SwitchImm:
331         out.print("SwitchImm");
332         return;
333     case SwitchChar:
334         out.print("SwitchChar");
335         return;
336     case SwitchString:
337         out.print("SwitchString");
338         return;
339     case SwitchCell:
340         out.print("SwitchCell");
341         return;
342     }
343     RELEASE_ASSERT_NOT_REACHED();
344 }
345 
346 void printInternal(PrintStream& out, Node* node)
347 {
348     if (!node) {
349         out.print("-");
350         return;
351     }
352     out.print("D@", node->index());
353     if (node->hasDoubleResult())
354         out.print("<Double>");
355     else if (node->hasInt52Result())
356         out.print("<Int52>");
357 }
