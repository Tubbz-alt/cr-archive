46 class JS_EXPORT_PRIVATE ScriptCallArgumentHandler {
47 public:
48     ScriptCallArgumentHandler(JSC::JSGlobalObject* globalObject) : m_globalObject(globalObject) { }
49 
50     void appendArgument(const char*);
51     void appendArgument(const String&);
52     void appendArgument(JSC::JSValue);
53     void appendArgument(long);
54     void appendArgument(long long);
55     void appendArgument(unsigned int);
56     void appendArgument(uint64_t);
57     void appendArgument(int);
58     void appendArgument(bool);
59 
60 protected:
61     JSC::MarkedArgumentBuffer m_arguments;
62     JSC::JSGlobalObject* m_globalObject;
63 
64 private:
65     // MarkedArgumentBuffer must be stack allocated, so prevent heap
66     // alloc of ScriptFunctionCall as well.
67     void* operator new(size_t) { ASSERT_NOT_REACHED(); return reinterpret_cast<void*>(0xbadbeef); }
68     void* operator new[](size_t) { ASSERT_NOT_REACHED(); return reinterpret_cast<void*>(0xbadbeef); }
69 };

74     ScriptFunctionCall(const ScriptObject& thisObject, const String& name, ScriptFunctionCallHandler handler = nullptr);
75     Expected<JSC::JSValue, NakedPtr<JSC::Exception>> call();

76 
77 protected:
78     ScriptFunctionCallHandler m_callHandler;
79     ScriptObject m_thisObject;
80     String m_name;
