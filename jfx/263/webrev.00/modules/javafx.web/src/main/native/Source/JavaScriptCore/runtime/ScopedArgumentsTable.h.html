 34 namespace JSC {
 35 
 36 // This class's only job is to hold onto the list of ScopeOffsets for each argument that a
 37 // function has. Most of the time, the BytecodeGenerator will create one of these and it will
 38 // never be modified subsequently. There is a rare case where a ScopedArguments object is created
 39 // and aliases one of these and then decides to modify it; in that case we do copy-on-write. This
 40 // makes sense because such modifications are so uncommon. You'd have to do something crazy like
 41 // "delete arguments[i]" or some variant of defineOwnProperty.
 42 class ScopedArgumentsTable final : public JSCell {
 43     friend class CachedScopedArgumentsTable;
 44 
 45 public:
 46     using Base = JSCell;
 47     static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
 48     static constexpr bool needsDestruction = true;
 49 
 50     template<typename CellType, SubspaceAccess mode>
 51     static IsoSubspace* subspaceFor(VM& vm)
 52     {
 53         return vm.scopedArgumentsTableSpace<mode>();
 54     }
 55 
 56 private:
 57     ScopedArgumentsTable(VM&);
 58     ~ScopedArgumentsTable();
 59 
 60 public:
 61     static ScopedArgumentsTable* create(VM&);
 62     static ScopedArgumentsTable* create(VM&, uint32_t length);
 63 
 64     static void destroy(JSCell*);
 65 
 66     ScopedArgumentsTable* clone(VM&);
 67 
 68     uint32_t length() const { return m_length; }
 69     ScopedArgumentsTable* setLength(VM&, uint32_t newLength);
 70 
 71     ScopeOffset get(uint32_t i) const { return at(i); }
 72 
 73     void lock()
 74     {
 75         m_locked = true;
 76     }
 77 
 78     ScopedArgumentsTable* set(VM&, uint32_t index, ScopeOffset);
 79 
 80     DECLARE_INFO;
 81 
 82     static Structure* createStructure(VM&, JSGlobalObject*, JSValue prototype);
 83 
 84     static ptrdiff_t offsetOfLength() { return OBJECT_OFFSETOF(ScopedArgumentsTable, m_length); }
 85     static ptrdiff_t offsetOfArguments() { return OBJECT_OFFSETOF(ScopedArgumentsTable, m_arguments); }
 86 
 87     typedef CagedUniquePtr<Gigacage::Primitive, ScopeOffset> ArgumentsPtr;
 88 
 89 private:
 90     ScopeOffset& at(uint32_t i) const
 91     {
 92         ASSERT_WITH_SECURITY_IMPLICATION(i < m_length);
 93         return m_arguments.get(length())[i];
 94     }
 95 
 96     uint32_t m_length;
 97     bool m_locked; // Being locked means that there are multiple references to this object and none of them expect to see the others' modifications. This means that modifications need to make a copy first.
 98     ArgumentsPtr m_arguments;
 99 };
100 
101 } // namespace JSC
