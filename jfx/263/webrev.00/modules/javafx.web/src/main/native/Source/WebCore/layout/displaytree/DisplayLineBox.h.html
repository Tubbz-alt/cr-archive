 26 #pragma once
 27 
 28 #if ENABLE(LAYOUT_FORMATTING_CONTEXT)
 29 
 30 #include "DisplayInlineRect.h"
 31 
 32 namespace WebCore {
 33 namespace Display {
 34 
 35 class LineBox {
 36     WTF_MAKE_FAST_ALLOCATED;
 37 public:
 38     struct Baseline {
 39         Baseline(InlineLayoutUnit ascent, InlineLayoutUnit descent);
 40 
 41         InlineLayoutUnit height() const { return ascent() + descent(); }
 42         InlineLayoutUnit ascent() const { return m_ascent; }
 43         InlineLayoutUnit descent() const { return m_descent; }
 44 
 45     private:
 46         InlineLayoutUnit m_ascent { 0 };
 47         InlineLayoutUnit m_descent { 0 };
 48     };
 49 
 50     LineBox(const InlineRect&, const InlineRect& scrollableOverflow, const InlineRect& inkOverflow, const Baseline&, InlineLayoutUnit baselineOffset, bool isConsideredEmpty);
 51 
 52     const InlineRect& rect() const { return m_rect; }
 53     const InlineRect& scrollableOverflow() const { return m_scrollableOverflow; }
 54     const InlineRect& inkOverflow() const { return m_inkOverflow; }
 55 
 56     InlineLayoutUnit left() const { return m_rect.left(); }
 57     InlineLayoutUnit right() const { return m_rect.right(); }
 58     InlineLayoutUnit top() const { return m_rect.top(); }
 59     InlineLayoutUnit bottom() const { return m_rect.bottom(); }
 60 
 61     InlineLayoutUnit width() const { return m_rect.width(); }
 62     InlineLayoutUnit height() const { return m_rect.height(); }
 63 
 64     const Baseline& baseline() const { return m_baseline; }
 65     // Baseline offset from line top. Note that offset does not necessarily equal to ascent.
 66     //
 67     // -------------------        line top         ------------------- (top align)
 68     //             ^                                              ^
 69     //             |                                  ^           |
 70     //   ^         | baseline offset                  |           | baseline offset
 71     //   |         |                                  |           |
 72     //   | ascent  |                                  | ascent    |
 73     //   |         |                                  v           v
 74     //   v         v                               ------------------- baseline
 75     //   ----------------- baseline                   ^
 76     //   ^                                            | descent
 77     //   | descent                                    v
 78     //   v
 79     // -------------------       line bottom       -------------------
 80     InlineLayoutUnit baselineOffset() const { return m_baselineOffset; }
 81     bool isConsideredEmpty() const { return m_isConsideredEmpty; }
 82 
 83 private:
 84     InlineRect m_rect;
 85     InlineRect m_scrollableOverflow;
 86     InlineRect m_inkOverflow;
 87     Baseline m_baseline;
 88     InlineLayoutUnit m_baselineOffset { 0 };
 89     bool m_isConsideredEmpty { true };
 90 };
 91 
 92 inline LineBox::LineBox(const InlineRect& rect, const InlineRect& scrollableOverflow, const InlineRect& inkOverflow, const Baseline& baseline, InlineLayoutUnit baselineOffset, bool isConsideredEmpty)
 93     : m_rect(rect)
 94     , m_scrollableOverflow(scrollableOverflow)
 95     , m_inkOverflow(inkOverflow)
 96     , m_baseline(baseline)
 97     , m_baselineOffset(baselineOffset)
 98     , m_isConsideredEmpty(isConsideredEmpty)
 99 {
100 }
101 
102 inline LineBox::Baseline::Baseline(InlineLayoutUnit ascent, InlineLayoutUnit descent)
103     : m_ascent(ascent)
104     , m_descent(descent)
105 {
106 }
107 
108 }
109 }
110 
111 #endif
