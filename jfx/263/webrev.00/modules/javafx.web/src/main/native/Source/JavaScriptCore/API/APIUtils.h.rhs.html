26 #ifndef APIUtils_h
27 #define APIUtils_h
28 
29 #include "CatchScope.h"
30 #include "Exception.h"
31 #include "JSCJSValue.h"
32 #include "JSGlobalObjectInspectorController.h"
33 #include "JSValueRef.h"
34 
35 enum class ExceptionStatus {
36     DidThrow,
37     DidNotThrow
38 };
39 
40 inline ExceptionStatus handleExceptionIfNeeded(JSC::CatchScope& scope, JSContextRef ctx, JSValueRef* returnedExceptionRef)
41 {
42     JSC::JSGlobalObject* globalObject = toJS(ctx);
43     if (UNLIKELY(scope.exception())) {
44         JSC::Exception* exception = scope.exception();
45         if (returnedExceptionRef)
46             *returnedExceptionRef = toRef(globalObject, exception->value());
47         scope.clearException();
48 #if ENABLE(REMOTE_INSPECTOR)
49         globalObject->inspectorController().reportAPIException(globalObject, exception);
50 #endif
51         return ExceptionStatus::DidThrow;
52     }
53     return ExceptionStatus::DidNotThrow;
54 }
55 
56 inline void setException(JSContextRef ctx, JSValueRef* returnedExceptionRef, JSC::JSValue exception)
57 {
58     JSC::JSGlobalObject* globalObject = toJS(ctx);
59     if (returnedExceptionRef)
60         *returnedExceptionRef = toRef(globalObject, exception);
61 #if ENABLE(REMOTE_INSPECTOR)
62     JSC::VM& vm = getVM(globalObject);
63     globalObject->inspectorController().reportAPIException(globalObject, JSC::Exception::create(vm, exception));
64 #endif
65 }
66 
67 #endif /* APIUtils_h */
