 42 Ref<Blob> ClipboardItem::blobFromString(const String& stringData, const String& type)
 43 {
 44     auto utf8 = stringData.utf8();
 45     return Blob::create(SharedBuffer::create(utf8.data(), utf8.length()), Blob::normalizedContentType(type));
 46 }
 47 
 48 static ClipboardItem::PresentationStyle clipboardItemPresentationStyle(const PasteboardItemInfo& info)
 49 {
 50     switch (info.preferredPresentationStyle) {
 51     case PasteboardItemPresentationStyle::Unspecified:
 52         return ClipboardItem::PresentationStyle::Unspecified;
 53     case PasteboardItemPresentationStyle::Inline:
 54         return ClipboardItem::PresentationStyle::Inline;
 55     case PasteboardItemPresentationStyle::Attachment:
 56         return ClipboardItem::PresentationStyle::Attachment;
 57     }
 58     ASSERT_NOT_REACHED();
 59     return ClipboardItem::PresentationStyle::Unspecified;
 60 }

 62 ClipboardItem::ClipboardItem(Vector<KeyValuePair<String, RefPtr<DOMPromise>>>&& items, const Options& options)
 63     : m_dataSource(makeUnique<ClipboardItemBindingsDataSource>(*this, WTFMove(items)))
 64     , m_presentationStyle(options.presentationStyle)
 65 {
 66 }
 67 
 68 ClipboardItem::ClipboardItem(Clipboard& clipboard, const PasteboardItemInfo& info)
 69     : m_clipboard(makeWeakPtr(clipboard))
 70     , m_navigator(makeWeakPtr(clipboard.navigator()))
 71     , m_dataSource(makeUnique<ClipboardItemPasteboardDataSource>(*this, info))
 72     , m_presentationStyle(clipboardItemPresentationStyle(info))
 73 {
 74 }

 76 Ref<ClipboardItem> ClipboardItem::create(Vector<KeyValuePair<String, RefPtr<DOMPromise>>>&& data, const Options& options)
 77 {
 78     return adoptRef(*new ClipboardItem(WTFMove(data), options));
 79 }
 80 
 81 Ref<ClipboardItem> ClipboardItem::create(Clipboard& clipboard, const PasteboardItemInfo& info)
 82 {
 83     return adoptRef(*new ClipboardItem(clipboard, info));
 84 }
