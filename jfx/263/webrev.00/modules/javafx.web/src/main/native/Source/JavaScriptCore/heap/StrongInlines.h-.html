28 #include "JSCJSValueInlines.h"

31 namespace JSC {
32 
33 template <typename T>
34 inline Strong<T>::Strong(VM& vm, ExternalType value)
35     : Handle<T>(vm.heap.handleSet()->allocate())
36 {
37     set(value);
38 }
39 
40 template <typename T>
41 inline Strong<T>::Strong(VM& vm, Handle<T> handle)
42     : Handle<T>(vm.heap.handleSet()->allocate())
43 {
44     set(handle.get());
45 }
46 
47 template <typename T>
48 inline void Strong<T>::set(VM& vm, ExternalType value)
49 {
50     if (!slot())
51         setSlot(vm.heap.handleSet()->allocate());
52     set(value);
53 }

55 } // namespace JSC
