31 namespace WebCore {
32 
33 template<typename T> struct Converter<IDLXPathNSResolver<T>> : DefaultConverter<IDLXPathNSResolver<T>> {
34     using ReturnType = RefPtr<T>;
35     using WrapperType = typename JSDOMWrapperConverterTraits<T>::WrapperClass;
36 
37     template<typename ExceptionThrower = DefaultExceptionThrower>
38     static ReturnType convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value, ExceptionThrower&& exceptionThrower = ExceptionThrower())
39     {
40         JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
41         auto scope = DECLARE_THROW_SCOPE(vm);
42         ReturnType object = WrapperType::toWrapped(vm, lexicalGlobalObject, value);
43         if (UNLIKELY(!object))
44             exceptionThrower(lexicalGlobalObject, scope);
45         return object;
46     }
47 };
48 
49 template<typename T> struct JSConverter<IDLXPathNSResolver<T>> {
50     static constexpr bool needsState = true;
51     static constexpr bool needsGlobalObject = true;
52 
53     template <typename U>
54     static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, const U& value)
55     {
56         return toJS(&lexicalGlobalObject, &globalObject, Detail::getPtrOrRef(value));
57     }
58 
59     template<typename U>
60     static JSC::JSValue convertNewlyCreated(JSC::JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, U&& value)
61     {
62         return toJSNewlyCreated(&lexicalGlobalObject, &globalObject, std::forward<U>(value));
63     }
64 };
