 72 bool JITFinalizer::finalizeCommon()
 73 {
 74     CodeBlock* codeBlock = m_plan.codeBlock();
 75     bool dumpDisassembly = shouldDumpDisassembly() || Options::asyncDisassembly();
 76 
 77     MacroAssemblerCodeRef<JSEntryPtrTag> b3CodeRef =
 78         FINALIZE_CODE_IF(dumpDisassembly, *b3CodeLinkBuffer, JSEntryPtrTag,
 79             "FTL B3 code for %s", toCString(CodeBlockWithJITType(codeBlock, JITType::FTLJIT)).data());
 80 
 81     MacroAssemblerCodeRef<JSEntryPtrTag> arityCheckCodeRef = entrypointLinkBuffer
 82         ? FINALIZE_CODE_IF(dumpDisassembly, *entrypointLinkBuffer, JSEntryPtrTag,
 83             "FTL entrypoint thunk for %s with B3 generated code at %p", toCString(CodeBlockWithJITType(codeBlock, JITType::FTLJIT)).data(), function)
 84         : MacroAssemblerCodeRef<JSEntryPtrTag>::createSelfManagedCodeRef(b3CodeRef.code());
 85 
 86     jitCode->initializeB3Code(b3CodeRef);
 87     jitCode->initializeArityCheckEntrypoint(arityCheckCodeRef);
 88 
 89     codeBlock->setJITCode(*jitCode);
 90 
 91     if (UNLIKELY(m_plan.compilation()))
 92         m_plan.vm()->m_perBytecodeProfiler->addCompilation(codeBlock, *m_plan.compilation());
 93 
 94     // The codeBlock is now responsible for keeping many things alive (e.g. frozen values)
 95     // that were previously kept alive by the plan.
 96     m_plan.vm()->heap.writeBarrier(codeBlock);
 97 
 98     return true;
 99 }
