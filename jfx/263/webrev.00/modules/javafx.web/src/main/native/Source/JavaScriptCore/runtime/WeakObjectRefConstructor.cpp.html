41 void WeakObjectRefConstructor::finishCreation(VM& vm, WeakObjectRefPrototype* prototype)
42 {
43     Base::finishCreation(vm, "WeakRef"_s, NameAdditionMode::WithoutStructureTransition);
44     putDirectWithoutTransition(vm, vm.propertyNames->prototype, prototype, PropertyAttribute::DontEnum | PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly);
45     putDirectWithoutTransition(vm, vm.propertyNames->length, jsNumber(1), PropertyAttribute::DontEnum | PropertyAttribute::ReadOnly);
46 }

48 static EncodedJSValue JSC_HOST_CALL callWeakRef(JSGlobalObject*, CallFrame*);
49 static EncodedJSValue JSC_HOST_CALL constructWeakRef(JSGlobalObject*, CallFrame*);
50 
51 WeakObjectRefConstructor::WeakObjectRefConstructor(VM& vm, Structure* structure)
52     : Base(vm, structure, callWeakRef, constructWeakRef)
53 {
54 }
55 
56 static EncodedJSValue JSC_HOST_CALL callWeakRef(JSGlobalObject* globalObject, CallFrame*)
57 {
58     VM& vm = globalObject->vm();
59     auto scope = DECLARE_THROW_SCOPE(vm);
60     return JSValue::encode(throwConstructorCannotBeCalledAsFunctionTypeError(globalObject, scope, "WeakRef"));
61 }
62 
63 static EncodedJSValue JSC_HOST_CALL constructWeakRef(JSGlobalObject* globalObject, CallFrame* callFrame)
64 {
65     VM& vm = globalObject->vm();
66     auto scope = DECLARE_THROW_SCOPE(vm);
67 
68     if (!callFrame->argument(0).isObject())
69         return throwVMTypeError(globalObject, scope, "First argument to WeakRef should be an object"_s);
70 
71     Structure* WeakObjectRefStructure = InternalFunction::createSubclassStructure(globalObject, callFrame->jsCallee(), callFrame->newTarget(), globalObject->weakObjectRefStructure());
72     RETURN_IF_EXCEPTION(scope, encodedJSValue());
73     RELEASE_AND_RETURN(scope, JSValue::encode(JSWeakObjectRef::create(vm, WeakObjectRefStructure, callFrame->uncheckedArgument(0).getObject())));
74 }
