 71 void PageRuntimeAgent::enable(ErrorString& errorString)
 72 {
 73     if (m_instrumentingAgents.pageRuntimeAgent() == this)
 74         return;
 75 
 76     InspectorRuntimeAgent::enable(errorString);
 77     if (!errorString.isEmpty())
 78         return;
 79 
 80     // Report initial contexts before enabling instrumentation as the reporting
 81     // can force creation of script state which could result in duplicate notifications.
 82     reportExecutionContextCreation();
 83 
 84     m_instrumentingAgents.setPageRuntimeAgent(this);

 85 }
 86 
 87 void PageRuntimeAgent::disable(ErrorString& errorString)
 88 {
 89     m_instrumentingAgents.setPageRuntimeAgent(nullptr);
 90 
 91     InspectorRuntimeAgent::disable(errorString);
 92 }

 94 void PageRuntimeAgent::frameNavigated(Frame& frame)
 95 {
 96     // Ensure execution context is created for the frame even if it doesn't have scripts.
 97     mainWorldExecState(&frame);
 98 }

100 void PageRuntimeAgent::didClearWindowObjectInWorld(Frame& frame, DOMWrapperWorld& world)
101 {
102     auto* pageAgent = m_instrumentingAgents.inspectorPageAgent();
103     if (!pageAgent)
104         return;
105 
106     notifyContextCreated(pageAgent->frameId(&frame), frame.script().globalObject(world), world);


107 }

109 InjectedScript PageRuntimeAgent::injectedScriptForEval(ErrorString& errorString, const int* executionContextId)
110 {
111     if (!executionContextId) {
112         JSC::JSGlobalObject* scriptState = mainWorldExecState(&m_inspectedPage.mainFrame());
113         InjectedScript result = injectedScriptManager().injectedScriptFor(scriptState);
114         if (result.hasNoValue())
115             errorString = "Internal error: main world execution context not found"_s;
116         return result;
117     }
118 
119     InjectedScript injectedScript = injectedScriptManager().injectedScriptForId(*executionContextId);
120     if (injectedScript.hasNoValue())
121         errorString = "Missing injected script for given executionContextId"_s;
122     return injectedScript;
123 }

135 void PageRuntimeAgent::reportExecutionContextCreation()
136 {
137     auto* pageAgent = m_instrumentingAgents.inspectorPageAgent();
138     if (!pageAgent)
139         return;
140 
141     for (auto* frame = &m_inspectedPage.mainFrame(); frame; frame = frame->tree().traverseNext()) {

142         if (!frame->script().canExecuteScripts(NotAboutToExecuteScript))
143             continue;
144 
145         auto frameId = pageAgent->frameId(frame);
146 
147         // Always send the main world first.
148         auto* mainGlobalObject = mainWorldExecState(frame);
149         notifyContextCreated(frameId, mainGlobalObject, mainThreadNormalWorld());
150 
151         for (auto& jsWindowProxy : frame->windowProxy().jsWindowProxiesAsVector()) {
152             auto* globalObject = jsWindowProxy->window();
153             if (globalObject == mainGlobalObject)
154                 continue;
155 
156             auto& securityOrigin = downcast<DOMWindow>(jsWindowProxy->wrapped()).document()->securityOrigin();
157             notifyContextCreated(frameId, globalObject, jsWindowProxy->world(), &securityOrigin);
158         }
159     }
160 }

162 static Inspector::Protocol::Runtime::ExecutionContextType toProtocol(DOMWrapperWorld::Type type)
163 {
164     switch (type) {
165     case DOMWrapperWorld::Type::Normal:
166         return Inspector::Protocol::Runtime::ExecutionContextType::Normal;
167     case DOMWrapperWorld::Type::User:
168         return Inspector::Protocol::Runtime::ExecutionContextType::User;
169     case DOMWrapperWorld::Type::Internal:
170         return Inspector::Protocol::Runtime::ExecutionContextType::Internal;
171     }
172 
173     ASSERT_NOT_REACHED();
174     return Inspector::Protocol::Runtime::ExecutionContextType::Internal;
175 }

177 void PageRuntimeAgent::notifyContextCreated(const String& frameId, JSC::JSGlobalObject* globalObject, const DOMWrapperWorld& world, SecurityOrigin* securityOrigin)
178 {
179     auto injectedScript = injectedScriptManager().injectedScriptFor(globalObject);
180     if (injectedScript.hasNoValue())
181         return;
182 
183     auto name = world.name();
184     if (name.isEmpty() && securityOrigin)
185         name = securityOrigin->toRawString();
186 
187     m_frontendDispatcher->executionContextCreated(Inspector::Protocol::Runtime::ExecutionContextDescription::create()
188         .setId(injectedScriptManager().injectedScriptIdFor(globalObject))
189         .setType(toProtocol(world.type()))
190         .setName(name)
191         .setFrameId(frameId)
192         .release());
193 }
