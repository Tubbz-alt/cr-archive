 58 Notification::Notification(Document& document, const String& title, const Options& options)

 79 void Notification::show()
 80 {
 81     // prevent double-showing
 82     if (m_state != Idle)
 83         return;
 84 
 85     auto* page = downcast<Document>(*scriptExecutionContext()).page();
 86     if (!page)
 87         return;
 88 
 89     auto& client = NotificationController::from(page)->client();
 90 
 91     if (client.checkPermission(scriptExecutionContext()) != Permission::Granted) {
 92         dispatchErrorEvent();
 93         return;
 94     }
 95     if (client.show(this)) {
 96         m_state = Showing;
 97         setPendingActivity(*this);
 98     }
 99 }

101 void Notification::close()
102 {
103     switch (m_state) {
104     case Idle:
105         break;
106     case Showing: {
107         auto* page = downcast<Document>(*scriptExecutionContext()).page();
108         if (page)
109             NotificationController::from(page)->client().cancel(this);
110         break;
111     }
112     case Closed:
113         break;
114     }
115 }

 81 void Notification::show()
 82 {
 83     // prevent double-showing
 84     if (m_state != Idle)
 85         return;
 86 
 87     auto* page = document()->page();
 88     if (!page)
 89         return;
 90 
 91     auto& client = NotificationController::from(page)->client();
 92 
 93     if (client.checkPermission(scriptExecutionContext()) != Permission::Granted) {
 94         dispatchErrorEvent();
 95         return;
 96     }
 97     if (client.show(this)) {
 98         m_state = Showing;
 99         setPendingActivity(*this);
100     }
101 }

103 void Notification::close()
104 {
105     switch (m_state) {
106     case Idle:
107         break;
108     case Showing: {
109         if (auto* page = document()->page())

111         break;
112     }
113     case Closed:
114         break;
115     }
116 }

151     auto* document = this->document();

188 auto Notification::permission(Document& document) -> Permission
189 {
190     auto* page = document.page();
191     if (!page)
192         return Permission::Default;
193 
194     if (!document.isSecureContext())
195         return Permission::Denied;
196 
197     return NotificationController::from(document.page())->client().checkPermission(&document);
198 }
199 
200 void Notification::requestPermission(Document& document, RefPtr<NotificationPermissionCallback>&& callback)
201 {
202     auto* page = document.page();
203     if (!page)
204         return;
205 
206     if (!document.isSecureContext()) {
207         if (callback) {
208             document.eventLoop().queueTask(TaskSource::DOMManipulation, [callback = WTFMove(callback)]() mutable {
209                 callback->handleEvent(Permission::Denied);
210             });
211         }
212         return;
213     }
214 
215     NotificationController::from(page)->client().requestPermission(&document, WTFMove(callback));
216 }
