 32 // We attempt to guess a value that is *AT LEAST* as large as the system's actual page size.
 33 // This is impossible to do correctly at build time, but JSC really needs it at build time, so
 34 // we have a RELEASE_ASSERT() inside WTF::pageSize to make sure it is set properly at runtime.
 35 // All of these values are going to be incorrect on systems configured to use larger than normal
 36 // page size, so on such systems it is expected that WebKit will crash until this value is changed
 37 // and recompiled. Sorry.
 38 //
 39 // macOS x86_64 uses 4 KiB, but Apple's aarch64 systems use 16 KiB. Use 16 KiB on all Apple systems
 40 // for consistency.
 41 //
 42 // Most Linux and Windows systems use a page size of 4 KiB.
 43 //
 44 // On Linux, Power systems normally use 64 KiB pages.
 45 //
 46 // aarch64 systems seem to be all over the place. Most Linux distros use 4 KiB, but RHEL uses
 47 // 64 KiB. (Apple uses 16 KiB.)
 48 //
 49 // Use 64 KiB for any unknown CPUs to be conservative.
 50 #if OS(DARWIN)
 51 constexpr size_t CeilingOnPageSize = 16 * KB;
 52 #elif OS(WINDOWS) || CPU(MIPS) || CPU(MIPS64) || CPU(X86) || CPU(X86_64) || CPU(ARM) || CPU(ARM64)
 53 constexpr size_t CeilingOnPageSize = 4 * KB;
 54 #elif CPU(UNKNOWN) || CPU(PPC) || CPU(PPC64) || CPU(PPC64LE)
 55 constexpr size_t CeilingOnPageSize = 64 * KB;
 56 #else
 57 #error Must set CeilingOnPageSize in PageBlock.h when adding a new CPU architecture!
 58 #endif
 59 
 60 WTF_EXPORT_PRIVATE size_t pageSize();
 61 WTF_EXPORT_PRIVATE size_t pageMask();
 62 inline bool isPageAligned(void* address) { return !(reinterpret_cast<intptr_t>(address) & (pageSize() - 1)); }
 63 inline bool isPageAligned(size_t size) { return !(size & (pageSize() - 1)); }
 64 inline bool isPowerOfTwo(size_t size) { return !(size & (size - 1)); }
 65 
 66 class PageBlock {
 67     WTF_MAKE_FAST_ALLOCATED;
 68 public:
 69     PageBlock();
 70     PageBlock(const PageBlock&);
 71     PageBlock(void*, size_t, bool hasGuardPages);
 72 
 73     void* base() const { return m_base; }
 74     size_t size() const { return m_size; }
 75 
 76     operator bool() const { return !!m_realBase; }
 77 
 78     bool contains(void* containedBase, size_t containedSize)
 79     {
 80         return containedBase >= m_base
 81             && (static_cast<char*>(containedBase) + containedSize) <= (static_cast<char*>(m_base) + m_size);
 82     }
 83 
 84 private:
 85     void* m_realBase;
 86     void* m_base;
 87     size_t m_size;
 88 };
 89 
 90 inline PageBlock::PageBlock()
 91     : m_realBase(0)
 92     , m_base(0)
 93     , m_size(0)
 94 {
 95 }
 96 
 97 inline PageBlock::PageBlock(const PageBlock& other)
 98     : m_realBase(other.m_realBase)
 99     , m_base(other.m_base)
100     , m_size(other.m_size)
101 {
102 }
103 
104 inline PageBlock::PageBlock(void* base, size_t size, bool hasGuardPages)
105     : m_realBase(base)
106     , m_base(static_cast<char*>(base) + ((base && hasGuardPages) ? pageSize() : 0))
107     , m_size(size)
108 {
109 }
