26 #pragma once
27 
28 #include "RenderElement.h"
29 
30 namespace WebCore {
31 
32 class RenderTreePosition {
33 public:
34     explicit RenderTreePosition(RenderElement& parent)
35         : m_parent(parent)
36     {
37     }
38     RenderTreePosition(RenderElement& parent, RenderObject* nextSibling)
39         : m_parent(parent)
40         , m_nextSibling(makeWeakPtr(nextSibling))
41         , m_hasValidNextSibling(true)
42     {
43     }
44 
45     RenderElement& parent() const { return m_parent; }
46     RenderObject* nextSibling() const { ASSERT(m_hasValidNextSibling); return m_nextSibling.get(); }
47 
48     void computeNextSibling(const Node&);
49     void moveToLastChild();
50     void invalidateNextSibling() { m_hasValidNextSibling = false; }
51     void invalidateNextSibling(const RenderObject&);
52 
53     RenderObject* nextSiblingRenderer(const Node&) const;
54 
55 private:
56     RenderElement& m_parent;
57     WeakPtr<RenderObject> m_nextSibling { nullptr };
58     bool m_hasValidNextSibling { false };
59 #if ASSERT_ENABLED
60     unsigned m_assertionLimitCounter { 0 };
61 #endif
62 };
63 
64 inline void RenderTreePosition::moveToLastChild()
65 {
66     m_nextSibling = nullptr;
67     m_hasValidNextSibling = true;
68 }
69 
70 } // namespace WebCore
