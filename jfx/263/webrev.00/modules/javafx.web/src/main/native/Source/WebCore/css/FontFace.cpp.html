 51 static bool populateFontFaceWithArrayBuffer(CSSFontFace& fontFace, Ref<JSC::ArrayBufferView>&& arrayBufferView)
 52 {
 53     auto source = makeUnique<CSSFontFaceSource>(fontFace, String(), nullptr, nullptr, WTFMove(arrayBufferView));
 54     fontFace.adoptSource(WTFMove(source));
 55     return false;
 56 }

 58 void FontFace::setErrorState()

 61     m_backing->setErrorState();

 64 Ref<FontFace> FontFace::create(Document& document, const String& family, Source&& source, const Descriptors& descriptors)
 65 {
 66     auto result = adoptRef(*new FontFace(document.fontSelector()));
 67     result->suspendIfNeeded();
 68 
 69     bool dataRequiresAsynchronousLoading = true;
 70 
 71     auto setFamilyResult = result->setFamily(document, family);
 72     if (setFamilyResult.hasException()) {
 73         result->setErrorState();
 74         return result;
 75     }
 76 
 77     auto sourceConversionResult = WTF::switchOn(source,
 78         [&] (String& string) -> ExceptionOr<void> {
 79             auto value = FontFace::parseString(string, CSSPropertySrc);
 80             if (!is<CSSValueList>(value))
 81                 return Exception { SyntaxError };
 82             CSSFontFace::appendSources(result->backing(), downcast<CSSValueList>(*value), &document, false);
 83             return { };
 84         },
 85         [&] (RefPtr<ArrayBufferView>& arrayBufferView) -> ExceptionOr<void> {
 86             dataRequiresAsynchronousLoading = populateFontFaceWithArrayBuffer(result->backing(), arrayBufferView.releaseNonNull());
 87             return { };
 88         },
 89         [&] (RefPtr<ArrayBuffer>& arrayBuffer) -> ExceptionOr<void> {
 90             unsigned byteLength = arrayBuffer->byteLength();
 91             auto arrayBufferView = JSC::Uint8Array::create(WTFMove(arrayBuffer), 0, byteLength);
 92             dataRequiresAsynchronousLoading = populateFontFaceWithArrayBuffer(result->backing(), WTFMove(arrayBufferView));
 93             return { };
 94         }
 95     );
 96 
 97     if (sourceConversionResult.hasException()) {
 98         result->setErrorState();
 99         return result;
100     }
101 
102     // These ternaries match the default strings inside the FontFaceDescriptors dictionary inside FontFace.idl.
103     auto setStyleResult = result->setStyle(descriptors.style.isEmpty() ? "normal"_s : descriptors.style);
104     if (setStyleResult.hasException()) {
105         result->setErrorState();
106         return result;
107     }
108     auto setWeightResult = result->setWeight(descriptors.weight.isEmpty() ? "normal"_s : descriptors.weight);
109     if (setWeightResult.hasException()) {
110         result->setErrorState();
111         return result;
112     }
113     auto setStretchResult = result->setStretch(descriptors.stretch.isEmpty() ? "normal"_s : descriptors.stretch);
114     if (setStretchResult.hasException()) {
115         result->setErrorState();
116         return result;
117     }
118     auto setUnicodeRangeResult = result->setUnicodeRange(descriptors.unicodeRange.isEmpty() ? "U+0-10FFFF"_s : descriptors.unicodeRange);
119     if (setUnicodeRangeResult.hasException()) {
120         result->setErrorState();
121         return result;
122     }
123     auto setFeatureSettingsResult = result->setFeatureSettings(descriptors.featureSettings.isEmpty() ? "normal"_s : descriptors.featureSettings);
124     if (setFeatureSettingsResult.hasException()) {
125         result->setErrorState();
126         return result;
127     }
128     auto setDisplayResult = result->setDisplay(descriptors.display.isEmpty() ? "auto"_s : descriptors.display);
129     if (setDisplayResult.hasException()) {
130         result->setErrorState();
131         return result;
132     }
133 
134     if (!dataRequiresAsynchronousLoading) {
135         result->backing().load();
136         auto status = result->backing().status();
137         ASSERT_UNUSED(status, status == CSSFontFace::Status::Success || status == CSSFontFace::Status::Failure);
138     }
139 
140     return result;
141 }

177 ExceptionOr<void> FontFace::setFamily(Document& document, const String& family)
178 {
179     if (family.isEmpty())
180         return Exception { SyntaxError };
181 
182     String familyNameToUse = family;
183     if (familyNameToUse.contains('\'') && document.quirks().shouldStripQuotationMarkInFontFaceSetFamily())
184         familyNameToUse = family.removeCharacters([](auto character) { return character == '\''; });
185 
186     // FIXME: https://bugs.webkit.org/show_bug.cgi?id=196381 Don't use a list here.
187     // See consumeFontFamilyDescriptor() in CSSPropertyParser.cpp for why we're using it.
188     auto list = CSSValueList::createCommaSeparated();
189     list->append(CSSValuePool::singleton().createFontFamilyValue(familyNameToUse));
190     bool success = m_backing->setFamilies(list);
191     if (!success)
192         return Exception { SyntaxError };
193     return { };
194 }

270 String FontFace::family() const
271 {
272     m_backing->updateStyleIfNeeded();
273 
274     const auto& families = m_backing->families();
275     if (!families.hasValue())
276         return "normal"_s;
277     auto familiesUnrwapped = families.value();
278     // FIXME: https://bugs.webkit.org/show_bug.cgi?id=196381 This is only here because CSSFontFace erroneously uses a list of values instead of a single value.
279     // See consumeFontFamilyDescriptor() in CSSPropertyParser.cpp.
280     if (familiesUnrwapped->length() == 1) {
281         if (familiesUnrwapped->item(0)) {
282             auto& item = *familiesUnrwapped->item(0);
283             if (item.isPrimitiveValue()) {
284                 auto& primitiveValue = downcast<CSSPrimitiveValue>(item);
285                 if (primitiveValue.isFontFamily()) {
286                     auto& fontFamily = primitiveValue.fontFamily();
287                     return fontFamily.familyName;
288                 }
289             }
290         }
291     }
292     return familiesUnrwapped->cssText();
293 }

295 String FontFace::style() const
296 {
297     m_backing->updateStyleIfNeeded();
298     const auto& styleWrapped = m_backing->italic();
299 
300     if (!styleWrapped.hasValue())
301         return "normal"_s;
302     auto style = styleWrapped.value();
303     auto minimum = ComputedStyleExtractor::fontStyleFromStyleValue(style.minimum, FontStyleAxis::ital);
304     auto maximum = ComputedStyleExtractor::fontStyleFromStyleValue(style.maximum, FontStyleAxis::ital);
305 
306     if (minimum.get().equals(maximum.get()))
307         return minimum->cssText();
308 
309     auto minimumNonKeyword = ComputedStyleExtractor::fontNonKeywordStyleFromStyleValue(style.minimum);
310     auto maximumNonKeyword = ComputedStyleExtractor::fontNonKeywordStyleFromStyleValue(style.maximum);
311 
312     ASSERT(minimumNonKeyword->fontStyleValue->valueID() == CSSValueOblique);
313     ASSERT(maximumNonKeyword->fontStyleValue->valueID() == CSSValueOblique);
314 
315     StringBuilder builder;
316     builder.append(minimumNonKeyword->fontStyleValue->cssText());
317     builder.append(' ');
318     if (minimum->obliqueValue.get() == maximum->obliqueValue.get())
319         builder.append(minimumNonKeyword->obliqueValue->cssText());
320     else {
321         builder.append(minimumNonKeyword->obliqueValue->cssText());
322         builder.append(' ');
323         builder.append(maximumNonKeyword->obliqueValue->cssText());
324     }
325     return builder.toString();
326 
327 }

329 String FontFace::weight() const
330 {
331     m_backing->updateStyleIfNeeded();
332     const auto& weightWrapped = m_backing->weight();
333     if (!weightWrapped.hasValue())
334         return "normal"_s;
335     auto weight = weightWrapped.value();
336     auto minimum = ComputedStyleExtractor::fontWeightFromStyleValue(weight.minimum);
337     auto maximum = ComputedStyleExtractor::fontWeightFromStyleValue(weight.maximum);
338 
339     if (minimum.get().equals(maximum.get()))
340         return minimum->cssText();
341 
342     auto minimumNonKeyword = ComputedStyleExtractor::fontNonKeywordWeightFromStyleValue(weight.minimum);
343     auto maximumNonKeyword = ComputedStyleExtractor::fontNonKeywordWeightFromStyleValue(weight.maximum);
344 
345     StringBuilder builder;
346     builder.append(minimumNonKeyword->cssText());
347     builder.append(' ');
348     builder.append(maximumNonKeyword->cssText());
349     return builder.toString();
350 }

375 String FontFace::unicodeRange() const
376 {
377     m_backing->updateStyleIfNeeded();
378     const auto& rangesWrapped = m_backing->ranges();
379     if (!rangesWrapped.hasValue())
380         return "U+0-10FFFF";
381     auto ranges = rangesWrapped.value();
382     if (!ranges.size())
383         return "U+0-10FFFF"_s;
384     auto values = CSSValueList::createCommaSeparated();
385     for (auto& range : ranges)
386         values->append(CSSUnicodeRangeValue::create(range.from, range.to));
387     return values->cssText();
388 }

390 String FontFace::featureSettings() const
391 {
392     m_backing->updateStyleIfNeeded();
393     const auto& featureSettingsWrapped = m_backing->featureSettings();
394     if (!featureSettingsWrapped.hasValue())
395         return "normal"_s;
396     auto featureSettings = featureSettingsWrapped.value();
397     if (!featureSettings.size())
398         return "normal"_s;
399     auto list = CSSValueList::createCommaSeparated();
400     for (auto& feature : featureSettings)
401         list->append(CSSFontFeatureValue::create(FontTag(feature.tag()), feature.value()));
402     return list->cssText();
403 }

405 String FontFace::display() const
406 {
407     m_backing->updateStyleIfNeeded();
408     const auto& loadingBehaviorWrapped = m_backing->loadingBehavior();
409     if (!loadingBehaviorWrapped.hasValue())
410         return "auto"_s;
411     return CSSValuePool::singleton().createValue(loadingBehaviorWrapped.value())->cssText();
412 }

414 auto FontFace::status() const -> LoadStatus
415 {
416     switch (m_backing->status()) {
417     case CSSFontFace::Status::Pending:
418         return LoadStatus::Unloaded;
419     case CSSFontFace::Status::Loading:
420         return LoadStatus::Loading;
421     case CSSFontFace::Status::TimedOut:
422         return LoadStatus::Error;
423     case CSSFontFace::Status::Success:
424         return LoadStatus::Loaded;
425     case CSSFontFace::Status::Failure:
426         return LoadStatus::Error;
427     }
428     ASSERT_NOT_REACHED();
429     return LoadStatus::Error;
430 }

432 void FontFace::adopt(CSSFontFace& newFace)
433 {
434     m_backing->removeClient(*this);
435     m_backing = newFace;
436     m_backing->addClient(*this);
437     newFace.setWrapper(*this);
438 }

440 void FontFace::fontStateChanged(CSSFontFace& face, CSSFontFace::Status, CSSFontFace::Status newState)
441 {
442     ASSERT_UNUSED(face, &face == m_backing.ptr());
443     switch (newState) {
444     case CSSFontFace::Status::Loading:
445         break;
446     case CSSFontFace::Status::TimedOut:
447         break;
448     case CSSFontFace::Status::Success:
449         // FIXME: This check should not be needed, but because FontFace's are sometimes adopted after they have already
450         // gone through a load cycle, we can sometimes come back through here and try to resolve the promise again.
451         if (!m_loadedPromise->isFulfilled())
452             m_loadedPromise->resolve(*this);
453         return;
454     case CSSFontFace::Status::Failure:
455         // FIXME: This check should not be needed, but because FontFace's are sometimes adopted after they have already
456         // gone through a load cycle, we can sometimes come back through here and try to resolve the promise again.
457         if (!m_loadedPromise->isFulfilled())
458             m_loadedPromise->reject(Exception { NetworkError });
459         return;
460     case CSSFontFace::Status::Pending:
461         ASSERT_NOT_REACHED();
462         return;
463     }
464 }

490 {
