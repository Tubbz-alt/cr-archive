 63 void JSGlobalObjectConsoleClient::messageWithTypeAndLevel(MessageType type, MessageLevel level, JSC::JSGlobalObject* globalObject, Ref<ScriptArguments>&& arguments)
 64 {
 65     if (JSGlobalObjectConsoleClient::logToSystemConsole())
 66         ConsoleClient::printConsoleMessageWithArguments(MessageSource::ConsoleAPI, type, level, globalObject, arguments.copyRef());
 67 
 68     if (LIKELY(!m_consoleAgent->developerExtrasEnabled()))
 69         return;
 70 
 71     String message;
 72     arguments->getFirstArgumentAsString(message);
 73     m_consoleAgent->addMessageToConsole(makeUnique<ConsoleMessage>(MessageSource::ConsoleAPI, type, level, message, WTFMove(arguments), globalObject));
 74 
 75     if (type == MessageType::Assert) {
 76         if (m_debuggerAgent)
 77             m_debuggerAgent->handleConsoleAssert(message);
 78     }
 79 }

 97 void JSGlobalObjectConsoleClient::profile(JSC::JSGlobalObject*, const String& title)
 98 {
 99     if (LIKELY(!m_consoleAgent->enabled()))
100         return;
101 
102     // Allow duplicate unnamed profiles. Disallow duplicate named profiles.
103     if (!title.isEmpty()) {
104         for (auto& existingTitle : m_profiles) {
105             if (existingTitle == title) {
106                 // FIXME: Send an enum to the frontend for localization?
107                 String warning = title.isEmpty() ? "Unnamed Profile already exists"_s : makeString("Profile \"", title, "\" already exists");
108                 m_consoleAgent->addMessageToConsole(makeUnique<ConsoleMessage>(MessageSource::ConsoleAPI, MessageType::Profile, MessageLevel::Warning, warning));
109                 return;
110             }
111         }
112     }
113 
114     m_profiles.append(title);
115     startConsoleProfile();
116 }
117 
118 void JSGlobalObjectConsoleClient::profileEnd(JSC::JSGlobalObject*, const String& title)
119 {
120     if (LIKELY(!m_consoleAgent->enabled()))
121         return;
122 
123     // Stop profiles in reverse order. If the title is empty, then stop the last profile.
124     // Otherwise, match the title of the profile to stop.
125     for (ptrdiff_t i = m_profiles.size() - 1; i >= 0; --i) {
126         if (title.isEmpty() || m_profiles[i] == title) {
127             m_profiles.remove(i);
128             if (m_profiles.isEmpty())
129                 stopConsoleProfile();
130             return;
131         }
132     }
133 
134     // FIXME: Send an enum to the frontend for localization?
135     String warning = title.isEmpty() ? "No profiles exist"_s : makeString("Profile \"", title, "\" does not exist");
136     m_consoleAgent->addMessageToConsole(makeUnique<ConsoleMessage>(MessageSource::ConsoleAPI, MessageType::ProfileEnd, MessageLevel::Warning, warning));
137 }

139 void JSGlobalObjectConsoleClient::startConsoleProfile()
140 {
141     ErrorString ignored;
142 
143     if (m_debuggerAgent) {
144         m_profileRestoreBreakpointActiveValue = m_debuggerAgent->breakpointsActive();
145         m_debuggerAgent->setBreakpointsActive(ignored, false);
146     }
147 
148     if (m_scriptProfilerAgent) {
149         const bool includeSamples = true;
150         m_scriptProfilerAgent->startTracking(ignored, &includeSamples);
151     }
152 }
153 
154 void JSGlobalObjectConsoleClient::stopConsoleProfile()
155 {
156     ErrorString ignored;
157 
158     if (m_scriptProfilerAgent)
159         m_scriptProfilerAgent->stopTracking(ignored);
160 
161     if (m_debuggerAgent)
162         m_debuggerAgent->setBreakpointsActive(ignored, m_profileRestoreBreakpointActiveValue);
163 }

229 void JSGlobalObjectConsoleClient::warnUnimplemented(const String& method)
230 {
