27 #ifndef TemporaryOpenGLSetting_h
28 #define TemporaryOpenGLSetting_h
29 
30 #include "GraphicsTypes3D.h"


31 
32 #include <wtf/Noncopyable.h>
33 
34 namespace WebCore {
35 
36 // TemporaryOpenGLSetting<> is useful for temporarily disabling (or enabling) a particular OpenGL
37 // feature with a particular scope. A TemporaryOpenGLSetting<> object returns the flag to its original
38 // value upon destruction, making it an alternative to checking, clearing, and resetting each flag
39 // at all of a block's exit points.
40 //
41 // Based on WTF::SetForScope<>
42 
43 class TemporaryOpenGLSetting {
44     WTF_MAKE_NONCOPYABLE(TemporaryOpenGLSetting);
45 public:
46     TemporaryOpenGLSetting(GC3Denum capability, GC3Denum scopedState);
47     ~TemporaryOpenGLSetting();
48 
49 private:
50     const GC3Denum m_capability;
51     const GC3Denum m_scopedState;
52     GC3Denum m_originalState;
53 };
54 
55 }
56 
57 using WebCore::TemporaryOpenGLSetting;
58 
59 #endif
