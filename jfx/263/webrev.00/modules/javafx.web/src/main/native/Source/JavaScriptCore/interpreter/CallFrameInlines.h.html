 35 inline Register& CallFrame::r(VirtualRegister reg)
 36 {
 37     if (reg.isConstant())
 38         return *reinterpret_cast<Register*>(&this->codeBlock()->constantRegister(reg));
 39     return this[reg.offset()];
 40 }
 41 
 42 inline Register& CallFrame::uncheckedR(VirtualRegister reg)
 43 {
 44     ASSERT(!reg.isConstant());
 45     return this[reg.offset()];
 46 }
 47 
 48 inline JSValue CallFrame::guaranteedJSValueCallee() const
 49 {
 50     ASSERT(!callee().isWasm());
 51     return this[static_cast<int>(CallFrameSlot::callee)].jsValue();
 52 }
 53 
 54 inline JSObject* CallFrame::jsCallee() const
 55 {
 56     ASSERT(!callee().isWasm());
 57     return this[static_cast<int>(CallFrameSlot::callee)].object();
 58 }
 59 
 60 inline CodeBlock* CallFrame::codeBlock() const
 61 {
 62     return this[static_cast<int>(CallFrameSlot::codeBlock)].Register::codeBlock();
 63 }
 64 
 65 inline SUPPRESS_ASAN CodeBlock* CallFrame::unsafeCodeBlock() const
 66 {
 67     return this[static_cast<int>(CallFrameSlot::codeBlock)].Register::asanUnsafeCodeBlock();
 68 }
 69 
 70 inline JSGlobalObject* CallFrame::lexicalGlobalObject(VM& vm) const
 71 {
 72     UNUSED_PARAM(vm);
 73 #if ENABLE(WEBASSEMBLY)
 74     if (callee().isWasm())
 75         return lexicalGlobalObjectFromWasmCallee(vm);
 76 #endif
 77     return jsCallee()->globalObject();
 78 }
 79 
 80 inline bool CallFrame::isStackOverflowFrame() const
 81 {
 82     if (callee().isWasm())
 83         return false;
 84     return jsCallee() == jsCallee()->globalObject()->stackOverflowFrameCallee();
 85 }
 86 
 87 inline bool CallFrame::isWasmFrame() const
 88 {
 89     return callee().isWasm();
 90 }
 91 
 92 inline void CallFrame::setCallee(JSObject* callee)
 93 {
 94     static_cast<Register*>(this)[static_cast<int>(CallFrameSlot::callee)] = callee;
 95 }
 96 
 97 inline void CallFrame::setCodeBlock(CodeBlock* codeBlock)
 98 {
 99     static_cast<Register*>(this)[static_cast<int>(CallFrameSlot::codeBlock)] = codeBlock;
100 }
101 
102 inline void CallFrame::setScope(int scopeRegisterOffset, JSScope* scope)
103 {
104     static_cast<Register*>(this)[scopeRegisterOffset] = scope;
105 }
106 
107 inline JSScope* CallFrame::scope(int scopeRegisterOffset) const
108 {
109     ASSERT(this[scopeRegisterOffset].Register::scope());
110     return this[scopeRegisterOffset].Register::scope();
111 }
112 
113 inline Register* CallFrame::topOfFrame()
114 {
115     if (!codeBlock())
116         return registers();
117     return topOfFrameInternal();
118 }
