41 JSValue toJSNewlyCreated(JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<WebAnimation>&& value)
42 {
43     if (value->isCSSAnimation())
44         return createWrapper<CSSAnimation>(globalObject, WTFMove(value));
45     if (value->isCSSTransition())
46         return createWrapper<CSSTransition>(globalObject, WTFMove(value));
47     return createWrapper<WebAnimation>(globalObject, WTFMove(value));
48 }
49 
50 JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, WebAnimation& value)
51 {
52     return wrap(lexicalGlobalObject, globalObject, value);
53 }

55 EncodedJSValue constructJSWebAnimation(JSGlobalObject* lexicalGlobalObject, CallFrame& callFrame)
56 {
57     VM& vm = lexicalGlobalObject->vm();
58     auto throwScope = DECLARE_THROW_SCOPE(vm);
59     UNUSED_PARAM(throwScope);
60     auto* jsConstructor = jsCast<JSDOMConstructorBase*>(callFrame.jsCallee());
61     ASSERT(jsConstructor);
62     auto* context = jsConstructor->scriptExecutionContext();
63     if (UNLIKELY(!context))
64         return throwConstructorScriptExecutionContextUnavailableError(*lexicalGlobalObject, throwScope, "Animation");
65     auto& document = downcast<Document>(*context);
66     auto effect = convert<IDLNullable<IDLInterface<AnimationEffect>>>(*lexicalGlobalObject, callFrame.argument(0), [](JSGlobalObject& lexicalGlobalObject, ThrowScope& scope) {
67         throwArgumentTypeError(lexicalGlobalObject, scope, 0, "effect", "Animation", nullptr, "AnimationEffect");
68     });
69     RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
70 
71     if (callFrame.argument(1).isUndefined()) {
72         auto object = WebAnimation::create(document, WTFMove(effect));
73         return JSValue::encode(toJSNewlyCreated<IDLInterface<WebAnimation>>(*lexicalGlobalObject, *jsConstructor->globalObject(), WTFMove(object)));
74     }
75 
76     auto timeline = convert<IDLNullable<IDLInterface<AnimationTimeline>>>(*lexicalGlobalObject, callFrame.uncheckedArgument(1), [](JSGlobalObject& lexicalGlobalObject, ThrowScope& scope) {
77         throwArgumentTypeError(lexicalGlobalObject, scope, 1, "timeline", "Animation", nullptr, "AnimationTimeline");
78     });
79     RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
80     auto object = WebAnimation::create(document, WTFMove(effect), WTFMove(timeline));
81     return JSValue::encode(toJSNewlyCreated<IDLInterface<WebAnimation>>(*lexicalGlobalObject, *jsConstructor->globalObject(), WTFMove(object)));
82 }
