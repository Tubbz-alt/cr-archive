26 #pragma once
27 
28 #if ENABLE(INDEXED_DATABASE)
29 
30 #include "Event.h"
31 #include "IDBResourceIdentifier.h"
32 #include <wtf/Optional.h>
33 
34 namespace WebCore {
35 
36 class IDBVersionChangeEvent final : public Event {
37     WTF_MAKE_ISO_ALLOCATED(IDBVersionChangeEvent);
38 public:
39     static Ref<IDBVersionChangeEvent> create(uint64_t oldVersion, uint64_t newVersion, const AtomString& eventType)
40     {
41         return adoptRef(*new IDBVersionChangeEvent(IDBResourceIdentifier::emptyValue(), oldVersion, newVersion, eventType));
42     }
43 
44     static Ref<IDBVersionChangeEvent> create(const IDBResourceIdentifier& requestIdentifier, uint64_t oldVersion, uint64_t newVersion, const AtomString& eventType)
45     {
46         return adoptRef(*new IDBVersionChangeEvent(requestIdentifier, oldVersion, newVersion, eventType));
47     }
48 
49     struct Init : EventInit {
50         uint64_t oldVersion { 0 };
51         Optional<uint64_t> newVersion;
52     };
53 
54     static Ref<IDBVersionChangeEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
55     {
56         return adoptRef(*new IDBVersionChangeEvent(type, initializer, isTrusted));
57     }
58 
59     const IDBResourceIdentifier& requestIdentifier() const { return m_requestIdentifier; }
60 
61     bool isVersionChangeEvent() const final { return true; }
62 
63     uint64_t oldVersion() const { return m_oldVersion; }
64     Optional<uint64_t> newVersion() const { return m_newVersion; }
65 
66 private:
67     IDBVersionChangeEvent(const IDBResourceIdentifier& requestIdentifier, uint64_t oldVersion, uint64_t newVersion, const AtomString& eventType);
68     IDBVersionChangeEvent(const AtomString&, const Init&, IsTrusted);
69 
70     EventInterface eventInterface() const;
71 
72     IDBResourceIdentifier m_requestIdentifier;
73     uint64_t m_oldVersion;
74     Optional<uint64_t> m_newVersion;
75 };
76 
77 } // namespace WebCore
78 
79 SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::IDBVersionChangeEvent)
80     static bool isType(const WebCore::Event& event) { return event.isVersionChangeEvent(); }
81 SPECIALIZE_TYPE_TRAITS_END()
82 
83 #endif // ENABLE(INDEXED_DATABASE)
