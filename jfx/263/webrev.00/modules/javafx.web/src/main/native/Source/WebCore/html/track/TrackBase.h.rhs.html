 26 #pragma once
 27 
 28 #if ENABLE(VIDEO_TRACK)
 29 
 30 #include <wtf/LoggerHelper.h>
 31 #include <wtf/WeakPtr.h>
 32 #include <wtf/text/AtomString.h>
 33 
 34 namespace WebCore {
 35 
 36 class Element;
 37 class HTMLMediaElement;
 38 class SourceBuffer;
 39 
 40 class TrackBase
 41     : public RefCounted<TrackBase>
 42 #if !RELEASE_LOG_DISABLED
 43     , private LoggerHelper
 44 #endif
 45 {
 46 public:
 47     virtual ~TrackBase() = default;
 48 
 49     enum Type { BaseTrack, TextTrack, AudioTrack, VideoTrack };
 50     Type type() const { return m_type; }
 51 
 52     virtual void setMediaElement(WeakPtr<HTMLMediaElement>);
 53     WeakPtr<HTMLMediaElement> mediaElement() { return m_mediaElement; }
 54     virtual Element* element();
 55 
 56     virtual AtomString id() const { return m_id; }
 57     virtual void setId(const AtomString& id) { m_id = id; }
 58 
 59     AtomString label() const { return m_label; }
 60     void setLabel(const AtomString& label) { m_label = label; }
 61 
 62     AtomString validBCP47Language() const;
 63     AtomString language() const { return m_language; }
 64     virtual void setLanguage(const AtomString&);
 65 
 66     virtual void clearClient() = 0;
 67 
 68     virtual int uniqueId() const { return m_uniqueId; }
 69 
 70 #if ENABLE(MEDIA_SOURCE)
 71     SourceBuffer* sourceBuffer() const { return m_sourceBuffer; }
 72     void setSourceBuffer(SourceBuffer* buffer) { m_sourceBuffer = buffer; }
 73 #endif
 74 
 75     virtual bool enabled() const = 0;
 76 
 77 #if !RELEASE_LOG_DISABLED
 78     const Logger& logger() const final { ASSERT(m_logger); return *m_logger.get(); }
 79     const void* logIdentifier() const final { return m_logIdentifier; }
 80     WTFLogChannel& logChannel() const final;
 81 #endif
 82 
 83 protected:
 84     TrackBase(Type, const AtomString& id, const AtomString& label, const AtomString& language);
 85 
 86     WeakPtr<HTMLMediaElement> m_mediaElement { nullptr };
 87 
 88 #if ENABLE(MEDIA_SOURCE)
 89     SourceBuffer* m_sourceBuffer { nullptr };
 90 #endif
 91 
 92 private:
 93     Type m_type;
 94     int m_uniqueId;
 95     AtomString m_id;
 96     AtomString m_label;
 97     AtomString m_language;
 98     AtomString m_validBCP47Language;
 99 #if !RELEASE_LOG_DISABLED
100     RefPtr<const Logger> m_logger;
101     const void* m_logIdentifier;
102 #endif
103 };
104 
105 class MediaTrackBase : public TrackBase {
106 public:
107     const AtomString& kind() const { return m_kind; }
108     virtual void setKind(const AtomString&);
109 
110 protected:
111     MediaTrackBase(Type, const AtomString& id, const AtomString& label, const AtomString& language);
112 
113     void setKindInternal(const AtomString&);
114 
115 private:
116     virtual bool isValidKind(const AtomString&) const = 0;
117 
118     AtomString m_kind;
119 };
120 
121 } // namespace WebCore
122 
123 #endif
