 43 // IsSameOrReference::value is true if T is the same type as U or U&. Else, it is false.
 44 
 45 template<typename T, typename U>
 46 static constexpr auto IsSameOrReferenceHelper(int) -> std::enable_if_t<std::is_same<T, U>::value || std::is_same<T, U&>::value, std::true_type>;
 47 
 48 template<typename T, typename U>
 49 static constexpr std::false_type IsSameOrReferenceHelper(...);
 50 
 51 template<class T, typename U>
 52 struct IsSameOrReference : public std::is_same<decltype(IsSameOrReferenceHelper<T, U>(0)), std::true_type> { };
 53 
 54 
 55 template<typename T, typename... Arguments, typename = std::enable_if_t<IsSameOrReference<T, B3::Air::Tmp>::value || IsSameOrReference<T, Reg>::value>>
 56 inline void appendAirArg(B3::Air::Inst& inst, T&& arg)
 57 {
 58     inst.args.append(std::forward<T>(arg));
 59 }
 60 
 61 template<typename T, typename... Arguments, typename = std::enable_if_t<!IsSameOrReference<T, B3::Air::Tmp>::value && !IsSameOrReference<T, Reg>::value>>
 62 inline void appendAirArg(B3::Air::Inst&, T&&, int = 0) { }
 63 
 64 inline void appendAirArgs(B3::Air::Inst&) { }
 65 
 66 template<typename T, typename... Arguments>
 67 inline void appendAirArgs(B3::Air::Inst& inst, T&& t, Arguments&&... others)
 68 {
 69     appendAirArg(inst, std::forward<T>(t));
 70     appendAirArgs(inst, std::forward<Arguments>(others)...);
 71 }

 95 class PrintSpecial : public Special {
 96 public:
 97     PrintSpecial(Printer::PrintRecordList*);
 98     ~PrintSpecial();
 99 
100     // You cannot use this register to pass arguments. It just so happens that this register is not
101     // used for arguments in the C calling convention. By the way, this is the only thing that causes
102     // this special to be specific to C calls.
103     static constexpr GPRReg scratchRegister = GPRInfo::nonArgGPR0;
104 
105 protected:
106     void forEachArg(Inst&, const ScopedLambda<Inst::EachArgCallback>&) final;
107     bool isValid(Inst&) final;
108     bool admitsStack(Inst&, unsigned argIndex) final;
109     bool admitsExtendedOffsetAddr(Inst&, unsigned) final;
110     void reportUsedRegisters(Inst&, const RegisterSet&) final;
111     MacroAssembler::Jump generate(Inst&, CCallHelpers&, GenerationContext&) final;
112     RegisterSet extraEarlyClobberedRegs(Inst&) final;
113     RegisterSet extraClobberedRegs(Inst&) final;
114 
115     void dumpImpl(PrintStream&) const final;
116     void deepDumpImpl(PrintStream&) const final;
117 
118 private:
119     static constexpr unsigned specialArgOffset = 0;
120     static constexpr unsigned numSpecialArgs = 1;
121     static constexpr unsigned calleeArgOffset = numSpecialArgs;
122     static constexpr unsigned numCalleeArgs = 1;
123     static constexpr unsigned returnGPArgOffset = numSpecialArgs + numCalleeArgs;
124     static constexpr unsigned numReturnGPArgs = 2;
125     static constexpr unsigned returnFPArgOffset = numSpecialArgs + numCalleeArgs + numReturnGPArgs;
126     static constexpr unsigned numReturnFPArgs = 1;
127     static constexpr unsigned argArgOffset =
128     numSpecialArgs + numCalleeArgs + numReturnGPArgs + numReturnFPArgs;
129 
130     Printer::PrintRecordList* m_printRecordList;
131 };
