48 static Lock providerIdLock;
49 
50 void SourceProvider::getID()
51 {
52     LockHolder lock(&providerIdLock);
53     if (!m_id) {
54         static intptr_t nextProviderID = 0;
55         m_id = ++nextProviderID;
56         RELEASE_ASSERT(m_id);
57     }
58 }
