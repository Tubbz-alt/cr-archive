 42 LayoutUnit TextUtil::width(const Box& inlineBox, unsigned from, unsigned to, LayoutUnit contentLogicalLeft)
 43 {
 44     auto& style = inlineBox.style();
 45     auto& font = style.fontCascade();
 46     if (!font.size() || from == to)
 47         return 0;
 48 
 49     auto text = inlineBox.textContent();

 50     ASSERT(to <= text.length());
 51 
 52     if (font.isFixedPitch())
 53         return fixedPitchWidth(text, style, from, to, contentLogicalLeft);
 54 
 55     auto hasKerningOrLigatures = font.enableKerning() || font.requiresShaping();
 56     auto measureWithEndSpace = hasKerningOrLigatures && to < text.length() && text[to] == ' ';
 57     if (measureWithEndSpace)
 58         ++to;
 59     LayoutUnit width;
 60     auto tabWidth = style.collapseWhiteSpace() ? TabSize(0) : style.tabSize();
 61 
 62     WebCore::TextRun run(StringView(text).substring(from, to - from), contentLogicalLeft);
 63     if (tabWidth)
 64         run.setTabSize(true, tabWidth);
 65     width = font.width(run);






 66 
 67     if (measureWithEndSpace)
 68         width -= (font.spaceWidth() + font.wordSpacing());
 69 
 70     return std::max<LayoutUnit>(0, width);
 71 }

 73 LayoutUnit TextUtil::fixedPitchWidth(String text, const RenderStyle& style, unsigned from, unsigned to, LayoutUnit contentLogicalLeft)
 74 {
 75     auto& font = style.fontCascade();
 76     auto monospaceCharacterWidth = font.spaceWidth();
 77     LayoutUnit width;
 78     for (auto i = from; i < to; ++i) {
 79         auto character = text[i];
 80         if (character >= ' ' || character == '\n')
 81             width += monospaceCharacterWidth;
 82         else if (character == '\t')
 83             width += style.collapseWhiteSpace() ? monospaceCharacterWidth : font.tabWidth(style.tabSize(), contentLogicalLeft + width);
 84 
 85         if (i > from && (character == ' ' || character == '\t' || character == '\n'))
 86             width += font.wordSpacing();
 87     }
 88 
 89     return width;
 90 }

 92 bool TextUtil::isTrimmableContent(const InlineItem& inlineItem)
 93 {
 94     if (!is<InlineTextItem>(inlineItem))
 95         return false;
 96     auto& inlineTextItem = downcast<InlineTextItem>(inlineItem);
 97     return inlineTextItem.isWhitespace() && inlineTextItem.style().collapseWhiteSpace();

100 TextUtil::SplitData TextUtil::split(const Box& inlineBox, unsigned startPosition, unsigned length, LayoutUnit textWidth, LayoutUnit availableWidth, LayoutUnit contentLogicalLeft)
101 {
102     // FIXME This should take hypens into account.
103     ASSERT(availableWidth >= 0);
104     auto left = startPosition;
105     // Pathological case of (extremely)long string and narrow lines.
106     // Adjust the range so that we can pick a reasonable midpoint.
107     LayoutUnit averageCharacterWidth = textWidth / length;
108     auto right = std::min<unsigned>(left + (2 * availableWidth / averageCharacterWidth).toUnsigned(), (startPosition + length - 1));

109     // Preserve the left width for the final split position so that we don't need to remeasure the left side again.
110     LayoutUnit leftSideWidth = 0;
111     while (left < right) {
112         auto middle = (left + right) / 2;
113         auto width = TextUtil::width(inlineBox, startPosition, middle + 1, contentLogicalLeft);
114         if (width < availableWidth) {
115             left = middle + 1;
116             leftSideWidth = width;
117         } else if (width > availableWidth)
118             right = middle;
119         else {
120             right = middle + 1;
121             leftSideWidth = width;
122             break;
123         }
124     }
125     return { startPosition, right - startPosition, leftSideWidth };
126 }

 40 InlineLayoutUnit TextUtil::width(const InlineTextItem& inlineTextItem, unsigned from, unsigned to, InlineLayoutUnit contentLogicalLeft)
 41 {
 42     // Fast path for collapsed whitespace.
 43     if (inlineTextItem.isCollapsible()) {
 44         auto font = inlineTextItem.style().fontCascade();
 45         return font.spaceWidth() + font.wordSpacing();
 46     }
 47     return TextUtil::width(inlineTextItem.layoutBox(), from, to, contentLogicalLeft);
 48 }

 50 InlineLayoutUnit TextUtil::width(const Box& inlineBox, unsigned from, unsigned to, InlineLayoutUnit contentLogicalLeft)
 51 {
 52     auto& style = inlineBox.style();
 53     auto& font = style.fontCascade();
 54     if (!font.size() || from == to)
 55         return 0;
 56 
 57     auto& textContext = *inlineBox.textContext();
 58     auto& text = textContext.content;
 59     ASSERT(to <= text.length());




 60     auto hasKerningOrLigatures = font.enableKerning() || font.requiresShaping();
 61     auto measureWithEndSpace = hasKerningOrLigatures && to < text.length() && text[to] == ' ';
 62     if (measureWithEndSpace)
 63         ++to;
 64     float width = 0;
 65     if (textContext.canUseSimplifiedContentMeasuring) {
 66         if (font.isFixedPitch())
 67             width = fixedPitchWidth(text, style, from, to, contentLogicalLeft);
 68         else
 69             width = font.widthForSimpleText(StringView(text).substring(from, to - from));
 70     } else {
 71         auto tabWidth = style.collapseWhiteSpace() ? TabSize(0) : style.tabSize();
 72         WebCore::TextRun run(StringView(text).substring(from, to - from), contentLogicalLeft);
 73         if (tabWidth)
 74             run.setTabSize(true, tabWidth);
 75         width = font.width(run);
 76     }
 77 
 78     if (measureWithEndSpace)
 79         width -= (font.spaceWidth() + font.wordSpacing());
 80 
 81     return std::max<InlineLayoutUnit>(0 , InlineLayoutUnit(width));
 82 }

103 TextUtil::SplitData TextUtil::split(const Box& inlineBox, unsigned startPosition, unsigned length, InlineLayoutUnit textWidth, InlineLayoutUnit availableWidth, InlineLayoutUnit contentLogicalLeft)
104 {

105     ASSERT(availableWidth >= 0);
106     auto left = startPosition;
107     // Pathological case of (extremely)long string and narrow lines.
108     // Adjust the range so that we can pick a reasonable midpoint.
109     InlineLayoutUnit averageCharacterWidth = textWidth / length;
110     unsigned offset = toLayoutUnit(2 * availableWidth / averageCharacterWidth).toUnsigned();
111     auto right = std::min<unsigned>(left + offset, (startPosition + length - 1));
112     // Preserve the left width for the final split position so that we don't need to remeasure the left side again.
113     InlineLayoutUnit leftSideWidth = 0;
114     while (left < right) {
115         auto middle = (left + right) / 2;
116         auto width = TextUtil::width(inlineBox, startPosition, middle + 1, contentLogicalLeft);
117         if (width < availableWidth) {
118             left = middle + 1;
119             leftSideWidth = width;
120         } else if (width > availableWidth)
121             right = middle;
122         else {
123             right = middle + 1;
124             leftSideWidth = width;
125             break;
126         }
127     }
128     return { startPosition, right - startPosition, leftSideWidth };
129 }

131 unsigned TextUtil::findNextBreakablePosition(LazyLineBreakIterator& lineBreakIterator, unsigned startPosition, const RenderStyle& style)
132 {
133     auto keepAllWordsForCJK = style.wordBreak() == WordBreak::KeepAll;
134     auto breakNBSP = style.autoWrap() && style.nbspMode() == NBSPMode::Space;
135 
136     if (keepAllWordsForCJK) {
137         if (breakNBSP)
138             return nextBreakablePositionKeepingAllWords(lineBreakIterator, startPosition);
139         return nextBreakablePositionKeepingAllWordsIgnoringNBSP(lineBreakIterator, startPosition);
140     }
141 
142     if (lineBreakIterator.mode() == LineBreakIteratorMode::Default) {
143         if (breakNBSP)
144             return WebCore::nextBreakablePosition(lineBreakIterator, startPosition);
145         return nextBreakablePositionIgnoringNBSP(lineBreakIterator, startPosition);
146     }
147 
148     if (breakNBSP)
149         return nextBreakablePositionWithoutShortcut(lineBreakIterator, startPosition);
150     return nextBreakablePositionIgnoringNBSPWithoutShortcut(lineBreakIterator, startPosition);
151 }
