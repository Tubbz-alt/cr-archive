 28 #if PLATFORM(MAC)
 29 
 30 #include "FloatRect.h"
 31 #include "PlatformScreen.h"
 32 #include <wtf/RetainPtr.h>
 33 #include <wtf/text/WTFString.h>
 34 
 35 typedef struct CGColorSpace *CGColorSpaceRef;
 36 
 37 namespace WebCore {
 38 
 39 struct ScreenData {
 40     FloatRect screenAvailableRect;
 41     FloatRect screenRect;
 42     RetainPtr<CGColorSpaceRef> colorSpace;
 43     int screenDepth { 0 };
 44     int screenDepthPerComponent { 0 };
 45     bool screenSupportsExtendedColor { false };
 46     bool screenHasInvertedColors { false };
 47     bool screenIsMonochrome { false };
 48     bool screenSupportsHighDynamicRange { false };
 49     uint32_t displayMask { 0 };
 50     IORegistryGPUID gpuID { 0 };
 51 
 52     enum EncodedColorSpaceDataType {
 53         Null,
 54         ColorSpaceName,
 55         ColorSpaceData,
 56     };
 57 
 58     template<class Encoder> void encode(Encoder&) const;
 59     template<class Decoder> static Optional<ScreenData> decode(Decoder&);
 60 };
 61 
 62 typedef HashMap<PlatformDisplayID, ScreenData> ScreenDataMap;
 63 
 64 struct ScreenProperties {
 65     PlatformDisplayID primaryDisplayID { 0 };
 66     ScreenDataMap screenDataMap;
 67 
 68     template<class Encoder> void encode(Encoder&) const;
 69     template<class Decoder> static Optional<ScreenProperties> decode(Decoder&);
 70 };
 71 
 72 template<class Encoder>
 73 void ScreenProperties::encode(Encoder& encoder) const
 74 {
 75     encoder << primaryDisplayID;
 76     encoder << screenDataMap;
 77 }
 78 
 79 template<class Decoder>
 80 Optional<ScreenProperties> ScreenProperties::decode(Decoder& decoder)
 81 {
 82     Optional<PlatformDisplayID> primaryDisplayID;
 83     decoder >> primaryDisplayID;
 84     if (!primaryDisplayID)
 85         return WTF::nullopt;
 86 
 87     Optional<ScreenDataMap> screenDataMap;
 88     decoder >> screenDataMap;
 89     if (!screenDataMap)
 90         return WTF::nullopt;
 91 
 92     return { { *primaryDisplayID, WTFMove(*screenDataMap) } };
 93 }
 94 
 95 template<class Encoder>
 96 void ScreenData::encode(Encoder& encoder) const
 97 {
 98     encoder << screenAvailableRect << screenRect << screenDepth << screenDepthPerComponent << screenSupportsExtendedColor << screenHasInvertedColors << screenIsMonochrome << screenSupportsHighDynamicRange << displayMask << gpuID;
 99 
100     if (colorSpace) {
101         // Try to encode the name.
102         if (auto name = adoptCF(CGColorSpaceCopyName(colorSpace.get()))) {
103             encoder.encodeEnum(ColorSpaceName);
104             encoder << String(name.get());
105             return;
106         }
107 
108         // Failing that, just encode the ICC data.
109         if (auto profileData = adoptCF(CGColorSpaceCopyICCData(colorSpace.get()))) {
110             encoder.encodeEnum(ColorSpaceData);
111 
112             Vector<uint8_t> iccData;
113             iccData.append(CFDataGetBytePtr(profileData.get()), CFDataGetLength(profileData.get()));
114 
115             encoder << iccData;
116             return;
117         }
118     }
119 
120     // The color space was null or failed to be encoded.
121     encoder << Null;
122 }
123 
124 template<class Decoder>
125 Optional<ScreenData> ScreenData::decode(Decoder& decoder)
126 {
127     Optional<FloatRect> screenAvailableRect;
128     decoder >> screenAvailableRect;
129     if (!screenAvailableRect)
130         return WTF::nullopt;
131 
132     Optional<FloatRect> screenRect;
133     decoder >> screenRect;
134     if (!screenRect)
135         return WTF::nullopt;
136 
137     Optional<int> screenDepth;
138     decoder >> screenDepth;
139     if (!screenDepth)
140         return WTF::nullopt;
141 
142     Optional<int> screenDepthPerComponent;
143     decoder >> screenDepthPerComponent;
144     if (!screenDepthPerComponent)
145         return WTF::nullopt;
146 
147     Optional<bool> screenSupportsExtendedColor;
148     decoder >> screenSupportsExtendedColor;
149     if (!screenSupportsExtendedColor)
150         return WTF::nullopt;
151 
152     Optional<bool> screenHasInvertedColors;
153     decoder >> screenHasInvertedColors;
154     if (!screenHasInvertedColors)
155         return WTF::nullopt;
156 
157     Optional<bool> screenIsMonochrome;
158     decoder >> screenIsMonochrome;
159     if (!screenIsMonochrome)
160         return WTF::nullopt;
161 
162     Optional<bool> screenSupportsHighDynamicRange;
163     decoder >> screenSupportsHighDynamicRange;
164     if (!screenSupportsHighDynamicRange)
165         return WTF::nullopt;
166 
167     Optional<uint32_t> displayMask;
168     decoder >> displayMask;
169     if (!displayMask)
170         return WTF::nullopt;
171 
172     Optional<IORegistryGPUID> gpuID;
173     decoder >> gpuID;
174     if (!gpuID)
175         return WTF::nullopt;
176 
177     EncodedColorSpaceDataType dataType;
178     if (!decoder.decodeEnum(dataType))
179         return WTF::nullopt;
180 
181     RetainPtr<CGColorSpaceRef> cgColorSpace;
182     switch (dataType) {
183     case Null:
184         break;
185     case ColorSpaceName: {
186         Optional<String> colorSpaceName;
187         decoder >> colorSpaceName;
188         ASSERT(colorSpaceName);
189         if (!colorSpaceName)
190             return WTF::nullopt;
191 
192         cgColorSpace = adoptCF(CGColorSpaceCreateWithName(colorSpaceName->createCFString().get()));
193         break;
194     }
195     case ColorSpaceData: {
196         Optional<Vector<uint8_t>> iccData;
197         decoder >> iccData;
198         ASSERT(iccData);
199         if (!iccData)
200             return WTF::nullopt;
201 
202         auto colorSpaceData = adoptCF(CFDataCreate(kCFAllocatorDefault, iccData->data(), iccData->size()));
203         cgColorSpace = adoptCF(CGColorSpaceCreateWithICCData(colorSpaceData.get()));
204         break;
205     }
206     }
207 
208     return { { WTFMove(*screenAvailableRect), WTFMove(*screenRect), WTFMove(cgColorSpace), WTFMove(*screenDepth), WTFMove(*screenDepthPerComponent), WTFMove(*screenSupportsExtendedColor), WTFMove(*screenHasInvertedColors), WTFMove(*screenIsMonochrome), WTFMove(*screenSupportsHighDynamicRange), WTFMove(*displayMask), WTFMove(*gpuID) } };
209 }

211 } // namespace WebCore
