41 ModuleNamespaceAccessCase::ModuleNamespaceAccessCase(VM& vm, JSCell* owner, CacheableIdentifier identifier, JSModuleNamespaceObject* moduleNamespaceObject, JSModuleEnvironment* moduleEnvironment, ScopeOffset scopeOffset)
42     : Base(vm, owner, ModuleNamespaceLoad, identifier, invalidOffset, nullptr, ObjectPropertyConditionSet(), nullptr)
43     , m_scopeOffset(scopeOffset)
44 {
45     m_moduleNamespaceObject.set(vm, owner, moduleNamespaceObject);
46     m_moduleEnvironment.set(vm, owner, moduleEnvironment);
47 }
48 
49 std::unique_ptr<AccessCase> ModuleNamespaceAccessCase::create(VM& vm, JSCell* owner, CacheableIdentifier identifier, JSModuleNamespaceObject* moduleNamespaceObject, JSModuleEnvironment* moduleEnvironment, ScopeOffset scopeOffset)
50 {
51     return std::unique_ptr<AccessCase>(new ModuleNamespaceAccessCase(vm, owner, identifier, moduleNamespaceObject, moduleEnvironment, scopeOffset));
52 }
53 
54 ModuleNamespaceAccessCase::~ModuleNamespaceAccessCase()
55 {
56 }

58 std::unique_ptr<AccessCase> ModuleNamespaceAccessCase::clone() const
59 {
60     std::unique_ptr<ModuleNamespaceAccessCase> result(new ModuleNamespaceAccessCase(*this));
61     result->resetState();
62     return result;
63 }
64 
65 void ModuleNamespaceAccessCase::emit(AccessGenerationState& state, MacroAssembler::JumpList& fallThrough)
66 {
67     CCallHelpers& jit = *state.jit;
68     JSValueRegs valueRegs = state.valueRegs;
69     GPRReg baseGPR = state.baseGPR;
70 
71     fallThrough.append(
72         jit.branchPtr(
73             CCallHelpers::NotEqual,
74             baseGPR,
75             CCallHelpers::TrustedImmPtr(m_moduleNamespaceObject.get())));
76 
77     jit.loadValue(&m_moduleEnvironment->variableAt(m_scopeOffset), valueRegs);
78     state.failAndIgnore.append(jit.branchIfEmpty(valueRegs));
79     state.succeed();
80 }
