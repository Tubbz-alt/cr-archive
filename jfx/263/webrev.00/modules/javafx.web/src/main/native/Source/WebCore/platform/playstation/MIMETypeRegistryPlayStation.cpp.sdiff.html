61 String MIMETypeRegistry::getMIMETypeForExtension(const String& extension)
62 {
63     for (auto& entry : commonMediaTypes) {
64         if (equalIgnoringASCIICase(extension, entry.extension.characters()))
65             return entry.type;
66     }
67     return emptyString();
68 }
