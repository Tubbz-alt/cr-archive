46 static Lock providerIdLock;
47 
48 void SourceProvider::getID()
49 {
50     LockHolder lock(&providerIdLock);
51     if (!m_id) {
52         static intptr_t nextProviderID = 0;
53         m_id = ++nextProviderID;
54         RELEASE_ASSERT(m_id);
55     }
56 }
