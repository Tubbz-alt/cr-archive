28 namespace JSC {

30 enum ExitKind : uint8_t {
31     ExitKindUnset,
32     BadType, // We exited because a type prediction was wrong.
33     BadCell, // We exited because we made an incorrect assumption about what cell we would see. Usually used for function checks.
34     BadIdent, // We exited because we made an incorrect assumption about what identifier we would see. Usually used for cached Id check in get_by_val.
35     BadExecutable, // We exited because we made an incorrect assumption about what executable we would see.
36     BadCache, // We exited because an inline cache was wrong.
37     BadConstantCache, // We exited because a cache on a weak constant (usually a prototype) was wrong.
38     BadIndexingType, // We exited because an indexing type was wrong.
39     BadTypeInfoFlags, // We exited because we made an incorrect assumption about what TypeInfo flags we would see.
40     Overflow, // We exited because of overflow.
41     NegativeZero, // We exited because we encountered negative zero.
42     Int52Overflow, // We exited because of an Int52 overflow.
43     StoreToHole, // We had a store to a hole.
44     LoadFromHole, // We had a load from a hole.
45     OutOfBounds, // We had an out-of-bounds access to an array.
46     InadequateCoverage, // We exited because we ended up in code that didn't have profiling coverage.
47     ArgumentsEscaped, // We exited because arguments escaped but we didn't expect them to.
48     ExoticObjectMode, // We exited because some exotic object that we were accessing was in an exotic mode (like Arguments with slow arguments).
49     VarargsOverflow, // We exited because a varargs call passed more arguments than we expected.
50     TDZFailure, // We exited because we were in the TDZ and accessed the variable.
51     HoistingFailed, // Something that was hoisted exited. So, assume that hoisting is a bad idea.
52     Uncountable, // We exited for none of the above reasons, and we should not count it. Most uses of this should be viewed as a FIXME.
53     UncountableInvalidation, // We exited because the code block was invalidated; this means that we've already counted the reasons why the code block was invalidated.
54     WatchdogTimerFired, // We exited because we need to service the watchdog timer.
55     DebuggerEvent, // We exited because we need to service the debugger.
56     ExceptionCheck, // We exited because a direct exception check showed that we threw an exception from a C call.
57     GenericUnwind, // We exited because a we arrived at this OSR exit from genericUnwind.
58 };
59 
60 const char* exitKindToString(ExitKind);
61 bool exitKindMayJettison(ExitKind);
