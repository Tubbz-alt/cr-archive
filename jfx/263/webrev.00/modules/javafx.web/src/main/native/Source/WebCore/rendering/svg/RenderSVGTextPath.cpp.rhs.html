38 RenderSVGTextPath::RenderSVGTextPath(SVGTextPathElement& element, RenderStyle&& style)
39     : RenderSVGInline(element, WTFMove(style))
40 {
41 }
42 
43 SVGTextPathElement& RenderSVGTextPath::textPathElement() const
44 {
45     return downcast<SVGTextPathElement>(RenderSVGInline::graphicsElement());
46 }
47 
48 SVGGeometryElement* RenderSVGTextPath::targetElement() const
49 {
50     auto target = SVGURIReference::targetElementFromIRIString(textPathElement().href(), textPathElement().treeScope());
51     if (!is<SVGGeometryElement>(target.element))
52         return nullptr;
53 
54     return downcast<SVGGeometryElement>(target.element.get());
55 }
56 
57 Path RenderSVGTextPath::layoutPath() const
58 {
59     auto element = targetElement();
60     if (!is<SVGGeometryElement>(element))
61         return Path();
62 
63     Path path = pathFromGraphicsElement(element);
64 
65     // Spec:  The transform attribute on the referenced 'path' element represents a
66     // supplemental transformation relative to the current user coordinate system for
67     // the current 'text' element, including any adjustments to the current user coordinate
68     // system due to a possible transform attribute on the current 'text' element.
69     // http://www.w3.org/TR/SVG/text.html#TextPathElement
70     path.transform(element->animatedLocalTransform());
71     return path;
72 }
73 
74 const SVGLengthValue& RenderSVGTextPath::startOffset() const
75 {
76     return textPathElement().startOffset();
77 }
78 
79 bool RenderSVGTextPath::exactAlignment() const
80 {
81     return textPathElement().spacing() == SVGTextPathSpacingExact;
82 }
83 
84 bool RenderSVGTextPath::stretchMethod() const
85 {
86     return textPathElement().method() == SVGTextPathMethodStretch;
87 }
