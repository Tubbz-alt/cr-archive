 31 bool WillChangeData::operator==(const WillChangeData& other) const
 32 {
 33     return m_animatableFeatures == other.m_animatableFeatures;
 34 }
 35 
 36 bool WillChangeData::containsScrollPosition() const
 37 {
 38     for (const auto& feature : m_animatableFeatures) {
 39         if (feature.feature() == ScrollPosition)
 40             return true;
 41     }
 42     return false;
 43 }
 44 
 45 bool WillChangeData::containsContents() const
 46 {
 47     for (const auto& feature : m_animatableFeatures) {
 48         if (feature.feature() == Contents)
 49             return true;
 50     }
 51     return false;
 52 }
 53 
 54 bool WillChangeData::containsProperty(CSSPropertyID property) const
 55 {
 56     for (const auto& feature : m_animatableFeatures) {
 57         if (feature.property() == property)
 58             return true;
 59     }
 60     return false;
 61 }

 63 // "If any non-initial value of a property would create a stacking context on the element,
 64 // specifying that property in will-change must create a stacking context on the element."
 65 bool WillChangeData::propertyCreatesStackingContext(CSSPropertyID property)
 66 {
 67     switch (property) {
 68     case CSSPropertyPerspective:
 69     case CSSPropertyTransform:
 70     case CSSPropertyTransformStyle:
 71     case CSSPropertyWebkitTransformStyle:
 72     case CSSPropertyClipPath:
 73     case CSSPropertyMask:
 74     case CSSPropertyOpacity:
 75     case CSSPropertyPosition:
 76     case CSSPropertyZIndex:
 77     case CSSPropertyWebkitBoxReflect:
 78 #if ENABLE(CSS_COMPOSITING)
 79     case CSSPropertyMixBlendMode:
 80     case CSSPropertyIsolation:
 81 #endif
 82     case CSSPropertyFilter:
 83 #if ENABLE(FILTERS_LEVEL_2)
 84     case CSSPropertyWebkitBackdropFilter:
 85 #endif
 86     case CSSPropertyWebkitMask:
 87     case CSSPropertyWebkitMaskImage:
 88     case CSSPropertyWebkitMaskBoxImage:
 89 #if ENABLE(OVERFLOW_SCROLLING_TOUCH)
 90     case CSSPropertyWebkitOverflowScrolling:
 91 #endif
 92         return true;
 93     default:
 94         return false;
 95     }
 96 }
 97 
 98 static bool propertyTriggersCompositing(CSSPropertyID property)
 99 {
100     switch (property) {
101     case CSSPropertyOpacity:
102     case CSSPropertyFilter:
103 #if ENABLE(FILTERS_LEVEL_2)
104     case CSSPropertyWebkitBackdropFilter:
105 #endif
106         return true;
107     default:
108         return false;
109     }
110 }
111 
112 static bool propertyTriggersCompositingOnBoxesOnly(CSSPropertyID property)
113 {
114     // Don't trigger for perspective and transform-style, because those
115     // only do compositing if they have a 3d-transformed descendant and
116     // we don't want to do compositing all the time.
117     // Similarly, we don't want -webkit-overflow-scrolling-touch to
118     // always composite if there's no scrollable overflow.
119     switch (property) {
120     case CSSPropertyTransform:
121         return true;
122     default:
123         return false;
124     }
125 }
126 
127 void WillChangeData::addFeature(Feature feature, CSSPropertyID propertyID)
128 {
129     ASSERT(feature == Property || propertyID == CSSPropertyInvalid);
130     m_animatableFeatures.append(AnimatableFeature(feature, propertyID));
131 
132     m_canCreateStackingContext |= propertyCreatesStackingContext(propertyID);
133 
134     m_canTriggerCompositingOnInline |= propertyTriggersCompositing(propertyID);
135     m_canTriggerCompositing |= m_canTriggerCompositingOnInline | propertyTriggersCompositingOnBoxesOnly(propertyID);
136 }
137 
138 WillChangeData::FeaturePropertyPair WillChangeData::featureAt(size_t index) const
139 {
140     if (index >= m_animatableFeatures.size())
141         return FeaturePropertyPair(Invalid, CSSPropertyInvalid);
142 
143     return m_animatableFeatures[index].featurePropertyPair();
144 }
