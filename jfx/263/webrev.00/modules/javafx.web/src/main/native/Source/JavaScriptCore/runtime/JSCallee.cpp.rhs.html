50 void JSCallee::finishCreation(VM& vm)
51 {
52     Base::finishCreation(vm);
53     ASSERT(inherits(vm, info()));
54 }

56 void JSCallee::visitChildren(JSCell* cell, SlotVisitor& visitor)
57 {
58     JSCallee* thisObject = jsCast<JSCallee*>(cell);
59     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
60     Base::visitChildren(thisObject, visitor);
61 
62     visitor.append(thisObject->m_scope);
63 }
