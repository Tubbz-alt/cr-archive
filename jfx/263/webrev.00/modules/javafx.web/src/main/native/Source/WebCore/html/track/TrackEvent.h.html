26 #pragma once

28 #if ENABLE(VIDEO_TRACK)
29 
30 #include "AudioTrack.h"
31 #include "Event.h"
32 #include "TextTrack.h"
33 #include "VideoTrack.h"
34 
35 namespace WebCore {
36 
37 class TrackEvent final : public Event {
38     WTF_MAKE_ISO_ALLOCATED(TrackEvent);
39 public:
40     virtual ~TrackEvent();
41 
42     static Ref<TrackEvent> create(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, Ref<TrackBase>&& track)
43     {
44         return adoptRef(*new TrackEvent(type, canBubble, cancelable, WTFMove(track)));
45     }
46 
47     using TrackEventTrack = Variant<RefPtr<VideoTrack>, RefPtr<AudioTrack>, RefPtr<TextTrack>>;
48 
49     struct Init : public EventInit {
50         Optional<TrackEventTrack> track;
51     };
52 
53     static Ref<TrackEvent> create(const AtomString& type, Init&& initializer, IsTrusted isTrusted = IsTrusted::No)
54     {
55         return adoptRef(*new TrackEvent(type, WTFMove(initializer), isTrusted));
56     }
57 
58     Optional<TrackEventTrack> track() const { return m_track; }
59 
60 private:
61     TrackEvent(const AtomString& type, CanBubble, IsCancelable, Ref<TrackBase>&&);
62     TrackEvent(const AtomString& type, Init&& initializer, IsTrusted);
63 
64     EventInterface eventInterface() const override;
65 
66     Optional<TrackEventTrack> m_track;
67 };
68 
69 } // namespace WebCore
70 
71 #endif // ENABLE(VIDEO_TRACK)
