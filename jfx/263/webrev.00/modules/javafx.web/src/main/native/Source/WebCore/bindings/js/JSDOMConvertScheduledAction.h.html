28 #include "IDLTypes.h"
29 #include "JSDOMConvertBase.h"
30 #include "JSDOMConvertStrings.h"
31 #include "ScheduledAction.h"
32 
33 namespace WebCore {
34 
35 template<> struct Converter<IDLScheduledAction> : DefaultConverter<IDLScheduledAction> {
36     static std::unique_ptr<ScheduledAction> convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value, JSDOMGlobalObject& globalObject)
37     {
38         JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
39         auto scope = DECLARE_THROW_SCOPE(vm);
40 
41         JSC::CallData callData;
42         if (getCallData(vm, value, callData) == JSC::CallType::None) {
43             auto code = Converter<IDLDOMString>::convert(lexicalGlobalObject, value);
44             RETURN_IF_EXCEPTION(scope, nullptr);
45             return ScheduledAction::create(globalObject.world(), WTFMove(code));
46         }
47 
48         // The value must be an object at this point because no non-object values are callable.
49         ASSERT(value.isObject());
50         return ScheduledAction::create(globalObject.world(), JSC::Strong<JSC::Unknown> { vm, JSC::asObject(value) });
51     }
52 };
53 
54 }
