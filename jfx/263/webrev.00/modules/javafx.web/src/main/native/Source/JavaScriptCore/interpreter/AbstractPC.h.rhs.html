26 #pragma once
27 
28 #include "MacroAssemblerCodeRef.h"
29 
30 namespace JSC {
31 
32 class CallFrame;
33 class VM;

34 struct Instruction;
35 
36 class AbstractPC {
37 public:
38     AbstractPC()
39         : m_mode(None)
40     {
41     }
42 
43     AbstractPC(VM&, CallFrame*);
44 
45 #if ENABLE(JIT)
46     AbstractPC(ReturnAddressPtr ptr)
47         : m_pointer(ptr.value())
48         , m_mode(JIT)
49     {
50     }
51 
52     bool hasJITReturnAddress() const { return m_mode == JIT; }
53     ReturnAddressPtr jitReturnAddress() const
54     {
55         ASSERT(hasJITReturnAddress());
56         return ReturnAddressPtr(m_pointer);
57     }
58 #endif
59 
60     bool isSet() const { return m_mode != None; }
61     bool operator!() const { return !isSet(); }
62 
63 private:
64 #if ENABLE(JIT)
65     const void* m_pointer { nullptr };
66 #endif
67 
68     enum Mode { None, JIT, Interpreter };
69     Mode m_mode;
70 };
71 
72 } // namespace JSC
