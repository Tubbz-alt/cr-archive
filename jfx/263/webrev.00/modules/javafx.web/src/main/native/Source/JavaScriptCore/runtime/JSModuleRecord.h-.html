28 #include "AbstractModuleRecord.h"

32 namespace JSC {
33 
34 class ModuleProgramExecutable;
35 
36 // Based on the Source Text Module Record
37 // http://www.ecma-international.org/ecma-262/6.0/#sec-source-text-module-records
38 class JSModuleRecord final : public AbstractModuleRecord {
39     friend class LLIntOffsetsExtractor;
40 public:
41     typedef AbstractModuleRecord Base;
42 
43     DECLARE_EXPORT_INFO;
44 
45     static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
46     static JSModuleRecord* create(ExecState*, VM&, Structure*, const Identifier&, const SourceCode&, const VariableEnvironment&, const VariableEnvironment&);
47 
48     void link(ExecState*, JSValue scriptFetcher);
49     JS_EXPORT_PRIVATE JSValue evaluate(ExecState*);
50 
51     const SourceCode& sourceCode() const { return m_sourceCode; }
52     const VariableEnvironment& declaredVariables() const { return m_declaredVariables; }
53     const VariableEnvironment& lexicalVariables() const { return m_lexicalVariables; }
54 
55 private:
56     JSModuleRecord(VM&, Structure*, const Identifier&, const SourceCode&, const VariableEnvironment&, const VariableEnvironment&);
57 
58     void finishCreation(ExecState*, VM&);
59 
60     static void visitChildren(JSCell*, SlotVisitor&);
61     static void destroy(JSCell*);
62 
63     void instantiateDeclarations(ExecState*, ModuleProgramExecutable*, JSValue scriptFetcher);
64 
65     SourceCode m_sourceCode;
66     VariableEnvironment m_declaredVariables;
67     VariableEnvironment m_lexicalVariables;
68     WriteBarrier<ModuleProgramExecutable> m_moduleProgramExecutable;
69 };
70 
71 } // namespace JSC
