26 #pragma once
27 
28 #include "Color.h"
29 #include "DecodingOptions.h"

30 #include "ImageOrientation.h"
31 #include "ImageTypes.h"
32 #include "IntSize.h"
33 #include "NativeImage.h"
34 #include <wtf/Seconds.h>
35 
36 namespace WebCore {
37 
38 class ImageFrame {
39     friend class ImageSource;
40 public:
41     enum class Caching { Metadata, MetadataAndImage };
42 
43     ImageFrame();
44     ImageFrame(const ImageFrame& other) { operator=(other); }
45 
46     ~ImageFrame();
47 
48     static const ImageFrame& defaultFrame();
49 
50     ImageFrame& operator=(const ImageFrame& other);
51 
52     unsigned clearImage();
53     unsigned clear();
54 
55     void setDecodingStatus(DecodingStatus);
56     DecodingStatus decodingStatus() const;
57 
58     bool isInvalid() const { return m_decodingStatus == DecodingStatus::Invalid; }
59     bool isPartial() const { return m_decodingStatus == DecodingStatus::Partial; }
60     bool isComplete() const { return m_decodingStatus == DecodingStatus::Complete; }
61 
62     IntSize size() const;

63     unsigned frameBytes() const { return hasNativeImage() ? (size().area() * sizeof(uint32_t)).unsafeGet() : 0; }
64     SubsamplingLevel subsamplingLevel() const { return m_subsamplingLevel; }
65 
66     NativeImagePtr nativeImage() const { return m_nativeImage; }
67 
68     void setOrientation(ImageOrientation orientation) { m_orientation = orientation; };
69     ImageOrientation orientation() const { return m_orientation; }
70 
71     void setDuration(const Seconds& duration) { m_duration = duration; }
72     Seconds duration() const { return m_duration; }
73 
74     void setHasAlpha(bool hasAlpha) { m_hasAlpha = hasAlpha; }
75     bool hasAlpha() const { return !hasMetadata() || m_hasAlpha; }
76 
77     bool hasNativeImage(const Optional<SubsamplingLevel>& = { }) const;
78     bool hasFullSizeNativeImage(const Optional<SubsamplingLevel>& = { }) const;
79     bool hasDecodedNativeImageCompatibleWithOptions(const Optional<SubsamplingLevel>&, const DecodingOptions&) const;
80     bool hasMetadata() const { return !size().isEmpty(); }
81 
82     Color singlePixelSolidColor() const;
83 
84 private:
85     DecodingStatus m_decodingStatus { DecodingStatus::Invalid };
86     IntSize m_size;
87 
88     NativeImagePtr m_nativeImage;
89     SubsamplingLevel m_subsamplingLevel { SubsamplingLevel::Default };
90     DecodingOptions m_decodingOptions;
91 
92     ImageOrientation m_orientation { ImageOrientation::None };
93     Seconds m_duration;
94     bool m_hasAlpha { true };
95 };
96 
97 }
