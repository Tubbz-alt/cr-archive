26 #pragma once

28 #include "TrackPrivateBase.h"
29 
30 #if ENABLE(VIDEO_TRACK)
31 
32 namespace WebCore {
33 
34 class AudioTrackPrivate;
35 
36 class AudioTrackPrivateClient : public TrackPrivateBaseClient {
37 public:
38     virtual void enabledChanged(bool) = 0;
39 };
40 
41 class AudioTrackPrivate : public TrackPrivateBase {
42 public:
43     static Ref<AudioTrackPrivate> create()
44     {
45         return adoptRef(*new AudioTrackPrivate);
46     }
47 
48     void setClient(AudioTrackPrivateClient* client) { m_client = client; }
49     AudioTrackPrivateClient* client() const override { return m_client; }
50 
51     virtual void setEnabled(bool enabled)
52     {
53         if (m_enabled == enabled)
54             return;
55         m_enabled = enabled;
56         if (m_client)
57             m_client->enabledChanged(enabled);
58     }
59 
60     bool enabled() const { return m_enabled; }
61 
62     enum Kind { Alternative, Description, Main, MainDesc, Translation, Commentary, None };
63     virtual Kind kind() const { return None; }
64 
65 #if !RELEASE_LOG_DISABLED
66     const char* logClassName() const override { return "AudioTrackPrivate"; }
67 #endif
68 
69 protected:
70     AudioTrackPrivate() = default;
71 
72 private:
73     AudioTrackPrivateClient* m_client { nullptr };
74     bool m_enabled { false };
75 };
76 
77 } // namespace WebCore
