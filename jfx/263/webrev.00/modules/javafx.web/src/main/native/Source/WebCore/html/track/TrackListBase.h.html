28 #if ENABLE(VIDEO_TRACK)

30 #include "ContextDestructionObserver.h"
31 #include "EventTarget.h"
32 #include "GenericEventQueue.h"
33 #include <wtf/RefCounted.h>
34 #include <wtf/Vector.h>
35 #include <wtf/WeakPtr.h>
36 
37 namespace WebCore {
38 
39 class HTMLMediaElement;
40 class Element;
41 class TrackBase;
42 
43 class TrackListBase : public RefCounted<TrackListBase>, public EventTargetWithInlineData, public ContextDestructionObserver {
44     WTF_MAKE_ISO_ALLOCATED(TrackListBase);
45 public:
46     virtual ~TrackListBase();
47 
48     virtual unsigned length() const;
49     virtual bool contains(TrackBase&) const;
50     virtual void remove(TrackBase&, bool scheduleEvent = true);
51 
52     // EventTarget
53     EventTargetInterface eventTargetInterface() const override = 0;
54     using RefCounted<TrackListBase>::ref;
55     using RefCounted<TrackListBase>::deref;
56     ScriptExecutionContext* scriptExecutionContext() const final { return ContextDestructionObserver::scriptExecutionContext(); }
57 
58     virtual void clearElement();
59     Element* element() const;
60     WeakPtr<HTMLMediaElement> mediaElement() const { return m_element; }
61 
62     // Needs to be public so tracks can call it
63     void scheduleChangeEvent();
64     bool isChangeEventScheduled() const;
65 
66     bool isAnyTrackEnabled() const;
67 
68 protected:
69     TrackListBase(WeakPtr<HTMLMediaElement>, ScriptExecutionContext*);
70 
71     void scheduleAddTrackEvent(Ref<TrackBase>&&);
72     void scheduleRemoveTrackEvent(Ref<TrackBase>&&);
73 
74     Vector<RefPtr<TrackBase>> m_inbandTracks;
75 
76 private:
77     void scheduleTrackEvent(const AtomString& eventName, Ref<TrackBase>&&);
78 
79     // EventTarget
80     void refEventTarget() final { ref(); }
81     void derefEventTarget() final { deref(); }
82 
83     WeakPtr<HTMLMediaElement> m_element;
84 
85     UniqueRef<MainThreadGenericEventQueue> m_asyncEventQueue;
86 };
87 
88 } // namespace WebCore

90 #endif // ENABLE(VIDEO_TRACK)
