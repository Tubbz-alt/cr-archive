26 #pragma once
27 
28 #include "JSObject.h"
29 
30 namespace JSC {
31 
32 class JSStringIterator final : public JSNonFinalObject {
33 public:
34     typedef JSNonFinalObject Base;





35 
36     DECLARE_EXPORT_INFO;
37 























38     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
39     {
40         return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
41     }
42 
43     static JSStringIterator* create(ExecState* exec, Structure* structure, JSString* iteratedString)
44     {
45         VM& vm = exec->vm();
46         JSStringIterator* instance = new (NotNull, allocateCell<JSStringIterator>(vm.heap)) JSStringIterator(vm, structure);
47         instance->finishCreation(vm, structure->globalObject(), iteratedString);
48         return instance;
49     }
50 
51     JSValue iteratedValue(ExecState*) const;
52     JSStringIterator* clone(ExecState*);



53 
54 private:
55     JSStringIterator(VM& vm, Structure* structure)
56         : Base(vm, structure)
57     {
58     }
59 
60     void finishCreation(VM&, JSGlobalObject*, JSString* iteratedString);
61 };
62 
63 } // namespace JSC
