 36 // The following functions convert values to integers as per the WebIDL specification.
 37 // The conversion fails if the value cannot be converted to a number or, if EnforceRange is specified,
 38 // the value is outside the range of the destination integer type.

 74     static int8_t convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)

 76         return convertToInteger<int8_t>(lexicalGlobalObject, value);
 77     }

280 template<> struct Converter<IDLFloat> : DefaultConverter<IDLFloat> {
281 
282     static inline float convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::ThrowScope& scope, double number)
283     {
284         if (UNLIKELY(!std::isfinite(number)))
285             throwNonFiniteTypeError(lexicalGlobalObject, scope);
286         return static_cast<float>(number);
287     }
288 
289     static float convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
290     {
291         JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
292         auto scope = DECLARE_THROW_SCOPE(vm);
293         double number = value.toNumber(&lexicalGlobalObject);
294         if (UNLIKELY(!std::isfinite(number)))
295             throwNonFiniteTypeError(lexicalGlobalObject, scope);
296         return static_cast<float>(number);
297     }
298 };
