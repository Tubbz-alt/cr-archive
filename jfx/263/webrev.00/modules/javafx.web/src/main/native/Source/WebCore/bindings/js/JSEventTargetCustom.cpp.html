46 EventTarget* JSEventTarget::toWrapped(VM& vm, JSValue value)
47 {
48     if (value.inherits<JSWindowProxy>(vm))
49         return &jsCast<JSWindowProxy*>(asObject(value))->wrapped();
50     if (value.inherits<JSDOMWindow>(vm))
51         return &jsCast<JSDOMWindow*>(asObject(value))->wrapped();
52     if (value.inherits<JSWorkerGlobalScope>(vm))
53         return &jsCast<JSWorkerGlobalScope*>(asObject(value))->wrapped();
54     if (value.inherits<JSEventTarget>(vm))
55         return &jsCast<JSEventTarget*>(asObject(value))->wrapped();
56     return nullptr;
57 }
58 
59 std::unique_ptr<JSEventTargetWrapper> jsEventTargetCast(VM& vm, JSValue thisValue)
60 {
61     if (auto* target = jsDynamicCast<JSEventTarget*>(vm, thisValue))
62         return makeUnique<JSEventTargetWrapper>(target->wrapped(), *target);
63     if (auto* window = toJSDOMWindow(vm, thisValue))
64         return makeUnique<JSEventTargetWrapper>(window->wrapped(), *window);
65     if (auto* scope = toJSWorkerGlobalScope(vm, thisValue))
66         return makeUnique<JSEventTargetWrapper>(scope->wrapped(), *scope);
67     return nullptr;
68 }
