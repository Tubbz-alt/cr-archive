27 @globalPrivate
28 function asyncFunctionResume(generator, promise, sentValue, resumeMode)
29 {
30     "use strict";
31 
32     @assert(@isPromise(promise));
33 
34     var state = @getGeneratorInternalField(generator, @generatorFieldState);
35     var value = @undefined;
36 
37     try {
38         @putGeneratorInternalField(generator, @generatorFieldState, @GeneratorStateExecuting);
39         value = @getGeneratorInternalField(generator, @generatorFieldNext).@call(@getGeneratorInternalField(generator, @generatorFieldThis), generator, state, sentValue, resumeMode, @getGeneratorInternalField(generator, @generatorFieldFrame));
40         if (@getGeneratorInternalField(generator, @generatorFieldState) === @GeneratorStateExecuting) {
41             @resolvePromiseWithFirstResolvingFunctionCallCheck(promise, value);
42             return promise;
43         }
44     } catch (error) {
45         @rejectPromiseWithFirstResolvingFunctionCallCheck(promise, error);
46         return promise;
47     }
48 
49     var capturedGenerator = generator;
50     var capturedPromise = promise;
51     @resolveWithoutPromise(value,
52         function(value) { @asyncFunctionResume(capturedGenerator, capturedPromise, value, @GeneratorResumeModeNormal); },
53         function(error) { @asyncFunctionResume(capturedGenerator, capturedPromise, error, @GeneratorResumeModeThrow); });
54 
55     return promise;
56 }
