33 ServiceWorkerJobData::ServiceWorkerJobData(SWServerConnectionIdentifier connectionIdentifier, const DocumentOrWorkerIdentifier& localSourceContext)
34     : m_identifier { connectionIdentifier, ServiceWorkerJobIdentifier::generateThreadSafe() }
35 {
36     WTF::switchOn(localSourceContext, [&](DocumentIdentifier documentIdentifier) {
37         sourceContext = ServiceWorkerClientIdentifier { connectionIdentifier, documentIdentifier };
38     }, [&](ServiceWorkerIdentifier serviceWorkerIdentifier) {
39         sourceContext = serviceWorkerIdentifier;
40     });
41 }
42 
43 ServiceWorkerRegistrationKey ServiceWorkerJobData::registrationKey() const
44 {
45     URL scope = scopeURL;
46     scope.removeFragmentIdentifier();
47     return { SecurityOriginData { topOrigin }, WTFMove(scope) };
48 }
49 
50 ServiceWorkerJobData ServiceWorkerJobData::isolatedCopy() const
51 {
52     ServiceWorkerJobData result;
53     result.m_identifier = identifier();
54     result.sourceContext = sourceContext;
55     result.type = type;
56 
57     result.scriptURL = scriptURL.isolatedCopy();
58     result.clientCreationURL = clientCreationURL.isolatedCopy();
59     result.topOrigin = topOrigin.isolatedCopy();
60     result.scopeURL = scopeURL.isolatedCopy();
61     result.registrationOptions = registrationOptions.isolatedCopy();
62 
63     return result;
64 }

66 // https://w3c.github.io/ServiceWorker/#dfn-job-equivalent
67 bool ServiceWorkerJobData::isEquivalent(const ServiceWorkerJobData& job) const
68 {
69     if (type != job.type)
70         return false;
71 
72     switch (type) {
73     case ServiceWorkerJobType::Register:
74     case ServiceWorkerJobType::Update:
75         return scopeURL == job.scopeURL
76             && scriptURL == job.scriptURL
77             && registrationOptions.type == job.registrationOptions.type
78             && registrationOptions.updateViaCache == job.registrationOptions.updateViaCache;
79     case ServiceWorkerJobType::Unregister:
80         return scopeURL == job.scopeURL;
81     }
82     return false;
83 }
