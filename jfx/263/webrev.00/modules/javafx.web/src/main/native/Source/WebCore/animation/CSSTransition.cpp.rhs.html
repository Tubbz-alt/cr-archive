40 Ref<CSSTransition> CSSTransition::create(Element& owningElement, CSSPropertyID property, MonotonicTime generationTime, const Animation& backingAnimation, const RenderStyle* oldStyle, const RenderStyle& newStyle, Seconds delay, Seconds duration, const RenderStyle& reversingAdjustedStartStyle, double reversingShorteningFactor)
41 {
42     ASSERT(oldStyle);
43     auto result = adoptRef(*new CSSTransition(owningElement, property, generationTime, backingAnimation, *oldStyle, newStyle, reversingAdjustedStartStyle, reversingShorteningFactor));
44     result->initialize(oldStyle, newStyle);
45     result->setTimingProperties(delay, duration);
46 
47     InspectorInstrumentation::didCreateWebAnimation(result.get());
48 
49     return result;
50 }

52 CSSTransition::CSSTransition(Element& element, CSSPropertyID property, MonotonicTime generationTime, const Animation& backingAnimation, const RenderStyle& oldStyle, const RenderStyle& targetStyle, const RenderStyle& reversingAdjustedStartStyle, double reversingShorteningFactor)
53     : DeclarativeAnimation(element, backingAnimation)
54     , m_property(property)
55     , m_generationTime(generationTime)
56     , m_targetStyle(RenderStyle::clonePtr(targetStyle))
57     , m_currentStyle(RenderStyle::clonePtr(oldStyle))
58     , m_reversingAdjustedStartStyle(RenderStyle::clonePtr(reversingAdjustedStartStyle))
59     , m_reversingShorteningFactor(reversingShorteningFactor)
60 {
61 }

69 void CSSTransition::setTimingProperties(Seconds delay, Seconds duration)
70 {
71     suspendEffectInvalidation();
72 
73     // This method is only called from CSSTransition::create() where we're guaranteed to have an effect.
74     auto* animationEffect = effect();
75     ASSERT(animationEffect);
76 
77     // In order for CSS Transitions to be seeked backwards, they need to have their fill mode set to backwards
78     // such that the original CSS value applied prior to the transition is used for a negative current time.
79     animationEffect->setFill(FillMode::Backwards);
80     animationEffect->setDelay(delay);
81     animationEffect->setIterationDuration(duration);
82     animationEffect->setTimingFunction(backingAnimation().timingFunction());
83     animationEffect->updateStaticTimingProperties();
84     effectTimingDidChange();
85 
86     unsuspendEffectInvalidation();
87 }
