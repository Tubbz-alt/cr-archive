33 #if USE(JSVALUE64)
34 void clearArrayMemset(WriteBarrier<Unknown>* base, unsigned count)
35 {
36 #if CPU(X86_64) && COMPILER(GCC_COMPATIBLE)
37     uint64_t zero = 0;
38     asm volatile (
39         "rep stosq\n\t"
40         : "+D"(base), "+c"(count)
41         : "a"(zero)
42         : "memory"
43         );
44 #else // not CPU(X86_64)
45     memset(base, 0, count * sizeof(WriteBarrier<Unknown>));
46 #endif // generic CPU
47 }
48 
49 void clearArrayMemset(double* base, unsigned count)
50 {
51 #if CPU(X86_64) && COMPILER(GCC_COMPATIBLE)
52     uint64_t pnan = bitwise_cast<uint64_t>(PNaN);
53     asm volatile (
54         "rep stosq\n\t"
55         : "+D"(base), "+c"(count)
56         : "a"(pnan)
57         : "memory"
58         );
59 #else // not CPU(X86_64)
60     // Oh no, we can't actually do any better than this!
61     for (unsigned i = count; i--;)
62         base[i] = PNaN;
63 #endif // generic CPU
64 }
65 #endif // USE(JSVALUE64)
