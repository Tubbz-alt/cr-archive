 40 namespace WebCore {
 41 
 42 class SWServer;
 43 class SWServerWorker;
 44 enum class ServiceWorkerRegistrationState : uint8_t;
 45 enum class ServiceWorkerState : uint8_t;
 46 struct ExceptionData;
 47 struct ServiceWorkerContextData;
 48 struct ServiceWorkerFetchResult;
 49 
 50 class SWServerRegistration : public CanMakeWeakPtr<SWServerRegistration> {
 51     WTF_MAKE_FAST_ALLOCATED;
 52 public:
 53     SWServerRegistration(SWServer&, const ServiceWorkerRegistrationKey&, ServiceWorkerUpdateViaCache, const URL& scopeURL, const URL& scriptURL);
 54     ~SWServerRegistration();
 55 
 56     const ServiceWorkerRegistrationKey& key() const { return m_registrationKey; }
 57     ServiceWorkerRegistrationIdentifier identifier() const { return m_identifier; }
 58 
 59     SWServerWorker* getNewestWorker();
 60     WEBCORE_EXPORT ServiceWorkerRegistrationData data() const;
 61 
 62     void setLastUpdateTime(WallTime);
 63     WallTime lastUpdateTime() const { return m_lastUpdateTime; }
 64     bool isStale() const { return m_lastUpdateTime && (WallTime::now() - m_lastUpdateTime) > 86400_s; }
 65 
 66     void setUpdateViaCache(ServiceWorkerUpdateViaCache);
 67     ServiceWorkerUpdateViaCache updateViaCache() const { return m_updateViaCache; }
 68 
 69     void updateRegistrationState(ServiceWorkerRegistrationState, SWServerWorker*);
 70     void updateWorkerState(SWServerWorker&, ServiceWorkerState);
 71     void fireUpdateFoundEvent();
 72 
 73     void addClientServiceWorkerRegistration(SWServerConnectionIdentifier);
 74     void removeClientServiceWorkerRegistration(SWServerConnectionIdentifier);
 75 
 76     void setPreInstallationWorker(SWServerWorker*);
 77     SWServerWorker* preInstallationWorker() const { return m_preInstallationWorker.get(); }
 78     SWServerWorker* installingWorker() const { return m_installingWorker.get(); }
 79     SWServerWorker* waitingWorker() const { return m_waitingWorker.get(); }
 80     SWServerWorker* activeWorker() const { return m_activeWorker.get(); }
 81 
 82     MonotonicTime creationTime() const { return m_creationTime; }
 83 
 84     bool hasClientsUsingRegistration() const { return !m_clientsUsingRegistration.isEmpty(); }
 85     void addClientUsingRegistration(const ServiceWorkerClientIdentifier&);
 86     void removeClientUsingRegistration(const ServiceWorkerClientIdentifier&);
 87     void unregisterServerConnection(SWServerConnectionIdentifier);
 88 
 89     void notifyClientsOfControllerChange();
 90     void controlClient(ServiceWorkerClientIdentifier);
 91 
 92     void clear();
 93     bool tryClear();
 94     void tryActivate();
 95     void didFinishActivation(ServiceWorkerIdentifier);
 96 
 97     bool isUnregistered() const;
 98 
 99     void forEachConnection(const WTF::Function<void(SWServer::Connection&)>&);
100 
101     WEBCORE_EXPORT bool shouldSoftUpdate(const FetchOptions&) const;
102     WEBCORE_EXPORT void scheduleSoftUpdate();
103     static constexpr Seconds softUpdateDelay { 1_s };
104 
105     URL scopeURLWithoutFragment() const { return m_scopeURL; }
106     URL scriptURL() const { return m_scriptURL; }
107 
108 private:
109     void activate();
110     void handleClientUnload();
111     void softUpdate();
112 
113     ServiceWorkerRegistrationIdentifier m_identifier;
114     ServiceWorkerRegistrationKey m_registrationKey;
115     ServiceWorkerUpdateViaCache m_updateViaCache;
116     URL m_scopeURL;
117     URL m_scriptURL;
118 
119     RefPtr<SWServerWorker> m_preInstallationWorker; // Implementation detail, not part of the specification.
120     RefPtr<SWServerWorker> m_installingWorker;
121     RefPtr<SWServerWorker> m_waitingWorker;
122     RefPtr<SWServerWorker> m_activeWorker;
123 
124     WallTime m_lastUpdateTime;
125 
126     HashCountedSet<SWServerConnectionIdentifier> m_connectionsWithClientRegistrations;
127     SWServer& m_server;
128 
129     MonotonicTime m_creationTime;
130     HashMap<SWServerConnectionIdentifier, HashSet<DocumentIdentifier>> m_clientsUsingRegistration;
131 
132     WebCore::Timer m_softUpdateTimer;
133 };
134 
135 } // namespace WebCore
