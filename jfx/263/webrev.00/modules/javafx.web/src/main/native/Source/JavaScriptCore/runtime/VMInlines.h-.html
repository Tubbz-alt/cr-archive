36 bool VM::ensureStackCapacityFor(Register* newTopOfStack)
37 {
38 #if !ENABLE(C_LOOP)
39     ASSERT(Thread::current().stack().isGrowingDownward());
40     return newTopOfStack >= m_softStackLimit;
41 #else
42     return ensureStackCapacityForCLoop(newTopOfStack);
43 #endif
44 
45 }
46 
47 bool VM::isSafeToRecurseSoft() const
48 {
49     bool safe = isSafeToRecurse(m_softStackLimit);
50 #if ENABLE(C_LOOP)
51     safe = safe && isSafeToRecurseSoftCLoop();
52 #endif
53     return safe;
54 }

56 template<typename Func>
57 void VM::logEvent(CodeBlock* codeBlock, const char* summary, const Func& func)
58 {
59     if (LIKELY(!m_perBytecodeProfiler))
60         return;
61 
62     m_perBytecodeProfiler->logEvent(codeBlock, summary, func());
63 }

65 inline CallFrame* VM::topJSCallFrame() const
66 {
67     CallFrame* frame = topCallFrame;
68     if (UNLIKELY(!frame))
69         return frame;
70     if (LIKELY(!frame->isWasmFrame() && !frame->isStackOverflowFrame()))
71         return frame;
72     EntryFrame* entryFrame = topEntryFrame;
73     do {
74         frame = frame->callerFrame(entryFrame);
75         ASSERT(!frame || !frame->isStackOverflowFrame());
76     } while (frame && frame->isWasmFrame());
77     return frame;
78 }
