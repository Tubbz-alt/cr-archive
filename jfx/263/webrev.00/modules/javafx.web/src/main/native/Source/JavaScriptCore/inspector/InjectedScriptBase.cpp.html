 62 bool InjectedScriptBase::hasAccessToInspectedScriptState() const
 63 {
 64     return m_environment && m_environment->canAccessInspectedScriptState(m_injectedScriptObject.globalObject());

 72 Expected<JSC::JSValue, NakedPtr<JSC::Exception>> InjectedScriptBase::callFunctionWithEvalEnabled(Deprecated::ScriptFunctionCall& function) const
 73 {
 74     JSC::JSGlobalObject* globalObject = m_injectedScriptObject.globalObject();
 75     JSC::DebuggerEvalEnabler evalEnabler(globalObject);
 76     return function.call();
 77 }
 78 
 79 Ref<JSON::Value> InjectedScriptBase::makeCall(Deprecated::ScriptFunctionCall& function)
 80 {
 81     if (hasNoValue() || !hasAccessToInspectedScriptState())
 82         return JSON::Value::null();
 83 
 84     auto result = callFunctionWithEvalEnabled(function);
 85     if (!result)
 86         return JSON::Value::create("Exception while making a call.");
 87 
 88     RefPtr<JSON::Value> resultJSONValue = toInspectorValue(m_injectedScriptObject.globalObject(), result.value());
 89     if (!resultJSONValue)
 90         return JSON::Value::create(makeString("Object has too long reference chain (must not be longer than ", JSON::Value::maxDepth, ')'));
 91 
 92     return resultJSONValue.releaseNonNull();
 93 }
 94 
 95 void InjectedScriptBase::makeEvalCall(ErrorString& errorString, Deprecated::ScriptFunctionCall& function, RefPtr<Protocol::Runtime::RemoteObject>& out_resultObject, Optional<bool>& out_wasThrown, Optional<int>& out_savedResultIndex)
 96 {
 97     checkCallResult(errorString, makeCall(function), out_resultObject, out_wasThrown, out_savedResultIndex);
 98 }

100 void InjectedScriptBase::makeAsyncCall(Deprecated::ScriptFunctionCall& function, AsyncCallCallback&& callback)
101 {
102     if (hasNoValue() || !hasAccessToInspectedScriptState()) {
103         checkAsyncCallResult(JSON::Value::null(), callback);
104         return;
105     }
106 
107     auto* globalObject = m_injectedScriptObject.globalObject();
108     JSC::VM& vm = globalObject->vm();
109 
110     JSC::JSNativeStdFunction* jsFunction = nullptr;
111     {
112         JSC::JSLockHolder locker(vm);
113 
114         jsFunction = JSC::JSNativeStdFunction::create(vm, globalObject, 1, String(), [&, callback = WTFMove(callback)] (JSC::JSGlobalObject* globalObject, JSC::CallFrame* callFrame) {
115             if (!callFrame)
116                 checkAsyncCallResult(JSON::Value::create("Exception while making a call."), callback);
117             else if (auto resultJSONValue = toInspectorValue(globalObject, callFrame->argument(0)))
118                 checkAsyncCallResult(resultJSONValue, callback);
119             else
120                 checkAsyncCallResult(JSON::Value::create(makeString("Object has too long reference chain (must not be longer than ", JSON::Value::maxDepth, ')')), callback);
121             return JSC::JSValue::encode(JSC::jsUndefined());
122         });
123     }
124 
125     function.appendArgument(JSC::JSValue(jsFunction));
126 
127     auto result = callFunctionWithEvalEnabled(function);
128     ASSERT_UNUSED(result, result.value().isUndefined());
129 
130     ASSERT(result);
131     if (!result) {
132         // Since `callback` is moved above, we can't call it if there's an exception while trying to
133         // execute the `JSNativeStdFunction` inside InjectedScriptSource.js.
134         jsFunction->function()(globalObject, nullptr);
135     }
136 }

138 void InjectedScriptBase::checkCallResult(ErrorString& errorString, RefPtr<JSON::Value> result, RefPtr<Protocol::Runtime::RemoteObject>& out_resultObject, Optional<bool>& out_wasThrown, Optional<int>& out_savedResultIndex)
139 {
140     if (!result) {
141         errorString = "Internal error: result value is empty"_s;
142         return;
143     }
144 
145     if (result->type() == JSON::Value::Type::String) {
146         result->asString(errorString);
147         ASSERT(errorString.length());
148         return;
149     }
150 
151     RefPtr<JSON::Object> resultTuple;
152     if (!result->asObject(resultTuple)) {
153         errorString = "Internal error: result is not an Object"_s;
154         return;
155     }
156 
157     RefPtr<JSON::Object> resultObject;
158     if (!resultTuple->getObject("result"_s, resultObject)) {
159         errorString = "Internal error: result is not a pair of value and wasThrown flag"_s;
160         return;
161     }
162 
163     bool wasThrown = false;
164     if (!resultTuple->getBoolean("wasThrown"_s, wasThrown)) {
165         errorString = "Internal error: result is not a pair of value and wasThrown flag"_s;
166         return;
167     }
168 
169     out_resultObject = BindingTraits<Protocol::Runtime::RemoteObject>::runtimeCast(resultObject);
170 
171     if (wasThrown)
172         out_wasThrown = wasThrown;
173 
174     int savedResultIndex;
175     if (resultTuple->getInteger("savedResultIndex"_s, savedResultIndex))
176         out_savedResultIndex = savedResultIndex;
177 }
178 
179 void InjectedScriptBase::checkAsyncCallResult(RefPtr<JSON::Value> result, const AsyncCallCallback& callback)
180 {
181     ErrorString errorString;
182     RefPtr<Protocol::Runtime::RemoteObject> resultObject;
183     Optional<bool> wasThrown;
184     Optional<int> savedResultIndex;
185 
186     checkCallResult(errorString, result, resultObject, wasThrown, savedResultIndex);
187 
188     callback(errorString, WTFMove(resultObject), wasThrown, savedResultIndex);
189 }
