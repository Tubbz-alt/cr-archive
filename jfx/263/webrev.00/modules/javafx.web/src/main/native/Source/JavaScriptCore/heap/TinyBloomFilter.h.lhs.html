26 #pragma once
27 
28 namespace JSC {
29 
30 typedef uintptr_t Bits;
31 
32 class TinyBloomFilter {
33 public:
34     TinyBloomFilter();

35 
36     void add(Bits);
37     void add(TinyBloomFilter&);
38     bool ruleOut(Bits) const; // True for 0.
39     void reset();

40 
41 private:
42     Bits m_bits;
43 };
44 
45 inline TinyBloomFilter::TinyBloomFilter()
46     : m_bits(0)
47 {
48 }
49 
50 inline void TinyBloomFilter::add(Bits bits)
51 {
52     m_bits |= bits;
53 }
54 
55 inline void TinyBloomFilter::add(TinyBloomFilter& other)
56 {
57     m_bits |= other.m_bits;
58 }
59 
60 inline bool TinyBloomFilter::ruleOut(Bits bits) const
61 {
62     if (!bits)
63         return true;
64 
65     if ((bits & m_bits) != bits)
66         return true;
67 
68     return false;
69 }
70 
71 inline void TinyBloomFilter::reset()
72 {
73     m_bits = 0;
74 }
75 
76 } // namespace JSC
