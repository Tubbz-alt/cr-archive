 26 #pragma once
 27 
 28 #if ENABLE(CONTENT_FILTERING)
 29 
 30 #include "CachedResourceHandle.h"
 31 #include "PlatformContentFilter.h"
 32 #include "ResourceError.h"
 33 #include <functional>
 34 #include <wtf/Forward.h>
 35 #include <wtf/UniqueRef.h>
 36 
 37 namespace WebCore {
 38 
 39 class CachedRawResource;
 40 class ContentFilterClient;
 41 class DocumentLoader;
 42 class ResourceRequest;
 43 class ResourceResponse;
 44 class SubstituteData;
 45 
 46 class ContentFilter {
 47     WTF_MAKE_FAST_ALLOCATED;
 48     WTF_MAKE_NONCOPYABLE(ContentFilter);
 49 
 50 public:
 51     template <typename T> static void addType() { types().append(type<T>()); }
 52 
 53     static std::unique_ptr<ContentFilter> create(ContentFilterClient&);
 54     ~ContentFilter();
 55 
 56     static const char* urlScheme() { return "x-apple-content-filter"; }
 57 
 58     void startFilteringMainResource(CachedRawResource&);
 59     void stopFilteringMainResource();
 60 
 61     bool continueAfterWillSendRequest(ResourceRequest&, const ResourceResponse&);
 62     bool continueAfterResponseReceived(const ResourceResponse&);
 63     bool continueAfterDataReceived(const char* data, int length);
 64     bool continueAfterNotifyFinished(CachedResource&);
 65 
 66     static bool continueAfterSubstituteDataRequest(const DocumentLoader& activeLoader, const SubstituteData&);
 67     bool willHandleProvisionalLoadFailure(const ResourceError&) const;
 68     void handleProvisionalLoadFailure(const ResourceError&);
 69 
 70 private:
 71     using State = PlatformContentFilter::State;
 72 
 73     struct Type {
 74         Function<UniqueRef<PlatformContentFilter>()> create;
 75     };
 76     template <typename T> static Type type();
 77     WEBCORE_EXPORT static Vector<Type>& types();
 78 
 79     using Container = Vector<UniqueRef<PlatformContentFilter>>;
 80     friend std::unique_ptr<ContentFilter> std::make_unique<ContentFilter>(Container&&, ContentFilterClient&);
 81     ContentFilter(Container&&, ContentFilterClient&);
 82 
 83     template <typename Function> void forEachContentFilterUntilBlocked(Function&&);
 84     void didDecide(State);
 85     void deliverResourceData(CachedResource&);
 86 
 87     Container m_contentFilters;
 88     ContentFilterClient& m_client;
 89     CachedResourceHandle<CachedRawResource> m_mainResource;
 90     const PlatformContentFilter* m_blockingContentFilter { nullptr };
 91     State m_state { State::Stopped };
 92     ResourceError m_blockedError;
 93     bool m_isLoadingBlockedPage { false };
 94 };
 95 
 96 template <typename T>
 97 ContentFilter::Type ContentFilter::type()
 98 {
 99     static_assert(std::is_base_of<PlatformContentFilter, T>::value, "Type must be a PlatformContentFilter.");
100     return { T::create };
101 }
102 
103 } // namespace WebCore
104 
105 #endif // ENABLE(CONTENT_FILTERING)
