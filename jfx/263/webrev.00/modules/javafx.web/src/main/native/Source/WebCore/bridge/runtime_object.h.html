 26 #ifndef KJS_RUNTIME_OBJECT_H
 27 #define KJS_RUNTIME_OBJECT_H
 28 
 29 #include "BridgeJSC.h"
 30 #include <JavaScriptCore/JSGlobalObject.h>
 31 
 32 namespace JSC {
 33 namespace Bindings {
 34 
 35 class WEBCORE_EXPORT RuntimeObject : public JSNonFinalObject {
 36 public:
 37     using Base = JSNonFinalObject;
 38     static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetPropertyNames | OverridesGetCallData;
 39     static constexpr bool needsDestruction = true;
 40 
 41     template<typename CellType, JSC::SubspaceAccess>
 42     static IsoSubspace* subspaceFor(JSC::VM& vm)
 43     {
 44         static_assert(sizeof(CellType) == sizeof(RuntimeObject), "RuntimeObject subclasses that add fields need to override subspaceFor<>()");
 45         static_assert(CellType::destroy == RuntimeObject::destroy);
 46         return subspaceForImpl(vm);
 47     }
 48 
 49     static RuntimeObject* create(VM& vm, Structure* structure, RefPtr<Instance>&& instance)
 50     {
 51         RuntimeObject* object = new (NotNull, allocateCell<RuntimeObject>(vm.heap)) RuntimeObject(vm, structure, WTFMove(instance));
 52         object->finishCreation(vm);
 53         return object;
 54     }
 55 
 56     static void destroy(JSCell*);
 57 
 58     static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
 59     static bool put(JSCell*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&);
 60     static bool deleteProperty(JSCell*, JSGlobalObject*, PropertyName);
 61     static JSValue defaultValue(const JSObject*, JSGlobalObject*, PreferredPrimitiveType);
 62     static CallType getCallData(JSCell*, CallData&);
 63     static ConstructType getConstructData(JSCell*, ConstructData&);
 64 
 65     static void getOwnPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, EnumerationMode);
 66 
 67     void invalidate();
 68 
 69     Instance* getInternalInstance() const { return m_instance.get(); }
 70 
 71     static Exception* throwInvalidAccessError(JSGlobalObject*, ThrowScope&);
 72 
 73     DECLARE_INFO;
 74 
 75     static ObjectPrototype* createPrototype(VM&, JSGlobalObject& globalObject)
 76     {
 77         return globalObject.objectPrototype();
 78     }
 79 
 80     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
 81     {
 82         return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
 83     }
 84 
 85 protected:
 86     RuntimeObject(VM&, Structure*, RefPtr<Instance>&&);
 87     void finishCreation(VM&);
 88 
 89 private:
 90     static EncodedJSValue fallbackObjectGetter(JSGlobalObject*, EncodedJSValue, PropertyName);
 91     static EncodedJSValue fieldGetter(JSGlobalObject*, EncodedJSValue, PropertyName);
 92     static EncodedJSValue methodGetter(JSGlobalObject*, EncodedJSValue, PropertyName);
 93 
 94     static IsoSubspace* subspaceForImpl(VM&);
 95 
 96     RefPtr<Instance> m_instance;
 97 };
 98 
 99 }
100 }
101 
102 #endif
