27 #pragma once
28 
29 #if ENABLE(IOS_TOUCH_EVENTS)
30 #include <WebKitAdditions/TouchEventIOS.h>
31 #elif ENABLE(TOUCH_EVENTS)
32 
33 #include "MouseRelatedEvent.h"
34 #include "TouchList.h"
35 
36 namespace WebCore {
37 
38 class TouchEvent final : public MouseRelatedEvent {
39     WTF_MAKE_ISO_ALLOCATED(TouchEvent);
40 public:
41     virtual ~TouchEvent();
42 
43     static Ref<TouchEvent> create(TouchList* touches, TouchList* targetTouches, TouchList* changedTouches,
44         const AtomString& type, RefPtr<WindowProxy>&& view, const IntPoint& globalLocation, OptionSet<Modifier> modifiers)
45     {
46         return adoptRef(*new TouchEvent(touches, targetTouches, changedTouches, type, WTFMove(view), globalLocation, modifiers));
47     }
48     static Ref<TouchEvent> createForBindings()
49     {
50         return adoptRef(*new TouchEvent);
51     }
52 
53     struct Init : MouseRelatedEventInit {
54         RefPtr<TouchList> touches;
55         RefPtr<TouchList> targetTouches;
56         RefPtr<TouchList> changedTouches;
57     };
58 
59     static Ref<TouchEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
60     {
61         return adoptRef(*new TouchEvent(type, initializer, isTrusted));
62     }
63 
64     void initTouchEvent(TouchList* touches, TouchList* targetTouches, TouchList* changedTouches, const AtomString& type,
65         RefPtr<WindowProxy>&&, int screenX, int screenY, int clientX, int clientY, bool ctrlKey, bool altKey, bool shiftKey, bool metaKey);
66 
67     TouchList* touches() const { return m_touches.get(); }
68     TouchList* targetTouches() const { return m_targetTouches.get(); }
69     TouchList* changedTouches() const { return m_changedTouches.get(); }
70 
71     void setTouches(RefPtr<TouchList>&& touches) { m_touches = touches; }
72     void setTargetTouches(RefPtr<TouchList>&& targetTouches) { m_targetTouches = targetTouches; }
73     void setChangedTouches(RefPtr<TouchList>&& changedTouches) { m_changedTouches = changedTouches; }
74 
75     bool isTouchEvent() const override;
76 
77     EventInterface eventInterface() const override;
78 
79 private:
80     TouchEvent();
81     TouchEvent(TouchList* touches, TouchList* targetTouches, TouchList* changedTouches, const AtomString& type,
82         RefPtr<WindowProxy>&&, const IntPoint& globalLocation, OptionSet<Modifier>);
83     TouchEvent(const AtomString&, const Init&, IsTrusted);
84 
85     RefPtr<TouchList> m_touches;
86     RefPtr<TouchList> m_targetTouches;
87     RefPtr<TouchList> m_changedTouches;
88 };
89 
90 } // namespace WebCore
91 
92 SPECIALIZE_TYPE_TRAITS_EVENT(TouchEvent)
93 
94 #endif // ENABLE(TOUCH_EVENTS)
