26 #pragma once
27 
28 #include "MouseEvent.h"
29 #include "MouseEventInit.h"
30 
31 namespace WebCore {
32 
33 class DataTransfer;
34 
35 struct DragEventInit : public MouseEventInit {
36     RefPtr<DataTransfer> dataTransfer;
37 };
38 
39 class DragEvent : public MouseEvent {
40     WTF_MAKE_ISO_ALLOCATED(DragEvent);
41 public:
42     using Init = DragEventInit;
43 
44     static Ref<DragEvent> create(const AtomString& eventType, DragEventInit&&);
45     static Ref<DragEvent> createForBindings();
46     static Ref<DragEvent> create(const AtomString& type, CanBubble, IsCancelable, IsComposed, MonotonicTime timestamp, RefPtr<WindowProxy>&&, int detail,
47         const IntPoint& screenLocation, const IntPoint& windowLocation, const IntPoint& movementDelta, OptionSet<Modifier>, short button, unsigned short buttons,
48         EventTarget* relatedTarget, double force, unsigned short syntheticClickType, DataTransfer* = nullptr, IsSimulated = IsSimulated::No, IsTrusted = IsTrusted::Yes);
49 
50 
51     DataTransfer* dataTransfer() const { return m_dataTransfer.get(); }
52 
53 private:
54     DragEvent(const AtomString& eventType, DragEventInit&&);
55     DragEvent(const AtomString& type, CanBubble, IsCancelable, IsComposed, MonotonicTime timestamp, RefPtr<WindowProxy>&&, int detail,
56         const IntPoint& screenLocation, const IntPoint& windowLocation, const IntPoint& movementDelta, OptionSet<Modifier>, short button, unsigned short buttons,
57         EventTarget* relatedTarget, double force, unsigned short syntheticClickType, DataTransfer*, IsSimulated, IsTrusted);
58     DragEvent();
59 
60     EventInterface eventInterface() const final;
61 
62     RefPtr<DataTransfer> m_dataTransfer;
63 };
64 
65 } // namespace WebCore
