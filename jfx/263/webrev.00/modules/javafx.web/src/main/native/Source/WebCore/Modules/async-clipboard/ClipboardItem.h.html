26 #pragma once
27 
28 #include <wtf/KeyValuePair.h>
29 #include <wtf/Ref.h>
30 #include <wtf/RefCounted.h>
31 #include <wtf/Vector.h>
32 #include <wtf/WeakPtr.h>
33 #include <wtf/text/WTFString.h>
34 
35 namespace WebCore {
36 
37 class Blob;
38 class Clipboard;
39 class ClipboardItemDataSource;
40 class DeferredPromise;
41 class DOMPromise;
42 class Navigator;
43 class PasteboardCustomData;
44 struct PasteboardItemInfo;
45 
46 class ClipboardItem : public RefCounted<ClipboardItem> {
47 public:
48     ~ClipboardItem();
49 
50     enum class PresentationStyle : uint8_t { Unspecified, Inline, Attachment };
51 
52     struct Options {
53         PresentationStyle presentationStyle { PresentationStyle::Unspecified };
54     };
55 
56     static Ref<ClipboardItem> create(Vector<KeyValuePair<String, RefPtr<DOMPromise>>>&&, const Options&);
57     static Ref<ClipboardItem> create(Clipboard&, const PasteboardItemInfo&);
58     static Ref<Blob> blobFromString(const String& stringData, const String& type);
59 
60     Vector<String> types() const;
61     void getType(const String&, Ref<DeferredPromise>&&);
62 
63     void collectDataForWriting(Clipboard& destination, CompletionHandler<void(Optional<PasteboardCustomData>)>&&);
64 
65     PresentationStyle presentationStyle() const { return m_presentationStyle; };
66     Navigator* navigator();
67     Clipboard* clipboard();
68 
69 private:
70     ClipboardItem(Vector<KeyValuePair<String, RefPtr<DOMPromise>>>&&, const Options&);
71     ClipboardItem(Clipboard&, const PasteboardItemInfo&);
72 
73     WeakPtr<Clipboard> m_clipboard;
74     WeakPtr<Navigator> m_navigator;
75     std::unique_ptr<ClipboardItemDataSource> m_dataSource;
76     PresentationStyle m_presentationStyle { PresentationStyle::Unspecified };
77 };
78 
79 } // namespace WebCore
