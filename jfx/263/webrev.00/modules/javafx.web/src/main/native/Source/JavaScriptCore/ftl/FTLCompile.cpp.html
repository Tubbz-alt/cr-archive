 60 void compile(State& state, Safepoint::Result& safepointResult)
 61 {
 62     Graph& graph = state.graph;
 63     CodeBlock* codeBlock = graph.m_codeBlock;
 64     VM& vm = graph.m_vm;
 65 
 66     if (shouldDumpDisassembly())
 67         state.proc->code().setDisassembler(makeUnique<B3::Air::Disassembler>());
 68 
 69     {
 70         GraphSafepoint safepoint(state.graph, safepointResult);
 71 
 72         B3::prepareForGeneration(*state.proc);
 73     }
 74 
 75     if (safepointResult.didGetCancelled())
 76         return;
 77     RELEASE_ASSERT(!state.graph.m_vm.heap.worldIsStopped());
 78 
 79     if (state.allocationFailed)
 80         return;
 81 
 82     std::unique_ptr<RegisterAtOffsetList> registerOffsets =
 83         makeUnique<RegisterAtOffsetList>(state.proc->calleeSaveRegisterAtOffsetList());
 84     if (shouldDumpDisassembly())
 85         dataLog(tierName, "Unwind info for ", CodeBlockWithJITType(codeBlock, JITType::FTLJIT), ": ", *registerOffsets, "\n");
 86     codeBlock->setCalleeSaveRegisters(WTFMove(registerOffsets));
 87     ASSERT(!(state.proc->frameSize() % sizeof(EncodedJSValue)));
 88     state.jitCode->common.frameRegisterCount = state.proc->frameSize() / sizeof(EncodedJSValue);
 89 
 90     int localsOffset =
 91         state.capturedValue->offsetFromFP() / sizeof(EncodedJSValue) + graph.m_nextMachineLocal;
 92     if (shouldDumpDisassembly()) {
 93         dataLog(tierName,
 94             "localsOffset = ", localsOffset, " for stack slot: ",
 95             pointerDump(state.capturedValue), " at ", RawPointer(state.capturedValue), "\n");
 96     }
 97 
 98     for (unsigned i = graph.m_inlineVariableData.size(); i--;) {
 99         InlineCallFrame* inlineCallFrame = graph.m_inlineVariableData[i].inlineCallFrame;
100 
101         if (inlineCallFrame->argumentCountRegister.isValid())
102             inlineCallFrame->argumentCountRegister += localsOffset;
103 
104         for (unsigned argument = inlineCallFrame->argumentsWithFixup.size(); argument-- > 1;) {
105             inlineCallFrame->argumentsWithFixup[argument] =
106                 inlineCallFrame->argumentsWithFixup[argument].withLocalsOffset(localsOffset);
107         }
108 
109         if (inlineCallFrame->isClosureCall) {
110             inlineCallFrame->calleeRecovery =
111                 inlineCallFrame->calleeRecovery.withLocalsOffset(localsOffset);
112         }
113 
114     }
115 
116     // Note that the scope register could be invalid here if the original code had CallEval but it
117     // got killed. That's because it takes the CallEval to cause the scope register to be kept alive
118     // unless the debugger is also enabled.
119     if (graph.needsScopeRegister() && codeBlock->scopeRegister().isValid())
120         codeBlock->setScopeRegister(codeBlock->scopeRegister() + localsOffset);
121 
122     for (OSRExitDescriptor& descriptor : state.jitCode->osrExitDescriptors) {
123         for (unsigned i = descriptor.m_values.size(); i--;)
124             descriptor.m_values[i] = descriptor.m_values[i].withLocalsOffset(localsOffset);
125         for (ExitTimeObjectMaterialization* materialization : descriptor.m_materializations)
126             materialization->accountForLocalsOffset(localsOffset);
127     }
128 
129     // We will add exception handlers while generating.
130     codeBlock->clearExceptionHandlers();
131 
132     CCallHelpers jit(codeBlock);
133     B3::generate(*state.proc, jit);
134 
135     // Emit the exception handler.
136     *state.exceptionHandler = jit.label();
137     jit.copyCalleeSavesToEntryFrameCalleeSavesBuffer(vm.topEntryFrame);
138     jit.move(MacroAssembler::TrustedImmPtr(&vm), GPRInfo::argumentGPR0);
139     jit.prepareCallOperation(vm);
140     CCallHelpers::Call call = jit.call(OperationPtrTag);
141     jit.jumpToExceptionHandler(vm);
142     jit.addLinkTask(
143         [=] (LinkBuffer& linkBuffer) {
144             linkBuffer.link(call, FunctionPtr<OperationPtrTag>(operationLookupExceptionHandler));
145         });
146 
147     state.finalizer->b3CodeLinkBuffer = makeUnique<LinkBuffer>(jit, codeBlock, JITCompilationCanFail);
148 
149     if (state.finalizer->b3CodeLinkBuffer->didFailToAllocate()) {
150         state.allocationFailed = true;
151         return;
152     }
153 
154     B3::PCToOriginMap originMap = state.proc->releasePCToOriginMap();
155     if (vm.shouldBuilderPCToCodeOriginMapping())
156         codeBlock->setPCToCodeOriginMap(makeUnique<PCToCodeOriginMap>(PCToCodeOriginMapBuilder(vm, WTFMove(originMap)), *state.finalizer->b3CodeLinkBuffer));
157 
158     CodeLocationLabel<JSEntryPtrTag> label = state.finalizer->b3CodeLinkBuffer->locationOf<JSEntryPtrTag>(state.proc->code().entrypointLabel(0));
159     state.generatedFunction = label;
160     state.jitCode->initializeB3Byproducts(state.proc->releaseByproducts());
161 
162     for (auto pair : state.graph.m_entrypointIndexToCatchBytecodeIndex) {
163         BytecodeIndex catchBytecodeIndex = pair.value;
164         unsigned entrypointIndex = pair.key;
165         Vector<FlushFormat> argumentFormats = state.graph.m_argumentFormats[entrypointIndex];
166         state.jitCode->common.appendCatchEntrypoint(catchBytecodeIndex, state.finalizer->b3CodeLinkBuffer->locationOf<ExceptionHandlerPtrTag>(state.proc->code().entrypointLabel(entrypointIndex)), WTFMove(argumentFormats));
167     }
168     state.jitCode->common.finalizeCatchEntrypoints();
169 
170     if (B3::Air::Disassembler* disassembler = state.proc->code().disassembler()) {
171         PrintStream& out = WTF::dataFile();
172 
173         out.print("Generated ", state.graph.m_plan.mode(), " code for ", CodeBlockWithJITType(state.graph.m_codeBlock, JITType::FTLJIT), ", instructions size = ", state.graph.m_codeBlock->instructionsSize(), ":\n");
174 
175         LinkBuffer& linkBuffer = *state.finalizer->b3CodeLinkBuffer;
176         B3::Value* currentB3Value = nullptr;
177         Node* currentDFGNode = nullptr;
178 
179         HashSet<B3::Value*> printedValues;
180         HashSet<Node*> printedNodes;
181         const char* dfgPrefix = "DFG " "    ";
182         const char* b3Prefix  = "b3  " "          ";
183         const char* airPrefix = "Air " "              ";
184         const char* asmPrefix = "asm " "                ";
185 
186         auto printDFGNode = [&] (Node* node) {
187             if (currentDFGNode == node)
188                 return;
189 
190             currentDFGNode = node;
191             if (!currentDFGNode)
192                 return;
193 
194             HashSet<Node*> localPrintedNodes;
195             WTF::Function<void(Node*)> printNodeRecursive = [&] (Node* node) {
196                 if (printedNodes.contains(node) || localPrintedNodes.contains(node))
197                     return;
198 
199                 localPrintedNodes.add(node);
200                 graph.doToChildren(node, [&] (Edge child) {
201                     printNodeRecursive(child.node());
202                 });
203                 graph.dump(out, dfgPrefix, node);
204             };
205             printNodeRecursive(node);
206             printedNodes.add(node);
207         };
208 
209         auto printB3Value = [&] (B3::Value* value) {
210             if (currentB3Value == value)
211                 return;
212 
213             currentB3Value = value;
214             if (!currentB3Value)
215                 return;
216 
217             printDFGNode(bitwise_cast<Node*>(value->origin().data()));
218 
219             HashSet<B3::Value*> localPrintedValues;
220             auto printValueRecursive = recursableLambda([&] (auto self, B3::Value* value) -> void {
221                 if (printedValues.contains(value) || localPrintedValues.contains(value))
222                     return;
223 
224                 localPrintedValues.add(value);
225                 for (unsigned i = 0; i < value->numChildren(); i++)
226                     self(value->child(i));
227                 out.print(b3Prefix);
228                 value->deepDump(state.proc.get(), out);
229                 out.print("\n");
230             });
231 
232             printValueRecursive(currentB3Value);
233             printedValues.add(value);
234         };
235 
236         auto forEachInst = scopedLambda<void(B3::Air::Inst&)>([&] (B3::Air::Inst& inst) {
237             printB3Value(inst.origin);
238         });
239 
240         disassembler->dump(state.proc->code(), out, linkBuffer, airPrefix, asmPrefix, forEachInst);
241         linkBuffer.didAlreadyDisassemble();
242     }
243 }
