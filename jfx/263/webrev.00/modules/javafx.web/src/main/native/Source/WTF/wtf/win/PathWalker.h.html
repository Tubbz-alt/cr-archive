28 #include <Windows.h>

32 namespace WTF {
33 
34 class PathWalker {
35     WTF_MAKE_FAST_ALLOCATED;
36     WTF_MAKE_NONCOPYABLE(PathWalker);
37 public:
38     PathWalker(const WTF::String& directory, const WTF::String& pattern);
39     ~PathWalker();
40 
41     bool isValid() const { return m_handle != INVALID_HANDLE_VALUE; }
42     const WIN32_FIND_DATAW& data() const { return m_data; }
43 
44     bool step();
45 
46 private:
47     HANDLE m_handle;
48     WIN32_FIND_DATAW m_data;
49 };
50 
51 } // namespace WTF
