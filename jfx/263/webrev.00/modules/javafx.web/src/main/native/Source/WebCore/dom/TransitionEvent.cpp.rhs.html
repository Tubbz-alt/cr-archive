36 TransitionEvent::TransitionEvent(const AtomString& type, const String& propertyName, double elapsedTime, const String& pseudoElement, Optional<Seconds> timelineTime, WebAnimation* animation)
37     : AnimationEventBase(type, animation, timelineTime)

39     , m_elapsedTime(elapsedTime)

44 TransitionEvent::TransitionEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)

49 {
50 }

52 TransitionEvent::~TransitionEvent() = default;
53 
54 const String& TransitionEvent::propertyName() const

58 
59 double TransitionEvent::elapsedTime() const
60 {
61     return m_elapsedTime;
62 }
63 
64 const String& TransitionEvent::pseudoElement() const
65 {
66     return m_pseudoElement;
67 }
68 
69 EventInterface TransitionEvent::eventInterface() const
70 {
71     return TransitionEventInterfaceType;
