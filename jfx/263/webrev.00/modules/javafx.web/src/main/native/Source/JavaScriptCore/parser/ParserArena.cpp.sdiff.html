 46 inline void ParserArena::deallocateObjects()
 47 {
 48     size_t size = m_deletableObjects.size();
 49     for (size_t i = 0; i < size; ++i)
 50         m_deletableObjects[i]->~ParserArenaDeletable();
 51 
 52     if (m_freeablePoolEnd)
 53         fastFree(freeablePool());
 54 
 55     size = m_freeablePools.size();
 56     for (size_t i = 0; i < size; ++i)
 57         fastFree(m_freeablePools[i]);
 58 }

 62     deallocateObjects();
 63 }

 65 void ParserArena::allocateFreeablePool()
 66 {
 67     if (m_freeablePoolEnd)
 68         m_freeablePools.append(freeablePool());
 69 
 70     char* pool = static_cast<char*>(fastMalloc(freeablePoolSize));
 71     m_freeableMemory = pool;
 72     m_freeablePoolEnd = pool + freeablePoolSize;
 73     ASSERT(freeablePool() == pool);
 74 }

 81 const Identifier& IdentifierArena::makeBigIntDecimalIdentifier(VM& vm, const Identifier& identifier, uint8_t radix)
 82 {
 83     if (radix == 10)
 84         return identifier;
 85 
 86     // FIXME: We are allocating a JSBigInt just to be able to use
 87     // JSBigInt::tryGetString when radix is not 10.
 88     // This creates some GC pressure, but since these identifiers
 89     // will only be created when BigInt literal is used as a property name,
 90     // it wont be much problematic, given such cases are very rare.
 91     // There is a lot of optimizations we can apply here when necessary.
 92     // https://bugs.webkit.org/show_bug.cgi?id=207627
 93 
 94     auto scope = DECLARE_CATCH_SCOPE(vm);
 95     JSBigInt* bigInt = JSBigInt::parseInt(nullptr, vm, identifier.string(), radix, JSBigInt::ErrorParseMode::ThrowExceptions, JSBigInt::ParseIntSign::Unsigned);
 96     scope.assertNoException();
 97     m_identifiers.append(Identifier::fromString(vm, JSBigInt::tryGetString(vm, bigInt, 10)));
 98     return m_identifiers.last();
 99 }
