33 inline double secondsToWebAnimationsAPITime(const Seconds time)
34 {
35     // Precision of time values
36     // https://drafts.csswg.org/web-animations-1/#precision-of-time-values
37 
38     // The internal representation of time values is implementation dependent however, it is RECOMMENDED that user
39     // agents be able to represent input time values with microsecond precision so that a time value (which nominally
40     // represents milliseconds) of 0.001 is distinguishable from 0.0.
41     auto roundedTime = std::round(time.microseconds()) / 1000;
42     if (roundedTime == -0)
43         return 0;
44     return roundedTime;
45 }
46 
47 const auto timeEpsilon = Seconds::fromMilliseconds(0.001);
