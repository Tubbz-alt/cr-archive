131 template<class HashType>
132 bool addToAccessControlAllowList(const String& string, unsigned start, unsigned end, HashSet<String, HashType>& set)
133 {
134     StringImpl* stringImpl = string.impl();
135     if (!stringImpl)
136         return true;
137 
138     // Skip white space from start.
139     while (start <= end && isHTTPSpace((*stringImpl)[start]))
140         ++start;
141 
142     // only white space
143     if (start > end)
144         return true;
145 
146     // Skip white space from end.
147     while (end && isHTTPSpace((*stringImpl)[end]))
148         --end;
149 
150     auto token = string.substring(start, end - start + 1);
151     if (!isValidHTTPToken(token))
152         return false;
153 
154     set.add(WTFMove(token));
155     return true;
156 }
