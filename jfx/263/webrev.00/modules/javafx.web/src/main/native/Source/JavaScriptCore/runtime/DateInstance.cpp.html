53 const GregorianDateTime* DateInstance::calculateGregorianDateTime(VM& vm) const
54 {
55     double milli = internalNumber();
56     if (std::isnan(milli))
57         return nullptr;
58 
59     if (!m_data)
60         m_data = vm.dateInstanceCache.add(milli);
61 
62     if (m_data->m_gregorianDateTimeCachedForMS != milli) {
63         msToGregorianDateTime(vm, milli, WTF::LocalTime, m_data->m_cachedGregorianDateTime);
64         m_data->m_gregorianDateTimeCachedForMS = milli;
65     }
66     return &m_data->m_cachedGregorianDateTime;
67 }
68 
69 const GregorianDateTime* DateInstance::calculateGregorianDateTimeUTC(VM& vm) const
70 {
71     double milli = internalNumber();
72     if (std::isnan(milli))
73         return nullptr;
74 
75     if (!m_data)
76         m_data = vm.dateInstanceCache.add(milli);
77 
78     if (m_data->m_gregorianDateTimeUTCCachedForMS != milli) {
79         msToGregorianDateTime(vm, milli, WTF::UTCTime, m_data->m_cachedGregorianDateTimeUTC);
80         m_data->m_gregorianDateTimeUTCCachedForMS = milli;
81     }
82     return &m_data->m_cachedGregorianDateTimeUTC;
83 }
