 37 const RegisterSet& StackmapGenerationParams::usedRegisters() const

 44 RegisterSet StackmapGenerationParams::unavailableRegisters() const
 45 {
 46     RegisterSet result = usedRegisters();
 47 
 48     RegisterSet unsavedCalleeSaves = RegisterSet::vmCalleeSaveRegisters();
 49     unsavedCalleeSaves.exclude(m_context.code->calleeSaveRegisters());
 50 
 51     result.merge(unsavedCalleeSaves);
 52 
 53     for (GPRReg gpr : m_gpScratch)
 54         result.clear(gpr);
 55     for (FPRReg fpr : m_fpScratch)
 56         result.clear(fpr);
 57 
 58     return result;
 59 }

 61 Vector<Box<CCallHelpers::Label>> StackmapGenerationParams::successorLabels() const
 62 {
 63     RELEASE_ASSERT(m_context.indexInBlock == m_context.currentBlock->size() - 1);
 64     RELEASE_ASSERT(m_value->effects().terminal);
 65 
 66     Vector<Box<CCallHelpers::Label>> result(m_context.currentBlock->numSuccessors());
 67     for (unsigned i = m_context.currentBlock->numSuccessors(); i--;)
 68         result[i] = m_context.blockLabels[m_context.currentBlock->successorBlock(i)];
 69     return result;
 70 }
 71 
 72 bool StackmapGenerationParams::fallsThroughToSuccessor(unsigned successorIndex) const
 73 {
 74     RELEASE_ASSERT(m_context.indexInBlock == m_context.currentBlock->size() - 1);
 75     RELEASE_ASSERT(m_value->effects().terminal);
 76 
 77     Air::BasicBlock* successor = m_context.currentBlock->successorBlock(successorIndex);
 78     Air::BasicBlock* nextBlock = m_context.code->findNextBlock(m_context.currentBlock);
 79     return successor == nextBlock;
 80 }
