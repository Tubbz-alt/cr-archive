 52 namespace Layout {
 53 
 54 struct Position {
 55     operator LayoutUnit() const { return value; }
 56     LayoutUnit value;
 57 };
 58 
 59 inline bool operator<(const Position& a, const Position& b)
 60 {
 61     return a.value < b.value;
 62 }
 63 
 64 inline bool operator==(const Position& a, const Position& b)
 65 {
 66     return a.value == b.value;
 67 }
 68 
 69 struct Point {
 70     // FIXME: Use Position<Horizontal>, Position<Vertical> to avoid top/left vs. x/y confusion.
 71     LayoutUnit x; // left
 72     LayoutUnit y; // top
 73 
 74     Point() = default;
 75     Point(LayoutUnit, LayoutUnit);
 76     Point(LayoutPoint);
 77     static Point max() { return { LayoutUnit::max(), LayoutUnit::max() }; }
 78 
 79     void move(LayoutSize);
 80     void moveBy(LayoutPoint);
 81     operator LayoutPoint() const { return { x, y }; }
 82 };
 83 
 84 // FIXME: Wrap these into structs.
 85 using PointInContextRoot = Point;
 86 using PositionInContextRoot = Position;
 87 
 88 inline Point::Point(LayoutPoint point)
 89     : x(point.x())
 90     , y(point.y())
 91 {
 92 }
 93 
 94 inline Point::Point(LayoutUnit x, LayoutUnit y)
 95     : x(x)
 96     , y(y)
 97 {
 98 }
 99 
100 inline void Point::move(LayoutSize offset)
101 {
102     x += offset.width();
103     y += offset.height();
104 }
105 
106 inline void Point::moveBy(LayoutPoint offset)
107 {
108     x += offset.x();
109     y += offset.y();
110 }
