 31 namespace WebCore {
 32 
 33 template<typename T> struct Converter<IDLCallbackFunction<T>> : DefaultConverter<IDLCallbackFunction<T>> {
 34 
 35     static constexpr bool conversionHasSideEffects = false;
 36 
 37     template<typename ExceptionThrower = DefaultExceptionThrower>
 38     static RefPtr<T> convert(JSC::ExecState& state, JSC::JSValue value, JSDOMGlobalObject& globalObject, ExceptionThrower&& exceptionThrower = ExceptionThrower())
 39     {
 40         JSC::VM& vm = state.vm();
 41         auto scope = DECLARE_THROW_SCOPE(vm);
 42 
 43         if (!value.isFunction(vm)) {
 44             exceptionThrower(state, scope);
 45             return nullptr;
 46         }
 47 
 48         return T::create(JSC::asObject(value), &globalObject);
 49     }
 50 };
 51 
 52 template<typename T> struct JSConverter<IDLCallbackFunction<T>> {
 53     static constexpr bool needsState = false;
 54     static constexpr bool needsGlobalObject = false;
 55 
 56     template <typename U>
 57     static JSC::JSValue convert(const U& value)
 58     {
 59         return toJS(Detail::getPtrOrRef(value));
 60     }
 61 
 62     template<typename U>
 63     static JSC::JSValue convertNewlyCreated(U&& value)
 64     {
 65         return toJSNewlyCreated(std::forward<U>(value));
 66     }
 67 };
 68 
 69 
 70 template<typename T> struct Converter<IDLCallbackInterface<T>> : DefaultConverter<IDLCallbackInterface<T>> {
 71     template<typename ExceptionThrower = DefaultExceptionThrower>
 72     static RefPtr<T> convert(JSC::ExecState& state, JSC::JSValue value, JSDOMGlobalObject& globalObject, ExceptionThrower&& exceptionThrower = ExceptionThrower())
 73     {
 74         JSC::VM& vm = state.vm();
 75         auto scope = DECLARE_THROW_SCOPE(vm);
 76 
 77         if (!value.isObject()) {
 78             exceptionThrower(state, scope);
 79             return nullptr;
 80         }
 81 
 82         return T::create(JSC::asObject(value), &globalObject);
 83     }
 84 };

 88     static constexpr bool needsGlobalObject = false;

 94     }
 95

 97     static JSC::JSValue convertNewlyCreated(U&& value)
