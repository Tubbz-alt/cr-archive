 79 # A writer that only updates file if it actually changed.
 80 class IncrementalFileWriter:
 81     def __init__(self, filepath, force_output):
 82         self._filepath = filepath
 83         self._output = ""
 84         self.force_output = force_output
 85 
 86     def write(self, text):
 87         self._output += text
 88 
 89     def close(self):
 90         text_changed = True
 91         self._output = self._output.rstrip() + "\n"
 92 
 93         try:
 94             if self.force_output:
 95                 raise
 96 
 97             read_file = open(self._filepath, "r")
 98             old_text = read_file.read()
 99             read_file.close()
100             text_changed = old_text != self._output
101         except:
102             # Ignore, just overwrite by default
103             pass
104 
105         if text_changed or self.force_output:
106             dirname = os.path.dirname(self._filepath)
107             if not os.path.isdir(dirname):
108                 os.makedirs(dirname)
109             out_file = open(self._filepath, "w")
110             out_file.write(self._output)
111             out_file.close()

114 def generate_from_specification(primary_specification_filepath=None,
115                                 supplemental_specification_filepaths=[],
116                                 concatenate_output=False,
117                                 output_dirpath=None,
118                                 force_output=False,
119                                 framework_name="",
120                                 platform_name="",
121                                 generate_frontend=True,
122                                 generate_backend=True):
123 
124     def load_specification(protocol, filepath, isSupplemental=False):
125         try:
126             with open(filepath, "r") as input_file:
127                 regex = re.compile(r"\/\*.*?\*\/", re.DOTALL)
128                 parsed_json = json.loads(re.sub(regex, "", input_file.read()))
129                 protocol.parse_specification(parsed_json, isSupplemental)
130         except ValueError as e:
131             raise Exception("Error parsing valid JSON in file: " + filepath + "\nParse error: " + str(e))
132 
133     platform = Platform.fromString(platform_name)
134     protocol = models.Protocol(framework_name)
135     for specification in supplemental_specification_filepaths:
136         load_specification(protocol, specification, isSupplemental=True)
137     load_specification(protocol, primary_specification_filepath, isSupplemental=False)
138 
139     protocol.resolve_types()
140 
141     generator_arguments = [protocol, platform, primary_specification_filepath]
142     generators = []
143 
144     if protocol.framework is Frameworks.Test:
145         generators.append(JSBackendCommandsGenerator(*generator_arguments))
146         generators.append(CppAlternateBackendDispatcherHeaderGenerator(*generator_arguments))
147         generators.append(CppBackendDispatcherHeaderGenerator(*generator_arguments))
148         generators.append(CppBackendDispatcherImplementationGenerator(*generator_arguments))
149         generators.append(CppFrontendDispatcherHeaderGenerator(*generator_arguments))
150         generators.append(CppFrontendDispatcherImplementationGenerator(*generator_arguments))
151         generators.append(CppProtocolTypesHeaderGenerator(*generator_arguments))
152         generators.append(CppProtocolTypesImplementationGenerator(*generator_arguments))
153         generators.append(ObjCBackendDispatcherHeaderGenerator(*generator_arguments))
154         generators.append(ObjCBackendDispatcherImplementationGenerator(*generator_arguments))
155         generators.append(ObjCConfigurationHeaderGenerator(*generator_arguments))
156         generators.append(ObjCConfigurationImplementationGenerator(*generator_arguments))
157         generators.append(ObjCFrontendDispatcherImplementationGenerator(*generator_arguments))
158         generators.append(ObjCHeaderGenerator(*generator_arguments))
159         generators.append(ObjCInternalHeaderGenerator(*generator_arguments))
160         generators.append(ObjCProtocolTypeConversionsHeaderGenerator(*generator_arguments))
161         generators.append(ObjCProtocolTypeConversionsImplementationGenerator(*generator_arguments))
162         generators.append(ObjCProtocolTypesImplementationGenerator(*generator_arguments))
163 
164     elif protocol.framework is Frameworks.JavaScriptCore:
165         generators.append(JSBackendCommandsGenerator(*generator_arguments))
166         generators.append(CppAlternateBackendDispatcherHeaderGenerator(*generator_arguments))
167         generators.append(CppBackendDispatcherHeaderGenerator(*generator_arguments))
168         generators.append(CppBackendDispatcherImplementationGenerator(*generator_arguments))
169         generators.append(CppFrontendDispatcherHeaderGenerator(*generator_arguments))
170         generators.append(CppFrontendDispatcherImplementationGenerator(*generator_arguments))
171         generators.append(CppProtocolTypesHeaderGenerator(*generator_arguments))
172         generators.append(CppProtocolTypesImplementationGenerator(*generator_arguments))
173 
174     elif protocol.framework is Frameworks.WebKit and generate_backend:
175         generators.append(CppBackendDispatcherHeaderGenerator(*generator_arguments))
176         generators.append(CppBackendDispatcherImplementationGenerator(*generator_arguments))
177         generators.append(CppFrontendDispatcherHeaderGenerator(*generator_arguments))
178         generators.append(CppFrontendDispatcherImplementationGenerator(*generator_arguments))
179         generators.append(CppProtocolTypesHeaderGenerator(*generator_arguments))
180         generators.append(CppProtocolTypesImplementationGenerator(*generator_arguments))
181 
182     elif protocol.framework is Frameworks.WebKit and generate_frontend:
183         generators.append(ObjCHeaderGenerator(*generator_arguments))
184         generators.append(ObjCProtocolTypeConversionsHeaderGenerator(*generator_arguments))
185         generators.append(ObjCProtocolTypeConversionsImplementationGenerator(*generator_arguments))
186         generators.append(ObjCProtocolTypesImplementationGenerator(*generator_arguments))
187 
188     elif protocol.framework is Frameworks.WebInspector:
189         generators.append(ObjCBackendDispatcherHeaderGenerator(*generator_arguments))
190         generators.append(ObjCBackendDispatcherImplementationGenerator(*generator_arguments))
191         generators.append(ObjCConfigurationHeaderGenerator(*generator_arguments))
192         generators.append(ObjCConfigurationImplementationGenerator(*generator_arguments))
193         generators.append(ObjCFrontendDispatcherImplementationGenerator(*generator_arguments))
194         generators.append(ObjCHeaderGenerator(*generator_arguments))
195         generators.append(ObjCInternalHeaderGenerator(*generator_arguments))
196         generators.append(ObjCProtocolTypeConversionsHeaderGenerator(*generator_arguments))
197         generators.append(ObjCProtocolTypesImplementationGenerator(*generator_arguments))
198 
199     elif protocol.framework is Frameworks.WebInspectorUI:
200         generators.append(JSBackendCommandsGenerator(*generator_arguments))
201 
202     single_output_file_contents = []
203 
204     for generator in generators:
205         # Only some generators care whether frontend or backend was specified, but it is
206         # set on all of them to avoid adding more constructor arguments or thinking too hard.
207         if generate_backend:
208             generator.set_generator_setting('generate_backend', True)
209         if generate_frontend:
210             generator.set_generator_setting('generate_frontend', True)
211 
212         output = generator.generate_output()
213         if concatenate_output:
214             single_output_file_contents.append('### Begin File: %s' % generator.output_filename())
215             single_output_file_contents.append(output)
216             single_output_file_contents.append('### End File: %s' % generator.output_filename())
217             single_output_file_contents.append('')
218         else:
219             output_file = IncrementalFileWriter(os.path.join(output_dirpath, generator.output_filename()), force_output)
220             output_file.write(output)
221             output_file.close()
222 
223     if concatenate_output:
224         filename = os.path.join(os.path.basename(primary_specification_filepath) + '-result')
225         output_file = IncrementalFileWriter(os.path.join(output_dirpath, filename), force_output)
226         output_file.write('\n'.join(single_output_file_contents))
227         output_file.close()

230 if __name__ == '__main__':
231     allowed_framework_names = ['JavaScriptCore', 'WebInspector', 'WebInspectorUI', 'WebKit', 'Test']
232     allowed_platform_names = ['iOS', 'macOS', 'all', 'generic']
233     cli_parser = optparse.OptionParser(usage="usage: %prog [options] PrimaryProtocol.json [SupplementalProtocol.json ...]")
234     cli_parser.add_option("-o", "--outputDir", help="Directory where generated files should be written.")
235     cli_parser.add_option("--framework", type="choice", choices=allowed_framework_names, help="The framework that the primary specification belongs to.")
236     cli_parser.add_option("--force", action="store_true", help="Force output of generated scripts, even if nothing changed.")
237     cli_parser.add_option("-v", "--debug", action="store_true", help="Log extra output for debugging the generator itself.")
238     cli_parser.add_option("-t", "--test", action="store_true", help="Enable test mode. Use unique output filenames created by prepending the input filename.")
239     cli_parser.add_option("--frontend", action="store_true", help="Generate code for the frontend-side of the protocol only.")
240     cli_parser.add_option("--backend", action="store_true", help="Generate code for the backend-side of the protocol only.")
241     cli_parser.add_option("--platform", default="generic", help="The platform of the backend being generated. For example, we compile WebKit2 for either macOS or iOS. This value is case-insensitive. Allowed values: %s" % ", ".join(allowed_platform_names))
242     options = None
243 
244     arg_options, arg_values = cli_parser.parse_args()
245     if (len(arg_values) < 1):
246         raise ParseException("At least one plain argument expected")
247 
248     if not arg_options.outputDir:
249         raise ParseException("Missing output directory")
250 
251     if arg_options.debug:
252         log.setLevel(logging.DEBUG)
253 
254     generate_backend = arg_options.backend;
255     generate_frontend = arg_options.frontend;
256     # Default to generating both the frontend and backend if neither is specified.
257     if not generate_backend and not generate_frontend:
258         generate_backend = True
259         generate_frontend = True
260 
261     options = {
262         'primary_specification_filepath': arg_values[0],
263         'supplemental_specification_filepaths': arg_values[1:],
264         'output_dirpath': arg_options.outputDir,
265         'concatenate_output': arg_options.test,
266         'framework_name': arg_options.framework,
267         'platform_name': arg_options.platform,
268         'force_output': arg_options.force,
269         'generate_backend': generate_backend,
270         'generate_frontend': generate_frontend,
271     }
272 
273     try:
274         generate_from_specification(**options)
275     except (ParseException, TypecheckException) as e:
276         if arg_options.test:
277             log.error(e.message)
278         else:
279             raise  # Force the build to fail.
