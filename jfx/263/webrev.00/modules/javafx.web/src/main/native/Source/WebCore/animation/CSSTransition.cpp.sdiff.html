62 void CSSTransition::setTimingProperties(Seconds delay, Seconds duration)
63 {
64     suspendEffectInvalidation();
65 
66     // This method is only called from CSSTransition::create() where we're guaranteed to have an effect.
67     auto* animationEffect = effect();
68     ASSERT(animationEffect);
69 
70     // In order for CSS Transitions to be seeked backwards, they need to have their fill mode set to backwards
71     // such that the original CSS value applied prior to the transition is used for a negative current time.
72     animationEffect->setFill(FillMode::Backwards);
73     animationEffect->setDelay(delay);
74     animationEffect->setIterationDuration(duration);



75 
76     unsuspendEffectInvalidation();
77 }

69 void CSSTransition::setTimingProperties(Seconds delay, Seconds duration)
70 {
71     suspendEffectInvalidation();
72 
73     // This method is only called from CSSTransition::create() where we're guaranteed to have an effect.
74     auto* animationEffect = effect();
75     ASSERT(animationEffect);
76 
77     // In order for CSS Transitions to be seeked backwards, they need to have their fill mode set to backwards
78     // such that the original CSS value applied prior to the transition is used for a negative current time.
79     animationEffect->setFill(FillMode::Backwards);
80     animationEffect->setDelay(delay);
81     animationEffect->setIterationDuration(duration);
82     animationEffect->setTimingFunction(backingAnimation().timingFunction());
83     animationEffect->updateStaticTimingProperties();
84     effectTimingDidChange();
85 
86     unsuspendEffectInvalidation();
87 }
