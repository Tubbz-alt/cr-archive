49     return adoptRef(*new ToFTLForOSREntryDeferredCompilationCallback(forcedOSREntryTrigger));

52 void ToFTLForOSREntryDeferredCompilationCallback::compilationDidBecomeReadyAsynchronously(
53     CodeBlock* codeBlock, CodeBlock* profiledDFGCodeBlock)
54 {
55     dataLogLnIf(Options::verboseOSR(),
56         "Optimizing compilation of ", *codeBlock, " (for ", *profiledDFGCodeBlock,
57         ") did become ready.");


58 
59     *m_forcedOSREntryTrigger = JITCode::TriggerReason::CompilationDone;
60 }
61 
62 void ToFTLForOSREntryDeferredCompilationCallback::compilationDidComplete(
63     CodeBlock* codeBlock, CodeBlock* profiledDFGCodeBlock, CompilationResult result)
64 {
65     dataLogLnIf(Options::verboseOSR(),
66         "Optimizing compilation of ", *codeBlock, " (for ", *profiledDFGCodeBlock,
67         ") result: ", result);


68 
69     JITCode* jitCode = profiledDFGCodeBlock->jitCode()->dfg();
70 
71     switch (result) {
72     case CompilationSuccessful: {
73         jitCode->setOSREntryBlock(codeBlock->vm(), profiledDFGCodeBlock, codeBlock);
74         BytecodeIndex osrEntryBytecode = codeBlock->jitCode()->ftlForOSREntry()->bytecodeIndex();
75         jitCode->tierUpEntryTriggers.set(osrEntryBytecode, JITCode::TriggerReason::CompilationDone);
76         break;
77     }
78     case CompilationFailed:
79         jitCode->osrEntryRetry = 0;
80         jitCode->abandonOSREntry = true;
81         profiledDFGCodeBlock->jitCode()->dfg()->setOptimizationThresholdBasedOnCompilationResult(
82             profiledDFGCodeBlock, result);
83         break;
84     case CompilationDeferred:
85         RELEASE_ASSERT_NOT_REACHED();
86     case CompilationInvalidated:
87         jitCode->osrEntryRetry = 0;
88         break;
89     }
90 
91     DeferredCompilationCallback::compilationDidComplete(codeBlock, profiledDFGCodeBlock, result);
92 }
