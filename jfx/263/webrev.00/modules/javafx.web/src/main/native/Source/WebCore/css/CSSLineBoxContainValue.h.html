26 #pragma once
27 
28 #include "CSSValue.h"
29 #include <wtf/OptionSet.h>
30 #include <wtf/Ref.h>
31 
32 namespace WebCore {
33 
34 class CSSPrimitiveValue;
35 
36 enum class LineBoxContain {
37     Block           = 1 << 0,
38     Inline          = 1 << 1,
39     Font            = 1 << 2,
40     Glyphs          = 1 << 3,
41     Replaced        = 1 << 4,
42     InlineBox       = 1 << 5,
43     InitialLetter   = 1 << 6,
44 };
45 
46 // Used for text-CSSLineBoxContain and box-CSSLineBoxContain
47 class CSSLineBoxContainValue final : public CSSValue {
48 public:
49     static Ref<CSSLineBoxContainValue> create(OptionSet<LineBoxContain> value)
50     {
51         return adoptRef(*new CSSLineBoxContainValue(value));
52     }
53 
54     String customCSSText() const;
55     bool equals(const CSSLineBoxContainValue& other) const { return m_value == other.m_value; }
56     OptionSet<LineBoxContain> value() const { return m_value; }
57 
58 private:
59     explicit CSSLineBoxContainValue(OptionSet<LineBoxContain>);
60 
61     OptionSet<LineBoxContain> m_value;
62 };
63 
64 } // namespace WebCore
65 
66 SPECIALIZE_TYPE_TRAITS_CSS_VALUE(CSSLineBoxContainValue, isLineBoxContainValue())
