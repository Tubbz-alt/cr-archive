 35 IdleCallbackController::IdleCallbackController(Document& document)
 36     : m_document(makeWeakPtr(document))
 37 {
 38 
 39 }
 40 int IdleCallbackController::queueIdleCallback(Ref<IdleRequestCallback>&& callback, Seconds)
 41 {
 42     bool startIdlePeriod = m_idleRequestCallbacks.isEmpty() && m_runnableIdleCallbacks.isEmpty();
 43 
 44     ++m_idleCallbackIdentifier;
 45     auto handle = m_idleCallbackIdentifier;
 46 
 47     m_idleRequestCallbacks.append({ handle, WTFMove(callback) });
 48 
 49     if (startIdlePeriod)
 50         queueTaskToStartIdlePeriod();
 51 
 52     // FIXME: Queue a task if timeout is positive.
 53 
 54     return handle;
 55 }
 56 
 57 void IdleCallbackController::removeIdleCallback(int signedIdentifier)
 58 {
 59     if (signedIdentifier <= 0)
 60         return;
 61     unsigned identifier = signedIdentifier;
 62 
 63     m_idleRequestCallbacks.removeAllMatching([identifier](auto& request) {
 64         return request.identifier == identifier;
 65     });
 66 
 67     m_runnableIdleCallbacks.removeAllMatching([identifier](auto& request) {
 68         return request.identifier == identifier;
 69     });
 70 }
 71 
 72 void IdleCallbackController::queueTaskToStartIdlePeriod()
 73 {
 74     m_document->eventLoop().queueTask(TaskSource::IdleTask, [protectedDocument = makeRef(*m_document), this] {
 75         RELEASE_ASSERT(protectedDocument->idleCallbackController() == this);
 76         startIdlePeriod();
 77     });
 78 }
 79 
 80 // https://w3c.github.io/requestidlecallback/#start-an-idle-period-algorithm
 81 static const auto deadlineCapToEnsureResponsiveness = 50_ms;
 82 void IdleCallbackController::startIdlePeriod()
 83 {
 84     auto now = MonotonicTime::now();
 85     if (m_lastDeadline > now)
 86         return;
 87 
 88     // FIXME: Take other tasks in the WindowEventLoop into account.
 89     auto deadline = now + deadlineCapToEnsureResponsiveness;
 90 
 91     for (auto& request : m_idleRequestCallbacks)
 92         m_runnableIdleCallbacks.append({ request.identifier, WTFMove(request.callback) });
 93     m_idleRequestCallbacks.clear();
 94 
 95     ASSERT(!m_runnableIdleCallbacks.isEmpty());
 96     queueTaskToInvokeIdleCallbacks(deadline);
 97 
 98     m_lastDeadline = deadline;
 99 }
100 
101 void IdleCallbackController::queueTaskToInvokeIdleCallbacks(MonotonicTime deadline)
102 {
103     m_document->eventLoop().queueTask(TaskSource::IdleTask, [protectedDocument = makeRef(*m_document), deadline, this] {
104         RELEASE_ASSERT(protectedDocument->idleCallbackController() == this);
105         invokeIdleCallbacks(deadline);
106     });
107 }
108 
109 // https://w3c.github.io/requestidlecallback/#invoke-idle-callbacks-algorithm
110 void IdleCallbackController::invokeIdleCallbacks(MonotonicTime deadline)
111 {
112     if (!m_document)
113         return;
114 
115     auto now = MonotonicTime::now();
116     if (now < deadline) {
117         // FIXME: Don't do this if there is a higher priority task in the event loop.
118         // https://github.com/w3c/requestidlecallback/issues/83
119         if (m_runnableIdleCallbacks.isEmpty())
120             return;
121 
122         auto request = m_runnableIdleCallbacks.takeFirst();
123         auto idleDeadline = IdleDeadline::create(deadline);
124         request.callback->handleEvent(idleDeadline.get());
125         if (!m_runnableIdleCallbacks.isEmpty())
126             queueTaskToInvokeIdleCallbacks(deadline);
127         return;
128     }
129 
130     if (!m_idleRequestCallbacks.isEmpty() || !m_runnableIdleCallbacks.isEmpty())
131         queueTaskToStartIdlePeriod();
132 }
