40 JSValue InjectedScriptHost::wrapper(JSGlobalObject* globalObject)
41 {
42     JSValue value = m_wrappers.getWrapper(globalObject);
43     if (value)
44         return value;
45 
46     VM& vm = globalObject->vm();
47     JSObject* prototype = JSInjectedScriptHost::createPrototype(vm, globalObject);
48     Structure* structure = JSInjectedScriptHost::createStructure(vm, globalObject, prototype);
49     JSInjectedScriptHost* injectedScriptHost = JSInjectedScriptHost::create(vm, structure, makeRef(*this));
50     m_wrappers.addWrapper(globalObject, injectedScriptHost);
51 
52     return injectedScriptHost;
53 }
54 
55 void InjectedScriptHost::clearAllWrappers()
56 {
57     m_wrappers.clearAllWrappers();
58 }
