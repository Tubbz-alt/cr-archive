119 inline bool FontCascadeFonts::isFixedPitch(const FontCascadeDescription& description)
120 {
121     if (m_pitch == UnknownPitch)
122         determinePitch(description);
123     return m_pitch == FixedPitch;
124 };
125 
126 inline const Font& FontCascadeFonts::primaryFont(const FontCascadeDescription& description)
127 {
128     ASSERT(isMainThread());
129     if (!m_cachedPrimaryFont) {
130         auto& primaryRanges = realizeFallbackRangesAt(description, 0);
131         m_cachedPrimaryFont = primaryRanges.glyphDataForCharacter(' ', ExternalResourceDownloadPolicy::Allow).font;
132         if (!m_cachedPrimaryFont)
133             m_cachedPrimaryFont = &primaryRanges.fontForFirstRange();
134         else if (m_cachedPrimaryFont->isInterstitial()) {
135             for (unsigned index = 1; ; ++index) {
136                 auto& localRanges = realizeFallbackRangesAt(description, index);
137                 if (localRanges.isNull())
138                     break;
139                 auto* font = localRanges.glyphDataForCharacter(' ', ExternalResourceDownloadPolicy::Forbid).font;
140                 if (font && !font->isInterstitial()) {
141                     m_cachedPrimaryFont = font;
142                     break;
143                 }
144             }
145         }
146     }
147     return *m_cachedPrimaryFont;
148 }
