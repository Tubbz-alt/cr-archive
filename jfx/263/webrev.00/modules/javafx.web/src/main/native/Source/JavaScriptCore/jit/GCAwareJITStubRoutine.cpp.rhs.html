 43 GCAwareJITStubRoutine::GCAwareJITStubRoutine(
 44     const MacroAssemblerCodeRef<JITStubRoutinePtrTag>& code, VM& vm)
 45     : JITStubRoutine(code)
 46 {
 47     vm.heap.m_jitStubRoutines->add(this);
 48 }
 49 
 50 GCAwareJITStubRoutine::~GCAwareJITStubRoutine() { }
 51 
 52 void GCAwareJITStubRoutine::observeZeroRefCount()
 53 {
 54     if (m_isJettisoned) {
 55         // This case is needed for when the system shuts down. It may be that
 56         // the JIT stub routine set gets deleted before we get around to deleting
 57         // this guy. In that case the GC informs us that we're jettisoned already
 58         // and that we should delete ourselves as soon as the ref count reaches
 59         // zero.
 60         delete this;
 61         return;
 62     }
 63 
 64     RELEASE_ASSERT(!m_refCount);
 65 
 66     m_isJettisoned = true;
 67 }
 68 
 69 void GCAwareJITStubRoutine::deleteFromGC()
 70 {
 71     ASSERT(m_isJettisoned);
 72     ASSERT(!m_refCount);
 73     ASSERT(!m_mayBeExecuting);
 74 
 75     delete this;
 76 }

 78 void GCAwareJITStubRoutine::markRequiredObjectsInternal(SlotVisitor&)

 82 MarkingGCAwareJITStubRoutine::MarkingGCAwareJITStubRoutine(
 83     const MacroAssemblerCodeRef<JITStubRoutinePtrTag>& code, VM& vm, const JSCell* owner,
 84     const Vector<JSCell*>& cells, Bag<CallLinkInfo>&& callLinkInfos)
 85     : GCAwareJITStubRoutine(code, vm)
 86     , m_cells(cells.size())
 87     , m_callLinkInfos(WTFMove(callLinkInfos))
 88 {
 89     for (unsigned i = cells.size(); i--;)
 90         m_cells[i].set(vm, owner, cells[i]);
 91 }
 92 
 93 MarkingGCAwareJITStubRoutine::~MarkingGCAwareJITStubRoutine()
 94 {
 95 }
 96 
 97 void MarkingGCAwareJITStubRoutine::markRequiredObjectsInternal(SlotVisitor& visitor)
 98 {
 99     for (auto& entry : m_cells)
100         visitor.append(entry);
101 }

104 GCAwareJITStubRoutineWithExceptionHandler::GCAwareJITStubRoutineWithExceptionHandler(
105     const MacroAssemblerCodeRef<JITStubRoutinePtrTag>& code, VM& vm,  const JSCell* owner, const Vector<JSCell*>& cells, Bag<CallLinkInfo>&& callLinkInfos,
106     CodeBlock* codeBlockForExceptionHandlers, DisposableCallSiteIndex exceptionHandlerCallSiteIndex)
107     : MarkingGCAwareJITStubRoutine(code, vm, owner, cells, WTFMove(callLinkInfos))
108     , m_codeBlockWithExceptionHandler(codeBlockForExceptionHandlers)
109     , m_exceptionHandlerCallSiteIndex(exceptionHandlerCallSiteIndex)
110 {
111     RELEASE_ASSERT(m_codeBlockWithExceptionHandler);
112     ASSERT(!!m_codeBlockWithExceptionHandler->handlerForIndex(exceptionHandlerCallSiteIndex.bits()));
113 }
114 
115 void GCAwareJITStubRoutineWithExceptionHandler::aboutToDie()
116 {
117     m_codeBlockWithExceptionHandler = nullptr;
118 }
119 
120 void GCAwareJITStubRoutineWithExceptionHandler::observeZeroRefCount()
121 {
122 #if ENABLE(DFG_JIT)
123     if (m_codeBlockWithExceptionHandler) {
124         m_codeBlockWithExceptionHandler->jitCode()->dfgCommon()->removeDisposableCallSiteIndex(m_exceptionHandlerCallSiteIndex);
125         m_codeBlockWithExceptionHandler->removeExceptionHandlerForCallSite(m_exceptionHandlerCallSiteIndex);
126         m_codeBlockWithExceptionHandler = nullptr;
127     }
128 #endif
129 
130     Base::observeZeroRefCount();
131 }

134 Ref<JITStubRoutine> createJITStubRoutine(
135     const MacroAssemblerCodeRef<JITStubRoutinePtrTag>& code,
136     VM& vm,
137     const JSCell* owner,
138     bool makesCalls,
139     const Vector<JSCell*>& cells,
140     Bag<CallLinkInfo>&& callLinkInfos,
141     CodeBlock* codeBlockForExceptionHandlers,
142     DisposableCallSiteIndex exceptionHandlerCallSiteIndex)
143 {
144     if (!makesCalls) {
145         // Allocating CallLinkInfos means we should have calls.
146         ASSERT(callLinkInfos.isEmpty());
147         return adoptRef(*new JITStubRoutine(code));
148     }
149 
150     if (codeBlockForExceptionHandlers) {
151         RELEASE_ASSERT(JITCode::isOptimizingJIT(codeBlockForExceptionHandlers->jitType()));
152         return adoptRef(*new GCAwareJITStubRoutineWithExceptionHandler(code, vm, owner, cells, WTFMove(callLinkInfos), codeBlockForExceptionHandlers, exceptionHandlerCallSiteIndex));
153     }
154 
155     if (cells.isEmpty() && callLinkInfos.isEmpty())
156         return GCAwareJITStubRoutine::create(code, vm);
157 
158     return adoptRef(*new MarkingGCAwareJITStubRoutine(code, vm, owner, cells, WTFMove(callLinkInfos)));
159 }
