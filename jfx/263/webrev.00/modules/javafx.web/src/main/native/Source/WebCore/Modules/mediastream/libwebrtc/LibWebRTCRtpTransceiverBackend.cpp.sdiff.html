 83 static inline ExceptionOr<webrtc::RtpCodecCapability> toRtpCodecCapability(const RTCRtpCodecCapability& codec)
 84 {
 85     webrtc::RtpCodecCapability rtcCodec;
 86     if (codec.mimeType.startsWith("video/"))
 87         rtcCodec.kind = cricket::MEDIA_TYPE_VIDEO;
 88     else if (codec.mimeType.startsWith("audio/"))
 89         rtcCodec.kind = cricket::MEDIA_TYPE_AUDIO;
 90     else
 91         return Exception { InvalidModificationError, "RTCRtpCodecCapability bad mimeType" };
 92 
 93     rtcCodec.name = codec.mimeType.substring(6).utf8().data();
 94     rtcCodec.clock_rate = codec.clockRate;
 95     if (codec.channels)
 96         rtcCodec.num_channels = *codec.channels;
 97 
 98     for (auto parameter : StringView(codec.sdpFmtpLine).split(';')) {
 99         auto position = parameter.find('=');
100         if (position == notFound)
101             return Exception { InvalidModificationError, "RTCRtpCodecCapability sdpFmtLine badly formated" };
102         rtcCodec.parameters.emplace(parameter.substring(0, position).utf8().data(), parameter.substring(position + 1).utf8().data());
103     }
104 
105     return rtcCodec;
106 }

108 ExceptionOr<void> LibWebRTCRtpTransceiverBackend::setCodecPreferences(const Vector<RTCRtpCodecCapability>& codecs)
109 {
110     std::vector<webrtc::RtpCodecCapability> rtcCodecs;
111     for (auto& codec : codecs) {
112         auto result = toRtpCodecCapability(codec);
113         if (result.hasException())
114             return result.releaseException();
115         rtcCodecs.push_back(result.releaseReturnValue());
116     }
117     auto result = m_rtcTransceiver->SetCodecPreferences(rtcCodecs);
118     if (!result.ok())
119         return toException(result);
120     return { };
121 }
