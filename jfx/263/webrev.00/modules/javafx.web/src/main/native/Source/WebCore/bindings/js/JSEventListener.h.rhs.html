 94 inline JSC::JSObject* JSEventListener::jsFunction(ScriptExecutionContext& scriptExecutionContext) const
 95 {
 96     // initializeJSFunction can trigger code that deletes this event listener
 97     // before we're done. It should always return null in this case.
 98     auto protect = makeRef(const_cast<JSEventListener&>(*this));
 99     JSC::Strong<JSC::JSObject> wrapper(m_isolatedWorld->vm(), m_wrapper.get());
100 
101     if (!m_jsFunction) {
102         auto* function = initializeJSFunction(scriptExecutionContext);
103         if (auto* wrapper = m_wrapper.get())
104             JSC::Heap::heap(wrapper)->writeBarrier(wrapper, function);
105         m_jsFunction = JSC::Weak<JSC::JSObject>(function);
106     }
107 
108     // Verify that we have a valid wrapper protecting our function from
109     // garbage collection. That is except for when we're not in the normal
110     // world and can have zombie m_jsFunctions.
111     ASSERT(!m_isolatedWorld->isNormal() || m_wrapper || !m_jsFunction);
112 
113     // If m_wrapper is null, then m_jsFunction is zombied, and should never be accessed.
114     if (!m_wrapper)
115         return nullptr;
116 
117     // Try to verify that m_jsFunction wasn't recycled. (Not exact, since an
118     // event listener can be almost anything, but this makes test-writing easier).
119     ASSERT(!m_jsFunction || static_cast<JSC::JSCell*>(m_jsFunction.get())->isObject());
120 
121     return m_jsFunction.get();
122 }
