68 void DesiredWeakReferences::reallyAdd(VM& vm, CommonData* common)
69 {

70     for (JSCell* target : m_references) {
71         if (Structure* structure = jsDynamicCast<Structure*>(vm, target)) {
72             common->weakStructureReferences.append(
73                 WriteBarrier<Structure>(vm, m_codeBlock, structure));

74         } else {
75             // There are weird relationships in how optimized CodeBlocks
76             // point to other CodeBlocks. We don't want to have them be
77             // part of the weak pointer set. For example, an optimized CodeBlock
78             // having a weak pointer to itself will cause it to get collected.
79             RELEASE_ASSERT(!jsDynamicCast<CodeBlock*>(vm, target));
80 

81             common->weakReferences.append(
82                 WriteBarrier<JSCell>(vm, m_codeBlock, target));
83         }
84     }
85 }

68 void DesiredWeakReferences::reallyAdd(VM& vm, CommonData* common)
69 {
70     ASSERT(vm.heap.isDeferred());
71     for (JSCell* target : m_references) {
72         if (Structure* structure = jsDynamicCast<Structure*>(vm, target)) {
73             ConcurrentJSLocker locker(m_codeBlock->m_lock);
74             // We do not emit WriteBarrier here since (1) GC is deferred and (2) we emit write-barrier on CodeBlock when finishing DFG::Plan::reallyAdd.
75             common->weakStructureReferences.append(structure->id());
76         } else {
77             // There are weird relationships in how optimized CodeBlocks
78             // point to other CodeBlocks. We don't want to have them be
79             // part of the weak pointer set. For example, an optimized CodeBlock
80             // having a weak pointer to itself will cause it to get collected.
81             RELEASE_ASSERT(!jsDynamicCast<CodeBlock*>(vm, target));
82 
83             ConcurrentJSLocker locker(m_codeBlock->m_lock);
84             common->weakReferences.append(
85                 WriteBarrier<JSCell>(vm, m_codeBlock, target));
86         }
87     }
88 }
