59 inline bool TextUtil::shouldPreserveTrailingWhitespace(const RenderStyle& style)
60 {
61     auto whitespace = style.whiteSpace();
62     return whitespace == WhiteSpace::Pre || whitespace == WhiteSpace::PreWrap || whitespace == WhiteSpace::BreakSpaces;
63 }
