 60 
 61 void initializeThreading()
 62 {
 63     static std::once_flag initializeThreadingOnceFlag;
 64 
 65     std::call_once(initializeThreadingOnceFlag, []{
 66         RELEASE_ASSERT(!g_jscConfig.initializeThreadingHasBeenCalled);
 67         g_jscConfig.initializeThreadingHasBeenCalled = true;
 68 
 69         WTF::initializeThreading();
 70         Options::initialize();
 71 
 72         initializePtrTagLookup();
 73 
 74 #if ENABLE(WRITE_BARRIER_PROFILING)
 75         WriteBarrierCounters::initialize();
 76 #endif
 77 
 78         ExecutableAllocator::initialize();
 79         VM::computeCanUseJIT();
 80 
 81         if (VM::canUseJIT() && Options::useSigillCrashAnalyzer())
 82             enableSigillCrashAnalyzer();
 83 
 84         LLInt::initialize();
 85 #ifndef NDEBUG
 86         DisallowGC::initialize();
 87         DisallowVMReentry::initialize();
 88 #endif
 89         initializeSuperSampler();
 90         Thread& thread = Thread::current();
 91         thread.setSavedLastStackTop(thread.stack().origin());
 92 
 93 #if ENABLE(WEBASSEMBLY)
 94         if (Wasm::isSupported()) {
 95             Wasm::Thunks::initialize();
 96             Wasm::CalleeRegistry::initialize();
 97         }
 98 #endif
 99 
100         if (VM::isInMiniMode())
101             WTF::fastEnableMiniMode();
102     });
103 }
