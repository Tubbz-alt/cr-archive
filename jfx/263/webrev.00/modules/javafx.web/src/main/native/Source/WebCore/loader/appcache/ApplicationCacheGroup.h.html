 27 
 28 #include "ApplicationCacheResourceLoader.h"
 29 #include "DOMApplicationCache.h"
 30 #include <wtf/Noncopyable.h>
 31 #include <wtf/HashMap.h>
 32 #include <wtf/HashSet.h>
 33 #include <wtf/URL.h>
 34 #include <wtf/WeakPtr.h>
 35 #include <wtf/text/WTFString.h>
 36 
 37 namespace WebCore {
 38 
 39 class ApplicationCache;
 40 class ApplicationCacheResource;
 41 class ApplicationCacheStorage;
 42 class Document;
 43 class DocumentLoader;
 44 class Frame;
 45 class SecurityOrigin;
 46 
 47 enum ApplicationCacheUpdateOption {
 48     ApplicationCacheUpdateWithBrowsingContext,
 49     ApplicationCacheUpdateWithoutBrowsingContext
 50 };
 51 
 52 class ApplicationCacheGroup : public CanMakeWeakPtr<ApplicationCacheGroup> {
 53     WTF_MAKE_NONCOPYABLE(ApplicationCacheGroup);
 54     WTF_MAKE_FAST_ALLOCATED;
 55 public:
 56     explicit ApplicationCacheGroup(Ref<ApplicationCacheStorage>&&, const URL& manifestURL);
 57     virtual ~ApplicationCacheGroup();
 58 
 59     enum UpdateStatus { Idle, Checking, Downloading };
 60 
 61     static ApplicationCache* cacheForMainRequest(const ResourceRequest&, DocumentLoader*);
 62     static ApplicationCache* fallbackCacheForMainRequest(const ResourceRequest&, DocumentLoader*);
 63 
 64     static void selectCache(Frame&, const URL& manifestURL);
 65     static void selectCacheWithoutManifestURL(Frame&);
 66 
 67     ApplicationCacheStorage& storage() { return m_storage; }
 68     const URL& manifestURL() const { return m_manifestURL; }
 69     const SecurityOrigin& origin() const { return m_origin.get(); }
 70     UpdateStatus updateStatus() const { return m_updateStatus; }
 71     void setUpdateStatus(UpdateStatus status);
 72 
 73     void setStorageID(unsigned storageID) { m_storageID = storageID; }
 74     unsigned storageID() const { return m_storageID; }
 75     void clearStorageID();
 76 
 77     void update(Frame&, ApplicationCacheUpdateOption); // FIXME: Frame should not be needed when updating without browsing context.
 78     void cacheDestroyed(ApplicationCache&);
 79 
 80     void abort(Frame&);
 81 
 82     bool cacheIsComplete(ApplicationCache& cache) { return m_caches.contains(&cache); }
 83 
 84     void stopLoadingInFrame(Frame&);
 85 
 86     ApplicationCache* newestCache() const { return m_newestCache.get(); }
 87     void setNewestCache(Ref<ApplicationCache>&&);
 88 
 89     void makeObsolete();
 90     bool isObsolete() const { return m_isObsolete; }
 91 
 92     void finishedLoadingMainResource(DocumentLoader&);
 93     void failedLoadingMainResource(DocumentLoader&);
 94 
 95     void disassociateDocumentLoader(DocumentLoader&);
 96 
 97 private:
 98     static void postListenerTask(const AtomString& eventType, const HashSet<DocumentLoader*>& set) { postListenerTask(eventType, 0, 0, set); }
 99     static void postListenerTask(const AtomString& eventType, DocumentLoader& loader)  { postListenerTask(eventType, 0, 0, loader); }
100     static void postListenerTask(const AtomString& eventType, int progressTotal, int progressDone, const HashSet<DocumentLoader*>&);
101     static void postListenerTask(const AtomString& eventType, int progressTotal, int progressDone, DocumentLoader&);
102 
103     void scheduleReachedMaxAppCacheSizeCallback();
104 
105     void didFinishLoadingManifest();
106     void didFailLoadingManifest(ApplicationCacheResourceLoader::Error);
107 
108     void didFailLoadingEntry(ApplicationCacheResourceLoader::Error, const URL&, unsigned type);
109     void didFinishLoadingEntry(const URL&);
110 
111     void didReachMaxAppCacheSize();
112     void didReachOriginQuota(int64_t totalSpaceNeeded);
113 
114     void startLoadingEntry();
115     void deliverDelayedMainResources();
116     void checkIfLoadIsComplete();
117     void cacheUpdateFailed();
118     void recalculateAvailableSpaceInQuota();
119     void manifestNotFound();
120 
121     void addEntry(const String&, unsigned type);
122 
123     void associateDocumentLoaderWithCache(DocumentLoader*, ApplicationCache*);
124 
125     void stopLoading();
126 
127     ResourceRequest createRequest(URL&&, ApplicationCacheResource*);
128 
129     Ref<ApplicationCacheStorage> m_storage;
130 
131     URL m_manifestURL;
132     Ref<SecurityOrigin> m_origin;
133     UpdateStatus m_updateStatus { Idle };
134 
135     // This is the newest complete cache in the group.
136     RefPtr<ApplicationCache> m_newestCache;
137 
138     // All complete caches in this cache group.
139     HashSet<ApplicationCache*> m_caches;
140 
141     // The cache being updated (if any). Note that cache updating does not immediately create a new
142     // ApplicationCache object, so this may be null even when update status is not Idle.
143     RefPtr<ApplicationCache> m_cacheBeingUpdated;
144 
145     // List of pending master entries, used during the update process to ensure that new master entries are cached.
146     HashSet<DocumentLoader*> m_pendingMasterResourceLoaders;
147     // How many of the above pending master entries have not yet finished downloading.
148     int m_downloadingPendingMasterResourceLoadersCount { 0 };
149 
150     // These are all the document loaders that are associated with a cache in this group.
151     HashSet<DocumentLoader*> m_associatedDocumentLoaders;
152 
153     // The URLs and types of pending cache entries.
154     HashMap<String, unsigned> m_pendingEntries;
155 
156     // The total number of items to be processed to update the cache group and the number that have been done.
157     int m_progressTotal { 0 };
158     int m_progressDone { 0 };
159 
160     // Frame used for fetching resources when updating.
161     // FIXME: An update started by a particular frame should not stop if it is destroyed, but there are other frames associated with the same cache group.
162     Frame* m_frame { nullptr };
163 
164     // An obsolete cache group is never stored, but the opposite is not true - storing may fail for multiple reasons, such as exceeding disk quota.
165     unsigned m_storageID { 0 };
166     bool m_isObsolete { false };
167 
168     // During update, this is used to handle asynchronously arriving results.
169     enum CompletionType {
170         None,
171         NoUpdate,
172         Failure,
173         Completed
174     };
175     CompletionType m_completionType { None };
176 
177     // This flag is set immediately after the ChromeClient::reachedMaxAppCacheSize() callback is invoked as a result of the storage layer failing to save a cache
178     // due to reaching the maximum size of the application cache database file. This flag is used by ApplicationCacheGroup::checkIfLoadIsComplete() to decide
179     // the course of action in case of this failure (i.e. call the ChromeClient callback or run the failure steps).
180     bool m_calledReachedMaxAppCacheSize { false };
181 
182     RefPtr<ApplicationCacheResource> m_currentResource;
183     RefPtr<ApplicationCacheResourceLoader> m_entryLoader;
184     unsigned long m_currentResourceIdentifier;
185 
186     RefPtr<ApplicationCacheResource> m_manifestResource;
187     RefPtr<ApplicationCacheResourceLoader> m_manifestLoader;
188 
189     int64_t m_availableSpaceInQuota;
190     bool m_originQuotaExceededPreviously { false };
191 
192     friend class ChromeClientCallbackTimer;
193 };
194 
195 } // namespace WebCore
