26 function performIteration(iterable)
27 {
28     "use strict";
29     // This is performing a spread operation on the iterable passed in,
30     // and returning the result in an array.
31     // https://tc39.github.io/ecma262/#sec-runtime-semantics-arrayaccumulation
32 
33     let result = [];
34 
35     let iterator = iterable.@iteratorSymbol();
36     let next = iterator.next;
37     let item;
38     let index = 0;
39     while (true) {
40         item = next.@call(iterator);
41         if (!@isObject(item))
42             @throwTypeError("Iterator result interface is not an object");
43         if (item.done)
44             return result;
45         @putByValDirect(result, index++, item.value);
46     }
47 }
