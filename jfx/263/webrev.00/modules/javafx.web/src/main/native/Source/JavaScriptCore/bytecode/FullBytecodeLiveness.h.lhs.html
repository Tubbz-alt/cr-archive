28 #include <wtf/FastBitVector.h>
29 
30 namespace JSC {
31 
32 class BytecodeLivenessAnalysis;

33 
34 typedef HashMap<unsigned, FastBitVector, WTF::IntHash<unsigned>, WTF::UnsignedWithZeroKeyHashTraits<unsigned>> BytecodeToBitmapMap;
35 
36 class FullBytecodeLiveness {
37     WTF_MAKE_FAST_ALLOCATED;
38 public:
39     const FastBitVector& getLiveness(unsigned bytecodeIndex) const
40     {
41         return m_map[bytecodeIndex];







42     }
43 
44     bool operandIsLive(int operand, unsigned bytecodeIndex) const
45     {
46         return operandIsAlwaysLive(operand) || operandThatIsNotAlwaysLiveIsLive(getLiveness(bytecodeIndex), operand);
47     }
48 
49 private:
50     friend class BytecodeLivenessAnalysis;
51 
52     Vector<FastBitVector, 0, UnsafeVectorOverflow> m_map;



53 };
