26 namespace WebCore {
27 using namespace JSC;
28 
29 inline void* root(OffscreenCanvas* canvas)
30 {
31     return canvas;
32 }
33 
34 bool JSOffscreenCanvasRenderingContext2DOwner::isReachableFromOpaqueRoots(JSC::Handle<JSC::Unknown> handle, void*, SlotVisitor& visitor, const char** reason)
35 {
36     if (UNLIKELY(reason))
37         *reason = "Canvas is opaque root";
38 
39     JSOffscreenCanvasRenderingContext2D* jsOffscreenCanvasRenderingContext = jsCast<JSOffscreenCanvasRenderingContext2D*>(handle.slot()->asCell());
40     void* root = WebCore::root(&jsOffscreenCanvasRenderingContext->wrapped().canvas());
41     return visitor.containsOpaqueRoot(root);
42 }
43 
44 void JSOffscreenCanvasRenderingContext2D::visitAdditionalChildren(SlotVisitor& visitor)
45 {
46     visitor.addOpaqueRoot(root(&wrapped().canvas()));
47 }
