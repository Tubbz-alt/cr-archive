 46 JSWebAssemblyModule* JSWebAssemblyModule::createStub(VM& vm, JSGlobalObject* globalObject, Structure* structure, Wasm::Module::ValidationResult&& result)
 47 {
 48     auto scope = DECLARE_THROW_SCOPE(vm);
 49     if (!result.has_value()) {
 50         auto* error = JSWebAssemblyCompileError::create(globalObject, vm, structure->globalObject()->webAssemblyCompileErrorStructure(), result.error());
 51         RETURN_IF_EXCEPTION(scope, nullptr);
 52         throwException(globalObject, scope, error);
 53         return nullptr;
 54     }
 55 
 56     auto* module = new (NotNull, allocateCell<JSWebAssemblyModule>(vm.heap)) JSWebAssemblyModule(vm, structure, result.value().releaseNonNull());
 57     module->finishCreation(vm);
 58     return module;
 59 }

 73 void JSWebAssemblyModule::finishCreation(VM& vm)
 74 {
 75     Base::finishCreation(vm);
 76     ASSERT(inherits(vm, info()));
 77 
 78     // On success, a new WebAssembly.Module object is returned with [[Module]] set to the validated Ast.module.
 79     SymbolTable* exportSymbolTable = SymbolTable::create(vm);
 80     const Wasm::ModuleInformation& moduleInformation = m_module->moduleInformation();
 81     for (auto& exp : moduleInformation.exports) {
 82         auto offset = exportSymbolTable->takeNextScopeOffset(NoLockingNecessary);
 83         String field = String::fromUTF8(exp.field);
 84         exportSymbolTable->set(NoLockingNecessary, AtomString(field).impl(), SymbolTableEntry(VarOffset(offset)));
 85     }
 86 
 87     m_exportSymbolTable.set(vm, this, exportSymbolTable);

 88 }

126 void JSWebAssemblyModule::visitChildren(JSCell* cell, SlotVisitor& visitor)
127 {
128     JSWebAssemblyModule* thisObject = jsCast<JSWebAssemblyModule*>(cell);
129     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
130 
131     Base::visitChildren(thisObject, visitor);
132     visitor.append(thisObject->m_exportSymbolTable);

133     for (unsigned i = 0; i < Wasm::NumberOfMemoryModes; ++i)
134         visitor.append(thisObject->m_codeBlocks[i]);
135 }
