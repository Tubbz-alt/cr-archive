41 JSSet* JSSet::clone(ExecState* exec, VM& vm, Structure* structure)
42 {
43     JSSet* instance = new (NotNull, allocateCell<JSSet>(vm.heap)) JSSet(vm, structure);
44     instance->finishCreation(exec, vm, this);
45     return instance;
46 }

48 bool JSSet::isIteratorProtocolFastAndNonObservable()
49 {
50     JSGlobalObject* globalObject = this->globalObject();
51     if (!globalObject->isSetPrototypeIteratorProtocolFastAndNonObservable())
52         return false;
53 
54     VM& vm = globalObject->vm();
55     Structure* structure = this->structure(vm);
56     // This is the fast case. Many sets will be an original set.
57     if (structure == globalObject->setStructure())
58         return true;
59 
60     if (getPrototypeDirect(vm) != globalObject->jsSetPrototype())
61         return false;
62 
63     if (getDirectOffset(vm, vm.propertyNames->iteratorSymbol) != invalidOffset)
64         return false;
65 
66     return true;
67 }
68 
69 bool JSSet::canCloneFastAndNonObservable(Structure* structure)
70 {
71     auto addFastAndNonObservable = [&] (Structure* structure) {
72         JSGlobalObject* globalObject = structure->globalObject();
73         if (!globalObject->isSetPrototypeAddFastAndNonObservable())
74             return false;
75 
76         if (structure->hasPolyProto())
77             return false;
78 
79         if (structure->storedPrototype() != globalObject->jsSetPrototype())
80             return false;
81 
82         return true;
83     };
84 
85     return isIteratorProtocolFastAndNonObservable() && addFastAndNonObservable(structure);
86 }
