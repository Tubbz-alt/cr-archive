 52 DOMWindow* domWindowFromExecState(JSC::ExecState* scriptState)
 53 {
 54     JSC::JSGlobalObject* globalObject = scriptState->lexicalGlobalObject();
 55     JSC::VM& vm = globalObject->vm();
 56     if (!globalObject->inherits<JSDOMWindowBase>(vm))
 57         return nullptr;
 58     return &JSC::jsCast<JSDOMWindowBase*>(globalObject)->wrapped();
 59 }

 84 JSC::ExecState* execStateFromNode(DOMWrapperWorld& world, Node* node)
 85 {
 86     if (!node)
 87         return nullptr;
 88     Frame* frame = node->document().frame();
 89     if (!frame)
 90         return nullptr;
 91     if (!frame->script().canExecuteScripts(NotAboutToExecuteScript))
 92         return nullptr;
 93     return frame->script().globalObject(world)->globalExec();
 94 }

 96 JSC::ExecState* execStateFromPage(DOMWrapperWorld& world, Page* page)
 97 {
 98     return page ? page->mainFrame().script().globalObject(world)->globalExec() : nullptr;
 99 }
