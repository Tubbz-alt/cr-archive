33 inline SVGEllipseElement::SVGEllipseElement(const QualifiedName& tagName, Document& document)
34     : SVGGeometryElement(tagName, document)

35 {
36     ASSERT(hasTagName(SVGNames::ellipseTag));
37 
38     static std::once_flag onceFlag;
39     std::call_once(onceFlag, [] {
40         PropertyRegistry::registerProperty<SVGNames::cxAttr, &SVGEllipseElement::m_cx>();
41         PropertyRegistry::registerProperty<SVGNames::cyAttr, &SVGEllipseElement::m_cy>();
42         PropertyRegistry::registerProperty<SVGNames::rxAttr, &SVGEllipseElement::m_rx>();
43         PropertyRegistry::registerProperty<SVGNames::ryAttr, &SVGEllipseElement::m_ry>();
44     });
45 }
46 
47 Ref<SVGEllipseElement> SVGEllipseElement::create(const QualifiedName& tagName, Document& document)
48 {
49     return adoptRef(*new SVGEllipseElement(tagName, document));
50 }
51 
52 void SVGEllipseElement::parseAttribute(const QualifiedName& name, const AtomString& value)
53 {
54     SVGParsingError parseError = NoError;
55 
56     if (name == SVGNames::cxAttr)
57         m_cx->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
58     else if (name == SVGNames::cyAttr)
59         m_cy->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
60     else if (name == SVGNames::rxAttr)
61         m_rx->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError, SVGLengthNegativeValuesMode::Forbid));
62     else if (name == SVGNames::ryAttr)
63         m_ry->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError, SVGLengthNegativeValuesMode::Forbid));
64 
65     reportAttributeParsingError(parseError, name, value);
66 
67     SVGGeometryElement::parseAttribute(name, value);

68 }
69 
70 void SVGEllipseElement::svgAttributeChanged(const QualifiedName& attrName)
71 {
72     if (PropertyRegistry::isKnownAttribute(attrName)) {
73         InstanceInvalidationGuard guard(*this);
74         invalidateSVGPresentationAttributeStyle();
75         return;
76     }
77 
78     SVGGeometryElement::svgAttributeChanged(attrName);

79 }
80 
81 RenderPtr<RenderElement> SVGEllipseElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
82 {
83     return createRenderer<RenderSVGEllipse>(*this, WTFMove(style));
84 }
