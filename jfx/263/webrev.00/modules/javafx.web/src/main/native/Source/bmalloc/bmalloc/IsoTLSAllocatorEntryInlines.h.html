26 #pragma once
27 
28 #include "IsoHeapImpl.h"
29 
30 namespace bmalloc {
31 
32 template<typename Config>
33 IsoTLSAllocatorEntry<Config>::IsoTLSAllocatorEntry(IsoHeapImpl<Config>& heap)
34     : m_heap(heap)
35 {
36 }
37 
38 template<typename Config>
39 IsoTLSAllocatorEntry<Config>::~IsoTLSAllocatorEntry()
40 {
41 }
42 
43 template<typename Config>
44 void IsoTLSAllocatorEntry<Config>::construct(void* dst)
45 {
46     new (dst) IsoAllocator<Config>(m_heap);
47 }
48 
49 template<typename Config>
50 void IsoTLSAllocatorEntry<Config>::scavenge(void* entry)
51 {
52     static_cast<IsoAllocator<Config>*>(entry)->scavenge(m_heap);
53 }
54 
55 } // namespace bmalloc
