 37 
 38 namespace WebCore {
 39 
 40 class KeyedDecoder;
 41 class KeyedEncoder;
 42 
 43 struct ResourceLoadStatistics {
 44     explicit ResourceLoadStatistics(const RegistrableDomain& domain)
 45         : registrableDomain { domain }
 46     {
 47     }
 48 
 49     ResourceLoadStatistics() = default;
 50 
 51     ResourceLoadStatistics(const ResourceLoadStatistics&) = delete;
 52     ResourceLoadStatistics& operator=(const ResourceLoadStatistics&) = delete;
 53     ResourceLoadStatistics(ResourceLoadStatistics&&) = default;
 54     ResourceLoadStatistics& operator=(ResourceLoadStatistics&&) = default;
 55 
 56     static constexpr Seconds NoExistingTimestamp { -1 };
 57 
 58     WEBCORE_EXPORT static WallTime reduceTimeResolution(WallTime);
 59 
 60     WEBCORE_EXPORT void encode(KeyedEncoder&) const;
 61     WEBCORE_EXPORT bool decode(KeyedDecoder&, unsigned modelVersion);
 62 
 63     WEBCORE_EXPORT String toString() const;
 64 
 65     WEBCORE_EXPORT void merge(const ResourceLoadStatistics&);
 66 
 67     RegistrableDomain registrableDomain;
 68 
 69     WallTime lastSeen;
 70 
 71     // User interaction
 72     bool hadUserInteraction { false };
 73     // Timestamp. Default value is negative, 0 means it was reset.
 74     WallTime mostRecentUserInteractionTime { WallTime::fromRawSeconds(-1) };
 75     bool grandfathered { false };
 76 
 77     // Storage access
 78     HashSet<RegistrableDomain> storageAccessUnderTopFrameDomains;
 79 
 80     // Top frame stats
 81     HashSet<RegistrableDomain> topFrameUniqueRedirectsTo;
 82     HashSet<RegistrableDomain> topFrameUniqueRedirectsFrom;
 83     HashSet<RegistrableDomain> topFrameLinkDecorationsFrom;
 84     bool gotLinkDecorationFromPrevalentResource { false };
 85     HashSet<RegistrableDomain> topFrameLoadedThirdPartyScripts;
 86 
 87     // Subframe stats
 88     HashSet<RegistrableDomain> subframeUnderTopFrameDomains;
 89 
 90     // Subresource stats
 91     HashSet<RegistrableDomain> subresourceUnderTopFrameDomains;
 92     HashSet<RegistrableDomain> subresourceUniqueRedirectsTo;
 93     HashSet<RegistrableDomain> subresourceUniqueRedirectsFrom;
 94 
 95     // Prevalent resource stats
 96     bool isPrevalentResource { false };
 97     bool isVeryPrevalentResource { false };
 98     unsigned dataRecordsRemoved { 0 };
 99     unsigned timesAccessedAsFirstPartyDueToUserInteraction { 0 };
100     unsigned timesAccessedAsFirstPartyDueToStorageAccessAPI { 0 };
101 
102     enum class NavigatorAPI : uint64_t {
103         AppVersion = 1 << 0,
104         UserAgent = 1 << 1,
105         Plugins = 1 << 2,
106         MimeTypes = 1 << 3,
107         CookieEnabled = 1 << 4,
108         JavaEnabled = 1 << 5,
109     };
110     enum class ScreenAPI : uint64_t {
111         Height = 1 << 0,
112         Width = 1 << 1,
113         ColorDepth = 1 << 2,
114         PixelDepth = 1 << 3,
115         AvailLeft = 1 << 4,
116         AvailTop = 1 << 5,
117         AvailHeight = 1 << 6,
118         AvailWidth = 1 << 7,
119     };
120 #if ENABLE(WEB_API_STATISTICS)
121     // This set represents the registrable domain of the top frame where web API
122     // were used in the top frame or one of its subframes.
123     HashCountedSet<String> topFrameRegistrableDomainsWhichAccessedWebAPIs;
124     HashSet<String> fontsFailedToLoad;
125     HashSet<String> fontsSuccessfullyLoaded;
126     CanvasActivityRecord canvasActivityRecord;
127     OptionSet<NavigatorAPI> navigatorFunctionsAccessed;
128     OptionSet<ScreenAPI> screenFunctionsAccessed;
129 #endif
130 };
131 
132 } // namespace WebCore
