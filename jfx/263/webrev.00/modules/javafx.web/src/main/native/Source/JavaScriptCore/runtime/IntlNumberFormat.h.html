 26 #pragma once
 27 
 28 #if ENABLE(INTL)
 29 
 30 #include "JSObject.h"
 31 #include <unicode/unum.h>
 32 #include <unicode/uvernum.h>
 33 
 34 #define HAVE_ICU_FORMAT_DOUBLE_FOR_FIELDS (U_ICU_VERSION_MAJOR_NUM >= 59)
 35 
 36 namespace JSC {
 37 
 38 class IntlNumberFormatConstructor;
 39 class JSBoundFunction;
 40 
 41 class IntlNumberFormat final : public JSNonFinalObject {
 42 public:
 43     using Base = JSNonFinalObject;
 44 
 45     static constexpr bool needsDestruction = true;
 46 
 47     static void destroy(JSCell* cell)
 48     {
 49         static_cast<IntlNumberFormat*>(cell)->IntlNumberFormat::~IntlNumberFormat();
 50     }
 51 
 52     template<typename CellType, SubspaceAccess mode>
 53     static IsoSubspace* subspaceFor(VM& vm)
 54     {
 55         return vm.intlNumberFormatSpace<mode>();
 56     }
 57 
 58     static IntlNumberFormat* create(VM&, Structure*);
 59     static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
 60 
 61     DECLARE_INFO;
 62 
 63     void initializeNumberFormat(JSGlobalObject*, JSValue locales, JSValue optionsValue);
 64     JSValue formatNumber(JSGlobalObject*, double number);
 65 #if HAVE(ICU_FORMAT_DOUBLE_FOR_FIELDS)
 66     JSValue formatToParts(JSGlobalObject*, double value);
 67 #endif
 68     JSObject* resolvedOptions(JSGlobalObject*);
 69 
 70     JSBoundFunction* boundFormat() const { return m_boundFormat.get(); }
 71     void setBoundFormat(VM&, JSBoundFunction*);
 72 
 73 protected:
 74     IntlNumberFormat(VM&, Structure*);
 75     void finishCreation(VM&);
 76     static void visitChildren(JSCell*, SlotVisitor&);
 77 
 78 private:
 79     enum class Style : uint8_t { Decimal, Percent, Currency };
 80     enum class CurrencyDisplay : uint8_t { Code, Symbol, Name };
 81 
 82     struct UNumberFormatDeleter {
 83         void operator()(UNumberFormat*) const;
 84     };
 85 
 86     static ASCIILiteral styleString(Style);
 87     static ASCIILiteral currencyDisplayString(CurrencyDisplay);
 88 
 89     String m_locale;
 90     String m_numberingSystem;
 91     String m_currency;
 92     std::unique_ptr<UNumberFormat, UNumberFormatDeleter> m_numberFormat;
 93     WriteBarrier<JSBoundFunction> m_boundFormat;
 94     unsigned m_minimumIntegerDigits { 1 };
 95     unsigned m_minimumFractionDigits { 0 };
 96     unsigned m_maximumFractionDigits { 3 };
 97     unsigned m_minimumSignificantDigits { 0 };
 98     unsigned m_maximumSignificantDigits { 0 };
 99     Style m_style { Style::Decimal };
100     CurrencyDisplay m_currencyDisplay;
101     bool m_useGrouping { true };
102     bool m_initializedNumberFormat { false };
103 
104 #if HAVE(ICU_FORMAT_DOUBLE_FOR_FIELDS)
105     struct UFieldPositionIteratorDeleter {
106         void operator()(UFieldPositionIterator*) const;
107     };
108 
109     struct IntlNumberFormatField {
110         int32_t type;
111         int32_t size;
112         IntlNumberFormatField(int32_t type, int32_t size)
113             : type(type)
114             , size(size)
115         { }
116     };
117 
118     static ASCIILiteral partTypeString(UNumberFormatFields, double);
119 #endif
120 };
121 
122 } // namespace JSC
123 
124 #endif // ENABLE(INTL)
