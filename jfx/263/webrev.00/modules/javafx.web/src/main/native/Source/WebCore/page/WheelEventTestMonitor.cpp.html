 43 WheelEventTestMonitor::WheelEventTestMonitor()
 44     : m_testForCompletionTimer(RunLoop::current(), this, &WheelEventTestMonitor::triggerTestTimerFired)
 45 {
 46 }
 47 
 48 void WheelEventTestMonitor::clearAllTestDeferrals()
 49 {
 50     ASSERT(isMainThread());
 51     m_deferCompletionReasons.clear();
 52     m_completionCallback = nullptr;
 53     m_testForCompletionTimer.stop();
 54     LOG_WITH_STREAM(WheelEventTestMonitor, stream << "      (=) WheelEventTestMonitor::clearAllTestDeferrals: cleared all test state.");
 55 }
 56 
 57 void WheelEventTestMonitor::setTestCallbackAndStartNotificationTimer(WTF::Function<void()>&& functionCallback)
 58 {
 59     ASSERT(isMainThread());
 60     m_completionCallback = WTFMove(functionCallback);
 61 
 62     if (!m_testForCompletionTimer.isActive())
 63         m_testForCompletionTimer.startRepeating(1_s / 60.);
 64 }
 65 
 66 void WheelEventTestMonitor::deferForReason(ScrollableAreaIdentifier identifier, DeferReason reason)
 67 {
 68     ASSERT(isMainThread());
 69     m_deferCompletionReasons.ensure(identifier, [] {
 70         return OptionSet<DeferReason>();
 71     }).iterator->value.add(reason);
 72 
 73     LOG_WITH_STREAM(WheelEventTestMonitor, stream << "      (=) WheelEventTestMonitor::deferForReason: id=" << identifier << ", reason=" << reason);
 74 }
 75 
 76 void WheelEventTestMonitor::removeDeferralForReason(ScrollableAreaIdentifier identifier, DeferReason reason)
 77 {
 78     ASSERT(isMainThread());
 79     auto it = m_deferCompletionReasons.find(identifier);
 80     if (it == m_deferCompletionReasons.end())
 81         return;
 82 
 83     LOG_WITH_STREAM(WheelEventTestMonitor, stream << "      (=) WheelEventTestMonitor::removeDeferralForReason: id=" << identifier << ", reason=" << reason);
 84     it->value.remove(reason);
 85 
 86     if (it->value.isEmpty())
 87         m_deferCompletionReasons.remove(it);
 88 }
 89 
 90 void WheelEventTestMonitor::triggerTestTimerFired()
 91 {
 92     ASSERT(isMainThread());
 93     if (!m_deferCompletionReasons.isEmpty()) {
 94         LOG_WITH_STREAM(WheelEventTestMonitor, stream << "  WheelEventTestMonitor::triggerTestTimerFired - scrolling still active, reasons " << m_deferCompletionReasons);
 95         return;
 96     }
 97 
 98     auto functionCallback = WTFMove(m_completionCallback);
 99     m_testForCompletionTimer.stop();
100 
101     LOG_WITH_STREAM(WheelEventTestMonitor, stream << "  WheelEventTestMonitor::triggerTestTimerFired: scrolling is idle, FIRING TEST");
102     if (functionCallback)
103         functionCallback();
104 }
105 
106 TextStream& operator<<(TextStream& ts, WheelEventTestMonitor::DeferReason reason)
107 {
108     switch (reason) {
109     case WheelEventTestMonitor::HandlingWheelEvent: ts << "handling wheel event"; break;
110     case WheelEventTestMonitor::RubberbandInProgress: ts << "rubberbanding"; break;
111     case WheelEventTestMonitor::ScrollSnapInProgress: ts << "scroll-snapping"; break;
112     case WheelEventTestMonitor::ScrollingThreadSyncNeeded: ts << "scrolling thread sync needed"; break;
113     case WheelEventTestMonitor::ContentScrollInProgress: ts << "content scrolling"; break;
114     }
115     return ts;
116 }
117 
118 TextStream& operator<<(TextStream& ts, const WheelEventTestMonitor::ScrollableAreaReasonMap& reasonMap)
119 {
120     for (const auto& regionReasonsPair : reasonMap)
121         ts << "   scroll region: " << regionReasonsPair.key << " reasons: " << regionReasonsPair.value;
122 
123     return ts;
124 }
