38 void VariableEvent::dump(PrintStream& out) const
39 {
40     switch (kind()) {
41     case Reset:
42         out.printf("Reset");
43         break;
44     case BirthToFill:
45         dumpFillInfo("BirthToFill", out);
46         break;
47     case BirthToSpill:
48         dumpSpillInfo("BirthToSpill", out);
49         break;
50     case Birth:
51         out.print("Birth(", id(), ")");
52         break;
53     case Fill:
54         dumpFillInfo("Fill", out);
55         break;
56     case Spill:
57         dumpSpillInfo("Spill", out);
58         break;
59     case Death:
60         out.print("Death(", id(), ")");
61         break;
62     case MovHintEvent:
63         out.print("MovHint(", id(), ", ", operand(), ")");
64         break;
65     case SetLocalEvent:
66         out.print(
67             "SetLocal(machine:", machineRegister(), " -> bytecode:", operand(),
68             ", ", dataFormatToString(dataFormat()), ")");
69         break;
70     default:
71         RELEASE_ASSERT_NOT_REACHED();
72         break;
73     }
74 }
75 
76 void VariableEvent::dumpFillInfo(const char* name, PrintStream& out) const
77 {
78     out.print(name, "(", id(), ", ");
79     if (dataFormat() == DataFormatDouble)
80         out.printf("%s", FPRInfo::debugName(fpr()));
81 #if USE(JSVALUE32_64)
82     else if (dataFormat() & DataFormatJS)
83         out.printf("%s:%s", GPRInfo::debugName(tagGPR()), GPRInfo::debugName(payloadGPR()));
84 #endif
85     else
86         out.printf("%s", GPRInfo::debugName(gpr()));
87     out.printf(", %s)", dataFormatToString(dataFormat()));
88 }
89 
90 void VariableEvent::dumpSpillInfo(const char* name, PrintStream& out) const
91 {
92     out.print(name, "(", id(), ", ", spillRegister(), ", ", dataFormatToString(dataFormat()), ")");
93 }
