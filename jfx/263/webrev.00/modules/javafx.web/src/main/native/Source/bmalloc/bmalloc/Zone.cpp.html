 31 template<typename T> static void remoteRead(task_t task, memory_reader_t reader, vm_address_t remotePointer, T& result)
 32 {
 33     void* tmp = nullptr;
 34     kern_return_t error = reader(task, remotePointer, sizeof(T), &tmp);
 35 
 36     // This read sometimes fails for unknown reasons (<rdar://problem/14093757>).
 37     // Avoid a crash by skipping the memcpy when this happens.
 38     if (error || !tmp) {
 39         fprintf(stderr, "bmalloc: error reading remote process: 0x%x\n", error);
 40         return;
 41     }
 42 
 43     memcpy(&result, tmp, sizeof(T));
 44 }

 46 // These function pointers are invoked unconditionally on all zones by various
 47 // system tools. We don't support any of these features, but we provide
 48 // just enough functionality not to crash.
 49 
 50 static size_t good_size(malloc_zone_t*, size_t size)
 51 {
 52     return size;
 53 }
 54 
 55 static boolean_t check(malloc_zone_t*)
 56 {
 57     return true;
 58 }
 59 
 60 static void print(malloc_zone_t*, boolean_t)
 61 {
 62 }
 63 
 64 static void log(malloc_zone_t*, void*)
 65 {
 66 }
 67 
 68 static void force_lock(malloc_zone_t*)
 69 {
 70 }
 71 
 72 static void force_unlock(malloc_zone_t*)
 73 {
 74 }

 76 static void statistics(malloc_zone_t*, malloc_statistics_t* statistics)
 77 {
 78     memset(statistics, 0, sizeof(malloc_statistics_t));
 79 }

 81 static size_t zoneSize(malloc_zone_t*, const void*)
 82 {
 83     // Our zone is not public API, so no pointer can belong to us.
 84     return 0;
 85 }

 87 // This function runs inside the leaks process.
 88 static kern_return_t enumerator(task_t task, void* context, unsigned type_mask, vm_address_t zone_address, memory_reader_t reader, vm_range_recorder_t recorder)
 89 {
 90     Zone remoteZone(task, reader, zone_address);
 91     for (auto& range : remoteZone.ranges()) {
 92         vm_range_t vmRange = { reinterpret_cast<vm_address_t>(range.begin()), range.size() };
 93 
 94         if ((type_mask & MALLOC_PTR_REGION_RANGE_TYPE))
 95             (*recorder)(task, context, MALLOC_PTR_REGION_RANGE_TYPE, &vmRange, 1);
 96 
 97         if ((type_mask & MALLOC_PTR_IN_USE_RANGE_TYPE))
 98             (*recorder)(task, context, MALLOC_PTR_IN_USE_RANGE_TYPE, &vmRange, 1);
 99     }
100 
101     return 0;
102 }

104 // The memory analysis API requires the contents of this struct to be a static
105 // constant in the program binary. The leaks process will load this struct
106 // out of the program binary (and not out of the running process).
107 static const malloc_introspection_t zoneIntrospect = {
108     .enumerator = bmalloc::enumerator,
109     .good_size = bmalloc::good_size,
110     .check = bmalloc::check,
111     .print = bmalloc::print,
112     .log = bmalloc::log,
113     .force_lock = bmalloc::force_lock,
114     .force_unlock = bmalloc::force_unlock,
115     .statistics = bmalloc::statistics
116 };

118 Zone::Zone(const LockHolder&)
119 {
120     malloc_zone_t::size = &bmalloc::zoneSize;
121     malloc_zone_t::zone_name = "WebKit Malloc";
122     malloc_zone_t::introspect = const_cast<malloc_introspection_t*>(&bmalloc::zoneIntrospect);
123     malloc_zone_t::version = 4;
124     malloc_zone_register(this);
125 }
126 
127 Zone::Zone(task_t task, memory_reader_t reader, vm_address_t remotePointer)
128 {
129     remoteRead(task, reader, remotePointer, *this);
130 }
