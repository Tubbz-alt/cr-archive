37 StructureChain::StructureChain(VM& vm, Structure* structure, StructureID* vector)
38     : Base(vm, structure)
39     , m_vector(vm, this, vector)
40 {
41 }
42 
43 StructureChain* StructureChain::create(VM& vm, JSObject* head)
44 {
45     // FIXME: Make StructureChain::create fail for large chain. Caching large chain is not so profitable.
46     // By making the size <= UINT16_MAX, we can store length in a high bits of auxiliary pointer.
47     // https://bugs.webkit.org/show_bug.cgi?id=200290
48     size_t size = 0;
49     for (JSObject* current = head; current; current = current->structure(vm)->storedPrototypeObject(current))
50         ++size;
51     ++size; // Sentinel nullptr.
52     size_t bytes = (Checked<size_t>(size) * sizeof(StructureID)).unsafeGet();
53     StructureID* vector = static_cast<StructureID*>(vm.jsValueGigacageAuxiliarySpace.allocateNonVirtual(vm, bytes, nullptr, AllocationFailureMode::Assert));
54     memset(vector, 0, bytes);
55     StructureChain* chain = new (NotNull, allocateCell<StructureChain>(vm.heap)) StructureChain(vm, vm.structureChainStructure.get(), vector);
56     chain->finishCreation(vm, head);
57     return chain;
58 }
59 
60 void StructureChain::finishCreation(VM& vm, JSObject* head)
61 {
62     Base::finishCreation(vm);
63     size_t i = 0;
64     for (JSObject* current = head; current; current = current->structure(vm)->storedPrototypeObject(current)) {
65         Structure* structure = current->structure(vm);
66         m_vector.get()[i++] = structure->id();
67         vm.heap.writeBarrier(this);
68     }
69 }
70 
71 void StructureChain::visitChildren(JSCell* cell, SlotVisitor& visitor)
72 {
73     StructureChain* thisObject = jsCast<StructureChain*>(cell);
74     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
75     Base::visitChildren(thisObject, visitor);
76     visitor.markAuxiliary(thisObject->m_vector.get());
77     VM& vm = visitor.vm();
78     for (auto* current = thisObject->m_vector.get(); *current; ++current) {
79         StructureID structureID = *current;
80         Structure* structure = vm.getStructure(structureID);
81         visitor.appendUnbarriered(structure);
82     }
83 }
