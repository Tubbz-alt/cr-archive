 29 #pragma once
 30 
 31 #include "ExceptionOr.h"
 32 #include "HTTPHeaderMap.h"
 33 #include <wtf/HashTraits.h>
 34 #include <wtf/Variant.h>
 35 #include <wtf/Vector.h>
 36 
 37 namespace WebCore {
 38 
 39 class FetchHeaders : public RefCounted<FetchHeaders> {
 40 public:
 41     enum class Guard {
 42         None,
 43         Immutable,
 44         Request,
 45         RequestNoCors,
 46         Response
 47     };
 48 
 49     using Init = Variant<Vector<Vector<String>>, Vector<WTF::KeyValuePair<String, String>>>;
 50     static ExceptionOr<Ref<FetchHeaders>> create(Optional<Init>&&);
 51 
 52     static Ref<FetchHeaders> create(Guard guard = Guard::None, HTTPHeaderMap&& headers = { }) { return adoptRef(*new FetchHeaders { guard, WTFMove(headers) }); }
 53     static Ref<FetchHeaders> create(const FetchHeaders& headers) { return adoptRef(*new FetchHeaders { headers }); }
 54 
 55     ExceptionOr<void> append(const String& name, const String& value);
 56     ExceptionOr<void> remove(const String&);
 57     ExceptionOr<String> get(const String&) const;
 58     ExceptionOr<bool> has(const String&) const;
 59     ExceptionOr<void> set(const String& name, const String& value);
 60 
 61     ExceptionOr<void> fill(const Init&);
 62     ExceptionOr<void> fill(const FetchHeaders&);
 63     void filterAndFill(const HTTPHeaderMap&, Guard);
 64 
 65     String fastGet(HTTPHeaderName name) const { return m_headers.get(name); }
 66     bool fastHas(HTTPHeaderName name) const { return m_headers.contains(name); }
 67     void fastSet(HTTPHeaderName name, const String& value) { m_headers.set(name, value); }
 68 
 69     class Iterator {
 70     public:
 71         explicit Iterator(FetchHeaders&);
 72         Optional<WTF::KeyValuePair<String, String>> next();
 73 
 74     private:
 75         Ref<FetchHeaders> m_headers;
 76         size_t m_currentIndex { 0 };
 77         Vector<String> m_keys;
 78     };
 79     Iterator createIterator() { return Iterator { *this }; }
 80 
 81     void setInternalHeaders(HTTPHeaderMap&& headers) { m_headers = WTFMove(headers); }
 82     const HTTPHeaderMap& internalHeaders() const { return m_headers; }
 83 
 84     void setGuard(Guard);
 85     Guard guard() const { return m_guard; }
 86 
 87 private:
 88     FetchHeaders(Guard, HTTPHeaderMap&&);
 89     explicit FetchHeaders(const FetchHeaders&);
 90 
 91     Guard m_guard;
 92     HTTPHeaderMap m_headers;
 93 };
 94 
 95 inline FetchHeaders::FetchHeaders(Guard guard, HTTPHeaderMap&& headers)
 96     : m_guard(guard)
 97     , m_headers(WTFMove(headers))
 98 {
 99 }
100 
101 inline FetchHeaders::FetchHeaders(const FetchHeaders& other)
102     : RefCounted<FetchHeaders>()
103     , m_guard(other.m_guard)
104     , m_headers(other.m_headers)
105 {
106 }
107 
108 inline void FetchHeaders::setGuard(Guard guard)
109 {
110     ASSERT(!m_headers.size());
111     m_guard = guard;
112 }
113 
114 } // namespace WebCore
115 
116 namespace WTF {
117 
118 template<> struct EnumTraits<WebCore::FetchHeaders::Guard> {
119     using values = EnumValues<
120     WebCore::FetchHeaders::Guard,
121     WebCore::FetchHeaders::Guard::None,
122     WebCore::FetchHeaders::Guard::Immutable,
123     WebCore::FetchHeaders::Guard::Request,
124     WebCore::FetchHeaders::Guard::RequestNoCors,
125     WebCore::FetchHeaders::Guard::Response
126     >;
127 };
