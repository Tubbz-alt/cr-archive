 43 ThreadGlobalData::ThreadGlobalData()
 44     : m_cachedResourceRequestInitiators(makeUnique<CachedResourceRequestInitiators>())
 45     , m_eventNames(EventNames::create())
 46     , m_threadTimers(makeUnique<ThreadTimers>())
 47     , m_qualifiedNameCache(makeUnique<QualifiedNameCache>())
 48 #ifndef NDEBUG
 49     , m_isMainThread(isMainThread())
 50 #endif
 51     , m_cachedConverterICU(makeUnique<ICUConverterWrapper>())
 52 {
 53     // This constructor will have been called on the main thread before being called on
 54     // any other thread, and is only called once per thread - this makes this a convenient
 55     // point to call methods that internally perform a one-time initialization that is not
 56     // threadsafe.
 57     Thread::current();
 58 }
 59 
 60 ThreadGlobalData::~ThreadGlobalData() = default;
 61 
 62 void ThreadGlobalData::destroy()
 63 {
 64     m_cachedConverterICU = nullptr;
 65 
 66     m_eventNames = nullptr;
 67     m_threadTimers = nullptr;
 68     m_qualifiedNameCache = nullptr;
 69 }

 71 #if USE(WEB_THREAD)
 72 static ThreadSpecific<RefPtr<ThreadGlobalData>>* staticData { nullptr };
 73 static ThreadGlobalData* sharedMainThreadStaticData { nullptr };
 74 
 75 void ThreadGlobalData::setWebCoreThreadData()
 76 {
 77     ASSERT(isWebThread());
 78     ASSERT(&threadGlobalData() != sharedMainThreadStaticData);
 79 
 80     // Set WebThread's ThreadGlobalData object to be the same as the main UI thread.
 81     **staticData = adoptRef(sharedMainThreadStaticData);
 82 
 83     ASSERT(&threadGlobalData() == sharedMainThreadStaticData);
 84 }
 85 
 86 ThreadGlobalData& threadGlobalData()
 87 {
 88     if (UNLIKELY(!staticData)) {
 89         staticData = new ThreadSpecific<RefPtr<ThreadGlobalData>>;
 90         auto& result = **staticData;
 91         ASSERT(!result);
 92         result = adoptRef(new ThreadGlobalData);
 93         // WebThread and main UI thread need to share the same object. Save it in a static
 94         // here, the WebThread will pick it up in setWebCoreThreadData().
 95         if (pthread_main_np()) {
 96             sharedMainThreadStaticData = result.get();
 97             result->ref();
 98         }
 99         return *result;
100     }
101 
102     auto& result = **staticData;
103     if (!result)
104         result = adoptRef(new ThreadGlobalData);
105     return *result;
106 }
107 
108 #else
109 
110 static ThreadSpecific<ThreadGlobalData>* staticData { nullptr };
111 
112 ThreadGlobalData& threadGlobalData()
113 {
114     if (UNLIKELY(!staticData))
115         staticData = new ThreadSpecific<ThreadGlobalData>;
116     return **staticData;
117 }

121 const MIMETypeRegistryThreadGlobalData& ThreadGlobalData::mimeTypeRegistryThreadGlobalData()
122 {
123     if (UNLIKELY(!m_MIMETypeRegistryThreadGlobalData))
124         m_MIMETypeRegistryThreadGlobalData = MIMETypeRegistry::createMIMETypeRegistryThreadGlobalData();
125     return *m_MIMETypeRegistryThreadGlobalData;
126 }
