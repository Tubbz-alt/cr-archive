42 namespace JSC {
43 
44 class ObjCCallbackFunctionImpl;
45 
46 class ObjCCallbackFunction : public InternalFunction {
47     friend struct APICallbackFunction;
48 public:
49     typedef InternalFunction Base;
50 
51     template<typename CellType, SubspaceAccess mode>
52     static IsoSubspace* subspaceFor(VM& vm)
53     {
54         return vm.objCCallbackFunctionSpace<mode>();
55     }
56 
57     static ObjCCallbackFunction* create(VM&, JSGlobalObject*, const String& name, std::unique_ptr<ObjCCallbackFunctionImpl>);
58     static constexpr bool needsDestruction = true;
59     static void destroy(JSCell*);
60 
61     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
62     {
63         ASSERT(globalObject);
64         return Structure::create(vm, globalObject, prototype, TypeInfo(InternalFunctionType, StructureFlags), info());
65     }
66 
67     DECLARE_EXPORT_INFO;
68 
69     ObjCCallbackFunctionImpl* impl() const { return m_impl.get(); }
70 
71 protected:
72     ObjCCallbackFunction(VM&, Structure*, JSObjectCallAsFunctionCallback, JSObjectCallAsConstructorCallback, std::unique_ptr<ObjCCallbackFunctionImpl>);
73 
74 private:
75     JSObjectCallAsFunctionCallback functionCallback() { return m_functionCallback; }
76     JSObjectCallAsConstructorCallback constructCallback() { return m_constructCallback; }
77 
78     JSObjectCallAsFunctionCallback m_functionCallback;
79     JSObjectCallAsConstructorCallback m_constructCallback;
80     std::unique_ptr<ObjCCallbackFunctionImpl> m_impl;
81 };

83 } // namespace JSC
