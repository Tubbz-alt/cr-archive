 49 bool stringLessThan(StringImpl& a, StringImpl& b)
 50 {
 51     unsigned minLength = std::min(a.length(), b.length());
 52     for (unsigned i = 0; i < minLength; ++i) {
 53         if (a[i] == b[i])
 54             continue;
 55         return a[i] < b[i];
 56     }
 57     return a.length() < b.length();
 58 }

 66 void printInternal(PrintStream& out, OptimizationFixpointState state)
 67 {
 68     switch (state) {
 69     case BeforeFixpoint:
 70         out.print("BeforeFixpoint");
 71         return;
 72     case FixpointNotConverged:
 73         out.print("FixpointNotConverged");
 74         return;
 75     case FixpointConverged:
 76         out.print("FixpointConverged");
 77         return;
 78     }
 79     RELEASE_ASSERT_NOT_REACHED();
 80 }
