 36 inline SVGCursorElement::SVGCursorElement(const QualifiedName& tagName, Document& document)
 37     : SVGElement(tagName, document)

 38     , SVGTests(this)
 39     , SVGURIReference(this)
 40 {
 41     ASSERT(hasTagName(SVGNames::cursorTag));
 42 
 43     static std::once_flag onceFlag;
 44     std::call_once(onceFlag, [] {
 45         PropertyRegistry::registerProperty<SVGNames::xAttr, &SVGCursorElement::m_x>();
 46         PropertyRegistry::registerProperty<SVGNames::yAttr, &SVGCursorElement::m_y>();
 47     });
 48 }
 49 
 50 Ref<SVGCursorElement> SVGCursorElement::create(const QualifiedName& tagName, Document& document)
 51 {
 52     return adoptRef(*new SVGCursorElement(tagName, document));
 53 }
 54 
 55 SVGCursorElement::~SVGCursorElement()
 56 {
 57     for (auto& client : m_clients)
 58         client->cursorElementRemoved(*this);
 59 }
 60 
 61 void SVGCursorElement::parseAttribute(const QualifiedName& name, const AtomString& value)
 62 {
 63     SVGParsingError parseError = NoError;
 64 
 65     if (name == SVGNames::xAttr)
 66         m_x->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
 67     else if (name == SVGNames::yAttr)
 68         m_y->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
 69 
 70     reportAttributeParsingError(parseError, name, value);
 71 
 72     SVGElement::parseAttribute(name, value);
 73     SVGTests::parseAttribute(name, value);

 74     SVGURIReference::parseAttribute(name, value);
 75 }
 76 
 77 void SVGCursorElement::addClient(CSSCursorImageValue& value)
 78 {
 79     m_clients.add(&value);
 80 }
 81 
 82 void SVGCursorElement::removeClient(CSSCursorImageValue& value)
 83 {
 84     m_clients.remove(&value);
 85 }
 86 
 87 void SVGCursorElement::svgAttributeChanged(const QualifiedName& attrName)
 88 {
 89     if (PropertyRegistry::isKnownAttribute(attrName)) {
 90         InstanceInvalidationGuard guard(*this);
 91         for (auto& client : m_clients)
 92             client->cursorElementChanged(*this);
 93         return;
 94     }
 95 
 96     SVGElement::svgAttributeChanged(attrName);
 97 }
 98 
 99 void SVGCursorElement::addSubresourceAttributeURLs(ListHashSet<URL>& urls) const
100 {
101     SVGElement::addSubresourceAttributeURLs(urls);
102 
103     addSubresourceURL(urls, document().completeURL(href()));
104 }
