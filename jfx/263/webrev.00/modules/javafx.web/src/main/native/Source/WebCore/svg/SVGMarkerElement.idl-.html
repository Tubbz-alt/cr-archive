26 interface SVGMarkerElement : SVGElement {
27     // Marker Unit Types
28     const unsigned short SVG_MARKERUNITS_UNKNOWN        = 0;
29     const unsigned short SVG_MARKERUNITS_USERSPACEONUSE = 1;
30     const unsigned short SVG_MARKERUNITS_STROKEWIDTH    = 2;
31 
32     // Marker Orientation Types
33     const unsigned short SVG_MARKER_ORIENT_UNKNOWN      = 0;
34     const unsigned short SVG_MARKER_ORIENT_AUTO         = 1;
35     const unsigned short SVG_MARKER_ORIENT_ANGLE        = 2;
36 
37     readonly attribute SVGAnimatedLength      refX;
38     readonly attribute SVGAnimatedLength      refY;
39     readonly attribute SVGAnimatedEnumeration markerUnits;
40     readonly attribute SVGAnimatedLength      markerWidth;
41     readonly attribute SVGAnimatedLength      markerHeight;
42     readonly attribute SVGAnimatedEnumeration orientType;
43     readonly attribute SVGAnimatedAngle       orientAngle;
44 
45     void setOrientToAuto();
46     void setOrientToAngle(SVGAngle angle);
47 };
48 
49 SVGMarkerElement implements SVGExternalResourcesRequired;
50 SVGMarkerElement implements SVGFitToViewBox;
