 38 MethodOfGettingAValueProfile MethodOfGettingAValueProfile::fromLazyOperand(
 39     CodeBlock* codeBlock, const LazyOperandValueProfileKey& key)
 40 {
 41     MethodOfGettingAValueProfile result;
 42     result.m_kind = LazyOperand;
 43     result.u.lazyOperand.codeBlock = codeBlock;
 44     result.u.lazyOperand.bytecodeOffset = key.bytecodeOffset();
 45     result.u.lazyOperand.operand = key.operand().offset();
 46     return result;
 47 }
 48 
 49 void MethodOfGettingAValueProfile::emitReportValue(CCallHelpers& jit, JSValueRegs regs) const
 50 {
 51     switch (m_kind) {
 52     case None:
 53         return;
 54 
 55     case Ready:
 56         jit.storeValue(regs, u.profile->specFailBucket(0));
 57         return;
 58 
 59     case LazyOperand: {
 60         LazyOperandValueProfileKey key(u.lazyOperand.bytecodeOffset, VirtualRegister(u.lazyOperand.operand));
 61 
 62         ConcurrentJSLocker locker(u.lazyOperand.codeBlock->m_lock);
 63         LazyOperandValueProfile* profile =
 64             u.lazyOperand.codeBlock->lazyOperandValueProfiles(locker).add(locker, key);
 65         jit.storeValue(regs, profile->specFailBucket(0));
 66         return;
 67     }
 68 
 69     case ArithProfileReady: {
 70         u.arithProfile->emitObserveResult(jit, regs, DoNotHaveTagRegisters);
 71         return;
 72     } }
 73 
 74     RELEASE_ASSERT_NOT_REACHED();
 75 }
 76 
 77 void MethodOfGettingAValueProfile::reportValue(JSValue value)
 78 {
 79     switch (m_kind) {
 80     case None:
 81         return;
 82 
 83     case Ready:
 84         *u.profile->specFailBucket(0) = JSValue::encode(value);
 85         return;
 86 
 87     case LazyOperand: {
 88         LazyOperandValueProfileKey key(u.lazyOperand.bytecodeOffset, VirtualRegister(u.lazyOperand.operand));
 89 
 90         ConcurrentJSLocker locker(u.lazyOperand.codeBlock->m_lock);
 91         LazyOperandValueProfile* profile =
 92             u.lazyOperand.codeBlock->lazyOperandValueProfiles(locker).add(locker, key);
 93         *profile->specFailBucket(0) = JSValue::encode(value);
 94         return;
 95     }
 96 
 97     case ArithProfileReady: {
 98         u.arithProfile->observeResult(value);
 99         return;
100     } }
101 
102     RELEASE_ASSERT_NOT_REACHED();
103 }
