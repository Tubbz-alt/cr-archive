 44 NativeExecutable* NativeExecutable::create(VM& vm, Ref<JITCode>&& callThunk, TaggedNativeFunction function, Ref<JITCode>&& constructThunk, TaggedNativeFunction constructor, const String& name)
 45 {
 46     NativeExecutable* executable;
 47     executable = new (NotNull, allocateCell<NativeExecutable>(vm.heap)) NativeExecutable(vm, function, constructor);
 48     executable->finishCreation(vm, WTFMove(callThunk), WTFMove(constructThunk), name);
 49     return executable;
 50 }
 51 
 52 void NativeExecutable::destroy(JSCell* cell)
 53 {
 54     static_cast<NativeExecutable*>(cell)->NativeExecutable::~NativeExecutable();
 55 }

 57 Structure* NativeExecutable::createStructure(VM& vm, JSGlobalObject* globalObject, JSValue proto)
 58 {
 59     return Structure::create(vm, globalObject, proto, TypeInfo(NativeExecutableType, StructureFlags), info());
 60 }
 61 
 62 void NativeExecutable::finishCreation(VM& vm, Ref<JITCode>&& callThunk, Ref<JITCode>&& constructThunk, const String& name)
 63 {
 64     Base::finishCreation(vm);
 65     m_jitCodeForCall = WTFMove(callThunk);
 66     m_jitCodeForConstruct = WTFMove(constructThunk);
 67     m_jitCodeForCallWithArityCheck = m_jitCodeForCall->addressForCall(MustCheckArity);
 68     m_jitCodeForConstructWithArityCheck = m_jitCodeForConstruct->addressForCall(MustCheckArity);
 69     m_name = name;
 70 
 71     assertIsTaggedWith(m_jitCodeForCall->addressForCall(ArityCheckNotRequired).executableAddress(), JSEntryPtrTag);
 72     assertIsTaggedWith(m_jitCodeForConstruct->addressForCall(ArityCheckNotRequired).executableAddress(), JSEntryPtrTag);
 73     assertIsTaggedWith(m_jitCodeForCallWithArityCheck.executableAddress(), JSEntryPtrTag);
 74     assertIsTaggedWith(m_jitCodeForConstructWithArityCheck.executableAddress(), JSEntryPtrTag);
 75 }

 77 NativeExecutable::NativeExecutable(VM& vm, TaggedNativeFunction function, TaggedNativeFunction constructor)
 78     : ExecutableBase(vm, vm.nativeExecutableStructure.get())
 79     , m_function(function)
 80     , m_constructor(constructor)
 81 {
 82 }
 83 
 84 const DOMJIT::Signature* NativeExecutable::signatureFor(CodeSpecializationKind kind) const
 85 {
 86     ASSERT(hasJITCodeFor(kind));
 87     return generatedJITCodeFor(kind)->signature();
 88 }
 89 
 90 Intrinsic NativeExecutable::intrinsic() const
 91 {
 92     return generatedJITCodeFor(CodeForCall)->intrinsic();
 93 }
 94 
 95 CodeBlockHash NativeExecutable::hashFor(CodeSpecializationKind kind) const
 96 {
 97     if (kind == CodeForCall)
 98         return CodeBlockHash(bitwise_cast<uintptr_t>(m_function));
 99 
100     RELEASE_ASSERT(kind == CodeForConstruct);
101     return CodeBlockHash(bitwise_cast<uintptr_t>(m_constructor));
102 }
