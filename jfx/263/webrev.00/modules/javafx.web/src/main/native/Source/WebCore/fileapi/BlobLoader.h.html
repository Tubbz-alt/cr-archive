26 #pragma once
27 
28 #include "Blob.h"
29 #include "Document.h"
30 #include "FileReaderLoader.h"
31 #include "FileReaderLoaderClient.h"
32 #include "SharedBuffer.h"
33 #include <JavaScriptCore/ArrayBuffer.h>
34 #include <wtf/CompletionHandler.h>
35 
36 namespace WebCore {
37 
38 class BlobLoader final : public FileReaderLoaderClient {
39     WTF_MAKE_FAST_ALLOCATED;
40 public:
41     BlobLoader(Document*, Blob&, CompletionHandler<void()>&&);
42     ~BlobLoader();
43 
44     bool isLoading() const { return !!m_loader; }
45     const RefPtr<JSC::ArrayBuffer>& result() const { return m_buffer; }
46     int errorCode() const { return m_errorCode; }
47 
48 private:
49     void didStartLoading() final { }
50     void didReceiveData() final { }
51 
52     void didFinishLoading() final;
53     void didFail(int errorCode) final;
54     void complete();
55 
56     std::unique_ptr<FileReaderLoader> m_loader;
57     RefPtr<JSC::ArrayBuffer> m_buffer;
58     int m_errorCode { 0 };
59     CompletionHandler<void()> m_completionHandler;
60 };
61 
62 inline BlobLoader::BlobLoader(WebCore::Document* document, Blob& blob, CompletionHandler<void()>&& completionHandler)
63     : m_loader(makeUnique<FileReaderLoader>(FileReaderLoader::ReadAsArrayBuffer, this))
64     , m_completionHandler(WTFMove(completionHandler))
65 {
66     m_loader->start(document, blob);
67 }
68 
69 inline BlobLoader::~BlobLoader()
70 {
71     if (m_loader)
72         m_loader->cancel();
73 }
74 
75 inline void BlobLoader::didFinishLoading()
76 {
77     m_buffer = m_loader->arrayBufferResult();
78     complete();
79 }
80 
81 inline void BlobLoader::didFail(int errorCode)
82 {
83     m_errorCode = errorCode;
84     complete();
85 }
86 
87 inline void BlobLoader::complete()
88 {
89     m_loader = nullptr;
90     m_completionHandler();
91 }
92 
93 } // namespace WebCore
