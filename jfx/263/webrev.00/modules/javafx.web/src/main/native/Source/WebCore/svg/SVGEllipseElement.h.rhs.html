24 #include "SVGGeometryElement.h"
25 #include "SVGNames.h"
26 
27 namespace WebCore {
28 
29 class SVGEllipseElement final : public SVGGeometryElement {
30     WTF_MAKE_ISO_ALLOCATED(SVGEllipseElement);
31 public:
32     static Ref<SVGEllipseElement> create(const QualifiedName&, Document&);
33 
34     const SVGLengthValue& cx() const { return m_cx->currentValue(); }
35     const SVGLengthValue& cy() const { return m_cy->currentValue(); }
36     const SVGLengthValue& rx() const { return m_rx->currentValue(); }
37     const SVGLengthValue& ry() const { return m_ry->currentValue(); }
38 
39     SVGAnimatedLength& cxAnimated() { return m_cx; }
40     SVGAnimatedLength& cyAnimated() { return m_cy; }
41     SVGAnimatedLength& rxAnimated() { return m_rx; }
42     SVGAnimatedLength& ryAnimated() { return m_ry; }
43 
44 private:
45     SVGEllipseElement(const QualifiedName&, Document&);
46 
47     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGEllipseElement, SVGGeometryElement>;
48     const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
49 
50     void parseAttribute(const QualifiedName&, const AtomString&) final;
51     void svgAttributeChanged(const QualifiedName&) final;
52 
53     bool isValid() const final { return SVGTests::isValid(); }
54     bool selfHasRelativeLengths() const final { return true; }
55 
56     RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) final;
57 
58     PropertyRegistry m_propertyRegistry { *this };
59     Ref<SVGAnimatedLength> m_cx { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
60     Ref<SVGAnimatedLength> m_cy { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
61     Ref<SVGAnimatedLength> m_rx { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
62     Ref<SVGAnimatedLength> m_ry { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
63 };
64 
65 } // namespace WebCore
