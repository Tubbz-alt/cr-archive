 69 bool ExtensionsGLOpenGLES::isEnabled(const String& name)
 70 {
 71     // Return false immediately if the extension is not supported by the drivers.
 72     bool enabled = ExtensionsGLOpenGLCommon::isEnabled(name);
 73     if (!enabled)
 74         return false;
 75 
 76     // For GL_EXT_robustness, check that the context supports robust access.
 77     if (name == "GL_EXT_robustness") {
 78         GLint robustAccess = GL_FALSE;
 79         m_context->getIntegerv(ExtensionsGL::CONTEXT_ROBUST_ACCESS, &robustAccess);
 80         return robustAccess == GL_TRUE;
 81     }
 82 
 83     return true;
 84 }

130 PlatformGLObject ExtensionsGLOpenGLES::createVertexArrayOES()
131 {
132     m_context->makeContextCurrent();
133     if (m_glGenVertexArraysOES) {
134         GLuint array = 0;
135         m_glGenVertexArraysOES(1, &array);
136         return array;
137     }
138 
139     m_context->synthesizeGLError(GL_INVALID_OPERATION);
140     return 0;
141 }

183 int ExtensionsGLOpenGLES::getGraphicsResetStatusARB()
184 {
185     // FIXME: This does not call getGraphicsResetStatusARB, but instead getGraphicsResetStatusEXT.
186     // The return codes from the two extensions are identical and their purpose is the same, so it
187     // may be best to rename getGraphicsResetStatusARB() to getGraphicsResetStatus().
188     if (m_contextResetStatus != GL_NO_ERROR)
189         return m_contextResetStatus;
190     if (m_glGetGraphicsResetStatusEXT) {
191         m_context->makeContextCurrent();
192         int reasonForReset = m_glGetGraphicsResetStatusEXT();
193         if (reasonForReset != GL_NO_ERROR)
194             m_contextResetStatus = reasonForReset;
195         return reasonForReset;
196     }
197 
198     m_context->synthesizeGLError(GL_INVALID_OPERATION);
199     return false;
200 }

202 void ExtensionsGLOpenGLES::readnPixelsEXT(int x, int y, GCGLsizei width, GCGLsizei height, GCGLenum format, GCGLenum type, GCGLsizei bufSize, void *data)
203 {
204     if (m_glReadnPixelsEXT) {
205         m_context->makeContextCurrent();
206         // FIXME: remove the two glFlush calls when the driver bug is fixed, i.e.,
207         // all previous rendering calls should be done before reading pixels.
208         ::glFlush();
209 
210         // FIXME: If non-BlackBerry platforms use this, they will need to implement
211         // their anti-aliasing code here.
212         m_glReadnPixelsEXT(x, y, width, height, format, type, bufSize, data);
213         return;
214     }
215 
216     m_context->synthesizeGLError(GL_INVALID_OPERATION);
217 }

274 bool ExtensionsGLOpenGLES::supportsExtension(const String& name)
275 {
276     if (m_availableExtensions.contains(name)) {
277         if (!m_supportsOESvertexArrayObject && name == "GL_OES_vertex_array_object") {
278             m_glBindVertexArrayOES = reinterpret_cast<PFNGLBINDVERTEXARRAYOESPROC>(eglGetProcAddress("glBindVertexArrayOES"));
279             m_glGenVertexArraysOES = reinterpret_cast<PFNGLGENVERTEXARRAYSOESPROC>(eglGetProcAddress("glGenVertexArraysOES"));
280             m_glDeleteVertexArraysOES = reinterpret_cast<PFNGLDELETEVERTEXARRAYSOESPROC>(eglGetProcAddress("glDeleteVertexArraysOES"));
281             m_glIsVertexArrayOES = reinterpret_cast<PFNGLISVERTEXARRAYOESPROC>(eglGetProcAddress("glIsVertexArrayOES"));
282             m_supportsOESvertexArrayObject = true;
283         } else if (!m_supportsIMGMultisampledRenderToTexture && name == "GL_IMG_multisampled_render_to_texture") {
284             m_glFramebufferTexture2DMultisampleIMG = reinterpret_cast<PFNGLFRAMEBUFFERTEXTURE2DMULTISAMPLEIMGPROC>(eglGetProcAddress("glFramebufferTexture2DMultisampleIMG"));
285             m_glRenderbufferStorageMultisampleIMG = reinterpret_cast<PFNGLRENDERBUFFERSTORAGEMULTISAMPLEIMGPROC>(eglGetProcAddress("glRenderbufferStorageMultisampleIMG"));
286             m_supportsIMGMultisampledRenderToTexture = true;
287         } else if (!m_glGetGraphicsResetStatusEXT && name == "GL_EXT_robustness") {
288             m_glGetGraphicsResetStatusEXT = reinterpret_cast<PFNGLGETGRAPHICSRESETSTATUSEXTPROC>(eglGetProcAddress("glGetGraphicsResetStatusEXT"));
289             m_glReadnPixelsEXT = reinterpret_cast<PFNGLREADNPIXELSEXTPROC>(eglGetProcAddress("glReadnPixelsEXT"));
290             m_glGetnUniformfvEXT = reinterpret_cast<PFNGLGETNUNIFORMFVEXTPROC>(eglGetProcAddress("glGetnUniformfvEXT"));
291             m_glGetnUniformivEXT = reinterpret_cast<PFNGLGETNUNIFORMIVEXTPROC>(eglGetProcAddress("glGetnUniformivEXT"));
292         } else if (!m_supportsANGLEinstancedArrays && name == "GL_ANGLE_instanced_arrays") {
293             m_glVertexAttribDivisorANGLE = reinterpret_cast<PFNGLVERTEXATTRIBDIVISORANGLEPROC>(eglGetProcAddress("glVertexAttribDivisorANGLE"));
294             m_glDrawArraysInstancedANGLE = reinterpret_cast<PFNGLDRAWARRAYSINSTANCEDANGLEPROC >(eglGetProcAddress("glDrawArraysInstancedANGLE"));
295             m_glDrawElementsInstancedANGLE = reinterpret_cast<PFNGLDRAWELEMENTSINSTANCEDANGLEPROC >(eglGetProcAddress("glDrawElementsInstancedANGLE"));
296             m_supportsANGLEinstancedArrays = true;
297         } else if (name == "GL_EXT_draw_buffers") {
298             // FIXME: implement the support.
299             return false;
300         }
301         return true;
302     }
303 
304     return false;
305 }
