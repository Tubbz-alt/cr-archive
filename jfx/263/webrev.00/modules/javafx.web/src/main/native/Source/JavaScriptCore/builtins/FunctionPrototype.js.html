26 function call(thisArgument)
27 {
28     "use strict";
29 
30     var argumentValues = [];
31     // Start from 1 to ignore thisArgument
32     for (var i = 1; i < arguments.length; i++)
33         @putByValDirect(argumentValues, i-1, arguments[i]);
34 
35     return this.@apply(thisArgument, argumentValues);
36 }
37 
38 function apply(thisValue, argumentValues)
39 {
40     "use strict";
41 
42     return this.@apply(thisValue, argumentValues);
43 }

45 @overriddenName="[Symbol.hasInstance]"
46 function symbolHasInstance(value)
47 {
48     "use strict";
49 
50     if (typeof this !== "function")
51         return false;
52 
53     if (@isBoundFunction(this))
54         return @hasInstanceBoundFunction(this, value);
55 
56     var target = this.prototype;
57     return @instanceOf(value, target);
58 }

60 function bind(thisValue)
61 {
62     "use strict";
63 
64     var target = this;
65     if (typeof target !== "function")
66         @throwTypeError("|this| is not a function inside Function.prototype.bind");
67 
68     var argumentCount = @argumentCount();
69     var boundArgs = null;
70     var numBoundArgs = 0;
71     if (argumentCount > 1) {
72         numBoundArgs = argumentCount - 1;
73         boundArgs = @createArgumentsButterfly();
74     }
75 
76     var length = 0;
77     if (@hasOwnLengthProperty(target)) {
78         var lengthValue = target.length;
79         if (typeof lengthValue === "number") {
80             lengthValue = lengthValue | 0;
81             // Note that we only care about positive lengthValues, however, this comparision
82             // against numBoundArgs suffices to prove we're not a negative number.
83             if (lengthValue > numBoundArgs)
84                 length = lengthValue - numBoundArgs;
85         }
86     }
87 
88     var name = target.name;
89     if (typeof name !== "string")
90         name = "";
91 
92     return @makeBoundFunction(target, thisValue, boundArgs, length, name);
93 }
