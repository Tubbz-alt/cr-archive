43 class CDMInstanceSession;

46 // Handle to a "real" CDM, not the JavaScript facade. This can be used
47 // from background threads (i.e. decryptors).
48 class ProxyCDM : public ThreadSafeRefCounted<ProxyCDM> {
49 public:
50     virtual ~ProxyCDM() = default;
51 };
52 
53 struct CDMKeySystemConfiguration;
54 
55 // JavaScript's handle to a CDMInstance, must be used from the
56 // main-thread only!
57 class CDMInstance : public RefCounted<CDMInstance> {
58 public:
59     virtual ~CDMInstance() = default;
60 
61     enum class ImplementationType {
62         Mock,
63         ClearKey,
64         FairPlayStreaming,
65     };
66     virtual ImplementationType implementationType() const = 0;
67 
68     enum SuccessValue {
69         Failed,
70         Succeeded,
71     };
72 
73     virtual SuccessValue initializeWithConfiguration(const CDMKeySystemConfiguration&) = 0;
74     virtual SuccessValue setDistinctiveIdentifiersAllowed(bool) = 0;
75     virtual SuccessValue setPersistentStateAllowed(bool) = 0;
76     virtual SuccessValue setServerCertificate(Ref<SharedBuffer>&&) = 0;
77     virtual SuccessValue setStorageDirectory(const String&) = 0;
78     virtual const String& keySystem() const = 0;
79     virtual RefPtr<CDMInstanceSession> createSession() = 0;
80     virtual RefPtr<ProxyCDM> proxyCDM() const = 0;
81 
82     enum class HDCPStatus {
83         Unknown,
84         Valid,
85         OutputRestricted,
86         OutputDownscaled,
87     };
88     virtual SuccessValue setHDCPStatus(HDCPStatus) { return Failed; }
89 };
90 
91 } // namespace WebCore
