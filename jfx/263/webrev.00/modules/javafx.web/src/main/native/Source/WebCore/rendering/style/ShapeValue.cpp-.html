34 bool ShapeValue::isImageValid() const
35 {
36     if (!m_image)
37         return false;
38     if (m_image->isCachedImage()) {
39         auto* cachedImage = m_image->cachedImage();
40         return cachedImage && cachedImage->hasImage();
41     }
42     return m_image->isGeneratedImage();
43 }
44 
45 bool ShapeValue::operator==(const ShapeValue& other) const
46 {
47     return m_type == other.m_type
48         && m_cssBox == other.m_cssBox
49         && arePointingToEqualData(m_shape, other.m_shape)
50         && arePointingToEqualData(m_image, other.m_image);
51 }
