 33 RealtimeVideoSource::RealtimeVideoSource(Ref<RealtimeVideoCaptureSource>&& source)
 34     : RealtimeVideoCaptureSource(String { source->name() }, String { source->persistentID() }, String { source->deviceIDHashSalt() })
 35     , m_source(WTFMove(source))
 36 {
 37     m_source->addObserver(*this);
 38     m_currentSettings = m_source->settings();
 39     setSize(m_source->size());
 40 }
 41 
 42 RealtimeVideoSource::~RealtimeVideoSource()
 43 {
 44     m_source->removeObserver(*this);
 45 }

 52 void RealtimeVideoSource::stopProducingData()

 57 bool RealtimeVideoSource::supportsSizeAndFrameRate(Optional<int> width, Optional<int> height, Optional<double> frameRate)
 58 {
 59     return m_source->supportsSizeAndFrameRate(width, height, frameRate);
 60 }
 61 
 62 void RealtimeVideoSource::setSizeAndFrameRate(Optional<int> width, Optional<int> height, Optional<double> frameRate)
 63 {
 64     if (!width && !height) {
 65         width = size().width();
 66         height = size().height();
 67     }
 68 
 69     m_source->clientUpdatedSizeAndFrameRate(width, height, frameRate);
 70     auto sourceSize = m_source->size();
 71     ASSERT(sourceSize.height());
 72     ASSERT(sourceSize.width());
 73 
 74     if (!width)
 75         width = sourceSize.width() * height.value() / sourceSize.height();
 76     m_currentSettings.setWidth(*width);
 77 
 78     if (!height)
 79         height = sourceSize.height() * width.value() / sourceSize.width();
 80     m_currentSettings.setHeight(*height);
 81 
 82     if (frameRate)
 83         m_currentSettings.setFrameRate(static_cast<float>(*frameRate));
 84 
 85     RealtimeMediaSource::setSizeAndFrameRate(width, height, frameRate);
 86 }

 88 void RealtimeVideoSource::sourceMutedChanged()
 89 {
 90     notifyMutedChange(m_source->muted());
 91 }
 92 
 93 void RealtimeVideoSource::sourceSettingsChanged()
 94 {
 95     auto rotation = m_source->sampleRotation();
 96     auto size = this->size();
 97     if (size.isEmpty())
 98         size = m_source->size();
 99     if (rotation == MediaSample::VideoRotation::Left || rotation == MediaSample::VideoRotation::Right)
100         size = size.transposedSize();
101 
102     m_currentSettings.setWidth(size.width());
103     m_currentSettings.setHeight(size.height());
104 
105     forEachObserver([&](auto& observer) {
106         observer.sourceSettingsChanged();
107     });
108 }
109 
110 bool RealtimeVideoSource::preventSourceFromStopping()
111 {
112     if (!isProducingData())
113         return false;
114 
115     bool hasObserverPreventingStopping = false;
116     forEachObserver([&](auto& observer) {
117         if (observer.preventSourceFromStopping())
118             hasObserverPreventingStopping = true;
119     });
120     return hasObserverPreventingStopping;
121 }

123 void RealtimeVideoSource::requestToEnd(RealtimeMediaSource::Observer&)
124 {
125     m_source->requestToEnd(*this);
126 }
127 
128 void RealtimeVideoSource::stopBeingObserved()
129 {
130     m_source->requestToEnd(*this);
131 }
132 
133 void RealtimeVideoSource::sourceStopped()
134 {
135     if (m_source->captureDidFail()) {
136         captureFailed();
137         return;
138     }
139     stop();
140     forEachObserver([](auto& observer) {
141         observer.sourceStopped();
142     });
143 }

145 void RealtimeVideoSource::videoSampleAvailable(MediaSample& sample)
146 {
147     if (!isProducingData())
148         return;
149 
150     if (auto mediaSample = adaptVideoSample(sample))
151         RealtimeMediaSource::videoSampleAvailable(*mediaSample);
152 }

154 Ref<RealtimeMediaSource> RealtimeVideoSource::clone()
155 {
156     auto source = create(m_source.copyRef());
157     source->m_currentSettings = m_currentSettings;
158     source->setSize(size());
159     return source;
160 }
