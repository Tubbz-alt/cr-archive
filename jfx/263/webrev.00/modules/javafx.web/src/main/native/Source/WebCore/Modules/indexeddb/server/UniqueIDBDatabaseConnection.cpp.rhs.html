 40 Ref<UniqueIDBDatabaseConnection> UniqueIDBDatabaseConnection::create(UniqueIDBDatabase& database, ServerOpenDBRequest& request)

 45 UniqueIDBDatabaseConnection::UniqueIDBDatabaseConnection(UniqueIDBDatabase& database, ServerOpenDBRequest& request)
 46     : m_database(&database)

 47     , m_connectionToClient(request.connection())
 48     , m_openRequestIdentifier(request.requestData().requestIdentifier())
 49 {
 50     server()->registerDatabaseConnection(*this);
 51     m_connectionToClient->registerDatabaseConnection(*this);
 52 }
 53 
 54 UniqueIDBDatabaseConnection::~UniqueIDBDatabaseConnection()
 55 {
 56     ASSERT(m_transactionMap.isEmpty());
 57 
 58     server()->unregisterDatabaseConnection(*this);
 59     m_connectionToClient->unregisterDatabaseConnection(*this);
 60 }

 62 bool UniqueIDBDatabaseConnection::hasNonFinishedTransactions() const
 63 {
 64     return !m_transactionMap.isEmpty();
 65 }

 67 void UniqueIDBDatabaseConnection::abortTransactionWithoutCallback(UniqueIDBDatabaseTransaction& transaction)
 68 {
 69     ASSERT(m_transactionMap.contains(transaction.info().identifier()));
 70 
 71     const auto& transactionIdentifier = transaction.info().identifier();
 72     RefPtr<UniqueIDBDatabaseConnection> protectedThis(this);
 73 
 74     m_database->abortTransaction(transaction, [this, protectedThis, transactionIdentifier](const IDBError&) {




 75         ASSERT(m_transactionMap.contains(transactionIdentifier));
 76         m_transactionMap.remove(transactionIdentifier);
 77     });
 78 }

114 UniqueIDBDatabaseTransaction& UniqueIDBDatabaseConnection::createVersionChangeTransaction(uint64_t newVersion)
115 {
116     LOG(IndexedDB, "UniqueIDBDatabaseConnection::createVersionChangeTransaction - %s - %" PRIu64, m_openRequestIdentifier.loggingString().utf8().data(), identifier());
117     ASSERT(!m_closePending);
118 
119     IDBTransactionInfo info = IDBTransactionInfo::versionChange(m_connectionToClient, m_database->info(), newVersion);
120 
121     Ref<UniqueIDBDatabaseTransaction> transaction = UniqueIDBDatabaseTransaction::create(*this, info);
122     m_transactionMap.set(transaction->info().identifier(), &transaction.get());
123 
124     return transaction.get();
125 }

127 void UniqueIDBDatabaseConnection::establishTransaction(const IDBTransactionInfo& info)
128 {
129     LOG(IndexedDB, "UniqueIDBDatabaseConnection::establishTransaction - %s - %" PRIu64, m_openRequestIdentifier.loggingString().utf8().data(), identifier());
130 
131     ASSERT(info.mode() != IDBTransactionMode::Versionchange);
132 
133     // No transactions should ever come from the client after the client has already told us
134     // the connection is closing.
135     ASSERT(!m_closePending);
136 



137     Ref<UniqueIDBDatabaseTransaction> transaction = UniqueIDBDatabaseTransaction::create(*this, info);
138     m_transactionMap.set(transaction->info().identifier(), &transaction.get());
139 
140     m_database->enqueueTransaction(WTFMove(transaction));
141 }

143 void UniqueIDBDatabaseConnection::didAbortTransaction(UniqueIDBDatabaseTransaction& transaction, const IDBError& error)
144 {
145     LOG(IndexedDB, "UniqueIDBDatabaseConnection::didAbortTransaction - %s - %" PRIu64, m_openRequestIdentifier.loggingString().utf8().data(), identifier());
146 
147     auto transactionIdentifier = transaction.info().identifier();
148     auto takenTransaction = m_transactionMap.take(transactionIdentifier);
149     ASSERT(takenTransaction);
150 
151     m_connectionToClient->didAbortTransaction(transactionIdentifier, error);


152 }
153 
154 void UniqueIDBDatabaseConnection::didCommitTransaction(UniqueIDBDatabaseTransaction& transaction, const IDBError& error)
155 {
156     LOG(IndexedDB, "UniqueIDBDatabaseConnection::didCommitTransaction - %s - %" PRIu64, m_openRequestIdentifier.loggingString().utf8().data(), identifier());
157 
158     auto transactionIdentifier = transaction.info().identifier();
159 
160     ASSERT(m_transactionMap.contains(transactionIdentifier) || !error.isNull());
161     m_transactionMap.remove(transactionIdentifier);
162 
163     m_connectionToClient->didCommitTransaction(transactionIdentifier, error);
164 }
