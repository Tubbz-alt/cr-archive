26 #pragma once
27 
28 #include "DeferredTrigger.h"
29 
30 namespace bmalloc {
31 
32 template<IsoPageTrigger trigger>
33 template<typename Config>
34 void DeferredTrigger<trigger>::didBecome(IsoPage<Config>& page)
35 {
36     if (page.isInUseForAllocation())
37         m_hasBeenDeferred = true;
38     else
39         page.directory().didBecome(&page, trigger);
40 }
41 
42 template<IsoPageTrigger trigger>
43 template<typename Config>
44 void DeferredTrigger<trigger>::handleDeferral(IsoPage<Config>& page)
45 {
46     RELEASE_BASSERT(!page.isInUseForAllocation());
47 
48     if (m_hasBeenDeferred) {
49         page.directory().didBecome(&page, trigger);
50         m_hasBeenDeferred = false;
51     }
52 }
53 
54 } // namespace bmalloc
