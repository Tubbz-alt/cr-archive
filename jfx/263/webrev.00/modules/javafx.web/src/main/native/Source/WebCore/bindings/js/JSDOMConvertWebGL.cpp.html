 70 // FIXME: This should use the IDLUnion JSConverter.
 71 JSValue convertToJSValue(JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, const WebGLAny& any)
 72 {
 73     return WTF::switchOn(any,
 74         [] (std::nullptr_t) {
 75             return jsNull();
 76         },
 77         [] (bool value) {
 78             return jsBoolean(value);
 79         },
 80         [] (int value) {
 81             return jsNumber(value);
 82         },
 83         [] (unsigned value) {
 84             return jsNumber(value);
 85         },
 86         [] (long long value) {
 87             return jsNumber(value);
 88         },
 89         [] (float value) {
 90             return jsNumber(value);
 91         },
 92         [&] (const String& value) {
 93             return jsStringWithCache(&lexicalGlobalObject, value);
 94         },
 95         [&] (const Vector<bool>& values) {
 96             MarkedArgumentBuffer list;
 97             for (auto& value : values)
 98                 list.append(jsBoolean(value));
 99             RELEASE_ASSERT(!list.hasOverflowed());
100             return constructArray(&globalObject, static_cast<JSC::ArrayAllocationProfile*>(nullptr), list);
101         },
102         [&] (const Vector<int>& values) {
103             MarkedArgumentBuffer list;
104             for (auto& value : values)
105                 list.append(jsNumber(value));
106             RELEASE_ASSERT(!list.hasOverflowed());
107             return constructArray(&globalObject, static_cast<JSC::ArrayAllocationProfile*>(nullptr), list);
108         },
109         [&] (const RefPtr<Float32Array>& array) {
110             return toJS(&lexicalGlobalObject, &globalObject, array.get());
111         },
112         [&] (const RefPtr<Int32Array>& array) {
113             return toJS(&lexicalGlobalObject, &globalObject, array.get());
114         },
115         [&] (const RefPtr<Uint8Array>& array) {
116             return toJS(&lexicalGlobalObject, &globalObject, array.get());
117         },
118         [&] (const RefPtr<Uint32Array>& array) {
119             return toJS(&lexicalGlobalObject, &globalObject, array.get());
120         },
121         [&] (const RefPtr<WebGLBuffer>& buffer) {
122             return toJS(&lexicalGlobalObject, &globalObject, buffer.get());
123         },
124         [&] (const RefPtr<WebGLFramebuffer>& buffer) {
125             return toJS(&lexicalGlobalObject, &globalObject, buffer.get());
126         },
127         [&] (const RefPtr<WebGLProgram>& program) {
128             return toJS(&lexicalGlobalObject, &globalObject, program.get());
129         },
130         [&] (const RefPtr<WebGLRenderbuffer>& buffer) {
131             return toJS(&lexicalGlobalObject, &globalObject, buffer.get());
132         },
133         [&] (const RefPtr<WebGLTexture>& texture) {
134             return toJS(&lexicalGlobalObject, &globalObject, texture.get());
135         },
136         [&] (const RefPtr<WebGLVertexArrayObjectOES>& array) {
137             return toJS(&lexicalGlobalObject, &globalObject, array.get());
138         }
139 #if ENABLE(WEBGL2)
140         ,
141         [&] (const RefPtr<WebGLSampler>& sampler) {
142             return toJS(&lexicalGlobalObject, &globalObject, sampler.get());
143         },
144         [&] (const RefPtr<WebGLTransformFeedback>& transformFeedback) {
145             return toJS(&lexicalGlobalObject, &globalObject, transformFeedback.get());
146         },
147         [&] (const RefPtr<WebGLVertexArrayObject>& array) {
148             return toJS(&lexicalGlobalObject, &globalObject, array.get());
149         }
150 #endif
151     );
152 }

154 JSValue convertToJSValue(JSGlobalObject& lexicalGlobalObject, JSDOMGlobalObject& globalObject, WebGLExtension& extension)
155 {
156     switch (extension.getName()) {
157     case WebGLExtension::WebGLLoseContextName:
158         return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLLoseContext&>(extension));
159     case WebGLExtension::EXTShaderTextureLODName:
160         return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTShaderTextureLOD&>(extension));
161     case WebGLExtension::EXTTextureFilterAnisotropicName:
162         return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTTextureFilterAnisotropic&>(extension));
163     case WebGLExtension::EXTsRGBName:
164         return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTsRGB&>(extension));
165     case WebGLExtension::EXTFragDepthName:
166         return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTFragDepth&>(extension));
167     case WebGLExtension::EXTBlendMinMaxName:
168         return toJS(&lexicalGlobalObject, &globalObject, static_cast<EXTBlendMinMax&>(extension));
169     case WebGLExtension::OESStandardDerivativesName:
170         return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESStandardDerivatives&>(extension));
171     case WebGLExtension::OESTextureFloatName:
172         return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESTextureFloat&>(extension));
173     case WebGLExtension::OESTextureFloatLinearName:
174         return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESTextureFloatLinear&>(extension));
175     case WebGLExtension::OESTextureHalfFloatName:
176         return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESTextureHalfFloat&>(extension));
177     case WebGLExtension::OESTextureHalfFloatLinearName:
178         return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESTextureHalfFloatLinear&>(extension));
179     case WebGLExtension::OESVertexArrayObjectName:
180         return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESVertexArrayObject&>(extension));
181     case WebGLExtension::OESElementIndexUintName:
182         return toJS(&lexicalGlobalObject, &globalObject, static_cast<OESElementIndexUint&>(extension));
183     case WebGLExtension::WebGLDebugRendererInfoName:
184         return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLDebugRendererInfo&>(extension));
185     case WebGLExtension::WebGLDebugShadersName:
186         return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLDebugShaders&>(extension));
187     case WebGLExtension::WebGLCompressedTextureATCName:
188         return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLCompressedTextureATC&>(extension));
189     case WebGLExtension::WebGLCompressedTextureETCName:
190         return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLCompressedTextureETC&>(extension));
191     case WebGLExtension::WebGLCompressedTextureETC1Name:
192         return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLCompressedTextureETC1&>(extension));
193     case WebGLExtension::WebGLCompressedTexturePVRTCName:
194         return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLCompressedTexturePVRTC&>(extension));
195     case WebGLExtension::WebGLCompressedTextureS3TCName:
196         return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLCompressedTextureS3TC&>(extension));
197     case WebGLExtension::WebGLCompressedTextureASTCName:
198         return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLCompressedTextureASTC&>(extension));
199     case WebGLExtension::WebGLDepthTextureName:
200         return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLDepthTexture&>(extension));
201     case WebGLExtension::WebGLDrawBuffersName:
202         return toJS(&lexicalGlobalObject, &globalObject, static_cast<WebGLDrawBuffers&>(extension));
203     case WebGLExtension::ANGLEInstancedArraysName:
204         return toJS(&lexicalGlobalObject, &globalObject, static_cast<ANGLEInstancedArrays&>(extension));
205     }
206     ASSERT_NOT_REACHED();
207     return jsNull();
208 }
