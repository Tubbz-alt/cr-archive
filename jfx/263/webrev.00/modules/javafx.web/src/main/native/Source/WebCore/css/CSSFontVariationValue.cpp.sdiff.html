43 String CSSFontVariationValue::customCSSText() const
44 {
45     StringBuilder builder;
46     builder.append('"');
47     for (char c : m_tag)
48         builder.append(c);
49     builder.appendLiteral("\" ");
50     builder.appendFixedPrecisionNumber(m_value);
51     return builder.toString();
