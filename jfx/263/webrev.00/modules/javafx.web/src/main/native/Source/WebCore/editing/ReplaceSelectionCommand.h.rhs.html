 26 #pragma once
 27 
 28 #include "CompositeEditCommand.h"
 29 #include "NodeTraversal.h"
 30 
 31 namespace WebCore {
 32 
 33 class DocumentFragment;
 34 class Range;
 35 class ReplacementFragment;
 36 
 37 class ReplaceSelectionCommand : public CompositeEditCommand {
 38 public:
 39     enum CommandOption {
 40         SelectReplacement = 1 << 0,
 41         SmartReplace = 1 << 1,
 42         MatchStyle = 1 << 2,
 43         PreventNesting = 1 << 3,
 44         MovingParagraph = 1 << 4,
 45         SanitizeFragment = 1 << 5,
 46         IgnoreMailBlockquote = 1 << 6,
 47     };
 48 
 49     static Ref<ReplaceSelectionCommand> create(Document& document, RefPtr<DocumentFragment>&& fragment, OptionSet<CommandOption> options, EditAction editingAction = EditAction::Insert)
 50     {
 51         return adoptRef(*new ReplaceSelectionCommand(document, WTFMove(fragment), options, editingAction));
 52     }
 53 
 54     VisibleSelection visibleSelectionForInsertedText() const { return m_visibleSelectionForInsertedText; }
 55 
 56     RefPtr<Range> insertedContentRange() const;
 57 
 58 private:
 59     ReplaceSelectionCommand(Document&, RefPtr<DocumentFragment>&&, OptionSet<CommandOption>, EditAction);
 60 
 61     String inputEventData() const final;
 62     RefPtr<DataTransfer> inputEventDataTransfer() const final;
 63     bool willApplyCommand() final;
 64     void doApply() override;
 65 
 66     class InsertedNodes {
 67     public:
 68         void respondToNodeInsertion(Node*);
 69         void willRemoveNodePreservingChildren(Node*);
 70         void willRemoveNode(Node*);
 71         void didReplaceNode(Node*, Node* newNode);
 72 
 73         bool isEmpty() { return !m_firstNodeInserted; }
 74         Node* firstNodeInserted() const { return m_firstNodeInserted.get(); }
 75         Node* lastLeafInserted() const
 76         {
 77             ASSERT(m_lastNodeInserted);
 78             return m_lastNodeInserted->lastDescendant();
 79         }
 80         Node* pastLastLeaf() const
 81         {
 82             ASSERT(m_lastNodeInserted);
 83             return NodeTraversal::next(*lastLeafInserted());
 84         }
 85 
 86     private:
 87         RefPtr<Node> m_firstNodeInserted;
 88         RefPtr<Node> m_lastNodeInserted;
 89     };
 90 
 91     Node* insertAsListItems(HTMLElement& listElement, Node* insertionNode, const Position&, InsertedNodes&);
 92 
 93     void updateNodesInserted(Node*);
 94     bool shouldRemoveEndBR(Node*, const VisiblePosition&);
 95 
 96     bool shouldMergeStart(bool, bool, bool);
 97     bool shouldMergeEnd(bool selectionEndWasEndOfParagraph);
 98     bool shouldMerge(const VisiblePosition&, const VisiblePosition&);
 99 
100     void mergeEndIfNeeded();
101 
102     void removeUnrenderedTextNodesAtEnds(InsertedNodes&);
103 
104     void removeRedundantStylesAndKeepStyleSpanInline(InsertedNodes&);
105     void inverseTransformColor(InsertedNodes&);
106     void makeInsertedContentRoundTrippableWithHTMLTreeBuilder(InsertedNodes&);
107     void moveNodeOutOfAncestor(Node&, Node& ancestor, InsertedNodes&);
108     void handleStyleSpans(InsertedNodes&);
109     void handlePasteAsQuotationNode();
110 
111     VisiblePosition positionAtStartOfInsertedContent() const;
112     VisiblePosition positionAtEndOfInsertedContent() const;
113 
114     bool shouldPerformSmartReplace() const;
115     bool shouldPerformSmartParagraphReplace() const;
116     void addSpacesForSmartReplace();
117     void addNewLinesForSmartReplace();
118     void completeHTMLReplacement(const Position& lastPositionToSelect);
119     void mergeTextNodesAroundPosition(Position&, Position& positionOnlyToBeUpdated);
120 
121     ReplacementFragment* ensureReplacementFragment();
122     bool performTrivialReplace(const ReplacementFragment&);
123 
124     VisibleSelection m_visibleSelectionForInsertedText;
125     Position m_startOfInsertedContent;
126     Position m_endOfInsertedContent;
127     RefPtr<EditingStyle> m_insertionStyle;
128     bool m_selectReplacement;
129     bool m_smartReplace;
130     bool m_matchStyle;
131     RefPtr<DocumentFragment> m_documentFragment;
132     std::unique_ptr<ReplacementFragment> m_replacementFragment;
133     String m_documentFragmentHTMLMarkup;
134     String m_documentFragmentPlainText;
135     bool m_preventNesting;
136     bool m_movingParagraph;
137     bool m_sanitizeFragment;
138     bool m_shouldMergeEnd;
139     bool m_ignoreMailBlockquote;
140 };
