 82 template<class Encoder>
 83 void ImagePaintingOptions::encode(Encoder& encoder) const
 84 {
 85     encoder << m_compositeOperator;
 86     encoder << m_blendMode;
 87     encoder << m_decodingMode;
 88     encoder << ImageOrientation::Orientation(m_orientation);
 89     encoder << m_interpolationQuality;
 90 }
 91 
 92 template<class Decoder>
 93 Optional<ImagePaintingOptions> ImagePaintingOptions::decode(Decoder& decoder)
 94 {
 95     Optional<CompositeOperator> compositeOperator;
 96     decoder >> compositeOperator;
 97     if (!compositeOperator)
 98         return WTF::nullopt;
 99 
100     Optional<BlendMode> blendMode;
101     decoder >> blendMode;
102     if (!blendMode)
103         return WTF::nullopt;
104 
105     Optional<DecodingMode> decodingMode;
106     decoder >> decodingMode;
107     if (!decodingMode)
108         return WTF::nullopt;
109 
110     Optional<ImageOrientation::Orientation> orientation;
111     decoder >> orientation;
112     if (!orientation)
113         return WTF::nullopt;
114 
115     Optional<InterpolationQuality> interpolationQuality;
116     decoder >> interpolationQuality;
117     if (!interpolationQuality)
118         return WTF::nullopt;
119 
120     return ImagePaintingOptions { *compositeOperator, *blendMode, *decodingMode, *orientation, *interpolationQuality };
121 }
