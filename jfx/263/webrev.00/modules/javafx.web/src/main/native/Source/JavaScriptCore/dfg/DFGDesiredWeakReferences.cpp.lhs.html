37 DesiredWeakReferences::DesiredWeakReferences()

42 DesiredWeakReferences::DesiredWeakReferences(CodeBlock* codeBlock)

49 }

51 void DesiredWeakReferences::addLazily(JSCell* cell)
52 {
53     if (cell)
54         m_references.add(cell);
55 }
56 
57 void DesiredWeakReferences::addLazily(JSValue value)
58 {
59     if (value.isCell())
60         addLazily(value.asCell());
61 }
62 
63 bool DesiredWeakReferences::contains(JSCell* cell)
64 {
65     return m_references.contains(cell);
66 }
67 
68 void DesiredWeakReferences::reallyAdd(VM& vm, CommonData* common)
69 {

70     for (JSCell* target : m_references) {
71         if (Structure* structure = jsDynamicCast<Structure*>(vm, target)) {
72             common->weakStructureReferences.append(
73                 WriteBarrier<Structure>(vm, m_codeBlock, structure));

74         } else {
75             // There are weird relationships in how optimized CodeBlocks
76             // point to other CodeBlocks. We don't want to have them be
77             // part of the weak pointer set. For example, an optimized CodeBlock
78             // having a weak pointer to itself will cause it to get collected.
79             RELEASE_ASSERT(!jsDynamicCast<CodeBlock*>(vm, target));
80 

81             common->weakReferences.append(
82                 WriteBarrier<JSCell>(vm, m_codeBlock, target));
83         }
84     }
85 }
86 
87 void DesiredWeakReferences::visitChildren(SlotVisitor& visitor)
88 {
89     for (JSCell* target : m_references)
90         visitor.appendUnbarriered(target);
91 }
