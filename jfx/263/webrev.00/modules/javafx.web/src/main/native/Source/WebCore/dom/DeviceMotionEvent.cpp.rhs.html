 34 WTF_MAKE_ISO_ALLOCATED_IMPL(DeviceMotionEvent);

 36 DeviceMotionEvent::~DeviceMotionEvent() = default;
 37 
 38 DeviceMotionEvent::DeviceMotionEvent()
 39     : m_deviceMotionData(DeviceMotionData::create())
 40 {
 41 }
 42 
 43 DeviceMotionEvent::DeviceMotionEvent(const AtomString& eventType, DeviceMotionData* deviceMotionData)
 44     : Event(eventType, CanBubble::No, IsCancelable::No)
 45     , m_deviceMotionData(deviceMotionData)
 46 {
 47 }
 48 
 49 static Optional<DeviceMotionEvent::Acceleration> convert(const DeviceMotionData::Acceleration* acceleration)
 50 {
 51     if (!acceleration)
 52         return WTF::nullopt;
 53 
 54     return DeviceMotionEvent::Acceleration { acceleration->x(), acceleration->y(), acceleration->z() };
 55 }
 56 
 57 static Optional<DeviceMotionEvent::RotationRate> convert(const DeviceMotionData::RotationRate* rotationRate)
 58 {
 59     if (!rotationRate)
 60         return WTF::nullopt;
 61 
 62     return DeviceMotionEvent::RotationRate { rotationRate->alpha(), rotationRate->beta(), rotationRate->gamma() };
 63 }
 64 
 65 static RefPtr<DeviceMotionData::Acceleration> convert(Optional<DeviceMotionEvent::Acceleration>&& acceleration)
 66 {
 67     if (!acceleration)
 68         return nullptr;
 69 
 70     if (!acceleration->x && !acceleration->y && !acceleration->z)
 71         return nullptr;
 72 
 73     return DeviceMotionData::Acceleration::create(acceleration->x, acceleration->y, acceleration->z);
 74 }
 75 
 76 static RefPtr<DeviceMotionData::RotationRate> convert(Optional<DeviceMotionEvent::RotationRate>&& rotationRate)
 77 {
 78     if (!rotationRate)
 79         return nullptr;
 80 
 81     if (!rotationRate->alpha && !rotationRate->beta && !rotationRate->gamma)
 82         return nullptr;
 83 
 84     return DeviceMotionData::RotationRate::create(rotationRate->alpha, rotationRate->beta, rotationRate->gamma);
 85 }
 86 
 87 Optional<DeviceMotionEvent::Acceleration> DeviceMotionEvent::acceleration() const
 88 {
 89     return convert(m_deviceMotionData->acceleration());
 90 }
 91 
 92 Optional<DeviceMotionEvent::Acceleration> DeviceMotionEvent::accelerationIncludingGravity() const
 93 {
 94     return convert(m_deviceMotionData->accelerationIncludingGravity());
 95 }
 96 
 97 Optional<DeviceMotionEvent::RotationRate> DeviceMotionEvent::rotationRate() const
 98 {
 99     return convert(m_deviceMotionData->rotationRate());
100 }
101 
102 Optional<double> DeviceMotionEvent::interval() const
103 {
104     return m_deviceMotionData->interval();
105 }
106 
107 void DeviceMotionEvent::initDeviceMotionEvent(const AtomString& type, bool bubbles, bool cancelable, Optional<DeviceMotionEvent::Acceleration>&& acceleration, Optional<DeviceMotionEvent::Acceleration>&& accelerationIncludingGravity, Optional<DeviceMotionEvent::RotationRate>&& rotationRate, Optional<double> interval)
108 {
109     if (isBeingDispatched())
110         return;
111 
112     initEvent(type, bubbles, cancelable);
113     m_deviceMotionData = DeviceMotionData::create(convert(WTFMove(acceleration)), convert(WTFMove(accelerationIncludingGravity)), convert(WTFMove(rotationRate)), interval);
114 }
115 
116 EventInterface DeviceMotionEvent::eventInterface() const
117 {
118 #if ENABLE(DEVICE_ORIENTATION)
119     return DeviceMotionEventInterfaceType;
120 #else
121     // FIXME: ENABLE(DEVICE_ORIENTATION) seems to be in a strange state where
122     // it is half-guarded by #ifdefs. DeviceMotionEvent.idl is guarded
123     // but DeviceMotionEvent.cpp itself is required by ungarded code.
124     return EventInterfaceType;
125 #endif
126 }
