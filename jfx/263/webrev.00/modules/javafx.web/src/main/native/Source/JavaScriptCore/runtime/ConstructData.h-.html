31 #include "CallData.h"
32 #include "JSCJSValue.h"
33 #include "NativeFunction.h"
34 
35 namespace JSC {
36 
37 class ArgList;
38 class ExecState;
39 class FunctionExecutable;
40 class JSObject;
41 class JSScope;
42 
43 enum class ConstructType : unsigned {
44     None,
45     Host,
46     JS
47 };
48 
49 struct ConstructData {
50     union {
51         struct {
52             TaggedNativeFunction function;
53         } native;
54         struct {
55             FunctionExecutable* functionExecutable;
56             JSScope* scope;
57         } js;
58     };
59 };
60 
61 // Convenience wrapper so you don't need to deal with CallData and CallType unless you are going to use them.
62 JS_EXPORT_PRIVATE JSObject* construct(ExecState*, JSValue functionObject, const ArgList&, const char* errorMessage);
63 JS_EXPORT_PRIVATE JSObject* construct(ExecState*, JSValue constructor, ConstructType, const ConstructData&, const ArgList&, JSValue newTarget);
64 
65 ALWAYS_INLINE JSObject* construct(ExecState* exec, JSValue constructorObject, ConstructType constructType, const ConstructData& constructData, const ArgList& args)
66 {
67     return construct(exec, constructorObject, constructType, constructData, args, constructorObject);
68 }
69 
70 JS_EXPORT_PRIVATE JSObject* profiledConstruct(ExecState*, ProfilingReason, JSValue constructor, ConstructType, const ConstructData&, const ArgList&, JSValue newTarget);
71 
72 ALWAYS_INLINE JSObject* profiledConstruct(ExecState* exec, ProfilingReason reason, JSValue constructorObject, ConstructType constructType, const ConstructData& constructData, const ArgList& args)
73 {
74     return profiledConstruct(exec, reason, constructorObject, constructType, constructData, args, constructorObject);
75 }

77 } // namespace JSC
