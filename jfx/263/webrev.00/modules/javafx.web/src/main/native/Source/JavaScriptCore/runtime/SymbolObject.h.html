22 #pragma once
23 
24 #include "JSWrapperObject.h"
25 #include "Symbol.h"
26 
27 namespace JSC {
28 
29 class SymbolObject final : public JSWrapperObject {
30 public:
31     using Base = JSWrapperObject;
32 
33     template<typename, SubspaceAccess mode>
34     static IsoSubspace* subspaceFor(VM& vm)
35     {
36         return vm.symbolObjectSpace<mode>();
37     }
38 
39     static SymbolObject* create(VM& vm, Structure* structure)
40     {
41         Symbol* symbol = Symbol::create(vm);
42         SymbolObject* object = new (NotNull, allocateCell<SymbolObject>(vm.heap)) SymbolObject(vm, structure);
43         object->finishCreation(vm, symbol);
44         return object;
45     }
46     static SymbolObject* create(VM& vm, Structure* structure, Symbol* symbol)
47     {
48         SymbolObject* object = new (NotNull, allocateCell<SymbolObject>(vm.heap)) SymbolObject(vm, structure);
49         object->finishCreation(vm, symbol);
50         return object;
51     }
52     static SymbolObject* create(VM&, JSGlobalObject*, Symbol*);
53 
54     DECLARE_EXPORT_INFO;
55 
56     Symbol* internalValue() const { return asSymbol(JSWrapperObject::internalValue()); }
57 
58     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
59     {
60         return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
61     }
62 
63     static JSValue defaultValue(const JSObject*, JSGlobalObject*, PreferredPrimitiveType);
64 
65     static String toStringName(const JSObject*, JSGlobalObject*);
66 
67 protected:
68     JS_EXPORT_PRIVATE void finishCreation(VM&, Symbol*);
69     JS_EXPORT_PRIVATE SymbolObject(VM&, Structure*);
70 };
71 
72 } // namespace JSC
