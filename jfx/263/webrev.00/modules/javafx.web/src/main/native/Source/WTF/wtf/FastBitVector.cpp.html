35 void FastBitVectorWordOwner::setEqualsSlow(const FastBitVectorWordOwner& other)
36 {
37     uint32_t* newArray = static_cast<uint32_t*>(
38         FastBitVectorMalloc::zeroedMalloc(other.arrayLength() * sizeof(uint32_t)));
39     memcpy(newArray, other.m_words, other.arrayLength() * sizeof(uint32_t));
40     if (m_words)
41         FastBitVectorMalloc::free(m_words);
42     m_words = newArray;
43     m_numBits = other.m_numBits;
44 }
45 
46 void FastBitVectorWordOwner::resizeSlow(size_t numBits)
47 {
48     size_t newLength = fastBitVectorArrayLength(numBits);
49 
50     RELEASE_ASSERT(newLength >= arrayLength());
51 
52     // Use fastCalloc instead of fastRealloc because we expect the common
53     // use case for this method to be initializing the size of the bitvector.
54 
55     uint32_t* newArray = static_cast<uint32_t*>(FastBitVectorMalloc::zeroedMalloc(newLength * sizeof(uint32_t)));
56     memcpy(newArray, m_words, arrayLength() * sizeof(uint32_t));
57     if (m_words)
58         FastBitVectorMalloc::free(m_words);
59     m_words = newArray;
60 }
61 
62 void FastBitVector::clearRange(size_t begin, size_t end)
63 {
64     if (end - begin < 32) {
65         for (size_t i = begin; i < end; ++i)
66             at(i) = false;
67         return;
68     }
69 
70     size_t endBeginSlop = (begin + 31) & ~31;
71     size_t beginEndSlop = end & ~31;
72 
73     for (size_t i = begin; i < endBeginSlop; ++i)
74         at(i) = false;
75     for (size_t i = beginEndSlop; i < end; ++i)
76         at(i) = false;
77     for (size_t i = endBeginSlop / 32; i < beginEndSlop / 32; ++i)
78         m_words.word(i) = 0;
79 }
