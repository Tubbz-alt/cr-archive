26 #pragma once
27 
28 #include "Event.h"
29 #include <wtf/Markable.h>
30 
31 namespace WebCore {
32 
33 class WebAnimation;
34 
35 class AnimationEventBase : public Event {
36     WTF_MAKE_ISO_ALLOCATED(AnimationEventBase);
37 public:
38     static Ref<AnimationEventBase> create(const AtomString& type, WebAnimation* animation, Optional<Seconds> timelineTime)
39     {
40         return adoptRef(*new AnimationEventBase(type, animation, timelineTime));
41     }
42 
43     virtual ~AnimationEventBase();
44 
45     virtual bool isAnimationPlaybackEvent() const { return false; }
46     virtual bool isAnimationEvent() const { return false; }
47     virtual bool isTransitionEvent() const { return false; }
48 
49     Optional<Seconds> timelineTime() const { return m_timelineTime; }
50     WebAnimation* animation() const { return m_animation.get(); }
51 
52 protected:
53     AnimationEventBase(const AtomString&, WebAnimation*, Optional<Seconds>);
54     AnimationEventBase(const AtomString&, const EventInit&, IsTrusted);
55 
56     RefPtr<WebAnimation> m_animation;
57     Markable<Seconds, Seconds::MarkableTraits> m_timelineTime;
58 };
59 
60 }
61 
62 #define SPECIALIZE_TYPE_TRAITS_ANIMATION_EVENT_BASE(ToValueTypeName, predicate) \
63 SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::ToValueTypeName) \
64 static bool isType(const WebCore::AnimationEventBase& value) { return value.predicate; } \
65 SPECIALIZE_TYPE_TRAITS_END()
