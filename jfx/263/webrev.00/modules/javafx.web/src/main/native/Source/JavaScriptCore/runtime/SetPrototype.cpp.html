 58 void SetPrototype::finishCreation(VM& vm, JSGlobalObject* globalObject)
 59 {
 60     Base::finishCreation(vm);
 61     ASSERT(inherits(vm, info()));
 62 
 63     JSC_NATIVE_INTRINSIC_FUNCTION_WITHOUT_TRANSITION(vm.propertyNames->add, setProtoFuncAdd, static_cast<unsigned>(PropertyAttribute::DontEnum), 1, JSSetAddIntrinsic);
 64     JSC_NATIVE_FUNCTION_WITHOUT_TRANSITION(vm.propertyNames->clear, setProtoFuncClear, static_cast<unsigned>(PropertyAttribute::DontEnum), 0);
 65     JSC_NATIVE_FUNCTION_WITHOUT_TRANSITION(vm.propertyNames->deleteKeyword, setProtoFuncDelete, static_cast<unsigned>(PropertyAttribute::DontEnum), 1);
 66     JSC_NATIVE_INTRINSIC_FUNCTION_WITHOUT_TRANSITION(vm.propertyNames->has, setProtoFuncHas, static_cast<unsigned>(PropertyAttribute::DontEnum), 1, JSSetHasIntrinsic);
 67     JSC_NATIVE_INTRINSIC_FUNCTION_WITHOUT_TRANSITION(vm.propertyNames->builtinNames().hasPrivateName(), setProtoFuncHas, static_cast<unsigned>(PropertyAttribute::DontEnum), 1, JSSetHasIntrinsic);
 68     JSC_NATIVE_INTRINSIC_FUNCTION_WITHOUT_TRANSITION(vm.propertyNames->builtinNames().addPrivateName(), setProtoFuncAdd, static_cast<unsigned>(PropertyAttribute::DontEnum), 1, JSSetAddIntrinsic);
 69 
 70     JSFunction* values = JSFunction::create(vm, setPrototypeValuesCodeGenerator(vm), globalObject);
 71     putDirectWithoutTransition(vm, vm.propertyNames->builtinNames().valuesPublicName(), values, static_cast<unsigned>(PropertyAttribute::DontEnum));
 72     putDirectWithoutTransition(vm, vm.propertyNames->builtinNames().keysPublicName(), values, static_cast<unsigned>(PropertyAttribute::DontEnum));
 73     putDirectWithoutTransition(vm, vm.propertyNames->iteratorSymbol, values, static_cast<unsigned>(PropertyAttribute::DontEnum));
 74     putDirectWithoutTransition(vm, vm.propertyNames->toStringTagSymbol, jsNontrivialString(vm, "Set"_s), PropertyAttribute::DontEnum | PropertyAttribute::ReadOnly);
 75 
 76     JSC_NATIVE_GETTER_WITHOUT_TRANSITION(vm.propertyNames->size, setProtoFuncSize, PropertyAttribute::DontEnum | PropertyAttribute::Accessor);
 77 
 78     globalObject->installSetPrototypeWatchpoint(this);
 79 }

 81 ALWAYS_INLINE static JSSet* getSet(JSGlobalObject* globalObject, JSValue thisValue)
 82 {
 83     VM& vm = globalObject->vm();
 84     auto scope = DECLARE_THROW_SCOPE(vm);
 85 
 86     if (UNLIKELY(!thisValue.isCell())) {
 87         throwVMError(globalObject, scope, createNotAnObjectError(globalObject, thisValue));
 88         return nullptr;
 89     }
 90     auto* set = jsDynamicCast<JSSet*>(vm, thisValue.asCell());
 91     if (LIKELY(set))
 92         return set;
 93     throwTypeError(globalObject, scope, "Set operation called on non-Set object"_s);
 94     return nullptr;
 95 }

 97 EncodedJSValue JSC_HOST_CALL setProtoFuncAdd(JSGlobalObject* globalObject, CallFrame* callFrame)
 98 {
 99     JSValue thisValue = callFrame->thisValue();
100     JSSet* set = getSet(globalObject, thisValue);
101     if (!set)
102         return JSValue::encode(jsUndefined());
103     set->add(globalObject, callFrame->argument(0));
104     return JSValue::encode(thisValue);
105 }
106 
107 EncodedJSValue JSC_HOST_CALL setProtoFuncClear(JSGlobalObject* globalObject, CallFrame* callFrame)
108 {
109     JSSet* set = getSet(globalObject, callFrame->thisValue());
110     if (!set)
111         return JSValue::encode(jsUndefined());
112     set->clear(globalObject);
113     return JSValue::encode(jsUndefined());
114 }
115 
116 EncodedJSValue JSC_HOST_CALL setProtoFuncDelete(JSGlobalObject* globalObject, CallFrame* callFrame)
117 {
118     JSSet* set = getSet(globalObject, callFrame->thisValue());
119     if (!set)
120         return JSValue::encode(jsUndefined());
121     return JSValue::encode(jsBoolean(set->remove(globalObject, callFrame->argument(0))));
122 }

124 EncodedJSValue JSC_HOST_CALL setProtoFuncHas(JSGlobalObject* globalObject, CallFrame* callFrame)
125 {
126     JSSet* set = getSet(globalObject, callFrame->thisValue());
127     if (!set)
128         return JSValue::encode(jsUndefined());
129     return JSValue::encode(jsBoolean(set->has(globalObject, callFrame->argument(0))));
130 }
131 
132 EncodedJSValue JSC_HOST_CALL setProtoFuncSize(JSGlobalObject* globalObject, CallFrame* callFrame)
133 {
134     JSSet* set = getSet(globalObject, callFrame->thisValue());
135     if (!set)
136         return JSValue::encode(jsUndefined());
137     return JSValue::encode(jsNumber(set->size()));
138 }
