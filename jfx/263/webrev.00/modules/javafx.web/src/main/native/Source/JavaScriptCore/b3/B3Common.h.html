 50 template<typename BitsType, typename InputType>
 51 inline bool isIdentical(InputType left, InputType right)
 52 {
 53     BitsType leftBits = bitwise_cast<BitsType>(left);
 54     BitsType rightBits = bitwise_cast<BitsType>(right);
 55     return leftBits == rightBits;
 56 }
 57 
 58 inline bool isIdentical(int32_t left, int32_t right)
 59 {
 60     return isIdentical<int32_t>(left, right);
 61 }
 62 
 63 inline bool isIdentical(int64_t left, int64_t right)
 64 {
 65     return isIdentical<int64_t>(left, right);
 66 }
 67 
 68 inline bool isIdentical(double left, double right)
 69 {
 70     return isIdentical<int64_t>(left, right);
 71 }
 72 
 73 inline bool isIdentical(float left, float right)
 74 {
 75     return isIdentical<int32_t>(left, right);
 76 }

 78 template<typename ResultType, typename InputType, typename BitsType>
 79 inline bool isRepresentableAsImpl(InputType originalValue)
 80 {
 81     // Convert the original value to the desired result type.
 82     ResultType result = static_cast<ResultType>(originalValue);
 83 
 84     // Convert the converted value back to the original type. The original value is representable
 85     // using the new type if such round-tripping doesn't lose bits.
 86     InputType newValue = static_cast<InputType>(result);
 87 
 88     return isIdentical<BitsType>(originalValue, newValue);
 89 }

115 template<typename IntType>
116 static IntType chillDiv(IntType numerator, IntType denominator)
117 {
118     if (!denominator)
119         return 0;
120     if (denominator == -1 && numerator == std::numeric_limits<IntType>::min())
121         return std::numeric_limits<IntType>::min();
122     return numerator / denominator;
123 }
124 
125 template<typename IntType>
126 static IntType chillMod(IntType numerator, IntType denominator)
127 {
128     if (!denominator)
129         return 0;
130     if (denominator == -1 && numerator == std::numeric_limits<IntType>::min())
131         return 0;
132     return numerator % denominator;
133 }

135 template<typename IntType>
136 static IntType chillUDiv(IntType numerator, IntType denominator)
137 {
138     typedef typename std::make_unsigned<IntType>::type UnsignedIntType;
139     UnsignedIntType unsignedNumerator = static_cast<UnsignedIntType>(numerator);
140     UnsignedIntType unsignedDenominator = static_cast<UnsignedIntType>(denominator);
141     if (!unsignedDenominator)
142         return 0;
143     return unsignedNumerator / unsignedDenominator;
144 }
145 
146 template<typename IntType>
147 static IntType chillUMod(IntType numerator, IntType denominator)
148 {
149     typedef typename std::make_unsigned<IntType>::type UnsignedIntType;
150     UnsignedIntType unsignedNumerator = static_cast<UnsignedIntType>(numerator);
151     UnsignedIntType unsignedDenominator = static_cast<UnsignedIntType>(denominator);
152     if (!unsignedDenominator)
153         return 0;
154     return unsignedNumerator % unsignedDenominator;
155 }

157 template<typename IntType>
158 static IntType rotateRight(IntType value, int32_t shift)
159 {
160     typedef typename std::make_unsigned<IntType>::type UnsignedIntType;
161     UnsignedIntType uValue = static_cast<UnsignedIntType>(value);
162     int32_t bits = sizeof(IntType) * 8;
163     int32_t mask = bits - 1;
164     shift &= mask;
165     return (uValue >> shift) | (uValue << ((bits - shift) & mask));
166 }
167 
168 template<typename IntType>
169 static IntType rotateLeft(IntType value, int32_t shift)
170 {
171     typedef typename std::make_unsigned<IntType>::type UnsignedIntType;
172     UnsignedIntType uValue = static_cast<UnsignedIntType>(value);
173     int32_t bits = sizeof(IntType) * 8;
174     int32_t mask = bits - 1;
175     shift &= mask;
176     return (uValue << shift) | (uValue >> ((bits - shift) & mask));
177 }
