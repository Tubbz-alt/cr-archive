22 #pragma once
23 
24 #include "SVGElement.h"

25 #include "SVGTests.h"
26 #include "SVGURIReference.h"
27 
28 namespace WebCore {
29 
30 class CSSCursorImageValue;
31 
32 class SVGCursorElement final : public SVGElement, public SVGTests, public SVGURIReference {
33     WTF_MAKE_ISO_ALLOCATED(SVGCursorElement);
34 public:
35     static Ref<SVGCursorElement> create(const QualifiedName&, Document&);
36 
37     virtual ~SVGCursorElement();
38 
39     void addClient(CSSCursorImageValue&);
40     void removeClient(CSSCursorImageValue&);
41 
42     const SVGLengthValue& x() const { return m_x->currentValue(); }
43     const SVGLengthValue& y() const { return m_y->currentValue(); }
44 
45     SVGAnimatedLength& xAnimated() { return m_x; }
46     SVGAnimatedLength& yAnimated() { return m_y; }
47 
48 private:
49     SVGCursorElement(const QualifiedName&, Document&);
50 
51     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGCursorElement, SVGElement, SVGTests, SVGURIReference>;
52     const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
53 
54     void parseAttribute(const QualifiedName&, const AtomString&) final;
55     void svgAttributeChanged(const QualifiedName&) final;
56 
57     bool isValid() const final { return SVGTests::isValid(); }
58     bool rendererIsNeeded(const RenderStyle&) final { return false; }
59 
60     void addSubresourceAttributeURLs(ListHashSet<URL>&) const final;
61 
62     PropertyRegistry m_propertyRegistry { *this };
63     Ref<SVGAnimatedLength> m_x { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
64     Ref<SVGAnimatedLength> m_y { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
65     HashSet<CSSCursorImageValue*> m_clients;
66 };
67 
68 } // namespace WebCore
