27 
28 #include "HashMapImpl.h"
29 #include "JSObject.h"
30 
31 namespace JSC {
32 
33 class JSSet final : public HashMapImpl<HashMapBucket<HashMapBucketDataKey>> {
34     using Base = HashMapImpl<HashMapBucket<HashMapBucketDataKey>>;
35 public:
36 
37     DECLARE_EXPORT_INFO;
38 
39     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
40     {
41         return Structure::create(vm, globalObject, prototype, TypeInfo(JSSetType, StructureFlags), info());
42     }
43 
44     static JSSet* create(ExecState* exec, VM& vm, Structure* structure)
45     {
46         return create(exec, vm, structure, 0);
47     }
48 
49     static JSSet* create(ExecState* exec, VM& vm, Structure* structure, uint32_t size)
50     {
51         JSSet* instance = new (NotNull, allocateCell<JSSet>(vm.heap)) JSSet(vm, structure, size);
52         instance->finishCreation(exec, vm);
53         return instance;
54     }
55 
56     bool isIteratorProtocolFastAndNonObservable();
57     bool canCloneFastAndNonObservable(Structure*);
58     JSSet* clone(ExecState*, VM&, Structure*);
59 
60 private:
61     JSSet(VM& vm, Structure* structure)
62         : Base(vm, structure)
63     {
64     }
65 
66     JSSet(VM& vm, Structure* structure, uint32_t sizeHint)
67         : Base(vm, structure, sizeHint)
68     {
69     }
70 
71     static String toStringName(const JSObject*, ExecState*);
72 };
73 
74 static_assert(std::is_final<JSSet>::value, "Required for JSType based casting");
75 
76 } // namespace JSC
