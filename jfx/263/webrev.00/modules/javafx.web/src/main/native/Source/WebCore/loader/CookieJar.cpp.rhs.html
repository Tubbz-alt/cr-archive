 45 Ref<CookieJar> CookieJar::create(Ref<StorageSessionProvider>&& storageSessionProvider)
 46 {
 47     return adoptRef(*new CookieJar(WTFMove(storageSessionProvider)));
 48 }

 50 IncludeSecureCookies CookieJar::shouldIncludeSecureCookies(const Document& document, const URL& url)
 51 {
 52     return (url.protocolIs("https") && !document.foundMixedContent().contains(SecurityContext::MixedContentType::Active)) ? IncludeSecureCookies::Yes : IncludeSecureCookies::No;
 53 }
 54 
 55 SameSiteInfo CookieJar::sameSiteInfo(const Document& document)
 56 {
 57     if (auto* loader = document.loader())
 58         return SameSiteInfo::create(loader->request());
 59     return { };
 60 }

 69 String CookieJar::cookies(Document& document, const URL& url) const
 70 {
 71     TraceScope scope(FetchCookiesStart, FetchCookiesEnd);
 72 
 73     auto includeSecureCookies = shouldIncludeSecureCookies(document, url);
 74 
 75     Optional<FrameIdentifier> frameID;
 76     Optional<PageIdentifier> pageID;
 77     if (auto* frame = document.frame()) {
 78         frameID = frame->loader().client().frameID();
 79         pageID = frame->loader().client().pageID();
 80     }
 81 
 82     std::pair<String, bool> result;
 83     if (auto* session = m_storageSessionProvider->storageSession())
 84         result = session->cookiesForDOM(document.firstPartyForCookies(), sameSiteInfo(document), url, frameID, pageID, includeSecureCookies, ShouldAskITP::Yes);
 85     else
 86         ASSERT_NOT_REACHED();
 87 
 88     if (result.second)
 89         document.setSecureCookiesAccessed();
 90 
 91     return result.first;
 92 }

 94 CookieRequestHeaderFieldProxy CookieJar::cookieRequestHeaderFieldProxy(const Document& document, const URL& url)
 95 {
 96     TraceScope scope(FetchCookiesStart, FetchCookiesEnd);
 97 
 98     Optional<FrameIdentifier> frameID;
 99     Optional<PageIdentifier> pageID;
100     if (auto* frame = document.frame()) {
101         frameID = frame->loader().client().frameID();
102         pageID = frame->loader().client().pageID();
103     }
104 
105     return { document.firstPartyForCookies(), sameSiteInfo(document), url, frameID, pageID, shouldIncludeSecureCookies(document, url) };
106 }

108 void CookieJar::setCookies(Document& document, const URL& url, const String& cookieString)
109 {
110     Optional<FrameIdentifier> frameID;
111     Optional<PageIdentifier> pageID;
112     if (auto* frame = document.frame()) {
113         frameID = frame->loader().client().frameID();
114         pageID = frame->loader().client().pageID();
115     }
116 
117     if (auto* session = m_storageSessionProvider->storageSession())
118         session->setCookiesFromDOM(document.firstPartyForCookies(), sameSiteInfo(document), url, frameID, pageID, ShouldAskITP::Yes, cookieString);
119     else
120         ASSERT_NOT_REACHED();
121 }

123 bool CookieJar::cookiesEnabled(const Document&) const
124 {
125     if (auto* session = m_storageSessionProvider->storageSession())
126         return session->cookieAcceptPolicy() != HTTPCookieAcceptPolicy::Never;
127 
128     ASSERT_NOT_REACHED();
129     return false;
130 }

132 std::pair<String, SecureCookiesAccessed> CookieJar::cookieRequestHeaderFieldValue(const URL& firstParty, const SameSiteInfo& sameSiteInfo, const URL& url, Optional<FrameIdentifier> frameID, Optional<PageIdentifier> pageID, IncludeSecureCookies includeSecureCookies) const
133 {
134     if (auto* session = m_storageSessionProvider->storageSession()) {
135         std::pair<String, bool> result = session->cookieRequestHeaderFieldValue(firstParty, sameSiteInfo, url, frameID, pageID, includeSecureCookies, ShouldAskITP::Yes);
136         return { result.first, result.second ? SecureCookiesAccessed::Yes : SecureCookiesAccessed::No };
137     }
138 
139     ASSERT_NOT_REACHED();
140     return { };
141 }
142 
143 String CookieJar::cookieRequestHeaderFieldValue(Document& document, const URL& url) const
144 {
145     Optional<FrameIdentifier> frameID;
146     Optional<PageIdentifier> pageID;
147     if (auto* frame = document.frame()) {
148         frameID = frame->loader().client().frameID();
149         pageID = frame->loader().client().pageID();
150     }
151 
152     auto result = cookieRequestHeaderFieldValue(document.firstPartyForCookies(), sameSiteInfo(document), url, frameID, pageID, shouldIncludeSecureCookies(document, url));
153     if (result.second == SecureCookiesAccessed::Yes)
154         document.setSecureCookiesAccessed();
155     return result.first;
156 }

158 bool CookieJar::getRawCookies(const Document& document, const URL& url, Vector<Cookie>& cookies) const
159 {
160     Optional<FrameIdentifier> frameID;
161     Optional<PageIdentifier> pageID;
162     if (auto* frame = document.frame()) {
163         frameID = frame->loader().client().frameID();
164         pageID = frame->loader().client().pageID();
165     }
166 
167     if (auto* session = m_storageSessionProvider->storageSession())
168         return session->getRawCookies(document.firstPartyForCookies(), sameSiteInfo(document), url, frameID, pageID, ShouldAskITP::Yes, cookies);
169 
170     ASSERT_NOT_REACHED();
171     return false;
172 }

174 void CookieJar::deleteCookie(const Document&, const URL& url, const String& cookieName)
175 {
176     if (auto* session = m_storageSessionProvider->storageSession())
177         session->deleteCookie(url, cookieName);
178     else
179         ASSERT_NOT_REACHED();
180 }
