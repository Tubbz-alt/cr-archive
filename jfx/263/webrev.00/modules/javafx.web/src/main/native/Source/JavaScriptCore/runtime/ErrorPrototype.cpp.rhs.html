 57 void ErrorPrototypeBase::finishCreation(VM& vm, const String& name)
 58 {
 59     Base::finishCreation(vm);
 60     ASSERT(inherits(vm, info()));
 61     putDirectWithoutTransition(vm, vm.propertyNames->name, jsString(vm, name), static_cast<unsigned>(PropertyAttribute::DontEnum));
 62     putDirectWithoutTransition(vm, vm.propertyNames->message, jsEmptyString(vm), static_cast<unsigned>(PropertyAttribute::DontEnum));
 63 }

 72 // ECMA-262 5.1, 15.11.4.4
 73 EncodedJSValue JSC_HOST_CALL errorProtoFuncToString(JSGlobalObject* globalObject, CallFrame* callFrame)
 74 {
 75     VM& vm = globalObject->vm();
 76     auto scope = DECLARE_THROW_SCOPE(vm);
 77 
 78     // 1. Let O be the this value.
 79     JSValue thisValue = callFrame->thisValue();
 80 
 81     // 2. If Type(O) is not Object, throw a TypeError exception.
 82     if (!thisValue.isObject())
 83         return throwVMTypeError(globalObject, scope);
 84     JSObject* thisObj = asObject(thisValue);
 85 
 86     // Guard against recursion!
 87     StringRecursionChecker checker(globalObject, thisObj);
 88     EXCEPTION_ASSERT(!scope.exception() || checker.earlyReturnValue());
 89     if (JSValue earlyReturnValue = checker.earlyReturnValue())
 90         return JSValue::encode(earlyReturnValue);
 91 
 92     // 3. Let name be the result of calling the [[Get]] internal method of O with argument "name".
 93     JSValue name = thisObj->get(globalObject, vm.propertyNames->name);
 94     RETURN_IF_EXCEPTION(scope, encodedJSValue());
 95 
 96     // 4. If name is undefined, then let name be "Error"; else let name be ToString(name).
 97     String nameString;
 98     if (name.isUndefined())
 99         nameString = "Error"_s;
100     else {
101         nameString = name.toWTFString(globalObject);
102         RETURN_IF_EXCEPTION(scope, encodedJSValue());
103     }
104 
105     // 5. Let msg be the result of calling the [[Get]] internal method of O with argument "message".
106     JSValue message = thisObj->get(globalObject, vm.propertyNames->message);
107     RETURN_IF_EXCEPTION(scope, encodedJSValue());
108 
109     // (sic)
110     // 6. If msg is undefined, then let msg be the empty String; else let msg be ToString(msg).
111     // 7. If msg is undefined, then let msg be the empty String; else let msg be ToString(msg).
112     String messageString;
113     if (message.isUndefined())
114         messageString = String();
115     else {
116         messageString = message.toWTFString(globalObject);
117         RETURN_IF_EXCEPTION(scope, encodedJSValue());
118     }
119 
120     // 8. If name is the empty String, return msg.
121     if (!nameString.length())
122         return JSValue::encode(message.isString() ? message : jsString(vm, messageString));
123 
124     // 9. If msg is the empty String, return name.
125     if (!messageString.length())
126         return JSValue::encode(name.isString() ? name : jsString(vm, nameString));
127 
128     // 10. Return the result of concatenating name, ":", a single space character, and msg.
129     RELEASE_AND_RETURN(scope, JSValue::encode(jsMakeNontrivialString(globalObject, nameString, ": ", messageString)));
130 }
