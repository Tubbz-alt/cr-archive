 60 struct ScrollableAreaParameters {
 61     ScrollElasticity horizontalScrollElasticity { ScrollElasticityNone };
 62     ScrollElasticity verticalScrollElasticity { ScrollElasticityNone };
 63 
 64     ScrollbarMode horizontalScrollbarMode { ScrollbarAuto };
 65     ScrollbarMode verticalScrollbarMode { ScrollbarAuto };
 66 
 67     bool hasEnabledHorizontalScrollbar { false };
 68     bool hasEnabledVerticalScrollbar { false };
 69 
 70     bool horizontalScrollbarHiddenByStyle { false };
 71     bool verticalScrollbarHiddenByStyle { false };
 72 
 73     bool useDarkAppearanceForScrollbars { false };
 74 
 75     bool operator==(const ScrollableAreaParameters& other) const
 76     {
 77         return horizontalScrollElasticity == other.horizontalScrollElasticity
 78             && verticalScrollElasticity == other.verticalScrollElasticity
 79             && horizontalScrollbarMode == other.horizontalScrollbarMode
 80             && verticalScrollbarMode == other.verticalScrollbarMode
 81             && hasEnabledHorizontalScrollbar == other.hasEnabledHorizontalScrollbar
 82             && horizontalScrollbarHiddenByStyle == other.horizontalScrollbarHiddenByStyle
 83             && verticalScrollbarHiddenByStyle == other.verticalScrollbarHiddenByStyle
 84             && hasEnabledVerticalScrollbar == other.hasEnabledVerticalScrollbar
 85             && useDarkAppearanceForScrollbars == other.useDarkAppearanceForScrollbars;
 86     }
 87 };
