28 namespace WebCore {
29 
30 StyleBackgroundData::StyleBackgroundData()
31     : background(FillLayerType::Background)
32     , color(RenderStyle::initialBackgroundColor())
33 {
34 }
35 
36 inline StyleBackgroundData::StyleBackgroundData(const StyleBackgroundData& other)
37     : RefCounted<StyleBackgroundData>()
38     , background(other.background)
39     , color(other.color)
40     , outline(other.outline)
41 {
42 }
43 
44 Ref<StyleBackgroundData> StyleBackgroundData::copy() const
45 {
46     return adoptRef(*new StyleBackgroundData(*this));
47 }
48 
49 bool StyleBackgroundData::operator==(const StyleBackgroundData& other) const
50 {
51     return background == other.background && color == other.color && outline == other.outline;
52 }
53 
54 bool StyleBackgroundData::isEquivalentForPainting(const StyleBackgroundData& other) const
55 {
56     if (background != other.background || color != other.color)
57         return false;
58     if (!outline.isVisible() && !other.outline.isVisible())
59         return true;
60     return outline == other.outline;
61 }
62 
63 void StyleBackgroundData::dump(TextStream& ts, DumpStyleValues behavior) const
64 {
65     if (behavior == DumpStyleValues::All || background != FillLayer(FillLayerType::Background))
66         ts.dumpProperty("background-image", background);
67     if (behavior == DumpStyleValues::All || color != RenderStyle::initialBackgroundColor())
68         ts.dumpProperty("background-color", color);
69     if (behavior == DumpStyleValues::All || outline != OutlineValue())
70         ts.dumpProperty("outline", outline);
71 }
72 
73 TextStream& operator<<(TextStream& ts, const StyleBackgroundData& backgroundData)
74 {
75     backgroundData.dump(ts);
76     return ts;
77 }
