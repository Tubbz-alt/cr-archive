31 #pragma once
32 
33 #if ENABLE(WEB_RTC)
34 
35 #include "MediaStreamTrack.h"
36 #include "RTCRtpReceiverBackend.h"
37 #include "RTCRtpSynchronizationSource.h"
38 #include "ScriptWrappable.h"
39 
40 namespace WebCore {
41 
42 class DeferredPromise;
43 class PeerConnectionBackend;
44 struct RTCRtpCapabilities;
45 
46 class RTCRtpReceiver final : public RefCounted<RTCRtpReceiver>, public ScriptWrappable  {
47     WTF_MAKE_ISO_ALLOCATED(RTCRtpReceiver);
48 public:
49     static Ref<RTCRtpReceiver> create(PeerConnectionBackend& connection, Ref<MediaStreamTrack>&& track, std::unique_ptr<RTCRtpReceiverBackend>&& backend)
50     {
51         return adoptRef(*new RTCRtpReceiver(connection, WTFMove(track), WTFMove(backend)));
52     }
53 
54     static Optional<RTCRtpCapabilities> getCapabilities(ScriptExecutionContext&, const String& kind);
55 
56     void stop();
57 
58     void setBackend(std::unique_ptr<RTCRtpReceiverBackend>&& backend) { m_backend = WTFMove(backend); }
59     RTCRtpParameters getParameters() { return m_backend ? m_backend->getParameters() : RTCRtpParameters(); }
60     Vector<RTCRtpContributingSource> getContributingSources() const { return m_backend ? m_backend->getContributingSources() : Vector<RTCRtpContributingSource> { }; }
61     Vector<RTCRtpSynchronizationSource> getSynchronizationSources() const { return m_backend ? m_backend->getSynchronizationSources() : Vector<RTCRtpSynchronizationSource> { }; }
62 
63     MediaStreamTrack& track() { return m_track.get(); }
64 
65     RTCRtpReceiverBackend* backend() { return m_backend.get(); }
66     void getStats(Ref<DeferredPromise>&&);
67 
68 private:
69     RTCRtpReceiver(PeerConnectionBackend&, Ref<MediaStreamTrack>&&, std::unique_ptr<RTCRtpReceiverBackend>&&);
70 
71     Ref<MediaStreamTrack> m_track;
72     std::unique_ptr<RTCRtpReceiverBackend> m_backend;
73     WeakPtr<PeerConnectionBackend> m_connection;
74 };
75 
76 } // namespace WebCore
77 
78 #endif // ENABLE(WEB_RTC)
