 34 inline JSFunction* JSFunction::createWithInvalidatedReallocationWatchpoint(
 35     VM& vm, FunctionExecutable* executable, JSScope* scope)
 36 {
 37     ASSERT(executable->singleton().hasBeenInvalidated());
 38     return createImpl(vm, executable, scope, selectStructureForNewFuncExp(scope->globalObject(vm), executable));
 39 }

 48 inline FunctionExecutable* JSFunction::jsExecutable() const
 49 {
 50     ASSERT(!isHostFunctionNonInline());
 51     return static_cast<FunctionExecutable*>(executable());
 52 }

 54 inline bool JSFunction::isHostFunction() const
 55 {
 56     ASSERT(executable());
 57     return executable()->isHostFunction();
 58 }

 60 inline Intrinsic JSFunction::intrinsic() const

 65 inline bool JSFunction::isBuiltinFunction() const
 66 {
 67     return !isHostFunction() && jsExecutable()->isBuiltinFunction();
 68 }

 70 inline bool JSFunction::isHostOrBuiltinFunction() const
 71 {
 72     return isHostFunction() || isBuiltinFunction();

 75 inline bool JSFunction::isClassConstructorFunction() const

 80 inline TaggedNativeFunction JSFunction::nativeFunction()

 86 inline TaggedNativeFunction JSFunction::nativeConstructor()

 92 inline bool isHostFunction(JSValue value, TaggedNativeFunction nativeFunction)
 93 {
 94     JSFunction* function = jsCast<JSFunction*>(getJSFunction(value));
 95     if (!function || !function->isHostFunction())
 96         return false;
 97     return function->nativeFunction() == nativeFunction;
 98 }

114 inline bool JSFunction::areNameAndLengthOriginal(VM&)
115 {
116     FunctionRareData* rareData = this->rareData();
117     if (!rareData)
118         return true;
119     if (rareData->hasModifiedName())
120         return false;
121     if (rareData->hasModifiedLength())
122         return false;
123     return true;
124 }

126 inline bool JSFunction::canUseAllocationProfile()
127 {
128     if (isHostOrBuiltinFunction()) {
129         if (isHostFunction())
130             return false;
131 
132         VM& vm = globalObject()->vm();
133         unsigned attributes;
134         JSValue prototype = getDirect(vm, vm.propertyNames->prototype, attributes);
135         if (!prototype || (attributes & PropertyAttribute::AccessorOrCustomAccessorOrValue))
136             return false;
137     }
138 
139     // If we don't have a prototype property, we're not guaranteed it's
140     // non-configurable. For example, user code can define the prototype
141     // as a getter. JS semantics require that the getter is called every
142     // time |construct| occurs with this function as new.target.
143     return jsExecutable()->hasPrototypeProperty();
144 }
145 
146 inline FunctionRareData* JSFunction::ensureRareDataAndAllocationProfile(JSGlobalObject* globalObject, unsigned inlineCapacity)
147 {
148     ASSERT(canUseAllocationProfile());
149     FunctionRareData* rareData = this->rareData();
150     if (!rareData)
151         return allocateAndInitializeRareData(globalObject, inlineCapacity);
152     if (UNLIKELY(!rareData->isObjectAllocationProfileInitialized()))
153         return initializeRareData(globalObject, inlineCapacity);
154     return rareData;
155 }
