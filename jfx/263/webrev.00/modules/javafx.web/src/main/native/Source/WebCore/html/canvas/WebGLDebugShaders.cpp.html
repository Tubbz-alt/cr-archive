38 WebGLDebugShaders::WebGLDebugShaders(WebGLRenderingContextBase& context)
39     : WebGLExtension(context)
40 {
41     context.graphicsContextGL()->getExtensions().ensureEnabled("GL_ANGLE_translated_shader_source");
42 }
43 
44 WebGLDebugShaders::~WebGLDebugShaders() = default;
45 
46 WebGLExtension::ExtensionName WebGLDebugShaders::getName() const
47 {
48     return WebGLDebugShadersName;
49 }
50 
51 String WebGLDebugShaders::getTranslatedShaderSource(WebGLShader& shader)
52 {
53     if (m_context.isContextLost())
54         return String();
55     if (!m_context.validateWebGLObject("getTranslatedShaderSource", &shader))
56         return emptyString();
57     return m_context.graphicsContextGL()->getExtensions().getTranslatedShaderSourceANGLE(shader.object());
58 }
