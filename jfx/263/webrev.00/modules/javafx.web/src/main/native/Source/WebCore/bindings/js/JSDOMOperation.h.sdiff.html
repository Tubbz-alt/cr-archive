32 class IDLOperation {

34     using ClassParameter = JSClass*;
35     using Operation = JSC::EncodedJSValue(JSC::ExecState*, ClassParameter, JSC::ThrowScope&);
36     using StaticOperation = JSC::EncodedJSValue(JSC::ExecState*, JSC::ThrowScope&);

38     static JSClass* cast(JSC::ExecState&);
39 
40     template<Operation operation, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::Throw>
41     static JSC::EncodedJSValue call(JSC::ExecState& state, const char* operationName)
42     {
43         auto throwScope = DECLARE_THROW_SCOPE(state.vm());
44 
45         auto* thisObject = cast(state);
46         if (shouldThrow != CastedThisErrorBehavior::Assert && UNLIKELY(!thisObject))
47             return throwThisTypeError(state, throwScope, JSClass::info()->className, operationName);
48 
49         ASSERT(thisObject);
50         ASSERT_GC_OBJECT_INHERITS(thisObject, JSClass::info());
51 
52         // FIXME: We should refactor the binding generated code to use references for state and thisObject.
53         return operation(&state, thisObject, throwScope);
54     }
55 
56     template<StaticOperation operation, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::Throw>
57     static JSC::EncodedJSValue callStatic(JSC::ExecState& state, const char*)
58     {
59         auto throwScope = DECLARE_THROW_SCOPE(state.vm());
60 
61         // FIXME: We should refactor the binding generated code to use references for state.
62         return operation(&state, throwScope);
63     }

40     template<Operation operation, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::Throw>

43         auto throwScope = DECLARE_THROW_SCOPE(JSC::getVM(&lexicalGlobalObject));
44

54     }

57     static JSC::EncodedJSValue callStatic(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, const char*)
