 28 #include "DeferGC.h"
 29 #include <wtf/Lock.h>
 30 #include <wtf/NoLock.h>
 31 #include <wtf/Optional.h>
 32 
 33 namespace JSC {
 34 
 35 using ConcurrentJSLock = Lock;
 36 using ConcurrentJSLockerImpl = LockHolder;
 37 
 38 static_assert(sizeof(ConcurrentJSLock) == 1, "Regardless of status of concurrent JS flag, size of ConurrentJSLock is always one byte.");
 39 
 40 class ConcurrentJSLockerBase : public AbstractLocker {
 41     WTF_MAKE_NONCOPYABLE(ConcurrentJSLockerBase);
 42 public:
 43     explicit ConcurrentJSLockerBase(ConcurrentJSLock& lockable)
 44         : m_locker(&lockable)
 45     {
 46     }
 47     explicit ConcurrentJSLockerBase(ConcurrentJSLock* lockable)
 48         : m_locker(lockable)
 49     {
 50     }
 51 
 52     explicit ConcurrentJSLockerBase(NoLockingNecessaryTag)
 53         : m_locker(NoLockingNecessary)
 54     {
 55     }
 56 
 57     ~ConcurrentJSLockerBase()
 58     {
 59     }
 60 
 61     void unlockEarly()
 62     {
 63         m_locker.unlockEarly();
 64     }
 65 
 66 private:
 67     ConcurrentJSLockerImpl m_locker;
 68 };
 69 
 70 class GCSafeConcurrentJSLocker : public ConcurrentJSLockerBase {
 71 public:
 72     GCSafeConcurrentJSLocker(ConcurrentJSLock& lockable, Heap& heap)
 73         : ConcurrentJSLockerBase(lockable)
 74         , m_deferGC(heap)
 75     {
 76     }
 77 
 78     GCSafeConcurrentJSLocker(ConcurrentJSLock* lockable, Heap& heap)
 79         : ConcurrentJSLockerBase(lockable)
 80         , m_deferGC(heap)
 81     {
 82     }
 83 
 84     ~GCSafeConcurrentJSLocker()
 85     {
 86         // We have to unlock early due to the destruction order of base
 87         // vs. derived classes. If we didn't, then we would destroy the
 88         // DeferGC object before unlocking the lock which could cause a GC
 89         // and resulting deadlock.
 90         unlockEarly();
 91     }
 92 
 93 private:
 94     DeferGC m_deferGC;
 95 };
 96 
 97 class ConcurrentJSLocker : public ConcurrentJSLockerBase {
 98 public:
 99     ConcurrentJSLocker(ConcurrentJSLock& lockable)
100         : ConcurrentJSLockerBase(lockable)
101 #if !defined(NDEBUG)
102         , m_disallowGC(std::in_place)
103 #endif
104     {
105     }
106 
107     ConcurrentJSLocker(ConcurrentJSLock* lockable)
108         : ConcurrentJSLockerBase(lockable)
109 #if !defined(NDEBUG)
110         , m_disallowGC(std::in_place)
111 #endif
112     {
113     }
114 
115     ConcurrentJSLocker(NoLockingNecessaryTag)
116         : ConcurrentJSLockerBase(NoLockingNecessary)
117 #if !defined(NDEBUG)
118         , m_disallowGC(WTF::nullopt)
119 #endif
120     {
121     }
122 
123     ConcurrentJSLocker(int) = delete;
124 
125 #if !defined(NDEBUG)
126 private:
127     Optional<DisallowGC> m_disallowGC;
128 #endif
129 };

131 } // namespace JSC
