34 Change determineChange(const RenderStyle& s1, const RenderStyle& s2)
35 {
36     if (s1.display() != s2.display())
37         return Detach;
38     if (s1.hasPseudoStyle(PseudoId::FirstLetter) != s2.hasPseudoStyle(PseudoId::FirstLetter))
39         return Detach;
40     // We just detach if a renderer acquires or loses a column-span, since spanning elements
41     // typically won't contain much content.
42     if (s1.columnSpan() != s2.columnSpan())
43         return Detach;
44     if (!s1.contentDataEquivalent(&s2))
45         return Detach;
46     // When text-combine property has been changed, we need to prepare a separate renderer object.
47     // When text-combine is on, we use RenderCombineText, otherwise RenderText.
48     // https://bugs.webkit.org/show_bug.cgi?id=55069
49     if (s1.hasTextCombine() != s2.hasTextCombine())
50         return Detach;
51 
52     if (s1 != s2) {
53         if (s1.inheritedNotEqual(&s2))
54             return Inherit;
55 
56         if (s1.alignItems() != s2.alignItems() || s1.justifyItems() != s2.justifyItems())
57             return Inherit;
58 

59         return NoInherit;
60     }
61 
62     // If the pseudoStyles have changed, we want any StyleChange that is not NoChange
63     // because setStyle will do the right thing with anything else.
64     if (s1.hasAnyPublicPseudoStyles()) {
65         for (PseudoId pseudoId = PseudoId::FirstPublicPseudoId; pseudoId < PseudoId::FirstInternalPseudoId; pseudoId = static_cast<PseudoId>(static_cast<unsigned>(pseudoId) + 1)) {
66             if (s1.hasPseudoStyle(pseudoId)) {
67                 RenderStyle* ps2 = s2.getCachedPseudoStyle(pseudoId);
68                 if (!ps2)
69                     return NoInherit;
70                 RenderStyle* ps1 = s1.getCachedPseudoStyle(pseudoId);
71                 if (!ps1 || *ps1 != *ps2)
72                     return NoInherit;
73             }
74         }
75     }
76 
77     return NoChange;
78 }
