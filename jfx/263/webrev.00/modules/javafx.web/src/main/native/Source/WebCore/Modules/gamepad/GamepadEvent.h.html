26 #pragma once
27 
28 #if ENABLE(GAMEPAD)
29 
30 #include "Event.h"
31 #include "Gamepad.h"
32 #include <wtf/RefPtr.h>
33 
34 namespace WebCore {
35 
36 class GamepadEvent final : public Event {
37     WTF_MAKE_ISO_ALLOCATED(GamepadEvent);
38 public:
39     ~GamepadEvent() = default;
40 
41     static Ref<GamepadEvent> create(const AtomString& eventType, Gamepad& gamepad)
42     {
43         return adoptRef(*new GamepadEvent(eventType, gamepad));
44     }
45 
46     struct Init : EventInit {
47         RefPtr<Gamepad> gamepad;
48     };
49 
50     static Ref<GamepadEvent> create(const AtomString& eventType, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
51     {
52         return adoptRef(*new GamepadEvent(eventType, initializer, isTrusted));
53     }
54 
55     Gamepad* gamepad() const { return m_gamepad.get(); }
56 
57     EventInterface eventInterface() const override { return GamepadEventInterfaceType; }
58 
59 private:
60     explicit GamepadEvent(const AtomString& eventType, Gamepad&);
61     GamepadEvent(const AtomString& eventType, const Init&, IsTrusted);
62 
63     RefPtr<Gamepad> m_gamepad;
64 };
65 
66 } // namespace WebCore
67 
68 #endif // ENABLE(GAMEPAD)
