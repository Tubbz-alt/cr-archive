 62 class IntSize;
 63 
 64 class FloatSize {
 65 public:
 66     FloatSize() { }
 67     FloatSize(float width, float height) : m_width(width), m_height(height) { }
 68     WEBCORE_EXPORT FloatSize(const IntSize&);
 69 
 70     static FloatSize narrowPrecision(double width, double height);
 71 
 72     float width() const { return m_width; }
 73     float height() const { return m_height; }
 74 
 75     void setWidth(float width) { m_width = width; }
 76     void setHeight(float height) { m_height = height; }
 77 
 78     bool isEmpty() const { return m_width <= 0 || m_height <= 0; }
 79     WEBCORE_EXPORT bool isZero() const;
 80     bool isExpressibleAsIntSize() const;
 81 
 82     float aspectRatio() const { return m_width / m_height; }
 83 
 84     void expand(float width, float height)
 85     {
 86         m_width += width;
 87         m_height += height;
 88     }
 89 
 90     void scale(float s)
 91     {
 92         m_width *= s;
 93         m_height *= s;
 94     }
 95 
 96     void scale(float scaleX, float scaleY)
 97     {
 98         m_width *= scaleX;
 99         m_height *= scaleY;
100     }
101 
102     FloatSize scaled(float s) const
103     {
104         return { m_width * s, m_height * s };
105     }
106 
107     FloatSize scaled(float scaleX, float scaleY) const
108     {
109         return { m_width * scaleX, m_height * scaleY };
110     }
111 
112     WEBCORE_EXPORT FloatSize constrainedBetween(const FloatSize& min, const FloatSize& max) const;
113 
114     FloatSize expandedTo(const FloatSize& other) const
115     {
116         return FloatSize(m_width > other.m_width ? m_width : other.m_width,
117             m_height > other.m_height ? m_height : other.m_height);
118     }
119 
120     FloatSize shrunkTo(const FloatSize& other) const
121     {
122        return FloatSize(m_width < other.m_width ? m_width : other.m_width,
123            m_height < other.m_height ? m_height : other.m_height);
124     }
125 
126     float diagonalLength() const
127     {
128         return std::hypot(m_width, m_height);
129     }
130 
131     float diagonalLengthSquared() const
132     {
133         return m_width * m_width + m_height * m_height;
134     }
135 
136     float area() const
137     {
138         return m_width * m_height;
139     }
140 
141     FloatSize transposedSize() const
142     {
143         return FloatSize(m_height, m_width);
144     }
145 
146 #if USE(CG)
147     WEBCORE_EXPORT explicit FloatSize(const CGSize&); // don't do this implicitly since it's lossy
148     WEBCORE_EXPORT operator CGSize() const;
149 #endif
150 
151 #if PLATFORM(MAC) && !defined(NSGEOMETRY_TYPES_SAME_AS_CGGEOMETRY_TYPES)
152     WEBCORE_EXPORT explicit FloatSize(const NSSize&); // don't do this implicitly since it's lossy
153     operator NSSize() const;
154 #endif
155 
156 #if PLATFORM(WIN)
157     WEBCORE_EXPORT FloatSize(const D2D1_SIZE_F&);
158     operator D2D1_SIZE_F() const;
159 #endif
160 
161     String toJSONString() const;
162     Ref<JSON::Object> toJSONObject() const;
163 
164 private:
165     float m_width { 0 };
166     float m_height { 0 };
167 };
168 
169 inline FloatSize& operator+=(FloatSize& a, const FloatSize& b)
170 {
171     a.setWidth(a.width() + b.width());
172     a.setHeight(a.height() + b.height());
173     return a;
174 }
175 
176 inline FloatSize& operator-=(FloatSize& a, const FloatSize& b)
177 {
178     a.setWidth(a.width() - b.width());
179     a.setHeight(a.height() - b.height());
180     return a;
181 }
182 
183 inline FloatSize operator+(const FloatSize& a, const FloatSize& b)
184 {
185     return FloatSize(a.width() + b.width(), a.height() + b.height());
186 }
187 
188 inline FloatSize operator-(const FloatSize& a, const FloatSize& b)
189 {
190     return FloatSize(a.width() - b.width(), a.height() - b.height());
191 }
192 
193 inline FloatSize operator-(const FloatSize& size)
194 {
195     return FloatSize(-size.width(), -size.height());
196 }
197 
198 inline FloatSize operator*(const FloatSize& a, float b)
199 {
200     return FloatSize(a.width() * b, a.height() * b);
201 }
202 
203 inline FloatSize operator*(float a, const FloatSize& b)
204 {
205     return FloatSize(a * b.width(), a * b.height());
206 }
207 
208 inline FloatSize operator*(const FloatSize& a, const FloatSize& b)
209 {
210     return FloatSize(a.width() * b.width(), a.height() * b.height());
211 }
212 
213 inline FloatSize operator/(const FloatSize& a, const FloatSize& b)
214 {
215     return FloatSize(a.width() / b.width(), a.height() / b.height());
216 }
217 
218 inline FloatSize operator/(const FloatSize& a, float b)
219 {
220     return FloatSize(a.width() / b, a.height() / b);
221 }
222 
223 inline FloatSize operator/(float a, const FloatSize& b)
224 {
225     return FloatSize(a / b.width(), a / b.height());
226 }
227 
228 inline bool operator==(const FloatSize& a, const FloatSize& b)
229 {
230     return a.width() == b.width() && a.height() == b.height();
231 }
232 
233 inline bool areEssentiallyEqual(const FloatSize& a, const FloatSize& b)
234 {
235     return WTF::areEssentiallyEqual(a.width(), b.width()) && WTF::areEssentiallyEqual(a.height(), b.height());
236 }
237 
238 inline bool operator!=(const FloatSize& a, const FloatSize& b)
239 {
240     return a.width() != b.width() || a.height() != b.height();
241 }
242 
243 inline IntSize roundedIntSize(const FloatSize& p)
244 {
245     return IntSize(clampToInteger(roundf(p.width())), clampToInteger(roundf(p.height())));
246 }
247 
248 inline IntSize flooredIntSize(const FloatSize& p)
249 {
250     return IntSize(clampToInteger(floorf(p.width())), clampToInteger(floorf(p.height())));
251 }
252 
253 inline IntSize expandedIntSize(const FloatSize& p)
254 {
255     return IntSize(clampToInteger(ceilf(p.width())), clampToInteger(ceilf(p.height())));
256 }
257 
258 inline IntPoint flooredIntPoint(const FloatSize& p)
259 {
260     return IntPoint(clampToInteger(floorf(p.width())), clampToInteger(floorf(p.height())));
261 }
