34 namespace WebCore {
35 
36 WTF_MAKE_ISO_ALLOCATED_IMPL(BlobEvent);
37 
38 Ref<BlobEvent> BlobEvent::create(const AtomString& type, Init&& init, IsTrusted isTrusted)
39 {
40     return adoptRef(*new BlobEvent(type, WTFMove(init), isTrusted));
41 }
42 
43 Ref<BlobEvent> BlobEvent::create(const AtomString& type, CanBubble canBubble, IsCancelable isCancelable, Ref<Blob>&& data)
44 {
45     return adoptRef(*new BlobEvent(type, canBubble, isCancelable, WTFMove(data)));
46 }
47 
48 BlobEvent::BlobEvent(const AtomString& type, Init&& init, IsTrusted isTrusted)
49     : Event(type, init, isTrusted)
50     , m_blob(init.data.releaseNonNull())
51 {
52 }
53 
54 BlobEvent::BlobEvent(const AtomString& type, CanBubble canBubble, IsCancelable isCancelable, Ref<Blob>&& data)
55     : Event(type, canBubble, isCancelable)
56     , m_blob(WTFMove(data))
57 {
58 }
59 
60 EventInterface BlobEvent::eventInterface() const
61 {
62     return BlobEventInterfaceType;
63 }
