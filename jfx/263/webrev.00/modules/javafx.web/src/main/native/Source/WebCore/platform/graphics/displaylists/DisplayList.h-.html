 42 namespace DisplayList {

 44 class Item;
 45 
 46 enum AsTextFlag {
 47     None                            = 0,
 48     IncludesPlatformOperations      = 1 << 0,
 49 };
 50 
 51 typedef unsigned AsTextFlags;
 52 
 53 class DisplayList {
 54     WTF_MAKE_NONCOPYABLE(DisplayList); WTF_MAKE_FAST_ALLOCATED;
 55     friend class Recorder;
 56     friend class Replayer;
 57 public:
 58     DisplayList() = default;
 59     DisplayList(DisplayList&&) = default;
 60 
 61     DisplayList& operator=(DisplayList&&) = default;
 62 
 63     void dump(WTF::TextStream&) const;
 64 
 65     const Vector<Ref<Item>>& list() const { return m_list; }
 66     Item& itemAt(size_t index)
 67     {
 68         ASSERT(index < m_list.size());
 69         return m_list[index].get();
 70     }
 71 
 72     void clear();
 73     void removeItemsFromIndex(size_t);
 74 
 75     size_t itemCount() const { return m_list.size(); }
 76     size_t sizeInBytes() const;
 77 
 78     String asText(AsTextFlags) const;
 79 
 80 #if !defined(NDEBUG) || !LOG_DISABLED
 81     WTF::CString description() const;
 82     void dump() const;
 83 #endif
 84 
 85 private:
 86     Item& append(Ref<Item>&& item)
 87     {
 88         m_list.append(WTFMove(item));
 89         return m_list.last().get();
 90     }
 91 
 92     // Less efficient append, only used for tracking replay.
 93     void appendItem(Item& item)
 94     {
 95         m_list.append(item);
 96     }
 97 
 98     static bool shouldDumpForFlags(AsTextFlags, const Item&);
 99 
100     Vector<Ref<Item>>& list() { return m_list; }
101 
102     Vector<Ref<Item>> m_list;
103 };
104 
105 } // DisplayList
106 
107 WTF::TextStream& operator<<(WTF::TextStream&, const DisplayList::DisplayList&);
108
