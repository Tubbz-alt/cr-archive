48 inline double normalizeNonFiniteValueOrFallBackTo(Optional<double> value, double fallbackValue)
49 {
50     // Normalize non-finite values (https://drafts.csswg.org/cssom-view/#normalize-non-finite-values).
51     return value ? (std::isfinite(*value) ? *value : 0) : fallbackValue;
52 }

57     ScrollToOptions options = value;
58     options.left = normalizeNonFiniteValueOrFallBackTo(value.left, x);
59     options.top = normalizeNonFiniteValueOrFallBackTo(value.top, y);
60     return options;
61 }
