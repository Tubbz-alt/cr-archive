43 WTF_MAKE_ISO_ALLOCATED_IMPL(RTCTrackEvent);
44 
45 Ref<RTCTrackEvent> RTCTrackEvent::create(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, RefPtr<RTCRtpReceiver>&& receiver, RefPtr<MediaStreamTrack>&& track, Vector<RefPtr<MediaStream>>&& streams, RefPtr<RTCRtpTransceiver>&& transceiver)
46 {
47     return adoptRef(*new RTCTrackEvent(type, canBubble, cancelable, WTFMove(receiver), WTFMove(track), WTFMove(streams), WTFMove(transceiver)));
48 }
49 
50 Ref<RTCTrackEvent> RTCTrackEvent::create(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
51 {
52     return adoptRef(*new RTCTrackEvent(type, initializer, isTrusted));
53 }
54 
55 RTCTrackEvent::RTCTrackEvent(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, RefPtr<RTCRtpReceiver>&& receiver, RefPtr<MediaStreamTrack>&& track, Vector<RefPtr<MediaStream>>&& streams, RefPtr<RTCRtpTransceiver>&& transceiver)
56     : Event(type, canBubble, cancelable)
57     , m_receiver(WTFMove(receiver))
58     , m_track(WTFMove(track))
59     , m_streams(WTFMove(streams))
60     , m_transceiver(WTFMove(transceiver))
61 {
62 }
63 
64 RTCTrackEvent::RTCTrackEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
65     : Event(type, initializer, isTrusted)
66     , m_receiver(initializer.receiver)
67     , m_track(initializer.track)
68     , m_streams(initializer.streams)
69     , m_transceiver(initializer.transceiver)
70 {
71 }
