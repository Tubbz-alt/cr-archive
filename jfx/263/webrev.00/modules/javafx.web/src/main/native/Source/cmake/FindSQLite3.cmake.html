 24 #[=======================================================================[.rst:
 25 FindSQLite3
 26 --------------
 27 
 28 Find SQLite3 headers and libraries.
 29 
 30 Imported Targets
 31 ^^^^^^^^^^^^^^^^
 32 
 33 ``SQLite::SQLite3``
 34   The SQLite3 library, if found.
 35 
 36 Result Variables
 37 ^^^^^^^^^^^^^^^^
 38 
 39 This will define the following variables in your project:
 40 
 41 ``SQLite3_FOUND``
 42   true if (the requested version of) SQLite3 is available.
 43 ``SQLite3_VERSION``
 44   the version of SQLite3.
 45 ``SQLite3_LIBRARIES``
 46   the libraries to link against to use SQLite3.
 47 ``SQLite3_INCLUDE_DIRS``
 48   where to find the SQLite3 headers.
 49 ``SQLite3_COMPILE_OPTIONS``
 50   this should be passed to target_compile_options(), if the
 51   target is not used for linking
 52 
 53 #]=======================================================================]
 54 
 55 # TODO: Remove when cmake_minimum_version bumped to 3.14.
 56 
 57 include(FindPkgConfig)
 58 pkg_check_modules(PC_SQLITE3 QUIET sqlite3)
 59 set(SQLite3_COMPILE_OPTIONS ${PC_SQLITE3_CFLAGS_OTHER})
 60 set(SQLite3_VERSION ${PC_SQLITE3_VERSION})
 61 
 62 find_path(SQLite3_INCLUDE_DIR
 63     NAMES sqlite3.h
 64     HINTS ${PC_SQLITE3_INCLUDEDIR} ${PC_SQLITE3_INCLUDE_DIR}
 65 )
 66 
 67 find_library(SQLite3_LIBRARY
 68     NAMES ${SQLite3_NAMES} sqlite3
 69     HINTS ${PC_SQLITE3_LIBDIR} ${PC_SQLITE3_LIBRARY_DIRS}
 70 )
 71 
 72 if (SQLite3_INCLUDE_DIR AND NOT SQLite3_VERSION)
 73     if (EXISTS "${SQLite3_INCLUDE_DIR}/sqlite3.h")
 74         file(STRINGS ${SQLite3_INCLUDE_DIR}/sqlite3.h _ver_line
 75             REGEX "^#define SQLITE_VERSION  *\"[0-9]+\\.[0-9]+\\.[0-9]+\""
 76             LIMIT_COUNT 1)
 77         string(REGEX MATCH "[0-9]+\\.[0-9]+\\.[0-9]+"
 78             SQLite3_VERSION "${_ver_line}")
 79         unset(_ver_line)
 80     endif ()
 81 endif ()
 82 
 83 include(FindPackageHandleStandardArgs)
 84 find_package_handle_standard_args(SQLite3
 85     FOUND_VAR SQLite3_FOUND
 86     REQUIRED_VARS SQLite3_LIBRARY SQLite3_INCLUDE_DIR
 87     VERSION_VAR SQLite3_VERSION
 88 )
 89 
 90 if (SQLite3_LIBRARY AND NOT TARGET SQLite::SQLite3)
 91     add_library(SQLite::SQLite3 UNKNOWN IMPORTED GLOBAL)
 92     set_target_properties(SQLite::SQLite3 PROPERTIES
 93         IMPORTED_LOCATION "${SQLite3_LIBRARY}"
 94         INTERFACE_COMPILE_OPTIONS "${SQLite3_COMPILE_OPTIONS}"
 95         INTERFACE_INCLUDE_DIRECTORIES "${SQLite3_INCLUDE_DIR}"
 96     )
 97 endif ()
 98 
 99 mark_as_advanced(SQLite3_INCLUDE_DIR SQLite3_LIBRARIES)
100 
101 if (SQLite3_FOUND)
102     set(SQLite3_LIBRARIES ${SQLite3_LIBRARY})
103     set(SQLite3_INCLUDE_DIRS ${SQLite3_INCLUDE_DIR})
104 endif ()
