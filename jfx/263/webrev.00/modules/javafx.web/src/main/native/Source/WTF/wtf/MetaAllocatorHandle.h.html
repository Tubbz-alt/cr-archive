 29 #pragma once
 30 
 31 #include <wtf/Assertions.h>
 32 #include <wtf/MetaAllocatorPtr.h>
 33 #include <wtf/RedBlackTree.h>
 34 #include <wtf/ThreadSafeRefCounted.h>
 35 
 36 namespace WTF {
 37 
 38 class MetaAllocator;
 39 class PrintStream;
 40 
 41 DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(MetaAllocatorHandle);
 42 class MetaAllocatorHandle : public ThreadSafeRefCounted<MetaAllocatorHandle>, public RedBlackTree<MetaAllocatorHandle, void*>::Node {
 43     WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(MetaAllocatorHandle);
 44 private:
 45     MetaAllocatorHandle(MetaAllocator*, void* start, size_t sizeInBytes, void* ownerUID);
 46 
 47 public:
 48     using MemoryPtr = MetaAllocatorPtr<HandleMemoryPtrTag>;
 49 
 50     WTF_EXPORT_PRIVATE ~MetaAllocatorHandle();
 51 
 52     MemoryPtr start() const
 53     {
 54         return m_start;
 55     }
 56 
 57     MemoryPtr end() const
 58     {
 59         return m_end;
 60     }
 61 
 62     uintptr_t startAsInteger() const
 63     {
 64         return m_start.untaggedPtr<uintptr_t>();
 65     }
 66 
 67     uintptr_t endAsInteger() const
 68     {
 69         return m_end.untaggedPtr<uintptr_t>();
 70     }
 71 
 72     size_t sizeInBytes() const
 73     {
 74         return m_end.untaggedPtr<size_t>() - m_start.untaggedPtr<size_t>();
 75     }
 76 
 77     bool containsIntegerAddress(uintptr_t address) const
 78     {
 79         return address >= startAsInteger() && address < endAsInteger();
 80     }
 81 
 82     bool contains(void* address) const
 83     {
 84         return containsIntegerAddress(reinterpret_cast<uintptr_t>(address));
 85     }
 86 
 87     WTF_EXPORT_PRIVATE void shrink(size_t newSizeInBytes);
 88 
 89     bool isManaged()
 90     {
 91         return !!m_allocator;
 92     }
 93 
 94     MetaAllocator* allocator()
 95     {
 96         ASSERT(m_allocator);
 97         return m_allocator;
 98     }
 99 
100     void* ownerUID()
101     {
102         return m_ownerUID;
103     }
104 
105     void* key()
106     {
107         return m_start.untaggedPtr();
108     }
109 
110     WTF_EXPORT_PRIVATE void dump(PrintStream& out) const;
111 
112 private:
113     friend class MetaAllocator;
114 
115     MetaAllocator* m_allocator;
116     MemoryPtr m_start;
117     MemoryPtr m_end;
118     void* m_ownerUID;
119 };
120 
121 }
