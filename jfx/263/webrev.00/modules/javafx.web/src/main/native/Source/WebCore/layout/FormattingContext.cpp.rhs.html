 70 void FormattingContext::computeOutOfFlowHorizontalGeometry(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints)
 71 {
 72     ASSERT(layoutBox.isOutOfFlowPositioned());


 73     auto compute = [&](Optional<LayoutUnit> usedWidth) {
 74         return geometry().outOfFlowHorizontalGeometry(layoutBox, horizontalConstraints, { usedWidth, { } });

 75     };
 76 
 77     auto containingBlockWidth = horizontalConstraints.logicalWidth;
 78     auto horizontalGeometry = compute({ });
 79     if (auto maxWidth = geometry().computedMaxWidth(layoutBox, containingBlockWidth)) {
 80         auto maxHorizontalGeometry = compute(maxWidth);
 81         if (horizontalGeometry.contentWidthAndMargin.contentWidth > maxHorizontalGeometry.contentWidthAndMargin.contentWidth)
 82             horizontalGeometry = maxHorizontalGeometry;
 83     }
 84 
 85     if (auto minWidth = geometry().computedMinWidth(layoutBox, containingBlockWidth)) {
 86         auto minHorizontalGeometry = compute(minWidth);
 87         if (horizontalGeometry.contentWidthAndMargin.contentWidth < minHorizontalGeometry.contentWidthAndMargin.contentWidth)
 88             horizontalGeometry = minHorizontalGeometry;
 89     }
 90 
 91     auto& displayBox = formattingState().displayBox(layoutBox);
 92     displayBox.setLeft(horizontalGeometry.left + horizontalGeometry.contentWidthAndMargin.usedMargin.start);
 93     displayBox.setContentBoxWidth(horizontalGeometry.contentWidthAndMargin.contentWidth);
 94     displayBox.setHorizontalMargin(horizontalGeometry.contentWidthAndMargin.usedMargin);
 95     displayBox.setHorizontalComputedMargin(horizontalGeometry.contentWidthAndMargin.computedMargin);
 96 }

 98 void FormattingContext::computeOutOfFlowVerticalGeometry(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints, const VerticalConstraints& verticalConstraints)
 99 {
100     ASSERT(layoutBox.isOutOfFlowPositioned());
101     auto compute = [&](Optional<LayoutUnit> usedHeight) {
102         return geometry().outOfFlowVerticalGeometry(layoutBox, horizontalConstraints, verticalConstraints, { usedHeight });

103     };
104 
105     auto containingBlockHeight = *verticalConstraints.logicalHeight;
106     auto verticalGeometry = compute({ });
107     if (auto maxHeight = geometry().computedMaxHeight(layoutBox, containingBlockHeight)) {
108         auto maxVerticalGeometry = compute(maxHeight);
109         if (verticalGeometry.contentHeightAndMargin.contentHeight > maxVerticalGeometry.contentHeightAndMargin.contentHeight)
110             verticalGeometry = maxVerticalGeometry;
111     }
112 
113     if (auto minHeight = geometry().computedMinHeight(layoutBox, containingBlockHeight)) {
114         auto minVerticalGeometry = compute(minHeight);
115         if (verticalGeometry.contentHeightAndMargin.contentHeight < minVerticalGeometry.contentHeightAndMargin.contentHeight)
116             verticalGeometry = minVerticalGeometry;
117     }
118 
119     auto& displayBox = formattingState().displayBox(layoutBox);
120     auto nonCollapsedVerticalMargin = verticalGeometry.contentHeightAndMargin.nonCollapsedMargin;
121     displayBox.setTop(verticalGeometry.top + nonCollapsedVerticalMargin.before);
122     displayBox.setContentBoxHeight(verticalGeometry.contentHeightAndMargin.contentHeight);
123     // Margins of absolutely positioned boxes do not collapse
124     displayBox.setVerticalMargin({ nonCollapsedVerticalMargin, { } });
125 }

134 void FormattingContext::layoutOutOfFlowContent(InvalidationState& invalidationState, const HorizontalConstraints& rootHorizontalConstraints, const VerticalConstraints& rootVerticalConstraints)
135 {
136     LOG_WITH_STREAM(FormattingContextLayout, stream << "Start: layout out-of-flow content -> context: " << &layoutState() << " root: " << &root());
137 
138     collectOutOfFlowDescendantsIfNeeded();
139 
140     auto horizontalConstraintsForOutOfFlowBox = [&] (const auto& outOfFlowBox) {
141         auto* containingBlock = outOfFlowBox.containingBlock();
142         ASSERT(containingBlock);
143         if (containingBlock == &root())
144             return rootHorizontalConstraints;
145         return Geometry::horizontalConstraintsForOutOfFlow(geometryForBox(*containingBlock));
146     };
147 
148     auto verticalConstraintsForOutOfFlowBox = [&] (const auto& outOfFlowBox) {
149         auto* containingBlock = outOfFlowBox.containingBlock();
150         ASSERT(containingBlock);
151         if (containingBlock == &root())
152             return rootVerticalConstraints;
153         return Geometry::verticalConstraintsForOutOfFlow(geometryForBox(*containingBlock));
154     };
155 
156     for (auto& outOfFlowBox : formattingState().outOfFlowBoxes()) {
157         ASSERT(outOfFlowBox->establishesFormattingContext());
158         if (!invalidationState.needsLayout(*outOfFlowBox))
159             continue;
160 
161         auto horizontalConstraints = horizontalConstraintsForOutOfFlowBox(*outOfFlowBox);
162         auto verticalConstraints = verticalConstraintsForOutOfFlowBox(*outOfFlowBox);
163         // Borders and paddings are resolved against the containing block's content box like if this box was an in-flow box.
164         auto& outOfFlowRootDisplayBox = geometryForBox(*outOfFlowBox);
165         computeBorderAndPadding(*outOfFlowBox, Geometry::horizontalConstraintsForInFlow(outOfFlowRootDisplayBox));
166         computeOutOfFlowHorizontalGeometry(*outOfFlowBox, horizontalConstraints);
167         if (!is<Container>(*outOfFlowBox) || !downcast<Container>(*outOfFlowBox).hasChild()) {
168             computeOutOfFlowVerticalGeometry(*outOfFlowBox, horizontalConstraints, verticalConstraints);
169             continue;
170         }
171 
172         auto& outOfFlowRootContainer = downcast<Container>(*outOfFlowBox);
173         auto formattingContext = LayoutContext::createFormattingContext(outOfFlowRootContainer, layoutState());
174         if (outOfFlowRootContainer.hasInFlowOrFloatingChild())
175             formattingContext->layoutInFlowContent(invalidationState, Geometry::horizontalConstraintsForInFlow(outOfFlowRootDisplayBox), Geometry::verticalConstraintsForInFlow(outOfFlowRootDisplayBox));
176         computeOutOfFlowVerticalGeometry(outOfFlowRootContainer, horizontalConstraints, verticalConstraints);
177         formattingContext->layoutOutOfFlowContent(invalidationState, Geometry::horizontalConstraintsForInFlow(outOfFlowRootDisplayBox), Geometry::verticalConstraintsForInFlow(outOfFlowRootDisplayBox));
178     }
179     LOG_WITH_STREAM(FormattingContextLayout, stream << "End: layout out-of-flow content -> context: " << &layoutState() << " root: " << &root());
180 }

182 static LayoutUnit mapHorizontalPositionToAncestor(const FormattingContext& formattingContext, LayoutUnit horizontalPosition, const Container& containingBlock, const Container& ancestor)
183 {
184     // "horizontalPosition" is in the coordinate system of the "containingBlock". -> map from containingBlock to ancestor.
185     if (&containingBlock == &ancestor)
186         return horizontalPosition;
187     ASSERT(containingBlock.isContainingBlockDescendantOf(ancestor));
188     for (auto* container = &containingBlock; container && container != &ancestor; container = container->containingBlock())
189         horizontalPosition += formattingContext.geometryForBox(*container).left();
190     return horizontalPosition;
191 }

193 // FIXME: turn these into templates.
194 LayoutUnit FormattingContext::mapTopToFormattingContextRoot(const Box& layoutBox) const
195 {
196     ASSERT(layoutBox.containingBlock());
197     auto& formattingContextRoot = root();
198     ASSERT(layoutBox.isContainingBlockDescendantOf(formattingContextRoot));
199     auto top = geometryForBox(layoutBox).top();
200     for (auto* container = layoutBox.containingBlock(); container && container != &formattingContextRoot; container = container->containingBlock())
201         top += geometryForBox(*container).top();
202     return top;
203 }

205 LayoutUnit FormattingContext::mapLeftToFormattingContextRoot(const Box& layoutBox) const
206 {
207     ASSERT(layoutBox.containingBlock());
208     return mapHorizontalPositionToAncestor(*this, geometryForBox(layoutBox).left(), *layoutBox.containingBlock(), root());
209 }

211 LayoutUnit FormattingContext::mapRightToFormattingContextRoot(const Box& layoutBox) const
212 {
213     ASSERT(layoutBox.containingBlock());
214     return mapHorizontalPositionToAncestor(*this, geometryForBox(layoutBox).right(), *layoutBox.containingBlock(), root());

217 const Display::Box& FormattingContext::geometryForBox(const Box& layoutBox, Optional<EscapeReason> escapeReason) const
218 {
219     UNUSED_PARAM(escapeReason);
220 #if ASSERT_ENABLED
221     auto isOkToAccessDisplayBox = [&] {
222         if (!layoutBox.isInitialContainingBlock() && &layoutBox.formattingContextRoot() == &root()) {
223             // This is the non-escape case of accessing a box's geometry information within the same formatting context.
224             return true;
225         }
226 
227         if (!escapeReason) {
228             // Any geometry access outside of the formatting context without a valid reason is considered an escape.
229             return false;
230         }
231 
232         if (*escapeReason == EscapeReason::BodyStrechesToViewportQuirk) {
233             ASSERT(layoutState().inQuirksMode());
234             return layoutBox.isInitialContainingBlock();
235         }
236 
237         if (*escapeReason == EscapeReason::StrokeOverflowNeedsViewportGeometry)
238             return layoutBox.isInitialContainingBlock();
239 
240         if (*escapeReason == EscapeReason::NeedsGeometryFromEstablishedFormattingContext) {
241             // This is the case when a formatting root collects geometry information from the established
242             // formatting context to be able to determine width/height.
243             // e.g <div>text content</div>. The <div> is a formatting root of the IFC.
244             // In order to compute the height of the <div>, we need to look inside the IFC and gather geometry information.
245             return &layoutBox.formattingContextRoot().formattingContextRoot() == &root();
246         }
247 
248         if (*escapeReason == EscapeReason::OutOfFlowBoxNeedsInFlowGeometry) {
249             // When computing the static position of an out-of-flow box, we need to gather sibling/parent geometry information
250             // as if the out-of-flow box was a simple inflow box.
251             // Now since the out-of-flow and the sibling/parent boxes could very well be in different containing block subtrees
252             // the formatting context they live in could also be very different.
253             return true;
254         }
255 
256         if (*escapeReason == EscapeReason::FloatBoxNeedsToBeInAbsoluteCoordinates) {
257             // Float box top/left values are mapped relative to the FloatState's root. Inline formatting contexts(A) inherit floats from parent
258             // block formatting contexts(B). Floats in these inline formatting contexts(A) need to be mapped to the parent, block formatting context(B).
259             auto& formattingContextRoot = layoutBox.formattingContextRoot();
260             return &formattingContextRoot == &root() || &formattingContextRoot == &root().formattingContextRoot();
261         }
262 
263         if (*escapeReason == EscapeReason::FindFixedHeightAncestorQuirk) {
264             ASSERT(layoutState().inQuirksMode());
265             // Find the first containing block with fixed height quirk. See Quirks::heightValueOfNearestContainingBlockWithFixedHeight
266             auto& targetFormattingRoot = layoutBox.formattingContextRoot();
267             auto* ancestorFormattingContextRoot = &root().formattingContextRoot();
268             while (true) {
269                 if (&targetFormattingRoot == ancestorFormattingContextRoot)
270                     return true;
271                 if (ancestorFormattingContextRoot->isInitialContainingBlock())
272                     return false;
273                 ancestorFormattingContextRoot = &ancestorFormattingContextRoot->formattingContextRoot();
274             }
275             return false;
276         }
277 
278         if (*escapeReason == EscapeReason::TableNeedsAccessToTableWrapper) {
279             // Tables are wrapped in a 2 level formatting context structure. A <table> element initiates a block formatting context for its principal table box
280             // where the caption and the table content live. It also initiates a table wrapper box which establishes the table formatting context.
281             // In many cases the TFC needs access to the parent (generated) BFC.
282             return &layoutBox == &root().formattingContextRoot();
283         }
284 
285         ASSERT_NOT_REACHED();
286         return false;
287     };
288 #endif
289     ASSERT(isOkToAccessDisplayBox());
290     ASSERT(layoutState().hasDisplayBox(layoutBox));
291     return layoutState().displayBoxForLayoutBox(layoutBox);
292 }

294 void FormattingContext::collectOutOfFlowDescendantsIfNeeded()
295 {
296     if (!formattingState().outOfFlowBoxes().isEmpty())
297         return;
298     auto& root = this->root();
299     if (!root.hasChild())
300         return;
301     if (!root.isPositioned() && !root.isInitialContainingBlock())
302         return;
303     // Collect the out-of-flow descendants at the formatting root level (as opposed to at the containing block level, though they might be the same).
304     // FIXME: Turn this into a register-self as boxes are being inserted.
305     for (auto& descendant : descendantsOfType<Box>(root)) {
306         if (!descendant.isOutOfFlowPositioned())
307             continue;
308         if (&descendant.formattingContextRoot() != &root)
309             continue;
310         formattingState().addOutOfFlowBox(descendant);
311     }
312 }

314 #ifndef NDEBUG
315 void FormattingContext::validateGeometryConstraintsAfterLayout() const
316 {
317     auto& formattingContextRoot = root();



318     // FIXME: add a descendantsOfType<> flavor that stops at nested formatting contexts
319     for (auto& layoutBox : descendantsOfType<Box>(formattingContextRoot)) {
320         if (&layoutBox.formattingContextRoot() != &formattingContextRoot)
321             continue;
322         auto& containingBlockGeometry = geometryForBox(*layoutBox.containingBlock());
323         auto& boxGeometry = geometryForBox(layoutBox);
324 
325         // 10.3.3 Block-level, non-replaced elements in normal flow
326         // 10.3.7 Absolutely positioned, non-replaced elements
327         if ((layoutBox.isBlockLevelBox() || layoutBox.isOutOfFlowPositioned()) && !layoutBox.replaced()) {
328             // margin-left + border-left-width + padding-left + width + padding-right + border-right-width + margin-right = width of containing block
329             auto containingBlockWidth = containingBlockGeometry.contentBoxWidth();
330             ASSERT(boxGeometry.horizontalMarginBorderAndPadding() + boxGeometry.contentBoxWidth() == containingBlockWidth);
331         }
332 
333         // 10.6.4 Absolutely positioned, non-replaced elements
334         if (layoutBox.isOutOfFlowPositioned() && !layoutBox.replaced()) {
335             // top + margin-top + border-top-width + padding-top + height + padding-bottom + border-bottom-width + margin-bottom + bottom = height of containing block
336             auto containingBlockHeight = containingBlockGeometry.contentBoxHeight();
337             ASSERT(boxGeometry.top() + boxGeometry.marginBefore() + boxGeometry.borderTop() + boxGeometry.paddingTop().valueOr(0) + boxGeometry.contentBoxHeight()
338                 + boxGeometry.paddingBottom().valueOr(0) + boxGeometry.borderBottom() + boxGeometry.marginAfter() == containingBlockHeight);
339         }
340     }
341 }
342 #endif
