 34 class AffineTransform;
 35 class CanvasBase;
 36 class CanvasRenderingContext;
 37 class Element;
 38 class GraphicsContext;
 39 class GraphicsContextStateSaver;
 40 class Image;
 41 class ImageBuffer;
 42 class FloatRect;
 43 class ScriptExecutionContext;
 44 class SecurityOrigin;
 45 
 46 class CanvasObserver {
 47 public:
 48     virtual ~CanvasObserver() = default;
 49 
 50     virtual bool isCanvasObserverProxy() const { return false; }
 51 
 52     virtual void canvasChanged(CanvasBase&, const FloatRect& changedRect) = 0;
 53     virtual void canvasResized(CanvasBase&) = 0;
 54     virtual void canvasDestroyed(CanvasBase&) = 0;
 55 };
 56 
 57 class CanvasBase {
 58 public:
 59     virtual ~CanvasBase();
 60 
 61     virtual void refCanvasBase() = 0;
 62     virtual void derefCanvasBase() = 0;
 63 
 64     virtual bool isHTMLCanvasElement() const { return false; }
 65     virtual bool isOffscreenCanvas() const { return false; }
 66     virtual bool isCustomPaintCanvas() const { return false; }
 67 
 68     virtual unsigned width() const { return m_size.width(); }
 69     virtual unsigned height() const { return m_size.height(); }
 70     const IntSize& size() const { return m_size; }
 71 
 72     ImageBuffer* buffer() const;
 73 
 74     virtual AffineTransform baseTransform() const;
 75 
 76     void makeRenderingResultsAvailable();
 77 
 78     size_t memoryCost() const;
 79     size_t externalMemoryCost() const;
 80 
 81     void setOriginClean() { m_originClean = true; }
 82     void setOriginTainted() { m_originClean = false; }
 83     bool originClean() const { return m_originClean; }
 84 
 85     virtual SecurityOrigin* securityOrigin() const { return nullptr; }
 86     ScriptExecutionContext* scriptExecutionContext() const { return canvasBaseScriptExecutionContext();  }
 87 
 88     virtual CanvasRenderingContext* renderingContext() const = 0;
 89 
 90     void addObserver(CanvasObserver&);
 91     void removeObserver(CanvasObserver&);
 92     void notifyObserversCanvasChanged(const FloatRect&);
 93     void notifyObserversCanvasResized();
 94     void notifyObserversCanvasDestroyed(); // Must be called in destruction before clearing m_context.
 95 
 96     HashSet<Element*> cssCanvasClients() const;
 97 
 98     virtual GraphicsContext* drawingContext() const;
 99     virtual GraphicsContext* existingDrawingContext() const;
100 
101     virtual void didDraw(const FloatRect&) = 0;
102 
103     virtual Image* copiedImage() const = 0;
104     bool callTracingActive() const;
105 
106 protected:
107     explicit CanvasBase(IntSize);
108 
109     virtual ScriptExecutionContext* canvasBaseScriptExecutionContext() const = 0;
110 
111     virtual void setSize(const IntSize& size) { m_size = size; }
112 
113     std::unique_ptr<ImageBuffer> setImageBuffer(std::unique_ptr<ImageBuffer>&&) const;
114     virtual bool hasCreatedImageBuffer() const { return false; }
115     static size_t activePixelMemory();
116 
117     void resetGraphicsContextState() const;
118 
119 private:
120     virtual void createImageBuffer() const { }
121 
122     mutable IntSize m_size;
123     mutable Lock m_imageBufferAssignmentLock;
124     mutable std::unique_ptr<ImageBuffer> m_imageBuffer;
125     mutable size_t m_imageBufferCost { 0 };
126     mutable std::unique_ptr<GraphicsContextStateSaver> m_contextStateSaver;
127 
128     bool m_originClean { true };
129 #if ASSERT_ENABLED
130     bool m_didNotifyObserversCanvasDestroyed { false };
131 #endif
132     HashSet<CanvasObserver*> m_observers;
133 };

135 } // namespace WebCore
