21 #pragma once
22 
23 #include "ErrorPrototype.h"
24 
25 namespace JSC {
26 
27 class NativeErrorPrototype final : public ErrorPrototypeBase {
28 private:
29     NativeErrorPrototype(VM&, Structure*);
30 
31 public:
32     typedef ErrorPrototypeBase Base;
33 
34     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
35     {
36         return Structure::create(vm, globalObject, prototype, TypeInfo(ErrorInstanceType, StructureFlags), info());
37     }
38 
39     static NativeErrorPrototype* create(VM& vm, Structure* structure, const String& name)
40     {
41         NativeErrorPrototype* prototype = new (NotNull, allocateCell<NativeErrorPrototype>(vm.heap)) NativeErrorPrototype(vm, structure);
42         prototype->finishCreation(vm, name);
43         return prototype;
44     }
45 };
46 
47 } // namespace JSC
