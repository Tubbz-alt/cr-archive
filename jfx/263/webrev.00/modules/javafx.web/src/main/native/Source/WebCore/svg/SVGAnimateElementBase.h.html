24 #pragma once
25 
26 #include "SVGAnimationElement.h"
27 #include "SVGNames.h"
28 
29 namespace WebCore {
30 
31 class SVGAttributeAnimator;
32 
33 class SVGAnimateElementBase : public SVGAnimationElement {
34     WTF_MAKE_ISO_ALLOCATED(SVGAnimateElementBase);
35 public:
36     bool isDiscreteAnimator() const;
37 
38 protected:
39     SVGAnimateElementBase(const QualifiedName&, Document&);
40 
41     bool hasValidAttributeType() const override;
42     virtual String animateRangeString(const String& string) const { return string; }
43 
44 private:
45     SVGAttributeAnimator* animator() const;
46     SVGAttributeAnimator* animatorIfExists() const { return m_animator.get(); }
47 
48     void setTargetElement(SVGElement*) override;
49     void setAttributeName(const QualifiedName&) override;
50     void resetAnimation() override;
51 
52     bool calculateFromAndToValues(const String& fromString, const String& toString) override;
53     bool calculateFromAndByValues(const String& fromString, const String& byString) override;
54     bool calculateToAtEndOfDurationValue(const String& toAtEndOfDurationString) override;
55 
56     void startAnimation() override;
57     void calculateAnimatedValue(float progress, unsigned repeatCount) override;
58     void applyResultsToTarget() override;
59     void stopAnimation(SVGElement* targetElement) override;
60     Optional<float> calculateDistance(const String& fromString, const String& toString) override;
61 
62     bool hasInvalidCSSAttributeType() const;
63 
64     mutable RefPtr<SVGAttributeAnimator> m_animator;
65     mutable Optional<bool> m_hasInvalidCSSAttributeType;
66 };
67 
68 } // namespace WebCore
69 
70 SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::SVGAnimateElementBase)
71     static bool isType(const WebCore::SVGElement& element)
72     {
73         return element.hasTagName(WebCore::SVGNames::animateTag) || element.hasTagName(WebCore::SVGNames::animateColorTag)
74             || element.hasTagName(WebCore::SVGNames::animateTransformTag) || element.hasTagName(WebCore::SVGNames::setTag);
75     }
76     static bool isType(const WebCore::Node& node) { return is<WebCore::SVGElement>(node) && isType(downcast<WebCore::SVGElement>(node)); }
77 SPECIALIZE_TYPE_TRAITS_END()
