39 namespace WebCore {
40 
41 using FetchResponsePromise = DOMPromiseDeferred<IDLInterface<FetchResponse>>;
42 
43 void DOMWindowFetch::fetch(DOMWindow& window, FetchRequest::Info&& input, FetchRequest::Init&& init, Ref<DeferredPromise>&& deferred)
44 {
45     FetchResponsePromise promise = WTFMove(deferred);
46 
47     auto* document = window.document();
48     if (!document) {
49         promise.reject(InvalidStateError);
50         return;
51     }
52 
53     auto request = FetchRequest::create(*document, WTFMove(input), WTFMove(init));
54     if (request.hasException()) {
55         promise.reject(request.releaseException());
56         return;
57     }
58 
59     FetchResponse::fetch(*document, request.releaseReturnValue().get(), [promise = WTFMove(promise)](ExceptionOr<FetchResponse&>&& result) mutable {
60         promise.settle(WTFMove(result));
61     });
62 }
