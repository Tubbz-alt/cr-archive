 23 
 24 #if USE(TEXTURE_MAPPER_GL)

 26 #include "TextureMapperGLHeaders.h"
 27 #include "TransformationMatrix.h"
 28 #include <wtf/HashMap.h>
 29 #include <wtf/NeverDestroyed.h>
 30 #include <wtf/Ref.h>
 31 #include <wtf/text/AtomStringHash.h>

 33 namespace WebCore {
 34 
 35 #define TEXMAP_DECLARE_VARIABLE(Accessor, Name, Type) \
 36     GLuint Accessor##Location() { \
 37         static NeverDestroyed<const AtomString> name(Name, AtomString::ConstructFromLiteral); \
 38         return getLocation(name.get(), Type); \
 39     }
 40 
 41 #define TEXMAP_DECLARE_UNIFORM(Accessor) TEXMAP_DECLARE_VARIABLE(Accessor, "u_"#Accessor, UniformVariable)
 42 #define TEXMAP_DECLARE_ATTRIBUTE(Accessor) TEXMAP_DECLARE_VARIABLE(Accessor, "a_"#Accessor, AttribVariable)
 43 #define TEXMAP_DECLARE_SAMPLER(Accessor) TEXMAP_DECLARE_VARIABLE(Accessor, "s_"#Accessor, UniformVariable)
 44 
 45 class TextureMapperShaderProgram : public RefCounted<TextureMapperShaderProgram> {
 46 public:
 47     enum Option {
 48         TextureRGB       = 1L << 0,
 49         Rect             = 1L << 1,
 50         SolidColor       = 1L << 2,
 51         Opacity          = 1L << 3,
 52         Antialiasing     = 1L << 5,
 53         GrayscaleFilter  = 1L << 6,
 54         SepiaFilter      = 1L << 7,
 55         SaturateFilter   = 1L << 8,
 56         HueRotateFilter  = 1L << 9,
 57         BrightnessFilter = 1L << 10,
 58         ContrastFilter   = 1L << 11,
 59         InvertFilter     = 1L << 12,
 60         OpacityFilter    = 1L << 13,
 61         BlurFilter       = 1L << 14,
 62         AlphaBlur        = 1L << 15,
 63         ContentTexture   = 1L << 16,
 64         ManualRepeat     = 1L << 17,
 65         TextureYUV       = 1L << 18,
 66         TextureNV12      = 1L << 19,
 67         TextureNV21      = 1L << 20,
 68         TexturePackedYUV = 1L << 21,
 69         TextureExternalOES = 1L << 22,
 70     };
 71 
 72     typedef unsigned Options;
 73 
 74     static Ref<TextureMapperShaderProgram> create(Options);
 75     virtual ~TextureMapperShaderProgram();
 76 
 77     GLuint programID() const { return m_id; }
 78 
 79     TEXMAP_DECLARE_ATTRIBUTE(vertex)
 80 
 81     TEXMAP_DECLARE_UNIFORM(modelViewMatrix)
 82     TEXMAP_DECLARE_UNIFORM(projectionMatrix)
 83     TEXMAP_DECLARE_UNIFORM(textureSpaceMatrix)
 84     TEXMAP_DECLARE_UNIFORM(textureColorSpaceMatrix)
 85     TEXMAP_DECLARE_UNIFORM(opacity)
 86     TEXMAP_DECLARE_UNIFORM(color)
 87     TEXMAP_DECLARE_UNIFORM(expandedQuadEdgesInScreenSpace)
 88     TEXMAP_DECLARE_UNIFORM(yuvToRgb)
 89     TEXMAP_DECLARE_SAMPLER(sampler)
 90     TEXMAP_DECLARE_SAMPLER(samplerY)
 91     TEXMAP_DECLARE_SAMPLER(samplerU)
 92     TEXMAP_DECLARE_SAMPLER(samplerV)
 93     TEXMAP_DECLARE_SAMPLER(mask)
 94 
 95     TEXMAP_DECLARE_UNIFORM(filterAmount)
 96     TEXMAP_DECLARE_UNIFORM(gaussianKernel)
 97     TEXMAP_DECLARE_UNIFORM(blurRadius)
 98     TEXMAP_DECLARE_UNIFORM(shadowOffset)
 99     TEXMAP_DECLARE_SAMPLER(contentTexture)
100     TEXMAP_DECLARE_SAMPLER(externalOESTexture)
101 
102     void setMatrix(GLuint, const TransformationMatrix&);
103 
104 private:
105     TextureMapperShaderProgram(const String& vertexShaderSource, const String& fragmentShaderSource);
106 
107     GLuint m_vertexShader;
108     GLuint m_fragmentShader;
109 
110     enum VariableType { UniformVariable, AttribVariable };
111     GLuint getLocation(const AtomString&, VariableType);
112 
113     GLuint m_id;
114     HashMap<AtomString, GLuint> m_variables;
115 };
116 
117 }
118 #endif // USE(TEXTURE_MAPPER_GL)
