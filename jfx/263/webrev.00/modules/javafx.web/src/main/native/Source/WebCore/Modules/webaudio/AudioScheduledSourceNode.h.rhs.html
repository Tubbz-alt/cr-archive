29 #pragma once
30 
31 #include "ActiveDOMObject.h"
32 #include "AudioNode.h"
33 
34 namespace WebCore {
35 
36 class AudioScheduledSourceNode : public AudioNode, public ActiveDOMObject {
37     WTF_MAKE_ISO_ALLOCATED(AudioScheduledSourceNode);
38 public:
39     // These are the possible states an AudioScheduledSourceNode can be in:
40     //
41     // UNSCHEDULED_STATE - Initial playback state. Created, but not yet scheduled.
42     // SCHEDULED_STATE - Scheduled to play but not yet playing.
43     // PLAYING_STATE - Generating sound.
44     // FINISHED_STATE - Finished generating sound.
45     //
46     // The state can only transition to the next state, except for the FINISHED_STATE which can
47     // never be changed.
48     enum PlaybackState {
49         // These must be defined with the same names and values as in the .idl file.
50         UNSCHEDULED_STATE = 0,
51         SCHEDULED_STATE = 1,
52         PLAYING_STATE = 2,
53         FINISHED_STATE = 3
54     };
55 
56     AudioScheduledSourceNode(AudioContext&, float sampleRate);
57 
58     ExceptionOr<void> startLater(double when);
59     ExceptionOr<void> stopLater(double when);
60 
61     void didBecomeMarkedForDeletion() override;
62 
63     unsigned short playbackState() const { return static_cast<unsigned short>(m_playbackState); }
64     bool isPlayingOrScheduled() const { return m_playbackState == PLAYING_STATE || m_playbackState == SCHEDULED_STATE; }
65     bool hasFinished() const { return m_playbackState == FINISHED_STATE; }
66 
67 protected:
68     // Get frame information for the current time quantum.
69     // We handle the transition into PLAYING_STATE and FINISHED_STATE here,
70     // zeroing out portions of the outputBus which are outside the range of startFrame and endFrame.
71     // Each frame time is relative to the context's currentSampleFrame().
72     // quantumFrameOffset: Offset frame in this time quantum to start rendering.
73     // nonSilentFramesToProcess: Number of frames rendering non-silence (will be <= quantumFrameSize).
74     void updateSchedulingInfo(size_t quantumFrameSize, AudioBus& outputBus, size_t& quantumFrameOffset, size_t& nonSilentFramesToProcess);
75 
76     // Called when we have no more sound to play or the noteOff() time has been reached.
77     virtual void finish();
78 
79     PlaybackState m_playbackState { UNSCHEDULED_STATE };
80 
81     RefPtr<PendingActivity<AudioScheduledSourceNode>> m_pendingActivity;
82     // m_startTime is the time to start playing based on the context's timeline (0 or a time less than the context's current time means "now").
83     double m_startTime { 0 }; // in seconds
84 
85     // m_endTime is the time to stop playing based on the context's timeline (0 or a time less than the context's current time means "now").
86     // If it hasn't been set explicitly, then the sound will not stop playing (if looping) or will stop when the end of the AudioBuffer
87     // has been reached.
88     double m_endTime; // in seconds
89 


90     static const double UnknownTime;





91 };
92 
93 } // namespace WebCore
