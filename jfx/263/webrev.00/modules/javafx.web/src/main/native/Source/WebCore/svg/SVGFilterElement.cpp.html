 41 inline SVGFilterElement::SVGFilterElement(const QualifiedName& tagName, Document& document)
 42     : SVGElement(tagName, document)
 43     , SVGURIReference(this)
 44 {
 45     // Spec: If the x/y attribute is not specified, the effect is as if a value of "-10%" were specified.
 46     // Spec: If the width/height attribute is not specified, the effect is as if a value of "120%" were specified.
 47     ASSERT(hasTagName(SVGNames::filterTag));
 48 
 49     static std::once_flag onceFlag;
 50     std::call_once(onceFlag, [] {
 51         PropertyRegistry::registerProperty<SVGNames::filterUnitsAttr, SVGUnitTypes::SVGUnitType, &SVGFilterElement::m_filterUnits>();
 52         PropertyRegistry::registerProperty<SVGNames::primitiveUnitsAttr, SVGUnitTypes::SVGUnitType, &SVGFilterElement::m_primitiveUnits>();
 53         PropertyRegistry::registerProperty<SVGNames::xAttr, &SVGFilterElement::m_x>();
 54         PropertyRegistry::registerProperty<SVGNames::yAttr, &SVGFilterElement::m_y>();
 55         PropertyRegistry::registerProperty<SVGNames::widthAttr, &SVGFilterElement::m_width>();
 56         PropertyRegistry::registerProperty<SVGNames::heightAttr, &SVGFilterElement::m_height>();
 57     });
 58 }

 60 Ref<SVGFilterElement> SVGFilterElement::create(const QualifiedName& tagName, Document& document)
 61 {
 62     return adoptRef(*new SVGFilterElement(tagName, document));
 63 }

 65 void SVGFilterElement::parseAttribute(const QualifiedName& name, const AtomString& value)
 66 {
 67     SVGParsingError parseError = NoError;
 68 
 69     if (name == SVGNames::filterUnitsAttr) {
 70         SVGUnitTypes::SVGUnitType propertyValue = SVGPropertyTraits<SVGUnitTypes::SVGUnitType>::fromString(value);
 71         if (propertyValue > 0)
 72             m_filterUnits->setBaseValInternal<SVGUnitTypes::SVGUnitType>(propertyValue);
 73     } else if (name == SVGNames::primitiveUnitsAttr) {
 74         SVGUnitTypes::SVGUnitType propertyValue = SVGPropertyTraits<SVGUnitTypes::SVGUnitType>::fromString(value);
 75         if (propertyValue > 0)
 76             m_primitiveUnits->setBaseValInternal<SVGUnitTypes::SVGUnitType>(propertyValue);
 77     } else if (name == SVGNames::xAttr)
 78         m_x->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
 79     else if (name == SVGNames::yAttr)
 80         m_y->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
 81     else if (name == SVGNames::widthAttr)
 82         m_width->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
 83     else if (name == SVGNames::heightAttr)
 84         m_height->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
 85 
 86     reportAttributeParsingError(parseError, name, value);
 87 
 88     SVGElement::parseAttribute(name, value);
 89     SVGURIReference::parseAttribute(name, value);
 90 }
 91 
 92 void SVGFilterElement::svgAttributeChanged(const QualifiedName& attrName)
 93 {
 94     if (PropertyRegistry::isAnimatedLengthAttribute(attrName)) {
 95         InstanceInvalidationGuard guard(*this);
 96         invalidateSVGPresentationAttributeStyle();
 97         return;
 98     }
 99 
100     if (PropertyRegistry::isKnownAttribute(attrName) || SVGURIReference::isKnownAttribute(attrName)) {
101         if (auto* renderer = this->renderer())
102             renderer->setNeedsLayout();
103         return;
104     }
105 
106     SVGElement::svgAttributeChanged(attrName);
107 }
108 
109 void SVGFilterElement::childrenChanged(const ChildChange& change)
110 {
111     SVGElement::childrenChanged(change);
112 
113     if (change.source == ChildChangeSource::Parser)
114         return;
115 
116     if (RenderObject* object = renderer())
117         object->setNeedsLayout();
118 }

125 bool SVGFilterElement::childShouldCreateRenderer(const Node& child) const
126 {
127     if (!child.isSVGElement())
128         return false;
129 
130     const SVGElement& svgElement = downcast<SVGElement>(child);
131 
132     static NeverDestroyed<HashSet<QualifiedName>> allowedChildElementTags;
133     if (allowedChildElementTags.get().isEmpty()) {
134         allowedChildElementTags.get().add(SVGNames::feBlendTag);
135         allowedChildElementTags.get().add(SVGNames::feColorMatrixTag);
136         allowedChildElementTags.get().add(SVGNames::feComponentTransferTag);
137         allowedChildElementTags.get().add(SVGNames::feCompositeTag);
138         allowedChildElementTags.get().add(SVGNames::feConvolveMatrixTag);
139         allowedChildElementTags.get().add(SVGNames::feDiffuseLightingTag);
140         allowedChildElementTags.get().add(SVGNames::feDisplacementMapTag);
141         allowedChildElementTags.get().add(SVGNames::feDistantLightTag);
142         allowedChildElementTags.get().add(SVGNames::feDropShadowTag);
143         allowedChildElementTags.get().add(SVGNames::feFloodTag);
144         allowedChildElementTags.get().add(SVGNames::feFuncATag);
145         allowedChildElementTags.get().add(SVGNames::feFuncBTag);
146         allowedChildElementTags.get().add(SVGNames::feFuncGTag);
147         allowedChildElementTags.get().add(SVGNames::feFuncRTag);
148         allowedChildElementTags.get().add(SVGNames::feGaussianBlurTag);
149         allowedChildElementTags.get().add(SVGNames::feImageTag);
150         allowedChildElementTags.get().add(SVGNames::feMergeTag);
151         allowedChildElementTags.get().add(SVGNames::feMergeNodeTag);
152         allowedChildElementTags.get().add(SVGNames::feMorphologyTag);
153         allowedChildElementTags.get().add(SVGNames::feOffsetTag);
154         allowedChildElementTags.get().add(SVGNames::fePointLightTag);
155         allowedChildElementTags.get().add(SVGNames::feSpecularLightingTag);
156         allowedChildElementTags.get().add(SVGNames::feSpotLightTag);
157         allowedChildElementTags.get().add(SVGNames::feTileTag);
158         allowedChildElementTags.get().add(SVGNames::feTurbulenceTag);
159     }
160 
161     return allowedChildElementTags.get().contains<SVGAttributeHashTranslator>(svgElement.tagQName());
162 }
