117 TextStream& operator<<(TextStream& ts, SelectionPart selectionPart)
118 {
119     switch (selectionPart) {
120     case SelectionBackground: ts << "selection-background"; break;
121     case SelectionForeground: ts << "selection-foreground"; break;
122     }
123     return ts;
124 }

126 TextStream& operator<<(TextStream& ts, ThemeFont themeFont)
127 {
128     switch (themeFont) {
129     case CaptionFont: ts << "caption-font"; break;
130     case IconFont: ts << "icon-font"; break;
131     case MenuFont: ts << "menu-font"; break;
132     case MessageBoxFont: ts << "messagebox-font"; break;
133     case SmallCaptionFont: ts << "small-caption-font"; break;
134     case StatusBarFont: ts << "statusbar-font"; break;
135     case MiniControlFont: ts << "minicontrol-font"; break;
136     case SmallControlFont: ts << "small-control-font"; break;
137     case ControlFont: ts << "control-font"; break;
138     }
139     return ts;
140 }
