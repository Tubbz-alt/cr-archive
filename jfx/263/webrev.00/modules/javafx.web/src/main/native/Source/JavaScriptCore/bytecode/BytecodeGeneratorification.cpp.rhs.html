 49 struct YieldData {
 50     InstructionStream::Offset point { 0 };
 51     VirtualRegister argument { 0 };
 52     FastBitVector liveness;
 53 };
 54 
 55 class BytecodeGeneratorification {
 56 public:
 57     typedef Vector<YieldData> Yields;
 58 
 59     struct GeneratorFrameData {
 60         InstructionStream::Offset m_point;
 61         VirtualRegister m_dst;
 62         VirtualRegister m_scope;
 63         VirtualRegister m_symbolTable;
 64         VirtualRegister m_initialValue;
 65     };
 66 
 67     BytecodeGeneratorification(BytecodeGenerator& bytecodeGenerator, UnlinkedCodeBlockGenerator* codeBlock, InstructionStreamWriter& instructions, SymbolTable* generatorFrameSymbolTable, int generatorFrameSymbolTableIndex)
 68         : m_bytecodeGenerator(bytecodeGenerator)
 69         , m_codeBlock(codeBlock)
 70         , m_instructions(instructions)
 71         , m_graph(m_codeBlock, m_instructions)
 72         , m_generatorFrameSymbolTable(codeBlock->vm(), generatorFrameSymbolTable)
 73         , m_generatorFrameSymbolTableIndex(generatorFrameSymbolTableIndex)
 74     {
 75         for (const auto& instruction : m_instructions) {
 76             switch (instruction->opcodeID()) {
 77             case op_enter: {
 78                 m_enterPoint = instruction.offset();
 79                 break;
 80             }
 81 
 82             case op_yield: {
 83                 auto bytecode = instruction->as<OpYield>();
 84                 unsigned liveCalleeLocalsIndex = bytecode.m_yieldPoint;
 85                 if (liveCalleeLocalsIndex >= m_yields.size())
 86                     m_yields.resize(liveCalleeLocalsIndex + 1);
 87                 YieldData& data = m_yields[liveCalleeLocalsIndex];
 88                 data.point = instruction.offset();
 89                 data.argument = bytecode.m_argument;
 90                 break;
 91             }
 92 
 93             case op_create_generator_frame_environment: {
 94                 auto bytecode = instruction->as<OpCreateGeneratorFrameEnvironment>();
 95                 GeneratorFrameData data;
 96                 data.m_point = instruction.offset();
 97                 data.m_dst = bytecode.m_dst;
 98                 data.m_scope = bytecode.m_scope;
 99                 data.m_symbolTable = bytecode.m_symbolTable;
100                 data.m_initialValue = bytecode.m_initialValue;
101                 m_generatorFrameData = WTFMove(data);
102                 break;
103             }
104 
105             default:
106                 break;



107             }
108         }
109     }
110 
111     struct Storage {
112         Identifier identifier;
113         unsigned identifierIndex;
114         ScopeOffset scopeOffset;
115     };
116 
117     void run();
118 
119     BytecodeGraph& graph() { return m_graph; }
120 
121     const Yields& yields() const
122     {
123         return m_yields;
124     }
125 
126     Yields& yields()
127     {
128         return m_yields;
129     }
130 
131     InstructionStream::Ref enterPoint() const
132     {
133         return m_instructions.at(m_enterPoint);
134     }
135 
136     Optional<GeneratorFrameData> generatorFrameData() const
137     {
138         return m_generatorFrameData;
139     }
140 
141     const InstructionStream& instructions() const
142     {
143         return m_instructions;
144     }
145 
146 private:
147     Storage storageForGeneratorLocal(VM& vm, unsigned index)
148     {
149         // We assign a symbol to a register. There is one-on-one corresponding between a register and a symbol.
150         // By doing so, we allocate the specific storage to save the given register.
151         // This allow us not to save all the live registers even if the registers are not overwritten from the previous resuming time.
152         // It means that, the register can be retrieved even if the immediate previous op_save does not save it.
153 
154         if (m_storages.size() <= index)
155             m_storages.resize(index + 1);
156         if (Optional<Storage> storage = m_storages[index])
157             return *storage;
158 
159         Identifier identifier = Identifier::from(vm, index);
160         unsigned identifierIndex = m_codeBlock->numberOfIdentifiers();
161         m_codeBlock->addIdentifier(identifier);
162         ScopeOffset scopeOffset = m_generatorFrameSymbolTable->takeNextScopeOffset(NoLockingNecessary);
163         m_generatorFrameSymbolTable->set(NoLockingNecessary, identifier.impl(), SymbolTableEntry(VarOffset(scopeOffset)));
164 
165         Storage storage = {
166             identifier,
167             identifierIndex,
168             scopeOffset
169         };
170         m_storages[index] = storage;
171         return storage;
172     }
173 
174     BytecodeGenerator& m_bytecodeGenerator;
175     InstructionStream::Offset m_enterPoint;
176     Optional<GeneratorFrameData> m_generatorFrameData;
177     UnlinkedCodeBlockGenerator* m_codeBlock;
178     InstructionStreamWriter& m_instructions;
179     BytecodeGraph m_graph;
180     Vector<Optional<Storage>> m_storages;
181     Yields m_yields;
182     Strong<SymbolTable> m_generatorFrameSymbolTable;
183     int m_generatorFrameSymbolTableIndex;
184 };
185 
186 class GeneratorLivenessAnalysis : public BytecodeLivenessPropagation {
187 public:
188     GeneratorLivenessAnalysis(BytecodeGeneratorification& generatorification)
189         : m_generatorification(generatorification)
190     {
191     }
192 
193     void run(UnlinkedCodeBlockGenerator* codeBlock, InstructionStreamWriter& instructions)
194     {
195         // Perform modified liveness analysis to determine which locals are live at the merge points.
196         // This produces the conservative results for the question, "which variables should be saved and resumed?".
197 
198         runLivenessFixpoint(codeBlock, instructions, m_generatorification.graph());
199 
200         for (YieldData& data : m_generatorification.yields())
201             data.liveness = getLivenessInfoAtBytecodeIndex(codeBlock, instructions, m_generatorification.graph(), BytecodeIndex(m_generatorification.instructions().at(data.point).next().offset()));
202     }
203 
204 private:
205     BytecodeGeneratorification& m_generatorification;
206 };
207 
208 void BytecodeGeneratorification::run()
209 {
210     // We calculate the liveness at each merge point. This gives us the information which registers should be saved and resumed conservatively.
211 
212     VM& vm = m_bytecodeGenerator.vm();
213     {
214         GeneratorLivenessAnalysis pass(*this);
215         pass.run(m_codeBlock, m_instructions);
216     }
217 
218     BytecodeRewriter rewriter(m_bytecodeGenerator, m_graph, m_codeBlock, m_instructions);
219 
220     // Setup the global switch for the generator.
221     {
222         auto nextToEnterPoint = enterPoint().next();
223         unsigned switchTableIndex = m_codeBlock->numberOfSwitchJumpTables();
224         VirtualRegister state = virtualRegisterForArgumentIncludingThis(static_cast<int32_t>(JSGenerator::GeneratorArgument::State));
225         auto& jumpTable = m_codeBlock->addSwitchJumpTable();
226         jumpTable.min = 0;
227         jumpTable.branchOffsets = RefCountedArray<int32_t>(m_yields.size() + 1);
228         std::fill(jumpTable.branchOffsets.begin(), jumpTable.branchOffsets.end(), 0);
229         jumpTable.add(0, nextToEnterPoint.offset());
230         for (unsigned i = 0; i < m_yields.size(); ++i)
231             jumpTable.add(i + 1, m_yields[i].point);
232 
233         rewriter.insertFragmentBefore(nextToEnterPoint, [&] (BytecodeRewriter::Fragment& fragment) {
234             fragment.appendInstruction<OpSwitchImm>(switchTableIndex, BoundLabel(nextToEnterPoint.offset()), state);
235         });
236     }
237 
238     for (const YieldData& data : m_yields) {
239         VirtualRegister scope = virtualRegisterForArgumentIncludingThis(static_cast<int32_t>(JSGenerator::GeneratorArgument::Frame));
240 
241         auto instruction = m_instructions.at(data.point);
242         // Emit save sequence.
243         rewriter.insertFragmentBefore(instruction, [&] (BytecodeRewriter::Fragment& fragment) {
244             data.liveness.forEachSetBit([&](size_t index) {
245                 VirtualRegister operand = virtualRegisterForLocal(index);
246                 Storage storage = storageForGeneratorLocal(vm, index);
247 
248                 fragment.appendInstruction<OpPutToScope>(
249                     scope, // scope
250                     storage.identifierIndex, // identifier
251                     operand, // value
252                     GetPutInfo(DoNotThrowIfNotFound, LocalClosureVar, InitializationMode::NotInitialization), // info
253                     SymbolTableOrScopeDepth::symbolTable(VirtualRegister { m_generatorFrameSymbolTableIndex }), // symbol table constant index
254                     storage.scopeOffset.offset() // scope offset
255                 );
256             });
257 
258             // Insert op_ret just after save sequence.
259             fragment.appendInstruction<OpRet>(data.argument);
260         });
261 
262         // Emit resume sequence.
263         rewriter.insertFragmentAfter(instruction, [&] (BytecodeRewriter::Fragment& fragment) {
264             data.liveness.forEachSetBit([&](size_t index) {
265                 VirtualRegister operand = virtualRegisterForLocal(index);
266                 Storage storage = storageForGeneratorLocal(vm, index);
267 
268                 fragment.appendInstruction<OpGetFromScope>(
269                     operand, // dst
270                     scope, // scope
271                     storage.identifierIndex, // identifier
272                     GetPutInfo(DoNotThrowIfNotFound, LocalClosureVar, InitializationMode::NotInitialization), // info
273                     0, // local scope depth
274                     storage.scopeOffset.offset() // scope offset
275                 );
276             });
277         });
278 
279         // Clip the unnecessary bytecodes.
280         rewriter.removeBytecode(instruction);
281     }
282 
283     if (m_generatorFrameData) {
284         auto instruction = m_instructions.at(m_generatorFrameData->m_point);
285         rewriter.insertFragmentAfter(instruction, [&] (BytecodeRewriter::Fragment& fragment) {
286             if (!m_generatorFrameSymbolTable->scopeSize()) {
287                 // This will cause us to put jsUndefined() into the generator frame's scope value.
288                 fragment.appendInstruction<OpMov>(m_generatorFrameData->m_dst, m_generatorFrameData->m_initialValue);
289             } else
290                 fragment.appendInstruction<OpCreateLexicalEnvironment>(m_generatorFrameData->m_dst, m_generatorFrameData->m_scope, m_generatorFrameData->m_symbolTable, m_generatorFrameData->m_initialValue);
291         });
292         rewriter.removeBytecode(instruction);
293     }
294 
295     rewriter.execute();
296 }
297 
298 void performGeneratorification(BytecodeGenerator& bytecodeGenerator, UnlinkedCodeBlockGenerator* codeBlock, InstructionStreamWriter& instructions, SymbolTable* generatorFrameSymbolTable, int generatorFrameSymbolTableIndex)
299 {
300     if (UNLIKELY(Options::dumpBytecodesBeforeGeneratorification()))
301         CodeBlockBytecodeDumper<UnlinkedCodeBlockGenerator>::dumpBlock(codeBlock, instructions, WTF::dataFile());
302 
303     BytecodeGeneratorification pass(bytecodeGenerator, codeBlock, instructions, generatorFrameSymbolTable, generatorFrameSymbolTableIndex);
304     pass.run();
305 }
