 24 #[=======================================================================[.rst:
 25 FindLibPSL
 26 --------------
 27 
 28 Find LibPSL headers and libraries.
 29 
 30 Imported Targets
 31 ^^^^^^^^^^^^^^^^
 32 
 33 ``LibPSL::LibPSL``
 34   The LibPSL library, if found.
 35 
 36 Result Variables
 37 ^^^^^^^^^^^^^^^^
 38 
 39 This will define the following variables in your project:
 40 
 41 ``LibPSL_FOUND``
 42   true if (the requested version of) LibPSL is available.
 43 ``LibPSL_VERSION``
 44   the version of LibPSL.
 45 ``LibPSL_LIBRARIES``
 46   the libraries to link against to use LibPSL.
 47 ``LibPSL_INCLUDE_DIRS``
 48   where to find the LibPSL headers.
 49 ``LibPSL_COMPILE_OPTIONS``
 50   this should be passed to target_compile_options(), if the
 51   target is not used for linking
 52 
 53 #]=======================================================================]
 54 
 55 include(FindPkgConfig)
 56 pkg_check_modules(PC_LIBPSL QUIET libpsl)
 57 set(LibPSL_COMPILE_OPTIONS ${PC_LIBPSL_CFLAGS_OTHER})
 58 set(LibPSL_VERSION ${PC_LIBPSL_VERSION})
 59 
 60 find_path(LibPSL_INCLUDE_DIR
 61     NAMES libpsl.h
 62     HINTS ${PC_LIBPSL_INCLUDEDIR} ${PC_LIBPSL_INCLUDE_DIR}
 63     PATH_SUFFIXES libpsl
 64 )
 65 
 66 find_library(LibPSL_LIBRARY
 67     NAMES ${LibPSL_NAMES} psl
 68     HINTS ${PC_LIBPSL_LIBDIR} ${PC_LIBPSL_LIBRARY_DIRS}
 69 )
 70 
 71 if (LibPSL_INCLUDE_DIR AND NOT LibPSL_VERSION)
 72     if (EXISTS "${LibPSL_INCLUDE_DIR}/libpsl.h")
 73         file(READ "${LibPSL_INCLUDE_DIR}/libpsl.h" LIBPSL_VERSION_CONTENT)
 74 
 75         string(REGEX MATCH "#define +PSL_VERSION_MAJOR +([0-9]+)" _dummy "${LIBPSL_VERSION_CONTENT}")
 76         set(LIBPSL_VERSION_MAJOR "${CMAKE_MATCH_1}")
 77 
 78         string(REGEX MATCH "#define +PSL_VERSION_MINOR +([0-9]+)" _dummy "${LIBPSL_VERSION_CONTENT}")
 79         set(LIBPSL_VERSION_MINOR "${CMAKE_MATCH_1}")
 80 
 81         string(REGEX MATCH "#define +PSL_VERSION_PATCH +([0-9]+)" _dummy "${LIBPSL_VERSION_CONTENT}")
 82         set(LIBPSL_VERSION_PATCH "${CMAKE_MATCH_1}")
 83 
 84         set(LIBPSL_VERSION "${LIBPSL_VERSION_MAJOR}.${LIBPSL_VERSION_MINOR}.${LIBPSL_VERSION_PATCH}")
 85     endif ()
 86 endif ()
 87 
 88 include(FindPackageHandleStandardArgs)
 89 find_package_handle_standard_args(LibPSL
 90     FOUND_VAR LibPSL_FOUND
 91     REQUIRED_VARS LibPSL_LIBRARY LibPSL_INCLUDE_DIR
 92     VERSION_VAR LibPSL_VERSION


 93 )
 94 
 95 if (LibPSL_LIBRARY AND NOT TARGET LibPSL::LibPSL)
 96     add_library(LibPSL::LibPSL UNKNOWN IMPORTED GLOBAL)
 97     set_target_properties(LibPSL::LibPSL PROPERTIES
 98         IMPORTED_LOCATION "${LibPSL_LIBRARY}"
 99         INTERFACE_COMPILE_OPTIONS "${LibPSL_COMPILE_OPTIONS}"
100         INTERFACE_INCLUDE_DIRECTORIES "${LibPSL_INCLUDE_DIR}"
101     )
102 endif ()
103 
104 mark_as_advanced(LibPSL_INCLUDE_DIR LIBPSL_LIBRARIES)
105 
106 if (LibPSL_FOUND)
107     set(LibPSL_LIBRARIES ${LibPSL_LIBRARY})
108     set(LibPSL_INCLUDE_DIRS ${LibPSL_INCLUDE_DIR})
109 endif ()
