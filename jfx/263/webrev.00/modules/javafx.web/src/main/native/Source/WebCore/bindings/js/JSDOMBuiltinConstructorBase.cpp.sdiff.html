30 void JSDOMBuiltinConstructorBase::callFunctionWithCurrentArguments(JSC::ExecState& state, JSC::JSObject& thisObject, JSC::JSFunction& function)
31 {
32     JSC::VM& vm = state.vm();
33     auto scope = DECLARE_THROW_SCOPE(vm);
34     JSC::CallData callData;
35     JSC::CallType callType = JSC::getCallData(vm, &function, callData);
36     ASSERT(callType != CallType::None);
37 
38     JSC::MarkedArgumentBuffer arguments;
39     for (unsigned i = 0; i < state.argumentCount(); ++i)
40         arguments.append(state.uncheckedArgument(i));
41     if (UNLIKELY(arguments.hasOverflowed())) {
42         throwOutOfMemoryError(&state, scope);
43         return;
44     }
45     JSC::call(&state, &function, callType, callData, &thisObject, arguments);
46 }

31 void JSDOMBuiltinConstructorBase::callFunctionWithCurrentArguments(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, JSC::JSObject& thisObject, JSC::JSFunction& function)
32 {
33     JSC::VM& vm = lexicalGlobalObject.vm();
34     auto scope = DECLARE_THROW_SCOPE(vm);
35     JSC::CallData callData;
36     JSC::CallType callType = JSC::getCallData(vm, &function, callData);
37     ASSERT(callType != CallType::None);
38 
39     JSC::MarkedArgumentBuffer arguments;
40     for (unsigned i = 0; i < callFrame.argumentCount(); ++i)
41         arguments.append(callFrame.uncheckedArgument(i));
42     if (UNLIKELY(arguments.hasOverflowed())) {
43         throwOutOfMemoryError(&lexicalGlobalObject, scope);
44         return;
45     }
46     JSC::call(&lexicalGlobalObject, &function, callType, callData, &thisObject, arguments);
47 }
