35 void CCallHelpers::logShadowChickenProloguePacket(GPRReg shadowPacket, GPRReg scratch1, GPRReg scope)
36 {
37     storePtr(GPRInfo::callFrameRegister, Address(shadowPacket, OBJECT_OFFSETOF(ShadowChicken::Packet, frame)));
38     loadPtr(Address(GPRInfo::callFrameRegister, OBJECT_OFFSETOF(CallerFrameAndPC, callerFrame)), scratch1);
39     storePtr(scratch1, Address(shadowPacket, OBJECT_OFFSETOF(ShadowChicken::Packet, callerFrame)));
40     loadPtr(addressFor(CallFrameSlot::callee), scratch1);
41     storePtr(scratch1, Address(shadowPacket, OBJECT_OFFSETOF(ShadowChicken::Packet, callee)));
42     storePtr(scope, Address(shadowPacket, OBJECT_OFFSETOF(ShadowChicken::Packet, scope)));
43 }
44 
45 void CCallHelpers::logShadowChickenTailPacket(GPRReg shadowPacket, JSValueRegs thisRegs, GPRReg scope, CodeBlock* codeBlock, CallSiteIndex callSiteIndex)
46 {
47     storePtr(GPRInfo::callFrameRegister, Address(shadowPacket, OBJECT_OFFSETOF(ShadowChicken::Packet, frame)));
48     storePtr(TrustedImmPtr(ShadowChicken::Packet::tailMarker()), Address(shadowPacket, OBJECT_OFFSETOF(ShadowChicken::Packet, callee)));
49     storeValue(thisRegs, Address(shadowPacket, OBJECT_OFFSETOF(ShadowChicken::Packet, thisValue)));
50     storePtr(scope, Address(shadowPacket, OBJECT_OFFSETOF(ShadowChicken::Packet, scope)));
51     storePtr(TrustedImmPtr(codeBlock), Address(shadowPacket, OBJECT_OFFSETOF(ShadowChicken::Packet, codeBlock)));
52     store32(TrustedImm32(callSiteIndex.bits()), Address(shadowPacket, OBJECT_OFFSETOF(ShadowChicken::Packet, callSiteIndex)));
53 }

55 void CCallHelpers::ensureShadowChickenPacket(VM& vm, GPRReg shadowPacket, GPRReg scratch1NonArgGPR, GPRReg scratch2)
56 {
57     ShadowChicken* shadowChicken = vm.shadowChicken();
58     RELEASE_ASSERT(shadowChicken);
59     ASSERT(!RegisterSet::argumentGPRS().get(scratch1NonArgGPR));
60     move(TrustedImmPtr(shadowChicken->addressOfLogCursor()), scratch1NonArgGPR);
61     loadPtr(Address(scratch1NonArgGPR), shadowPacket);
62     Jump ok = branchPtr(Below, shadowPacket, TrustedImmPtr(shadowChicken->logEnd()));
63     setupArguments<decltype(operationProcessShadowChickenLog)>(TrustedImmPtr(&vm));
64     prepareCallOperation(vm);
65     move(TrustedImmPtr(tagCFunctionPtr<OperationPtrTag>(operationProcessShadowChickenLog)), scratch1NonArgGPR);
66     call(scratch1NonArgGPR, OperationPtrTag);
67     move(TrustedImmPtr(shadowChicken->addressOfLogCursor()), scratch1NonArgGPR);
68     loadPtr(Address(scratch1NonArgGPR), shadowPacket);
69     ok.link(this);
70     addPtr(TrustedImm32(sizeof(ShadowChicken::Packet)), shadowPacket, scratch2);
71     storePtr(scratch2, Address(scratch1NonArgGPR));
72 }
