53 JSInternalPromise* JSInternalPromise::then(ExecState* exec, JSFunction* onFulfilled, JSFunction* onRejected)
54 {
55     VM& vm = exec->vm();
56     auto scope = DECLARE_THROW_SCOPE(vm);
57 
58     JSObject* function = jsCast<JSObject*>(get(exec, vm.propertyNames->builtinNames().thenPublicName()));
59     RETURN_IF_EXCEPTION(scope, nullptr);
60     CallData callData;
61     CallType callType = JSC::getCallData(vm, function, callData);
62     ASSERT(callType != CallType::None);
63 
64     MarkedArgumentBuffer arguments;
65     arguments.append(onFulfilled ? onFulfilled : jsUndefined());
66     arguments.append(onRejected ? onRejected : jsUndefined());
67     ASSERT(!arguments.hasOverflowed());
68 
69     RELEASE_AND_RETURN(scope, jsCast<JSInternalPromise*>(call(exec, function, callType, callData, this, arguments)));
70 }
