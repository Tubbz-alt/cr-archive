 32 namespace WebCore {

 34 WTF_MAKE_ISO_ALLOCATED_IMPL(DeviceOrientationEvent);
 35 
 36 DeviceOrientationEvent::~DeviceOrientationEvent() = default;
 37 
 38 DeviceOrientationEvent::DeviceOrientationEvent()
 39     : m_orientation(DeviceOrientationData::create())
 40 {
 41 }
 42 
 43 DeviceOrientationEvent::DeviceOrientationEvent(const AtomString& eventType, DeviceOrientationData* orientation)
 44     : Event(eventType, CanBubble::No, IsCancelable::No)
 45     , m_orientation(orientation)
 46 {
 47 }
 48 
 49 Optional<double> DeviceOrientationEvent::alpha() const
 50 {
 51     return m_orientation->alpha();
 52 }
 53 
 54 Optional<double> DeviceOrientationEvent::beta() const
 55 {
 56     return m_orientation->beta();
 57 }
 58 
 59 Optional<double> DeviceOrientationEvent::gamma() const
 60 {
 61     return m_orientation->gamma();
 62 }
 63 
 64 #if PLATFORM(IOS_FAMILY)
 65 
 66 Optional<double> DeviceOrientationEvent::compassHeading() const
 67 {
 68     return m_orientation->compassHeading();
 69 }
 70 
 71 Optional<double> DeviceOrientationEvent::compassAccuracy() const
 72 {
 73     return m_orientation->compassAccuracy();
 74 }
 75 
 76 void DeviceOrientationEvent::initDeviceOrientationEvent(const AtomString& type, bool bubbles, bool cancelable, Optional<double> alpha, Optional<double> beta, Optional<double> gamma, Optional<double> compassHeading, Optional<double> compassAccuracy)
 77 {
 78     if (isBeingDispatched())
 79         return;
 80 
 81     initEvent(type, bubbles, cancelable);
 82     m_orientation = DeviceOrientationData::create(alpha, beta, gamma, compassHeading, compassAccuracy);
 83 }
 84 
 85 #else
 86 
 87 Optional<bool> DeviceOrientationEvent::absolute() const
 88 {
 89     return m_orientation->absolute();
 90 }
 91 
 92 void DeviceOrientationEvent::initDeviceOrientationEvent(const AtomString& type, bool bubbles, bool cancelable, Optional<double> alpha, Optional<double> beta, Optional<double> gamma, Optional<bool> absolute)
 93 {
 94     if (isBeingDispatched())
 95         return;
 96 
 97     initEvent(type, bubbles, cancelable);
 98     m_orientation = DeviceOrientationData::create(alpha, beta, gamma, absolute);
 99 }
100 
101 #endif
102 
103 EventInterface DeviceOrientationEvent::eventInterface() const
104 {
105 #if ENABLE(DEVICE_ORIENTATION)
106     return DeviceOrientationEventInterfaceType;
107 #else
108     // FIXME: ENABLE(DEVICE_ORIENTATION) seems to be in a strange state where
109     // it is half-guarded by #ifdefs. DeviceOrientationEvent.idl is guarded
110     // but DeviceOrientationEvent.cpp itself is required by ungarded code.
111     return EventInterfaceType;
112 #endif
113 }
