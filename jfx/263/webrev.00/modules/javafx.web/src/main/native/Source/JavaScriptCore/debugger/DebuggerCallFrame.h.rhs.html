29 #pragma once
30 
31 #include "CallFrame.h"
32 #include "DebuggerPrimitives.h"
33 #include "ShadowChicken.h"
34 #include "Strong.h"
35 #include <wtf/NakedPtr.h>
36 #include <wtf/RefCounted.h>
37 #include <wtf/text/TextPosition.h>
38 
39 namespace JSC {
40 
41 class DebuggerScope;
42 class Exception;
43 
44 class DebuggerCallFrame : public RefCounted<DebuggerCallFrame> {
45 public:
46     enum Type { ProgramType, FunctionType };
47 
48     static Ref<DebuggerCallFrame> create(VM&, CallFrame*);
49 
50     JS_EXPORT_PRIVATE RefPtr<DebuggerCallFrame> callerFrame();
51     JSGlobalObject* globalObject();
52     JS_EXPORT_PRIVATE SourceID sourceID() const;
53 
54     // line and column are in base 0 e.g. the first line is line 0.
55     int line() const { return m_position.m_line.zeroBasedInt(); }
56     int column() const { return m_position.m_column.zeroBasedInt(); }
57     JS_EXPORT_PRIVATE const TextPosition& position() const { return m_position; }
58 
59     JS_EXPORT_PRIVATE JSGlobalObject* deprecatedVMEntryGlobalObject() const;
60     JS_EXPORT_PRIVATE DebuggerScope* scope();
61     JS_EXPORT_PRIVATE String functionName() const;
62     JS_EXPORT_PRIVATE Type type() const;
63     JS_EXPORT_PRIVATE JSValue thisValue(VM&) const;
64     JSValue evaluateWithScopeExtension(const String&, JSObject* scopeExtensionObject, NakedPtr<Exception>&);
65 
66     bool isValid() const { return !!m_validMachineFrame || isTailDeleted(); }
67     JS_EXPORT_PRIVATE void invalidate();
68 
69     // The following are only public for the Debugger's use only. They will be
70     // made private soon. Other clients should not use these.
71 
72     JS_EXPORT_PRIVATE TextPosition currentPosition(VM&);
73     JS_EXPORT_PRIVATE static TextPosition positionForCallFrame(VM&, CallFrame*);
74     JS_EXPORT_PRIVATE static SourceID sourceIDForCallFrame(CallFrame*);
75 
76     bool isTailDeleted() const { return m_shadowChickenFrame.isTailDeleted; }
77 
78 private:
79     DebuggerCallFrame(VM&, CallFrame*, const ShadowChicken::Frame&);
80 
81     CallFrame* m_validMachineFrame;
82     RefPtr<DebuggerCallFrame> m_caller;
83     TextPosition m_position;
84     // The DebuggerPausedScope is responsible for calling invalidate() which,
85     // in turn, will clear this strong ref.
86     Strong<DebuggerScope> m_scope;
87     ShadowChicken::Frame m_shadowChickenFrame;
88 };
89 
90 } // namespace JSC
