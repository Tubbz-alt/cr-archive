24 #pragma once
25 
26 #include "Event.h"
27 
28 namespace WebCore {
29 
30 class DataTransfer;
31 
32 class ClipboardEvent final : public Event {
33     WTF_MAKE_ISO_ALLOCATED(ClipboardEvent);
34 public:
35     virtual ~ClipboardEvent();
36 
37     struct Init : EventInit {
38         RefPtr<DataTransfer> clipboardData;
39     };
40 
41     static Ref<ClipboardEvent> create(const AtomString& type, Ref<DataTransfer>&& dataTransfer)
42     {
43         return adoptRef(*new ClipboardEvent(type, WTFMove(dataTransfer)));
44     }
45 
46     static Ref<ClipboardEvent> create(const AtomString& type, const Init& init)
47     {
48         return adoptRef(*new ClipboardEvent(type, init));
49     }
50 
51     DataTransfer* clipboardData() const { return m_clipboardData.get(); }
52 
53 private:
54     ClipboardEvent(const AtomString& type, Ref<DataTransfer>&&);
55     ClipboardEvent(const AtomString& type, const Init&);
56 
57     EventInterface eventInterface() const final;
58     bool isClipboardEvent() const final;
59 
60     RefPtr<DataTransfer> m_clipboardData;
61 };
62 
63 } // namespace WebCore
