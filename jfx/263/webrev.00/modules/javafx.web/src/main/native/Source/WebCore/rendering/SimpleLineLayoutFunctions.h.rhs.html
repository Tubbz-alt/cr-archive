 71 inline LayoutUnit computeFlowHeight(const RenderBlockFlow& flow, const Layout& layout)
 72 {
 73     auto flowHeight = lineHeightFromFlow(flow) * layout.lineCount();
 74     if (!layout.hasLineStruts())
 75         return flowHeight;
 76     for (auto& strutEntry : layout.struts())
 77         flowHeight += strutEntry.offset;
 78     return flowHeight;
 79 }
 80 
 81 inline LayoutUnit computeFlowFirstLineBaseline(const RenderBlockFlow& flow, const Layout& layout)
 82 {
 83     ASSERT_UNUSED(layout, layout.lineCount());
 84     return flow.borderAndPaddingBefore() + baselineFromFlow(flow);
 85 }
 86 
 87 inline LayoutUnit computeFlowLastLineBaseline(const RenderBlockFlow& flow, const Layout& layout)
 88 {
 89     ASSERT(layout.lineCount());
 90     return flow.borderAndPaddingBefore() + lineHeightFromFlow(flow) * (layout.lineCount() - 1) + baselineFromFlow(flow);
 91 }
 92 





































 93 inline LayoutUnit lineHeightFromFlow(const RenderBlockFlow& flow)
 94 {
 95     return flow.lineHeight(false, HorizontalLine, PositionOfInteriorLineBoxes);
 96 }
 97 
 98 inline LayoutUnit baselineFromFlow(const RenderBlockFlow& flow)
 99 {
100     return flow.baselinePosition(AlphabeticBaseline, false, HorizontalLine, PositionOfInteriorLineBoxes);
101 }
