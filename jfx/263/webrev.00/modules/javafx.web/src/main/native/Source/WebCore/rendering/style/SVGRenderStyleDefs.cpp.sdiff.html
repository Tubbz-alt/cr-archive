309 TextStream& operator<<(TextStream& ts, AlignmentBaseline value)
310 {
311     switch (value) {
312     case AlignmentBaseline::Auto: ts << "auto"; break;
313     case AlignmentBaseline::Baseline: ts << "baseline"; break;
314     case AlignmentBaseline::BeforeEdge: ts << "before-edge"; break;
315     case AlignmentBaseline::TextBeforeEdge: ts << "text-before-edge"; break;
316     case AlignmentBaseline::Middle: ts << "middle"; break;
317     case AlignmentBaseline::Central: ts << "central"; break;
318     case AlignmentBaseline::AfterEdge: ts << "after-edge"; break;
319     case AlignmentBaseline::TextAfterEdge: ts << "text-after-edge"; break;
320     case AlignmentBaseline::Ideographic: ts << "ideographic"; break;
321     case AlignmentBaseline::Alphabetic: ts << "alphabetic"; break;
322     case AlignmentBaseline::Hanging: ts << "hanging"; break;
323     case AlignmentBaseline::Mathematical: ts << "mathematical"; break;
324     }
325     return ts;
326 }
