26 #pragma once
27 
28 #include "FloatRect.h"
29 #include <wtf/Vector.h>
30 #include <wtf/text/WTFString.h>
31 
32 namespace WebCore {
33 
34 enum ReasonForDismissingAlternativeText {
35     ReasonForDismissingAlternativeTextCancelled = 0,
36     ReasonForDismissingAlternativeTextIgnored,
37     ReasonForDismissingAlternativeTextAccepted
38 };
39 
40 enum AlternativeTextType {
41     AlternativeTextTypeCorrection = 0,
42     AlternativeTextTypeReversion,
43     AlternativeTextTypeSpellingSuggestions,
44     AlternativeTextTypeDictationAlternatives
45 };
46 
47 enum class AutocorrectionResponse {
48     Edited,
49     Reverted,
50     Accepted
51 };
52 
53 class AlternativeTextClient {
54 public:
55     virtual ~AlternativeTextClient() = default;
56     virtual void pageDestroyed() = 0;
57 #if USE(AUTOCORRECTION_PANEL)
58     virtual void showCorrectionAlternative(AlternativeTextType, const FloatRect& boundingBoxOfReplacedString, const String& replacedString, const String& replacmentString, const Vector<String>& alternativeReplacementStrings) = 0;
59     virtual void dismissAlternative(ReasonForDismissingAlternativeText) = 0;
60     virtual String dismissAlternativeSoon(ReasonForDismissingAlternativeText) = 0;
61     virtual void recordAutocorrectionResponse(AutocorrectionResponse, const String& replacedString, const String& replacementString) = 0;
62 #endif
63 #if USE(DICTATION_ALTERNATIVES)
64     virtual void showDictationAlternativeUI(const WebCore::FloatRect& boundingBoxOfDictatedText, uint64_t dictationContext) = 0;
65     virtual void removeDictationAlternatives(uint64_t dictationContext) = 0;
66     virtual Vector<String> dictationAlternatives(uint64_t dictationContext) = 0;
67 #endif
68 };
69 
70 } // namespace WebCore
