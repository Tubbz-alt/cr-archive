32 #include "CSSParserToken.h"

34 #include "CSSValue.h"

36 #include <wtf/HashSet.h>

38 #include <wtf/text/WTFString.h>
39 
40 namespace WebCore {
41 

42 class CSSParserTokenRange;
43 
44 namespace Style {
45 class BuilderState;
46 }
47 
48 class CSSVariableReferenceValue : public CSSValue {
49 public:
50     static Ref<CSSVariableReferenceValue> create(const CSSParserTokenRange& range)
51     {
52         return adoptRef(*new CSSVariableReferenceValue(CSSVariableData::create(range)));
53     }
54 
55     bool equals(const CSSVariableReferenceValue& other) const { return m_data.get() == other.m_data.get(); }
56     String customCSSText() const;
57 
58     RefPtr<CSSVariableData> resolveVariableReferences(Style::BuilderState&) const;
59 
60 private:
61     CSSVariableReferenceValue(Ref<CSSVariableData>&& data)
62         : CSSValue(VariableReferenceClass)
63         , m_data(WTFMove(data))
64     {
65     }
66 
67     Ref<CSSVariableData> m_data;
68     mutable String m_stringValue;
69     mutable bool m_serialized { false };
70 };
71 
72 } // namespace WebCore
73 
74 SPECIALIZE_TYPE_TRAITS_CSS_VALUE(CSSVariableReferenceValue, isVariableReferenceValue())
