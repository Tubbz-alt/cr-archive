 57 static void appendChild(Container& parent, Box& newChild)
 58 {
 59     if (!parent.hasChild()) {
 60         parent.setFirstChild(newChild);
 61         parent.setLastChild(newChild);
 62         newChild.setParent(parent);
 63         return;
 64     }
 65 
 66     auto& lastChild = const_cast<Box&>(*parent.lastChild());
 67     lastChild.setNextSibling(newChild);
 68     newChild.setPreviousSibling(lastChild);
 69     newChild.setParent(parent);
 70     parent.setLastChild(newChild);
 71 }

 86 static Optional<LayoutSize> accumulatedOffsetForInFlowPositionedContinuation(const RenderBox& block)

 87 {
 88     // FIXE: This is a workaround of the continuation logic when the relatively positioned parent inline box
 89     // becomes a sibling box of this block and only reachable through the continuation link which we don't have here.
 90     if (!block.isAnonymous() || !block.isInFlowPositioned() || !block.isContinuation())
 91         return { };
 92     return block.relativePositionOffset();
 93 }

 95 std::unique_ptr<Box> TreeBuilder::createLayoutBox(const RenderElement& parentRenderer, const RenderObject& childRenderer)





























 96 {
 97     auto elementAttributes = [] (const RenderElement& renderer) -> Optional<Box::ElementAttributes> {
 98         if (renderer.isDocumentElementRenderer())
 99             return Box::ElementAttributes { Box::ElementType::Document };
100         if (auto* element = renderer.element()) {
101             if (element->hasTagName(HTMLNames::bodyTag))
102                 return Box::ElementAttributes { Box::ElementType::Body };
103             if (element->hasTagName(HTMLNames::colTag))
104                 return Box::ElementAttributes { Box::ElementType::TableColumn };
105             if (element->hasTagName(HTMLNames::trTag))
106                 return Box::ElementAttributes { Box::ElementType::TableRow };
107             if (element->hasTagName(HTMLNames::colgroupTag))
108                 return Box::ElementAttributes { Box::ElementType::TableColumnGroup };
109             if (element->hasTagName(HTMLNames::theadTag))
110                 return Box::ElementAttributes { Box::ElementType::TableHeaderGroup };
111             if (element->hasTagName(HTMLNames::tbodyTag))
112                 return Box::ElementAttributes { Box::ElementType::TableBodyGroup };
113             if (element->hasTagName(HTMLNames::tfootTag))
114                 return Box::ElementAttributes { Box::ElementType::TableFooterGroup };
115             if (element->hasTagName(HTMLNames::imgTag))
116                 return Box::ElementAttributes { Box::ElementType::Image };
117             if (element->hasTagName(HTMLNames::iframeTag))
118                 return Box::ElementAttributes { Box::ElementType::IFrame };
119             // FIXME wbr should not be considered as hard linebreak.
120             if (element->hasTagName(HTMLNames::brTag) || element->hasTagName(HTMLNames::wbrTag))
121                 return Box::ElementAttributes { Box::ElementType::HardLineBreak };
122             return Box::ElementAttributes { Box::ElementType::GenericElement };
123         }
124         return WTF::nullopt;
125     };
126 
127     std::unique_ptr<Box> childLayoutBox;
128     if (is<RenderText>(childRenderer)) {
129         // FIXME: Clearly there must be a helper function for this.
130         if (parentRenderer.style().display() == DisplayType::Inline)
131             childLayoutBox = makeUnique<Box>(downcast<RenderText>(childRenderer).originalText(), RenderStyle::clone(parentRenderer.style()));



132         else
133             childLayoutBox = makeUnique<Box>(downcast<RenderText>(childRenderer).originalText(), RenderStyle::createAnonymousStyleWithDisplay(parentRenderer.style(), DisplayType::Inline));
134         childLayoutBox->setIsAnonymous();
135         return childLayoutBox;
136     }
137 
138     auto& renderer = downcast<RenderElement>(childRenderer);
139     auto displayType = renderer.style().display();
140     if (is<RenderLineBreak>(renderer))
141         return makeUnique<Box>(elementAttributes(renderer), RenderStyle::clone(renderer.style()));
142 
143     if (is<RenderTable>(renderer)) {
144         // Construct the principal table wrapper box (and not the table box itself).
145         childLayoutBox = makeUnique<Container>(Box::ElementAttributes { Box::ElementType::TableWrapperBox }, RenderStyle::clone(renderer.style()));
146         childLayoutBox->setIsAnonymous();
147     } else if (is<RenderReplaced>(renderer)) {
148         if (displayType == DisplayType::Block)
149             childLayoutBox = makeUnique<Box>(elementAttributes(renderer), RenderStyle::clone(renderer.style()));
150         else
151             childLayoutBox = makeUnique<Box>(elementAttributes(renderer), RenderStyle::clone(renderer.style()));
152         // FIXME: We don't yet support all replaced elements and this is temporary anyway.
153         if (childLayoutBox->replaced())
154             childLayoutBox->replaced()->setIntrinsicSize(downcast<RenderReplaced>(renderer).intrinsicSize());
155         if (is<RenderImage>(renderer)) {
156             auto& imageRenderer = downcast<RenderImage>(renderer);
157             if (imageRenderer.imageResource().errorOccurred())
158                 childLayoutBox->replaced()->setIntrinsicRatio(1);
159         }
160     } else {
161         if (displayType == DisplayType::Block) {
162             if (auto offset = accumulatedOffsetForInFlowPositionedContinuation(downcast<RenderBox>(renderer))) {
163                 auto style = RenderStyle::clonePtr(renderer.style());
164                 style->setTop({ offset->height(), Fixed });
165                 style->setLeft({ offset->width(), Fixed });
166                 childLayoutBox = makeUnique<Container>(elementAttributes(renderer), WTFMove(*style));
167             } else
168                 childLayoutBox = makeUnique<Container>(elementAttributes(renderer), RenderStyle::clone(renderer.style()));
169         } else if (displayType == DisplayType::Inline)
170             childLayoutBox = makeUnique<Container>(elementAttributes(renderer), RenderStyle::clone(renderer.style()));
171         else if (displayType == DisplayType::InlineBlock)
172             childLayoutBox = makeUnique<Container>(elementAttributes(renderer), RenderStyle::clone(renderer.style()));
173         else if (displayType == DisplayType::TableCaption || displayType == DisplayType::TableCell) {
174             childLayoutBox = makeUnique<Container>(elementAttributes(renderer), RenderStyle::clone(renderer.style()));
175         } else if (displayType == DisplayType::TableRowGroup || displayType == DisplayType::TableHeaderGroup || displayType == DisplayType::TableFooterGroup
176             || displayType == DisplayType::TableRow || displayType == DisplayType::TableColumnGroup || displayType == DisplayType::TableColumn) {
177             childLayoutBox = makeUnique<Container>(elementAttributes(renderer), RenderStyle::clone(renderer.style()));











178         } else {
179             ASSERT_NOT_IMPLEMENTED_YET();
180             return { };


























181         }
182     }
183 
184     if (is<RenderTableCell>(renderer)) {
185         auto& cellElement = downcast<HTMLTableCellElement>(*renderer.element());
186         auto rowSpan = cellElement.rowSpan();
187         if (rowSpan > 1)
188             childLayoutBox->setRowSpan(rowSpan);
189 
190         auto columnSpan = cellElement.colSpan();
191         if (columnSpan > 1)
192             childLayoutBox->setColumnSpan(columnSpan);
193     }

194 
195     if (childRenderer.isAnonymous())
196         childLayoutBox->setIsAnonymous();


197 




198     return childLayoutBox;
199 }

201 void TreeBuilder::createTableStructure(const RenderTable& tableRenderer, Container& tableWrapperBox)
202 {
203     // Create caption and table box.
204     auto* tableChild = tableRenderer.firstChild();
205     while (is<RenderTableCaption>(tableChild)) {
206         auto& captionRenderer = *tableChild;
207         auto captionBox = createLayoutBox(tableRenderer, captionRenderer);
208         appendChild(tableWrapperBox, *captionBox);
209         auto& captionContainer = downcast<Container>(*captionBox);
210         TreeBuilder::createSubTree(downcast<RenderElement>(captionRenderer), captionContainer);
211         // Temporary
212         captionBox.release();
213         tableChild = tableChild->nextSibling();
214     }
215 
216     auto tableBox = makeUnique<Container>(Box::ElementAttributes { Box::ElementType::TableBox }, RenderStyle::clone(tableRenderer.style()));
217     appendChild(tableWrapperBox, *tableBox);
218     auto* sectionRenderer = tableChild;
219     while (sectionRenderer) {
220         auto sectionBox = createLayoutBox(tableRenderer, *sectionRenderer);
221         appendChild(*tableBox, *sectionBox);
222         auto& sectionContainer = downcast<Container>(*sectionBox);
223         TreeBuilder::createSubTree(downcast<RenderElement>(*sectionRenderer), sectionContainer);
224         sectionBox.release();
225         sectionRenderer = sectionRenderer->nextSibling();
226     }
227     // Temporary
228     tableBox.release();
229 }

231 void TreeBuilder::createSubTree(const RenderElement& rootRenderer, Container& rootContainer)
232 {
233     for (auto& childRenderer : childrenOfType<RenderObject>(rootRenderer)) {
234         auto childLayoutBox = createLayoutBox(rootRenderer, childRenderer);
235         appendChild(rootContainer, *childLayoutBox);
236         if (childLayoutBox->isTableWrapperBox())
237             createTableStructure(downcast<RenderTable>(childRenderer), downcast<Container>(*childLayoutBox));
238         else if (is<Container>(*childLayoutBox))
239             createSubTree(downcast<RenderElement>(childRenderer), downcast<Container>(*childLayoutBox));
240         // Temporary
241         childLayoutBox.release();
242     }
243 }

246 static void outputInlineRuns(TextStream& stream, const LayoutState& layoutState, const Container& inlineFormattingRoot, unsigned depth)
247 {
248     auto& inlineFormattingState = downcast<InlineFormattingState>(layoutState.establishedFormattingState(inlineFormattingRoot));
249     auto& inlineRuns = inlineFormattingState.inlineRuns();
250     auto& lineBoxes = inlineFormattingState.lineBoxes();




251 
252     unsigned printedCharacters = 0;
253     while (++printedCharacters <= depth * 3)
254         stream << " ";

255 
256     stream << "lines are -> ";
257     for (auto& lineBox : lineBoxes)
258         stream << "[" << lineBox.logicalLeft() << "," << lineBox.logicalTop() << " " << lineBox.logicalWidth() << "x" << lineBox.logicalHeight() << "] ";
259     stream.nextLine();
260 
261     for (auto& inlineRun : inlineRuns) {
262         unsigned printedCharacters = 0;
263         while (++printedCharacters <= depth * 3)
264             stream << " ";
265         if (inlineRun->textContext())

266             stream << "inline text box";
267         else
268             stream << "inline box";
269         stream << " at (" << inlineRun->logicalLeft() << "," << inlineRun->logicalTop() << ") size " << inlineRun->logicalWidth() << "x" << inlineRun->logicalHeight();
270         if (inlineRun->textContext())
271             stream << " run(" << inlineRun->textContext()->start() << ", " << inlineRun->textContext()->end() << ")";
272         stream.nextLine();
273     }
274 }

276 static void outputLayoutBox(TextStream& stream, const Box& layoutBox, const Display::Box* displayBox, unsigned depth)
277 {
278     unsigned printedCharacters = 0;
279     while (++printedCharacters <= depth * 2)
280         stream << " ";
281 
282     if (layoutBox.isFloatingPositioned())
283         stream << "[float] ";
284 
285     if (layoutBox.isInitialContainingBlock())
286         stream << "Initial containing block";
287     else if (layoutBox.isDocumentBox())
288         stream << "HTML";
289     else if (layoutBox.isBodyBox())
290         stream << "BODY";
291     else if (layoutBox.isTableWrapperBox())
292         stream << "TABLE principal";
293     else if (layoutBox.isTableBox())
294         stream << "TABLE";
295     else if (layoutBox.isTableCaption())
296         stream << "CAPTION";
297     else if (layoutBox.isTableHeader())
298         stream << "THEAD";
299     else if (layoutBox.isTableBody())
300         stream << "TBODY";
301     else if (layoutBox.isTableFooter())
302         stream << "TFOOT";




303     else if (layoutBox.isTableCell())
304         stream << "TD";
305     else if (layoutBox.isTableRow())
306         stream << "TR";
307     else if (layoutBox.isInlineBlockBox())
308         stream << "Inline-block container";
309     else if (layoutBox.isInlineLevelBox()) {
310         if (layoutBox.isInlineContainer())
311             stream << "SPAN inline container";
312         else if (layoutBox.replaced())
313             stream << "IMG replaced inline box";
314         else if (layoutBox.isAnonymous())
315             stream << "anonymous inline box";
316         else if (layoutBox.isLineBreakBox())
317             stream << "BR line break";
318         else
319             stream << "inline box";
320     } else if (layoutBox.isBlockLevelBox())
321         stream << "block box";
322     else
323         stream << "unknown box";
324 
325     // FIXME: Inline text runs don't create display boxes yet.
326     if (displayBox)
327         stream << " at (" << displayBox->left() << "," << displayBox->top() << ") size " << displayBox->width() << "x" << displayBox->height();
328     stream << " layout box->(" << &layoutBox << ")";
329     if (layoutBox.isInlineLevelBox() && layoutBox.isAnonymous())
330         stream << " text content [\"" << layoutBox.textContent().utf8().data() << "\"]";
331 
332     stream.nextLine();
333 }

335 static void outputLayoutTree(const LayoutState* layoutState, TextStream& stream, const Container& rootContainer, unsigned depth)
336 {
337     for (auto& child : childrenOfType<Box>(rootContainer)) {
338         Display::Box* displayBox = nullptr;
339         // Not all boxes generate display boxes.
340         if (layoutState && layoutState->hasDisplayBox(child))
341             displayBox = &layoutState->displayBoxForLayoutBox(child);
342 
343         outputLayoutBox(stream, child, displayBox, depth);
344         if (layoutState && child.establishesInlineFormattingContext())
345             outputInlineRuns(stream, *layoutState, downcast<Container>(child), depth + 1);
346 
347         if (is<Container>(child))
348             outputLayoutTree(layoutState, stream, downcast<Container>(child), depth + 1);
349     }
350 }

352 void showLayoutTree(const Box& layoutBox, const LayoutState* layoutState)
353 {
354     TextStream stream(TextStream::LineMode::MultipleLine, TextStream::Formatting::SVGStyleRect);
355 
356     auto& initialContainingBlock = layoutBox.initialContainingBlock();
357     outputLayoutBox(stream, initialContainingBlock, layoutState ? &layoutState->displayBoxForLayoutBox(initialContainingBlock) : nullptr, 0);
358     outputLayoutTree(layoutState, stream, initialContainingBlock, 1);
359     WTFLogAlways("%s", stream.release().utf8().data());
360 }

107 static bool canUseSimplifiedTextMeasuring(const StringView& content, const FontCascade& font, bool whitespaceIsCollapsed)
108 {
109     if (font.codePath(TextRun(content)) == FontCascade::Complex)
110         return false;
111 
112     if (font.wordSpacing() || font.letterSpacing())
113         return false;
114 
115     for (unsigned i = 0; i < content.length(); ++i) {
116         if ((!whitespaceIsCollapsed && content[i] == '\t') || content[i] == noBreakSpace || content[i] >= HiraganaLetterSmallA)
117             return false;
118     }
119     return true;
120 }

169 Box* TreeBuilder::createLayoutBox(const Container& parentContainer, const RenderObject& childRenderer)
170 {
171     auto elementAttributes = [] (const RenderElement& renderer) -> Optional<Box::ElementAttributes> {
172         if (renderer.isDocumentElementRenderer())
173             return Box::ElementAttributes { Box::ElementType::Document };
174         if (auto* element = renderer.element()) {
175             if (element->hasTagName(HTMLNames::bodyTag))
176                 return Box::ElementAttributes { Box::ElementType::Body };












177             if (element->hasTagName(HTMLNames::imgTag))
178                 return Box::ElementAttributes { Box::ElementType::Image };
179             if (element->hasTagName(HTMLNames::iframeTag))
180                 return Box::ElementAttributes { Box::ElementType::IFrame };
181             // FIXME wbr should not be considered as hard linebreak.
182             if (element->hasTagName(HTMLNames::brTag) || element->hasTagName(HTMLNames::wbrTag))
183                 return Box::ElementAttributes { Box::ElementType::HardLineBreak };
184             return Box::ElementAttributes { Box::ElementType::GenericElement };
185         }
186         return WTF::nullopt;
187     };
188 
189     Box* childLayoutBox = nullptr;
190     if (is<RenderText>(childRenderer)) {
191         auto& textRenderer = downcast<RenderText>(childRenderer);
192         // RenderText::text() has already applied text-transform and text-security properties.
193         String text = textRenderer.text();
194         auto textContent = TextContext { text, canUseSimplifiedTextMeasuring(text, parentContainer.style().fontCascade(), parentContainer.style().collapseWhiteSpace()) };
195         if (parentContainer.style().display() == DisplayType::Inline)
196             childLayoutBox = &createTextBox(WTFMove(textContent), RenderStyle::clone(parentContainer.style()));
197         else
198             childLayoutBox = &createTextBox(WTFMove(textContent), RenderStyle::createAnonymousStyleWithDisplay(parentContainer.style(), DisplayType::Inline));












199         childLayoutBox->setIsAnonymous();













200     } else {
201         auto& renderer = downcast<RenderElement>(childRenderer);
202         auto displayType = renderer.style().display();
203 
204         auto clonedStyle = RenderStyle::clone(renderer.style());
205 
206         if (is<RenderLineBreak>(renderer)) {
207             clonedStyle.setDisplay(DisplayType::Inline);
208             clonedStyle.setFloating(Float::No);
209             childLayoutBox = &createBox(elementAttributes(renderer), WTFMove(clonedStyle));
210         } else if (is<RenderTable>(renderer)) {
211             // Construct the principal table wrapper box (and not the table box itself).
212             childLayoutBox = &createContainer(Box::ElementAttributes { Box::ElementType::TableWrapperBox }, WTFMove(clonedStyle));
213             childLayoutBox->setIsAnonymous();
214         } else if (is<RenderReplaced>(renderer)) {
215             if (displayType == DisplayType::Block)
216                 childLayoutBox = &createBox(elementAttributes(renderer), WTFMove(clonedStyle));
217             else
218                 childLayoutBox = &createBox(elementAttributes(renderer), WTFMove(clonedStyle));
219             // FIXME: We don't yet support all replaced elements and this is temporary anyway.
220             if (childLayoutBox->replaced())
221                 childLayoutBox->replaced()->setIntrinsicSize(downcast<RenderReplaced>(renderer).intrinsicSize());
222             if (is<RenderImage>(renderer)) {
223                 auto& imageRenderer = downcast<RenderImage>(renderer);
224                 if (imageRenderer.shouldDisplayBrokenImageIcon())
225                     childLayoutBox->replaced()->setIntrinsicRatio(1);
226                 if (imageRenderer.cachedImage())
227                     childLayoutBox->replaced()->setCachedImage(*imageRenderer.cachedImage());
228             }
229         } else {
230             if (displayType == DisplayType::Block) {
231                 if (auto offset = accumulatedOffsetForInFlowPositionedContinuation(downcast<RenderBox>(renderer))) {
232                     clonedStyle.setTop({ offset->height(), Fixed });
233                     clonedStyle.setLeft({ offset->width(), Fixed });
234                     childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
235                 } else
236                     childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
237             } else if (displayType == DisplayType::Inline)
238                 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
239             else if (displayType == DisplayType::InlineBlock)
240                 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
241             else if (displayType == DisplayType::TableCaption || displayType == DisplayType::TableCell) {
242                 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
243             } else if (displayType == DisplayType::TableRowGroup || displayType == DisplayType::TableHeaderGroup || displayType == DisplayType::TableFooterGroup
244                 || displayType == DisplayType::TableRow || displayType == DisplayType::TableColumnGroup) {
245                 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
246             } else if (displayType == DisplayType::TableColumn) {
247                 childLayoutBox = &createContainer(elementAttributes(renderer), WTFMove(clonedStyle));
248                 auto& tableColElement = static_cast<HTMLTableColElement&>(*renderer.element());
249                 auto columnWidth = tableColElement.width();
250                 if (!columnWidth.isEmpty())
251                     childLayoutBox->setColumnWidth(columnWidth.toInt());
252                 if (tableColElement.span() > 1)
253                     childLayoutBox->setColumnSpan(tableColElement.span());
254             } else {
255                 ASSERT_NOT_IMPLEMENTED_YET();
256                 return nullptr;
257             }
258         }







259 
260         if (is<RenderTableCell>(renderer)) {
261             auto& cellElement = downcast<HTMLTableCellElement>(*renderer.element());
262             auto rowSpan = cellElement.rowSpan();
263             if (rowSpan > 1)
264                 childLayoutBox->setRowSpan(rowSpan);
265 
266             auto columnSpan = cellElement.colSpan();
267             if (columnSpan > 1)
268                 childLayoutBox->setColumnSpan(columnSpan);
269         }
270 
271         if (childRenderer.isAnonymous())
272             childLayoutBox->setIsAnonymous();
273     }
274     m_layoutTreeContent.addLayoutBoxForRenderer(childRenderer, *childLayoutBox);
275     return childLayoutBox;
276 }

278 void TreeBuilder::buildTableStructure(const RenderTable& tableRenderer, Container& tableWrapperBox)
279 {
280     // Create caption and table box.
281     auto* tableChild = tableRenderer.firstChild();
282     while (is<RenderTableCaption>(tableChild)) {
283         auto& captionRenderer = *tableChild;
284         auto* captionBox = createLayoutBox(tableWrapperBox, captionRenderer);
285         appendChild(tableWrapperBox, *captionBox);
286         auto& captionContainer = downcast<Container>(*captionBox);
287         buildSubTree(downcast<RenderElement>(captionRenderer), captionContainer);


288         tableChild = tableChild->nextSibling();
289     }
290 
291     auto& tableBox = createContainer(Box::ElementAttributes { Box::ElementType::TableBox }, RenderStyle::clone(tableRenderer.style()));
292     appendChild(tableWrapperBox, tableBox);
293     auto* sectionRenderer = tableChild;
294     while (sectionRenderer) {
295         auto* sectionBox = createLayoutBox(tableBox, *sectionRenderer);
296         appendChild(tableBox, *sectionBox);
297         auto& sectionContainer = downcast<Container>(*sectionBox);
298         buildSubTree(downcast<RenderElement>(*sectionRenderer), sectionContainer);

299         sectionRenderer = sectionRenderer->nextSibling();
300     }


301 }
302 
303 void TreeBuilder::buildSubTree(const RenderElement& parentRenderer, Container& parentContainer)
304 {
305     for (auto& childRenderer : childrenOfType<RenderObject>(parentRenderer)) {
306         auto* childLayoutBox = createLayoutBox(parentContainer, childRenderer);
307         appendChild(parentContainer, *childLayoutBox);
308         if (childLayoutBox->isTableWrapperBox())
309             buildTableStructure(downcast<RenderTable>(childRenderer), downcast<Container>(*childLayoutBox));
310         else if (is<Container>(*childLayoutBox))
311             buildSubTree(downcast<RenderElement>(childRenderer), downcast<Container>(*childLayoutBox));


312     }
313 }

316 static void outputInlineRuns(TextStream& stream, const LayoutState& layoutState, const Container& inlineFormattingRoot, unsigned depth)
317 {
318     auto& inlineFormattingState = layoutState.establishedInlineFormattingState(inlineFormattingRoot);
319     auto* displayInlineContent = inlineFormattingState.displayInlineContent();
320     if (!displayInlineContent)
321         return;
322 
323     auto& displayRuns = displayInlineContent->runs;
324     auto& lineBoxes = displayInlineContent->lineBoxes;
325 
326     unsigned printedCharacters = 0;
327     while (++printedCharacters <= depth * 2)
328         stream << " ";
329     stream << "  ";
330 
331     stream << "lines are -> ";
332     for (auto& lineBox : lineBoxes)
333         stream << "[" << lineBox.left() << "," << lineBox.top() << " " << lineBox.width() << "x" << lineBox.height() << "] ";
334     stream.nextLine();
335 
336     for (auto& displayRun : displayRuns) {
337         unsigned printedCharacters = 0;
338         while (++printedCharacters <= depth * 2)
339             stream << " ";
340         stream << "  ";
341         if (displayRun.textContext())
342             stream << "inline text box";
343         else
344             stream << "inline box";
345         stream << " at (" << displayRun.left() << "," << displayRun.top() << ") size " << displayRun.width() << "x" << displayRun.height();
346         if (displayRun.textContext())
347             stream << " run(" << displayRun.textContext()->start() << ", " << displayRun.textContext()->end() << ")";
348         stream.nextLine();
349     }
350 }

352 static void outputLayoutBox(TextStream& stream, const Box& layoutBox, const Display::Box* displayBox, unsigned depth)
353 {
354     unsigned printedCharacters = 0;
355     while (++printedCharacters <= depth * 2)
356         stream << " ";
357 
358     if (layoutBox.isFloatingPositioned())
359         stream << "[float] ";
360 
361     if (layoutBox.isInitialContainingBlock())
362         stream << "Initial containing block";
363     else if (layoutBox.isDocumentBox())
364         stream << "HTML";
365     else if (layoutBox.isBodyBox())
366         stream << "BODY";
367     else if (layoutBox.isTableWrapperBox())
368         stream << "TABLE principal";
369     else if (layoutBox.isTableBox())
370         stream << "TABLE";
371     else if (layoutBox.isTableCaption())
372         stream << "CAPTION";
373     else if (layoutBox.isTableHeader())
374         stream << "THEAD";
375     else if (layoutBox.isTableBody())
376         stream << "TBODY";
377     else if (layoutBox.isTableFooter())
378         stream << "TFOOT";
379     else if (layoutBox.isTableColumnGroup())
380         stream << "COL GROUP";
381     else if (layoutBox.isTableColumn())
382         stream << "COL";
383     else if (layoutBox.isTableCell())
384         stream << "TD";
385     else if (layoutBox.isTableRow())
386         stream << "TR";
387     else if (layoutBox.isInlineBlockBox())
388         stream << "Inline-block";
389     else if (layoutBox.isInlineLevelBox()) {
390         if (layoutBox.isInlineBox())
391             stream << "SPAN inline box";
392         else if (layoutBox.replaced())
393             stream << "IMG replaced inline box";
394         else if (layoutBox.isAnonymous())
395             stream << "anonymous inline box";
396         else if (layoutBox.isLineBreakBox())
397             stream << "BR line break";
398         else
399             stream << "other inline level box";
400     } else if (layoutBox.isBlockLevelBox())
401         stream << "block box";
402     else
403         stream << "unknown box";
404 
405     // FIXME: Inline text runs don't create display boxes yet.
406     if (displayBox)
407         stream << " at (" << displayBox->left() << "," << displayBox->top() << ") size " << displayBox->width() << "x" << displayBox->height();
408     stream << " layout box->(" << &layoutBox << ")";
409     if (layoutBox.isInlineLevelBox() && layoutBox.isAnonymous())
410         stream << " text content [\"" << layoutBox.textContext()->content.utf8().data() << "\"]";
411 
412     stream.nextLine();
413 }

415 static void outputLayoutTree(const LayoutState* layoutState, TextStream& stream, const Container& rootContainer, unsigned depth)
416 {
417     for (auto& child : childrenOfType<Box>(rootContainer)) {
418         if (layoutState) {
419             // Not all boxes generate display boxes.
420             if (layoutState->hasDisplayBox(child))
421                 outputLayoutBox(stream, child, &layoutState->displayBoxForLayoutBox(child), depth);
422             if (child.establishesInlineFormattingContext())
423                 outputInlineRuns(stream, *layoutState, downcast<Container>(child), depth + 1);
424         } else
425             outputLayoutBox(stream, child, nullptr, depth);
426 
427         if (is<Container>(child))
428             outputLayoutTree(layoutState, stream, downcast<Container>(child), depth + 1);
429     }
430 }

432 void showLayoutTree(const Box& layoutBox, const LayoutState* layoutState)
433 {
434     TextStream stream(TextStream::LineMode::MultipleLine, TextStream::Formatting::SVGStyleRect);
435 
436     auto& initialContainingBlock = layoutBox.initialContainingBlock();
437     outputLayoutBox(stream, initialContainingBlock, layoutState ? &layoutState->displayBoxForLayoutBox(initialContainingBlock) : nullptr, 0);
438     outputLayoutTree(layoutState, stream, initialContainingBlock, 1);
439     WTFLogAlways("%s", stream.release().utf8().data());
440 }
