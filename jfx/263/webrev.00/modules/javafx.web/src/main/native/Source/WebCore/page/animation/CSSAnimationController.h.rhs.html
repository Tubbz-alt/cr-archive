 39 class CSSAnimationControllerPrivate;
 40 class Document;

 41 class Frame;
 42 class LayoutRect;
 43 class RenderElement;
 44 
 45 struct AnimationUpdate {
 46     std::unique_ptr<RenderStyle> style;
 47     OptionSet<AnimationImpact> impact;
 48 
 49 };
 50 
 51 class CSSAnimationController {
 52     WTF_MAKE_FAST_ALLOCATED;
 53 public:
 54     explicit CSSAnimationController(Frame&);
 55     ~CSSAnimationController();
 56 
 57     void cancelAnimations(Element&);
 58     AnimationUpdate updateAnimations(Element&, const RenderStyle& newStyle, const RenderStyle* oldStyle);
 59     std::unique_ptr<RenderStyle> animatedStyleForRenderer(RenderElement&);
 60 
 61     // If possible, compute the visual extent of any transform animation on the given renderer
 62     // using the given rect, returning the result in the rect. Return false if there is some
 63     // transform animation but we were unable to cheaply compute its affect on the extent.
 64     bool computeExtentOfAnimation(RenderElement&, LayoutRect&) const;
 65 
 66     // This is called when an accelerated animation or transition has actually started to animate.
 67     void notifyAnimationStarted(RenderElement&, MonotonicTime startTime);
 68 
 69     WEBCORE_EXPORT bool pauseAnimationAtTime(Element&, const AtomString& name, double t); // To be used only for testing
 70     WEBCORE_EXPORT bool pauseTransitionAtTime(Element&, const String& property, double t); // To be used only for testing
 71     WEBCORE_EXPORT unsigned numberOfActiveAnimations(Document*) const; // To be used only for testing
 72 
 73     // "Running" here means the animation is running or paused.
 74     bool isRunningAnimationOnRenderer(RenderElement&, CSSPropertyID) const;
 75     bool isRunningAcceleratedAnimationOnRenderer(RenderElement&, CSSPropertyID) const;
 76 
 77     WEBCORE_EXPORT bool isSuspended() const;
 78     WEBCORE_EXPORT void suspendAnimations();
 79     WEBCORE_EXPORT void resumeAnimations();
 80     void serviceAnimations();
 81 
 82     void updateThrottlingState();
 83     WEBCORE_EXPORT Seconds animationInterval() const;
 84 
 85     WEBCORE_EXPORT void suspendAnimationsForDocument(Document*);
 86     WEBCORE_EXPORT void resumeAnimationsForDocument(Document*);
 87     WEBCORE_EXPORT bool animationsAreSuspendedForDocument(Document*);
 88     void detachFromDocument(Document*);
 89     void startAnimationsIfNotSuspended(Document*);
 90 
 91     void beginAnimationUpdate();
 92     void endAnimationUpdate();
 93 
 94     WEBCORE_EXPORT bool allowsNewAnimationsWhileSuspended() const;
 95     WEBCORE_EXPORT void setAllowsNewAnimationsWhileSuspended(bool);
 96 
 97     static bool supportsAcceleratedAnimationOfProperty(CSSPropertyID);
 98 
 99     bool hasAnimations() const;
100 
101 private:
102     const std::unique_ptr<CSSAnimationControllerPrivate> m_data;
103 };
104 
105 class AnimationUpdateBlock {
106 public:
107     AnimationUpdateBlock(CSSAnimationController* animationController)
108         : m_animationController(animationController)
109     {
110         if (m_animationController)
111             m_animationController->beginAnimationUpdate();
112     }
113 
114     ~AnimationUpdateBlock()
115     {
116         if (m_animationController)
117             m_animationController->endAnimationUpdate();
118     }
119 
120     CSSAnimationController* m_animationController;
121 };
