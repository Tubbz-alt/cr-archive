 37 Ref<CustomPaintCanvas> CustomPaintCanvas::create(ScriptExecutionContext& context, unsigned width, unsigned height)
 38 {
 39     return adoptRef(*new CustomPaintCanvas(context, width, height));
 40 }
 41 
 42 CustomPaintCanvas::CustomPaintCanvas(ScriptExecutionContext& context, unsigned width, unsigned height)
 43     : CanvasBase(IntSize(width, height))
 44     , ContextDestructionObserver(&context)
 45 {
 46 }
 47 
 48 CustomPaintCanvas::~CustomPaintCanvas()
 49 {
 50     notifyObserversCanvasDestroyed();
 51 
 52     m_context = nullptr; // Ensure this goes away before the ImageBuffer.
 53     setImageBuffer(nullptr);
 54 }
 55 
 56 ExceptionOr<RefPtr<PaintRenderingContext2D>> CustomPaintCanvas::getContext()
 57 {
 58     if (m_context)
 59         return { RefPtr<PaintRenderingContext2D> { &downcast<PaintRenderingContext2D>(*m_context) } };
 60 
 61     m_context = PaintRenderingContext2D::create(*this);
 62     downcast<PaintRenderingContext2D>(*m_context).setUsesDisplayListDrawing(true);
 63     downcast<PaintRenderingContext2D>(*m_context).setTracksDisplayListReplay(false);
 64 
 65     return { RefPtr<PaintRenderingContext2D> { &downcast<PaintRenderingContext2D>(*m_context) } };
 66 }
 67 
 68 void CustomPaintCanvas::replayDisplayList(GraphicsContext* ctx) const
 69 {
 70     ASSERT(!m_destinationGraphicsContext);
 71     if (!width() || !height())
 72         return;
 73 
 74     // FIXME: Using an intermediate buffer is not needed if there are no composite operations.
 75     auto clipBounds = ctx->clipBounds();
 76 
 77     auto image = ImageBuffer::createCompatibleBuffer(clipBounds.size(), *ctx);
 78     if (!image)
 79         return;
 80 
 81     m_destinationGraphicsContext = &image->context();
 82     m_destinationGraphicsContext->translate(-clipBounds.location());
 83     if (m_context)
 84         m_context->paintRenderingResultsToCanvas();
 85     m_destinationGraphicsContext = nullptr;
 86 
 87     ctx->drawImageBuffer(*image, clipBounds);
 88 }
 89 
 90 Image* CustomPaintCanvas::copiedImage() const
 91 {
 92     ASSERT(!m_destinationGraphicsContext);
 93     if (!width() || !height())
 94         return nullptr;
 95 
 96     m_copiedBuffer = ImageBuffer::create(size(), RenderingMode::Unaccelerated, 1, ColorSpace::SRGB, nullptr);
 97     if (!m_copiedBuffer)
 98         return nullptr;
 99 
100     m_destinationGraphicsContext = &m_copiedBuffer->context();
101     if (m_context)
102         m_context->paintRenderingResultsToCanvas();
103     m_destinationGraphicsContext = nullptr;
104 
105     m_copiedImage = m_copiedBuffer->copyImage(DontCopyBackingStore, PreserveResolution::Yes);
106     return m_copiedImage.get();
107 }
108 
109 GraphicsContext* CustomPaintCanvas::drawingContext() const
110 {
111     return m_destinationGraphicsContext;
112 }
113 
114 GraphicsContext* CustomPaintCanvas::existingDrawingContext() const
115 {
116     return drawingContext();
