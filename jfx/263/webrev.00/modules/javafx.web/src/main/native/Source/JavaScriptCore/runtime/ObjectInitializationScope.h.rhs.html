26 #pragma once
27 
28 #include "DeferGC.h"
29 #include "DisallowVMReentry.h"
30 #include "VM.h"
31 
32 namespace JSC {
33 
34 class VM;
35 class JSObject;
36 
37 #ifdef NDEBUG
38 
39 class ObjectInitializationScope {
40 public:
41     ALWAYS_INLINE ObjectInitializationScope(VM& vm)
42         : m_vm(vm)
43     { }
44     ALWAYS_INLINE ~ObjectInitializationScope()
45     {
46         m_vm.heap.mutatorFence();
47     }
48 
49     ALWAYS_INLINE VM& vm() const { return m_vm; }
50     ALWAYS_INLINE void notifyAllocated(JSObject*, bool) { }
51     ALWAYS_INLINE void notifyInitialized(JSObject*) { }
52 
53 private:
54     VM& m_vm;
55 };
56 
57 #else // not NDEBUG
58 
59 class ObjectInitializationScope {
60 public:
61     JS_EXPORT_PRIVATE ObjectInitializationScope(VM&);
62     JS_EXPORT_PRIVATE ~ObjectInitializationScope();
63 
64     VM& vm() const { return m_vm; }
65     void notifyAllocated(JSObject*, bool wasCreatedUninitialized);
66     void notifyInitialized(JSObject*);
67 
68 private:
69     void verifyPropertiesAreInitialized(JSObject*);
70 
71     VM& m_vm;
72     DisallowGC m_disallowGC;
73     DisallowVMReentry m_disallowVMReentry;
74     JSObject* m_object { nullptr };
75 };
76 
77 #endif // NDEBUG
78 
79 } // namespace JSC
