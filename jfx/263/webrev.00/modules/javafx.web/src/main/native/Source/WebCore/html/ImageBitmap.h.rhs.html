 27 
 28 #include "IDLTypes.h"
 29 #include "ScriptWrappable.h"
 30 #include <wtf/RefCounted.h>

 32 namespace JSC {
 33 class ArrayBuffer;
 34 }
 35 
 36 using JSC::ArrayBuffer;
 37 
 38 namespace WebCore {
 39 
 40 class Blob;
 41 class CanvasBase;
 42 class HTMLCanvasElement;
 43 class HTMLImageElement;
 44 class HTMLVideoElement;
 45 class ImageBitmapImageObserver;
 46 class ImageBuffer;
 47 class ImageData;
 48 class IntRect;
 49 class IntSize;
 50 #if ENABLE(OFFSCREEN_CANVAS)
 51 class OffscreenCanvas;
 52 #endif
 53 class PendingImageBitmap;
 54 class ScriptExecutionContext;
 55 class TypedOMCSSImageValue;
 56 struct ImageBitmapOptions;
 57 
 58 template<typename IDLType> class DOMPromiseDeferred;
 59 
 60 class ImageBitmap final : public ScriptWrappable, public RefCounted<ImageBitmap> {
 61     WTF_MAKE_ISO_ALLOCATED(ImageBitmap);
 62 public:
 63     using Source = Variant<
 64         RefPtr<HTMLImageElement>,
 65 #if ENABLE(VIDEO)
 66         RefPtr<HTMLVideoElement>,
 67 #endif
 68         RefPtr<HTMLCanvasElement>,
 69         RefPtr<ImageBitmap>,
 70 #if ENABLE(OFFSCREEN_CANVAS)
 71         RefPtr<OffscreenCanvas>,
 72 #endif
 73 #if ENABLE(CSS_TYPED_OM)
 74         RefPtr<TypedOMCSSImageValue>,
 75 #endif
 76         RefPtr<Blob>,
 77         RefPtr<ImageData>
 78     >;
 79 
 80     using Promise = DOMPromiseDeferred<IDLInterface<ImageBitmap>>;
 81 
 82     static void createPromise(ScriptExecutionContext&, Source&&, ImageBitmapOptions&&, Promise&&);
 83     static void createPromise(ScriptExecutionContext&, Source&&, ImageBitmapOptions&&, int sx, int sy, int sw, int sh, Promise&&);
 84 
 85     static Ref<ImageBitmap> create(IntSize);
 86     static Ref<ImageBitmap> create(std::pair<std::unique_ptr<ImageBuffer>, bool>&&);
 87 
 88     ~ImageBitmap();
 89 
 90     unsigned width() const;
 91     unsigned height() const;
 92     void close();
 93 
 94     bool isDetached() const { return m_detached; }
 95 
 96     ImageBuffer* buffer() { return m_bitmapData.get(); }
 97 
 98     bool originClean() const { return m_originClean; }
 99 
100     std::unique_ptr<ImageBuffer> transferOwnershipAndClose();
101 
102     static Vector<std::pair<std::unique_ptr<ImageBuffer>, bool>> detachBitmaps(Vector<RefPtr<ImageBitmap>>&&);
103 
104 private:
105     friend class ImageBitmapImageObserver;
106     friend class PendingImageBitmap;
107 
108     static Ref<ImageBitmap> create(std::unique_ptr<ImageBuffer>&&);
109     ImageBitmap(std::unique_ptr<ImageBuffer>&&);
110 
111     static void createPromise(ScriptExecutionContext&, RefPtr<HTMLImageElement>&, ImageBitmapOptions&&, Optional<IntRect>, Promise&&);
112 #if ENABLE(VIDEO)
113     static void createPromise(ScriptExecutionContext&, RefPtr<HTMLVideoElement>&, ImageBitmapOptions&&, Optional<IntRect>, Promise&&);
114 #endif

115     static void createPromise(ScriptExecutionContext&, RefPtr<ImageBitmap>&, ImageBitmapOptions&&, Optional<IntRect>, Promise&&);
116     static void createPromise(ScriptExecutionContext&, RefPtr<HTMLCanvasElement>&, ImageBitmapOptions&&, Optional<IntRect>, Promise&&);
117 #if ENABLE(OFFSCREEN_CANVAS)
118     static void createPromise(ScriptExecutionContext&, RefPtr<OffscreenCanvas>&, ImageBitmapOptions&&, Optional<IntRect>, Promise&&);
119 #endif
120     static void createPromise(ScriptExecutionContext&, CanvasBase&, ImageBitmapOptions&&, Optional<IntRect>, Promise&&);
121     static void createPromise(ScriptExecutionContext&, RefPtr<Blob>&, ImageBitmapOptions&&, Optional<IntRect>, Promise&&);
122     static void createPromise(ScriptExecutionContext&, RefPtr<ImageData>&, ImageBitmapOptions&&, Optional<IntRect>, Promise&&);
123     static void createPromise(ScriptExecutionContext&, RefPtr<TypedOMCSSImageValue>&, ImageBitmapOptions&&, Optional<IntRect>, Promise&&);
124     static void createFromBuffer(Ref<ArrayBuffer>&&, String mimeType, long long expectedContentLength, const URL&, ImageBitmapOptions&&, Optional<IntRect>, Promise&&);
125 
126     std::unique_ptr<ImageBuffer> m_bitmapData;
127     bool m_detached { false };
128     bool m_originClean { true };
129 };

131 }
