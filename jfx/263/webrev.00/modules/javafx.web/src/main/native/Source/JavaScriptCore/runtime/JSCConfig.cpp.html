38 namespace JSC {

40 alignas(ConfigSizeToProtect) JS_EXPORT_PRIVATE Config g_jscConfig;
41 
42 void Config::disableFreezingForTesting()
43 {
44     RELEASE_ASSERT(!g_jscConfig.isPermanentlyFrozen);
45     g_jscConfig.disabledFreezingForTesting = true;
46 }
47 
48 void Config::enableRestrictedOptions()
49 {
50     RELEASE_ASSERT(!g_jscConfig.isPermanentlyFrozen);
51     g_jscConfig.restrictedOptionsEnabled = true;
52 }
53 
54 void Config::permanentlyFreeze()
55 {
56     RELEASE_ASSERT(roundUpToMultipleOf(pageSize(), ConfigSizeToProtect) == ConfigSizeToProtect);
57 
58     if (!g_jscConfig.isPermanentlyFrozen)
59         g_jscConfig.isPermanentlyFrozen = true;
60 
61     int result = 0;
62 #if OS(DARWIN)
63     enum {
64         AllowPermissionChangesAfterThis = false,
65         DisallowPermissionChangesAfterThis = true
66     };
67 
68     // There's no going back now!
69     result = vm_protect(mach_task_self(), reinterpret_cast<vm_address_t>(&g_jscConfig), ConfigSizeToProtect, DisallowPermissionChangesAfterThis, VM_PROT_READ);
70 #elif OS(LINUX)
71     result = mprotect(&g_jscConfig, ConfigSizeToProtect, PROT_READ);
72 #elif OS(WINDOWS)
73     // FIXME: Implement equivalent, maybe with VirtualProtect.
74     // Also need to fix WebKitTestRunner.
75 #endif
76     RELEASE_ASSERT(!result);
77     RELEASE_ASSERT(g_jscConfig.isPermanentlyFrozen);
78 }
