 39 JSValue Global::get() const
 40 {
 41     switch (m_type) {
 42     case Wasm::Type::I32:
 43         return jsNumber(bitwise_cast<int32_t>(static_cast<uint32_t>(m_value.m_primitive)));
 44     case Wasm::Type::F32:
 45         return jsNumber(purifyNaN(static_cast<double>(bitwise_cast<float>(static_cast<uint32_t>(m_value.m_primitive)))));
 46     case Wasm::Type::F64:
 47         return jsNumber(purifyNaN(bitwise_cast<double>(m_value.m_primitive)));
 48     case Wasm::Anyref:
 49     case Wasm::Funcref:
 50         return m_value.m_anyref.get();
 51     default:
 52         return jsUndefined();
 53     }
 54 }
 55 
 56 void Global::set(JSGlobalObject* globalObject, JSValue argument)
 57 {
 58     VM& vm = globalObject->vm();
 59     auto throwScope = DECLARE_THROW_SCOPE(vm);
 60     ASSERT(m_mutability != Wasm::GlobalInformation::Immutable);
 61     switch (m_type) {
 62     case Wasm::Type::I32: {
 63         int32_t value = argument.toInt32(globalObject);
 64         RETURN_IF_EXCEPTION(throwScope, void());
 65         m_value.m_primitive = static_cast<uint64_t>(static_cast<uint32_t>(value));
 66         break;
 67     }
 68     case Wasm::Type::F32: {
 69         float value = argument.toFloat(globalObject);
 70         RETURN_IF_EXCEPTION(throwScope, void());
 71         m_value.m_primitive = static_cast<uint64_t>(bitwise_cast<uint32_t>(value));
 72         break;
 73     }
 74     case Wasm::Type::F64: {
 75         double value = argument.toNumber(globalObject);
 76         RETURN_IF_EXCEPTION(throwScope, void());
 77         m_value.m_primitive = bitwise_cast<uint64_t>(value);
 78         break;
 79     }
 80     case Wasm::Anyref: {
 81         RELEASE_ASSERT(m_owner);
 82         m_value.m_anyref.set(m_owner->vm(), m_owner, argument);
 83         break;
 84     }
 85     case Wasm::Funcref: {
 86         RELEASE_ASSERT(m_owner);
 87         if (!isWebAssemblyHostFunction(vm, argument) && !argument.isNull()) {
 88             throwException(globalObject, throwScope, createJSWebAssemblyRuntimeError(globalObject, vm, "Funcref must be an exported wasm function"));
 89             return;
 90         }
 91         m_value.m_anyref.set(m_owner->vm(), m_owner, argument);
 92         break;
 93     }
 94     default:
 95         break;
 96     }
 97 }
 98 
 99 void Global::visitAggregate(SlotVisitor& visitor)
100 {
101     switch (m_type) {
102     case Wasm::Type::Anyref:
103     case Wasm::Type::Funcref: {
104         RELEASE_ASSERT(m_owner);
105         visitor.append(m_value.m_anyref);
106         break;
107     }
108     default:
109         break;
110     }
111 }
