58 bool WebGPUComputePipeline::cloneShaderModules(const WebGPUDevice& device)
59 {
60     if (m_computeShader.module) {
61         const auto& computeSource = m_computeShader.module->source();
62         m_computeShader.module = WebGPUShaderModule::create(GPUShaderModule::tryCreate(device.device(), { computeSource }), computeSource);
63         return true;
64     }
65     return false;
66 }
67 
68 bool WebGPUComputePipeline::recompile(const WebGPUDevice& device)

69 {
70     if (m_computePipeline && m_computeShader.module) {
71         if (auto* gpuComputeShaderModule = m_computeShader.module->module()) {
72             GPUProgrammableStageDescriptor computeStage(makeRef(*gpuComputeShaderModule), { m_computeShader.entryPoint });
73             return m_computePipeline->recompile(device.device(), WTFMove(computeStage));
74         }
75     }
76     return false;
77 }
