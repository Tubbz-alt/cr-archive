33 JSValue PerGlobalObjectWrapperWorld::getWrapper(JSGlobalObject* globalObject)
34 {
35     auto it = m_wrappers.find(globalObject);
36     if (it != m_wrappers.end())
37         return it->value.get();
38     return JSValue();
39 }
40 
41 void PerGlobalObjectWrapperWorld::addWrapper(JSGlobalObject* globalObject, JSObject* object)
42 {
43     Strong<JSObject> wrapper(globalObject->vm(), object);
44     m_wrappers.add(globalObject, wrapper);
45 }
46 
47 void PerGlobalObjectWrapperWorld::clearAllWrappers()
48 {
49     m_wrappers.clear();
50 }
