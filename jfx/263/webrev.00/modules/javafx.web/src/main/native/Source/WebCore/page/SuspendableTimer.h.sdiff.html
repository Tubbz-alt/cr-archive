 80 class SuspendableTimer final : public SuspendableTimerBase {
 81     WTF_MAKE_FAST_ALLOCATED;
 82 public:
 83     template <typename TimerFiredClass, typename TimerFiredBaseClass>
 84     SuspendableTimer(ScriptExecutionContext* context, TimerFiredClass& object, void (TimerFiredBaseClass::*function)())
 85         : SuspendableTimerBase(context)
 86         , m_function(std::bind(function, &object))
 87     {
 88     }
 89 
 90     SuspendableTimer(ScriptExecutionContext* context, WTF::Function<void ()>&& function)
 91         : SuspendableTimerBase(context)
 92         , m_function(WTFMove(function))
 93     {
 94     }
 95 
 96 private:
 97     void fired() final
 98     {
 99         m_function();
100     }
101 
102     const char* activeDOMObjectName() const final;
103 
104     WTF::Function<void ()> m_function;
105 };
