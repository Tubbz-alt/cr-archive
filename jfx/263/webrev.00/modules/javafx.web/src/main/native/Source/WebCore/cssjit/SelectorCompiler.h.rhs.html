28 #if ENABLE(CSS_SELECTOR_JIT)
29 
30 #include "CompiledSelector.h"
31 #include "SelectorChecker.h"
32 


33 namespace WebCore {
34 
35 class CSSSelector;
36 class Element;
37 
38 namespace SelectorCompiler {
39 
40 enum class SelectorContext {
41     // Rule Collector needs a resolvingMode and can modify the tree as it matches.
42     RuleCollector,
43 
44     // Query Selector does not modify the tree and never match :visited.
45     QuerySelector
46 };
47 
48 typedef unsigned (*RuleCollectorSimpleSelectorChecker)(const Element*, unsigned*);
49 typedef unsigned (*QuerySelectorSimpleSelectorChecker)(const Element*);
50 
51 typedef unsigned (*RuleCollectorSelectorCheckerWithCheckingContext)(const Element*, SelectorChecker::CheckingContext*, unsigned*);
52 typedef unsigned (*QuerySelectorSelectorCheckerWithCheckingContext)(const Element*, const SelectorChecker::CheckingContext*);
53 
54 void compileSelector(CompiledSelector&, const CSSSelector*, SelectorContext);
55 
56 inline RuleCollectorSimpleSelectorChecker ruleCollectorSimpleSelectorCheckerFunction(CompiledSelector& compiledSelector)
57 {
58     ASSERT(compiledSelector.status == SelectorCompilationStatus::SimpleSelectorChecker);
59     return WTF::untagCFunctionPtr<RuleCollectorSimpleSelectorChecker, CSSSelectorPtrTag>(compiledSelector.codeRef.code().executableAddress());
60 }
61 
62 inline QuerySelectorSimpleSelectorChecker querySelectorSimpleSelectorCheckerFunction(CompiledSelector& compiledSelector)
63 {
64     ASSERT(compiledSelector.status == SelectorCompilationStatus::SimpleSelectorChecker);
65     return WTF::untagCFunctionPtr<QuerySelectorSimpleSelectorChecker, CSSSelectorPtrTag>(compiledSelector.codeRef.code().executableAddress());
66 }
67 
68 inline RuleCollectorSelectorCheckerWithCheckingContext ruleCollectorSelectorCheckerFunctionWithCheckingContext(CompiledSelector& compiledSelector)
69 {
70     ASSERT(compiledSelector.status == SelectorCompilationStatus::SelectorCheckerWithCheckingContext);
71     return WTF::untagCFunctionPtr<RuleCollectorSelectorCheckerWithCheckingContext, CSSSelectorPtrTag>(compiledSelector.codeRef.code().executableAddress());
72 }
73 
74 inline QuerySelectorSelectorCheckerWithCheckingContext querySelectorSelectorCheckerFunctionWithCheckingContext(CompiledSelector& compiledSelector)
75 {
76     ASSERT(compiledSelector.status == SelectorCompilationStatus::SelectorCheckerWithCheckingContext);
77     return WTF::untagCFunctionPtr<QuerySelectorSelectorCheckerWithCheckingContext, CSSSelectorPtrTag>(compiledSelector.codeRef.code().executableAddress());
78 }
79 
80 } // namespace SelectorCompiler
81 } // namespace WebCore
82 
83 #endif // ENABLE(CSS_SELECTOR_JIT)
