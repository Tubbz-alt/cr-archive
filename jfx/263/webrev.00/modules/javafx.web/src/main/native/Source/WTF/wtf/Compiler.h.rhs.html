392 /* IGNORE_WARNINGS */
393 
394 /* Can't use WTF_CONCAT() and STRINGIZE() because they are defined in
395  * StdLibExtras.h, which includes this file. */
396 #define _COMPILER_CONCAT_I(a, b) a ## b
397 #define _COMPILER_CONCAT(a, b) _COMPILER_CONCAT_I(a, b)
398 
399 #define _COMPILER_STRINGIZE(exp) #exp
400 
401 #define _COMPILER_WARNING_NAME(warning) "-W" warning
402 
403 #if COMPILER(GCC) || COMPILER(CLANG)
404 #define IGNORE_WARNINGS_BEGIN_COND(cond, compiler, warning) \
405     _Pragma(_COMPILER_STRINGIZE(compiler diagnostic push)) \
406     _COMPILER_CONCAT(IGNORE_WARNINGS_BEGIN_IMPL_, cond)(compiler, warning)
407 
408 #define IGNORE_WARNINGS_BEGIN_IMPL_1(compiler, warning) \
409     _Pragma(_COMPILER_STRINGIZE(compiler diagnostic ignored warning))
410 #define IGNORE_WARNINGS_BEGIN_IMPL_0(compiler, warning)
411 #define IGNORE_WARNINGS_BEGIN_IMPL_(compiler, warning)
412 
413 
414 #define IGNORE_WARNINGS_END_IMPL(compiler) _Pragma(_COMPILER_STRINGIZE(compiler diagnostic pop))
415 
416 #if defined(__has_warning)
417 #define _IGNORE_WARNINGS_BEGIN_IMPL(compiler, warning) \
418     IGNORE_WARNINGS_BEGIN_COND(__has_warning(warning), compiler, warning)
419 #else
420 #define _IGNORE_WARNINGS_BEGIN_IMPL(compiler, warning) IGNORE_WARNINGS_BEGIN_COND(1, compiler, warning)
421 #endif
422 
423 #define IGNORE_WARNINGS_BEGIN_IMPL(compiler, warning) \
424     _IGNORE_WARNINGS_BEGIN_IMPL(compiler, _COMPILER_WARNING_NAME(warning))

426 #endif // COMPILER(GCC) || COMPILER(CLANG)
