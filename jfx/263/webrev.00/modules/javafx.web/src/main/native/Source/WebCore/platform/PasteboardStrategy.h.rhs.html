26 #ifndef PasteboardStrategy_h
27 #define PasteboardStrategy_h
28 
29 #include <wtf/Forward.h>
30 
31 namespace WebCore {
32 
33 class Color;
34 class PasteboardCustomData;
35 class SelectionData;
36 class SharedBuffer;
37 struct PasteboardImage;
38 struct PasteboardItemInfo;
39 struct PasteboardURL;
40 struct PasteboardWebContent;

41 
42 class PasteboardStrategy {
43 public:
44 #if PLATFORM(IOS_FAMILY)
45     virtual void writeToPasteboard(const PasteboardURL&, const String& pasteboardName) = 0;
46     virtual void writeToPasteboard(const PasteboardWebContent&, const String& pasteboardName) = 0;
47     virtual void writeToPasteboard(const PasteboardImage&, const String& pasteboardName) = 0;
48     virtual void writeToPasteboard(const String& pasteboardType, const String&, const String& pasteboardName) = 0;






49     virtual void updateSupportedTypeIdentifiers(const Vector<String>& identifiers, const String& pasteboardName) = 0;
50 #endif // PLATFORM(IOS_FAMILY)
51 #if PLATFORM(COCOA)
52     virtual void getTypes(Vector<String>& types, const String& pasteboardName) = 0;
53     virtual RefPtr<SharedBuffer> bufferForType(const String& pasteboardType, const String& pasteboardName) = 0;
54     virtual void getPathnamesForType(Vector<String>& pathnames, const String& pasteboardType, const String& pasteboardName) = 0;
55     virtual String stringForType(const String& pasteboardType, const String& pasteboardName) = 0;
56     virtual Vector<String> allStringsForType(const String& pasteboardType, const String& pasteboardName) = 0;
57     virtual int64_t changeCount(const String& pasteboardName) = 0;
58     virtual String uniqueName() = 0;
59     virtual Color color(const String& pasteboardName) = 0;
60     virtual URL url(const String& pasteboardName) = 0;
61     virtual int getNumberOfFiles(const String& pasteboardName) = 0;
62 
63     virtual int64_t addTypes(const Vector<String>& pasteboardTypes, const String& pasteboardName) = 0;
64     virtual int64_t setTypes(const Vector<String>& pasteboardTypes, const String& pasteboardName) = 0;
65     virtual int64_t setBufferForType(SharedBuffer*, const String& pasteboardType, const String& pasteboardName) = 0;
66     virtual int64_t setURL(const PasteboardURL&, const String& pasteboardName) = 0;
67     virtual int64_t setColor(const Color&, const String& pasteboardName) = 0;
68     virtual int64_t setStringForType(const String&, const String& pasteboardType, const String& pasteboardName) = 0;
69 #endif
70     virtual String readStringFromPasteboard(size_t index, const String& pasteboardType, const String& pasteboardName) = 0;
71     virtual RefPtr<SharedBuffer> readBufferFromPasteboard(size_t index, const String& pasteboardType, const String& pasteboardName) = 0;
72     virtual URL readURLFromPasteboard(size_t index, const String& pasteboardName, String& title) = 0;
73     virtual Optional<PasteboardItemInfo> informationForItemAtIndex(size_t index, const String& pasteboardName, int64_t changeCount) = 0;
74     virtual Optional<Vector<PasteboardItemInfo>> allPasteboardItemInfo(const String& pasteboardName, int64_t changeCount) = 0;
75     virtual int getPasteboardItemsCount(const String& pasteboardName) = 0;
76 
77     virtual Vector<String> typesSafeForDOMToReadAndWrite(const String& pasteboardName, const String& origin) = 0;
78     virtual int64_t writeCustomData(const Vector<PasteboardCustomData>&, const String& pasteboardName) = 0;
79 
80 #if PLATFORM(GTK)
81     virtual void writeToClipboard(const String& pasteboardName, const SelectionData&) = 0;
82     virtual Ref<SelectionData> readFromClipboard(const String& pasteboardName) = 0;
83 #endif // PLATFORM(GTK)
84 
85 #if USE(LIBWPE)
86     virtual void getTypes(Vector<String>& types) = 0;

87     virtual void writeToPasteboard(const PasteboardWebContent&) = 0;
88     virtual void writeToPasteboard(const String& pasteboardType, const String&) = 0;
89 #endif
90 
91 protected:
92     virtual ~PasteboardStrategy()
93     {
94     }
95 };
96 
97 }
98 
99 #endif // !PasteboardStrategy_h
