 88 
 89 class GraphicsLayerClient {
 90 public:
 91     virtual ~GraphicsLayerClient() = default;
 92 
 93     virtual void tiledBackingUsageChanged(const GraphicsLayer*, bool /*usingTiledBacking*/) { }
 94 
 95     // Callback for when hardware-accelerated animation started.
 96     virtual void notifyAnimationStarted(const GraphicsLayer*, const String& /*animationKey*/, MonotonicTime /*time*/) { }
 97     virtual void notifyAnimationEnded(const GraphicsLayer*, const String& /*animationKey*/) { }
 98 
 99     // Notification that a layer property changed that requires a subsequent call to flushCompositingState()
100     // to appear on the screen.
101     virtual void notifyFlushRequired(const GraphicsLayer*) { }
102 
103     // Notification that this layer requires a flush before the next display refresh.
104     virtual void notifyFlushBeforeDisplayRefresh(const GraphicsLayer*) { }
105 
106     virtual void paintContents(const GraphicsLayer*, GraphicsContext&, const FloatRect& /* inClip */, GraphicsLayerPaintBehavior) { }
107     virtual void didChangePlatformLayerForLayer(const GraphicsLayer*) { }
108 
109     // Provides current transform (taking transform-origin and animations into account). Input matrix has been
110     // initialized to identity already. Returns false if the layer has no transform.
111     virtual bool getCurrentTransform(const GraphicsLayer*, TransformationMatrix&) const { return false; }
112 
113     // Allows the client to modify a layer position used during the visibleRect calculation, for example to ignore
114     // scroll overhang.
115     virtual void customPositionForVisibleRectComputation(const GraphicsLayer*, FloatPoint&) const { }
116 
117     // Multiplier for backing store size, related to high DPI.
118     virtual float deviceScaleFactor() const { return 1; }
119     // Page scale factor.
120     virtual float pageScaleFactor() const { return 1; }
121     virtual float zoomedOutPageScaleFactor() const { return 0; }
122 
123     virtual float contentsScaleMultiplierForNewTiles(const GraphicsLayer*) const { return 1; }
124     virtual bool paintsOpaquelyAtNonIntegralScales(const GraphicsLayer*) const { return false; }
125 
126     virtual bool isTrackingRepaints() const { return false; }
127 
128     virtual bool shouldSkipLayerInDump(const GraphicsLayer*, LayerTreeAsTextBehavior) const { return false; }
129     virtual bool shouldDumpPropertyForLayer(const GraphicsLayer*, const char*, LayerTreeAsTextBehavior) const { return true; }
130 
131     virtual bool shouldAggressivelyRetainTiles(const GraphicsLayer*) const { return false; }
132     virtual bool shouldTemporarilyRetainTileCohorts(const GraphicsLayer*) const { return true; }
133 
134     virtual bool useGiantTiles() const { return false; }
135 
136     virtual bool needsPixelAligment() const { return false; }
137 
138     virtual bool needsIOSDumpRenderTreeMainFrameRenderViewLayerIsAlwaysOpaqueHack(const GraphicsLayer&) const { return false; }
139 
140     virtual void logFilledVisibleFreshTile(unsigned) { };
141 
142 #ifndef NDEBUG
143     // RenderLayerBacking overrides this to verify that it is not
144     // currently painting contents. An ASSERT fails, if it is.
145     // This is executed in GraphicsLayer construction and destruction
146     // to verify that we don't create or destroy GraphicsLayers
147     // while painting.
148     virtual void verifyNotPainting() { }
149 #endif
150 };
