 34 SuspendableTimer::SuspendableTimer(ScriptExecutionContext& context)
 35     : ActiveDOMObject(&context)
 36 {
 37 }

 39 SuspendableTimer::~SuspendableTimer() = default;
 40 
 41 bool SuspendableTimer::hasPendingActivity() const
 42 {
 43     return isActive();
 44 }
 45 
 46 void SuspendableTimer::stop()
 47 {
 48     if (!m_suspended)
 49         TimerBase::stop();
 50 
 51     // Make it less likely that SuspendableTimer is accidentally revived and fires after being stopped.
 52     // It is not allowed for an ActiveDOMObject to become active again after stop().
 53     m_suspended = true;
 54     m_savedIsActive = false;
 55 
 56     didStop();
 57 }
 58 
 59 void SuspendableTimer::suspend(ReasonForSuspension)
 60 {
 61     ASSERT(!m_suspended);
 62     m_suspended = true;
 63 
 64     m_savedIsActive = TimerBase::isActive();
 65     if (m_savedIsActive) {
 66         m_savedNextFireInterval = TimerBase::nextUnalignedFireInterval();
 67         m_savedRepeatInterval = TimerBase::repeatInterval();
 68         TimerBase::stop();
 69     }
 70 }
 71 
 72 void SuspendableTimer::resume()
 73 {
 74     ASSERT(m_suspended);
 75     m_suspended = false;
 76 
 77     if (m_savedIsActive)
 78         start(m_savedNextFireInterval, m_savedRepeatInterval);
 79 }

 81 bool SuspendableTimer::canSuspendForDocumentSuspension() const
 82 {
 83     return true;

120 Seconds SuspendableTimer::repeatInterval() const

122     if (!m_suspended)

129 void SuspendableTimer::augmentFireInterval(Seconds delta)
130 {
