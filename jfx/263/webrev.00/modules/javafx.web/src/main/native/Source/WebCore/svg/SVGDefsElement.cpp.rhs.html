33 inline SVGDefsElement::SVGDefsElement(const QualifiedName& tagName, Document& document)
34     : SVGGraphicsElement(tagName, document)

35 {
36     ASSERT(hasTagName(SVGNames::defsTag));
37 }
38 
39 Ref<SVGDefsElement> SVGDefsElement::create(const QualifiedName& tagName, Document& document)
40 {
41     return adoptRef(*new SVGDefsElement(tagName, document));
42 }
43 
44 bool SVGDefsElement::isValid() const
45 {
46     return SVGTests::isValid();
47 }
48 
49 RenderPtr<RenderElement> SVGDefsElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
50 {
51     return createRenderer<RenderSVGHiddenContainer>(*this, WTFMove(style));
52 }
