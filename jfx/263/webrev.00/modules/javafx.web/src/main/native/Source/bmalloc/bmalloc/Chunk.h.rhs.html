 26 #ifndef Chunk_h
 27 #define Chunk_h
 28 
 29 #include "Object.h"
 30 #include "Sizes.h"
 31 #include "SmallLine.h"

 36 namespace bmalloc {
 37 
 38 class Chunk : public ListNode<Chunk> {
 39 public:
 40     static Chunk* get(void*);
 41     static size_t metadataSize(size_t pageSize);
 42 
 43     Chunk(size_t pageSize);
 44 
 45     void ref() { ++m_refCount; }
 46     void deref() { BASSERT(m_refCount); --m_refCount; }
 47     unsigned refCount() { return m_refCount; }
 48 
 49     size_t offset(void*);
 50 
 51     char* address(size_t offset);
 52     SmallPage* page(size_t offset);
 53     SmallLine* line(size_t offset);
 54 
 55     char* bytes() { return reinterpret_cast<char*>(this); }
 56     SmallLine* lines() { return &m_lines[0]; }
 57     SmallPage* pages() { return &m_pages[0]; }
 58 
 59     List<SmallPage>& freePages() { return m_freePages; }
 60 
 61 private:
 62     size_t m_refCount { };
 63     List<SmallPage> m_freePages { };
 64 
 65     std::array<SmallLine, chunkSize / smallLineSize> m_lines { };
 66     std::array<SmallPage, chunkSize / smallPageSize> m_pages { };
 67 };
 68 
 69 struct ChunkHash {
 70     static unsigned hash(Chunk* key)
 71     {
 72         return static_cast<unsigned>(
 73             reinterpret_cast<uintptr_t>(key) / chunkSize);
 74     }
 75 };
 76 
 77 inline size_t Chunk::metadataSize(size_t pageSize)
 78 {
 79     // We align to at least the page size so we can service aligned allocations
 80     // at equal and smaller powers of two, and also so we can vmDeallocatePhysicalPages().
 81     return roundUpToMultipleOfNonPowerOfTwo(pageSize, sizeof(Chunk));
 82 }
 83 
 84 template<typename Function> void forEachPage(Chunk* chunk, size_t pageSize, Function function)
 85 {
 86     Object begin(chunk, Chunk::metadataSize(pageSize));
 87     Object end(chunk, chunkSize);
 88 
 89     for (auto it = begin; it + pageSize <= end; it = it + pageSize)
 90         function(it.page());
 91 }
 92 
 93 inline Chunk::Chunk(size_t pageSize)
 94 {
 95     size_t smallPageCount = pageSize / smallPageSize;
 96     forEachPage(this, pageSize, [&](SmallPage* page) {
 97         for (size_t i = 0; i < smallPageCount; ++i)
 98             page[i].setSlide(i);
 99     });
100 }
101 
102 inline Chunk* Chunk::get(void* address)
103 {
104     return static_cast<Chunk*>(mask(address, chunkMask));
105 }
106 
107 inline size_t Chunk::offset(void* address)
108 {
109     BASSERT(address >= this);
110     BASSERT(address < bytes() + chunkSize);
111     return static_cast<char*>(address) - bytes();
112 }
113 
114 inline char* Chunk::address(size_t offset)
115 {
116     return bytes() + offset;
117 }
118 
119 inline SmallPage* Chunk::page(size_t offset)
120 {
121     size_t pageNumber = offset / smallPageSize;
122     SmallPage* page = &m_pages[pageNumber];
123     return page - page->slide();
124 }
125 
126 inline SmallLine* Chunk::line(size_t offset)
127 {
128     size_t lineNumber = offset / smallLineSize;
129     return &m_lines[lineNumber];
130 }
131 
132 inline char* SmallLine::begin()
133 {
134     Chunk* chunk = Chunk::get(this);
135     size_t lineNumber = this - chunk->lines();
136     size_t offset = lineNumber * smallLineSize;
137     return &reinterpret_cast<char*>(chunk)[offset];
138 }
139 
140 inline char* SmallLine::end()
141 {
142     return begin() + smallLineSize;
143 }
144 
145 inline SmallLine* SmallPage::begin()
146 {
147     BASSERT(!m_slide);
148     Chunk* chunk = Chunk::get(this);
149     size_t pageNumber = this - chunk->pages();
150     size_t lineNumber = pageNumber * smallPageLineCount;
151     return &chunk->lines()[lineNumber];
152 }
153 
154 inline Object::Object(void* object)
155     : m_chunk(Chunk::get(object))
156     , m_offset(m_chunk->offset(object))
157 {
158 }
159 
160 inline Object::Object(Chunk* chunk, void* object)
161     : m_chunk(chunk)
162     , m_offset(m_chunk->offset(object))
163 {
164     BASSERT(chunk == Chunk::get(object));
165 }
166 
167 inline char* Object::address()
168 {
169     return m_chunk->address(m_offset);
170 }
171 
172 inline SmallLine* Object::line()
173 {
174     return m_chunk->line(m_offset);
175 }
176 
177 inline SmallPage* Object::page()
178 {
179     return m_chunk->page(m_offset);
180 }
181 
182 }; // namespace bmalloc

184 #endif // Chunk
