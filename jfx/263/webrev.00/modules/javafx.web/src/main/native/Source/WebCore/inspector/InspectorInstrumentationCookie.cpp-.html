36 namespace WebCore {

38 InspectorInstrumentationCookie::InspectorInstrumentationCookie()
39     : m_timelineAgentId(0)
40 {
41 }
42 
43 InspectorInstrumentationCookie::InspectorInstrumentationCookie(InstrumentingAgents& agents, int timelineAgentId)
44     : m_instrumentingAgents(&agents)
45     , m_timelineAgentId(timelineAgentId)
46 {
47 }
48 
49 InspectorInstrumentationCookie::InspectorInstrumentationCookie(const InspectorInstrumentationCookie& other)
50     : m_instrumentingAgents(other.m_instrumentingAgents)
51     , m_timelineAgentId(other.m_timelineAgentId)
52 {
53 }
54 
55 InspectorInstrumentationCookie& InspectorInstrumentationCookie::operator=(const InspectorInstrumentationCookie& other)
56 {
57     if (this != &other) {
58         m_instrumentingAgents = other.m_instrumentingAgents;
59         m_timelineAgentId = other.m_timelineAgentId;
60     }
61 
62     return *this;
63 }
64 
65 InspectorInstrumentationCookie::~InspectorInstrumentationCookie() = default;

67 } // namespace WebCore
