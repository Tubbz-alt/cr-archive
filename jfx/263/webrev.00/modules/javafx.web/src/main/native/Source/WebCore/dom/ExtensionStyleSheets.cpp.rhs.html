 55 ExtensionStyleSheets::ExtensionStyleSheets(Document& document)
 56     : m_document(document)
 57 {
 58 }

 60 static Ref<CSSStyleSheet> createExtensionsStyleSheet(Document& document, URL url, const String& text, UserStyleLevel level)
 61 {
 62     auto contents = StyleSheetContents::create(url, CSSParserContext(document, url));
 63     auto styleSheet = CSSStyleSheet::create(contents.get(), document, true);
 64 
 65     contents->setIsUserStyleSheet(level == UserStyleUserLevel);
 66     contents->parseString(text);
 67 
 68     return styleSheet;
 69 }
 70 
 71 CSSStyleSheet* ExtensionStyleSheets::pageUserSheet()
 72 {
 73     if (m_pageUserSheet)
 74         return m_pageUserSheet.get();
 75 
 76     Page* owningPage = m_document.page();
 77     if (!owningPage)
 78         return 0;
 79 
 80     String userSheetText = owningPage->userStyleSheet();
 81     if (userSheetText.isEmpty())
 82         return 0;
 83 
 84     m_pageUserSheet = createExtensionsStyleSheet(m_document, m_document.settings().userStyleSheetLocation(), userSheetText, UserStyleUserLevel);
 85 
 86     return m_pageUserSheet.get();
 87 }
 88 
 89 void ExtensionStyleSheets::clearPageUserSheet()
 90 {
 91     if (m_pageUserSheet) {
 92         m_pageUserSheet = nullptr;
 93         m_document.styleScope().didChangeStyleSheetEnvironment();
 94     }
 95 }
 96 
 97 void ExtensionStyleSheets::updatePageUserSheet()
 98 {
 99     clearPageUserSheet();
100     if (pageUserSheet())
101         m_document.styleScope().didChangeStyleSheetEnvironment();
102 }

116 void ExtensionStyleSheets::updateInjectedStyleSheetCache() const
117 {
118     if (m_injectedStyleSheetCacheValid)
119         return;
120 
121     m_injectedStyleSheetCacheValid = true;
122     m_injectedUserStyleSheets.clear();
123     m_injectedAuthorStyleSheets.clear();
124     m_injectedStyleSheetToSource.clear();
125 
126     Page* owningPage = m_document.page();
127     if (!owningPage)
128         return;
129 
130     owningPage->userContentProvider().forEachUserStyleSheet([&](const UserStyleSheet& userStyleSheet) {
131         if (userStyleSheet.injectedFrames() == InjectInTopFrameOnly && m_document.ownerElement())
132             return;
133 
134         if (!UserContentURLPattern::matchesPatterns(m_document.url(), userStyleSheet.whitelist(), userStyleSheet.blacklist()))
135             return;
136 
137         auto sheet = createExtensionsStyleSheet(const_cast<Document&>(m_document), userStyleSheet.url(), userStyleSheet.source(), userStyleSheet.level());
138 
139         m_injectedStyleSheetToSource.set(sheet.copyRef(), userStyleSheet.source());
140 
141         if (userStyleSheet.level() == UserStyleUserLevel)
142             m_injectedUserStyleSheets.append(WTFMove(sheet));
143         else
144             m_injectedAuthorStyleSheets.append(WTFMove(sheet));
145     });
146 }
147 
148 void ExtensionStyleSheets::invalidateInjectedStyleSheetCache()
149 {
150     m_injectedStyleSheetCacheValid = false;
151     m_document.styleScope().didChangeStyleSheetEnvironment();
152 }

154 void ExtensionStyleSheets::addUserStyleSheet(Ref<StyleSheetContents>&& userSheet)
155 {
156     ASSERT(userSheet.get().isUserStyleSheet());
157     m_userStyleSheets.append(CSSStyleSheet::create(WTFMove(userSheet), m_document));
158     m_document.styleScope().didChangeStyleSheetEnvironment();
159 }
160 
161 void ExtensionStyleSheets::addAuthorStyleSheetForTesting(Ref<StyleSheetContents>&& authorSheet)
162 {
163     ASSERT(!authorSheet.get().isUserStyleSheet());
164     m_authorStyleSheetsForTesting.append(CSSStyleSheet::create(WTFMove(authorSheet), m_document));
165     m_document.styleScope().didChangeStyleSheetEnvironment();
166 }

168 #if ENABLE(CONTENT_EXTENSIONS)
169 void ExtensionStyleSheets::addDisplayNoneSelector(const String& identifier, const String& selector, uint32_t selectorID)
170 {
171     auto result = m_contentExtensionSelectorSheets.add(identifier, nullptr);
172     if (result.isNewEntry) {
173         result.iterator->value = ContentExtensionStyleSheet::create(m_document);
174         m_userStyleSheets.append(&result.iterator->value->styleSheet());
175     }
176 
177     if (result.iterator->value->addDisplayNoneSelector(selector, selectorID))
178         m_document.styleScope().didChangeStyleSheetEnvironment();
179 }
180 
181 void ExtensionStyleSheets::maybeAddContentExtensionSheet(const String& identifier, StyleSheetContents& sheet)
182 {
183     ASSERT(sheet.isUserStyleSheet());
184 
185     if (m_contentExtensionSheets.contains(identifier))
186         return;
187 
188     Ref<CSSStyleSheet> cssSheet = CSSStyleSheet::create(sheet, m_document);
189     m_contentExtensionSheets.set(identifier, &cssSheet.get());
190     m_userStyleSheets.append(adoptRef(cssSheet.leakRef()));
191     m_document.styleScope().didChangeStyleSheetEnvironment();
192 
193 }
194 #endif // ENABLE(CONTENT_EXTENSIONS)

196 String ExtensionStyleSheets::contentForInjectedStyleSheet(const RefPtr<CSSStyleSheet>& styleSheet) const
197 {
198     return m_injectedStyleSheetToSource.get(styleSheet);
199 }
200 
201 void ExtensionStyleSheets::detachFromDocument()
202 {
203     if (m_pageUserSheet)
204         m_pageUserSheet->detachFromDocument();
205     for (auto& sheet : m_injectedUserStyleSheets)
206         sheet->detachFromDocument();
207     for (auto& sheet :  m_injectedAuthorStyleSheets)
208         sheet->detachFromDocument();
209     for (auto& sheet : m_userStyleSheets)
210         sheet->detachFromDocument();
211     for (auto& sheet : m_authorStyleSheetsForTesting)
212         sheet->detachFromDocument();
213 }
