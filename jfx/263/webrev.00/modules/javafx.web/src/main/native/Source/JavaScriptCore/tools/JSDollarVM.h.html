33 struct DollarVMAssertScope {
34     DollarVMAssertScope() { RELEASE_ASSERT(Options::useDollarVM()); }
35     ~DollarVMAssertScope() { RELEASE_ASSERT(Options::useDollarVM()); }
36 };
37 
38 class JSDollarVM final : public JSNonFinalObject {
39 public:
40     typedef JSNonFinalObject Base;
41 
42     DECLARE_EXPORT_INFO;
43 
44     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
45     {
46         DollarVMAssertScope assertScope;
47         return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
48     }
49 
50     static JSDollarVM* create(VM& vm, Structure* structure)
51     {
52         DollarVMAssertScope assertScope;
53         JSDollarVM* instance = new (NotNull, allocateCell<JSDollarVM>(vm.heap)) JSDollarVM(vm, structure);
54         instance->finishCreation(vm);
55         return instance;
56     }
57 
58     Structure* objectDoingSideEffectPutWithoutCorrectSlotStatusStructure() { return m_objectDoingSideEffectPutWithoutCorrectSlotStatusStructure.get(); }
59 
60 private:
61     JSDollarVM(VM& vm, Structure* structure)
62         : Base(vm, structure)
63     {
64         DollarVMAssertScope assertScope;
65     }
66 
67     void finishCreation(VM&);
68     void addFunction(VM&, JSGlobalObject*, const char* name, NativeFunction, unsigned arguments);
69     void addConstructibleFunction(VM&, JSGlobalObject*, const char* name, NativeFunction, unsigned arguments);
70 
71     static void visitChildren(JSCell*, SlotVisitor&);
72 
73     WriteBarrier<Structure> m_objectDoingSideEffectPutWithoutCorrectSlotStatusStructure;
74 };
