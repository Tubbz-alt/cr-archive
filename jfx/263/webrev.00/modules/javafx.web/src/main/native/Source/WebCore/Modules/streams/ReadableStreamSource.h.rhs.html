33 #include "ReadableStreamDefaultController.h"
34 #include <wtf/Optional.h>
35 
36 namespace WebCore {
37 
38 template<typename IDLType> class DOMPromiseDeferred;
39 
40 class ReadableStreamSource : public RefCounted<ReadableStreamSource> {
41 public:
42     virtual ~ReadableStreamSource();
43 
44     void start(ReadableStreamDefaultController&&, DOMPromiseDeferred<void>&&);
45     void pull(DOMPromiseDeferred<void>&&);
46     void cancel(JSC::JSValue);
47 
48     bool isPulling() const { return !!m_promise; }
49 
50 protected:
51     ReadableStreamDefaultController& controller() { return m_controller.value(); }
52     const ReadableStreamDefaultController& controller() const { return m_controller.value(); }
53 
54     void startFinished();
55     void pullFinished();
56     void cancelFinished();
57     void clean();
58 
59     virtual void setActive() = 0;
60     virtual void setInactive() = 0;
61 
62     virtual void doStart() = 0;
63     virtual void doPull() = 0;
64     virtual void doCancel() = 0;
65 
66 private:
67     std::unique_ptr<DOMPromiseDeferred<void>> m_promise;
68     Optional<ReadableStreamDefaultController> m_controller;
69 };
