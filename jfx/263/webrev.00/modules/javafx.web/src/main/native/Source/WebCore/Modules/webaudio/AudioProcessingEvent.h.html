25 #pragma once
26 
27 #include "AudioBuffer.h"
28 #include "Event.h"
29 #include <wtf/RefPtr.h>
30 
31 namespace WebCore {
32 
33 class AudioBuffer;
34 
35 class AudioProcessingEvent final : public Event {
36     WTF_MAKE_ISO_ALLOCATED(AudioProcessingEvent);
37 public:
38     static Ref<AudioProcessingEvent> create(RefPtr<AudioBuffer>&& inputBuffer, RefPtr<AudioBuffer>&& outputBuffer, double playbackTime)
39     {
40         return adoptRef(*new AudioProcessingEvent(WTFMove(inputBuffer), WTFMove(outputBuffer), playbackTime));
41     }
42 
43     static Ref<AudioProcessingEvent> createForBindings()
44     {
45         return adoptRef(*new AudioProcessingEvent);
46     }
47 
48     virtual ~AudioProcessingEvent();
49 
50     AudioBuffer* inputBuffer() { return m_inputBuffer.get(); }
51     AudioBuffer* outputBuffer() { return m_outputBuffer.get(); }
52     double playbackTime() const { return m_playbackTime; }
53 
54     EventInterface eventInterface() const override;
55 
56 private:
57     AudioProcessingEvent();
58     AudioProcessingEvent(RefPtr<AudioBuffer>&& inputBuffer, RefPtr<AudioBuffer>&& outputBuffer, double playbackTime);
59 
60     RefPtr<AudioBuffer> m_inputBuffer;
61     RefPtr<AudioBuffer> m_outputBuffer;
62     double m_playbackTime;
63 };
64 
65 } // namespace WebCore
