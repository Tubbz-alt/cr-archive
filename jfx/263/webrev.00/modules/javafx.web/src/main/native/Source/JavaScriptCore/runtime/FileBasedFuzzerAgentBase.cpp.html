38 String FileBasedFuzzerAgentBase::createLookupKey(const String& sourceFilename, OpcodeID opcodeId, int startLocation, int endLocation)
39 {
40     StringBuilder lookupKey;
41     lookupKey.append(sourceFilename);
42     lookupKey.append("|");
43     lookupKey.append(opcodeNames[opcodeAliasForLookupKey(opcodeId)]);
44     lookupKey.append("|");
45     lookupKey.append(startLocation);
46     lookupKey.append("|");
47     lookupKey.append(endLocation);
48     return lookupKey.toString();
49 }
50 
51 OpcodeID FileBasedFuzzerAgentBase::opcodeAliasForLookupKey(const OpcodeID& opcodeId)
52 {
53     if (opcodeId == op_call_varargs || opcodeId == op_call_eval || opcodeId == op_tail_call || opcodeId == op_tail_call_varargs)
54         return op_call;
55     if (opcodeId == op_get_direct_pname || opcodeId == op_get_by_val_with_this)
56         return op_get_by_val;
57     if (opcodeId == op_construct_varargs)
58         return op_construct;
59     return opcodeId;
60 }

62 SpeculatedType FileBasedFuzzerAgentBase::getPrediction(CodeBlock* codeBlock, const CodeOrigin& codeOrigin, SpeculatedType original)
63 {
64     auto locker = holdLock(m_lock);
65 
66     ScriptExecutable* ownerExecutable = codeBlock->ownerExecutable();
67     const auto& sourceURL = ownerExecutable->sourceURL();
68     if (sourceURL.isEmpty())
69         return original;
70 
71     PredictionTarget predictionTarget;
72     BytecodeIndex bytecodeIndex = codeOrigin.bytecodeIndex();
73     codeBlock->expressionRangeForBytecodeIndex(bytecodeIndex, predictionTarget.divot, predictionTarget.startOffset, predictionTarget.endOffset, predictionTarget.line, predictionTarget.column);
74 
75     Vector<String> urlParts = sourceURL.split('/');
76     predictionTarget.sourceFilename = urlParts.isEmpty() ? sourceURL : urlParts.last();
77 
78     const InstructionStream& instructions = codeBlock->instructions();
79     const Instruction* anInstruction = instructions.at(bytecodeIndex).ptr();
80     predictionTarget.opcodeId = anInstruction->opcodeID();
81 
82     int startLocation = predictionTarget.divot - predictionTarget.startOffset;
83     int endLocation = predictionTarget.divot + predictionTarget.endOffset;
84     predictionTarget.lookupKey = createLookupKey(predictionTarget.sourceFilename, predictionTarget.opcodeId, startLocation, endLocation);
85     return getPredictionInternal(codeBlock, predictionTarget, original);
86 }
