32 namespace JSC {

34 STATIC_ASSERT_IS_TRIVIALLY_DESTRUCTIBLE(SymbolObject);
35 
36 const ClassInfo SymbolObject::s_info = { "Symbol", &Base::s_info, nullptr, nullptr, CREATE_METHOD_TABLE(SymbolObject) };
37 
38 SymbolObject::SymbolObject(VM& vm, Structure* structure)
39     : Base(vm, structure)
40 {
41 }
42 
43 void SymbolObject::finishCreation(VM& vm, Symbol* symbol)
44 {
45     Base::finishCreation(vm);
46     ASSERT(inherits(vm, info()));
47     setInternalValue(vm, symbol);
48 }
49 
50 String SymbolObject::toStringName(const JSObject*, ExecState*)
51 {
52     return "Object"_s;
53 }
54 
55 JSValue SymbolObject::defaultValue(const JSObject* object, ExecState*, PreferredPrimitiveType)
56 {
57     const SymbolObject* symbolObject = jsCast<const SymbolObject*>(object);
58     return symbolObject->internalValue();
59 }
