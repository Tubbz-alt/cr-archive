34 NavigatorIsLoggedIn* NavigatorIsLoggedIn::from(Navigator& navigator)
35 {
36     auto* supplement = static_cast<NavigatorIsLoggedIn*>(Supplement<Navigator>::from(&navigator, supplementName()));
37     if (!supplement) {
38         auto newSupplement = makeUnique<NavigatorIsLoggedIn>(navigator);
39         supplement = newSupplement.get();
40         provideTo(&navigator, supplementName(), WTFMove(newSupplement));
41     }
42     return supplement;
43 }
44 
45 const char* NavigatorIsLoggedIn::supplementName()
46 {
47     return "NavigatorIsLoggedIn";
48 }
49 
50 void NavigatorIsLoggedIn::setLoggedIn(Navigator& navigator, Ref<DeferredPromise>&& promise)
51 {
52     NavigatorIsLoggedIn::from(navigator)->setLoggedIn(WTFMove(promise));
53 }
54 
55 void NavigatorIsLoggedIn::setLoggedOut(Navigator& navigator, Ref<DeferredPromise>&& promise)
56 {
57     NavigatorIsLoggedIn::from(navigator)->setLoggedOut(WTFMove(promise));
58 }
59 
60 void NavigatorIsLoggedIn::isLoggedIn(Navigator& navigator, Ref<DeferredPromise>&& promise)
61 {
62     NavigatorIsLoggedIn::from(navigator)->isLoggedIn(WTFMove(promise));
63 }
64 
65 void NavigatorIsLoggedIn::setLoggedIn(Ref<DeferredPromise>&& promise)
66 {
67     if (m_navigator.cookieEnabled())
68         promise->resolve();
69     else
70         promise->reject();
71 }
72 
73 void NavigatorIsLoggedIn::setLoggedOut(Ref<DeferredPromise>&& promise)
74 {
75     promise->resolve();
76 }
77 
78 void NavigatorIsLoggedIn::isLoggedIn(Ref<DeferredPromise>&& promise)
79 {
80     if (m_navigator.cookieEnabled())
81         promise->resolve<IDLBoolean>(true);
82     else
83         promise->resolve<IDLBoolean>(false);
84 }
