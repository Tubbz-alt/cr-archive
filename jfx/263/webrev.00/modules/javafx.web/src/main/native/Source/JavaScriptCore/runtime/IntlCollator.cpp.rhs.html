 93 static Vector<String> sortLocaleData(const String& locale, size_t keyIndex)
 94 {
 95     // 9.1 Internal slots of Service Constructors & 10.2.3 Internal slots (ECMA-402 2.0)
 96     Vector<String> keyLocaleData;
 97     switch (keyIndex) {
 98     case indexOfExtensionKeyCo: {
 99         // 10.2.3 "The first element of [[sortLocaleData]][locale].co and [[searchLocaleData]][locale].co must be null for all locale values."
100         keyLocaleData.append({ });
101 
102         UErrorCode status = U_ZERO_ERROR;
103         UEnumeration* enumeration = ucol_getKeywordValuesForLocale("collation", locale.utf8().data(), false, &status);
104         if (U_SUCCESS(status)) {
105             const char* collation;
106             while ((collation = uenum_next(enumeration, nullptr, &status)) && U_SUCCESS(status)) {
107                 // 10.2.3 "The values "standard" and "search" must not be used as elements in any [[sortLocaleData]][locale].co and [[searchLocaleData]][locale].co array."
108                 if (!strcmp(collation, "standard") || !strcmp(collation, "search"))
109                     continue;
110 
111                 // Map keyword values to BCP 47 equivalents.
112                 if (!strcmp(collation, "dictionary"))
113                     collation = "dict";
114                 else if (!strcmp(collation, "gb2312han"))
115                     collation = "gb2312";
116                 else if (!strcmp(collation, "phonebook"))
117                     collation = "phonebk";
118                 else if (!strcmp(collation, "traditional"))
119                     collation = "trad";
120 
121                 keyLocaleData.append(collation);
122             }
123             uenum_close(enumeration);
124         }
125         break;
126     }
127     case indexOfExtensionKeyKn:
128         keyLocaleData.reserveInitialCapacity(2);
129         keyLocaleData.uncheckedAppend("false"_s);
130         keyLocaleData.uncheckedAppend("true"_s);
131         break;
132     case indexOfExtensionKeyKf:
133         keyLocaleData.reserveInitialCapacity(3);
134         keyLocaleData.uncheckedAppend("false"_s);
135         keyLocaleData.uncheckedAppend("lower"_s);
136         keyLocaleData.uncheckedAppend("upper"_s);
137         break;
138     default:
139         ASSERT_NOT_REACHED();
140     }
141     return keyLocaleData;
142 }

144 static Vector<String> searchLocaleData(const String&, size_t keyIndex)
145 {
146     // 9.1 Internal slots of Service Constructors & 10.2.3 Internal slots (ECMA-402 2.0)
147     Vector<String> keyLocaleData;
148     switch (keyIndex) {
149     case indexOfExtensionKeyCo:
150         // 10.2.3 "The first element of [[sortLocaleData]][locale].co and [[searchLocaleData]][locale].co must be null for all locale values."
151         keyLocaleData.reserveInitialCapacity(1);
152         keyLocaleData.append({ });
153         break;
154     case indexOfExtensionKeyKn:
155         keyLocaleData.reserveInitialCapacity(2);
156         keyLocaleData.uncheckedAppend("false"_s);
157         keyLocaleData.uncheckedAppend("true"_s);
158         break;
159     case indexOfExtensionKeyKf:
160         keyLocaleData.reserveInitialCapacity(3);
161         keyLocaleData.uncheckedAppend("false"_s);
162         keyLocaleData.uncheckedAppend("lower"_s);
163         keyLocaleData.uncheckedAppend("upper"_s);
164         break;
165     default:
166         ASSERT_NOT_REACHED();
167     }
168     return keyLocaleData;
169 }

171 void IntlCollator::initializeCollator(JSGlobalObject* globalObject, JSValue locales, JSValue optionsValue)
172 {
173     VM& vm = globalObject->vm();
174     auto scope = DECLARE_THROW_SCOPE(vm);
175 
176     // 10.1.1 InitializeCollator (collator, locales, options) (ECMA-402)
177     // https://tc39.github.io/ecma402/#sec-initializecollator
178 
179     auto requestedLocales = canonicalizeLocaleList(globalObject, locales);
180     RETURN_IF_EXCEPTION(scope, void());
181 
182     JSValue options = optionsValue;
183     if (!optionsValue.isUndefined()) {
184         options = optionsValue.toObject(globalObject);


185         RETURN_IF_EXCEPTION(scope, void());
186     }
187 
188     String usageString = intlStringOption(globalObject, options, vm.propertyNames->usage, { "sort", "search" }, "usage must be either \"sort\" or \"search\"", "sort");
189     RETURN_IF_EXCEPTION(scope, void());
190     if (usageString == "sort")
191         m_usage = Usage::Sort;
192     else if (usageString == "search")
193         m_usage = Usage::Search;
194     else
195         ASSERT_NOT_REACHED();
196 
197     auto localeData = (m_usage == Usage::Sort) ? sortLocaleData : searchLocaleData;
198 
199     HashMap<String, String> opt;
200 
201     String matcher = intlStringOption(globalObject, options, vm.propertyNames->localeMatcher, { "lookup", "best fit" }, "localeMatcher must be either \"lookup\" or \"best fit\"", "best fit");
202     RETURN_IF_EXCEPTION(scope, void());
203     opt.add("localeMatcher"_s, matcher);
204 
205     {
206         String numericString;
207         bool usesFallback;
208         bool numeric = intlBooleanOption(globalObject, options, vm.propertyNames->numeric, usesFallback);
209         RETURN_IF_EXCEPTION(scope, void());
210         if (!usesFallback)
211             numericString = numeric ? "true"_s : "false"_s;
212         if (!numericString.isNull())
213             opt.add("kn"_s, numericString);
214     }
215     {
216         String caseFirst = intlStringOption(globalObject, options, vm.propertyNames->caseFirst, { "upper", "lower", "false" }, "caseFirst must be either \"upper\", \"lower\", or \"false\"", nullptr);
217         RETURN_IF_EXCEPTION(scope, void());
218         if (!caseFirst.isNull())
219             opt.add("kf"_s, caseFirst);
220     }
221 
222     auto& availableLocales = intlCollatorAvailableLocales();
223     auto result = resolveLocale(globalObject, availableLocales, requestedLocales, opt, relevantCollatorExtensionKeys, WTF_ARRAY_LENGTH(relevantCollatorExtensionKeys), localeData);
224 
225     m_locale = result.get("locale"_s);
226     if (m_locale.isEmpty()) {
227         throwTypeError(globalObject, scope, "failed to initialize Collator due to invalid locale"_s);
228         return;
229     }
230 
231     const String& collation = result.get("co"_s);
232     m_collation = collation.isNull() ? "default"_s : collation;
233     m_numeric = result.get("kn"_s) == "true";
234 
235     const String& caseFirst = result.get("kf"_s);
236     if (caseFirst == "lower")
237         m_caseFirst = CaseFirst::Lower;
238     else if (caseFirst == "upper")
239         m_caseFirst = CaseFirst::Upper;
240     else
241         m_caseFirst = CaseFirst::False;
242 
243     String sensitivityString = intlStringOption(globalObject, options, vm.propertyNames->sensitivity, { "base", "accent", "case", "variant" }, "sensitivity must be either \"base\", \"accent\", \"case\", or \"variant\"", nullptr);
244     RETURN_IF_EXCEPTION(scope, void());
245     if (sensitivityString == "base")
246         m_sensitivity = Sensitivity::Base;
247     else if (sensitivityString == "accent")
248         m_sensitivity = Sensitivity::Accent;
249     else if (sensitivityString == "case")
250         m_sensitivity = Sensitivity::Case;
251     else
252         m_sensitivity = Sensitivity::Variant;
253 
254     bool usesFallback;
255     bool ignorePunctuation = intlBooleanOption(globalObject, options, vm.propertyNames->ignorePunctuation, usesFallback);
256     if (usesFallback)
257         ignorePunctuation = false;
258     RETURN_IF_EXCEPTION(scope, void());
259     m_ignorePunctuation = ignorePunctuation;
260 
261     m_initializedCollator = true;
262 }

264 void IntlCollator::createCollator(JSGlobalObject* globalObject)
265 {
266     VM& vm = globalObject->vm();
267     auto scope = DECLARE_CATCH_SCOPE(vm);
268     ASSERT(!m_collator);
269 
270     if (!m_initializedCollator) {
271         initializeCollator(globalObject, jsUndefined(), jsUndefined());
272         scope.assertNoException();
273     }
274 
275     UErrorCode status = U_ZERO_ERROR;
276     auto collator = std::unique_ptr<UCollator, UCollatorDeleter>(ucol_open(m_locale.utf8().data(), &status));
277     if (U_FAILURE(status))
278         return;
279 
280     UColAttributeValue strength = UCOL_PRIMARY;
281     UColAttributeValue caseLevel = UCOL_OFF;
282     UColAttributeValue caseFirst = UCOL_OFF;
283     switch (m_sensitivity) {
284     case Sensitivity::Base:
285         break;
286     case Sensitivity::Accent:
287         strength = UCOL_SECONDARY;
288         break;
289     case Sensitivity::Case:
290         caseLevel = UCOL_ON;
291         break;
292     case Sensitivity::Variant:
293         strength = UCOL_TERTIARY;
294         break;
295     }
296     switch (m_caseFirst) {
297     case CaseFirst::False:
298         break;
299     case CaseFirst::Lower:
300         caseFirst = UCOL_LOWER_FIRST;
301         break;
302     case CaseFirst::Upper:
303         caseFirst = UCOL_UPPER_FIRST;
304         break;
305     }
306 
307     ucol_setAttribute(collator.get(), UCOL_STRENGTH, strength, &status);
308     ucol_setAttribute(collator.get(), UCOL_CASE_LEVEL, caseLevel, &status);
309     ucol_setAttribute(collator.get(), UCOL_CASE_FIRST, caseFirst, &status);
310     ucol_setAttribute(collator.get(), UCOL_NUMERIC_COLLATION, m_numeric ? UCOL_ON : UCOL_OFF, &status);
311 
312     // FIXME: Setting UCOL_ALTERNATE_HANDLING to UCOL_SHIFTED causes punctuation and whitespace to be
313     // ignored. There is currently no way to ignore only punctuation.
314     ucol_setAttribute(collator.get(), UCOL_ALTERNATE_HANDLING, m_ignorePunctuation ? UCOL_SHIFTED : UCOL_DEFAULT, &status);
315 
316     // "The method is required to return 0 when comparing Strings that are considered canonically
317     // equivalent by the Unicode standard."
318     ucol_setAttribute(collator.get(), UCOL_NORMALIZATION_MODE, UCOL_ON, &status);
319     if (U_FAILURE(status))
320         return;
321 
322     m_collator = WTFMove(collator);
323 }

325 JSValue IntlCollator::compareStrings(JSGlobalObject* globalObject, StringView x, StringView y)
326 {
327     VM& vm = globalObject->vm();
328     auto scope = DECLARE_THROW_SCOPE(vm);
329 
330     // 10.3.4 CompareStrings abstract operation (ECMA-402 2.0)
331     if (!m_collator) {
332         createCollator(globalObject);
333         if (!m_collator)
334             return throwException(globalObject, scope, createError(globalObject, "Failed to compare strings."_s));
335     }
336 
337     UErrorCode status = U_ZERO_ERROR;
338     UCollationResult result = UCOL_EQUAL;
339     if (x.is8Bit() && y.is8Bit() && x.isAllASCII() && y.isAllASCII())
340         result = ucol_strcollUTF8(m_collator.get(), bitwise_cast<const char*>(x.characters8()), x.length(), bitwise_cast<const char*>(y.characters8()), y.length(), &status);
341     else {
342         auto getCharacters = [&] (const StringView& view, Vector<UChar>& buffer) -> const UChar* {
343             if (!view.is8Bit())
344                 return view.characters16();
345             buffer.resize(view.length());
346             StringImpl::copyCharacters(buffer.data(), view.characters8(), view.length());
347             return buffer.data();
348         };
349 
350         Vector<UChar> xBuffer;
351         Vector<UChar> yBuffer;
352         const UChar* xCharacters = getCharacters(x, xBuffer);
353         const UChar* yCharacters = getCharacters(y, yBuffer);
354         result = ucol_strcoll(m_collator.get(), xCharacters, x.length(), yCharacters, y.length());
355     }
356     if (U_FAILURE(status))
357         return throwException(globalObject, scope, createError(globalObject, "Failed to compare strings."_s));
358     return jsNumber(result);
359 }

361 ASCIILiteral IntlCollator::usageString(Usage usage)
362 {
363     switch (usage) {
364     case Usage::Sort:

403 JSObject* IntlCollator::resolvedOptions(JSGlobalObject* globalObject)
404 {
405     VM& vm = globalObject->vm();
406     auto scope = DECLARE_THROW_SCOPE(vm);
407 
408     // 10.3.5 Intl.Collator.prototype.resolvedOptions() (ECMA-402 2.0)
409     // The function returns a new object whose properties and attributes are set as if
410     // constructed by an object literal assigning to each of the following properties the
411     // value of the corresponding internal slot of this Collator object (see 10.4): locale,
412     // usage, sensitivity, ignorePunctuation, collation, as well as those properties shown
413     // in Table 1 whose keys are included in the %Collator%[[relevantExtensionKeys]]
414     // internal slot of the standard built-in object that is the initial value of
415     // Intl.Collator.
416 
417     if (!m_initializedCollator) {
418         initializeCollator(globalObject, jsUndefined(), jsUndefined());
419         scope.assertNoException();
420     }
421 
422     JSObject* options = constructEmptyObject(globalObject);
423     options->putDirect(vm, vm.propertyNames->locale, jsString(vm, m_locale));
424     options->putDirect(vm, vm.propertyNames->usage, jsNontrivialString(vm, usageString(m_usage)));
425     options->putDirect(vm, vm.propertyNames->sensitivity, jsNontrivialString(vm, sensitivityString(m_sensitivity)));
426     options->putDirect(vm, vm.propertyNames->ignorePunctuation, jsBoolean(m_ignorePunctuation));
427     options->putDirect(vm, vm.propertyNames->collation, jsString(vm, m_collation));
428     options->putDirect(vm, vm.propertyNames->numeric, jsBoolean(m_numeric));
429     options->putDirect(vm, vm.propertyNames->caseFirst, jsNontrivialString(vm, caseFirstString(m_caseFirst)));
430     return options;
431 }
