29 #include <wtf/IsoMallocInlines.h>
30 #include <wtf/NeverDestroyed.h>

32 #if ENABLE(WIRELESS_PLAYBACK_TARGET)
33 
34 namespace WebCore {
35 
36 WTF_MAKE_ISO_ALLOCATED_IMPL(WebKitPlaybackTargetAvailabilityEvent);
37 
38 static const AtomString& stringForPlaybackTargetAvailability(bool available)
39 {
40     static NeverDestroyed<AtomString> availableString("available", AtomString::ConstructFromLiteral);
41     static NeverDestroyed<AtomString> notAvailableString("not-available", AtomString::ConstructFromLiteral);
42 
43     return available ? availableString : notAvailableString;
44 }
45 
46 WebKitPlaybackTargetAvailabilityEvent::WebKitPlaybackTargetAvailabilityEvent(const AtomString& eventType, bool available)
47     : Event(eventType, CanBubble::No, IsCancelable::No)
48     , m_availability(stringForPlaybackTargetAvailability(available))
49 {
50 }
51 
52 WebKitPlaybackTargetAvailabilityEvent::WebKitPlaybackTargetAvailabilityEvent(const AtomString& eventType, const Init& initializer, IsTrusted isTrusted)
53     : Event(eventType, initializer, isTrusted)
54     , m_availability(initializer.availability)
55 {
56 }
57 
58 } // namespace WebCore

60 #endif // ENABLE(WIRELESS_PLAYBACK_TARGET)
