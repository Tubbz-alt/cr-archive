 26 #pragma once
 27 
 28 #include "SVGLengthContext.h"
 29 #include "SVGLengthValue.h"
 30 #include "SVGValueProperty.h"
 31 
 32 namespace WebCore {
 33 
 34 class SVGLength : public SVGValueProperty<SVGLengthValue> {
 35     using Base = SVGValueProperty<SVGLengthValue>;
 36     using Base::Base;
 37     using Base::m_value;
 38 
 39 public:
 40     // Forward declare these enums in the w3c naming scheme, for IDL generation
 41     enum {
 42         SVG_LENGTHTYPE_UNKNOWN      = static_cast<unsigned>(SVGLengthType::Unknown),
 43         SVG_LENGTHTYPE_NUMBER       = static_cast<unsigned>(SVGLengthType::Number),
 44         SVG_LENGTHTYPE_PERCENTAGE   = static_cast<unsigned>(SVGLengthType::Percentage),
 45         SVG_LENGTHTYPE_EMS          = static_cast<unsigned>(SVGLengthType::Ems),
 46         SVG_LENGTHTYPE_EXS          = static_cast<unsigned>(SVGLengthType::Exs),
 47         SVG_LENGTHTYPE_PX           = static_cast<unsigned>(SVGLengthType::Pixels),
 48         SVG_LENGTHTYPE_CM           = static_cast<unsigned>(SVGLengthType::Centimeters),
 49         SVG_LENGTHTYPE_MM           = static_cast<unsigned>(SVGLengthType::Millimeters),
 50         SVG_LENGTHTYPE_IN           = static_cast<unsigned>(SVGLengthType::Inches),
 51         SVG_LENGTHTYPE_PT           = static_cast<unsigned>(SVGLengthType::Points),
 52         SVG_LENGTHTYPE_PC           = static_cast<unsigned>(SVGLengthType::Picas)
 53     };
 54 
 55     static Ref<SVGLength> create()
 56     {
 57         return adoptRef(*new SVGLength());
 58     }
 59 
 60     static Ref<SVGLength> create(const SVGLengthValue& value)
 61     {
 62         return adoptRef(*new SVGLength(value));
 63     }
 64 
 65     static Ref<SVGLength> create(SVGPropertyOwner* owner, SVGPropertyAccess access, const SVGLengthValue& value = { })
 66     {
 67         return adoptRef(*new SVGLength(owner, access, value));
 68     }
 69 
 70     template<typename T>
 71     static ExceptionOr<Ref<SVGLength>> create(ExceptionOr<T>&& value)
 72     {
 73         if (value.hasException())
 74             return value.releaseException();
 75         return adoptRef(*new SVGLength(value.releaseReturnValue()));
 76     }
 77 
 78     Ref<SVGLength> clone() const
 79     {
 80         return SVGLength::create(m_value);
 81     }
 82 
 83     unsigned short unitType()  const
 84     {
 85         return static_cast<unsigned>(m_value.lengthType());
 86     }
 87 
 88     ExceptionOr<float> valueForBindings()
 89     {
 90         return m_value.valueForBindings(SVGLengthContext { contextElement() });
 91     }
 92 
 93     ExceptionOr<void> setValueForBindings(float value)
 94     {
 95         if (isReadOnly())
 96             return Exception { NoModificationAllowedError };
 97 
 98         auto result = m_value.setValue(SVGLengthContext { contextElement() }, value);
 99         if (result.hasException())
100             return result;
101 
102         commitChange();
103         return result;
104     }
105 
106     float valueInSpecifiedUnits()
107     {
108         return m_value.valueInSpecifiedUnits();
109     }
110 
111     ExceptionOr<void> setValueInSpecifiedUnits(float valueInSpecifiedUnits)
112     {
113         if (isReadOnly())
114             return Exception { NoModificationAllowedError };
115 
116         m_value.setValueInSpecifiedUnits(valueInSpecifiedUnits);
117         commitChange();
118         return { };
119     }
120 
121     ExceptionOr<void> setValueAsString(const String& value)
122     {
123         if (isReadOnly())
124             return Exception { NoModificationAllowedError };
125 
126         auto result = m_value.setValueAsString(value);
127         if (result.hasException())
128             return result;
129 
130         commitChange();
131         return result;
132     }
133 
134     ExceptionOr<void> newValueSpecifiedUnits(unsigned short unitType, float valueInSpecifiedUnits)
135     {
136         if (isReadOnly())
137             return Exception { NoModificationAllowedError };
138 
139         if (unitType == SVG_LENGTHTYPE_UNKNOWN || unitType > SVG_LENGTHTYPE_PC)
140             return Exception { NotSupportedError };

141 
142         m_value = { valueInSpecifiedUnits, static_cast<SVGLengthType>(unitType), m_value.lengthMode() };
143         commitChange();
144         return { };
145     }
146 
147     ExceptionOr<void> convertToSpecifiedUnits(unsigned short unitType)
148     {
149         if (isReadOnly())
150             return Exception { NoModificationAllowedError };
151 
152         if (unitType == SVG_LENGTHTYPE_UNKNOWN || unitType > SVG_LENGTHTYPE_PC)
153             return Exception { NotSupportedError };
154 
155         auto result = m_value.convertToSpecifiedUnits(SVGLengthContext { contextElement() }, static_cast<SVGLengthType>(unitType));
156         if (result.hasException())
157             return result;
158 
159         commitChange();
160         return result;
161     }
162 
163     String valueAsString() const override
164     {
165         return m_value.valueAsString();
166     }
167 };
168 
169 } // namespace WebCore
