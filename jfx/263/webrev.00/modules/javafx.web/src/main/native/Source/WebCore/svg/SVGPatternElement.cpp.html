 48 inline SVGPatternElement::SVGPatternElement(const QualifiedName& tagName, Document& document)
 49     : SVGElement(tagName, document)
 50     , SVGFitToViewBox(this)
 51     , SVGTests(this)
 52     , SVGURIReference(this)
 53 {
 54     ASSERT(hasTagName(SVGNames::patternTag));
 55 
 56     static std::once_flag onceFlag;
 57     std::call_once(onceFlag, [] {
 58         PropertyRegistry::registerProperty<SVGNames::xAttr, &SVGPatternElement::m_x>();
 59         PropertyRegistry::registerProperty<SVGNames::yAttr, &SVGPatternElement::m_y>();
 60         PropertyRegistry::registerProperty<SVGNames::widthAttr, &SVGPatternElement::m_width>();
 61         PropertyRegistry::registerProperty<SVGNames::heightAttr, &SVGPatternElement::m_height>();
 62         PropertyRegistry::registerProperty<SVGNames::patternUnitsAttr, SVGUnitTypes::SVGUnitType, &SVGPatternElement::m_patternUnits>();
 63         PropertyRegistry::registerProperty<SVGNames::patternContentUnitsAttr, SVGUnitTypes::SVGUnitType, &SVGPatternElement::m_patternContentUnits>();
 64         PropertyRegistry::registerProperty<SVGNames::patternTransformAttr, &SVGPatternElement::m_patternTransform>();
 65     });
 66 }

 73 void SVGPatternElement::parseAttribute(const QualifiedName& name, const AtomString& value)
 74 {
 75     if (name == SVGNames::patternUnitsAttr) {
 76         auto propertyValue = SVGPropertyTraits<SVGUnitTypes::SVGUnitType>::fromString(value);
 77         if (propertyValue > 0)
 78             m_patternUnits->setBaseValInternal<SVGUnitTypes::SVGUnitType>(propertyValue);
 79         return;
 80     }
 81     if (name == SVGNames::patternContentUnitsAttr) {
 82         auto propertyValue = SVGPropertyTraits<SVGUnitTypes::SVGUnitType>::fromString(value);
 83         if (propertyValue > 0)
 84             m_patternContentUnits->setBaseValInternal<SVGUnitTypes::SVGUnitType>(propertyValue);
 85         return;
 86     }
 87     if (name == SVGNames::patternTransformAttr) {
 88         m_patternTransform->baseVal()->parse(value);
 89         return;
 90     }
 91 
 92     SVGParsingError parseError = NoError;
 93 
 94     if (name == SVGNames::xAttr)
 95         m_x->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
 96     else if (name == SVGNames::yAttr)
 97         m_y->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
 98     else if (name == SVGNames::widthAttr)
 99         m_width->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError, SVGLengthNegativeValuesMode::Forbid));
100     else if (name == SVGNames::heightAttr)
101         m_height->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError, SVGLengthNegativeValuesMode::Forbid));
102 
103     reportAttributeParsingError(parseError, name, value);
104 
105     SVGElement::parseAttribute(name, value);
106     SVGURIReference::parseAttribute(name, value);
107     SVGTests::parseAttribute(name, value);
108     SVGFitToViewBox::parseAttribute(name, value);
109 }

111 void SVGPatternElement::svgAttributeChanged(const QualifiedName& attrName)
112 {
113     if (PropertyRegistry::isAnimatedLengthAttribute(attrName)) {
114         InstanceInvalidationGuard guard(*this);
115         invalidateSVGPresentationAttributeStyle();
116         return;
117     }
118 
119     if (PropertyRegistry::isKnownAttribute(attrName) || SVGFitToViewBox::isKnownAttribute(attrName) || SVGURIReference::isKnownAttribute(attrName)) {
120         if (RenderObject* object = renderer())
121             object->setNeedsLayout();
122         return;
123     }
124 
125     SVGElement::svgAttributeChanged(attrName);
126 }
127 
128 void SVGPatternElement::childrenChanged(const ChildChange& change)
129 {
130     SVGElement::childrenChanged(change);
131 
132     if (change.source == ChildChangeSource::Parser)
133         return;
134 
135     if (RenderObject* object = renderer())
136         object->setNeedsLayout();
137 }

144 void SVGPatternElement::collectPatternAttributes(PatternAttributes& attributes) const
145 {
146     if (!attributes.hasX() && hasAttribute(SVGNames::xAttr))
147         attributes.setX(x());
148 
149     if (!attributes.hasY() && hasAttribute(SVGNames::yAttr))
150         attributes.setY(y());
151 
152     if (!attributes.hasWidth() && hasAttribute(SVGNames::widthAttr))
153         attributes.setWidth(width());
154 
155     if (!attributes.hasHeight() && hasAttribute(SVGNames::heightAttr))
156         attributes.setHeight(height());
157 
158     if (!attributes.hasViewBox() && hasAttribute(SVGNames::viewBoxAttr) && hasValidViewBox())
159         attributes.setViewBox(viewBox());
160 
161     if (!attributes.hasPreserveAspectRatio() && hasAttribute(SVGNames::preserveAspectRatioAttr))
162         attributes.setPreserveAspectRatio(preserveAspectRatio());
163 
164     if (!attributes.hasPatternUnits() && hasAttribute(SVGNames::patternUnitsAttr))
165         attributes.setPatternUnits(patternUnits());
166 
167     if (!attributes.hasPatternContentUnits() && hasAttribute(SVGNames::patternContentUnitsAttr))
168         attributes.setPatternContentUnits(patternContentUnits());
169 
170     if (!attributes.hasPatternTransform() && hasAttribute(SVGNames::patternTransformAttr))
171         attributes.setPatternTransform(patternTransform().concatenate());
172 
173     if (!attributes.hasPatternContentElement() && childElementCount())
174         attributes.setPatternContentElement(this);
175 }
176 
177 AffineTransform SVGPatternElement::localCoordinateSpaceTransform(SVGLocatable::CTMScope) const
178 {
179     return patternTransform().concatenate();
