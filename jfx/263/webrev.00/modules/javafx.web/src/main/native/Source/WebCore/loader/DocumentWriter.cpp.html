 56 static inline bool canReferToParentFrameEncoding(const Frame* frame, const Frame* parentFrame)
 57 {
 58     return parentFrame && parentFrame->document()->securityOrigin().canAccess(frame->document()->securityOrigin());
 59 }

 61 // This is only called by ScriptController::executeIfJavaScriptURL
 62 // and always contains the result of evaluating a javascript: url.
 63 // This is the <iframe src="javascript:'html'"> case.
 64 void DocumentWriter::replaceDocumentWithResultOfExecutingJavascriptURL(const String& source, Document* ownerDocument)
 65 {
 66     m_frame->loader().stopAllLoaders();
 67 
 68     // If we are in the midst of changing the frame's document, don't execute script
 69     // that modifies the document further:
 70     if (m_frame->documentIsBeingReplaced())
 71         return;
 72 
 73     begin(m_frame->document()->url(), true, ownerDocument);
 74 
 75     // begin() might fire an unload event, which will result in a situation where no new document has been attached,
 76     // and the old document has been detached. Therefore, bail out if no document is attached.
 77     if (!m_frame->document())
 78         return;
 79 
 80     if (!source.isNull()) {
 81         if (!m_hasReceivedSomeData) {
 82             m_hasReceivedSomeData = true;
 83             m_frame->document()->setCompatibilityMode(DocumentCompatibilityMode::NoQuirksMode);
 84         }
 85 
 86         // FIXME: This should call DocumentParser::appendBytes instead of append
 87         // to support RawDataDocumentParsers.
 88         if (DocumentParser* parser = m_frame->document()->parser())
 89             parser->append(source.impl());
 90     }
 91 
 92     end();
 93 }

 95 void DocumentWriter::clear()
 96 {
 97     m_decoder = nullptr;
 98     m_hasReceivedSomeData = false;
 99     if (!m_encodingWasChosenByUser)
100         m_encoding = String();

108 Ref<Document> DocumentWriter::createDocument(const URL& url)
109 {
110     if (!m_frame->loader().stateMachine().isDisplayingInitialEmptyDocument() && m_frame->loader().client().shouldAlwaysUsePluginDocument(m_mimeType))
111         return PluginDocument::create(*m_frame, url);
112 #if PLATFORM(IOS_FAMILY)
113     if (MIMETypeRegistry::isPDFMIMEType(m_mimeType) && (m_frame->isMainFrame() || !m_frame->settings().useImageDocumentForSubframePDF()))
114         return SinkDocument::create(*m_frame, url);
115 #endif
116     if (!m_frame->loader().client().hasHTMLView())
117         return Document::createNonRenderedPlaceholder(*m_frame, url);
118     return DOMImplementation::createDocument(m_mimeType, m_frame, url);
119 }

121 bool DocumentWriter::begin(const URL& urlReference, bool dispatch, Document* ownerDocument)
122 {
123     // We grab a local copy of the URL because it's easy for callers to supply
124     // a URL that will be deallocated during the execution of this function.
125     // For example, see <https://bugs.webkit.org/show_bug.cgi?id=66360>.
126     URL url = urlReference;
127 
128     // Create a new document before clearing the frame, because it may need to
129     // inherit an aliased security context.
130     Ref<Document> document = createDocument(url);
131 
132     // If the new document is for a Plugin but we're supposed to be sandboxed from Plugins,
133     // then replace the document with one whose parser will ignore the incoming data (bug 39323)
134     if (document->isPluginDocument() && document->isSandboxed(SandboxPlugins))
135         document = SinkDocument::create(*m_frame, url);
136 
137     // FIXME: Do we need to consult the content security policy here about blocked plug-ins?
138 
139     bool shouldReuseDefaultView = m_frame->loader().stateMachine().isDisplayingInitialEmptyDocument() && m_frame->document()->isSecureTransitionTo(url);
140 
141     // Temporarily extend the lifetime of the existing document so that FrameLoader::clear() doesn't destroy it as
142     // we need to retain its ongoing set of upgraded requests in new navigation contexts per <http://www.w3.org/TR/upgrade-insecure-requests/>
143     // and we may also need to inherit its Content Security Policy below.
144     RefPtr<Document> existingDocument = m_frame->document();
145 
146     WTF::Function<void()> handleDOMWindowCreation = [this, document = document.copyRef(), url] {
147         if (m_frame->loader().stateMachine().isDisplayingInitialEmptyDocument() && m_frame->document()->isSecureTransitionTo(url))
148             document->takeDOMWindowFrom(*m_frame->document());
149         else
150             document->createDOMWindow();
151     };
152 
153     m_frame->loader().clear(document.ptr(), !shouldReuseDefaultView, !shouldReuseDefaultView, true, WTFMove(handleDOMWindowCreation));
154     clear();
155 
156     // m_frame->loader().clear() might fire unload event which could remove the view of the document.
157     // Bail out if document has no view.
158     if (!document->view())
159         return false;
160 
161     if (!shouldReuseDefaultView)
162         m_frame->script().updatePlatformScriptObjects();
163 
164     m_frame->loader().setOutgoingReferrer(url);
165     m_frame->setDocument(document.copyRef());
166 
167     if (m_decoder)
168         document->setDecoder(m_decoder.get());
169     if (ownerDocument) {
170         // |document| is the result of evaluating a JavaScript URL.
171         document->setCookieURL(ownerDocument->cookieURL());
172         document->setSecurityOriginPolicy(ownerDocument->securityOriginPolicy());
173         document->setStrictMixedContentMode(ownerDocument->isStrictMixedContentMode());
174 
175         document->setContentSecurityPolicy(makeUnique<ContentSecurityPolicy>(URL { url }, document));
176         document->contentSecurityPolicy()->copyStateFrom(ownerDocument->contentSecurityPolicy());
177         document->contentSecurityPolicy()->setInsecureNavigationRequestsToUpgrade(ownerDocument->contentSecurityPolicy()->takeNavigationRequestsToUpgrade());
178     } else if (existingDocument) {
179         if (url.protocolIsData() || url.protocolIsBlob()) {
180             document->setContentSecurityPolicy(makeUnique<ContentSecurityPolicy>(URL { url }, document));
181             document->contentSecurityPolicy()->copyStateFrom(existingDocument->contentSecurityPolicy());
182 
183             // Fix up 'self' for blob: and data:, which is inherited from its embedding document or opener.
184             auto* parentFrame = m_frame->tree().parent();
185             if (auto* ownerFrame = parentFrame ? parentFrame : m_frame->loader().opener())
186                 document->contentSecurityPolicy()->updateSourceSelf(ownerFrame->document()->securityOrigin());
187         }
188         document->contentSecurityPolicy()->setInsecureNavigationRequestsToUpgrade(existingDocument->contentSecurityPolicy()->takeNavigationRequestsToUpgrade());
189     }
190 
191     m_frame->loader().didBeginDocument(dispatch);
192 
193     document->implicitOpen();
194 
195     // We grab a reference to the parser so that we'll always send data to the
196     // original parser, even if the document acquires a new parser (e.g., via
197     // document.open).
198     m_parser = document->parser();
199 
200     if (m_frame->view() && m_frame->loader().client().hasHTMLView())
201         m_frame->view()->setContentsSize(IntSize());
202 
203     m_state = State::Started;
204     return true;
205 }

207 TextResourceDecoder& DocumentWriter::decoder()
208 {
209     if (!m_decoder) {
210         m_decoder = TextResourceDecoder::create(m_mimeType,
211             m_frame->settings().defaultTextEncodingName(),
212             m_frame->settings().usesEncodingDetector());
213         Frame* parentFrame = m_frame->tree().parent();
214         // Set the hint encoding to the parent frame encoding only if
215         // the parent and the current frames share the security origin.
216         // We impose this condition because somebody can make a child frame
217         // containing a carefully crafted html/javascript in one encoding
218         // that can be mistaken for hintEncoding (or related encoding) by
219         // an auto detector. When interpreted in the latter, it could be
220         // an attack vector.
221         // FIXME: This might be too cautious for non-7bit-encodings and
222         // we may consider relaxing this later after testing.
223         if (canReferToParentFrameEncoding(m_frame, parentFrame))
224             m_decoder->setHintEncoding(parentFrame->document()->decoder());
225         if (m_encoding.isEmpty()) {
226             if (canReferToParentFrameEncoding(m_frame, parentFrame))
227                 m_decoder->setEncoding(parentFrame->document()->textEncoding(), TextResourceDecoder::EncodingFromParentFrame);
228         } else {
229             m_decoder->setEncoding(m_encoding,
230                 m_encodingWasChosenByUser ? TextResourceDecoder::UserChosenEncoding : TextResourceDecoder::EncodingFromHTTPHeader);
231         }
232         m_frame->document()->setDecoder(m_decoder.get());
233     }
234     return *m_decoder;
235 }

237 void DocumentWriter::reportDataReceived()
238 {
239     ASSERT(m_decoder);
240     if (m_hasReceivedSomeData)
241         return;
242     m_hasReceivedSomeData = true;
243     if (m_decoder->encoding().usesVisualOrdering())
244         m_frame->document()->setVisuallyOrdered();
245     m_frame->document()->resolveStyle(Document::ResolveStyleType::Rebuild);
246 }

248 void DocumentWriter::addData(const char* bytes, size_t length)
249 {
250     // FIXME: Change these to ASSERT once https://bugs.webkit.org/show_bug.cgi?id=80427 has been resolved.
251     RELEASE_ASSERT(m_state != State::NotStarted);
252     if (m_state == State::Finished) {
253         ASSERT_NOT_REACHED();
254         return;
255     }
256     ASSERT(m_parser);
257     m_parser->appendBytes(*this, bytes, length);
258 }
259 
260 void DocumentWriter::insertDataSynchronously(const String& markup)
261 {
262     ASSERT(m_state != State::NotStarted);
263     ASSERT(m_state != State::Finished);
264     ASSERT(m_parser);
265     m_parser->insert(markup);
266 }

268 void DocumentWriter::end()
269 {
270     ASSERT(m_frame->page());
271     ASSERT(m_frame->document());
272 
273     // The parser is guaranteed to be released after this point. begin() would
274     // have to be called again before we can start writing more data.
275     m_state = State::Finished;
276 
277     // http://bugs.webkit.org/show_bug.cgi?id=10854
278     // The frame's last ref may be removed and it can be deleted by checkCompleted(),
279     // so we'll add a protective refcount
280     Ref<Frame> protect(*m_frame);
281 
282     if (!m_parser)
283         return;
284     // FIXME: m_parser->finish() should imply m_parser->flush().
285     m_parser->flush(*this);
286     if (!m_parser)
287         return;
288     m_parser->finish();
289     m_parser = nullptr;
290 }
