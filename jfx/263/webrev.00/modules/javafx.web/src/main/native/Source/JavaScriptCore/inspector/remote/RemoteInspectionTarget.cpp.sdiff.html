41 void RemoteInspectionTarget::setRemoteDebuggingAllowed(bool allowed)
42 {
43     if (m_allowed == allowed)
44         return;
45 
46     m_allowed = allowed;
47 
48     if (m_allowed && automaticInspectionAllowed())
49         RemoteInspector::singleton().updateAutomaticInspectionCandidate(this);
50     else
51         RemoteInspector::singleton().updateTarget(this);
52 }

54 void RemoteInspectionTarget::pauseWaitingForAutomaticInspection()

56     ASSERT(targetIdentifier());

58     ASSERT(automaticInspectionAllowed());
59 
60     EventLoop loop;
61     while (RemoteInspector::singleton().waitingForAutomaticInspection(targetIdentifier()) && !loop.ended())
62         loop.cycle();

55 void RemoteInspectionTarget::pauseWaitingForAutomaticInspection()
56 {
57     ASSERT(targetIdentifier());
58     ASSERT(m_allowed);
59     ASSERT(automaticInspectionAllowed());
60 
61     while (RemoteInspector::singleton().waitingForAutomaticInspection(targetIdentifier())) {
62         if (RunLoop::cycle(JSGlobalObjectScriptDebugServer::runLoopMode()) == RunLoop::CycleResult::Stop)
63             break;
64     }
65 }
