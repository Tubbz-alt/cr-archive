 42 namespace Layout {
 43 
 44 class Container;
 45 class LayoutState;
 46 class TreeBuilder;
 47 
 48 class Box : public CanMakeWeakPtr<Box> {
 49     WTF_MAKE_ISO_ALLOCATED(Box);
 50 public:
 51     enum class ElementType {
 52         Document,
 53         Body,
 54         TableWrapperBox, // The table generates a principal block container box called the table wrapper box that contains the table box and any caption boxes.
 55         TableBox, // The table box is a block-level box that contains the table's internal table boxes.
 56         Image,
 57         IFrame,
 58         HardLineBreak,
 59         GenericElement
 60     };
 61 
 62     struct ElementAttributes {
 63         ElementType elementType;
 64     };
 65 
 66     enum BaseTypeFlag {
 67         BoxFlag               = 1 << 0,
 68         ContainerFlag         = 1 << 1
 69     };
 70     typedef unsigned BaseTypeFlags;
 71 
 72     Box(Optional<ElementAttributes>, RenderStyle&&);
 73     Box(TextContext&&, RenderStyle&&);
 74     ~Box();
 75 
 76     bool establishesFormattingContext() const;
 77     bool establishesBlockFormattingContext() const;
 78     bool establishesInlineFormattingContext() const;
 79     bool establishesTableFormattingContext() const;
 80     bool establishesIndependentFormattingContext() const;
 81 
 82     bool isInFlow() const { return !isFloatingOrOutOfFlowPositioned(); }
 83     bool isPositioned() const { return isInFlowPositioned() || isOutOfFlowPositioned(); }
 84     bool isInFlowPositioned() const { return isRelativelyPositioned() || isStickyPositioned(); }
 85     bool isOutOfFlowPositioned() const { return isAbsolutelyPositioned(); }
 86     bool isRelativelyPositioned() const;
 87     bool isStickyPositioned() const;
 88     bool isAbsolutelyPositioned() const;
 89     bool isFixedPositioned() const;
 90     bool isFloatingPositioned() const;
 91     bool isLeftFloatingPositioned() const;
 92     bool isRightFloatingPositioned() const;
 93     bool hasFloatClear() const;
 94     bool isFloatAvoider() const;
 95 
 96     bool isFloatingOrOutOfFlowPositioned() const { return isFloatingPositioned() || isOutOfFlowPositioned(); }
 97 
 98     const Container* containingBlock() const;
 99     const Container& formattingContextRoot() const;
100     const Container& initialContainingBlock() const;
101 
102     bool isDescendantOf(const Container&) const;
103     bool isContainingBlockDescendantOf(const Container&) const;
104 
105     bool isAnonymous() const { return m_isAnonymous; }
106 
107     bool isBlockLevelBox() const;
108     bool isInlineLevelBox() const;
109     bool isInlineBox() const;
110     bool isAtomicInlineLevelBox() const;
111     bool isInlineBlockBox() const;
112     bool isInlineTableBox() const;
113     bool isBlockContainerBox() const;
114     bool isInitialContainingBlock() const;
115 
116     bool isDocumentBox() const { return m_elementAttributes && m_elementAttributes.value().elementType == ElementType::Document; }
117     bool isBodyBox() const { return m_elementAttributes && m_elementAttributes.value().elementType == ElementType::Body; }
118     bool isTableWrapperBox() const { return m_elementAttributes && m_elementAttributes.value().elementType == ElementType::TableWrapperBox; }
119     bool isTableBox() const { return m_elementAttributes && m_elementAttributes.value().elementType == ElementType::TableBox; }
120     bool isTableCaption() const { return style().display() == DisplayType::TableCaption; }
121     bool isTableHeader() const { return style().display() == DisplayType::TableHeaderGroup; }
122     bool isTableBody() const { return style().display() == DisplayType::TableRowGroup; }
123     bool isTableFooter() const { return style().display() == DisplayType::TableFooterGroup; }
124     bool isTableRow() const { return style().display() == DisplayType::TableRow; }
125     bool isTableColumnGroup() const { return style().display() == DisplayType::TableColumnGroup; }
126     bool isTableColumn() const { return style().display() == DisplayType::TableColumn; }
127     bool isTableCell() const { return style().display() == DisplayType::TableCell; }
128     bool isReplaced() const { return isImage() || isIFrame(); }
129     bool isIFrame() const { return m_elementAttributes && m_elementAttributes.value().elementType == ElementType::IFrame; }
130     bool isImage() const { return m_elementAttributes && m_elementAttributes.value().elementType == ElementType::Image; }
131     bool isLineBreakBox() const { return m_elementAttributes && m_elementAttributes.value().elementType == ElementType::HardLineBreak; }
132 
133     const Container* parent() const { return m_parent; }
134     const Box* nextSibling() const { return m_nextSibling; }
135     const Box* nextInFlowSibling() const;
136     const Box* nextInFlowOrFloatingSibling() const;
137     const Box* previousSibling() const { return m_previousSibling; }
138     const Box* previousInFlowSibling() const;
139     const Box* previousInFlowOrFloatingSibling() const;
140 
141     // FIXME: This is currently needed for style updates.
142     Box* nextSibling() { return m_nextSibling; }
143 
144     bool isContainer() const { return m_baseTypeFlags & ContainerFlag; }
145 
146     bool isPaddingApplicable() const;
147     bool isOverflowVisible() const;
148 
149     void updateStyle(const RenderStyle& newStyle);
150     const RenderStyle& style() const { return m_style; }
151 
152     const Replaced* replaced() const;
153     // FIXME: Temporary until after intrinsic size change is tracked by Replaced.
154     Replaced* replaced();
155     bool hasTextContent() const { return !!m_textContext; }
156     const Optional<TextContext>& textContext() const { return m_textContext; }
157 
158     // FIXME: Find a better place for random DOM things.
159     void setRowSpan(unsigned);
160     unsigned rowSpan() const;
161 
162     void setColumnSpan(unsigned);
163     unsigned columnSpan() const;
164 
165     void setColumnWidth(LayoutUnit);
166     Optional<LayoutUnit> columnWidth() const;
167 
168     void setParent(Container& parent) { m_parent = &parent; }
169     void setNextSibling(Box& nextSibling) { m_nextSibling = &nextSibling; }
170     void setPreviousSibling(Box& previousSibling) { m_previousSibling = &previousSibling; }
171 
172     void setIsAnonymous() { m_isAnonymous = true; }
173 
174     bool canCacheForLayoutState(const LayoutState&) const;
175     Display::Box* cachedDisplayBoxForLayoutState(const LayoutState&) const;
176     void setCachedDisplayBoxForLayoutState(LayoutState&, std::unique_ptr<Display::Box>) const;
177 
178 protected:
179     Box(Optional<ElementAttributes>, Optional<TextContext>, RenderStyle&&, BaseTypeFlags);
180 
181 private:
182     class BoxRareData {
183         WTF_MAKE_FAST_ALLOCATED;
184     public:
185         BoxRareData() = default;
186 
187         std::unique_ptr<Replaced> replaced;
188         unsigned rowSpan { 1 };
189         unsigned columnSpan { 1 };
190         Optional<LayoutUnit> columnWidth;
191     };
192 
193     bool hasRareData() const { return m_hasRareData; }
194     void setHasRareData(bool hasRareData) { m_hasRareData = hasRareData; }
195     const BoxRareData& rareData() const;
196     BoxRareData& ensureRareData();
197     void removeRareData();
198 
199     typedef HashMap<const Box*, std::unique_ptr<BoxRareData>> RareDataMap;
200 
201     static RareDataMap& rareDataMap();
202 
203     RenderStyle m_style;
204     Optional<ElementAttributes> m_elementAttributes;
205 
206     Container* m_parent { nullptr };
207     Box* m_previousSibling { nullptr };
208     Box* m_nextSibling { nullptr };
209 
210     const Optional<TextContext> m_textContext;
211 
212     // First LayoutState gets a direct cache.
213     mutable WeakPtr<LayoutState> m_cachedLayoutState;
214     mutable std::unique_ptr<Display::Box> m_cachedDisplayBoxForLayoutState;
215 
216     unsigned m_baseTypeFlags : 6;
217     bool m_hasRareData : 1;
218     bool m_isAnonymous : 1;
219 };
220 
221 }
222 }
