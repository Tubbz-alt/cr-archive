26 #ifndef BumpAllocator_h
27 #define BumpAllocator_h
28 
29 #include "BAssert.h"
30 #include "BumpRange.h"
31 #include "ObjectType.h"
32 
33 namespace bmalloc {
34 
35 // Helper object for allocating small objects.
36 
37 class BumpAllocator {
38 public:
39     BumpAllocator();
40     void init(size_t);
41 
42     size_t size() { return m_size; }
43 
44     bool isNull() { return !m_ptr; }
45     void clear();
46 
47     bool canAllocate() { return !!m_remaining; }
48     void* allocate();
49 
50     void refill(const BumpRange&);
51 
52 private:
53     char* m_ptr;
54     unsigned m_size;
55     unsigned m_remaining;
56 };
57 
58 inline BumpAllocator::BumpAllocator()
59     : m_ptr()
60     , m_size()
61     , m_remaining()
62 {
63 }
64 
65 inline void BumpAllocator::init(size_t size)
66 {
67     m_ptr = nullptr;
68     m_size = size;
69     m_remaining = 0;
70 }
71 
72 inline void* BumpAllocator::allocate()
73 {
74     BASSERT(m_remaining);
75 
76     --m_remaining;
77     char* result = m_ptr;
78     m_ptr += m_size;
79     return result;
80 }
81 
82 inline void BumpAllocator::refill(const BumpRange& bumpRange)
83 {
84     BASSERT(!canAllocate());
85     m_ptr = bumpRange.begin;
86     m_remaining = bumpRange.objectCount;
87     BASSERT(canAllocate());
88 }
89 
90 inline void BumpAllocator::clear()
91 {
92     m_ptr = nullptr;
93     m_remaining = 0;
94 }
95 
96 } // namespace bmalloc
97 
98 #endif // BumpAllocator_h
