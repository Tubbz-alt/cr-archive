26 #include "config.h"

29 #if ENABLE(LAYOUT_FORMATTING_CONTEXT)
30 
31 #include <wtf/NeverDestroyed.h>
32 
33 namespace WebCore {
34 namespace Layout {
35 
36 static Phase& phase()
37 {
38     static NeverDestroyed<Phase> phase;
39     return phase;
40 }
41 
42 bool Phase::isInTreeBuilding()
43 {
44     return *phase() && (*phase()).value() == Type::TreeBuilding;
45 }
46 
47 bool Phase::isInLayout()
48 {
49     return *phase() && (*phase()).value() == Type::Layout;
50 }
51 
52 bool Phase::isInInvalidation()
53 {
54     return *phase() && (*phase()).value() == Type::Invalidation;
55 }
56 
57 PhaseScope::PhaseScope(Phase::Type type)
58 {
59     // Should never nest states like calling TreeBuilding from Layout.
60     ASSERT(!(*phase()).hasValue());
61     phase().set(type);
62 }
63 
64 PhaseScope::~PhaseScope()
65 {
66     phase().reset();
67 }
68 
69 }
70 }
