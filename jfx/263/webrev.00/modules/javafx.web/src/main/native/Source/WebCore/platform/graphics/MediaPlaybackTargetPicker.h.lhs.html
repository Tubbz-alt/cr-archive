26 #ifndef MediaPlaybackTargetPicker_h
27 #define MediaPlaybackTargetPicker_h
28 
29 #if ENABLE(WIRELESS_PLAYBACK_TARGET)
30 
31 #include <wtf/Ref.h>
32 #include <wtf/RunLoop.h>
33 
34 namespace WebCore {
35 
36 class FloatRect;
37 class MediaPlaybackTarget;
38 
39 class MediaPlaybackTargetPicker {
40 public:
41     class Client {
42     protected:
43         virtual ~Client() = default;
44 
45     public:
46         virtual void setPlaybackTarget(Ref<MediaPlaybackTarget>&&) = 0;
47         virtual void externalOutputDeviceAvailableDidChange(bool) = 0;

48 
49         void invalidate();
50     };
51 
52     virtual ~MediaPlaybackTargetPicker();
53 
54     virtual void showPlaybackTargetPicker(const FloatRect&, bool checkActiveRoute, bool useDarkAppearance);
55     virtual void startingMonitoringPlaybackTargets();
56     virtual void stopMonitoringPlaybackTargets();
57     virtual void invalidatePlaybackTargets();
58 
59     void availableDevicesDidChange() { addPendingAction(OutputDeviceAvailabilityChanged); }
60     void currentDeviceDidChange() { addPendingAction(CurrentDeviceDidChange); }

61 
62 protected:
63     explicit MediaPlaybackTargetPicker(Client&);
64 
65     enum ActionType {
66         OutputDeviceAvailabilityChanged = 1 << 0,
67         CurrentDeviceDidChange = 1 << 1,

68     };
69     typedef unsigned PendingActionFlags;
70 
71     void addPendingAction(PendingActionFlags);
72     void pendingActionTimerFired();
73     Client* client() const { return m_client; }
74     void setClient(Client* client) { m_client = client; }
75 
76 private:
77     virtual bool externalOutputDeviceAvailable() = 0;
78     virtual Ref<MediaPlaybackTarget> playbackTarget() = 0;
79 
80     PendingActionFlags m_pendingActionFlags { 0 };
81     Client* m_client;
82     RunLoop::Timer<MediaPlaybackTargetPicker> m_pendingActionTimer;
83 };
84 
85 } // namespace WebCore
86 
87 #endif // ENABLE(WIRELESS_PLAYBACK_TARGET)
88 
89 #endif // MediaPlaybackTargetPicker_h
