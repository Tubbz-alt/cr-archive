26 #include "config.h"
27 #include "FastMallocAlignedMemoryAllocator.h"
28 
29 #include <mutex>
30 #include <wtf/FastMalloc.h>

32 namespace JSC {
33 
34 FastMallocAlignedMemoryAllocator::FastMallocAlignedMemoryAllocator()
35 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
36     : m_heap("WebKit FastMallocAlignedMemoryAllocator")
37 #endif
38 {
39 }
40 
41 FastMallocAlignedMemoryAllocator::~FastMallocAlignedMemoryAllocator()
42 {
43 }
44 
45 void* FastMallocAlignedMemoryAllocator::tryAllocateAlignedMemory(size_t alignment, size_t size)
46 {
47 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
48     return m_heap.memalign(alignment, size, true);
49 #else
50     return tryFastAlignedMalloc(alignment, size);
51 #endif
52 
53 }
54 
55 void FastMallocAlignedMemoryAllocator::freeAlignedMemory(void* basePtr)
56 {
57 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
58     return m_heap.free(basePtr);
59 #else
60     fastAlignedFree(basePtr);
61 #endif
62 
63 }
64 
65 void FastMallocAlignedMemoryAllocator::dump(PrintStream& out) const
66 {
67     out.print("FastMalloc");
68 }
69 
70 void* FastMallocAlignedMemoryAllocator::tryAllocateMemory(size_t size)
71 {
72 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
73     return m_heap.malloc(size);
74 #else
75     return FastMalloc::tryMalloc(size);
76 #endif
77 }
78 
79 void FastMallocAlignedMemoryAllocator::freeMemory(void* pointer)
80 {
81 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
82     return m_heap.free(pointer);
83 #else
84     FastMalloc::free(pointer);
85 #endif
86 }
87 
88 void* FastMallocAlignedMemoryAllocator::tryReallocateMemory(void* pointer, size_t size)
89 {
90 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
91     return m_heap.realloc(pointer, size);
92 #else
93     return FastMalloc::tryRealloc(pointer, size);
94 #endif
95 }
96 
97 } // namespace JSC
