45 void JSNativeStdFunction::visitChildren(JSCell* cell, SlotVisitor& visitor)
46 {
47     JSNativeStdFunction* thisObject = jsCast<JSNativeStdFunction*>(cell);

49     Base::visitChildren(thisObject, visitor);

50 }

52 void JSNativeStdFunction::finishCreation(VM& vm, NativeExecutable* executable, int length, const String& name)
53 {
54     Base::finishCreation(vm, executable, length, name);
55     ASSERT(inherits(vm, info()));

56 }
57 
58 static EncodedJSValue JSC_HOST_CALL runStdFunction(JSGlobalObject* globalObject, CallFrame* callFrame)
59 {
60     JSNativeStdFunction* function = jsCast<JSNativeStdFunction*>(callFrame->jsCallee());
61     ASSERT(function);
62     return function->function()(globalObject, callFrame);
63 }
64 
65 JSNativeStdFunction* JSNativeStdFunction::create(VM& vm, JSGlobalObject* globalObject, int length, const String& name, NativeStdFunction&& nativeStdFunction, Intrinsic intrinsic, NativeFunction nativeConstructor)
66 {
67     NativeExecutable* executable = vm.getHostFunction(runStdFunction, intrinsic, nativeConstructor, nullptr, name);

68     Structure* structure = globalObject->nativeStdFunctionStructure();
69     JSNativeStdFunction* function = new (NotNull, allocateCell<JSNativeStdFunction>(vm.heap)) JSNativeStdFunction(vm, executable, globalObject, structure, WTFMove(nativeStdFunction));
70     function->finishCreation(vm, executable, length, name);
71     return function;
72 }
