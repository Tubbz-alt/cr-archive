37 Ref<WebGLVertexArrayObjectOES> WebGLVertexArrayObjectOES::create(WebGLRenderingContextBase& context, Type type)
38 {
39     return adoptRef(*new WebGLVertexArrayObjectOES(context, type));
40 }
41 
42 WebGLVertexArrayObjectOES::WebGLVertexArrayObjectOES(WebGLRenderingContextBase& context, Type type)
43     : WebGLVertexArrayObjectBase(context, type)
44 {
45     switch (type) {
46     case Type::Default:
47         break;
48     case Type::User:
49         setObject(this->context()->graphicsContextGL()->getExtensions().createVertexArrayOES());
50         break;
51     }
52 }
53 
54 WebGLVertexArrayObjectOES::~WebGLVertexArrayObjectOES()
55 {
56     deleteObject(nullptr);
57 }
58 
59 void WebGLVertexArrayObjectOES::deleteObjectImpl(GraphicsContextGLOpenGL* context3d, PlatformGLObject object)
60 {
61     switch (m_type) {
62     case Type::Default:
63         break;
64     case Type::User:
65         context3d->getExtensions().deleteVertexArrayOES(object);
66         break;
67     }
68 
69     if (m_boundElementArrayBuffer)
70         m_boundElementArrayBuffer->onDetached(context3d);
71 
72     for (auto& state : m_vertexAttribState) {
73         if (state.bufferBinding)
74             state.bufferBinding->onDetached(context3d);
75     }
76 }
77 }
