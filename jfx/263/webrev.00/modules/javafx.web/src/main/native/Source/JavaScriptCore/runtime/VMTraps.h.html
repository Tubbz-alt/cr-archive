 43 class VMTraps {
 44     typedef uint8_t BitField;
 45 public:
 46     enum class Error {
 47         None,
 48         LockUnavailable,
 49         NotJITCode
 50     };
 51 
 52     enum EventType {
 53         // Sorted in servicing priority order from highest to lowest.
 54         NeedDebuggerBreak,
 55         NeedShellTimeoutCheck,
 56         NeedTermination,
 57         NeedWatchdogCheck,
 58         NumberOfEventTypes, // This entry must be last in this list.
 59         Invalid
 60     };
 61 
 62     class Mask {
 63     public:
 64         enum AllEventTypes { AllEventTypesTag };
 65         constexpr Mask(AllEventTypes)
 66             : m_mask(std::numeric_limits<BitField>::max())
 67         { }
 68         static constexpr Mask allEventTypes() { return Mask(AllEventTypesTag); }
 69 
 70         constexpr Mask(const Mask&) = default;
 71         constexpr Mask(Mask&&) = default;
 72 
 73         template<typename... Arguments>
 74         constexpr Mask(Arguments... args)
 75             : m_mask(0)
 76         {
 77             init(args...);
 78         }
 79 
 80         BitField bits() const { return m_mask; }
 81 
 82     private:
 83         template<typename... Arguments>
 84         constexpr void init(EventType eventType, Arguments... args)
 85         {
 86             ASSERT(eventType < NumberOfEventTypes);
 87             m_mask |= (1 << eventType);
 88             init(args...);
 89         }
 90 
 91         constexpr void init() { }
 92 
 93         BitField m_mask;
 94     };
 95 
 96     static constexpr Mask interruptingTraps() { return Mask(NeedShellTimeoutCheck, NeedTermination, NeedWatchdogCheck); }
 97 
 98     ~VMTraps();
 99     VMTraps();
100 
101     void willDestroyVM();
102 
103     bool needTrapHandling() { return m_needTrapHandling; }
104     bool needTrapHandling(Mask mask) { return m_needTrapHandling & mask.bits(); }
105     void* needTrapHandlingAddress() { return &m_needTrapHandling; }
106 
107     void notifyGrabAllLocks()
108     {
109         if (needTrapHandling())
110             invalidateCodeBlocksOnStack();
111     }
112 
113     JS_EXPORT_PRIVATE void fireTrap(EventType);
114 
115     void handleTraps(JSGlobalObject*, CallFrame*, VMTraps::Mask);
116 
117     void tryInstallTrapBreakpoints(struct SignalContext&, StackBounds);
118 
119 private:
120     VM& vm() const;
121 
122     bool hasTrapForEvent(Locker<Lock>&, EventType eventType, Mask mask)
123     {
124         ASSERT(eventType < NumberOfEventTypes);
125         return (m_trapsBitField & mask.bits() & (1 << eventType));
126     }
127     void setTrapForEvent(Locker<Lock>&, EventType eventType)
128     {
129         ASSERT(eventType < NumberOfEventTypes);
130         m_trapsBitField |= (1 << eventType);
131     }
132     void clearTrapForEvent(Locker<Lock>&, EventType eventType)
133     {
134         ASSERT(eventType < NumberOfEventTypes);
135         m_trapsBitField &= ~(1 << eventType);
136     }
137 
138     EventType takeTopPriorityTrap(Mask);
139 
140 #if ENABLE(SIGNAL_BASED_VM_TRAPS)
141     class SignalSender;
142     friend class SignalSender;
143 
144     void invalidateCodeBlocksOnStack();
145     void invalidateCodeBlocksOnStack(CallFrame* topCallFrame);
146     void invalidateCodeBlocksOnStack(Locker<Lock>& codeBlockSetLocker, CallFrame* topCallFrame);
147 
148     void addSignalSender(SignalSender*);
149     void removeSignalSender(SignalSender*);
150 #else
151     friend class SignalSender;
152     void invalidateCodeBlocksOnStack() { }
153     void invalidateCodeBlocksOnStack(CallFrame*) { }
154 #endif
155 
156     Box<Lock> m_lock;
157     Ref<AutomaticThreadCondition> m_condition;
158     union {
159         BitField m_needTrapHandling { 0 };
160         BitField m_trapsBitField;
161     };
162     bool m_needToInvalidatedCodeBlocks { false };
163     bool m_isShuttingDown { false };
164 
165 #if ENABLE(SIGNAL_BASED_VM_TRAPS)
166     RefPtr<SignalSender> m_signalSender;
167 #endif
168 
169     friend class LLIntOffsetsExtractor;
170 };
