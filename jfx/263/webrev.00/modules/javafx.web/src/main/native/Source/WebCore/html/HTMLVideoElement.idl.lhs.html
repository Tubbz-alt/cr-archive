26 [
27     Conditional=VIDEO,

28     JSGenerateToNativeObject,
29 ] interface HTMLVideoElement : HTMLMediaElement {
30     [CEReactions=NotNeeded, Reflect] attribute unsigned long width;
31     [CEReactions=NotNeeded, Reflect] attribute unsigned long height;
32     readonly attribute unsigned long videoWidth;
33     readonly attribute unsigned long videoHeight;
34     [CEReactions=NotNeeded, Reflect, URL] attribute USVString poster;
35 
36     [CEReactions=NotNeeded, Reflect] attribute boolean playsInline;
37     readonly attribute boolean webkitSupportsFullscreen;
38     readonly attribute boolean webkitDisplayingFullscreen;
39 
40     [MayThrowException] void webkitEnterFullscreen();
41     void webkitExitFullscreen();
42 
43     // Maintain "FullScreen" capitalization scheme for backwards compatibility.
44     // https://bugs.webkit.org/show_bug.cgi?id=36081
45     [ImplementedAs=webkitEnterFullscreen, MayThrowException] void webkitEnterFullScreen();
46     [ImplementedAs=webkitExitFullscreen] void webkitExitFullScreen();
47 
48     [Conditional=WIRELESS_PLAYBACK_TARGET] attribute boolean webkitWirelessVideoPlaybackDisabled;
49 
50     // The number of frames that have been decoded and made available for playback.
51     [Conditional=MEDIA_STATISTICS] readonly attribute unsigned long webkitDecodedFrameCount;
52 
53     // The number of decoded frames that have been dropped by the player for performance reasons during playback.
54     [Conditional=MEDIA_STATISTICS] readonly attribute unsigned long webkitDroppedFrameCount;
55 
56     [Conditional=VIDEO_PRESENTATION_MODE] boolean webkitSupportsPresentationMode(VideoPresentationMode mode);
57     [Conditional=VIDEO_PRESENTATION_MODE] readonly attribute VideoPresentationMode webkitPresentationMode;
58     [Conditional=VIDEO_PRESENTATION_MODE] void webkitSetPresentationMode(VideoPresentationMode mode);
59 };
60 
61 [Conditional=VIDEO_PRESENTATION_MODE] enum VideoPresentationMode { "fullscreen", "picture-in-picture", "inline" };
