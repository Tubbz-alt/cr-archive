21 #pragma once
22 
23 #include "InternalFunction.h"
24 
25 namespace JSC {
26 
27 class NumberPrototype;
28 class GetterSetter;
29 
30 class NumberConstructor final : public InternalFunction {
31 public:
32     typedef InternalFunction Base;
33     static const unsigned StructureFlags = Base::StructureFlags | ImplementsHasInstance | HasStaticPropertyTable;
34 
35     static NumberConstructor* create(VM& vm, Structure* structure, NumberPrototype* numberPrototype, GetterSetter*)
36     {
37         NumberConstructor* constructor = new (NotNull, allocateCell<NumberConstructor>(vm.heap)) NumberConstructor(vm, structure);
38         constructor->finishCreation(vm, numberPrototype);
39         return constructor;
40     }
41 
42     DECLARE_INFO;
43 
44     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue proto)
45     {
46         return Structure::create(vm, globalObject, proto, TypeInfo(InternalFunctionType, StructureFlags), info());
47     }
48 
49     static bool isIntegerImpl(JSValue value)
50     {
51         if (value.isInt32())
52             return true;
53         if (!value.isDouble())
54             return false;
55 
56         double number = value.asDouble();
57         return std::isfinite(number) && trunc(number) == number;
58     }
59 
60 protected:
61     void finishCreation(VM&, NumberPrototype*);
62 
63 private:
64     NumberConstructor(VM&, Structure*);
65 };
66 
67 } // namespace JSC
