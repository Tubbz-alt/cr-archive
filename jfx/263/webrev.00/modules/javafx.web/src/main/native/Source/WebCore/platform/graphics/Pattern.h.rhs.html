 57 namespace WebCore {
 58 
 59 class AffineTransform;
 60 class GraphicsContext;
 61 class Image;
 62 class ImageHandle;
 63 
 64 class Pattern final : public RefCounted<Pattern> {
 65 public:
 66     WEBCORE_EXPORT static Ref<Pattern> create(Ref<Image>&& tileImage, bool repeatX, bool repeatY);
 67     WEBCORE_EXPORT ~Pattern();
 68 
 69     Image& tileImage() const { return m_tileImage.get(); }
 70 
 71     // Pattern space is an abstract space that maps to the default user space by the transformation 'userSpaceTransformation'
 72 #if !USE(DIRECT2D)
 73     PlatformPatternPtr createPlatformPattern(const AffineTransform& userSpaceTransformation) const;
 74 #else

 75     PlatformPatternPtr createPlatformPattern(const GraphicsContext&, float alpha, const AffineTransform& userSpaceTransformation) const;
 76 #endif
 77     WEBCORE_EXPORT void setPatternSpaceTransform(const AffineTransform& patternSpaceTransformation);
 78     const AffineTransform& patternSpaceTransform() const { return m_patternSpaceTransformation; };
 79     bool repeatX() const { return m_repeatX; }
 80     bool repeatY() const { return m_repeatY; }
 81 
 82 #if PLATFORM(JAVA)
 83     const AffineTransform& getPatternSpaceTransform() const { return m_patternSpaceTransformation; }
 84 #endif
 85 
 86     template<class Encoder> void encode(Encoder&) const;
 87     template<class Decoder> static Optional<Ref<Pattern>> decode(Decoder&);
 88 
 89 private:
 90     Pattern(Ref<Image>&&, bool repeatX, bool repeatY);
 91 
 92     Ref<Image> m_tileImage;
 93     AffineTransform m_patternSpaceTransformation;
 94     bool m_repeatX;
 95     bool m_repeatY;
 96 };
 97 
 98 template<class Encoder>
 99 void Pattern::encode(Encoder& encoder) const
100 {
101     ImageHandle imageHandle;
102     imageHandle.image = m_tileImage.ptr();
103     encoder << imageHandle;
104     encoder << m_patternSpaceTransformation;
105     encoder << m_repeatX;
106     encoder << m_repeatY;
107 }
108 
109 template<class Decoder>
110 Optional<Ref<Pattern>> Pattern::decode(Decoder& decoder)
111 {
112     Optional<ImageHandle> imageHandle;
113     decoder >> imageHandle;
114     if (!imageHandle)
115         return WTF::nullopt;
116 
117     Optional<AffineTransform> patternSpaceTransformation;
118     decoder >> patternSpaceTransformation;
119     if (!patternSpaceTransformation)
120         return WTF::nullopt;
121 
122     Optional<bool> repeatX;
123     decoder >> repeatX;
124     if (!repeatX)
125         return WTF::nullopt;
126 
127     Optional<bool> repeatY;
128     decoder >> repeatY;
129     if (!repeatY)
130         return WTF::nullopt;
131 
132     auto pattern = Pattern::create(imageHandle->image.releaseNonNull(), *repeatX, *repeatY);
133     pattern->setPatternSpaceTransform(*patternSpaceTransformation);
134     return pattern;
135 }
