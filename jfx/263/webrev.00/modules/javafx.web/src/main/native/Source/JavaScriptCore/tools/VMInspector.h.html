 26 #pragma once
 27 
 28 #include "CallFrame.h"
 29 #include "VM.h"
 30 #include <wtf/DoublyLinkedList.h>
 31 #include <wtf/Expected.h>
 32 #include <wtf/Lock.h>
 33 
 34 namespace JSC {
 35 
 36 class VMInspector {
 37     WTF_MAKE_FAST_ALLOCATED;
 38     WTF_MAKE_NONCOPYABLE(VMInspector);
 39     VMInspector() = default;
 40 public:
 41     enum class Error {
 42         None,
 43         TimedOut
 44     };
 45 
 46     typedef WTF::Locker<Lock> Locker;
 47 
 48     static VMInspector& instance();
 49 
 50     void add(VM*);
 51     void remove(VM*);
 52 
 53     Lock& getLock() { return m_lock; }
 54 
 55     enum class FunctorStatus {
 56         Continue,
 57         Done
 58     };
 59 
 60     template <typename Functor>
 61     void iterate(const Locker&, const Functor& functor) { iterate(functor); }
 62 
 63     JS_EXPORT_PRIVATE static void forEachVM(Function<FunctorStatus(VM&)>&&);
 64 
 65     Expected<Locker, Error> lock(Seconds timeout = Seconds::infinity());
 66 
 67     Expected<bool, Error> isValidExecutableMemory(const Locker&, void*);
 68     Expected<CodeBlock*, Error> codeBlockForMachinePC(const Locker&, void*);
 69 
 70     JS_EXPORT_PRIVATE static bool currentThreadOwnsJSLock(JSGlobalObject*);
 71     JS_EXPORT_PRIVATE static void gc(JSGlobalObject*);
 72     JS_EXPORT_PRIVATE static void edenGC(JSGlobalObject*);
 73     JS_EXPORT_PRIVATE static bool isInHeap(Heap*, void*);
 74     JS_EXPORT_PRIVATE static bool isValidCell(Heap*, JSCell*);
 75     JS_EXPORT_PRIVATE static bool isValidCodeBlock(JSGlobalObject*, CodeBlock*);
 76     JS_EXPORT_PRIVATE static CodeBlock* codeBlockForFrame(JSGlobalObject*, CallFrame* topCallFrame, unsigned frameNumber);
 77     JS_EXPORT_PRIVATE static void dumpCallFrame(JSGlobalObject*, CallFrame*, unsigned framesToSkip = 0);
 78     JS_EXPORT_PRIVATE static void dumpRegisters(CallFrame*);
 79     JS_EXPORT_PRIVATE static void dumpStack(JSGlobalObject*, CallFrame* topCallFrame, unsigned framesToSkip = 0);
 80     JS_EXPORT_PRIVATE static void dumpValue(JSValue);
 81     JS_EXPORT_PRIVATE static void dumpCellMemory(JSCell*);
 82     JS_EXPORT_PRIVATE static void dumpCellMemoryToStream(JSCell*, PrintStream&);
 83     JS_EXPORT_PRIVATE static void dumpSubspaceHashes(VM*);
 84 
 85     enum VerifierAction { ReleaseAssert, Custom };
 86 
 87     using VerifyFunctor = bool(bool condition, const char* description, ...);
 88     static bool unusedVerifier(bool, const char*, ...) { return false; }
 89 
 90     template<VerifierAction, VerifyFunctor = unusedVerifier>
 91     static bool verifyCellSize(VM&, JSCell*, size_t allocatorCellSize);
 92 
 93     template<VerifierAction, VerifyFunctor = unusedVerifier>
 94     static bool verifyCell(VM&, JSCell*);
 95 
 96 private:
 97     template <typename Functor> void iterate(const Functor& functor)
 98     {
 99         for (VM* vm = m_vmList.head(); vm; vm = vm->next()) {
100             FunctorStatus status = functor(*vm);
101             if (status == FunctorStatus::Done)
102                 return;
103         }
104     }
105 
106     Lock m_lock;
107     DoublyLinkedList<VM> m_vmList;
108 };
109 
110 } // namespace JSC
