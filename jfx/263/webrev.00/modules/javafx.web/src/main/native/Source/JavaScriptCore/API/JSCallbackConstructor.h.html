26 #ifndef JSCallbackConstructor_h
27 #define JSCallbackConstructor_h
28 
29 #include "JSObject.h"
30 #include "JSObjectRef.h"
31 
32 namespace JSC {
33 
34 class JSCallbackConstructor final : public JSNonFinalObject {
35 public:
36     using Base = JSNonFinalObject;
37     static constexpr unsigned StructureFlags = Base::StructureFlags | ImplementsHasInstance | ImplementsDefaultHasInstance;
38     static constexpr bool needsDestruction = true;
39 
40     template<typename CellType, SubspaceAccess mode>
41     static IsoSubspace* subspaceFor(VM& vm)
42     {
43         return vm.callbackConstructorSpace<mode>();
44     }
45 
46     static JSCallbackConstructor* create(JSGlobalObject* globalObject, Structure* structure, JSClassRef classRef, JSObjectCallAsConstructorCallback callback)
47     {
48         VM& vm = getVM(globalObject);
49         JSCallbackConstructor* constructor = new (NotNull, allocateCell<JSCallbackConstructor>(vm.heap)) JSCallbackConstructor(globalObject, structure, classRef, callback);
50         constructor->finishCreation(globalObject, classRef);
51         return constructor;
52     }
53 
54     ~JSCallbackConstructor();
55     static void destroy(JSCell*);
56     JSClassRef classRef() const { return m_class; }
57     JSObjectCallAsConstructorCallback callback() const { return m_callback; }
58     DECLARE_INFO;
59 
60     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue proto)
61     {
62         return Structure::create(vm, globalObject, proto, TypeInfo(ObjectType, StructureFlags), info());
63     }
64 
65 protected:
66     JSCallbackConstructor(JSGlobalObject*, Structure*, JSClassRef, JSObjectCallAsConstructorCallback);
67     void finishCreation(JSGlobalObject*, JSClassRef);
68 
69 private:
70     friend struct APICallbackFunction;
71 
72     static ConstructType getConstructData(JSCell*, ConstructData&);
73 
74     JSObjectCallAsConstructorCallback constructCallback() { return m_callback; }
75 
76     JSClassRef m_class;
77     JSObjectCallAsConstructorCallback m_callback;
78 };
79 
80 } // namespace JSC
81 
82 #endif // JSCallbackConstructor_h
