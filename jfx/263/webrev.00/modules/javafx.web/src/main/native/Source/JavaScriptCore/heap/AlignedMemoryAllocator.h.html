26 #pragma once
27 
28 #include <wtf/PrintStream.h>
29 #include <wtf/SinglyLinkedListWithTail.h>
30 
31 namespace JSC {
32 
33 class BlockDirectory;
34 class Heap;
35 class Subspace;
36 
37 class AlignedMemoryAllocator {
38     WTF_MAKE_NONCOPYABLE(AlignedMemoryAllocator);
39     WTF_MAKE_FAST_ALLOCATED;
40 public:
41     AlignedMemoryAllocator();
42     virtual ~AlignedMemoryAllocator();
43 
44     virtual void* tryAllocateAlignedMemory(size_t alignment, size_t size) = 0;
45     virtual void freeAlignedMemory(void*) = 0;
46 
47     virtual void dump(PrintStream&) const = 0;
48 
49     void registerDirectory(Heap&, BlockDirectory*);
50     BlockDirectory* firstDirectory() const { return m_directories.first(); }
51 
52     void registerSubspace(Subspace*);
53 
54     // Some of derived memory allocators do not have these features because they do not use them.
55     // For example, IsoAlignedMemoryAllocator does not have "realloc" feature since it never extends / shrinks the allocated memory region.
56     virtual void* tryAllocateMemory(size_t) = 0;
57     virtual void freeMemory(void*) = 0;
58     virtual void* tryReallocateMemory(void*, size_t) = 0;
59 
60 private:
61     SinglyLinkedListWithTail<BlockDirectory> m_directories;
62     SinglyLinkedListWithTail<Subspace> m_subspaces;
63 };
64 
65 } // namespace WTF
