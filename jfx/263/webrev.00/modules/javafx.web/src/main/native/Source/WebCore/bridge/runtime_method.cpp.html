 61 EncodedJSValue RuntimeMethod::lengthGetter(JSGlobalObject* exec, EncodedJSValue thisValue, PropertyName)
 62 {
 63     VM& vm = exec->vm();
 64     auto scope = DECLARE_THROW_SCOPE(vm);
 65 
 66     RuntimeMethod* thisObject = jsDynamicCast<RuntimeMethod*>(vm, JSValue::decode(thisValue));
 67     if (!thisObject)
 68         return throwVMTypeError(exec, scope);
 69     return JSValue::encode(jsNumber(thisObject->m_method->numParameters()));
 70 }
 71 
 72 bool RuntimeMethod::getOwnPropertySlot(JSObject* object, JSGlobalObject* exec, PropertyName propertyName, PropertySlot &slot)
 73 {
 74     VM& vm = exec->vm();
 75     RuntimeMethod* thisObject = jsCast<RuntimeMethod*>(object);
 76     if (propertyName == vm.propertyNames->length) {
 77         slot.setCacheableCustom(thisObject, PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum, thisObject->lengthGetter);
 78         return true;
 79     }
 80 
 81     return InternalFunction::getOwnPropertySlot(thisObject, exec, propertyName, slot);
 82 }

 89 static EncodedJSValue JSC_HOST_CALL callRuntimeMethod(JSGlobalObject* globalObject, CallFrame* callFrame)
 90 {
 91     VM& vm = globalObject->vm();
 92     auto scope = DECLARE_THROW_SCOPE(vm);
 93 
 94     RuntimeMethod* method = static_cast<RuntimeMethod*>(callFrame->jsCallee());
 95 
 96     if (!method->method())
 97         return JSValue::encode(jsUndefined());
 98 
 99     RefPtr<Instance> instance;
100 
101     JSValue thisValue = callFrame->thisValue();
102     if (thisValue.inherits<RuntimeObject>(vm)) {
103         RuntimeObject* runtimeObject = static_cast<RuntimeObject*>(asObject(thisValue));
104         instance = runtimeObject->getInternalInstance();
105         if (!instance)
106             return JSValue::encode(RuntimeObject::throwInvalidAccessError(globalObject, scope));
107     } else {
108         // Calling a runtime object of a plugin element?
109         if (thisValue.inherits<JSHTMLElement>(vm))
110             instance = pluginInstance(jsCast<JSHTMLElement*>(asObject(thisValue))->wrapped());
111         if (!instance)
112             return throwVMTypeError(globalObject, scope);
113     }
114     ASSERT(instance);
115 
116     instance->begin();
117     JSValue result = instance->invokeMethod(globalObject, callFrame, method);
118     instance->end();
119     return JSValue::encode(result);
120 }
