 44 class AudioSourceProvider;
 45 class Document;
 46 
 47 struct MediaTrackConstraints;
 48 
 49 template<typename IDLType> class DOMPromiseDeferred;
 50 
 51 class MediaStreamTrack
 52     : public RefCounted<MediaStreamTrack>
 53     , public ActiveDOMObject
 54     , public EventTargetWithInlineData
 55     , private MediaStreamTrackPrivate::Observer
 56     , private PlatformMediaSessionClient
 57 #if !RELEASE_LOG_DISABLED
 58     , private LoggerHelper
 59 #endif
 60 {
 61     WTF_MAKE_ISO_ALLOCATED(MediaStreamTrack);
 62 public:
 63     class Observer {
 64     public:
 65         virtual ~Observer() = default;
 66         virtual void trackDidEnd() = 0;
 67     };
 68 
 69     static Ref<MediaStreamTrack> create(ScriptExecutionContext&, Ref<MediaStreamTrackPrivate>&&);
 70     virtual ~MediaStreamTrack();
 71 
 72     static void endCapture(Document&);
 73 
 74     static MediaProducer::MediaStateFlags captureState(Document&);
 75     static void updateCaptureAccordingToMutedState(Document&);
 76 
 77     virtual bool isCanvas() const { return false; }
 78 
 79     const AtomString& kind() const;
 80     WEBCORE_EXPORT const String& id() const;
 81     const String& label() const;
 82 
 83     const AtomString& contentHint() const;
 84     void setContentHint(const String&);
 85 
 86     bool enabled() const;
 87     void setEnabled(bool);
 88 
 89     bool muted() const;
 90     void setMuted(MediaProducer::MutedStateFlags);
 91 
 92     enum class State { Live, Ended };
 93     State readyState() const;
 94 
 95     bool ended() const;
 96 
 97     virtual RefPtr<MediaStreamTrack> clone();
 98 
 99     enum class StopMode { Silently, PostEvent };
100     void stopTrack(StopMode = StopMode::Silently);
101 
102     bool isCaptureTrack() const { return m_isCaptureTrack; }
103 
104     struct TrackSettings {
105         Optional<int> width;
106         Optional<int> height;
107         Optional<double> aspectRatio;
108         Optional<double> frameRate;
109         String facingMode;
110         Optional<double> volume;
111         Optional<int> sampleRate;
112         Optional<int> sampleSize;
113         Optional<bool> echoCancellation;
114         Optional<bool> displaySurface;
115         String logicalSurface;
116         String deviceId;
117         String groupId;
118     };
119     TrackSettings getSettings() const;
120 
121     struct TrackCapabilities {
122         Optional<LongRange> width;
123         Optional<LongRange> height;
124         Optional<DoubleRange> aspectRatio;
125         Optional<DoubleRange> frameRate;
126         Optional<Vector<String>> facingMode;
127         Optional<DoubleRange> volume;
128         Optional<LongRange> sampleRate;
129         Optional<LongRange> sampleSize;
130         Optional<Vector<bool>> echoCancellation;
131         String deviceId;
132         String groupId;
133     };
134     TrackCapabilities getCapabilities() const;
135 
136     const MediaTrackConstraints& getConstraints() const { return m_constraints; }
137     void applyConstraints(const Optional<MediaTrackConstraints>&, DOMPromiseDeferred<void>&&);
138 
139     RealtimeMediaSource& source() const { return m_private->source(); }
140     MediaStreamTrackPrivate& privateTrack() { return m_private.get(); }
141 
142     AudioSourceProvider* audioSourceProvider();
143 
144     MediaProducer::MediaStateFlags mediaState() const;
145 
146     void addObserver(Observer&);
147     void removeObserver(Observer&);
148 
149     using RefCounted::ref;
150     using RefCounted::deref;
151 
152     // ActiveDOMObject API.
153     bool hasPendingActivity() const final;
154 
155     void setIdForTesting(String&& id) { m_private->setIdForTesting(WTFMove(id)); }
156 
157 #if !RELEASE_LOG_DISABLED
158     const Logger& logger() const final { return m_private->logger(); }
159     const void* logIdentifier() const final { return m_private->logIdentifier(); }
160 #endif
161 
162 protected:
163     MediaStreamTrack(ScriptExecutionContext&, Ref<MediaStreamTrackPrivate>&&);
164 
165     ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
166 
167     Ref<MediaStreamTrackPrivate> m_private;
168 
169 private:
170     explicit MediaStreamTrack(MediaStreamTrack&);
171 
172     void configureTrackRendering();
173 
174     Document* document() const;
175 
176     // ActiveDOMObject API.
177     void stop() final { stopTrack(); }
178     const char* activeDOMObjectName() const override;
179     void suspend(ReasonForSuspension) final;
180 
181     // EventTarget
182     void refEventTarget() final { ref(); }
183     void derefEventTarget() final { deref(); }
184     EventTargetInterface eventTargetInterface() const final { return MediaStreamTrackEventTargetInterfaceType; }
185 
186     // MediaStreamTrackPrivate::Observer
187     void trackStarted(MediaStreamTrackPrivate&) final;
188     void trackEnded(MediaStreamTrackPrivate&) final;
189     void trackMutedChanged(MediaStreamTrackPrivate&) final;
190     void trackSettingsChanged(MediaStreamTrackPrivate&) final;
191     void trackEnabledChanged(MediaStreamTrackPrivate&) final;
192 
193     // PlatformMediaSessionClient
194     PlatformMediaSession::MediaType mediaType() const final;
195     PlatformMediaSession::MediaType presentationType() const final;
196     PlatformMediaSession::CharacteristicsFlags characteristics() const final;
197     void mayResumePlayback(bool shouldResume) final;
198     void suspendPlayback() final;
199     bool canReceiveRemoteControlCommands() const final { return false; }
200     void didReceiveRemoteControlCommand(PlatformMediaSession::RemoteControlCommandType, const PlatformMediaSession::RemoteCommandArgument*) final { }
201     bool supportsSeeking() const final { return false; }
202     bool shouldOverrideBackgroundPlaybackRestriction(PlatformMediaSession::InterruptionType) const final { return false; }
203     String sourceApplicationIdentifier() const final;
204     bool canProduceAudio() const final;
205     Document* hostingDocument() const final { return document(); }
206     bool processingUserGestureForMedia() const final;
207     bool shouldOverridePauseDuringRouteChange() const final { return true; }
208 
209 #if !RELEASE_LOG_DISABLED
210     const char* logClassName() const final { return "MediaStreamTrack"; }
211     WTFLogChannel& logChannel() const final;
212 #endif
213 
214     Vector<Observer*> m_observers;
215 
216     MediaTrackConstraints m_constraints;
217     std::unique_ptr<DOMPromiseDeferred<void>> m_promise;
218 
219     bool m_ended { false };
220     const bool m_isCaptureTrack { false };
221     std::unique_ptr<PlatformMediaSession> m_mediaSession;
222 };
