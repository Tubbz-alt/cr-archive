 26 #pragma once
 27 
 28 #include "DebuggerLocation.h"
 29 #include "JSObject.h"
 30 
 31 namespace JSC {
 32 
 33 class DebuggerCallFrame;
 34 class JSScope;
 35 
 36 class DebuggerScope final : public JSNonFinalObject {
 37 public:
 38     using Base = JSNonFinalObject;
 39     static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetPropertyNames;
 40 
 41     template<typename CellType, SubspaceAccess mode>
 42     static IsoSubspace* subspaceFor(VM& vm)
 43     {
 44         return vm.debuggerScopeSpace<mode>();
 45     }
 46 
 47     JS_EXPORT_PRIVATE static DebuggerScope* create(VM& vm, JSScope* scope);
 48 
 49     static void visitChildren(JSCell*, SlotVisitor&);
 50     static String className(const JSObject*, VM&);
 51     static String toStringName(const JSObject*, JSGlobalObject*);
 52     static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
 53     static bool put(JSCell*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&);
 54     static bool deleteProperty(JSCell*, JSGlobalObject*, PropertyName);
 55     static void getOwnPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, EnumerationMode);
 56     static bool defineOwnProperty(JSObject*, JSGlobalObject*, PropertyName, const PropertyDescriptor&, bool shouldThrow);
 57 
 58     DECLARE_EXPORT_INFO;
 59 
 60     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject)
 61     {
 62         return Structure::create(vm, globalObject, jsNull(), TypeInfo(ObjectType, StructureFlags), info());
 63     }
 64     class iterator {
 65     public:
 66         iterator(DebuggerScope* node)
 67             : m_node(node)
 68         {
 69         }
 70 
 71         DebuggerScope* get() { return m_node; }
 72         iterator& operator++() { m_node = m_node->next(); return *this; }
 73         // postfix ++ intentionally omitted
 74 
 75         bool operator==(const iterator& other) const { return m_node == other.m_node; }
 76         bool operator!=(const iterator& other) const { return m_node != other.m_node; }
 77 
 78     private:
 79         DebuggerScope* m_node;
 80     };
 81 
 82     iterator begin();
 83     iterator end();
 84     DebuggerScope* next();
 85 
 86     void invalidateChain();
 87     bool isValid() const { return !!m_scope; }
 88 
 89     bool isCatchScope() const;
 90     bool isFunctionNameScope() const;
 91     bool isWithScope() const;
 92     bool isGlobalScope() const;
 93     bool isClosureScope() const;
 94     bool isGlobalLexicalEnvironment() const;
 95     bool isNestedLexicalScope() const;
 96 
 97     String name() const;
 98     DebuggerLocation location() const;
 99 
100     JSValue caughtValue(JSGlobalObject*) const;
101 
102 private:
103     DebuggerScope(VM&, Structure*, JSScope*);
104     void finishCreation(VM&);
105 
106     JSScope* jsScope() const { return m_scope.get(); }
107 
108     WriteBarrier<JSScope> m_scope;
109     WriteBarrier<DebuggerScope> m_next;
110 
111     friend class DebuggerCallFrame;
112 };
113 
114 inline DebuggerScope::iterator DebuggerScope::begin()
115 {
116     return iterator(this);
117 }
118 
119 inline DebuggerScope::iterator DebuggerScope::end()
120 {
121     return iterator(0);
122 }
123 
124 } // namespace JSC
