 61 class PromiseWrapper;

 64 struct CryptoKeyPair;
 65 struct JsonWebKey;

 67 class CryptoKeyRSA final : public CryptoKey {
 68 public:
 69     static Ref<CryptoKeyRSA> create(CryptoAlgorithmIdentifier identifier, CryptoAlgorithmIdentifier hash, bool hasHash, CryptoKeyType type, PlatformRSAKeyContainer&& platformKey, bool extractable, CryptoKeyUsageBitmap usage)
 70     {
 71         return adoptRef(*new CryptoKeyRSA(identifier, hash, hasHash, type, WTFMove(platformKey), extractable, usage));
 72     }
 73     static RefPtr<CryptoKeyRSA> create(CryptoAlgorithmIdentifier, CryptoAlgorithmIdentifier hash, bool hasHash, const CryptoKeyRSAComponents&, bool extractable, CryptoKeyUsageBitmap);
 74     virtual ~CryptoKeyRSA() = default;
 75 
 76     bool isRestrictedToHash(CryptoAlgorithmIdentifier&) const;
 77 
 78     size_t keySizeInBits() const;
 79 
 80     using KeyPairCallback = WTF::Function<void(CryptoKeyPair&&)>;
 81     using VoidCallback = WTF::Function<void()>;
 82     static void generatePair(CryptoAlgorithmIdentifier, CryptoAlgorithmIdentifier hash, bool hasHash, unsigned modulusLength, const Vector<uint8_t>& publicExponent, bool extractable, CryptoKeyUsageBitmap, KeyPairCallback&&, VoidCallback&& failureCallback, ScriptExecutionContext*);
 83     static RefPtr<CryptoKeyRSA> importJwk(CryptoAlgorithmIdentifier, Optional<CryptoAlgorithmIdentifier> hash, JsonWebKey&&, bool extractable, CryptoKeyUsageBitmap);
 84     static RefPtr<CryptoKeyRSA> importSpki(CryptoAlgorithmIdentifier, Optional<CryptoAlgorithmIdentifier> hash, Vector<uint8_t>&&, bool extractable, CryptoKeyUsageBitmap);
 85     static RefPtr<CryptoKeyRSA> importPkcs8(CryptoAlgorithmIdentifier, Optional<CryptoAlgorithmIdentifier> hash, Vector<uint8_t>&&, bool extractable, CryptoKeyUsageBitmap);
 86 
 87     PlatformRSAKey platformKey() const { return m_platformKey.get(); }
 88     JsonWebKey exportJwk() const;
 89     ExceptionOr<Vector<uint8_t>> exportSpki() const;
 90     ExceptionOr<Vector<uint8_t>> exportPkcs8() const;
 91 
 92     std::unique_ptr<CryptoKeyRSAComponents> exportData() const;
 93 
 94     CryptoAlgorithmIdentifier hashAlgorithmIdentifier() const { return m_hash; }
 95 
 96 private:
 97     CryptoKeyRSA(CryptoAlgorithmIdentifier, CryptoAlgorithmIdentifier hash, bool hasHash, CryptoKeyType, PlatformRSAKeyContainer&&, bool extractable, CryptoKeyUsageBitmap);
 98 
 99     CryptoKeyClass keyClass() const final { return CryptoKeyClass::RSA; }
100 
101     KeyAlgorithm algorithm() const final;
102 
103     PlatformRSAKeyContainer m_platformKey;
104 
105     bool m_restrictedToSpecificHash;
106     CryptoAlgorithmIdentifier m_hash;
107 };
108 
109 } // namespace WebCore
