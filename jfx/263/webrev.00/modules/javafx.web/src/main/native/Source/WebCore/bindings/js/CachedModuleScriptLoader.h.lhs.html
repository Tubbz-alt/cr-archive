27 
28 #include "CachedResourceClient.h"
29 #include "CachedResourceHandle.h"
30 #include <wtf/Ref.h>
31 #include <wtf/RefCounted.h>
32 #include <wtf/RefPtr.h>

33 
34 namespace WebCore {
35 
36 class CachedModuleScriptLoaderClient;
37 class CachedScript;
38 class CachedScriptFetcher;
39 class DeferredPromise;
40 class Document;
41 class JSDOMGlobalObject;
42 class ModuleFetchParameters;
43 
44 class CachedModuleScriptLoader final : public RefCounted<CachedModuleScriptLoader>, private CachedResourceClient {
45 public:
46     static Ref<CachedModuleScriptLoader> create(CachedModuleScriptLoaderClient&, DeferredPromise&, CachedScriptFetcher&, RefPtr<ModuleFetchParameters>&&);
47 
48     virtual ~CachedModuleScriptLoader();
49 
50     bool load(Document&, const URL& sourceURL);
51 
52     CachedScriptFetcher& scriptFetcher() { return m_scriptFetcher.get(); }
53     CachedScript* cachedScript() { return m_cachedScript.get(); }
54     ModuleFetchParameters* parameters() { return m_parameters.get(); }

55 
56     void clearClient()
57     {
58         ASSERT(m_client);
59         m_client = nullptr;
60     }
61 
62 private:
63     CachedModuleScriptLoader(CachedModuleScriptLoaderClient&, DeferredPromise&, CachedScriptFetcher&, RefPtr<ModuleFetchParameters>&&);
64 
65     void notifyFinished(CachedResource&) final;
66 
67     CachedModuleScriptLoaderClient* m_client { nullptr };
68     RefPtr<DeferredPromise> m_promise;
69     Ref<CachedScriptFetcher> m_scriptFetcher;
70     RefPtr<ModuleFetchParameters> m_parameters;
71     CachedResourceHandle<CachedScript> m_cachedScript;

72 };
73 
74 } // namespace WebCore
