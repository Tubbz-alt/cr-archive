 35 OBJC_CLASS NSPasteboard;
 36 OBJC_CLASS NSPasteboardItem;

 40 OBJC_CLASS UIPasteboard;

 47 namespace WebCore {
 48 
 49 class Color;
 50 class PasteboardCustomData;
 51 class SelectionData;
 52 class SharedBuffer;
 53 struct PasteboardImage;
 54 struct PasteboardItemInfo;
 55 struct PasteboardURL;
 56 struct PasteboardWebContent;
 57 
 58 class PlatformPasteboard {
 59 public:
 60     WEBCORE_EXPORT explicit PlatformPasteboard(const String& pasteboardName);
 61 #if PLATFORM(IOS_FAMILY) || USE(LIBWPE)
 62     WEBCORE_EXPORT PlatformPasteboard();
 63     WEBCORE_EXPORT void updateSupportedTypeIdentifiers(const Vector<String>& types);
 64 #endif
 65     WEBCORE_EXPORT Optional<PasteboardItemInfo> informationForItemAtIndex(size_t index, int64_t changeCount);
 66     WEBCORE_EXPORT Optional<Vector<PasteboardItemInfo>> allPasteboardItemInfo(int64_t changeCount);
 67     WEBCORE_EXPORT static String uniqueName();
 68 
 69     enum class IncludeImageTypes : bool { No, Yes };
 70     static String platformPasteboardTypeForSafeTypeForDOMToReadAndWrite(const String& domType, IncludeImageTypes = IncludeImageTypes::No);
 71 
 72     WEBCORE_EXPORT void getTypes(Vector<String>& types);
 73     WEBCORE_EXPORT RefPtr<SharedBuffer> bufferForType(const String& pasteboardType);
 74     WEBCORE_EXPORT void getPathnamesForType(Vector<String>& pathnames, const String& pasteboardType) const;
 75     WEBCORE_EXPORT String stringForType(const String& pasteboardType) const;
 76     WEBCORE_EXPORT Vector<String> allStringsForType(const String& pasteboardType) const;
 77     WEBCORE_EXPORT int64_t changeCount() const;
 78     WEBCORE_EXPORT Color color();
 79     WEBCORE_EXPORT URL url();
 80 
 81     // Take ownership of the pasteboard, and return new change count.
 82     WEBCORE_EXPORT int64_t addTypes(const Vector<String>& pasteboardTypes);
 83     WEBCORE_EXPORT int64_t setTypes(const Vector<String>& pasteboardTypes);
 84 
 85     // These methods will return 0 if pasteboard ownership has been taken from us.
 86     WEBCORE_EXPORT int64_t copy(const String& fromPasteboard);
 87     WEBCORE_EXPORT int64_t setBufferForType(SharedBuffer*, const String& pasteboardType);
 88     WEBCORE_EXPORT int64_t setURL(const PasteboardURL&);
 89     WEBCORE_EXPORT int64_t setColor(const Color&);
 90     WEBCORE_EXPORT int64_t setStringForType(const String&, const String& pasteboardType);
 91     WEBCORE_EXPORT void write(const PasteboardWebContent&);
 92     WEBCORE_EXPORT void write(const PasteboardImage&);
 93     WEBCORE_EXPORT void write(const String& pasteboardType, const String&);
 94     WEBCORE_EXPORT void write(const PasteboardURL&);
 95     WEBCORE_EXPORT RefPtr<SharedBuffer> readBuffer(size_t index, const String& pasteboardType) const;
 96     WEBCORE_EXPORT String readString(size_t index, const String& pasteboardType) const;
 97     WEBCORE_EXPORT URL readURL(size_t index, String& title) const;
 98     WEBCORE_EXPORT int count() const;
 99     WEBCORE_EXPORT int numberOfFiles() const;
100     WEBCORE_EXPORT int64_t write(const Vector<PasteboardCustomData>&);
101     WEBCORE_EXPORT int64_t write(const PasteboardCustomData&);
102     WEBCORE_EXPORT Vector<String> typesSafeForDOMToReadAndWrite(const String& origin) const;
103 
104 #if PLATFORM(GTK)
105     WEBCORE_EXPORT void writeToClipboard(const SelectionData&, WTF::Function<void()>&& primarySelectionCleared);
106     WEBCORE_EXPORT Ref<SelectionData> readFromClipboard();
107 #endif
108 
109 private:
110 #if PLATFORM(IOS_FAMILY)
111     bool allowReadingURLAtIndex(const URL&, int index) const;
112 #endif
113 
114 #if PLATFORM(MAC)
115     NSPasteboardItem *itemAtIndex(size_t index) const;
116 #endif
117 
118 #if PLATFORM(MAC)
119     RetainPtr<NSPasteboard> m_pasteboard;
120 #endif
121 #if PLATFORM(IOS_FAMILY)
122     RetainPtr<id> m_pasteboard;
123 #endif
124 #if PLATFORM(GTK)
125     GtkClipboard* m_clipboard;
126 #endif
127 #if USE(LIBWPE)
128     struct wpe_pasteboard* m_pasteboard;
129 #endif
130 };
131 
132 }

134 #endif // !PlatformPasteboard_h
