30 #include "JSValueInWrappedObject.h"

36 namespace JSC {
37 class JSObject;
38 }
39 
40 namespace WebCore {
41 
42 class PaymentMethodChangeEvent final : public PaymentRequestUpdateEvent {
43     WTF_MAKE_ISO_ALLOCATED(PaymentMethodChangeEvent);
44 public:
45     template<typename... Args> static Ref<PaymentMethodChangeEvent> create(Args&&... args)
46     {
47         return adoptRef(*new PaymentMethodChangeEvent(std::forward<Args>(args)...));
48     }
49 
50     using MethodDetailsFunction = std::function<JSC::Strong<JSC::JSObject>(JSC::JSGlobalObject&)>;
51     using MethodDetailsType = Variant<JSValueInWrappedObject, MethodDetailsFunction>;
52 
53     const String& methodName() const { return m_methodName; }
54     const MethodDetailsType& methodDetails() const { return m_methodDetails; }
55     JSValueInWrappedObject& cachedMethodDetails() { return m_cachedMethodDetails; }
56 
57     // Event
58     EventInterface eventInterface() const override;
59 
60     struct Init final : PaymentRequestUpdateEventInit {
61         String methodName;
62         JSC::Strong<JSC::JSObject> methodDetails;
63     };
64 
65 private:
66     PaymentMethodChangeEvent(const AtomString& type, Init&&);
67     PaymentMethodChangeEvent(const AtomString& type, const String& methodName, MethodDetailsFunction&&);
68 
69     String m_methodName;
70     MethodDetailsType m_methodDetails;
71     JSValueInWrappedObject m_cachedMethodDetails;
72 };
73 
74 } // namespace WebCore

76 #endif // ENABLE(PAYMENT_REQUEST)
