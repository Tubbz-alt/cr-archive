 66 void WebDebuggerAgent::didAddEventListener(EventTarget& target, const AtomString& eventType, EventListener& listener, bool capture)
 67 {
 68     if (!breakpointsActive())
 69         return;
 70 
 71     auto& eventListeners = target.eventListeners(eventType);
 72     auto position = eventListeners.findMatching([&](auto& registeredListener) {
 73         return &registeredListener->callback() == &listener && registeredListener->useCapture() == capture;
 74     });
 75     if (position == notFound)
 76         return;
 77 
 78     auto& registeredListener = eventListeners.at(position);
 79     if (m_registeredEventListeners.contains(registeredListener.get()))
 80         return;
 81 
 82     JSC::JSGlobalObject* scriptState = target.scriptExecutionContext()->execState();
 83     if (!scriptState)
 84         return;
 85 
 86     int identifier = m_nextEventListenerIdentifier++;
 87     m_registeredEventListeners.set(registeredListener.get(), identifier);
 88 
 89     didScheduleAsyncCall(scriptState, InspectorDebuggerAgent::AsyncCallType::EventListener, identifier, registeredListener->isOnce());
 90 }
 91 
 92 void WebDebuggerAgent::willRemoveEventListener(EventTarget& target, const AtomString& eventType, EventListener& listener, bool capture)
 93 {
 94     auto& eventListeners = target.eventListeners(eventType);
 95     size_t listenerIndex = eventListeners.findMatching([&](auto& registeredListener) {
 96         return &registeredListener->callback() == &listener && registeredListener->useCapture() == capture;
 97     });
 98 
 99     if (listenerIndex == notFound)
100         return;
101 
102     int identifier = m_registeredEventListeners.take(eventListeners[listenerIndex].get());
103     didCancelAsyncCall(InspectorDebuggerAgent::AsyncCallType::EventListener, identifier);
104 }
105 
106 void WebDebuggerAgent::willHandleEvent(const RegisteredEventListener& listener)
107 {
108     auto it = m_registeredEventListeners.find(&listener);
109     if (it == m_registeredEventListeners.end())
110         return;
111 
112     willDispatchAsyncCall(InspectorDebuggerAgent::AsyncCallType::EventListener, it->value);
113 }
114 
115 void WebDebuggerAgent::didPostMessage(const TimerBase& timer, JSC::JSGlobalObject& state)
116 {
117     if (!breakpointsActive())
118         return;
119 
120     if (m_postMessageTimers.contains(&timer))
121         return;
122 
123     int postMessageIdentifier = m_nextPostMessageIdentifier++;
124     m_postMessageTimers.set(&timer, postMessageIdentifier);
125 
126     didScheduleAsyncCall(&state, InspectorDebuggerAgent::AsyncCallType::PostMessage, postMessageIdentifier, true);
127 }
128 
129 void WebDebuggerAgent::didFailPostMessage(const TimerBase& timer)
130 {
131     auto it = m_postMessageTimers.find(&timer);
132     if (it == m_postMessageTimers.end())
133         return;
134 
135     didCancelAsyncCall(InspectorDebuggerAgent::AsyncCallType::PostMessage, it->value);
136 
137     m_postMessageTimers.remove(it);
138 }
139 
140 void WebDebuggerAgent::willDispatchPostMessage(const TimerBase& timer)
141 {
142     auto it = m_postMessageTimers.find(&timer);
143     if (it == m_postMessageTimers.end())
144         return;
145 
146     willDispatchAsyncCall(InspectorDebuggerAgent::AsyncCallType::PostMessage, it->value);
147 }
148 
149 void WebDebuggerAgent::didDispatchPostMessage(const TimerBase& timer)
150 {
151     auto it = m_postMessageTimers.find(&timer);
152     if (it == m_postMessageTimers.end())
153         return;
154 
155     didDispatchAsyncCall();
156 
157     m_postMessageTimers.remove(it);
158 }
159 
160 void WebDebuggerAgent::didClearAsyncStackTraceData()
161 {
162     m_registeredEventListeners.clear();
163     m_postMessageTimers.clear();
164     m_nextEventListenerIdentifier = 1;
165     m_nextPostMessageIdentifier = 1;
166 }
