  46 // Because |format| is used as the second parameter to va_start, it cannot be a reference
  47 // type according to section 18.7/3 of the C++ N1905 standard.
  48 String formatLocalizedString(String format, ...)
  49 {
  50 #if USE(CF)
  51     va_list arguments;
  52     va_start(arguments, format);
  53 
  54     ALLOW_NONLITERAL_FORMAT_BEGIN
  55     auto result = adoptCF(CFStringCreateWithFormatAndArguments(0, 0, format.createCFString().get(), arguments));
  56     ALLOW_NONLITERAL_FORMAT_END
  57 
  58     va_end(arguments);
  59     return result.get();
  60 #elif USE(GLIB)
  61     va_list arguments;
  62     va_start(arguments, format);
  63     GUniquePtr<gchar> result(g_strdup_vprintf(format.utf8().data(), arguments));
  64     va_end(arguments);
  65     return String::fromUTF8(result.get());
  66 #else
  67     notImplemented();
  68     return format;
  69 #endif
  70 }

  83 static String truncatedStringForLookupMenuItem(const String& original)
  84 {
  85     // Truncate the string if it's too long. This number is roughly the same as the one used by AppKit.
  86     unsigned maxNumberOfGraphemeClustersInLookupMenuItem = 24;
  87 
  88     String trimmed = original.stripWhiteSpace();
  89     unsigned numberOfCharacters = numCodeUnitsInGraphemeClusters(trimmed, maxNumberOfGraphemeClustersInLookupMenuItem);
  90     return numberOfCharacters == trimmed.length() ? trimmed : makeString(trimmed.left(numberOfCharacters), horizontalEllipsis);
  91 }

 790 String imageTitle(const String& filename, const IntSize& size)
 791 {
 792 #if USE(CF)
 793     auto locale = adoptCF(CFLocaleCopyCurrent());
 794     auto formatter = adoptCF(CFNumberFormatterCreate(0, locale.get(), kCFNumberFormatterDecimalStyle));
 795 
 796     int widthInt = size.width();
 797     auto width = adoptCF(CFNumberCreate(0, kCFNumberIntType, &widthInt));
 798     auto widthString = adoptCF(CFNumberFormatterCreateStringWithNumber(0, formatter.get(), width.get()));
 799 
 800     int heightInt = size.height();
 801     auto height = adoptCF(CFNumberCreate(0, kCFNumberIntType, &heightInt));
 802     auto heightString = adoptCF(CFNumberFormatterCreateStringWithNumber(0, formatter.get(), height.get()));
 803 
 804     return formatLocalizedString(WEB_UI_CFSTRING("%@ %@×%@ pixels", "window title for a standalone image (uses multiplication symbol, not x)"), filename.createCFString().get(), widthString.get(), heightString.get());
 805 #elif USE(GLIB)
 806     return formatLocalizedString(WEB_UI_STRING("%s %d×%d pixels", "window title for a standalone image (uses multiplication symbol, not x)"), filename.utf8().data(), size.width(), size.height());
 807 #else
 808     return formatLocalizedString(WEB_UI_STRING("<filename> %d×%d pixels", "window title for a standalone image (uses multiplication symbol, not x)"), size.width(), size.height()).replace("<filename>", filename);
 809 #endif
 810 }

 918 String localizedMediaTimeDescription(float time)
 919 {
 920     if (!std::isfinite(time))
 921         return WEB_UI_STRING("indefinite time", "accessibility help text for an indefinite media controller time value");
 922 
 923     int seconds = static_cast<int>(fabsf(time));
 924     int days = seconds / (60 * 60 * 24);
 925     int hours = seconds / (60 * 60);
 926     int minutes = (seconds / 60) % 60;
 927     seconds %= 60;
 928 
 929     if (days)
 930         return formatLocalizedString(WEB_UI_STRING("%1$d days %2$d hours %3$d minutes %4$d seconds", "accessibility help text for media controller time value >= 1 day"), days, hours, minutes, seconds);
 931     if (hours)
 932         return formatLocalizedString(WEB_UI_STRING("%1$d hours %2$d minutes %3$d seconds", "accessibility help text for media controller time value >= 60 minutes"), hours, minutes, seconds);
 933     if (minutes)
 934         return formatLocalizedString(WEB_UI_STRING("%1$d minutes %2$d seconds", "accessibility help text for media controller time value >= 60 seconds"), minutes, seconds);
 935     return formatLocalizedString(WEB_UI_STRING("%1$d seconds", "accessibility help text for media controller time value < 60 seconds"), seconds);
 936 }
