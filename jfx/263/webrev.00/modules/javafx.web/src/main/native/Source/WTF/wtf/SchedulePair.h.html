29 #pragma once
30 
31 #include <wtf/HashSet.h>
32 #include <wtf/RetainPtr.h>
33 #include <wtf/ThreadSafeRefCounted.h>
34 #include <wtf/text/WTFString.h>
35 
36 #if PLATFORM(COCOA)
37 OBJC_CLASS NSRunLoop;
38 #endif
39 
40 namespace WTF {
41 
42 class SchedulePair : public ThreadSafeRefCounted<SchedulePair> {
43 public:
44     static Ref<SchedulePair> create(CFRunLoopRef runLoop, CFStringRef mode) { return adoptRef(*new SchedulePair(runLoop, mode)); }
45 
46 #if PLATFORM(COCOA)
47     static Ref<SchedulePair> create(NSRunLoop* runLoop, CFStringRef mode) { return adoptRef(*new SchedulePair(runLoop, mode)); }
48     NSRunLoop* nsRunLoop() const { return m_nsRunLoop.get(); }
49 #endif
50 
51     CFRunLoopRef runLoop() const { return m_runLoop.get(); }
52     CFStringRef mode() const { return m_mode.get(); }
53 
54     WTF_EXPORT_PRIVATE bool operator==(const SchedulePair& other) const;
55 
56 private:
57     SchedulePair(CFRunLoopRef runLoop, CFStringRef mode)
58         : m_runLoop(runLoop)
59     {
60         if (mode)
61             m_mode = adoptCF(CFStringCreateCopy(0, mode));
62     }
63 
64 #if PLATFORM(COCOA)
65     WTF_EXPORT_PRIVATE SchedulePair(NSRunLoop*, CFStringRef);
66     RetainPtr<NSRunLoop> m_nsRunLoop;
67 #endif
68 
69     RetainPtr<CFRunLoopRef> m_runLoop;
70     RetainPtr<CFStringRef> m_mode;
71 };
72 
73 struct SchedulePairHash {
74     static unsigned hash(const RefPtr<SchedulePair>& pair)
75     {
76         uintptr_t hashCodes[2] = { reinterpret_cast<uintptr_t>(pair->runLoop()), pair->mode() ? CFHash(pair->mode()) : 0 };
77         return StringHasher::hashMemory<sizeof(hashCodes)>(hashCodes);
78     }
79 
80     static bool equal(const RefPtr<SchedulePair>& a, const RefPtr<SchedulePair>& b) { return a == b; }
81 
82     static constexpr bool safeToCompareToEmptyOrDeleted = true;
83 };
84 
85 typedef HashSet<RefPtr<SchedulePair>, SchedulePairHash> SchedulePairHashSet;
86 
87 } // namespace WTF
88 
89 using WTF::SchedulePair;
90 using WTF::SchedulePairHashSet;
