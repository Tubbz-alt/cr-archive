37 namespace WebCore {
38 
39 using FetchResponsePromise = DOMPromiseDeferred<IDLInterface<FetchResponse>>;
40 
41 void WorkerGlobalScopeFetch::fetch(WorkerGlobalScope& scope, FetchRequest::Info&& input, FetchRequest::Init&& init, Ref<DeferredPromise>&& deferred)
42 {
43     FetchResponsePromise promise = WTFMove(deferred);
44 
45     auto request = FetchRequest::create(scope, WTFMove(input), WTFMove(init));
46     if (request.hasException()) {
47         promise.reject(request.releaseException());
48         return;
49     }
50 
51     FetchResponse::fetch(scope, request.releaseReturnValue().get(), [promise = WTFMove(promise)](ExceptionOr<FetchResponse&>&& result) mutable {
52         promise.settle(WTFMove(result));
53     });
54 }
