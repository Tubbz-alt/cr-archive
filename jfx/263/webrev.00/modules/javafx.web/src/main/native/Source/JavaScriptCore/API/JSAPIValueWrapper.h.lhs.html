23 #pragma once
24 
25 #include "CallFrame.h"
26 #include "JSCJSValue.h"
27 #include "JSCast.h"
28 #include "Structure.h"
29 
30 namespace JSC {
31 
32 class JSAPIValueWrapper final : public JSCell {
33     friend JSValue jsAPIValueWrapper(ExecState*, JSValue);
34 public:
35     typedef JSCell Base;
36     static const unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;






37 
38     JSValue value() const { return m_value.get(); }
39 
40     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
41     {
42         return Structure::create(vm, globalObject, prototype, TypeInfo(APIValueWrapperType, OverridesGetPropertyNames), info());
43     }
44 
45     DECLARE_EXPORT_INFO;
46 
47     static JSAPIValueWrapper* create(VM& vm, JSValue value)
48     {
49         JSAPIValueWrapper* wrapper = new (NotNull, allocateCell<JSAPIValueWrapper>(vm.heap)) JSAPIValueWrapper(vm);
50         wrapper->finishCreation(vm, value);
51         return wrapper;
52     }
53 
54 protected:
55     void finishCreation(VM& vm, JSValue value)
56     {
57         Base::finishCreation(vm);
58         m_value.set(vm, this, value);
59         ASSERT(!value.isCell());
60     }
61 
62 private:
63     JSAPIValueWrapper(VM& vm)
64         : JSCell(vm, vm.apiWrapperStructure.get())
65     {
66     }
67 
68     WriteBarrier<Unknown> m_value;
69 };
70 
71 } // namespace JSC
