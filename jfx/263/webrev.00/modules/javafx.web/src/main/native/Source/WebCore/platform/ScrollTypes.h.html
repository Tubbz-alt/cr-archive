 49 enum ScrollLogicalDirection : uint8_t {
 50     ScrollBlockDirectionBackward,
 51     ScrollBlockDirectionForward,
 52     ScrollInlineDirectionBackward,
 53     ScrollInlineDirectionForward
 54 };
 55 
 56 inline ScrollDirection logicalToPhysical(ScrollLogicalDirection direction, bool isVertical, bool isFlipped)
 57 {
 58     switch (direction) {
 59     case ScrollBlockDirectionBackward: {
 60         if (isVertical) {
 61             if (!isFlipped)
 62                 return ScrollUp;
 63             return ScrollDown;
 64         } else {
 65             if (!isFlipped)
 66                 return ScrollLeft;
 67             return ScrollRight;
 68         }
 69         break;
 70     }
 71     case ScrollBlockDirectionForward: {
 72         if (isVertical) {
 73             if (!isFlipped)
 74                 return ScrollDown;
 75             return ScrollUp;
 76         } else {
 77             if (!isFlipped)
 78                 return ScrollRight;
 79             return ScrollLeft;
 80         }
 81         break;
 82     }
 83     case ScrollInlineDirectionBackward: {
 84         if (isVertical) {
 85             if (!isFlipped)
 86                 return ScrollLeft;
 87             return ScrollRight;
 88         } else {
 89             if (!isFlipped)
 90                 return ScrollUp;
 91             return ScrollDown;
 92         }
 93         break;
 94     }
 95     case ScrollInlineDirectionForward: {
 96         if (isVertical) {
 97             if (!isFlipped)
 98                 return ScrollRight;
 99             return ScrollLeft;
100         } else {
101             if (!isFlipped)
102                 return ScrollDown;
103             return ScrollUp;
104         }
105         break;
106     }
107     default:
108         ASSERT_NOT_REACHED();
109         break;
110     }
111     return ScrollUp;
112 }
