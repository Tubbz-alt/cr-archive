31 find_package(PkgConfig)
32 pkg_check_modules(GEOCLUE2 geoclue-2.0)
33 
34 set(VERSION_OK FALSE)
35 if (GEOCLUE2_VERSION)
36     set(VERSION_OK TRUE)
37     if (GeoClue2_FIND_VERSION_EXACT)
38         if (NOT("${GeoClue2_FIND_VERSION}" VERSION_EQUAL "${GEOCLUE2_VERSION}"))
39             set(VERSION_OK FALSE)
40         endif ()
41     else ()
42         if ("${GEOCLUE2_VERSION}" VERSION_LESS "${GeoClue2_FIND_VERSION}")
43             set(VERSION_OK FALSE)
44         endif ()
45     endif ()
46 endif ()
47 
48 include(FindPackageHandleStandardArgs)
49 find_package_handle_standard_args(GeoClue2 REQUIRED_VARS VERSION_OK
50                                   FOUND_VAR GEOCLUE2_FOUND)
