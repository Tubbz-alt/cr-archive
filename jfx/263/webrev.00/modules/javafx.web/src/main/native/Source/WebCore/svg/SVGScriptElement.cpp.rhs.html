33 inline SVGScriptElement::SVGScriptElement(const QualifiedName& tagName, Document& document, bool wasInsertedByParser, bool alreadyStarted)
34     : SVGElement(tagName, document)

35     , SVGURIReference(this)
36     , ScriptElement(*this, wasInsertedByParser, alreadyStarted)
37     , m_loadEventTimer(*this, &SVGElement::loadEventTimerFired)
38 {
39     ASSERT(hasTagName(SVGNames::scriptTag));
40 }
41 
42 Ref<SVGScriptElement> SVGScriptElement::create(const QualifiedName& tagName, Document& document, bool insertedByParser)
43 {
44     return adoptRef(*new SVGScriptElement(tagName, document, insertedByParser, false));
45 }
46 
47 void SVGScriptElement::parseAttribute(const QualifiedName& name, const AtomString& value)
48 {
49     SVGElement::parseAttribute(name, value);
50     SVGURIReference::parseAttribute(name, value);

51 }
52 
53 void SVGScriptElement::svgAttributeChanged(const QualifiedName& attrName)
54 {
55     InstanceInvalidationGuard guard(*this);
56 
57     if (SVGURIReference::isKnownAttribute(attrName)) {
58         handleSourceAttribute(href());
59         return;
60     }
61 
62     SVGElement::svgAttributeChanged(attrName);

63 }
64 
65 Node::InsertedIntoAncestorResult SVGScriptElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
66 {
67     SVGElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);


68     return ScriptElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
69 }
70 
71 void SVGScriptElement::didFinishInsertingNode()

73     ScriptElement::didFinishInsertingNode();
74 }
75 
76 void SVGScriptElement::childrenChanged(const ChildChange& change)
77 {
78     SVGElement::childrenChanged(change);
79     ScriptElement::childrenChanged(change);
80 }

82 void SVGScriptElement::addSubresourceAttributeURLs(ListHashSet<URL>& urls) const
83 {
84     SVGElement::addSubresourceAttributeURLs(urls);
85 
86     addSubresourceURL(urls, document().completeURL(href()));
87 }
88 Ref<Element> SVGScriptElement::cloneElementWithoutAttributesAndChildren(Document& targetDocument)
89 {
90     return adoptRef(*new SVGScriptElement(tagQName(), targetDocument, false, alreadyStarted()));
91 }

93 void SVGScriptElement::dispatchErrorEvent()
