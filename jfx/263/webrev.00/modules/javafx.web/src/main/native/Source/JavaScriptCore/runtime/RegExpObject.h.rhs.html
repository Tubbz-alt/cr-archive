 21 #pragma once
 22 
 23 #include "JSObject.h"
 24 #include "RegExp.h"
 25 #include "ThrowScope.h"
 26 #include "TypeError.h"
 27 
 28 namespace JSC {
 29 
 30 class RegExpObject final : public JSNonFinalObject {
 31 public:
 32     using Base = JSNonFinalObject;
 33     static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetPropertyNames;
 34 
 35     template<typename CellType, SubspaceAccess mode>
 36     static IsoSubspace* subspaceFor(VM& vm)
 37     {
 38         static_assert(!CellType::needsDestruction, "");
 39         return &vm.regExpObjectSpace;
 40     }
 41 
 42     static constexpr uintptr_t lastIndexIsNotWritableFlag = 1;
 43 
 44     static RegExpObject* create(VM& vm, Structure* structure, RegExp* regExp)
 45     {
 46         RegExpObject* object = new (NotNull, allocateCell<RegExpObject>(vm.heap)) RegExpObject(vm, structure, regExp);
 47         object->finishCreation(vm);
 48         return object;
 49     }
 50 
 51     static RegExpObject* create(VM& vm, Structure* structure, RegExp* regExp, JSValue lastIndex)
 52     {
 53         auto* object = create(vm, structure, regExp);
 54         object->m_lastIndex.set(vm, object, lastIndex);
 55         return object;
 56     }
 57 
 58     void setRegExp(VM& vm, RegExp* regExp)
 59     {
 60         uintptr_t result = (m_regExpAndLastIndexIsNotWritableFlag & lastIndexIsNotWritableFlag) | bitwise_cast<uintptr_t>(regExp);
 61         m_regExpAndLastIndexIsNotWritableFlag = result;
 62         vm.heap.writeBarrier(this, regExp);
 63     }
 64 
 65     RegExp* regExp() const
 66     {
 67         return bitwise_cast<RegExp*>(m_regExpAndLastIndexIsNotWritableFlag & (~lastIndexIsNotWritableFlag));
 68     }
 69 
 70     bool setLastIndex(JSGlobalObject* globalObject, size_t lastIndex)
 71     {
 72         VM& vm = getVM(globalObject);
 73         auto scope = DECLARE_THROW_SCOPE(vm);
 74 
 75         if (LIKELY(lastIndexIsWritable())) {
 76             m_lastIndex.setWithoutWriteBarrier(jsNumber(lastIndex));
 77             return true;
 78         }
 79         throwTypeError(globalObject, scope, ReadonlyPropertyWriteError);
 80         return false;
 81     }
 82     bool setLastIndex(JSGlobalObject* globalObject, JSValue lastIndex, bool shouldThrow)
 83     {
 84         VM& vm = getVM(globalObject);
 85         auto scope = DECLARE_THROW_SCOPE(vm);
 86 
 87         if (LIKELY(lastIndexIsWritable())) {
 88             m_lastIndex.set(vm, this, lastIndex);
 89             return true;
 90         }
 91         return typeError(globalObject, scope, shouldThrow, ReadonlyPropertyWriteError);
 92     }
 93     JSValue getLastIndex() const
 94     {
 95         return m_lastIndex.get();
 96     }
 97 
 98     bool test(JSGlobalObject* globalObject, JSString* string) { return !!match(globalObject, string); }
 99     bool testInline(JSGlobalObject* globalObject, JSString* string) { return !!matchInline(globalObject, string); }
100     JSValue exec(JSGlobalObject*, JSString*);
101     JSValue execInline(JSGlobalObject*, JSString*);
102     MatchResult match(JSGlobalObject*, JSString*);
103     JSValue matchGlobal(JSGlobalObject*, JSString*);
104 
105     static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
106     static bool put(JSCell*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&);
107 
108     DECLARE_EXPORT_INFO;
109 
110     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
111     {
112         return Structure::create(vm, globalObject, prototype, TypeInfo(RegExpObjectType, StructureFlags), info());
113     }
114 
115     static ptrdiff_t offsetOfRegExpAndLastIndexIsNotWritableFlag()
116     {
117         return OBJECT_OFFSETOF(RegExpObject, m_regExpAndLastIndexIsNotWritableFlag);
118     }
119 
120     static ptrdiff_t offsetOfLastIndex()
121     {
122         return OBJECT_OFFSETOF(RegExpObject, m_lastIndex);
123     }
124 
125     static size_t allocationSize(Checked<size_t> inlineCapacity)
126     {
127         ASSERT_UNUSED(inlineCapacity, !inlineCapacity);
128         return sizeof(RegExpObject);
129     }
130 
131 protected:
132     JS_EXPORT_PRIVATE RegExpObject(VM&, Structure*, RegExp*);
133     JS_EXPORT_PRIVATE void finishCreation(VM&);
134 
135     static void visitChildren(JSCell*, SlotVisitor&);
136 
137     bool lastIndexIsWritable() const
138     {
139         return !(m_regExpAndLastIndexIsNotWritableFlag & lastIndexIsNotWritableFlag);
140     }
141 
142     void setLastIndexIsNotWritable()
143     {
144         m_regExpAndLastIndexIsNotWritableFlag = (m_regExpAndLastIndexIsNotWritableFlag | lastIndexIsNotWritableFlag);
145     }
146 
147     JS_EXPORT_PRIVATE static bool deleteProperty(JSCell*, JSGlobalObject*, PropertyName);
148     JS_EXPORT_PRIVATE static void getOwnNonIndexPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, EnumerationMode);
149     JS_EXPORT_PRIVATE static void getPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, EnumerationMode);
150     JS_EXPORT_PRIVATE static void getGenericPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, EnumerationMode);
151     JS_EXPORT_PRIVATE static bool defineOwnProperty(JSObject*, JSGlobalObject*, PropertyName, const PropertyDescriptor&, bool shouldThrow);
152 
153 private:
154     MatchResult matchInline(JSGlobalObject*, JSString*);
155 
156     uintptr_t m_regExpAndLastIndexIsNotWritableFlag { 0 };
157     WriteBarrier<Unknown> m_lastIndex;
158 };
159 
160 } // namespace JSC
