40 float SVGTextLayoutEngineSpacing::calculateCSSKerningAndSpacing(const SVGRenderStyle* style, SVGElement* contextElement, const UChar* currentCharacter)
41 {
42     float kerning = 0;
43     auto kerningLength = style->kerning();
44     if (kerningLength.unitType() == LengthTypePercentage)
45         kerning = kerningLength.valueAsPercentage() * m_font.pixelSize();
46     else {
47         SVGLengthContext lengthContext(contextElement);
48         kerning = kerningLength.value(lengthContext);
49     }
50 
51     const UChar* lastCharacter = m_lastCharacter;
52     m_lastCharacter = currentCharacter;
53 
54     if (!kerning && !m_font.letterSpacing() && !m_font.wordSpacing())
55         return 0;
56 
57     float spacing = m_font.letterSpacing() + kerning;
58     if (currentCharacter && lastCharacter && m_font.wordSpacing()) {
59         if (FontCascade::treatAsSpace(*currentCharacter) && !FontCascade::treatAsSpace(*lastCharacter))
60             spacing += m_font.wordSpacing();
61     }
62 
63     return spacing;
64 }
