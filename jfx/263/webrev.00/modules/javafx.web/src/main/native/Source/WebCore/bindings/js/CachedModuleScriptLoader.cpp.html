43 Ref<CachedModuleScriptLoader> CachedModuleScriptLoader::create(CachedModuleScriptLoaderClient& client, DeferredPromise& promise, CachedScriptFetcher& scriptFetcher, RefPtr<ModuleFetchParameters>&& parameters)
44 {
45     return adoptRef(*new CachedModuleScriptLoader(client, promise, scriptFetcher, WTFMove(parameters)));
46 }
47 
48 CachedModuleScriptLoader::CachedModuleScriptLoader(CachedModuleScriptLoaderClient& client, DeferredPromise& promise, CachedScriptFetcher& scriptFetcher, RefPtr<ModuleFetchParameters>&& parameters)
49     : m_client(&client)
50     , m_promise(&promise)
51     , m_scriptFetcher(scriptFetcher)
52     , m_parameters(WTFMove(parameters))
53 {
54 }
55 
56 CachedModuleScriptLoader::~CachedModuleScriptLoader()
57 {
58     if (m_cachedScript) {
59         m_cachedScript->removeClient(*this);
60         m_cachedScript = nullptr;
61     }
62 }
63 
64 bool CachedModuleScriptLoader::load(Document& document, const URL& sourceURL)
65 {
66     ASSERT(!m_cachedScript);
67     String integrity = m_parameters ? m_parameters->integrity() : String { };
68     m_cachedScript = m_scriptFetcher->requestModuleScript(document, sourceURL, WTFMove(integrity));
69     if (!m_cachedScript)
70         return false;
71     m_sourceURL = sourceURL;
72 
73     // If the content is already cached, this immediately calls notifyFinished.
74     m_cachedScript->addClient(*this);
75     return true;
76 }
77 
78 void CachedModuleScriptLoader::notifyFinished(CachedResource& resource)
79 {
80     ASSERT_UNUSED(resource, &resource == m_cachedScript);
81     ASSERT(m_cachedScript);
82     ASSERT(m_promise);
83 
84     Ref<CachedModuleScriptLoader> protectedThis(*this);
85     if (m_client)
86         m_client->notifyFinished(*this, WTFMove(m_promise));
87 
88     // Remove the client after calling notifyFinished to keep the data buffer in
89     // CachedResource alive while notifyFinished processes the resource.
90     m_cachedScript->removeClient(*this);
91     m_cachedScript = nullptr;
92 }
