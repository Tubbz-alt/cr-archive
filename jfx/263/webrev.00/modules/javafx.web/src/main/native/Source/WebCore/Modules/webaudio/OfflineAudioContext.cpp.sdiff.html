44 ExceptionOr<Ref<OfflineAudioContext>> OfflineAudioContext::create(ScriptExecutionContext& context, unsigned numberOfChannels, size_t numberOfFrames, float sampleRate)
45 {
46     // FIXME: Add support for workers.
47     if (!is<Document>(context))
48         return Exception { NotSupportedError };
49     if (!numberOfChannels || numberOfChannels > 10 || !numberOfFrames || !isSampleRateRangeGood(sampleRate))
50         return Exception { SyntaxError };
51     auto renderTarget = AudioBuffer::create(numberOfChannels, numberOfFrames, sampleRate);
52     if (!renderTarget)
53         return Exception { SyntaxError };
54 
55     auto audioContext = adoptRef(*new OfflineAudioContext(downcast<Document>(context), renderTarget.get()));
56     audioContext->suspendIfNeeded();
57     return audioContext;
58 }
