 49 ExceptionOr<Ref<MediaRecorder>> MediaRecorder::create(Document& document, Ref<MediaStream>&& stream, Options&& options)
 50 {
 51     auto privateInstance = MediaRecorder::createMediaRecorderPrivate(document, stream->privateStream());
 52     if (!privateInstance)
 53         return Exception { NotSupportedError, "The MediaRecorder is unsupported on this platform"_s };
 54     auto recorder = adoptRef(*new MediaRecorder(document, WTFMove(stream), WTFMove(privateInstance), WTFMove(options)));
 55     recorder->suspendIfNeeded();
 56     recorder->m_private->setErrorCallback([recorder = recorder.copyRef()](auto&& exception) mutable {
 57         recorder->dispatchError(WTFMove(*exception));
 58     });
 59     return recorder;
 60 }

 62 void MediaRecorder::setCustomPrivateRecorderCreator(creatorFunction creator)
 63 {
 64     m_customCreator = creator;
 65 }

 67 std::unique_ptr<MediaRecorderPrivate> MediaRecorder::createMediaRecorderPrivate(Document& document, const MediaStreamPrivate& stream)
 68 {
 69     if (m_customCreator)
 70         return m_customCreator();
 71 
 72 #if PLATFORM(COCOA)
 73     auto* page = document.page();
 74     if (!page)
 75         return nullptr;
 76 
 77     return page->mediaRecorderProvider().createMediaRecorderPrivate(stream);
 78 #else
 79     UNUSED_PARAM(document);
 80     UNUSED_PARAM(stream);
 81     return nullptr;
 82 #endif
 83 }

 85 MediaRecorder::MediaRecorder(Document& document, Ref<MediaStream>&& stream, std::unique_ptr<MediaRecorderPrivate>&& privateImpl, Options&& option)
 86     : ActiveDOMObject(document)
 87     , m_options(WTFMove(option))
 88     , m_stream(WTFMove(stream))
 89     , m_private(WTFMove(privateImpl))
 90 {
 91     m_tracks = WTF::map(m_stream->getTracks(), [] (auto&& track) -> Ref<MediaStreamTrackPrivate> {
 92         return track->privateTrack();
 93     });
 94     m_stream->addObserver(this);
 95 }
 96 
 97 MediaRecorder::~MediaRecorder()
 98 {
 99     m_stream->removeObserver(this);
100     stopRecordingInternal();
101 }

114 void MediaRecorder::suspend(ReasonForSuspension reason)
115 {
116     if (reason != ReasonForSuspension::BackForwardCache)
117         return;
118 
119     if (!m_isActive || state() == RecordingState::Inactive)
120         return;
121 
122     stopRecordingInternal();
123 
124     scheduleDeferredTask([this] {
125         dispatchEvent(MediaRecorderErrorEvent::create(eventNames().errorEvent, Exception { UnknownError, "MediaStream recording was interrupted"_s }));
126     });
127 }

134 ExceptionOr<void> MediaRecorder::startRecording(Optional<int> timeslice)
135 {
136     UNUSED_PARAM(timeslice);
137     if (state() != RecordingState::Inactive)
138         return Exception { InvalidStateError, "The MediaRecorder's state must be inactive in order to start recording"_s };
139 
140     for (auto& track : m_tracks)
141         track->addObserver(*this);
142 
143     m_state = RecordingState::Recording;
144     return { };
145 }
146 
147 ExceptionOr<void> MediaRecorder::stopRecording()
148 {
149     if (state() == RecordingState::Inactive)
150         return Exception { InvalidStateError, "The MediaRecorder's state cannot be inactive"_s };
151 
152     scheduleDeferredTask([this] {
153         if (!m_isActive || state() == RecordingState::Inactive)
154             return;
155 
156         stopRecordingInternal();
157         ASSERT(m_state == RecordingState::Inactive);
158         m_private->fetchData([this, protectedThis = makeRef(*this)](auto&& buffer, auto& mimeType) {
159             if (!m_isActive)
160                 return;
161 
162             dispatchEvent(BlobEvent::create(eventNames().dataavailableEvent, Event::CanBubble::No, Event::IsCancelable::No, buffer ? Blob::create(buffer.releaseNonNull(), mimeType) : Blob::create()));
163 
164             if (!m_isActive)
165                 return;
166 
167             dispatchEvent(Event::create(eventNames().stopEvent, Event::CanBubble::No, Event::IsCancelable::No));
168         });
169     });
170     return { };
171 }
172 
173 ExceptionOr<void> MediaRecorder::requestData()
174 {
175     if (state() == RecordingState::Inactive)
176         return Exception { InvalidStateError, "The MediaRecorder's state cannot be inactive"_s };
177 
178     m_private->fetchData([this, protectedThis = makeRef(*this)](auto&& buffer, auto& mimeType) {
179         if (!m_isActive)
180             return;
181 
182         dispatchEvent(BlobEvent::create(eventNames().dataavailableEvent, Event::CanBubble::No, Event::IsCancelable::No, buffer ? Blob::create(buffer.releaseNonNull(), mimeType) : Blob::create()));
183     });
184     return { };
185 }
186 
187 void MediaRecorder::stopRecordingInternal()
188 {
189     if (state() != RecordingState::Recording)
190         return;
191 
192     for (auto& track : m_tracks)
193         track->removeObserver(*this);
194 
195     m_state = RecordingState::Inactive;
196     m_private->stopRecording();
197 }
198 
199 void MediaRecorder::didAddOrRemoveTrack()
200 {
201     scheduleDeferredTask([this] {
202         if (!m_isActive || state() == RecordingState::Inactive)
203             return;
204         stopRecordingInternal();
205         dispatchError(Exception { UnknownError, "Track cannot be added to or removed from the MediaStream while recording is happening"_s });
206     });
207 }
208 
209 void MediaRecorder::dispatchError(Exception&& exception)
210 {
211     if (!m_isActive)
212         return;
213     dispatchEvent(MediaRecorderErrorEvent::create(eventNames().errorEvent, WTFMove(exception)));
214 }
215 
216 void MediaRecorder::trackEnded(MediaStreamTrackPrivate&)
217 {
218     auto position = m_tracks.findMatching([](auto& track) {
219         return !track->ended();
220     });
221     if (position != notFound)
222         return;
223 
224     stopRecording();
225 }

237 void MediaRecorder::scheduleDeferredTask(Function<void()>&& function)
238 {
239     ASSERT(function);
240     auto* document = this->document();
241     if (!document)
242         return;
243 
244     document->eventLoop().queueTask(TaskSource::Networking, [pendingActivity = makePendingActivity(*this), function = WTFMove(function)] {
245         function();
246     });
247 }
