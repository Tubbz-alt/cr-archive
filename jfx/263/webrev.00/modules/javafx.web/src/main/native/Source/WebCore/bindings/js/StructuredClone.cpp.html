 37 enum class CloneMode {
 38     Full,
 39     Partial,
 40 };
 41 
 42 static EncodedJSValue cloneArrayBufferImpl(JSGlobalObject* lexicalGlobalObject, CallFrame* callFrame, CloneMode mode)
 43 {
 44     VM& vm = lexicalGlobalObject->vm();
 45 
 46     ASSERT(lexicalGlobalObject);
 47     ASSERT(callFrame->argumentCount());
 48     ASSERT(callFrame->lexicalGlobalObject(vm) == lexicalGlobalObject);
 49 
 50     auto* buffer = toUnsharedArrayBuffer(vm, callFrame->uncheckedArgument(0));
 51     if (!buffer) {
 52         auto scope = DECLARE_THROW_SCOPE(vm);
 53         throwDataCloneError(*lexicalGlobalObject, scope);
 54         return { };
 55     }
 56     if (mode == CloneMode::Partial) {
 57         ASSERT(callFrame->argumentCount() == 3);
 58         int srcByteOffset = static_cast<int>(callFrame->uncheckedArgument(1).toNumber(lexicalGlobalObject));
 59         int srcLength = static_cast<int>(callFrame->uncheckedArgument(2).toNumber(lexicalGlobalObject));
 60         return JSValue::encode(JSArrayBuffer::create(lexicalGlobalObject->vm(), lexicalGlobalObject->arrayBufferStructure(ArrayBufferSharingMode::Default), buffer->slice(srcByteOffset, srcByteOffset + srcLength)));
 61     }
 62     return JSValue::encode(JSArrayBuffer::create(lexicalGlobalObject->vm(), lexicalGlobalObject->arrayBufferStructure(ArrayBufferSharingMode::Default), ArrayBuffer::tryCreate(buffer->data(), buffer->byteLength())));
 63 }
 64 
 65 EncodedJSValue JSC_HOST_CALL cloneArrayBuffer(JSGlobalObject* globalObject, CallFrame* callFrame)
 66 {
 67     return cloneArrayBufferImpl(globalObject, callFrame, CloneMode::Partial);
 68 }
 69 
 70 EncodedJSValue JSC_HOST_CALL structuredCloneArrayBuffer(JSGlobalObject* globalObject, CallFrame* callFrame)
 71 {
 72     return cloneArrayBufferImpl(globalObject, callFrame, CloneMode::Full);
 73 }

 75 EncodedJSValue JSC_HOST_CALL structuredCloneArrayBufferView(JSGlobalObject* globalObject, CallFrame* callFrame)
 76 {
 77     ASSERT(callFrame);
 78     ASSERT(callFrame->argumentCount());
 79 
 80     JSValue value = callFrame->uncheckedArgument(0);
 81     VM& vm = globalObject->vm();
 82     auto* bufferView = jsDynamicCast<JSArrayBufferView*>(vm, value);
 83     ASSERT(bufferView);
 84 
 85     auto* buffer = bufferView->unsharedBuffer();
 86     if (!buffer) {
 87         auto scope = DECLARE_THROW_SCOPE(vm);
 88         throwDataCloneError(*globalObject, scope);
 89         return { };
 90     }
 91     auto bufferClone = ArrayBuffer::tryCreate(buffer->data(), buffer->byteLength());
 92     Structure* structure = bufferView->structure(vm);
 93 
 94     if (jsDynamicCast<JSInt8Array*>(vm, value))
 95         return JSValue::encode(JSInt8Array::create(globalObject, structure, WTFMove(bufferClone), bufferView->byteOffset(), bufferView->length()));
 96     if (jsDynamicCast<JSInt16Array*>(vm, value))
 97         return JSValue::encode(JSInt16Array::create(globalObject, structure, WTFMove(bufferClone), bufferView->byteOffset(), bufferView->length()));
 98     if (jsDynamicCast<JSInt32Array*>(vm, value))
 99         return JSValue::encode(JSInt32Array::create(globalObject, structure, WTFMove(bufferClone), bufferView->byteOffset(), bufferView->length()));
100     if (jsDynamicCast<JSUint8Array*>(vm, value))
101         return JSValue::encode(JSUint8Array::create(globalObject, structure, WTFMove(bufferClone), bufferView->byteOffset(), bufferView->length()));
102     if (jsDynamicCast<JSUint8ClampedArray*>(vm, value))
103         return JSValue::encode(JSUint8ClampedArray::create(globalObject, structure, WTFMove(bufferClone), bufferView->byteOffset(), bufferView->length()));
104     if (jsDynamicCast<JSUint16Array*>(vm, value))
105         return JSValue::encode(JSUint16Array::create(globalObject, structure, WTFMove(bufferClone), bufferView->byteOffset(), bufferView->length()));
106     if (jsDynamicCast<JSUint32Array*>(vm, value))
107         return JSValue::encode(JSUint32Array::create(globalObject, structure, WTFMove(bufferClone), bufferView->byteOffset(), bufferView->length()));
108     if (jsDynamicCast<JSFloat32Array*>(vm, value))
109         return JSValue::encode(JSFloat32Array::create(globalObject, structure, WTFMove(bufferClone), bufferView->byteOffset(), bufferView->length()));
110     if (jsDynamicCast<JSFloat64Array*>(vm, value))
111         return JSValue::encode(JSFloat64Array::create(globalObject, structure, WTFMove(bufferClone), bufferView->byteOffset(), bufferView->length()));
112     if (jsDynamicCast<JSDataView*>(vm, value))
113         return JSValue::encode(JSDataView::create(globalObject, structure, WTFMove(bufferClone), bufferView->byteOffset(), bufferView->length()));
114 
115     ASSERT_NOT_REACHED();
116     return JSValue::encode(jsUndefined());
117 }
