 35 ActiveDOMObject::ActiveDOMObject(ScriptExecutionContext* scriptExecutionContext)
 36     : ContextDestructionObserver(scriptExecutionContext)
 37     , m_pendingActivityCount(0)
 38 #if !ASSERT_DISABLED
 39     , m_suspendIfNeededWasCalled(false)
 40 #endif
 41 {
 42     ASSERT(!is<Document>(m_scriptExecutionContext) || &downcast<Document>(m_scriptExecutionContext)->contextDocument() == downcast<Document>(m_scriptExecutionContext));
 43     if (!m_scriptExecutionContext)







 44         return;
 45 
 46     ASSERT(m_scriptExecutionContext->isContextThread());
 47     m_scriptExecutionContext->didCreateActiveDOMObject(*this);















 48 }
 49 
 50 ActiveDOMObject::~ActiveDOMObject()
 51 {
 52     ASSERT(canAccessThreadLocalDataForThread(m_creationThread));
 53 
 54     // ActiveDOMObject may be inherited by a sub-class whose life-cycle
 55     // exceeds that of the associated ScriptExecutionContext. In those cases,
 56     // m_scriptExecutionContext would/should have been nullified by
 57     // ContextDestructionObserver::contextDestroyed() (which we implement /
 58     // inherit). Hence, we should ensure that this is not 0 before use it
 59     // here.
 60     if (!m_scriptExecutionContext)
 61         return;
 62 
 63     ASSERT(m_suspendIfNeededWasCalled);
 64     ASSERT(m_scriptExecutionContext->isContextThread());
 65     m_scriptExecutionContext->willDestroyActiveDOMObject(*this);
 66 }
 67 
 68 void ActiveDOMObject::suspendIfNeeded()
 69 {
 70 #if !ASSERT_DISABLED
 71     ASSERT(!m_suspendIfNeededWasCalled);
 72     m_suspendIfNeededWasCalled = true;
 73 #endif
 74     if (!m_scriptExecutionContext)
 75         return;
 76 
 77     m_scriptExecutionContext->suspendActiveDOMObjectIfNeeded(*this);
 78 }
 79 
 80 #if !ASSERT_DISABLED
 81 
 82 void ActiveDOMObject::assertSuspendIfNeededWasCalled() const
 83 {


 84     ASSERT(m_suspendIfNeededWasCalled);
 85 }
 86 
 87 #endif
