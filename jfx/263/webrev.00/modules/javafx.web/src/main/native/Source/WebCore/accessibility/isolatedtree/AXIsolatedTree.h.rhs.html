37 class Page;

39 class AXIsolatedTree : public ThreadSafeRefCounted<AXIsolatedTree> {
40     WTF_MAKE_NONCOPYABLE(AXIsolatedTree); WTF_MAKE_FAST_ALLOCATED;
41 
42 public:
43     static Ref<AXIsolatedTree> create();
44     virtual ~AXIsolatedTree();
45 
46     static Ref<AXIsolatedTree> createTreeForPageID(PageIdentifier);
47     static void removeTreeForPageID(PageIdentifier);
48 
49     WEBCORE_EXPORT static RefPtr<AXIsolatedTree> treeForPageID(PageIdentifier);
50     WEBCORE_EXPORT static RefPtr<AXIsolatedTree> treeForID(AXIsolatedTreeID);
51     AXObjectCache* axObjectCache() const { return m_axObjectCache; }
52     void setAXObjectCache(AXObjectCache* axObjectCache) { m_axObjectCache = axObjectCache; }
53 
54     WEBCORE_EXPORT RefPtr<AXIsolatedObject> rootNode();
55     WEBCORE_EXPORT RefPtr<AXIsolatedObject> focusedUIElement();
56     RefPtr<AXIsolatedObject> nodeForID(AXID) const;
57     static RefPtr<AXIsolatedObject> nodeInTreeForID(AXIsolatedTreeID, AXID);
58 
59     // Call on main thread
60     void appendNodeChanges(Vector<Ref<AXIsolatedObject>>&);
61     void removeNode(AXID);
62 
63     void setRootNode(Ref<AXIsolatedObject>&);
64     void setFocusedNodeID(AXID);
65 
66     // Call on AX thread
67     void applyPendingChanges();
68 

69     AXIsolatedTreeID treeIdentifier() const { return m_treeID; }
70 
71 private:
72     AXIsolatedTree();
73 
74     static HashMap<AXIsolatedTreeID, Ref<AXIsolatedTree>>& treeIDCache();
75     static HashMap<PageIdentifier, Ref<AXIsolatedTree>>& treePageCache();
76 
77     AXObjectCache* m_axObjectCache { nullptr };
78 
79     // Only access on AX thread requesting data.
80     HashMap<AXID, Ref<AXIsolatedObject>> m_readerThreadNodeMap;
81 
82     // Written to by main thread under lock, accessed and applied by AX thread.
83     Vector<Ref<AXIsolatedObject>> m_pendingAppends;
84     Vector<AXID> m_pendingRemovals;
85     AXID m_pendingFocusedNodeID;

86     Lock m_changeLogLock;
87 
88     AXIsolatedTreeID m_treeID;
89     AXID m_rootNodeID { InvalidAXID };
90     AXID m_focusedNodeID { InvalidAXID };

91 };
