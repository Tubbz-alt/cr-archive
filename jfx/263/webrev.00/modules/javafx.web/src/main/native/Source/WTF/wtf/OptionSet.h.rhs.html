 26 #pragma once
 27 
 28 #include <initializer_list>
 29 #include <iterator>
 30 #include <type_traits>
 31 #include <wtf/Assertions.h>
 32 #include <wtf/MathExtras.h>
 33 
 34 namespace WTF {
 35 
 36 // OptionSet is a class that represents a set of enumerators in a space-efficient manner. The enumerators
 37 // must be powers of two greater than 0. This class is useful as a replacement for passing a bitmask of
 38 // enumerators around.
 39 template<typename T> class OptionSet {
 40     WTF_MAKE_FAST_ALLOCATED;
 41     static_assert(std::is_enum<T>::value, "T is not an enum type");

 42 
 43 public:
 44     using StorageType = std::make_unsigned_t<std::underlying_type_t<T>>;
 45 
 46     template<typename StorageType> class Iterator {
 47         WTF_MAKE_FAST_ALLOCATED;
 48     public:
 49         // Isolate the rightmost set bit.
 50         T operator*() const { return static_cast<T>(m_value & -m_value); }
 51 
 52         // Iterates from smallest to largest enum value by turning off the rightmost set bit.
 53         Iterator& operator++()
 54         {
 55             m_value &= m_value - 1;
 56             return *this;
 57         }
 58 
 59         Iterator& operator++(int) = delete;
 60 
 61         bool operator==(const Iterator& other) const { return m_value == other.m_value; }
 62         bool operator!=(const Iterator& other) const { return m_value != other.m_value; }
 63 
 64     private:
 65         Iterator(StorageType value) : m_value(value) { }
 66         friend OptionSet;
 67 
 68         StorageType m_value;
 69     };
 70     using iterator = Iterator<StorageType>;
 71 
 72     static constexpr OptionSet fromRaw(StorageType storageType)
 73     {
 74         return OptionSet(static_cast<T>(storageType), FromRawValue);
 75     }
 76 
 77     constexpr OptionSet() = default;
 78 
 79     constexpr OptionSet(T t)
 80         : m_storage(static_cast<StorageType>(t))
 81     {
 82 #ifndef NDEBUG
 83         // This assertion will conflict with the constexpr attribute if we enable it on NDEBUG builds.
 84         ASSERT_WITH_MESSAGE(!m_storage || hasOneBitSet(m_storage), "Enumerator is not a zero or a positive power of two.");
 85 #endif
 86     }
 87 
 88     constexpr OptionSet(std::initializer_list<T> initializerList)
 89     {
 90         for (auto& option : initializerList) {
 91 #ifndef NDEBUG
 92             // This assertion will conflict with the constexpr attribute if we enable it on NDEBUG builds.
 93             ASSERT_WITH_MESSAGE(hasOneBitSet(static_cast<StorageType>(option)), "Enumerator is not a positive power of two.");
 94 #endif
 95             m_storage |= static_cast<StorageType>(option);
 96         }
 97     }
 98 
 99     constexpr StorageType toRaw() const { return m_storage; }
100 
101     constexpr bool isEmpty() const { return !m_storage; }
102 
103     constexpr iterator begin() const { return m_storage; }
104     constexpr iterator end() const { return 0; }
105 
106     constexpr explicit operator bool() { return !isEmpty(); }
107 
108     constexpr bool contains(T option) const
109     {
110         return containsAny(option);
111     }
112 
113     constexpr bool containsAny(OptionSet optionSet) const
114     {
115         return !!(*this & optionSet);
116     }
117 
118     constexpr bool containsAll(OptionSet optionSet) const
119     {
120         return (*this & optionSet) == optionSet;
121     }
122 
123     constexpr void add(OptionSet optionSet)
124     {
125         m_storage |= optionSet.m_storage;
126     }
127 
128     constexpr void remove(OptionSet optionSet)
129     {
130         m_storage &= ~optionSet.m_storage;
131     }
132 
133     constexpr friend bool operator==(OptionSet lhs, OptionSet rhs)
134     {
135         return lhs.m_storage == rhs.m_storage;
136     }
137 
138     constexpr friend bool operator!=(OptionSet lhs, OptionSet rhs)
139     {
140         return lhs.m_storage != rhs.m_storage;
141     }
142 
143     constexpr friend OptionSet operator|(OptionSet lhs, OptionSet rhs)
144     {
145         return fromRaw(lhs.m_storage | rhs.m_storage);
146     }
147 
148     constexpr friend OptionSet operator&(OptionSet lhs, OptionSet rhs)
149     {
150         return fromRaw(lhs.m_storage & rhs.m_storage);
151     }
152 
153     constexpr friend OptionSet operator-(OptionSet lhs, OptionSet rhs)
154     {
155         return fromRaw(lhs.m_storage & ~rhs.m_storage);
156     }
157 
158     constexpr friend OptionSet operator^(OptionSet lhs, OptionSet rhs)
159     {
160         return fromRaw(lhs.m_storage ^ rhs.m_storage);
161     }
162 
163 private:
164     enum InitializationTag { FromRawValue };
165     constexpr OptionSet(T t, InitializationTag)
166         : m_storage(static_cast<StorageType>(t))
167     {
168     }
169     StorageType m_storage { 0 };
170 };
171 
172 }
173 
174 using WTF::OptionSet;
