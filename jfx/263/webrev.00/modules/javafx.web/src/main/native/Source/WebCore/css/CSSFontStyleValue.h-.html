28 #include "CSSPrimitiveValue.h"
29 #include "CSSValue.h"
30 #include "FontSelectionAlgorithm.h"
31 
32 namespace WebCore {
33 
34 class CSSFontStyleValue final : public CSSValue {
35 public:
36     static Ref<CSSFontStyleValue> create(Ref<CSSPrimitiveValue>&& fontStyleValue)
37     {
38         return adoptRef(*new CSSFontStyleValue(WTFMove(fontStyleValue), nullptr));
39     }
40     static Ref<CSSFontStyleValue> create(Ref<CSSPrimitiveValue>&& fontStyleValue, RefPtr<CSSPrimitiveValue>&& obliqueValue)
41     {
42         return adoptRef(*new CSSFontStyleValue(WTFMove(fontStyleValue), WTFMove(obliqueValue)));
43     }
44 
45     String customCSSText() const;
46 
47     bool equals(const CSSFontStyleValue&) const;
48 
49     bool isItalicOrOblique() const
50     {
51         if (!obliqueValue) {
52             auto valueID = fontStyleValue->valueID();
53             return valueID == CSSValueItalic || valueID == CSSValueOblique;
54         }
55         return obliqueValue->value<float>(CSSPrimitiveValue::CSS_DEG) >= static_cast<float>(italicValue());
56     }
57 
58     Ref<CSSPrimitiveValue> fontStyleValue;
59     RefPtr<CSSPrimitiveValue> obliqueValue;
60 
61 private:
62     CSSFontStyleValue(Ref<CSSPrimitiveValue>&& fontStyleValue, RefPtr<CSSPrimitiveValue>&& obliqueValue)
63         : CSSValue(FontStyleClass)
64         , fontStyleValue(WTFMove(fontStyleValue))
65         , obliqueValue(WTFMove(obliqueValue))
66     {
67     }
68 };
69 
70 }

72 SPECIALIZE_TYPE_TRAITS_CSS_VALUE(CSSFontStyleValue, isFontStyleValue())
