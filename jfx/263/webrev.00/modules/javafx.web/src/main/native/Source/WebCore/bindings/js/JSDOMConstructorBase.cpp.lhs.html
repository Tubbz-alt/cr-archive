27 namespace WebCore {
28 using namespace JSC;

30 STATIC_ASSERT_IS_TRIVIALLY_DESTRUCTIBLE(JSDOMConstructorBase);
31 
32 static EncodedJSValue JSC_HOST_CALL callThrowTypeError(ExecState* exec)
33 {
34     VM& vm = exec->vm();
35     auto scope = DECLARE_THROW_SCOPE(vm);
36     throwTypeError(exec, scope, "Constructor requires 'new' operator"_s);
37     return JSValue::encode(jsNull());
38 }
39 
40 CallType JSDOMConstructorBase::getCallData(JSCell*, CallData& callData)
41 {
42     callData.native.function = callThrowTypeError;
43     return CallType::Host;
44 }
45 
46 String JSDOMConstructorBase::className(const JSObject*, JSC::VM&)
47 {
48     return "Function"_s;
49 }
50 
51 String JSDOMConstructorBase::toStringName(const JSObject* object, JSC::ExecState* exec)
52 {
53     VM& vm = exec->vm();
54     const ClassInfo* info = object->classInfo(vm);
55     ASSERT(info);
56     return info->methodTable.className(object, vm);
57 }
