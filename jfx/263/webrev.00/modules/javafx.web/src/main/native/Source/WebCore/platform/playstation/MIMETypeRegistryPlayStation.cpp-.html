31 struct TypeExtensionPair {
32     ASCIILiteral type;
33     ASCIILiteral extension;
34 };
35 
36 const TypeExtensionPair commonMediaTypes[] = {
37     { "bmp"_s, "image/bmp"_s },
38     { "css"_s, "text/css"_s },
39     { "gif"_s, "image/gif"_s },
40     { "html"_s, "text/html"_s },
41     { "htm"_s, "text/html"_s },
42     { "ico"_s, "image/x-icon"_s },
43     { "jpeg"_s, "image/jpeg"_s },
44     { "jpg"_s, "image/jpeg"_s },
45     { "js"_s, "application/x-javascript"_s },
46     { "pdf"_s, "application/pdf"_s },
47     { "png"_s, "image/png"_s },
48     { "rss"_s, "application/rss+xml"_s },
49     { "svg"_s, "image/svg+xml"_s },
50     { "swf"_s, "application/x-shockwave-flash"_s },
51     { "text"_s, "text/plain"_s },
52     { "txt"_s, "text/plain"_s },
53     { "xbm"_s, "image/x-xbitmap"_s },
54     { "xml"_s, "text/xml"_s },
55     { "xsl"_s, "text/xsl"_s },
56     { "xhtml"_s, "application/xhtml+xml"_s },
57     { "wml"_s, "text/vnd.wap.wml"_s },
58     { "wmlc"_s, "application/vnd.wap.wmlc"_s },
59 };
60 
61 String MIMETypeRegistry::getMIMETypeForExtension(const String& extension)
62 {
63     for (auto& entry : commonMediaTypes) {
64         if (equalIgnoringASCIICase(extension, entry.extension.characters()))
65             return entry.type;
66     }
67     return emptyString();
68 }
69 
70 bool MIMETypeRegistry::isApplicationPluginMIMEType(const String&)
71 {
72     return false;
73 }
74 
75 String MIMETypeRegistry::getPreferredExtensionForMIMEType(const String& mimeType)
76 {
77     for (auto& entry : commonMediaTypes) {
78         if (equalIgnoringASCIICase(mimeType, entry.type.characters()))
79             return entry.extension;
80     }
81     return emptyString();
82 }
