24 #include "SVGExternalResourcesRequired.h"
25 #include "SVGGeometryElement.h"
26 #include "SVGNames.h"

28 namespace WebCore {
29 
30 class SVGCircleElement final : public SVGGeometryElement, public SVGExternalResourcesRequired {
31     WTF_MAKE_ISO_ALLOCATED(SVGCircleElement);
32 public:
33     static Ref<SVGCircleElement> create(const QualifiedName&, Document&);
34 
35     const SVGLengthValue& cx() const { return m_cx->currentValue(); }
36     const SVGLengthValue& cy() const { return m_cy->currentValue(); }
37     const SVGLengthValue& r() const { return m_r->currentValue(); }
38 
39     SVGAnimatedLength& cxAnimated() { return m_cx; }
40     SVGAnimatedLength& cyAnimated() { return m_cy; }
41     SVGAnimatedLength& rAnimated() { return m_r; }
42 
43 private:
44     SVGCircleElement(const QualifiedName&, Document&);
45 
46     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGCircleElement, SVGGeometryElement, SVGExternalResourcesRequired>;
47     const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
48 
49     void parseAttribute(const QualifiedName&, const AtomString&) final;
50     void svgAttributeChanged(const QualifiedName&) final;
51 
52     bool isValid() const final { return SVGTests::isValid(); }
53     bool selfHasRelativeLengths() const final { return true; }
54 
55     RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) final;
56 
57     PropertyRegistry m_propertyRegistry { *this };
58     Ref<SVGAnimatedLength> m_cx { SVGAnimatedLength::create(this, LengthModeWidth) };
59     Ref<SVGAnimatedLength> m_cy { SVGAnimatedLength::create(this, LengthModeHeight) };
60     Ref<SVGAnimatedLength> m_r { SVGAnimatedLength::create(this, LengthModeOther) };
61 };
