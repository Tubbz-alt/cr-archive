 26 #pragma once
 27 
 28 #if ENABLE(REMOTE_INSPECTOR)
 29 
 30 #include "RemoteInspectorSocket.h"
 31 
 32 #include <wtf/Condition.h>
 33 #include <wtf/Function.h>
 34 #include <wtf/HashMap.h>
 35 #include <wtf/Lock.h>
 36 #include <wtf/Threading.h>
 37 #include <wtf/Vector.h>
 38 
 39 namespace Inspector {
 40 
 41 class RemoteInspectorSocketEndpoint {
 42     WTF_MAKE_FAST_ALLOCATED;
 43 public:
 44     class Client {
 45     public:
 46         virtual void didReceive(ConnectionID, Vector<uint8_t>&&) = 0;
 47         virtual void didClose(ConnectionID) = 0;
 48     };
 49 
 50     class Listener {
 51     public:
 52         virtual bool didAccept(ConnectionID acceptedID, ConnectionID listenerID, Socket::Domain) = 0;
 53         virtual void didClose(ConnectionID) = 0;
 54     };
 55 
 56     static RemoteInspectorSocketEndpoint& singleton();
 57 
 58     RemoteInspectorSocketEndpoint();
 59     ~RemoteInspectorSocketEndpoint();
 60 
 61     Optional<ConnectionID> connectInet(const char* serverAddr, uint16_t serverPort, Client&);
 62     Optional<ConnectionID> listenInet(const char* address, uint16_t port, Listener&, Client&);
 63     void invalidateClient(Client&);
 64     void invalidateListener(Listener&);
 65 
 66     void send(ConnectionID, const uint8_t* data, size_t);
 67 
 68     Optional<ConnectionID> createClient(PlatformSocketType, Client&);
 69     Optional<ConnectionID> createListener(PlatformSocketType, Listener&, Client&);
 70 
 71     Optional<uint16_t> getPort(ConnectionID) const;
 72 
 73 protected:
 74     struct Connection {
 75         WTF_MAKE_STRUCT_FAST_ALLOCATED;
 76         explicit Connection(Client& client)
 77             : client(client)
 78         {
 79         }
 80 
 81         ConnectionID id;
 82         Vector<uint8_t> sendBuffer;
 83         PlatformSocketType socket { INVALID_SOCKET_VALUE };
 84         PollingDescriptor poll;
 85         Client& client;
 86         Listener* listener { };
 87     };
 88 
 89     ConnectionID generateConnectionID();
 90     std::unique_ptr<Connection> makeConnection(PlatformSocketType, Client&);
 91 
 92     void recvIfEnabled(ConnectionID);
 93     void sendIfEnabled(ConnectionID);
 94     void workerThread();
 95     void wakeupWorkerThread();
 96     void acceptInetSocketIfEnabled(ConnectionID);
 97     bool isListening(ConnectionID);
 98 
 99     mutable Lock m_connectionsLock;
100     HashMap<ConnectionID, std::unique_ptr<Connection>> m_connections;
101     HashMap<ConnectionID, std::unique_ptr<Connection>> m_listeners;
102 
103     PlatformSocketType m_wakeupSendSocket { INVALID_SOCKET_VALUE };
104     PlatformSocketType m_wakeupReceiveSocket { INVALID_SOCKET_VALUE };
105 
106     RefPtr<Thread> m_workerThread;
107     std::atomic<bool> m_shouldAbortWorkerThread { false };
108 };
109 
110 } // namespace Inspector
111 
112 #endif // ENABLE(REMOTE_INSPECTOR)
