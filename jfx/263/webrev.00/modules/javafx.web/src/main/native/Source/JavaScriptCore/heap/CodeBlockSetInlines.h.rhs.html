36 inline void CodeBlockSet::mark(const AbstractLocker&, CodeBlock* codeBlock)
37 {
38     if (!codeBlock)
39         return;
40 
41     // Conservative root scanning in Eden collection can only find PreciseAllocation that is allocated in this Eden cycle.
42     // Since CodeBlockSet::m_currentlyExecuting is strongly assuming that this catches all the currently executing CodeBlock,
43     // we now have a restriction that all CodeBlock needs to be a non-precise-allocation.
44     ASSERT(!codeBlock->isPreciseAllocation());
45     m_currentlyExecuting.add(codeBlock);
46 }
47 
48 template<typename Functor>
49 void CodeBlockSet::iterate(const Functor& functor)
50 {
51     auto locker = holdLock(m_lock);
52     iterate(locker, functor);
53 }
54 
55 template<typename Functor>
56 void CodeBlockSet::iterate(const AbstractLocker&, const Functor& functor)
57 {
58     for (CodeBlock* codeBlock : m_codeBlocks)
59         functor(codeBlock);
60 }
61 
62 template<typename Functor>
63 void CodeBlockSet::iterateViaSubspaces(VM& vm, const Functor& functor)
64 {
65     vm.forEachCodeBlockSpace(
66         [&] (auto& spaceAndSet) {
67             spaceAndSet.space.forEachLiveCell(
68                 [&] (HeapCell* cell, HeapCell::Kind) {
69                     functor(jsCast<CodeBlock*>(static_cast<JSCell*>(cell)));
70                 });
71         });
72 }
73 
74 template<typename Functor>
75 void CodeBlockSet::iterateCurrentlyExecuting(const Functor& functor)
76 {
77     LockHolder locker(&m_lock);
78     for (CodeBlock* codeBlock : m_currentlyExecuting)
79         functor(codeBlock);
80 }
