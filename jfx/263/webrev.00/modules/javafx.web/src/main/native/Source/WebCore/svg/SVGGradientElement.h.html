 32 enum SVGSpreadMethodType {
 33     SVGSpreadMethodUnknown = 0,
 34     SVGSpreadMethodPad,
 35     SVGSpreadMethodReflect,
 36     SVGSpreadMethodRepeat
 37 };
 38 
 39 template<>
 40 struct SVGPropertyTraits<SVGSpreadMethodType> {
 41     static unsigned highestEnumValue() { return SVGSpreadMethodRepeat; }
 42 
 43     static String toString(SVGSpreadMethodType type)
 44     {
 45         switch (type) {
 46         case SVGSpreadMethodUnknown:
 47             return emptyString();
 48         case SVGSpreadMethodPad:
 49             return "pad"_s;
 50         case SVGSpreadMethodReflect:
 51             return "reflect"_s;
 52         case SVGSpreadMethodRepeat:
 53             return "repeat"_s;
 54         }
 55 
 56         ASSERT_NOT_REACHED();
 57         return emptyString();
 58     }
 59 
 60     static SVGSpreadMethodType fromString(const String& value)
 61     {
 62         if (value == "pad")
 63             return SVGSpreadMethodPad;
 64         if (value == "reflect")
 65             return SVGSpreadMethodReflect;
 66         if (value == "repeat")
 67             return SVGSpreadMethodRepeat;
 68         return SVGSpreadMethodUnknown;
 69     }
 70 };
 71 
 72 class SVGGradientElement : public SVGElement, public SVGURIReference {
 73     WTF_MAKE_ISO_ALLOCATED(SVGGradientElement);
 74 public:
 75     enum {
 76         SVG_SPREADMETHOD_UNKNOWN = SVGSpreadMethodUnknown,
 77         SVG_SPREADMETHOD_PAD = SVGSpreadMethodReflect,
 78         SVG_SPREADMETHOD_REFLECT = SVGSpreadMethodRepeat,
 79         SVG_SPREADMETHOD_REPEAT = SVGSpreadMethodUnknown
 80     };
 81 
 82     Vector<Gradient::ColorStop> buildStops();
 83 
 84     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGGradientElement, SVGElement, SVGURIReference>;
 85 
 86     SVGSpreadMethodType spreadMethod() const { return m_spreadMethod->currentValue<SVGSpreadMethodType>(); }
 87     SVGUnitTypes::SVGUnitType gradientUnits() const { return m_gradientUnits->currentValue<SVGUnitTypes::SVGUnitType>(); }
 88     const SVGTransformList& gradientTransform() const { return m_gradientTransform->currentValue(); }
 89 
 90     SVGAnimatedEnumeration& spreadMethodAnimated() { return m_spreadMethod; }
 91     SVGAnimatedEnumeration& gradientUnitsAnimated() { return m_gradientUnits; }
 92     SVGAnimatedTransformList& gradientTransformAnimated() { return m_gradientTransform; }
 93 
 94 protected:
 95     SVGGradientElement(const QualifiedName&, Document&);
 96 
 97     void parseAttribute(const QualifiedName&, const AtomString&) override;
 98     void svgAttributeChanged(const QualifiedName&) override;
 99 
100 private:
101     bool needsPendingResourceHandling() const override { return false; }
102     void childrenChanged(const ChildChange&) override;
103 
104     Ref<SVGAnimatedEnumeration> m_spreadMethod { SVGAnimatedEnumeration::create(this, SVGSpreadMethodPad) };
105     Ref<SVGAnimatedEnumeration> m_gradientUnits { SVGAnimatedEnumeration::create(this, SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX) };
106     Ref<SVGAnimatedTransformList> m_gradientTransform { SVGAnimatedTransformList::create(this) };
107 };
108 
109 } // namespace WebCore

111 SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::SVGGradientElement)
112 static bool isType(const WebCore::SVGElement& element)
113 {
114     return element.hasTagName(WebCore::SVGNames::radialGradientTag) || element.hasTagName(WebCore::SVGNames::linearGradientTag);
115 }
116 static bool isType(const WebCore::Node& node)
117 {
118     return is<WebCore::SVGElement>(node) && isType(downcast<WebCore::SVGElement>(node));
119 }
