1 # The settings in this file are the WebKit project default values, and
  2 # are recommended for most ports. Ports can override these settings in
  3 # Options*.cmake, but should do so only if there is strong reason to
  4 # deviate from the defaults of the WebKit project (e.g. if the feature
  5 # requires platform-specific implementation that does not exist).
  6 
  7 set(_WEBKIT_AVAILABLE_OPTIONS "")
  8 
  9 set(PUBLIC YES)
 10 set(PRIVATE NO)
 11 
 12 macro(_ENSURE_OPTION_MODIFICATION_IS_ALLOWED)
 13     if (NOT _SETTING_WEBKIT_OPTIONS)
 14         message(FATAL_ERROR "Options must be set between WEBKIT_OPTION_BEGIN and WEBKIT_OPTION_END")
 15     endif ()
 16 endmacro()
 17 
 18 macro(_ENSURE_IS_WEBKIT_OPTION _name)
 19     list(FIND _WEBKIT_AVAILABLE_OPTIONS ${_name} ${_name}_OPTION_INDEX)
 20     if (${_name}_OPTION_INDEX EQUAL -1)
 21         message(FATAL_ERROR "${_name} is not a valid WebKit option")
 22     endif ()
 23 endmacro()
 24 
 25 macro(WEBKIT_OPTION_DEFINE _name _description _public _initial_value)
 26     _ENSURE_OPTION_MODIFICATION_IS_ALLOWED()
 27 
 28     set(_WEBKIT_AVAILABLE_OPTIONS_DESCRIPTION_${_name} ${_description})
 29     set(_WEBKIT_AVAILABLE_OPTIONS_IS_PUBLIC_${_name} ${_public})
 30     set(_WEBKIT_AVAILABLE_OPTIONS_INITIAL_VALUE_${_name} ${_initial_value})
 31     set(_WEBKIT_AVAILABLE_OPTIONS_${_name}_CONFLICTS "")
 32     set(_WEBKIT_AVAILABLE_OPTIONS_${_name}_DEPENDENCIES "")
 33     list(APPEND _WEBKIT_AVAILABLE_OPTIONS ${_name})
 34 
 35     EXPOSE_VARIABLE_TO_BUILD(${_name})
 36 endmacro()
 37 
 38 macro(WEBKIT_OPTION_DEFAULT_PORT_VALUE _name _public _value)
 39     _ENSURE_OPTION_MODIFICATION_IS_ALLOWED()
 40     _ENSURE_IS_WEBKIT_OPTION(${_name})
 41 
 42     set(_WEBKIT_AVAILABLE_OPTIONS_IS_PUBLIC_${_name} ${_public})
 43     set(_WEBKIT_AVAILABLE_OPTIONS_INITIAL_VALUE_${_name} ${_value})
 44 endmacro()

 46 macro(WEBKIT_OPTION_CONFLICT _name _conflict)
 47     _ENSURE_OPTION_MODIFICATION_IS_ALLOWED()
 48     _ENSURE_IS_WEBKIT_OPTION(${_name})
 49     _ENSURE_IS_WEBKIT_OPTION(${_conflict})
 50 
 51     list(APPEND _WEBKIT_AVAILABLE_OPTIONS_${_name}_CONFLICTS ${_conflict})
 52 endmacro()

 54 macro(WEBKIT_OPTION_DEPEND _name _depend)
 55     _ENSURE_OPTION_MODIFICATION_IS_ALLOWED()
 56     _ENSURE_IS_WEBKIT_OPTION(${_name})
 57     _ENSURE_IS_WEBKIT_OPTION(${_depend})
 58 
 59     list(APPEND _WEBKIT_AVAILABLE_OPTIONS_${_name}_DEPENDENCIES ${_depend})
 60 endmacro()

261 macro(_WEBKIT_OPTION_ENFORCE_DEPENDS _name)
262     foreach (_dependency ${_WEBKIT_AVAILABLE_OPTIONS_${_name}_DEPENDENCIES})
263         if (NOT ${_dependency})
264             message(STATUS "Disabling ${_name} since ${_dependency} is disabled.")
265             set(${_name} OFF)
266             set(_OPTION_CHANGED TRUE)
267             break ()
268         endif ()
269     endforeach ()
270 endmacro()
271 
272 macro(_WEBKIT_OPTION_ENFORCE_ALL_DEPENDS)
273     set(_OPTION_CHANGED TRUE)
274     while (${_OPTION_CHANGED})
275         set(_OPTION_CHANGED FALSE)
276         foreach (_name ${_WEBKIT_AVAILABLE_OPTIONS})
277             if (${_name})
278                 _WEBKIT_OPTION_ENFORCE_DEPENDS(${_name})
279             endif ()
280         endforeach ()
281     endwhile ()
282 endmacro()

284 macro(_WEBKIT_OPTION_ENFORCE_CONFLICTS _name)
285     foreach (_conflict ${_WEBKIT_AVAILABLE_OPTIONS_${_name}_CONFLICTS})
286         if (${_conflict})
287             message(FATAL_ERROR "${_name} conflicts with ${_conflict}. You must disable one or the other.")
288         endif ()
289     endforeach ()
290 endmacro()

292 macro(_WEBKIT_OPTION_ENFORCE_ALL_CONFLICTS)
293     foreach (_name ${_WEBKIT_AVAILABLE_OPTIONS})
294         if (${_name})
295             _WEBKIT_OPTION_ENFORCE_CONFLICTS(${_name})
296         endif ()
297     endforeach ()
298 endmacro()

300 macro(WEBKIT_OPTION_END)
301     set(_SETTING_WEBKIT_OPTIONS FALSE)
302 
303     list(SORT _WEBKIT_AVAILABLE_OPTIONS)
304     set(_MAX_FEATURE_LENGTH 0)
305     foreach (_name ${_WEBKIT_AVAILABLE_OPTIONS})
306         string(LENGTH ${_name} _name_length)
307         if (_name_length GREATER _MAX_FEATURE_LENGTH)
308             set(_MAX_FEATURE_LENGTH ${_name_length})
309         endif ()
310 
311         option(${_name} "${_WEBKIT_AVAILABLE_OPTIONS_DESCRIPTION_${_name}}" ${_WEBKIT_AVAILABLE_OPTIONS_INITIAL_VALUE_${_name}})
312         if (NOT ${_WEBKIT_AVAILABLE_OPTIONS_IS_PUBLIC_${_name}})
313             mark_as_advanced(FORCE ${_name})
314         endif ()
315     endforeach ()
316 
317     # Run through every possible depends to make sure we have disabled anything
318     # that could cause an unnecessary conflict before processing conflicts.
319     _WEBKIT_OPTION_ENFORCE_ALL_DEPENDS()
320     _WEBKIT_OPTION_ENFORCE_ALL_CONFLICTS()
321 
322     foreach (_name ${_WEBKIT_AVAILABLE_OPTIONS})
323         if (${_name})
324             list(APPEND FEATURE_DEFINES ${_name})
325             set(FEATURE_DEFINES_WITH_SPACE_SEPARATOR "${FEATURE_DEFINES_WITH_SPACE_SEPARATOR} ${_name}")
326         endif ()
327     endforeach ()
328 endmacro()

330 macro(PRINT_WEBKIT_OPTIONS)
331     message(STATUS "Enabled features:")
332 
333     set(_should_print_dots ON)
334     foreach (_name ${_WEBKIT_AVAILABLE_OPTIONS})
335         if (${_WEBKIT_AVAILABLE_OPTIONS_IS_PUBLIC_${_name}})
336             string(LENGTH ${_name} _name_length)
337             set(_message " ${_name} ")
338 
339             # Print dots on every other row, for readability.
340             foreach (IGNORE RANGE ${_name_length} ${_MAX_FEATURE_LENGTH})
341                 if (${_should_print_dots})
342                     set(_message "${_message}.")
343                 else ()
344                     set(_message "${_message} ")
345                 endif ()
346             endforeach ()
347 
348             set(_should_print_dots (NOT ${_should_print_dots}))
349 
350             set(_message "${_message} ${${_name}}")

362 macro(SET_AND_EXPOSE_TO_BUILD _variable_name)
363     # It's important to handle the case where the value isn't passed, because often
364     # during configuration an empty variable is the result of a failed package search.
365     if (${ARGC} GREATER 1)
366         set(_variable_value ${ARGV1})
367     else ()
368         set(_variable_value OFF)
369     endif ()
370 
371     set(${_variable_name} ${_variable_value})
372     EXPOSE_VARIABLE_TO_BUILD(${_variable_name})
373 endmacro()

375 macro(_ADD_CONFIGURATION_LINE_TO_HEADER_STRING _string _variable_name _output_variable_name)
376     if (${${_variable_name}})
377         set(${_string} "${_file_contents}#define ${_output_variable_name} 1\n")
378     else ()
379         set(${_string} "${_file_contents}#define ${_output_variable_name} 0\n")
380     endif ()
381 endmacro()

383 macro(CREATE_CONFIGURATION_HEADER)
384     list(SORT _WEBKIT_CONFIG_FILE_VARIABLES)
385     set(_file_contents "#ifndef CMAKECONFIG_H\n")
386     set(_file_contents "${_file_contents}#define CMAKECONFIG_H\n\n")
387 
388     foreach (_variable_name ${_WEBKIT_CONFIG_FILE_VARIABLES})
389         _ADD_CONFIGURATION_LINE_TO_HEADER_STRING(_file_contents ${_variable_name} ${_variable_name})
390     endforeach ()
391     set(_file_contents "${_file_contents}\n#endif /* CMAKECONFIG_H */\n")
392 
393     file(WRITE "${CMAKE_BINARY_DIR}/cmakeconfig.h.tmp" "${_file_contents}")
394     execute_process(COMMAND ${CMAKE_COMMAND}
395         -E copy_if_different
396         "${CMAKE_BINARY_DIR}/cmakeconfig.h.tmp"
397         "${CMAKE_BINARY_DIR}/cmakeconfig.h"
398     )
399     file(REMOVE "${CMAKE_BINARY_DIR}/cmakeconfig.h.tmp")
400 endmacro()
