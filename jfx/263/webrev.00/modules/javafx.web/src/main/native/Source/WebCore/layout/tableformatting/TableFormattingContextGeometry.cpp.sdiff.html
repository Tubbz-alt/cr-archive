37 ContentHeightAndMargin TableFormattingContext::Geometry::tableCellHeightAndMargin(const Box& layoutBox) const
38 {
39     ASSERT(layoutBox.isInFlow());
40 
41     auto height = computedContentHeight(layoutBox);
42     if (!height)
43         height = contentHeightForFormattingContextRoot(layoutBox);
44 
45     // Margins don't apply to internal table elements.
46     return ContentHeightAndMargin { *height, { } };
47 }
