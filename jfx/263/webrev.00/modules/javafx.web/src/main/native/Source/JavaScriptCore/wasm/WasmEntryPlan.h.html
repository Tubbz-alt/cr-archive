 36 #include <wtf/Vector.h>
 37 
 38 namespace JSC {
 39 
 40 class CallLinkInfo;
 41 
 42 namespace Wasm {
 43 
 44 class EntryPlan : public Plan, public StreamingParserClient {
 45 public:
 46     using Base = Plan;
 47     enum AsyncWork : uint8_t { FullCompile, Validation };
 48 
 49     // Note: CompletionTask should not hold a reference to the Plan otherwise there will be a reference cycle.
 50     EntryPlan(Context*, Ref<ModuleInformation>, AsyncWork, CompletionTask&&);
 51     JS_EXPORT_PRIVATE EntryPlan(Context*, Vector<uint8_t>&&, AsyncWork, CompletionTask&&);
 52 
 53     virtual ~EntryPlan() = default;
 54 
 55     void prepare();
 56 
 57     void compileFunctions(CompilationEffort);
 58 
 59     Ref<ModuleInformation>&& takeModuleInformation()
 60     {
 61         RELEASE_ASSERT(!failed() && !hasWork());
 62         return WTFMove(m_moduleInformation);
 63     }
 64 
 65     Vector<MacroAssemblerCodeRef<WasmEntryPtrTag>>&& takeWasmToWasmExitStubs()
 66     {
 67         RELEASE_ASSERT(!failed() && !hasWork());
 68         return WTFMove(m_wasmToWasmExitStubs);
 69     }
 70 
 71     Vector<Vector<UnlinkedWasmToWasmCall>> takeWasmToWasmCallsites()
 72     {
 73         RELEASE_ASSERT(!failed() && !hasWork());
 74         return WTFMove(m_unlinkedWasmToWasmCalls);
 75     }
 76 
 77     enum class State : uint8_t {
 78         Initial,
 79         Validated,
 80         Prepared,
 81         Compiled,
 82         Completed // We should only move to Completed if we are holding the lock.
 83     };
 84 
 85     bool multiThreaded() const override { return m_state >= State::Prepared; }
 86 
 87 private:
 88     class ThreadCountHolder;
 89     friend class ThreadCountHolder;
 90 
 91 protected:
 92     // For some reason friendship doesn't extend to parent classes...
 93     using Base::m_lock;
 94 
 95     bool parseAndValidateModule(const uint8_t*, size_t);
 96 
 97     const char* stateString(State);
 98     void moveToState(State);
 99     bool isComplete() const override { return m_state == State::Completed; }
100     void complete(const AbstractLocker&) override;
101 
102     virtual bool prepareImpl() = 0;
103     virtual void compileFunction(uint32_t functionIndex) = 0;
104     virtual void didCompleteCompilation(const AbstractLocker&) = 0;
105 
106     template<typename T>
107     bool tryReserveCapacity(Vector<T>& vector, size_t size, const char* what)
108     {
109         if (UNLIKELY(!vector.tryReserveCapacity(size))) {
110             fail(holdLock(m_lock), WTF::makeString("Failed allocating enough space for ", size, what));
111             return false;
112         }
113         return true;
114     }
115 
116     Vector<uint8_t> m_source;
117     Vector<MacroAssemblerCodeRef<WasmEntryPtrTag>> m_wasmToWasmExitStubs;
118     HashSet<uint32_t, typename DefaultHash<uint32_t>::Hash, WTF::UnsignedWithZeroKeyHashTraits<uint32_t>> m_exportedFunctionIndices;
119 
120     Vector<Vector<UnlinkedWasmToWasmCall>> m_unlinkedWasmToWasmCalls;
121     StreamingParser m_streamingParser;
122     State m_state;
123 
124     const AsyncWork m_asyncWork;
125     uint8_t m_numberOfActiveThreads { 0 };
126     uint32_t m_currentIndex { 0 };
127     uint32_t m_numberOfFunctions { 0 };
128 };
129 
130 
131 } } // namespace JSC::Wasm
