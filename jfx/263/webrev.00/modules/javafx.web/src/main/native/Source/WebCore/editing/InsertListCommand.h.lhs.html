26 #pragma once
27 
28 #include "CompositeEditCommand.h"
29 
30 namespace WebCore {
31 
32 class HTMLElement;
33 class HTMLQualifiedName;
34 
35 class InsertListCommand final : public CompositeEditCommand {
36 public:
37     enum class Type : uint8_t { OrderedList, UnorderedList };
38 
39     static Ref<InsertListCommand> create(Document& document, Type listType)
40     {
41         return adoptRef(*new InsertListCommand(document, listType));
42     }
43 
44     static RefPtr<HTMLElement> insertList(Document&, Type);
45 
46     bool preservesTypingStyle() const final { return true; }
47 
48 private:
49     InsertListCommand(Document&, Type);
50 
51     void doApply() final;
52     EditAction editingAction() const final;
53 
54     HTMLElement& fixOrphanedListChild(Node&);
55     bool selectionHasListOfType(const VisibleSelection& selection, const QualifiedName&);
56     Ref<HTMLElement> mergeWithNeighboringLists(HTMLElement&);
57     void doApplyForSingleParagraph(bool forceCreateList, const HTMLQualifiedName&, Range* currentSelection);
58     void unlistifyParagraph(const VisiblePosition& originalStart, HTMLElement* listNode, Node* listChildNode);
59     RefPtr<HTMLElement> listifyParagraph(const VisiblePosition& originalStart, const QualifiedName& listTag);
60     RefPtr<HTMLElement> m_listElement;
61     Type m_type;
62 };
63 
64 } // namespace WebCore
