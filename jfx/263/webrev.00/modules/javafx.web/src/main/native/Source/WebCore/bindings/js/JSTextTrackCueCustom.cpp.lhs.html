41 bool JSTextTrackCueOwner::isReachableFromOpaqueRoots(JSC::Handle<JSC::Unknown> handle, void*, SlotVisitor& visitor, const char** reason)
42 {
43     JSTextTrackCue* jsTextTrackCue = jsCast<JSTextTrackCue*>(handle.slot()->asCell());
44     TextTrackCue& textTrackCue = jsTextTrackCue->wrapped();
45 
46     // If the cue is firing event listeners, its wrapper is reachable because
47     // the wrapper is responsible for marking those event listeners.
48     if (textTrackCue.isFiringEventListeners()) {
49         if (UNLIKELY(reason))
50             *reason = "TextTrackCue is firing event listeners";
51         return true;
52     }
53 
54     // If the cue is not associated with a track, it is not reachable.
55     if (!textTrackCue.track())
56         return false;
57 
58     if (UNLIKELY(reason))
59         *reason = "TextTrack is an opaque root";
60 
61     return visitor.containsOpaqueRoot(root(textTrackCue.track()));
62 }

64 JSValue toJSNewlyCreated(ExecState*, JSDOMGlobalObject* globalObject, Ref<TextTrackCue>&& cue)
65 {
66     switch (cue->cueType()) {
67     case TextTrackCue::Data:
68         return createWrapper<DataCue>(globalObject, WTFMove(cue));
69     case TextTrackCue::WebVTT:
70     case TextTrackCue::Generic:
71         return createWrapper<VTTCue>(globalObject, WTFMove(cue));
72     default:
73         ASSERT_NOT_REACHED();
74         return jsNull();
75     }



76 }
77 
78 JSValue toJS(ExecState* state, JSDOMGlobalObject* globalObject, TextTrackCue& cue)
79 {
80     return wrap(state, globalObject, cue);
81 }
82 
83 void JSTextTrackCue::visitAdditionalChildren(SlotVisitor& visitor)
84 {
85     if (TextTrack* textTrack = wrapped().track())
86         visitor.addOpaqueRoot(root(textTrack));
87 }
