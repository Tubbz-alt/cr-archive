 71 bool EventTarget::addEventListener(const AtomString& eventType, Ref<EventListener>&& listener, const AddEventListenerOptions& options)
 72 {
 73 #if ASSERT_ENABLED
 74     listener->checkValidityForEventTarget(*this);
 75 #endif
 76 
 77     auto passive = options.passive;
 78 
 79     if (!passive.hasValue() && Quirks::shouldMakeEventListenerPassive(*this, eventType, listener.get()))
 80         passive = true;









 81 
 82     bool listenerCreatedFromScript = listener->type() == EventListener::JSEventListenerType && !listener->wasCreatedFromMarkup();
 83     auto listenerRef = listener.copyRef();
 84 
 85     if (!ensureEventTargetData().eventListenerMap.add(eventType, WTFMove(listener), { options.capture, passive.valueOr(false), options.once }))
 86         return false;
 87 
 88     if (listenerCreatedFromScript)
 89         InspectorInstrumentation::didAddEventListener(*this, eventType, listenerRef.get(), options.capture);
 90 
 91     return true;
 92 }

 94 void EventTarget::addEventListenerForBindings(const AtomString& eventType, RefPtr<EventListener>&& listener, AddEventListenerOptionsOrBoolean&& variant)
 95 {
 96     if (!listener)
 97         return;
 98 
 99     auto visitor = WTF::makeVisitor([&](const AddEventListenerOptions& options) {
100         addEventListener(eventType, listener.releaseNonNull(), options);
101     }, [&](bool capture) {
102         addEventListener(eventType, listener.releaseNonNull(), capture);
103     });
104 
105     WTF::visit(visitor, variant);
106 }

108 void EventTarget::removeEventListenerForBindings(const AtomString& eventType, RefPtr<EventListener>&& listener, ListenerOptionsOrBoolean&& variant)
109 {
110     if (!listener)
111         return;
112 
113     auto visitor = WTF::makeVisitor([&](const ListenerOptions& options) {
114         removeEventListener(eventType, *listener, options);
115     }, [&](bool capture) {
116         removeEventListener(eventType, *listener, capture);
117     });
118 
119     WTF::visit(visitor, variant);
120 }

122 bool EventTarget::removeEventListener(const AtomString& eventType, EventListener& listener, const ListenerOptions& options)
123 {
124     auto* data = eventTargetData();
125     if (!data)
126         return false;
127 
128     InspectorInstrumentation::willRemoveEventListener(*this, eventType, listener, options.capture);
129 
130     return data->eventListenerMap.remove(eventType, listener, options.capture);
131 }

133 bool EventTarget::setAttributeEventListener(const AtomString& eventType, RefPtr<EventListener>&& listener, DOMWrapperWorld& isolatedWorld)
134 {
135     auto* existingListener = attributeEventListener(eventType, isolatedWorld);
136     if (!listener) {
137         if (existingListener)
138             removeEventListener(eventType, *existingListener, false);
139         return false;
140     }
141     if (existingListener) {
142         InspectorInstrumentation::willRemoveEventListener(*this, eventType, *existingListener, false);
143 
144 #if ASSERT_ENABLED
145         listener->checkValidityForEventTarget(*this);
146 #endif
147 
148         auto listenerPointer = listener.copyRef();
149         eventTargetData()->eventListenerMap.replace(eventType, *existingListener, listener.releaseNonNull(), { });
150 
151         InspectorInstrumentation::didAddEventListener(*this, eventType, *listenerPointer, false);
152 
153         return true;
154     }
155     return addEventListener(eventType, listener.releaseNonNull());
156 }

158 EventListener* EventTarget::attributeEventListener(const AtomString& eventType, DOMWrapperWorld& isolatedWorld)
159 {
160     for (auto& eventListener : eventListeners(eventType)) {
161         auto& listener = eventListener->callback();
162         if (!listener.isAttribute())
163             continue;
164 
165         auto& listenerWorld = downcast<JSEventListener>(listener).isolatedWorld();
166         if (&listenerWorld == &isolatedWorld)
167             return &listener;
168     }
169 
170     return nullptr;
171 }

179 ExceptionOr<bool> EventTarget::dispatchEventForBindings(Event& event)
180 {


181     if (!event.isInitialized() || event.isBeingDispatched())
182         return Exception { InvalidStateError };
183 
184     if (!scriptExecutionContext())
185         return false;
186 
187     event.setUntrusted();
188 
189     dispatchEvent(event);
190     return event.legacyReturnValue();
191 }

193 void EventTarget::dispatchEvent(Event& event)
194 {
195     // FIXME: We should always use EventDispatcher.
196     ASSERT(event.isInitialized());
197     ASSERT(!event.isBeingDispatched());
198 
199     event.setTarget(this);
200     event.setCurrentTarget(this);
201     event.setEventPhase(Event::AT_TARGET);
202     event.resetBeforeDispatch();
203     fireEventListeners(event, EventInvokePhase::Capturing);
204     fireEventListeners(event, EventInvokePhase::Bubbling);
205     event.resetAfterDispatch();
206 }

212 static const AtomString& legacyType(const Event& event)
213 {
214     if (event.type() == eventNames().animationendEvent)
215         return eventNames().webkitAnimationEndEvent;
216 
217     if (event.type() == eventNames().animationstartEvent)
218         return eventNames().webkitAnimationStartEvent;
219 
220     if (event.type() == eventNames().animationiterationEvent)
221         return eventNames().webkitAnimationIterationEvent;
222 
223     if (event.type() == eventNames().transitionendEvent)
224         return eventNames().webkitTransitionEndEvent;
225 
226     // FIXME: This legacy name is not part of the specification (https://dom.spec.whatwg.org/#dispatching-events).
227     if (event.type() == eventNames().wheelEvent)
228         return eventNames().mousewheelEvent;
229 
230     return nullAtom();
231 }

233 // https://dom.spec.whatwg.org/#concept-event-listener-invoke
234 void EventTarget::fireEventListeners(Event& event, EventInvokePhase phase)
235 {
236     ASSERT_WITH_SECURITY_IMPLICATION(ScriptDisallowedScope::isEventAllowedInMainThread());
237     ASSERT(event.isInitialized());
238 
239     auto* data = eventTargetData();
240     if (!data)
241         return;
242 
243     SetForScope<bool> firingEventListenersScope(data->isFiringEventListeners, true);
244 
245     if (auto* listenersVector = data->eventListenerMap.find(event.type())) {
246         innerInvokeEventListeners(event, *listenersVector, phase);
247         return;
248     }
249 
250     // Only fall back to legacy types for trusted events.
251     if (!event.isTrusted())
252         return;
253 
254     const AtomString& legacyTypeName = legacyType(event);
255     if (!legacyTypeName.isNull()) {
256         if (auto* legacyListenersVector = data->eventListenerMap.find(legacyTypeName)) {
257             AtomString typeName = event.type();
258             event.setType(legacyTypeName);
259             innerInvokeEventListeners(event, *legacyListenersVector, phase);
260             event.setType(typeName);
261         }
262     }
263 }

265 // Intentionally creates a copy of the listeners vector to avoid event listeners added after this point from being run.
266 // Note that removal still has an effect due to the removed field in RegisteredEventListener.
267 // https://dom.spec.whatwg.org/#concept-event-listener-inner-invoke
268 void EventTarget::innerInvokeEventListeners(Event& event, EventListenerVector listeners, EventInvokePhase phase)
269 {
270     Ref<EventTarget> protectedThis(*this);
271     ASSERT(!listeners.isEmpty());
272     ASSERT(scriptExecutionContext());
273 
274     auto& context = *scriptExecutionContext();
275     bool contextIsDocument = is<Document>(context);

276     if (contextIsDocument)
277         InspectorInstrumentation::willDispatchEvent(downcast<Document>(context), event);
278 
279     for (auto& registeredListener : listeners) {
280         if (UNLIKELY(registeredListener->wasRemoved()))
281             continue;
282 
283         if (phase == EventInvokePhase::Capturing && !registeredListener->useCapture())
284             continue;
285         if (phase == EventInvokePhase::Bubbling && registeredListener->useCapture())
286             continue;
287 
288         if (InspectorInstrumentation::isEventListenerDisabled(*this, event.type(), registeredListener->callback(), registeredListener->useCapture()))
289             continue;
290 
291         // If stopImmediatePropagation has been called, we just break out immediately, without
292         // handling any more events on this target.
293         if (event.immediatePropagationStopped())
294             break;
295 
296         // Do this before invocation to avoid reentrancy issues.
297         if (registeredListener->isOnce())
298             removeEventListener(event.type(), registeredListener->callback(), ListenerOptions(registeredListener->useCapture()));
299 
300         if (registeredListener->isPassive())
301             event.setInPassiveListener(true);
302 
303 #if ASSERT_ENABLED
304         registeredListener->callback().checkValidityForEventTarget(*this);
305 #endif
306 
307         InspectorInstrumentation::willHandleEvent(context, event, *registeredListener);
308         registeredListener->callback().handleEvent(context, event);
309         InspectorInstrumentation::didHandleEvent(context);
310 
311         if (registeredListener->isPassive())
312             event.setInPassiveListener(false);
313     }
314 
315     if (contextIsDocument)
316         InspectorInstrumentation::didDispatchEvent(downcast<Document>(context), event);
317 }

319 Vector<AtomString> EventTarget::eventTypes()

321     if (auto* data = eventTargetData())
322         return data->eventListenerMap.eventTypes();

326 const EventListenerVector& EventTarget::eventListeners(const AtomString& eventType)
327 {
328     auto* data = eventTargetData();
329     auto* listenerVector = data ? data->eventListenerMap.find(eventType) : nullptr;
330     static NeverDestroyed<EventListenerVector> emptyVector;
331     return listenerVector ? *listenerVector : emptyVector.get();
332 }

334 void EventTarget::removeAllEventListeners()
335 {
336     auto& threadData = threadGlobalData();
337     RELEASE_ASSERT(!threadData.isInRemoveAllEventListeners());
338 
339     threadData.setIsInRemoveAllEventListeners(true);
340 
341     auto* data = eventTargetData();
342     if (data)
343         data->eventListenerMap.clear();
344 
345     threadData.setIsInRemoveAllEventListeners(false);
346 }

348 void EventTarget::visitJSEventListeners(JSC::SlotVisitor& visitor)
349 {
350     EventTargetData* data = eventTargetDataConcurrently();
351     if (!data)
352         return;
353 
354     auto locker = holdLock(data->eventListenerMap.lock());
355     EventListenerIterator iterator(&data->eventListenerMap);
356     while (auto* listener = iterator.nextListener())
357         listener->visitJSFunction(visitor);
358 }
