 77 static inline bool isReachableFromDOM(Node* node, SlotVisitor& visitor, const char** reason)
 78 {
 79     if (!node->isConnected()) {
 80         if (is<Element>(*node)) {
 81             auto& element = downcast<Element>(*node);
 82 
 83             // If a wrapper is the last reference to an image element
 84             // that is loading but not in the document, the wrapper is observable
 85             // because it is the only thing keeping the image element alive, and if
 86             // the element is destroyed, its load event will not fire.
 87             // FIXME: The DOM should manage this issue without the help of JavaScript wrappers.
 88             if (is<HTMLImageElement>(element)) {
 89                 if (downcast<HTMLImageElement>(element).hasPendingActivity()) {
 90                     if (UNLIKELY(reason))
 91                         *reason = "Image element with pending activity";
 92                     return true;
 93                 }
 94             }
 95 #if ENABLE(VIDEO)
 96             else if (is<HTMLAudioElement>(element)) {
 97                 if (!downcast<HTMLAudioElement>(element).paused()) {
 98                     if (UNLIKELY(reason))
 99                         *reason = "Audio element which is not paused";
100                     return true;
101                 }
102             }
103 #endif
104         }
105 
106         // If a node is firing event listeners, its wrapper is observable because
107         // its wrapper is responsible for marking those event listeners.
108         if (node->isFiringEventListeners()) {
109             if (UNLIKELY(reason))
110                 *reason = "Node which is firing event listeners";
111             return true;
112         }
113         if (GCReachableRefMap::contains(*node)) {
114             if (UNLIKELY(reason))
115                 *reason = "Node is scheduled to be used in an async script invocation)";
116             return true;
117         }
118     }
119 
120     if (UNLIKELY(reason))
121         *reason = "Connected node";
122 
123     return visitor.containsOpaqueRoot(root(node));
124 }

126 bool JSNodeOwner::isReachableFromOpaqueRoots(JSC::Handle<JSC::Unknown> handle, void*, SlotVisitor& visitor, const char** reason)
127 {
128     JSNode* jsNode = jsCast<JSNode*>(handle.slot()->asCell());
129     return isReachableFromDOM(&jsNode->wrapped(), visitor, reason);
130 }

132 JSScope* JSNode::pushEventHandlerScope(JSGlobalObject* lexicalGlobalObject, JSScope* node) const
133 {
134     if (inherits<JSHTMLElement>(lexicalGlobalObject->vm()))
135         return jsCast<const JSHTMLElement*>(this)->pushEventHandlerScope(lexicalGlobalObject, node);
136     return node;

144 static ALWAYS_INLINE JSValue createWrapperInline(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, Ref<Node>&& node)
145 {
146     ASSERT(!getCachedWrapper(globalObject->world(), node));
147 
148     JSDOMObject* wrapper;
149     switch (node->nodeType()) {
150         case Node::ELEMENT_NODE:
151             if (is<HTMLElement>(node))
152                 wrapper = createJSHTMLWrapper(globalObject, static_reference_cast<HTMLElement>(WTFMove(node)));
153             else if (is<SVGElement>(node))
154                 wrapper = createJSSVGWrapper(globalObject, static_reference_cast<SVGElement>(WTFMove(node)));
155 #if ENABLE(MATHML)
156             else if (is<MathMLElement>(node))
157                 wrapper = createJSMathMLWrapper(globalObject, static_reference_cast<MathMLElement>(WTFMove(node)));
158 #endif
159             else
160                 wrapper = createWrapper<Element>(globalObject, WTFMove(node));
161             break;
162         case Node::ATTRIBUTE_NODE:
163             wrapper = createWrapper<Attr>(globalObject, WTFMove(node));
164             break;
165         case Node::TEXT_NODE:
166             wrapper = createWrapper<Text>(globalObject, WTFMove(node));
167             break;
168         case Node::CDATA_SECTION_NODE:
169             wrapper = createWrapper<CDATASection>(globalObject, WTFMove(node));
170             break;
171         case Node::PROCESSING_INSTRUCTION_NODE:
172             wrapper = createWrapper<ProcessingInstruction>(globalObject, WTFMove(node));
173             break;
174         case Node::COMMENT_NODE:
175             wrapper = createWrapper<Comment>(globalObject, WTFMove(node));
176             break;
177         case Node::DOCUMENT_NODE:
178             // we don't want to cache the document itself in the per-document dictionary
179             return toJS(lexicalGlobalObject, globalObject, downcast<Document>(node.get()));
180         case Node::DOCUMENT_TYPE_NODE:
181             wrapper = createWrapper<DocumentType>(globalObject, WTFMove(node));
182             break;
183         case Node::DOCUMENT_FRAGMENT_NODE:
184             if (node->isShadowRoot())
185                 wrapper = createWrapper<ShadowRoot>(globalObject, WTFMove(node));
186             else
187                 wrapper = createWrapper<DocumentFragment>(globalObject, WTFMove(node));
188             break;
189         default:
190             wrapper = createWrapper<Node>(globalObject, WTFMove(node));
191     }
192 
193     return wrapper;
194 }
195 
196 JSValue createWrapper(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, Ref<Node>&& node)
197 {
198     return createWrapperInline(lexicalGlobalObject, globalObject, WTFMove(node));
199 }

201 JSValue toJSNewlyCreated(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, Ref<Node>&& node)
202 {
203     return createWrapperInline(lexicalGlobalObject, globalObject, WTFMove(node));
204 }

206 JSC::JSObject* getOutOfLineCachedWrapper(JSDOMGlobalObject* globalObject, Node& node)
207 {
208     ASSERT(!globalObject->world().isNormal());
209     return globalObject->world().wrappers().get(&node);
210 }

212 void willCreatePossiblyOrphanedTreeByRemovalSlowCase(Node* root)
213 {
214     JSC::JSGlobalObject* lexicalGlobalObject = mainWorldExecState(root->document().frame());
215     if (!lexicalGlobalObject)
216         return;
217 
218     JSLockHolder lock(lexicalGlobalObject);
219     toJS(lexicalGlobalObject, static_cast<JSDOMGlobalObject*>(lexicalGlobalObject), *root);
220 }
