 26 #pragma once
 27 
 28 #include "TextGranularity.h"
 29 #include "VisiblePosition.h"
 30 
 31 namespace WebCore {
 32 
 33 class Position;
 34 class StaticRange;
 35 
 36 const EAffinity SEL_DEFAULT_AFFINITY = DOWNSTREAM;
 37 enum SelectionDirection : uint8_t { DirectionForward, DirectionBackward, DirectionRight, DirectionLeft };
 38 
 39 class VisibleSelection {
 40 public:
 41     enum SelectionType { NoSelection, CaretSelection, RangeSelection };
 42 
 43     WEBCORE_EXPORT VisibleSelection();
 44 
 45     VisibleSelection(const Position&, EAffinity, bool isDirectional = false);
 46     VisibleSelection(const Position&, const Position&, EAffinity = SEL_DEFAULT_AFFINITY, bool isDirectional = false);
 47 
 48     WEBCORE_EXPORT VisibleSelection(const Range&, EAffinity = SEL_DEFAULT_AFFINITY, bool isDirectional = false);
 49     WEBCORE_EXPORT VisibleSelection(const StaticRange&, EAffinity = SEL_DEFAULT_AFFINITY, bool isDirectional = false);
 50 
 51     WEBCORE_EXPORT VisibleSelection(const VisiblePosition&, bool isDirectional = false);
 52     WEBCORE_EXPORT VisibleSelection(const VisiblePosition&, const VisiblePosition&, bool isDirectional = false);
 53 
 54     WEBCORE_EXPORT static VisibleSelection selectionFromContentsOfNode(Node*);
 55 
 56     SelectionType selectionType() const { return m_selectionType; }
 57 
 58     void setAffinity(EAffinity affinity) { m_affinity = affinity; }
 59     EAffinity affinity() const { return m_affinity; }
 60 
 61     void setBase(const Position&);
 62     void setBase(const VisiblePosition&);
 63     void setExtent(const Position&);
 64     void setExtent(const VisiblePosition&);
 65 
 66     Position base() const { return m_base; }
 67     Position extent() const { return m_extent; }
 68     Position start() const { return m_start; }
 69     Position end() const { return m_end; }
 70 
 71     VisiblePosition visibleStart() const { return VisiblePosition(m_start, isRange() ? DOWNSTREAM : affinity()); }
 72     VisiblePosition visibleEnd() const { return VisiblePosition(m_end, isRange() ? UPSTREAM : affinity()); }
 73     VisiblePosition visibleBase() const { return VisiblePosition(m_base, isRange() ? (isBaseFirst() ? UPSTREAM : DOWNSTREAM) : affinity()); }
 74     VisiblePosition visibleExtent() const { return VisiblePosition(m_extent, isRange() ? (isBaseFirst() ? DOWNSTREAM : UPSTREAM) : affinity()); }
 75 
 76     bool isNone() const { return selectionType() == NoSelection; }
 77     bool isCaret() const { return selectionType() == CaretSelection; }
 78     bool isRange() const { return selectionType() == RangeSelection; }
 79     bool isCaretOrRange() const { return selectionType() != NoSelection; }
 80     bool isNonOrphanedRange() const { return isRange() && !start().isOrphan() && !end().isOrphan(); }
 81     bool isNoneOrOrphaned() const { return isNone() || start().isOrphan() || end().isOrphan(); }
 82 
 83     bool isBaseFirst() const { return m_baseIsFirst; }
 84     bool isDirectional() const { return m_isDirectional; }
 85     void setIsDirectional(bool isDirectional) { m_isDirectional = isDirectional; }
 86 
 87     WEBCORE_EXPORT bool isAll(EditingBoundaryCrossingRule) const;
 88 
 89     void appendTrailingWhitespace();
 90 
 91     WEBCORE_EXPORT bool expandUsingGranularity(TextGranularity granularity);
 92 
 93     // We don't yet support multi-range selections, so we only ever have one range to return.
 94     WEBCORE_EXPORT RefPtr<Range> firstRange() const;
 95 
 96     // FIXME: Most callers probably don't want this function, but are using it
 97     // for historical reasons.  toNormalizedRange contracts the range around
 98     // text, and moves the caret upstream before returning the range.
 99     WEBCORE_EXPORT RefPtr<Range> toNormalizedRange() const;
100 
101     WEBCORE_EXPORT Element* rootEditableElement() const;
102     WEBCORE_EXPORT bool isContentEditable() const;
103     bool hasEditableStyle() const;
104     WEBCORE_EXPORT bool isContentRichlyEditable() const;
105     // Returns a shadow tree node for legacy shadow trees, a child of the
106     // ShadowRoot node for new shadow trees, or 0 for non-shadow trees.
107     Node* nonBoundaryShadowTreeRootNode() const;
108 
109     WEBCORE_EXPORT bool isInPasswordField() const;
110 
111     WEBCORE_EXPORT static Position adjustPositionForEnd(const Position& currentPosition, Node* startContainerNode);
112     WEBCORE_EXPORT static Position adjustPositionForStart(const Position& currentPosition, Node* startContainerNode);
113 
114 #if ENABLE(TREE_DEBUGGING)
115     void debugPosition() const;
116     void formatForDebugger(char* buffer, unsigned length) const;
117     void showTreeForThis() const;
118 #endif
119 
120     void setWithoutValidation(const Position&, const Position&);
121 
122 private:
123     void validate(TextGranularity = CharacterGranularity);
124 
125     // Support methods for validate()
126     void setBaseAndExtentToDeepEquivalents();
127     void setStartAndEndFromBaseAndExtentRespectingGranularity(TextGranularity);
128     void adjustSelectionToAvoidCrossingShadowBoundaries();
129     void adjustSelectionToAvoidCrossingEditingBoundaries();
130     void updateSelectionType();
131 
132     // We need to store these as Positions because VisibleSelection is
133     // used to store values in editing commands for use when
134     // undoing the command. We need to be able to create a selection that, while currently
135     // invalid, will be valid once the changes are undone.
136 
137     Position m_base;   // Where the first click happened
138     Position m_extent; // Where the end click happened
139     Position m_start;  // Leftmost position when expanded to respect granularity
140     Position m_end;    // Rightmost position when expanded to respect granularity
141 
142     EAffinity m_affinity;           // the upstream/downstream affinity of the caret
143 
144     // these are cached, can be recalculated by validate()
145     SelectionType m_selectionType; // None, Caret, Range
146     bool m_baseIsFirst : 1; // True if base is before the extent
147     bool m_isDirectional : 1; // Non-directional ignores m_baseIsFirst and selection always extends on shift + arrow key.
148 };
149 
150 inline bool operator==(const VisibleSelection& a, const VisibleSelection& b)
151 {
152     return a.start() == b.start() && a.end() == b.end() && a.affinity() == b.affinity() && a.isBaseFirst() == b.isBaseFirst()
153         && a.isDirectional() == b.isDirectional();
154 }
155 
156 inline bool operator!=(const VisibleSelection& a, const VisibleSelection& b)
157 {
158     return !(a == b);
159 }
160 
161 WEBCORE_EXPORT WTF::TextStream& operator<<(WTF::TextStream&, const VisibleSelection&);
162 
163 } // namespace WebCore
