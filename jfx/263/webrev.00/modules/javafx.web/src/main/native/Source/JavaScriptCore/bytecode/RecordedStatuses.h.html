33 namespace JSC {
34 
35 struct RecordedStatuses {
36     RecordedStatuses() { }
37 
38     RecordedStatuses& operator=(const RecordedStatuses& other) = delete;
39 
40     RecordedStatuses& operator=(RecordedStatuses&& other);
41 
42     RecordedStatuses(const RecordedStatuses& other) = delete;
43 
44     RecordedStatuses(RecordedStatuses&& other);
45 
46     CallLinkStatus* addCallLinkStatus(const CodeOrigin&, const CallLinkStatus&);
47     GetByStatus* addGetByStatus(const CodeOrigin&, const GetByStatus&);
48     PutByIdStatus* addPutByIdStatus(const CodeOrigin&, const PutByIdStatus&);
49     InByIdStatus* addInByIdStatus(const CodeOrigin&, const InByIdStatus&);
50 
51     void visitAggregate(SlotVisitor&);
52     void markIfCheap(SlotVisitor&);
53 
54     void finalizeWithoutDeleting(VM&);
55     void finalize(VM&);
56 
57     void shrinkToFit();
58 
59     template<typename Func>
60     void forEachVector(const Func& func)
61     {
62         func(calls);
63         func(gets);
64         func(puts);
65         func(ins);
66     }
67 
68     Vector<std::pair<CodeOrigin, std::unique_ptr<CallLinkStatus>>> calls;
69     Vector<std::pair<CodeOrigin, std::unique_ptr<GetByStatus>>> gets;
70     Vector<std::pair<CodeOrigin, std::unique_ptr<PutByIdStatus>>> puts;
71     Vector<std::pair<CodeOrigin, std::unique_ptr<InByIdStatus>>> ins;
72 };
73 
74 } // namespace JSC
