188 function(add_library target type)
189     _add_library(${ARGV})
190     if ("${type}" STREQUAL "SHARED")
191         sign(${target})
192     endif ()
193 endfunction()

195 macro(WEBKIT_EXECUTABLE _target)
196     _WEBKIT_EXECUTABLE(${_target})
197     playstation_setup_libc(${_target})
198     playstation_setup_fp(${_target})
199     if (NOT ${_target} MATCHES "^LLInt")
200         sign(${_target})
201     endif ()
202     if (PLAYSTATION_${_target}_WRAP)
203         foreach (WRAP ${PLAYSTATION_${_target}_WRAP})
204             target_link_options(${_target} PRIVATE -Wl,--wrap=${WRAP})
205         endforeach ()
206     endif ()
207 endmacro()
