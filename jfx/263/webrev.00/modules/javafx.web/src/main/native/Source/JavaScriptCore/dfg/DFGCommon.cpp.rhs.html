 51 bool stringLessThan(StringImpl& a, StringImpl& b)
 52 {
 53     unsigned minLength = std::min(a.length(), b.length());
 54     for (unsigned i = 0; i < minLength; ++i) {
 55         if (a[i] == b[i])
 56             continue;
 57         return a[i] < b[i];
 58     }
 59     return a.length() < b.length();
 60 }

 70     switch (state) {

 82 }
