26 enum AnimationPlayState {
27     "idle",
28     "running",
29     "paused",
30     "finished"
31 };
32 
33 enum AnimationReplaceState {
34     "active",
35     "removed",
36     "persisted"
37 };
38 
39 [
40     ActiveDOMObject,
41     EnabledAtRuntime=WebAnimations,
42     InterfaceName=Animation,
43     CustomConstructor(),
44     CustomToJSObject
45 ] interface WebAnimation : EventTarget {
46     attribute DOMString id;
47     attribute AnimationEffect? effect;
48     [RuntimeConditionallyReadWrite=WebAnimationsMutableTimelines] attribute AnimationTimeline? timeline;
49     attribute double? startTime;
50     [MayThrowException, ImplementedAs=bindingsCurrentTime] attribute double? currentTime;
51     attribute double playbackRate;
52     [ImplementedAs=bindingsPlayState] readonly attribute AnimationPlayState playState;
53     [ImplementedAs=bindingsReplaceState] readonly attribute AnimationReplaceState replaceState;
54     [ImplementedAs=bindingsPending] readonly attribute boolean pending;
55     attribute EventHandler onfinish;
56     attribute EventHandler oncancel;
57     attribute EventHandler onremove;
58     [ImplementedAs=bindingsReady] readonly attribute Promise<WebAnimation> ready;
59     [ImplementedAs=bindingsFinished] readonly attribute Promise<WebAnimation> finished;
60     void cancel();
61     [MayThrowException] void finish();
62     [MayThrowException, ImplementedAs=bindingsPlay] void play();
63     [MayThrowException, ImplementedAs=bindingsPause] void pause();
64     void updatePlaybackRate(double playbackRate);
65     [MayThrowException] void reverse();
66     void persist();
67     [MayThrowException] void commitStyles();
68 };
