26 #pragma once
27 
28 #include "BMalloced.h"
29 
30 namespace bmalloc {
31 
32 class IsoTLS;
33 class IsoTLSLayout;
34 
35 class BEXPORT IsoTLSEntry {
36     MAKE_BMALLOCED;
37 public:
38     IsoTLSEntry(size_t alignment, size_t size);
39     virtual ~IsoTLSEntry();
40 
41     size_t offset() const { return m_offset; }
42     size_t alignment() const { return m_alignment; }
43     size_t size() const { return m_size; }
44     size_t extent() const { return m_offset + m_size; }
45 
46     virtual void construct(void* entry) = 0;
47     virtual void move(void* src, void* dst) = 0;
48     virtual void destruct(void* entry) = 0;
49     virtual void scavenge(void* entry) = 0;
50 
51     template<typename Func>
52     void walkUpToInclusive(IsoTLSEntry*, const Func&);
53 
54 private:
55     friend class IsoTLS;
56     friend class IsoTLSLayout;
57 
58     IsoTLSEntry* m_next { nullptr };
59 
60     size_t m_offset; // Computed in constructor.
61     size_t m_alignment;
62     size_t m_size;
63 };
64 
65 template<typename EntryType>
66 class DefaultIsoTLSEntry : public IsoTLSEntry {
67 public:
68     DefaultIsoTLSEntry();
69     ~DefaultIsoTLSEntry();
70 
71 protected:
72     // This clones src onto dst and then destructs src. Therefore, entry destructors cannot do
73     // scavenging.
74     void move(void* src, void* dst) override;
75 
76     // Likewise, this is separate from scavenging. When the TLS is shutting down, we will be asked to
77     // scavenge and then we will be asked to destruct.
78     void destruct(void* entry) override;
79 
80     void scavenge(void* entry) override;
81 };

83 } // namespace bmalloc
