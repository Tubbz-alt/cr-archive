 33 namespace WTF {
 34 
 35 static void retainSendRight(mach_port_t port)
 36 {
 37     if (port == MACH_PORT_NULL)
 38         return;
 39 
 40     auto kr = KERN_SUCCESS;
 41     if (port != MACH_PORT_DEAD)
 42         kr = mach_port_mod_refs(mach_task_self(), port, MACH_PORT_RIGHT_SEND, 1);
 43 
 44     if (kr == KERN_INVALID_RIGHT || port == MACH_PORT_DEAD)
 45         kr = mach_port_mod_refs(mach_task_self(), port, MACH_PORT_RIGHT_DEAD_NAME, 1);
 46 
 47     if (kr != KERN_SUCCESS) {
 48         LOG_ERROR("mach_port_mod_refs error for port %d: %s (%x)", port, mach_error_string(kr), kr);
 49         if (kr == KERN_INVALID_RIGHT)
 50             CRASH();
 51     }
 52 }
 53 
 54 static void releaseSendRight(mach_port_t port)
 55 {
 56     if (port == MACH_PORT_NULL)
 57         return;
 58 
 59     deallocateSendRightSafely(port);
 60 }
 61 
 62 void deallocateSendRightSafely(mach_port_t port)
 63 {
 64     auto kr = mach_port_deallocate(mach_task_self(), port);
 65     if (kr == KERN_SUCCESS)
 66         return;
 67 
 68     LOG_ERROR("mach_port_deallocate error for port %d: %s (%#x)", port, mach_error_string(kr), kr);
 69     if (kr == KERN_INVALID_RIGHT)
 70         CRASH();
 71 }
 72 
 73 MachSendRight MachSendRight::adopt(mach_port_t port)
 74 {
 75     return MachSendRight(port);
 76 }
 77 
 78 MachSendRight MachSendRight::create(mach_port_t port)
 79 {
 80     retainSendRight(port);
 81 
 82     return adopt(port);
 83 }
 84 
 85 MachSendRight::MachSendRight(mach_port_t port)
 86     : m_port(port)
 87 {
 88 }
 89 
 90 MachSendRight::MachSendRight(MachSendRight&& other)
 91     : m_port(other.leakSendRight())
 92 {
 93 }
 94 
 95 MachSendRight::~MachSendRight()
 96 {
 97     releaseSendRight(m_port);
 98 }
 99 
100 MachSendRight& MachSendRight::operator=(MachSendRight&& other)
101 {
102     if (this != &other) {
103         releaseSendRight(m_port);
104         m_port = other.leakSendRight();
105     }
106 
107     return *this;
108 }
109 
110 MachSendRight MachSendRight::copySendRight() const
111 {
112     return create(m_port);
113 }
114 
115 mach_port_t MachSendRight::leakSendRight()
116 {
117     return std::exchange(m_port, MACH_PORT_NULL);
118 }
