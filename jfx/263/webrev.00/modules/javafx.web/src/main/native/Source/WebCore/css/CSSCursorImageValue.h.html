43 class CSSCursorImageValue final : public CSSValue {
44 public:
45     static Ref<CSSCursorImageValue> create(Ref<CSSValue>&& imageValue, bool hasHotSpot, const IntPoint& hotSpot, LoadedFromOpaqueSource loadedFromOpaqueSource)
46     {
47         return adoptRef(*new CSSCursorImageValue(WTFMove(imageValue), hasHotSpot, hotSpot, loadedFromOpaqueSource));
48     }
49 
50     ~CSSCursorImageValue();
51 
52     bool hasHotSpot() const { return m_hasHotSpot; }
53 
54     IntPoint hotSpot() const
55     {
56         if (m_hasHotSpot)
57             return m_hotSpot;
58         return IntPoint(-1, -1);
59     }
60 
61     const URL& imageURL() const { return m_originalURL; }
62 
63     String customCSSText() const;
64 
65     ImageWithScale selectBestFitImage(const Document&);
66 
67     void removeReferencedElement(SVGElement*);
68 
69     bool equals(const CSSCursorImageValue&) const;
70 
71     void cursorElementRemoved(SVGCursorElement&);
72     void cursorElementChanged(SVGCursorElement&);
73 
74 private:
75     CSSCursorImageValue(Ref<CSSValue>&& imageValue, bool hasHotSpot, const IntPoint& hotSpot, LoadedFromOpaqueSource);
76 
77     SVGCursorElement* updateCursorElement(const Document&);
78 
79     URL m_originalURL;
80     Ref<CSSValue> m_imageValue;
81 
82     bool m_hasHotSpot;
83     IntPoint m_hotSpot;
84     HashSet<SVGCursorElement*> m_cursorElements;
85     LoadedFromOpaqueSource m_loadedFromOpaqueSource { LoadedFromOpaqueSource::No };
86 };
