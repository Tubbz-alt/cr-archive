26 #ifndef _PLAYER_EVENT_DISPATCHER_H_
27 #define _PLAYER_EVENT_DISPATCHER_H_
28 
29 #include <list>
30 #include <string>
31 #include <stdint.h>
32 
33 using namespace std;
34 
35 class CVideoFrame;
36 class CAudioTrack;
37 class CVideoTrack;
38 
39 class CPlayerEventDispatcher
40 {
41 public:
42     virtual ~CPlayerEventDispatcher() {}
43 
44     virtual bool SendPlayerMediaErrorEvent(int errorCode) = 0;
45     virtual bool SendPlayerHaltEvent(const char* message, double msgTime) = 0;
46     virtual bool SendPlayerStateEvent(int newState, double presentTime) = 0;
47     virtual bool SendNewFrameEvent(CVideoFrame* pVideoFrame) = 0;
48     virtual bool SendFrameSizeChangedEvent(int width, int height) = 0;
49     virtual bool SendAudioTrackEvent(CAudioTrack* pTrack) = 0;
50     virtual bool SendVideoTrackEvent(CVideoTrack* pTrack) = 0;
51     virtual bool SendMarkerEvent(string name, double time) = 0;
52     virtual bool SendBufferProgressEvent(double clipDuration, int64_t start, int64_t stop, int64_t position) = 0;
53     virtual bool SendDurationUpdateEvent(double time) = 0;
54     virtual bool SendAudioSpectrumEvent(double time, double duration) = 0;
55     virtual void Warning(int warningCode, const char* warningMessage) = 0;
56 };
57 #endif // _PLAYER_EVENT_DISPATCHER_H_
