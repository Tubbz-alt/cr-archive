26 package com.sun.media.jfxmedia.events;

28 import com.sun.media.jfxmedia.effects.AudioSpectrum;
29 
30 public class AudioSpectrumEvent extends PlayerEvent {
31     private AudioSpectrum source;
32     private double        timestamp;
33     private double        duration;

34 
35     public AudioSpectrumEvent(AudioSpectrum source, double timestamp, double duration) {










36         this.source = source;
37         this.timestamp = timestamp;
38         this.duration = duration;

39     }
40 
41     public final AudioSpectrum getSource() {
42         return source;
43     }
44 




45     public final double getTimestamp() {
46         return timestamp;
47     }
48 
49     public final double getDuration() {
50         return duration;
51     }




52 }
