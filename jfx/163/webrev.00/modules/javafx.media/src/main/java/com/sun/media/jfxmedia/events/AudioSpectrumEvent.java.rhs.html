26 package com.sun.media.jfxmedia.events;
27 
28 import com.sun.media.jfxmedia.effects.AudioSpectrum;
29 
30 public class AudioSpectrumEvent extends PlayerEvent {
31     private AudioSpectrum source;
32     private double        timestamp;
33     private double        duration;
34     private boolean       queryTimestamp;
35 
36     /*
37      * Value of timestamp will be ignored if queryTimestamp is set true and
38      * timestamp will be requested from EventQueueThread when spectrum event is
39      * received instead. We do not use -1.0 (GST_CLOCK_TIME_NONE), since
40      * GStreamer might send us such events in case if something fails, so we using
41      * queryTimestamp to know for sure that we need to ask for timestamp from
42      * event queue. Note: Only OSX platfrom sets it true. GStreamer platfrom
43      * should not use it unless such usage is tested.
44      */
45     public AudioSpectrumEvent(AudioSpectrum source, double timestamp,
46                               double duration, boolean queryTimestamp) {
47         this.source = source;
48         this.timestamp = timestamp;
49         this.duration = duration;
50         this.queryTimestamp = queryTimestamp;
51     }
52 
53     public final AudioSpectrum getSource() {
54         return source;
55     }
56 
57     public final void setTimestamp(double timestamp) {
58         this.timestamp = timestamp;
59     }
60 
61     public final double getTimestamp() {
62         return timestamp;
63     }
64 
65     public final double getDuration() {
66         return duration;
67     }
68 
69     public final boolean queryTimestamp() {
70         return queryTimestamp;
71     }
72 }
