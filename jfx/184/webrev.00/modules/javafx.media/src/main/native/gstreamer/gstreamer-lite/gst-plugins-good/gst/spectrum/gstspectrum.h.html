 29 G_BEGIN_DECLS
 30 
 31 #define GST_TYPE_SPECTRUM            (gst_spectrum_get_type())
 32 #define GST_SPECTRUM(obj)            (G_TYPE_CHECK_INSTANCE_CAST((obj),GST_TYPE_SPECTRUM,GstSpectrum))
 33 #define GST_IS_SPECTRUM(obj)         (G_TYPE_CHECK_INSTANCE_TYPE((obj),GST_TYPE_SPECTRUM))
 34 #define GST_SPECTRUM_CLASS(klass)    (G_TYPE_CHECK_CLASS_CAST((klass), GST_TYPE_SPECTRUM,GstSpectrumClass))
 35 #define GST_IS_SPECTRUM_CLASS(klass) (G_TYPE_CHECK_CLASS_TYPE((klass), GST_TYPE_SPECTRUM))
 36 typedef struct _GstSpectrum GstSpectrum;
 37 typedef struct _GstSpectrumClass GstSpectrumClass;
 38 typedef struct _GstSpectrumChannel GstSpectrumChannel;
 39 
 40 typedef void (*GstSpectrumInputData)(const guint8 * in, gfloat * out,
 41     guint len, guint channels, gfloat max_value, guint op, guint nfft);
 42 
 43 #if defined (GSTREAMER_LITE) && defined (OSX)
 44 // Used to overwrite post_message callback to get spectrum messages in OSXPlatform.
 45 // We cannot use GST_ELEMENT_GET_CLASS(spectrum)->post_message, since it will
 46 // change callback for all instances of spectrum elements and it will conflict
 47 // with GStreamer platform.
 48 typedef gboolean (*PostMessageCallbackProc)(GstElement * element,
 49                                             GstMessage * message);
 50 #endif // GSTREAMER_LITE and OSX
 51 
 52 struct _GstSpectrumChannel
 53 {
 54   gfloat *input;
 55   gfloat *input_tmp;
 56   GstFFTF32Complex *freqdata;
 57   gfloat *spect_magnitude;      /* accumulated mangitude and phase */
 58   gfloat *spect_phase;          /* will be scaled by num_fft before sending */
 59   GstFFTF32 *fft_ctx;
 60 };
 61 
 62 struct _GstSpectrum
 63 {
 64   GstAudioFilter parent;
 65 
 66   /* properties */
 67   gboolean post_messages;       /* whether or not to post messages */
 68   gboolean message_magnitude;
 69   gboolean message_phase;
 70   guint64 interval;             /* how many nanoseconds between emits */
 71   guint64 frames_per_interval;  /* how many frames per interval */
 72   guint64 frames_todo;
 73   guint bands;                  /* number of spectrum bands */
 74   gint threshold;               /* energy level treshold */
 75   gboolean multi_channel;       /* send separate channel results */
 76 
 77   guint64 num_frames;           /* frame count (1 sample per channel)
 78                                  * since last emit */
 79   guint64 num_fft;              /* number of FFTs since last emit */
 80   GstClockTime message_ts;      /* starttime for next message */
 81 
 82   /* <private> */
 83   GstSpectrumChannel *channel_data;
 84   guint num_channels;
 85 
 86   guint input_pos;
 87   guint64 error_per_interval;
 88   guint64 accumulated_error;
 89 
 90   GMutex lock;
 91 
 92   GstSpectrumInputData input_data;
 93 
 94 #if defined (GSTREAMER_LITE) && defined (OSX)
 95   guint bps_user; // User provided values to avoid more complex spectrum initialization
 96   guint bpf_user;
 97   void *user_data;
 98   PostMessageCallbackProc post_message_callback;
 99 #endif // GSTREAMER_LITE and OSX
100 };
101 
102 struct _GstSpectrumClass
103 {
104   GstAudioFilterClass parent_class;
105 };
