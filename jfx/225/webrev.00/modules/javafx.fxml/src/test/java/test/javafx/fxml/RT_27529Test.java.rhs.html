26 package test.javafx.fxml;
27 
28 import java.io.IOException;
29 import java.net.URL;
30 import java.util.Arrays;
31 import java.util.ResourceBundle;
32 import javafx.fxml.FXMLLoader;
33 import org.junit.Test;
34 
35 import static org.junit.Assert.*;
36 
37 public class RT_27529Test {
38 
39     @Test
40     public void testListAndArrayWithResources() throws IOException {
41         FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource("rt_27529_1.fxml"),
42             ResourceBundle.getBundle("test/javafx/fxml/rt_27529"));
43 
44         Widget widget = (Widget)fxmlLoader.load();
45         assertEquals(Arrays.asList(new String[]{"a", "b", "c"}), widget.getStyles());
46         assertTrue(Arrays.equals(new String[]{"a", "b", "c"}, widget.getNames()));
47         assertTrue(Arrays.equals(new float[] {1.0f, 2.0f, 3.0f}, widget.getRatios()));
48     }
49 
50     @Test
51     public void testListAndArrayWithEscapes() throws IOException {
52         System.err.println("Below warnings about - deprecated escape sequence - are expected from this test.");
53         FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource("rt_27529_2.fxml"),
54             ResourceBundle.getBundle("test/javafx/fxml/rt_27529"));
55         fxmlLoader.load();
56 
57         Widget widget = (Widget)fxmlLoader.getNamespace().get("widget1");
58         assertEquals(Arrays.asList(new String[]{"@a", "%b", "$c", "@c", "%d", "$e"}), widget.getStyles());
59         assertTrue(Arrays.equals(  new String[]{"@a", "%b", "$c", "@c", "%d", "$e"}, widget.getNames()));
60     }
61 
62     @Test
63     public void testListAndArrayWithRelativePath() throws IOException {
64         FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource("rt_27529_3.fxml"),
65             ResourceBundle.getBundle("test/javafx/fxml/rt_27529"));
66 
67         Widget widget = (Widget)fxmlLoader.load();
68         assertEquals(Arrays.asList(new String[]{
69             new URL(fxmlLoader.getLocation(), "a").toString(),
70             new URL(fxmlLoader.getLocation(), "b").toString(),
71             new URL(fxmlLoader.getLocation(), "c").toString()}), widget.getStyles());
72     }
73 
74     @Test
75     public void testListAndArrayWithReference() throws IOException {
76         FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource("rt_27529_4.fxml"),
77             ResourceBundle.getBundle("test/javafx/fxml/rt_27529"));
78 
79         fxmlLoader.load();
80         Widget widget = (Widget)fxmlLoader.getNamespace().get("widget1");
81         assertEquals(Arrays.asList(new String[]{"ABC", "ABC"}), widget.getStyles());
82         assertTrue(Arrays.equals(new String[]{"ABC", "ABC"}, widget.getNames()));
83         assertTrue(Arrays.equals(new float[] {1.0f, 1.0f}, widget.getRatios()));
84     }
85 }
