26 package javafx.scene.control.skin;
27 
28 import com.sun.javafx.scene.control.ContextMenuContent;
29 import javafx.scene.control.ContextMenu;
30 import javafx.scene.control.MenuButton;
31 import javafx.scene.control.Skin;
32 import javafx.scene.control.skin.MenuBarSkin;
33 import javafx.scene.control.skin.MenuButtonSkinBase;
34 import javafx.scene.layout.VBox;
35 
36 
37 /**
38  *
39  */
40 public class MenuBarSkinShim {
41 
42     // can only access the getNodeForMenu method in MenuBarSkin from this package.
43     public static MenuButton getNodeForMenu(MenuBarSkin skin, int i) {
44         return skin.getNodeForMenu(i);
45     }
46 
47     public static Skin getPopupSkin(MenuButton mb) {
48         return ((MenuButtonSkinBase)mb.getSkin()).popup.getSkin();
49     }
50 
51     public static ContextMenuContent getMenuContent(MenuButton mb) {
52         ContextMenuContent cmc = (ContextMenuContent)getPopupSkin(mb).getNode();
53         return cmc;
54     }
55 
56     public static int getFocusedMenuIndex(MenuBarSkin skin) {
57         return skin.getFocusedMenuIndex();
58     }
59 
60     public static void setFocusedMenuIndex(MenuBarSkin skin, int index) {
61         skin.setFocusedMenuIndex(index);
62     }
63 
64 }
