26 package javafx.scene.control.skin;
27 
28 import java.util.List;
29 
30 import javafx.scene.control.ContextMenu;
31 import javafx.scene.control.TabPane;
32 import javafx.scene.layout.StackPane;
33 import javafx.scene.Node;
34 
35 public class TabPaneSkinShim {
36 
37     public static ContextMenu getTabsMenu(TabPaneSkin tpSkin) {
38         return tpSkin.test_getTabsMenu();
39     }
40 
41     public static void disableAnimations(TabPaneSkin tpSkin) {
42         tpSkin.test_disableAnimations();
43     }
44 
45     public static List<Node> getTabHeaders(TabPane tabPane) {
46         StackPane headersRegion = (StackPane) tabPane.lookup(".headers-region");
47         return headersRegion.getChildren();
48     }
49 }
