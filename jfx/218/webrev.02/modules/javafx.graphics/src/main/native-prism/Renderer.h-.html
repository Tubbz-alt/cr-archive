 26 #ifndef RENDERER_H
 27 #define RENDERER_H
 28 
 29 #include "PathConsumer.h"
 30 #include "AlphaConsumer.h"
 31 #include "Curve.h"
 32 
 33 #ifdef __cplusplus
 34 extern "C" {
 35 #endif
 36 
 37 #define INIT_CROSSINGS_SIZE   10
 38 typedef struct {
 39     jint *crossings;
 40     jint crossingsSIZE;
 41     jint *edgePtrs;
 42     jint edgePtrsSIZE;
 43     jint edgeCount;
 44 
 45     // crossing bounds. The bounds are not necessarily tight (the scan line
 46     // at minY, for example, might have no crossings). The x bounds will
 47     // be accumulated as crossings are computed.
 48     jint nextY;
 49 } ScanlineIterator;
 50 
 51 // common to all types of input path segments.
 52 #define YMAX         0
 53 #define CURX         1
 54 
 55 // NEXT and OR are meant to be indices into "int" fields, but arrays must
 56 // be homogenous, so every field is a float. However floats can represent
 57 // exactly up to 26 bit ints, so we're ok.
 58 #define OR           2
 59 #define SLOPE        3
 60 #define NEXT         4
 61 #define SIZEOF_EDGE  5
 62 
 63 #define WIND_EVEN_ODD   0
 64 #define WIND_NON_ZERO   1
 65 
 66 #define DEC_BND   1.0f
 67 #define INC_BND   0.4f
 68 
 69 typedef struct {
 70     PathConsumer consumer;
 71 
 72     ScanlineIterator iterator;
 73 
 74     jint sampleRowMin;
 75     jint sampleRowMax;
 76     jfloat edgeMinX;
 77     jfloat edgeMaxX;
 78 
 79     jfloat *edges;
 80     jint edgesSIZE;
 81     jint *edgeBuckets;
 82     jint edgeBucketsSIZE;
 83     jint numEdges;
 84 
 85     // Bounds of the drawing region, at subpixel precision.
 86     jint boundsMinX, boundsMinY, boundsMaxX, boundsMaxY;
 87 
 88     // Current winding rule
 89     jint windingRule;
 90 
 91     // Current drawing position, i.e., final point of last segment
 92     jfloat x0, y0;
 93 
 94     // Position of most recent 'moveTo' command
 95     jfloat pix_sx0, pix_sy0;
 96 
 97     Curve c;
 98 } Renderer;
 99 
100 extern void Renderer_setup(jint subpixelLgPositionsX, jint subpixelLgPositionsY);
101 
102 extern void Renderer_init(Renderer *pRenderer);
103 
104 extern void Renderer_reset(Renderer *pRenderer,
105                            jint pix_boundsX, jint pix_boundsY,
106                            jint pix_boundsWidth, jint pix_boundsHeight,
107                            jint windingRule);
108 
109 extern void Renderer_destroy(Renderer *pRenderer);
110 
111 extern void Renderer_getOutputBounds(Renderer *pRenderer, jint bounds[]);
112 
113 extern void Renderer_produceAlphas(Renderer *pRenderer, AlphaConsumer *pAC);
114 
115 #ifdef __cplusplus
116 }
117 #endif
118 
119 #endif /* RENDERER_H */
