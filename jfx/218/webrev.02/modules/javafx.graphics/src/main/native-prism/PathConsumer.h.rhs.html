26 #ifndef PATHCONSUMER_H
27 #define PATHCONSUMER_H
28 
29 #ifdef __cplusplus
30 extern "C" {
31 #endif
32 
33 #include <jni.h>
34 
35 // Pisces error codes
36 #define ERROR_NONE   0
37 #define ERROR_OOM    1
38 #define ERROR_AIOOBE 2
39 
40 typedef struct _PathConsumer PathConsumer;
41 
42 typedef jint MoveToFunc(PathConsumer *pConsumer,
43                         jfloat x0, jfloat y0);
44 typedef jint LineToFunc(PathConsumer *pConsumer,
45                         jfloat x1, jfloat y1);
46 typedef jint QuadToFunc(PathConsumer *pConsumer,
47                         jfloat xc, jfloat yc,
48                         jfloat x1, jfloat y1);
49 typedef jint CurveToFunc(PathConsumer *pConsumer,
50                          jfloat xc0, jfloat yc0,
51                          jfloat xc1, jfloat yc1,
52                          jfloat x1, jfloat y1);
53 typedef jint ClosePathFunc(PathConsumer *pConsumer);
54 typedef jint PathDoneFunc(PathConsumer *pConsumer);
55 
56 struct _PathConsumer {
57     MoveToFunc       *moveTo;
58     LineToFunc       *lineTo;
59     QuadToFunc       *quadTo;
60     CurveToFunc      *curveTo;
61     ClosePathFunc    *closePath;
62     PathDoneFunc     *pathDone;
63 };
64 
65 extern void PathConsumer_init(PathConsumer *pConsumer,
66                               MoveToFunc       *moveTo,
67                               LineToFunc       *lineTo,
68                               QuadToFunc       *quadTo,
69                               CurveToFunc      *curveTo,
70                               ClosePathFunc    *closePath,
71                               PathDoneFunc     *pathDone);
72 
73 #ifdef __cplusplus
74 }
75 #endif
76 
77 #endif /* PATHCONSUMER_H */
