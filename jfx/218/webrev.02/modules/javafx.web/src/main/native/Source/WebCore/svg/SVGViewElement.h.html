32 class SVGViewElement final : public SVGElement, public SVGFitToViewBox, public SVGZoomAndPan {

37     using SVGElement::ref;
38     using SVGElement::deref;

40     const SVGSVGElement* targetElement() const { return m_targetElement.get(); }

42     void resetTargetElement() { m_targetElement = nullptr; }
43 
44 private:
45     SVGViewElement(const QualifiedName&, Document&);
46 
47     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGViewElement, SVGElement, SVGFitToViewBox>;

49 
50     void parseAttribute(const QualifiedName&, const AtomString&) final;
51     void svgAttributeChanged(const QualifiedName&) override;
52 
53     bool rendererIsNeeded(const RenderStyle&) final { return false; }
54 
55     PropertyRegistry m_propertyRegistry { *this };
56     WeakPtr<SVGSVGElement> m_targetElement { nullptr };
57 };
