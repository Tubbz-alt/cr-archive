26 // NOTE: This type is specified as 'any' in the IndexedDB specification, but is always
27 // constrained to this union.
28 typedef (DOMString or sequence<DOMString>) IDBKeyPath;
29 
30 [
31     Conditional=INDEXED_DATABASE,
32     GenerateIsReachable=Impl,
33     JSCustomMarkFunction,
34     SkipVTableValidation,
35 ] interface IDBIndex {
36     attribute DOMString name;
37     readonly attribute IDBObjectStore objectStore;
38     readonly attribute IDBKeyPath? keyPath;
39     readonly attribute boolean multiEntry;
40     readonly attribute boolean unique;
41 
42     [CallWith=ExecState, MayThrowException] IDBRequest openCursor(optional IDBKeyRange? range = null, optional IDBCursorDirection direction = "next");
43     [CallWith=ExecState, MayThrowException] IDBRequest openCursor(any key, optional IDBCursorDirection direction = "next");
44 
45     [CallWith=ExecState, MayThrowException] IDBRequest openKeyCursor(optional IDBKeyRange? range = null, optional IDBCursorDirection direction = "next");
46     [CallWith=ExecState, MayThrowException] IDBRequest openKeyCursor(any key, optional IDBCursorDirection direction = "next");
47 
48     [CallWith=ExecState, MayThrowException] IDBRequest get(IDBKeyRange? key);
49     [CallWith=ExecState, MayThrowException] IDBRequest get(any key);
50     [CallWith=ExecState, MayThrowException] IDBRequest getKey(IDBKeyRange? key);
51     [CallWith=ExecState, MayThrowException] IDBRequest getKey(any key);
52     [CallWith=ExecState, MayThrowException] IDBRequest getAll(optional IDBKeyRange? range = null, optional [EnforceRange] unsigned long count);
53     [CallWith=ExecState, MayThrowException] IDBRequest getAll(any key, optional [EnforceRange] unsigned long count);
54     [CallWith=ExecState, MayThrowException] IDBRequest getAllKeys(optional IDBKeyRange? range = null, optional [EnforceRange] unsigned long count);
55     [CallWith=ExecState, MayThrowException] IDBRequest getAllKeys(any key, optional [EnforceRange] unsigned long count);
56     [CallWith=ExecState, MayThrowException] IDBRequest count(optional IDBKeyRange? range = null);
57     [CallWith=ExecState, MayThrowException] IDBRequest count(any key);
58 };
