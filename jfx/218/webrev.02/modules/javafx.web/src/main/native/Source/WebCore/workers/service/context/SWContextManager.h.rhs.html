 26 #pragma once

 28 #if ENABLE(SERVICE_WORKER)
 29 
 30 #include "ExceptionOr.h"
 31 #include "ServiceWorkerClientData.h"
 32 #include "ServiceWorkerClientQueryOptions.h"
 33 #include "ServiceWorkerIdentifier.h"
 34 #include "ServiceWorkerThreadProxy.h"
 35 #include <wtf/CompletionHandler.h>
 36 #include <wtf/HashMap.h>
 37 
 38 namespace WebCore {
 39 
 40 class SerializedScriptValue;
 41 class ServiceWorkerGlobalScope;
 42 
 43 class SWContextManager {
 44 public:
 45     WEBCORE_EXPORT static SWContextManager& singleton();
 46 
 47     class Connection {
 48         WTF_MAKE_FAST_ALLOCATED;
 49     public:
 50         virtual ~Connection() { }
 51 
 52         virtual void establishConnection(CompletionHandler<void()>&&) = 0;
 53         virtual void postMessageToServiceWorkerClient(const ServiceWorkerClientIdentifier& destinationIdentifier, const MessageWithMessagePorts&, ServiceWorkerIdentifier source, const String& sourceOrigin) = 0;
 54         virtual void serviceWorkerStarted(Optional<ServiceWorkerJobDataIdentifier>, ServiceWorkerIdentifier, bool doesHandleFetch) = 0;
 55         virtual void serviceWorkerFailedToStart(Optional<ServiceWorkerJobDataIdentifier>, ServiceWorkerIdentifier, const String& message) = 0;
 56         virtual void didFinishInstall(Optional<ServiceWorkerJobDataIdentifier>, ServiceWorkerIdentifier, bool wasSuccessful) = 0;
 57         virtual void didFinishActivation(ServiceWorkerIdentifier) = 0;
 58         virtual void setServiceWorkerHasPendingEvents(ServiceWorkerIdentifier, bool) = 0;
 59         virtual void workerTerminated(ServiceWorkerIdentifier) = 0;
 60         virtual void skipWaiting(ServiceWorkerIdentifier, CompletionHandler<void()>&&) = 0;
 61         virtual void setScriptResource(ServiceWorkerIdentifier, const URL&, const ServiceWorkerContextData::ImportedScript&) = 0;
 62 
 63         using FindClientByIdentifierCallback = CompletionHandler<void(ExceptionOr<Optional<ServiceWorkerClientData>>&&)>;
 64         virtual void findClientByIdentifier(ServiceWorkerIdentifier, ServiceWorkerClientIdentifier, FindClientByIdentifierCallback&&) = 0;
 65         virtual void matchAll(ServiceWorkerIdentifier, const ServiceWorkerClientQueryOptions&, ServiceWorkerClientsMatchAllCallback&&) = 0;
 66         virtual void claim(ServiceWorkerIdentifier, CompletionHandler<void()>&&) = 0;
 67 
 68         virtual void didFailHeartBeatCheck(ServiceWorkerIdentifier) = 0;
 69 
 70         virtual bool isThrottleable() const = 0;
 71 
 72         bool isClosed() const { return m_isClosed; }
 73         bool shouldUseShortTimeout() const { return m_shouldUseShortTimeout; }
 74 
 75     protected:
 76         void setAsClosed() { m_isClosed = true; }
 77         void setShouldUseShortTimeout(bool value) { m_shouldUseShortTimeout = value; }
 78 
 79     private:
 80         bool m_isClosed { false };
 81         bool m_shouldUseShortTimeout { false };
 82     };
 83 
 84     WEBCORE_EXPORT void setConnection(std::unique_ptr<Connection>&&);
 85     WEBCORE_EXPORT Connection* connection() const;
 86 
 87     WEBCORE_EXPORT void registerServiceWorkerThreadForInstall(Ref<ServiceWorkerThreadProxy>&&);
 88     WEBCORE_EXPORT ServiceWorkerThreadProxy* serviceWorkerThreadProxy(ServiceWorkerIdentifier) const;
 89     WEBCORE_EXPORT void postMessageToServiceWorker(ServiceWorkerIdentifier destination, MessageWithMessagePorts&&, ServiceWorkerOrClientData&& sourceData);
 90     WEBCORE_EXPORT void fireInstallEvent(ServiceWorkerIdentifier);
 91     WEBCORE_EXPORT void fireActivateEvent(ServiceWorkerIdentifier);
 92     WEBCORE_EXPORT void terminateWorker(ServiceWorkerIdentifier, Seconds timeout, Function<void()>&&);
 93 
 94     void forEachServiceWorkerThread(const WTF::Function<void(ServiceWorkerThreadProxy&)>&);
 95 
 96     WEBCORE_EXPORT bool postTaskToServiceWorker(ServiceWorkerIdentifier, WTF::Function<void(ServiceWorkerGlobalScope&)>&&);
 97 
 98     using ServiceWorkerCreationCallback = void(uint64_t);
 99     void setServiceWorkerCreationCallback(ServiceWorkerCreationCallback* callback) { m_serviceWorkerCreationCallback = callback; }
100 
101     ServiceWorkerThreadProxy* workerByID(ServiceWorkerIdentifier identifier) { return m_workerMap.get(identifier); }
102 
103     WEBCORE_EXPORT void stopAllServiceWorkers();
104 
105     static constexpr Seconds workerTerminationTimeout { 10_s };
106     static constexpr Seconds syncWorkerTerminationTimeout { 100_ms }; // Only used by layout tests.
107 
108 private:
109     SWContextManager() = default;
110 
111     void startedServiceWorker(Optional<ServiceWorkerJobDataIdentifier>, ServiceWorkerIdentifier, const String& exceptionMessage, bool doesHandleFetch);
112     NO_RETURN_DUE_TO_CRASH void serviceWorkerFailedToTerminate(ServiceWorkerIdentifier);
113 
114     void stopWorker(ServiceWorkerThreadProxy&, Seconds, Function<void()>&&);
115 
116     HashMap<ServiceWorkerIdentifier, Ref<ServiceWorkerThreadProxy>> m_workerMap;
117     std::unique_ptr<Connection> m_connection;
118     ServiceWorkerCreationCallback* m_serviceWorkerCreationCallback { nullptr };
119 
120     class ServiceWorkerTerminationRequest {
121         WTF_MAKE_FAST_ALLOCATED;
122     public:
123         ServiceWorkerTerminationRequest(SWContextManager&, ServiceWorkerIdentifier, Seconds timeout);
124 
125     private:
126         Timer m_timeoutTimer;
127     };
128     HashMap<ServiceWorkerIdentifier, std::unique_ptr<ServiceWorkerTerminationRequest>> m_pendingServiceWorkerTerminationRequests;
129 };
130 
131 } // namespace WebCore

133 #endif // ENABLE(SERVICE_WORKER)
