26 #pragma once
27 
28 #include "CodeBlock.h"
29 #include "Register.h"
30 #include "StackAlignment.h"
31 #include <wtf/ForbidHeapAllocation.h>
32 
33 namespace JSC {
34 
35 struct JS_EXPORT_PRIVATE ProtoCallFrame {
36     WTF_FORBID_HEAP_ALLOCATION;
37 public:
38     // CodeBlock, Callee, ArgumentCount, and |this|.
39     static constexpr unsigned numberOfRegisters { 4 };
40 
41     Register codeBlockValue;
42     Register calleeValue;
43     Register argCountAndCodeOriginValue;
44     Register thisArg;
45     uint32_t paddedArgCount;
46     bool hasArityMismatch;
47     JSValue *args;

48 
49     void init(CodeBlock*, JSObject*, JSValue, int, JSValue* otherArgs = 0);
50 
51     CodeBlock* codeBlock() const { return codeBlockValue.Register::codeBlock(); }
52     void setCodeBlock(CodeBlock* codeBlock) { codeBlockValue = codeBlock; }
53 
54     JSObject* callee() const { return calleeValue.Register::object(); }
55     void setCallee(JSObject* callee) { calleeValue = callee; }




56 
57     int argumentCountIncludingThis() const { return argCountAndCodeOriginValue.payload(); }
58     int argumentCount() const { return argumentCountIncludingThis() - 1; }
59     void setArgumentCountIncludingThis(int count) { argCountAndCodeOriginValue.payload() = count; }
60     void setPaddedArgCount(uint32_t argCount) { paddedArgCount = argCount; }
61 
62     void clearCurrentVPC() { argCountAndCodeOriginValue.tag() = 0; }
63 
64     JSValue thisValue() const { return thisArg.Register::jsValue(); }
65     void setThisValue(JSValue value) { thisArg = value; }
66 
67     bool needArityCheck() { return hasArityMismatch; }
68 
69     JSValue argument(size_t argumentIndex)
70     {
71         ASSERT(static_cast<int>(argumentIndex) < argumentCount());
72         return args[argumentIndex];
73     }
74     void setArgument(size_t argumentIndex, JSValue value)
75     {
76         ASSERT(static_cast<int>(argumentIndex) < argumentCount());
77         args[argumentIndex] = value;
78     }
79 };
80 
81 inline void ProtoCallFrame::init(CodeBlock* codeBlock, JSObject* callee, JSValue thisValue, int argCountIncludingThis, JSValue* otherArgs)
82 {
83     this->args = otherArgs;
84     this->setCodeBlock(codeBlock);
85     this->setCallee(callee);
86     this->setArgumentCountIncludingThis(argCountIncludingThis);
87     if (codeBlock && argCountIncludingThis < codeBlock->numParameters())
88         this->hasArityMismatch = true;
89     else
90         this->hasArityMismatch = false;
91 
92     // Round up argCountIncludingThis to keep the stack frame size aligned.
93     size_t paddedArgsCount = roundArgumentCountToAlignFrame(argCountIncludingThis);
94     this->setPaddedArgCount(paddedArgsCount);
95     this->clearCurrentVPC();
96     this->setThisValue(thisValue);
97 }
98 
99 } // namespace JSC
