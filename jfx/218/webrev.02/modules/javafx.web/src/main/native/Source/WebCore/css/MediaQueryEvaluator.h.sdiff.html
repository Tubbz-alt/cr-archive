49 struct MediaQueryDynamicResults {
50     Vector<MediaQueryResult> viewport;
51     Vector<MediaQueryResult> appearance;
52     Vector<MediaQueryResult> accessibilitySettings;
53 
54     void append(const MediaQueryDynamicResults& other)
55     {
56         viewport.appendVector(other.viewport);
57         appearance.appendVector(other.appearance);
58         accessibilitySettings.appendVector(other.accessibilitySettings);
59     }
60     bool isEmpty() const { return viewport.isEmpty() && appearance.isEmpty() && accessibilitySettings.isEmpty(); }
61 };
