 50 Plan::Plan(Context* context, Ref<ModuleInformation> info, CompletionTask&& task)
 51     : m_moduleInformation(WTFMove(info))
 52 {
 53     m_completionTasks.append(std::make_pair(context, WTFMove(task)));
 54 }
 55 Plan::Plan(Context* context, CompletionTask&& task)
 56     : m_moduleInformation(ModuleInformation::create())
 57 {
 58     m_completionTasks.append(std::make_pair(context, WTFMove(task)));
 59 }
 60 
 61 void Plan::runCompletionTasks(const AbstractLocker&)
 62 {
 63     ASSERT(isComplete() && !hasWork());
 64 
 65     for (auto& task : m_completionTasks)
 66         task.second->run(*this);
 67     m_completionTasks.clear();
 68     m_completed.notifyAll();
 69 }
 70 
 71 void Plan::addCompletionTask(Context* context, CompletionTask&& task)
 72 {
 73     LockHolder locker(m_lock);
 74     if (!isComplete())
 75         m_completionTasks.append(std::make_pair(context, WTFMove(task)));
 76     else
 77         task->run(*this);
 78 }
 79 
 80 void Plan::waitForCompletion()
 81 {
 82     LockHolder locker(m_lock);
 83     if (!isComplete()) {
 84         m_completed.wait(m_lock);
 85     }
 86 }

 88 bool Plan::tryRemoveContextAndCancelIfLast(Context& context)
 89 {
 90     LockHolder locker(m_lock);
 91 
 92     if (ASSERT_ENABLED) {
 93         // We allow the first completion task to not have a Context.
 94         for (unsigned i = 1; i < m_completionTasks.size(); ++i)
 95             ASSERT(m_completionTasks[i].first);
 96     }
 97 
 98     bool removedAnyTasks = false;
 99     m_completionTasks.removeAllMatching([&] (const std::pair<Context*, CompletionTask>& pair) {
100         bool shouldRemove = pair.first == &context;
101         removedAnyTasks |= shouldRemove;
102         return shouldRemove;
103     });
104 
105     if (!removedAnyTasks)
106         return false;
107 
108     if (isComplete()) {
109         // We trivially cancel anything that's completed.
110         return true;
111     }
112 
113     // FIXME: Make 0 index not so magical: https://bugs.webkit.org/show_bug.cgi?id=171395
114     if (m_completionTasks.isEmpty() || (m_completionTasks.size() == 1 && !m_completionTasks[0].first)) {
115         fail(locker, "WebAssembly Plan was cancelled. If you see this error message please file a bug at bugs.webkit.org!"_s);
116         return true;
117     }
118 
119     return false;
120 }
121 
122 void Plan::fail(const AbstractLocker& locker, String&& errorMessage)
123 {
124     if (failed())
125         return;
126     ASSERT(errorMessage);
127     dataLogLnIf(WasmPlanInternal::verbose, "failing with message: ", errorMessage);
128     m_errorMessage = WTFMove(errorMessage);
129     complete(locker);
130 }
