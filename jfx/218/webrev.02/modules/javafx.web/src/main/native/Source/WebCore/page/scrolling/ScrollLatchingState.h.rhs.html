26 #pragma once
27 
28 #include <wtf/RefPtr.h>
29 
30 namespace WebCore {
31 
32 class ContainerNode;
33 class Element;
34 class Frame;
35 
36 class ScrollLatchingState final {
37 public:
38     ScrollLatchingState();
39     ~ScrollLatchingState();
40 
41     void clear();
42 
43     Element* wheelEventElement() { return m_wheelEventElement.get(); }
44     void setWheelEventElement(RefPtr<Element>&&);
45     Frame* frame() { return m_frame; }
46     void setFrame(Frame* frame) { m_frame = frame; }
47 
48     bool widgetIsLatched() const { return m_widgetIsLatched; }
49     void setWidgetIsLatched(bool isOverWidget);
50 
51     Element* previousWheelScrolledElement() { return m_previousWheelScrolledElement.get(); }
52     void setPreviousWheelScrolledElement(RefPtr<Element>&&);
53 
54     ContainerNode* scrollableContainer() { return m_scrollableContainer.get(); }
55     void setScrollableContainer(RefPtr<ContainerNode>&&);
56     bool startedGestureAtScrollLimit() const { return m_startedGestureAtScrollLimit; }
57     void setStartedGestureAtScrollLimit(bool startedAtLimit) { m_startedGestureAtScrollLimit = startedAtLimit; }
58 
59 private:
60     RefPtr<Element> m_wheelEventElement;
61     RefPtr<Element> m_previousWheelScrolledElement;
62     RefPtr<ContainerNode> m_scrollableContainer;
63 
64     Frame* m_frame { nullptr };
65 
66     bool m_widgetIsLatched { false };
67     bool m_startedGestureAtScrollLimit { false };
68 };
69 
70 } // namespace WebCore
