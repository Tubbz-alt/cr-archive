28 #include "LineMetadata.h"
29 #include "Mutex.h"
30 #include "Sizes.h"
31 #include "StaticPerProcess.h"
32 #include "Vector.h"
33 #include <array>

35 
36 namespace bmalloc {
37 
38 class HeapConstants : public StaticPerProcess<HeapConstants> {
39 public:
40     HeapConstants(const LockHolder&);
41     ~HeapConstants() = delete;
42 
43     inline size_t pageClass(size_t sizeClass) const { return m_pageClasses[sizeClass]; }
44     inline size_t smallLineCount() const { return bmalloc::smallLineCount(m_vmPageSizePhysical); }
45     inline unsigned char startOffset(size_t sizeClass, size_t lineNumber) const { return lineMetadata(sizeClass, lineNumber).startOffset; }
46     inline unsigned char objectCount(size_t sizeClass, size_t lineNumber) const { return lineMetadata(sizeClass, lineNumber).objectCount; }
47 
48 private:
49     void initializeLineMetadata();
50     void initializePageMetadata();
51 
52     inline const LineMetadata& lineMetadata(size_t sizeClass, size_t lineNumber) const
53     {
54         return m_smallLineMetadata[sizeClass * smallLineCount() + lineNumber];
55     }
56 
57     size_t m_vmPageSizePhysical;
58     const LineMetadata* m_smallLineMetadata { };
59     Vector<LineMetadata> m_smallLineMetadataStorage;
60     std::array<size_t, sizeClassCount> m_pageClasses;
61 };
62 DECLARE_STATIC_PER_PROCESS_STORAGE(HeapConstants);
63 
64 } // namespace bmalloc
