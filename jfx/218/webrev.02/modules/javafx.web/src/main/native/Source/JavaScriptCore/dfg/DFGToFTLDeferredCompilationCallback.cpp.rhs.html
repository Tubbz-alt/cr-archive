37 ToFTLDeferredCompilationCallback::ToFTLDeferredCompilationCallback()

41 ToFTLDeferredCompilationCallback::~ToFTLDeferredCompilationCallback() { }
42 
43 Ref<ToFTLDeferredCompilationCallback> ToFTLDeferredCompilationCallback::create()
44 {
45     return adoptRef(*new ToFTLDeferredCompilationCallback());
46 }
47 
48 void ToFTLDeferredCompilationCallback::compilationDidBecomeReadyAsynchronously(
49     CodeBlock* codeBlock, CodeBlock* profiledDFGCodeBlock)
50 {
51     dataLogLnIf(Options::verboseOSR(),
52         "Optimizing compilation of ", codeBlock, " (for ", profiledDFGCodeBlock,
53         ") did become ready.");


54 
55     profiledDFGCodeBlock->jitCode()->dfg()->forceOptimizationSlowPathConcurrently(
56         profiledDFGCodeBlock);
57 }
58 
59 void ToFTLDeferredCompilationCallback::compilationDidComplete(
60     CodeBlock* codeBlock, CodeBlock* profiledDFGCodeBlock, CompilationResult result)
61 {
62     dataLogLnIf(Options::verboseOSR(),
63         "Optimizing compilation of ", codeBlock, " (for ", profiledDFGCodeBlock,
64         ") result: ", result);


65 
66     if (profiledDFGCodeBlock->replacement() != profiledDFGCodeBlock) {
67         dataLogLnIf(Options::verboseOSR(),
68             "Dropping FTL code block ", codeBlock, " on the floor because the "
69             "DFG code block ", profiledDFGCodeBlock, " was jettisoned.");


70         return;
71     }
72 
73     if (result == CompilationSuccessful)
74         codeBlock->ownerExecutable()->installCode(codeBlock);
75 
76     profiledDFGCodeBlock->jitCode()->dfg()->setOptimizationThresholdBasedOnCompilationResult(
77         profiledDFGCodeBlock, result);
78 
79     DeferredCompilationCallback::compilationDidComplete(codeBlock, profiledDFGCodeBlock, result);
80 }
