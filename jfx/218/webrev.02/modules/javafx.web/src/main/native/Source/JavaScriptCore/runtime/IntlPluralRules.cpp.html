101 void IntlPluralRules::initializePluralRules(JSGlobalObject* globalObject, JSValue locales, JSValue optionsValue)
102 {
103     VM& vm = globalObject->vm();
104     auto scope = DECLARE_THROW_SCOPE(vm);
105 
106     // 13.1.1 InitializePluralRules (pluralRules, locales, options)
107     // https://tc39.github.io/ecma402/#sec-initializepluralrules
108     Vector<String> requestedLocales = canonicalizeLocaleList(globalObject, locales);
109     RETURN_IF_EXCEPTION(scope, void());
110 
111     JSObject* options;
112     if (optionsValue.isUndefined())
113         options = constructEmptyObject(vm, globalObject->nullPrototypeObjectStructure());
114     else {
115         options = optionsValue.toObject(globalObject);
116         RETURN_IF_EXCEPTION(scope, void());
117     }
118 
119     HashMap<String, String> localeOpt;
120     String localeMatcher = intlStringOption(globalObject, options, vm.propertyNames->localeMatcher, { "lookup", "best fit" }, "localeMatcher must be either \"lookup\" or \"best fit\"", "best fit");
121     RETURN_IF_EXCEPTION(scope, void());
122     localeOpt.add(vm.propertyNames->localeMatcher.string(), localeMatcher);
123 
124     const HashSet<String>& availableLocales = intlPluralRulesAvailableLocales();
125     HashMap<String, String> resolved = resolveLocale(globalObject, availableLocales, requestedLocales, localeOpt, nullptr, 0, IntlPRInternal::localeData);
126     m_locale = resolved.get(vm.propertyNames->locale.string());
127     if (m_locale.isEmpty()) {
128         throwTypeError(globalObject, scope, "failed to initialize PluralRules due to invalid locale"_s);
129         return;
130     }
131 
132     String typeString = intlStringOption(globalObject, options, Identifier::fromString(vm, "type"), { "cardinal", "ordinal" }, "type must be \"cardinal\" or \"ordinal\"", "cardinal");
133     RETURN_IF_EXCEPTION(scope, void());
134     m_type = typeString == "ordinal" ? UPLURAL_TYPE_ORDINAL : UPLURAL_TYPE_CARDINAL;
135 
136     unsigned minimumIntegerDigits = intlNumberOption(globalObject, options, Identifier::fromString(vm, "minimumIntegerDigits"), 1, 21, 1);
137     RETURN_IF_EXCEPTION(scope, void());
138     m_minimumIntegerDigits = minimumIntegerDigits;
139 
140     unsigned minimumFractionDigitsDefault = 0;
141     unsigned minimumFractionDigits = intlNumberOption(globalObject, options, Identifier::fromString(vm, "minimumFractionDigits"), 0, 20, minimumFractionDigitsDefault);
142     RETURN_IF_EXCEPTION(scope, void());
143     m_minimumFractionDigits = minimumFractionDigits;
144 
145     unsigned maximumFractionDigitsDefault = std::max(minimumFractionDigits, 3u);
146     unsigned maximumFractionDigits = intlNumberOption(globalObject, options, Identifier::fromString(vm, "maximumFractionDigits"), minimumFractionDigits, 20, maximumFractionDigitsDefault);
147     RETURN_IF_EXCEPTION(scope, void());
148     m_maximumFractionDigits = maximumFractionDigits;
149 
150     JSValue minimumSignificantDigitsValue = options->get(globalObject, Identifier::fromString(vm, "minimumSignificantDigits"));
151     RETURN_IF_EXCEPTION(scope, void());
152 
153     JSValue maximumSignificantDigitsValue = options->get(globalObject, Identifier::fromString(vm, "maximumSignificantDigits"));
154     RETURN_IF_EXCEPTION(scope, void());
155 
156     if (!minimumSignificantDigitsValue.isUndefined() || !maximumSignificantDigitsValue.isUndefined()) {
157         unsigned minimumSignificantDigits = intlNumberOption(globalObject, options, Identifier::fromString(vm, "minimumSignificantDigits"), 1, 21, 1);
158         RETURN_IF_EXCEPTION(scope, void());
159         unsigned maximumSignificantDigits = intlNumberOption(globalObject, options, Identifier::fromString(vm, "maximumSignificantDigits"), minimumSignificantDigits, 21, 21);
160         RETURN_IF_EXCEPTION(scope, void());
161         m_minimumSignificantDigits = minimumSignificantDigits;
162         m_maximumSignificantDigits = maximumSignificantDigits;
163     }
164 
165     UErrorCode status = U_ZERO_ERROR;
166     m_numberFormat = std::unique_ptr<UNumberFormat, UNumberFormatDeleter>(unum_open(UNUM_DECIMAL, nullptr, 0, m_locale.utf8().data(), nullptr, &status));
167     if (U_FAILURE(status)) {
168         throwTypeError(globalObject, scope, "failed to initialize PluralRules"_s);
169         return;
170     }
171 
172     if (m_minimumSignificantDigits) {
173         unum_setAttribute(m_numberFormat.get(), UNUM_SIGNIFICANT_DIGITS_USED, true);
174         unum_setAttribute(m_numberFormat.get(), UNUM_MIN_SIGNIFICANT_DIGITS, m_minimumSignificantDigits.value());
175         unum_setAttribute(m_numberFormat.get(), UNUM_MAX_SIGNIFICANT_DIGITS, m_maximumSignificantDigits.value());
176     } else {
177         unum_setAttribute(m_numberFormat.get(), UNUM_MIN_INTEGER_DIGITS, m_minimumIntegerDigits);
178         unum_setAttribute(m_numberFormat.get(), UNUM_MIN_FRACTION_DIGITS, m_minimumFractionDigits);
179         unum_setAttribute(m_numberFormat.get(), UNUM_MAX_FRACTION_DIGITS, m_maximumFractionDigits);
180     }
181 
182     status = U_ZERO_ERROR;
183     m_pluralRules = std::unique_ptr<UPluralRules, UPluralRulesDeleter>(uplrules_openForType(m_locale.utf8().data(), m_type, &status));
184     if (U_FAILURE(status)) {
185         throwTypeError(globalObject, scope, "failed to initialize PluralRules"_s);
186         return;
187     }
188 
189     m_initializedPluralRules = true;
190 }

192 JSObject* IntlPluralRules::resolvedOptions(JSGlobalObject* globalObject)
193 {
194     VM& vm = globalObject->vm();
195     auto scope = DECLARE_THROW_SCOPE(vm);
196 
197     // 13.4.4 Intl.PluralRules.prototype.resolvedOptions ()
198     // https://tc39.github.io/ecma402/#sec-intl.pluralrules.prototype.resolvedoptions
199     if (UNLIKELY(!m_initializedPluralRules)) {
200         throwTypeError(globalObject, scope, "Intl.PluralRules.prototype.resolvedOptions called on value that's not an object initialized as a PluralRules"_s);
201         return nullptr;
202     }
203 
204     JSObject* options = constructEmptyObject(globalObject);
205     options->putDirect(vm, vm.propertyNames->locale, jsNontrivialString(vm, m_locale));
206     options->putDirect(vm, Identifier::fromString(vm, "type"), jsNontrivialString(vm, m_type == UPLURAL_TYPE_ORDINAL ? "ordinal"_s : "cardinal"_s));
207     options->putDirect(vm, Identifier::fromString(vm, "minimumIntegerDigits"), jsNumber(m_minimumIntegerDigits));
208     options->putDirect(vm, Identifier::fromString(vm, "minimumFractionDigits"), jsNumber(m_minimumFractionDigits));
209     options->putDirect(vm, Identifier::fromString(vm, "maximumFractionDigits"), jsNumber(m_maximumFractionDigits));
210     if (m_minimumSignificantDigits) {
211         options->putDirect(vm, Identifier::fromString(vm, "minimumSignificantDigits"), jsNumber(m_minimumSignificantDigits.value()));
212         options->putDirect(vm, Identifier::fromString(vm, "maximumSignificantDigits"), jsNumber(m_maximumSignificantDigits.value()));
213     }
214 
215 #if HAVE(ICU_PLURALRULES_KEYWORDS)
216     JSArray* categories = JSArray::tryCreate(vm, globalObject->arrayStructureForIndexingTypeDuringAllocation(ArrayWithContiguous), 0);
217     if (UNLIKELY(!categories)) {
218         throwOutOfMemoryError(globalObject, scope);
219         return nullptr;
220     }
221 
222     UErrorCode status = U_ZERO_ERROR;
223     auto keywords = std::unique_ptr<UEnumeration, UEnumerationDeleter>(uplrules_getKeywords(m_pluralRules.get(), &status));
224     ASSERT(U_SUCCESS(status));
225     int32_t resultLength;
226 
227     // Category names are always ASCII, so use char[].
228     unsigned index = 0;
229     while (const char* result = uenum_next(keywords.get(), &resultLength, &status)) {
230         ASSERT(U_SUCCESS(status));
231         categories->putDirectIndex(globalObject, index++, jsNontrivialString(vm, String(result, resultLength)));
232         RETURN_IF_EXCEPTION(scope, { });
233     }
234     options->putDirect(vm, Identifier::fromString(vm, "pluralCategories"), categories);
235 #endif
236 
237     RELEASE_AND_RETURN(scope, options);
238 }

240 JSValue IntlPluralRules::select(JSGlobalObject* globalObject, double value)
241 {
242     VM& vm = globalObject->vm();
243     auto scope = DECLARE_THROW_SCOPE(vm);
244 
245     // 13.1.4 ResolvePlural (pluralRules, n)
246     // https://tc39.github.io/ecma402/#sec-resolveplural
247     if (!m_initializedPluralRules)
248         return throwTypeError(globalObject, scope, "Intl.PluralRules.prototype.select called on value that's not an object initialized as a PluralRules"_s);
249 
250     if (!std::isfinite(value))
251         return jsNontrivialString(vm, "other"_s);
252 
253 #if HAVE(ICU_PLURALRULES_WITH_FORMAT)
254     UErrorCode status = U_ZERO_ERROR;
255     Vector<UChar, 8> result(8);
256     auto length = uplrules_selectWithFormat(m_pluralRules.get(), value, m_numberFormat.get(), result.data(), result.size(), &status);
257     if (U_FAILURE(status))
258         return throwTypeError(globalObject, scope, "failed to select plural value"_s);
259 #else
260     UErrorCode status = U_ZERO_ERROR;
261     Vector<UChar, 32> buffer(32);
262     auto length = unum_formatDouble(m_numberFormat.get(), value, buffer.data(), buffer.size(), nullptr, &status);
263     if (status == U_BUFFER_OVERFLOW_ERROR) {
264         buffer.grow(length);
265         status = U_ZERO_ERROR;
266         unum_formatDouble(m_numberFormat.get(), value, buffer.data(), length, nullptr, &status);
267     }
268     if (U_FAILURE(status))
269         return throwTypeError(globalObject, scope, "failed to select plural value"_s);
270 
271     double formatted = unum_parseDouble(m_numberFormat.get(), buffer.data(), length, nullptr, &status);
272     if (U_FAILURE(status))
273         return throwTypeError(globalObject, scope, "failed to select plural value"_s);
274 
275     // Can only be 'zero', 'one', 'two', 'few', 'many' or 'other'
276     status = U_ZERO_ERROR;
277     Vector<UChar, 8> result(8);
278     length = uplrules_select(m_pluralRules.get(), formatted, result.data(), result.size(), &status);
279     if (U_FAILURE(status))
280         return throwTypeError(globalObject, scope, "failed to select plural value"_s);
281 #endif
282 
283     return jsString(vm, String(result.data(), length));
284 }
