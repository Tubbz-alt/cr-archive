 35 template<typename T>
 36 static inline void shrinkToFit(T& segmentedVector)
 37 {
 38     while (segmentedVector.size() && !segmentedVector.last().refCount())
 39         segmentedVector.removeLast();
 40 }
 41 
 42 template<typename Traits>
 43 BytecodeGeneratorBase<Traits>::BytecodeGeneratorBase(typename Traits::CodeBlock codeBlock, uint32_t virtualRegisterCountForCalleeSaves)
 44     : m_codeBlock(WTFMove(codeBlock))
 45 {
 46     allocateCalleeSaveSpace(virtualRegisterCountForCalleeSaves);
 47 }
 48 
 49 template<typename Traits>
 50 Ref<GenericLabel<Traits>> BytecodeGeneratorBase<Traits>::newLabel()
 51 {
 52     shrinkToFit(m_labels);
 53 
 54     // Allocate new label ID.
 55     m_labels.append();
 56     return m_labels.last();
 57 }
 58 
 59 template<typename Traits>
 60 Ref<GenericLabel<Traits>> BytecodeGeneratorBase<Traits>::newEmittedLabel()
 61 {
 62     auto label = newLabel();
 63     emitLabel(label.get());
 64     return label;
 65 }
 66 
 67 template<typename Traits>
 68 void BytecodeGeneratorBase<Traits>::reclaimFreeRegisters()
 69 {
 70     shrinkToFit(m_calleeLocals);
 71 }
 72 
 73 template<typename Traits>
 74 void BytecodeGeneratorBase<Traits>::emitLabel(GenericLabel<Traits>& label)
 75 {
 76     unsigned newLabelIndex = m_writer.position();
 77     label.setLocation(*this, newLabelIndex);
 78 
 79     if (m_codeBlock->numberOfJumpTargets()) {
 80         unsigned lastLabelIndex = m_codeBlock->lastJumpTarget();
 81         ASSERT(lastLabelIndex <= newLabelIndex);
 82         if (newLabelIndex == lastLabelIndex) {
 83             // Peephole optimizations have already been disabled by emitting the last label
 84             return;
 85         }
 86     }
 87 
 88     m_codeBlock->addJumpTarget(newLabelIndex);
 89 
 90     m_lastOpcodeID = Traits::opcodeForDisablingOptimizations;
 91 }
 92 
 93 template<typename Traits>
 94 void BytecodeGeneratorBase<Traits>::recordOpcode(typename Traits::OpcodeID opcodeID)
 95 {
 96     ASSERT(m_lastOpcodeID == Traits::opcodeForDisablingOptimizations || (m_lastOpcodeID == m_lastInstruction->opcodeID<typename Traits::OpcodeTraits>() && m_writer.position() == m_lastInstruction.offset() + m_lastInstruction->size<typename Traits::OpcodeTraits>()));
 97     m_lastInstruction = m_writer.ref();
 98     m_lastOpcodeID = opcodeID;
 99 }

101 template<typename Traits>
102 void BytecodeGeneratorBase<Traits>::alignWideOpcode16()
103 {
104 #if CPU(NEEDS_ALIGNED_ACCESS)
105     size_t opcodeSize = 1;
106     size_t prefixAndOpcodeSize = opcodeSize + PaddingBySize<OpcodeSize::Wide16>::value;
107     while ((m_writer.position() + prefixAndOpcodeSize) % OpcodeSize::Wide16)
108         Traits::OpNop::template emit<OpcodeSize::Narrow>(this);
109 #endif
110 }
111 
112 template<typename Traits>
113 void BytecodeGeneratorBase<Traits>::alignWideOpcode32()
114 {
115 #if CPU(NEEDS_ALIGNED_ACCESS)
116     size_t opcodeSize = 1;
117     size_t prefixAndOpcodeSize = opcodeSize + PaddingBySize<OpcodeSize::Wide16>::value;
118     while ((m_writer.position() + prefixAndOpcodeSize) % OpcodeSize::Wide32)
119         Traits::OpNop::template emit<OpcodeSize::Narrow>(this);
120 #endif
121 }

160 template<typename Traits>
161 RegisterID* BytecodeGeneratorBase<Traits>::newRegister()
162 {
163     m_calleeLocals.append(virtualRegisterForLocal(m_calleeLocals.size()));
164     int numCalleeLocals = std::max<int>(m_codeBlock->numCalleeLocals(), m_calleeLocals.size());
165     numCalleeLocals = WTF::roundUpToMultipleOf(stackAlignmentRegisters(), numCalleeLocals);
166     m_codeBlock->setNumCalleeLocals(numCalleeLocals);
167     return &m_calleeLocals.last();
168 }
169 
170 template<typename Traits>
171 RegisterID* BytecodeGeneratorBase<Traits>::newTemporary()
172 {
173     reclaimFreeRegisters();
174 
175     RegisterID* result = newRegister();
176     result->setTemporary();
177     return result;
178 }
179 
180 // Adds an anonymous local var slot. To give this slot a name, add it to symbolTable().
181 template<typename Traits>
182 RegisterID* BytecodeGeneratorBase<Traits>::addVar()
183 {
184     int numVars = m_codeBlock->numVars();
185     m_codeBlock->setNumVars(numVars + 1);
186     RegisterID* result = newRegister();
187     ASSERT(VirtualRegister(result->index()).toLocal() == numVars);
188     result->ref(); // We should never free this slot.
189     return result;
190 }
191 
192 template<typename Traits>
193 void BytecodeGeneratorBase<Traits>::allocateCalleeSaveSpace(uint32_t virtualRegisterCountForCalleeSaves)
194 {
195     for (size_t i = 0; i < virtualRegisterCountForCalleeSaves; i++)
196         addVar();
197 }
