49 class ServiceWorker final : public RefCounted<ServiceWorker>, public EventTargetWithInlineData, public ActiveDOMObject {
50     WTF_MAKE_ISO_ALLOCATED(ServiceWorker);
51 public:
52     using State = ServiceWorkerState;
53     static Ref<ServiceWorker> getOrCreate(ScriptExecutionContext&, ServiceWorkerData&&);
54 
55     virtual ~ServiceWorker();
56 
57     const URL& scriptURL() const { return m_data.scriptURL; }
58 
59     State state() const { return m_data.state; }
60 
61     void updateState(State);
62 
63     ExceptionOr<void> postMessage(JSC::JSGlobalObject&, JSC::JSValue message, PostMessageOptions&&);
64 
65     ServiceWorkerIdentifier identifier() const { return m_data.identifier; }
66     ServiceWorkerRegistrationIdentifier registrationIdentifier() const { return m_data.registrationIdentifier; }
67 
68     using RefCounted::ref;
69     using RefCounted::deref;
70 
71 private:
72     ServiceWorker(ScriptExecutionContext&, ServiceWorkerData&&);
73     void updatePendingActivityForEventDispatch();
74 
75     EventTargetInterface eventTargetInterface() const final;
76     ScriptExecutionContext* scriptExecutionContext() const final;
77     void refEventTarget() final { ref(); }
78     void derefEventTarget() final { deref(); }
79 
80     // ActiveDOMObject.
81     const char* activeDOMObjectName() const final;
82     void stop() final;
83 
84     bool isAlwaysOnLoggingAllowed() const;
85 
86     SWClientConnection& swConnection();
87 
88     ServiceWorkerData m_data;
89     bool m_isStopped { false };
90     RefPtr<PendingActivity<ServiceWorker>> m_pendingActivityForEventDispatch;
91 };
