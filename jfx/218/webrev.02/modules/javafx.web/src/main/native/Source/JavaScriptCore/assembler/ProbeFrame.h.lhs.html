26 #pragma once
27 
28 #if ENABLE(MASM_PROBE)
29 
30 #include "CallFrame.h"
31 #include "ProbeStack.h"
32 
33 namespace JSC {
34 namespace Probe {
35 
36 class Frame {
37 public:
38     Frame(void* frameBase, Stack& stack)
39         : m_frameBase { reinterpret_cast<uint8_t*>(frameBase) }
40         , m_stack { stack }
41     { }
42 
43     template<typename T = JSValue>
44     T argument(int argument)
45     {
46         return get<T>(CallFrame::argumentOffset(argument) * sizeof(Register));
47     }
48     template<typename T = JSValue>
49     T operand(int operand)
50     {
51         return get<T>(static_cast<VirtualRegister>(operand).offset() * sizeof(Register));
52     }
53     template<typename T = JSValue>
54     T operand(int operand, ptrdiff_t offset)
55     {
56         return get<T>(static_cast<VirtualRegister>(operand).offset() * sizeof(Register) + offset);
57     }
58 
59     template<typename T>
60     void setArgument(int argument, T value)
61     {
62         return set<T>(CallFrame::argumentOffset(argument) * sizeof(Register), value);
63     }
64     template<typename T>
65     void setOperand(int operand, T value)
66     {
67         set<T>(static_cast<VirtualRegister>(operand).offset() * sizeof(Register), value);
68     }
69     template<typename T>
70     void setOperand(int operand, ptrdiff_t offset, T value)
71     {
72         set<T>(static_cast<VirtualRegister>(operand).offset() * sizeof(Register) + offset, value);
73     }
74 
75     template<typename T = JSValue>
76     T get(ptrdiff_t offset)
77     {
78         return m_stack.get<T>(m_frameBase + offset);
79     }
80     template<typename T>
81     void set(ptrdiff_t offset, T value)
82     {
83         m_stack.set<T>(m_frameBase + offset, value);
84     }
85 
86 private:
87     uint8_t* m_frameBase;
88     Stack& m_stack;
89 };
90 
91 } // namespace Probe
92 } // namespace JSC
93 
94 #endif // ENABLE(MASM_PROBE)
