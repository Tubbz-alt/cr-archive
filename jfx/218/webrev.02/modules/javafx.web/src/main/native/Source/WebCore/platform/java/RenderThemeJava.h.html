 63     void adjustTextFieldStyle(RenderStyle&, const Element*) const override;
 64     bool paintTextField(const RenderObject&, const PaintInfo&, const FloatRect&) override;

 66     void adjustSearchFieldStyle(RenderStyle&, const Element*) const override;
 67     bool paintSearchField(const RenderObject&, const PaintInfo&, const IntRect&) override;

 84 #if ENABLE(VIDEO)

 87 
 88     String formatMediaControlsCurrentTime(float currentTime, float duration) const override;
 89     String formatMediaControlsRemainingTime(float currentTime, float duration) const override;
 90 
 91     bool paintMediaPlayButton(const RenderObject&, const PaintInfo&, const IntRect&) override;

137     bool paintWidget(int widgetIndex, const RenderObject& o,

142 #endif
