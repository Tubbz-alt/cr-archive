36 namespace JSC {
37 
38 class ArrayBuffer;
39 class JSArrayBuffer;
40 
41 class JSWebAssemblyMemory final : public JSDestructibleObject {
42 public:
43     typedef JSDestructibleObject Base;


44 
45     template<typename CellType, SubspaceAccess>
46     static CompleteSubspace* subspaceFor(VM& vm)
47     {
48         // We hold onto a lot of memory, so it makes a lot of sense to be swept eagerly.
49         return &vm.eagerlySweptDestructibleObjectSpace;
50     }
51 
52     static JSWebAssemblyMemory* create(ExecState*, VM&, Structure*);
53     static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
54 
55     DECLARE_EXPORT_INFO;
56 
57     void adopt(Ref<Wasm::Memory>&&);
58     Wasm::Memory& memory() { return m_memory.get(); }
59     JSArrayBuffer* buffer(VM& vm, JSGlobalObject*);
60     Wasm::PageCount grow(VM&, ExecState*, uint32_t delta);
61     void growSuccessCallback(VM&, Wasm::PageCount oldPageCount, Wasm::PageCount newPageCount);
62 
63 private:
64     JSWebAssemblyMemory(VM&, Structure*);
65     void finishCreation(VM&);
66     static void destroy(JSCell*);
67     static void visitChildren(JSCell*, SlotVisitor&);
68 
69     Ref<Wasm::Memory> m_memory;
70     WriteBarrier<JSArrayBuffer> m_bufferWrapper;
71     RefPtr<ArrayBuffer> m_buffer;
72 };
73 
74 } // namespace JSC
