 26 #pragma once
 27 
 28 #include <wtf/DumbPtrTraits.h>
 29 #include <wtf/FastMalloc.h>
 30 #include <wtf/Noncopyable.h>
 31 #include <wtf/Packed.h>
 32 
 33 namespace WTF {
 34 
 35 DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(BagNode);

 36 template<typename T, typename PassedPtrTraits = DumbPtrTraits<T>>
 37 class BagNode {
 38     WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(BagNode);
 39 public:
 40     using PtrTraits = typename PassedPtrTraits::template RebindTraits<BagNode>;
 41 
 42     template<typename... Args>
 43     BagNode(Args&&... args)
 44         : m_item(std::forward<Args>(args)...)
 45     { }
 46 
 47     T m_item;
 48     typename PtrTraits::StorageType m_next { nullptr };
 49 };
 50 


 51 template<typename T, typename PassedPtrTraits = DumbPtrTraits<T>>
 52 class Bag final {
 53     WTF_MAKE_NONCOPYABLE(Bag);
 54     WTF_MAKE_FAST_ALLOCATED;
 55     using Node = BagNode<T, PassedPtrTraits>;
 56     using PtrTraits = typename PassedPtrTraits::template RebindTraits<Node>;
 57 
 58 public:
 59     Bag() = default;
 60 
 61     template<typename U>
 62     Bag(Bag<T, U>&& other)
 63     {
 64         ASSERT(!m_head);
 65         m_head = other.unwrappedHead();
 66         other.m_head = nullptr;
 67     }
 68 
 69     ~Bag()
 70     {
 71         clear();
 72     }
 73 
 74     void clear()
 75     {
 76         Node* head = this->unwrappedHead();
 77         while (head) {
 78             Node* current = head;
 79             head = Node::PtrTraits::unwrap(current->m_next);
 80             delete current;
 81         }
 82         m_head = nullptr;
 83     }
 84 
 85     template<typename... Args>
 86     T* add(Args&&... args)
 87     {
 88         Node* newNode = new Node(std::forward<Args>(args)...);
 89         newNode->m_next = unwrappedHead();
 90         m_head = newNode;
 91         return &newNode->m_item;
 92     }
 93 
 94     class iterator {
 95         WTF_MAKE_FAST_ALLOCATED;
 96     public:
 97         iterator()
 98             : m_node(0)
 99         {
100         }
101 
102         // This is sort of cheating; it's equivalent to iter == end().
103         bool operator!() const { return !m_node; }
104 
105         T* operator*() const { return &m_node->m_item; }
106 
107         iterator& operator++()
108         {
109             m_node = Node::PtrTraits::unwrap(m_node->m_next);
110             return *this;
111         }
112 
113         bool operator==(const iterator& other) const
114         {
115             return m_node == other.m_node;
116         }
117 
118         bool operator!=(const iterator& other) const
119         {
120             return !(*this == other);
121         }
122 
123     private:
124         template<typename, typename> friend class WTF::Bag;
125         Node* m_node;
126     };
127 
128     iterator begin()
129     {
130         iterator result;
131         result.m_node = unwrappedHead();
132         return result;
133     }
134 
135     const iterator begin() const
136     {
137         iterator result;
138         result.m_node = unwrappedHead();
139         return result;
140     }
141 
142 
143     iterator end() const { return iterator(); }
144 
145     bool isEmpty() const { return !m_head; }
146 
147 private:
148     Node* unwrappedHead() const { return PtrTraits::unwrap(m_head); }
149 
150     typename PtrTraits::StorageType m_head { nullptr };
151 };
152 
153 template<typename T>
154 using PackedBag = Bag<T, PackedPtrTraits<T>>;
155 
156 } // namespace WTF
157 
158 using WTF::Bag;
159 using WTF::PackedBag;
