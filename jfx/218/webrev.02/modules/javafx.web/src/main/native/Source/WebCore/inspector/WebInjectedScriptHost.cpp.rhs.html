 49 JSValue WebInjectedScriptHost::subtype(JSGlobalObject* exec, JSValue value)
 50 {
 51     VM& vm = exec->vm();
 52     if (value.inherits<JSNode>(vm))
 53         return jsNontrivialString(vm, "node"_s);
 54     if (value.inherits<JSNodeList>(vm))
 55         return jsNontrivialString(vm, "array"_s);
 56     if (value.inherits<JSHTMLCollection>(vm))
 57         return jsNontrivialString(vm, "array"_s);
 58     if (value.inherits<JSDOMException>(vm))
 59         return jsNontrivialString(vm, "error"_s);
 60 
 61     return jsUndefined();
 62 }

 64 static JSObject* constructInternalProperty(VM& vm, JSGlobalObject* exec, const String& name, JSValue value)

 65 {
 66     auto* object = constructEmptyObject(exec);
 67     object->putDirect(vm, Identifier::fromString(vm, "name"), jsString(vm, name));
 68     object->putDirect(vm, Identifier::fromString(vm, "value"), value);
 69     return object;
 70 }

 72 #if ENABLE(PAYMENT_REQUEST)
 73 static JSObject* objectForPaymentOptions(VM& vm, JSGlobalObject* exec, const PaymentOptions& paymentOptions)
 74 {
 75     auto* object = constructEmptyObject(exec);
 76     object->putDirect(vm, Identifier::fromString(vm, "requestPayerName"), jsBoolean(paymentOptions.requestPayerName));
 77     object->putDirect(vm, Identifier::fromString(vm, "requestPayerEmail"), jsBoolean(paymentOptions.requestPayerEmail));
 78     object->putDirect(vm, Identifier::fromString(vm, "requestPayerPhone"), jsBoolean(paymentOptions.requestPayerPhone));
 79     object->putDirect(vm, Identifier::fromString(vm, "requestShipping"), jsBoolean(paymentOptions.requestShipping));
 80     object->putDirect(vm, Identifier::fromString(vm, "shippingType"), jsNontrivialString(vm, convertEnumerationToString(paymentOptions.shippingType)));
 81     return object;
 82 }

 92 static JSObject* objectForPaymentItem(VM& vm, JSGlobalObject* exec, const PaymentItem& paymentItem)

111 static JSObject* objectForPaymentDetailsModifier(VM& vm, JSGlobalObject* exec, const PaymentDetailsModifier& modifier)
112 {
113     auto* additionalDisplayItems = constructEmptyArray(exec, nullptr);
114     for (unsigned i = 0; i < modifier.additionalDisplayItems.size(); ++i)
115         additionalDisplayItems->putDirectIndex(exec, i, objectForPaymentItem(vm, exec, modifier.additionalDisplayItems[i]));
116 
117     auto* object = constructEmptyObject(exec);
118     object->putDirect(vm, Identifier::fromString(vm, "supportedMethods"), jsString(vm, modifier.supportedMethods));
119     object->putDirect(vm, Identifier::fromString(vm, "total"), !modifier.total ? jsNull() : objectForPaymentItem(vm, exec, *modifier.total));
120     object->putDirect(vm, Identifier::fromString(vm, "additionalDisplayItems"), additionalDisplayItems);
121     object->putDirect(vm, Identifier::fromString(vm, "data"), !modifier.data ? jsNull() : modifier.data.get());
122     return object;
123 }
124 
125 static JSObject* objectForPaymentDetails(VM& vm, JSGlobalObject* exec, const PaymentDetailsInit& paymentDetails)
126 {
127     auto* displayItems = constructEmptyArray(exec, nullptr);
128     for (unsigned i = 0; i < paymentDetails.displayItems.size(); ++i)
129         displayItems->putDirectIndex(exec, i, objectForPaymentItem(vm, exec, paymentDetails.displayItems[i]));
130 
131     auto* shippingOptions = constructEmptyArray(exec, nullptr);
132     for (unsigned i = 0; i < paymentDetails.shippingOptions.size(); ++i)
133         shippingOptions->putDirectIndex(exec, i, objectForPaymentShippingOption(vm, exec, paymentDetails.shippingOptions[i]));
134 
135     auto* modifiers = constructEmptyArray(exec, nullptr);
136     for (unsigned i = 0; i < paymentDetails.modifiers.size(); ++i)
137         modifiers->putDirectIndex(exec, i, objectForPaymentDetailsModifier(vm, exec, paymentDetails.modifiers[i]));
138 
139     auto* object = constructEmptyObject(exec);
140     object->putDirect(vm, Identifier::fromString(vm, "id"), jsString(vm, paymentDetails.id));
141     object->putDirect(vm, Identifier::fromString(vm, "total"), objectForPaymentItem(vm, exec, paymentDetails.total));
142     object->putDirect(vm, Identifier::fromString(vm, "displayItems"), displayItems);
143     object->putDirect(vm, Identifier::fromString(vm, "shippingOptions"), shippingOptions);
144     object->putDirect(vm, Identifier::fromString(vm, "modifiers"), modifiers);
145     return object;
146 }

148 static JSString* jsStringForPaymentRequestState(VM& vm, PaymentRequest::State state)
149 {
150     switch (state) {
151     case PaymentRequest::State::Created:
152         return jsNontrivialString(vm, "created"_s);
153     case PaymentRequest::State::Interactive:
154         return jsNontrivialString(vm, "interactive"_s);
155     case PaymentRequest::State::Closed:
156         return jsNontrivialString(vm, "closed"_s);
157     }
158 
159     ASSERT_NOT_REACHED();
160     return jsEmptyString(vm);
161 }

164 JSValue WebInjectedScriptHost::getInternalProperties(VM& vm, JSGlobalObject* exec, JSC::JSValue value)
165 {

166     auto scope = DECLARE_THROW_SCOPE(vm);
167 
168     if (auto* worker = JSWorker::toWrapped(vm, value)) {
169         unsigned index = 0;
170         auto* array = constructEmptyArray(exec, nullptr);
171         array->putDirectIndex(exec, index++, constructInternalProperty(vm, exec, "terminated"_s, jsBoolean(worker->wasTerminated())));
172         RETURN_IF_EXCEPTION(scope, { });
173         return array;
174     }
175 
176 #if ENABLE(PAYMENT_REQUEST)
177     if (PaymentRequest* paymentRequest = JSPaymentRequest::toWrapped(vm, value)) {
178         unsigned index = 0;
179         auto* array = constructEmptyArray(exec, nullptr);
180         array->putDirectIndex(exec, index++, constructInternalProperty(vm, exec, "options"_s, objectForPaymentOptions(vm, exec, paymentRequest->paymentOptions())));
181         array->putDirectIndex(exec, index++, constructInternalProperty(vm, exec, "details"_s, objectForPaymentDetails(vm, exec, paymentRequest->paymentDetails())));
182         array->putDirectIndex(exec, index++, constructInternalProperty(vm, exec, "state"_s, jsStringForPaymentRequestState(vm, paymentRequest->state())));
183         RETURN_IF_EXCEPTION(scope, { });
184         return array;
185     }




186 #endif
187 
188     return { };
189 }
