37 class SaneStringGetByValSlowPathGenerator : public JumpingSlowPathGenerator<MacroAssembler::Jump> {
38 public:
39     SaneStringGetByValSlowPathGenerator(
40         const MacroAssembler::Jump& from, SpeculativeJIT* jit, JSValueRegs resultRegs, CCallHelpers::TrustedImmPtr globalObject, GPRReg baseReg, GPRReg propertyReg)
41         : JumpingSlowPathGenerator<MacroAssembler::Jump>(from, jit)
42         , m_resultRegs(resultRegs)
43         , m_globalObject(globalObject)
44         , m_baseReg(baseReg)
45         , m_propertyReg(propertyReg)
46     {
47         jit->silentSpillAllRegistersImpl(false, m_plans, extractResult(resultRegs));
48     }
49 
50 protected:
51     void generateInternal(SpeculativeJIT* jit) override
52     {
53         linkFrom(jit);
54 
55         MacroAssembler::Jump isNeg = jit->m_jit.branch32(
56             MacroAssembler::LessThan, m_propertyReg, MacroAssembler::TrustedImm32(0));
57 
58 #if USE(JSVALUE64)
59         jit->m_jit.move(
60             MacroAssembler::TrustedImm64(JSValue::encode(jsUndefined())), m_resultRegs.gpr());
61 #else
62         jit->m_jit.move(
63             MacroAssembler::TrustedImm32(JSValue::UndefinedTag), m_resultRegs.tagGPR());
64         jit->m_jit.move(
65             MacroAssembler::TrustedImm32(0), m_resultRegs.payloadGPR());
66 #endif
67         jumpTo(jit);
68 
69         isNeg.link(&jit->m_jit);
70 
71         for (unsigned i = 0; i < m_plans.size(); ++i)
72             jit->silentSpill(m_plans[i]);
73         jit->callOperation(operationGetByValStringInt, extractResult(m_resultRegs), m_globalObject, m_baseReg, m_propertyReg);
74         for (unsigned i = m_plans.size(); i--;)
75             jit->silentFill(m_plans[i]);
76         jit->m_jit.exceptionCheck();
77 
78         jumpTo(jit);
79     }
80 
81 private:
82     JSValueRegs m_resultRegs;
83     CCallHelpers::TrustedImmPtr m_globalObject;
84     GPRReg m_baseReg;
85     GPRReg m_propertyReg;
86     Vector<SilentRegisterSavePlan, 2> m_plans;
87 };
88 
89 } } // namespace JSC::DFG
