 45 ModuleProgramExecutable::ModuleProgramExecutable(JSGlobalObject* globalObject, const SourceCode& source)
 46     : Base(globalObject->vm().moduleProgramExecutableStructure.get(), globalObject->vm(), source, false, DerivedContextType::None, false, EvalContextType::None, NoIntrinsic)
 47 {
 48     ASSERT(source.provider()->sourceType() == SourceProviderSourceType::Module);
 49     VM& vm = globalObject->vm();
 50     if (vm.typeProfiler() || vm.controlFlowProfiler())
 51         vm.functionHasExecutedCache()->insertUnexecutedRange(sourceID(), typeProfilingStartOffset(vm), typeProfilingEndOffset(vm));
 52 }

 54 ModuleProgramExecutable* ModuleProgramExecutable::create(JSGlobalObject* globalObject, const SourceCode& source)
 55 {
 56     VM& vm = globalObject->vm();
 57     auto scope = DECLARE_THROW_SCOPE(vm);
 58 
 59     ModuleProgramExecutable* executable = new (NotNull, allocateCell<ModuleProgramExecutable>(vm.heap)) ModuleProgramExecutable(globalObject, source);
 60     executable->finishCreation(globalObject->vm());
 61 
 62     ParserError error;
 63     OptionSet<CodeGenerationMode> codeGenerationMode = globalObject->defaultCodeGenerationMode();
 64     UnlinkedModuleProgramCodeBlock* unlinkedModuleProgramCode = vm.codeCache()->getUnlinkedModuleProgramCodeBlock(
 65         vm, executable, executable->source(), codeGenerationMode, error);
 66 
 67     if (globalObject->hasDebugger())
 68         globalObject->debugger()->sourceParsed(globalObject, executable->source().provider(), error.line(), error.message());
 69 
 70     if (error.isValid()) {
 71         throwVMError(globalObject, scope, error.toErrorObject(globalObject, executable->source()));
 72         return nullptr;
 73     }
 74 
 75     executable->m_unlinkedModuleProgramCodeBlock.set(globalObject->vm(), executable, unlinkedModuleProgramCode);
 76 
 77     executable->m_moduleEnvironmentSymbolTable.set(globalObject->vm(), executable, jsCast<SymbolTable*>(unlinkedModuleProgramCode->constantRegister(VirtualRegister(unlinkedModuleProgramCode->moduleEnvironmentSymbolTableConstantRegisterOffset())).get())->cloneScopePart(globalObject->vm()));
 78 
 79     return executable;
 80 }

 82 void ModuleProgramExecutable::destroy(JSCell* cell)

 84     static_cast<ModuleProgramExecutable*>(cell)->ModuleProgramExecutable::~ModuleProgramExecutable();

 87 auto ModuleProgramExecutable::ensureTemplateObjectMap(VM&) -> TemplateObjectMap&
 88 {
 89     return ensureTemplateObjectMapImpl(m_templateObjectMap);
 90 }
 91 
 92 void ModuleProgramExecutable::visitChildren(JSCell* cell, SlotVisitor& visitor)
 93 {
 94     ModuleProgramExecutable* thisObject = jsCast<ModuleProgramExecutable*>(cell);
 95     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
 96     Base::visitChildren(thisObject, visitor);
 97     visitor.append(thisObject->m_unlinkedModuleProgramCodeBlock);
 98     visitor.append(thisObject->m_moduleEnvironmentSymbolTable);
 99     visitor.append(thisObject->m_moduleProgramCodeBlock);
100     if (TemplateObjectMap* map = thisObject->m_templateObjectMap.get()) {
101         auto locker = holdLock(thisObject->cellLock());
102         for (auto& entry : *map)
103             visitor.append(entry.value);
104     }
105 }
