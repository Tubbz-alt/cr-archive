 36 inline static unsigned determineDeltaMode(const PlatformWheelEvent& event)
 37 {
 38     return event.granularity() == ScrollByPageWheelEvent ? WheelEvent::DOM_DELTA_PAGE : WheelEvent::DOM_DELTA_PIXEL;
 39 }
 40 
 41 inline WheelEvent::WheelEvent() = default;
 42 
 43 inline WheelEvent::WheelEvent(const AtomString& type, const Init& initializer)
 44     : MouseEvent(type, initializer)
 45     , m_wheelDelta(initializer.wheelDeltaX ? initializer.wheelDeltaX : -initializer.deltaX, initializer.wheelDeltaY ? initializer.wheelDeltaY : -initializer.deltaY)
 46     , m_deltaX(initializer.deltaX ? initializer.deltaX : -initializer.wheelDeltaX)
 47     , m_deltaY(initializer.deltaY ? initializer.deltaY : -initializer.wheelDeltaY)
 48     , m_deltaZ(initializer.deltaZ)
 49     , m_deltaMode(initializer.deltaMode)
 50 {
 51 }
 52 
 53 inline WheelEvent::WheelEvent(const PlatformWheelEvent& event, RefPtr<WindowProxy>&& view)
 54     : MouseEvent(eventNames().wheelEvent, CanBubble::Yes, IsCancelable::Yes, IsComposed::Yes, event.timestamp().approximateMonotonicTime(), WTFMove(view), 0,
 55         event.globalPosition(), event.position() , { }, event.modifiers(), 0, 0, nullptr, 0, 0, IsSimulated::No, IsTrusted::Yes)
 56     , m_wheelDelta(event.wheelTicksX() * TickMultiplier, event.wheelTicksY() * TickMultiplier)
 57     , m_deltaX(-event.deltaX())
 58     , m_deltaY(-event.deltaY())
 59     , m_deltaMode(determineDeltaMode(event))
 60     , m_underlyingPlatformEvent(event)
 61 {
 62 }
 63 
 64 Ref<WheelEvent> WheelEvent::create(const PlatformWheelEvent& event, RefPtr<WindowProxy>&& view)
 65 {
 66     return adoptRef(*new WheelEvent(event, WTFMove(view)));
 67 }
 68 
 69 Ref<WheelEvent> WheelEvent::createForBindings()
 70 {
 71     return adoptRef(*new WheelEvent);
 72 }
 73 
 74 Ref<WheelEvent> WheelEvent::create(const AtomString& type, const Init& initializer)
 75 {
 76     return adoptRef(*new WheelEvent(type, initializer));
 77 }
 78 
 79 void WheelEvent::initWebKitWheelEvent(int rawDeltaX, int rawDeltaY, RefPtr<WindowProxy>&& view, int screenX, int screenY, int pageX, int pageY, bool ctrlKey, bool altKey, bool shiftKey, bool metaKey)
 80 {
 81     if (isBeingDispatched())
 82         return;
 83 
 84     initMouseEvent(eventNames().wheelEvent, true, true, WTFMove(view), 0, screenX, screenY, pageX, pageY, ctrlKey, altKey, shiftKey, metaKey, 0, nullptr);
 85 
 86     // Normalize to 120 multiple for compatibility with IE.
 87     m_wheelDelta = { rawDeltaX * TickMultiplier, rawDeltaY * TickMultiplier };
 88     m_deltaX = -rawDeltaX;
 89     m_deltaY = -rawDeltaY;
 90 
 91     m_deltaMode = DOM_DELTA_PIXEL;
 92 
 93     m_underlyingPlatformEvent = WTF::nullopt;
 94 }
