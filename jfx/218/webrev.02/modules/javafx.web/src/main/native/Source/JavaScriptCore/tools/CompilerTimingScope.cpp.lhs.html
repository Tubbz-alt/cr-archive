26 #include "config.h"
27 #include "CompilerTimingScope.h"
28 
29 #include "Options.h"
30 #include <wtf/DataLog.h>
31 #include <wtf/HashMap.h>
32 #include <wtf/Lock.h>


33 
34 namespace JSC {
35 
36 namespace {
37 
38 class CompilerTimingScopeState {
39     WTF_MAKE_NONCOPYABLE(CompilerTimingScopeState);
40     WTF_MAKE_FAST_ALLOCATED;
41 public:
42     CompilerTimingScopeState() { }
43 
44     Seconds addToTotal(const char* compilerName, const char* name, Seconds duration)
45     {
46         auto locker = holdLock(lock);
47         return totals.add(std::make_pair(compilerName, name), Seconds(0)).iterator->value += duration;

















48     }
49 
50 private:
51     HashMap<std::pair<const char*, const char*>, Seconds> totals;
52     Lock lock;
53 };
54 
55 CompilerTimingScopeState& compilerTimingScopeState()
56 {
57     static Atomic<CompilerTimingScopeState*> s_state;
58     return ensurePointer(s_state, [] { return new CompilerTimingScopeState(); });
59 }
60 
61 } // anonymous namespace
62 
63 CompilerTimingScope::CompilerTimingScope(const char* compilerName, const char* name)
64     : m_compilerName(compilerName)
65     , m_name(name)
66 {
67     if (Options::logPhaseTimes())
68         m_before = MonotonicTime::now();
69 }
70 
71 CompilerTimingScope::~CompilerTimingScope()
72 {
73     if (Options::logPhaseTimes()) {
74         Seconds duration = MonotonicTime::now() - m_before;
75         dataLog(
76             "[", m_compilerName, "] ", m_name, " took: ", duration.milliseconds(), " ms ",
77             "(total: ", compilerTimingScopeState().addToTotal(m_compilerName, m_name, duration).milliseconds(),
78             " ms).\n");



79     }
80 }
81 
82 } // namespace JSC
