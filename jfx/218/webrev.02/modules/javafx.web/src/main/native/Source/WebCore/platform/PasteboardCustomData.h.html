 26 #pragma once
 27 
 28 #include <wtf/Function.h>
 29 #include <wtf/HashMap.h>
 30 #include <wtf/Variant.h>
 31 #include <wtf/Vector.h>
 32 #include <wtf/text/WTFString.h>
 33 
 34 namespace WebCore {
 35 
 36 class SharedBuffer;
 37 
 38 class PasteboardCustomData {
 39 public:
 40     struct Entry {
 41         WEBCORE_EXPORT Entry();
 42         WEBCORE_EXPORT Entry(const Entry&);
 43         WEBCORE_EXPORT Entry(Entry&&);
 44         WEBCORE_EXPORT Entry& operator=(const Entry& otherData);
 45         WEBCORE_EXPORT Entry& operator=(Entry&& otherData);
 46         Entry(const String& type);
 47 
 48         String type;
 49         String customData;
 50         Variant<String, Ref<SharedBuffer>> platformData;
 51     };
 52 
 53     WEBCORE_EXPORT PasteboardCustomData();
 54     WEBCORE_EXPORT PasteboardCustomData(String&& origin, Vector<Entry>&&);
 55     WEBCORE_EXPORT PasteboardCustomData(PasteboardCustomData&&);
 56     WEBCORE_EXPORT PasteboardCustomData(const PasteboardCustomData&);
 57     WEBCORE_EXPORT ~PasteboardCustomData();
 58 
 59     const String& origin() const { return m_origin; }
 60     void setOrigin(const String& origin) { m_origin = origin; }
 61 
 62     WEBCORE_EXPORT Ref<SharedBuffer> createSharedBuffer() const;
 63     WEBCORE_EXPORT static PasteboardCustomData fromSharedBuffer(const SharedBuffer&);
 64 
 65     String readString(const String& type) const;
 66     RefPtr<SharedBuffer> readBuffer(const String& type) const;
 67     String readStringInCustomData(const String& type) const;
 68 
 69     void writeString(const String& type, const String& value);
 70     void writeData(const String& type, Ref<SharedBuffer>&& data);
 71     void writeStringInCustomData(const String& type, const String& value);
 72 
 73     void clear();
 74     void clear(const String& type);
 75 
 76 #if PLATFORM(COCOA)
 77     WEBCORE_EXPORT static const char* cocoaType();
 78 #endif
 79 
 80     void forEachType(Function<void(const String&)>&&) const;
 81     void forEachPlatformString(Function<void(const String& type, const String& data)>&&) const;
 82     void forEachPlatformStringOrBuffer(Function<void(const String& type, const Variant<String, Ref<SharedBuffer>>& data)>&&) const;
 83     void forEachCustomString(Function<void(const String& type, const String& data)>&&) const;
 84 
 85     bool hasData() const;
 86     bool hasSameOriginCustomData() const;
 87 
 88     Vector<String> orderedTypes() const;
 89     WEBCORE_EXPORT PasteboardCustomData& operator=(const PasteboardCustomData& otherData);
 90 
 91     const Vector<Entry>& data() const { return m_data; }
 92 
 93 private:
 94     HashMap<String, String> sameOriginCustomStringData() const;
 95     Entry& addOrMoveEntryToEnd(const String&);
 96 
 97     String m_origin;
 98     Vector<Entry> m_data;
 99 };
100 
101 } // namespace WebCore
