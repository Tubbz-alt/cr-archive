 85         throwVMError(callFrame, scope, createNotAnObjectError(callFrame, thisValue));
 86         return nullptr;
 87     }
 88     auto* set = jsDynamicCast<JSSet*>(vm, thisValue.asCell());

 90         return set;

 92     return nullptr;
