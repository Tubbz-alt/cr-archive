28 #if ENABLE(SERVICE_WORKER)

30 #include "SWServer.h"
31 #include "ServiceWorkerJobData.h"
32 #include "Timer.h"
33 #include <wtf/Deque.h>
34 
35 namespace WebCore {
36 
37 class SWServerJobQueue {
38     WTF_MAKE_FAST_ALLOCATED;
39 public:
40     explicit SWServerJobQueue(SWServer&, const ServiceWorkerRegistrationKey&);
41     SWServerJobQueue(const SWServerRegistration&) = delete;
42     ~SWServerJobQueue();
43 
44     const ServiceWorkerJobData& firstJob() const { return m_jobQueue.first(); }
45     const ServiceWorkerJobData& lastJob() const { return m_jobQueue.last(); }
46     void enqueueJob(const ServiceWorkerJobData& jobData) { m_jobQueue.append(jobData); }
47     size_t size() const { return m_jobQueue.size(); }
48 
49     void runNextJob();
50 
51     void scriptFetchFinished(SWServer::Connection&, const ServiceWorkerFetchResult&);
52     void scriptContextFailedToStart(const ServiceWorkerJobDataIdentifier&, ServiceWorkerIdentifier, const String& message);
53     void scriptContextStarted(const ServiceWorkerJobDataIdentifier&, ServiceWorkerIdentifier);
54     void didFinishInstall(const ServiceWorkerJobDataIdentifier&, ServiceWorkerIdentifier, bool wasSuccessful);
55     void didResolveRegistrationPromise();
56     void cancelJobsFromConnection(SWServerConnectionIdentifier);
57     void cancelJobsFromServiceWorker(ServiceWorkerIdentifier);
58 
59     bool isCurrentlyProcessingJob(const ServiceWorkerJobDataIdentifier&) const;
60 
61 private:
62     void jobTimerFired();
63     void runNextJobSynchronously();
64     void rejectCurrentJob(const ExceptionData&);
65     void finishCurrentJob();
66 
67     void runRegisterJob(const ServiceWorkerJobData&);
68     void runUnregisterJob(const ServiceWorkerJobData&);
69     void runUpdateJob(const ServiceWorkerJobData&);
70 
71     void install(SWServerRegistration&, ServiceWorkerIdentifier);
72 
73     void removeAllJobsMatching(const WTF::Function<bool(ServiceWorkerJobData&)>&);
74 
75     Deque<ServiceWorkerJobData> m_jobQueue;
76 
77     Timer m_jobTimer;
78     SWServer& m_server;
79     ServiceWorkerRegistrationKey m_registrationKey;
80 };
81 
82 } // namespace WebCore

84 #endif // ENABLE(SERVICE_WORKER)
