26 #pragma once
27 
28 #include "ResourceError.h"
29 #include "ResourceHandleClient.h"
30 #include "ResourceResponse.h"
31 #include <wtf/Function.h>
32 #include <wtf/MessageQueue.h>

33 
34 namespace WebCore {
35 














36 class SynchronousLoaderClient final : public ResourceHandleClient {
37 public:

38     virtual ~SynchronousLoaderClient();
39 
40     void setAllowStoredCredentials(bool allow) { m_allowStoredCredentials = allow; }
41     const ResourceResponse& response() const { return m_response; }
42     Vector<char>& mutableData() { return m_data; }
43     const ResourceError& error() const { return m_error; }
44     MessageQueue<Function<void()>>& messageQueue() { return m_messageQueue; }
45 
46     WEBCORE_EXPORT static ResourceError platformBadResponseError();
47 
48 private:
49     void willSendRequestAsync(ResourceHandle*, ResourceRequest&&, ResourceResponse&&, CompletionHandler<void(ResourceRequest&&)>&&) override;
50     bool shouldUseCredentialStorage(ResourceHandle*) override;
51     void didReceiveAuthenticationChallenge(ResourceHandle*, const AuthenticationChallenge&) override;
52     void didReceiveResponseAsync(ResourceHandle*, ResourceResponse&&, CompletionHandler<void()>&&) override;
53     void didReceiveData(ResourceHandle*, const char*, unsigned, int /*encodedDataLength*/) override;
54     void didFinishLoading(ResourceHandle*) override;
55     void didFail(ResourceHandle*, const ResourceError&) override;
56 #if USE(PROTECTION_SPACE_AUTH_CALLBACK)
57     void canAuthenticateAgainstProtectionSpaceAsync(ResourceHandle*, const ProtectionSpace&, CompletionHandler<void(bool)>&&) override;
58 #endif
59 
60     bool m_allowStoredCredentials { false };
61     ResourceResponse m_response;
62     Vector<char> m_data;
63     ResourceError m_error;
64     MessageQueue<Function<void()>> m_messageQueue;
65 };
66 }
