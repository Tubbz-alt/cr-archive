25 #include "config.h"
26 #include "StyleImageSet.h"
27 
28 #include "CSSImageSetValue.h"
29 
30 namespace WebCore {
31 
32 Ref<StyleImageSet> StyleImageSet::create(CSSImageSetValue& cssValue)
33 {
34     return adoptRef(*new StyleImageSet(cssValue));
35 }
36 
37 StyleImageSet::StyleImageSet(CSSImageSetValue& cssValue)
38     : m_cssValue(cssValue)
39 {
40     m_isImageSet = true;
41 }
42 
43 bool StyleImageSet::operator==(const StyleImage& other) const
44 {
45     return is<StyleImageSet>(other) && equals(downcast<StyleImageSet>(other));
46 }
47 
48 StyleImageSet::~StyleImageSet() = default;
49 
50 Ref<CSSValue> StyleImageSet::cssValue() const
51 {
52     return m_cssValue.copyRef();
53 }
54 
55 ImageWithScale StyleImageSet::selectBestFitImage(const Document& document) const
56 {
57     return m_cssValue->selectBestFitImage(document);
58 }
