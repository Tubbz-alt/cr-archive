 38 static constexpr unsigned FirstInternalAttribute = 1 << 6; // Use for transitions that don't have to do with property additions.
 39 
 40 // Support for attributes used to indicate transitions not related to properties.
 41 // If any of these are used, the string portion of the key should be 0.
 42 enum class NonPropertyTransition : unsigned {
 43     AllocateUndecided,
 44     AllocateInt32,
 45     AllocateDouble,
 46     AllocateContiguous,
 47     AllocateArrayStorage,
 48     AllocateSlowPutArrayStorage,
 49     SwitchToSlowPutArrayStorage,
 50     AddIndexedAccessors,
 51     PreventExtensions,
 52     Seal,
 53     Freeze
 54 };
 55 using TransitionPropertyAttributes = uint16_t;
 56 
 57 inline unsigned toAttributes(NonPropertyTransition transition)
 58 {
 59     return static_cast<unsigned>(transition) + FirstInternalAttribute;
 60 }
 61 
 62 inline bool changesIndexingType(NonPropertyTransition transition)
 63 {
 64     switch (transition) {
 65     case NonPropertyTransition::AllocateUndecided:
 66     case NonPropertyTransition::AllocateInt32:
 67     case NonPropertyTransition::AllocateDouble:
 68     case NonPropertyTransition::AllocateContiguous:
 69     case NonPropertyTransition::AllocateArrayStorage:
 70     case NonPropertyTransition::AllocateSlowPutArrayStorage:
 71     case NonPropertyTransition::SwitchToSlowPutArrayStorage:
 72     case NonPropertyTransition::AddIndexedAccessors:
 73         return true;
 74     default:
 75         return false;
 76     }
 77 }
 78 
 79 inline IndexingType newIndexingType(IndexingType oldType, NonPropertyTransition transition)
 80 {
 81     switch (transition) {
 82     case NonPropertyTransition::AllocateUndecided:
 83         ASSERT(!hasIndexedProperties(oldType));
 84         return oldType | UndecidedShape;
 85     case NonPropertyTransition::AllocateInt32:
 86         ASSERT(!hasIndexedProperties(oldType) || hasUndecided(oldType) || oldType == CopyOnWriteArrayWithInt32);
 87         return (oldType & ~IndexingShapeAndWritabilityMask) | Int32Shape;
 88     case NonPropertyTransition::AllocateDouble:
 89         ASSERT(!hasIndexedProperties(oldType) || hasUndecided(oldType) || hasInt32(oldType) || oldType == CopyOnWriteArrayWithDouble);
 90         return (oldType & ~IndexingShapeAndWritabilityMask) | DoubleShape;
 91     case NonPropertyTransition::AllocateContiguous:
 92         ASSERT(!hasIndexedProperties(oldType) || hasUndecided(oldType) || hasInt32(oldType) || hasDouble(oldType) || oldType == CopyOnWriteArrayWithContiguous);
 93         return (oldType & ~IndexingShapeAndWritabilityMask) | ContiguousShape;
 94     case NonPropertyTransition::AllocateArrayStorage:
 95         ASSERT(!hasIndexedProperties(oldType) || hasUndecided(oldType) || hasInt32(oldType) || hasDouble(oldType) || hasContiguous(oldType));
 96         return (oldType & ~IndexingShapeAndWritabilityMask) | ArrayStorageShape;
 97     case NonPropertyTransition::AllocateSlowPutArrayStorage:
 98         ASSERT(!hasIndexedProperties(oldType) || hasUndecided(oldType) || hasInt32(oldType) || hasDouble(oldType) || hasContiguous(oldType));
 99         return (oldType & ~IndexingShapeAndWritabilityMask) | SlowPutArrayStorageShape;
100     case NonPropertyTransition::SwitchToSlowPutArrayStorage:
101         ASSERT(hasArrayStorage(oldType));
102         return (oldType & ~IndexingShapeAndWritabilityMask) | SlowPutArrayStorageShape;
103     case NonPropertyTransition::AddIndexedAccessors:
104         return oldType | MayHaveIndexedAccessors;
105     default:
106         return oldType;
107     }
108 }
109 
110 inline bool preventsExtensions(NonPropertyTransition transition)
111 {
112     switch (transition) {
113     case NonPropertyTransition::PreventExtensions:
114     case NonPropertyTransition::Seal:
115     case NonPropertyTransition::Freeze:
116         return true;
117     default:
118         return false;
119     }
120 }
121 
122 inline bool setsDontDeleteOnAllProperties(NonPropertyTransition transition)
123 {
124     switch (transition) {
125     case NonPropertyTransition::Seal:
126     case NonPropertyTransition::Freeze:
127         return true;
128     default:
129         return false;
130     }
131 }
132 
133 inline bool setsReadOnlyOnNonAccessorProperties(NonPropertyTransition transition)
134 {
135     switch (transition) {
136     case NonPropertyTransition::Freeze:
137         return true;
138     default:
139         return false;
140     }
141 }

143 class StructureTransitionTable {
144     static constexpr intptr_t UsingSingleSlotFlag = 1;
145 
146 
147 #if CPU(ADDRESS64)
148     struct Hash {
149         // Logically, Key is a tuple of (1) UniquedStringImpl*, (2) unsigned attributes, and (3) bool isAddition.
150         // We encode (2) and (3) into (1)'s empty bits since a pointer is 48bit and lower 3 bits are usable because of alignment.
151         struct Key {
152             friend struct Hash;
153             static_assert(WTF_OS_CONSTANT_EFFECTIVE_ADDRESS_WIDTH <= 48);
154             static constexpr uintptr_t isAdditionMask = 1ULL;
155             static constexpr uintptr_t stringMask = ((1ULL << 48) - 1) & (~isAdditionMask);
156             static constexpr unsigned attributesShift = 48;
157             static constexpr uintptr_t hashTableDeletedValue = 0x2;
158             static_assert(sizeof(TransitionPropertyAttributes) * 8 <= 16);
159             static_assert(hashTableDeletedValue < alignof(UniquedStringImpl));
160 
161             // Highest 16 bits are for TransitionPropertyAttributes.
162             // Lowest 1 bit is for isAddition flag.
163             // Remaining bits are for UniquedStringImpl*.
164             Key(UniquedStringImpl* impl, unsigned attributes, bool isAddition)
165                 : m_encodedData(bitwise_cast<uintptr_t>(impl) | (static_cast<uintptr_t>(attributes) << attributesShift) | (static_cast<uintptr_t>(isAddition) & isAdditionMask))
166             {
167                 ASSERT(impl == this->impl());
168                 ASSERT(isAddition == this->isAddition());
169                 ASSERT(attributes == this->attributes());
170             }
171 
172             Key() = default;
173 
174             Key(WTF::HashTableDeletedValueType)
175                 : m_encodedData(hashTableDeletedValue)
176             { }
177 
178             bool isHashTableDeletedValue() const { return m_encodedData == hashTableDeletedValue; }
179 
180             UniquedStringImpl* impl() const { return bitwise_cast<UniquedStringImpl*>(m_encodedData & stringMask); }
181             bool isAddition() const { return m_encodedData & isAdditionMask; }
182             unsigned attributes() const { return m_encodedData >> attributesShift; }
183 
184             friend bool operator==(const Key& a, const Key& b)
185             {
186                 return a.m_encodedData == b.m_encodedData;
187             }
188 
189             friend bool operator!=(const Key& a, const Key& b)
190             {
191                 return a.m_encodedData != b.m_encodedData;
192             }
193 
194         private:
195             uintptr_t m_encodedData { 0 };
196         };
197         using KeyTraits = SimpleClassHashTraits<Key>;
198 
199         static unsigned hash(const Key& p)
200         {
201             return IntHash<uintptr_t>::hash(p.m_encodedData);
202         }
203 
204         static bool equal(const Key& a, const Key& b)
205         {
206             return a == b;
207         }
208 
209         static constexpr bool safeToCompareToEmptyOrDeleted = true;
210     };
211 #else
212     struct Hash {
213         using Key = std::tuple<UniquedStringImpl*, unsigned, bool>;
214         using KeyTraits = HashTraits<Key>;
215 
216         static unsigned hash(const Key& p)
217         {
218             return PtrHash<UniquedStringImpl*>::hash(std::get<0>(p)) + std::get<1>(p);
219         }
220 
221         static bool equal(const Key& a, const Key& b)
222         {
223             return a == b;
224         }
225 
226         static constexpr bool safeToCompareToEmptyOrDeleted = true;
227     };
228 #endif
229 
230     typedef WeakGCMap<Hash::Key, Structure, Hash, Hash::KeyTraits> TransitionMap;
231 
232 public:
233     StructureTransitionTable()
234         : m_data(UsingSingleSlotFlag)
235     {
236     }
237 
238     ~StructureTransitionTable()
239     {
240         if (!isUsingSingleSlot()) {
241             delete map();
242             return;
243         }
244 
245         WeakImpl* impl = this->weakImpl();
246         if (!impl)
247             return;
248         WeakSet::deallocate(impl);
249     }
250 
251     void add(VM&, Structure*);
252     bool contains(UniquedStringImpl*, unsigned attributes, bool isAddition) const;
253     Structure* get(UniquedStringImpl*, unsigned attributes, bool isAddition) const;
254 
255 private:
256     friend class SingleSlotTransitionWeakOwner;
257 
258     bool isUsingSingleSlot() const
259     {
260         return m_data & UsingSingleSlotFlag;
261     }
262 
263     TransitionMap* map() const
264     {
265         ASSERT(!isUsingSingleSlot());
266         return bitwise_cast<TransitionMap*>(m_data);
267     }
268 
269     WeakImpl* weakImpl() const
270     {
271         ASSERT(isUsingSingleSlot());
272         return bitwise_cast<WeakImpl*>(m_data & ~UsingSingleSlotFlag);
273     }
274 
275     void setMap(TransitionMap* map)
276     {
277         ASSERT(isUsingSingleSlot());
278 
279         if (WeakImpl* impl = this->weakImpl())
280             WeakSet::deallocate(impl);
281 
282         // This implicitly clears the flag that indicates we're using a single transition
283         m_data = bitwise_cast<intptr_t>(map);
284 
285         ASSERT(!isUsingSingleSlot());
286     }
287 
288     Structure* singleTransition() const;
289     void setSingleTransition(Structure*);
290 
291     intptr_t m_data;
292 };
