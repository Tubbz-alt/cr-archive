 37 #include "PlatformTouchEventIOS.h"

 39 
 40 namespace WebCore {
 41 
 42 class PointerEvent final : public MouseEvent {
 43     WTF_MAKE_ISO_ALLOCATED(PointerEvent);
 44 public:
 45     struct Init : MouseEventInit {
 46         PointerID pointerId { mousePointerID };
 47         double width { 1 };
 48         double height { 1 };
 49         float pressure { 0 };
 50         float tangentialPressure { 0 };
 51         long tiltX { 0 };
 52         long tiltY { 0 };
 53         long twist { 0 };
 54         String pointerType { PointerEvent::mousePointerType() };
 55         bool isPrimary { false };
 56     };
 57 
 58     enum class IsPrimary : uint8_t { No, Yes };
 59 
 60     static Ref<PointerEvent> create(const AtomString& type, Init&& initializer)
 61     {
 62         return adoptRef(*new PointerEvent(type, WTFMove(initializer)));
 63     }
 64 
 65     static Ref<PointerEvent> createForPointerCapture(const AtomString& type, PointerID pointerId, bool isPrimary, String pointerType)
 66     {
 67         Init initializer;
 68         initializer.bubbles = true;
 69         initializer.pointerId = pointerId;
 70         initializer.isPrimary = isPrimary;
 71         initializer.pointerType = pointerType;
 72         return adoptRef(*new PointerEvent(type, WTFMove(initializer)));
 73     }
 74 
 75     static Ref<PointerEvent> createForBindings()
 76     {
 77         return adoptRef(*new PointerEvent);
 78     }
 79 
 80     static RefPtr<PointerEvent> create(short button, const MouseEvent&);
 81     static Ref<PointerEvent> create(const String& type, short button, const MouseEvent&);
 82     static Ref<PointerEvent> create(const String& type, PointerID, const String& pointerType, IsPrimary = IsPrimary::No);
 83 
 84 #if ENABLE(TOUCH_EVENTS) && PLATFORM(IOS_FAMILY)
 85     static Ref<PointerEvent> create(const PlatformTouchEvent&, unsigned touchIndex, bool isPrimary, Ref<WindowProxy>&&);
 86     static Ref<PointerEvent> create(const String& type, const PlatformTouchEvent&, unsigned touchIndex, bool isPrimary, Ref<WindowProxy>&&);
 87 #endif
 88 
 89     static const String& mousePointerType();
 90     static const String& penPointerType();
 91     static const String& touchPointerType();
 92 
 93     virtual ~PointerEvent();
 94 
 95     PointerID pointerId() const { return m_pointerId; }
 96     double width() const { return m_width; }
 97     double height() const { return m_height; }
 98     float pressure() const { return m_pressure; }
 99     float tangentialPressure() const { return m_tangentialPressure; }
100     long tiltX() const { return m_tiltX; }
101     long tiltY() const { return m_tiltY; }
102     long twist() const { return m_twist; }
103     String pointerType() const { return m_pointerType; }
104     bool isPrimary() const { return m_isPrimary; }
105 
106     bool isPointerEvent() const final { return true; }
107 
108     // https://w3c.github.io/pointerevents/#attributes-and-default-actions
109     // Many user agents expose non-standard attributes fromElement and toElement in MouseEvents to
110     // support legacy content. In those user agents, the values of those (inherited) attributes in
111     // PointerEvents must be null to encourage the use of the standardized alternates (i.e. target
112     // and relatedTarget).
113     RefPtr<Node> toElement() const final { return nullptr; }
114     RefPtr<Node> fromElement() const final { return nullptr; }
115 
116     EventInterface eventInterface() const override;
117 
118 private:
119     static bool typeIsEnterOrLeave(const AtomString& type) { return type == eventNames().pointerenterEvent || type == eventNames().pointerleaveEvent; }
120     static CanBubble typeCanBubble(const AtomString& type) { return typeIsEnterOrLeave(type) ? CanBubble::No : CanBubble::Yes; }
121     static IsCancelable typeIsCancelable(const AtomString& type) { return typeIsEnterOrLeave(type) ? IsCancelable::No : IsCancelable::Yes; }
122     static IsComposed typeIsComposed(const AtomString& type) { return typeIsEnterOrLeave(type) ? IsComposed::No : IsComposed::Yes; }
123 
124     PointerEvent();
125     PointerEvent(const AtomString&, Init&&);
126     PointerEvent(const AtomString& type, short button, const MouseEvent&);
127     PointerEvent(const AtomString& type, PointerID, const String& pointerType, IsPrimary);
128 #if ENABLE(TOUCH_EVENTS) && PLATFORM(IOS_FAMILY)
129     PointerEvent(const AtomString& type, const PlatformTouchEvent&, IsCancelable isCancelable, unsigned touchIndex, bool isPrimary, Ref<WindowProxy>&&);
130 #endif
131 
132     PointerID m_pointerId { mousePointerID };
133     double m_width { 1 };
134     double m_height { 1 };
135     float m_pressure { 0 };
136     float m_tangentialPressure { 0 };
137     long m_tiltX { 0 };
138     long m_tiltY { 0 };
139     long m_twist { 0 };
140     String m_pointerType { PointerEvent::mousePointerType() };
141     bool m_isPrimary { false };
142 };
143 
144 } // namespace WebCore
145 
146 SPECIALIZE_TYPE_TRAITS_EVENT(PointerEvent)

148 #endif // ENABLE(POINTER_EVENTS)
