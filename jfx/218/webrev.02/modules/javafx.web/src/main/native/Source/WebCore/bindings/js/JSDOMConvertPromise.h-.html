34 template<typename T> struct Converter<IDLPromise<T>> : DefaultConverter<IDLPromise<T>> {
35     using ReturnType = RefPtr<DOMPromise>;
36 
37     // https://heycam.github.io/webidl/#es-promise
38     template<typename ExceptionThrower = DefaultExceptionThrower>
39     static ReturnType convert(JSC::ExecState& state, JSC::JSValue value, ExceptionThrower&& exceptionThrower = ExceptionThrower())
40     {
41         JSC::VM& vm = state.vm();
42         auto scope = DECLARE_THROW_SCOPE(vm);
43         auto* globalObject = JSC::jsDynamicCast<JSDOMGlobalObject*>(vm, state.lexicalGlobalObject());
44         if (!globalObject)
45             return nullptr;
46 
47         // 1. Let resolve be the original value of %Promise%.resolve.
48         // 2. Let promise be the result of calling resolve with %Promise% as the this value and V as the single argument value.
49         auto* promise = JSC::JSPromise::resolve(*globalObject, value);
50         if (scope.exception()) {
51             exceptionThrower(state, scope);
52             return nullptr;
53         }
54         ASSERT(promise);
55 
56         // 3. Return the IDL promise type value that is a reference to the same object as promise.
57         return DOMPromise::create(*globalObject, *promise);
58     }
59 };
60 
61 template<typename T> struct JSConverter<IDLPromise<T>> {
62     static constexpr bool needsState = true;
63     static constexpr bool needsGlobalObject = true;
64 
65     static JSC::JSValue convert(JSC::ExecState&, JSDOMGlobalObject&, DOMPromise& promise)
66     {
67         return promise.promise();
68     }
69 
70     template<template<typename> class U>
71     static JSC::JSValue convert(JSC::ExecState& state, JSDOMGlobalObject& globalObject, U<T>& promiseProxy)
72     {
73         return promiseProxy.promise(state, globalObject);
74     }
75 };
