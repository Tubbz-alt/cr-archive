 28 #if ENABLE(INTL)
 29 
 30 #include "JSObject.h"
 31 #include <unicode/udat.h>
 32 #include <unicode/uvernum.h>
 33 
 34 #define JSC_ICU_HAS_UFIELDPOSITER (U_ICU_VERSION_MAJOR_NUM >= 55)
 35 
 36 namespace JSC {
 37 
 38 class IntlDateTimeFormatConstructor;
 39 class JSBoundFunction;
 40 
 41 class IntlDateTimeFormat final : public JSNonFinalObject {
 42 public:
 43     using Base = JSNonFinalObject;
 44 
 45     static constexpr bool needsDestruction = true;
 46 
 47     static void destroy(JSCell* cell)
 48     {
 49         static_cast<IntlDateTimeFormat*>(cell)->IntlDateTimeFormat::~IntlDateTimeFormat();
 50     }
 51 
 52     template<typename CellType, SubspaceAccess mode>
 53     static IsoSubspace* subspaceFor(VM& vm)
 54     {
 55         return vm.intlDateTimeFormatSpace<mode>();
 56     }
 57 
 58     static IntlDateTimeFormat* create(VM&, Structure*);
 59     static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
 60 
 61     DECLARE_INFO;
 62 
 63     void initializeDateTimeFormat(JSGlobalObject*, JSValue locales, JSValue options);
 64     JSValue format(JSGlobalObject*, double value);
 65 #if JSC_ICU_HAS_UFIELDPOSITER
 66     JSValue formatToParts(JSGlobalObject*, double value);
 67 #endif
 68     JSObject* resolvedOptions(JSGlobalObject*);
 69 
 70     JSBoundFunction* boundFormat() const { return m_boundFormat.get(); }
 71     void setBoundFormat(VM&, JSBoundFunction*);
 72 
 73 protected:
 74     IntlDateTimeFormat(VM&, Structure*);
 75     void finishCreation(VM&);

 76     static void visitChildren(JSCell*, SlotVisitor&);
 77 
 78 private:
 79     enum class Weekday : uint8_t { None, Narrow, Short, Long };
 80     enum class Era : uint8_t { None, Narrow, Short, Long };
 81     enum class Year : uint8_t { None, TwoDigit, Numeric };
 82     enum class Month : uint8_t { None, TwoDigit, Numeric, Narrow, Short, Long };
 83     enum class Day : uint8_t { None, TwoDigit, Numeric };
 84     enum class Hour : uint8_t { None, TwoDigit, Numeric };
 85     enum class Minute : uint8_t { None, TwoDigit, Numeric };
 86     enum class Second : uint8_t { None, TwoDigit, Numeric };
 87     enum class TimeZoneName : uint8_t { None, Short, Long };
 88 
 89     struct UDateFormatDeleter {
 90         void operator()(UDateFormat*) const;
 91     };
 92 
 93     void setFormatsFromPattern(const StringView&);
 94     static ASCIILiteral weekdayString(Weekday);
 95     static ASCIILiteral eraString(Era);
 96     static ASCIILiteral yearString(Year);
 97     static ASCIILiteral monthString(Month);
 98     static ASCIILiteral dayString(Day);
 99     static ASCIILiteral hourString(Hour);
100     static ASCIILiteral minuteString(Minute);
101     static ASCIILiteral secondString(Second);
102     static ASCIILiteral timeZoneNameString(TimeZoneName);
103 
104     WriteBarrier<JSBoundFunction> m_boundFormat;
105     std::unique_ptr<UDateFormat, UDateFormatDeleter> m_dateFormat;
106 
107     String m_locale;
108     String m_calendar;
109     String m_numberingSystem;
110     String m_timeZone;
111     String m_hourCycle;
112     Weekday m_weekday { Weekday::None };
113     Era m_era { Era::None };
114     Year m_year { Year::None };
115     Month m_month { Month::None };
116     Day m_day { Day::None };
117     Hour m_hour { Hour::None };
118     Minute m_minute { Minute::None };
119     Second m_second { Second::None };
120     TimeZoneName m_timeZoneName { TimeZoneName::None };
121     bool m_initializedDateTimeFormat { false };
122 
123 #if JSC_ICU_HAS_UFIELDPOSITER
124     struct UFieldPositionIteratorDeleter {
125         void operator()(UFieldPositionIterator*) const;
126     };
127 
128     static ASCIILiteral partTypeString(UDateFormatField);
129 #endif
130 };
131 
132 } // namespace JSC
133 
134 #endif // ENABLE(INTL)
