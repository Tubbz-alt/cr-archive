 40 BytecodeLivenessAnalysis::BytecodeLivenessAnalysis(CodeBlock* codeBlock)
 41     : m_graph(codeBlock, codeBlock->instructions())
 42 {
 43     runLivenessFixpoint(codeBlock, codeBlock->instructions(), m_graph);
 44 
 45     if (UNLIKELY(Options::dumpBytecodeLivenessResults()))
 46         dumpResults(codeBlock);
 47 }

 49 void BytecodeLivenessAnalysis::getLivenessInfoAtBytecodeIndex(CodeBlock* codeBlock, BytecodeIndex bytecodeIndex, FastBitVector& result)
 50 {
 51     BytecodeBasicBlock* block = m_graph.findBasicBlockForBytecodeOffset(bytecodeIndex.offset());
 52     ASSERT(block);
 53     ASSERT(!block->isEntryBlock());
 54     ASSERT(!block->isExitBlock());
 55     result.resize(block->out().numBits());
 56     computeLocalLivenessForBytecodeIndex(codeBlock, codeBlock->instructions(), m_graph, *block, bytecodeIndex, result);
 57 }

 59 FastBitVector BytecodeLivenessAnalysis::getLivenessInfoAtBytecodeIndex(CodeBlock* codeBlock, BytecodeIndex bytecodeIndex)
 60 {
 61     FastBitVector out;
 62     getLivenessInfoAtBytecodeIndex(codeBlock, bytecodeIndex, out);
 63     return out;

 66 void BytecodeLivenessAnalysis::computeFullLiveness(CodeBlock* codeBlock, FullBytecodeLiveness& result)
 67 {
 68     FastBitVector out;
 69 
 70     result.m_beforeUseVector.resize(codeBlock->instructions().size());
 71     result.m_afterUseVector.resize(codeBlock->instructions().size());
 72 
 73     for (BytecodeBasicBlock& block : m_graph.basicBlocksInReverseOrder()) {
 74         if (block.isEntryBlock() || block.isExitBlock())
 75             continue;
 76 
 77         out = block.out();
 78 
 79         auto use = [&] (unsigned bitIndex) {
 80             // This is the use functor, so we set the bit.
 81             out[bitIndex] = true;
 82         };
 83 
 84         auto def = [&] (unsigned bitIndex) {
 85             // This is the def functor, so we clear the bit.
 86             out[bitIndex] = false;
 87         };
 88 
 89         auto& instructions = codeBlock->instructions();
 90         unsigned cursor = block.totalLength();
 91         for (unsigned i = block.delta().size(); i--;) {
 92             cursor -= block.delta()[i];
 93             BytecodeIndex bytecodeIndex = BytecodeIndex(block.leaderOffset() + cursor);
 94 
 95             stepOverInstructionDef(codeBlock, instructions, m_graph, bytecodeIndex, def);
 96             stepOverInstructionUseInExceptionHandler(codeBlock, instructions, m_graph, bytecodeIndex, use);
 97             result.m_afterUseVector[bytecodeIndex.offset()] = out; // AfterUse point.
 98             stepOverInstructionUse(codeBlock, instructions, m_graph, bytecodeIndex, use);
 99             result.m_beforeUseVector[bytecodeIndex.offset()] = out; // BeforeUse point.
100         }
101     }
102 }

104 void BytecodeLivenessAnalysis::computeKills(CodeBlock* codeBlock, BytecodeKills& result)
105 {
106     UNUSED_PARAM(result);
107     FastBitVector out;
108 
109     result.m_codeBlock = codeBlock;
110     result.m_killSets = makeUniqueArray<BytecodeKills::KillSet>(codeBlock->instructions().size());
111 
112     for (BytecodeBasicBlock& block : m_graph.basicBlocksInReverseOrder()) {
113         if (block.isEntryBlock() || block.isExitBlock())
114             continue;
115 
116         out = block.out();
117 
118         unsigned cursor = block.totalLength();
119         for (unsigned i = block.delta().size(); i--;) {
120             cursor -= block.delta()[i];
121             BytecodeIndex bytecodeIndex = BytecodeIndex(block.leaderOffset() + cursor);
122             stepOverInstruction(
123                 codeBlock, codeBlock->instructions(), m_graph, bytecodeIndex,
124                 [&] (unsigned index) {
125                     // This is for uses.
126                     if (out[index])
127                         return;
128                     result.m_killSets[bytecodeIndex.offset()].add(index);
129                     out[index] = true;
130                 },
131                 [&] (unsigned index) {
132                     // This is for defs.
133                     out[index] = false;
134                 });
135         }
136     }
137 }

139 void BytecodeLivenessAnalysis::dumpResults(CodeBlock* codeBlock)
140 {
141     dataLog("\nDumping bytecode liveness for ", *codeBlock, ":\n");
142     const auto& instructions = codeBlock->instructions();
143     unsigned i = 0;
144 
145     unsigned numberOfBlocks = m_graph.size();
146     Vector<FastBitVector> predecessors(numberOfBlocks);
147     for (BytecodeBasicBlock& block : m_graph)
148         predecessors[block.index()].resize(numberOfBlocks);
149     for (BytecodeBasicBlock& block : m_graph) {
150         for (unsigned successorIndex : block.successors()) {
151             unsigned blockIndex = block.index();
152             predecessors[successorIndex][blockIndex] = true;
153         }
154     }
155 
156     auto dumpBitVector = [] (FastBitVector& bits) {
157         for (unsigned j = 0; j < bits.numBits(); j++) {
158             if (bits[j])
159                 dataLogF(" %u", j);
160         }
161     };
162 
163     for (BytecodeBasicBlock& block : m_graph) {
164         dataLogF("\nBytecode basic block %u: %p (offset: %u, length: %u)\n", i++, &block, block.leaderOffset(), block.totalLength());
165 
166         dataLogF("Predecessors:");
167         dumpBitVector(predecessors[block.index()]);
168         dataLogF("\n");
169 
170         dataLogF("Successors:");
171         FastBitVector successors;
172         successors.resize(numberOfBlocks);
173         for (unsigned successorIndex : block.successors())
174             successors[successorIndex] = true;
175         dumpBitVector(successors); // Dump in sorted order.
176         dataLogF("\n");
177 
178         if (block.isEntryBlock()) {
179             dataLogF("Entry block %p\n", &block);
180             continue;
181         }
182         if (block.isExitBlock()) {
183             dataLogF("Exit block: %p\n", &block);
184             continue;
185         }
186         for (unsigned bytecodeOffset = block.leaderOffset(); bytecodeOffset < block.leaderOffset() + block.totalLength();) {
187             const auto currentInstruction = instructions.at(bytecodeOffset);
188 
189             dataLogF("Live variables:");
190             FastBitVector liveBefore = getLivenessInfoAtBytecodeIndex(codeBlock, BytecodeIndex(bytecodeOffset));
191             dumpBitVector(liveBefore);
192             dataLogF("\n");
193             codeBlock->dumpBytecode(WTF::dataFile(), currentInstruction);
194 
195             bytecodeOffset += currentInstruction->size();
196         }
197 
198         dataLogF("Live variables:");
199         FastBitVector liveAfter = block.out();
200         dumpBitVector(liveAfter);
201         dataLogF("\n");
202     }
203 }

205 template<typename EnumType1, typename EnumType2>
206 constexpr bool enumValuesEqualAsIntegral(EnumType1 v1, EnumType2 v2)
207 {
208     using IntType1 = typename std::underlying_type<EnumType1>::type;
209     using IntType2 = typename std::underlying_type<EnumType2>::type;
210     if constexpr (sizeof(IntType1) > sizeof(IntType2))
211         return static_cast<IntType1>(v1) == static_cast<IntType1>(v2);
212     else
213         return static_cast<IntType2>(v1) == static_cast<IntType2>(v2);
214 }

216 Bitmap<maxNumCheckpointTmps> tmpLivenessForCheckpoint(const CodeBlock& codeBlock, BytecodeIndex bytecodeIndex)
217 {
218     Bitmap<maxNumCheckpointTmps> result;
219     uint8_t checkpoint = bytecodeIndex.checkpoint();
220 
221     if (!checkpoint)
222         return result;
223 
224     switch (codeBlock.instructions().at(bytecodeIndex)->opcodeID()) {
225     case op_call_varargs:
226     case op_tail_call_varargs:
227     case op_construct_varargs: {
228         static_assert(enumValuesEqualAsIntegral(OpCallVarargs::makeCall, OpTailCallVarargs::makeCall) && enumValuesEqualAsIntegral(OpCallVarargs::argCountIncludingThis, OpTailCallVarargs::argCountIncludingThis));
229         static_assert(enumValuesEqualAsIntegral(OpCallVarargs::makeCall, OpConstructVarargs::makeCall) && enumValuesEqualAsIntegral(OpCallVarargs::argCountIncludingThis, OpConstructVarargs::argCountIncludingThis));
230         if (checkpoint == OpCallVarargs::makeCall)
231             result.set(OpCallVarargs::argCountIncludingThis);
232         return result;
233     }
234     default:
235         break;
236     }
237     RELEASE_ASSERT_NOT_REACHED();
238 }
