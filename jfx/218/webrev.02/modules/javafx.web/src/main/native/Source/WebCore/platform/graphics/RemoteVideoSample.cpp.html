 49 #if HAVE(IOSURFACE)
 50 static inline std::unique_ptr<IOSurface> transferBGRAPixelBufferToIOSurface(CVPixelBufferRef pixelBuffer)
 51 {
 52 #if USE(ACCELERATE)
 53     ASSERT(CVPixelBufferGetPixelFormatType(pixelBuffer) == kCVPixelFormatType_32BGRA);
 54 
 55     auto result = CVPixelBufferLockBaseAddress(pixelBuffer, kCVPixelBufferLock_ReadOnly);
 56     ASSERT(result == kCVReturnSuccess);
 57     if (result != kCVReturnSuccess) {
 58         RELEASE_LOG_ERROR(Media, "transferBGRAPixelBufferToIOSurface CVPixelBufferLockBaseAddress() returned error code %d", result);
 59         return nullptr;
 60     }
 61 
 62     IntSize size { static_cast<int>(CVPixelBufferGetWidth(pixelBuffer)), static_cast<int>(CVPixelBufferGetHeight(pixelBuffer)) };
 63     auto ioSurface =  IOSurface::create(size, sRGBColorSpaceRef(), IOSurface::Format::RGBA);
 64 
 65     IOSurface::Locker lock(*ioSurface);
 66     vImage_Buffer src;
 67     src.width = size.width();
 68     src.height = size.height();
 69     src.rowBytes = CVPixelBufferGetBytesPerRow(pixelBuffer);
 70     src.data = CVPixelBufferGetBaseAddress(pixelBuffer);
 71 
 72     vImage_Buffer dest;
 73     dest.width = size.width();
 74     dest.height = size.height();
 75     dest.rowBytes = ioSurface->bytesPerRow();
 76     dest.data = lock.surfaceBaseAddress();
 77 
 78     vImageUnpremultiplyData_BGRA8888(&src, &dest, kvImageNoFlags);
 79 
 80     result = CVPixelBufferUnlockBaseAddress(pixelBuffer, kCVPixelBufferLock_ReadOnly);
 81     ASSERT(result == kCVReturnSuccess);
 82     if (result != kCVReturnSuccess) {
 83         RELEASE_LOG_ERROR(Media, "transferBGRAPixelBufferToIOSurface CVPixelBufferUnlockBaseAddress() returned error code %d", result);
 84         return nullptr;
 85     }
 86     return ioSurface;
 87 #else
 88     RELEASE_LOG_ERROR(Media, "transferBGRAPixelBufferToIOSurface cannot convert to IOSurface");
 89     return nullptr;
 90 #endif
 91 }

 93 std::unique_ptr<RemoteVideoSample> RemoteVideoSample::create(MediaSample& sample)
 94 {
 95     ASSERT(sample.platformSample().type == PlatformSample::CMSampleBufferType);
 96 
 97     auto imageBuffer = CMSampleBufferGetImageBuffer(sample.platformSample().sample.cmSampleBuffer);
 98     if (!imageBuffer) {
 99         RELEASE_LOG_ERROR(Media, "RemoteVideoSample::create: CMSampleBufferGetImageBuffer returned nullptr");
100         return nullptr;
101     }
102 
103     std::unique_ptr<IOSurface> ioSurface;
104     auto surface = CVPixelBufferGetIOSurface(imageBuffer);
105     if (!surface) {
106         // Special case for canvas data that is RGBA, not IOSurface backed.
107         auto pixelFormatType = CVPixelBufferGetPixelFormatType(imageBuffer);
108         if (pixelFormatType != kCVPixelFormatType_32BGRA) {
109             RELEASE_LOG_ERROR(Media, "RemoteVideoSample::create does not support non IOSurface backed samples that are not BGRA");
110             return nullptr;
111         }
112 
113         ioSurface = transferBGRAPixelBufferToIOSurface(imageBuffer);
114         if (!ioSurface)
115             return nullptr;
116 
117         surface = ioSurface->surface();
118     }
119 
120     return std::unique_ptr<RemoteVideoSample>(new RemoteVideoSample(surface, sRGBColorSpaceRef(), sample.presentationTime(), sample.videoRotation(), sample.videoMirrored()));
121 }
122 
123 std::unique_ptr<RemoteVideoSample> RemoteVideoSample::create(CVPixelBufferRef imageBuffer, MediaTime&& presentationTime, MediaSample::VideoRotation rotation)
124 {
125     auto surface = CVPixelBufferGetIOSurface(imageBuffer);
126     if (!surface) {
127         RELEASE_LOG_ERROR(Media, "RemoteVideoSample::create: CVPixelBufferGetIOSurface returned nullptr");
128         return nullptr;
129     }
130 
131     return std::unique_ptr<RemoteVideoSample>(new RemoteVideoSample(surface, sRGBColorSpaceRef(), WTFMove(presentationTime), rotation, false));
132 }
133 
134 RemoteVideoSample::RemoteVideoSample(IOSurfaceRef surface, CGColorSpaceRef colorSpace, MediaTime&& time, MediaSample::VideoRotation rotation, bool mirrored)
135     : m_ioSurface(WebCore::IOSurface::createFromSurface(surface, colorSpace))
136     , m_rotation(rotation)
137     , m_time(WTFMove(time))
138     , m_videoFormat(IOSurfaceGetPixelFormat(surface))
139     , m_size(IntSize(IOSurfaceGetWidth(surface), IOSurfaceGetHeight(surface)))
140     , m_mirrored(mirrored)
141 {
142 }
143 
144 IOSurfaceRef RemoteVideoSample::surface()
145 {
146     if (!m_ioSurface && m_sendRight)
147         m_ioSurface = WebCore::IOSurface::createFromSendRight(WTFMove(m_sendRight), sRGBColorSpaceRef());
148 
149     return m_ioSurface ? m_ioSurface->surface() : nullptr;
150 }
