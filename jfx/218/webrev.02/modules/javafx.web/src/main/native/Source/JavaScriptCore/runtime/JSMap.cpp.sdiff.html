48 bool JSMap::isIteratorProtocolFastAndNonObservable()
49 {
50     JSGlobalObject* globalObject = this->globalObject();
51     if (!globalObject->isMapPrototypeIteratorProtocolFastAndNonObservable())
52         return false;
53 
54     VM& vm = globalObject->vm();
55     Structure* structure = this->structure(vm);
56     // This is the fast case. Many maps will be an original map.
57     if (structure == globalObject->mapStructure())
58         return true;
59 
60     if (getPrototypeDirect(vm) != globalObject->mapPrototype())
61         return false;
62 
63     if (getDirectOffset(vm, vm.propertyNames->iteratorSymbol) != invalidOffset)
64         return false;
