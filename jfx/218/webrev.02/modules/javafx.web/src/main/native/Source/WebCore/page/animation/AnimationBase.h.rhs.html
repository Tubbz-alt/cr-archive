 46 enum class AnimateChange {
 47     StyleBlended            = 1 << 0, // Style was changed.
 48     StateChange             = 1 << 1, // Animation state() changed.
 49     RunningStateChange      = 1 << 2, // Animation "running or paused" changed.
 50 };
 51 
 52 enum class AnimationImpact {
 53     RequiresRecomposite     = 1 << 0,
 54     ForcesStackingContext   = 1 << 1
 55 };

 57 class AnimationBase : public RefCounted<AnimationBase>
 58     , public CSSPropertyBlendingClient {
 59     friend class CompositeAnimation;
 60     friend class CSSPropertyAnimation;
 61     WTF_MAKE_FAST_ALLOCATED;
 62 public:
 63     AnimationBase(const Animation& transition, Element&, CompositeAnimation&);
 64     virtual ~AnimationBase();
 65 
 66     Element* element() const { return m_element.get(); }
 67     const RenderStyle& currentStyle() const override;
 68     RenderElement* renderer() const override;
 69     virtual void clear();
 70 
 71     double duration() const;
 72 
 73     // Animations and Transitions go through the states below. When entering the STARTED state
 74     // the animation is started. This may or may not require deferred response from the animator.
 75     // If so, we stay in this state until that response is received (and it returns the start time).
 76     // Otherwise, we use the current time as the start time and go immediately to AnimationState::Looping
 77     // or AnimationState::Ending.
 78     enum class AnimationState : uint8_t {
 79         New,                        // animation just created, animation not running yet
 80         StartWaitTimer,             // start timer running, waiting for fire
 81         StartWaitStyleAvailable,    // waiting for style setup so we can start animations
 82         StartWaitResponse,          // animation started, waiting for response
 83         Looping,                    // response received, animation running, loop timer running, waiting for fire
 84         Ending,                     // received, animation running, end timer running, waiting for fire
 85         PausedNew,                  // in pause mode when animation was created
 86         PausedWaitTimer,            // in pause mode when animation started
 87         PausedWaitStyleAvailable,   // in pause mode when waiting for style setup
 88         PausedWaitResponse,         // animation paused when in STARTING state
 89         PausedRun,                  // animation paused when in LOOPING or ENDING state
 90         Done,                       // end timer fired, animation finished and removed
 91         FillingForwards             // animation has ended and is retaining its final value
 92     };
 93 
 94     enum class AnimationStateInput : uint8_t {
 95         MakeNew,           // reset back to new from any state
 96         StartAnimation,    // animation requests a start
 97         RestartAnimation,  // force a restart from any state
 98         StartTimerFired,   // start timer fired
 99         StyleAvailable,    // style is setup, ready to start animating
100         StartTimeSet,      // m_startTime was set
101         LoopTimerFired,    // loop timer fired
102         EndTimerFired,     // end timer fired
103         PauseOverride,     // pause an animation due to override
104         ResumeOverride,    // resume an overridden animation
105         PlayStateRunning,  // play state paused -> running
106         PlayStatePaused,   // play state running -> paused
107         EndAnimation       // force an end from any state
108     };
109 
110     // Called when animation is in AnimationState::New to start animation
111     void updateStateMachine(AnimationStateInput, double param);
112 
113     // Animation has actually started, at passed time
114     void onAnimationStartResponse(MonotonicTime startTime)
115     {
116         updateStateMachine(AnimationStateInput::StartTimeSet, startTime.secondsSinceEpoch().seconds());
117     }
118 
119     // Called to change to or from paused state
120     void updatePlayState(AnimationPlayState);
121     bool playStatePlaying() const;
122 
123     bool waitingToStart() const { return m_animationState == AnimationState::New || m_animationState == AnimationState::StartWaitTimer || m_animationState == AnimationState::PausedNew; }
124     bool preActive() const
125     {
126         return m_animationState == AnimationState::New || m_animationState == AnimationState::StartWaitTimer || m_animationState == AnimationState::StartWaitStyleAvailable || m_animationState == AnimationState::StartWaitResponse;
127     }
128 
129     bool postActive() const { return m_animationState == AnimationState::Done; }
130     bool fillingForwards() const { return m_animationState == AnimationState::FillingForwards; }
131     bool active() const { return !postActive() && !preActive(); }
132     bool running() const { return !isNew() && !postActive(); }
133     bool paused() const { return m_pauseTime || m_animationState == AnimationState::PausedNew; }
134 
135     static bool isPausedState(AnimationState state) { return state >= AnimationState::PausedNew && state <= AnimationState::PausedRun; }
136     static bool isRunningState(AnimationState state) { return state >= AnimationState::StartWaitStyleAvailable && state < AnimationState::Done; }
137 
138     bool inPausedState() const { return isPausedState(m_animationState); }
139     bool inRunningState() const { return isRunningState(m_animationState); }
140 
141     bool isNew() const { return m_animationState == AnimationState::New || m_animationState == AnimationState::PausedNew; }
142     bool waitingForStartTime() const { return m_animationState == AnimationState::StartWaitResponse; }
143     bool waitingForStyleAvailable() const { return m_animationState == AnimationState::StartWaitStyleAvailable; }
144 
145     bool isAccelerated() const { return m_isAccelerated; }
146 
147     virtual Optional<Seconds> timeToNextService();
148 
149     double progress(double scale = 1, double offset = 0, const TimingFunction* = nullptr) const;
150 
151     virtual void getAnimatedStyle(std::unique_ptr<RenderStyle>& /*animatedStyle*/) = 0;
152 
153     virtual bool computeExtentOfTransformAnimation(LayoutRect&) const = 0;
154 
155     virtual bool shouldFireEvents() const { return false; }
156 
157     void fireAnimationEventsIfNeeded();
158 
159     bool animationsMatch(const Animation&) const;
160 
161     const Animation& animation() const { return m_animation; }
162     void setAnimation(const Animation& animation) { m_animation = const_cast<Animation&>(animation); }
163 
164     // Return true if this animation is overridden. This will only be the case for
165     // ImplicitAnimations and is used to determine whether or not we should force
166     // set the start time. If an animation is overridden, it will probably not get
167     // back the AnimationStateInput::StartTimeSet input.
168     virtual bool overridden() const { return false; }
169 
170     // Does this animation/transition involve the given property?
171     virtual bool affectsProperty(CSSPropertyID /*property*/) const { return false; }
172 
173     bool isAnimatingProperty(CSSPropertyID property, bool acceleratedOnly) const
174     {
175         if (acceleratedOnly && !m_isAccelerated)
176             return false;
177 
178         if (!affectsProperty(property))
179             return false;
180 
181         return inRunningState() || inPausedState();
182     }
183 
184     bool transformFunctionListsMatch() const override { return m_transformFunctionListsMatch; }
185     bool filterFunctionListsMatch() const override { return m_filterFunctionListsMatch; }
186 #if ENABLE(FILTERS_LEVEL_2)
187     bool backdropFilterFunctionListsMatch() const override { return m_backdropFilterFunctionListsMatch; }
188 #endif
189     bool colorFilterFunctionListsMatch() const override { return m_colorFilterFunctionListsMatch; }
190 
191     // Freeze the animation; used by DumpRenderTree.
192     void freezeAtTime(double t);
193 
194     // Play and pause API
195     void play();
196     void pause();
197 
198     double beginAnimationUpdateTime() const;
199 
200     double getElapsedTime() const;
201     // Setting the elapsed time will adjust the start time and possibly pause time.
202     void setElapsedTime(double);
203 
204     void styleAvailable()
205     {
206         ASSERT(waitingForStyleAvailable());
207         updateStateMachine(AnimationStateInput::StyleAvailable, -1);
208     }
209 
210 protected:
211     virtual void overrideAnimations() { }
212     virtual void resumeOverriddenAnimations() { }
213 
214     CompositeAnimation* compositeAnimation() { return m_compositeAnimation; }
215 
216     // These are called when the corresponding timer fires so subclasses can do any extra work
217     virtual void onAnimationStart(double /*elapsedTime*/) { }
218     virtual void onAnimationIteration(double /*elapsedTime*/) { }
219     virtual void onAnimationEnd(double /*elapsedTime*/) { }
220 
221     // timeOffset is an offset from the current time when the animation should start. Negative values are OK.
222     // Return value indicates whether to expect an asynchronous notifyAnimationStarted() callback.
223     virtual bool startAnimation(double /*timeOffset*/) { return false; }
224     // timeOffset is the time at which the animation is being paused.
225     virtual void pauseAnimation(double /*timeOffset*/) { }
226     virtual void endAnimation(bool /*fillingForwards*/ = false) { }
227 
228     virtual const RenderStyle& unanimatedStyle() const = 0;
229 
230     void goIntoEndingOrLoopingState();
231 
232     AnimationState state() const { return m_animationState; }
233 
234     static void setNeedsStyleRecalc(Element*);
235 
236     void getTimeToNextEvent(Seconds& time, bool& isLooping) const;
237 
238     double fractionalTime(double scale, double elapsedTime, double offset) const;
239 
240     // These return true if we can easily compute a bounding box by applying the style's transform to the bounds rect.
241     bool computeTransformedExtentViaTransformList(const FloatRect& rendererBox, const RenderStyle&, LayoutRect& bounds) const;
242     bool computeTransformedExtentViaMatrix(const FloatRect& rendererBox, const RenderStyle&, LayoutRect& bounds) const;
243 
244 protected:
245     bool m_isAccelerated { false };
246     bool m_transformFunctionListsMatch { false };
247     bool m_filterFunctionListsMatch { false };
248 #if ENABLE(FILTERS_LEVEL_2)
249     bool m_backdropFilterFunctionListsMatch { false };
250 #endif
251 
252 private:
253     RefPtr<Element> m_element;
254 
255 protected:
256     CompositeAnimation* m_compositeAnimation; // Ideally this would be a reference, but it has to be cleared if an animation is destroyed inside an event callback.
257     Ref<Animation> m_animation;
258 
259     Optional<double> m_startTime;
260     Optional<double> m_pauseTime;
261     double m_requestedStartTime { 0 };
262     Optional<double> m_totalDuration;
263     Optional<double> m_nextIterationDuration;
264 
265     AnimationState m_animationState { AnimationState::New };
266     bool m_colorFilterFunctionListsMatch { false };
267 };
