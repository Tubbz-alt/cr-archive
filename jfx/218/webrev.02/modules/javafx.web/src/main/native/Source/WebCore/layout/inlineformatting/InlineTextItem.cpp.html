 41 static inline bool isWhitespaceCharacter(UChar character, bool preserveNewline)
 42 {
 43     return character == ' ' || character == '\t' || (character == '\n' && !preserveNewline);
 44 }
 45 
 46 static unsigned moveToNextNonWhitespacePosition(const StringView& textContent, unsigned startPosition, bool preserveNewline)
 47 {
 48     auto nextNonWhiteSpacePosition = startPosition;
 49     while (nextNonWhiteSpacePosition < textContent.length() && isWhitespaceCharacter(textContent[nextNonWhiteSpacePosition], preserveNewline))
 50         ++nextNonWhiteSpacePosition;
 51     return nextNonWhiteSpacePosition - startPosition;
 52 }
 53 
 54 static unsigned moveToNextBreakablePosition(unsigned startPosition, LazyLineBreakIterator& lineBreakIterator, const RenderStyle& style)
 55 {
 56     auto textLength = lineBreakIterator.stringView().length();
 57     auto startPositionForNextBreakablePosition = startPosition;
 58     while (startPositionForNextBreakablePosition < textLength) {
 59         auto nextBreakablePosition = TextUtil::findNextBreakablePosition(lineBreakIterator, startPositionForNextBreakablePosition, style);
 60         // Oftentimes the next breakable position comes back as the start position (most notably hyphens).
 61         if (nextBreakablePosition != startPosition)
 62             return nextBreakablePosition - startPosition;
 63         ++startPositionForNextBreakablePosition;
 64     }
 65     return textLength - startPosition;
 66 }

 68 void InlineTextItem::createAndAppendTextItems(InlineItems& inlineContent, const Box& inlineBox)
 69 {
 70     auto& textContext = *inlineBox.textContext();
 71     auto text = textContext.content;
 72     if (!text.length())
 73         return inlineContent.append(InlineTextItem::createEmptyItem(inlineBox));
 74 
 75     auto& style = inlineBox.style();
 76     auto& font = style.fontCascade();
 77     LazyLineBreakIterator lineBreakIterator(text);
 78     unsigned currentPosition = 0;
 79 
 80     auto inlineItemWidth = [&](auto startPosition, auto length) -> Optional<InlineLayoutUnit> {
 81         if (!textContext.canUseSimplifiedContentMeasuring)
 82             return { };
 83         return TextUtil::width(inlineBox, startPosition, startPosition + length);
 84     };
 85 
 86     while (currentPosition < text.length()) {
 87         auto isSegmentBreakCandidate = [](auto character) {
 88             return character == '\n';
 89         };
 90 
 91         // Segment breaks with preserve new line style (white-space: pre, pre-wrap, break-spaces and pre-line) compute to forced line break.
 92         if (isSegmentBreakCandidate(text[currentPosition]) && style.preserveNewline()) {
 93             inlineContent.append(InlineSoftLineBreakItem::createSoftLineBreakItem(inlineBox, currentPosition));
 94             ++currentPosition;
 95             continue;
 96         }
 97 
 98         if (isWhitespaceCharacter(text[currentPosition], style.preserveNewline())) {
 99             auto appendWhitespaceItem = [&] (auto startPosition, auto itemLength) {
100                 auto simpleSingleWhitespaceContent = textContext.canUseSimplifiedContentMeasuring && (itemLength == 1 || style.collapseWhiteSpace());
101                 auto width = simpleSingleWhitespaceContent ? makeOptional(InlineLayoutUnit { font.spaceWidth() }) : inlineItemWidth(startPosition, itemLength);
102                 inlineContent.append(InlineTextItem::createWhitespaceItem(inlineBox, startPosition, itemLength, width));
103             };
104 
105             auto length = moveToNextNonWhitespacePosition(text, currentPosition, style.preserveNewline());
106             if (style.whiteSpace() == WhiteSpace::BreakSpaces) {
107                 // https://www.w3.org/TR/css-text-3/#white-space-phase-1
108                 // For break-spaces, a soft wrap opportunity exists after every space and every tab.
109                 // FIXME: if this turns out to be a perf hit with too many individual whitespace inline items, we should transition this logic to line breaking.
110                 for (unsigned i = 0; i < length; ++i)
111                     appendWhitespaceItem(currentPosition + i, 1);
112             } else
113                 appendWhitespaceItem(currentPosition, length);
114             currentPosition += length;
115             continue;
116         }
117 
118         auto length = moveToNextBreakablePosition(currentPosition, lineBreakIterator, style);
119         inlineContent.append(InlineTextItem::createNonWhitespaceItem(inlineBox, currentPosition, length, inlineItemWidth(currentPosition, length)));
120         currentPosition += length;
121     }
122 }

124 bool InlineTextItem::isEmptyContent() const
125 {
126     // FIXME: We should check for more zero width content and not just U+200B.
127     return !m_length || (m_length == 1 && layoutBox().textContext()->content[start()] == zeroWidthSpace);
128 }
