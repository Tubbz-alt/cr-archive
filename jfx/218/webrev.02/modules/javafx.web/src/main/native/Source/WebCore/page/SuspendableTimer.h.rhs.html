 28 
 29 #include "ActiveDOMObject.h"
 30 #include "Timer.h"
 31 
 32 #include <wtf/Function.h>
 33 #include <wtf/Seconds.h>
 34 
 35 namespace WebCore {
 36 
 37 class SuspendableTimerBase : private TimerBase, public ActiveDOMObject {
 38     WTF_MAKE_FAST_ALLOCATED;
 39 public:
 40     explicit SuspendableTimerBase(ScriptExecutionContext*);
 41     virtual ~SuspendableTimerBase();
 42 
 43     // A hook for derived classes to perform cleanup.
 44     virtual void didStop();
 45 
 46     // Part of TimerBase interface used by SuspendableTimerBase clients, modified to work when suspended.
 47     bool isActive() const { return TimerBase::isActive() || (m_suspended && m_savedIsActive); }
 48     bool isSuspended() const { return m_suspended; }
 49 
 50     Seconds repeatInterval() const;
 51 
 52     void startRepeating(Seconds repeatInterval);
 53     void startOneShot(Seconds interval);
 54 
 55     void augmentFireInterval(Seconds delta);
 56     void augmentRepeatInterval(Seconds delta);
 57 
 58     using TimerBase::didChangeAlignmentInterval;
 59     using TimerBase::operator new;
 60     using TimerBase::operator delete;
 61 
 62     void cancel(); // Equivalent to TimerBase::stop(), whose name conflicts with ActiveDOMObject::stop().
 63 
 64 private:
 65     void fired() override = 0;
 66 
 67     // ActiveDOMObject API.
 68     bool hasPendingActivity() const final;
 69     void stop() final;

 70     void suspend(ReasonForSuspension) final;
 71     void resume() final;
 72 
 73     Seconds m_savedNextFireInterval;
 74     Seconds m_savedRepeatInterval;
 75 
 76     bool m_suspended { false };
 77     bool m_savedIsActive { false };
 78 };
 79 
 80 class SuspendableTimer final : public SuspendableTimerBase {
 81     WTF_MAKE_FAST_ALLOCATED;
 82 public:
 83     template <typename TimerFiredClass, typename TimerFiredBaseClass>
 84     SuspendableTimer(ScriptExecutionContext* context, TimerFiredClass& object, void (TimerFiredBaseClass::*function)())
 85         : SuspendableTimerBase(context)
 86         , m_function(std::bind(function, &object))
 87     {
 88     }
 89 
 90     SuspendableTimer(ScriptExecutionContext* context, WTF::Function<void ()>&& function)
 91         : SuspendableTimerBase(context)
 92         , m_function(WTFMove(function))
 93     {
 94     }
 95 
 96 private:
 97     void fired() final
 98     {
 99         m_function();
100     }
101 
102     const char* activeDOMObjectName() const final;
103 
104     WTF::Function<void ()> m_function;
105 };

107 } // namespace WebCore
