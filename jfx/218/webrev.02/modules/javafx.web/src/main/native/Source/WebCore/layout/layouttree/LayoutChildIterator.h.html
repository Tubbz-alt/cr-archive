 26 #pragma once
 27 
 28 #include "LayoutIterator.h"
 29 
 30 #if ENABLE(LAYOUT_FORMATTING_CONTEXT)
 31 
 32 namespace WebCore {
 33 namespace Layout {
 34 
 35 template <typename T>
 36 class LayoutChildIterator : public LayoutIterator<T> {
 37 public:
 38     LayoutChildIterator(const Container& parent);
 39     LayoutChildIterator(const Container& parent, const T* current);
 40     LayoutChildIterator& operator++();
 41 };
 42 
 43 template <typename T>
 44 class LayoutChildIteratorAdapter {
 45 public:
 46     LayoutChildIteratorAdapter(const Container& parent);
 47     LayoutChildIterator<T> begin() const;
 48     LayoutChildIterator<T> end() const;
 49     const T* first() const;
 50     const T* last() const;
 51 
 52 private:
 53     const Container& m_parent;
 54 };
 55 
 56 template <typename T> LayoutChildIteratorAdapter<T> childrenOfType(const Container&);
 57 
 58 // LayoutChildIterator
 59 
 60 template <typename T>
 61 inline LayoutChildIterator<T>::LayoutChildIterator(const Container& parent)
 62     : LayoutIterator<T>(&parent)
 63 {
 64 }
 65 
 66 template <typename T>
 67 inline LayoutChildIterator<T>::LayoutChildIterator(const Container& parent, const T* current)
 68     : LayoutIterator<T>(&parent, current)
 69 {
 70 }
 71 
 72 template <typename T>
 73 inline LayoutChildIterator<T>& LayoutChildIterator<T>::operator++()
 74 {
 75     return static_cast<LayoutChildIterator<T>&>(LayoutIterator<T>::traverseNextSibling());
 76 }
 77 
 78 // LayoutChildIteratorAdapter
 79 
 80 template <typename T>
 81 inline LayoutChildIteratorAdapter<T>::LayoutChildIteratorAdapter(const Container& parent)
 82     : m_parent(parent)
 83 {
 84 }
 85 
 86 template <typename T>
 87 inline LayoutChildIterator<T> LayoutChildIteratorAdapter<T>::begin() const
 88 {
 89     return LayoutChildIterator<T>(m_parent, Traversal::firstChild<T>(m_parent));
 90 }
 91 
 92 template <typename T>
 93 inline LayoutChildIterator<T> LayoutChildIteratorAdapter<T>::end() const
 94 {
 95     return LayoutChildIterator<T>(m_parent);
 96 }
 97 
 98 template <typename T>
 99 inline const T* LayoutChildIteratorAdapter<T>::first() const
100 {
101     return Traversal::firstChild<T>(m_parent);
102 }
103 
104 template <typename T>
105 inline const T* LayoutChildIteratorAdapter<T>::last() const
106 {
107     return Traversal::lastChild<T>(m_parent);
108 }
109 
110 template <typename T>
111 inline LayoutChildIteratorAdapter<T> childrenOfType(const Container& parent)
112 {
113     return LayoutChildIteratorAdapter<T>(parent);
114 }
115 
116 }
117 }
118 #endif
