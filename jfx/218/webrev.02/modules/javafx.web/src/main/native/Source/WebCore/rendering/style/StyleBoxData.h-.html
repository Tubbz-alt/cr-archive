25 #pragma once
26 
27 #include "Length.h"
28 #include "RenderStyleConstants.h"
29 #include <wtf/RefCounted.h>
30 #include <wtf/Ref.h>
31 
32 namespace WebCore {
33 
34 class StyleBoxData : public RefCounted<StyleBoxData> {
35 public:
36     static Ref<StyleBoxData> create() { return adoptRef(*new StyleBoxData); }
37     Ref<StyleBoxData> copy() const;
38 
39     bool operator==(const StyleBoxData& o) const;
40     bool operator!=(const StyleBoxData& o) const
41     {
42         return !(*this == o);
43     }
44 
45     const Length& width() const { return m_width; }
46     const Length& height() const { return m_height; }
47 
48     const Length& minWidth() const { return m_minWidth; }
49     const Length& minHeight() const { return m_minHeight; }
50 
51     const Length& maxWidth() const { return m_maxWidth; }
52     const Length& maxHeight() const { return m_maxHeight; }
53 
54     const Length& verticalAlign() const { return m_verticalAlign; }
55 
56     int zIndex() const { return m_zIndex; }
57     bool hasAutoZIndex() const { return m_hasAutoZIndex; }
58 
59     BoxSizing boxSizing() const { return static_cast<BoxSizing>(m_boxSizing); }
60 #if ENABLE(CSS_BOX_DECORATION_BREAK)
61     BoxDecorationBreak boxDecorationBreak() const { return static_cast<BoxDecorationBreak>(m_boxDecorationBreak); }
62 #endif
63 
64 private:
65     friend class RenderStyle;
66 
67     StyleBoxData();
68     StyleBoxData(const StyleBoxData&);
69 
70     Length m_width;
71     Length m_height;
72 
73     Length m_minWidth;
74     Length m_maxWidth;
75 
76     Length m_minHeight;
77     Length m_maxHeight;
78 
79     Length m_verticalAlign;
80 
81     int m_zIndex;
82     unsigned m_hasAutoZIndex : 1;
83     unsigned m_boxSizing : 1; // BoxSizing
84 #if ENABLE(CSS_BOX_DECORATION_BREAK)
85     unsigned m_boxDecorationBreak : 1; // BoxDecorationBreak
86 #endif
87 };
88 
89 } // namespace WebCore
