54 JSValue Bytecodes::toJS(ExecState* exec) const
55 {
56     VM& vm = exec->vm();
57     JSObject* result = constructEmptyObject(exec);
58 
59     result->putDirect(vm, vm.propertyNames->bytecodesID, jsNumber(m_id));
60     result->putDirect(vm, vm.propertyNames->inferredName, jsString(vm, String::fromUTF8(m_inferredName)));
61     result->putDirect(vm, vm.propertyNames->sourceCode, jsString(vm, String::fromUTF8(m_sourceCode)));
62     result->putDirect(vm, vm.propertyNames->hash, jsString(vm, String::fromUTF8(toCString(m_hash))));
63     result->putDirect(vm, vm.propertyNames->instructionCount, jsNumber(m_instructionCount));
64     addSequenceProperties(exec, result);
65 
66     return result;
