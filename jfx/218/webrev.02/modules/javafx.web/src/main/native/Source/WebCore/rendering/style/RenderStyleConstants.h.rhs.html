  47 // The difference between two styles.  The following values are used:
  48 // - StyleDifference::Equal - The two styles are identical
  49 // - StyleDifference::RecompositeLayer - The layer needs its position and transform updated, but no repaint
  50 // - StyleDifference::Repaint - The object just needs to be repainted.
  51 // - StyleDifference::RepaintIfTextOrBorderOrOutline - The object needs to be repainted if it contains text or a border or outline.
  52 // - StyleDifference::RepaintLayer - The layer and its descendant layers needs to be repainted.
  53 // - StyleDifference::LayoutPositionedMovementOnly - Only the position of this positioned object has been updated
  54 // - StyleDifference::SimplifiedLayout - Only overflow needs to be recomputed
  55 // - StyleDifference::SimplifiedLayoutAndPositionedMovement - Both positioned movement and simplified layout updates are required.
  56 // - StyleDifference::Layout - A full layout is required.
  57 enum class StyleDifference {
  58     Equal,
  59     RecompositeLayer,
  60     Repaint,
  61     RepaintIfTextOrBorderOrOutline,
  62     RepaintLayer,
  63     LayoutPositionedMovementOnly,
  64     SimplifiedLayout,
  65     SimplifiedLayoutAndPositionedMovement,
  66     Layout,
  67     NewStyle
  68 };

  78     Filter      = 1 << 2,
  79     ClipRect    = 1 << 3,

  81     WillChange  = 1 << 5,

 114 class PseudoIdSet {
 115 public:
 116     PseudoIdSet()
 117         : m_data(0)
 118     {
 119     }
 120 
 121     PseudoIdSet(std::initializer_list<PseudoId> initializerList)
 122         : m_data(0)
 123     {
 124         for (PseudoId pseudoId : initializerList)
 125             add(pseudoId);
 126     }
 127 
 128     static PseudoIdSet fromMask(unsigned rawPseudoIdSet)
 129     {
 130         return PseudoIdSet(rawPseudoIdSet);
 131     }
 132 
 133     bool has(PseudoId pseudoId) const
 134     {
 135         ASSERT((sizeof(m_data) * 8) > static_cast<unsigned>(pseudoId));
 136         return m_data & (1U << static_cast<unsigned>(pseudoId));
 137     }
 138 
 139     void add(PseudoId pseudoId)
 140     {
 141         ASSERT((sizeof(m_data) * 8) > static_cast<unsigned>(pseudoId));
 142         m_data |= (1U << static_cast<unsigned>(pseudoId));
 143     }
 144 
 145     void remove(PseudoId pseudoId)
 146     {
 147         ASSERT((sizeof(m_data) * 8) > static_cast<unsigned>(pseudoId));
 148         m_data &= ~(1U << static_cast<unsigned>(pseudoId));
 149     }
 150 
 151     void merge(PseudoIdSet source)
 152     {
 153         m_data |= source.m_data;
 154     }
 155 
 156     PseudoIdSet operator &(const PseudoIdSet& pseudoIdSet) const
 157     {
 158         return PseudoIdSet(m_data & pseudoIdSet.m_data);
 159     }
 160 
 161     PseudoIdSet operator |(const PseudoIdSet& pseudoIdSet) const
 162     {
 163         return PseudoIdSet(m_data | pseudoIdSet.m_data);
 164     }
 165 
 166     explicit operator bool() const
 167     {
 168         return m_data;
 169     }
 170 
 171     unsigned data() const { return m_data; }
 172 
 173     static ptrdiff_t dataMemoryOffset() { return OBJECT_OFFSETOF(PseudoIdSet, m_data); }
 174 
 175 private:
 176     explicit PseudoIdSet(unsigned rawPseudoIdSet)
 177         : m_data(rawPseudoIdSet)
 178     {
 179     }
 180 
 181     unsigned m_data;
 182 };
