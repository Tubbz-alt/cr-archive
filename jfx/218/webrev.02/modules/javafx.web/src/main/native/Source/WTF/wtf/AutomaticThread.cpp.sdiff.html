 49 void AutomaticThreadCondition::notifyOne(const AbstractLocker& locker)
 50 {
 51     for (AutomaticThread* thread : m_threads) {
 52         if (thread->isWaiting(locker)) {
 53             thread->notify(locker);
 54             return;

154         m_isRunningCondition.wait(*m_lock);
