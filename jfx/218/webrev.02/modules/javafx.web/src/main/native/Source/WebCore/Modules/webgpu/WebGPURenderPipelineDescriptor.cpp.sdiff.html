35 Optional<GPURenderPipelineDescriptor> WebGPURenderPipelineDescriptor::tryCreateGPURenderPipelineDescriptor(GPUErrorScopes& errorScopes) const
36 {
37     auto pipelineLayout = layout ? makeRefPtr(layout->pipelineLayout()) : nullptr;
38 
39     auto vertex = vertexStage.tryCreateGPUPipelineStageDescriptor();
40 
41     Optional<GPUPipelineStageDescriptor> fragment;
42     if (fragmentStage)
43         fragment = fragmentStage->tryCreateGPUPipelineStageDescriptor();
44

46         errorScopes.generatePrefixedError("Invalid GPUPipelineStageDescriptor!");
47         return WTF::nullopt;
48     }
49 
50     // FIXME: Web GPU validation, e.g. fail if colorStates is larger than (max number of supported color states).
