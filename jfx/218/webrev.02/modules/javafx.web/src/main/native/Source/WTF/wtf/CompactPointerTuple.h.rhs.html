 26 #pragma once
 27 
 28 #include <type_traits>
 29 #include <wtf/FastMalloc.h>
 30 #include <wtf/StdLibExtras.h>
 31 
 32 namespace WTF {
 33 
 34 // The goal of this class is folding a pointer and 1 byte value into 8 bytes in both 32bit and 64bit architectures.
 35 // 32bit architecture just has a pair of byte and pointer, which should be 8 bytes.
 36 // In 64bit, we use the upper 5 bits and lower 3 bits (zero due to alignment) since these bits are safe to use even
 37 // with 5-level page tables where the effective pointer width is 57bits.
 38 template<typename PointerType, typename Type>
 39 class CompactPointerTuple final {
 40     WTF_MAKE_FAST_ALLOCATED;
 41 public:
 42     static_assert(sizeof(Type) == 1, "");
 43     static_assert(std::is_pointer<PointerType>::value, "");
 44     static_assert(std::is_integral<Type>::value || std::is_enum<Type>::value, "");
 45 
 46     CompactPointerTuple() = default;
 47 
 48 #if CPU(ADDRESS64)
 49 public:
 50     static constexpr uint64_t encodeType(uint8_t type)
 51     {
 52         // Encode 8bit type UUUDDDDD into 64bit data DDDDD..56bit..UUU.
 53         return (static_cast<uint64_t>(type) << 59) | (static_cast<uint64_t>(type) >> 5);
 54     }
 55     static constexpr uint8_t decodeType(uint64_t value)
 56     {
 57         // Decode 64bit data DDDDD..56bit..UUU into 8bit type UUUDDDDD.
 58         return static_cast<uint8_t>((value >> 59) | (value << 5));
 59     }
 60 
 61     static constexpr uint64_t typeMask = encodeType(UINT8_MAX);
 62     static_assert(0xF800000000000007ULL == typeMask, "");
 63     static constexpr uint64_t pointerMask = ~typeMask;
 64 
 65     CompactPointerTuple(PointerType pointer, Type type)
 66         : m_data(bitwise_cast<uint64_t>(pointer) | encodeType(static_cast<uint8_t>(type)))
 67     {
 68         ASSERT((bitwise_cast<uint64_t>(pointer) & 0b111) == 0x0);
 69     }
 70 
 71     PointerType pointer() const { return bitwise_cast<PointerType>(m_data & pointerMask); }
 72     void setPointer(PointerType pointer)
 73     {
 74         static_assert(alignof(typename std::remove_pointer<PointerType>::type) >= alignof(void*), "");
 75         ASSERT((bitwise_cast<uint64_t>(pointer) & 0b111) == 0x0);
 76         m_data = CompactPointerTuple(pointer, type()).m_data;
 77     }
 78 
 79     Type type() const { return static_cast<Type>(decodeType(m_data)); }
 80     void setType(Type type)
 81     {
 82         m_data = CompactPointerTuple(pointer(), type).m_data;
 83     }
 84 
 85 private:
 86     uint64_t m_data { 0 };
 87 #else
 88 public:
 89     CompactPointerTuple(PointerType pointer, Type type)
 90         : m_pointer(pointer)
 91         , m_type(type)
 92     {
 93     }
 94 
 95     PointerType pointer() const { return m_pointer; }
 96     void setPointer(PointerType pointer) { m_pointer = pointer; }
 97     Type type() const { return m_type; }
 98     void setType(Type type) { m_type = type; }
 99 
100 private:
101     PointerType m_pointer { nullptr };
102     Type m_type { 0 };
103 #endif
104 };
105 
106 } // namespace WTF
107 
108 using WTF::CompactPointerTuple;
