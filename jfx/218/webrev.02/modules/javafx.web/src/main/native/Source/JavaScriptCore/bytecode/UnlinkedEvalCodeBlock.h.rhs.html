26 #pragma once
27 
28 #include "UnlinkedGlobalCodeBlock.h"
29 
30 namespace JSC {
31 
32 class CachedEvalCodeBlock;
33 
34 class UnlinkedEvalCodeBlock final : public UnlinkedGlobalCodeBlock {
35 public:
36     typedef UnlinkedGlobalCodeBlock Base;
37     static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
38 
39     template<typename CellType, SubspaceAccess mode>
40     static IsoSubspace* subspaceFor(VM& vm)
41     {
42         return vm.unlinkedEvalCodeBlockSpace<mode>();
43     }
44 
45     static UnlinkedEvalCodeBlock* create(VM& vm, const ExecutableInfo& info, OptionSet<CodeGenerationMode> codeGenerationMode)
46     {
47         UnlinkedEvalCodeBlock* instance = new (NotNull, allocateCell<UnlinkedEvalCodeBlock>(vm.heap)) UnlinkedEvalCodeBlock(vm, vm.unlinkedEvalCodeBlockStructure.get(), info, codeGenerationMode);
48         instance->finishCreation(vm);
49         return instance;
50     }
51 
52     static void destroy(JSCell*);
53 
54     const Identifier& variable(unsigned index) { return m_variables[index]; }
55     unsigned numVariables() { return m_variables.size(); }
56     void adoptVariables(Vector<Identifier, 0, UnsafeVectorOverflow>& variables)
57     {
58         ASSERT(m_variables.isEmpty());
59         m_variables.swap(variables);
60     }
61 
62     const Identifier& functionHoistingCandidate(unsigned index) { return m_functionHoistingCandidates[index]; }
63     unsigned numFunctionHoistingCandidates() { return m_functionHoistingCandidates.size(); }
64     void adoptFunctionHoistingCandidates(Vector<Identifier, 0, UnsafeVectorOverflow>&& functionHoistingCandidates)
65     {
66         ASSERT(m_functionHoistingCandidates.isEmpty());
67         m_functionHoistingCandidates = WTFMove(functionHoistingCandidates);
68     }
69 private:
70     friend CachedEvalCodeBlock;
71 
72     UnlinkedEvalCodeBlock(VM& vm, Structure* structure, const ExecutableInfo& info, OptionSet<CodeGenerationMode> codeGenerationMode)
73         : Base(vm, structure, EvalCode, info, codeGenerationMode)
74     {
75     }
76 
77     UnlinkedEvalCodeBlock(Decoder&, const CachedEvalCodeBlock&);
78 
79     Vector<Identifier, 0, UnsafeVectorOverflow> m_variables;
80     Vector<Identifier, 0, UnsafeVectorOverflow> m_functionHoistingCandidates;
81 
82 public:
83     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue proto)
84     {
85         return Structure::create(vm, globalObject, proto, TypeInfo(UnlinkedEvalCodeBlockType, StructureFlags), info());
86     }
87 
88     DECLARE_INFO;
89 };
90 
91 }
