 27 #ifndef ThreadGlobalData_h
 28 #define ThreadGlobalData_h
 29 
 30 #include <wtf/ThreadSafeRefCounted.h>
 31 #include <wtf/text/StringHash.h>
 32 
 33 namespace JSC {
 34 class CallFrame;
 35 class JSGlobalObject;
 36 }
 37 
 38 namespace WebCore {
 39 
 40 class QualifiedNameCache;
 41 class ThreadTimers;
 42 
 43 struct CachedResourceRequestInitiators;
 44 struct EventNames;
 45 struct ICUConverterWrapper;
 46 struct MIMETypeRegistryThreadGlobalData;
 47 
 48 #if USE(WEB_THREAD)
 49 class ThreadGlobalData : public ThreadSafeRefCounted<ThreadGlobalData> {
 50 #else
 51 class ThreadGlobalData {
 52 #endif
 53     WTF_MAKE_NONCOPYABLE(ThreadGlobalData);
 54     WTF_MAKE_FAST_ALLOCATED;
 55 public:
 56     WEBCORE_EXPORT ThreadGlobalData();
 57     WEBCORE_EXPORT ~ThreadGlobalData();
 58     void destroy(); // called on workers to clean up the ThreadGlobalData before the thread exits.
 59 
 60     const CachedResourceRequestInitiators& cachedResourceRequestInitiators() { return *m_cachedResourceRequestInitiators; }
 61     EventNames& eventNames() { return *m_eventNames; }
 62     ThreadTimers& threadTimers() { return *m_threadTimers; }
 63     QualifiedNameCache& qualifiedNameCache() { return *m_qualifiedNameCache; }
 64 
 65     ICUConverterWrapper& cachedConverterICU() { return *m_cachedConverterICU; }
 66 
 67     JSC::JSGlobalObject* currentState() const { return m_currentState; }
 68     void setCurrentState(JSC::JSGlobalObject* state) { m_currentState = state; }
 69 
 70 #if USE(WEB_THREAD)
 71     void setWebCoreThreadData();
 72 #endif
 73 
 74     bool isInRemoveAllEventListeners() const { return m_isInRemoveAllEventListeners; }
 75     void setIsInRemoveAllEventListeners(bool value) { m_isInRemoveAllEventListeners = value; }
 76 
 77     const MIMETypeRegistryThreadGlobalData& mimeTypeRegistryThreadGlobalData();
 78 
 79 private:
 80     std::unique_ptr<CachedResourceRequestInitiators> m_cachedResourceRequestInitiators;
 81     std::unique_ptr<EventNames> m_eventNames;
 82     std::unique_ptr<ThreadTimers> m_threadTimers;
 83     std::unique_ptr<QualifiedNameCache> m_qualifiedNameCache;
 84     JSC::JSGlobalObject* m_currentState { nullptr };
 85 
 86 #ifndef NDEBUG
 87     bool m_isMainThread;
 88 #endif
 89 
 90     bool m_isInRemoveAllEventListeners { false };
 91 
 92     std::unique_ptr<ICUConverterWrapper> m_cachedConverterICU;
 93     std::unique_ptr<MIMETypeRegistryThreadGlobalData> m_MIMETypeRegistryThreadGlobalData;
 94 
 95     WEBCORE_EXPORT friend ThreadGlobalData& threadGlobalData();
 96 };
 97 
 98 #if USE(WEB_THREAD)
 99 WEBCORE_EXPORT ThreadGlobalData& threadGlobalData();
100 #else
101 WEBCORE_EXPORT ThreadGlobalData& threadGlobalData() PURE_FUNCTION;
102 #endif
103 
104 } // namespace WebCore
105 
106 #endif // ThreadGlobalData_h
