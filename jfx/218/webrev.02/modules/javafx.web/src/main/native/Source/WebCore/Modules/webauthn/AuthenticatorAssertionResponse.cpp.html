35 Ref<AuthenticatorAssertionResponse> AuthenticatorAssertionResponse::create(Ref<ArrayBuffer>&& rawId, Ref<ArrayBuffer>&& authenticatorData, Ref<ArrayBuffer>&& signature, RefPtr<ArrayBuffer>&& userHandle, Optional<AuthenticationExtensionsClientOutputs>&& extensions)
36 {
37     auto response = adoptRef(*new AuthenticatorAssertionResponse(WTFMove(rawId), WTFMove(authenticatorData), WTFMove(signature), WTFMove(userHandle)));
38     if (extensions)
39         response->setExtensions(WTFMove(*extensions));
40     return response;
41 }
42 
43 Ref<AuthenticatorAssertionResponse> AuthenticatorAssertionResponse::create(const Vector<uint8_t>& rawId, const Vector<uint8_t>& authenticatorData, const Vector<uint8_t>& signature,  const Vector<uint8_t>& userHandle)
44 {
45     RefPtr<ArrayBuffer> userhandleBuffer;
46     if (!userHandle.isEmpty())
47         userhandleBuffer = ArrayBuffer::create(userHandle.data(), userHandle.size());
48     return create(ArrayBuffer::create(rawId.data(), rawId.size()), ArrayBuffer::create(authenticatorData.data(), authenticatorData.size()), ArrayBuffer::create(signature.data(), signature.size()), WTFMove(userhandleBuffer), WTF::nullopt);
49 }

51 AuthenticatorAssertionResponse::AuthenticatorAssertionResponse(Ref<ArrayBuffer>&& rawId, Ref<ArrayBuffer>&& authenticatorData, Ref<ArrayBuffer>&& signature, RefPtr<ArrayBuffer>&& userHandle)
52     : AuthenticatorResponse(WTFMove(rawId))
53     , m_authenticatorData(WTFMove(authenticatorData))
54     , m_signature(WTFMove(signature))
55     , m_userHandle(WTFMove(userHandle))
56 {
57 }
58 
59 AuthenticatorResponseData AuthenticatorAssertionResponse::data() const
60 {
61     auto data = AuthenticatorResponse::data();
62     data.isAuthenticatorAttestationResponse = false;
63     data.authenticatorData = m_authenticatorData.copyRef();
64     data.signature = m_signature.copyRef();
65     data.userHandle = m_userHandle;
66     return data;
67 }
