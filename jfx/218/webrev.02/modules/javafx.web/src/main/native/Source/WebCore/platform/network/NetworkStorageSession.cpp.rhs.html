 50 void NetworkStorageSession::permitProcessToUseCookieAPI(bool value)
 51 {
 52     m_processMayUseCookieAPI = value;

 61 bool NetworkStorageSession::shouldBlockThirdPartyCookies(const RegistrableDomain& registrableDomain) const
 62 {
 63     if (!m_isResourceLoadStatisticsEnabled || registrableDomain.isEmpty())
 64         return false;
 65 
 66     ASSERT(!(m_registrableDomainsToBlockAndDeleteCookiesFor.contains(registrableDomain) && m_registrableDomainsToBlockButKeepCookiesFor.contains(registrableDomain)));
 67 
 68     return m_registrableDomainsToBlockAndDeleteCookiesFor.contains(registrableDomain)
 69         || m_registrableDomainsToBlockButKeepCookiesFor.contains(registrableDomain);
 70 }
 71 
 72 bool NetworkStorageSession::shouldBlockThirdPartyCookiesButKeepFirstPartyCookiesFor(const RegistrableDomain& registrableDomain) const
 73 {
 74     if (!m_isResourceLoadStatisticsEnabled || registrableDomain.isEmpty())
 75         return false;
 76 
 77     ASSERT(!(m_registrableDomainsToBlockAndDeleteCookiesFor.contains(registrableDomain) && m_registrableDomainsToBlockButKeepCookiesFor.contains(registrableDomain)));
 78 
 79     return m_registrableDomainsToBlockButKeepCookiesFor.contains(registrableDomain);
 80 }

 82 bool NetworkStorageSession::hasHadUserInteractionAsFirstParty(const RegistrableDomain& registrableDomain) const
 83 {
 84     if (registrableDomain.isEmpty())
 85         return false;

 87     return m_registrableDomainsWithUserInteractionAsFirstParty.contains(registrableDomain);

 90 bool NetworkStorageSession::shouldBlockCookies(const ResourceRequest& request, Optional<FrameIdentifier> frameID, Optional<PageIdentifier> pageID) const
 91 {
 92     if (!m_isResourceLoadStatisticsEnabled)
 93         return false;
 94 
 95     return shouldBlockCookies(request.firstPartyForCookies(), request.url(), frameID, pageID);
 96 }
 97 
 98 bool NetworkStorageSession::shouldBlockCookies(const URL& firstPartyForCookies, const URL& resource, Optional<FrameIdentifier> frameID, Optional<PageIdentifier> pageID) const
 99 {
100     if (!m_isResourceLoadStatisticsEnabled)
101         return false;
102 
103     RegistrableDomain firstPartyDomain { firstPartyForCookies };
104     if (firstPartyDomain.isEmpty())
105         return false;
106 
107     RegistrableDomain resourceDomain { resource };
108     if (resourceDomain.isEmpty())
109         return false;
110 
111     if (firstPartyDomain == resourceDomain)
112         return false;
113 
114     if (pageID && hasStorageAccess(resourceDomain, firstPartyDomain, frameID, pageID.value()))
115         return false;
116 
117     switch (m_thirdPartyCookieBlockingMode) {
118     case ThirdPartyCookieBlockingMode::All:
119         return true;
120     case ThirdPartyCookieBlockingMode::AllOnSitesWithoutUserInteraction:
121         if (!hasHadUserInteractionAsFirstParty(firstPartyDomain))
122             return true;
123         FALLTHROUGH;
124     case ThirdPartyCookieBlockingMode::OnlyAccordingToPerDomainPolicy:
125         return shouldBlockThirdPartyCookies(resourceDomain);
126     }
127     ASSERT_NOT_REACHED();
128     return false;
129 }

131 Optional<Seconds> NetworkStorageSession::maxAgeCacheCap(const ResourceRequest& request)
132 {
133     if (m_cacheMaxAgeCapForPrevalentResources && shouldBlockCookies(request, WTF::nullopt, WTF::nullopt))
134         return m_cacheMaxAgeCapForPrevalentResources;
135     return WTF::nullopt;
136 }

170 bool NetworkStorageSession::hasStorageAccess(const RegistrableDomain& resourceDomain, const RegistrableDomain& firstPartyDomain, Optional<FrameIdentifier> frameID, PageIdentifier pageID) const
171 {
172     if (frameID) {
173         auto framesGrantedIterator = m_framesGrantedStorageAccess.find(pageID);
174         if (framesGrantedIterator != m_framesGrantedStorageAccess.end()) {
175             auto it = framesGrantedIterator->value.find(frameID.value());
176             if (it != framesGrantedIterator->value.end() && it->value == resourceDomain)
177                 return true;
178         }
179     }
180 
181     if (!firstPartyDomain.isEmpty()) {
182         auto pagesGrantedIterator = m_pagesGrantedStorageAccess.find(pageID);
183         if (pagesGrantedIterator != m_pagesGrantedStorageAccess.end()) {
184             auto it = pagesGrantedIterator->value.find(firstPartyDomain);
185             if (it != pagesGrantedIterator->value.end() && it->value == resourceDomain)
186                 return true;
187         }
188     }
189 
190     return false;
191 }

193 Vector<String> NetworkStorageSession::getAllStorageAccessEntries() const
194 {
195     Vector<String> entries;
196     for (auto& innerMap : m_framesGrantedStorageAccess.values()) {
197         for (auto& value : innerMap.values())
198             entries.append(value.string());
199     }
200     return entries;
201 }

203 void NetworkStorageSession::grantStorageAccess(const RegistrableDomain& resourceDomain, const RegistrableDomain& firstPartyDomain, Optional<FrameIdentifier> frameID, PageIdentifier pageID)
204 {
205     if (!frameID) {
206         if (firstPartyDomain.isEmpty())
207             return;
208         auto pagesGrantedIterator = m_pagesGrantedStorageAccess.find(pageID);
209         if (pagesGrantedIterator == m_pagesGrantedStorageAccess.end()) {
210             HashMap<RegistrableDomain, RegistrableDomain> entry;
211             entry.add(firstPartyDomain, resourceDomain);
212             m_pagesGrantedStorageAccess.add(pageID, entry);
213         } else {
214             auto firstPartyDomainIterator = pagesGrantedIterator->value.find(firstPartyDomain);
215             if (firstPartyDomainIterator == pagesGrantedIterator->value.end())
216                 pagesGrantedIterator->value.add(firstPartyDomain, resourceDomain);
217             else
218                 firstPartyDomainIterator->value = resourceDomain;
219         }
220         return;
221     }
222 
223     auto pagesGrantedIterator = m_framesGrantedStorageAccess.find(pageID);
224     if (pagesGrantedIterator == m_framesGrantedStorageAccess.end()) {
225         HashMap<FrameIdentifier, RegistrableDomain> entry;
226         entry.add(frameID.value(), resourceDomain);
227         m_framesGrantedStorageAccess.add(pageID, entry);
228     } else {
229         auto framesGrantedIterator = pagesGrantedIterator->value.find(frameID.value());
230         if (framesGrantedIterator == pagesGrantedIterator->value.end())
231             pagesGrantedIterator->value.add(frameID.value(), resourceDomain);
232         else
233             framesGrantedIterator->value = resourceDomain;
234     }
235 }

237 void NetworkStorageSession::removeStorageAccessForFrame(FrameIdentifier frameID, PageIdentifier pageID)
238 {
239     auto iteration = m_framesGrantedStorageAccess.find(pageID);
240     if (iteration == m_framesGrantedStorageAccess.end())
241         return;
242 
243     iteration->value.remove(frameID);
244 }

286 Optional<Seconds> NetworkStorageSession::clientSideCookieCap(const RegistrableDomain& firstParty, Optional<PageIdentifier> pageID) const
287 {
288     if (!m_ageCapForClientSideCookies || !pageID || m_navigatedToWithLinkDecorationByPrevalentResource.isEmpty())
289         return m_ageCapForClientSideCookies;
290 
291     auto domainIterator = m_navigatedToWithLinkDecorationByPrevalentResource.find(*pageID);
292     if (domainIterator == m_navigatedToWithLinkDecorationByPrevalentResource.end())
293         return m_ageCapForClientSideCookies;
294 
295     if (domainIterator->value == firstParty)
296         return m_ageCapForClientSideCookiesShort;
297 
298     return m_ageCapForClientSideCookies;
299 }
