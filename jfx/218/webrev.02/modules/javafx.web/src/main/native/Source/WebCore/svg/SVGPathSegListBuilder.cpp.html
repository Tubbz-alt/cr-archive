 34 SVGPathSegListBuilder::SVGPathSegListBuilder(SVGPathSegList& pathSegList)
 35     : m_pathSegList(pathSegList)
 36 {
 37 }
 38 
 39 void SVGPathSegListBuilder::moveTo(const FloatPoint& targetPoint, bool, PathCoordinateMode mode)
 40 {
 41     if (mode == AbsoluteCoordinates)
 42         m_pathSegList.append(SVGPathSegMovetoAbs::create(targetPoint.x(), targetPoint.y()));
 43     else
 44         m_pathSegList.append(SVGPathSegMovetoRel::create(targetPoint.x(), targetPoint.y()));
 45 }
 46 
 47 void SVGPathSegListBuilder::lineTo(const FloatPoint& targetPoint, PathCoordinateMode mode)
 48 {
 49     if (mode == AbsoluteCoordinates)
 50         m_pathSegList.append(SVGPathSegLinetoAbs::create(targetPoint.x(), targetPoint.y()));
 51     else
 52         m_pathSegList.append(SVGPathSegLinetoRel::create(targetPoint.x(), targetPoint.y()));
 53 }
 54 
 55 void SVGPathSegListBuilder::lineToHorizontal(float x, PathCoordinateMode mode)
 56 {
 57     if (mode == AbsoluteCoordinates)
 58         m_pathSegList.append(SVGPathSegLinetoHorizontalAbs::create(x));
 59     else
 60         m_pathSegList.append(SVGPathSegLinetoHorizontalRel::create(x));
 61 }
 62 
 63 void SVGPathSegListBuilder::lineToVertical(float y, PathCoordinateMode mode)
 64 {
 65     if (mode == AbsoluteCoordinates)
 66         m_pathSegList.append(SVGPathSegLinetoVerticalAbs::create(y));
 67     else
 68         m_pathSegList.append(SVGPathSegLinetoVerticalRel::create(y));
 69 }
 70 
 71 void SVGPathSegListBuilder::curveToCubic(const FloatPoint& point1, const FloatPoint& point2, const FloatPoint& targetPoint, PathCoordinateMode mode)
 72 {
 73     if (mode == AbsoluteCoordinates)
 74         m_pathSegList.append(SVGPathSegCurvetoCubicAbs::create(targetPoint.x(), targetPoint.y(), point1.x(), point1.y(), point2.x(), point2.y()));
 75     else
 76         m_pathSegList.append(SVGPathSegCurvetoCubicRel::create(targetPoint.x(), targetPoint.y(), point1.x(), point1.y(), point2.x(), point2.y()));
 77 }
 78 
 79 void SVGPathSegListBuilder::curveToCubicSmooth(const FloatPoint& point2, const FloatPoint& targetPoint, PathCoordinateMode mode)
 80 {
 81     if (mode == AbsoluteCoordinates)
 82         m_pathSegList.append(SVGPathSegCurvetoCubicSmoothAbs::create(targetPoint.x(), targetPoint.y(), point2.x(), point2.y()));
 83     else
 84         m_pathSegList.append(SVGPathSegCurvetoCubicSmoothRel::create(targetPoint.x(), targetPoint.y(), point2.x(), point2.y()));
 85 }
 86 
 87 void SVGPathSegListBuilder::curveToQuadratic(const FloatPoint& point1, const FloatPoint& targetPoint, PathCoordinateMode mode)
 88 {
 89     if (mode == AbsoluteCoordinates)
 90         m_pathSegList.append(SVGPathSegCurvetoQuadraticAbs::create(targetPoint.x(), targetPoint.y(), point1.x(), point1.y()));
 91     else
 92         m_pathSegList.append(SVGPathSegCurvetoQuadraticRel::create(targetPoint.x(), targetPoint.y(), point1.x(), point1.y()));
 93 }
 94 
 95 void SVGPathSegListBuilder::curveToQuadraticSmooth(const FloatPoint& targetPoint, PathCoordinateMode mode)
 96 {
 97     if (mode == AbsoluteCoordinates)
 98         m_pathSegList.append(SVGPathSegCurvetoQuadraticSmoothAbs::create(targetPoint.x(), targetPoint.y()));
 99     else
100         m_pathSegList.append(SVGPathSegCurvetoQuadraticSmoothRel::create(targetPoint.x(), targetPoint.y()));
101 }
102 
103 void SVGPathSegListBuilder::arcTo(float r1, float r2, float angle, bool largeArcFlag, bool sweepFlag, const FloatPoint& targetPoint, PathCoordinateMode mode)
104 {
105     if (mode == AbsoluteCoordinates)
106         m_pathSegList.append(SVGPathSegArcAbs::create(targetPoint.x(), targetPoint.y(), r1, r2, angle, largeArcFlag, sweepFlag));
107     else
108         m_pathSegList.append(SVGPathSegArcRel::create(targetPoint.x(), targetPoint.y(), r1, r2, angle, largeArcFlag, sweepFlag));
109 }
110 
111 void SVGPathSegListBuilder::closePath()
112 {
113     m_pathSegList.append(SVGPathSegClosePath::create());
114 }
