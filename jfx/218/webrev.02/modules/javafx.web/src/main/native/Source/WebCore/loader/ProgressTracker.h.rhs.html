26 #pragma once
27 
28 #include "Timer.h"
29 #include <wtf/Forward.h>
30 #include <wtf/HashMap.h>
31 #include <wtf/Noncopyable.h>
32 #include <wtf/RefPtr.h>
33 #include <wtf/UniqueRef.h>
34 
35 namespace WebCore {
36 
37 class Frame;
38 class ResourceResponse;
39 class ProgressTrackerClient;
40 struct ProgressItem;
41 
42 class ProgressTracker {
43     WTF_MAKE_NONCOPYABLE(ProgressTracker);
44     WTF_MAKE_FAST_ALLOCATED;
45 public:
46     explicit ProgressTracker(UniqueRef<ProgressTrackerClient>&&);
47     ~ProgressTracker();
48 
49     ProgressTrackerClient& client() { return m_client.get(); }
50 
51     static unsigned long createUniqueIdentifier();
52 
53     WEBCORE_EXPORT double estimatedProgress() const;
54 
55     void progressStarted(Frame&);
56     void progressCompleted(Frame&);
57 
58     void incrementProgress(unsigned long identifier, const ResourceResponse&);
59     void incrementProgress(unsigned long identifier, unsigned bytesReceived);
60     void completeProgress(unsigned long identifier);
61 
62     long long totalPageAndResourceBytesToLoad() const { return m_totalPageAndResourceBytesToLoad; }
63     long long totalBytesReceived() const { return m_totalBytesReceived; }
64 
65     bool isMainLoadProgressing() const;
66 
67 private:
68     void reset();
69     void finalProgressComplete();
70 
71     void progressHeartbeatTimerFired();
72     bool isAlwaysOnLoggingAllowed() const;
73 
74     static unsigned long s_uniqueIdentifier;
75 
76     UniqueRef<ProgressTrackerClient> m_client;
77     RefPtr<Frame> m_originatingProgressFrame;
78     HashMap<unsigned long, std::unique_ptr<ProgressItem>> m_progressItems;
79     Timer m_progressHeartbeatTimer;
80 
81     long long m_totalPageAndResourceBytesToLoad { 0 };
82     long long m_totalBytesReceived { 0 };
83     long long m_totalBytesReceivedBeforePreviousHeartbeat { 0 };
84 
85     double m_lastNotifiedProgressValue { 0 };
86     double m_progressValue { 0 };
87 
88     MonotonicTime m_mainLoadCompletionTime;
89     MonotonicTime m_lastNotifiedProgressTime;
90 
91     int m_numProgressTrackedFrames { 0 };
92     unsigned m_heartbeatsWithNoProgress { 0 };
93 
94     bool m_finalProgressChangedSent { false };
95     bool m_isMainLoad { false };
96 };
97 
98 } // namespace WebCore
