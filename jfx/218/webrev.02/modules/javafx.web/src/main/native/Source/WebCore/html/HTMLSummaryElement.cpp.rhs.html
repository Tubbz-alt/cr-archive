 53 Ref<HTMLSummaryElement> HTMLSummaryElement::create(const QualifiedName& tagName, Document& document)
 54 {
 55     Ref<HTMLSummaryElement> summary = adoptRef(*new HTMLSummaryElement(tagName, document));
 56     summary->addShadowRoot(ShadowRoot::create(document, makeUnique<SummarySlotElement>()));
 57     return summary;
 58 }

 71 void HTMLSummaryElement::didAddUserAgentShadowRoot(ShadowRoot& root)

 77 RefPtr<HTMLDetailsElement> HTMLSummaryElement::detailsElement() const
 78 {
 79     auto* parent = parentElement();
 80     if (parent && is<HTMLDetailsElement>(*parent))
 81         return downcast<HTMLDetailsElement>(parent);
 82     // Fallback summary element is in the shadow tree.
 83     auto* host = shadowHost();
 84     if (host && is<HTMLDetailsElement>(*host))
 85         return downcast<HTMLDetailsElement>(host);
 86     return nullptr;
 87 }
 88 
 89 bool HTMLSummaryElement::isActiveSummary() const
 90 {
 91     RefPtr<HTMLDetailsElement> details = detailsElement();
 92     if (!details)
 93         return false;
 94     return details->isActiveSummary(*this);
 95 }

 97 static bool isClickableControl(EventTarget* target)
 98 {
 99     if (!is<Element>(target))
100         return false;
101     auto& element = downcast<Element>(*target);
102     return is<HTMLFormControlElement>(element) || is<HTMLFormControlElement>(element.shadowHost());
103 }

105 int HTMLSummaryElement::defaultTabIndex() const
106 {
107     return isActiveSummary() ? 0 : -1;
108 }

110 bool HTMLSummaryElement::supportsFocus() const
111 {
112     return isActiveSummary();

115 void HTMLSummaryElement::defaultEventHandler(Event& event)
116 {
117     if (isActiveSummary() && renderer()) {
118         if (event.type() == eventNames().DOMActivateEvent && !isClickableControl(event.target())) {
119             if (RefPtr<HTMLDetailsElement> details = detailsElement())
120                 details->toggleOpen();
121             event.setDefaultHandled();
122             return;
123         }
124 
125         if (is<KeyboardEvent>(event)) {
126             KeyboardEvent& keyboardEvent = downcast<KeyboardEvent>(event);
127             if (keyboardEvent.type() == eventNames().keydownEvent && keyboardEvent.keyIdentifier() == "U+0020") {
128                 setActive(true, true);
129                 // No setDefaultHandled() - IE dispatches a keypress in this case.
130                 return;
131             }
132             if (keyboardEvent.type() == eventNames().keypressEvent) {
133                 switch (keyboardEvent.charCode()) {
134                 case '\r':
135                     dispatchSimulatedClick(&event);
136                     keyboardEvent.setDefaultHandled();
137                     return;
138                 case ' ':
139                     // Prevent scrolling down the page.
140                     keyboardEvent.setDefaultHandled();
141                     return;
142                 }
143             }
144             if (keyboardEvent.type() == eventNames().keyupEvent && keyboardEvent.keyIdentifier() == "U+0020") {
145                 if (active())
146                     dispatchSimulatedClick(&event);
147                 keyboardEvent.setDefaultHandled();
148                 return;
149             }
150         }
151     }
152 
153     HTMLElement::defaultEventHandler(event);
154 }
155 
156 bool HTMLSummaryElement::willRespondToMouseClickEvents()
157 {
158     if (isActiveSummary() && renderer())
159         return true;
160 
161     return HTMLElement::willRespondToMouseClickEvents();
162 }
