 58 template<typename T> struct Converter<IDLInterface<T>> : DefaultConverter<IDLInterface<T>> {
 59     using ReturnType = typename JSDOMWrapperConverterTraits<T>::ToWrappedReturnType;
 60     using WrapperType = typename JSDOMWrapperConverterTraits<T>::WrapperClass;
 61 
 62     template<typename ExceptionThrower = DefaultExceptionThrower>
 63     static ReturnType convert(JSC::ExecState& state, JSC::JSValue value, ExceptionThrower&& exceptionThrower = ExceptionThrower())
 64     {
 65         auto& vm = state.vm();
 66         auto scope = DECLARE_THROW_SCOPE(vm);
 67         ReturnType object = JSToWrappedOverloader<T>::toWrapped(state, value);
 68         if (UNLIKELY(!object))
 69             exceptionThrower(state, scope);
 70         return object;
 71     }
