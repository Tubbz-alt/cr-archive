 65 EncodedJSValue RuntimeArray::lengthGetter(JSGlobalObject* lexicalGlobalObject, EncodedJSValue thisValue, PropertyName)
 66 {
 67     VM& vm = lexicalGlobalObject->vm();
 68     auto scope = DECLARE_THROW_SCOPE(vm);
 69 
 70     RuntimeArray* thisObject = jsDynamicCast<RuntimeArray*>(vm, JSValue::decode(thisValue));
 71     if (!thisObject)
 72         return throwVMTypeError(lexicalGlobalObject, scope);
 73     return JSValue::encode(jsNumber(thisObject->getLength()));
 74 }

 76 void RuntimeArray::getOwnPropertyNames(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyNameArray& propertyNames, EnumerationMode mode)
 77 {
 78     VM& vm = lexicalGlobalObject->vm();
 79     RuntimeArray* thisObject = jsCast<RuntimeArray*>(object);
 80     unsigned length = thisObject->getLength();
 81     for (unsigned i = 0; i < length; ++i)
 82         propertyNames.add(Identifier::from(vm, i));
 83 
 84     if (mode.includeDontEnumProperties())
 85         propertyNames.add(vm.propertyNames->length);
 86 
 87     JSObject::getOwnPropertyNames(thisObject, lexicalGlobalObject, propertyNames, mode);
 88 }

 90 bool RuntimeArray::getOwnPropertySlot(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot)
 91 {
 92     VM& vm = lexicalGlobalObject->vm();
 93     RuntimeArray* thisObject = jsCast<RuntimeArray*>(object);
 94     if (propertyName == vm.propertyNames->length) {
 95         slot.setCacheableCustom(thisObject, PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum, thisObject->lengthGetter);
 96         return true;
 97     }
 98 
 99     Optional<uint32_t> index = parseIndex(propertyName);
100     if (index && index.value() < thisObject->getLength()) {
101         slot.setValue(thisObject, PropertyAttribute::DontDelete | PropertyAttribute::DontEnum,
102             thisObject->getConcreteArray()->valueAt(lexicalGlobalObject, index.value()));
103         return true;
104     }
105 
106     return JSObject::getOwnPropertySlot(thisObject, lexicalGlobalObject, propertyName, slot);
107 }
108 
109 bool RuntimeArray::getOwnPropertySlotByIndex(JSObject* object, JSGlobalObject* lexicalGlobalObject, unsigned index, PropertySlot& slot)
110 {
111     RuntimeArray* thisObject = jsCast<RuntimeArray*>(object);
112     if (index < thisObject->getLength()) {
113         slot.setValue(thisObject, PropertyAttribute::DontDelete | PropertyAttribute::DontEnum,
114             thisObject->getConcreteArray()->valueAt(lexicalGlobalObject, index));
115         return true;
116     }
117 
118     return JSObject::getOwnPropertySlotByIndex(thisObject, lexicalGlobalObject, index, slot);
119 }

121 bool RuntimeArray::put(JSCell* cell, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, JSValue value, PutPropertySlot& slot)
122 {
123     VM& vm = lexicalGlobalObject->vm();
124     auto scope = DECLARE_THROW_SCOPE(vm);
125 
126     RuntimeArray* thisObject = jsCast<RuntimeArray*>(cell);
127     if (propertyName == vm.propertyNames->length) {
128         throwException(lexicalGlobalObject, scope, createRangeError(lexicalGlobalObject, "Range error"));
129         return false;
130     }
131 
132     if (Optional<uint32_t> index = parseIndex(propertyName))
133         return thisObject->getConcreteArray()->setValueAt(lexicalGlobalObject, index.value(), value);
134 
135     RELEASE_AND_RETURN(scope, JSObject::put(thisObject, lexicalGlobalObject, propertyName, value, slot));
136 }
137 
138 bool RuntimeArray::putByIndex(JSCell* cell, JSGlobalObject* lexicalGlobalObject, unsigned index, JSValue value, bool)
139 {
140     VM& vm = lexicalGlobalObject->vm();
141     auto scope = DECLARE_THROW_SCOPE(vm);
142 
143     RuntimeArray* thisObject = jsCast<RuntimeArray*>(cell);
144     if (index >= thisObject->getLength()) {
145         throwException(lexicalGlobalObject, scope, createRangeError(lexicalGlobalObject, "Range error"));
146         return false;
147     }
148 
149     return thisObject->getConcreteArray()->setValueAt(lexicalGlobalObject, index, value);
150 }
