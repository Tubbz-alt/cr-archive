33 using namespace JSC;

35 namespace Inspector {
36 
37 JSGlobalObjectScriptDebugServer::JSGlobalObjectScriptDebugServer(JSGlobalObject& globalObject)
38     : ScriptDebugServer(globalObject.vm())
39     , m_globalObject(globalObject)
40 {
41 }
42 
43 void JSGlobalObjectScriptDebugServer::attachDebugger()
44 {
45     attach(&m_globalObject);
46 }
47 
48 void JSGlobalObjectScriptDebugServer::detachDebugger(bool isBeingDestroyed)
49 {
50     detach(&m_globalObject, isBeingDestroyed ? Debugger::GlobalObjectIsDestructing : Debugger::TerminatingDebuggingSession);
51     if (!isBeingDestroyed)
52         recompileAllJSFunctions();
53 }
54 
55 void JSGlobalObjectScriptDebugServer::runEventLoopWhilePaused()
56 {
57     // Drop all locks so another thread can work in the VM while we are nested.
58     JSC::JSLock::DropAllLocks dropAllLocks(&m_globalObject.vm());
59 
60     EventLoop loop;
61     while (!m_doneProcessingDebuggerEvents && !loop.ended())
62         loop.cycle();
63 }
