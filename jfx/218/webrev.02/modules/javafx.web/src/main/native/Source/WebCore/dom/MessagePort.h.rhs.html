 28 
 29 #include "ActiveDOMObject.h"
 30 #include "EventTarget.h"
 31 #include "ExceptionOr.h"
 32 #include "MessagePortChannel.h"
 33 #include "MessagePortIdentifier.h"

 35 #include "PostMessageOptions.h"
 36 #include <wtf/WeakPtr.h>
 37 
 38 namespace JSC {
 39 class CallFrame;
 40 class JSObject;
 41 class JSValue;
 42 }
 43 
 44 namespace WebCore {
 45 
 46 class Frame;
 47 
 48 class MessagePort final : public ActiveDOMObject, public EventTargetWithInlineData, public CanMakeWeakPtr<MessagePort, WeakPtrFactoryInitialization::Eager> {
 49     WTF_MAKE_NONCOPYABLE(MessagePort);
 50     WTF_MAKE_ISO_ALLOCATED(MessagePort);
 51 public:
 52     static Ref<MessagePort> create(ScriptExecutionContext&, const MessagePortIdentifier& local, const MessagePortIdentifier& remote);
 53     virtual ~MessagePort();
 54 
 55     ExceptionOr<void> postMessage(JSC::JSGlobalObject&, JSC::JSValue message, PostMessageOptions&&);
 56 
 57     void start();
 58     void close();
 59     void entangle();
 60 
 61     // Returns nullptr if the passed-in vector is empty.
 62     static ExceptionOr<TransferredMessagePortArray> disentanglePorts(Vector<RefPtr<MessagePort>>&&);
 63     static Vector<RefPtr<MessagePort>> entanglePorts(ScriptExecutionContext&, TransferredMessagePortArray&&);
 64 
 65     WEBCORE_EXPORT static bool isExistingMessagePortLocallyReachable(const MessagePortIdentifier&);
 66     WEBCORE_EXPORT static void notifyMessageAvailable(const MessagePortIdentifier&);
 67 
 68     WEBCORE_EXPORT void messageAvailable();
 69     bool started() const { return m_started; }
 70     bool closed() const { return m_closed; }
 71 
 72     void dispatchMessages();
 73 
 74     // Returns null if there is no entangled port, or if the entangled port is run by a different thread.
 75     // This is used solely to enable a GC optimization. Some platforms may not be able to determine ownership
 76     // of the remote port (since it may live cross-process) - those platforms may always return null.
 77     MessagePort* locallyEntangledPort() const;
 78 
 79     const MessagePortIdentifier& identifier() const { return m_identifier; }
 80     const MessagePortIdentifier& remoteIdentifier() const { return m_remoteIdentifier; }
 81 
 82     WEBCORE_EXPORT void ref() const;
 83     WEBCORE_EXPORT void deref() const;
 84 
 85     // ActiveDOMObject
 86     const char* activeDOMObjectName() const final;

 87     void contextDestroyed() final;
 88     void stop() final { close(); }
 89     bool hasPendingActivity() const final;
 90 
 91     WEBCORE_EXPORT bool isLocallyReachable() const;
 92 
 93     // EventTargetWithInlineData.
 94     EventTargetInterface eventTargetInterface() const final { return MessagePortEventTargetInterfaceType; }
 95     ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
 96     void refEventTarget() final { ref(); }
 97     void derefEventTarget() final { deref(); }
 98 
 99     void dispatchEvent(Event&) final;
100 
101 private:
102     explicit MessagePort(ScriptExecutionContext&, const MessagePortIdentifier& local, const MessagePortIdentifier& remote);
103 
104     bool addEventListener(const AtomString& eventType, Ref<EventListener>&&, const AddEventListenerOptions&) final;
105     bool removeEventListener(const AtomString& eventType, EventListener&, const ListenerOptions&) final;
106 
107     void disentangle();
108 
109     void registerLocalActivity();
110 
111     // A port starts out its life entangled, and remains entangled until it is closed or is cloned.
112     bool isEntangled() const { return !m_closed && m_entangled; }
113 
114     void updateActivity(MessagePortChannelProvider::HasActivity);
115 
116     bool m_started { false };
117     bool m_closed { false };
118     bool m_entangled { true };
119 
120     // Flags to manage querying the remote port for GC purposes
121     mutable bool m_mightBeEligibleForGC { false };
122     mutable bool m_hasHadLocalActivitySinceLastCheck { false };
123     mutable bool m_isRemoteEligibleForGC { false };
124     mutable bool m_isAskingRemoteAboutGC { false };
125     bool m_hasMessageEventListener { false };
126 
127     MessagePortIdentifier m_identifier;
128     MessagePortIdentifier m_remoteIdentifier;
129 
130     mutable std::atomic<unsigned> m_refCount { 1 };
131 };
132 
133 } // namespace WebCore
