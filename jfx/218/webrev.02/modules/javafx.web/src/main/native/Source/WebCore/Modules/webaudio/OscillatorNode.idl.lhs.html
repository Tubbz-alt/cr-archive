26 enum OscillatorType {
27     "sine",
28     "square",
29     "sawtooth",
30     "triangle",
31     "custom"
32 };
33 
34 [
35     Conditional=WEB_AUDIO,
36     JSGenerateToJSObject,

37 ] interface OscillatorNode : AudioNode {
38     attribute OscillatorType type;
39 
40     // Playback state constants.
41     const unsigned short UNSCHEDULED_STATE = 0;
42     const unsigned short SCHEDULED_STATE = 1;
43     const unsigned short PLAYING_STATE = 2;
44     const unsigned short FINISHED_STATE = 3;
45 
46     readonly attribute unsigned short playbackState;
47 
48     readonly attribute AudioParam frequency; // in Hertz
49     readonly attribute AudioParam detune; // in Cents
50 
51     [MayThrowException] void start(optional unrestricted double when = 0);
52     [MayThrowException] void stop(optional unrestricted double when = 0);
53 
54     void setPeriodicWave(PeriodicWave? wave); // FIXME: The parameter should not be nullable.
55 
56     attribute EventHandler onended;
57 };
