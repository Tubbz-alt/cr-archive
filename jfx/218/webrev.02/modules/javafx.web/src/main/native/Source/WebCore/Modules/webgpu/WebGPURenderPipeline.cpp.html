59 bool WebGPURenderPipeline::cloneShaderModules(const WebGPUDevice& device)
60 {
61     if (m_vertexShader.module) {
62         bool sharesVertexFragmentShaderModule = m_fragmentShader.module == m_vertexShader.module;
63 
64         const auto& vertexSource = m_vertexShader.module->source();
65         m_vertexShader.module = WebGPUShaderModule::create(GPUShaderModule::tryCreate(device.device(), { vertexSource }), vertexSource);
66 
67         if (m_fragmentShader.module) {
68             if (sharesVertexFragmentShaderModule)
69                 m_fragmentShader.module = m_vertexShader.module;
70             else {
71                 const auto& fragmentSource = m_fragmentShader.module->source();
72                 m_fragmentShader.module = WebGPUShaderModule::create(GPUShaderModule::tryCreate(device.device(), { fragmentSource }), fragmentSource);
73             }
74         }
75 
76         return true;
77     }
78     return false;
79 }
80 
81 bool WebGPURenderPipeline::recompile(const WebGPUDevice& device)
82 {
83     if (m_renderPipeline && m_vertexShader.module) {
84         if (auto* gpuVertexShaderModule = m_vertexShader.module->module()) {
85             GPUProgrammableStageDescriptor vertexStage(makeRef(*gpuVertexShaderModule), { m_vertexShader.entryPoint });
86             Optional<GPUProgrammableStageDescriptor> fragmentStage;
87             if (m_fragmentShader.module) {
88                 if (auto* gpuFragmentShaderModule = m_fragmentShader.module->module())
89                     fragmentStage = GPUProgrammableStageDescriptor(makeRef(*gpuFragmentShaderModule), { m_fragmentShader.entryPoint });
90             }
91             return m_renderPipeline->recompile(device.device(), WTFMove(vertexStage), WTFMove(fragmentStage));
92         }
93     }
94     return false;
95 }
