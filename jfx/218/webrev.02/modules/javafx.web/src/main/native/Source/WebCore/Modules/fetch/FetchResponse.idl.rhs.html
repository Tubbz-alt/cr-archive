29 typedef (Blob or BufferSource or DOMFormData or URLSearchParams or ReadableStream or USVString) BodyInit;
30 typedef (sequence<sequence<ByteString>> or record<ByteString, ByteString>) HeadersInit;
31 
32 enum FetchResponseType { "basic", "cors", "default", "error", "opaque", "opaqueredirect" };
33 
34 dictionary FetchResponseInit {
35     unsigned short status = 200;
36     ByteString statusText = "";
37     HeadersInit headers;
38 };
39 
40 [
41     ActiveDOMObject,
42     Constructor(optional BodyInit? body = null, optional FetchResponseInit init),
43     ConstructorCallWith=ScriptExecutionContext,
44     ConstructorMayThrowException,
45     EnabledAtRuntime=FetchAPI,
46     ExportMacro=WEBCORE_EXPORT,
47     Exposed=(Window,Worker),
48     InterfaceName=Response,
49 ] interface FetchResponse {
50     [CallWith=ScriptExecutionContext, NewObject] static FetchResponse error();
51     [CallWith=ScriptExecutionContext, MayThrowException, NewObject] static FetchResponse redirect(USVString url, optional unsigned short status = 302);
52 
53     readonly attribute FetchResponseType type;
54 
55     readonly attribute USVString url;
56     readonly attribute boolean redirected;
57     readonly attribute unsigned short status;
58     readonly attribute boolean ok;
59     readonly attribute ByteString statusText;
60     readonly attribute FetchHeaders headers; // FIXME: Should be [SameObject].
61     // FIXME: Add support for trailers.
62     // readonly attribute Promise<Headers> trailer;
63 
64     [CallWith=ScriptExecutionContext, MayThrowException, NewObject] FetchResponse clone();
65 };
