40 VMEntryScope::VMEntryScope(VM& vm, JSGlobalObject* globalObject)
41     : m_vm(vm)
42     , m_globalObject(globalObject)
43 {
44     ASSERT(!DisallowVMReentry::isInEffectOnCurrentThread());
45     ASSERT(Thread::current().stack().isGrowingDownward());
46     if (!vm.entryScope) {
47         vm.entryScope = this;
48 
49         // Reset the date cache between JS invocations to force the VM to
50         // observe time zone changes.
51         vm.resetDateCache();
52 
53         if (vm.watchdog())
54             vm.watchdog()->enteredVM();
55 
56 #if ENABLE(SAMPLING_PROFILER)
57         if (SamplingProfiler* samplingProfiler = vm.samplingProfiler())
58             samplingProfiler->noticeVMEntry();
59 #endif
60         if (Options::useTracePoints())
61             tracePoint(VMEntryScopeStart);
62     }
63 
64     vm.clearLastException();
65 }
