 43 JSWeakObjectMapRef JSWeakObjectMapCreate(JSContextRef context, void* privateData, JSWeakMapDestroyedCallback callback)
 44 {
 45     JSGlobalObject* globalObject = toJS(context);
 46     VM& vm = globalObject->vm();
 47     JSLockHolder locker(vm);
 48     auto map = OpaqueJSWeakObjectMap::create(vm, privateData, callback);
 49     globalObject->registerWeakMap(map.ptr());
 50     return map.ptr();
 51 }
 52 
 53 void JSWeakObjectMapSet(JSContextRef ctx, JSWeakObjectMapRef map, void* key, JSObjectRef object)
 54 {
 55     if (!ctx) {
 56         ASSERT_NOT_REACHED();
 57         return;
 58     }
 59     JSGlobalObject* globalObject = toJS(ctx);
 60     VM& vm = globalObject->vm();
 61     JSLockHolder locker(vm);
 62     JSObject* obj = toJS(object);
 63     if (!obj)
 64         return;
 65     ASSERT(obj->inherits<JSProxy>(vm)
 66         || obj->inherits<JSCallbackObject<JSGlobalObject>>(vm)
 67         || obj->inherits<JSCallbackObject<JSNonFinalObject>>(vm));
 68     map->map().set(key, obj);
 69 }
 70 
 71 JSObjectRef JSWeakObjectMapGet(JSContextRef ctx, JSWeakObjectMapRef map, void* key)
 72 {
 73     if (!ctx) {
 74         ASSERT_NOT_REACHED();
 75         return 0;
 76     }
 77     JSGlobalObject* globalObject = toJS(ctx);
 78     JSLockHolder locker(globalObject);
 79     return toRef(jsCast<JSObject*>(map->map().get(key)));
 80 }
 81 
 82 void JSWeakObjectMapRemove(JSContextRef ctx, JSWeakObjectMapRef map, void* key)
 83 {
 84     if (!ctx) {
 85         ASSERT_NOT_REACHED();
 86         return;
 87     }
 88     JSGlobalObject* globalObject = toJS(ctx);
 89     JSLockHolder locker(globalObject);
 90     map->map().remove(key);
 91 }
