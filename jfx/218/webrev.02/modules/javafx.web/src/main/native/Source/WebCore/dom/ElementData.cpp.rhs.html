 39 void ElementData::destroy()
 40 {
 41     if (is<UniqueElementData>(*this))
 42         delete downcast<UniqueElementData>(this);
 43     else
 44         delete downcast<ShareableElementData>(this);
 45 }

 64 static size_t sizeForShareableElementDataWithAttributeCount(unsigned count)
 65 {
 66     return sizeof(ShareableElementData) + sizeof(Attribute) * count;
 67 }
 68 
 69 Ref<ShareableElementData> ShareableElementData::createWithAttributes(const Vector<Attribute>& attributes)
 70 {
 71     void* slot = ShareableElementDataMalloc::malloc(sizeForShareableElementDataWithAttributeCount(attributes.size()));
 72     return adoptRef(*new (NotNull, slot) ShareableElementData(attributes));
 73 }

 75 Ref<UniqueElementData> UniqueElementData::create()
 76 {
 77     return adoptRef(*new UniqueElementData);
 78 }

 80 ShareableElementData::ShareableElementData(const Vector<Attribute>& attributes)
 81     : ElementData(attributes.size())
 82 {
 83     unsigned attributeArraySize = arraySize();
 84     for (unsigned i = 0; i < attributeArraySize; ++i)
 85         new (NotNull, &m_attributeArray[i]) Attribute(attributes[i]);
 86 }
 87 
 88 ShareableElementData::~ShareableElementData()
 89 {
 90     unsigned attributeArraySize = arraySize();
 91     for (unsigned i = 0; i < attributeArraySize; ++i)
 92         m_attributeArray[i].~Attribute();
 93 }
 94 
 95 ShareableElementData::ShareableElementData(const UniqueElementData& other)
 96     : ElementData(other, false)
 97 {
 98     ASSERT(!other.m_presentationAttributeStyle);
 99 
100     if (other.m_inlineStyle) {
101         ASSERT(!other.m_inlineStyle->hasCSSOMWrapper());
102         m_inlineStyle = other.m_inlineStyle->immutableCopyIfNeeded();
103     }
104 
105     unsigned attributeArraySize = arraySize();
106     for (unsigned i = 0; i < attributeArraySize; ++i)
107         new (NotNull, &m_attributeArray[i]) Attribute(other.m_attributeVector.at(i));
108 }

110 inline uint32_t ElementData::arraySizeAndFlagsFromOther(const ElementData& other, bool isUnique)
111 {
112     if (isUnique) {
113         // Set isUnique and ignore arraySize.
114         return (other.m_arraySizeAndFlags | s_flagIsUnique) & s_flagsMask;
115     }
116     // Clear isUnique and set arraySize.
117     return (other.m_arraySizeAndFlags & (s_flagsMask & ~s_flagIsUnique)) | other.length() << s_flagCount;
118 }

141 UniqueElementData::UniqueElementData(const ShareableElementData& other)
142     : ElementData(other, true)
143 {
144     // An ShareableElementData should never have a mutable inline StyleProperties attached.
145     ASSERT(!other.m_inlineStyle || !other.m_inlineStyle->isMutable());
146     m_inlineStyle = other.m_inlineStyle;
147 
148     unsigned otherLength = other.length();
149     m_attributeVector.reserveCapacity(otherLength);
150     for (unsigned i = 0; i < otherLength; ++i)
151         m_attributeVector.uncheckedAppend(other.m_attributeArray[i]);
152 }

154 Ref<UniqueElementData> ElementData::makeUniqueCopy() const
155 {
156     if (isUnique())
157         return adoptRef(*new UniqueElementData(static_cast<const UniqueElementData&>(*this)));
158     return adoptRef(*new UniqueElementData(static_cast<const ShareableElementData&>(*this)));
159 }

161 Ref<ShareableElementData> UniqueElementData::makeShareableCopy() const
162 {
163     void* slot = ShareableElementDataMalloc::malloc(sizeForShareableElementDataWithAttributeCount(m_attributeVector.size()));
164     return adoptRef(*new (NotNull, slot) ShareableElementData(*this));
165 }

167 bool ElementData::isEquivalent(const ElementData* other) const
168 {
169     if (!other)
170         return isEmpty();
171 
172     if (length() != other->length())
173         return false;
174 
175     for (const Attribute& attribute : attributesIterator()) {
176         const Attribute* otherAttr = other->findAttributeByName(attribute.name());
177         if (!otherAttr || attribute.value() != otherAttr->value())
178             return false;
179     }
180 
181     return true;
182 }
183 
184 Attribute* UniqueElementData::findAttributeByName(const QualifiedName& name)
185 {
186     for (auto& attribute : m_attributeVector) {
187         if (attribute.name().matches(name))
188             return &attribute;
189     }
190     return nullptr;
191 }

193 const Attribute* ElementData::findLanguageAttribute() const
194 {
195     ASSERT(XMLNames::langAttr->localName() == HTMLNames::langAttr->localName());
196 
197     const Attribute* attributes = attributeBase();
198     // Spec: xml:lang takes precedence over html:lang -- http://www.w3.org/TR/xhtml1/#C_7
199     const Attribute* languageAttribute = nullptr;
200     for (unsigned i = 0, count = length(); i < count; ++i) {
201         const QualifiedName& name = attributes[i].name();
202         if (name.localName() != HTMLNames::langAttr->localName())
203             continue;
204         if (name.namespaceURI() == XMLNames::langAttr->namespaceURI())
205             return &attributes[i];
206         if (name.namespaceURI() == HTMLNames::langAttr->namespaceURI())
207             languageAttribute = &attributes[i];
208     }
209     return languageAttribute;
210 }
