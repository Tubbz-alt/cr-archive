26 // We keep this function small very carefully to encourage inlining.
27 @globalPrivate
28 function mapIteratorNext(bucket, kind)
29 {
30     "use strict";
31     var value;
32 
33     bucket = @mapBucketNext(bucket);
34     @putByIdDirectPrivate(this, "mapBucket", bucket);
35     var done = bucket === @sentinelMapBucket;
36     if (!done) {
37         var key = @mapBucketKey(bucket);
38         value = @mapBucketValue(bucket);
39         if (kind === @iterationKindEntries)
40             value = [ key, value ]
41         else if (kind === @iterationKindKey)
42             value = key;
43     }
44     return { value, done };
45 }
46 
47 function next()
48 {
49     "use strict";
50 
51     if (@isUndefinedOrNull(this))
52         @throwTypeError("%MapIteratorPrototype%.next requires that |this| not be null or undefined");
53 
54     var bucket = @getByIdDirectPrivate(this, "mapBucket");
55     if (bucket === @undefined)
56         @throwTypeError("%MapIteratorPrototype%.next requires that |this| be a Map Iterator instance");
57     return @mapIteratorNext.@call(this, bucket, @getByIdDirectPrivate(this, "mapIteratorKind"));
58 }
