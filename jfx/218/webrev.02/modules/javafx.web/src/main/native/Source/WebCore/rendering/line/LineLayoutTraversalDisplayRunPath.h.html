 36 static FloatPoint linePosition(float left, float top)
 37 {
 38     return FloatPoint(left, roundf(top));
 39 }
 40 
 41 class DisplayRunPath {
 42 public:
 43     DisplayRunPath(const Display::InlineContent& inlineContent, size_t startIndex, size_t endIndex)
 44         : m_inlineContent(&inlineContent)
 45         , m_endIndex(endIndex)
 46         , m_runIndex(startIndex)
 47     { }
 48     DisplayRunPath(DisplayRunPath&&) = default;
 49     DisplayRunPath(const DisplayRunPath&) = default;
 50     DisplayRunPath& operator=(const DisplayRunPath&) = default;
 51     DisplayRunPath& operator=(DisplayRunPath&&) = default;
 52 
 53     FloatRect rect() const;
 54 
 55     float baselineOffset() const { return lineBox().baselineOffset(); }
 56 
 57     bool isLeftToRightDirection() const { return true; }
 58     bool isHorizontal() const { return true; }
 59     bool dirOverride() const { return false; }
 60     bool isLineBreak() const { return run().isLineBreak(); }
 61 
 62     bool useLineBreakBoxRenderTreeDumpQuirk() const
 63     {
 64         if (!m_runIndex)
 65             return false;
 66         auto& previous = runs()[m_runIndex - 1];
 67         return previous.lineIndex() == run().lineIndex();
 68     }
 69 
 70     bool hasHyphen() const { return run().textContext()->needsHyphen(); }
 71     StringView text() const { return run().textContext()->content(); }
 72     unsigned localStartOffset() const { return run().textContext()->start(); }
 73     unsigned localEndOffset() const { return run().textContext()->end(); }
 74     unsigned length() const { return run().textContext()->length(); }
 75 
 76     bool isLastOnLine() const
 77     {
 78         if (isLast())
 79             return true;
 80         auto& next = runs()[m_runIndex + 1];
 81         return run().lineIndex() != next.lineIndex();
 82     }
 83     bool isLast() const
 84     {
 85         return m_runIndex + 1 == m_endIndex;
 86     };
 87 
 88     void traverseNextTextBoxInVisualOrder()
 89     {
 90         ASSERT(!atEnd());
 91         ++m_runIndex;
 92     }
 93     void traverseNextTextBoxInTextOrder() {  traverseNextTextBoxInVisualOrder(); }
 94 
 95     bool operator==(const DisplayRunPath& other) const { return m_inlineContent == other.m_inlineContent && m_runIndex == other.m_runIndex; }
 96     bool atEnd() const { return m_runIndex == m_endIndex; }
 97 
 98 private:
 99     const Display::InlineContent::Runs& runs() const { return m_inlineContent->runs; }
100     const Display::Run& run() const { return runs()[m_runIndex]; }
101     const Display::LineBox& lineBox() const { return m_inlineContent->lineBoxForRun(run()); }
102 
103     RefPtr<const Display::InlineContent> m_inlineContent;
104     size_t m_endIndex { 0 };
105     size_t m_runIndex { 0 };
106 };
107 
108 inline FloatRect DisplayRunPath::rect() const
109 {
110     auto rect = run().rect();
111     auto position = linePosition(rect.left(), rect.top());
112     return { position, rect.size() };
113 }
114 
115 }
