20 #pragma once
21 
22 #include "SVGGraphicsElement.h"
23 #include "SVGNames.h"
24 #include "SVGURIReference.h"
25 
26 namespace WebCore {
27 
28 class SVGForeignObjectElement final : public SVGGraphicsElement {
29     WTF_MAKE_ISO_ALLOCATED(SVGForeignObjectElement);
30 public:
31     static Ref<SVGForeignObjectElement> create(const QualifiedName&, Document&);
32 
33     const SVGLengthValue& x() const { return m_x->currentValue(); }
34     const SVGLengthValue& y() const { return m_y->currentValue(); }
35     const SVGLengthValue& width() const { return m_width->currentValue(); }
36     const SVGLengthValue& height() const { return m_height->currentValue(); }
37 
38     SVGAnimatedLength& xAnimated() { return m_x; }
39     SVGAnimatedLength& yAnimated() { return m_y; }
40     SVGAnimatedLength& widthAnimated() { return m_width; }
41     SVGAnimatedLength& heightAnimated() { return m_height; }
42 
43 private:
44     SVGForeignObjectElement(const QualifiedName&, Document&);
45 
46     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGForeignObjectElement, SVGGraphicsElement>;
47     const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
48 
49     void parseAttribute(const QualifiedName&, const AtomString&) final;
50     void svgAttributeChanged(const QualifiedName&) final;
51 
52     bool rendererIsNeeded(const RenderStyle&) final;
53     bool childShouldCreateRenderer(const Node&) const final;
54     RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) final;
55 
56     bool isValid() const final { return SVGTests::isValid(); }
57     bool selfHasRelativeLengths() const final { return true; }
58 
59     PropertyRegistry m_propertyRegistry { *this };
60     Ref<SVGAnimatedLength> m_x { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
61     Ref<SVGAnimatedLength> m_y { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
62     Ref<SVGAnimatedLength> m_width { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
63     Ref<SVGAnimatedLength> m_height { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
64 };
65 
66 } // namespace WebCore
