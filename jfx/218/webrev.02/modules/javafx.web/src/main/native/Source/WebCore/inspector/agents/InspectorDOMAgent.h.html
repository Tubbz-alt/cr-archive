280     struct InspectorEventListener {
281         int identifier { 1 };
282         RefPtr<EventTarget> eventTarget;
283         RefPtr<EventListener> eventListener;
284         AtomString eventType;
285         bool useCapture { false };
286         bool disabled { false };
287         bool hasBreakpoint { false };
288 
289         InspectorEventListener() { }
290 
291         InspectorEventListener(int identifier, EventTarget& target, const AtomString& type, EventListener& listener, bool capture)
292             : identifier(identifier)
293             , eventTarget(&target)
294             , eventListener(&listener)
295             , eventType(type)
296             , useCapture(capture)
297         {
298         }
299 
300         bool matches(EventTarget& target, const AtomString& type, EventListener& listener, bool capture)
301         {
302             if (eventTarget.get() != &target)
303                 return false;
304             if (eventListener.get() != &listener)
305                 return false;
306             if (eventType != type)
307                 return false;
308             if (useCapture != capture)
309                 return false;
310             return true;
311         }
312     };
