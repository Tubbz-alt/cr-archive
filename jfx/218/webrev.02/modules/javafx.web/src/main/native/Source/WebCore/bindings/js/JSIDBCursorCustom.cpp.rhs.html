39 JSC::JSValue JSIDBCursor::key(JSC::JSGlobalObject& lexicalGlobalObject) const
40 {
41     return cachedPropertyValue(lexicalGlobalObject, *this, wrapped().keyWrapper(), [&] {
42         return toJS(lexicalGlobalObject, lexicalGlobalObject, wrapped().key());
43     });
44 }
45 
46 JSC::JSValue JSIDBCursor::primaryKey(JSC::JSGlobalObject& lexicalGlobalObject) const
47 {
48     return cachedPropertyValue(lexicalGlobalObject, *this, wrapped().primaryKeyWrapper(), [&] {
49         return toJS(lexicalGlobalObject, lexicalGlobalObject, wrapped().primaryKey());
50     });
51 }
52 
53 void JSIDBCursor::visitAdditionalChildren(SlotVisitor& visitor)
54 {
55     auto& cursor = wrapped();
56     if (auto* request = cursor.request())
57         visitor.addOpaqueRoot(request);
58     cursor.keyWrapper().visit(visitor);
59     cursor.primaryKeyWrapper().visit(visitor);
60 }
61 
62 JSValue toJSNewlyCreated(JSC::JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<IDBCursor>&& cursor)
63 {
64     if (is<IDBCursorWithValue>(cursor))
65         return createWrapper<IDBCursorWithValue>(globalObject, WTFMove(cursor));
66     return createWrapper<IDBCursor>(globalObject, WTFMove(cursor));
67 }
68 
69 JSValue toJS(JSC::JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, IDBCursor& cursor)
70 {
71     return wrap(lexicalGlobalObject, globalObject, cursor);
72 }
