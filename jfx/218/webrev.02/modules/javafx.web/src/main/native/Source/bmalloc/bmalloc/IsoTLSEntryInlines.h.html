32 template<typename Func>
33 void IsoTLSEntry::walkUpToInclusive(IsoTLSEntry* last, const Func& func)
34 {
35     IsoTLSEntry* current = this;
36     for (;;) {
37         func(current);
38         if (current == last)
39             return;
40         current = current->m_next;
41     }
42 }
43 
44 template<typename EntryType>
45 DefaultIsoTLSEntry<EntryType>::DefaultIsoTLSEntry()
46     : IsoTLSEntry(sizeof(EntryType))
47 {
48     static_assert(sizeof(EntryType) <= UINT32_MAX);
49     static_assert(sizeof(void*) == alignof(EntryType), "Because IsoTLSEntry includes vtable, it should be the same to the pointer");
50 }
51 
52 template<typename EntryType>
53 void DefaultIsoTLSEntry<EntryType>::move(void* passedSrc, void* dst)
54 {
55     EntryType* src = static_cast<EntryType*>(passedSrc);
56     new (dst) EntryType(std::move(*src));
57     src->~EntryType();
58 }
59 
60 template<typename EntryType>
61 void DefaultIsoTLSEntry<EntryType>::destruct(void* passedEntry)
62 {
63     EntryType* entry = static_cast<EntryType*>(passedEntry);
64     entry->~EntryType();
65 }
