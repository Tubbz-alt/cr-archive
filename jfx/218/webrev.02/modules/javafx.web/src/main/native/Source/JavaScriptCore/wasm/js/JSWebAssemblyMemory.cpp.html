 40 JSWebAssemblyMemory* JSWebAssemblyMemory::tryCreate(JSGlobalObject* globalObject, VM& vm, Structure* structure)
 41 {
 42     auto throwScope = DECLARE_THROW_SCOPE(vm);
 43 
 44     auto exception = [&] (JSObject* error) {
 45         throwException(globalObject, throwScope, error);
 46         return nullptr;
 47     };
 48 
 49     if (!globalObject->webAssemblyEnabled())
 50         return exception(createEvalError(globalObject, globalObject->webAssemblyDisabledErrorMessage()));
 51 
 52     auto* memory = new (NotNull, allocateCell<JSWebAssemblyMemory>(vm.heap)) JSWebAssemblyMemory(vm, structure);
 53     memory->finishCreation(vm);
 54     return memory;
 55 }

 57 void JSWebAssemblyMemory::adopt(Ref<Wasm::Memory>&& memory)
 58 {
 59     m_memory.swap(memory);
 60     ASSERT(m_memory->refCount() == 1);
 61     m_memory->check();
 62 }

 75 JSArrayBuffer* JSWebAssemblyMemory::buffer(VM& vm, JSGlobalObject* globalObject)
 76 {
 77     if (m_bufferWrapper)
 78         return m_bufferWrapper.get();
 79 
 80     // We can't use a ref here since it doesn't have a copy constructor...
 81     Ref<Wasm::Memory> protectedMemory = m_memory.get();
 82     auto destructor = createSharedTask<void(void*)>([protectedMemory = WTFMove(protectedMemory)] (void*) { });
 83     m_buffer = ArrayBuffer::createFromBytes(memory().memory(), memory().size(), WTFMove(destructor));
 84     m_buffer->makeWasmMemory();
 85     m_bufferWrapper.set(vm, this, JSArrayBuffer::create(vm, globalObject->arrayBufferStructure(ArrayBufferSharingMode::Default), m_buffer.get()));
 86     RELEASE_ASSERT(m_bufferWrapper);
 87     return m_bufferWrapper.get();
 88 }

 90 Wasm::PageCount JSWebAssemblyMemory::grow(VM& vm, JSGlobalObject* globalObject, uint32_t delta)
 91 {
 92     auto throwScope = DECLARE_THROW_SCOPE(vm);
 93 
 94     auto grown = memory().grow(Wasm::PageCount(delta));
 95     if (!grown) {
 96         switch (grown.error()) {
 97         case Wasm::Memory::GrowFailReason::InvalidDelta:
 98             throwException(globalObject, throwScope, createRangeError(globalObject, "WebAssembly.Memory.grow expects the delta to be a valid page count"_s));
 99             break;
100         case Wasm::Memory::GrowFailReason::InvalidGrowSize:
101             throwException(globalObject, throwScope, createRangeError(globalObject, "WebAssembly.Memory.grow expects the grown size to be a valid page count"_s));
102             break;
103         case Wasm::Memory::GrowFailReason::WouldExceedMaximum:
104             throwException(globalObject, throwScope, createRangeError(globalObject, "WebAssembly.Memory.grow would exceed the memory's declared maximum size"_s));
105             break;
106         case Wasm::Memory::GrowFailReason::OutOfMemory:
107             throwException(globalObject, throwScope, createOutOfMemoryError(globalObject));
108             break;
109         }
110         return Wasm::PageCount();
111     }
112 
113     return grown.value();
114 }

116 void JSWebAssemblyMemory::growSuccessCallback(VM& vm, Wasm::PageCount oldPageCount, Wasm::PageCount newPageCount)
117 {
118     // We need to clear out the old array buffer because it might now be pointing to stale memory.
119     // Neuter the old array.
120     if (m_buffer) {
121         m_buffer->neuter(vm);
122         m_buffer = nullptr;
123         m_bufferWrapper.clear();
124     }
125 
126     memory().check();
127 
128     vm.heap.reportExtraMemoryAllocated(newPageCount.bytes() - oldPageCount.bytes());
129 }
130 
131 void JSWebAssemblyMemory::finishCreation(VM& vm)
132 {
133     Base::finishCreation(vm);
134     ASSERT(inherits(vm, info()));
135     vm.heap.reportExtraMemoryAllocated(memory().size());
136 }

138 void JSWebAssemblyMemory::destroy(JSCell* cell)
139 {
140     auto memory = static_cast<JSWebAssemblyMemory*>(cell);
141     memory->JSWebAssemblyMemory::~JSWebAssemblyMemory();
142 }
143 
144 void JSWebAssemblyMemory::visitChildren(JSCell* cell, SlotVisitor& visitor)
145 {
146     auto* thisObject = jsCast<JSWebAssemblyMemory*>(cell);
147     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
148 
149     Base::visitChildren(thisObject, visitor);
150     visitor.append(thisObject->m_bufferWrapper);
151     visitor.reportExtraMemoryVisited(thisObject->memory().size());
152 }
