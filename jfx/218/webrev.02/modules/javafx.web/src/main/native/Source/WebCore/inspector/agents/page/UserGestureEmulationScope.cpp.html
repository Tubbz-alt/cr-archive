31 #include "config.h"
32 #include "UserGestureEmulationScope.h"
33 
34 #include "Chrome.h"
35 #include "ChromeClient.h"
36 #include "Page.h"
37 #include "UserGestureIndicator.h"
38 #include <wtf/Optional.h>
39 
40 namespace WebCore {
41 
42 UserGestureEmulationScope::UserGestureEmulationScope(Page& inspectedPage, bool emulateUserGesture)
43     : m_pageChromeClient(inspectedPage.chrome().client())
44     , m_gestureIndicator(emulateUserGesture ? Optional<ProcessingUserGestureState>(ProcessingUserGesture) : WTF::nullopt)
45     , m_emulateUserGesture(emulateUserGesture)
46     , m_userWasInteracting(false)
47 {
48     if (m_emulateUserGesture) {
49         m_userWasInteracting = m_pageChromeClient.userIsInteracting();
50         if (!m_userWasInteracting)
51             m_pageChromeClient.setUserIsInteracting(true);
52     }
53 }
54 
55 UserGestureEmulationScope::~UserGestureEmulationScope()
56 {
57     if (m_emulateUserGesture && !m_userWasInteracting && m_pageChromeClient.userIsInteracting())
58         m_pageChromeClient.setUserIsInteracting(false);
59 }
60 
61 } // namespace WebCore
