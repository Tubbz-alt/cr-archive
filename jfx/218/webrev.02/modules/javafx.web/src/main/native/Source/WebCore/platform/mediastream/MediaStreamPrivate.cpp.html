 48 Ref<MediaStreamPrivate> MediaStreamPrivate::create(Ref<const Logger>&& logger, Ref<RealtimeMediaSource>&& source)
 49 {
 50     auto loggerCopy = logger.copyRef();
 51     return MediaStreamPrivate::create(WTFMove(logger), MediaStreamTrackPrivateVector::from(MediaStreamTrackPrivate::create(WTFMove(loggerCopy), WTFMove(source))));
 52 }
 53 
 54 Ref<MediaStreamPrivate> MediaStreamPrivate::create(Ref<const Logger>&& logger, const Vector<Ref<RealtimeMediaSource>>& audioSources, const Vector<Ref<RealtimeMediaSource>>& videoSources)
 55 {
 56     MediaStreamTrackPrivateVector tracks;
 57     tracks.reserveInitialCapacity(audioSources.size() + videoSources.size());
 58 
 59     for (auto& source : audioSources)
 60         tracks.uncheckedAppend(MediaStreamTrackPrivate::create(logger.copyRef(), source.copyRef()));
 61 
 62     for (auto& source : videoSources)
 63         tracks.uncheckedAppend(MediaStreamTrackPrivate::create(logger.copyRef(), source.copyRef()));
 64 
 65     return MediaStreamPrivate::create(WTFMove(logger), tracks);
 66 }

 68 MediaStreamPrivate::MediaStreamPrivate(Ref<const Logger>&& logger, const MediaStreamTrackPrivateVector& tracks, String&& id)
 69     : m_id(WTFMove(id))
 70 #if !RELEASE_LOG_DISABLED
 71     , m_logger(WTFMove(logger))
 72     , m_logIdentifier(uniqueLogIdentifier())
 73 #endif
 74 {
 75     UNUSED_PARAM(logger);
 76     ASSERT(!m_id.isEmpty());
 77 
 78     for (auto& track : tracks) {
 79         track->addObserver(*this);
 80         m_trackSet.add(track->id(), track);
 81     }
 82 
 83     updateActiveState(NotifyClientOption::DontNotify);
 84 }
 85 
 86 MediaStreamPrivate::~MediaStreamPrivate()
 87 {
 88     for (auto& track : m_trackSet.values())
 89         track->removeObserver(*this);
 90 }

 92 void MediaStreamPrivate::addObserver(MediaStreamPrivate::Observer& observer)
 93 {
 94     RELEASE_ASSERT(isMainThread());
 95     m_observers.add(&observer);
 96 }
 97 
 98 void MediaStreamPrivate::removeObserver(MediaStreamPrivate::Observer& observer)
 99 {
100     RELEASE_ASSERT(isMainThread());
101     m_observers.remove(&observer);
102 }
103 
104 void MediaStreamPrivate::forEachObserver(const WTF::Function<void(Observer&)>& apply) const
105 {
106     RELEASE_ASSERT(isMainThread());
107     for (auto* observer : copyToVector(m_observers)) {
108         if (!m_observers.contains(observer))
109             continue;
110         apply(*observer);
111     }
112 }

119 void MediaStreamPrivate::updateActiveState(NotifyClientOption notifyClientOption)
120 {
121     bool newActiveState = false;
122     for (auto& track : m_trackSet.values()) {
123         if (!track->ended()) {
124             newActiveState = true;
125             break;
126         }
127     }
128 
129     updateActiveVideoTrack();
130 
131     // A stream is active if it has at least one un-ended track.
132     if (newActiveState == m_isActive)
133         return;
134 
135     m_isActive = newActiveState;
136 
137     if (notifyClientOption == NotifyClientOption::Notify) {
138         forEachObserver([](auto& observer) {
139             observer.activeStatusChanged();
140         });
141     }
142 }

144 void MediaStreamPrivate::addTrack(RefPtr<MediaStreamTrackPrivate>&& track, NotifyClientOption notifyClientOption)
145 {
146     if (m_trackSet.contains(track->id()))
147         return;
148 
149     ALWAYS_LOG(LOGIDENTIFIER, track->logIdentifier());
150 
151     track->addObserver(*this);
152     m_trackSet.add(track->id(), track);
153 
154     if (notifyClientOption == NotifyClientOption::Notify) {
155         forEachObserver([&track](auto& observer) {
156             observer.didAddTrack(*track.get());
157         });
158     }
159 
160     updateActiveState(notifyClientOption);
161     characteristicsChanged();
162 }
163 
164 void MediaStreamPrivate::removeTrack(MediaStreamTrackPrivate& track, NotifyClientOption notifyClientOption)
165 {
166     if (!m_trackSet.remove(track.id()))
167         return;
168 
169     ALWAYS_LOG(LOGIDENTIFIER, track.logIdentifier());
170     track.removeObserver(*this);
171 
172     if (notifyClientOption == NotifyClientOption::Notify) {
173         forEachObserver([&track](auto& observer) {
174             observer.didRemoveTrack(track);
175         });
176     }
177 
178     updateActiveState(NotifyClientOption::Notify);
179     characteristicsChanged();
180 }

196 bool MediaStreamPrivate::isProducingData() const
197 {
198     for (auto& track : m_trackSet.values()) {
199         if (track->isProducingData())
200             return true;
201     }
202     return false;
203 }
204 
205 bool MediaStreamPrivate::hasVideo() const
206 {
207     for (auto& track : m_trackSet.values()) {
208         if (track->type() == RealtimeMediaSource::Type::Video && track->isActive())
209             return true;
210     }
211     return false;
212 }
213 
214 bool MediaStreamPrivate::hasAudio() const
215 {
216     for (auto& track : m_trackSet.values()) {
217         if (track->type() == RealtimeMediaSource::Type::Audio && track->isActive())
218             return true;
219     }
220     return false;
221 }
222 
223 bool MediaStreamPrivate::muted() const
224 {
225     for (auto& track : m_trackSet.values()) {
226         if (!track->muted() && !track->ended())
227             return false;
228     }
229     return true;
230 }

232 FloatSize MediaStreamPrivate::intrinsicSize() const
233 {
234     FloatSize size;
235 
236     if (m_activeVideoTrack) {
237         const RealtimeMediaSourceSettings& setting = m_activeVideoTrack->settings();
238         size.setWidth(setting.width());
239         size.setHeight(setting.height());
240     }
241 
242     return size;
243 }
244 
245 void MediaStreamPrivate::updateActiveVideoTrack()
246 {
247     m_activeVideoTrack = nullptr;
248     for (auto& track : m_trackSet.values()) {
249         if (!track->ended() && track->type() == RealtimeMediaSource::Type::Video) {
250             m_activeVideoTrack = track.get();
251             break;
252         }
253     }
254 }

311 void MediaStreamPrivate::monitorOrientation(OrientationNotifier& notifier)
312 {
313     for (auto& track : m_trackSet.values()) {
314         if (track->source().isCaptureSource() && track->type() == RealtimeMediaSource::Type::Video)
315             track->source().monitorOrientation(notifier);
316     }
317 }
