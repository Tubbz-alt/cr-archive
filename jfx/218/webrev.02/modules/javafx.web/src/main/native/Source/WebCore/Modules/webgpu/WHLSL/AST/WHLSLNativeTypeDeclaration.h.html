 30 #include "WHLSLCodeLocation.h"
 31 #include "WHLSLNamedType.h"

 37 namespace WebCore {
 38 
 39 namespace WHLSL {
 40 
 41 namespace AST {
 42 
 43 class NativeTypeDeclaration final : public NamedType {
 44     WTF_MAKE_FAST_ALLOCATED;
 45 public:
 46     NativeTypeDeclaration(CodeLocation location, String&& name, TypeArguments&& typeArguments)
 47         : NamedType(Kind::NativeTypeDeclaration, location, WTFMove(name))
 48         , m_typeArguments(WTFMove(typeArguments))
 49     {
 50     }
 51 
 52     ~NativeTypeDeclaration() = default;
 53 
 54     NativeTypeDeclaration(const NativeTypeDeclaration&) = delete;
 55     NativeTypeDeclaration(NativeTypeDeclaration&&) = default;
 56 
 57     TypeArguments& typeArguments() { return m_typeArguments; }
 58 
 59     bool isInt() const { return m_isInt; }
 60     bool isNumber() const { return m_isNumber; }
 61     bool isFloating() const { return m_isFloating; }
 62     bool isAtomic() const { return m_isAtomic; }
 63     bool isVector() const { return m_isVector; }
 64     bool isMatrix() const { return m_isMatrix; }
 65     bool isOpaqueType() const { return m_isOpaqueType; }
 66     bool isTexture() const { return m_isTexture; }
 67     bool isTextureArray() const { return m_isTextureArray; }
 68     bool isDepthTexture() const { return m_isDepthTexture; }
 69     bool isWritableTexture() const { return m_isWritableTexture; }
 70     bool isCubeTexture() const { return m_isCubeTexture; }
 71     uint textureDimension() const { return m_textureDimension; }
 72     bool isSigned() const { return m_isSigned; }
 73     const std::function<bool(int)>& canRepresentInteger() const { return m_canRepresentInteger; }
 74     const std::function<bool(unsigned)>& canRepresentUnsignedInteger() const { return m_canRepresentUnsignedInteger; }
 75     const std::function<bool(float)>& canRepresentFloat() const { return m_canRepresentFloat; }
 76     const std::function<int64_t(int64_t)>& successor() const { return m_successor; }
 77     const std::function<int64_t(int)>& formatValueFromInteger() const { return m_formatValueFromInteger; }
 78     const std::function<int64_t(unsigned)>& formatValueFromUnsignedInteger() const { return m_formatValueFromUnsignedInteger; }
 79     void iterateAllValues(const std::function<bool(int64_t)>& callback) { m_iterateAllValues(callback); }
 80 private:
 81     unsigned matrixDimension(unsigned typeArgumentIndex)
 82     {
 83         ASSERT(isMatrix());
 84         ASSERT(typeArguments().size() == 3);
 85         return WTF::get<AST::ConstantExpression>(typeArguments()[typeArgumentIndex]).integerLiteral().value();
 86     }
 87 public:
 88     unsigned numberOfMatrixRows()
 89     {
 90         return matrixDimension(2);
 91     }
 92     unsigned numberOfMatrixColumns()
 93     {
 94         return matrixDimension(1);
 95     }
 96 
 97     TypeReference& vectorTypeArgument()
 98     {
 99         ASSERT(isVector());
100         return WTF::get<Ref<AST::TypeReference>>(typeArguments()[0]);
101     }
102 
103     unsigned vectorSize()
104     {
105         ASSERT(isVector());
106         return WTF::get<AST::ConstantExpression>(typeArguments()[1]).integerLiteral().value();
107     }
108 
109     TypeReference& matrixTypeArgument()
110     {
111         ASSERT(isMatrix());
112         return WTF::get<Ref<AST::TypeReference>>(typeArguments()[0]);
113     }
114 
115     void setIsInt() { m_isInt = true; }
116     void setIsNumber() { m_isNumber = true; }
117     void setIsFloating() { m_isFloating = true; }
118     void setIsAtomic() { m_isAtomic = true; }
119     void setIsVector() { m_isVector = true; }
120     void setIsMatrix() { m_isMatrix = true; }
121     void setIsOpaqueType() { m_isOpaqueType = true; }
122     void setIsTexture() { m_isTexture = true; }
123     void setIsTextureArray() { m_isTextureArray = true; }
124     void setIsDepthTexture() { m_isDepthTexture = true; }
125     void setIsWritableTexture() { m_isWritableTexture = true; }
126     void setIsCubeTexture() { m_isCubeTexture = true; }
127     void setTextureDimension(uint textureDimension) { m_textureDimension = textureDimension; }
128     void setIsSigned() { m_isSigned = true; }
129     void setCanRepresentInteger(std::function<bool(int)>&& canRepresent) { m_canRepresentInteger = WTFMove(canRepresent); }
130     void setCanRepresentUnsignedInteger(std::function<bool(unsigned)>&& canRepresent) { m_canRepresentUnsignedInteger = WTFMove(canRepresent); }
131     void setCanRepresentFloat(std::function<bool(float)>&& canRepresent) { m_canRepresentFloat = WTFMove(canRepresent); }
132     void setSuccessor(std::function<int64_t(int64_t)>&& successor) { m_successor = WTFMove(successor); }
133     void setFormatValueFromInteger(std::function<int64_t(int)>&& formatValue) { m_formatValueFromInteger = WTFMove(formatValue); }
134     void setFormatValueFromUnsignedInteger(std::function<int64_t(unsigned)>&& formatValue) { m_formatValueFromUnsignedInteger = WTFMove(formatValue); }
135     void setIterateAllValues(std::function<void(const std::function<bool(int64_t)>&)>&& iterateAllValues) { m_iterateAllValues = WTFMove(iterateAllValues); }
136 
137 private:
138     TypeArguments m_typeArguments;
139     std::function<bool(int)> m_canRepresentInteger;
140     std::function<bool(unsigned)> m_canRepresentUnsignedInteger;
141     std::function<bool(float)> m_canRepresentFloat;
142     std::function<int64_t(int64_t)> m_successor;
143     std::function<int64_t(int)> m_formatValueFromInteger;
144     std::function<int64_t(unsigned)> m_formatValueFromUnsignedInteger;
145     std::function<void(const std::function<bool(int64_t)>&)> m_iterateAllValues;
146     uint m_textureDimension { 0 };
147     bool m_isInt { false };
148     bool m_isNumber { false };
149     bool m_isFloating { false };
150     bool m_isAtomic { false };
151     bool m_isVector { false };
152     bool m_isMatrix { false };
153     bool m_isOpaqueType { false };
154     bool m_isTexture { false };
155     bool m_isTextureArray { false };
156     bool m_isDepthTexture { false };
157     bool m_isWritableTexture { false };
158     bool m_isCubeTexture { false };
159     bool m_isSigned { false };
160 };
161 
162 } // namespace AST

164 }
