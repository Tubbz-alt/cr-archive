 25 #ifndef HRTFPanner_h
 26 #define HRTFPanner_h
 27 
 28 #include "DelayDSPKernel.h"
 29 #include "FFTConvolver.h"

 30 #include "Panner.h"
 31 
 32 namespace WebCore {
 33 
 34 class HRTFDatabaseLoader;
 35 
 36 class HRTFPanner final : public Panner {
 37 public:
 38     explicit HRTFPanner(float sampleRate, HRTFDatabaseLoader*);
 39     virtual ~HRTFPanner();
 40 
 41     // Panner
 42     void pan(double azimuth, double elevation, const AudioBus* inputBus, AudioBus* outputBus, size_t framesToProcess) override;
 43     void reset() override;
 44 
 45     size_t fftSize() const { return fftSizeForSampleRate(m_sampleRate); }
 46     static size_t fftSizeForSampleRate(float sampleRate);
 47 
 48     float sampleRate() const { return m_sampleRate; }
 49 
 50     double tailTime() const override;
 51     double latencyTime() const override;
 52 
 53 private:
 54     // Given an azimuth angle in the range -180 -> +180, returns the corresponding azimuth index for the database,
 55     // and azimuthBlend which is an interpolation value from 0 -> 1.
 56     int calculateDesiredAzimuthIndexAndBlend(double azimuth, double& azimuthBlend);
 57 
 58     RefPtr<HRTFDatabaseLoader> m_databaseLoader;
 59 
 60     float m_sampleRate;
 61 
 62     // We maintain two sets of convolvers for smooth cross-faded interpolations when
 63     // then azimuth and elevation are dynamically changing.
 64     // When the azimuth and elevation are not changing, we simply process with one of the two sets.
 65     // Initially we use CrossfadeSelection1 corresponding to m_convolverL1 and m_convolverR1.
 66     // Whenever the azimuth or elevation changes, a crossfade is initiated to transition
 67     // to the new position. So if we're currently processing with CrossfadeSelection1, then
 68     // we transition to CrossfadeSelection2 (and vice versa).
 69     // If we're in the middle of a transition, then we wait until it is complete before
 70     // initiating a new transition.
 71 
 72     // Selects either the convolver set (m_convolverL1, m_convolverR1) or (m_convolverL2, m_convolverR2).
 73     enum CrossfadeSelection {
 74         CrossfadeSelection1,
 75         CrossfadeSelection2
 76     };
 77 
 78     CrossfadeSelection m_crossfadeSelection;
 79 
 80     // azimuth/elevation for CrossfadeSelection1.
 81     int m_azimuthIndex1;
 82     double m_elevation1;
 83 
 84     // azimuth/elevation for CrossfadeSelection2.
 85     int m_azimuthIndex2;
 86     double m_elevation2;
 87 
 88     // A crossfade value 0 <= m_crossfadeX <= 1.
 89     float m_crossfadeX;
 90 
 91     // Per-sample-frame crossfade value increment.
 92     float m_crossfadeIncr;
 93 
 94     FFTConvolver m_convolverL1;
 95     FFTConvolver m_convolverR1;
 96     FFTConvolver m_convolverL2;
 97     FFTConvolver m_convolverR2;
 98 
 99     DelayDSPKernel m_delayLineL;
100     DelayDSPKernel m_delayLineR;
101 
102     AudioFloatArray m_tempL1;
103     AudioFloatArray m_tempR1;
104     AudioFloatArray m_tempL2;
105     AudioFloatArray m_tempR2;
106 };
107 
108 } // namespace WebCore
109 
110 #endif // HRTFPanner_h
