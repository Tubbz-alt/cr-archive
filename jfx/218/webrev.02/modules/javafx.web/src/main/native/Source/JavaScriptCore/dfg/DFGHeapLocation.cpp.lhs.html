 33 void HeapLocation::dump(PrintStream& out) const
 34 {
 35     out.print(m_kind, ":", m_heap);
 36 
 37     if (!m_base)
 38         return;
 39 
 40     out.print("[", m_base);
 41     if (m_index)
 42         out.print(", ", m_index);
 43     out.print("]");
 44 }
 45 
 46 } } // namespace JSC::DFG
