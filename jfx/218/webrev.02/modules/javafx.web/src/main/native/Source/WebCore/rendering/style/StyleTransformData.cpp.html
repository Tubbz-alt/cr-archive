22 #include "config.h"
23 #include "StyleTransformData.h"
24 
25 #include "RenderStyle.h"
26 
27 namespace WebCore {
28 
29 DEFINE_ALLOCATOR_WITH_HEAP_IDENTIFIER(StyleTransformData);
30 
31 StyleTransformData::StyleTransformData()
32     : operations(RenderStyle::initialTransform())
33     , x(RenderStyle::initialTransformOriginX())
34     , y(RenderStyle::initialTransformOriginY())
35     , z(RenderStyle::initialTransformOriginZ())
36     , transformBox(RenderStyle::initialTransformBox())
37 {
38 }
39 
40 inline StyleTransformData::StyleTransformData(const StyleTransformData& other)
41     : RefCounted<StyleTransformData>()
42     , operations(other.operations)
43     , x(other.x)
44     , y(other.y)
45     , z(other.z)
46     , transformBox(other.transformBox)
47 {
48 }
49 
50 Ref<StyleTransformData> StyleTransformData::copy() const
51 {
52     return adoptRef(*new StyleTransformData(*this));
53 }
54 
55 bool StyleTransformData::operator==(const StyleTransformData& other) const
56 {
57     return x == other.x && y == other.y && z == other.z && transformBox == other.transformBox && operations == other.operations;
58 }
59 
60 } // namespace WebCore
