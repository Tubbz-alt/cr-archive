31 namespace WebCore {
32 
33 CSSLineBoxContainValue::CSSLineBoxContainValue(OptionSet<LineBoxContain> value)
34     : CSSValue(LineBoxContainClass)
35     , m_value(value)
36 {
37 }
38 
39 String CSSLineBoxContainValue::customCSSText() const
40 {
41     StringBuilder text;
42 
43     if (m_value.contains(LineBoxContain::Block))
44         text.appendLiteral("block");
45 
46     if (m_value.contains(LineBoxContain::Inline)) {
47         if (!text.isEmpty())
48             text.append(' ');
49         text.appendLiteral("inline");
50     }
51     if (m_value.contains(LineBoxContain::Font)) {
52         if (!text.isEmpty())
53             text.append(' ');
54         text.appendLiteral("font");
55     }
56     if (m_value.contains(LineBoxContain::Glyphs)) {
57         if (!text.isEmpty())
58             text.append(' ');
59         text.appendLiteral("glyphs");
60     }
61     if (m_value.contains(LineBoxContain::Replaced)) {
62         if (!text.isEmpty())
63             text.append(' ');
64         text.appendLiteral("replaced");
65     }
66     if (m_value.contains(LineBoxContain::InlineBox)) {
67         if (!text.isEmpty())
68             text.append(' ');
69         text.appendLiteral("inline-box");
70     }
71     if (m_value.contains(LineBoxContain::InitialLetter)) {
72         if (!text.isEmpty())
73             text.append(' ');
74         text.appendLiteral("initial-letter");
75     }
76 
77     return text.toString();
78 }
