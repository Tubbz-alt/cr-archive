 37 using ClassChangeVector = Vector<AtomStringImpl*, 4>;
 38 
 39 static ClassChangeVector collectClasses(const SpaceSplitString& classes)
 40 {
 41     ClassChangeVector result;
 42     result.reserveCapacity(classes.size());
 43     for (unsigned i = 0; i < classes.size(); ++i)
 44         result.uncheckedAppend(classes[i].impl());
 45     return result;
 46 }
 47 
 48 static ClassChangeVector computeClassChange(const SpaceSplitString& oldClasses, const SpaceSplitString& newClasses)
 49 {
 50     unsigned oldSize = oldClasses.size();
 51     unsigned newSize = newClasses.size();
 52 
 53     if (!oldSize)
 54         return collectClasses(newClasses);
 55     if (!newSize)
 56         return collectClasses(oldClasses);
 57 
 58     ClassChangeVector changedClasses;
 59 
 60     BitVector remainingClassBits;
 61     remainingClassBits.ensureSize(oldSize);
 62     // Class vectors tend to be very short. This is faster than using a hash table.
 63     for (unsigned i = 0; i < newSize; ++i) {
 64         bool foundFromBoth = false;
 65         for (unsigned j = 0; j < oldSize; ++j) {
 66             if (newClasses[i] == oldClasses[j]) {
 67                 remainingClassBits.quickSet(j);
 68                 foundFromBoth = true;
 69             }
 70         }
 71         if (foundFromBoth)
 72             continue;
 73         changedClasses.append(newClasses[i].impl());
 74     }
 75     for (unsigned i = 0; i < oldSize; ++i) {
 76         // If the bit is not set the corresponding class has been removed.
 77         if (remainingClassBits.quickGet(i))
 78             continue;
 79         changedClasses.append(oldClasses[i].impl());
 80     }
 81 
 82     return changedClasses;
 83 }
 84 
 85 void ClassChangeInvalidation::computeInvalidation(const SpaceSplitString& oldClasses, const SpaceSplitString& newClasses)
 86 {
 87     auto changedClasses = computeClassChange(oldClasses, newClasses);
 88 
 89     bool shouldInvalidateCurrent = false;
 90     bool mayAffectStyleInShadowTree = false;
 91 
 92     traverseRuleFeatures(m_element, [&] (const RuleFeatureSet& features, bool mayAffectShadowTree) {
 93         for (auto* changedClass : changedClasses) {
 94             if (mayAffectShadowTree && features.classRules.contains(changedClass))
 95                 mayAffectStyleInShadowTree = true;
 96             if (features.classesAffectingHost.contains(changedClass))
 97                 shouldInvalidateCurrent = true;
 98         }
 99     });
100 
101     if (mayAffectStyleInShadowTree) {
102         // FIXME: We should do fine-grained invalidation for shadow tree.
103         m_element.invalidateStyleForSubtree();
104     }
105 
106     if (shouldInvalidateCurrent)
107         m_element.invalidateStyle();
108 
109     auto& ruleSets = m_element.styleResolver().ruleSets();
110 
111     for (auto* changedClass : changedClasses) {
112         if (auto* invalidationRuleSets = ruleSets.classInvalidationRuleSets(changedClass)) {
113             for (auto& invalidationRuleSet : *invalidationRuleSets)
114                 Invalidator::addToMatchElementRuleSets(m_matchElementRuleSets, invalidationRuleSet);
115         }
116     }
117 }
