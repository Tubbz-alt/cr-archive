45 
46 WorkerScriptDebugServer::WorkerScriptDebugServer(WorkerGlobalScope& context)
47     : ScriptDebugServer(context.script()->vm())
48     , m_workerGlobalScope(context)
49 {
50 }
51 
52 void WorkerScriptDebugServer::attachDebugger()
53 {
54     m_workerGlobalScope.script()->attachDebugger(this);
55 }
56 
57 void WorkerScriptDebugServer::detachDebugger(bool isBeingDestroyed)
58 {
59     if (m_workerGlobalScope.script())
60         m_workerGlobalScope.script()->detachDebugger(this);
61     if (!isBeingDestroyed)
62         recompileAllJSFunctions();
63 }
64 
65 void WorkerScriptDebugServer::recompileAllJSFunctions()
66 {
67     JSC::JSLockHolder lock(vm());
68     JSC::Debugger::recompileAllJSFunctions();
69 }
70 
71 void WorkerScriptDebugServer::runEventLoopWhilePaused()
72 {
73     TimerBase::fireTimersInNestedEventLoop();
74 
75     MessageQueueWaitResult result;
76     do {
77         result = m_workerGlobalScope.thread().runLoop().runInDebuggerMode(m_workerGlobalScope);
78     } while (result != MessageQueueTerminated && !m_doneProcessingDebuggerEvents);
79 }
80 
81 void WorkerScriptDebugServer::reportException(JSC::ExecState* exec, JSC::Exception* exception) const
82 {
83     WebCore::reportException(exec, exception);
84 }
