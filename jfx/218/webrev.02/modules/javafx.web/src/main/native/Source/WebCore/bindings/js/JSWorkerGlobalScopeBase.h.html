 50 class JSWorkerGlobalScopeBase : public JSDOMGlobalObject {
 51 public:
 52     using Base = JSDOMGlobalObject;
 53 
 54     template<typename, JSC::SubspaceAccess>
 55     static void subspaceFor(JSC::VM&) { RELEASE_ASSERT_NOT_REACHED(); }
 56 
 57     static void destroy(JSC::JSCell*);
 58 
 59     DECLARE_INFO;
 60 
 61     WorkerGlobalScope& wrapped() const { return *m_wrapped; }
 62     JSC::JSProxy* proxy() const { ASSERT(m_proxy); return m_proxy.get(); }
 63     ScriptExecutionContext* scriptExecutionContext() const;
 64 
 65     static JSC::Structure* createStructure(JSC::VM& vm, JSC::JSGlobalObject* globalObject, JSC::JSValue prototype)
 66     {
 67         return JSC::Structure::create(vm, globalObject, prototype, JSC::TypeInfo(JSC::GlobalObjectType, StructureFlags), info());
 68     }
 69 
 70     static const JSC::GlobalObjectMethodTable s_globalObjectMethodTable;
 71 
 72     static bool supportsRichSourceInfo(const JSC::JSGlobalObject*);
 73     static bool shouldInterruptScript(const JSC::JSGlobalObject*);
 74     static bool shouldInterruptScriptBeforeTimeout(const JSC::JSGlobalObject*);
 75     static JSC::RuntimeFlags javaScriptRuntimeFlags(const JSC::JSGlobalObject*);
 76     static void queueMicrotaskToEventLoop(JSC::JSGlobalObject&, Ref<JSC::Microtask>&&);
 77 
 78     void clearDOMGuardedObjects();
 79 
 80 protected:
 81     JSWorkerGlobalScopeBase(JSC::VM&, JSC::Structure*, RefPtr<WorkerGlobalScope>&&);
 82     void finishCreation(JSC::VM&, JSC::JSProxy*);
 83 
 84     static void visitChildren(JSC::JSCell*, JSC::SlotVisitor&);
 85 
 86 private:
 87     RefPtr<WorkerGlobalScope> m_wrapped;
 88     JSC::WriteBarrier<JSC::JSProxy> m_proxy;
 89 };

 92 // Always ignores the execState and passed globalObject, WorkerGlobalScope is itself a globalObject and will always use its own prototype chain.
