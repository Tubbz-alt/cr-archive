28 #if ENABLE(WEBGPU)
29 
30 #include "WebGPUPipeline.h"
31 #include <wtf/Forward.h>
32 
33 namespace WebCore {
34 
35 class GPUPipeline;
36 class GPURenderPipeline;
37 class GPUErrorScopes;
38 class WebGPUDevice;
39 class WebGPUShaderModule;
40 
41 class WebGPURenderPipeline final : public WebGPUPipeline {
42 public:
43     virtual ~WebGPURenderPipeline();
44 
45     static Ref<WebGPURenderPipeline> create(WebGPUDevice&, RefPtr<GPURenderPipeline>&&, GPUErrorScopes&, WebGPUPipeline::ShaderData&& vertexShader, WebGPUPipeline::ShaderData&& fragmentShader);
46 
47     bool isRenderPipeline() const { return true; }
48 
49     bool isValid() const { return renderPipeline(); }
50     const GPURenderPipeline* renderPipeline() const { return m_renderPipeline.get(); }
51     RefPtr<WebGPUShaderModule> vertexShader() const { return m_vertexShader.module; }
52     RefPtr<WebGPUShaderModule> fragmentShader() const { return m_fragmentShader.module; }
53 
54     bool cloneShaderModules(const WebGPUDevice&);
55     bool recompile(const WebGPUDevice&);
56 
57 private:
58     WebGPURenderPipeline(WebGPUDevice&, RefPtr<GPURenderPipeline>&&, GPUErrorScopes&, WebGPUPipeline::ShaderData&& vertexShader, WebGPUPipeline::ShaderData&& fragmentShader);
59 
60     RefPtr<GPURenderPipeline> m_renderPipeline;
61 
62     // Preserved for Web Inspector recompilation.
63     WebGPUPipeline::ShaderData m_vertexShader;
64     WebGPUPipeline::ShaderData m_fragmentShader;
65 };
66 
67 } // namespace WebCore

69 SPECIALIZE_TYPE_TRAITS_WEBGPUPIPELINE(WebCore::WebGPURenderPipeline, isRenderPipeline())
