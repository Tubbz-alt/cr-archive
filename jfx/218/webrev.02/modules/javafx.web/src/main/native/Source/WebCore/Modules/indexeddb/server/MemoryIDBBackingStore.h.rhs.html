 40 class MemoryObjectStore;
 41 
 42 class MemoryIDBBackingStore final : public IDBBackingStore {
 43     WTF_MAKE_FAST_ALLOCATED;
 44 public:


 45     MemoryIDBBackingStore(PAL::SessionID, const IDBDatabaseIdentifier&);
 46     ~MemoryIDBBackingStore();
 47 
 48     IDBError getOrEstablishDatabaseInfo(IDBDatabaseInfo&) final;
 49     void setDatabaseInfo(const IDBDatabaseInfo&);
 50 
 51     void removeObjectStoreForVersionChangeAbort(MemoryObjectStore&);
 52     void restoreObjectStoreForVersionChangeAbort(Ref<MemoryObjectStore>&&);
 53 
 54 private:
 55     IDBError beginTransaction(const IDBTransactionInfo&) final;
 56     IDBError abortTransaction(const IDBResourceIdentifier& transactionIdentifier) final;
 57     IDBError commitTransaction(const IDBResourceIdentifier& transactionIdentifier) final;
 58     IDBError createObjectStore(const IDBResourceIdentifier& transactionIdentifier, const IDBObjectStoreInfo&) final;
 59     IDBError deleteObjectStore(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier) final;
 60     IDBError renameObjectStore(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, const String& newName) final;
 61     IDBError clearObjectStore(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier) final;
 62     IDBError createIndex(const IDBResourceIdentifier& transactionIdentifier, const IDBIndexInfo&) final;
 63     IDBError deleteIndex(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, uint64_t indexIdentifier) final;
 64     IDBError renameIndex(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, uint64_t indexIdentifier, const String& newName) final;
 65     IDBError keyExistsInObjectStore(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, const IDBKeyData&, bool& keyExists) final;
 66     IDBError deleteRange(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, const IDBKeyRangeData&) final;
 67     IDBError addRecord(const IDBResourceIdentifier& transactionIdentifier, const IDBObjectStoreInfo&, const IDBKeyData&, const IDBValue&) final;
 68     IDBError getRecord(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, const IDBKeyRangeData&, IDBGetRecordDataType, IDBGetResult& outValue) final;
 69     IDBError getAllRecords(const IDBResourceIdentifier& transactionIdentifier, const IDBGetAllRecordsData&, IDBGetAllResult& outValue) final;
 70     IDBError getIndexRecord(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, uint64_t indexIdentifier, IndexedDB::IndexRecordType, const IDBKeyRangeData&, IDBGetResult& outValue) final;
 71     IDBError getCount(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, uint64_t indexIdentifier, const IDBKeyRangeData&, uint64_t& outCount) final;
 72     IDBError generateKeyNumber(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, uint64_t& keyNumber) final;
 73     IDBError revertGeneratedKeyNumber(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, uint64_t keyNumber) final;
 74     IDBError maybeUpdateKeyGeneratorNumber(const IDBResourceIdentifier& transactionIdentifier, uint64_t objectStoreIdentifier, double newKeyNumber) final;
 75     IDBError openCursor(const IDBResourceIdentifier& transactionIdentifier, const IDBCursorInfo&, IDBGetResult& outResult) final;
 76     IDBError iterateCursor(const IDBResourceIdentifier& transactionIdentifier, const IDBResourceIdentifier& cursorIdentifier, const IDBIterateCursorData&, IDBGetResult& outResult) final;

 77 
 78     IDBObjectStoreInfo* infoForObjectStore(uint64_t objectStoreIdentifier) final;
 79     void deleteBackingStore() final;
 80 
 81     bool supportsSimultaneousTransactions() final { return true; }
 82     bool isEphemeral() final { return true; }
 83 



 84     bool hasTransaction(const IDBResourceIdentifier& identifier) const final { return m_transactions.contains(identifier); }
 85 

 86     RefPtr<MemoryObjectStore> takeObjectStoreByIdentifier(uint64_t identifier);
 87 

 88     void close() final;
 89 
 90     void registerObjectStore(Ref<MemoryObjectStore>&&);
 91     void unregisterObjectStore(MemoryObjectStore&);
 92 
 93     IDBDatabaseIdentifier m_identifier;
 94     PAL::SessionID m_sessionID;
 95     std::unique_ptr<IDBDatabaseInfo> m_databaseInfo;
 96 
 97     HashMap<IDBResourceIdentifier, std::unique_ptr<MemoryBackingStoreTransaction>> m_transactions;
 98 


 99     HashMap<uint64_t, RefPtr<MemoryObjectStore>> m_objectStoresByIdentifier;
100     HashMap<String, MemoryObjectStore*> m_objectStoresByName;
101 };
