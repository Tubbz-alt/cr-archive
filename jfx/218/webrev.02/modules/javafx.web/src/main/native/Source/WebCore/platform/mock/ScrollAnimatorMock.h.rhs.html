37 // A Mock implementation of ScrollAnimator used to test the scroll events
38 // received by the scroll animator. Tests can enable this scroll animator using
39 // the internal setting enableMockScrollAnimator.
40 class ScrollAnimatorMock final : public ScrollAnimator {
41 public:
42     ScrollAnimatorMock(ScrollableArea&, WTF::Function<void(const String&)>&&);
43     virtual ~ScrollAnimatorMock();
44 
45 private:
46 
47 #if ENABLE(RUBBER_BANDING)
48     bool allowsHorizontalStretching(const PlatformWheelEvent&) const override { return false; }
49     bool allowsVerticalStretching(const PlatformWheelEvent&) const override { return false; }
50     IntSize stretchAmount() const override { return IntSize(); }
51     bool pinnedInDirection(const FloatSize&) const override { return false; }
52     bool canScrollHorizontally() const override { return false; }
53     bool canScrollVertically() const override { return false; }
54     bool shouldRubberBandInDirection(ScrollDirection) const override { return false; }
55     void immediateScrollBy(const FloatSize&) override { }
56     void immediateScrollByWithoutContentEdgeConstraints(const FloatSize&) override { }
57     void adjustScrollPositionToBoundsIfNecessary() override { }
58 #endif
59 
60     void didAddVerticalScrollbar(Scrollbar*) override;
61     void didAddHorizontalScrollbar(Scrollbar*) override;
62     void willRemoveVerticalScrollbar(Scrollbar*) override;
63     void willRemoveHorizontalScrollbar(Scrollbar*) override;
64     void mouseEnteredContentArea() override;
65     void mouseMovedInContentArea() override;
66     void mouseExitedContentArea() override;
67     void mouseEnteredScrollbar(Scrollbar*) const override;
68     void mouseExitedScrollbar(Scrollbar*) const override;
69     void mouseIsDownInScrollbar(Scrollbar*, bool) const override;
70 
71     WTF::Function<void(const String&)> m_logger;
72     Scrollbar* m_verticalScrollbar { nullptr };
73     Scrollbar* m_horizontalScrollbar { nullptr };
74 };
75 
76 } // namespace WebCore
