26 #pragma once
27 
28 #if ENABLE(JIT)
29 
30 #include "MacroAssembler.h"
31 #include <wtf/Vector.h>
32 #include <wtf/text/CString.h>
33 
34 namespace JSC {
35 
36 class CodeBlock;
37 class LinkBuffer;
38 
39 namespace Profiler {
40 class Compilation;
41 }
42 
43 class JITDisassembler {
44     WTF_MAKE_FAST_ALLOCATED;
45 public:
46     JITDisassembler(CodeBlock*);
47     ~JITDisassembler();
48 
49     void setStartOfCode(MacroAssembler::Label label) { m_startOfCode = label; }
50     void setForBytecodeMainPath(unsigned bytecodeIndex, MacroAssembler::Label label)
51     {
52         m_labelForBytecodeIndexInMainPath[bytecodeIndex] = label;
53     }
54     void setForBytecodeSlowPath(unsigned bytecodeIndex, MacroAssembler::Label label)
55     {
56         m_labelForBytecodeIndexInSlowPath[bytecodeIndex] = label;
57     }
58     void setEndOfSlowPath(MacroAssembler::Label label) { m_endOfSlowPath = label; }
59     void setEndOfCode(MacroAssembler::Label label) { m_endOfCode = label; }
60 
61     void dump(LinkBuffer&);
62     void dump(PrintStream&, LinkBuffer&);
63     void reportToProfiler(Profiler::Compilation*, LinkBuffer&);
64 
65 private:
66     void dumpHeader(PrintStream&, LinkBuffer&);
67     MacroAssembler::Label firstSlowLabel();
68 
69     struct DumpedOp {
70         unsigned index;
71         CString disassembly;
72     };
73     Vector<DumpedOp> dumpVectorForInstructions(LinkBuffer&, const char* prefix, Vector<MacroAssembler::Label>& labels, MacroAssembler::Label endLabel);
74 
75     void dumpForInstructions(PrintStream&, LinkBuffer&, const char* prefix, Vector<MacroAssembler::Label>& labels, MacroAssembler::Label endLabel);
76     void reportInstructions(Profiler::Compilation*, LinkBuffer&, const char* prefix, Vector<MacroAssembler::Label>& labels, MacroAssembler::Label endLabel);
77 
78     void dumpDisassembly(PrintStream&, LinkBuffer&, MacroAssembler::Label from, MacroAssembler::Label to);
79 
80     CodeBlock* m_codeBlock;
81     MacroAssembler::Label m_startOfCode;
82     Vector<MacroAssembler::Label> m_labelForBytecodeIndexInMainPath;
83     Vector<MacroAssembler::Label> m_labelForBytecodeIndexInSlowPath;
84     MacroAssembler::Label m_endOfSlowPath;
85     MacroAssembler::Label m_endOfCode;
86 };
87 
88 } // namespace JSC
89 
90 #endif // ENABLE(JIT)
