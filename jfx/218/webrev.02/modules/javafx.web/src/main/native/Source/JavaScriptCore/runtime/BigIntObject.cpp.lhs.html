37 const ClassInfo BigIntObject::s_info = { "BigInt", &Base::s_info, nullptr, nullptr, CREATE_METHOD_TABLE(BigIntObject) };
38 
39 
40 BigIntObject* BigIntObject::create(VM& vm, JSGlobalObject* globalObject, JSBigInt* bigInt)
41 {
42     BigIntObject* object = new (NotNull, allocateCell<BigIntObject>(vm.heap)) BigIntObject(vm, globalObject->bigIntObjectStructure());
43     object->finishCreation(vm, bigInt);
44     return object;
45 }
46 
47 BigIntObject::BigIntObject(VM& vm, Structure* structure)
48     : Base(vm, structure)
49 {
50 }
51 
52 void BigIntObject::finishCreation(VM& vm, JSBigInt* bigInt)
53 {
54     Base::finishCreation(vm);
55     ASSERT(inherits(vm, info()));
56     setInternalValue(vm, bigInt);
57 }
58 
59 String BigIntObject::toStringName(const JSObject*, ExecState*)
60 {
61     return "Object"_s;
62 }
63 
64 JSValue BigIntObject::defaultValue(const JSObject* object, ExecState*, PreferredPrimitiveType)
65 {
66     const BigIntObject* bigIntObject = jsCast<const BigIntObject*>(object);
67     return bigIntObject->internalValue();
68 }
