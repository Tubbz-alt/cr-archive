 40 namespace WebCore {
 41 
 42 class FloatRect;
 43 
 44 class InspectorFrontendClient {
 45 public:
 46     enum class DockSide {
 47         Undocked = 0,
 48         Right,
 49         Left,
 50         Bottom,
 51     };
 52 
 53     virtual ~InspectorFrontendClient() = default;
 54 
 55     WEBCORE_EXPORT virtual void windowObjectCleared() = 0;
 56     virtual void frontendLoaded() = 0;
 57 
 58     virtual void startWindowDrag() = 0;
 59     virtual void moveWindowBy(float x, float y) = 0;
 60 
 61     // Information about the debuggable.
 62     virtual bool isRemote() const = 0;
 63     virtual String localizedStringsURL() const = 0;
 64     virtual String backendCommandsURL() const = 0;
 65     virtual Inspector::DebuggableType debuggableType() const = 0;
 66     virtual String targetPlatformName() const = 0;
 67     virtual String targetBuildVersion() const = 0;
 68     virtual String targetProductVersion() const = 0;
 69     virtual bool targetIsSimulator() const = 0;
 70     virtual unsigned inspectionLevel() const = 0;


 71 
 72     virtual void bringToFront() = 0;
 73     virtual void closeWindow() = 0;
 74     virtual void reopen() = 0;
 75     virtual void resetState() = 0;
 76 
 77     virtual UserInterfaceLayoutDirection userInterfaceLayoutDirection() const = 0;
 78 
 79     WEBCORE_EXPORT virtual void requestSetDockSide(DockSide) = 0;
 80     WEBCORE_EXPORT virtual void changeAttachedWindowHeight(unsigned) = 0;
 81     WEBCORE_EXPORT virtual void changeAttachedWindowWidth(unsigned) = 0;
 82 
 83     WEBCORE_EXPORT virtual void changeSheetRect(const FloatRect&) = 0;
 84 
 85     WEBCORE_EXPORT virtual void openInNewTab(const String& url) = 0;
 86 
 87     virtual bool canSave() = 0;
 88     virtual void save(const WTF::String& url, const WTF::String& content, bool base64Encoded, bool forceSaveAs) = 0;
 89     virtual void append(const WTF::String& url, const WTF::String& content) = 0;
 90 
 91     virtual void inspectedURLChanged(const String&) = 0;
 92     virtual void showCertificate(const CertificateInfo&) = 0;
 93 
 94 #if ENABLE(INSPECTOR_TELEMETRY)
 95     virtual bool supportsDiagnosticLogging() { return false; }
 96     virtual bool diagnosticLoggingAvailable() { return false; }
 97     virtual void logDiagnosticEvent(const String& /* eventName */, const DiagnosticLoggingClient::ValueDictionary&) { }
 98 #endif
 99 
100     virtual void pagePaused() { }
101     virtual void pageUnpaused() { }
102 
103     WEBCORE_EXPORT virtual void sendMessageToBackend(const String&) = 0;
104 
105     WEBCORE_EXPORT virtual bool isUnderTest() = 0;
106 };

108 } // namespace WebCore
