24 #pragma once
25 
26 #include "EventModifierInit.h"
27 #include "PlatformEvent.h"
28 #include "UIEvent.h"
29 
30 namespace WebCore {
31 
32 class UIEventWithKeyState : public UIEvent {
33     WTF_MAKE_ISO_ALLOCATED(UIEventWithKeyState);
34 public:
35     using Modifier = PlatformEvent::Modifier;
36 
37     bool ctrlKey() const { return m_modifiers.contains(Modifier::ControlKey); }
38     bool shiftKey() const { return m_modifiers.contains(Modifier::ShiftKey); }
39     bool altKey() const { return m_modifiers.contains(Modifier::AltKey); }
40     bool metaKey() const { return m_modifiers.contains(Modifier::MetaKey); }
41     bool altGraphKey() const { return m_modifiers.contains(Modifier::AltGraphKey); }
42     bool capsLockKey() const { return m_modifiers.contains(Modifier::CapsLockKey); }
43 
44     OptionSet<Modifier> modifierKeys() const { return m_modifiers; }
45 
46     WEBCORE_EXPORT bool getModifierState(const String& keyIdentifier) const;
47 
48 protected:
49     UIEventWithKeyState() = default;
50 
51     UIEventWithKeyState(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, IsComposed isComposed,
52         RefPtr<WindowProxy>&& view, int detail, OptionSet<Modifier> modifiers)
53         : UIEvent(type, canBubble, cancelable, isComposed, WTFMove(view), detail)
54         , m_modifiers(modifiers)
55     {
56     }
57 
58     UIEventWithKeyState(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, IsComposed isComposed,
59         MonotonicTime timestamp, RefPtr<WindowProxy>&& view, int detail, OptionSet<Modifier> modifiers, IsTrusted isTrusted)
60         : UIEvent(type, canBubble, cancelable, isComposed, timestamp, WTFMove(view), detail, isTrusted)
61         , m_modifiers(modifiers)
62     {
63     }
64 
65     UIEventWithKeyState(const AtomString& type, const EventModifierInit& initializer)
66         : UIEvent(type, initializer)
67         , m_modifiers(modifiersFromInitializer(initializer))
68     {
69     }
70 
71     void setModifierKeys(bool ctrlKey, bool altKey, bool shiftKey, bool metaKey, bool altGraphKey = false);
72 
73 private:
74     OptionSet<Modifier> m_modifiers;
75 
76     static OptionSet<Modifier> modifiersFromInitializer(const EventModifierInit& initializer);
77 };
78 
79 UIEventWithKeyState* findEventWithKeyState(Event*);
80 
81 } // namespace WebCore
