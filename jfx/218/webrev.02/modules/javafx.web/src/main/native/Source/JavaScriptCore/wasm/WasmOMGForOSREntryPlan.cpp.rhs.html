 56 OMGForOSREntryPlan::OMGForOSREntryPlan(Context* context, Ref<Module>&& module, Ref<Callee>&& callee, uint32_t functionIndex, uint32_t loopIndex, MemoryMode mode, CompletionTask&& task)
 57     : Base(context, makeRef(const_cast<ModuleInformation&>(module->moduleInformation())), WTFMove(task))
 58     , m_module(WTFMove(module))
 59     , m_codeBlock(*m_module->codeBlockFor(mode))
 60     , m_callee(WTFMove(callee))
 61     , m_functionIndex(functionIndex)
 62     , m_loopIndex(loopIndex)
 63 {
 64     setMode(mode);
 65     ASSERT(m_codeBlock->runnable());
 66     ASSERT(m_codeBlock.ptr() == m_module->codeBlockFor(m_mode));
 67     dataLogLnIf(WasmOMGForOSREntryPlanInternal::verbose, "Starting OMGForOSREntry plan for ", functionIndex, " of module: ", RawPointer(&m_module.get()));
 68 }

 70 void OMGForOSREntryPlan::work(CompilationEffort)
 71 {
 72     ASSERT(m_codeBlock->runnable());
 73     ASSERT(m_codeBlock.ptr() == m_module->codeBlockFor(mode()));
 74     const FunctionData& function = m_moduleInformation->functions[m_functionIndex];
 75 
 76     const uint32_t functionIndexSpace = m_functionIndex + m_module->moduleInformation().importFunctionCount();
 77     ASSERT(functionIndexSpace < m_module->moduleInformation().functionIndexSpaceSize());
 78 
 79     SignatureIndex signatureIndex = m_moduleInformation->internalFunctionSignatureIndices[m_functionIndex];
 80     const Signature& signature = SignatureInformation::get(signatureIndex);

 81 
 82     Vector<UnlinkedWasmToWasmCall> unlinkedCalls;
 83     CompilationContext context;
 84     unsigned osrEntryScratchBufferSize = 0;
 85     auto parseAndCompileResult = parseAndCompile(context, function, signature, unlinkedCalls, osrEntryScratchBufferSize, m_moduleInformation.get(), m_mode, CompilationMode::OMGForOSREntryMode, m_functionIndex, m_loopIndex);
 86 
 87     if (UNLIKELY(!parseAndCompileResult)) {
 88         fail(holdLock(m_lock), makeString(parseAndCompileResult.error(), "when trying to tier up ", String::number(m_functionIndex)));
 89         return;
 90     }
 91 
 92     Entrypoint omgEntrypoint;
 93     LinkBuffer linkBuffer(*context.wasmEntrypointJIT, nullptr, JITCompilationCanFail);
 94     if (UNLIKELY(linkBuffer.didFailToAllocate())) {
 95         Base::fail(holdLock(m_lock), makeString("Out of executable memory while tiering up function at index ", String::number(m_functionIndex)));
 96         return;
 97     }
 98 
 99     omgEntrypoint.compilation = makeUnique<B3::Compilation>(
100         FINALIZE_WASM_CODE_FOR_MODE(CompilationMode::OMGForOSREntryMode, linkBuffer, B3CompilationPtrTag, "WebAssembly OMGForOSREntry function[%i] %s name %s", m_functionIndex, signature.toString().ascii().data(), makeString(IndexOrName(functionIndexSpace, m_moduleInformation->nameSection->get(functionIndexSpace))).ascii().data()),
101         WTFMove(context.wasmEntrypointByproducts));
102 
103     omgEntrypoint.calleeSaveRegisters = WTFMove(parseAndCompileResult.value()->entrypoint.calleeSaveRegisters);
104 
105     MacroAssemblerCodePtr<WasmEntryPtrTag> entrypoint;
106     ASSERT(m_codeBlock.ptr() == m_module->codeBlockFor(mode()));
107     Ref<OMGForOSREntryCallee> callee = OMGForOSREntryCallee::create(WTFMove(omgEntrypoint), functionIndexSpace, m_moduleInformation->nameSection->get(functionIndexSpace), osrEntryScratchBufferSize, m_loopIndex, WTFMove(unlinkedCalls));
108     {
109         MacroAssembler::repatchPointer(parseAndCompileResult.value()->calleeMoveLocation, CalleeBits::boxWasm(callee.ptr()));
110         entrypoint = callee->entrypoint();
111 
112         auto locker = holdLock(m_codeBlock->m_lock);
113         for (auto& call : callee->wasmToWasmCallsites()) {
114             MacroAssemblerCodePtr<WasmEntryPtrTag> entrypoint;
115             if (call.functionIndexSpace < m_module->moduleInformation().importFunctionCount())
116                 entrypoint = m_codeBlock->m_wasmToWasmExitStubs[call.functionIndexSpace].code();
117             else
118                 entrypoint = m_codeBlock->wasmEntrypointCalleeFromFunctionIndexSpace(call.functionIndexSpace).entrypoint().retagged<WasmEntryPtrTag>();
119 
120             MacroAssembler::repatchNearCall(call.callLocation, CodeLocationLabel<WasmEntryPtrTag>(entrypoint));
121         }
122     }
123     resetInstructionCacheOnAllThreads();
124     WTF::storeStoreFence();
125     {
126         auto locker = holdLock(m_codeBlock->m_lock);
127         {
128             switch (m_callee->compilationMode()) {
129             case CompilationMode::LLIntMode: {
130                 LLIntCallee* llintCallee = static_cast<LLIntCallee*>(m_callee.ptr());
131                 auto locker = holdLock(llintCallee->tierUpCounter().m_lock);
132                 llintCallee->setOSREntryCallee(callee.copyRef());
133                 llintCallee->tierUpCounter().m_loopCompilationStatus = LLIntTierUpCounter::CompilationStatus::Compiled;
134                 break;
135             }
136             case CompilationMode::BBQMode: {
137                 BBQCallee* bbqCallee = static_cast<BBQCallee*>(m_callee.ptr());
138                 auto locker = holdLock(bbqCallee->tierUpCount()->getLock());
139                 bbqCallee->setOSREntryCallee(callee.copyRef());
140                 bbqCallee->tierUpCount()->osrEntryTriggers()[m_loopIndex] = TierUpCount::TriggerReason::CompilationDone;
141                 bbqCallee->tierUpCount()->m_compilationStatusForOMGForOSREntry = TierUpCount::CompilationStatus::Compiled;
142                 break;
143             }
144             default:
145                 RELEASE_ASSERT_NOT_REACHED();
146             }
147         }
148         WTF::storeStoreFence();
149         // It is possible that a new OMG callee is added while we release m_codeBlock->lock.
150         // Until we add OMGForOSREntry callee to BBQCallee's m_osrEntryCallee, this new OMG function linking does not happen for this OMGForOSREntry callee.
151         // We re-link this OMGForOSREntry callee again not to miss that chance.
152         for (auto& call : callee->wasmToWasmCallsites()) {
153             MacroAssemblerCodePtr<WasmEntryPtrTag> entrypoint;
154             if (call.functionIndexSpace < m_module->moduleInformation().importFunctionCount())
155                 entrypoint = m_codeBlock->m_wasmToWasmExitStubs[call.functionIndexSpace].code();
156             else
157                 entrypoint = m_codeBlock->wasmEntrypointCalleeFromFunctionIndexSpace(call.functionIndexSpace).entrypoint().retagged<WasmEntryPtrTag>();
158 
159             MacroAssembler::repatchNearCall(call.callLocation, CodeLocationLabel<WasmEntryPtrTag>(entrypoint));
160         }
161     }
162     dataLogLnIf(WasmOMGForOSREntryPlanInternal::verbose, "Finished OMGForOSREntry ", m_functionIndex);
163     complete(holdLock(m_lock));
164 }
