28 #if ENABLE(SERVICE_WORKER)

30 #include "SWServer.h"
31 #include "ServiceWorkerJobData.h"

33 #include <wtf/Deque.h>
34 
35 namespace WebCore {
36 
37 class SWServerWorker;
38 
39 class SWServerJobQueue {
40     WTF_MAKE_FAST_ALLOCATED;
41 public:
42     explicit SWServerJobQueue(SWServer&, const ServiceWorkerRegistrationKey&);
43     SWServerJobQueue(const SWServerRegistration&) = delete;
44     ~SWServerJobQueue();
45 
46     const ServiceWorkerJobData& firstJob() const { return m_jobQueue.first(); }
47     const ServiceWorkerJobData& lastJob() const { return m_jobQueue.last(); }
48     void enqueueJob(ServiceWorkerJobData&& jobData) { m_jobQueue.append(WTFMove(jobData)); }
49     size_t size() const { return m_jobQueue.size(); }
50 
51     void runNextJob();
52 
53     void scriptFetchFinished(const ServiceWorkerFetchResult&);
54     void scriptContextFailedToStart(const ServiceWorkerJobDataIdentifier&, ServiceWorkerIdentifier, const String& message);
55     void scriptContextStarted(const ServiceWorkerJobDataIdentifier&, ServiceWorkerIdentifier);
56     void didFinishInstall(const ServiceWorkerJobDataIdentifier&, SWServerWorker&, bool wasSuccessful);
57     void didResolveRegistrationPromise();
58     void cancelJobsFromConnection(SWServerConnectionIdentifier);
59     void cancelJobsFromServiceWorker(ServiceWorkerIdentifier);
60 
61     bool isCurrentlyProcessingJob(const ServiceWorkerJobDataIdentifier&) const;
62 
63 private:
64     void jobTimerFired();
65     void runNextJobSynchronously();
66     void rejectCurrentJob(const ExceptionData&);
67     void finishCurrentJob();
68 
69     void runRegisterJob(const ServiceWorkerJobData&);
70     void runUnregisterJob(const ServiceWorkerJobData&);
71     void runUpdateJob(const ServiceWorkerJobData&);
72 
73     void install(SWServerRegistration&, ServiceWorkerIdentifier);
74 
75     void removeAllJobsMatching(const WTF::Function<bool(ServiceWorkerJobData&)>&);
76 
77     Deque<ServiceWorkerJobData> m_jobQueue;
78 
79     Timer m_jobTimer;
80     SWServer& m_server;
81     ServiceWorkerRegistrationKey m_registrationKey;
82 };
83 
84 } // namespace WebCore
