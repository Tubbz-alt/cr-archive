 36 namespace WebCore {

 38 static SWServerToContextConnectionIdentifier generateServerToContextConnectionIdentifier()
 39 {
 40     return SWServerToContextConnectionIdentifier::generate();
 41 }
 42 
 43 SWServerToContextConnection::SWServerToContextConnection(RegistrableDomain&& registrableDomain)
 44     : m_identifier(generateServerToContextConnectionIdentifier())
 45     , m_registrableDomain(WTFMove(registrableDomain))
 46 {
 47 }
 48 
 49 SWServerToContextConnection::~SWServerToContextConnection()
 50 {
 51 }
 52 
 53 void SWServerToContextConnection::scriptContextFailedToStart(const Optional<ServiceWorkerJobDataIdentifier>& jobDataIdentifier, ServiceWorkerIdentifier serviceWorkerIdentifier, const String& message)
 54 {
 55     if (auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier))
 56         worker->scriptContextFailedToStart(jobDataIdentifier, message);
 57 }
 58 
 59 void SWServerToContextConnection::scriptContextStarted(const Optional<ServiceWorkerJobDataIdentifier>& jobDataIdentifier, ServiceWorkerIdentifier serviceWorkerIdentifier, bool doesHandleFetch)
 60 {
 61     if (auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier))
 62         worker->scriptContextStarted(jobDataIdentifier, doesHandleFetch);
 63 }
 64 
 65 void SWServerToContextConnection::didFinishInstall(const Optional<ServiceWorkerJobDataIdentifier>& jobDataIdentifier, ServiceWorkerIdentifier serviceWorkerIdentifier, bool wasSuccessful)
 66 {
 67     if (auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier))
 68         worker->didFinishInstall(jobDataIdentifier, wasSuccessful);
 69 }
 70 
 71 void SWServerToContextConnection::didFinishActivation(ServiceWorkerIdentifier serviceWorkerIdentifier)
 72 {
 73     if (auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier))
 74         worker->didFinishActivation();
 75 }
 76 
 77 void SWServerToContextConnection::setServiceWorkerHasPendingEvents(ServiceWorkerIdentifier serviceWorkerIdentifier, bool hasPendingEvents)
 78 {
 79     if (auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier))
 80         worker->setHasPendingEvents(hasPendingEvents);
 81 }
 82 
 83 void SWServerToContextConnection::workerTerminated(ServiceWorkerIdentifier serviceWorkerIdentifier)
 84 {
 85     if (auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier))
 86         worker->contextTerminated();
 87 }
 88 
 89 void SWServerToContextConnection::findClientByIdentifier(uint64_t requestIdentifier, ServiceWorkerIdentifier serviceWorkerIdentifier, ServiceWorkerClientIdentifier clientId)
 90 {
 91     if (auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier))
 92         worker->contextConnection()->findClientByIdentifierCompleted(requestIdentifier, worker->findClientByIdentifier(clientId), false);
 93 }
 94 
 95 void SWServerToContextConnection::matchAll(uint64_t requestIdentifier, ServiceWorkerIdentifier serviceWorkerIdentifier, const ServiceWorkerClientQueryOptions& options)
 96 {
 97     if (auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier)) {
 98         worker->matchAll(options, [&] (auto&& data) {
 99             worker->contextConnection()->matchAllCompleted(requestIdentifier, data);
100         });
101     }
102 }
103 
104 void SWServerToContextConnection::claim(uint64_t requestIdentifier, ServiceWorkerIdentifier serviceWorkerIdentifier)
105 {
106     if (auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier)) {
107         worker->claim();
108         worker->contextConnection()->claimCompleted(requestIdentifier);
109     }
110 }
111 
112 void SWServerToContextConnection::skipWaiting(ServiceWorkerIdentifier serviceWorkerIdentifier, CompletionHandler<void()>&& completionHandler)
113 {
114     if (auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier))
115         worker->skipWaiting();
116 
117     completionHandler();
118 }
119 
120 void SWServerToContextConnection::setScriptResource(ServiceWorkerIdentifier serviceWorkerIdentifier, URL&& scriptURL, String&& script, URL&& responseURL, String&& mimeType)
121 {
122     if (auto* worker = SWServerWorker::existingWorkerForIdentifier(serviceWorkerIdentifier))
123         worker->setScriptResource(WTFMove(scriptURL), ServiceWorkerContextData::ImportedScript { WTFMove(script), WTFMove(responseURL), WTFMove(mimeType) });
124 }
125 
126 void SWServerToContextConnection::didFailHeartBeatCheck(ServiceWorkerIdentifier identifier)
127 {
128     if (auto* worker = SWServerWorker::existingWorkerForIdentifier(identifier))
129         worker->didFailHeartBeatCheck();
130 }
