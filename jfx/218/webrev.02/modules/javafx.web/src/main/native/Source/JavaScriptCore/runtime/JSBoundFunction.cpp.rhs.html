 38 EncodedJSValue JSC_HOST_CALL boundThisNoArgsFunctionCall(JSGlobalObject* globalObject, CallFrame* callFrame)
 39 {
 40     JSBoundFunction* boundFunction = jsCast<JSBoundFunction*>(callFrame->jsCallee());
 41 
 42     JSImmutableButterfly* boundArgs = boundFunction->boundArgs();
 43 
 44     MarkedArgumentBuffer args;
 45     if (boundArgs) {
 46         // Starts with 1 since the first one is |this|.
 47         for (unsigned i = 1; i < boundArgs->length(); ++i)
 48             args.append(boundArgs->get(i));
 49     }
 50     for (unsigned i = 0; i < callFrame->argumentCount(); ++i)
 51         args.append(callFrame->uncheckedArgument(i));
 52     RELEASE_ASSERT(!args.hasOverflowed());
 53 
 54     JSFunction* targetFunction = jsCast<JSFunction*>(boundFunction->targetFunction());
 55     ExecutableBase* executable = targetFunction->executable();
 56     if (executable->hasJITCodeForCall()) {
 57         // Force the executable to cache its arity entrypoint.
 58         executable->entrypointFor(CodeForCall, MustCheckArity);
 59     }
 60     CallData callData;
 61     CallType callType = getCallData(globalObject->vm(), targetFunction, callData);
 62     ASSERT(callType != CallType::None);
 63     return JSValue::encode(call(globalObject, targetFunction, callType, callData, boundFunction->boundThis(), args));
 64 }

 66 EncodedJSValue JSC_HOST_CALL boundFunctionCall(JSGlobalObject* globalObject, CallFrame* callFrame)
 67 {
 68     VM& vm = globalObject->vm();
 69     auto scope = DECLARE_THROW_SCOPE(vm);
 70     JSBoundFunction* boundFunction = jsCast<JSBoundFunction*>(callFrame->jsCallee());
 71 
 72     JSImmutableButterfly* boundArgs = boundFunction->boundArgs();
 73 
 74     MarkedArgumentBuffer args;
 75     if (boundArgs) {
 76         // Starts with 1 since the first one is |this|.
 77         for (unsigned i = 1; i < boundArgs->length(); ++i)
 78             args.append(boundArgs->get(i));
 79     }
 80     for (unsigned i = 0; i < callFrame->argumentCount(); ++i)
 81         args.append(callFrame->uncheckedArgument(i));
 82     if (UNLIKELY(args.hasOverflowed())) {
 83         throwOutOfMemoryError(globalObject, scope);
 84         return encodedJSValue();
 85     }
 86 
 87     JSObject* targetFunction = boundFunction->targetFunction();
 88     CallData callData;
 89     CallType callType = getCallData(vm, targetFunction, callData);
 90     ASSERT(callType != CallType::None);
 91     RELEASE_AND_RETURN(scope, JSValue::encode(call(globalObject, targetFunction, callType, callData, boundFunction->boundThis(), args)));
 92 }

151 {
152     JSBoundFunction* boundObject = jsCast<JSBoundFunction*>(callFrame->uncheckedArgument(0));
153     JSValue value = callFrame->uncheckedArgument(1);

155     return JSValue::encode(jsBoolean(boundObject->targetFunction()->hasInstance(globalObject, value)));

158 inline Structure* getBoundFunctionStructure(VM& vm, JSGlobalObject* globalObject, JSObject* targetFunction)
159 {
160     auto scope = DECLARE_THROW_SCOPE(vm);
161     JSValue prototype = targetFunction->getPrototype(vm, globalObject);
162     RETURN_IF_EXCEPTION(scope, nullptr);
163     JSFunction* targetJSFunction = jsDynamicCast<JSFunction*>(vm, targetFunction);
164 
165     // We only cache the structure of the bound function if the bindee is a JSFunction since there
166     // isn't any good place to put the structure on Internal Functions.
167     if (targetJSFunction) {
168         Structure* structure = targetJSFunction->ensureRareData(vm)->getBoundFunctionStructure();
169         if (structure && structure->storedPrototype() == prototype && structure->globalObject() == globalObject)
170             return structure;
171     }
172 
173     Structure* result = globalObject->boundFunctionStructure();
174 
175     // It would be nice if the structure map was keyed global objects in addition to the other things. Unfortunately, it is not
176     // currently. Whoever works on caching structure changes for prototype transitions should consider this problem as well.
177     // See: https://bugs.webkit.org/show_bug.cgi?id=152738
178     if (prototype.isObject() && prototype.getObject()->globalObject(vm) == globalObject) {
179         result = vm.structureCache.emptyStructureForPrototypeFromBaseStructure(globalObject, prototype.getObject(), result);
180         ASSERT_WITH_SECURITY_IMPLICATION(result->globalObject() == globalObject);
181     } else
182         result = Structure::create(vm, globalObject, prototype, result->typeInfo(), result->classInfo());
183 
184     if (targetJSFunction)
185         targetJSFunction->ensureRareData(vm)->setBoundFunctionStructure(vm, result);
186 
187     return result;
188 }

190 JSBoundFunction* JSBoundFunction::create(VM& vm, JSGlobalObject* globalObject, JSObject* targetFunction, JSValue boundThis, JSImmutableButterfly* boundArgs, int length, JSString* nameMayBeNull)
191 {
192     auto scope = DECLARE_THROW_SCOPE(vm);
193 
194     if (nameMayBeNull) {
195         nameMayBeNull->value(globalObject); // Resolving rope.
196         RETURN_IF_EXCEPTION(scope, nullptr);
197     }
198 
199     bool isJSFunction = getJSFunction(targetFunction);
200     ConstructData constructData;
201     ConstructType constructType = JSC::getConstructData(vm, targetFunction, constructData);
202     bool canConstruct = constructType != ConstructType::None;
203 
204     NativeExecutable* executable = vm.getBoundFunction(isJSFunction, canConstruct);
205     Structure* structure = getBoundFunctionStructure(vm, globalObject, targetFunction);






206     RETURN_IF_EXCEPTION(scope, nullptr);
207     JSBoundFunction* function = new (NotNull, allocateCell<JSBoundFunction>(vm.heap)) JSBoundFunction(vm, executable, globalObject, structure, targetFunction, boundThis, boundArgs, nameMayBeNull, length);
208 
209     function->finishCreation(vm, executable, length);
210     return function;
211 }

218 JSBoundFunction::JSBoundFunction(VM& vm, NativeExecutable* executable, JSGlobalObject* globalObject, Structure* structure, JSObject* targetFunction, JSValue boundThis, JSImmutableButterfly* boundArgs, JSString* nameMayBeNull, int length)
219     : Base(vm, executable, globalObject, structure)
220     , m_targetFunction(vm, this, targetFunction)
221     , m_boundThis(vm, this, boundThis)
222     , m_boundArgs(vm, this, boundArgs, WriteBarrier<JSImmutableButterfly>::MayBeNull)
223     , m_nameMayBeNull(vm, this, nameMayBeNull, WriteBarrier<JSString>::MayBeNull)
224     , m_length(length)
225 {
226     ASSERT(!m_nameMayBeNull || !m_nameMayBeNull->isRope());
227 }

229 JSArray* JSBoundFunction::boundArgsCopy(JSGlobalObject* globalObject)
230 {
231     VM& vm = globalObject->vm();
232     auto scope = DECLARE_THROW_SCOPE(vm);
233     JSArray* result = constructEmptyArray(this->globalObject(), nullptr);
234     RETURN_IF_EXCEPTION(scope, nullptr);
235     if (m_boundArgs) {
236         // Starts with 1 since the first one is bound |this|.
237         for (unsigned i = 1; i < m_boundArgs->length(); ++i) {
238             result->push(globalObject, m_boundArgs->get(i));
239             RETURN_IF_EXCEPTION(scope, nullptr);
240         }
241     }
242     return result;
243 }

245 void JSBoundFunction::finishCreation(VM& vm, NativeExecutable* executable, int length)
246 {
247     String name; // We lazily create our 'name' string property.
248     Base::finishCreation(vm, executable, length, name);
249     ASSERT(inherits(vm, info()));
250 }

252 void JSBoundFunction::visitChildren(JSCell* cell, SlotVisitor& visitor)
253 {
254     JSBoundFunction* thisObject = jsCast<JSBoundFunction*>(cell);
255     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
256     Base::visitChildren(thisObject, visitor);
257 
258     visitor.append(thisObject->m_targetFunction);
259     visitor.append(thisObject->m_boundThis);
260     visitor.append(thisObject->m_boundArgs);
261     visitor.append(thisObject->m_nameMayBeNull);
262 }
