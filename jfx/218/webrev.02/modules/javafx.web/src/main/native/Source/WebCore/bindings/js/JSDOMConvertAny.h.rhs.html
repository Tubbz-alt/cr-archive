26 #pragma once
27 
28 #include "IDLTypes.h"
29 #include "JSDOMConvertBase.h"
30 
31 namespace WebCore {
32 
33 template<> struct Converter<IDLAny> : DefaultConverter<IDLAny> {
34     using ReturnType = JSC::JSValue;
35 
36     static constexpr bool conversionHasSideEffects = false;
37 
38     static JSC::JSValue convert(JSC::JSGlobalObject&, JSC::JSValue value)
39     {
40         return value;
41     }
42 
43     static JSC::JSValue convert(const JSC::Strong<JSC::Unknown>& value)
44     {
45         return value.get();
46     }
47 };
48 
49 template<> struct JSConverter<IDLAny> {
50     static constexpr bool needsState = false;
51     static constexpr bool needsGlobalObject = false;
52 
53     static JSC::JSValue convert(const JSC::JSValue& value)
54     {
55         return value;
56     }
57 
58     static JSC::JSValue convert(const JSC::Strong<JSC::Unknown>& value)
59     {
60         return value.get();
61     }
62 };
63 
64 template<> struct VariadicConverter<IDLAny> {
65     using Item = typename IDLAny::ImplementationType;
66 
67     static Optional<Item> convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
68     {
69         auto& vm = JSC::getVM(&lexicalGlobalObject);
70         auto scope = DECLARE_THROW_SCOPE(vm);
71 
72         auto result = Converter<IDLAny>::convert(lexicalGlobalObject, value);
73         RETURN_IF_EXCEPTION(scope, WTF::nullopt);
74 
75         return Item { vm, result };
76     }
77 };
