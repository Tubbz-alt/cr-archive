 39 static bool canDecodeMedia(const MediaDecodingConfiguration& configuration)
 40 {
 41     // The mock implementation supports only local file playback.
 42     if (configuration.type == MediaDecodingType::MediaSource)
 43         return false;
 44 
 45     // Maxing out video decoding support at 720P.
 46     auto videoConfig = configuration.video;
 47     if (videoConfig && videoConfig->width > 1280 && videoConfig->height > 720)
 48         return false;
 49 
 50     // Only the "mock-with-alpha" codec supports alphaChannel
 51     if (videoConfig && videoConfig->alphaChannel && videoConfig->alphaChannel.value()) {
 52         if (ContentType(videoConfig->contentType).parameter(ContentType::codecsParameter()) != "mock-with-alpha")
 53             return false;
 54     }
 55 
 56     // Only the "mock-with-hdr" codec supports HDR)
 57     if (videoConfig && (videoConfig->colorGamut || videoConfig->hdrMetadataType || videoConfig->transferFunction)) {
 58         if (ContentType(videoConfig->contentType).parameter(ContentType::codecsParameter()) != "mock-with-hdr")
 59             return false;
 60     }
 61 
 62     // Audio decoding support limited to audio/mp4.
 63     auto audioConfig = configuration.audio;
 64     if (audioConfig)
 65         return ContentType(audioConfig->contentType).containerType() == "audio/mp4";
 66 
 67     return true;
 68 }
 69 
 70 static bool canSmoothlyDecodeMedia(const MediaDecodingConfiguration& configuration)
 71 {
 72     auto videoConfig = configuration.video;
 73     if (videoConfig && videoConfig->framerate > 30)
 74         return false;
 75 
 76     auto audioConfig = configuration.audio;
 77     if (audioConfig)
 78         return audioConfig->channels == "2";
 79 
 80     return true;
 81 }
 82 
 83 static bool canPowerEfficientlyDecodeMedia(const MediaDecodingConfiguration& configuration)
 84 {
 85     auto videoConfig = configuration.video;
 86     if (videoConfig && ContentType(videoConfig->contentType).containerType() != "video/mp4")
 87         return false;
 88 
 89     auto audioConfig = configuration.audio;
 90     if (audioConfig)
 91         return audioConfig->bitrate <= 1000;
 92 
 93     return true;
 94 }

 96 static bool canEncodeMedia(const MediaEncodingConfiguration& configuration)
 97 {
 98     // The mock implementation supports only local file playback.
 99     if (configuration.type == MediaEncodingType::Record)
100         return false;
101 
102     // Maxing out video encoding support at 720P.
103     auto videoConfig = configuration.video;
104     if (videoConfig && videoConfig->width > 1280 && videoConfig->height > 720)
105         return false;
106 
107     // Only the "mock-with-alpha" codec supports alphaChannel
108     if (videoConfig && videoConfig->alphaChannel && videoConfig->alphaChannel.value()) {
109         if (ContentType(videoConfig->contentType).parameter(ContentType::codecsParameter()) != "mock-with-alpha")
110             return false;
111     }
112 
113     // Audio encoding support limited to audio/mp4.
114     auto audioConfig = configuration.audio;
115     if (audioConfig && ContentType(audioConfig->contentType).containerType() != "audio/mp4")
116         return false;
117 
118     return true;
119 }

121 static bool canSmoothlyEncodeMedia(const MediaEncodingConfiguration& configuration)
122 {
123     auto videoConfig = configuration.video;
124     if (videoConfig && videoConfig->framerate > 30)
125         return false;
126 
127     auto audioConfig = configuration.audio;
128     if (audioConfig && audioConfig->channels != "2")
129         return false;
130 
131     return true;
132 }
133 
134 static bool canPowerEfficientlyEncodeMedia(const MediaEncodingConfiguration& configuration)
135 {
136     auto videoConfig = configuration.video;
137     if (videoConfig && ContentType(videoConfig->contentType).containerType() != "video/mp4")
138         return false;
139 
140     auto audioConfig = configuration.audio;
141     if (audioConfig && audioConfig->bitrate > 1000)
142         return false;
143 
144     return true;
145 }

147 void MediaEngineConfigurationFactoryMock::createDecodingConfiguration(MediaDecodingConfiguration&& configuration, DecodingConfigurationCallback&& callback)
148 {
149     if (!canDecodeMedia(configuration)) {
150         MediaCapabilitiesDecodingInfo info { WTFMove(configuration) };
151         callback(WTFMove(info));
152         return;
153     }
154     callback({{ true, canSmoothlyDecodeMedia(configuration), canPowerEfficientlyDecodeMedia(configuration) }, WTFMove(configuration)});
155 }
156 
157 void MediaEngineConfigurationFactoryMock::createEncodingConfiguration(MediaEncodingConfiguration&& configuration, EncodingConfigurationCallback&& callback)
158 {
159     if (!canEncodeMedia(configuration)) {
160         callback({{ }, WTFMove(configuration) });
161         return;
162     }
163     callback({{ true, canSmoothlyEncodeMedia(configuration), canPowerEfficientlyEncodeMedia(configuration) }, WTFMove(configuration)});
164 }
