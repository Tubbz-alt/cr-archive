 97 JS_EXPORT_PRIVATE void* startOfFixedExecutableMemoryPoolImpl();
 98 JS_EXPORT_PRIVATE void* endOfFixedExecutableMemoryPoolImpl();

100 template<typename T = void*>
101 T startOfFixedExecutableMemoryPool()
102 {
103     return bitwise_cast<T>(startOfFixedExecutableMemoryPoolImpl());
104 }
105 
106 template<typename T = void*>
107 T endOfFixedExecutableMemoryPool()
108 {
109     return bitwise_cast<T>(endOfFixedExecutableMemoryPoolImpl());
110 }

116 static ALWAYS_INLINE void* performJITMemcpy(void *dst, const void *src, size_t n)








117 {
118 #if CPU(ARM64)
119     static constexpr size_t instructionSize = sizeof(unsigned);
120     RELEASE_ASSERT(roundUpToMultipleOf<instructionSize>(dst) == dst);
121     RELEASE_ASSERT(roundUpToMultipleOf<instructionSize>(src) == src);
122 #endif
123     if (isJITPC(dst)) {
124         RELEASE_ASSERT(reinterpret_cast<uint8_t*>(dst) + n <= endOfFixedExecutableMemoryPool());
125 
126         if (UNLIKELY(Options::dumpJITMemoryPath()))
127             dumpJITMemory(dst, src, n);
128 #if ENABLE(FAST_JIT_PERMISSIONS)
129 #if ENABLE(SEPARATED_WX_HEAP)
130         if (g_jscConfig.useFastPermisionsJITCopy)
131 #endif
132         {
133             os_thread_self_restrict_rwx_to_rw();
134             memcpy(dst, src, n);
135             RELEASE_ASSERT(!Gigacage::contains(src));
136             os_thread_self_restrict_rwx_to_rx();
137             return dst;
138         }
139 #endif // ENABLE(FAST_JIT_PERMISSIONS)
140 
141 #if ENABLE(SEPARATED_WX_HEAP)
142         if (g_jscConfig.jitWriteSeparateHeaps) {
143             // Use execute-only write thunk for writes inside the JIT region. This is a variant of
144             // memcpy that takes an offset into the JIT region as its destination (first) parameter.
145             off_t offset = (off_t)((uintptr_t)dst - startOfFixedExecutableMemoryPool<uintptr_t>());
146             retagCodePtr<JITThunkPtrTag, CFunctionPtrTag>(g_jscConfig.jitWriteSeparateHeaps)(offset, src, n);
147             RELEASE_ASSERT(!Gigacage::contains(src));
148             return dst;
149         }
150 #endif
151     }
152 
153     // Use regular memcpy for writes outside the JIT region.
154     return memcpy(dst, src, n);
155 }
