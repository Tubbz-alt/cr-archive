 27 #pragma once
 28 
 29 #include "AffineTransform.h"
 30 #include "FloatSize.h"
 31 #include <wtf/EnumTraits.h>
 32 
 33 namespace WebCore {
 34 
 35 struct ImageOrientation {
 36     enum Orientation {
 37         FromImage         = 0, // Orientation from the image should be respected.
 38 
 39         // This range intentionally matches the orientation values from the EXIF spec.
 40         // See JEITA CP-3451, page 18. http://www.exif.org/Exif2-2.PDF
 41         OriginTopLeft     = 1, // default
 42         OriginTopRight    = 2, // mirror along y-axis
 43         OriginBottomRight = 3, // 180 degree rotation
 44         OriginBottomLeft  = 4, // mirror along the x-axis
 45         OriginLeftTop     = 5, // mirror along x-axis + 270 degree CW rotation
 46         OriginRightTop    = 6, // 90 degree CW rotation
 47         OriginRightBottom = 7, // mirror along x-axis + 90 degree CW rotation
 48         OriginLeftBottom  = 8, // 270 degree CW rotation
 49 
 50         None              = OriginTopLeft
 51     };
 52 
 53     ImageOrientation() = default;
 54 
 55     ImageOrientation(Orientation orientation)
 56         : m_orientation(orientation)
 57     {
 58     }
 59 
 60     static Orientation fromEXIFValue(int exifValue)
 61     {
 62         return isValidEXIFOrientation(exifValue) ? static_cast<Orientation>(exifValue) : None;
 63     }
 64 
 65     operator Orientation() const { return m_orientation; }
 66 
 67     bool usesWidthAsHeight() const
 68     {
 69         ASSERT(isValidEXIFOrientation(m_orientation));
 70         // Values 5 through 8 all flip the width/height.
 71         return m_orientation >= OriginLeftTop && m_orientation <= OriginLeftBottom;
 72     }
 73 
 74     AffineTransform transformFromDefault(const FloatSize& drawnSize) const
 75     {
 76         float w = drawnSize.width();
 77         float h = drawnSize.height();
 78 
 79         switch (m_orientation) {
 80         case FromImage:
 81             ASSERT_NOT_REACHED();
 82             return AffineTransform();
 83         case OriginTopLeft:
 84             return AffineTransform();
 85         case OriginTopRight:
 86             return AffineTransform(-1,  0,  0,  1,  w, 0);
 87         case OriginBottomRight:
 88             return AffineTransform(-1,  0,  0, -1,  w, h);
 89         case OriginBottomLeft:
 90             return AffineTransform( 1,  0,  0, -1,  0, h);
 91         case OriginLeftTop:
 92             return AffineTransform( 0,  1,  1,  0,  0, 0);
 93         case OriginRightTop:
 94             return AffineTransform( 0,  1, -1,  0,  w, 0);
 95         case OriginRightBottom:
 96             return AffineTransform( 0, -1, -1,  0,  w, h);
 97         case OriginLeftBottom:
 98             return AffineTransform( 0, -1,  1,  0,  0, h);
 99         }
100 
101         ASSERT_NOT_REACHED();
102         return AffineTransform();
103     }
104 
105 private:
106     static const Orientation EXIFFirst = OriginTopLeft;
107     static const Orientation EXIFLast = OriginLeftBottom;
108     static const Orientation First = FromImage;
109     static const Orientation Last = EXIFLast;
110 
111     static bool isValidOrientation(int orientation)
112     {
113         return orientation >= static_cast<int>(First) && orientation <= static_cast<int>(Last);
114     }
115 
116     static bool isValidEXIFOrientation(int orientation)
117     {
118         return orientation >= static_cast<int>(EXIFFirst) && orientation <= static_cast<int>(EXIFLast);
119     }
120 
121     Orientation m_orientation { None };
122 };
123 
124 } // namespace WebCore
125

135     WebCore::ImageOrientation::Orientation::OriginBottomLeft,
136     WebCore::ImageOrientation::Orientation::OriginLeftTop,
137     WebCore::ImageOrientation::Orientation::OriginRightTop,
138     WebCore::ImageOrientation::Orientation::OriginRightBottom,
139     WebCore::ImageOrientation::Orientation::OriginLeftBottom
140     >;
141 };
