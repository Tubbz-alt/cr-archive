 46 namespace Inspector {
 47 
 48 class ConsoleFrontendDispatcher;
 49 class InjectedScriptManager;
 50 class ScriptArguments;
 51 class ScriptCallStack;
 52 
 53 class JS_EXPORT_PRIVATE ConsoleMessage {
 54     WTF_MAKE_NONCOPYABLE(ConsoleMessage);
 55     WTF_MAKE_FAST_ALLOCATED;
 56 public:
 57     ConsoleMessage(MessageSource, MessageType, MessageLevel, const String& message, unsigned long requestIdentifier = 0);
 58     ConsoleMessage(MessageSource, MessageType, MessageLevel, const String& message, const String& url, unsigned line, unsigned column, JSC::JSGlobalObject* = nullptr, unsigned long requestIdentifier = 0);
 59     ConsoleMessage(MessageSource, MessageType, MessageLevel, const String& message, Ref<ScriptCallStack>&&, unsigned long requestIdentifier = 0);
 60     ConsoleMessage(MessageSource, MessageType, MessageLevel, const String& message, Ref<ScriptArguments>&&, Ref<ScriptCallStack>&&, unsigned long requestIdentifier = 0);
 61     ConsoleMessage(MessageSource, MessageType, MessageLevel, const String& message, Ref<ScriptArguments>&&, JSC::JSGlobalObject* = nullptr, unsigned long requestIdentifier = 0);
 62     ConsoleMessage(MessageSource, MessageType, MessageLevel, Vector<JSONLogValue>&&, JSC::JSGlobalObject*, unsigned long requestIdentifier = 0);
 63     ~ConsoleMessage();
 64 
 65     void addToFrontend(ConsoleFrontendDispatcher&, InjectedScriptManager&, bool generatePreview);
 66     void updateRepeatCountInConsole(ConsoleFrontendDispatcher&);
 67 
 68     MessageSource source() const { return m_source; }
 69     MessageType type() const { return m_type; }
 70     MessageLevel level() const { return m_level; }
 71     const String& message() const { return m_message; }
 72     const String& url() const { return m_url; }
 73     unsigned line() const { return m_line; }
 74     unsigned column() const { return m_column; }
 75 
 76     JSC::JSGlobalObject* globalObject() const;
 77 
 78     void incrementCount() { ++m_repeatCount; }
 79 
 80     const RefPtr<ScriptArguments>& arguments() const { return m_arguments; }
 81     unsigned argumentCount() const;
 82 
 83     bool isEqual(ConsoleMessage* msg) const;
 84 
 85     void clear();
 86 
 87 private:
 88     void autogenerateMetadata(JSC::JSGlobalObject* = nullptr);
 89 
 90     MessageSource m_source;
 91     MessageType m_type;
 92     MessageLevel m_level;
 93     String m_message;
 94     RefPtr<ScriptArguments> m_arguments;
 95     RefPtr<ScriptCallStack> m_callStack;
 96     Vector<JSONLogValue> m_jsonLogValues;
 97     String m_url;
 98     JSC::Strong<JSC::JSGlobalObject> m_globalObject;
 99     unsigned m_line { 0 };
100     unsigned m_column { 0 };
101     unsigned m_repeatCount { 1 };
102     String m_requestId;
103 };
