 276 JNIEXPORT jstring JNICALL Java_com_sun_webkit_dom_DocumentImpl_getReadyStateImpl(JNIEnv* env, jclass, jlong peer)
 277 {
 278     WebCore::JSMainThreadNullState state;
 279     auto readyState = IMPL->readyState();
 280     const char* readyStateStr { };
 281     switch (readyState) {
 282     case WebCore::Document::Loading:
 283         readyStateStr = "loading";
 284         break;
 285     case WebCore::Document::Interactive:
 286         readyStateStr = "interactive";
 287         break;
 288     case WebCore::Document::Complete:
 289         readyStateStr = "complete";
 290         break;
 291     default:
 292         ASSERT_NOT_REACHED();
 293     }
 294     return JavaReturn<String>(env, String(readyStateStr));
 295 }
