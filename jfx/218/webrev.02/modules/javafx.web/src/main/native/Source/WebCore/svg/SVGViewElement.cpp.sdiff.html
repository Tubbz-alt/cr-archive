51 void SVGViewElement::parseAttribute(const QualifiedName& name, const AtomString& value)
52 {
53     if (name == SVGNames::viewTargetAttr) {
54         m_viewTarget->reset(value);
55         return;
56     }
57 
58     SVGElement::parseAttribute(name, value);
59     SVGExternalResourcesRequired::parseAttribute(name, value);
60     SVGFitToViewBox::parseAttribute(name, value);
61     SVGZoomAndPan::parseAttribute(name, value);
62 }

54 void SVGViewElement::svgAttributeChanged(const QualifiedName& attrName)
55 {
56     // We ignore changes to SVGNames::viewTargetAttr, which is deprecated and unused in WebCore.
57     if (PropertyRegistry::isKnownAttribute(attrName))
58         return;
59 
60     if (SVGFitToViewBox::isKnownAttribute(attrName)) {
61         if (m_targetElement) {
62             m_targetElement->inheritViewAttributes(*this);
63             if (auto* renderer = m_targetElement->renderer())
64                 RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
65         }
66 
67         return;
68     }
69 
70     SVGElement::svgAttributeChanged(attrName);
71 }
