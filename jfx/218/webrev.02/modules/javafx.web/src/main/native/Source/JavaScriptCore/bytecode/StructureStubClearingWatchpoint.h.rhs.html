 30 #include "AdaptiveInferredPropertyValueWatchpointBase.h"
 31 #include "ObjectPropertyCondition.h"
 32 #include "Watchpoint.h"
 33 #include <wtf/Bag.h>
 34 #include <wtf/FastMalloc.h>
 35 #include <wtf/Noncopyable.h>
 36 
 37 namespace JSC {
 38 
 39 class CodeBlock;
 40 class StructureStubInfo;
 41 class WatchpointsOnStructureStubInfo;
 42 
 43 class StructureTransitionStructureStubClearingWatchpoint final : public Watchpoint {
 44     WTF_MAKE_NONCOPYABLE(StructureTransitionStructureStubClearingWatchpoint);
 45     WTF_MAKE_FAST_ALLOCATED;
 46 public:
 47     StructureTransitionStructureStubClearingWatchpoint(const ObjectPropertyCondition& key, WatchpointsOnStructureStubInfo& holder)
 48         : Watchpoint(Watchpoint::Type::StructureTransitionStructureStubClearing)
 49         , m_holder(&holder)
 50         , m_key(key)
 51     {
 52     }
 53 
 54     void fireInternal(VM&, const FireDetail&);
 55 
 56 private:
 57     // Own destructor may not be called. Keep members trivially destructible.
 58     JSC_WATCHPOINT_FIELD(PackedPtr<WatchpointsOnStructureStubInfo>, m_holder);
 59     JSC_WATCHPOINT_FIELD(ObjectPropertyCondition, m_key);
 60 };
 61 
 62 class AdaptiveValueStructureStubClearingWatchpoint final : public AdaptiveInferredPropertyValueWatchpointBase {
 63     using Base = AdaptiveInferredPropertyValueWatchpointBase;
 64     WTF_MAKE_NONCOPYABLE(AdaptiveValueStructureStubClearingWatchpoint);
 65     WTF_MAKE_FAST_ALLOCATED;
 66 
 67     void handleFire(VM&, const FireDetail&) override;
 68 
 69 public:
 70     AdaptiveValueStructureStubClearingWatchpoint(const ObjectPropertyCondition& key, WatchpointsOnStructureStubInfo& holder)
 71         : Base(key)
 72         , m_holder(&holder)
 73     {
 74         RELEASE_ASSERT(key.condition().kind() == PropertyCondition::Equivalence);
 75     }
 76 
 77 
 78 private:
 79     PackedPtr<WatchpointsOnStructureStubInfo> m_holder;
 80 };
 81 
 82 class WatchpointsOnStructureStubInfo {
 83     WTF_MAKE_NONCOPYABLE(WatchpointsOnStructureStubInfo);
 84     WTF_MAKE_FAST_ALLOCATED;
 85 public:
 86     WatchpointsOnStructureStubInfo(CodeBlock* codeBlock, StructureStubInfo* stubInfo)
 87         : m_codeBlock(codeBlock)
 88         , m_stubInfo(stubInfo)
 89     {
 90     }
 91 
 92     using Node = Variant<StructureTransitionStructureStubClearingWatchpoint, AdaptiveValueStructureStubClearingWatchpoint>;
 93 
 94     Node& addWatchpoint(const ObjectPropertyCondition& key);
 95 
 96     static void ensureReferenceAndInstallWatchpoint(
 97         std::unique_ptr<WatchpointsOnStructureStubInfo>& holderRef,
 98         CodeBlock*, StructureStubInfo*, const ObjectPropertyCondition& key);
 99     static Watchpoint* ensureReferenceAndAddWatchpoint(
100         std::unique_ptr<WatchpointsOnStructureStubInfo>& holderRef,
101         CodeBlock*, StructureStubInfo*);
102 
103     CodeBlock* codeBlock() const { return m_codeBlock; }
104     StructureStubInfo* stubInfo() const { return m_stubInfo; }
105 
106     bool isValid() const;
107 
108 private:
109     CodeBlock* m_codeBlock;
110     StructureStubInfo* m_stubInfo;
111     // FIXME: use less memory for the entries in this Bag:
112     // https://bugs.webkit.org/show_bug.cgi?id=202380
113     Bag<WTF::Variant<StructureTransitionStructureStubClearingWatchpoint, AdaptiveValueStructureStubClearingWatchpoint>> m_watchpoints;
114 };

116 } // namespace JSC
