 40 HTTPHeaderMap::HTTPHeaderMap()
 41 {
 42 }

 44 HTTPHeaderMap HTTPHeaderMap::isolatedCopy() const
 45 {
 46     HTTPHeaderMap map;
 47     map.m_commonHeaders = crossThreadCopy(m_commonHeaders);
 48     map.m_uncommonHeaders = crossThreadCopy(m_uncommonHeaders);
 49     return map;
 50 }
 51 
 52 String HTTPHeaderMap::get(const String& name) const
 53 {
 54     HTTPHeaderName headerName;
 55     if (findHTTPHeaderName(name, headerName))
 56         return get(headerName);
 57 
 58     return getUncommonHeader(name);
 59 }
 60 
 61 String HTTPHeaderMap::getUncommonHeader(const String& name) const
 62 {
 63     auto index = m_uncommonHeaders.findMatching([&](auto& header) {
 64         return equalIgnoringASCIICase(header.key, name);
 65     });
 66     return index != notFound ? m_uncommonHeaders[index].value : String();
 67 }
 68 
 69 #if USE(CF)
 70 
 71 void HTTPHeaderMap::set(CFStringRef name, const String& value)
 72 {
 73     // Fast path: avoid constructing a temporary String in the common header case.
 74     if (auto* nameCharacters = CFStringGetCStringPtr(name, kCFStringEncodingASCII)) {
 75         unsigned length = CFStringGetLength(name);
 76         HTTPHeaderName headerName;
 77         if (findHTTPHeaderName(StringView(nameCharacters, length), headerName))
 78             set(headerName, value);
 79         else
 80             setUncommonHeader(String(nameCharacters, length), value);
 81 
 82         return;
 83     }
 84 
 85     set(String(name), value);
 86 }
 87 
 88 #endif // USE(CF)
 89 
 90 void HTTPHeaderMap::set(const String& name, const String& value)
 91 {
 92     HTTPHeaderName headerName;
 93     if (findHTTPHeaderName(name, headerName)) {
 94         set(headerName, value);
 95         return;
 96     }
 97 
 98     setUncommonHeader(name, value);
 99 }
100 
101 void HTTPHeaderMap::setUncommonHeader(const String& name, const String& value)
102 {
103     auto index = m_uncommonHeaders.findMatching([&](auto& header) {
104         return equalIgnoringASCIICase(header.key, name);
105     });
106     if (index == notFound)
107         m_uncommonHeaders.append(UncommonHeader { name, value });
108     else
109         m_uncommonHeaders[index].value = value;
110 }
111 
112 void HTTPHeaderMap::add(const String& name, const String& value)
113 {
114     HTTPHeaderName headerName;
115     if (findHTTPHeaderName(name, headerName)) {
116         add(headerName, value);
117         return;
118     }
119     auto index = m_uncommonHeaders.findMatching([&](auto& header) {
120         return equalIgnoringASCIICase(header.key, name);
121     });
122     if (index == notFound)
123         m_uncommonHeaders.append(UncommonHeader { name, value });
124     else
125         m_uncommonHeaders[index].value = makeString(m_uncommonHeaders[index].value, ", ", value);
126 }
127 
128 void HTTPHeaderMap::append(const String& name, const String& value)
129 {
130     ASSERT(!contains(name));
131 
132     HTTPHeaderName headerName;
133     if (findHTTPHeaderName(name, headerName))
134         m_commonHeaders.append(CommonHeader { headerName, value });
135     else
136         m_uncommonHeaders.append(UncommonHeader { name, value });
137 }
138 
139 bool HTTPHeaderMap::addIfNotPresent(HTTPHeaderName headerName, const String& value)
140 {
141     if (contains(headerName))
142         return false;
143 
144     m_commonHeaders.append(CommonHeader { headerName, value });
145     return true;
146 }
147 
148 bool HTTPHeaderMap::contains(const String& name) const
149 {
150     HTTPHeaderName headerName;
151     if (findHTTPHeaderName(name, headerName))
152         return contains(headerName);
153 
154     return m_uncommonHeaders.findMatching([&](auto& header) {
155         return equalIgnoringASCIICase(header.key, name);
156     }) != notFound;
157 }
158 
159 bool HTTPHeaderMap::remove(const String& name)
160 {
161     HTTPHeaderName headerName;
162     if (findHTTPHeaderName(name, headerName))
163         return remove(headerName);
164 
165     return m_uncommonHeaders.removeFirstMatching([&](auto& header) {
166         return equalIgnoringASCIICase(header.key, name);
167     });
168 }
169 
170 String HTTPHeaderMap::get(HTTPHeaderName name) const
171 {
172     auto index = m_commonHeaders.findMatching([&](auto& header) {
173         return header.key == name;
174     });
175     return index != notFound ? m_commonHeaders[index].value : String();
176 }
177 
178 void HTTPHeaderMap::set(HTTPHeaderName name, const String& value)
179 {
180     auto index = m_commonHeaders.findMatching([&](auto& header) {
181         return header.key == name;
182     });
183     if (index == notFound)
184         m_commonHeaders.append(CommonHeader { name, value });
185     else
186         m_commonHeaders[index].value = value;
187 }
188 
189 bool HTTPHeaderMap::contains(HTTPHeaderName name) const
190 {
191     return m_commonHeaders.findMatching([&](auto& header) {
192         return header.key == name;
193     }) != notFound;
194 }
195 
196 bool HTTPHeaderMap::remove(HTTPHeaderName name)
197 {
198     return m_commonHeaders.removeFirstMatching([&](auto& header) {
199         return header.key == name;
200     });
201 }
202 
203 void HTTPHeaderMap::add(HTTPHeaderName name, const String& value)
204 {
205     auto index = m_commonHeaders.findMatching([&](auto& header) {
206         return header.key == name;
207     });
208     if (index != notFound)
209         m_commonHeaders[index].value = makeString(m_commonHeaders[index].value, ", ", value);
210     else
211         m_commonHeaders.append(CommonHeader { name, value });
212 }
