42 // https://html.spec.whatwg.org/multipage/webappapis.html#window-event-loop
43 class WindowEventLoop final : public EventLoop {
44 public:
45     static Ref<WindowEventLoop> eventLoopForSecurityOrigin(const SecurityOrigin&);
46 
47     virtual ~WindowEventLoop();
48 
49     void queueMutationObserverCompoundMicrotask();
50     Vector<GCReachableRef<HTMLSlotElement>>& signalSlotList() { return m_signalSlotList; }
51     HashSet<RefPtr<MutationObserver>>& activeMutationObservers() { return m_activeObservers; }
52     HashSet<RefPtr<MutationObserver>>& suspendedMutationObservers() { return m_suspendedObservers; }
53 
54     CustomElementQueue& backupElementQueue();
55 
56 private:
57     static Ref<WindowEventLoop> create(const String&);
58     WindowEventLoop(const String&);
59 
60     void scheduleToRun() final;
61     bool isContextThread() const final;
62     MicrotaskQueue& microtaskQueue() final;
63 
64     void didReachTimeToRun();
65 
66     String m_agentClusterKey;
67     Timer m_timer;
68     std::unique_ptr<MicrotaskQueue> m_microtaskQueue;
69 
70     // Each task scheduled in event loop is associated with a document so that it can be suspened or stopped
71     // when the associated document is suspened or stopped. This task group is used to schedule a task
72     // which is not scheduled to a specific document, and should only be used when it's absolutely required.
73     EventLoopTaskGroup m_perpetualTaskGroupForSimilarOriginWindowAgents;
74 
75     bool m_mutationObserverCompoundMicrotaskQueuedFlag { false };
76     bool m_deliveringMutationRecords { false }; // FIXME: This flag doesn't exist in the spec.
77     Vector<GCReachableRef<HTMLSlotElement>> m_signalSlotList; // https://dom.spec.whatwg.org/#signal-slot-list
78     HashSet<RefPtr<MutationObserver>> m_activeObservers;
79     HashSet<RefPtr<MutationObserver>> m_suspendedObservers;
80 
81     std::unique_ptr<CustomElementQueue> m_customElementQueue;
82     bool m_processingBackupElementQueue { false };
83 };
84 
85 } // namespace WebCore
