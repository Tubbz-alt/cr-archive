31 namespace JSC {

33 inline void Operand::dump(PrintStream& out) const
34 {
35     if (isTmp())
36         out.print("tmp", value());
37     else
38         out.print(virtualRegister());
39 }
40 
41 template<typename T>
42 void Operands<T>::dumpInContext(PrintStream& out, DumpContext* context) const
43 {
44     CommaPrinter comma(" ");
45     for (size_t argumentIndex = numberOfArguments(); argumentIndex--;) {
46         if (!argument(argumentIndex))
47             continue;
48         out.print(comma, "arg", argumentIndex, ":", inContext(argument(argumentIndex), context));
49     }
50     for (size_t localIndex = 0; localIndex < numberOfLocals(); ++localIndex) {
51         if (!local(localIndex))
52             continue;
53         out.print(comma, "loc", localIndex, ":", inContext(local(localIndex), context));
54     }
55     for (size_t tmpIndex = 0; tmpIndex < numberOfTmps(); ++tmpIndex) {
56         if (!tmp(tmpIndex))
57             continue;
58         out.print(comma, "tmp", tmpIndex, ":", inContext(tmp(tmpIndex), context));
59     }
60 }
61
