26 #pragma once

28 #if ENABLE(INDEXED_DATABASE)
29 
30 #include "IDBKeyData.h"
31 
32 namespace WebCore {
33 
34 struct IDBIterateCursorData {
35     IDBKeyData keyData;
36     IDBKeyData primaryKeyData;
37     unsigned count;
38     IndexedDB::CursorIterateOption option { IndexedDB::CursorIterateOption::Reply };
39 
40     WEBCORE_EXPORT IDBIterateCursorData isolatedCopy() const;
41 
42 #if !LOG_DISABLED
43     String loggingString() const;
44 #endif
45 
46     template<class Encoder> void encode(Encoder&) const;
47     template<class Decoder> static bool decode(Decoder&, IDBIterateCursorData&);
48 };
49 
50 template<class Encoder>
51 void IDBIterateCursorData::encode(Encoder& encoder) const
52 {
53     encoder << keyData << primaryKeyData << static_cast<uint64_t>(count);
54     encoder.encodeEnum(option);
55 }
56 
57 template<class Decoder>
58 bool IDBIterateCursorData::decode(Decoder& decoder, IDBIterateCursorData& iteratorCursorData)
59 {
60     Optional<IDBKeyData> keyData;
61     decoder >> keyData;
62     if (!keyData)
63         return false;
64     iteratorCursorData.keyData = WTFMove(*keyData);
65 
66     Optional<IDBKeyData> primaryKeyData;
67     decoder >> primaryKeyData;
68     if (!primaryKeyData)
69         return false;
70     iteratorCursorData.primaryKeyData = WTFMove(*primaryKeyData);
71 
72     uint64_t count;
73     if (!decoder.decode(count))
74         return false;
75 
76     if (count > std::numeric_limits<unsigned>::max())
77         return false;

78     iteratorCursorData.count = static_cast<unsigned>(count);
79 
80     if (!decoder.decodeEnum(iteratorCursorData.option))
81         return false;
82 
83     return true;
84 }
85 
86 } // namespace WebCore
