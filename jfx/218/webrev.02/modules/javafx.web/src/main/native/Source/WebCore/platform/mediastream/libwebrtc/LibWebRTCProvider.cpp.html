 73 static inline rtc::SocketAddress prepareSocketAddress(const rtc::SocketAddress& address, bool disableNonLocalhostConnections)
 74 {
 75     auto result = address;
 76     if (disableNonLocalhostConnections)
 77         result.SetIP("127.0.0.1");
 78     return result;
 79 }

 81 class BasicPacketSocketFactory : public rtc::BasicPacketSocketFactory {
 82     WTF_MAKE_FAST_ALLOCATED;
 83 public:
 84     explicit BasicPacketSocketFactory(rtc::Thread& networkThread)
 85         : m_socketFactory(makeUniqueRefWithoutFastMallocCheck<rtc::BasicPacketSocketFactory>(&networkThread))
 86     {
 87     }
 88 
 89     void setDisableNonLocalhostConnections(bool disableNonLocalhostConnections) { m_disableNonLocalhostConnections = disableNonLocalhostConnections; }
 90 
 91     rtc::AsyncPacketSocket* CreateUdpSocket(const rtc::SocketAddress& address, uint16_t minPort, uint16_t maxPort) final
 92     {
 93         return m_socketFactory->CreateUdpSocket(prepareSocketAddress(address, m_disableNonLocalhostConnections), minPort, maxPort);
 94     }
 95 
 96     rtc::AsyncPacketSocket* CreateServerTcpSocket(const rtc::SocketAddress& address, uint16_t minPort, uint16_t maxPort, int options) final
 97     {
 98         return m_socketFactory->CreateServerTcpSocket(prepareSocketAddress(address, m_disableNonLocalhostConnections), minPort, maxPort, options);
 99     }
100 
101     rtc::AsyncPacketSocket* CreateClientTcpSocket(const rtc::SocketAddress& localAddress, const rtc::SocketAddress& remoteAddress, const rtc::ProxyInfo& info, const std::string& name, const rtc::PacketSocketTcpOptions& options)
102     {
103         return m_socketFactory->CreateClientTcpSocket(prepareSocketAddress(localAddress, m_disableNonLocalhostConnections), remoteAddress, info, name, options);
104     }
105 
106 private:
107     bool m_disableNonLocalhostConnections { false };
108     UniqueRef<rtc::BasicPacketSocketFactory> m_socketFactory;
109 };

124 static void doReleaseLogging(rtc::LoggingSeverity severity, const char* message)
125 {
126 #if RELEASE_LOG_DISABLED
127     UNUSED_PARAM(severity);
128     UNUSED_PARAM(message);
129 #else
130     if (severity == rtc::LS_ERROR)
131         RELEASE_LOG_ERROR(WebRTC, "LibWebRTC error: %{public}s", message);
132     else
133         RELEASE_LOG(WebRTC, "LibWebRTC message: %{public}s", message);
134 #endif
135 }
136 
137 static void setLogging(rtc::LoggingSeverity level)
138 {
139     rtc::LogMessage::SetLogOutput(level, (level == rtc::LS_NONE) ? nullptr : doReleaseLogging);
140 }

142 static rtc::LoggingSeverity computeLogLevel()
143 {
144 #if defined(NDEBUG)
145 #if !LOG_DISABLED || !RELEASE_LOG_DISABLED
146     if (LogWebRTC.state != WTFLogChannelState::On)
147         return rtc::LS_ERROR;
148 
149     switch (LogWebRTC.level) {
150     case WTFLogLevel::Always:
151     case WTFLogLevel::Error:
152         return rtc::LS_ERROR;
153     case WTFLogLevel::Warning:
154         return rtc::LS_WARNING;
155     case WTFLogLevel::Info:
156         return rtc::LS_INFO;
157     case WTFLogLevel::Debug:
158         return rtc::LS_VERBOSE;
159     }
160 #else
161     return rtc::LS_NONE;
162 #endif
163 #else
164     return (LogWebRTC.state != WTFLogChannelState::On) ? rtc::LS_WARNING : rtc::LS_INFO;
165 #endif
166 }

168 static void initializePeerConnectionFactoryAndThreads(PeerConnectionFactoryAndThreads& factoryAndThreads)
169 {
170     ASSERT(!factoryAndThreads.networkThread);
171 
172     factoryAndThreads.networkThread = factoryAndThreads.networkThreadWithSocketServer ? rtc::Thread::CreateWithSocketServer() : rtc::Thread::Create();
173     factoryAndThreads.networkThread->SetName("WebKitWebRTCNetwork", nullptr);
174     bool result = factoryAndThreads.networkThread->Start();
175     ASSERT_UNUSED(result, result);
176 
177     factoryAndThreads.signalingThread = rtc::Thread::Create();
178     factoryAndThreads.signalingThread->SetName("WebKitWebRTCSignaling", nullptr);
179 
180     result = factoryAndThreads.signalingThread->Start();
181     ASSERT(result);
182 
183     factoryAndThreads.audioDeviceModule = makeUnique<LibWebRTCAudioModule>();
184 }

186 static inline PeerConnectionFactoryAndThreads& staticFactoryAndThreads()
187 {
188     static NeverDestroyed<PeerConnectionFactoryAndThreads> factoryAndThreads;
189     return factoryAndThreads.get();
190 }
191 
192 static inline PeerConnectionFactoryAndThreads& getStaticFactoryAndThreads(bool useNetworkThreadWithSocketServer)
193 {
194     auto& factoryAndThreads = staticFactoryAndThreads();
195 
196     ASSERT(!factoryAndThreads.networkThread || factoryAndThreads.networkThreadWithSocketServer == useNetworkThreadWithSocketServer);
197 
198     if (!factoryAndThreads.networkThread) {
199         factoryAndThreads.networkThreadWithSocketServer = useNetworkThreadWithSocketServer;
200         initializePeerConnectionFactoryAndThreads(factoryAndThreads);
201     }
202     return factoryAndThreads;
203 }

205 struct ThreadMessageData : public rtc::MessageData {
206     ThreadMessageData(Function<void()>&& callback)
207         : callback(WTFMove(callback))
208     { }
209     Function<void()> callback;
210 };

212 void PeerConnectionFactoryAndThreads::OnMessage(rtc::Message* message)
213 {
214     ASSERT(message->message_id == 1);
215     auto* data = static_cast<ThreadMessageData*>(message->pdata);
216     data->callback();
217     delete data;
218 }

245 webrtc::PeerConnectionFactoryInterface* LibWebRTCProvider::factory()
246 {
247     if (m_factory)
248         return m_factory.get();
249 
250     if (!webRTCAvailable())
251         return nullptr;
252 
253     auto& factoryAndThreads = getStaticFactoryAndThreads(m_useNetworkThreadWithSocketServer);
254 
255     m_factory = createPeerConnectionFactory(factoryAndThreads.networkThread.get(), factoryAndThreads.networkThread.get(), factoryAndThreads.audioDeviceModule.get());
256 
257     return m_factory;
258 }

290 rtc::scoped_refptr<webrtc::PeerConnectionInterface> LibWebRTCProvider::createPeerConnection(webrtc::PeerConnectionObserver& observer, rtc::PacketSocketFactory*, webrtc::PeerConnectionInterface::RTCConfiguration&& configuration)
291 {
292     // Default WK1 implementation.
293     ASSERT(m_useNetworkThreadWithSocketServer);
294     auto& factoryAndThreads = getStaticFactoryAndThreads(m_useNetworkThreadWithSocketServer);
295 
296     if (!factoryAndThreads.networkManager)
297         factoryAndThreads.networkManager = makeUniqueWithoutFastMallocCheck<rtc::BasicNetworkManager>();
298 
299     if (!factoryAndThreads.packetSocketFactory)
300         factoryAndThreads.packetSocketFactory = makeUnique<BasicPacketSocketFactory>(*factoryAndThreads.networkThread);
301     factoryAndThreads.packetSocketFactory->setDisableNonLocalhostConnections(m_disableNonLocalhostConnections);
302 
303     return createPeerConnection(observer, *factoryAndThreads.networkManager, *factoryAndThreads.packetSocketFactory, WTFMove(configuration), nullptr);
304 }

340 rtc::scoped_refptr<webrtc::PeerConnectionInterface> LibWebRTCProvider::createPeerConnection(webrtc::PeerConnectionObserver& observer, rtc::NetworkManager& networkManager, rtc::PacketSocketFactory& packetSocketFactory, webrtc::PeerConnectionInterface::RTCConfiguration&& configuration, std::unique_ptr<webrtc::AsyncResolverFactory>&& asyncResolveFactory)
341 {
342     auto& factoryAndThreads = getStaticFactoryAndThreads(m_useNetworkThreadWithSocketServer);
343 
344     std::unique_ptr<cricket::BasicPortAllocator> portAllocator;
345     factoryAndThreads.signalingThread->Invoke<void>(RTC_FROM_HERE, [&]() {
346         auto basicPortAllocator = makeUniqueWithoutFastMallocCheck<cricket::BasicPortAllocator>(&networkManager, &packetSocketFactory);
347         if (!m_enableEnumeratingAllNetworkInterfaces)
348             basicPortAllocator->set_flags(basicPortAllocator->flags() | cricket::PORTALLOCATOR_DISABLE_ADAPTER_ENUMERATION);
349         portAllocator = WTFMove(basicPortAllocator);
350     });
351 
352     auto* factory = this->factory();
353     if (!factory)
354         return nullptr;
355 
356     webrtc::PeerConnectionDependencies dependencies { &observer };
357     dependencies.allocator = WTFMove(portAllocator);
358     dependencies.async_resolver_factory = WTFMove(asyncResolveFactory);
359 
360     return m_factory->CreatePeerConnection(configuration, WTFMove(dependencies));
361 }

372 static inline Optional<cricket::MediaType> typeFromKind(const String& kind)

393 static inline RTCRtpCapabilities toRTCRtpCapabilities(const webrtc::RtpCapabilities& rtpCapabilities)
394 {
395     RTCRtpCapabilities capabilities;
396 
397     capabilities.codecs.reserveInitialCapacity(rtpCapabilities.codecs.size());
398     for (auto& codec : rtpCapabilities.codecs)
399         capabilities.codecs.uncheckedAppend(RTCRtpCodecCapability { fromStdString(codec.mime_type()), static_cast<uint32_t>(codec.clock_rate ? *codec.clock_rate : 0), toChannels(codec.num_channels), { } });
400 
401     capabilities.headerExtensions.reserveInitialCapacity(rtpCapabilities.header_extensions.size());
402     for (auto& header : rtpCapabilities.header_extensions)
403         capabilities.headerExtensions.uncheckedAppend(RTCRtpCapabilities::HeaderExtensionCapability { fromStdString(header.uri) });
404 
405     return capabilities;
406 }

408 Optional<RTCRtpCapabilities> LibWebRTCProvider::receiverCapabilities(const String& kind)
409 {
410     auto mediaType = typeFromKind(kind);
411     if (!mediaType)
412         return { };
413 
414     auto* factory = this->factory();
415     if (!factory)
416         return { };
417 
418     return toRTCRtpCapabilities(factory->GetRtpReceiverCapabilities(*mediaType));
419 }

421 Optional<RTCRtpCapabilities> LibWebRTCProvider::senderCapabilities(const String& kind)
422 {
423     auto mediaType = typeFromKind(kind);
424     if (!mediaType)
425         return { };
426 
427     auto* factory = this->factory();
428     if (!factory)
429         return { };
430 
431     return toRTCRtpCapabilities(factory->GetRtpSenderCapabilities(*mediaType));
432 }
