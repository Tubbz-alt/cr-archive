 27 interface SVGPathElement : SVGGeometryElement {
 28     unsigned long getPathSegAtLength(optional unrestricted float distance = NaN);
 29 
 30     SVGPathSegClosePath createSVGPathSegClosePath();
 31 
 32     SVGPathSegMovetoAbs createSVGPathSegMovetoAbs(optional unrestricted float x = NaN,
 33                                                   optional unrestricted float y = NaN);
 34     SVGPathSegMovetoRel createSVGPathSegMovetoRel(optional unrestricted float x = NaN,
 35                                                   optional unrestricted float y = NaN);
 36 
 37     SVGPathSegLinetoAbs createSVGPathSegLinetoAbs(optional unrestricted float x = NaN,
 38                                                   optional unrestricted float y = NaN);
 39     SVGPathSegLinetoRel createSVGPathSegLinetoRel(optional unrestricted float x = NaN,
 40                                                   optional unrestricted float y = NaN);
 41 
 42     SVGPathSegCurvetoCubicAbs createSVGPathSegCurvetoCubicAbs(optional unrestricted float x = NaN,
 43                                                               optional unrestricted float y = NaN,
 44                                                               optional unrestricted float x1 = NaN,
 45                                                               optional unrestricted float y1 = NaN,
 46                                                               optional unrestricted float x2 = NaN,
 47                                                               optional unrestricted float y2 = NaN);
 48     SVGPathSegCurvetoCubicRel createSVGPathSegCurvetoCubicRel(optional unrestricted float x = NaN,
 49                                                               optional unrestricted float y = NaN,
 50                                                               optional unrestricted float x1 = NaN,
 51                                                               optional unrestricted float y1 = NaN,
 52                                                               optional unrestricted float x2 = NaN,
 53                                                               optional unrestricted float y2 = NaN);
 54 
 55     SVGPathSegCurvetoQuadraticAbs createSVGPathSegCurvetoQuadraticAbs(optional unrestricted float x = NaN,
 56                                                                       optional unrestricted float y = NaN,
 57                                                                       optional unrestricted float x1 = NaN,
 58                                                                       optional unrestricted float y1 = NaN);
 59     SVGPathSegCurvetoQuadraticRel createSVGPathSegCurvetoQuadraticRel(optional unrestricted float x = NaN,
 60                                                                       optional unrestricted float y = NaN,
 61                                                                       optional unrestricted float x1 = NaN,
 62                                                                       optional unrestricted float y1 = NaN);
 63 
 64     SVGPathSegArcAbs createSVGPathSegArcAbs(optional unrestricted float x = NaN,
 65                                             optional unrestricted float y = NaN,
 66                                             optional unrestricted float r1 = NaN,
 67                                             optional unrestricted float r2 = NaN,
 68                                             optional unrestricted float angle = NaN,
 69                                             optional boolean largeArcFlag = false,
 70                                             optional boolean sweepFlag = false);
 71     SVGPathSegArcRel createSVGPathSegArcRel(optional unrestricted float x = NaN,
 72                                             optional unrestricted float y = NaN,
 73                                             optional unrestricted float r1 = NaN,
 74                                             optional unrestricted float r2 = NaN,
 75                                             optional unrestricted float angle = NaN,
 76                                             optional boolean largeArcFlag = false,
 77                                             optional boolean sweepFlag = false);
 78 
 79     SVGPathSegLinetoHorizontalAbs createSVGPathSegLinetoHorizontalAbs(optional unrestricted float x = NaN);
 80     SVGPathSegLinetoHorizontalRel createSVGPathSegLinetoHorizontalRel(optional unrestricted float x = NaN);
 81 
 82     SVGPathSegLinetoVerticalAbs createSVGPathSegLinetoVerticalAbs(optional unrestricted float y = NaN);
 83     SVGPathSegLinetoVerticalRel createSVGPathSegLinetoVerticalRel(optional unrestricted float y = NaN);
 84 
 85     SVGPathSegCurvetoCubicSmoothAbs createSVGPathSegCurvetoCubicSmoothAbs(optional unrestricted float x = NaN,
 86                                                                           optional unrestricted float y = NaN,
 87                                                                           optional unrestricted float x2 = NaN,
 88                                                                           optional unrestricted float y2 = NaN);
 89     SVGPathSegCurvetoCubicSmoothRel createSVGPathSegCurvetoCubicSmoothRel(optional unrestricted float x = NaN,
 90                                                                           optional unrestricted float y = NaN,
 91                                                                           optional unrestricted float x2 = NaN,
 92                                                                           optional unrestricted float y2 = NaN);
 93 
 94     SVGPathSegCurvetoQuadraticSmoothAbs createSVGPathSegCurvetoQuadraticSmoothAbs(optional unrestricted float x = NaN,
 95                                                                                   optional unrestricted float y = NaN);
 96     SVGPathSegCurvetoQuadraticSmoothRel createSVGPathSegCurvetoQuadraticSmoothRel(optional unrestricted float x = NaN,
 97                                                                                   optional unrestricted float y = NaN);
 98 
 99     readonly attribute SVGPathSegList pathSegList;
100     readonly attribute SVGPathSegList normalizedPathSegList;
101     readonly attribute SVGPathSegList animatedPathSegList;
102     readonly attribute SVGPathSegList animatedNormalizedPathSegList;
103 };
