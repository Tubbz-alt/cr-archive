26 #pragma once
27 
28 #include "ArrayBuffer.h"
29 #include "JSObject.h"
30 
31 namespace JSC {
32 
33 class JSArrayBuffer final : public JSNonFinalObject {
34 public:
35     using Base = JSNonFinalObject;
36     static constexpr unsigned StructureFlags = Base::StructureFlags;
37 
38     template<typename CellType, SubspaceAccess mode>
39     static IsoSubspace* subspaceFor(VM& vm)
40     {
41         return vm.arrayBufferSpace<mode>();
42     }
43 
44 protected:
45     JSArrayBuffer(VM&, Structure*, RefPtr<ArrayBuffer>&&);
46     void finishCreation(VM&, JSGlobalObject*);
47 
48 public:
49     // This function will register the new wrapper with the vm's TypedArrayController.
50     JS_EXPORT_PRIVATE static JSArrayBuffer* create(VM&, Structure*, RefPtr<ArrayBuffer>&&);
51 
52     ArrayBuffer* impl() const { return m_impl; }
53 
54     static Structure* createStructure(VM&, JSGlobalObject*, JSValue prototype);
55 
56     JS_EXPORT_PRIVATE bool isShared() const;
57     ArrayBufferSharingMode sharingMode() const;
58 
59     DECLARE_EXPORT_INFO;
60 
61     // This is the default DOM unwrapping. It calls toUnsharedArrayBuffer().
62     static ArrayBuffer* toWrapped(VM&, JSValue);
63 
64 protected:
65     static size_t estimatedSize(JSCell*, VM&);
66 
67 private:
68     ArrayBuffer* m_impl;
69 };
70 
71 inline ArrayBuffer* toPossiblySharedArrayBuffer(VM& vm, JSValue value)
72 {
73     JSArrayBuffer* wrapper = jsDynamicCast<JSArrayBuffer*>(vm, value);
74     if (!wrapper)
75         return nullptr;
76     return wrapper->impl();
77 }
78 
79 inline ArrayBuffer* toUnsharedArrayBuffer(VM& vm, JSValue value)
80 {
81     ArrayBuffer* result = toPossiblySharedArrayBuffer(vm, value);
82     if (!result || result->isShared())
83         return nullptr;
84     return result;
85 }
86 
87 inline ArrayBuffer* JSArrayBuffer::toWrapped(VM& vm, JSValue value)
88 {
89     return toUnsharedArrayBuffer(vm, value);
90 }
91 
92 } // namespace JSC
