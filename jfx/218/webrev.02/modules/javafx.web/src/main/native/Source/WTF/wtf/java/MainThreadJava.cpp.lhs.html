33 void scheduleDispatchFunctionsOnMainThread()
34 {
35     AttachThreadAsNonDaemonToJavaEnv autoAttach;
36     JNIEnv* env = autoAttach.env();
37     static JGClass jMainThreadCls(env->FindClass("com/sun/webkit/MainThread"));
38 
39     static jmethodID mid = env->GetStaticMethodID(
40             jMainThreadCls,
41             "fwkScheduleDispatchFunctions",
42             "()V");
43 
44     ASSERT(mid);
45 
46     env->CallStaticVoidMethod(jMainThreadCls, mid);
47     WTF::CheckAndClearException(env);
48 }

59 bool isMainThread()
60 {
61     AttachThreadAsNonDaemonToJavaEnv autoAttach;
62     JNIEnv* env = autoAttach.env();
63     static JGClass jMainThreadCls(env->FindClass("com/sun/webkit/MainThread"));
64 
65     static jmethodID mid = env->GetStaticMethodID(
66             jMainThreadCls,
67             "fwkIsMainThread",
68             "()Z");
69 
70     ASSERT(mid);
71 
72     jboolean isMainThread = env->CallStaticBooleanMethod(jMainThreadCls, mid);
73     WTF::CheckAndClearException(env);
74     return isMainThread == JNI_TRUE;
75 }
