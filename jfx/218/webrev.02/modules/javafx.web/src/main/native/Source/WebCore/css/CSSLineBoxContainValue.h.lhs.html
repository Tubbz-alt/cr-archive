26 #pragma once
27 
28 #include "CSSValue.h"

29 #include <wtf/Ref.h>
30 
31 namespace WebCore {
32 
33 class CSSPrimitiveValue;
34 
35 enum LineBoxContainFlags {
36     LineBoxContainNone = 0x0,
37     LineBoxContainBlock = 0x1,
38     LineBoxContainInline = 0x2,
39     LineBoxContainFont = 0x4,
40     LineBoxContainGlyphs = 0x8,
41     LineBoxContainReplaced = 0x10,
42     LineBoxContainInlineBox = 0x20,
43     LineBoxContainInitialLetter = 0x40
44 };
45 typedef unsigned LineBoxContain;
46 
47 // Used for text-CSSLineBoxContain and box-CSSLineBoxContain
48 class CSSLineBoxContainValue final : public CSSValue {
49 public:
50     static Ref<CSSLineBoxContainValue> create(LineBoxContain value)
51     {
52         return adoptRef(*new CSSLineBoxContainValue(value));
53     }
54 
55     String customCSSText() const;
56     bool equals(const CSSLineBoxContainValue& other) const { return m_value == other.m_value; }
57     LineBoxContain value() const { return m_value; }
58 
59 private:
60     explicit CSSLineBoxContainValue(LineBoxContain);
61 
62     LineBoxContain m_value;
63 };
64 
65 } // namespace WebCore
66 
67 SPECIALIZE_TYPE_TRAITS_CSS_VALUE(CSSLineBoxContainValue, isLineBoxContainValue())
