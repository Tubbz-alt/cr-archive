 42 void BitVector::setSlow(const BitVector& other)
 43 {
 44     uintptr_t newBitsOrPointer;
 45     if (other.isInline() || other.isEmptyOrDeletedValue())
 46         newBitsOrPointer = other.m_bitsOrPointer;
 47     else {
 48         OutOfLineBits* newOutOfLineBits = OutOfLineBits::create(other.size());
 49         memcpy(newOutOfLineBits->bits(), other.bits(), byteCount(other.size()));
 50         newBitsOrPointer = bitwise_cast<uintptr_t>(newOutOfLineBits) >> 1;
 51     }
 52     if (!isInline() && !isEmptyOrDeletedValue())
 53         OutOfLineBits::destroy(outOfLineBits());
 54     m_bitsOrPointer = newBitsOrPointer;
 55 }
 56 
 57 void BitVector::resize(size_t numBits)
 58 {
 59     if (numBits <= maxInlineBits()) {
 60         if (isInline())
 61             return;
 62 
 63         OutOfLineBits* myOutOfLineBits = outOfLineBits();
 64         m_bitsOrPointer = makeInlineBits(*myOutOfLineBits->bits());
 65         OutOfLineBits::destroy(myOutOfLineBits);
 66         return;
 67     }
 68 
 69     resizeOutOfLine(numBits);
 70 }
 71 
 72 void BitVector::clearAll()
 73 {
 74     if (isInline())
 75         m_bitsOrPointer = makeInlineBits(0);
 76     else
 77         memset(outOfLineBits()->bits(), 0, byteCount(size()));
 78 }

 80 BitVector::OutOfLineBits* BitVector::OutOfLineBits::create(size_t numBits)
 81 {
 82     numBits = (numBits + bitsInPointer() - 1) & ~(static_cast<size_t>(bitsInPointer()) - 1);
 83     size_t size = sizeof(OutOfLineBits) + sizeof(uintptr_t) * (numBits / bitsInPointer());
 84     OutOfLineBits* result = new (NotNull, BitVectorMalloc::malloc(size)) OutOfLineBits(numBits);
 85     return result;
 86 }
 87 
 88 void BitVector::OutOfLineBits::destroy(OutOfLineBits* outOfLineBits)
 89 {
 90     BitVectorMalloc::free(outOfLineBits);
 91 }

 93 void BitVector::resizeOutOfLine(size_t numBits)
 94 {
 95     ASSERT(numBits > maxInlineBits());
 96     OutOfLineBits* newOutOfLineBits = OutOfLineBits::create(numBits);
 97     size_t newNumWords = newOutOfLineBits->numWords();
 98     if (isInline()) {
 99         // Make sure that all of the bits are zero in case we do a no-op resize.
100         *newOutOfLineBits->bits() = m_bitsOrPointer & ~(static_cast<uintptr_t>(1) << maxInlineBits());
101         memset(newOutOfLineBits->bits() + 1, 0, (newNumWords - 1) * sizeof(void*));
102     } else {
103         if (numBits > size()) {
104             size_t oldNumWords = outOfLineBits()->numWords();
105             memcpy(newOutOfLineBits->bits(), outOfLineBits()->bits(), oldNumWords * sizeof(void*));
106             memset(newOutOfLineBits->bits() + oldNumWords, 0, (newNumWords - oldNumWords) * sizeof(void*));
107         } else
108             memcpy(newOutOfLineBits->bits(), outOfLineBits()->bits(), newOutOfLineBits->numWords() * sizeof(void*));
109         OutOfLineBits::destroy(outOfLineBits());
110     }
111     m_bitsOrPointer = bitwise_cast<uintptr_t>(newOutOfLineBits) >> 1;
112 }

114 void BitVector::mergeSlow(const BitVector& other)
115 {
116     if (other.isInline()) {
117         ASSERT(!isInline());
118         *bits() |= cleanseInlineBits(other.m_bitsOrPointer);
119         return;
120     }
121 
122     ensureSize(other.size());
123     ASSERT(!isInline());
124     ASSERT(!other.isInline());
125 
126     OutOfLineBits* a = outOfLineBits();
127     const OutOfLineBits* b = other.outOfLineBits();
128     for (unsigned i = a->numWords(); i--;)
129         a->bits()[i] |= b->bits()[i];
130 }

132 void BitVector::filterSlow(const BitVector& other)
133 {
134     if (other.isInline()) {
135         ASSERT(!isInline());
136         *bits() &= cleanseInlineBits(other.m_bitsOrPointer);
137         return;
138     }
139 
140     if (isInline()) {
141         ASSERT(!other.isInline());
142         m_bitsOrPointer &= *other.outOfLineBits()->bits();
143         m_bitsOrPointer |= (static_cast<uintptr_t>(1) << maxInlineBits());
144         ASSERT(isInline());
145         return;
146     }
147 
148     OutOfLineBits* a = outOfLineBits();
149     const OutOfLineBits* b = other.outOfLineBits();
150     for (unsigned i = std::min(a->numWords(), b->numWords()); i--;)
151         a->bits()[i] &= b->bits()[i];
152 
153     for (unsigned i = b->numWords(); i < a->numWords(); ++i)
154         a->bits()[i] = 0;
155 }
156 
157 void BitVector::excludeSlow(const BitVector& other)
158 {
159     if (other.isInline()) {
160         ASSERT(!isInline());
161         *bits() &= ~cleanseInlineBits(other.m_bitsOrPointer);
162         return;
163     }
164 
165     if (isInline()) {
166         ASSERT(!other.isInline());
167         m_bitsOrPointer &= ~*other.outOfLineBits()->bits();
168         m_bitsOrPointer |= (static_cast<uintptr_t>(1) << maxInlineBits());
169         ASSERT(isInline());
170         return;
171     }
172 
173     OutOfLineBits* a = outOfLineBits();
174     const OutOfLineBits* b = other.outOfLineBits();
175     for (unsigned i = std::min(a->numWords(), b->numWords()); i--;)
176         a->bits()[i] &= ~b->bits()[i];
177 }

179 size_t BitVector::bitCountSlow() const
180 {
181     ASSERT(!isInline());
182     const OutOfLineBits* bits = outOfLineBits();
183     size_t result = 0;
184     for (unsigned i = bits->numWords(); i--;)
185         result += bitCount(bits->bits()[i]);
186     return result;
187 }
188 
189 bool BitVector::isEmptySlow() const
190 {
191     ASSERT(!isInline());
192     const OutOfLineBits* bits = outOfLineBits();
193     for (unsigned i = bits->numWords(); i--;) {
194         if (bits->bits()[i])
195             return false;
196     }
197     return true;
198 }

200 bool BitVector::equalsSlowCase(const BitVector& other) const
201 {
202     bool result = equalsSlowCaseFast(other);
203     ASSERT(result == equalsSlowCaseSimple(other));
204     return result;
205 }
206 
207 bool BitVector::equalsSlowCaseFast(const BitVector& other) const
208 {
209     if (isInline() != other.isInline())
210         return equalsSlowCaseSimple(other);
211 
212     const OutOfLineBits* myBits = outOfLineBits();
213     const OutOfLineBits* otherBits = other.outOfLineBits();
214 
215     size_t myNumWords = myBits->numWords();
216     size_t otherNumWords = otherBits->numWords();
217     size_t minNumWords;
218     size_t maxNumWords;
219 
220     const OutOfLineBits* longerBits;
221     if (myNumWords < otherNumWords) {
222         minNumWords = myNumWords;
223         maxNumWords = otherNumWords;
224         longerBits = otherBits;
225     } else {
226         minNumWords = otherNumWords;
227         maxNumWords = myNumWords;
228         longerBits = myBits;
229     }
230 
231     for (size_t i = minNumWords; i < maxNumWords; ++i) {
232         if (longerBits->bits()[i])
233             return false;
234     }
235 
236     for (size_t i = minNumWords; i--;) {
237         if (myBits->bits()[i] != otherBits->bits()[i])
238             return false;
239     }
240 
241     return true;
242 }

244 bool BitVector::equalsSlowCaseSimple(const BitVector& other) const
245 {
246     // This is really cheesy, but probably good enough for now.
247     for (unsigned i = std::max(size(), other.size()); i--;) {
248         if (get(i) != other.get(i))
249             return false;
250     }
251     return true;
252 }

254 uintptr_t BitVector::hashSlowCase() const
255 {
256     ASSERT(!isInline());
257     const OutOfLineBits* bits = outOfLineBits();
258     uintptr_t result = 0;
259     for (unsigned i = bits->numWords(); i--;)
260         result ^= bits->bits()[i];
261     return result;
262 }

264 void BitVector::dump(PrintStream& out) const
265 {
266     for (size_t i = 0; i < size(); ++i) {
267         if (get(i))
268             out.printf("1");
269         else
270             out.printf("-");
271     }
272 }
