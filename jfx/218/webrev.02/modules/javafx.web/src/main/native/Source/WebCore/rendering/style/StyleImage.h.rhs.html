24 #pragma once
25 
26 #include "CSSValue.h"
27 #include "FloatSize.h"
28 #include "Image.h"
29 #include <wtf/RefCounted.h>
30 #include <wtf/RefPtr.h>
31 #include <wtf/TypeCasts.h>
32 
33 namespace WebCore {
34 
35 class CachedImage;
36 class CachedResourceLoader;
37 class CSSValue;
38 class RenderElement;
39 class RenderObject;
40 struct ResourceLoaderOptions;
41 
42 typedef const void* WrappedImagePtr;
43 
44 class StyleImage : public RefCounted<StyleImage> {
45 public:
46     virtual ~StyleImage() = default;
47 
48     virtual bool operator==(const StyleImage& other) const = 0;
49 
50     virtual Ref<CSSValue> cssValue() const = 0;
51 
52     virtual bool canRender(const RenderElement*, float /*multiplier*/) const { return true; }
53     virtual bool isPending() const = 0;
54     virtual void load(CachedResourceLoader&, const ResourceLoaderOptions&) = 0;
55     virtual bool isLoaded() const { return true; }
56     virtual bool errorOccurred() const { return false; }
57     virtual FloatSize imageSize(const RenderElement*, float multiplier) const = 0;
58     virtual void computeIntrinsicDimensions(const RenderElement*, Length& intrinsicWidth, Length& intrinsicHeight, FloatSize& intrinsicRatio) = 0;
59     virtual bool imageHasRelativeWidth() const = 0;
60     virtual bool imageHasRelativeHeight() const = 0;
61     virtual bool usesImageContainerSize() const = 0;
62     virtual void setContainerContextForRenderer(const RenderElement&, const FloatSize&, float) = 0;
63     virtual void addClient(RenderElement*) = 0;
64     virtual void removeClient(RenderElement*) = 0;
65     virtual RefPtr<Image> image(RenderElement*, const FloatSize&) const = 0;
66     virtual WrappedImagePtr data() const = 0;
67     virtual float imageScaleFactor() const { return 1; }
68     virtual bool knownToBeOpaque(const RenderElement*) const = 0;
69     virtual CachedImage* cachedImage() const { return 0; }
70     virtual StyleImage* selectedImage() { return this; }
71     virtual const StyleImage* selectedImage() const { return this; }
72 
73     ALWAYS_INLINE bool isCachedImage() const { return m_isCachedImage; }
74     ALWAYS_INLINE bool isGeneratedImage() const { return m_isGeneratedImage; }
75     ALWAYS_INLINE bool isCursorImage() const { return m_isCursorImage; }
76     ALWAYS_INLINE bool isImageSet() const { return m_isImageSet; }
77 
78     bool hasCachedImage() const { return m_isCachedImage || selectedImage()->isCachedImage(); }
79 
80 protected:
81     StyleImage()
82         : m_isCachedImage(false)
83         , m_isGeneratedImage(false)
84         , m_isImageSet(false)
85         , m_isCursorImage(false)
86     {
87     }
88     bool m_isCachedImage : 1;
89     bool m_isGeneratedImage : 1;
90     bool m_isImageSet : 1;
91     bool m_isCursorImage : 1;
92 };
93 
94 } // namespace WebCore
95

98     static bool isType(const WebCore::StyleImage& image) { return image.predicate(); } \
99 SPECIALIZE_TYPE_TRAITS_END()
