41 WidthAndMargin InlineFormattingContext::Geometry::inlineBlockWidthAndMargin(LayoutState& layoutState, const Box& formattingContextRoot, UsedHorizontalValues usedValues)
42 {
43     ASSERT(formattingContextRoot.isInFlow());
44 
45     // 10.3.10 'Inline-block', replaced elements in normal flow
46 
47     // Exactly as inline replaced elements.
48     if (formattingContextRoot.replaced())
49         return inlineReplacedWidthAndMargin(layoutState, formattingContextRoot, usedValues);
50 
51     // 10.3.9 'Inline-block', non-replaced elements in normal flow
52 
53     // If 'width' is 'auto', the used value is the shrink-to-fit width as for floating elements.
54     // A computed value of 'auto' for 'margin-left' or 'margin-right' becomes a used value of '0'.
55     // #1
56     auto width = computedValueIfNotAuto(formattingContextRoot.style().logicalWidth(), usedValues.containingBlockWidth.valueOr(0));
57     if (!width)
58         width = shrinkToFitWidth(layoutState, formattingContextRoot, usedValues);
59 
60     // #2
61     auto computedHorizontalMargin = Geometry::computedHorizontalMargin(formattingContextRoot, usedValues);
62 
63     return WidthAndMargin { *width, { computedHorizontalMargin.start.valueOr(0_lu), computedHorizontalMargin.end.valueOr(0_lu) }, computedHorizontalMargin };
64 }

66 HeightAndMargin InlineFormattingContext::Geometry::inlineBlockHeightAndMargin(const LayoutState& layoutState, const Box& layoutBox)
67 {
68     ASSERT(layoutBox.isInFlow());
69 
70     // 10.6.2 Inline replaced elements, block-level replaced elements in normal flow, 'inline-block' replaced elements in normal flow and floating replaced elements
71     if (layoutBox.replaced())
72         return inlineReplacedHeightAndMargin(layoutState, layoutBox, { });
73 
74     // 10.6.6 Complicated cases
75     // - 'Inline-block', non-replaced elements.
76     auto usedHorizontalValues = UsedHorizontalValues { layoutState.displayBoxForLayoutBox(*layoutBox.containingBlock()).contentBoxWidth() };
77     return complicatedCases(layoutState, layoutBox, { }, usedHorizontalValues);

77 Optional<InlineLayoutUnit> InlineFormattingContext::Geometry::computedTextIndent(const Container& formattingContextRoot, const HorizontalConstraints& horizontalConstraints) const
78 {
79     auto textIndent = formattingContextRoot.style().textIndent();
80     if (textIndent == RenderStyle::initialTextIndent())
81         return { };
82     return InlineLayoutUnit { minimumValueForLength(textIndent, horizontalConstraints.logicalWidth) };
