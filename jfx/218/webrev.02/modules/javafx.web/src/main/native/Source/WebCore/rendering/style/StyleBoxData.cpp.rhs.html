 22 #include "config.h"
 23 #include "StyleBoxData.h"
 24 
 25 #include "RenderStyle.h"
 26 #include "RenderStyleConstants.h"
 27 
 28 namespace WebCore {
 29 
 30 struct SameSizeAsStyleBoxData : public RefCounted<SameSizeAsStyleBoxData> {
 31     Length length[7];
 32     int m_zIndex[2];
 33     uint32_t bitfields;
 34 };
 35 
 36 COMPILE_ASSERT(sizeof(StyleBoxData) == sizeof(SameSizeAsStyleBoxData), StyleBoxData_should_not_grow);
 37 
 38 DEFINE_ALLOCATOR_WITH_HEAP_IDENTIFIER(StyleBoxData);
 39 
 40 StyleBoxData::StyleBoxData()
 41     : m_minWidth(RenderStyle::initialMinSize())
 42     , m_maxWidth(RenderStyle::initialMaxSize())
 43     , m_minHeight(RenderStyle::initialMinSize())
 44     , m_maxHeight(RenderStyle::initialMaxSize())
 45     , m_specifiedZIndex(0)
 46     , m_usedZIndex(0)
 47     , m_hasAutoSpecifiedZIndex(true)
 48     , m_hasAutoUsedZIndex(true)
 49     , m_boxSizing(static_cast<unsigned>(BoxSizing::ContentBox))
 50 #if ENABLE(CSS_BOX_DECORATION_BREAK)
 51     , m_boxDecorationBreak(static_cast<unsigned>(BoxDecorationBreak::Slice))
 52 #endif
 53 {
 54 }
 55 
 56 inline StyleBoxData::StyleBoxData(const StyleBoxData& o)
 57     : RefCounted<StyleBoxData>()
 58     , m_width(o.m_width)
 59     , m_height(o.m_height)
 60     , m_minWidth(o.m_minWidth)
 61     , m_maxWidth(o.m_maxWidth)
 62     , m_minHeight(o.m_minHeight)
 63     , m_maxHeight(o.m_maxHeight)
 64     , m_verticalAlign(o.m_verticalAlign)
 65     , m_specifiedZIndex(o.m_specifiedZIndex)
 66     , m_usedZIndex(o.m_usedZIndex)
 67     , m_hasAutoSpecifiedZIndex(o.m_hasAutoSpecifiedZIndex)
 68     , m_hasAutoUsedZIndex(o.m_hasAutoUsedZIndex)
 69     , m_boxSizing(o.m_boxSizing)
 70 #if ENABLE(CSS_BOX_DECORATION_BREAK)
 71     , m_boxDecorationBreak(o.m_boxDecorationBreak)
 72 #endif
 73 {
 74 }
 75 
 76 Ref<StyleBoxData> StyleBoxData::copy() const
 77 {
 78     return adoptRef(*new StyleBoxData(*this));
 79 }
 80 
 81 bool StyleBoxData::operator==(const StyleBoxData& o) const
 82 {
 83     return m_width == o.m_width
 84         && m_height == o.m_height
 85         && m_minWidth == o.m_minWidth
 86         && m_maxWidth == o.m_maxWidth
 87         && m_minHeight == o.m_minHeight
 88         && m_maxHeight == o.m_maxHeight
 89         && m_verticalAlign == o.m_verticalAlign
 90         && m_specifiedZIndex == o.m_specifiedZIndex
 91         && m_hasAutoSpecifiedZIndex == o.m_hasAutoSpecifiedZIndex
 92         && m_usedZIndex == o.m_usedZIndex
 93         && m_hasAutoUsedZIndex == o.m_hasAutoUsedZIndex
 94         && m_boxSizing == o.m_boxSizing
 95 #if ENABLE(CSS_BOX_DECORATION_BREAK)
 96         && m_boxDecorationBreak == o.m_boxDecorationBreak
 97 #endif
 98         ;
 99 }
