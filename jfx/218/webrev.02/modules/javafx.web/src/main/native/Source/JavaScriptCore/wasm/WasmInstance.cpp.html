 41 size_t globalMemoryByteSize(Module& module)
 42 {
 43     return (Checked<size_t>(module.moduleInformation().globals.size()) * sizeof(Register)).unsafeGet();
 44 }
 45 }

 47 Instance::Instance(Context* context, Ref<Module>&& module, EntryFrame** pointerToTopEntryFrame, void** pointerToActualStackLimit, StoreTopCallFrameCallback&& storeTopCallFrame)
 48     : m_context(context)
 49     , m_module(WTFMove(module))
 50     , m_globals(MallocPtr<Global::Value, VMMalloc>::malloc(globalMemoryByteSize(m_module.get())))
 51     , m_globalsToMark(m_module.get().moduleInformation().globals.size())
 52     , m_globalsToBinding(m_module.get().moduleInformation().globals.size())
 53     , m_pointerToTopEntryFrame(pointerToTopEntryFrame)
 54     , m_pointerToActualStackLimit(pointerToActualStackLimit)
 55     , m_storeTopCallFrame(WTFMove(storeTopCallFrame))
 56     , m_numImportFunctions(m_module->moduleInformation().importFunctionCount())
 57 {
 58     for (unsigned i = 0; i < m_numImportFunctions; ++i)
 59         new (importFunctionInfo(i)) ImportFunctionInfo();
 60     memset(static_cast<void*>(m_globals.get()), 0, globalMemoryByteSize(m_module.get()));
 61     for (unsigned i = 0; i < m_module->moduleInformation().globals.size(); ++i) {
 62         const Wasm::GlobalInformation& global = m_module.get().moduleInformation().globals[i];
 63         if (global.bindingMode == Wasm::GlobalInformation::BindingMode::Portable) {
 64             // This is kept alive by JSWebAssemblyInstance -> JSWebAssemblyGlobal -> binding.
 65             m_globalsToBinding.set(i);
 66         } else if (isSubtype(global.type, Anyref)) {
 67             // This is kept alive by JSWebAssemblyInstance -> binding.
 68             m_globalsToMark.set(i);
 69         }
 70     }
 71     memset(bitwise_cast<char*>(this) + offsetOfTablePtr(m_numImportFunctions, 0), 0, m_module->moduleInformation().tableCount() * sizeof(Table*));
 72 }

 74 Ref<Instance> Instance::create(Context* context, Ref<Module>&& module, EntryFrame** pointerToTopEntryFrame, void** pointerToActualStackLimit, StoreTopCallFrameCallback&& storeTopCallFrame)
 75 {
 76     return adoptRef(*new (NotNull, fastMalloc(allocationSize(module->moduleInformation().importFunctionCount(), module->moduleInformation().tableCount()))) Instance(context, WTFMove(module), pointerToTopEntryFrame, pointerToActualStackLimit, WTFMove(storeTopCallFrame)));
 77 }

 81 size_t Instance::extraMemoryAllocated() const
 82 {
 83     return globalMemoryByteSize(m_module.get()) + allocationSize(m_numImportFunctions, m_module->moduleInformation().tableCount());
 84 }

 86 void Instance::setGlobal(unsigned i, JSValue value)
 87 {
 88     Global::Value* slot = m_globals.get() + i;
 89     if (m_globalsToBinding.get(i)) {
 90         Wasm::Global* global = getGlobalBinding(i);
 91         if (!global)
 92             return;
 93         global->valuePointer()->m_anyref.set(owner<JSWebAssemblyInstance>()->vm(), global->owner<JSWebAssemblyGlobal>(), value);
 94         return;
 95     }
 96     ASSERT(m_owner);
 97     slot->m_anyref.set(owner<JSWebAssemblyInstance>()->vm(), owner<JSWebAssemblyInstance>(), value);
 98 }

100 JSValue Instance::getFunctionWrapper(unsigned i) const
101 {
102     JSValue value = m_functionWrappers.get(i).get();
103     if (value.isEmpty())
104         return jsNull();
105     return value;
106 }
107 
108 void Instance::setFunctionWrapper(unsigned i, JSValue value)
109 {
110     ASSERT(m_owner);
111     ASSERT(value.isFunction(owner<JSWebAssemblyInstance>()->vm()));
112     ASSERT(!m_functionWrappers.contains(i));
113     auto locker = holdLock(owner<JSWebAssemblyInstance>()->cellLock());
114     m_functionWrappers.set(i, WriteBarrier<Unknown>(owner<JSWebAssemblyInstance>()->vm(), owner<JSWebAssemblyInstance>(), value));
115     ASSERT(getFunctionWrapper(i) == value);
116 }

118 Table* Instance::table(unsigned i)
119 {
120     RELEASE_ASSERT(i < m_module->moduleInformation().tableCount());
121     return *bitwise_cast<Table**>(bitwise_cast<char*>(this) + offsetOfTablePtr(m_numImportFunctions, i));
122 }
123 
124 void Instance::setTable(unsigned i, Ref<Table>&& table)
125 {
126     RELEASE_ASSERT(i < m_module->moduleInformation().tableCount());
127     ASSERT(!this->table(i));
128     *bitwise_cast<Table**>(bitwise_cast<char*>(this) + offsetOfTablePtr(m_numImportFunctions, i)) = &table.leakRef();
129 }
130 
131 void Instance::linkGlobal(unsigned i, Ref<Global>&& global)
132 {
133     m_globals.get()[i].m_pointer = global->valuePointer();
134     m_linkedGlobals.set(i, WTFMove(global));
135 }
