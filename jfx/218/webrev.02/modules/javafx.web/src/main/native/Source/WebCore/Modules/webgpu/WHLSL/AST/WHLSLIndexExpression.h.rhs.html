30 #include "WHLSLCodeLocation.h"
31 #include "WHLSLPropertyAccessExpression.h"
32 #include <wtf/FastMalloc.h>
33 #include <wtf/UniqueRef.h>
34 
35 namespace WebCore {
36 
37 namespace WHLSL {
38 
39 namespace AST {
40 
41 class IndexExpression final : public PropertyAccessExpression {
42     WTF_MAKE_FAST_ALLOCATED;
43 public:
44     IndexExpression(CodeLocation location, UniqueRef<Expression>&& base, UniqueRef<Expression>&& index)
45         : PropertyAccessExpression(location, Kind::Index, WTFMove(base))
46         , m_index(WTFMove(index))
47     {
48 #if CPU(ADDRESS32)
49         UNUSED_PARAM(m_padding);
50 #endif
51     }
52 
53     ~IndexExpression() = default;
54 
55     IndexExpression(const IndexExpression&) = delete;
56     IndexExpression(IndexExpression&&) = default;
57 















58     Expression& indexExpression() { return m_index; }
59     UniqueRef<Expression>& indexReference() { return m_index; }
60     UniqueRef<Expression> takeIndex() { return WTFMove(m_index); }
61 
62 private:
63     UniqueRef<Expression> m_index;
64 #if CPU(ADDRESS32)
65     // This is used to allow for replaceWith into a bigger type.
66     char m_padding[4];
67 #endif
68 };
69 
70 } // namespace AST
71 
72 }
73 
74 }
75 
76 DEFINE_DEFAULT_DELETE(IndexExpression)
77 
78 SPECIALIZE_TYPE_TRAITS_WHLSL_EXPRESSION(IndexExpression, isIndexExpression())
