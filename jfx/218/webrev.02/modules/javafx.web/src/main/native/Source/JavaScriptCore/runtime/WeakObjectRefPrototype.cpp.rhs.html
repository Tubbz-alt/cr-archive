35 const ClassInfo WeakObjectRefPrototype::s_info = { "WeakRef", &Base::s_info, nullptr, nullptr, CREATE_METHOD_TABLE(WeakObjectRefPrototype) };

37 static EncodedJSValue JSC_HOST_CALL protoFuncWeakRefDeref(JSGlobalObject*, CallFrame*);
38 
39 void WeakObjectRefPrototype::finishCreation(VM& vm, JSGlobalObject* globalObject)
40 {
41     Base::finishCreation(vm);
42     ASSERT(inherits(vm, info()));
43 
44     // FIXME: It wouldn't be hard to make this an intrinsic.
45     JSC_NATIVE_FUNCTION_WITHOUT_TRANSITION(vm.propertyNames->deref, protoFuncWeakRefDeref, static_cast<unsigned>(PropertyAttribute::DontEnum), 0);
46 
47     putDirectWithoutTransition(vm, vm.propertyNames->toStringTagSymbol, jsNontrivialString(vm, "WeakRef"_s), PropertyAttribute::DontEnum | PropertyAttribute::ReadOnly);
48 }
49 
50 ALWAYS_INLINE static JSWeakObjectRef* getWeakRef(JSGlobalObject* globalObject, JSValue value)
51 {
52     VM& vm = globalObject->vm();
53     auto scope = DECLARE_THROW_SCOPE(vm);
54 
55     if (UNLIKELY(!value.isObject())) {
56         throwTypeError(globalObject, scope, "Called WeakRef function on non-object"_s);
57         return nullptr;
58     }
59 
60     auto* ref = jsDynamicCast<JSWeakObjectRef*>(vm, asObject(value));
61     if (LIKELY(ref))
62         return ref;
63 
64     throwTypeError(globalObject, scope, "Called WeakRef function on a non-WeakRef object"_s);
65     return nullptr;
66 }
67 
68 EncodedJSValue JSC_HOST_CALL protoFuncWeakRefDeref(JSGlobalObject* globalObject, CallFrame* callFrame)
69 {
70     VM& vm = globalObject->vm();
71     auto* ref = getWeakRef(globalObject, callFrame->thisValue());
72     if (!ref)
73         return JSValue::encode(jsUndefined());
74 
75     auto* value = ref->deref(vm);
76     return value ? JSValue::encode(value) : JSValue::encode(jsNull());
77 }
