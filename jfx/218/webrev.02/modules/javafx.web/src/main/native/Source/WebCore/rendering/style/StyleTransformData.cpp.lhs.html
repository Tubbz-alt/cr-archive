22 #include "config.h"
23 #include "StyleTransformData.h"
24 
25 #include "RenderStyle.h"
26 
27 namespace WebCore {
28 


29 StyleTransformData::StyleTransformData()
30     : operations(RenderStyle::initialTransform())
31     , x(RenderStyle::initialTransformOriginX())
32     , y(RenderStyle::initialTransformOriginY())
33     , z(RenderStyle::initialTransformOriginZ())
34     , transformBox(RenderStyle::initialTransformBox())
35 {
36 }
37 
38 inline StyleTransformData::StyleTransformData(const StyleTransformData& other)
39     : RefCounted<StyleTransformData>()
40     , operations(other.operations)
41     , x(other.x)
42     , y(other.y)
43     , z(other.z)
44     , transformBox(other.transformBox)
45 {
46 }
47 
48 Ref<StyleTransformData> StyleTransformData::copy() const
49 {
50     return adoptRef(*new StyleTransformData(*this));
51 }
52 
53 bool StyleTransformData::operator==(const StyleTransformData& other) const
54 {
55     return x == other.x && y == other.y && z == other.z && transformBox == other.transformBox && operations == other.operations;
56 }
