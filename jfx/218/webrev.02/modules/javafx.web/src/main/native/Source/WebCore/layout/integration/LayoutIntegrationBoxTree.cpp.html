 37 static constexpr size_t smallTreeThreshold = 8;
 38 
 39 static RenderStyle rootBoxStyle(const RenderStyle& style)
 40 {
 41     auto clonedStyle = RenderStyle::clone(style);
 42     clonedStyle.setDisplay(DisplayType::Block);
 43     return clonedStyle;
 44 }
 45 
 46 BoxTree::BoxTree(const RenderBlockFlow& flow)
 47     : m_root({ }, rootBoxStyle(flow.style()))
 48 {
 49     if (flow.isAnonymous())
 50         m_root.setIsAnonymous();
 51 
 52     buildTree(flow);
 53 }
 54 
 55 void BoxTree::buildTree(const RenderBlockFlow& flow)
 56 {
 57     for (auto& childRenderer : childrenOfType<RenderObject>(flow)) {
 58         std::unique_ptr<Layout::Box> childBox;
 59         if (is<RenderText>(childRenderer)) {
 60             auto& textRenderer = downcast<RenderText>(childRenderer);
 61             auto textContent = Layout::TextContext { textRenderer.text(), textRenderer.canUseSimplifiedTextMeasuring() };
 62             childBox = makeUnique<Layout::Box>(WTFMove(textContent), RenderStyle::createAnonymousStyleWithDisplay(m_root.style(), DisplayType::Inline));
 63             childBox->setIsAnonymous();
 64         } else if (childRenderer.isLineBreak()) {
 65             auto clonedStyle = RenderStyle::clone(childRenderer.style());
 66             clonedStyle.setDisplay(DisplayType::Inline);
 67             clonedStyle.setFloating(Float::No);
 68             childBox = makeUnique<Layout::Box>(Layout::Box::ElementAttributes { Layout::Box::ElementType::HardLineBreak }, WTFMove(clonedStyle));
 69         }
 70         ASSERT(childBox);
 71 
 72         m_root.appendChild(*childBox);
 73         m_boxes.append({ WTFMove(childBox), &childRenderer });
 74     }
 75 }
 76 
 77 const Layout::Box* BoxTree::layoutBoxForRenderer(const RenderObject& renderer) const
 78 {
 79     if (m_boxes.size() <= smallTreeThreshold) {
 80         auto index = m_boxes.findMatching([&](auto& entry) {
 81             return entry.renderer == &renderer;
 82         });
 83         if (index == notFound)
 84             return nullptr;
 85         return m_boxes[index].box.get();
 86     }
 87 
 88     if (m_rendererToBoxMap.isEmpty()) {
 89         for (auto& entry : m_boxes)
 90             m_rendererToBoxMap.add(entry.renderer, entry.box.get());
 91     }
 92     return m_rendererToBoxMap.get(&renderer);
 93 }
 94 
 95 const RenderObject* BoxTree::rendererForLayoutBox(const Layout::Box& box) const
 96 {
 97     if (m_boxes.size() <= smallTreeThreshold) {
 98         auto index = m_boxes.findMatching([&](auto& entry) {
 99             return entry.box.get() == &box;
100         });
101         if (index == notFound)
102             return nullptr;
103         return m_boxes[index].renderer;
104     }
105 
106     if (m_boxToRendererMap.isEmpty()) {
107         for (auto& entry : m_boxes)
108             m_boxToRendererMap.add(entry.box.get(), entry.renderer);
109     }
110     return m_boxToRendererMap.get(&box);
111 }
