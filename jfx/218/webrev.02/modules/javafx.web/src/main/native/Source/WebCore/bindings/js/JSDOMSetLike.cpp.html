 35 void DOMSetAdapter::clear()
 36 {
 37     clearBackingSet(m_lexicalGlobalObject, m_backingSet);
 38 }

 40 std::pair<bool, std::reference_wrapper<JSC::JSObject>> getBackingSet(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& setLike)
 41 {
 42     auto& vm = lexicalGlobalObject.vm();
 43     auto backingSet = setLike.get(&lexicalGlobalObject, static_cast<JSVMClientData*>(vm.clientData)->builtinNames().backingSetPrivateName());
 44     if (backingSet.isUndefined()) {
 45         auto& vm = lexicalGlobalObject.vm();
 46         auto scope = DECLARE_CATCH_SCOPE(vm);
 47 
 48         backingSet = JSC::JSSet::create(&lexicalGlobalObject, vm, lexicalGlobalObject.setStructure());
 49         scope.releaseAssertNoException();
 50 
 51         setLike.putDirect(vm, static_cast<JSVMClientData*>(vm.clientData)->builtinNames().backingSetPrivateName(), backingSet, static_cast<unsigned>(JSC::PropertyAttribute::DontEnum));
 52         return { true, *JSC::asObject(backingSet) };
 53     }
 54     return { false, *JSC::asObject(backingSet) };
 55 }

 57 void clearBackingSet(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& backingSet)
 58 {
 59     auto& vm = JSC::getVM(&lexicalGlobalObject);
 60     JSC::JSValue function = backingSet.get(&lexicalGlobalObject, vm.propertyNames->clear);
 61     ASSERT(!function.isUndefined());
 62 
 63     JSC::CallData callData;
 64     JSC::CallType callType = JSC::getCallData(vm, function, callData);
 65     ASSERT(callType != JSC::CallType::None);
 66     JSC::MarkedArgumentBuffer arguments;
 67     JSC::call(&lexicalGlobalObject, function, callType, callData, &backingSet, arguments);
 68 }

 70 void addToBackingSet(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSObject& backingSet, JSC::JSValue item)

 89 JSC::JSValue forwardFunctionCallToBackingSet(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, JSC::JSObject& backingSet, const JSC::Identifier& functionName)
 90 {
 91     JSC::JSValue function = backingSet.get(&lexicalGlobalObject, functionName);
 92     ASSERT(!function.isUndefined());
 93 
 94     JSC::CallData callData;
 95     JSC::CallType callType = JSC::getCallData(lexicalGlobalObject.vm(), function, callData);
 96     ASSERT(callType != JSC::CallType::None);
 97     JSC::MarkedArgumentBuffer arguments;
 98     for (size_t cptr = 0; cptr < callFrame.argumentCount(); ++cptr)
 99         arguments.append(callFrame.uncheckedArgument(cptr));
100     ASSERT(!arguments.hasOverflowed());
101     return JSC::call(&lexicalGlobalObject, function, callType, callData, &backingSet, arguments);
102 }

104 JSC::JSValue forwardForEachCallToBackingSet(JSDOMGlobalObject& globalObject, JSC::CallFrame& callFrame, JSC::JSObject& setLike)
105 {
106     auto result = getBackingSet(globalObject, setLike);
107     ASSERT(!result.first);
108 
109     auto* function = globalObject.builtinInternalFunctions().jsDOMBindingInternals().m_forEachWrapperFunction.get();
110     ASSERT(function);
111 
112     JSC::CallData callData;
113     JSC::CallType callType = JSC::getCallData(globalObject.vm(), function, callData);
114     ASSERT(callType != JSC::CallType::None);
115 
116     JSC::MarkedArgumentBuffer arguments;
117     arguments.append(&result.second.get());
118     for (size_t cptr = 0; cptr < callFrame.argumentCount(); ++cptr)
119         arguments.append(callFrame.uncheckedArgument(cptr));
120     ASSERT(!arguments.hasOverflowed());
121     return JSC::call(&globalObject, function, callType, callData, &setLike, arguments);
122 }
