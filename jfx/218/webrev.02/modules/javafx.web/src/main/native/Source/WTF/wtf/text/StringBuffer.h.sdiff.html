38 template <typename CharType>
39 class StringBuffer {
40     WTF_MAKE_NONCOPYABLE(StringBuffer);
41     WTF_MAKE_FAST_ALLOCATED;
42 public:
43     explicit StringBuffer(unsigned length)
44         : m_length(length)
45         , m_data(m_length ? static_cast<CharType*>(fastMalloc((Checked<size_t>(m_length) * sizeof(CharType)).unsafeGet())) : nullptr)
46     {
47     }
48 
49     ~StringBuffer()
50     {
51         fastFree(m_data);
52     }
53 
54     void shrink(unsigned newLength)
55     {
56         ASSERT(newLength <= m_length);
57         m_length = newLength;
58     }
59 
60     void resize(unsigned newLength)
61     {
62         if (newLength > m_length) {
63             if (newLength > std::numeric_limits<unsigned>::max() / sizeof(UChar))
64                 CRASH();
65             m_data = static_cast<UChar*>(fastRealloc(m_data, newLength * sizeof(UChar)));
66         }
67         m_length = newLength;
68     }
69 
70     unsigned length() const { return m_length; }
71     CharType* characters() { return m_data; }
72 
73     CharType& operator[](unsigned i) { ASSERT_WITH_SECURITY_IMPLICATION(i < m_length); return m_data[i]; }
74 
75     MallocPtr<CharType> release()
76     {
77         CharType* data = m_data;
78         m_data = 0;
79         return adoptMallocPtr(data);
80     }
81 
82 private:
83     unsigned m_length;
84     CharType* m_data;
85 };

50     {
51     }
52 
53     ~StringBuffer()
54     {
55         StringBufferMalloc::free(m_data);
56     }
57 
58     void shrink(unsigned newLength)
59     {
60         ASSERT(newLength <= m_length);
61         m_length = newLength;
62     }
63 
64     void resize(unsigned newLength)
65     {
66         if (newLength > m_length) {
67             if (newLength > std::numeric_limits<unsigned>::max() / sizeof(UChar))
68                 CRASH();
69             m_data = static_cast<UChar*>(StringBufferMalloc::realloc(m_data, newLength * sizeof(UChar)));
70         }
71         m_length = newLength;
72     }
73 
74     unsigned length() const { return m_length; }
75     CharType* characters() { return m_data; }
76 
77     CharType& operator[](unsigned i) { ASSERT_WITH_SECURITY_IMPLICATION(i < m_length); return m_data[i]; }
78 
79     MallocPtr<CharType, StringBufferMalloc> release()
80     {
81         CharType* data = m_data;
82         m_data = 0;
83         return adoptMallocPtr<CharType, StringBufferMalloc>(data);
84     }
85 
86 private:
87     unsigned m_length;
88     CharType* m_data;
89 };
