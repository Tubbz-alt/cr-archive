34 class IsoSubspacePerVM::AutoremovingIsoSubspace : public IsoSubspace {
35 public:
36     AutoremovingIsoSubspace(IsoSubspacePerVM& perVM, CString name, Heap& heap, HeapCellType* heapCellType, size_t size)
37         : IsoSubspace(name, heap, heapCellType, size, /* numberOfLowerTierCells */ 0)
38         , m_perVM(perVM)
39     {
40     }
41 
42     ~AutoremovingIsoSubspace()
43     {
44         auto locker = holdLock(m_perVM.m_lock);
45         m_perVM.m_subspacePerVM.remove(&space().heap().vm());
46     }
47 
48 private:
49     IsoSubspacePerVM& m_perVM;
50 };
51 
52 IsoSubspacePerVM::IsoSubspacePerVM(Function<SubspaceParameters(VM&)> subspaceParameters)
53     : m_subspaceParameters(WTFMove(subspaceParameters))
54 {
55 }
56 
57 IsoSubspacePerVM::~IsoSubspacePerVM()
58 {
59     UNREACHABLE_FOR_PLATFORM();
60 }
61 
62 IsoSubspace& IsoSubspacePerVM::forVM(VM& vm)
63 {
64     auto locker = holdLock(m_lock);
65     auto result = m_subspacePerVM.add(&vm, nullptr);
66     if (result.isNewEntry) {
67         SubspaceParameters params = m_subspaceParameters(vm);
68         result.iterator->value = new AutoremovingIsoSubspace(*this, params.name, vm.heap, params.heapCellType, params.size);
69     }
70     return *result.iterator->value;
71 }
