 41 bool Pasteboard::isSafeTypeForDOMToReadAndWrite(const String& type)
 42 {
 43     return type == "text/plain" || type == "text/html" || type == "text/uri-list";
 44 }
 45 
 46 bool Pasteboard::canExposeURLToDOMWhenPasteboardContainsFiles(const String& urlString)
 47 {
 48     URL url({ }, urlString);
 49     return url.protocolIsInHTTPFamily() || url.protocolIsBlob() || url.protocolIsData();
 50 }

 65 Optional<Vector<PasteboardItemInfo>> Pasteboard::allPasteboardItemInfo() const
 66 {
 67 #if PLATFORM(COCOA)
 68     if (auto* strategy = platformStrategies()->pasteboardStrategy())
 69         return strategy->allPasteboardItemInfo(name(), m_changeCount);
 70 #endif
 71     return WTF::nullopt;
 72 }
 73 
 74 Optional<PasteboardItemInfo> Pasteboard::pasteboardItemInfo(size_t index) const
 75 {
 76 #if PLATFORM(COCOA)
 77     if (auto* strategy = platformStrategies()->pasteboardStrategy())
 78         return strategy->informationForItemAtIndex(index, name(), m_changeCount);
 79 #else
 80     UNUSED_PARAM(index);
 81 #endif
 82     return WTF::nullopt;
 83 }
 84 
 85 String Pasteboard::readString(size_t index, const String& type)
 86 {
 87     if (auto* strategy = platformStrategies()->pasteboardStrategy())
 88         return strategy->readStringFromPasteboard(index, type, name());
 89     return { };
 90 }
 91 
 92 RefPtr<WebCore::SharedBuffer> Pasteboard::readBuffer(size_t index, const String& type)
 93 {
 94     if (auto* strategy = platformStrategies()->pasteboardStrategy())
 95         return strategy->readBufferFromPasteboard(index, type, name());
 96     return nullptr;
 97 }
 98 
 99 URL Pasteboard::readURL(size_t index, String& title)
100 {
101     if (auto* strategy = platformStrategies()->pasteboardStrategy())
102         return strategy->readURLFromPasteboard(index, name(), title);
103     return { };
104 }
