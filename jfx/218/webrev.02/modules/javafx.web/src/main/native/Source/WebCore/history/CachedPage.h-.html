26 #pragma once
27 
28 #include "CachedFrame.h"
29 
30 namespace WebCore {
31 
32 class Document;
33 class DocumentLoader;
34 class Page;
35 
36 class CachedPage {
37     WTF_MAKE_FAST_ALLOCATED;
38 public:
39     explicit CachedPage(Page&);
40     ~CachedPage();
41 
42     void restore(Page&);
43     void clear();
44 
45     Page& page() const { return m_page; }
46     Document* document() const { return m_cachedMainFrame->document(); }
47     DocumentLoader* documentLoader() const { return m_cachedMainFrame->documentLoader(); }
48 
49     bool hasExpired() const;
50 
51     CachedFrame* cachedMainFrame() { return m_cachedMainFrame.get(); }
52 
53 #if ENABLE(VIDEO_TRACK)
54     void markForCaptionPreferencesChanged() { m_needsCaptionPreferencesChanged = true; }
55 #endif
56 
57     void markForDeviceOrPageScaleChanged() { m_needsDeviceOrPageScaleChanged = true; }
58 
59     void markForContentsSizeChanged() { m_needsUpdateContentsSize = true; }
60 
61 private:
62     Page& m_page;
63     MonotonicTime m_expirationTime;
64     std::unique_ptr<CachedFrame> m_cachedMainFrame;
65 #if ENABLE(VIDEO_TRACK)
66     bool m_needsCaptionPreferencesChanged { false };
67 #endif
68     bool m_needsDeviceOrPageScaleChanged { false };
69     bool m_needsUpdateContentsSize { false };
70 };
71 
72 } // namespace WebCore
