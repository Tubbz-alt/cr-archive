26 #pragma once
27 
28 #if ENABLE(WEBGL)
29 
30 #include "GraphicsContextGLOpenGL.h"
31 #include "WebGLBuffer.h"
32 #include "WebGLContextObject.h"
33 
34 namespace WebCore {
35 
36 class WebGLVertexArrayObjectBase : public WebGLContextObject {
37 public:
38     enum class Type { Default, User };
39 
40     // Cached values for vertex attrib range checks
41     struct VertexAttribState {
42         bool isBound() const { return bufferBinding && bufferBinding->object(); }
43         bool validateBinding() const { return !enabled || isBound(); }
44 
45         bool enabled { false };
46         RefPtr<WebGLBuffer> bufferBinding;
47         GCGLsizei bytesPerElement { 0 };
48         GCGLint size { 4 };
49         GCGLenum type { GraphicsContextGL::FLOAT };
50         bool normalized { false };
51         GCGLsizei stride { 16 };
52         GCGLsizei originalStride { 0 };
53         GCGLintptr offset { 0 };
54         GCGLuint divisor { 0 };
55     };
56 
57     bool isDefaultObject() const { return m_type == Type::Default; }
58 
59     bool hasEverBeenBound() const { return object() && m_hasEverBeenBound; }
60     void setHasEverBeenBound() { m_hasEverBeenBound = true; }
61 
62     WebGLBuffer* getElementArrayBuffer() const { return m_boundElementArrayBuffer.get(); }
63     void setElementArrayBuffer(WebGLBuffer*);
64 
65     VertexAttribState& getVertexAttribState(int index) { return m_vertexAttribState[index]; }
66     void setVertexAttribState(GCGLuint, GCGLsizei, GCGLint, GCGLenum, GCGLboolean, GCGLsizei, GCGLintptr, WebGLBuffer*);
67     void unbindBuffer(WebGLBuffer&);
68 
69     void setVertexAttribDivisor(GCGLuint index, GCGLuint divisor);
70 
71 protected:
72     WebGLVertexArrayObjectBase(WebGLRenderingContextBase&, Type);
73     void deleteObjectImpl(GraphicsContextGLOpenGL*, PlatformGLObject) override = 0;
74 
75     Type m_type;
76     bool m_hasEverBeenBound { false };
77     RefPtr<WebGLBuffer> m_boundElementArrayBuffer;
78     Vector<VertexAttribState> m_vertexAttribState;
79 };
80 
81 } // namespace WebCore
82 
83 #endif
