 46 static const int defaultMemLevel = 8;
 47 static const size_t bufferIncrementUnit = 4096;
 48 
 49 WebSocketDeflater::WebSocketDeflater(int windowBits, ContextTakeOverMode contextTakeOverMode)
 50     : m_windowBits(windowBits)
 51     , m_contextTakeOverMode(contextTakeOverMode)
 52 {
 53     ASSERT(m_windowBits >= 8);
 54     ASSERT(m_windowBits <= 15);
 55     m_stream = makeUniqueWithoutFastMallocCheck<z_stream>();
 56     memset(m_stream.get(), 0, sizeof(z_stream));
 57 }
 58 
 59 bool WebSocketDeflater::initialize()
 60 {
 61     return deflateInit2(m_stream.get(), Z_DEFAULT_COMPRESSION, Z_DEFLATED, -m_windowBits, defaultMemLevel, Z_DEFAULT_STRATEGY) == Z_OK;
 62 }
 63 
 64 WebSocketDeflater::~WebSocketDeflater()
 65 {
 66     int result = deflateEnd(m_stream.get());
 67     if (result != Z_OK)
 68         LOG(Network, "WebSocketDeflater %p Destructor deflateEnd() failed: %d is returned", this, result);
 69 }

 71 static void setStreamParameter(z_stream* stream, const char* inputData, size_t inputLength, char* outputData, size_t outputLength)
 72 {
 73     stream->next_in = reinterpret_cast<Bytef*>(const_cast<char*>(inputData));
 74     stream->avail_in = inputLength;
 75     stream->next_out = reinterpret_cast<Bytef*>(outputData);
 76     stream->avail_out = outputLength;
 77 }
 78 
 79 bool WebSocketDeflater::addBytes(const char* data, size_t length)
 80 {
 81     if (!length)
 82         return false;
 83 
 84     size_t maxLength = deflateBound(m_stream.get(), length);
 85     size_t writePosition = m_buffer.size();
 86     m_buffer.grow(writePosition + maxLength);
 87     setStreamParameter(m_stream.get(), data, length, m_buffer.data() + writePosition, maxLength);
 88     int result = deflate(m_stream.get(), Z_NO_FLUSH);
 89     if (result != Z_OK || m_stream->avail_in > 0)
 90         return false;
 91 
 92     m_buffer.shrink(writePosition + maxLength - m_stream->avail_out);
 93     return true;
 94 }
 95 
 96 bool WebSocketDeflater::finish()
 97 {
 98     while (true) {
 99         size_t writePosition = m_buffer.size();
100         m_buffer.grow(writePosition + bufferIncrementUnit);
101         size_t availableCapacity = m_buffer.size() - writePosition;
102         setStreamParameter(m_stream.get(), 0, 0, m_buffer.data() + writePosition, availableCapacity);
103         int result = deflate(m_stream.get(), Z_SYNC_FLUSH);
104         if (m_stream->avail_out) {
105             m_buffer.shrink(writePosition + availableCapacity - m_stream->avail_out);
106             if (result == Z_OK)
107                 break;
108             if (result != Z_BUF_ERROR)
109                 return false;
110         }
111     }
112     // Remove 4 octets from the tail as the specification requires.
113     if (m_buffer.size() <= 4)
114         return false;
115     m_buffer.shrink(m_buffer.size() - 4);
116     return true;
117 }
118 
119 void WebSocketDeflater::reset()
120 {
121     m_buffer.clear();
122     if (m_contextTakeOverMode == DoNotTakeOverContext)
123         deflateReset(m_stream.get());
124 }

126 WebSocketInflater::WebSocketInflater(int windowBits)
127     : m_windowBits(windowBits)
128 {
129     m_stream = makeUniqueWithoutFastMallocCheck<z_stream>();
130     memset(m_stream.get(), 0, sizeof(z_stream));
131 }
132 
133 bool WebSocketInflater::initialize()
134 {
135     return inflateInit2(m_stream.get(), -m_windowBits) == Z_OK;
136 }
137 
138 WebSocketInflater::~WebSocketInflater()
139 {
140     int result = inflateEnd(m_stream.get());
141     if (result != Z_OK)
142         LOG(Network, "WebSocketInflater %p Destructor inflateEnd() failed: %d is returned", this, result);
143 }
144 
145 bool WebSocketInflater::addBytes(const char* data, size_t length)
146 {
147     if (!length)
148         return false;
149 
150     size_t consumedSoFar = 0;
151     while (consumedSoFar < length) {
152         size_t writePosition = m_buffer.size();
153         m_buffer.grow(writePosition + bufferIncrementUnit);
154         size_t availableCapacity = m_buffer.size() - writePosition;
155         size_t remainingLength = length - consumedSoFar;
156         setStreamParameter(m_stream.get(), data + consumedSoFar, remainingLength, m_buffer.data() + writePosition, availableCapacity);
157         int result = inflate(m_stream.get(), Z_NO_FLUSH);
158         consumedSoFar += remainingLength - m_stream->avail_in;
159         m_buffer.shrink(writePosition + availableCapacity - m_stream->avail_out);
160         if (result == Z_BUF_ERROR)
161             continue;
162         if (result == Z_STREAM_END) {
163             // Received a block with BFINAL set to 1. Reset decompression state.
164             if (inflateReset(m_stream.get()) != Z_OK)
165                 return false;
166             continue;
167         }
168         if (result != Z_OK)
169             return false;
170         ASSERT(remainingLength > m_stream->avail_in);
171     }
172     ASSERT(consumedSoFar == length);
173     return true;
174 }
175 
176 bool WebSocketInflater::finish()
177 {
178     static const char* strippedFields = "\0\0\xff\xff";
179     static const size_t strippedLength = 4;
180 
181     // Appends 4 octests of 0x00 0x00 0xff 0xff
182     size_t consumedSoFar = 0;
183     while (consumedSoFar < strippedLength) {
184         size_t writePosition = m_buffer.size();
185         m_buffer.grow(writePosition + bufferIncrementUnit);
186         size_t availableCapacity = m_buffer.size() - writePosition;
187         size_t remainingLength = strippedLength - consumedSoFar;
188         setStreamParameter(m_stream.get(), strippedFields + consumedSoFar, remainingLength, m_buffer.data() + writePosition, availableCapacity);
189         int result = inflate(m_stream.get(), Z_FINISH);
190         consumedSoFar += remainingLength - m_stream->avail_in;
191         m_buffer.shrink(writePosition + availableCapacity - m_stream->avail_out);
192         if (result == Z_BUF_ERROR)
193             continue;
194         if (result != Z_OK && result != Z_STREAM_END)
195             return false;
196         ASSERT(remainingLength > m_stream->avail_in);
197     }
198     ASSERT(consumedSoFar == strippedLength);
199 
200     return true;
201 }
202 
203 void WebSocketInflater::reset()
204 {
205     m_buffer.clear();
206 }
