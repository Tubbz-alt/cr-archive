35 class WEBCORE_EXPORT RuntimeMethod : public InternalFunction {
36 public:
37     typedef InternalFunction Base;
38     static const unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetCallData;
39 
40     template<typename CellType, JSC::SubspaceAccess>
41     static IsoSubspace* subspaceFor(VM& vm)
42     {
43         static_assert(sizeof(CellType) == sizeof(RuntimeMethod), "RuntimeMethod subclasses that add fields need to override subspaceFor<>()");

44         return subspaceForImpl(vm);
45     }
46 
47     static RuntimeMethod* create(ExecState*, JSGlobalObject* globalObject, Structure* structure, const String& name, Bindings::Method* method)
48     {
49         VM& vm = globalObject->vm();
50         RuntimeMethod* runtimeMethod = new (NotNull, allocateCell<RuntimeMethod>(vm.heap)) RuntimeMethod(globalObject, structure, method);
51         runtimeMethod->finishCreation(vm, name);
52         return runtimeMethod;
53     }
54 
55     Bindings::Method* method() const { return m_method; }
56 
57     DECLARE_INFO;
58 
59     static FunctionPrototype* createPrototype(VM&, JSGlobalObject& globalObject)
60     {
61         return globalObject.functionPrototype();
62     }
63 
64     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
65     {
66         return Structure::create(vm, globalObject, prototype, TypeInfo(InternalFunctionType, StructureFlags), info());
67     }
68 
69 protected:
70     RuntimeMethod(JSGlobalObject*, Structure*, Bindings::Method*);
71     void finishCreation(VM&, const String&);
72 
73     static bool getOwnPropertySlot(JSObject*, ExecState*, PropertyName, PropertySlot&);
74 
75 private:
76     static EncodedJSValue lengthGetter(ExecState*, EncodedJSValue, PropertyName);
77 
78     static IsoSubspace* subspaceForImpl(VM&);

35 class WEBCORE_EXPORT RuntimeMethod : public InternalFunction {
36 public:
37     using Base = InternalFunction;
38     static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetCallData;
39 
40     template<typename CellType, JSC::SubspaceAccess>
41     static IsoSubspace* subspaceFor(JSC::VM& vm)
42     {
43         static_assert(sizeof(CellType) == sizeof(RuntimeMethod), "RuntimeMethod subclasses that add fields need to override subspaceFor<>()");
44         static_assert(CellType::destroy == JSC::JSCell::destroy);
45         return subspaceForImpl(vm);
46     }
47 
48     static RuntimeMethod* create(JSGlobalObject*, JSGlobalObject* globalObject, Structure* structure, const String& name, Bindings::Method* method)
49     {
50         VM& vm = globalObject->vm();
51         RuntimeMethod* runtimeMethod = new (NotNull, allocateCell<RuntimeMethod>(vm.heap)) RuntimeMethod(vm, structure, method);
52         runtimeMethod->finishCreation(vm, name);
53         return runtimeMethod;
54     }
55 
56     Bindings::Method* method() const { return m_method; }
57 
58     DECLARE_INFO;
59 
60     static FunctionPrototype* createPrototype(VM&, JSGlobalObject& globalObject)
61     {
62         return globalObject.functionPrototype();
63     }
64 
65     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
66     {
67         return Structure::create(vm, globalObject, prototype, TypeInfo(InternalFunctionType, StructureFlags), info());
68     }
69 
70 protected:
71     RuntimeMethod(VM&, Structure*, Bindings::Method*);
72     void finishCreation(VM&, const String&);
73 
74     static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
75 
76 private:
77     static EncodedJSValue lengthGetter(JSGlobalObject*, EncodedJSValue, PropertyName);
78 
79     static IsoSubspace* subspaceForImpl(VM&);
80 
81     Bindings::Method* m_method;
82 };

84 } // namespace JSC
