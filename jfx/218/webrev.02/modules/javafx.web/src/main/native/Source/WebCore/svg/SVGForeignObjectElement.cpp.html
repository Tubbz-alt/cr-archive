 38 inline SVGForeignObjectElement::SVGForeignObjectElement(const QualifiedName& tagName, Document& document)
 39     : SVGGraphicsElement(tagName, document)
 40 {
 41     ASSERT(hasTagName(SVGNames::foreignObjectTag));
 42     static std::once_flag onceFlag;
 43     std::call_once(onceFlag, [] {
 44         PropertyRegistry::registerProperty<SVGNames::xAttr, &SVGForeignObjectElement::m_x>();
 45         PropertyRegistry::registerProperty<SVGNames::yAttr, &SVGForeignObjectElement::m_y>();
 46         PropertyRegistry::registerProperty<SVGNames::widthAttr, &SVGForeignObjectElement::m_width>();
 47         PropertyRegistry::registerProperty<SVGNames::heightAttr, &SVGForeignObjectElement::m_height>();
 48     });
 49 }

 51 Ref<SVGForeignObjectElement> SVGForeignObjectElement::create(const QualifiedName& tagName, Document& document)
 52 {
 53     return adoptRef(*new SVGForeignObjectElement(tagName, document));
 54 }
 55 
 56 void SVGForeignObjectElement::parseAttribute(const QualifiedName& name, const AtomString& value)
 57 {
 58     SVGParsingError parseError = NoError;
 59 
 60     if (name == SVGNames::xAttr)
 61         m_x->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
 62     else if (name == SVGNames::yAttr)
 63         m_y->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
 64     else if (name == SVGNames::widthAttr)
 65         m_width->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
 66     else if (name == SVGNames::heightAttr)
 67         m_height->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
 68 
 69     reportAttributeParsingError(parseError, name, value);
 70 
 71     SVGGraphicsElement::parseAttribute(name, value);
 72 }
 73 
 74 void SVGForeignObjectElement::svgAttributeChanged(const QualifiedName& attrName)
 75 {
 76     if (attrName == SVGNames::widthAttr || attrName == SVGNames::heightAttr) {
 77         invalidateSVGPresentationAttributeStyle();
 78         return;
 79     }
 80 
 81     if (attrName == SVGNames::xAttr || attrName == SVGNames::yAttr) {
 82         updateRelativeLengthsInformation();
 83         if (auto renderer = this->renderer())
 84             RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
 85         return;
 86     }
 87 
 88     SVGGraphicsElement::svgAttributeChanged(attrName);
 89 }

 93     return createRenderer<RenderSVGForeignObject>(*this, WTFMove(style));

 96 bool SVGForeignObjectElement::childShouldCreateRenderer(const Node& child) const
 97 {
 98     // Disallow arbitary SVG content. Only allow proper <svg xmlns="svgNS"> subdocuments.
 99     if (child.isSVGElement())
100         return child.hasTagName(SVGNames::svgTag);
101 
102     // Skip over SVG rules which disallow non-SVG kids
103     return StyledElement::childShouldCreateRenderer(child);
104 }
105 
106 bool SVGForeignObjectElement::rendererIsNeeded(const RenderStyle& style)
107 {
108     // Suppress foreignObject renderers in SVG hidden containers.
109     // (https://bugs.webkit.org/show_bug.cgi?id=87297)
110     // Note that we currently do not support foreignObject instantiation via <use>, hence it is safe
111     // to use parentElement() here. If that changes, this method should be updated to use
112     // parentOrShadowHostElement() instead.
113     auto ancestor = makeRefPtr(parentElement());
114     while (ancestor && ancestor->isSVGElement()) {
115         if (ancestor->renderer() && ancestor->renderer()->isSVGHiddenContainer())
116             return false;
117 
118         ancestor = ancestor->parentElement();
119     }
120 
121     return SVGGraphicsElement::rendererIsNeeded(style);
122 }
