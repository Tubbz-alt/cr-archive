 55 JNIEXPORT void JNICALL Java_com_sun_javafx_webkit_drt_DumpRenderTree_initTest
 56     (JNIEnv* env, jclass cls, jstring testPath, jstring pixelsHash)
 57 {
 58     const char* testPathChars = env->GetStringUTFChars(testPath, NULL);
 59     const char* pixelsHashChars = env->GetStringUTFChars(pixelsHash, NULL);
 60 
 61     ASSERT(!gTestRunner);
 62     gTestRunner = TestRunner::create(testPathChars, pixelsHashChars);
 63     ASSERT(!gGCController);
 64     gGCController = std::make_unique<GCController>();
 65 
 66     DRT::WorkQueue::singleton().clear();
 67 
 68     env->ReleaseStringUTFChars(testPath, testPathChars);
 69     env->ReleaseStringUTFChars(pixelsHash, pixelsHashChars);
 70 }

 72 JNIEXPORT void JNICALL Java_com_sun_javafx_webkit_drt_DumpRenderTree_didClearWindowObject
 73     (JNIEnv* env, jclass cls, jlong pContext, jlong pWindowObject,
 74     jobject eventSender)
 75 {
 76     if(!gTestRunner || !gGCController)
 77         return;
 78     ASSERT(pContext);
 79     ASSERT(pWindowObject);
 80     ASSERT(eventSender);
 81 
 82     gContext = static_cast<JSGlobalContextRef>(jlong_to_ptr(pContext));
 83     JSObjectRef windowObject =
 84             static_cast<JSObjectRef>(jlong_to_ptr(pWindowObject));
 85 
 86     JSValueRef exception = 0;
 87 
 88     gTestRunner->makeWindowObject(gContext, windowObject, &exception);
 89     ASSERT(!exception);
 90 
 91     JLObject jlEventSender(eventSender, true);
 92     makeEventSender(gContext, windowObject, jlEventSender, &exception);
 93     ASSERT(!exception);
 94     WebCoreTestSupport::injectInternalsObject(gContext);
 95     gGCController->makeWindowObject(gContext, windowObject, &exception);
 96     ASSERT(!exception);
 97 }

 99 JNIEXPORT void JNICALL Java_com_sun_javafx_webkit_drt_DumpRenderTree_dispose
100     (JNIEnv* env, jclass cls)

108 }

145 JNIEXPORT jobjectArray JNICALL Java_com_sun_javafx_webkit_drt_DumpRenderTree_openPanelFiles
146     (JNIEnv* env, jclass)
147 {
148     ASSERT(gTestRunner);
149     const auto& openFiles = gTestRunner->openPanelFiles();
150     static JGClass stringCls = env->FindClass("java/lang/String");
151     ASSERT(stringCls);
152     jobjectArray files = env->NewObjectArray(openFiles.size(), stringCls, env->NewStringUTF(""));
153     for (auto i = 0; i < openFiles.size(); i++) {
154         env->SetObjectArrayElement(files, i, env->NewStringUTF(openFiles[i].c_str()));
155     }
156     return files;
157 }
