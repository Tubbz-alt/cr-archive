26 #ifndef MediaPlaybackTargetPicker_h
27 #define MediaPlaybackTargetPicker_h
28 
29 #if ENABLE(WIRELESS_PLAYBACK_TARGET)
30 
31 #include <wtf/Ref.h>
32 #include <wtf/RunLoop.h>
33 
34 namespace WebCore {
35 
36 class FloatRect;
37 class MediaPlaybackTarget;
38 
39 class MediaPlaybackTargetPicker {
40 public:
41     class Client {
42     protected:
43         virtual ~Client() = default;
44 
45     public:
46         virtual void setPlaybackTarget(Ref<MediaPlaybackTarget>&&) = 0;
47         virtual void externalOutputDeviceAvailableDidChange(bool) = 0;
48         virtual void playbackTargetPickerWasDismissed() = 0;
49 
50         void invalidate();
51     };
52 
53     virtual ~MediaPlaybackTargetPicker();
54 
55     virtual void showPlaybackTargetPicker(const FloatRect&, bool checkActiveRoute, bool useDarkAppearance);
56     virtual void startingMonitoringPlaybackTargets();
57     virtual void stopMonitoringPlaybackTargets();
58     virtual void invalidatePlaybackTargets();
59 
60     void availableDevicesDidChange() { addPendingAction(OutputDeviceAvailabilityChanged); }
61     void currentDeviceDidChange() { addPendingAction(CurrentDeviceDidChange); }
62     void playbackTargetPickerWasDismissed() { addPendingAction(PlaybackTargetPickerWasDismissed); }
63 
64 protected:
65     explicit MediaPlaybackTargetPicker(Client&);
66 
67     enum ActionType {
68         OutputDeviceAvailabilityChanged = 1 << 0,
69         CurrentDeviceDidChange = 1 << 1,
70         PlaybackTargetPickerWasDismissed = 1 << 2,
71     };
72     typedef unsigned PendingActionFlags;
73 
74     void addPendingAction(PendingActionFlags);
75     void pendingActionTimerFired();
76     Client* client() const { return m_client; }
77     void setClient(Client* client) { m_client = client; }
78 
79 private:
80     virtual bool externalOutputDeviceAvailable() = 0;
81     virtual Ref<MediaPlaybackTarget> playbackTarget() = 0;
82 
83     PendingActionFlags m_pendingActionFlags { 0 };
84     Client* m_client;
85     RunLoop::Timer<MediaPlaybackTargetPicker> m_pendingActionTimer;
86 };
87 
88 } // namespace WebCore
89 
90 #endif // ENABLE(WIRELESS_PLAYBACK_TARGET)
91 
92 #endif // MediaPlaybackTargetPicker_h
