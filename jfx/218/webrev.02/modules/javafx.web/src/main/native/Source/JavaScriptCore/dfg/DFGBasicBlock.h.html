 49 struct BasicBlock : RefCounted<BasicBlock> {
 50     WTF_MAKE_STRUCT_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(BasicBlock);
 51     BasicBlock(
 52         BytecodeIndex bytecodeBegin, unsigned numArguments, unsigned numLocals, unsigned numTmps,
 53         float executionCount);
 54     ~BasicBlock();
 55 
 56     void ensureLocals(unsigned newNumLocals);
 57     void ensureTmps(unsigned newNumTmps);
 58 
 59     size_t size() const { return m_nodes.size(); }
 60     bool isEmpty() const { return !size(); }
 61     Node*& at(size_t i) { return m_nodes[i]; }
 62     Node* at(size_t i) const { return m_nodes[i]; }
 63     Node* tryAt(size_t i) const
 64     {
 65         if (i >= size())
 66             return nullptr;
 67         return at(i);
 68     }
 69     Node*& operator[](size_t i) { return at(i); }
 70     Node* operator[](size_t i) const { return at(i); }
 71     Node* last() const
 72     {
 73         RELEASE_ASSERT(!!size());
 74         return at(size() - 1);
 75     }
 76 
 77     // Use this to find both the index of the terminal and the terminal itself in one go. May
 78     // return a clear NodeAndIndex if the basic block currently lacks a terminal. That may happen
 79     // in the middle of IR transformations within a phase but should never be the case in between
 80     // phases.
 81     //
 82     // The reason why this is more than just "at(size() - 1)" is that we may place non-terminal
 83     // liveness marking instructions after the terminal. This is supposed to happen infrequently
 84     // but some basic blocks - most notably return blocks - will have liveness markers for all of
 85     // the flushed variables right after the return.
 86     //
 87     // It turns out that doing this linear search is basically perf-neutral, so long as we force
 88     // the method to be inlined. Hence the ALWAYS_INLINE.
 89     ALWAYS_INLINE NodeAndIndex findTerminal() const
 90     {
 91         size_t i = size();
 92         while (i--) {
 93             Node* node = at(i);
 94             if (node->isTerminal())
 95                 return NodeAndIndex(node, i);
 96             switch (node->op()) {
 97             // The bitter end can contain Phantoms and the like. There will probably only be one or two nodes after the terminal. They are all no-ops and will not have any checked children.
 98             case Check: // This is here because it's our universal no-op.
 99             case CheckVarargs:
100             case Phantom:
101             case PhantomLocal:
102             case Flush:
103                 break;
104             default:
105                 return NodeAndIndex();
106             }
107         }
108         return NodeAndIndex();
109     }
110 
111     ALWAYS_INLINE Node* terminal() const
112     {
113         return findTerminal().node;
114     }
115 
116     void resize(size_t size) { m_nodes.resize(size); }
117     void grow(size_t size) { m_nodes.grow(size); }
118 
119     void append(Node* node) { m_nodes.append(node); }
120     void insertBeforeTerminal(Node* node)
121     {
122         NodeAndIndex result = findTerminal();
123         if (!result)
124             append(node);
125         else
126             m_nodes.insert(result.index, node);
127     }
128 
129     void replaceTerminal(Graph&, Node*);
130 
131     size_t numNodes() const { return phis.size() + size(); }
132     Node* node(size_t i) const
133     {
134         if (i < phis.size())
135             return phis[i];
136         return at(i - phis.size());
137     }
138     bool isPhiIndex(size_t i) const { return i < phis.size(); }
139 
140     bool isInPhis(Node* node) const;
141     bool isInBlock(Node* myNode) const;
142 
143     BlockNodeList::iterator begin() { return m_nodes.begin(); }
144     BlockNodeList::iterator end() { return m_nodes.end(); }
145 
146     unsigned numSuccessors() { return terminal()->numSuccessors(); }
147 
148     BasicBlock*& successor(unsigned index)
149     {
150         return terminal()->successor(index);
151     }
152     BasicBlock*& successorForCondition(bool condition)
153     {
154         return terminal()->successorForCondition(condition);
155     }
156 
157     Node::SuccessorsIterable successors()
158     {
159         return terminal()->successors();
160     }
161 
162     void removePredecessor(BasicBlock* block);
163     void replacePredecessor(BasicBlock* from, BasicBlock* to);
164

180 #endif
181     }

184     // for other purposes due to inlining.
185     BytecodeIndex bytecodeBegin;

189     StructureClobberState cfaStructureClobberStateAtHead;
190     StructureClobberState cfaStructureClobberStateAtTail;
191     BranchDirection cfaBranchDirection;
192     bool cfaHasVisited;
193     bool cfaShouldRevisit;
194     bool cfaThinksShouldTryConstantFolding { false };
195     bool cfaDidFinish;
196     bool intersectionOfCFAHasVisited;
197     bool isOSRTarget;
198     bool isCatchEntrypoint;
199 
200 #if ASSERT_ENABLED
201     bool isLinked;
202 #endif
203     bool isReachable;
204 
205     Vector<Node*> phis;
206     PredecessorList predecessors;
207 
208     Operands<Node*> variablesAtHead;
209     Operands<Node*> variablesAtTail;
210 
211     Operands<AbstractValue> valuesAtHead;
212     Operands<AbstractValue> valuesAtTail;
213 
214     // The intersection of assumptions we have made previously at the head of this block. Note
215     // that under normal circumstances, each time we run the CFA, we will get strictly more precise
216     // results. But we don't actually require this to be the case. It's fine for the CFA to loosen
217     // up for any odd reason. It's fine when this happens, because anything that the CFA proves
218     // must be true from that point forward, except if some registered watchpoint fires, in which
219     // case the code won't ever run. So, the CFA proving something less precise later on is just an
220     // outcome of the CFA being imperfect; the more precise thing that it had proved earlier is no
221     // less true.
222     //
223     // But for the purpose of OSR entry, we need to make sure that we remember what assumptions we
224     // had used for optimizing any given basic block. That's what this is for.
225     //
226     // It's interesting that we could use this to make the CFA more precise: all future CFAs could
227     // filter their results with this thing to sort of maintain maximal precision. Because we
228     // expect CFA to usually be monotonically more precise each time we run it to fixpoint, this
229     // would not be a productive optimization: it would make setting up a basic block more
230     // expensive and would only benefit bizarre pathological cases.
231     Operands<AbstractValue> intersectionOfPastValuesAtHead;
232 
233     float executionCount;

235     struct SSAData {
236         WTF_MAKE_FAST_ALLOCATED;
237     public:
238         void invalidate()
239         {
240             liveAtTail.clear();
241             liveAtHead.clear();
242             valuesAtHead.clear();
243             valuesAtTail.clear();
244         }
245 
246         AvailabilityMap availabilityAtHead;
247         AvailabilityMap availabilityAtTail;
248 
249         Vector<NodeFlowProjection> liveAtHead;
250         Vector<NodeFlowProjection> liveAtTail;
251         Vector<NodeAbstractValuePair> valuesAtHead;
252         Vector<NodeAbstractValuePair> valuesAtTail;
253 
254         SSAData(BasicBlock*);
255         ~SSAData();
256     };
257     std::unique_ptr<SSAData> ssa;
258 
259 private:
260     friend class InsertionSet;
261     BlockNodeList m_nodes;
262 };
