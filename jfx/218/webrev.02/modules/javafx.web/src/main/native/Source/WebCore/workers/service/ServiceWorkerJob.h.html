41 namespace WebCore {
42 
43 class DeferredPromise;
44 class Exception;
45 class ScriptExecutionContext;
46 enum class ServiceWorkerJobType;
47 struct ServiceWorkerRegistrationData;
48 
49 class ServiceWorkerJob : public WorkerScriptLoaderClient {
50     WTF_MAKE_FAST_ALLOCATED;
51 public:
52     ServiceWorkerJob(ServiceWorkerJobClient&, RefPtr<DeferredPromise>&&, ServiceWorkerJobData&&);
53     WEBCORE_EXPORT ~ServiceWorkerJob();
54 
55     void failedWithException(const Exception&);
56     void resolvedWithRegistration(ServiceWorkerRegistrationData&&, ShouldNotifyWhenResolved);
57     void resolvedWithUnregistrationResult(bool);
58     void startScriptFetch(FetchOptions::Cache);
59 
60     using Identifier = ServiceWorkerJobIdentifier;
61     Identifier identifier() const { return m_jobData.identifier().jobIdentifier; }
62 
63     const ServiceWorkerJobData& data() const { return m_jobData; }
64     bool hasPromise() const { return !!m_promise; }
65     RefPtr<DeferredPromise> takePromise();
66 
67     void fetchScriptWithContext(ScriptExecutionContext&, FetchOptions::Cache);
68 
69     const DocumentOrWorkerIdentifier& contextIdentifier() { return m_contextIdentifier; }
70 
71     bool cancelPendingLoad();
72 
73     WEBCORE_EXPORT static ResourceError validateServiceWorkerResponse(const ServiceWorkerJobData&, const ResourceResponse&);
74 
75 private:
76     // WorkerScriptLoaderClient
77     void didReceiveResponse(unsigned long identifier, const ResourceResponse&) final;
78     void notifyFinished() final;
79 
80     ServiceWorkerJobClient& m_client;
81     ServiceWorkerJobData m_jobData;
82     RefPtr<DeferredPromise> m_promise;
83 
84     bool m_completed { false };
85 
86     DocumentOrWorkerIdentifier m_contextIdentifier;
87     RefPtr<WorkerScriptLoader> m_scriptLoader;
88 
89 #if ASSERT_ENABLED
90     Ref<Thread> m_creationThread { Thread::current() };
91 #endif
92 };
