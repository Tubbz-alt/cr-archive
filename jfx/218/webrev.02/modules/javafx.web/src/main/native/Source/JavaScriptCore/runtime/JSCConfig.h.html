28 #include "OptionsList.h"

32 namespace JSC {
33 
34 class ExecutableAllocator;
35 class FixedVMPoolExecutableAllocator;
36 class VM;
37 
38 constexpr size_t ConfigSizeToProtect = CeilingOnPageSize;
39 
40 #if ENABLE(SEPARATED_WX_HEAP)
41 using JITWriteSeparateHeapsFunction = void (*)(off_t, const void*, size_t);
42 #endif
43 
44 struct Config {
45     JS_EXPORT_PRIVATE static void disableFreezingForTesting();
46     JS_EXPORT_PRIVATE static void enableRestrictedOptions();
47     JS_EXPORT_PRIVATE static void permanentlyFreeze();
48 
49     static void configureForTesting()
50     {
51         disableFreezingForTesting();
52         enableRestrictedOptions();
53     }
54 
55     union {
56         struct {
57             // All the fields in this struct should be chosen such that their
58             // initial value is 0 / null / falsy because Config is instantiated
59             // as a global singleton.
60 
61             bool isPermanentlyFrozen;
62             bool disabledFreezingForTesting;
63             bool restrictedOptionsEnabled;
64             bool jitDisabled;
65 
66             // The following HasBeenCalled flags are for auditing call_once initialization functions.
67             bool initializeThreadingHasBeenCalled;
68 
69             ExecutableAllocator* executableAllocator;
70             FixedVMPoolExecutableAllocator* fixedVMPoolExecutableAllocator;
71             void* startExecutableMemory;
72             void* endExecutableMemory;
73             uintptr_t startOfFixedWritableMemoryPool;
74 
75 #if ENABLE(SEPARATED_WX_HEAP)
76             JITWriteSeparateHeapsFunction jitWriteSeparateHeaps;
77             bool useFastPermisionsJITCopy;
78 #endif
79 
80             OptionsStorage options;
81 
82             void (*shellTimeoutCheckCallback)(VM&);
83         };
84         char ensureSize[ConfigSizeToProtect];
85     };
86 };
87 
88 extern "C" alignas(ConfigSizeToProtect) JS_EXPORT_PRIVATE Config g_jscConfig;
89 
90 static_assert(sizeof(Config) == ConfigSizeToProtect, "");
91 
92 } // namespace JSC
