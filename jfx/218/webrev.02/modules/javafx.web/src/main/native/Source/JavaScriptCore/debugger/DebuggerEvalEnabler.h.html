26 #pragma once
27 
28 #include "CallFrame.h"
29 #include "JSGlobalObject.h"
30 
31 namespace JSC {
32 
33 class DebuggerEvalEnabler {
34 public:
35     enum class Mode {
36         EvalOnCurrentCallFrame,
37         EvalOnGlobalObjectAtDebuggerEntry,
38     };
39 
40     DebuggerEvalEnabler(JSGlobalObject* globalObject, Mode mode = Mode::EvalOnCurrentCallFrame)
41         : m_globalObject(globalObject)
42 #if ASSERT_ENABLED
43         , m_mode(mode)
44 #endif
45 
46     {
47         UNUSED_PARAM(mode);
48         if (globalObject) {
49             m_evalWasDisabled = !globalObject->evalEnabled();
50             if (m_evalWasDisabled)
51                 globalObject->setEvalEnabled(true, globalObject->evalDisabledErrorMessage());
52 #if ASSERT_ENABLED
53             if (m_mode == Mode::EvalOnGlobalObjectAtDebuggerEntry)
54                 globalObject->setGlobalObjectAtDebuggerEntry(globalObject);
55 #endif
56         }
57     }
58 
59     ~DebuggerEvalEnabler()
60     {
61         if (m_globalObject) {
62             JSGlobalObject* globalObject = m_globalObject;
63             if (m_evalWasDisabled)
64                 globalObject->setEvalEnabled(false, globalObject->evalDisabledErrorMessage());
65 #if ASSERT_ENABLED
66             if (m_mode == Mode::EvalOnGlobalObjectAtDebuggerEntry)
67                 globalObject->setGlobalObjectAtDebuggerEntry(nullptr);
68 #endif
69         }
70     }
71 
72 private:
73     JSGlobalObject* m_globalObject;
74     bool m_evalWasDisabled { false };
75 #if ASSERT_ENABLED
76     DebuggerEvalEnabler::Mode m_mode;
77 #endif
78 };
79 
80 } // namespace JSC
