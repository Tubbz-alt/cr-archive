 33 SVGURIReference::SVGURIReference(SVGElement* contextElement)
 34     : m_href(SVGAnimatedString::create(contextElement))
 35 {
 36     static std::once_flag onceFlag;
 37     std::call_once(onceFlag, [] {
 38         PropertyRegistry::registerProperty<SVGNames::hrefAttr, &SVGURIReference::m_href>();
 39         PropertyRegistry::registerProperty<XLinkNames::hrefAttr, &SVGURIReference::m_href>();
 40     });
 41 }

 43 bool SVGURIReference::isKnownAttribute(const QualifiedName& attributeName)
 44 {
 45     return PropertyRegistry::isKnownAttribute(attributeName);
 46 }
 47 
 48 SVGElement& SVGURIReference::contextElement() const
 49 {
 50     return *m_href->contextElement();
 51 }
 52 
 53 void SVGURIReference::parseAttribute(const QualifiedName& name, const AtomString& value)
 54 {
 55     if (name.matches(SVGNames::hrefAttr))
 56         m_href->setBaseValInternal(value.isNull() ? contextElement().getAttribute(XLinkNames::hrefAttr) : value);
 57     else if (name.matches(XLinkNames::hrefAttr) && !contextElement().hasAttribute(SVGNames::hrefAttr))
 58         m_href->setBaseValInternal(value);
 59 }

 61 String SVGURIReference::fragmentIdentifierFromIRIString(const String& url, const Document& document)
 62 {
 63     size_t start = url.find('#');
 64     if (start == notFound)
 65         return emptyString();
 66 
 67     if (!start)
 68         return url.substring(1);
 69 
 70     URL base = URL(document.baseURL(), url.substring(0, start));
 71     String fragmentIdentifier = url.substring(start);
 72     URL kurl(base, fragmentIdentifier);
 73     if (equalIgnoringFragmentIdentifier(kurl, document.url()))
 74         return fragmentIdentifier.substring(1);
 75 
 76     // The url doesn't have any fragment identifier.
 77     return emptyString();
 78 }

 80 auto SVGURIReference::targetElementFromIRIString(const String& iri, const TreeScope& treeScope, RefPtr<Document> externalDocument) -> TargetElementResult
 81 {
 82     // If there's no fragment identifier contained within the IRI string, we can't lookup an element.
 83     size_t startOfFragmentIdentifier = iri.find('#');
 84     if (startOfFragmentIdentifier == notFound)
 85         return { };
 86 
 87     // Exclude the '#' character when determining the fragmentIdentifier.
 88     auto id = iri.substring(startOfFragmentIdentifier + 1);
 89     if (id.isEmpty())
 90         return { };
 91 
 92     auto& document = treeScope.documentScope();
 93     auto url = document.completeURL(iri);
 94     if (externalDocument) {
 95         // Enforce that the referenced url matches the url of the document that we've loaded for it!
 96         ASSERT(equalIgnoringFragmentIdentifier(url, externalDocument->url()));
 97         return { externalDocument->getElementById(id), WTFMove(id) };
 98     }
 99 
100     // Exit early if the referenced url is external, and we have no externalDocument given.
101     if (isExternalURIReference(iri, document))
102         return { nullptr, WTFMove(id) };
103 
104     return { treeScope.getElementById(id), WTFMove(id) };
105 }

107 bool SVGURIReference::haveLoadedRequiredResources() const
108 {
109     if (href().isEmpty() || !isExternalURIReference(href(), contextElement().document()))
110         return true;
111     return errorOccurred() || haveFiredLoadEvent();
112 }
113 
114 void SVGURIReference::dispatchLoadEvent()
115 {
116     if (haveFiredLoadEvent())
117         return;
118 
119     // Dispatch the load event
120     setHaveFiredLoadEvent(true);
121     ASSERT(contextElement().haveLoadedRequiredResources());
122 
123     contextElement().sendLoadEventIfPossible();
124 }
