26 #pragma once
27 
28 #include "DeviceOrientationOrMotionEvent.h"
29 #include "Event.h"
30 
31 namespace WebCore {
32 
33 class DeviceMotionData;
34 
35 class DeviceMotionEvent final : public Event, public DeviceOrientationOrMotionEvent {
36     WTF_MAKE_ISO_ALLOCATED(DeviceMotionEvent);
37 public:
38     virtual ~DeviceMotionEvent();
39 
40     // FIXME: Merge this with DeviceMotionData::Acceleration
41     struct Acceleration {
42         Optional<double> x;
43         Optional<double> y;
44         Optional<double> z;
45     };
46 
47     // FIXME: Merge this with DeviceMotionData::RotationRate
48     struct RotationRate {
49         Optional<double> alpha;
50         Optional<double> beta;
51         Optional<double> gamma;
52     };
53 
54     static Ref<DeviceMotionEvent> create(const AtomString& eventType, DeviceMotionData* deviceMotionData)
55     {
56         return adoptRef(*new DeviceMotionEvent(eventType, deviceMotionData));
57     }
58 
59     static Ref<DeviceMotionEvent> createForBindings()
60     {
61         return adoptRef(*new DeviceMotionEvent);
62     }
63 
64     Optional<Acceleration> acceleration() const;
65     Optional<Acceleration> accelerationIncludingGravity() const;
66     Optional<RotationRate> rotationRate() const;
67     Optional<double> interval() const;
68 
69     void initDeviceMotionEvent(const AtomString& type, bool bubbles, bool cancelable, Optional<Acceleration>&&, Optional<Acceleration>&&, Optional<RotationRate>&&, Optional<double>);
70 
71 private:
72     DeviceMotionEvent();
73     DeviceMotionEvent(const AtomString& eventType, DeviceMotionData*);
74 
75     EventInterface eventInterface() const override;
76 
77     RefPtr<DeviceMotionData> m_deviceMotionData;
78 };
79 
80 } // namespace WebCore
