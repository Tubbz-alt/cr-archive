 37     static Ref<SVGUseElement> create(const QualifiedName&, Document&);

 40     void invalidateShadowTree();

 42     void updateShadowTree();
 43 
 44     RenderElement* rendererClipChild() const;
 45 
 46     const SVGLengthValue& x() const { return m_x->currentValue(); }

 51     SVGAnimatedLength& xAnimated() { return m_x; }

 67     void parseAttribute(const QualifiedName&, const AtomString&) override;
 68     void svgAttributeChanged(const QualifiedName&) override;

 70     RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) override;

 75     Document* externalDocument() const;

 78     SVGElement* findTarget(String* targetID = nullptr) const;

 80     void cloneTarget(ContainerNode&, SVGElement& target) const;
 81     RefPtr<SVGElement> targetClone() const;

 83     void expandUseElementsInShadowTree() const;

 88     void clearShadowTree();

 91     bool haveLoadedRequiredResources() override { return SVGURIReference::haveLoadedRequiredResources(); }

 94     void setErrorOccurred(bool errorOccurred) override { m_errorOccurred = errorOccurred; }
 95     bool errorOccurred() const override { return m_errorOccurred; }
 96     Timer* loadEventTimer() override { return &m_loadEventTimer; }
 97 
 98     bool isValid() const override { return SVGTests::isValid(); }
 99 
100     PropertyRegistry m_propertyRegistry { *this };
