56 ExceptionOr<void> Crypto::getRandomValues(ArrayBufferView& array)
57 {
58     if (!isInt(array.getType()))
59         return Exception { TypeMismatchError };
60     if (array.byteLength() > 65536)
61         return Exception { QuotaExceededError };
62 #if OS(DARWIN) && ENABLE(SUBTLE_CRYPTO)
63     auto rc = CCRandomGenerateBytes(array.baseAddress(), array.byteLength());
64     RELEASE_ASSERT(rc == kCCSuccess);
65 #else
66     cryptographicallyRandomValues(array.baseAddress(), array.byteLength());
67 #endif
68     return { };
69 }
