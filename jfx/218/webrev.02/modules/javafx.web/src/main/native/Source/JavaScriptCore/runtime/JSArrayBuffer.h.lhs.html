28 #include "ArrayBuffer.h"

31 namespace JSC {
32 
33 class JSArrayBuffer final : public JSNonFinalObject {
34 public:
35     using Base = JSNonFinalObject;
36     static const unsigned StructureFlags = Base::StructureFlags;






37 
38 protected:
39     JSArrayBuffer(VM&, Structure*, RefPtr<ArrayBuffer>&&);
40     void finishCreation(VM&, JSGlobalObject*);
41 
42 public:
43     // This function will register the new wrapper with the vm's TypedArrayController.
44     JS_EXPORT_PRIVATE static JSArrayBuffer* create(VM&, Structure*, RefPtr<ArrayBuffer>&&);
45 
46     ArrayBuffer* impl() const { return m_impl; }
47 
48     static Structure* createStructure(VM&, JSGlobalObject*, JSValue prototype);
49 
50     JS_EXPORT_PRIVATE bool isShared() const;
51     ArrayBufferSharingMode sharingMode() const;
52 
53     DECLARE_EXPORT_INFO;
54 
55     // This is the default DOM unwrapping. It calls toUnsharedArrayBuffer().
56     static ArrayBuffer* toWrapped(VM&, JSValue);
57 
58 protected:
59     static size_t estimatedSize(JSCell*, VM&);
60 
61 private:
62     ArrayBuffer* m_impl;
63 };
64 
65 inline ArrayBuffer* toPossiblySharedArrayBuffer(VM& vm, JSValue value)
66 {
67     JSArrayBuffer* wrapper = jsDynamicCast<JSArrayBuffer*>(vm, value);
68     if (!wrapper)
69         return nullptr;
70     return wrapper->impl();
71 }
72 
73 inline ArrayBuffer* toUnsharedArrayBuffer(VM& vm, JSValue value)
74 {
75     ArrayBuffer* result = toPossiblySharedArrayBuffer(vm, value);
76     if (!result || result->isShared())
77         return nullptr;
78     return result;
79 }
80 
81 inline ArrayBuffer* JSArrayBuffer::toWrapped(VM& vm, JSValue value)
82 {
83     return toUnsharedArrayBuffer(vm, value);
84 }
