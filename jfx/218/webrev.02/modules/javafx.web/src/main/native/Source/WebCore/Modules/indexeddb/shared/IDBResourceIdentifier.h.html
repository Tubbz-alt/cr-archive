 45 using IDBConnectionIdentifier = ProcessIdentifier;
 46 
 47 class IDBResourceIdentifier {
 48     WTF_MAKE_FAST_ALLOCATED;
 49 public:
 50     explicit IDBResourceIdentifier(const IDBClient::IDBConnectionProxy&);
 51     IDBResourceIdentifier(const IDBClient::IDBConnectionProxy&, const IDBRequest&);
 52     explicit IDBResourceIdentifier(const IDBServer::IDBConnectionToClient&);
 53 
 54     static IDBResourceIdentifier deletedValue();
 55     WEBCORE_EXPORT bool isHashTableDeletedValue() const;
 56 
 57     static IDBResourceIdentifier emptyValue();
 58     bool isEmpty() const
 59     {
 60         return !m_resourceNumber && !m_idbConnectionIdentifier;
 61     }
 62 
 63     unsigned hash() const
 64     {
 65         uint64_t hashCodes[2] = { m_idbConnectionIdentifier.toUInt64(), m_resourceNumber };
 66         return StringHasher::hashMemory<sizeof(hashCodes)>(hashCodes);
 67     }
 68 
 69     bool operator==(const IDBResourceIdentifier& other) const
 70     {
 71         return m_idbConnectionIdentifier == other.m_idbConnectionIdentifier
 72             && m_resourceNumber == other.m_resourceNumber;
 73     }
 74 
 75     IDBConnectionIdentifier connectionIdentifier() const { return m_idbConnectionIdentifier; }
 76 
 77     WEBCORE_EXPORT IDBResourceIdentifier isolatedCopy() const;
 78 
 79 #if !LOG_DISABLED
 80     String loggingString() const;
 81 #endif
 82 
 83     WEBCORE_EXPORT IDBResourceIdentifier();
 84 
 85     template<class Encoder> void encode(Encoder&) const;
 86     template<class Decoder> static bool decode(Decoder&, IDBResourceIdentifier&);
 87 
 88 private:
 89     IDBResourceIdentifier(IDBConnectionIdentifier, uint64_t resourceIdentifier);
 90     IDBConnectionIdentifier m_idbConnectionIdentifier;
 91     uint64_t m_resourceNumber { 0 };
 92 };
 93 
 94 struct IDBResourceIdentifierHash {
 95     static unsigned hash(const IDBResourceIdentifier& a) { return a.hash(); }
 96     static bool equal(const IDBResourceIdentifier& a, const IDBResourceIdentifier& b) { return a == b; }
 97     static const bool safeToCompareToEmptyOrDeleted = false;
 98 };
 99 
100 struct IDBResourceIdentifierHashTraits : WTF::CustomHashTraits<IDBResourceIdentifier> {
101     static const bool hasIsEmptyValueFunction = true;
102     static const bool emptyValueIsZero = false;
103 
104     static IDBResourceIdentifier emptyValue()
105     {
106         return IDBResourceIdentifier::emptyValue();
107     }
108 
109     static bool isEmptyValue(const IDBResourceIdentifier& identifier)
110     {
111         return identifier.isEmpty();
112     }
113 
114     static void constructDeletedValue(IDBResourceIdentifier& identifier)
115     {
116         identifier = IDBResourceIdentifier::deletedValue();
117     }
118 
119     static bool isDeletedValue(const IDBResourceIdentifier& identifier)
120     {
121         return identifier.isHashTableDeletedValue();
122     }
123 };
124 
125 template<class Encoder>
126 void IDBResourceIdentifier::encode(Encoder& encoder) const
127 {
128     encoder << m_idbConnectionIdentifier << m_resourceNumber;
129 }
130 
131 template<class Decoder>
132 bool IDBResourceIdentifier::decode(Decoder& decoder, IDBResourceIdentifier& identifier)
133 {
134     if (!decoder.decode(identifier.m_idbConnectionIdentifier))
135         return false;
136 
137     if (!decoder.decode(identifier.m_resourceNumber))
138         return false;
139 
140     return true;
141 }
