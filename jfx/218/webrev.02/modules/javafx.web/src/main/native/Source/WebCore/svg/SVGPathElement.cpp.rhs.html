 38 inline SVGPathElement::SVGPathElement(const QualifiedName& tagName, Document& document)
 39     : SVGGeometryElement(tagName, document)

 40 {
 41     ASSERT(hasTagName(SVGNames::pathTag));
 42 
 43     static std::once_flag onceFlag;
 44     std::call_once(onceFlag, [] {
 45         PropertyRegistry::registerProperty<SVGNames::dAttr, &SVGPathElement::m_pathSegList>();
 46     });
 47 }

 49 Ref<SVGPathElement> SVGPathElement::create(const QualifiedName& tagName, Document& document)
 50 {
 51     return adoptRef(*new SVGPathElement(tagName, document));
 52 }

 54 void SVGPathElement::parseAttribute(const QualifiedName& name, const AtomString& value)
 55 {
 56     if (name == SVGNames::dAttr) {
 57         if (!m_pathSegList->baseVal()->parse(value))
 58             document().accessSVGExtensions().reportError("Problem parsing d=\"" + value + "\"");
 59         return;
 60     }
 61 
 62     SVGGeometryElement::parseAttribute(name, value);

 63 }
 64 
 65 void SVGPathElement::svgAttributeChanged(const QualifiedName& attrName)
 66 {
 67     if (attrName == SVGNames::dAttr) {
 68         InstanceInvalidationGuard guard(*this);
 69         invalidateMPathDependencies();
 70 
 71         if (auto* renderer = downcast<RenderSVGPath>(this->renderer())) {
 72             renderer->setNeedsShapeUpdate();
 73             RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
 74         }
 75 
 76         return;
 77     }
 78 
 79     SVGGeometryElement::svgAttributeChanged(attrName);

 80 }
 81 
 82 void SVGPathElement::invalidateMPathDependencies()
 83 {
 84     // <mpath> can only reference <path> but this dependency is not handled in
 85     // markForLayoutAndParentResourceInvalidation so we update any mpath dependencies manually.
 86     if (HashSet<SVGElement*>* dependencies = document().accessSVGExtensions().setOfElementsReferencingTarget(*this)) {
 87         for (auto* element : *dependencies) {
 88             if (is<SVGMPathElement>(*element))
 89                 downcast<SVGMPathElement>(*element).targetPathChanged();
 90         }
 91     }
 92 }

 94 Node::InsertedIntoAncestorResult SVGPathElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)

112 ExceptionOr<Ref<SVGPoint>> SVGPathElement::getPointAtLength(float distance) const
113 {
114     // Spec: Clamp distance to [0, length].
115     distance = clampTo<float>(distance, 0, getTotalLength());
116 
117     // Spec: Return a newly created, detached SVGPoint object.
118     return SVGPoint::create(getPointAtLengthOfSVGPathByteStream(pathByteStream(), distance));
119 }

124 }

126 FloatRect SVGPathElement::getBBox(StyleUpdateStrategy styleUpdateStrategy)
127 {
128     if (styleUpdateStrategy == AllowStyleUpdate)
129         document().updateLayoutIgnorePendingStylesheets();
130 
131     RenderSVGPath* renderer = downcast<RenderSVGPath>(this->renderer());
132 
133     // FIXME: Eventually we should support getBBox for detached elements.
134     // FIXME: If the path is null it means we're calling getBBox() before laying out this element,
135     // which is an error.
136     if (!renderer || !renderer->hasPath())
137         return { };
138 
139     return renderer->path().boundingRect();
140 }
