 35 DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(CStringBuffer);
 36 
 37 // CStringBuffer is the ref-counted storage class for the characters in a CString.
 38 // The data is implicitly allocated 1 character longer than length(), as it is zero-terminated.
 39 class CStringBuffer final : public RefCounted<CStringBuffer> {
 40     WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(CStringBuffer);
 41 public:
 42     const char* data() { return mutableData(); }
 43     size_t length() const { return m_length; }
 44 
 45 private:
 46     friend class CString;
 47 
 48     static Ref<CStringBuffer> createUninitialized(size_t length);
 49 
 50     CStringBuffer(size_t length) : m_length(length) { }
 51     char* mutableData() { return reinterpret_cast_ptr<char*>(this + 1); }
 52 
 53     const size_t m_length;
 54 };
 55 
 56 // A container for a null-terminated char array supporting copy-on-write
 57 // assignment.  The contained char array may be null.
 58 class CString final {
 59     WTF_MAKE_FAST_ALLOCATED;
 60 public:
 61     CString() { }
 62     WTF_EXPORT_PRIVATE CString(const char*);
 63     WTF_EXPORT_PRIVATE CString(const char*, size_t length);
 64     CString(CStringBuffer* buffer) : m_buffer(buffer) { }
 65     WTF_EXPORT_PRIVATE static CString newUninitialized(size_t length, char*& characterBuffer);
 66     CString(HashTableDeletedValueType) : m_buffer(HashTableDeletedValue) { }
 67 
 68     const char* data() const
 69     {
 70         return m_buffer ? m_buffer->data() : 0;
 71     }
 72     WTF_EXPORT_PRIVATE char* mutableData();
 73     size_t length() const
 74     {
 75         return m_buffer ? m_buffer->length() : 0;
 76     }
 77 
 78     bool isNull() const { return !m_buffer; }
 79     bool isSafeToSendToAnotherThread() const;
 80 
 81     CStringBuffer* buffer() const { return m_buffer.get(); }
 82 
 83     bool isHashTableDeletedValue() const { return m_buffer.isHashTableDeletedValue(); }
 84 
 85     WTF_EXPORT_PRIVATE unsigned hash() const;
 86 
 87 private:
 88     void copyBufferIfNeeded();
 89     void init(const char*, size_t length);
 90     RefPtr<CStringBuffer> m_buffer;
 91 };
 92 
 93 WTF_EXPORT_PRIVATE bool operator==(const CString& a, const CString& b);
 94 inline bool operator!=(const CString& a, const CString& b) { return !(a == b); }
 95 WTF_EXPORT_PRIVATE bool operator==(const CString& a, const char* b);
 96 inline bool operator!=(const CString& a, const char* b) { return !(a == b); }
 97 WTF_EXPORT_PRIVATE bool operator<(const CString& a, const CString& b);
 98 
 99 struct CStringHash {
100     static unsigned hash(const CString& string) { return string.hash(); }
101     WTF_EXPORT_PRIVATE static bool equal(const CString& a, const CString& b);
102     static constexpr bool safeToCompareToEmptyOrDeleted = true;
103 };
104 
105 template<typename T> struct DefaultHash;
106 template<> struct DefaultHash<CString> {
107     typedef CStringHash Hash;
108 };
109 
110 template<typename T> struct HashTraits;
111 template<> struct HashTraits<CString> : SimpleClassHashTraits<CString> { };
