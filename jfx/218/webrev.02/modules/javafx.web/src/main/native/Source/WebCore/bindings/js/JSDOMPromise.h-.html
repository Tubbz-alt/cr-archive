27 #pragma once
28 
29 #include "JSDOMGuardedObject.h"
30 #include <JavaScriptCore/JSPromise.h>
31 
32 namespace WebCore {
33 
34 class DOMPromise : public DOMGuarded<JSC::JSPromise> {
35 public:
36     static Ref<DOMPromise> create(JSDOMGlobalObject& globalObject, JSC::JSPromise& promise)
37     {
38         return adoptRef(*new DOMPromise(globalObject, promise));
39     }
40 
41     JSC::JSPromise* promise() const
42     {
43         ASSERT(!isSuspended());
44         return guarded();
45     }
46 
47     void whenSettled(std::function<void()>&&);
48     JSC::JSValue result() const;
49 
50     enum class Status { Pending, Fulfilled, Rejected };
51     Status status() const;
52 
53     static void whenPromiseIsSettled(JSDOMGlobalObject*, JSC::JSObject* promise, std::function<void()>&&);
54 
55 private:
56     DOMPromise(JSDOMGlobalObject& globalObject, JSC::JSPromise& promise)
57         : DOMGuarded<JSC::JSPromise>(globalObject, promise)
58     {
59     }
60 };
61 
62 } // namespace WebCore
