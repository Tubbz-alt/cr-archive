 39 Ref<FetchEvent> FetchEvent::createForTesting(ScriptExecutionContext& context)
 40 {
 41     FetchEvent::Init init;
 42     init.request = FetchRequest::create(context, { }, FetchHeaders::create(FetchHeaders::Guard::Immutable, { }), { }, { }, { });
 43     return FetchEvent::create("fetch", WTFMove(init), Event::IsTrusted::Yes);
 44 }

 46 FetchEvent::FetchEvent(const AtomString& type, Init&& initializer, IsTrusted isTrusted)
 47     : ExtendableEvent(type, initializer, isTrusted)
 48     , m_request(initializer.request.releaseNonNull())
 49     , m_clientId(WTFMove(initializer.clientId))
 50     , m_reservedClientId(WTFMove(initializer.reservedClientId))
 51     , m_targetClientId(WTFMove(initializer.targetClientId))
 52 {
 53 }
 54 
 55 FetchEvent::~FetchEvent()
 56 {
 57     if (auto callback = WTFMove(m_onResponse))
 58         callback(makeUnexpected(ResourceError { errorDomainWebKitServiceWorker, 0, m_request->url(), "Fetch event is destroyed."_s, ResourceError::Type::Cancellation }));
 59 }

 61 ResourceError FetchEvent::createResponseError(const URL& url, const String& errorMessage)
 62 {
 63     return ResourceError { errorDomainWebKitServiceWorker, 0, url, makeString("FetchEvent.respondWith received an error: ", errorMessage), ResourceError::Type::General };
 64 
 65 }

 67 ExceptionOr<void> FetchEvent::respondWith(Ref<DOMPromise>&& promise)
 68 {
 69     if (!isBeingDispatched())
 70         return Exception { InvalidStateError, "Event is not being dispatched"_s };
 71 
 72     if (m_respondWithEntered)
 73         return Exception { InvalidStateError, "Event respondWith flag is set"_s };
 74 
 75     m_respondPromise = WTFMove(promise);
 76     addExtendLifetimePromise(*m_respondPromise);
 77 
 78     m_respondPromise->whenSettled([this, weakThis = makeWeakPtr(*this)] () {
 79         if (!weakThis)
 80             return;
 81         promiseIsSettled();
 82     });
 83 
 84     stopPropagation();
 85     stopImmediatePropagation();
 86 
 87     m_respondWithEntered = true;
 88     m_waitToRespond = true;
 89 
 90     return { };
 91 }
 92 
 93 void FetchEvent::onResponse(ResponseCallback&& callback)
 94 {
 95     ASSERT(!m_onResponse);
 96     m_onResponse = WTFMove(callback);
 97 }
 98 
 99 void FetchEvent::respondWithError(ResourceError&& error)
100 {
101     m_respondWithError = true;
102     processResponse(makeUnexpected(WTFMove(error)));
103 }
104 
105 void FetchEvent::processResponse(Expected<Ref<FetchResponse>, ResourceError>&& result)
106 {
107     m_respondPromise = nullptr;
108     m_waitToRespond = false;
109     if (auto callback = WTFMove(m_onResponse))
110         callback(WTFMove(result));
111 }
112 
113 void FetchEvent::promiseIsSettled()
114 {
115     if (m_respondPromise->status() == DOMPromise::Status::Rejected) {
116         auto reason = m_respondPromise->result().toWTFString(m_respondPromise->globalObject());
117         respondWithError(createResponseError(m_request->url(), reason));
118         return;
119     }
120 
121     ASSERT(m_respondPromise->status() == DOMPromise::Status::Fulfilled);
122     auto response = JSFetchResponse::toWrapped(m_respondPromise->globalObject()->vm(), m_respondPromise->result());
123     if (!response) {
124         respondWithError(createResponseError(m_request->url(), "Returned response is null."_s));
125         return;
126     }
127 
128     if (response->isDisturbedOrLocked()) {
129         respondWithError(createResponseError(m_request->url(), "Response is disturbed or locked."_s));
130         return;
131     }
132 
133     processResponse(makeRef(*response));
134 }
