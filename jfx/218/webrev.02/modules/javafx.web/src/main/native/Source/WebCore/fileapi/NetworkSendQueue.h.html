45 class WEBCORE_EXPORT NetworkSendQueue {
46 public:
47     using WriteString = Function<void(const String&)>;
48     using WriteRawData = Function<void(const char*, size_t)>;
49     enum class Continue { No, Yes };
50     using ProcessError = Function<Continue(int)>;
51     NetworkSendQueue(Document&, WriteString&&, WriteRawData&&, ProcessError&&);
52     ~NetworkSendQueue();
53 
54     void enqueue(const String&);
55     void enqueue(const JSC::ArrayBuffer&, unsigned byteOffset, unsigned byteLength);
56     void enqueue(Blob&);
57 
58     void clear();
59 
60 private:
61     void processMessages();
62 
63     using Message = Variant<String, Ref<SharedBuffer>, UniqueRef<BlobLoader>>;
64     Deque<Message> m_queue;
65 
66     WTF::WeakPtr<Document> m_document;
67 
68     WriteString m_writeString;
69     WriteRawData m_writeRawData;
70     ProcessError m_processError;
71 };
