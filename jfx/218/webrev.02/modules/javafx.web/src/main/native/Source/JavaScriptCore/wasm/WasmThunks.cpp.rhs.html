 43 MacroAssemblerCodeRef<JITThunkPtrTag> throwExceptionFromWasmThunkGenerator(const AbstractLocker&)
 44 {
 45     CCallHelpers jit;
 46 
 47     // The thing that jumps here must move ExceptionType into the argumentGPR1 before jumping here.
 48     // We're allowed to use temp registers here. We are not allowed to use callee saves.
 49     jit.loadWasmContextInstance(GPRInfo::argumentGPR2);
 50     jit.loadPtr(CCallHelpers::Address(GPRInfo::argumentGPR2, Instance::offsetOfPointerToTopEntryFrame()), GPRInfo::argumentGPR0);
 51     jit.loadPtr(CCallHelpers::Address(GPRInfo::argumentGPR0), GPRInfo::argumentGPR0);
 52     jit.copyCalleeSavesToEntryFrameCalleeSavesBuffer(GPRInfo::argumentGPR0);
 53     jit.move(GPRInfo::callFrameRegister, GPRInfo::argumentGPR0);
 54 
 55     CCallHelpers::Call call = jit.call(OperationPtrTag);
 56     jit.farJump(GPRInfo::returnValueGPR, ExceptionHandlerPtrTag);
 57     jit.breakpoint(); // We should not reach this.
 58 


 59     LinkBuffer linkBuffer(jit, GLOBAL_THUNK_ID);
 60     linkBuffer.link(call, FunctionPtr<OperationPtrTag>(operationWasmToJSException));
 61     return FINALIZE_WASM_CODE(linkBuffer, JITThunkPtrTag, "Throw exception from Wasm");
 62 }

 64 MacroAssemblerCodeRef<JITThunkPtrTag> throwStackOverflowFromWasmThunkGenerator(const AbstractLocker& locker)
 65 {
 66     CCallHelpers jit;
 67 
 68     int32_t stackSpace = WTF::roundUpToMultipleOf(stackAlignmentBytes(), RegisterSet::calleeSaveRegisters().numberOfSetRegisters() * sizeof(Register));
 69     ASSERT(static_cast<unsigned>(stackSpace) < Options::softReservedZoneSize());
 70     jit.addPtr(CCallHelpers::TrustedImm32(-stackSpace), GPRInfo::callFrameRegister, MacroAssembler::stackPointerRegister);
 71     jit.move(CCallHelpers::TrustedImm32(static_cast<uint32_t>(ExceptionType::StackOverflow)), GPRInfo::argumentGPR1);
 72     auto jumpToExceptionHandler = jit.jump();
 73     LinkBuffer linkBuffer(jit, GLOBAL_THUNK_ID);
 74     linkBuffer.link(jumpToExceptionHandler, CodeLocationLabel<JITThunkPtrTag>(Thunks::singleton().stub(locker, throwExceptionFromWasmThunkGenerator).code()));
 75     return FINALIZE_WASM_CODE(linkBuffer, JITThunkPtrTag, "Throw stack overflow from Wasm");
 76 }

 78 MacroAssemblerCodeRef<JITThunkPtrTag> triggerOMGEntryTierUpThunkGenerator(const AbstractLocker&)
 79 {
 80     // We expect that the user has already put the function index into GPRInfo::argumentGPR1
 81     CCallHelpers jit;
 82 
 83     jit.emitFunctionPrologue();
 84 
 85     const unsigned extraPaddingBytes = 0;
 86     RegisterSet registersToSpill = RegisterSet::allRegisters();
 87     registersToSpill.exclude(RegisterSet::registersToNotSaveForCCall());
 88     unsigned numberOfStackBytesUsedForRegisterPreservation = ScratchRegisterAllocator::preserveRegistersToStackForCall(jit, registersToSpill, extraPaddingBytes);
 89 
 90     jit.loadWasmContextInstance(GPRInfo::argumentGPR0);
 91     jit.move(MacroAssembler::TrustedImmPtr(tagCFunctionPtr<OperationPtrTag>(operationWasmTriggerTierUpNow)), GPRInfo::argumentGPR2);
 92     jit.call(GPRInfo::argumentGPR2, OperationPtrTag);
 93 
 94     ScratchRegisterAllocator::restoreRegistersFromStackForCall(jit, registersToSpill, RegisterSet(), numberOfStackBytesUsedForRegisterPreservation, extraPaddingBytes);
 95 
 96     jit.emitFunctionEpilogue();
 97     jit.ret();
 98     LinkBuffer linkBuffer(jit, GLOBAL_THUNK_ID);
 99     return FINALIZE_WASM_CODE(linkBuffer, JITThunkPtrTag, "Trigger OMG entry tier up");
100 }

114 MacroAssemblerCodeRef<JITThunkPtrTag> Thunks::stub(ThunkGenerator generator)
115 {
116     auto locker = holdLock(m_lock);
117     return stub(locker, generator);
118 }
119 
120 MacroAssemblerCodeRef<JITThunkPtrTag> Thunks::stub(const AbstractLocker& locker, ThunkGenerator generator)
121 {
122     ASSERT(!!generator);
123     {
124         auto addResult = m_stubs.add(generator, MacroAssemblerCodeRef<JITThunkPtrTag>());
125         if (!addResult.isNewEntry)
126             return addResult.iterator->value;
127     }
128 
129     MacroAssemblerCodeRef<JITThunkPtrTag> code = generator(locker);
130     // We specifically don't use the iterator here to allow generator to recursively change m_stubs.
131     m_stubs.set(generator, code);
132     return code;
133 }
134 
135 MacroAssemblerCodeRef<JITThunkPtrTag> Thunks::existingStub(ThunkGenerator generator)
136 {
137     auto locker = holdLock(m_lock);
138 
139     auto iter = m_stubs.find(generator);
140     if (iter != m_stubs.end())
141         return iter->value;
142 
143     return MacroAssemblerCodeRef<JITThunkPtrTag>();
144 }
