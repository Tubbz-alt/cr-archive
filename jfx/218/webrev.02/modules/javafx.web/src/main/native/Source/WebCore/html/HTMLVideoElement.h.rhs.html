 46     WEBCORE_EXPORT unsigned videoWidth() const;

 49     WEBCORE_EXPORT ExceptionOr<void> webkitEnterFullscreen();

 54     void ancestorWillEnterFullscreen() final;

 97 #endif

105 #endif

112     void scheduleResizeEvent() final;
113     void scheduleResizeEventIfSizeChanged() final;
114     bool rendererIsNeeded(const RenderStyle&) final;
115     void didAttachRenderers() final;
