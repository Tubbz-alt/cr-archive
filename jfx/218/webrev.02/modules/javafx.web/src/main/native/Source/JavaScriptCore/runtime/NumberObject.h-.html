21 #pragma once
22 
23 #include "JSWrapperObject.h"
24 
25 namespace JSC {
26 
27 class NumberObject : public JSWrapperObject {
28 protected:
29     NumberObject(VM&, Structure*);
30     void finishCreation(VM&);
31 
32 public:
33     using Base = JSWrapperObject;
34 
35     static NumberObject* create(VM& vm, Structure* structure)
36     {
37         NumberObject* number = new (NotNull, allocateCell<NumberObject>(vm.heap)) NumberObject(vm, structure);
38         number->finishCreation(vm);
39         return number;
40     }
41 
42     DECLARE_EXPORT_INFO;
43 
44     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
45     {
46         return Structure::create(vm, globalObject, prototype, TypeInfo(NumberObjectType, StructureFlags), info());
47     }
48 };
49 
50 JS_EXPORT_PRIVATE NumberObject* constructNumber(ExecState*, JSGlobalObject*, JSValue);
51 
52 } // namespace JSC
