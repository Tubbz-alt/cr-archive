21 [
22     CustomConstructor(),
23     CustomPushEventHandlerScope,
24     ExportMacro=WEBCORE_EXPORT,
25     JSGenerateToNativeObject,
26 ] interface HTMLElement : Element {
27     // Metadata attributes.
28     [CEReactions, Reflect] attribute DOMString title;
29     [CEReactions, Reflect] attribute DOMString lang;
30     [CEReactions] attribute boolean translate;
31     [CEReactions] attribute DOMString dir;
32 
33     // User interaction.
34     [CEReactions, Reflect] attribute boolean hidden;
35     void click();
36     [CEReactions, Reflect] attribute DOMString accessKey;
37     // readonly attribute DOMString accessKeyLabel; // FIXME: Not supported.
38     [CEReactions] attribute boolean draggable;
39     // [CEReactions] attribute HTMLMenuElement? contextMenu; // FIXME: Not supported.
40     [CEReactions] attribute boolean spellcheck;
41     // void forceSpellCheck(); // FIXME: Not supported.
42 
43     [CEReactions] attribute [TreatNullAs=EmptyString] DOMString innerText;
44 
45     // FIXME: Should be in a separate ElementContentEditable interface.
46     // https://html.spec.whatwg.org/#elementcontenteditable
47     [CEReactions] attribute DOMString contentEditable;
48     readonly attribute boolean isContentEditable;
49 
50     // Extensions from CSSOM-view specification (https://drafts.csswg.org/cssom-view/#extensions-to-the-htmlelement-interface).
51     [ImplementedAs=offsetParentForBindings] readonly attribute Element? offsetParent;
52     [ImplementedAs=offsetTopForBindings] readonly attribute double offsetTop; // FIXME: Should be of type long.
53     [ImplementedAs=offsetLeftForBindings] readonly attribute double offsetLeft; // FIXME: Should be of type long.
54     readonly attribute double offsetWidth; // FIXME: Should be of type long.
55     readonly attribute double offsetHeight; // FIXME: Should be of type long.
56 
57     // Non Standard but may get added to the specification (https://github.com/whatwg/html/issues/668).
58     [CEReactions] attribute [TreatNullAs=EmptyString] DOMString outerText;
59 
60     // iOS autocorrect / autocapitalization extensions.
61     [Conditional=AUTOCORRECT, CEReactions] attribute boolean autocorrect;
62     [Conditional=AUTOCAPITALIZE, CEReactions] attribute [TreatNullAs=EmptyString] DOMString autocapitalize;
63 
64     // FIXME: We are the only browser to support this now that Blink dropped it (http://crbug.com/688943).
65     [CEReactions, Reflect] attribute DOMString webkitdropzone;
66 
67     attribute DOMString inputMode;
68     [EnabledBySetting=EnterKeyHint] attribute DOMString enterKeyHint;
69 };
70 
71 HTMLElement implements ElementCSSInlineStyle;
72 HTMLElement implements GlobalEventHandlers;
73 HTMLElement implements HTMLOrForeignElement;
