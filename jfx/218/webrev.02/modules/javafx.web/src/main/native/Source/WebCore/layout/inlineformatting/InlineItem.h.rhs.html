30 #include "LayoutBox.h"
31 #include "LayoutUnits.h"
32 
33 namespace WebCore {
34 namespace Layout {
35 
36 class InlineItem {

37 public:
38     enum class Type : uint8_t { Text, HardLineBreak, SoftLineBreak, Box, Float, ContainerStart, ContainerEnd };
39     InlineItem(const Box& layoutBox, Type);
40 
41     Type type() const { return m_type; }
42     const Box& layoutBox() const { return *m_layoutBox; }
43     const RenderStyle& style() const { return layoutBox().style(); }
44 
45     bool isText() const { return type() == Type::Text; }
46     bool isBox() const { return type() == Type::Box; }

47     bool isFloat() const { return type() == Type::Float; }
48     bool isLineBreak() const { return isSoftLineBreak() || isHardLineBreak(); }
49     bool isSoftLineBreak() const { return type() == Type::SoftLineBreak; }
50     bool isHardLineBreak() const { return type() == Type::HardLineBreak; }
51     bool isContainerStart() const { return type() == Type::ContainerStart; }
52     bool isContainerEnd() const { return type() == Type::ContainerEnd; }
53 
54 private:
55     const Box* m_layoutBox { nullptr };
56     Type m_type { };
57 
58 protected:
59     // For InlineTextItem
60     enum class TextItemType  : uint8_t { Undefined, Whitespace, NonWhitespace };
61     TextItemType m_textItemType { TextItemType::Undefined };
62     bool m_hasWidth { false };
63     bool m_isCollapsible { false };
64     InlineLayoutUnit m_width { };
65     unsigned m_length { 0 };
66 
67     // For InlineTextItem and InlineSoftLineBreakItem
68     unsigned m_startOrPosition { 0 };
69 };
70 
71 inline InlineItem::InlineItem(const Box& layoutBox, Type type)
72     : m_layoutBox(&layoutBox)
73     , m_type(type)
74 {
75 }
76

79     static bool isType(const WebCore::Layout::InlineItem& inlineItem) { return inlineItem.predicate; } \
80 SPECIALIZE_TYPE_TRAITS_END()
81 
82 }
83 }
