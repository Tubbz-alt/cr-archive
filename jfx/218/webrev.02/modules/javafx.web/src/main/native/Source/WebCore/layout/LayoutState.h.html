 50 class LayoutState : public CanMakeWeakPtr<LayoutState> {
 51     WTF_MAKE_ISO_ALLOCATED(LayoutState);
 52 public:
 53     LayoutState(const Document&, const Container& rootContainer);
 54     ~LayoutState();
 55 
 56     FormattingState& ensureFormattingState(const Container& formattingContextRoot);
 57     InlineFormattingState& ensureInlineFormattingState(const Container& formattingContextRoot);
 58     BlockFormattingState& ensureBlockFormattingState(const Container& formattingContextRoot);
 59     TableFormattingState& ensureTableFormattingState(const Container& formattingContextRoot);
 60 
 61     FormattingState& establishedFormattingState(const Container& formattingRoot) const;
 62     InlineFormattingState& establishedInlineFormattingState(const Container& formattingContextRoot) const;
 63     BlockFormattingState& establishedBlockFormattingState(const Container& formattingContextRoot) const;
 64     TableFormattingState& establishedTableFormattingState(const Container& formattingContextRoot) const;
 65 
 66     FormattingState& formattingStateForBox(const Box&) const;
 67     bool hasInlineFormattingState(const Container& formattingRoot) const { return m_inlineFormattingStates.contains(&formattingRoot); }
 68 
 69 #ifndef NDEBUG
 70     void registerFormattingContext(const FormattingContext&);
 71     void deregisterFormattingContext(const FormattingContext& formattingContext) { m_formattingContextList.remove(&formattingContext); }
 72 #endif
 73 
 74     Display::Box& displayBoxForRootLayoutBox();
 75     Display::Box& ensureDisplayBoxForLayoutBox(const Box&);
 76     const Display::Box& displayBoxForLayoutBox(const Box&) const;
 77 
 78     bool hasDisplayBox(const Box&) const;
 79 
 80     enum class QuirksMode { No, Limited, Yes };
 81     bool inQuirksMode() const { return m_quirksMode == QuirksMode::Yes; }
 82     bool inLimitedQuirksMode() const { return m_quirksMode == QuirksMode::Limited; }
 83     bool inNoQuirksMode() const { return m_quirksMode == QuirksMode::No; }
 84 
 85     const Container& root() const { return *m_rootContainer; }
 86 
 87     // LFC integration only. Full LFC has proper ICB access.
 88     void setViewportSize(const LayoutSize&);
 89     LayoutSize viewportSize() const;
 90     bool isIntegratedRootBoxFirstChild() const { return m_isIntegratedRootBoxFirstChild; }
 91     void setIsIntegratedRootBoxFirstChild(bool);
 92 
 93 private:
 94     void setQuirksMode(QuirksMode quirksMode) { m_quirksMode = quirksMode; }
 95     Display::Box& ensureDisplayBoxForLayoutBoxSlow(const Box&);
 96 
 97     HashMap<const Container*, std::unique_ptr<InlineFormattingState>> m_inlineFormattingStates;
 98     HashMap<const Container*, std::unique_ptr<BlockFormattingState>> m_blockFormattingStates;
 99     HashMap<const Container*, std::unique_ptr<TableFormattingState>> m_tableFormattingStates;
100 
101     std::unique_ptr<InlineFormattingState> m_rootInlineFormattingStateForIntegration;
102 
103 #ifndef NDEBUG
104     HashSet<const FormattingContext*> m_formattingContextList;
105 #endif
106     HashMap<const Box*, std::unique_ptr<Display::Box>> m_layoutToDisplayBox;
107     QuirksMode m_quirksMode { QuirksMode::No };
108 
109     WeakPtr<const Container> m_rootContainer;
110 
111     // LFC integration only.
112     LayoutSize m_viewportSize;
113     bool m_isIntegratedRootBoxFirstChild { false };
114 };
115 
116 inline bool LayoutState::hasDisplayBox(const Box& layoutBox) const
117 {
118     if (layoutBox.cachedDisplayBoxForLayoutState(*this))
119         return true;
120     return m_layoutToDisplayBox.contains(&layoutBox);
121 }
122 
123 inline Display::Box& LayoutState::ensureDisplayBoxForLayoutBox(const Box& layoutBox)
124 {
125     if (auto* displayBox = layoutBox.cachedDisplayBoxForLayoutState(*this))
126         return *displayBox;
127     return ensureDisplayBoxForLayoutBoxSlow(layoutBox);
128 }
129 
130 inline const Display::Box& LayoutState::displayBoxForLayoutBox(const Box& layoutBox) const
131 {
132     if (auto* displayBox = layoutBox.cachedDisplayBoxForLayoutState(*this))
133         return *displayBox;
134     ASSERT(m_layoutToDisplayBox.contains(&layoutBox));
135     return *m_layoutToDisplayBox.get(&layoutBox);
136 }
137 
138 #ifndef NDEBUG
139 inline void LayoutState::registerFormattingContext(const FormattingContext& formattingContext)
140 {
141     // Multiple formatting contexts of the same root within a layout frame indicates defective layout logic.
142     ASSERT(!m_formattingContextList.contains(&formattingContext));
143     m_formattingContextList.add(&formattingContext);
144 }
145 #endif
146 
147 // These Layout::Box function are here to allow inlining.
148 inline bool Box::canCacheForLayoutState(const LayoutState& layoutState) const
149 {
150     return !m_cachedLayoutState || m_cachedLayoutState.get() == &layoutState;
151 }
152 
153 inline Display::Box* Box::cachedDisplayBoxForLayoutState(const LayoutState& layoutState) const
154 {
155     if (m_cachedLayoutState.get() != &layoutState)
156         return nullptr;
157     return m_cachedDisplayBoxForLayoutState.get();
158 }
