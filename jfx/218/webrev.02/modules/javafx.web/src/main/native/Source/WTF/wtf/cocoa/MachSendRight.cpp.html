 43 static void retainSendRight(mach_port_t port)
 44 {
 45     if (port == MACH_PORT_NULL)
 46         return;
 47 
 48     auto kr = KERN_SUCCESS;
 49     if (port != MACH_PORT_DEAD)
 50         kr = mach_port_mod_refs(mach_task_self(), port, MACH_PORT_RIGHT_SEND, 1);
 51 
 52     if (kr == KERN_INVALID_RIGHT || port == MACH_PORT_DEAD)
 53         kr = mach_port_mod_refs(mach_task_self(), port, MACH_PORT_RIGHT_DEAD_NAME, 1);
 54 
 55     if (kr != KERN_SUCCESS) {
 56         LOG_ERROR("mach_port_mod_refs error for port %d: %s (%x)", port, mach_error_string(kr), kr);
 57         if (kr == KERN_INVALID_RIGHT)
 58             CRASH();
 59     }
 60 }
 61 
 62 static void releaseSendRight(mach_port_t port)
 63 {
 64     if (port == MACH_PORT_NULL)
 65         return;
 66 
 67     deallocateSendRightSafely(port);
 68 }
 69 
 70 void deallocateSendRightSafely(mach_port_t port)
 71 {
 72     auto kr = mach_port_deallocate(mach_task_self(), port);
 73     if (kr == KERN_SUCCESS)
 74         return;
 75 
 76     RELEASE_LOG_ERROR(Process, "mach_port_deallocate error for port %d: %{private}s (%#x)", port, mach_error_string(kr), kr);
 77     if (kr == KERN_INVALID_RIGHT || kr == KERN_INVALID_NAME)
 78         CRASH();
 79 }
 80 
 81 MachSendRight MachSendRight::adopt(mach_port_t port)
 82 {
 83     return MachSendRight(port);
 84 }
 85 
 86 MachSendRight MachSendRight::create(mach_port_t port)
 87 {
 88     retainSendRight(port);
 89 
 90     return adopt(port);
 91 }
 92 
 93 MachSendRight::MachSendRight(mach_port_t port)
 94     : m_port(port)
 95 {
 96 }
 97 
 98 MachSendRight::MachSendRight(MachSendRight&& other)
 99     : m_port(other.leakSendRight())
100 {
101 }
102 
103 MachSendRight::~MachSendRight()
104 {
105     releaseSendRight(m_port);
106 }
107 
108 MachSendRight& MachSendRight::operator=(MachSendRight&& other)
109 {
110     if (this != &other) {
111         releaseSendRight(m_port);
112         m_port = other.leakSendRight();
113     }
114 
115     return *this;
116 }
117 
118 MachSendRight MachSendRight::copySendRight() const
119 {
120     return create(m_port);
121 }
122 
123 mach_port_t MachSendRight::leakSendRight()
124 {
125     return std::exchange(m_port, MACH_PORT_NULL);
126 }
