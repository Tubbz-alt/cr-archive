 45 static ExceptionOr<Vector<String>> convertAndValidate(Document& document, unsigned version, const Vector<String>& supportedNetworks, const PaymentCoordinator& paymentCoordinator)
 46 {



 47     Vector<String> result;
 48     result.reserveInitialCapacity(supportedNetworks.size());
 49     for (auto& supportedNetwork : supportedNetworks) {
 50         auto validatedNetwork = paymentCoordinator.validatedPaymentNetwork(document, version, supportedNetwork);
 51         if (!validatedNetwork)
 52             return Exception { TypeError, makeString("\"", supportedNetwork, "\" is not a valid payment network.") };
 53         result.uncheckedAppend(*validatedNetwork);
 54     }
 55 
 56     return WTFMove(result);
 57 }
 58 
 59 ExceptionOr<ApplePaySessionPaymentRequest> convertAndValidate(Document& document, unsigned version, ApplePayRequestBase& request, const PaymentCoordinator& paymentCoordinator)
 60 {
 61     if (!version || !paymentCoordinator.supportsVersion(document, version))
 62         return Exception { InvalidAccessError, makeString('"', version, "\" is not a supported version.") };
 63 
 64     ApplePaySessionPaymentRequest result;
 65     result.setVersion(version);
 66     result.setCountryCode(request.countryCode);
 67 
 68     auto merchantCapabilities = convertAndValidate(request.merchantCapabilities);
 69     if (merchantCapabilities.hasException())
 70         return merchantCapabilities.releaseException();
 71     result.setMerchantCapabilities(merchantCapabilities.releaseReturnValue());
 72 
 73     if (requiresSupportedNetworks(version, request) && request.supportedNetworks.isEmpty())
 74         return Exception { TypeError, "At least one supported network must be provided." };
 75 
 76     auto supportedNetworks = convertAndValidate(document, version, request.supportedNetworks, paymentCoordinator);
 77     if (supportedNetworks.hasException())
 78         return supportedNetworks.releaseException();
 79     result.setSupportedNetworks(supportedNetworks.releaseReturnValue());
 80 
 81     if (request.requiredBillingContactFields) {
 82         auto requiredBillingContactFields = convertAndValidate(version, *request.requiredBillingContactFields);
 83         if (requiredBillingContactFields.hasException())
 84             return requiredBillingContactFields.releaseException();
 85         result.setRequiredBillingContactFields(requiredBillingContactFields.releaseReturnValue());
 86     }
 87 
 88     if (request.billingContact)
 89         result.setBillingContact(PaymentContact::fromApplePayPaymentContact(version, *request.billingContact));
 90 
 91     if (request.requiredShippingContactFields) {
 92         auto requiredShippingContactFields = convertAndValidate(version, *request.requiredShippingContactFields);
 93         if (requiredShippingContactFields.hasException())
 94             return requiredShippingContactFields.releaseException();
 95         result.setRequiredShippingContactFields(requiredShippingContactFields.releaseReturnValue());
 96     }
 97 
 98     if (request.shippingContact)
 99         result.setShippingContact(PaymentContact::fromApplePayPaymentContact(version, *request.shippingContact));
100 
101     result.setApplicationData(request.applicationData);
102 
103     if (version >= 3)
104         result.setSupportedCountries(WTFMove(request.supportedCountries));
105 
106     finishConverting(result, request);
107 
108     return WTFMove(result);
109 }
