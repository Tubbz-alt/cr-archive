 37 namespace WebCore {
 38 
 39 static uint64_t nextClientResourceNumber()
 40 {
 41     static std::atomic<uint64_t> currentNumber(1);
 42     return currentNumber += 2;
 43 }
 44 
 45 static uint64_t nextServerResourceNumber()
 46 {
 47     ASSERT(isMainThread());
 48     static uint64_t currentNumber = 0;
 49     return currentNumber += 2;
 50 }
 51 
 52 IDBResourceIdentifier::IDBResourceIdentifier()
 53 {
 54 }
 55 
 56 IDBResourceIdentifier::IDBResourceIdentifier(uint64_t connectionIdentifier, uint64_t resourceIdentifier)
 57     : m_idbConnectionIdentifier(connectionIdentifier)
 58     , m_resourceNumber(resourceIdentifier)
 59 {
 60 }
 61 
 62 IDBResourceIdentifier::IDBResourceIdentifier(const IDBClient::IDBConnectionProxy& connectionProxy)
 63     : m_idbConnectionIdentifier(connectionProxy.serverConnectionIdentifier())
 64     , m_resourceNumber(nextClientResourceNumber())
 65 {
 66 }
 67 
 68 IDBResourceIdentifier::IDBResourceIdentifier(const IDBClient::IDBConnectionProxy& connectionProxy, const IDBRequest& request)
 69     : m_idbConnectionIdentifier(connectionProxy.serverConnectionIdentifier())
 70     , m_resourceNumber(request.resourceIdentifier().m_resourceNumber)
 71 {
 72 }
 73 
 74 IDBResourceIdentifier::IDBResourceIdentifier(const IDBServer::IDBConnectionToClient& connection)
 75     : m_idbConnectionIdentifier(connection.identifier())
 76     , m_resourceNumber(nextServerResourceNumber())
 77 {
 78 }
 79 
 80 IDBResourceIdentifier IDBResourceIdentifier::isolatedCopy() const
 81 {
 82     return IDBResourceIdentifier(m_idbConnectionIdentifier, m_resourceNumber);
 83 }
 84 
 85 IDBResourceIdentifier IDBResourceIdentifier::emptyValue()
 86 {
 87     return IDBResourceIdentifier(0, 0);
 88 }
 89 
 90 IDBResourceIdentifier IDBResourceIdentifier::deletedValue()
 91 {
 92     return IDBResourceIdentifier(std::numeric_limits<uint64_t>::max(), std::numeric_limits<uint64_t>::max());
 93 }
 94 
 95 bool IDBResourceIdentifier::isHashTableDeletedValue() const
 96 {
 97     return m_idbConnectionIdentifier == std::numeric_limits<uint64_t>::max()
 98         && m_resourceNumber == std::numeric_limits<uint64_t>::max();
 99 }
100 
101 #if !LOG_DISABLED
102 
103 String IDBResourceIdentifier::loggingString() const
104 {
105     return makeString('<', m_idbConnectionIdentifier, ", ", m_resourceNumber, '>');
106 }
