33 StaticPasteboard::StaticPasteboard() = default;
34 StaticPasteboard::~StaticPasteboard() = default;
35 
36 bool StaticPasteboard::hasData()
37 {
38     return m_customData.hasData();
39 }
40 
41 Vector<String> StaticPasteboard::typesSafeForBindings(const String&)
42 {
43     return m_customData.orderedTypes();
44 }
45 
46 Vector<String> StaticPasteboard::typesForLegacyUnsafeBindings()
47 {
48     return m_customData.orderedTypes();
49 }
50 
51 String StaticPasteboard::readString(const String& type)
52 {
53     return m_customData.readString(type);
54 }
55 
56 String StaticPasteboard::readStringInCustomData(const String& type)
57 {
58     return m_customData.readStringInCustomData(type);
59 }
60 
61 void StaticPasteboard::writeString(const String& type, const String& value)
62 {
63     m_customData.writeString(type, value);
64 }
65 
66 void StaticPasteboard::writeData(const String& type, Ref<SharedBuffer>&& data)
67 {
68     m_customData.writeData(type, WTFMove(data));
69 }
70 
71 void StaticPasteboard::writeStringInCustomData(const String& type, const String& value)
72 {
73     m_customData.writeStringInCustomData(type, value);
74 }
75 
76 void StaticPasteboard::clear()
77 {
78     m_customData.clear();
79 }
80 
81 void StaticPasteboard::clear(const String& type)
82 {
83     m_customData.clear(type);
84 }
85 
86 PasteboardCustomData StaticPasteboard::takeCustomData()
87 {
88     return std::exchange(m_customData, { });
89 }
