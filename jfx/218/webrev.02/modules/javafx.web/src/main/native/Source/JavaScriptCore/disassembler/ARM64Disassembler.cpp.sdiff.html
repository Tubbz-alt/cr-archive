36 bool tryToDisassemble(const MacroAssemblerCodePtr<DisassemblyPtrTag>& codePtr, size_t size, const char* prefix, PrintStream& out)
37 {
38     A64DOpcode arm64Opcode;
39 
40     uint32_t* currentPC = codePtr.untaggedExecutableAddress<uint32_t*>();
41     size_t byteCount = size;
42 
43     while (byteCount) {
44         char pcString[20];
45         snprintf(pcString, sizeof(pcString), "0x%lx", reinterpret_cast<unsigned long>(currentPC));
46         out.printf("%s%16s: %s\n", prefix, pcString, arm64Opcode.disassemble(currentPC));
47         currentPC++;
48         byteCount -= sizeof(uint32_t);
49     }
50 
51     return true;
52 }
