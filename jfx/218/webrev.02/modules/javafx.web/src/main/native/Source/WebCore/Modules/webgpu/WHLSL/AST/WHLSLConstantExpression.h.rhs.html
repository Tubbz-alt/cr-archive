 44 // FIXME: https://bugs.webkit.org/show_bug.cgi?id=198158 macOS Sierra doesn't seem to support putting Variants inside Variants,
 45 // so this is a wrapper class to make sure that doesn't happen. As soon as we don't
 46 // have to support Sierra, this can be migrated to a Variant proper.
 47 class ConstantExpression final {
 48     WTF_MAKE_FAST_ALLOCATED;
 49 public:
 50     ConstantExpression(IntegerLiteral&& integerLiteral)
 51         : m_variant(WTFMove(integerLiteral))
 52     {
 53     }
 54 
 55     ConstantExpression(UnsignedIntegerLiteral&& unsignedIntegerLiteral)
 56         : m_variant(WTFMove(unsignedIntegerLiteral))
 57     {
 58     }
 59 
 60     ConstantExpression(FloatLiteral&& floatLiteral)
 61         : m_variant(WTFMove(floatLiteral))
 62     {
 63     }
 64 





 65     ConstantExpression(BooleanLiteral&& booleanLiteral)
 66         : m_variant(WTFMove(booleanLiteral))
 67     {
 68     }
 69 
 70     ConstantExpression(EnumerationMemberLiteral&& enumerationMemberLiteral)
 71         : m_variant(WTFMove(enumerationMemberLiteral))
 72     {
 73     }
 74 
 75     ConstantExpression(const ConstantExpression&) = delete;
 76     ConstantExpression(ConstantExpression&&) = default;
 77 
 78     ConstantExpression& operator=(const ConstantExpression&) = delete;
 79     ConstantExpression& operator=(ConstantExpression&&) = default;
 80 
 81     IntegerLiteral& integerLiteral()
 82     {
 83         ASSERT(WTF::holds_alternative<IntegerLiteral>(m_variant));
 84         return WTF::get<IntegerLiteral>(m_variant);
 85     }
 86 
 87     template <typename Visitor> auto visit(const Visitor& visitor) -> decltype(WTF::visit(visitor, std::declval<Variant<IntegerLiteral, UnsignedIntegerLiteral, FloatLiteral, BooleanLiteral, EnumerationMemberLiteral>&>()))
 88     {
 89         return WTF::visit(visitor, m_variant);
 90     }
 91 
 92     template <typename Visitor> auto visit(const Visitor& visitor) const -> decltype(WTF::visit(visitor, std::declval<Variant<IntegerLiteral, UnsignedIntegerLiteral, FloatLiteral, BooleanLiteral, EnumerationMemberLiteral>&>()))
 93     {
 94         return WTF::visit(visitor, m_variant);
 95     }
 96 
 97     ConstantExpression clone() const
 98     {
 99         return WTF::visit(WTF::makeVisitor([&](const IntegerLiteral& integerLiteral) -> ConstantExpression {
100             return integerLiteral.clone();
101         }, [&](const UnsignedIntegerLiteral& unsignedIntegerLiteral) -> ConstantExpression {
102             return unsignedIntegerLiteral.clone();
103         }, [&](const FloatLiteral& floatLiteral) -> ConstantExpression {
104             return floatLiteral.clone();


105         }, [&](const BooleanLiteral& booleanLiteral) -> ConstantExpression {
106             return booleanLiteral.clone();
107         }, [&](const EnumerationMemberLiteral& enumerationMemberLiteral) -> ConstantExpression {
108             return enumerationMemberLiteral.clone();
109         }), m_variant);
110     }
111 
112     bool matches(const ConstantExpression& other) const
113     {
114         Optional<bool> result;
115         double value;
116         visit(WTF::makeVisitor([&](const IntegerLiteral& integerLiteral) {
117             value = integerLiteral.value();
118         }, [&](const UnsignedIntegerLiteral& unsignedIntegerLiteral) {
119             value = unsignedIntegerLiteral.value();
120         }, [&](const FloatLiteral& floatLiteral) {
121             value = floatLiteral.value();


122         }, [&](const BooleanLiteral& booleanLiteral) {
123             if (WTF::holds_alternative<BooleanLiteral>(other.m_variant)) {
124                 const auto& otherBooleanLiteral = WTF::get<BooleanLiteral>(other.m_variant);
125                 result = booleanLiteral.value() == otherBooleanLiteral.value();
126             } else
127                 result = false;
128         }, [&](const EnumerationMemberLiteral& enumerationMemberLiteral) {
129             if (WTF::holds_alternative<EnumerationMemberLiteral>(other.m_variant)) {
130                 const auto& otherMemberReference = WTF::get<EnumerationMemberLiteral>(other.m_variant);
131                 result = enumerationMemberLiteral.enumerationMember() == otherMemberReference.enumerationMember();
132             } else
133                 result = false;
134         }));
135 
136         if (result)
137             return *result;
138 
139         other.visit(WTF::makeVisitor([&](const IntegerLiteral& integerLiteral) {
140             result = value == integerLiteral.value();
141         }, [&](const UnsignedIntegerLiteral& unsignedIntegerLiteral) {
142             result = value == unsignedIntegerLiteral.value();
143         }, [&](const FloatLiteral& floatLiteral) {
144             result = value == floatLiteral.value();


145         }, [&](const BooleanLiteral&) {
146             result = false;
147         }, [&](const EnumerationMemberLiteral&) {
148             result = false;
149         }));
150 
151         ASSERT(result);
152         return *result;
153     }
154 
155 private:
156     Variant<
157         IntegerLiteral,
158         UnsignedIntegerLiteral,
159         FloatLiteral,

160         BooleanLiteral,
161         EnumerationMemberLiteral
162         > m_variant;
163 };
