 71 void DragClientJava::startDrag(DragItem item, DataTransfer& dataTransfer, Frame&)
 72 {
 73     auto& dragImage = item.image;
 74     auto dragImageOrigin = item.dragLocationInContentCoordinates;
 75     auto eventPos = item.eventPositionInContentCoordinates;
 76     auto dragSourceAction = item.sourceAction;
 77 
 78     JNIEnv* env = WTF::GetJavaEnv();
 79     static jmethodID mid = env->GetMethodID(
 80         PG_GetWebPageClass(env),
 81         "fwkStartDrag", "("
 82         "Ljava/lang/Object;"
 83         "II"
 84         "II"
 85         "[Ljava/lang/String;"
 86         "[Ljava/lang/Object;"
 87         "Z"
 88         ")V");
 89     ASSERT(mid);
 90 
 91     static JGClass clsString(env->FindClass("java/lang/String"));
 92     static JGClass clsObject(env->FindClass("java/lang/Object"));
 93 
 94     // we are temporary changing dataTransfer security context
 95     // for transfer-to-Java purposes.
 96     auto actualStoreMode = dataTransfer.storeMode();
 97     dataTransfer.setStoreMode(DataTransfer::StoreMode::Readonly);
 98 
 99     Vector<String> mimeTypes(dataTransfer.types());
100     JLObjectArray jmimeTypes(env->NewObjectArray(mimeTypes.size(), clsString, NULL));
101     JLObjectArray jvalues(env->NewObjectArray(mimeTypes.size(), clsObject, NULL));
102     WTF::CheckAndClearException(env); // OOME
103 
104     auto document = WebPage::pageFromJObject(m_webPage)->mainFrame().document();
105     if (document) {
106         int index = 0;
107         for(const auto& mime : mimeTypes) {
108             String value = dataTransfer.getData(*document, mime);
109 
110             env->SetObjectArrayElement(
111                 jmimeTypes,
112                 index,
113                 (jstring)mime.toJavaString(env));
114 
115             env->SetObjectArrayElement(
116                 jvalues,
117                 index,
118                 (jstring)value.toJavaString(env));
119             index++;
120         }
121     }
122     // restore the original store mode
123     dataTransfer.setStoreMode(actualStoreMode);
124 
125     // Attention! [jimage] can be the instance of WCImage or WCImageFrame class.
126     // The nature of raster is too different to make a conversion inside the native code.
127     jobject jimage = dragImage.get() && dragImage.get()->javaImage()
128                   ? jobject(*(dragImage.get()->javaImage())) : nullptr;
129 
130     bool isImageSource = dragSourceAction & DragSourceActionImage;
131 
132     env->CallVoidMethod(m_webPage, mid, jimage,
133         eventPos.x() - dragImageOrigin.x(),
134         eventPos.y() - dragImageOrigin.y(),
135         eventPos.x(),
136         eventPos.y(),
137         jobjectArray(jmimeTypes),
138         jobjectArray(jvalues),
139         bool_to_jbool(isImageSource));
140     WTF::CheckAndClearException(env);
141 }
