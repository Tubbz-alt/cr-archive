 58 ALWAYS_INLINE const char* name(Kind kind)
 59 {
 60     switch (kind) {


 61     case Primitive:
 62         return "Primitive";
 63     case JSValue:
 64         return "JSValue";
 65     case NumberOfKinds:
 66         break;
 67     }
 68     RELEASE_ASSERT_NOT_REACHED();
 69     return nullptr;
 70 }
