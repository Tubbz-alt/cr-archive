35 const ClassInfo JSStringIterator::s_info = { "String Iterator", &Base::s_info, nullptr, nullptr, CREATE_METHOD_TABLE(JSStringIterator) };
36 
37 void JSStringIterator::finishCreation(VM& vm, JSGlobalObject*, JSString* iteratedString)
38 {
39     Base::finishCreation(vm);
40     ASSERT(inherits(vm, info()));
41     putDirect(vm, vm.propertyNames->builtinNames().iteratedStringPrivateName(), iteratedString);
42     putDirect(vm, vm.propertyNames->builtinNames().stringIteratorNextIndexPrivateName(), jsNumber(0));
43 }
44 
45 JSValue JSStringIterator::iteratedValue(ExecState* exec) const
46 {
47     return getDirect(exec->vm(), exec->vm().propertyNames->builtinNames().iteratedStringPrivateName());




48 }
49 
50 JSStringIterator* JSStringIterator::clone(ExecState* exec)
51 {
52     VM& vm = exec->vm();
53     JSValue iteratedString = getDirect(vm, vm.propertyNames->builtinNames().iteratedStringPrivateName());
54     JSValue nextIndex = getDirect(vm, vm.propertyNames->builtinNames().stringIteratorNextIndexPrivateName());
55 
56     auto clone = JSStringIterator::create(exec, exec->jsCallee()->globalObject(vm)->stringIteratorStructure(), asString(iteratedString));
57     clone->putDirect(vm, vm.propertyNames->builtinNames().stringIteratorNextIndexPrivateName(), nextIndex);
58     return clone;
59 }
