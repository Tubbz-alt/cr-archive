 48 class JSWebAssemblyInstance final : public JSNonFinalObject {
 49 public:
 50     using Base = JSNonFinalObject;
 51     static constexpr bool needsDestruction = true;
 52     static void destroy(JSCell*);
 53 
 54     template<typename CellType, SubspaceAccess mode>
 55     static IsoSubspace* subspaceFor(VM& vm)
 56     {
 57         return vm.webAssemblyInstanceSpace<mode>();
 58     }
 59 
 60     static Identifier createPrivateModuleKey();
 61 
 62     static JSWebAssemblyInstance* tryCreate(VM&, JSGlobalObject*, const Identifier& moduleKey, JSWebAssemblyModule*, JSObject* importObject, Structure*, Ref<Wasm::Module>&&, Wasm::CreationMode);
 63     static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
 64 
 65     DECLARE_EXPORT_INFO;
 66 
 67     void finalizeCreation(VM&, JSGlobalObject*, Ref<Wasm::CodeBlock>&&, JSObject* importObject, Wasm::CreationMode);
 68 
 69     Wasm::Instance& instance() { return m_instance.get(); }
 70     JSModuleNamespaceObject* moduleNamespaceObject() { return m_moduleNamespaceObject.get(); }

 71 
 72     JSWebAssemblyMemory* memory() { return m_memory.get(); }
 73     void setMemory(VM& vm, JSWebAssemblyMemory* value) {
 74         ASSERT(!memory());
 75         m_memory.set(vm, this, value);
 76         instance().setMemory(makeRef(memory()->memory()));
 77     }
 78     Wasm::MemoryMode memoryMode() { return memory()->memory().mode(); }
 79 
 80     JSWebAssemblyTable* table(unsigned i) { return m_tables[i].get(); }
 81     void setTable(VM& vm, uint32_t index, JSWebAssemblyTable* value)
 82     {
 83         ASSERT(index < m_tables.size());
 84         ASSERT(!table(index));
 85         m_tables[index].set(vm, this, value);
 86         instance().setTable(index, makeRef(*table(index)->table()));
 87     }
 88 
 89     void linkGlobal(VM& vm, uint32_t index, JSWebAssemblyGlobal* value)
 90     {
 91         ASSERT(value == value->global()->owner<JSWebAssemblyGlobal>());
 92         instance().linkGlobal(index, makeRef(*value->global()));
 93         vm.heap.writeBarrier(this, value);
 94     }
 95 
 96     JSGlobalObject* globalObject() const { return m_globalObject.get(); }
 97     JSWebAssemblyModule* module() const { return m_module.get(); }
 98 
 99     static size_t offsetOfInstance() { return OBJECT_OFFSETOF(JSWebAssemblyInstance, m_instance); }
100     static size_t offsetOfModule() { return OBJECT_OFFSETOF(JSWebAssemblyInstance, m_module); }
101     static size_t offsetOfGlobalObject() { return OBJECT_OFFSETOF(JSWebAssemblyInstance, m_globalObject); }
102     static size_t offsetOfVM() { return OBJECT_OFFSETOF(JSWebAssemblyInstance, m_vm); }
103 
104 protected:
105     JSWebAssemblyInstance(VM&, Structure*, Ref<Wasm::Instance>&&);
106     void finishCreation(VM&, JSWebAssemblyModule*, JSModuleNamespaceObject*);

107     static void visitChildren(JSCell*, SlotVisitor&);
108 
109 private:
110     Ref<Wasm::Instance> m_instance;
111     VM* m_vm;
112 
113     WriteBarrier<JSGlobalObject> m_globalObject;
114     WriteBarrier<JSWebAssemblyModule> m_module;
115     WriteBarrier<JSWebAssemblyCodeBlock> m_codeBlock;
116     WriteBarrier<JSModuleNamespaceObject> m_moduleNamespaceObject;
117     WriteBarrier<JSWebAssemblyMemory> m_memory;
118     Vector<WriteBarrier<JSWebAssemblyTable>> m_tables;

119 };
