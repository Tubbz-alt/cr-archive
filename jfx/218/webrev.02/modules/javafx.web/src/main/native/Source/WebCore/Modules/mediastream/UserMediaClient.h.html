39 namespace WebCore {
40 
41 class CaptureDevice;
42 class Document;
43 class Page;
44 class UserMediaRequest;
45 
46 class UserMediaClient {
47 public:
48     virtual void pageDestroyed() = 0;
49 
50     virtual void requestUserMediaAccess(UserMediaRequest&) = 0;
51     virtual void cancelUserMediaAccessRequest(UserMediaRequest&) = 0;
52 
53     virtual void enumerateMediaDevices(Document&, CompletionHandler<void(const Vector<CaptureDevice>&, const String&)>&&) = 0;
54 
55     enum DeviceChangeObserverTokenType { };
56     using DeviceChangeObserverToken = ObjectIdentifier<DeviceChangeObserverTokenType>;
57     virtual DeviceChangeObserverToken addDeviceChangeObserver(WTF::Function<void()>&&) = 0;
58     virtual void removeDeviceChangeObserver(DeviceChangeObserverToken) = 0;
59 
60 protected:
61     virtual ~UserMediaClient() = default;
62 };
63 
64 WEBCORE_EXPORT void provideUserMediaTo(Page*, UserMediaClient*);
65 
66 } // namespace WebCore

68 #endif // ENABLE(MEDIA_STREAM)
