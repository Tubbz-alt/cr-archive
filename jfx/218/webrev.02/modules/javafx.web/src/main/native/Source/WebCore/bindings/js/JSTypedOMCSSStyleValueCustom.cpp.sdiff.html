38 JSValue toJSNewlyCreated(ExecState*, JSDOMGlobalObject* globalObject, Ref<TypedOMCSSStyleValue>&& value)
39 {
40     if (value->isUnitValue())

42     if (value->isUnparsedValue())
43         return createWrapper<TypedOMCSSUnparsedValue>(globalObject, WTFMove(value));

46 
47     ASSERT_NOT_REACHED();
48     return createWrapper<TypedOMCSSStyleValue>(globalObject, WTFMove(value));
49 }
