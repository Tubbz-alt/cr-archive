 78 ResourceHandle::ResourceHandle(NetworkingContext* context, const ResourceRequest& request, ResourceHandleClient* client, bool defersLoading, bool shouldContentSniff, bool shouldContentEncodingSniff)
 79     : d(makeUnique<ResourceHandleInternal>(this, context, request, client, defersLoading, shouldContentSniff && shouldContentSniffURL(request.url()), shouldContentEncodingSniff))
 80 {
 81     if (!request.url().isValid()) {
 82         scheduleFailure(InvalidURLFailure);
 83         return;
 84     }
 85 
 86     if (!portAllowed(request.url())) {
 87         scheduleFailure(BlockedFailure);
 88         return;
 89     }
 90 }
 91 
 92 RefPtr<ResourceHandle> ResourceHandle::create(NetworkingContext* context, const ResourceRequest& request, ResourceHandleClient* client, bool defersLoading, bool shouldContentSniff, bool shouldContentEncodingSniff)
 93 {
 94     if (auto constructor = builtinResourceHandleConstructorMap().get(request.url().protocol().toStringWithoutCopying()))
 95         return constructor(request, client);
 96 
 97     auto newHandle = adoptRef(*new ResourceHandle(context, request, client, defersLoading, shouldContentSniff, shouldContentEncodingSniff));
 98 
 99     if (newHandle->d->m_scheduledFailureType != NoFailure)
100         return newHandle;
101 
102     if (newHandle->start())
103         return newHandle;
104 
105     return nullptr;
106 }

108 void ResourceHandle::scheduleFailure(FailureType type)
109 {
110     d->m_scheduledFailureType = type;
111     d->m_failureTimer.startOneShot(0_s);
112 }
113 
114 void ResourceHandle::failureTimerFired()
115 {
116     if (!client())
117         return;
118 
119     switch (d->m_scheduledFailureType) {
120         case NoFailure:
121             ASSERT_NOT_REACHED();
122             return;
123         case BlockedFailure:
124             d->m_scheduledFailureType = NoFailure;
125             client()->wasBlocked(this);
126             return;
127         case InvalidURLFailure:
128             d->m_scheduledFailureType = NoFailure;
129             client()->cannotShowURL(this);
130             return;
131     }
132 
133     ASSERT_NOT_REACHED();
134 }

136 void ResourceHandle::loadResourceSynchronously(NetworkingContext* context, const ResourceRequest& request, StoredCredentialsPolicy storedCredentialsPolicy, ResourceError& error, ResourceResponse& response, Vector<char>& data)
137 {
138     if (auto constructor = builtinResourceHandleSynchronousLoaderMap().get(request.url().protocol().toStringWithoutCopying())) {
139         constructor(context, request, storedCredentialsPolicy, error, response, data);
140         return;
141     }
142 
143     platformLoadResourceSynchronously(context, request, storedCredentialsPolicy, error, response, data);
144 }

156 void ResourceHandle::didReceiveResponse(ResourceResponse&& response, CompletionHandler<void()>&& completionHandler)
157 {
158     if (response.isHTTP09()) {
159         auto url = response.url();
160         Optional<uint16_t> port = url.port();
161         if (port && !WTF::isDefaultPortForProtocol(port.value(), url.protocol())) {
162             cancel();
163             String message = "Cancelled load from '" + url.stringCenterEllipsizedToLength() + "' because it is using HTTP/0.9.";
164             d->m_client->didFail(this, { String(), 0, url, message });
165             completionHandler();
166             return;
167         }
168     }
169     client()->didReceiveResponseAsync(this, WTFMove(response), WTFMove(completionHandler));
170 }

217 bool ResourceHandle::shouldContentSniffURL(const URL& url)
218 {
219 #if PLATFORM(COCOA)
220     if (shouldForceContentSniffing)
221         return true;
222 #endif
223     // We shouldn't content sniff file URLs as their MIME type should be established via their extension.
224     return !url.protocolIs("file");
225 }

232 void ResourceHandle::setDefersLoading(bool defers)
233 {
234     LOG(Network, "Handle %p setDefersLoading(%s)", this, defers ? "true" : "false");
235 
236     ASSERT(d->m_defersLoading != defers); // Deferring is not counted, so calling setDefersLoading() repeatedly is likely to be in error.
237     d->m_defersLoading = defers;
238 
239     if (defers) {
240         ASSERT(d->m_failureTimer.isActive() == (d->m_scheduledFailureType != NoFailure));
241         if (d->m_failureTimer.isActive())
242             d->m_failureTimer.stop();
243     } else if (d->m_scheduledFailureType != NoFailure) {
244         ASSERT(!d->m_failureTimer.isActive());
245         d->m_failureTimer.startOneShot(0_s);
246     }
247 
248     platformSetDefersLoading(defers);
249 }
