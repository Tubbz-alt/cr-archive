43 void* GigacageAlignedMemoryAllocator::tryAllocateAlignedMemory(size_t alignment, size_t size)
44 {
45 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
46     return m_heap.memalign(alignment, size, true);
47 #else
48     return Gigacage::tryAlignedMalloc(m_kind, alignment, size);
49 #endif
50 }

52 void GigacageAlignedMemoryAllocator::freeAlignedMemory(void* basePtr)
53 {
54 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
55     return m_heap.free(basePtr);
56 #else
57     Gigacage::alignedFree(m_kind, basePtr);
58 #endif
59 }
