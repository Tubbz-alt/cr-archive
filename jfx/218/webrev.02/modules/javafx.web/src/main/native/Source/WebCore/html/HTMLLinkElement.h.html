 42 typedef EventSender<HTMLLinkElement> LinkEventSender;

 44 class HTMLLinkElement final : public HTMLElement, public CachedStyleSheetClient, public LinkLoaderClient {
 45     WTF_MAKE_ISO_ALLOCATED(HTMLLinkElement);
 46 public:
 47     static Ref<HTMLLinkElement> create(const QualifiedName&, Document&, bool createdByParser);
 48     virtual ~HTMLLinkElement();
 49 
 50     URL href() const;
 51     WEBCORE_EXPORT const AtomString& rel() const;
 52 
 53     String target() const final;
 54 
 55     const AtomString& type() const;
 56 
 57     Optional<LinkIconType> iconType() const;
 58 
 59     CSSStyleSheet* sheet() const { return m_sheet.get(); }
 60 
 61     bool styleSheetIsLoading() const;
 62 
 63     bool isDisabled() const { return m_disabledState == Disabled; }
 64     bool isEnabledViaScript() const { return m_disabledState == EnabledViaScript; }
 65     DOMTokenList& sizes();
 66 
 67     WEBCORE_EXPORT void setCrossOrigin(const AtomString&);
 68     WEBCORE_EXPORT String crossOrigin() const;
 69     WEBCORE_EXPORT void setAs(const AtomString&);
 70     WEBCORE_EXPORT String as() const;
 71 
 72     void dispatchPendingEvent(LinkEventSender*);
 73     static void dispatchPendingLoadEvents();
 74 
 75     WEBCORE_EXPORT DOMTokenList& relList();
 76 
 77 #if ENABLE(APPLICATION_MANIFEST)
 78     bool isApplicationManifest() const { return m_relAttribute.isApplicationManifest; }
 79 #endif
 80 
 81     void allowPrefetchLoadAndErrorForTesting() { m_allowPrefetchLoadAndErrorForTesting = true; }
 82 
 83 private:
 84     void parseAttribute(const QualifiedName&, const AtomString&) final;
 85 
 86     bool shouldLoadLink() final;
 87     void process();
 88     static void processCallback(Node*);
 89     void clearSheet();
 90 
 91     InsertedIntoAncestorResult insertedIntoAncestor(InsertionType, ContainerNode&) final;
 92     void didFinishInsertingNode() final;
 93     void removedFromAncestor(RemovalType, ContainerNode&) final;
 94 
 95     void initializeStyleSheet(Ref<StyleSheetContents>&&, const CachedCSSStyleSheet&, MediaQueryParserContext);
 96 
 97     // from CachedResourceClient
 98     void setCSSStyleSheet(const String& href, const URL& baseURL, const String& charset, const CachedCSSStyleSheet*) final;
 99     bool sheetLoaded() final;
100     void notifyLoadedSheetAndAllCriticalSubresources(bool errorOccurred) final;
101     void startLoadingDynamicSheet() final;
102 
103     void linkLoaded() final;
104     void linkLoadingErrored() final;
105 
106     bool isAlternate() const { return m_disabledState == Unset && m_relAttribute.isAlternate; }
107 
108     void setDisabledState(bool);
109 
110     bool isURLAttribute(const Attribute&) const final;
111 
112     void defaultEventHandler(Event&) final;
113     void handleClick(Event&);
114 
115     HTMLLinkElement(const QualifiedName&, Document&, bool createdByParser);
116 
117     void addSubresourceAttributeURLs(ListHashSet<URL>&) const final;
118 
119     void finishParsingChildren() final;
120 
121     enum PendingSheetType : uint8_t { Unknown, ActiveSheet, InactiveSheet };
122     void addPendingSheet(PendingSheetType);
123 
124     void removePendingSheet();
125 
126     LinkLoader m_linkLoader;
127     Style::Scope* m_styleScope { nullptr };
128     CachedResourceHandle<CachedCSSStyleSheet> m_cachedSheet;
129     RefPtr<CSSStyleSheet> m_sheet;
130     enum DisabledState : uint8_t {
131         Unset,
132         EnabledViaScript,
133         Disabled
134     };
135 
136     String m_type;
137     String m_media;
138     String m_integrityMetadataForPendingSheetRequest;
139     std::unique_ptr<DOMTokenList> m_sizes;
140     std::unique_ptr<DOMTokenList> m_relList;
141     DisabledState m_disabledState;
142     LinkRelAttribute m_relAttribute;
143     bool m_loading : 1;
144     bool m_createdByParser : 1;
145     bool m_firedLoad : 1;
146     bool m_loadedResource : 1;
147     bool m_isHandlingBeforeLoad : 1;
148     bool m_allowPrefetchLoadAndErrorForTesting : 1;
149     PendingSheetType m_pendingSheetType;
150 };
