26 #pragma once
27 
28 #include "JSSegmentedVariableObject.h"
29 
30 namespace JSC {
31 
32 class JSGlobalLexicalEnvironment final : public JSSegmentedVariableObject {
33 
34 public:
35     using Base = JSSegmentedVariableObject;
36 
37     static const unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot;






38 
39     static JSGlobalLexicalEnvironment* create(VM& vm, Structure* structure, JSScope* parentScope)
40     {
41         JSGlobalLexicalEnvironment* result =
42             new (NotNull, allocateCell<JSGlobalLexicalEnvironment>(vm.heap)) JSGlobalLexicalEnvironment(vm, structure, parentScope);
43         result->finishCreation(vm);
44         result->symbolTable()->setScopeType(SymbolTable::ScopeType::GlobalLexicalScope);
45         return result;
46     }
47 
48     static bool getOwnPropertySlot(JSObject*, ExecState*, PropertyName, PropertySlot&);
49     static bool put(JSCell*, ExecState*, PropertyName, JSValue, PutPropertySlot&);
50 
51     static void destroy(JSCell*);
52     // We don't need a destructor because we use a finalizer instead.
53     static const bool needsDestruction = false;
54 
55     bool isEmpty() const { return !symbolTable()->size(); }
56     bool isConstVariable(UniquedStringImpl*);
57 
58     DECLARE_INFO;
59 
60     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject)
61     {
62         return Structure::create(vm, globalObject, jsNull(), TypeInfo(GlobalLexicalEnvironmentType, StructureFlags), info());
63     }
64 
65 protected:
66     JSGlobalLexicalEnvironment(VM& vm, Structure* structure, JSScope* scope)
67         : Base(vm, structure, scope)
68     {
69     }
70 };
71 
72 } // namespace JSC
