 34 bool CheckAndClearException(JNIEnv* env)
 35 {
 36     if (JNI_TRUE == env->ExceptionCheck()) {
 37         env->ExceptionDescribe();
 38         env->ExceptionClear();
 39         return true;
 40     }
 41     return false;
 42 }

 45 jclass PL_GetClass(JNIEnv* env)
 46 {
 47     static JGClass cls(
 48         env->FindClass("com/sun/webkit/perf/PerfLogger"));
 49     return cls;
 50 }
 51 
 52 JLObject PL_GetLogger(JNIEnv* env, const char* name)
 53 {
 54     static jmethodID mid =
 55         env->GetStaticMethodID(PL_GetClass(env),
 56             "getLogger",
 57             "(Ljava/lang/String;)Lcom/sun/webkit/perf/PerfLogger;");
 58     ASSERT(mid);
 59 
 60     JLObject jlogger(env->CallStaticObjectMethod(PL_GetClass(env), mid,
 61         (jstring)JLString(env->NewStringUTF(name))));
 62     CheckAndClearException(env);
 63 
 64     return jlogger;
 65 }
 66 
 67 void PL_ResumeCount(JNIEnv* env, jobject perfLogger, const char* probe)
 68 {
 69     static jmethodID mid =
 70         env->GetMethodID(PL_GetClass(env),
 71             "resumeCount",
 72             "(Ljava/lang/String;)V");
 73     ASSERT(mid);
 74 
 75     env->CallVoidMethod(perfLogger, mid,
 76         (jstring)JLString(env->NewStringUTF(probe)));
 77     CheckAndClearException(env);
 78 }

 80 void PL_SuspendCount(JNIEnv* env, jobject perfLogger, const char* probe)

 89         (jstring)JLString(env->NewStringUTF(probe)));
 90     CheckAndClearException(env);
 91 }
 92 
 93 bool PL_IsEnabled(JNIEnv* env, jobject perfLogger)
 94 {
 95     static jmethodID mid =
 96         env->GetMethodID(PL_GetClass(env),
 97             "isEnabled",
 98             "()Z");
 99     ASSERT(mid);
100 
101     static jboolean isEnabled = env->CallBooleanMethod(perfLogger, mid);
102     CheckAndClearException(env);
103 
104     return isEnabled;
105 }

112 #include <crtdbg.h>

114 JNIEXPORT jint JNICALL JNI_OnLoad(JavaVM* vm, void*)

115 {
116 #if PLATFORM(JAVA_WIN) && !defined(NDEBUG)
117     _CrtSetReportMode( _CRT_ERROR, _CRTDBG_MODE_FILE );
118     _CrtSetReportFile( _CRT_ERROR, _CRTDBG_FILE_STDERR );
119 
120     // Get current flag
121     int tmpFlag = _CrtSetDbgFlag( _CRTDBG_REPORT_FLAG );
122 
123     // Turn on leak-checking bit
124     tmpFlag |= _CRTDBG_CHECK_CRT_DF | _CRTDBG_LEAK_CHECK_DF; //| _CRTDBG_CHECK_EVERY_1024_DF;
125 
126     // Set flag to the new value
127     _CrtSetDbgFlag( tmpFlag );
128 #endif
129     jvm = vm;
130     return JNI_VERSION_1_2;
131 }
132 
133 JNIEXPORT void JNICALL JNI_OnUnload(JavaVM* , void*)
134 {
135 #if PLATFORM(JAVA_WIN) && !defined(NDEBUG)
136     _CrtDumpMemoryLeaks();
137 #endif
138     jvm = 0;
139 }

145 BOOL WINAPI DllMain(HINSTANCE hinstDLL, DWORD fdwReason, LPVOID lpvReserved)
146 {
147     if (fdwReason == DLL_PROCESS_ATTACH) {
148 #if defined(_MSC_VER) && _MSC_VER >= 1800 && _MSC_VER < 1900 && defined(_M_X64) || defined(__x86_64__)
149         // The VS2013 runtime has a bug where it mis-detects AVX-capable processors
150         // if the feature has been disabled in firmware. This causes us to crash
151         // in some of the math functions. For now, we disable those optimizations
152         // because Microsoft is not going to fix the problem in VS2013.
153         // FIXME: Remove this workaround when we switch to VS2015+.
154         _set_FMA3_enable(0);
155 #endif
156     }
157 
158     return TRUE;
159 }
