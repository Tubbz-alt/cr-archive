 28 // This is guarded by ENABLE_DFG_JIT only because it uses some value profiles
 29 // that are currently only used if the DFG is enabled (i.e. they are not
 30 // available in the profile-only configuration). Hopefully someday all of
 31 // these #if's will disappear...
 32 #if ENABLE(DFG_JIT)
 33 
 34 #include "BytecodeIndex.h"
 35 #include "GPRInfo.h"
 36 #include "Operands.h"
 37 
 38 namespace JSC {
 39 
 40 class UnaryArithProfile;
 41 class BinaryArithProfile;
 42 class CCallHelpers;
 43 class CodeBlock;
 44 class LazyOperandValueProfileKey;

 45 struct ValueProfile;
 46 
 47 class MethodOfGettingAValueProfile {
 48 public:
 49     MethodOfGettingAValueProfile()
 50         : m_kind(None)
 51     {
 52     }
 53 
 54     MethodOfGettingAValueProfile(ValueProfile* profile)
 55     {
 56         if (profile) {
 57             m_kind = Ready;
 58             u.profile = profile;
 59         } else
 60             m_kind = None;
 61     }
 62 
 63     MethodOfGettingAValueProfile(UnaryArithProfile* profile)
 64     {
 65         if (profile) {
 66             m_kind = UnaryArithProfileReady;
 67             u.unaryArithProfile = profile;
 68         } else
 69             m_kind = None;
 70     }
 71 
 72     MethodOfGettingAValueProfile(BinaryArithProfile* profile)
 73     {
 74         if (profile) {
 75             m_kind = BinaryArithProfileReady;
 76             u.binaryArithProfile = profile;
 77         } else
 78             m_kind = None;
 79     }
 80 
 81     static MethodOfGettingAValueProfile fromLazyOperand(
 82         CodeBlock*, const LazyOperandValueProfileKey&);
 83 
 84     explicit operator bool() const { return m_kind != None; }
 85 
 86     void emitReportValue(CCallHelpers&, JSValueRegs) const;
 87     void reportValue(JSValue);
 88 
 89 private:
 90     enum Kind {
 91         None,
 92         Ready,
 93         UnaryArithProfileReady,
 94         BinaryArithProfileReady,
 95         LazyOperand
 96     };
 97 
 98     Kind m_kind;
 99     union Data {
100         Data()
101             : profile(nullptr)
102         { }
103 
104         ValueProfile* profile;
105         UnaryArithProfile* unaryArithProfile;
106         BinaryArithProfile* binaryArithProfile;
107         struct {
108             CodeBlock* codeBlock;
109             BytecodeIndex bytecodeOffset;
110             Operand operand;
111         } lazyOperand;
112     } u;
113 };
114 
115 } // namespace JSC
116 
117 #endif // ENABLE(DFG_JIT)
