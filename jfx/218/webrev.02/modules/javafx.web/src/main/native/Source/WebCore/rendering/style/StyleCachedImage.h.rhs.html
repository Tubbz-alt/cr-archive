41 
42     bool operator==(const StyleImage& other) const final;

44     CachedImage* cachedImage() const final;

46     WrappedImagePtr data() const final { return m_cachedImage.get(); }
47 
48     Ref<CSSValue> cssValue() const final;
49 
50     bool canRender(const RenderElement*, float multiplier) const final;
51     bool isPending() const final;
52     void load(CachedResourceLoader&, const ResourceLoaderOptions&) final;
53     bool isLoaded() const final;

59     bool usesImageContainerSize() const final;

61     void addClient(RenderElement*) final;
62     void removeClient(RenderElement*) final;
63     RefPtr<Image> image(RenderElement*, const FloatSize&) const final;
64     float imageScaleFactor() const final;
65     bool knownToBeOpaque(const RenderElement*) const final;
66 
67 private:
68     StyleCachedImage(CSSImageValue&, float);
69     URL imageURL();

74     mutable CachedResourceHandle<CachedImage> m_cachedImage;
75 };
