26 #pragma once
27 
28 #if ENABLE(JIT)
29 
30 #include "CodeLocation.h"
31 #include <wtf/StdLibExtras.h>
32 #include <wtf/Vector.h>
33 
34 namespace JSC {
35 
36 class JITCodeMap {
37 private:
38     struct Entry {
39         Entry() { }
40 
41         Entry(unsigned bytecodeIndex, CodeLocationLabel<JSEntryPtrTag> codeLocation)
42             : m_bytecodeIndex(bytecodeIndex)
43             , m_codeLocation(codeLocation)
44         { }
45 
46         inline unsigned bytecodeIndex() const { return m_bytecodeIndex; }
47         inline CodeLocationLabel<JSEntryPtrTag> codeLocation() { return m_codeLocation; }
48 
49     private:
50         unsigned m_bytecodeIndex;
51         CodeLocationLabel<JSEntryPtrTag> m_codeLocation;
52     };
53 
54 public:
55     void append(unsigned bytecodeIndex, CodeLocationLabel<JSEntryPtrTag> codeLocation)
56     {
57         m_entries.append({ bytecodeIndex, codeLocation });
58     }
59 
60     void finish() { m_entries.shrinkToFit(); }
61 
62     CodeLocationLabel<JSEntryPtrTag> find(unsigned bytecodeIndex) const
63     {
64         auto* entry =
65             binarySearch<Entry, unsigned>(m_entries,
66                 m_entries.size(), bytecodeIndex, [] (Entry* entry) {
67                     return entry->bytecodeIndex();
68                 });
69         if (!entry)
70             return CodeLocationLabel<JSEntryPtrTag>();
71         return entry->codeLocation();
72     }
73 
74     explicit operator bool() const { return m_entries.size(); }
75 
76 private:
77     Vector<Entry> m_entries;
78 };
79 
80 } // namespace JSC
81 
82 #endif // ENABLE(JIT)
