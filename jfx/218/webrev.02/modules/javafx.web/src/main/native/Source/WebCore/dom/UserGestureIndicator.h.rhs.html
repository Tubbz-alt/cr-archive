 36 namespace WebCore {

 38 class Document;
 39 
 40 enum ProcessingUserGestureState {
 41     ProcessingUserGesture,
 42     ProcessingPotentialUserGesture,
 43     NotProcessingUserGesture
 44 };
 45 
 46 enum class UserGestureType { EscapeKey, Other };
 47 
 48 class UserGestureToken : public RefCounted<UserGestureToken>, public CanMakeWeakPtr<UserGestureToken> {
 49 public:
 50     static Ref<UserGestureToken> create(ProcessingUserGestureState state, UserGestureType gestureType)
 51     {
 52         return adoptRef(*new UserGestureToken(state, gestureType));
 53     }
 54 
 55     WEBCORE_EXPORT ~UserGestureToken();
 56 
 57     ProcessingUserGestureState state() const { return m_state; }
 58     bool processingUserGesture() const { return m_scope == GestureScope::All && m_state == ProcessingUserGesture; }
 59     bool processingUserGestureForMedia() const { return m_state == ProcessingUserGesture || m_state == ProcessingPotentialUserGesture; }
 60     UserGestureType gestureType() const { return m_gestureType; }
 61 
 62     void addDestructionObserver(WTF::Function<void (UserGestureToken&)>&& observer)
 63     {
 64         m_destructionObservers.append(WTFMove(observer));
 65     }
 66 
 67     DOMPasteAccessPolicy domPasteAccessPolicy() const { return m_domPasteAccessPolicy; }
 68     void didRequestDOMPasteAccess(DOMPasteAccessResponse response)
 69     {
 70         switch (response) {
 71         case DOMPasteAccessResponse::DeniedForGesture:
 72             m_domPasteAccessPolicy = DOMPasteAccessPolicy::Denied;
 73             break;
 74         case DOMPasteAccessResponse::GrantedForCommand:
 75             break;
 76         case DOMPasteAccessResponse::GrantedForGesture:
 77             m_domPasteAccessPolicy = DOMPasteAccessPolicy::Granted;
 78             break;
 79         }
 80     }
 81     void resetDOMPasteAccess() { m_domPasteAccessPolicy = DOMPasteAccessPolicy::NotRequestedYet; }
 82 
 83     enum class GestureScope { All, MediaOnly };
 84     void setScope(GestureScope scope) { m_scope = scope; }
 85     void resetScope() { m_scope = GestureScope::All; }
 86 
 87     bool hasExpired(Seconds expirationInterval) const
 88     {
 89         return m_startTime + expirationInterval < MonotonicTime::now();
 90     }
 91 
 92     MonotonicTime startTime() const { return m_startTime; }
 93 
 94 private:
 95     UserGestureToken(ProcessingUserGestureState state, UserGestureType gestureType)
 96         : m_state(state)
 97         , m_gestureType(gestureType)
 98     {
 99     }
100 
101     ProcessingUserGestureState m_state = NotProcessingUserGesture;
102     Vector<WTF::Function<void (UserGestureToken&)>> m_destructionObservers;
103     UserGestureType m_gestureType;
104     DOMPasteAccessPolicy m_domPasteAccessPolicy { DOMPasteAccessPolicy::NotRequestedYet };
105     GestureScope m_scope { GestureScope::All };
106     MonotonicTime m_startTime { MonotonicTime::now() };
107 };
108 
109 class UserGestureIndicator {
110     WTF_MAKE_FAST_ALLOCATED;
111     WTF_MAKE_NONCOPYABLE(UserGestureIndicator);
112 public:
113     WEBCORE_EXPORT static RefPtr<UserGestureToken> currentUserGesture();
114 
115     WEBCORE_EXPORT static bool processingUserGesture();
116     WEBCORE_EXPORT static bool processingUserGestureForMedia();
117 
118     // If a document is provided, its last known user gesture timestamp is updated.
119     enum class ProcessInteractionStyle { Immediate, Delayed };
120     WEBCORE_EXPORT explicit UserGestureIndicator(Optional<ProcessingUserGestureState>, Document* = nullptr, UserGestureType = UserGestureType::Other, ProcessInteractionStyle = ProcessInteractionStyle::Immediate);
121     WEBCORE_EXPORT explicit UserGestureIndicator(RefPtr<UserGestureToken>, UserGestureToken::GestureScope = UserGestureToken::GestureScope::All);
122     WEBCORE_EXPORT ~UserGestureIndicator();
123 
124 private:
125     RefPtr<UserGestureToken> m_previousToken;
126 };
