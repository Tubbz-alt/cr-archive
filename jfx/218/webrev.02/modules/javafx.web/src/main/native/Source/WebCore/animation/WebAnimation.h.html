 42 namespace WebCore {
 43 
 44 class AnimationEffect;
 45 class AnimationEventBase;
 46 class AnimationTimeline;
 47 class Document;
 48 class Element;
 49 class RenderStyle;
 50 
 51 template<typename IDLType> class DOMPromiseProxyWithResolveCallback;
 52 
 53 class WebAnimation : public RefCounted<WebAnimation>, public CanMakeWeakPtr<WebAnimation>, public EventTargetWithInlineData, public ActiveDOMObject {
 54     WTF_MAKE_ISO_ALLOCATED(WebAnimation);
 55 public:
 56     static Ref<WebAnimation> create(Document&, AnimationEffect*);
 57     static Ref<WebAnimation> create(Document&, AnimationEffect*, AnimationTimeline*);
 58     ~WebAnimation();
 59 
 60     static HashSet<WebAnimation*>& instances(const LockHolder&);
 61     static Lock& instancesMutex();
 62 
 63     virtual bool isDeclarativeAnimation() const { return false; }
 64     virtual bool isCSSAnimation() const { return false; }
 65     virtual bool isCSSTransition() const { return false; }
 66 
 67     const String& id() const { return m_id; }
 68     void setId(const String& id) { m_id = id; }
 69 
 70     AnimationEffect* effect() const { return m_effect.get(); }
 71     void setEffect(RefPtr<AnimationEffect>&&);
 72     AnimationTimeline* timeline() const;
 73     virtual void setTimeline(RefPtr<AnimationTimeline>&&);
 74 
 75     Optional<Seconds> currentTime() const;
 76     ExceptionOr<void> setCurrentTime(Optional<Seconds>);
 77 
 78     enum class Silently : uint8_t { Yes, No };
 79     double playbackRate() const { return m_playbackRate + 0; }
 80     void setPlaybackRate(double);
 81 
 82     enum class PlayState : uint8_t { Idle, Running, Paused, Finished };
 83     PlayState playState() const;
 84 
 85     enum class ReplaceState : uint8_t { Active, Removed, Persisted };
 86     ReplaceState replaceState() const { return m_replaceState; }
 87     void setReplaceState(ReplaceState replaceState) { m_replaceState = replaceState; }
 88 
 89     bool pending() const { return hasPendingPauseTask() || hasPendingPlayTask(); }
 90 
 91     using ReadyPromise = DOMPromiseProxyWithResolveCallback<IDLInterface<WebAnimation>>;
 92     ReadyPromise& ready() { return m_readyPromise.get(); }
 93 
 94     using FinishedPromise = DOMPromiseProxyWithResolveCallback<IDLInterface<WebAnimation>>;
 95     FinishedPromise& finished() { return m_finishedPromise.get(); }
 96 
 97     virtual void cancel();
 98     void cancel(Silently);
 99     ExceptionOr<void> finish();
100     ExceptionOr<void> play();
101     void updatePlaybackRate(double);
102     ExceptionOr<void> pause();
103     ExceptionOr<void> reverse();
104     void persist();
105     ExceptionOr<void> commitStyles();
106 
107     virtual Optional<double> startTime() const;
108     virtual void setStartTime(Optional<double>);
109     virtual Optional<double> bindingsCurrentTime() const;
110     virtual ExceptionOr<void> setBindingsCurrentTime(Optional<double>);
111     virtual PlayState bindingsPlayState() const { return playState(); }
112     virtual ReplaceState bindingsReplaceState() const { return replaceState(); }
113     virtual bool bindingsPending() const { return pending(); }
114     virtual ReadyPromise& bindingsReady() { return ready(); }
115     virtual FinishedPromise& bindingsFinished() { return finished(); }
116     virtual ExceptionOr<void> bindingsPlay() { return play(); }
117     virtual ExceptionOr<void> bindingsPause() { return pause(); }
118 
119     bool needsTick() const;
120     virtual void tick();
121     Seconds timeToNextTick() const;
122     virtual void resolve(RenderStyle&);
123     void effectTargetDidChange(Element* previousTarget, Element* newTarget);
124     void acceleratedStateDidChange();
125     void applyPendingAcceleratedActions();
126     void willChangeRenderer();
127 
128     bool isRunningAccelerated() const;
129     bool isCompletelyAccelerated() const;
130     bool isRelevant() const { return m_isRelevant; }
131     void updateRelevance();
132     void effectTimingDidChange(Optional<ComputedEffectTiming> = WTF::nullopt);
133     void suspendEffectInvalidation();
134     void unsuspendEffectInvalidation();
135     void setSuspended(bool);
136     bool isSuspended() const { return m_isSuspended; }
137     bool isReplaceable() const;
138     void remove();
139     void enqueueAnimationPlaybackEvent(const AtomString&, Optional<Seconds>, Optional<Seconds>);
140 
141     uint64_t globalPosition() const { return m_globalPosition; }
142     void setGlobalPosition(uint64_t globalPosition) { m_globalPosition = globalPosition; }
143 
144     bool hasPendingActivity() const final;
145 
146     virtual bool canHaveGlobalPosition() { return true; }
147 
148     // ContextDestructionObserver.
149     ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
150     void contextDestroyed() final;
151 
152     using RefCounted::ref;
153     using RefCounted::deref;
154 
155 protected:
156     explicit WebAnimation(Document&);
157 
158     void enqueueAnimationEvent(Ref<AnimationEventBase>&&);
159 
160 private:
161     enum class DidSeek : uint8_t { Yes, No };
162     enum class SynchronouslyNotify : uint8_t { Yes, No };
163     enum class RespectHoldTime : uint8_t { Yes, No };
164     enum class AutoRewind : uint8_t { Yes, No };
165     enum class TimeToRunPendingTask : uint8_t { NotScheduled, ASAP, WhenReady };
166 
167     void timingDidChange(DidSeek, SynchronouslyNotify, Silently = Silently::No);
168     void updateFinishedState(DidSeek, SynchronouslyNotify);
169     Seconds effectEndTime() const;
170     WebAnimation& readyPromiseResolve();
171     WebAnimation& finishedPromiseResolve();
172     Optional<Seconds> currentTime(RespectHoldTime) const;
173     ExceptionOr<void> silentlySetCurrentTime(Optional<Seconds>);
174     void finishNotificationSteps();
175     bool hasPendingPauseTask() const { return m_timeToRunPendingPauseTask != TimeToRunPendingTask::NotScheduled; }
176     bool hasPendingPlayTask() const { return m_timeToRunPendingPlayTask != TimeToRunPendingTask::NotScheduled; }
177     ExceptionOr<void> play(AutoRewind);
178     void runPendingPauseTask();
179     void runPendingPlayTask();
180     void resetPendingTasks(Silently = Silently::No);
181     void setEffectInternal(RefPtr<AnimationEffect>&&, bool = false);
182     void setTimelineInternal(RefPtr<AnimationTimeline>&&);
183     bool isEffectInvalidationSuspended() { return m_suspendCount; }
184     bool computeRelevance();
185     void invalidateEffect();
186     double effectivePlaybackRate() const;
187     void applyPendingPlaybackRate();
188 
189     RefPtr<AnimationEffect> m_effect;
190     WeakPtr<AnimationTimeline> m_timeline;
191     UniqueRef<ReadyPromise> m_readyPromise;
192     UniqueRef<FinishedPromise> m_finishedPromise;
193     Markable<Seconds, Seconds::MarkableTraits> m_previousCurrentTime;
194     Markable<Seconds, Seconds::MarkableTraits> m_startTime;
195     Markable<Seconds, Seconds::MarkableTraits> m_holdTime;
196     MarkableDouble m_pendingPlaybackRate;
197     double m_playbackRate { 1 };
198     String m_id;
199 
200     int m_suspendCount { 0 };
201 
202     bool m_isSuspended { false };
203     bool m_finishNotificationStepsMicrotaskPending;
204     bool m_isRelevant;
205     bool m_shouldSkipUpdatingFinishedStateWhenResolving;
206     bool m_hasScheduledEventsDuringTick { false };
207     TimeToRunPendingTask m_timeToRunPendingPlayTask { TimeToRunPendingTask::NotScheduled };
208     TimeToRunPendingTask m_timeToRunPendingPauseTask { TimeToRunPendingTask::NotScheduled };
209     ReplaceState m_replaceState { ReplaceState::Active };
210     uint64_t m_globalPosition { 0 };
211 
212     // ActiveDOMObject.
213     const char* activeDOMObjectName() const final;
214     void suspend(ReasonForSuspension) final;
215     void resume() final;
216     void stop() final;
217 
218     // EventTarget
219     EventTargetInterface eventTargetInterface() const final { return WebAnimationEventTargetInterfaceType; }
220     void refEventTarget() final { ref(); }
221     void derefEventTarget() final { deref(); }
222 };
223 
224 } // namespace WebCore
