 26 SVGPathByteStreamSource::SVGPathByteStreamSource(const SVGPathByteStream& stream)
 27 {
 28     m_streamCurrent = stream.begin();
 29     m_streamEnd = stream.end();
 30 }
 31 
 32 bool SVGPathByteStreamSource::hasMoreData() const
 33 {
 34     return m_streamCurrent < m_streamEnd;
 35 }
 36 
 37 bool SVGPathByteStreamSource::parseSVGSegmentType(SVGPathSegType& pathSegType)
 38 {
 39     pathSegType = static_cast<SVGPathSegType>(readSVGSegmentType());
 40     return true;
 41 }
 42 
 43 SVGPathSegType SVGPathByteStreamSource::nextCommand(SVGPathSegType)

 46 }
