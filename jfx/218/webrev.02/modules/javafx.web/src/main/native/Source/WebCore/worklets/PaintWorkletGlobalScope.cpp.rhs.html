 54 double PaintWorkletGlobalScope::devicePixelRatio() const
 55 {
 56     if (!responsibleDocument() || !responsibleDocument()->domWindow())
 57         return 1.0;
 58     return responsibleDocument()->domWindow()->devicePixelRatio();

 70 // https://drafts.css-houdini.org/css-paint-api/#registering-custom-paint
 71 ExceptionOr<void> PaintWorkletGlobalScope::registerPaint(JSC::JSGlobalObject& globalObject, const String& name, Strong<JSObject> paintConstructor)
 72 {
 73     auto& vm = paintConstructor->vm();
 74     JSC::JSLockHolder lock(vm);
 75     auto scope = DECLARE_THROW_SCOPE(vm);
 76 
 77     // Validate that paintConstructor is a VoidFunction
 78     CallData callData;
 79     if (JSC::getCallData(vm, paintConstructor.get(), callData) == JSC::CallType::None)
 80         return Exception { TypeError, "paintConstructor must be callable" };
 81 
 82     if (name.isEmpty())
 83         return Exception { TypeError, "The first argument must not be the empty string" };
 84 
 85     {
 86         auto locker = holdLock(paintDefinitionLock());
 87 
 88         if (paintDefinitionMap().contains(name))
 89             return Exception { InvalidModificationError, "This name has already been registered" };
 90 
 91         Vector<String> inputProperties;
 92 
 93         JSValue inputPropertiesIterableValue = paintConstructor->get(&globalObject, Identifier::fromString(vm, "inputProperties"));
 94         RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
 95 
 96         if (!inputPropertiesIterableValue.isUndefined())
 97             inputProperties = convert<IDLSequence<IDLDOMString>>(globalObject, inputPropertiesIterableValue);
 98         RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
 99 
100         // FIXME: Validate input properties here (step 7).
101 
102         Vector<String> inputArguments;
103 
104         JSValue inputArgumentsIterableValue = paintConstructor->get(&globalObject, Identifier::fromString(vm, "inputArguments"));
105         RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
106 
107         if (!inputArgumentsIterableValue.isUndefined())
108             inputArguments = convert<IDLSequence<IDLDOMString>>(globalObject, inputArgumentsIterableValue);
109         RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
110 
111         // FIXME: Parse syntax for inputArguments here (steps 11 and 12).
112 
113         JSValue contextOptionsValue = paintConstructor->get(&globalObject, Identifier::fromString(vm, "contextOptions"));
114         RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
115         UNUSED_PARAM(contextOptionsValue);
116 
117         // FIXME: Convert to PaintRenderingContext2DSettings here (step 14).
118 
119         if (!paintConstructor->isConstructor(vm))
120             return Exception { TypeError, "The second argument must be a constructor" };
121 
122         JSValue prototypeValue = paintConstructor->get(&globalObject, vm.propertyNames->prototype);
123         RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
124 
125         if (!prototypeValue.isObject())
126             return Exception { TypeError, "The second argument must have a prototype that is an object" };
127 
128         JSValue paintValue = prototypeValue.get(&globalObject, Identifier::fromString(vm, "paint"));
129         RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
130 
131         if (paintValue.isUndefined())
132             return Exception { TypeError, "The class must have a paint method" };
133 
134         RefPtr<JSCSSPaintCallback> paint = convert<IDLCallbackFunction<JSCSSPaintCallback>>(globalObject, paintValue, *jsCast<JSDOMGlobalObject*>(&globalObject));
135         RETURN_IF_EXCEPTION(scope, Exception { ExistingExceptionError });
136 
137         auto paintDefinition = makeUnique<PaintDefinition>(name, paintConstructor.get(), paint.releaseNonNull(), WTFMove(inputProperties), WTFMove(inputArguments));
138         paintDefinitionMap().add(name, WTFMove(paintDefinition));
139     }
140 
141     // This is for the case when we have already visited the paint definition map, and the GC is currently running in the background.
142     vm.heap.writeBarrier(&globalObject);
143 
144     // FIXME: construct documentDefinition (step 22).
145 
146     // FIXME: we should only repaint affected custom paint images <https://bugs.webkit.org/show_bug.cgi?id=192322>.
147     if (responsibleDocument() && responsibleDocument()->renderView())
148         responsibleDocument()->renderView()->repaintRootContents();
149 
150     return { };
151 }
