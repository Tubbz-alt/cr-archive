36 Ref<StyleCursorImage> StyleCursorImage::create(CSSCursorImageValue& cssValue)
37 {
38     return adoptRef(*new StyleCursorImage(cssValue));
39 }
40 
41 bool StyleCursorImage::operator==(const StyleImage& other) const
42 {
43     return is<StyleCursorImage>(other) && equals(downcast<StyleCursorImage>(other));
44 }
45 
46 Ref<CSSValue> StyleCursorImage::cssValue() const
47 {
48     return m_cssValue.copyRef();
49 }
50 
51 StyleCursorImage::StyleCursorImage(CSSCursorImageValue& cssValue)
52     : m_cssValue(cssValue)
53 {
54     m_isCursorImage = true;
55 }
56 
57 StyleCursorImage::~StyleCursorImage() = default;
58 
59 ImageWithScale StyleCursorImage::selectBestFitImage(const Document& document) const
60 {
61     return m_cssValue->selectBestFitImage(document);
62 }
63 
64 void StyleCursorImage::setContainerContextForRenderer(const RenderElement& renderer, const FloatSize& containerSize, float containerZoom)
65 {
66     if (!hasCachedImage())
67         return;
68     cachedImage()->setContainerContextForClient(renderer, LayoutSize(containerSize), containerZoom, m_cssValue->imageURL());
69 }
