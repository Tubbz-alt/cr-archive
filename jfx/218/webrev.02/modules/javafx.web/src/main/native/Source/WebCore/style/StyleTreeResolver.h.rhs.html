 48 DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(TreeResolverScope);
 49 class TreeResolver {
 50 public:
 51     TreeResolver(Document&);
 52     ~TreeResolver();
 53 
 54     std::unique_ptr<Update> resolve();
 55 
 56 private:
 57     std::unique_ptr<RenderStyle> styleForElement(Element&, const RenderStyle& inheritedStyle);
 58 
 59     void resolveComposedTree();
 60 
 61     ElementUpdates resolveElement(Element&);
 62 
 63     ElementUpdate createAnimatedElementUpdate(std::unique_ptr<RenderStyle>, Element&, Change);
 64     ElementUpdate resolvePseudoStyle(Element&, const ElementUpdate&, PseudoId);
 65 
 66     struct Scope : RefCounted<Scope> {
 67         WTF_MAKE_STRUCT_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(TreeResolverScope);
 68         Resolver& resolver;
 69         SelectorFilter selectorFilter;
 70         SharingResolver sharingResolver;
 71         ShadowRoot* shadowRoot { nullptr };
 72         Scope* enclosingScope { nullptr };
 73 
 74         Scope(Document&);
 75         Scope(ShadowRoot&, Scope& enclosingScope);
 76         ~Scope();
 77     };
 78 
 79     struct Parent {
 80         Element* element;
 81         const RenderStyle& style;
 82         Change change { NoChange };
 83         DescendantsToResolve descendantsToResolve { DescendantsToResolve::None };
 84         bool didPushScope { false };
 85 
 86         Parent(Document&);
 87         Parent(Element&, const RenderStyle&, Change, DescendantsToResolve);
 88     };
 89 
 90     Scope& scope() { return m_scopeStack.last(); }
 91     Parent& parent() { return m_parentStack.last(); }
 92 
 93     void pushScope(ShadowRoot&);
 94     void pushEnclosingScope();
 95     void popScope();
 96 
 97     void pushParent(Element&, const RenderStyle&, Change, DescendantsToResolve);
 98     void popParent();
 99     void popParentsToDepth(unsigned depth);
100 
101     const RenderStyle* parentBoxStyle() const;
102     const RenderStyle* parentBoxStyleForPseudo(const ElementUpdate&) const;
103 
104     Document& m_document;
105     std::unique_ptr<RenderStyle> m_documentElementStyle;
106 
107     Vector<Ref<Scope>, 4> m_scopeStack;
108     Vector<Parent, 32> m_parentStack;
109     bool m_didSeePendingStylesheet { false };
110 
111     std::unique_ptr<Update> m_update;
112 };
