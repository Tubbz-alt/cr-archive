29 #ifndef SincResampler_h
30 #define SincResampler_h
31 
32 #include "AudioArray.h"
33 #include "AudioSourceProvider.h"
34 #include <wtf/RefPtr.h>
35 
36 namespace WebCore {
37 
38 // SincResampler is a high-quality sample-rate converter.
39 
40 class SincResampler final {
41     WTF_MAKE_FAST_ALLOCATED;
42 public:
43     // scaleFactor == sourceSampleRate / destinationSampleRate
44     // kernelSize can be adjusted for quality (higher is better)
45     // numberOfKernelOffsets is used for interpolation and is the number of sub-sample kernel shifts.
46     SincResampler(double scaleFactor, unsigned kernelSize = 32, unsigned numberOfKernelOffsets = 32);
47 
48     // Processes numberOfSourceFrames from source to produce numberOfSourceFrames / scaleFactor frames in destination.
49     void process(const float* source, float* destination, unsigned numberOfSourceFrames);
50 
51     // Process with input source callback function for streaming applications.
52     void process(AudioSourceProvider*, float* destination, size_t framesToProcess);
53 
54 protected:
55     void initializeKernel();
56     void consumeSource(float* buffer, unsigned numberOfSourceFrames);
57 
58     double m_scaleFactor;
59     unsigned m_kernelSize;
60     unsigned m_numberOfKernelOffsets;
61 
62     // m_kernelStorage has m_numberOfKernelOffsets kernels back-to-back, each of size m_kernelSize.
63     // The kernel offsets are sub-sample shifts of a windowed sinc() shifted from 0.0 to 1.0 sample.
64     AudioFloatArray m_kernelStorage;
65 
66     // m_virtualSourceIndex is an index on the source input buffer with sub-sample precision.
67     // It must be double precision to avoid drift.
68     double m_virtualSourceIndex;
69 
70     // This is the number of destination frames we generate per processing pass on the buffer.
71     unsigned m_blockSize;
72 
73     // Source is copied into this buffer for each processing pass.
74     AudioFloatArray m_inputBuffer;
75 
76     const float* m_source;
77     unsigned m_sourceFramesAvailable;
78 
79     // m_sourceProvider is used to provide the audio input stream to the resampler.
80     AudioSourceProvider* m_sourceProvider;
81 
82     // The buffer is primed once at the very beginning of processing.
83     bool m_isBufferPrimed;
84 
85     RefPtr<AudioBus> m_internalBus;
86 };
87 
88 } // namespace WebCore
89 
90 #endif // SincResampler_h
