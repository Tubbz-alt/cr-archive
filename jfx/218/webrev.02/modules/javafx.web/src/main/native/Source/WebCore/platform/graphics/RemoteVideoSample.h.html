 30 #include "MediaSample.h"
 31 #include "RemoteVideoSample.h"
 32 #include <wtf/MachSendRight.h>
 33 #include <wtf/MediaTime.h>

 35 #if HAVE(IOSURFACE)
 36 #include "IOSurface.h"
 37 #endif
 38 
 39 typedef struct __CVBuffer* CVPixelBufferRef;
 40 
 41 namespace WebCore {
 42 
 43 class RemoteVideoSample {
 44 public:
 45     RemoteVideoSample() = default;
 46     RemoteVideoSample(RemoteVideoSample&&) = default;
 47     RemoteVideoSample& operator=(RemoteVideoSample&&) = default;
 48     ~RemoteVideoSample() = default;
 49 
 50 #if HAVE(IOSURFACE)
 51     WEBCORE_EXPORT static std::unique_ptr<RemoteVideoSample> create(MediaSample&);
 52     WEBCORE_EXPORT static std::unique_ptr<RemoteVideoSample> create(CVPixelBufferRef, MediaTime&& presentationTime, MediaSample::VideoRotation = MediaSample::VideoRotation::None);
 53     WEBCORE_EXPORT IOSurfaceRef surface();
 54 #endif
 55 
 56     const MediaTime& time() const { return m_time; }
 57     uint32_t videoFormat() const { return m_videoFormat; }
 58     IntSize size() const { return m_size; }
 59     MediaSample::VideoRotation rotation() const { return m_rotation; }
 60 
 61     template<class Encoder> void encode(Encoder& encoder) const
 62     {
 63 #if HAVE(IOSURFACE)
 64         if (m_ioSurface)
 65             encoder << m_ioSurface->createSendRight();
 66         else
 67             encoder << WTF::MachSendRight();
 68 #endif
 69         encoder << m_rotation;
 70         encoder << m_time;
 71         encoder << m_videoFormat;
 72         encoder << m_size;
 73         encoder << m_mirrored;
 74     }
 75 
 76     template<class Decoder> static bool decode(Decoder& decoder, RemoteVideoSample& sample)
 77     {
 78 #if HAVE(IOSURFACE)
 79         MachSendRight sendRight;
 80         if (!decoder.decode(sendRight))
 81             return false;
 82         sample.m_sendRight = WTFMove(sendRight);
 83 #endif
 84         MediaSample::VideoRotation rotation;
 85         if (!decoder.decode(rotation))
 86             return false;
 87         sample.m_rotation = rotation;
 88 
 89         MediaTime time;
 90         if (!decoder.decode(time))
 91             return false;
 92         sample.m_time = WTFMove(time);
 93 
 94         uint32_t format;
 95         if (!decoder.decode(format))
 96             return false;
 97         sample.m_videoFormat = format;
 98 
 99         IntSize size;
100         if (!decoder.decode(size))
101             return false;
102         sample.m_size = WTFMove(size);
103 
104         bool mirrored;
105         if (!decoder.decode(mirrored))
106             return false;
107         sample.m_mirrored = mirrored;
108 
109         return true;
110     }
111 
112 private:
113 
114 #if HAVE(IOSURFACE)
115     RemoteVideoSample(IOSurfaceRef, CGColorSpaceRef, MediaTime&&, MediaSample::VideoRotation, bool);
116 
117     std::unique_ptr<WebCore::IOSurface> m_ioSurface;
118     WTF::MachSendRight m_sendRight;
119 #endif
120     MediaSample::VideoRotation m_rotation { MediaSample::VideoRotation::None };
121     MediaTime m_time;
122     uint32_t m_videoFormat { 0 };
123     IntSize m_size;
124     bool m_mirrored { false };
125 };
126 
127 }

129 #endif // ENABLE(MEDIA_STREAM)
