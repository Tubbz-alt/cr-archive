40 namespace JSC {
41 
42 class Identifier;
43 class JSCell;
44 class SlotVisitor;
45 
46 class CacheableIdentifier {
47 public:
48     CacheableIdentifier() = default;
49     inline CacheableIdentifier(const Identifier&);
50     inline CacheableIdentifier(JSCell* identifier);
51 
52     CacheableIdentifier(const CacheableIdentifier&) = default;
53     CacheableIdentifier(CacheableIdentifier&&) = default;
54 
55     CacheableIdentifier(std::nullptr_t)
56         : m_bits(0)
57     { }
58 
59     bool isUid() const { return m_bits & s_uidTag; }
60     bool isCell() const { return !isUid(); }
61     inline bool isSymbolCell() const;
62     inline bool isStringCell() const;
63 
64     bool isSymbol() const { return m_bits && uid()->isSymbol(); }
65 
66     inline JSCell* cell() const;
67     UniquedStringImpl* uid() const;
68 
69     explicit operator bool() const { return m_bits; }
70 
71     CacheableIdentifier& operator=(const CacheableIdentifier&) = default;
72     CacheableIdentifier& operator=(CacheableIdentifier&&) = default;
73 
74     bool operator==(const CacheableIdentifier& other) const;
75     bool operator!=(const CacheableIdentifier& other) const;
76     bool operator==(const Identifier& other) const;
77 
78     static inline bool isCacheableIdentifierCell(JSCell*);
79     static inline bool isCacheableIdentifierCell(JSValue);
80 
81     inline void visitAggregate(SlotVisitor&) const;
82 
83     JS_EXPORT_PRIVATE void dump(PrintStream&) const;
84 
85 private:
86     inline void setCellBits(JSCell*);
87     inline void setUidBits(UniquedStringImpl*);
88 
89     // CacheableIdentifier can either hold a cell pointer or a uid. To discern which
90     // it is holding, we tag the low bit if we have a uid. We choose to tag the uid
91     // instead of the cell because this keeps the bits of the cell pointer form
92     // unpolluted, and therefore, it can be scanned by our conservative GC to keep the
93     // cell alive when the CacheableIdentifier is on the stack.
94     static constexpr uintptr_t s_uidTag = 1;
95     uintptr_t m_bits;
96 };
97 
98 } // namespace JSC
