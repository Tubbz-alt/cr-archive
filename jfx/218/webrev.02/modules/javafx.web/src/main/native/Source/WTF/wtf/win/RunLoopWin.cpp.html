 33 static const UINT PerformWorkMessage = WM_USER + 1;
 34 static const LPCWSTR kRunLoopMessageWindowClassName = L"RunLoopMessageWindow";
 35 
 36 LRESULT CALLBACK RunLoop::RunLoopWndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam)
 37 {
 38     if (RunLoop* runLoop = static_cast<RunLoop*>(getWindowPointer(hWnd, 0)))
 39         return runLoop->wndProc(hWnd, message, wParam, lParam);
 40 
 41     if (message == WM_CREATE) {
 42         LPCREATESTRUCT createStruct = reinterpret_cast<LPCREATESTRUCT>(lParam);
 43 
 44         // Associate the RunLoop with the window.
 45         setWindowPointer(hWnd, 0, createStruct->lpCreateParams);
 46         return 0;
 47     }
 48 
 49     return ::DefWindowProc(hWnd, message, wParam, lParam);
 50 }
 51 
 52 LRESULT RunLoop::wndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam)
 53 {
 54     switch (message) {
 55     case PerformWorkMessage:
 56         performWork();
 57         return 0;
 58     case WM_TIMER:
 59         bitwise_cast<RunLoop::TimerBase*>(wParam)->timerFired();
 60         return 0;
 61     }
 62 
 63     return ::DefWindowProc(hWnd, message, wParam, lParam);
 64 }
 65 
 66 void RunLoop::run()
 67 {
 68     MSG message;
 69     while (BOOL result = ::GetMessage(&message, 0, 0, 0)) {
 70         if (result == -1)
 71             break;
 72         ::TranslateMessage(&message);
 73         ::DispatchMessage(&message);
 74     }
 75 }
 76 
 77 void RunLoop::iterate()
 78 {
 79     MSG message;
 80     while (::PeekMessage(&message, 0, 0, 0, PM_REMOVE)) {
 81         ::TranslateMessage(&message);
 82         ::DispatchMessage(&message);
 83     }
 84 }
 85 
 86 void RunLoop::stop()
 87 {
 88     // RunLoop::stop() can be called from threads unrelated to this RunLoop.
 89     // We should post a message that call PostQuitMessage in RunLoop's thread.
 90     dispatch([] {
 91         ::PostQuitMessage(0);
 92     });
 93 }
 94 
 95 void RunLoop::registerRunLoopMessageWindowClass()
 96 {
 97     WNDCLASS windowClass = { };
 98     windowClass.lpfnWndProc     = RunLoop::RunLoopWndProc;
 99     windowClass.cbWndExtra      = sizeof(RunLoop*);
100     windowClass.lpszClassName   = kRunLoopMessageWindowClassName;
101     bool result = ::RegisterClass(&windowClass);
102     RELEASE_ASSERT(result);
103 }
104 
105 RunLoop::RunLoop()
106 {
107     m_runLoopMessageWindow = ::CreateWindow(kRunLoopMessageWindowClassName, 0, 0,
108         CW_USEDEFAULT, 0, CW_USEDEFAULT, 0, HWND_MESSAGE, 0, 0, this);
109     ASSERT(::IsWindow(m_runLoopMessageWindow));
110 }

116 void RunLoop::wakeUp()
117 {
118     // FIXME: No need to wake up the run loop if we've already called dispatch
119     // before the run loop has had the time to respond.
120     ::PostMessage(m_runLoopMessageWindow, PerformWorkMessage, reinterpret_cast<WPARAM>(this), 0);
121 }

123 RunLoop::CycleResult RunLoop::cycle(RunLoopMode)
124 {
125     MSG message;
126     if (!::GetMessage(&message, 0, 0, 0))
127         return CycleResult::Stop;
128 
129     ::TranslateMessage(&message);
130     ::DispatchMessage(&message);
131 
132     return CycleResult::Continue;
133 }

137 void RunLoop::TimerBase::timerFired()
138 {
139     {
140         LockHolder locker(m_runLoop->m_loopLock);
141 
142         if (!m_isActive)
143             return;
144 
145         if (!m_isRepeating) {
146             m_isActive = false;
147             ::KillTimer(m_runLoop->m_runLoopMessageWindow, bitwise_cast<uintptr_t>(this));
148         } else
149             m_nextFireDate = MonotonicTime::now() + m_interval;
150     }
151 
152     fired();
153 }
154 
155 RunLoop::TimerBase::TimerBase(RunLoop& runLoop)
156     : m_runLoop(runLoop)
157 {
158 }
159 
160 RunLoop::TimerBase::~TimerBase()
161 {
162     stop();
163 }
164 
165 void RunLoop::TimerBase::start(Seconds nextFireInterval, bool repeat)
166 {
167     LockHolder locker(m_runLoop->m_loopLock);
168     m_isRepeating = repeat;
169     m_isActive = true;
170     m_interval = nextFireInterval;
171     m_nextFireDate = MonotonicTime::now() + m_interval;
172     ::SetTimer(m_runLoop->m_runLoopMessageWindow, bitwise_cast<uintptr_t>(this), nextFireInterval.millisecondsAs<UINT>(), 0);
173 }
174 
175 void RunLoop::TimerBase::stop()
176 {
177     LockHolder locker(m_runLoop->m_loopLock);
178     if (!isActive(locker))
179         return;
180 
181     m_isActive = false;
182     ::KillTimer(m_runLoop->m_runLoopMessageWindow, bitwise_cast<uintptr_t>(this));
183 }
184 
185 bool RunLoop::TimerBase::isActive(const AbstractLocker&) const
186 {
187     return m_isActive;
188 }
189 
190 bool RunLoop::TimerBase::isActive() const
191 {
192     LockHolder locker(m_runLoop->m_loopLock);
193     return isActive(locker);
194 }
195 
196 Seconds RunLoop::TimerBase::secondsUntilFire() const
197 {
198     LockHolder locker(m_runLoop->m_loopLock);
199     if (isActive(locker))
200         return std::max<Seconds>(m_nextFireDate - MonotonicTime::now(), 0_s);
201     return 0_s;
202 }
