26 #pragma once
27 
28 #include "ArrayBuffer.h"
29 #include "InternalFunction.h"
30 
31 namespace JSC {
32 
33 class JSArrayBufferPrototype;
34 class GetterSetter;
35 
36 template<ArrayBufferSharingMode sharingMode>
37 class JSGenericArrayBufferConstructor final : public InternalFunction {
38 public:
39     using Base = InternalFunction;
40 
41 protected:
42     JSGenericArrayBufferConstructor(VM&, Structure*);
43     void finishCreation(VM&, JSArrayBufferPrototype*, GetterSetter* speciesSymbol);
44 
45     static EncodedJSValue JSC_HOST_CALL constructArrayBuffer(JSGlobalObject*, CallFrame*);
46 
47 public:
48     static JSGenericArrayBufferConstructor* create(VM& vm, Structure* structure, JSArrayBufferPrototype* prototype, GetterSetter* speciesSymbol)
49     {
50         JSGenericArrayBufferConstructor* result =
51             new (NotNull, allocateCell<JSGenericArrayBufferConstructor>(vm.heap)) JSGenericArrayBufferConstructor(vm, structure);
52         result->finishCreation(vm, prototype, speciesSymbol);
53         return result;
54     }
55 
56     static Structure* createStructure(VM&, JSGlobalObject*, JSValue prototype);
57 
58     static const ClassInfo s_info; // This is never accessed directly, since that would break linkage on some compilers.
59     static const ClassInfo* info();
60 };
61 
62 using JSArrayBufferConstructor = JSGenericArrayBufferConstructor<ArrayBufferSharingMode::Default>;
63 using JSSharedArrayBufferConstructor = JSGenericArrayBufferConstructor<ArrayBufferSharingMode::Shared>;
64 STATIC_ASSERT_ISO_SUBSPACE_SHARABLE(JSArrayBufferConstructor, InternalFunction);
65 STATIC_ASSERT_ISO_SUBSPACE_SHARABLE(JSSharedArrayBufferConstructor, InternalFunction);
66 
67 } // namespace JSC
