 35 namespace WebCore {

 36 namespace Layout {
 37 
 38 class Box;
 39 class FloatingState;
 40 enum class StyleDiff;
 41 
 42 class FormattingState {
 43     WTF_MAKE_ISO_ALLOCATED(FormattingState);
 44 public:


 45     FloatingState& floatingState() const { return m_floatingState; }
 46 
 47     void markNeedsLayout(const Box&, StyleDiff);
 48     bool needsLayout(const Box&);
 49 
 50     void setIntrinsicWidthConstraintsForBox(const Box&,  FormattingContext::IntrinsicWidthConstraints);
 51     Optional<FormattingContext::IntrinsicWidthConstraints> intrinsicWidthConstraintsForBox(const Box&) const;
 52     void clearIntrinsicWidthConstraints(const Box&);
 53 
 54     void setIntrinsicWidthConstraints(FormattingContext::IntrinsicWidthConstraints intrinsicWidthConstraints) { m_intrinsicWidthConstraints = intrinsicWidthConstraints; }
 55     Optional<FormattingContext::IntrinsicWidthConstraints> intrinsicWidthConstraints() const { return m_intrinsicWidthConstraints; }
 56 
 57     bool isBlockFormattingState() const { return m_type == Type::Block; }
 58     bool isInlineFormattingState() const { return m_type == Type::Inline; }
 59     bool isTableFormattingState() const { return m_type == Type::Table; }
 60 
 61     LayoutState& layoutState() const { return m_layoutState; }
 62 
 63     Display::Box& displayBox(const Box& layoutBox);
 64     // Since we layout the out-of-flow boxes at the end of the formatting context layout, it's okay to store them in the formatting state -as opposed to the containing block level.
 65     using OutOfFlowBoxList = Vector<WeakPtr<const Box>>;
 66     void addOutOfFlowBox(const Box& outOfFlowBox) { m_outOfFlowBoxes.append(makeWeakPtr(outOfFlowBox)); }
 67     void removeOutOfFlowBox(const Box&);
 68     const OutOfFlowBoxList& outOfFlowBoxes() const { return m_outOfFlowBoxes; }
 69 
 70 protected:
 71     enum class Type { Block, Inline, Table };
 72     FormattingState(Ref<FloatingState>&&, Type, LayoutState&);
 73     ~FormattingState();
 74 
 75 private:
 76     LayoutState& m_layoutState;
 77     Ref<FloatingState> m_floatingState;
 78     HashMap<const Box*, FormattingContext::IntrinsicWidthConstraints> m_intrinsicWidthConstraintsForBoxes;
 79     Optional<FormattingContext::IntrinsicWidthConstraints> m_intrinsicWidthConstraints;
 80     // FIXME: This needs WeakListHashSet
 81     OutOfFlowBoxList m_outOfFlowBoxes;
 82     Type m_type;
 83 };
 84 
 85 inline void FormattingState::setIntrinsicWidthConstraintsForBox(const Box& layoutBox, FormattingContext::IntrinsicWidthConstraints intrinsicWidthConstraints)
 86 {
 87     ASSERT(!m_intrinsicWidthConstraintsForBoxes.contains(&layoutBox));
 88     ASSERT(&m_layoutState.formattingStateForBox(layoutBox) == this);
 89     m_intrinsicWidthConstraintsForBoxes.set(&layoutBox, intrinsicWidthConstraints);
 90 }
 91 
 92 inline void FormattingState::clearIntrinsicWidthConstraints(const Box& layoutBox)
 93 {
 94     m_intrinsicWidthConstraints = { };
 95     m_intrinsicWidthConstraintsForBoxes.remove(&layoutBox);
 96 }
 97 
 98 inline Optional<FormattingContext::IntrinsicWidthConstraints> FormattingState::intrinsicWidthConstraintsForBox(const Box& layoutBox) const
 99 {
100     ASSERT(&m_layoutState.formattingStateForBox(layoutBox) == this);
101     auto iterator = m_intrinsicWidthConstraintsForBoxes.find(&layoutBox);
102     if (iterator == m_intrinsicWidthConstraintsForBoxes.end())
103         return { };
104     return iterator->value;
105 }
106 
107 }
108 }
