 38 struct FormDataElement {
 39     struct EncodedFileData;
 40     struct EncodedBlobData;
 41     using Data = Variant<Vector<char>, EncodedFileData, EncodedBlobData>;
 42 
 43     FormDataElement() = default;
 44     explicit FormDataElement(Data&& data)
 45         : data(WTFMove(data)) { }
 46     explicit FormDataElement(Vector<char>&& array)
 47         : data(WTFMove(array)) { }
 48     FormDataElement(const String& filename, int64_t fileStart, int64_t fileLength, Optional<WallTime> expectedFileModificationTime)
 49         : data(EncodedFileData { filename, fileStart, fileLength, expectedFileModificationTime }) { }
 50     explicit FormDataElement(const URL& blobURL)
 51         : data(EncodedBlobData { blobURL }) { }
 52 
 53     uint64_t lengthInBytes(const Function<uint64_t(const URL&)>&) const;
 54     uint64_t lengthInBytes() const;
 55 
 56     FormDataElement isolatedCopy() const;
 57 
 58     template<typename Encoder> void encode(Encoder& encoder) const
 59     {
 60         encoder << data;
 61     }
 62     template<typename Decoder> static Optional<FormDataElement> decode(Decoder& decoder)
 63     {
 64         Optional<Data> data;
 65         decoder >> data;
 66         if (!data)
 67             return WTF::nullopt;
 68         return FormDataElement(WTFMove(*data));
 69     }
 70 
 71     struct EncodedFileData {
 72         String filename;
 73         int64_t fileStart { 0 };
 74         int64_t fileLength { 0 };
 75         Optional<WallTime> expectedFileModificationTime;
 76 
 77         bool fileModificationTimeMatchesExpectation() const;
 78 
 79         EncodedFileData isolatedCopy() const
 80         {
 81             return { filename.isolatedCopy(), fileStart, fileLength, expectedFileModificationTime };
 82         }
 83 
 84         bool operator==(const EncodedFileData& other) const
 85         {
 86             return filename == other.filename
 87                 && fileStart == other.fileStart
 88                 && fileLength == other.fileLength
 89                 && expectedFileModificationTime == other.expectedFileModificationTime;
 90         }
 91         template<typename Encoder> void encode(Encoder& encoder) const
 92         {
 93             encoder << filename << fileStart << fileLength << expectedFileModificationTime;
 94         }
 95         template<typename Decoder> static Optional<EncodedFileData> decode(Decoder& decoder)
 96         {
 97             Optional<String> filename;
 98             decoder >> filename;
 99             if (!filename)
100                 return WTF::nullopt;
101 
102             Optional<int64_t> fileStart;
103             decoder >> fileStart;
104             if (!fileStart)
105                 return WTF::nullopt;
106 
107             Optional<int64_t> fileLength;
108             decoder >> fileLength;
109             if (!fileLength)
110                 return WTF::nullopt;
111 
112             Optional<Optional<WallTime>> expectedFileModificationTime;
113             decoder >> expectedFileModificationTime;
114             if (!expectedFileModificationTime)
115                 return WTF::nullopt;
116 
117             return {{
118                 WTFMove(*filename),
119                 WTFMove(*fileStart),
120                 WTFMove(*fileLength),
121                 WTFMove(*expectedFileModificationTime)
122             }};
123         }
124 
125     };
126 
127     struct EncodedBlobData {
128         URL url;
129 
130         bool operator==(const EncodedBlobData& other) const
131         {
132             return url == other.url;
133         }
134         template<typename Encoder> void encode(Encoder& encoder) const
135         {
136             encoder << url;
137         }
138         template<typename Decoder> static Optional<EncodedBlobData> decode(Decoder& decoder)
139         {
140             Optional<URL> url;
141             decoder >> url;
142             if (!url)
143                 return WTF::nullopt;
144 
145             return {{ WTFMove(*url) }};
146         }
147     };
148 
149     bool operator==(const FormDataElement& other) const
150     {
151         if (&other == this)
152             return true;
153         if (data.index() != other.data.index())
154             return false;
155         if (!data.index())
156             return WTF::get<0>(data) == WTF::get<0>(other.data);
157         if (data.index() == 1)
158             return WTF::get<1>(data) == WTF::get<1>(other.data);
159         return WTF::get<2>(data) == WTF::get<2>(other.data);
160     }
161     bool operator!=(const FormDataElement& other) const
162     {
163         return !(*this == other);
164     }
165 
166     Data data;
167 };

185 class FormData : public RefCounted<FormData> {
186 public:
187     enum EncodingType {
188         FormURLEncoded, // for application/x-www-form-urlencoded
189         TextPlain, // for text/plain
190         MultipartFormData // for multipart/form-data
191     };
192 
193     WEBCORE_EXPORT static Ref<FormData> create();
194     WEBCORE_EXPORT static Ref<FormData> create(const void*, size_t);
195     static Ref<FormData> create(const CString&);
196     static Ref<FormData> create(Vector<char>&&);
197     static Ref<FormData> create(const Vector<char>&);
198     static Ref<FormData> create(const Vector<uint8_t>&);
199     static Ref<FormData> create(const DOMFormData&, EncodingType = FormURLEncoded);
200     static Ref<FormData> createMultiPart(const DOMFormData&);
201     WEBCORE_EXPORT ~FormData();
202 
203     // FIXME: Both these functions perform a deep copy of m_elements, but differ in handling of other data members.
204     // How much of that is intentional? We need better names that explain the difference.
205     Ref<FormData> copy() const;
206     WEBCORE_EXPORT Ref<FormData> isolatedCopy() const;
207 
208     template<typename Encoder>
209     void encode(Encoder&) const;
210     template<typename Decoder>
211     static RefPtr<FormData> decode(Decoder&);
212 
213     WEBCORE_EXPORT void appendData(const void* data, size_t);
214     void appendFile(const String& filePath);
215     WEBCORE_EXPORT void appendFileRange(const String& filename, long long start, long long length, Optional<WallTime> expectedModificationTime);
216     WEBCORE_EXPORT void appendBlob(const URL& blobURL);
217 
218     WEBCORE_EXPORT Vector<char> flatten() const; // omits files
219     String flattenToString() const; // omits files
220 
221     // Resolve all blob references so we only have file and data.
222     // If the FormData has no blob references to resolve, this is returned.
223     WEBCORE_EXPORT Ref<FormData> resolveBlobReferences(BlobRegistryImpl* = nullptr);
224 
225     WEBCORE_EXPORT FormDataForUpload prepareForUpload();
226 
227     bool isEmpty() const { return m_elements.isEmpty(); }
228     const Vector<FormDataElement>& elements() const { return m_elements; }
229     const Vector<char>& boundary() const { return m_boundary; }
230 
231     RefPtr<SharedBuffer> asSharedBuffer() const;
232 
233     bool alwaysStream() const { return m_alwaysStream; }
234     void setAlwaysStream(bool alwaysStream) { m_alwaysStream = alwaysStream; }
235 
236     // Identifies a particular form submission instance.  A value of 0 is used
237     // to indicate an unspecified identifier.
238     void setIdentifier(int64_t identifier) { m_identifier = identifier; }
239     int64_t identifier() const { return m_identifier; }
240 
241     bool containsPasswordData() const { return m_containsPasswordData; }
242     void setContainsPasswordData(bool containsPasswordData) { m_containsPasswordData = containsPasswordData; }
243 
244     static EncodingType parseEncodingType(const String& type)
245     {
246         if (equalLettersIgnoringASCIICase(type, "text/plain"))
247             return TextPlain;
248         if (equalLettersIgnoringASCIICase(type, "multipart/form-data"))
249             return MultipartFormData;
250         return FormURLEncoded;
251     }
252 
253     uint64_t lengthInBytes() const;
254 
255     WEBCORE_EXPORT URL asBlobURL() const;
256 
257 private:
258     FormData();
259     FormData(const FormData&);
260 
261     void appendMultiPartFileValue(const File&, Vector<char>& header, TextEncoding&);
262     void appendMultiPartStringValue(const String&, Vector<char>& header, TextEncoding&);
263     void appendMultiPartKeyValuePairItems(const DOMFormData&);
264     void appendNonMultiPartKeyValuePairItems(const DOMFormData&, EncodingType);
265 
266     Vector<FormDataElement> m_elements;
267 
268     int64_t m_identifier { 0 };
269     bool m_alwaysStream { false };
270     Vector<char> m_boundary;
271     bool m_containsPasswordData { false };
272     mutable Optional<uint64_t> m_lengthInBytes;
273 };

275 inline bool operator==(const FormData& a, const FormData& b)
276 {
277     return a.elements() == b.elements();
278 }
279 
280 inline bool operator!=(const FormData& a, const FormData& b)
281 {
282     return !(a == b);
283 }

285 template<typename Encoder>
286 void FormData::encode(Encoder& encoder) const
287 {
288     encoder << m_alwaysStream;
289     encoder << m_boundary;
290     encoder << m_elements;
291     encoder << m_identifier;
292     // FIXME: Does not encode m_containsPasswordData. Why is that OK?
293 }
294 
295 template<typename Decoder>
296 RefPtr<FormData> FormData::decode(Decoder& decoder)
297 {
298     auto data = FormData::create();
299 
300     if (!decoder.decode(data->m_alwaysStream))
301         return nullptr;
302 
303     if (!decoder.decode(data->m_boundary))
304         return nullptr;
305 
306     if (!decoder.decode(data->m_elements))
307         return nullptr;
308 
309     if (!decoder.decode(data->m_identifier))
310         return nullptr;
311 
312     return data;
313 }
