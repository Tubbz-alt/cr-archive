60 void InspectorInstrumentationObject::finishCreation(VM& vm, JSGlobalObject*)
61 {
62     Base::finishCreation(vm);
63     ASSERT(inherits(vm, info()));
64     putDirectWithoutTransition(vm, vm.propertyNames->isEnabled, jsBoolean(false));
65 }
66 
67 bool InspectorInstrumentationObject::isEnabled(VM& vm) const
68 {
69     return getDirect(vm, vm.propertyNames->isEnabled).asBoolean();
70 }
71 
72 void InspectorInstrumentationObject::enable(VM& vm)
73 {
74     putDirect(vm, vm.propertyNames->isEnabled, jsBoolean(true));
75 }
76 
77 void InspectorInstrumentationObject::disable(VM& vm)
78 {
79     putDirect(vm, vm.propertyNames->isEnabled, jsBoolean(false));
80 }

84 EncodedJSValue JSC_HOST_CALL inspectorInstrumentationObjectLog(JSGlobalObject* globalObject, CallFrame* callFrame)
85 {
86     VM& vm = globalObject->vm();
87     auto scope = DECLARE_THROW_SCOPE(vm);
88     JSValue target = callFrame->argument(0);
89     String value = target.toWTFString(globalObject);
90     RETURN_IF_EXCEPTION(scope, encodedJSValue());
91     dataLog(value, "\n");
92     return JSValue::encode(jsUndefined());
93 }
