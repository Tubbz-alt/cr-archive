26 #include "config.h"
27 #include <wtf/DebugHeap.h>
28 
29 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
30 
31 #include <cstdlib>
32 #include <thread>
33 
34 namespace WTF {
35 
36 DebugHeap::DebugHeap(const char* heapName)
37     : m_zone(malloc_create_zone(0, 0))
38 {
39     malloc_set_zone_name(m_zone, heapName);
40 }
41 
42 void* DebugHeap::malloc(size_t size)
43 {
44     void* result = malloc_zone_malloc(m_zone, size);
45     if (!result)
46         CRASH();
47     return result;
48 }
49 
50 void* DebugHeap::calloc(size_t numElements, size_t elementSize)
51 {
52     void* result = malloc_zone_calloc(m_zone, numElements, elementSize);
53     if (!result)
54         CRASH();
55     return result;
56 }
57 
58 void* DebugHeap::memalign(size_t alignment, size_t size, bool crashOnFailure)
59 {
60     void* result = malloc_zone_memalign(m_zone, alignment, size);
61     if (!result && crashOnFailure)
62         CRASH();
63     return result;
64 }
65 
66 void* DebugHeap::realloc(void* object, size_t size)
67 {
68     void* result = malloc_zone_realloc(m_zone, object, size);
69     if (!result)
70         CRASH();
71     return result;
72 }
73 
74 void DebugHeap::free(void* object)
75 {
76     malloc_zone_free(m_zone, object);
77 }
78 
79 } // namespace WTF
80 
81 #endif // ENABLE(MALLOC_HEAP_BREAKDOWN)
