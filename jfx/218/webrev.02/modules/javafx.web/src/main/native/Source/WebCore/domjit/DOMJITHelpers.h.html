 57 template<typename WrappedNode>
 58 JSC::EncodedJSValue JIT_OPERATION toWrapperSlow(JSC::JSGlobalObject* globalObject, void* result)
 59 {
 60     ASSERT(result);
 61     ASSERT(globalObject);
 62     JSC::VM& vm = globalObject->vm();
 63     JSC::CallFrame* callFrame = DECLARE_CALL_FRAME(vm);
 64     JSC::JITOperationPrologueCallFrameTracer tracer(vm, callFrame);
 65     return JSC::JSValue::encode(toJS(globalObject, static_cast<JSDOMGlobalObject*>(globalObject), *static_cast<WrappedNode*>(result)));
 66 }

 68 template<typename WrappedType>
 69 void tryLookUpWrapperCache(CCallHelpers& jit, CCallHelpers::JumpList& failureCases, GPRReg wrapped, GPRReg resultGPR)
 70 {
 71     jit.loadPtr(CCallHelpers::Address(wrapped, ScriptWrappable::offsetOfWrapper<WrappedType>()), resultGPR);
 72     failureCases.append(jit.branchTestPtr(CCallHelpers::Zero, resultGPR));
 73     failureCases.append(branchIfNotWeakIsLive(jit, resultGPR));
 74     jit.loadPtr(CCallHelpers::Address(resultGPR, JSC::WeakImpl::offsetOfJSValue() + JSC::JSValue::offsetOfPayload()), resultGPR);
 75 }

 77 template<typename WrappedType, typename ToJSFunction>
 78 void toWrapper(CCallHelpers& jit, JSC::SnippetParams& params, GPRReg wrapped, GPRReg globalObject, JSValueRegs result, ToJSFunction function, JSC::JSValue globalObjectConstant)
 79 {
 80     ASSERT(wrapped != result.payloadGPR());
 81     ASSERT(globalObject != result.payloadGPR());
 82     GPRReg payloadGPR = result.payloadGPR();
 83     CCallHelpers::JumpList slowCases;
 84 
 85     if (globalObjectConstant) {
 86         if (!JSC::jsCast<JSDOMGlobalObject*>(globalObjectConstant)->worldIsNormal()) {
 87             slowCases.append(jit.jump());
 88             params.addSlowPathCall(slowCases, jit, function, result, globalObject, wrapped);
 89             return;
 90         }
 91     } else
 92         slowCases.append(branchIfNotWorldIsNormal(jit, globalObject));
 93 
 94     tryLookUpWrapperCache<WrappedType>(jit, slowCases, wrapped, payloadGPR);
 95     jit.boxCell(payloadGPR, result);
 96     params.addSlowPathCall(slowCases, jit, function, result, globalObject, wrapped);
 97 }

 99 inline CCallHelpers::Jump branchIfDOMWrapper(CCallHelpers& jit, GPRReg target)
100 {
101     return jit.branch8(
102         CCallHelpers::AboveOrEqual,
103         CCallHelpers::Address(target, JSC::JSCell::typeInfoTypeOffset()),
104         CCallHelpers::TrustedImm32(JSC::JSType(JSDOMWrapperType)));
105 }

112         CCallHelpers::TrustedImm32(JSC::JSType(JSDOMWrapperType)));
113 }
