27 
28 #include "GPRInfo.h"
29 
30 namespace JSC {
31 
32 struct EntryFrame;
33 class ExecState;

34 class JSObject;
35 class VM;
36 
37 struct VMEntryRecord {
38     /*
39      * This record stored in a vmEntryTo{JavaScript,Host} allocated frame. It is allocated on the stack
40      * after callee save registers where local variables would go.
41      */
42     VM* m_vm;
43     ExecState* m_prevTopCallFrame;
44     EntryFrame* m_prevTopEntryFrame;
45     JSObject* m_callee;
46 
47     JSObject* callee() const { return m_callee; }
48 
49 #if !ENABLE(C_LOOP) && NUMBER_OF_CALLEE_SAVES_REGISTERS > 0
50     CPURegister calleeSaveRegistersBuffer[NUMBER_OF_CALLEE_SAVES_REGISTERS];
51 #elif ENABLE(C_LOOP)
52     CPURegister calleeSaveRegistersBuffer[1];
53 #endif
54 
55     ExecState* prevTopCallFrame() { return m_prevTopCallFrame; }
56     SUPPRESS_ASAN ExecState* unsafePrevTopCallFrame() { return m_prevTopCallFrame; }
57 
58     EntryFrame* prevTopEntryFrame() { return m_prevTopEntryFrame; }
59     SUPPRESS_ASAN EntryFrame* unsafePrevTopEntryFrame() { return m_prevTopEntryFrame; }
60 };
61 
62 extern "C" VMEntryRecord* vmEntryRecord(EntryFrame*);
63 
64 } // namespace JSC
