 41 RealtimeOutgoingAudioSource::RealtimeOutgoingAudioSource(Ref<MediaStreamTrackPrivate>&& source)
 42     : m_audioSource(WTFMove(source))
 43 {
 44 }
 45 
 46 RealtimeOutgoingAudioSource::~RealtimeOutgoingAudioSource()
 47 {
 48 ASSERT(!m_audioSource->hasObserver(*this));
 49 #if ASSERT_ENABLED
 50     auto locker = holdLock(m_sinksLock);
 51 #endif
 52     ASSERT(m_sinks.isEmpty());
 53 
 54     stop();
 55 }
 56 
 57 void RealtimeOutgoingAudioSource::observeSource()
 58 {
 59     ASSERT(!m_audioSource->hasObserver(*this));
 60     m_audioSource->addObserver(*this);
 61     initializeConverter();
 62 }
 63 
 64 void RealtimeOutgoingAudioSource::unobserveSource()
 65 {
 66     m_audioSource->removeObserver(*this);
 67 }
 68 
 69 void RealtimeOutgoingAudioSource::setSource(Ref<MediaStreamTrackPrivate>&& newSource)
 70 {
 71     ALWAYS_LOG("Changing source to ", newSource->logIdentifier());


 72 
 73     ASSERT(!m_audioSource->hasObserver(*this));

 74     m_audioSource = WTFMove(newSource);



 75     sourceUpdated();


 76 }
 77 
 78 void RealtimeOutgoingAudioSource::initializeConverter()
 79 {
 80     m_muted = m_audioSource->muted();
 81     m_enabled = m_audioSource->enabled();
 82 }
 83 
 84 void RealtimeOutgoingAudioSource::sourceMutedChanged()
 85 {
 86     m_muted = m_audioSource->muted();
 87 }
 88 
 89 void RealtimeOutgoingAudioSource::sourceEnabledChanged()
 90 {
 91     m_enabled = m_audioSource->enabled();
 92 }
 93 
 94 void RealtimeOutgoingAudioSource::AddSink(webrtc::AudioTrackSinkInterface* sink)
 95 {

 96     auto locker = holdLock(m_sinksLock);
 97     m_sinks.add(sink);






 98 }
 99 
100 void RealtimeOutgoingAudioSource::RemoveSink(webrtc::AudioTrackSinkInterface* sink)
101 {

102     auto locker = holdLock(m_sinksLock);
103     m_sinks.remove(sink);




104 }
105 
106 void RealtimeOutgoingAudioSource::sendAudioFrames(const void* audioData, int bitsPerSample, int sampleRate, size_t numberOfChannels, size_t numberOfFrames)
107 {
108 #if !RELEASE_LOG_DISABLED
109     if (!(++m_chunksSent % 200))
110         ALWAYS_LOG(LOGIDENTIFIER, "chunk ", m_chunksSent);
111 #endif
112 
113     auto locker = holdLock(m_sinksLock);
114     for (auto sink : m_sinks)
115         sink->OnData(audioData, bitsPerSample, sampleRate, numberOfChannels, numberOfFrames);
116 }
