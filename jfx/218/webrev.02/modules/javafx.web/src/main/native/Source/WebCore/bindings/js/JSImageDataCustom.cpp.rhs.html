40 JSValue toJSNewlyCreated(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, Ref<ImageData>&& imageData)
41 {
42     VM& vm = lexicalGlobalObject->vm();
43     auto* data = imageData->data();
44     auto* wrapper = createWrapper<ImageData>(globalObject, WTFMove(imageData));
45     Identifier dataName = Identifier::fromString(vm, "data");
46     wrapper->putDirect(vm, dataName, toJS(lexicalGlobalObject, globalObject, data), PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly);
47     // FIXME: Adopt reportExtraMemoryVisited, and switch to reportExtraMemoryAllocated.
48     // https://bugs.webkit.org/show_bug.cgi?id=142595
49     vm.heap.deprecatedReportExtraMemory(data->length());
50 
51     return wrapper;
52 }
53 
54 JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, ImageData& imageData)
55 {
56     return wrap(lexicalGlobalObject, globalObject, imageData);
57 }
