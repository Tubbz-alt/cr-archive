36 WebGLDepthTexture::WebGLDepthTexture(WebGLRenderingContextBase& context)
37     : WebGLExtension(context)
38 {
39 }
40 
41 WebGLDepthTexture::~WebGLDepthTexture() = default;
42 
43 WebGLExtension::ExtensionName WebGLDepthTexture::getName() const
44 {
45     return WebGLDepthTextureName;
46 }
47 
48 bool WebGLDepthTexture::supported(GraphicsContext3D& context)
49 {
50     Extensions3D& extensions = context.getExtensions();
51     return extensions.supports("GL_OES_depth_texture")
52         || extensions.supports("GL_ARB_depth_texture");
53 }
