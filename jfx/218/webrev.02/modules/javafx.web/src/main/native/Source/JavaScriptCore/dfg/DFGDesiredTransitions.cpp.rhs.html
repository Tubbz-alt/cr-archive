37 DesiredTransition::DesiredTransition(CodeBlock* codeBlock, CodeBlock* codeOriginOwner, Structure* oldStructure, Structure* newStructure)
38     : m_codeBlock(codeBlock)
39     , m_codeOriginOwner(codeOriginOwner)
40     , m_oldStructure(oldStructure)
41     , m_newStructure(newStructure)
42 {
43 }
44 
45 void DesiredTransition::reallyAdd(VM& vm, CommonData* common)
46 {
47     ConcurrentJSLocker locker(m_codeBlock->m_lock);
48     common->transitions.append(
49         WeakReferenceTransition(
50             vm, m_codeBlock,
51             m_codeOriginOwner,
52             m_oldStructure, m_newStructure));
53 }
54 
55 void DesiredTransition::visitChildren(SlotVisitor& visitor)
56 {
57     visitor.appendUnbarriered(m_codeOriginOwner);
58     visitor.appendUnbarriered(m_oldStructure);
59     visitor.appendUnbarriered(m_newStructure);
60 }
61 
62 DesiredTransitions::DesiredTransitions()
63 {
64 }
65 
66 DesiredTransitions::~DesiredTransitions()
67 {
68 }
69 
70 void DesiredTransitions::addLazily(CodeBlock* codeBlock, CodeBlock* codeOriginOwner, Structure* oldStructure, Structure* newStructure)
71 {
72     m_transitions.append(DesiredTransition(codeBlock, codeOriginOwner, oldStructure, newStructure));
73 }
74 
75 void DesiredTransitions::reallyAdd(VM& vm, CommonData* common)
76 {
77     for (unsigned i = 0; i < m_transitions.size(); i++)
78         m_transitions[i].reallyAdd(vm, common);
79 }
80 
81 void DesiredTransitions::visitChildren(SlotVisitor& visitor)
82 {
83     for (unsigned i = 0; i < m_transitions.size(); i++)
84         m_transitions[i].visitChildren(visitor);
85 }
