35 namespace JSC {
36 class SlotVisitor;

39 namespace WebCore {
40 
41 class ScriptExecutionContext;
42 
43 // A base class that prevents binding callbacks from executing when
44 // active dom objects are stopped or suspended.
45 //
46 // Should only be created, used, and destroyed on the script execution
47 // context thread.
48 class ActiveDOMCallback : public ContextDestructionObserver {
49 public:
50     ActiveDOMCallback(ScriptExecutionContext*);
51     virtual ~ActiveDOMCallback();
52 
53     WEBCORE_EXPORT bool canInvokeCallback() const;
54 
55     virtual void visitJSFunction(JSC::SlotVisitor&) { }
56 };
57 
58 } // namespace WebCore
