 36 namespace WebCore {

 38 class Document;
 39 class Element;
 40 class VisiblePosition;
 41 
 42 class TextManipulationController : public CanMakeWeakPtr<TextManipulationController> {
 43     WTF_MAKE_FAST_ALLOCATED;
 44 public:
 45     TextManipulationController(Document&);
 46 
 47     enum TokenIdentifierType { };
 48     using TokenIdentifier = ObjectIdentifier<TokenIdentifierType>;
 49 
 50     struct ManipulationToken {
 51         TokenIdentifier identifier;
 52         String content;
 53         bool isExcluded { false };
 54 
 55         template<class Encoder> void encode(Encoder&) const;
 56         template<class Decoder> static Optional<ManipulationToken> decode(Decoder&);
 57     };
 58 
 59     enum ItemIdentifierType { };
 60     using ItemIdentifier = ObjectIdentifier<ItemIdentifierType>;
 61 
 62     struct ExclusionRule {
 63         enum class Type : uint8_t { Exclude, Include };
 64 
 65         struct ElementRule {
 66             AtomString localName;
 67 
 68             template<class Encoder> void encode(Encoder&) const;
 69             template<class Decoder> static Optional<ElementRule> decode(Decoder&);
 70         };
 71 
 72         struct AttributeRule {
 73             AtomString name;
 74             String value;
 75 
 76             template<class Encoder> void encode(Encoder&) const;
 77             template<class Decoder> static Optional<AttributeRule> decode(Decoder&);
 78         };
 79 
 80         struct ClassRule {
 81             AtomString className;
 82 
 83             template<class Encoder> void encode(Encoder&) const;
 84             template<class Decoder> static Optional<ClassRule> decode(Decoder&);
 85         };
 86 
 87         Type type;
 88         WTF::Variant<ElementRule, AttributeRule, ClassRule> rule;
 89 
 90         bool match(const Element&) const;
 91 
 92         template<class Encoder> void encode(Encoder&) const;
 93         template<class Decoder> static Optional<ExclusionRule> decode(Decoder&);
 94     };
 95 
 96     using ManipulationItemCallback = WTF::Function<void(Document&, ItemIdentifier, const Vector<ManipulationToken>&)>;
 97     WEBCORE_EXPORT void startObservingParagraphs(ManipulationItemCallback&&, Vector<ExclusionRule>&& = { });
 98 
 99     void didCreateRendererForElement(Element&);
100 
101     enum class ManipulationResult : uint8_t {
102         Success,
103         ContentChanged,
104         InvalidItem,
105         InvalidToken,
106         ExclusionViolation,
107     };
108     WEBCORE_EXPORT ManipulationResult completeManipulation(ItemIdentifier, const Vector<ManipulationToken>&);
109 
110 private:
111     void observeParagraphs(VisiblePosition& start, VisiblePosition& end);
112     void scheduleObservartionUpdate();
113 
114     struct ManipulationItem {
115         Position start;
116         Position end;
117         Vector<ManipulationToken> tokens;
118     };
119 
120     void addItem(const Position& startOfParagraph, const Position& endOfParagraph, Vector<ManipulationToken>&&);
121     ManipulationResult replace(const ManipulationItem&, const Vector<ManipulationToken>&);
122 
123     WeakPtr<Document> m_document;
124     WeakHashSet<Element> m_mutatedElements;
125     WeakHashSet<Element> m_recentlyInsertedElements;
126     ManipulationItemCallback m_callback;
127     Vector<ExclusionRule> m_exclusionRules;
128     HashMap<ItemIdentifier, ManipulationItem> m_items;
129     ItemIdentifier m_itemIdentifier;
130     TokenIdentifier m_tokenIdentifier;
131 };
132 
133 template<class Encoder>
134 void TextManipulationController::ManipulationToken::encode(Encoder& encoder) const
135 {
136     encoder << identifier << content << isExcluded;
137 }
138 
139 template<class Decoder>
140 Optional<TextManipulationController::ManipulationToken> TextManipulationController::ManipulationToken::decode(Decoder& decoder)
141 {
142     ManipulationToken result;
143     if (!decoder.decode(result.identifier))
144         return WTF::nullopt;
145     if (!decoder.decode(result.content))
146         return WTF::nullopt;
147     if (!decoder.decode(result.isExcluded))
148         return WTF::nullopt;
149     return result;
150 }
151 
152 template<class Encoder>
153 void TextManipulationController::ExclusionRule::encode(Encoder& encoder) const
154 {
155     encoder << type << rule;
156 }
157 
158 template<class Decoder>
159 Optional<TextManipulationController::ExclusionRule> TextManipulationController::ExclusionRule::decode(Decoder& decoder)
160 {
161     ExclusionRule result;
162     if (!decoder.decode(result.type))
163         return WTF::nullopt;
164     if (!decoder.decode(result.rule))
165         return WTF::nullopt;
166     return result;
167 }
168 
169 template<class Encoder>
170 void TextManipulationController::ExclusionRule::ElementRule::encode(Encoder& encoder) const
171 {
172     encoder << localName;
173 }
174 
175 template<class Decoder>
176 Optional<TextManipulationController::ExclusionRule::ElementRule> TextManipulationController::ExclusionRule::ElementRule::decode(Decoder& decoder)
177 {
178     ElementRule result;
179     if (!decoder.decode(result.localName))
180         return WTF::nullopt;
181     return result;
182 }
183 
184 template<class Encoder>
185 void TextManipulationController::ExclusionRule::AttributeRule::encode(Encoder& encoder) const
186 {
187     encoder << name << value;
188 }
189 
190 template<class Decoder>
191 Optional<TextManipulationController::ExclusionRule::AttributeRule> TextManipulationController::ExclusionRule::AttributeRule::decode(Decoder& decoder)
192 {
193     AttributeRule result;
194     if (!decoder.decode(result.name))
195         return WTF::nullopt;
196     if (!decoder.decode(result.value))
197         return WTF::nullopt;
198     return result;
199 }
200 
201 template<class Encoder>
202 void TextManipulationController::ExclusionRule::ClassRule::encode(Encoder& encoder) const
203 {
204     encoder << className;
205 }
206 
207 template<class Decoder>
208 Optional<TextManipulationController::ExclusionRule::ClassRule> TextManipulationController::ExclusionRule::ClassRule::decode(Decoder& decoder)
209 {
210     ClassRule result;
211     if (!decoder.decode(result.className))
212         return WTF::nullopt;
213     return result;
214 }
