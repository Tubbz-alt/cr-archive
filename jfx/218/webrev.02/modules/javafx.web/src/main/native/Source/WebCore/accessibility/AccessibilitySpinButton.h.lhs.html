26 #pragma once
27 
28 #include "AccessibilityMockObject.h"
29 #include "SpinButtonElement.h"
30 
31 namespace WebCore {
32 
33 class AccessibilitySpinButton final : public AccessibilityMockObject {
34 public:
35     static Ref<AccessibilitySpinButton> create();
36     virtual ~AccessibilitySpinButton();
37 
38     void setSpinButtonElement(SpinButtonElement* spinButton) { m_spinButtonElement = spinButton; }
39 
40     AccessibilityObject* incrementButton();
41     AccessibilityObject* decrementButton();
42 
43     void step(int amount);
44 
45 private:
46     AccessibilitySpinButton();
47 
48     AccessibilityRole roleValue() const override { return AccessibilityRole::SpinButton; }
49     bool isSpinButton() const override { return true; }
50     bool isNativeSpinButton() const override { return true; }
51     void addChildren() override;
52     LayoutRect elementRect() const override;
53 
54     SpinButtonElement* m_spinButtonElement;
55 };
56 
57 class AccessibilitySpinButtonPart final : public AccessibilityMockObject {
58 public:
59     static Ref<AccessibilitySpinButtonPart> create();
60     virtual ~AccessibilitySpinButtonPart() = default;
61 
62     bool isIncrementor() const { return m_isIncrementor; }
63     void setIsIncrementor(bool value) { m_isIncrementor = value; }
64 
65 private:
66     AccessibilitySpinButtonPart();
67 
68     bool press() override;
69     AccessibilityRole roleValue() const override { return AccessibilityRole::Button; }
70     bool isSpinButtonPart() const override { return true; }
71     LayoutRect elementRect() const override;
72 
73     unsigned m_isIncrementor : 1;
74 };
75 
76 } // namespace WebCore
77 
78 SPECIALIZE_TYPE_TRAITS_ACCESSIBILITY(AccessibilitySpinButton, isNativeSpinButton())
79 SPECIALIZE_TYPE_TRAITS_ACCESSIBILITY(AccessibilitySpinButtonPart, isSpinButtonPart())
