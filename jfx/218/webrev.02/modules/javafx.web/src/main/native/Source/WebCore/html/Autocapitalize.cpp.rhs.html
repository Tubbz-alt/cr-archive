33 namespace WebCore {
34 
35 AutocapitalizeType autocapitalizeTypeForAttributeValue(const AtomString& attributeValue)
36 {
37     // Omitted / missing values are the Default state.
38     if (attributeValue.isEmpty())
39         return AutocapitalizeTypeDefault;
40 
41     if (equalLettersIgnoringASCIICase(attributeValue, "on") || equalLettersIgnoringASCIICase(attributeValue, "sentences"))
42         return AutocapitalizeTypeSentences;
43     if (equalLettersIgnoringASCIICase(attributeValue, "off") || equalLettersIgnoringASCIICase(attributeValue, "none"))
44         return AutocapitalizeTypeNone;
45     if (equalLettersIgnoringASCIICase(attributeValue, "words"))
46         return AutocapitalizeTypeWords;
47     if (equalLettersIgnoringASCIICase(attributeValue, "characters"))
48         return AutocapitalizeTypeAllCharacters;
49 
50     // Unrecognized values fall back to "on".
51     return AutocapitalizeTypeSentences;
52 }
53 
54 const AtomString& stringForAutocapitalizeType(AutocapitalizeType type)
55 {
56     switch (type) {
57     case AutocapitalizeTypeDefault:
58         return nullAtom();
59     case AutocapitalizeTypeNone: {
60         static NeverDestroyed<const AtomString> valueNone("none", AtomString::ConstructFromLiteral);
61         return valueNone;
62     }
63     case AutocapitalizeTypeSentences: {
64         static NeverDestroyed<const AtomString> valueSentences("sentences", AtomString::ConstructFromLiteral);
65         return valueSentences;
66     }
67     case AutocapitalizeTypeWords: {
68         static NeverDestroyed<const AtomString> valueWords("words", AtomString::ConstructFromLiteral);
69         return valueWords;
70     }
71     case AutocapitalizeTypeAllCharacters: {
72         static NeverDestroyed<const AtomString> valueAllCharacters("characters", AtomString::ConstructFromLiteral);
73         return valueAllCharacters;
74     }
75     }
76 
77     ASSERT_NOT_REACHED();
78     return nullAtom();
79 }
