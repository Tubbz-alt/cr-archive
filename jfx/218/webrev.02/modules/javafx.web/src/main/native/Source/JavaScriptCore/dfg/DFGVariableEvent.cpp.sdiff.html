37 void VariableEvent::dump(PrintStream& out) const
38 {
39     switch (kind()) {
40     case Reset:
41         out.printf("Reset");
42         break;
43     case BirthToFill:
44         dumpFillInfo("BirthToFill", out);
45         break;
46     case BirthToSpill:
47         dumpSpillInfo("BirthToSpill", out);
48         break;
49     case Birth:
50         out.print("Birth(", id(), ")");
51         break;
52     case Fill:
53         dumpFillInfo("Fill", out);
54         break;
55     case Spill:
56         dumpSpillInfo("Spill", out);
57         break;
58     case Death:
59         out.print("Death(", id(), ")");
60         break;
61     case MovHintEvent:
62         out.print("MovHint(", id(), ", ", bytecodeRegister(), ")");
63         break;
64     case SetLocalEvent:
65         out.print(
66             "SetLocal(machine:", machineRegister(), " -> bytecode:", bytecodeRegister(),
67             ", ", dataFormatToString(dataFormat()), ")");
68         break;
69     default:
70         RELEASE_ASSERT_NOT_REACHED();
71         break;
72     }
73 }

75 void VariableEvent::dumpFillInfo(const char* name, PrintStream& out) const
76 {
77     out.print(name, "(", id(), ", ");
78     if (dataFormat() == DataFormatDouble)
79         out.printf("%s", FPRInfo::debugName(fpr()));
80 #if USE(JSVALUE32_64)
81     else if (dataFormat() & DataFormatJS)
82         out.printf("%s:%s", GPRInfo::debugName(tagGPR()), GPRInfo::debugName(payloadGPR()));
83 #endif
84     else
85         out.printf("%s", GPRInfo::debugName(gpr()));
86     out.printf(", %s)", dataFormatToString(dataFormat()));
