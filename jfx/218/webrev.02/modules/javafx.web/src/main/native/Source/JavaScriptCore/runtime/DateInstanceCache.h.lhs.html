34 namespace JSC {
35 
36 class DateInstanceData : public RefCounted<DateInstanceData> {
37 public:
38     static Ref<DateInstanceData> create() { return adoptRef(*new DateInstanceData); }
39 
40     double m_gregorianDateTimeCachedForMS;





41     GregorianDateTime m_cachedGregorianDateTime;
42     double m_gregorianDateTimeUTCCachedForMS;
43     GregorianDateTime m_cachedGregorianDateTimeUTC;
44 
45 private:
46     DateInstanceData()
47         : m_gregorianDateTimeCachedForMS(PNaN)
48         , m_gregorianDateTimeUTCCachedForMS(PNaN)
49     {
50     }
51 };
52 
53 class DateInstanceCache {
54 public:
55     DateInstanceCache()
56     {
57         reset();
58     }
59 
60     void reset()
61     {
62         for (size_t i = 0; i < cacheSize; ++i)
63             m_cache[i].key = PNaN;
64     }
65 
66     DateInstanceData* add(double d)
67     {
68         CacheEntry& entry = lookup(d);
69         if (d == entry.key)
70             return entry.value.get();
71 
72         entry.key = d;
73         entry.value = DateInstanceData::create();
74         return entry.value.get();
75     }
76 
77 private:
78     static const size_t cacheSize = 16;
79 
80     struct CacheEntry {
81         double key;
82         RefPtr<DateInstanceData> value;
83     };
84 
85     CacheEntry& lookup(double d) { return m_cache[WTF::FloatHash<double>::hash(d) & (cacheSize - 1)]; }
86 
87     std::array<CacheEntry, cacheSize> m_cache;
88 };
