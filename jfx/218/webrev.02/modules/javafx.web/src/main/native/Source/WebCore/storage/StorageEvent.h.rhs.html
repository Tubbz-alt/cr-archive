33 class Storage;

35 class StorageEvent final : public Event {
36     WTF_MAKE_ISO_ALLOCATED(StorageEvent);
37 public:
38     static Ref<StorageEvent> create(const AtomString& type, const String& key, const String& oldValue, const String& newValue, const String& url, Storage* storageArea);
39     static Ref<StorageEvent> createForBindings();
40 
41     struct Init : EventInit {
42         String key;
43         String oldValue;
44         String newValue;
45         String url;
46         RefPtr<Storage> storageArea;
47     };
48 
49     static Ref<StorageEvent> create(const AtomString&, const Init&, IsTrusted = IsTrusted::No);
50     virtual ~StorageEvent();
51 
52     const String& key() const { return m_key; }
53     const String& oldValue() const { return m_oldValue; }
54     const String& newValue() const { return m_newValue; }
55     const String& url() const { return m_url; }
56     Storage* storageArea() const { return m_storageArea.get(); }
57 
58     void initStorageEvent(const AtomString& type, bool canBubble, bool cancelable, const String& key, const String& oldValue, const String& newValue, const String& url, Storage* storageArea);
59 
60     // Needed once we support init<blank>EventNS

62 
63     EventInterface eventInterface() const override;
64 
65 private:
66     StorageEvent();
67     StorageEvent(const AtomString& type, const String& key, const String& oldValue, const String& newValue, const String& url, Storage* storageArea);
68     StorageEvent(const AtomString&, const Init&, IsTrusted);
69 
70     String m_key;
71     String m_oldValue;
72     String m_newValue;
73     String m_url;
74     RefPtr<Storage> m_storageArea;
75 };
