28 #if ENABLE(WEBGL)
29 
30 #include "ContextDestructionObserver.h"
31 #include "WebGLSharedObject.h"
32 #include <wtf/HashMap.h>
33 #include <wtf/HashFunctions.h>
34 #include <wtf/Lock.h>
35 #include <wtf/Vector.h>
36 
37 namespace WebCore {
38 
39 class ScriptExecutionContext;
40 class WebGLRenderingContextBase;
41 class WebGLShader;
42 
43 class WebGLProgram final : public WebGLSharedObject, public ContextDestructionObserver {
44 public:
45     static Ref<WebGLProgram> create(WebGLRenderingContextBase&);
46     virtual ~WebGLProgram();
47 
48     static HashMap<WebGLProgram*, WebGLRenderingContextBase*>& instances(const WTF::LockHolder&);
49     static Lock& instancesMutex();
50 
51     void contextDestroyed() final;
52 
53     unsigned numActiveAttribLocations();
54     GCGLint getActiveAttribLocation(GCGLuint index);
55 
56     bool isUsingVertexAttrib0();
57 
58     bool getLinkStatus();
59     void setLinkStatus(bool);
60 
61     unsigned getLinkCount() const { return m_linkCount; }
62 
63     // This is to be called everytime after the program is successfully linked.
64     // We don't deal with integer overflow here, assuming in reality a program
65     // will never be linked so many times.
66     // Also, we invalidate the cached program info.
67     void increaseLinkCount();
68 
69     WebGLShader* getAttachedShader(GCGLenum);
70     bool attachShader(WebGLShader*);
71     bool detachShader(WebGLShader*);
72 
73 protected:
74     WebGLProgram(WebGLRenderingContextBase&);
75 
76     void deleteObjectImpl(GraphicsContextGLOpenGL*, PlatformGLObject) override;
77 
78 private:
79     void cacheActiveAttribLocations(GraphicsContextGLOpenGL*);
80     void cacheInfoIfNeeded();
81 
82     Vector<GCGLint> m_activeAttribLocations;
83 
84     GCGLint m_linkStatus { 0 };
85 
86     // This is used to track whether a WebGLUniformLocation belongs to this program or not.
87     unsigned m_linkCount { 0 };
88 
89     RefPtr<WebGLShader> m_vertexShader;
90     RefPtr<WebGLShader> m_fragmentShader;
91 
92     bool m_infoValid { true };
93 };
94 
95 } // namespace WebCore
