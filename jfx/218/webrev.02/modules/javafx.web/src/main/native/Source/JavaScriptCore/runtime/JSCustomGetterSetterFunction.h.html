26 #pragma once
27 
28 #include "JSFunction.h"
29 
30 namespace JSC {
31 
32 class CustomGetterSetter;
33 
34 class JSCustomGetterSetterFunction final : public JSFunction {
35 public:
36     typedef JSFunction Base;
37 
38     // The Type is set to the number of arguments the resultant function will have.
39     enum class Type { Getter = 0, Setter = 1 };
40 
41     static constexpr unsigned StructureFlags = Base::StructureFlags;
42 
43     template<typename CellType, SubspaceAccess mode>
44     static IsoSubspace* subspaceFor(VM& vm)
45     {
46         return vm.customGetterSetterFunctionSpace<mode>();
47     }
48 
49     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
50     {
51         ASSERT(globalObject);
52         return Structure::create(vm, globalObject, prototype, TypeInfo(JSFunctionType, StructureFlags), info());
53     }
54 
55     static JSCustomGetterSetterFunction* create(VM&, JSGlobalObject*, CustomGetterSetter*, const Type, const PropertyName&);
56 
57     DECLARE_EXPORT_INFO;
58 
59 protected:
60     static void visitChildren(JSCell*, SlotVisitor&);
61 
62 private:
63     JSCustomGetterSetterFunction(VM&, NativeExecutable*, JSGlobalObject*, Structure*, Type, const PropertyName&);
64     void finishCreation(VM&, NativeExecutable*, CustomGetterSetter*, const String&);
65 
66     static EncodedJSValue JSC_HOST_CALL customGetterSetterFunctionCall(JSGlobalObject*, CallFrame*);
67 
68     CustomGetterSetter* customGetterSetter() const { return m_getterSetter.get(); }
69     bool isSetter() const { return m_type == Type::Setter; }
70     const PropertyName& propertyName() const { return m_propertyName; }
71 
72     WriteBarrier<CustomGetterSetter> m_getterSetter;
73     Type m_type;
74     PropertyName m_propertyName;
75 };
76 
77 } // namespace JSC
