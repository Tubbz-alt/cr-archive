 69 #if ENABLE(CONTEXT_MENUS)
 70 class FrontendMenuProvider : public ContextMenuProvider {
 71 public:
 72     static Ref<FrontendMenuProvider> create(InspectorFrontendHost* frontendHost, Deprecated::ScriptObject frontendApiObject, const Vector<ContextMenuItem>& items)
 73     {
 74         return adoptRef(*new FrontendMenuProvider(frontendHost, frontendApiObject, items));
 75     }
 76 
 77     void disconnect()
 78     {
 79         m_frontendApiObject = { };
 80         m_frontendHost = nullptr;
 81     }
 82 
 83 private:
 84     FrontendMenuProvider(InspectorFrontendHost* frontendHost, Deprecated::ScriptObject frontendApiObject, const Vector<ContextMenuItem>& items)
 85         : m_frontendHost(frontendHost)
 86         , m_frontendApiObject(frontendApiObject)
 87         , m_items(items)
 88     {
 89     }
 90 
 91     ~FrontendMenuProvider() override
 92     {
 93         contextMenuCleared();
 94     }
 95 
 96     void populateContextMenu(ContextMenu* menu) override
 97     {
 98         for (auto& item : m_items)
 99             menu->appendItem(item);
100     }
101 
102     void contextMenuItemSelected(ContextMenuAction action, const String&) override
103     {
104         if (m_frontendHost) {
105             UserGestureIndicator gestureIndicator(ProcessingUserGesture);
106             int itemNumber = action - ContextMenuItemBaseCustomTag;
107 
108             Deprecated::ScriptFunctionCall function(m_frontendApiObject, "contextMenuItemSelected", WebCore::functionCallHandlerFromAnyThread);
109             function.appendArgument(itemNumber);
110             function.call();
111         }
112     }
113 
114     void contextMenuCleared() override
115     {
116         if (m_frontendHost) {
117             Deprecated::ScriptFunctionCall function(m_frontendApiObject, "contextMenuCleared", WebCore::functionCallHandlerFromAnyThread);
118             function.call();
119 
120             m_frontendHost->m_menuProvider = nullptr;
121         }
122         m_items.clear();
123     }
124 
125     InspectorFrontendHost* m_frontendHost;
126     Deprecated::ScriptObject m_frontendApiObject;
127     Vector<ContextMenuItem> m_items;
128 };
129 #endif

155 void InspectorFrontendHost::addSelfToGlobalObjectInWorld(DOMWrapperWorld& world)
156 {
157     auto& lexicalGlobalObject = *execStateFromPage(world, m_frontendPage);
158     auto& vm = lexicalGlobalObject.vm();
159     JSC::JSLockHolder lock(vm);
160     auto scope = DECLARE_CATCH_SCOPE(vm);
161 
162     auto& globalObject = *JSC::jsCast<JSDOMGlobalObject*>(&lexicalGlobalObject);
163     globalObject.putDirect(vm, JSC::Identifier::fromString(vm, "InspectorFrontendHost"), toJS<IDLInterface<InspectorFrontendHost>>(lexicalGlobalObject, globalObject, *this));
164     if (UNLIKELY(scope.exception()))
165         reportException(&lexicalGlobalObject, scope.exception());
166 }

355 void InspectorFrontendHost::killText(const String& text, bool shouldPrependToKillRing, bool shouldStartNewSequence)
356 {
357     if (!m_frontendPage)
358         return;
359 
360     Editor& editor = m_frontendPage->focusController().focusedOrMainFrame().editor();
361     editor.setStartNewKillRingSequence(shouldStartNewSequence);
362     Editor::KillRingInsertionMode insertionMode = shouldPrependToKillRing ? Editor::KillRingInsertionMode::PrependText : Editor::KillRingInsertionMode::AppendText;
363     editor.addTextToKillRing(text, insertionMode);
364 }

406 static void populateContextMenu(Vector<InspectorFrontendHost::ContextMenuItem>&& items, ContextMenu& menu)
407 {
408     for (auto& item : items) {
409         if (item.type == "separator") {
410             menu.appendItem({ SeparatorType, ContextMenuItemTagNoAction, { } });
411             continue;
412         }
413 
414         if (item.type == "subMenu" && item.subItems) {
415             ContextMenu subMenu;
416             populateContextMenu(WTFMove(*item.subItems), subMenu);
417 
418             menu.appendItem({ SubmenuType, ContextMenuItemTagNoAction, item.label, &subMenu });
419             continue;
420         }
421 
422         auto type = item.type == "checkbox" ? CheckableActionType : ActionType;
423         auto action = static_cast<ContextMenuAction>(ContextMenuItemBaseCustomTag + item.id.valueOr(0));
424         ContextMenuItem menuItem = { type, action, item.label };
425         if (item.enabled)
426             menuItem.setEnabled(*item.enabled);
427         if (item.checked)
428             menuItem.setChecked(*item.checked);
429         menu.appendItem(menuItem);
430     }
431 }
432 #endif

434 void InspectorFrontendHost::showContextMenu(Event& event, Vector<ContextMenuItem>&& items)
435 {
436 #if ENABLE(CONTEXT_MENUS)
437     ASSERT(m_frontendPage);
438 
439     auto& lexicalGlobalObject = *execStateFromPage(debuggerWorld(), m_frontendPage);
440     auto& vm = lexicalGlobalObject.vm();
441     auto value = lexicalGlobalObject.get(&lexicalGlobalObject, JSC::Identifier::fromString(vm, "InspectorFrontendAPI"));
442     ASSERT(value);
443     ASSERT(value.isObject());
444     auto* frontendAPIObject = asObject(value);
445 
446     ContextMenu menu;
447     populateContextMenu(WTFMove(items), menu);
448 
449     auto menuProvider = FrontendMenuProvider::create(this, { &lexicalGlobalObject, frontendAPIObject }, menu.items());
450     m_menuProvider = menuProvider.ptr();
451     m_frontendPage->contextMenuController().showContextMenu(event, menuProvider);
452 #else
453     UNUSED_PARAM(event);
454     UNUSED_PARAM(items);
455 #endif
456 }

458 void InspectorFrontendHost::dispatchEventAsContextMenuEvent(Event& event)
459 {
460 #if ENABLE(CONTEXT_MENUS) && USE(ACCESSIBILITY_CONTEXT_MENUS)
461     if (!is<MouseEvent>(event))
462         return;
463 
464     auto& mouseEvent = downcast<MouseEvent>(event);
465     auto& frame = *downcast<Node>(mouseEvent.target())->document().frame();
466     m_frontendPage->contextMenuController().showContextMenuAt(frame, roundedIntPoint(mouseEvent.absoluteLocation()));
467 #else
468     UNUSED_PARAM(event);
469 #endif
470 }

521 bool InspectorFrontendHost::showCertificate(const String& serializedCertificate)
522 {
523     if (!m_client)
524         return false;
525 
526     Vector<uint8_t> data;
527     if (!base64Decode(serializedCertificate, data))
528         return false;
529 
530     CertificateInfo certificateInfo;
531     WTF::Persistence::Decoder decoder(data.data(), data.size());
532     if (!decoder.decode(certificateInfo))
533         return false;
534 
535     if (certificateInfo.isEmpty())
536         return false;
537 
538     m_client->showCertificate(certificateInfo);
539     return true;
540 }

557 static Optional<DiagnosticLoggingClient::ValuePayload> valuePayloadFromJSONValue(const RefPtr<JSON::Value>& value)
558 {
559     switch (value->type()) {
560     case JSON::Value::Type::Array:
561     case JSON::Value::Type::Null:
562     case JSON::Value::Type::Object:
563         ASSERT_NOT_REACHED();
564         return WTF::nullopt;
565 
566     case JSON::Value::Type::Boolean:
567         bool boolValue;
568         value->asBoolean(boolValue);
569         return DiagnosticLoggingClient::ValuePayload(boolValue);
570 
571     case JSON::Value::Type::Double:
572         double doubleValue;
573         value->asDouble(doubleValue);
574         return DiagnosticLoggingClient::ValuePayload(doubleValue);
575 
576     case JSON::Value::Type::Integer:
577         long long intValue;
578         value->asInteger(intValue);
579         return DiagnosticLoggingClient::ValuePayload(intValue);
580 
581     case JSON::Value::Type::String:
582         String stringValue;
583         value->asString(stringValue);
584         return DiagnosticLoggingClient::ValuePayload(stringValue);
585     }
586 
587     ASSERT_NOT_REACHED();
588     return WTF::nullopt;
589 }

591 void InspectorFrontendHost::logDiagnosticEvent(const String& eventName, const String& payloadString)
592 {
593     if (!supportsDiagnosticLogging())
594         return;
595 
596     RefPtr<JSON::Value> payloadValue;
597     if (!JSON::Value::parseJSON(payloadString, payloadValue))
598         return;
599 
600     RefPtr<JSON::Object> payloadObject;
601     if (!payloadValue->asObject(payloadObject))
602         return;
603 
604     DiagnosticLoggingClient::ValueDictionary dictionary;
605     for (const auto& [key, value] : *payloadObject) {
606         if (auto valuePayload = valuePayloadFromJSONValue(value))
607             dictionary.set(key, WTFMove(valuePayload.value()));
608     }
609 
610     m_client->logDiagnosticEvent(makeString("WebInspector."_s, eventName), dictionary);
611 }
