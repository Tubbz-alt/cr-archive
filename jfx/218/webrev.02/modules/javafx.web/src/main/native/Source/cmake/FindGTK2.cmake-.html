32 find_package(PkgConfig)
33 pkg_check_modules(GTK2 gtk+-2.0)
34 
35 set(VERSION_OK TRUE)
36 if (GTK2_VERSION)
37     if (GTK2_FIND_VERSION_EXACT)
38         if (NOT("${GTK2_FIND_VERSION}" VERSION_EQUAL "${GTK2_VERSION}"))
39             set(VERSION_OK FALSE)
40         endif ()
41     else ()
42         if ("${GTK2_VERSION}" VERSION_LESS "${GTK2_FIND_VERSION}")
43             set(VERSION_OK FALSE)
44         endif ()
45     endif ()
46 endif ()
47 
48 include(FindPackageHandleStandardArgs)
49 FIND_PACKAGE_HANDLE_STANDARD_ARGS(GTK2 DEFAULT_MSG GTK2_INCLUDE_DIRS GTK2_LIBRARIES VERSION_OK)
