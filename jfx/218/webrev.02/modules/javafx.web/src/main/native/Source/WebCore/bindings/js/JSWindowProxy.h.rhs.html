58 
59     DECLARE_INFO;
60

63     void setWindow(AbstractDOMWindow&);
64 
65     WindowProxy* windowProxy() const;
66 
67     AbstractDOMWindow& wrapped() const;
68     static WEBCORE_EXPORT WindowProxy* toWrapped(JSC::VM&, JSC::JSValue);
69 
70     DOMWrapperWorld& world() { return m_world; }
71 
72     void attachDebugger(JSC::Debugger*);
73 
74 private:
75     JSWindowProxy(JSC::VM&, JSC::Structure&, DOMWrapperWorld&);
76     void finishCreation(JSC::VM&, AbstractDOMWindow&);
77     static JSC::IsoSubspace* subspaceForImpl(JSC::VM&);
78 
79     Ref<DOMWrapperWorld> m_world;
80 };

82 // JSWindowProxy is a little odd in that it's not a traditional wrapper and has no back pointer.
83 // It is, however, strongly owned by AbstractFrame via its WindowProxy, so we can get one from a WindowProxy.
84 WEBCORE_EXPORT JSC::JSValue toJS(JSC::JSGlobalObject*, WindowProxy&);
85 inline JSC::JSValue toJS(JSC::JSGlobalObject* lexicalGlobalObject, WindowProxy* windowProxy) { return windowProxy ? toJS(lexicalGlobalObject, *windowProxy) : JSC::jsNull(); }
86 inline JSC::JSValue toJS(JSC::JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject*, WindowProxy& windowProxy) { return toJS(lexicalGlobalObject, windowProxy); }
87 inline JSC::JSValue toJS(JSC::JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, WindowProxy* windowProxy) { return windowProxy ? toJS(lexicalGlobalObject, globalObject, *windowProxy) : JSC::jsNull(); }
88 
89 JSWindowProxy* toJSWindowProxy(WindowProxy&, DOMWrapperWorld&);
90 inline JSWindowProxy* toJSWindowProxy(WindowProxy* windowProxy, DOMWrapperWorld& world) { return windowProxy ? toJSWindowProxy(*windowProxy, world) : nullptr; }
