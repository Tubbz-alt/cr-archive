 69 bool Extensions3DOpenGLES::isEnabled(const String& name)
 70 {
 71     // Return false immediately if the extension is not supported by the drivers.
 72     bool enabled = Extensions3DOpenGLCommon::isEnabled(name);
 73     if (!enabled)
 74         return false;
 75 
 76     // For GL_EXT_robustness, check that the context supports robust access.
 77     if (name == "GL_EXT_robustness") {
 78         GLint robustAccess = GL_FALSE;
 79         m_context->getIntegerv(Extensions3D::CONTEXT_ROBUST_ACCESS, &robustAccess);
 80         return robustAccess == GL_TRUE;
 81     }
 82 
 83     return true;
 84 }

130 Platform3DObject Extensions3DOpenGLES::createVertexArrayOES()
131 {
132     m_context->makeContextCurrent();
133     if (m_glGenVertexArraysOES) {
134         GLuint array = 0;
135         m_glGenVertexArraysOES(1, &array);
136         return array;
137     }
138 
139     m_context->synthesizeGLError(GL_INVALID_OPERATION);
140     return 0;
141 }

183 int Extensions3DOpenGLES::getGraphicsResetStatusARB()
184 {
185     // FIXME: This does not call getGraphicsResetStatusARB, but instead getGraphicsResetStatusEXT.
186     // The return codes from the two extensions are identical and their purpose is the same, so it
187     // may be best to rename getGraphicsResetStatusARB() to getGraphicsResetStatus().
188     if (m_contextResetStatus != GL_NO_ERROR)
189         return m_contextResetStatus;
190     if (m_glGetGraphicsResetStatusEXT) {
191         m_context->makeContextCurrent();
192         int reasonForReset = m_glGetGraphicsResetStatusEXT();
193         if (reasonForReset != GL_NO_ERROR) {
194             ASSERT(m_contextLostCallback);
195             if (m_contextLostCallback)
196                 m_contextLostCallback->onContextLost();
197             m_contextResetStatus = reasonForReset;
198         }
199         return reasonForReset;
200     }
201 
202     m_context->synthesizeGLError(GL_INVALID_OPERATION);
203     return false;
204 }

206 void Extensions3DOpenGLES::readnPixelsEXT(int x, int y, GC3Dsizei width, GC3Dsizei height, GC3Denum format, GC3Denum type, GC3Dsizei bufSize, void *data)
207 {
208     if (m_glReadnPixelsEXT) {
209         m_context->makeContextCurrent();
210         // FIXME: remove the two glFlush calls when the driver bug is fixed, i.e.,
211         // all previous rendering calls should be done before reading pixels.
212         ::glFlush();
213 
214         // FIXME: If non-BlackBerry platforms use this, they will need to implement
215         // their anti-aliasing code here.
216         m_glReadnPixelsEXT(x, y, width, height, format, type, bufSize, data);
217         return;
218     }
219 
220     m_context->synthesizeGLError(GL_INVALID_OPERATION);
221 }

278 bool Extensions3DOpenGLES::supportsExtension(const String& name)
279 {
280     if (m_availableExtensions.contains(name)) {
281         if (!m_supportsOESvertexArrayObject && name == "GL_OES_vertex_array_object") {
282             m_glBindVertexArrayOES = reinterpret_cast<PFNGLBINDVERTEXARRAYOESPROC>(eglGetProcAddress("glBindVertexArrayOES"));
283             m_glGenVertexArraysOES = reinterpret_cast<PFNGLGENVERTEXARRAYSOESPROC>(eglGetProcAddress("glGenVertexArraysOES"));
284             m_glDeleteVertexArraysOES = reinterpret_cast<PFNGLDELETEVERTEXARRAYSOESPROC>(eglGetProcAddress("glDeleteVertexArraysOES"));
285             m_glIsVertexArrayOES = reinterpret_cast<PFNGLISVERTEXARRAYOESPROC>(eglGetProcAddress("glIsVertexArrayOES"));
286             m_supportsOESvertexArrayObject = true;
287         } else if (!m_supportsIMGMultisampledRenderToTexture && name == "GL_IMG_multisampled_render_to_texture") {
288             m_glFramebufferTexture2DMultisampleIMG = reinterpret_cast<PFNGLFRAMEBUFFERTEXTURE2DMULTISAMPLEIMGPROC>(eglGetProcAddress("glFramebufferTexture2DMultisampleIMG"));
289             m_glRenderbufferStorageMultisampleIMG = reinterpret_cast<PFNGLRENDERBUFFERSTORAGEMULTISAMPLEIMGPROC>(eglGetProcAddress("glRenderbufferStorageMultisampleIMG"));
290             m_supportsIMGMultisampledRenderToTexture = true;
291         } else if (!m_glGetGraphicsResetStatusEXT && name == "GL_EXT_robustness") {
292             m_glGetGraphicsResetStatusEXT = reinterpret_cast<PFNGLGETGRAPHICSRESETSTATUSEXTPROC>(eglGetProcAddress("glGetGraphicsResetStatusEXT"));
293             m_glReadnPixelsEXT = reinterpret_cast<PFNGLREADNPIXELSEXTPROC>(eglGetProcAddress("glReadnPixelsEXT"));
294             m_glGetnUniformfvEXT = reinterpret_cast<PFNGLGETNUNIFORMFVEXTPROC>(eglGetProcAddress("glGetnUniformfvEXT"));
295             m_glGetnUniformivEXT = reinterpret_cast<PFNGLGETNUNIFORMIVEXTPROC>(eglGetProcAddress("glGetnUniformivEXT"));
296         } else if (!m_supportsANGLEinstancedArrays && name == "GL_ANGLE_instanced_arrays") {
297             m_glVertexAttribDivisorANGLE = reinterpret_cast<PFNGLVERTEXATTRIBDIVISORANGLEPROC>(eglGetProcAddress("glVertexAttribDivisorANGLE"));
298             m_glDrawArraysInstancedANGLE = reinterpret_cast<PFNGLDRAWARRAYSINSTANCEDANGLEPROC >(eglGetProcAddress("glDrawArraysInstancedANGLE"));
299             m_glDrawElementsInstancedANGLE = reinterpret_cast<PFNGLDRAWELEMENTSINSTANCEDANGLEPROC >(eglGetProcAddress("glDrawElementsInstancedANGLE"));
300             m_supportsANGLEinstancedArrays = true;
301         } else if (name == "GL_EXT_draw_buffers") {
302             // FIXME: implement the support.
303             return false;
304         }
305         return true;
306     }
307 
308     return false;
309 }
