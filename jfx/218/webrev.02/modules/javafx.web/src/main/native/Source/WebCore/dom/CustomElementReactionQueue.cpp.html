 47 class CustomElementReactionQueueItem {
 48 public:
 49     enum class Type {
 50         ElementUpgrade,
 51         Connected,
 52         Disconnected,
 53         Adopted,
 54         AttributeChanged,
 55     };
 56 
 57     CustomElementReactionQueueItem(Type type)
 58         : m_type(type)
 59     { }
 60 
 61     CustomElementReactionQueueItem(Document& oldDocument, Document& newDocument)
 62         : m_type(Type::Adopted)
 63         , m_oldDocument(&oldDocument)
 64         , m_newDocument(&newDocument)
 65     { }
 66 
 67     CustomElementReactionQueueItem(const QualifiedName& attributeName, const AtomString& oldValue, const AtomString& newValue)
 68         : m_type(Type::AttributeChanged)
 69         , m_attributeName(attributeName)
 70         , m_oldValue(oldValue)
 71         , m_newValue(newValue)
 72     { }
 73 
 74     Type type() const { return m_type; }
 75 
 76     void invoke(Element& element, JSCustomElementInterface& elementInterface)
 77     {
 78         switch (m_type) {
 79         case Type::ElementUpgrade:
 80             elementInterface.upgradeElement(element);
 81             break;
 82         case Type::Connected:
 83             elementInterface.invokeConnectedCallback(element);
 84             break;
 85         case Type::Disconnected:
 86             elementInterface.invokeDisconnectedCallback(element);
 87             break;
 88         case Type::Adopted:
 89             elementInterface.invokeAdoptedCallback(element, *m_oldDocument, *m_newDocument);
 90             break;
 91         case Type::AttributeChanged:
 92             ASSERT(m_attributeName);
 93             elementInterface.invokeAttributeChangedCallback(element, m_attributeName.value(), m_oldValue, m_newValue);
 94             break;
 95         }
 96     }
 97 
 98 private:
 99     Type m_type;
100     RefPtr<Document> m_oldDocument;
101     RefPtr<Document> m_newDocument;
102     Optional<QualifiedName> m_attributeName;
103     AtomString m_oldValue;
104     AtomString m_newValue;
105 };

107 CustomElementReactionQueue::CustomElementReactionQueue(JSCustomElementInterface& elementInterface)
108     : m_interface(elementInterface)
109 { }

111 CustomElementReactionQueue::~CustomElementReactionQueue()
112 {
113     ASSERT(m_items.isEmpty());
114 }

116 void CustomElementReactionQueue::clear()
117 {
118     m_items.clear();
119 }

121 void CustomElementReactionQueue::enqueueElementUpgrade(Element& element, bool alreadyScheduledToUpgrade)
122 {
123     ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
124     ASSERT(element.reactionQueue());
125     auto& queue = *element.reactionQueue();
126     if (alreadyScheduledToUpgrade) {
127         ASSERT(queue.m_items.size() == 1);
128         ASSERT(queue.m_items[0].type() == CustomElementReactionQueueItem::Type::ElementUpgrade);
129     } else {
130         queue.m_items.append({CustomElementReactionQueueItem::Type::ElementUpgrade});
131         enqueueElementOnAppropriateElementQueue(element);
132     }
133 }
134 
135 void CustomElementReactionQueue::enqueueElementUpgradeIfDefined(Element& element)
136 {
137     ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
138     ASSERT(element.isCustomElementUpgradeCandidate());
139     auto* window = element.document().domWindow();
140     if (!window)
141         return;
142 
143     auto* registry = window->customElementRegistry();
144     if (!registry)
145         return;
146 
147     auto* elementInterface = registry->findInterface(element);
148     if (!elementInterface)
149         return;
150 
151     element.enqueueToUpgrade(*elementInterface);
152 }

154 void CustomElementReactionQueue::enqueueConnectedCallbackIfNeeded(Element& element)
155 {
156     ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
157     ASSERT(element.isDefinedCustomElement());
158     ASSERT(element.document().refCount() > 0);
159     ASSERT(element.reactionQueue());
160     auto& queue = *element.reactionQueue();
161     if (!queue.m_interface->hasConnectedCallback())
162         return;
163     queue.m_items.append({CustomElementReactionQueueItem::Type::Connected});
164     enqueueElementOnAppropriateElementQueue(element);
165 }
166 
167 void CustomElementReactionQueue::enqueueDisconnectedCallbackIfNeeded(Element& element)
168 {
169     ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
170     ASSERT(element.isDefinedCustomElement());
171     if (element.document().refCount() <= 0)
172         return; // Don't enqueue disconnectedCallback if the entire document is getting destructed.
173     ASSERT(element.reactionQueue());
174     auto& queue = *element.reactionQueue();
175     if (!queue.m_interface->hasDisconnectedCallback())
176         return;
177     queue.m_items.append({CustomElementReactionQueueItem::Type::Disconnected});
178     enqueueElementOnAppropriateElementQueue(element);
179 }
180 
181 void CustomElementReactionQueue::enqueueAdoptedCallbackIfNeeded(Element& element, Document& oldDocument, Document& newDocument)
182 {
183     ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
184     ASSERT(element.isDefinedCustomElement());
185     ASSERT(element.document().refCount() > 0);
186     ASSERT(element.reactionQueue());
187     auto& queue = *element.reactionQueue();
188     if (!queue.m_interface->hasAdoptedCallback())
189         return;
190     queue.m_items.append({oldDocument, newDocument});
191     enqueueElementOnAppropriateElementQueue(element);
192 }
193 
194 void CustomElementReactionQueue::enqueueAttributeChangedCallbackIfNeeded(Element& element, const QualifiedName& attributeName, const AtomString& oldValue, const AtomString& newValue)
195 {
196     ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
197     ASSERT(element.isDefinedCustomElement());
198     ASSERT(element.document().refCount() > 0);
199     ASSERT(element.reactionQueue());
200     auto& queue = *element.reactionQueue();
201     if (!queue.m_interface->observesAttribute(attributeName.localName()))
202         return;
203     queue.m_items.append({attributeName, oldValue, newValue});
204     enqueueElementOnAppropriateElementQueue(element);
205 }

207 void CustomElementReactionQueue::enqueuePostUpgradeReactions(Element& element)
208 {
209     ASSERT(CustomElementReactionDisallowedScope::isReactionAllowed());
210     ASSERT(element.isCustomElementUpgradeCandidate());
211     if (!element.hasAttributes() && !element.isConnected())
212         return;
213 
214     ASSERT(element.reactionQueue());
215     auto& queue = *element.reactionQueue();
216 
217     if (element.hasAttributes()) {
218         for (auto& attribute : element.attributesIterator()) {
219             if (queue.m_interface->observesAttribute(attribute.localName()))
220                 queue.m_items.append({attribute.name(), nullAtom(), attribute.value()});
221         }
222     }
223 
224     if (element.isConnected() && queue.m_interface->hasConnectedCallback())
225         queue.m_items.append({CustomElementReactionQueueItem::Type::Connected});
226 }

228 bool CustomElementReactionQueue::observesStyleAttribute() const
229 {
230     return m_interface->observesAttribute(HTMLNames::styleAttr->localName());
231 }

233 void CustomElementReactionQueue::invokeAll(Element& element)
234 {
235     while (!m_items.isEmpty()) {
236         Vector<CustomElementReactionQueueItem> items = WTFMove(m_items);
237         for (auto& item : items)
238             item.invoke(element, m_interface.get());
239     }
240 }

242 inline void CustomElementQueue::add(Element& element)
243 {
244     ASSERT(!m_invoking);
245     // FIXME: Avoid inserting the same element multiple times.
246     m_elements.append(element);
247 }
248 
249 inline void CustomElementQueue::invokeAll()
250 {
251     RELEASE_ASSERT(!m_invoking);
252     SetForScope<bool> invoking(m_invoking, true);
253     unsigned originalSize = m_elements.size();
254     // It's possible for more elements to be enqueued if some IDL attributes were missing CEReactions.
255     // Invoke callbacks slightly later here instead of crashing / ignoring those cases.
256     for (unsigned i = 0; i < m_elements.size(); ++i) {
257         auto& element = m_elements[i].get();
258         auto* queue = element.reactionQueue();
259         ASSERT(queue);
260         queue->invokeAll(element);
261     }
262     ASSERT_UNUSED(originalSize, m_elements.size() == originalSize);
263     m_elements.clear();
264 }
265 
266 inline void CustomElementQueue::processQueue(JSC::JSGlobalObject* state)
267 {
268     if (!state) {
269         invokeAll();
270         return;
271     }
272 
273     auto& vm = state->vm();
274     JSC::JSLockHolder lock(vm);
275 
276     JSC::Exception* previousException = nullptr;
277     {
278         auto catchScope = DECLARE_CATCH_SCOPE(vm);
279         previousException = catchScope.exception();
280         if (previousException)
281             catchScope.clearException();
282     }
283 
284     invokeAll();
285 
286     if (previousException) {
287         auto throwScope = DECLARE_THROW_SCOPE(vm);
288         throwException(state, throwScope, previousException);
289     }
290 }

292 // https://html.spec.whatwg.org/multipage/custom-elements.html#enqueue-an-element-on-the-appropriate-element-queue
293 void CustomElementReactionQueue::enqueueElementOnAppropriateElementQueue(Element& element)
294 {
295     ASSERT(element.reactionQueue());
296     if (!CustomElementReactionStack::s_currentProcessingStack) {
297         element.document().windowEventLoop().backupElementQueue().add(element);
298         return;
299     }
300 
301     auto*& queue = CustomElementReactionStack::s_currentProcessingStack->m_queue;
302     if (!queue) // We use a raw pointer to avoid genearing code to delete it in ~CustomElementReactionStack.
303         queue = new CustomElementQueue;
304     queue->add(element);
305 }

311 CustomElementReactionStack* CustomElementReactionStack::s_currentProcessingStack = nullptr;
312 
313 void CustomElementReactionStack::processQueue(JSC::JSGlobalObject* state)
314 {
315     ASSERT(m_queue);
316     m_queue->processQueue(state);
317     delete m_queue;
318     m_queue = nullptr;
319 }
320 
321 void CustomElementReactionQueue::processBackupQueue(CustomElementQueue& backupElementQueue)
322 {
323     backupElementQueue.processQueue(nullptr);
