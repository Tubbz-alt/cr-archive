 58 PropertyTable::PropertyTable(VM& vm, unsigned initialCapacity)
 59     : JSCell(vm, vm.propertyTableStructure.get())
 60     , m_indexSize(sizeForCapacity(initialCapacity))
 61     , m_indexMask(m_indexSize - 1)
 62     , m_index(static_cast<unsigned*>(PropertyTableMalloc::zeroedMalloc(dataSize())))
 63     , m_keyCount(0)
 64     , m_deletedCount(0)
 65 {
 66     ASSERT(isPowerOf2(m_indexSize));
 67 }
 68 
 69 PropertyTable::PropertyTable(VM& vm, const PropertyTable& other)
 70     : JSCell(vm, vm.propertyTableStructure.get())
 71     , m_indexSize(other.m_indexSize)
 72     , m_indexMask(other.m_indexMask)
 73     , m_index(static_cast<unsigned*>(PropertyTableMalloc::malloc(dataSize())))
 74     , m_keyCount(other.m_keyCount)
 75     , m_deletedCount(other.m_deletedCount)
 76 {
 77     ASSERT(isPowerOf2(m_indexSize));
 78 
 79     memcpy(m_index, other.m_index, dataSize());
 80 
 81     iterator end = this->end();
 82     for (iterator iter = begin(); iter != end; ++iter)
 83         iter->key->ref();
 84 
 85     // Copy the m_deletedOffsets vector.
 86     Vector<PropertyOffset>* otherDeletedOffsets = other.m_deletedOffsets.get();
 87     if (otherDeletedOffsets)
 88         m_deletedOffsets = makeUnique<Vector<PropertyOffset>>(*otherDeletedOffsets);
 89 }
 90 
 91 PropertyTable::PropertyTable(VM& vm, unsigned initialCapacity, const PropertyTable& other)
 92     : JSCell(vm, vm.propertyTableStructure.get())
 93     , m_indexSize(sizeForCapacity(initialCapacity))
 94     , m_indexMask(m_indexSize - 1)
 95     , m_index(static_cast<unsigned*>(PropertyTableMalloc::zeroedMalloc(dataSize())))
 96     , m_keyCount(0)
 97     , m_deletedCount(0)
 98 {
 99     ASSERT(isPowerOf2(m_indexSize));
100     ASSERT(initialCapacity >= other.m_keyCount);
101 
102     const_iterator end = other.end();
103     for (const_iterator iter = other.begin(); iter != end; ++iter) {
104         ASSERT(canInsert());
105         reinsert(*iter);
106         iter->key->ref();
107     }
108 
109     // Copy the m_deletedOffsets vector.
110     Vector<PropertyOffset>* otherDeletedOffsets = other.m_deletedOffsets.get();
111     if (otherDeletedOffsets)
112         m_deletedOffsets = makeUnique<Vector<PropertyOffset>>(*otherDeletedOffsets);
113 }
114 
115 void PropertyTable::destroy(JSCell* cell)
116 {
117     static_cast<PropertyTable*>(cell)->PropertyTable::~PropertyTable();
118 }
119 
120 PropertyTable::~PropertyTable()
121 {
122     iterator end = this->end();
123     for (iterator iter = begin(); iter != end; ++iter)
124         iter->key->deref();
125 
126     PropertyTableMalloc::free(m_index);
127 
128 }
