 45 namespace WebCore {
 46 
 47 class ContentType;
 48 class SourceBuffer;
 49 class SourceBufferList;
 50 class SourceBufferPrivate;

 53 class MediaSource final
 54     : public MediaSourcePrivateClient
 55     , public ActiveDOMObject
 56     , public EventTargetWithInlineData
 57     , public URLRegistrable
 58 #if !RELEASE_LOG_DISABLED
 59     , private LoggerHelper
 60 #endif
 61 {
 62     WTF_MAKE_ISO_ALLOCATED(MediaSource);
 63 public:
 64     static void setRegistry(URLRegistry*);
 65     static MediaSource* lookup(const String& url) { return s_registry ? static_cast<MediaSource*>(s_registry->lookup(url)) : nullptr; }
 66 
 67     static Ref<MediaSource> create(ScriptExecutionContext&);
 68     virtual ~MediaSource();
 69 
 70     void addedToRegistry();
 71     void removedFromRegistry();
 72     void openIfInEndedState();
 73     bool isOpen() const;
 74     bool isClosed() const;
 75     bool isEnded() const;
 76     void sourceBufferDidChangeActiveState(SourceBuffer&, bool);
 77 
 78     enum class EndOfStreamError { Network, Decode };
 79     void streamEndedWithError(Optional<EndOfStreamError>);
 80 
 81     MediaTime duration() const final;
 82     void durationChanged(const MediaTime&) final;
 83     std::unique_ptr<PlatformTimeRanges> buffered() const final;
 84 
 85     bool attachToElement(HTMLMediaElement&);
 86     void detachFromElement(HTMLMediaElement&);
 87     void monitorSourceBuffers() override;
 88     bool isSeeking() const { return m_pendingSeekTime.isValid(); }
 89     Ref<TimeRanges> seekable();
 90     ExceptionOr<void> setLiveSeekableRange(double start, double end);
 91     ExceptionOr<void> clearLiveSeekableRange();
 92 
 93     ExceptionOr<void> setDuration(double);
 94     ExceptionOr<void> setDurationInternal(const MediaTime&);
 95     MediaTime currentTime() const;
 96 
 97     enum class ReadyState { Closed, Open, Ended };
 98     ReadyState readyState() const { return m_readyState; }
 99     ExceptionOr<void> endOfStream(Optional<EndOfStreamError>);
100 
101     HTMLMediaElement* mediaElement() const { return m_mediaElement.get(); }
102 
103     SourceBufferList* sourceBuffers() { return m_sourceBuffers.get(); }
104     SourceBufferList* activeSourceBuffers() { return m_activeSourceBuffers.get(); }
105     ExceptionOr<Ref<SourceBuffer>> addSourceBuffer(const String& type);
106     ExceptionOr<void> removeSourceBuffer(SourceBuffer&);
107     static bool isTypeSupported(const String& type);
108 
109     ScriptExecutionContext* scriptExecutionContext() const final;
110 
111     using RefCounted::ref;
112     using RefCounted::deref;
113 
114     bool hasPendingActivity() const final;
115 
116     static const MediaTime& currentTimeFudgeFactor();
117     static bool contentTypeShouldGenerateTimestamps(const ContentType&);
118 
119 #if !RELEASE_LOG_DISABLED
120     const Logger& logger() const final { return m_logger.get(); }
121     const void* logIdentifier() const final { return m_logIdentifier; }
122     const char* logClassName() const final { return "MediaSource"; }
123     WTFLogChannel& logChannel() const final;
124     void setLogIdentifier(const void*) final;
125 #endif
126 
127 private:
128     explicit MediaSource(ScriptExecutionContext&);
129 
130     void stop() final;
131     const char* activeDOMObjectName() const final;
132 
133     void setPrivateAndOpen(Ref<MediaSourcePrivate>&&) final;
134     void seekToTime(const MediaTime&) final;
135 
136     void refEventTarget() final { ref(); }
137     void derefEventTarget() final { deref(); }
138     EventTargetInterface eventTargetInterface() const final;
139 
140     URLRegistry& registry() const final;
141 
142     void setReadyState(ReadyState);
143     void onReadyStateChange(ReadyState oldState, ReadyState newState);
144 
145     Vector<PlatformTimeRanges> activeRanges() const;
146 
147     ExceptionOr<Ref<SourceBufferPrivate>> createSourceBufferPrivate(const ContentType&);
148     void scheduleEvent(const AtomString& eventName);
149 
150     bool hasBufferedTime(const MediaTime&);
151     bool hasCurrentTime();
152     bool hasFutureTime();
153 
154     void regenerateActiveSourceBuffers();
155 
156     void completeSeek();
157 
158     static URLRegistry* s_registry;
159 
160     RefPtr<MediaSourcePrivate> m_private;
161     RefPtr<SourceBufferList> m_sourceBuffers;
162     RefPtr<SourceBufferList> m_activeSourceBuffers;
163     mutable std::unique_ptr<PlatformTimeRanges> m_buffered;
164     std::unique_ptr<PlatformTimeRanges> m_liveSeekable;
165     WeakPtr<HTMLMediaElement> m_mediaElement;
166     MediaTime m_duration;
167     MediaTime m_pendingSeekTime;
168     ReadyState m_readyState { ReadyState::Closed };
169     UniqueRef<MainThreadGenericEventQueue> m_asyncEventQueue;
170 #if !RELEASE_LOG_DISABLED
171     Ref<const Logger> m_logger;
172     const void* m_logIdentifier { nullptr };
173 #endif
174 };
