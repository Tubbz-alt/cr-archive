35 const ClassInfo JSStringIterator::s_info = { "String Iterator", &Base::s_info, nullptr, nullptr, CREATE_METHOD_TABLE(JSStringIterator) };
36 
37 void JSStringIterator::finishCreation(VM& vm, JSString* iteratedString)
38 {
39     Base::finishCreation(vm);
40     ASSERT(inherits(vm, info()));
41     internalField(Field::Index).set(vm, this, jsNumber(0));
42     internalField(Field::IteratedString).set(vm, this, iteratedString);
43 }
44 
45 JSStringIterator* JSStringIterator::clone(JSGlobalObject* globalObject)
46 {
47     VM& vm = globalObject->vm();
48     JSString* iteratedString = jsCast<JSString*>(this->iteratedString());
49     auto* clone = JSStringIterator::create(vm, globalObject->stringIteratorStructure(), iteratedString);
50     clone->internalField(Field::Index).set(vm, clone, this->index());
51     return clone;
52 }
53 
54 void JSStringIterator::visitChildren(JSCell* cell, SlotVisitor& visitor)
55 {
56     auto* thisObject = jsCast<JSStringIterator*>(cell);
57     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
58     Base::visitChildren(thisObject, visitor);
59 }
