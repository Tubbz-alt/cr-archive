 29 #include "HTMLNames.h"

 32 namespace WebCore {
 33 
 34 WTF_MAKE_ISO_ALLOCATED_IMPL(HTMLDialogElement);
 35 
 36 using namespace HTMLNames;
 37 
 38 HTMLDialogElement::HTMLDialogElement(const QualifiedName& tagName, Document& document)
 39     : HTMLElement(tagName, document)
 40 {
 41 }
 42 
 43 bool HTMLDialogElement::isOpen() const





 44 {
 45     return m_isOpen;
 46 }
 47 
 48 const String& HTMLDialogElement::returnValue()
 49 {
 50     return m_returnValue;
 51 }
 52 
 53 void HTMLDialogElement::setReturnValue(String&& returnValue)
 54 {
 55     m_returnValue = WTFMove(returnValue);
 56 }
 57 
 58 void HTMLDialogElement::show()
 59 {
 60     // If the element already has an open attribute, then return.
 61     if (isOpen())
 62         return;
 63 
 64     toggleOpen();
 65 }
 66 
 67 ExceptionOr<void> HTMLDialogElement::showModal()
 68 {
 69     // If subject already has an open attribute, then throw an "InvalidStateError" DOMException.
 70     if (isOpen())
 71         return Exception { InvalidStateError };
 72 
 73     // If subject is not connected, then throw an "InvalidStateError" DOMException.
 74     if (!isConnected())
 75         return Exception { InvalidStateError };
 76 
 77     toggleOpen();
 78     return { };
 79 }
 80 
 81 void HTMLDialogElement::close(const String& returnValue)
 82 {
 83     if (!isOpen())
 84         return;
 85 
 86     toggleOpen();
 87     if (!returnValue.isNull())
 88         m_returnValue = returnValue;
 89 }
 90 
 91 void HTMLDialogElement::parseAttribute(const QualifiedName& name, const AtomString& value)
 92 {
 93     if (name == HTMLNames::openAttr) {
 94         m_isOpen = !value.isNull();
 95         return;
 96     }
 97 
 98     HTMLElement::parseAttribute(name, value);
 99 }
100 
101 void HTMLDialogElement::toggleOpen()
102 {
103     m_isOpen = !m_isOpen;
104     setAttributeWithoutSynchronization(HTMLNames::openAttr, m_isOpen ? emptyAtom() : nullAtom());
105 }
