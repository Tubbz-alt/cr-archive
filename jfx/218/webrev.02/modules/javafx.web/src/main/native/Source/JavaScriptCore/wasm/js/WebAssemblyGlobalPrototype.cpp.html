 55 static ALWAYS_INLINE JSWebAssemblyGlobal* getGlobal(JSGlobalObject* globalObject, VM& vm, JSValue v)
 56 {
 57     auto throwScope = DECLARE_THROW_SCOPE(vm);
 58     JSWebAssemblyGlobal* result = jsDynamicCast<JSWebAssemblyGlobal*>(vm, v);
 59     if (!result) {
 60         throwException(globalObject, throwScope,
 61             createTypeError(globalObject, "expected |this| value to be an instance of WebAssembly.Global"_s));
 62         return nullptr;
 63     }
 64     return result;
 65 }

 67 static EncodedJSValue JSC_HOST_CALL webAssemblyGlobalProtoFuncValueOf(JSGlobalObject* globalObject, CallFrame* callFrame)
 68 {
 69     VM& vm = globalObject->vm();
 70     auto throwScope = DECLARE_THROW_SCOPE(vm);
 71 
 72     JSWebAssemblyGlobal* global = getGlobal(globalObject, vm, callFrame->thisValue());
 73     RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
 74 
 75     switch (global->global()->type()) {
 76     case Wasm::Type::I64:
 77         return JSValue::encode(throwException(globalObject, throwScope, createTypeError(globalObject, "WebAssembly.Global.prototype.valueOf does not work with i64 type"_s)));
 78     default:
 79         return JSValue::encode(global->global()->get());
 80     }
 81 }

 99 static EncodedJSValue JSC_HOST_CALL webAssemblyGlobalProtoSetterFuncValue(JSGlobalObject* globalObject, CallFrame* callFrame)
100 {
101     VM& vm = globalObject->vm();
102     auto throwScope = DECLARE_THROW_SCOPE(vm);
103 
104     if (UNLIKELY(callFrame->argumentCount() < 1))
105         return JSValue::encode(throwException(globalObject, throwScope, createNotEnoughArgumentsError(globalObject)));
106 
107     JSWebAssemblyGlobal* global = getGlobal(globalObject, vm, callFrame->thisValue());
108     RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
109 
110     if (global->global()->mutability() == Wasm::GlobalInformation::Mutability::Immutable)
111         return JSValue::encode(throwException(globalObject, throwScope, createTypeError(globalObject, "WebAssembly.Global.prototype.value attempts to modify immutable global value"_s)));
112 
113     switch (global->global()->type()) {
114     case Wasm::Type::I64:
115         return JSValue::encode(throwException(globalObject, throwScope, createTypeError(globalObject, "WebAssembly.Global.prototype.value does not work with i64 type"_s)));
116     default:
117         global->global()->set(globalObject, callFrame->argument(0));
118         RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
119         return JSValue::encode(jsUndefined());
120     }
121 }

135 void WebAssemblyGlobalPrototype::finishCreation(VM& vm, JSGlobalObject* globalObject)
136 {
137     Base::finishCreation(vm);
138     ASSERT(inherits(vm, info()));
139 
140     JSFunction* valueGetterFunction = JSFunction::create(vm, globalObject, 0, "get value"_s, webAssemblyGlobalProtoGetterFuncValue, NoIntrinsic);
141     JSFunction* valueSetterFunction = JSFunction::create(vm, globalObject, 1, "set value"_s, webAssemblyGlobalProtoSetterFuncValue, NoIntrinsic);
142     GetterSetter* valueAccessor = GetterSetter::create(vm, globalObject, valueGetterFunction, valueSetterFunction);
143     putDirectNonIndexAccessorWithoutTransition(vm, Identifier::fromString(vm, "value"_s), valueAccessor, static_cast<unsigned>(PropertyAttribute::Accessor));
144 }
