 95 inline JSScope::JSScope(VM& vm, Structure* structure, JSScope* next)
 96     : Base(vm, structure)

100 
101 class ScopeChainIterator {
102 public:
103     ScopeChainIterator(JSScope* node)
104         : m_node(node)
105     {
106     }
107 
108     JSObject* get() const { return JSScope::objectAtScope(m_node); }
109     JSObject* operator->() const { return JSScope::objectAtScope(m_node); }
110     JSScope* scope() const { return m_node; }
111 
112     ScopeChainIterator& operator++() { m_node = m_node->next(); return *this; }
113 
114     // postfix ++ intentionally omitted
115 
116     bool operator==(const ScopeChainIterator& other) const { return m_node == other.m_node; }
117     bool operator!=(const ScopeChainIterator& other) const { return m_node != other.m_node; }
118 
119 private:
120     JSScope* m_node;
121 };
122 
123 inline ScopeChainIterator JSScope::begin()
124 {
125     return ScopeChainIterator(this);
126 }
127 
128 inline ScopeChainIterator JSScope::end()
129 {
130     return ScopeChainIterator(0);
131 }
132 
133 inline JSScope* JSScope::next()
134 {
135     return m_next.get();
136 }
137 
















138 inline size_t JSScope::offsetOfNext()
139 {
140     return OBJECT_OFFSETOF(JSScope, m_next);
