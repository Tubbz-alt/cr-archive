34 namespace WebCore {

35 namespace Layout {
36 
37 class FloatAvoider;
38 class FloatBox;
39 class FormattingContext;
40 class Box;


41 class LayoutState;
42 
43 // FloatingContext is responsible for adjusting the position of a box in the current formatting context
44 // by taking the floating boxes into account.
45 class FloatingContext {
46     WTF_MAKE_ISO_ALLOCATED(FloatingContext);
47 public:
48     FloatingContext(const Container& floatingContextRoot, const FormattingContext&, FloatingState&);
49 
50     FloatingState& floatingState() const { return m_floatingState; }
51 
52     Point positionForFloat(const Box&) const;
53     Optional<Point> positionForFormattingContextRoot(const Box&) const;
54 
55     struct ClearancePosition {
56         Optional<Position> position;
57         Optional<LayoutUnit> clearance;
58     };
59     ClearancePosition verticalPositionWithClearance(const Box&) const;
60 
61     bool isEmpty() const { return m_floatingState.floats().isEmpty(); }
62 
63     struct Constraints {
64         Optional<PointInContextRoot> left;
65         Optional<PointInContextRoot> right;
66     };
67     Constraints constraints(LayoutUnit logicalTop, LayoutUnit logicalBottom) const;
68     void append(const Box&);
69 
70 private:
71     LayoutState& layoutState() const { return m_floatingState.layoutState(); }
72     const FormattingContext& formattingContext() const { return m_formattingContext; }
73     const Container& root() const { return *m_root; }
74 
75     void findPositionForFloatBox(FloatBox&) const;
76     void findPositionForFormattingContextRoot(FloatAvoider&) const;
77 
78     struct AbsoluteCoordinateValuesForFloatAvoider;
79     AbsoluteCoordinateValuesForFloatAvoider absoluteDisplayBoxCoordinates(const Box&) const;
80     Display::Box mapToFloatingStateRoot(const Box&) const;
81     LayoutUnit mapTopToFloatingStateRoot(const Box&) const;
82     Point mapPointFromFormattingContextRootToFloatingStateRoot(Point) const;
83 
84     WeakPtr<const Container> m_root;
85     const FormattingContext& m_formattingContext;
86     FloatingState& m_floatingState;
87 };
88 
89 }
90 }
91 #endif
