 60 // The allocations and releases in ResourceHandleInternal are
 61 // Cocoa-exception-free (either simple Foundation classes or
 62 // WebCoreResourceLoaderImp which avoids doing work in dealloc).
 63 
 64 namespace WebCore {
 65 
 66 DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(ResourceHandleInternal);
 67 class ResourceHandleInternal {
 68     WTF_MAKE_NONCOPYABLE(ResourceHandleInternal);
 69     WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(ResourceHandleInternal);
 70 public:
 71     ResourceHandleInternal(ResourceHandle* loader, NetworkingContext* context, const ResourceRequest& request, ResourceHandleClient* client, bool defersLoading, bool shouldContentSniff, bool shouldContentEncodingSniff)
 72         : m_context(context)
 73         , m_client(client)
 74         , m_firstRequest(request)
 75         , m_lastHTTPMethod(request.httpMethod())
 76         , m_partition(request.cachePartition())
 77         , m_defersLoading(defersLoading)
 78         , m_shouldContentSniff(shouldContentSniff)
 79         , m_shouldContentEncodingSniff(shouldContentEncodingSniff)
 80 #if USE(CFURLCONNECTION)
 81         , m_currentRequest(request)
 82 #endif
 83         , m_failureTimer(*loader, &ResourceHandle::failureTimerFired)
 84     {
 85         const URL& url = m_firstRequest.url();
 86         m_user = url.user();
 87         m_pass = url.pass();
 88         m_firstRequest.removeCredentials();
 89     }
 90 
 91     ~ResourceHandleInternal();
 92 
 93     ResourceHandleClient* client() { return m_client; }
 94 
 95     RefPtr<NetworkingContext> m_context;
 96     ResourceHandleClient* m_client;
 97     ResourceRequest m_firstRequest;
 98     String m_lastHTTPMethod;
 99     String m_partition;
100 
101     // Suggested credentials for the current redirection step.
102     String m_user;
103     String m_pass;
104 
105     Credential m_initialCredential;
106 
107     int status { 0 };
108 
109     bool m_defersLoading;
110     bool m_shouldContentSniff;
111     bool m_shouldContentEncodingSniff;
112 #if USE(CFURLCONNECTION)
113     RetainPtr<CFURLConnectionRef> m_connection;
114     ResourceRequest m_currentRequest;
115     RefPtr<ResourceHandleCFURLConnectionDelegate> m_connectionDelegate;
116 #endif
117 #if PLATFORM(COCOA)
118     RetainPtr<NSURLConnection> m_connection;
119     RetainPtr<id> m_delegate;
120 #endif
121 #if PLATFORM(COCOA)
122     bool m_startWhenScheduled { false };
123 #endif
124 #if PLATFORM(COCOA) || USE(CFURLCONNECTION)
125     RetainPtr<CFURLStorageSessionRef> m_storageSession;
126 #endif
127 #if USE(CURL)
128     std::unique_ptr<CurlResourceHandleDelegate> m_delegate;
129 
130     bool m_cancelled { false };
131     unsigned m_redirectCount { 0 };
132     unsigned m_authFailureCount { 0 };
133     bool m_addedCacheValidationHeaders { false };
134     RefPtr<CurlRequest> m_curlRequest;
135     RefPtr<SynchronousLoaderMessageQueue> m_messageQueue;
136     MonotonicTime m_startTime;
137 #endif
138 
139 #if PLATFORM(JAVA)
140     std::unique_ptr<URLLoader> m_loader;
141     int m_redirectCount { 0 };
142 #endif
143 
144 #if PLATFORM(COCOA)
145     // We need to keep a reference to the original challenge to be able to cancel it.
146     // It is almost identical to m_currentWebChallenge.nsURLAuthenticationChallenge(), but has a different sender.
147     NSURLAuthenticationChallenge *m_currentMacChallenge { nil };
148 #endif
149 
150     AuthenticationChallenge m_currentWebChallenge;
151     ResourceHandle::FailureType m_scheduledFailureType { ResourceHandle::NoFailure };
152     Timer m_failureTimer;
153 };
