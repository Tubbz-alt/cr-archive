 50 class TreeScope {
 51     friend class Document;
 52 
 53 public:
 54     TreeScope* parentTreeScope() const { return m_parentTreeScope; }
 55     void setParentTreeScope(TreeScope&);
 56 
 57     Element* focusedElementInScope();
 58     Element* pointerLockElement() const;
 59 
 60     WEBCORE_EXPORT Element* getElementById(const AtomString&) const;
 61     WEBCORE_EXPORT Element* getElementById(const String&) const;
 62     Element* getElementById(StringView) const;
 63     const Vector<Element*>* getAllElementsById(const AtomString&) const;
 64     bool hasElementWithId(const AtomStringImpl&) const;
 65     bool containsMultipleElementsWithId(const AtomString& id) const;
 66     void addElementById(const AtomStringImpl& elementId, Element&, bool notifyObservers = true);
 67     void removeElementById(const AtomStringImpl& elementId, Element&, bool notifyObservers = true);
 68 
 69     WEBCORE_EXPORT Element* getElementByName(const AtomString&) const;
 70     bool hasElementWithName(const AtomStringImpl&) const;
 71     bool containsMultipleElementsWithName(const AtomString&) const;
 72     void addElementByName(const AtomStringImpl&, Element&);
 73     void removeElementByName(const AtomStringImpl&, Element&);
 74 
 75     Document& documentScope() const { return m_documentScope.get(); }
 76     static ptrdiff_t documentScopeMemoryOffset() { return OBJECT_OFFSETOF(TreeScope, m_documentScope); }
 77 
 78     // https://dom.spec.whatwg.org/#retarget
 79     Node& retargetToScope(Node&) const;
 80 
 81     WEBCORE_EXPORT Node* ancestorNodeInThisScope(Node*) const;
 82     WEBCORE_EXPORT Element* ancestorElementInThisScope(Element*) const;
 83 
 84     void addImageMap(HTMLMapElement&);
 85     void removeImageMap(HTMLMapElement&);
 86     HTMLMapElement* getImageMap(const AtomString&) const;
 87 
 88     void addImageElementByUsemap(const AtomStringImpl&, HTMLImageElement&);
 89     void removeImageElementByUsemap(const AtomStringImpl&, HTMLImageElement&);
 90     HTMLImageElement* imageElementByUsemap(const AtomStringImpl&) const;
 91 
 92     // For accessibility.
 93     bool shouldCacheLabelsByForAttribute() const { return !!m_labelsByForAttribute; }
 94     void addLabel(const AtomStringImpl& forAttributeValue, HTMLLabelElement&);
 95     void removeLabel(const AtomStringImpl& forAttributeValue, HTMLLabelElement&);
 96     HTMLLabelElement* labelElementForId(const AtomString& forAttributeValue);
 97 
 98     WEBCORE_EXPORT RefPtr<Element> elementFromPoint(double clientX, double clientY);
 99     WEBCORE_EXPORT Vector<RefPtr<Element>> elementsFromPoint(double clientX, double clientY);
100     WEBCORE_EXPORT Vector<RefPtr<Element>> elementsFromPoint(const FloatPoint&);
101 
102     // Find first anchor with the given name.
103     // First searches for an element with the given ID, but if that fails, then looks
104     // for an anchor with the given name. ID matching is always case sensitive, but
105     // Anchor name matching is case sensitive in strict mode and not case sensitive in
106     // quirks mode for historical compatibility reasons.
107     Element* findAnchor(const String& name);
108 
109     ContainerNode& rootNode() const { return m_rootNode; }
110 
111     IdTargetObserverRegistry& idTargetObserverRegistry() const { return *m_idTargetObserverRegistry.get(); }
112 
113     RadioButtonGroups& radioButtonGroups();
114 
115 protected:
116     TreeScope(ShadowRoot&, Document&);
117     explicit TreeScope(Document&);
118     ~TreeScope();
119 
120     void destroyTreeScopeData();
121     void setDocumentScope(Document& document)
122     {
123         m_documentScope = document;
124     }
125 
126     Node* nodeFromPoint(const LayoutPoint& clientPoint, LayoutPoint* localPoint);
127 
128 private:
129 
130     ContainerNode& m_rootNode;
131     std::reference_wrapper<Document> m_documentScope;
132     TreeScope* m_parentTreeScope;
133 
134     std::unique_ptr<TreeScopeOrderedMap> m_elementsById;
135     std::unique_ptr<TreeScopeOrderedMap> m_elementsByName;
136     std::unique_ptr<TreeScopeOrderedMap> m_imageMapsByName;
137     std::unique_ptr<TreeScopeOrderedMap> m_imagesByUsemap;
138     std::unique_ptr<TreeScopeOrderedMap> m_labelsByForAttribute;
139 
140     std::unique_ptr<IdTargetObserverRegistry> m_idTargetObserverRegistry;
141 
142     std::unique_ptr<RadioButtonGroups> m_radioButtonGroups;
143 };
144 
145 inline bool TreeScope::hasElementWithId(const AtomStringImpl& id) const
146 {
147     return m_elementsById && m_elementsById->contains(id);
148 }
149 
150 inline bool TreeScope::containsMultipleElementsWithId(const AtomString& id) const
151 {
152     return m_elementsById && id.impl() && m_elementsById->containsMultiple(*id.impl());
153 }
154 
155 inline bool TreeScope::hasElementWithName(const AtomStringImpl& id) const
156 {
157     return m_elementsByName && m_elementsByName->contains(id);
158 }
159 
160 inline bool TreeScope::containsMultipleElementsWithName(const AtomString& name) const
161 {
162     return m_elementsByName && name.impl() && m_elementsByName->containsMultiple(*name.impl());
163 }
