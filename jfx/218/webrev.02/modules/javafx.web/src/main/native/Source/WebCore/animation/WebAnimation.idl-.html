26 enum AnimationPlayState {
27     "idle",
28     "running",
29     "paused",
30     "finished"
31 };
32 
33 [
34     ActiveDOMObject,
35     EnabledAtRuntime=WebAnimations,
36     InterfaceName=Animation,
37     CustomConstructor(),
38     CustomToJSObject
39 ] interface WebAnimation : EventTarget {
40     attribute DOMString id;
41     attribute AnimationEffect? effect;
42     attribute AnimationTimeline? timeline;
43     attribute double? startTime;
44     [MayThrowException, ImplementedAs=bindingsCurrentTime] attribute double? currentTime;
45     attribute double playbackRate;
46     [ImplementedAs=bindingsPlayState] readonly attribute AnimationPlayState playState;
47     [ImplementedAs=bindingsPending] readonly attribute boolean pending;
48     attribute EventHandler onfinish;
49     attribute EventHandler oncancel;
50     [ImplementedAs=bindingsReady] readonly attribute Promise<WebAnimation> ready;
51     [ImplementedAs=bindingsFinished] readonly attribute Promise<WebAnimation> finished;
52     void cancel();
53     [MayThrowException] void finish();
54     [MayThrowException, ImplementedAs=bindingsPlay] void play();
55     [MayThrowException, ImplementedAs=bindingsPause] void pause();
56     void updatePlaybackRate(double playbackRate);
57     [MayThrowException] void reverse();
58 };
