26 #pragma once
27 
28 #include "Event.h"
29 
30 namespace WebCore {
31 
32 class WebKitAnimationEvent final : public Event {
33     WTF_MAKE_ISO_ALLOCATED(WebKitAnimationEvent);
34 public:
35     static Ref<WebKitAnimationEvent> create(const AtomString& type, const String& animationName, double elapsedTime, const String& pseudoElement)
36     {
37         return adoptRef(*new WebKitAnimationEvent(type, animationName, elapsedTime, pseudoElement));
38     }
39 
40     struct Init : EventInit {
41         String animationName;
42         double elapsedTime { 0.0 };
43         String pseudoElement;
44     };
45 
46     static Ref<WebKitAnimationEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
47     {
48         return adoptRef(*new WebKitAnimationEvent(type, initializer, isTrusted));
49     }
50 
51     virtual ~WebKitAnimationEvent();
52 
53     const String& animationName() const;
54     double elapsedTime() const;
55     const String& pseudoElement() const;
56 
57     EventInterface eventInterface() const override;
58 
59 private:
60     WebKitAnimationEvent(const AtomString& type, const String& animationName, double elapsedTime, const String& pseudoElement);
61     WebKitAnimationEvent(const AtomString&, const Init&, IsTrusted);
62 
63     String m_animationName;
64     double m_elapsedTime;
65     String m_pseudoElement;
66 };
67 
68 } // namespace WebCore
