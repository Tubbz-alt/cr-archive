26 [
27     Conditional=VIDEO,
28     ExportMacro=WEBCORE_EXPORT,
29     JSGenerateToNativeObject,
30 ] interface HTMLVideoElement : HTMLMediaElement {
31     [CEReactions=NotNeeded, Reflect] attribute unsigned long width;
32     [CEReactions=NotNeeded, Reflect] attribute unsigned long height;
33     readonly attribute unsigned long videoWidth;
34     readonly attribute unsigned long videoHeight;
35     [CEReactions=NotNeeded, Reflect, URL] attribute USVString poster;
36 
37     [CEReactions=NotNeeded, Reflect] attribute boolean playsInline;
38     readonly attribute boolean webkitSupportsFullscreen;
39     readonly attribute boolean webkitDisplayingFullscreen;
40 
41     [MayThrowException] void webkitEnterFullscreen();
42     void webkitExitFullscreen();
43 
44     // Maintain "FullScreen" capitalization scheme for backwards compatibility.
45     // https://bugs.webkit.org/show_bug.cgi?id=36081
46     [ImplementedAs=webkitEnterFullscreen, MayThrowException] void webkitEnterFullScreen();
47     [ImplementedAs=webkitExitFullscreen] void webkitExitFullScreen();
48 
49     [Conditional=WIRELESS_PLAYBACK_TARGET] attribute boolean webkitWirelessVideoPlaybackDisabled;
50 
51     // The number of frames that have been decoded and made available for playback.
52     [Conditional=MEDIA_STATISTICS] readonly attribute unsigned long webkitDecodedFrameCount;
53 
54     // The number of decoded frames that have been dropped by the player for performance reasons during playback.
55     [Conditional=MEDIA_STATISTICS] readonly attribute unsigned long webkitDroppedFrameCount;
56 
57     [Conditional=VIDEO_PRESENTATION_MODE] boolean webkitSupportsPresentationMode(VideoPresentationMode mode);
58     [Conditional=VIDEO_PRESENTATION_MODE] readonly attribute VideoPresentationMode webkitPresentationMode;
59     [Conditional=VIDEO_PRESENTATION_MODE] void webkitSetPresentationMode(VideoPresentationMode mode);
60 };
61 
62 [Conditional=VIDEO_PRESENTATION_MODE] enum VideoPresentationMode { "inline", "fullscreen", "picture-in-picture" };
