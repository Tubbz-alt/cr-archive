 69 MacroAssembler::Jump PrintSpecial::generate(Inst& inst, CCallHelpers& jit, GenerationContext&)
 70 {
 71     size_t currentArg = 1; // Skip the PrintSpecial arg.
 72     for (auto& term : *m_printRecordList) {
 73         if (term.printer == Printer::printAirArg) {
 74             const Arg& arg = inst.args[currentArg++];
 75             switch (arg.kind()) {
 76             case Arg::Tmp:
 77                 term = Printer::Printer<MacroAssembler::RegisterID>(arg.gpr());
 78                 break;
 79             case Arg::Addr:
 80             case Arg::ExtendedOffsetAddr:
 81                 term = Printer::Printer<MacroAssembler::Address>(arg.asAddress());
 82                 break;
 83             default:
 84                 RELEASE_ASSERT_NOT_REACHED();
 85                 break;
 86             }
 87         }
 88     }
 89     jit.print(m_printRecordList);
 90     return CCallHelpers::Jump();
 91 }
