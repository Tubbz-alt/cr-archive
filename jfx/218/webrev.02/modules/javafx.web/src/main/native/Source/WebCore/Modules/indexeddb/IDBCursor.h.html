 26 #pragma once
 27 
 28 #if ENABLE(INDEXED_DATABASE)
 29 
 30 #include "ExceptionOr.h"
 31 #include "IDBCursorDirection.h"
 32 #include "IDBCursorInfo.h"
 33 #include "IDBKeyPath.h"
 34 #include "IDBRequest.h"
 35 #include "IDBValue.h"
 36 #include "JSValueInWrappedObject.h"
 37 #include <JavaScriptCore/Strong.h>
 38 #include <wtf/Variant.h>
 39 #include <wtf/WeakPtr.h>
 40 
 41 namespace WebCore {
 42 
 43 class IDBGetResult;
 44 class IDBIndex;
 45 class IDBObjectStore;
 46 class IDBTransaction;
 47 
 48 class IDBCursor : public ScriptWrappable, public RefCounted<IDBCursor> {
 49     WTF_MAKE_ISO_ALLOCATED(IDBCursor);
 50 public:
 51     static Ref<IDBCursor> create(IDBObjectStore&, const IDBCursorInfo&);
 52     static Ref<IDBCursor> create(IDBIndex&, const IDBCursorInfo&);
 53 
 54     virtual ~IDBCursor();
 55 
 56     using Source = Variant<RefPtr<IDBObjectStore>, RefPtr<IDBIndex>>;
 57 
 58     const Source& source() const;
 59     IDBCursorDirection direction() const;
 60 
 61     IDBKey* key() { return m_key.get(); };
 62     IDBKey* primaryKey() { return m_primaryKey.get(); };
 63     IDBValue value() { return m_value; };
 64     const Optional<IDBKeyPath>& primaryKeyPath() { return m_keyPath; };
 65     JSValueInWrappedObject& keyWrapper() { return m_keyWrapper; }
 66     JSValueInWrappedObject& primaryKeyWrapper() { return m_primaryKeyWrapper; }
 67     JSValueInWrappedObject& valueWrapper() { return m_valueWrapper; }
 68 
 69     ExceptionOr<Ref<IDBRequest>> update(JSC::JSGlobalObject&, JSC::JSValue);
 70     ExceptionOr<void> advance(unsigned);
 71     ExceptionOr<void> continueFunction(JSC::JSGlobalObject&, JSC::JSValue key);
 72     ExceptionOr<void> continuePrimaryKey(JSC::JSGlobalObject&, JSC::JSValue key, JSC::JSValue primaryKey);
 73     ExceptionOr<Ref<IDBRequest>> deleteFunction(JSC::JSGlobalObject&);
 74 
 75     ExceptionOr<void> continueFunction(const IDBKeyData&);
 76 
 77     const IDBCursorInfo& info() const { return m_info; }
 78 
 79     void setRequest(IDBRequest& request) { m_request = makeWeakPtr(&request); }
 80     void clearRequest() { m_request.clear(); }
 81     void clearWrappers();
 82     IDBRequest* request() { return m_request.get(); }
 83 
 84     bool setGetResult(IDBRequest&, const IDBGetResult&, uint64_t operationID);
 85 
 86     virtual bool isKeyCursorWithValue() const { return false; }
 87 
 88     Optional<IDBGetResult> iterateWithPrefetchedRecords(unsigned count, uint64_t lastWriteOperationID);
 89     void clearPrefetchedRecords();
 90 
 91 protected:
 92     IDBCursor(IDBObjectStore&, const IDBCursorInfo&);
 93     IDBCursor(IDBIndex&, const IDBCursorInfo&);
 94 
 95 private:
 96     bool sourcesDeleted() const;
 97     IDBObjectStore& effectiveObjectStore() const;
 98     IDBTransaction& transaction() const;
 99 
100     void uncheckedIterateCursor(const IDBKeyData&, unsigned count);
101     void uncheckedIterateCursor(const IDBKeyData&, const IDBKeyData&);
102 
103     IDBCursorInfo m_info;
104     Source m_source;
105     WeakPtr<IDBRequest> m_request;
106 
107     bool m_gotValue { false };
108 
109     RefPtr<IDBKey> m_key;
110     RefPtr<IDBKey> m_primaryKey;
111     IDBKeyData m_keyData;
112     IDBKeyData m_primaryKeyData;
113     IDBValue m_value;
114     Optional<IDBKeyPath> m_keyPath;
115 
116     JSValueInWrappedObject m_keyWrapper;
117     JSValueInWrappedObject m_primaryKeyWrapper;
118     JSValueInWrappedObject m_valueWrapper;
119 
120     Deque<IDBCursorRecord> m_prefetchedRecords;
121     uint64_t m_prefetchOperationID { 0 };
122 };
123 
124 
125 inline const IDBCursor::Source& IDBCursor::source() const
126 {
127     return m_source;
128 }
129 
130 inline IDBCursorDirection IDBCursor::direction() const
131 {
132     return m_info.cursorDirection();
133 }
134 
135 } // namespace WebCore

137 #endif // ENABLE(INDEXED_DATABASE)
