 44 Ref<ScrollingTreeFixedNode> ScrollingTreeFixedNode::create(ScrollingTree& scrollingTree, ScrollingNodeID nodeID)
 45 {
 46     return adoptRef(*new ScrollingTreeFixedNode(scrollingTree, nodeID));
 47 }

 49 ScrollingTreeFixedNode::ScrollingTreeFixedNode(ScrollingTree& scrollingTree, ScrollingNodeID nodeID)
 50     : ScrollingTreeNode(scrollingTree, ScrollingNodeType::Fixed, nodeID)
 51 {
 52     scrollingTree.fixedOrStickyNodeAdded();
 53 }
 54 
 55 ScrollingTreeFixedNode::~ScrollingTreeFixedNode()
 56 {
 57     scrollingTree().fixedOrStickyNodeRemoved();
 58 }

 60 void ScrollingTreeFixedNode::commitStateBeforeChildren(const ScrollingStateNode& stateNode)
 61 {
 62     auto& fixedStateNode = downcast<ScrollingStateFixedNode>(stateNode);
 63 
 64     if (fixedStateNode.hasChangedProperty(ScrollingStateNode::Layer)) {
 65         auto* layer = static_cast<Nicosia::PlatformLayer*>(fixedStateNode.layer());
 66         m_layer = downcast<Nicosia::CompositionLayer>(layer);
 67     }
 68 
 69     if (fixedStateNode.hasChangedProperty(ScrollingStateFixedNode::ViewportConstraints))
 70         m_constraints = fixedStateNode.viewportConstraints();
 71 }

 73 void ScrollingTreeFixedNode::applyLayerPositions()
 74 {
 75     auto computeLayerPosition = [&] {
 76         FloatSize overflowScrollDelta;
 77         // FIXME: This code is wrong in complex cases where the fixed element is inside a positioned node as
 78         //        the scroll container order does not match the scrolling tree ancestor order.
 79         for (auto* node = parent(); node; node = node->parent()) {
 80             if (is<ScrollingTreeFrameScrollingNode>(*node)) {
 81                 // Fixed nodes are positioned relative to the containing frame scrolling node.
 82                 // We bail out after finding one.
 83                 auto layoutViewport = downcast<ScrollingTreeFrameScrollingNode>(*node).layoutViewport();
 84                 return m_constraints.layerPositionForViewportRect(layoutViewport) - overflowScrollDelta;
 85             }
 86 
 87             if (is<ScrollingTreeOverflowScrollingNode>(*node)) {
 88                 // To keep the layer still during async scrolling we adjust by how much the position has changed since layout.
 89                 auto& overflowNode = downcast<ScrollingTreeOverflowScrollingNode>(*node);
 90                 auto localDelta = overflowNode.lastCommittedScrollPosition() - overflowNode.currentScrollPosition();
 91                 overflowScrollDelta += localDelta;
 92             }
 93         }
 94         ASSERT_NOT_REACHED();
 95         return FloatPoint();
 96     };
 97 
 98     auto layerPosition = computeLayerPosition();
 99 
100     LOG_WITH_STREAM(Scrolling, stream << "ScrollingTreeFixedNode " << scrollingNodeID() << " relatedNodeScrollPositionDidChange: viewportRectAtLastLayout " << m_constraints.viewportRectAtLastLayout() << " last layer pos " << m_constraints.layerPositionAtLastLayout() << " layerPosition " << layerPosition);
101 
102     ASSERT(m_layer);
103     m_layer->updateState(
104         [&layerPosition](Nicosia::CompositionLayer::LayerState& state)
105         {
106             state.position = layerPosition;
107             state.delta.positionChanged = true;
108         });
109 }

111 void ScrollingTreeFixedNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
112 {
113     ts << "fixed node";
114     ScrollingTreeNode::dumpProperties(ts, behavior);
115     ts.dumpProperty("fixed constraints", m_constraints);
116 
117     if (behavior & ScrollingStateTreeAsTextBehaviorIncludeLayerPositions) {
118         FloatPoint layerTopLeft;
119         ASSERT(m_layer);
120         m_layer->accessCommitted(
121             [this, &layerTopLeft](Nicosia::CompositionLayer::LayerState& state)
122             {
123                 layerTopLeft = state.position - toFloatSize(state.anchorPoint.xy()) * state.size + m_constraints.alignmentOffset();
124             });
125 
126         ts.dumpProperty("layer top left", layerTopLeft);
127     }
128 }
