20 #pragma once
21 
22 #include "JSDOMConstructorBase.h"
23 
24 namespace WebCore {
25 
26 class JSDOMBuiltinConstructorBase : public JSDOMConstructorBase {
27 public:
28     using Base = JSDOMConstructorBase;
29 
30     template<typename CellType, JSC::SubspaceAccess>
31     static JSC::IsoSubspace* subspaceFor(JSC::VM& vm)
32     {
33         static_assert(sizeof(CellType) == sizeof(JSDOMBuiltinConstructorBase));
34         STATIC_ASSERT_ISO_SUBSPACE_SHARABLE(CellType, JSDOMBuiltinConstructorBase);
35         static_assert(CellType::destroy == JSC::JSCell::destroy, "JSDOMBuiltinConstructor<JSClass> is not destructible actually");
36         return subspaceForImpl(vm);
37     }
38 
39 protected:
40     JSDOMBuiltinConstructorBase(JSC::Structure* structure, JSDOMGlobalObject& globalObject)
41         : JSDOMConstructorBase(structure, globalObject)
42     {
43     }
44 
45     static void visitChildren(JSC::JSCell*, JSC::SlotVisitor&);
46 
47     JSC::JSFunction* initializeFunction();
48     void setInitializeFunction(JSC::VM&, JSC::JSFunction&);
49 
50     static void callFunctionWithCurrentArguments(JSC::JSGlobalObject&, JSC::CallFrame&, JSC::JSObject& thisObject, JSC::JSFunction&);
51 
52 private:
53     static JSC::IsoSubspace* subspaceForImpl(JSC::VM&);
54 
55     JSC::WriteBarrier<JSC::JSFunction> m_initializeFunction;
56 };
57 
58 inline JSC::JSFunction* JSDOMBuiltinConstructorBase::initializeFunction()
59 {
60     return m_initializeFunction.get();
61 }
62 
63 inline void JSDOMBuiltinConstructorBase::setInitializeFunction(JSC::VM& vm, JSC::JSFunction& function)
64 {
65     m_initializeFunction.set(vm, this, &function);
66 }
67 
68 
69 } // namespace WebCore
