 44 void NavigatorEME::requestMediaKeySystemAccess(Navigator&, Document& document, const String& keySystem, Vector<MediaKeySystemConfiguration>&& supportedConfigurations, Ref<DeferredPromise>&& promise)
 45 {
 46     // https://w3c.github.io/encrypted-media/#dom-navigator-requestmediakeysystemaccess
 47     // W3C Editor's Draft 09 November 2016
 48     LOG(EME, "EME - request media key system access for %s", keySystem.utf8().data());
 49 
 50     // When this method is invoked, the user agent must run the following steps:
 51     // 1. If keySystem is the empty string, return a promise rejected with a newly created TypeError.
 52     // 2. If supportedConfigurations is empty, return a promise rejected with a newly created TypeError.
 53     if (keySystem.isEmpty() || supportedConfigurations.isEmpty()) {
 54         promise->reject(TypeError);
 55         return;
 56     }
 57 
 58     document.postTask([keySystem, supportedConfigurations = WTFMove(supportedConfigurations), promise = WTFMove(promise), &document] (ScriptExecutionContext&) mutable {
 59         // 3. Let document be the calling context's Document.
 60         // 4. Let origin be the origin of document.
 61         // 5. Let promise be a new promise.
 62         // 6. Run the following steps in parallel:
 63         // 6.1. If keySystem is not one of the Key Systems supported by the user agent, reject promise with a NotSupportedError.
 64         //      String comparison is case-sensitive.
 65         if (!CDM::supportsKeySystem(keySystem)) {
 66             LOG(EME, "EME - %s is not supported", keySystem.utf8().data());
 67             promise->reject(NotSupportedError);
 68             return;
 69         }
 70 
 71         // 6.2. Let implementation be the implementation of keySystem.
 72         auto implementation = CDM::create(document, keySystem);
 73         tryNextSupportedConfiguration(WTFMove(implementation), WTFMove(supportedConfigurations), WTFMove(promise));
 74     });
 75 }
 76 
 77 static void tryNextSupportedConfiguration(RefPtr<CDM>&& implementation, Vector<MediaKeySystemConfiguration>&& supportedConfigurations, RefPtr<DeferredPromise>&& promise)
 78 {
 79     // 6.3. For each value in supportedConfigurations:
 80     if (!supportedConfigurations.isEmpty()) {
 81         // 6.3.1. Let candidate configuration be the value.
 82         // 6.3.2. Let supported configuration be the result of executing the Get Supported Configuration
 83         //        algorithm on implementation, candidate configuration, and origin.
 84         MediaKeySystemConfiguration candidateConfiguration = WTFMove(supportedConfigurations.first());
 85         supportedConfigurations.remove(0);
 86 
 87         CDM::SupportedConfigurationCallback callback = [implementation = implementation, supportedConfigurations = WTFMove(supportedConfigurations), promise] (Optional<MediaKeySystemConfiguration> supportedConfiguration) mutable {
 88             // 6.3.3. If supported configuration is not NotSupported, run the following steps:
 89             if (supportedConfiguration) {
 90                 // 6.3.3.1. Let access be a new MediaKeySystemAccess object, and initialize it as follows:
 91                 // 6.3.3.1.1. Set the keySystem attribute to keySystem.
 92                 // 6.3.3.1.2. Let the configuration value be supported configuration.
 93                 // 6.3.3.1.3. Let the cdm implementation value be implementation.
 94 
 95                 // Obtain reference to the key system string before the `implementation` RefPtr<> is cleared out.
 96                 const String& keySystem = implementation->keySystem();
 97                 auto access = MediaKeySystemAccess::create(keySystem, WTFMove(supportedConfiguration.value()), implementation.releaseNonNull());
 98 
 99                 // 6.3.3.2. Resolve promise with access and abort the parallel steps of this algorithm.
100                 promise->resolveWithNewlyCreated<IDLInterface<MediaKeySystemAccess>>(WTFMove(access));
101                 return;
102             }
103 
104             tryNextSupportedConfiguration(WTFMove(implementation), WTFMove(supportedConfigurations), WTFMove(promise));
105         };
106         implementation->getSupportedConfiguration(WTFMove(candidateConfiguration), WTFMove(callback));
107         return;
108     }
109 
110 
111     // 6.4. Reject promise with a NotSupportedError.
112     promise->reject(NotSupportedError);
113 }
