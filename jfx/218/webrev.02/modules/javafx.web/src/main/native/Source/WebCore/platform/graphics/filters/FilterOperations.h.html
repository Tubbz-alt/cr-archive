26 #pragma once
27 
28 #include "FilterOperation.h"
29 #include "IntRectExtent.h"
30 #include <wtf/RefPtr.h>
31 #include <wtf/Vector.h>
32 
33 namespace WebCore {
34 
35 class FilterOperations {
36     WTF_MAKE_FAST_ALLOCATED;
37 public:
38     bool operator==(const FilterOperations&) const;
39     bool operator!=(const FilterOperations& other) const { return !(*this == other); }
40 
41     void clear() { m_operations.clear(); }
42 
43     Vector<RefPtr<FilterOperation>>& operations() { return m_operations; }
44     const Vector<RefPtr<FilterOperation>>& operations() const { return m_operations; }
45 
46     bool isEmpty() const { return m_operations.isEmpty(); }
47     size_t size() const { return m_operations.size(); }
48     const FilterOperation* at(size_t index) const { return index < m_operations.size() ? m_operations[index].get() : nullptr; }
49 
50     bool operationsMatch(const FilterOperations&) const;
51 
52     bool hasOutsets() const { return !outsets().isZero(); }
53     IntOutsets outsets() const;
54 
55     bool hasFilterThatAffectsOpacity() const;
56     bool hasFilterThatMovesPixels() const;
57     bool hasFilterThatShouldBeRestrictedBySecurityOrigin() const;
58 
59     bool hasReferenceFilter() const;
60 
61     bool transformColor(Color&) const;
62     bool inverseTransformColor(Color&) const;
63 
64 private:
65     Vector<RefPtr<FilterOperation>> m_operations;
66 };
67 
68 WEBCORE_EXPORT WTF::TextStream& operator<<(WTF::TextStream&, const FilterOperations&);
69 
70 } // namespace WebCore
