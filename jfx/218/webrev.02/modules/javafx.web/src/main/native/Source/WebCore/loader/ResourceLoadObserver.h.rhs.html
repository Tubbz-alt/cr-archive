31 namespace WebCore {
32 
33 class Document;
34 class Frame;


35 class ResourceRequest;
36 class ResourceResponse;



37 
38 class ResourceLoadObserver {
39     WTF_MAKE_FAST_ALLOCATED;
40 public:
41     // https://fetch.spec.whatwg.org/#request-destination-script-like
42     enum class FetchDestinationIsScriptLike : bool { Yes, No };
























43 
44     WEBCORE_EXPORT static ResourceLoadObserver& shared();
45     WEBCORE_EXPORT static ResourceLoadObserver* sharedIfExists();
46     WEBCORE_EXPORT static void setShared(ResourceLoadObserver&);
47 
48     virtual ~ResourceLoadObserver() { }
49 
50     virtual void logSubresourceLoading(const Frame*, const ResourceRequest& /* newRequest */, const ResourceResponse& /* redirectResponse */, FetchDestinationIsScriptLike) { }
51     virtual void logWebSocketLoading(const URL& /* targetURL */, const URL& /* mainFrameURL */) { }
52     virtual void logUserInteractionWithReducedTimeResolution(const Document&) { }
53     virtual void logFontLoad(const Document&, const String& /* familyName */, bool /* loadStatus */) { }
54     virtual void logCanvasRead(const Document&) { }
55     virtual void logCanvasWriteOrMeasure(const Document&, const String& /* textWritten */) { }
56     virtual void logNavigatorAPIAccessed(const Document&, const ResourceLoadStatistics::NavigatorAPI) { }
57     virtual void logScreenAPIAccessed(const Document&, const ResourceLoadStatistics::ScreenAPI) { }
58     virtual void logSubresourceLoadingForTesting(const RegistrableDomain& /* firstPartyDomain */, const RegistrableDomain& /* thirdPartyDomain */, bool /* shouldScheduleNotification */) { }
59 
60     virtual String statisticsForURL(const URL&) { return { }; }
61     virtual void updateCentralStatisticsStore() { }
62     virtual void clearState() { }
63 
64     virtual bool hasStatistics() const { return false; }






65 };
