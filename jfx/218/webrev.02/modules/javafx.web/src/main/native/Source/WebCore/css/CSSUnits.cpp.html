 28 CSSUnitCategory unitCategory(CSSUnitType type)
 29 {
 30     switch (type) {
 31     case CSSUnitType::CSS_NUMBER:
 32         return CSSUnitCategory::Number;
 33     case CSSUnitType::CSS_PERCENTAGE:
 34         return CSSUnitCategory::Percent;
 35     case CSSUnitType::CSS_PX:
 36     case CSSUnitType::CSS_CM:
 37     case CSSUnitType::CSS_MM:
 38     case CSSUnitType::CSS_IN:
 39     case CSSUnitType::CSS_PT:
 40     case CSSUnitType::CSS_PC:
 41     case CSSUnitType::CSS_Q:
 42         return CSSUnitCategory::Length;
 43     case CSSUnitType::CSS_MS:
 44     case CSSUnitType::CSS_S:
 45         return CSSUnitCategory::Time;
 46     case CSSUnitType::CSS_DEG:
 47     case CSSUnitType::CSS_RAD:
 48     case CSSUnitType::CSS_GRAD:
 49     case CSSUnitType::CSS_TURN:
 50         return CSSUnitCategory::Angle;
 51     case CSSUnitType::CSS_HZ:
 52     case CSSUnitType::CSS_KHZ:
 53         return CSSUnitCategory::Frequency;
 54     case CSSUnitType::CSS_DPPX:
 55     case CSSUnitType::CSS_DPI:
 56     case CSSUnitType::CSS_DPCM:
 57         return CSSUnitCategory::Resolution;
 58     default:
 59         return CSSUnitCategory::Other;
 60     }
 61 }
 62 
 63 CSSUnitType canonicalUnitTypeForCategory(CSSUnitCategory category)
 64 {
 65     switch (category) {
 66     case CSSUnitCategory::Number:
 67         return CSSUnitType::CSS_NUMBER;
 68     case CSSUnitCategory::Length:
 69         return CSSUnitType::CSS_PX;
 70     case CSSUnitCategory::Percent:
 71         return CSSUnitType::CSS_UNKNOWN; // Cannot convert between numbers and percent.
 72     case CSSUnitCategory::Time:
 73         return CSSUnitType::CSS_MS;
 74     case CSSUnitCategory::Angle:
 75         return CSSUnitType::CSS_DEG;
 76     case CSSUnitCategory::Frequency:
 77         return CSSUnitType::CSS_HZ;
 78 #if ENABLE(CSS_IMAGE_RESOLUTION) || ENABLE(RESOLUTION_MEDIA_QUERY)
 79     case CSSUnitCategory::Resolution:
 80         return CSSUnitType::CSS_DPPX;
 81 #endif
 82     default:
 83         return CSSUnitType::CSS_UNKNOWN;
 84     }
 85 }
 86 
 87 CSSUnitType canonicalUnitType(CSSUnitType unitType)
 88 {
 89     return canonicalUnitTypeForCategory(unitCategory(unitType));
 90 }

 92 TextStream& operator<<(TextStream& ts, CSSUnitCategory category)
 93 {
 94     switch (category) {
 95     case CSSUnitCategory::Number: ts << "Number"; break;
 96     case CSSUnitCategory::Percent: ts << "Percent"; break;
 97     case CSSUnitCategory::Length: ts << "Length"; break;
 98     case CSSUnitCategory::Angle: ts << "Angle"; break;
 99     case CSSUnitCategory::Time: ts << "Time"; break;
100     case CSSUnitCategory::Frequency: ts << "Frequency"; break;
101     case CSSUnitCategory::Resolution: ts << "Resolution"; break;
102     case CSSUnitCategory::Other: ts << "Other"; break;
103     }
104     return ts;
105 }
