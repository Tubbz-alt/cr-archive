 26 #pragma once
 27 
 28 #include "WorkerGlobalScopeProxy.h"
 29 #include "WorkerDebuggerProxy.h"
 30 #include "WorkerLoaderProxy.h"
 31 #include "WorkerObjectProxy.h"
 32 #include <wtf/MonotonicTime.h>
 33 #include <wtf/ThreadSafeRefCounted.h>
 34 
 35 namespace WebCore {
 36 
 37 class DedicatedWorkerThread;
 38 class WorkerInspectorProxy;
 39 
 40 class WorkerMessagingProxy final : public ThreadSafeRefCounted<WorkerMessagingProxy>, public WorkerGlobalScopeProxy, public WorkerObjectProxy, public WorkerLoaderProxy, public WorkerDebuggerProxy {
 41     WTF_MAKE_FAST_ALLOCATED;
 42 public:
 43     explicit WorkerMessagingProxy(Worker&);
 44     virtual ~WorkerMessagingProxy();
 45 
 46     // Implementation of WorkerLoaderProxy.
 47     // This method is used in the main thread to post task back to the worker thread.
 48     bool postTaskForModeToWorkerGlobalScope(ScriptExecutionContext::Task&&, const String& mode) final;
 49 
 50 private:
 51     // Implementations of WorkerGlobalScopeProxy.
 52     // (Only use these functions in the worker object thread.)
 53     void startWorkerGlobalScope(const URL& scriptURL, const String& name, const String& userAgent, bool isOnline, const String& sourceCode, const ContentSecurityPolicyResponseHeaders&, bool shouldBypassMainWorldContentSecurityPolicy, MonotonicTime timeOrigin, ReferrerPolicy, JSC::RuntimeFlags) final;
 54     void terminateWorkerGlobalScope() final;
 55     void postMessageToWorkerGlobalScope(MessageWithMessagePorts&&) final;
 56     bool hasPendingActivity() const final;
 57     void workerObjectDestroyed() final;
 58     void notifyNetworkStateChange(bool isOnline) final;
 59     void suspendForBackForwardCache() final;
 60     void resumeForBackForwardCache() final;
 61 
 62     // Implementation of WorkerObjectProxy.
 63     // (Only use these functions in the worker context thread.)
 64     void postMessageToWorkerObject(MessageWithMessagePorts&&) final;
 65     void postExceptionToWorkerObject(const String& errorMessage, int lineNumber, int columnNumber, const String& sourceURL) final;
 66     void confirmMessageFromWorkerObject(bool hasPendingActivity) final;
 67     void reportPendingActivity(bool hasPendingActivity) final;
 68     void workerGlobalScopeClosed() final;
 69     void workerGlobalScopeDestroyed() final;
 70 
 71     // Implementation of WorkerDebuggerProxy.
 72     // (Only use these functions in the worker context thread.)
 73     void postMessageToDebugger(const String&) final;
 74     void setResourceCachingDisabledByWebInspector(bool) final;
 75 
 76     // Implementation of WorkerLoaderProxy.
 77     // These functions are called on different threads to schedule loading
 78     // requests and to send callbacks back to WorkerGlobalScope.
 79     bool isWorkerMessagingProxy() const final { return true; }
 80     void postTaskToLoader(ScriptExecutionContext::Task&&) final;
 81     Ref<CacheStorageConnection> createCacheStorageConnection() final;
 82 
 83     void workerThreadCreated(DedicatedWorkerThread&);
 84 
 85     // Only use this method on the worker object thread.
 86     bool askedToTerminate() const { return m_askedToTerminate; }
 87 
 88     void workerGlobalScopeDestroyedInternal();
 89     void reportPendingActivityInternal(bool confirmingMessage, bool hasPendingActivity);
 90     Worker* workerObject() const { return m_workerObject; }
 91 
 92     RefPtr<ScriptExecutionContext> m_scriptExecutionContext;
 93     std::unique_ptr<WorkerInspectorProxy> m_inspectorProxy;
 94     Worker* m_workerObject;
 95     bool m_mayBeDestroyed { false };
 96     RefPtr<DedicatedWorkerThread> m_workerThread;
 97 
 98     unsigned m_unconfirmedMessageCount { 0 }; // Unconfirmed messages from worker object to worker thread.
 99     bool m_workerThreadHadPendingActivity { false }; // The latest confirmation from worker thread reported that it was still active.
100 
101     bool m_askedToSuspend { false };
102     bool m_askedToTerminate { false };
103 
104     Vector<std::unique_ptr<ScriptExecutionContext::Task>> m_queuedEarlyTasks; // Tasks are queued here until there's a thread object created.
105 };
106 
107 } // namespace WebCore
108 
109 SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::WorkerMessagingProxy)
110     static bool isType(const WebCore::WorkerLoaderProxy& proxy) { return proxy.isWorkerMessagingProxy(); }
111 SPECIALIZE_TYPE_TRAITS_END()
