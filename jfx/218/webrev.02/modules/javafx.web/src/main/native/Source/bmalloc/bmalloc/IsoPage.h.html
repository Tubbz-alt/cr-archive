 37 class IsoHeapImplBase;
 38 template<typename Config> class IsoDirectoryBase;
 39 template<typename Config> class IsoHeapImpl;
 40 
 41 class IsoPageBase {
 42 public:
 43     static constexpr size_t pageSize = 16384;
 44 
 45     explicit IsoPageBase(bool isShared)
 46         : m_isShared(isShared)
 47         , m_eligibilityHasBeenNoted(true)
 48         , m_isInUseForAllocation(false)
 49     {
 50     }
 51 
 52     static IsoPageBase* pageFor(void*);
 53 
 54     bool isShared() const { return m_isShared; }
 55 
 56 protected:
 57     BEXPORT static void* allocatePageMemory();
 58 
 59     bool m_isShared : 1;
 60     bool m_eligibilityHasBeenNoted : 1;
 61     bool m_isInUseForAllocation : 1;
 62 };
 63 
 64 template<typename Config>
 65 class IsoPage : public IsoPageBase {
 66 public:
 67     static constexpr unsigned numObjects = pageSize / Config::objectSize;
 68 
 69     static_assert(numObjects, "IsoHeap size should allow at least one allocation per page");
 70 
 71     static IsoPage* tryCreate(IsoDirectoryBase<Config>& directory, unsigned index);
 72 
 73     // It's expected that you will only use this with placement new and direct destruction.
 74     IsoPage(IsoDirectoryBase<Config>& directory, unsigned index);
 75 
 76     static IsoPage* pageFor(void*);
 77 
 78     unsigned index() const { return m_index; }
 79 
 80     void free(const LockHolder&, void*);
 81 
 82     // Called after this page is already selected for allocation.
 83     FreeList startAllocating(const LockHolder&);
 84 
 85     // Called after the allocator picks another page to replace this one.
 86     void stopAllocating(const LockHolder&, FreeList);
 87 
 88     IsoDirectoryBase<Config>& directory() { return m_directory; }
 89     bool isInUseForAllocation() const { return m_isInUseForAllocation; }
 90 
 91     template<typename Func>
 92     void forEachLiveObject(const LockHolder&, const Func&);
 93 
 94     IsoHeapImpl<Config>& heap();
 95 
 96 private:
 97     static constexpr unsigned indexOfFirstObject()
 98     {
 99         return (sizeof(IsoPage) + Config::objectSize - 1) / Config::objectSize;
100     }
101 
102     // The possible states of a page are as follows. We mark these states by their corresponding
103     // eligible, empty, and committed bits (respectively).
104     //
105     // 000 - Deallocated. It has no objects and its memory is not paged in.
106     // 111 - Empty.
107     // 101 - Eligible for allocation, meaning that there is at least one free object in the page.
108     // 001 - Full.
109     // 001 - Currently being used for allocation.
110     //
111     // Note that the last two states have identical representation in the directory, which is fine - in
112     // both cases we are basically telling the directory that this page is off limits. But we keep track
113     // of the distinction internally.
114 
115     // We manage the bitvector ourselves. This bitvector works in a special way to enable very fast
116     // freeing.
117 
118     // This must have a trivial destructor.
119 
120     DeferredTrigger<IsoPageTrigger::Eligible> m_eligibilityTrigger;
121     DeferredTrigger<IsoPageTrigger::Empty> m_emptyTrigger;
122     uint8_t m_numNonEmptyWords { 0 };
123     static_assert(bitsArrayLength(numObjects) <= UINT8_MAX);
124     unsigned m_index { UINT_MAX };
125     IsoDirectoryBase<Config>& m_directory;
126     unsigned m_allocBits[bitsArrayLength(numObjects)];
127 };
