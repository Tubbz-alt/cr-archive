25 #pragma once
26 
27 #include "TransformationMatrix.h"
28 
29 #include <JavaScriptCore/Float32Array.h>
30 #include <wtf/RefPtr.h>
31 
32 namespace WebCore {
33 
34 class VRStageParameters : public RefCounted<VRStageParameters> {
35 public:
36     static RefPtr<VRStageParameters> create(const Optional<TransformationMatrix>& sittingToStandingTransform, const Optional<FloatSize>& playAreaBounds)
37     {
38         if (!sittingToStandingTransform || !playAreaBounds)
39             return nullptr;
40 
41         return adoptRef(*new VRStageParameters(sittingToStandingTransform.value(), playAreaBounds.value()));
42     }
43 
44     Ref<Float32Array> sittingToStandingTransform() const;
45 
46     float sizeX() const;
47     float sizeZ() const;
48 
49 private:
50     VRStageParameters(const TransformationMatrix& sittingToStandingTransform, const FloatSize& playAreaBounds);
51 
52     FloatSize m_playAreaBounds;
53     TransformationMatrix m_sittingToStandingTransform;
54 };
55 
56 } // namespace WebCore
