125 void RunLoop::TimerBase::timerFired()
126 {
127     {
128         LockHolder locker(m_runLoop->m_loopLock);
129 
130         if (!m_isActive)
131             return;
132 
133         if (!m_isRepeating) {
134             m_isActive = false;
135             ::KillTimer(m_runLoop->m_runLoopMessageWindow, bitwise_cast<uintptr_t>(this));
136         } else
137             m_nextFireDate = MonotonicTime::now() + m_interval;
138     }
139 
140     fired();
141 }

123 RunLoop::CycleResult RunLoop::cycle(RunLoopMode)
124 {
125     MSG message;
126     if (!::GetMessage(&message, 0, 0, 0))
127         return CycleResult::Stop;
128 
129     ::TranslateMessage(&message);
130     ::DispatchMessage(&message);
131 
132     return CycleResult::Continue;
133 }
