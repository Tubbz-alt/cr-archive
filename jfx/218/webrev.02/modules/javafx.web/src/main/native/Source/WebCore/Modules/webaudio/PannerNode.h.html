 44 // PannerNode is an AudioNode with one input and one output.
 45 // It positions a sound in 3D space, with the exact effect dependent on the panning model.
 46 // It has a position and an orientation in 3D space which is relative to the position and orientation of the context's AudioListener.
 47 // A distance effect will attenuate the gain as the position moves away from the listener.
 48 // A cone effect will attenuate the gain as the orientation moves away from the listener.
 49 // All of these effects follow the OpenAL specification very closely.
 50 
 51 class PannerNode final : public AudioNode {
 52     WTF_MAKE_ISO_ALLOCATED(PannerNode);
 53 public:
 54     static Ref<PannerNode> create(AudioContext& context, float sampleRate)
 55     {
 56         return adoptRef(*new PannerNode(context, sampleRate));
 57     }
 58 
 59     virtual ~PannerNode();
 60 
 61     // AudioNode
 62     void process(size_t framesToProcess) override;
 63     void pullInputs(size_t framesToProcess) override;
 64     void reset() override;
 65     void initialize() override;
 66     void uninitialize() override;
 67 
 68     // Listener
 69     AudioListener* listener();
 70 
 71     // Panning model
 72     PanningModelType panningModel() const { return m_panningModel; }
 73     void setPanningModel(PanningModelType);
 74 
 75     // Position
 76     FloatPoint3D position() const { return m_position; }
 77     void setPosition(float x, float y, float z) { m_position = FloatPoint3D(x, y, z); }
 78 
 79     // Orientation
 80     FloatPoint3D orientation() const { return m_position; }
 81     void setOrientation(float x, float y, float z) { m_orientation = FloatPoint3D(x, y, z); }
 82 
 83     // Velocity
 84     FloatPoint3D velocity() const { return m_velocity; }
 85     void setVelocity(float x, float y, float z) { m_velocity = FloatPoint3D(x, y, z); }
 86 
 87     // Distance parameters
 88     DistanceModelType distanceModel() const;
 89     void setDistanceModel(DistanceModelType);
 90 
 91     double refDistance() { return m_distanceEffect.refDistance(); }
 92     void setRefDistance(double refDistance) { m_distanceEffect.setRefDistance(refDistance); }
 93 
 94     double maxDistance() { return m_distanceEffect.maxDistance(); }
 95     void setMaxDistance(double maxDistance) { m_distanceEffect.setMaxDistance(maxDistance); }
 96 
 97     double rolloffFactor() { return m_distanceEffect.rolloffFactor(); }
 98     void setRolloffFactor(double rolloffFactor) { m_distanceEffect.setRolloffFactor(rolloffFactor); }
 99 
100     // Sound cones - angles in degrees
101     double coneInnerAngle() const { return m_coneEffect.innerAngle(); }
102     void setConeInnerAngle(double angle) { m_coneEffect.setInnerAngle(angle); }
103 
104     double coneOuterAngle() const { return m_coneEffect.outerAngle(); }
105     void setConeOuterAngle(double angle) { m_coneEffect.setOuterAngle(angle); }
106 
107     double coneOuterGain() const { return m_coneEffect.outerGain(); }
108     void setConeOuterGain(double angle) { m_coneEffect.setOuterGain(angle); }
109 
110     void getAzimuthElevation(double* outAzimuth, double* outElevation);
111     float dopplerRate();
112 
113     // Accessors for dynamically calculated gain values.
114     AudioParam* distanceGain() { return m_distanceGain.get(); }
115     AudioParam* coneGain() { return m_coneGain.get(); }
116 
117     double tailTime() const override { return m_panner ? m_panner->tailTime() : 0; }
118     double latencyTime() const override { return m_panner ? m_panner->latencyTime() : 0; }
119 
120 private:
121     PannerNode(AudioContext&, float sampleRate);
122 
123     // Returns the combined distance and cone gain attenuation.
124     float distanceConeGain();
125 
126     // Notifies any AudioBufferSourceNodes connected to us either directly or indirectly about our existence.
127     // This is in order to handle the pitch change necessary for the doppler shift.
128     void notifyAudioSourcesConnectedToNode(AudioNode*, HashSet<AudioNode*>& visitedNodes);
129 
130     std::unique_ptr<Panner> m_panner;
131     PanningModelType m_panningModel;
132 
133     FloatPoint3D m_position;
134     FloatPoint3D m_orientation;
135     FloatPoint3D m_velocity;
136 
137     // Gain
138     RefPtr<AudioParam> m_distanceGain;
139     RefPtr<AudioParam> m_coneGain;
140     DistanceEffect m_distanceEffect;
141     ConeEffect m_coneEffect;
142     float m_lastGain;
143 
144     // HRTF Database loader
145     RefPtr<HRTFDatabaseLoader> m_hrtfDatabaseLoader;
146 
147     unsigned m_connectionCount;
148 
149     // Synchronize process() and setPanningModel() which can change the panner.
150     mutable Lock m_pannerMutex;
151 };
