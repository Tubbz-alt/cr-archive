30 #include "HTMLDocument.h"

32 namespace WebCore {
33 
34 class MediaDocument final : public HTMLDocument {
35     WTF_MAKE_ISO_ALLOCATED(MediaDocument);
36 public:
37     static Ref<MediaDocument> create(Frame* frame, const URL& url)
38     {
39         return adoptRef(*new MediaDocument(frame, url));
40     }
41     virtual ~MediaDocument();
42 

43     void mediaElementNaturalSizeChanged(const IntSize&);
44     String outgoingReferrer() const { return m_outgoingReferrer; }
45 
46 private:
47     MediaDocument(Frame*, const URL&);
48 
49     Ref<DocumentParser> createParser() override;
50 
51     void defaultEventHandler(Event&) override;
52 
53     void replaceMediaElementTimerFired();
54 

55     String m_outgoingReferrer;
56 };
57 
58 } // namespace WebCore
59 
60 SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::MediaDocument)
61     static bool isType(const WebCore::Document& document) { return document.isMediaDocument(); }
62     static bool isType(const WebCore::Node& node) { return is<WebCore::Document>(node) && isType(downcast<WebCore::Document>(node)); }
63 SPECIALIZE_TYPE_TRAITS_END()
