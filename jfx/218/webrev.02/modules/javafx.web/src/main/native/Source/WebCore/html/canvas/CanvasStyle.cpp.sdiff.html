 46 static bool isCurrentColorString(const String& colorString)
 47 {
 48     return equalLettersIgnoringASCIICase(colorString, "currentcolor");

 51 static Color parseColor(const String& colorString)
 52 {








 53     Color color = CSSParser::parseColor(colorString);
 54     if (color.isValid())
 55         return color;
 56     return CSSParser::parseSystemColor(colorString, nullptr);
 57 }

 59 Color currentColor(HTMLCanvasElement* canvas)
 60 {
 61     if (!canvas || !canvas->isConnected() || !canvas->inlineStyle())




 62         return Color::black;
 63     Color color = CSSParser::parseColor(canvas->inlineStyle()->getPropertyValue(CSSPropertyColor));
 64     if (!color.isValid())
 65         return Color::black;
 66     return color;
 67 }

 70 {
 71     if (isCurrentColorString(colorString))
 72         return currentColor(canvas);

136 bool CanvasStyle::isEquivalentColor(const CanvasStyle& other) const
137 {
138     if (WTF::holds_alternative<Color>(m_style) && WTF::holds_alternative<Color>(other.m_style))
139         return WTF::get<Color>(m_style) == WTF::get<Color>(other.m_style);
140 
141     if (WTF::holds_alternative<CMYKAColor>(m_style) && WTF::holds_alternative<CMYKAColor>(other.m_style)) {
142         auto& a = WTF::get<CMYKAColor>(m_style);
143         auto& b = WTF::get<CMYKAColor>(other.m_style);
144         return a.c == b.c && a.m == b.m && a.y == b.y && a.k == b.k && a.a == b.a;
145     }
146 
147     return false;
148 }

 54 Color parseColor(const String& colorString, CanvasBase& canvasBase)
 55 {
 56 #if ENABLE(OFFSCREEN_CANVAS)
 57     if (canvasBase.isOffscreenCanvas()) {
 58         auto& canvas = downcast<OffscreenCanvas>(canvasBase);
 59         return CSSParser::parseColorWorkerSafe(colorString, canvas.cssValuePool());
 60     }
 61 #else
 62     UNUSED_PARAM(canvasBase);
 63 #endif
 64     Color color = CSSParser::parseColor(colorString);
 65     if (color.isValid())
 66         return color;
 67     return CSSParser::parseSystemColor(colorString, nullptr);
 68 }
 69 
 70 Color currentColor(CanvasBase& canvasBase)
 71 {
 72     if (!is<HTMLCanvasElement>(canvasBase))
 73         return Color::black;
 74 
 75     auto& canvas = downcast<HTMLCanvasElement>(canvasBase);
 76     if (!canvas.isConnected() || !canvas.inlineStyle())
 77         return Color::black;
 78     Color color = CSSParser::parseColor(canvas.inlineStyle()->getPropertyValue(CSSPropertyColor));
 79     if (!color.isValid())
 80         return Color::black;
 81     return color;
 82 }

 85 {
 86     if (isCurrentColorString(colorString))
 87         return currentColor(canvasBase);
 88 
 89     return parseColor(colorString, canvasBase);
 90 }
