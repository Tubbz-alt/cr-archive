 75 class WorkerThread : public ThreadSafeRefCounted<WorkerThread> {
 76 public:
 77     virtual ~WorkerThread();
 78 
 79     static HashSet<WorkerThread*>& workerThreads(const LockHolder&);
 80     static Lock& workerThreadsMutex();
 81 
 82     void stop(WTF::Function<void()>&& terminatedCallback);
 83 
 84     void suspend();
 85     void resume();
 86 
 87     Thread* thread() const { return m_thread.get(); }
 88     WorkerRunLoop& runLoop() { return m_runLoop; }
 89     WorkerLoaderProxy& workerLoaderProxy() const { return m_workerLoaderProxy; }
 90     WorkerDebuggerProxy& workerDebuggerProxy() const { return m_workerDebuggerProxy; }
 91     WorkerReportingProxy& workerReportingProxy() const { return m_workerReportingProxy; }
 92 
 93     // Number of active worker threads.
 94     WEBCORE_EXPORT static unsigned workerThreadCount();
 95     static void releaseFastMallocFreeMemoryInAllThreads();
 96 
 97 #if ENABLE(NOTIFICATIONS)
 98     NotificationClient* getNotificationClient() { return m_notificationClient; }
 99     void setNotificationClient(NotificationClient* client) { m_notificationClient = client; }
100 #endif
101 
102     void startRunningDebuggerTasks();
103     void stopRunningDebuggerTasks();
104 
105     JSC::RuntimeFlags runtimeFlags() const { return m_runtimeFlags; }
106 
107     String identifier() const { return m_identifier; }
108 
109 protected:
110     WorkerThread(const WorkerParameters&, const String& sourceCode, WorkerLoaderProxy&, WorkerDebuggerProxy&, WorkerReportingProxy&, WorkerThreadStartMode, const SecurityOrigin& topOrigin, IDBClient::IDBConnectionProxy*, SocketProvider*, JSC::RuntimeFlags);
111 
112     // Factory method for creating a new worker context for the thread.
113     virtual Ref<WorkerGlobalScope> createWorkerGlobalScope(const WorkerParameters&, Ref<SecurityOrigin>&&, Ref<SecurityOrigin>&& topOrigin) = 0;
114 
115     // Executes the event loop for the worker thread. Derived classes can override to perform actions before/after entering the event loop.
116     virtual void runEventLoop();
117 
118     WorkerGlobalScope* workerGlobalScope() { return m_workerGlobalScope.get(); }
119 
120     IDBClient::IDBConnectionProxy* idbConnectionProxy();
121     SocketProvider* socketProvider();
122 
123     void start(Function<void(const String&)>&& evaluateCallback);
124 
125 private:
126     void workerThread();
127     virtual bool isServiceWorkerThread() const { return false; }
128 
129     virtual void finishedEvaluatingScript() { }
130 
131     RefPtr<Thread> m_thread;
132     String m_identifier;
133     WorkerRunLoop m_runLoop;
134     WorkerLoaderProxy& m_workerLoaderProxy;
135     WorkerDebuggerProxy& m_workerDebuggerProxy;
136     WorkerReportingProxy& m_workerReportingProxy;
137     JSC::RuntimeFlags m_runtimeFlags;
138     bool m_pausedForDebugger { false };
139 
140     RefPtr<WorkerGlobalScope> m_workerGlobalScope;
141     Lock m_threadCreationAndWorkerGlobalScopeMutex;
142 
143     std::unique_ptr<WorkerThreadStartupData> m_startupData;
144 
145     WTF::Function<void(const String&)> m_evaluateCallback;
146 
147 #if ENABLE(NOTIFICATIONS)
148     NotificationClient* m_notificationClient { nullptr };
149 #endif
150 
151 #if ENABLE(INDEXED_DATABASE)
152     RefPtr<IDBClient::IDBConnectionProxy> m_idbConnectionProxy;
153 #endif
154     RefPtr<SocketProvider> m_socketProvider;
155 
156     WTF::Function<void()> m_stoppedCallback;
157     BinarySemaphore m_suspensionSemaphore;
158     bool m_isSuspended { false };
159 };
