26 #pragma once
27 
28 #include <type_traits>
29 #include <utility>
30 
31 namespace bmalloc {
32 
33 template<typename ExitFunction>
34 class ScopeExit {
35 public:
36     explicit ScopeExit(ExitFunction&& exitFunction)
37         : m_exitFunction(exitFunction)
38     {
39     }
40 
41     ~ScopeExit()
42     {
43         m_exitFunction();
44     }
45 
46 private:
47     ExitFunction m_exitFunction;
48 };
49 
50 template<typename ExitFunction>
51 ScopeExit<ExitFunction> makeScopeExit(ExitFunction&& exitFunction)
52 {
53     return ScopeExit<ExitFunction>(std::forward<ExitFunction>(exitFunction));
54 }
55 
56 } // namespace bmalloc
