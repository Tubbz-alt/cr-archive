 45 class LayoutState;

 47 class LayoutTreeContent : public CanMakeWeakPtr<LayoutTreeContent> {
 48     WTF_MAKE_ISO_ALLOCATED(LayoutTreeContent);
 49 public:
 50     LayoutTreeContent(const RenderBox&, std::unique_ptr<Container>);
 51     ~LayoutTreeContent();
 52 
 53     const Container& rootLayoutBox() const { return *m_rootLayoutBox; }
 54     Container& rootLayoutBox() { return *m_rootLayoutBox; }
 55     const RenderBox& rootRenderer() const { return m_rootRenderer; }
 56 
 57     void addBox(std::unique_ptr<Box> box)
 58     {
 59         ASSERT(!box->isContainer());
 60         m_boxes.add(WTFMove(box));
 61     }
 62     void addContainer(std::unique_ptr<Container> container) { m_containers.add(WTFMove(container)); }
 63 
 64     Box* layoutBoxForRenderer(const RenderObject& renderer) { return m_renderObjectToLayoutBox.get(&renderer); }
 65     const Box* layoutBoxForRenderer(const RenderObject& renderer) const { return m_renderObjectToLayoutBox.get(&renderer); }
 66 
 67     const RenderObject* rendererForLayoutBox(const Box& box) const { return m_layoutBoxToRenderObject.get(&box); }
 68 
 69     void addLayoutBoxForRenderer(const RenderObject&, Box&);
 70 
 71 private:
 72     const RenderBox& m_rootRenderer;
 73     std::unique_ptr<Container> m_rootLayoutBox;
 74     HashSet<std::unique_ptr<Box>> m_boxes;
 75     HashSet<std::unique_ptr<Container>> m_containers;
 76 
 77     HashMap<const RenderObject*, Box*> m_renderObjectToLayoutBox;
 78     HashMap<const Box*, const RenderObject*> m_layoutBoxToRenderObject;
 79 };
 80

 92 
 93     Box& createBox(Optional<Box::ElementAttributes>, RenderStyle&&);
 94     Box& createTextBox(TextContext&&, RenderStyle&&);
 95     Container& createContainer(Optional<Box::ElementAttributes>, RenderStyle&&);
 96 
 97     LayoutTreeContent& m_layoutTreeContent;
 98 };
