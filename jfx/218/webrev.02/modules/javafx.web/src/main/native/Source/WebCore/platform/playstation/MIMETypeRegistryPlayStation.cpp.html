31 static const std::initializer_list<TypeExtensionPair>& platformMediaTypes()
32 {
33     static std::initializer_list<TypeExtensionPair> platformMediaTypes = {
34         { "image/bmp"_s, "bmp"_s },
35         { "text/css"_s, "css"_s },
36         { "image/gif"_s, "gif"_s },
37         { "text/html"_s, "htm"_s },
38         { "text/html"_s, "html"_s },
39         { "image/x-icon"_s, "ico"_s },
40         { "image/jpeg"_s, "jpeg"_s },
41         { "image/jpeg"_s, "jpg"_s },
42         { "application/x-javascript"_s, "js"_s },
43         { "application/pdf"_s, "pdf"_s },
44         { "image/png"_s, "png"_s },
45         { "application/rss+xml"_s, "rss"_s },
46         { "image/svg+xml"_s, "svg"_s },
47         { "application/x-shockwave-flash"_s, "swf"_s },
48         { "text/plain"_s, "text"_s },
49         { "text/plain"_s, "txt"_s },
50         { "text/vnd.wap.wml"_s, "wml"_s },
51         { "application/vnd.wap.wmlc"_s, "wmlc"_s },
52         { "image/x-xbitmap"_s, "xbm"_s },
53         { "application/xhtml+xml"_s, "xhtml"_s },
54         { "text/xml"_s, "xml"_s },
55         { "text/xsl"_s, "xsl"_s },
56     };
57     return platformMediaTypes;
58 }
59 
60 String MIMETypeRegistry::getMIMETypeForExtension(const String& extension)
61 {
62     for (auto& entry : platformMediaTypes()) {
63         if (equalIgnoringASCIICase(extension, entry.extension.characters()))
64             return entry.type;
65     }
66     return emptyString();
67 }
68 
69 bool MIMETypeRegistry::isApplicationPluginMIMEType(const String&)
70 {
71     return false;
72 }
73 
74 String MIMETypeRegistry::getPreferredExtensionForMIMEType(const String& mimeType)
75 {
76     for (auto& entry : platformMediaTypes()) {
77         if (equalIgnoringASCIICase(mimeType, entry.type.characters()))
78             return entry.extension;
79     }
80     return emptyString();
81 }
