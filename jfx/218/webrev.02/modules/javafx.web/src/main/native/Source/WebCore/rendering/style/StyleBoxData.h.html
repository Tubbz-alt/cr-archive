25 #pragma once
26 
27 #include "Length.h"
28 #include "RenderStyleConstants.h"
29 #include <wtf/RefCounted.h>
30 #include <wtf/Ref.h>
31 
32 namespace WebCore {
33 
34 DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(StyleBoxData);
35 class StyleBoxData : public RefCounted<StyleBoxData> {
36     WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(StyleBoxData);
37 public:
38     static Ref<StyleBoxData> create() { return adoptRef(*new StyleBoxData); }
39     Ref<StyleBoxData> copy() const;
40 
41     bool operator==(const StyleBoxData& o) const;
42     bool operator!=(const StyleBoxData& o) const
43     {
44         return !(*this == o);
45     }
46 
47     const Length& width() const { return m_width; }
48     const Length& height() const { return m_height; }
49 
50     const Length& minWidth() const { return m_minWidth; }
51     const Length& minHeight() const { return m_minHeight; }
52 
53     const Length& maxWidth() const { return m_maxWidth; }
54     const Length& maxHeight() const { return m_maxHeight; }
55 
56     const Length& verticalAlign() const { return m_verticalAlign; }
57 
58     int specifiedZIndex() const { return m_specifiedZIndex; }
59     bool hasAutoSpecifiedZIndex() const { return m_hasAutoSpecifiedZIndex; }
60 
61     int usedZIndex() const { return m_usedZIndex; }
62     bool hasAutoUsedZIndex() const { return m_hasAutoUsedZIndex; }
63 
64     BoxSizing boxSizing() const { return static_cast<BoxSizing>(m_boxSizing); }
65 #if ENABLE(CSS_BOX_DECORATION_BREAK)
66     BoxDecorationBreak boxDecorationBreak() const { return static_cast<BoxDecorationBreak>(m_boxDecorationBreak); }
67 #endif
68 
69 private:
70     friend class RenderStyle;
71 
72     StyleBoxData();
73     StyleBoxData(const StyleBoxData&);
74 
75     Length m_width;
76     Length m_height;
77 
78     Length m_minWidth;
79     Length m_maxWidth;
80 
81     Length m_minHeight;
82     Length m_maxHeight;
83 
84     Length m_verticalAlign;
85 
86     int m_specifiedZIndex;
87     int m_usedZIndex;
88     unsigned m_hasAutoSpecifiedZIndex : 1;
89     unsigned m_hasAutoUsedZIndex : 1;
90     unsigned m_boxSizing : 1; // BoxSizing
91 #if ENABLE(CSS_BOX_DECORATION_BREAK)
92     unsigned m_boxDecorationBreak : 1; // BoxDecorationBreak
93 #endif
94 };
95 
96 } // namespace WebCore
