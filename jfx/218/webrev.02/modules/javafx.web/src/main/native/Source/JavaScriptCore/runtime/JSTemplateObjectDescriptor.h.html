27 #pragma once
28 
29 #include "Structure.h"
30 #include "TemplateObjectDescriptor.h"
31 
32 namespace JSC {
33 
34 class JSTemplateObjectDescriptor final : public JSCell {
35 public:
36     using Base = JSCell;
37 
38     static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
39     static constexpr bool needsDestruction = true;
40     template<typename CellType, SubspaceAccess mode>
41     static IsoSubspace* subspaceFor(VM& vm)
42     {
43         return vm.templateObjectDescriptorSpace<mode>();
44     }
45     DECLARE_INFO;
46 
47     static JSTemplateObjectDescriptor* create(VM&, Ref<TemplateObjectDescriptor>&&, int);
48 
49     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
50     {
51         return Structure::create(vm, globalObject, prototype, TypeInfo(CellType, StructureFlags), info());
52     }
53 
54     const TemplateObjectDescriptor& descriptor() const { return m_descriptor.get(); }
55 
56     JSArray* createTemplateObject(JSGlobalObject*);
57 
58     int endOffset() const { return m_endOffset; }
59 
60 protected:
61     static void destroy(JSCell*);
62 
63 private:
64     JSTemplateObjectDescriptor(VM&, Ref<TemplateObjectDescriptor>&&, int);
65 
66     Ref<TemplateObjectDescriptor> m_descriptor;
67     int m_endOffset { 0 };
68 };
69 
70 } // namespace JSC
