127 RefPtr<GenericTypedArrayView<Adaptor>>
128 GenericTypedArrayView<Adaptor>::subarray(int start, int end) const
129 {
130     unsigned offset, length;
131     calculateOffsetAndLength(start, end, this->length(), &offset, &length);
