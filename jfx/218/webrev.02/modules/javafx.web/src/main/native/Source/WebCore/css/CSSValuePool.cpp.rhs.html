 39 StaticCSSValuePool::StaticCSSValuePool()
 40 {
 41     m_inheritedValue.construct(CSSValue::StaticCSSValue);
 42     m_implicitInitialValue.construct(CSSValue::StaticCSSValue, true);
 43     m_explicitInitialValue.construct(CSSValue::StaticCSSValue, false);
 44     m_unsetValue.construct(CSSValue::StaticCSSValue);
 45     m_revertValue.construct(CSSValue::StaticCSSValue);
 46 
 47     m_transparentColor.construct(CSSValue::StaticCSSValue, Color(Color::transparent));
 48     m_whiteColor.construct(CSSValue::StaticCSSValue, Color(Color::white));
 49     m_blackColor.construct(CSSValue::StaticCSSValue, Color(Color::black));
 50 
 51     for (unsigned i = firstCSSValueKeyword; i <= lastCSSValueKeyword; ++i)
 52         m_identifierValues[i].construct(CSSValue::StaticCSSValue, static_cast<CSSValueID>(i));
 53 
 54     for (unsigned i = 0; i < (maximumCacheableIntegerValue + 1); ++i) {
 55         m_pixelValues[i].construct(CSSValue::StaticCSSValue, i, CSSUnitType::CSS_PX);
 56         m_percentValues[i].construct(CSSValue::StaticCSSValue, i, CSSUnitType::CSS_PERCENTAGE);
 57         m_numberValues[i].construct(CSSValue::StaticCSSValue, i, CSSUnitType::CSS_NUMBER);
 58     }
 59 }
 60 
 61 void StaticCSSValuePool::init()
 62 {
 63     static std::once_flag onceKey;
 64     std::call_once(onceKey, []() {
 65         staticCSSValuePool.construct();
 66     });
 67 }
 68 
 69 CSSValuePool::CSSValuePool()
 70 {
 71     StaticCSSValuePool::init();
 72 }

 74 CSSValuePool& CSSValuePool::singleton()
 75 {
 76     ASSERT(isMainThread());
 77     static NeverDestroyed<CSSValuePool> pool;
 78     return pool;
 79 }

 81 Ref<CSSPrimitiveValue> CSSValuePool::createIdentifierValue(CSSValueID ident)
 82 {
 83     RELEASE_ASSERT(ident >= firstCSSValueKeyword && ident <= lastCSSValueKeyword);
 84     return staticCSSValuePool->m_identifierValues[ident].get();
 85 }
 86 
 87 Ref<CSSPrimitiveValue> CSSValuePool::createIdentifierValue(CSSPropertyID ident)
 88 {
 89     return CSSPrimitiveValue::createIdentifier(ident);
 90 }

 92 Ref<CSSPrimitiveValue> CSSValuePool::createColorValue(const Color& color)
 93 {
 94     // These are the empty and deleted values of the hash table.
 95     if (color == Color::transparent)
 96         return staticCSSValuePool->m_transparentColor.get();
 97     if (color == Color::white)
 98         return staticCSSValuePool->m_whiteColor.get();
 99     // Just because it is common.
100     if (color == Color::black)
101         return staticCSSValuePool->m_blackColor.get();
102 
103     // Remove one entry at random if the cache grows too large.
104     // FIXME: Use TinyLRUCache instead?
105     const int maximumColorCacheSize = 512;
106     if (m_colorValueCache.size() >= maximumColorCacheSize)
107         m_colorValueCache.remove(m_colorValueCache.random());
108 
109     return *m_colorValueCache.ensure(color, [&color] {
110         return CSSPrimitiveValue::create(color);
111     }).iterator->value;
112 }
113 
114 Ref<CSSPrimitiveValue> CSSValuePool::createValue(double value, CSSUnitType type)
115 {
116     ASSERT(std::isfinite(value));
117 
118     if (value < 0 || value > StaticCSSValuePool::maximumCacheableIntegerValue)
119         return CSSPrimitiveValue::create(value, type);
120 
121     int intValue = static_cast<int>(value);
122     if (value != intValue)
123         return CSSPrimitiveValue::create(value, type);
124 
125     switch (type) {
126     case CSSUnitType::CSS_PX:
127         return staticCSSValuePool->m_pixelValues[intValue].get();
128     case CSSUnitType::CSS_PERCENTAGE:
129         return staticCSSValuePool->m_percentValues[intValue].get();
130     case CSSUnitType::CSS_NUMBER:
131         return staticCSSValuePool->m_numberValues[intValue].get();
132     default:
133         return CSSPrimitiveValue::create(value, type);
134     }
135 }

137 Ref<CSSPrimitiveValue> CSSValuePool::createFontFamilyValue(const String& familyName, FromSystemFontID fromSystemFontID)
138 {
139     // Remove one entry at random if the cache grows too large.
140     // FIXME: Use TinyLRUCache instead?
141     const int maximumFontFamilyCacheSize = 128;
142     if (m_fontFamilyValueCache.size() >= maximumFontFamilyCacheSize)
143         m_fontFamilyValueCache.remove(m_fontFamilyValueCache.random());
144 
145     bool isFromSystemID = fromSystemFontID == FromSystemFontID::Yes;
146     return *m_fontFamilyValueCache.ensure({ familyName, isFromSystemID }, [&familyName, isFromSystemID] {
147         return CSSPrimitiveValue::create(CSSFontFamily { familyName, isFromSystemID });
148     }).iterator->value;
149 }
150 
151 RefPtr<CSSValueList> CSSValuePool::createFontFaceValue(const AtomString& string)
152 {
153     // Remove one entry at random if the cache grows too large.
154     // FIXME: Use TinyLRUCache instead?
155     const int maximumFontFaceCacheSize = 128;
156     if (m_fontFaceValueCache.size() >= maximumFontFaceCacheSize)
157         m_fontFaceValueCache.remove(m_fontFaceValueCache.random());
158 
159     return m_fontFaceValueCache.ensure(string, [&string] () -> RefPtr<CSSValueList> {
160         auto result = CSSParser::parseSingleValue(CSSPropertyFontFamily, string);
161         if (!is<CSSValueList>(result))
162             return nullptr;
163         // FIXME: Make downcast work on RefPtr, remove the get() below, and save one reference count churn.
164         return downcast<CSSValueList>(result.get());
165     }).iterator->value;
166 }
167 
168 void CSSValuePool::drain()
169 {
170     m_colorValueCache.clear();
171     m_fontFaceValueCache.clear();
172     m_fontFamilyValueCache.clear();
173 }
