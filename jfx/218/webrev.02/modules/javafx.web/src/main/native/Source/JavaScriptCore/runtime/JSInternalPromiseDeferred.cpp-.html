40 JSInternalPromiseDeferred* JSInternalPromiseDeferred::tryCreate(ExecState* exec, JSGlobalObject* globalObject)
41 {
42     VM& vm = exec->vm();
43     auto scope = DECLARE_THROW_SCOPE(vm);
44 
45     DeferredData data = createDeferredData(exec, globalObject, globalObject->internalPromiseConstructor());
46     RETURN_IF_EXCEPTION(scope, nullptr);
47 
48     JSInternalPromiseDeferred* result = new (NotNull, allocateCell<JSInternalPromiseDeferred>(vm.heap)) JSInternalPromiseDeferred(vm);
49     result->finishCreation(vm, data.promise, data.resolve, data.reject);
50     return result;
51 }
52 
53 JSInternalPromiseDeferred::JSInternalPromiseDeferred(VM& vm)
54     : Base(vm, vm.internalPromiseDeferredStructure.get())
55 {
56 }
57 
58 JSInternalPromise* JSInternalPromiseDeferred::promise() const
59 {
60     return jsCast<JSInternalPromise*>(Base::promise());
61 }
62 
63 JSInternalPromise* JSInternalPromiseDeferred::resolve(ExecState* exec, JSValue value)
64 {
65     Base::resolve(exec, value);
66     return promise();
67 }
68 
69 JSInternalPromise* JSInternalPromiseDeferred::reject(ExecState* exec, JSValue reason)
70 {
71     Base::reject(exec, reason);
72     return promise();
73 }
74 
75 JSInternalPromise* JSInternalPromiseDeferred::reject(ExecState* exec, Exception* reason)
76 {
77     return reject(exec, reason->value());
78 }
