42 namespace WebCore {
43 
44 class MessagePort;
45 class ServiceWorker;
46 class ServiceWorkerClient;
47 
48 using ExtendableMessageEventSource = Variant<RefPtr<ServiceWorkerClient>, RefPtr<ServiceWorker>, RefPtr<MessagePort>>;
49 
50 class ExtendableMessageEvent final : public ExtendableEvent {
51     WTF_MAKE_ISO_ALLOCATED(ExtendableMessageEvent);
52 public:
53     struct Init : ExtendableEventInit {
54         JSC::JSValue data;
55         String origin;
56         String lastEventId;
57         Optional<ExtendableMessageEventSource> source;
58         Vector<RefPtr<MessagePort>> ports;
59     };
60 
61     static Ref<ExtendableMessageEvent> create(JSC::JSGlobalObject& state, const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
62     {
63         return adoptRef(*new ExtendableMessageEvent(state, type, initializer, isTrusted));
64     }
65 
66     static Ref<ExtendableMessageEvent> create(Vector<RefPtr<MessagePort>>&&, RefPtr<SerializedScriptValue>&&, const String& origin = { }, const String& lastEventId = { }, Optional<ExtendableMessageEventSource>&& source = WTF::nullopt);
67 
68     ~ExtendableMessageEvent();
69 
70     SerializedScriptValue* data() const { return m_data.get(); }
71     const String& origin() const { return m_origin; }
72     const String& lastEventId() const { return m_lastEventId; }
73     const Optional<ExtendableMessageEventSource>& source() const { return m_source; }
74     const Vector<RefPtr<MessagePort>>& ports() const { return m_ports; }
75 
76     EventInterface eventInterface() const final { return ExtendableMessageEventInterfaceType; }
77 
78 private:
79     ExtendableMessageEvent(JSC::JSGlobalObject&, const AtomString&, const Init&, IsTrusted);
80     ExtendableMessageEvent(RefPtr<SerializedScriptValue>&& data, const String& origin, const String& lastEventId, Optional<ExtendableMessageEventSource>&&, Vector<RefPtr<MessagePort>>&&);
81 
82     RefPtr<SerializedScriptValue> m_data;
83     String m_origin;
84     String m_lastEventId;
85     Optional<ExtendableMessageEventSource> m_source;
86     Vector<RefPtr<MessagePort>> m_ports;
87 };
88 
89 } // namespace WebCore
