 45     VideoTextureCopierCV(GraphicsContextGLOpenGL&);

 47

 51     typedef CVOpenGLTextureRef TextureType;

 53 
 54     bool copyImageToPlatformTexture(CVPixelBufferRef, size_t width, size_t height, PlatformGLObject outputTexture, GCGLenum outputTarget, GCGLint level, GCGLenum internalFormat, GCGLenum format, GCGLenum type, bool premultiplyAlpha, bool flipY);
 55     bool copyVideoTextureToPlatformTexture(TextureType, size_t width, size_t height, PlatformGLObject outputTexture, GCGLenum outputTarget, GCGLint level, GCGLenum internalFormat, GCGLenum format, GCGLenum type, bool premultiplyAlpha, bool flipY, bool swapColorChannels = false);

 57     GraphicsContextGLOpenGL& context() { return m_context; }
 58 
 59 private:
 60     bool copyVideoTextureToPlatformTexture(PlatformGLObject inputTexture, GCGLenum inputTarget, size_t width, size_t height, PlatformGLObject outputTexture, GCGLenum outputTarget, GCGLint level, GCGLenum internalFormat, GCGLenum format, GCGLenum type, bool premultiplyAlpha, bool flipY, bool swapColorChannels);

 62     bool initializeContextObjects();
 63     bool initializeUVContextObjects();

 66     unsigned lastTextureSeed(GCGLuint texture)

 68         auto iterator = m_lastTextureSeed.find(texture);

 70     }

 81     void detachIOSurfaceFromTexture(void* handle);

 89     PlatformGLObject m_framebuffer { 0 };

106     GCGLint m_uvTextureSizeUniformLocation { -1 };

114     TextureSeedMap m_lastTextureSeed;
