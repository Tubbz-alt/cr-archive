61 void ToFTLDeferredCompilationCallback::compilationDidComplete(
62     CodeBlock* codeBlock, CodeBlock* profiledDFGCodeBlock, CompilationResult result)
63 {
64     if (Options::verboseOSR()) {
65         dataLog(
66             "Optimizing compilation of ", codeBlock, " (for ", profiledDFGCodeBlock,
67             ") result: ", result, "\n");
68     }
69 
70     if (profiledDFGCodeBlock->replacement() != profiledDFGCodeBlock) {
71         if (Options::verboseOSR()) {
72             dataLog(
73                 "Dropping FTL code block ", codeBlock, " on the floor because the "
74                 "DFG code block ", profiledDFGCodeBlock, " was jettisoned.\n");
75         }
76         return;
77     }
78 
79     if (result == CompilationSuccessful)
80         codeBlock->ownerExecutable()->installCode(codeBlock);
81 
82     profiledDFGCodeBlock->jitCode()->dfg()->setOptimizationThresholdBasedOnCompilationResult(
83         profiledDFGCodeBlock, result);
84 
85     DeferredCompilationCallback::compilationDidComplete(codeBlock, profiledDFGCodeBlock, result);
86 }
