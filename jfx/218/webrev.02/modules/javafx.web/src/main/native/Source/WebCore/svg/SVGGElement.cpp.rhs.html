52 RenderPtr<RenderElement> SVGGElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
53 {
54     // SVG 1.1 testsuite explicitly uses constructs like <g display="none"><linearGradient>
55     // We still have to create renderers for the <g> & <linearGradient> element, though the
56     // subtree may be hidden - we only want the resource renderers to exist so they can be
57     // referenced from somewhere else.
58     if (style.display() == DisplayType::None)
59         return createRenderer<RenderSVGHiddenContainer>(*this, WTFMove(style));
60 
61     return createRenderer<RenderSVGTransformableContainer>(*this, WTFMove(style));
62 }
63 
64 bool SVGGElement::rendererIsNeeded(const RenderStyle&)
65 {
66     // Unlike SVGElement::rendererIsNeeded(), we still create renderers, even if
67     // display is set to 'none' - which is special to SVG <g> container elements.
68     return parentOrShadowHostElement() && parentOrShadowHostElement()->isSVGElement();
69 }
