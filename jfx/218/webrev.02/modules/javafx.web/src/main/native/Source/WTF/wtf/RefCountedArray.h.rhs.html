 26 #pragma once
 27 
 28 #include <wtf/DumbPtrTraits.h>
 29 #include <wtf/FastMalloc.h>
 30 #include <wtf/StdLibExtras.h>
 31 #include <wtf/Vector.h>
 32 
 33 // This implements a reference counted array for POD** values, which is optimized for:
 34 // - An empty array only uses one word.
 35 // - A copy of the array only uses one word (i.e. assignment means aliasing).
 36 // - The vector can't grow beyond 2^32-1 elements.
 37 // - In all other regards this has similar space usage to a Vector.
 38 //
 39 // ** This could be modified to support non-POD values quite easily. It just
 40 //    hasn't been, so far, because there has been no need. Moreover, even now,
 41 //    it's used for things that aren't quite POD according to the official
 42 //    defintion, such as JSC::Instruction.
 43 
 44 namespace WTF {
 45 
 46 DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(RefCountedArray);
 47 
 48 template<typename T, typename PtrTraits = DumbPtrTraits<T>>
 49 class RefCountedArray {
 50     enum CommonCopyConstructorTag { CommonCopyConstructor };
 51 
 52 public:
 53     RefCountedArray() = default;
 54 
 55     RefCountedArray(const RefCountedArray& other)
 56         : RefCountedArray(CommonCopyConstructor, other)
 57     { }
 58 
 59     template<typename OtherTraits>
 60     RefCountedArray(const RefCountedArray<T, OtherTraits>& other)
 61         : RefCountedArray(CommonCopyConstructor, other)
 62     { }
 63 
 64     explicit RefCountedArray(size_t size)
 65     {
 66         if (!size) {
 67             // NOTE: JSC's LowLevelInterpreter relies on this being nullptr when the size is zero.
 68             PtrTraits::exchange(m_data, nullptr);
 69             return;
 70         }
 71 
 72         T* data = (static_cast<Header*>(RefCountedArrayMalloc::malloc(Header::size() + sizeof(T) * size)))->payload();
 73         m_data = data;
 74         Header::fromPayload(data)->refCount = 1;
 75         Header::fromPayload(data)->length = size;
 76         ASSERT(Header::fromPayload(data)->length == size);
 77         VectorTypeOperations<T>::initializeIfNonPOD(begin(), end());
 78     }
 79 
 80     template<typename OtherTraits = PtrTraits>
 81     RefCountedArray<T, OtherTraits> clone() const
 82     {
 83         RefCountedArray<T, OtherTraits> result(size());
 84         const T* data = this->data();
 85         T* resultData = result.data();
 86         for (unsigned i = size(); i--;)
 87             resultData[i] = data[i];
 88         return result;
 89     }
 90 
 91     template<size_t inlineCapacity, typename OverflowHandler>
 92     explicit RefCountedArray(const Vector<T, inlineCapacity, OverflowHandler>& other)
 93     {
 94         if (other.isEmpty()) {
 95             PtrTraits::exchange(m_data, nullptr);
 96             return;
 97         }
 98 
 99         T* data = (static_cast<Header*>(RefCountedArrayMalloc::malloc(Header::size() + sizeof(T) * other.size())))->payload();
100         m_data = data;
101         Header::fromPayload(data)->refCount = 1;
102         Header::fromPayload(data)->length = other.size();
103         ASSERT(Header::fromPayload(data)->length == other.size());
104         VectorTypeOperations<T>::uninitializedCopy(other.begin(), other.end(), data);
105     }
106 
107     template<typename OtherTraits = PtrTraits>
108     RefCountedArray& operator=(const RefCountedArray<T, OtherTraits>& other)
109     {
110         return assign<OtherTraits>(other);
111     }
112 
113     RefCountedArray& operator=(const RefCountedArray& other)
114     {
115         return assign<PtrTraits>(other);
116     }
117 
118     template<size_t inlineCapacity, typename OverflowHandler>
119     RefCountedArray& operator=(const Vector<T, inlineCapacity, OverflowHandler>& other)
120     {
121         T* oldData = data();
122         if (other.isEmpty())
123             PtrTraits::exchange(m_data, nullptr);
124         else {
125             T* data = (static_cast<Header*>(RefCountedArrayMalloc::malloc(Header::size() + sizeof(T) * other.size())))->payload();
126             m_data = data;
127             Header::fromPayload(data)->refCount = 1;
128             Header::fromPayload(data)->length = other.size();
129             ASSERT(Header::fromPayload(data)->length == other.size());
130             VectorTypeOperations<T>::uninitializedCopy(other.begin(), other.end(), data);
131         }
132         if (!oldData)
133             return *this;
134         if (--Header::fromPayload(oldData)->refCount)
135             return *this;
136         VectorTypeOperations<T>::destruct(oldData, oldData + Header::fromPayload(oldData)->length);
137 
138         RefCountedArrayMalloc::free(Header::fromPayload(oldData));
139         return *this;
140     }
141 
142     ~RefCountedArray()
143     {
144         if (!m_data)
145             return;
146         T* data = this->data();
147         if (--Header::fromPayload(data)->refCount)
148             return;
149         VectorTypeOperations<T>::destruct(begin(), end());
150         RefCountedArrayMalloc::free(Header::fromPayload(data));
151     }
152 
153     unsigned refCount() const
154     {
155         if (!m_data)
156             return 0;
157         return Header::fromPayload(data())->refCount;
158     }
159 
160     size_t size() const
161     {
162         if (!m_data)
163             return 0;
164         return Header::fromPayload(data())->length;
165     }
166 
167     bool isEmpty() const { return size() == 0; }
168 
169     size_t byteSize() const { return size() * sizeof(T); }
170 
171     T* data() { return PtrTraits::unwrap(m_data); }
172     T* begin() { return data(); }
173     T* end()
174     {
175         if (!m_data)
176             return 0;
177         T* data = this->data();
178         return data + Header::fromPayload(data)->length;
179     }
180 
181     const T* data() const { return const_cast<RefCountedArray*>(this)->data(); }
182     const T* begin() const { return const_cast<RefCountedArray*>(this)->begin(); }
183     const T* end() const { return const_cast<RefCountedArray*>(this)->end(); }
184 
185     T& at(size_t i)
186     {
187         ASSERT_WITH_SECURITY_IMPLICATION(i < size());
188         return begin()[i];
189     }
190 
191     const T& at(size_t i) const
192     {
193         ASSERT_WITH_SECURITY_IMPLICATION(i < size());
194         return begin()[i];
195     }
196 
197     T& operator[](size_t i) { return at(i); }
198     const T& operator[](size_t i) const { return at(i); }
199 
200     T& first() { return (*this)[0]; }
201     const T& first() const { return (*this)[0]; }
202     T& last() { return (*this)[size() - 1]; }
203     const T& last() const { return (*this)[size() - 1]; }
204 
205     template<typename OtherTraits = PtrTraits>
206     bool operator==(const RefCountedArray<T, OtherTraits>& other) const
207     {
208         T* data = const_cast<T*>(this->data());
209         T* otherData = const_cast<T*>(other.data());
210         if (data == otherData)
211             return true;
212         if (!data || !otherData)
213             return false;
214         unsigned length = Header::fromPayload(data)->length;
215         if (length != Header::fromPayload(otherData)->length)
216             return false;
217         for (unsigned i = 0; i < length; ++i) {
218             if (data[i] != otherData[i])
219                 return false;
220         }
221         return true;
222     }
223 
224     bool operator==(const RefCountedArray& other) const { return this->operator==<PtrTraits>(other); }
225 
226 private:
227     template<typename OtherTraits = PtrTraits>
228     RefCountedArray& assign(const RefCountedArray<T, OtherTraits>& other)
229     {
230         T* oldData = data();
231         T* otherData = const_cast<T*>(other.data());
232         if (otherData)
233             Header::fromPayload(otherData)->refCount++;
234         m_data = otherData;
235 
236         if (!oldData)
237             return *this;
238         if (--Header::fromPayload(oldData)->refCount)
239             return *this;
240         VectorTypeOperations<T>::destruct(oldData, oldData + Header::fromPayload(oldData)->length);
241 
242         RefCountedArrayMalloc::free(Header::fromPayload(oldData));
243         return *this;
244     }
245 
246     struct Header {
247         unsigned refCount;
248         unsigned length;
249 
250         static size_t size()
251         {
252             return (sizeof(Header) + 7) & ~7;
253         }
254 
255         T* payload()
256         {
257             char* result = reinterpret_cast<char*>(this) + size();
258             ASSERT(!(bitwise_cast<uintptr_t>(result) & 7));
259             return reinterpret_cast_ptr<T*>(result);
260         }
261 
262         static Header* fromPayload(T* payload)
263         {
264             return reinterpret_cast_ptr<Header*>(reinterpret_cast<char*>(payload) - size());
265         }
266 
267         static const Header* fromPayload(const T* payload)
268         {
269             return fromPayload(const_cast<T*>(payload));
270         }
271     };
272 
273     template<typename OtherTraits>
274     RefCountedArray(CommonCopyConstructorTag, const RefCountedArray<T, OtherTraits>& other)
275         : m_data(const_cast<T*>(other.data()))
276     {
277         if (m_data)
278             Header::fromPayload(data())->refCount++;
279     }
280 
281     friend class JSC::LLIntOffsetsExtractor;
282     typename PtrTraits::StorageType m_data { nullptr };
283 };
284 
285 } // namespace WTF
286
