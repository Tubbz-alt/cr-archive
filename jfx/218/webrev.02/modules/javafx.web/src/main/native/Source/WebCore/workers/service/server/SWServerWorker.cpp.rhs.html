 50 // FIXME: Use r-value references for script and contentSecurityPolicy
 51 SWServerWorker::SWServerWorker(SWServer& server, SWServerRegistration& registration, const URL& scriptURL, const String& script, const ContentSecurityPolicyResponseHeaders& contentSecurityPolicy, String&& referrerPolicy, WorkerType type, ServiceWorkerIdentifier identifier, HashMap<URL, ServiceWorkerContextData::ImportedScript>&& scriptResourceMap)
 52     : m_server(makeWeakPtr(server))
 53     , m_registrationKey(registration.key())
 54     , m_registration(makeWeakPtr(registration))
 55     , m_data { identifier, scriptURL, ServiceWorkerState::Redundant, type, registration.identifier() }
 56     , m_script(script)
 57     , m_contentSecurityPolicy(contentSecurityPolicy)
 58     , m_referrerPolicy(WTFMove(referrerPolicy))
 59     , m_registrableDomain(m_data.scriptURL)
 60     , m_scriptResourceMap(WTFMove(scriptResourceMap))
 61 {
 62     m_data.scriptURL.removeFragmentIdentifier();
 63 
 64     auto result = allWorkers().add(identifier, this);
 65     ASSERT_UNUSED(result, result.isNewEntry);
 66 
 67     ASSERT(m_server->getRegistration(m_registrationKey) == &registration);
 68 }
 69 
 70 SWServerWorker::~SWServerWorker()
 71 {
 72     ASSERT(m_whenActivatedHandlers.isEmpty());
 73     callWhenActivatedHandler(false);
 74 
 75     auto taken = allWorkers().take(identifier());
 76     ASSERT_UNUSED(taken, taken == this);
 77 }

120 void SWServerWorker::didFinishInstall(const Optional<ServiceWorkerJobDataIdentifier>& jobDataIdentifier, bool wasSuccessful)
121 {
122     auto state = this->state();
123     if (state == ServiceWorkerState::Redundant)
124         return;
125 
126     ASSERT(m_server);
127     RELEASE_ASSERT(state == ServiceWorkerState::Installing);
128     if (m_server)
129         m_server->didFinishInstall(jobDataIdentifier, *this, wasSuccessful);
130 }
131 
132 void SWServerWorker::didFinishActivation()
133 {
134     auto state = this->state();
135     if (state == ServiceWorkerState::Redundant)
136         return;
137 
138     ASSERT(m_server);
139     RELEASE_ASSERT(state == ServiceWorkerState::Activating);
140     if (m_server)
141         m_server->didFinishActivation(*this);
142 }

203         return;
204 
205     // Do tryClear/tryActivate, as per https://w3c.github.io/ServiceWorker/#wait-until-method.

214 void SWServerWorker::whenActivated(CompletionHandler<void(bool)>&& handler)
215 {
216     if (state() == ServiceWorkerState::Activated) {
217         handler(true);
218         return;
219     }
220     ASSERT(state() == ServiceWorkerState::Activating);
221     m_whenActivatedHandlers.append(WTFMove(handler));
222 }
223 
224 void SWServerWorker::setState(ServiceWorkerState state)
225 {
226     if (state == ServiceWorkerState::Redundant)
227         terminate();
228 
229     m_data.state = state;
230 
231     ASSERT(m_registration || state == ServiceWorkerState::Redundant);
232     if (m_registration) {
233         m_registration->forEachConnection([&](auto& connection) {

234             connection.updateWorkerStateInClient(this->identifier(), state);
235         });
236     }
237 
238     if (state == ServiceWorkerState::Activated || state == ServiceWorkerState::Redundant)
239         callWhenActivatedHandler(state == ServiceWorkerState::Activated);
240 }
241 
242 void SWServerWorker::callWhenActivatedHandler(bool success)
243 {
244     auto whenActivatedHandlers = WTFMove(m_whenActivatedHandlers);
245     for (auto& handler : whenActivatedHandlers)
246         handler(success);
247 }
248 
249 void SWServerWorker::setState(State state)
250 {
251     ASSERT(state != State::Running || m_registration);
252     m_state = state;
253 
254     switch (state) {
255     case State::Running:
256         m_shouldSkipHandleFetch = false;
257         break;
258     case State::Terminating:
259     case State::NotRunning:
260         callWhenActivatedHandler(false);
261         break;
262     }
263 }
