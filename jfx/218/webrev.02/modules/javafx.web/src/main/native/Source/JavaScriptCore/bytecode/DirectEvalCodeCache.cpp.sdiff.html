33 void DirectEvalCodeCache::setSlow(ExecState* exec, JSCell* owner, const String& evalSource, CallSiteIndex callSiteIndex, DirectEvalExecutable* evalExecutable)
34 {
35     if (!evalExecutable->allowDirectEvalCache())
36         return;
37 
38     LockHolder locker(m_lock);
39     m_cacheMap.set(CacheKey(evalSource, callSiteIndex), WriteBarrier<DirectEvalExecutable>(exec->vm(), owner, evalExecutable));
40 }
41 
42 void DirectEvalCodeCache::clear()
43 {
44     LockHolder locker(m_lock);
45     m_cacheMap.clear();
46 }

48 void DirectEvalCodeCache::visitAggregate(SlotVisitor& visitor)
49 {
50     LockHolder locker(m_lock);
51     EvalCacheMap::iterator end = m_cacheMap.end();
52     for (EvalCacheMap::iterator ptr = m_cacheMap.begin(); ptr != end; ++ptr)
53         visitor.append(ptr->value);
54 }
