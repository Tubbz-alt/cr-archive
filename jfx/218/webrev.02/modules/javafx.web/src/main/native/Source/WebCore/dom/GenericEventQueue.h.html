35 namespace WebCore {

37 class Event;
38 class EventTarget;
39 class Timer;
40 class ScriptExecutionContext;

42 // All instances of MainThreadGenericEventQueue use a shared Timer for dispatching events.
43 // FIXME: We should port call sites to the HTML event loop and remove this class.
44 class MainThreadGenericEventQueue : public ActiveDOMObject {
45     WTF_MAKE_FAST_ALLOCATED;
46 public:
47     static UniqueRef<MainThreadGenericEventQueue> create(EventTarget&);
48 
49     void enqueueEvent(RefPtr<Event>&&);
50     void close();
51 
52     void cancelAllEvents();
53     bool hasPendingEvents() const;
54     bool hasPendingEventsOfType(const AtomString&) const;
55 
56     void setPaused(bool);
57 
58     bool isSuspended() const { return m_isSuspended; }
59 
60 private:
61     friend UniqueRef<MainThreadGenericEventQueue> WTF::makeUniqueRefWithoutFastMallocCheck<MainThreadGenericEventQueue, WebCore::EventTarget&>(WebCore::EventTarget&);
62     explicit MainThreadGenericEventQueue(EventTarget&);
63 
64     void dispatchOneEvent();
65 
66     const char* activeDOMObjectName() const final;
67     void stop() final;
68     void suspend(ReasonForSuspension) final;
69     void resume() final;
70 
71     void rescheduleAllEventsIfNeeded();
72     bool isSuspendedOrPausedByClient() const { return m_isSuspended || m_isPausedByClient; }
73 
74     EventTarget& m_owner;
75     UniqueRef<GenericTaskQueue<Timer>> m_taskQueue;
76     Deque<RefPtr<Event>> m_pendingEvents;
77     bool m_isClosed { false };
78     bool m_isPausedByClient { false };
79     bool m_isSuspended { false };
80 };
81 
82 } // namespace WebCore
