 33 // Checking sorting is too slow for general use; turn it on explicitly when working on this class.
 34 #undef CHECK_SORTING
 35 
 36 #ifdef CHECK_SORTING
 37 #define ASSERT_SORTED(begin, end) ASSERT(std::is_sorted(begin, end, cueSortsBefore))
 38 #else
 39 #define ASSERT_SORTED(begin, end) ((void)0)
 40 #endif

 42 namespace WebCore {
 43 
 44 static inline bool cueSortsBefore(const RefPtr<TextTrackCue>& a, const RefPtr<TextTrackCue>& b)
 45 {
 46     if (a->startMediaTime() < b->startMediaTime())
 47         return true;
 48 
 49     return a->startMediaTime() == b->startMediaTime() && a->endMediaTime() > b->endMediaTime();
 50 }
 51 
 52 unsigned TextTrackCueList::cueIndex(const TextTrackCue& cue) const
 53 {
 54     ASSERT(m_vector.contains(&cue));
 55     return m_vector.find(&cue);
 56 }
 57 
 58 TextTrackCue* TextTrackCueList::item(unsigned index) const
 59 {
 60     if (index >= m_vector.size())
 61         return nullptr;
 62     return m_vector[index].get();
 63 }
 64 
 65 TextTrackCue* TextTrackCueList::getCueById(const String& id) const
 66 {
 67     for (auto& cue : m_vector) {
 68         if (cue->id() == id)
 69             return cue.get();
 70     }
 71     return nullptr;
 72 }
 73 
 74 TextTrackCueList& TextTrackCueList::activeCues()
 75 {
 76     if (!m_activeCues)
 77         m_activeCues = create();
 78 
 79     Vector<RefPtr<TextTrackCue>> activeCuesVector;
 80     for (auto& cue : m_vector) {
 81         if (cue->isActive())
 82             activeCuesVector.append(cue);
 83     }
 84     ASSERT_SORTED(activeCuesVector.begin(), activeCuesVector.end());
 85     m_activeCues->m_vector = WTFMove(activeCuesVector);
 86 
 87     // FIXME: This list of active cues is not updated as cues are added, removed, become active, and become inactive.
 88     // Instead it is only updated each time this function is called again. That is not consistent with other dynamic DOM lists.
 89     return *m_activeCues;
 90 }
 91 
 92 void TextTrackCueList::add(Ref<TextTrackCue>&& cue)
 93 {
 94     ASSERT(!m_vector.contains(cue.ptr()));
 95     ASSERT(cue->startMediaTime() >= MediaTime::zeroTime());
 96     ASSERT(cue->endMediaTime() >= MediaTime::zeroTime());
 97 
 98     RefPtr<TextTrackCue> cueRefPtr { WTFMove(cue) };
 99     unsigned insertionPosition = std::upper_bound(m_vector.begin(), m_vector.end(), cueRefPtr, cueSortsBefore) - m_vector.begin();
100     ASSERT_SORTED(m_vector.begin(), m_vector.end());
101     m_vector.insert(insertionPosition, WTFMove(cueRefPtr));
102     ASSERT_SORTED(m_vector.begin(), m_vector.end());
103 }
104 
105 void TextTrackCueList::remove(TextTrackCue& cue)
106 {
107     ASSERT_SORTED(m_vector.begin(), m_vector.end());
108     m_vector.remove(cueIndex(cue));
109     ASSERT_SORTED(m_vector.begin(), m_vector.end());
110 }
111 
112 void TextTrackCueList::clear()
113 {
114     m_vector.clear();
115     if (m_activeCues)
116         m_activeCues->m_vector.clear();
117 }
118 
119 void TextTrackCueList::updateCueIndex(const TextTrackCue& cue)
120 {
121     auto cuePosition = m_vector.begin() + cueIndex(cue);
122     auto afterCuePosition = cuePosition + 1;
123 
124     ASSERT_SORTED(m_vector.begin(), cuePosition);
125     ASSERT_SORTED(afterCuePosition, m_vector.end());
126 
127     auto reinsertionPosition = std::upper_bound(m_vector.begin(), cuePosition, *cuePosition, cueSortsBefore);
128     if (reinsertionPosition != cuePosition)
129         std::rotate(reinsertionPosition, cuePosition, afterCuePosition);
130     else {
131         reinsertionPosition = std::upper_bound(afterCuePosition, m_vector.end(), *cuePosition, cueSortsBefore);
132         if (reinsertionPosition != afterCuePosition)
133             std::rotate(cuePosition, afterCuePosition, reinsertionPosition);
134     }
135 
136     ASSERT_SORTED(m_vector.begin(), m_vector.end());
137 }
