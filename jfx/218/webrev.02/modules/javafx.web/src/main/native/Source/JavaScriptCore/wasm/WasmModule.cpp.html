 52 static Module::ValidationResult makeValidationResult(LLIntPlan& plan)
 53 {
 54     ASSERT(!plan.hasWork());
 55     if (plan.failed())
 56         return Unexpected<String>(plan.errorMessage());
 57     return Module::ValidationResult(Module::create(plan));
 58 }
 59 
 60 static Plan::CompletionTask makeValidationCallback(Module::AsyncValidationCallback&& callback)
 61 {
 62     return createSharedTask<Plan::CallbackType>([callback = WTFMove(callback)] (Plan& plan) {
 63         ASSERT(!plan.hasWork());
 64         callback->run(makeValidationResult(static_cast<LLIntPlan&>(plan)));
 65     });
 66 }

 68 Module::ValidationResult Module::validateSync(Context* context, Vector<uint8_t>&& source)
 69 {
 70     Ref<LLIntPlan> plan = adoptRef(*new LLIntPlan(context, WTFMove(source), EntryPlan::Validation, Plan::dontFinalize()));
 71     Wasm::ensureWorklist().enqueue(plan.get());
 72     plan->waitForCompletion();
 73     return makeValidationResult(plan.get());
 74 }
 75 
 76 void Module::validateAsync(Context* context, Vector<uint8_t>&& source, Module::AsyncValidationCallback&& callback)
 77 {
 78     Ref<Plan> plan = adoptRef(*new LLIntPlan(context, WTFMove(source), EntryPlan::Validation, makeValidationCallback(WTFMove(callback))));
 79     Wasm::ensureWorklist().enqueue(WTFMove(plan));
 80 }

 82 Ref<CodeBlock> Module::getOrCreateCodeBlock(Context* context, MemoryMode mode)
 83 {
 84     RefPtr<CodeBlock> codeBlock;
 85     auto locker = holdLock(m_lock);
 86     codeBlock = m_codeBlocks[static_cast<uint8_t>(mode)];
 87     // If a previous attempt at a compile errored out, let's try again.
 88     // Compilations from valid modules can fail because OOM and cancellation.
 89     // It's worth retrying.
 90     // FIXME: We might want to back off retrying at some point:
 91     // https://bugs.webkit.org/show_bug.cgi?id=170607
 92     if (!codeBlock || (codeBlock->compilationFinished() && !codeBlock->runnable())) {
 93         RefPtr<LLIntCallees> llintCallees = nullptr;
 94         if (Options::useWasmLLInt())
 95             llintCallees = m_llintCallees;
 96         codeBlock = CodeBlock::create(context, mode, const_cast<ModuleInformation&>(moduleInformation()), llintCallees);
 97         m_codeBlocks[static_cast<uint8_t>(mode)] = codeBlock;
 98     }
 99     return codeBlock.releaseNonNull();
100 }
101 
102 Ref<CodeBlock> Module::compileSync(Context* context, MemoryMode mode)
103 {
104     Ref<CodeBlock> codeBlock = getOrCreateCodeBlock(context, mode);
105     codeBlock->waitUntilFinished();
106     return codeBlock;
107 }
108 
109 void Module::compileAsync(Context* context, MemoryMode mode, CodeBlock::AsyncCompilationCallback&& task)
110 {
111     Ref<CodeBlock> codeBlock = getOrCreateCodeBlock(context, mode);
112     codeBlock->compileAsync(context, WTFMove(task));
113 }
