26 #pragma once
27 
28 #include "ObjectPropertyCondition.h"
29 #include "Watchpoint.h"
30 #include <wtf/FastMalloc.h>
31 #include <wtf/Noncopyable.h>
32 
33 namespace JSC {
34 
35 // FIXME: This isn't actually a Watchpoint. We should probably have a name which better reflects that:
36 // https://bugs.webkit.org/show_bug.cgi?id=202381
37 class AdaptiveInferredPropertyValueWatchpointBase {
38     WTF_MAKE_NONCOPYABLE(AdaptiveInferredPropertyValueWatchpointBase);
39     WTF_MAKE_FAST_ALLOCATED;
40 
41 public:
42     AdaptiveInferredPropertyValueWatchpointBase(const ObjectPropertyCondition&);
43 
44     const ObjectPropertyCondition& key() const { return m_key; }
45 
46     void install(VM&);
47 
48     virtual ~AdaptiveInferredPropertyValueWatchpointBase() = default;
49 
50     class StructureWatchpoint final : public Watchpoint {
51     public:
52         StructureWatchpoint()
53             : Watchpoint(Watchpoint::Type::AdaptiveInferredPropertyValueStructure)
54         { }
55 
56         void fireInternal(VM&, const FireDetail&);
57     };
58     // Own destructor may not be called. Keep members trivially destructible.
59     static_assert(sizeof(StructureWatchpoint) == sizeof(Watchpoint), "");
60 
61     class PropertyWatchpoint final : public Watchpoint {
62     public:
63         PropertyWatchpoint()
64             : Watchpoint(Watchpoint::Type::AdaptiveInferredPropertyValueProperty)
65         { }
66 
67         void fireInternal(VM&, const FireDetail&);
68     };
69     // Own destructor may not be called. Keep members trivially destructible.
70     static_assert(sizeof(PropertyWatchpoint) == sizeof(Watchpoint), "");
71 
72 protected:
73     virtual bool isValid() const;
74     virtual void handleFire(VM&, const FireDetail&) = 0;
75 
76 private:
77     void fire(VM&, const FireDetail&);
78 
79     ObjectPropertyCondition m_key;
80     StructureWatchpoint m_structureWatchpoint;
81     PropertyWatchpoint m_propertyWatchpoint;
82 };
83 
84 } // namespace JSC
