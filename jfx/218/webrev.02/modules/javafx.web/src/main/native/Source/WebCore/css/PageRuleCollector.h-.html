22 #pragma once
23 
24 #include "DocumentRuleSets.h"
25 #include "StyleResolver.h"
26 #include <wtf/Forward.h>
27 
28 namespace WebCore {
29 
30 class StyleRulePage;
31 
32 class PageRuleCollector {
33 public:
34     PageRuleCollector(StyleResolver::State& state, DocumentRuleSets& ruleSets)
35         : m_state(state)
36         , m_ruleSets(ruleSets) { }
37 
38     void matchAllPageRules(int pageIndex);
39     StyleResolver::MatchResult& matchedResult() { return m_result; }
40 
41 private:
42     bool isLeftPage(int pageIndex) const;
43     bool isRightPage(int pageIndex) const { return !isLeftPage(pageIndex); }
44     bool isFirstPage(int pageIndex) const;
45     String pageName(int pageIndex) const;
46 
47     void matchPageRules(RuleSet* rules, bool isLeftPage, bool isFirstPage, const String& pageName);
48     void matchPageRulesForList(Vector<StyleRulePage*>& matchedRules, const Vector<StyleRulePage*>& rules, bool isLeftPage, bool isFirstPage, const String& pageName);
49 
50     const StyleResolver::State& m_state;
51     DocumentRuleSets& m_ruleSets;
52 
53     StyleResolver::MatchResult m_result;
54 };
55 
56 } // namespace WebCore
