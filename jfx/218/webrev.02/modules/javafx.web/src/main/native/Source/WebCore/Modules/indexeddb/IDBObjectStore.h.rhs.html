 46 class IDBIndex;

 55 enum class ObjectStoreOverwriteMode;

 58 class IDBObjectStore final : public ActiveDOMObject {
 59     WTF_MAKE_NONCOPYABLE(IDBObjectStore);
 60     WTF_MAKE_FAST_ALLOCATED;
 61 public:
 62     IDBObjectStore(ScriptExecutionContext&, const IDBObjectStoreInfo&, IDBTransaction&);
 63     ~IDBObjectStore();
 64 
 65     const String& name() const;
 66     ExceptionOr<void> setName(const String&);
 67     const Optional<IDBKeyPath>& keyPath() const;
 68     Ref<DOMStringList> indexNames() const;
 69     IDBTransaction& transaction();
 70     bool autoIncrement() const;
 71 
 72     struct IndexParameters {
 73         bool unique;
 74         bool multiEntry;
 75     };
 76 
 77     ExceptionOr<Ref<IDBRequest>> openCursor(JSC::JSGlobalObject&, RefPtr<IDBKeyRange>&&, IDBCursorDirection);
 78     ExceptionOr<Ref<IDBRequest>> openCursor(JSC::JSGlobalObject&, JSC::JSValue key, IDBCursorDirection);
 79     ExceptionOr<Ref<IDBRequest>> openKeyCursor(JSC::JSGlobalObject&, RefPtr<IDBKeyRange>&&, IDBCursorDirection);
 80     ExceptionOr<Ref<IDBRequest>> openKeyCursor(JSC::JSGlobalObject&, JSC::JSValue key, IDBCursorDirection);
 81     ExceptionOr<Ref<IDBRequest>> get(JSC::JSGlobalObject&, JSC::JSValue key);
 82     ExceptionOr<Ref<IDBRequest>> get(JSC::JSGlobalObject&, IDBKeyRange*);
 83     ExceptionOr<Ref<IDBRequest>> getKey(JSC::JSGlobalObject&, JSC::JSValue key);
 84     ExceptionOr<Ref<IDBRequest>> getKey(JSC::JSGlobalObject&, IDBKeyRange*);
 85     ExceptionOr<Ref<IDBRequest>> add(JSC::JSGlobalObject&, JSC::JSValue, JSC::JSValue key);
 86     ExceptionOr<Ref<IDBRequest>> put(JSC::JSGlobalObject&, JSC::JSValue, JSC::JSValue key);
 87     ExceptionOr<Ref<IDBRequest>> deleteFunction(JSC::JSGlobalObject&, IDBKeyRange*);
 88     ExceptionOr<Ref<IDBRequest>> deleteFunction(JSC::JSGlobalObject&, JSC::JSValue key);
 89     ExceptionOr<Ref<IDBRequest>> clear(JSC::JSGlobalObject&);
 90     ExceptionOr<Ref<IDBIndex>> createIndex(JSC::JSGlobalObject&, const String& name, IDBKeyPath&&, const IndexParameters&);
 91     ExceptionOr<Ref<IDBIndex>> index(const String& name);
 92     ExceptionOr<void> deleteIndex(const String& name);
 93     ExceptionOr<Ref<IDBRequest>> count(JSC::JSGlobalObject&, IDBKeyRange*);
 94     ExceptionOr<Ref<IDBRequest>> count(JSC::JSGlobalObject&, JSC::JSValue key);
 95     ExceptionOr<Ref<IDBRequest>> getAll(JSC::JSGlobalObject&, RefPtr<IDBKeyRange>&&, Optional<uint32_t> count);
 96     ExceptionOr<Ref<IDBRequest>> getAll(JSC::JSGlobalObject&, JSC::JSValue key, Optional<uint32_t> count);
 97     ExceptionOr<Ref<IDBRequest>> getAllKeys(JSC::JSGlobalObject&, RefPtr<IDBKeyRange>&&, Optional<uint32_t> count);
 98     ExceptionOr<Ref<IDBRequest>> getAllKeys(JSC::JSGlobalObject&, JSC::JSValue key, Optional<uint32_t> count);
 99 
100     ExceptionOr<Ref<IDBRequest>> putForCursorUpdate(JSC::JSGlobalObject&, JSC::JSValue, RefPtr<IDBKey>);
101 
102     void markAsDeleted();
103     bool isDeleted() const { return m_deleted; }
104 
105     const IDBObjectStoreInfo& info() const { return m_info; }
106 
107     void rollbackForVersionChangeAbort();
108 
109     void ref();
110     void deref();
111 
112     void visitReferencedIndexes(JSC::SlotVisitor&) const;
113     void renameReferencedIndex(IDBIndex&, const String& newName);
114 
115     bool hasPendingActivity() const final;
116 
117 private:
118     enum class InlineKeyCheck { Perform, DoNotPerform };
119     ExceptionOr<Ref<IDBRequest>> putOrAdd(JSC::JSGlobalObject&, JSC::JSValue, RefPtr<IDBKey>, IndexedDB::ObjectStoreOverwriteMode, InlineKeyCheck);
120     ExceptionOr<Ref<IDBRequest>> doCount(JSC::JSGlobalObject&, const IDBKeyRangeData&);
121     ExceptionOr<Ref<IDBRequest>> doDelete(JSC::JSGlobalObject&, WTF::Function<ExceptionOr<RefPtr<IDBKeyRange>>()> &&);
122     ExceptionOr<Ref<IDBRequest>> doOpenCursor(JSC::JSGlobalObject&, IDBCursorDirection, WTF::Function<ExceptionOr<RefPtr<IDBKeyRange>>()>&&);
123     ExceptionOr<Ref<IDBRequest>> doOpenKeyCursor(JSC::JSGlobalObject&, IDBCursorDirection, WTF::Function<ExceptionOr<RefPtr<IDBKeyRange>>()>&&);
124     ExceptionOr<Ref<IDBRequest>> doGetAll(JSC::JSGlobalObject&, Optional<uint32_t> count, WTF::Function<ExceptionOr<RefPtr<IDBKeyRange>>()> &&);
125     ExceptionOr<Ref<IDBRequest>> doGetAllKeys(JSC::JSGlobalObject&, Optional<uint32_t> count, WTF::Function<ExceptionOr<RefPtr<IDBKeyRange>>()> &&);
126 
127     const char* activeDOMObjectName() const final;


128 
129     IDBObjectStoreInfo m_info;
130     IDBObjectStoreInfo m_originalInfo;
131 
132     // IDBObjectStore objects are always owned by their referencing IDBTransaction.
133     // ObjectStores will never outlive transactions so its okay to keep a raw C++ reference here.
134     IDBTransaction& m_transaction;
135 
136     bool m_deleted { false };
137 
138     mutable Lock m_referencedIndexLock;
139     HashMap<String, std::unique_ptr<IDBIndex>> m_referencedIndexes;
140     HashMap<uint64_t, std::unique_ptr<IDBIndex>> m_deletedIndexes;
141 };
