33 void scheduleDispatchFunctionsOnMainThread()
34 {
35     AttachThreadAsNonDaemonToJavaEnv autoAttach;
36     JNIEnv* env = autoAttach.env();
37     static JGClass jMainThreadCls(env->FindClass("com/sun/webkit/MainThread"));
38 
39     static jmethodID mid = env->GetStaticMethodID(
40             jMainThreadCls,
41             "fwkScheduleDispatchFunctions",
42             "()V");
43 
44     ASSERT(mid);
45 
46     env->CallStaticVoidMethod(jMainThreadCls, mid);
47     WTF::CheckAndClearException(env);
48 }
