 68 EncodedJSValue JSC_HOST_CALL webAssemblyModuleCustomSections(JSGlobalObject* globalObject, CallFrame* callFrame)
 69 {
 70     VM& vm = globalObject->vm();

 71     auto throwScope = DECLARE_THROW_SCOPE(vm);
 72 
 73     if (UNLIKELY(callFrame->argumentCount() < 2))
 74         return JSValue::encode(throwException(globalObject, throwScope, createNotEnoughArgumentsError(globalObject)));
 75 
 76     JSWebAssemblyModule* module = jsDynamicCast<JSWebAssemblyModule*>(vm, callFrame->uncheckedArgument(0));
 77     if (!module)
 78         return JSValue::encode(throwException(globalObject, throwScope, createTypeError(globalObject, "WebAssembly.Module.customSections called with non WebAssembly.Module argument"_s)));
 79 
 80     const String sectionNameString = callFrame->uncheckedArgument(1).getString(globalObject);
 81     RETURN_IF_EXCEPTION(throwScope, { });
 82 
 83     JSArray* result = constructEmptyArray(globalObject, nullptr);
 84     RETURN_IF_EXCEPTION(throwScope, { });
 85 
 86     const auto& customSections = module->moduleInformation().customSections;
 87     for (const Wasm::CustomSection& section : customSections) {
 88         if (String::fromUTF8(section.name) == sectionNameString) {
 89             auto buffer = ArrayBuffer::tryCreate(section.payload.data(), section.payload.size());
 90             if (!buffer)
 91                 return JSValue::encode(throwException(globalObject, throwScope, createOutOfMemoryError(globalObject)));
 92 
 93             result->push(globalObject, JSArrayBuffer::create(vm, globalObject->arrayBufferStructure(ArrayBufferSharingMode::Default), WTFMove(buffer)));
 94             RETURN_IF_EXCEPTION(throwScope, { });
 95         }
 96     }
 97 
 98     return JSValue::encode(result);
 99 }

101 EncodedJSValue JSC_HOST_CALL webAssemblyModuleImports(JSGlobalObject* globalObject, CallFrame* callFrame)
102 {
103     VM& vm = globalObject->vm();

104     auto throwScope = DECLARE_THROW_SCOPE(vm);
105 
106     JSWebAssemblyModule* module = jsDynamicCast<JSWebAssemblyModule*>(vm, callFrame->argument(0));
107     if (!module)
108         return JSValue::encode(throwException(globalObject, throwScope, createTypeError(globalObject, "WebAssembly.Module.imports called with non WebAssembly.Module argument"_s)));
109 
110     JSArray* result = constructEmptyArray(globalObject, nullptr);
111     RETURN_IF_EXCEPTION(throwScope, { });
112 
113     const auto& imports = module->moduleInformation().imports;
114     if (imports.size()) {
115         Identifier module = Identifier::fromString(vm, "module");
116         Identifier name = Identifier::fromString(vm, "name");
117         Identifier kind = Identifier::fromString(vm, "kind");
118         for (const Wasm::Import& imp : imports) {
119             JSObject* obj = constructEmptyObject(globalObject);
120             RETURN_IF_EXCEPTION(throwScope, { });
121             obj->putDirect(vm, module, jsString(vm, String::fromUTF8(imp.module)));
122             obj->putDirect(vm, name, jsString(vm, String::fromUTF8(imp.field)));
123             obj->putDirect(vm, kind, jsString(vm, String(makeString(imp.kind))));
124             result->push(globalObject, obj);
125             RETURN_IF_EXCEPTION(throwScope, { });
126         }
127     }
128 
129     return JSValue::encode(result);
130 }

161 static EncodedJSValue JSC_HOST_CALL constructJSWebAssemblyModule(JSGlobalObject* globalObject, CallFrame* callFrame)
162 {
163     VM& vm = globalObject->vm();
164     auto scope = DECLARE_THROW_SCOPE(vm);
165 
166     Vector<uint8_t> source = createSourceBufferFromValue(vm, globalObject, callFrame->argument(0));
167     RETURN_IF_EXCEPTION(scope, { });
168 
169     RELEASE_AND_RETURN(scope, JSValue::encode(WebAssemblyModuleConstructor::createModule(globalObject, callFrame, WTFMove(source))));
170 }
171 
172 static EncodedJSValue JSC_HOST_CALL callJSWebAssemblyModule(JSGlobalObject* globalObject, CallFrame*)
173 {
174     VM& vm = globalObject->vm();
175     auto scope = DECLARE_THROW_SCOPE(vm);
176     return JSValue::encode(throwConstructorCannotBeCalledAsFunctionTypeError(globalObject, scope, "WebAssembly.Module"));
177 }

179 JSWebAssemblyModule* WebAssemblyModuleConstructor::createModule(JSGlobalObject* globalObject, CallFrame* callFrame, Vector<uint8_t>&& buffer)
180 {
181     VM& vm = globalObject->vm();
182     auto scope = DECLARE_THROW_SCOPE(vm);
183 
184     auto* structure = InternalFunction::createSubclassStructure(globalObject, callFrame->jsCallee(), callFrame->newTarget(), globalObject->webAssemblyModuleStructure());
185     RETURN_IF_EXCEPTION(scope, nullptr);
186 
187     RELEASE_AND_RETURN(scope, JSWebAssemblyModule::createStub(vm, globalObject, structure, Wasm::Module::validateSync(&vm.wasmContext, WTFMove(buffer))));
188 }

190 WebAssemblyModuleConstructor* WebAssemblyModuleConstructor::create(VM& vm, Structure* structure, WebAssemblyModulePrototype* thisPrototype)
191 {
192     auto* constructor = new (NotNull, allocateCell<WebAssemblyModuleConstructor>(vm.heap)) WebAssemblyModuleConstructor(vm, structure);
193     constructor->finishCreation(vm, thisPrototype);
194     return constructor;
195 }

197 Structure* WebAssemblyModuleConstructor::createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
198 {
199     return Structure::create(vm, globalObject, prototype, TypeInfo(InternalFunctionType, StructureFlags), info());
200 }

202 void WebAssemblyModuleConstructor::finishCreation(VM& vm, WebAssemblyModulePrototype* prototype)
203 {
204     Base::finishCreation(vm, "Module"_s, NameAdditionMode::WithoutStructureTransition);
205     putDirectWithoutTransition(vm, vm.propertyNames->prototype, prototype, PropertyAttribute::DontEnum | PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly);
206     putDirectWithoutTransition(vm, vm.propertyNames->length, jsNumber(1), PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum);
207 }
