 28 #include "ActiveDOMObject.h"
 29 #include "CSSFontFace.h"
 30 #include "CSSPropertyNames.h"
 31 #include "IDLTypes.h"

 34 #include <wtf/WeakPtr.h>

 36 namespace JSC {
 37 class ArrayBuffer;
 38 class ArrayBufferView;
 39 }
 40 
 41 namespace WebCore {
 42 
 43 template<typename IDLType> class DOMPromiseProxyWithResolveCallback;
 44 
 45 class FontFace final : public RefCounted<FontFace>, public CanMakeWeakPtr<FontFace>, public ActiveDOMObject, private CSSFontFace::Client {
 46 public:
 47     struct Descriptors {
 48         String style;
 49         String weight;
 50         String stretch;
 51         String unicodeRange;

 52         String featureSettings;
 53         String display;
 54     };
 55 
 56     using Source = Variant<String, RefPtr<JSC::ArrayBuffer>, RefPtr<JSC::ArrayBufferView>>;
 57     static Ref<FontFace> create(Document&, const String& family, Source&&, const Descriptors&);
 58     static Ref<FontFace> create(CSSFontFace&);
 59     virtual ~FontFace();
 60 
 61     ExceptionOr<void> setFamily(Document&, const String&);
 62     ExceptionOr<void> setStyle(const String&);
 63     ExceptionOr<void> setWeight(const String&);
 64     ExceptionOr<void> setStretch(const String&);
 65     ExceptionOr<void> setUnicodeRange(const String&);

 66     ExceptionOr<void> setFeatureSettings(const String&);
 67     ExceptionOr<void> setDisplay(const String&);
 68 
 69     String family() const;
 70     String style() const;
 71     String weight() const;
 72     String stretch() const;
 73     String unicodeRange() const;

 74     String featureSettings() const;
 75     String display() const;
 76 
 77     enum class LoadStatus { Unloaded, Loading, Loaded, Error };
 78     LoadStatus status() const;
 79 
 80     using LoadedPromise = DOMPromiseProxyWithResolveCallback<IDLInterface<FontFace>>;
 81     LoadedPromise& loadedForBindings();
 82     LoadedPromise& loadForBindings();
 83 
 84     void adopt(CSSFontFace&);
 85 
 86     CSSFontFace& backing() { return m_backing; }
 87 
 88     static RefPtr<CSSValue> parseString(const String&, CSSPropertyID);
 89 
 90     void fontStateChanged(CSSFontFace&, CSSFontFace::Status oldState, CSSFontFace::Status newState) final;
 91 
 92     void ref() final { RefCounted::ref(); }
 93     void deref() final { RefCounted::deref(); }
 94 
 95     bool hasPendingActivity() const final;
 96 
 97 private:
 98     explicit FontFace(CSSFontSelector&);
 99     explicit FontFace(CSSFontFace&);
100 
101     const char* activeDOMObjectName() const final;
102 
103     // Callback for LoadedPromise.
104     FontFace& loadedPromiseResolve();
105     void setErrorState();
106     Ref<CSSFontFace> m_backing;
107     UniqueRef<LoadedPromise> m_loadedPromise;
108     bool m_mayLoadedPromiseBeScriptObservable { false };
109 };
