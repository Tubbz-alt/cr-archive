146 class BytecodeIntrinsicRegistry {
147     WTF_MAKE_FAST_ALLOCATED;
148     WTF_MAKE_NONCOPYABLE(BytecodeIntrinsicRegistry);
149 public:
150     explicit BytecodeIntrinsicRegistry(VM&);
151 
152     typedef RegisterID* (BytecodeIntrinsicNode::* EmitterType)(BytecodeGenerator&, RegisterID*);
153 
154     enum class Type : uint8_t {
155         Emitter = 0,
156         LinkTimeConstant = 1,
157     };
158 
159     class Entry {
160     public:
161         Entry()
162             : m_type(Type::Emitter)
163         {
164             m_emitter = nullptr;
165         }
166 
167         Entry(EmitterType emitter)
168             : m_type(Type::Emitter)
169         {
170             m_emitter = emitter;
171         }
172 
173         Entry(LinkTimeConstant linkTimeConstant)
174             : m_type(Type::LinkTimeConstant)
175         {
176             m_linkTimeConstant = linkTimeConstant;
177         }
178 
179         Type type() const { return m_type; }
180         LinkTimeConstant linkTimeConstant() const { return m_linkTimeConstant; }
181         EmitterType emitter() const { return m_emitter; }
182 
183     private:
184         union {
185             EmitterType m_emitter;
186             LinkTimeConstant m_linkTimeConstant;
187         };
188         Type m_type;
189     };
190 
191     Optional<Entry> lookup(const Identifier&) const;
192 
193 #define JSC_DECLARE_BYTECODE_INTRINSIC_CONSTANT_GENERATORS(name) JSValue name##Value(BytecodeGenerator&);
194     JSC_COMMON_BYTECODE_INTRINSIC_CONSTANTS_EACH_NAME(JSC_DECLARE_BYTECODE_INTRINSIC_CONSTANT_GENERATORS)
195 #undef JSC_DECLARE_BYTECODE_INTRINSIC_CONSTANT_GENERATORS
196 
197 private:
198     VM& m_vm;
199     HashMap<RefPtr<UniquedStringImpl>, Entry, IdentifierRepHash> m_bytecodeIntrinsicMap;
200
