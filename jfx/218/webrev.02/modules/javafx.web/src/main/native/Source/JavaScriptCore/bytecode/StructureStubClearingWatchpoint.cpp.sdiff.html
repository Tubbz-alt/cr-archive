 37 void StructureStubClearingWatchpoint::fireInternal(VM& vm, const FireDetail&)
 38 {
 39     if (!m_holder->isValid())
 40         return;
 41 
 42     if (!m_key || !m_key.isWatchable(PropertyCondition::EnsureWatchability)) {
 43         // This will implicitly cause my own demise: stub reset removes all watchpoints.
 44         // That works, because deleting a watchpoint removes it from the set's list, and
 45         // the set's list traversal for firing is robust against the set changing.
 46         ConcurrentJSLocker locker(m_holder->codeBlock()->m_lock);
 47         m_holder->stubInfo()->reset(m_holder->codeBlock());
 48         return;
 49     }
 50 
 51     if (m_key.kind() == PropertyCondition::Presence) {
 52         // If this was a presence condition, let's watch the property for replacements. This is profitable
 53         // for the DFG, which will want the replacement set to be valid in order to do constant folding.
 54         m_key.object()->structure(vm)->startWatchingPropertyForReplacements(vm, m_key.offset());
 55     }
 56 
 57     m_key.object()->structure(vm)->addTransitionWatchpoint(this);
 58 }

 65 WatchpointsOnStructureStubInfo::Node& WatchpointsOnStructureStubInfo::addWatchpoint(const ObjectPropertyCondition& key)
 66 {
 67     if (!key || key.condition().kind() != PropertyCondition::Equivalence)
 68         return *m_watchpoints.add(WTF::in_place<StructureTransitionStructureStubClearingWatchpoint>, key, *this);
 69     ASSERT(key.condition().kind() == PropertyCondition::Equivalence);
 70     return *m_watchpoints.add(WTF::in_place<AdaptiveValueStructureStubClearingWatchpoint>, key, *this);
 71 }

 73 void WatchpointsOnStructureStubInfo::ensureReferenceAndInstallWatchpoint(
 74     std::unique_ptr<WatchpointsOnStructureStubInfo>& holderRef, CodeBlock* codeBlock,
 75     StructureStubInfo* stubInfo, const ObjectPropertyCondition& key)
 76 {
 77     if (!holderRef)
 78         holderRef = makeUnique<WatchpointsOnStructureStubInfo>(codeBlock, stubInfo);
 79     else {
 80         ASSERT(holderRef->m_codeBlock == codeBlock);
 81         ASSERT(holderRef->m_stubInfo == stubInfo);
 82     }
 83 
 84     ASSERT(!!key);
 85     auto& watchpointVariant = holderRef->addWatchpoint(key);
 86     if (key.kind() == PropertyCondition::Equivalence) {
 87         auto& adaptiveWatchpoint = WTF::get<AdaptiveValueStructureStubClearingWatchpoint>(watchpointVariant);
 88         adaptiveWatchpoint.install(codeBlock->vm());
 89     } else {
 90         auto* structureTransitionWatchpoint = &WTF::get<StructureTransitionStructureStubClearingWatchpoint>(watchpointVariant);
 91         key.object()->structure()->addTransitionWatchpoint(structureTransitionWatchpoint);
 92     }
 93 }
