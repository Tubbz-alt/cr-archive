35 inline constexpr bool isDynamicallySizedType(JSType type)
36 {
37     if (type == DirectArgumentsType
38         || type == FinalObjectType
39         || type == LexicalEnvironmentType
40         || type == ModuleEnvironmentType)
41         return true;
42     return false;
43 }
44 
45 inline size_t cellSize(VM& vm, JSCell* cell)
46 {
47     Structure* structure = cell->structure(vm);
48     const ClassInfo* classInfo = structure->classInfo();
49     JSType cellType = cell->type();
50 
51     if (isDynamicallySizedType(cellType)) {
52         switch (cellType) {
53         case DirectArgumentsType: {
54             auto* args = jsCast<DirectArguments*>(cell);
55             return DirectArguments::allocationSize(args->m_minCapacity);
56         }
57         case FinalObjectType:
58             return JSFinalObject::allocationSize(structure->inlineCapacity());
59         case LexicalEnvironmentType: {
60             auto* env = jsCast<JSLexicalEnvironment*>(cell);
61             return JSLexicalEnvironment::allocationSize(env->symbolTable());
62         }
63         case ModuleEnvironmentType: {
64             auto* env = jsCast<JSModuleEnvironment*>(cell);
65             return JSModuleEnvironment::allocationSize(env->symbolTable());
66         }
67         default:
68             RELEASE_ASSERT_NOT_REACHED();
69         }
70     }
71     return classInfo->staticClassSize;
72 }
