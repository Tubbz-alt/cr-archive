25 #pragma once
26 
27 #include "Event.h"
28 
29 namespace WebCore {
30 
31 class BeforeUnloadEvent final : public Event {
32     WTF_MAKE_ISO_ALLOCATED(BeforeUnloadEvent);
33 public:
34     static Ref<BeforeUnloadEvent> create()
35     {
36         return adoptRef(*new BeforeUnloadEvent);
37     }
38 
39     static Ref<BeforeUnloadEvent> createForBindings()
40     {
41         return adoptRef(*new BeforeUnloadEvent { ForBindings });
42     }
43 
44     String returnValue() const { return m_returnValue; }
45     void setReturnValue(const String& returnValue) { m_returnValue = returnValue; }
46 
47 private:
48     enum ForBindingsFlag { ForBindings };
49     BeforeUnloadEvent();
50     BeforeUnloadEvent(ForBindingsFlag);
51 
52     EventInterface eventInterface() const final { return BeforeUnloadEventInterfaceType; }
53     bool isBeforeUnloadEvent() const final;
54 
55     String m_returnValue;
56 };
57 
58 } // namespace WebCore
59 
60 SPECIALIZE_TYPE_TRAITS_EVENT(BeforeUnloadEvent)
