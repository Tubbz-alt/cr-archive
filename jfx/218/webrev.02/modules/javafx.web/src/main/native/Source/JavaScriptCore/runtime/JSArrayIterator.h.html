26 #pragma once
27 
28 #include "JSInternalFieldObjectImpl.h"
29 
30 namespace JSC {
31 
32 class JSArrayIterator final : public JSInternalFieldObjectImpl<4> {
33 public:
34     using Base = JSInternalFieldObjectImpl<4>;
35 
36     enum class Field : uint8_t {
37         Index = 0,
38         IteratedObject,
39         Kind,
40     };
41 
42     static size_t allocationSize(Checked<size_t> inlineCapacity)
43     {
44         ASSERT_UNUSED(inlineCapacity, inlineCapacity == 0U);
45         return sizeof(JSArrayIterator);
46     }
47 
48     template<typename CellType, SubspaceAccess mode>
49     static IsoSubspace* subspaceFor(VM& vm)
50     {
51         return vm.arrayIteratorSpace<mode>();
52     }
53 
54     static std::array<JSValue, numberOfInternalFields> initialValues()
55     {
56         return { {
57             jsNumber(0),
58             jsNull(),
59             jsNumber(0),
60         } };
61     }
62 
63     const WriteBarrier<Unknown>& internalField(Field field) const { return Base::internalField(static_cast<uint32_t>(field)); }
64     WriteBarrier<Unknown>& internalField(Field field) { return Base::internalField(static_cast<uint32_t>(field)); }
65 
66     IterationKind kind() const { return static_cast<IterationKind>(internalField(Field::Kind).get().asUInt32()); }
67     JSObject* iteratedObject() const { return jsCast<JSObject*>(internalField(Field::IteratedObject).get()); }
68 
69     static JSArrayIterator* create(VM&, Structure*, JSObject* iteratedObject, JSValue kind);
70     static JSArrayIterator* createWithInitialValues(VM&, Structure*);
71     static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
72 
73     DECLARE_INFO;
74 
75     static void visitChildren(JSCell*, SlotVisitor&);
76 
77 private:
78     JSArrayIterator(VM&, Structure*);
79     void finishCreation(VM&);
80 };
81 
82 } // namespace JSC
