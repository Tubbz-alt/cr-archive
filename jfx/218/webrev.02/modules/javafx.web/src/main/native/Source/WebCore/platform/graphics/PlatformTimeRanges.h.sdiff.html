 47     MediaTime end(unsigned index) const;

 57 
 58     void add(const MediaTime& start, const MediaTime& end);
 59     void clear();
 60 
 61     bool contain(const MediaTime&) const;
 62 
 63     size_t find(const MediaTime&) const;

 67     MediaTime totalDuration() const;
 68

 70 
 71     template<class Encoder> void encode(Encoder&) const;
 72     template<class Decoder> static Optional<PlatformTimeRanges> decode(Decoder&);
 73 
 74 private:
 75     // We consider all the Ranges to be semi-bounded as follow: [start, end[
 76     struct Range {
 77         Range() { }
 78         Range(const MediaTime& start, const MediaTime& end)
 79             : m_start(start)
 80             , m_end(end)
 81         {
 82         }
 83 
 84         MediaTime m_start;
 85         MediaTime m_end;
 86 
 87         template<class Encoder>
 88         void encode(Encoder& encoder) const
 89         {
 90             encoder << m_start;
 91             encoder << m_end;
 92         }
 93 
 94         template <class Decoder>
 95         static Optional<Range> decode(Decoder& decoder)
 96         {
 97             Optional<MediaTime> start;
 98             decoder >> start;
 99             if (!start)
100                 return WTF::nullopt;
101 
102             Optional<MediaTime> end;
103             decoder >> end;
104             if (!end)
105                 return WTF::nullopt;
106 
107             return {{ WTFMove(*start), WTFMove(*end) }};
108         }
109 
110         inline bool isPointInRange(const MediaTime& point) const
111         {
112             return m_start <= point && point < m_end;
113         }
114 
115         inline bool isOverlappingRange(const Range& range) const
116         {
117             return isPointInRange(range.m_start) || isPointInRange(range.m_end) || range.isPointInRange(m_start);
118         }
119 
120         inline bool isContiguousWithRange(const Range& range) const
121         {
122             return range.m_start == m_end || range.m_end == m_start;
123         }
124 
125         inline Range unionWithOverlappingOrContiguousRange(const Range& range) const
126         {
127             Range ret;
128 
129             ret.m_start = std::min(m_start, range.m_start);
130             ret.m_end = std::max(m_end, range.m_end);
131 
132             return ret;
133         }
134 
135         inline bool isBeforeRange(const Range& range) const
136         {
137             return range.m_start >= m_end;
138         }
139     };
140 
141     PlatformTimeRanges(Vector<Range>&&);
142 
143     Vector<Range> m_ranges;
144 };

152 template <class Decoder>
153 Optional<PlatformTimeRanges> PlatformTimeRanges::decode(Decoder& decoder)
154 {
155     Optional<Vector<Range>> buffered;
156     decoder >> buffered;
157     if (!buffered)
158         return WTF::nullopt;
159
