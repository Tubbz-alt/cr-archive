 28 class StringObject : public JSWrapperObject {

 33     template<typename, SubspaceAccess mode>

 35     {
 36         return &vm.stringObjectSpace;
 37     }
 38 
 39     static StringObject* create(VM& vm, Structure* structure)
 40     {
 41         JSString* string = jsEmptyString(vm);
 42         StringObject* object = new (NotNull, allocateCell<StringObject>(vm.heap)) StringObject(vm, structure);
 43         object->finishCreation(vm, string);
 44         return object;
 45     }
 46     static StringObject* create(VM& vm, Structure* structure, JSString* string)
 47     {
 48         StringObject* object = new (NotNull, allocateCell<StringObject>(vm.heap)) StringObject(vm, structure);
 49         object->finishCreation(vm, string);
 50         return object;
 51     }

 82 // Helper for producing a JSString for 'string', where 'string' was been produced by
 83 // calling ToString on 'originalValue'. In cases where 'originalValue' already was a
 84 // string primitive we can just use this, otherwise we need to allocate a new JSString.
 85 // FIXME: Basically any use of this is bad. toString() returns a JSString* so we don't need to
 86 // pass around the originalValue; we could just pass around the JSString*. Then you don't need
 87 // this function. You just use the JSString* that toString() returned.
 88 static inline JSString* jsStringWithReuse(JSGlobalObject* globalObject, JSValue originalValue, const String& string)
 89 {
 90     if (originalValue.isString()) {
 91         ASSERT(asString(originalValue)->value(globalObject) == string);
 92         return asString(originalValue);
 93     }
 94     return jsString(getVM(globalObject), string);
 95 }
 96 
 97 // Helper that tries to use the JSString substring sharing mechanism if 'originalValue' is a JSString.
 98 // FIXME: Basically any use of this is bad. toString() returns a JSString* so we don't need to
 99 // pass around the originalValue; we could just pass around the JSString*. And since we've
100 // resolved it, we know that we can just allocate the substring cell directly.
101 // https://bugs.webkit.org/show_bug.cgi?id=158140
102 static inline JSString* jsSubstring(JSGlobalObject* globalObject, JSValue originalValue, const String& string, unsigned offset, unsigned length)
103 {
104     if (originalValue.isString()) {
105         ASSERT(asString(originalValue)->value(globalObject) == string);
106         return jsSubstring(globalObject, asString(originalValue), offset, length);
107     }
108     return jsSubstring(getVM(globalObject), string, offset, length);
109 }
