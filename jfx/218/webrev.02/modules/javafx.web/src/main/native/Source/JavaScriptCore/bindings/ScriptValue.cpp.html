 42 static RefPtr<JSON::Value> jsToInspectorValue(JSGlobalObject* globalObject, JSValue value, int maxDepth)
 43 {
 44     if (!value) {
 45         ASSERT_NOT_REACHED();
 46         return nullptr;
 47     }
 48 
 49     if (!maxDepth)
 50         return nullptr;
 51 
 52     maxDepth--;
 53 
 54     if (value.isUndefinedOrNull())
 55         return JSON::Value::null();
 56     if (value.isBoolean())
 57         return JSON::Value::create(value.asBoolean());
 58     if (value.isNumber() && value.isDouble())
 59         return JSON::Value::create(value.asNumber());
 60     if (value.isNumber() && value.isAnyInt())
 61         return JSON::Value::create(static_cast<int>(value.asAnyInt()));
 62     if (value.isString())
 63         return JSON::Value::create(asString(value)->value(globalObject));
 64 
 65     if (value.isObject()) {
 66         if (isJSArray(value)) {
 67             auto inspectorArray = JSON::Array::create();
 68             auto& array = *asArray(value);
 69             unsigned length = array.length();
 70             for (unsigned i = 0; i < length; i++) {
 71                 auto elementValue = jsToInspectorValue(globalObject, array.getIndex(globalObject, i), maxDepth);
 72                 if (!elementValue)
 73                     return nullptr;
 74                 inspectorArray->pushValue(WTFMove(elementValue));
 75             }
 76             return inspectorArray;
 77         }
 78         VM& vm = globalObject->vm();
 79         auto inspectorObject = JSON::Object::create();
 80         auto& object = *value.getObject();
 81         PropertyNameArray propertyNames(vm, PropertyNameMode::Strings, PrivateSymbolMode::Exclude);
 82         object.methodTable(vm)->getOwnPropertyNames(&object, globalObject, propertyNames, EnumerationMode());
 83         for (auto& name : propertyNames) {
 84             auto inspectorValue = jsToInspectorValue(globalObject, object.get(globalObject, name), maxDepth);
 85             if (!inspectorValue)
 86                 return nullptr;
 87             inspectorObject->setValue(name.string(), WTFMove(inspectorValue));
 88         }
 89         return inspectorObject;
 90     }
 91 
 92     ASSERT_NOT_REACHED();
 93     return nullptr;
 94 }
 95 
 96 RefPtr<JSON::Value> toInspectorValue(JSGlobalObject* globalObject, JSValue value)
 97 {
 98     // FIXME: Maybe we should move the JSLockHolder stuff to the callers since this function takes a JSValue directly.
 99     // Doing the locking here made sense when we were trying to abstract the difference between multiple JavaScript engines.
100     JSLockHolder holder(globalObject);
101     return jsToInspectorValue(globalObject, value, JSON::Value::maxDepth);
102 }
