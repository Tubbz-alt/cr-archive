32 template<typename Func>
33 void IsoTLSEntry::walkUpToInclusive(IsoTLSEntry* last, const Func& func)
34 {
35     IsoTLSEntry* current = this;
36     for (;;) {
37         func(current);
38         if (current == last)
39             return;
40         current = current->m_next;
41     }
42 }
43 
44 template<typename EntryType>
45 DefaultIsoTLSEntry<EntryType>::DefaultIsoTLSEntry()
46     : IsoTLSEntry(alignof(EntryType), sizeof(EntryType))
47 {
48 }
49 
50 template<typename EntryType>
51 DefaultIsoTLSEntry<EntryType>::~DefaultIsoTLSEntry()
52 {


53 }
54 
55 template<typename EntryType>
56 void DefaultIsoTLSEntry<EntryType>::move(void* passedSrc, void* dst)
57 {
58     EntryType* src = static_cast<EntryType*>(passedSrc);
59     new (dst) EntryType(std::move(*src));
60     src->~EntryType();
61 }
62 
63 template<typename EntryType>
64 void DefaultIsoTLSEntry<EntryType>::destruct(void* passedEntry)
65 {
66     EntryType* entry = static_cast<EntryType*>(passedEntry);
67     entry->~EntryType();
68 }
69 
70 template<typename EntryType>
71 void DefaultIsoTLSEntry<EntryType>::scavenge(void* passedEntry)
72 {
73     EntryType* entry = static_cast<EntryType*>(passedEntry);
74     entry->scavenge();
75 }
