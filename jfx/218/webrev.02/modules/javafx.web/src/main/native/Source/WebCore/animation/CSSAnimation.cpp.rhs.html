 40 Ref<CSSAnimation> CSSAnimation::create(Element& owningElement, const Animation& backingAnimation, const RenderStyle* oldStyle, const RenderStyle& newStyle)
 41 {
 42     auto result = adoptRef(*new CSSAnimation(owningElement, backingAnimation));
 43     result->initialize(oldStyle, newStyle);
 44 
 45     InspectorInstrumentation::didCreateWebAnimation(result.get());
 46 
 47     return result;
 48 }

 56 void CSSAnimation::syncPropertiesWithBackingAnimation()
 57 {
 58     DeclarativeAnimation::syncPropertiesWithBackingAnimation();
 59 
 60     if (!effect())
 61         return;
 62 
 63     suspendEffectInvalidation();
 64 
 65     auto& animation = backingAnimation();
 66     auto* animationEffect = effect();
 67 
 68     auto previousTiming = animationEffect->getComputedTiming();
 69 
 70     switch (animation.fillMode()) {
 71     case AnimationFillMode::None:
 72         animationEffect->setFill(FillMode::None);
 73         break;
 74     case AnimationFillMode::Backwards:
 75         animationEffect->setFill(FillMode::Backwards);
 76         break;
 77     case AnimationFillMode::Forwards:
 78         animationEffect->setFill(FillMode::Forwards);
 79         break;
 80     case AnimationFillMode::Both:
 81         animationEffect->setFill(FillMode::Both);
 82         break;
 83     }
 84 
 85     switch (animation.direction()) {
 86     case Animation::AnimationDirectionNormal:
 87         animationEffect->setDirection(PlaybackDirection::Normal);
 88         break;
 89     case Animation::AnimationDirectionAlternate:
 90         animationEffect->setDirection(PlaybackDirection::Alternate);
 91         break;
 92     case Animation::AnimationDirectionReverse:
 93         animationEffect->setDirection(PlaybackDirection::Reverse);
 94         break;
 95     case Animation::AnimationDirectionAlternateReverse:
 96         animationEffect->setDirection(PlaybackDirection::AlternateReverse);
 97         break;
 98     }
 99 
100     auto iterationCount = animation.iterationCount();
101     animationEffect->setIterations(iterationCount == Animation::IterationCountInfinite ? std::numeric_limits<double>::infinity() : iterationCount);
102 
103     animationEffect->setDelay(Seconds(animation.delay()));
104     animationEffect->setIterationDuration(Seconds(animation.duration()));
105     animationEffect->updateStaticTimingProperties();
106     effectTimingDidChange(previousTiming);
107 
108     // Synchronize the play state
109     if (animation.playState() == AnimationPlayState::Playing && playState() == WebAnimation::PlayState::Paused) {
110         if (!m_stickyPaused)
111             play();
112     } else if (animation.playState() == AnimationPlayState::Paused && playState() == WebAnimation::PlayState::Running)
113         pause();
114 
115     unsuspendEffectInvalidation();
116 }
117 
118 ExceptionOr<void> CSSAnimation::bindingsPlay()
119 {
120     m_stickyPaused = false;
121     return DeclarativeAnimation::bindingsPlay();
122 }
123 
124 ExceptionOr<void> CSSAnimation::bindingsPause()
125 {
126     m_stickyPaused = true;
127     return DeclarativeAnimation::bindingsPause();
128 }

132     return AnimationEvent::create(eventType, m_animationName, elapsedTime, pseudoId, timelineTime, this);
