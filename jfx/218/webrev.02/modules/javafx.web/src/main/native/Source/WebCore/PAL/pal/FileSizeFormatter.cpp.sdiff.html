33 String fileSizeDescription(uint64_t size)
34 {
35     // FIXME: These strings should be localized, but that would require bringing LocalizedStrings into PAL.
36     // See <https://bugs.webkit.org/show_bug.cgi?id=179019> for more details.

39     if (size < 1000000)
40         return makeString(FormattedNumber::fixedWidth(size / 1000., 1), " KB");
41     if (size < 1000000000)
42         return makeString(FormattedNumber::fixedWidth(size / 1000000., 1), " MB");
43     return makeString(FormattedNumber::fixedWidth(size / 1000000000., 1), " GB");
44 }
