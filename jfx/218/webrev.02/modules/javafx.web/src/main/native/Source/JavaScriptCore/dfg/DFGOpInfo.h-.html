36 // This type used in passing an immediate argument to Node constructor;
37 // distinguishes an immediate value (typically an index into a CodeBlock data structure -
38 // a constant index, argument, or identifier) from a Node*.
39 struct OpInfo {
40     OpInfo() : m_value(0) { }
41     template<
42         typename IntegralType,
43         typename Constraint = typename std::enable_if<(std::is_integral<IntegralType>::value || std::is_enum<IntegralType>::value) && sizeof(IntegralType) <= sizeof(uint64_t)>::type>
44     explicit OpInfo(IntegralType value)
45         : m_value(static_cast<uint64_t>(value)) { }
46     explicit OpInfo(RegisteredStructure structure) : m_value(static_cast<uint64_t>(bitwise_cast<uintptr_t>(structure))) { }
47 
48 
49     template <typename T>
50     explicit OpInfo(T* ptr)
51     {
52         static_assert(!std::is_base_of<HeapCell, T>::value, "To make an OpInfo with a cell in it, the cell must be registered or frozen.");
53         static_assert(!std::is_base_of<StructureSet, T>::value, "To make an OpInfo with a structure set in, make sure to use RegisteredStructureSet.");
54         m_value = static_cast<uint64_t>(reinterpret_cast<uintptr_t>(ptr));
55     }
56 
57     uint64_t m_value;
58 };
