 48 static String decompressAndDecodeStandardLibrary()
 49 {
 50     auto decompressedStandardLibrary = gunzip(WHLSLStandardLibrary, sizeof(WHLSLStandardLibrary));
 51     return String::fromUTF8(decompressedStandardLibrary.data(), decompressedStandardLibrary.size());
 52 }
 53 
 54 class NameFinder : public Visitor {
 55 public:
 56     HashSet<String> takeFunctionNames()
 57     {
 58         HashSet<String> result;
 59         std::swap(result, m_functionNames);
 60         return result;
 61     }
 62 
 63 private:
 64     void visit(AST::CallExpression& callExpression) override
 65     {
 66         m_functionNames.add(callExpression.name());
 67         Visitor::visit(callExpression);
 68     }
 69 








 70     HashSet<String> m_functionNames;
 71 };
 72 
 73 Expected<void, Error> includeStandardLibrary(Program& program, Parser& parser, bool parseFullStandardLibrary)
 74 {
 75     static NeverDestroyed<String> standardLibrary(decompressAndDecodeStandardLibrary());
 76     if (parseFullStandardLibrary) {
 77         auto parseResult = parser.parse(program, standardLibrary.get(), ParsingMode::StandardLibrary, AST::NameSpace::StandardLibrary);
 78         if (!parseResult)
 79             return makeUnexpected(parseResult.error());
 80         return { };
 81     }
 82 
 83     static NeverDestroyed<HashMap<String, SubstringLocation>> standardLibraryFunctionMap(computeStandardLibraryFunctionMap());
 84 
 85     auto stringView = StringView(standardLibrary.get()).substring(0, firstFunctionOffsetInStandardLibrary());
 86     auto parseResult = parser.parse(program, stringView, ParsingMode::StandardLibrary, AST::NameSpace::StandardLibrary);
 87     if (!parseResult)
 88         return makeUnexpected(parseResult.error());
 89 
 90     NameFinder nameFinder;
 91     nameFinder.Visitor::visit(program);
 92     HashSet<String> functionNames = nameFinder.takeFunctionNames();
 93     HashSet<String> allFunctionNames;
 94     // The name of a call to a cast operator is the name of the type, so we can't match them up correctly.
 95     // Instead, just include all casting operators.
 96     functionNames.add("operator cast"_str);
 97     // We need to make sure that an author can't write a function with the same signature as anything in the standard library.
 98     // If they do so, we need to make sure it collides, so we include all potential duplicates here, and the "checkDuplicateFunctions" pass enforces it.
 99     for (auto& functionDefinition : program.functionDefinitions())
100         functionNames.add(functionDefinition->name());
101     while (!functionNames.isEmpty()) {
102         auto nativeFunctionDeclarationsCount = program.nativeFunctionDeclarations().size();
103         auto functionDefinitionsCount = program.functionDefinitions().size();
104         for (const auto& name : functionNames) {
105             if (allFunctionNames.contains(name))
106                 continue;
107             auto iterator = standardLibraryFunctionMap.get().find(name);
108             if (iterator == standardLibraryFunctionMap.get().end())
109                 continue;
110             auto stringView = StringView(standardLibrary.get()).substring(iterator->value.start, iterator->value.end - iterator->value.start);
111             if (verbose) {
112                 dataLogLn("---------------------------");
113                 dataLogLn(stringView);
114                 dataLogLn("---------------------------");
115             }
116             auto start = program.functionDefinitions().size();
117             auto parseResult = parser.parse(program, stringView, ParsingMode::StandardLibrary, AST::NameSpace::StandardLibrary);
118             if (!parseResult)
119                 return makeUnexpected(parseResult.error());
120             if (verbose) {
121                 if (program.functionDefinitions().size() != start)
122                     dataLogLn("non native stdlib function: '", name, "'");
123             }
124             allFunctionNames.add(name);
125         }
126         for ( ; nativeFunctionDeclarationsCount < program.nativeFunctionDeclarations().size(); ++nativeFunctionDeclarationsCount)
127             nameFinder.Visitor::visit(program.nativeFunctionDeclarations()[nativeFunctionDeclarationsCount]);
128         for ( ; functionDefinitionsCount < program.functionDefinitions().size(); ++functionDefinitionsCount)
129             nameFinder.Visitor::visit(program.functionDefinitions()[functionDefinitionsCount]);
130         functionNames = nameFinder.takeFunctionNames();
131     }
132     return { };
133 }
