50 static JSValue createNewElementWrapper(JSDOMGlobalObject* globalObject, Ref<Element>&& element)
51 {
52     if (is<HTMLElement>(element))
53         return createJSHTMLWrapper(globalObject, static_reference_cast<HTMLElement>(WTFMove(element)));
54     if (is<SVGElement>(element))
55         return createJSSVGWrapper(globalObject, static_reference_cast<SVGElement>(WTFMove(element)));




56     return createWrapper<Element>(globalObject, WTFMove(element));
57 }

59 JSValue toJS(ExecState*, JSDOMGlobalObject* globalObject, Element& element)
60 {
61     if (auto* wrapper = getCachedWrapper(globalObject->world(), element))
62         return wrapper;
63     return createNewElementWrapper(globalObject, element);
64 }
65 
66 JSValue toJSNewlyCreated(ExecState*, JSDOMGlobalObject* globalObject, Ref<Element>&& element)
67 {
68     if (element->isDefinedCustomElement())
69         return getCachedWrapper(globalObject->world(), element);
70     ASSERT(!getCachedWrapper(globalObject->world(), element));
71     return createNewElementWrapper(globalObject, WTFMove(element));
72 }

52 static JSValue createNewElementWrapper(JSDOMGlobalObject* globalObject, Ref<Element>&& element)
53 {
54     if (is<HTMLElement>(element))
55         return createJSHTMLWrapper(globalObject, static_reference_cast<HTMLElement>(WTFMove(element)));
56     if (is<SVGElement>(element))
57         return createJSSVGWrapper(globalObject, static_reference_cast<SVGElement>(WTFMove(element)));
58 #if ENABLE(MATHML)
59     if (is<MathMLElement>(element))
60         return createJSMathMLWrapper(globalObject, static_reference_cast<MathMLElement>(WTFMove(element)));
61 #endif
62     return createWrapper<Element>(globalObject, WTFMove(element));
63 }

66 {

68         return wrapper;

70 }

76     ASSERT(!getCachedWrapper(globalObject->world(), element));
77     return createNewElementWrapper(globalObject, WTFMove(element));
