22 #include "config.h"
23 #include "StyleSurroundData.h"
24 
25 namespace WebCore {
26 
27 StyleSurroundData::StyleSurroundData()
28     : margin(Fixed)
29     , padding(Fixed)
30 {
31 }
32 
33 inline StyleSurroundData::StyleSurroundData(const StyleSurroundData& o)
34     : RefCounted<StyleSurroundData>()
35     , offset(o.offset)
36     , margin(o.margin)
37     , padding(o.padding)
38     , border(o.border)
39 {
40 }
41 
42 Ref<StyleSurroundData> StyleSurroundData::copy() const
43 {
44     return adoptRef(*new StyleSurroundData(*this));
45 }
46 
47 bool StyleSurroundData::operator==(const StyleSurroundData& o) const
48 {
49     return offset == o.offset && margin == o.margin && padding == o.padding && border == o.border;
50 }

52 } // namespace WebCore
