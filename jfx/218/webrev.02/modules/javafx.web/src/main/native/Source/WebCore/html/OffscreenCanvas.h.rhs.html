 73 class OffscreenCanvas final : public RefCounted<OffscreenCanvas>, public CanvasBase, public EventTargetWithInlineData, private ContextDestructionObserver {
 74     WTF_MAKE_ISO_ALLOCATED(OffscreenCanvas);
 75 public:
 76 
 77     struct ImageEncodeOptions {
 78         String type = "image/png";
 79         double quality = 1.0;
 80     };
 81 
 82     enum class RenderingContextType {
 83         _2d,
 84         Webgl
 85     };
 86 
 87     static Ref<OffscreenCanvas> create(ScriptExecutionContext&, unsigned width, unsigned height);
 88     static Ref<OffscreenCanvas> create(ScriptExecutionContext&, std::unique_ptr<DetachedOffscreenCanvas>&&);
 89     virtual ~OffscreenCanvas();
 90 
 91     unsigned width() const final;

 92     unsigned height() const final;
 93     void setWidth(unsigned);
 94     void setHeight(unsigned);
 95 
 96     CanvasRenderingContext* renderingContext() const final { return m_context.get(); }

 97 
 98     ExceptionOr<OffscreenRenderingContext> getContext(JSC::JSGlobalObject&, RenderingContextType, Vector<JSC::Strong<JSC::Unknown>>&& arguments);
 99     ExceptionOr<RefPtr<ImageBitmap>> transferToImageBitmap();
100     void convertToBlob(ImageEncodeOptions&&, Ref<DeferredPromise>&&);


101 
102     void didDraw(const FloatRect&) final;

103 




104     Image* copiedImage() const final { return nullptr; }
105     bool hasCreatedImageBuffer() const final { return m_hasCreatedImageBuffer; }
106 
107     SecurityOrigin* securityOrigin() const final;
108 
109     bool canDetach() const;
110     std::unique_ptr<DetachedOffscreenCanvas> detach();
111 
112     CSSValuePool& cssValuePool();
113 
114     using RefCounted::ref;
115     using RefCounted::deref;
116 
117 private:
118 
119     OffscreenCanvas(ScriptExecutionContext&, unsigned width, unsigned height);
120 
121     bool isOffscreenCanvas() const final { return true; }
122 
123     ScriptExecutionContext* scriptExecutionContext() const final { return ContextDestructionObserver::scriptExecutionContext(); }
124     ScriptExecutionContext* canvasBaseScriptExecutionContext() const final { return ContextDestructionObserver::scriptExecutionContext(); }
125 
126     EventTargetInterface eventTargetInterface() const final { return OffscreenCanvasEventTargetInterfaceType; }
127     void refEventTarget() final { ref(); }
128     void derefEventTarget() final { deref(); }
129 
130     void refCanvasBase() final { ref(); }
131     void derefCanvasBase() final { deref(); }
132 
133     void setSize(const IntSize&) final;
134     void createImageBuffer() const final;
135     std::unique_ptr<ImageBuffer> takeImageBuffer() const;
136 
137     void reset();
138 
139     std::unique_ptr<CanvasRenderingContext> m_context;
140 
141     // m_hasCreatedImageBuffer means we tried to malloc the buffer. We didn't necessarily get it.
142     mutable bool m_hasCreatedImageBuffer { false };
143 
144     bool m_detached { false };
145 };
