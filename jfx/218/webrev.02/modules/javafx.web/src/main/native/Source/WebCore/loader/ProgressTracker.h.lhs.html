27 
28 #include "Timer.h"
29 #include <wtf/Forward.h>
30 #include <wtf/HashMap.h>
31 #include <wtf/Noncopyable.h>
32 #include <wtf/RefPtr.h>

33 
34 namespace WebCore {
35 
36 class Frame;
37 class ResourceResponse;
38 class ProgressTrackerClient;
39 struct ProgressItem;
40 
41 class ProgressTracker {
42     WTF_MAKE_NONCOPYABLE(ProgressTracker); WTF_MAKE_FAST_ALLOCATED;

43 public:
44     explicit ProgressTracker(ProgressTrackerClient&);
45     ~ProgressTracker();
46 


47     static unsigned long createUniqueIdentifier();
48 
49     WEBCORE_EXPORT double estimatedProgress() const;
50 
51     void progressStarted(Frame&);
52     void progressCompleted(Frame&);
53 
54     void incrementProgress(unsigned long identifier, const ResourceResponse&);
55     void incrementProgress(unsigned long identifier, unsigned bytesReceived);
56     void completeProgress(unsigned long identifier);
57 
58     long long totalPageAndResourceBytesToLoad() const { return m_totalPageAndResourceBytesToLoad; }
59     long long totalBytesReceived() const { return m_totalBytesReceived; }
60 
61     bool isMainLoadProgressing() const;
62 
63 private:
64     void reset();
65     void finalProgressComplete();
66 
67     void progressHeartbeatTimerFired();
68     bool isAlwaysOnLoggingAllowed() const;
69 
70     static unsigned long s_uniqueIdentifier;
71 
72     ProgressTrackerClient& m_client;
73     RefPtr<Frame> m_originatingProgressFrame;
74     HashMap<unsigned long, std::unique_ptr<ProgressItem>> m_progressItems;
75     Timer m_progressHeartbeatTimer;
76 
77     long long m_totalPageAndResourceBytesToLoad { 0 };
78     long long m_totalBytesReceived { 0 };
79     long long m_totalBytesReceivedBeforePreviousHeartbeat { 0 };
80 
81     double m_lastNotifiedProgressValue { 0 };
82     double m_progressValue { 0 };
83 
84     MonotonicTime m_mainLoadCompletionTime;
85     MonotonicTime m_lastNotifiedProgressTime;
86 
87     int m_numProgressTrackedFrames { 0 };
88     unsigned m_heartbeatsWithNoProgress { 0 };
89 
90     bool m_finalProgressChangedSent { false };
91     bool m_isMainLoad { false };
92 };
93 
94 } // namespace WebCore
