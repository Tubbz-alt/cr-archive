26 #pragma once
27 
28 #if ENABLE(VIDEO_TRACK)
29 
30 #include "TrackPrivateBase.h"
31 
32 namespace WebCore {
33 
34 class VideoTrackPrivateClient : public TrackPrivateBaseClient {
35 public:
36     virtual void selectedChanged(bool) = 0;
37 };
38 
39 class VideoTrackPrivate : public TrackPrivateBase {
40 public:
41     void setClient(VideoTrackPrivateClient* client) { m_client = client; }
42     VideoTrackPrivateClient* client() const override { return m_client; }
43 
44     virtual void setSelected(bool selected)
45     {
46         if (m_selected == selected)
47             return;
48         m_selected = selected;
49         if (m_client)
50             m_client->selectedChanged(m_selected);
51     }
52     virtual bool selected() const { return m_selected; }
53 
54     enum Kind { Alternative, Captions, Main, Sign, Subtitles, Commentary, None };
55     virtual Kind kind() const { return None; }
56 
57 #if !RELEASE_LOG_DISABLED
58     const char* logClassName() const final { return "VideoTrackPrivate"; }
59 #endif
60 
61 protected:
62     VideoTrackPrivate() = default;
63 
64 private:
65     VideoTrackPrivateClient* m_client { nullptr };
66     bool m_selected { false };
67 };
68 
69 } // namespace WebCore
