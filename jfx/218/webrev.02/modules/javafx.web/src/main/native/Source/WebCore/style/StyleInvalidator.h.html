47 class Invalidator {
48 public:
49     Invalidator(const Vector<StyleSheetContents*>&, const MediaQueryEvaluator&);
50     Invalidator(const InvalidationRuleSetVector&);
51 
52     ~Invalidator();
53 
54     bool dirtiesAllStyle() const { return m_dirtiesAllStyle; }
55     void invalidateStyle(Document&);
56     void invalidateStyle(Scope&);
57     void invalidateStyle(ShadowRoot&);

60     static void invalidateShadowParts(ShadowRoot&);

67

71     void invalidateStyleForTree(Element&, SelectorFilter*);
72     void invalidateStyleForDescendants(Element&, SelectorFilter*);
73     void invalidateInShadowTreeIfNeeded(Element&);
74     void invalidateStyleWithMatchElement(Element&, MatchElement);
75 
76     struct RuleInformation {
77         bool hasSlottedPseudoElementRules { false };
78         bool hasHostPseudoClassRules { false };
79         bool hasShadowPseudoElementRules { false };
80         bool hasPartPseudoElementRules { false };
81     };
82     RuleInformation collectRuleInformation();
83 
84     RefPtr<RuleSet> m_ownedRuleSet;
85     const InvalidationRuleSetVector m_ruleSets;
86 
87     RuleInformation m_ruleInformation;
88 
89     bool m_dirtiesAllStyle { false };
90     bool m_didInvalidateHostChildren { false };
91 };
