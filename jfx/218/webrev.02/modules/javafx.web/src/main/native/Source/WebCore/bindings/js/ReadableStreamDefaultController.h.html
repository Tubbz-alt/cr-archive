 44 class ReadableStreamDefaultController {
 45 public:
 46     explicit ReadableStreamDefaultController(JSReadableStreamDefaultController* controller) : m_jsController(controller) { }
 47 
 48     static JSC::JSValue invoke(JSC::JSGlobalObject&, JSC::JSObject&, const char*, JSC::JSValue);
 49 
 50     bool enqueue(RefPtr<JSC::ArrayBuffer>&&);
 51 
 52     void error(const Exception&);
 53 
 54     void close() { invoke(globalObject(), jsController(), "close", JSC::jsUndefined()); }
 55 
 56 private:
 57     void error(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value) { invoke(lexicalGlobalObject, jsController(), "error", value); }
 58     void enqueue(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value) { invoke(lexicalGlobalObject, jsController(), "enqueue", value); }
 59     JSReadableStreamDefaultController& jsController() const;
 60 
 61     JSDOMGlobalObject& globalObject() const;
 62 
 63     // The owner of ReadableStreamDefaultController is responsible to keep uncollected the JSReadableStreamDefaultController.
 64     JSReadableStreamDefaultController* m_jsController { nullptr };
 65 };
 66 
 67 inline JSReadableStreamDefaultController& ReadableStreamDefaultController::jsController() const
 68 {
 69     ASSERT(m_jsController);
 70     return *m_jsController;
 71 }
 72 
 73 inline JSDOMGlobalObject& ReadableStreamDefaultController::globalObject() const
 74 {
 75     ASSERT(m_jsController);
 76     ASSERT(m_jsController->globalObject());
 77     return *static_cast<JSDOMGlobalObject*>(m_jsController->globalObject());
 78 }
 79 
 80 inline bool ReadableStreamDefaultController::enqueue(RefPtr<JSC::ArrayBuffer>&& buffer)
 81 {
 82     auto& globalObject = this->globalObject();
 83     JSC::VM& vm = globalObject.vm();
 84     JSC::JSLockHolder locker(vm);
 85     auto scope = DECLARE_THROW_SCOPE(vm);
 86     JSC::JSGlobalObject& lexicalGlobalObject = globalObject;
 87 
 88     if (!buffer) {
 89         error(lexicalGlobalObject, createOutOfMemoryError(&lexicalGlobalObject));
 90         return false;
 91     }
 92     auto length = buffer->byteLength();
 93     auto chunk = JSC::Uint8Array::create(WTFMove(buffer), 0, length);
 94     enqueue(lexicalGlobalObject, toJS(&lexicalGlobalObject, &globalObject, chunk.ptr()));
 95     scope.assertNoException();
 96     return true;
 97 }
 98 
 99 inline void ReadableStreamDefaultController::error(const Exception& exception)
100 {
101     JSC::JSGlobalObject& lexicalGlobalObject = globalObject();
102     JSC::JSLockHolder locker(&lexicalGlobalObject);
103     error(lexicalGlobalObject, createDOMException(&lexicalGlobalObject, exception.code(), exception.message()));
104 }
