 26 #pragma once
 27 
 28 #include "JSGlobalObject.h"
 29 #include "JSObject.h"
 30 #include "JSScope.h"
 31 
 32 namespace JSC {
 33 
 34 class JSGlobalObject;
 35 class LLIntOffsetsExtractor;
 36 
 37 
 38 class JSCallee : public JSNonFinalObject {
 39     friend class JIT;
 40 #if ENABLE(DFG_JIT)
 41     friend class DFG::SpeculativeJIT;
 42     friend class DFG::JITCompiler;
 43 #endif
 44     friend class VM;
 45 
 46 public:
 47     using Base = JSNonFinalObject;
 48     static constexpr unsigned StructureFlags = Base::StructureFlags | ImplementsHasInstance | ImplementsDefaultHasInstance;
 49 
 50     template<typename CellType, SubspaceAccess>
 51     static IsoSubspace* subspaceFor(VM& vm)
 52     {
 53         return &vm.calleeSpace;
 54     }
 55 
 56     static JSCallee* create(VM& vm, JSGlobalObject* globalObject, JSScope* scope)
 57     {
 58         JSCallee* callee = new (NotNull, allocateCell<JSCallee>(vm.heap)) JSCallee(vm, scope, globalObject->calleeStructure());
 59         callee->finishCreation(vm);
 60         return callee;
 61     }
 62 
 63     JSScope* scope()
 64     {
 65         return m_scope.get();
 66     }
 67 
 68     // This method may be called for host functions, in which case it
 69     // will return an arbitrary value. This should only be used for
 70     // optimized paths in which the return value does not matter for
 71     // host functions, and checking whether the function is a host
 72     // function is deemed too expensive.
 73     JSScope* scopeUnchecked()
 74     {
 75         return m_scope.get();
 76     }
 77 
 78     void setScope(VM& vm, JSScope* scope)
 79     {
 80         m_scope.set(vm, this, scope);
 81     }
 82 
 83     DECLARE_EXPORT_INFO;
 84 
 85     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
 86     {
 87         ASSERT(globalObject);
 88         return Structure::create(vm, globalObject, prototype, TypeInfo(JSCalleeType, StructureFlags), info());
 89     }
 90 
 91     static inline ptrdiff_t offsetOfScopeChain()
 92     {
 93         return OBJECT_OFFSETOF(JSCallee, m_scope);
 94     }
 95 
 96 protected:
 97     JS_EXPORT_PRIVATE JSCallee(VM&, JSGlobalObject*, Structure*);
 98     JSCallee(VM&, JSScope*, Structure*);
 99 
100     void finishCreation(VM&);
101     using Base::finishCreation;
102 
103     static void visitChildren(JSCell*, SlotVisitor&);
104 
105 private:
106     friend class LLIntOffsetsExtractor;
107 
108     WriteBarrier<JSScope> m_scope;
109 };
110 
111 } // namespace JSC
