110 Optional<BytecodeIntrinsicRegistry::Entry> BytecodeIntrinsicRegistry::lookup(const Identifier& ident) const
111 {
112     if (!ident.isPrivateName())
113         return WTF::nullopt;
114     auto iterator = m_bytecodeIntrinsicMap.find(ident.impl());
115     if (iterator == m_bytecodeIntrinsicMap.end())
116         return WTF::nullopt;
117     return iterator->value;
118 }
