27 
28 #if ENABLE(JIT)
29 
30 #include "BytecodeIndex.h"
31 #include "MacroAssembler.h"
32 #include <wtf/Vector.h>
33 #include <wtf/text/CString.h>
34 
35 namespace JSC {
36 
37 class CodeBlock;
38 class LinkBuffer;
39 
40 namespace Profiler {
41 class Compilation;
42 }
43 
44 class JITDisassembler {
45     WTF_MAKE_FAST_ALLOCATED;
46 public:
47     JITDisassembler(CodeBlock*);
48     ~JITDisassembler();
49 
50     void setStartOfCode(MacroAssembler::Label label) { m_startOfCode = label; }
51     void setForBytecodeMainPath(unsigned bytecodeIndex, MacroAssembler::Label label)
52     {
53         m_labelForBytecodeIndexInMainPath[bytecodeIndex] = label;
54     }
55     void setForBytecodeSlowPath(unsigned bytecodeIndex, MacroAssembler::Label label)
56     {
57         m_labelForBytecodeIndexInSlowPath[bytecodeIndex] = label;
58     }
59     void setEndOfSlowPath(MacroAssembler::Label label) { m_endOfSlowPath = label; }
60     void setEndOfCode(MacroAssembler::Label label) { m_endOfCode = label; }
61 
62     void dump(LinkBuffer&);
63     void dump(PrintStream&, LinkBuffer&);
64     void reportToProfiler(Profiler::Compilation*, LinkBuffer&);
65 
66 private:
67     void dumpHeader(PrintStream&, LinkBuffer&);
68     MacroAssembler::Label firstSlowLabel();
69 
70     struct DumpedOp {
71         BytecodeIndex bytecodeIndex;
72         CString disassembly;
73     };
74     Vector<DumpedOp> dumpVectorForInstructions(LinkBuffer&, const char* prefix, Vector<MacroAssembler::Label>& labels, MacroAssembler::Label endLabel);
75 
76     void dumpForInstructions(PrintStream&, LinkBuffer&, const char* prefix, Vector<MacroAssembler::Label>& labels, MacroAssembler::Label endLabel);
77     void reportInstructions(Profiler::Compilation*, LinkBuffer&, const char* prefix, Vector<MacroAssembler::Label>& labels, MacroAssembler::Label endLabel);
78 
79     void dumpDisassembly(PrintStream&, LinkBuffer&, MacroAssembler::Label from, MacroAssembler::Label to);
80 
81     CodeBlock* m_codeBlock;
82     MacroAssembler::Label m_startOfCode;
83     Vector<MacroAssembler::Label> m_labelForBytecodeIndexInMainPath;
84     Vector<MacroAssembler::Label> m_labelForBytecodeIndexInSlowPath;
85     MacroAssembler::Label m_endOfSlowPath;
86     MacroAssembler::Label m_endOfCode;
87 };
88 
89 } // namespace JSC
90 
91 #endif // ENABLE(JIT)
