29 namespace WebCore {

31 EnterKeyHint enterKeyHintForAttributeValue(const String& value)
32 {
33     if (equalIgnoringASCIICase(value, "enter"))
34         return EnterKeyHint::Enter;
35     if (equalIgnoringASCIICase(value, "done"))
36         return EnterKeyHint::Done;
37     if (equalIgnoringASCIICase(value, "go"))
38         return EnterKeyHint::Go;
39     if (equalIgnoringASCIICase(value, "next"))
40         return EnterKeyHint::Next;
41     if (equalIgnoringASCIICase(value, "previous"))
42         return EnterKeyHint::Previous;
43     if (equalIgnoringASCIICase(value, "search"))
44         return EnterKeyHint::Search;
45     if (equalIgnoringASCIICase(value, "send"))
46         return EnterKeyHint::Send;
47     return EnterKeyHint::Unspecified;
48 }
49 
50 String attributeValueForEnterKeyHint(EnterKeyHint hint)
51 {
52     switch (hint) {
53     case EnterKeyHint::Unspecified:
54         return emptyAtom();
55     case EnterKeyHint::Enter:
56         return "enter";
57     case EnterKeyHint::Done:
58         return "done";
59     case EnterKeyHint::Go:
60         return "go";
61     case EnterKeyHint::Next:
62         return "next";
63     case EnterKeyHint::Previous:
64         return "previous";
65     case EnterKeyHint::Search:
66         return "search";
67     case EnterKeyHint::Send:
68         return "send";
69     }
70     ASSERT_NOT_REACHED();
71     return nullAtom();
72 }
