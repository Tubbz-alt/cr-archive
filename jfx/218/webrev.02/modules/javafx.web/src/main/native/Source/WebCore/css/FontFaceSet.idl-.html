26 enum FontFaceSetLoadStatus {
27     "loading",
28     "loaded"
29 };
30 
31 [
32     ConstructorCallWith=Document,
33     Constructor(sequence<FontFace> initialFaces),
34 ] interface FontFaceSet : EventTarget {
35     boolean has(FontFace font);
36 
37     // FIXME: We should add support for the setlike declaration.
38     iterable<FontFace>;
39 
40     readonly attribute long size;
41 
42     FontFaceSet add(FontFace font);
43     [ImplementedAs=remove] boolean delete(FontFace font);
44     void clear();
45 
46     attribute EventHandler onloading;
47     attribute EventHandler onloadingdone;
48     attribute EventHandler onloadingerror;
49 
50     Promise<sequence<FontFace>> load(DOMString font, optional DOMString text = " ");
51     [MayThrowException] boolean check(DOMString font, optional DOMString text = " ");
52 
53     readonly attribute Promise<FontFaceSet> ready;
54     readonly attribute FontFaceSetLoadStatus status;
55 };
