 33 namespace WebCore {
 34 
 35 class TokenPreloadScanner {
 36     WTF_MAKE_NONCOPYABLE(TokenPreloadScanner);
 37 public:
 38     explicit TokenPreloadScanner(const URL& documentURL, float deviceScaleFactor = 1.0);
 39 
 40     void scan(const HTMLToken&, PreloadRequestStream&, Document&);
 41 
 42     void setPredictedBaseElementURL(const URL& url) { m_predictedBaseElementURL = url; }
 43 
 44     bool inPicture() { return !m_pictureSourceState.isEmpty(); }
 45 
 46 private:
 47     enum class TagId {
 48         // These tags are scanned by the StartTagScanner.
 49         Img,
 50         Input,
 51         Link,
 52         Script,
 53         Meta,
 54         Source,
 55 
 56         // These tags are not scanned by the StartTagScanner.
 57         Unknown,
 58         Style,
 59         Base,
 60         Template,
 61         Picture
 62     };
 63 
 64     class StartTagScanner;
 65 
 66     static TagId tagIdFor(const HTMLToken::DataVector&);
 67 
 68     static String initiatorFor(TagId);
 69 
 70     void updatePredictedBaseURL(const HTMLToken&);
 71 
 72     CSSPreloadScanner m_cssScanner;
 73     const URL m_documentURL;
 74     const float m_deviceScaleFactor { 1 };
 75 
 76     URL m_predictedBaseElementURL;
 77     bool m_inStyle { false };
 78 
 79     Vector<bool> m_pictureSourceState;
 80 
 81     unsigned m_templateCount { 0 };
 82 };
 83 
 84 class HTMLPreloadScanner {
 85     WTF_MAKE_FAST_ALLOCATED;
 86 public:
 87     HTMLPreloadScanner(const HTMLParserOptions&, const URL& documentURL, float deviceScaleFactor = 1.0);
 88 
 89     void appendToEnd(const SegmentedString&);
 90     void scan(HTMLResourcePreloader&, Document&);
 91 
 92 private:
 93     TokenPreloadScanner m_scanner;
 94     SegmentedString m_source;
 95     HTMLTokenizer m_tokenizer;
 96 };
