51 
52     static JSWebAssemblyMemory* tryCreate(JSGlobalObject*, VM&, Structure*);

55     DECLARE_EXPORT_INFO;

60     Wasm::PageCount grow(VM&, JSGlobalObject*, uint32_t delta);

63 private:
64     JSWebAssemblyMemory(VM&, Structure*);
65     void finishCreation(VM&);
66     static void visitChildren(JSCell*, SlotVisitor&);
67 
68     Ref<Wasm::Memory> m_memory;
69     WriteBarrier<JSArrayBuffer> m_bufferWrapper;
70     RefPtr<ArrayBuffer> m_buffer;
71 };
