45     struct Init : EventInit {
46         RefPtr<Uint8Array> initData;
47     };

49     static Ref<WebKitMediaKeyNeededEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)

51         return adoptRef(*new WebKitMediaKeyNeededEvent(type, initializer, isTrusted));

56     Uint8Array* initData() const { return m_initData.get(); }

59     WebKitMediaKeyNeededEvent(const AtomString& type, Uint8Array* initData);

61 
62     RefPtr<Uint8Array> m_initData;
63 };
