27 #pragma once
28 
29 #include "JSGlobalObject.h"
30 #include "JSObject.h"
31 #include "ScriptFetcher.h"
32 #include <wtf/RefPtr.h>
33 
34 namespace JSC {
35 
36 class JSScriptFetcher final : public JSCell {
37 public:
38     using Base = JSCell;
39 
40     static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
41     static constexpr bool needsDestruction = true;
42 
43     DECLARE_EXPORT_INFO;
44 
45     template<typename CellType, SubspaceAccess mode>
46     static IsoSubspace* subspaceFor(VM& vm)
47     {
48         return vm.scriptFetcherSpace<mode>();
49     }
50 
51     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
52     {
53         return Structure::create(vm, globalObject, prototype, TypeInfo(JSScriptFetcherType, StructureFlags), info());
54     }
55 
56     static JSScriptFetcher* create(VM& vm, Structure* structure, RefPtr<ScriptFetcher>&& scriptFetcher)
57     {
58         auto* result = new (NotNull, allocateCell<JSScriptFetcher>(vm.heap)) JSScriptFetcher(vm, structure, WTFMove(scriptFetcher));
59         result->finishCreation(vm);
60         return result;
61     }
62 
63     static JSScriptFetcher* create(VM& vm, RefPtr<ScriptFetcher>&& scriptFetcher)
64     {
65         return create(vm, vm.scriptFetcherStructure.get(), WTFMove(scriptFetcher));
66     }
67 
68     ScriptFetcher* fetcher() const
69     {
70         return m_fetcher.get();
71     }
72 
73     static void destroy(JSCell*);
74 
75 private:
76     JSScriptFetcher(VM& vm, Structure* structure, RefPtr<ScriptFetcher>&& scriptFetcher)
77         : Base(vm, structure)
78         , m_fetcher(WTFMove(scriptFetcher))
79     {
80     }
81 
82     RefPtr<ScriptFetcher> m_fetcher;
83 };
84 
85 } // namespace JSC
