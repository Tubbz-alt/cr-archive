26 #pragma once
27 
28 #include "CryptoAlgorithm.h"
29 
30 #if ENABLE(WEB_CRYPTO)
31 
32 namespace WebCore {
33 
34 class CryptoAlgorithmAesCbcCfbParams;
35 class CryptoKeyAES;
36 
37 class CryptoAlgorithmAES_CBC final : public CryptoAlgorithm {
38 public:
39     enum class Padding : uint8_t {
40         Yes,
41         No
42     };
43 
44     static constexpr const char* s_name = "AES-CBC";
45     static constexpr CryptoAlgorithmIdentifier s_identifier = CryptoAlgorithmIdentifier::AES_CBC;
46     static Ref<CryptoAlgorithm> create();
47 
48     // Operations can be performed directly.
49     WEBCORE_EXPORT static ExceptionOr<Vector<uint8_t>> platformEncrypt(const CryptoAlgorithmAesCbcCfbParams&, const CryptoKeyAES&, const Vector<uint8_t>&, Padding padding = Padding::Yes);
50     WEBCORE_EXPORT static ExceptionOr<Vector<uint8_t>> platformDecrypt(const CryptoAlgorithmAesCbcCfbParams&, const CryptoKeyAES&, const Vector<uint8_t>&, Padding padding = Padding::Yes);
51 
52 private:
53     CryptoAlgorithmAES_CBC() = default;
54     CryptoAlgorithmIdentifier identifier() const final;
55 
56     void encrypt(const CryptoAlgorithmParameters&, Ref<CryptoKey>&&, Vector<uint8_t>&&, VectorCallback&&, ExceptionCallback&&, ScriptExecutionContext&, WorkQueue&) final;
57     void decrypt(const CryptoAlgorithmParameters&, Ref<CryptoKey>&&, Vector<uint8_t>&&, VectorCallback&&, ExceptionCallback&&, ScriptExecutionContext&, WorkQueue&) final;
58     void generateKey(const CryptoAlgorithmParameters&, bool extractable, CryptoKeyUsageBitmap, KeyOrKeyPairCallback&&, ExceptionCallback&&, ScriptExecutionContext&) final;
59     void importKey(CryptoKeyFormat, KeyData&&, const CryptoAlgorithmParameters&, bool extractable, CryptoKeyUsageBitmap, KeyCallback&&, ExceptionCallback&&) final;
60     void exportKey(CryptoKeyFormat, Ref<CryptoKey>&&, KeyDataCallback&&, ExceptionCallback&&) final;
61     ExceptionOr<size_t> getKeyLength(const CryptoAlgorithmParameters&) final;
62 };
63 
64 } // namespace WebCore
65 
66 #endif // ENABLE(WEB_CRYPTO)
