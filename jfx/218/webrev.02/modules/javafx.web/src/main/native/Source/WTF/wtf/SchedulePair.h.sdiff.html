73 struct SchedulePairHash {
74     static unsigned hash(const RefPtr<SchedulePair>& pair)
75     {
76         uintptr_t hashCodes[2] = { reinterpret_cast<uintptr_t>(pair->runLoop()), pair->mode() ? CFHash(pair->mode()) : 0 };
77         return StringHasher::hashMemory<sizeof(hashCodes)>(hashCodes);
78     }
79 
80     static bool equal(const RefPtr<SchedulePair>& a, const RefPtr<SchedulePair>& b) { return a == b; }
81 
82     static const bool safeToCompareToEmptyOrDeleted = true;
83 };
