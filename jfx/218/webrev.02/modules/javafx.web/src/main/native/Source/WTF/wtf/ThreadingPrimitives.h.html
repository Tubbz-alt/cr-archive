 53 using ThreadFunction = void (*)(void* argument);
 54 
 55 #if USE(PTHREADS)
 56 using PlatformThreadHandle = pthread_t;
 57 using PlatformMutex = pthread_mutex_t;
 58 using PlatformCondition = pthread_cond_t;
 59 using ThreadSpecificKey = pthread_key_t;
 60 #elif OS(WINDOWS)
 61 using ThreadIdentifier = uint32_t;
 62 using PlatformThreadHandle = HANDLE;
 63 using PlatformMutex = SRWLOCK;
 64 using PlatformCondition = CONDITION_VARIABLE;
 65 using ThreadSpecificKey = DWORD;
 66 #else
 67 #error "Not supported platform"
 68 #endif
 69 
 70 class Mutex final {
 71     WTF_MAKE_NONCOPYABLE(Mutex);
 72     WTF_MAKE_FAST_ALLOCATED;
 73 public:
 74     constexpr Mutex() = default;
 75     WTF_EXPORT_PRIVATE ~Mutex();
 76 
 77     WTF_EXPORT_PRIVATE void lock();
 78     WTF_EXPORT_PRIVATE bool tryLock();
 79     WTF_EXPORT_PRIVATE void unlock();
 80 
 81     PlatformMutex& impl() { return m_mutex; }
 82 
 83 private:
 84 #if USE(PTHREADS)
 85     PlatformMutex m_mutex = PTHREAD_MUTEX_INITIALIZER;
 86 #elif OS(WINDOWS)
 87     PlatformMutex m_mutex = SRWLOCK_INIT;
 88 #endif
 89 };
 90 
 91 typedef Locker<Mutex> MutexLocker;
 92 
 93 class ThreadCondition final {
 94     WTF_MAKE_NONCOPYABLE(ThreadCondition);
 95     WTF_MAKE_FAST_ALLOCATED;
 96 public:
 97     constexpr ThreadCondition() = default;
 98     WTF_EXPORT_PRIVATE ~ThreadCondition();
 99 
100     WTF_EXPORT_PRIVATE void wait(Mutex& mutex);
101     // Returns true if the condition was signaled before absoluteTime, false if the absoluteTime was reached or is in the past.
102     WTF_EXPORT_PRIVATE bool timedWait(Mutex&, WallTime absoluteTime);
103     WTF_EXPORT_PRIVATE void signal();
104     WTF_EXPORT_PRIVATE void broadcast();
105 
106 private:
107 #if USE(PTHREADS)
108     PlatformCondition m_condition = PTHREAD_COND_INITIALIZER;
109 #elif OS(WINDOWS)
110     PlatformCondition m_condition = CONDITION_VARIABLE_INIT;
111 #endif
112 };

114 #if USE(PTHREADS)
115 
116 static constexpr ThreadSpecificKey InvalidThreadSpecificKey = PTHREAD_KEYS_MAX;
117 
118 inline void threadSpecificKeyCreate(ThreadSpecificKey* key, void (*destructor)(void *))
119 {
120     int error = pthread_key_create(key, destructor);
121     if (error)
122         CRASH();
123 }
124 
125 inline void threadSpecificKeyDelete(ThreadSpecificKey key)
126 {
127     int error = pthread_key_delete(key);
128     if (error)
129         CRASH();
130 }
131 
132 inline void threadSpecificSet(ThreadSpecificKey key, void* value)
133 {
134     pthread_setspecific(key, value);
135 }
136 
137 inline void* threadSpecificGet(ThreadSpecificKey key)
138 {
139     return pthread_getspecific(key);
140 }
