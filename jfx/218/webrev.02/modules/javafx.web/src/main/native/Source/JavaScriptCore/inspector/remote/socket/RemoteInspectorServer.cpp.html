34 namespace Inspector {
35 
36 RemoteInspectorServer& RemoteInspectorServer::singleton()
37 {
38     static NeverDestroyed<RemoteInspectorServer> server;
39     return server;
40 }
41 
42 RemoteInspectorServer::~RemoteInspectorServer()
43 {
44     auto& endpoint = Inspector::RemoteInspectorSocketEndpoint::singleton();
45     endpoint.invalidateListener(*this);
46 }
47 
48 bool RemoteInspectorServer::start(const char* address, uint16_t port)
49 {
50     auto& endpoint = Inspector::RemoteInspectorSocketEndpoint::singleton();
51     m_server = endpoint.listenInet(address, port, *this, RemoteInspector::singleton());
52     return isRunning();
53 }
54 
55 bool RemoteInspectorServer::didAccept(ConnectionID acceptedID, ConnectionID, Socket::Domain)
56 {
57     ASSERT(!isMainThread());
58 
59     auto& inspector = RemoteInspector::singleton();
60     if (inspector.isConnected()) {
61         LOG_ERROR("RemoteInspector can accept only 1 client");
62 
63         return false;
64     }
65     inspector.connect(acceptedID);
66     return true;
67 }

69 } // namespace Inspector
