 38 UIEvent::UIEvent(const AtomString& eventType, CanBubble canBubble, IsCancelable isCancelable, IsComposed isComposed, RefPtr<WindowProxy>&& viewArg, int detailArg)

 45 UIEvent::UIEvent(const AtomString& eventType, CanBubble canBubble, IsCancelable isCancelable, IsComposed isComposed, MonotonicTime timestamp, RefPtr<WindowProxy>&& viewArg, int detailArg, IsTrusted isTrusted)

 52 UIEvent::UIEvent(const AtomString& eventType, const UIEventInit& initializer)
 53     : Event(eventType, initializer, IsTrusted::No)
 54     , m_view(initializer.view.get())
 55     , m_detail(initializer.detail)
 56 {
 57 }
 58 
 59 UIEvent::~UIEvent() = default;
 60 
 61 void UIEvent::initUIEvent(const AtomString& typeArg, bool canBubbleArg, bool cancelableArg, RefPtr<WindowProxy>&& viewArg, int detailArg)
 62 {
 63     if (isBeingDispatched())
 64         return;
 65 
 66     initEvent(typeArg, canBubbleArg, cancelableArg);
 67 
 68     m_view = viewArg;
 69     m_detail = detailArg;
 70 }
