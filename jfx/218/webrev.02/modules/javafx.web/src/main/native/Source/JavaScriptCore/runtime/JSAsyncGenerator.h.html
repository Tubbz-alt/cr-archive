 26 #pragma once
 27 
 28 #include "JSGenerator.h"
 29 #include "JSInternalFieldObjectImpl.h"
 30 
 31 namespace JSC {
 32 
 33 class JSAsyncGenerator final : public JSInternalFieldObjectImpl<8> {
 34 public:
 35     using Base = JSInternalFieldObjectImpl<8>;
 36 
 37     // JSAsyncGenerator has one inline storage slot, which is pointing internalField(0).
 38     static size_t allocationSize(Checked<size_t> inlineCapacity)
 39     {
 40         ASSERT_UNUSED(inlineCapacity, inlineCapacity == 0U);
 41         return sizeof(JSAsyncGenerator);
 42     }
 43 
 44     template<typename CellType, SubspaceAccess mode>
 45     static IsoSubspace* subspaceFor(VM& vm)
 46     {
 47         return vm.asyncGeneratorSpace<mode>();
 48     }
 49 
 50     enum class AsyncGeneratorState : int32_t {
 51         Completed = -1,
 52         Executing = -2,
 53         SuspendedStart = -3,
 54         SuspendedYield = -4,
 55         AwaitingReturn = -5,
 56     };
 57     static_assert(static_cast<int32_t>(AsyncGeneratorState::Completed) == static_cast<int32_t>(JSGenerator::GeneratorState::Completed));
 58     static_assert(static_cast<int32_t>(AsyncGeneratorState::Executing) == static_cast<int32_t>(JSGenerator::GeneratorState::Executing));
 59 
 60     enum class AsyncGeneratorSuspendReason : int32_t {
 61         None = 0,
 62         Yield = -1,
 63         Await = -2
 64     };
 65 
 66     enum class Field : uint32_t {
 67         // FIXME: JSAsyncGenerator should support PolyProto, since generator tends to be created with poly proto mode.
 68         // We reserve the first internal field for PolyProto property. This offset is identical to JSFinalObject's first inline storage slot which will be used for PolyProto.
 69         PolyProto = 0,
 70         State,
 71         Next,
 72         This,
 73         Frame,
 74         SuspendReason,
 75         QueueFirst,
 76         QueueLast,
 77     };
 78     static_assert(numberOfInternalFields == 8);
 79     static std::array<JSValue, numberOfInternalFields> initialValues()
 80     {
 81         return { {
 82             jsNull(),
 83             jsNumber(static_cast<int32_t>(AsyncGeneratorState::SuspendedStart)),
 84             jsUndefined(),
 85             jsUndefined(),
 86             jsUndefined(),
 87             jsNumber(static_cast<int32_t>(AsyncGeneratorSuspendReason::None)),
 88             jsNull(),
 89             jsNull(),
 90         } };
 91     }
 92 
 93     static JSAsyncGenerator* create(VM&, Structure*);
 94     static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
 95 
 96     DECLARE_EXPORT_INFO;
 97 
 98     static void visitChildren(JSCell*, SlotVisitor&);
 99 
100 protected:
101     JSAsyncGenerator(VM&, Structure*);
102     void finishCreation(VM&);
103 };
104 
105 } // namespace JSC
