 30 namespace WebCore {
 31 
 32 class MediaPlayerEnums {
 33 public:
 34     enum class NetworkState : uint8_t {
 35         Empty,
 36         Idle,
 37         Loading,
 38         Loaded,
 39         FormatError,
 40         NetworkError,
 41         DecodeError
 42     };
 43 
 44     enum class ReadyState : uint8_t {
 45         HaveNothing,
 46         HaveMetadata,
 47         HaveCurrentData,
 48         HaveFutureData,
 49         HaveEnoughData
 50     };
 51 
 52     enum class MovieLoadType : uint8_t {
 53         Unknown,
 54         Download,
 55         StoredStream,
 56         LiveStream
 57     };
 58 
 59     enum class Preload : uint8_t {
 60         None,
 61         MetaData,
 62         Auto
 63     };
 64 
 65     enum class VideoGravity : uint8_t {
 66         Resize,
 67         ResizeAspect,
 68         ResizeAspectFill
 69     };
 70 
 71     enum class SupportsType : uint8_t {
 72         IsNotSupported,
 73         IsSupported,
 74         MayBeSupported
 75     };
 76 
 77     enum {
 78         VideoFullscreenModeNone = 0,
 79         VideoFullscreenModeStandard = 1 << 0,
 80         VideoFullscreenModePictureInPicture = 1 << 1,
 81     };
 82     typedef uint32_t VideoFullscreenMode;
 83 
 84     enum class BufferingPolicy : uint8_t {
 85         Default,
 86         LimitReadAhead,
 87         MakeResourcesPurgeable,
 88         PurgeResources,
 89     };
 90 
 91     enum class MediaEngineIdentifier : uint8_t {
 92         AVFoundation,
 93         AVFoundationMSE,
 94         AVFoundationMediaStream,
 95         AVFoundationCF,
 96         GStreamer,
 97         GStreamerMSE,
 98         HolePunch,
 99         MediaFoundation,
100         MockMSE,
101     };
102 };
103 
104 WTF::String convertEnumerationToString(MediaPlayerEnums::ReadyState);
105 WTF::String convertEnumerationToString(MediaPlayerEnums::NetworkState);
106 WTF::String convertEnumerationToString(MediaPlayerEnums::Preload);
107 WTF::String convertEnumerationToString(MediaPlayerEnums::SupportsType);
108 WTF::String convertEnumerationToString(MediaPlayerEnums::BufferingPolicy);
109 
110 } // namespace WebCore
