 21 #pragma once
 22 
 23 #include "JSObject.h"
 24 #include "RuntimeType.h"
 25 #include "StackFrame.h"
 26 
 27 namespace JSC {
 28 
 29 class ErrorInstance : public JSNonFinalObject {
 30 public:
 31     using Base = JSNonFinalObject;
 32     static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetPropertyNames;
 33     static constexpr bool needsDestruction = true;
 34 
 35     static void destroy(JSCell* cell)
 36     {
 37         static_cast<ErrorInstance*>(cell)->ErrorInstance::~ErrorInstance();
 38     }
 39 
 40     template<typename CellType, SubspaceAccess mode>
 41     static IsoSubspace* subspaceFor(VM& vm)
 42     {
 43         return vm.errorInstanceSpace<mode>();
 44     }
 45 
 46     enum SourceTextWhereErrorOccurred { FoundExactSource, FoundApproximateSource };
 47     typedef String (*SourceAppender) (const String& originalMessage, const String& sourceText, RuntimeType, SourceTextWhereErrorOccurred);
 48 
 49     DECLARE_EXPORT_INFO;
 50 
 51     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
 52     {
 53         return Structure::create(vm, globalObject, prototype, TypeInfo(ErrorInstanceType, StructureFlags), info());
 54     }
 55 
 56     static ErrorInstance* create(JSGlobalObject* globalObject, VM& vm, Structure* structure, const String& message, SourceAppender appender = nullptr, RuntimeType type = TypeNothing, bool useCurrentFrame = true)
 57     {
 58         ErrorInstance* instance = new (NotNull, allocateCell<ErrorInstance>(vm.heap)) ErrorInstance(vm, structure);
 59         instance->m_sourceAppender = appender;
 60         instance->m_runtimeTypeForCause = type;
 61         instance->finishCreation(globalObject, vm, message, useCurrentFrame);
 62         return instance;
 63     }
 64 
 65     static ErrorInstance* create(JSGlobalObject*, Structure*, JSValue message, SourceAppender = nullptr, RuntimeType = TypeNothing, bool useCurrentFrame = true);
 66 
 67     bool hasSourceAppender() const { return !!m_sourceAppender; }
 68     SourceAppender sourceAppender() const { return m_sourceAppender; }
 69     void setSourceAppender(SourceAppender appender) { m_sourceAppender = appender; }
 70     void clearSourceAppender() { m_sourceAppender = nullptr; }
 71     void setRuntimeTypeForCause(RuntimeType type) { m_runtimeTypeForCause = type; }
 72     RuntimeType runtimeTypeForCause() const { return m_runtimeTypeForCause; }
 73     void clearRuntimeTypeForCause() { m_runtimeTypeForCause = TypeNothing; }
 74 
 75     void setStackOverflowError() { m_stackOverflowError = true; }
 76     bool isStackOverflowError() const { return m_stackOverflowError; }
 77     void setOutOfMemoryError() { m_outOfMemoryError = true; }
 78     bool isOutOfMemoryError() const { return m_outOfMemoryError; }
 79 
 80     void setNativeGetterTypeError() { m_nativeGetterTypeError = true; }
 81     bool isNativeGetterTypeError() const { return m_nativeGetterTypeError; }
 82 
 83     JS_EXPORT_PRIVATE String sanitizedToString(JSGlobalObject*);
 84 
 85     Vector<StackFrame>* stackTrace() { return m_stackTrace.get(); }
 86 
 87     bool materializeErrorInfoIfNeeded(VM&);
 88     bool materializeErrorInfoIfNeeded(VM&, PropertyName);
 89 
 90     void finalizeUnconditionally(VM&);
 91 
 92 protected:
 93     explicit ErrorInstance(VM&, Structure*);
 94 
 95     void finishCreation(JSGlobalObject*, VM&, const String&, bool useCurrentFrame = true);
 96 
 97     static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
 98     static void getOwnNonIndexPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, EnumerationMode);
 99     static void getStructurePropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, EnumerationMode);
100     static bool defineOwnProperty(JSObject*, JSGlobalObject*, PropertyName, const PropertyDescriptor&, bool shouldThrow);
101     static bool put(JSCell*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&);
102     static bool deleteProperty(JSCell*, JSGlobalObject*, PropertyName);
103 
104     void computeErrorInfo(VM&);
105 
106     SourceAppender m_sourceAppender { nullptr };
107     std::unique_ptr<Vector<StackFrame>> m_stackTrace;
108     unsigned m_line;
109     unsigned m_column;
110     String m_sourceURL;
111     String m_stackString;
112     RuntimeType m_runtimeTypeForCause { TypeNothing };
113     bool m_stackOverflowError { false };
114     bool m_outOfMemoryError { false };
115     bool m_errorInfoMaterialized { false };
116     bool m_nativeGetterTypeError { false };
117 };
118 
119 } // namespace JSC
