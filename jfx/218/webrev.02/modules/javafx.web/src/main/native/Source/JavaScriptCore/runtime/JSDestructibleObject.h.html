26 #pragma once
27 
28 #include "JSObject.h"
29 
30 namespace JSC {
31 
32 struct ClassInfo;
33 
34 class JSDestructibleObject : public JSNonFinalObject {
35 public:
36     typedef JSNonFinalObject Base;
37 
38     static constexpr bool needsDestruction = true;
39 
40     template<typename CellType, SubspaceAccess>
41     static CompleteSubspace* subspaceFor(VM& vm)
42     {
43         return &vm.destructibleObjectSpace;
44     }
45 
46     const ClassInfo* classInfo() const { return m_classInfo; }
47 
48     static ptrdiff_t classInfoOffset() { return OBJECT_OFFSETOF(JSDestructibleObject, m_classInfo); }
49 
50 protected:
51     JSDestructibleObject(VM& vm, Structure* structure, Butterfly* butterfly = 0)
52         : JSNonFinalObject(vm, structure, butterfly)
53         , m_classInfo(structure->classInfo())
54     {
55         ASSERT(m_classInfo);
56     }
57 
58 private:
59     const ClassInfo* m_classInfo;
60 };
61 
62 } // namespace JSC
