27 function next()
28 {
29     "use strict";
30 
31     if (!@isArrayIterator(this))




32         @throwTypeError("%ArrayIteratorPrototype%.next requires that |this| be an Array Iterator instance");
33 
34     var array = @getArrayIteratorInternalField(this, @arrayIteratorFieldIteratedObject);
35     var kind = @getArrayIteratorInternalField(this, @arrayIteratorFieldKind);
36     return @arrayIteratorNextHelper.@call(this, array, kind);












































37 }
38 
39 // FIXME: We have to have this because we can't implement this all as one function and meet our inlining heuristics.
40 // Collectively, next and this have ~130 bytes of bytecodes but our limit is 120.
41 @globalPrivate
42 function arrayIteratorNextHelper(array, kind)
43 {
44     "use strict";
45     var done = true;
46     var value;
47 
48     var index = @getArrayIteratorInternalField(this, @arrayIteratorFieldIndex);
49     if (index !== -1) {

50         var length = array.length >>> 0;
51         if (index < length) {
52             @putArrayIteratorInternalField(this, @arrayIteratorFieldIndex, index + 1);


53             done = false;
54             if (kind === @iterationKindKey)
55                 value = index;
56             else {
57                 value = array[index];
58                 if (kind === @iterationKindEntries)
59                     value = [index, value];
60             }
61         } else
62             @putArrayIteratorInternalField(this, @arrayIteratorFieldIndex, -1);
63     }
64 
65     return { value, done };
66 }
