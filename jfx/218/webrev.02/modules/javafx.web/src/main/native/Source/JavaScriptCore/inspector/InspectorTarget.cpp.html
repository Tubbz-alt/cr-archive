31 #include "config.h"
32 #include "InspectorTarget.h"
33 
34 namespace Inspector {
35 
36 void InspectorTarget::pause()
37 {
38     ASSERT(!m_isPaused);
39     m_isPaused = true;
40 }
41 
42 void InspectorTarget::resume()
43 {
44     ASSERT(m_isPaused);
45     m_isPaused = false;
46 
47     if (m_resumeCallback) {
48         m_resumeCallback();
49         m_resumeCallback = nullptr;
50     }
51 }
52 
53 void InspectorTarget::setResumeCallback(WTF::Function<void()>&& callback)
54 {
55     ASSERT(!m_resumeCallback);
56     m_resumeCallback = WTFMove(callback);
57 }
58 
59 } // namespace Inspector
