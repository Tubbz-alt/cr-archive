 41 struct CSSParserContext {
 42     WTF_MAKE_FAST_ALLOCATED;
 43 public:
 44 
 45     CSSParserContext(CSSParserMode, const URL& baseURL = URL());
 46     WEBCORE_EXPORT CSSParserContext(const Document&, const URL& baseURL = URL(), const String& charset = emptyString());
 47 
 48     URL baseURL;
 49     String charset;
 50     CSSParserMode mode { HTMLStandardMode };
 51     Optional<StyleRuleType> enclosingRuleType;
 52     bool isHTMLDocument { false };
 53 #if ENABLE(TEXT_AUTOSIZING)
 54     bool textAutosizingEnabled { false };
 55 #endif
 56 #if ENABLE(OVERFLOW_SCROLLING_TOUCH)
 57     bool legacyOverflowScrollingTouchEnabled { false };
 58 #endif
 59     bool enforcesCSSMIMETypeInNoQuirksMode { true };
 60     bool useLegacyBackgroundSizeShorthandBehavior { false };
 61     bool springTimingFunctionEnabled { false };
 62     bool constantPropertiesEnabled { false };
 63     bool colorFilterEnabled { false };
 64 #if ENABLE(ATTACHMENT_ELEMENT)
 65     bool attachmentEnabled { false };
 66 #endif
 67     bool deferredCSSParserEnabled { false };
 68     bool scrollBehaviorEnabled { false };
 69 
 70     // This is only needed to support getMatchedCSSRules.
 71     bool hasDocumentSecurityOrigin { false };
 72 
 73     bool useSystemAppearance { false };
 74 
 75     URL completeURL(const String& url) const;









 76 
 77     bool isContentOpaque { false };
 78 };

 80 bool operator==(const CSSParserContext&, const CSSParserContext&);
 81 inline bool operator!=(const CSSParserContext& a, const CSSParserContext& b) { return !(a == b); }

 84 
 85 struct CSSParserContextHash {
 86     static unsigned hash(const CSSParserContext& key)
 87     {
 88         unsigned hash = 0;
 89         if (!key.baseURL.isNull())
 90             hash ^= WTF::URLHash::hash(key.baseURL);
 91         if (!key.charset.isEmpty())
 92             hash ^= StringHash::hash(key.charset);
 93         unsigned bits = key.isHTMLDocument                  << 0
 94 #if ENABLE(TEXT_AUTOSIZING)
 95             & key.textAutosizingEnabled                     << 1
 96 #endif
 97 #if ENABLE(OVERFLOW_SCROLLING_TOUCH)
 98             & key.legacyOverflowScrollingTouchEnabled       << 2
 99 #endif
100             & key.enforcesCSSMIMETypeInNoQuirksMode         << 3
101             & key.useLegacyBackgroundSizeShorthandBehavior  << 4
102             & key.springTimingFunctionEnabled               << 5
103             & key.constantPropertiesEnabled                 << 6
104             & key.colorFilterEnabled                        << 7
105             & key.deferredCSSParserEnabled                  << 8
106             & key.hasDocumentSecurityOrigin                 << 9
107             & key.useSystemAppearance                       << 10
108 #if ENABLE(ATTACHMENT_ELEMENT)
109             & key.attachmentEnabled                         << 11
110 #endif
111             & key.scrollBehaviorEnabled                     << 12
112             & key.mode                                      << 13; // Keep this last.
113         hash ^= WTF::intHash(bits);
114         return hash;
115     }
116     static bool equal(const CSSParserContext& a, const CSSParserContext& b)
117     {
118         return a == b;
119     }
120     static const bool safeToCompareToEmptyOrDeleted = false;
121 };
