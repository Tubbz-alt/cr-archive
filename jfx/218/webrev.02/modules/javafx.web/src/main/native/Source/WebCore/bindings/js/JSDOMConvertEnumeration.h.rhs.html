33 // Specialized by generated code for IDL enumeration conversion.
34 template<typename T> Optional<T> parseEnumeration(JSC::JSGlobalObject&, JSC::JSValue);
35 template<typename T> const char* expectedEnumerationValues();
36 
37 // Specialized by generated code for IDL enumeration conversion.
38 template<typename T> JSC::JSString* convertEnumerationToJS(JSC::JSGlobalObject&, T);
39 
40 
41 template<typename T> struct Converter<IDLEnumeration<T>> : DefaultConverter<IDLEnumeration<T>> {
42     template<typename ExceptionThrower = DefaultExceptionThrower>
43     static T convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value, ExceptionThrower&& exceptionThrower = ExceptionThrower())
44     {
45         auto& vm = JSC::getVM(&lexicalGlobalObject);
46         auto throwScope = DECLARE_THROW_SCOPE(vm);
47 
48         auto result = parseEnumeration<T>(lexicalGlobalObject, value);
49         RETURN_IF_EXCEPTION(throwScope, { });
50 
51         if (UNLIKELY(!result)) {
52             exceptionThrower(lexicalGlobalObject, throwScope);
53             return { };
54         }
55         return result.value();
56     }
57 };
58 
59 template<typename T> struct JSConverter<IDLEnumeration<T>> {
60     static constexpr bool needsState = true;
61     static constexpr bool needsGlobalObject = false;
62 
63     static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, T value)
64     {
65         return convertEnumerationToJS(lexicalGlobalObject, value);
66     }
67 };
