34 const ClassInfo JSSetIterator::s_info = { "Set Iterator", nullptr, nullptr, nullptr, CREATE_METHOD_TABLE(JSSetIterator) };

36 void JSSetIterator::finishCreation(VM& vm, JSSet* iteratedObject)
37 {
38     Base::finishCreation(vm);
39     m_set.set(vm, this, iteratedObject);
40     setIterator(vm, m_set->head());
41 }
42 
43 void JSSetIterator::visitChildren(JSCell* cell, SlotVisitor& visitor)
44 {
45     JSSetIterator* thisObject = jsCast<JSSetIterator*>(cell);
46     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
47     Base::visitChildren(thisObject, visitor);
48     visitor.append(thisObject->m_set);
49     visitor.append(thisObject->m_iter);
50 }
51 
52 JSValue JSSetIterator::createPair(CallFrame* callFrame, JSValue key, JSValue value)
53 {
54     MarkedArgumentBuffer args;
55     args.append(key);
56     args.append(value);
57     ASSERT(!args.hasOverflowed());
58     JSGlobalObject* globalObject = callFrame->jsCallee()->globalObject(callFrame->vm());
59     return constructArray(callFrame, 0, globalObject, args);
60 }
