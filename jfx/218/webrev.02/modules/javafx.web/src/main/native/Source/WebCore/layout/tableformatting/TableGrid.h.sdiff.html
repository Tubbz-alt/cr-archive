 40 class TableGrid {

 49     using SlotPosition = IntPoint;
 50 
 51     // Cell represents a <td> or <th>. It can span multiple slots in the grid.
 52     using CellSize = IntSize;
 53     struct CellInfo : public CanMakeWeakPtr<CellInfo> {
 54         WTF_MAKE_STRUCT_FAST_ALLOCATED;
 55         CellInfo(const Box& tableCellBox, SlotPosition, CellSize);
 56 
 57         const Box& tableCellBox;
 58         SlotPosition position;
 59         CellSize size;
 60     };
 61     using CellList = WTF::ListHashSet<std::unique_ptr<CellInfo>>;
 62     CellList& cells() { return m_cellList; }
 63 
 64     // Column represents a vertical set of slots in the grid. A column has min/max and final width.
 65     class ColumnsContext;
 66     class Column {
 67     public:
 68         void setWidthConstraints(FormattingContext::IntrinsicWidthConstraints);
 69         FormattingContext::IntrinsicWidthConstraints widthConstraints() const;
 70 



 71         void setLogicalWidth(LayoutUnit);
 72         LayoutUnit logicalWidth() const;
 73 
 74         void setLogicalLeft(LayoutUnit);
 75         LayoutUnit logicalLeft() const;
 76 
 77         LayoutUnit logicalRight() const { return logicalLeft() + logicalWidth(); }
 78 
 79     private:
 80         friend class ColumnsContext;
 81         Column() = default;
 82 
 83         FormattingContext::IntrinsicWidthConstraints m_widthConstraints;
 84         LayoutUnit m_computedLogicalWidth;
 85         LayoutUnit m_computedLogicalLeft;
 86 #ifndef NDEBUG


 87         bool m_hasWidthConstraints { false };
 88         bool m_hasComputedWidth { false };
 89         bool m_hasComputedLeft { false };
 90 #endif
 91     };
 92 
 93     class ColumnsContext {
 94     public:
 95         using ColumnList = Vector<Column>;
 96         ColumnList& columns() { return m_columns; }
 97         const ColumnList& columns() const { return m_columns; }

 98 
 99         enum class WidthConstraintsType { Minimum, Maximum };
100         void useAsLogicalWidth(WidthConstraintsType);
101 
102     private:
103         friend class TableGrid;
104         void addColumn();
105 
106         ColumnList m_columns;
107     };

108     ColumnsContext& columnsContext() { return m_columnsContext; }

 41 class TableGrid {
 42     WTF_MAKE_ISO_ALLOCATED(TableGrid);

 45 
 46     void appendCell(const Box&);
 47     void insertCell(const Box&, const Box& before);
 48     void removeCell(const Box&);
 49 
 50     void setHorizontalSpacing(LayoutUnit horizontalSpacing) { m_horizontalSpacing = horizontalSpacing; }
 51     LayoutUnit horizontalSpacing() const { return m_horizontalSpacing; }
 52     LayoutUnit totalHorizontalSpacing() const { return (columnsContext().columns().size() + 1) * horizontalSpacing(); }
 53 
 54     void setVerticalSpacing(LayoutUnit verticalSpacing) { m_verticalSpacing = verticalSpacing; }
 55     LayoutUnit verticalSpacing() const { return m_verticalSpacing; }
 56 
 57     using SlotPosition = IntPoint;
 58 
 59     // Cell represents a <td> or <th>. It can span multiple slots in the grid.
 60     using CellSize = IntSize;
 61     struct CellInfo : public CanMakeWeakPtr<CellInfo> {
 62         WTF_MAKE_STRUCT_FAST_ALLOCATED;
 63         CellInfo(const Box& tableCellBox, SlotPosition, CellSize);
 64 
 65         const Box& tableCellBox;
 66         SlotPosition position;
 67         CellSize size;
 68     };
 69     using CellList = WTF::ListHashSet<std::unique_ptr<CellInfo>>;
 70     CellList& cells() { return m_cellList; }
 71 
 72     // Column represents a vertical set of slots in the grid. A column has min/max and final width.
 73     class ColumnsContext;
 74     class Column {
 75     public:
 76         void setWidthConstraints(FormattingContext::IntrinsicWidthConstraints);
 77         FormattingContext::IntrinsicWidthConstraints widthConstraints() const;
 78 
 79         void setLogicalLeft(LayoutUnit);
 80         LayoutUnit logicalLeft() const;
 81         LayoutUnit logicalRight() const { return logicalLeft() + logicalWidth(); }
 82         void setLogicalWidth(LayoutUnit);
 83         LayoutUnit logicalWidth() const;
 84 
 85         bool hasFixedWidth() const;

 86 
 87         const Box* columnBox() const { return m_columnBox.get(); }
 88 
 89     private:
 90         friend class ColumnsContext;
 91         Column(const Box* columnBox);
 92 
 93         FormattingContext::IntrinsicWidthConstraints m_widthConstraints;
 94         LayoutUnit m_computedLogicalWidth;
 95         LayoutUnit m_computedLogicalLeft;
 96         WeakPtr<const Box> m_columnBox;
 97 
 98 #if ASSERT_ENABLED
 99         bool m_hasWidthConstraints { false };
100         bool m_hasComputedWidth { false };
101         bool m_hasComputedLeft { false };
102 #endif
103     };
104 
105     class ColumnsContext {
106     public:
107         using ColumnList = Vector<Column>;

109         const ColumnList& columns() const { return m_columns; }

111 
112         LayoutUnit logicalWidth() const { return columns().last().logicalRight() - columns().first().logicalLeft(); }

114     private:

120     ColumnsContext& columnsContext() { return m_columnsContext; }

135
