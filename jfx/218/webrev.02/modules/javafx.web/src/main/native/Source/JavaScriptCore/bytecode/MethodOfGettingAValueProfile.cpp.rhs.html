 38 MethodOfGettingAValueProfile MethodOfGettingAValueProfile::fromLazyOperand(
 39     CodeBlock* codeBlock, const LazyOperandValueProfileKey& key)
 40 {
 41     MethodOfGettingAValueProfile result;
 42     result.m_kind = LazyOperand;
 43     result.u.lazyOperand.codeBlock = codeBlock;
 44     result.u.lazyOperand.bytecodeOffset = key.bytecodeIndex();
 45     result.u.lazyOperand.operand = key.operand();
 46     return result;
 47 }
 48 
 49 void MethodOfGettingAValueProfile::emitReportValue(CCallHelpers& jit, JSValueRegs regs) const
 50 {
 51     switch (m_kind) {
 52     case None:
 53         return;
 54 
 55     case Ready:
 56         jit.storeValue(regs, u.profile->specFailBucket(0));
 57         return;
 58 
 59     case LazyOperand: {
 60         LazyOperandValueProfileKey key(u.lazyOperand.bytecodeOffset, u.lazyOperand.operand);
 61 
 62         ConcurrentJSLocker locker(u.lazyOperand.codeBlock->m_lock);
 63         LazyOperandValueProfile* profile =
 64             u.lazyOperand.codeBlock->lazyOperandValueProfiles(locker).add(locker, key);
 65         jit.storeValue(regs, profile->specFailBucket(0));
 66         return;
 67     }
 68 
 69     case UnaryArithProfileReady: {
 70         u.unaryArithProfile->emitObserveResult(jit, regs, DoNotHaveTagRegisters);
 71         return;
 72     }
 73 
 74     case BinaryArithProfileReady: {
 75         u.binaryArithProfile->emitObserveResult(jit, regs, DoNotHaveTagRegisters);
 76         return;
 77     }
 78     }
 79 
 80     RELEASE_ASSERT_NOT_REACHED();
 81 }
 82 
 83 void MethodOfGettingAValueProfile::reportValue(JSValue value)
 84 {
 85     switch (m_kind) {
 86     case None:
 87         return;
 88 
 89     case Ready:
 90         *u.profile->specFailBucket(0) = JSValue::encode(value);
 91         return;
 92 
 93     case LazyOperand: {
 94         LazyOperandValueProfileKey key(u.lazyOperand.bytecodeOffset, u.lazyOperand.operand);
 95 
 96         ConcurrentJSLocker locker(u.lazyOperand.codeBlock->m_lock);
 97         LazyOperandValueProfile* profile =
 98             u.lazyOperand.codeBlock->lazyOperandValueProfiles(locker).add(locker, key);
 99         *profile->specFailBucket(0) = JSValue::encode(value);
100         return;
101     }
102 
103     case UnaryArithProfileReady: {
104         u.unaryArithProfile->observeResult(value);
105         return;
106     }
107 
108     case BinaryArithProfileReady: {
109         u.binaryArithProfile->observeResult(value);
110         return;
111     }
112     }
113 
114     RELEASE_ASSERT_NOT_REACHED();
115 }
