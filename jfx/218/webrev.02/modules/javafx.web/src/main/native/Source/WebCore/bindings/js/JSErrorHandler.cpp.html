 50 inline JSErrorHandler::JSErrorHandler(JSObject& listener, JSObject& wrapper, bool isAttribute, DOMWrapperWorld& world)
 51     : JSEventListener(&listener, &wrapper, isAttribute, world)

 55 Ref<JSErrorHandler> JSErrorHandler::create(JSC::JSObject& listener, JSC::JSObject& wrapper, bool isAttribute, DOMWrapperWorld& world)
 56 {
 57     return adoptRef(*new JSErrorHandler(listener, wrapper, isAttribute, world));
 58 }

 60 JSErrorHandler::~JSErrorHandler() = default;

 62 void JSErrorHandler::handleEvent(ScriptExecutionContext& scriptExecutionContext, Event& event)
 63 {
 64     if (!is<ErrorEvent>(event))
 65         return JSEventListener::handleEvent(scriptExecutionContext, event);
 66 
 67     VM& vm = scriptExecutionContext.vm();
 68     JSLockHolder lock(vm);
 69 
 70     JSObject* jsFunction = this->jsFunction(scriptExecutionContext);
 71     if (!jsFunction)
 72         return;
 73 
 74     auto* globalObject = toJSDOMGlobalObject(scriptExecutionContext, isolatedWorld());
 75     if (!globalObject)
 76         return;
 77 
 78     CallData callData;
 79     CallType callType = jsFunction->methodTable(vm)->getCallData(jsFunction, callData);
 80 
 81     if (callType != CallType::None) {
 82         Ref<JSErrorHandler> protectedThis(*this);
 83 
 84         Event* savedEvent = globalObject->currentEvent();
 85         globalObject->setCurrentEvent(&event);
 86 
 87         auto& errorEvent = downcast<ErrorEvent>(event);
 88 
 89         MarkedArgumentBuffer args;
 90         args.append(toJS<IDLDOMString>(*globalObject, errorEvent.message()));
 91         args.append(toJS<IDLUSVString>(*globalObject, errorEvent.filename()));
 92         args.append(toJS<IDLUnsignedLong>(errorEvent.lineno()));
 93         args.append(toJS<IDLUnsignedLong>(errorEvent.colno()));
 94         args.append(errorEvent.error(*globalObject));
 95         ASSERT(!args.hasOverflowed());
 96 
 97         VM& vm = globalObject->vm();
 98         VMEntryScope entryScope(vm, vm.entryScope ? vm.entryScope->globalObject() : globalObject);
 99 
100         JSExecState::instrumentFunctionCall(&scriptExecutionContext, callType, callData);
101 
102         NakedPtr<JSC::Exception> exception;
103         JSValue returnValue = JSExecState::profiledCall(globalObject, JSC::ProfilingReason::Other, jsFunction, callType, callData, globalObject, args, exception);
104 
105         InspectorInstrumentation::didCallFunction(&scriptExecutionContext);
106 
107         globalObject->setCurrentEvent(savedEvent);
108 
109         if (exception)
110             reportException(globalObject, exception);
111         else {
112             if (returnValue.isTrue())
113                 event.preventDefault();
114         }
115     }
116 }
