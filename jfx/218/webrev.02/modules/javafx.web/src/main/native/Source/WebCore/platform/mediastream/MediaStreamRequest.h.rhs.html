26 #pragma once
27 
28 #if ENABLE(MEDIA_STREAM)
29 
30 #include "MediaConstraints.h"
31 
32 namespace WebCore {
33 
34 struct MediaStreamRequest {
35     enum class Type { UserMedia, DisplayMedia };
36     Type type { Type::UserMedia };
37     MediaConstraints audioConstraints;
38     MediaConstraints videoConstraints;
39     bool isUserGesturePriviledged { false };
40 
41     template<class Encoder>
42     void encode(Encoder& encoder) const
43     {
44         encoder.encodeEnum(type);
45         encoder << audioConstraints;
46         encoder << videoConstraints;
47         encoder << isUserGesturePriviledged;
48     }
49 
50     template <class Decoder> static Optional<MediaStreamRequest> decode(Decoder& decoder)
51     {
52         MediaStreamRequest request;
53         if (decoder.decodeEnum(request.type) && decoder.decode(request.audioConstraints) && decoder.decode(request.videoConstraints) && decoder.decode(request.isUserGesturePriviledged))
54             return request;
55 
56         return WTF::nullopt;
57     }
58 };
59 
60 } // namespace WebCore
61 
62 #else
63 
64 namespace WebCore {
65 
66 struct MediaStreamRequest {
67     enum class Type { UserMedia, DisplayMedia };
68     Type type;
69 };
70 
71 } // namespace WebCore
72 
73 #endif // ENABLE(MEDIA_STREAM)

79         WebCore::MediaStreamRequest::Type,
80         WebCore::MediaStreamRequest::Type::UserMedia,
81         WebCore::MediaStreamRequest::Type::DisplayMedia
82     >;
83 };
