33 // This is an Arguments-class object that we create when you do function.arguments, or you say
34 // "arguments" inside a function in strict mode. It behaves almpst entirely like an ordinary
35 // JavaScript object. All of the arguments values are simply copied from the stack (possibly via
36 // some sophisticated ValueRecovery's if an optimizing compiler is in play) and the appropriate
37 // properties of the object are populated. The only reason why we need a special class is to make
38 // the object claim to be "Arguments" from a toString standpoint, and to avoid materializing the
39 // caller/callee/@@iterator properties unless someone asks for them.
40 class ClonedArguments final : public JSNonFinalObject {
41 public:
42     using Base = JSNonFinalObject;
43     static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetPropertyNames;
44 
45     template<typename CellType, SubspaceAccess mode>
46     static IsoSubspace* subspaceFor(VM& vm)
47     {
48         static_assert(!CellType::needsDestruction, "");
49         return &vm.clonedArgumentsSpace;
50     }
51 
52 private:
53     ClonedArguments(VM&, Structure*, Butterfly*);
54 
55 public:
56     static ClonedArguments* createEmpty(VM&, Structure*, JSFunction* callee, unsigned length);
57     static ClonedArguments* createEmpty(JSGlobalObject*, JSFunction* callee, unsigned length);
58     static ClonedArguments* createWithInlineFrame(JSGlobalObject*, CallFrame* targetFrame, InlineCallFrame*, ArgumentsMode);
59     static ClonedArguments* createWithMachineFrame(JSGlobalObject*, CallFrame* targetFrame, ArgumentsMode);
60     static ClonedArguments* createByCopyingFrom(JSGlobalObject*, Structure*, Register* argumentsStart, unsigned length, JSFunction* callee);
61 
62     static Structure* createStructure(VM&, JSGlobalObject*, JSValue prototype);
63     static Structure* createSlowPutStructure(VM&, JSGlobalObject*, JSValue prototype);
64 
65     static void visitChildren(JSCell*, SlotVisitor&);
66 
67     DECLARE_INFO;
68 
69 private:
70     static Structure* createStructure(VM&, JSGlobalObject*, JSValue prototype, IndexingType);
71 
72     static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
73     static void getOwnPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, EnumerationMode);
74     static bool put(JSCell*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&);
75     static bool deleteProperty(JSCell*, JSGlobalObject*, PropertyName);
76     static bool defineOwnProperty(JSObject*, JSGlobalObject*, PropertyName, const PropertyDescriptor&, bool shouldThrow);
77 
78     bool specialsMaterialized() const { return !m_callee; }
79     void materializeSpecials(JSGlobalObject*);
80     void materializeSpecialsIfNecessary(JSGlobalObject*);
81 
82     WriteBarrier<JSFunction> m_callee; // Set to nullptr when we materialize all of our special properties.
83 };
