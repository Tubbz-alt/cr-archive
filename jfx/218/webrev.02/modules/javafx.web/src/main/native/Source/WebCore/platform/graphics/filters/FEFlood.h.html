24 #include "Color.h"
25 #include "Filter.h"
26 #include "FilterEffect.h"
27 
28 namespace WebCore {
29 
30 class FEFlood : public FilterEffect {
31 public:
32     static Ref<FEFlood> create(Filter&, const Color&, float);
33 
34     const Color& floodColor() const { return m_floodColor; }
35     bool setFloodColor(const Color&);
36 
37     float floodOpacity() const { return m_floodOpacity; }
38     bool setFloodOpacity(float);
39 
40 #if !USE(CG)
41     // feFlood does not perform color interpolation of any kind, so the result is always in the current
42     // color space regardless of the value of color-interpolation-filters.
43     void setOperatingColorSpace(ColorSpace) override { FilterEffect::setResultColorSpace(ColorSpace::SRGB); }
44     void setResultColorSpace(ColorSpace) override { FilterEffect::setResultColorSpace(ColorSpace::SRGB); }
45 #endif
46 
47 private:
48     FEFlood(Filter&, const Color&, float);
49 
50     const char* filterName() const final { return "FEFlood"; }
51 
52     void platformApplySoftware() override;
53 
54     void determineAbsolutePaintRect() override { setAbsolutePaintRect(enclosingIntRect(maxEffectRect())); }
55 
56     WTF::TextStream& externalRepresentation(WTF::TextStream&, RepresentationType) const override;
57 
58     Color m_floodColor;
59     float m_floodOpacity;
60 };

62 } // namespace WebCore
