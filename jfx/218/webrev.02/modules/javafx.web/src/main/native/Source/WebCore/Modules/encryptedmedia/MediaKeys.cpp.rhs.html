 57 ExceptionOr<Ref<MediaKeySession>> MediaKeys::createSession(ScriptExecutionContext& context, MediaKeySessionType sessionType)
 58 {
 59     // https://w3c.github.io/encrypted-media/#dom-mediakeys-setservercertificate
 60     // W3C Editor's Draft 09 November 2016
 61     LOG(EME, "EME - check if a new session can be created");
 62 
 63     // When this method is invoked, the user agent must run the following steps:
 64     // 1. If this object's supported session types value does not contain sessionType, throw [WebIDL] a NotSupportedError.
 65     if (!m_supportedSessionTypes.contains(sessionType))
 66         return Exception(NotSupportedError);
 67 
 68     // 2. If the implementation does not support MediaKeySession operations in the current state, throw [WebIDL] an InvalidStateError.
 69     if (!m_implementation->supportsSessions())
 70         return Exception(InvalidStateError);
 71 
 72     auto instanceSession = m_instance->createSession();
 73     if (!instanceSession)
 74         return Exception(InvalidStateError);
 75 
 76     // 3. Let session be a new MediaKeySession object, and initialize it as follows:
 77     // NOTE: Continued in MediaKeySession.
 78     // 4. Return session.
 79     auto session = MediaKeySession::create(context, makeWeakPtr(*this), sessionType, m_useDistinctiveIdentifier, m_implementation.copyRef(), instanceSession.releaseNonNull());
 80     m_sessions.append(session.copyRef());
 81     return session;
 82 }

 84 void MediaKeys::setServerCertificate(ScriptExecutionContext& context, const BufferSource& serverCertificate, Ref<DeferredPromise>&& promise)
 85 {
 86     // https://w3c.github.io/encrypted-media/#dom-mediakeys-setservercertificate
 87     // W3C Editor's Draft 09 November 2016
 88 
 89     // When this method is invoked, the user agent must run the following steps:
 90     // 1. If the Key System implementation represented by this object's cdm implementation value does not support
 91     //    server certificates, return a promise resolved with false.
 92     if (!m_implementation->supportsServerCertificates()) {
 93         promise->resolve<IDLBoolean>(false);
 94         return;
 95     }
 96 
 97     // 2. If serverCertificate is an empty array, return a promise rejected with a new a newly created TypeError.
 98     if (!serverCertificate.length()) {
 99         promise->reject(TypeError);
100         return;
101     }
102 
103     // 3. Let certificate be a copy of the contents of the serverCertificate parameter.
104     auto certificate = SharedBuffer::create(serverCertificate.data(), serverCertificate.length());
105 
106     // 4. Let promise be a new promise.
107     // 5. Run the following steps in parallel:
108 
109     context.eventLoop().queueTask(TaskSource::Networking, [this, certificate = WTFMove(certificate), promise = WTFMove(promise)] () mutable {
110         // 5.1. Use this object's cdm instance to process certificate.
111         if (m_instance->setServerCertificate(WTFMove(certificate)) == CDMInstance::Failed) {
112             // 5.2. If the preceding step failed, resolve promise with a new DOMException whose name is the appropriate error name.
113             promise->reject(InvalidStateError);
114             return;
115         }
116 
117         // 5.1. Resolve promise with true.
118         promise->resolve<IDLBoolean>(true);
119     });
120 
121     // 6. Return promise.
122 }

124 void MediaKeys::attachCDMClient(CDMClient& client)
125 {
126     ASSERT(!m_cdmClients.contains(&client));
127     m_cdmClients.append(&client);
128 }
129 
130 void MediaKeys::detachCDMClient(CDMClient& client)
131 {
132     ASSERT(m_cdmClients.contains(&client));
133     m_cdmClients.removeFirst(&client);
134 }
135 
136 void MediaKeys::attemptToResumePlaybackOnClients()
137 {
138     for (auto* cdmClient : m_cdmClients)
139         cdmClient->cdmClientAttemptToResumePlaybackIfNecessary();
140 }
141 
142 bool MediaKeys::hasOpenSessions() const
143 {
144     return std::any_of(m_sessions.begin(), m_sessions.end(),
145         [](auto& session) {
146             return !session->isClosed();
147         });
148 }
