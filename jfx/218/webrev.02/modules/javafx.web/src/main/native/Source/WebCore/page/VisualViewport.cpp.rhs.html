 53 ScriptExecutionContext* VisualViewport::scriptExecutionContext() const
 54 {
 55     auto window = this->window();
 56     if (!window)
 57         return nullptr;
 58     return static_cast<ContextDestructionObserver*>(window)->scriptExecutionContext();
 59 }

 61 bool VisualViewport::addEventListener(const AtomString& eventType, Ref<EventListener>&& listener, const AddEventListenerOptions& options)
 62 {
 63     if (!EventTarget::addEventListener(eventType, WTFMove(listener), options))
 64         return false;
 65 
 66     if (auto* frame = this->frame())
 67         frame->document()->addListenerTypeIfNeeded(eventType);
 68     return true;
 69 }

 71 void VisualViewport::updateFrameLayout() const
 72 {
 73     ASSERT(frame());
 74     frame()->document()->updateLayoutIgnorePendingStylesheets(Document::RunPostLayoutTasks::Synchronously);
 75 }

131 double VisualViewport::scale() const
132 {
133     // Subframes always have scale 1 since they aren't scaled relative to their parent frame.
134     auto* frame = this->frame();
135     if (!frame || !frame->isMainFrame())
136         return 1;
137 
138     updateFrameLayout();
139     return m_scale;
140 }
141 
142 void VisualViewport::update()
143 {
144     double offsetLeft = 0;
145     double offsetTop = 0;
146     m_pageLeft = 0;
147     m_pageTop = 0;
148     double width = 0;
149     double height = 0;
150     double scale = 1;
151 
152     auto frame = makeRefPtr(this->frame());
153     if (frame) {
154         if (auto* view = frame->view()) {
155             auto visualViewportRect = view->visualViewportRect();
156             auto layoutViewportRect = view->layoutViewportRect();
157             auto pageZoomFactor = frame->pageZoomFactor();
158             ASSERT(pageZoomFactor);
159             offsetLeft = (visualViewportRect.x() - layoutViewportRect.x()) / pageZoomFactor;
160             offsetTop = (visualViewportRect.y() - layoutViewportRect.y()) / pageZoomFactor;
161             m_pageLeft = visualViewportRect.x() / pageZoomFactor;
162             m_pageTop = visualViewportRect.y() / pageZoomFactor;
163             width = visualViewportRect.width() / pageZoomFactor;
164             height = visualViewportRect.height() / pageZoomFactor;
165         }
166         if (auto* page = frame->page())
167             scale = page->pageScaleFactor();
168     }
169 
170     RefPtr<Document> document = frame ? frame->document() : nullptr;
171     if (m_offsetLeft != offsetLeft || m_offsetTop != offsetTop) {
172         if (document)
173             document->setNeedsVisualViewportScrollEvent();
174         m_offsetLeft = offsetLeft;
175         m_offsetTop = offsetTop;
176     }
177     if (m_width != width || m_height != height || m_scale != scale) {
178         if (document)
179             document->setNeedsVisualViewportResize();
180         m_width = width;
181         m_height = height;
182         m_scale = scale;
183     }
184 }
