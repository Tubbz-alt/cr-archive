27 
28 namespace WebCore {
29 
30 class ResourceRequest;
31 
32 struct SameSiteInfo {
33     WEBCORE_EXPORT static SameSiteInfo create(const ResourceRequest&);
34 
35     bool isSameSite { false };
36     bool isTopSite { false };

37 
38     template <class Encoder> void encode(Encoder&) const;
39     template <class Decoder> static bool decode(Decoder&, SameSiteInfo&);
40 };
41 
42 template <class Encoder>
43 void SameSiteInfo::encode(Encoder& encoder) const
44 {
45     encoder << isSameSite;
46     encoder << isTopSite;

47 }
48 
49 template <class Decoder>
50 bool SameSiteInfo::decode(Decoder& decoder, SameSiteInfo& info)
51 {
52     if (!decoder.decode(info.isSameSite))
53         return false;
54     if (!decoder.decode(info.isTopSite))
55         return false;


56     return true;
57 }
