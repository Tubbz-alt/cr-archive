 28 #include <memory>
 29 #include <wtf/FastMalloc.h>
 30 
 31 namespace WTF {
 32 
 33 namespace Detail {
 34 
 35 template<typename Out, typename... In>
 36 class CallableWrapperBase {
 37     WTF_MAKE_FAST_ALLOCATED;
 38 public:
 39     virtual ~CallableWrapperBase() { }
 40     virtual Out call(In...) = 0;
 41 };
 42 
 43 template<typename, typename, typename...> class CallableWrapper;
 44 
 45 template<typename CallableType, typename Out, typename... In>
 46 class CallableWrapper : public CallableWrapperBase<Out, In...> {
 47 public:
 48     explicit CallableWrapper(CallableType&& callable)
 49         : m_callable(WTFMove(callable)) { }
 50     CallableWrapper(const CallableWrapper&) = delete;
 51     CallableWrapper& operator=(const CallableWrapper&) = delete;
 52     Out call(In... in) final { return m_callable(std::forward<In>(in)...); }
 53 private:
 54     CallableType m_callable;
 55 };
 56 
 57 } // namespace Detail
 58 
 59 template<typename> class Function;
 60 
 61 template <typename Out, typename... In>
 62 class Function<Out(In...)> {
 63     WTF_MAKE_FAST_ALLOCATED;
 64 public:
 65     using Impl = Detail::CallableWrapperBase<Out, In...>;
 66 
 67     Function() = default;
 68     Function(std::nullptr_t) { }
 69     Function(Impl* impl)
 70         : m_callableWrapper(impl)
 71     { }
 72 
 73     template<typename CallableType, class = typename std::enable_if<!(std::is_pointer<CallableType>::value && std::is_function<typename std::remove_pointer<CallableType>::type>::value) && std::is_rvalue_reference<CallableType&&>::value>::type>
 74     Function(CallableType&& callable)
 75         : m_callableWrapper(makeUnique<Detail::CallableWrapper<CallableType, Out, In...>>(WTFMove(callable))) { }
 76 
 77     template<typename FunctionType, class = typename std::enable_if<std::is_pointer<FunctionType>::value && std::is_function<typename std::remove_pointer<FunctionType>::type>::value>::type>
 78     Function(FunctionType f)
 79         : m_callableWrapper(makeUnique<Detail::CallableWrapper<FunctionType, Out, In...>>(WTFMove(f))) { }
 80 
 81     Out operator()(In... in) const
 82     {
 83         ASSERT(m_callableWrapper);
 84         return m_callableWrapper->call(std::forward<In>(in)...);
 85     }
 86 
 87     explicit operator bool() const { return !!m_callableWrapper; }
 88 
 89     template<typename CallableType, class = typename std::enable_if<!(std::is_pointer<CallableType>::value && std::is_function<typename std::remove_pointer<CallableType>::type>::value) && std::is_rvalue_reference<CallableType&&>::value>::type>
 90     Function& operator=(CallableType&& callable)
 91     {
 92         m_callableWrapper = makeUnique<Detail::CallableWrapper<CallableType, Out, In...>>(WTFMove(callable));
 93         return *this;
 94     }
 95 
 96     template<typename FunctionType, class = typename std::enable_if<std::is_pointer<FunctionType>::value && std::is_function<typename std::remove_pointer<FunctionType>::type>::value>::type>
 97     Function& operator=(FunctionType f)
 98     {
 99         m_callableWrapper = makeUnique<Detail::CallableWrapper<FunctionType, Out, In...>>(WTFMove(f));
100         return *this;
101     }
102 
103     Function& operator=(std::nullptr_t)
104     {
105         m_callableWrapper = nullptr;
106         return *this;
107     }
108 
109     Impl* leakImpl()
110     {
111         return m_callableWrapper.release();
112     }
113 
114 private:
115     std::unique_ptr<Impl> m_callableWrapper;
116 };
117 
118 } // namespace WTF
