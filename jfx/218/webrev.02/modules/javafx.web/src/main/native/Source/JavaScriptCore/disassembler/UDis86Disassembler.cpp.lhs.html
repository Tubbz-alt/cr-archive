34 namespace JSC {

36 bool tryToDisassembleWithUDis86(const MacroAssemblerCodePtr<DisassemblyPtrTag>& codePtr, size_t size, const char* prefix, PrintStream& out)
37 {
38     ud_t disassembler;
39     ud_init(&disassembler);
40     ud_set_input_buffer(&disassembler, codePtr.untaggedExecutableAddress<unsigned char*>(), size);
41 #if CPU(X86_64)
42     ud_set_mode(&disassembler, 64);
43 #else
44     ud_set_mode(&disassembler, 32);
45 #endif
46     ud_set_pc(&disassembler, codePtr.untaggedExecutableAddress<uintptr_t>());
47     ud_set_syntax(&disassembler, UD_SYN_ATT);
48 
49     uint64_t currentPC = disassembler.pc;
50     while (ud_disassemble(&disassembler)) {
51         char pcString[20];
52         snprintf(pcString, sizeof(pcString), "0x%" PRIxPTR, static_cast<uintptr_t>(currentPC));
53         out.printf("%s%16s: %s\n", prefix, pcString, ud_insn_asm(&disassembler));
54         currentPC = disassembler.pc;
55     }
56 
57     return true;
58 }
59 
60 } // namespace JSC
