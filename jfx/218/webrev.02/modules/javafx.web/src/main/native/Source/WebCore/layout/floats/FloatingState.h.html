 44 class Box;
 45 class FloatingContext;
 46 class LayoutState;

 48 // FloatingState holds the floating boxes per formatting context.
 49 class FloatingState : public RefCounted<FloatingState> {
 50     WTF_MAKE_ISO_ALLOCATED(FloatingState);
 51 public:
 52     static Ref<FloatingState> create(LayoutState& layoutState, const Container& formattingContextRoot) { return adoptRef(*new FloatingState(layoutState, formattingContextRoot)); }
 53 
 54     const Container& root() const { return *m_formattingContextRoot; }
 55 
 56     Optional<PositionInContextRoot> top(const Container& formattingContextRoot) const;
 57     Optional<PositionInContextRoot> leftBottom(const Container& formattingContextRoot) const;
 58     Optional<PositionInContextRoot> rightBottom(const Container& formattingContextRoot) const;
 59     Optional<PositionInContextRoot> bottom(const Container& formattingContextRoot) const;
 60 
 61     class FloatItem {
 62     public:
 63         FloatItem(const Box&, Display::Box absoluteDisplayBox);
 64 
 65         enum class Position { Left, Right };
 66         FloatItem(Position, Display::Box absoluteDisplayBox);
 67 
 68         bool isLeftPositioned() const { return m_position == Position::Left; }
 69         bool isDescendantOfFormattingRoot(const Container&) const;
 70 
 71         Display::Rect rectWithMargin() const { return m_absoluteDisplayBox.rectWithMargin(); }
 72         UsedHorizontalMargin horizontalMargin() const { return m_absoluteDisplayBox.horizontalMargin(); }
 73         PositionInContextRoot bottom() const { return { m_absoluteDisplayBox.bottom() }; }
 74 
 75     private:
 76         WeakPtr<const Box> m_layoutBox;
 77         Position m_position;
 78         Display::Box m_absoluteDisplayBox;
 79     };
 80     using FloatList = Vector<FloatItem>;
 81     const FloatList& floats() const { return m_floats; }
 82     const FloatItem* last() const { return floats().isEmpty() ? nullptr : &m_floats.last(); }
 83 
 84     void append(FloatItem);
 85     void clear() { m_floats.clear(); }
 86 
 87 private:
 88     friend class FloatingContext;
 89     FloatingState(LayoutState&, const Container& formattingContextRoot);
 90 
 91     LayoutState& layoutState() const { return m_layoutState; }
 92 
 93     Optional<PositionInContextRoot> bottom(const Container& formattingContextRoot, Clear) const;
 94 
 95     LayoutState& m_layoutState;
 96     WeakPtr<const Container> m_formattingContextRoot;
 97     FloatList m_floats;
 98 };
 99 
100 inline Optional<PositionInContextRoot> FloatingState::leftBottom(const Container& formattingContextRoot) const
101 {
102     ASSERT(formattingContextRoot.establishesFormattingContext());
103     return bottom(formattingContextRoot, Clear::Left);
104 }
105 
106 inline Optional<PositionInContextRoot> FloatingState::rightBottom(const Container& formattingContextRoot) const
107 {
108     ASSERT(formattingContextRoot.establishesFormattingContext());
109     return bottom(formattingContextRoot, Clear::Right);
110 }
111 
112 inline Optional<PositionInContextRoot> FloatingState::bottom(const Container& formattingContextRoot) const
113 {
114     ASSERT(formattingContextRoot.establishesFormattingContext());
115     return bottom(formattingContextRoot, Clear::Both);
116 }
117 
118 inline bool FloatingState::FloatItem::isDescendantOfFormattingRoot(const Container& formattingContextRoot) const
119 {
120     ASSERT(formattingContextRoot.establishesFormattingContext());
121     return m_layoutBox->isContainingBlockDescendantOf(downcast<Container>(formattingContextRoot));
122 }
