21 #pragma once
22 
23 #include "LayoutUnit.h"
24 #include <wtf/FastMalloc.h>
25 #include <wtf/Noncopyable.h>
26 
27 namespace WebCore {
28 
29 class RenderTable;
30 

31 class TableLayout {
32     WTF_MAKE_NONCOPYABLE(TableLayout); WTF_MAKE_FAST_ALLOCATED;

33 public:
34     explicit TableLayout(RenderTable* table)
35         : m_table(table)
36     {
37     }
38 
39     virtual ~TableLayout() = default;
40 
41     virtual void computeIntrinsicLogicalWidths(LayoutUnit& minWidth, LayoutUnit& maxWidth) = 0;
42     virtual LayoutUnit scaledWidthFromPercentColumns() const { return 0_lu; }
43     virtual void applyPreferredLogicalWidthQuirks(LayoutUnit& minWidth, LayoutUnit& maxWidth) const = 0;
44     virtual void layout() = 0;
45 
46 protected:
47     // FIXME: Once we enable SATURATED_LAYOUT_ARITHMETHIC, this should just be LayoutUnit::nearlyMax().
48     // Until then though, using nearlyMax causes overflow in some tests, so we just pick a large number.
49     const static int tableMaxWidth = 1000000;
50 
51     RenderTable* m_table;
52 };
53 
54 } // namespace WebCore
