26 #ifndef ResourceLoadPriority_h
27 #define ResourceLoadPriority_h
28 
29 namespace WebCore {
30 
31 enum class ResourceLoadPriority : uint8_t {
32     VeryLow,
33     Low,
34     Medium,
35     High,
36     VeryHigh,
37     Lowest = VeryLow,
38     Highest = VeryHigh,
39 };
40 static constexpr unsigned bitWidthOfResourceLoadPriority = 3;
41 static_assert(static_cast<unsigned>(ResourceLoadPriority::Highest) <= ((1U << bitWidthOfResourceLoadPriority) - 1));
42 
43 static const unsigned resourceLoadPriorityCount { static_cast<unsigned>(ResourceLoadPriority::Highest) + 1 };
44 
45 inline ResourceLoadPriority& operator++(ResourceLoadPriority& priority)
46 {
47     ASSERT(priority != ResourceLoadPriority::Highest);
48     return priority = static_cast<ResourceLoadPriority>(static_cast<int>(priority) + 1);
49 }
50 
51 inline ResourceLoadPriority& operator--(ResourceLoadPriority& priority)
52 {
53     ASSERT(priority != ResourceLoadPriority::Lowest);
54     return priority = static_cast<ResourceLoadPriority>(static_cast<int>(priority) - 1);
55 }
56 
57 }
58 
59 #endif
