60 const GregorianDateTime* DateInstance::calculateGregorianDateTime(ExecState* exec) const
61 {
62     double milli = internalNumber();
63     if (std::isnan(milli))
64         return 0;
65 
66     VM& vm = exec->vm();
67     if (!m_data)
68         m_data = vm.dateInstanceCache.add(milli);
69 
70     if (m_data->m_gregorianDateTimeCachedForMS != milli) {
71         msToGregorianDateTime(vm, milli, WTF::LocalTime, m_data->m_cachedGregorianDateTime);
72         m_data->m_gregorianDateTimeCachedForMS = milli;
73     }
74     return &m_data->m_cachedGregorianDateTime;
75 }
76 
77 const GregorianDateTime* DateInstance::calculateGregorianDateTimeUTC(ExecState* exec) const
78 {
79     double milli = internalNumber();
80     if (std::isnan(milli))
81         return 0;
82 
83     VM& vm = exec->vm();
84     if (!m_data)
85         m_data = vm.dateInstanceCache.add(milli);
86 
87     if (m_data->m_gregorianDateTimeUTCCachedForMS != milli) {
88         msToGregorianDateTime(vm, milli, WTF::UTCTime, m_data->m_cachedGregorianDateTimeUTC);
89         m_data->m_gregorianDateTimeUTCCachedForMS = milli;
90     }
91     return &m_data->m_cachedGregorianDateTimeUTC;
92 }
