33 void JITBitOrGenerator::generateFastPath(CCallHelpers& jit)
34 {
35     ASSERT(!m_leftOperand.isConstInt32() || !m_rightOperand.isConstInt32());
36 
37     m_didEmitFastPath = true;
38 
39     if (m_leftOperand.isConstInt32() || m_rightOperand.isConstInt32()) {
40         JSValueRegs var = m_leftOperand.isConstInt32() ? m_right : m_left;
41         SnippetOperand& constOpr = m_leftOperand.isConstInt32() ? m_leftOperand : m_rightOperand;
42 
43         // Try to do intVar | intConstant.
44         m_slowPathJumpList.append(jit.branchIfNotInt32(var));
45 
46         jit.moveValueRegs(var, m_result);
47         if (constOpr.asConstInt32()) {
48 #if USE(JSVALUE64)
49             jit.or32(CCallHelpers::Imm32(constOpr.asConstInt32()), m_result.payloadGPR());
50             jit.or64(GPRInfo::tagTypeNumberRegister, m_result.payloadGPR());
51 #else
52             jit.or32(CCallHelpers::Imm32(constOpr.asConstInt32()), m_result.payloadGPR());
53 #endif
54         }
55 
56     } else {
57         ASSERT(!m_leftOperand.isConstInt32() && !m_rightOperand.isConstInt32());
58 
59         // Try to do intVar | intVar.
60         m_slowPathJumpList.append(jit.branchIfNotInt32(m_left));
61         m_slowPathJumpList.append(jit.branchIfNotInt32(m_right));
62 
63         jit.moveValueRegs(m_left, m_result);
64 #if USE(JSVALUE64)
65         jit.or64(m_right.payloadGPR(), m_result.payloadGPR());
66 #else
67         jit.or32(m_right.payloadGPR(), m_result.payloadGPR());
68 #endif
69     }
70 }
