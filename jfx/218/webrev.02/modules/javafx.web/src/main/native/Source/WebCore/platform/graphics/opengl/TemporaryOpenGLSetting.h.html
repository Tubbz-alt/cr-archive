37 // TemporaryOpenGLSetting<> is useful for temporarily disabling (or enabling) a particular OpenGL
38 // feature with a particular scope. A TemporaryOpenGLSetting<> object returns the flag to its original
39 // value upon destruction, making it an alternative to checking, clearing, and resetting each flag
40 // at all of a block's exit points.
41 //
42 // Based on WTF::SetForScope<>
43 
44 class TemporaryOpenGLSetting {
45     WTF_MAKE_NONCOPYABLE(TemporaryOpenGLSetting);
46 public:
47     TemporaryOpenGLSetting(GCGLenum capability, GCGLenum scopedState);
48     ~TemporaryOpenGLSetting();
49 
50 private:
51     const GCGLenum m_capability;
52     const GCGLenum m_scopedState;
53     GCGLenum m_originalState;
54 };
55 
56 }
