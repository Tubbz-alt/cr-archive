34 namespace WebCore {
35 
36 WebGLDepthTexture::WebGLDepthTexture(WebGLRenderingContextBase& context)
37     : WebGLExtension(context)
38 {
39     context.graphicsContextGL()->getExtensions().ensureEnabled("GL_OES_depth_texture");
40     context.graphicsContextGL()->getExtensions().ensureEnabled("GL_ARB_depth_texture");
41 }
42 
43 WebGLDepthTexture::~WebGLDepthTexture() = default;
44 
45 WebGLExtension::ExtensionName WebGLDepthTexture::getName() const
46 {
47     return WebGLDepthTextureName;
48 }
49 
50 bool WebGLDepthTexture::supported(GraphicsContextGLOpenGL& context)
51 {
52     ExtensionsGL& extensions = context.getExtensions();
53     return extensions.supports("GL_OES_depth_texture")
54         || extensions.supports("GL_ARB_depth_texture");
55 }
