37 DesiredTransition::DesiredTransition(CodeBlock* codeBlock, CodeBlock* codeOriginOwner, Structure* oldStructure, Structure* newStructure)
38     : m_codeBlock(codeBlock)
39     , m_codeOriginOwner(codeOriginOwner)
40     , m_oldStructure(oldStructure)
41     , m_newStructure(newStructure)
42 {
43 }
44 
45 void DesiredTransition::reallyAdd(VM& vm, CommonData* common)
46 {
47     common->transitions.append(
48         WeakReferenceTransition(
49             vm, m_codeBlock,
50             m_codeOriginOwner,
51             m_oldStructure, m_newStructure));
52 }
53 
54 void DesiredTransition::visitChildren(SlotVisitor& visitor)
55 {
56     visitor.appendUnbarriered(m_codeOriginOwner);
57     visitor.appendUnbarriered(m_oldStructure);
58     visitor.appendUnbarriered(m_newStructure);
59 }
60 
61 DesiredTransitions::DesiredTransitions()
62 {
63 }
64 
65 DesiredTransitions::~DesiredTransitions()
66 {
67 }
68 
69 void DesiredTransitions::addLazily(CodeBlock* codeBlock, CodeBlock* codeOriginOwner, Structure* oldStructure, Structure* newStructure)
70 {
71     m_transitions.append(DesiredTransition(codeBlock, codeOriginOwner, oldStructure, newStructure));
72 }
73 
74 void DesiredTransitions::reallyAdd(VM& vm, CommonData* common)
75 {
76     for (unsigned i = 0; i < m_transitions.size(); i++)
77         m_transitions[i].reallyAdd(vm, common);
78 }
79 
80 void DesiredTransitions::visitChildren(SlotVisitor& visitor)
81 {
82     for (unsigned i = 0; i < m_transitions.size(); i++)
83         m_transitions[i].visitChildren(visitor);
84 }
