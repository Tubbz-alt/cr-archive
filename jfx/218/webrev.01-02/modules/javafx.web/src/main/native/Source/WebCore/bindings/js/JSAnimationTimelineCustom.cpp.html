39 JSValue toJSNewlyCreated(JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<AnimationTimeline>&& value)
40 {
41     if (value->isDocumentTimeline())
42         return createWrapper<DocumentTimeline>(globalObject, WTFMove(value));
43     return createWrapper<AnimationTimeline>(globalObject, WTFMove(value));
44 }
45 
46 JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, AnimationTimeline& value)
47 {
48     return wrap(lexicalGlobalObject, globalObject, value);
49 }
