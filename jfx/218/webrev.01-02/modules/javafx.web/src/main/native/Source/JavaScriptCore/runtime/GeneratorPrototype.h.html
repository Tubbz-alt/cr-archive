31 namespace JSC {

33 // %GeneratorPrototype% intrinsic.
34 // https://tc39.github.io/ecma262/#sec-properties-of-generator-prototype
35 class GeneratorPrototype final : public JSNonFinalObject {
36 public:
37     typedef JSNonFinalObject Base;
38     static constexpr unsigned StructureFlags = Base::StructureFlags | HasStaticPropertyTable;
39 
40     static GeneratorPrototype* create(VM& vm, JSGlobalObject*, Structure* structure)
41     {
42         GeneratorPrototype* prototype = new (NotNull, allocateCell<GeneratorPrototype>(vm.heap)) GeneratorPrototype(vm, structure);
43         prototype->finishCreation(vm);
44         return prototype;
45     }
46 
47     DECLARE_INFO;
48 
49     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
50     {
51         return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
52     }
53 
54 private:
55     GeneratorPrototype(VM& vm, Structure* structure)
56         : Base(vm, structure)
57     {
58     }
59     void finishCreation(VM&);
60 };
61 
62 } // namespace JSC
