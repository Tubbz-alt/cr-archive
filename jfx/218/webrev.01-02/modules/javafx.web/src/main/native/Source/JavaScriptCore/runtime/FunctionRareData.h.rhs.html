 44 class FunctionRareData final : public JSCell {
 45     friend class JIT;
 46     friend class DFG::SpeculativeJIT;
 47     friend class DFG::JITCompiler;
 48     friend class VM;
 49 
 50 public:
 51     typedef JSCell Base;
 52     static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
 53 
 54     static FunctionRareData* create(VM&, ExecutableBase*);
 55 
 56     static constexpr bool needsDestruction = true;
 57 
 58     template<typename CellType, SubspaceAccess mode>
 59     static IsoSubspace* subspaceFor(VM& vm)
 60     {
 61         return vm.functionRareDataSpace<mode>();
 62     }
 63 

 64     static void destroy(JSCell*);
 65 
 66     static Structure* createStructure(VM&, JSGlobalObject*, JSValue prototype);
 67 
 68     static void visitChildren(JSCell*, SlotVisitor&);
 69 
 70     DECLARE_INFO;
 71 
 72     static inline ptrdiff_t offsetOfObjectAllocationProfile() { return OBJECT_OFFSETOF(FunctionRareData, m_objectAllocationProfile); }
 73     static inline ptrdiff_t offsetOfAllocationProfileWatchpointSet() { return OBJECT_OFFSETOF(FunctionRareData, m_allocationProfileWatchpointSet); }
 74     static inline ptrdiff_t offsetOfInternalFunctionAllocationProfile() { return OBJECT_OFFSETOF(FunctionRareData, m_internalFunctionAllocationProfile); }
 75     static inline ptrdiff_t offsetOfBoundFunctionStructure() { return OBJECT_OFFSETOF(FunctionRareData, m_boundFunctionStructure); }
 76     static inline ptrdiff_t offsetOfExecutable() { return OBJECT_OFFSETOF(FunctionRareData, m_executable); }
 77     static inline ptrdiff_t offsetOfAllocationProfileClearingWatchpoint() { return OBJECT_OFFSETOF(FunctionRareData, m_allocationProfileClearingWatchpoint); }


 78 
 79     ObjectAllocationProfileWithPrototype* objectAllocationProfile()
 80     {
 81         return &m_objectAllocationProfile;
 82     }
 83 
 84     Structure* objectAllocationStructure() { return m_objectAllocationProfile.structure(); }
 85     JSObject* objectAllocationPrototype() { return m_objectAllocationProfile.prototype(); }
 86 
 87     InlineWatchpointSet& allocationProfileWatchpointSet()
 88     {
 89         return m_allocationProfileWatchpointSet;
 90     }
 91 
 92     void clear(const char* reason);
 93 
 94     void initializeObjectAllocationProfile(VM&, JSGlobalObject*, JSObject* prototype, size_t inlineCapacity, JSFunction* constructor);
 95 
 96     bool isObjectAllocationProfileInitialized() { return !m_objectAllocationProfile.isNull(); }
 97 
 98     Structure* internalFunctionAllocationStructure() { return m_internalFunctionAllocationProfile.structure(); }
 99     Structure* createInternalFunctionAllocationStructureFromBase(VM& vm, JSGlobalObject* baseGlobalObject, JSObject* prototype, Structure* baseStructure)
100     {
101         initializeAllocationProfileWatchpointSet();
102         return m_internalFunctionAllocationProfile.createAllocationStructureFromBase(vm, baseGlobalObject, this, prototype, baseStructure);
103     }
104     void clearInternalFunctionAllocationProfile(const char* reason)
105     {
106         m_internalFunctionAllocationProfile.clear();
107         m_allocationProfileWatchpointSet.fireAll(vm(), reason);
108     }
109 
110     void initializeAllocationProfileWatchpointSet()
111     {
112         if (m_allocationProfileWatchpointSet.isStillValid())
113             m_allocationProfileWatchpointSet.startWatching();
114     }
115 
116     Structure* getBoundFunctionStructure() { return m_boundFunctionStructure.get(); }
117     void setBoundFunctionStructure(VM& vm, Structure* structure) { m_boundFunctionStructure.set(vm, this, structure); }
118 
119     ExecutableBase* executable() const { return m_executable.get(); }
120 
121     bool hasReifiedLength() const { return m_hasReifiedLength; }
122     void setHasReifiedLength() { m_hasReifiedLength = true; }
123     bool hasReifiedName() const { return m_hasReifiedName; }
124     void setHasReifiedName() { m_hasReifiedName = true; }
125 
126     bool hasModifiedLength() const { return m_hasModifiedLength; }
127     void setHasModifiedLength()
128     {
129         m_hasModifiedLength = true;
130     }
131     bool hasModifiedName() const { return m_hasModifiedName; }
132     void setHasModifiedName()
133     {
134         m_hasModifiedName = true;
135     }
136 
137     bool hasAllocationProfileClearingWatchpoint() const { return !!m_allocationProfileClearingWatchpoint; }
138     Watchpoint* createAllocationProfileClearingWatchpoint();
139     class AllocationProfileClearingWatchpoint;
140 
141 protected:
142     explicit FunctionRareData(VM&, ExecutableBase*);
143     ~FunctionRareData();
144 
145 private:
146     friend class LLIntOffsetsExtractor;
147 
148     // Ideally, there would only be one allocation profile for subclassing but due to Reflect.construct we
149     // have two. There are some pros and cons in comparison to our current system to using the same profile
150     // for both JS constructors and subclasses of builtin constructors:
151     //
152     // 1) + Uses less memory.
153     // 2) + Conceptually simplier as there is only one profile.
154     // 3) - We would need a check in all JSFunction object creations (both with classes and without) that the
155     //      new.target's profiled structure has a JSFinalObject ClassInfo. This is needed, for example, if we have
156     //      `Reflect.construct(Array, args, myConstructor)` since myConstructor will be the new.target of Array
157     //      the Array constructor will set the allocation profile of myConstructor to hold an Array structure
158     //
159     // We don't really care about 1) since this memory is rare and small in total. 2) is unfortunate but is
160     // probably outweighed by the cost of 3).
161     ObjectAllocationProfileWithPrototype m_objectAllocationProfile;
162     InlineWatchpointSet m_allocationProfileWatchpointSet;
163     InternalFunctionAllocationProfile m_internalFunctionAllocationProfile;
164     WriteBarrier<Structure> m_boundFunctionStructure;
165     WriteBarrier<ExecutableBase> m_executable;
166     std::unique_ptr<AllocationProfileClearingWatchpoint> m_allocationProfileClearingWatchpoint;
167     bool m_hasReifiedLength : 1;
168     bool m_hasReifiedName : 1;
169     bool m_hasModifiedLength : 1;
170     bool m_hasModifiedName : 1;
171 };
172 
173 class FunctionRareData::AllocationProfileClearingWatchpoint final : public Watchpoint {
174 public:
175     AllocationProfileClearingWatchpoint(FunctionRareData* rareData)
176         : Watchpoint(Watchpoint::Type::FunctionRareDataAllocationProfileClearing)
177         , m_rareData(rareData)
178     { }
179 
180     void fireInternal(VM&, const FireDetail&);
181 
182 private:
183     // Own destructor may not be called. Keep members trivially destructible.
184     JSC_WATCHPOINT_FIELD(PackedCellPtr<FunctionRareData>, m_rareData);
185 };
186 
187 inline Watchpoint* FunctionRareData::createAllocationProfileClearingWatchpoint()
188 {
189     RELEASE_ASSERT(!hasAllocationProfileClearingWatchpoint());
190     m_allocationProfileClearingWatchpoint = makeUnique<AllocationProfileClearingWatchpoint>(this);
191     return m_allocationProfileClearingWatchpoint.get();
192 }
