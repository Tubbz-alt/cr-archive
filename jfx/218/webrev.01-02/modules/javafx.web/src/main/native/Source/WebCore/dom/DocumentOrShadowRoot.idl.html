27 // https://dom.spec.whatwg.org/#documentorshadowroot
28 [
29     NoInterfaceObject,
30 ] interface DocumentOrShadowRoot {
31     // Extensions from Shadow DOM API (https://w3c.github.io/webcomponents/spec/shadow/#extensions-to-the-documentorshadowroot-mixin).
32     // DOMSelection? getSelection(); // FIXME: We currently have this on Document only.
33     Element? elementFromPoint(double x, double y);
34     sequence<Element> elementsFromPoint(double x, double y);
35     // CaretPosition? caretPositionFromPoint(double x, double y); // FIXME: Implement this.
36     readonly attribute Element? activeElement;
37 
38     // Extensions from the CSSOM specification (https://drafts.csswg.org/cssom/#extensions-to-the-document-interface).
39     readonly attribute StyleSheetList styleSheets; // FIXME: Should be [SameObject].
40 
41     // Extensions from Pointer Lock API (https://w3c.github.io/pointerlock/#extensions-to-the-documentorshadowroot-mixin).
42     [Conditional=POINTER_LOCK] readonly attribute Element? pointerLockElement;
43 
44     // Extensions from Picture-in-Picture API (https://wicg.github.io/picture-in-picture/#documentorshadowroot-extension)
45     [Conditional=PICTURE_IN_PICTURE_API, EnabledBySetting=PictureInPictureAPI] readonly attribute Element? pictureInPictureElement;
46 };
