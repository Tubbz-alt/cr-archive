44 void EventLoop::cycle()
45 {
46 #if OS(WINDOWS)
47     MSG msg;
48     if (!GetMessage(&msg, 0, 0, 0)) {
49         m_ended = true;
50         return;
51     }
52     TranslateMessage(&msg);
53     DispatchMessage(&msg);
54 #elif PLATFORM(WATCHOS)
55     // FIXME: <rdar://problem/25972777>. In order for auto-attach to work, we need to
56     // run in the default run loop mode otherwise we do not receive the XPC messages
57     // necessary to setup the relay connection and negotiate an auto-attach debugger.
58     CFTimeInterval timeInterval = 0.05;
59     CFRunLoopRunInMode(kCFRunLoopDefaultMode, timeInterval, true);
60 #elif USE(CF)
61     // Run the RunLoop in a custom run loop mode to prevent default observers
62     // to run and potentially evaluate JavaScript in this context while we are
63     // nested. Only the debugger should control things until we continue.
64     // FIXME: This is not a perfect solution, as background threads are not
65     // paused and can still access and evalute script in the JSContext.
66     CFTimeInterval timeInterval = 0.05;
67     CFRunLoopRunInMode(remoteInspectorRunLoopMode(), timeInterval, true);
68 #elif USE(GLIB)
69     g_main_context_iteration(NULL, FALSE);
70 #endif
71 }
