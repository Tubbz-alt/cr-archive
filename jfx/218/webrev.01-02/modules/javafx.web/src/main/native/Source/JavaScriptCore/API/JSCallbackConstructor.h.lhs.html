26 #ifndef JSCallbackConstructor_h
27 #define JSCallbackConstructor_h
28 
29 #include "JSDestructibleObject.h"
30 #include "JSObjectRef.h"
31 
32 namespace JSC {
33 
34 class JSCallbackConstructor final : public JSDestructibleObject {
35 public:
36     typedef JSDestructibleObject Base;
37     static const unsigned StructureFlags = Base::StructureFlags | ImplementsHasInstance | ImplementsDefaultHasInstance;

38 
39     static JSCallbackConstructor* create(ExecState* exec, JSGlobalObject* globalObject, Structure* structure, JSClassRef classRef, JSObjectCallAsConstructorCallback callback)

40     {
41         VM& vm = exec->vm();





42         JSCallbackConstructor* constructor = new (NotNull, allocateCell<JSCallbackConstructor>(vm.heap)) JSCallbackConstructor(globalObject, structure, classRef, callback);
43         constructor->finishCreation(globalObject, classRef);
44         return constructor;
45     }
46 
47     ~JSCallbackConstructor();
48     static void destroy(JSCell*);
49     JSClassRef classRef() const { return m_class; }
50     JSObjectCallAsConstructorCallback callback() const { return m_callback; }
51     DECLARE_INFO;
52 
53     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue proto)
54     {
55         return Structure::create(vm, globalObject, proto, TypeInfo(ObjectType, StructureFlags), info());
56     }
57 
58 protected:
59     JSCallbackConstructor(JSGlobalObject*, Structure*, JSClassRef, JSObjectCallAsConstructorCallback);
60     void finishCreation(JSGlobalObject*, JSClassRef);
61 
62 private:
63     friend struct APICallbackFunction;
64 
65     static ConstructType getConstructData(JSCell*, ConstructData&);
66 
67     JSObjectCallAsConstructorCallback constructCallback() { return m_callback; }
68 
69     JSClassRef m_class;
70     JSObjectCallAsConstructorCallback m_callback;
71 };
72 
73 } // namespace JSC
74 
75 #endif // JSCallbackConstructor_h
