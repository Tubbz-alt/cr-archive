26 #pragma once
27 
28 #include "JSGlobalObject.h"
29 
30 OBJC_CLASS JSScript;
31 
32 namespace JSC {
33 
34 class JSAPIGlobalObject final : public JSGlobalObject {
35 public:
36     using Base = JSGlobalObject;
37 
38     DECLARE_EXPORT_INFO;
39     static const GlobalObjectMethodTable s_globalObjectMethodTable;
40 
41     static constexpr bool needsDestruction = true;
42     template<typename CellType, SubspaceAccess mode>
43     static IsoSubspace* subspaceFor(VM& vm)
44     {
45         return vm.apiGlobalObjectSpace<mode>();
46     }
47 
48     static JSAPIGlobalObject* create(VM& vm, Structure* structure)
49     {
50         auto* object = new (NotNull, allocateCell<JSAPIGlobalObject>(vm.heap)) JSAPIGlobalObject(vm, structure);
51         object->finishCreation(vm);
52         return object;
53     }
54 
55     static Structure* createStructure(VM& vm, JSValue prototype)
56     {
57         auto* result = Structure::create(vm, 0, prototype, TypeInfo(GlobalObjectType, StructureFlags), info());
58         result->setTransitionWatchpointIsLikelyToBeFired(true);
59         return result;
60     }
61 
62     static JSInternalPromise* moduleLoaderImportModule(JSGlobalObject*, JSModuleLoader*, JSString* moduleNameValue, JSValue parameters, const SourceOrigin&);
63     static Identifier moduleLoaderResolve(JSGlobalObject*, JSModuleLoader*, JSValue keyValue, JSValue referrerValue, JSValue);
64     static JSInternalPromise* moduleLoaderFetch(JSGlobalObject*, JSModuleLoader*, JSValue, JSValue, JSValue);
65     static JSObject* moduleLoaderCreateImportMetaProperties(JSGlobalObject*, JSModuleLoader*, JSValue, JSModuleRecord*, JSValue);
66     static JSValue moduleLoaderEvaluate(JSGlobalObject*, JSModuleLoader*, JSValue, JSValue, JSValue);
67 
68     JSValue loadAndEvaluateJSScriptModule(const JSLockHolder&, JSScript *);
69 
70 private:
71     JSAPIGlobalObject(VM& vm, Structure* structure)
72         : Base(vm, structure, &s_globalObjectMethodTable)
73     { }
74 };
75 
76 }
