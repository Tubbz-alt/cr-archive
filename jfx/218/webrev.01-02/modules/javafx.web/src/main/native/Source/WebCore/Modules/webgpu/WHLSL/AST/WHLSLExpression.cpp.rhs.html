 39 void Expression::destroy(Expression& expression)
 40 {
 41     switch (expression.kind()) {
 42     case Expression::Kind::Assignment:
 43         delete &downcast<AssignmentExpression>(expression);
 44         break;
 45     case Expression::Kind::BooleanLiteral:
 46         delete &downcast<BooleanLiteral>(expression);
 47         break;
 48     case Expression::Kind::Call:
 49         delete &downcast<CallExpression>(expression);
 50         break;
 51     case Expression::Kind::Comma:
 52         delete &downcast<CommaExpression>(expression);
 53         break;
 54     case Expression::Kind::Dereference:
 55         delete &downcast<DereferenceExpression>(expression);
 56         break;
 57     case Expression::Kind::FloatLiteral:
 58         delete &downcast<FloatLiteral>(expression);
 59         break;
 60     case Expression::Kind::IntegerLiteral:
 61         delete &downcast<IntegerLiteral>(expression);
 62         break;
 63     case Expression::Kind::Logical:
 64         delete &downcast<LogicalExpression>(expression);
 65         break;
 66     case Expression::Kind::LogicalNot:
 67         delete &downcast<LogicalNotExpression>(expression);
 68         break;
 69     case Expression::Kind::MakeArrayReference:
 70         delete &downcast<MakeArrayReferenceExpression>(expression);
 71         break;
 72     case Expression::Kind::MakePointer:
 73         delete &downcast<MakePointerExpression>(expression);
 74         break;



 75     case Expression::Kind::Dot:
 76         delete &downcast<DotExpression>(expression);
 77         break;
 78     case Expression::Kind::GlobalVariableReference:
 79         delete &downcast<GlobalVariableReference>(expression);
 80         break;
 81     case Expression::Kind::Index:
 82         delete &downcast<IndexExpression>(expression);
 83         break;
 84     case Expression::Kind::ReadModifyWrite:
 85         delete &downcast<ReadModifyWriteExpression>(expression);
 86         break;
 87     case Expression::Kind::Ternary:
 88         delete &downcast<TernaryExpression>(expression);
 89         break;
 90     case Expression::Kind::UnsignedIntegerLiteral:
 91         delete &downcast<UnsignedIntegerLiteral>(expression);
 92         break;
 93     case Expression::Kind::EnumerationMemberLiteral:
 94         delete &downcast<EnumerationMemberLiteral>(expression);
 95         break;
 96     case Expression::Kind::VariableReference:
 97         delete &downcast<VariableReference>(expression);
 98         break;
 99     }
100 }

102 void Expression::destruct(Expression& expression)
103 {
104     switch (expression.kind()) {
105     case Expression::Kind::Assignment:
106         downcast<AssignmentExpression>(expression).~AssignmentExpression();
107         break;
108     case Expression::Kind::BooleanLiteral:
109         downcast<BooleanLiteral>(expression).~BooleanLiteral();
110         break;
111     case Expression::Kind::Call:
112         downcast<CallExpression>(expression).~CallExpression();
113         break;
114     case Expression::Kind::Comma:
115         downcast<CommaExpression>(expression).~CommaExpression();
116         break;
117     case Expression::Kind::Dereference:
118         downcast<DereferenceExpression>(expression).~DereferenceExpression();
119         break;
120     case Expression::Kind::FloatLiteral:
121         downcast<FloatLiteral>(expression).~FloatLiteral();
122         break;
123     case Expression::Kind::IntegerLiteral:
124         downcast<IntegerLiteral>(expression).~IntegerLiteral();
125         break;
126     case Expression::Kind::Logical:
127         downcast<LogicalExpression>(expression).~LogicalExpression();
128         break;
129     case Expression::Kind::LogicalNot:
130         downcast<LogicalNotExpression>(expression).~LogicalNotExpression();
131         break;
132     case Expression::Kind::MakeArrayReference:
133         downcast<MakeArrayReferenceExpression>(expression).~MakeArrayReferenceExpression();
134         break;
135     case Expression::Kind::MakePointer:
136         downcast<MakePointerExpression>(expression).~MakePointerExpression();
137         break;



138     case Expression::Kind::Dot:
139         downcast<DotExpression>(expression).~DotExpression();
140         break;
141     case Expression::Kind::GlobalVariableReference:
142         downcast<GlobalVariableReference>(expression).~GlobalVariableReference();
143         break;
144     case Expression::Kind::Index:
145         downcast<IndexExpression>(expression).~IndexExpression();
146         break;
147     case Expression::Kind::ReadModifyWrite:
148         downcast<ReadModifyWriteExpression>(expression).~ReadModifyWriteExpression();
149         break;
150     case Expression::Kind::Ternary:
151         downcast<TernaryExpression>(expression).~TernaryExpression();
152         break;
153     case Expression::Kind::UnsignedIntegerLiteral:
154         downcast<UnsignedIntegerLiteral>(expression).~UnsignedIntegerLiteral();
155         break;
156     case Expression::Kind::EnumerationMemberLiteral:
157         downcast<EnumerationMemberLiteral>(expression).~EnumerationMemberLiteral();
158         break;
159     case Expression::Kind::VariableReference:
160         downcast<VariableReference>(expression).~VariableReference();
161         break;
162     }
163 }

165 bool Expression::mayBeEffectful() const
166 {
167     auto& expression = const_cast<Expression&>(*this);





168 
169     switch (expression.kind()) {
170     case Expression::Kind::BooleanLiteral:
171     case Expression::Kind::FloatLiteral:
172     case Expression::Kind::IntegerLiteral:
173     case Expression::Kind::UnsignedIntegerLiteral:
174     case Expression::Kind::EnumerationMemberLiteral:
175     case Expression::Kind::GlobalVariableReference:
176     case Expression::Kind::VariableReference:
177         return false;
178 
179     case Expression::Kind::Dereference:
180         return downcast<DereferenceExpression>(expression).pointer().mayBeEffectful();
181 
182     case Expression::Kind::Logical:
183         return downcast<LogicalExpression>(expression).left().mayBeEffectful() || downcast<LogicalExpression>(expression).right().mayBeEffectful();
184 
185     case Expression::Kind::LogicalNot:
186         return downcast<LogicalNotExpression>(expression).operand().mayBeEffectful();
187 
188     case Expression::Kind::MakeArrayReference:
189         return downcast<MakeArrayReferenceExpression>(expression).leftValue().mayBeEffectful();
190 
191     case Expression::Kind::MakePointer:
192         return downcast<MakePointerExpression>(expression).leftValue().mayBeEffectful();
193 
194     case Expression::Kind::Dot:
195         return downcast<DotExpression>(expression).base().mayBeEffectful();
196 
197     case Expression::Kind::Index:
198         return downcast<IndexExpression>(expression).base().mayBeEffectful() || downcast<IndexExpression>(expression).indexExpression().mayBeEffectful();
199 
200     default:
201         break;
202     }
203 
204     return true;
205 }
