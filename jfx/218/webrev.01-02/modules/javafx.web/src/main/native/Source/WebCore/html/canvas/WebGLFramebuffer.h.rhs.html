 26 #pragma once
 27 
 28 #if ENABLE(WEBGL)
 29 
 30 #include "WebGLContextObject.h"
 31 #include "WebGLSharedObject.h"
 32 #include <wtf/HashMap.h>
 33 #include <wtf/RefCounted.h>
 34 #include <wtf/Vector.h>
 35 
 36 namespace WebCore {
 37 
 38 class WebGLRenderbuffer;
 39 class WebGLTexture;
 40 
 41 class WebGLFramebuffer final : public WebGLContextObject {
 42 public:
 43     class WebGLAttachment : public RefCounted<WebGLAttachment> {
 44     public:
 45         virtual ~WebGLAttachment();
 46 
 47         virtual GCGLsizei getWidth() const = 0;
 48         virtual GCGLsizei getHeight() const = 0;
 49         virtual GCGLenum getFormat() const = 0;
 50         virtual WebGLSharedObject* getObject() const = 0;
 51         virtual bool isSharedObject(WebGLSharedObject*) const = 0;
 52         virtual bool isValid() const = 0;
 53         virtual bool isInitialized() const = 0;
 54         virtual void setInitialized() = 0;
 55         virtual void onDetached(GraphicsContextGLOpenGL*) = 0;
 56         virtual void attach(GraphicsContextGLOpenGL*, GCGLenum attachment) = 0;
 57         virtual void unattach(GraphicsContextGLOpenGL*, GCGLenum attachment) = 0;
 58 
 59     protected:
 60         WebGLAttachment();
 61     };
 62 
 63     virtual ~WebGLFramebuffer();
 64 
 65     static Ref<WebGLFramebuffer> create(WebGLRenderingContextBase&);
 66 
 67     void setAttachmentForBoundFramebuffer(GCGLenum attachment, GCGLenum texTarget, WebGLTexture*, GCGLint level);
 68     void setAttachmentForBoundFramebuffer(GCGLenum attachment, WebGLRenderbuffer*);
 69     // If an object is attached to the currently bound framebuffer, remove it.
 70     void removeAttachmentFromBoundFramebuffer(WebGLSharedObject*);
 71     // If a given attachment point for the currently bound framebuffer is not null, remove the attached object.
 72     void removeAttachmentFromBoundFramebuffer(GCGLenum);
 73     WebGLSharedObject* getAttachmentObject(GCGLenum) const;
 74 
 75     GCGLenum getColorBufferFormat() const;
 76     GCGLsizei getColorBufferWidth() const;
 77     GCGLsizei getColorBufferHeight() const;
 78 
 79     // This should always be called before drawArray, drawElements, clear,
 80     // readPixels, copyTexImage2D, copyTexSubImage2D if this framebuffer is
 81     // currently bound.
 82     // Return false if the framebuffer is incomplete; otherwise initialize
 83     // the buffers if they haven't been initialized and
 84     // needToInitializeAttachments is true.
 85     bool onAccess(GraphicsContextGLOpenGL*, const char** reason);
 86 
 87     // Software version of glCheckFramebufferStatus(), except that when
 88     // FRAMEBUFFER_COMPLETE is returned, it is still possible for
 89     // glCheckFramebufferStatus() to return FRAMEBUFFER_UNSUPPORTED,
 90     // depending on hardware implementation.
 91     GCGLenum checkStatus(const char** reason) const;
 92 
 93     bool hasEverBeenBound() const { return object() && m_hasEverBeenBound; }
 94 
 95     void setHasEverBeenBound() { m_hasEverBeenBound = true; }
 96 
 97     bool hasStencilBuffer() const;
 98 
 99     // Wrapper for drawBuffersEXT/drawBuffersARB to work around a driver bug.
100     void drawBuffers(const Vector<GCGLenum>& bufs);
101 
102     GCGLenum getDrawBuffer(GCGLenum);
103 
104 protected:
105     WebGLFramebuffer(WebGLRenderingContextBase&);
106 
107     void deleteObjectImpl(GraphicsContextGLOpenGL*, PlatformGLObject) override;
108 
109 private:
110     WebGLAttachment* getAttachment(GCGLenum) const;
111 
112     // Return false if framebuffer is incomplete.
113     bool initializeAttachments(GraphicsContextGLOpenGL*, const char** reason);
114 
115     // Check if the framebuffer is currently bound.
116     bool isBound() const;
117 
118     // attach 'attachment' at 'attachmentPoint'.
119     void attach(GCGLenum attachment, GCGLenum attachmentPoint);
120 
121     // Check if a new drawBuffers call should be issued. This is called when we add or remove an attachment.
122     void drawBuffersIfNecessary(bool force);
123 
124     typedef WTF::HashMap<GCGLenum, RefPtr<WebGLAttachment>> AttachmentMap;
125 
126     AttachmentMap m_attachments;
127 
128     bool m_hasEverBeenBound;
129 
130     Vector<GCGLenum> m_drawBuffers;
131     Vector<GCGLenum> m_filteredDrawBuffers;
132 };
133 
134 } // namespace WebCore
135 
136 #endif
