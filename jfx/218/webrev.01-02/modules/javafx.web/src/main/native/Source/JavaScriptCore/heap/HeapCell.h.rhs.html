 40 #if COMPILER(GCC_COMPATIBLE)
 41 ALWAYS_INLINE void keepAlive(const void* pointer)
 42 {
 43     asm volatile ("" : : "r"(pointer) : "memory");
 44 }
 45 #else
 46 JS_EXPORT_PRIVATE void keepAlive(const void*);
 47 #endif
 48 
 49 class HeapCell {
 50 public:
 51     enum Kind : int8_t {
 52         JSCell,
 53         JSCellWithInteriorPointers,
 54         Auxiliary
 55     };
 56 
 57     HeapCell() { }
 58 
 59     // We're intentionally only zapping the bits for the structureID and leaving
 60     // the rest of the cell header bits intact for crash analysis uses.
 61     enum ZapReason : int8_t { Unspecified, Destruction, StopAllocating };
 62     void zap(ZapReason reason)
 63     {
 64         uint32_t* cellWords = bitwise_cast<uint32_t*>(this);
 65         cellWords[0] = 0;
 66         // Leaving cellWords[1] alone for crash analysis if needed.
 67         cellWords[2] = reason;
 68     }
 69     bool isZapped() const { return !*bitwise_cast<const uint32_t*>(this); }
 70 
 71     bool isLive();
 72 
 73     bool isPreciseAllocation() const;
 74     CellContainer cellContainer() const;
 75     MarkedBlock& markedBlock() const;
 76     PreciseAllocation& preciseAllocation() const;
 77 
 78     // If you want performance and you know that your cell is small, you can do this instead:
 79     // ASSERT(!cell->isPreciseAllocation());
 80     // cell->markedBlock().vm()
 81     // We currently only use this hack for callees to make CallFrame::vm() fast. It's not
 82     // recommended to use it for too many other things, since the large allocation cutoff is
 83     // a runtime option and its default value is small (400 bytes).
 84     Heap* heap() const;
 85     VM& vm() const;
 86 
 87     size_t cellSize() const;
 88     CellAttributes cellAttributes() const;
 89     DestructionMode destructionMode() const;
 90     Kind cellKind() const;
 91     Subspace* subspace() const;
 92 
 93     // Call use() after the last point where you need `this` pointer to be kept alive. You usually don't
 94     // need to use this, but it might be necessary if you're otherwise referring to an object's innards
 95     // but not the object itself.
 96     ALWAYS_INLINE void use() const

 97     {
 98         keepAlive(this);
 99     }



100 };
101 
102 inline bool isJSCellKind(HeapCell::Kind kind)
103 {
104     return kind == HeapCell::JSCell || kind == HeapCell::JSCellWithInteriorPointers;
105 }
106 
107 inline bool hasInteriorPointers(HeapCell::Kind kind)
108 {
109     return kind == HeapCell::Auxiliary || kind == HeapCell::JSCellWithInteriorPointers;
110 }
