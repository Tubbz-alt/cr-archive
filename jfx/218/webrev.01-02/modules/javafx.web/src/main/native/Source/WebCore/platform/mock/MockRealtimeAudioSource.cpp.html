 48 CaptureSourceOrError MockRealtimeAudioSource::create(String&& deviceID, String&& name, String&& hashSalt, const MediaConstraints* constraints)
 49 {
 50 #ifndef NDEBUG
 51     auto device = MockRealtimeMediaSourceCenter::mockDeviceWithPersistentID(deviceID);
 52     ASSERT(device);
 53     if (!device)
 54         return { "No mock microphone device"_s };
 55 #endif
 56 
 57     auto source = adoptRef(*new MockRealtimeAudioSource(WTFMove(deviceID), WTFMove(name), WTFMove(hashSalt)));
 58     if (constraints && source->applyConstraints(*constraints))
 59         return { };
 60 
 61     return CaptureSourceOrError(WTFMove(source));
 62 }

 65 MockRealtimeAudioSource::MockRealtimeAudioSource(String&& deviceID, String&& name, String&& hashSalt)
 66     : RealtimeMediaSource(RealtimeMediaSource::Type::Audio, WTFMove(name), WTFMove(deviceID), WTFMove(hashSalt))
 67     , m_workQueue(WorkQueue::create("MockRealtimeAudioSource Render Queue"))
 68     , m_timer(RunLoop::current(), this, &MockRealtimeAudioSource::tick)
 69 {
 70     auto device = MockRealtimeMediaSourceCenter::mockDeviceWithPersistentID(persistentID());
 71     ASSERT(device);
 72     m_device = *device;
 73 
 74     setSampleRate(WTF::get<MockMicrophoneProperties>(m_device.properties).defaultSampleRate);
 75     initializeEchoCancellation(true);
 76 }

 78 MockRealtimeAudioSource::~MockRealtimeAudioSource()
 79 {
 80 #if PLATFORM(IOS_FAMILY)
 81     RealtimeMediaSourceCenter::singleton().audioCaptureFactory().unsetActiveSource(*this);
 82 #endif
 83 }

 85 const RealtimeMediaSourceSettings& MockRealtimeAudioSource::settings()
 86 {
 87     if (!m_currentSettings) {
 88         RealtimeMediaSourceSettings settings;
 89         settings.setDeviceId(hashedId());
 90         settings.setVolume(volume());
 91         settings.setEchoCancellation(echoCancellation());
 92         settings.setSampleRate(sampleRate());
 93 
 94         RealtimeMediaSourceSupportedConstraints supportedConstraints;
 95         supportedConstraints.setSupportsDeviceId(true);
 96         supportedConstraints.setSupportsVolume(true);
 97         supportedConstraints.setSupportsEchoCancellation(true);
 98         supportedConstraints.setSupportsSampleRate(true);
 99         settings.setSupportedConstraints(supportedConstraints);
100 
101         m_currentSettings = WTFMove(settings);
102     }
103     return m_currentSettings.value();
104 }
105 
106 void MockRealtimeAudioSource::setChannelCount(unsigned channelCount)
107 {
108     if (channelCount > 2)
109         return;
110 
111     m_channelCount = channelCount;
112     settingsDidChange(RealtimeMediaSourceSettings::Flag::SampleRate);
113 }
114 
115 const RealtimeMediaSourceCapabilities& MockRealtimeAudioSource::capabilities()
116 {
117     if (!m_capabilities) {
118         RealtimeMediaSourceCapabilities capabilities(settings().supportedConstraints());
119 
120         capabilities.setDeviceId(hashedId());
121         capabilities.setVolume(CapabilityValueOrRange(0.0, 1.0));
122         capabilities.setEchoCancellation(RealtimeMediaSourceCapabilities::EchoCancellation::ReadWrite);
123         capabilities.setSampleRate(CapabilityValueOrRange(44100, 48000));
124 
125         m_capabilities = WTFMove(capabilities);
126     }
127     return m_capabilities.value();
128 }
129 
130 void MockRealtimeAudioSource::settingsDidChange(OptionSet<RealtimeMediaSourceSettings::Flag>)
131 {
132     m_currentSettings = WTF::nullopt;
133 }

135 void MockRealtimeAudioSource::startProducingData()
136 {
137 #if PLATFORM(IOS_FAMILY)
138     RealtimeMediaSourceCenter::singleton().audioCaptureFactory().setActiveSource(*this);
139     PlatformMediaSessionManager::sharedManager().sessionCanProduceAudioChanged();
140     ASSERT(AudioSession::sharedSession().category() == AudioSession::PlayAndRecord);
141 #endif
142 
143     if (!sampleRate())
144         setSampleRate(WTF::get<MockMicrophoneProperties>(m_device.properties).defaultSampleRate);
145 
146     m_startTime = MonotonicTime::now();
147     m_timer.startRepeating(renderInterval());
148 }
149 
150 void MockRealtimeAudioSource::stopProducingData()
151 {
152     m_timer.stop();
153     m_startTime = MonotonicTime::nan();
154 }
155 
156 void MockRealtimeAudioSource::tick()
157 {
158     if (std::isnan(m_lastRenderTime))
159         m_lastRenderTime = MonotonicTime::now();
160 
161     MonotonicTime now = MonotonicTime::now();
162 
163     if (m_delayUntil) {
164         if (m_delayUntil < now)
165             return;
166         m_delayUntil = MonotonicTime();
167     }
168 
169     Seconds delta = now - m_lastRenderTime;
170     m_lastRenderTime = now;
171 
172     m_workQueue->dispatch([this, delta, protectedThis = makeRef(*this)] {
173         render(delta);
174     });
175 }
176 
177 void MockRealtimeAudioSource::delaySamples(Seconds delta)
178 {
179     m_delayUntil = MonotonicTime::now() + delta;
180 }
