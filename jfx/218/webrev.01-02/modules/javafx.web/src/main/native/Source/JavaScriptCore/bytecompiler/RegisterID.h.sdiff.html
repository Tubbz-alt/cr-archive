 36 namespace JSC {
 37 
 38     class RegisterID {
 39         WTF_MAKE_NONCOPYABLE(RegisterID);
 40 
 41         friend class VirtualRegister;
 42     public:
 43         RegisterID()
 44             : m_refCount(0)
 45             , m_isTemporary(false)
 46 #ifndef NDEBUG
 47             , m_didSetIndex(false)
 48 #endif
 49         {
 50         }
 51 
 52         RegisterID(VirtualRegister virtualRegister)
 53             : m_refCount(0)
 54             , m_virtualRegister(virtualRegister)
 55             , m_isTemporary(false)
 56 #ifndef NDEBUG
 57             , m_didSetIndex(true)
 58 #endif
 59         {
 60         }
 61 
 62         explicit RegisterID(int index)
 63             : m_refCount(0)
 64             , m_virtualRegister(VirtualRegister(index))
 65             , m_isTemporary(false)
 66 #ifndef NDEBUG
 67             , m_didSetIndex(true)
 68 #endif
 69         {
 70         }
 71 
 72         void setIndex(int index)
 73         {
 74 #ifndef NDEBUG
 75             m_didSetIndex = true;
 76 #endif
 77             m_virtualRegister = VirtualRegister(index);
 78         }
 79 
 80         void setTemporary()
 81         {
 82             m_isTemporary = true;
 83         }
 84 
 85         int index() const
 86         {
 87             ASSERT(m_didSetIndex);
 88             return m_virtualRegister.offset();
 89         }
 90 
 91         VirtualRegister virtualRegister() const
 92         {
 93             ASSERT(m_virtualRegister.isValid());
 94             return m_virtualRegister;
 95         }
 96 
 97         bool isTemporary()

103         {
104             ++m_refCount;
105         }
106 
107         void deref()
108         {
109             --m_refCount;
110             ASSERT(m_refCount >= 0);
111         }
112 
113         int refCount() const
114         {
115             return m_refCount;
116         }
117 
118     private:
119 
120         int m_refCount;
121         VirtualRegister m_virtualRegister;
122         bool m_isTemporary;
123 #ifndef NDEBUG
124         bool m_didSetIndex;
125 #endif
126     };
127 } // namespace JSC
