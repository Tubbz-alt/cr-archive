26 #pragma once
27 
28 #include <wtf/RefCounted.h>
29 #include <wtf/Vector.h>
30 #include <wtf/text/WTFString.h>
31 
32 namespace WebCore {
33 class DOMSetAdapter;
34 
35 class InternalsSetLike : public RefCounted<InternalsSetLike> {
36 public:
37     static Ref<InternalsSetLike> create() { return adoptRef(*new InternalsSetLike); }
38 
39     void clearFromSetLike() { m_items.clear(); }
40     bool addToSetLike(const String&);
41     bool removeFromSetLike(const String& item) { return m_items.removeFirst(item); }
42     void initializeSetLike(DOMSetAdapter&);
43 
44     const Vector<String>& items() const { return m_items; }
45 
46 private:
47     InternalsSetLike();
48     Vector<String> m_items;
49 };
50 
51 inline bool InternalsSetLike::addToSetLike(const String& value)
52 {
53     bool hasValue = removeFromSetLike(value);
54     m_items.append(value);
55     return !hasValue;
56 }
57 
58 } // namespace WebCore
