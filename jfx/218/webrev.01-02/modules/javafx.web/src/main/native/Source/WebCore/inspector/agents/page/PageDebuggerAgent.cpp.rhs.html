 93 String PageDebuggerAgent::sourceMapURLForScript(const Script& script)
 94 {
 95     static NeverDestroyed<String> sourceMapHTTPHeader(MAKE_STATIC_STRING_IMPL("SourceMap"));
 96     static NeverDestroyed<String> sourceMapHTTPHeaderDeprecated(MAKE_STATIC_STRING_IMPL("X-SourceMap"));
 97 
 98     if (!script.url.isEmpty()) {
 99         CachedResource* resource = InspectorPageAgent::cachedResource(&m_inspectedPage.mainFrame(), URL({ }, script.url));
100         if (resource) {
101             String sourceMapHeader = resource->response().httpHeaderField(sourceMapHTTPHeader);
102             if (!sourceMapHeader.isEmpty())
103                 return sourceMapHeader;
104 
105             sourceMapHeader = resource->response().httpHeaderField(sourceMapHTTPHeaderDeprecated);
106             if (!sourceMapHeader.isEmpty())
107                 return sourceMapHeader;
108         }
109     }
110 
111     return InspectorDebuggerAgent::sourceMapURLForScript(script);
112 }

129 InjectedScript PageDebuggerAgent::injectedScriptForEval(ErrorString& errorString, const int* executionContextId)
130 {
131     if (!executionContextId) {
132         JSC::JSGlobalObject* scriptState = mainWorldExecState(&m_inspectedPage.mainFrame());
133         return injectedScriptManager().injectedScriptFor(scriptState);
134     }
135 
136     InjectedScript injectedScript = injectedScriptManager().injectedScriptForId(*executionContextId);
137     if (injectedScript.hasNoValue())
138         errorString = "Missing injected script for given executionContextId."_s;
139 
140     return injectedScript;
141 }

143 void PageDebuggerAgent::didClearWindowObjectInWorld(Frame& frame, DOMWrapperWorld& world)
144 {
145     if (!frame.isMainFrame() || &world != &mainThreadNormalWorld())
146         return;
147 
148     didClearGlobalObject();

171 void PageDebuggerAgent::didRequestAnimationFrame(int callbackId, Document& document)
172 {
173     if (!breakpointsActive())
174         return;
175 
176     JSC::JSGlobalObject* scriptState = document.execState();
177     if (!scriptState)
178         return;
179 
180     didScheduleAsyncCall(scriptState, InspectorDebuggerAgent::AsyncCallType::RequestAnimationFrame, callbackId, true);
181 }
182 
183 void PageDebuggerAgent::willFireAnimationFrame(int callbackId)
184 {
185     willDispatchAsyncCall(InspectorDebuggerAgent::AsyncCallType::RequestAnimationFrame, callbackId);
186 }
187 
188 void PageDebuggerAgent::didCancelAnimationFrame(int callbackId)
189 {
190     didCancelAsyncCall(InspectorDebuggerAgent::AsyncCallType::RequestAnimationFrame, callbackId);
191 }
