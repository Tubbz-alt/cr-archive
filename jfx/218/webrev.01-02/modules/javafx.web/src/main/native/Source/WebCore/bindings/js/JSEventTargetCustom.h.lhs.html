35 // Wrapper type for JSEventTarget's castedThis because JSDOMWindow and JSWorkerGlobalScope do not inherit JSEventTarget.
36 class JSEventTargetWrapper {
37     WTF_MAKE_FAST_ALLOCATED;
38 public:
39     JSEventTargetWrapper(EventTarget& wrapped, JSC::JSObject& wrapper)
40         : m_wrapped(wrapped)
41         , m_wrapper(wrapper)
42     { }
43 
44     EventTarget& wrapped() { return m_wrapped; }
45 
46     operator JSC::JSObject&() { return m_wrapper; }
47 
48 private:
49     EventTarget& m_wrapped;
50     JSC::JSObject& m_wrapper;
51 };
52 
53 std::unique_ptr<JSEventTargetWrapper> jsEventTargetCast(JSC::VM&, JSC::JSValue thisValue);
54 
55 template<> class IDLOperation<JSEventTarget> {
56 public:
57     using ClassParameter = JSEventTargetWrapper*;
58     using Operation = JSC::EncodedJSValue(JSC::ExecState*, ClassParameter, JSC::ThrowScope&);
59 
60     template<Operation operation, CastedThisErrorBehavior = CastedThisErrorBehavior::Throw>
61     static JSC::EncodedJSValue call(JSC::ExecState& state, const char* operationName)
62     {
63         auto& vm = state.vm();
64         auto throwScope = DECLARE_THROW_SCOPE(vm);
65 
66         auto thisObject = jsEventTargetCast(vm, state.thisValue().toThis(&state, JSC::NotStrictMode));
67         if (UNLIKELY(!thisObject))
68             return throwThisTypeError(state, throwScope, "EventTarget", operationName);
69 
70         auto& wrapped = thisObject->wrapped();
71         if (is<DOMWindow>(wrapped)) {
72             auto& window = downcast<DOMWindow>(wrapped);
73             if (!window.frame() || !BindingSecurity::shouldAllowAccessToDOMWindow(&state, window, ThrowSecurityError))
74                 return JSC::JSValue::encode(JSC::jsUndefined());
75         }
76 
77         return operation(&state, thisObject.get(), throwScope);
78     }
79 
80 };
