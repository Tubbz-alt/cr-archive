 59 IntlPluralRulesConstructor* IntlPluralRulesConstructor::create(VM& vm, Structure* structure, IntlPluralRulesPrototype* pluralRulesPrototype)
 60 {
 61     IntlPluralRulesConstructor* constructor = new (NotNull, allocateCell<IntlPluralRulesConstructor>(vm.heap)) IntlPluralRulesConstructor(vm, structure);
 62     constructor->finishCreation(vm, pluralRulesPrototype);
 63     return constructor;
 64 }

 79 void IntlPluralRulesConstructor::finishCreation(VM& vm, IntlPluralRulesPrototype* pluralRulesPrototype)
 80 {
 81     Base::finishCreation(vm, "PluralRules"_s, NameAdditionMode::WithoutStructureTransition);
 82     putDirectWithoutTransition(vm, vm.propertyNames->prototype, pluralRulesPrototype, PropertyAttribute::DontEnum | PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly);
 83     putDirectWithoutTransition(vm, vm.propertyNames->length, jsNumber(0), PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum);
 84     pluralRulesPrototype->putDirectWithoutTransition(vm, vm.propertyNames->constructor, this, static_cast<unsigned>(PropertyAttribute::DontEnum));
 85 }

 87 static EncodedJSValue JSC_HOST_CALL constructIntlPluralRules(JSGlobalObject* globalObject, CallFrame* callFrame)
 88 {
 89     VM& vm = globalObject->vm();
 90     auto scope = DECLARE_THROW_SCOPE(vm);
 91 
 92     // 13.2.1 Intl.PluralRules ([ locales [ , options ] ])
 93     // https://tc39.github.io/ecma402/#sec-intl.pluralrules
 94     Structure* structure = InternalFunction::createSubclassStructure(globalObject, callFrame->jsCallee(), callFrame->newTarget(), jsCast<IntlPluralRulesConstructor*>(callFrame->jsCallee())->pluralRulesStructure(vm));
 95     RETURN_IF_EXCEPTION(scope, encodedJSValue());
 96     IntlPluralRules* pluralRules = IntlPluralRules::create(vm, structure);
 97     ASSERT(pluralRules);
 98 
 99     scope.release();
100     pluralRules->initializePluralRules(globalObject, callFrame->argument(0), callFrame->argument(1));
101     return JSValue::encode(pluralRules);
102 }

114 EncodedJSValue JSC_HOST_CALL IntlPluralRulesConstructorFuncSupportedLocalesOf(JSGlobalObject* globalObject, CallFrame* callFrame)
115 {
116     VM& vm = globalObject->vm();
117     auto scope = DECLARE_THROW_SCOPE(vm);
118 
119     // 13.3.2 Intl.PluralRules.supportedLocalesOf (locales [, options ])
120     // https://tc39.github.io/ecma402/#sec-intl.pluralrules.supportedlocalesof
121     const HashSet<String>& availableLocales = intlPluralRulesAvailableLocales();

122 
123     Vector<String> requestedLocales = canonicalizeLocaleList(globalObject, callFrame->argument(0));
124     RETURN_IF_EXCEPTION(scope, encodedJSValue());
125 
126     RELEASE_AND_RETURN(scope, JSValue::encode(supportedLocales(globalObject, availableLocales, requestedLocales, callFrame->argument(1))));
127 }
