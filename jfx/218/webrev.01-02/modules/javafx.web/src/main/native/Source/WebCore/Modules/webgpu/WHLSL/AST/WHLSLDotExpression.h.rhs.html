41 class DotExpression final : public PropertyAccessExpression {

45         : PropertyAccessExpression(location, Kind::Dot, WTFMove(base))

49     }

51     ~DotExpression() = default;

53     DotExpression(const DotExpression&) = delete;

55 















56     String& fieldName() { return m_fieldName; }
57 
58 private:
59     String m_fieldName;
60     // This is used to allow for replaceWith into a bigger type.
61     char m_padding[12];
62 };
