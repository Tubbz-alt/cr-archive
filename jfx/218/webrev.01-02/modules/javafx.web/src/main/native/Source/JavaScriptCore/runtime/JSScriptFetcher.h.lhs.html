27 #pragma once
28 
29 #include "JSGlobalObject.h"
30 #include "JSObject.h"
31 #include "ScriptFetcher.h"
32 #include <wtf/RefPtr.h>
33 
34 namespace JSC {
35 
36 class JSScriptFetcher final : public JSCell {
37 public:
38     using Base = JSCell;
39 
40     static const unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
41     static const bool needsDestruction = true;
42 
43     DECLARE_EXPORT_INFO;
44 






45     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
46     {
47         return Structure::create(vm, globalObject, prototype, TypeInfo(JSScriptFetcherType, StructureFlags), info());
48     }
49 
50     static JSScriptFetcher* create(VM& vm, Structure* structure, RefPtr<ScriptFetcher>&& scriptFetcher)
51     {
52         auto* result = new (NotNull, allocateCell<JSScriptFetcher>(vm.heap)) JSScriptFetcher(vm, structure, WTFMove(scriptFetcher));
53         result->finishCreation(vm);
54         return result;
55     }
56 
57     static JSScriptFetcher* create(VM& vm, RefPtr<ScriptFetcher>&& scriptFetcher)
58     {
59         return create(vm, vm.scriptFetcherStructure.get(), WTFMove(scriptFetcher));
60     }
61 
62     ScriptFetcher* fetcher() const
63     {
64         return m_fetcher.get();
65     }
66 
67     static void destroy(JSCell*);
68 
69 private:
70     JSScriptFetcher(VM& vm, Structure* structure, RefPtr<ScriptFetcher>&& scriptFetcher)
71         : Base(vm, structure)
72         , m_fetcher(WTFMove(scriptFetcher))
73     {
74     }
75 
76     RefPtr<ScriptFetcher> m_fetcher;
77 };
78 
79 } // namespace JSC
