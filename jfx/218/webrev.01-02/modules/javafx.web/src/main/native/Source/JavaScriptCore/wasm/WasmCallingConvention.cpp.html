35 const JSCallingConvention& jsCallingConvention()
36 {
37     static LazyNeverDestroyed<JSCallingConvention> staticJSCallingConvention;
38     static std::once_flag staticJSCCallingConventionFlag;
39     std::call_once(staticJSCCallingConventionFlag, [] () {
40         RegisterSet callerSaveRegisters = RegisterSet::allRegisters();
41         callerSaveRegisters.exclude(RegisterSet::calleeSaveRegisters());
42 
43         staticJSCallingConvention.construct(Vector<Reg>(), Vector<Reg>(), RegisterSet::calleeSaveRegisters(), WTFMove(callerSaveRegisters));
44     });
45 
46     return staticJSCallingConvention;
47 }

49 const WasmCallingConvention& wasmCallingConvention()
50 {
51     static LazyNeverDestroyed<WasmCallingConvention> staticWasmCallingConvention;
52     static std::once_flag staticWasmCallingConventionFlag;
53     std::call_once(staticWasmCallingConventionFlag, [] () {
54         Vector<Reg> gprArgumentRegisters(GPRInfo::numberOfArgumentRegisters);
55         for (unsigned i = 0; i < GPRInfo::numberOfArgumentRegisters; ++i)
56             gprArgumentRegisters[i] = GPRInfo::toArgumentRegister(i);
57 
58         Vector<Reg> fprArgumentRegisters(FPRInfo::numberOfArgumentRegisters);
59         for (unsigned i = 0; i < FPRInfo::numberOfArgumentRegisters; ++i)
60             fprArgumentRegisters[i] = FPRInfo::toArgumentRegister(i);
61 
62         RegisterSet scratch = RegisterSet::allGPRs();
63         scratch.exclude(RegisterSet::calleeSaveRegisters());
64         scratch.exclude(RegisterSet::macroScratchRegisters());
65         scratch.exclude(RegisterSet::reservedHardwareRegisters());
66         scratch.exclude(RegisterSet::stackRegisters());
67         for (Reg reg : gprArgumentRegisters)
68             scratch.clear(reg);
69 
70         Vector<GPRReg> scratchGPRs;
71         for (Reg reg : scratch)
72             scratchGPRs.append(reg.gpr());
73         RELEASE_ASSERT(scratchGPRs.size() >= 2);
74 
75         RegisterSet callerSaveRegisters = RegisterSet::allRegisters();
76         callerSaveRegisters.exclude(RegisterSet::calleeSaveRegisters());
77 
78         staticWasmCallingConvention.construct(WTFMove(gprArgumentRegisters), WTFMove(fprArgumentRegisters), WTFMove(scratchGPRs), RegisterSet::calleeSaveRegisters(), WTFMove(callerSaveRegisters));
79     });
80 
81     return staticWasmCallingConvention;
82 }
