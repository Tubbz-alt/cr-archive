 45 static bool jsDOMWindowPropertiesGetOwnPropertySlotNamedItemGetter(JSDOMWindowProperties* thisObject, DOMWindow& window, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot)
 46 {
 47     // Check for child frames by name before built-in properties to match Mozilla. This does
 48     // not match IE, but some sites end up naming frames things that conflict with window
 49     // properties that are in Moz but not IE. Since we have some of these, we have to do it
 50     // the Moz way.
 51     if (auto* frame = window.frame()) {
 52         if (auto* scopedChild = frame->tree().scopedChild(propertyNameToAtomString(propertyName))) {
 53             slot.setValue(thisObject, JSC::PropertyAttribute::ReadOnly | JSC::PropertyAttribute::DontDelete | JSC::PropertyAttribute::DontEnum, toJS(lexicalGlobalObject, scopedChild->document()->domWindow()));
 54             return true;
 55         }
 56     }
 57 
 58     if (!BindingSecurity::shouldAllowAccessToDOMWindow(lexicalGlobalObject, window, ThrowSecurityError))
 59         return false;
 60 
 61     // FIXME: Search the whole frame hierarchy somewhere around here.
 62     // We need to test the correct priority order.
 63 
 64     // Allow shortcuts like 'Image1' instead of document.images.Image1
 65     auto* document = window.document();
 66     if (is<HTMLDocument>(document)) {
 67         auto& htmlDocument = downcast<HTMLDocument>(*document);
 68         auto* atomicPropertyName = propertyName.publicName();
 69         if (atomicPropertyName && htmlDocument.hasWindowNamedItem(*atomicPropertyName)) {
 70             JSValue namedItem;
 71             if (UNLIKELY(htmlDocument.windowNamedItemContainsMultipleElements(*atomicPropertyName))) {
 72                 Ref<HTMLCollection> collection = document->windowNamedItems(atomicPropertyName);
 73                 ASSERT(collection->length() > 1);
 74                 namedItem = toJS(lexicalGlobalObject, thisObject->globalObject(), collection);
 75             } else
 76                 namedItem = toJS(lexicalGlobalObject, thisObject->globalObject(), htmlDocument.windowNamedItem(*atomicPropertyName));
 77             slot.setValue(thisObject, JSC::PropertyAttribute::ReadOnly | JSC::PropertyAttribute::DontDelete | JSC::PropertyAttribute::DontEnum, namedItem);
 78             return true;
 79         }
 80     }
 81 
 82     return false;
 83 }

 85 bool JSDOMWindowProperties::getOwnPropertySlot(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot)
 86 {
 87     auto* thisObject = jsCast<JSDOMWindowProperties*>(object);
 88     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
 89     if (Base::getOwnPropertySlot(thisObject, lexicalGlobalObject, propertyName, slot))
 90         return true;
 91     JSValue proto = thisObject->getPrototypeDirect(lexicalGlobalObject->vm());
 92     if (proto.isObject() && jsCast<JSObject*>(proto)->hasProperty(lexicalGlobalObject, propertyName))
 93         return false;
 94 
 95     auto& vm = lexicalGlobalObject->vm();
 96 
 97     // FIXME: We should probably add support for JSRemoteDOMWindowBase too.
 98     auto* jsWindow = jsDynamicCast<JSDOMWindowBase*>(vm, thisObject->globalObject());
 99     if (!jsWindow)
100         return false;
101 
102     auto& window = jsWindow->wrapped();
103     return jsDOMWindowPropertiesGetOwnPropertySlotNamedItemGetter(thisObject, window, lexicalGlobalObject, propertyName, slot);
104 }

106 bool JSDOMWindowProperties::getOwnPropertySlotByIndex(JSObject* object, JSGlobalObject* lexicalGlobalObject, unsigned index, PropertySlot& slot)
107 {
108     VM& vm = lexicalGlobalObject->vm();
109     return getOwnPropertySlot(object, lexicalGlobalObject, Identifier::from(vm, index), slot);
110 }
