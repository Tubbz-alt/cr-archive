29 #include <wtf/text/StringBuilder.h>

31 namespace WebCore {

33 String CSSCubicBezierTimingFunctionValue::customCSSText() const
34 {
35     return makeString("cubic-bezier(", m_x1, ", ", m_y1, ", ", m_x2, ", ", m_y2, ')');
36 }
37 
38 bool CSSCubicBezierTimingFunctionValue::equals(const CSSCubicBezierTimingFunctionValue& other) const
39 {
40     return m_x1 == other.m_x1 && m_x2 == other.m_x2 && m_y1 == other.m_y1 && m_y2 == other.m_y2;
41 }
42 
43 String CSSStepsTimingFunctionValue::customCSSText() const
44 {
45     StringBuilder builder;
46     builder.appendLiteral("steps(");
47     builder.appendNumber(m_steps);
48     if (m_stepAtStart)
49         builder.appendLiteral(", start)");
50     else
51         builder.appendLiteral(", end)");
52     return builder.toString();
53 }
54 
55 bool CSSStepsTimingFunctionValue::equals(const CSSStepsTimingFunctionValue& other) const
56 {
57     return m_steps == other.m_steps && m_stepAtStart == other.m_stepAtStart;
58 }
59 
60 String CSSSpringTimingFunctionValue::customCSSText() const
61 {
62     StringBuilder builder;
63     builder.appendLiteral("spring(");
64     builder.append(FormattedNumber::fixedPrecision(m_mass));
65     builder.append(' ');
66     builder.append(FormattedNumber::fixedPrecision(m_stiffness));
67     builder.append(' ');
68     builder.append(FormattedNumber::fixedPrecision(m_damping));
69     builder.append(' ');
70     builder.append(FormattedNumber::fixedPrecision(m_initialVelocity));
71     builder.append(')');
72     return builder.toString();
73 }
74 
75 bool CSSSpringTimingFunctionValue::equals(const CSSSpringTimingFunctionValue& other) const
76 {
77     return m_mass == other.m_mass && m_stiffness == other.m_stiffness && m_damping == other.m_damping && m_initialVelocity == other.m_initialVelocity;
78 }
