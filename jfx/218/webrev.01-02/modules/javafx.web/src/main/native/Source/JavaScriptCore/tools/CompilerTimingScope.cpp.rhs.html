 36 namespace JSC {
 37 
 38 namespace {
 39 
 40 class CompilerTimingScopeState {
 41     WTF_MAKE_NONCOPYABLE(CompilerTimingScopeState);
 42     WTF_MAKE_FAST_ALLOCATED;
 43 public:
 44     CompilerTimingScopeState() { }
 45 
 46     Seconds addToTotal(const char* compilerName, const char* name, Seconds duration)
 47     {
 48         auto locker = holdLock(lock);
 49 
 50         for (auto& tuple : totals) {
 51             if (String(std::get<0>(tuple)) == String(compilerName) && String(std::get<1>(tuple)) == String(name)) {
 52                 std::get<2>(tuple) += duration;
 53                 return std::get<2>(tuple);
 54             }
 55         }
 56 
 57         totals.append({ compilerName, name, duration });
 58         return duration;
 59     }
 60 
 61     void logTotals()
 62     {
 63         for (auto& tuple : totals) {
 64             dataLogLn(
 65                 "[", std::get<0>(tuple), "] ", std::get<1>(tuple), " total ms: ", std::get<2>(tuple).milliseconds());
 66         }
 67     }
 68 
 69 private:
 70     Vector<std::tuple<const char*, const char*, Seconds>> totals;
 71     Lock lock;
 72 };
 73 
 74 CompilerTimingScopeState& compilerTimingScopeState()
 75 {
 76     static Atomic<CompilerTimingScopeState*> s_state;
 77     return ensurePointer(s_state, [] { return new CompilerTimingScopeState(); });
 78 }
 79 
 80 } // anonymous namespace
 81 
 82 CompilerTimingScope::CompilerTimingScope(const char* compilerName, const char* name)
 83     : m_compilerName(compilerName)
 84     , m_name(name)
 85 {
 86     if (UNLIKELY(Options::logPhaseTimes() || Options::reportTotalPhaseTimes()))
 87         m_before = MonotonicTime::now();
 88 }
 89 
 90 CompilerTimingScope::~CompilerTimingScope()
 91 {
 92     if (UNLIKELY(Options::logPhaseTimes() || Options::reportTotalPhaseTimes())) {
 93         Seconds duration = MonotonicTime::now() - m_before;
 94         auto total = compilerTimingScopeState().addToTotal(m_compilerName, m_name, duration);
 95         if (Options::logPhaseTimes()) {
 96             dataLog(
 97                 "[", m_compilerName, "] ", m_name, " took: ", duration.milliseconds(), " ms ",
 98                 "(total: ", total.milliseconds(),
 99                 " ms).\n");
100         }
101     }
102 }
103 
104 void logTotalPhaseTimes()
105 {
106     compilerTimingScopeState().logTotals();
107 }
