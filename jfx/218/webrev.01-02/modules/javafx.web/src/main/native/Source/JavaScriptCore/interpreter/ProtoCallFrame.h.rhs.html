26 #pragma once
27 
28 #include "CodeBlock.h"
29 #include "Register.h"
30 #include "StackAlignment.h"
31 #include <wtf/ForbidHeapAllocation.h>
32 
33 namespace JSC {
34 
35 struct JS_EXPORT_PRIVATE ProtoCallFrame {
36     WTF_FORBID_HEAP_ALLOCATION;
37 public:
38     // CodeBlock, Callee, ArgumentCount, and |this|.
39     static constexpr unsigned numberOfRegisters { 4 };
40 
41     Register codeBlockValue;
42     Register calleeValue;
43     Register argCountAndCodeOriginValue;
44     Register thisArg;
45     uint32_t paddedArgCount;
46     bool hasArityMismatch;
47     JSValue *args;
48     JSGlobalObject* globalObject;
49 
50     inline void init(CodeBlock*, JSGlobalObject*, JSObject*, JSValue, int, JSValue* otherArgs = 0);
51 
52     inline CodeBlock* codeBlock() const;
53     inline void setCodeBlock(CodeBlock*);
54 
55     inline JSObject* callee() const;
56     inline void setCallee(JSObject*);
57     void setGlobalObject(JSGlobalObject* object)
58     {
59         globalObject = object;
60     }
61 
62     int argumentCountIncludingThis() const { return argCountAndCodeOriginValue.payload(); }
63     int argumentCount() const { return argumentCountIncludingThis() - 1; }
64     void setArgumentCountIncludingThis(int count) { argCountAndCodeOriginValue.payload() = count; }
65     void setPaddedArgCount(uint32_t argCount) { paddedArgCount = argCount; }
66 
67     void clearCurrentVPC() { argCountAndCodeOriginValue.tag() = 0; }
68 
69     JSValue thisValue() const { return thisArg.Register::jsValue(); }
70     void setThisValue(JSValue value) { thisArg = value; }
71 
72     bool needArityCheck() { return hasArityMismatch; }
73 
74     JSValue argument(size_t argumentIndex)
75     {
76         ASSERT(static_cast<int>(argumentIndex) < argumentCount());
77         return args[argumentIndex];
78     }
79     void setArgument(size_t argumentIndex, JSValue value)
80     {
81         ASSERT(static_cast<int>(argumentIndex) < argumentCount());
82         args[argumentIndex] = value;
83     }
84 };
85 


















86 } // namespace JSC
