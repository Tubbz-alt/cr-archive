33 namespace WebCore {
34 
35 WTF_MAKE_ISO_ALLOCATED_IMPL(IDBVersionChangeEvent);
36 
37 IDBVersionChangeEvent::IDBVersionChangeEvent(const IDBResourceIdentifier& requestIdentifier, uint64_t oldVersion, uint64_t newVersion, const AtomString& name)
38     : Event(name, CanBubble::No, IsCancelable::No)
39     , m_requestIdentifier(requestIdentifier)
40     , m_oldVersion(oldVersion)
41 {
42     if (newVersion)
43         m_newVersion = newVersion;
44     else
45         m_newVersion = WTF::nullopt;
46 }
47 
48 IDBVersionChangeEvent::IDBVersionChangeEvent(const AtomString& name, const Init& init, IsTrusted isTrusted)
49     : Event(name, init, isTrusted)
50     , m_oldVersion(init.oldVersion)
51     , m_newVersion(init.newVersion)
52 {
53 }
54 
55 EventInterface IDBVersionChangeEvent::eventInterface() const
56 {
57     return IDBVersionChangeEventInterfaceType;
58 }
59 
60 } // namespace WebCore
