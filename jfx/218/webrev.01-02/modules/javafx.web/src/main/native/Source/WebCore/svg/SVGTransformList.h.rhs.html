 22 #pragma once
 23 
 24 #include "DOMMatrix2DInit.h"
 25 #include "SVGTransform.h"
 26 #include "SVGTransformable.h"
 27 #include "SVGValuePropertyList.h"
 28 
 29 namespace WebCore {
 30 
 31 class SVGTransformList final : public SVGValuePropertyList<SVGTransform> {
 32     friend class SVGViewSpec;
 33     using Base = SVGValuePropertyList<SVGTransform>;
 34     using Base::Base;
 35 
 36 public:
 37     static Ref<SVGTransformList> create()
 38     {
 39         return adoptRef(*new SVGTransformList());
 40     }
 41 
 42     static Ref<SVGTransformList> create(SVGPropertyOwner* owner, SVGPropertyAccess access)
 43     {
 44         return adoptRef(*new SVGTransformList(owner, access));
 45     }
 46 
 47     static Ref<SVGTransformList> create(const SVGTransformList& other, SVGPropertyAccess access)
 48     {
 49         return adoptRef(*new SVGTransformList(other, access));
 50     }
 51 
 52     ExceptionOr<Ref<SVGTransform>> createSVGTransformFromMatrix(DOMMatrix2DInit&& matrixInit)
 53     {
 54         auto svgTransform =  SVGTransform::create();
 55         svgTransform->setMatrix(WTFMove(matrixInit));
 56         return svgTransform;
 57     }
 58 
 59     ExceptionOr<RefPtr<SVGTransform>> consolidate()
 60     {
 61         auto result = canAlterList();
 62         if (result.hasException())
 63             return result.releaseException();
 64         ASSERT(result.releaseReturnValue());
 65 
 66         // Spec: If the list was empty, then a value of null is returned.
 67         if (m_items.isEmpty())
 68             return nullptr;
 69 
 70         if (m_items.size() == 1)
 71             return makeRefPtr(at(0).get());
 72 
 73         auto newItem = SVGTransform::create(concatenate());
 74         clearItems();
 75 
 76         auto item = append(WTFMove(newItem));
 77         commitChange();
 78         return makeRefPtr(item.get());
 79     }
 80 
 81     void parse(const String& value)
 82     {
 83         clearItems();
 84 
 85         auto upconvertedCharacters = StringView(value).upconvertedCharacters();
 86         const UChar* start = upconvertedCharacters;
 87         if (!parse(start, start + value.length()))
 88             clearItems();
 89     }
 90 
 91     AffineTransform concatenate() const
 92     {
 93         AffineTransform result;
 94         for (const auto& transform : m_items)
 95             result *= transform->matrix()->value();
 96         return result;
 97     }
 98 
 99     String valueAsString() const override
100     {
101         StringBuilder builder;
102         for (const auto& transfrom : m_items) {
103             if (builder.length())
104                 builder.append(' ');
105 
106             builder.append(transfrom->value().valueAsString());
107         }
108         return builder.toString();
109     }
110 
111 private:
112     bool parse(const UChar*& start, const UChar* end)
113     {
114         bool delimParsed = false;
115         while (start < end) {
116             delimParsed = false;
117             SVGTransformValue::SVGTransformType type = SVGTransformValue::SVG_TRANSFORM_UNKNOWN;
118             skipOptionalSVGSpaces(start, end);
119 
120             if (!SVGTransformable::parseAndSkipType(start, end, type))
121                 return false;
122 
123             Ref<SVGTransform> transform = SVGTransform::create(type);
124             if (!SVGTransformable::parseTransformValue(type, start, end, transform->value()))
125                 return false;
126 
127             append(WTFMove(transform));
128             skipOptionalSVGSpaces(start, end);
129             if (start < end && *start == ',') {
130                 delimParsed = true;
131                 ++start;
132             }
133 
134             skipOptionalSVGSpaces(start, end);
135         }
136         return !delimParsed;
137     }
138 };
