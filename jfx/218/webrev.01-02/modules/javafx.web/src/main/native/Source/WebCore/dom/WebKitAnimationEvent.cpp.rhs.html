26 #include "config.h"
27 #include "WebKitAnimationEvent.h"
28 
29 #include <wtf/IsoMallocInlines.h>
30 
31 namespace WebCore {
32 
33 WTF_MAKE_ISO_ALLOCATED_IMPL(WebKitAnimationEvent);
34 
35 WebKitAnimationEvent::WebKitAnimationEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
36     : Event(type, initializer, isTrusted)
37     , m_animationName(initializer.animationName)
38     , m_elapsedTime(initializer.elapsedTime)
39     , m_pseudoElement(initializer.pseudoElement)
40 {
41 }
42 
43 WebKitAnimationEvent::WebKitAnimationEvent(const AtomString& type, const String& animationName, double elapsedTime, const String& pseudoElement)
44     : Event(type, CanBubble::Yes, IsCancelable::Yes)
45     , m_animationName(animationName)
46     , m_elapsedTime(elapsedTime)
47     , m_pseudoElement(pseudoElement)
48 {
49 }
50 
51 WebKitAnimationEvent::~WebKitAnimationEvent() = default;
52 
53 const String& WebKitAnimationEvent::animationName() const
54 {
55     return m_animationName;
56 }
57 
58 double WebKitAnimationEvent::elapsedTime() const
59 {
60     return m_elapsedTime;
61 }
62 
63 const String& WebKitAnimationEvent::pseudoElement() const
64 {
65     return m_pseudoElement;
66 }
67 
68 EventInterface WebKitAnimationEvent::eventInterface() const
69 {
70     return WebKitAnimationEventInterfaceType;
71 }
