35 void AbstractHeap::Payload::dump(PrintStream& out) const
36 {
37     if (isTop())
38         out.print("TOP");
39     else
40         out.print(value());
41 }
42 
43 void AbstractHeap::dump(PrintStream& out) const
44 {
45     out.print(kind());
46     if (kind() == InvalidAbstractHeap || kind() == World || kind() == Heap || payload().isTop())
47         return;
48     if (kind() == DOMState) {
49         out.print("(", DOMJIT::HeapRange::fromRaw(payload().value32()), ")");
50         return;
51     }
52     out.print("(", payload(), ")");
53 }

61 void printInternal(PrintStream& out, AbstractHeapKind kind)
62 {
63     switch (kind) {
64 #define ABSTRACT_HEAP_DUMP(name)            \
65     case AbstractHeapKind::name:            \
66         out.print(#name);                   \
67         return;
68     FOR_EACH_ABSTRACT_HEAP_KIND(ABSTRACT_HEAP_DUMP)
69 #undef ABSTRACT_HEAP_DUMP
70     }
71     RELEASE_ASSERT_NOT_REACHED();
72 }
