33 void JSStyleSheet::visitAdditionalChildren(JSC::SlotVisitor& visitor)
34 {
35     visitor.addOpaqueRoot(root(&wrapped()));
36 }
37 
38 JSC::JSValue toJSNewlyCreated(JSC::JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<StyleSheet>&& styleSheet)
39 {
40     if (styleSheet->isCSSStyleSheet())
41         return createWrapper<CSSStyleSheet>(globalObject, WTFMove(styleSheet));
42     return createWrapper<StyleSheet>(globalObject, WTFMove(styleSheet));
43 }
44 
45 JSC::JSValue toJS(JSC::JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, StyleSheet& stylesheet)
46 {
47     return wrap(lexicalGlobalObject, globalObject, stylesheet);
48 }
