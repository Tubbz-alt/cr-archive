 40 inline SVGImageElement::SVGImageElement(const QualifiedName& tagName, Document& document)
 41     : SVGGraphicsElement(tagName, document)
 42     , SVGURIReference(this)
 43     , m_imageLoader(*this)
 44 {
 45     static std::once_flag onceFlag;
 46     std::call_once(onceFlag, [] {
 47         PropertyRegistry::registerProperty<SVGNames::xAttr, &SVGImageElement::m_x>();
 48         PropertyRegistry::registerProperty<SVGNames::yAttr, &SVGImageElement::m_y>();
 49         PropertyRegistry::registerProperty<SVGNames::widthAttr, &SVGImageElement::m_width>();
 50         PropertyRegistry::registerProperty<SVGNames::heightAttr, &SVGImageElement::m_height>();
 51         PropertyRegistry::registerProperty<SVGNames::preserveAspectRatioAttr, &SVGImageElement::m_preserveAspectRatio>();
 52     });
 53 }

 60 bool SVGImageElement::hasSingleSecurityOrigin() const
 61 {
 62     auto* renderer = downcast<RenderSVGImage>(this->renderer());
 63     if (!renderer || !renderer->imageResource().cachedImage())
 64         return true;
 65     auto* image = renderer->imageResource().cachedImage()->image();
 66     return !image || image->hasSingleSecurityOrigin();
 67 }

 69 void SVGImageElement::parseAttribute(const QualifiedName& name, const AtomString& value)
 70 {
 71     if (name == SVGNames::preserveAspectRatioAttr) {
 72         SVGPreserveAspectRatioValue preserveAspectRatio;
 73         preserveAspectRatio.parse(value);
 74         m_preserveAspectRatio->setBaseValInternal(preserveAspectRatio);
 75         return;
 76     }
 77 
 78     SVGParsingError parseError = NoError;
 79 
 80     if (name == SVGNames::xAttr)
 81         m_x->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
 82     else if (name == SVGNames::yAttr)
 83         m_y->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
 84     else if (name == SVGNames::widthAttr)
 85         m_width->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError, SVGLengthNegativeValuesMode::Forbid));
 86     else if (name == SVGNames::heightAttr)
 87         m_height->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError, SVGLengthNegativeValuesMode::Forbid));
 88 
 89     reportAttributeParsingError(parseError, name, value);
 90 
 91     SVGGraphicsElement::parseAttribute(name, value);
 92     SVGURIReference::parseAttribute(name, value);
 93 }

 95 void SVGImageElement::svgAttributeChanged(const QualifiedName& attrName)
 96 {
 97     if (attrName == SVGNames::xAttr || attrName == SVGNames::yAttr) {
 98         InstanceInvalidationGuard guard(*this);
 99         updateRelativeLengthsInformation();
100 
101         if (auto* renderer = this->renderer()) {
102             if (downcast<RenderSVGImage>(*renderer).updateImageViewport())
103                 RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
104         }
105         return;
106     }
107 
108     if (attrName == SVGNames::widthAttr || attrName == SVGNames::heightAttr) {
109         InstanceInvalidationGuard guard(*this);
110         invalidateSVGPresentationAttributeStyle();
111         return;
112     }
113 
114     if (attrName == SVGNames::preserveAspectRatioAttr) {
115         InstanceInvalidationGuard guard(*this);
116         if (auto* renderer = this->renderer())
117             RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
118         return;
119     }
120 
121     if (SVGURIReference::isKnownAttribute(attrName)) {
122         m_imageLoader.updateFromElementIgnoringPreviousError();
123         return;
124     }
125 
126     SVGGraphicsElement::svgAttributeChanged(attrName);
127 }

129 RenderPtr<RenderElement> SVGImageElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)

131     return createRenderer<RenderSVGImage>(*this, WTFMove(style));

133 
134 bool SVGImageElement::haveLoadedRequiredResources()
135 {
136     return !m_imageLoader.hasPendingActivity();
137 }
138 
139 void SVGImageElement::didAttachRenderers()
140 {
141     if (auto* imageObj = downcast<RenderSVGImage>(renderer())) {
142         if (imageObj->imageResource().cachedImage())
143             return;
144 
145         imageObj->imageResource().setCachedImage(m_imageLoader.image());
146     }
147 }

149 Node::InsertedIntoAncestorResult SVGImageElement::insertedIntoAncestor(InsertionType insertionType, ContainerNode& parentOfInsertedTree)
150 {
151     SVGGraphicsElement::insertedIntoAncestor(insertionType, parentOfInsertedTree);
152     if (!insertionType.connectedToDocument)
153         return InsertedIntoAncestorResult::Done;
154     // Update image loader, as soon as we're living in the tree.
155     // We can only resolve base URIs properly, after that!
156     m_imageLoader.updateFromElement();
157     return InsertedIntoAncestorResult::Done;
158 }

160 const AtomString& SVGImageElement::imageSourceURL() const
161 {
162     return getAttribute(SVGNames::hrefAttr, XLinkNames::hrefAttr);
163 }

165 void SVGImageElement::addSubresourceAttributeURLs(ListHashSet<URL>& urls) const
166 {
167     SVGGraphicsElement::addSubresourceAttributeURLs(urls);
168 
169     addSubresourceURL(urls, document().completeURL(href()));
170 }
