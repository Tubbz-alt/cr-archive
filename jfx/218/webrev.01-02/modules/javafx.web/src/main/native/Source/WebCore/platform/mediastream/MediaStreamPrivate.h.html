 36 
 37 #if ENABLE(MEDIA_STREAM)
 38 
 39 #include "FloatSize.h"
 40 #include "MediaStreamTrackPrivate.h"
 41 #include <wtf/Function.h>
 42 #include <wtf/HashMap.h>
 43 #include <wtf/MediaTime.h>
 44 #include <wtf/RefPtr.h>
 45 #include <wtf/UUID.h>
 46 #include <wtf/Vector.h>
 47 
 48 namespace WebCore {
 49 
 50 class MediaStream;
 51 class OrientationNotifier;
 52 
 53 class MediaStreamPrivate final
 54     : public MediaStreamTrackPrivate::Observer
 55     , public RefCounted<MediaStreamPrivate>
 56 #if !RELEASE_LOG_DISABLED
 57     , private LoggerHelper
 58 #endif
 59 {
 60 public:
 61     class Observer {
 62     public:
 63         virtual ~Observer() = default;
 64 
 65         virtual void characteristicsChanged() { }
 66         virtual void activeStatusChanged() { }
 67         virtual void didAddTrack(MediaStreamTrackPrivate&) { }
 68         virtual void didRemoveTrack(MediaStreamTrackPrivate&) { }
 69     };
 70 
 71     static Ref<MediaStreamPrivate> create(Ref<const Logger>&&, Ref<RealtimeMediaSource>&&);
 72     static Ref<MediaStreamPrivate> create(Ref<const Logger>&&, const Vector<Ref<RealtimeMediaSource>>& audioSources, const Vector<Ref<RealtimeMediaSource>>& videoSources);
 73     static Ref<MediaStreamPrivate> create(Ref<const Logger>&& logger, const MediaStreamTrackPrivateVector& tracks, String&& id = createCanonicalUUIDString()) { return adoptRef(*new MediaStreamPrivate(WTFMove(logger), tracks, WTFMove(id))); }
 74 
 75     virtual ~MediaStreamPrivate();
 76 
 77     enum class NotifyClientOption { Notify, DontNotify };
 78 
 79     void addObserver(Observer&);
 80     void removeObserver(Observer&);
 81 
 82     String id() const { return m_id; }
 83 
 84     WEBCORE_EXPORT MediaStreamTrackPrivateVector tracks() const;
 85     MediaStreamTrackPrivate* activeVideoTrack() { return m_activeVideoTrack; }
 86 
 87     bool active() const { return m_isActive; }
 88     void updateActiveState(NotifyClientOption);
 89 
 90     void addTrack(RefPtr<MediaStreamTrackPrivate>&&, NotifyClientOption = NotifyClientOption::Notify);
 91     void removeTrack(MediaStreamTrackPrivate&, NotifyClientOption = NotifyClientOption::Notify);
 92 
 93     void startProducingData();
 94     void stopProducingData();
 95     bool isProducingData() const;
 96 
 97     bool hasVideo() const;
 98     bool hasAudio() const;
 99     bool muted() const;
100 
101     FloatSize intrinsicSize() const;
102 
103     void monitorOrientation(OrientationNotifier&);
104 
105 #if !RELEASE_LOG_DISABLED
106     const Logger& logger() const final { return m_logger; }
107     const void* logIdentifier() const final { return m_logIdentifier; }
108 #endif
109 
110 private:
111     MediaStreamPrivate(Ref<const Logger>&&, const MediaStreamTrackPrivateVector&, String&&);
112 
113     // MediaStreamTrackPrivate::Observer
114     void trackStarted(MediaStreamTrackPrivate&) override;
115     void trackEnded(MediaStreamTrackPrivate&) override;
116     void trackMutedChanged(MediaStreamTrackPrivate&) override;
117     void trackSettingsChanged(MediaStreamTrackPrivate&) override;
118     void trackEnabledChanged(MediaStreamTrackPrivate&) override;
119 
120     void characteristicsChanged();
121     void updateActiveVideoTrack();
122 
123     void forEachObserver(const WTF::Function<void(Observer&)>&) const;
124 
125 #if !RELEASE_LOG_DISABLED
126     const char* logClassName() const final { return "MediaStreamPrivate"; }
127     WTFLogChannel& logChannel() const final;
128 #endif
129 
130     HashSet<Observer*> m_observers;
131     String m_id;
132     MediaStreamTrackPrivate* m_activeVideoTrack { nullptr };
133     HashMap<String, RefPtr<MediaStreamTrackPrivate>> m_trackSet;
134     bool m_isActive { false };
135 #if !RELEASE_LOG_DISABLED
136     Ref<const Logger> m_logger;
137     const void* m_logIdentifier;
138 #endif
139 };
140 
141 typedef Vector<RefPtr<MediaStreamPrivate>> MediaStreamPrivateVector;
142 
143 } // namespace WebCore
144 
145 #endif // ENABLE(MEDIA_STREAM)

147 #endif // MediaStreamPrivate_h
