26 #pragma once
27 
28 #include "JSCast.h"
29 #include "JSPromise.h"
30 #include "Structure.h"
31 
32 namespace JSC {
33 
34 class Exception;
35 class JSPromiseConstructor;
36 class JSFunction;
37 
38 class JSPromiseDeferred : public JSCell {
39 public:
40     typedef JSCell Base;
41     static const unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
42 
43     struct DeferredData {
44         WTF_FORBID_HEAP_ALLOCATION;
45     public:
46         JSPromise* promise { nullptr };
47         JSFunction* resolve { nullptr };
48         JSFunction* reject { nullptr };
49     };
50     static DeferredData createDeferredData(ExecState*, JSGlobalObject*, JSPromiseConstructor*);
51 
52     JS_EXPORT_PRIVATE static JSPromiseDeferred* tryCreate(ExecState*, JSGlobalObject*);
53     JS_EXPORT_PRIVATE static JSPromiseDeferred* create(VM&, JSPromise*, JSFunction* resolve, JSFunction* reject);
54 
55     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
56     {
57         return Structure::create(vm, globalObject, prototype, TypeInfo(CellType, StructureFlags), info());
58     }
59 
60     DECLARE_EXPORT_INFO;
61 
62     JSPromise* promise() const { return m_promise.get(); }
63     JSFunction* resolve() const { return m_resolve.get(); }
64     JSFunction* reject() const { return m_reject.get(); }
65 
66     JS_EXPORT_PRIVATE void resolve(ExecState*, JSValue);
67     JS_EXPORT_PRIVATE void reject(ExecState*, JSValue);
68     JS_EXPORT_PRIVATE void reject(ExecState*, Exception*);
69 
70 #ifndef NDEBUG
71     void promiseAsyncPending() { m_promiseIsAsyncPending = true; }
72 #endif
73 
74 protected:
75     JSPromiseDeferred(VM&, Structure*);
76     void finishCreation(VM&, JSPromise*, JSFunction* resolve, JSFunction* reject);
77     static void visitChildren(JSCell*, SlotVisitor&);
78 
79 private:
80     JSPromiseDeferred(VM&);
81 
82 #ifndef NDEBUG
83     bool m_promiseIsAsyncPending { false };
84 #endif
85 
86     WriteBarrier<JSPromise> m_promise;
87     WriteBarrier<JSFunction> m_resolve;
88     WriteBarrier<JSFunction> m_reject;
89 };
90 
91 } // namespace JSC
