 39 void Type::destroy(Type& type)
 40 {
 41     switch (type.kind()) {
 42     case Kind::TypeReference:
 43         delete &downcast<TypeReference>(type);
 44         break;
 45     case Kind::Pointer:
 46         delete &downcast<PointerType>(type);
 47         break;
 48     case Kind::ArrayReference:
 49         delete &downcast<ArrayReferenceType>(type);
 50         break;
 51     case Kind::Array:
 52         delete &downcast<ArrayType>(type);
 53         break;
 54 
 55     case Kind::TypeDefinition:
 56         delete &downcast<TypeDefinition>(type);
 57         break;
 58     case Kind::StructureDefinition:
 59         delete &downcast<StructureDefinition>(type);
 60         break;
 61     case Kind::EnumerationDefinition:
 62         delete &downcast<EnumerationDefinition>(type);
 63         break;
 64     case Kind::NativeTypeDeclaration:
 65         delete &downcast<NativeTypeDeclaration>(type);
 66         break;
 67 
 68     case Kind::FloatLiteral:
 69         delete &downcast<FloatLiteralType>(type);
 70         break;
 71     case Kind::IntegerLiteral:
 72         delete &downcast<IntegerLiteralType>(type);
 73         break;



 74     case Kind::UnsignedIntegerLiteral:
 75         delete &downcast<UnsignedIntegerLiteralType>(type);
 76         break;
 77     }
 78 }

 80 void Type::destruct(Type& type)
 81 {
 82     switch (type.kind()) {
 83     case Kind::TypeReference:
 84         downcast<TypeReference>(type).~TypeReference();
 85         break;
 86     case Kind::Pointer:
 87         downcast<PointerType>(type).~PointerType();
 88         break;
 89     case Kind::ArrayReference:
 90         downcast<ArrayReferenceType>(type).~ArrayReferenceType();
 91         break;
 92     case Kind::Array:
 93         downcast<ArrayType>(type).~ArrayType();
 94         break;
 95 
 96     case Kind::TypeDefinition:
 97         downcast<TypeDefinition>(type).~TypeDefinition();
 98         break;
 99     case Kind::StructureDefinition:
100         downcast<StructureDefinition>(type).~StructureDefinition();
101         break;
102     case Kind::EnumerationDefinition:
103         downcast<EnumerationDefinition>(type).~EnumerationDefinition();
104         break;
105     case Kind::NativeTypeDeclaration:
106         downcast<NativeTypeDeclaration>(type).~NativeTypeDeclaration();
107         break;
108 
109     case Kind::FloatLiteral:
110         downcast<FloatLiteralType>(type).~FloatLiteralType();
111         break;
112     case Kind::IntegerLiteral:
113         downcast<IntegerLiteralType>(type).~IntegerLiteralType();
114         break;



115     case Kind::UnsignedIntegerLiteral:
116         downcast<UnsignedIntegerLiteralType>(type).~UnsignedIntegerLiteralType();
117         break;
118     }
119 }

121 Type& Type::unifyNode()
122 {
123     switch (kind()) {
124     case Kind::TypeReference:
125         return downcast<TypeReference>(*this).unifyNodeImpl();
126     case Kind::Pointer:
127         return downcast<PointerType>(*this).unifyNodeImpl();
128     case Kind::ArrayReference:
129         return downcast<ArrayReferenceType>(*this).unifyNodeImpl();
130     case Kind::Array:
131         return downcast<ArrayType>(*this).unifyNodeImpl();
132 
133     case Kind::TypeDefinition:
134         return downcast<TypeDefinition>(*this).unifyNodeImpl();
135     case Kind::StructureDefinition:
136         return downcast<StructureDefinition>(*this).unifyNodeImpl();
137     case Kind::EnumerationDefinition:
138         return downcast<EnumerationDefinition>(*this).unifyNodeImpl();
139     case Kind::NativeTypeDeclaration:
140         return downcast<NativeTypeDeclaration>(*this).unifyNodeImpl();
141 
142     default:
143         RELEASE_ASSERT_NOT_REACHED();
144     }
145 }

147 bool ResolvableType::canResolve(const Type& type) const
148 {
149     switch (kind()) {
150     case Kind::FloatLiteral:
151         return downcast<FloatLiteralType>(*this).canResolve(type);
152     case Kind::IntegerLiteral:
153         return downcast<IntegerLiteralType>(*this).canResolve(type);


154     case Kind::UnsignedIntegerLiteral:
155         return downcast<UnsignedIntegerLiteralType>(*this).canResolve(type);
156     default:
157         RELEASE_ASSERT_NOT_REACHED();
158     }
159 }

161 unsigned ResolvableType::conversionCost(const UnnamedType& type) const
162 {
163     switch (kind()) {
164     case Kind::FloatLiteral:
165         return downcast<FloatLiteralType>(*this).conversionCost(type);
166     case Kind::IntegerLiteral:
167         return downcast<IntegerLiteralType>(*this).conversionCost(type);


168     case Kind::UnsignedIntegerLiteral:
169         return downcast<UnsignedIntegerLiteralType>(*this).conversionCost(type);
170     default:
171         RELEASE_ASSERT_NOT_REACHED();
172     }
173 }

175 String UnnamedType::toString() const
176 {
177     switch (kind()) {
178     case Kind::TypeReference:
179         return downcast<TypeReference>(*this).toString();
180     case Kind::Pointer:
181         return downcast<PointerType>(*this).toString();
182     case Kind::ArrayReference:
183         return downcast<ArrayReferenceType>(*this).toString();
184     case Kind::Array:
185         return downcast<ArrayType>(*this).toString();
186     default:
187         RELEASE_ASSERT_NOT_REACHED();
188     }
