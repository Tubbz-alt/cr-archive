 26 #pragma once
 27 
 28 #include "JSDestructibleObject.h"
 29 
 30 namespace JSC {
 31 
 32 class JSProxy : public JSDestructibleObject {
 33 public:
 34     using Base = JSDestructibleObject;
 35     static const unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesGetPropertyNames | InterceptsGetOwnPropertySlotByIndexEvenWhenLengthIsNotZero;







 36 
 37     static JSProxy* create(VM& vm, Structure* structure, JSObject* target)
 38     {
 39         JSProxy* proxy = new (NotNull, allocateCell<JSProxy>(vm.heap)) JSProxy(vm, structure);
 40         proxy->finishCreation(vm, target);
 41         return proxy;
 42     }
 43 
 44     static JSProxy* create(VM& vm, Structure* structure)
 45     {
 46         JSProxy* proxy = new (NotNull, allocateCell<JSProxy>(vm.heap)) JSProxy(vm, structure);
 47         proxy->finishCreation(vm);
 48         return proxy;
 49     }
 50 
 51     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype, JSType proxyType)
 52     {
 53         ASSERT(proxyType == ImpureProxyType || proxyType == PureForwardingProxyType);
 54         return Structure::create(vm, globalObject, prototype, TypeInfo(proxyType, StructureFlags), info());
 55     }
 56 
 57     DECLARE_EXPORT_INFO;
 58 
 59     JSObject* target() const { return m_target.get(); }
 60     static ptrdiff_t targetOffset() { return OBJECT_OFFSETOF(JSProxy, m_target); }
 61 
 62     JS_EXPORT_PRIVATE void setTarget(VM&, JSGlobalObject*);
 63 
 64 protected:
 65     JSProxy(VM& vm, Structure* structure)
 66         : Base(vm, structure)
 67     {
 68     }
 69 
 70     void finishCreation(VM& vm)
 71     {
 72         Base::finishCreation(vm);
 73     }
 74 
 75     void finishCreation(VM& vm, JSObject* target)
 76     {
 77         Base::finishCreation(vm);
 78         m_target.set(vm, this, target);
 79     }
 80 
 81     JS_EXPORT_PRIVATE static void visitChildren(JSCell*, SlotVisitor&);
 82 
 83     JS_EXPORT_PRIVATE static String className(const JSObject*, VM&);
 84     JS_EXPORT_PRIVATE static String toStringName(const JSObject*, ExecState*);
 85     JS_EXPORT_PRIVATE static bool getOwnPropertySlot(JSObject*, ExecState*, PropertyName, PropertySlot&);
 86     JS_EXPORT_PRIVATE static bool getOwnPropertySlotByIndex(JSObject*, ExecState*, unsigned, PropertySlot&);
 87     JS_EXPORT_PRIVATE static bool put(JSCell*, ExecState*, PropertyName, JSValue, PutPropertySlot&);
 88     JS_EXPORT_PRIVATE static bool putByIndex(JSCell*, ExecState*, unsigned, JSValue, bool shouldThrow);
 89     JS_EXPORT_PRIVATE static bool deleteProperty(JSCell*, ExecState*, PropertyName);
 90     JS_EXPORT_PRIVATE static bool deletePropertyByIndex(JSCell*, ExecState*, unsigned);
 91     JS_EXPORT_PRIVATE static void getOwnPropertyNames(JSObject*, ExecState*, PropertyNameArray&, EnumerationMode);
 92     JS_EXPORT_PRIVATE static void getPropertyNames(JSObject*, ExecState*, PropertyNameArray&, EnumerationMode);
 93     JS_EXPORT_PRIVATE static uint32_t getEnumerableLength(ExecState*, JSObject*);
 94     JS_EXPORT_PRIVATE static void getStructurePropertyNames(JSObject*, ExecState*, PropertyNameArray&, EnumerationMode);
 95     JS_EXPORT_PRIVATE static void getGenericPropertyNames(JSObject*, ExecState*, PropertyNameArray&, EnumerationMode);
 96     JS_EXPORT_PRIVATE static bool defineOwnProperty(JSObject*, ExecState*, PropertyName, const PropertyDescriptor&, bool shouldThrow);
 97     JS_EXPORT_PRIVATE static bool setPrototype(JSObject*, ExecState*, JSValue, bool shouldThrowIfCantSet);
 98     JS_EXPORT_PRIVATE static JSValue getPrototype(JSObject*, ExecState*);
 99     JS_EXPORT_PRIVATE static bool isExtensible(JSObject*, ExecState*);
100     JS_EXPORT_PRIVATE static bool preventExtensions(JSObject*, ExecState*);
101 
102 private:
103     WriteBarrier<JSObject> m_target;
104 };
105 
106 } // namespace JSC
