26 #pragma once
27 
28 #include "HistoryItem.h"
29 #include <wtf/Forward.h>
30 #include <wtf/ListHashSet.h>
31 #include <wtf/Noncopyable.h>
32 
33 namespace WebCore {
34 
35 class CachedPage;
36 class Frame;
37 class Page;
38 
39 enum class PruningReason { None, ProcessSuspended, MemoryPressure, ReachedMaxSize };
40 
41 class BackForwardCache {
42     WTF_MAKE_NONCOPYABLE(BackForwardCache);
43     WTF_MAKE_FAST_ALLOCATED;
44 public:
45     // Function to obtain the global back/forward cache.
46     WEBCORE_EXPORT static BackForwardCache& singleton();
47 
48     bool canCache(Page&) const;
49 
50     // Used when memory is low to prune some cached pages.
51     WEBCORE_EXPORT void pruneToSizeNow(unsigned maxSize, PruningReason);
52     WEBCORE_EXPORT void setMaxSize(unsigned); // number of pages to cache.
53     unsigned maxSize() const { return m_maxSize; }
54 
55     WEBCORE_EXPORT bool addIfCacheable(HistoryItem&, Page*); // Prunes if maxSize() is exceeded.
56     WEBCORE_EXPORT void remove(HistoryItem&);
57     CachedPage* get(HistoryItem&, Page*);
58     std::unique_ptr<CachedPage> take(HistoryItem&, Page*);
59 
60     void removeAllItemsForPage(Page&);
61 
62     unsigned pageCount() const { return m_items.size(); }
63     WEBCORE_EXPORT unsigned frameCount() const;
64 
65     void markPagesForDeviceOrPageScaleChanged(Page&);
66     void markPagesForContentsSizeChanged(Page&);
67 #if ENABLE(VIDEO_TRACK)
68     void markPagesForCaptionPreferencesChanged();
69 #endif
70 
71 private:
72     BackForwardCache();
73     ~BackForwardCache() = delete; // Make sure nobody accidentally calls delete -- WebCore does not delete singletons.
74 
75     static bool canCachePageContainingThisFrame(Frame&);
76 
77     void prune(PruningReason);
78     void dump() const;
79 
80     ListHashSet<RefPtr<HistoryItem>> m_items;
81     unsigned m_maxSize {0};
82 
83 #if ASSERT_ENABLED
84     bool m_isInRemoveAllItemsForPage { false };
85 #endif
86 
87     friend class WTF::NeverDestroyed<BackForwardCache>;
88 };
89 
90 } // namespace WebCore
