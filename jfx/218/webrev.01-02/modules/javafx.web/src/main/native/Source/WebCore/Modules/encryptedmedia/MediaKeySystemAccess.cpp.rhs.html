 57 void MediaKeySystemAccess::createMediaKeys(Ref<DeferredPromise>&& promise)
 58 {
 59     // https://w3c.github.io/encrypted-media/#dom-mediakeysystemaccess-createmediakeys
 60     // W3C Editor's Draft 09 November 2016
 61 
 62     // When this method is invoked, the user agent must run the following steps:
 63     // 1. Let promise be a new promise.
 64     // 2. Run the following steps in parallel:
 65     m_taskQueue.enqueueTask([this, promise = WTFMove(promise)] () mutable {
 66         // 2.1. Let configuration be the value of this object's configuration value.
 67         // 2.2. Let use distinctive identifier be true if the value of configuration's distinctiveIdentifier member is "required" and false otherwise.
 68         bool useDistinctiveIdentifier = m_configuration->distinctiveIdentifier == MediaKeysRequirement::Required;
 69 
 70         // 2.3. Let persistent state allowed be true if the value of configuration's persistentState member is "required" and false otherwise.
 71         bool persistentStateAllowed = m_configuration->persistentState == MediaKeysRequirement::Required;
 72 
 73         // 2.4. Load and initialize the Key System implementation represented by this object's cdm implementation value if necessary.
 74         m_implementation->loadAndInitialize();
 75 
 76         // 2.5. Let instance be a new instance of the Key System implementation represented by this object's cdm implementation value.
 77         auto instance = m_implementation->createInstance();
 78         if (!instance) {
 79             promise->reject(InvalidStateError);
 80             return;
 81         }
 82 
 83         // 2.6. Initialize instance to enable, disable and/or select Key System features using configuration.
 84         if (instance->initializeWithConfiguration(*m_configuration) == CDMInstance::Failed) {
 85             promise->reject(NotAllowedError);
 86             return;
 87         }
 88 
 89         // 2.7. If use distinctive identifier is false, prevent instance from using Distinctive Identifier(s) and Distinctive Permanent Identifier(s).
 90         if (!useDistinctiveIdentifier && instance->setDistinctiveIdentifiersAllowed(false) == CDMInstance::Failed) {
 91             promise->reject(NotAllowedError);
 92             return;
 93         }
 94 
 95         // 2.8. If persistent state allowed is false, prevent instance from persisting any state related to the application or origin of this object's Document.
 96         if (!persistentStateAllowed && instance->setPersistentStateAllowed(false) == CDMInstance::Failed) {
 97             promise->reject(NotAllowedError);
 98             return;
 99         }
100 
101         // 2.9. If any of the preceding steps failed, reject promise with a new DOMException whose name is the appropriate error name.
102         // 2.10. Let media keys be a new MediaKeys object, and initialize it as follows:
103         // 2.10.1. Let the use distinctive identifier value be use distinctive identifier.
104         // 2.10.2. Let the persistent state allowed value be persistent state allowed.
105         // 2.10.3. Let the supported session types value be be the value of configuration's sessionTypes member.
106         // 2.10.4. Let the cdm implementation value be this object's cdm implementation value.
107         // 2.10.5. Let the cdm instance value be instance.
108         auto mediaKeys = MediaKeys::create(useDistinctiveIdentifier, persistentStateAllowed, m_configuration->sessionTypes, m_implementation.copyRef(), instance.releaseNonNull());
109 
110         // 2.11. Resolve promise with media keys.
111         promise->resolveWithNewlyCreated<IDLInterface<MediaKeys>>(WTFMove(mediaKeys));
112     });
113 
114     // 3. Return promise.
115 }
