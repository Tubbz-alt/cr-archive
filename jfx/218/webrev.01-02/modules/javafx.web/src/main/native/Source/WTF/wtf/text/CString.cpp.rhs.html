 36 DEFINE_ALLOCATOR_WITH_HEAP_IDENTIFIER(CStringBuffer);
 37 
 38 Ref<CStringBuffer> CStringBuffer::createUninitialized(size_t length)
 39 {
 40     RELEASE_ASSERT(length < (std::numeric_limits<unsigned>::max() - sizeof(CStringBuffer)));
 41 
 42     // The +1 is for the terminating null character.
 43     size_t size = sizeof(CStringBuffer) + length + 1;
 44     CStringBuffer* stringBuffer = static_cast<CStringBuffer*>(CStringBufferMalloc::malloc(size));
 45     return adoptRef(*new (NotNull, stringBuffer) CStringBuffer(length));
 46 }
 47 
 48 CString::CString(const char* str)
 49 {
 50     if (!str)
 51         return;
 52 
 53     init(str, strlen(str));
 54 }
 55 
 56 CString::CString(const char* str, size_t length)
 57 {
 58     if (!str) {
 59         ASSERT(!length);
 60         return;
 61     }
 62 
 63     init(str, length);
 64 }
 65 
 66 void CString::init(const char* str, size_t length)
 67 {
 68     ASSERT(str);
 69 
 70     m_buffer = CStringBuffer::createUninitialized(length);
 71     memcpy(m_buffer->mutableData(), str, length);
 72     m_buffer->mutableData()[length] = '\0';
 73 }
 74 
 75 char* CString::mutableData()
 76 {
 77     copyBufferIfNeeded();
 78     if (!m_buffer)
 79         return 0;
 80     return m_buffer->mutableData();
 81 }
 82 
 83 CString CString::newUninitialized(size_t length, char*& characterBuffer)
 84 {
 85     CString result;
 86     result.m_buffer = CStringBuffer::createUninitialized(length);
 87     char* bytes = result.m_buffer->mutableData();
 88     bytes[length] = '\0';
 89     characterBuffer = bytes;
 90     return result;
 91 }
 92 
 93 void CString::copyBufferIfNeeded()
 94 {
 95     if (!m_buffer || m_buffer->hasOneRef())
 96         return;
 97 
 98     RefPtr<CStringBuffer> buffer = WTFMove(m_buffer);
 99     size_t length = buffer->length();
100     m_buffer = CStringBuffer::createUninitialized(length);
101     memcpy(m_buffer->mutableData(), buffer->data(), length + 1);
102 }
103 
104 bool CString::isSafeToSendToAnotherThread() const
105 {
106     return !m_buffer || m_buffer->hasOneRef();
107 }
108 
109 bool operator==(const CString& a, const CString& b)
110 {
111     if (a.isNull() != b.isNull())
112         return false;
113     if (a.length() != b.length())
114         return false;
115     return !memcmp(a.data(), b.data(), a.length());
116 }
117 
118 bool operator==(const CString& a, const char* b)
119 {
120     if (a.isNull() != !b)
121         return false;
122     if (!b)
123         return true;
124     return !strcmp(a.data(), b);
125 }
126 
127 unsigned CString::hash() const
128 {
129     if (isNull())
130         return 0;
131     StringHasher hasher;
132     for (const char* ptr = data(); *ptr; ++ptr)
133         hasher.addCharacter(*ptr);
134     return hasher.hash();
135 }
136 
137 bool operator<(const CString& a, const CString& b)
138 {
139     if (a.isNull())
140         return !b.isNull();
141     if (b.isNull())
142         return false;
143     return strcmp(a.data(), b.data()) < 0;
144 }
145 
146 bool CStringHash::equal(const CString& a, const CString& b)
147 {
148     if (a.isHashTableDeletedValue())
149         return b.isHashTableDeletedValue();
150     if (b.isHashTableDeletedValue())
151         return false;
152     return a == b;
153 }
