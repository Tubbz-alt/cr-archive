40 // InlineFormattingState holds the state for a particular inline formatting context tree.
41 class InlineFormattingState : public FormattingState {
42     WTF_MAKE_ISO_ALLOCATED(InlineFormattingState);
43 public:
44     InlineFormattingState(Ref<FloatingState>&&, LayoutState&);
45     ~InlineFormattingState();
46 
47     InlineItems& inlineItems() { return m_inlineItems; }
48     const InlineItems& inlineItems() const { return m_inlineItems; }
49     void addInlineItem(InlineItem&& inlineItem) { m_inlineItems.append(WTFMove(inlineItem)); }


50 
51     const Display::InlineContent* displayInlineContent() const { return m_displayInlineContent.get(); }
52     Display::InlineContent& ensureDisplayInlineContent();
53     void clearDisplayInlineContent() { m_displayInlineContent = nullptr; }
54 
55 private:
56     // Cacheable input to line layout.
57     InlineItems m_inlineItems;
58 
59     RefPtr<Display::InlineContent> m_displayInlineContent;
60 };
61 
62 inline Display::InlineContent& InlineFormattingState::ensureDisplayInlineContent()
63 {
64     if (!m_displayInlineContent)
65         m_displayInlineContent = adoptRef(*new Display::InlineContent);
66     return *m_displayInlineContent;
67 }
68 
69 }
