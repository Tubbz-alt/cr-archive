40 VMEntryScope::VMEntryScope(VM& vm, JSGlobalObject* globalObject)
41     : m_vm(vm)
42     , m_globalObject(globalObject)
43 {
44     ASSERT(!DisallowVMReentry::isInEffectOnCurrentThread());
45     if (!vm.entryScope) {
46         vm.entryScope = this;
47 
48         // Reset the date cache between JS invocations to force the VM to
49         // observe time zone changes.
50         vm.resetDateCache();
51 
52         if (vm.watchdog())
53             vm.watchdog()->enteredVM();
54 
55 #if ENABLE(SAMPLING_PROFILER)
56         if (SamplingProfiler* samplingProfiler = vm.samplingProfiler())
57             samplingProfiler->noticeVMEntry();
58 #endif
59         if (Options::useTracePoints())
60             tracePoint(VMEntryScopeStart);
61     }
62 
63     vm.clearLastException();
64 }
65 
66 void VMEntryScope::addDidPopListener(Function<void ()>&& listener)
67 {
68     m_didPopListeners.append(WTFMove(listener));
69 }
70 
71 VMEntryScope::~VMEntryScope()
72 {
73     if (m_vm.entryScope != this)
74         return;
75 
76     if (Options::useTracePoints())
77         tracePoint(VMEntryScopeEnd);
78 
79     if (m_vm.watchdog())
80         m_vm.watchdog()->exitedVM();
81 
82     m_vm.entryScope = nullptr;
83 
84     for (auto& listener : m_didPopListeners)
85         listener();
86 
87     m_vm.clearScratchBuffers();
88 }
