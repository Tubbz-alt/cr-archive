26 #pragma once
27 

28 #include "AnimationPlaybackEventInit.h"
29 #include "Event.h"
30 #include <wtf/Markable.h>
31 
32 namespace WebCore {
33 
34 class AnimationPlaybackEvent final : public Event {

35 public:
36     static Ref<AnimationPlaybackEvent> create(const AtomString& type, Optional<Seconds> currentTime, Optional<Seconds> timelineTime)
37     {
38         return adoptRef(*new AnimationPlaybackEvent(type, currentTime, timelineTime));
39     }
40 
41     static Ref<AnimationPlaybackEvent> create(const AtomString& type, const AnimationPlaybackEventInit& initializer, IsTrusted isTrusted = IsTrusted::No)
42     {
43         return adoptRef(*new AnimationPlaybackEvent(type, initializer, isTrusted));
44     }
45 
46     virtual ~AnimationPlaybackEvent();
47 


48     Optional<double> bindingsCurrentTime() const;
49     Optional<Seconds> currentTime() const { return m_currentTime; }
50     Optional<double> bindingsTimelineTime() const;
51     Optional<Seconds> timelineTime() const { return m_timelineTime; }
52 
53     EventInterface eventInterface() const override { return AnimationPlaybackEventInterfaceType; }
54 
55 private:
56     AnimationPlaybackEvent(const AtomString&, Optional<Seconds>, Optional<Seconds>);
57     AnimationPlaybackEvent(const AtomString&, const AnimationPlaybackEventInit&, IsTrusted);
58 
59     Markable<Seconds, Seconds::MarkableTraits> m_currentTime;
60     Markable<Seconds, Seconds::MarkableTraits> m_timelineTime;
61 };
62 
63 }
