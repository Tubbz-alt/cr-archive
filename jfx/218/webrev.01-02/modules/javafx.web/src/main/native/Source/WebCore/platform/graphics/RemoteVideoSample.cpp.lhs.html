45 std::unique_ptr<RemoteVideoSample> RemoteVideoSample::create(MediaSample&& sample)











































46 {
47     ASSERT(sample.platformSample().type == PlatformSample::CMSampleBufferType);
48 
49     auto imageBuffer = CMSampleBufferGetImageBuffer(sample.platformSample().sample.cmSampleBuffer);
50     if (!imageBuffer) {
51         RELEASE_LOG_ERROR(Media, "RemoteVideoSample::create: CMSampleBufferGetImageBuffer returned nullptr");
52         return nullptr;
53     }
54 






















55     auto surface = CVPixelBufferGetIOSurface(imageBuffer);
56     if (!surface) {
57         RELEASE_LOG_ERROR(Media, "RemoteVideoSample::create: CVPixelBufferGetIOSurface returned nullptr");
58         return nullptr;
59     }
60 
61     return std::unique_ptr<RemoteVideoSample>(new RemoteVideoSample(surface, sRGBColorSpaceRef(), sample.presentationTime(), sample.videoRotation(), sample.videoMirrored()));
62 }
63 
64 RemoteVideoSample::RemoteVideoSample(IOSurfaceRef surface, CGColorSpaceRef colorSpace, MediaTime&& time, MediaSample::VideoRotation rotation, bool mirrored)
65     : m_ioSurface(WebCore::IOSurface::createFromSurface(surface, colorSpace))
66     , m_rotation(rotation)
67     , m_time(WTFMove(time))
68     , m_videoFormat(IOSurfaceGetPixelFormat(surface))
69     , m_size(IntSize(IOSurfaceGetWidth(surface), IOSurfaceGetHeight(surface)))
70     , m_mirrored(mirrored)
71 {
72 }
73 
74 IOSurfaceRef RemoteVideoSample::surface()
75 {
76     if (!m_ioSurface && m_sendRight)
77         m_ioSurface = WebCore::IOSurface::createFromSendRight(WTFMove(m_sendRight), sRGBColorSpaceRef());
78 
79     return m_ioSurface ? m_ioSurface->surface() : nullptr;
80 }
