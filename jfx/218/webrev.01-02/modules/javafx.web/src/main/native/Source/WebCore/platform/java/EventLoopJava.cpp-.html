26 #include "config.h"
27 #include "EventLoop.h"
28 
29 #include "PlatformJavaClasses.h"
30 
31 namespace EventLoopJava {
32 
33 static JGClass eventLoopClass;
34 static jmethodID cycleMethod;
35 
36 static void initRefs(JNIEnv* env)
37 {
38     if (!eventLoopClass) {
39         eventLoopClass = JLClass(env->FindClass(
40                 "com/sun/webkit/EventLoop"));
41         ASSERT(eventLoopClass);
42 
43         cycleMethod = env->GetStaticMethodID(eventLoopClass, "fwkCycle", "()V");
44         ASSERT(cycleMethod);
45     }
46 }
47 }
48 
49 namespace WebCore {
50 
51 void EventLoop::cycle()
52 {
53     using namespace EventLoopJava;
54     WC_GETJAVAENV_CHKRET(env);
55     initRefs(env);
56 
57     env->CallStaticVoidMethod(eventLoopClass, cycleMethod);
58     WTF::CheckAndClearException(env);
59 }
60 
61 } // namespace WebCore
