38 const PinnedRegisterInfo& PinnedRegisterInfo::get()
39 {
40     static LazyNeverDestroyed<PinnedRegisterInfo> staticPinnedRegisterInfo;
41     static std::once_flag staticPinnedRegisterInfoFlag;
42     std::call_once(staticPinnedRegisterInfoFlag, [] () {
43         unsigned numberOfPinnedRegisters = 2;
44         if (!Context::useFastTLS())
45             ++numberOfPinnedRegisters;
46         GPRReg baseMemoryPointer = GPRInfo::regCS3;
47         GPRReg sizeRegister = GPRInfo::regCS4;
48         GPRReg wasmContextInstancePointer = InvalidGPRReg;
49         if (!Context::useFastTLS())
50             wasmContextInstancePointer = GPRInfo::regCS0;
51 
52         staticPinnedRegisterInfo.construct(sizeRegister, baseMemoryPointer, wasmContextInstancePointer);
53     });
54 
55     return staticPinnedRegisterInfo.get();
56 }

58 PinnedRegisterInfo::PinnedRegisterInfo(GPRReg sizeRegister, GPRReg baseMemoryPointer, GPRReg wasmContextInstancePointer)
59     : sizeRegister(sizeRegister)
60     , baseMemoryPointer(baseMemoryPointer)
61     , wasmContextInstancePointer(wasmContextInstancePointer)

63 }

65 MemoryInformation::MemoryInformation(PageCount initial, PageCount maximum, bool isImport)
66     : m_initial(initial)
67     , m_maximum(maximum)
68     , m_isImport(isImport)
69 {
70     RELEASE_ASSERT(!!m_initial);
71     RELEASE_ASSERT(!m_maximum || m_maximum >= m_initial);
72     ASSERT(!!*this);
73 }
