112 ExceptionOr<void> MediaRecorder::startRecording(Optional<int> timeslice)
113 {
114     UNUSED_PARAM(timeslice);
115     if (state() != RecordingState::Inactive)
116         return Exception { InvalidStateError, "The MediaRecorder's state must be inactive in order to start recording"_s };
117 
118     for (auto& track : m_tracks)
119         track->addObserver(*this);
120 
121     m_state = RecordingState::Recording;
122     return { };
123 }

125 ExceptionOr<void> MediaRecorder::stopRecording()
126 {
127     if (state() == RecordingState::Inactive)
128         return Exception { InvalidStateError, "The MediaRecorder's state cannot be inactive"_s };
129 
130     scheduleDeferredTask([this] {
131         if (!m_isActive || state() == RecordingState::Inactive)
132             return;
133 
134         stopRecordingInternal();
135         ASSERT(m_state == RecordingState::Inactive);
136         dispatchEvent(BlobEvent::create(eventNames().dataavailableEvent, Event::CanBubble::No, Event::IsCancelable::No, createRecordingDataBlob()));




















137         if (!m_isActive)
138             return;
139         dispatchEvent(Event::create(eventNames().stopEvent, Event::CanBubble::No, Event::IsCancelable::No));

140     });
141     return { };
142 }

144 void MediaRecorder::stopRecordingInternal()
145 {
146     if (state() != RecordingState::Recording)
147         return;
148 
149     for (auto& track : m_tracks)
150         track->removeObserver(*this);
151 
152     m_state = RecordingState::Inactive;
153     m_private->stopRecording();
154 }

156 Ref<Blob> MediaRecorder::createRecordingDataBlob()
157 {
158     auto data = m_private->fetchData();
159     if (!data)
160         return Blob::create(scriptExecutionContext()->sessionID());
161     return Blob::create(scriptExecutionContext()->sessionID(), *data, m_private->mimeType());
162 }

196 void MediaRecorder::scheduleDeferredTask(Function<void()>&& function)
197 {
198     ASSERT(function);
199     auto* scriptExecutionContext = this->scriptExecutionContext();
200     if (!scriptExecutionContext)
201         return;
202 
203     scriptExecutionContext->postTask([protectedThis = makeRef(*this), function = WTFMove(function)] (auto&) {
204         function();
205     });

 49 ExceptionOr<Ref<MediaRecorder>> MediaRecorder::create(Document& document, Ref<MediaStream>&& stream, Options&& options)
 50 {
 51     auto privateInstance = MediaRecorder::createMediaRecorderPrivate(document, stream->privateStream());
 52     if (!privateInstance)
 53         return Exception { NotSupportedError, "The MediaRecorder is unsupported on this platform"_s };
 54     auto recorder = adoptRef(*new MediaRecorder(document, WTFMove(stream), WTFMove(privateInstance), WTFMove(options)));
 55     recorder->suspendIfNeeded();
 56     recorder->m_private->setErrorCallback([recorder = recorder.copyRef()](auto&& exception) mutable {
 57         recorder->dispatchError(WTFMove(*exception));
 58     });
 59     return recorder;
 60 }

114 void MediaRecorder::suspend(ReasonForSuspension reason)
115 {
116     if (reason != ReasonForSuspension::BackForwardCache)
117         return;
118 
119     if (!m_isActive || state() == RecordingState::Inactive)
120         return;
121 
122     stopRecordingInternal();
123 
124     scheduleDeferredTask([this] {
125         dispatchEvent(MediaRecorderErrorEvent::create(eventNames().errorEvent, Exception { UnknownError, "MediaStream recording was interrupted"_s }));
126     });
127 }

147 ExceptionOr<void> MediaRecorder::stopRecording()
148 {
149     if (state() == RecordingState::Inactive)
150         return Exception { InvalidStateError, "The MediaRecorder's state cannot be inactive"_s };
151 
152     scheduleDeferredTask([this] {
153         if (!m_isActive || state() == RecordingState::Inactive)
154             return;
155 
156         stopRecordingInternal();
157         ASSERT(m_state == RecordingState::Inactive);
158         m_private->fetchData([this, protectedThis = makeRef(*this)](auto&& buffer, auto& mimeType) {
159             if (!m_isActive)
160                 return;
161 
162             dispatchEvent(BlobEvent::create(eventNames().dataavailableEvent, Event::CanBubble::No, Event::IsCancelable::No, buffer ? Blob::create(buffer.releaseNonNull(), mimeType) : Blob::create()));
163 
164             if (!m_isActive)
165                 return;
166 
167             dispatchEvent(Event::create(eventNames().stopEvent, Event::CanBubble::No, Event::IsCancelable::No));
168         });
169     });
170     return { };
171 }

237 void MediaRecorder::scheduleDeferredTask(Function<void()>&& function)
238 {
239     ASSERT(function);
240     auto* document = this->document();
241     if (!document)
242         return;
243 
244     document->eventLoop().queueTask(TaskSource::Networking, [pendingActivity = makePendingActivity(*this), function = WTFMove(function)] {
245         function();
246     });
