 28 #include "JSCJSValue.h"
 29 #include <wtf/PrintStream.h>
 30 
 31 namespace JSC { namespace Profiler {
 32 
 33 class UID {
 34 public:
 35     UID()
 36         : m_uid(0)
 37     {
 38     }
 39 
 40     static UID fromInt(uint64_t value)
 41     {
 42         UID result;
 43         result.m_uid = value;
 44         return result;
 45     }
 46 
 47     UID(WTF::HashTableDeletedValueType)
 48         : m_uid(std::numeric_limits<uint64_t>::max())
 49     {
 50     }
 51 
 52     static UID create();
 53 
 54     uint64_t toInt() const
 55     {
 56         return m_uid;
 57     }
 58 
 59     bool operator==(const UID& other) const
 60     {
 61         return m_uid == other.m_uid;
 62     }
 63 
 64     bool operator!=(const UID& other) const
 65     {
 66         return !(*this == other);
 67     }
 68 
 69     explicit operator bool() const
 70     {
 71         return *this != UID();
 72     }
 73 
 74     bool isHashTableDeletedValue() const
 75     {
 76         return *this != UID(WTF::HashTableDeletedValue);
 77     }
 78 
 79     unsigned hash() const
 80     {
 81         return IntHash<uint64_t>::hash(m_uid);
 82     }
 83 
 84     void dump(PrintStream&) const;
 85     JSValue toJS(JSGlobalObject*) const;
 86 
 87 private:
 88     uint64_t m_uid;
 89 };
 90 
 91 struct UIDHash {
 92     static unsigned hash(const UID& key) { return key.hash(); }
 93     static bool equal(const UID& a, const UID& b) { return a == b; }
 94     static constexpr bool safeToCompareToEmptyOrDeleted = true;
 95 };
 96 
 97 } } // namespace JSC::Profiler
 98 
 99 namespace WTF {

103     typedef JSC::Profiler::UIDHash Hash;
104 };
