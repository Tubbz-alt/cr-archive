 49 class TransactionOperation : public ThreadSafeRefCounted<TransactionOperation> {
 50     friend IDBRequestData::IDBRequestData(TransactionOperation&);
 51 public:
 52     virtual ~TransactionOperation()
 53     {
 54         ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
 55     }
 56 
 57     void perform()
 58     {
 59         ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
 60         ASSERT(m_performFunction);
 61         m_performFunction();
 62         m_performFunction = { };
 63     }
 64 
 65     void transitionToCompleteOnThisThread(const IDBResultData& data)
 66     {
 67         ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
 68         m_transaction->operationCompletedOnServer(data, *this);
 69     }
 70 
 71     void transitionToComplete(const IDBResultData& data, RefPtr<TransactionOperation>&& lastRef)
 72     {
 73         ASSERT(isMainThread());
 74 
 75         if (canCurrentThreadAccessThreadLocalData(originThread()))
 76             transitionToCompleteOnThisThread(data);
 77         else {
 78             m_transaction->performCallbackOnOriginThread(*this, &TransactionOperation::transitionToCompleteOnThisThread, data);
 79             m_transaction->callFunctionOnOriginThread([lastRef = WTFMove(lastRef)]() {
 80             });
 81         }
 82     }
 83 
 84     void doComplete(const IDBResultData& data)
 85     {
 86         ASSERT(canCurrentThreadAccessThreadLocalData(originThread()));
 87 
 88         if (m_performFunction)
 89             m_performFunction = { };
 90 
 91         // Due to race conditions between the server sending an "operation complete" message and the client
 92         // forcefully aborting an operation, it's unavoidable that this method might be called twice.
 93         // It's okay to handle that gracefully with an early return.
 94         if (m_didComplete)
 95             return;
 96         m_didComplete = true;
 97 
 98         if (m_completeFunction) {
 99             m_completeFunction(data);
100             // m_completeFunction should not hold ref to this TransactionOperation after its execution.
101             m_completeFunction = { };
102         }
103         m_transaction->operationCompletedOnClient(*this);
104     }
105 
106     const IDBResourceIdentifier& identifier() const { return m_identifier; }
107 
108     Thread& originThread() const { return m_originThread.get(); }
109 
110     IDBRequest* idbRequest() { return m_idbRequest.get(); }
111 
112     bool nextRequestCanGoToServer() const { return m_nextRequestCanGoToServer && m_idbRequest; }
113     void setNextRequestCanGoToServer(bool nextRequestCanGoToServer) { m_nextRequestCanGoToServer = nextRequestCanGoToServer; }
114 
115     uint64_t operationID() const { return m_operationID; }
116 
117 protected:
118     TransactionOperation(IDBTransaction& transaction)
119         : m_transaction(transaction)
120         , m_identifier(transaction.connectionProxy())
121         , m_operationID(transaction.generateOperationID())
122     {
123     }
124 
125     TransactionOperation(IDBTransaction&, IDBRequest&);
126 
127     Ref<IDBTransaction> m_transaction;
128     IDBResourceIdentifier m_identifier;
129     uint64_t m_objectStoreIdentifier { 0 };
130     uint64_t m_indexIdentifier { 0 };
131     std::unique_ptr<IDBResourceIdentifier> m_cursorIdentifier;
132     IndexedDB::IndexRecordType m_indexRecordType;
133     Function<void()> m_performFunction;
134     Function<void(const IDBResultData&)> m_completeFunction;
135 
136 private:
137     IDBResourceIdentifier transactionIdentifier() const { return m_transaction->info().identifier(); }
138     uint64_t objectStoreIdentifier() const { return m_objectStoreIdentifier; }
139     uint64_t indexIdentifier() const { return m_indexIdentifier; }
140     IDBResourceIdentifier* cursorIdentifier() const { return m_cursorIdentifier.get(); }
141     IDBTransaction& transaction() { return m_transaction.get(); }
142     IndexedDB::IndexRecordType indexRecordType() const { return m_indexRecordType; }
143 
144     Ref<Thread> m_originThread { Thread::current() };
145     RefPtr<IDBRequest> m_idbRequest;
146     bool m_nextRequestCanGoToServer { true };
147     bool m_didComplete { false };
148 
149     uint64_t m_operationID { 0 };
150 };
151 
152 class TransactionOperationImpl final : public TransactionOperation {
153 public:
154     template<typename... Args> static Ref<TransactionOperationImpl> create(Args&&... args) { return adoptRef(*new TransactionOperationImpl(std::forward<Args>(args)...)); }
155 private:
156     TransactionOperationImpl(IDBTransaction& transaction, Function<void(const IDBResultData&)> completeMethod, Function<void(TransactionOperation&)> performMethod)
157         : TransactionOperation(transaction)
158     {
159         ASSERT(performMethod);
160         m_performFunction = [protectedThis = makeRef(*this), performMethod = WTFMove(performMethod)] {
161             performMethod(protectedThis.get());
162         };
163 
164         if (completeMethod) {
165             m_completeFunction = [protectedThis = makeRef(*this), completeMethod = WTFMove(completeMethod)] (const IDBResultData& resultData) {
166                 completeMethod(resultData);
167             };
168         }
169     }
170 
171     TransactionOperationImpl(IDBTransaction& transaction, IDBRequest& request, Function<void(const IDBResultData&)> completeMethod, Function<void(TransactionOperation&)> performMethod)
172         : TransactionOperation(transaction, request)
173     {
174         ASSERT(performMethod);
175         m_performFunction = [protectedThis = makeRef(*this), performMethod = WTFMove(performMethod)] {
176             performMethod(protectedThis.get());
177         };
178 
179         if (completeMethod) {
180             m_completeFunction = [protectedThis = makeRef(*this), completeMethod = WTFMove(completeMethod)] (const IDBResultData& resultData) {
181                 completeMethod(resultData);
182             };
183         }
184     }
185 };
