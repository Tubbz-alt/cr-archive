38 JSArrayBuffer* SimpleTypedArrayController::toJS(
39     ExecState* exec, JSGlobalObject* globalObject, ArrayBuffer* native)
40 {
41     if (JSArrayBuffer* buffer = native->m_wrapper.get())
42         return buffer;
43 
44     // The JSArrayBuffer::create function will register the wrapper in finishCreation.
45     JSArrayBuffer* result = JSArrayBuffer::create(
46         exec->vm(), globalObject->arrayBufferStructure(native->sharingMode()), native);
47     return result;
48 }
49 
50 void SimpleTypedArrayController::registerWrapper(JSGlobalObject*, ArrayBuffer* native, JSArrayBuffer* wrapper)
51 {
52     ASSERT(!native->m_wrapper);
53     native->m_wrapper = Weak<JSArrayBuffer>(wrapper, &m_owner);
54 }
55 
56 bool SimpleTypedArrayController::isAtomicsWaitAllowedOnCurrentThread()
57 {
58     return true;
59 }
60 
61 bool SimpleTypedArrayController::JSArrayBufferOwner::isReachableFromOpaqueRoots(JSC::Handle<JSC::Unknown> handle, void*, JSC::SlotVisitor& visitor, const char** reason)
62 {
63     if (UNLIKELY(reason))
64         *reason = "JSArrayBuffer is opaque root";
65     auto& wrapper = *JSC::jsCast<JSC::JSArrayBuffer*>(handle.slot()->asCell());
66     return visitor.containsOpaqueRoot(wrapper.impl());
67 }
