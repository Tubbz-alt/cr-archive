40 void CommandLineAPIModule::injectIfNeeded(InjectedScriptManager* injectedScriptManager, const InjectedScript& injectedScript)
41 {
42     CommandLineAPIModule module;
43     module.ensureInjected(injectedScriptManager, injectedScript);
44 }
45 
46 CommandLineAPIModule::CommandLineAPIModule()
47     : InjectedScriptModule("CommandLineAPI"_s)
48 {
49 }
50 
51 String CommandLineAPIModule::source() const
52 {
53     return StringImpl::createWithoutCopying(CommandLineAPIModuleSource_js, sizeof(CommandLineAPIModuleSource_js));
54 }
55 
56 JSValue CommandLineAPIModule::host(InjectedScriptManager* injectedScriptManager, JSGlobalObject* lexicalGlobalObject) const
57 {
58     // CommandLineAPIModule should only ever be used by a WebInjectedScriptManager.
59     WebInjectedScriptManager* pageInjectedScriptManager = static_cast<WebInjectedScriptManager*>(injectedScriptManager);
60     ASSERT(pageInjectedScriptManager->commandLineAPIHost());
61 
62     JSDOMGlobalObject* globalObject = jsCast<JSDOMGlobalObject*>(lexicalGlobalObject);
63     return pageInjectedScriptManager->commandLineAPIHost()->wrapper(lexicalGlobalObject, globalObject);
64 }
