26 #pragma once
27 
28 #include "JSObject.h"
29 
30 namespace JSC {
31 
32 class JSONObject final : public JSNonFinalObject {
33 public:
34     typedef JSNonFinalObject Base;
35     static const unsigned StructureFlags = Base::StructureFlags | HasStaticPropertyTable;
36 
37     static JSONObject* create(VM& vm, Structure* structure)
38     {
39         JSONObject* object = new (NotNull, allocateCell<JSONObject>(vm.heap)) JSONObject(vm, structure);
40         object->finishCreation(vm);
41         return object;
42     }
43 
44     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
45     {
46         return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
47     }
48 
49     DECLARE_INFO;
50 
51 protected:
52     void finishCreation(VM&);
53 
54 private:
55     JSONObject(VM&, Structure*);
56 };
57 
58 JS_EXPORT_PRIVATE JSValue JSONParse(ExecState*, const String&);
59 JS_EXPORT_PRIVATE String JSONStringify(ExecState*, JSValue, JSValue space);
60 JS_EXPORT_PRIVATE String JSONStringify(ExecState*, JSValue, unsigned indent);
61 
62 } // namespace JSC
