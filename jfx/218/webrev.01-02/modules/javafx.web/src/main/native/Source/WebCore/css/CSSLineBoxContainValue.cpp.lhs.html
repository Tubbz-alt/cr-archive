31 namespace WebCore {
32 
33 CSSLineBoxContainValue::CSSLineBoxContainValue(unsigned value)
34     : CSSValue(LineBoxContainClass)
35     , m_value(value)
36 {
37 }
38 
39 String CSSLineBoxContainValue::customCSSText() const
40 {
41     StringBuilder text;
42 
43     if (m_value & LineBoxContainBlock)
44         text.appendLiteral("block");
45     if (m_value & LineBoxContainInline) {

46         if (!text.isEmpty())
47             text.append(' ');
48         text.appendLiteral("inline");
49     }
50     if (m_value & LineBoxContainFont) {
51         if (!text.isEmpty())
52             text.append(' ');
53         text.appendLiteral("font");
54     }
55     if (m_value & LineBoxContainGlyphs) {
56         if (!text.isEmpty())
57             text.append(' ');
58         text.appendLiteral("glyphs");
59     }
60     if (m_value & LineBoxContainReplaced) {
61         if (!text.isEmpty())
62             text.append(' ');
63         text.appendLiteral("replaced");
64     }
65     if (m_value & LineBoxContainInlineBox) {
66         if (!text.isEmpty())
67             text.append(' ');
68         text.appendLiteral("inline-box");
69     }
70     if (m_value & LineBoxContainInitialLetter) {
71         if (!text.isEmpty())
72             text.append(' ');
73         text.appendLiteral("initial-letter");
74     }
75 
76     return text.toString();
77 }
