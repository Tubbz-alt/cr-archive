39 JSValue RegExpGlobalData::getBackref(JSGlobalObject* globalObject, unsigned i)
40 {
41     VM& vm = globalObject->vm();
42     auto scope = DECLARE_THROW_SCOPE(vm);
43 
44     JSArray* array = m_cachedResult.lastResult(globalObject, globalObject);
45     RETURN_IF_EXCEPTION(scope, { });
46 
47     if (i < array->length()) {
48         JSValue result = JSValue(array).get(globalObject, i);
49         RETURN_IF_EXCEPTION(scope, { });
50         ASSERT(result.isString() || result.isUndefined());
51         if (!result.isUndefined())
52             return result;
53     }
54     return jsEmptyString(vm);
55 }
56 
57 JSValue RegExpGlobalData::getLastParen(JSGlobalObject* globalObject)
58 {
59     VM& vm = globalObject->vm();
60     auto scope = DECLARE_THROW_SCOPE(vm);
61 
62     JSArray* array = m_cachedResult.lastResult(globalObject, globalObject);
63     RETURN_IF_EXCEPTION(scope, { });
64 
65     unsigned length = array->length();
66     if (length > 1) {
67         JSValue result = JSValue(array).get(globalObject, length - 1);
68         RETURN_IF_EXCEPTION(scope, { });
69         ASSERT(result.isString() || result.isUndefined());
70         if (!result.isUndefined())
71             return result;
72     }
73     return jsEmptyString(vm);
74 }
