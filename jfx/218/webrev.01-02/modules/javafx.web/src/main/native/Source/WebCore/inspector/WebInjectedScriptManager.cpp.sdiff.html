 72 void WebInjectedScriptManager::discardInjectedScriptsFor(DOMWindow* window)
 73 {
 74     if (m_scriptStateToId.isEmpty())
 75         return;
 76 
 77     Vector<long> idsToRemove;
 78     for (const auto& it : m_idToInjectedScript) {
 79         JSC::JSGlobalObject* lexicalGlobalObject = it.value.globalObject();
 80         // JDK-8164076
 81         if (lexicalGlobalObject == nullptr || window != domWindowFromExecState(lexicalGlobalObject))
 82             continue;
 83         m_scriptStateToId.remove(lexicalGlobalObject);
 84         idsToRemove.append(it.key);
 85     }
 86 
 87     for (auto& id : idsToRemove)
 88         m_idToInjectedScript.remove(id);
 89 
 90     // Now remove script states that have id but no injected script.
 91     Vector<JSC::JSGlobalObject*> scriptStatesToRemove;
 92     for (const auto& it : m_scriptStateToId) {
 93         JSC::JSGlobalObject* lexicalGlobalObject = it.key;
 94         if (window == domWindowFromExecState(lexicalGlobalObject))
 95             scriptStatesToRemove.append(lexicalGlobalObject);
 96     }
 97 
 98     for (auto& lexicalGlobalObject : scriptStatesToRemove)
 99         m_scriptStateToId.remove(lexicalGlobalObject);
