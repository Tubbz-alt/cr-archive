33 namespace WebCore {
34 
35 class ReadableStreamSink;
36 class ReadableStreamSource;
37 
38 class ReadableStream final : public DOMGuarded<JSReadableStream> {
39 public:
40     static Ref<ReadableStream> create(JSDOMGlobalObject& globalObject, JSReadableStream& readableStream) { return adoptRef(*new ReadableStream(globalObject, readableStream)); }
41 
42     static Ref<ReadableStream> create(JSC::ExecState&, RefPtr<ReadableStreamSource>&&);
43 
44     WEBCORE_EXPORT static bool isDisturbed(JSC::ExecState&, JSC::JSValue);
45 
46     std::pair<Ref<ReadableStream>, Ref<ReadableStream>> tee();
47 
48     void lock();
49     void pipeTo(ReadableStreamSink&);
50     bool isLocked() const;
51     bool isDisturbed() const;
52 
53     JSReadableStream* readableStream() const { return guarded(); }
54 
55 protected:
56     ReadableStream(JSDOMGlobalObject& globalObject, JSReadableStream& readableStream) : DOMGuarded<JSReadableStream>(globalObject, readableStream) { }
57 };
58 
59 struct JSReadableStreamWrapperConverter {
60     static RefPtr<ReadableStream> toWrapped(JSC::ExecState& state, JSC::JSValue value)
61     {
62         JSC::VM& vm = state.vm();
63         auto* globalObject = JSC::jsDynamicCast<JSDOMGlobalObject*>(vm, state.lexicalGlobalObject());
64         if (!globalObject)
65             return nullptr;
66 
67         auto* readableStream = JSC::jsDynamicCast<JSReadableStream*>(vm, value);
68         if (!readableStream)
69             return nullptr;
70 
71         return ReadableStream::create(*globalObject, *readableStream);
72     }
73 };
74 
75 template<> struct JSDOMWrapperConverterTraits<ReadableStream> {
76     using WrapperClass = JSReadableStreamWrapperConverter;
77     using ToWrappedReturnType = RefPtr<ReadableStream>;
78     static constexpr bool needsState = true;
79 };
80 
81 inline JSC::JSValue toJS(JSC::ExecState*, JSC::JSGlobalObject*, ReadableStream* stream)
82 {
83     return stream ? stream->readableStream() : JSC::jsUndefined();
84 }
