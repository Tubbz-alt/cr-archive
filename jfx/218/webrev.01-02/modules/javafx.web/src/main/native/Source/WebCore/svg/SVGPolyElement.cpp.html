36 SVGPolyElement::SVGPolyElement(const QualifiedName& tagName, Document& document)
37     : SVGGeometryElement(tagName, document)
38 {
39     static std::once_flag onceFlag;
40     std::call_once(onceFlag, [] {
41         PropertyRegistry::registerProperty<SVGNames::pointsAttr, &SVGPolyElement::m_points>();
42     });
43 }
44 
45 void SVGPolyElement::parseAttribute(const QualifiedName& name, const AtomString& value)
46 {
47     if (name == SVGNames::pointsAttr) {
48         if (!m_points->baseVal()->parse(value))
49             document().accessSVGExtensions().reportError("Problem parsing points=\"" + value + "\"");
50         return;
51     }
52 
53     SVGGeometryElement::parseAttribute(name, value);
54 }
55 
56 void SVGPolyElement::svgAttributeChanged(const QualifiedName& attrName)
57 {
58     if (attrName == SVGNames::pointsAttr) {
59         if (auto* renderer = downcast<RenderSVGPath>(this->renderer())) {
60             InstanceInvalidationGuard guard(*this);
61             renderer->setNeedsShapeUpdate();
62             RenderSVGResource::markForLayoutAndParentResourceInvalidation(*renderer);
63         }
64         return;
65     }
66 
67     SVGGeometryElement::svgAttributeChanged(attrName);
68 }
69 
70 size_t SVGPolyElement::approximateMemoryCost() const
71 {
72     size_t pointsCost = m_points->baseVal()->items().size() * sizeof(FloatPoint);
73     // We need to account for the memory which is allocated by the RenderSVGPath::m_path.
74     return sizeof(*this) + (renderer() ? pointsCost * 2 + sizeof(RenderSVGPath) : pointsCost);
75 }
