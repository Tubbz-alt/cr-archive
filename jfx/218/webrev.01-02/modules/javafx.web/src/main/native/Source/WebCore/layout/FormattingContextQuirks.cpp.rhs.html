37 LayoutUnit FormattingContext::Quirks::heightValueOfNearestContainingBlockWithFixedHeight(const Box& layoutBox)
38 {
39     // In quirks mode, we go and travers the containing block chain to find a block level box with fixed height value, even if it means leaving
40     // the current formatting context. FIXME: surely we need to do some tricks here when block direction support is added.
41     auto& formattingContext = this->formattingContext();
42     auto* containingBlock = layoutBox.containingBlock();
43     LayoutUnit bodyAndDocumentVerticalMarginPaddingAndBorder;
44     while (containingBlock) {
45         auto containingBlockHeight = containingBlock->style().logicalHeight();
46         if (containingBlockHeight.isFixed())
47             return LayoutUnit(containingBlockHeight.value() - bodyAndDocumentVerticalMarginPaddingAndBorder);
48 
49         // If the only fixed value box we find is the ICB, then ignore the body and the document (vertical) margin, padding and border. So much quirkiness.
50         // -and it's totally insane because now we freely travel across formatting context boundaries and computed margins are nonexistent.
51         if (containingBlock->isBodyBox() || containingBlock->isDocumentBox()) {
52             auto& boxGeometry = formattingContext.geometryForBox(*containingBlock, FormattingContext::EscapeReason::FindFixedHeightAncestorQuirk);
53 
54             auto& containingBlockDisplayBox = formattingContext.geometryForBox(*containingBlock->containingBlock(), FormattingContext::EscapeReason::FindFixedHeightAncestorQuirk);
55             auto horizontalConstraints = Geometry::horizontalConstraintsForInFlow(containingBlockDisplayBox);
56             auto verticalMargin = formattingContext.geometry().computedVerticalMargin(*containingBlock, horizontalConstraints);
57             auto verticalPadding = boxGeometry.paddingTop().valueOr(0) + boxGeometry.paddingBottom().valueOr(0);
58             auto verticalBorder = boxGeometry.borderTop() + boxGeometry.borderBottom();
59             bodyAndDocumentVerticalMarginPaddingAndBorder += verticalMargin.before.valueOr(0) + verticalMargin.after.valueOr(0) + verticalPadding + verticalBorder;
60         }
61 
62         containingBlock = containingBlock->containingBlock();
63     }
64     // Initial containing block has to have a height.
65     return formattingContext.geometryForBox(layoutBox.initialContainingBlock(), FormattingContext::EscapeReason::FindFixedHeightAncestorQuirk).contentBox().height() - bodyAndDocumentVerticalMarginPaddingAndBorder;
66 }
67 
68 }
