28 #include "IDLTypes.h"
29 #include "JSDOMConvertBase.h"
30 
31 namespace WebCore {
32 
33 // Specialized by generated code for IDL dictionary conversion.
34 template<typename T> T convertDictionary(JSC::ExecState&, JSC::JSValue);
35 
36 
37 template<typename T> struct Converter<IDLDictionary<T>> : DefaultConverter<IDLDictionary<T>> {
38     using ReturnType = T;
39 
40     static ReturnType convert(JSC::ExecState& state, JSC::JSValue value)
41     {
42         return convertDictionary<T>(state, value);
43     }
44 };
45 
46 template<typename T> struct JSConverter<IDLDictionary<T>> {
47     static constexpr bool needsState = true;
48     static constexpr bool needsGlobalObject = true;
49 
50     static JSC::JSValue convert(JSC::ExecState& state, JSDOMGlobalObject& globalObject, const T& dictionary)
51     {
52         return convertDictionaryToJS(state, globalObject, dictionary);
53     }
54 };
