36 // An existing accessibility object is considered to be invalid whether it's already
37 // detached or if it's not but just updating the layout will detach it anyway.
38 #define returnIfWebKitAccessibleIsInvalid(webkitAccessible) G_STMT_START { \
39     if (!webkitAccessible || webkitAccessibleIsDetached(webkitAccessible)) \
40         return; \
41     auto& coreObject = webkitAccessibleGetAccessibilityObject(webkitAccessible); \
42     if (!coreObject.document())                                         \
43         return;                                                         \
44     coreObject.updateBackingStore();                                    \
45     if (webkitAccessibleIsDetached(webkitAccessible))                   \
46         return;                                                         \
47     ; } G_STMT_END
48 
49 #define returnValIfWebKitAccessibleIsInvalid(webkitAccessible, val) G_STMT_START { \
50     if (!webkitAccessible || webkitAccessibleIsDetached(webkitAccessible)) \
51         return (val);                                                   \
52     auto& coreObject = webkitAccessibleGetAccessibilityObject(webkitAccessible); \
53     if (!coreObject.document())                                         \
54         return (val);                                                   \
55     coreObject.updateBackingStore();                                    \
56     if (webkitAccessibleIsDetached(webkitAccessible))                   \
57         return (val);                                                   \
58     ; } G_STMT_END
