 34 namespace WebCore {
 35 namespace Layout {
 36 
 37 class InlineFormattingState;
 38 class InvalidationState;

 39 
 40 // This class implements the layout logic for inline formatting contexts.
 41 // https://www.w3.org/TR/CSS22/visuren.html#inline-formatting
 42 class InlineFormattingContext final : public FormattingContext {
 43     WTF_MAKE_ISO_ALLOCATED(InlineFormattingContext);
 44 public:
 45     InlineFormattingContext(const Container& formattingContextRoot, InlineFormattingState&);
 46     void layoutInFlowContent(InvalidationState&, const HorizontalConstraints&, const VerticalConstraints&) override;
 47 
 48 private:
 49     IntrinsicWidthConstraints computedIntrinsicWidthConstraints() override;
 50 
 51     class Quirks : public FormattingContext::Quirks {
 52     public:
 53         bool lineDescentNeedsCollapsing(const LineBuilder::RunList&) const;
 54         LineBuilder::Constraints::HeightAndBaseline lineHeightConstraints(const Container& formattingRoot) const;

 55 
 56     private:
 57         friend class InlineFormattingContext;
 58         Quirks(const InlineFormattingContext&);


 59 
 60         const InlineFormattingContext& formattingContext() const { return downcast<InlineFormattingContext>(FormattingContext::Quirks::formattingContext()); }



 61 




 62     };
 63     InlineFormattingContext::Quirks quirks() const { return Quirks(*this); }
 64 
 65     class Geometry : public FormattingContext::Geometry {
 66     public:
 67         ContentHeightAndMargin inlineBlockHeightAndMargin(const Box&, const HorizontalConstraints&, const OverrideVerticalValues&) const;
 68         ContentWidthAndMargin inlineBlockWidthAndMargin(const Box&, const HorizontalConstraints&, const OverrideHorizontalValues&);
 69         Optional<InlineLayoutUnit> computedTextIndent(const Container& formattingContextRoot, const HorizontalConstraints&) const;
 70 
 71     private:
 72         friend class InlineFormattingContext;
 73         Geometry(const InlineFormattingContext&);
 74 
 75         const InlineFormattingContext& formattingContext() const { return downcast<InlineFormattingContext>(FormattingContext::Geometry::formattingContext()); }
 76 
 77     };
 78     InlineFormattingContext::Geometry geometry() const { return Geometry(*this); }
 79 
 80     void lineLayout(InlineItems&, LineLayoutContext::InlineItemRange, const HorizontalConstraints&, const VerticalConstraints&);
 81 
 82     void computeIntrinsicWidthForFormattingRoot(const Box&);
 83     InlineLayoutUnit computedIntrinsicWidthForConstraint(const HorizontalConstraints&) const;






 84 
 85     void computeHorizontalMargin(const Box&, const HorizontalConstraints&);
 86     void computeHeightAndMargin(const Box&, const HorizontalConstraints&);
 87     void computeWidthAndMargin(const Box&, const HorizontalConstraints&);
 88 
 89     void collectInlineContentIfNeeded();
 90     LineBuilder::Constraints constraintsForLine(const HorizontalConstraints&, InlineLayoutUnit lineLogicalTop);
 91     void setDisplayBoxesForLine(const LineLayoutContext::LineContent&, const HorizontalConstraints&);
 92     void invalidateFormattingState(const InvalidationState&);
 93 
 94     const InlineFormattingState& formattingState() const { return downcast<InlineFormattingState>(FormattingContext::formattingState()); }
 95     InlineFormattingState& formattingState() { return downcast<InlineFormattingState>(FormattingContext::formattingState()); }
 96     // FIXME: Come up with a structure that requires no friending.
 97     friend class LineBuilder;
 98 };
 99 
100 inline InlineFormattingContext::Geometry::Geometry(const InlineFormattingContext& inlineFormattingContext)
101     : FormattingContext::Geometry(inlineFormattingContext)
102 {
103 }
104 
105 inline InlineFormattingContext::Quirks::Quirks(const InlineFormattingContext& inlineFormattingContext)
106     : FormattingContext::Quirks(inlineFormattingContext)
107 {
108 }
109 
110 }
111 }
