 31 #pragma once
 32 
 33 #include "ActiveDOMObject.h"
 34 #include "EventTarget.h"
 35 #include "ExceptionOr.h"
 36 #include "FileError.h"
 37 #include "FileReaderLoader.h"
 38 #include "FileReaderLoaderClient.h"
 39 #include <wtf/HashMap.h>
 40 #include <wtf/UniqueRef.h>
 41 
 42 namespace JSC {
 43 class ArrayBuffer;
 44 }
 45 
 46 namespace WebCore {
 47 
 48 class Blob;
 49 
 50 class FileReader final : public RefCounted<FileReader>, public ActiveDOMObject, public EventTargetWithInlineData, private FileReaderLoaderClient {
 51     WTF_MAKE_ISO_ALLOCATED(FileReader);
 52 public:
 53     static Ref<FileReader> create(ScriptExecutionContext&);
 54 
 55     virtual ~FileReader();
 56 
 57     enum ReadyState {
 58         EMPTY = 0,
 59         LOADING = 1,
 60         DONE = 2
 61     };
 62 
 63     ExceptionOr<void> readAsArrayBuffer(Blob*);
 64     ExceptionOr<void> readAsBinaryString(Blob*);
 65     ExceptionOr<void> readAsText(Blob*, const String& encoding);
 66     ExceptionOr<void> readAsDataURL(Blob*);
 67     void abort();
 68 
 69     void doAbort();
 70 
 71     ReadyState readyState() const { return m_state; }
 72     RefPtr<FileError> error() { return m_error; }
 73     FileReaderLoader::ReadType readType() const { return m_readType; }
 74     Optional<Variant<String, RefPtr<JSC::ArrayBuffer>>> result() const;
 75 
 76     using RefCounted::ref;
 77     using RefCounted::deref;
 78 
 79     // ActiveDOMObject.
 80     bool hasPendingActivity() const final;
 81 
 82 private:
 83     explicit FileReader(ScriptExecutionContext&);
 84 
 85     // ActiveDOMObject.
 86     const char* activeDOMObjectName() const final;

 87     void stop() final;
 88 
 89     EventTargetInterface eventTargetInterface() const final { return FileReaderEventTargetInterfaceType; }
 90     ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
 91     void refEventTarget() final { ref(); }
 92     void derefEventTarget() final { deref(); }
 93 
 94     void enqueueTask(Function<void()>&&);
 95 
 96     void didStartLoading() final;
 97     void didReceiveData() final;
 98     void didFinishLoading() final;
 99     void didFail(int errorCode) final;
100 
101     ExceptionOr<void> readInternal(Blob&, FileReaderLoader::ReadType);
102     void fireErrorEvent(int httpStatusCode);
103     void fireEvent(const AtomString& type);
104 
105     ReadyState m_state { EMPTY };
106     bool m_aborting { false };
107     RefPtr<Blob> m_blob;
108     FileReaderLoader::ReadType m_readType { FileReaderLoader::ReadAsBinaryString };
109     String m_encoding;
110     std::unique_ptr<FileReaderLoader> m_loader;
111     RefPtr<FileError> m_error;
112     MonotonicTime m_lastProgressNotificationTime { MonotonicTime::nan() };
113     HashMap<uint64_t, Function<void()>> m_pendingTasks;
114 };
115 
116 } // namespace WebCore
