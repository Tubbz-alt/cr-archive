 42 namespace WebCore {
 43 
 44 class CSSAnimation;
 45 class CSSTransition;
 46 class DeclarativeAnimation;
 47 class Element;
 48 
 49 class AnimationTimeline : public RefCounted<AnimationTimeline>, public CanMakeWeakPtr<AnimationTimeline> {
 50 public:
 51     virtual bool isDocumentTimeline() const { return false; }
 52 
 53     void forgetAnimation(WebAnimation*);
 54     virtual void animationTimingDidChange(WebAnimation&);
 55     virtual void removeAnimation(WebAnimation&);
 56 
 57     Optional<double> bindingsCurrentTime();
 58     virtual Optional<Seconds> currentTime() { return m_currentTime; }
 59 
 60     enum class Ordering : uint8_t { Sorted, Unsorted };
 61     Vector<RefPtr<WebAnimation>> animationsForElement(Element&, Ordering ordering = Ordering::Unsorted) const;
 62 
 63     void elementWasRemoved(Element&);
 64     void removeAnimationsForElement(Element&);
 65 
 66     void willChangeRendererForElement(Element&);
 67     void willDestroyRendererForElement(Element&);
 68     void cancelDeclarativeAnimationsForElement(Element&);
 69 
 70     virtual void animationWasAddedToElement(WebAnimation&, Element&);
 71     virtual void animationWasRemovedFromElement(WebAnimation&, Element&);
 72 
 73     void removeDeclarativeAnimationFromListsForOwningElement(WebAnimation&, Element&);
 74 
 75     void updateCSSAnimationsForElement(Element&, const RenderStyle* currentStyle, const RenderStyle& afterChangeStyle);
 76     void updateCSSTransitionsForElement(Element&, const RenderStyle& currentStyle, const RenderStyle& afterChangeStyle);
 77 
 78     using AnimationCollection = ListHashSet<RefPtr<WebAnimation>>;
 79     using ElementToAnimationsMap = HashMap<Element*, AnimationCollection>;
 80     using PropertyToTransitionMap = HashMap<CSSPropertyID, RefPtr<CSSTransition>>;
 81 
 82     virtual ~AnimationTimeline();
 83 
 84 protected:
 85     explicit AnimationTimeline();
 86 
 87     Vector<WeakPtr<WebAnimation>> m_allAnimations;
 88     AnimationCollection m_animations;
 89     HashMap<Element*, PropertyToTransitionMap> m_elementToCompletedCSSTransitionByCSSPropertyID;
 90 
 91 private:
 92     using CSSAnimationCollection = ListHashSet<RefPtr<CSSAnimation>>;
 93     using ElementToCSSAnimationsMap = HashMap<Element*, CSSAnimationCollection>;
 94 
 95     void updateGlobalPosition(WebAnimation&);
 96     PropertyToTransitionMap& ensureRunningTransitionsByProperty(Element&);
 97     void updateCSSTransitionsForElementAndProperty(Element&, CSSPropertyID, const RenderStyle& currentStyle, const RenderStyle& afterChangeStyle, PropertyToTransitionMap&, PropertyToTransitionMap&, const MonotonicTime);
 98     void removeCSSAnimationCreatedByMarkup(Element&, CSSAnimation&);
 99 
100     ElementToAnimationsMap m_elementToAnimationsMap;
101     ElementToAnimationsMap m_elementToCSSAnimationsMap;
102     ElementToAnimationsMap m_elementToCSSTransitionsMap;
103     ElementToCSSAnimationsMap m_elementToCSSAnimationsCreatedByMarkupMap;
104     HashMap<Element*, PropertyToTransitionMap> m_elementToRunningCSSTransitionByCSSPropertyID;
105 
106     Markable<Seconds, Seconds::MarkableTraits> m_currentTime;
107 };
108 
109 } // namespace WebCore
