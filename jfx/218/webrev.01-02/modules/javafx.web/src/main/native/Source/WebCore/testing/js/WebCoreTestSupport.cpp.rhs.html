 57 void injectInternalsObject(JSContextRef context)
 58 {
 59     JSGlobalObject* lexicalGlobalObject = toJS(context);
 60     VM& vm = lexicalGlobalObject->vm();
 61     JSLockHolder lock(vm);
 62     JSDOMGlobalObject* globalObject = jsCast<JSDOMGlobalObject*>(lexicalGlobalObject);
 63     ScriptExecutionContext* scriptContext = globalObject->scriptExecutionContext();
 64     if (is<Document>(*scriptContext)) {
 65         globalObject->putDirect(vm, Identifier::fromString(vm, Internals::internalsId), toJS(lexicalGlobalObject, globalObject, Internals::create(downcast<Document>(*scriptContext))));
 66         Options::useDollarVM() = true;
 67         globalObject->exposeDollarVM(vm);
 68     }
 69 }
 70 
 71 void resetInternalsObject(JSContextRef context)
 72 {
 73     JSGlobalObject* lexicalGlobalObject = toJS(context);
 74     JSLockHolder lock(lexicalGlobalObject);
 75     JSDOMGlobalObject* globalObject = jsCast<JSDOMGlobalObject*>(lexicalGlobalObject);
 76     ScriptExecutionContext* scriptContext = globalObject->scriptExecutionContext();
 77     Page* page = downcast<Document>(scriptContext)->frame()->page();
 78     Internals::resetToConsistentState(*page);
 79     InternalSettings::from(page)->resetToConsistentState();
 80 }

 82 void monitorWheelEvents(WebCore::Frame& frame)
 83 {
 84     Page* page = frame.page();
 85     if (!page)
 86         return;
 87 
 88     page->ensureWheelEventTestMonitor();
 89 }
 90 
 91 void setTestCallbackAndStartNotificationTimer(WebCore::Frame& frame, JSContextRef context, JSObjectRef jsCallbackFunction)
 92 {
 93     Page* page = frame.page();
 94     if (!page || !page->isMonitoringWheelEvents())
 95         return;
 96 
 97     JSValueProtect(context, jsCallbackFunction);
 98 
 99     page->ensureWheelEventTestMonitor().setTestCallbackAndStartNotificationTimer([=](void) {
100         JSObjectCallAsFunction(context, jsCallbackFunction, nullptr, 0, nullptr, nullptr);
101         JSValueUnprotect(context, jsCallbackFunction);
102     });
103 }

201 void setupNewlyCreatedServiceWorker(uint64_t serviceWorkerIdentifier)
202 {
203 #if ENABLE(SERVICE_WORKER)
204     auto identifier = makeObjectIdentifier<ServiceWorkerIdentifierType>(serviceWorkerIdentifier);
205     SWContextManager::singleton().postTaskToServiceWorker(identifier, [identifier] (ServiceWorkerGlobalScope& globalScope) {
206         auto* script = globalScope.script();
207         if (!script)
208             return;
209 
210         auto& state = *globalScope.execState();
211         auto& vm = state.vm();
212         JSLockHolder locker(vm);
213         auto* contextWrapper = script->workerGlobalScopeWrapper();
214         contextWrapper->putDirect(vm, Identifier::fromString(vm, Internals::internalsId), toJS(&state, contextWrapper, ServiceWorkerInternals::create(identifier)));
215     });
216 #else
217     UNUSED_PARAM(serviceWorkerIdentifier);
218 #endif
219 }
