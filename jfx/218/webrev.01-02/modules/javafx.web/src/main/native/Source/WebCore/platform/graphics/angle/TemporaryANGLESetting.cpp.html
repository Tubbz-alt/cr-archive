34 namespace WebCore {
35 
36 TemporaryANGLESetting::TemporaryANGLESetting(GLenum capability, GLenum scopedState)
37     : m_capability(capability)
38     , m_scopedState(scopedState)
39 {
40     m_originalState = gl::IsEnabled(m_capability);
41     if (m_originalState == m_scopedState)
42         return;
43 
44     if (GL_TRUE == m_scopedState)
45         gl::Enable(m_capability);
46     else
47         gl::Disable(m_capability);
48 }
49 
50 TemporaryANGLESetting::~TemporaryANGLESetting()
51 {
52     if (m_originalState == m_scopedState)
53         return;
54 
55     if (GL_TRUE == m_originalState)
56         gl::Enable(m_capability);
57     else
58         gl::Disable(m_capability);
59 }
60 
61 }

63 #endif // ENABLE(GRAPHICS_CONTEXT_GL) && USE(ANGLE)
