+     ASSERT(continuation());

+         if (is<RenderBlock>(*nextInContinuation)) {
+             auto& blockBox = downcast<RenderBlock>(*nextInContinuation);

+                 blockBox.height() + blockBox.collapsedMarginBefore() + blockBox.collapsedMarginAfter() };
