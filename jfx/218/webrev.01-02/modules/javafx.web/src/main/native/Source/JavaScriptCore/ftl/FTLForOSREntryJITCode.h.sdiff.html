36 // - OSR entry only happens through special OSR entry compilations. They have their
37 //   own CodeBlock and their own JITCode.

40 //
41 // - Each OSR entry compilation allows entry through only one bytecode index.

50

52     unsigned bytecodeIndex() const { return m_bytecodeIndex; }
53 
54     void countEntryFailure() { m_entryFailureCount++; }
55     unsigned entryFailureCount() const { return m_entryFailureCount; }
56 
57     ForOSREntryJITCode* ftlForOSREntry();
58 
59 private:
60     ScratchBuffer* m_entryBuffer; // Only for OSR entry code blocks.
