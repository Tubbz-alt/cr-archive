 36 namespace WebCore {
 37 
 38 class CacheStorageProvider;
 39 class ContentSecurityPolicyResponseHeaders;
 40 class ExtendableEvent;
 41 class MessagePortChannel;
 42 class SerializedScriptValue;
 43 class WorkerObjectProxy;
 44 struct MessageWithMessagePorts;

 45 struct ServiceWorkerClientIdentifier;

 46 
 47 class ServiceWorkerThread : public WorkerThread {
 48 public:
 49     template<typename... Args> static Ref<ServiceWorkerThread> create(Args&&... args)
 50     {
 51         return adoptRef(*new ServiceWorkerThread(std::forward<Args>(args)...));
 52     }
 53     virtual ~ServiceWorkerThread();
 54 
 55     WorkerObjectProxy& workerObjectProxy() const { return m_workerObjectProxy; }
 56 
 57     void start(Function<void(const String&, bool)>&&);

 58 
 59     void willPostTaskToFireInstallEvent();
 60     void willPostTaskToFireActivateEvent();
 61     void willPostTaskToFireMessageEvent();
 62 
 63     void queueTaskToFireFetchEvent(Ref<ServiceWorkerFetch::Client>&&, Optional<ServiceWorkerClientIdentifier>&&, ResourceRequest&&, String&& referrer, FetchOptions&&);
 64     void queueTaskToPostMessage(MessageWithMessagePorts&&, ServiceWorkerOrClientData&& sourceData);
 65     void queueTaskToFireInstallEvent();
 66     void queueTaskToFireActivateEvent();
 67 
 68     const ServiceWorkerContextData& contextData() const { return m_data; }
 69 
 70     ServiceWorkerIdentifier identifier() const { return m_data.serviceWorkerIdentifier; }
 71     bool doesHandleFetch() const { return m_doesHandleFetch; }
 72 
 73     void startFetchEventMonitoring();
 74     void stopFetchEventMonitoring() { m_isHandlingFetchEvent = false; }
 75 
 76 protected:
 77     Ref<WorkerGlobalScope> createWorkerGlobalScope(const WorkerParameters&, Ref<SecurityOrigin>&&, Ref<SecurityOrigin>&& topOrigin) final;
 78     void runEventLoop() override;
 79 
 80 private:
 81     WEBCORE_EXPORT ServiceWorkerThread(const ServiceWorkerContextData&, String&& userAgent, WorkerLoaderProxy&, WorkerDebuggerProxy&, IDBClient::IDBConnectionProxy*, SocketProvider*);
 82 
 83     bool isServiceWorkerThread() const final { return true; }
 84     void finishedEvaluatingScript() final;
 85 
 86     void finishedFiringInstallEvent(bool hasRejectedAnyPromise);
 87     void finishedFiringActivateEvent();
 88     void finishedFiringMessageEvent();
 89     void finishedStarting();
 90 
 91     void startHeartBeatTimer();
 92     void heartBeatTimerFired();
 93     void installEventTimerFired();
 94 
 95     ServiceWorkerContextData m_data;
 96     WorkerObjectProxy& m_workerObjectProxy;
 97     bool m_doesHandleFetch { false };
 98 
 99     bool m_isHandlingFetchEvent { false };
100     uint64_t m_messageEventCount { 0 };
101     enum class State { Idle, Starting, Installing, Activating };
102     State m_state { State::Idle };
103     bool m_ongoingHeartBeatCheck { false };
104 
105     static constexpr Seconds heartBeatTimeout { 60_s };
106     static constexpr Seconds heartBeatTimeoutForTest { 1_s };
107     Seconds m_heartBeatTimeout { heartBeatTimeout };
108     Timer m_heartBeatTimer;
109 };

111 } // namespace WebCore
