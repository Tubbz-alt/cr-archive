 40 template<typename Source>
 41 static inline bool updateTrackSource(Source& source, MediaStreamTrack* track)


 42 {
 43     if (!track) {
 44         source.stop();
 45         return true;
 46     }
 47     return source.setSource(track->privateTrack());






















 48 }

 50 void LibWebRTCRtpSenderBackend::replaceTrack(ScriptExecutionContext& context, RTCRtpSender& sender, RefPtr<MediaStreamTrack>&& track, DOMPromiseDeferred<void>&& promise)
 51 {
 52     if (!m_peerConnectionBackend) {
 53         promise.reject(Exception { InvalidStateError, "No WebRTC backend"_s });
 54         return;
 55     }
 56 
 57     auto* currentTrack = sender.track();
 58 
 59     ASSERT(!track || !currentTrack || currentTrack->source().type() == track->source().type());
 60     if (currentTrack) {
 61     switch (currentTrack->source().type()) {
 62     case RealtimeMediaSource::Type::None:
 63         ASSERT_NOT_REACHED();
 64         promise.reject(InvalidModificationError);
 65         break;
 66     case RealtimeMediaSource::Type::Audio:
 67         if (!updateTrackSource(*audioSource(), track.get())) {
 68             promise.reject(InvalidModificationError);
 69             return;
 70         }
 71         break;
 72     case RealtimeMediaSource::Type::Video:
 73         if (!updateTrackSource(*videoSource(), track.get())) {
 74             promise.reject(InvalidModificationError);
 75             return;
 76         }
 77         break;
 78     }
 79     }
 80 
 81     // FIXME: Remove this postTask once this whole function is executed as part of the RTCPeerConnection operation queue.
 82     context.postTask([protectedSender = makeRef(sender), promise = WTFMove(promise), track = WTFMove(track), this](ScriptExecutionContext&) mutable {
 83         if (protectedSender->isStopped())
 84             return;
 85 
 86         if (!track) {
 87             protectedSender->setTrackToNull();
 88             promise.resolve();
 89             return;
 90         }
 91 
 92         bool hasTrack = protectedSender->track();
 93         protectedSender->setTrack(track.releaseNonNull());
 94 
 95         if (hasTrack) {
 96             promise.resolve();
 97             return;
 98         }
 99 
100         if (RuntimeEnabledFeatures::sharedFeatures().webRTCUnifiedPlanEnabled()) {
101             m_source = nullptr;
102             m_peerConnectionBackend->setSenderSourceFromTrack(*this, *protectedSender->track());
103             promise.resolve();
104             return;
105         }
106 
107         auto result = m_peerConnectionBackend->addTrack(*protectedSender->track(), { });
108         if (result.hasException()) {
109             promise.reject(result.releaseException());
110             return;
111         }
112         promise.resolve();
113     });
114 }

 53 void LibWebRTCRtpSenderBackend::startSource()

 58         source->start();

 60     });

 74 void LibWebRTCRtpSenderBackend::replaceTrack(ScriptExecutionContext& context, RTCRtpSender& sender, RefPtr<MediaStreamTrack>&& track, DOMPromiseDeferred<void>&& promise)
 75 {
 76     if (!m_peerConnectionBackend) {
 77         promise.reject(Exception { InvalidStateError, "No WebRTC backend"_s });
 78         return;
 79     }
 80 
 81     if (!track)
 82         stopSource();
 83     else if (sender.track()) {
 84         switchOn(m_source, [&](Ref<RealtimeOutgoingAudioSource>& source) {
 85             ASSERT(track->source().type() == RealtimeMediaSource::Type::Audio);
 86             source->stop();
 87             source->setSource(track->privateTrack());
 88             source->start();
 89         }, [&](Ref<RealtimeOutgoingVideoSource>& source) {
 90             ASSERT(track->source().type() == RealtimeMediaSource::Type::Video);
 91             source->stop();
 92             source->setSource(track->privateTrack());
 93             source->start();
 94         }, [](std::nullptr_t&) {
 95             ASSERT_NOT_REACHED();
 96         });






 97     }
 98 
 99     // FIXME: Remove this postTask once this whole function is executed as part of the RTCPeerConnection operation queue.
100     context.postTask([protectedSender = makeRef(sender), promise = WTFMove(promise), track = WTFMove(track), this](ScriptExecutionContext&) mutable {
101         if (protectedSender->isStopped())
102             return;
103 
104         if (!track) {
105             protectedSender->setTrackToNull();
106             promise.resolve();
107             return;
108         }
109 
110         bool hasTrack = protectedSender->track();
111         protectedSender->setTrack(track.releaseNonNull());
112 
113         if (hasTrack) {
114             promise.resolve();
115             return;
116         }
117 
118         m_peerConnectionBackend->setSenderSourceFromTrack(*this, *protectedSender->track());











119         promise.resolve();
120     });
121 }

161 RealtimeOutgoingVideoSource* LibWebRTCRtpSenderBackend::videoSource()
162 {
163     return switchOn(m_source,
164         [](Ref<RealtimeOutgoingVideoSource>& source) { return source.ptr(); },
165         [](const auto&) -> RealtimeOutgoingVideoSource* { return nullptr; }
166     );

169 bool LibWebRTCRtpSenderBackend::hasSource() const
170 {
171     return switchOn(m_source,
172         [](const std::nullptr_t&) { return false; },
173         [](const auto&) { return true; }
174     );
175 }

177 void LibWebRTCRtpSenderBackend::setSource(Source&& source)
178 {
179     stopSource();
180     m_source = WTFMove(source);
181     startSource();
182 }

184 void LibWebRTCRtpSenderBackend::takeSource(LibWebRTCRtpSenderBackend& backend)
185 {
186     ASSERT(backend.hasSource());
187     stopSource();
188     m_source = WTFMove(backend.m_source);
189     backend.m_source = nullptr;
