26 #pragma once
27 
28 namespace WebCore {
29 
30 class ResourceRequest;
31 
32 struct SameSiteInfo {
33     WEBCORE_EXPORT static SameSiteInfo create(const ResourceRequest&);
34 
35     bool isSameSite { false };
36     bool isTopSite { false };
37     bool isSafeHTTPMethod { false };
38 
39     template <class Encoder> void encode(Encoder&) const;
40     template <class Decoder> static bool decode(Decoder&, SameSiteInfo&);
41 };
42 
43 template <class Encoder>
44 void SameSiteInfo::encode(Encoder& encoder) const
45 {
46     encoder << isSameSite;
47     encoder << isTopSite;
48     encoder << isSafeHTTPMethod;
49 }
50 
51 template <class Decoder>
52 bool SameSiteInfo::decode(Decoder& decoder, SameSiteInfo& info)
53 {
54     if (!decoder.decode(info.isSameSite))
55         return false;
56     if (!decoder.decode(info.isTopSite))
57         return false;
58     if (!decoder.decode(info.isSafeHTTPMethod))
59         return false;
60     return true;
61 }

63 } // namespace WebCore
