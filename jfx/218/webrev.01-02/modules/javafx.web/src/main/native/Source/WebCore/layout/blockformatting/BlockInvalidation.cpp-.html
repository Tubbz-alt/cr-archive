43 static bool invalidationStopsAtFormattingContextBoundary(const Container& formattingContextRoot, const Box&, StyleDiff)
44 {
45     UNUSED_PARAM(formattingContextRoot);
46 
47     ASSERT(formattingContextRoot.establishesFormattingContext());
48     return true;
49 }
50 
51 static LayoutState::UpdateType computeUpdateType(const Box&, StyleDiff, BlockFormattingState&)
52 {
53     return LayoutState::UpdateType::All;
54 }
55 
56 static LayoutState::UpdateType computeUpdateTypeForAncestor(const Container&, StyleDiff, BlockFormattingState&)
57 {
58     return LayoutState::UpdateType::All;
59 }
60 
61 InvalidationResult BlockInvalidation::invalidate(const Box& layoutBox, StyleDiff styleDiff, LayoutState& layoutState,
62     BlockFormattingState& formattingState)
63 {
64     // Invalidate this box and the containing block chain all the way up to the formatting context root (and beyond if needed).
65     layoutState.markNeedsUpdate(layoutBox, computeUpdateType(layoutBox, styleDiff, formattingState));
66     for (auto* containingBlock = layoutBox.containingBlock(); containingBlock; containingBlock = containingBlock->containingBlock()) {
67         if (containingBlock->establishesFormattingContext() && invalidationStopsAtFormattingContextBoundary(*containingBlock, layoutBox, styleDiff))
68             return { containingBlock };
69         layoutState.markNeedsUpdate(*containingBlock, computeUpdateTypeForAncestor(*containingBlock, styleDiff, formattingState));
70     }
71     // Invalidation always stops at the initial containing block.
72     ASSERT_NOT_REACHED();
73     return { nullptr };
74 }
