 54 Ref<SharedBuffer> PasteboardCustomData::createSharedBuffer() const
 55 {
 56     const static unsigned currentCustomDataSerializationVersion = 1;
 57 
 58     WTF::Persistence::Encoder encoder;
 59     encoder << currentCustomDataSerializationVersion;
 60     encoder << origin;
 61     encoder << sameOriginCustomData;
 62     encoder << orderedTypes;
 63     return SharedBuffer::create(encoder.buffer(), encoder.bufferSize());
 64 }
 65 
 66 PasteboardCustomData PasteboardCustomData::fromSharedBuffer(const SharedBuffer& buffer)
 67 {
 68     const static unsigned maxSupportedDataSerializationVersionNumber = 1;
 69 
 70     PasteboardCustomData result;
 71     WTF::Persistence::Decoder decoder { reinterpret_cast<const uint8_t*>(buffer.data()), buffer.size() };
 72     unsigned version;
 73     if (!decoder.decode(version) || version > maxSupportedDataSerializationVersionNumber)
 74         return { };
 75 
 76     if (!decoder.decode(result.origin))
 77         return { };
 78 
 79     if (!decoder.decode(result.sameOriginCustomData))
 80         return { };
 81 
 82     if (!decoder.decode(result.orderedTypes))
 83         return { };
 84 
 85     return result;






 86 }
