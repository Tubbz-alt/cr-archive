 40 // Inputs: PropertyID, isImportant bool, CSSParserTokenRange.
 41 // Outputs: Vector of CSSProperties
 42 
 43 class CSSPropertyParser {
 44     WTF_MAKE_NONCOPYABLE(CSSPropertyParser);
 45 public:
 46     static bool parseValue(CSSPropertyID, bool important,
 47         const CSSParserTokenRange&, const CSSParserContext&,
 48         Vector<CSSProperty, 256>&, StyleRuleType);
 49 
 50     // Parses a non-shorthand CSS property
 51     static RefPtr<CSSValue> parseSingleValue(CSSPropertyID, const CSSParserTokenRange&, const CSSParserContext&);
 52     static bool canParseTypedCustomPropertyValue(const String& syntax, const CSSParserTokenRange&, const CSSParserContext&);
 53     static RefPtr<CSSCustomPropertyValue> parseTypedCustomPropertyValue(const String& name, const String& syntax, const CSSParserTokenRange&, const Style::BuilderState&, const CSSParserContext&);
 54     static void collectParsedCustomPropertyValueDependencies(const String& syntax, bool isRoot, HashSet<CSSPropertyID>& dependencies, const CSSParserTokenRange&, const CSSParserContext&);
 55 
 56 private:
 57     CSSPropertyParser(const CSSParserTokenRange&, const CSSParserContext&, Vector<CSSProperty, 256>*, bool consumeWhitespace = true);
 58 
 59     // FIXME: Rename once the CSSParserValue-based parseValue is removed
 60     bool parseValueStart(CSSPropertyID, bool important);
 61     bool consumeCSSWideKeyword(CSSPropertyID, bool important);
 62     RefPtr<CSSValue> parseSingleValue(CSSPropertyID, CSSPropertyID = CSSPropertyInvalid);
 63     bool canParseTypedCustomPropertyValue(const String& syntax);
 64     RefPtr<CSSCustomPropertyValue> parseTypedCustomPropertyValue(const String& name, const String& syntax, const Style::BuilderState&);
 65     void collectParsedCustomPropertyValueDependencies(const String& syntax, bool isRoot, HashSet<CSSPropertyID>& dependencies);
 66 
 67     bool inQuirksMode() const { return m_context.mode == HTMLQuirksMode; }
 68 
 69     bool parseViewportDescriptor(CSSPropertyID propId, bool important);
 70     bool parseFontFaceDescriptor(CSSPropertyID);
 71 
 72     void addProperty(CSSPropertyID, CSSPropertyID, Ref<CSSValue>&&, bool important, bool implicit = false);
 73     void addExpandedPropertyForValue(CSSPropertyID propId, Ref<CSSValue>&&, bool);
 74 
 75     bool consumeBorder(RefPtr<CSSValue>& width, RefPtr<CSSValue>& style, RefPtr<CSSValue>& color);
 76 
 77     bool parseShorthand(CSSPropertyID, bool important);
 78     bool consumeShorthandGreedily(const StylePropertyShorthand&, bool important);
 79     bool consume2ValueShorthand(const StylePropertyShorthand&, bool important);
 80     bool consume4ValueShorthand(const StylePropertyShorthand&, bool important);
 81 
 82     // Legacy parsing allows <string>s for animation-name
 83     bool consumeAnimationShorthand(const StylePropertyShorthand&, bool important);
 84     bool consumeBackgroundShorthand(const StylePropertyShorthand&, bool important);
 85     bool consumeOverflowShorthand(bool important);
 86 
 87     bool consumeColumns(bool important);
 88 
 89     bool consumeGridItemPositionShorthand(CSSPropertyID, bool important);
 90     bool consumeGridTemplateRowsAndAreasAndColumns(CSSPropertyID, bool important);
 91     bool consumeGridTemplateShorthand(CSSPropertyID, bool important);
 92     bool consumeGridShorthand(bool important);
 93     bool consumeGridAreaShorthand(bool important);
 94 
 95     bool consumePlaceContentShorthand(bool important);
 96     bool consumePlaceItemsShorthand(bool important);
 97     bool consumePlaceSelfShorthand(bool important);
 98 
 99     bool consumeFont(bool important);
100     bool consumeFontVariantShorthand(bool important);
101     bool consumeSystemFont(bool important);
102 
103     bool consumeBorderSpacing(bool important);
104 
105     // CSS3 Parsing Routines (for properties specific to CSS3)
106     bool consumeBorderImage(CSSPropertyID, bool important);
107 
108     bool consumeFlex(bool important);
109 
110     bool consumeLegacyBreakProperty(CSSPropertyID, bool important);
111 
112     bool consumeTransformOrigin(bool important);
113     bool consumePerspectiveOrigin(bool important);
114 
115 private:
116     // Inputs:
117     CSSParserTokenRange m_range;
118     const CSSParserContext& m_context;
119 
120     // Outputs:
121     Vector<CSSProperty, 256>* m_parsedProperties;
122 };
