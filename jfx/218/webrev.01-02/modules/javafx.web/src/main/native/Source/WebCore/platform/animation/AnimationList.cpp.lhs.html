27 #define FILL_UNSET_PROPERTY(test, propGet, propSet) \
28 for (i = 0; i < size() && animation(i).test(); ++i) { } \
29 if (i) { \
30     for (size_t j = 0; i < size(); ++i, ++j) \
31         animation(i).propSet(animation(j).propGet()); \
32 }
33 


34 AnimationList::AnimationList(const AnimationList& other)

35 {
36     m_animations.reserveInitialCapacity(other.size());
37     for (auto& animation : other.m_animations)
38         m_animations.uncheckedAppend(Animation::create(animation.get()));
39 }
40 
41 void AnimationList::fillUnsetProperties()
42 {
43     size_t i;
44     FILL_UNSET_PROPERTY(isDelaySet, delay, setDelay);
45     FILL_UNSET_PROPERTY(isDirectionSet, direction, setDirection);
46     FILL_UNSET_PROPERTY(isDurationSet, duration, setDuration);
47     FILL_UNSET_PROPERTY(isFillModeSet, fillMode, setFillMode);
48     FILL_UNSET_PROPERTY(isIterationCountSet, iterationCount, setIterationCount);
49     FILL_UNSET_PROPERTY(isPlayStateSet, playState, setPlayState);
50     FILL_UNSET_PROPERTY(isNameSet, name, setName);
51     FILL_UNSET_PROPERTY(isTimingFunctionSet, timingFunction, setTimingFunction);
52     FILL_UNSET_PROPERTY(isPropertySet, property, setProperty);
53 }
54 
55 bool AnimationList::operator==(const AnimationList& other) const
56 {
57     if (size() != other.size())
58         return false;
59     for (size_t i = 0; i < size(); ++i) {
60         if (animation(i) != other.animation(i))
61             return false;
62     }
63     return true;
64 }
