 40 FloatAvoider::FloatAvoider(const Box& layoutBox, Display::Box absoluteDisplayBox, LayoutPoint containingBlockAbsoluteTopLeft, HorizontalEdges containingBlockAbsoluteContentBox)
 41     : m_layoutBox(makeWeakPtr(layoutBox))
 42     , m_absoluteDisplayBox(absoluteDisplayBox)
 43     , m_containingBlockAbsoluteTopLeft(containingBlockAbsoluteTopLeft)
 44     , m_containingBlockAbsoluteContentBox(containingBlockAbsoluteContentBox)
 45 {
 46     ASSERT(m_layoutBox->establishesBlockFormattingContext());
 47     m_absoluteDisplayBox.setLeft({ initialHorizontalPosition() });
 48 }
 49 
 50 void FloatAvoider::setHorizontalConstraints(HorizontalConstraints horizontalConstraints)
 51 {
 52     if ((isLeftAligned() && !horizontalConstraints.left) || (!isLeftAligned() && !horizontalConstraints.right)) {
 53         // No constraints? Set horizontal position back to the inital value.
 54         m_absoluteDisplayBox.setLeft(initialHorizontalPosition());
 55         return;
 56     }
 57 
 58     auto constrainWithContainingBlock = [&](auto left) -> PositionInContextRoot {
 59         // Horizontal position is constrained by the containing block's content box.
 60         // Compute the horizontal position for the new floating by taking both the contining block and the current left/right floats into account.
 61         if (isLeftAligned())
 62             return std::max<PositionInContextRoot>({ m_containingBlockAbsoluteContentBox.left + marginStart() }, left);
 63 
 64         // Make sure it does not overflow the containing block on the right.
 65         return std::min<PositionInContextRoot>(left, { m_containingBlockAbsoluteContentBox.right - marginBoxWidth() + marginStart() });
 66     };
 67 
 68     auto positionCandidate = horizontalPositionCandidate(horizontalConstraints);
 69     m_absoluteDisplayBox.setLeft(constrainWithContainingBlock(positionCandidate));
 70 }
 71 
 72 void FloatAvoider::setVerticalConstraint(PositionInContextRoot verticalConstraint)
 73 {
 74     m_absoluteDisplayBox.setTop(verticalPositionCandidate(verticalConstraint));
 75 }
 76 
 77 PositionInContextRoot FloatAvoider::horizontalPositionCandidate(HorizontalConstraints horizontalConstraints)
 78 {
 79     return { isLeftAligned() ? *horizontalConstraints.left : *horizontalConstraints.right - rect().width() };
 80 }
 81 
 82 PositionInContextRoot FloatAvoider::verticalPositionCandidate(PositionInContextRoot verticalConstraint)
 83 {
 84     return verticalConstraint;
 85 }
 86 
 87 PositionInContextRoot FloatAvoider::initialHorizontalPosition() const
 88 {
 89     // Align the box with the containing block's content box.
 90     auto left = isLeftAligned() ? m_containingBlockAbsoluteContentBox.left : m_containingBlockAbsoluteContentBox.right - marginBoxWidth();
 91     left += marginStart();
 92     return { left };
 93 }
 94 
 95 bool FloatAvoider::overflowsContainingBlock() const
 96 {
 97     auto left = displayBox().left() - marginStart();
 98     if (m_containingBlockAbsoluteContentBox.left > left)
 99         return true;
100 
101     auto right = displayBox().right() + marginEnd();
102     return m_containingBlockAbsoluteContentBox.right < right;
103 }
104 
105 Display::Rect FloatAvoider::rectInContainingBlock() const
106 {
107     // From formatting root coordinate system back to containing block's.
108     if (m_containingBlockAbsoluteTopLeft.isZero())
109         return m_absoluteDisplayBox.rect();
110 
111     return {
112         m_absoluteDisplayBox.top() - m_containingBlockAbsoluteTopLeft.y(),
113         m_absoluteDisplayBox.left() - m_containingBlockAbsoluteTopLeft.x(),
114         m_absoluteDisplayBox.width(),
115         m_absoluteDisplayBox.height()
116     };
117 }
