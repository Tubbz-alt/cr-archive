 25 #pragma once
 26 
 27 #if ENABLE(POINTER_EVENTS)
 28 
 29 #include "ExceptionOr.h"
 30 #include "PointerID.h"
 31 #include <wtf/HashMap.h>
 32 
 33 namespace WebCore {
 34 
 35 class Document;
 36 class Element;
 37 class EventTarget;
 38 class IntPoint;
 39 class MouseEvent;
 40 class Page;
 41 class PlatformTouchEvent;
 42 class PointerEvent;
 43 class WindowProxy;
 44 
 45 class PointerCaptureController {
 46     WTF_MAKE_NONCOPYABLE(PointerCaptureController);
 47     WTF_MAKE_FAST_ALLOCATED;
 48 public:
 49     explicit PointerCaptureController(Page&);
 50 
 51     Element* pointerCaptureElement(Document*, PointerID);
 52     ExceptionOr<void> setPointerCapture(Element*, PointerID);
 53     ExceptionOr<void> releasePointerCapture(Element*, PointerID);
 54     bool hasPointerCapture(Element*, PointerID);
 55     void reset();
 56 
 57     void pointerLockWasApplied();
 58     void elementWasRemoved(Element&);
 59 
 60     RefPtr<PointerEvent> pointerEventForMouseEvent(const MouseEvent&);
 61 
 62 #if ENABLE(TOUCH_EVENTS) && PLATFORM(IOS_FAMILY)
 63     void dispatchEventForTouchAtIndex(EventTarget&, const PlatformTouchEvent&, unsigned, bool isPrimary, WindowProxy&);
 64 #endif
 65 
 66     WEBCORE_EXPORT void touchWithIdentifierWasRemoved(PointerID);
 67     bool hasCancelledPointerEventForIdentifier(PointerID);
 68     bool preventsCompatibilityMouseEventsForIdentifier(PointerID);
 69     void dispatchEvent(PointerEvent&, EventTarget*);
 70     WEBCORE_EXPORT void cancelPointer(PointerID, const IntPoint&);
 71     void processPendingPointerCapture(PointerID);
 72 
 73 private:
 74     struct CapturingData {
 75         RefPtr<Element> pendingTargetOverride;
 76         RefPtr<Element> targetOverride;
 77 #if ENABLE(TOUCH_EVENTS) && PLATFORM(IOS_FAMILY)
 78         RefPtr<Element> previousTarget;
 79 #endif
 80         String pointerType;
 81         bool cancelled { false };
 82         bool isPrimary { false };
 83         bool preventsCompatibilityMouseEvents { false };
 84         bool pointerIsPressed { false };
 85         short previousMouseButton { -1 };
 86     };
 87 
 88     CapturingData& ensureCapturingDataForPointerEvent(const PointerEvent&);
 89     void pointerEventWillBeDispatched(const PointerEvent&, EventTarget*);
 90     void pointerEventWasDispatched(const PointerEvent&);
 91 
 92     Page& m_page;
 93     // While PointerID is defined as int32_t, we use int64_t here so that we may use a value outside of the int32_t range to have safe
 94     // empty and removed values, allowing any int32_t to be provided through the API for lookup in this hashmap.
 95     using PointerIdToCapturingDataMap = HashMap<int64_t, CapturingData, WTF::IntHash<int64_t>, WTF::SignedWithZeroKeyHashTraits<int64_t>>;
 96     PointerIdToCapturingDataMap m_activePointerIdsToCapturingData;
 97     bool m_processingPendingPointerCapture { false };
 98 };
 99 
100 } // namespace WebCore
101 
102 #endif // ENABLE(POINTER_EVENTS)
