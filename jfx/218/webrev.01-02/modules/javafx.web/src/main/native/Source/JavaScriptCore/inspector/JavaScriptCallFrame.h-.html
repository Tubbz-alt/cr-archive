26 #pragma once
27 
28 #include "DebuggerCallFrame.h"
29 #include "JSCJSValueInlines.h"
30 #include <wtf/Forward.h>
31 #include <wtf/RefCounted.h>
32 #include <wtf/text/TextPosition.h>
33 
34 namespace Inspector {
35 
36 class JavaScriptCallFrame : public RefCounted<JavaScriptCallFrame> {
37 public:
38     static Ref<JavaScriptCallFrame> create(Ref<JSC::DebuggerCallFrame>&& debuggerCallFrame)
39     {
40         return adoptRef(*new JavaScriptCallFrame(WTFMove(debuggerCallFrame)));
41     }
42 
43     JavaScriptCallFrame* caller();
44     intptr_t sourceID() const { return m_debuggerCallFrame->sourceID(); }
45     const TextPosition position() const { return m_debuggerCallFrame->position(); }
46     int line() const { return m_debuggerCallFrame->line(); }
47     int column() const { return m_debuggerCallFrame->column(); }
48 
49     String functionName() const { return m_debuggerCallFrame->functionName(); }
50     JSC::DebuggerCallFrame::Type type() const { return m_debuggerCallFrame->type(); }
51     JSC::DebuggerScope* scopeChain() const { return m_debuggerCallFrame->scope(); }
52     JSC::JSGlobalObject* vmEntryGlobalObject() const { return m_debuggerCallFrame->vmEntryGlobalObject(); }
53     bool isTailDeleted() const { return m_debuggerCallFrame->isTailDeleted(); }
54 
55     JSC::JSValue thisValue() const { return m_debuggerCallFrame->thisValue(); }
56     JSC::JSValue evaluateWithScopeExtension(const String& script, JSC::JSObject* scopeExtension, NakedPtr<JSC::Exception>& exception) const { return m_debuggerCallFrame->evaluateWithScopeExtension(script, scopeExtension, exception); }
57 
58 private:
59     JavaScriptCallFrame(Ref<JSC::DebuggerCallFrame>&&);
60 
61     Ref<JSC::DebuggerCallFrame> m_debuggerCallFrame;
62     RefPtr<JavaScriptCallFrame> m_caller;
63 };
64 
65 } // namespace Inspector
