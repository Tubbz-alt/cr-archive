40 void BitmapTextureJava::updateContents(TextureMapper& mapper, GraphicsLayer* sourceLayer, const IntRect& targetRect, const IntPoint& sourceOffset, float /*scale*/)
41 {
42     GraphicsContext& context = m_image->context();
43     // Share RenderThemeJava context
44     context.platformContext()->setJRenderTheme(static_cast<TextureMapperJava&>(mapper).graphicsContext()->platformContext()->jRenderTheme());
45 
46     context.clearRect(targetRect);
47 
48     IntRect sourceRect(targetRect);
49     sourceRect.setLocation(sourceOffset);
50     context.save();
51     context.clip(targetRect);
52     context.translate(targetRect.x() - sourceOffset.x(), targetRect.y() - sourceOffset.y());
53     sourceLayer->paintGraphicsLayerContents(context, sourceRect);
54     context.restore();
55 }

57 void BitmapTextureJava::didReset()
58 {
59     float devicePixelRatio = 1.0;
60     m_image = ImageBuffer::create(contentSize(), Accelerated, devicePixelRatio);
61 }
62 
63 void BitmapTextureJava::updateContents(Image* image, const IntRect& targetRect, const IntPoint& offset)
64 {
65     m_image->context().drawImage(*image, targetRect, IntRect(offset, targetRect.size()), CompositeCopy);
66 }
