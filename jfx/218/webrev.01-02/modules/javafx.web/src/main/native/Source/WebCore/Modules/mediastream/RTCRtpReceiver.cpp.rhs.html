34 #if ENABLE(WEB_RTC)
35 
36 #include "JSDOMPromiseDeferred.h"
37 #include "PeerConnectionBackend.h"
38 #include "RTCRtpCapabilities.h"
39 #include <wtf/IsoMallocInlines.h>
40 
41 namespace WebCore {
42 
43 WTF_MAKE_ISO_ALLOCATED_IMPL(RTCRtpReceiver);
44 
45 RTCRtpReceiver::RTCRtpReceiver(PeerConnectionBackend& connection, Ref<MediaStreamTrack>&& track, std::unique_ptr<RTCRtpReceiverBackend>&& backend)
46     : m_track(WTFMove(track))
47     , m_backend(WTFMove(backend))
48     , m_connection(makeWeakPtr(&connection))
49 {
50 }
51 
52 void RTCRtpReceiver::stop()
53 {
54     if (!m_backend)
55         return;
56 
57     m_backend = nullptr;
58     m_track->stopTrack(MediaStreamTrack::StopMode::PostEvent);
59 }
60 
61 void RTCRtpReceiver::getStats(Ref<DeferredPromise>&& promise)
62 {
63     if (!m_connection) {
64         promise->reject(InvalidStateError);
65         return;
66     }
67     m_connection->getStats(*this, WTFMove(promise));
68 }
69 
70 Optional<RTCRtpCapabilities> RTCRtpReceiver::getCapabilities(ScriptExecutionContext& context, const String& kind)
71 {
72     return PeerConnectionBackend::receiverCapabilities(context, kind);
73 }
74 
75 } // namespace WebCore
76 
77 #endif // ENABLE(WEB_RTC)
