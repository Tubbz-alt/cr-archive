26 #pragma once
27 
28 #include "CSSPropertyNames.h"
29 #include "DeclarativeAnimation.h"
30 #include <wtf/MonotonicTime.h>
31 #include <wtf/Ref.h>
32 
33 namespace WebCore {
34 
35 class Animation;
36 class Element;
37 class RenderStyle;
38 
39 class CSSTransition final : public DeclarativeAnimation {
40     WTF_MAKE_ISO_ALLOCATED(CSSTransition);
41 public:
42     static Ref<CSSTransition> create(Element&, CSSPropertyID, MonotonicTime generationTime, const Animation&, const RenderStyle* oldStyle, const RenderStyle& newStyle, Seconds delay, Seconds duration, const RenderStyle& reversingAdjustedStartStyle, double);
43     ~CSSTransition() = default;
44 
45     bool isCSSTransition() const override { return true; }
46     String transitionProperty() const { return getPropertyNameString(m_property); }
47     CSSPropertyID property() const { return m_property; }
48     MonotonicTime generationTime() const { return m_generationTime; }
49     const RenderStyle& targetStyle() const { return *m_targetStyle; }
50     const RenderStyle& currentStyle() const { return *m_currentStyle; }
51     const RenderStyle& reversingAdjustedStartStyle() const { return *m_reversingAdjustedStartStyle; }
52     double reversingShorteningFactor() const { return m_reversingShorteningFactor; }
53 
54     void resolve(RenderStyle&) final;
55 
56 private:
57     CSSTransition(Element&, CSSPropertyID, MonotonicTime generationTime, const Animation&, const RenderStyle& oldStyle, const RenderStyle& targetStyle, const RenderStyle& reversingAdjustedStartStyle, double);
58     void setTimingProperties(Seconds delay, Seconds duration);
59     Ref<AnimationEventBase> createEvent(const AtomString& eventType, double elapsedTime, const String& pseudoId, Optional<Seconds> timelineTime) final;
60 
61     CSSPropertyID m_property;
62     MonotonicTime m_generationTime;
63     std::unique_ptr<RenderStyle> m_targetStyle;
64     std::unique_ptr<RenderStyle> m_currentStyle;
65     std::unique_ptr<RenderStyle> m_reversingAdjustedStartStyle;
66     double m_reversingShorteningFactor;
67 
68 };
69 
70 } // namespace WebCore
71 
72 SPECIALIZE_TYPE_TRAITS_WEB_ANIMATION(CSSTransition, isCSSTransition())
