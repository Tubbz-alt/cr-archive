 46 extern "C" JS_EXPORT_PRIVATE Opcode g_opcodeMap[numOpcodeIDs + numWasmOpcodeIDs];
 47 extern "C" JS_EXPORT_PRIVATE Opcode g_opcodeMapWide16[numOpcodeIDs + numWasmOpcodeIDs];
 48 extern "C" JS_EXPORT_PRIVATE Opcode g_opcodeMapWide32[numOpcodeIDs + numWasmOpcodeIDs];
 49 
 50 class Data {
 51 
 52 public:
 53     static void performAssertions(VM&);
 54 
 55 private:
 56     static uint8_t s_exceptionInstructions[maxOpcodeLength + 1];
 57     static uint8_t s_wasmExceptionInstructions[maxOpcodeLength + 1];
 58 
 59     friend void initialize();
 60 
 61     friend Instruction* exceptionInstructions();
 62     friend Instruction* wasmExceptionInstructions();
 63     friend Opcode* opcodeMap();
 64     friend Opcode* opcodeMapWide16();
 65     friend Opcode* opcodeMapWide32();
 66     friend Opcode getOpcode(OpcodeID);
 67     friend Opcode getOpcodeWide16(OpcodeID);
 68     friend Opcode getOpcodeWide32(OpcodeID);
 69     template<PtrTag tag> friend MacroAssemblerCodePtr<tag> getCodePtr(OpcodeID);
 70     template<PtrTag tag> friend MacroAssemblerCodePtr<tag> getWide16CodePtr(OpcodeID);
 71     template<PtrTag tag> friend MacroAssemblerCodePtr<tag> getWide32CodePtr(OpcodeID);
 72     template<PtrTag tag> friend MacroAssemblerCodeRef<tag> getCodeRef(OpcodeID);
 73 };
 74 
 75 void initialize();
 76 
 77 inline Instruction* exceptionInstructions()
 78 {
 79     return reinterpret_cast<Instruction*>(Data::s_exceptionInstructions);
 80 }
 81 
 82 inline Instruction* wasmExceptionInstructions()
 83 {
 84     return bitwise_cast<Instruction*>(Data::s_wasmExceptionInstructions);
 85 }
 86 
 87 inline Opcode* opcodeMap()
 88 {
 89     return g_opcodeMap;
 90 }
 91 
 92 inline Opcode* opcodeMapWide16()
 93 {
 94     return g_opcodeMapWide16;
 95 }
 96 
 97 inline Opcode* opcodeMapWide32()
 98 {
 99     return g_opcodeMapWide32;
100 }
101 
102 inline Opcode getOpcode(OpcodeID id)
103 {
104 #if ENABLE(COMPUTED_GOTO_OPCODES)
105     return g_opcodeMap[id];
106 #else
107     return static_cast<Opcode>(id);
108 #endif
109 }
110 
111 inline Opcode getOpcodeWide16(OpcodeID id)
112 {
113 #if ENABLE(COMPUTED_GOTO_OPCODES)
114     return g_opcodeMapWide16[id];
115 #else
116     return static_cast<Opcode>(id + numOpcodeIDs);
117 #endif
118 }
119 
120 inline Opcode getOpcodeWide32(OpcodeID id)
121 {
122 #if ENABLE(COMPUTED_GOTO_OPCODES)
123     return g_opcodeMapWide32[id];
124 #else
125     return static_cast<Opcode>(id + numOpcodeIDs*2);
126 #endif
127 }
128 
129 template<PtrTag tag>
130 ALWAYS_INLINE MacroAssemblerCodePtr<tag> getCodePtr(OpcodeID opcodeID)
131 {
132     void* address = reinterpret_cast<void*>(getOpcode(opcodeID));
133     address = retagCodePtr<BytecodePtrTag, tag>(address);
134     return MacroAssemblerCodePtr<tag>::createFromExecutableAddress(address);
135 }
136 
137 template<PtrTag tag>
138 ALWAYS_INLINE MacroAssemblerCodePtr<tag> getWide16CodePtr(OpcodeID opcodeID)
139 {
140     void* address = reinterpret_cast<void*>(getOpcodeWide16(opcodeID));
141     address = retagCodePtr<BytecodePtrTag, tag>(address);
142     return MacroAssemblerCodePtr<tag>::createFromExecutableAddress(address);
143 }
144 
145 template<PtrTag tag>
146 ALWAYS_INLINE MacroAssemblerCodePtr<tag> getWide32CodePtr(OpcodeID opcodeID)
147 {
148     void* address = reinterpret_cast<void*>(getOpcodeWide32(opcodeID));
149     address = retagCodePtr<BytecodePtrTag, tag>(address);
150     return MacroAssemblerCodePtr<tag>::createFromExecutableAddress(address);
151 }
152 
153 template<PtrTag tag>
154 ALWAYS_INLINE MacroAssemblerCodePtr<tag> getCodePtr(const Instruction& instruction)
155 {
156     if (instruction.isWide16())
157         return getWide16CodePtr<tag>(instruction.opcodeID());
158     if (instruction.isWide32())
159         return getWide32CodePtr<tag>(instruction.opcodeID());
160     return getCodePtr<tag>(instruction.opcodeID());
161 }
162 
163 template<PtrTag tag>
164 ALWAYS_INLINE MacroAssemblerCodeRef<tag> getCodeRef(OpcodeID opcodeID)
165 {
166     return MacroAssemblerCodeRef<tag>::createSelfManagedCodeRef(getCodePtr<tag>(opcodeID));
167 }
168 
169 #if ENABLE(JIT)
170 template<PtrTag tag>
171 ALWAYS_INLINE LLIntCode getCodeFunctionPtr(OpcodeID opcodeID)
172 {
173     ASSERT(opcodeID >= NUMBER_OF_BYTECODE_IDS);
174 #if COMPILER(MSVC)
175     return reinterpret_cast<LLIntCode>(getCodePtr<tag>(opcodeID).executableAddress());
176 #else
177     return reinterpret_cast<LLIntCode>(getCodePtr<tag>(opcodeID).template executableAddress());
178 #endif
179 }
180 
181 #else
182 ALWAYS_INLINE void* getCodePtr(OpcodeID id)
183 {
184     return reinterpret_cast<void*>(getOpcode(id));
185 }
186 
187 ALWAYS_INLINE void* getWide16CodePtr(OpcodeID id)
188 {
189     return reinterpret_cast<void*>(getOpcodeWide16(id));
190 }
191 
192 ALWAYS_INLINE void* getWide32CodePtr(OpcodeID id)
193 {
194     return reinterpret_cast<void*>(getOpcodeWide32(id));
195 }
196 #endif
197 
198 ALWAYS_INLINE void* getCodePtr(JSC::EncodedJSValue glueHelper())
199 {
200     return bitwise_cast<void*>(glueHelper);
201 }
