31 namespace WebCore {
32 
33 template<typename T> struct Converter<IDLSerializedScriptValue<T>> : DefaultConverter<IDLSerializedScriptValue<T>> {
34     static RefPtr<T> convert(JSC::ExecState& state, JSC::JSValue value)
35     {
36         return T::create(state, value);
37     }
38 };
39 
40 template<typename T> struct JSConverter<IDLSerializedScriptValue<T>> {
41     static constexpr bool needsState = true;
42     static constexpr bool needsGlobalObject = true;
43 
44     static JSC::JSValue convert(JSC::ExecState& state, JSDOMGlobalObject& globalObject, RefPtr<T> value)
45     {
46         return value ? value->deserialize(state, &globalObject) : JSC::jsNull();
47     }
48 };
