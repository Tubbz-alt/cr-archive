33 namespace WebCore {
34 
35 class CachedImage;
36 class CachedResourceLoader;

37 class Document;
38 









39 class CSSImageSetValue final : public CSSValueList {
40 public:
41     static Ref<CSSImageSetValue> create(LoadedFromOpaqueSource loadedFromOpaqueSource)
42     {
43         return adoptRef(*new CSSImageSetValue(loadedFromOpaqueSource));
44     }
45     ~CSSImageSetValue();
46 
47     std::pair<CachedImage*, float>  loadBestFitImage(CachedResourceLoader&, const ResourceLoaderOptions&);
48     CachedImage* cachedImage() const { return m_cachedImage.get(); }
49 
50     String customCSSText() const;
51 
52     struct ImageWithScale {
53         URL imageURL;
54         float scaleFactor;
55     };
56 
57     bool traverseSubresources(const WTF::Function<bool (const CachedResource&)>& handler) const;
58 
59     void updateDeviceScaleFactor(const Document&);
60 
61     URL bestImageForScaleFactorURL() { return bestImageForScaleFactor().imageURL; }
62 
63 protected:
64     ImageWithScale bestImageForScaleFactor();
65 
66 private:
67     explicit CSSImageSetValue(LoadedFromOpaqueSource);
68 
69     void fillImageSet();
70     static inline bool compareByScaleFactor(ImageWithScale first, ImageWithScale second) { return first.scaleFactor < second.scaleFactor; }
71 
72     CachedResourceHandle<CachedImage> m_cachedImage;
73     bool m_accessedBestFitImage { false };
74     float m_bestFitImageScaleFactor { 1 };
75     float m_deviceScaleFactor { 1 };
76 
77     Vector<ImageWithScale> m_imagesInSet;
78     LoadedFromOpaqueSource m_loadedFromOpaqueSource { LoadedFromOpaqueSource::No };
79 };
80 
81 } // namespace WebCore
