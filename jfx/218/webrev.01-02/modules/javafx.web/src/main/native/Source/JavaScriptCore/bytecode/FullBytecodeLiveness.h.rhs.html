34 class BytecodeLivenessAnalysis;

36 
37 // Note: Full bytecode liveness does not track any information about the liveness of temps.
38 // If you want tmp liveness for a checkpoint ask tmpLivenessForCheckpoint.
39 class FullBytecodeLiveness {
40     WTF_MAKE_FAST_ALLOCATED;
41 public:
42     const FastBitVector& getLiveness(BytecodeIndex bytecodeIndex, LivenessCalculationPoint point) const
43     {
44         // FIXME: What should this do when we have checkpoints?
45         switch (point) {
46         case LivenessCalculationPoint::BeforeUse:
47             return m_beforeUseVector[bytecodeIndex.offset()];
48         case LivenessCalculationPoint::AfterUse:
49             return m_afterUseVector[bytecodeIndex.offset()];
50         }
51         RELEASE_ASSERT_NOT_REACHED();
52     }
53 
54     bool virtualRegisterIsLive(VirtualRegister reg, BytecodeIndex bytecodeIndex, LivenessCalculationPoint point) const
55     {
56         return virtualRegisterIsAlwaysLive(reg) || virtualRegisterThatIsNotAlwaysLiveIsLive(getLiveness(bytecodeIndex, point), reg);
57     }
58 
59 private:
60     friend class BytecodeLivenessAnalysis;
61 
62     // FIXME: Use FastBitVector's view mechanism to make them compact.
63     // https://bugs.webkit.org/show_bug.cgi?id=204427<Paste>
64     Vector<FastBitVector, 0, UnsafeVectorOverflow> m_beforeUseVector;
65     Vector<FastBitVector, 0, UnsafeVectorOverflow> m_afterUseVector;
66 };
