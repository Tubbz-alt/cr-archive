26 #include "config.h"
27 #include "StyleChange.h"
28 
29 #include "RenderStyle.h"
30 
31 namespace WebCore {
32 namespace Style {
33 
34 Change determineChange(const RenderStyle& s1, const RenderStyle& s2)
35 {
36     if (s1.display() != s2.display())
37         return Detach;
38     if (s1.hasPseudoStyle(PseudoId::FirstLetter) != s2.hasPseudoStyle(PseudoId::FirstLetter))
39         return Detach;
40     // We just detach if a renderer acquires or loses a column-span, since spanning elements
41     // typically won't contain much content.
42     if (s1.columnSpan() != s2.columnSpan())
43         return Detach;
44     if (!s1.contentDataEquivalent(&s2))
45         return Detach;
46     // When text-combine property has been changed, we need to prepare a separate renderer object.
47     // When text-combine is on, we use RenderCombineText, otherwise RenderText.
48     // https://bugs.webkit.org/show_bug.cgi?id=55069
49     if (s1.hasTextCombine() != s2.hasTextCombine())
50         return Detach;
51 
52     if (!s1.inheritedEqual(s2))
53         return Inherit;

54 
55     if (!s1.descendantAffectingNonInheritedPropertiesEqual(s2))
56         return Inherit;
57 
58     if (s1 != s2)
59         return NoInherit;

60 
61     // If the pseudoStyles have changed, we want any StyleChange that is not NoChange
62     // because setStyle will do the right thing with anything else.
63     if (s1.hasAnyPublicPseudoStyles()) {
64         for (PseudoId pseudoId = PseudoId::FirstPublicPseudoId; pseudoId < PseudoId::FirstInternalPseudoId; pseudoId = static_cast<PseudoId>(static_cast<unsigned>(pseudoId) + 1)) {
65             if (s1.hasPseudoStyle(pseudoId)) {
66                 RenderStyle* ps2 = s2.getCachedPseudoStyle(pseudoId);
67                 if (!ps2)
68                     return NoInherit;
69                 RenderStyle* ps1 = s1.getCachedPseudoStyle(pseudoId);
70                 if (!ps1 || *ps1 != *ps2)
71                     return NoInherit;
72             }
73         }
74     }
75 
76     return NoChange;
77 }
78 
79 }
80 }
