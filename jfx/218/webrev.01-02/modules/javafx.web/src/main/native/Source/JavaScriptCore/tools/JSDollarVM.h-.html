26 #pragma once
27 
28 #include "JSObject.h"
29 
30 namespace JSC {
31 
32 class JSDollarVM final : public JSNonFinalObject {
33 public:
34     typedef JSNonFinalObject Base;
35 
36     DECLARE_EXPORT_INFO;
37 
38     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
39     {
40         return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
41     }
42 
43     static JSDollarVM* create(VM& vm, Structure* structure)
44     {
45         JSDollarVM* instance = new (NotNull, allocateCell<JSDollarVM>(vm.heap)) JSDollarVM(vm, structure);
46         instance->finishCreation(vm);
47         return instance;
48     }
49 
50 private:
51     JSDollarVM(VM& vm, Structure* structure)
52         : Base(vm, structure)
53     {
54     }
55 
56     void finishCreation(VM&);
57     void addFunction(VM&, JSGlobalObject*, const char* name, NativeFunction, unsigned arguments);
58     void addConstructibleFunction(VM&, JSGlobalObject*, const char* name, NativeFunction, unsigned arguments);
59 };
60 
61 } // namespace JSC
