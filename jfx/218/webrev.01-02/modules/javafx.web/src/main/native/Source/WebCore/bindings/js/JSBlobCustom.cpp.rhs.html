40 using namespace JSC;

42 JSValue toJSNewlyCreated(JSC::JSGlobalObject*, JSDOMGlobalObject* globalObject, Ref<Blob>&& blob)
43 {
44     if (is<File>(blob))
45         return createWrapper<File>(globalObject, WTFMove(blob));
46     return createWrapper<Blob>(globalObject, WTFMove(blob));
47 }
48 
49 JSValue toJS(JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, Blob& blob)
50 {
51     return wrap(lexicalGlobalObject, globalObject, blob);
52 }
