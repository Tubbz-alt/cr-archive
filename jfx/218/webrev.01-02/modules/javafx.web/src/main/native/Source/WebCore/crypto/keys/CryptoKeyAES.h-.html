26 #pragma once
27 
28 #include "CryptoAlgorithmIdentifier.h"
29 #include "CryptoKey.h"
30 #include "ExceptionOr.h"
31 #include <wtf/Function.h>
32 #include <wtf/Vector.h>
33 
34 #if ENABLE(WEB_CRYPTO)
35 
36 namespace WebCore {
37 
38 class CryptoAlgorithmParameters;
39 
40 struct JsonWebKey;
41 
42 class CryptoKeyAES final : public CryptoKey {
43 public:
44     static const int s_length128 = 128;
45     static const int s_length192 = 192;
46     static const int s_length256 = 256;
47 
48     static Ref<CryptoKeyAES> create(CryptoAlgorithmIdentifier algorithm, const Vector<uint8_t>& key, bool extractable, CryptoKeyUsageBitmap usage)
49     {
50         return adoptRef(*new CryptoKeyAES(algorithm, key, extractable, usage));
51     }
52     virtual ~CryptoKeyAES();
53 
54     static bool isValidAESAlgorithm(CryptoAlgorithmIdentifier);
55 
56     static RefPtr<CryptoKeyAES> generate(CryptoAlgorithmIdentifier, size_t lengthBits, bool extractable, CryptoKeyUsageBitmap);
57     static RefPtr<CryptoKeyAES> importRaw(CryptoAlgorithmIdentifier, Vector<uint8_t>&& keyData, bool extractable, CryptoKeyUsageBitmap);
58     using CheckAlgCallback = Function<bool(size_t, const String&)>;
59     static RefPtr<CryptoKeyAES> importJwk(CryptoAlgorithmIdentifier, JsonWebKey&&, bool extractable, CryptoKeyUsageBitmap, CheckAlgCallback&&);
60 
61     CryptoKeyClass keyClass() const final { return CryptoKeyClass::AES; }
62 
63     const Vector<uint8_t>& key() const { return m_key; }
64     JsonWebKey exportJwk() const;
65 
66     static ExceptionOr<size_t> getKeyLength(const CryptoAlgorithmParameters&);
67 
68 private:
69     CryptoKeyAES(CryptoAlgorithmIdentifier, const Vector<uint8_t>& key, bool extractable, CryptoKeyUsageBitmap);
70     CryptoKeyAES(CryptoAlgorithmIdentifier, Vector<uint8_t>&& key, bool extractable, CryptoKeyUsageBitmap);
71 
72     KeyAlgorithm algorithm() const final;
73 
74     Vector<uint8_t> m_key;
75 };
76 
77 } // namespace WebCore
78 
79 SPECIALIZE_TYPE_TRAITS_CRYPTO_KEY(CryptoKeyAES, CryptoKeyClass::AES)
80 
81 #endif // ENABLE(WEB_CRYPTO)
