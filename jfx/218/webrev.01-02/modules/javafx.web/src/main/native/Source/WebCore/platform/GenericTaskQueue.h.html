 39 template <typename T>
 40 class TaskDispatcher {
 41 public:
 42     explicit TaskDispatcher(T* context)
 43         : m_context(context)
 44     {
 45     }
 46 
 47     void postTask(Function<void()>&& function)
 48     {
 49         ASSERT(m_context);
 50         m_context->enqueueTaskForDispatcher(WTFMove(function));
 51     }
 52 
 53 private:
 54     T* m_context;
 55 };
 56 
 57 template<>
 58 class TaskDispatcher<Timer> : public CanMakeWeakPtr<TaskDispatcher<Timer>> {
 59 public:
 60     TaskDispatcher();
 61     void postTask(Function<void()>&&);
 62 
 63 private:
 64     static Timer& sharedTimer();
 65     static WTF::Lock& sharedLock();
 66     static void sharedTimerFired();
 67     static Deque<WeakPtr<TaskDispatcher<Timer>>>& pendingDispatchers();
 68 
 69     void dispatchOneTask();
 70 
 71     Deque<Function<void()>> m_pendingTasks;
 72 };
 73 
 74 template <typename T>
 75 class GenericTaskQueue : public CanMakeWeakPtr<GenericTaskQueue<T>> {
 76     WTF_MAKE_FAST_ALLOCATED;
 77 public:
 78     GenericTaskQueue()
 79         : m_dispatcher()
 80     {
 81     }
 82 
 83     explicit GenericTaskQueue(T& t)
 84         : m_dispatcher(&t)
 85     {
 86     }
 87 
 88     explicit GenericTaskQueue(T* t)
 89         : m_dispatcher(t)
 90         , m_isClosed(!t)
 91     {
 92     }
 93 
 94     typedef WTF::Function<void ()> TaskFunction;
 95 
 96     void enqueueTask(TaskFunction&& task)
 97     {
 98         if (m_isClosed)
 99             return;
100 
101         ++m_pendingTasks;
102         m_dispatcher.postTask([weakThis = makeWeakPtr(*this), task = WTFMove(task)] {
103             if (!weakThis)
104                 return;
105             ASSERT(weakThis->m_pendingTasks);
106             --weakThis->m_pendingTasks;
107             task();
108         });
109     }
110 
111     void close()
112     {
113         cancelAllTasks();
114         m_isClosed = true;
115     }
116 
117     void cancelAllTasks()
118     {
119         CanMakeWeakPtr<GenericTaskQueue<T>>::weakPtrFactory().revokeAll();
120         m_pendingTasks = 0;
121     }
122 
123     bool hasPendingTasks() const { return m_pendingTasks; }
124     bool isClosed() const { return m_isClosed; }
125 
126 private:
127     TaskDispatcher<T> m_dispatcher;
128     unsigned m_pendingTasks { 0 };
129     bool m_isClosed { false };
130 };
