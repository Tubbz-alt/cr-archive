 26 #pragma once
 27 
 28 #include "EventTarget.h"
 29 #include <wtf/IsoMalloc.h>
 30 #include <wtf/Optional.h>
 31 #include <wtf/Vector.h>
 32 #include <wtf/WeakPtr.h>
 33 
 34 namespace WebCore {
 35 
 36 class ClipboardItem;
 37 class DeferredPromise;
 38 class Frame;
 39 class Navigator;
 40 class Pasteboard;
 41 class PasteboardCustomData;
 42 
 43 class Clipboard final : public RefCounted<Clipboard>, public EventTargetWithInlineData, public CanMakeWeakPtr<Clipboard> {
 44     WTF_MAKE_ISO_ALLOCATED(Clipboard);
 45 public:
 46     static Ref<Clipboard> create(Navigator&);
 47     ~Clipboard();
 48 
 49     EventTargetInterface eventTargetInterface() const final;
 50     ScriptExecutionContext* scriptExecutionContext() const final;
 51 
 52     Frame* frame() const;
 53     Navigator* navigator();
 54 
 55     using RefCounted::ref;
 56     using RefCounted::deref;
 57 
 58     void readText(Ref<DeferredPromise>&&);
 59     void writeText(const String& data, Ref<DeferredPromise>&&);
 60 
 61     void read(Ref<DeferredPromise>&&);
 62     void write(const Vector<RefPtr<ClipboardItem>>& data, Ref<DeferredPromise>&&);
 63 
 64     void getType(ClipboardItem&, const String& type, Ref<DeferredPromise>&&);
 65 
 66 private:
 67     Clipboard(Navigator&);
 68 
 69     struct Session {
 70         std::unique_ptr<Pasteboard> pasteboard;
 71         Vector<Ref<ClipboardItem>> items;
 72         int64_t changeCount;
 73     };
 74 
 75     void refEventTarget() final { ref(); }
 76     void derefEventTarget() final { deref(); }
 77 
 78     Pasteboard& activePasteboard();
 79 
 80     class ItemWriter : public RefCounted<ItemWriter> {
 81     public:
 82         static Ref<ItemWriter> create(Clipboard& clipboard, Ref<DeferredPromise>&& promise)
 83         {
 84             return adoptRef(*new ItemWriter(clipboard, WTFMove(promise)));
 85         }
 86 
 87         ~ItemWriter();
 88 
 89         void write(const Vector<RefPtr<ClipboardItem>>&);
 90         void invalidate();
 91 
 92     private:
 93         ItemWriter(Clipboard&, Ref<DeferredPromise>&&);
 94 
 95         void setData(Optional<PasteboardCustomData>&&, size_t index);
 96         void didSetAllData();
 97         void reject();
 98 
 99         WeakPtr<Clipboard> m_clipboard;
100         Vector<Optional<PasteboardCustomData>> m_dataToWrite;
101         RefPtr<DeferredPromise> m_promise;
102         unsigned m_pendingItemCount;
103         std::unique_ptr<Pasteboard> m_pasteboard;
104 #if PLATFORM(COCOA)
105         int64_t m_changeCountAtStart { 0 };
106 #endif
107     };
108 
109     void didResolveOrReject(ItemWriter&);
110 
111     Optional<Session> m_activeSession;
112     WeakPtr<Navigator> m_navigator;
113     Vector<Optional<PasteboardCustomData>> m_dataToWrite;
114     RefPtr<ItemWriter> m_activeItemWriter;
115 };
116 
117 } // namespace WebCore
