37 inline double secondsToWebAnimationsAPITime(const Seconds time)
38 {
39     // Precision of time values
40     // https://drafts.csswg.org/web-animations-1/#precision-of-time-values
41 
42     // The internal representation of time values is implementation dependent however, it is RECOMMENDED that user
43     // agents be able to represent input time values with microsecond precision so that a time value (which nominally
44     // represents milliseconds) of 0.001 is distinguishable from 0.0.
45     auto roundedTime = std::round(time.microseconds()) / 1000;
46     if (roundedTime == -0)
47         return 0;
48     return roundedTime;
49 }
50 
51 const auto timeEpsilon = Seconds::fromMilliseconds(0.001);

53 struct WebAnimationsMarkableDoubleTraits {
54     static bool isEmptyValue(double value)
55     {
56         return std::isnan(value);
57     }
58 
59     static constexpr double emptyValue()
60     {
61         return std::numeric_limits<double>::quiet_NaN();
62     }
63 };
