 52 void BlockFormattingContext::layout() const
 53 {
 54     // 9.4.1 Block formatting contexts
 55     // In a block formatting context, boxes are laid out one after the other, vertically, beginning at the top of a containing block.
 56     // The vertical distance between two sibling boxes is determined by the 'margin' properties.
 57     // Vertical margins between adjacent block-level boxes in a block formatting context collapse.
 58     if (!is<Container>(root()))
 59         return;
 60 
 61     LOG_WITH_STREAM(FormattingContextLayout, stream << "[Start] -> block formatting context -> formatting root(" << &root() << ")");



 62 
 63     auto& formattingRoot = downcast<Container>(root());
 64     LayoutQueue layoutQueue;
 65     FloatingContext floatingContext(formattingState().floatingState());









































 66     // This is a post-order tree traversal layout.
 67     // The root container layout is done in the formatting context it lives in, not that one it creates, so let's start with the first child.
 68     if (auto* firstChild = formattingRoot.firstInFlowOrFloatingChild())
 69         layoutQueue.append(firstChild);
 70     // 1. Go all the way down to the leaf node
 71     // 2. Compute static position and width as we traverse down
 72     // 3. As we climb back on the tree, compute height and finialize position
 73     // (Any subtrees with new formatting contexts need to layout synchronously)
 74     while (!layoutQueue.isEmpty()) {
 75         // Traverse down on the descendants and compute width/static position until we find a leaf node.
 76         while (true) {
 77             auto& layoutBox = *layoutQueue.last();







 78 
 79             if (layoutBox.establishesFormattingContext()) {
 80                 layoutFormattingContextRoot(floatingContext, layoutBox);
 81                 layoutQueue.removeLast();
 82                 // Since this box is a formatting context root, it takes care of its entire subtree.
 83                 // Continue with next sibling if exists.
 84                 if (!layoutBox.nextInFlowOrFloatingSibling())
 85                     break;
 86                 layoutQueue.append(layoutBox.nextInFlowOrFloatingSibling());
 87                 continue;

 88             }
 89 
 90             LOG_WITH_STREAM(FormattingContextLayout, stream << "[Compute] -> [Position][Border][Padding][Width][Margin] -> for layoutBox(" << &layoutBox << ")");
 91             computeBorderAndPadding(layoutBox);
 92             computeWidthAndMargin(layoutBox);
 93             computeStaticPosition(floatingContext, layoutBox);
 94             if (!is<Container>(layoutBox) || !downcast<Container>(layoutBox).hasInFlowOrFloatingChild())
 95                 break;
 96             layoutQueue.append(downcast<Container>(layoutBox).firstInFlowOrFloatingChild());
 97         }
 98 
 99         // Climb back on the ancestors and compute height/final position.
100         while (!layoutQueue.isEmpty()) {
101             // All inflow descendants (if there are any) are laid out by now. Let's compute the box's height.
102             auto& layoutBox = *layoutQueue.takeLast();


103 
104             LOG_WITH_STREAM(FormattingContextLayout, stream << "[Compute] -> [Height][Margin] -> for layoutBox(" << &layoutBox << ")");
105             // Formatting root boxes are special-cased and they don't come here.
106             ASSERT(!layoutBox.establishesFormattingContext());
107             computeHeightAndMargin(layoutBox);
108             // Move in-flow positioned children to their final position.
109             placeInFlowPositionedChildren(layoutBox);
110             if (auto* nextSibling = layoutBox.nextInFlowOrFloatingSibling()) {
111                 layoutQueue.append(nextSibling);
112                 break;









113             }



114         }
115     }
116     // Place the inflow positioned children.
117     placeInFlowPositionedChildren(formattingRoot);
118     LOG_WITH_STREAM(FormattingContextLayout, stream << "[End] -> block formatting context -> formatting root(" << &root() << ")");
119 }

121 Optional<LayoutUnit> BlockFormattingContext::usedAvailableWidthForFloatAvoider(const FloatingContext& floatingContext, const Box& layoutBox) const
122 {
123     // Normally the available width for an in-flow block level box is the width of the containing block's content box.
124     // However (and can't find it anywhere in the spec) non-floating positioned float avoider block level boxes are constrained by existing floats.
125     if (!layoutBox.isFloatAvoider() || layoutBox.isFloatingPositioned())
126         return { };
127     auto& floatingState = floatingContext.floatingState();
128     if (floatingState.isEmpty())
129         return { };
130     // Vertical static position is not computed yet, so let's just estimate it for now.
131     auto& formattingRoot = downcast<Container>(root());
132     auto verticalPosition = FormattingContext::mapTopToAncestor(layoutState(), layoutBox, formattingRoot);
133     auto constraints = floatingState.constraints({ verticalPosition }, formattingRoot);

134     if (!constraints.left && !constraints.right)
135         return { };
136     auto& containingBlock = downcast<Container>(*layoutBox.containingBlock());
137     auto& containingBlockDisplayBox = layoutState().displayBoxForLayoutBox(containingBlock);
138     auto availableWidth = containingBlockDisplayBox.contentBoxWidth();
139 
140     LayoutUnit containingBlockLeft;
141     LayoutUnit containingBlockRight = containingBlockDisplayBox.right();
142     if (&containingBlock != &formattingRoot) {
143         // Move containing block left/right to the root's coordinate system.
144         containingBlockLeft = FormattingContext::mapLeftToAncestor(layoutState(), containingBlock, formattingRoot);
145         containingBlockRight = FormattingContext::mapRightToAncestor(layoutState(), containingBlock, formattingRoot);
146     }
147     auto containingBlockContentBoxLeft = containingBlockLeft + containingBlockDisplayBox.borderLeft() + containingBlockDisplayBox.paddingLeft().valueOr(0);
148     auto containingBlockContentBoxRight = containingBlockRight - containingBlockDisplayBox.borderRight() + containingBlockDisplayBox.paddingRight().valueOr(0);
149 
150     // Shrink the available space if the floats are actually intruding at this vertical position.
151     if (constraints.left)
152         availableWidth -= std::max<LayoutUnit>(0, constraints.left->x - containingBlockContentBoxLeft);
153     if (constraints.right)
154         availableWidth -= std::max<LayoutUnit>(0, containingBlockContentBoxRight - constraints.right->x);
155     return availableWidth;
156 }

158 void BlockFormattingContext::layoutFormattingContextRoot(FloatingContext& floatingContext, const Box& layoutBox) const
159 {
160     ASSERT(layoutBox.establishesFormattingContext());
161     // Start laying out this formatting root in the formatting contenxt it lives in.
162     LOG_WITH_STREAM(FormattingContextLayout, stream << "[Compute] -> [Position][Border][Padding][Width][Margin] -> for layoutBox(" << &layoutBox << ")");
163     computeBorderAndPadding(layoutBox);
164     computeStaticVerticalPosition(floatingContext, layoutBox);
165 
166     computeWidthAndMargin(layoutBox, usedAvailableWidthForFloatAvoider(floatingContext, layoutBox));
167     computeStaticHorizontalPosition(layoutBox);
168     // Swich over to the new formatting context (the one that the root creates).
169     auto formattingContext = layoutState().createFormattingContext(layoutBox);
170     formattingContext->layout();
171 
172     // Come back and finalize the root's geometry.
173     LOG_WITH_STREAM(FormattingContextLayout, stream << "[Compute] -> [Height][Margin] -> for layoutBox(" << &layoutBox << ")");
174     computeHeightAndMargin(layoutBox);
175     // Now that we computed the root's height, we can go back and layout the out-of-flow content.
176     formattingContext->layoutOutOfFlowContent();
177 
178     // Float related final positioning.
179     if (layoutBox.isFloatingPositioned()) {
180         computeFloatingPosition(floatingContext, layoutBox);
181         floatingContext.floatingState().append(layoutBox);
182     } else if (layoutBox.establishesBlockFormattingContext())
183         computePositionToAvoidFloats(floatingContext, layoutBox);
184 }

186 void BlockFormattingContext::placeInFlowPositionedChildren(const Box& layoutBox) const
187 {
188     if (!is<Container>(layoutBox))
189         return;
190 
191     LOG_WITH_STREAM(FormattingContextLayout, stream << "Start: move in-flow positioned children -> parent: " << &layoutBox);
192     auto& container = downcast<Container>(layoutBox);
193     for (auto& childBox : childrenOfType<Box>(container)) {
194         if (!childBox.isInFlowPositioned())
195             continue;
196 
197         auto computeInFlowPositionedPosition = [&] {
198             auto& layoutState = this->layoutState();
199             auto positionOffset = Geometry::inFlowPositionedPositionOffset(layoutState, childBox);
200 
201             auto& displayBox = layoutState.displayBoxForLayoutBox(childBox);
202             auto topLeft = displayBox.topLeft();
203 
204             topLeft.move(positionOffset);
205 
206             displayBox.setTopLeft(topLeft);
207         };
208 
209         computeInFlowPositionedPosition();
210     }
211     LOG_WITH_STREAM(FormattingContextLayout, stream << "End: move in-flow positioned children -> parent: " << &layoutBox);
212 }

214 void BlockFormattingContext::computeStaticVerticalPosition(const FloatingContext& floatingContext, const Box& layoutBox) const
215 {
216     auto& layoutState = this->layoutState();
217     layoutState.displayBoxForLayoutBox(layoutBox).setTop(Geometry::staticVerticalPosition(layoutState, layoutBox));
218     if (layoutBox.hasFloatClear())
219         computeEstimatedVerticalPositionForFloatClear(floatingContext, layoutBox);
220     else if (layoutBox.establishesFormattingContext())
221         computeEstimatedVerticalPositionForFormattingRoot(layoutBox);
222 }

235 void BlockFormattingContext::computeEstimatedVerticalPosition(const Box& layoutBox) const
236 {
237     auto& layoutState = this->layoutState();
238     auto estimatedMarginBefore = MarginCollapse::estimatedMarginBefore(layoutState, layoutBox);
239     setEstimatedMarginBefore(layoutBox, estimatedMarginBefore);
240 
241     auto& displayBox = layoutState.displayBoxForLayoutBox(layoutBox);
242     auto nonCollapsedValues = UsedVerticalMargin::NonCollapsedValues { estimatedMarginBefore.nonCollapsedValue, { } };
243     auto collapsedValues = UsedVerticalMargin::CollapsedValues { estimatedMarginBefore.collapsedValue, { }, estimatedMarginBefore.isCollapsedThrough };

244     auto verticalMargin = UsedVerticalMargin { nonCollapsedValues, collapsedValues };
245     displayBox.setVerticalMargin(verticalMargin);
246     displayBox.setTop(verticalPositionWithMargin(layoutBox, verticalMargin));
247 #if !ASSERT_DISABLED
248     displayBox.setHasEstimatedMarginBefore();
249 #endif
250 }

252 void BlockFormattingContext::computeEstimatedVerticalPositionForAncestors(const Box& layoutBox) const
253 {
254     // We only need to estimate margin top for float related layout (formatting context roots avoid floats).
255     ASSERT(layoutBox.isFloatingPositioned() || layoutBox.hasFloatClear() || layoutBox.establishesBlockFormattingContext() || layoutBox.establishesInlineFormattingContext());
256 
257     // In order to figure out whether a box should avoid a float, we need to know the final positions of both (ignore relative positioning for now).
258     // In block formatting context the final position for a normal flow box includes
259     // 1. the static position and
260     // 2. the corresponding (non)collapsed margins.
261     // Now the vertical margins are computed when all the descendants are finalized, because the margin values might be depending on the height of the box
262     // (and the height might be based on the content).
263     // So when we get to the point where we intersect the box with the float to decide if the box needs to move, we don't yet have the final vertical position.
264     //
265     // The idea here is that as long as we don't cross the block formatting context boundary, we should be able to pre-compute the final top position.

266     for (auto* ancestor = layoutBox.containingBlock(); ancestor && !ancestor->establishesBlockFormattingContext(); ancestor = ancestor->containingBlock()) {
267         // FIXME: with incremental layout, we might actually have a valid (non-estimated) margin top as well.
268         if (hasEstimatedMarginBefore(*ancestor))
269             return;
270         computeEstimatedVerticalPosition(*ancestor);








271     }
272 }

274 void BlockFormattingContext::computeEstimatedVerticalPositionForFormattingRoot(const Box& layoutBox) const
275 {
276     ASSERT(layoutBox.establishesFormattingContext());
277     ASSERT(!layoutBox.hasFloatClear());









278 







279     if (layoutBox.isFloatingPositioned()) {
280         computeEstimatedVerticalPositionForAncestors(layoutBox);

281         return;
282     }






283 
284     computeEstimatedVerticalPosition(layoutBox);
285     computeEstimatedVerticalPositionForAncestors(layoutBox);
286 
287     // If the inline formatting root is also the root for the floats (happens when the root box also establishes a block formatting context)
288     // the floats are in the coordinate system of this root. No need to find the final vertical position.
289     auto inlineContextInheritsFloats = layoutBox.establishesInlineFormattingContextOnly();
290     if (inlineContextInheritsFloats) {
291         computeEstimatedVerticalPosition(layoutBox);
292         computeEstimatedVerticalPositionForAncestors(layoutBox);
293     }
294 }

296 void BlockFormattingContext::computeEstimatedVerticalPositionForFloatClear(const FloatingContext& floatingContext, const Box& layoutBox) const
297 {
298     ASSERT(layoutBox.hasFloatClear());
299     if (floatingContext.floatingState().isEmpty())
300         return;
301     // The static position with clear requires margin esitmation to see if clearance is needed.
302     computeEstimatedVerticalPosition(layoutBox);
303     computeEstimatedVerticalPositionForAncestors(layoutBox);
304     auto verticalPositionAndClearance = floatingContext.verticalPositionWithClearance(layoutBox);
305     if (!verticalPositionAndClearance.position) {
306         ASSERT(!verticalPositionAndClearance.clearance);
307         return;
308     }
309 
310     auto& displayBox = layoutState().displayBoxForLayoutBox(layoutBox);
311     ASSERT(*verticalPositionAndClearance.position >= displayBox.top());
312     displayBox.setTop(*verticalPositionAndClearance.position);
313     if (verticalPositionAndClearance.clearance)
314         displayBox.setHasClearance();
315 }

320     for (auto* ancestor = layoutBox.containingBlock(); ancestor && !ancestor->establishesBlockFormattingContext(); ancestor = ancestor->containingBlock()) {

329 void BlockFormattingContext::computeFloatingPosition(const FloatingContext& floatingContext, const Box& layoutBox) const
330 {
331     ASSERT(layoutBox.isFloatingPositioned());
332     ASSERT(hasPrecomputedMarginBefore(layoutBox));
333     layoutState().displayBoxForLayoutBox(layoutBox).setTopLeft(floatingContext.positionForFloat(layoutBox));

336 void BlockFormattingContext::computePositionToAvoidFloats(const FloatingContext& floatingContext, const Box& layoutBox) const
337 {
338     auto& layoutState = this->layoutState();
339     // Formatting context roots avoid floats.
340     ASSERT(layoutBox.establishesBlockFormattingContext());
341     ASSERT(!layoutBox.isFloatingPositioned());
342     ASSERT(!layoutBox.hasFloatClear());
343     ASSERT(hasPrecomputedMarginBefore(layoutBox));
344 
345     if (floatingContext.floatingState().isEmpty())
346         return;
347 
348     if (auto adjustedPosition = floatingContext.positionForFormattingContextRoot(layoutBox))
349         layoutState.displayBoxForLayoutBox(layoutBox).setTopLeft(*adjustedPosition);
350 }

352 void BlockFormattingContext::computeWidthAndMargin(const Box& layoutBox, Optional<LayoutUnit> usedAvailableWidth) const
353 {
354     auto& layoutState = this->layoutState();


355 
356     LayoutUnit availableWidth;
357     if (usedAvailableWidth)
358         availableWidth = *usedAvailableWidth;
359     else
360         availableWidth = layoutState.displayBoxForLayoutBox(*layoutBox.containingBlock()).contentBoxWidth();
361 
362     auto compute = [&](Optional<LayoutUnit> usedWidth) -> WidthAndMargin {
363         auto usedValues = UsedHorizontalValues { availableWidth, usedWidth, { } };
364         if (layoutBox.isInFlow())
365             return Geometry::inFlowWidthAndMargin(layoutState, layoutBox, usedValues);
366 
367         if (layoutBox.isFloatingPositioned())
368             return Geometry::floatingWidthAndMargin(layoutState, layoutBox, usedValues);
369 
370         ASSERT_NOT_REACHED();
371         return { };
372     };
373 
374     auto widthAndMargin = compute({ });
375 
376     if (auto maxWidth = Geometry::computedValueIfNotAuto(layoutBox.style().logicalMaxWidth(), availableWidth)) {

377         auto maxWidthAndMargin = compute(maxWidth);
378         if (widthAndMargin.width > maxWidthAndMargin.width)
379             widthAndMargin = maxWidthAndMargin;
380     }
381 
382     auto minWidth = Geometry::computedValueIfNotAuto(layoutBox.style().logicalMinWidth(), availableWidth).valueOr(0);
383     auto minWidthAndMargin = compute(minWidth);
384     if (widthAndMargin.width < minWidthAndMargin.width)
385         widthAndMargin = minWidthAndMargin;
386 
387     auto& displayBox = layoutState.displayBoxForLayoutBox(layoutBox);
388     displayBox.setContentBoxWidth(widthAndMargin.width);
389     displayBox.setHorizontalMargin(widthAndMargin.usedMargin);
390     displayBox.setHorizontalComputedMargin(widthAndMargin.computedMargin);
391 }

393 void BlockFormattingContext::computeHeightAndMargin(const Box& layoutBox) const
394 {
395     auto& layoutState = this->layoutState();
396 
397     auto compute = [&](UsedVerticalValues usedValues) -> HeightAndMargin {
398 
399         if (layoutBox.isInFlow())
400             return Geometry::inFlowHeightAndMargin(layoutState, layoutBox, usedValues);
401 
402         if (layoutBox.isFloatingPositioned())
403             return Geometry::floatingHeightAndMargin(layoutState, layoutBox, usedValues, UsedHorizontalValues { layoutState.displayBoxForLayoutBox(*layoutBox.containingBlock()).contentBoxWidth() });
404 
405         ASSERT_NOT_REACHED();
406         return { };
407     };
408 
409     auto heightAndMargin = compute({ });
410     if (auto maxHeight = Geometry::computedMaxHeight(layoutState, layoutBox)) {
411         if (heightAndMargin.height > *maxHeight) {
412             auto maxHeightAndMargin = compute({ *maxHeight });
413             // Used height should remain the same.
414             ASSERT((layoutState.inQuirksMode() && (layoutBox.isBodyBox() || layoutBox.isDocumentBox())) || maxHeightAndMargin.height == *maxHeight);
415             heightAndMargin = { *maxHeight, maxHeightAndMargin.nonCollapsedMargin };
416         }
417     }
418 
419     if (auto minHeight = Geometry::computedMinHeight(layoutState, layoutBox)) {
420         if (heightAndMargin.height < *minHeight) {
421             auto minHeightAndMargin = compute({ *minHeight });
422             // Used height should remain the same.
423             ASSERT((layoutState.inQuirksMode() && (layoutBox.isBodyBox() || layoutBox.isDocumentBox())) || minHeightAndMargin.height == *minHeight);
424             heightAndMargin = { *minHeight, minHeightAndMargin.nonCollapsedMargin };
425         }
426     }
427 
428     // 1. Compute collapsed margins.
429     // 2. Adjust vertical position using the collapsed values
430     // 3. Adjust previous in-flow sibling margin after using this margin.
431     auto collapsedMargin = MarginCollapse::collapsedVerticalValues(layoutState, layoutBox, heightAndMargin.nonCollapsedMargin);
432     auto verticalMargin = UsedVerticalMargin { heightAndMargin.nonCollapsedMargin, collapsedMargin };
433     auto& displayBox = layoutState.displayBoxForLayoutBox(layoutBox);
434 
435     // Out of flow boxes don't need vertical adjustment after margin collapsing.
436     if (layoutBox.isOutOfFlowPositioned()) {
437         ASSERT(!hasEstimatedMarginBefore(layoutBox));
438         displayBox.setContentBoxHeight(heightAndMargin.height);
439         displayBox.setVerticalMargin(verticalMargin);
440         return;
441     }
442 
443     ASSERT(!hasEstimatedMarginBefore(layoutBox) || estimatedMarginBefore(layoutBox).usedValue() == verticalMargin.before());
444     removeEstimatedMarginBefore(layoutBox);
445     displayBox.setTop(verticalPositionWithMargin(layoutBox, verticalMargin));
446     displayBox.setContentBoxHeight(heightAndMargin.height);
447     displayBox.setVerticalMargin(verticalMargin);
448 
449     MarginCollapse::updatePositiveNegativeMarginValues(layoutState, layoutBox);

450     // Adjust the previous sibling's margin bottom now that this box's vertical margin is computed.
451     MarginCollapse::updateMarginAfterForPreviousSibling(layoutState, layoutBox);
452 }

454 FormattingContext::IntrinsicWidthConstraints BlockFormattingContext::computedIntrinsicWidthConstraints() const
455 {
456     auto& layoutState = this->layoutState();
457     auto& formattingRoot = root();
458     auto& formattingState = this->formattingState();
459     ASSERT(!formattingState.intrinsicWidthConstraints());
460 
461     // Visit the in-flow descendants and compute their min/max intrinsic width if needed.
462     // 1. Go all the way down to the leaf node
463     // 2. Check if actually need to visit all the boxes as we traverse down (already computed, container's min/max does not depend on descendants etc)
464     // 3. As we climb back on the tree, compute min/max intrinsic width
465     // (Any subtrees with new formatting contexts need to layout synchronously)
466     Vector<const Box*> queue;
467     if (is<Container>(formattingRoot) && downcast<Container>(formattingRoot).hasInFlowOrFloatingChild())
468         queue.append(downcast<Container>(formattingRoot).firstInFlowOrFloatingChild());
469 
470     IntrinsicWidthConstraints constraints;
471     while (!queue.isEmpty()) {
472         while (true) {
473             auto& layoutBox = *queue.last();
474             auto hasInFlowOrFloatingChild = is<Container>(layoutBox) && downcast<Container>(layoutBox).hasInFlowOrFloatingChild();
475             auto skipDescendants = formattingState.intrinsicWidthConstraintsForBox(layoutBox) || !hasInFlowOrFloatingChild || layoutBox.establishesFormattingContext() || layoutBox.style().width().isFixed();
476             if (skipDescendants)
477                 break;
478             queue.append(downcast<Container>(layoutBox).firstInFlowOrFloatingChild());
479         }
480         // Compute min/max intrinsic width bottom up if needed.
481         while (!queue.isEmpty()) {
482             auto& layoutBox = *queue.takeLast();
483             auto desdendantConstraints = formattingState.intrinsicWidthConstraintsForBox(layoutBox);
484             if (!desdendantConstraints) {
485                 desdendantConstraints = Geometry::intrinsicWidthConstraints(layoutState, layoutBox);
486                 formattingState.setIntrinsicWidthConstraintsForBox(layoutBox, *desdendantConstraints);
487             }
488             constraints.minimum = std::max(constraints.minimum, desdendantConstraints->minimum);
489             constraints.maximum = std::max(constraints.maximum, desdendantConstraints->maximum);
490             // Move over to the next sibling or take the next box in the queue.
491             if (auto* nextSibling = layoutBox.nextInFlowOrFloatingSibling()) {
492                 queue.append(nextSibling);
493                 break;
494             }
495         }
496     }
497     formattingState.setIntrinsicWidthConstraints(constraints);
498     return constraints;
499 }

501 LayoutUnit BlockFormattingContext::verticalPositionWithMargin(const Box& layoutBox, const UsedVerticalMargin& verticalMargin) const
502 {
503     ASSERT(!layoutBox.isOutOfFlowPositioned());
504     // Now that we've computed the final margin before, let's shift the box's vertical position if needed.
505     // 1. Check if the box has clearance. If so, we've already precomputed/finalized the top value and vertical margin does not impact it anymore.
506     // 2. Check if the margin before collapses with the previous box's margin after. if not -> return previous box's bottom including margin after + marginBefore
507     // 3. Check if the previous box's margins collapse through. If not -> return previous box' bottom excluding margin after + marginBefore (they are supposed to be equal)
508     // 4. Go to previous box and start from step #1 until we hit the parent box.
509     auto& layoutState = this->layoutState();
510     auto& displayBox = layoutState.displayBoxForLayoutBox(layoutBox);
511     if (displayBox.hasClearance())
512         return displayBox.top();
513 
514     auto* currentLayoutBox = &layoutBox;
515     while (currentLayoutBox) {
516         if (!currentLayoutBox->previousInFlowSibling())
517             break;
518         auto& previousInFlowSibling = *currentLayoutBox->previousInFlowSibling();
519         if (!MarginCollapse::marginBeforeCollapsesWithPreviousSiblingMarginAfter(layoutState, *currentLayoutBox)) {
520             auto& previousDisplayBox = layoutState.displayBoxForLayoutBox(previousInFlowSibling);
521             return previousDisplayBox.rectWithMargin().bottom() + verticalMargin.before();
522         }
523 
524         if (!MarginCollapse::marginsCollapseThrough(layoutState, previousInFlowSibling)) {
525             auto& previousDisplayBox = layoutState.displayBoxForLayoutBox(previousInFlowSibling);
526             return previousDisplayBox.bottom() + verticalMargin.before();
527         }
528         currentLayoutBox = &previousInFlowSibling;
529     }
530 
531     auto& containingBlock = *layoutBox.containingBlock();
532     auto containingBlockContentBoxTop = layoutState.displayBoxForLayoutBox(containingBlock).contentBoxTop();
533     // Adjust vertical position depending whether this box directly or indirectly adjoins with its parent.
534     auto directlyAdjoinsParent = !layoutBox.previousInFlowSibling();
535     if (directlyAdjoinsParent) {
536         // If the top and bottom margins of a box are adjoining, then it is possible for margins to collapse through it.
537         // In this case, the position of the element depends on its relationship with the other elements whose margins are being collapsed.
538         if (verticalMargin.collapsedValues().isCollapsedThrough) {
539             // If the element's margins are collapsed with its parent's top margin, the top border edge of the box is defined to be the same as the parent's.
540             if (MarginCollapse::marginBeforeCollapsesWithParentMarginBefore(layoutState, layoutBox))
541                 return containingBlockContentBoxTop;
542             // Otherwise, either the element's parent is not taking part in the margin collapsing, or only the parent's bottom margin is involved.
543             // The position of the element's top border edge is the same as it would have been if the element had a non-zero bottom border.
544             auto beforeMarginWithBottomBorder = MarginCollapse::marginBeforeIgnoringCollapsingThrough(layoutState, layoutBox, verticalMargin.nonCollapsedValues());
545             return containingBlockContentBoxTop + beforeMarginWithBottomBorder;
546         }
547         // Non-collapsed through box vertical position depending whether the margin collapses.
548         if (MarginCollapse::marginBeforeCollapsesWithParentMarginBefore(layoutState, layoutBox))
549             return containingBlockContentBoxTop;
550 
551         return containingBlockContentBoxTop + verticalMargin.before();
552     }
553     // At this point this box indirectly (via collapsed through previous in-flow siblings) adjoins the parent. Let's check if it margin collapses with the parent.

554     ASSERT(containingBlock.firstInFlowChild());
555     ASSERT(containingBlock.firstInFlowChild() != &layoutBox);
556     if (MarginCollapse::marginBeforeCollapsesWithParentMarginBefore(layoutState, *containingBlock.firstInFlowChild()))
557         return containingBlockContentBoxTop;
558 
559     return containingBlockContentBoxTop + verticalMargin.before();
560 }

 54 void BlockFormattingContext::layoutInFlowContent(InvalidationState& invalidationState, const HorizontalConstraints& rootHorizontalConstraints, const VerticalConstraints& rootVerticalConstraints)
 55 {
 56     // 9.4.1 Block formatting contexts
 57     // In a block formatting context, boxes are laid out one after the other, vertically, beginning at the top of a containing block.
 58     // The vertical distance between two sibling boxes is determined by the 'margin' properties.
 59     // Vertical margins between adjacent block-level boxes in a block formatting context collapse.



 60     LOG_WITH_STREAM(FormattingContextLayout, stream << "[Start] -> block formatting context -> formatting root(" << &root() << ")");
 61     auto& formattingRoot = root();
 62     ASSERT(formattingRoot.hasInFlowOrFloatingChild());
 63     auto floatingContext = FloatingContext { formattingRoot, *this, formattingState().floatingState() };
 64 

 65     LayoutQueue layoutQueue;
 66     enum class LayoutDirection { Child, Sibling };
 67     auto appendNextToLayoutQueue = [&] (const auto& layoutBox, auto direction) {
 68         if (direction == LayoutDirection::Child) {
 69             if (!is<Container>(layoutBox))
 70                 return false;
 71             for (auto* child = downcast<Container>(layoutBox).firstInFlowOrFloatingChild(); child; child = child->nextInFlowOrFloatingSibling()) {
 72                 if (!invalidationState.needsLayout(*child))
 73                     continue;
 74                 layoutQueue.append(child);
 75                 return true;
 76             }
 77             return false;
 78         }
 79 
 80         if (direction == LayoutDirection::Sibling) {
 81             for (auto* nextSibling = layoutBox.nextInFlowOrFloatingSibling(); nextSibling; nextSibling = nextSibling->nextInFlowOrFloatingSibling()) {
 82                 if (!invalidationState.needsLayout(*nextSibling))
 83                     continue;
 84                 layoutQueue.append(nextSibling);
 85                 return true;
 86             }
 87             return false;
 88         }
 89         ASSERT_NOT_REACHED();
 90         return false;
 91     };
 92 
 93     auto horizontalConstraintsForLayoutBox = [&] (const auto& layoutBox) -> ConstraintsPair<HorizontalConstraints> {
 94         auto* containingBlock = layoutBox.containingBlock();
 95         ASSERT(containingBlock);
 96         if (containingBlock == &formattingRoot)
 97             return { rootHorizontalConstraints, rootHorizontalConstraints };
 98         return { rootHorizontalConstraints, Geometry::horizontalConstraintsForInFlow(geometryForBox(*containingBlock)) };
 99     };
100 
101     auto verticalConstraintsForLayoutBox = [&] (const auto& layoutBox) -> ConstraintsPair<VerticalConstraints> {
102         auto* containingBlock = layoutBox.containingBlock();
103         ASSERT(containingBlock);
104         if (containingBlock == &formattingRoot)
105             return { rootVerticalConstraints, rootVerticalConstraints };
106         return { rootVerticalConstraints, Geometry::verticalConstraintsForInFlow(geometryForBox(*containingBlock)) };
107     };
108     // This is a post-order tree traversal layout.
109     // The root container layout is done in the formatting context it lives in, not that one it creates, so let's start with the first child.
110     appendNextToLayoutQueue(formattingRoot, LayoutDirection::Child);

111     // 1. Go all the way down to the leaf node
112     // 2. Compute static position and width as we traverse down
113     // 3. As we climb back on the tree, compute height and finialize position
114     // (Any subtrees with new formatting contexts need to layout synchronously)
115     while (!layoutQueue.isEmpty()) {
116         // Traverse down on the descendants and compute width/static position until we find a leaf node.
117         while (true) {
118             auto& layoutBox = *layoutQueue.last();
119             auto horizontalConstraints = horizontalConstraintsForLayoutBox(layoutBox);
120             auto verticalConstraints = verticalConstraintsForLayoutBox(layoutBox);
121 
122             computeBorderAndPadding(layoutBox, horizontalConstraints.containingBlock);
123             computeStaticVerticalPosition(layoutBox, verticalConstraints);
124             computeWidthAndMargin(floatingContext, layoutBox, horizontalConstraints);
125             computeStaticHorizontalPosition(layoutBox, horizontalConstraints);
126 
127             if (layoutBox.establishesFormattingContext()) {
128                 if (is<Container>(layoutBox) && downcast<Container>(layoutBox).hasInFlowOrFloatingChild()) {
129                     // Layout the inflow descendants of this formatting context root.
130                     precomputeVerticalPositionForFormattingRoot(floatingContext, layoutBox, horizontalConstraints, verticalConstraints);
131                     auto& rootDisplayBox = geometryForBox(layoutBox);
132                     auto horizontalConstraintsForFormattingContext = Geometry::horizontalConstraintsForInFlow(rootDisplayBox);
133                     auto verticalConstraintsForFormattingContext = Geometry::verticalConstraintsForInFlow(rootDisplayBox);
134                     LayoutContext::createFormattingContext(downcast<Container>(layoutBox), layoutState())->layoutInFlowContent(invalidationState, horizontalConstraintsForFormattingContext, verticalConstraintsForFormattingContext);
135                 }
136                 break;
137             }
138             if (!appendNextToLayoutQueue(layoutBox, LayoutDirection::Child))





139                 break;

140         }
141 
142         // Climb back on the ancestors and compute height/final position.
143         while (!layoutQueue.isEmpty()) {

144             auto& layoutBox = *layoutQueue.takeLast();
145             auto horizontalConstraints = horizontalConstraintsForLayoutBox(layoutBox);
146             auto verticalConstraints = verticalConstraintsForLayoutBox(layoutBox);
147 
148             // All inflow descendants (if there are any) are laid out by now. Let's compute the box's height.
149             computeHeightAndMargin(layoutBox, horizontalConstraints, verticalConstraints);
150 
151             if (layoutBox.establishesFormattingContext()) {
152                 // Now that we computed the root's height, we can layout the out-of-flow descendants.
153                 if (is<Container>(layoutBox) && downcast<Container>(layoutBox).hasChild()) {
154                     auto& rootDisplayBox = geometryForBox(layoutBox);
155                     auto horizontalConstraintsForOutOfFlow =  Geometry::horizontalConstraintsForOutOfFlow(rootDisplayBox);
156                     auto verticalConstraintsForOutOfFlow = Geometry::verticalConstraintsForOutOfFlow(rootDisplayBox);
157                     LayoutContext::createFormattingContext(downcast<Container>(layoutBox), layoutState())->layoutOutOfFlowContent(invalidationState, horizontalConstraintsForOutOfFlow, verticalConstraintsForOutOfFlow);
158                 }
159             }
160             // Resolve final positions.
161             placeInFlowPositionedChildren(layoutBox, horizontalConstraints);
162             if (layoutBox.isFloatAvoider()) {
163                 computePositionToAvoidFloats(floatingContext, layoutBox, horizontalConstraints, verticalConstraints);
164                 if (layoutBox.isFloatingPositioned())
165                     floatingContext.append(layoutBox);
166             }
167 
168             if (appendNextToLayoutQueue(layoutBox, LayoutDirection::Sibling))
169                 break;
170         }
171     }
172     // Place the inflow positioned children.
173     placeInFlowPositionedChildren(formattingRoot, ConstraintsPair<HorizontalConstraints> { rootHorizontalConstraints, rootHorizontalConstraints });
174     LOG_WITH_STREAM(FormattingContextLayout, stream << "[End] -> block formatting context -> formatting root(" << &root() << ")");
175 }

177 Optional<LayoutUnit> BlockFormattingContext::usedAvailableWidthForFloatAvoider(const FloatingContext& floatingContext, const Box& layoutBox) const
178 {
179     // Normally the available width for an in-flow block level box is the width of the containing block's content box.
180     // However (and can't find it anywhere in the spec) non-floating positioned float avoider block level boxes are constrained by existing floats.
181     ASSERT(layoutBox.isFloatAvoider());
182     if (floatingContext.isEmpty())


183         return { };
184     // Vertical static position is not computed yet, so let's just pre-compute it for now.
185     auto verticalPosition = mapTopToFormattingContextRoot(layoutBox);
186     // FIXME: Check if the non-yet-computed height affects this computation - and whether we have to resolve it
187     // at a later point or not (can't find this in the spec).
188     auto constraints = floatingContext.constraints(verticalPosition, verticalPosition);
189     if (!constraints.left && !constraints.right)
190         return { };
191     auto& containingBlock = *layoutBox.containingBlock();
192     auto& containingBlockGeometry = geometryForBox(containingBlock);
193     auto availableWidth = containingBlockGeometry.contentBoxWidth();
194 
195     LayoutUnit containingBlockLeft;
196     LayoutUnit containingBlockRight = containingBlockGeometry.right();
197     if (&containingBlock != &root()) {
198         // Move containing block left/right to the root's coordinate system.
199         containingBlockLeft = mapLeftToFormattingContextRoot(containingBlock);
200         containingBlockRight = mapRightToFormattingContextRoot(containingBlock);
201     }
202     auto containingBlockContentBoxLeft = containingBlockLeft + containingBlockGeometry.borderLeft() + containingBlockGeometry.paddingLeft().valueOr(0);
203     auto containingBlockContentBoxRight = containingBlockRight - containingBlockGeometry.borderRight() + containingBlockGeometry.paddingRight().valueOr(0);
204 
205     // Shrink the available space if the floats are actually intruding at this vertical position.
206     if (constraints.left)
207         availableWidth -= std::max<LayoutUnit>(0, constraints.left->x - containingBlockContentBoxLeft);
208     if (constraints.right)
209         availableWidth -= std::max<LayoutUnit>(0, containingBlockContentBoxRight - constraints.right->x);
210     return availableWidth;
211 }

213 void BlockFormattingContext::placeInFlowPositionedChildren(const Box& layoutBox, const ConstraintsPair<HorizontalConstraints>& horizontalConstraints)




























214 {
215     if (!is<Container>(layoutBox))
216         return;

217     LOG_WITH_STREAM(FormattingContextLayout, stream << "Start: move in-flow positioned children -> parent: " << &layoutBox);
218     auto& container = downcast<Container>(layoutBox);
219     for (auto& childBox : childrenOfType<Box>(container)) {
220         if (!childBox.isInFlowPositioned())
221             continue;
222         auto positionOffset = geometry().inFlowPositionedPositionOffset(childBox, horizontalConstraints.containingBlock);
223         formattingState().displayBox(childBox).move(positionOffset);












224     }
225     LOG_WITH_STREAM(FormattingContextLayout, stream << "End: move in-flow positioned children -> parent: " << &layoutBox);
226 }

238 void BlockFormattingContext::precomputeVerticalPosition(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints, const VerticalConstraints& verticalConstraints)
239 {
240     auto computedVerticalMargin = geometry().computedVerticalMargin(layoutBox, horizontalConstraints);
241     auto usedNonCollapsedMargin = UsedVerticalMargin::NonCollapsedValues { computedVerticalMargin.before.valueOr(0), computedVerticalMargin.after.valueOr(0) };
242     auto precomputedMarginBefore = marginCollapse().precomputedMarginBefore(layoutBox, usedNonCollapsedMargin);
243     setPrecomputedMarginBefore(layoutBox, precomputedMarginBefore);
244 
245     auto& displayBox = formattingState().displayBox(layoutBox);
246     auto nonCollapsedValues = UsedVerticalMargin::NonCollapsedValues { precomputedMarginBefore.nonCollapsedValue, { } };
247     auto collapsedValues = UsedVerticalMargin::CollapsedValues { precomputedMarginBefore.collapsedValue, { }, precomputedMarginBefore.isCollapsedThrough };
248     auto verticalMargin = UsedVerticalMargin { nonCollapsedValues, collapsedValues };
249     displayBox.setVerticalMargin(verticalMargin);
250     displayBox.setTop(verticalPositionWithMargin(layoutBox, verticalMargin, verticalConstraints));
251 #if ASSERT_ENABLED
252     displayBox.setHasPrecomputedMarginBefore();
253 #endif

256 void BlockFormattingContext::precomputeVerticalPositionForAncestors(const Box& layoutBox, const ConstraintsPair<HorizontalConstraints>& horizontalConstraints, const ConstraintsPair<VerticalConstraints>& verticalConstraints)
257 {
258     // We only need to pre-compute the margin top for float related layout (formatting context roots avoid floats).
259     ASSERT(layoutBox.isFloatAvoider() || layoutBox.establishesInlineFormattingContext());
260 
261     // In order to figure out whether a box should avoid a float, we need to know the final positions of both (ignore relative positioning for now).
262     // In block formatting context the final position for a normal flow box includes
263     // 1. the static position and
264     // 2. the corresponding (non)collapsed margins.
265     // Now the vertical margins are computed when all the descendants are finalized, because the margin values might be depending on the height of the box
266     // (and the height might be based on the content).
267     // So when we get to the point where we intersect the box with the float to decide if the box needs to move, we don't yet have the final vertical position.
268     //
269     // The idea here is that as long as we don't cross the block formatting context boundary, we should be able to pre-compute the final top position.
270     // FIXME: we currently don't account for the "clear" property when computing the final position for an ancestor.
271     for (auto* ancestor = layoutBox.containingBlock(); ancestor && !ancestor->establishesBlockFormattingContext(); ancestor = ancestor->containingBlock()) {
272         // FIXME: with incremental layout, we might actually have a valid (non-precomputed) margin top as well.
273         if (hasPrecomputedMarginBefore(*ancestor))
274             return;
275         auto horizontalConstraintsForAncestor = [&] {
276             auto* containingBlock = layoutBox.containingBlock();
277             return containingBlock == &root() ? horizontalConstraints.root : Geometry::horizontalConstraintsForInFlow(geometryForBox(*containingBlock));
278         };
279         auto verticalConstraintsForAncestor = [&] {
280             auto* containingBlock = layoutBox.containingBlock();
281             return containingBlock == &root() ? verticalConstraints.root : Geometry::verticalConstraintsForInFlow(geometryForBox(*containingBlock));
282         };
283         precomputeVerticalPosition(*ancestor, horizontalConstraintsForAncestor(), verticalConstraintsForAncestor());
284     }
285 }

287 void BlockFormattingContext::precomputeVerticalPositionForFormattingRoot(const FloatingContext& floatingContext, const Box& layoutBox, const ConstraintsPair<HorizontalConstraints>& horizontalConstraints, const ConstraintsPair<VerticalConstraints>& verticalConstraints)
288 {
289     ASSERT(layoutBox.establishesFormattingContext());
290     if (layoutBox.isFloatAvoider() && !layoutBox.hasFloatClear())
291         return;
292     // We need the final vertical position when the formatting context inherits floats from the parent FC.
293     precomputeVerticalPositionForAncestors(layoutBox, horizontalConstraints, verticalConstraints);
294     precomputeVerticalPosition(layoutBox, horizontalConstraints.containingBlock, verticalConstraints.containingBlock);
295     if (layoutBox.hasFloatClear()) {
296         // if we just let the box sit at the static vertical position, we might find unrelated float boxes there (boxes that we need to clear).
297         computeVerticalPositionForFloatClear(floatingContext, layoutBox);
298     }
299 }

301 void BlockFormattingContext::computePositionToAvoidFloats(const FloatingContext& floatingContext, const Box& layoutBox, const ConstraintsPair<HorizontalConstraints>& horizontalConstraints, const ConstraintsPair<VerticalConstraints>& verticalConstraints)
302 {
303     ASSERT(layoutBox.isFloatAvoider());
304     // In order to position a float avoider we need to know its vertical position relative to its formatting context root (and not just its containing block),
305     // because all the already-placed floats (floats that we are trying to avoid here) in this BFC might belong
306     // to a different set of containing blocks (but they all descendants of the BFC root).
307     // However according to the BFC rules, at this point of the layout flow we don't yet have computed vertical positions for the ancestors.
308     if (layoutBox.isFloatingPositioned()) {
309         precomputeVerticalPositionForAncestors(layoutBox, horizontalConstraints, verticalConstraints);
310         formattingState().displayBox(layoutBox).setTopLeft(floatingContext.positionForFloat(layoutBox));
311         return;
312     }
313     // Non-float positioned float avoiders (formatting context roots and clear boxes) should be fine unless there are floats in this context.
314     if (floatingContext.isEmpty())
315         return;
316     precomputeVerticalPositionForAncestors(layoutBox, horizontalConstraints, verticalConstraints);
317     if (layoutBox.hasFloatClear())
318         return computeVerticalPositionForFloatClear(floatingContext, layoutBox);
319 
320     ASSERT(layoutBox.establishesFormattingContext());
321     if (auto adjustedPosition = floatingContext.positionForFormattingContextRoot(layoutBox))
322         formattingState().displayBox(layoutBox).setTopLeft(*adjustedPosition);







323 }

325 void BlockFormattingContext::computeVerticalPositionForFloatClear(const FloatingContext& floatingContext, const Box& layoutBox)
326 {
327     ASSERT(layoutBox.hasFloatClear());
328     if (floatingContext.isEmpty())
329         return;



330     auto verticalPositionAndClearance = floatingContext.verticalPositionWithClearance(layoutBox);
331     if (!verticalPositionAndClearance.position) {
332         ASSERT(!verticalPositionAndClearance.clearance);
333         return;
334     }
335 
336     auto& displayBox = formattingState().displayBox(layoutBox);
337     ASSERT(*verticalPositionAndClearance.position >= displayBox.top());
338     displayBox.setTop(*verticalPositionAndClearance.position);
339     if (verticalPositionAndClearance.clearance)
340         displayBox.setHasClearance();
341 }

343 void BlockFormattingContext::computeWidthAndMargin(const FloatingContext& floatingContext, const Box& layoutBox, const ConstraintsPair<HorizontalConstraints>& horizontalConstraints)



































344 {
345     auto compute = [&](Optional<LayoutUnit> usedWidth) -> ContentWidthAndMargin {
346         if (layoutBox.isFloatingPositioned())
347             return geometry().floatingWidthAndMargin(layoutBox, horizontalConstraints.containingBlock, { usedWidth, { } });
348 
349         if (layoutBox.isFloatAvoider()) {
350             auto availableWidth = usedAvailableWidthForFloatAvoider(floatingContext, layoutBox).valueOr(horizontalConstraints.containingBlock.logicalWidth);
351             return geometry().inFlowWidthAndMargin(layoutBox, { horizontalConstraints.containingBlock.logicalLeft, availableWidth }, { usedWidth, { } });
352         }

353 


354         if (layoutBox.isInFlow())
355             return geometry().inFlowWidthAndMargin(layoutBox, horizontalConstraints.containingBlock, { usedWidth, { } });



356 
357         ASSERT_NOT_REACHED();
358         return { };
359     };
360 
361     auto contentWidthAndMargin = compute({ });
362 
363     auto availableWidth = horizontalConstraints.containingBlock.logicalWidth;
364     if (auto maxWidth = geometry().computedMaxWidth(layoutBox, availableWidth)) {
365         auto maxWidthAndMargin = compute(maxWidth);
366         if (contentWidthAndMargin.contentWidth > maxWidthAndMargin.contentWidth)
367             contentWidthAndMargin = maxWidthAndMargin;
368     }
369 
370     auto minWidth = geometry().computedMinWidth(layoutBox, availableWidth).valueOr(0);
371     auto minWidthAndMargin = compute(minWidth);
372     if (contentWidthAndMargin.contentWidth < minWidthAndMargin.contentWidth)
373         contentWidthAndMargin = minWidthAndMargin;
374 
375     auto& displayBox = formattingState().displayBox(layoutBox);
376     displayBox.setContentBoxWidth(contentWidthAndMargin.contentWidth);
377     displayBox.setHorizontalMargin(contentWidthAndMargin.usedMargin);
378     displayBox.setHorizontalComputedMargin(contentWidthAndMargin.computedMargin);
379 }

381 void BlockFormattingContext::computeHeightAndMargin(const Box& layoutBox, const ConstraintsPair<HorizontalConstraints>& horizontalConstraints, const ConstraintsPair<VerticalConstraints>& verticalConstraints)
382 {
383     auto compute = [&](Optional<LayoutUnit> usedHeight) -> ContentHeightAndMargin {



384         if (layoutBox.isInFlow())
385             return geometry().inFlowHeightAndMargin(layoutBox, horizontalConstraints.containingBlock, { usedHeight });
386 
387         if (layoutBox.isFloatingPositioned())
388             return geometry().floatingHeightAndMargin(layoutBox, horizontalConstraints.containingBlock, { usedHeight });
389 
390         ASSERT_NOT_REACHED();
391         return { };
392     };
393 
394     auto contentHeightAndMargin = compute({ });
395     if (auto maxHeight = geometry().computedMaxHeight(layoutBox)) {
396         if (contentHeightAndMargin.contentHeight > *maxHeight) {
397             auto maxHeightAndMargin = compute(maxHeight);
398             // Used height should remain the same.
399             ASSERT((layoutState().inQuirksMode() && (layoutBox.isBodyBox() || layoutBox.isDocumentBox())) || maxHeightAndMargin.contentHeight == *maxHeight);
400             contentHeightAndMargin = { *maxHeight, maxHeightAndMargin.nonCollapsedMargin };
401         }
402     }
403 
404     if (auto minHeight = geometry().computedMinHeight(layoutBox)) {
405         if (contentHeightAndMargin.contentHeight < *minHeight) {
406             auto minHeightAndMargin = compute(minHeight);
407             // Used height should remain the same.
408             ASSERT((layoutState().inQuirksMode() && (layoutBox.isBodyBox() || layoutBox.isDocumentBox())) || minHeightAndMargin.contentHeight == *minHeight);
409             contentHeightAndMargin = { *minHeight, minHeightAndMargin.nonCollapsedMargin };
410         }
411     }
412 
413     // 1. Compute collapsed margins.
414     // 2. Adjust vertical position using the collapsed values
415     // 3. Adjust previous in-flow sibling margin after using this margin.
416     auto collapsedMargin = marginCollapse().collapsedVerticalValues(layoutBox, contentHeightAndMargin.nonCollapsedMargin);
417     auto verticalMargin = UsedVerticalMargin { contentHeightAndMargin.nonCollapsedMargin, collapsedMargin };
418     auto& displayBox = formattingState().displayBox(layoutBox);
419 
420     // Out of flow boxes don't need vertical adjustment after margin collapsing.
421     if (layoutBox.isOutOfFlowPositioned()) {
422         ASSERT(!hasPrecomputedMarginBefore(layoutBox));
423         displayBox.setContentBoxHeight(contentHeightAndMargin.contentHeight);
424         displayBox.setVerticalMargin(verticalMargin);
425         return;
426     }
427 
428     ASSERT(!hasPrecomputedMarginBefore(layoutBox) || precomputedMarginBefore(layoutBox).usedValue() == verticalMargin.before());
429     removePrecomputedMarginBefore(layoutBox);
430     displayBox.setTop(verticalPositionWithMargin(layoutBox, verticalMargin, verticalConstraints.containingBlock));
431     displayBox.setContentBoxHeight(contentHeightAndMargin.contentHeight);
432     displayBox.setVerticalMargin(verticalMargin);
433 
434     auto marginCollapse = this->marginCollapse();
435     formattingState().setPositiveAndNegativeVerticalMargin(layoutBox, marginCollapse.resolvedPositiveNegativeMarginValues(layoutBox, contentHeightAndMargin.nonCollapsedMargin));
436     // Adjust the previous sibling's margin bottom now that this box's vertical margin is computed.
437     MarginCollapse::updateMarginAfterForPreviousSibling(*this, marginCollapse, layoutBox);
438 }

440 FormattingContext::IntrinsicWidthConstraints BlockFormattingContext::computedIntrinsicWidthConstraints()
441 {


442     auto& formattingState = this->formattingState();
443     ASSERT(!formattingState.intrinsicWidthConstraints());
444 
445     // Visit the in-flow descendants and compute their min/max intrinsic width if needed.
446     // 1. Go all the way down to the leaf node
447     // 2. Check if actually need to visit all the boxes as we traverse down (already computed, container's min/max does not depend on descendants etc)
448     // 3. As we climb back on the tree, compute min/max intrinsic width
449     // (Any subtrees with new formatting contexts need to layout synchronously)
450     Vector<const Box*> queue;
451     if (root().hasInFlowOrFloatingChild())
452         queue.append(root().firstInFlowOrFloatingChild());
453 
454     IntrinsicWidthConstraints constraints;
455     while (!queue.isEmpty()) {
456         while (true) {
457             auto& layoutBox = *queue.last();
458             auto hasInFlowOrFloatingChild = is<Container>(layoutBox) && downcast<Container>(layoutBox).hasInFlowOrFloatingChild();
459             auto skipDescendants = formattingState.intrinsicWidthConstraintsForBox(layoutBox) || !hasInFlowOrFloatingChild || layoutBox.establishesFormattingContext() || layoutBox.style().width().isFixed();
460             if (skipDescendants)
461                 break;
462             queue.append(downcast<Container>(layoutBox).firstInFlowOrFloatingChild());
463         }
464         // Compute min/max intrinsic width bottom up if needed.
465         while (!queue.isEmpty()) {
466             auto& layoutBox = *queue.takeLast();
467             auto desdendantConstraints = formattingState.intrinsicWidthConstraintsForBox(layoutBox);
468             if (!desdendantConstraints) {
469                 desdendantConstraints = geometry().intrinsicWidthConstraints(layoutBox);
470                 formattingState.setIntrinsicWidthConstraintsForBox(layoutBox, *desdendantConstraints);
471             }
472             constraints.minimum = std::max(constraints.minimum, desdendantConstraints->minimum);
473             constraints.maximum = std::max(constraints.maximum, desdendantConstraints->maximum);
474             // Move over to the next sibling or take the next box in the queue.
475             if (auto* nextSibling = layoutBox.nextInFlowOrFloatingSibling()) {
476                 queue.append(nextSibling);
477                 break;
478             }
479         }
480     }
481     formattingState.setIntrinsicWidthConstraints(constraints);
482     return constraints;
483 }

485 LayoutUnit BlockFormattingContext::verticalPositionWithMargin(const Box& layoutBox, const UsedVerticalMargin& verticalMargin,  const VerticalConstraints& verticalConstraints) const
486 {
487     ASSERT(!layoutBox.isOutOfFlowPositioned());
488     // Now that we've computed the final margin before, let's shift the box's vertical position if needed.
489     // 1. Check if the box has clearance. If so, we've already precomputed/finalized the top value and vertical margin does not impact it anymore.
490     // 2. Check if the margin before collapses with the previous box's margin after. if not -> return previous box's bottom including margin after + marginBefore
491     // 3. Check if the previous box's margins collapse through. If not -> return previous box' bottom excluding margin after + marginBefore (they are supposed to be equal)
492     // 4. Go to previous box and start from step #1 until we hit the parent box.
493     auto& boxGeometry = geometryForBox(layoutBox);
494     if (boxGeometry.hasClearance())
495         return boxGeometry.top();

496 
497     auto* currentLayoutBox = &layoutBox;
498     while (currentLayoutBox) {
499         if (!currentLayoutBox->previousInFlowSibling())
500             break;
501         auto& previousInFlowSibling = *currentLayoutBox->previousInFlowSibling();
502         if (!marginCollapse().marginBeforeCollapsesWithPreviousSiblingMarginAfter(*currentLayoutBox)) {
503             auto& previousBoxGeometry = geometryForBox(previousInFlowSibling);
504             return previousBoxGeometry.rectWithMargin().bottom() + verticalMargin.before();
505         }
506 
507         if (!marginCollapse().marginsCollapseThrough(previousInFlowSibling)) {
508             auto& previousBoxGeometry = geometryForBox(previousInFlowSibling);
509             return previousBoxGeometry.bottom() + verticalMargin.before();
510         }
511         currentLayoutBox = &previousInFlowSibling;
512     }
513 
514     auto containingBlockContentBoxTop = verticalConstraints.logicalTop;

515     // Adjust vertical position depending whether this box directly or indirectly adjoins with its parent.
516     auto directlyAdjoinsParent = !layoutBox.previousInFlowSibling();
517     if (directlyAdjoinsParent) {
518         // If the top and bottom margins of a box are adjoining, then it is possible for margins to collapse through it.
519         // In this case, the position of the element depends on its relationship with the other elements whose margins are being collapsed.
520         if (verticalMargin.collapsedValues().isCollapsedThrough) {
521             // If the element's margins are collapsed with its parent's top margin, the top border edge of the box is defined to be the same as the parent's.
522             if (marginCollapse().marginBeforeCollapsesWithParentMarginBefore(layoutBox))
523                 return containingBlockContentBoxTop;
524             // Otherwise, either the element's parent is not taking part in the margin collapsing, or only the parent's bottom margin is involved.
525             // The position of the element's top border edge is the same as it would have been if the element had a non-zero bottom border.
526             auto beforeMarginWithBottomBorder = marginCollapse().marginBeforeIgnoringCollapsingThrough(layoutBox, verticalMargin.nonCollapsedValues());
527             return containingBlockContentBoxTop + beforeMarginWithBottomBorder;
528         }
529         // Non-collapsed through box vertical position depending whether the margin collapses.
530         if (marginCollapse().marginBeforeCollapsesWithParentMarginBefore(layoutBox))
531             return containingBlockContentBoxTop;
532 
533         return containingBlockContentBoxTop + verticalMargin.before();
534     }
535     // At this point this box indirectly (via collapsed through previous in-flow siblings) adjoins the parent. Let's check if it margin collapses with the parent.
536     auto& containingBlock = *layoutBox.containingBlock();
537     ASSERT(containingBlock.firstInFlowChild());
538     ASSERT(containingBlock.firstInFlowChild() != &layoutBox);
539     if (marginCollapse().marginBeforeCollapsesWithParentMarginBefore(*containingBlock.firstInFlowChild()))
540         return containingBlockContentBoxTop;
541 
542     return containingBlockContentBoxTop + verticalMargin.before();
543 }

545 void BlockFormattingContext::setPrecomputedMarginBefore(const Box& layoutBox, const PrecomputedMarginBefore& precomputedMarginBefore)
546 {
547     // Can't cross formatting context boundary.
548     ASSERT(&layoutState().formattingStateForBox(layoutBox) == &formattingState());
549     m_precomputedMarginBeforeList.set(&layoutBox, precomputedMarginBefore);
550 }

553 {
