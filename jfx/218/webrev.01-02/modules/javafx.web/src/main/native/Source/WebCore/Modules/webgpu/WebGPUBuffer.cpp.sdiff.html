 80 void WebGPUBuffer::rejectOrRegisterPromiseCallback(BufferMappingPromise&& promise, bool isRead)
 81 {
 82     errorScopes().setErrorPrefix(makeString("GPUBuffer.map", isRead ? "Read" : "Write", "Async(): "));
 83 
 84     if (!m_buffer) {
 85         errorScopes().generatePrefixedError("Invalid operation: invalid GPUBuffer!");
 86         promise.reject();
 87         return;
 88     }
 89 
 90     m_buffer->registerMappingCallback([promise = WTFMove(promise), protectedErrorScopes = makeRef(errorScopes())] (JSC::ArrayBuffer* arrayBuffer) mutable {
 91         if (arrayBuffer)
 92             promise.resolve(*arrayBuffer);
 93         else {
 94             // FIXME: It's possible to hit this code path in response to an unmap call, where
 95             // the ArrayBuffer will definitely be null, and thus an out of memory error makes no sense.
 96             // https://bugs.webkit.org/show_bug.cgi?id=204166
 97             protectedErrorScopes->generateError("", GPUErrorFilter::OutOfMemory);
 98             promise.reject();
 99         }
100     }, isRead, errorScopes());
101 }
