32 JSValue toJSNewlyCreated(ExecState*, JSDOMGlobalObject* globalObject, Ref<HTMLCollection>&& collection)
33 {
34     switch (collection->type()) {
35     case FormControls:
36         return createWrapper<HTMLFormControlsCollection>(globalObject, WTFMove(collection));
37     case SelectOptions:
38         return createWrapper<HTMLOptionsCollection>(globalObject, WTFMove(collection));
39     case DocAll:
40         return createWrapper<HTMLAllCollection>(globalObject, WTFMove(collection));
41     default:
42         break;
43     }
44 
45     return createWrapper<HTMLCollection>(globalObject, WTFMove(collection));
46 }
47 
48 JSValue toJS(ExecState* state, JSDOMGlobalObject* globalObject, HTMLCollection& collection)
49 {
50     return wrap(state, globalObject, collection);
51 }
