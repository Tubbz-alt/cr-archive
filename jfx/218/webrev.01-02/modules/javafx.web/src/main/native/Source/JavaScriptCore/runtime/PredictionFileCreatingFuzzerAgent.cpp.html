37 SpeculatedType PredictionFileCreatingFuzzerAgent::getPredictionInternal(CodeBlock*, PredictionTarget& predictionTarget, SpeculatedType original)
38 {
39     switch (predictionTarget.opcodeId) {
40     case op_to_this:
41     case op_bitand:
42     case op_bitor:
43     case op_bitxor:
44     case op_bitnot:
45     case op_lshift:
46     case op_rshift:
47     case op_get_by_val:
48     case op_get_argument:
49     case op_get_from_arguments:
50     case op_get_from_scope:
51     case op_to_number:
52     case op_get_by_id:
53     case op_get_by_id_with_this:
54     case op_get_by_val_with_this:
55     case op_get_direct_pname:
56     case op_construct:
57     case op_construct_varargs:
58     case op_call:
59     case op_call_eval:
60     case op_call_varargs:
61     case op_tail_call:
62     case op_tail_call_varargs:
63         dataLogF("%s:%" PRIx64 "\n", predictionTarget.lookupKey.utf8().data(), original);
64         break;
65 
66     default:
67         RELEASE_ASSERT_WITH_MESSAGE(false, "unhandled opcode: %s", opcodeNames[predictionTarget.opcodeId]);
68     }
69     return original;
70 }
