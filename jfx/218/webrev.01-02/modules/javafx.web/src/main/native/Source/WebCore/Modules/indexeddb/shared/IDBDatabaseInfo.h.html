 26 #pragma once
 27 
 28 #if ENABLE(INDEXED_DATABASE)
 29 
 30 #include "IDBObjectStoreInfo.h"
 31 #include <wtf/HashMap.h>
 32 
 33 namespace WebCore {
 34 
 35 class IDBDatabaseInfo {
 36     WTF_MAKE_FAST_ALLOCATED;
 37 public:
 38     explicit IDBDatabaseInfo(const String& name, uint64_t version, uint64_t maxIndexID);
 39 
 40     enum IsolatedCopyTag { IsolatedCopy };
 41     IDBDatabaseInfo(const IDBDatabaseInfo&, IsolatedCopyTag);
 42 
 43     IDBDatabaseInfo isolatedCopy() const;
 44 
 45     const String& name() const { return m_name; }
 46 
 47     void setVersion(uint64_t version) { m_version = version; }
 48     uint64_t version() const { return m_version; }
 49 
 50     bool hasObjectStore(const String& name) const;
 51     IDBObjectStoreInfo createNewObjectStore(const String& name, Optional<IDBKeyPath>&&, bool autoIncrement);
 52     void addExistingObjectStore(const IDBObjectStoreInfo&);
 53     IDBObjectStoreInfo* infoForExistingObjectStore(uint64_t objectStoreIdentifier);
 54     IDBObjectStoreInfo* infoForExistingObjectStore(const String& objectStoreName);
 55     const IDBObjectStoreInfo* infoForExistingObjectStore(uint64_t objectStoreIdentifier) const;
 56     const IDBObjectStoreInfo* infoForExistingObjectStore(const String& objectStoreName) const;
 57 
 58     void renameObjectStore(uint64_t objectStoreIdentifier, const String& newName);
 59 
 60     Vector<String> objectStoreNames() const;
 61     const HashMap<uint64_t, IDBObjectStoreInfo>& objectStoreMap() const { return m_objectStoreMap; }
 62 
 63     void deleteObjectStore(const String& objectStoreName);
 64     void deleteObjectStore(uint64_t objectStoreIdentifier);
 65 
 66     WEBCORE_EXPORT IDBDatabaseInfo();
 67 
 68     template<class Encoder> void encode(Encoder&) const;
 69     template<class Decoder> static bool decode(Decoder&, IDBDatabaseInfo&);
 70 
 71     void setMaxIndexID(uint64_t maxIndexID);
 72     uint64_t generateNextIndexID() { return ++m_maxIndexID; }
 73 
 74 #if !LOG_DISABLED
 75     String loggingString() const;
 76 #endif
 77 
 78 private:
 79     IDBObjectStoreInfo* getInfoForExistingObjectStore(const String& objectStoreName);
 80     IDBObjectStoreInfo* getInfoForExistingObjectStore(uint64_t objectStoreIdentifier);
 81 
 82     String m_name;
 83     uint64_t m_version { 0 };
 84     uint64_t m_maxObjectStoreID { 0 };
 85     uint64_t m_maxIndexID { 0 };
 86 
 87     HashMap<uint64_t, IDBObjectStoreInfo> m_objectStoreMap;
 88 
 89 };
 90 
 91 template<class Encoder>
 92 void IDBDatabaseInfo::encode(Encoder& encoder) const
 93 {
 94     encoder << m_name << m_version << m_maxObjectStoreID << m_maxIndexID << m_objectStoreMap;
 95 }
 96 
 97 template<class Decoder>
 98 bool IDBDatabaseInfo::decode(Decoder& decoder, IDBDatabaseInfo& info)
 99 {
100     if (!decoder.decode(info.m_name))
101         return false;
102 
103     if (!decoder.decode(info.m_version))
104         return false;
105 
106     if (!decoder.decode(info.m_maxObjectStoreID))
107         return false;
108 
109     if (!decoder.decode(info.m_maxIndexID))
110         return false;
111 
112     if (!decoder.decode(info.m_objectStoreMap))
113         return false;
114 
115     return true;
116 }
117 
118 } // namespace WebCore
119 
120 #endif // ENABLE(INDEXED_DATABASE)
