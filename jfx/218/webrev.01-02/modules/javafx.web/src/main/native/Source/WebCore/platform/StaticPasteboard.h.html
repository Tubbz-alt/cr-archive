26 #pragma once
27 
28 #include "Pasteboard.h"
29 #include <wtf/HashMap.h>
30 #include <wtf/Vector.h>
31 #include <wtf/text/StringHash.h>
32 
33 namespace WebCore {
34 
35 class SharedBuffer;
36 
37 class StaticPasteboard final : public Pasteboard {
38 public:
39     StaticPasteboard();
40     ~StaticPasteboard();
41 
42     PasteboardCustomData takeCustomData();
43 
44     bool isStatic() const final { return true; }
45 
46     bool hasData() final;
47     Vector<String> typesSafeForBindings(const String&) final;
48     Vector<String> typesForLegacyUnsafeBindings() final;
49     String readOrigin() final { return { }; }
50     String readString(const String& type) final;
51     String readStringInCustomData(const String& type) final;
52 
53     void writeString(const String& type, const String& data) final;
54     void writeData(const String& type, Ref<SharedBuffer>&& data);
55     void writeStringInCustomData(const String& type, const String& data);
56     void clear() final;
57     void clear(const String& type) final;
58 
59     void read(PasteboardPlainText&, PlainTextURLReadingPolicy = PlainTextURLReadingPolicy::AllowURL, Optional<size_t> = WTF::nullopt) final { }
60     void read(PasteboardWebContentReader&, WebContentReadingPolicy, Optional<size_t> = WTF::nullopt) final { }
61 
62     void write(const PasteboardURL&) final { }
63     void write(const PasteboardImage&) final { }
64     void write(const PasteboardWebContent&) final { }
65 
66     void writeCustomData(const Vector<PasteboardCustomData>&) final { }
67 
68     Pasteboard::FileContentState fileContentState() final { return FileContentState::NoFileOrImageData; }
69     bool canSmartReplace() final { return false; }
70 
71     void writeMarkup(const String&) final { }
72     void writePlainText(const String&, SmartReplaceOption) final { }
73 
74 #if ENABLE(DRAG_SUPPORT)
75     void setDragImage(DragImage, const IntPoint&) final { }
76 #endif
77 
78 private:
79     PasteboardCustomData m_customData;
80 };
81 
82 }
83 
84 SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::StaticPasteboard)
85     static bool isType(const WebCore::Pasteboard& pasteboard) { return pasteboard.isStatic(); }
86 SPECIALIZE_TYPE_TRAITS_END()
