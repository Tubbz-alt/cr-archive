 55 void ScrollingTreePositionedNode::commitStateBeforeChildren(const ScrollingStateNode& stateNode)
 56 {
 57     const ScrollingStatePositionedNode& positionedStateNode = downcast<ScrollingStatePositionedNode>(stateNode);
 58 
 59     if (positionedStateNode.hasChangedProperty(ScrollingStateNode::Layer)) {
 60         auto* layer = static_cast<Nicosia::PlatformLayer*>(positionedStateNode.layer());
 61         m_layer = downcast<Nicosia::CompositionLayer>(layer);
 62     }
 63 
 64     if (positionedStateNode.hasChangedProperty(ScrollingStatePositionedNode::RelatedOverflowScrollingNodes))
 65         m_relatedOverflowScrollingNodes = positionedStateNode.relatedOverflowScrollingNodes();
 66 
 67     if (positionedStateNode.hasChangedProperty(ScrollingStatePositionedNode::LayoutConstraintData))
 68         m_constraints = positionedStateNode.layoutConstraints();
 69 
 70     if (!m_relatedOverflowScrollingNodes.isEmpty())
 71         scrollingTree().activePositionedNodes().add(*this);
 72 }
 73 
 74 void ScrollingTreePositionedNode::applyLayerPositions()
 75 {
 76     FloatSize delta;
 77     for (auto nodeID : m_relatedOverflowScrollingNodes) {
 78         if (auto* node = scrollingTree().nodeForID(nodeID)) {
 79             if (is<ScrollingTreeOverflowScrollingNode>(node)) {
 80                 auto& overflowNode = downcast<ScrollingTreeOverflowScrollingNode>(*node);
 81                 delta += overflowNode.scrollDeltaSinceLastCommit();
 82             }
 83         }
 84     }
 85 
 86     FloatPoint layerPosition = m_constraints.layerPositionAtLastLayout() + delta;
 87 
 88     LOG_WITH_STREAM(Scrolling, stream << "ScrollingTreePositionedNode " << scrollingNodeID() << " applyLayerPositions: overflow delta " << delta << " moving layer to " << layerPosition);
 89 
 90     layerPosition -= m_constraints.alignmentOffset();
 91 
 92     ASSERT(m_layer);
 93     m_layer->updateState(
 94         [&layerPosition](Nicosia::CompositionLayer::LayerState& state)
 95         {
 96             state.position = layerPosition;
 97             state.delta.positionChanged = true;
 98         });
 99 }

101 void ScrollingTreePositionedNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
102 {
103     ts << "positioned node";
104     ScrollingTreeNode::dumpProperties(ts, behavior);
105 
106     ts.dumpProperty("layout constraints", m_constraints);
107     ts.dumpProperty("related overflow nodes", m_relatedOverflowScrollingNodes.size());
108 
109     if (behavior & ScrollingStateTreeAsTextBehaviorIncludeNodeIDs) {
110         if (!m_relatedOverflowScrollingNodes.isEmpty()) {
111             TextStream::GroupScope scope(ts);
112             ts << "overflow nodes";
113             for (auto nodeID : m_relatedOverflowScrollingNodes)
114                 ts << "\n" << indent << "nodeID " << nodeID;
115         }
116     }
117 }
