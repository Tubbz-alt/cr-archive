31 STATIC_ASSERT_IS_TRIVIALLY_DESTRUCTIBLE(JSDOMConstructorBase);
32 
33 static EncodedJSValue JSC_HOST_CALL callThrowTypeError(JSGlobalObject* globalObject, CallFrame*)
34 {
35     VM& vm = globalObject->vm();
36     auto scope = DECLARE_THROW_SCOPE(vm);
37     throwTypeError(globalObject, scope, "Constructor requires 'new' operator"_s);
38     return JSValue::encode(jsNull());
39 }
40 
41 CallType JSDOMConstructorBase::getCallData(JSCell*, CallData& callData)
42 {
43     callData.native.function = callThrowTypeError;
44     return CallType::Host;
45 }
46 
47 String JSDOMConstructorBase::className(const JSObject*, JSC::VM&)
48 {
49     return "Function"_s;
50 }
51 
52 String JSDOMConstructorBase::toStringName(const JSObject* object, JSC::JSGlobalObject* lexicalGlobalObject)
53 {
54     VM& vm = lexicalGlobalObject->vm();
55     const ClassInfo* info = object->classInfo(vm);
56     ASSERT(info);
57     return info->methodTable.className(object, vm);
58 }
59 
60 JSC::IsoSubspace* JSDOMConstructorBase::subspaceForImpl(JSC::VM& vm)
61 {
62     return &static_cast<JSVMClientData*>(vm.clientData)->domConstructorSpace();
63 }
