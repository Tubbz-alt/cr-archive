26 #include "config.h"
27 #include "ScrollbarThemeMock.h"
28 
29 // FIXME: This is a layering violation.
30 #include "DeprecatedGlobalSettings.h"
31 #include "Scrollbar.h"
32 
33 namespace WebCore {
34 
35 static const int cScrollbarThickness[] = { 15, 11 };
36 
37 IntRect ScrollbarThemeMock::trackRect(Scrollbar& scrollbar, bool)
38 {
39     return scrollbar.frameRect();
40 }
41 
42 int ScrollbarThemeMock::scrollbarThickness(ScrollbarControlSize controlSize, ScrollbarExpansionState)
43 {
44     return cScrollbarThickness[controlSize];
45 }
46 
47 void ScrollbarThemeMock::paintTrackBackground(GraphicsContext& context, Scrollbar& scrollbar, const IntRect& trackRect)
48 {
49     context.fillRect(trackRect, scrollbar.enabled() ? Color::lightGray : SimpleColor { 0xFFE0E0E0 });
50 }
51 
52 void ScrollbarThemeMock::paintThumb(GraphicsContext& context, Scrollbar& scrollbar, const IntRect& thumbRect)
53 {
54     if (scrollbar.enabled())
55         context.fillRect(thumbRect, Color::darkGray);
56 }
57 
58 bool ScrollbarThemeMock::usesOverlayScrollbars() const
59 {
60     // FIXME: This is a layering violation, but ScrollbarThemeMock is also created depending on settings in platform layer,
61     // we should fix it in both places.
62     return DeprecatedGlobalSettings::usesOverlayScrollbars();
63 }
