26 #pragma once
27 
28 #include "BitmapTextureJava.h"
29 #include "ImageBuffer.h"
30 #include "TextureMapper.h"
31 
32 #if USE(TEXTURE_MAPPER)
33 namespace WebCore {
34 
35 class TextureMapperJava final : public TextureMapper {
36     WTF_MAKE_FAST_ALLOCATED;
37 public:
38     TextureMapperJava();
39 
40     // TextureMapper implementation
41     void drawBorder(const Color&, float borderWidth, const FloatRect&, const TransformationMatrix&) final;
42     void drawNumber(int number, const Color&, const FloatPoint&, const TransformationMatrix&) final;
43     void drawTexture(const BitmapTexture&, const FloatRect& targetRect, const TransformationMatrix&, float opacity, unsigned exposedEdges) final;
44     void drawSolidColor(const FloatRect&, const TransformationMatrix&, const Color&, bool) final;
45     void beginClip(const TransformationMatrix&, const FloatRect&) final;
46     void bindSurface(BitmapTexture* surface) final { m_currentSurface = surface;}
47     void endClip() final { graphicsContext()->restore(); }
48     IntRect clipBounds() final { return currentContext()->clipBounds(); }
49     IntSize maxTextureSize() const final;
50     Ref<BitmapTexture> createTexture() final { return BitmapTextureJava::create(); }
51     Ref<BitmapTexture> createTexture(GCGLint) final { return createTexture(); }
52     void clearColor(const Color&) final;
53 
54     inline GraphicsContext* currentContext()
55     {
56         return m_currentSurface ? static_cast<BitmapTextureJava*>(m_currentSurface.get())->graphicsContext() : graphicsContext();
57     }
58 
59     void setGraphicsContext(GraphicsContext* context) { m_context = context; }
60     GraphicsContext* graphicsContext() { return m_context; }
61 private:
62     RefPtr<BitmapTexture> m_currentSurface;
63     GraphicsContext* m_context;
64 };
65 
66 }
67 #endif // USE(TEXTURE_MAPPER)
