 26 function catch(onRejected)
 27 {
 28     "use strict";
 29 
 30     return this.then(@undefined, onRejected);
 31 }
 32 
 33 function then(onFulfilled, onRejected)
 34 {
 35     "use strict";
 36 
 37     if (!@isPromise(this))
 38         @throwTypeError("|this| is not a Promise");
 39 
 40     var constructor = @speciesConstructor(this, @Promise);
 41 
 42     var promise;
 43     var promiseOrCapability;
 44     if (constructor === @Promise) {
 45         promiseOrCapability = @newPromise();
 46         promise = promiseOrCapability;
 47     } else {
 48         promiseOrCapability = @newPromiseCapabilitySlow(constructor);
 49         promise = promiseOrCapability.@promise;
 50     }
 51 
 52     if (typeof onFulfilled !== "function")
 53         onFulfilled = function (argument) { return argument; };
 54 
 55     if (typeof onRejected !== "function")
 56         onRejected = function (argument) { throw argument; };
 57 
 58     var reaction = @newPromiseReaction(promiseOrCapability, onFulfilled, onRejected);
 59 
 60     var flags = @getPromiseInternalField(this, @promiseFieldFlags);
 61     var state = flags & @promiseStateMask;
 62     if (state === @promiseStatePending) {
 63         reaction.@next = @getPromiseInternalField(this, @promiseFieldReactionsOrResult);
 64         @putPromiseInternalField(this, @promiseFieldReactionsOrResult, reaction);
 65     } else {
 66         if (state === @promiseStateRejected && !(flags & @promiseFlagsIsHandled))
 67             @hostPromiseRejectionTracker(this, @promiseRejectionHandle);
 68         @enqueueJob(@promiseReactionJob, state, reaction, @getPromiseInternalField(this, @promiseFieldReactionsOrResult));
 69     }
 70     @putPromiseInternalField(this, @promiseFieldFlags, @getPromiseInternalField(this, @promiseFieldFlags) | @promiseFlagsIsHandled);
 71 
 72     return promise;
 73 }

 75 function finally(onFinally)
 76 {
 77     "use strict";
 78 
 79     if (!@isObject(this))
 80         @throwTypeError("|this| is not an object");
 81 
 82     var constructor = @speciesConstructor(this, @Promise);
 83 
 84     @assert(@isConstructor(constructor));
 85 
 86     var thenFinally;
 87     var catchFinally;
 88 
 89     if (typeof onFinally !== "function") {
 90         thenFinally = onFinally;
 91         catchFinally = onFinally;
 92     } else {
 93         thenFinally = @getThenFinally(onFinally, constructor);
 94         catchFinally = @getCatchFinally(onFinally, constructor);
 95     }
 96 
 97     return this.then(thenFinally, catchFinally);
 98 }
 99 
100 @globalPrivate
101 function getThenFinally(onFinally, constructor)
102 {
103     "use strict";
104 
105     return function(value)
106     {
107         @assert(typeof onFinally === "function");
108         var result = onFinally();
109 
110         @assert(@isConstructor(constructor));
111         var resultCapability = @newPromiseCapability(constructor);
112 
113         resultCapability.@resolve.@call(@undefined, result);
114 
115         var promise = resultCapability.@promise;
116         var valueThunk = function () { return value; };
117 
118         return promise.then(valueThunk);
119     }
120 }
121 
122 @globalPrivate
123 function getCatchFinally(onFinally, constructor)
124 {
125     "use strict";
126 
127     return function(reason)
128     {
129         @assert(typeof onFinally === "function");
130         var result = onFinally();
131 
132         @assert(@isConstructor(constructor));
133         var resultCapability = @newPromiseCapability(constructor);
134 
135         resultCapability.@resolve.@call(@undefined, result);
136 
137         var promise = resultCapability.@promise;
138         var thrower = function () { throw reason; };
139 
140         return promise.then(thrower);
141     }
142 }
