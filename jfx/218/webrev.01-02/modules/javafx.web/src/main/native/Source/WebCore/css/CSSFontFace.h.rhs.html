 57 DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(CSSFontFace);
 58 class CSSFontFace final : public RefCounted<CSSFontFace> {
 59     WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(CSSFontFace);
 60 public:
 61     static Ref<CSSFontFace> create(CSSFontSelector* fontSelector, StyleRuleFontFace* cssConnection = nullptr, FontFace* wrapper = nullptr, bool isLocalFallback = false)
 62     {
 63         return adoptRef(*new CSSFontFace(fontSelector, cssConnection, wrapper, isLocalFallback));
 64     }
 65     virtual ~CSSFontFace();
 66 
 67     // FIXME: These functions don't need to have boolean return values.
 68     // Callers only call this with known-valid CSS values.
 69     bool setFamilies(CSSValue&);
 70     void setStyle(CSSValue&);
 71     void setWeight(CSSValue&);
 72     void setStretch(CSSValue&);
 73     bool setUnicodeRange(CSSValue&);






 74     void setFeatureSettings(CSSValue&);
 75     void setLoadingBehavior(CSSValue&);
 76 
 77     // Pending => Loading  => TimedOut
 78     //              ||  \\    //  ||
 79     //              ||   \\  //   ||
 80     //              ||    \\//    ||
 81     //              ||     //     ||
 82     //              ||    //\\    ||
 83     //              ||   //  \\   ||
 84     //              \/  \/    \/  \/
 85     //             Success    Failure
 86     enum class Status : uint8_t { Pending, Loading, TimedOut, Success, Failure };
 87 
 88     struct UnicodeRange;
 89 
 90     // Optional return values to represent default string for members of FontFace.h
 91     const Optional<CSSValueList*> families() const { return m_status == Status::Failure ? WTF::nullopt : static_cast<Optional<CSSValueList*>>(m_families.get()); }
 92     Optional<FontSelectionRange> weight() const { return m_status == Status::Failure ? WTF::nullopt : static_cast<Optional<FontSelectionRange>>(m_fontSelectionCapabilities.computeWeight()); }
 93     Optional<FontSelectionRange> stretch() const { return m_status == Status::Failure ? WTF::nullopt : static_cast<Optional<FontSelectionRange>>(m_fontSelectionCapabilities.computeWidth()); }
 94     Optional<FontSelectionRange> italic() const { return m_status == Status::Failure ? WTF::nullopt : static_cast<Optional<FontSelectionRange>>(m_fontSelectionCapabilities.computeSlope()); }
 95     Optional<FontSelectionCapabilities> fontSelectionCapabilities() const { return m_status == Status::Failure ? WTF::nullopt : static_cast<Optional<FontSelectionCapabilities>>(m_fontSelectionCapabilities.computeFontSelectionCapabilities()); }
 96     const Optional<Vector<UnicodeRange>> ranges() const { return m_status == Status::Failure ? WTF::nullopt : static_cast<Optional<Vector<UnicodeRange>>>(m_ranges); }
 97     const Optional<FontFeatureSettings> featureSettings() const { return m_status == Status::Failure ? WTF::nullopt : static_cast<Optional<FontFeatureSettings>>(m_featureSettings); }
 98     Optional<FontLoadingBehavior> loadingBehavior() const { return m_status == Status::Failure ? WTF::nullopt :  static_cast<Optional<FontLoadingBehavior>>(m_loadingBehavior); }
 99     void setWeight(FontSelectionRange weight) { m_fontSelectionCapabilities.weight = weight; }
100     void setStretch(FontSelectionRange stretch) { m_fontSelectionCapabilities.width = stretch; }
101     void setStyle(FontSelectionRange italic) { m_fontSelectionCapabilities.slope = italic; }
102     void setFontSelectionCapabilities(FontSelectionCapabilities capabilities) { m_fontSelectionCapabilities = capabilities; }
103     bool isLocalFallback() const { return m_isLocalFallback; }
104     Status status() const { return m_status; }
105     StyleRuleFontFace* cssConnection() const { return m_cssConnection.get(); }
106 
107     class Client;
108     void addClient(Client&);
109     void removeClient(Client&);
110 
111     bool computeFailureState() const;
112 
113     void opportunisticallyStartFontDataURLLoading(CSSFontSelector&);
114 
115     void adoptSource(std::unique_ptr<CSSFontFaceSource>&&);
116     void sourcesPopulated() { m_sourcesPopulated = true; }
117 
118     void fontLoaded(CSSFontFaceSource&);
119 
120     void load();
121 
122     RefPtr<Font> font(const FontDescription&, bool syntheticBold, bool syntheticItalic, ExternalResourceDownloadPolicy);
123 
124     static void appendSources(CSSFontFace&, CSSValueList&, Document*, bool isInitiatingElementInUserAgentShadowTree);
125 
126     class Client {
127     public:
128         virtual ~Client() = default;
129         virtual void fontLoaded(CSSFontFace&) { }
130         virtual void fontStateChanged(CSSFontFace&, Status /*oldState*/, Status /*newState*/) { }
131         virtual void fontPropertyChanged(CSSFontFace&, CSSValueList* /*oldFamilies*/ = nullptr) { }
132         virtual void ref() = 0;
133         virtual void deref() = 0;
134     };
135 











136     struct UnicodeRange {
137         UChar32 from;
138         UChar32 to;
139         bool operator==(const UnicodeRange& other) const { return from == other.from && to == other.to; }
140         bool operator!=(const UnicodeRange& other) const { return !(*this == other); }
141     };
142 
143     bool rangesMatchCodePoint(UChar32) const;
144 
145     // We don't guarantee that the FontFace wrapper will be the same every time you ask for it.
146     Ref<FontFace> wrapper();
147     void setWrapper(FontFace&);
148     FontFace* existingWrapper();
149 
150     struct FontLoadTiming {
151         Seconds blockPeriod;
152         Seconds swapPeriod;
153     };
154     FontLoadTiming fontLoadTiming() const;
155     bool shouldIgnoreFontLoadCompletions() const;
156 
157     bool purgeable() const;
158 
159     AllowUserInstalledFonts allowUserInstalledFonts() const;

160 
161     void updateStyleIfNeeded();
162 
163 #if ENABLE(SVG_FONTS)
164     bool hasSVGFontFaceSource() const;
165 #endif
166     void setErrorState();
167 
168     Document* document() const;
169 
170 private:
171     CSSFontFace(CSSFontSelector*, StyleRuleFontFace*, FontFace*, bool isLocalFallback);
172 
173     size_t pump(ExternalResourceDownloadPolicy);
174     void setStatus(Status);
175     void notifyClientsOfFontPropertyChange();
176 
177     void initializeWrapper();
178 
179     void fontLoadEventOccurred();
180     void timeoutFired();
181 
182     RefPtr<CSSValueList> m_families;
183     Vector<UnicodeRange> m_ranges;
184 
185     FontFeatureSettings m_featureSettings;

186     FontLoadingBehavior m_loadingBehavior { FontLoadingBehavior::Auto };
187 
188     Vector<std::unique_ptr<CSSFontFaceSource>, 0, CrashOnOverflow, 0> m_sources;
189     WeakPtr<CSSFontSelector> m_fontSelector; // FIXME: Ideally this data member would go away (https://bugs.webkit.org/show_bug.cgi?id=208351).
190     RefPtr<StyleRuleFontFace> m_cssConnection;
191     HashSet<Client*> m_clients;
192     WeakPtr<FontFace> m_wrapper;
193     FontSelectionSpecifiedCapabilities m_fontSelectionCapabilities;
194 
195     Status m_status { Status::Pending };
196     bool m_isLocalFallback { false };
197     bool m_sourcesPopulated { false };
198     bool m_mayBePurged { true };
199 
200     Timer m_timeoutTimer;
201 };
