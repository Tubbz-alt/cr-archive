 44 class EventLoopTask {
 45     WTF_MAKE_NONCOPYABLE(EventLoopTask);
 46     WTF_MAKE_FAST_ALLOCATED;
 47 
 48 public:
 49     virtual ~EventLoopTask() = default;
 50 
 51     TaskSource taskSource() { return m_taskSource; }
 52     virtual void execute() = 0;
 53 
 54     EventLoopTaskGroup* group() const { return m_group.get(); }
 55 
 56 protected:
 57     EventLoopTask(TaskSource, EventLoopTaskGroup&);
 58 
 59 private:
 60     const TaskSource m_taskSource;
 61     WeakPtr<EventLoopTaskGroup> m_group;
 62 };

 64 // https://html.spec.whatwg.org/multipage/webappapis.html#event-loop
 65 class EventLoop : public RefCounted<EventLoop>, public CanMakeWeakPtr<EventLoop> {
 66 public:
 67     virtual ~EventLoop() = default;
 68 
 69     typedef Function<void ()> TaskFunction;
 70     void queueTask(std::unique_ptr<EventLoopTask>&&);
 71 
 72     // https://html.spec.whatwg.org/multipage/webappapis.html#queue-a-microtask
 73     void queueMicrotask(std::unique_ptr<EventLoopTask>&&);
 74 
 75     // https://html.spec.whatwg.org/multipage/webappapis.html#perform-a-microtask-checkpoint
 76     void performMicrotaskCheckpoint();
 77     virtual MicrotaskQueue& microtaskQueue() = 0;
 78 
 79     void resumeGroup(EventLoopTaskGroup&);
 80     void stopGroup(EventLoopTaskGroup&);
 81 
 82 protected:
 83     EventLoop() = default;
 84     void run();
 85     void clearAllTasks();
 86 
 87 private:
 88     void scheduleToRunIfNeeded();
 89     virtual void scheduleToRun() = 0;
 90     virtual bool isContextThread() const = 0;
 91 
 92     // Use a global queue instead of multiple task queues since HTML5 spec allows UA to pick arbitrary queue.
 93     Vector<std::unique_ptr<EventLoopTask>> m_tasks;
 94     WeakHashSet<EventLoopTaskGroup> m_groupsWithSuspenedTasks;
 95     bool m_isScheduledToRun { false };
 96 };
 97 
 98 class EventLoopTaskGroup : public CanMakeWeakPtr<EventLoopTaskGroup> {
 99     WTF_MAKE_NONCOPYABLE(EventLoopTaskGroup);
100     WTF_MAKE_FAST_ALLOCATED;
101 
102 public:
103     EventLoopTaskGroup(EventLoop& eventLoop)
104         : m_eventLoop(makeWeakPtr(eventLoop))
105     {
106     }
107 
108     bool hasSameEventLoopAs(EventLoopTaskGroup& otherGroup)
109     {
110         ASSERT(m_eventLoop);
111         return m_eventLoop == otherGroup.m_eventLoop;
112     }
113 
114     bool matchesTask(EventLoopTask& task) const
115     {
116         auto* group = task.group();
117         return group == this;
118     }
119 
120     void stopAndDiscardAllTasks()
121     {
122         m_state = State::Stopped;
123         if (auto* eventLoop = m_eventLoop.get())
124             eventLoop->stopGroup(*this);
125     }
126 
127     void suspend()
128     {
129         ASSERT(m_state != State::Stopped);
130         m_state = State::Suspended;
131         // We don't remove suspended tasks to preserve the ordering.
132         // EventLoop::run checks whether each task's group is suspended or not.
133     }
134 
135     void resume()
136     {
137         ASSERT(m_state != State::Stopped);
138         m_state = State::Running;
139         if (auto* eventLoop = m_eventLoop.get())
140             eventLoop->resumeGroup(*this);
141     }
142 
143     bool isStoppedPermanently() { return m_state == State::Stopped; }
144     bool isSuspended() { return m_state == State::Suspended; }
145 
146     void queueTask(std::unique_ptr<EventLoopTask>&&);
147     WEBCORE_EXPORT void queueTask(TaskSource, EventLoop::TaskFunction&&);
148 
149     // https://html.spec.whatwg.org/multipage/webappapis.html#queue-a-microtask
150     WEBCORE_EXPORT void queueMicrotask(EventLoop::TaskFunction&&);
151     MicrotaskQueue& microtaskQueue() { return m_eventLoop->microtaskQueue(); }
152 
153     // https://html.spec.whatwg.org/multipage/webappapis.html#perform-a-microtask-checkpoint
154     void performMicrotaskCheckpoint();
155 
156 private:
157     enum class State : uint8_t { Running, Suspended, Stopped };
158 
159     WeakPtr<EventLoop> m_eventLoop;
160     State m_state { State::Running };
161 };

163 inline EventLoopTask::EventLoopTask(TaskSource source, EventLoopTaskGroup& group)
164     : m_taskSource(source)
165     , m_group(makeWeakPtr(group))
166 { }
