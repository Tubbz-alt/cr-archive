31 namespace JSC {

34 class GetterSetter;

37 
38 class DOMAttributeGetterSetter final : public CustomGetterSetter {
39 public:
40     using Base = CustomGetterSetter;
41 






42     static DOMAttributeGetterSetter* create(VM& vm, CustomGetter customGetter, CustomSetter customSetter, DOMAttributeAnnotation domAttribute)
43     {
44         DOMAttributeGetterSetter* customGetterSetter = new (NotNull, allocateCell<DOMAttributeGetterSetter>(vm.heap)) DOMAttributeGetterSetter(vm, customGetter, customSetter, domAttribute);
45         customGetterSetter->finishCreation(vm);
46         return customGetterSetter;
47     }
48 
49     DOMAttributeAnnotation domAttribute() const { return m_domAttribute; }
50 
51     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
52     {
53         return Structure::create(vm, globalObject, prototype, TypeInfo(CustomGetterSetterType, StructureFlags), info());
54     }
55 
56     DECLARE_EXPORT_INFO;
57 
58 private:
59     DOMAttributeGetterSetter(VM& vm, CustomGetter getter, CustomSetter setter, DOMAttributeAnnotation domAttribute)
60         : Base(vm, vm.domAttributeGetterSetterStructure.get(), getter, setter)
61         , m_domAttribute(domAttribute)
62     {
63     }
64 
65     DOMAttributeAnnotation m_domAttribute;
66 };
