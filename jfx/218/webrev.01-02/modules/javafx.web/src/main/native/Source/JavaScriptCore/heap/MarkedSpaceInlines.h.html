33 ALWAYS_INLINE Heap& MarkedSpace::heap() const
34 {
35     return *bitwise_cast<Heap*>(bitwise_cast<uintptr_t>(this) - OBJECT_OFFSETOF(Heap, m_objectSpace));
36 }
37 
38 template<typename Functor> inline void MarkedSpace::forEachLiveCell(HeapIterationScope&, const Functor& functor)
39 {
40     ASSERT(isIterating());
41     forEachLiveCell(functor);
42 }
43 
44 template<typename Functor> inline void MarkedSpace::forEachLiveCell(const Functor& functor)
45 {
46     BlockIterator end = m_blocks.set().end();
47     for (BlockIterator it = m_blocks.set().begin(); it != end; ++it) {
48         IterationStatus result = (*it)->handle().forEachLiveCell(
49             [&] (size_t, HeapCell* cell, HeapCell::Kind kind) -> IterationStatus {
50                 return functor(cell, kind);
51             });
52         if (result == IterationStatus::Done)
53             return;
54     }
55     for (PreciseAllocation* allocation : m_preciseAllocations) {
56         if (allocation->isLive()) {
57             if (functor(allocation->cell(), allocation->attributes().cellKind) == IterationStatus::Done)
58                 return;
59         }
60     }
61 }
62 
63 template<typename Functor> inline void MarkedSpace::forEachDeadCell(HeapIterationScope&, const Functor& functor)
64 {
65     ASSERT(isIterating());
66     BlockIterator end = m_blocks.set().end();
67     for (BlockIterator it = m_blocks.set().begin(); it != end; ++it) {
68         if ((*it)->handle().forEachDeadCell(functor) == IterationStatus::Done)
69             return;
70     }
71     for (PreciseAllocation* allocation : m_preciseAllocations) {
72         if (!allocation->isLive()) {
73             if (functor(allocation->cell(), allocation->attributes().cellKind) == IterationStatus::Done)
74                 return;
75         }
76     }
77 }
