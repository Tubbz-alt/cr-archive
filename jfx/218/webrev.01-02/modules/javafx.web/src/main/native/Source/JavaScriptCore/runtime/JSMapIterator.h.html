 27 
 28 #include "IterationKind.h"
 29 #include "JSMap.h"
 30 #include "JSObject.h"

 32 namespace JSC {
 33 
 34 // Now, it is only used for serialization.
 35 class JSMapIterator final : public JSCell {
 36     typedef HashMapBucket<HashMapBucketDataKeyValue> HashMapBucketType;
 37 public:
 38     using Base = JSCell;
 39 
 40     DECLARE_EXPORT_INFO;
 41 
 42     template<typename CellType, SubspaceAccess mode>
 43     static IsoSubspace* subspaceFor(VM& vm)
 44     {
 45         return vm.mapIteratorSpace<mode>();
 46     }
 47 
 48     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
 49     {
 50         return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
 51     }
 52 
 53     static JSMapIterator* create(VM& vm, Structure* structure, JSMap* iteratedObject, IterationKind kind)
 54     {
 55         JSMapIterator* instance = new (NotNull, allocateCell<JSMapIterator>(vm.heap)) JSMapIterator(vm, structure, iteratedObject, kind);
 56         instance->finishCreation(vm, iteratedObject);
 57         return instance;
 58     }
 59 
 60     ALWAYS_INLINE HashMapBucketType* advanceIter(JSGlobalObject* globalObject)
 61     {
 62         HashMapBucketType* prev = m_iter.get();
 63         if (!prev)
 64             return nullptr;
 65         VM& vm = getVM(globalObject);
 66         HashMapBucketType* bucket = m_iter->next();
 67         while (bucket && bucket->deleted())
 68             bucket = bucket->next();
 69         if (!bucket) {
 70             setIterator(vm, nullptr);
 71             return nullptr;
 72         }
 73         setIterator(vm, bucket); // We keep m_iter on the last value since the first thing we do in this function is call next().
 74         return bucket;
 75     }
 76     bool next(JSGlobalObject* globalObject, JSValue& value)
 77     {
 78         HashMapBucketType* bucket = advanceIter(globalObject);
 79         if (!bucket)
 80             return false;
 81 
 82         if (m_kind == IterationKind::Values)
 83             value = bucket->value();
 84         else if (m_kind == IterationKind::Keys)
 85             value = bucket->key();
 86         else
 87             value = createPair(globalObject, bucket->key(), bucket->value());
 88         return true;
 89     }
 90 
 91     bool nextKeyValue(JSGlobalObject* globalObject, JSValue& key, JSValue& value)
 92     {
 93         HashMapBucketType* bucket = advanceIter(globalObject);
 94         if (!bucket)
 95             return false;
 96 
 97         key = bucket->key();
 98         value = bucket->value();
 99         return true;
100     }
101 
102     IterationKind kind() const { return m_kind; }
103     JSValue iteratedValue() const { return m_map.get(); }
104 
105 private:
106     JSMapIterator(VM& vm, Structure* structure, JSMap*, IterationKind kind)
107         : Base(vm, structure)
108         , m_kind(kind)
109     { }
110 
111     void setIterator(VM& vm, HashMapBucketType* bucket)
112     {
113         m_iter.setMayBeNull(vm, this, bucket);
114     }
115 
116     JS_EXPORT_PRIVATE void finishCreation(VM&, JSMap*);
117     JSValue createPair(JSGlobalObject*, JSValue, JSValue);
118     static void visitChildren(JSCell*, SlotVisitor&);
119 
120     WriteBarrier<JSMap> m_map;
121     WriteBarrier<HashMapBucketType> m_iter;
122     IterationKind m_kind;
123 };
124 STATIC_ASSERT_IS_TRIVIALLY_DESTRUCTIBLE(JSMapIterator);
125 
126 } // namespace JSC
