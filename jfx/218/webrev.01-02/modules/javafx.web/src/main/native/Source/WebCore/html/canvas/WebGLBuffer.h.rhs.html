 26 #pragma once
 27 
 28 #if ENABLE(WEBGL)
 29 
 30 #include "WebGLSharedObject.h"
 31 #include <wtf/RefPtr.h>
 32 
 33 namespace JSC {
 34 class ArrayBuffer;
 35 class ArrayBufferView;
 36 }
 37 
 38 namespace WebCore {
 39 
 40 class WebGLBuffer final : public WebGLSharedObject {
 41 public:
 42     static Ref<WebGLBuffer> create(WebGLRenderingContextBase&);
 43     virtual ~WebGLBuffer();
 44 
 45     bool associateBufferData(GCGLsizeiptr size);
 46     bool associateBufferData(JSC::ArrayBuffer*);
 47     bool associateBufferData(JSC::ArrayBufferView*);
 48     bool associateBufferSubData(GCGLintptr offset, JSC::ArrayBuffer*);
 49     bool associateBufferSubData(GCGLintptr offset, JSC::ArrayBufferView*);
 50     bool associateCopyBufferSubData(const WebGLBuffer& readBuffer, GCGLintptr readOffset, GCGLintptr writeOffset, GCGLsizeiptr);
 51 
 52     void disassociateBufferData();
 53 
 54     GCGLsizeiptr byteLength() const;
 55     const RefPtr<JSC::ArrayBuffer> elementArrayBuffer() const { return m_elementArrayBuffer; }
 56 
 57     // Gets the cached max index for the given type if one has been set.
 58     Optional<unsigned> getCachedMaxIndex(GCGLenum type);
 59     // Sets the cached max index for the given type.
 60     void setCachedMaxIndex(GCGLenum type, unsigned value);
 61 
 62     GCGLenum getTarget() const { return m_target; }
 63     void setTarget(GCGLenum);
 64 
 65     GCGLenum arrayBufferOrElementArrayBuffer() const { return m_arrayBufferOrElementArrayBuffer; }
 66 
 67     bool hasEverBeenBound() const { return object() && m_target; }
 68 
 69 protected:
 70     WebGLBuffer(WebGLRenderingContextBase&);
 71 
 72     void deleteObjectImpl(GraphicsContextGLOpenGL*, PlatformGLObject) override;
 73 
 74 private:
 75     GCGLenum m_target { 0 };
 76 
 77     RefPtr<JSC::ArrayBuffer> m_elementArrayBuffer;
 78     GCGLsizeiptr m_byteLength { 0 };
 79 
 80     GCGLenum m_arrayBufferOrElementArrayBuffer { 0 };
 81 
 82     // Optimization for index validation. For each type of index
 83     // (i.e., UNSIGNED_SHORT), cache the maximum index in the
 84     // entire buffer.
 85     //
 86     // This is sufficient to eliminate a lot of work upon each
 87     // draw call as long as all bound array buffers are at least
 88     // that size.
 89     struct MaxIndexCacheEntry {
 90         GCGLenum type;
 91         unsigned maxIndex;
 92     };
 93     // OpenGL ES 2.0 only has two valid index types (UNSIGNED_BYTE
 94     // and UNSIGNED_SHORT) plus one extension (UNSIGNED_INT).
 95     MaxIndexCacheEntry m_maxIndexCache[4];
 96     unsigned m_nextAvailableCacheEntry { 0 };
 97 
 98     // Clears all of the cached max indices.
 99     void clearCachedMaxIndices();
100 
101     // Helper function called by the three associateBufferData().
102     bool associateBufferDataImpl(const void* data, GCGLsizeiptr byteLength);
103     // Helper function called by the two associateBufferSubData().
104     bool associateBufferSubDataImpl(GCGLintptr offset, const void* data, GCGLsizeiptr byteLength);
105 };
106 
107 } // namespace WebCore
108 
109 #endif
