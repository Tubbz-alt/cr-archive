 26 #pragma once
 27 
 28 #include "ImageFrame.h"
 29 
 30 #include <wtf/Forward.h>
 31 #include <wtf/Optional.h>
 32 #include <wtf/RunLoop.h>
 33 #include <wtf/SynchronizedFixedQueue.h>
 34 #include <wtf/WeakPtr.h>
 35 #include <wtf/WorkQueue.h>
 36 #include <wtf/text/TextStream.h>

 38 namespace WebCore {
 39 
 40 class BitmapImage;
 41 class GraphicsContext;
 42 class ImageDecoder;
 43 class SharedBuffer;
 44 
 45 class ImageSource : public ThreadSafeRefCounted<ImageSource>, public CanMakeWeakPtr<ImageSource> {
 46     friend class BitmapImage;
 47 public:
 48     ~ImageSource();
 49 
 50     static Ref<ImageSource> create(BitmapImage* image, AlphaOption alphaOption = AlphaOption::Premultiplied, GammaAndColorProfileOption gammaAndColorProfileOption = GammaAndColorProfileOption::Applied)
 51     {
 52         return adoptRef(*new ImageSource(image, alphaOption, gammaAndColorProfileOption));
 53     }
 54 
 55     static Ref<ImageSource> create(NativeImagePtr&& nativeImage)
 56     {
 57         return adoptRef(*new ImageSource(WTFMove(nativeImage)));
 58     }
 59 
 60     void setData(SharedBuffer* data, bool allDataReceived);
 61     void resetData(SharedBuffer* data);
 62     EncodedDataStatus dataChanged(SharedBuffer* data, bool allDataReceived);
 63     bool isAllDataReceived();
 64 
 65     unsigned decodedSize() const { return m_decodedSize; }
 66     void destroyAllDecodedData() { destroyDecodedData(frameCount(), frameCount()); }
 67     void destroyAllDecodedDataExcludeFrame(size_t excludeFrame) { destroyDecodedData(frameCount(), excludeFrame); }
 68     void destroyDecodedDataBeforeFrame(size_t beforeFrame) { destroyDecodedData(beforeFrame, beforeFrame); }
 69     void destroyIncompleteDecodedData();
 70     void clearFrameBufferCache(size_t beforeFrame);
 71 
 72     void growFrames();
 73     void clearMetadata();
 74     void clearImage() { m_image = nullptr; }
 75     URL sourceURL() const;
 76     String mimeType() const;
 77     long long expectedContentLength() const;
 78 
 79     // Asynchronous image decoding
 80     bool canUseAsyncDecoding();
 81     void startAsyncDecodingQueue();
 82     void requestFrameAsyncDecodingAtIndex(size_t, SubsamplingLevel, const Optional<IntSize>& = { });
 83     void stopAsyncDecodingQueue();
 84     bool hasAsyncDecodingQueue() const { return m_decodingQueue; }
 85     bool isAsyncDecodingQueueIdle() const;
 86     void setFrameDecodingDurationForTesting(Seconds duration) { m_frameDecodingDurationForTesting = duration; }
 87     Seconds frameDecodingDurationForTesting() const { return m_frameDecodingDurationForTesting; }
 88 
 89     // Image metadata which is calculated either by the ImageDecoder or directly
 90     // from the NativeImage if this class was created for a memory image.
 91     EncodedDataStatus encodedDataStatus();
 92     bool isSizeAvailable() { return encodedDataStatus() >= EncodedDataStatus::SizeAvailable; }
 93     WEBCORE_EXPORT size_t frameCount();
 94     RepetitionCount repetitionCount();
 95     String uti();
 96     String filenameExtension();
 97     Optional<IntPoint> hotSpot();
 98     ImageOrientation orientation();
 99 
100     // Image metadata which is calculated from the first ImageFrame.
101     WEBCORE_EXPORT IntSize size(ImageOrientation = ImageOrientation::FromImage);
102     IntSize sizeRespectingOrientation();
103     Color singlePixelSolidColor();
104     SubsamplingLevel maximumSubsamplingLevel();
105 
106     // ImageFrame metadata which does not require caching the ImageFrame.
107     bool frameIsBeingDecodedAndIsCompatibleWithOptionsAtIndex(size_t, const DecodingOptions&);
108     DecodingStatus frameDecodingStatusAtIndex(size_t);
109     bool frameHasAlphaAtIndex(size_t);
110     bool frameHasImageAtIndex(size_t);
111     bool frameHasFullSizeNativeImageAtIndex(size_t, const Optional<SubsamplingLevel>&);
112     bool frameHasDecodedNativeImageCompatibleWithOptionsAtIndex(size_t, const Optional<SubsamplingLevel>&, const DecodingOptions&);
113     SubsamplingLevel frameSubsamplingLevelAtIndex(size_t);
114 
115     // ImageFrame metadata which forces caching or re-caching the ImageFrame.
116     IntSize frameSizeAtIndex(size_t, SubsamplingLevel = SubsamplingLevel::Default);
117     unsigned frameBytesAtIndex(size_t, SubsamplingLevel = SubsamplingLevel::Default);
118     WEBCORE_EXPORT Seconds frameDurationAtIndex(size_t);
119     ImageOrientation frameOrientationAtIndex(size_t);
120 
121 #if USE(DIRECT2D)
122     void setTargetContext(const GraphicsContext*);
123 #endif
124     NativeImagePtr createFrameImageAtIndex(size_t, SubsamplingLevel = SubsamplingLevel::Default);
125     NativeImagePtr frameImageAtIndex(size_t);
126     NativeImagePtr frameImageAtIndexCacheIfNeeded(size_t, SubsamplingLevel = SubsamplingLevel::Default);
127 
128 private:
129     ImageSource(BitmapImage*, AlphaOption = AlphaOption::Premultiplied, GammaAndColorProfileOption = GammaAndColorProfileOption::Applied);
130     ImageSource(NativeImagePtr&&);
131 
132     template<typename T, T (ImageDecoder::*functor)() const>
133     T metadata(const T& defaultValue, Optional<T>* cachedValue = nullptr);
134 
135     template<typename T, typename... Args>
136     T frameMetadataAtIndex(size_t, T (ImageFrame::*functor)(Args...) const, Args&&...);
137 
138     template<typename T, typename... Args>
139     T frameMetadataAtIndexCacheIfNeeded(size_t, T (ImageFrame::*functor)() const,  Optional<T>* cachedValue, Args&&...);
140 
141     bool ensureDecoderAvailable(SharedBuffer* data);
142     bool isDecoderAvailable() const { return m_decoder; }
143     void destroyDecodedData(size_t frameCount, size_t excludeFrame);
144     void decodedSizeChanged(long long decodedSize);
145     void didDecodeProperties(unsigned decodedPropertiesSize);
146     void decodedSizeIncreased(unsigned decodedSize);
147     void decodedSizeDecreased(unsigned decodedSize);
148     void decodedSizeReset(unsigned decodedSize);
149     void encodedDataStatusChanged(EncodedDataStatus);
150 
151     void setNativeImage(NativeImagePtr&&);
152     void cacheMetadataAtIndex(size_t, SubsamplingLevel, DecodingStatus = DecodingStatus::Invalid);
153     void cacheNativeImageAtIndex(NativeImagePtr&&, size_t, SubsamplingLevel, const DecodingOptions&, DecodingStatus = DecodingStatus::Invalid);
154     void cacheNativeImageAtIndexAsync(NativeImagePtr&&, size_t, SubsamplingLevel, const DecodingOptions&, DecodingStatus);
155 
156     struct ImageFrameRequest;
157     static const int BufferSize = 8;
158     WorkQueue& decodingQueue();
159     SynchronizedFixedQueue<ImageFrameRequest, BufferSize>& frameRequestQueue();
160 
161     const ImageFrame& frameAtIndexCacheIfNeeded(size_t, ImageFrame::Caching, const Optional<SubsamplingLevel>& = { });
162 
163     void dump(TextStream&);
164 
165     BitmapImage* m_image { nullptr };
166     RefPtr<ImageDecoder> m_decoder;
167     AlphaOption m_alphaOption { AlphaOption::Premultiplied };
168     GammaAndColorProfileOption m_gammaAndColorProfileOption { GammaAndColorProfileOption::Applied };
169 
170     unsigned m_decodedSize { 0 };
171     unsigned m_decodedPropertiesSize { 0 };
172     Vector<ImageFrame, 1> m_frames;
173 
174     // Asynchronous image decoding.
175     struct ImageFrameRequest {
176         size_t index;
177         SubsamplingLevel subsamplingLevel;
178         DecodingOptions decodingOptions;
179         DecodingStatus decodingStatus;
180         bool operator==(const ImageFrameRequest& other) const
181         {
182             return index == other.index && subsamplingLevel == other.subsamplingLevel && decodingOptions == other.decodingOptions && decodingStatus == other.decodingStatus;
183         }
184     };
185     using FrameRequestQueue = SynchronizedFixedQueue<ImageFrameRequest, BufferSize>;
186     using FrameCommitQueue = Deque<ImageFrameRequest, BufferSize>;
187     RefPtr<FrameRequestQueue> m_frameRequestQueue;
188     FrameCommitQueue m_frameCommitQueue;
189     RefPtr<WorkQueue> m_decodingQueue;
190     Seconds m_frameDecodingDurationForTesting;
191 
192     // Image metadata.
193     Optional<EncodedDataStatus> m_encodedDataStatus;
194     Optional<size_t> m_frameCount;
195     Optional<RepetitionCount> m_repetitionCount;
196     Optional<String> m_uti;
197     Optional<String> m_filenameExtension;
198     Optional<Optional<IntPoint>> m_hotSpot;
199 
200     // Image metadata which is calculated from the first ImageFrame.
201     Optional<IntSize> m_size;
202     Optional<ImageOrientation> m_orientation;
203     Optional<Color> m_singlePixelSolidColor;
204     Optional<SubsamplingLevel> m_maximumSubsamplingLevel;
205 
206     RunLoop& m_runLoop;
207 };
