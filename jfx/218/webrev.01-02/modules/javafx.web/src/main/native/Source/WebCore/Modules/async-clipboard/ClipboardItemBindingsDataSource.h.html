34 namespace WebCore {
35 
36 class Blob;
37 class DOMPromise;
38 class FileReaderLoader;
39 class PasteboardCustomData;
40 class ScriptExecutionContext;
41 class SharedBuffer;
42 
43 class ClipboardItemBindingsDataSource : public ClipboardItemDataSource {
44     WTF_MAKE_FAST_ALLOCATED;
45 public:
46     ClipboardItemBindingsDataSource(ClipboardItem&, Vector<KeyValuePair<String, RefPtr<DOMPromise>>>&&);
47     ~ClipboardItemBindingsDataSource();
48 
49 private:
50     Vector<String> types() const final;
51     void getType(const String&, Ref<DeferredPromise>&&) final;
52     void collectDataForWriting(Clipboard& destination, CompletionHandler<void(Optional<PasteboardCustomData>)>&&) final;
53 
54     void invokeCompletionHandler();
55 
56     using BufferOrString = Variant<String, Ref<SharedBuffer>>;
57     class ClipboardItemTypeLoader : public FileReaderLoaderClient, public RefCounted<ClipboardItemTypeLoader>, public CanMakeWeakPtr<ClipboardItemTypeLoader> {
58     public:
59         static Ref<ClipboardItemTypeLoader> create(const String& type, CompletionHandler<void()>&& completionHandler)
60         {
61             return adoptRef(*new ClipboardItemTypeLoader(type, WTFMove(completionHandler)));
62         }
63 
64         ~ClipboardItemTypeLoader();
65 
66         void didResolveToString(const String&);
67         void didFailToResolve();
68         void didResolveToBlob(ScriptExecutionContext&, Ref<Blob>&&);
69 
70         const String& type() { return m_type; }
71         const BufferOrString& data() { return m_data; }
72 
73     private:
74         ClipboardItemTypeLoader(const String& type, CompletionHandler<void()>&&);
75 
76         void sanitizeDataIfNeeded();
77         void invokeCompletionHandler();
78 
79         // FileReaderLoaderClient methods.
80         void didStartLoading() final { }
81         void didReceiveData() final { }
82         void didFinishLoading() final;
83         void didFail(int) final;
84 
85         String m_type;
86         BufferOrString m_data;
87         std::unique_ptr<FileReaderLoader> m_blobLoader;
88         CompletionHandler<void()> m_completionHandler;
89     };
90 
91     unsigned m_numberOfPendingClipboardTypes { 0 };
92     CompletionHandler<void(Optional<PasteboardCustomData>)> m_completionHandler;
93     Vector<Ref<ClipboardItemTypeLoader>> m_itemTypeLoaders;
94     WeakPtr<Clipboard> m_writingDestination;
95 
96     Vector<KeyValuePair<String, RefPtr<DOMPromise>>> m_itemPromises;
97 };

99 } // namespace WebCore
