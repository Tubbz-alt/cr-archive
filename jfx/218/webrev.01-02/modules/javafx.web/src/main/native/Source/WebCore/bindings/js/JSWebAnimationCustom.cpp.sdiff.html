54 EncodedJSValue JSC_HOST_CALL constructJSWebAnimation(ExecState& state)
55 {
56     VM& vm = state.vm();
57     auto throwScope = DECLARE_THROW_SCOPE(vm);
58     UNUSED_PARAM(throwScope);
59     auto* jsConstructor = jsCast<JSDOMConstructorBase*>(state.jsCallee());
60     ASSERT(jsConstructor);
61     auto* context = jsConstructor->scriptExecutionContext();
62     if (UNLIKELY(!context))
63         return throwConstructorScriptExecutionContextUnavailableError(state, throwScope, "Animation");
64     auto& document = downcast<Document>(*context);
65     auto effect = convert<IDLNullable<IDLInterface<AnimationEffect>>>(state, state.argument(0), [](ExecState& state, ThrowScope& scope) {
66         throwArgumentTypeError(state, scope, 0, "effect", "Animation", nullptr, "AnimationEffect");
67     });
68     RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
69 
70     if (state.argument(1).isUndefined()) {
71         auto object = WebAnimation::create(document, WTFMove(effect));
72         return JSValue::encode(toJSNewlyCreated<IDLInterface<WebAnimation>>(state, *jsConstructor->globalObject(), WTFMove(object)));
73     }
74 
75     auto timeline = convert<IDLNullable<IDLInterface<AnimationTimeline>>>(state, state.uncheckedArgument(1), [](ExecState& state, ThrowScope& scope) {
76         throwArgumentTypeError(state, scope, 1, "timeline", "Animation", nullptr, "AnimationTimeline");
77     });
78     RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
79     auto object = WebAnimation::create(document, WTFMove(effect), WTFMove(timeline));
80     return JSValue::encode(toJSNewlyCreated<IDLInterface<WebAnimation>>(state, *jsConstructor->globalObject(), WTFMove(object)));
81 }

61     ASSERT(jsConstructor);

63     if (UNLIKELY(!context))
64         return throwConstructorScriptExecutionContextUnavailableError(*lexicalGlobalObject, throwScope, "Animation");

66     auto effect = convert<IDLNullable<IDLInterface<AnimationEffect>>>(*lexicalGlobalObject, callFrame.argument(0), [](JSGlobalObject& lexicalGlobalObject, ThrowScope& scope) {
67         throwArgumentTypeError(lexicalGlobalObject, scope, 0, "effect", "Animation", nullptr, "AnimationEffect");
68     });
69     RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
