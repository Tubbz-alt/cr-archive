406 def isBranch(instruction)
407     instruction =~ /^b/
408 end
409 
410 def hasFallThrough(instruction)
411     instruction != "ret" and instruction != "jmp"
412 end
413 
414 def isPowerOfTwo(value)
415     return false if value <= 0
416     (value & (value - 1)).zero?
417 end
