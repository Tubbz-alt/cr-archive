21 #pragma once
22 
23 #include "DeprecatedCSSOMPrimitiveValue.h"
24 #include <wtf/Ref.h>
25 
26 namespace WebCore {
27 
28 class DeprecatedCSSOMRGBColor final : public RefCounted<DeprecatedCSSOMRGBColor> {
29 public:
30     static Ref<DeprecatedCSSOMRGBColor> create(CSSStyleDeclaration& owner, const Color& color)
31     {
32         return adoptRef(*new DeprecatedCSSOMRGBColor(owner, color));
33     }
34 
35     DeprecatedCSSOMPrimitiveValue& red() { return m_red; }
36     DeprecatedCSSOMPrimitiveValue& green() { return m_green; }
37     DeprecatedCSSOMPrimitiveValue& blue() { return m_blue; }
38     DeprecatedCSSOMPrimitiveValue& alpha() { return m_alpha; }
39 
40     Color color() const { return m_color; }
41 
42 private:
43     template<typename NumberType> static Ref<DeprecatedCSSOMPrimitiveValue> createWrapper(CSSStyleDeclaration& owner, NumberType number)
44     {
45         return CSSPrimitiveValue::create(number, CSSUnitType::CSS_NUMBER)->createDeprecatedCSSOMPrimitiveWrapper(owner);
46     }
47 
48     DeprecatedCSSOMRGBColor(CSSStyleDeclaration& owner, const Color& color)
49         : m_color(color)
50         , m_red(createWrapper(owner, color.red()))
51         , m_green(createWrapper(owner, color.green()))
52         , m_blue(createWrapper(owner, color.blue()))
53         , m_alpha(createWrapper(owner, color.alphaAsFloat()))
54     {
55     }
56 
57     Color m_color;
58     Ref<DeprecatedCSSOMPrimitiveValue> m_red;
59     Ref<DeprecatedCSSOMPrimitiveValue> m_green;
60     Ref<DeprecatedCSSOMPrimitiveValue> m_blue;
61     Ref<DeprecatedCSSOMPrimitiveValue> m_alpha;
62 };
63 
64 } // namespace WebCore
