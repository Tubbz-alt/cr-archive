26 function next()
27 {
28     "use strict";
29 
30     if (@isUndefinedOrNull(this))
31         @throwTypeError("%StringIteratorPrototype%.next requires that |this| not be null or undefined");
32 
33     var position = @getByIdDirectPrivate(this, "stringIteratorNextIndex");
34     if (position === @undefined)
35         @throwTypeError("%StringIteratorPrototype%.next requires that |this| be a String Iterator instance");
36 
37     var done = true;
38     var value = @undefined;
39 
40     var string = @getByIdDirectPrivate(this, "iteratedString");
41     if (string !== @undefined) {

42         var length = string.length >>> 0;
43         if (position >= length)
44             @putByIdDirectPrivate(this, "iteratedString", @undefined);
45         else {
46             done = false;
47 
48             var first = string.@charCodeAt(position);
49             if (first < 0xD800 || first > 0xDBFF || position + 1 === length)
50                 value = string[position];
51             else {
52                 var second = string.@charCodeAt(position + 1);
53                 if (second < 0xDC00 || second > 0xDFFF)
54                     value = string[position];
55                 else
56                     value = string[position] + string[position + 1];
57             }
58 
59             @putByIdDirectPrivate(this, "stringIteratorNextIndex", position + value.length);
60         }
61     }
62 
63     return { value, done };
64 }
