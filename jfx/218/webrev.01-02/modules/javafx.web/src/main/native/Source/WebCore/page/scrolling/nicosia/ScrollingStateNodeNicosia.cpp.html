34 namespace WebCore {
35 
36 void LayerRepresentation::retainPlatformLayer(void* typelessLayer)
37 {
38     if (auto* layer = makePlatformLayerTyped(typelessLayer))
39         layer->ref();
40 }
41 
42 void LayerRepresentation::releasePlatformLayer(void* typelessLayer)
43 {
44     if (auto* layer = makePlatformLayerTyped(typelessLayer))
45         layer->deref();
46 }
47 
48 PlatformLayer* LayerRepresentation::makePlatformLayerTyped(void* typelessLayer)
49 {
50     return static_cast<PlatformLayer*>(typelessLayer);
51 }
52 
53 void* LayerRepresentation::makePlatformLayerTypeless(PlatformLayer* layer)
54 {
55     return layer;
56 }
