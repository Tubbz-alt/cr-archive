75 SceneIntegration::UpdateScope::UpdateScope(Ref<SceneIntegration>&& sceneIntegration)
76     : m_sceneIntegration(WTFMove(sceneIntegration))
77     , m_locker(m_sceneIntegration->m_client.lock)
78 {
79 }
80 
81 SceneIntegration::UpdateScope::~UpdateScope()
82 {
83     if (!m_sceneIntegration->m_client.scene)
84         return;
85 
86     m_sceneIntegration->m_client.scene->accessState(
87         [](Nicosia::Scene::State& state)
88         {
89             for (auto& compositionLayer : state.layers)
90                 compositionLayer->flushState([](auto&) { });
91         });
92 
93     auto& sceneIntegrationObj = m_sceneIntegration.get();
94     if (sceneIntegrationObj.m_client.object)
95         sceneIntegrationObj.m_client.object->requestUpdate();
96 }
