22 #include "config.h"
23 #include "SVGSymbolElement.h"
24 
25 #include "RenderSVGHiddenContainer.h"
26 #include "SVGFitToViewBox.h"
27 #include "SVGNames.h"
28 #include <wtf/IsoMallocInlines.h>
29 
30 namespace WebCore {
31 
32 WTF_MAKE_ISO_ALLOCATED_IMPL(SVGSymbolElement);
33 
34 inline SVGSymbolElement::SVGSymbolElement(const QualifiedName& tagName, Document& document)
35     : SVGElement(tagName, document)
36     , SVGExternalResourcesRequired(this)
37     , SVGFitToViewBox(this)
38 {
39     ASSERT(hasTagName(SVGNames::symbolTag));
40 }
41 
42 Ref<SVGSymbolElement> SVGSymbolElement::create(const QualifiedName& tagName, Document& document)
43 {
44     return adoptRef(*new SVGSymbolElement(tagName, document));
45 }
46 
47 void SVGSymbolElement::parseAttribute(const QualifiedName& name, const AtomString& value)
48 {
49     SVGElement::parseAttribute(name, value);
50     SVGExternalResourcesRequired::parseAttribute(name, value);
51     SVGFitToViewBox::parseAttribute(name, value);
52 }
53 
54 void SVGSymbolElement::svgAttributeChanged(const QualifiedName& attrName)
55 {
56     SVGElement::svgAttributeChanged(attrName);
57     SVGExternalResourcesRequired::svgAttributeChanged(attrName);
58 }
59 
60 bool SVGSymbolElement::selfHasRelativeLengths() const
61 {
62     return hasAttribute(SVGNames::viewBoxAttr);
63 }
64 
65 RenderPtr<RenderElement> SVGSymbolElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
66 {
67     return createRenderer<RenderSVGHiddenContainer>(*this, WTFMove(style));
68 }
