30 #include "GraphicsContext3D.h"


31 #include <wtf/StdLibExtras.h>
32 #include <wtf/UniqueArray.h>
33 
34 namespace WebCore {
35 
36 class FormatConverter {
37 public:
38     FormatConverter(unsigned width, unsigned height, const void* srcStart, void* dstStart, int srcStride, int dstStride)
39         : m_width(width)
40         , m_height(height)
41         , m_srcStart(srcStart)
42         , m_dstStart(dstStart)
43         , m_srcStride(srcStride)
44         , m_dstStride(dstStride)
45         , m_success(false)
46     {
47         const unsigned MaxNumberOfComponents = 4;
48         const unsigned MaxBytesPerComponent  = 4;
49         m_unpackedIntermediateSrcData = makeUniqueArray<uint8_t>((Checked<size_t>(m_width) * MaxNumberOfComponents * MaxBytesPerComponent).unsafeGet());
50 
51         ASSERT(m_unpackedIntermediateSrcData.get());
52     }
53 
54     void convert(GraphicsContext3D::DataFormat srcFormat, GraphicsContext3D::DataFormat dstFormat, GraphicsContext3D::AlphaOp);
55     bool success() const { return m_success; }
56 
57 private:
58     template<GraphicsContext3D::DataFormat SrcFormat>
59     ALWAYS_INLINE void convert(GraphicsContext3D::DataFormat dstFormat, GraphicsContext3D::AlphaOp);
60 
61     template<GraphicsContext3D::DataFormat SrcFormat, GraphicsContext3D::DataFormat DstFormat>
62     ALWAYS_INLINE void convert(GraphicsContext3D::AlphaOp);
63 
64     template<GraphicsContext3D::DataFormat SrcFormat, GraphicsContext3D::DataFormat DstFormat, GraphicsContext3D::AlphaOp alphaOp>
65     ALWAYS_INLINE void convert();
66 
67     const unsigned m_width, m_height;
68     const void* const m_srcStart;
69     void* const m_dstStart;
70     const int m_srcStride, m_dstStride;
71     bool m_success;
72     UniqueArray<uint8_t> m_unpackedIntermediateSrcData;
73 };
74 
75 } // namespace WebCore
76 
77 #endif // ENABLE(GRAPHICS_CONTEXT_3D)
