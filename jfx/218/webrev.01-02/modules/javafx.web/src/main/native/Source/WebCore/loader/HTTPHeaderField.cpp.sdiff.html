 33 bool isTokenCharacter(UChar c)
 34 {
 35     return isASCIIAlpha(c) || isASCIIDigit(c)
 36         || c == '!' || c == '#' || c == '$'
 37         || c == '%' || c == '&' || c == '\''
 38         || c == '*' || c == '+' || c == '-'
 39         || c == '.' || c == '^' || c == '_'
 40         || c == '`' || c == '|' || c == '~';
 41 }
 42 
 43 static bool isDelimiter(UChar c)
 44 {
 45     return c == '(' || c == ')' || c == ','
 46         || c == '/' || c == ':' || c == ';'
 47         || c == '<' || c == '=' || c == '>'
 48         || c == '?' || c == '@' || c == '['
 49         || c == '\\' || c == ']' || c == '{'
 50         || c == '}' || c == '"';
 51 }

 55     return isTokenCharacter(c) || isDelimiter(c);
