44 RefPtr<TypedOMCSSStyleValue> StylePropertyMapReadOnly::reifyValue(CSSValue* value, Document& document, Element*)
45 {
46     if (!value)
47         return nullptr;
48 
49     // FIXME: Properly reify all length values.
50     if (is<CSSPrimitiveValue>(*value) && downcast<CSSPrimitiveValue>(*value).primitiveType() == CSSPrimitiveValue::CSS_PX)
51         return TypedOMCSSUnitValue::create(downcast<CSSPrimitiveValue>(*value).doubleValue(), "px");
52 
53     if (is<CSSImageValue>(*value))
54         return TypedOMCSSImageValue::create(downcast<CSSImageValue>(*value), document);
55 
56     // FIXME: should use raw TypedOMCSSStyleValue
57     return TypedOMCSSUnparsedValue::create(value->cssText());
58 }

60 RefPtr<TypedOMCSSStyleValue> StylePropertyMapReadOnly::customPropertyValueOrDefault(const String& name, Document& document, CSSValue* inputValue, Element* element)
61 {
62     if (!inputValue) {
63         auto* registered = document.getCSSRegisteredCustomPropertySet().get(name);
64 
65         if (registered && registered->initialValue()) {
66             auto value = registered->initialValueCopy();

70         return TypedOMCSSUnparsedValue::create(emptyString());
