38 using FetchResponsePromise = DOMPromiseDeferred<IDLInterface<FetchResponse>>;

40 void WorkerGlobalScopeFetch::fetch(WorkerGlobalScope& scope, FetchRequest::Info&& input, FetchRequest::Init&& init, Ref<DeferredPromise>&& deferred)
41 {
42     FetchResponsePromise promise = WTFMove(deferred);
43 
44     auto request = FetchRequest::create(scope, WTFMove(input), WTFMove(init));
45     if (request.hasException()) {
46         promise.reject(request.releaseException());
47         return;
48     }
49 
50     FetchResponse::fetch(scope, request.releaseReturnValue().get(), [promise = WTFMove(promise)](ExceptionOr<FetchResponse&>&& result) mutable {
51         promise.settle(WTFMove(result));
52     });
