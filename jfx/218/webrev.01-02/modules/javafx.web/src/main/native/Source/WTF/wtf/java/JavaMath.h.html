26 #pragma once
27 
28 #include <type_traits>
29 #include <cmath>
30 #include <limits>
31 #include <utility>
32 
33 namespace javamath
34 {
35     template <class T>
36     T hypot(T x, T y, T z) noexcept
37     {
38         static_assert(std::is_floating_point_v<T>);
39         x = std::abs(x);
40         y = std::abs(y);
41         z = std::abs(z);
42 
43         constexpr T infValue = std::numeric_limits<T>::infinity();
44         if (x == infValue || y == infValue || z == infValue) {
45             return infValue;
46         }
47 
48         if (y > x) {
49             std::swap(x, y);
50         }
51 
52         if (z > x) {
53             std::swap(x, z);
54         }
55 
56         constexpr T epsValue = std::numeric_limits<T>::epsilon();
57         if (x * epsValue >= y && x * epsValue >= z) {
58             return x;
59         }
60 
61         const auto yx = y / x;
62         const auto zx = z / x;
63 
64         return x * std::sqrt(1 + yx * yx + zx * zx);
65     }
66 }
