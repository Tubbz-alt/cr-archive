40 static bool isSecure(DocumentLoader& documentLoader)
41 {
42     if (!documentLoader.response().url().protocolIs("https"))
43         return false;
44 
45     if (!documentLoader.response().certificateInfo() || documentLoader.response().certificateInfo()->containsNonRootSHA1SignedCertificate())
46         return false;
47 
48     return true;
49 }

51 ExceptionOr<void> PaymentSession::canCreateSession(Document& document)
52 {
53     if (!document.frame())
54         return Exception { InvalidAccessError, "Trying to start an Apple Pay session from an inactive document." };
55 
56     if (!isSecure(*document.loader()))
57         return Exception { InvalidAccessError, "Trying to start an Apple Pay session from an insecure document." };
58 
59     auto& topDocument = document.topDocument();
60     if (&document != &topDocument) {
61         auto& topOrigin = topDocument.topOrigin();
62 
63         if (!document.securityOrigin().isSameSchemeHostPort(topOrigin))
64             return Exception { InvalidAccessError, "Trying to start an Apple Pay session from a document with an different security origin than its top-level frame." };
65 
66         for (auto* ancestorDocument = document.parentDocument(); ancestorDocument != &topDocument; ancestorDocument = ancestorDocument->parentDocument()) {
67             if (!isSecure(*ancestorDocument->loader()))
68                 return Exception { InvalidAccessError, "Trying to start an Apple Pay session from a document with an insecure parent frame." };
69 
70             if (!ancestorDocument->securityOrigin().isSameSchemeHostPort(topOrigin))
71                 return Exception { InvalidAccessError, "Trying to start an Apple Pay session from a document with an different security origin than its top-level frame." };
72         }
73     }
74 
75     return { };
76 }
77 
78 bool PaymentSession::enabledForContext(ScriptExecutionContext& context)
79 {
80     auto& document = downcast<Document>(context);
81     if (auto page = document.page())
82         return page->paymentCoordinator().shouldEnableApplePayAPIs(document);
83 
84     return false;
85 }
