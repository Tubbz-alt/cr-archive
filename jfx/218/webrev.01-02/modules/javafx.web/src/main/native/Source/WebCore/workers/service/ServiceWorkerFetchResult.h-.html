30 #include "ContentSecurityPolicyResponseHeaders.h"
31 #include "ResourceError.h"
32 #include "ServiceWorkerRegistrationKey.h"
33 #include "ServiceWorkerTypes.h"
34 
35 namespace WebCore {
36 
37 struct ServiceWorkerFetchResult {
38     ServiceWorkerJobDataIdentifier jobDataIdentifier;
39     ServiceWorkerRegistrationKey registrationKey;
40     String script;
41     ContentSecurityPolicyResponseHeaders contentSecurityPolicy;
42     String referrerPolicy;
43     ResourceError scriptError;
44 
45     template<class Encoder> void encode(Encoder&) const;
46     template<class Decoder> static bool decode(Decoder&, ServiceWorkerFetchResult&);
47 };
48 
49 template<class Encoder>
50 void ServiceWorkerFetchResult::encode(Encoder& encoder) const
51 {
52     encoder << jobDataIdentifier << registrationKey << script << contentSecurityPolicy << referrerPolicy << scriptError;
53 }
54 
55 template<class Decoder>
56 bool ServiceWorkerFetchResult::decode(Decoder& decoder, ServiceWorkerFetchResult& result)
57 {
58     Optional<ServiceWorkerJobDataIdentifier> jobDataIdentifier;
59     decoder >> jobDataIdentifier;
60     if (!jobDataIdentifier)
61         return false;
62     result.jobDataIdentifier = WTFMove(*jobDataIdentifier);
63 
64     auto registrationKey = ServiceWorkerRegistrationKey::decode(decoder);
65     if (!registrationKey)
66         return false;
67     std::swap(*registrationKey, result.registrationKey);
68 
69     if (!decoder.decode(result.script))
70         return false;
71     if (!decoder.decode(result.contentSecurityPolicy))
72         return false;
73     if (!decoder.decode(result.referrerPolicy))
74         return false;
75     if (!decoder.decode(result.scriptError))
76         return false;
77 
78     return true;
79 }
80 
81 } // namespace WebCore
82 
83 #endif // ENABLE(SERVICE_WORKER)
