26 #pragma once
27 
28 #include "DisallowScope.h"

29 #include <wtf/ThreadSpecific.h>
30 
31 namespace JSC {
32 
33 class DisallowVMReentry : public DisallowScope<DisallowVMReentry> {
34     WTF_MAKE_NONCOPYABLE(DisallowVMReentry);
35     typedef DisallowScope<DisallowVMReentry> Base;
36 public:
37 #ifdef NDEBUG
38 
39     ALWAYS_INLINE DisallowVMReentry(bool = false) { }
40     ALWAYS_INLINE static void initialize() { }
41 
42 #else // not NDEBUG
43 
44     DisallowVMReentry(bool enabled = true)
45         : Base(enabled)
46     { }
47 
48     static void initialize()
49     {
50         WTF::threadSpecificKeyCreate(&s_scopeReentryCount, 0);
51     }
52 
53 private:
54     static uintptr_t scopeReentryCount()
55     {
56         return reinterpret_cast<uintptr_t>(WTF::threadSpecificGet(s_scopeReentryCount));
57     }
58     static void setScopeReentryCount(uintptr_t value)
59     {
60         WTF::threadSpecificSet(s_scopeReentryCount, reinterpret_cast<void*>(value));
61     }
62 
63     JS_EXPORT_PRIVATE static WTF::ThreadSpecificKey s_scopeReentryCount;
64 
65 #endif // NDEBUG
66 
67     friend class DisallowScope<DisallowVMReentry>;
68 };
69 
70 } // namespace JSC
