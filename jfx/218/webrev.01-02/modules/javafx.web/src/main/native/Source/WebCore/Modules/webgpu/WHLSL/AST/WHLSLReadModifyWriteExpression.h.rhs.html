 34 #include <wtf/FastMalloc.h>
 35 #include <wtf/UniqueRef.h>
 36 
 37 namespace WebCore {
 38 
 39 namespace WHLSL {
 40 
 41 namespace AST {
 42 
 43 /*
 44  *  1. Evaluate m_leftValue
 45  *  2. Assign the result to m_oldValue
 46  *  3. Evaluate m_newValueExpression
 47  *  4. Assign the result to m_newValue
 48  *  5. Assign the result to m_leftValue
 49  *  6. Evaluate m_resultExpression
 50  *  7. Return the result
 51  */
 52 class ReadModifyWriteExpression final : public Expression {
 53     WTF_MAKE_FAST_ALLOCATED;
 54 public:
 55     ReadModifyWriteExpression(CodeLocation location, UniqueRef<Expression> leftValue)
 56         : Expression(location, Kind::ReadModifyWrite)
 57         , m_leftValue(WTFMove(leftValue))
 58         , m_oldValue(makeUniqueRef<VariableDeclaration>(location, Qualifiers(), nullptr, String(), nullptr, nullptr))
 59         , m_newValue(makeUniqueRef<VariableDeclaration>(location, Qualifiers(), nullptr, String(), nullptr, nullptr))
 60     {
 61     }
 62 
 63     ReadModifyWriteExpression(CodeLocation location, UniqueRef<Expression> leftValue, UniqueRef<VariableDeclaration> oldValue, UniqueRef<VariableDeclaration> newValue)
 64         : Expression(location, Kind::ReadModifyWrite)
 65         , m_leftValue(WTFMove(leftValue))
 66         , m_oldValue(WTFMove(oldValue))
 67         , m_newValue(WTFMove(newValue))
 68     {
 69     }
 70 
 71 
 72     ~ReadModifyWriteExpression() = default;
 73 
 74     ReadModifyWriteExpression(const ReadModifyWriteExpression&) = delete;
 75     ReadModifyWriteExpression(ReadModifyWriteExpression&&) = default;
 76 
 77     void setNewValueExpression(UniqueRef<Expression>&& newValueExpression)
 78     {
 79         m_newValueExpression = WTFMove(newValueExpression);
 80     }
 81 
 82     void setResultExpression(UniqueRef<Expression>&& resultExpression)
 83     {
 84         m_resultExpression = WTFMove(resultExpression);
 85     }
 86 
 87     UniqueRef<VariableReference> oldVariableReference()
 88     {
 89         return makeUniqueRef<VariableReference>(VariableReference::wrap(m_oldValue));
 90     }
 91 
 92     UniqueRef<VariableReference> newVariableReference()
 93     {
 94         return makeUniqueRef<VariableReference>(VariableReference::wrap(m_newValue));
 95     }
 96 
 97     Expression& leftValue() { return m_leftValue; }
 98     UniqueRef<Expression>& leftValueReference() { return m_leftValue; }
 99     VariableDeclaration& oldValue() { return m_oldValue; }
100     VariableDeclaration& newValue() { return m_newValue; }
101     Expression& newValueExpression()
102     {
103         ASSERT(m_newValueExpression);
104         return *m_newValueExpression;
105     }
106     Expression& resultExpression()
107     {
108         ASSERT(m_resultExpression);
109         return *m_resultExpression;
110     }
111     UniqueRef<Expression> takeLeftValue() { return WTFMove(m_leftValue); }
112     UniqueRef<VariableDeclaration> takeOldValue() { return WTFMove(m_oldValue); }
113     UniqueRef<VariableDeclaration> takeNewValue() { return WTFMove(m_newValue); }
114     UniqueRef<Expression> takeNewValueExpression()
115     {
116         auto result = WTFMove(m_newValueExpression.value());
117         m_newValueExpression.reset();
118         return result;
119     }
120     UniqueRef<Expression> takeResultExpression()
121     {
122         auto result = WTFMove(m_resultExpression.value());
123         m_resultExpression.reset();
124         return result;
125     }
126 
127 private:
128     UniqueRef<Expression> m_leftValue;
129     UniqueRef<VariableDeclaration> m_oldValue;
130     UniqueRef<VariableDeclaration> m_newValue;
131     Optional<UniqueRef<Expression>> m_newValueExpression;
132     Optional<UniqueRef<Expression>> m_resultExpression;
133 };
134 
135 } // namespace AST
136 
137 }
138 
139 }
