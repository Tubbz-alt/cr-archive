 26 #pragma once
 27 
 28 #include "AnimationTimeline.h"
 29 #include "DocumentTimelineOptions.h"
 30 #include "GenericTaskQueue.h"
 31 #include "Timer.h"
 32 #include <wtf/Markable.h>
 33 #include <wtf/Ref.h>
 34 #include <wtf/WeakPtr.h>
 35 
 36 namespace WebCore {
 37 
 38 class AnimationEventBase;
 39 class RenderElement;
 40 
 41 class DocumentTimeline final : public AnimationTimeline
 42 {
 43 public:
 44     static Ref<DocumentTimeline> create(Document&);
 45     static Ref<DocumentTimeline> create(Document&, DocumentTimelineOptions&&);
 46     ~DocumentTimeline();
 47 
 48     bool isDocumentTimeline() const final { return true; }
 49 
 50     Vector<RefPtr<WebAnimation>> getAnimations() const;
 51 
 52     Document* document() const { return m_document.get(); }
 53 
 54     Optional<Seconds> currentTime() override;
 55 
 56     void animationTimingDidChange(WebAnimation&) override;
 57     void removeAnimation(WebAnimation&) override;
 58     void animationWasAddedToElement(WebAnimation&, Element&) final;
 59     void animationWasRemovedFromElement(WebAnimation&, Element&) final;
 60 
 61     // If possible, compute the visual extent of any transform animation on the given renderer
 62     // using the given rect, returning the result in the rect. Return false if there is some
 63     // transform animation but we were unable to cheaply compute its effect on the extent.
 64     bool computeExtentOfAnimation(RenderElement&, LayoutRect&) const;
 65     std::unique_ptr<RenderStyle> animatedStyleForRenderer(RenderElement& renderer);
 66     bool isRunningAnimationOnRenderer(RenderElement&, CSSPropertyID) const;
 67     bool isRunningAcceleratedAnimationOnRenderer(RenderElement&, CSSPropertyID) const;
 68     void animationAcceleratedRunningStateDidChange(WebAnimation&);
 69     void applyPendingAcceleratedAnimations();
 70     bool runningAnimationsForElementAreAllAccelerated(Element&) const;

 71     void detachFromDocument();
 72 
 73     void enqueueAnimationEvent(AnimationEventBase&);
 74 
 75     bool scheduledUpdate() const { return m_animationResolutionScheduled; }
 76     void updateCurrentTime(DOMHighResTimeStamp timestamp);
 77     void updateAnimationsAndSendEvents();
 78 
 79     void updateThrottlingState();
 80     WEBCORE_EXPORT Seconds animationInterval() const;
 81     WEBCORE_EXPORT void suspendAnimations();
 82     WEBCORE_EXPORT void resumeAnimations();
 83     WEBCORE_EXPORT bool animationsAreSuspended();
 84     WEBCORE_EXPORT unsigned numberOfActiveAnimationsForTesting() const;
 85     WEBCORE_EXPORT Vector<std::pair<String, double>> acceleratedAnimationsForElement(Element&) const;
 86     WEBCORE_EXPORT unsigned numberOfAnimationTimelineInvalidationsForTesting() const;
 87 
 88 private:
 89     DocumentTimeline(Document&, Seconds);
 90 
 91     DOMHighResTimeStamp liveCurrentTime() const;
 92     void cacheCurrentTime(DOMHighResTimeStamp);
 93     void maybeClearCachedCurrentTime();
 94     void scheduleInvalidationTaskIfNeeded();

 95     void scheduleAnimationResolution();
 96     void clearTickScheduleTimer();
 97     void internalUpdateAnimationsAndSendEvents();

 98     void updateListOfElementsWithRunningAcceleratedAnimationsForElement(Element&);
 99     void transitionDidComplete(RefPtr<CSSTransition>);
100     void scheduleNextTick();
101     void removeReplacedAnimations();
102     bool animationCanBeRemoved(WebAnimation&);
103     bool shouldRunUpdateAnimationsAndSendEventsIgnoringSuspensionState() const;
104 
105     Timer m_tickScheduleTimer;
106     GenericTaskQueue<Timer> m_currentTimeClearingTaskQueue;
107     HashSet<RefPtr<WebAnimation>> m_acceleratedAnimationsPendingRunningStateChange;
108     HashSet<Element*> m_elementsWithRunningAcceleratedAnimations;
109     Vector<Ref<AnimationEventBase>> m_pendingAnimationEvents;
110     WeakPtr<Document> m_document;
111     Markable<Seconds, Seconds::MarkableTraits> m_cachedCurrentTime;
112     Seconds m_originTime;
113     unsigned m_numberOfAnimationTimelineInvalidationsForTesting { 0 };
114     bool m_isSuspended { false };
115     bool m_waitingOnVMIdle { false };
116     bool m_animationResolutionScheduled { false };
117     bool m_shouldScheduleAnimationResolutionForNewPendingEvents { true };
118 };
119 
120 } // namespace WebCore
