36 class ExtendedColor : public RefCounted<ExtendedColor> {
37 public:
38     static Ref<ExtendedColor> create(float r, float g, float b, float a, ColorSpace = ColorSpace::ColorSpaceSRGB);
39 
40     float red() const { return m_red; }
41     float green() const { return m_green; }
42     float blue() const { return m_blue; }
43     float alpha() const { return m_alpha; }
44 
45     ColorSpace colorSpace() const { return m_colorSpace; }
46 
47     WEBCORE_EXPORT String cssText() const;
48 
49 private:
50     ExtendedColor(float r, float g, float b, float a, ColorSpace colorSpace)
51         : m_red(r)
52         , m_green(g)
53         , m_blue(b)
54         , m_alpha(a)
55         , m_colorSpace(colorSpace)
56     { }
57 
58     float m_red { 0 };
59     float m_green { 0 };
60     float m_blue { 0 };
61     float m_alpha { 0 };
62 
63     ColorSpace m_colorSpace { ColorSpace::ColorSpaceSRGB };
64 };
