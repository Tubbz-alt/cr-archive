29 #pragma once
30 
31 #include <memory>
32 #include <wtf/Lock.h>
33 #include <wtf/RefPtr.h>
34 #include <wtf/ThreadSafeRefCounted.h>
35 
36 namespace Nicosia {
37 
38 class Scene;
39 
40 class SceneIntegration : public ThreadSafeRefCounted<SceneIntegration> {
41 public:
42     class Client {
43     public:
44         virtual ~Client();
45 
46         virtual void requestUpdate() = 0;
47     };
48 
49     static Ref<SceneIntegration> create(Scene& scene, Client& client)
50     {
51         return adoptRef(*new SceneIntegration(scene, client));
52     }
53     ~SceneIntegration();
54 
55     void setClient(Client&);
56     void invalidate();
57 
58     void requestUpdate();
59 
60     class UpdateScope {
61         WTF_MAKE_FAST_ALLOCATED;
62     public:
63         explicit UpdateScope(Ref<SceneIntegration>&&);
64         ~UpdateScope();
65 
66     private:
67         Ref<SceneIntegration> m_sceneIntegration;
68         LockHolder m_locker;
69     };
70 
71     std::unique_ptr<UpdateScope> createUpdateScope();
72 
73 private:
74     explicit SceneIntegration(Scene&, Client&);
75 
76     struct {
77         Lock lock;
78         RefPtr<Scene> scene;
79         Client* object { nullptr };
80     } m_client;
81 };
82 
83 } // namespace Nicosia
