37 // Define the two types of JSCallbackObjects we support.
38 template <> const ClassInfo JSCallbackObject<JSNonFinalObject>::s_info = { "CallbackObject", &Base::s_info, nullptr, nullptr, CREATE_METHOD_TABLE(JSCallbackObject) };
39 template <> const ClassInfo JSCallbackObject<JSGlobalObject>::s_info = { "CallbackGlobalObject", &Base::s_info, nullptr, nullptr, CREATE_METHOD_TABLE(JSCallbackObject) };
40 
41 template<> const bool JSCallbackObject<JSNonFinalObject>::needsDestruction = true;
42 template<> const bool JSCallbackObject<JSGlobalObject>::needsDestruction = true;
43 
44 template<>
45 JSCallbackObject<JSGlobalObject>* JSCallbackObject<JSGlobalObject>::create(VM& vm, JSClassRef classRef, Structure* structure)
46 {
47     JSCallbackObject<JSGlobalObject>* callbackObject = new (NotNull, allocateCell<JSCallbackObject<JSGlobalObject>>(vm.heap)) JSCallbackObject(vm, classRef, structure);
48     callbackObject->finishCreation(vm);
49     return callbackObject;
50 }
51 
52 template <>
53 Structure* JSCallbackObject<JSNonFinalObject>::createStructure(VM& vm, JSGlobalObject* globalObject, JSValue proto)
54 {
55     return Structure::create(vm, globalObject, proto, TypeInfo(ObjectType, StructureFlags), info());
56 }
57 
58 template <>
59 Structure* JSCallbackObject<JSGlobalObject>::createStructure(VM& vm, JSGlobalObject* globalObject, JSValue proto)
60 {
61     return Structure::create(vm, globalObject, proto, TypeInfo(GlobalObjectType, StructureFlags), info());
62 }
63 
64 template <>
65 IsoSubspace* JSCallbackObject<JSNonFinalObject>::subspaceForImpl(VM& vm, SubspaceAccess mode)
66 {
67     switch (mode) {
68     case SubspaceAccess::OnMainThread:
69         return vm.callbackObjectSpace<SubspaceAccess::OnMainThread>();
70     case SubspaceAccess::Concurrently:
71         return vm.callbackObjectSpace<SubspaceAccess::Concurrently>();
72     }
73     RELEASE_ASSERT_NOT_REACHED();
74     return nullptr;
75 }
76 
77 template <>
78 IsoSubspace* JSCallbackObject<JSGlobalObject>::subspaceForImpl(VM& vm, SubspaceAccess mode)
79 {
80     switch (mode) {
81     case SubspaceAccess::OnMainThread:
82         return vm.callbackGlobalObjectSpace<SubspaceAccess::OnMainThread>();
83     case SubspaceAccess::Concurrently:
84         return vm.callbackGlobalObjectSpace<SubspaceAccess::Concurrently>();
85     }
86     RELEASE_ASSERT_NOT_REACHED();
87     return nullptr;
88 }
