 45 void PolymorphicCallNode::unlink(VM& vm)
 46 {
 47     if (m_callLinkInfo) {
 48         dataLogLnIf(Options::dumpDisassembly(), "Unlinking polymorphic call at ", m_callLinkInfo->callReturnLocation(), ", ", m_callLinkInfo->codeOrigin());


 49         m_callLinkInfo->unlink(vm);
 50     }
 51 
 52     if (isOnList())
 53         remove();
 54 }

 66 PolymorphicCallStubRoutine::PolymorphicCallStubRoutine(
 67     const MacroAssemblerCodeRef<JITStubRoutinePtrTag>& codeRef, VM& vm, const JSCell* owner, CallFrame* callerFrame,
 68     CallLinkInfo& info, const Vector<PolymorphicCallCase>& cases,
 69     UniqueArray<uint32_t>&& fastCounts)
 70     : GCAwareJITStubRoutine(codeRef, vm)
 71     , m_variants(cases.size())
 72     , m_fastCounts(WTFMove(fastCounts))
 73 {
 74     for (unsigned index = 0; index < cases.size(); ++index) {
 75         const PolymorphicCallCase& callCase = cases[index];
 76         m_variants[index].set(vm, owner, callCase.variant().rawCalleeCell());
 77         if (shouldDumpDisassemblyFor(callerFrame->codeBlock()))
 78             dataLog("Linking polymorphic call in ", FullCodeOrigin(callerFrame->codeBlock(), callerFrame->codeOrigin()), " to ", callCase.variant(), ", codeBlock = ", pointerDump(callCase.codeBlock()), "\n");
 79         if (CodeBlock* codeBlock = callCase.codeBlock())
 80             codeBlock->linkIncomingPolymorphicCall(callerFrame, m_callNodes.add(&info));
 81     }

 82     WTF::storeStoreFence();
 83 }

 87 CallVariantList PolymorphicCallStubRoutine::variants() const
 88 {
 89     CallVariantList result;
 90     for (size_t i = 0; i < m_variants.size(); ++i)
 91         result.append(CallVariant(m_variants[i].get()));
 92     return result;
 93 }

 95 bool PolymorphicCallStubRoutine::hasEdges() const
 96 {
 97     // The FTL does not count edges in its poly call stub routines. If the FTL went poly call, then
 98     // it's not meaningful to keep profiling - we can just leave it at that. Remember, the FTL would
 99     // have had full edge profiling from the DFG, and based on this information, it would have
100     // decided to go poly.
101     //
102     // There probably are very-difficult-to-imagine corner cases where the FTL not doing edge
103     // profiling is bad for polyvariant inlining. But polyvariant inlining is profitable sometimes
104     // while not having to increment counts is profitable always. So, we let the FTL run faster and
105     // not keep counts.
106     return !!m_fastCounts;
107 }
108 
109 CallEdgeList PolymorphicCallStubRoutine::edges() const
110 {
111     RELEASE_ASSERT(m_fastCounts);
112 
113     CallEdgeList result;
114     for (size_t i = 0; i < m_variants.size(); ++i)
115         result.append(CallEdge(CallVariant(m_variants[i].get()), m_fastCounts[i]));
116     return result;
117 }

119 void PolymorphicCallStubRoutine::clearCallNodesFor(CallLinkInfo* info)
120 {
121     for (Bag<PolymorphicCallNode>::iterator iter = m_callNodes.begin(); !!iter; ++iter) {
122         PolymorphicCallNode& node = **iter;
123         // All nodes should point to info, but okay to be a little paranoid.
124         if (node.hasCallLinkInfo(info))
125             node.clearCallLinkInfo();
126     }
127 }
128 
129 bool PolymorphicCallStubRoutine::visitWeak(VM& vm)
130 {
131     bool isStillLive = true;
132     forEachDependentCell([&](JSCell* cell) {
133         isStillLive &= vm.heap.isMarked(cell);
134     });
135     return isStillLive;
136 }
137 
138 void PolymorphicCallStubRoutine::markRequiredObjectsInternal(SlotVisitor& visitor)
139 {
140     for (auto& variant : m_variants)
141         visitor.append(variant);
142 }
