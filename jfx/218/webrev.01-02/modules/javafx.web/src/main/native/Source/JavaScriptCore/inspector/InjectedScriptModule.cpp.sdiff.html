 57 void InjectedScriptModule::ensureInjected(InjectedScriptManager* injectedScriptManager, const InjectedScript& injectedScript)
 58 {
 59     ASSERT(!injectedScript.hasNoValue());
 60     if (injectedScript.hasNoValue())
 61         return;
 62 
 63     // FIXME: Make the InjectedScript a module itself.
 64     JSC::JSLockHolder locker(injectedScript.globalObject());
 65     Deprecated::ScriptFunctionCall function(injectedScript.injectedScriptObject(), "hasInjectedModule"_s, injectedScriptManager->inspectorEnvironment().functionCallHandler());
 66     function.appendArgument(name());
 67     auto hasInjectedModuleResult = injectedScript.callFunctionWithEvalEnabled(function);
 68     ASSERT(hasInjectedModuleResult);
 69     if (!hasInjectedModuleResult) {
 70         auto& error = hasInjectedModuleResult.error();
 71         ASSERT(error);
 72         unsigned line = 0;
 73         unsigned column = 0;
 74         auto& stack = error->stack();
 75         if (stack.size() > 0)
 76             stack[0].computeLineAndColumn(line, column);
 77         WTFLogAlways("Error when calling 'hasInjectedModule' for '%s': %s (%d:%d)\n", name().utf8().data(), error->value().toWTFString(injectedScript.globalObject()).utf8().data(), line, column);
 78         WTFLogAlways("%s\n", source().utf8().data());
 79         RELEASE_ASSERT_NOT_REACHED();
 80     }
 81     if (!hasInjectedModuleResult.value().isBoolean() || !hasInjectedModuleResult.value().asBoolean()) {
 82         Deprecated::ScriptFunctionCall function(injectedScript.injectedScriptObject(), "injectModule"_s, injectedScriptManager->inspectorEnvironment().functionCallHandler());
 83         function.appendArgument(name());
 84         function.appendArgument(source());
 85         function.appendArgument(host(injectedScriptManager, injectedScript.globalObject()));
 86         auto injectModuleResult = injectedScript.callFunctionWithEvalEnabled(function);
 87         if (!injectModuleResult) {
 88             auto& error = injectModuleResult.error();
 89             ASSERT(error);
 90             unsigned line = 0;
 91             unsigned column = 0;
 92             auto& stack = error->stack();
 93             if (stack.size() > 0)
 94                 stack[0].computeLineAndColumn(line, column);
 95             WTFLogAlways("Error when calling 'injectModule' for '%s': %s (%d:%d)\n", name().utf8().data(), error->value().toWTFString(injectedScript.globalObject()).utf8().data(), line, column);
 96             WTFLogAlways("%s\n", source().utf8().data());
 97             RELEASE_ASSERT_NOT_REACHED();
 98         }
 99     }
100 }
