62 inline Node* EventPath::eventTargetRespectingTargetRules(Node& referenceNode)
63 {
64     if (is<PseudoElement>(referenceNode))
65         return downcast<PseudoElement>(referenceNode).hostElement();
66 
67     // Events sent to elements inside an SVG use element's shadow tree go to the use element.
68     if (is<SVGElement>(referenceNode)) {
69         if (auto useElement = downcast<SVGElement>(referenceNode).correspondingUseElement())
70             return useElement.get();
71     }
72 
73     return &referenceNode;
74 }
