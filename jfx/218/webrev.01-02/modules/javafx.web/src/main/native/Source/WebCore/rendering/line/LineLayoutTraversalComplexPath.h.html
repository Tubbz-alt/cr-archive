 26 #pragma once
 27 
 28 #include "InlineTextBox.h"
 29 #include "RenderText.h"
 30 #include <wtf/Vector.h>
 31 
 32 namespace WebCore {
 33 namespace LineLayoutTraversal {
 34 
 35 class ComplexPath {
 36 public:
 37     ComplexPath(const InlineBox* inlineBox, Vector<const InlineTextBox*>&& sortedInlineTextBoxes = { }, size_t sortedInlineTextBoxIndex = 0)
 38         : m_inlineBox(inlineBox)
 39         , m_sortedInlineTextBoxes(WTFMove(sortedInlineTextBoxes))
 40         , m_sortedInlineTextBoxIndex(sortedInlineTextBoxIndex)
 41     { }
 42 
 43     FloatRect rect() const { return m_inlineBox->frameRect(); }
 44 
 45     bool isLeftToRightDirection() const { return m_inlineBox->isLeftToRightDirection(); }
 46     bool isHorizontal() const { return m_inlineBox->isHorizontal(); }
 47     bool dirOverride() const { return m_inlineBox->dirOverride(); }
 48     bool isLineBreak() const { return m_inlineBox->isLineBreak(); }
 49     float baselineOffset() const { return m_inlineBox->baselinePosition(AlphabeticBaseline); }
 50 
 51     bool useLineBreakBoxRenderTreeDumpQuirk() const
 52     {
 53         return !m_inlineBox->behavesLikeText();
 54     }
 55 
 56     bool hasHyphen() const { return inlineTextBox()->hasHyphen(); }
 57     StringView text() const { return StringView(inlineTextBox()->renderer().text()).substring(inlineTextBox()->start(), inlineTextBox()->len()); }
 58     unsigned localStartOffset() const { return inlineTextBox()->start(); }
 59     unsigned localEndOffset() const { return inlineTextBox()->end(); }
 60     unsigned length() const { return inlineTextBox()->len(); }
 61 
 62     inline bool isLastOnLine() const
 63     {
 64         auto* next = nextInlineTextBoxInTextOrder();
 65         return !next || &inlineTextBox()->root() != &next->root();
 66     }
 67     inline bool isLast() const { return !nextInlineTextBoxInTextOrder(); };
 68 
 69     void traverseNextTextBoxInVisualOrder() { m_inlineBox = inlineTextBox()->nextTextBox(); }
 70     void traverseNextTextBoxInTextOrder()
 71     {
 72         m_inlineBox = nextInlineTextBoxInTextOrder();
 73         if (!m_sortedInlineTextBoxes.isEmpty())
 74             ++m_sortedInlineTextBoxIndex;
 75     }
 76 
 77     bool operator==(const ComplexPath& other) const { return m_inlineBox == other.m_inlineBox; }
 78     bool atEnd() const { return !m_inlineBox; }
 79 
 80 private:
 81     const InlineTextBox* inlineTextBox() const { return downcast<InlineTextBox>(m_inlineBox); }
 82     const InlineTextBox* nextInlineTextBoxInTextOrder() const;
 83 
 84     const InlineBox* m_inlineBox;
 85     Vector<const InlineTextBox*> m_sortedInlineTextBoxes;
 86     size_t m_sortedInlineTextBoxIndex { 0 };
 87 };
 88 
 89 inline const InlineTextBox* ComplexPath::nextInlineTextBoxInTextOrder() const
 90 {
 91     if (!m_sortedInlineTextBoxes.isEmpty()) {
 92         if (m_sortedInlineTextBoxIndex + 1 < m_sortedInlineTextBoxes.size())
 93             return m_sortedInlineTextBoxes[m_sortedInlineTextBoxIndex + 1];
 94         return nullptr;
 95     }
 96     return inlineTextBox()->nextTextBox();
 97 }
 98 
 99 }
