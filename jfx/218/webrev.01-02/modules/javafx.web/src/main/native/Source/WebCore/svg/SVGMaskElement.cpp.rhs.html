 42 inline SVGMaskElement::SVGMaskElement(const QualifiedName& tagName, Document& document)
 43     : SVGElement(tagName, document)

 44     , SVGTests(this)
 45 {
 46     // Spec: If the x/y attribute is not specified, the effect is as if a value of "-10%" were specified.
 47     // Spec: If the width/height attribute is not specified, the effect is as if a value of "120%" were specified.
 48     ASSERT(hasTagName(SVGNames::maskTag));
 49 
 50     static std::once_flag onceFlag;
 51     std::call_once(onceFlag, [] {
 52         PropertyRegistry::registerProperty<SVGNames::xAttr, &SVGMaskElement::m_x>();
 53         PropertyRegistry::registerProperty<SVGNames::yAttr, &SVGMaskElement::m_y>();
 54         PropertyRegistry::registerProperty<SVGNames::widthAttr, &SVGMaskElement::m_width>();
 55         PropertyRegistry::registerProperty<SVGNames::heightAttr, &SVGMaskElement::m_height>();
 56         PropertyRegistry::registerProperty<SVGNames::maskUnitsAttr, SVGUnitTypes::SVGUnitType, &SVGMaskElement::m_maskUnits>();
 57         PropertyRegistry::registerProperty<SVGNames::maskContentUnitsAttr, SVGUnitTypes::SVGUnitType, &SVGMaskElement::m_maskContentUnits>();
 58     });
 59 }

 63     return adoptRef(*new SVGMaskElement(tagName, document));
 64 }

 66 void SVGMaskElement::parseAttribute(const QualifiedName& name, const AtomString& value)
 67 {
 68     if (name == SVGNames::maskUnitsAttr) {
 69         auto propertyValue = SVGPropertyTraits<SVGUnitTypes::SVGUnitType>::fromString(value);
 70         if (propertyValue > 0)
 71             m_maskUnits->setBaseValInternal<SVGUnitTypes::SVGUnitType>(propertyValue);
 72         return;
 73     }
 74     if (name == SVGNames::maskContentUnitsAttr) {
 75         auto propertyValue = SVGPropertyTraits<SVGUnitTypes::SVGUnitType>::fromString(value);
 76         if (propertyValue > 0)
 77             m_maskContentUnits->setBaseValInternal<SVGUnitTypes::SVGUnitType>(propertyValue);
 78         return;
 79     }
 80 
 81     SVGParsingError parseError = NoError;
 82 
 83     if (name == SVGNames::xAttr)
 84         m_x->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
 85     else if (name == SVGNames::yAttr)
 86         m_y->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
 87     else if (name == SVGNames::widthAttr)
 88         m_width->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Width, value, parseError));
 89     else if (name == SVGNames::heightAttr)
 90         m_height->setBaseValInternal(SVGLengthValue::construct(SVGLengthMode::Height, value, parseError));
 91 
 92     reportAttributeParsingError(parseError, name, value);
 93 
 94     SVGElement::parseAttribute(name, value);
 95     SVGTests::parseAttribute(name, value);

 96 }

 98 void SVGMaskElement::svgAttributeChanged(const QualifiedName& attrName)
 99 {
100     if (PropertyRegistry::isAnimatedLengthAttribute(attrName)) {
101         InstanceInvalidationGuard guard(*this);
102         invalidateSVGPresentationAttributeStyle();
103         return;
104     }
105 
106     if (PropertyRegistry::isKnownAttribute(attrName)) {
107         if (auto* renderer = this->renderer())
108             renderer->setNeedsLayout();
109         return;
110     }
111 
112     SVGElement::svgAttributeChanged(attrName);

113 }
114 
115 void SVGMaskElement::childrenChanged(const ChildChange& change)
116 {
117     SVGElement::childrenChanged(change);
118 
119     if (change.source == ChildChangeSource::Parser)
120         return;
121 
122     if (RenderObject* object = renderer())
123         object->setNeedsLayout();
124 }

126 RenderPtr<RenderElement> SVGMaskElement::createElementRenderer(RenderStyle&& style, const RenderTreePosition&)
127 {
128     return createRenderer<RenderSVGResourceMasker>(*this, WTFMove(style));
