 32 namespace WebCore {
 33 
 34 class Document;
 35 class Frame;
 36 class FrameView;
 37 class LayoutScope;
 38 class LayoutSize;
 39 class RenderBlockFlow;
 40 class RenderBox;
 41 class RenderObject;
 42 class RenderElement;
 43 class RenderLayoutState;
 44 class RenderView;
 45 #if ENABLE(LAYOUT_FORMATTING_CONTEXT)
 46 namespace Layout {
 47 class LayoutState;
 48 class LayoutTreeContent;
 49 }

 51 
 52 class FrameViewLayoutContext {
 53 public:
 54     FrameViewLayoutContext(FrameView&);
 55     ~FrameViewLayoutContext();
 56 
 57     void layout();
 58     bool needsLayout() const;
 59 
 60     // We rely on the side-effects of layout, like compositing updates, to update state in various subsystems
 61     // whose dependencies are poorly defined. This call triggers such updates.
 62     void setNeedsLayoutAfterViewConfigurationChange();
 63 
 64     void scheduleLayout();
 65     void scheduleSubtreeLayout(RenderElement& layoutRoot);
 66     void unscheduleLayout();
 67 
 68     void startDisallowingLayout() { ++m_layoutDisallowedCount; }
 69     void endDisallowingLayout() { ASSERT(m_layoutDisallowedCount > 0); --m_layoutDisallowedCount; }
 70 
 71     void disableSetNeedsLayout();
 72     void enableSetNeedsLayout();
 73 
 74     enum class LayoutPhase : uint8_t {
 75         OutsideLayout,
 76         InPreLayout,
 77         InRenderTreeLayout,
 78         InViewSizeAdjust,
 79         InPostLayout
 80     };
 81     LayoutPhase layoutPhase() const { return m_layoutPhase; }
 82     bool isLayoutNested() const { return m_layoutNestedState == LayoutNestedState::Nested; }
 83     bool isLayoutPending() const { return m_layoutTimer.isActive(); }
 84     bool isInLayout() const { return layoutPhase() != LayoutPhase::OutsideLayout; }
 85     bool isInRenderTreeLayout() const { return layoutPhase() == LayoutPhase::InRenderTreeLayout; }
 86     bool inPaintableState() const { return layoutPhase() != LayoutPhase::InRenderTreeLayout && layoutPhase() != LayoutPhase::InViewSizeAdjust && (layoutPhase() != LayoutPhase::InPostLayout || inAsynchronousTasks()); }
 87 
 88     unsigned layoutCount() const { return m_layoutCount; }
 89 
 90     RenderElement* subtreeLayoutRoot() const;
 91     void clearSubtreeLayoutRoot() { m_subtreeLayoutRoot.clear(); }
 92     void convertSubtreeLayoutToFullLayout();
 93 
 94     void reset();
 95     void resetFirstLayoutFlag() { m_firstLayout = true; }
 96     bool didFirstLayout() const { return !m_firstLayout; }
 97 
 98     void setNeedsFullRepaint() { m_needsFullRepaint = true; }
 99     bool needsFullRepaint() const { return m_needsFullRepaint; }
100 
101     void flushAsynchronousTasks();
102 
103     RenderLayoutState* layoutState() const PURE_FUNCTION;
104     // Returns true if layoutState should be used for its cached offset and clip.
105     bool isPaintOffsetCacheEnabled() const { return !m_paintOffsetCacheDisableCount && layoutState(); }
106 #ifndef NDEBUG
107     void checkLayoutState();
108 #endif
109     // layoutDelta is used transiently during layout to store how far an object has moved from its
110     // last layout location, in order to repaint correctly.
111     // If we're doing a full repaint m_layoutState will be 0, but in that case layoutDelta doesn't matter.
112     LayoutSize layoutDelta() const;
113     void addLayoutDelta(const LayoutSize& delta);
114 #if ASSERT_ENABLED
115     bool layoutDeltaMatches(const LayoutSize& delta);
116 #endif
117     using LayoutStateStack = Vector<std::unique_ptr<RenderLayoutState>>;
118 
119 #if ENABLE(LAYOUT_FORMATTING_CONTEXT)
120     const Layout::LayoutState* layoutFormattingState() const { return m_layoutState.get(); }
121     Layout::LayoutTreeContent* layoutTreeContent() const { return m_layoutTreeContent.get(); }
122     void invalidateLayoutTreeContent();
123 #endif
124 
125 private:
126     friend class LayoutScope;
127     friend class LayoutStateMaintainer;
128     friend class LayoutStateDisabler;
129     friend class SubtreeLayoutStateMaintainer;
130     friend class PaginatedLayoutStateMaintainer;
131 
132     bool canPerformLayout() const;
133     bool layoutDisallowed() const { return m_layoutDisallowedCount; }
134     bool isLayoutSchedulingEnabled() const { return m_layoutSchedulingIsEnabled; }
135 
136     void layoutTimerFired();
137     void runAsynchronousTasks();
138     void runOrScheduleAsynchronousTasks();
139     bool inAsynchronousTasks() const { return m_inAsynchronousTasks; }
140 
141     void setSubtreeLayoutRoot(RenderElement&);
142 
143 #if ENABLE(TEXT_AUTOSIZING)
144     void applyTextSizingIfNeeded(RenderElement& layoutRoot);
145 #endif
146     void updateStyleForLayout();
147 
148     bool handleLayoutWithFrameFlatteningIfNeeded();
149     void startLayoutAtMainFrameViewIfNeeded();
150 
151     // These functions may only be accessed by LayoutStateMaintainer.
152     // Subtree push/pop
153     void pushLayoutState(RenderElement&);
154     bool pushLayoutStateForPaginationIfNeeded(RenderBlockFlow&);
155     bool pushLayoutState(RenderBox& renderer, const LayoutSize& offset, LayoutUnit pageHeight = 0_lu, bool pageHeightChanged = false);
156     void popLayoutState();
157 
158     // Suspends the LayoutState optimization. Used under transforms that cannot be represented by
159     // LayoutState (common in SVG) and when manipulating the render tree during layout in ways
160     // that can trigger repaint of a non-child (e.g. when a list item moves its list marker around).
161     // Note that even when disabled, LayoutState is still used to store layoutDelta.
162     // These functions may only be accessed by LayoutStateMaintainer or LayoutStateDisabler.
163     void disablePaintOffsetCache() { m_paintOffsetCacheDisableCount++; }
164     void enablePaintOffsetCache() { ASSERT(m_paintOffsetCacheDisableCount > 0); m_paintOffsetCacheDisableCount--; }
165 #if ENABLE(LAYOUT_FORMATTING_CONTEXT)
166     void layoutUsingFormattingContext();
167 #endif
168 
169     Frame& frame() const;
170     FrameView& view() const;
171     RenderView* renderView() const;
172     Document* document() const;
173 
174     FrameView& m_frameView;
175     Timer m_layoutTimer;
176     Timer m_asynchronousTasksTimer;
177     WeakPtr<RenderElement> m_subtreeLayoutRoot;
178 
179     bool m_layoutSchedulingIsEnabled { true };
180     bool m_delayedLayout { false };
181     bool m_firstLayout { true };
182     bool m_needsFullRepaint { true };
183     bool m_inAsynchronousTasks { false };
184     bool m_setNeedsLayoutWasDeferred { false };
185     LayoutPhase m_layoutPhase { LayoutPhase::OutsideLayout };
186     enum class LayoutNestedState : uint8_t  { NotInLayout, NotNested, Nested };
187     LayoutNestedState m_layoutNestedState { LayoutNestedState::NotInLayout };
188     unsigned m_layoutCount { 0 };
189     unsigned m_disableSetNeedsLayoutCount { 0 };
190     int m_layoutDisallowedCount { 0 };
191     unsigned m_paintOffsetCacheDisableCount { 0 };
192     LayoutStateStack m_layoutStateStack;
193 #if ENABLE(LAYOUT_FORMATTING_CONTEXT)
194     std::unique_ptr<Layout::LayoutState> m_layoutState;
195     std::unique_ptr<Layout::LayoutTreeContent> m_layoutTreeContent;
196 #endif
197 };
198 
199 } // namespace WebCore
