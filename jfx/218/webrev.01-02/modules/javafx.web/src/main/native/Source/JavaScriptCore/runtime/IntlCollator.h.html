 27 
 28 #if ENABLE(INTL)
 29 
 30 #include "JSObject.h"
 31 
 32 struct UCollator;
 33 
 34 namespace JSC {
 35 
 36 class IntlCollatorConstructor;
 37 class JSBoundFunction;
 38 
 39 class IntlCollator final : public JSNonFinalObject {
 40 public:
 41     using Base = JSNonFinalObject;
 42 
 43     static constexpr bool needsDestruction = true;
 44 
 45     static void destroy(JSCell* cell)
 46     {
 47         static_cast<IntlCollator*>(cell)->IntlCollator::~IntlCollator();
 48     }
 49 
 50     template<typename CellType, SubspaceAccess mode>
 51     static IsoSubspace* subspaceFor(VM& vm)
 52     {
 53         return vm.intlCollatorSpace<mode>();
 54     }
 55 
 56     static IntlCollator* create(VM&, Structure*);
 57     static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
 58 
 59     DECLARE_INFO;
 60 
 61     void initializeCollator(JSGlobalObject*, JSValue locales, JSValue optionsValue);
 62     JSValue compareStrings(JSGlobalObject*, StringView, StringView);
 63     JSObject* resolvedOptions(JSGlobalObject*);
 64 
 65     JSBoundFunction* boundCompare() const { return m_boundCompare.get(); }
 66     void setBoundCompare(VM&, JSBoundFunction*);
 67 
 68 protected:
 69     IntlCollator(VM&, Structure*);
 70     void finishCreation(VM&);
 71     static void visitChildren(JSCell*, SlotVisitor&);
 72 
 73 private:
 74     enum class Usage : uint8_t { Sort, Search };
 75     enum class Sensitivity : uint8_t { Base, Accent, Case, Variant };
 76     enum class CaseFirst : uint8_t { Upper, Lower, False };
 77 
 78     struct UCollatorDeleter {
 79         void operator()(UCollator*) const;
 80     };
 81 
 82     void createCollator(JSGlobalObject*);
 83     static ASCIILiteral usageString(Usage);
 84     static ASCIILiteral sensitivityString(Sensitivity);
 85     static ASCIILiteral caseFirstString(CaseFirst);
 86 
 87     String m_locale;
 88     String m_collation;
 89     WriteBarrier<JSBoundFunction> m_boundCompare;
 90     std::unique_ptr<UCollator, UCollatorDeleter> m_collator;
 91     Usage m_usage;
 92     Sensitivity m_sensitivity;
 93     CaseFirst m_caseFirst;
 94     bool m_numeric;
 95     bool m_ignorePunctuation;
 96     bool m_initializedCollator { false };
 97 };
 98 
 99 } // namespace JSC

101 #endif // ENABLE(INTL)
