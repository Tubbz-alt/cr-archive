59 void WebGLVertexArrayObjectOES::deleteObjectImpl(GraphicsContext3D* context3d, Platform3DObject object)
60 {
61     switch (m_type) {
62     case Type::Default:
63         break;
64     case Type::User:
65         context3d->getExtensions().deleteVertexArrayOES(object);
66         break;
67     }
68 
69     if (m_boundElementArrayBuffer)
70         m_boundElementArrayBuffer->onDetached(context3d);
71 
72     for (auto& state : m_vertexAttribState) {
73         if (state.bufferBinding)
74             state.bufferBinding->onDetached(context3d);
75     }
76 }
77 }
