44 bool PaintingEngineBasic::paint(GraphicsLayer& layer, Ref<Buffer>&& buffer, const IntRect& sourceRect, const IntRect& mappedSourceRect, const IntRect& targetRect, float contentsScale)
45 {
46     buffer->beginPainting();
47 
48     bool supportsAlpha = buffer->supportsAlpha();
49     PaintingContext::paint(buffer,
50         [&layer, sourceRect, mappedSourceRect, targetRect, contentsScale, supportsAlpha]
51         (GraphicsContext& context)
52         {
53             context.save();
54             context.clip(targetRect);
55             context.translate(targetRect.x(), targetRect.y());
56 
57             if (supportsAlpha) {
58                 context.setCompositeOperation(CompositeCopy);
59                 context.fillRect(IntRect(IntPoint::zero(), sourceRect.size()), Color::transparent);
60                 context.setCompositeOperation(CompositeSourceOver);
61             }
62 
63             context.translate(-sourceRect.x(), -sourceRect.y());
64             context.scale(FloatSize(contentsScale, contentsScale));
65 
66             layer.paintGraphicsLayerContents(context, mappedSourceRect);
67 
68             context.restore();
69         });
70

72
