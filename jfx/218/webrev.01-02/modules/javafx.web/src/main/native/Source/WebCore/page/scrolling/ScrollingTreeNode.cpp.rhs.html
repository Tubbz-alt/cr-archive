 38 ScrollingTreeNode::ScrollingTreeNode(ScrollingTree& scrollingTree, ScrollingNodeType nodeType, ScrollingNodeID nodeID)
 39     : m_scrollingTree(scrollingTree)
 40     , m_nodeType(nodeType)
 41     , m_nodeID(nodeID)
 42     , m_parent(nullptr)
 43 {
 44 }
 45 
 46 ScrollingTreeNode::~ScrollingTreeNode() = default;
 47 
 48 void ScrollingTreeNode::appendChild(Ref<ScrollingTreeNode>&& childNode)
 49 {
 50     RELEASE_ASSERT(m_scrollingTree.inCommitTreeState());
 51 
 52     childNode->setParent(this);
 53 
 54     m_children.append(WTFMove(childNode));


 55 }
 56 
 57 void ScrollingTreeNode::removeChild(ScrollingTreeNode& node)
 58 {
 59     RELEASE_ASSERT(m_scrollingTree.inCommitTreeState());

 60 
 61     size_t index = m_children.findMatching([&](auto& child) {
 62         return &node == child.ptr();
 63     });
 64 
 65     // The index will be notFound if the node to remove is a deeper-than-1-level descendant or
 66     // if node is the root state node.
 67     if (index != notFound) {
 68         m_children.remove(index);
 69         return;
 70     }
 71 
 72     for (auto& child : m_children)
 73         child->removeChild(node);
 74 }
 75 
 76 void ScrollingTreeNode::removeAllChildren()
 77 {
 78     RELEASE_ASSERT(m_scrollingTree.inCommitTreeState());
 79 
 80     m_children.clear();
 81 }
 82 
 83 bool ScrollingTreeNode::isRootNode() const
 84 {
 85     return m_scrollingTree.rootNode() == this;
 86 }
 87 
 88 void ScrollingTreeNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
 89 {
 90     if (behavior & ScrollingStateTreeAsTextBehaviorIncludeNodeIDs)
 91         ts.dumpProperty("nodeID", scrollingNodeID());
 92 }
 93 
 94 ScrollingTreeFrameScrollingNode* ScrollingTreeNode::enclosingFrameNodeIncludingSelf()
 95 {
 96     auto* node = this;
 97     while (node && !node->isFrameScrollingNode())
 98         node = node->parent();
 99 
100     return downcast<ScrollingTreeFrameScrollingNode>(node);
101 }
102 
103 ScrollingTreeScrollingNode* ScrollingTreeNode::enclosingScrollingNodeIncludingSelf()
104 {
105     auto* node = this;
106     while (node && !node->isScrollingNode())
107         node = node->parent();
108 
109     return downcast<ScrollingTreeScrollingNode>(node);
110 }
111 
112 void ScrollingTreeNode::dump(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
113 {
114     dumpProperties(ts, behavior);
115 
116     for (auto& child : m_children) {
117         TextStream::GroupScope scope(ts);
118         child->dump(ts, behavior);


119     }
120 }

122 ScrollingTreeScrollingNode* ScrollingTreeNode::scrollingNodeForPoint(LayoutPoint parentPoint) const
123 {
124     LayoutPoint localPoint = parentToLocalPoint(parentPoint);
125     LayoutPoint contentsPoint = localToContentsPoint(localPoint);
126 
127     for (auto& child : WTF::makeReversedRange(m_children)) {
128         if (auto* node = child->scrollingNodeForPoint(contentsPoint))
129             return node;


130     }
131 
132     return nullptr;
133 }
