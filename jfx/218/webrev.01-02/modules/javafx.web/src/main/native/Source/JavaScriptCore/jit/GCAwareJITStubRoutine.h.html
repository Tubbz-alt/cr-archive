 40 // Use this stub routine if you know that your code might be on stack when
 41 // either GC or other kinds of stub deletion happen. Basicaly, if your stub
 42 // routine makes calls (either to JS code or to C++ code) then you should
 43 // assume that it's possible for that JS or C++ code to do something that
 44 // causes the system to try to delete your routine. Using this routine type
 45 // ensures that the actual deletion is delayed until the GC proves that the
 46 // routine is no longer running. You can also subclass this routine if you
 47 // want to mark additional objects during GC in those cases where the
 48 // routine is known to be executing, or if you want to force this routine to
 49 // keep other routines alive (for example due to the use of a slow-path
 50 // list which does not get reclaimed all at once).
 51 class GCAwareJITStubRoutine : public JITStubRoutine {
 52 public:
 53     GCAwareJITStubRoutine(const MacroAssemblerCodeRef<JITStubRoutinePtrTag>&, VM&);
 54     virtual ~GCAwareJITStubRoutine();
 55 
 56     static Ref<JITStubRoutine> create(const MacroAssemblerCodeRef<JITStubRoutinePtrTag>& code, VM& vm)
 57     {
 58         return adoptRef(*new GCAwareJITStubRoutine(code, vm));
 59     }
 60 
 61     void markRequiredObjects(SlotVisitor& visitor)
 62     {
 63         markRequiredObjectsInternal(visitor);
 64     }
 65 
 66     void deleteFromGC();
 67 
 68 protected:
 69     void observeZeroRefCount() override;
 70 
 71     virtual void markRequiredObjectsInternal(SlotVisitor&);
 72 
 73 private:
 74     friend class JITStubRoutineSet;
 75 
 76     bool m_mayBeExecuting { false };
 77     bool m_isJettisoned { false };
 78 };
 79 
 80 // Use this if you want to mark one additional object during GC if your stub
 81 // routine is known to be executing.
 82 class MarkingGCAwareJITStubRoutine : public GCAwareJITStubRoutine {
 83 public:
 84     MarkingGCAwareJITStubRoutine(
 85         const MacroAssemblerCodeRef<JITStubRoutinePtrTag>&, VM&, const JSCell* owner, const Vector<JSCell*>&, Bag<CallLinkInfo>&&);
 86     virtual ~MarkingGCAwareJITStubRoutine();
 87 
 88 protected:
 89     void markRequiredObjectsInternal(SlotVisitor&) override;
 90 
 91 private:
 92     Vector<WriteBarrier<JSCell>> m_cells;
 93     Bag<CallLinkInfo> m_callLinkInfos;
 94 };

 97 // The stub has exception handlers in it. So it clears itself from exception
 98 // handling table when it dies. It also frees space in CodeOrigin table
 99 // for new exception handlers to use the same DisposableCallSiteIndex.
100 class GCAwareJITStubRoutineWithExceptionHandler : public MarkingGCAwareJITStubRoutine {
101 public:
102     typedef GCAwareJITStubRoutine Base;
103 
104     GCAwareJITStubRoutineWithExceptionHandler(const MacroAssemblerCodeRef<JITStubRoutinePtrTag>&, VM&, const JSCell* owner, const Vector<JSCell*>&, Bag<CallLinkInfo>&&, CodeBlock*, DisposableCallSiteIndex);
105 
106     void aboutToDie() override;
107     void observeZeroRefCount() override;
108 
109 private:
110     CodeBlock* m_codeBlockWithExceptionHandler;
111     DisposableCallSiteIndex m_exceptionHandlerCallSiteIndex;
112 };

114 // Helper for easily creating a GC-aware JIT stub routine. For the varargs,
115 // pass zero or more JSCell*'s. This will either create a JITStubRoutine, a
116 // GCAwareJITStubRoutine, or an ObjectMarkingGCAwareJITStubRoutine as
117 // appropriate. Generally you only need to pass pointers that will be used
118 // after the first call to C++ or JS.
119 //
120 // Ref<JITStubRoutine> createJITStubRoutine(
121 //    const MacroAssemblerCodeRef<JITStubRoutinePtrTag>& code,
122 //    VM& vm,
123 //    const JSCell* owner,
124 //    bool makesCalls,
125 //    ...);
126 //
127 // Note that we don't actually use C-style varargs because that leads to
128 // strange type-related problems. For example it would preclude us from using
129 // our custom of passing '0' as NULL pointer. Besides, when I did try to write
130 // this function using varargs, I ended up with more code than this simple
131 // way.
132 
133 Ref<JITStubRoutine> createJITStubRoutine(
134     const MacroAssemblerCodeRef<JITStubRoutinePtrTag>&, VM&, const JSCell* owner, bool makesCalls,
135     const Vector<JSCell*>&, Bag<CallLinkInfo>&& callLinkInfos,
136     CodeBlock* codeBlockForExceptionHandlers, DisposableCallSiteIndex exceptionHandlingCallSiteIndex);
