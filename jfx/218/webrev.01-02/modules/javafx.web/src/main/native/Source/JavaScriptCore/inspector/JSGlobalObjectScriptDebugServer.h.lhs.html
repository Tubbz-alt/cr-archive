26 #pragma once
27 
28 #include "ScriptDebugServer.h"

29 
30 namespace Inspector {
31 
32 class JSGlobalObjectScriptDebugServer final : public ScriptDebugServer {
33     WTF_MAKE_NONCOPYABLE(JSGlobalObjectScriptDebugServer);
34     WTF_MAKE_FAST_ALLOCATED;
35 public:
36     JSGlobalObjectScriptDebugServer(JSC::JSGlobalObject&);
37     virtual ~JSGlobalObjectScriptDebugServer() { }
38 
39     JSC::JSGlobalObject& globalObject() const { return m_globalObject; }
40 


41 private:
42     void attachDebugger() override;
43     void detachDebugger(bool isBeingDestroyed) override;
44 
45     void didPause(JSC::JSGlobalObject*) override { }
46     void didContinue(JSC::JSGlobalObject*) override { }
47     void runEventLoopWhilePaused() override;
48     bool isContentScript(JSC::ExecState*) const override { return false; }
49 
50     // NOTE: Currently all exceptions are reported at the API boundary through reportAPIException.
51     // Until a time comes where an exception can be caused outside of the API (e.g. setTimeout
52     // or some other async operation in a pure JSContext) we can ignore exceptions reported here.
53     void reportException(JSC::ExecState*, JSC::Exception*) const override { }
54 
55     JSC::JSGlobalObject& m_globalObject;
56 };
57 
58 } // namespace Inspector
