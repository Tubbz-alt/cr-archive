39 ContentWidthAndMargin InlineFormattingContext::Geometry::inlineBlockWidthAndMargin(const Box& formattingContextRoot, const HorizontalConstraints& horizontalConstraints, const OverrideHorizontalValues& overrideHorizontalValues)
40 {
41     ASSERT(formattingContextRoot.isInFlow());
42 
43     // 10.3.10 'Inline-block', replaced elements in normal flow
44 
45     // Exactly as inline replaced elements.
46     if (formattingContextRoot.replaced())
47         return inlineReplacedWidthAndMargin(formattingContextRoot, horizontalConstraints, overrideHorizontalValues);
48 
49     // 10.3.9 'Inline-block', non-replaced elements in normal flow
50 
51     // If 'width' is 'auto', the used value is the shrink-to-fit width as for floating elements.
52     // A computed value of 'auto' for 'margin-left' or 'margin-right' becomes a used value of '0'.
53     // #1
54     auto width = computedValueIfNotAuto(formattingContextRoot.style().logicalWidth(), horizontalConstraints.logicalWidth);
55     if (!width)
56         width = shrinkToFitWidth(formattingContextRoot, horizontalConstraints.logicalWidth);
57 
58     // #2
59     auto computedHorizontalMargin = Geometry::computedHorizontalMargin(formattingContextRoot, horizontalConstraints);
60 
61     return ContentWidthAndMargin { *width, { computedHorizontalMargin.start.valueOr(0_lu), computedHorizontalMargin.end.valueOr(0_lu) }, computedHorizontalMargin };
62 }
63 
64 ContentHeightAndMargin InlineFormattingContext::Geometry::inlineBlockHeightAndMargin(const Box& layoutBox, const HorizontalConstraints& horizontalConstraints, const OverrideVerticalValues& overrideVerticalValues) const
65 {
66     ASSERT(layoutBox.isInFlow());
67 
68     // 10.6.2 Inline replaced elements, block-level replaced elements in normal flow, 'inline-block' replaced elements in normal flow and floating replaced elements
69     if (layoutBox.replaced())
70         return inlineReplacedHeightAndMargin(layoutBox, horizontalConstraints, { }, overrideVerticalValues);
71 
72     // 10.6.6 Complicated cases
73     // - 'Inline-block', non-replaced elements.
74     return complicatedCases(layoutBox, horizontalConstraints, overrideVerticalValues);
75 }

77 Optional<InlineLayoutUnit> InlineFormattingContext::Geometry::computedTextIndent(const Container& formattingContextRoot, const HorizontalConstraints& horizontalConstraints) const
78 {
79     auto textIndent = formattingContextRoot.style().textIndent();
80     if (textIndent == RenderStyle::initialTextIndent())
81         return { };
82     return InlineLayoutUnit { minimumValueForLength(textIndent, horizontalConstraints.logicalWidth) };
83 }
