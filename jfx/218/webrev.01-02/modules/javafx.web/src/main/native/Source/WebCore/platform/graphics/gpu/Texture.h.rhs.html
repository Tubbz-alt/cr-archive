31 #ifndef Texture_h
32 #define Texture_h
33 
34 #include "TilingData.h"
35 #include <wtf/RefCounted.h>
36 #include <wtf/RefPtr.h>
37 #include <wtf/Vector.h>
38 
39 namespace WebCore {
40 class GraphicsContextGLOpenGL;
41 
42 class IntRect;
43 
44 class Texture : public RefCounted<Texture> {
45 public:
46     ~Texture();
47     enum Format { RGBA8, BGRA8 };
48     static RefPtr<Texture> create(GraphicsContextGLOpenGL*, Format, int width, int height);
49     void bindTile(int tile);
50     void load(void* pixels);
51     void updateSubRect(void* pixels, const IntRect&);
52     Format format() const { return m_format; }
53     const TilingData& tiles() const { return m_tiles; }
54 private:
55     Texture(GraphicsContextGLOpenGL*, std::unique_ptr<Vector<unsigned>> tileTextureIds, Format, int width, int height, int maxTextureSize);
56     GraphicsContextGLOpenGL* m_context;
57     Format m_format;
58     TilingData m_tiles;
59     std::unique_ptr<Vector<unsigned>> m_tileTextureIds;
60 };
61 
62 }
63 
64 #endif // Texture_h
