40 void WebGLObject::setObject(PlatformGLObject object)
41 {
42     ASSERT(!m_object);
43     ASSERT(!m_deleted);
44     m_object = object;
45 }
46 
47 void WebGLObject::deleteObject(GraphicsContextGLOpenGL* context3d)
48 {
49     m_deleted = true;
50     if (!m_object)
51         return;
52 
53     if (!hasGroupOrContext())
54         return;
55 
56     if (!m_attachmentCount) {
57         if (!context3d)
58             context3d = getAGraphicsContextGL();
59 
60         if (context3d)
61             deleteObjectImpl(context3d, m_object);
62 
63         m_object = 0;
64     }
65 }
66 
67 void WebGLObject::detach()
68 {
69     m_attachmentCount = 0; // Make sure OpenGL resource is deleted.
70 }
71 
72 void WebGLObject::onDetached(GraphicsContextGLOpenGL* context3d)
73 {
74     if (m_attachmentCount)
75         --m_attachmentCount;
76     if (m_deleted)
77         deleteObject(context3d);
78 }
