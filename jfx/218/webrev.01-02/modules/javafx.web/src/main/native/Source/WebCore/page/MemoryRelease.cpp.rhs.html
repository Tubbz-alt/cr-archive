 63 static void releaseNoncriticalMemory(MaintainMemoryCache maintainMemoryCache)
 64 {
 65     RenderTheme::singleton().purgeCaches();
 66 
 67     FontCache::singleton().purgeInactiveFontData();
 68 
 69     clearWidthCaches();
 70     TextPainter::clearGlyphDisplayLists();
 71 
 72     for (auto* document : Document::allDocuments()) {
 73         document->clearSelectorQueryCache();
 74 
 75 #if ENABLE(LAYOUT_FORMATTING_CONTEXT)
 76         if (auto* renderView = document->renderView())
 77             LayoutIntegration::LineLayout::releaseCaches(*renderView);
 78 #endif
 79     }
 80 
 81     if (maintainMemoryCache == MaintainMemoryCache::No)
 82         MemoryCache::singleton().pruneDeadResourcesToSize(0);
 83 
 84     InlineStyleSheetOwner::clearCache();
 85 }
 86 
 87 static void releaseCriticalMemory(Synchronous synchronous, MaintainBackForwardCache maintainBackForwardCache, MaintainMemoryCache maintainMemoryCache)
 88 {
 89     // Right now, the only reason we call release critical memory while not under memory pressure is if the process is about to be suspended.
 90     if (maintainBackForwardCache == MaintainBackForwardCache::No) {
 91         PruningReason pruningReason = MemoryPressureHandler::singleton().isUnderMemoryPressure() ? PruningReason::MemoryPressure : PruningReason::ProcessSuspended;
 92         BackForwardCache::singleton().pruneToSizeNow(0, pruningReason);
 93     }
 94 
 95     if (maintainMemoryCache == MaintainMemoryCache::No) {
 96         auto shouldDestroyDecodedDataForAllLiveResources = true;
 97         MemoryCache::singleton().pruneLiveResourcesToSize(0, shouldDestroyDecodedDataForAllLiveResources);
 98     }
 99 
100     CSSValuePool::singleton().drain();
101 
102     for (auto& document : copyToVectorOf<RefPtr<Document>>(Document::allDocuments())) {
103         document->styleScope().releaseMemory();
104         document->fontSelector().emptyCaches();
105         document->cachedResourceLoader().garbageCollectDocumentResources();
106     }
107 
108     GCController::singleton().deleteAllCode(JSC::DeleteAllCodeIfNotCollecting);
109 
110 #if ENABLE(VIDEO)
111     for (auto* mediaElement : HTMLMediaElement::allMediaElements()) {
112         if (mediaElement->paused())
113             mediaElement->purgeBufferedDataIfPossible();
114     }
115 #endif
116 
117     if (synchronous == Synchronous::Yes) {
118         GCController::singleton().garbageCollectNow();
119     } else {
120 #if PLATFORM(IOS_FAMILY)
121         GCController::singleton().garbageCollectNowIfNotDoneRecently();
122 #else
123         GCController::singleton().garbageCollectSoon();
124 #endif
125     }
126 }

128 void releaseMemory(Critical critical, Synchronous synchronous, MaintainBackForwardCache maintainBackForwardCache, MaintainMemoryCache maintainMemoryCache)
129 {
130     TraceScope scope(MemoryPressureHandlerStart, MemoryPressureHandlerEnd, static_cast<uint64_t>(critical), static_cast<uint64_t>(synchronous));
131 
132     if (critical == Critical::Yes) {
133         // Return unused pages back to the OS now as this will likely give us a little memory to work with.
134         WTF::releaseFastMallocFreeMemory();
135         releaseCriticalMemory(synchronous, maintainBackForwardCache, maintainMemoryCache);
136     }
137 
138     releaseNoncriticalMemory(maintainMemoryCache);
139 
140     platformReleaseMemory(critical);
141 
142     if (synchronous == Synchronous::Yes) {
143         // FastMalloc has lock-free thread specific caches that can only be cleared from the thread itself.
144         WorkerThread::releaseFastMallocFreeMemoryInAllThreads();
145 #if ENABLE(ASYNC_SCROLLING) && !PLATFORM(IOS_FAMILY)
146         ScrollingThread::dispatch(WTF::releaseFastMallocFreeMemory);
147 #endif
148         WTF::releaseFastMallocFreeMemory();
149     }
150 
151 #if ENABLE(RESOURCE_USAGE)
152     Page::forEachPage([&](Page& page) {
153         InspectorInstrumentation::didHandleMemoryPressure(page, critical);
154     });
155 #endif
156 }

158 #if !RELEASE_LOG_DISABLED
159 static unsigned pageCount()
160 {
161     unsigned count = 0;
162     Page::forEachPage([&] (Page& page) {
163         if (!page.isUtilityPage())
164             ++count;
165     });
166     return count;
167 }

170 void logMemoryStatisticsAtTimeOfDeath()
171 {
172 #if !RELEASE_LOG_DISABLED
173 #if PLATFORM(COCOA)
174     auto pageSize = vmPageSize();
175     auto pages = pagesPerVMTag();
176 
177     RELEASE_LOG(MemoryPressure, "Dirty memory per VM tag at time of death:");
178     for (unsigned i = 0; i < 256; ++i) {
179         size_t dirty = pages[i].dirty * pageSize;
180         if (!dirty)
181             continue;
182         String tagName = displayNameForVMTag(i);
183         if (!tagName)
184             tagName = makeString("Tag ", i);
185         RELEASE_LOG(MemoryPressure, "%16s: %lu MB", tagName.latin1().data(), dirty / MB);
186     }
187 #endif
188 
189     auto& vm = commonVM();
190     RELEASE_LOG(MemoryPressure, "Memory usage statistics at time of death:");
191     RELEASE_LOG(MemoryPressure, "GC heap size: %zu", vm.heap.size());
192     RELEASE_LOG(MemoryPressure, "GC heap extra memory size: %zu", vm.heap.extraMemorySize());
193 #if ENABLE(RESOURCE_USAGE)
194     RELEASE_LOG(MemoryPressure, "GC heap external memory: %zu", vm.heap.externalMemorySize());
195 #endif
196     RELEASE_LOG(MemoryPressure, "Global object count: %zu", vm.heap.globalObjectCount());
197 
198     RELEASE_LOG(MemoryPressure, "Page count: %u", pageCount());
199     RELEASE_LOG(MemoryPressure, "Document count: %u", Document::allDocuments().size());
200     RELEASE_LOG(MemoryPressure, "Live JavaScript objects:");
201     for (auto& it : *vm.heap.objectTypeCounts())
202         RELEASE_LOG(MemoryPressure, "  %s: %d", it.key, it.value);
203 #endif
204 }
