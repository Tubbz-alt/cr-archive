 73 JSValue ErrorEvent::error(JSGlobalObject& globalObject)
 74 {
 75     JSValue error = m_error;
 76     if (!error)
 77         return jsNull();
 78 
 79     if (!isWorldCompatible(globalObject, error)) {
 80         // We need to make sure ErrorEvents do not leak their error property across isolated DOM worlds.
 81         // Ideally, we would check that the worlds have different privileges but that's not possible yet.
 82         auto serializedError = trySerializeError(globalObject);
 83         if (!serializedError)
 84             return jsNull();
 85         return serializedError->deserialize(globalObject, &globalObject);
 86     }
 87 
 88     return error;
 89 }
 90 
 91 RefPtr<SerializedScriptValue> ErrorEvent::trySerializeError(JSGlobalObject& exec)
 92 {
 93     if (!m_serializedError && !m_triedToSerialize) {
 94         m_serializedError = SerializedScriptValue::create(exec, m_error, SerializationErrorMode::NonThrowing);
 95         m_triedToSerialize = true;
 96     }
 97     return m_serializedError;
 98 }
