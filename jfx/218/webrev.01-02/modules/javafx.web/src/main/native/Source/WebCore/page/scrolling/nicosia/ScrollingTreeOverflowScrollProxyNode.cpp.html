 42 Ref<ScrollingTreeOverflowScrollProxyNode> ScrollingTreeOverflowScrollProxyNode::create(ScrollingTree& scrollingTree, ScrollingNodeID nodeID)
 43 {
 44     return adoptRef(*new ScrollingTreeOverflowScrollProxyNode(scrollingTree, nodeID));

 47 ScrollingTreeOverflowScrollProxyNode::ScrollingTreeOverflowScrollProxyNode(ScrollingTree& scrollingTree, ScrollingNodeID nodeID)
 48     : ScrollingTreeNode(scrollingTree, ScrollingNodeType::OverflowProxy, nodeID)
 49 {
 50 }

 52 ScrollingTreeOverflowScrollProxyNode::~ScrollingTreeOverflowScrollProxyNode() = default;

 54 void ScrollingTreeOverflowScrollProxyNode::commitStateBeforeChildren(const ScrollingStateNode& stateNode)
 55 {
 56     const ScrollingStateOverflowScrollProxyNode& overflowProxyStateNode = downcast<ScrollingStateOverflowScrollProxyNode>(stateNode);
 57 
 58     if (overflowProxyStateNode.hasChangedProperty(ScrollingStateNode::Layer)) {
 59         auto* layer = static_cast<Nicosia::PlatformLayer*>(stateNode.layer());
 60         m_layer = downcast<Nicosia::CompositionLayer>(layer);
 61     }
 62 
 63     if (overflowProxyStateNode.hasChangedProperty(ScrollingStateOverflowScrollProxyNode::OverflowScrollingNode))
 64         m_overflowScrollingNodeID = overflowProxyStateNode.overflowScrollingNode();
 65 
 66     if (m_overflowScrollingNodeID) {
 67         scrollingTree().overflowRelatedNodes().ensure(m_overflowScrollingNodeID,
 68             [] {
 69                 return Vector<ScrollingNodeID>();
 70             }).iterator->value.append(scrollingNodeID());
 71 
 72         scrollingTree().activeOverflowScrollProxyNodes().add(*this);
 73     }
 74 }
 75 
 76 void ScrollingTreeOverflowScrollProxyNode::applyLayerPositions()
 77 {
 78     FloatPoint scrollOffset;
 79     if (auto* node = scrollingTree().nodeForID(m_overflowScrollingNodeID)) {
 80         if (is<ScrollingTreeOverflowScrollingNode>(node)) {
 81             auto& overflowNode = downcast<ScrollingTreeOverflowScrollingNode>(*node);
 82             scrollOffset = overflowNode.currentScrollOffset();
 83         }
 84     }
 85 
 86     LOG_WITH_STREAM(Scrolling, stream << "ScrollingTreeOverflowScrollProxyNode " << scrollingNodeID() << " applyLayerPositions: setting bounds origin to " << scrollOffset);
 87     m_layer->updateState(
 88         [&scrollOffset](Nicosia::CompositionLayer::LayerState& state)
 89         {
 90             state.boundsOrigin = scrollOffset;
 91             state.delta.boundsOriginChanged = true;
 92         });
 93 }
 94 
 95 void ScrollingTreeOverflowScrollProxyNode::dumpProperties(TextStream& ts, ScrollingStateTreeAsTextBehavior behavior) const
 96 {
 97     ts << "overflow scroll proxy node";
 98     ScrollingTreeNode::dumpProperties(ts, behavior);
 99 
100     if (auto* relatedOverflowNode = scrollingTree().nodeForID(m_overflowScrollingNodeID)) {
101         auto scrollPosition = downcast<ScrollingTreeOverflowScrollingNode>(*relatedOverflowNode).currentScrollPosition();
102         ts.dumpProperty("related overflow scrolling node scroll position", scrollPosition);
103     }
104 
105     if (behavior & ScrollingStateTreeAsTextBehaviorIncludeNodeIDs)
106         ts.dumpProperty("overflow scrolling node", m_overflowScrollingNodeID);
107 }
