 39 // Utilities for finding the last points where a node is live in DFG SSA. This accounts for liveness due
 40 // to OSR exit. This is usually used for enumerating over all of the program points where a node is live,
 41 // by exploring all blocks where the node is live at tail and then exploring all program points where the
 42 // node is killed. A prerequisite to using these utilities is having liveness and OSR availability
 43 // computed.
 44 
 45 // This tells you those things that die on the boundary between nodeBefore and nodeAfter. It is
 46 // conservative in the sense that it might resort to telling you some things that are still live at
 47 // nodeAfter.
 48 template<typename Functor>
 49 void forAllKilledOperands(Graph& graph, Node* nodeBefore, Node* nodeAfter, const Functor& functor)
 50 {
 51     CodeOrigin before = nodeBefore->origin.forExit;
 52 
 53     if (!nodeAfter) {
 54         graph.forAllLiveInBytecode(before, functor);
 55         return;
 56     }
 57 
 58     CodeOrigin after = nodeAfter->origin.forExit;
 59 
 60     Operand alreadyNoted;
 61     // If we MovHint something that is live at the time, then we kill the old value.
 62     if (nodeAfter->containsMovHint()) {
 63         Operand operand = nodeAfter->unlinkedOperand();
 64         if (graph.isLiveInBytecode(operand, after)) {
 65             functor(operand);
 66             alreadyNoted = operand;
 67         }
 68     }
 69 
 70     if (before == after)
 71         return;
 72 
 73     // It's easier to do this if the inline call frames are the same. This is way faster than the
 74     // other loop, below.
 75     auto* beforeInlineCallFrame = before.inlineCallFrame();
 76     if (beforeInlineCallFrame == after.inlineCallFrame()) {

 77         CodeBlock* codeBlock = graph.baselineCodeBlockFor(beforeInlineCallFrame);
 78         if (after.bytecodeIndex().checkpoint()) {
 79             ASSERT(before.bytecodeIndex().checkpoint() != after.bytecodeIndex().checkpoint());
 80             ASSERT_WITH_MESSAGE(before.bytecodeIndex().offset() == after.bytecodeIndex().offset(), "When the DFG does code motion it should change the forExit origin to match the surrounding bytecodes.");
 81 
 82             auto liveBefore = tmpLivenessForCheckpoint(*codeBlock, before.bytecodeIndex());
 83             auto liveAfter = tmpLivenessForCheckpoint(*codeBlock, after.bytecodeIndex());
 84             liveAfter.invert();
 85             liveBefore.filter(liveAfter);
 86 
 87             liveBefore.forEachSetBit([&] (size_t tmp) {
 88                 functor(remapOperand(beforeInlineCallFrame, Operand::tmp(tmp)));
 89             });
 90             // No locals can die at a checkpoint.
 91             return;
 92         }
 93 
 94         FullBytecodeLiveness& fullLiveness = graph.livenessFor(codeBlock);
 95         const FastBitVector& liveBefore = fullLiveness.getLiveness(before.bytecodeIndex(), LivenessCalculationPoint::BeforeUse);
 96         const FastBitVector& liveAfter = fullLiveness.getLiveness(after.bytecodeIndex(), LivenessCalculationPoint::BeforeUse);
 97 
 98         (liveBefore & ~liveAfter).forEachSetBit(
 99             [&] (size_t relativeLocal) {
100                 functor(remapOperand(beforeInlineCallFrame, virtualRegisterForLocal(relativeLocal)));
101             });
102         return;
103     }
104 
105     ASSERT_WITH_MESSAGE(!after.bytecodeIndex().checkpoint(), "Transitioning across a checkpoint but before and after don't share an inlineCallFrame.");
106 
107     // Detect kills the super conservative way: it is killed if it was live before and dead after.
108     BitVector liveAfter = graph.localsAndTmpsLiveInBytecode(after);
109     unsigned numLocals = graph.block(0)->variablesAtHead.numberOfLocals();
110     graph.forAllLocalsAndTmpsLiveInBytecode(
111         before,
112         [&] (Operand operand) {
113             if (operand == alreadyNoted)
114                 return;
115             unsigned offset = operand.isTmp() ? numLocals + operand.value() : operand.toLocal();
116             if (liveAfter.get(offset))
117                 return;
118             functor(operand);
119         });
120 }

122 // Tells you all of the nodes that would no longer be live across the node at this nodeIndex.
123 template<typename Functor>
124 void forAllKilledNodesAtNodeIndex(
125     Graph& graph, AvailabilityMap& availabilityMap, BasicBlock* block, unsigned nodeIndex,
126     const Functor& functor)
127 {
128     static constexpr unsigned seenInClosureFlag = 1;
129     static constexpr unsigned calledFunctorFlag = 2;
130     HashMap<Node*, unsigned> flags;
131 
132     ASSERT(nodeIndex);
133     Node* node = block->at(nodeIndex);
134 
135     graph.doToChildren(
136         node,
137         [&] (Edge edge) {
138             if (edge.doesKill()) {
139                 auto& result = flags.add(edge.node(), 0).iterator->value;
140                 if (!(result & calledFunctorFlag)) {
141                     functor(edge.node());
142                     result |= calledFunctorFlag;
143                 }
144             }
145         });
146 
147     Node* before = block->at(nodeIndex - 1);


148 
149     forAllKilledOperands(
150         graph, before, node,
151         [&] (Operand operand) {
152             availabilityMap.closeStartingWithLocal(
153                 operand,
154                 [&] (Node* node) -> bool {
155                     return flags.get(node) & seenInClosureFlag;
156                 },
157                 [&] (Node* node) -> bool {
158                     auto& resultFlags = flags.add(node, 0).iterator->value;
159                     bool result = resultFlags & seenInClosureFlag;
160                     if (!(resultFlags & calledFunctorFlag))
161                         functor(node);
162                     resultFlags |= seenInClosureFlag | calledFunctorFlag;
163                     return result;
164                 });
165         });
166 }

168 // Tells you all of the places to start searching from in a basic block. Gives you the node index at which
169 // the value is either no longer live. This pretends that nodes are dead at the end of the block, so that
170 // you can use this to do per-basic-block analyses.
171 template<typename Functor>
172 void forAllKillsInBlock(
173     Graph& graph, const CombinedLiveness& combinedLiveness, BasicBlock* block,
174     const Functor& functor)
175 {
176     for (Node* node : combinedLiveness.liveAtTail[block])
177         functor(block->size(), node);
178 
179     LocalOSRAvailabilityCalculator localAvailability(graph);
180     localAvailability.beginBlock(block);
181     // Start at the second node, because the functor is expected to only inspect nodes from the start of
182     // the block up to nodeIndex (exclusive), so if nodeIndex is zero then the functor has nothing to do.
183     for (unsigned nodeIndex = 1; nodeIndex < block->size(); ++nodeIndex) {
184         dataLogLnIf(ForAllKillsInternal::verbose, "local availability at index: ", nodeIndex, " ", localAvailability.m_availability);
185         forAllKilledNodesAtNodeIndex(
186             graph, localAvailability.m_availability, block, nodeIndex,
187             [&] (Node* node) {
188                 functor(nodeIndex, node);
189             });
190         localAvailability.executeNode(block->at(nodeIndex));
191     }
192 }
