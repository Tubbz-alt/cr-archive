32 [
33     ActiveDOMObject,
34     Conditional=MEDIA_STREAM,
35     SecureContext
36 ] interface MediaDevices : EventTarget {
37     attribute EventHandler ondevicechange;
38     Promise<sequence<MediaDeviceInfo>> enumerateDevices();
39 
40     MediaTrackSupportedConstraints getSupportedConstraints();
41     [PrivateIdentifier, PublicIdentifier] Promise<MediaStream> getUserMedia(optional MediaStreamConstraints constraints);
42     [EnabledAtRuntime=ScreenCapture] Promise<MediaStream> getDisplayMedia(optional DisplayMediaStreamConstraints constraints);
43 };
44 
45 [
46     Conditional=MEDIA_STREAM,
47 ] dictionary DisplayMediaStreamConstraints {
48     (boolean or MediaTrackConstraints) video = true;
49     (boolean or MediaTrackConstraints) audio = false;
50 };
51 
52 [
53     Conditional=MEDIA_STREAM,
54 ] dictionary MediaStreamConstraints {
55     (boolean or MediaTrackConstraints) video = false;
56     (boolean or MediaTrackConstraints) audio = false;
57 };
58 
59 [
60     Conditional=MEDIA_STREAM,
61     EnabledAtRuntime=ScreenCapture
62 ] enum DisplayCaptureSurfaceType {
63     "monitor",
64     "window",
65     "application",
66     "browser"
67 };
