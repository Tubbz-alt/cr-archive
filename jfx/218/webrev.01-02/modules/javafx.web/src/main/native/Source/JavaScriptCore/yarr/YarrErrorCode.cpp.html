 33 const char* errorMessage(ErrorCode error)
 34 {
 35 #define REGEXP_ERROR_PREFIX "Invalid regular expression: "
 36     // The order of this array must match the ErrorCode enum.
 37     static const char* errorMessages[] = {
 38         nullptr,                                                                    // NoError
 39         REGEXP_ERROR_PREFIX "regular expression too large",                         // PatternTooLarge
 40         REGEXP_ERROR_PREFIX "numbers out of order in {} quantifier",                // QuantifierOutOfOrder
 41         REGEXP_ERROR_PREFIX "nothing to repeat",                                    // QuantifierWithoutAtom
 42         REGEXP_ERROR_PREFIX "number too large in {} quantifier",                    // QuantifierTooLarge
 43         REGEXP_ERROR_PREFIX "incomplete {} quantifier for Unicode pattern",         // QuantifierIncomplete
 44         REGEXP_ERROR_PREFIX "missing )",                                            // MissingParentheses
 45         REGEXP_ERROR_PREFIX "unmatched ] or } bracket for Unicode pattern",         // BracketUnmatched
 46         REGEXP_ERROR_PREFIX "unmatched parentheses",                                // ParenthesesUnmatched
 47         REGEXP_ERROR_PREFIX "unrecognized character after (?",                      // ParenthesesTypeInvalid
 48         REGEXP_ERROR_PREFIX "invalid group specifier name",                         // InvalidGroupName
 49         REGEXP_ERROR_PREFIX "duplicate group specifier name",                       // DuplicateGroupName
 50         REGEXP_ERROR_PREFIX "missing terminating ] for character class",            // CharacterClassUnmatched
 51         REGEXP_ERROR_PREFIX "range out of order in character class",                // CharacterClassRangeOutOfOrder
 52         REGEXP_ERROR_PREFIX "invalid range in character class for Unicode pattern", // CharacterClassRangeInvalid
 53         REGEXP_ERROR_PREFIX "\\ at end of pattern",                                 // EscapeUnterminated
 54         REGEXP_ERROR_PREFIX "invalid Unicode {} escape",                            // InvalidUnicodeEscape
 55         REGEXP_ERROR_PREFIX "invalid backreference for Unicode pattern",            // InvalidBackreference
 56         REGEXP_ERROR_PREFIX "invalid escaped character for Unicode pattern",        // InvalidIdentityEscape
 57         REGEXP_ERROR_PREFIX "invalid \\c escape for Unicode pattern",               // InvalidControlLetterEscape
 58         REGEXP_ERROR_PREFIX "invalid property expression",                          // InvalidUnicodePropertyExpression
 59         REGEXP_ERROR_PREFIX "too many nested disjunctions",                         // TooManyDisjunctions
 60         REGEXP_ERROR_PREFIX "pattern exceeds string length limits",                 // OffsetTooLarge
 61         REGEXP_ERROR_PREFIX "invalid flags"                                         // InvalidRegularExpressionFlags
 62     };
 63 
 64     return errorMessages[static_cast<unsigned>(error)];
 65 }
 66 
 67 JSObject* errorToThrow(JSGlobalObject* globalObject, ErrorCode error)
 68 {
 69     switch (error) {
 70     case ErrorCode::NoError:
 71         ASSERT_NOT_REACHED();
 72         return nullptr;
 73     case ErrorCode::PatternTooLarge:
 74     case ErrorCode::QuantifierOutOfOrder:
 75     case ErrorCode::QuantifierWithoutAtom:
 76     case ErrorCode::QuantifierTooLarge:
 77     case ErrorCode::QuantifierIncomplete:
 78     case ErrorCode::MissingParentheses:
 79     case ErrorCode::BracketUnmatched:
 80     case ErrorCode::ParenthesesUnmatched:
 81     case ErrorCode::ParenthesesTypeInvalid:
 82     case ErrorCode::InvalidGroupName:
 83     case ErrorCode::DuplicateGroupName:
 84     case ErrorCode::CharacterClassUnmatched:
 85     case ErrorCode::CharacterClassRangeOutOfOrder:
 86     case ErrorCode::CharacterClassRangeInvalid:
 87     case ErrorCode::EscapeUnterminated:
 88     case ErrorCode::InvalidUnicodeEscape:
 89     case ErrorCode::InvalidBackreference:
 90     case ErrorCode::InvalidIdentityEscape:
 91     case ErrorCode::InvalidControlLetterEscape:
 92     case ErrorCode::InvalidUnicodePropertyExpression:
 93     case ErrorCode::OffsetTooLarge:
 94     case ErrorCode::InvalidRegularExpressionFlags:
 95         return createSyntaxError(globalObject, errorMessage(error));
 96     case ErrorCode::TooManyDisjunctions:
 97         return createOutOfMemoryError(globalObject, errorMessage(error));
 98     }
 99 
100     ASSERT_NOT_REACHED();
101     return nullptr;
102 }
