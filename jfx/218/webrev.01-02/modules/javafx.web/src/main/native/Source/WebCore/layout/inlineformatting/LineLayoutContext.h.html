33 namespace WebCore {
34 namespace Layout {
35 
36 struct LineCandidateContent;
37 
38 class LineLayoutContext {
39 public:
40     LineLayoutContext(const InlineFormattingContext&, const Container& formattingContextRoot, const InlineItems&);
41 
42     struct LineContent {
43         struct PartialContent {
44             bool trailingContentNeedsHyphen { false };
45             unsigned overflowContentLength { 0 };
46         };
47         Optional<unsigned> trailingInlineItemIndex;
48         Optional<PartialContent> partialContent;
49         Vector<const InlineItem*> floats;
50         const LineBuilder::RunList runList;
51         const LineBoxBuilder lineBox;
52     };
53     struct InlineItemRange {
54         bool isEmpty() const { return start == end; }
55         size_t start { 0 };
56         size_t end { 0 };
57     };
58     LineContent layoutLine(LineBuilder&, const InlineItemRange, Optional<unsigned> partialLeadingContentLength);
59     using FloatList = Vector<const InlineItem*>;
60 
61 private:
62     void nextContentForLine(LineCandidateContent&, unsigned inlineItemIndex, const InlineItemRange layoutRange, Optional<unsigned> overflowLength, InlineLayoutUnit currentLogicalRight);
63     struct Result {
64         LineBreaker::IsEndOfLine isEndOfLine { LineBreaker::IsEndOfLine::No };
65         size_t committedCount { 0 };
66         Optional <LineContent::PartialContent> partialContent { };
67         const InlineItem* revertTo { nullptr };
68     };
69     Result tryAddingFloatItems(LineBuilder&, const FloatList&);
70     Result tryAddingInlineItems(LineBreaker&, LineBuilder&, const LineCandidateContent&);
71     void rebuildLineForRevert(LineBuilder&, const InlineItem& revertTo, const InlineItemRange layoutRange);
72     void commitPartialContent(LineBuilder&, const LineBreaker::RunList&, const LineBreaker::Result::PartialTrailingContent&);
73     LineContent close(LineBuilder&, const InlineItemRange layoutRange, unsigned committedInlineItemCount, Optional<LineContent::PartialContent>);
74 
75     InlineLayoutUnit inlineItemWidth(const InlineItem&, InlineLayoutUnit contentLogicalLeft) const;
76 
77     const InlineFormattingContext& formattingContext() const { return m_inlineFormattingContext; }
78     const Container& root() const { return m_formattingContextRoot; }
79 
80     const InlineFormattingContext& m_inlineFormattingContext;
81     const Container& m_formattingContextRoot;
82     const InlineItems& m_inlineItems;
83     FloatList m_floats;
84     Optional<InlineTextItem> m_partialLeadingTextItem;
85     Optional<InlineTextItem> m_partialTrailingTextItem;
86     unsigned m_successiveHyphenatedLineCount { 0 };
87 };
88 
89 }
