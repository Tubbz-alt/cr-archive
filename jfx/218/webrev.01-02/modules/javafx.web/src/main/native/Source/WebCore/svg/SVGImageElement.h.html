22 #pragma once
23 
24 #include "SVGGraphicsElement.h"
25 #include "SVGImageLoader.h"
26 #include "SVGURIReference.h"
27 
28 namespace WebCore {
29 
30 class SVGImageElement final : public SVGGraphicsElement, public SVGURIReference {
31     WTF_MAKE_ISO_ALLOCATED(SVGImageElement);
32 public:
33     static Ref<SVGImageElement> create(const QualifiedName&, Document&);
34 
35     bool hasSingleSecurityOrigin() const;
36     const AtomString& imageSourceURL() const final;
37 
38     const SVGLengthValue& x() const { return m_x->currentValue(); }
39     const SVGLengthValue& y() const { return m_y->currentValue(); }
40     const SVGLengthValue& width() const { return m_width->currentValue(); }
41     const SVGLengthValue& height() const { return m_height->currentValue(); }
42     const SVGPreserveAspectRatioValue& preserveAspectRatio() const { return m_preserveAspectRatio->currentValue(); }
43 
44     SVGAnimatedLength& xAnimated() { return m_x; }
45     SVGAnimatedLength& yAnimated() { return m_y; }
46     SVGAnimatedLength& widthAnimated() { return m_width; }
47     SVGAnimatedLength& heightAnimated() { return m_height; }
48     SVGAnimatedPreserveAspectRatio& preserveAspectRatioAnimated() { return m_preserveAspectRatio; }
49 
50 private:
51     SVGImageElement(const QualifiedName&, Document&);
52 
53     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGImageElement, SVGGraphicsElement, SVGURIReference>;
54     const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
55 
56     void parseAttribute(const QualifiedName&, const AtomString&) final;
57     void svgAttributeChanged(const QualifiedName&) final;
58 
59     void didAttachRenderers() final;
60     InsertedIntoAncestorResult insertedIntoAncestor(InsertionType, ContainerNode&) final;
61     RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) final;
62 
63     void addSubresourceAttributeURLs(ListHashSet<URL>&) const final;
64     bool haveLoadedRequiredResources() final;
65 
66     bool isValid() const final { return SVGTests::isValid(); }
67     bool selfHasRelativeLengths() const final { return true; }
68     void didMoveToNewDocument(Document& oldDocument, Document& newDocument) final;
69 
70     PropertyRegistry m_propertyRegistry { *this };
71     Ref<SVGAnimatedLength> m_x { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
72     Ref<SVGAnimatedLength> m_y { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
73     Ref<SVGAnimatedLength> m_width { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
74     Ref<SVGAnimatedLength> m_height { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
75     Ref<SVGAnimatedPreserveAspectRatio> m_preserveAspectRatio { SVGAnimatedPreserveAspectRatio::create(this) };
76     SVGImageLoader m_imageLoader;
77 };
78 
79 } // namespace WebCore
