 37 StyleRuleKeyframe::StyleRuleKeyframe(Ref<StyleProperties>&& properties)
 38     : StyleRuleBase(StyleRuleType::Keyframe)
 39     , m_properties(WTFMove(properties))
 40 {
 41 }
 42 
 43 StyleRuleKeyframe::StyleRuleKeyframe(std::unique_ptr<Vector<double>> keys, Ref<StyleProperties>&& properties)
 44     : StyleRuleBase(StyleRuleType::Keyframe)
 45     , m_properties(WTFMove(properties))
 46     , m_keys(*keys)
 47 {
 48 }
 49 
 50 StyleRuleKeyframe::~StyleRuleKeyframe() = default;
 51 
 52 MutableStyleProperties& StyleRuleKeyframe::mutableProperties()
 53 {
 54     if (!is<MutableStyleProperties>(m_properties.get()))
 55         m_properties = m_properties->mutableCopy();
 56     return downcast<MutableStyleProperties>(m_properties.get());
 57 }
 58 
 59 String StyleRuleKeyframe::keyText() const
 60 {
 61     StringBuilder keyText;
 62     for (size_t i = 0; i < m_keys.size(); ++i) {
 63         if (i)
 64             keyText.append(',');
 65         keyText.append(m_keys[i] * 100, '%');
 66     }
 67     return keyText.toString();
 68 }
 69 
 70 bool StyleRuleKeyframe::setKeyText(const String& keyText)
 71 {
 72     ASSERT(!keyText.isNull());
 73     auto keys = CSSParser::parseKeyframeKeyList(keyText);
 74     if (!keys || keys->isEmpty())
 75         return false;
 76     m_keys = *keys;
 77     return true;
 78 }
 79 
 80 String StyleRuleKeyframe::cssText() const
 81 {
 82     StringBuilder result;
 83     result.append(keyText());
 84     result.appendLiteral(" { ");
 85     String decls = m_properties->asText();
 86     result.append(decls);
 87     if (!decls.isEmpty())
 88         result.append(' ');
 89     result.append('}');
 90     return result.toString();
 91 }
 92 
 93 CSSKeyframeRule::CSSKeyframeRule(StyleRuleKeyframe& keyframe, CSSKeyframesRule* parent)
 94     : CSSRule(0)
 95     , m_keyframe(keyframe)
 96 {
 97     setParentRule(parent);
 98 }
 99 
100 CSSKeyframeRule::~CSSKeyframeRule()
101 {
102     if (m_propertiesCSSOMWrapper)
103         m_propertiesCSSOMWrapper->clearParentRule();
104 }
105 
106 CSSStyleDeclaration& CSSKeyframeRule::style()
107 {
108     if (!m_propertiesCSSOMWrapper)
109         m_propertiesCSSOMWrapper = StyleRuleCSSStyleDeclaration::create(m_keyframe->mutableProperties(), *this);
110     return *m_propertiesCSSOMWrapper;
111 }
112 
113 void CSSKeyframeRule::reattach(StyleRuleBase&)
114 {
115     // No need to reattach, the underlying data is shareable on mutation.
116     ASSERT_NOT_REACHED();
117 }
