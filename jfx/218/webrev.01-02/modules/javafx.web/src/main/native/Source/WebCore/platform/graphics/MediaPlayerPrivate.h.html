197     virtual void setShouldMaintainAspectRatio(bool) { }
198 
199     virtual bool hasSingleSecurityOrigin() const { return false; }
200     virtual bool didPassCORSAccessCheck() const { return false; }

225     virtual void setPrivateBrowsingMode(bool) { }

258     virtual void endSimulatedHDCPError() { }

260     virtual String languageOfPrimaryAudioTrack() const { return emptyString(); }
261 
262     virtual size_t extraMemoryCost() const
263     {
264         MediaTime duration = this->durationMediaTime();
265         if (!duration)
266             return 0;
267 
268         unsigned long long extra = totalBytes() * buffered()->totalDuration().toDouble() / duration.toDouble();
269         return static_cast<unsigned>(extra);
270     }
