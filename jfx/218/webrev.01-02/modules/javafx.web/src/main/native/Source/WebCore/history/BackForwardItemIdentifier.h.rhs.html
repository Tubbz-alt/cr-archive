 33 namespace WebCore {
 34 
 35 struct BackForwardItemIdentifier {
 36     ProcessIdentifier processIdentifier;
 37     enum ItemIdentifierType { };
 38     ObjectIdentifier<ItemIdentifierType> itemIdentifier;
 39 
 40     unsigned hash() const;
 41     explicit operator bool() const { return processIdentifier && itemIdentifier; }
 42 
 43     template<class Encoder> void encode(Encoder&) const;
 44     template<class Decoder> static Optional<BackForwardItemIdentifier> decode(Decoder&);
 45 
 46     String string() const { return makeString(processIdentifier.toUInt64(), '-', itemIdentifier.toUInt64()); }
 47 
 48     WEBCORE_EXPORT bool isValid() const;
 49 
 50 #if !LOG_DISABLED
 51     const char* logString() const;
 52 #endif
 53 };
 54 
 55 #if !LOG_DISABLED
 56 
 57 inline const char* BackForwardItemIdentifier::logString() const
 58 {
 59     return debugString(makeString(processIdentifier.toUInt64(), '-', itemIdentifier.toUInt64()));
 60 }
 61 
 62 #endif
 63 
 64 inline bool operator==(const BackForwardItemIdentifier& a, const BackForwardItemIdentifier& b)
 65 {
 66     return a.processIdentifier == b.processIdentifier &&  a.itemIdentifier == b.itemIdentifier;
 67 }
 68 
 69 inline bool operator!=(const BackForwardItemIdentifier& a, const BackForwardItemIdentifier& b)
 70 {
 71     return !(a == b);
 72 }
 73 
 74 template<class Encoder>
 75 void BackForwardItemIdentifier::encode(Encoder& encoder) const
 76 {
 77     encoder << processIdentifier << itemIdentifier;
 78 }
 79 
 80 template<class Decoder>
 81 Optional<BackForwardItemIdentifier> BackForwardItemIdentifier::decode(Decoder& decoder)
 82 {
 83     Optional<ProcessIdentifier> processIdentifier;
 84     decoder >> processIdentifier;
 85     if (!processIdentifier)
 86         return WTF::nullopt;
 87 
 88     Optional<ObjectIdentifier<ItemIdentifierType>> itemIdentifier;
 89     decoder >> itemIdentifier;
 90     if (!itemIdentifier)
 91         return WTF::nullopt;
 92 
 93     BackForwardItemIdentifier result = { WTFMove(*processIdentifier), WTFMove(*itemIdentifier) };
 94     if (!result.isValid())
 95         return WTF::nullopt;
 96     return result;
 97 }
 98 
 99 inline unsigned BackForwardItemIdentifier::hash() const
100 {
101     return computeHash(processIdentifier.toUInt64(), itemIdentifier.toUInt64());
102 }
