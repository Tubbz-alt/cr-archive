26 typedef (ArrayBuffer or ArrayBufferView) BinaryData;
27 
28 enum FontFaceLoadStatus {
29     "unloaded",
30     "loading",
31     "loaded",
32     "error"
33 };
34 
35 dictionary FontFaceDescriptors {
36     DOMString style = "normal";
37     DOMString weight = "normal";
38     DOMString stretch = "normal";
39     DOMString unicodeRange = "U+0-10FFFF";
40     DOMString featureSettings = "normal";
41     DOMString display = "auto";
42 };
43 
44 [
45     ActiveDOMObject,
46     ConstructorCallWith=Document,
47     Constructor(DOMString family, (DOMString or BinaryData) source, optional FontFaceDescriptors descriptors)
48 ] interface FontFace {
49     [SetterCallWith=Document] attribute DOMString family;
50     attribute DOMString style;
51     attribute DOMString weight;
52     attribute DOMString stretch;
53     attribute DOMString unicodeRange;
54     attribute DOMString featureSettings;
55     attribute DOMString display;
56 
57     readonly attribute FontFaceLoadStatus status;
58 
59     [ReturnsOwnPromise, PromiseProxy, ImplementedAs=loadForBindings] Promise<FontFace> load();
60     [ImplementedAs=loadedForBindings] readonly attribute Promise<FontFace> loaded;
61 };
