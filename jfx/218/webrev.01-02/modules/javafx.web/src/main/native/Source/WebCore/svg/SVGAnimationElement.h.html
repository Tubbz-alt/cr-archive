 36 // If we have 'currentColor' or 'inherit' as animation value, we need to grab
 37 // the value during the animation since the value can be animated itself.
 38 enum AnimatedPropertyValueType { RegularPropertyValue, CurrentColorValue, InheritValue };
 39 
 40 class SVGAnimationElement : public SVGSMILElement, public SVGTests {
 41     WTF_MAKE_ISO_ALLOCATED(SVGAnimationElement);
 42 public:
 43     float getStartTime() const;
 44     float getCurrentTime() const;
 45     float getSimpleDuration() const;
 46 
 47     void beginElement();
 48     void beginElementAt(float offset);
 49     void endElement();
 50     void endElementAt(float offset);
 51 
 52     static bool isTargetAttributeCSSProperty(SVGElement*, const QualifiedName&);
 53 
 54     bool isAdditive() const override;
 55     bool isAccumulated() const;
 56     AnimationMode animationMode() const { return m_animationMode; }
 57     CalcMode calcMode() const { return m_calcMode; }
 58 
 59     AnimatedPropertyValueType fromPropertyValueType() const { return m_fromPropertyValueType; }
 60     AnimatedPropertyValueType toPropertyValueType() const { return m_toPropertyValueType; }
 61 
 62     void animateAdditiveNumber(float percentage, unsigned repeatCount, float fromNumber, float toNumber, float toAtEndOfDurationNumber, float& animatedNumber)
 63     {
 64         float number;
 65         if (calcMode() == CalcMode::Discrete)
 66             number = percentage < 0.5 ? fromNumber : toNumber;
 67         else
 68             number = (toNumber - fromNumber) * percentage + fromNumber;
 69 
 70         if (isAccumulated() && repeatCount)
 71             number += toAtEndOfDurationNumber * repeatCount;
 72 
 73         if (isAdditive() && animationMode() != AnimationMode::To)
 74             animatedNumber += number;
 75         else
 76             animatedNumber = number;
 77     }
 78 
 79     enum class AttributeType : uint8_t { CSS, XML, Auto };
 80     AttributeType attributeType() const { return m_attributeType; }
 81 
 82     void computeCSSPropertyValue(SVGElement*, CSSPropertyID, String& value);
 83     virtual void determinePropertyValueTypes(const String& from, const String& to);
 84 
 85 protected:
 86     SVGAnimationElement(const QualifiedName&, Document&);
 87 
 88     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGAnimationElement, SVGElement, SVGTests>;
 89     const SVGPropertyRegistry& propertyRegistry() const override { return m_propertyRegistry; }
 90 
 91     virtual void resetAnimation();

 93     static bool isSupportedAttribute(const QualifiedName&);
 94     void parseAttribute(const QualifiedName&, const AtomString&) override;
 95     void svgAttributeChanged(const QualifiedName&) override;

109 
110     void setAttributeName(const QualifiedName&) override { }
111 
112     virtual void updateAnimationMode();
113     void setAnimationMode(AnimationMode animationMode) { m_animationMode = animationMode; }
114     void setCalcMode(CalcMode calcMode) { m_calcMode = calcMode; }
115 
116 private:

118     void setAttributeType(const AtomString&);

120     virtual bool calculateToAtEndOfDurationValue(const String& toAtEndOfDurationString) = 0;
121     virtual bool calculateFromAndToValues(const String& fromString, const String& toString) = 0;
122     virtual bool calculateFromAndByValues(const String& fromString, const String& byString) = 0;
123     virtual void calculateAnimatedValue(float percent, unsigned repeatCount) = 0;
124     virtual Optional<float> calculateDistance(const String& /*fromString*/, const String& /*toString*/) = 0;

126     void currentValuesForValuesAnimation(float percent, float& effectivePercent, String& from, String& to);
127     void calculateKeyTimesForCalcModePaced();
128     float calculatePercentFromKeyPoints(float percent) const;

147     PropertyRegistry m_propertyRegistry { *this };
