23 #include "ErrorPrototype.h"

25 namespace JSC {
26 
27 class NativeErrorPrototype final : public ErrorPrototype {
28 private:
29     NativeErrorPrototype(VM&, Structure*);
30 
31 public:
32     typedef ErrorPrototype Base;
33 
34     static NativeErrorPrototype* create(VM& vm, Structure* structure, const String& name)
35     {
36         NativeErrorPrototype* prototype = new (NotNull, allocateCell<NativeErrorPrototype>(vm.heap)) NativeErrorPrototype(vm, structure);
37         prototype->finishCreation(vm, name);
38         return prototype;
39     }
40 };

42 } // namespace JSC
