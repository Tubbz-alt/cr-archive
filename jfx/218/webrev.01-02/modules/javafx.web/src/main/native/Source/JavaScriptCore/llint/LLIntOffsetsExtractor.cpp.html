 92 const int64_t* LLIntOffsetsExtractor::dummy()
 93 {
 94 // This is a file generated by offlineasm/generate_offset_extractor.rb, and contains code
 95 // to create a table of offsets, sizes, and a header identifying what combination of
 96 // Platform.h macros we have set. We include it inside of a method on LLIntOffsetsExtractor
 97 // because the fields whose offsets we're extracting are mostly private. So we make their
 98 // classes friends with LLIntOffsetsExtractor, and include the header here, to get the C++
 99 // compiler to kindly step aside and yield to our best intentions.
100 #include "LLIntDesiredOffsets.h"
101     return offsetExtractorTable;
102 }
