35 void JSGlobalLexicalEnvironment::destroy(JSCell* cell)
36 {
37     static_cast<JSGlobalLexicalEnvironment*>(cell)->JSGlobalLexicalEnvironment::~JSGlobalLexicalEnvironment();
38 }
39 
40 bool JSGlobalLexicalEnvironment::getOwnPropertySlot(JSObject* object, JSGlobalObject*, PropertyName propertyName, PropertySlot& slot)
41 {
42     JSGlobalLexicalEnvironment* thisObject = jsCast<JSGlobalLexicalEnvironment*>(object);
43     return symbolTableGet(thisObject, propertyName, slot);
44 }
45 
46 bool JSGlobalLexicalEnvironment::put(JSCell* cell, JSGlobalObject* globalObject, PropertyName propertyName, JSValue value, PutPropertySlot& slot)
47 {
48     JSGlobalLexicalEnvironment* thisObject = jsCast<JSGlobalLexicalEnvironment*>(cell);
49     ASSERT(!Heap::heap(value) || Heap::heap(value) == Heap::heap(thisObject));
50     bool alwaysThrowWhenAssigningToConstProperty = true;
51     bool ignoreConstAssignmentError = slot.isInitialization();
52     bool putResult = false;
53     symbolTablePutTouchWatchpointSet(thisObject, globalObject, propertyName, value, alwaysThrowWhenAssigningToConstProperty, ignoreConstAssignmentError, putResult);
54     return putResult;
55 }
56 
57 bool JSGlobalLexicalEnvironment::isConstVariable(UniquedStringImpl* impl)
58 {
59     SymbolTableEntry entry = symbolTable()->get(impl);
60     ASSERT(!entry.isNull());
61     return entry.isReadOnly();
62 }
