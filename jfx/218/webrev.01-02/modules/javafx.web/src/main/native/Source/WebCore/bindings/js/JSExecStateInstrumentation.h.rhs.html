33 namespace WebCore {
34 
35 template<typename Type, Type jsType, class DataType>
36 inline void JSExecState::instrumentFunctionInternal(ScriptExecutionContext* context, Type callType, const DataType& callData)
37 {
38     if (!InspectorInstrumentation::timelineAgentTracking(context))
39         return;
40 
41     String resourceName;
42     int lineNumber = 1;
43     int columnNumber = 1;
44     if (callType == jsType) {
45         resourceName = callData.js.functionExecutable->sourceURL();
46         lineNumber = callData.js.functionExecutable->firstLine();
47         columnNumber = callData.js.functionExecutable->startColumn();
48     } else
49         resourceName = "undefined";
50     InspectorInstrumentation::willCallFunction(context, resourceName, lineNumber, columnNumber);
51 }
52 
53 inline void JSExecState::instrumentFunctionCall(ScriptExecutionContext* context, JSC::CallType type, const JSC::CallData& data)
54 {
55     instrumentFunctionInternal<JSC::CallType, JSC::CallType::JS, JSC::CallData>(context, type, data);
56 }
57 
58 inline void JSExecState::instrumentFunctionConstruct(ScriptExecutionContext* context, JSC::ConstructType type, const JSC::ConstructData& data)
59 {
60     instrumentFunctionInternal<JSC::ConstructType, JSC::ConstructType::JS, JSC::ConstructData>(context, type, data);
61 }
