44 Ref<WebGPUComputePipeline> WebGPUComputePipeline::create(WebGPUDevice& device, RefPtr<GPUComputePipeline>&& pipeline, GPUErrorScopes& errorScopes, WebGPUPipeline::ShaderData&& computeShader)
45 {
46     return adoptRef(*new WebGPUComputePipeline(device, WTFMove(pipeline), errorScopes, WTFMove(computeShader)));
47 }
48 
49 WebGPUComputePipeline::WebGPUComputePipeline(WebGPUDevice& device, RefPtr<GPUComputePipeline>&& pipeline, GPUErrorScopes& errorScopes, WebGPUPipeline::ShaderData&& computeShader)
50     : WebGPUPipeline(device, errorScopes)
51     , m_computePipeline(WTFMove(pipeline))
52     , m_computeShader(WTFMove(computeShader))
53 {
54 }
55 
56 WebGPUComputePipeline::~WebGPUComputePipeline() = default;
57 
58 bool WebGPUComputePipeline::cloneShaderModules(const WebGPUDevice& device)
59 {
60     if (m_computeShader.module) {
61         const auto& computeSource = m_computeShader.module->source();
62         m_computeShader.module = WebGPUShaderModule::create(GPUShaderModule::tryCreate(device.device(), { computeSource }), computeSource);
63         return true;
64     }
65     return false;
66 }
67 
68 bool WebGPUComputePipeline::recompile(const WebGPUDevice& device)
69 {
70     if (m_computePipeline && m_computeShader.module) {
71         if (auto* gpuComputeShaderModule = m_computeShader.module->module()) {
72             GPUProgrammableStageDescriptor computeStage(makeRef(*gpuComputeShaderModule), { m_computeShader.entryPoint });
73             return m_computePipeline->recompile(device.device(), WTFMove(computeStage));
74         }
75     }
76     return false;
77 }
