 41 class RenderBlockFlow;
 42

 44 
 45 class RunResolver;
 46 
 47 bool canUseFor(const RenderBlockFlow&);
 48 AvoidanceReasonFlags canUseForWithReason(const RenderBlockFlow&, IncludeReasons);
 49 
 50 struct Run {
 51 #if COMPILER(MSVC)
 52     Run() { }
 53 #endif
 54     Run(unsigned start, unsigned end, float logicalLeft, float logicalRight, bool isEndOfLine, bool hasHyphen, bool isLineBreak)
 55         : end(end)
 56         , start(start)
 57         , isEndOfLine(isEndOfLine)
 58         , hasHyphen(hasHyphen)
 59         , isLineBreak(isLineBreak)
 60         , logicalLeft(logicalLeft)
 61         , logicalRight(logicalRight)
 62     { }
 63 
 64     unsigned end;
 65     unsigned start : 29;
 66     unsigned isEndOfLine : 1;
 67     unsigned hasHyphen : 1;
 68     unsigned isLineBreak : 1;
 69     float logicalLeft;
 70     float logicalRight;
 71     // TODO: Move these optional items out of SimpleLineLayout::Run to a supplementary structure.
 72     float expansion { 0 };
 73     ExpansionBehavior expansionBehavior { ForbidLeadingExpansion | ForbidTrailingExpansion };
 74 };
 75 
 76 struct SimpleLineStrut {
 77     unsigned lineBreak;
 78     float offset;
 79 };
 80 
 81 class Layout : public RefCounted<Layout> {
 82     WTF_MAKE_FAST_ALLOCATED;
 83 public:
 84     using RunVector = Vector<Run, 10>;
 85     using SimpleLineStruts = Vector<SimpleLineStrut, 4>;
 86     static Ref<Layout> create(const RunVector&, unsigned lineCount, const RenderBlockFlow&);
 87 
 88     ~Layout();
 89 
 90     unsigned lineCount() const { return m_lineCount; }
 91 
 92     unsigned runCount() const { return m_runCount; }
 93     const Run& runAt(unsigned i) const { return m_runs[i]; }
 94 
 95     void setIsPaginated() { m_isPaginated = true; }
 96     bool isPaginated() const { return m_isPaginated; }
 97     bool hasLineStruts() const { return !m_lineStruts.isEmpty(); }
 98     void setLineStruts(SimpleLineStruts&& lineStruts) { m_lineStruts = lineStruts; }
 99     const SimpleLineStruts& struts() const { return m_lineStruts; }
100     const RunResolver& runResolver() const;
101 
102 private:
103     Layout(const RunVector&, unsigned lineCount, const RenderBlockFlow&);
104 
105     unsigned m_lineCount;
106     unsigned m_runCount;
107     bool m_isPaginated { false };
108     SimpleLineStruts m_lineStruts;
109     const RenderBlockFlow& m_blockFlowRenderer;
110     mutable std::unique_ptr<RunResolver> m_runResolver;
111     Run m_runs[0];
112 };
113 
114 Ref<Layout> create(RenderBlockFlow&);
115 
116 }
117 }
