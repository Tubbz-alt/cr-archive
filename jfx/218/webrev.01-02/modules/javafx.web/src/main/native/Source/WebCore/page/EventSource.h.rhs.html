 32 #pragma once
 33 
 34 #include "ActiveDOMObject.h"
 35 #include "EventTarget.h"
 36 #include "ExceptionOr.h"
 37 #include "SuspendableTimer.h"
 38 #include "ThreadableLoaderClient.h"
 39 #include "Timer.h"
 40 #include <wtf/URL.h>
 41 #include <wtf/Vector.h>
 42 
 43 namespace WebCore {
 44 
 45 class MessageEvent;
 46 class TextResourceDecoder;
 47 class ThreadableLoader;
 48 
 49 class EventSource final : public RefCounted<EventSource>, public EventTargetWithInlineData, private ThreadableLoaderClient, public ActiveDOMObject {
 50     WTF_MAKE_ISO_ALLOCATED(EventSource);
 51 public:
 52     struct Init {
 53         bool withCredentials;
 54     };
 55     static ExceptionOr<Ref<EventSource>> create(ScriptExecutionContext&, const String& url, const Init&);
 56     virtual ~EventSource();
 57 
 58     const String& url() const;
 59     bool withCredentials() const;
 60 
 61     using State = short;
 62     static const State CONNECTING = 0;
 63     static const State OPEN = 1;
 64     static const State CLOSED = 2;
 65 
 66     State readyState() const;
 67 
 68     void close();
 69 
 70     using RefCounted::ref;
 71     using RefCounted::deref;
 72 
 73 private:
 74     EventSource(ScriptExecutionContext&, const URL&, const Init&);
 75 
 76     EventTargetInterface eventTargetInterface() const final { return EventSourceEventTargetInterfaceType; }
 77     ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
 78 
 79     void refEventTarget() final { ref(); }
 80     void derefEventTarget() final { deref(); }
 81 
 82     void dispatchErrorEvent();
 83     void doExplicitLoadCancellation();
 84 
 85     // ThreadableLoaderClient
 86     void didReceiveResponse(unsigned long, const ResourceResponse&) final;
 87     void didReceiveData(const char*, int) final;
 88     void didFinishLoading(unsigned long) final;
 89     void didFail(const ResourceError&) final;
 90 
 91     // ActiveDOMObject
 92     void stop() final;
 93     const char* activeDOMObjectName() const final;
 94     void suspend(ReasonForSuspension) final;
 95     void resume() final;
 96 
 97     void connect();
 98     void networkRequestEnded();
 99     void scheduleInitialConnect();
100     void scheduleReconnect();
101     void abortConnectionAttempt();
102     void parseEventStream();
103     void parseEventStreamLine(unsigned position, Optional<unsigned> fieldLength, unsigned lineLength);
104     void dispatchMessageEvent();
105 
106     bool responseIsValid(const ResourceResponse&) const;
107 
108     static const uint64_t defaultReconnectDelay;
109 
110     URL m_url;
111     bool m_withCredentials;
112     State m_state { CONNECTING };
113 
114     Ref<TextResourceDecoder> m_decoder;
115     RefPtr<ThreadableLoader> m_loader;
116     SuspendableTimer m_connectTimer;
117     Vector<UChar> m_receiveBuffer;
118     bool m_discardTrailingNewline { false };
119     bool m_requestInFlight { false };
120     bool m_isSuspendedForBackForwardCache { false };
121     bool m_isDoingExplicitCancellation { false };
122     bool m_shouldReconnectOnResume { false };
123 
124     AtomString m_eventName;
125     Vector<UChar> m_data;
126     String m_currentlyParsedEventId;
127     String m_lastEventId;
128     uint64_t m_reconnectDelay { defaultReconnectDelay };
129     String m_eventStreamOrigin;
130 };
131 
132 inline const String& EventSource::url() const
133 {
134     return m_url.string();
135 }
136 
137 inline bool EventSource::withCredentials() const
138 {
139     return m_withCredentials;
140 }
141 
142 inline EventSource::State EventSource::readyState() const
143 {
144     return m_state;
145 }
146 
147 } // namespace WebCore
