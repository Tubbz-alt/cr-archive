 30 #include <wtf/text/WTFString.h>

 32 namespace WebCore {
 33 
 34 class CaptureDevice;
 35 class CaptureDeviceManager;
 36 class RealtimeMediaSource;
 37 
 38 struct CaptureSourceOrError;
 39 struct MediaConstraints;
 40 
 41 class SingleSourceFactory {
 42 public:
 43     WEBCORE_EXPORT void setActiveSource(RealtimeMediaSource&);
 44     WEBCORE_EXPORT void unsetActiveSource(RealtimeMediaSource&);
 45 
 46 protected:
 47     RealtimeMediaSource* activeSource() { return m_activeSource; }
 48 
 49 private:
 50     RealtimeMediaSource* m_activeSource { nullptr };
 51 };
 52 
 53 class AudioCaptureFactory
 54 #if PLATFORM(IOS_FAMILY)
 55     : public SingleSourceFactory
 56 #endif
 57 {
 58 public:
 59     virtual ~AudioCaptureFactory() = default;
 60     virtual CaptureSourceOrError createAudioCaptureSource(const CaptureDevice&, String&&, const MediaConstraints*) = 0;
 61     virtual CaptureDeviceManager& audioCaptureDeviceManager() = 0;
 62     virtual void setAudioCapturePageState(bool interrupted, bool pageMuted)
 63     {
 64         UNUSED_PARAM(interrupted);
 65         UNUSED_PARAM(pageMuted);
 66     }
 67 
 68 protected:
 69     AudioCaptureFactory() = default;
 70 };
 71 
 72 class VideoCaptureFactory
 73 #if PLATFORM(IOS_FAMILY)
 74     : public SingleSourceFactory
 75 #endif
 76 {
 77 public:
 78     virtual ~VideoCaptureFactory() = default;
 79     virtual CaptureSourceOrError createVideoCaptureSource(const CaptureDevice&, String&&, const MediaConstraints*) = 0;
 80     virtual CaptureDeviceManager& videoCaptureDeviceManager() = 0;
 81     virtual void setVideoCapturePageState(bool interrupted, bool pageMuted)
 82     {
 83         UNUSED_PARAM(interrupted);
 84         UNUSED_PARAM(pageMuted);
 85     }
 86 
 87 protected:
 88     VideoCaptureFactory() = default;
 89 };
 90 
 91 class DisplayCaptureFactory {
 92 public:
 93     virtual ~DisplayCaptureFactory() = default;
 94     virtual CaptureSourceOrError createDisplayCaptureSource(const CaptureDevice&, const MediaConstraints*) = 0;
 95     virtual CaptureDeviceManager& displayCaptureDeviceManager() = 0;
 96 
 97 protected:
 98     DisplayCaptureFactory() = default;
 99 };
