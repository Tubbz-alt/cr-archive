 48 static EncodedJSValue JSC_HOST_CALL constructJSWebAssemblyGlobal(JSGlobalObject* globalObject, CallFrame* callFrame)
 49 {
 50     VM& vm = globalObject->vm();
 51     auto throwScope = DECLARE_THROW_SCOPE(vm);
 52 
 53     JSObject* globalDescriptor;
 54     {
 55         JSValue argument = callFrame->argument(0);
 56         if (!argument.isObject())
 57             return JSValue::encode(throwException(globalObject, throwScope, createTypeError(globalObject, "WebAssembly.Global expects its first argument to be an object"_s)));
 58         globalDescriptor = jsCast<JSObject*>(argument);
 59     }
 60 
 61     Wasm::GlobalInformation::Mutability mutability;
 62     {
 63         Identifier mutableIdent = Identifier::fromString(vm, "mutable");
 64         JSValue mutableValue = globalDescriptor->get(globalObject, mutableIdent);
 65         RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
 66         bool mutableBoolean = mutableValue.toBoolean(globalObject);
 67         RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
 68         if (mutableBoolean)
 69             mutability = Wasm::GlobalInformation::Mutable;
 70         else
 71             mutability = Wasm::GlobalInformation::Immutable;
 72     }
 73 
 74     Wasm::Type type;
 75     {
 76         Identifier valueIdent = Identifier::fromString(vm, "value");
 77         JSValue valueValue = globalDescriptor->get(globalObject, valueIdent);
 78         RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
 79         String valueString = valueValue.toWTFString(globalObject);
 80         RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
 81         if (valueString == "i32")
 82             type = Wasm::Type::I32;
 83         else if (valueString == "i64")
 84             type = Wasm::Type::I64;
 85         else if (valueString == "f32")
 86             type = Wasm::Type::F32;
 87         else if (valueString == "f64")
 88             type = Wasm::Type::F64;
 89         else
 90             return JSValue::encode(throwException(globalObject, throwScope, createTypeError(globalObject, "WebAssembly.Global expects its 'value' field to be the string 'i32', 'i64', 'f32', or 'f64'"_s)));
 91     }
 92 
 93     uint64_t initialValue = 0;
 94     JSValue argument = callFrame->argument(1);
 95     if (!argument.isUndefined()) {
 96         switch (type) {
 97         case Wasm::Type::I32: {
 98             int32_t value = argument.toInt32(globalObject);
 99             RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
100             initialValue = static_cast<uint64_t>(bitwise_cast<uint32_t>(value));
101             break;
102         }
103         case Wasm::Type::I64: {
104             return JSValue::encode(throwException(globalObject, throwScope, createTypeError(globalObject, "WebAssembly.Global does not accept i64 initial value"_s)));
105         }
106         case Wasm::Type::F32: {
107             float value = argument.toFloat(globalObject);
108             RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
109             initialValue = static_cast<uint64_t>(bitwise_cast<uint32_t>(value));
110             break;
111         }
112         case Wasm::Type::F64: {
113             double value = argument.toNumber(globalObject);
114             RETURN_IF_EXCEPTION(throwScope, encodedJSValue());
115             initialValue = bitwise_cast<uint64_t>(value);
116             break;
117         }
118         default:
119             RELEASE_ASSERT_NOT_REACHED();
120         }
121     }
122 
123     Ref<Wasm::Global> wasmGlobal = Wasm::Global::create(type, mutability, initialValue);
124     RELEASE_AND_RETURN(throwScope, JSValue::encode(JSWebAssemblyGlobal::tryCreate(globalObject, vm, globalObject->webAssemblyGlobalStructure(), WTFMove(wasmGlobal))));
125 }

127 static EncodedJSValue JSC_HOST_CALL callJSWebAssemblyGlobal(JSGlobalObject* globalObject, CallFrame*)
128 {
129     VM& vm = globalObject->vm();
130     auto scope = DECLARE_THROW_SCOPE(vm);
131     return JSValue::encode(throwConstructorCannotBeCalledAsFunctionTypeError(globalObject, scope, "WebAssembly.Global"));
132 }

134 WebAssemblyGlobalConstructor* WebAssemblyGlobalConstructor::create(VM& vm, Structure* structure, WebAssemblyGlobalPrototype* thisPrototype)
135 {
136     auto* constructor = new (NotNull, allocateCell<WebAssemblyGlobalConstructor>(vm.heap)) WebAssemblyGlobalConstructor(vm, structure);
137     constructor->finishCreation(vm, thisPrototype);
138     return constructor;
139 }

141 Structure* WebAssemblyGlobalConstructor::createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
142 {
143     return Structure::create(vm, globalObject, prototype, TypeInfo(InternalFunctionType, StructureFlags), info());
144 }

146 void WebAssemblyGlobalConstructor::finishCreation(VM& vm, WebAssemblyGlobalPrototype* prototype)
147 {
148     Base::finishCreation(vm, "Global"_s, NameAdditionMode::WithoutStructureTransition);
149     putDirectWithoutTransition(vm, vm.propertyNames->prototype, prototype, PropertyAttribute::DontEnum | PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly);
150     putDirectWithoutTransition(vm, vm.propertyNames->length, jsNumber(1), PropertyAttribute::ReadOnly | PropertyAttribute::DontEnum);
151 }
