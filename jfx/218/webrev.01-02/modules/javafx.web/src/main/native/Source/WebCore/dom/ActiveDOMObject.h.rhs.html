 44 enum class ReasonForSuspension {
 45     JavaScriptDebuggerPaused,
 46     WillDeferLoading,
 47     BackForwardCache,
 48     PageWillBeSuspended,
 49 };
 50 
 51 class WEBCORE_EXPORT ActiveDOMObject : public ContextDestructionObserver {
 52 public:
 53     // The suspendIfNeeded must be called exactly once after object construction to update
 54     // the suspended state to match that of the ScriptExecutionContext.
 55     void suspendIfNeeded();
 56     void assertSuspendIfNeededWasCalled() const;
 57 
 58     virtual bool hasPendingActivity() const;
 59 




 60     // However, the suspend function will sometimes be called even if canSuspendForDocumentSuspension() returns false.
 61     // That happens in step-by-step JS debugging for example - in this case it would be incorrect
 62     // to stop the object. Exact semantics of suspend is up to the object in cases like that.
 63 
 64     virtual const char* activeDOMObjectName() const = 0;
 65 
 66     // These functions must not have a side effect of creating or destroying
 67     // any ActiveDOMObject. That means they must not result in calls to arbitrary JavaScript.

 68     virtual void suspend(ReasonForSuspension);
 69     virtual void resume();
 70 
 71     // This function must not have a side effect of creating an ActiveDOMObject.
 72     // That means it must not result in calls to arbitrary JavaScript.
 73     // It can, however, have a side effect of deleting an ActiveDOMObject.
 74     virtual void stop();
 75 
 76     template<typename T> void setPendingActivity(T& thisObject)
 77     {
 78         ASSERT(&thisObject == this);
 79         thisObject.ref();
 80         ++m_pendingActivityCount;
 81     }
 82 
 83     template<typename T> void unsetPendingActivity(T& thisObject)
 84     {
 85         ASSERT(m_pendingActivityCount > 0);
 86         --m_pendingActivityCount;
 87         thisObject.deref();
 88     }
 89 
 90     template<class T>
 91     class PendingActivity : public RefCounted<PendingActivity<T>> {
 92     public:
 93         explicit PendingActivity(T& thisObject)
 94             : m_thisObject(thisObject)
 95         {
 96             ++(m_thisObject->m_pendingActivityCount);
 97         }
 98 
 99         ~PendingActivity()
100         {
101             ASSERT(m_thisObject->m_pendingActivityCount > 0);
102             --(m_thisObject->m_pendingActivityCount);
103         }
104 
105     private:
106         Ref<T> m_thisObject;
107     };
108 
109     template<class T> Ref<PendingActivity<T>> makePendingActivity(T& thisObject)
110     {
111         ASSERT(&thisObject == this);
112         return adoptRef(*new PendingActivity<T>(thisObject));
113     }
114 
115     bool isContextStopped() const;
116     bool isAllowedToRunScript() const;
117 
118     template<typename T>
119     static void queueTaskKeepingObjectAlive(T& object, TaskSource source, Function<void ()>&& task)
120     {
121         object.queueTaskInEventLoop(source, [protectedObject = makeRef(object), activity = object.ActiveDOMObject::makePendingActivity(object), task = WTFMove(task)] () {
122             task();
123         });
124     }
125 
126     template<typename EventTargetType, typename EventType>
127     static void queueTaskToDispatchEvent(EventTargetType& target, TaskSource source, Ref<EventType>&& event)
128     {
129         target.queueTaskToDispatchEventInternal(target, source, WTFMove(event));
130     }
131 
132 protected:
133     explicit ActiveDOMObject(ScriptExecutionContext*);
134     explicit ActiveDOMObject(Document*);
135     explicit ActiveDOMObject(Document&);
136     virtual ~ActiveDOMObject();
137 
138 private:
139     enum CheckedScriptExecutionContextType { CheckedScriptExecutionContext };
140     ActiveDOMObject(ScriptExecutionContext*, CheckedScriptExecutionContextType);
141 
142     void queueTaskInEventLoop(TaskSource, Function<void ()>&&);
143     void queueTaskToDispatchEventInternal(EventTarget&, TaskSource, Ref<Event>&&);
144 
145     unsigned m_pendingActivityCount { 0 };
146 #if ASSERT_ENABLED
147     bool m_suspendIfNeededWasCalled { false };
148     Ref<Thread> m_creationThread { Thread::current() };
149 #endif
150 
151     friend class ActiveDOMObjectEventDispatchTask;
152 };
