26 #pragma once
27 
28 #include "JSCast.h"
29 #include "PropertySlot.h"
30 #include "PutPropertySlot.h"
31 #include "Structure.h"
32 
33 namespace JSC {
34 
35 class CustomGetterSetter : public JSCell {
36 public:
37     using Base = JSCell;
38     static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
39 
40     using CustomGetter = PropertySlot::GetValueFunc;
41     using CustomSetter = PutPropertySlot::PutValueFunc;
42 
43     template<typename CellType, SubspaceAccess>
44     static IsoSubspace* subspaceFor(VM& vm)
45     {
46         return &vm.customGetterSetterSpace;
47     }
48 
49     static CustomGetterSetter* create(VM& vm, CustomGetter customGetter, CustomSetter customSetter)
50     {
51         CustomGetterSetter* customGetterSetter = new (NotNull, allocateCell<CustomGetterSetter>(vm.heap)) CustomGetterSetter(vm, vm.customGetterSetterStructure.get(), customGetter, customSetter);
52         customGetterSetter->finishCreation(vm);
53         return customGetterSetter;
54     }
55 
56     CustomGetterSetter::CustomGetter getter() const { return m_getter; }
57     CustomGetterSetter::CustomSetter setter() const { return m_setter; }
58 
59     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
60     {
61         return Structure::create(vm, globalObject, prototype, TypeInfo(CustomGetterSetterType, StructureFlags), info());
62     }
63 
64     DECLARE_EXPORT_INFO;
65 
66 protected:
67     CustomGetterSetter(VM& vm, Structure* structure, CustomGetter getter, CustomSetter setter)
68         : JSCell(vm, structure)
69         , m_getter(getter)
70         , m_setter(setter)
71     {
72     }
73 
74 private:
75     CustomGetter m_getter;
76     CustomSetter m_setter;
77 };
78 
79 JS_EXPORT_PRIVATE bool callCustomSetter(JSGlobalObject*, CustomGetterSetter::CustomSetter, bool isAccessor, JSValue thisValue, JSValue);
80 JS_EXPORT_PRIVATE bool callCustomSetter(JSGlobalObject*, JSValue customGetterSetter, bool isAccessor, JSObject* slotBase, JSValue thisValue, JSValue);
81 
82 } // namespace JSC
