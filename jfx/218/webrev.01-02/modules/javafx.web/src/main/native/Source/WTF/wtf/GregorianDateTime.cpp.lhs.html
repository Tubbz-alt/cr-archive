38 void GregorianDateTime::setToCurrentLocalTime()
39 {
40 #if OS(WINDOWS)
41     SYSTEMTIME systemTime;
42     GetLocalTime(&systemTime);
43     TIME_ZONE_INFORMATION timeZoneInformation;
44     DWORD timeZoneId = GetTimeZoneInformation(&timeZoneInformation);
45 
46     LONG bias = 0;
47     if (timeZoneId != TIME_ZONE_ID_INVALID) {
48         bias = timeZoneInformation.Bias;
49         if (timeZoneId == TIME_ZONE_ID_DAYLIGHT)
50             bias += timeZoneInformation.DaylightBias;
51         else if ((timeZoneId == TIME_ZONE_ID_STANDARD) || (timeZoneId == TIME_ZONE_ID_UNKNOWN))
52             bias += timeZoneInformation.StandardBias;
53         else
54             ASSERT(0);
55     }
56 
57     m_year = systemTime.wYear;
58     m_month = systemTime.wMonth - 1;
59     m_monthDay = systemTime.wDay;
60     m_yearDay = dayInYear(m_year, m_month, m_monthDay);
61     m_weekDay = systemTime.wDayOfWeek;
62     m_hour = systemTime.wHour;
63     m_minute = systemTime.wMinute;
64     m_second = systemTime.wSecond;
65     m_utcOffset = -bias * secondsPerMinute;
66     m_isDST = timeZoneId == TIME_ZONE_ID_DAYLIGHT ? 1 : 0;
67 #else
68     tm localTM;
69     time_t localTime = time(0);
70 #if HAVE(LOCALTIME_R)
71     localtime_r(&localTime, &localTM);
72 #else
73     localtime_s(&localTime, &localTM);
74 #endif
75 
76     m_year = localTM.tm_year + 1900;
77     m_month = localTM.tm_mon;
78     m_monthDay = localTM.tm_mday;
79     m_yearDay = localTM.tm_yday;
80     m_weekDay = localTM.tm_wday;
81     m_hour = localTM.tm_hour;
82     m_minute = localTM.tm_min;
83     m_second = localTM.tm_sec;
84     m_isDST = localTM.tm_isdst;
85 #if HAVE(TM_GMTOFF)
86     m_utcOffset = localTM.tm_gmtoff;
87 #else
88     m_utcOffset = calculateLocalTimeOffset(localTime * msPerSecond).offset / msPerSecond;
89 #endif
90 #endif
91 }
