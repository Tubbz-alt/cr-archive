35 // TemporaryANGLESetting<> is useful for temporarily disabling (or enabling) a particular OpenGL
36 // feature with a particular scope. A TemporaryANGLESetting<> object returns the flag to its original
37 // value upon destruction, making it an alternative to checking, clearing, and resetting each flag
38 // at all of a block's exit points.
39 //
40 // Based on WTF::SetForScope<>
41 
42 class TemporaryANGLESetting {
43     WTF_MAKE_NONCOPYABLE(TemporaryANGLESetting);
44 public:
45     TemporaryANGLESetting(GCGLenum capability, GCGLenum scopedState);
46     ~TemporaryANGLESetting();
47 
48 private:
49     const GCGLenum m_capability;
50     const GCGLenum m_scopedState;
51     GCGLenum m_originalState;
52 };
53 
54 }
55 
56 using WebCore::TemporaryANGLESetting;
