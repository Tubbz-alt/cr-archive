23 #include "config.h"

26 #include <wtf/IsoMallocInlines.h>

28 namespace WebCore {
29 
30 WTF_MAKE_ISO_ALLOCATED_IMPL(MutationEvent);
31 
32 MutationEvent::MutationEvent(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, Node* relatedNode, const String& prevValue, const String& newValue)
33     : Event(type, canBubble, cancelable)
34     , m_relatedNode(relatedNode)
35     , m_prevValue(prevValue)
36     , m_newValue(newValue)
37 {
38 }
39 
40 void MutationEvent::initMutationEvent(const AtomString& type, bool canBubble, bool cancelable, Node* relatedNode, const String& prevValue, const String& newValue, const String& attrName, unsigned short attrChange)
41 {
42     if (isBeingDispatched())
43         return;
44 
45     initEvent(type, canBubble, cancelable);
46 
47     m_relatedNode = relatedNode;
48     m_prevValue = prevValue;
49     m_newValue = newValue;
50     m_attrName = attrName;
51     m_attrChange = attrChange;
52 }
53 
54 EventInterface MutationEvent::eventInterface() const
55 {
56     return MutationEventInterfaceType;
57 }
