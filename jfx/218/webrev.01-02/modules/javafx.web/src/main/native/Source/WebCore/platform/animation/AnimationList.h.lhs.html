25 #pragma once
26 
27 #include "Animation.h"


28 #include <wtf/Vector.h>
29 
30 namespace WebCore {
31 
32 class AnimationList {
33     WTF_MAKE_FAST_ALLOCATED;
34 public:
35     AnimationList() { }
36     AnimationList(const AnimationList&);
37     AnimationList(AnimationList&&) = default;
38 
39     void fillUnsetProperties();
40     bool operator==(const AnimationList&) const;
41     bool operator!=(const AnimationList& other) const
42     {
43         return !(*this == other);
44     }
45 
46     size_t size() const { return m_animations.size(); }
47     bool isEmpty() const { return m_animations.isEmpty(); }
48 
49     void resize(size_t n) { m_animations.resize(n); }
50     void remove(size_t i) { m_animations.remove(i); }
51     void append(Ref<Animation>&& animation) { m_animations.append(WTFMove(animation)); }
52 
53     Animation& animation(size_t i) { return m_animations[i].get(); }
54     const Animation& animation(size_t i) const { return m_animations[i].get(); }
55 
56 private:



57     AnimationList& operator=(const AnimationList&);
58     AnimationList& operator=(AnimationList&&) = default;
59 
60     Vector<Ref<Animation>, 0, CrashOnOverflow, 0> m_animations;
61 };
62 

63 
64 } // namespace WebCore
