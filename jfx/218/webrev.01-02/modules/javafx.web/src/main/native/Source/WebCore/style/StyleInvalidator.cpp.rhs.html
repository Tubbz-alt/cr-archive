 48 static bool shouldDirtyAllStyle(const Vector<RefPtr<StyleRuleBase>>& rules)
 49 {
 50     for (auto& rule : rules) {
 51         if (is<StyleRuleMedia>(*rule)) {
 52             const auto* childRules = downcast<StyleRuleMedia>(*rule).childRulesWithoutDeferredParsing();
 53             if (childRules && shouldDirtyAllStyle(*childRules))
 54                 return true;
 55             continue;
 56         }
 57         // FIXME: At least font faces don't need full recalc in all cases.
 58         if (!is<StyleRule>(*rule))
 59             return true;
 60     }
 61     return false;
 62 }
 63 
 64 static bool shouldDirtyAllStyle(const StyleSheetContents& sheet)
 65 {
 66     for (auto& import : sheet.importRules()) {
 67         if (!import->styleSheet())
 68             continue;
 69         if (shouldDirtyAllStyle(*import->styleSheet()))
 70             return true;
 71     }
 72     if (shouldDirtyAllStyle(sheet.childRules()))
 73         return true;
 74     return false;
 75 }
 76 
 77 static bool shouldDirtyAllStyle(const Vector<StyleSheetContents*>& sheets)
 78 {
 79     for (auto& sheet : sheets) {
 80         if (shouldDirtyAllStyle(*sheet))
 81             return true;
 82     }
 83     return false;
 84 }

 86 Invalidator::Invalidator(const Vector<StyleSheetContents*>& sheets, const MediaQueryEvaluator& mediaQueryEvaluator)
 87     : m_ownedRuleSet(RuleSet::create())
 88     , m_ruleSets({ m_ownedRuleSet })
 89     , m_dirtiesAllStyle(shouldDirtyAllStyle(sheets))
 90 {
 91     if (m_dirtiesAllStyle)
 92         return;
 93 
 94     m_ownedRuleSet->disableAutoShrinkToFit();
 95     for (auto& sheet : sheets)
 96         m_ownedRuleSet->addRulesFromSheet(*sheet, mediaQueryEvaluator);
 97 
 98     m_ruleInformation = collectRuleInformation();
 99 }

110 Invalidator::RuleInformation Invalidator::collectRuleInformation()
111 {
112     RuleInformation information;
113     for (auto& ruleSet : m_ruleSets) {
114         if (!ruleSet->slottedPseudoElementRules().isEmpty())
115             information.hasSlottedPseudoElementRules = true;
116         if (!ruleSet->hostPseudoClassRules().isEmpty())
117             information.hasHostPseudoClassRules = true;
118         if (ruleSet->hasShadowPseudoElementRules())
119             information.hasShadowPseudoElementRules = true;
120         if (!ruleSet->partPseudoElementRules().isEmpty())
121             information.hasPartPseudoElementRules = true;
122     }
123     return information;
124 }

126 Invalidator::CheckDescendants Invalidator::invalidateIfNeeded(Element& element, const SelectorFilter* filter)
127 {
128     invalidateInShadowTreeIfNeeded(element);
129 
130     bool shouldCheckForSlots = m_ruleInformation.hasSlottedPseudoElementRules && !m_didInvalidateHostChildren;
131     if (shouldCheckForSlots && is<HTMLSlotElement>(element)) {
132         auto* containingShadowRoot = element.containingShadowRoot();
133         if (containingShadowRoot && containingShadowRoot->host()) {
134             for (auto& possiblySlotted : childrenOfType<Element>(*containingShadowRoot->host()))
135                 possiblySlotted.invalidateStyleInternal();
136         }
137         // No need to do this again.
138         m_didInvalidateHostChildren = true;
139     }
140 
141     switch (element.styleValidity()) {
142     case Style::Validity::Valid: {
143         for (auto& ruleSet : m_ruleSets) {
144             ElementRuleCollector ruleCollector(element, *ruleSet, filter);
145             ruleCollector.setMode(SelectorChecker::Mode::CollectingRulesIgnoringVirtualPseudoElements);
146 
147             if (ruleCollector.matchesAnyAuthorRules()) {
148                 element.invalidateStyleInternal();
149                 break;
150             }
151         }
152 


153         return CheckDescendants::Yes;
154     }
155     case Style::Validity::ElementInvalid:
156         return CheckDescendants::Yes;
157     case Style::Validity::SubtreeInvalid:
158     case Style::Validity::SubtreeAndRenderersInvalid:
159         if (shouldCheckForSlots)
160             return CheckDescendants::Yes;
161         return CheckDescendants::No;
162     }
163     ASSERT_NOT_REACHED();
164     return CheckDescendants::Yes;
165 }

167 void Invalidator::invalidateStyleForTree(Element& root, SelectorFilter* filter)
168 {
169     if (invalidateIfNeeded(root, filter) == CheckDescendants::No)
170         return;
171     invalidateStyleForDescendants(root, filter);
172 }
173 
174 void Invalidator::invalidateStyleForDescendants(Element& root, SelectorFilter* filter)
175 {
176     Vector<Element*, 20> parentStack;
177     Element* previousElement = &root;
178     auto descendants = descendantsOfType<Element>(root);
179     for (auto it = descendants.begin(), end = descendants.end(); it != end;) {
180         auto& descendant = *it;
181         auto* parent = descendant.parentElement();
182         if (parentStack.isEmpty() || parentStack.last() != parent) {
183             if (parent == previousElement) {
184                 parentStack.append(parent);
185                 if (filter)
186                     filter->pushParentInitializingIfNeeded(*parent);
187             } else {
188                 while (parentStack.last() != parent) {
189                     parentStack.removeLast();
190                     if (filter)
191                         filter->popParent();
192                 }
193             }
194         }
195         previousElement = &descendant;
196 
197         if (invalidateIfNeeded(descendant, filter) == CheckDescendants::Yes)
198             it.traverseNext();
199         else
200             it.traverseNextSkippingChildren();
201     }
202 }

204 void Invalidator::invalidateStyle(Document& document)
205 {
206     ASSERT(!m_dirtiesAllStyle);
207 
208     Element* documentElement = document.documentElement();
209     if (!documentElement)
210         return;
211 
212     SelectorFilter filter;
213     invalidateStyleForTree(*documentElement, &filter);
214 }
215 
216 void Invalidator::invalidateStyle(Scope& scope)
217 {
218     if (m_dirtiesAllStyle) {
219         invalidateAllStyle(scope);
220         return;
221     }
222 
223     if (auto* shadowRoot = scope.shadowRoot()) {
224         invalidateStyle(*shadowRoot);
225         return;
226     }
227 
228     invalidateStyle(scope.document());
229 }

231 void Invalidator::invalidateStyle(ShadowRoot& shadowRoot)
232 {
233     ASSERT(!m_dirtiesAllStyle);
234 
235     if (m_ruleInformation.hasHostPseudoClassRules && shadowRoot.host())
236         shadowRoot.host()->invalidateStyleInternal();
237 
238     for (auto& child : childrenOfType<Element>(shadowRoot)) {
239         SelectorFilter filter;
240         invalidateStyleForTree(child, &filter);
241     }
242 }
243 
244 void Invalidator::invalidateStyle(Element& element)
245 {
246     ASSERT(!m_dirtiesAllStyle);
247 
248     // Don't use SelectorFilter as the rule sets here tend to be small and the filter would have setup cost deep in the tree.
249     invalidateStyleForTree(element, nullptr);
250 }

252 void Invalidator::invalidateStyleWithMatchElement(Element& element, MatchElement matchElement)
253 {
254     switch (matchElement) {
255     case MatchElement::Subject: {
256         invalidateIfNeeded(element, nullptr);
257         break;
258     }
259     case MatchElement::Parent: {
260         auto children = childrenOfType<Element>(element);
261         for (auto& child : children)
262             invalidateIfNeeded(child, nullptr);
263         break;
264     }
265     case MatchElement::Ancestor: {
266         SelectorFilter filter;
267         invalidateStyleForDescendants(element, &filter);
268         break;
269     }
270     case MatchElement::DirectSibling:
271         if (auto* sibling = element.nextElementSibling())
272             invalidateIfNeeded(*sibling, nullptr);
273         break;
274     case MatchElement::IndirectSibling:
275         for (auto* sibling = element.nextElementSibling(); sibling; sibling = sibling->nextElementSibling())
276             invalidateIfNeeded(*sibling, nullptr);
277         break;
278     case MatchElement::AnySibling: {
279         auto parentChildren = childrenOfType<Element>(*element.parentNode());
280         for (auto& parentChild : parentChildren)
281             invalidateIfNeeded(parentChild, nullptr);
282         break;
283     }
284     case MatchElement::ParentSibling:
285         for (auto* sibling = element.nextElementSibling(); sibling; sibling = sibling->nextElementSibling()) {
286             auto siblingChildren = childrenOfType<Element>(*sibling);
287             for (auto& siblingChild : siblingChildren)
288                 invalidateIfNeeded(siblingChild, nullptr);
289         }
290         break;
291     case MatchElement::AncestorSibling: {
292         SelectorFilter filter;
293         for (auto* sibling = element.nextElementSibling(); sibling; sibling = sibling->nextElementSibling()) {
294             filter.popParentsUntil(element.parentElement());
295             invalidateStyleForDescendants(*sibling, &filter);
296         }
297         break;
298     }
299     case MatchElement::Host:
300         invalidateInShadowTreeIfNeeded(element);
301         break;
302     }
303 }

305 void Invalidator::invalidateShadowParts(ShadowRoot& shadowRoot)
306 {
307     if (!RuntimeEnabledFeatures::sharedFeatures().cssShadowPartsEnabled())
308         return;
309 
310     if (shadowRoot.mode() == ShadowRootMode::UserAgent)
311         return;
312 
313     for (auto& descendant : descendantsOfType<Element>(shadowRoot)) {
314         // FIXME: We could only invalidate part names that actually show up in rules.
315         if (!descendant.partNames().isEmpty())
316             descendant.invalidateStyleInternal();
317 
318         auto* nestedShadowRoot = descendant.shadowRoot();
319         if (nestedShadowRoot && !nestedShadowRoot->partMappings().isEmpty())
320             invalidateShadowParts(*nestedShadowRoot);
321     }
322 }

324 void Invalidator::invalidateInShadowTreeIfNeeded(Element& element)
325 {
326     auto* shadowRoot = element.shadowRoot();
327     if (!shadowRoot)
328         return;
329 
330     // FIXME: This could do actual rule matching too.
331     if (m_ruleInformation.hasShadowPseudoElementRules)
332         element.invalidateStyleForSubtreeInternal();
333 
334     // FIXME: More fine-grained invalidation for ::part()
335     if (m_ruleInformation.hasPartPseudoElementRules)
336         invalidateShadowParts(*shadowRoot);
337 }

339 void Invalidator::addToMatchElementRuleSets(Invalidator::MatchElementRuleSets& matchElementRuleSets, const InvalidationRuleSet& invalidationRuleSet)
340 {
341     matchElementRuleSets.ensure(invalidationRuleSet.matchElement, [] {
342         return InvalidationRuleSetVector { };
343     }).iterator->value.append(invalidationRuleSet.ruleSet.copyRef());
344 }

346 void Invalidator::invalidateWithMatchElementRuleSets(Element& element, const MatchElementRuleSets& matchElementRuleSets)
347 {
348     SetForScope<bool> isInvalidating(element.styleResolver().ruleSets().isInvalidatingStyleWithRuleSets(), true);
349 
350     for (auto& matchElementAndRuleSet : matchElementRuleSets) {
351         Invalidator invalidator(matchElementAndRuleSet.value);
352         invalidator.invalidateStyleWithMatchElement(element, matchElementAndRuleSet.key);
353     }
354 }

356 void Invalidator::invalidateAllStyle(Scope& scope)
357 {
358     if (auto* shadowRoot = scope.shadowRoot()) {
359         for (auto& shadowChild : childrenOfType<Element>(*shadowRoot))
360             shadowChild.invalidateStyleForSubtreeInternal();
361         invalidateHostAndSlottedStyleIfNeeded(*shadowRoot);
362         return;
363     }
364 
365     scope.document().scheduleFullStyleRebuild();
366 }
367 
368 void Invalidator::invalidateHostAndSlottedStyleIfNeeded(ShadowRoot& shadowRoot)
369 {
370     auto& host = *shadowRoot.host();
371     auto* resolver = shadowRoot.styleScope().resolverIfExists();
372     if (!resolver)
373         return;
374     auto& authorStyle = resolver->ruleSets().authorStyle();
375 
376     if (!authorStyle.hostPseudoClassRules().isEmpty())
377         host.invalidateStyleInternal();
378 
379     if (!authorStyle.slottedPseudoElementRules().isEmpty()) {
380         for (auto& shadowChild : childrenOfType<Element>(host))
381             shadowChild.invalidateStyleInternal();
382     }
383 }
