 40 static const Container& initialContainingBlock(const Box& layoutBox)
 41 {
 42     auto* containingBlock = layoutBox.containingBlock();
 43     while (containingBlock->containingBlock())
 44         containingBlock = containingBlock->containingBlock();
 45     return *containingBlock;
 46 }
 47 
 48 static bool isQuirkContainer(const Box& layoutBox)
 49 {
 50     return layoutBox.isBodyBox() || layoutBox.isDocumentBox() || layoutBox.isTableCell();
 51 }
 52 
 53 bool BlockFormattingContext::Quirks::needsStretching(const Box& layoutBox) const
 54 {
 55     // In quirks mode, body stretches to html and html to the initial containing block (height: auto only).
 56     if (!layoutState().inQuirksMode())
 57         return false;
 58 
 59     if (!layoutBox.isDocumentBox() && !layoutBox.isBodyBox())
 60         return false;
 61 
 62     return layoutBox.style().logicalHeight().isAuto();
 63 }

 65 ContentHeightAndMargin BlockFormattingContext::Quirks::stretchedInFlowHeight(const Box& layoutBox, ContentHeightAndMargin contentHeightAndMargin)
 66 {
 67     ASSERT(layoutBox.isInFlow());
 68     ASSERT(layoutBox.isDocumentBox() || layoutBox.isBodyBox());
 69 
 70     auto& formattingContext = this->formattingContext();
 71     auto& documentBox = layoutBox.isDocumentBox() ? layoutBox : *layoutBox.parent();
 72     auto& documentBoxGeometry = formattingContext.geometryForBox(documentBox, EscapeReason::BodyStrechesToViewportQuirk);
 73 
 74     auto& initialContainingBlockGeometry = formattingContext.geometryForBox(initialContainingBlock(layoutBox), EscapeReason::BodyStrechesToViewportQuirk);
 75     auto strechedHeight = initialContainingBlockGeometry.contentBoxHeight();
 76     strechedHeight -= documentBoxGeometry.verticalBorder() + documentBoxGeometry.verticalPadding().valueOr(0);
 77 
 78     LayoutUnit totalVerticalMargin;
 79     if (layoutBox.isDocumentBox()) {
 80         // Document box's margins do not collapse.
 81         auto verticalMargin = contentHeightAndMargin.nonCollapsedMargin;
 82         totalVerticalMargin = verticalMargin.before + verticalMargin.after;
 83     } else if (layoutBox.isBodyBox()) {
 84         // Here is the quirky part for body box:
 85         // Stretch the body using the initial containing block's height and shrink it with document box's margin/border/padding.
 86         // This looks extremely odd when html has non-auto height.
 87         auto documentBoxVerticalMargin = formattingContext.geometry().computedVerticalMargin(documentBox, Geometry::horizontalConstraintsForInFlow(initialContainingBlockGeometry));
 88         strechedHeight -= (documentBoxVerticalMargin.before.valueOr(0) + documentBoxVerticalMargin.after.valueOr(0));
 89 
 90         auto& bodyBoxGeometry = formattingContext.geometryForBox(layoutBox);
 91         strechedHeight -= bodyBoxGeometry.verticalBorder() + bodyBoxGeometry.verticalPadding().valueOr(0);
 92 
 93         auto nonCollapsedMargin = contentHeightAndMargin.nonCollapsedMargin;
 94         auto collapsedMargin = formattingContext.marginCollapse().collapsedVerticalValues(layoutBox, nonCollapsedMargin);
 95         totalVerticalMargin = collapsedMargin.before.valueOr(nonCollapsedMargin.before);
 96         totalVerticalMargin += collapsedMargin.isCollapsedThrough ? nonCollapsedMargin.after : collapsedMargin.after.valueOr(nonCollapsedMargin.after);
 97     }
 98 
 99     // Stretch but never overstretch with the margins.
100     if (contentHeightAndMargin.contentHeight + totalVerticalMargin < strechedHeight)
101         contentHeightAndMargin.contentHeight = strechedHeight - totalVerticalMargin;
102 
103     return contentHeightAndMargin;
104 }
105 
106 bool BlockFormattingContext::Quirks::shouldIgnoreCollapsedQuirkMargin(const Box& layoutBox) const
107 {
108     return layoutState().inQuirksMode() && isQuirkContainer(layoutBox);
109 }
