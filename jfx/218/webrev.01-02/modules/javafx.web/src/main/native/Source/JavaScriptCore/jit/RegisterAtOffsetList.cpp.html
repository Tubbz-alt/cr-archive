35 DEFINE_ALLOCATOR_WITH_HEAP_IDENTIFIER(RegisterAtOffsetList);
36 
37 RegisterAtOffsetList::RegisterAtOffsetList() { }
38 
39 RegisterAtOffsetList::RegisterAtOffsetList(RegisterSet registerSet, OffsetBaseType offsetBaseType)
40 {
41     size_t numberOfRegisters = registerSet.numberOfSetRegisters();
42     ptrdiff_t offset = 0;
43 
44     if (offsetBaseType == FramePointerBased)
45         offset = -(static_cast<ptrdiff_t>(numberOfRegisters) * sizeof(CPURegister));
46 
47     m_registers.reserveInitialCapacity(numberOfRegisters);
48     registerSet.forEach([&] (Reg reg) {
49         m_registers.append(RegisterAtOffset(reg, offset));
50         offset += sizeof(CPURegister);
51     });
52 }
53 
54 void RegisterAtOffsetList::dump(PrintStream& out) const
55 {
56     out.print(listDump(m_registers));
57 }
58 
59 RegisterAtOffset* RegisterAtOffsetList::find(Reg reg) const
60 {
61     return tryBinarySearch<RegisterAtOffset, Reg>(m_registers, m_registers.size(), reg, RegisterAtOffset::getReg);
62 }
63 
64 unsigned RegisterAtOffsetList::indexOf(Reg reg) const
65 {
66     if (RegisterAtOffset* pointer = find(reg))
67         return pointer - m_registers.begin();
68     return UINT_MAX;
69 }
70 
71 const RegisterAtOffsetList& RegisterAtOffsetList::llintBaselineCalleeSaveRegisters()
72 {
73     static std::once_flag onceKey;
74     static LazyNeverDestroyed<RegisterAtOffsetList> result;
75     std::call_once(onceKey, [] {
76         result.construct(RegisterSet::llintBaselineCalleeSaveRegisters());
77     });
78     return result.get();
79 }
