37 ANGLEInstancedArrays::ANGLEInstancedArrays(WebGLRenderingContextBase& context)
38     : WebGLExtension(context)
39 {
40 #if USE(ANGLE)
41     context.graphicsContextGL()->getExtensions().ensureEnabled("GL_ANGLE_instanced_arrays");
42 #endif
43 }
44 
45 ANGLEInstancedArrays::~ANGLEInstancedArrays() = default;
46 
47 WebGLExtension::ExtensionName ANGLEInstancedArrays::getName() const
48 {
49     return ANGLEInstancedArraysName;
50 }
51 
52 bool ANGLEInstancedArrays::supported(WebGLRenderingContextBase& context)
53 {
54 #if PLATFORM(COCOA)
55 #if USE(ANGLE)
56     return context.graphicsContextGL()->getExtensions().supports("GL_ANGLE_instanced_arrays");
57 #else
58     UNUSED_PARAM(context);
59     return true;
60 #endif
61 #elif PLATFORM(GTK)
62     return context.graphicsContextGL()->getExtensions().supports("GL_ANGLE_instanced_arrays");
63 #else
64     UNUSED_PARAM(context);
65     return false;
66 #endif
67 }
68 
69 void ANGLEInstancedArrays::drawArraysInstancedANGLE(GCGLenum mode, GCGLint first, GCGLsizei count, GCGLsizei primcount)
70 {
71     if (m_context.isContextLost())
72         return;
73     m_context.drawArraysInstanced(mode, first, count, primcount);
74 }
75 
76 void ANGLEInstancedArrays::drawElementsInstancedANGLE(GCGLenum mode, GCGLsizei count, GCGLenum type, long long offset, GCGLsizei primcount)
77 {
78     if (m_context.isContextLost())
79         return;
80     m_context.drawElementsInstanced(mode, count, type, offset, primcount);
81 }
82 
83 void ANGLEInstancedArrays::vertexAttribDivisorANGLE(GCGLuint index, GCGLuint divisor)
84 {
85     if (m_context.isContextLost())
86         return;
87     m_context.vertexAttribDivisor(index, divisor);
88 }
