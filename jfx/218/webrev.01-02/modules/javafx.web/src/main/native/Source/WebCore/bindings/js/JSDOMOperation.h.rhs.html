26 #include "JSDOMCastedThisErrorBehavior.h"
27 #include "JSDOMExceptionHandling.h"
28 
29 namespace WebCore {
30 
31 template<typename JSClass>
32 class IDLOperation {
33 public:
34     using ClassParameter = JSClass*;
35     using Operation = JSC::EncodedJSValue(JSC::JSGlobalObject*, JSC::CallFrame*, ClassParameter, JSC::ThrowScope&);
36     using StaticOperation = JSC::EncodedJSValue(JSC::JSGlobalObject*, JSC::CallFrame*, JSC::ThrowScope&);
37 
38     static JSClass* cast(JSC::JSGlobalObject&, JSC::CallFrame&);
39 
40     template<Operation operation, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::Throw>
41     static JSC::EncodedJSValue call(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, const char* operationName)
42     {
43         auto throwScope = DECLARE_THROW_SCOPE(JSC::getVM(&lexicalGlobalObject));
44 
45         auto* thisObject = cast(lexicalGlobalObject, callFrame);
46         if (shouldThrow != CastedThisErrorBehavior::Assert && UNLIKELY(!thisObject))
47             return throwThisTypeError(lexicalGlobalObject, throwScope, JSClass::info()->className, operationName);
48 
49         ASSERT(thisObject);
50         ASSERT_GC_OBJECT_INHERITS(thisObject, JSClass::info());
51 
52         // FIXME: We should refactor the binding generated code to use references for lexicalGlobalObject and thisObject.
53         return operation(&lexicalGlobalObject, &callFrame, thisObject, throwScope);
54     }
55 
56     template<StaticOperation operation, CastedThisErrorBehavior shouldThrow = CastedThisErrorBehavior::Throw>
57     static JSC::EncodedJSValue callStatic(JSC::JSGlobalObject& lexicalGlobalObject, JSC::CallFrame& callFrame, const char*)
58     {
59         auto throwScope = DECLARE_THROW_SCOPE(JSC::getVM(&lexicalGlobalObject));
60 
61         // FIXME: We should refactor the binding generated code to use references for lexicalGlobalObject.
62         return operation(&lexicalGlobalObject, &callFrame, throwScope);
63     }
64 };
65 
66 } // namespace WebCore
