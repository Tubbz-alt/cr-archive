 76 void DragClientJava::startDrag(DragItem item, DataTransfer& dataTransfer, Frame&)
 77 {
 78     auto& dragImage = item.image;
 79     auto dragImageOrigin = item.dragLocationInContentCoordinates;
 80     auto eventPos = item.eventPositionInContentCoordinates;
 81     auto dragSourceAction = item.sourceAction;
 82 
 83     JNIEnv* env = WTF::GetJavaEnv();
 84     static jmethodID mid = env->GetMethodID(
 85         PG_GetWebPageClass(env),
 86         "fwkStartDrag", "("
 87         "Ljava/lang/Object;"
 88         "II"
 89         "II"
 90         "[Ljava/lang/String;"
 91         "[Ljava/lang/Object;"
 92         "Z"
 93         ")V");
 94     ASSERT(mid);
 95 
 96     static JGClass clsString(env->FindClass("java/lang/String"));
 97     static JGClass clsObject(env->FindClass("java/lang/Object"));
 98 
 99     // we are temporary changing dataTransfer security context
100     // for transfer-to-Java purposes.
101     auto actualStoreMode = dataTransfer.storeMode();
102     dataTransfer.setStoreMode(DataTransfer::StoreMode::Readonly);
103 
104     Vector<String> mimeTypes(dataTransfer.types());
105     JLObjectArray jmimeTypes(env->NewObjectArray(mimeTypes.size(), clsString, NULL));
106     JLObjectArray jvalues(env->NewObjectArray(mimeTypes.size(), clsObject, NULL));
107     WTF::CheckAndClearException(env); // OOME
108 
109     auto document = WebPage::pageFromJObject(m_webPage)->mainFrame().document();
110     if (document) {
111         int index = 0;
112         for(const auto& mime : mimeTypes) {
113             String value = dataTransfer.getData(*document, mime);
114 
115             env->SetObjectArrayElement(
116                 jmimeTypes,
117                 index,
118                 (jstring)mime.toJavaString(env));
119 
120             env->SetObjectArrayElement(
121                 jvalues,
122                 index,
123                 (jstring)value.toJavaString(env));
124             index++;
125         }
126     }
127     // restore the original store mode
128     dataTransfer.setStoreMode(actualStoreMode);
129 
130     // Attention! [jimage] can be the instance of WCImage or WCImageFrame class.
131     // The nature of raster is too different to make a conversion inside the native code.
132     jobject jimage = dragImage.get() && dragImage.get()->javaImage()
133                   ? jobject(*(dragImage.get()->javaImage())) : nullptr;
134 
135     bool isImageSource = dragSourceAction & DragSourceActionImage;
136 
137     env->CallVoidMethod(m_webPage, mid, jimage,
138         eventPos.x() - dragImageOrigin.x(),
139         eventPos.y() - dragImageOrigin.y(),
140         eventPos.x(),
141         eventPos.y(),
142         jobjectArray(jmimeTypes),
143         jobjectArray(jvalues),
144         bool_to_jbool(isImageSource));
145     WTF::CheckAndClearException(env);
146 }
