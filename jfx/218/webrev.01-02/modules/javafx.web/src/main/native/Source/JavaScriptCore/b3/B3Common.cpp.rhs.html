40 bool shouldDumpIR(B3CompilationMode mode)
41 {
42 #if ENABLE(FTL_JIT)
43     return FTL::verboseCompilationEnabled() || FTL::shouldDumpDisassembly() || shouldDumpIRAtEachPhase(mode);
44 #else
45     return shouldDumpIRAtEachPhase(mode);
46 #endif
47 }
48 
49 bool shouldDumpIRAtEachPhase(B3CompilationMode mode)
50 {
51     if (mode == B3Mode)
52         return Options::dumpGraphAtEachPhase() || Options::dumpB3GraphAtEachPhase();
53     return Options::dumpGraphAtEachPhase() || Options::dumpAirGraphAtEachPhase();
54 }
55 
56 bool shouldValidateIR()
57 {
58     return DFG::validationEnabled() || shouldValidateIRAtEachPhase();
59 }
60 
61 bool shouldValidateIRAtEachPhase()
62 {
63     return Options::validateGraphAtEachPhase();
64 }
65 
66 bool shouldSaveIRBeforePhase()
67 {
68     return Options::verboseValidationFailure();
69 }
70 
71 Optional<GPRReg> pinnedExtendedOffsetAddrRegister()
72 {
73 #if CPU(ARM64)
74     return MacroAssembler::dataTempRegister;
75 #elif CPU(X86_64)
76     return WTF::nullopt;
77 #else
78 #error Unhandled architecture.
79 #endif
80 }
