 46 template<typename Config>
 47 void* IsoAllocator<Config>::allocate(IsoHeapImpl<Config>& heap, bool abortOnFailure)
 48 {
 49     static constexpr bool verbose = false;
 50     void* result = m_freeList.allocate<Config>(
 51         [&] () -> void* {
 52             return allocateSlow(heap, abortOnFailure);
 53         });
 54     if (verbose)
 55         fprintf(stderr, "%p: allocated %p of size %u\n", &heap, result, Config::objectSize);
 56     return result;
 57 }
 58 
 59 template<typename Config>
 60 BNO_INLINE void* IsoAllocator<Config>::allocateSlow(IsoHeapImpl<Config>& heap, bool abortOnFailure)
 61 {
 62     LockHolder locker(heap.lock);
 63 
 64     AllocationMode allocationMode = heap.updateAllocationMode();
 65     if (allocationMode == AllocationMode::Shared) {
 66         if (m_currentPage) {
 67             m_currentPage->stopAllocating(locker, m_freeList);
 68             m_currentPage = nullptr;
 69             m_freeList.clear();
 70         }
 71         return heap.allocateFromShared(locker, abortOnFailure);
 72     }
 73 
 74     BASSERT(allocationMode == AllocationMode::Fast);
 75 
 76     EligibilityResult<Config> result = heap.takeFirstEligible(locker);
 77     if (result.kind != EligibilityKind::Success) {
 78         RELEASE_BASSERT(result.kind == EligibilityKind::OutOfMemory);
 79         RELEASE_BASSERT(!abortOnFailure);
 80         return nullptr;
 81     }
 82 
 83     if (m_currentPage)
 84         m_currentPage->stopAllocating(locker, m_freeList);
 85 
 86     m_currentPage = result.page;
 87     m_freeList = m_currentPage->startAllocating(locker);
 88 
 89     return m_freeList.allocate<Config>([] () { BCRASH(); return nullptr; });
 90 }
 91 
 92 template<typename Config>
 93 void IsoAllocator<Config>::scavenge(IsoHeapImpl<Config>& heap)
 94 {
 95     if (m_currentPage) {
 96         LockHolder locker(heap.lock);
 97         m_currentPage->stopAllocating(locker, m_freeList);
 98         m_currentPage = nullptr;
 99         m_freeList.clear();
100     }
101 }
