48 namespace Inspector {
49 
50 typedef String ErrorString;
51 typedef WTF::Function<void(ErrorString&, RefPtr<Protocol::Runtime::RemoteObject>&&, Optional<bool>&, Optional<int>&)> AsyncCallCallback;
52 
53 class JS_EXPORT_PRIVATE InjectedScriptBase {
54 public:
55     virtual ~InjectedScriptBase();
56 
57     const String& name() const { return m_name; }
58     bool hasNoValue() const { return m_injectedScriptObject.hasNoValue(); }
59     JSC::JSGlobalObject* globalObject() const { return m_injectedScriptObject.globalObject(); }
60 
61 protected:
62     InjectedScriptBase(const String& name);
63     InjectedScriptBase(const String& name, Deprecated::ScriptObject, InspectorEnvironment*);
64 
65     InspectorEnvironment* inspectorEnvironment() const { return m_environment; }
66 
67     bool hasAccessToInspectedScriptState() const;
68 
69     const Deprecated::ScriptObject& injectedScriptObject() const;
70     Expected<JSC::JSValue, NakedPtr<JSC::Exception>> callFunctionWithEvalEnabled(Deprecated::ScriptFunctionCall&) const;
71     Ref<JSON::Value> makeCall(Deprecated::ScriptFunctionCall&);
72     void makeEvalCall(ErrorString&, Deprecated::ScriptFunctionCall&, RefPtr<Protocol::Runtime::RemoteObject>& resultObject, Optional<bool>& wasThrown, Optional<int>& savedResultIndex);
73     void makeAsyncCall(Deprecated::ScriptFunctionCall&, AsyncCallCallback&&);
74 
75 private:
76     void checkCallResult(ErrorString&, RefPtr<JSON::Value> result, RefPtr<Protocol::Runtime::RemoteObject>& resultObject, Optional<bool>& wasThrown, Optional<int>& savedResultIndex);
77     void checkAsyncCallResult(RefPtr<JSON::Value> result, const AsyncCallCallback&);
78 
79     String m_name;
80     Deprecated::ScriptObject m_injectedScriptObject;
81     InspectorEnvironment* m_environment { nullptr };
82 };
