25 #pragma once
26 
27 #include "Length.h"
28 #include "NinePieceImage.h"
29 #include "RenderStyleConstants.h"
30 #include <wtf/RefCounted.h>
31 
32 namespace WebCore {
33 
34 class StyleReflection : public RefCounted<StyleReflection> {
35 public:
36     static Ref<StyleReflection> create()
37     {
38         return adoptRef(*new StyleReflection);
39     }
40 
41     bool operator==(const StyleReflection& o) const
42     {
43         return m_direction == o.m_direction && m_offset == o.m_offset && m_mask == o.m_mask;
44     }
45     bool operator!=(const StyleReflection& o) const { return !(*this == o); }
46 
47     ReflectionDirection direction() const { return m_direction; }
48     const Length& offset() const { return m_offset; }
49     const NinePieceImage& mask() const { return m_mask; }
50 
51     void setDirection(ReflectionDirection dir) { m_direction = dir; }
52     void setOffset(Length offset) { m_offset = WTFMove(offset); }
53     void setMask(const NinePieceImage& image) { m_mask = image; }
54 
55 private:
56     StyleReflection()
57         : m_offset(0, Fixed)
58         , m_mask(NinePieceImage::Type::Mask)
59     {
60     }
61 
62     ReflectionDirection m_direction { ReflectionDirection::Below };
63     Length m_offset;
64     NinePieceImage m_mask;
65 };
66 
67 } // namespace WebCore
