 49 bool isCurrentColorString(const String& colorString)
 50 {
 51     return equalLettersIgnoringASCIICase(colorString, "currentcolor");
 52 }
 53 
 54 Color parseColor(const String& colorString, CanvasBase& canvasBase)
 55 {
 56 #if ENABLE(OFFSCREEN_CANVAS)
 57     if (canvasBase.isOffscreenCanvas()) {
 58         auto& canvas = downcast<OffscreenCanvas>(canvasBase);
 59         return CSSParser::parseColorWorkerSafe(colorString, canvas.cssValuePool());
 60     }
 61 #else
 62     UNUSED_PARAM(canvasBase);
 63 #endif
 64     Color color = CSSParser::parseColor(colorString);
 65     if (color.isValid())
 66         return color;
 67     return CSSParser::parseSystemColor(colorString, nullptr);
 68 }
 69 
 70 Color currentColor(CanvasBase& canvasBase)
 71 {
 72     if (!is<HTMLCanvasElement>(canvasBase))
 73         return Color::black;
 74 
 75     auto& canvas = downcast<HTMLCanvasElement>(canvasBase);
 76     if (!canvas.isConnected() || !canvas.inlineStyle())
 77         return Color::black;
 78     Color color = CSSParser::parseColor(canvas.inlineStyle()->getPropertyValue(CSSPropertyColor));
 79     if (!color.isValid())
 80         return Color::black;
 81     return color;
 82 }
 83 
 84 Color parseColorOrCurrentColor(const String& colorString, CanvasBase& canvasBase)
 85 {
 86     if (isCurrentColorString(colorString))
 87         return currentColor(canvasBase);
 88 
 89     return parseColor(colorString, canvasBase);
 90 }

127 CanvasStyle CanvasStyle::createFromString(const String& colorString, CanvasBase& canvasBase)
128 {
129     if (isCurrentColorString(colorString))
130         return CurrentColor { WTF::nullopt };
131 
132     Color color = parseColor(colorString, canvasBase);
133     if (!color.isValid())
134         return { };
135 
136     return color;
137 }
138 
139 CanvasStyle CanvasStyle::createFromStringWithOverrideAlpha(const String& colorString, float alpha, CanvasBase& canvasBase)
140 {
141     if (isCurrentColorString(colorString))
142         return CurrentColor { alpha };
143 
144     Color color = parseColor(colorString, canvasBase);
145     if (!color.isValid())
146         return { };
147 
148     return Color { colorWithOverrideAlpha(color.rgb(), alpha) };
149 }

151 bool CanvasStyle::isEquivalentColor(const CanvasStyle& other) const
152 {
153     if (WTF::holds_alternative<Color>(m_style) && WTF::holds_alternative<Color>(other.m_style))
154         return WTF::get<Color>(m_style) == WTF::get<Color>(other.m_style);
155 
156     if (WTF::holds_alternative<CMYKAColor>(m_style) && WTF::holds_alternative<CMYKAColor>(other.m_style)) {
157         auto& a = WTF::get<CMYKAColor>(m_style);
158         auto& b = WTF::get<CMYKAColor>(other.m_style);
159         return a.c == b.c && a.m == b.m && a.y == b.y && a.k == b.k && a.a == b.a;
160     }
161 
162     return false;
163 }
164 
165 bool CanvasStyle::isEquivalentRGBA(float r, float g, float b, float a) const
166 {
167     return WTF::holds_alternative<Color>(m_style) && WTF::get<Color>(m_style) == Color { r, g, b, a };
168 }
169 
170 bool CanvasStyle::isEquivalentCMYKA(float c, float m, float y, float k, float a) const
171 {
172     if (!WTF::holds_alternative<CMYKAColor>(m_style))
173         return false;
174 
175     auto& channels = WTF::get<CMYKAColor>(m_style);
176     return c == channels.c && m == channels.m && y == channels.y && k == channels.k && a == channels.a;
177 }

179 void CanvasStyle::applyStrokeColor(GraphicsContext& context) const
180 {
181     WTF::switchOn(m_style,
182         [&context] (const Color& color) {
183             context.setStrokeColor(color);
184         },
185         [&context] (const CMYKAColor& color) {
186             // FIXME: Do this through platform-independent GraphicsContext API.
187             // We'll need a fancier Color abstraction to support CMYKA correctly
188 #if USE(CG)
189             CGContextSetCMYKStrokeColor(context.platformContext(), color.c, color.m, color.y, color.k, color.a);
190 #else
191             context.setStrokeColor(color.color);
192 #endif
193         },
194         [&context] (const RefPtr<CanvasGradient>& gradient) {
195             context.setStrokeGradient(gradient->gradient());
196         },
197         [&context] (const RefPtr<CanvasPattern>& pattern) {
198             context.setStrokePattern(pattern->pattern());
199         },
200         [] (const CurrentColor&) {
201             ASSERT_NOT_REACHED();
202         },
203         [] (const Invalid&) {
204             ASSERT_NOT_REACHED();
205         }
206     );
207 }
