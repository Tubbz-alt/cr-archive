 60 WorkerInspectorController::WorkerInspectorController(WorkerGlobalScope& workerGlobalScope)
 61     : m_instrumentingAgents(InstrumentingAgents::create(*this))
 62     , m_injectedScriptManager(makeUnique<WebInjectedScriptManager>(*this, WebInjectedScriptHost::create()))
 63     , m_frontendRouter(FrontendRouter::create())
 64     , m_backendDispatcher(BackendDispatcher::create(m_frontendRouter.copyRef()))
 65     , m_executionStopwatch(Stopwatch::create())
 66     , m_scriptDebugServer(workerGlobalScope)
 67     , m_workerGlobalScope(workerGlobalScope)
 68 {
 69     ASSERT(workerGlobalScope.isContextThread());
 70 
 71     auto workerContext = workerAgentContext();
 72 
 73     auto consoleAgent = makeUnique<WorkerConsoleAgent>(workerContext);
 74     m_instrumentingAgents->setWebConsoleAgent(consoleAgent.get());
 75     m_agents.append(WTFMove(consoleAgent));
 76 }
 77 
 78 WorkerInspectorController::~WorkerInspectorController()
 79 {
 80     ASSERT(!m_frontendRouter->hasFrontends());
 81     ASSERT(!m_forwardingChannel);
 82 
 83     m_instrumentingAgents->reset();
 84 }
 85 
 86 void WorkerInspectorController::workerTerminating()
 87 {
 88     m_injectedScriptManager->disconnect();
 89 
 90     disconnectFrontend(Inspector::DisconnectReason::InspectedTargetDestroyed);
 91 
 92     m_agents.discardValues();
 93 }
 94 
 95 void WorkerInspectorController::connectFrontend()
 96 {
 97     ASSERT(!m_frontendRouter->hasFrontends());
 98     ASSERT(!m_forwardingChannel);
 99 
100     createLazyAgents();
101 
102     callOnMainThread([] {
103         InspectorInstrumentation::frontendCreated();
104     });
105 
106     m_executionStopwatch->reset();
107     m_executionStopwatch->start();
108 
109     m_forwardingChannel = makeUnique<WorkerToPageFrontendChannel>(m_workerGlobalScope);
110     m_frontendRouter->connectFrontend(*m_forwardingChannel.get());
111     m_agents.didCreateFrontendAndBackend(&m_frontendRouter.get(), &m_backendDispatcher.get());
112 }
113 
114 void WorkerInspectorController::disconnectFrontend(Inspector::DisconnectReason reason)
115 {
116     if (!m_frontendRouter->hasFrontends())
117         return;
118 
119     ASSERT(m_forwardingChannel);
120 
121     callOnMainThread([] {
122         InspectorInstrumentation::frontendDeleted();
123     });
124 
125     m_agents.willDestroyFrontendAndBackend(reason);
126     m_frontendRouter->disconnectFrontend(*m_forwardingChannel.get());
127     m_forwardingChannel = nullptr;
128 }
129 
130 void WorkerInspectorController::dispatchMessageFromFrontend(const String& message)
131 {
132     m_backendDispatcher->dispatch(message);
133 }

135 WorkerAgentContext WorkerInspectorController::workerAgentContext()
136 {
137     AgentContext baseContext = {
138         *this,
139         *m_injectedScriptManager,
140         m_frontendRouter.get(),
141         m_backendDispatcher.get(),
142     };
143 
144     WebAgentContext webContext = {
145         baseContext,
146         m_instrumentingAgents.get(),
147     };
148 
149     WorkerAgentContext workerContext = {
150         webContext,
151         m_workerGlobalScope,
152     };
153 
154     return workerContext;
155 }
156 
157 void WorkerInspectorController::createLazyAgents()
158 {
159     if (m_didCreateLazyAgents)
160         return;
161 
162     m_didCreateLazyAgents = true;
163 
164     m_injectedScriptManager->connect();
165 
166     auto workerContext = workerAgentContext();
167 
168     m_agents.append(makeUnique<WorkerRuntimeAgent>(workerContext));
169 
170 #if ENABLE(SERVICE_WORKER)
171     if (is<ServiceWorkerGlobalScope>(m_workerGlobalScope)) {
172         m_agents.append(makeUnique<ServiceWorkerAgent>(workerContext));
173         m_agents.append(makeUnique<WorkerNetworkAgent>(workerContext));
174     }
175 #endif
176 
177     m_agents.append(makeUnique<WebHeapAgent>(workerContext));
178 
179     auto debuggerAgent = makeUnique<WorkerDebuggerAgent>(workerContext);
180     auto debuggerAgentPtr = debuggerAgent.get();
181     m_agents.append(WTFMove(debuggerAgent));
182 
183     m_agents.append(makeUnique<WorkerDOMDebuggerAgent>(workerContext, debuggerAgentPtr));
184     m_agents.append(makeUnique<WorkerAuditAgent>(workerContext));
185 
186     if (auto& commandLineAPIHost = m_injectedScriptManager->commandLineAPIHost())
187         commandLineAPIHost->init(m_instrumentingAgents.copyRef());
188 }
