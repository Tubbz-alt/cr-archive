 50 WorkletGlobalScope::WorkletGlobalScope(Document& document, ScriptSourceCode&& code)
 51     : m_document(makeWeakPtr(document))

 52     , m_script(makeUnique<WorkletScriptController>(this))
 53     , m_topOrigin(SecurityOrigin::createUnique())

 54     , m_code(WTFMove(code))
 55 {
 56     auto addResult = allWorkletGlobalScopesSet().add(this);
 57     ASSERT_UNUSED(addResult, addResult);
 58 
 59     auto* frame = document.frame();
 60     m_jsRuntimeFlags = frame ? frame->settings().javaScriptRuntimeFlags() : JSC::RuntimeFlags();
 61     ASSERT(document.page());
 62 
 63     setSecurityOriginPolicy(SecurityOriginPolicy::create(m_topOrigin.copyRef()));
 64     setContentSecurityPolicy(makeUnique<ContentSecurityPolicy>(URL { m_code.url() }, *this));
 65 }
 66 
 67 WorkletGlobalScope::~WorkletGlobalScope()
 68 {
 69     ASSERT(!m_script);
 70     removeFromContextsMap();
 71     auto removeResult = allWorkletGlobalScopesSet().remove(this);
 72     ASSERT_UNUSED(removeResult, removeResult);
 73 }
 74 
 75 void WorkletGlobalScope::prepareForDestruction()
 76 {
 77     if (!m_script)
 78         return;
 79     if (m_defaultTaskGroup)
 80         m_defaultTaskGroup->stopAndDiscardAllTasks();
 81     stopActiveDOMObjects();

 82     removeAllEventListeners();
 83     if (m_eventLoop)
 84         m_eventLoop->clearMicrotaskQueue();
 85     removeRejectedPromiseTracker();
 86     m_script->vm().notifyNeedTermination();
 87     m_script = nullptr;
 88 }
 89 
 90 auto WorkletGlobalScope::allWorkletGlobalScopesSet() -> WorkletGlobalScopesSet&
 91 {
 92     static NeverDestroyed<WorkletGlobalScopesSet> scopes;
 93     return scopes;
 94 }
 95 
 96 EventLoopTaskGroup& WorkletGlobalScope::eventLoop()
 97 {
 98     if (UNLIKELY(!m_defaultTaskGroup)) {
 99         m_eventLoop = WorkerEventLoop::create(*this);
100         m_defaultTaskGroup = makeUnique<EventLoopTaskGroup>(*m_eventLoop);
101         if (activeDOMObjectsAreStopped())
102             m_defaultTaskGroup->stopAndDiscardAllTasks();
103     }
104     return *m_defaultTaskGroup;
105 }

107 String WorkletGlobalScope::userAgent(const URL& url) const
108 {
109     if (!m_document)
110         return "";
111     return m_document->userAgent(url);
112 }
113 
114 void WorkletGlobalScope::evaluate()
115 {
116     m_script->evaluate(m_code);
117 }
118 
119 bool WorkletGlobalScope::isJSExecutionForbidden() const
120 {
121     return !m_script || m_script->isExecutionForbidden();
122 }
123 
124 void WorkletGlobalScope::disableEval(const String& errorMessage)
125 {
126     m_script->disableEval(errorMessage);
127 }
128 
129 void WorkletGlobalScope::disableWebAssembly(const String& errorMessage)
130 {
131     m_script->disableWebAssembly(errorMessage);
132 }
133 
134 URL WorkletGlobalScope::completeURL(const String& url, ForceUTF8) const
135 {
136     if (url.isNull())
137         return URL();
138     return URL(m_code.url(), url);
139 }

141 void WorkletGlobalScope::logExceptionToConsole(const String& errorMessage, const String& sourceURL, int lineNumber, int columnNumber, RefPtr<ScriptCallStack>&& stack)
142 {
143     if (!m_document || isJSExecutionForbidden())
144         return;
145     m_document->logExceptionToConsole(errorMessage, sourceURL, lineNumber, columnNumber, WTFMove(stack));
146 }
147 
148 void WorkletGlobalScope::addConsoleMessage(std::unique_ptr<Inspector::ConsoleMessage>&& message)
149 {
150     if (!m_document || isJSExecutionForbidden() || !message)
151         return;
152     m_document->addConsoleMessage(makeUnique<Inspector::ConsoleMessage>(message->source(), message->type(), message->level(), message->message(), 0));
153 }
154 
155 void WorkletGlobalScope::addConsoleMessage(MessageSource source, MessageLevel level, const String& message, unsigned long requestIdentifier)
156 {
157     if (!m_document || isJSExecutionForbidden())
158         return;
159     m_document->addConsoleMessage(source, level, message, requestIdentifier);
160 }
161 
162 void WorkletGlobalScope::addMessage(MessageSource source, MessageLevel level, const String& messageText, const String& sourceURL, unsigned lineNumber, unsigned columnNumber, RefPtr<ScriptCallStack>&& callStack, JSC::JSGlobalObject*, unsigned long requestIdentifier)
163 {
164     if (!m_document || isJSExecutionForbidden())
165         return;
166     m_document->addMessage(source, level, messageText, sourceURL, lineNumber, columnNumber, WTFMove(callStack), nullptr, requestIdentifier);
167 }
