 40 DOMFormData::DOMFormData(const TextEncoding& encoding)
 41     : m_encoding(encoding)
 42 {
 43 }
 44 
 45 DOMFormData::DOMFormData(HTMLFormElement* form)
 46     : m_encoding(UTF8Encoding())
 47 {
 48     if (!form)
 49         return;
 50 
 51     ASSERT(isMainThread());
 52     for (auto& element : form->copyAssociatedElementsVector()) {
 53         if (!element->asHTMLElement().isDisabledFormControl())
 54             element->appendFormData(*this, true);
 55     }
 56 }

 58 // https://xhr.spec.whatwg.org/#create-an-entry
 59 auto DOMFormData::createFileEntry(const String& name, Blob& blob, const String& filename) -> Item
 60 {
 61     if (!blob.isFile())
 62         return { name, File::create(blob, filename.isNull() ? "blob"_s : filename) };
 63 
 64     if (!filename.isNull())
 65         return { name, File::create(downcast<File>(blob), filename) };
 66 
 67     return { name, RefPtr<File> { &downcast<File>(blob) } };
 68 }

 70 void DOMFormData::append(const String& name, const String& value)
 71 {
 72     m_items.append({ name, value });
 73 }
 74 
 75 void DOMFormData::append(const String& name, Blob& blob, const String& filename)
 76 {
 77     m_items.append(createFileEntry(name, blob, filename));
 78 }
 79 
 80 void DOMFormData::remove(const String& name)
 81 {
 82     m_items.removeAllMatching([&name] (const auto& item) {
 83         return item.name == name;
 84     });
 85 }
 86 
 87 auto DOMFormData::get(const String& name) -> Optional<FormDataEntryValue>
 88 {
 89     for (auto& item : m_items) {
 90         if (item.name == name)
 91             return item.data;
 92     }
 93 
 94     return WTF::nullopt;
 95 }
 96 
 97 auto DOMFormData::getAll(const String& name) -> Vector<FormDataEntryValue>
 98 {
 99     Vector<FormDataEntryValue> result;
100 
101     for (auto& item : m_items) {
102         if (item.name == name)
103             result.append(item.data);
104     }
105 
106     return result;
107 }
108 
109 bool DOMFormData::has(const String& name)
110 {
111     for (auto& item : m_items) {
112         if (item.name == name)
113             return true;
114     }
115 
116     return false;
117 }

119 void DOMFormData::set(const String& name, const String& value)
120 {
121     set(name, { name, value });
122 }
123 
124 void DOMFormData::set(const String& name, Blob& blob, const String& filename)
125 {
126     set(name, createFileEntry(name, blob, filename));
127 }
128 
129 void DOMFormData::set(const String& name, Item&& item)
130 {
131     Optional<size_t> initialMatchLocation;
132 
133     // Find location of the first item with a matching name.
134     for (size_t i = 0; i < m_items.size(); ++i) {
135         if (name == m_items[i].name) {
136             initialMatchLocation = i;
137             break;
138         }
139     }
140 
141     if (initialMatchLocation) {
142         m_items[*initialMatchLocation] = WTFMove(item);
143 
144         m_items.removeAllMatching([&name] (const auto& item) {
145             return item.name == name;
146         }, *initialMatchLocation + 1);
147         return;
148     }
149 
150     m_items.append(WTFMove(item));
151 }

153 DOMFormData::Iterator::Iterator(DOMFormData& target)
154     : m_target(target)
155 {
156 }
157 
158 Optional<KeyValuePair<String, DOMFormData::FormDataEntryValue>> DOMFormData::Iterator::next()
159 {
160     auto& items = m_target->items();
161     if (m_index >= items.size())
162         return WTF::nullopt;
163 
164     auto& item = items[m_index++];
165     return makeKeyValuePair(item.name, item.data);
166 }
