31 namespace WebCore {
32 
33 WTF_MAKE_ISO_ALLOCATED_IMPL(WebKitTransitionEvent);
34 
35 WebKitTransitionEvent::WebKitTransitionEvent(const AtomString& type, const String& propertyName, double elapsedTime, const String& pseudoElement)
36     : Event(type, CanBubble::Yes, IsCancelable::Yes)
37     , m_propertyName(propertyName)
38     , m_elapsedTime(elapsedTime)
39     , m_pseudoElement(pseudoElement)
40 {
41 }
42 
43 WebKitTransitionEvent::WebKitTransitionEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
44     : Event(type, initializer, isTrusted)
45     , m_propertyName(initializer.propertyName)
46     , m_elapsedTime(initializer.elapsedTime)
47     , m_pseudoElement(initializer.pseudoElement)
48 {
49 }
50 
51 WebKitTransitionEvent::~WebKitTransitionEvent() = default;
52 
53 const String& WebKitTransitionEvent::propertyName() const
54 {
55     return m_propertyName;
56 }
57 
58 double WebKitTransitionEvent::elapsedTime() const
59 {
60     return m_elapsedTime;
61 }
62 
63 const String& WebKitTransitionEvent::pseudoElement() const
64 {
65     return m_pseudoElement;
66 }
67 
68 EventInterface WebKitTransitionEvent::eventInterface() const
69 {
70     return WebKitTransitionEventInterfaceType;
71 }
