26 #pragma once
27 
28 #include "ExceptionOr.h"
29 #include <wtf/Ref.h>
30 #include <wtf/RefCounted.h>

31 
32 namespace WebCore {
33 
34 class Node;
35 class Range;
36 
37 class StaticRange : public RefCounted<StaticRange> {
38 public:
39     ~StaticRange();
40 
41     static Ref<StaticRange> createFromRange(const Range&);
42     static Ref<StaticRange> create(Ref<Node>&& startContainer, unsigned startOffset, Ref<Node>&& endContainer, unsigned endOffset);
43 
44     struct Init {
45         RefPtr<Node> startContainer;
46         unsigned long startOffset { 0 };
47         RefPtr<Node> endContainer;
48         unsigned long endOffset { 0 };
49     };
50 
51     static ExceptionOr<Ref<StaticRange>> create(Init&&);
52 
53     unsigned startOffset() const { return m_startOffset; }
54     unsigned endOffset() const { return m_endOffset; }
55     Node* startContainer() const;
56     Node* endContainer() const;
57     bool collapsed() const;
58 
59     bool operator==(const StaticRange&) const;
60 
61 private:
62     StaticRange(Ref<Node>&& startContainer, unsigned startOffset, Ref<Node>&& endContainer, unsigned endOffset);
63 
64     Ref<Node> m_startContainer;
65     unsigned m_startOffset;
66     Ref<Node> m_endContainer;
67     unsigned m_endOffset;
68 };
69 
70 }
