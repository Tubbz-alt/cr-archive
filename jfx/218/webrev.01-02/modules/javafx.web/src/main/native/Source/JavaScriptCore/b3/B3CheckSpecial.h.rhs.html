 40 // We want to lower Check instructions to a branch, but then we want to route that branch to our
 41 // out-of-line code instead of doing anything else. For this reason, a CheckSpecial will remember
 42 // which branch opcode we have selected along with the number of args in the overload we want. It
 43 // will create an Inst with that opcode plus the appropriate args from the owning Inst whenever you
 44 // call any of the callbacks.
 45 //
 46 // Note that for CheckAdd, CheckSub, and CheckMul we expect that the B3 arguments are the reverse
 47 // of the Air arguments (Add(a, b) => Add32 b, a). Except:
 48 // - CheckSub(0, x), which turns into BranchNeg32 x.
 49 // - CheckMul(a, b), which turns into Mul32 b, a but we pass Any for a's ValueRep.
 50 
 51 class CheckSpecial : public StackmapSpecial {
 52 public:
 53     // Support for hash consing these things.
 54     class Key {
 55     public:
 56         Key()
 57             : m_stackmapRole(SameAsRep)
 58             , m_numArgs(0)
 59         {
 60         }
 61 
 62         Key(Air::Kind kind, unsigned numArgs, RoleMode stackmapRole = SameAsRep)
 63             : m_kind(kind)
 64             , m_stackmapRole(stackmapRole)
 65             , m_numArgs(numArgs)
 66         {
 67         }
 68 
 69         explicit Key(const Air::Inst&);
 70 
 71         bool operator==(const Key& other) const
 72         {
 73             return m_kind == other.m_kind
 74                 && m_numArgs == other.m_numArgs
 75                 && m_stackmapRole == other.m_stackmapRole;
 76         }
 77 
 78         bool operator!=(const Key& other) const
 79         {
 80             return !(*this == other);
 81         }
 82 
 83         explicit operator bool() const { return *this != Key(); }
 84 
 85         Air::Kind kind() const { return m_kind; }
 86         unsigned numArgs() const { return m_numArgs; }
 87         RoleMode stackmapRole() const { return m_stackmapRole; }
 88 
 89         void dump(PrintStream& out) const;
 90 
 91         Key(WTF::HashTableDeletedValueType)
 92             : m_stackmapRole(SameAsRep)
 93             , m_numArgs(1)
 94         {
 95         }
 96 
 97         bool isHashTableDeletedValue() const
 98         {
 99             return *this == Key(WTF::HashTableDeletedValue);
100         }
101 
102         unsigned hash() const
103         {
104             // Seriously, we don't need to be smart here. It just doesn't matter.
105             return m_kind.hash() + m_numArgs + m_stackmapRole;
106         }
107 
108     private:
109         Air::Kind m_kind;
110         RoleMode m_stackmapRole;
111         unsigned m_numArgs;
112     };
113 
114     CheckSpecial(Air::Kind, unsigned numArgs, RoleMode stackmapRole = SameAsRep);
115     CheckSpecial(const Key&);
116     ~CheckSpecial();
117 
118 protected:
119     // Constructs and returns the Inst representing the branch that this will use.
120     Air::Inst hiddenBranch(const Air::Inst&) const;
121 
122     void forEachArg(Air::Inst&, const ScopedLambda<Air::Inst::EachArgCallback>&) final;
123     bool isValid(Air::Inst&) final;
124     bool admitsStack(Air::Inst&, unsigned argIndex) final;
125     bool admitsExtendedOffsetAddr(Air::Inst&, unsigned) final;
126     Optional<unsigned> shouldTryAliasingDef(Air::Inst&) final;
127 
128     // NOTE: the generate method will generate the hidden branch and then register a LatePath that
129     // generates the stackmap. Super crazy dude!
130 
131     MacroAssembler::Jump generate(Air::Inst&, CCallHelpers&, Air::GenerationContext&) final;
132 
133     void dumpImpl(PrintStream&) const final;
134     void deepDumpImpl(PrintStream&) const final;
135 
136 private:
137     Air::Kind m_checkKind;
138     RoleMode m_stackmapRole;
139     unsigned m_numCheckArgs;
140 };
141 
142 struct CheckSpecialKeyHash {
143     static unsigned hash(const CheckSpecial::Key& key) { return key.hash(); }
144     static bool equal(const CheckSpecial::Key& a, const CheckSpecial::Key& b) { return a == b; }
145     static constexpr bool safeToCompareToEmptyOrDeleted = true;
146 };
