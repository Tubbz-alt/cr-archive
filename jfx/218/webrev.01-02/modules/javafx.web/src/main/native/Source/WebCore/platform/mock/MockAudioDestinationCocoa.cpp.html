31 #include "CAAudioStreamDescription.h"
32 #include "WebAudioBufferList.h"
33 #include <wtf/threads/BinarySemaphore.h>
34 
35 namespace WebCore {
36 
37 const int kRenderBufferSize = 128;
38 
39 MockAudioDestinationCocoa::MockAudioDestinationCocoa(AudioIOCallback& callback, float sampleRate)
40     : AudioDestinationCocoa(callback, sampleRate)
41     , m_workQueue(WorkQueue::create("MockAudioDestinationCocoa Render Queue"))
42     , m_timer(RunLoop::current(), this, &MockAudioDestinationCocoa::tick)
43 {
44 }
45 
46 MockAudioDestinationCocoa::~MockAudioDestinationCocoa() = default;
47 
48 void MockAudioDestinationCocoa::start()
49 {
50     m_timer.startRepeating(Seconds { m_numberOfFramesToProcess / sampleRate() });
51     setIsPlaying(true);
52 }
53 
54 void MockAudioDestinationCocoa::stop()
55 {
56     m_timer.stop();
57     setIsPlaying(false);
58 
59     BinarySemaphore semaphore;
60     m_workQueue->dispatch([&semaphore] {
61         semaphore.signal();
62     });
63     semaphore.wait();
64 }
65 
66 void MockAudioDestinationCocoa::tick()
67 {
68     m_workQueue->dispatch([this, sampleRate = sampleRate(), numberOfFramesToProcess = m_numberOfFramesToProcess] {
69         AudioStreamBasicDescription streamFormat;
70         setAudioStreamBasicDescription(streamFormat, sampleRate);
71 
72         WebAudioBufferList webAudioBufferList { streamFormat, numberOfFramesToProcess };
73         AudioDestinationCocoa::inputProc(this, 0, 0, 0, numberOfFramesToProcess, webAudioBufferList.list());
74     });
75 }
76 
77 } // namespace WebCore
78 
79 #endif // ENABLE(WEB_AUDIO)
