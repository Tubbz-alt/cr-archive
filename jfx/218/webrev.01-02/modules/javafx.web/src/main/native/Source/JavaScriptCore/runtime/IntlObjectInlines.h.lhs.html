37 template<typename IntlInstance, typename Constructor, typename Factory>
38 JSValue constructIntlInstanceWithWorkaroundForLegacyIntlConstructor(ExecState& state, JSValue thisValue, Constructor* callee, Factory factory)
39 {
40     // FIXME: Workaround to provide compatibility with ECMA-402 1.0 call/apply patterns.
41     // https://bugs.webkit.org/show_bug.cgi?id=153679
42     VM& vm = state.vm();
43     auto scope = DECLARE_THROW_SCOPE(vm);
44 
45     if (!jsDynamicCast<IntlInstance*>(vm, thisValue)) {
46         JSValue prototype = callee->getDirect(vm, vm.propertyNames->prototype);
47         bool hasInstance = JSObject::defaultHasInstance(&state, thisValue, prototype);
48         RETURN_IF_EXCEPTION(scope, JSValue());
49         if (hasInstance) {
50             JSObject* thisObject = thisValue.toObject(&state);
51             RETURN_IF_EXCEPTION(scope, JSValue());
52 
53             IntlInstance* instance = factory(vm);
54             RETURN_IF_EXCEPTION(scope, JSValue());
55 
56             thisObject->putDirect(vm, vm.propertyNames->builtinNames().intlSubstituteValuePrivateName(), instance);
57             return thisObject;
58         }
59     }
60     RELEASE_AND_RETURN(scope, factory(vm));
61 }
62 
63 } // namespace JSC
