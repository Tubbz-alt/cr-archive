34 namespace WebCore {
35 


36 class AuthenticatorResponse : public RefCounted<AuthenticatorResponse> {
37 public:
38     enum class Type {
39         Assertion,
40         Attestation
41     };
42 
43     explicit AuthenticatorResponse(Ref<ArrayBuffer>&& clientDataJSON)
44         : m_clientDataJSON(WTFMove(clientDataJSON))
45     {
46     }
47     virtual ~AuthenticatorResponse() = default;
48 
49     virtual Type type() const = 0;







50 
51     ArrayBuffer* clientDataJSON() const { return m_clientDataJSON.ptr(); }

52 
53 private:
54     Ref<ArrayBuffer> m_clientDataJSON;


55 };
56 
57 } // namespace WebCore
