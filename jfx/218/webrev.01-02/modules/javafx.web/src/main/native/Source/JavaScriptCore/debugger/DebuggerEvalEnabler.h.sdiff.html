33 class DebuggerEvalEnabler {
34 public:
35     enum class Mode {
36         EvalOnCurrentCallFrame,
37         EvalOnCallFrameAtDebuggerEntry,
38     };
39 
40     DebuggerEvalEnabler(const ExecState* exec, Mode mode = Mode::EvalOnCurrentCallFrame)
41         : m_exec(exec)
42 #if !ASSERT_DISABLED
43         , m_mode(mode)
44 #endif
45 
46     {
47         UNUSED_PARAM(mode);
48         if (exec) {
49             JSGlobalObject* globalObject = exec->lexicalGlobalObject();
50             m_evalWasDisabled = !globalObject->evalEnabled();
51             if (m_evalWasDisabled)
52                 globalObject->setEvalEnabled(true, globalObject->evalDisabledErrorMessage());
53 #if !ASSERT_DISABLED
54             if (m_mode == Mode::EvalOnCallFrameAtDebuggerEntry)
55                 globalObject->setCallFrameAtDebuggerEntry(exec);
56 #endif
57         }
58     }
59 
60     ~DebuggerEvalEnabler()
61     {
62         if (m_exec) {
63             JSGlobalObject* globalObject = m_exec->lexicalGlobalObject();
64             if (m_evalWasDisabled)
65                 globalObject->setEvalEnabled(false, globalObject->evalDisabledErrorMessage());
66 #if !ASSERT_DISABLED
67             if (m_mode == Mode::EvalOnCallFrameAtDebuggerEntry)
68                 globalObject->setCallFrameAtDebuggerEntry(nullptr);
69 #endif
70         }
71     }
72 
73 private:
74     const ExecState* m_exec;
75     bool m_evalWasDisabled { false };
76 #if !ASSERT_DISABLED
77     DebuggerEvalEnabler::Mode m_mode;
78 #endif
79 };

81 } // namespace JSC
