35 inline SignatureInformation& SignatureInformation::singleton()
36 {
37     std::call_once(signatureInformationFlag, [] () {
38         theOne = new SignatureInformation;
39     });
40     return *theOne;
41 }
42 
43 inline const Signature& SignatureInformation::get(SignatureIndex index)
44 {
45     ASSERT(index != Signature::invalidIndex);
46     return *bitwise_cast<const Signature*>(index);
47 }
48 
49 inline SignatureIndex SignatureInformation::get(const Signature& signature)
50 {
51     if (ASSERT_ENABLED) {
52         SignatureInformation& info = singleton();
53         auto locker = holdLock(info.m_lock);
54         ASSERT_UNUSED(info, info.m_signatureSet.contains(SignatureHash { makeRef(const_cast<Signature&>(signature)) }));
55     }
56     return bitwise_cast<SignatureIndex>(&signature);
57 }
