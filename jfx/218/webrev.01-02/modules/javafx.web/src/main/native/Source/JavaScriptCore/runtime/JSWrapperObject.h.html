22 #pragma once
23 
24 #include "JSObject.h"
25 
26 namespace JSC {
27 
28 // This class is used as a base for classes such as String,
29 // Number, Boolean and Symbol which are wrappers for primitive types.
30 class JSWrapperObject : public JSNonFinalObject {
31 public:
32     using Base = JSNonFinalObject;
33 
34     template<typename, SubspaceAccess>
35     static void subspaceFor(VM&)
36     {
37         RELEASE_ASSERT_NOT_REACHED();
38     }
39 
40     static size_t allocationSize(Checked<size_t> inlineCapacity)
41     {
42         ASSERT_UNUSED(inlineCapacity, !inlineCapacity);
43         return sizeof(JSWrapperObject);
44     }
45 
46     JSValue internalValue() const;
47     void setInternalValue(VM&, JSValue);
48 
49     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
50     {
51         return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
52     }
53 
54     static ptrdiff_t internalValueOffset() { return OBJECT_OFFSETOF(JSWrapperObject, m_internalValue); }
55     static ptrdiff_t internalValueCellOffset()
56     {
57 #if USE(JSVALUE64)
58         return internalValueOffset();
59 #else
60         return internalValueOffset() + OBJECT_OFFSETOF(EncodedValueDescriptor, asBits.payload);
61 #endif
62     }
63 
64 protected:
65     explicit JSWrapperObject(VM&, Structure*);
66 
67     JS_EXPORT_PRIVATE static void visitChildren(JSCell*, SlotVisitor&);
68 
69 private:
70     WriteBarrier<Unknown> m_internalValue;
71 };
72 
73 inline JSWrapperObject::JSWrapperObject(VM& vm, Structure* structure)
74     : Base(vm, structure)
75 {
76 }
77 
78 inline JSValue JSWrapperObject::internalValue() const
79 {
80     return m_internalValue.get();
81 }
82 
83 inline void JSWrapperObject::setInternalValue(VM& vm, JSValue value)
84 {
85     ASSERT(value);
86     ASSERT(!value.isObject());
87     m_internalValue.set(vm, this, value);
88 }
89 
90 } // namespace JSC
