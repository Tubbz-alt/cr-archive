 31 #include "ConcurrentJSLock.h"
 32 #include "JSObject.h"
 33 #include "JSSymbolTableObject.h"
 34 #include "SymbolTable.h"
 35 #include <wtf/SegmentedVector.h>

 37 namespace JSC {
 38 
 39 class LLIntOffsetsExtractor;
 40 
 41 // This is a mostly drop-in replacement for JSLexicalEnvironment, except that it preserves
 42 // the invariant that after a variable is created, its address in memory will not change
 43 // so long as the JSSegmentedVariableObject is alive. This allows optimizations based
 44 // on getting the address of the variable and remembering it. As well, unlike a
 45 // JSLexicalEnvironment, this will manage the memory for the registers itself and neither
 46 // requires nor allows for the subclasses to manage that memory. Finally,
 47 // JSSegmentedVariableObject has its own GC tracing functionality, since it knows the
 48 // exact dimensions of the variables array at all times.
 49 
 50 class JSSegmentedVariableObject : public JSSymbolTableObject {
 51     friend class JIT;
 52     friend class LLIntOffsetsExtractor;

 53 public:
 54     using Base = JSSymbolTableObject;
 55 
 56     DECLARE_INFO;
 57 
 58     static constexpr bool needsDestruction = true;

 59     template<typename CellType, SubspaceAccess>
 60     static void subspaceFor(VM&) { RELEASE_ASSERT_NOT_REACHED(); }



 61 
 62     bool isValidScopeOffset(ScopeOffset offset)
 63     {
 64         return !!offset && offset.offset() < m_variables.size();
 65     }
 66 
 67     // This is not thread-safe, since m_variables is a segmented vector, and its spine can resize with
 68     // malloc/free if new variables - unrelated to the one you are accessing - are added. You can get
 69     // around this by grabbing m_lock, or finding some other way to get to the variable pointer (global
 70     // variable access bytecode instructions will have a direct pointer already).
 71     WriteBarrier<Unknown>& variableAt(ScopeOffset offset) { return m_variables[offset.offset()]; }
 72 
 73     // This is a slow method call, which searches the register bank to find the index
 74     // given a pointer. It will CRASH() if it does not find the register. Only use this
 75     // in debug code (like bytecode dumping).
 76     JS_EXPORT_PRIVATE ScopeOffset findVariableIndex(void*);
 77 
 78     WriteBarrier<Unknown>* assertVariableIsInThisObject(WriteBarrier<Unknown>* variablePointer)
 79     {
 80         if (ASSERT_ENABLED)
 81             findVariableIndex(variablePointer);
 82         return variablePointer;
 83     }
 84 
 85     // Adds numberOfRegistersToAdd registers, initializes them to Undefined, and returns
 86     // the index of the first one added.
 87     JS_EXPORT_PRIVATE ScopeOffset addVariables(unsigned numberOfVariablesToAdd, JSValue);
 88 
 89     JS_EXPORT_PRIVATE static void visitChildren(JSCell*, SlotVisitor&);
 90     JS_EXPORT_PRIVATE static void analyzeHeap(JSCell*, HeapAnalyzer&);
 91 




 92 protected:
 93     JSSegmentedVariableObject(VM&, Structure*, JSScope*);
 94 
 95     ~JSSegmentedVariableObject();
 96 
 97     void finishCreation(VM&);
 98 
 99 private:
100     SegmentedVector<WriteBarrier<Unknown>, 16> m_variables;

101 #ifndef NDEBUG
102     bool m_alreadyDestroyed { false }; // We use these assertions to check that we aren't doing ancient hacks that result in this being destroyed more than once.
103 #endif
104 };
105 
106 } // namespace JSC
