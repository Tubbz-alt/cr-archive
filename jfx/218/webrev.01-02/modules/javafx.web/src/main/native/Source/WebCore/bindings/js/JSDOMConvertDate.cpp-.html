33 // FIXME: This should get passed a global object rather than getting it out of the ExecState.
34 JSValue jsDate(ExecState& state, double value)
35 {
36     return DateInstance::create(state.vm(), state.lexicalGlobalObject()->dateStructure(), value);
37 }
38 
39 double valueToDate(ExecState& state, JSValue value)
40 {
41     if (value.isNumber())
42         return value.asNumber();
43     if (!value.inherits<DateInstance>(state.vm()))
44         return std::numeric_limits<double>::quiet_NaN();
45     return static_cast<DateInstance*>(value.toObject(&state))->internalNumber();
46 }
