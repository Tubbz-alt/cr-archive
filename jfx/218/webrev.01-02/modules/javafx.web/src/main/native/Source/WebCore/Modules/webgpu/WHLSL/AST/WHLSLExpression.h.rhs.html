 30 #include "WHLSLAddressSpace.h"
 31 #include "WHLSLCodeLocation.h"
 32 #include "WHLSLDefaultDelete.h"
 33 #include "WHLSLUnnamedType.h"
 34 #include <wtf/FastMalloc.h>
 35 #include <wtf/Optional.h>
 36 #include <wtf/UniqueRef.h>
 37 
 38 namespace WebCore {
 39 
 40 namespace WHLSL {
 41 
 42 namespace AST {
 43 
 44 class Expression {
 45     WTF_MAKE_FAST_ALLOCATED;
 46 
 47 protected:
 48     ~Expression() = default;
 49 
 50 public:
 51 
 52     enum class Kind : uint8_t {
 53         Assignment,
 54         BooleanLiteral,
 55         Call,
 56         Comma,
 57         Dereference,
 58         Dot,
 59         GlobalVariableReference,
 60         FloatLiteral,
 61         Index,
 62         IntegerLiteral,
 63         Logical,
 64         LogicalNot,
 65         MakeArrayReference,
 66         MakePointer,

 67         ReadModifyWrite,
 68         Ternary,
 69         UnsignedIntegerLiteral,
 70         VariableReference,
 71         EnumerationMemberLiteral,
 72     };
 73 
 74     Expression(CodeLocation codeLocation, Kind kind)
 75         : m_codeLocation(codeLocation)
 76         , m_kind(kind)
 77     {
 78     }
 79 
 80     static void destroy(Expression&);
 81     static void destruct(Expression&);
 82 
 83     Expression(const Expression&) = delete;
 84     Expression(Expression&&) = default;
 85 
 86     Expression& operator=(const Expression&) = delete;
 87     Expression& operator=(Expression&&) = default;
 88 
 89     UnnamedType* maybeResolvedType() { return m_type ? &*m_type : nullptr; }
 90 
 91     UnnamedType& resolvedType()
 92     {
 93         ASSERT(m_type);
 94         return *m_type;
 95     }
 96 
 97     void setType(Ref<UnnamedType> type)
 98     {
 99         ASSERT(!m_type);
100         m_type = WTFMove(type);
101     }
102 
103     const TypeAnnotation* maybeTypeAnnotation() const { return m_typeAnnotation ? &*m_typeAnnotation : nullptr; }
104 
105     const TypeAnnotation& typeAnnotation() const
106     {
107         ASSERT(m_typeAnnotation);
108         return *m_typeAnnotation;
109     }
110 
111     void setTypeAnnotation(TypeAnnotation&& typeAnnotation)
112     {
113         ASSERT(!m_typeAnnotation);
114         m_typeAnnotation = WTFMove(typeAnnotation);
115     }
116 
117     void copyTypeTo(Expression& other) const
118     {
119         if (auto* resolvedType = const_cast<Expression*>(this)->maybeResolvedType())
120             other.setType(*resolvedType);


121     }
122 
123     bool mayBeEffectful() const;
124 
125     Kind kind() const  { return m_kind; }
126     bool isAssignmentExpression() const { return kind() == Kind::Assignment; }
127     bool isBooleanLiteral() const { return kind() == Kind::BooleanLiteral; }
128     bool isCallExpression() const { return kind() == Kind::Call; }
129     bool isCommaExpression() const { return kind() == Kind::Comma; }
130     bool isDereferenceExpression() const { return kind() == Kind::Dereference; }
131     bool isDotExpression() const { return kind() == Kind::Dot; }
132     bool isGlobalVariableReference() const { return kind() == Kind::GlobalVariableReference; }
133     bool isFloatLiteral() const { return kind() == Kind::FloatLiteral; }
134     bool isIndexExpression() const { return kind() == Kind::Index; }
135     bool isIntegerLiteral() const { return kind() == Kind::IntegerLiteral; }
136     bool isLogicalExpression() const { return kind() == Kind::Logical; }
137     bool isLogicalNotExpression() const { return kind() == Kind::LogicalNot; }
138     bool isMakeArrayReferenceExpression() const { return kind() == Kind::MakeArrayReference; }
139     bool isMakePointerExpression() const { return kind() == Kind::MakePointer; }

140     bool isPropertyAccessExpression() const { return isDotExpression() || isIndexExpression(); }
141     bool isReadModifyWriteExpression() const { return kind() == Kind::ReadModifyWrite; }
142     bool isTernaryExpression() const { return kind() == Kind::Ternary; }
143     bool isUnsignedIntegerLiteral() const { return kind() == Kind::UnsignedIntegerLiteral; }
144     bool isVariableReference() const { return kind() == Kind::VariableReference; }
145     bool isEnumerationMemberLiteral() const { return kind() == Kind::EnumerationMemberLiteral; }
146 
147     CodeLocation codeLocation() const { return m_codeLocation; }
148     void updateCodeLocation(CodeLocation location) { m_codeLocation = location; }
149 
150 private:
151     CodeLocation m_codeLocation;
152     RefPtr<UnnamedType> m_type;
153     Optional<TypeAnnotation> m_typeAnnotation;
154     Kind m_kind;
155 };
156 
157 } // namespace AST
158 
159 }
160 
161 }
