34 namespace WebCore {
35 
36 WTF_MAKE_ISO_ALLOCATED_IMPL(RTCDTMFToneChangeEvent);
37 
38 Ref<RTCDTMFToneChangeEvent> RTCDTMFToneChangeEvent::create(const String& tone)
39 {
40     return adoptRef(*new RTCDTMFToneChangeEvent(tone));
41 }
42 
43 Ref<RTCDTMFToneChangeEvent> RTCDTMFToneChangeEvent::create(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
44 {
45     return adoptRef(*new RTCDTMFToneChangeEvent(type, initializer, isTrusted));
46 }
47 
48 RTCDTMFToneChangeEvent::RTCDTMFToneChangeEvent(const String& tone)
49     : Event(eventNames().tonechangeEvent, CanBubble::No, IsCancelable::No)
50     , m_tone(tone)
51 {
52 }
53 
54 RTCDTMFToneChangeEvent::RTCDTMFToneChangeEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
55     : Event(type, initializer, isTrusted)
56     , m_tone(initializer.tone)
57 {
58 }
59 
60 RTCDTMFToneChangeEvent::~RTCDTMFToneChangeEvent() = default;
61 
62 const String& RTCDTMFToneChangeEvent::tone() const
63 {
64     return m_tone;
65 }
66 
67 EventInterface RTCDTMFToneChangeEvent::eventInterface() const
68 {
69     return RTCDTMFToneChangeEventInterfaceType;
70 }
