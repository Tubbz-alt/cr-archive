43 // Initially these helper functions were intended to perform KEK wrapping and unwrapping,
44 // but this is not required anymore, despite the function names and the Mac implementation
45 // still indicating otherwise.
46 // See https://bugs.webkit.org/show_bug.cgi?id=173883 for more info.
47 
48 bool wrapSerializedCryptoKey(const Vector<uint8_t>& masterKey, const Vector<uint8_t>& key, Vector<uint8_t>& result)
49 {
50     UNUSED_PARAM(masterKey);
51 
52     // No wrapping performed -- the serialized key data is copied into the `result` variable.
53     result = Vector<uint8_t>(key);
54     return true;
55 }
56 
57 bool unwrapSerializedCryptoKey(const Vector<uint8_t>& masterKey, const Vector<uint8_t>& wrappedKey, Vector<uint8_t>& key)
58 {
59     UNUSED_PARAM(masterKey);
60 
61     // No unwrapping performed -- the serialized key data is copied into the `key` variable.
62     key = Vector<uint8_t>(wrappedKey);
63     return true;
64 }
