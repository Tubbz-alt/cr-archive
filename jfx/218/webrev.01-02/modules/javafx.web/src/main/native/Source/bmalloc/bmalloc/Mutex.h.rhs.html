29 #include "BAssert.h"
30 #include <atomic>
31 #include <mutex>
32 #include <thread>
33 
34 // A fast replacement for std::mutex.
35 
36 namespace bmalloc {
37 
38 class Mutex;
39 
40 using UniqueLockHolder = std::unique_lock<Mutex>;
41 using LockHolder = std::lock_guard<Mutex>;
42 
43 class Mutex {
44 public:
45     constexpr Mutex() = default;
46 
47     void lock();
48     bool try_lock();
49     void unlock();
50 
51 private:
52     BEXPORT void lockSlowCase();
53 
54     std::atomic<bool> m_flag { false };
55     std::atomic<bool> m_isSpinning { false };
56 };
57 
58 static inline void sleep(
59     UniqueLockHolder& lock, std::chrono::milliseconds duration)
60 {
61     if (duration == std::chrono::milliseconds(0))
62         return;
63 
64     lock.unlock();
65     std::this_thread::sleep_for(duration);
66     lock.lock();
67 }
68 
69 static inline void waitUntilFalse(
70     UniqueLockHolder& lock, std::chrono::milliseconds sleepDuration,
71     bool& condition)
72 {
73     while (condition) {
74         condition = false;
75         sleep(lock, sleepDuration);
76     }
77 }
78 
79 inline bool Mutex::try_lock()
80 {
81     return !m_flag.exchange(true, std::memory_order_acquire);
82 }
83 
84 inline void Mutex::lock()
85 {
86     if (!try_lock())
87         lockSlowCase();
88 }
89 
90 inline void Mutex::unlock()
91 {
92     m_flag.store(false, std::memory_order_release);
93 }
94 
95 } // namespace bmalloc
