 53 JSValue JSXMLHttpRequest::response(ExecState& state) const
 54 {
 55     auto cacheResult = [&] (JSValue value) -> JSValue {
 56         m_response.set(state.vm(), this, value);
 57         return value;
 58     };
 59 
 60 
 61     if (wrapped().responseCacheIsValid())
 62         return m_response.get();
 63 
 64     auto type = wrapped().responseType();
 65 
 66     switch (type) {
 67     case XMLHttpRequest::ResponseType::EmptyString:
 68     case XMLHttpRequest::ResponseType::Text: {
 69         auto scope = DECLARE_THROW_SCOPE(state.vm());
 70         return cacheResult(toJS<IDLNullable<IDLUSVString>>(state, scope, wrapped().responseText()));
 71     }
 72     default:
 73         break;
 74     }
 75 
 76     if (!wrapped().doneWithoutErrors())
 77         return cacheResult(jsNull());
 78 
 79     JSValue value;
 80     switch (type) {
 81     case XMLHttpRequest::ResponseType::EmptyString:
 82     case XMLHttpRequest::ResponseType::Text:
 83         ASSERT_NOT_REACHED();
 84         return jsUndefined();
 85 
 86     case XMLHttpRequest::ResponseType::Json:
 87         value = toJS<IDLJSON>(*globalObject()->globalExec(), wrapped().responseTextIgnoringResponseType());
 88         if (!value)
 89             value = jsNull();
 90         break;
 91 
 92     case XMLHttpRequest::ResponseType::Document: {
 93         auto document = wrapped().responseXML();
 94         ASSERT(!document.hasException());
 95         value = toJS<IDLInterface<Document>>(state, *globalObject(), document.releaseReturnValue());
 96         break;
 97     }
 98 
 99     case XMLHttpRequest::ResponseType::Blob:
100         value = toJSNewlyCreated<IDLInterface<Blob>>(state, *globalObject(), wrapped().createResponseBlob());
101         break;
102 
103     case XMLHttpRequest::ResponseType::Arraybuffer:
104         value = toJS<IDLInterface<ArrayBuffer>>(state, *globalObject(), wrapped().createResponseArrayBuffer());
105         break;
106     }
107 
108     wrapped().didCacheResponse();
109     return cacheResult(value);
110 }
