 44 class ShadowRoot final : public DocumentFragment, public TreeScope {
 45     WTF_MAKE_ISO_ALLOCATED(ShadowRoot);
 46 public:
 47 
 48     enum class DelegatesFocus : uint8_t { Yes, No };
 49 
 50     static Ref<ShadowRoot> create(Document& document, ShadowRootMode type, DelegatesFocus delegatesFocus = DelegatesFocus::No)
 51     {
 52         return adoptRef(*new ShadowRoot(document, type, delegatesFocus));
 53     }
 54 
 55     static Ref<ShadowRoot> create(Document& document, std::unique_ptr<SlotAssignment>&& assignment)
 56     {
 57         return adoptRef(*new ShadowRoot(document, WTFMove(assignment)));
 58     }
 59 
 60     virtual ~ShadowRoot();
 61 
 62     using TreeScope::rootNode;
 63 
 64     Style::Scope& styleScope();
 65     StyleSheetList& styleSheets();
 66 
 67     bool resetStyleInheritance() const { return m_resetStyleInheritance; }
 68     void setResetStyleInheritance(bool);
 69 
 70     bool delegatesFocus() const { return m_delegatesFocus; }
 71     bool containsFocusedElement() const { return m_containsFocusedElement; }
 72     void setContainsFocusedElement(bool flag) { m_containsFocusedElement = flag; }
 73 
 74     Element* host() const { return m_host; }
 75     void setHost(Element* host) { m_host = host; }
 76 
 77     String innerHTML() const;
 78     ExceptionOr<void> setInnerHTML(const String&);
 79 
 80     Element* activeElement() const;
 81 
 82     ShadowRootMode mode() const { return m_type; }
 83     bool shouldFireSlotchangeEvent() const { return m_type != ShadowRootMode::UserAgent && !m_hasBegunDeletingDetachedChildren; }
 84 
 85     void removeAllEventListeners() override;
 86 
 87     HTMLSlotElement* findAssignedSlot(const Node&);
 88 
 89     void renameSlotElement(HTMLSlotElement&, const AtomString& oldName, const AtomString& newName);
 90     void addSlotElementByName(const AtomString&, HTMLSlotElement&);
 91     void removeSlotElementByName(const AtomString&, HTMLSlotElement&, ContainerNode& oldParentOfRemovedTree);
 92     void slotFallbackDidChange(HTMLSlotElement&);
 93     void resolveSlotsBeforeNodeInsertionOrRemoval();
 94     void willRemoveAllChildren(ContainerNode&);
 95 
 96     void didRemoveAllChildrenOfShadowHost();
 97     void didChangeDefaultSlot();
 98     void hostChildElementDidChange(const Element&);
 99     void hostChildElementDidChangeSlotAttribute(Element&, const AtomString& oldValue, const AtomString& newValue);
100 
101     const Vector<Node*>* assignedNodesForSlot(const HTMLSlotElement&);
102 
103     void moveShadowRootToNewParentScope(TreeScope&, Document&);
104     void moveShadowRootToNewDocument(Document&);
105 
106     using PartMappings = HashMap<AtomString, Vector<AtomString, 1>>;
107     const PartMappings& partMappings() const;
108     void invalidatePartMappings();
109 
110 #if ENABLE(PICTURE_IN_PICTURE_API)
111     HTMLVideoElement* pictureInPictureElement() const;
112 #endif
113 
114 private:
115     ShadowRoot(Document&, ShadowRootMode, DelegatesFocus);
116     ShadowRoot(Document&, std::unique_ptr<SlotAssignment>&&);
117 
118     bool childTypeAllowed(NodeType) const override;
119 
120     Ref<Node> cloneNodeInternal(Document&, CloningOperation) override;
121 
122     Node::InsertedIntoAncestorResult insertedIntoAncestor(InsertionType, ContainerNode&) override;
123     void removedFromAncestor(RemovalType, ContainerNode& insertionPoint) override;
124 
125     void childrenChanged(const ChildChange&) override;
126 
127     bool m_resetStyleInheritance { false };
128     bool m_hasBegunDeletingDetachedChildren { false };
129     bool m_delegatesFocus { false };
130     bool m_containsFocusedElement { false };
131     ShadowRootMode m_type { ShadowRootMode::UserAgent };
132 
133     Element* m_host { nullptr };
134     RefPtr<StyleSheetList> m_styleSheetList;
135 
136     std::unique_ptr<Style::Scope> m_styleScope;
137     std::unique_ptr<SlotAssignment> m_slotAssignment;
138     mutable Optional<PartMappings> m_partMappings;
139 };
140 
141 inline Element* ShadowRoot::activeElement() const
142 {
143     return treeScope().focusedElementInScope();
144 }
145 
146 inline ShadowRoot* Node::shadowRoot() const
147 {
148     if (!is<Element>(*this))
149         return nullptr;
150     return downcast<Element>(*this).shadowRoot();
151 }
152 
153 inline ContainerNode* Node::parentOrShadowHostNode() const
154 {
155     ASSERT(isMainThreadOrGCThread());
156     if (is<ShadowRoot>(*this))
157         return downcast<ShadowRoot>(*this).host();
158     return parentNode();
159 }
160 
161 inline bool hasShadowRootParent(const Node& node)
162 {
163     return node.parentNode() && node.parentNode()->isShadowRoot();
164 }
