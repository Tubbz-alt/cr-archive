31 #include "NativeFunction.h"
32 #include <wtf/NakedPtr.h>
33 
34 namespace JSC {
35 
36 class ArgList;
37 class Exception;
38 class ExecState;
39 class FunctionExecutable;
40 class JSObject;
41 class JSScope;
42 
43 enum class CallType : unsigned {
44     None,
45     Host,
46     JS
47 };
48 
49 struct CallData {
50     union {
51         struct {
52             TaggedNativeFunction function;
53         } native;
54         struct {
55             FunctionExecutable* functionExecutable;
56             JSScope* scope;
57         } js;
58     };
59 };
60 
61 enum class ProfilingReason {
62     API,
63     Microtask,
64     Other
65 };
66 
67 // Convenience wrapper so you don't need to deal with CallData and CallType unless you are going to use them.
68 JS_EXPORT_PRIVATE JSValue call(ExecState*, JSValue functionObject, const ArgList&, const char* errorMessage);
69 JS_EXPORT_PRIVATE JSValue call(ExecState*, JSValue functionObject, JSValue thisValue, const ArgList&, const char* errorMessage);
70 
71 JS_EXPORT_PRIVATE JSValue call(ExecState*, JSValue functionObject, CallType, const CallData&, JSValue thisValue, const ArgList&);
72 JS_EXPORT_PRIVATE JSValue call(ExecState*, JSValue functionObject, CallType, const CallData&, JSValue thisValue, const ArgList&, NakedPtr<Exception>& returnedException);
73 
74 JS_EXPORT_PRIVATE JSValue profiledCall(ExecState*, ProfilingReason, JSValue functionObject, CallType, const CallData&, JSValue thisValue, const ArgList&);
75 JS_EXPORT_PRIVATE JSValue profiledCall(ExecState*, ProfilingReason, JSValue functionObject, CallType, const CallData&, JSValue thisValue, const ArgList&, NakedPtr<Exception>& returnedException);
76 
77 } // namespace JSC
