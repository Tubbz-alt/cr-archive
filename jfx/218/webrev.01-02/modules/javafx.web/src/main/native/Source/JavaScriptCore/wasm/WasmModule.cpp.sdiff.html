 49 static Module::ValidationResult makeValidationResult(BBQPlan& plan)
 50 {
 51     ASSERT(!plan.hasWork());
 52     if (plan.failed())
 53         return Unexpected<String>(plan.errorMessage());
 54     return Module::ValidationResult(Module::create(plan.takeModuleInformation()));
 55 }

 58 {
 59     return createSharedTask<Plan::CallbackType>([callback = WTFMove(callback)] (Plan& plan) {

 61         callback->run(makeValidationResult(static_cast<BBQPlan&>(plan)));
 62     });

 78 Ref<CodeBlock> Module::getOrCreateCodeBlock(Context* context, MemoryMode mode, CreateEmbedderWrapper&& createEmbedderWrapper, ThrowWasmException throwWasmException)
 79 {
 80     RefPtr<CodeBlock> codeBlock;
 81     auto locker = holdLock(m_lock);
 82     codeBlock = m_codeBlocks[static_cast<uint8_t>(mode)];
 83     // If a previous attempt at a compile errored out, let's try again.
 84     // Compilations from valid modules can fail because OOM and cancellation.
 85     // It's worth retrying.
 86     // FIXME: We might want to back off retrying at some point:
 87     // https://bugs.webkit.org/show_bug.cgi?id=170607
 88     if (!codeBlock || (codeBlock->compilationFinished() && !codeBlock->runnable())) {
 89         codeBlock = CodeBlock::create(context, mode, const_cast<ModuleInformation&>(moduleInformation()), WTFMove(createEmbedderWrapper), throwWasmException);



 90         m_codeBlocks[static_cast<uint8_t>(mode)] = codeBlock;
 91     }
 92     return codeBlock.releaseNonNull();
 93 }
 94 
 95 Ref<CodeBlock> Module::compileSync(Context* context, MemoryMode mode, CreateEmbedderWrapper&& createEmbedderWrapper, ThrowWasmException throwWasmException)
 96 {
 97     Ref<CodeBlock> codeBlock = getOrCreateCodeBlock(context, mode, WTFMove(createEmbedderWrapper), throwWasmException);
 98     codeBlock->waitUntilFinished();
 99     return codeBlock;
100 }

102 void Module::compileAsync(Context* context, MemoryMode mode, CodeBlock::AsyncCompilationCallback&& task, CreateEmbedderWrapper&& createEmbedderWrapper, ThrowWasmException throwWasmException)
103 {
104     Ref<CodeBlock> codeBlock = getOrCreateCodeBlock(context, mode, WTFMove(createEmbedderWrapper), throwWasmException);
105     codeBlock->compileAsync(context, WTFMove(task));

 82 Ref<CodeBlock> Module::getOrCreateCodeBlock(Context* context, MemoryMode mode)
 83 {
 84     RefPtr<CodeBlock> codeBlock;
 85     auto locker = holdLock(m_lock);
 86     codeBlock = m_codeBlocks[static_cast<uint8_t>(mode)];
 87     // If a previous attempt at a compile errored out, let's try again.
 88     // Compilations from valid modules can fail because OOM and cancellation.
 89     // It's worth retrying.
 90     // FIXME: We might want to back off retrying at some point:
 91     // https://bugs.webkit.org/show_bug.cgi?id=170607
 92     if (!codeBlock || (codeBlock->compilationFinished() && !codeBlock->runnable())) {
 93         RefPtr<LLIntCallees> llintCallees = nullptr;
 94         if (Options::useWasmLLInt())
 95             llintCallees = m_llintCallees;
 96         codeBlock = CodeBlock::create(context, mode, const_cast<ModuleInformation&>(moduleInformation()), llintCallees);
 97         m_codeBlocks[static_cast<uint8_t>(mode)] = codeBlock;
 98     }
 99     return codeBlock.releaseNonNull();
100 }

102 Ref<CodeBlock> Module::compileSync(Context* context, MemoryMode mode)
103 {
104     Ref<CodeBlock> codeBlock = getOrCreateCodeBlock(context, mode);
105     codeBlock->waitUntilFinished();
106     return codeBlock;
107 }
