28 #include <wtf/HashTraits.h>
29 #include <wtf/OptionSet.h>
30 
31 namespace WTF {
32 
33 template<typename T> struct DefaultHash<OptionSet<T>> {
34     struct Hash {
35         static unsigned hash(OptionSet<T> key)
36         {
37             return IntHash<typename OptionSet<T>::StorageType>::hash(key.toRaw());
38         }
39 
40         static bool equal(OptionSet<T> a, OptionSet<T> b)
41         {
42             return a == b;
43         }
44 
45         static constexpr bool safeToCompareToEmptyOrDeleted = true;
46     };
47 };
48 
49 template<typename T> struct HashTraits<OptionSet<T>> : GenericHashTraits<OptionSet<T>> {
50     using StorageTraits = UnsignedWithZeroKeyHashTraits<typename OptionSet<T>::StorageType>;
51 
52     static OptionSet<T> emptyValue()
53     {
54         return OptionSet<T>::fromRaw(StorageTraits::emptyValue());
55     }
56 
57     static void constructDeletedValue(OptionSet<T>& slot)
58     {
59         typename OptionSet<T>::StorageType storage;
60         StorageTraits::constructDeletedValue(storage);
61         slot = OptionSet<T>::fromRaw(storage);
62     }
63 
64     static bool isDeletedValue(OptionSet<T> value)
65     {
66         return StorageTraits::isDeletedValue(value.toRaw());
67     }
68 };

70 } // namespace WTF
