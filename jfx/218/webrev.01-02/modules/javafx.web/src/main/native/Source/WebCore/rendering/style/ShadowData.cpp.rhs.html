 30 ShadowData::ShadowData(const ShadowData& o)
 31     : m_location(o.m_location)

 32     , m_spread(o.m_spread)
 33     , m_radius(o.m_radius)
 34     , m_color(o.m_color)
 35     , m_style(o.m_style)
 36     , m_isWebkitBoxShadow(o.m_isWebkitBoxShadow)
 37     , m_next(o.m_next ? makeUnique<ShadowData>(*o.m_next) : nullptr)
 38 {
 39 }

 41 Optional<ShadowData> ShadowData::clone(const ShadowData* data)
 42 {
 43     if (!data)
 44         return WTF::nullopt;
 45     return *data;
 46 }
 47 
 48 bool ShadowData::operator==(const ShadowData& o) const
 49 {
 50     if (!arePointingToEqualData(m_next, o.m_next))
 51         return false;
 52 
 53     return m_location == o.m_location
 54         && m_radius == o.m_radius
 55         && m_spread == o.m_spread
 56         && m_style == o.m_style
 57         && m_color == o.m_color
 58         && m_isWebkitBoxShadow == o.m_isWebkitBoxShadow;
 59 }

 61 static inline void calculateShadowExtent(const ShadowData* shadow, LayoutUnit additionalOutlineSize, LayoutUnit& shadowLeft, LayoutUnit& shadowRight, LayoutUnit& shadowTop, LayoutUnit& shadowBottom)
 62 {
 63     do {
 64         LayoutUnit extentAndSpread = shadow->paintingExtent() + shadow->spread() + additionalOutlineSize;
 65         if (shadow->style() == ShadowStyle::Normal) {
 66             shadowLeft = std::min(shadow->x() - extentAndSpread, shadowLeft);
 67             shadowRight = std::max(shadow->x() + extentAndSpread, shadowRight);
 68             shadowTop = std::min(shadow->y() - extentAndSpread, shadowTop);
 69             shadowBottom = std::max(shadow->y() + extentAndSpread, shadowBottom);
 70         }
 71 
 72         shadow = shadow->next();
 73     } while (shadow);
 74 }
 75 
 76 void ShadowData::adjustRectForShadow(LayoutRect& rect, int additionalOutlineSize) const
 77 {
 78     LayoutUnit shadowLeft;
 79     LayoutUnit shadowRight;
 80     LayoutUnit shadowTop;
 81     LayoutUnit shadowBottom;
 82     calculateShadowExtent(this, additionalOutlineSize, shadowLeft, shadowRight, shadowTop, shadowBottom);
 83 
 84     rect.move(shadowLeft, shadowTop);
 85     rect.setWidth(rect.width() - shadowLeft + shadowRight);
 86     rect.setHeight(rect.height() - shadowTop + shadowBottom);
 87 }
 88 
 89 void ShadowData::adjustRectForShadow(FloatRect& rect, int additionalOutlineSize) const
 90 {
 91     LayoutUnit shadowLeft = 0;
 92     LayoutUnit shadowRight = 0;
 93     LayoutUnit shadowTop = 0;
 94     LayoutUnit shadowBottom = 0;
 95     calculateShadowExtent(this, additionalOutlineSize, shadowLeft, shadowRight, shadowTop, shadowBottom);
 96 
 97     rect.move(shadowLeft, shadowTop);
 98     rect.setWidth(rect.width() - shadowLeft + shadowRight);
 99     rect.setHeight(rect.height() - shadowTop + shadowBottom);
100 }

102 TextStream& operator<<(TextStream& ts, const ShadowData& data)
103 {
104     ts.dumpProperty("location", data.location());
105     ts.dumpProperty("radius", data.radius());
106     ts.dumpProperty("spread", data.spread());
107     ts.dumpProperty("color", data.color());
108 
109     return ts;
110 }
