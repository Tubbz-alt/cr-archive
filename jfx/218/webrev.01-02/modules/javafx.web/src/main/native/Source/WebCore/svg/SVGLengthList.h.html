 26 #pragma once
 27 
 28 #include "SVGLength.h"
 29 #include "SVGValuePropertyList.h"
 30 #include <wtf/text/StringBuilder.h>
 31 
 32 namespace WebCore {
 33 
 34 class SVGLengthList : public SVGValuePropertyList<SVGLength> {
 35     using Base = SVGValuePropertyList<SVGLength>;
 36     using Base::Base;
 37 
 38 public:
 39     static Ref<SVGLengthList> create(SVGLengthMode lengthMode = SVGLengthMode::Other)
 40     {
 41         return adoptRef(*new SVGLengthList(lengthMode));
 42     }
 43 
 44     static Ref<SVGLengthList> create(SVGPropertyOwner* owner, SVGPropertyAccess access, SVGLengthMode lengthMode)
 45     {
 46         return adoptRef(*new SVGLengthList(owner, access, lengthMode));
 47     }
 48 
 49     static Ref<SVGLengthList> create(const SVGLengthList& other, SVGPropertyAccess access)
 50     {
 51         return adoptRef(*new SVGLengthList(other, access));
 52     }
 53 
 54     SVGLengthMode lengthMode() const { return m_lengthMode; }
 55 
 56     bool parse(const String& value)
 57     {
 58         clearItems();
 59 
 60         auto upconvertedCharacters = StringView(value).upconvertedCharacters();
 61         const UChar* ptr = upconvertedCharacters;
 62         const UChar* end = ptr + value.length();
 63         while (ptr < end) {
 64             const UChar* start = ptr;
 65             while (ptr < end && *ptr != ',' && !isSVGSpace(*ptr))
 66                 ptr++;
 67             if (ptr == start)
 68                 break;
 69 
 70             String valueString(start, ptr - start);
 71             SVGLengthValue value(m_lengthMode);
 72             if (value.setValueAsString(valueString).hasException())
 73                 break;
 74 
 75             append(SVGLength::create(value));
 76             skipOptionalSVGSpacesOrDelimiter(ptr, end);
 77         }
 78 
 79         return ptr == end;
 80     }
 81 
 82     String valueAsString() const override
 83     {
 84         StringBuilder builder;
 85 
 86         for (const auto& length : m_items) {
 87             if (builder.length())
 88                 builder.append(' ');
 89 
 90             builder.append(length->value().valueAsString());
 91         }
 92 
 93         return builder.toString();
 94     }
 95 
 96 private:
 97     SVGLengthList(SVGLengthMode lengthMode)
 98         : m_lengthMode(lengthMode)
 99     {
100     }
101 
102     SVGLengthList(SVGPropertyOwner* owner, SVGPropertyAccess access, SVGLengthMode lengthMode)
103         : Base(owner, access)
104         , m_lengthMode(lengthMode)
105     {
106     }
107 
108     SVGLengthList(const SVGLengthList& other, SVGPropertyAccess access)
109         : Base(other, access)
110         , m_lengthMode(other.lengthMode())
111     {
112     }
113 
114     SVGLengthMode m_lengthMode { SVGLengthMode::Other };
115 };
116 
117 } // namespace WebCore
