26 #pragma once
27 
28 #include "ProcessIdentifier.h"
29 #include <wtf/CompletionHandler.h>
30 #include <wtf/Vector.h>
31 
32 namespace WebCore {
33 
34 class ScriptExecutionContext;
35 struct MessagePortIdentifier;
36 struct MessageWithMessagePorts;
37 
38 class MessagePortChannelProvider {
39 public:
40     static MessagePortChannelProvider& fromContext(ScriptExecutionContext&);
41     static MessagePortChannelProvider& singleton();
42     WEBCORE_EXPORT static void setSharedProvider(MessagePortChannelProvider&);
43 
44     virtual ~MessagePortChannelProvider() { }
45 
46     // Operations that WebProcesses perform
47     virtual void createNewMessagePortChannel(const MessagePortIdentifier& local, const MessagePortIdentifier& remote) = 0;
48     virtual void entangleLocalPortInThisProcessToRemote(const MessagePortIdentifier& local, const MessagePortIdentifier& remote) = 0;
49     virtual void messagePortDisentangled(const MessagePortIdentifier& local) = 0;
50     virtual void messagePortClosed(const MessagePortIdentifier& local) = 0;
51 
52     virtual void takeAllMessagesForPort(const MessagePortIdentifier&, CompletionHandler<void(Vector<MessageWithMessagePorts>&&, Function<void()>&&)>&&) = 0;
53 
54     virtual void postMessageToRemote(MessageWithMessagePorts&&, const MessagePortIdentifier& remoteTarget) = 0;
55 
56     enum class HasActivity {
57         Yes,
58         No,
59     };
60     virtual void checkRemotePortForActivity(const MessagePortIdentifier& remoteTarget, CompletionHandler<void(HasActivity)>&& callback) = 0;
61 
62 private:
63 
64 };
65 
66 } // namespace WebCore
67 
68 namespace WTF {
69 
70 template<> struct EnumTraits<WebCore::MessagePortChannelProvider::HasActivity> {
71     using values = EnumValues<
72         WebCore::MessagePortChannelProvider::HasActivity,
73         WebCore::MessagePortChannelProvider::HasActivity::No,
74         WebCore::MessagePortChannelProvider::HasActivity::Yes
75     >;
76 };
