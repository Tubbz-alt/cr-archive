31 find_path(LIBPSL_INCLUDE_DIRS




































32     NAMES libpsl.h

33     PATH_SUFFIXES libpsl
34 )
35 
36 find_library(LIBPSL_LIBRARIES
37     NAMES psl

38 )
39 
40 if (LIBPSL_INCLUDE_DIRS)
41     if (EXISTS "${LIBPSL_INCLUDE_DIRS}/libpsl.h")
42         file(READ "${LIBPSL_INCLUDE_DIRS}/libpsl.h" LIBPSL_VERSION_CONTENT)
43 
44         string(REGEX MATCH "#define +PSL_VERSION_MAJOR +([0-9]+)" _dummy "${LIBPSL_VERSION_CONTENT}")
45         set(LIBPSL_VERSION_MAJOR "${CMAKE_MATCH_1}")
46 
47         string(REGEX MATCH "#define +PSL_VERSION_MINOR +([0-9]+)" _dummy "${LIBPSL_VERSION_CONTENT}")
48         set(LIBPSL_VERSION_MINOR "${CMAKE_MATCH_1}")
49 
50         string(REGEX MATCH "#define +PSL_VERSION_PATCH +([0-9]+)" _dummy "${LIBPSL_VERSION_CONTENT}")
51         set(LIBPSL_VERSION_PATCH "${CMAKE_MATCH_1}")
52 
53         set(LIBPSL_VERSION "${LIBPSL_VERSION_MAJOR}.${LIBPSL_VERSION_MINOR}.${LIBPSL_VERSION_PATCH}")
54     endif ()
55 endif ()
