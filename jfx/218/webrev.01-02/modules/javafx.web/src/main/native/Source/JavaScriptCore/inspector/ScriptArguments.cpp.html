 42 Ref<ScriptArguments> ScriptArguments::create(JSC::JSGlobalObject* globalObject, Vector<JSC::Strong<JSC::Unknown>>&& arguments)
 43 {
 44     return adoptRef(*new ScriptArguments(globalObject, WTFMove(arguments)));
 45 }

 47 ScriptArguments::ScriptArguments(JSC::JSGlobalObject* globalObject, Vector<JSC::Strong<JSC::Unknown>>&& arguments)
 48     : m_globalObject(globalObject->vm(), globalObject)
 49     , m_arguments(WTFMove(arguments))
 50 {
 51 }

 53 ScriptArguments::~ScriptArguments() = default;

 55 JSC::JSValue ScriptArguments::argumentAt(size_t index) const
 56 {
 57     ASSERT(m_arguments.size() > index);
 58     return m_arguments[index].get();
 59 }
 60 
 61 JSC::JSGlobalObject* ScriptArguments::globalObject() const
 62 {
 63     return m_globalObject.get();
 64 }
 65 
 66 bool ScriptArguments::getFirstArgumentAsString(String& result) const
 67 {
 68     if (!argumentCount())
 69         return false;
 70 
 71     auto* globalObject = this->globalObject();
 72     if (!globalObject) {
 73         ASSERT_NOT_REACHED();
 74         return false;
 75     }
 76 
 77     auto value = argumentAt(0);
 78     if (JSC::jsDynamicCast<JSC::ProxyObject*>(globalObject->vm(), value)) {
 79         result = "[object Proxy]"_s;
 80         return true;
 81     }
 82 
 83     auto scope = DECLARE_CATCH_SCOPE(globalObject->vm());
 84     result = value.toWTFString(globalObject);
 85     scope.clearException();
 86     return true;
 87 }
 88 
 89 bool ScriptArguments::isEqual(const ScriptArguments& other) const
 90 {
 91     auto size = m_arguments.size();
 92 
 93     if (size != other.m_arguments.size())
 94         return false;
 95 
 96     if (!size)
 97         return true;
 98 
 99     auto* globalObject = this->globalObject();
100     if (!globalObject)
101         return false;
102 
103     for (size_t i = 0; i < size; ++i) {
104         auto a = m_arguments[i].get();
105         auto b = other.m_arguments[i].get();
106         if (!a || !b) {
107             if (a != b)
108                 return false;
109         } else {
110             auto scope = DECLARE_CATCH_SCOPE(globalObject->vm());
111             bool result = JSC::JSValue::strictEqual(globalObject, a, b);
112             scope.clearException();
113             if (!result)
114                 return false;
115         }
116     }
117 
118     return true;
119 }
