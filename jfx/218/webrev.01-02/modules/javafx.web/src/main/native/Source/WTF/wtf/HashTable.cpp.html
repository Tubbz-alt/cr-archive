31 #if DUMP_HASHTABLE_STATS
32 
33 std::atomic<unsigned> HashTableStats::numAccesses;
34 std::atomic<unsigned> HashTableStats::numRehashes;
35 std::atomic<unsigned> HashTableStats::numRemoves;
36 std::atomic<unsigned> HashTableStats::numReinserts;
37 
38 unsigned HashTableStats::numCollisions;
39 unsigned HashTableStats::collisionGraph[4096];
40 unsigned HashTableStats::maxCollisions;
41 
42 static Lock hashTableStatsMutex;
43 
44 void HashTableStats::recordCollisionAtCount(unsigned count)
45 {
46     std::lock_guard<Lock> lock(hashTableStatsMutex);
47 
48     if (count > maxCollisions)
49         maxCollisions = count;
50     numCollisions++;
51     collisionGraph[count]++;
52 }
53 
54 void HashTableStats::dumpStats()
55 {
56     std::lock_guard<Lock> lock(hashTableStatsMutex);
57 
58     dataLogF("\nWTF::HashTable statistics\n\n");
59     dataLogF("%u accesses\n", numAccesses.load());
60     dataLogF("%d total collisions, average %.2f probes per access\n", numCollisions, 1.0 * (numAccesses + numCollisions) / numAccesses);
61     dataLogF("longest collision chain: %d\n", maxCollisions);
62     for (unsigned i = 1; i <= maxCollisions; i++) {
63         dataLogF("  %u lookups with exactly %u collisions (%.2f%% , %.2f%% with this many or more)\n", collisionGraph[i], i, 100.0 * (collisionGraph[i] - collisionGraph[i+1]) / numAccesses, 100.0 * collisionGraph[i] / numAccesses);
64     }
65     dataLogF("%d rehashes\n", numRehashes.load());
66     dataLogF("%d reinserts\n", numReinserts.load());
67 }
68 
69 #endif
