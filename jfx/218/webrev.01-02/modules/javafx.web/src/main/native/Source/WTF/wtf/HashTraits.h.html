 39 template<typename T> struct GenericHashTraitsBase<false, T> {
 40     // The emptyValueIsZero flag is used to optimize allocation of empty hash tables with zeroed memory.
 41     static constexpr bool emptyValueIsZero = false;
 42 
 43     // The hasIsEmptyValueFunction flag allows the hash table to automatically generate code to check
 44     // for the empty value when it can be done with the equality operator, but allows custom functions
 45     // for cases like String that need them.
 46     static constexpr bool hasIsEmptyValueFunction = false;
 47 
 48     // Used by WeakPtr to indicate that the value may become deleted without being explicitly removed.
 49     static constexpr bool hasIsReleasedWeakValueFunction = false;
 50 
 51     // The starting table size. Can be overridden when we know beforehand that
 52     // a hash table will have at least N entries.
 53     static constexpr unsigned minimumTableSize = 8;
 54 };
 55 
 56 // Default integer traits disallow both 0 and -1 as keys (max value instead of -1 for unsigned).
 57 template<typename T> struct GenericHashTraitsBase<true, T> : GenericHashTraitsBase<false, T> {
 58     static constexpr bool emptyValueIsZero = true;
 59     static void constructDeletedValue(T& slot) { slot = static_cast<T>(-1); }
 60     static bool isDeletedValue(T value) { return value == static_cast<T>(-1); }
 61 };

 63 template<typename T> struct GenericHashTraits : GenericHashTraitsBase<std::is_integral<T>::value, T> {
 64     typedef T TraitType;
 65     typedef T EmptyValueType;
 66 
 67     static T emptyValue() { return T(); }
 68 
 69     template<typename U, typename V>
 70     static void assignToEmpty(U& emptyValue, V&& value)
 71     {
 72         emptyValue = std::forward<V>(value);
 73     }
 74 
 75     template <typename Traits>
 76     static void constructEmptyValue(T& slot)
 77     {
 78         new (NotNull, std::addressof(slot)) T(Traits::emptyValue());
 79     }
 80 
 81     // Type for return value of functions that do not transfer ownership, such as get.
 82     typedef T PeekType;
 83     template<typename U> static U&& peek(U&& value) { return std::forward<U>(value); }
 84 
 85     typedef T TakeType;
 86     template<typename U> static TakeType take(U&& value) { return std::forward<U>(value); }
 87 };

146 template<typename T, typename Deleter> struct HashTraits<std::unique_ptr<T, Deleter>> : SimpleClassHashTraits<std::unique_ptr<T, Deleter>> {
147     typedef std::nullptr_t EmptyValueType;
148     static EmptyValueType emptyValue() { return nullptr; }
149 
150     static void constructDeletedValue(std::unique_ptr<T, Deleter>& slot) { new (NotNull, std::addressof(slot)) std::unique_ptr<T, Deleter> { reinterpret_cast<T*>(-1) }; }
151     static bool isDeletedValue(const std::unique_ptr<T, Deleter>& value) { return value.get() == reinterpret_cast<T*>(-1); }
152 
153     typedef T* PeekType;
154     static T* peek(const std::unique_ptr<T, Deleter>& value) { return value.get(); }
155     static T* peek(std::nullptr_t) { return nullptr; }
156 
157     static void customDeleteBucket(std::unique_ptr<T, Deleter>& value)
158     {
159         // The custom delete function exists to avoid a dead store before the value is destructed.
160         // The normal destruction sequence of a bucket would be:
161         // 1) Call the destructor of unique_ptr.
162         // 2) unique_ptr store a zero for its internal pointer.
163         // 3) unique_ptr destroys its value.
164         // 4) Call constructDeletedValue() to set the bucket as destructed.
165         //
166         // The problem is the call in (3) prevents the compile from eliminating the dead store in (2)
167         // becase a side effect of free() could be observing the value.
168         //
169         // This version of deleteBucket() ensures the dead 2 stores changing "value"
170         // are on the same side of the function call.
171         ASSERT(!isDeletedValue(value));
172         T* pointer = value.release();
173         constructDeletedValue(value);
174 
175         // The null case happens if a caller uses std::move() to remove the pointer before calling remove()
176         // with an iterator. This is very uncommon.
177         if (LIKELY(pointer))
178             Deleter()(pointer);
179     }
180 };

182 template<typename P> struct HashTraits<RefPtr<P>> : SimpleClassHashTraits<RefPtr<P>> {
183     static P* emptyValue() { return nullptr; }
184 
185     typedef P* PeekType;
186     static PeekType peek(const RefPtr<P>& value) { return value.get(); }
187     static PeekType peek(P* value) { return value; }
188 
189     static void customDeleteBucket(RefPtr<P>& value)
190     {
191         // See unique_ptr's customDeleteBucket() for an explanation.
192         ASSERT(!SimpleClassHashTraits<RefPtr<P>>::isDeletedValue(value));
193         auto valueToBeDestroyed = WTFMove(value);
194         SimpleClassHashTraits<RefPtr<P>>::constructDeletedValue(value);
195     }
196 };

198 template<typename P> struct RefHashTraits : SimpleClassHashTraits<Ref<P>> {
199     static constexpr bool emptyValueIsZero = true;
200     static Ref<P> emptyValue() { return HashTableEmptyValue; }
201 
202     template <typename>
203     static void constructEmptyValue(Ref<P>& slot)
204     {
205         new (NotNull, std::addressof(slot)) Ref<P>(HashTableEmptyValue);
206     }
207 
208     static constexpr bool hasIsEmptyValueFunction = true;
209     static bool isEmptyValue(const Ref<P>& value) { return value.isHashTableEmptyValue(); }
210 
211     static void assignToEmpty(Ref<P>& emptyValue, Ref<P>&& newValue) { ASSERT(isEmptyValue(emptyValue)); emptyValue.assignToHashTableEmptyValue(WTFMove(newValue)); }
212 
213     typedef P* PeekType;
214     static PeekType peek(const Ref<P>& value) { return const_cast<PeekType>(value.ptrAllowingHashTableEmptyValue()); }
215     static PeekType peek(P* value) { return value; }
216 
217     typedef Optional<Ref<P>> TakeType;
218     static TakeType take(Ref<P>&& value) { return isEmptyValue(value) ? WTF::nullopt : Optional<Ref<P>>(WTFMove(value)); }
219 };

243 // This struct template is an implementation detail of the isHashTraitsEmptyValue function,
244 // which selects either the emptyValue function or the isEmptyValue function to check for empty values.
245 template<typename Traits, bool hasEmptyValueFunction> struct HashTraitsEmptyValueChecker;
246 template<typename Traits> struct HashTraitsEmptyValueChecker<Traits, true> {
247     template<typename T> static bool isEmptyValue(const T& value) { return Traits::isEmptyValue(value); }
248 };
249 template<typename Traits> struct HashTraitsEmptyValueChecker<Traits, false> {
250     template<typename T> static bool isEmptyValue(const T& value) { return value == Traits::emptyValue(); }
251 };
252 template<typename Traits, typename T> inline bool isHashTraitsEmptyValue(const T& value)
253 {
254     return HashTraitsEmptyValueChecker<Traits, Traits::hasIsEmptyValueFunction>::isEmptyValue(value);
255 }

257 template<typename Traits, bool hasIsReleasedWeakValueFunction> struct HashTraitsReleasedWeakValueChecker;
258 template<typename Traits> struct HashTraitsReleasedWeakValueChecker<Traits, true> {
259     template<typename T> static bool isReleasedWeakValue(const T& value) { return Traits::isReleasedWeakValue(value); }
260 };
261 template<typename Traits> struct HashTraitsReleasedWeakValueChecker<Traits, false> {
262     template<typename T> static bool isReleasedWeakValue(const T&) { return false; }
263 };
264 template<typename Traits, typename T> inline bool isHashTraitsReleasedWeakValue(const T& value)

266     return HashTraitsReleasedWeakValueChecker<Traits, Traits::hasIsReleasedWeakValueFunction>::isReleasedWeakValue(value);

269 template<typename Traits, typename T>
270 struct HashTraitHasCustomDelete {
271     static T& bucketArg;
272     template<typename X> static std::true_type TestHasCustomDelete(X*, decltype(X::customDeleteBucket(bucketArg))* = nullptr);
273     static std::false_type TestHasCustomDelete(...);
274     typedef decltype(TestHasCustomDelete(static_cast<Traits*>(nullptr))) ResultType;
275     static constexpr bool value = ResultType::value;
276 };
277 
278 template<typename Traits, typename T>
279 typename std::enable_if<HashTraitHasCustomDelete<Traits, T>::value>::type
280 hashTraitsDeleteBucket(T& value)
281 {
282     Traits::customDeleteBucket(value);
283 }
284 
285 template<typename Traits, typename T>
286 typename std::enable_if<!HashTraitHasCustomDelete<Traits, T>::value>::type
287 hashTraitsDeleteBucket(T& value)
288 {
289     value.~T();
290     Traits::constructDeletedValue(value);
291 }

293 template<typename FirstTraitsArg, typename SecondTraitsArg>
294 struct PairHashTraits : GenericHashTraits<std::pair<typename FirstTraitsArg::TraitType, typename SecondTraitsArg::TraitType>> {
295     typedef FirstTraitsArg FirstTraits;
296     typedef SecondTraitsArg SecondTraits;
297     typedef std::pair<typename FirstTraits::TraitType, typename SecondTraits::TraitType> TraitType;
298     typedef std::pair<typename FirstTraits::EmptyValueType, typename SecondTraits::EmptyValueType> EmptyValueType;
299 
300     static constexpr bool emptyValueIsZero = FirstTraits::emptyValueIsZero && SecondTraits::emptyValueIsZero;
301     static EmptyValueType emptyValue() { return std::make_pair(FirstTraits::emptyValue(), SecondTraits::emptyValue()); }
302 
303     static constexpr unsigned minimumTableSize = FirstTraits::minimumTableSize;
304 
305     static void constructDeletedValue(TraitType& slot) { FirstTraits::constructDeletedValue(slot.first); }

312 template<typename FirstTrait, typename... Traits>
313 struct TupleHashTraits : GenericHashTraits<std::tuple<typename FirstTrait::TraitType, typename Traits::TraitType...>> {
314     typedef std::tuple<typename FirstTrait::TraitType, typename Traits::TraitType...> TraitType;
315     typedef std::tuple<typename FirstTrait::EmptyValueType, typename Traits::EmptyValueType...> EmptyValueType;
316 
317     // We should use emptyValueIsZero = Traits::emptyValueIsZero &&... whenever we switch to C++17. We can't do anything
318     // better here right now because GCC can't do C++.
319     template<typename BoolType>
320     static constexpr bool allTrue(BoolType value) { return value; }
321     template<typename BoolType, typename... BoolTypes>
322     static constexpr bool allTrue(BoolType value, BoolTypes... values) { return value && allTrue(values...); }
323     static constexpr bool emptyValueIsZero = allTrue(FirstTrait::emptyValueIsZero, Traits::emptyValueIsZero...);
324     static EmptyValueType emptyValue() { return std::make_tuple(FirstTrait::emptyValue(), Traits::emptyValue()...); }
325 
326     static constexpr unsigned minimumTableSize = FirstTrait::minimumTableSize;
327 
328     static void constructDeletedValue(TraitType& slot) { FirstTrait::constructDeletedValue(std::get<0>(slot)); }
329     static bool isDeletedValue(const TraitType& value) { return FirstTrait::isDeletedValue(std::get<0>(value)); }
330 };

335 template<typename KeyTraitsArg, typename ValueTraitsArg>
336 struct KeyValuePairHashTraits : GenericHashTraits<KeyValuePair<typename KeyTraitsArg::TraitType, typename ValueTraitsArg::TraitType>> {
337     typedef KeyTraitsArg KeyTraits;
338     typedef ValueTraitsArg ValueTraits;
339     typedef KeyValuePair<typename KeyTraits::TraitType, typename ValueTraits::TraitType> TraitType;
340     typedef KeyValuePair<typename KeyTraits::EmptyValueType, typename ValueTraits::EmptyValueType> EmptyValueType;
341     typedef typename ValueTraitsArg::TraitType ValueType;
342 
343     static constexpr bool emptyValueIsZero = KeyTraits::emptyValueIsZero && ValueTraits::emptyValueIsZero;
344     static EmptyValueType emptyValue() { return KeyValuePair<typename KeyTraits::EmptyValueType, typename ValueTraits::EmptyValueType>(KeyTraits::emptyValue(), ValueTraits::emptyValue()); }
345 
346     template <typename>
347     static void constructEmptyValue(TraitType& slot)
348     {
349         KeyTraits::template constructEmptyValue<KeyTraits>(slot.key);
350         ValueTraits::template constructEmptyValue<ValueTraits>(slot.value);
351     }
352 
353     static constexpr unsigned minimumTableSize = KeyTraits::minimumTableSize;
354 
355     static void constructDeletedValue(TraitType& slot) { KeyTraits::constructDeletedValue(slot.key); }
356     static bool isDeletedValue(const TraitType& value) { return KeyTraits::isDeletedValue(value.key); }
357 
358     static void customDeleteBucket(TraitType& value)
359     {
360         static_assert(std::is_trivially_destructible<KeyValuePair<int, int>>::value,
361             "The wrapper itself has to be trivially destructible for customDeleteBucket() to make sense, since we do not destruct the wrapper itself.");
362 
363         hashTraitsDeleteBucket<KeyTraits>(value.key);
364         value.value.~ValueType();
365     }
366 };

385 // Useful for classes that want complete control over what is empty and what is deleted,
386 // and how to construct both.
387 template<typename T>
388 struct CustomHashTraits : public GenericHashTraits<T> {
389     static constexpr bool emptyValueIsZero = false;
390     static constexpr bool hasIsEmptyValueFunction = true;
391 
392     static void constructDeletedValue(T& slot)
393     {
394         new (NotNull, std::addressof(slot)) T(T::DeletedValue);
395     }
396 
397     static bool isDeletedValue(const T& value)
398     {
399         return value.isDeletedValue();
400     }
401 
402     static T emptyValue()
403     {
404         return T(T::EmptyValue);
405     }
406 
407     static bool isEmptyValue(const T& value)
408     {
409         return value.isEmptyValue();
410     }
411 };
