40 class ConsoleClient {
41 public:
42     virtual ~ConsoleClient() { }
43 
44     JS_EXPORT_PRIVATE static void printConsoleMessage(MessageSource, MessageType, MessageLevel, const String& message, const String& url, unsigned lineNumber, unsigned columnNumber);
45     JS_EXPORT_PRIVATE static void printConsoleMessageWithArguments(MessageSource, MessageType, MessageLevel, JSC::JSGlobalObject*, Ref<Inspector::ScriptArguments>&&);
46 
47     void logWithLevel(JSGlobalObject*, Ref<Inspector::ScriptArguments>&&, MessageLevel);
48     void clear(JSGlobalObject*);
49     void dir(JSGlobalObject*, Ref<Inspector::ScriptArguments>&&);
50     void dirXML(JSGlobalObject*, Ref<Inspector::ScriptArguments>&&);
51     void table(JSGlobalObject*, Ref<Inspector::ScriptArguments>&&);
52     void trace(JSGlobalObject*, Ref<Inspector::ScriptArguments>&&);
53     void assertion(JSGlobalObject*, Ref<Inspector::ScriptArguments>&&);
54     void group(JSGlobalObject*, Ref<Inspector::ScriptArguments>&&);
55     void groupCollapsed(JSGlobalObject*, Ref<Inspector::ScriptArguments>&&);
56     void groupEnd(JSGlobalObject*, Ref<Inspector::ScriptArguments>&&);
57 
58     virtual void messageWithTypeAndLevel(MessageType, MessageLevel, JSC::JSGlobalObject*, Ref<Inspector::ScriptArguments>&&) = 0;
59     virtual void count(JSGlobalObject*, const String& label) = 0;
60     virtual void countReset(JSGlobalObject*, const String& label) = 0;
61     virtual void profile(JSGlobalObject*, const String& title) = 0;
62     virtual void profileEnd(JSGlobalObject*, const String& title) = 0;
63     virtual void takeHeapSnapshot(JSGlobalObject*, const String& title) = 0;
64     virtual void time(JSGlobalObject*, const String& label) = 0;
65     virtual void timeLog(JSGlobalObject*, const String& label, Ref<Inspector::ScriptArguments>&&) = 0;
66     virtual void timeEnd(JSGlobalObject*, const String& label) = 0;
67     virtual void timeStamp(JSGlobalObject*, Ref<Inspector::ScriptArguments>&&) = 0;
68     virtual void record(JSGlobalObject*, Ref<Inspector::ScriptArguments>&&) = 0;
69     virtual void recordEnd(JSGlobalObject*, Ref<Inspector::ScriptArguments>&&) = 0;
70     virtual void screenshot(JSGlobalObject*, Ref<Inspector::ScriptArguments>&&) = 0;
71 
72 private:
73     enum ArgumentRequirement { ArgumentRequired, ArgumentNotRequired };
74     void internalMessageWithTypeAndLevel(MessageType, MessageLevel, JSC::JSGlobalObject*, Ref<Inspector::ScriptArguments>&&, ArgumentRequirement);
75 };
