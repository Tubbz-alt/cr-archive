26 function next()
27 {
28     "use strict";
29 
30     if (!@isStringIterator(this))
31         @throwTypeError("%StringIteratorPrototype%.next requires that |this| be a String Iterator instance");
32 
33     var done = true;
34     var value = @undefined;
35 
36     var position = @getStringIteratorInternalField(this, @stringIteratorFieldIndex);
37     if (position !== -1) {
38         var string = @getStringIteratorInternalField(this, @stringIteratorFieldIteratedString);
39         var length = string.length >>> 0;
40         if (position >= length)
41             @putStringIteratorInternalField(this, @stringIteratorFieldIndex, -1);
42         else {
43             done = false;
44 
45             var first = string.@charCodeAt(position);
46             if (first < 0xD800 || first > 0xDBFF || position + 1 === length)
47                 value = string[position];
48             else {
49                 var second = string.@charCodeAt(position + 1);
50                 if (second < 0xDC00 || second > 0xDFFF)
51                     value = string[position];
52                 else
53                     value = string[position] + string[position + 1];
54             }
55             @putStringIteratorInternalField(this, @stringIteratorFieldIndex, position + value.length);
56         }
57     }
58 
59     return { value, done };
60 }
