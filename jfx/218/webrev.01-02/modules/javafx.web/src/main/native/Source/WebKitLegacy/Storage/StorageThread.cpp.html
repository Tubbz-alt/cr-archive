 37 namespace WebCore {
 38 
 39 static HashSet<StorageThread*>& activeStorageThreads()
 40 {
 41     ASSERT(isMainThread());
 42     static NeverDestroyed<HashSet<StorageThread*>> threads;
 43     return threads;
 44 }
 45 
 46 StorageThread::StorageThread(Type type)
 47     : m_type(type)
 48 {
 49     ASSERT(isMainThread());
 50 }
 51 
 52 StorageThread::~StorageThread()
 53 {
 54     ASSERT(isMainThread());
 55     ASSERT(!m_thread);
 56 }
 57 
 58 void StorageThread::start()
 59 {
 60     ASSERT(isMainThread());
 61     if (!m_thread) {
 62         if (m_type == Type::LocalStorage) {
 63             m_thread = Thread::create("LocalStorage", [this] {
 64                 threadEntryPoint();
 65             });
 66         } else {
 67             ASSERT(m_type == Type::IndexedDB);
 68             m_thread = Thread::create("IndexedDB", [this] {
 69                 threadEntryPoint();
 70             });
 71         }
 72     }
 73     activeStorageThreads().add(this);
 74 }
 75 
 76 void StorageThread::threadEntryPoint()
 77 {
 78 #if PLATFORM(JAVA)
 79     WTF::AttachThreadAsDaemonToJavaEnv autoAttach;
 80 #endif
 81     ASSERT(!isMainThread());
 82 
 83     while (auto function = m_queue.waitForMessage()) {
 84         AutodrainedPool pool;
 85         (*function)();
 86     }
 87 }
 88 
 89 void StorageThread::dispatch(Function<void ()>&& function)
 90 {
 91     ASSERT(isMainThread());
 92     ASSERT(!m_queue.killed() && m_thread);
 93     m_queue.append(makeUnique<Function<void ()>>(WTFMove(function)));
 94 }
 95 
 96 void StorageThread::terminate()
 97 {
 98     ASSERT(isMainThread());
 99     ASSERT(!m_queue.killed() && m_thread);
100     activeStorageThreads().remove(this);
101     // Even in weird, exceptional cases, don't wait on a nonexistent thread to terminate.
102     if (!m_thread)
103         return;
104 
105     m_queue.append(makeUnique<Function<void ()>>([this] {
106         performTerminate();
107     }));
108     m_thread->waitForCompletion();
109     ASSERT(m_queue.killed());
110     m_thread = nullptr;
111 }
112 
113 void StorageThread::performTerminate()
114 {
115     ASSERT(!isMainThread());
116     m_queue.kill();
117 }
118 
119 void StorageThread::releaseFastMallocFreeMemoryInAllThreads()
120 {
121     for (auto& thread : activeStorageThreads())
122         thread->dispatch(&WTF::releaseFastMallocFreeMemory);
123 }
