33 namespace WebCore {
34 namespace XPath {
35 
36 bool isRootDomNode(Node* node)
37 {
38     return node && !node->parentNode();
39 }
40 
41 String stringValue(Node* node)
42 {
43     switch (node->nodeType()) {
44         case Node::ATTRIBUTE_NODE:
45         case Node::PROCESSING_INSTRUCTION_NODE:
46         case Node::COMMENT_NODE:
47         case Node::TEXT_NODE:
48         case Node::CDATA_SECTION_NODE:
49             return node->nodeValue();
50         default:
51             if (isRootDomNode(node) || node->isElementNode())
52                 return TextNodeTraversal::contentsAsString(*node);
53     }
54     return String();
55 }
56 
57 bool isValidContextNode(Node* node)
58 {
59     if (!node)
60         return false;
61     switch (node->nodeType()) {
62         case Node::ATTRIBUTE_NODE:
63         case Node::CDATA_SECTION_NODE:
64         case Node::COMMENT_NODE:
65         case Node::DOCUMENT_NODE:
66         case Node::ELEMENT_NODE:
67         case Node::PROCESSING_INSTRUCTION_NODE:
68         case Node::TEXT_NODE:
69             return true;
70         case Node::DOCUMENT_FRAGMENT_NODE:
71         case Node::DOCUMENT_TYPE_NODE:
72             return false;
73     }
74     ASSERT_NOT_REACHED();
75     return false;
76 }
