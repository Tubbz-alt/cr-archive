 39 template<typename Type>
 40 void* IsoTLS::allocate(api::IsoHeap<Type>& handle, bool abortOnFailure)
 41 {
 42     return allocateImpl<typename api::IsoHeap<Type>::Config>(handle, abortOnFailure);
 43 }
 44 
 45 template<typename Type>
 46 void IsoTLS::deallocate(api::IsoHeap<Type>& handle, void* p)
 47 {
 48     if (!p)
 49         return;
 50     deallocateImpl<typename api::IsoHeap<Type>::Config>(handle, p);
 51 }
 52 
 53 template<typename Type>
 54 void IsoTLS::scavenge(api::IsoHeap<Type>& handle)
 55 {
 56     IsoTLS* tls = get();
 57     if (!tls)
 58         return;
 59     if (!handle.isInitialized())
 60         return;
 61     unsigned offset = handle.allocatorOffset();
 62     if (offset < tls->m_extent)
 63         reinterpret_cast<IsoAllocator<typename api::IsoHeap<Type>::Config>*>(tls->m_data + offset)->scavenge(handle.impl());
 64     offset = handle.deallocatorOffset();
 65     if (offset < tls->m_extent)
 66         reinterpret_cast<IsoDeallocator<typename api::IsoHeap<Type>::Config>*>(tls->m_data + offset)->scavenge();
 67     handle.impl().scavengeNow();
 68 }
 69 
 70 template<typename Config, typename Type>
 71 void* IsoTLS::allocateImpl(api::IsoHeap<Type>& handle, bool abortOnFailure)
 72 {
 73     unsigned offset = handle.allocatorOffset();
 74     IsoTLS* tls = get();
 75     if (!tls || offset >= tls->m_extent)
 76         return allocateSlow<Config>(handle, abortOnFailure);
 77     return tls->allocateFast<Config>(handle, offset, abortOnFailure);
 78 }
 79 
 80 template<typename Config, typename Type>
 81 void* IsoTLS::allocateFast(api::IsoHeap<Type>& handle, unsigned offset, bool abortOnFailure)
 82 {
 83     return reinterpret_cast<IsoAllocator<Config>*>(m_data + offset)->allocate(handle.impl(), abortOnFailure);
 84 }
 85 
 86 template<typename Config, typename Type>
 87 BNO_INLINE void* IsoTLS::allocateSlow(api::IsoHeap<Type>& handle, bool abortOnFailure)
 88 {
 89     for (;;) {
 90         switch (s_mallocFallbackState) {
 91         case MallocFallbackState::Undecided:
 92             determineMallocFallbackState();
 93             continue;
 94         case MallocFallbackState::FallBackToMalloc:
 95 #if BENABLE_MALLOC_HEAP_BREAKDOWN
 96             return malloc_zone_malloc(handle.m_zone, Config::objectSize);
 97 #else
 98             return api::tryMalloc(Config::objectSize);
 99 #endif
100         case MallocFallbackState::DoNotFallBack:
101             break;
102         }
103         break;
104     }
105 
106     // If debug heap is enabled, s_mallocFallbackState becomes MallocFallbackState::FallBackToMalloc.
107     BASSERT(!Environment::get()->isDebugHeapEnabled());
108 
109     IsoTLS* tls = ensureHeapAndEntries(handle);
110 
111     return tls->allocateFast<Config>(handle, handle.allocatorOffset(), abortOnFailure);
112 }
113 
114 template<typename Config, typename Type>
115 void IsoTLS::deallocateImpl(api::IsoHeap<Type>& handle, void* p)
116 {
117     unsigned offset = handle.deallocatorOffset();
118     IsoTLS* tls = get();
119     // Note that this bounds check would be here even if we didn't have to support DebugHeap,
120     // since we don't want unpredictable behavior if offset or m_extent ever got corrupted.
121     if (!tls || offset >= tls->m_extent)
122         deallocateSlow<Config>(handle, p);
123     else
124         tls->deallocateFast<Config>(handle, offset, p);
125 }
126 
127 template<typename Config, typename Type>
128 void IsoTLS::deallocateFast(api::IsoHeap<Type>& handle, unsigned offset, void* p)
129 {
130     reinterpret_cast<IsoDeallocator<Config>*>(m_data + offset)->deallocate(handle, p);
131 }
132 
133 template<typename Config, typename Type>
134 BNO_INLINE void IsoTLS::deallocateSlow(api::IsoHeap<Type>& handle, void* p)
135 {
136     for (;;) {
137         switch (s_mallocFallbackState) {
138         case MallocFallbackState::Undecided:
139             determineMallocFallbackState();
140             continue;
141         case MallocFallbackState::FallBackToMalloc:
142 #if BENABLE_MALLOC_HEAP_BREAKDOWN
143             return malloc_zone_free(handle.m_zone, p);
144 #else
145             return api::free(p);
146 #endif
147         case MallocFallbackState::DoNotFallBack:
148             break;
149         }
150         break;
151     }
152 
153     // If debug heap is enabled, s_mallocFallbackState becomes MallocFallbackState::FallBackToMalloc.
154     BASSERT(!Environment::get()->isDebugHeapEnabled());
155 
156     RELEASE_BASSERT(handle.isInitialized());
157 
158     IsoTLS* tls = ensureEntries(std::max(handle.allocatorOffset(), handle.deallocatorOffset()));
159 
160     tls->deallocateFast<Config>(handle, handle.deallocatorOffset(), p);
161 }
162 
163 inline IsoTLS* IsoTLS::get()
164 {
165 #if HAVE_PTHREAD_MACHDEP_H
166     return static_cast<IsoTLS*>(_pthread_getspecific_direct(tlsKey));
167 #else
168     if (!s_didInitialize)
169         return nullptr;
170     return static_cast<IsoTLS*>(pthread_getspecific(s_tlsKey));
171 #endif
172 }
173 
174 inline void IsoTLS::set(IsoTLS* tls)
175 {
176 #if HAVE_PTHREAD_MACHDEP_H
177     _pthread_setspecific_direct(tlsKey, tls);
178 #else
179     pthread_setspecific(s_tlsKey, tls);
180 #endif
181 }

183 template<typename Type>
184 void IsoTLS::ensureHeap(api::IsoHeap<Type>& handle)
185 {
186     if (!handle.isInitialized()) {
187         LockHolder locker(handle.m_initializationLock);
188         if (!handle.isInitialized())
189             handle.initialize();
190     }
191 }
192 
193 template<typename Type>
194 BNO_INLINE IsoTLS* IsoTLS::ensureHeapAndEntries(api::IsoHeap<Type>& handle)
195 {
196     RELEASE_BASSERT(
197         !get()
198         || handle.allocatorOffset() >= get()->m_extent
199         || handle.deallocatorOffset() >= get()->m_extent);
200     ensureHeap(handle);
201     return ensureEntries(std::max(handle.allocatorOffset(), handle.deallocatorOffset()));
202 }
