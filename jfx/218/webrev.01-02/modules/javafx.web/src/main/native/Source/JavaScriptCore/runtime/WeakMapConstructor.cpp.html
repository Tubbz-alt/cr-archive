 41 void WeakMapConstructor::finishCreation(VM& vm, WeakMapPrototype* prototype)
 42 {
 43     Base::finishCreation(vm, "WeakMap"_s, NameAdditionMode::WithoutStructureTransition);
 44     putDirectWithoutTransition(vm, vm.propertyNames->prototype, prototype, PropertyAttribute::DontEnum | PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly);
 45     putDirectWithoutTransition(vm, vm.propertyNames->length, jsNumber(0), PropertyAttribute::DontEnum | PropertyAttribute::ReadOnly);
 46 }

 48 static EncodedJSValue JSC_HOST_CALL callWeakMap(JSGlobalObject*, CallFrame*);
 49 static EncodedJSValue JSC_HOST_CALL constructWeakMap(JSGlobalObject*, CallFrame*);
 50 
 51 WeakMapConstructor::WeakMapConstructor(VM& vm, Structure* structure)
 52     : Base(vm, structure, callWeakMap, constructWeakMap)
 53 {
 54 }
 55 
 56 static EncodedJSValue JSC_HOST_CALL callWeakMap(JSGlobalObject* globalObject, CallFrame*)
 57 {
 58     VM& vm = globalObject->vm();
 59     auto scope = DECLARE_THROW_SCOPE(vm);
 60     return JSValue::encode(throwConstructorCannotBeCalledAsFunctionTypeError(globalObject, scope, "WeakMap"));
 61 }
 62 
 63 static EncodedJSValue JSC_HOST_CALL constructWeakMap(JSGlobalObject* globalObject, CallFrame* callFrame)
 64 {
 65     VM& vm = globalObject->vm();
 66     auto scope = DECLARE_THROW_SCOPE(vm);
 67 
 68     Structure* weakMapStructure = InternalFunction::createSubclassStructure(globalObject, callFrame->jsCallee(), callFrame->newTarget(), globalObject->weakMapStructure());
 69     RETURN_IF_EXCEPTION(scope, encodedJSValue());
 70     JSWeakMap* weakMap = JSWeakMap::create(vm, weakMapStructure);
 71     JSValue iterable = callFrame->argument(0);
 72     if (iterable.isUndefinedOrNull())
 73         return JSValue::encode(weakMap);
 74 
 75     JSValue adderFunction = weakMap->JSObject::get(globalObject, vm.propertyNames->set);
 76     RETURN_IF_EXCEPTION(scope, encodedJSValue());
 77 
 78     CallData adderFunctionCallData;
 79     CallType adderFunctionCallType = getCallData(vm, adderFunction, adderFunctionCallData);
 80     if (adderFunctionCallType == CallType::None)
 81         return JSValue::encode(throwTypeError(globalObject, scope));
 82 
 83     scope.release();
 84     forEachInIterable(globalObject, iterable, [&](VM& vm, JSGlobalObject* globalObject, JSValue nextItem) {
 85         auto scope = DECLARE_THROW_SCOPE(vm);
 86         if (!nextItem.isObject()) {
 87             throwTypeError(globalObject, scope);
 88             return;
 89         }
 90 
 91         JSValue key = nextItem.get(globalObject, static_cast<unsigned>(0));
 92         RETURN_IF_EXCEPTION(scope, void());
 93 
 94         JSValue value = nextItem.get(globalObject, static_cast<unsigned>(1));
 95         RETURN_IF_EXCEPTION(scope, void());
 96 
 97         MarkedArgumentBuffer arguments;
 98         arguments.append(key);
 99         arguments.append(value);
100         ASSERT(!arguments.hasOverflowed());
101         scope.release();
102         call(globalObject, adderFunction, adderFunctionCallType, adderFunctionCallData, weakMap, arguments);
103     });
104 
105     return JSValue::encode(weakMap);
106 }
