35 namespace Nicosia {
36 
37 SceneIntegration::SceneIntegration(Scene& scene, Client& client)
38 {
39     m_client.scene = makeRef(scene);
40     m_client.object = &client;
41 }
42 
43 SceneIntegration::~SceneIntegration()
44 {
45     ASSERT(!m_client.object);
46 }
47 
48 void SceneIntegration::setClient(Client& client)
49 {
50     LockHolder locker(m_client.lock);
51     m_client.object = &client;
52 }
53 
54 void SceneIntegration::invalidate()
55 {
56     LockHolder locker(m_client.lock);
57     m_client.scene = nullptr;
58     m_client.object = nullptr;
59 }
60 
61 void SceneIntegration::requestUpdate()
62 {
63     LockHolder locker(m_client.lock);
64     if (m_client.object)
65         m_client.object->requestUpdate();
66 }
67 
68 std::unique_ptr<SceneIntegration::UpdateScope> SceneIntegration::createUpdateScope()
69 {
70     return makeUnique<UpdateScope>(makeRef(*this));
71 }
72 
73 SceneIntegration::Client::~Client() = default;
74 
75 SceneIntegration::UpdateScope::UpdateScope(Ref<SceneIntegration>&& sceneIntegration)
76     : m_sceneIntegration(WTFMove(sceneIntegration))
77     , m_locker(m_sceneIntegration->m_client.lock)
78 {
79 }
80 
81 SceneIntegration::UpdateScope::~UpdateScope()
82 {
83     if (!m_sceneIntegration->m_client.scene)
84         return;
85 
86     m_sceneIntegration->m_client.scene->accessState(
87         [](Nicosia::Scene::State& state)
88         {
89             for (auto& compositionLayer : state.layers)
90                 compositionLayer->flushState([](auto&) { });
91         });
92 
93     auto& sceneIntegrationObj = m_sceneIntegration.get();
94     if (sceneIntegrationObj.m_client.object)
95         sceneIntegrationObj.m_client.object->requestUpdate();
96 }
