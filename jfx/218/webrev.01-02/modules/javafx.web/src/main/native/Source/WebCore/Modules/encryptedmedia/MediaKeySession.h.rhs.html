 46 namespace WebCore {
 47 
 48 class BufferSource;
 49 class CDM;
 50 class DeferredPromise;
 51 class MediaKeyStatusMap;
 52 class MediaKeys;
 53 class SharedBuffer;
 54 
 55 template<typename IDLType> class DOMPromiseProxy;
 56 
 57 class MediaKeySession final : public RefCounted<MediaKeySession>, public EventTargetWithInlineData, public ActiveDOMObject, public CDMInstanceSessionClient {
 58     WTF_MAKE_ISO_ALLOCATED(MediaKeySession);
 59 public:
 60     static Ref<MediaKeySession> create(ScriptExecutionContext&, WeakPtr<MediaKeys>&&, MediaKeySessionType, bool useDistinctiveIdentifier, Ref<CDM>&&, Ref<CDMInstanceSession>&&);
 61     virtual ~MediaKeySession();
 62 
 63     using RefCounted<MediaKeySession>::ref;
 64     using RefCounted<MediaKeySession>::deref;
 65 
 66     bool isClosed() const { return m_closed; }
 67 
 68     const String& sessionId() const;
 69     double expiration() const;
 70     Ref<MediaKeyStatusMap> keyStatuses() const;
 71 
 72     void generateRequest(const AtomString&, const BufferSource&, Ref<DeferredPromise>&&);
 73     void load(const String&, Ref<DeferredPromise>&&);
 74     void update(const BufferSource&, Ref<DeferredPromise>&&);
 75     void close(Ref<DeferredPromise>&&);
 76     void remove(Ref<DeferredPromise>&&);
 77 
 78     using ClosedPromise = DOMPromiseProxy<IDLVoid>;
 79     ClosedPromise& closed() { return m_closedPromise.get(); }
 80 
 81     const Vector<std::pair<Ref<SharedBuffer>, MediaKeyStatus>>& statuses() const { return m_statuses; }
 82 
 83     // ActiveDOMObject
 84     bool hasPendingActivity() const override;
 85 
 86 private:
 87     MediaKeySession(ScriptExecutionContext&, WeakPtr<MediaKeys>&&, MediaKeySessionType, bool useDistinctiveIdentifier, Ref<CDM>&&, Ref<CDMInstanceSession>&&);
 88     void enqueueMessage(MediaKeyMessageType, const SharedBuffer&);
 89     void updateExpiration(double);
 90     void sessionClosed();
 91     String mediaKeysStorageDirectory() const;
 92 
 93     // CDMInstanceSessionClient
 94     void updateKeyStatuses(CDMInstanceSessionClient::KeyStatusVector&&) override;
 95     void sendMessage(CDMMessageType, Ref<SharedBuffer>&& message) final;
 96     void sessionIdChanged(const String&) final;
 97 
 98     // EventTarget
 99     EventTargetInterface eventTargetInterface() const override { return MediaKeySessionEventTargetInterfaceType; }
100     ScriptExecutionContext* scriptExecutionContext() const override { return ActiveDOMObject::scriptExecutionContext(); }
101     void refEventTarget() override { ref(); }
102     void derefEventTarget() override { deref(); }
103 
104     // ActiveDOMObject

105     const char* activeDOMObjectName() const override;


106 
107     WeakPtr<MediaKeys> m_keys;
108     String m_sessionId;
109     double m_expiration;
110     UniqueRef<ClosedPromise> m_closedPromise;
111     Ref<MediaKeyStatusMap> m_keyStatuses;
112     bool m_closed { false };
113     bool m_uninitialized { true };
114     bool m_callable { false };
115     bool m_useDistinctiveIdentifier;
116     MediaKeySessionType m_sessionType;
117     Ref<CDM> m_implementation;
118     Ref<CDMInstanceSession> m_instanceSession;


119     Vector<Ref<SharedBuffer>> m_recordOfKeyUsage;
120     double m_firstDecryptTime { 0 };
121     double m_latestDecryptTime { 0 };
122     Vector<std::pair<Ref<SharedBuffer>, MediaKeyStatus>> m_statuses;
123 };
124 
125 } // namespace WebCore
