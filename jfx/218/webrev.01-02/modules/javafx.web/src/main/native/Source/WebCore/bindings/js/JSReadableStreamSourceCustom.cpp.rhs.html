39 JSValue JSReadableStreamSource::start(JSGlobalObject& lexicalGlobalObject, CallFrame& callFrame, Ref<DeferredPromise>&& promise)
40 {
41     VM& vm = lexicalGlobalObject.vm();
42 
43     // FIXME: Why is it ok to ASSERT the argument count here?
44     ASSERT(callFrame.argumentCount());
45     JSReadableStreamDefaultController* controller = jsDynamicCast<JSReadableStreamDefaultController*>(vm, callFrame.uncheckedArgument(0));
46     ASSERT(controller);
47 
48     m_controller.set(vm, this, controller);
49 
50     wrapped().start(ReadableStreamDefaultController(controller), WTFMove(promise));
51 
52     return jsUndefined();
53 }
54 
55 JSValue JSReadableStreamSource::pull(JSGlobalObject&, CallFrame&, Ref<DeferredPromise>&& promise)
56 {
57     wrapped().pull(WTFMove(promise));
58     return jsUndefined();
59 }
60 
61 JSValue JSReadableStreamSource::controller(JSGlobalObject&) const
62 {
63     ASSERT_NOT_REACHED();
64     return jsUndefined();
65 }
