 56 static Ref<ArrayBuffer> produceClientDataJson(ClientDataType type, const BufferSource& challenge, const SecurityOrigin& origin)
 57 {
 58     auto object = JSON::Object::create();
 59     switch (type) {
 60     case ClientDataType::Create:
 61         object->setString("type"_s, "webauthn.create"_s);
 62         break;
 63     case ClientDataType::Get:
 64         object->setString("type"_s, "webauthn.get"_s);
 65         break;
 66     }
 67     object->setString("challenge"_s, WTF::base64URLEncode(challenge.data(), challenge.length()));
 68     object->setString("origin"_s, origin.toRawString());
 69 
 70     auto utf8JSONString = object->toJSONString().utf8();
 71     return ArrayBuffer::create(utf8JSONString.data(), utf8JSONString.length());
 72 }
 73 
 74 static Vector<uint8_t> produceClientDataJsonHash(const ArrayBuffer& clientDataJson)
 75 {
 76     auto crypto = PAL::CryptoDigest::create(PAL::CryptoDigest::Algorithm::SHA_256);
 77     crypto->addBytes(clientDataJson.data(), clientDataJson.byteLength());
 78     return crypto->computeHash();
 79 }

 81 static bool needsAppIdQuirks(const String& host, const String& appId)
 82 {
 83     // FIXME(197524): Remove this quirk in 2023. As an early adopter of U2F features, Google has a large number of
 84     // existing device registrations that authenticate 'google.com' against 'gstatic.com'. Firefox and other browsers
 85     // have agreed to grant an exception to the AppId rules for a limited time period (5 years from January, 2018) to
 86     // allow existing Google users to seamlessly transition to proper WebAuthN behavior.
 87     if (equalLettersIgnoringASCIICase(host, "google.com") || host.endsWithIgnoringASCIICase(".google.com"))
 88         return (appId == "https://www.gstatic.com/securitykey/origins.json"_s) || (appId == "https://www.gstatic.com/securitykey/a/google.com/origins.json"_s);
 89     return false;
 90 }
 91 
 92 // The following roughly implements Step 1-3 of the spec to avoid the complexity of making unnecessary network requests:
 93 // https://fidoalliance.org/specs/fido-v2.0-id-20180227/fido-appid-and-facets-v2.0-id-20180227.html#determining-if-a-caller-s-facetid-is-authorized-for-an-appid
 94 // It follows what Chrome and Firefox do, see:
 95 // https://bugzilla.mozilla.org/show_bug.cgi?id=1244959#c8
 96 // https://bugs.chromium.org/p/chromium/issues/detail?id=818303
 97 static String processAppIdExtension(const SecurityOrigin& facetId, const String& appId)
 98 {
 99     // Step 1. Skipped since facetId should always be secure origins.
100     ASSERT(LegacySchemeRegistry::shouldTreatURLSchemeAsSecure(facetId.protocol()));
101 
102     // Step 2. Follow Chrome and Firefox to use the origin directly without adding a trailing slash.
103     if (appId.isEmpty())
104         return facetId.toString();
105 
106     // Step 3. Relax the comparison to same site.
107     URL appIdURL(URL(), appId);
108     if (!appIdURL.isValid() || facetId.protocol() != appIdURL.protocol() || (RegistrableDomain(appIdURL) != RegistrableDomain::uncheckedCreateFromHost(facetId.host()) && !needsAppIdQuirks(facetId.host(), appId)))
109         return String();
110     return appId;
111 }
112 
113 // The default behaviour for google.com is to always turn on the legacy AppID support.
114 static bool processGoogleLegacyAppIdSupportExtension(const Optional<AuthenticationExtensionsClientInputs>& extensions, const String& rpId)
115 {
116     if (rpId != "google.com"_s)
117         return false;
118     if (!extensions)
119         return true;
120     return extensions->googleLegacyAppidSupport;
121 }

135 void AuthenticatorCoordinator::create(const Document& document, const PublicKeyCredentialCreationOptions& options, bool sameOriginWithAncestors, RefPtr<AbortSignal>&& abortSignal, CredentialPromise&& promise) const
136 {
137     using namespace AuthenticatorCoordinatorInternal;
138 
139     const auto& callerOrigin = document.securityOrigin();
140     auto* frame = document.frame();
141     ASSERT(frame);
142     // The following implements https://www.w3.org/TR/webauthn/#createCredential as of 5 December 2017.

143     // Step 1, 3, 16 are handled by the caller.
144     // Step 2.
145     if (!sameOriginWithAncestors) {
146         promise.reject(Exception { NotAllowedError, "The origin of the document is not the same as its ancestors."_s });
147         return;
148     }
149 
150     // Step 5. Skipped since SecurityOrigin doesn't have the concept of "opaque origin".
151     // Step 6. The effective domain may be represented in various manners, such as a domain or an ip address.
152     // Only the domain format of host is permitted in WebAuthN.
153     if (URL::hostIsIPAddress(callerOrigin.domain())) {
154         promise.reject(Exception { SecurityError, "The effective domain of the document is not a valid domain."_s });
155         return;
156     }
157 
158     // Step 7.
159     if (!options.rp.id.isEmpty() && !callerOrigin.isMatchingRegistrableDomainSuffix(options.rp.id)) {
160         promise.reject(Exception { SecurityError, "The provided RP ID is not a registrable domain suffix of the effective domain of the document."_s });
161         return;
162     }
163     if (options.rp.id.isEmpty())
164         options.rp.id = callerOrigin.domain();
165 
166     // Step 8-10.
167     // Most of the jobs are done by bindings. However, we can't know if the JSValue of options.pubKeyCredParams
168     // is empty or not. Return NotSupportedError as long as it is empty.
169     if (options.pubKeyCredParams.isEmpty()) {
170         promise.reject(Exception { NotSupportedError, "No desired properties of the to be created credential are provided."_s });
171         return;
172     }
173 
174     // Step 11-12.
175     // Only Google Legacy AppID Support Extension is supported.
176     options.extensions = AuthenticationExtensionsClientInputs { String(), processGoogleLegacyAppIdSupportExtension(options.extensions, options.rp.id) };
177 
178     // Step 13-15.
179     auto clientDataJson = produceClientDataJson(ClientDataType::Create, options.challenge, callerOrigin);
180     auto clientDataJsonHash = produceClientDataJsonHash(clientDataJson);
181 
182     // Step 4, 17-21.
183     if (!m_client) {
184         promise.reject(Exception { UnknownError, "Unknown internal error."_s });
185         return;
186     }
187 
188     auto callback = [clientDataJson = WTFMove(clientDataJson), promise = WTFMove(promise), abortSignal = WTFMove(abortSignal)] (AuthenticatorResponseData&& data, ExceptionData&& exception) mutable {
189         if (abortSignal && abortSignal->aborted()) {
190             promise.reject(Exception { AbortError, "Aborted by AbortSignal."_s });
191             return;
192         }
193 
194         if (auto response = AuthenticatorResponse::tryCreate(WTFMove(data))) {
195             response->setClientDataJSON(WTFMove(clientDataJson));
196             promise.resolve(PublicKeyCredential::create(response.releaseNonNull()).ptr());
197             return;
198         }
199         ASSERT(!exception.message.isNull());
200         promise.reject(exception.toException());
201     };
202     // Async operations are dispatched and handled in the messenger.
203     m_client->makeCredential(*frame, callerOrigin, clientDataJsonHash, options, WTFMove(callback));
204 }

206 void AuthenticatorCoordinator::discoverFromExternalSource(const Document& document, const PublicKeyCredentialRequestOptions& options, bool sameOriginWithAncestors, RefPtr<AbortSignal>&& abortSignal, CredentialPromise&& promise) const
207 {
208     using namespace AuthenticatorCoordinatorInternal;
209 
210     auto& callerOrigin = document.securityOrigin();
211     auto* frame = document.frame();
212     ASSERT(frame);
213     // The following implements https://www.w3.org/TR/webauthn/#createCredential as of 5 December 2017.
214     // Step 1, 3, 13 are handled by the caller.
215     // Step 2.
216     if (!sameOriginWithAncestors) {
217         promise.reject(Exception { NotAllowedError, "The origin of the document is not the same as its ancestors."_s });
218         return;
219     }
220 
221     // Step 5. Skipped since SecurityOrigin doesn't have the concept of "opaque origin".
222     // Step 6. The effective domain may be represented in various manners, such as a domain or an ip address.
223     // Only the domain format of host is permitted in WebAuthN.
224     if (URL::hostIsIPAddress(callerOrigin.domain())) {
225         promise.reject(Exception { SecurityError, "The effective domain of the document is not a valid domain."_s });
226         return;
227     }
228 
229     // Step 7.
230     if (!options.rpId.isEmpty() && !callerOrigin.isMatchingRegistrableDomainSuffix(options.rpId)) {
231         promise.reject(Exception { SecurityError, "The provided RP ID is not a registrable domain suffix of the effective domain of the document."_s });
232         return;
233     }
234     if (options.rpId.isEmpty())
235         options.rpId = callerOrigin.domain();
236 
237     // Step 8-9.
238     // Only FIDO AppID Extension is supported.
239     if (options.extensions && !options.extensions->appid.isNull()) {
240         // The following implements https://www.w3.org/TR/webauthn/#sctn-appid-extension as of 4 March 2019.
241         auto appid = processAppIdExtension(callerOrigin, options.extensions->appid);
242         if (!appid) {
243             promise.reject(Exception { SecurityError, "The origin of the document is not authorized for the provided App ID."_s });
244             return;
245         }
246         options.extensions->appid = appid;
247     }
248 
249     // Step 10-12.
250     auto clientDataJson = produceClientDataJson(ClientDataType::Get, options.challenge, callerOrigin);
251     auto clientDataJsonHash = produceClientDataJsonHash(clientDataJson);
252 
253     // Step 4, 14-19.
254     if (!m_client) {
255         promise.reject(Exception { UnknownError, "Unknown internal error."_s });
256         return;
257     }
258 
259     auto callback = [clientDataJson = WTFMove(clientDataJson), promise = WTFMove(promise), abortSignal = WTFMove(abortSignal)] (AuthenticatorResponseData&& data, ExceptionData&& exception) mutable {
260         if (abortSignal && abortSignal->aborted()) {
261             promise.reject(Exception { AbortError, "Aborted by AbortSignal."_s });
262             return;
263         }
264 
265         if (auto response = AuthenticatorResponse::tryCreate(WTFMove(data))) {
266             response->setClientDataJSON(WTFMove(clientDataJson));
267             promise.resolve(PublicKeyCredential::create(response.releaseNonNull()).ptr());
268             return;
269         }
270         ASSERT(!exception.message.isNull());
271         promise.reject(exception.toException());
272     };
273     // Async operations are dispatched and handled in the messenger.
274     m_client->getAssertion(*frame, callerOrigin, clientDataJsonHash, options, WTFMove(callback));
275 }

277 void AuthenticatorCoordinator::isUserVerifyingPlatformAuthenticatorAvailable(DOMPromiseDeferred<IDLBoolean>&& promise) const
278 {
279     // The following implements https://www.w3.org/TR/webauthn/#isUserVerifyingPlatformAuthenticatorAvailable
280     // as of 5 December 2017.
281     if (!m_client)  {
282         promise.reject(Exception { UnknownError, "Unknown internal error."_s });
283         return;
284     }
285 
286     // FIXME(182767): We should consider more on the assessment of the return value. Right now, we return true/false
287     // immediately according to platform specific procedures.
288     auto completionHandler = [promise = WTFMove(promise)] (bool result) mutable {
289         promise.resolve(result);
290     };
291     // Async operation are dispatched and handled in the messenger.
292     m_client->isUserVerifyingPlatformAuthenticatorAvailable(WTFMove(completionHandler));
293 }
