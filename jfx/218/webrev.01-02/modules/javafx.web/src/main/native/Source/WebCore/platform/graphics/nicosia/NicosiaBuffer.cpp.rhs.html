36 Ref<Buffer> Buffer::create(const WebCore::IntSize& size, Flags flags)
37 {
38     return adoptRef(*new Buffer(size, flags));
39 }
40 
41 Buffer::Buffer(const WebCore::IntSize& size, Flags flags)
42     : m_size(size)
43     , m_flags(flags)
44 {
45     auto checkedArea = size.area() * 4;
46     m_data = MallocPtr<unsigned char>::tryZeroedMalloc(checkedArea.unsafeGet());




47 }
48 
49 Buffer::~Buffer() = default;
50 
51 void Buffer::beginPainting()
52 {
53     LockHolder locker(m_painting.lock);
54     ASSERT(m_painting.state == PaintingState::Complete);
55     m_painting.state = PaintingState::InProgress;
56 }
57 
58 void Buffer::completePainting()
59 {
60     LockHolder locker(m_painting.lock);
61     ASSERT(m_painting.state == PaintingState::InProgress);
62     m_painting.state = PaintingState::Complete;
63     m_painting.condition.notifyOne();
64 }
65 
66 void Buffer::waitUntilPaintingComplete()
67 {
68     LockHolder locker(m_painting.lock);
69     m_painting.condition.wait(m_painting.lock,
70         [this] { return m_painting.state == PaintingState::Complete; });
71 }
