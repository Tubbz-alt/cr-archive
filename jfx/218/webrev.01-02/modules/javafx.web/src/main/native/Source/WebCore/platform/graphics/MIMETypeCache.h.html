45     bool supportsContainerType(const String&);

48     virtual const HashSet<String, ASCIICaseInsensitiveHash>& staticContainerTypeList();

51     HashSet<String, ASCIICaseInsensitiveHash>& supportedTypes();

54     virtual bool isAvailable() const;
55     bool isEmpty() const;

62     Optional<HashMap<String, MediaPlayerEnums::SupportsType, ASCIICaseInsensitiveHash>> m_cachedResults;
63 };
