35 void AbstractHeap::Payload::dump(PrintStream& out) const
36 {
37     if (isTop())
38         out.print("TOP");
39     else
40         out.print(value());
41 }
42 
43 void AbstractHeap::Payload::dumpAsOperand(PrintStream& out) const
44 {
45     if (isTop())
46         out.print("TOP");
47     else
48         out.print(Operand::fromBits(value()));
49 }
50 
51 void AbstractHeap::dump(PrintStream& out) const
52 {
53     out.print(kind());
54     if (kind() == InvalidAbstractHeap || kind() == World || kind() == Heap || payload().isTop())
55         return;
56     if (kind() == DOMState) {
57         out.print("(", DOMJIT::HeapRange::fromRaw(payload().value32()), ")");
58         return;
59     }
60     if (kind() == Stack) {
61         out.print("(");
62         payload().dumpAsOperand(out);
63         out.print(")");
64         return;
65     }
66 
67     out.print("(", payload(), ")");
68 }

76 void printInternal(PrintStream& out, AbstractHeapKind kind)
77 {
78     switch (kind) {
79 #define ABSTRACT_HEAP_DUMP(name)            \
80     case AbstractHeapKind::name:            \
81         out.print(#name);                   \
82         return;
83     FOR_EACH_ABSTRACT_HEAP_KIND(ABSTRACT_HEAP_DUMP)
84 #undef ABSTRACT_HEAP_DUMP
85     }
86     RELEASE_ASSERT_NOT_REACHED();
87 }
