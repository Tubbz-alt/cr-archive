30 #include "ContentSecurityPolicyResponseHeaders.h"
31 #include "ResourceError.h"
32 #include "ServiceWorkerRegistrationKey.h"

35 namespace WebCore {
36 
37 struct ServiceWorkerFetchResult {
38     ServiceWorkerJobDataIdentifier jobDataIdentifier;
39     ServiceWorkerRegistrationKey registrationKey;
40     String script;
41     ContentSecurityPolicyResponseHeaders contentSecurityPolicy;
42     String referrerPolicy;
43     ResourceError scriptError;
44 
45     ServiceWorkerFetchResult isolatedCopy() const { return { jobDataIdentifier, registrationKey.isolatedCopy(), script.isolatedCopy(), contentSecurityPolicy.isolatedCopy(), referrerPolicy.isolatedCopy(), scriptError.isolatedCopy() }; }
46 
47     template<class Encoder> void encode(Encoder&) const;
48     template<class Decoder> static bool decode(Decoder&, ServiceWorkerFetchResult&);
49 };
50 
51 inline ServiceWorkerFetchResult serviceWorkerFetchError(ServiceWorkerJobDataIdentifier jobDataIdentifier, ServiceWorkerRegistrationKey&& registrationKey, ResourceError&& error)
52 {
53     return { jobDataIdentifier, WTFMove(registrationKey), { }, { }, { }, WTFMove(error) };
54 }
55 
56 template<class Encoder>
57 void ServiceWorkerFetchResult::encode(Encoder& encoder) const
58 {
59     encoder << jobDataIdentifier << registrationKey << script << contentSecurityPolicy << referrerPolicy << scriptError;
60 }
61 
62 template<class Decoder>
63 bool ServiceWorkerFetchResult::decode(Decoder& decoder, ServiceWorkerFetchResult& result)
64 {
65     Optional<ServiceWorkerJobDataIdentifier> jobDataIdentifier;
66     decoder >> jobDataIdentifier;
67     if (!jobDataIdentifier)
68         return false;
69     result.jobDataIdentifier = WTFMove(*jobDataIdentifier);
70 
71     auto registrationKey = ServiceWorkerRegistrationKey::decode(decoder);
72     if (!registrationKey)
73         return false;
74     std::swap(*registrationKey, result.registrationKey);
75 
76     if (!decoder.decode(result.script))
77         return false;
78     if (!decoder.decode(result.contentSecurityPolicy))
79         return false;
80     if (!decoder.decode(result.referrerPolicy))
81         return false;
82     if (!decoder.decode(result.scriptError))
83         return false;
84 
85     return true;
86 }

88 } // namespace WebCore
