30 #include "GPUPipeline.h"

32 #include "WHLSLPrepare.h"
33 #include <wtf/RefCounted.h>
34 #include <wtf/RefPtr.h>
35 #include <wtf/RetainPtr.h>
36 
37 OBJC_PROTOCOL(MTLComputePipelineState);
38 
39 namespace WebCore {
40 
41 class GPUDevice;
42 class GPUErrorScopes;
43 class GPUPipelineLayout;
44 
45 struct GPUComputePipelineDescriptor;
46 
47 using PlatformComputePipeline = MTLComputePipelineState;
48 using PlatformComputePipelineSmartPtr = RetainPtr<MTLComputePipelineState>;
49 
50 class GPUComputePipeline final : public GPUPipeline {
51 public:
52     virtual ~GPUComputePipeline();
53 
54     static RefPtr<GPUComputePipeline> tryCreate(const GPUDevice&, const GPUComputePipelineDescriptor&, GPUErrorScopes&);
55 
56     bool isComputePipeline() const { return true; }
57 
58     bool recompile(const GPUDevice&, GPUProgrammableStageDescriptor&& computeStage);
59 
60     const PlatformComputePipeline* platformComputePipeline() const { return m_platformComputePipeline.get(); }
61 
62     WHLSL::ComputeDimensions computeDimensions() const { return m_computeDimensions; }
63 
64 private:
65     GPUComputePipeline(PlatformComputePipelineSmartPtr&&, WHLSL::ComputeDimensions, const RefPtr<GPUPipelineLayout>&);
66 
67     PlatformComputePipelineSmartPtr m_platformComputePipeline;
68     WHLSL::ComputeDimensions m_computeDimensions { 0, 0, 0 };
69 
70     // Preserved for Web Inspector recompilation.
71     RefPtr<GPUPipelineLayout> m_layout;
72 };
73 
74 } // namespace WebCore
75 
76 SPECIALIZE_TYPE_TRAITS_GPUPIPELINE(WebCore::GPUComputePipeline, isComputePipeline())
