27 // 25.3.3.3 GeneratorResume ( generator, value )
28 // 25.3.3.4 GeneratorResumeAbrupt(generator, abruptCompletion)
29 @globalPrivate
30 function generatorResume(generator, state, generatorThis, sentValue, value, resumeMode)
31 {
32     "use strict";
33 
34     var done = state === @GeneratorStateCompleted;
35     if (!done) {
36         try {
37             @putGeneratorInternalField(generator, @generatorFieldState, @GeneratorStateExecuting);
38             value = @getGeneratorInternalField(generator, @generatorFieldNext).@call(generatorThis, generator, state, sentValue, resumeMode, @getGeneratorInternalField(generator, @generatorFieldFrame));
39             if (@getGeneratorInternalField(generator, @generatorFieldState) === @GeneratorStateExecuting) {
40                 @putGeneratorInternalField(generator, @generatorFieldState, @GeneratorStateCompleted);
41                 done = true;
42             }
43         } catch (error) {
44             @putGeneratorInternalField(generator, @generatorFieldState, @GeneratorStateCompleted);
45             throw error;
46         }
47     }
48     return { value, done };
49 }
50 
51 function next(value)
52 {
53     "use strict";
54 
55     if (!@isGenerator(this))
56         @throwTypeError("|this| should be a generator");
57 
58     var state = @getGeneratorInternalField(this, @generatorFieldState);
59     if (state === @GeneratorStateExecuting)
60         @throwTypeError("Generator is executing");
61 
62     return @generatorResume(this, state, @getGeneratorInternalField(this, @generatorFieldThis), value, @undefined, @GeneratorResumeModeNormal);
63 }
64 
65 function return(value)
66 {
67     "use strict";
68 
69     if (!@isGenerator(this))
70         @throwTypeError("|this| should be a generator");
71 
72     var state = @getGeneratorInternalField(this, @generatorFieldState);
73     if (state === @GeneratorStateExecuting)
74         @throwTypeError("Generator is executing");
75 
76     return @generatorResume(this, state, @getGeneratorInternalField(this, @generatorFieldThis), value, value, @GeneratorResumeModeReturn);
77 }
78 
79 function throw(exception)
80 {
81     "use strict";
82 
83     if (!@isGenerator(this))
84         @throwTypeError("|this| should be a generator");
85 
86     var state = @getGeneratorInternalField(this, @generatorFieldState);
87     if (state === @GeneratorStateExecuting)
88         @throwTypeError("Generator is executing");
89 
90     if (state === @GeneratorStateCompleted)
91         throw exception;
92 
93     return @generatorResume(this, state, @getGeneratorInternalField(this, @generatorFieldThis), exception, @undefined, @GeneratorResumeModeThrow);
94 }
