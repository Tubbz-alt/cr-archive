38 // Provides for the OSR exit profiling functionality that is common between the DFG
39 // and the FTL.
40 
41 struct OSRExitBase {
42     OSRExitBase(ExitKind kind, CodeOrigin origin, CodeOrigin originForProfile, bool wasHoisted)
43         : m_kind(kind)
44         , m_wasHoisted(wasHoisted)
45         , m_codeOrigin(origin)
46         , m_codeOriginForExitProfile(originForProfile)
47     {
48         ASSERT(m_codeOrigin.isSet());
49         ASSERT(m_codeOriginForExitProfile.isSet());
50     }
51 
52     uint32_t m_count { 0 };
53     ExitKind m_kind;
54     bool m_wasHoisted;
55 
56     CodeOrigin m_codeOrigin;
57     CodeOrigin m_codeOriginForExitProfile;
58     CallSiteIndex m_exceptionHandlerCallSiteIndex;
59 
60     ALWAYS_INLINE bool isExceptionHandler() const
61     {
62         return m_kind == ExceptionCheck || m_kind == GenericUnwind;
63     }
64 
65     // True if this exit is used as an exception handler for unwinding. This happens to only be set when
66     // isExceptionHandler is true, but all this actually means is that the OSR exit will assume that the
67     // machine state is as it would be coming out of genericUnwind.
68     ALWAYS_INLINE bool isGenericUnwindHandler() const
69     {
70         return m_kind == GenericUnwind;
71     }
72 





73 protected:
74     void considerAddingAsFrequentExitSite(CodeBlock* profiledCodeBlock, ExitingJITType jitType)
75     {
76         if (m_count)
77             considerAddingAsFrequentExitSiteSlow(profiledCodeBlock, jitType);
78     }
79 
80 private:
81     void considerAddingAsFrequentExitSiteSlow(CodeBlock* profiledCodeBlock, ExitingJITType);
82 };
83 
84 } } // namespace JSC::DFG
