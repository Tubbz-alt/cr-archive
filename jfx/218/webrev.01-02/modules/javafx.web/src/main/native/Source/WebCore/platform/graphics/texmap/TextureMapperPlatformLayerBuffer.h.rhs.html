 26 #pragma once
 27 
 28 #if USE(COORDINATED_GRAPHICS)
 29 
 30 #include "BitmapTextureGL.h"
 31 #include "TextureMapperGLHeaders.h"
 32 #include "TextureMapperPlatformLayer.h"
 33 #include <wtf/MonotonicTime.h>
 34 #include <wtf/Variant.h>
 35 
 36 namespace WebCore {
 37 
 38 class TextureMapperPlatformLayerBuffer : public TextureMapperPlatformLayer {
 39     WTF_MAKE_NONCOPYABLE(TextureMapperPlatformLayerBuffer);
 40     WTF_MAKE_FAST_ALLOCATED();
 41 public:
 42     TextureMapperPlatformLayerBuffer(RefPtr<BitmapTexture>&&, TextureMapperGL::Flags = 0);
 43 
 44     TextureMapperPlatformLayerBuffer(GLuint textureID, const IntSize&, TextureMapperGL::Flags, GLint internalFormat);
 45 
 46     struct RGBTexture {
 47         GLuint id;
 48     };
 49     struct YUVTexture {
 50         unsigned numberOfPlanes;
 51         std::array<GLuint, 3> planes;
 52         std::array<unsigned, 3> yuvPlane;
 53         std::array<unsigned, 3> yuvPlaneOffset;
 54         std::array<GLfloat, 9> yuvToRgbMatrix;
 55     };
 56     struct ExternalOESTexture {
 57         GLuint id;
 58     };
 59     using TextureVariant = WTF::Variant<RGBTexture, YUVTexture, ExternalOESTexture>;
 60 
 61     TextureMapperPlatformLayerBuffer(TextureVariant&&, const IntSize&, TextureMapperGL::Flags, GLint internalFormat);
 62 
 63     virtual ~TextureMapperPlatformLayerBuffer();
 64 
 65     void paintToTextureMapper(TextureMapper&, const FloatRect&, const TransformationMatrix& modelViewMatrix = TransformationMatrix(), float opacity = 1.0) final;
 66 
 67     bool canReuseWithoutReset(const IntSize&, GLint internalFormat);
 68     BitmapTextureGL& textureGL() { return static_cast<BitmapTextureGL&>(*m_texture); }
 69 
 70     inline void markUsed() { m_timeLastUsed = MonotonicTime::now(); }
 71     MonotonicTime lastUsedTime() const { return m_timeLastUsed; }
 72 
 73     class UnmanagedBufferDataHolder {
 74         WTF_MAKE_NONCOPYABLE(UnmanagedBufferDataHolder);
 75         WTF_MAKE_FAST_ALLOCATED();
 76     public:
 77         UnmanagedBufferDataHolder() = default;
 78         virtual ~UnmanagedBufferDataHolder() = default;
 79 
 80 #if USE(GSTREAMER_GL)
 81         virtual void waitForCPUSync() = 0;
 82 #endif // USE(GSTREAMER_GL)
 83     };
 84 
 85     bool hasManagedTexture() const { return m_hasManagedTexture; }
 86     void setUnmanagedBufferDataHolder(std::unique_ptr<UnmanagedBufferDataHolder> holder) { m_unmanagedBufferDataHolder = WTFMove(holder); }
 87     void setExtraFlags(TextureMapperGL::Flags flags) { m_extraFlags = flags; }
 88 
 89     std::unique_ptr<TextureMapperPlatformLayerBuffer> clone();
 90 
 91     class HolePunchClient {
 92         WTF_MAKE_FAST_ALLOCATED();
 93     public:
 94         virtual ~HolePunchClient() = default;
 95         virtual void setVideoRectangle(const IntRect&) = 0;
 96     };
 97 
 98     void setHolePunchClient(std::unique_ptr<HolePunchClient>&& client) { m_holePunchClient = WTFMove(client); }
 99 
100     const TextureVariant& textureVariant() { return m_variant; }
101 
102 private:
103 
104     RefPtr<BitmapTexture> m_texture;
105     MonotonicTime m_timeLastUsed;
106 
107     TextureVariant m_variant;
108     IntSize m_size;
109     GLint m_internalFormat;
110     TextureMapperGL::Flags m_extraFlags;
111     bool m_hasManagedTexture;
112     std::unique_ptr<UnmanagedBufferDataHolder> m_unmanagedBufferDataHolder;
113     std::unique_ptr<HolePunchClient> m_holePunchClient;
114 };
115 
116 } // namespace WebCore
117 
118 #endif // USE(COORDINATED_GRAPHICS)
