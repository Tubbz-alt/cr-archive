 26 #pragma once
 27 
 28 #if ENABLE(PAYMENT_REQUEST)
 29 
 30 #include "ActiveDOMObject.h"
 31 #include "ContextDestructionObserver.h"
 32 #include "EventTarget.h"
 33 #include "JSValueInWrappedObject.h"
 34 #include "PaymentAddress.h"
 35 #include "PaymentComplete.h"
 36 #include <wtf/WeakPtr.h>
 37 
 38 namespace WebCore {
 39 
 40 class Document;
 41 class PaymentRequest;
 42 struct PaymentValidationErrors;
 43 
 44 template<typename IDLType> class DOMPromiseDeferred;
 45 
 46 class PaymentResponse final : public ActiveDOMObject, public EventTargetWithInlineData, public RefCounted<PaymentResponse> {
 47     WTF_MAKE_ISO_ALLOCATED(PaymentResponse);
 48 public:
 49     using DetailsFunction = Function<JSC::Strong<JSC::JSObject>(JSC::JSGlobalObject&)>;
 50 
 51     static Ref<PaymentResponse> create(ScriptExecutionContext* context, PaymentRequest& request)
 52     {
 53         auto response = adoptRef(*new PaymentResponse(context, request));
 54         response->finishConstruction();
 55         return response;
 56     }
 57 
 58     ~PaymentResponse();
 59 
 60     const String& requestId() const { return m_requestId; }
 61     void setRequestId(const String& requestId) { m_requestId = requestId; }
 62 
 63     const String& methodName() const { return m_methodName; }
 64     void setMethodName(const String& methodName) { m_methodName = methodName; }
 65 
 66     const DetailsFunction& detailsFunction() const { return m_detailsFunction; }
 67     void setDetailsFunction(DetailsFunction&&);
 68 
 69     JSValueInWrappedObject& cachedDetails() { return m_cachedDetails; }
 70 
 71     PaymentAddress* shippingAddress() const { return m_shippingAddress.get(); }
 72     void setShippingAddress(PaymentAddress* shippingAddress) { m_shippingAddress = shippingAddress; }
 73 
 74     const String& shippingOption() const { return m_shippingOption; }
 75     void setShippingOption(const String& shippingOption) { m_shippingOption = shippingOption; }
 76 
 77     const String& payerName() const { return m_payerName; }
 78     void setPayerName(const String& payerName) { m_payerName = payerName; }
 79 
 80     const String& payerEmail() const { return m_payerEmail; }
 81     void setPayerEmail(const String& payerEmail) { m_payerEmail = payerEmail; }
 82 
 83     const String& payerPhone() const { return m_payerPhone; }
 84     void setPayerPhone(const String& payerPhone) { m_payerPhone = payerPhone; }
 85 
 86     void complete(Optional<PaymentComplete>&&, DOMPromiseDeferred<void>&&);
 87     void retry(PaymentValidationErrors&&, DOMPromiseDeferred<void>&&);
 88     void abortWithException(Exception&&);
 89     bool hasRetryPromise() const { return !!m_retryPromise; }
 90     void settleRetryPromise(ExceptionOr<void>&& = { });
 91 
 92     using RefCounted<PaymentResponse>::ref;
 93     using RefCounted<PaymentResponse>::deref;
 94 
 95 private:
 96     PaymentResponse(ScriptExecutionContext*, PaymentRequest&);
 97     void finishConstruction();
 98 
 99     // ActiveDOMObject
100     const char* activeDOMObjectName() const final { return "PaymentResponse"; }
101     void stop() final;
102     void suspend(ReasonForSuspension) final;
103 
104     // EventTarget
105     EventTargetInterface eventTargetInterface() const final { return PaymentResponseEventTargetInterfaceType; }
106     ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
107     void refEventTarget() final { ref(); }
108     void derefEventTarget() final { deref(); }
109 
110     enum class State {
111         Created,
112         Completed,
113         Stopped,
114     };
115 
116     WeakPtr<PaymentRequest> m_request;
117     String m_requestId;
118     String m_methodName;
119     DetailsFunction m_detailsFunction;
120     JSValueInWrappedObject m_cachedDetails;
121     RefPtr<PaymentAddress> m_shippingAddress;
122     String m_shippingOption;
123     String m_payerName;
124     String m_payerEmail;
125     String m_payerPhone;
126     State m_state { State::Created };
127     std::unique_ptr<DOMPromiseDeferred<void>> m_retryPromise;
128     RefPtr<PendingActivity<PaymentResponse>> m_pendingActivity;
129 };
130 
131 } // namespace WebCore
132 
133 #endif // ENABLE(PAYMENT_REQUEST)
