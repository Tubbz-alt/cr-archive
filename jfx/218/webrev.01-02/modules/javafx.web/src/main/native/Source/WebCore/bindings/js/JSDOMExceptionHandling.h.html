75 // Convert a DOM implementation exception into a JavaScript exception in the execution lexicalGlobalObject.
76 WEBCORE_EXPORT void propagateExceptionSlowPath(JSC::JSGlobalObject&, JSC::ThrowScope&, Exception&&);
77 
78 ALWAYS_INLINE void propagateException(JSC::JSGlobalObject& lexicalGlobalObject, JSC::ThrowScope& throwScope, Exception&& exception)
79 {
80     if (throwScope.exception())
81         return;
82     propagateExceptionSlowPath(lexicalGlobalObject, throwScope, WTFMove(exception));
83 }
84 
85 inline void propagateException(JSC::JSGlobalObject& lexicalGlobalObject, JSC::ThrowScope& throwScope, ExceptionOr<void>&& value)
86 {
87     if (UNLIKELY(value.hasException()))
88         propagateException(lexicalGlobalObject, throwScope, value.releaseException());
89 }
