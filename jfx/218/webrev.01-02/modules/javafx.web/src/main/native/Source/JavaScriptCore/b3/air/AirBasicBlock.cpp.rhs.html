 36 const char* const BasicBlock::dumpPrefix = "#";
 37 
 38 void BasicBlock::setSuccessors(FrequentedBlock target)
 39 {
 40     m_successors.resize(1);
 41     m_successors[0] = target;
 42 }
 43 
 44 void BasicBlock::setSuccessors(FrequentedBlock taken, FrequentedBlock notTaken)
 45 {
 46     m_successors.resize(2);
 47     m_successors[0] = taken;
 48     m_successors[1] = notTaken;
 49 }
 50 
 51 bool BasicBlock::addPredecessor(BasicBlock* block)
 52 {
 53     return B3::addPredecessor(this, block);
 54 }
 55 
 56 bool BasicBlock::removePredecessor(BasicBlock* block)
 57 {
 58     return B3::removePredecessor(this, block);
 59 }
 60 
 61 bool BasicBlock::replacePredecessor(BasicBlock* from, BasicBlock* to)
 62 {
 63     return B3::replacePredecessor(this, from, to);
 64 }
 65 
 66 void BasicBlock::dump(PrintStream& out) const
 67 {
 68     out.print(dumpPrefix, m_index);
 69 }
 70 
 71 void BasicBlock::deepDump(PrintStream& out) const
 72 {
 73     dumpHeader(out);
 74     for (const Inst& inst : *this)
 75         out.print(tierName, "    ", inst, "\n");
 76     dumpFooter(out);
 77 }
 78 
 79 void BasicBlock::dumpHeader(PrintStream& out) const
 80 {
 81     out.print(tierName, "BB", *this, ": ; frequency = ", m_frequency, "\n");
 82     if (predecessors().size())
 83         out.print(tierName, "  Predecessors: ", pointerListDump(predecessors()), "\n");
 84 }
 85 
 86 void BasicBlock::dumpFooter(PrintStream& out) const
 87 {
 88     if (successors().size())
 89         out.print(tierName, "  Successors: ", listDump(successors()), "\n");
 90 }
