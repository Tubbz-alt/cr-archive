21 #include "JSOffscreenCanvasRenderingContext2D.h"

23 namespace WebCore {
24 using namespace JSC;
25 
26 inline void* root(OffscreenCanvas* canvas)
27 {
28     return canvas;
29 }
30 
31 bool JSOffscreenCanvasRenderingContext2DOwner::isReachableFromOpaqueRoots(JSC::Handle<JSC::Unknown> handle, void*, SlotVisitor& visitor, const char** reason)
32 {
33     if (UNLIKELY(reason))
34         *reason = "Canvas is opaque root";
35 
36     JSOffscreenCanvasRenderingContext2D* jsOffscreenCanvasRenderingContext = jsCast<JSOffscreenCanvasRenderingContext2D*>(handle.slot()->asCell());
37     void* root = WebCore::root(&jsOffscreenCanvasRenderingContext->wrapped().canvas());
38     return visitor.containsOpaqueRoot(root);
39 }
40 
41 void JSOffscreenCanvasRenderingContext2D::visitAdditionalChildren(SlotVisitor& visitor)
42 {
43     visitor.addOpaqueRoot(root(&wrapped().canvas()));
44 }
