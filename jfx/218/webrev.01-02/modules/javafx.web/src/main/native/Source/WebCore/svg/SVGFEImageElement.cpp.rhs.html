 41 inline SVGFEImageElement::SVGFEImageElement(const QualifiedName& tagName, Document& document)
 42     : SVGFilterPrimitiveStandardAttributes(tagName, document)

 43     , SVGURIReference(this)
 44 {
 45     ASSERT(hasTagName(SVGNames::feImageTag));
 46 
 47     static std::once_flag onceFlag;
 48     std::call_once(onceFlag, [] {
 49         PropertyRegistry::registerProperty<SVGNames::preserveAspectRatioAttr, &SVGFEImageElement::m_preserveAspectRatio>();
 50     });
 51 }

 55     return adoptRef(*new SVGFEImageElement(tagName, document));
 56 }

 58 SVGFEImageElement::~SVGFEImageElement()
 59 {

 61 }

 63 bool SVGFEImageElement::hasSingleSecurityOrigin() const
 64 {
 65     if (!m_cachedImage)
 66         return true;
 67     auto* image = m_cachedImage->image();
 68     return !image || image->hasSingleSecurityOrigin();
 69 }
 70 
 71 void SVGFEImageElement::clearResourceReferences()
 72 {
 73     if (m_cachedImage) {
 74         m_cachedImage->removeClient(*this);
 75         m_cachedImage = nullptr;
 76     }
 77 
 78     document().accessSVGExtensions().removeAllTargetReferencesForElement(*this);
 79 }
 80 
 81 void SVGFEImageElement::requestImageResource()
 82 {
 83     ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
 84     options.contentSecurityPolicyImposition = isInUserAgentShadowTree() ? ContentSecurityPolicyImposition::SkipPolicyCheck : ContentSecurityPolicyImposition::DoPolicyCheck;
 85 
 86     CachedResourceRequest request(ResourceRequest(document().completeURL(href())), options);
 87     request.setInitiator(*this);
 88     m_cachedImage = document().cachedResourceLoader().requestImage(WTFMove(request)).value_or(nullptr);
 89 
 90     if (m_cachedImage)
 91         m_cachedImage->addClient(*this);
 92 }
 93 
 94 void SVGFEImageElement::buildPendingResource()
 95 {
 96     clearResourceReferences();
 97     if (!isConnected())
 98         return;
 99 
100     auto target = SVGURIReference::targetElementFromIRIString(href(), treeScope());
101     if (!target.element) {
102         if (target.identifier.isEmpty())
103             requestImageResource();
104         else {
105             document().accessSVGExtensions().addPendingResource(target.identifier, *this);
106             ASSERT(hasPendingResources());
107         }
108     } else if (is<SVGElement>(*target.element)) {
109         // Register us with the target in the dependencies map. Any change of hrefElement
110         // that leads to relayout/repainting now informs us, so we can react to it.
111         document().accessSVGExtensions().addElementReferencingTarget(*this, downcast<SVGElement>(*target.element));
112     }
113 
114     invalidate();
115 }

117 void SVGFEImageElement::parseAttribute(const QualifiedName& name, const AtomString& value)
118 {
119     if (name == SVGNames::preserveAspectRatioAttr) {
120         SVGPreserveAspectRatioValue preserveAspectRatio;
121         preserveAspectRatio.parse(value);
122         m_preserveAspectRatio->setBaseValInternal(preserveAspectRatio);
123         return;
124     }
125 
126     SVGFilterPrimitiveStandardAttributes::parseAttribute(name, value);
127     SVGURIReference::parseAttribute(name, value);

128 }
129 
130 void SVGFEImageElement::svgAttributeChanged(const QualifiedName& attrName)
131 {
132     if (attrName == SVGNames::preserveAspectRatioAttr) {
133         InstanceInvalidationGuard guard(*this);
134         invalidate();
135         return;
136     }
137 
138     if (SVGURIReference::isKnownAttribute(attrName)) {
139         InstanceInvalidationGuard guard(*this);
140         buildPendingResource();
141         return;
142     }
143 
144     SVGFilterPrimitiveStandardAttributes::svgAttributeChanged(attrName);
145 }

165 void SVGFEImageElement::notifyFinished(CachedResource&)
166 {
167     if (!isConnected())
168         return;
169 
170     auto parent = makeRefPtr(parentElement());
171 
172     if (!parent || !parent->hasTagName(SVGNames::filterTag))
173         return;
174 
175     RenderElement* parentRenderer = parent->renderer();
176     if (!parentRenderer)
177         return;
178 
179     RenderSVGResource::markForLayoutAndParentResourceInvalidation(*parentRenderer);
180 }
181 
182 RefPtr<FilterEffect> SVGFEImageElement::build(SVGFilterBuilder*, Filter& filter) const
183 {
184     if (m_cachedImage)
185         return FEImage::createWithImage(filter, m_cachedImage->imageForRenderer(renderer()), preserveAspectRatio());
186 
187     auto target = SVGURIReference::targetElementFromIRIString(href(), treeScope());
188     if (isDescendantOrShadowDescendantOf(target.element.get()))
189         return nullptr;
190 
191     return FEImage::createWithIRIReference(filter, treeScope(), href(), preserveAspectRatio());
192 }
