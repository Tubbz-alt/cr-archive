43 class SMILTimeContainer final : public RefCounted<SMILTimeContainer>  {
44     WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(SMILTimeContainer);
45 public:
46     static Ref<SMILTimeContainer> create(SVGSVGElement& owner) { return adoptRef(*new SMILTimeContainer(owner)); }
47 
48     void schedule(SVGSMILElement*, SVGElement*, const QualifiedName&);
49     void unschedule(SVGSMILElement*, SVGElement*, const QualifiedName&);
50     void notifyIntervalsChanged();
51 
52     WEBCORE_EXPORT Seconds animationFrameDelay() const;
53 
54     SMILTime elapsed() const;
55 
56     bool isActive() const;
57     bool isPaused() const;
58     bool isStarted() const;
59 
60     void begin();
61     void pause();
62     void resume();
63     void setElapsed(SMILTime);
64 
65     void setDocumentOrderIndexesDirty() { m_documentOrderIndexesDirty = true; }
66 
67 private:
68     SMILTimeContainer(SVGSVGElement& owner);
69 
70     void timerFired();
71     void startTimer(SMILTime elapsed, SMILTime fireTime, SMILTime minimumDelay = 0);
72     void updateAnimations(SMILTime elapsed, bool seekToTime = false);
73 
74     typedef std::pair<SVGElement*, QualifiedName> ElementAttributePair;
75     typedef Vector<SVGSMILElement*> AnimationsVector;
76     typedef HashMap<ElementAttributePair, std::unique_ptr<AnimationsVector>> GroupedAnimationsMap;
77 
78     void processAnimations(const AnimationsVector&, Function<void(SVGSMILElement*)>&&);
79     void processScheduledAnimations(Function<void(SVGSMILElement*)>&&);
80     void updateDocumentOrderIndexes();
81     void sortByPriority(AnimationsVector& smilElements, SMILTime elapsed);
82 
83     MonotonicTime m_beginTime;
84     MonotonicTime m_pauseTime;
85     Seconds m_accumulatedActiveTime { 0_s };
86     MonotonicTime m_resumeTime;
87     Seconds m_presetStartTime { 0_s };
88 
89     bool m_documentOrderIndexesDirty { false };
90     Timer m_timer;
91     GroupedAnimationsMap m_scheduledAnimations;
92     SVGSVGElement& m_ownerSVGElement;
93 };
