 48 DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(BlockDirectory);
 49 
 50 class BlockDirectory {
 51     WTF_MAKE_NONCOPYABLE(BlockDirectory);
 52     WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(BlockDirectory);
 53 
 54     friend class LLIntOffsetsExtractor;
 55 
 56 public:
 57     BlockDirectory(size_t cellSize);
 58     ~BlockDirectory();
 59     void setSubspace(Subspace*);
 60     void lastChanceToFinalize();
 61     void prepareForAllocation();
 62     void stopAllocating();
 63     void stopAllocatingForGood();
 64     void resumeAllocating();
 65     void beginMarkingForFullCollection();
 66     void endMarking();
 67     void snapshotUnsweptForEdenCollection();
 68     void snapshotUnsweptForFullCollection();
 69     void sweep();
 70     void shrink();
 71     void assertNoUnswept();
 72     size_t cellSize() const { return m_cellSize; }
 73     const CellAttributes& attributes() const { return m_attributes; }
 74     bool needsDestruction() const { return m_attributes.destruction == NeedsDestruction; }
 75     DestructionMode destruction() const { return m_attributes.destruction; }
 76     HeapCell::Kind cellKind() const { return m_attributes.cellKind; }
 77 
 78     bool isFreeListedCell(const void* target);
 79 
 80     template<typename Functor> void forEachBlock(const Functor&);
 81     template<typename Functor> void forEachNotEmptyBlock(const Functor&);
 82 
 83     RefPtr<SharedTask<MarkedBlock::Handle*()>> parallelNotEmptyBlockSource();
 84 
 85     void addBlock(MarkedBlock::Handle*);
 86     void removeBlock(MarkedBlock::Handle*);
 87 
 88     bool isPagedOut(MonotonicTime deadline);
 89 
 90     Lock& bitvectorLock() { return m_bitvectorLock; }
 91 
 92 #define BLOCK_DIRECTORY_BIT_ACCESSORS(lowerBitName, capitalBitName)     \
 93     bool is ## capitalBitName(const AbstractLocker&, size_t index) const { return m_bits.is ## capitalBitName(index); } \
 94     bool is ## capitalBitName(const AbstractLocker& locker, MarkedBlock::Handle* block) const { return is ## capitalBitName(locker, block->index()); } \
 95     void setIs ## capitalBitName(const AbstractLocker&, size_t index, bool value) { m_bits.setIs ## capitalBitName(index, value); } \
 96     void setIs ## capitalBitName(const AbstractLocker& locker, MarkedBlock::Handle* block, bool value) { setIs ## capitalBitName(locker, block->index(), value); }
 97     FOR_EACH_BLOCK_DIRECTORY_BIT(BLOCK_DIRECTORY_BIT_ACCESSORS)
 98 #undef BLOCK_DIRECTORY_BIT_ACCESSORS
 99 
100     template<typename Func>
101     void forEachBitVector(const AbstractLocker&, const Func& func)
102     {
103 #define BLOCK_DIRECTORY_BIT_CALLBACK(lowerBitName, capitalBitName) \
104         func(m_bits.lowerBitName());
105         FOR_EACH_BLOCK_DIRECTORY_BIT(BLOCK_DIRECTORY_BIT_CALLBACK);
106 #undef BLOCK_DIRECTORY_BIT_CALLBACK
107     }
108 
109     template<typename Func>
110     void forEachBitVectorWithName(const AbstractLocker&, const Func& func)
111     {
112 #define BLOCK_DIRECTORY_BIT_CALLBACK(lowerBitName, capitalBitName) \
113         func(m_bits.lowerBitName(), #capitalBitName);
114         FOR_EACH_BLOCK_DIRECTORY_BIT(BLOCK_DIRECTORY_BIT_CALLBACK);
115 #undef BLOCK_DIRECTORY_BIT_CALLBACK
116     }
117 
118     BlockDirectory* nextDirectory() const { return m_nextDirectory; }
119     BlockDirectory* nextDirectoryInSubspace() const { return m_nextDirectoryInSubspace; }
120     BlockDirectory* nextDirectoryInAlignedMemoryAllocator() const { return m_nextDirectoryInAlignedMemoryAllocator; }
121 
122     void setNextDirectory(BlockDirectory* directory) { m_nextDirectory = directory; }
123     void setNextDirectoryInSubspace(BlockDirectory* directory) { m_nextDirectoryInSubspace = directory; }
124     void setNextDirectoryInAlignedMemoryAllocator(BlockDirectory* directory) { m_nextDirectoryInAlignedMemoryAllocator = directory; }
125 
126     MarkedBlock::Handle* findEmptyBlockToSteal();
127 
128     MarkedBlock::Handle* findBlockToSweep();
129 
130     Subspace* subspace() const { return m_subspace; }
131     MarkedSpace& markedSpace() const;
132 
133     void dump(PrintStream&) const;
134     void dumpBits(PrintStream& = WTF::dataFile());
135 
136 private:
137     friend class IsoCellSet;
138     friend class LocalAllocator;
139     friend class LocalSideAllocator;
140     friend class MarkedBlock;
141 
142     MarkedBlock::Handle* findBlockForAllocation(LocalAllocator&);
143 
144     MarkedBlock::Handle* tryAllocateBlock(Heap&);
145 
146     Vector<MarkedBlock::Handle*> m_blocks;
147     Vector<unsigned> m_freeBlockIndices;
148 
149     // Mutator uses this to guard resizing the bitvectors. Those things in the GC that may run
150     // concurrently to the mutator must lock this when accessing the bitvectors.
151     BlockDirectoryBits m_bits;
152     Lock m_bitvectorLock;
153     Lock m_localAllocatorsLock;
154     CellAttributes m_attributes;
155 
156     unsigned m_cellSize;
157 
158     // After you do something to a block based on one of these cursors, you clear the bit in the
159     // corresponding bitvector and leave the cursor where it was. We can use unsigned instead of size_t since
160     // this number is bound by capacity of Vector m_blocks, which must be within unsigned.
161     unsigned m_emptyCursor { 0 };
162     unsigned m_unsweptCursor { 0 }; // Points to the next block that is a candidate for incremental sweeping.
163 
164     // FIXME: All of these should probably be references.
165     // https://bugs.webkit.org/show_bug.cgi?id=166988
166     Subspace* m_subspace { nullptr };
167     BlockDirectory* m_nextDirectory { nullptr };
168     BlockDirectory* m_nextDirectoryInSubspace { nullptr };
169     BlockDirectory* m_nextDirectoryInAlignedMemoryAllocator { nullptr };
170 
171     SentinelLinkedList<LocalAllocator, BasicRawSentinelNode<LocalAllocator>> m_localAllocators;
172 };

174 } // namespace JSC
