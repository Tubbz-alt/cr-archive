26 #pragma once
27 
28 #include <wtf/FastMalloc.h>
29 
30 namespace JSC {
31 
32 struct ClassInfo;
33 
34 namespace DOMJIT {
35 class GetterSetter;
36 }
37 
38 struct DOMAttributeAnnotation {
39     WTF_MAKE_FAST_ALLOCATED;
40 public:
41     const ClassInfo* classInfo;
42     const DOMJIT::GetterSetter* domJIT;
43 };
44 
45 inline bool operator==(const DOMAttributeAnnotation& left, const DOMAttributeAnnotation& right)
46 {
47     return left.classInfo == right.classInfo && left.domJIT == right.domJIT;
48 }
49 
50 inline bool operator!=(const DOMAttributeAnnotation& left, const DOMAttributeAnnotation& right)
51 {
52     return !(left == right);
53 }
54 
55 } // namespace JSC
