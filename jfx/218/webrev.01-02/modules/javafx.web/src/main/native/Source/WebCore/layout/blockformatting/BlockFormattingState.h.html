36 // BlockFormattingState holds the state for a particular block formatting context tree.
37 class BlockFormattingState : public FormattingState {
38     WTF_MAKE_ISO_ALLOCATED(BlockFormattingState);
39 public:
40     BlockFormattingState(Ref<FloatingState>&&, LayoutState&);
41     ~BlockFormattingState();
42 
43     void setPositiveAndNegativeVerticalMargin(const Box& layoutBox, PositiveAndNegativeVerticalMargin verticalMargin) { m_positiveAndNegativeVerticalMargin.set(&layoutBox, verticalMargin); }
44     bool hasPositiveAndNegativeVerticalMargin(const Box& layoutBox) const { return m_positiveAndNegativeVerticalMargin.contains(&layoutBox); }
45     PositiveAndNegativeVerticalMargin positiveAndNegativeVerticalMargin(const Box& layoutBox) const { return m_positiveAndNegativeVerticalMargin.get(&layoutBox); }
46 
47 private:
48     HashMap<const Box*, PositiveAndNegativeVerticalMargin> m_positiveAndNegativeVerticalMargin;
49 };
50 
51 }
52 }
