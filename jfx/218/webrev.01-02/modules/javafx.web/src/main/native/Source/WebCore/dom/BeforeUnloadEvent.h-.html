25 #pragma once
26 
27 #include "Event.h"
28 
29 namespace WebCore {
30 
31 class BeforeUnloadEvent final : public Event {
32 public:
33     static Ref<BeforeUnloadEvent> create()
34     {
35         return adoptRef(*new BeforeUnloadEvent);
36     }
37 
38     static Ref<BeforeUnloadEvent> createForBindings()
39     {
40         return adoptRef(*new BeforeUnloadEvent { ForBindings });
41     }
42 
43     String returnValue() const { return m_returnValue; }
44     void setReturnValue(const String& returnValue) { m_returnValue = returnValue; }
45 
46 private:
47     enum ForBindingsFlag { ForBindings };
48     BeforeUnloadEvent();
49     BeforeUnloadEvent(ForBindingsFlag);
50 
51     EventInterface eventInterface() const final { return BeforeUnloadEventInterfaceType; }
52     bool isBeforeUnloadEvent() const final;
53 
54     String m_returnValue;
55 };
56 
57 } // namespace WebCore
58 
59 SPECIALIZE_TYPE_TRAITS_EVENT(BeforeUnloadEvent)
