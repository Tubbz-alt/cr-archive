 47 Ref<ScriptProcessorNode> ScriptProcessorNode::create(AudioContext& context, float sampleRate, size_t bufferSize, unsigned numberOfInputChannels, unsigned numberOfOutputChannels)
 48 {
 49     return adoptRef(*new ScriptProcessorNode(context, sampleRate, bufferSize, numberOfInputChannels, numberOfOutputChannels));

 52 ScriptProcessorNode::ScriptProcessorNode(AudioContext& context, float sampleRate, size_t bufferSize, unsigned numberOfInputChannels, unsigned numberOfOutputChannels)
 53     : AudioNode(context, sampleRate)
 54     , ActiveDOMObject(context.scriptExecutionContext())
 55     , m_doubleBufferIndex(0)
 56     , m_doubleBufferIndexForEvent(0)
 57     , m_bufferSize(bufferSize)
 58     , m_bufferReadWriteIndex(0)
 59     , m_isRequestOutstanding(false)
 60     , m_numberOfInputChannels(numberOfInputChannels)
 61     , m_numberOfOutputChannels(numberOfOutputChannels)
 62     , m_internalInputBus(AudioBus::create(numberOfInputChannels, AudioNode::ProcessingSizeInFrames, false))
 63 {
 64     // Regardless of the allowed buffer sizes, we still need to process at the granularity of the AudioNode.
 65     if (m_bufferSize < AudioNode::ProcessingSizeInFrames)
 66         m_bufferSize = AudioNode::ProcessingSizeInFrames;
 67 
 68     ASSERT(numberOfInputChannels <= AudioContext::maxNumberOfChannels());
 69 
 70     setNodeType(NodeTypeJavaScript);
 71     addInput(makeUnique<AudioNodeInput>(this));
 72     addOutput(makeUnique<AudioNodeOutput>(this, numberOfOutputChannels));
 73 
 74     initialize();
 75     suspendIfNeeded();
 76     m_pendingActivity = makePendingActivity(*this);
 77 }
 78 
 79 ScriptProcessorNode::~ScriptProcessorNode()
 80 {
 81     ASSERT(!hasPendingActivity());
 82     uninitialize();
 83 }
 84 
 85 void ScriptProcessorNode::initialize()
 86 {
 87     if (isInitialized())
 88         return;
 89 
 90     float sampleRate = context().sampleRate();
 91 
 92     // Create double buffers on both the input and output sides.
 93     // These AudioBuffers will be directly accessed in the main thread by JavaScript.
 94     for (unsigned i = 0; i < 2; ++i) {
 95         auto inputBuffer = m_numberOfInputChannels ? AudioBuffer::create(m_numberOfInputChannels, bufferSize(), sampleRate) : 0;
 96         auto outputBuffer = m_numberOfOutputChannels ? AudioBuffer::create(m_numberOfOutputChannels, bufferSize(), sampleRate) : 0;
 97 
 98         m_inputBuffers.append(inputBuffer);
 99         m_outputBuffers.append(outputBuffer);
100     }
101 
102     AudioNode::initialize();
103 }
104 
105 void ScriptProcessorNode::uninitialize()
106 {
107     if (!isInitialized())
108         return;
109 
110     m_inputBuffers.clear();
111     m_outputBuffers.clear();
112 
113     AudioNode::uninitialize();
114 }

116 void ScriptProcessorNode::didBecomeMarkedForDeletion()
117 {
118     ASSERT(context().isGraphOwner());
119     m_pendingActivity = nullptr;
120     ASSERT(!hasPendingActivity());
121 }

123 void ScriptProcessorNode::process(size_t framesToProcess)
124 {
125     // Discussion about inputs and outputs:
126     // As in other AudioNodes, ScriptProcessorNode uses an AudioBus for its input and output (see inputBus and outputBus below).
127     // Additionally, there is a double-buffering for input and output which is exposed directly to JavaScript (see inputBuffer and outputBuffer below).
128     // This node is the producer for inputBuffer and the consumer for outputBuffer.
129     // The JavaScript code is the consumer of inputBuffer and the producer for outputBuffer.
130 
131     // Get input and output busses.
132     AudioBus* inputBus = this->input(0)->bus();
133     AudioBus* outputBus = this->output(0)->bus();
134 
135     // Get input and output buffers. We double-buffer both the input and output sides.
136     unsigned doubleBufferIndex = this->doubleBufferIndex();
137     bool isDoubleBufferIndexGood = doubleBufferIndex < 2 && doubleBufferIndex < m_inputBuffers.size() && doubleBufferIndex < m_outputBuffers.size();
138     ASSERT(isDoubleBufferIndexGood);
139     if (!isDoubleBufferIndexGood)
140         return;
141 
142     AudioBuffer* inputBuffer = m_inputBuffers[doubleBufferIndex].get();
143     AudioBuffer* outputBuffer = m_outputBuffers[doubleBufferIndex].get();
144 
145     // Check the consistency of input and output buffers.
146     unsigned numberOfInputChannels = m_internalInputBus->numberOfChannels();
147     bool buffersAreGood = outputBuffer && bufferSize() == outputBuffer->length() && m_bufferReadWriteIndex + framesToProcess <= bufferSize();
148 
149     // If the number of input channels is zero, it's ok to have inputBuffer = 0.
150     if (m_internalInputBus->numberOfChannels())
151         buffersAreGood = buffersAreGood && inputBuffer && bufferSize() == inputBuffer->length();
152 
153     ASSERT(buffersAreGood);
154     if (!buffersAreGood)
155         return;
156 
157     // We assume that bufferSize() is evenly divisible by framesToProcess - should always be true, but we should still check.
158     bool isFramesToProcessGood = framesToProcess && bufferSize() >= framesToProcess && !(bufferSize() % framesToProcess);
159     ASSERT(isFramesToProcessGood);
160     if (!isFramesToProcessGood)
161         return;
162 
163     unsigned numberOfOutputChannels = outputBus->numberOfChannels();
164 
165     bool channelsAreGood = (numberOfInputChannels == m_numberOfInputChannels) && (numberOfOutputChannels == m_numberOfOutputChannels);
166     ASSERT(channelsAreGood);
167     if (!channelsAreGood)
168         return;
169 
170     for (unsigned i = 0; i < numberOfInputChannels; i++)
171         m_internalInputBus->setChannelMemory(i, inputBuffer->channelData(i)->data() + m_bufferReadWriteIndex, framesToProcess);
172 
173     if (numberOfInputChannels)
174         m_internalInputBus->copyFrom(*inputBus);
175 
176     // Copy from the output buffer to the output.
177     for (unsigned i = 0; i < numberOfOutputChannels; ++i)
178         memcpy(outputBus->channel(i)->mutableData(), outputBuffer->channelData(i)->data() + m_bufferReadWriteIndex, sizeof(float) * framesToProcess);
179 
180     // Update the buffering index.
181     m_bufferReadWriteIndex = (m_bufferReadWriteIndex + framesToProcess) % bufferSize();
182 
183     // m_bufferReadWriteIndex will wrap back around to 0 when the current input and output buffers are full.
184     // When this happens, fire an event and swap buffers.
185     if (!m_bufferReadWriteIndex) {
186         // Avoid building up requests on the main thread to fire process events when they're not being handled.
187         // This could be a problem if the main thread is very busy doing other things and is being held up handling previous requests.
188         if (m_isRequestOutstanding) {
189             // We're late in handling the previous request. The main thread must be very busy.
190             // The best we can do is clear out the buffer ourself here.
191             outputBuffer->zero();
192         } else {
193             // Reference ourself so we don't accidentally get deleted before fireProcessEvent() gets called.
194             ref();
195 
196             // Fire the event on the main thread, not this one (which is the realtime audio thread).
197             m_doubleBufferIndexForEvent = m_doubleBufferIndex;
198             m_isRequestOutstanding = true;
199 
200             callOnMainThread([this] {
201                 fireProcessEvent();
202 
203                 // De-reference to match the ref() call in process().
204                 deref();
205             });
206         }
207 
208         swapBuffers();
209     }
210 }
211 
212 void ScriptProcessorNode::fireProcessEvent()
213 {
214     ASSERT(isMainThread() && m_isRequestOutstanding);
215 
216     bool isIndexGood = m_doubleBufferIndexForEvent < 2;
217     ASSERT(isIndexGood);
218     if (!isIndexGood)
219         return;
220 
221     AudioBuffer* inputBuffer = m_inputBuffers[m_doubleBufferIndexForEvent].get();
222     AudioBuffer* outputBuffer = m_outputBuffers[m_doubleBufferIndexForEvent].get();
223     ASSERT(outputBuffer);
224     if (!outputBuffer)
225         return;
226 
227     // Avoid firing the event if the document has already gone away.
228     if (!context().isStopped()) {
229         // Let the audio thread know we've gotten to the point where it's OK for it to make another request.
230         m_isRequestOutstanding = false;
231 
232         // Calculate playbackTime with the buffersize which needs to be processed each time when onaudioprocess is called.
233         // The outputBuffer being passed to JS will be played after exhausting previous outputBuffer by double-buffering.
234         double playbackTime = (context().currentSampleFrame() + m_bufferSize) / static_cast<double>(context().sampleRate());
235 
236         // Call the JavaScript event handler which will do the audio processing.
237         dispatchEvent(AudioProcessingEvent::create(inputBuffer, outputBuffer, playbackTime));
238     }
239 }

241 void ScriptProcessorNode::reset()
242 {
243     m_bufferReadWriteIndex = 0;
244     m_doubleBufferIndex = 0;
245 
246     for (unsigned i = 0; i < 2; ++i) {
247         m_inputBuffers[i]->zero();
248         m_outputBuffers[i]->zero();
249     }
250 }
