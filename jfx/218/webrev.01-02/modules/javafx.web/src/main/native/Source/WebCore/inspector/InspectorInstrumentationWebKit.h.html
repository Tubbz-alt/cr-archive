26 #pragma once
27 
28 #include "InspectorInstrumentationPublic.h"
29 #include <wtf/CompletionHandler.h>
30 
31 namespace WebCore {
32 
33 class Frame;
34 class ResourceResponse;
35 class SharedBuffer;
36 
37 class WEBCORE_EXPORT InspectorInstrumentationWebKit {
38 public:
39     static bool shouldInterceptResponse(const Frame*, const ResourceResponse&);
40     static void interceptResponse(const Frame*, const ResourceResponse&, unsigned long identifier, CompletionHandler<void(const ResourceResponse&, RefPtr<SharedBuffer>)>&&);
41 
42 private:
43     static bool shouldInterceptResponseInternal(const Frame&, const ResourceResponse&);
44     static void interceptResponseInternal(const Frame&, const ResourceResponse&, unsigned long identifier, CompletionHandler<void(const ResourceResponse&, RefPtr<SharedBuffer>)>&&);
45 };
46 
47 inline bool InspectorInstrumentationWebKit::shouldInterceptResponse(const Frame* frame, const ResourceResponse& response)
48 {
49     FAST_RETURN_IF_NO_FRONTENDS(false);
50     if (!frame)
51         return false;
52 
53     return shouldInterceptResponseInternal(*frame, response);
54 }
55 
56 inline void InspectorInstrumentationWebKit::interceptResponse(const Frame* frame, const ResourceResponse& response, unsigned long identifier, CompletionHandler<void(const ResourceResponse&, RefPtr<SharedBuffer>)>&& handler)
57 {
58     ASSERT(InspectorInstrumentationWebKit::shouldInterceptResponse(frame, response));
59     interceptResponseInternal(*frame, response, identifier, WTFMove(handler));
60 }
61 
62 }
