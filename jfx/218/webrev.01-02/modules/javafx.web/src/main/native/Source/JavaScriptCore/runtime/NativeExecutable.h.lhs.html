26 #pragma once
27 
28 #include "ExecutableBase.h"
29 
30 namespace JSC {
31 
32 class NativeExecutable final : public ExecutableBase {
33     friend class JIT;
34     friend class LLIntOffsetsExtractor;
35 public:
36     typedef ExecutableBase Base;
37     static const unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
38 
39     static NativeExecutable* create(VM&, Ref<JITCode>&& callThunk, TaggedNativeFunction, Ref<JITCode>&& constructThunk, TaggedNativeFunction constructor, const String& name);
40 
41     static void destroy(JSCell*);
42 
43     template<typename CellType, SubspaceAccess>
44     static IsoSubspace* subspaceFor(VM& vm)
45     {
46         return &vm.nativeExecutableSpace;
47     }
48 
49     CodeBlockHash hashFor(CodeSpecializationKind) const;
50 
51     TaggedNativeFunction function() { return m_function; }
52     TaggedNativeFunction constructor() { return m_constructor; }
53 
54     TaggedNativeFunction nativeFunctionFor(CodeSpecializationKind kind)
55     {
56         if (kind == CodeForCall)
57             return function();
58         ASSERT(kind == CodeForConstruct);
59         return constructor();
60     }
61 
62     static ptrdiff_t offsetOfNativeFunctionFor(CodeSpecializationKind kind)
63     {
64         if (kind == CodeForCall)
65             return OBJECT_OFFSETOF(NativeExecutable, m_function);
66         ASSERT(kind == CodeForConstruct);
67         return OBJECT_OFFSETOF(NativeExecutable, m_constructor);
68     }
69 
70     static Structure* createStructure(VM&, JSGlobalObject*, JSValue proto);
71 
72     DECLARE_INFO;
73 
74     const String& name() const { return m_name; }
75 
76     const DOMJIT::Signature* signatureFor(CodeSpecializationKind) const;
77     Intrinsic intrinsic() const;
78 
79 protected:
80     void finishCreation(VM&, Ref<JITCode>&& callThunk, Ref<JITCode>&& constructThunk, const String& name);
81 
82 private:
83     NativeExecutable(VM&, TaggedNativeFunction, TaggedNativeFunction constructor);
84 
85     TaggedNativeFunction m_function;
86     TaggedNativeFunction m_constructor;
87 
88     String m_name;
89 };
90 
91 } // namespace JSC
