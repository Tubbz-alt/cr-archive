33 namespace JSC {
34 
35 class CallLinkInfo;
36 
37 namespace Wasm {
38 
39 class LLIntCallee;
40 class EmbedderEntrypointCallee;
41 
42 using EmbedderEntrypointCalleeMap = HashMap<uint32_t, RefPtr<EmbedderEntrypointCallee>, typename DefaultHash<uint32_t>::Hash, WTF::UnsignedWithZeroKeyHashTraits<uint32_t>>;
43 
44 class LLIntPlan final : public EntryPlan {
45     using Base = EntryPlan;
46 
47 public:
48     JS_EXPORT_PRIVATE LLIntPlan(Context*, Vector<uint8_t>&&, AsyncWork, CompletionTask&&);
49     LLIntPlan(Context*, Ref<ModuleInformation>, const Ref<LLIntCallee>*, CompletionTask&&);
50 
51     MacroAssemblerCodeRef<B3CompilationPtrTag>&& takeEntryThunks()
52     {
53         RELEASE_ASSERT(!failed() && !hasWork());
54         return WTFMove(m_entryThunks);
55     }
56 
57     Vector<Ref<LLIntCallee>>&& takeCallees()
58     {
59         RELEASE_ASSERT(!failed() && !hasWork());
60         return WTFMove(m_calleesVector);
61     }
62 
63     EmbedderEntrypointCalleeMap&& takeEmbedderCallees()
64     {
65         RELEASE_ASSERT(!failed() && !hasWork());
66         return WTFMove(m_embedderCallees);
67     }
68 
69     bool hasWork() const override
70     {
71         return m_state < State::Compiled;
72     }
73 
74     void work(CompilationEffort) override;
75 
76     bool didReceiveFunctionData(unsigned, const FunctionData&) override;
77 
78 protected:
79     bool prepareImpl() override;
80     void compileFunction(uint32_t functionIndex) override;
81     void didCompleteCompilation(const AbstractLocker&) override;
82 
83 private:
84     Vector<std::unique_ptr<FunctionCodeBlock>> m_wasmInternalFunctions;
85     const Ref<LLIntCallee>* m_callees { nullptr };
86     Vector<Ref<LLIntCallee>> m_calleesVector;
87     EmbedderEntrypointCalleeMap m_embedderCallees;
88     MacroAssemblerCodeRef<B3CompilationPtrTag> m_entryThunks;
89 };
90 
91 
92 } } // namespace JSC::Wasm
