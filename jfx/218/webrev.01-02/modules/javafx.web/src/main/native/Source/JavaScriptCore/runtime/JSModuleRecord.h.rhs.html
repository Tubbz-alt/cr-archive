26 #pragma once
27 
28 #include "AbstractModuleRecord.h"
29 #include "SourceCode.h"
30 #include "VariableEnvironment.h"
31 
32 namespace JSC {
33 
34 class ModuleProgramExecutable;
35 
36 // Based on the Source Text Module Record
37 // http://www.ecma-international.org/ecma-262/6.0/#sec-source-text-module-records
38 class JSModuleRecord final : public AbstractModuleRecord {
39     friend class LLIntOffsetsExtractor;
40 public:
41     using Base = AbstractModuleRecord;
42 
43     DECLARE_EXPORT_INFO;
44 
45     static constexpr bool needsDestruction = true;
46     static void destroy(JSCell*);
47 
48     template<typename CellType, SubspaceAccess mode>
49     static IsoSubspace* subspaceFor(VM& vm)
50     {
51         return vm.jsModuleRecordSpace<mode>();
52     }
53 
54     static Structure* createStructure(VM&, JSGlobalObject*, JSValue);
55     static JSModuleRecord* create(JSGlobalObject*, VM&, Structure*, const Identifier&, const SourceCode&, const VariableEnvironment&, const VariableEnvironment&);
56 
57     void link(JSGlobalObject*, JSValue scriptFetcher);
58     JS_EXPORT_PRIVATE JSValue evaluate(JSGlobalObject*);
59 
60     const SourceCode& sourceCode() const { return m_sourceCode; }
61     const VariableEnvironment& declaredVariables() const { return m_declaredVariables; }
62     const VariableEnvironment& lexicalVariables() const { return m_lexicalVariables; }
63 
64 private:
65     JSModuleRecord(VM&, Structure*, const Identifier&, const SourceCode&, const VariableEnvironment&, const VariableEnvironment&);
66 
67     void finishCreation(JSGlobalObject*, VM&);
68 
69     static void visitChildren(JSCell*, SlotVisitor&);

70 
71     void instantiateDeclarations(JSGlobalObject*, ModuleProgramExecutable*, JSValue scriptFetcher);
72 
73     SourceCode m_sourceCode;
74     VariableEnvironment m_declaredVariables;
75     VariableEnvironment m_lexicalVariables;
76     WriteBarrier<ModuleProgramExecutable> m_moduleProgramExecutable;
77 };
78 
79 } // namespace JSC
