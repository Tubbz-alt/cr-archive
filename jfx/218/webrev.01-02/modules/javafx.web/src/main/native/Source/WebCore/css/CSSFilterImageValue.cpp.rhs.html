 42 CSSFilterImageValue::~CSSFilterImageValue()
 43 {
 44     if (m_cachedImage)
 45         m_cachedImage->removeClient(m_filterSubimageObserver);
 46 }

 53 FloatSize CSSFilterImageValue::fixedSize(const RenderElement& renderer)
 54 {
 55     // FIXME: Skip Content Security Policy check when filter is applied to an element in a user agent shadow tree.
 56     // See <https://bugs.webkit.org/show_bug.cgi?id=146663>.
 57     ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
 58 
 59     CachedResourceLoader& cachedResourceLoader = renderer.document().cachedResourceLoader();
 60     CachedImage* cachedImage = cachedImageForCSSValue(m_imageValue, cachedResourceLoader, options);
 61 
 62     if (!cachedImage)
 63         return FloatSize();
 64 
 65     return cachedImage->imageForRenderer(&renderer)->size();
 66 }

 78 void CSSFilterImageValue::loadSubimages(CachedResourceLoader& cachedResourceLoader, const ResourceLoaderOptions& options)
 79 {
 80     CachedResourceHandle<CachedImage> oldCachedImage = m_cachedImage;
 81 
 82     m_cachedImage = CSSImageGeneratorValue::cachedImageForCSSValue(m_imageValue, cachedResourceLoader, options);
 83 
 84     if (m_cachedImage != oldCachedImage) {
 85         if (oldCachedImage)
 86             oldCachedImage->removeClient(m_filterSubimageObserver);
 87         if (m_cachedImage)
 88             m_cachedImage->addClient(m_filterSubimageObserver);
 89     }
 90 
 91     for (auto& filterOperation : m_filterOperations.operations()) {
 92         if (!is<ReferenceFilterOperation>(filterOperation))
 93             continue;
 94         auto& referenceFilterOperation = downcast<ReferenceFilterOperation>(*filterOperation);
 95         referenceFilterOperation.loadExternalDocumentIfNeeded(cachedResourceLoader, options);
 96     }
 97 
 98     m_filterSubimageObserver.setReady(true);
 99 }

101 RefPtr<Image> CSSFilterImageValue::image(RenderElement& renderer, const FloatSize& size)
102 {


103     if (size.isEmpty())
104         return nullptr;
105 
106     // FIXME: Skip Content Security Policy check when filter is applied to an element in a user agent shadow tree.
107     // See <https://bugs.webkit.org/show_bug.cgi?id=146663>.
108     ResourceLoaderOptions options = CachedResourceLoader::defaultCachedResourceOptions();
109     auto* cachedImage = cachedImageForCSSValue(m_imageValue, renderer.document().cachedResourceLoader(), options);
110     if (!cachedImage)
111         return &Image::nullImage();
112 
113     auto* image = cachedImage->imageForRenderer(&renderer);
114     if (!image)
115         return &Image::nullImage();
116 
117     // Transform Image into ImageBuffer.
118     // FIXME (149424): This buffer should not be unconditionally unaccelerated.
119     auto texture = ImageBuffer::create(size, RenderingMode::Unaccelerated);
120     if (!texture)
121         return &Image::nullImage();
122 
123     auto imageRect = FloatRect { { }, size };
124     texture->context().drawImage(*image, imageRect);
125 
126     auto cssFilter = CSSFilter::create();
127     cssFilter->setSourceImage(WTFMove(texture));
128     cssFilter->setSourceImageRect(imageRect);
129     cssFilter->setFilterRegion(imageRect);
130     if (!cssFilter->build(renderer, m_filterOperations, FilterConsumer::FilterFunction))
131         return &Image::nullImage();
132     cssFilter->apply();
133 
134     return cssFilter->output()->copyImage();
135 }

137 void CSSFilterImageValue::filterImageChanged(const IntRect&)
138 {
139     for (auto& client : clients())
140         client.key->imageChanged(static_cast<WrappedImagePtr>(this));
141 }
142 
143 void CSSFilterImageValue::createFilterOperations(Style::BuilderState& builderState)
144 {
145     m_filterOperations.clear();
146     builderState.createFilterOperations(m_filterValue, m_filterOperations);
147 }

149 void CSSFilterImageValue::FilterSubimageObserverProxy::imageChanged(CachedImage*, const IntRect* rect)
150 {
151     if (m_ready)
152         m_ownerValue->filterImageChanged(*rect);
153 }
154 
155 bool CSSFilterImageValue::traverseSubresources(const WTF::Function<bool (const CachedResource&)>& handler) const
156 {
157     if (!m_cachedImage)
158         return false;
159     return handler(*m_cachedImage);
160 }
161 
162 bool CSSFilterImageValue::equals(const CSSFilterImageValue& other) const
163 {
164     return equalInputImages(other) && compareCSSValue(m_filterValue, other.m_filterValue);
165 }
166 
167 bool CSSFilterImageValue::equalInputImages(const CSSFilterImageValue& other) const
168 {
169     return compareCSSValue(m_imageValue, other.m_imageValue);
170 }
