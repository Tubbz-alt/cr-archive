33 void JITBitXorGenerator::generateFastPath(CCallHelpers& jit)
34 {
35     ASSERT(!m_leftOperand.isConstInt32() || !m_rightOperand.isConstInt32());
36 
37     m_didEmitFastPath = true;
38 
39     if (m_leftOperand.isConstInt32() || m_rightOperand.isConstInt32()) {
40         JSValueRegs var = m_leftOperand.isConstInt32() ? m_right : m_left;
41         SnippetOperand& constOpr = m_leftOperand.isConstInt32() ? m_leftOperand : m_rightOperand;
42 
43         // Try to do intVar ^ intConstant.
44         m_slowPathJumpList.append(jit.branchIfNotInt32(var));
45 
46         jit.moveValueRegs(var, m_result);
47 #if USE(JSVALUE64)
48         jit.xor32(CCallHelpers::Imm32(constOpr.asConstInt32()), m_result.payloadGPR());
49         jit.or64(GPRInfo::tagTypeNumberRegister, m_result.payloadGPR());
50 #else
51         jit.xor32(CCallHelpers::Imm32(constOpr.asConstInt32()), m_result.payloadGPR());
52 #endif
53 
54     } else {
55         ASSERT(!m_leftOperand.isConstInt32() && !m_rightOperand.isConstInt32());
56 
57         // Try to do intVar ^ intVar.
58         m_slowPathJumpList.append(jit.branchIfNotInt32(m_left));
59         m_slowPathJumpList.append(jit.branchIfNotInt32(m_right));
60 
61         jit.moveValueRegs(m_left, m_result);
62 #if USE(JSVALUE64)
63         jit.xor64(m_right.payloadGPR(), m_result.payloadGPR());
64         jit.or64(GPRInfo::tagTypeNumberRegister, m_result.payloadGPR());
65 #else
66         jit.xor32(m_right.payloadGPR(), m_result.payloadGPR());
67 #endif
68     }
69 }
70 
71 } // namespace JSC
