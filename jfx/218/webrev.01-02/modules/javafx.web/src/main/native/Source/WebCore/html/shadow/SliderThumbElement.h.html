 34 #include "HTMLDivElement.h"
 35 #include "RenderBlockFlow.h"
 36 #include <wtf/Forward.h>

 38 namespace WebCore {
 39 
 40 class HTMLInputElement;
 41 class TouchEvent;
 42 
 43 class SliderThumbElement final : public HTMLDivElement {
 44     WTF_MAKE_ISO_ALLOCATED(SliderThumbElement);
 45 public:
 46     static Ref<SliderThumbElement> create(Document&);
 47 
 48     void setPositionFromValue();
 49     void dragFrom(const LayoutPoint&);
 50     RefPtr<HTMLInputElement> hostInput() const;
 51     void setPositionFromPoint(const LayoutPoint&);
 52 
 53 #if ENABLE(IOS_TOUCH_EVENTS)
 54     void handleTouchEvent(TouchEvent&);
 55 #endif
 56 
 57     void hostDisabledStateChanged();
 58 
 59 private:
 60     SliderThumbElement(Document&);
 61 
 62     RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) final;
 63 
 64     Ref<Element> cloneElementWithoutAttributesAndChildren(Document&) final;
 65     bool isDisabledFormControl() const final;
 66     bool matchesReadWritePseudoClass() const final;
 67     RefPtr<Element> focusDelegate() final;
 68 
 69     void defaultEventHandler(Event&) final;
 70     bool willRespondToMouseMoveEvents() final;
 71     bool willRespondToMouseClickEvents() final;
 72 
 73 #if ENABLE(IOS_TOUCH_EVENTS)
 74     void didAttachRenderers() final;
 75 #endif
 76     void willDetachRenderers() final;
 77 
 78     Optional<Style::ElementStyle> resolveCustomStyle(const RenderStyle&, const RenderStyle*) final;
 79     const AtomString& shadowPseudoId() const final;
 80 
 81     void startDragging();
 82     void stopDragging();
 83 
 84 #if ENABLE(IOS_TOUCH_EVENTS)
 85     unsigned exclusiveTouchIdentifier() const;
 86     void setExclusiveTouchIdentifier(unsigned);
 87     void clearExclusiveTouchIdentifier();
 88 
 89     void handleTouchStart(TouchEvent&);
 90     void handleTouchMove(TouchEvent&);
 91     void handleTouchEndAndCancel(TouchEvent&);
 92 
 93     bool shouldAcceptTouchEvents();
 94     void registerForTouchEvents();
 95     void unregisterForTouchEvents();
 96 #endif
 97 
 98     AtomString m_shadowPseudoId;
 99     bool m_inDragMode { false };
100 
101 #if ENABLE(IOS_TOUCH_EVENTS)
102     // FIXME: Currently it is safe to use 0, but this may need to change
103     // if touch identifiers change in the future and can be 0.
104     static const unsigned NoIdentifier = 0;
105     unsigned m_exclusiveTouchIdentifier { NoIdentifier };
106     bool m_isRegisteredAsTouchEventListener { false };
107 #endif
108 };
109 
110 inline Ref<SliderThumbElement> SliderThumbElement::create(Document& document)
111 {
112     return adoptRef(*new SliderThumbElement(document));
113 }

129 class SliderContainerElement final : public HTMLDivElement {
130     WTF_MAKE_ISO_ALLOCATED(SliderContainerElement);
131 public:
132     static Ref<SliderContainerElement> create(Document&);
133 
134 private:
135     SliderContainerElement(Document&);
136     RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) final;
137     Optional<Style::ElementStyle> resolveCustomStyle(const RenderStyle&, const RenderStyle*) final;
138     const AtomString& shadowPseudoId() const final;
139     bool isSliderContainerElement() const final { return true; }
140 
141     AtomString m_shadowPseudoId;
142 };
