26 #pragma once
27 
28 #if ENABLE(WEBGL)
29 
30 #include "GraphicsTypesGL.h"
31 #include <wtf/RefCounted.h>
32 
33 namespace WebCore {
34 
35 class GraphicsContextGLOpenGL;
36 class WebGLContextGroup;
37 class WebGLRenderingContextBase;
38 
39 class WebGLObject : public RefCounted<WebGLObject> {
40 public:
41     virtual ~WebGLObject() = default;
42 
43     PlatformGLObject object() const { return m_object; }
44 
45     // deleteObject may not always delete the OpenGL resource.  For programs and
46     // shaders, deletion is delayed until they are no longer attached.
47     // FIXME: revisit this when resource sharing between contexts are implemented.
48     void deleteObject(GraphicsContextGLOpenGL*);
49 
50     void onAttached() { ++m_attachmentCount; }
51     void onDetached(GraphicsContextGLOpenGL*);
52 
53     // This indicates whether the client side issue a delete call already, not
54     // whether the OpenGL resource is deleted.
55     // object()==0 indicates the OpenGL resource is deleted.
56     bool isDeleted() { return m_deleted; }
57 
58     // True if this object belongs to the group or context.
59     virtual bool validate(const WebGLContextGroup*, const WebGLRenderingContextBase&) const = 0;
60 
61 protected:
62     WebGLObject() = default;
63 
64     // setObject should be only called once right after creating a WebGLObject.
65     void setObject(PlatformGLObject);
66 
67     // deleteObjectImpl should be only called once to delete the OpenGL resource.
68     virtual void deleteObjectImpl(GraphicsContextGLOpenGL*, PlatformGLObject) = 0;
69 
70     virtual bool hasGroupOrContext() const = 0;
71 
72     virtual void detach();
73 
74     virtual GraphicsContextGLOpenGL* getAGraphicsContextGL() const = 0;
75 
76 private:
77     PlatformGLObject m_object { 0 };
78     unsigned m_attachmentCount { 0 };
79     bool m_deleted { false };
80 };
81 
82 inline PlatformGLObject objectOrZero(WebGLObject* object)
83 {
84     return object ? object->object() : 0;
85 }
86 
87 } // namespace WebCore
88 
89 #endif
