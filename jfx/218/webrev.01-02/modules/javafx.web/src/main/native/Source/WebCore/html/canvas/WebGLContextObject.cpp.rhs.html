26 #include "config.h"
27 #include "WebGLContextObject.h"
28 
29 #if ENABLE(WEBGL)
30 
31 #include "WebGLRenderingContextBase.h"
32 
33 namespace WebCore {
34 
35 WebGLContextObject::WebGLContextObject(WebGLRenderingContextBase& context)
36     : m_context(&context)
37 {
38 }
39 
40 WebGLContextObject::~WebGLContextObject()
41 {
42     if (m_context)
43         m_context->removeContextObject(*this);
44 }
45 
46 void WebGLContextObject::detachContext()
47 {
48     detach();
49     if (m_context) {
50         deleteObject(m_context->graphicsContextGL());
51         m_context->removeContextObject(*this);
52         m_context = nullptr;
53     }
54 }
55 
56 GraphicsContextGLOpenGL* WebGLContextObject::getAGraphicsContextGL() const
57 {
58     return m_context ? m_context->graphicsContextGL() : nullptr;
59 }
60 
61 }

63 #endif // ENABLE(WEBGL)
