28 #if ENABLE(WEBGL)

31 #include "ExtensionsGL.h"
32 #include "WebGLRenderingContextBase.h"

34 namespace WebCore {
35 
36 WebGLCompressedTextureETC1::WebGLCompressedTextureETC1(WebGLRenderingContextBase& context)
37     : WebGLExtension(context)
38 {
39     context.graphicsContextGL()->getExtensions().ensureEnabled("GL_OES_compressed_ETC1_RGB8_texture");
40 
41     context.addCompressedTextureFormat(ExtensionsGL::ETC1_RGB8_OES);
42 }
43 
44 WebGLCompressedTextureETC1::~WebGLCompressedTextureETC1() = default;
45 
46 WebGLExtension::ExtensionName WebGLCompressedTextureETC1::getName() const
47 {
48     return WebGLCompressedTextureETC1Name;
49 }
50 
51 bool WebGLCompressedTextureETC1::supported(WebGLRenderingContextBase& context)
52 {
53     return context.graphicsContextGL()->getExtensions().supports("GL_OES_compressed_ETC1_RGB8_texture");
54 }
55 
56 } // namespace WebCore

58 #endif // ENABLE(WEBGL)
