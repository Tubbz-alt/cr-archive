 38     static Ref<SVGPathElement> create(const QualifiedName&, Document&);
 39 
 40     static Ref<SVGPathSegClosePath> createSVGPathSegClosePath() { return SVGPathSegClosePath::create(); }
 41     static Ref<SVGPathSegMovetoAbs> createSVGPathSegMovetoAbs(float x, float y) { return SVGPathSegMovetoAbs::create(x, y); }
 42     static Ref<SVGPathSegMovetoRel> createSVGPathSegMovetoRel(float x, float y) { return SVGPathSegMovetoRel::create(x, y); }
 43     static Ref<SVGPathSegLinetoAbs> createSVGPathSegLinetoAbs(float x, float y) { return SVGPathSegLinetoAbs::create(x, y); }
 44     static Ref<SVGPathSegLinetoRel> createSVGPathSegLinetoRel(float x, float y) { return SVGPathSegLinetoRel::create(x, y); }
 45     static Ref<SVGPathSegCurvetoCubicAbs> createSVGPathSegCurvetoCubicAbs(float x, float y, float x1, float y1, float x2, float y2)
 46     {
 47         return SVGPathSegCurvetoCubicAbs::create(x, y, x1, y1, x2, y2);
 48     }
 49     static Ref<SVGPathSegCurvetoCubicRel> createSVGPathSegCurvetoCubicRel(float x, float y, float x1, float y1, float x2, float y2)
 50     {
 51         return SVGPathSegCurvetoCubicRel::create(x, y, x1, y1, x2, y2);
 52     }
 53     static Ref<SVGPathSegCurvetoQuadraticAbs> createSVGPathSegCurvetoQuadraticAbs(float x, float y, float x1, float y1)
 54     {
 55         return SVGPathSegCurvetoQuadraticAbs::create(x, y, x1, y1);
 56     }
 57     static Ref<SVGPathSegCurvetoQuadraticRel> createSVGPathSegCurvetoQuadraticRel(float x, float y, float x1, float y1)
 58     {
 59         return SVGPathSegCurvetoQuadraticRel::create(x, y, x1, y1);
 60     }
 61     static Ref<SVGPathSegArcAbs> createSVGPathSegArcAbs(float x, float y, float r1, float r2, float angle, bool largeArcFlag, bool sweepFlag)
 62     {
 63         return SVGPathSegArcAbs::create(x, y, r1, r2, angle, largeArcFlag, sweepFlag);
 64     }
 65     static Ref<SVGPathSegArcRel> createSVGPathSegArcRel(float x, float y, float r1, float r2, float angle, bool largeArcFlag, bool sweepFlag)
 66     {
 67         return SVGPathSegArcRel::create(x, y, r1, r2, angle, largeArcFlag, sweepFlag);
 68     }
 69     static Ref<SVGPathSegLinetoHorizontalAbs> createSVGPathSegLinetoHorizontalAbs(float x) { return SVGPathSegLinetoHorizontalAbs::create(x); }
 70     static Ref<SVGPathSegLinetoHorizontalRel> createSVGPathSegLinetoHorizontalRel(float x) { return SVGPathSegLinetoHorizontalRel::create(x); }
 71     static Ref<SVGPathSegLinetoVerticalAbs> createSVGPathSegLinetoVerticalAbs(float y) { return SVGPathSegLinetoVerticalAbs::create(y); }
 72     static Ref<SVGPathSegLinetoVerticalRel> createSVGPathSegLinetoVerticalRel(float y) { return SVGPathSegLinetoVerticalRel::create(y); }
 73     static Ref<SVGPathSegCurvetoCubicSmoothAbs> createSVGPathSegCurvetoCubicSmoothAbs(float x, float y, float x2, float y2)
 74     {
 75         return SVGPathSegCurvetoCubicSmoothAbs::create(x, y, x2, y2);
 76     }
 77     static Ref<SVGPathSegCurvetoCubicSmoothRel> createSVGPathSegCurvetoCubicSmoothRel(float x, float y, float x2, float y2)
 78     {
 79         return SVGPathSegCurvetoCubicSmoothRel::create(x, y, x2, y2);
 80     }
 81     static Ref<SVGPathSegCurvetoQuadraticSmoothAbs> createSVGPathSegCurvetoQuadraticSmoothAbs(float x, float y)
 82     {
 83         return SVGPathSegCurvetoQuadraticSmoothAbs::create(x, y);
 84     }
 85     static Ref<SVGPathSegCurvetoQuadraticSmoothRel> createSVGPathSegCurvetoQuadraticSmoothRel(float x, float y)
 86     {
 87         return SVGPathSegCurvetoQuadraticSmoothRel::create(x, y);
 88     }
 89 
 90     float getTotalLength() const final;

 92     unsigned getPathSegAtLength(float distance) const;
 93 
 94     FloatRect getBBox(StyleUpdateStrategy = AllowStyleUpdate) final;
 95 
 96     Ref<SVGPathSegList>& pathSegList() { return m_pathSegList->baseVal(); }
 97     RefPtr<SVGPathSegList>& animatedPathSegList() { return m_pathSegList->animVal(); }
 98

102

106 
107 private:
108     SVGPathElement(const QualifiedName&, Document&);
109 
110     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGPathElement, SVGGeometryElement>;
111     const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
112 
113     void parseAttribute(const QualifiedName&, const AtomString&) final;
114     void svgAttributeChanged(const QualifiedName&) final;
115 
116     bool isValid() const final { return SVGTests::isValid(); }
117     bool supportsMarkers() const final { return true; }
118 
119     RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) final;
120 
121     Node::InsertedIntoAncestorResult insertedIntoAncestor(InsertionType, ContainerNode&) final;
122     void removedFromAncestor(RemovalType, ContainerNode&) final;
123 
124     void invalidateMPathDependencies();
125 
126 private:
127     PropertyRegistry m_propertyRegistry { *this };
128     Ref<SVGAnimatedPathSegList> m_pathSegList { SVGAnimatedPathSegList::create(this) };
129 };
