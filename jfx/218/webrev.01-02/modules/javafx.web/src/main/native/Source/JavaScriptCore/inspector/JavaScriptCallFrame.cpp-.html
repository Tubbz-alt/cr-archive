31 namespace Inspector {
32 
33 JavaScriptCallFrame::JavaScriptCallFrame(Ref<DebuggerCallFrame>&& debuggerCallFrame)
34     : m_debuggerCallFrame(WTFMove(debuggerCallFrame))
35 {
36 }
37 
38 JavaScriptCallFrame* JavaScriptCallFrame::caller()
39 {
40     if (m_caller)
41         return m_caller.get();
42 
43     RefPtr<DebuggerCallFrame> debuggerCallerFrame = m_debuggerCallFrame->callerFrame();
44     if (!debuggerCallerFrame)
45         return nullptr;
46 
47     m_caller = create(debuggerCallerFrame.releaseNonNull());
48     return m_caller.get();
49 }
