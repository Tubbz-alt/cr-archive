 63 class CSSCalcExpressionNode : public RefCounted<CSSCalcExpressionNode> {
 64 public:
 65     enum Type {
 66         CssCalcPrimitiveValue = 1,
 67         CssCalcOperation,
 68         CssCalcNegate,
 69         CssCalcInvert,
 70     };
 71 
 72     virtual ~CSSCalcExpressionNode() = default;
 73     virtual bool isZero() const = 0;
 74     virtual std::unique_ptr<CalcExpressionNode> createCalcExpression(const CSSToLengthConversionData&) const = 0;
 75     virtual double doubleValue(CSSUnitType) const = 0;
 76     virtual double computeLengthPx(const CSSToLengthConversionData&) const = 0;
 77     virtual bool equals(const CSSCalcExpressionNode& other) const { return m_category == other.m_category; }

 78     virtual Type type() const = 0;
 79     virtual CSSUnitType primitiveType() const = 0;
 80 
 81     virtual void collectDirectComputationalDependencies(HashSet<CSSPropertyID>&) const = 0;
 82     virtual void collectDirectRootComputationalDependencies(HashSet<CSSPropertyID>&) const = 0;
 83 
 84     CalculationCategory category() const { return m_category; }
 85 
 86     virtual void dump(TextStream&) const = 0;
 87 
 88 protected:
 89     CSSCalcExpressionNode(CalculationCategory category)
 90         : m_category(category)

 91     {
 92     }
 93 
 94 private:
 95     CalculationCategory m_category;

 96 };
 97 
 98 class CSSCalcValue final : public CSSValue {
 99 public:
100     static RefPtr<CSSCalcValue> create(CSSValueID function, const CSSParserTokenRange&, CalculationCategory destinationCategory, ValueRange);
101 
102     static RefPtr<CSSCalcValue> create(const CalculationValue&, const RenderStyle&);
103 
104     CalculationCategory category() const { return m_expression->category(); }

105     double doubleValue() const;


106     double computeLengthPx(const CSSToLengthConversionData&) const;
107     CSSUnitType primitiveType() const { return m_expression->primitiveType(); }
108 
109     Ref<CalculationValue> createCalculationValue(const CSSToLengthConversionData&) const;
110     void setPermittedValueRange(ValueRange);
111 
112     void collectDirectComputationalDependencies(HashSet<CSSPropertyID>&) const;
113     void collectDirectRootComputationalDependencies(HashSet<CSSPropertyID>&) const;
114 
115     String customCSSText() const;
116     bool equals(const CSSCalcValue&) const;
117 
118     static bool isCalcFunction(CSSValueID);
119 
120     void dump(TextStream&) const;
121 
122 private:
123     CSSCalcValue(Ref<CSSCalcExpressionNode>&&, bool shouldClampToNonNegative);
124 
125     double clampToPermittedRange(double) const;
126 
127     const Ref<CSSCalcExpressionNode> m_expression;
128     bool m_shouldClampToNonNegative;
129 };

131 inline CSSCalcValue::CSSCalcValue(Ref<CSSCalcExpressionNode>&& expression, bool shouldClampToNonNegative)
132     : CSSValue(CalculationClass)
133     , m_expression(WTFMove(expression))
134     , m_shouldClampToNonNegative(shouldClampToNonNegative)
135 {
136 }
137 
138 inline Ref<CalculationValue> CSSCalcValue::createCalculationValue(const CSSToLengthConversionData& conversionData) const
139 {
140     return CalculationValue::create(m_expression->createCalcExpression(conversionData),
141         m_shouldClampToNonNegative ? ValueRangeNonNegative : ValueRangeAll);
142 }
143 
144 inline void CSSCalcValue::setPermittedValueRange(ValueRange range)
145 {
146     m_shouldClampToNonNegative = range != ValueRangeAll;
147 }
148 
149 inline void CSSCalcValue::collectDirectComputationalDependencies(HashSet<CSSPropertyID>& values) const
150 {
151     m_expression->collectDirectComputationalDependencies(values);
152 }
153 
154 inline void CSSCalcValue::collectDirectRootComputationalDependencies(HashSet<CSSPropertyID>& values) const
155 {
156     m_expression->collectDirectRootComputationalDependencies(values);
157 }
