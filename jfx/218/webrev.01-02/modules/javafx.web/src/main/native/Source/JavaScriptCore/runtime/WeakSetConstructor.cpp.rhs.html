41 void WeakSetConstructor::finishCreation(VM& vm, WeakSetPrototype* prototype)
42 {
43     Base::finishCreation(vm, "WeakSet"_s, NameAdditionMode::WithoutStructureTransition);
44     putDirectWithoutTransition(vm, vm.propertyNames->prototype, prototype, PropertyAttribute::DontEnum | PropertyAttribute::DontDelete | PropertyAttribute::ReadOnly);
45     putDirectWithoutTransition(vm, vm.propertyNames->length, jsNumber(0), PropertyAttribute::DontEnum | PropertyAttribute::ReadOnly);
46 }

51 WeakSetConstructor::WeakSetConstructor(VM& vm, Structure* structure)
52     : Base(vm, structure, callWeakSet, constructWeakSet)
53 {
54 }

56 static EncodedJSValue JSC_HOST_CALL callWeakSet(JSGlobalObject* globalObject, CallFrame*)
57 {
58     VM& vm = globalObject->vm();
59     auto scope = DECLARE_THROW_SCOPE(vm);
60     return JSValue::encode(throwConstructorCannotBeCalledAsFunctionTypeError(globalObject, scope, "WeakSet"));
61 }
62 
63 static EncodedJSValue JSC_HOST_CALL constructWeakSet(JSGlobalObject* globalObject, CallFrame* callFrame)
64 {
65     VM& vm = globalObject->vm();
66     auto scope = DECLARE_THROW_SCOPE(vm);
67 
68     Structure* weakSetStructure = InternalFunction::createSubclassStructure(globalObject, callFrame->jsCallee(), callFrame->newTarget(), globalObject->weakSetStructure());

69     RETURN_IF_EXCEPTION(scope, encodedJSValue());
70     JSWeakSet* weakSet = JSWeakSet::create(vm, weakSetStructure);
71     JSValue iterable = callFrame->argument(0);
72     if (iterable.isUndefinedOrNull())
73         return JSValue::encode(weakSet);
74 
75     JSValue adderFunction = weakSet->JSObject::get(globalObject, vm.propertyNames->add);
76     RETURN_IF_EXCEPTION(scope, encodedJSValue());
77 
78     CallData adderFunctionCallData;
79     CallType adderFunctionCallType = getCallData(vm, adderFunction, adderFunctionCallData);
80     if (adderFunctionCallType == CallType::None)
81         return JSValue::encode(throwTypeError(globalObject, scope));
82 
83     scope.release();
84     forEachInIterable(globalObject, iterable, [&](VM&, JSGlobalObject* globalObject, JSValue nextValue) {
85         MarkedArgumentBuffer arguments;
86         arguments.append(nextValue);
87         ASSERT(!arguments.hasOverflowed());
88         call(globalObject, adderFunction, adderFunctionCallType, adderFunctionCallData, weakSet, arguments);
89     });
90 
91     return JSValue::encode(weakSet);
92 }
