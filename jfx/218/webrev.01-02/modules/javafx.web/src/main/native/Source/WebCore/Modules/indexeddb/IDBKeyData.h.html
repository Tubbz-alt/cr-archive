 42 public:

 45         , m_isNull(true)

 47     }
 48

 50

 52     IDBKeyData(const IDBKeyData&, IsolatedCopyTag);
 53 
 54     static IDBKeyData minimum()
 55     {
 56         IDBKeyData result;
 57         result.m_type = IndexedDB::KeyType::Min;
 58         result.m_isNull = false;
 59         return result;
 60     }
 61 
 62     static IDBKeyData maximum()
 63     {
 64         IDBKeyData result;
 65         result.m_type = IndexedDB::KeyType::Max;
 66         result.m_isNull = false;
 67         return result;
 68     }
 69 
 70     WEBCORE_EXPORT RefPtr<IDBKey> maybeCreateIDBKey() const;
 71 
 72     WEBCORE_EXPORT IDBKeyData isolatedCopy() const;
 73 
 74     WEBCORE_EXPORT void encode(KeyedEncoder&) const;
 75     WEBCORE_EXPORT static bool decode(KeyedDecoder&, IDBKeyData&);
 76 
 77     // compare() has the same semantics as strcmp().
 78     //   - Returns negative if this IDBKeyData is less than other.
 79     //   - Returns positive if this IDBKeyData is greater than other.
 80     //   - Returns zero if this IDBKeyData is equal to other.
 81     WEBCORE_EXPORT int compare(const IDBKeyData& other) const;
 82

114     }
115

117     bool operator!=(const IDBKeyData& other) const
118     {
119         return !(*this == other);
120     }
121 
122     unsigned hash() const
123     {
124         Vector<unsigned> hashCodes;
125         hashCodes.append(static_cast<unsigned>(m_type));
126         hashCodes.append(m_isNull ? 1 : 0);
127         hashCodes.append(m_isDeletedValue ? 1 : 0);
128         switch (m_type) {
129         case IndexedDB::KeyType::Invalid:
130         case IndexedDB::KeyType::Max:
131         case IndexedDB::KeyType::Min:
132             break;
133         case IndexedDB::KeyType::Number:
134         case IndexedDB::KeyType::Date:
135             hashCodes.append(StringHasher::hashMemory<sizeof(double)>(&WTF::get<double>(m_value)));
136             break;
137         case IndexedDB::KeyType::String:
138             hashCodes.append(StringHash::hash(WTF::get<String>(m_value)));
139             break;
140         case IndexedDB::KeyType::Binary: {
141             auto* data = WTF::get<ThreadSafeDataBuffer>(m_value).data();
142             if (!data)
143                 hashCodes.append(0);
144             else
145                 hashCodes.append(StringHasher::hashMemory(data->data(), data->size()));
146             break;
147         }
148         case IndexedDB::KeyType::Array:
149             for (auto& key : WTF::get<Vector<IDBKeyData>>(m_value))
150                 hashCodes.append(key.hash());
151             break;
152         }
153 
154         return StringHasher::hashMemory(hashCodes.data(), hashCodes.size() * sizeof(unsigned));
155     }
156 
157     static IDBKeyData deletedValue();
158     bool isDeletedValue() const { return m_isDeletedValue; }
159 
160     String string() const
161     {
162         ASSERT(m_type == IndexedDB::KeyType::String);
163         return WTF::get<String>(m_value);
164     }
165 
166     double date() const
167     {
168         ASSERT(m_type == IndexedDB::KeyType::Date);
169         return WTF::get<double>(m_value);
170     }
171 
172     double number() const
173     {
174         ASSERT(m_type == IndexedDB::KeyType::Number);
175         return WTF::get<double>(m_value);
176     }
177 
178     const ThreadSafeDataBuffer& binary() const
179     {
180         ASSERT(m_type == IndexedDB::KeyType::Binary);
181         return WTF::get<ThreadSafeDataBuffer>(m_value);
182     }
183 
184     const Vector<IDBKeyData>& array() const
185     {
186         ASSERT(m_type == IndexedDB::KeyType::Array);
187         return WTF::get<Vector<IDBKeyData>>(m_value);
188     }
189 
190     size_t size() const;
191 
192 private:
193     static void isolatedCopy(const IDBKeyData& source, IDBKeyData& destination);
194 
195     IndexedDB::KeyType m_type;
196     Variant<Vector<IDBKeyData>, String, double, ThreadSafeDataBuffer> m_value;
197 
198     bool m_isNull { false };
199     bool m_isDeletedValue { false };
200 };

234 template<class Encoder>
235 void IDBKeyData::encode(Encoder& encoder) const
236 {
237     encoder << m_isNull;
238     if (m_isNull)
239         return;
240 
241     encoder.encodeEnum(m_type);
242 
243     switch (m_type) {
244     case IndexedDB::KeyType::Invalid:
245     case IndexedDB::KeyType::Max:
246     case IndexedDB::KeyType::Min:
247         break;
248     case IndexedDB::KeyType::Array:
249         encoder << WTF::get<Vector<IDBKeyData>>(m_value);
250         break;
251     case IndexedDB::KeyType::Binary:
252         encoder << WTF::get<ThreadSafeDataBuffer>(m_value);
253         break;
254     case IndexedDB::KeyType::String:
255         encoder << WTF::get<String>(m_value);
256         break;
257     case IndexedDB::KeyType::Date:
258     case IndexedDB::KeyType::Number:
259         encoder << WTF::get<double>(m_value);
260         break;
261     }
262 }

264 template<class Decoder>
265 Optional<IDBKeyData> IDBKeyData::decode(Decoder& decoder)
266 {
267     IDBKeyData keyData;
268     if (!decoder.decode(keyData.m_isNull))
269         return WTF::nullopt;
270 
271     if (keyData.m_isNull)
272         return keyData;
273 
274     if (!decoder.decodeEnum(keyData.m_type))
275         return WTF::nullopt;
276 
277     switch (keyData.m_type) {
278     case IndexedDB::KeyType::Invalid:
279     case IndexedDB::KeyType::Max:
280     case IndexedDB::KeyType::Min:
281         break;
282     case IndexedDB::KeyType::Array:
283         keyData.m_value = Vector<IDBKeyData>();
284         if (!decoder.decode(WTF::get<Vector<IDBKeyData>>(keyData.m_value)))
285             return WTF::nullopt;
286         break;
287     case IndexedDB::KeyType::Binary:
288         keyData.m_value = ThreadSafeDataBuffer();
289         if (!decoder.decode(WTF::get<ThreadSafeDataBuffer>(keyData.m_value)))
290             return WTF::nullopt;
291         break;
292     case IndexedDB::KeyType::String:
293         keyData.m_value = String();
294         if (!decoder.decode(WTF::get<String>(keyData.m_value)))
295             return WTF::nullopt;
296         break;
297     case IndexedDB::KeyType::Date:
298     case IndexedDB::KeyType::Number:
299         keyData.m_value = 0.0;
300         if (!decoder.decode(WTF::get<double>(keyData.m_value)))
301             return WTF::nullopt;
302         break;
303     }
304 
305     return keyData;
306 }
