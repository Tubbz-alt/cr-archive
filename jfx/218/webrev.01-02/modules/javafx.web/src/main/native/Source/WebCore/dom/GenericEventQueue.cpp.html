 39 MainThreadGenericEventQueue::MainThreadGenericEventQueue(EventTarget& owner)
 40     : ActiveDOMObject(owner.scriptExecutionContext())
 41     , m_owner(owner)
 42     , m_taskQueue(makeUniqueRef<GenericTaskQueue<Timer>>())
 43 {
 44 }
 45 
 46 void MainThreadGenericEventQueue::enqueueEvent(RefPtr<Event>&& event)
 47 {
 48     if (m_isClosed)
 49         return;
 50 
 51     if (event->target() == &m_owner)
 52         event->setTarget(nullptr);
 53 
 54     m_pendingEvents.append(WTFMove(event));
 55 
 56     if (isSuspendedOrPausedByClient())
 57         return;
 58 
 59     m_taskQueue->enqueueTask(std::bind(&MainThreadGenericEventQueue::dispatchOneEvent, this));
 60 }
 61 
 62 void MainThreadGenericEventQueue::dispatchOneEvent()
 63 {
 64     ASSERT(!m_pendingEvents.isEmpty());
 65 
 66     Ref<EventTarget> protect(m_owner);
 67     RefPtr<Event> event = m_pendingEvents.takeFirst();
 68     EventTarget& target = event->target() ? *event->target() : m_owner;
 69     ASSERT_WITH_MESSAGE(!target.scriptExecutionContext()->activeDOMObjectsAreStopped(),
 70         "An attempt to dispatch an event on a stopped target by EventTargetInterface=%d (nodeName=%s target=%p owner=%p)",
 71         m_owner.eventTargetInterface(), m_owner.isNode() ? static_cast<Node&>(m_owner).nodeName().ascii().data() : "", &target, &m_owner);
 72     target.dispatchEvent(*event);
 73 }
 74 
 75 void MainThreadGenericEventQueue::close()
 76 {
 77     m_isClosed = true;
 78 
 79     m_taskQueue->close();
 80     m_pendingEvents.clear();
 81 }
 82 
 83 void MainThreadGenericEventQueue::cancelAllEvents()
 84 {
 85     m_taskQueue->cancelAllTasks();
 86     m_pendingEvents.clear();
 87 }
 88 
 89 bool MainThreadGenericEventQueue::hasPendingEvents() const
 90 {
 91     return !m_pendingEvents.isEmpty();
 92 }
 93 
 94 bool MainThreadGenericEventQueue::hasPendingEventsOfType(const AtomString& type) const
 95 {
 96     for (auto& event : m_pendingEvents) {
 97         if (event->type() == type)
 98             return true;
 99     }
100 
101     return false;
102 }
103 
104 void MainThreadGenericEventQueue::setPaused(bool shouldPause)
105 {
106     if (m_isPausedByClient == shouldPause)
107         return;
108 
109     m_isPausedByClient = shouldPause;
110     if (shouldPause)
111         m_taskQueue->cancelAllTasks();
112     else
113         rescheduleAllEventsIfNeeded();
114 }
115 
116 void MainThreadGenericEventQueue::suspend(ReasonForSuspension)
117 {
118     if (m_isSuspended)
119         return;
120 
121     m_isSuspended = true;
122     m_taskQueue->cancelAllTasks();
123 }
124 
125 void MainThreadGenericEventQueue::resume()
126 {
127     if (!m_isSuspended)
128         return;
129 
130     m_isSuspended = false;
131     rescheduleAllEventsIfNeeded();
132 }
133 
134 void MainThreadGenericEventQueue::rescheduleAllEventsIfNeeded()
135 {
136     if (isSuspendedOrPausedByClient())
137         return;
138 
139     for (unsigned i = 0; i < m_pendingEvents.size(); ++i)
140         m_taskQueue->enqueueTask(std::bind(&MainThreadGenericEventQueue::dispatchOneEvent, this));
141 }
142 
143 void MainThreadGenericEventQueue::stop()
144 {
145     close();
146 }
147 
148 const char* MainThreadGenericEventQueue::activeDOMObjectName() const
149 {
150     return "MainThreadGenericEventQueue";
151 }
152 
153 UniqueRef<MainThreadGenericEventQueue> MainThreadGenericEventQueue::create(EventTarget& eventTarget)
154 {
155     auto eventQueue = makeUniqueRef<MainThreadGenericEventQueue>(eventTarget);
156     eventQueue->suspendIfNeeded();
157     return eventQueue;
158 }
