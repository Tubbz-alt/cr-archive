26 #pragma once
27 
28 #include "UnlinkedGlobalCodeBlock.h"
29 
30 namespace JSC {
31 
32 class CachedProgramCodeBlock;
33 
34 class UnlinkedProgramCodeBlock final : public UnlinkedGlobalCodeBlock {
35 public:
36     typedef UnlinkedGlobalCodeBlock Base;
37     static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
38 
39     template<typename CellType, SubspaceAccess mode>
40     static IsoSubspace* subspaceFor(VM& vm)
41     {
42         return vm.unlinkedProgramCodeBlockSpace<mode>();
43     }
44 
45     static UnlinkedProgramCodeBlock* create(VM& vm, const ExecutableInfo& info, OptionSet<CodeGenerationMode> codeGenerationMode)
46     {
47         UnlinkedProgramCodeBlock* instance = new (NotNull, allocateCell<UnlinkedProgramCodeBlock>(vm.heap)) UnlinkedProgramCodeBlock(vm, vm.unlinkedProgramCodeBlockStructure.get(), info, codeGenerationMode);
48         instance->finishCreation(vm);
49         return instance;
50     }
51 
52     static void destroy(JSCell*);
53 
54     void setVariableDeclarations(const VariableEnvironment& environment) { m_varDeclarations = environment; }
55     const VariableEnvironment& variableDeclarations() const { return m_varDeclarations; }
56 
57     void setLexicalDeclarations(const VariableEnvironment& environment) { m_lexicalDeclarations = environment; }
58     const VariableEnvironment& lexicalDeclarations() const { return m_lexicalDeclarations; }
59 
60 private:
61     friend CachedProgramCodeBlock;
62 
63     UnlinkedProgramCodeBlock(VM& vm, Structure* structure, const ExecutableInfo& info, OptionSet<CodeGenerationMode> codeGenerationMode)
64         : Base(vm, structure, GlobalCode, info, codeGenerationMode)
65     {
66     }
67 
68     UnlinkedProgramCodeBlock(Decoder&, const CachedProgramCodeBlock&);
69 
70     VariableEnvironment m_varDeclarations;
71     VariableEnvironment m_lexicalDeclarations;
72 
73 public:
74     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue proto)
75     {
76         return Structure::create(vm, globalObject, proto, TypeInfo(UnlinkedProgramCodeBlockType, StructureFlags), info());
77     }
78 
79     DECLARE_INFO;
80 };
81 
82 }
