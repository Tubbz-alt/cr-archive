35 namespace WebCore {

38 JSValue toJSNewlyCreated(ExecState*, JSDOMGlobalObject* globalObject, Ref<TypedOMCSSStyleValue>&& value)
39 {
40     if (value->isUnitValue())
41         return createWrapper<TypedOMCSSUnitValue>(globalObject, WTFMove(value));
42     if (value->isUnparsedValue())
43         return createWrapper<TypedOMCSSUnparsedValue>(globalObject, WTFMove(value));
44     if (value->isImageValue())
45         return createWrapper<TypedOMCSSImageValue>(globalObject, WTFMove(value));
46 
47     ASSERT_NOT_REACHED();
48     return createWrapper<TypedOMCSSStyleValue>(globalObject, WTFMove(value));
49 }
50 
51 JSValue toJS(ExecState* state, JSDOMGlobalObject* globalObject, TypedOMCSSStyleValue& object)
52 {
53     return wrap(state, globalObject, object);
54 }
