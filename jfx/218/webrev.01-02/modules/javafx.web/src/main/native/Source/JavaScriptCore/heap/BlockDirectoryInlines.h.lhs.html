32 namespace JSC {
33 
34 template <typename Functor> inline void BlockDirectory::forEachBlock(const Functor& functor)
35 {
36     m_live.forEachSetBit(
37         [&] (size_t index) {
38             functor(m_blocks[index]);
39         });
40 }
41 
42 template <typename Functor> inline void BlockDirectory::forEachNotEmptyBlock(const Functor& functor)
43 {
44     m_markingNotEmpty.forEachSetBit(
45         [&] (size_t index) {
46             functor(m_blocks[index]);
47         });
48 }
