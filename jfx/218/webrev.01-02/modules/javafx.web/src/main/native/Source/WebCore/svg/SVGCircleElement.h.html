22 #pragma once
23 
24 #include "SVGGeometryElement.h"
25 #include "SVGNames.h"
26 
27 namespace WebCore {
28 
29 class SVGCircleElement final : public SVGGeometryElement {
30     WTF_MAKE_ISO_ALLOCATED(SVGCircleElement);
31 public:
32     static Ref<SVGCircleElement> create(const QualifiedName&, Document&);
33 
34     const SVGLengthValue& cx() const { return m_cx->currentValue(); }
35     const SVGLengthValue& cy() const { return m_cy->currentValue(); }
36     const SVGLengthValue& r() const { return m_r->currentValue(); }
37 
38     SVGAnimatedLength& cxAnimated() { return m_cx; }
39     SVGAnimatedLength& cyAnimated() { return m_cy; }
40     SVGAnimatedLength& rAnimated() { return m_r; }
41 
42 private:
43     SVGCircleElement(const QualifiedName&, Document&);
44 
45     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGCircleElement, SVGGeometryElement>;
46     const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
47 
48     void parseAttribute(const QualifiedName&, const AtomString&) final;
49     void svgAttributeChanged(const QualifiedName&) final;
50 
51     bool isValid() const final { return SVGTests::isValid(); }
52     bool selfHasRelativeLengths() const final { return true; }
53 
54     RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) final;
55 
56     PropertyRegistry m_propertyRegistry { *this };
57     Ref<SVGAnimatedLength> m_cx { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
58     Ref<SVGAnimatedLength> m_cy { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
59     Ref<SVGAnimatedLength> m_r { SVGAnimatedLength::create(this, SVGLengthMode::Other) };
60 };
61 
62 } // namespace WebCore
