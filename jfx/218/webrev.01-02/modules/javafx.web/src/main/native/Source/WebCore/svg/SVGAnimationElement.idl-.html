27 interface SVGAnimationElement : SVGElement {
28     readonly attribute SVGElement targetElement;
29 
30     unrestricted float getStartTime();
31     unrestricted float getCurrentTime();
32     unrestricted float getSimpleDuration();
33 
34     void beginElement();
35     void beginElementAt(optional unrestricted float offset = NaN);
36     void endElement();
37     void endElementAt(optional unrestricted float offset = NaN);
38 };
