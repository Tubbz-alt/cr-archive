28 #include "ResourceError.h"
29 #include "ResourceHandleClient.h"
30 #include "ResourceResponse.h"
31 #include <wtf/Function.h>
32 #include <wtf/MessageQueue.h>

34 
35 namespace WebCore {
36 
37 class SynchronousLoaderMessageQueue : public ThreadSafeRefCounted<SynchronousLoaderMessageQueue> {
38 public:
39     static Ref<SynchronousLoaderMessageQueue> create() { return adoptRef(*new SynchronousLoaderMessageQueue); }
40 
41     void append(std::unique_ptr<Function<void()>>&& task) { m_queue.append(WTFMove(task)); }
42     void kill() { m_queue.kill(); }
43     bool killed() const { return m_queue.killed(); }
44     std::unique_ptr<Function<void()>> waitForMessage() { return m_queue.waitForMessage(); }
45 
46 private:
47     SynchronousLoaderMessageQueue() = default;
48     MessageQueue<Function<void()>> m_queue;
49 };
50 
51 class SynchronousLoaderClient final : public ResourceHandleClient {
52 public:
53     SynchronousLoaderClient();
54     virtual ~SynchronousLoaderClient();
55 
56     void setAllowStoredCredentials(bool allow) { m_allowStoredCredentials = allow; }
57     const ResourceResponse& response() const { return m_response; }
58     Vector<char>& mutableData() { return m_data; }
59     const ResourceError& error() const { return m_error; }
60     SynchronousLoaderMessageQueue& messageQueue() { return m_messageQueue.get(); }
61 
62     WEBCORE_EXPORT static ResourceError platformBadResponseError();
63 
64 private:
65     void willSendRequestAsync(ResourceHandle*, ResourceRequest&&, ResourceResponse&&, CompletionHandler<void(ResourceRequest&&)>&&) override;
66     bool shouldUseCredentialStorage(ResourceHandle*) override;
67     void didReceiveAuthenticationChallenge(ResourceHandle*, const AuthenticationChallenge&) override;
68     void didReceiveResponseAsync(ResourceHandle*, ResourceResponse&&, CompletionHandler<void()>&&) override;
69     void didReceiveData(ResourceHandle*, const char*, unsigned, int /*encodedDataLength*/) override;
70     void didFinishLoading(ResourceHandle*) override;
71     void didFail(ResourceHandle*, const ResourceError&) override;
72 #if USE(PROTECTION_SPACE_AUTH_CALLBACK)
73     void canAuthenticateAgainstProtectionSpaceAsync(ResourceHandle*, const ProtectionSpace&, CompletionHandler<void(bool)>&&) override;
74 #endif
75 
76     bool m_allowStoredCredentials { false };
77     ResourceResponse m_response;
78     Vector<char> m_data;
79     ResourceError m_error;
80     Ref<SynchronousLoaderMessageQueue> m_messageQueue;
81 };
82 }
