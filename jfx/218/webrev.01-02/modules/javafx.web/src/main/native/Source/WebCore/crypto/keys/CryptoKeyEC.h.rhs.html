 58 namespace WebCore {
 59 
 60 struct JsonWebKey;
 61 
 62 class CryptoKeyEC final : public CryptoKey {
 63 public:
 64     enum class NamedCurve {
 65         P256,
 66         P384,
 67         P521,
 68     };
 69 
 70     static Ref<CryptoKeyEC> create(CryptoAlgorithmIdentifier identifier, NamedCurve curve, CryptoKeyType type, PlatformECKeyContainer&& platformKey, bool extractable, CryptoKeyUsageBitmap usages)
 71     {
 72         return adoptRef(*new CryptoKeyEC(identifier, curve, type, WTFMove(platformKey), extractable, usages));
 73     }
 74     virtual ~CryptoKeyEC() = default;
 75 
 76     WEBCORE_EXPORT static ExceptionOr<CryptoKeyPair> generatePair(CryptoAlgorithmIdentifier, const String& curve, bool extractable, CryptoKeyUsageBitmap);
 77     WEBCORE_EXPORT static RefPtr<CryptoKeyEC> importRaw(CryptoAlgorithmIdentifier, const String& curve, Vector<uint8_t>&& keyData, bool extractable, CryptoKeyUsageBitmap);
 78     static RefPtr<CryptoKeyEC> importJwk(CryptoAlgorithmIdentifier, const String& curve, JsonWebKey&&, bool extractable, CryptoKeyUsageBitmap);
 79     static RefPtr<CryptoKeyEC> importSpki(CryptoAlgorithmIdentifier, const String& curve, Vector<uint8_t>&& keyData, bool extractable, CryptoKeyUsageBitmap);
 80     static RefPtr<CryptoKeyEC> importPkcs8(CryptoAlgorithmIdentifier, const String& curve, Vector<uint8_t>&& keyData, bool extractable, CryptoKeyUsageBitmap);
 81 
 82     WEBCORE_EXPORT ExceptionOr<Vector<uint8_t>> exportRaw() const;
 83     ExceptionOr<JsonWebKey> exportJwk() const;
 84     ExceptionOr<Vector<uint8_t>> exportSpki() const;
 85     ExceptionOr<Vector<uint8_t>> exportPkcs8() const;
 86 
 87     size_t keySizeInBits() const;
 88     NamedCurve namedCurve() const { return m_curve; }
 89     String namedCurveString() const;
 90     PlatformECKey platformKey() const { return m_platformKey.get(); }
 91     static bool isValidECAlgorithm(CryptoAlgorithmIdentifier);
 92 
 93 private:
 94     CryptoKeyEC(CryptoAlgorithmIdentifier, NamedCurve, CryptoKeyType, PlatformECKeyContainer&&, bool extractable, CryptoKeyUsageBitmap);
 95 
 96     CryptoKeyClass keyClass() const final { return CryptoKeyClass::EC; }
 97 
 98     KeyAlgorithm algorithm() const final;
 99 
100     static bool platformSupportedCurve(NamedCurve);
101     static Optional<CryptoKeyPair> platformGeneratePair(CryptoAlgorithmIdentifier, NamedCurve, bool extractable, CryptoKeyUsageBitmap);
102     static RefPtr<CryptoKeyEC> platformImportRaw(CryptoAlgorithmIdentifier, NamedCurve, Vector<uint8_t>&& keyData, bool extractable, CryptoKeyUsageBitmap);
103     static RefPtr<CryptoKeyEC> platformImportJWKPublic(CryptoAlgorithmIdentifier, NamedCurve, Vector<uint8_t>&& x, Vector<uint8_t>&& y, bool extractable, CryptoKeyUsageBitmap);
104     static RefPtr<CryptoKeyEC> platformImportJWKPrivate(CryptoAlgorithmIdentifier, NamedCurve, Vector<uint8_t>&& x, Vector<uint8_t>&& y, Vector<uint8_t>&& d, bool extractable, CryptoKeyUsageBitmap);
105     static RefPtr<CryptoKeyEC> platformImportSpki(CryptoAlgorithmIdentifier, NamedCurve, Vector<uint8_t>&& keyData, bool extractable, CryptoKeyUsageBitmap);
106     static RefPtr<CryptoKeyEC> platformImportPkcs8(CryptoAlgorithmIdentifier, NamedCurve, Vector<uint8_t>&& keyData, bool extractable, CryptoKeyUsageBitmap);
107     Vector<uint8_t> platformExportRaw() const;
108     bool platformAddFieldElements(JsonWebKey&) const;
109     Vector<uint8_t> platformExportSpki() const;
110     Vector<uint8_t> platformExportPkcs8() const;
111 
112     PlatformECKeyContainer m_platformKey;
113     NamedCurve m_curve;
114 };
115 
116 } // namespace WebCore
