 31 namespace WebCore {
 32 
 33 class OutlineValue;
 34 
 35 class BorderData {
 36 friend class RenderStyle;
 37 public:
 38     BorderData()
 39         : m_topLeftRadius { { 0, Fixed }, { 0, Fixed } }
 40         , m_topRightRadius { { 0, Fixed }, { 0, Fixed } }
 41         , m_bottomLeftRadius { { 0, Fixed }, { 0, Fixed } }
 42         , m_bottomRightRadius { { 0, Fixed }, { 0, Fixed } }
 43     {
 44     }
 45 
 46     bool hasBorder() const
 47     {
 48         bool haveImage = m_image.hasImage();
 49         return m_left.nonZero(!haveImage) || m_right.nonZero(!haveImage) || m_top.nonZero(!haveImage) || m_bottom.nonZero(!haveImage);
 50     }
 51 
 52     bool hasVisibleBorder() const
 53     {
 54         bool haveImage = m_image.hasImage();
 55         return m_left.isVisible(!haveImage) || m_right.isVisible(!haveImage) || m_top.isVisible(!haveImage) || m_bottom.isVisible(!haveImage);
 56     }
 57 
 58     bool hasFill() const
 59     {
 60         return m_image.hasImage() && m_image.fill();
 61     }
 62 
 63     bool hasBorderRadius() const
 64     {
 65         return !m_topLeftRadius.width.isZero()
 66             || !m_topRightRadius.width.isZero()
 67             || !m_bottomLeftRadius.width.isZero()
 68             || !m_bottomRightRadius.width.isZero();
 69     }
 70 
 71     float borderLeftWidth() const
 72     {
 73         if (!m_image.hasImage() && (m_left.style() == BorderStyle::None || m_left.style() == BorderStyle::Hidden))
 74             return 0;
 75         return m_left.width();
 76     }
 77 
 78     float borderRightWidth() const
 79     {
 80         if (!m_image.hasImage() && (m_right.style() == BorderStyle::None || m_right.style() == BorderStyle::Hidden))
 81             return 0;
 82         return m_right.width();
 83     }
 84 
 85     float borderTopWidth() const
 86     {
 87         if (!m_image.hasImage() && (m_top.style() == BorderStyle::None || m_top.style() == BorderStyle::Hidden))
 88             return 0;
 89         return m_top.width();
 90     }
 91 
 92     float borderBottomWidth() const
 93     {
 94         if (!m_image.hasImage() && (m_bottom.style() == BorderStyle::None || m_bottom.style() == BorderStyle::Hidden))
 95             return 0;
 96         return m_bottom.width();
 97     }
 98 
 99     FloatBoxExtent borderWidth() const
100     {
101         return FloatBoxExtent(borderTopWidth(), borderRightWidth(), borderBottomWidth(), borderLeftWidth());
102     }
103 
104     bool operator==(const BorderData& o) const
105     {
106         return m_left == o.m_left && m_right == o.m_right && m_top == o.m_top && m_bottom == o.m_bottom && m_image == o.m_image
107             && m_topLeftRadius == o.m_topLeftRadius && m_topRightRadius == o.m_topRightRadius && m_bottomLeftRadius == o.m_bottomLeftRadius && m_bottomRightRadius == o.m_bottomRightRadius;
108     }
109 
110     bool operator!=(const BorderData& o) const
111     {
112         return !(*this == o);
113     }
114 
115     const BorderValue& left() const { return m_left; }
116     const BorderValue& right() const { return m_right; }
117     const BorderValue& top() const { return m_top; }
118     const BorderValue& bottom() const { return m_bottom; }
119 
120     const NinePieceImage& image() const { return m_image; }
121 
122     const LengthSize& topLeftRadius() const { return m_topLeftRadius; }
123     const LengthSize& topRightRadius() const { return m_topRightRadius; }
124     const LengthSize& bottomLeftRadius() const { return m_bottomLeftRadius; }
125     const LengthSize& bottomRightRadius() const { return m_bottomRightRadius; }
126 
127     void dump(TextStream&, DumpStyleValues = DumpStyleValues::All) const;
128 
129 private:
130     BorderValue m_left;
131     BorderValue m_right;
132     BorderValue m_top;
133     BorderValue m_bottom;
134 
135     NinePieceImage m_image;
136 
137     LengthSize m_topLeftRadius;
138     LengthSize m_topRightRadius;
139     LengthSize m_bottomLeftRadius;
140     LengthSize m_bottomRightRadius;
141 };
142

144 WTF::TextStream& operator<<(WTF::TextStream&, const OutlineValue&);
145 WTF::TextStream& operator<<(WTF::TextStream&, const BorderData&);
