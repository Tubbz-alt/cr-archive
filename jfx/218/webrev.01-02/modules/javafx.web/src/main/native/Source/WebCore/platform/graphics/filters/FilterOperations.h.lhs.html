26 #pragma once
27 
28 #include "FilterOperation.h"
29 #include "IntRectExtent.h"
30 #include <wtf/RefPtr.h>
31 #include <wtf/Vector.h>
32 
33 namespace WebCore {
34 
35 using FilterOutsets = IntRectExtent;
36 
37 class FilterOperations {
38     WTF_MAKE_FAST_ALLOCATED;
39 public:
40     bool operator==(const FilterOperations&) const;
41     bool operator!=(const FilterOperations& other) const { return !(*this == other); }
42 
43     void clear() { m_operations.clear(); }
44 
45     Vector<RefPtr<FilterOperation>>& operations() { return m_operations; }
46     const Vector<RefPtr<FilterOperation>>& operations() const { return m_operations; }
47 
48     bool isEmpty() const { return m_operations.isEmpty(); }
49     size_t size() const { return m_operations.size(); }
50     const FilterOperation* at(size_t index) const { return index < m_operations.size() ? m_operations[index].get() : nullptr; }
51 
52     bool operationsMatch(const FilterOperations&) const;
53 
54     bool hasOutsets() const;
55     FilterOutsets outsets() const;
56 
57     bool hasFilterThatAffectsOpacity() const;
58     bool hasFilterThatMovesPixels() const;
59     bool hasFilterThatShouldBeRestrictedBySecurityOrigin() const;
60 
61     bool hasReferenceFilter() const;
62 
63     bool transformColor(Color&) const;
64     bool inverseTransformColor(Color&) const;
65 
66 private:
67     Vector<RefPtr<FilterOperation>> m_operations;
68 };
69 
70 WEBCORE_EXPORT WTF::TextStream& operator<<(WTF::TextStream&, const FilterOperations&);
71 
72 } // namespace WebCore
