37 bool InlineFormattingContext::Quirks::lineDescentNeedsCollapsing(const LineBuilder::RunList& runList) const
38 {
39     // Collapse line descent in limited and full quirk mode when there's no baseline aligned content or
40     // the baseline aligned content has no descent.
41     auto& layoutState = this->layoutState();
42     if (!layoutState.inQuirksMode() && !layoutState.inLimitedQuirksMode())
43         return false;
44 
45     for (auto& run : runList) {
46         auto& layoutBox = run.layoutBox();
47         if (run.isContainerEnd() || layoutBox.style().verticalAlign() != VerticalAlign::Baseline)
48             continue;
49 
50         if (run.isLineBreak())
51             return false;
52         if (run.isText())
53             return false;
54         if (run.isContainerStart()) {
55             auto& boxGeometry = formattingContext().geometryForBox(layoutBox);
56             if (boxGeometry.horizontalBorder() || (boxGeometry.horizontalPadding() && boxGeometry.horizontalPadding().value()))
57                 return false;
58             continue;
59         }
60         if (run.isBox()) {
61             if (layoutBox.isInlineBlockBox() && layoutBox.establishesInlineFormattingContext()) {
62                 auto& formattingState = layoutState.establishedInlineFormattingState(downcast<Container>(layoutBox));
63                 auto inlineBlockBaseline = formattingState.displayInlineContent()->lineBoxes.last().baseline();
64                 if (inlineBlockBaseline.descent())
65                     return false;
66             }
67             continue;
68         }
69         ASSERT_NOT_REACHED();
70     }
71     return true;
72 }

74 LineBuilder::Constraints::HeightAndBaseline InlineFormattingContext::Quirks::lineHeightConstraints(const Container& formattingRoot) const
75 {
76     // computedLineHeight takes font-size into account when line-height is not set.
77     // Strut is the imaginary box that we put on every line. It sets the initial vertical constraints for each new line.
78     InlineLayoutUnit strutHeight = formattingRoot.style().computedLineHeight();
79     auto strutBaselineOffset = LineBuilder::halfLeadingMetrics(formattingRoot.style().fontMetrics(), strutHeight).ascent();
80     if (layoutState().inNoQuirksMode())
81         return { strutHeight, strutBaselineOffset, { } };
82 
83     auto lineHeight = formattingRoot.style().lineHeight();
84     if (lineHeight.isPercentOrCalculated()) {
85         auto initialBaselineOffset = LineBuilder::halfLeadingMetrics(formattingRoot.style().fontMetrics(), 0_lu).ascent();
86         return { initialBaselineOffset, initialBaselineOffset, LineBoxBuilder::Baseline { strutBaselineOffset, strutHeight - strutBaselineOffset } };
87     }
88     // FIXME: The only reason why we use intValue() here is to match current inline tree (integral)behavior.
89     InlineLayoutUnit initialLineHeight = lineHeight.intValue();
90     auto initialBaselineOffset = LineBuilder::halfLeadingMetrics(formattingRoot.style().fontMetrics(), initialLineHeight).ascent();
91     return { initialLineHeight, initialBaselineOffset, LineBoxBuilder::Baseline { strutBaselineOffset, strutHeight - strutBaselineOffset } };
92 }
