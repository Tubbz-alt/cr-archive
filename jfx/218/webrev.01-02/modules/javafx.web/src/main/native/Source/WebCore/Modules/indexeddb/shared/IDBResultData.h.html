137 template<class Encoder>
138 void IDBResultData::encode(Encoder& encoder) const
139 {
140     encoder << m_requestIdentifier << m_error << m_databaseConnectionIdentifier << m_resultInteger;
141 
142     encoder.encodeEnum(m_type);
143 
144     encoder << !!m_databaseInfo;
145     if (m_databaseInfo)
146         encoder << *m_databaseInfo;
147 
148     encoder << !!m_transactionInfo;
149     if (m_transactionInfo)
150         encoder << *m_transactionInfo;
151 
152     encoder << !!m_resultKey;
153     if (m_resultKey)
154         encoder << *m_resultKey;
155 
156     encoder << !!m_getResult;
157     if (m_getResult)
158         encoder << *m_getResult;
159 
160     encoder << !!m_getAllResult;
161     if (m_getAllResult)
162         encoder << *m_getAllResult;
163 }
164 
165 template<class Decoder> Optional<IDBResultData> IDBResultData::decode(Decoder& decoder)
166 {
167     IDBResultData result;
168     if (!decoder.decode(result.m_requestIdentifier))
169         return WTF::nullopt;
170 
171     if (!decoder.decode(result.m_error))
172         return WTF::nullopt;
173 
174     if (!decoder.decode(result.m_databaseConnectionIdentifier))
175         return WTF::nullopt;
176 
177     if (!decoder.decode(result.m_resultInteger))
178         return WTF::nullopt;
179 
180     if (!decoder.decodeEnum(result.m_type))
181         return WTF::nullopt;
182 
183     bool hasObject;
184 
185     if (!decoder.decode(hasObject))
186         return WTF::nullopt;
187     if (hasObject) {
188         auto object = makeUnique<IDBDatabaseInfo>();
189         if (!decoder.decode(*object))
190             return WTF::nullopt;
191         result.m_databaseInfo = WTFMove(object);
192     }
193 
194     if (!decoder.decode(hasObject))
195         return WTF::nullopt;
196     if (hasObject) {
197         auto object = makeUnique<IDBTransactionInfo>();
198         if (!decoder.decode(*object))
199             return WTF::nullopt;
200         result.m_transactionInfo = WTFMove(object);
201     }
202 
203     if (!decoder.decode(hasObject))
204         return WTF::nullopt;
205     if (hasObject) {
206         auto object = makeUnique<IDBKeyData>();
207         Optional<IDBKeyData> optional;
208         decoder >> optional;
209         if (!optional)
210             return WTF::nullopt;
211         *object = WTFMove(*optional);
212         result.m_resultKey = WTFMove(object);
213     }
214 
215     if (!decoder.decode(hasObject))
216         return WTF::nullopt;
217     if (hasObject) {
218         auto object = makeUnique<IDBGetResult>();
219         if (!decoder.decode(*object))
220             return WTF::nullopt;
221         result.m_getResult = WTFMove(object);
222     }
223 
224     if (!decoder.decode(hasObject))
225         return WTF::nullopt;
226     if (hasObject) {
227         auto object = makeUnique<IDBGetAllResult>();
228         if (!decoder.decode(*object))
229             return WTF::nullopt;
230         result.m_getAllResult = WTFMove(object);
231     }
232 
233     return result;
234 }
