 33 template<typename Type>
 34 void GenericArguments<Type>::visitChildren(JSCell* thisCell, SlotVisitor& visitor)
 35 {
 36     Type* thisObject = static_cast<Type*>(thisCell);
 37     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
 38     Base::visitChildren(thisCell, visitor);
 39 
 40     if (thisObject->m_modifiedArgumentsDescriptor)
 41         visitor.markAuxiliary(thisObject->m_modifiedArgumentsDescriptor.getUnsafe());
 42 }

 44 template<typename Type>
 45 bool GenericArguments<Type>::getOwnPropertySlot(JSObject* object, JSGlobalObject* globalObject, PropertyName ident, PropertySlot& slot)
 46 {
 47     Type* thisObject = jsCast<Type*>(object);
 48     VM& vm = globalObject->vm();
 49 
 50     if (!thisObject->overrodeThings()) {
 51         if (ident == vm.propertyNames->length) {
 52             slot.setValue(thisObject, static_cast<unsigned>(PropertyAttribute::DontEnum), jsNumber(thisObject->internalLength()));
 53             return true;
 54         }
 55         if (ident == vm.propertyNames->callee) {
 56             slot.setValue(thisObject, static_cast<unsigned>(PropertyAttribute::DontEnum), thisObject->callee());
 57             return true;
 58         }
 59         if (ident == vm.propertyNames->iteratorSymbol) {
 60             slot.setValue(thisObject, static_cast<unsigned>(PropertyAttribute::DontEnum), thisObject->globalObject(vm)->arrayProtoValuesFunction());
 61             return true;
 62         }
 63     }
 64 
 65     if (Optional<uint32_t> index = parseIndex(ident))
 66         return GenericArguments<Type>::getOwnPropertySlotByIndex(thisObject, globalObject, *index, slot);
 67 
 68     return Base::getOwnPropertySlot(thisObject, globalObject, ident, slot);
 69 }

 71 template<typename Type>
 72 bool GenericArguments<Type>::getOwnPropertySlotByIndex(JSObject* object, JSGlobalObject* globalObject, unsigned index, PropertySlot& slot)
 73 {
 74     Type* thisObject = jsCast<Type*>(object);
 75 
 76     if (!thisObject->isModifiedArgumentDescriptor(index) && thisObject->isMappedArgument(index)) {
 77         slot.setValue(thisObject, static_cast<unsigned>(PropertyAttribute::None), thisObject->getIndexQuickly(index));
 78         return true;
 79     }
 80 
 81     bool result = Base::getOwnPropertySlotByIndex(object, globalObject, index, slot);
 82 
 83     if (thisObject->isMappedArgument(index)) {
 84         ASSERT(result);
 85         slot.setValue(thisObject, slot.attributes(), thisObject->getIndexQuickly(index));
 86         return true;
 87     }
 88 
 89     return result;
 90 }

 92 template<typename Type>
 93 void GenericArguments<Type>::getOwnPropertyNames(JSObject* object, JSGlobalObject* globalObject, PropertyNameArray& array, EnumerationMode mode)
 94 {
 95     VM& vm = globalObject->vm();
 96     Type* thisObject = jsCast<Type*>(object);
 97 
 98     if (array.includeStringProperties()) {
 99         for (unsigned i = 0; i < thisObject->internalLength(); ++i) {
100             if (!thisObject->isMappedArgument(i))
101                 continue;
102             array.add(Identifier::from(vm, i));
103         }
104     }
105 
106     if (mode.includeDontEnumProperties() && !thisObject->overrodeThings()) {
107         array.add(vm.propertyNames->length);
108         array.add(vm.propertyNames->callee);
109         if (array.includeSymbolProperties())
110             array.add(vm.propertyNames->iteratorSymbol);
111     }
112     Base::getOwnPropertyNames(thisObject, globalObject, array, mode);
113 }

115 template<typename Type>
116 bool GenericArguments<Type>::put(JSCell* cell, JSGlobalObject* globalObject, PropertyName ident, JSValue value, PutPropertySlot& slot)
117 {
118     Type* thisObject = jsCast<Type*>(cell);
119     VM& vm = globalObject->vm();
120     auto scope = DECLARE_THROW_SCOPE(vm);
121 
122     if (!thisObject->overrodeThings()
123         && (ident == vm.propertyNames->length
124             || ident == vm.propertyNames->callee
125             || ident == vm.propertyNames->iteratorSymbol)) {
126         thisObject->overrideThings(globalObject);
127         RETURN_IF_EXCEPTION(scope, false);
128         PutPropertySlot dummy = slot; // This put is not cacheable, so we shadow the slot that was given to us.
129         RELEASE_AND_RETURN(scope, Base::put(thisObject, globalObject, ident, value, dummy));
130     }
131 
132     // https://tc39.github.io/ecma262/#sec-arguments-exotic-objects-set-p-v-receiver
133     // Fall back to the OrdinarySet when the receiver is altered from the thisObject.
134     if (UNLIKELY(isThisValueAltered(slot, thisObject)))
135         RELEASE_AND_RETURN(scope, ordinarySetSlow(globalObject, thisObject, ident, value, slot.thisValue(), slot.isStrictMode()));
136 
137     Optional<uint32_t> index = parseIndex(ident);
138     if (index && thisObject->isMappedArgument(index.value())) {
139         thisObject->setIndexQuickly(vm, index.value(), value);
140         return true;
141     }
142 
143     RELEASE_AND_RETURN(scope, Base::put(thisObject, globalObject, ident, value, slot));
144 }

146 template<typename Type>
147 bool GenericArguments<Type>::putByIndex(JSCell* cell, JSGlobalObject* globalObject, unsigned index, JSValue value, bool shouldThrow)
148 {
149     Type* thisObject = jsCast<Type*>(cell);
150     VM& vm = globalObject->vm();
151 
152     if (thisObject->isMappedArgument(index)) {
153         thisObject->setIndexQuickly(vm, index, value);
154         return true;
155     }
156 
157     return Base::putByIndex(cell, globalObject, index, value, shouldThrow);
158 }

160 template<typename Type>
161 bool GenericArguments<Type>::deleteProperty(JSCell* cell, JSGlobalObject* globalObject, PropertyName ident)
162 {
163     Type* thisObject = jsCast<Type*>(cell);
164     VM& vm = globalObject->vm();
165     auto scope = DECLARE_THROW_SCOPE(vm);
166 
167     if (!thisObject->overrodeThings()
168         && (ident == vm.propertyNames->length
169             || ident == vm.propertyNames->callee
170             || ident == vm.propertyNames->iteratorSymbol)) {
171         thisObject->overrideThings(globalObject);
172         RETURN_IF_EXCEPTION(scope, false);
173     }
174 
175     if (Optional<uint32_t> index = parseIndex(ident))
176         RELEASE_AND_RETURN(scope, GenericArguments<Type>::deletePropertyByIndex(thisObject, globalObject, *index));
177 
178     RELEASE_AND_RETURN(scope, Base::deleteProperty(thisObject, globalObject, ident));
179 }

181 template<typename Type>
182 bool GenericArguments<Type>::deletePropertyByIndex(JSCell* cell, JSGlobalObject* globalObject, unsigned index)
183 {
184     VM& vm = globalObject->vm();
185     auto scope = DECLARE_THROW_SCOPE(vm);
186 
187     Type* thisObject = jsCast<Type*>(cell);

188 
189     bool propertyMightBeInJSObjectStorage = thisObject->isModifiedArgumentDescriptor(index) || !thisObject->isMappedArgument(index);
190     bool deletedProperty = true;
191     if (propertyMightBeInJSObjectStorage) {
192         deletedProperty = Base::deletePropertyByIndex(cell, globalObject, index);
193         RETURN_IF_EXCEPTION(scope, true);
194     }
195 
196     if (deletedProperty) {
197         // Deleting an indexed property unconditionally unmaps it.
198         if (thisObject->isMappedArgument(index)) {
199             // We need to check that the property was mapped so we don't write to random memory.
200             thisObject->unmapArgument(globalObject, index);
201             RETURN_IF_EXCEPTION(scope, true);
202         }
203         thisObject->setModifiedArgumentDescriptor(globalObject, index);
204         RETURN_IF_EXCEPTION(scope, true);
205     }
206 
207     return deletedProperty;
208 }

210 template<typename Type>
211 bool GenericArguments<Type>::defineOwnProperty(JSObject* object, JSGlobalObject* globalObject, PropertyName ident, const PropertyDescriptor& descriptor, bool shouldThrow)
212 {
213     Type* thisObject = jsCast<Type*>(object);
214     VM& vm = globalObject->vm();
215     auto scope = DECLARE_THROW_SCOPE(vm);
216 
217     if (ident == vm.propertyNames->length
218         || ident == vm.propertyNames->callee
219         || ident == vm.propertyNames->iteratorSymbol) {
220         thisObject->overrideThingsIfNecessary(globalObject);
221         RETURN_IF_EXCEPTION(scope, false);
222     } else {
223         Optional<uint32_t> optionalIndex = parseIndex(ident);
224         if (optionalIndex) {
225             uint32_t index = optionalIndex.value();
226             if (!descriptor.isAccessorDescriptor() && thisObject->isMappedArgument(optionalIndex.value())) {
227                 // If the property is not deleted and we are using a non-accessor descriptor, then
228                 // make sure that the aliased argument sees the value.
229                 if (descriptor.value())
230                     thisObject->setIndexQuickly(vm, index, descriptor.value());
231 
232                 // If the property is not deleted and we are using a non-accessor, writable,
233                 // configurable and enumerable descriptor and isn't modified, then we are done.
234                 // The argument continues to be aliased.
235                 if (descriptor.writable() && descriptor.configurable() && descriptor.enumerable() && !thisObject->isModifiedArgumentDescriptor(index))
236                     return true;
237 
238                 if (!thisObject->isModifiedArgumentDescriptor(index)) {
239                     // If it is a new entry, we need to put direct to initialize argument[i] descriptor properly
240                     JSValue value = thisObject->getIndexQuickly(index);
241                     ASSERT(value);
242                     object->putDirectMayBeIndex(globalObject, ident, value);
243                     scope.assertNoException();
244 
245                     thisObject->setModifiedArgumentDescriptor(globalObject, index);
246                     RETURN_IF_EXCEPTION(scope, false);
247                 }
248             }
249 
250             if (thisObject->isMappedArgument(index)) {
251                 // Just unmap arguments if its descriptor contains {writable: false}.
252                 // Check https://tc39.github.io/ecma262/#sec-createunmappedargumentsobject
253                 // and https://tc39.github.io/ecma262/#sec-createmappedargumentsobject to verify that all data
254                 // property from arguments object are {writable: true, configurable: true, enumerable: true} by default
255                 if ((descriptor.writablePresent() && !descriptor.writable()) || descriptor.isAccessorDescriptor()) {
256                     if (!descriptor.isAccessorDescriptor()) {
257                         JSValue value = thisObject->getIndexQuickly(index);
258                         ASSERT(value);
259                         object->putDirectMayBeIndex(globalObject, ident, value);
260                         scope.assertNoException();
261                     }
262                     thisObject->unmapArgument(globalObject, index);
263                     RETURN_IF_EXCEPTION(scope, false);
264                     thisObject->setModifiedArgumentDescriptor(globalObject, index);
265                     RETURN_IF_EXCEPTION(scope, false);
266                 }
267             }
268         }
269     }
270 
271     // Now just let the normal object machinery do its thing.
272     RELEASE_AND_RETURN(scope, Base::defineOwnProperty(object, globalObject, ident, descriptor, shouldThrow));
273 }

275 template<typename Type>
276 void GenericArguments<Type>::initModifiedArgumentsDescriptor(JSGlobalObject* globalObject, unsigned argsLength)
277 {
278     VM& vm = globalObject->vm();
279     auto scope = DECLARE_THROW_SCOPE(vm);
280 
281     RELEASE_ASSERT(!m_modifiedArgumentsDescriptor);
282 
283     if (argsLength) {
284         void* backingStore = vm.gigacageAuxiliarySpace(m_modifiedArgumentsDescriptor.kind).allocateNonVirtual(vm, WTF::roundUpToMultipleOf<8>(argsLength), nullptr, AllocationFailureMode::ReturnNull);
285         if (UNLIKELY(!backingStore)) {
286             throwOutOfMemoryError(globalObject, scope);
287             return;
288         }
289         bool* modifiedArguments = static_cast<bool*>(backingStore);
290         m_modifiedArgumentsDescriptor.set(vm, this, modifiedArguments, argsLength);
291         for (unsigned i = argsLength; i--;)
292             modifiedArguments[i] = false;
293     }
294 }
295 
296 template<typename Type>
297 void GenericArguments<Type>::initModifiedArgumentsDescriptorIfNecessary(JSGlobalObject* globalObject, unsigned argsLength)
298 {
299     if (!m_modifiedArgumentsDescriptor)
300         initModifiedArgumentsDescriptor(globalObject, argsLength);
301 }

303 template<typename Type>
304 void GenericArguments<Type>::setModifiedArgumentDescriptor(JSGlobalObject* globalObject, unsigned index, unsigned length)
305 {
306     VM& vm = globalObject->vm();
307     auto scope = DECLARE_THROW_SCOPE(vm);
308 
309     initModifiedArgumentsDescriptorIfNecessary(globalObject, length);
310     RETURN_IF_EXCEPTION(scope, void());
311     if (index < length)
312         m_modifiedArgumentsDescriptor.at(index, length) = true;
313 }
314 
315 template<typename Type>
316 bool GenericArguments<Type>::isModifiedArgumentDescriptor(unsigned index, unsigned length)
317 {
318     if (!m_modifiedArgumentsDescriptor)
319         return false;
320     if (index < length)
321         return m_modifiedArgumentsDescriptor.at(index, length);
322     return false;
323 }

325 template<typename Type>
326 void GenericArguments<Type>::copyToArguments(JSGlobalObject* globalObject, JSValue* firstElementDest, unsigned offset, unsigned length)
327 {
328     VM& vm = globalObject->vm();
329     auto scope = DECLARE_THROW_SCOPE(vm);
330 
331     Type* thisObject = static_cast<Type*>(this);
332     for (unsigned i = 0; i < length; ++i) {
333         if (thisObject->isMappedArgument(i + offset))
334             firstElementDest[i] = thisObject->getIndexQuickly(i + offset);
335         else {
336             firstElementDest[i] = get(globalObject, i + offset);
337             RETURN_IF_EXCEPTION(scope, void());
338         }
339     }
340 }
