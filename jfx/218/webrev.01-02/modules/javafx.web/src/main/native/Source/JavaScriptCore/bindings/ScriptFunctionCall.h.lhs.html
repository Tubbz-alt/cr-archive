42 namespace Deprecated {
43 
44 class JS_EXPORT_PRIVATE ScriptCallArgumentHandler {
45 public:
46     ScriptCallArgumentHandler(JSC::ExecState* state) : m_exec(state) { }
47 
48     void appendArgument(const char*);
49     void appendArgument(const String&);
50     void appendArgument(JSC::JSValue);
51     void appendArgument(long);
52     void appendArgument(long long);
53     void appendArgument(unsigned int);
54     void appendArgument(uint64_t);
55     void appendArgument(int);
56     void appendArgument(bool);
57 
58 protected:
59     JSC::MarkedArgumentBuffer m_arguments;
60     JSC::ExecState* m_exec;
61 
62 private:
63     // MarkedArgumentBuffer must be stack allocated, so prevent heap
64     // alloc of ScriptFunctionCall as well.
65     void* operator new(size_t) { ASSERT_NOT_REACHED(); return reinterpret_cast<void*>(0xbadbeef); }
66     void* operator new[](size_t) { ASSERT_NOT_REACHED(); return reinterpret_cast<void*>(0xbadbeef); }
67 };
68 
69 class JS_EXPORT_PRIVATE ScriptFunctionCall : public ScriptCallArgumentHandler {
70 public:
71     typedef JSC::JSValue (*ScriptFunctionCallHandler)(JSC::ExecState* exec, JSC::JSValue functionObject, JSC::CallType callType, const JSC::CallData& callData, JSC::JSValue thisValue, const JSC::ArgList& args, NakedPtr<JSC::Exception>&);
72     ScriptFunctionCall(const ScriptObject& thisObject, const String& name, ScriptFunctionCallHandler handler = nullptr);
73     JSC::JSValue call(bool& hadException);
74     JSC::JSValue call();
75 
76 protected:
77     ScriptFunctionCallHandler m_callHandler;
78     ScriptObject m_thisObject;
79     String m_name;
80 };
