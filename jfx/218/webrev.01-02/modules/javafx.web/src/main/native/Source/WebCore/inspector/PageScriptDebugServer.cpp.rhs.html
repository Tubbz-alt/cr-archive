 56 PageScriptDebugServer::PageScriptDebugServer(Page& page)
 57     : ScriptDebugServer(WebCore::commonVM())
 58     , m_page(page)
 59 {
 60 }
 61 
 62 void PageScriptDebugServer::attachDebugger()
 63 {
 64     m_page.setDebugger(this);
 65 }
 66 
 67 void PageScriptDebugServer::detachDebugger(bool isBeingDestroyed)
 68 {
 69     m_page.setDebugger(nullptr);
 70     if (!isBeingDestroyed)
 71         recompileAllJSFunctions();
 72 }
 73 
 74 void PageScriptDebugServer::recompileAllJSFunctions()
 75 {
 76     JSLockHolder lock(vm());
 77     Debugger::recompileAllJSFunctions();
 78 }
 79 
 80 void PageScriptDebugServer::didPause(JSGlobalObject*)
 81 {
 82     setJavaScriptPaused(m_page.group(), true);
 83 }
 84 
 85 void PageScriptDebugServer::didContinue(JSGlobalObject*)
 86 {
 87     setJavaScriptPaused(m_page.group(), false);
 88 }

 90 void PageScriptDebugServer::runEventLoopWhilePaused()
 91 {
 92 #if PLATFORM(IOS_FAMILY)
 93     // On iOS, running an EventLoop causes us to run a nested WebRunLoop.
 94     // Since the WebThread is autoreleased at the end of run loop iterations
 95     // we need to gracefully handle releasing and reacquiring the lock.
 96     if (WebThreadIsEnabled()) {
 97         ASSERT(WebThreadIsLockedOrDisabled());
 98         JSC::JSLock::DropAllLocks dropAllLocks(vm());
 99         WebRunLoopEnableNested();
100 
101         runEventLoopWhilePausedInternal();
102 
103         WebRunLoopDisableNested();
104         ASSERT(WebThreadIsLockedOrDisabled());
105         return;
106     }
107 #endif
108 
109     runEventLoopWhilePausedInternal();
110 }
111 
112 void PageScriptDebugServer::runEventLoopWhilePausedInternal()
113 {
114     TimerBase::fireTimersInNestedEventLoop();
115 
116     m_page.incrementNestedRunLoopCount();
117 
118     while (!m_doneProcessingDebuggerEvents) {
119         if (RunLoop::cycle() == RunLoop::CycleResult::Stop)
120             break;
121     }
122 
123     m_page.decrementNestedRunLoopCount();
124 }

126 bool PageScriptDebugServer::isContentScript(JSGlobalObject* state) const
127 {
128     return &currentWorld(*state) != &mainThreadNormalWorld();
129 }

131 void PageScriptDebugServer::reportException(JSGlobalObject* state, JSC::Exception* exception) const

133     WebCore::reportException(state, exception);

136 void PageScriptDebugServer::setJavaScriptPaused(const PageGroup& pageGroup, bool paused)
137 {
138     setMainThreadCallbacksPaused(paused);
139 
140     for (auto& page : pageGroup.pages()) {
141         for (Frame* frame = &page->mainFrame(); frame; frame = frame->tree().traverseNext())
142             setJavaScriptPaused(*frame, paused);
143 
144         if (auto* frontendClient = page->inspectorController().inspectorFrontendClient()) {
145             if (paused)
146                 frontendClient->pagePaused();
147             else
148                 frontendClient->pageUnpaused();
149         }
150     }
151 }
152 
153 void PageScriptDebugServer::setJavaScriptPaused(Frame& frame, bool paused)
154 {
155     if (!frame.script().canExecuteScripts(NotAboutToExecuteScript))
156         return;
157 
158     frame.script().setPaused(paused);
159 
160     ASSERT(frame.document());
161     auto& document = *frame.document();
162     if (paused) {
163         document.suspendScriptedAnimationControllerCallbacks();
164         document.suspendActiveDOMObjects(ReasonForSuspension::JavaScriptDebuggerPaused);
165     } else {
166         document.resumeActiveDOMObjects(ReasonForSuspension::JavaScriptDebuggerPaused);
167         document.resumeScriptedAnimationControllerCallbacks();
168     }
169 
170     if (auto* view = frame.view()) {
171         for (auto& child : view->children()) {
172             if (!is<PluginViewBase>(child))
173                 continue;
174             downcast<PluginViewBase>(child.get()).setJavaScriptPaused(paused);
175         }
176     }
177 }
