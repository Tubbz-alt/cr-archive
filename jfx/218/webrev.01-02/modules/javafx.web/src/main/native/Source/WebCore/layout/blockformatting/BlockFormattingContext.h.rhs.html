 44 // This class implements the layout logic for block formatting contexts.
 45 // https://www.w3.org/TR/CSS22/visuren.html#block-formatting
 46 class BlockFormattingContext final : public FormattingContext {
 47     WTF_MAKE_ISO_ALLOCATED(BlockFormattingContext);
 48 public:
 49     BlockFormattingContext(const Container& formattingContextRoot, BlockFormattingState&);
 50 
 51     void layoutInFlowContent(InvalidationState&, const HorizontalConstraints&, const VerticalConstraints&) override;
 52 
 53 private:


 54 
 55     template<typename T>
 56     struct ConstraintsPair {
 57         const T root;
 58         const T containingBlock;
 59     };
 60     void placeInFlowPositionedChildren(const Box&, const ConstraintsPair<HorizontalConstraints>&);
 61 
 62     void computeWidthAndMargin(const FloatingContext&, const Box&, const ConstraintsPair<HorizontalConstraints>&);
 63     void computeHeightAndMargin(const Box&, const ConstraintsPair<HorizontalConstraints>&, const ConstraintsPair<VerticalConstraints>&);
 64 
 65     void computeStaticHorizontalPosition(const Box&, const ConstraintsPair<HorizontalConstraints>&);
 66     void computeStaticVerticalPosition(const Box&, const ConstraintsPair<VerticalConstraints>&);
 67     void computePositionToAvoidFloats(const FloatingContext&, const Box&, const ConstraintsPair<HorizontalConstraints>&, const ConstraintsPair<VerticalConstraints>&);
 68     void computeVerticalPositionForFloatClear(const FloatingContext&, const Box&);

 69 
 70     void precomputeVerticalPosition(const Box&, const HorizontalConstraints&, const VerticalConstraints&);
 71     void precomputeVerticalPositionForAncestors(const Box&, const ConstraintsPair<HorizontalConstraints>&, const ConstraintsPair<VerticalConstraints>&);
 72     void precomputeVerticalPositionForFormattingRoot(const FloatingContext&, const Box&, const ConstraintsPair<HorizontalConstraints>&, const ConstraintsPair<VerticalConstraints>&);

 73 
 74     IntrinsicWidthConstraints computedIntrinsicWidthConstraints() override;
 75     LayoutUnit verticalPositionWithMargin(const Box&, const UsedVerticalMargin&, const VerticalConstraints&) const;
 76 
 77     // This class implements positioning and sizing for boxes participating in a block formatting context.
 78     class Geometry : public FormattingContext::Geometry {
 79     public:
 80         ContentHeightAndMargin inFlowHeightAndMargin(const Box&, const HorizontalConstraints&, const OverrideVerticalValues&);
 81         ContentWidthAndMargin inFlowWidthAndMargin(const Box&, const HorizontalConstraints&, const OverrideHorizontalValues&);
 82 
 83         Point staticPosition(const Box&, const HorizontalConstraints&, const VerticalConstraints&) const;
 84         LayoutUnit staticVerticalPosition(const Box&, const VerticalConstraints&) const;
 85         LayoutUnit staticHorizontalPosition(const Box&, const HorizontalConstraints&) const;
 86 
 87         IntrinsicWidthConstraints intrinsicWidthConstraints(const Box&);
 88 
 89     private:
 90         friend class BlockFormattingContext;
 91         Geometry(const BlockFormattingContext&);
 92 
 93         ContentHeightAndMargin inFlowNonReplacedHeightAndMargin(const Box&, const HorizontalConstraints&, const OverrideVerticalValues&);
 94         ContentWidthAndMargin inFlowNonReplacedWidthAndMargin(const Box&, const HorizontalConstraints&, const OverrideHorizontalValues&) const;
 95         ContentWidthAndMargin inFlowReplacedWidthAndMargin(const Box&, const HorizontalConstraints&, const OverrideHorizontalValues&) const;
 96         Point staticPositionForOutOfFlowPositioned(const Box&) const;
 97 
 98         const BlockFormattingContext& formattingContext() const { return downcast<BlockFormattingContext>(FormattingContext::Geometry::formattingContext()); }
 99     };
100     BlockFormattingContext::Geometry geometry() const { return Geometry(*this); }
101 
102     // This class implements margin collapsing for block formatting context.
103     class MarginCollapse {
104     public:
105         UsedVerticalMargin::CollapsedValues collapsedVerticalValues(const Box&, UsedVerticalMargin::NonCollapsedValues);
106 
107         PrecomputedMarginBefore precomputedMarginBefore(const Box&, UsedVerticalMargin::NonCollapsedValues);
108         LayoutUnit marginBeforeIgnoringCollapsingThrough(const Box&, UsedVerticalMargin::NonCollapsedValues);
109         static void updateMarginAfterForPreviousSibling(BlockFormattingContext&, const MarginCollapse&, const Box&);
110         PositiveAndNegativeVerticalMargin resolvedPositiveNegativeMarginValues(const Box&, const UsedVerticalMargin::NonCollapsedValues&);
111 
112         bool marginBeforeCollapsesWithParentMarginBefore(const Box&) const;
113         bool marginBeforeCollapsesWithFirstInFlowChildMarginBefore(const Box&) const;
114         bool marginBeforeCollapsesWithParentMarginAfter(const Box&) const;
115         bool marginBeforeCollapsesWithPreviousSiblingMarginAfter(const Box&) const;
116 
117         bool marginAfterCollapsesWithParentMarginAfter(const Box&) const;
118         bool marginAfterCollapsesWithLastInFlowChildMarginAfter(const Box&) const;
119         bool marginAfterCollapsesWithParentMarginBefore(const Box&) const;
120         bool marginAfterCollapsesWithNextSiblingMarginBefore(const Box&) const;
121         bool marginAfterCollapsesWithSiblingMarginBeforeWithClearance(const Box&) const;
122 
123         bool marginsCollapseThrough(const Box&) const;
124 
125     private:
126         friend class BlockFormattingContext;
127         MarginCollapse(const BlockFormattingContext&);
128 
129         enum class MarginType { Before, After };
130         PositiveAndNegativeVerticalMargin::Values positiveNegativeValues(const Box&, MarginType) const;
131         PositiveAndNegativeVerticalMargin::Values positiveNegativeMarginBefore(const Box&, UsedVerticalMargin::NonCollapsedValues) const;
132         PositiveAndNegativeVerticalMargin::Values positiveNegativeMarginAfter(const Box&, UsedVerticalMargin::NonCollapsedValues) const;
133         bool hasClearance(const Box&) const;
134 
135         LayoutState& layoutState() { return m_blockFormattingContext.layoutState(); }
136         const LayoutState& layoutState() const { return m_blockFormattingContext.layoutState(); }
137         const BlockFormattingContext& formattingContext() const { return m_blockFormattingContext; }
138 
139         const BlockFormattingContext& m_blockFormattingContext;
140     };
141     MarginCollapse marginCollapse() const { return MarginCollapse(*this); }
142 
143     class Quirks : public FormattingContext::Quirks {
144     public:
145         bool needsStretching(const Box&) const;
146         ContentHeightAndMargin stretchedInFlowHeight(const Box&, ContentHeightAndMargin);
147 
148         bool shouldIgnoreCollapsedQuirkMargin(const Box&) const;
149         bool shouldIgnoreMarginBefore(const Box&) const;
150         bool shouldIgnoreMarginAfter(const Box&) const;
151 
152     private:
153         friend class BlockFormattingContext;
154         Quirks(const BlockFormattingContext&);
155 
156         const BlockFormattingContext& formattingContext() const { return downcast<BlockFormattingContext>(FormattingContext::Quirks::formattingContext()); }
157 



158     };
159     BlockFormattingContext::Quirks quirks() const { return Quirks(*this); }
160 
161     void setPrecomputedMarginBefore(const Box&, const PrecomputedMarginBefore&);
162     void removePrecomputedMarginBefore(const Box& layoutBox) { m_precomputedMarginBeforeList.remove(&layoutBox); }




163     bool hasPrecomputedMarginBefore(const Box&) const;
164     Optional<LayoutUnit> usedAvailableWidthForFloatAvoider(const FloatingContext&, const Box&) const;
165 #if ASSERT_ENABLED
166     PrecomputedMarginBefore precomputedMarginBefore(const Box& layoutBox) const { return m_precomputedMarginBeforeList.get(&layoutBox); }
167 #endif
168 
169     const BlockFormattingState& formattingState() const { return downcast<BlockFormattingState>(FormattingContext::formattingState()); }
170     BlockFormattingState& formattingState() { return downcast<BlockFormattingState>(FormattingContext::formattingState()); }
171 
172 private:
173     HashMap<const Box*, PrecomputedMarginBefore> m_precomputedMarginBeforeList;
174 };
