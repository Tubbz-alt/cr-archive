 38 const RegisterSet& StackmapGenerationParams::usedRegisters() const
 39 {
 40     ASSERT(m_context.code->needsUsedRegisters());
 41 
 42     return m_value->m_usedRegisters;
 43 }
 44 
 45 RegisterSet StackmapGenerationParams::unavailableRegisters() const
 46 {
 47     RegisterSet result = usedRegisters();
 48 
 49     RegisterSet unsavedCalleeSaves = RegisterSet::vmCalleeSaveRegisters();
 50     unsavedCalleeSaves.exclude(m_context.code->calleeSaveRegisters());
 51 
 52     result.merge(unsavedCalleeSaves);
 53 
 54     for (GPRReg gpr : m_gpScratch)
 55         result.clear(gpr);
 56     for (FPRReg fpr : m_fpScratch)
 57         result.clear(fpr);
 58 
 59     return result;
 60 }
 61 
 62 Vector<Box<MacroAssembler::Label>> StackmapGenerationParams::successorLabels() const
 63 {
 64     RELEASE_ASSERT(m_context.indexInBlock == m_context.currentBlock->size() - 1);
 65     RELEASE_ASSERT(m_value->effects().terminal);
 66 
 67     Vector<Box<MacroAssembler::Label>> result(m_context.currentBlock->numSuccessors());
 68     for (unsigned i = m_context.currentBlock->numSuccessors(); i--;)
 69         result[i] = m_context.blockLabels[m_context.currentBlock->successorBlock(i)];
 70     return result;
 71 }
 72 
 73 bool StackmapGenerationParams::fallsThroughToSuccessor(unsigned successorIndex) const
 74 {
 75     RELEASE_ASSERT(m_context.indexInBlock == m_context.currentBlock->size() - 1);
 76     RELEASE_ASSERT(m_value->effects().terminal);
 77 
 78     Air::BasicBlock* successor = m_context.currentBlock->successorBlock(successorIndex);
 79     Air::BasicBlock* nextBlock = m_context.code->findNextBlock(m_context.currentBlock);
 80     return successor == nextBlock;
 81 }

 83 Procedure& StackmapGenerationParams::proc() const
 84 {
 85     return m_context.code->proc();

 88 Air::Code& StackmapGenerationParams::code() const
 89 {
