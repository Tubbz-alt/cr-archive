 46 bool isOnAccessControlSimpleRequestMethodWhitelist(const String& method)
 47 {
 48     return method == "GET" || method == "HEAD" || method == "POST";
 49 }
 50 
 51 bool isSimpleCrossOriginAccessRequest(const String& method, const HTTPHeaderMap& headerMap)
 52 {
 53     if (!isOnAccessControlSimpleRequestMethodWhitelist(method))
 54         return false;
 55 
 56     for (const auto& header : headerMap) {
 57         if (!header.keyAsHTTPHeaderName || !isCrossOriginSafeRequestHeader(header.keyAsHTTPHeaderName.value(), header.value))
 58             return false;
 59     }
 60 
 61     return true;
 62 }

 64 void updateRequestReferrer(ResourceRequest& request, ReferrerPolicy referrerPolicy, const String& outgoingReferrer)
 65 {
 66     String newOutgoingReferrer = SecurityPolicy::generateReferrerHeader(referrerPolicy, request.url(), outgoingReferrer);
 67     if (newOutgoingReferrer.isEmpty())
 68         request.clearHTTPReferrer();
 69     else
 70         request.setHTTPReferrer(newOutgoingReferrer);
 71 }

 73 void updateRequestForAccessControl(ResourceRequest& request, SecurityOrigin& securityOrigin, StoredCredentialsPolicy storedCredentialsPolicy)
 74 {
 75     request.removeCredentials();
 76     request.setAllowCookies(storedCredentialsPolicy == StoredCredentialsPolicy::Use);
 77     request.setHTTPOrigin(securityOrigin.toString());
 78 }

 80 ResourceRequest createAccessControlPreflightRequest(const ResourceRequest& request, SecurityOrigin& securityOrigin, const String& referrer)
 81 {
 82     ResourceRequest preflightRequest(request.url());
 83     static const double platformDefaultTimeout = 0;
 84     preflightRequest.setTimeoutInterval(platformDefaultTimeout);
 85     updateRequestForAccessControl(preflightRequest, securityOrigin, StoredCredentialsPolicy::DoNotUse);
 86     preflightRequest.setHTTPMethod("OPTIONS");
 87     preflightRequest.setHTTPHeaderField(HTTPHeaderName::AccessControlRequestMethod, request.httpMethod());
 88     preflightRequest.setPriority(request.priority());
 89     if (!referrer.isNull())
 90         preflightRequest.setHTTPReferrer(referrer);
 91 
 92     const HTTPHeaderMap& requestHeaderFields = request.httpHeaderFields();
 93 
 94     if (!requestHeaderFields.isEmpty()) {
 95         Vector<String> unsafeHeaders;
 96         for (auto& headerField : requestHeaderFields) {
 97             if (!headerField.keyAsHTTPHeaderName || !isCrossOriginSafeRequestHeader(*headerField.keyAsHTTPHeaderName, headerField.value))
 98                 unsafeHeaders.append(headerField.key.convertToASCIILowercase());
 99         }
100 
101         std::sort(unsafeHeaders.begin(), unsafeHeaders.end(), WTF::codePointCompareLessThan);
102 
103         StringBuilder headerBuffer;
104 
105         bool appendComma = false;
106         for (const auto& headerField : unsafeHeaders) {
107             if (appendComma)
108                 headerBuffer.append(',');
109             else
110                 appendComma = true;
111 
112             headerBuffer.append(headerField);
113         }
114         if (!headerBuffer.isEmpty())
115             preflightRequest.setHTTPHeaderField(HTTPHeaderName::AccessControlRequestHeaders, headerBuffer.toString());
116     }
117 
118     return preflightRequest;
119 }

121 // https://html.spec.whatwg.org/multipage/urls-and-fetching.html#create-a-potential-cors-request
122 CachedResourceRequest createPotentialAccessControlRequest(ResourceRequest&& request, ResourceLoaderOptions&& options, Document& document, const String& crossOriginAttribute, SameOriginFlag sameOriginFlag)
123 {
124     ASSERT(options.mode == FetchOptions::Mode::NoCors);
125     if (!crossOriginAttribute.isNull())
126         options.mode = FetchOptions::Mode::Cors;
127     else if (sameOriginFlag == SameOriginFlag::Yes)
128         options.mode = FetchOptions::Mode::SameOrigin;
129 
130     if (crossOriginAttribute.isNull()) {
131         CachedResourceRequest cachedRequest { WTFMove(request), WTFMove(options) };
132         cachedRequest.setOrigin(document.securityOrigin());
133         return cachedRequest;
134     }
135 
136     FetchOptions::Credentials credentials = equalLettersIgnoringASCIICase(crossOriginAttribute, "omit")
137         ? FetchOptions::Credentials::Omit : equalLettersIgnoringASCIICase(crossOriginAttribute, "use-credentials")
138         ? FetchOptions::Credentials::Include : FetchOptions::Credentials::SameOrigin;
139     options.credentials = credentials;
140     switch (credentials) {
141     case FetchOptions::Credentials::Include:
142         options.storedCredentialsPolicy = StoredCredentialsPolicy::Use;
143         break;
144     case FetchOptions::Credentials::SameOrigin:
145         options.storedCredentialsPolicy = document.securityOrigin().canRequest(request.url()) ? StoredCredentialsPolicy::Use : StoredCredentialsPolicy::DoNotUse;
146         break;
147     case FetchOptions::Credentials::Omit:
148         options.storedCredentialsPolicy = StoredCredentialsPolicy::DoNotUse;
149     }
150 
151     CachedResourceRequest cachedRequest { WTFMove(request), WTFMove(options) };
152     updateRequestForAccessControl(cachedRequest.resourceRequest(), document.securityOrigin(), options.storedCredentialsPolicy);
153     return cachedRequest;
154 }

156 String validateCrossOriginRedirectionURL(const URL& redirectURL)
157 {
158     if (!LegacySchemeRegistry::shouldTreatURLSchemeAsCORSEnabled(redirectURL.protocol().toStringWithoutCopying()))
159         return makeString("not allowed to follow a cross-origin CORS redirection with non CORS scheme");
160 
161     if (redirectURL.hasUsername() || redirectURL.hasPassword())
162         return makeString("redirection URL ", redirectURL.string(), " has credentials");
163 
164     return { };
165 }

167 OptionSet<HTTPHeadersToKeepFromCleaning> httpHeadersToKeepFromCleaning(const HTTPHeaderMap& headers)
168 {
169     OptionSet<HTTPHeadersToKeepFromCleaning> headersToKeep;
170     if (headers.contains(HTTPHeaderName::ContentType))
171         headersToKeep.add(HTTPHeadersToKeepFromCleaning::ContentType);
172     if (headers.contains(HTTPHeaderName::Referer))
173         headersToKeep.add(HTTPHeadersToKeepFromCleaning::Referer);
174     if (headers.contains(HTTPHeaderName::Origin))
175         headersToKeep.add(HTTPHeadersToKeepFromCleaning::Origin);
176     if (headers.contains(HTTPHeaderName::UserAgent))
177         headersToKeep.add(HTTPHeadersToKeepFromCleaning::UserAgent);
178     if (headers.contains(HTTPHeaderName::AcceptEncoding))
179         headersToKeep.add(HTTPHeadersToKeepFromCleaning::AcceptEncoding);
180     return headersToKeep;
181 }

183 void cleanHTTPRequestHeadersForAccessControl(ResourceRequest& request, OptionSet<HTTPHeadersToKeepFromCleaning> headersToKeep)
184 {
185     // Remove headers that may have been added by the network layer that cause access control to fail.
186     if (!headersToKeep.contains(HTTPHeadersToKeepFromCleaning::ContentType)) {
187         auto contentType = request.httpContentType();
188         if (!contentType.isNull() && !isCrossOriginSafeRequestHeader(HTTPHeaderName::ContentType, contentType))
189             request.clearHTTPContentType();
190     }
191     if (!headersToKeep.contains(HTTPHeadersToKeepFromCleaning::Referer))
192         request.clearHTTPReferrer();
193     if (!headersToKeep.contains(HTTPHeadersToKeepFromCleaning::Origin))
194         request.clearHTTPOrigin();
195     if (!headersToKeep.contains(HTTPHeadersToKeepFromCleaning::UserAgent))
196         request.clearHTTPUserAgent();
197     if (!headersToKeep.contains(HTTPHeadersToKeepFromCleaning::AcceptEncoding))
198         request.clearHTTPAcceptEncoding();
199 }

201 bool passesAccessControlCheck(const ResourceResponse& response, StoredCredentialsPolicy storedCredentialsPolicy, SecurityOrigin& securityOrigin, String& errorDescription)
202 {
203     // A wildcard Access-Control-Allow-Origin can not be used if credentials are to be sent,
204     // even with Access-Control-Allow-Credentials set to true.
205     const String& accessControlOriginString = response.httpHeaderField(HTTPHeaderName::AccessControlAllowOrigin);
206     if (accessControlOriginString == "*" && storedCredentialsPolicy == StoredCredentialsPolicy::DoNotUse)
207         return true;
208 
209     String securityOriginString = securityOrigin.toString();
210     if (accessControlOriginString != securityOriginString) {
211         if (accessControlOriginString == "*")
212             errorDescription = "Cannot use wildcard in Access-Control-Allow-Origin when credentials flag is true."_s;
213         else if (accessControlOriginString.find(',') != notFound)
214             errorDescription = "Access-Control-Allow-Origin cannot contain more than one origin."_s;
215         else
216             errorDescription = makeString("Origin ", securityOriginString, " is not allowed by Access-Control-Allow-Origin.");
217         return false;
218     }
219 
220     if (storedCredentialsPolicy == StoredCredentialsPolicy::Use) {
221         const String& accessControlCredentialsString = response.httpHeaderField(HTTPHeaderName::AccessControlAllowCredentials);
222         if (accessControlCredentialsString != "true") {
223             errorDescription = "Credentials flag is true, but Access-Control-Allow-Credentials is not \"true\".";
224             return false;
225         }
226     }
227 
228     return true;
229 }

231 bool validatePreflightResponse(const ResourceRequest& request, const ResourceResponse& response, StoredCredentialsPolicy storedCredentialsPolicy, SecurityOrigin& securityOrigin, String& errorDescription)
232 {
233     if (!response.isSuccessful()) {
234         errorDescription = "Preflight response is not successful"_s;
235         return false;
236     }
237 
238     if (!passesAccessControlCheck(response, storedCredentialsPolicy, securityOrigin, errorDescription))
239         return false;
240 
241     auto result = makeUnique<CrossOriginPreflightResultCacheItem>(storedCredentialsPolicy);
242     if (!result->parse(response)
243         || !result->allowsCrossOriginMethod(request.httpMethod(), storedCredentialsPolicy, errorDescription)
244         || !result->allowsCrossOriginHeaders(request.httpHeaderFields(), storedCredentialsPolicy, errorDescription)) {
245         return false;
246     }
247 
248     CrossOriginPreflightResultCache::singleton().appendEntry(securityOrigin.toString(), request.url(), WTFMove(result));
249     return true;
250 }

252 static inline bool shouldCrossOriginResourcePolicyCancelLoad(const SecurityOrigin& origin, const ResourceResponse& response)
253 {
254     if (origin.canRequest(response.url()))
255         return false;
256 
257     auto policy = parseCrossOriginResourcePolicyHeader(response.httpHeaderField(HTTPHeaderName::CrossOriginResourcePolicy));
258 
259     if (policy == CrossOriginResourcePolicy::SameOrigin)
260         return true;
261 
262     if (policy == CrossOriginResourcePolicy::SameSite) {
263         if (origin.isUnique())
264             return true;
265 #if ENABLE(PUBLIC_SUFFIX_LIST)
266         if (!RegistrableDomain::uncheckedCreateFromHost(origin.host()).matches(response.url()))
267             return true;
268 #endif
269         if (origin.protocol() == "http" && response.url().protocol() == "https")
270             return true;
271     }
272 
273     return false;
274 }

276 Optional<ResourceError> validateCrossOriginResourcePolicy(const SecurityOrigin& origin, const URL& requestURL, const ResourceResponse& response)
277 {
278     if (shouldCrossOriginResourcePolicyCancelLoad(origin, response))
279         return ResourceError { errorDomainWebKitInternal, 0, requestURL, makeString("Cancelled load to ", response.url().stringCenterEllipsizedToLength(), " because it violates the resource's Cross-Origin-Resource-Policy response header."), ResourceError::Type::AccessControl };
280     return WTF::nullopt;
281 }

283 Optional<ResourceError> validateRangeRequestedFlag(const ResourceRequest& request, const ResourceResponse& response)
284 {
285     if (response.isRangeRequested() && response.httpStatusCode() == 206 && response.type() == ResourceResponse::Type::Opaque && !request.hasHTTPHeaderField(HTTPHeaderName::Range))
286         return ResourceError({ }, 0, response.url(), { }, ResourceError::Type::General);
287     return WTF::nullopt;
288 }
