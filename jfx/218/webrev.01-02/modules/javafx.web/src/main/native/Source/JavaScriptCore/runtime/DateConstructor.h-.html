25 namespace JSC {

27 class DatePrototype;
28 class GetterSetter;
29 
30 class DateConstructor final : public InternalFunction {
31 public:
32     typedef InternalFunction Base;
33     static const unsigned StructureFlags = Base::StructureFlags | HasStaticPropertyTable;
34 
35     static DateConstructor* create(VM& vm, Structure* structure, DatePrototype* datePrototype, GetterSetter*)
36     {
37         DateConstructor* constructor = new (NotNull, allocateCell<DateConstructor>(vm.heap)) DateConstructor(vm, structure);
38         constructor->finishCreation(vm, datePrototype);
39         return constructor;
40     }
41 
42     DECLARE_INFO;
43 
44     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
45     {
46         return Structure::create(vm, globalObject, prototype, TypeInfo(InternalFunctionType, StructureFlags), info());
47     }
48 
49 protected:
50     void finishCreation(VM&, DatePrototype*);
51 
52 private:
53     DateConstructor(VM&, Structure*);
54 };
