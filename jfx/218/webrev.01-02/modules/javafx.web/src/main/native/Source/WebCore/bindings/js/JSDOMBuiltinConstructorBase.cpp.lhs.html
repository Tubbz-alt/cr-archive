30 void JSDOMBuiltinConstructorBase::callFunctionWithCurrentArguments(JSC::ExecState& state, JSC::JSObject& thisObject, JSC::JSFunction& function)
31 {
32     JSC::VM& vm = state.vm();
33     auto scope = DECLARE_THROW_SCOPE(vm);
34     JSC::CallData callData;
35     JSC::CallType callType = JSC::getCallData(vm, &function, callData);
36     ASSERT(callType != CallType::None);
37 
38     JSC::MarkedArgumentBuffer arguments;
39     for (unsigned i = 0; i < state.argumentCount(); ++i)
40         arguments.append(state.uncheckedArgument(i));
41     if (UNLIKELY(arguments.hasOverflowed())) {
42         throwOutOfMemoryError(&state, scope);
43         return;
44     }
45     JSC::call(&state, &function, callType, callData, &thisObject, arguments);
46 }
47 
48 void JSDOMBuiltinConstructorBase::visitChildren(JSC::JSCell* cell, JSC::SlotVisitor& visitor)
49 {
50     auto* thisObject = jsCast<JSDOMBuiltinConstructorBase*>(cell);
51     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
52     Base::visitChildren(thisObject, visitor);
53     visitor.append(thisObject->m_initializeFunction);
54 }
