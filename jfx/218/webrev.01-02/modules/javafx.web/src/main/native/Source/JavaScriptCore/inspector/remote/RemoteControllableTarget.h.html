36 
37 namespace Inspector {
38 
39 class FrontendChannel;
40 
41 using TargetID = unsigned;
42 
43 class JS_EXPORT_PRIVATE RemoteControllableTarget {
44 public:
45     virtual ~RemoteControllableTarget();
46 
47     void init();
48     void update();
49 
50     virtual void connect(FrontendChannel&, bool isAutomaticConnection = false, bool immediatelyPause = false) = 0;
51     virtual void disconnect(FrontendChannel&) = 0;
52 
53     TargetID targetIdentifier() const { return m_identifier; }
54     void setTargetIdentifier(TargetID identifier) { m_identifier = identifier; }
55 
56     enum class Type {
57         Automation,
58         JavaScript,
59         Page,
60         ServiceWorker,
61         WebPage,
62     };
63     virtual Type type() const = 0;
64     virtual bool remoteControlAllowed() const = 0;
65     virtual void dispatchMessageFromRemote(const String& message) = 0;
66 
67 #if USE(CF)
68     // The dispatch block will be scheduled on a global run loop if null is returned.
69     virtual CFRunLoopRef targetRunLoop() const { return nullptr; }
70 #endif
71 
72 private:
73     TargetID m_identifier { 0 };
74 };
75 
76 } // namespace Inspector
77 
78 #define SPECIALIZE_TYPE_TRAITS_CONTROLLABLE_TARGET(ToClassName, ToClassType) \
79 SPECIALIZE_TYPE_TRAITS_BEGIN(ToClassName) \
80     static bool isType(const Inspector::RemoteControllableTarget& target) { return target.type() == Inspector::RemoteControllableTarget::Type::ToClassType; } \
81 SPECIALIZE_TYPE_TRAITS_END()

83 #endif // ENABLE(REMOTE_INSPECTOR)
