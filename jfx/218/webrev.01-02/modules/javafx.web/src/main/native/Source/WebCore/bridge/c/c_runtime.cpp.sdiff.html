41 JSValue CField::valueFromInstance(ExecState* exec, const Instance* inst) const
42 {
43     const CInstance* instance = static_cast<const CInstance*>(inst);
44     NPObject* obj = instance->getObject();
45     if (obj->_class->getProperty) {
46         NPVariant property;
47         VOID_TO_NPVARIANT(property);
48 
49         bool result;
50         {
51             JSLock::DropAllLocks dropAllLocks(exec);
52             result = obj->_class->getProperty(obj, _fieldIdentifier, &property);
53             CInstance::moveGlobalExceptionToExecState(exec);
54         }
55         if (result) {
56             JSValue result = convertNPVariantToValue(exec, &property, instance->rootObject());
57             _NPN_ReleaseVariantValue(&property);
58             return result;
59         }
60     }
61     return jsUndefined();
62 }
