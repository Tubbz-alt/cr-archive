154     // Enumerate all the SVGMemberAccessors recursively. The functor will be called and will
155     // be given the pair<QualifiedName, SVGMemberAccessor> till the functor returns false.
156     template<typename Functor>
157     static bool enumerateRecursively(const Functor& functor)
158     {
159         for (const auto& entry : attributeNameToAccessorMap()) {
160             if (!functor(entry))
161                 return false;
162         }
163         return enumerateRecursivelyBaseTypes(functor);
164     }
165 
166     // Returns true if OwnerType owns a property whose name is attributeName.
167     static bool isKnownAttribute(const QualifiedName& attributeName)
168     {
169         return findAccessor(attributeName);
170     }
171 
172     // Returns true if OwnerType owns a property whose name is attributeName
173     // and its type is SVGAnimatedLength.
174     static bool isAnimatedLengthAttribute(const QualifiedName& attributeName)
175     {
176         if (const auto* accessor = findAccessor(attributeName))
177             return accessor->isAnimatedLength();
178         return false;
179     }
180 
181     QualifiedName propertyAttributeName(const SVGProperty& property) const override
182     {
183         QualifiedName attributeName = nullQName();
184         enumerateRecursively([&](const auto& entry) -> bool {
185             if (!entry.value->matches(m_owner, property))
186                 return true;
187             attributeName = entry.key;
188             return false;
189         });
190         return attributeName;
191     }

193     QualifiedName animatedPropertyAttributeName(const SVGAnimatedProperty& animatedProperty) const override
194     {
195         QualifiedName attributeName = nullQName();
196         enumerateRecursively([&](const auto& entry) -> bool {
197             if (!entry.value->matches(m_owner, animatedProperty))
198                 return true;
199             attributeName = entry.key;
200             return false;
201         });
202         return attributeName;
203     }

205     void setAnimatedPropertDirty(const QualifiedName& attributeName, SVGAnimatedProperty& animatedProperty) const override
206     {
207         enumerateRecursively([&](const auto& entry) -> bool {
208             if (!entry.key.matches(attributeName))
209                 return true;
210             entry.value->setDirty(m_owner, animatedProperty);
211             return false;
212         });
213     }

215     // Detach all the properties recursively from their OwnerTypes.
216     void detachAllProperties() const override
217     {
218         enumerateRecursively([&](const auto& entry) -> bool {
219             entry.value->detach(m_owner);
220             return true;
221         });
222     }

224     // Finds the property whose name is attributeName and returns the synchronize
225     // string through the associated SVGMemberAccessor.
226     Optional<String> synchronize(const QualifiedName& attributeName) const override
227     {
228         Optional<String> value;
229         enumerateRecursively([&](const auto& entry) -> bool {
230             if (!entry.key.matches(attributeName))
231                 return true;
232             value = entry.value->synchronize(m_owner);
233             return false;
234         });
235         return value;
236     }
237 
238     // Enumerate recursively the SVGMemberAccessors of the OwnerType and all its BaseTypes.
239     // Collect all the pairs <AttributeName, String> only for the dirty properties.
240     HashMap<QualifiedName, String> synchronizeAllAttributes() const override
241     {
242         HashMap<QualifiedName, String> map;
243         enumerateRecursively([&](const auto& entry) -> bool {
244             if (auto string = entry.value->synchronize(m_owner))
245                 map.add(entry.key, *string);
246             return true;
247         });
248         return map;
249     }

251     bool isAnimatedPropertyAttribute(const QualifiedName& attributeName) const override
252     {
253         bool isAnimatedPropertyAttribute = false;
254         enumerateRecursively([&attributeName, &isAnimatedPropertyAttribute](const auto& entry) -> bool {
255             if (!entry.key.matches(attributeName))
256                 return true;
257             isAnimatedPropertyAttribute = entry.value->isAnimatedProperty();
258             return false;
259         });
260         return isAnimatedPropertyAttribute;
261     }

263     bool isAnimatedStylePropertyAttribute(const QualifiedName& attributeName) const override
264     {
265         static NeverDestroyed<HashSet<QualifiedName::QualifiedNameImpl*>> animatedStyleAttributes = std::initializer_list<QualifiedName::QualifiedNameImpl*> {
266             SVGNames::cxAttr->impl(),
267             SVGNames::cyAttr->impl(),
268             SVGNames::rAttr->impl(),
269             SVGNames::rxAttr->impl(),
270             SVGNames::ryAttr->impl(),
271             SVGNames::heightAttr->impl(),
272             SVGNames::widthAttr->impl(),
273             SVGNames::xAttr->impl(),
274             SVGNames::yAttr->impl()
275         };
276         return isAnimatedLengthAttribute(attributeName) && animatedStyleAttributes.get().contains(attributeName.impl());
277     }

279     RefPtr<SVGAttributeAnimator> createAnimator(const QualifiedName& attributeName, AnimationMode animationMode, CalcMode calcMode, bool isAccumulated, bool isAdditive) const override
280     {
281         RefPtr<SVGAttributeAnimator> animator;
282         enumerateRecursively([&](const auto& entry) -> bool {
283             if (!entry.key.matches(attributeName))
284                 return true;
285             animator = entry.value->createAnimator(m_owner, attributeName, animationMode, calcMode, isAccumulated, isAdditive);
286             return false;
287         });
288         return animator;
289     }

291     void appendAnimatedInstance(const QualifiedName& attributeName, SVGAttributeAnimator& animator) const override
292     {
293         enumerateRecursively([&](const auto& entry) -> bool {
294             if (!entry.key.matches(attributeName))
295                 return true;
296             entry.value->appendAnimatedInstance(m_owner, animator);
297             return false;
298         });
299     }

301 private:
302     // Singleton map for every OwnerType.
303     static HashMap<QualifiedName, const SVGMemberAccessor<OwnerType>*>& attributeNameToAccessorMap()
304     {
305         static NeverDestroyed<HashMap<QualifiedName, const SVGMemberAccessor<OwnerType>*>> attributeNameToAccessorMap;
306         return attributeNameToAccessorMap;
307     }
308 
309     static void registerProperty(const QualifiedName& attributeName, const SVGMemberAccessor<OwnerType>& propertyAccessor)
310     {
311         attributeNameToAccessorMap().add(attributeName, &propertyAccessor);
312     }
313 
314     // This is a template function with parameter 'I' whose default value = 0. So you can call it without any parameter
315     // from enumerateRecursively(). It returns true and is enable_if<I == sizeof...(BaseTypes)>. So it is mainly for
316     // breaking the recursion.
317     template<typename Functor, size_t I = 0>
318     static typename std::enable_if<I == sizeof...(BaseTypes), bool>::type enumerateRecursivelyBaseTypes(const Functor&) { return true; }
319 
320     // This version of animatedTypesBaseTypes() is enable_if<I < sizeof...(BaseTypes)>.
321     template<typename Functor, size_t I = 0>
322     static typename std::enable_if<I < sizeof...(BaseTypes), bool>::type enumerateRecursivelyBaseTypes(const Functor& functor)
323     {
324         // Get the base type at index 'I' using std::tuple and std::tuple_element.
325         using BaseType = typename std::tuple_element<I, typename std::tuple<BaseTypes...>>::type;
326         if (!BaseType::PropertyRegistry::enumerateRecursively(functor))
327             return false;
328         // BaseType does not want to break the recursion. So recurse to the next BaseType.
329         return enumerateRecursivelyBaseTypes<Functor, I + 1>(functor);
330     }
331 
332     static const SVGMemberAccessor<OwnerType>* findAccessor(const QualifiedName& attributeName)
333     {
334         // Here we need to loop through the entries in the map and use matches() to compare them with attributeName.
335         // m_map.contains() uses QualifiedName::operator==() which compares the impl pointers only while matches()
336         // compares the contents if the impl pointers differ.
337         auto it = std::find_if(attributeNameToAccessorMap().begin(), attributeNameToAccessorMap().end(), [&attributeName](const auto& entry) -> bool {
338             return entry.key.matches(attributeName);
339         });
340         return it != attributeNameToAccessorMap().end() ? it->value : nullptr;
341     }
