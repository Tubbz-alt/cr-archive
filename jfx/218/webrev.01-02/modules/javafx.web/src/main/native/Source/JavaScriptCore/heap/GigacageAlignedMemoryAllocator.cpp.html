26 #include "config.h"
27 #include "GigacageAlignedMemoryAllocator.h"
28 
29 namespace JSC {
30 
31 GigacageAlignedMemoryAllocator::GigacageAlignedMemoryAllocator(Gigacage::Kind kind)
32     : m_kind(kind)
33 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
34     , m_heap(makeString("WebKit GigacageAlignedMemoryAllocator ", Gigacage::name(m_kind)).utf8().data())
35 #endif
36 {
37 }
38 
39 GigacageAlignedMemoryAllocator::~GigacageAlignedMemoryAllocator()
40 {
41 }
42 
43 void* GigacageAlignedMemoryAllocator::tryAllocateAlignedMemory(size_t alignment, size_t size)
44 {
45 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
46     return m_heap.memalign(alignment, size, true);
47 #else
48     return Gigacage::tryAlignedMalloc(m_kind, alignment, size);
49 #endif
50 }
51 
52 void GigacageAlignedMemoryAllocator::freeAlignedMemory(void* basePtr)
53 {
54 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
55     return m_heap.free(basePtr);
56 #else
57     Gigacage::alignedFree(m_kind, basePtr);
58 #endif
59 }
60 
61 void GigacageAlignedMemoryAllocator::dump(PrintStream& out) const
62 {
63     out.print(Gigacage::name(m_kind), "Gigacage");
64 }
65 
66 void* GigacageAlignedMemoryAllocator::tryAllocateMemory(size_t size)
67 {
68 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
69     return m_heap.malloc(size);
70 #else
71     return Gigacage::tryMalloc(m_kind, size);
72 #endif
73 }
74 
75 void GigacageAlignedMemoryAllocator::freeMemory(void* pointer)
76 {
77 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
78     m_heap.free(pointer);
79 #else
80     Gigacage::free(m_kind, pointer);
81 #endif
82 }
83 
84 void* GigacageAlignedMemoryAllocator::tryReallocateMemory(void* pointer, size_t size)
85 {
86 #if ENABLE(MALLOC_HEAP_BREAKDOWN)
87     return m_heap.realloc(pointer, size);
88 #else
89     return Gigacage::tryRealloc(m_kind, pointer, size);
90 #endif
91 }
92 
93 } // namespace JSC
