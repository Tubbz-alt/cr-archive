31 namespace WebCore {
32 
33 CSSCanvasValue::~CSSCanvasValue()
34 {
35     if (m_element)
36         m_element->removeObserver(m_canvasObserver);
37 }
38 
39 String CSSCanvasValue::customCSSText() const
40 {
41     return makeString("-webkit-canvas(", m_name, ')');
42 }
43 
44 void CSSCanvasValue::canvasChanged(HTMLCanvasElement&, const FloatRect& changedRect)
45 {
46     IntRect imageChangeRect = enclosingIntRect(changedRect);
47     for (auto it = clients().begin(), end = clients().end(); it != end; ++it)
48         it->key->imageChanged(static_cast<WrappedImagePtr>(this), &imageChangeRect);
49 }
50 
51 void CSSCanvasValue::canvasResized(HTMLCanvasElement&)
52 {
53     for (auto it = clients().begin(), end = clients().end(); it != end; ++it)
54         it->key->imageChanged(static_cast<WrappedImagePtr>(this));
55 }
56 
57 void CSSCanvasValue::canvasDestroyed(HTMLCanvasElement& element)
58 {
59     ASSERT_UNUSED(&element, &element == m_element);
60     m_element = nullptr;
61 }
62 
63 FloatSize CSSCanvasValue::fixedSize(const RenderElement* renderer)
64 {
65     if (HTMLCanvasElement* elt = element(renderer->document()))
66         return FloatSize(elt->width(), elt->height());
67     return FloatSize();
68 }
69 
70 HTMLCanvasElement* CSSCanvasValue::element(Document& document)
71 {
72      if (!m_element) {
73         m_element = document.getCSSCanvasElement(m_name);
74         if (!m_element)
75             return nullptr;
76         m_element->addObserver(m_canvasObserver);
77     }
78     return m_element;
79 }
80 
81 RefPtr<Image> CSSCanvasValue::image(RenderElement* renderer, const FloatSize& /*size*/)
82 {
83     ASSERT(clients().contains(renderer));
84     HTMLCanvasElement* element = this->element(renderer->document());
85     if (!element || !element->buffer())
86         return nullptr;
87     return element->copiedImage();
88 }
89 
90 bool CSSCanvasValue::equals(const CSSCanvasValue& other) const
91 {
92     return m_name == other.m_name;
93 }
