 26 #pragma once
 27 
 28 #if ENABLE(INDEXED_DATABASE)
 29 
 30 #include "IDBIndexInfo.h"
 31 #include "IDBKeyPath.h"
 32 #include <wtf/HashMap.h>
 33 #include <wtf/text/WTFString.h>
 34 
 35 namespace WebCore {
 36 
 37 class IDBObjectStoreInfo {
 38 public:
 39     WEBCORE_EXPORT IDBObjectStoreInfo();
 40     IDBObjectStoreInfo(uint64_t identifier, const String& name, Optional<IDBKeyPath>&&, bool autoIncrement);
 41 
 42     uint64_t identifier() const { return m_identifier; }
 43     const String& name() const { return m_name; }
 44     const Optional<IDBKeyPath>& keyPath() const { return m_keyPath; }
 45     bool autoIncrement() const { return m_autoIncrement; }
 46 
 47     void rename(const String& newName) { m_name = newName; }
 48 
 49     WEBCORE_EXPORT IDBObjectStoreInfo isolatedCopy() const;
 50 
 51     IDBIndexInfo createNewIndex(uint64_t indexID, const String& name, IDBKeyPath&&, bool unique, bool multiEntry);
 52     void addExistingIndex(const IDBIndexInfo&);
 53     bool hasIndex(const String& name) const;
 54     bool hasIndex(uint64_t indexIdentifier) const;
 55     IDBIndexInfo* infoForExistingIndex(const String& name);
 56     IDBIndexInfo* infoForExistingIndex(uint64_t identifier);
 57 
 58     Vector<String> indexNames() const;
 59     const HashMap<uint64_t, IDBIndexInfo>& indexMap() const { return m_indexMap; }
 60 
 61     void deleteIndex(const String& indexName);
 62     void deleteIndex(uint64_t indexIdentifier);
 63 
 64     template<class Encoder> void encode(Encoder&) const;
 65     template<class Decoder> static bool decode(Decoder&, IDBObjectStoreInfo&);
 66 
 67 #if !LOG_DISABLED
 68     String loggingString(int indent = 0) const;
 69     String condensedLoggingString() const;
 70 #endif
 71 
 72 private:
 73     uint64_t m_identifier { 0 };
 74     String m_name;
 75     Optional<IDBKeyPath> m_keyPath;
 76     bool m_autoIncrement { false };
 77 
 78     HashMap<uint64_t, IDBIndexInfo> m_indexMap;
 79 };
 80 
 81 template<class Encoder>
 82 void IDBObjectStoreInfo::encode(Encoder& encoder) const
 83 {
 84     encoder << m_identifier << m_name << m_keyPath << m_autoIncrement << m_indexMap;
 85 }
 86 
 87 template<class Decoder>
 88 bool IDBObjectStoreInfo::decode(Decoder& decoder, IDBObjectStoreInfo& info)
 89 {
 90     if (!decoder.decode(info.m_identifier))
 91         return false;
 92 
 93     if (!decoder.decode(info.m_name))
 94         return false;
 95 
 96     if (!decoder.decode(info.m_keyPath))
 97         return false;
 98 
 99     if (!decoder.decode(info.m_autoIncrement))
100         return false;
101 
102     if (!decoder.decode(info.m_indexMap))
103         return false;
104 
105     return true;
106 }
107 
108 } // namespace WebCore
109 
110 #endif // ENABLE(INDEXED_DATABASE)
