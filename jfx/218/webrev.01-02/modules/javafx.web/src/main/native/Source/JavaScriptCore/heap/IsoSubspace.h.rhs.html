33 namespace JSC {

35 class IsoAlignedMemoryAllocator;
36 class IsoCellSet;
37 
38 class IsoSubspace : public Subspace {
39 public:
40     JS_EXPORT_PRIVATE IsoSubspace(CString name, Heap&, HeapCellType*, size_t size, uint8_t numberOfLowerTierCells);
41     JS_EXPORT_PRIVATE ~IsoSubspace();
42 
43     size_t cellSize() { return m_directory.cellSize(); }
44 
45     Allocator allocatorFor(size_t, AllocatorForMode) override;
46     Allocator allocatorForNonVirtual(size_t, AllocatorForMode);
47 
48     void* allocate(VM&, size_t, GCDeferralContext*, AllocationFailureMode) override;
49     void* allocateNonVirtual(VM&, size_t, GCDeferralContext*, AllocationFailureMode);
50 
51     void sweepLowerTierCell(PreciseAllocation*);
52     void clearIsoCellSetBit(PreciseAllocation*);
53 
54     void* tryAllocateFromLowerTier();
55     void destroyLowerTierFreeList();
56 
57     void sweep();
58 
59 private:
60     friend class IsoCellSet;
61 
62     void didResizeBits(unsigned newSize) override;
63     void didRemoveBlock(unsigned blockIndex) override;
64     void didBeginSweepingToFreeList(MarkedBlock::Handle*) override;
65 

66     BlockDirectory m_directory;
67     LocalAllocator m_localAllocator;
68     std::unique_ptr<IsoAlignedMemoryAllocator> m_isoAlignedMemoryAllocator;
69     SentinelLinkedList<PreciseAllocation, PackedRawSentinelNode<PreciseAllocation>> m_lowerTierFreeList;
70     SentinelLinkedList<IsoCellSet, PackedRawSentinelNode<IsoCellSet>> m_cellSets;
71 };
72 
73 ALWAYS_INLINE Allocator IsoSubspace::allocatorForNonVirtual(size_t size, AllocatorForMode)
74 {
75     RELEASE_ASSERT(WTF::roundUpToMultipleOf<MarkedBlock::atomSize>(size) == cellSize());
76     return Allocator(&m_localAllocator);
77 }

79 #define ISO_SUBSPACE_INIT(heap, heapCellType, type) ("Isolated " #type " Space", (heap), (heapCellType), sizeof(type), type::numberOfLowerTierCells)
80 
81 template<typename T>
82 struct isAllocatedFromIsoSubspace {
83     static constexpr bool value = std::is_same<std::result_of_t<decltype(T::template subspaceFor<T, SubspaceAccess::OnMainThread>)&(VM&)>, IsoSubspace*>::value;
84 };
