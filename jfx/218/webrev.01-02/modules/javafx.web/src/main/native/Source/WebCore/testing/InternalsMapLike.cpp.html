26 #include "config.h"
27 #include "InternalsMapLike.h"
28 
29 #include "IDLTypes.h"
30 #include "JSDOMMapLike.h"
31 #include <wtf/Vector.h>
32 
33 namespace WebCore {
34 
35 InternalsMapLike::InternalsMapLike()
36 {
37     m_values.add("init", 0);
38 }
39 
40 void InternalsMapLike::initializeMapLike(DOMMapAdapter& map)
41 {
42     for (auto& keyValue : m_values)
43         map.set<IDLDOMString, IDLUnsignedLong>(keyValue.key, keyValue.value);
44 }
45 
46 void InternalsMapLike::setFromMapLike(String&& key, unsigned value)
47 {
48     m_values.set(WTFMove(key), value);
49 }
50 
51 void InternalsMapLike::clear()
52 {
53     m_values.clear();
54 }
55 
56 bool InternalsMapLike::remove(const String& key)
57 {
58     return m_values.remove(key);
59 }
60 
61 Vector<String> InternalsMapLike::inspectKeys() const
62 {
63     auto result = copyToVector(m_values.keys());
64     std::sort(result.begin(), result.end(), WTF::codePointCompareLessThan);
65     return result;
66 }
67 
68 Vector<unsigned> InternalsMapLike::inspectValues() const
69 {
70     auto result = copyToVector(m_values.values());
71     std::sort(result.begin(), result.end(), std::less<unsigned>());
72     return result;
73 }

75 } // namespace WebCore
