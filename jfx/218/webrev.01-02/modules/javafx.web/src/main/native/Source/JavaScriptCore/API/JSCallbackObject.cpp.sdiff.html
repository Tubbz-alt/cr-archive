64 template <>
65 IsoSubspace* JSCallbackObject<JSNonFinalObject>::subspaceForImpl(VM& vm, SubspaceAccess mode)
66 {
67     switch (mode) {
68     case SubspaceAccess::OnMainThread:
69         return vm.callbackObjectSpace<SubspaceAccess::OnMainThread>();
70     case SubspaceAccess::Concurrently:
71         return vm.callbackObjectSpace<SubspaceAccess::Concurrently>();
72     }
73     RELEASE_ASSERT_NOT_REACHED();
74     return nullptr;
75 }

77 template <>
78 IsoSubspace* JSCallbackObject<JSGlobalObject>::subspaceForImpl(VM& vm, SubspaceAccess mode)
79 {
80     switch (mode) {
81     case SubspaceAccess::OnMainThread:
82         return vm.callbackGlobalObjectSpace<SubspaceAccess::OnMainThread>();
83     case SubspaceAccess::Concurrently:
84         return vm.callbackGlobalObjectSpace<SubspaceAccess::Concurrently>();
85     }
86     RELEASE_ASSERT_NOT_REACHED();
