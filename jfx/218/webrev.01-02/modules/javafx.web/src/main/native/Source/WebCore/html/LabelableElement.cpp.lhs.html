37 LabelableElement::LabelableElement(const QualifiedName& tagName, Document& document)
38     : HTMLElement(tagName, document)
39 {
40 }
41 
42 LabelableElement::~LabelableElement() = default;
43 
44 RefPtr<NodeList> LabelableElement::labels()
45 {
46     if (!supportLabels())
47         return nullptr;
48 
49     return ensureRareData().ensureNodeLists().addCacheWithAtomicName<LabelsNodeList>(*this, starAtom());
50 }
