26 #pragma once
27 
28 #include "SubstituteResource.h"
29 #include <wtf/WeakPtr.h>
30 
31 namespace WebCore {
32 
33 class ApplicationCacheResource : public SubstituteResource, public CanMakeWeakPtr<ApplicationCacheResource> {
34 public:
35     enum Type {
36         Master = 1 << 0,
37         Manifest = 1 << 1,
38         Explicit = 1 << 2,
39         Foreign = 1 << 3,
40         Fallback = 1 << 4
41     };
42 
43     static Ref<ApplicationCacheResource> create(const URL&, const ResourceResponse&, unsigned type, RefPtr<SharedBuffer>&& = SharedBuffer::create(), const String& path = String());
44 
45     unsigned type() const { return m_type; }
46     void addType(unsigned type);
47 
48     void setStorageID(unsigned storageID) { m_storageID = storageID; }
49     unsigned storageID() const { return m_storageID; }
50     void clearStorageID() { m_storageID = 0; }
51     int64_t estimatedSizeInStorage();
52 
53     const String& path() const { return m_path; }
54     void setPath(const String& path) { m_path = path; }
55 
56 #ifndef NDEBUG
57     static void dumpType(unsigned type);
58 #endif
59 
60 private:
61     ApplicationCacheResource(URL&&, ResourceResponse&&, unsigned type, Ref<SharedBuffer>&&, const String& path);
62 
63     void deliver(ResourceLoader&) override;
64 
65     unsigned m_type;
66     unsigned m_storageID;
67     int64_t m_estimatedSizeInStorage;
68     String m_path;
69 };
70 
71 } // namespace WebCore
