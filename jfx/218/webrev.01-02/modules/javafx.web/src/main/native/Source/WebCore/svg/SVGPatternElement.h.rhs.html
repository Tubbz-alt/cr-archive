22 #pragma once
23 
24 #include "SVGElement.h"

25 #include "SVGFitToViewBox.h"
26 #include "SVGNames.h"
27 #include "SVGTests.h"
28 #include "SVGURIReference.h"
29 #include "SVGUnitTypes.h"
30 
31 namespace WebCore {
32 
33 struct PatternAttributes;
34 
35 class SVGPatternElement final : public SVGElement, public SVGFitToViewBox, public SVGTests, public SVGURIReference {
36     WTF_MAKE_ISO_ALLOCATED(SVGPatternElement);
37 public:
38     static Ref<SVGPatternElement> create(const QualifiedName&, Document&);
39 
40     void collectPatternAttributes(PatternAttributes&) const;
41 
42     AffineTransform localCoordinateSpaceTransform(SVGLocatable::CTMScope) const final;
43 
44     const SVGLengthValue& x() const { return m_x->currentValue(); }
45     const SVGLengthValue& y() const { return m_y->currentValue(); }
46     const SVGLengthValue& width() const { return m_width->currentValue(); }
47     const SVGLengthValue& height() const { return m_height->currentValue(); }
48     SVGUnitTypes::SVGUnitType patternUnits() const { return m_patternUnits->currentValue<SVGUnitTypes::SVGUnitType>(); }
49     SVGUnitTypes::SVGUnitType patternContentUnits() const { return m_patternContentUnits->currentValue<SVGUnitTypes::SVGUnitType>(); }
50     const SVGTransformList& patternTransform() const { return m_patternTransform->currentValue(); }
51 
52     SVGAnimatedLength& xAnimated() { return m_x; }
53     SVGAnimatedLength& yAnimated() { return m_y; }
54     SVGAnimatedLength& widthAnimated() { return m_width; }
55     SVGAnimatedLength& heightAnimated() { return m_height; }
56     SVGAnimatedEnumeration& patternUnitsAnimated() { return m_patternUnits; }
57     SVGAnimatedEnumeration& patternContentUnitsAnimated() { return m_patternContentUnits; }
58     SVGAnimatedTransformList& patternTransformAnimated() { return m_patternTransform; }
59 
60 private:
61     SVGPatternElement(const QualifiedName&, Document&);
62 
63     using PropertyRegistry = SVGPropertyOwnerRegistry<SVGPatternElement, SVGElement, SVGFitToViewBox, SVGTests, SVGURIReference>;
64     const SVGPropertyRegistry& propertyRegistry() const final { return m_propertyRegistry; }
65 
66     void parseAttribute(const QualifiedName&, const AtomString&) final;
67     void svgAttributeChanged(const QualifiedName&) final;
68     void childrenChanged(const ChildChange&) final;
69 
70     RenderPtr<RenderElement> createElementRenderer(RenderStyle&&, const RenderTreePosition&) final;
71 
72     bool isValid() const final { return SVGTests::isValid(); }
73     bool needsPendingResourceHandling() const final { return false; }
74     bool selfHasRelativeLengths() const final { return true; }
75 
76     PropertyRegistry m_propertyRegistry { *this };
77     Ref<SVGAnimatedLength> m_x { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
78     Ref<SVGAnimatedLength> m_y { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
79     Ref<SVGAnimatedLength> m_width { SVGAnimatedLength::create(this, SVGLengthMode::Width) };
80     Ref<SVGAnimatedLength> m_height { SVGAnimatedLength::create(this, SVGLengthMode::Height) };
81     Ref<SVGAnimatedEnumeration> m_patternUnits { SVGAnimatedEnumeration::create(this, SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX) };
82     Ref<SVGAnimatedEnumeration> m_patternContentUnits { SVGAnimatedEnumeration::create(this, SVGUnitTypes::SVG_UNIT_TYPE_USERSPACEONUSE) };
83     Ref<SVGAnimatedTransformList> m_patternTransform { SVGAnimatedTransformList::create(this) };
84 };
85 
86 } // namespace WebCore
