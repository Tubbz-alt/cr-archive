42 Instruction* returnToThrow(VM& vm)
43 {
44     UNUSED_PARAM(vm);
45 #if LLINT_TRACING
46     if (UNLIKELY(Options::traceLLIntSlowPath())) {
47         auto scope = DECLARE_CATCH_SCOPE(vm);

48         dataLog("Throwing exception ", JSValue(scope.exception()), " (returnToThrow).\n");
49     }
50 #endif
51     return LLInt::exceptionInstructions();
52 }
53 
54 void* callToThrow(VM& vm)
55 {
56     UNUSED_PARAM(vm);
57 #if LLINT_TRACING
58     if (UNLIKELY(Options::traceLLIntSlowPath())) {
59         auto scope = DECLARE_CATCH_SCOPE(vm);

60         dataLog("Throwing exception ", JSValue(scope.exception()), " (callToThrow).\n");
61     }
62 #endif
63     return LLInt::getCodePtr<ExceptionHandlerPtrTag>(llint_throw_during_call_trampoline).executableAddress();
64 }
