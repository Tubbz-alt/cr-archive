 26 #pragma once
 27 
 28 #include <atomic>
 29 #include <mutex>
 30 #include <wtf/HashTraits.h>
 31 #include <wtf/NeverDestroyed.h>
 32 #include <wtf/text/TextStream.h>
 33 #include <wtf/text/WTFString.h>
 34 
 35 namespace WTF {
 36 
 37 class ObjectIdentifierBase {
 38 protected:
 39     WTF_EXPORT_PRIVATE static uint64_t generateIdentifierInternal();
 40     WTF_EXPORT_PRIVATE static uint64_t generateThreadSafeIdentifierInternal();
 41 };
 42 
 43 template<typename T> class ObjectIdentifier : private ObjectIdentifierBase {
 44 public:
 45     static ObjectIdentifier generate()
 46     {
 47         RELEASE_ASSERT(!m_generationProtected);
 48         return ObjectIdentifier { generateIdentifierInternal() };
 49     }
 50 
 51     static ObjectIdentifier generateThreadSafe()
 52     {
 53         RELEASE_ASSERT(!m_generationProtected);
 54         return ObjectIdentifier { generateThreadSafeIdentifierInternal() };
 55     }
 56 
 57     static void enableGenerationProtection()
 58     {
 59         m_generationProtected = true;
 60     }
 61 
 62     ObjectIdentifier() = default;
 63 
 64     ObjectIdentifier(HashTableDeletedValueType) : m_identifier(hashTableDeletedValue()) { }
 65     bool isHashTableDeletedValue() const { return m_identifier == hashTableDeletedValue(); }
 66 
 67     template<typename Encoder> void encode(Encoder& encoder) const
 68     {
 69         ASSERT(isValidIdentifier(m_identifier));
 70         encoder << m_identifier;
 71     }
 72     template<typename Decoder> static Optional<ObjectIdentifier> decode(Decoder& decoder)
 73     {
 74         Optional<uint64_t> identifier;
 75         decoder >> identifier;
 76         if (!identifier || !isValidIdentifier(*identifier))
 77             return WTF::nullopt;

 78         return ObjectIdentifier { *identifier };
 79     }
 80 
 81     bool operator==(const ObjectIdentifier& other) const
 82     {
 83         return m_identifier == other.m_identifier;
 84     }
 85 
 86     bool operator!=(const ObjectIdentifier& other) const
 87     {
 88         return m_identifier != other.m_identifier;
 89     }
 90 
 91     uint64_t toUInt64() const { return m_identifier; }
 92     explicit operator bool() const { return m_identifier; }
 93 
 94     String loggingString() const
 95     {
 96         return String::number(m_identifier);
 97     }
 98 
 99     struct MarkableTraits {
100         static bool isEmptyValue(ObjectIdentifier identifier)
101         {
102             return !identifier.m_identifier;
103         }
104 
105         static constexpr ObjectIdentifier emptyValue()
106         {
107             return ObjectIdentifier();
108         }
109     };
110 
111 private:
112     template<typename U> friend ObjectIdentifier<U> makeObjectIdentifier(uint64_t);
113     friend struct HashTraits<ObjectIdentifier>;
114     template<typename U> friend struct ObjectIdentifierHash;
115 
116     static uint64_t hashTableDeletedValue() { return std::numeric_limits<uint64_t>::max(); }
117     static bool isValidIdentifier(uint64_t identifier) { return identifier && identifier != hashTableDeletedValue(); }
118 
119     explicit constexpr ObjectIdentifier(uint64_t identifier)
120         : m_identifier(identifier)
121     {
122     }
123 
124     uint64_t m_identifier { 0 };
125     inline static bool m_generationProtected { false };
126 };
127 
128 template<typename T> inline ObjectIdentifier<T> makeObjectIdentifier(uint64_t identifier)
129 {
130     return ObjectIdentifier<T> { identifier };
131 }
132 
133 template<typename T> struct ObjectIdentifierHash {
134     static unsigned hash(const ObjectIdentifier<T>& identifier) { return intHash(identifier.m_identifier); }
135     static bool equal(const ObjectIdentifier<T>& a, const ObjectIdentifier<T>& b) { return a == b; }
136     static constexpr bool safeToCompareToEmptyOrDeleted = true;
137 };
138 
139 template<typename T> struct HashTraits<ObjectIdentifier<T>> : SimpleClassHashTraits<ObjectIdentifier<T>> { };
140 
141 template<typename T> struct DefaultHash<ObjectIdentifier<T>> {
142     typedef ObjectIdentifierHash<T> Hash;
143 };
144 
145 template<typename T>
146 TextStream& operator<<(TextStream& ts, const ObjectIdentifier<T>& identifier)
147 {
148     ts << identifier.toUInt64();
149     return ts;
150 }
151 
152 } // namespace WTF
