32 #pragma once
33 
34 #include "Event.h"
35 #include "JSValueInWrappedObject.h"
36 #include "SerializedScriptValue.h"
37 #include <JavaScriptCore/Strong.h>
38 #include <wtf/text/WTFString.h>
39 
40 namespace WebCore {
41 
42 class ErrorEvent final : public Event {
43     WTF_MAKE_ISO_ALLOCATED(ErrorEvent);
44 public:
45     static Ref<ErrorEvent> create(const String& message, const String& fileName, unsigned lineNumber, unsigned columnNumber, JSC::Strong<JSC::Unknown> error)
46     {
47         return adoptRef(*new ErrorEvent(message, fileName, lineNumber, columnNumber, error));
48     }
49 
50     struct Init : EventInit {
51         String message;
52         String filename;
53         unsigned lineno { 0 };
54         unsigned colno { 0 };
55         JSC::JSValue error;
56     };
57 
58     static Ref<ErrorEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
59     {
60         return adoptRef(*new ErrorEvent(type, initializer, isTrusted));
61     }
62 
63     virtual ~ErrorEvent();
64 
65     const String& message() const { return m_message; }
66     const String& filename() const { return m_fileName; }
67     unsigned lineno() const { return m_lineNumber; }
68     unsigned colno() const { return m_columnNumber; }
69     JSC::JSValue error(JSC::JSGlobalObject&);
70 
71     const JSValueInWrappedObject& originalError() const { return m_error; }
72     SerializedScriptValue* serializedError() const { return m_serializedError.get(); }
73 
74     EventInterface eventInterface() const override;
75 
76     RefPtr<SerializedScriptValue> trySerializeError(JSC::JSGlobalObject&);
77 
78 private:
79     ErrorEvent(const String& message, const String& fileName, unsigned lineNumber, unsigned columnNumber, JSC::Strong<JSC::Unknown> error);
80     ErrorEvent(const AtomString&, const Init&, IsTrusted);
81 
82     bool isErrorEvent() const override;
83 
84     String m_message;
85     String m_fileName;
86     unsigned m_lineNumber;
87     unsigned m_columnNumber;
88     JSValueInWrappedObject m_error;
89     RefPtr<SerializedScriptValue> m_serializedError;
90     bool m_triedToSerialize { false };
91 };
92 
93 } // namespace WebCore
94 
95 SPECIALIZE_TYPE_TRAITS_EVENT(ErrorEvent)
