32 // It represents that the string impl is uniqued in some ways.
33 // When the given 2 string impls are both uniqued string impls, we can compare it just using pointer comparison.
34 class UniquedStringImpl : public StringImpl {
35 private:
36     UniquedStringImpl() = delete;
37 protected:
38     UniquedStringImpl(CreateSymbolTag, const LChar* characters, unsigned length) : StringImpl(CreateSymbol, characters, length) { }
39     UniquedStringImpl(CreateSymbolTag, const UChar* characters, unsigned length) : StringImpl(CreateSymbol, characters, length) { }
40     UniquedStringImpl(CreateSymbolTag) : StringImpl(CreateSymbol) { }
41 };
42 
43 #if !ASSERT_DISABLED
44 // UniquedStringImpls created from StaticStringImpl will ASSERT
45 // in the generic ValueCheck<T>::checkConsistency
46 // as they are not allocated by fastMalloc.
47 // We don't currently have any way to detect that case
48 // so we ignore the consistency check for all UniquedStringImpls*.
49 template<> struct
50 ValueCheck<UniquedStringImpl*> {
51     static void checkConsistency(const UniquedStringImpl*) { }
52 };
53 
54 template<> struct
55 ValueCheck<const UniquedStringImpl*> {
56     static void checkConsistency(const UniquedStringImpl*) { }
57 };
58 #endif
59 
60 } // namespace WTF
61 
62 using WTF::UniquedStringImpl;
