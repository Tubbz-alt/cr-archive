 35 /*
 36 | <--- one message for send / didReceiveData ---> |
 37 +--------------+----------------------------------+--------------
 38 |    size      |               data               | (next message)
 39 |  4byte (NBO) |          variable length         |
 40 +--------------+----------------------------------+--------------
 41                | <------------ size ------------> |
 42 */
 43 
 44 MessageParser::MessageParser(Function<void(Vector<uint8_t>&&)>&& listener)
 45     : m_listener(WTFMove(listener))
 46 {
 47 }
 48 
 49 Vector<uint8_t> MessageParser::createMessage(const uint8_t* data, size_t size)
 50 {
 51     if (!data || !size || size > UINT_MAX)
 52         return Vector<uint8_t>();
 53 
 54     auto messageBuffer = Vector<uint8_t>(size + sizeof(uint32_t));
 55     uint32_t uintSize = static_cast<uint32_t>(size);
 56     uint32_t nboSize = htonl(uintSize);
 57     memcpy(&messageBuffer[0], &nboSize, sizeof(uint32_t));
 58     memcpy(&messageBuffer[sizeof(uint32_t)], data, uintSize);
 59     return messageBuffer;
 60 }
 61 
 62 void MessageParser::pushReceivedData(const uint8_t* data, size_t size)
 63 {
 64     if (!data || !size || !m_listener)
 65         return;
 66 
 67     m_buffer.reserveCapacity(m_buffer.size() + size);
 68     m_buffer.append(data, size);
 69 
 70     if (!parse())
 71         clearReceivedData();
 72 }
 73 
 74 void MessageParser::clearReceivedData()
 75 {
 76     m_buffer.clear();
 77 }
 78 
 79 bool MessageParser::parse()
 80 {
 81     while (!m_buffer.isEmpty()) {
 82         if (m_buffer.size() < sizeof(uint32_t)) {
 83             // Wait for more data.
 84             return true;
 85         }
 86 
 87         uint32_t dataSize = 0;
 88         memcpy(&dataSize, &m_buffer[0], sizeof(uint32_t));
 89         dataSize = ntohl(dataSize);
 90         if (!dataSize) {
 91             LOG_ERROR("Message Parser received an invalid message size");
 92             return false;
 93         }
 94 
 95         size_t messageSize = (sizeof(uint32_t) + dataSize);
 96         if (m_buffer.size() < messageSize) {
 97             // Wait for more data.
 98             return true;
 99         }
100 
101         // FIXME: This should avoid re-creating a new data Vector.
102         auto dataBuffer = Vector<uint8_t>(dataSize);
103         memcpy(&dataBuffer[0], &m_buffer[sizeof(uint32_t)], dataSize);
104 
105         m_listener(WTFMove(dataBuffer));
106 
107         m_buffer.remove(0, messageSize);
108     }
109 
110     return true;
111 }
