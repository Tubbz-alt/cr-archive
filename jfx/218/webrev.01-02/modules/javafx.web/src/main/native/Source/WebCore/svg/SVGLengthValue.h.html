 22 #pragma once
 23 
 24 #include "ExceptionOr.h"
 25 #include "SVGParsingError.h"
 26 #include "SVGPropertyTraits.h"
 27 
 28 namespace WebCore {
 29 
 30 class CSSPrimitiveValue;
 31 class SVGLengthContext;
 32 
 33 enum class SVGLengthType : uint8_t {
 34     Unknown = 0,
 35     Number,
 36     Percentage,
 37     Ems,
 38     Exs,
 39     Pixels,
 40     Centimeters,
 41     Millimeters,
 42     Inches,
 43     Points,
 44     Picas
 45 };
 46 
 47 enum class SVGLengthMode : uint8_t {
 48     Width,
 49     Height,
 50     Other
 51 };
 52 
 53 enum class SVGLengthNegativeValuesMode : uint8_t {
 54     Allow,
 55     Forbid
 56 };
 57 
 58 class SVGLengthValue {
 59     WTF_MAKE_FAST_ALLOCATED;
 60 public:
 61     SVGLengthValue(SVGLengthMode = SVGLengthMode::Other, const String& valueAsString = String());
 62     SVGLengthValue(float valueInSpecifiedUnits, SVGLengthType, SVGLengthMode = SVGLengthMode::Other);
 63     SVGLengthValue(const SVGLengthContext&, float, SVGLengthType = SVGLengthType::Number, SVGLengthMode = SVGLengthMode::Other);
 64 
 65     static SVGLengthValue construct(SVGLengthMode, const String&, SVGParsingError&, SVGLengthNegativeValuesMode = SVGLengthNegativeValuesMode::Allow);
 66     static SVGLengthValue blend(const SVGLengthValue& from, const SVGLengthValue& to, float progress);
 67 
 68     static SVGLengthValue fromCSSPrimitiveValue(const CSSPrimitiveValue&);
 69     static Ref<CSSPrimitiveValue> toCSSPrimitiveValue(const SVGLengthValue&);
 70 
 71     SVGLengthType lengthType() const { return m_lengthType; }
 72     SVGLengthMode lengthMode() const { return m_lengthMode; }
 73 
 74     bool isZero() const { return !m_valueInSpecifiedUnits;  }
 75     bool isRelative() const { return m_lengthType == SVGLengthType::Percentage || m_lengthType == SVGLengthType::Ems || m_lengthType == SVGLengthType::Exs; }
 76 
 77     float value(const SVGLengthContext&) const;
 78     float valueAsPercentage() const { return m_lengthType == SVGLengthType::Percentage ? m_valueInSpecifiedUnits / 100 : m_valueInSpecifiedUnits; }
 79     float valueInSpecifiedUnits() const { return m_valueInSpecifiedUnits; }
 80 
 81     String valueAsString() const;
 82     ExceptionOr<float> valueForBindings(const SVGLengthContext&) const;
 83 
 84     void setValueInSpecifiedUnits(float value) { m_valueInSpecifiedUnits = value; }
 85     ExceptionOr<void> setValue(const SVGLengthContext&, float);
 86     ExceptionOr<void> setValue(const SVGLengthContext&, float, SVGLengthType, SVGLengthMode);
 87 
 88     ExceptionOr<void> setValueAsString(const String&);
 89     ExceptionOr<void> setValueAsString(const String&, SVGLengthMode);
 90 
 91     ExceptionOr<void> convertToSpecifiedUnits(const SVGLengthContext&, SVGLengthType);
 92 
 93 private:
 94     float m_valueInSpecifiedUnits { 0 };
 95     SVGLengthType m_lengthType { SVGLengthType::Number };
 96     SVGLengthMode m_lengthMode { SVGLengthMode::Other };
 97 };
 98 
 99 inline bool operator==(const SVGLengthValue& a, const SVGLengthValue& b)
100 {
101     return a.valueInSpecifiedUnits() == b.valueInSpecifiedUnits() && a.lengthType() == b.lengthType() && a.lengthMode() == b.lengthMode();
102 }
103 
104 inline bool operator!=(const SVGLengthValue& a, const SVGLengthValue& b)
105 {
106     return a.valueInSpecifiedUnits() != b.valueInSpecifiedUnits() || a.lengthType() != b.lengthType() || a.lengthMode() != b.lengthMode();
107 }
108 
109 WTF::TextStream& operator<<(WTF::TextStream&, const SVGLengthValue&);

111 } // namespace WebCore
