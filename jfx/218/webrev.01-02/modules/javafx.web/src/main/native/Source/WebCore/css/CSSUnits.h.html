 87     CSS_VALUE_ID = 118,
 88 
 89     // This value is used to handle quirky margins in reflow roots (body, td, and th) like WinIE.
 90     // The basic idea is that a stylesheet can use the value __qem (for quirky em) instead of em.
 91     // When the quirky value is used, if you're in quirks mode, the margin will collapse away
 92     // inside a table cell. This quirk is specified in the HTML spec but our impl is different.
 93     CSS_QUIRKY_EMS = 120
 94 
 95     // Note that CSSValue allocates 7 bits for m_primitiveUnitType, so there can be no value here > 127.
 96 };
