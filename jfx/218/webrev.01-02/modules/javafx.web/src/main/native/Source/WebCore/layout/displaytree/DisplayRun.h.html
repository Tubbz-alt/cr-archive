 35 namespace WebCore {
 36 
 37 class CachedImage;
 38 
 39 namespace Display {
 40 
 41 struct Run {
 42     WTF_MAKE_STRUCT_FAST_ALLOCATED;
 43     struct TextContext {
 44         WTF_MAKE_STRUCT_FAST_ALLOCATED;
 45     public:
 46         struct ExpansionContext;
 47         TextContext(unsigned position, unsigned length, const String&);
 48 
 49         unsigned start() const { return m_start; }
 50         unsigned end() const { return start() + length(); }
 51         unsigned length() const { return m_length; }
 52         StringView content() const { return StringView(m_contentString).substring(m_start, m_length); }
 53 
 54         struct ExpansionContext {
 55             ExpansionBehavior behavior { DefaultExpansion };
 56             InlineLayoutUnit horizontalExpansion { 0 };
 57         };
 58         void setExpansion(ExpansionContext expansionContext) { m_expansionContext = expansionContext; }
 59         Optional<ExpansionContext> expansion() const { return m_expansionContext; }
 60 
 61         bool needsHyphen() const { return m_needsHyphen; }
 62 
 63         void expand(unsigned delta) { m_length += delta; }
 64         void setNeedsHyphen() { m_needsHyphen = true; }
 65 
 66     private:
 67         unsigned m_start { 0 };
 68         unsigned m_length { 0 };
 69         bool m_needsHyphen { false };
 70         String m_contentString;
 71         Optional<ExpansionContext> m_expansionContext;
 72     };
 73 
 74     Run(size_t lineIndex, const Layout::Box&, const InlineRect&, const InlineRect& inkOverflow, Optional<TextContext> = WTF::nullopt);
 75 
 76     size_t lineIndex() const { return m_lineIndex; }
 77 
 78     const InlineRect& rect() const { return m_rect; }
 79     const InlineRect& inkOverflow() const { return m_inkOverflow; }
 80 
 81     InlineLayoutPoint topLeft() const { return m_rect.topLeft(); }
 82     InlineLayoutUnit left() const { return m_rect.left(); }
 83     InlineLayoutUnit right() const { return m_rect.right(); }
 84     InlineLayoutUnit top() const { return m_rect.top(); }
 85     InlineLayoutUnit bottom() const { return m_rect.bottom(); }
 86 
 87     InlineLayoutUnit width() const { return m_rect.width(); }
 88     InlineLayoutUnit height() const { return m_rect.height(); }
 89 
 90     void setWidth(InlineLayoutUnit width) { m_rect.setWidth(width); }
 91     void setTop(InlineLayoutUnit top) { m_rect.setTop(top); }
 92     void setlLeft(InlineLayoutUnit left) { m_rect.setLeft(left); }
 93     void moveVertically(InlineLayoutUnit delta) { m_rect.moveVertically(delta); }
 94     void moveHorizontally(InlineLayoutUnit delta) { m_rect.moveHorizontally(delta); }
 95     void expandVertically(InlineLayoutUnit delta) { m_rect.expandVertically(delta); }
 96     void expandHorizontally(InlineLayoutUnit delta) { m_rect.expandHorizontally(delta); }
 97 
 98     void setTextContext(const TextContext&& textContext) { m_textContext.emplace(textContext); }
 99     const Optional<TextContext>& textContext() const { return m_textContext; }
100     Optional<TextContext>& textContext() { return m_textContext; }
101 
102     void setImage(CachedImage& image) { m_cachedImage = &image; }
103     CachedImage* image() const { return m_cachedImage; }
104 
105     bool isLineBreak() const { return layoutBox().isLineBreakBox() || (textContext() && textContext()->content() == "\n" && style().preserveNewline()); }
106 
107     const Layout::Box& layoutBox() const { return *m_layoutBox; }
108     const RenderStyle& style() const { return m_layoutBox->style(); }
109 
110 private:
111     // FIXME: Find out the Display::Run <-> paint style setup.
112     const size_t m_lineIndex;
113     WeakPtr<const Layout::Box> m_layoutBox;
114     CachedImage* m_cachedImage { nullptr };
115     InlineRect m_rect;
116     InlineRect m_inkOverflow;
117     Optional<TextContext> m_textContext;
118 };
119 
120 inline Run::Run(size_t lineIndex, const Layout::Box& layoutBox, const InlineRect& rect, const InlineRect& inkOverflow, Optional<TextContext> textContext)
121     : m_lineIndex(lineIndex)
122     , m_layoutBox(makeWeakPtr(layoutBox))
123     , m_rect(rect)
124     , m_inkOverflow(inkOverflow)
125     , m_textContext(textContext)
126 {
127 }
128 
129 inline Run::TextContext::TextContext(unsigned start, unsigned length, const String& contentString)
130     : m_start(start)
131     , m_length(length)
132     , m_contentString(contentString)
133 {
134 }
135 
136 }
137 }
