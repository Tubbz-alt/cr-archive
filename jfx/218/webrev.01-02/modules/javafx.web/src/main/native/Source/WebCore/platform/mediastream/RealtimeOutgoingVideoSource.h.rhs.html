 30 
 31 #if USE(LIBWEBRTC)
 32 
 33 #include "LibWebRTCMacros.h"
 34 #include "MediaStreamTrackPrivate.h"
 35 #include <Timer.h>
 36 
 37 ALLOW_UNUSED_PARAMETERS_BEGIN
 38 
 39 #include <webrtc/api/media_stream_interface.h>
 40 #include <webrtc/common_video/include/i420_buffer_pool.h>
 41 
 42 ALLOW_UNUSED_PARAMETERS_END
 43 
 44 #include <wtf/LoggerHelper.h>
 45 #include <wtf/Optional.h>
 46 #include <wtf/ThreadSafeRefCounted.h>
 47 
 48 namespace WebCore {
 49 
 50 class RealtimeOutgoingVideoSource
 51     : public ThreadSafeRefCounted<RealtimeOutgoingVideoSource, WTF::DestructionThread::Main>
 52     , public webrtc::VideoTrackSourceInterface
 53     , private MediaStreamTrackPrivate::Observer
 54 #if !RELEASE_LOG_DISABLED
 55     , private LoggerHelper
 56 #endif
 57 {
 58 public:
 59     static Ref<RealtimeOutgoingVideoSource> create(Ref<MediaStreamTrackPrivate>&& videoSource);
 60     ~RealtimeOutgoingVideoSource();
 61 
 62     void start() { observeSource(); }
 63     void stop();
 64     void setSource(Ref<MediaStreamTrackPrivate>&&);
 65     MediaStreamTrackPrivate& source() const { return m_videoSource.get(); }
 66 
 67     void AddRef() const final { ref(); }
 68     rtc::RefCountReleaseStatus Release() const final
 69     {
 70         deref();
 71         return rtc::RefCountReleaseStatus::kOtherRefsRemained;
 72     }
 73 
 74     void setApplyRotation(bool shouldApplyRotation) { m_shouldApplyRotation = shouldApplyRotation; }
 75 
 76 protected:
 77     explicit RealtimeOutgoingVideoSource(Ref<MediaStreamTrackPrivate>&&);
 78 
 79     void sendFrame(rtc::scoped_refptr<webrtc::VideoFrameBuffer>&&);
 80     bool isSilenced() const { return m_muted || !m_enabled; }
 81 
 82     virtual rtc::scoped_refptr<webrtc::VideoFrameBuffer> createBlackFrame(size_t width, size_t height) = 0;
 83 
 84     bool m_shouldApplyRotation { false };
 85     webrtc::VideoRotation m_currentRotation { webrtc::kVideoRotation_0 };
 86 
 87 #if !RELEASE_LOG_DISABLED
 88     // LoggerHelper API
 89     const Logger& logger() const final { return m_logger.get(); }
 90     const void* logIdentifier() const final { return m_logIdentifier; }
 91     const char* logClassName() const final { return "RealtimeOutgoingVideoSource"; }
 92     WTFLogChannel& logChannel() const final;
 93 #endif
 94 
 95 private:
 96     void sendBlackFramesIfNeeded();
 97     void sendOneBlackFrame();
 98     void initializeFromSource();
 99     void updateBlackFramesSending();
100 
101     void observeSource();
102     void unobserveSource();
103 
104     // Notifier API
105     void RegisterObserver(webrtc::ObserverInterface*) final { }
106     void UnregisterObserver(webrtc::ObserverInterface*) final { }
107 
108     // VideoTrackSourceInterface API
109     bool is_screencast() const final { return false; }
110     absl::optional<bool> needs_denoising() const final { return absl::optional<bool>(); }
111     bool GetStats(Stats*) final { return false; };
112 
113     // MediaSourceInterface API
114     SourceState state() const final { return SourceState(); }
115     bool remote() const final { return true; }
116 
117     // rtc::VideoSourceInterface<webrtc::VideoFrame> API
118     void AddOrUpdateSink(rtc::VideoSinkInterface<webrtc::VideoFrame>*, const rtc::VideoSinkWants&) final;
119     void RemoveSink(rtc::VideoSinkInterface<webrtc::VideoFrame>*) final;
120 
121     void sourceMutedChanged();
122     void sourceEnabledChanged();
123 
124     // MediaStreamTrackPrivate::Observer API
125     void trackMutedChanged(MediaStreamTrackPrivate&) final { sourceMutedChanged(); }
126     void trackEnabledChanged(MediaStreamTrackPrivate&) final { sourceEnabledChanged(); }
127     void trackSettingsChanged(MediaStreamTrackPrivate&) final { initializeFromSource(); }
128     void sampleBufferUpdated(MediaStreamTrackPrivate&, MediaSample&) override { }
129     void trackEnded(MediaStreamTrackPrivate&) final { }
130 
131     Ref<MediaStreamTrackPrivate> m_videoSource;

132     Timer m_blackFrameTimer;
133     rtc::scoped_refptr<webrtc::VideoFrameBuffer> m_blackFrame;
134 
135     mutable RecursiveLock m_sinksLock;
136     HashSet<rtc::VideoSinkInterface<webrtc::VideoFrame>*> m_sinks;
137 
138     bool m_enabled { true };
139     bool m_muted { false };
140     uint32_t m_width { 0 };
141     uint32_t m_height { 0 };
142 
143 #if !RELEASE_LOG_DISABLED
144     Ref<const Logger> m_logger;
145     const void* m_logIdentifier;
146     MonotonicTime m_lastFrameLogTime;
147     unsigned m_frameCount { 0 };
148 #endif
149 };
150 
151 } // namespace WebCore
152 
153 #endif // USE(LIBWEBRTC)
