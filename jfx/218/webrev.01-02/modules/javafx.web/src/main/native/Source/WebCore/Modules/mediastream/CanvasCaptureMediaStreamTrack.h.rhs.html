 25 #pragma once
 26 
 27 #if ENABLE(MEDIA_STREAM)
 28 
 29 #include "CanvasBase.h"
 30 #include "MediaStreamTrack.h"
 31 #include "Timer.h"
 32 #include <wtf/TypeCasts.h>
 33 
 34 namespace WebCore {
 35 
 36 class Document;
 37 class HTMLCanvasElement;
 38 class Image;
 39 
 40 class CanvasCaptureMediaStreamTrack final : public MediaStreamTrack {
 41     WTF_MAKE_ISO_ALLOCATED(CanvasCaptureMediaStreamTrack);
 42 public:
 43     static Ref<CanvasCaptureMediaStreamTrack> create(Document&, Ref<HTMLCanvasElement>&&, Optional<double>&& frameRequestRate);
 44 
 45     HTMLCanvasElement& canvas() { return m_canvas.get(); }
 46     void requestFrame() { static_cast<Source&>(source()).requestFrame(); }
 47 
 48     RefPtr<MediaStreamTrack> clone() final;
 49 
 50 private:
 51     const char* activeDOMObjectName() const override;
 52 
 53     class Source final : public RealtimeMediaSource, private CanvasObserver {
 54     public:
 55         static Ref<Source> create(HTMLCanvasElement&, Optional<double>&& frameRequestRate);
 56 
 57         void requestFrame() { m_shouldEmitFrame = true; }
 58         Optional<double> frameRequestRate() const { return m_frameRequestRate; }
 59 
 60     private:
 61         Source(HTMLCanvasElement&, Optional<double>&&);
 62 
 63         // CanvasObserver API
 64         void canvasChanged(CanvasBase&, const FloatRect&) final;
 65         void canvasResized(CanvasBase&) final;
 66         void canvasDestroyed(CanvasBase&) final;
 67 
 68         // RealtimeMediaSource API
 69         void startProducingData() final;
 70         void stopProducingData()  final;
 71         const RealtimeMediaSourceCapabilities& capabilities() final { return RealtimeMediaSourceCapabilities::emptyCapabilities(); }
 72         const RealtimeMediaSourceSettings& settings() final;
 73         void settingsDidChange(OptionSet<RealtimeMediaSourceSettings::Flag>) final;
 74 
 75         void captureCanvas();
 76         void requestFrameTimerFired();
 77 
 78         bool m_shouldEmitFrame { true };
 79         Optional<double> m_frameRequestRate;
 80         Timer m_requestFrameTimer;
 81         Timer m_canvasChangedTimer;
 82         Optional<RealtimeMediaSourceSettings> m_currentSettings;
 83         HTMLCanvasElement* m_canvas;
 84         RefPtr<Image> m_currentImage;
 85     };
 86 
 87     CanvasCaptureMediaStreamTrack(Document&, Ref<HTMLCanvasElement>&&, Ref<Source>&&);
 88     CanvasCaptureMediaStreamTrack(Document&, Ref<HTMLCanvasElement>&&, Ref<MediaStreamTrackPrivate>&&);
 89 
 90     bool isCanvas() const final { return true; }
 91 
 92     Ref<HTMLCanvasElement> m_canvas;
 93 };
 94 
 95 }
 96 
 97 SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::CanvasCaptureMediaStreamTrack)
 98 static bool isType(const WebCore::MediaStreamTrack& track) { return track.isCanvas(); }
 99 SPECIALIZE_TYPE_TRAITS_END()
