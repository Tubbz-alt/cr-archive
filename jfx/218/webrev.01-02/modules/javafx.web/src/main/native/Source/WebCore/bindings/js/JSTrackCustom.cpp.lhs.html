40 JSC::JSValue toJS(JSC::ExecState* state, JSDOMGlobalObject* globalObject, TrackBase& track)
41 {
42     switch (track.type()) {
43     case TrackBase::BaseTrack:
44         // This should never happen.
45         ASSERT_NOT_REACHED();
46         break;
47 
48     case TrackBase::AudioTrack:
49         return wrap(state, globalObject, downcast<AudioTrack>(track));
50     case TrackBase::VideoTrack:
51         return wrap(state, globalObject, downcast<VideoTrack>(track));
52     case TrackBase::TextTrack:
53         return wrap(state, globalObject, downcast<TextTrack>(track));
54     }
55 
56     return jsNull();
57 }
