33 namespace WebCore {

35 Ref<WebGLUniformLocation> WebGLUniformLocation::create(WebGLProgram* program, GCGLint location, GCGLenum type)
36 {
37     return adoptRef(*new WebGLUniformLocation(program, location, type));
38 }
39 
40 WebGLUniformLocation::WebGLUniformLocation(WebGLProgram* program, GCGLint location, GCGLenum type)
41     : m_program(program)
42     , m_location(location)
43     , m_type(type)
44 {
45     ASSERT(m_program);
46     m_linkCount = m_program->getLinkCount();
47 }
48 
49 WebGLProgram* WebGLUniformLocation::program() const
50 {
51     // If the program has been linked again, then this UniformLocation is no
52     // longer valid.
53     if (m_program->getLinkCount() != m_linkCount)
54         return 0;
55     return m_program.get();
56 }
57 
58 GCGLint WebGLUniformLocation::location() const
59 {
60     // If the program has been linked again, then this UniformLocation is no
61     // longer valid.
62     ASSERT(m_program->getLinkCount() == m_linkCount);
63     return m_location;
64 }
65 
66 GCGLenum WebGLUniformLocation::type() const
67 {
68     // If the program has been linked again, then this UniformLocation is no
69     // longer valid.
70     ASSERT(m_program->getLinkCount() == m_linkCount);
71     return m_type;
72 }
73 
74 }
