40 class GetCallerStrictnessFunctor {
41 public:
42     GetCallerStrictnessFunctor()
43         : m_iterations(0)
44         , m_callerIsStrict(false)
45     {
46     }
47 
48     StackVisitor::Status operator()(StackVisitor& visitor) const
49     {
50         ++m_iterations;
51         if (m_iterations < 2)
52             return StackVisitor::Continue;
53 
54         CodeBlock* codeBlock = visitor->codeBlock();
55         m_callerIsStrict = codeBlock && codeBlock->isStrictMode();
56         return StackVisitor::Done;
57     }
58 
59     bool callerIsStrict() const { return m_callerIsStrict; }
60 
61 private:
62     mutable int m_iterations;
63     mutable bool m_callerIsStrict;
64 };
65 
66 static bool callerIsStrict(ExecState* exec)
67 {
68     GetCallerStrictnessFunctor iter;
69     exec->iterate(iter);
70     return iter.callerIsStrict();
71 }
72 
73 namespace NullSetterFunctionInternal {
74 static EncodedJSValue JSC_HOST_CALL callReturnUndefined(ExecState* exec)
75 {
76     VM& vm = exec->vm();
77     auto scope = DECLARE_THROW_SCOPE(vm);
78 
79     if (callerIsStrict(exec))
80         return JSValue::encode(throwTypeError(exec, scope, "Setting a property that has only a getter"_s));
81     return JSValue::encode(jsUndefined());
82 }
83 }

86     : Base(vm, structure, NullSetterFunctionInternal::callReturnUndefined, nullptr)
