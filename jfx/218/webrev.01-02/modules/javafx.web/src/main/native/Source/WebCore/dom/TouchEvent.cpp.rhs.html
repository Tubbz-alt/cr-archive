36 namespace WebCore {
37 
38 WTF_MAKE_ISO_ALLOCATED_IMPL(TouchEvent);
39 
40 TouchEvent::TouchEvent() = default;
41 
42 TouchEvent::TouchEvent(TouchList* touches, TouchList* targetTouches, TouchList* changedTouches, const AtomString& type,
43     RefPtr<WindowProxy>&& view, const IntPoint& globalLocation, OptionSet<Modifier> modifiers)
44     : MouseRelatedEvent(type, IsCancelable::Yes, MonotonicTime::now(), WTFMove(view), globalLocation, modifiers)
45     , m_touches(touches)
46     , m_targetTouches(targetTouches)
47     , m_changedTouches(changedTouches)
48 {
49 }
50 
51 TouchEvent::TouchEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
52     : MouseRelatedEvent(type, initializer, isTrusted)
53     , m_touches(initializer.touches ? initializer.touches : TouchList::create())
54     , m_targetTouches(initializer.targetTouches ? initializer.targetTouches : TouchList::create())
55     , m_changedTouches(initializer.changedTouches ? initializer.changedTouches : TouchList::create())
56 {
57 }
58 
59 TouchEvent::~TouchEvent() = default;
60 
61 void TouchEvent::initTouchEvent(TouchList* touches, TouchList* targetTouches,
62         TouchList* changedTouches, const AtomString& type,
63         RefPtr<WindowProxy>&& view, int screenX, int screenY, int clientX, int clientY,
64         bool ctrlKey, bool altKey, bool shiftKey, bool metaKey)
65 {
66     if (isBeingDispatched())
67         return;
68 
69     initUIEvent(type, true, true, WTFMove(view), 0);
70 
71     m_touches = touches;
72     m_targetTouches = targetTouches;
73     m_changedTouches = changedTouches;
74     m_screenLocation = IntPoint(screenX, screenY);
75     setModifierKeys(ctrlKey, altKey, shiftKey, metaKey);
76     initCoordinates(IntPoint(clientX, clientY));
77 }
78 
79 EventInterface TouchEvent::eventInterface() const
80 {
81     return TouchEventInterfaceType;
82 }
83 
84 bool TouchEvent::isTouchEvent() const
85 {
86     return true;
87 }
