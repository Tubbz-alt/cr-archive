 39 static RefPtr<UserGestureToken>& currentToken()
 40 {
 41     ASSERT(isMainThread());
 42     static NeverDestroyed<RefPtr<UserGestureToken>> token;
 43     return token;
 44 }
 45 
 46 UserGestureToken::~UserGestureToken()
 47 {
 48     for (auto& observer : m_destructionObservers)
 49         observer(*this);
 50 }
 51 
 52 UserGestureIndicator::UserGestureIndicator(Optional<ProcessingUserGestureState> state, Document* document, UserGestureType gestureType, ProcessInteractionStyle processInteractionStyle)
 53     : m_previousToken { currentToken() }
 54 {
 55     ASSERT(isMainThread());
 56 
 57     if (state)
 58         currentToken() = UserGestureToken::create(state.value(), gestureType);
 59 
 60     if (document && currentToken()->processingUserGesture() && state) {
 61         document->updateLastHandledUserGestureTimestamp(currentToken()->startTime());
 62         if (processInteractionStyle == ProcessInteractionStyle::Immediate)
 63             ResourceLoadObserver::shared().logUserInteractionWithReducedTimeResolution(document->topDocument());
 64         document->topDocument().setUserDidInteractWithPage(true);
 65         if (auto* frame = document->frame()) {
 66             if (!frame->hasHadUserInteraction()) {
 67                 for (; frame; frame = frame->tree().parent())
 68                     frame->setHasHadUserInteraction();
 69             }
 70         }
 71 
 72         if (auto* window = document->domWindow())
 73             window->notifyActivated(currentToken()->startTime());
 74     }
 75 }
 76 
 77 UserGestureIndicator::UserGestureIndicator(RefPtr<UserGestureToken> token, UserGestureToken::GestureScope scope)
 78 {
 79     // Silently ignore UserGestureIndicators on non main threads.
 80     if (!isMainThread())
 81         return;
 82 
 83     // It is only safe to use currentToken() on the main thread.
 84     m_previousToken = currentToken();
 85 
 86     if (token) {
 87         token->setScope(scope);
 88         currentToken() = token;
 89     }
 90 }
 91 
 92 UserGestureIndicator::~UserGestureIndicator()
 93 {
 94     if (!isMainThread())
 95         return;
 96 
 97     if (auto token = currentToken()) {
 98         token->resetDOMPasteAccess();
 99         token->resetScope();
100     }
101 
102     currentToken() = m_previousToken;
103 }
104 
105 RefPtr<UserGestureToken> UserGestureIndicator::currentUserGesture()
106 {
107     if (!isMainThread())
108         return nullptr;
109 
110     return currentToken();
111 }
112 
113 bool UserGestureIndicator::processingUserGesture()
114 {
115     if (!isMainThread())
116         return false;
117 
118     return currentToken() ? currentToken()->processingUserGesture() : false;
119 }
120 
121 bool UserGestureIndicator::processingUserGestureForMedia()
122 {
123     if (!isMainThread())
124         return false;
125 
126     return currentToken() ? currentToken()->processingUserGestureForMedia() : false;
127 }
