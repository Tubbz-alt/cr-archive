 71 JSObject* Instance::createRuntimeObject(ExecState* exec)
 72 {
 73     ASSERT(m_rootObject);
 74     ASSERT(m_rootObject->isValid());
 75     if (RuntimeObject* existingObject = m_runtimeObject.get())
 76         return existingObject;
 77 
 78     JSLockHolder lock(exec);
 79     RuntimeObject* newObject = newRuntimeObject(exec);
 80     m_runtimeObject = JSC::Weak<RuntimeObject>(newObject);
 81     m_rootObject->addRuntimeObject(exec->vm(), newObject);
 82     return newObject;
 83 }
