 20 #pragma once
 21 
 22 #include "SVGLengthValue.h"
 23 #include "SVGPreserveAspectRatioValue.h"
 24 #include "SVGUnitTypes.h"
 25 
 26 namespace WebCore {
 27 
 28 class SVGPatternElement;
 29 
 30 struct PatternAttributes {
 31     PatternAttributes()
 32         : m_x()
 33         , m_y()
 34         , m_width()
 35         , m_height()
 36         , m_viewBox()
 37         , m_preserveAspectRatio()
 38         , m_patternUnits(SVGUnitTypes::SVG_UNIT_TYPE_OBJECTBOUNDINGBOX)
 39         , m_patternContentUnits(SVGUnitTypes::SVG_UNIT_TYPE_USERSPACEONUSE)
 40         , m_patternContentElement(nullptr)
 41         , m_xSet(false)
 42         , m_ySet(false)
 43         , m_widthSet(false)
 44         , m_heightSet(false)
 45         , m_viewBoxSet(false)
 46         , m_preserveAspectRatioSet(false)
 47         , m_patternUnitsSet(false)
 48         , m_patternContentUnitsSet(false)
 49         , m_patternTransformSet(false)
 50         , m_patternContentElementSet(false)
 51     {
 52     }
 53 
 54     SVGLengthValue x() const { return m_x; }
 55     SVGLengthValue y() const { return m_y; }
 56     SVGLengthValue width() const { return m_width; }
 57     SVGLengthValue height() const { return m_height; }
 58     FloatRect viewBox() const { return m_viewBox; }
 59     SVGPreserveAspectRatioValue preserveAspectRatio() const { return m_preserveAspectRatio; }
 60     SVGUnitTypes::SVGUnitType patternUnits() const { return m_patternUnits; }
 61     SVGUnitTypes::SVGUnitType patternContentUnits() const { return m_patternContentUnits; }
 62     AffineTransform patternTransform() const { return m_patternTransform; }
 63     const SVGPatternElement* patternContentElement() const { return m_patternContentElement; }
 64 
 65     void setX(SVGLengthValue value)
 66     {
 67         m_x = value;
 68         m_xSet = true;
 69     }
 70 
 71     void setY(SVGLengthValue value)
 72     {
 73         m_y = value;
 74         m_ySet = true;
 75     }
 76 
 77     void setWidth(SVGLengthValue value)
 78     {
 79         m_width = value;
 80         m_widthSet = true;
 81     }
 82 
 83     void setHeight(SVGLengthValue value)
 84     {
 85         m_height = value;
 86         m_heightSet = true;
 87     }
 88 
 89     void setViewBox(const FloatRect& value)
 90     {
 91         m_viewBox = value;
 92         m_viewBoxSet = true;
 93     }
 94 
 95     void setPreserveAspectRatio(SVGPreserveAspectRatioValue value)
 96     {
 97         m_preserveAspectRatio = value;
 98         m_preserveAspectRatioSet = true;
 99     }
100 
101     void setPatternUnits(SVGUnitTypes::SVGUnitType value)
102     {
103         m_patternUnits = value;
104         m_patternUnitsSet = true;
105     }
106 
107     void setPatternContentUnits(SVGUnitTypes::SVGUnitType value)
108     {
109         m_patternContentUnits = value;
110         m_patternContentUnitsSet = true;
111     }
112 
113     void setPatternTransform(const AffineTransform& value)
114     {
115         m_patternTransform = value;
116         m_patternTransformSet = true;
117     }
118 
119     void setPatternContentElement(const SVGPatternElement* value)
120     {
121         m_patternContentElement = value;
122         m_patternContentElementSet = true;
123     }
124 
125     bool hasX() const { return m_xSet; }
126     bool hasY() const { return m_ySet; }
127     bool hasWidth() const { return m_widthSet; }
128     bool hasHeight() const { return m_heightSet; }
129     bool hasViewBox() const { return m_viewBoxSet; }
130     bool hasPreserveAspectRatio() const { return m_preserveAspectRatioSet; }
131     bool hasPatternUnits() const { return m_patternUnitsSet; }
132     bool hasPatternContentUnits() const { return m_patternContentUnitsSet; }
133     bool hasPatternTransform() const { return m_patternTransformSet; }
134     bool hasPatternContentElement() const { return m_patternContentElementSet; }
135 
136 private:
137     // Properties
138     SVGLengthValue m_x;
139     SVGLengthValue m_y;
140     SVGLengthValue m_width;
141     SVGLengthValue m_height;
142     FloatRect m_viewBox;
143     SVGPreserveAspectRatioValue m_preserveAspectRatio;
144     SVGUnitTypes::SVGUnitType m_patternUnits;
145     SVGUnitTypes::SVGUnitType m_patternContentUnits;
146     AffineTransform m_patternTransform;
147     const SVGPatternElement* m_patternContentElement;
148 
149     // Property states
150     bool m_xSet : 1;
151     bool m_ySet : 1;
152     bool m_widthSet : 1;
153     bool m_heightSet : 1;
154     bool m_viewBoxSet : 1;
155     bool m_preserveAspectRatioSet : 1;
156     bool m_patternUnitsSet : 1;
157     bool m_patternContentUnitsSet : 1;
158     bool m_patternTransformSet : 1;
159     bool m_patternContentElementSet : 1;
160 };
161 
162 } // namespace WebCore
