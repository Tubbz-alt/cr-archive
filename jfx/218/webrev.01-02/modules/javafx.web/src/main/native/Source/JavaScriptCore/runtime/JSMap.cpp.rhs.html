41 JSMap* JSMap::clone(JSGlobalObject* globalObject, VM& vm, Structure* structure)
42 {
43     JSMap* instance = new (NotNull, allocateCell<JSMap>(vm.heap)) JSMap(vm, structure);
44     instance->finishCreation(globalObject, vm, this);
45     return instance;
46 }

48 bool JSMap::isIteratorProtocolFastAndNonObservable()
49 {
50     JSGlobalObject* globalObject = this->globalObject();
51     if (!globalObject->isMapPrototypeIteratorProtocolFastAndNonObservable())
52         return false;
53 
54     VM& vm = globalObject->vm();
55     Structure* structure = this->structure(vm);
56     // This is the fast case. Many maps will be an original map.
57     if (structure == globalObject->mapStructure())
58         return true;
59 
60     if (getPrototypeDirect(vm) != globalObject->mapPrototype())
61         return false;
62 
63     if (getDirectOffset(vm, vm.propertyNames->iteratorSymbol) != invalidOffset)
64         return false;
65 
66     return true;
67 }
68 
69 bool JSMap::canCloneFastAndNonObservable(Structure* structure)
70 {
71     auto setFastAndNonObservable = [&] (Structure* structure) {
72         JSGlobalObject* globalObject = structure->globalObject();
73         if (!globalObject->isMapPrototypeSetFastAndNonObservable())
74             return false;
75 
76         if (structure->hasPolyProto())
77             return false;
78 
79         if (structure->storedPrototype() != globalObject->mapPrototype())
80             return false;
81 
82         return true;
83     };
84 
85     return isIteratorProtocolFastAndNonObservable() && setFastAndNonObservable(structure);
86 }
