26 #pragma once
27 
28 #include <wtf/CompletionHandler.h>
29 #include <wtf/MainThread.h>
30 #include <wtf/ThreadSafeRefCounted.h>
31 
32 namespace WTF {
33 
34 class CallbackAggregator : public ThreadSafeRefCounted<CallbackAggregator> {
35 public:
36     static Ref<CallbackAggregator> create(CompletionHandler<void()>&& callback) { return adoptRef(*new CallbackAggregator(WTFMove(callback))); }
37 
38     ~CallbackAggregator()
39     {
40         ASSERT(m_wasConstructedOnMainThread == isMainThread());
41         if (m_callback)
42             m_callback();
43     }
44 
45 private:
46     explicit CallbackAggregator(CompletionHandler<void()>&& callback)
47         : m_callback(WTFMove(callback))
48 #if !ASSERT_DISABLED
49         , m_wasConstructedOnMainThread(isMainThread())
50 #endif
51     {
52     }
53 
54     CompletionHandler<void()> m_callback;
55 #if !ASSERT_DISABLED
56     bool m_wasConstructedOnMainThread;
57 #endif
58 };
59 
60 } // namespace WTF
61 
62 using WTF::CallbackAggregator;
