31 namespace WebCore {
32 
33 bool getCommonCryptoDigestAlgorithm(CryptoAlgorithmIdentifier hashFunction, CCDigestAlgorithm& algorithm)
34 {
35     switch (hashFunction) {
36     case CryptoAlgorithmIdentifier::SHA_1:
37         ALLOW_DEPRECATED_DECLARATIONS_BEGIN
38         algorithm = kCCDigestSHA1;
39         ALLOW_DEPRECATED_DECLARATIONS_END
40         return true;
41     case CryptoAlgorithmIdentifier::SHA_224:
42         algorithm = kCCDigestSHA224;
43         return true;
44     case CryptoAlgorithmIdentifier::SHA_256:
45         algorithm = kCCDigestSHA256;
46         return true;
47     case CryptoAlgorithmIdentifier::SHA_384:
48         algorithm = kCCDigestSHA384;
49         return true;
50     case CryptoAlgorithmIdentifier::SHA_512:
51         algorithm = kCCDigestSHA512;
52         return true;
53     default:
54         return false;
55     }
56 }
