30 #include "MediaStreamTrack.h"

33 namespace WebCore {
34 
35 WTF_MAKE_ISO_ALLOCATED_IMPL(MediaStreamTrackEvent);
36 
37 Ref<MediaStreamTrackEvent> MediaStreamTrackEvent::create(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, RefPtr<MediaStreamTrack>&& track)
38 {
39     return adoptRef(*new MediaStreamTrackEvent(type, canBubble, cancelable, WTFMove(track)));
40 }
41 
42 Ref<MediaStreamTrackEvent> MediaStreamTrackEvent::create(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
43 {
44     return adoptRef(*new MediaStreamTrackEvent(type, initializer, isTrusted));
45 }
46 
47 MediaStreamTrackEvent::MediaStreamTrackEvent(const AtomString& type, CanBubble canBubble, IsCancelable cancelable, RefPtr<MediaStreamTrack>&& track)
48     : Event(type, canBubble, cancelable)
49     , m_track(WTFMove(track))
50 {
51 }
52 
53 MediaStreamTrackEvent::MediaStreamTrackEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
54     : Event(type, initializer, isTrusted)
55     , m_track(initializer.track)
56 {
57 }
58 
59 MediaStreamTrackEvent::~MediaStreamTrackEvent() = default;
60 
61 MediaStreamTrack* MediaStreamTrackEvent::track() const
62 {
63     return m_track.get();
64 }
65 
66 EventInterface MediaStreamTrackEvent::eventInterface() const
67 {
68     return MediaStreamTrackEventInterfaceType;
69 }

71 } // namespace WebCore
