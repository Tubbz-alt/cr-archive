34 namespace JSC {
35 
36 MetadataTable::MetadataTable(UnlinkedMetadataTable& unlinkedMetadata)
37 {
38     new (&linkingData()) UnlinkedMetadataTable::LinkingData {
39         unlinkedMetadata,
40         1,
41     };
42 }
43 
44 struct DeallocTable {
45     template<typename Op>
46     static void withOpcodeType(MetadataTable* table)
47     {
48         table->forEach<Op>([](auto& entry) {
49             entry.~Metadata();
50         });
51     }
52 };
53 
54 MetadataTable::~MetadataTable()
55 {
56     for (unsigned i = 0; i < NUMBER_OF_BYTECODE_WITH_METADATA; i++)
57         getOpcodeType<DeallocTable>(static_cast<OpcodeID>(i), this);
58     linkingData().~LinkingData();
59 }
60 
61 void MetadataTable::destroy(MetadataTable* table)
62 {
63     Ref<UnlinkedMetadataTable> unlinkedMetadata = WTFMove(table->linkingData().unlinkedMetadata);
64     table->~MetadataTable();
65     // Since UnlinkedMetadata::unlink frees the underlying memory of MetadataTable.
66     // We need to destroy LinkingData before calling it.
67     unlinkedMetadata->unlink(*table);
68 }
69 
70 size_t MetadataTable::sizeInBytes()
71 {
72     return linkingData().unlinkedMetadata->sizeInBytes(*this);
73 }
