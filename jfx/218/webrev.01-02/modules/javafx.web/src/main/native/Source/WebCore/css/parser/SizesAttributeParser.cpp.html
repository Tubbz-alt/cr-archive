 47 float SizesAttributeParser::computeLength(double value, CSSUnitType type, const Document& document)
 48 {
 49     auto* renderer = document.renderView();
 50     if (!renderer)
 51         return 0;
 52     auto& style = renderer->style();
 53 
 54     CSSToLengthConversionData conversionData(&style, &style, renderer);
 55 
 56     // Because we evaluate "sizes" at parse time (before style has been resolved), the font metrics used for these specific units
 57     // are not available. The font selector's internal consistency isn't guaranteed just yet, so we can just temporarily clear
 58     // the pointer to it for the duration of the unit evaluation. This is acceptible because the style always comes from the
 59     // RenderView, which has its font information hardcoded in resolveForDocument() to be -webkit-standard, whose operations
 60     // don't require a font selector.
 61     if (type == CSSUnitType::CSS_EXS || type == CSSUnitType::CSS_CHS) {
 62         RefPtr<FontSelector> fontSelector = style.fontCascade().fontSelector();
 63         style.fontCascade().update(nullptr);
 64         float result = CSSPrimitiveValue::computeNonCalcLengthDouble(conversionData, type, value);
 65         style.fontCascade().update(fontSelector.get());
 66         return result;
 67     }
 68 
 69     return clampTo<float>(CSSPrimitiveValue::computeNonCalcLengthDouble(conversionData, type, value));
 70 }

 72 SizesAttributeParser::SizesAttributeParser(const String& attribute, const Document& document, MediaQueryDynamicResults* mediaQueryDynamicResults)
 73     : m_document(document)
 74     , m_mediaQueryDynamicResults(mediaQueryDynamicResults)
 75 {
 76     m_isValid = parse(CSSTokenizer(attribute).tokenRange());
 77 }
 78 
 79 float SizesAttributeParser::length()
 80 {
 81     if (m_isValid)
 82         return effectiveSize();
 83     return effectiveSizeDefaultValue();
 84 }
 85 
 86 bool SizesAttributeParser::calculateLengthInPixels(CSSParserTokenRange range, float& result)
 87 {
 88     const CSSParserToken& startToken = range.peek();
 89     CSSParserTokenType type = startToken.type();
 90     if (type == DimensionToken) {
 91         if (!CSSPrimitiveValue::isLength(startToken.unitType()))
 92             return false;
 93         result = computeLength(startToken.numericValue(), startToken.unitType(), m_document);
 94         if (result >= 0)
 95             return true;
 96     } else if (type == FunctionToken) {
 97         SizesCalcParser calcParser(range, m_document);
 98         if (!calcParser.isValid())
 99             return false;
100         result = calcParser.result();
101         return true;
102     } else if (type == NumberToken && !startToken.numericValue()) {
103         result = 0;
104         return true;
105     }
106 
107     return false;
108 }

110 bool SizesAttributeParser::mediaConditionMatches(const MediaQuerySet& mediaCondition)
111 {
112     // A Media Condition cannot have a media type other than screen.
113     auto* renderer = m_document.renderView();
114     if (!renderer)
115         return false;
116     auto& style = renderer->style();
117     return MediaQueryEvaluator { "screen", m_document, &style }.evaluate(mediaCondition, m_mediaQueryDynamicResults);
118 }

120 bool SizesAttributeParser::parse(CSSParserTokenRange range)
121 {
122     // Split on a comma token and parse the result tokens as (media-condition, length) pairs
123     while (!range.atEnd()) {
124         const CSSParserToken* mediaConditionStart = &range.peek();
125         // The length is the last component value before the comma which isn't whitespace or a comment
126         const CSSParserToken* lengthTokenStart = &range.peek();
127         const CSSParserToken* lengthTokenEnd = &range.peek();
128         while (!range.atEnd() && range.peek().type() != CommaToken) {
129             lengthTokenStart = &range.peek();
130             range.consumeComponentValue();
131             lengthTokenEnd = &range.peek();
132             range.consumeWhitespace();
133         }
134         range.consume();
135 
136         float length;
137         if (!calculateLengthInPixels(range.makeSubRange(lengthTokenStart, lengthTokenEnd), length))
138             continue;
139         RefPtr<MediaQuerySet> mediaCondition = MediaQueryParser::parseMediaCondition(range.makeSubRange(mediaConditionStart, lengthTokenStart), MediaQueryParserContext(m_document));
140         if (!mediaCondition || !mediaConditionMatches(*mediaCondition))
141             continue;
142         m_length = length;
143         m_lengthWasSet = true;
144         return true;
145     }
146     return false;
147 }
148 
149 float SizesAttributeParser::effectiveSize()
150 {
151     if (m_lengthWasSet)
152         return m_length;
153     return effectiveSizeDefaultValue();
154 }
155 
156 unsigned SizesAttributeParser::effectiveSizeDefaultValue()
157 {
158     auto* renderer = m_document.renderView();
159     if (!renderer)
160         return 0;
161     auto& style = renderer->style();
162     return clampTo<float>(CSSPrimitiveValue::computeNonCalcLengthDouble({ &style, &style, renderer }, CSSUnitType::CSS_VW, 100.0));
163 }
