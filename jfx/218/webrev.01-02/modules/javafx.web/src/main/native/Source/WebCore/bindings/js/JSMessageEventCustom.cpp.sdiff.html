48 JSC::JSValue JSMessageEvent::ports(JSC::ExecState& state) const
49 {
50     auto throwScope = DECLARE_THROW_SCOPE(state.vm());
51     return cachedPropertyValue(state, *this, wrapped().cachedPorts(), [&] {
52         JSC::JSValue ports = toJS<IDLFrozenArray<IDLInterface<MessagePort>>>(state, *globalObject(), throwScope, wrapped().ports());
53         return ports;
54     });
55 }

57 JSC::JSValue JSMessageEvent::data(JSC::ExecState& state) const
58 {
59     return cachedPropertyValue(state, *this, wrapped().cachedData(), [this, &state] {
60         return WTF::switchOn(wrapped().data(), [] (JSC::JSValue data) {
61             return data ? data : JSC::jsNull();
62         }, [this, &state] (const Ref<SerializedScriptValue>& data) {
63             // FIXME: Is it best to handle errors by returning null rather than throwing an exception?
64             return data->deserialize(state, globalObject(), wrapped().ports(), SerializationErrorMode::NonThrowing);
65         }, [&state] (const String& data) {
66             return toJS<IDLDOMString>(state, data);
67         }, [this, &state] (const Ref<Blob>& data) {
68             return toJS<IDLInterface<Blob>>(state, *globalObject(), data);
69         }, [this, &state] (const Ref<ArrayBuffer>& data) {
70             return toJS<IDLInterface<ArrayBuffer>>(state, *globalObject(), data);
71         });
72     });
73 }

75 void JSMessageEvent::visitAdditionalChildren(JSC::SlotVisitor& visitor)
76 {
77     WTF::switchOn(wrapped().data(), [&visitor] (const JSValueInWrappedObject& data) {
78         data.visit(visitor);
79     }, [] (const Ref<SerializedScriptValue>&) {
80     }, [] (const String&) {
81     }, [] (const Ref<Blob>&) {
82     }, [] (const Ref<ArrayBuffer>&) {
83     });
84 
85     wrapped().cachedData().visit(visitor);
86     wrapped().cachedPorts().visit(visitor);
87 }
