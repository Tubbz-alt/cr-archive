30 #include "WasmFormat.h"
31 #include "WasmLimits.h"
32 #include "WriteBarrier.h"
33 #include <wtf/Ref.h>
34 #include <wtf/ThreadSafeRefCounted.h>
35 
36 namespace JSC { namespace Wasm {
37 
38 class Instance;
39 
40 class Global final : public ThreadSafeRefCounted<Global> {
41     WTF_MAKE_NONCOPYABLE(Global);
42     WTF_MAKE_FAST_ALLOCATED(Global);
43 public:
44     union Value {
45         uint64_t m_primitive;
46         WriteBarrierBase<Unknown> m_anyref;
47         Value* m_pointer;
48     };
49 
50     static Ref<Global> create(Wasm::Type type, Wasm::GlobalInformation::Mutability mutability, uint64_t initialValue = 0)
51     {
52         return adoptRef(*new Global(type, mutability, initialValue));
53     }
54 
55     Wasm::Type type() const { return m_type; }
56     Wasm::GlobalInformation::Mutability mutability() const { return m_mutability; }
57     JSValue get() const;
58     uint64_t getPrimitive() const { return m_value.m_primitive; }
59     void set(JSGlobalObject*, JSValue);
60     void visitAggregate(SlotVisitor&);
61 
62     template<typename T> T* owner() const { return reinterpret_cast<T*>(m_owner); }
63     void setOwner(JSObject* owner)
64     {
65         ASSERT(!m_owner);
66         ASSERT(owner);
67         m_owner = owner;
68     }
69 
70     static ptrdiff_t offsetOfValue() { return OBJECT_OFFSETOF(Global, m_value); }
71     static ptrdiff_t offsetOfOwner() { return OBJECT_OFFSETOF(Global, m_owner); }
72 
73     static Global& fromBinding(Value& value)
74     {
75         return *bitwise_cast<Global*>(bitwise_cast<uint8_t*>(&value) - offsetOfValue());
76     }
77 
78     Value* valuePointer() { return &m_value; }
79 
80 private:
81     Global(Wasm::Type type, Wasm::GlobalInformation::Mutability mutability, uint64_t initialValue)
82         : m_type(type)
83         , m_mutability(mutability)
84         , m_value()
85     {
86         m_value.m_primitive = initialValue;
87     }
88 
89     Wasm::Type m_type;
90     Wasm::GlobalInformation::Mutability m_mutability;
91     JSObject* m_owner { nullptr };
92     Value m_value;
93 };
94 
95 } } // namespace JSC::Wasm
96 
97 #endif // ENABLE(WEBASSEMBLY)
