30 #include "RegistrableDomain.h"

36 namespace WebCore {
37 
38 class SWServer;
39 struct ServiceWorkerClientData;
40 struct ServiceWorkerClientIdentifier;
41 struct ServiceWorkerContextData;
42 struct ServiceWorkerJobDataIdentifier;
43 
44 class SWServerToContextConnection {
45     WTF_MAKE_FAST_ALLOCATED;
46 public:
47     WEBCORE_EXPORT virtual ~SWServerToContextConnection();
48 
49     SWServerToContextConnectionIdentifier identifier() const { return m_identifier; }
50 
51     // Messages to the SW host process
52     virtual void installServiceWorkerContext(const ServiceWorkerContextData&, const String& userAgent) = 0;
53     virtual void fireInstallEvent(ServiceWorkerIdentifier) = 0;
54     virtual void fireActivateEvent(ServiceWorkerIdentifier) = 0;
55     virtual void terminateWorker(ServiceWorkerIdentifier) = 0;
56     virtual void syncTerminateWorker(ServiceWorkerIdentifier) = 0;
57     virtual void findClientByIdentifierCompleted(uint64_t requestIdentifier, const Optional<ServiceWorkerClientData>&, bool hasSecurityError) = 0;
58     virtual void matchAllCompleted(uint64_t requestIdentifier, const Vector<ServiceWorkerClientData>&) = 0;
59     virtual void claimCompleted(uint64_t requestIdentifier) = 0;

60 
61     // Messages back from the SW host process
62     WEBCORE_EXPORT void scriptContextFailedToStart(const Optional<ServiceWorkerJobDataIdentifier>&, ServiceWorkerIdentifier, const String& message);
63     WEBCORE_EXPORT void scriptContextStarted(const Optional<ServiceWorkerJobDataIdentifier>&, ServiceWorkerIdentifier, bool doesHandleFetch);
64     WEBCORE_EXPORT void didFinishInstall(const Optional<ServiceWorkerJobDataIdentifier>&, ServiceWorkerIdentifier, bool wasSuccessful);
65     WEBCORE_EXPORT void didFinishActivation(ServiceWorkerIdentifier);
66     WEBCORE_EXPORT void setServiceWorkerHasPendingEvents(ServiceWorkerIdentifier, bool hasPendingEvents);
67     WEBCORE_EXPORT void skipWaiting(ServiceWorkerIdentifier, CompletionHandler<void()>&&);
68     WEBCORE_EXPORT void workerTerminated(ServiceWorkerIdentifier);
69     WEBCORE_EXPORT void findClientByIdentifier(uint64_t clientIdRequestIdentifier, ServiceWorkerIdentifier, ServiceWorkerClientIdentifier);
70     WEBCORE_EXPORT void matchAll(uint64_t requestIdentifier, ServiceWorkerIdentifier, const ServiceWorkerClientQueryOptions&);
71     WEBCORE_EXPORT void claim(uint64_t requestIdentifier, ServiceWorkerIdentifier);
72     WEBCORE_EXPORT void setScriptResource(ServiceWorkerIdentifier, URL&& scriptURL, String&& script, URL&& responseURL, String&& mimeType);
73     WEBCORE_EXPORT void didFailHeartBeatCheck(ServiceWorkerIdentifier);

74 
75     const RegistrableDomain& registrableDomain() const { return m_registrableDomain; }
76 
77     virtual void connectionIsNoLongerNeeded() = 0;
78 
79 protected:
80     WEBCORE_EXPORT explicit SWServerToContextConnection(RegistrableDomain&&);
81 
82 private:
83     SWServerToContextConnectionIdentifier m_identifier;
84     RegistrableDomain m_registrableDomain;
85 };
86 
87 } // namespace WebCore
88 
89 #endif // ENABLE(SERVICE_WORKER)
