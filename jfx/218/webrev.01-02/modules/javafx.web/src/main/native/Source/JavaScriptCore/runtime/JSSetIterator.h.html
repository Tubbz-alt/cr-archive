 28 #include "IterationKind.h"
 29 #include "JSObject.h"
 30 #include "JSSet.h"

 32 namespace JSC {
 33 
 34 // Now, it is only used for serialization.
 35 class JSSetIterator final : public JSCell {
 36     typedef HashMapBucket<HashMapBucketDataKey> HashMapBucketType;
 37 public:
 38     using Base = JSCell;
 39 
 40     DECLARE_EXPORT_INFO;
 41 
 42     template<typename CellType, SubspaceAccess mode>
 43     static IsoSubspace* subspaceFor(VM& vm)
 44     {
 45         return vm.setIteratorSpace<mode>();
 46     }
 47 
 48     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
 49     {
 50         return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
 51     }
 52 
 53     static JSSetIterator* create(VM& vm, Structure* structure, JSSet* iteratedObject, IterationKind kind)
 54     {
 55         JSSetIterator* instance = new (NotNull, allocateCell<JSSetIterator>(vm.heap)) JSSetIterator(vm, structure, iteratedObject, kind);
 56         instance->finishCreation(vm, iteratedObject);
 57         return instance;
 58     }
 59 
 60     ALWAYS_INLINE HashMapBucketType* advanceIter(JSGlobalObject* globalObject)
 61     {
 62         HashMapBucketType* prev = m_iter.get();
 63         if (!prev)
 64             return nullptr;
 65         VM& vm = getVM(globalObject);
 66         HashMapBucketType* bucket = m_iter->next();
 67         while (bucket && bucket->deleted())
 68             bucket = bucket->next();
 69         if (!bucket) {
 70             setIterator(vm, nullptr);
 71             return nullptr;
 72         }
 73         setIterator(vm, bucket); // We keep m_iter on the last value since the first thing we do in this function is call next().
 74         return bucket;
 75     }
 76 
 77     bool next(JSGlobalObject* globalObject, JSValue& value)
 78     {
 79         HashMapBucketType* bucket = advanceIter(globalObject);
 80         if (!bucket)
 81             return false;
 82 
 83         if (m_kind == IterationKind::Values || m_kind == IterationKind::Keys)
 84             value = bucket->key();
 85         else
 86             value = createPair(globalObject, bucket->key(), bucket->key());
 87         return true;
 88     }
 89 
 90     IterationKind kind() const { return m_kind; }
 91     JSValue iteratedValue() const { return m_set.get(); }
 92 
 93 private:
 94     JSSetIterator(VM& vm, Structure* structure, JSSet*, IterationKind kind)
 95         : Base(vm, structure)
 96         , m_kind(kind)
 97     {
 98     }
 99 
100     void setIterator(VM& vm, HashMapBucketType* bucket)
101     {
102         m_iter.setMayBeNull(vm, this, bucket);
103     }
104 
105     JS_EXPORT_PRIVATE void finishCreation(VM&, JSSet*);
106     JS_EXPORT_PRIVATE JSValue createPair(JSGlobalObject*, JSValue, JSValue);
107     static void visitChildren(JSCell*, SlotVisitor&);
108 
109     WriteBarrier<JSSet> m_set;
110     WriteBarrier<HashMapBucketType> m_iter;
111     IterationKind m_kind;
112 };
113 STATIC_ASSERT_IS_TRIVIALLY_DESTRUCTIBLE(JSSetIterator);
114 
115 } // namespace JSC
