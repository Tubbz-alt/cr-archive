26 #pragma once
27 
28 #include "Color.h"
29 #include <wtf/Optional.h>
30 
31 namespace WebCore {
32 
33 struct CompositionHighlight {
34     CompositionHighlight() = default;
35     CompositionHighlight(unsigned startOffset, unsigned endOffset, const Color& c)
36         : startOffset(startOffset)
37         , endOffset(endOffset)
38         , color(c)
39     {
40     }
41 
42     unsigned startOffset { 0 };
43     unsigned endOffset { 0 };
44     Color color { Color::compositionFill };
45 
46     template<class Encoder> void encode(Encoder&) const;
47     template<class Decoder> static Optional<CompositionHighlight> decode(Decoder&);
48 };
49 
50 template<class Encoder>
51 void CompositionHighlight::encode(Encoder& encoder) const
52 {
53     encoder << startOffset;
54     encoder << endOffset;
55     encoder << color;
56 }
57 
58 template<class Decoder>
59 Optional<CompositionHighlight> CompositionHighlight::decode(Decoder& decoder)
60 {
61     Optional<unsigned> startOffset;
62     decoder >> startOffset;
63     if (!startOffset)
64         return WTF::nullopt;
65 
66     Optional<unsigned> endOffset;
67     decoder >> endOffset;
68     if (!endOffset)
69         return WTF::nullopt;
70 
71     Optional<Color> color;
72     decoder >> color;
73     if (!color)
74         return WTF::nullopt;
75 
76     return {{ *startOffset, *endOffset, *color }};
77 }
78 
79 } // namespace WebCore
