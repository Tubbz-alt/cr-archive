 28 #pragma once
 29 
 30 #if ENABLE(MEDIA_STREAM)
 31 
 32 #include "RealtimeMediaSource.h"
 33 #include <wtf/LoggerHelper.h>
 34 #include <wtf/RefCounted.h>
 35 #include <wtf/WeakPtr.h>
 36 
 37 namespace WebCore {
 38 
 39 class AudioSourceProvider;
 40 class GraphicsContext;
 41 class MediaSample;
 42 class RealtimeMediaSourceCapabilities;
 43 class WebAudioSourceProvider;
 44 
 45 class MediaStreamTrackPrivate final
 46     : public RefCounted<MediaStreamTrackPrivate>
 47     , public CanMakeWeakPtr<MediaStreamTrackPrivate, WeakPtrFactoryInitialization::Eager>
 48     , public RealtimeMediaSource::Observer
 49 #if !RELEASE_LOG_DISABLED
 50     , public LoggerHelper
 51 #endif
 52 {
 53 public:
 54     class Observer {
 55     public:
 56         virtual ~Observer() = default;
 57 
 58         virtual void trackStarted(MediaStreamTrackPrivate&) { };
 59         virtual void trackEnded(MediaStreamTrackPrivate&) = 0;
 60         virtual void trackMutedChanged(MediaStreamTrackPrivate&) = 0;
 61         virtual void trackSettingsChanged(MediaStreamTrackPrivate&) = 0;
 62         virtual void trackEnabledChanged(MediaStreamTrackPrivate&) = 0;
 63         virtual void sampleBufferUpdated(MediaStreamTrackPrivate&, MediaSample&) { };
 64         virtual void readyStateChanged(MediaStreamTrackPrivate&) { };
 65 
 66         // May get called on a background thread.
 67         virtual void audioSamplesAvailable(MediaStreamTrackPrivate&, const MediaTime&, const PlatformAudioData&, const AudioStreamDescription&, size_t) { };
 68     };
 69 
 70     static Ref<MediaStreamTrackPrivate> create(Ref<const Logger>&&, Ref<RealtimeMediaSource>&&);
 71     static Ref<MediaStreamTrackPrivate> create(Ref<const Logger>&&, Ref<RealtimeMediaSource>&&, String&& id);
 72 
 73     virtual ~MediaStreamTrackPrivate();
 74 
 75     const String& id() const { return m_id; }
 76     const String& label() const;
 77 
 78     bool isActive() const { return enabled() && !ended() && !muted(); }
 79 
 80     bool ended() const { return m_isEnded; }
 81 
 82     enum class HintValue { Empty, Speech, Music, Motion, Detail, Text };
 83     HintValue contentHint() const { return m_contentHint; }
 84     void setContentHint(HintValue);
 85 
 86     void startProducingData() { m_source->start(); }
 87     void stopProducingData() { m_source->stop(); }
 88     bool isProducingData() { return m_source->isProducingData(); }
 89 
 90     bool isIsolated() const { return m_source->isIsolated(); }
 91 
 92     bool muted() const;
 93     void setMuted(bool muted) { m_source->setMuted(muted); }
 94 
 95     bool isCaptureTrack() const;
 96 
 97     bool enabled() const { return m_isEnabled; }
 98     void setEnabled(bool);
 99 
100     Ref<MediaStreamTrackPrivate> clone();
101 
102     RealtimeMediaSource& source() { return m_source.get(); }
103     WEBCORE_EXPORT RealtimeMediaSource::Type type() const;
104 
105     void endTrack();
106 
107     void addObserver(Observer&);
108     void removeObserver(Observer&);
109 #if ASSERT_ENABLED
110     bool hasObserver(Observer&) const;
111 #endif
112 
113     WEBCORE_EXPORT const RealtimeMediaSourceSettings& settings() const;
114     const RealtimeMediaSourceCapabilities& capabilities() const;
115 
116     void applyConstraints(const MediaConstraints&, RealtimeMediaSource::ApplyConstraintsHandler&&);
117 
118     AudioSourceProvider* audioSourceProvider();
119 
120     void paintCurrentFrameInContext(GraphicsContext&, const FloatRect&);
121 
122     enum class ReadyState { None, Live, Ended };
123     ReadyState readyState() const { return m_readyState; }
124 
125     void setIdForTesting(String&& id) { m_id = WTFMove(id); }
126 
127 #if !RELEASE_LOG_DISABLED
128     const Logger& logger() const final { return m_logger; }
129     const void* logIdentifier() const final { return m_logIdentifier; }
130 #endif
131 
132 private:
133     MediaStreamTrackPrivate(Ref<const Logger>&&, Ref<RealtimeMediaSource>&&, String&& id);
134 
135     // RealtimeMediaSourceObserver
136     void sourceStarted() final;
137     void sourceStopped() final;
138     void sourceMutedChanged() final;
139     void sourceSettingsChanged() final;
140     bool preventSourceFromStopping() final;
141     void videoSampleAvailable(MediaSample&) final;
142     void audioSamplesAvailable(const MediaTime&, const PlatformAudioData&, const AudioStreamDescription&, size_t) final;
143 
144     void updateReadyState();
145 
146     void forEachObserver(const WTF::Function<void(Observer&)>&) const;
147 
148 #if !RELEASE_LOG_DISABLED
149     const char* logClassName() const final { return "MediaStreamTrackPrivate"; }
150     WTFLogChannel& logChannel() const final;
151 #endif
152 
153     mutable RecursiveLock m_observersLock;
154     HashSet<Observer*> m_observers;
155     Ref<RealtimeMediaSource> m_source;
156 
157     String m_id;
158     ReadyState m_readyState { ReadyState::None };
159     bool m_isEnabled { true };
160     bool m_isEnded { false };
161     bool m_haveProducedData { false };
162     bool m_hasSentStartProducedData { false };
163     HintValue m_contentHint { HintValue::Empty };
164     RefPtr<WebAudioSourceProvider> m_audioSourceProvider;
165     Ref<const Logger> m_logger;
166 #if !RELEASE_LOG_DISABLED
167     const void* m_logIdentifier;
168 #endif
169 };
170 
171 typedef Vector<RefPtr<MediaStreamTrackPrivate>> MediaStreamTrackPrivateVector;
172 
173 #if ASSERT_ENABLED
174 inline bool MediaStreamTrackPrivate::hasObserver(Observer& observer) const
175 {
176     auto locker = holdLock(m_observersLock);
177     return m_observers.contains(&observer);
178 }
179 #endif
180 
181 } // namespace WebCore

183 #endif // ENABLE(MEDIA_STREAM)
