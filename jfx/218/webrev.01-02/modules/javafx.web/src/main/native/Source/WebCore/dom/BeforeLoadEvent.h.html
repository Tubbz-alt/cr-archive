27 #pragma once
28 
29 #include "Event.h"
30 #include "EventNames.h"
31 
32 namespace WebCore {
33 
34 class BeforeLoadEvent final : public Event {
35     WTF_MAKE_ISO_ALLOCATED(BeforeLoadEvent);
36 public:
37     static Ref<BeforeLoadEvent> create(const String& url)
38     {
39         return adoptRef(*new BeforeLoadEvent(url));
40     }
41 
42     struct Init : EventInit {
43         String url;
44     };
45 
46     static Ref<BeforeLoadEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
47     {
48         return adoptRef(*new BeforeLoadEvent(type, initializer, isTrusted));
49     }
50 
51     const String& url() const { return m_url; }
52 
53     EventInterface eventInterface() const override { return BeforeLoadEventInterfaceType; }
54 
55 private:
56     explicit BeforeLoadEvent(const String& url)
57         : Event(eventNames().beforeloadEvent, CanBubble::No, IsCancelable::Yes)
58         , m_url(url)
59     {
60     }
61 
62     BeforeLoadEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
63         : Event(type, initializer, isTrusted)
64         , m_url(initializer.url)
65     {
66     }
67 
68     String m_url;
69 };
70 
71 } // namespace WebCore
