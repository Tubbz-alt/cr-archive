20 #pragma once
21 
22 #include "Length.h"
23 
24 namespace WebCore {
25 
26 struct LengthSize {
27     Length width;
28     Length height;
29 
30     bool isEmpty() const { return width.isZero() || height.isZero(); }

31 };
32 
33 ALWAYS_INLINE bool operator==(const LengthSize& a, const LengthSize& b)
34 {
35     return a.width == b.width && a.height == b.height;
36 }
37 
38 inline LengthSize blend(const LengthSize& from, const LengthSize& to, double progress)
39 {
40     return { blend(from.width, to.width, progress), blend(from.height, to.height, progress) };
41 }
42 
43 WTF::TextStream& operator<<(WTF::TextStream&, const LengthSize&);
44 
45 } // namespace WebCore
