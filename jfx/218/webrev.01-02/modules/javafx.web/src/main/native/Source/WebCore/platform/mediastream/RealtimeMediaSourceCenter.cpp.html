 67 void RealtimeMediaSourceCenter::createMediaStream(Ref<const Logger>&& logger, NewMediaStreamHandler&& completionHandler, String&& hashSalt, CaptureDevice&& audioDevice, CaptureDevice&& videoDevice, const MediaStreamRequest& request)
 68 {
 69     Vector<Ref<RealtimeMediaSource>> audioSources;
 70     Vector<Ref<RealtimeMediaSource>> videoSources;
 71     String invalidConstraint;
 72 
 73     if (audioDevice) {
 74         auto audioSource = audioCaptureFactory().createAudioCaptureSource(WTFMove(audioDevice), String { hashSalt }, &request.audioConstraints);
 75         if (audioSource)
 76             audioSources.append(audioSource.source());
 77         else {
 78 #if !LOG_DISABLED
 79             if (!audioSource.errorMessage.isEmpty())
 80                 LOG(Media, "RealtimeMediaSourceCenter::createMediaStream(%p), audio constraints failed to apply: %s", this, audioSource.errorMessage.utf8().data());
 81 #endif
 82             completionHandler(makeUnexpected(makeString("Failed to create MediaStream audio source: ", audioSource.errorMessage)));
 83             return;
 84         }
 85     }
 86 
 87     if (videoDevice) {
 88         CaptureSourceOrError videoSource;
 89         if (videoDevice.type() == CaptureDevice::DeviceType::Camera)
 90             videoSource = videoCaptureFactory().createVideoCaptureSource(WTFMove(videoDevice), WTFMove(hashSalt), &request.videoConstraints);
 91         else
 92             videoSource = displayCaptureFactory().createDisplayCaptureSource(WTFMove(videoDevice), &request.videoConstraints);
 93 
 94         if (videoSource)
 95             videoSources.append(videoSource.source());
 96         else {
 97 #if !LOG_DISABLED
 98             if (!videoSource.errorMessage.isEmpty())
 99                 LOG(Media, "RealtimeMediaSourceCenter::createMediaStream(%p), video constraints failed to apply: %s", this, videoSource.errorMessage.utf8().data());
100 #endif
101             completionHandler(makeUnexpected(makeString("Failed to create MediaStream video source: ", videoSource.errorMessage)));
102             return;
103         }
104     }
105 
106     completionHandler(MediaStreamPrivate::create(WTFMove(logger), audioSources, videoSources));
107 }

109 Vector<CaptureDevice> RealtimeMediaSourceCenter::getMediaStreamDevices()
110 {
111     Vector<CaptureDevice> result;
112     for (auto& device : audioCaptureFactory().audioCaptureDeviceManager().captureDevices()) {
113         if (device.enabled())
114             result.append(device);
115     }
116     for (auto& device : videoCaptureFactory().videoCaptureDeviceManager().captureDevices()) {
117         if (device.enabled())
118             result.append(device);
119     }
120     for (auto& device : displayCaptureFactory().displayCaptureDeviceManager().captureDevices()) {
121         if (device.enabled())
122             result.append(device);
123     }
124 
125     return result;
126 }

128 static void addStringToSHA1(SHA1& sha1, const String& string)
129 {
130     if (string.isEmpty())
131         return;
132 
133     if (string.is8Bit() && string.isAllASCII()) {
134         const uint8_t nullByte = 0;
135         sha1.addBytes(string.characters8(), string.length());
136         sha1.addBytes(&nullByte, 1);
137         return;
138     }
139 
140     auto utf8 = string.utf8();
141     sha1.addBytes(reinterpret_cast<const uint8_t*>(utf8.data()), utf8.length() + 1); // Include terminating null byte.
142 }

144 String RealtimeMediaSourceCenter::hashStringWithSalt(const String& id, const String& hashSalt)
145 {
146     if (id.isEmpty() || hashSalt.isEmpty())
147         return emptyString();
148 
149     SHA1 sha1;
150 
151     addStringToSHA1(sha1, id);
152     addStringToSHA1(sha1, hashSalt);
153 
154     SHA1::Digest digest;
155     sha1.computeHash(digest);
156 
157     return SHA1::hexDigest(digest).data();
158 }

174 void RealtimeMediaSourceCenter::getDisplayMediaDevices(const MediaStreamRequest& request, Vector<DeviceInfo>& diaplayDeviceInfo, String& firstInvalidConstraint)
175 {
176     if (!request.videoConstraints.isValid)
177         return;
178 
179     String invalidConstraint;
180     for (auto& device : displayCaptureFactory().displayCaptureDeviceManager().captureDevices()) {
181         if (!device.enabled())
182             return;
183 
184         auto sourceOrError = displayCaptureFactory().createDisplayCaptureSource(device, { });
185         if (sourceOrError && sourceOrError.captureSource->supportsConstraints(request.videoConstraints, invalidConstraint))
186             diaplayDeviceInfo.append({sourceOrError.captureSource->fitnessScore(), device});
187 
188         if (!invalidConstraint.isEmpty() && firstInvalidConstraint.isEmpty())
189             firstInvalidConstraint = invalidConstraint;
190     }
191 }

193 void RealtimeMediaSourceCenter::getUserMediaDevices(const MediaStreamRequest& request, String&& hashSalt, Vector<DeviceInfo>& audioDeviceInfo, Vector<DeviceInfo>& videoDeviceInfo, String& firstInvalidConstraint)
194 {
195     String invalidConstraint;
196     if (request.audioConstraints.isValid) {
197         for (auto& device : audioCaptureFactory().audioCaptureDeviceManager().captureDevices()) {
198             if (!device.enabled())
199                 continue;
200 
201             auto sourceOrError = audioCaptureFactory().createAudioCaptureSource(device, String { hashSalt }, { });
202             if (sourceOrError && sourceOrError.captureSource->supportsConstraints(request.audioConstraints, invalidConstraint))
203                 audioDeviceInfo.append({sourceOrError.captureSource->fitnessScore(), device});
204 
205             if (!invalidConstraint.isEmpty() && firstInvalidConstraint.isEmpty())
206                 firstInvalidConstraint = invalidConstraint;
207         }
208     }
209 
210     if (request.videoConstraints.isValid) {
211         for (auto& device : videoCaptureFactory().videoCaptureDeviceManager().captureDevices()) {
212             if (!device.enabled())
213                 continue;
214 
215             auto sourceOrError = videoCaptureFactory().createVideoCaptureSource(device, String { hashSalt }, { });
216             if (sourceOrError && sourceOrError.captureSource->supportsConstraints(request.videoConstraints, invalidConstraint))
217                 videoDeviceInfo.append({sourceOrError.captureSource->fitnessScore(), device});
218 
219             if (!invalidConstraint.isEmpty() && firstInvalidConstraint.isEmpty())
220                 firstInvalidConstraint = invalidConstraint;
221         }
222     }
223 }

225 void RealtimeMediaSourceCenter::validateRequestConstraints(ValidConstraintsHandler&& validHandler, InvalidConstraintsHandler&& invalidHandler, const MediaStreamRequest& request, String&& deviceIdentifierHashSalt)
226 {
227     struct {
228         bool operator()(const DeviceInfo& a, const DeviceInfo& b)
229         {
230             return a.fitnessScore > b.fitnessScore;
231         }
232     } sortBasedOnFitnessScore;
233 
234     Vector<DeviceInfo> audioDeviceInfo;
235     Vector<DeviceInfo> videoDeviceInfo;
236     String firstInvalidConstraint;
237 
238     if (request.type == MediaStreamRequest::Type::DisplayMedia)
239         getDisplayMediaDevices(request, videoDeviceInfo, firstInvalidConstraint);
240     else
241         getUserMediaDevices(request, String { deviceIdentifierHashSalt }, audioDeviceInfo, videoDeviceInfo, firstInvalidConstraint);
242 
243     if ((request.audioConstraints.isValid && audioDeviceInfo.isEmpty()) || (request.videoConstraints.isValid && videoDeviceInfo.isEmpty())) {
244         invalidHandler(firstInvalidConstraint);
245         return;
246     }
247 
248     Vector<CaptureDevice> audioDevices;
249     if (!audioDeviceInfo.isEmpty()) {
250         std::sort(audioDeviceInfo.begin(), audioDeviceInfo.end(), sortBasedOnFitnessScore);
251         audioDevices = WTF::map(audioDeviceInfo, [] (auto& info) {
252             return info.device;
253         });
254     }
255 
256     Vector<CaptureDevice> videoDevices;
257     if (!videoDeviceInfo.isEmpty()) {
258         std::sort(videoDeviceInfo.begin(), videoDeviceInfo.end(), sortBasedOnFitnessScore);
259         videoDevices = WTF::map(videoDeviceInfo, [] (auto& info) {
260             return info.device;
261         });
262     }
263 
264     validHandler(WTFMove(audioDevices), WTFMove(videoDevices), WTFMove(deviceIdentifierHashSalt));
265 }
