 33 template<typename T> struct Converter<IDLCallbackFunction<T>> : DefaultConverter<IDLCallbackFunction<T>> {
 34 
 35     static constexpr bool conversionHasSideEffects = false;
 36 
 37     template<typename ExceptionThrower = DefaultExceptionThrower>
 38     static RefPtr<T> convert(JSC::ExecState& state, JSC::JSValue value, JSDOMGlobalObject& globalObject, ExceptionThrower&& exceptionThrower = ExceptionThrower())
 39     {
