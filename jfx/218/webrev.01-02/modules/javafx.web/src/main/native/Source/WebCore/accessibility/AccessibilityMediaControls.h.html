 36 namespace WebCore {

 38 class AccessibilityMediaControl : public AccessibilityRenderObject {
 39 public:
 40     static Ref<AccessibilityObject> create(RenderObject*);
 41     virtual ~AccessibilityMediaControl() = default;
 42 
 43     AccessibilityRole roleValue() const override;
 44 
 45     String title() const override;
 46     String accessibilityDescription() const override;
 47     String helpText() const override;
 48 
 49 protected:
 50     explicit AccessibilityMediaControl(RenderObject*);
 51     MediaControlElementType controlType() const;
 52     const String& controlTypeName() const;
 53     bool computeAccessibilityIsIgnored() const override;
 54 
 55 private:
 56     void accessibilityText(Vector<AccessibilityText>&) const override;
 57 };

 59 
 60 class AccessibilityMediaTimeline final : public AccessibilitySlider {
 61 public:
 62     static Ref<AccessibilityObject> create(RenderObject*);
 63     virtual ~AccessibilityMediaTimeline() = default;
 64 
 65     String helpText() const override;
 66     String valueDescription() const override;
 67 
 68 private:
 69     explicit AccessibilityMediaTimeline(RenderObject*);
 70 
 71     bool isMediaTimeline() const override { return true; }
 72 };

 75 class AccessibilityMediaControlsContainer final : public AccessibilityMediaControl {
 76 public:
 77     static Ref<AccessibilityObject> create(RenderObject*);
 78     virtual ~AccessibilityMediaControlsContainer() = default;
 79 
 80     AccessibilityRole roleValue() const override { return AccessibilityRole::Toolbar; }
 81 
 82     String helpText() const override;
 83     String accessibilityDescription() const override;
 84 
 85 private:
 86     explicit AccessibilityMediaControlsContainer(RenderObject*);
 87     bool controllingVideoElement() const;
 88     const String& elementTypeName() const;
 89     bool computeAccessibilityIsIgnored() const override;
 90 };

 93 class AccessibilityMediaTimeDisplay final : public AccessibilityMediaControl {
 94 public:
 95     static Ref<AccessibilityObject> create(RenderObject*);
 96     virtual ~AccessibilityMediaTimeDisplay() = default;
 97 
 98     AccessibilityRole roleValue() const override { return AccessibilityRole::ApplicationTimer; }
 99 
100     String stringValue() const override;
101     String accessibilityDescription() const override;
102 
103 private:
104     explicit AccessibilityMediaTimeDisplay(RenderObject*);
105     bool isMediaControlLabel() const override { return true; }
106     bool computeAccessibilityIsIgnored() const override;
107 };
