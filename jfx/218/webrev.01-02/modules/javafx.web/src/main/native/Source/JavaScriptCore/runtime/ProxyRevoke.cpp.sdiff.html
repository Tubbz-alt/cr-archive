60 static EncodedJSValue JSC_HOST_CALL performProxyRevoke(ExecState* exec)
61 {
62     ProxyRevoke* proxyRevoke = jsCast<ProxyRevoke*>(exec->jsCallee());
63     JSValue proxyValue = proxyRevoke->proxy();
64     if (proxyValue.isNull())
65         return JSValue::encode(jsUndefined());
66 
67     ProxyObject* proxy = jsCast<ProxyObject*>(proxyValue);
68     VM& vm = exec->vm();
69     proxy->revoke(vm);
70     proxyRevoke->setProxyToNull(vm);
71     return JSValue::encode(jsUndefined());
72 }
