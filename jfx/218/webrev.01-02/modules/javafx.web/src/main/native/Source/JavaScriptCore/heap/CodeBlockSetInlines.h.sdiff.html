36 inline void CodeBlockSet::mark(const AbstractLocker&, CodeBlock* codeBlock)
37 {
38     if (!codeBlock)
39         return;
40 
41     // Conservative root scanning in Eden collection can only find PreciseAllocation that is allocated in this Eden cycle.
42     // Since CodeBlockSet::m_currentlyExecuting is strongly assuming that this catches all the currently executing CodeBlock,
43     // we now have a restriction that all CodeBlock needs to be a non-precise-allocation.
44     ASSERT(!codeBlock->isPreciseAllocation());
45     m_currentlyExecuting.add(codeBlock);
46 }
