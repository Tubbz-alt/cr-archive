26 #pragma once
27 
28 #include "AnimationEventBase.h"
29 #include "AnimationPlaybackEventInit.h"

30 #include <wtf/Markable.h>
31 
32 namespace WebCore {
33 
34 class AnimationPlaybackEvent final : public AnimationEventBase {
35     WTF_MAKE_ISO_ALLOCATED(AnimationPlaybackEvent);
36 public:
37     static Ref<AnimationPlaybackEvent> create(const AtomString& type, Optional<Seconds> currentTime, Optional<Seconds> timelineTime, WebAnimation* animation)
38     {
39         return adoptRef(*new AnimationPlaybackEvent(type, currentTime, timelineTime, animation));
40     }
41 
42     static Ref<AnimationPlaybackEvent> create(const AtomString& type, const AnimationPlaybackEventInit& initializer, IsTrusted isTrusted = IsTrusted::No)
43     {
44         return adoptRef(*new AnimationPlaybackEvent(type, initializer, isTrusted));
45     }
46 
47     virtual ~AnimationPlaybackEvent();
48 
49     bool isAnimationPlaybackEvent() const final { return true; }
50 
51     Optional<double> bindingsCurrentTime() const;
52     Optional<Seconds> currentTime() const { return m_currentTime; }
53     Optional<double> bindingsTimelineTime() const;

54 
55     EventInterface eventInterface() const override { return AnimationPlaybackEventInterfaceType; }
56 
57 private:
58     AnimationPlaybackEvent(const AtomString&, Optional<Seconds>, Optional<Seconds>, WebAnimation*);
59     AnimationPlaybackEvent(const AtomString&, const AnimationPlaybackEventInit&, IsTrusted);
60 
61     Markable<Seconds, Seconds::MarkableTraits> m_currentTime;

62 };
63 
64 }
65 
66 SPECIALIZE_TYPE_TRAITS_ANIMATION_EVENT_BASE(AnimationPlaybackEvent, isAnimationPlaybackEvent())
