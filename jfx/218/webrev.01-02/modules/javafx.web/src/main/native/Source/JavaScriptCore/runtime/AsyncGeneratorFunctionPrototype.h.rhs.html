27 #include "JSObject.h"

29 namespace JSC {
30 
31 class AsyncGeneratorFunctionPrototype final : public JSNonFinalObject {
32 public:
33     using Base = JSNonFinalObject;
34     static constexpr unsigned StructureFlags = Base::StructureFlags | HasStaticPropertyTable;
35 
36     DECLARE_INFO;
37 
38     static AsyncGeneratorFunctionPrototype* create(VM& vm, Structure* structure)
39     {
40         AsyncGeneratorFunctionPrototype* prototype = new (NotNull, allocateCell<AsyncGeneratorFunctionPrototype>(vm.heap)) AsyncGeneratorFunctionPrototype(vm, structure);
41         prototype->finishCreation(vm);
42         return prototype;
43     }
44 
45     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue proto)
46     {
47         return Structure::create(vm, globalObject, proto, TypeInfo(ObjectType, StructureFlags), info());
48     }
49 
50 protected:
51     void finishCreation(VM&);
52 
53 private:
54     AsyncGeneratorFunctionPrototype(VM&, Structure*);
55 };
56 
57 } // namespace JSC
