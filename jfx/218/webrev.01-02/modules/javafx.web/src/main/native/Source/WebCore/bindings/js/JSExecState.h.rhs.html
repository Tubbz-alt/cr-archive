 40 class ScriptExecutionContext;

 42 class JSExecState {
 43     WTF_MAKE_NONCOPYABLE(JSExecState);
 44     WTF_FORBID_HEAP_ALLOCATION;
 45     friend class JSMainThreadNullState;
 46 public:
 47     static JSC::JSGlobalObject* currentState()
 48     {
 49         return threadGlobalData().currentState();
 50     };
 51 
 52     static JSC::JSValue call(JSC::JSGlobalObject* lexicalGlobalObject, JSC::JSValue functionObject, JSC::CallType callType, const JSC::CallData& callData, JSC::JSValue thisValue, const JSC::ArgList& args, NakedPtr<JSC::Exception>& returnedException)
 53     {
 54         JSExecState currentState(lexicalGlobalObject);
 55         return JSC::call(lexicalGlobalObject, functionObject, callType, callData, thisValue, args, returnedException);
 56     };
 57 
 58     static JSC::JSValue evaluate(JSC::JSGlobalObject* lexicalGlobalObject, const JSC::SourceCode& source, JSC::JSValue thisValue, NakedPtr<JSC::Exception>& returnedException)
 59     {
 60         JSExecState currentState(lexicalGlobalObject);
 61         return JSC::evaluate(lexicalGlobalObject, source, thisValue, returnedException);
 62     };
 63 
 64     static JSC::JSValue evaluate(JSC::JSGlobalObject* lexicalGlobalObject, const JSC::SourceCode& source, JSC::JSValue thisValue = JSC::JSValue())
 65     {
 66         NakedPtr<JSC::Exception> unused;
 67         return evaluate(lexicalGlobalObject, source, thisValue, unused);
 68     };
 69 
 70     static JSC::JSValue profiledCall(JSC::JSGlobalObject* lexicalGlobalObject, JSC::ProfilingReason reason, JSC::JSValue functionObject, JSC::CallType callType, const JSC::CallData& callData, JSC::JSValue thisValue, const JSC::ArgList& args, NakedPtr<JSC::Exception>& returnedException)
 71     {
 72         JSExecState currentState(lexicalGlobalObject);
 73         return JSC::profiledCall(lexicalGlobalObject, reason, functionObject, callType, callData, thisValue, args, returnedException);
 74     }
 75 
 76     static JSC::JSValue profiledEvaluate(JSC::JSGlobalObject* lexicalGlobalObject, JSC::ProfilingReason reason, const JSC::SourceCode& source, JSC::JSValue thisValue, NakedPtr<JSC::Exception>& returnedException)
 77     {
 78         JSExecState currentState(lexicalGlobalObject);
 79         return JSC::profiledEvaluate(lexicalGlobalObject, reason, source, thisValue, returnedException);
 80     }
 81 
 82     static JSC::JSValue profiledEvaluate(JSC::JSGlobalObject* lexicalGlobalObject, JSC::ProfilingReason reason, const JSC::SourceCode& source, JSC::JSValue thisValue = JSC::JSValue())
 83     {
 84         NakedPtr<JSC::Exception> unused;
 85         return profiledEvaluate(lexicalGlobalObject, reason, source, thisValue, unused);
 86     }
 87 
 88     static void runTask(JSC::JSGlobalObject* lexicalGlobalObject, JSC::Microtask& task)
 89     {
 90         JSExecState currentState(lexicalGlobalObject);
 91         task.run(lexicalGlobalObject);
 92     }
 93 
 94     static JSC::JSInternalPromise& loadModule(JSC::JSGlobalObject& lexicalGlobalObject, const String& moduleName, JSC::JSValue parameters, JSC::JSValue scriptFetcher)
 95     {
 96         JSExecState currentState(&lexicalGlobalObject);
 97         return *JSC::loadModule(&lexicalGlobalObject, moduleName, parameters, scriptFetcher);
 98     }
 99 
100     static JSC::JSInternalPromise& loadModule(JSC::JSGlobalObject& lexicalGlobalObject, const JSC::SourceCode& sourceCode, JSC::JSValue scriptFetcher)
101     {
102         JSExecState currentState(&lexicalGlobalObject);
103         return *JSC::loadModule(&lexicalGlobalObject, sourceCode, scriptFetcher);
104     }
105 
106     static JSC::JSValue linkAndEvaluateModule(JSC::JSGlobalObject& lexicalGlobalObject, const JSC::Identifier& moduleKey, JSC::JSValue scriptFetcher, NakedPtr<JSC::Exception>& returnedException)
107     {
108         JSC::VM& vm = JSC::getVM(&lexicalGlobalObject);
109         auto scope = DECLARE_CATCH_SCOPE(vm);
110 
111         JSExecState currentState(&lexicalGlobalObject);
112         auto returnValue = JSC::linkAndEvaluateModule(&lexicalGlobalObject, moduleKey, scriptFetcher);
113         if (UNLIKELY(scope.exception())) {
114             returnedException = scope.exception();
115             scope.clearException();
116             return JSC::jsUndefined();
117         }
118         return returnValue;
119     }
120 
121     static void instrumentFunctionCall(ScriptExecutionContext*, JSC::CallType, const JSC::CallData&);
122     static void instrumentFunctionConstruct(ScriptExecutionContext*, JSC::ConstructType, const JSC::ConstructData&);
123 
124 private:
125     explicit JSExecState(JSC::JSGlobalObject* lexicalGlobalObject)
126         : m_previousState(currentState())
127         , m_lock(lexicalGlobalObject)
128     {
129         setCurrentState(lexicalGlobalObject);
130     };
131 
132     ~JSExecState()
133     {
134         JSC::VM& vm = currentState()->vm();
135         auto scope = DECLARE_CATCH_SCOPE(vm);
136         scope.assertNoException();
137 
138         JSC::JSGlobalObject* lexicalGlobalObject = currentState();
139         bool didExitJavaScript = lexicalGlobalObject && !m_previousState;
140 
141         setCurrentState(m_previousState);
142 
143         if (didExitJavaScript)
144             didLeaveScriptContext(lexicalGlobalObject);
145     }
146 
147     static void setCurrentState(JSC::JSGlobalObject* lexicalGlobalObject)
148     {
149         threadGlobalData().setCurrentState(lexicalGlobalObject);
150     }
151 
152     template<typename Type, Type jsType, typename DataType> static void instrumentFunctionInternal(ScriptExecutionContext*, Type, const DataType&);
153 
154     JSC::JSGlobalObject* m_previousState;
155     JSC::JSLockHolder m_lock;
156 
157     static void didLeaveScriptContext(JSC::JSGlobalObject*);
158 };
159 
160 // Null lexicalGlobalObject prevents origin security checks.
161 // Used by non-JavaScript bindings (ObjC, GObject).
162 class JSMainThreadNullState {
163     WTF_MAKE_NONCOPYABLE(JSMainThreadNullState);
164     WTF_FORBID_HEAP_ALLOCATION;
165 public:
166     explicit JSMainThreadNullState()
167         : m_previousState(JSExecState::currentState())
168         , m_customElementReactionStack(m_previousState)
169     {
170         ASSERT(isMainThread());
171         JSExecState::setCurrentState(nullptr);
172     }
173 
174     ~JSMainThreadNullState()
175     {
176         ASSERT(isMainThread());
177         JSExecState::setCurrentState(m_previousState);
178     }
179 
180 private:
181     JSC::JSGlobalObject* m_previousState;
182     CustomElementReactionStack m_customElementReactionStack;
183 };
