36 RefPtr<XPathNSResolver> JSXPathNSResolver::toWrapped(VM& vm, ExecState& state, JSValue value)
37 {
38     if (value.inherits<JSXPathNSResolver>(vm))
39         return &jsCast<JSXPathNSResolver*>(asObject(value))->wrapped();
40 
41     auto result = JSCustomXPathNSResolver::create(state, value);
42     if (UNLIKELY(result.hasException())) {
43         auto scope = DECLARE_THROW_SCOPE(vm);
44         propagateException(state, scope, result.releaseException());
45         return nullptr;
46     }
47     return result.releaseReturnValue();
