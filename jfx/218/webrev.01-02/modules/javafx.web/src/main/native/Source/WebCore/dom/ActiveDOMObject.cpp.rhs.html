 36 static inline ScriptExecutionContext* suitableScriptExecutionContext(ScriptExecutionContext* scriptExecutionContext)





 37 {
 38     // For detached documents, make sure we observe their context document instead.
 39     return is<Document>(scriptExecutionContext) ? &downcast<Document>(*scriptExecutionContext).contextDocument() : scriptExecutionContext;
 40 }
 41 
 42 inline ActiveDOMObject::ActiveDOMObject(ScriptExecutionContext* context, CheckedScriptExecutionContextType)
 43     : ContextDestructionObserver(context)
 44 {
 45     ASSERT(!is<Document>(context) || &downcast<Document>(context)->contextDocument() == downcast<Document>(context));
 46     if (!context)
 47         return;
 48 
 49     ASSERT(context->isContextThread());
 50     context->didCreateActiveDOMObject(*this);
 51 }

 68 ActiveDOMObject::~ActiveDOMObject()
 69 {
 70     ASSERT(canCurrentThreadAccessThreadLocalData(m_creationThread));
 71 
 72     // ActiveDOMObject may be inherited by a sub-class whose life-cycle
 73     // exceeds that of the associated ScriptExecutionContext. In those cases,
 74     // m_scriptExecutionContext would/should have been nullified by
 75     // ContextDestructionObserver::contextDestroyed() (which we implement /
 76     // inherit). Hence, we should ensure that this is not 0 before use it
 77     // here.
 78     if (!m_scriptExecutionContext)
 79         return;
 80 
 81     ASSERT(m_suspendIfNeededWasCalled);
 82     ASSERT(m_scriptExecutionContext->isContextThread());
 83     m_scriptExecutionContext->willDestroyActiveDOMObject(*this);
 84 }
 85 
 86 void ActiveDOMObject::suspendIfNeeded()
 87 {
 88 #if ASSERT_ENABLED
 89     ASSERT(!m_suspendIfNeededWasCalled);
 90     m_suspendIfNeededWasCalled = true;
 91 #endif
 92     if (!m_scriptExecutionContext)
 93         return;
 94 
 95     m_scriptExecutionContext->suspendActiveDOMObjectIfNeeded(*this);
 96 }

 98 #if ASSERT_ENABLED
 99 
100 void ActiveDOMObject::assertSuspendIfNeededWasCalled() const
101 {
102     if (!m_suspendIfNeededWasCalled)
103         WTFLogAlways("Failed to call suspendIfNeeded() for %s", activeDOMObjectName());
104     ASSERT(m_suspendIfNeededWasCalled);
105 }

107 #endif // ASSERT_ENABLED

126 bool ActiveDOMObject::isContextStopped() const

136 void ActiveDOMObject::queueTaskInEventLoop(TaskSource source, Function<void ()>&& function)
137 {
138     auto* context = scriptExecutionContext();
139     if (!context)
140         return;
141     context->eventLoop().queueTask(source, WTFMove(function));
142 }
143 
144 class ActiveDOMObjectEventDispatchTask : public EventLoopTask {
145 public:
146     ActiveDOMObjectEventDispatchTask(TaskSource source, EventLoopTaskGroup& group, ActiveDOMObject& object, EventTarget& target, Ref<Event>&& event)
147         : EventLoopTask(source, group)
148         , m_object(object)
149         , m_target(target)
150         , m_event(WTFMove(event))
151     {
152         ++m_object.m_pendingActivityCount;
153     }
154 
155     ~ActiveDOMObjectEventDispatchTask()
156     {
157         ASSERT(m_object.m_pendingActivityCount);
158         --m_object.m_pendingActivityCount;
159     }
160 
161     void execute() final { m_target->dispatchEvent(m_event.get()); }
162 
163 private:
164     ActiveDOMObject& m_object;
165     Ref<EventTarget> m_target;
166     Ref<Event> m_event;
167 };
168 
169 void ActiveDOMObject::queueTaskToDispatchEventInternal(EventTarget& target, TaskSource source, Ref<Event>&& event)
170 {
171     ASSERT(!event->target() || &target == event->target());
172     auto* context = scriptExecutionContext();
173     if (!context)
174         return;
175     auto& eventLoopTaskGroup = context->eventLoop();
176     auto task = makeUnique<ActiveDOMObjectEventDispatchTask>(source, eventLoopTaskGroup, *this, target, WTFMove(event));
177     eventLoopTaskGroup.queueTask(WTFMove(task));
178 }
