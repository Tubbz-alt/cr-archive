36 JSArrayIterator* JSArrayIterator::create(VM& vm, Structure* structure, JSObject* iteratedObject, JSValue kind)
37 {
38     ASSERT(kind.asUInt32() <= static_cast<uint32_t>(IterationKind::Entries));
39     JSArrayIterator* iterator = new (NotNull, allocateCell<JSArrayIterator>(vm.heap)) JSArrayIterator(vm, structure);
40     iterator->finishCreation(vm);
41     iterator->internalField(Field::IteratedObject).set(vm, iterator, iteratedObject);
42     iterator->internalField(Field::Kind).set(vm, iterator, kind);
43     return iterator;
44 }
45 
46 JSArrayIterator* JSArrayIterator::createWithInitialValues(VM& vm, Structure* structure)
47 {
48     JSArrayIterator* iterator = new (NotNull, allocateCell<JSArrayIterator>(vm.heap)) JSArrayIterator(vm, structure);
49     iterator->finishCreation(vm);
50     return iterator;
51 }
52 
53 Structure* JSArrayIterator::createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
54 {
55     return Structure::create(vm, globalObject, prototype, TypeInfo(JSArrayIteratorType, StructureFlags), info());
56 }
57 
58 JSArrayIterator::JSArrayIterator(VM& vm, Structure* structure)
59     : Base(vm, structure)
60 {
61 }
62 
63 void JSArrayIterator::finishCreation(VM& vm)
64 {
65     Base::finishCreation(vm);
66     auto values = initialValues();
67     for (unsigned index = 0; index < values.size(); ++index)
68         Base::internalField(index).set(vm, this, values[index]);
69 }
70 
71 void JSArrayIterator::visitChildren(JSCell* cell, SlotVisitor& visitor)
72 {
73     auto* thisObject = jsCast<JSArrayIterator*>(cell);
74     ASSERT_GC_OBJECT_INHERITS(thisObject, info());
75     Base::visitChildren(thisObject, visitor);
76 }
