 33 
 34 #if ENABLE(WEB_RTC)
 35 
 36 #include "RTCIceTransport.h"
 37 #include "RTCRtpReceiver.h"
 38 #include "RTCRtpSender.h"
 39 #include "RTCRtpTransceiverBackend.h"
 40 #include "RTCRtpTransceiverDirection.h"
 41 #include "ScriptWrappable.h"
 42 #include <wtf/RefCounted.h>
 43 #include <wtf/RefPtr.h>
 44 #include <wtf/text/WTFString.h>
 45 
 46 namespace WebCore {
 47 
 48 struct RTCRtpCodecCapability;
 49 
 50 class RTCRtpTransceiver final : public RefCounted<RTCRtpTransceiver>, public ScriptWrappable {
 51     WTF_MAKE_ISO_ALLOCATED(RTCRtpTransceiver);
 52 public:
 53     static Ref<RTCRtpTransceiver> create(Ref<RTCRtpSender>&& sender, Ref<RTCRtpReceiver>&& receiver, std::unique_ptr<RTCRtpTransceiverBackend>&& backend) { return adoptRef(*new RTCRtpTransceiver(WTFMove(sender), WTFMove(receiver), WTFMove(backend))); }
 54     virtual ~RTCRtpTransceiver() = default;
 55 
 56     bool hasSendingDirection() const;
 57     void enableSendingDirection();
 58     void disableSendingDirection();
 59 
 60     RTCRtpTransceiverDirection direction() const;
 61     Optional<RTCRtpTransceiverDirection> currentDirection() const;
 62     void setDirection(RTCRtpTransceiverDirection);
 63     String mid() const;
 64 
 65     RTCRtpSender& sender() { return m_sender.get(); }
 66     RTCRtpReceiver& receiver() { return m_receiver.get(); }
 67 
 68     bool stopped() const;
 69     void stop();
 70     ExceptionOr<void> setCodecPreferences(const Vector<RTCRtpCodecCapability>&);
 71 
 72     // FIXME: Temporary solution to keep track of ICE states for this transceiver. Later, each
 73     // sender and receiver will have up to two DTLS transports, which in turn will have an ICE
 74     // transport each.
 75     RTCIceTransport& iceTransport() { return m_iceTransport.get(); }
 76 
 77     RTCRtpTransceiverBackend* backend() { return m_backend.get(); }
 78 
 79 private:
 80     RTCRtpTransceiver(Ref<RTCRtpSender>&&, Ref<RTCRtpReceiver>&&, std::unique_ptr<RTCRtpTransceiverBackend>&&);
 81 
 82     RTCRtpTransceiverDirection m_direction;
 83 
 84     Ref<RTCRtpSender> m_sender;
 85     Ref<RTCRtpReceiver> m_receiver;
 86 
 87     bool m_stopped { false };
 88 
 89     Ref<RTCIceTransport> m_iceTransport;
 90     std::unique_ptr<RTCRtpTransceiverBackend> m_backend;
 91 };
 92 
 93 class RtpTransceiverSet {
 94 public:
 95     const Vector<RefPtr<RTCRtpTransceiver>>& list() const { return m_transceivers; }
 96     void append(Ref<RTCRtpTransceiver>&&);
 97 
 98     Vector<std::reference_wrapper<RTCRtpSender>> senders() const;
 99     Vector<std::reference_wrapper<RTCRtpReceiver>> receivers() const;
100 
101 private:
102     Vector<RefPtr<RTCRtpTransceiver>> m_transceivers;
103 };
104 
105 } // namespace WebCore

107 #endif // ENABLE(WEB_RTC)
