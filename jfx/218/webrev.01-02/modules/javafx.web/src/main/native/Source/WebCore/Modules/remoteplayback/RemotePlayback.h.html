 27 
 28 #if ENABLE(WIRELESS_PLAYBACK_TARGET)
 29 
 30 #include "ActiveDOMObject.h"
 31 #include "EventTarget.h"
 32 #include "GenericEventQueue.h"
 33 #include <wtf/HashMap.h>
 34 #include <wtf/Ref.h>
 35 #include <wtf/RefCounted.h>
 36 
 37 namespace WebCore {
 38 
 39 class DeferredPromise;
 40 class HTMLMediaElement;
 41 class MediaPlaybackTarget;
 42 class RemotePlaybackAvailabilityCallback;
 43 
 44 class RemotePlayback final : public RefCounted<RemotePlayback>, public ActiveDOMObject, public EventTargetWithInlineData {
 45     WTF_MAKE_ISO_ALLOCATED(RemotePlayback);
 46 public:
 47     static Ref<RemotePlayback> create(HTMLMediaElement&);
 48     ~RemotePlayback();
 49 
 50     void watchAvailability(Ref<RemotePlaybackAvailabilityCallback>&&, Ref<DeferredPromise>&&);
 51     void cancelWatchAvailability(Optional<int32_t> id, Ref<DeferredPromise>&&);
 52     void prompt(Ref<DeferredPromise>&&);
 53 
 54     bool hasAvailabilityCallbacks() const;
 55     void availabilityChanged(bool);
 56     void playbackTargetPickerWasDismissed();
 57     void shouldPlayToRemoteTargetChanged(bool);
 58     void isPlayingToRemoteTargetChanged(bool);
 59 
 60     enum class State {
 61         Connecting,
 62         Connected,
 63         Disconnected,
 64     };
 65     State state() const { return m_state; }
 66 
 67     void invalidate();
 68 
 69     using RefCounted::ref;
 70     using RefCounted::deref;
 71 
 72 private:
 73     explicit RemotePlayback(HTMLMediaElement&);
 74 
 75     void refEventTarget() final { ref(); }
 76     void derefEventTarget() final { deref(); }
 77 
 78     void setState(State);
 79     void establishConnection();
 80     void disconnect();
 81 
 82     enum class UpdateResults {
 83         Unchanged,
 84         Changed,
 85     };
 86     UpdateResults updateAvailability();
 87 
 88     // ActiveDOMObject.
 89     const char* activeDOMObjectName() const final;
 90     void stop() final;
 91 
 92     // EventTargetWithInlineData.
 93     EventTargetInterface eventTargetInterface() const final { return RemotePlaybackEventTargetInterfaceType; }
 94     ScriptExecutionContext* scriptExecutionContext() const final { return ActiveDOMObject::scriptExecutionContext(); }
 95 
 96     WeakPtr<HTMLMediaElement> m_mediaElement;
 97     uint32_t m_nextId { 0 };
 98 
 99     using CallbackMap = HashMap<int32_t, Ref<RemotePlaybackAvailabilityCallback>>;
100     CallbackMap m_callbackMap;
101 
102     using PromiseVector = Vector<Ref<DeferredPromise>>;
103     PromiseVector m_availabilityPromises;
104     PromiseVector m_cancelAvailabilityPromises;
105     PromiseVector m_promptPromises;
106     State m_state { State::Disconnected };
107     bool m_available { false };
108 
109     UniqueRef<MainThreadGenericEventQueue> m_eventQueue;
110     GenericTaskQueue<Timer> m_taskQueue;
111 };
112 
113 }

115 #endif // ENABLE(WIRELESS_PLAYBACK_TARGET)
