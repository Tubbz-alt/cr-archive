26 #pragma once
27 
28 #include "JSCJSValue.h"
29 
30 namespace JSC {
31 
32 template<typename Adaptor>
33 typename Adaptor::Type toNativeFromValue(JSValue value)
34 {
35     if (value.isInt32())
36         return Adaptor::toNativeFromInt32(value.asInt32());
37     return Adaptor::toNativeFromDouble(value.asDouble());
38 }
39 
40 template<typename Adaptor>
41 typename Adaptor::Type toNativeFromValue(ExecState* exec, JSValue value)
42 {
43     if (value.isInt32())
44         return Adaptor::toNativeFromInt32(value.asInt32());
45     if (value.isNumber())
46         return Adaptor::toNativeFromDouble(value.asDouble());
47     return Adaptor::toNativeFromDouble(value.toNumber(exec));
48 }
49 
50 template<typename Adaptor>
51 Optional<typename Adaptor::Type> toNativeFromValueWithoutCoercion(JSValue value)
52 {
53     if (!value.isNumber())
54         return WTF::nullopt;
55     if (value.isInt32())
56         return Adaptor::toNativeFromInt32WithoutCoercion(value.asInt32());
57     return Adaptor::toNativeFromDoubleWithoutCoercion(value.asDouble());
58 }
