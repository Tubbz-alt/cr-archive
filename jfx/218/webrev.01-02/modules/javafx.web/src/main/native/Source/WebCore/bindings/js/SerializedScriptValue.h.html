140 template<class Encoder>
141 void SerializedScriptValue::encode(Encoder& encoder) const
142 {
143     encoder << m_data;
144 
145     auto hasArray = m_arrayBufferContentsArray && m_arrayBufferContentsArray->size();
146     encoder << hasArray;
147 
148     if (!hasArray)
149         return;
150 
151     encoder << static_cast<uint64_t>(m_arrayBufferContentsArray->size());
152     for (const auto &arrayBufferContents : *m_arrayBufferContentsArray) {
153         encoder << arrayBufferContents.sizeInBytes();
154         encoder.encodeFixedLengthData(static_cast<const uint8_t*>(arrayBufferContents.data()), arrayBufferContents.sizeInBytes(), 1);
155     }
156 }
157 
158 template<class Decoder>
159 RefPtr<SerializedScriptValue> SerializedScriptValue::decode(Decoder& decoder)
160 {
161     Vector<uint8_t> data;
162     if (!decoder.decode(data))
163         return nullptr;
164 
165     bool hasArray;
166     if (!decoder.decode(hasArray))
167         return nullptr;
168 
169     if (!hasArray)
170         return adoptRef(*new SerializedScriptValue(WTFMove(data)));
171 
172     uint64_t arrayLength;
173     if (!decoder.decode(arrayLength))
174         return nullptr;
175     ASSERT(arrayLength);
176 
177     auto arrayBufferContentsArray = makeUnique<ArrayBufferContentsArray>();
178     while (arrayLength--) {
179         unsigned bufferSize;
180         if (!decoder.decode(bufferSize))
181             return nullptr;
182 
183         auto buffer = Gigacage::tryMalloc(Gigacage::Primitive, bufferSize);
184         if (!decoder.decodeFixedLengthData(static_cast<uint8_t*>(buffer), bufferSize, 1)) {
185             Gigacage::free(Gigacage::Primitive, buffer);
186             return nullptr;
187         }
188         arrayBufferContentsArray->append({ buffer, bufferSize, ArrayBuffer::primitiveGigacageDestructor() });
189     }
190 
191     return adoptRef(*new SerializedScriptValue(WTFMove(data), WTFMove(arrayBufferContentsArray)));
192 }
