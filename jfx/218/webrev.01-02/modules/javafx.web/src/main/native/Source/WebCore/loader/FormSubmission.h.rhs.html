 31 #pragma once
 32 
 33 #include "FormState.h"
 34 #include "FrameLoaderTypes.h"
 35 #include <wtf/URL.h>
 36 #include <wtf/WeakPtr.h>
 37 
 38 namespace WebCore {
 39 
 40 class Event;
 41 class FormData;
 42 class FrameLoadRequest;
 43 
 44 class FormSubmission : public RefCounted<FormSubmission>, public CanMakeWeakPtr<FormSubmission> {
 45 public:
 46     enum class Method : bool { Get, Post };
 47 
 48     class Attributes {
 49     public:
 50         Method method() const { return m_method; }
 51         static Method parseMethodType(const String&);
 52         void updateMethodType(const String&);
 53         static ASCIILiteral methodString(Method method) { return method == Method::Post ? "post"_s : "get"_s; }
 54 
 55         const String& action() const { return m_action; }
 56         void parseAction(const String&);
 57 
 58         const String& target() const { return m_target; }
 59         void setTarget(const String& target) { m_target = target; }
 60 
 61         const String& encodingType() const { return m_encodingType; }
 62         static String parseEncodingType(const String&);
 63         void updateEncodingType(const String&);
 64         bool isMultiPartForm() const { return m_isMultiPartForm; }
 65 
 66         const String& acceptCharset() const { return m_acceptCharset; }
 67         void setAcceptCharset(const String& value) { m_acceptCharset = value; }
 68 
 69     private:
 70         Method m_method { Method::Get };
 71         bool m_isMultiPartForm { false };
 72         String m_action;
 73         String m_target;
 74         String m_encodingType { "application/x-www-form-urlencoded"_s };
 75         String m_acceptCharset;
 76     };
 77 
 78     static Ref<FormSubmission> create(HTMLFormElement&, const Attributes&, Event*, LockHistory, FormSubmissionTrigger);
 79 
 80     void populateFrameLoadRequest(FrameLoadRequest&);
 81     URL requestURL() const;
 82 
 83     Method method() const { return m_method; }
 84     const URL& action() const { return m_action; }
 85     const String& target() const { return m_target; }
 86     const String& contentType() const { return m_contentType; }
 87     FormState& state() const { return *m_formState; }
 88     Ref<FormState> takeState() { return m_formState.releaseNonNull(); }
 89     FormData& data() const { return m_formData; }
 90     const String boundary() const { return m_boundary; }
 91     LockHistory lockHistory() const { return m_lockHistory; }
 92     Event* event() const { return m_event.get(); }
 93     const String& referrer() const { return m_referrer; }
 94     const String& origin() const { return m_origin; }
 95 
 96     void clearTarget() { m_target = { }; }
 97     void setReferrer(const String& referrer) { m_referrer = referrer; }
 98     void setOrigin(const String& origin) { m_origin = origin; }
 99 
100     void cancel() { m_wasCancelled = true; }
101     bool wasCancelled() const { return m_wasCancelled; }
102 
103 private:
104     FormSubmission(Method, const URL& action, const String& target, const String& contentType, Ref<FormState>&&, Ref<FormData>&&, const String& boundary, LockHistory, Event*);
105 
106     // FIXME: Hold an instance of Attributes instead of individual members.
107     Method m_method;
108     bool m_wasCancelled { false };
109     URL m_action;
110     String m_target;
111     String m_contentType;
112     RefPtr<FormState> m_formState;
113     Ref<FormData> m_formData;
114     String m_boundary;
115     LockHistory m_lockHistory;
116     RefPtr<Event> m_event;
117     String m_referrer;
118     String m_origin;
119 };
120 
121 } // namespace WebCore
