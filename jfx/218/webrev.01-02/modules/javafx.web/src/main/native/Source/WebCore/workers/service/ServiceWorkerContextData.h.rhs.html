 37 #if ENABLE(SERVICE_WORKER)
 38 
 39 namespace WebCore {
 40 
 41 struct ServiceWorkerContextData {
 42     struct ImportedScript {
 43         String script;
 44         URL responseURL;
 45         String mimeType;
 46 
 47         ImportedScript isolatedCopy() const { return { script.isolatedCopy(), responseURL.isolatedCopy(), mimeType.isolatedCopy() }; }
 48 
 49         template<class Encoder> void encode(Encoder& encoder) const
 50         {
 51             encoder << script << responseURL << mimeType;
 52         }
 53 
 54         template<class Decoder> static bool decode(Decoder& decoder, ImportedScript& script)
 55         {
 56             ImportedScript importedScript;
 57             if (!decoder.decode(importedScript.script))
 58                 return false;
 59 
 60             if (!decoder.decode(importedScript.responseURL))
 61                 return false;
 62 
 63             if (!decoder.decode(importedScript.mimeType))
 64                 return false;
 65 
 66             script = WTFMove(importedScript);
 67             return true;
 68         }
 69     };
 70 
 71     Optional<ServiceWorkerJobDataIdentifier> jobDataIdentifier;
 72     ServiceWorkerRegistrationData registration;
 73     ServiceWorkerIdentifier serviceWorkerIdentifier;
 74     String script;
 75     ContentSecurityPolicyResponseHeaders contentSecurityPolicy;
 76     String referrerPolicy;
 77     URL scriptURL;
 78     WorkerType workerType;

 79     bool loadedFromDisk;
 80     HashMap<URL, ImportedScript> scriptResourceMap;
 81 
 82     template<class Encoder> void encode(Encoder&) const;
 83     template<class Decoder> static Optional<ServiceWorkerContextData> decode(Decoder&);
 84 
 85     ServiceWorkerContextData isolatedCopy() const;
 86 };
 87 
 88 template<class Encoder>
 89 void ServiceWorkerContextData::encode(Encoder& encoder) const
 90 {
 91     encoder << jobDataIdentifier << registration << serviceWorkerIdentifier << script << contentSecurityPolicy << referrerPolicy << scriptURL << workerType << loadedFromDisk;
 92     encoder << scriptResourceMap;
 93 }
 94 
 95 template<class Decoder>
 96 Optional<ServiceWorkerContextData> ServiceWorkerContextData::decode(Decoder& decoder)
 97 {
 98     Optional<Optional<ServiceWorkerJobDataIdentifier>> jobDataIdentifier;
 99     decoder >> jobDataIdentifier;
100     if (!jobDataIdentifier)
101         return WTF::nullopt;
102 
103     Optional<ServiceWorkerRegistrationData> registration;
104     decoder >> registration;
105     if (!registration)
106         return WTF::nullopt;
107 
108     auto serviceWorkerIdentifier = ServiceWorkerIdentifier::decode(decoder);
109     if (!serviceWorkerIdentifier)
110         return WTF::nullopt;
111 
112     String script;
113     if (!decoder.decode(script))
114         return WTF::nullopt;
115 
116     ContentSecurityPolicyResponseHeaders contentSecurityPolicy;
117     if (!decoder.decode(contentSecurityPolicy))
118         return WTF::nullopt;
119 
120     String referrerPolicy;
121     if (!decoder.decode(referrerPolicy))
122         return WTF::nullopt;
123 
124     URL scriptURL;
125     if (!decoder.decode(scriptURL))
126         return WTF::nullopt;
127 
128     WorkerType workerType;
129     if (!decoder.decodeEnum(workerType))
130         return WTF::nullopt;
131 





132     bool loadedFromDisk;
133     if (!decoder.decode(loadedFromDisk))
134         return WTF::nullopt;
135 
136     HashMap<URL, ImportedScript> scriptResourceMap;
137     if (!decoder.decode(scriptResourceMap))
138         return WTF::nullopt;
139 
140     return {{ WTFMove(*jobDataIdentifier), WTFMove(*registration), WTFMove(*serviceWorkerIdentifier), WTFMove(script), WTFMove(contentSecurityPolicy), WTFMove(referrerPolicy), WTFMove(scriptURL), workerType, loadedFromDisk, WTFMove(scriptResourceMap) }};
141 }
