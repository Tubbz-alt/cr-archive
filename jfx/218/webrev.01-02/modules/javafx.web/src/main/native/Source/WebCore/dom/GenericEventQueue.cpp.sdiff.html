 47 void GenericEventQueue::enqueueEvent(RefPtr<Event>&& event)
 48 {
 49     if (m_isClosed)
 50         return;
 51 
 52     if (event->target() == &m_owner)
 53         event->setTarget(nullptr);
 54 
 55     m_pendingEvents.append(WTFMove(event));
 56 
 57     if (m_isSuspended)
 58         return;
 59 
 60     m_taskQueue.enqueueTask(std::bind(&GenericEventQueue::dispatchOneEvent, this));
 61 }
 62 
 63 void GenericEventQueue::dispatchOneEvent()
 64 {
 65     ASSERT(!m_pendingEvents.isEmpty());
 66 
 67     Ref<EventTarget> protect(m_owner);
 68     RefPtr<Event> event = m_pendingEvents.takeFirst();
 69     EventTarget& target = event->target() ? *event->target() : m_owner;
 70     ASSERT_WITH_MESSAGE(!target.scriptExecutionContext()->activeDOMObjectsAreStopped(),
 71         "An attempt to dispatch an event on a stopped target by EventTargetInterface=%d (nodeName=%s target=%p owner=%p)",
 72         m_owner.eventTargetInterface(), m_owner.isNode() ? static_cast<Node&>(m_owner).nodeName().ascii().data() : "", &target, &m_owner);
 73     target.dispatchEvent(*event);
 74 }

104 void MainThreadGenericEventQueue::setPaused(bool shouldPause)
105 {
106     if (m_isPausedByClient == shouldPause)
107         return;
108 
109     m_isPausedByClient = shouldPause;
110     if (shouldPause)
111         m_taskQueue->cancelAllTasks();
112     else
113         rescheduleAllEventsIfNeeded();
114 }
115 
116 void MainThreadGenericEventQueue::suspend(ReasonForSuspension)
117 {
118     if (m_isSuspended)
119         return;
120 
121     m_isSuspended = true;
122     m_taskQueue->cancelAllTasks();
123 }
124 
125 void MainThreadGenericEventQueue::resume()
126 {
127     if (!m_isSuspended)
128         return;
129 
130     m_isSuspended = false;
131     rescheduleAllEventsIfNeeded();
132 }
133 
134 void MainThreadGenericEventQueue::rescheduleAllEventsIfNeeded()
135 {
136     if (isSuspendedOrPausedByClient())
137         return;
138 
139     for (unsigned i = 0; i < m_pendingEvents.size(); ++i)
140         m_taskQueue->enqueueTask(std::bind(&MainThreadGenericEventQueue::dispatchOneEvent, this));
141 }

143 void MainThreadGenericEventQueue::stop()
144 {
