 26 #pragma once
 27 
 28 #if ENABLE(VIDEO_TRACK) && ENABLE(MEDIA_STREAM)
 29 
 30 #include <wtf/LoggerHelper.h>
 31 
 32 namespace WTF {
 33 class MediaTime;
 34 }
 35 
 36 namespace WebCore {
 37 
 38 class AudioStreamDescription;
 39 class PlatformAudioData;
 40 
 41 class WEBCORE_EXPORT AudioMediaStreamTrackRenderer : public LoggerHelper {
 42     WTF_MAKE_FAST_ALLOCATED;
 43 public:
 44     static std::unique_ptr<AudioMediaStreamTrackRenderer> create();
 45     virtual ~AudioMediaStreamTrackRenderer() = default;
 46 
 47     virtual void start() = 0;
 48     virtual void stop() = 0;
 49     virtual void clear() = 0;
 50     // May be called on a background thread.
 51     virtual void pushSamples(const WTF::MediaTime&, const PlatformAudioData&, const AudioStreamDescription&, size_t) = 0;
 52 
 53     virtual void setMuted(bool);
 54     virtual void setVolume(float);
 55     float volume() const;
 56 
 57 #if !RELEASE_LOG_DISABLED
 58     void setLogger(const Logger&, const void*);
 59 #endif
 60 
 61     using RendererCreator = std::unique_ptr<AudioMediaStreamTrackRenderer> (*)();
 62     static void setCreator(RendererCreator);
 63 
 64 protected:
 65 #if !RELEASE_LOG_DISABLED
 66     const Logger& logger() const final;
 67     const void* logIdentifier() const final;
 68 
 69     const char* logClassName() const final;
 70     WTFLogChannel& logChannel() const final;
 71 #endif
 72 
 73     bool isMuted() const;
 74 
 75 private:
 76     static RendererCreator m_rendererCreator;
 77 
 78     // Main thread writable members
 79     bool m_muted { false };
 80     float m_volume { 1 };
 81 
 82 #if !RELEASE_LOG_DISABLED
 83     RefPtr<const Logger> m_logger;
 84     const void* m_logIdentifier;
 85 #endif
 86 };
 87 
 88 inline void AudioMediaStreamTrackRenderer::setMuted(bool value)
 89 {
 90     m_muted = value;
 91 }
 92 
 93 inline void AudioMediaStreamTrackRenderer::setVolume(float volume)
 94 {
 95     m_volume = volume;
 96 }
 97 
 98 inline float AudioMediaStreamTrackRenderer::volume() const
 99 {
100     return m_volume;
101 }
102 
103 
104 inline bool AudioMediaStreamTrackRenderer::isMuted() const
105 {
106     return m_muted;
107 }
108 
109 #if !RELEASE_LOG_DISABLED
110 inline const Logger& AudioMediaStreamTrackRenderer::logger() const
111 {
112     return *m_logger;
113 
114 }
115 
116 inline const void* AudioMediaStreamTrackRenderer::logIdentifier() const
117 {
118     return m_logIdentifier;
119 }
120 
121 inline const char* AudioMediaStreamTrackRenderer::logClassName() const
122 {
123     return "AudioMediaStreamTrackRenderer";
124 }
125 #endif
126 
127 }
128 
129 #endif // ENABLE(VIDEO_TRACK) && ENABLE(MEDIA_STREAM)
