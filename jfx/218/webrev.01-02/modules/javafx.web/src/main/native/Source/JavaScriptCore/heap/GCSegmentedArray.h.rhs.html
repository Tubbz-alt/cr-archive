 35 DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(GCSegmentedArray);

 37 template <typename T>
 38 class GCArraySegment : public DoublyLinkedListNode<GCArraySegment<T>> {
 39     friend class WTF::DoublyLinkedListNode<GCArraySegment<T>>;
 40 public:
 41     GCArraySegment()
 42         : DoublyLinkedListNode<GCArraySegment<T>>()
 43 #if ASSERT_ENABLED
 44         , m_top(0)
 45 #endif
 46     {
 47     }
 48 
 49     static GCArraySegment* create();
 50     static void destroy(GCArraySegment*);
 51 
 52     T* data()
 53     {
 54         return bitwise_cast<T*>(this + 1);
 55     }
 56 
 57     static constexpr size_t blockSize = 4 * KB;
 58 
 59     GCArraySegment* m_prev;
 60     GCArraySegment* m_next;
 61 #if ASSERT_ENABLED
 62     size_t m_top;
 63 #endif
 64 };
 65 
 66 template <typename T> class GCSegmentedArrayIterator;
 67 
 68 template <typename T>
 69 class GCSegmentedArray {
 70     WTF_MAKE_FAST_ALLOCATED;
 71     WTF_MAKE_NONCOPYABLE(GCSegmentedArray);
 72     friend class GCSegmentedArrayIterator<T>;
 73     friend class GCSegmentedArrayIterator<const T>;
 74 public:
 75     GCSegmentedArray();
 76     ~GCSegmentedArray();
 77 
 78     void append(T);
 79 
 80     bool canRemoveLast();
 81     const T removeLast();
 82     bool refill();
 83 
 84     size_t size();
 85     bool isEmpty();
 86 
 87     void fillVector(Vector<T>&);
 88     void clear();
 89 
 90     typedef GCSegmentedArrayIterator<T> iterator;
 91     iterator begin() const { return GCSegmentedArrayIterator<T>(m_segments.head(), m_top); }
 92     iterator end() const { return GCSegmentedArrayIterator<T>(); }
 93 
 94 protected:
 95     template <size_t size> struct CapacityFromSize {
 96         static constexpr size_t value = (size - sizeof(GCArraySegment<T>)) / sizeof(T);
 97     };
 98 
 99     void expand();
100 
101     size_t postIncTop();
102     size_t preDecTop();
103     void setTopForFullSegment();
104     void setTopForEmptySegment();
105     size_t top();
106 
107     void validatePrevious();
108 
109     DoublyLinkedList<GCArraySegment<T>> m_segments;
110 
111     JS_EXPORT_PRIVATE static const size_t s_segmentCapacity = CapacityFromSize<GCArraySegment<T>::blockSize>::value;
112     size_t m_top;
113     size_t m_numberOfSegments;
114 };
115 
116 template <typename T>
117 class GCSegmentedArrayIterator {
118     friend class GCSegmentedArray<T>;
119 public:
120     GCSegmentedArrayIterator()
121         : m_currentSegment(0)
122         , m_currentOffset(0)
123     {
124     }
125 
126     T& get() { return m_currentSegment->data()[m_currentOffset]; }
127     T& operator*() { return get(); }
128     T& operator->() { return get(); }
129 
130     bool operator==(const GCSegmentedArrayIterator& other)
131     {
132         return m_currentSegment == other.m_currentSegment && m_currentOffset == other.m_currentOffset;
133     }
134 
135     bool operator!=(const GCSegmentedArrayIterator& other)
136     {
137         return !(*this == other);
138     }
139 
140     GCSegmentedArrayIterator& operator++()
141     {
142         ASSERT(m_currentSegment);
143 
144         m_currentOffset++;
145 
146         if (m_currentOffset >= m_offsetLimit) {
147             m_currentSegment = m_currentSegment->next();
148             m_currentOffset = 0;
149             m_offsetLimit = GCSegmentedArray<T>::s_segmentCapacity;
150         }
151 
152         return *this;
153     }
154 
155 private:
156     GCSegmentedArrayIterator(GCArraySegment<T>* start, size_t top)
157         : m_currentSegment(start)
158         , m_currentOffset(0)
159         , m_offsetLimit(top)
160     {
161         if (!m_offsetLimit)
162             m_currentSegment = nullptr;
163     }
164 
165     GCArraySegment<T>* m_currentSegment;
166     size_t m_currentOffset;
167     size_t m_offsetLimit;
168 };
