26 #pragma once
27 
28 #include "GenericTaskQueue.h"
29 
30 namespace WebCore {
31 
32 template <typename T>
33 class DeferrableTask : public CanMakeWeakPtr<DeferrableTask<T>> {
34 public:
35     DeferrableTask()
36         : m_dispatcher()
37     {
38     }
39 
40     DeferrableTask(T& t)
41         : m_dispatcher(t)
42     {
43     }
44 
45     typedef WTF::Function<void ()> TaskFunction;
46 
47     void scheduleTask(TaskFunction&& task)
48     {
49         if (m_isClosed)
50             return;
51 
52         cancelTask();
53 
54         m_pendingTask = true;
55         m_dispatcher.postTask([weakThis = makeWeakPtr(*this), task = WTFMove(task)] {
56             if (!weakThis)
57                 return;
58             ASSERT(weakThis->m_pendingTask);
59             weakThis->m_pendingTask = false;
60             task();
61         });
62     }
63 
64     void close()
65     {
66         cancelTask();
67         m_isClosed = true;
68     }
69 
70     void cancelTask()
71     {
72         CanMakeWeakPtr<DeferrableTask<T>>::weakPtrFactory().revokeAll();
73         m_pendingTask = false;
74     }
75     bool hasPendingTask() const { return m_pendingTask; }
76 
77 private:
78     TaskDispatcher<T> m_dispatcher;
79     bool m_pendingTask { false };
80     bool m_isClosed { false };
81 };
82 
83 }
