30 class ExecState;

33 namespace Yarr {
34 
35 enum class ErrorCode : uint8_t {
36     NoError = 0,
37     PatternTooLarge,
38     QuantifierOutOfOrder,
39     QuantifierWithoutAtom,
40     QuantifierTooLarge,
41     MissingParentheses,
42     ParenthesesUnmatched,
43     ParenthesesTypeInvalid,
44     InvalidGroupName,
45     DuplicateGroupName,
46     CharacterClassUnmatched,
47     CharacterClassOutOfOrder,
48     EscapeUnterminated,
49     InvalidUnicodeEscape,
50     InvalidBackreference,
51     InvalidIdentityEscape,
52     InvalidUnicodePropertyExpression,
53     TooManyDisjunctions,
54     OffsetTooLarge,
55     InvalidRegularExpressionFlags,
56 };
57 
58 JS_EXPORT_PRIVATE const char* errorMessage(ErrorCode);
59 inline bool hasError(ErrorCode errorCode)
60 {
61     return errorCode != ErrorCode::NoError;
62 }
63 
64 inline bool hasHardError(ErrorCode errorCode)
65 {
66     // TooManyDisjunctions means that we ran out stack compiling.
67     // All other errors are due to problems in the expression.
68     return hasError(errorCode) && errorCode != ErrorCode::TooManyDisjunctions;
69 }
70 JS_EXPORT_PRIVATE JSObject* errorToThrow(ExecState*, ErrorCode);
71 
72 } } // namespace JSC::Yarr
