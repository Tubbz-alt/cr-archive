35 ExitFlag hasBadCacheExitSite(CodeBlock* profiledBlock, unsigned bytecodeIndex)
36 {
37 #if ENABLE(DFG_JIT)
38     UnlinkedCodeBlock* unlinkedCodeBlock = profiledBlock->unlinkedCodeBlock();
39     ConcurrentJSLocker locker(unlinkedCodeBlock->m_lock);
40     auto exitFlag = [&] (ExitKind exitKind) -> ExitFlag {
41         auto withInlined = [&] (ExitingInlineKind inlineKind) -> ExitFlag {
42             return ExitFlag(unlinkedCodeBlock->hasExitSite(locker, DFG::FrequentExitSite(bytecodeIndex, exitKind, ExitFromAnything, inlineKind)), inlineKind);
43         };
44         return withInlined(ExitFromNotInlined) | withInlined(ExitFromInlined);
45     };
46     return exitFlag(BadCache) | exitFlag(BadConstantCache) | exitFlag(BadType);
47 #else
48     UNUSED_PARAM(profiledBlock);
49     UNUSED_PARAM(bytecodeIndex);
50     return ExitFlag();
51 #endif
52 }
