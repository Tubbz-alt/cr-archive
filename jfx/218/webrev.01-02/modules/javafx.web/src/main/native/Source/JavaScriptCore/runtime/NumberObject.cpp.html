31 STATIC_ASSERT_IS_TRIVIALLY_DESTRUCTIBLE(NumberObject);

33 const ClassInfo NumberObject::s_info = { "Number", &Base::s_info, nullptr, nullptr, CREATE_METHOD_TABLE(NumberObject) };
34 
35 NumberObject::NumberObject(VM& vm, Structure* structure)
36     : Base(vm, structure)
37 {
38 }
39 
40 void NumberObject::finishCreation(VM& vm)
41 {
42     Base::finishCreation(vm);
43     ASSERT(inherits(vm, info()));
44     ASSERT(type() == NumberObjectType);
45 }
46 
47 NumberObject* constructNumber(JSGlobalObject* globalObject, JSValue number)
48 {
49     NumberObject* object = NumberObject::create(globalObject->vm(), globalObject->numberObjectStructure());
50     object->setInternalValue(globalObject->vm(), number);
51     return object;
52 }
