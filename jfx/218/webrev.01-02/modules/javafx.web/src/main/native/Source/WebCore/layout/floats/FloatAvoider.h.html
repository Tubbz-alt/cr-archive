37 namespace WebCore {

39 namespace Display {
40 class Rect;

42 
43 namespace Layout {
44 
45 class FloatAvoider {
46     WTF_MAKE_ISO_ALLOCATED(FloatAvoider);
47 public:
48     FloatAvoider(const Box&, Display::Box absoluteDisplayBox, LayoutPoint containingBlockAbsoluteTopLeft, HorizontalEdges containingBlockAbsoluteContentBox);
49     virtual ~FloatAvoider() = default;
50 
51     virtual Display::Rect rect() const { return m_absoluteDisplayBox.rect(); }
52     Display::Rect rectInContainingBlock() const;
53 
54     struct HorizontalConstraints {
55         Optional<PositionInContextRoot> left;
56         Optional<PositionInContextRoot> right;
57     };
58     void setHorizontalConstraints(HorizontalConstraints);
59     void setVerticalConstraint(PositionInContextRoot);
60 
61     bool overflowsContainingBlock() const;
62 
63 protected:
64     virtual bool isLeftAligned() const { return layoutBox().style().isLeftToRightDirection(); }
65     PositionInContextRoot initialHorizontalPosition() const;
66 
67     void resetHorizontalConstraints();
68 
69     virtual PositionInContextRoot horizontalPositionCandidate(HorizontalConstraints);
70     virtual PositionInContextRoot verticalPositionCandidate(PositionInContextRoot);
71 
72     LayoutUnit marginBefore() const { return displayBox().marginBefore(); }
73     LayoutUnit marginAfter() const { return displayBox().marginAfter(); }
74     // Do not use the used values here because they computed as if this box was not a float avoider.
75     LayoutUnit marginStart() const { return displayBox().computedMarginStart().valueOr(0); }
76     LayoutUnit marginEnd() const { return displayBox().computedMarginEnd().valueOr(0); }
77 
78     LayoutUnit marginBoxWidth() const { return marginStart() + displayBox().width() + marginEnd(); }
79 
80     const Box& layoutBox() const { return *m_layoutBox; }
81     const Display::Box& displayBox() const { return m_absoluteDisplayBox; }
82     Display::Box& displayBox() { return m_absoluteDisplayBox; }
83 
84 private:
85     WeakPtr<const Box> m_layoutBox;
86     // These coordinate values are relative to the formatting root's border box.
87     Display::Box m_absoluteDisplayBox;
88     LayoutPoint m_containingBlockAbsoluteTopLeft;
89     HorizontalEdges m_containingBlockAbsoluteContentBox;
90 };
91 
92 }
93 }
