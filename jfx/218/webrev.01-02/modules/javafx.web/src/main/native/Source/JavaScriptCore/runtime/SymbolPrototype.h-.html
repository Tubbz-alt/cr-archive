32 namespace JSC {

34 // In the ES6 spec, Symbol.prototype object is an ordinary JS object, not one of the symbol wrapper object instance.
35 class SymbolPrototype final : public JSNonFinalObject {
36 public:
37     typedef JSNonFinalObject Base;
38     static const unsigned StructureFlags = Base::StructureFlags | HasStaticPropertyTable;
39 
40     static SymbolPrototype* create(VM& vm, JSGlobalObject* globalObject, Structure* structure)
41     {
42         SymbolPrototype* prototype = new (NotNull, allocateCell<SymbolPrototype>(vm.heap)) SymbolPrototype(vm, structure);
43         prototype->finishCreation(vm, globalObject);
44         return prototype;
45     }
46 
47     DECLARE_INFO;
48 
49     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
50     {
51         return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
52     }
53 
54 protected:
55     SymbolPrototype(VM&, Structure*);
56     void finishCreation(VM&, JSGlobalObject*);
57 };
58 STATIC_ASSERT_IS_TRIVIALLY_DESTRUCTIBLE(SymbolPrototype);

60 } // namespace JSC
