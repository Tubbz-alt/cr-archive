55 StorageEvent::StorageEvent(const AtomString& type, const String& key, const String& oldValue, const String& newValue, const String& url, Storage* storageArea)
56     : Event(type, CanBubble::No, IsCancelable::No)
57     , m_key(key)
58     , m_oldValue(oldValue)
59     , m_newValue(newValue)
60     , m_url(url)
61     , m_storageArea(storageArea)
62 {
63 }
64 
65 StorageEvent::StorageEvent(const AtomString& type, const Init& initializer, IsTrusted isTrusted)
66     : Event(type, initializer, isTrusted)
67     , m_key(initializer.key)
68     , m_oldValue(initializer.oldValue)
69     , m_newValue(initializer.newValue)
70     , m_url(initializer.url)
71     , m_storageArea(initializer.storageArea)
72 {
73 }
74 
75 void StorageEvent::initStorageEvent(const AtomString& type, bool canBubble, bool cancelable, const String& key, const String& oldValue, const String& newValue, const String& url, Storage* storageArea)
76 {
77     if (isBeingDispatched())
78         return;
79 
80     initEvent(type, canBubble, cancelable);
81 
82     m_key = key;
83     m_oldValue = oldValue;
84     m_newValue = newValue;
85     m_url = url;
86     m_storageArea = storageArea;
87 }
88 
89 EventInterface StorageEvent::eventInterface() const
90 {
91     return StorageEventInterfaceType;
92 }
