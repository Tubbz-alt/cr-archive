 80 class WebGPUDevice : public RefCounted<WebGPUDevice>, public EventTargetWithInlineData, public CanMakeWeakPtr<WebGPUDevice> {
 81     WTF_MAKE_ISO_ALLOCATED(WebGPUDevice);
 82 public:
 83     virtual ~WebGPUDevice();
 84 
 85     static RefPtr<WebGPUDevice> tryCreate(ScriptExecutionContext&, Ref<const WebGPUAdapter>&&);
 86 
 87     static HashSet<WebGPUDevice*>& instances(const LockHolder&);
 88     static Lock& instancesMutex();
 89 
 90     const WebGPUAdapter& adapter() const { return m_adapter.get(); }
 91     GPUDevice& device() { return m_device.get(); }
 92     const GPUDevice& device() const { return m_device.get(); }
 93 
 94     Ref<WebGPUBuffer> createBuffer(const GPUBufferDescriptor&) const;
 95     Vector<JSC::JSValue> createBufferMapped(JSC::JSGlobalObject&, const GPUBufferDescriptor&) const;
 96     Ref<WebGPUTexture> createTexture(const GPUTextureDescriptor&) const;
 97     Ref<WebGPUSampler> createSampler(const GPUSamplerDescriptor&) const;
 98 
 99     Ref<WebGPUBindGroupLayout> createBindGroupLayout(const GPUBindGroupLayoutDescriptor&) const;
100     Ref<WebGPUPipelineLayout> createPipelineLayout(const WebGPUPipelineLayoutDescriptor&) const;
101     Ref<WebGPUBindGroup> createBindGroup(const WebGPUBindGroupDescriptor&) const;
102 
103     Ref<WebGPUShaderModule> createShaderModule(const WebGPUShaderModuleDescriptor&) const;
104     Ref<WebGPURenderPipeline> createRenderPipeline(const WebGPURenderPipelineDescriptor&);
105     Ref<WebGPUComputePipeline> createComputePipeline(const WebGPUComputePipelineDescriptor&);
106 
107     Ref<WebGPUCommandEncoder> createCommandEncoder() const;
108 
109     Ref<WebGPUQueue> getQueue() const;
110 
111     void pushErrorScope(GPUErrorFilter filter) { m_errorScopes->pushErrorScope(filter); }
112     void popErrorScope(ErrorPromise&&);
113 
114     ScriptExecutionContext* scriptExecutionContext() const final { return &m_scriptExecutionContext; }
115 
116     using RefCounted::ref;
117     using RefCounted::deref;
118 
119 private:
120     WebGPUDevice(ScriptExecutionContext&, Ref<const WebGPUAdapter>&&, Ref<GPUDevice>&&);
121 
122     // EventTarget
123     EventTargetInterface eventTargetInterface() const final { return WebGPUDeviceEventTargetInterfaceType; }
124     void refEventTarget() final { ref(); }
125     void derefEventTarget() final { deref(); }
126 
127     void dispatchUncapturedError(GPUError&&);
128 
129     ScriptExecutionContext& m_scriptExecutionContext;
130 
131     Ref<const WebGPUAdapter> m_adapter;
132     Ref<GPUDevice> m_device;
133     mutable RefPtr<WebGPUQueue> m_queue;
134 
135     Ref<GPUErrorScopes> m_errorScopes;
136 };
