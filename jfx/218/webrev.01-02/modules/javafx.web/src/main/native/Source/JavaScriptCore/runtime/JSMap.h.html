26 #pragma once
27 
28 #include "HashMapImpl.h"
29 #include "JSObject.h"
30 
31 namespace JSC {
32 
33 class JSMap final : public HashMapImpl<HashMapBucket<HashMapBucketDataKeyValue>> {
34     using Base = HashMapImpl<HashMapBucket<HashMapBucketDataKeyValue>>;
35 public:
36 
37     DECLARE_EXPORT_INFO;
38 
39     template<typename CellType, SubspaceAccess mode>
40     static IsoSubspace* subspaceFor(VM& vm)
41     {
42         return vm.mapSpace<mode>();
43     }
44 
45     static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
46     {
47         return Structure::create(vm, globalObject, prototype, TypeInfo(JSMapType, StructureFlags), info());
48     }
49 
50     static JSMap* create(JSGlobalObject* globalObject, VM& vm, Structure* structure)
51     {
52         JSMap* instance = new (NotNull, allocateCell<JSMap>(vm.heap)) JSMap(vm, structure);
53         instance->finishCreation(globalObject, vm);
54         return instance;
55     }
56 
57     ALWAYS_INLINE void set(JSGlobalObject* globalObject, JSValue key, JSValue value)
58     {
59         add(globalObject, key, value);
60     }
61 
62     bool isIteratorProtocolFastAndNonObservable();
63     bool canCloneFastAndNonObservable(Structure*);
64     JSMap* clone(JSGlobalObject*, VM&, Structure*);
65 
66 private:
67     JSMap(VM& vm, Structure* structure)
68         : Base(vm, structure)
69     {
70     }
71 
72     static String toStringName(const JSObject*, JSGlobalObject*);
73 };
74 
75 static_assert(std::is_final<JSMap>::value, "Required for JSType based casting");
76 
77 } // namespace JSC
