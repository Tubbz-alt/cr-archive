34 inline bool JSArrayBufferView::isShared()
35 {
36     switch (m_mode) {
37     case WastefulTypedArray:
38         return existingBufferInButterfly()->isShared();
39     case DataViewMode:
40         return jsCast<JSDataView*>(this)->possiblySharedBuffer()->isShared();
41     default:
42         return false;
43     }
44 }
45 
46 inline ArrayBuffer* JSArrayBufferView::possiblySharedBuffer()

47 {



48     switch (m_mode) {
49     case WastefulTypedArray:
50         return existingBufferInButterfly();
51     case DataViewMode:
52         return jsCast<JSDataView*>(this)->possiblySharedBuffer();
53     case FastTypedArray:
54     case OversizeTypedArray:
55         return slowDownAndWasteMemory();
56     }
57     ASSERT_NOT_REACHED();
58     return nullptr;
59 }
60 





61 inline ArrayBuffer* JSArrayBufferView::existingBufferInButterfly()
62 {
63     ASSERT(m_mode == WastefulTypedArray);
64     return butterfly()->indexingHeader()->arrayBuffer();
65 }
66 
67 inline RefPtr<ArrayBufferView> JSArrayBufferView::unsharedImpl()
68 {
69     RefPtr<ArrayBufferView> result = possiblySharedImpl();
70     RELEASE_ASSERT(!result->isShared());
71     return result;
72 }
73 
74 inline unsigned JSArrayBufferView::byteOffset()

75 {
76     if (!hasArrayBuffer())
77         return 0;
78 
79     ArrayBuffer* buffer = possiblySharedBuffer();
80     ASSERT(!vector() == !buffer->data());






81 
82     ptrdiff_t delta =
83         bitwise_cast<uint8_t*>(vector()) - static_cast<uint8_t*>(buffer->data());
84 
85     unsigned result = static_cast<unsigned>(delta);
86     ASSERT(static_cast<ptrdiff_t>(result) == delta);






87     return result;
88 }
89 










90 inline RefPtr<ArrayBufferView> JSArrayBufferView::toWrapped(VM& vm, JSValue value)
91 {
92     if (JSArrayBufferView* view = jsDynamicCast<JSArrayBufferView*>(vm, value)) {
93         if (!view->isShared())
94             return view->unsharedImpl();
95     }
96     return nullptr;
97 }
