42 class ScriptFunctionCall;

45 namespace Inspector {
46 
47 typedef String ErrorString;
48 typedef WTF::Function<void(ErrorString&, RefPtr<Protocol::Runtime::RemoteObject>&&, Optional<bool>&, Optional<int>&)> AsyncCallCallback;
49 
50 class JS_EXPORT_PRIVATE InjectedScriptBase {
51 public:
52     virtual ~InjectedScriptBase();
53 
54     const String& name() const { return m_name; }
55     bool hasNoValue() const { return m_injectedScriptObject.hasNoValue(); }
56     JSC::ExecState* scriptState() const { return m_injectedScriptObject.scriptState(); }
57 
58 protected:
59     InjectedScriptBase(const String& name);
60     InjectedScriptBase(const String& name, Deprecated::ScriptObject, InspectorEnvironment*);
61 
62     InspectorEnvironment* inspectorEnvironment() const { return m_environment; }
63 
64     bool hasAccessToInspectedScriptState() const;
65 
66     const Deprecated::ScriptObject& injectedScriptObject() const;
67     JSC::JSValue callFunctionWithEvalEnabled(Deprecated::ScriptFunctionCall&, bool& hadException) const;
68     Ref<JSON::Value> makeCall(Deprecated::ScriptFunctionCall&);
69     void makeEvalCall(ErrorString&, Deprecated::ScriptFunctionCall&, RefPtr<Protocol::Runtime::RemoteObject>& resultObject, Optional<bool>& wasThrown, Optional<int>& savedResultIndex);
70     void makeAsyncCall(Deprecated::ScriptFunctionCall&, AsyncCallCallback&&);
71 
72 private:
73     void checkCallResult(ErrorString&, RefPtr<JSON::Value> result, RefPtr<Protocol::Runtime::RemoteObject>& resultObject, Optional<bool>& wasThrown, Optional<int>& savedResultIndex);
74     void checkAsyncCallResult(RefPtr<JSON::Value> result, const AsyncCallCallback&);
75 
76     String m_name;
77     Deprecated::ScriptObject m_injectedScriptObject;
78     InspectorEnvironment* m_environment { nullptr };
79 };
