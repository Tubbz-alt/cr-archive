34 // This is a special that recognizes that there are two uses of Patchpoint: Void and and non-Void.
35 // In the Void case, the syntax of the Air Patch instruction is:
36 //
37 //     Patch &patchpoint, args...
38 //
39 // Where "args..." are the lowered arguments to the Patchpoint instruction. In the non-Void case
40 // we will have:
41 //
42 //     Patch &patchpoint, result, args...
43 
44 class PatchpointSpecial : public StackmapSpecial {
45 public:
46     JS_EXPORT_PRIVATE PatchpointSpecial();
47     virtual ~PatchpointSpecial();
48 
49 protected:
50     void forEachArg(Air::Inst&, const ScopedLambda<Air::Inst::EachArgCallback>&) final;
51     bool isValid(Air::Inst&) final;
52     bool admitsStack(Air::Inst&, unsigned argIndex) final;
53     bool admitsExtendedOffsetAddr(Air::Inst&, unsigned) final;
54 
55     // NOTE: the generate method will generate the hidden branch and then register a LatePath that
56     // generates the stackmap. Super crazy dude!
57 
58     CCallHelpers::Jump generate(Air::Inst&, CCallHelpers&, Air::GenerationContext&) final;
59 
60     bool isTerminal(Air::Inst&) final;
61 
62     void dumpImpl(PrintStream&) const final;
63     void deepDumpImpl(PrintStream&) const final;
64 };
65 
66 } } // namespace JSC::B3
