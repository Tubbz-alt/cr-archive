26 #ifndef PATHCONSUMER_H
27 #define PATHCONSUMER_H
28 
29 #ifdef __cplusplus
30 extern "C" {
31 #endif
32 
33 #include <jni.h>
34 





35 typedef struct _PathConsumer PathConsumer;
36 
37 typedef void MoveToFunc(PathConsumer *pConsumer,
38                         jfloat x0, jfloat y0);
39 typedef void LineToFunc(PathConsumer *pConsumer,
40                         jfloat x1, jfloat y1);
41 typedef void QuadToFunc(PathConsumer *pConsumer,
42                         jfloat xc, jfloat yc,
43                         jfloat x1, jfloat y1);
44 typedef void CurveToFunc(PathConsumer *pConsumer,
45                          jfloat xc0, jfloat yc0,
46                          jfloat xc1, jfloat yc1,
47                          jfloat x1, jfloat y1);
48 typedef void ClosePathFunc(PathConsumer *pConsumer);
49 typedef void PathDoneFunc(PathConsumer *pConsumer);
50 
51 struct _PathConsumer {
52     MoveToFunc       *moveTo;
53     LineToFunc       *lineTo;
54     QuadToFunc       *quadTo;
55     CurveToFunc      *curveTo;
56     ClosePathFunc    *closePath;
57     PathDoneFunc     *pathDone;
58 };
59 
60 extern void PathConsumer_init(PathConsumer *pConsumer,
61                               MoveToFunc       *moveTo,
62                               LineToFunc       *lineTo,
63                               QuadToFunc       *quadTo,
64                               CurveToFunc      *curveTo,
65                               ClosePathFunc    *closePath,
66                               PathDoneFunc     *pathDone);
67 
68 #ifdef __cplusplus
69 }
70 #endif
71 
72 #endif /* PATHCONSUMER_H */
