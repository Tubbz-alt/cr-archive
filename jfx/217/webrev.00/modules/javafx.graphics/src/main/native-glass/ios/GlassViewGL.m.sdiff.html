129     int code = com_sun_glass_events_KeyEvent_VK_UNDEFINED;
130     if (unicode == com_sun_glass_events_KeyEvent_VK_ENTER) {
131          code = unicode;
132     }
133     [self->delegate sendJavaKeyEventWithType:com_sun_glass_events_KeyEvent_PRESS
