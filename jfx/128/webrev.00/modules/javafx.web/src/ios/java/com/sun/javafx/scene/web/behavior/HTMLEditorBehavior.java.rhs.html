26 package com.sun.javafx.scene.web.behavior;
27 
28 import com.sun.javafx.scene.ParentHelper;
29 import javafx.scene.web.HTMLEditor;
30 import java.util.ArrayList;
31 import java.util.List;
32 import com.sun.javafx.scene.control.behavior.BehaviorBase;
33 import com.sun.javafx.scene.control.inputmap.InputMap;
34 import com.sun.javafx.scene.control.inputmap.KeyBinding;
35 import com.sun.javafx.scene.control.behavior.FocusTraversalInputMap;
36 
37 import static com.sun.javafx.scene.control.inputmap.InputMap.KeyMapping;
38 import javafx.scene.web.HTMLEditorSkin;
39 import static javafx.scene.input.KeyCode.B;
40 import static javafx.scene.input.KeyCode.I;
41 import static javafx.scene.input.KeyCode.U;
42 import static javafx.scene.input.KeyCode.F12;
43 import static javafx.scene.input.KeyCode.TAB;
44 
45 
46 /**
47  * HTML editor behavior.
48  */
49 public class HTMLEditorBehavior extends BehaviorBase<HTMLEditor> {
50     private final InputMap<HTMLEditor> inputMap;
51 
52     public HTMLEditorBehavior(HTMLEditor htmlEditor) {
53         super(htmlEditor);
54 
55         this.inputMap = createInputMap();
56         addDefaultMapping(inputMap,
57                 new KeyMapping(new KeyBinding(B).shortcut(), e -> keyboardShortcuts(HTMLEditorSkin.Command.BOLD)),
58                 new KeyMapping(new KeyBinding(I).shortcut(), e -> keyboardShortcuts(HTMLEditorSkin.Command.ITALIC)),
59                 new KeyMapping(new KeyBinding(U).shortcut(), e -> keyboardShortcuts(HTMLEditorSkin.Command.UNDERLINE)),
60 
61                 new KeyMapping(new KeyBinding(F12), e -> ParentHelper.getTraversalEngine(getNode()).selectFirst().requestFocus()),
62                 new KeyMapping(new KeyBinding(TAB).ctrl(), FocusTraversalInputMap::traverseNext),
63                 new KeyMapping(new KeyBinding(TAB).ctrl().shift(), FocusTraversalInputMap::traversePrevious)
64         );
65     }
66 
67     @Override public InputMap<HTMLEditor> getInputMap() {
68         return inputMap;
69     }
70 
71     private void keyboardShortcuts(HTMLEditorSkin.Command command) {
72         HTMLEditor editor = getNode();
73         HTMLEditorSkin editorSkin = (HTMLEditorSkin)editor.getSkin();
74         editorSkin.performCommand(command);
75     }
76 }
