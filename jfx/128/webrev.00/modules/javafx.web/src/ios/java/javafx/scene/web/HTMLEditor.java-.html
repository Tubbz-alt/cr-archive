26 package javafx.scene.web;
27 
28 
29 import javafx.css.StyleableProperty;
30 
31 import javafx.scene.control.Control;
32 
33 import javafx.scene.control.Skin;
34 
35 
36 /**
37  * A control that allows for users to edit text, and apply styling to this text.
38  * The underlying data model is HTML, although this is not shown visually to the
39  * end-user.
40  * @since JavaFX 2.0
41  */
42 public class HTMLEditor extends Control {
43 
44     /**
45      * Creates a new instance of the HTMLEditor control.
46      */
47     public HTMLEditor() {
48         ((StyleableProperty)super.skinClassNameProperty()).applyStyle(
49             null,
50             "com.sun.javafx.scene.web.skin.HTMLEditorSkin"
51         );
52         getStyleClass().add("html-editor");
53     }
54 
55     @Override protected Skin<?> createDefaultSkin() {
56         return new HTMLEditorSkin(this);
57     }
58 
59     /**
60      * Returns the HTML content of the editor.
61      */
62     public String getHtmlText() {
63         return ((HTMLEditorSkin)getSkin()).getHTMLText();
64     }
65 
66     /**
67      * Sets the HTML content of the editor. Note that if the contentEditable
68      * property on the <body> tag of the provided HTML is not set to true, the
69      * HTMLEditor will become read-only. You can ensure that the text remains
70      * editable by ensuring the body appears as such:
71      * <code>
72      * &lt;body contentEditable="true"&gt;
73      * </code>
74      *
75      * @param htmlText The full HTML markup to put into the editor. This should
76      *      include all normal HTML elements, starting with
77      *      <code>&lt;html&gt;</code>, and including a <code>&lt;body&gt;</code>.
78      */
79     public void setHtmlText(String htmlText) {
80         ((HTMLEditorSkin)getSkin()).setHTMLText(htmlText);
81     }
82 }
