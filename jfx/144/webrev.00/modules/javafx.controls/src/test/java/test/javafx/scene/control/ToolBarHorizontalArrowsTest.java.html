 53 /**
 54  * Test basic horizontal navigation mappings for ToolBar.
 55  * It is parameterized on NodeOrientation
 56  */
 57 @RunWith(Parameterized.class)
 58 public class ToolBarHorizontalArrowsTest {
 59     @Parameterized.Parameters
 60     public static Collection<?> implementations() {
 61         return Arrays.asList(new Object[][] {
 62             {NodeOrientation.LEFT_TO_RIGHT},
 63             {NodeOrientation.RIGHT_TO_LEFT}
 64         });
 65     }
 66 
 67     private ToolBar toolBar;
 68     private Button btn1;
 69     private Button btn2;
 70     private Button btn3;
 71     private Button btn4;
 72     private Button btn5;
 73 
 74     private KeyEventFirer keyboard;
 75     private StageLoader stageLoader;
 76     private NodeOrientation orientation;
 77 
 78     public ToolBarHorizontalArrowsTest(NodeOrientation val) {
 79         orientation = val;
 80     }
 81 
 82     @Before public void setup() {
 83         toolBar = new ToolBar();
 84         toolBar.setNodeOrientation(orientation);
 85 
 86         // Create 5 buttons and add them in order to the toolBar
 87         btn1 = new Button("Btn1");
 88         btn2 = new Button("Btn2");
 89         btn3 = new Button("Btn3");
 90         btn4 = new Button("Btn4");
 91         btn5 = new Button("Btn5");
 92 
 93         toolBar.getItems().addAll(btn1, btn2, btn3, btn4, btn5);
 94 
 95         ToolBarSkin toolbarSkin = new ToolBarSkin(toolBar);
 96         toolBar.setSkin(toolbarSkin);
 97 
 98         stageLoader = new StageLoader(toolBar);
 99         stageLoader.getStage().show();
100         ((StubToolkit)Toolkit.getToolkit()).firePulse();
101 
102         toolBar.setFocusTraversable(true);
103 
104         keyboard = new KeyEventFirer(toolBar, toolBar.getScene());
105     }
106 
107     @After
108     public void tearDown() {
109         toolBar.getSkin().dispose();
110         stageLoader.dispose();
111     }
112 
113     // ---------------- Helper methods -------------------------
114     /**
115      * Toggles the parameter nodeOrientation and
116      * sets the toolBar's orientation to the new toggled value
117      */
118     private void toggleNodeOrientation() {
119         orientation = (orientation == NodeOrientation.LEFT_TO_RIGHT?
120             NodeOrientation.RIGHT_TO_LEFT : NodeOrientation.LEFT_TO_RIGHT);
121         toolBar.setNodeOrientation(orientation);
122     }
123 
124     /**
125      * Orientation-aware forward horizontal navigation with arrow keys.
126      * @param modifiers the modifiers to use on keyboard
127      */
128     private void forward(KeyModifier... modifiers) {
129         if (orientation == NodeOrientation.LEFT_TO_RIGHT) {
130             keyboard.doRightArrowPress(modifiers);
131         } else if (orientation == NodeOrientation.RIGHT_TO_LEFT) {
132             keyboard.doLeftArrowPress(modifiers);
133         }
134     }
135 
136     /**
137      * Orientation-aware backward horizontal navigation with arrow keys.
138      * @param modifiers the modifiers to use on keyboard
139      */
140     private void backward(KeyModifier... modifiers) {
141         if (orientation == NodeOrientation.LEFT_TO_RIGHT) {
142             keyboard.doLeftArrowPress(modifiers);
143         } else if (orientation == NodeOrientation.RIGHT_TO_LEFT) {
144             keyboard.doRightArrowPress(modifiers);
145         }
146     }

151     /**
152      * Test forward focus movements with TAB key
153      */
154     @Test
155     public void testForwardFocus() {
156         assertTrue(toolBar.isFocusTraversable());
157 
158         toolBar.getScene().getWindow().requestFocus();
159         assertTrue(btn1.isFocused());
160 
161         keyboard.doKeyPress(KeyCode.TAB);
162         assertTrue(btn2.isFocused());
163 
164         keyboard.doKeyPress(KeyCode.TAB);
165         assertTrue(btn3.isFocused());
166 
167         keyboard.doKeyPress(KeyCode.TAB);
168         assertTrue(btn4.isFocused());
169 
170         keyboard.doKeyPress(KeyCode.TAB);
171         assertTrue(btn5.isFocused());
172     }
173 
174     /**
175      * Test backward focus movements with SHIFT+TAB keys
176      */
177     @Test
178     public void testBackwardFocus() {
179         assertTrue(toolBar.isFocusTraversable());
180 
181         toolBar.getScene().getWindow().requestFocus();
182         btn5.requestFocus();
183         assertTrue(btn5.isFocused());
184 
185         keyboard.doKeyPress(KeyCode.TAB, KeyModifier.SHIFT);
186         assertTrue(btn4.isFocused());
187 
188         keyboard.doKeyPress(KeyCode.TAB, KeyModifier.SHIFT);
189         assertTrue(btn3.isFocused());
190 
191         keyboard.doKeyPress(KeyCode.TAB, KeyModifier.SHIFT);
192         assertTrue(btn2.isFocused());
193 
194         keyboard.doKeyPress(KeyCode.TAB, KeyModifier.SHIFT);
195         assertTrue(btn1.isFocused());
196     }

324     /**
325      * Test focus movements when focus is at extreme child Nodes of the ToolBar
326      */
327     @Test
328     public void testFocusExtremeNodesOfToolBar() {
329         assertTrue(toolBar.isFocusTraversable());
330 
331         // Test backward movement when focus is at the first Button in the toolBar

340         assertTrue(btn5.isFocused());
