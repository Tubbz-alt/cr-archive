27 package com.sun.javatest.util;
28 
29 
30 import org.junit.Assert;
31 import org.junit.Test;
32 
33 public class ConsoleLoggingTestDefaultLevel extends ConsoleLoggingTestBase {
34 
35     @Test
36     public void test() {
37         savedSystemErr.clear();
38         Log.info("info message 345-");
39         Log.warning("warning message 345-");
40         Log.fine("fine message-");
41         Log.finer("finer message-");
42         Log.finest("finest message-");
43         Log.error("error message 890-");
44         warningFromOtherMethod();
45         Assert.assertEquals(4, savedSystemErr.size());
46         checkSystemErrLineEndsWith(0, "[INFO] (ConsoleLoggingTestDefaultLevel.test) info message 345-");
47         checkSystemErrLineEndsWith(1, "[WARNING] (ConsoleLoggingTestDefaultLevel.test) warning message 345-");
48         checkSystemErrLineEndsWith(2, "[SEVERE] (ConsoleLoggingTestDefaultLevel.test) error message 890-");
49         checkSystemErrLineEndsWith(3,
50                 "[WARNING] (ConsoleLoggingTestDefaultLevel.warningFromOtherMethod) warning message from other method");
51     }
52 
53     private void warningFromOtherMethod() {
54         Log.warning("warning message from other method");
55     }
56 
57 }
