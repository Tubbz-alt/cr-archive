  32 /**
  33  * Configuration parameters for a test run. Methods are provided
  34  * to access the parameters, and to access objects which contain
  35  * the permanent representation of the parameters, which is otherwise
  36  * undefined. Different representations include implementations
  37  * based on configuration interviews, and on simple files. A default
  38  * implementation, based on configuration interviews, is available
  39  * to simplify the task of providing configuration parameters in
  40  * almost all cases.
  41  *
  42  * @since 3.0.2
  43  */
  44 public interface Parameters {
  45     /**
  46      * Get the test suite for which these parameters apply.
  47      *
  48      * @return the test suite for which these parameters apply.
  49      * @see #setTestSuite
  50      */
  51     TestSuite getTestSuite();
  52 
  53     /**
  54      * Set the test suite for which these parameters apply.
  55      *
  56      * @param ts the test suite for which these parameters apply
  57      * @see #getTestSuite
  58      */
  59     void setTestSuite(TestSuite ts);
  60 
  61     /**
  62      * Get the work directory in which to store the results of the test run.
  63      *
  64      * @return the work directory in which to store the results of the test run.
  65      * @see #setWorkDirectory
  66      */
  67     WorkDirectory getWorkDirectory();
  68 
  69     /**
  70      * Set the work directory for which these parameters apply.
  71      *
  72      * @param wd the work directory for which these parameters apply
  73      * @see #getWorkDirectory
  74      */
  75     void setWorkDirectory(WorkDirectory wd);
  76 
  77     /**
  78      * Get the paths identifying the tests or folders of tests within
  79      * the test suite to be run.
  80      *
  81      * @return an array of paths identifying the tests to be run
  82      * @see Parameters.TestsParameters#getTests
  83      */
  84     String[] getTests();
  85 
  86     /**
  87      * Get an object which provides access to the paths identifying the tests
  88      * or folders of tests to be run.
  89      *
  90      * @return an object which provides access to the paths identifying tests to be run.
  91      */
  92     TestsParameters getTestsParameters();
  93 
  94     /**
  95      * Get an exclude list which identifies tests or test cases
  96      * to be excluded from the test run.
  97      *
  98      * @return an exclude list  identifying tests or test cases
  99      * to be excluded from the test run.
 100      * @see #getExcludeListFilter
 101      * @see Parameters.ExcludeListParameters#getExcludeList
 102      */
 103     ExcludeList getExcludeList();
 104 
 105     /**
 106      * Get an object which provides access to the exclude list which
 107      * identifies tests or test cases to be excluded from the test run.
 108      *
 109      * @return an object which provides access to the exclude list
 110      * identifying tests or test cases to be excluded from the test run.
 111      */
 112     ExcludeListParameters getExcludeListParameters();
 113 
 114     /**
 115      * Get a keywords object which identifies tests to be run according
 116      * to their keywords.
 117      *
 118      * @return a keywords object which identifies tests to be run according
 119      * to their keywords.
 120      * @see Parameters.KeywordsParameters#getKeywords
 121      */
 122     Keywords getKeywords();
 123 
 124     /**
 125      * Get an object which provides access to the keywords object which
 126      * identifies tests to be run according to their keywords.
 127      *
 128      * @return an object which provides access to the keywords object
 129      * which identifies tests to be run according to their keywords.
 130      */
 131     KeywordsParameters getKeywordsParameters();
 132 
 133     /**
 134      * Get an array of booleans which identify tests to be run according to their
 135      * prior execution status. The array can be indexed by the constants
 136      * Status.PASSED, Status.FAILED, Status.ERROR, and Status.NOT_RUN.
 137      * For each of those values, if the corresponding boolean in the array
 138      * is true, a test will be selected if its status matches the index.
 139      * If the array is null, all tests will be selected.
 140      *
 141      * @return an array of booleans which identifying tests to be run according
 142      * to their prior execution status, or null if no such criteria is required.
 143      * @see Parameters.PriorStatusParameters#getPriorStatusValues
 144      */
 145     boolean[] getPriorStatusValues();
 146 
 147     /**
 148      * Get an object which provides access to an array of booleans which
 149      * identify tests to be run according to their prior execution status.
 150      *
 151      * @return an object which provides access to an array of booleans which
 152      * identify tests to be run according to their prior execution status,
 153      * or null if no such selection criteria is required.
 154      */
 155     PriorStatusParameters getPriorStatusParameters();
 156 
 157     /**
 158      * Get the environment of test-suite-specific configuration values,
 159      * to be passed to the script used to run each test.
 160      *
 161      * @return an environment to be passed to the script used to run
 162      * each test.
 163      * @see Parameters.EnvParameters#getEnv
 164      */
 165     TestEnvironment getEnv();
 166 
 167     /**
 168      * Get an object which provides access to the environment of
 169      * test-suite-specific configuration values to be used when each
 170      * test is run.
 171      *
 172      * @return an object which provides access to the environment to
 173      * be used when each test is run.
 174      */
 175     EnvParameters getEnvParameters();
 176 
 177     /**
 178      * Get an integer specifying the maximum number of tests that may
 179      * be run in parallel.
 180      *
 181      * @return an integer specifying the maximum number of tests that
 182      * may be run in parallel
 183      * @see Parameters.ConcurrencyParameters#getConcurrency
 184      */
 185     int getConcurrency();
 186 
 187     /**
 188      * Get an object which provides access to the integer specifying
 189      * the maximum number of tests that may be run in parallel.
 190      *
 191      * @return an object which provides access to the integer specifying
 192      * the maximum number of tests that may be run in parallel.
 193      */
 194     ConcurrencyParameters getConcurrencyParameters();
 195 
 196     /**
 197      * Get an integer specifying a scale factor to be applied to the
 198      * standard timeout for the test.
 199      *
 200      * @return an integer specifying a scale factor to be applied to
 201      * the standard timeout for each test.
 202      * @see Parameters.TimeoutFactorParameters#getTimeoutFactor
 203      */
 204     float getTimeoutFactor();
 205 
 206     /**
 207      * Get an object which provides access to the integer specifying
 208      * a scale factor to be applied to the standard timeout for the test.
 209      *
 210      * @return an object which provides access to the integer specifying
 211      * a scale factor to be applied to the standard timeout for each test.
 212      */
 213     TimeoutFactorParameters getTimeoutFactorParameters();
 214 
 215     /**
 216      * Get a filter which will filter tests according to the result
 217      * of getExcludeList(). If the result of getExcludeList is null
 218      * or an empty exclude list, the result of this method will also be null.
 219      *
 220      * @return a filter which will filter tests according to the result
 221      * of getExcludeList().
 222      * @see #getExcludeList
 223      */
 224     TestFilter getExcludeListFilter();
 225 
 226     /**
 227      * Get a filter which will filter tests according to the result
 228      * of getKeywords(). If the result of getKeywords is null,
 229      * the result of this method will also be null.
 230      *
 231      * @return a filter which will filter tests according to the result
 232      * of getKeywords().
 233      * @see #getKeywords
 234      */
 235     TestFilter getKeywordsFilter();
 236 
 237     /**
 238      * Get a filter which will filter tests according to the result
 239      * of getPriorStatusValus(). If the result of getPriorStatusValues is null,
 240      * the result of this method will also be null.
 241      *
 242      * @return a filter which will filter tests according to the result
 243      * of getPriorStatusValues().
 244      * @see #getPriorStatusValues
 245      */
 246     TestFilter getPriorStatusFilter();
 247 
 248     /**
 249      * Get a test-suite specific filter which will filter tests according
 250      * to test-suite-specific criteria, as perhaps determined by
 251      * a configuration interview. For example, if the platform being tested
 252      * does not support some optional feature, the tests for that feature
 253      * could be automatically filtered out. If no such filter is required,
 254      * null can be returned.
 255      *
 256      * @return a test-suite-specific filter, or null if no such filter is
 257      * required.
 258      */
 259     TestFilter getRelevantTestFilter();
 260 
 261     /**
 262      * Get an array of the non-null filters returned from
 263      * getExcludeListFilter, getKeywordsFilter, getPriorStatusFilter,
 264      * and getRelevantTestFilter.
 265      *
 266      * @return an array of the non-null filters returned by the
 267      * various getXXXFilter methods.
 268      * @see #getExcludeListFilter
 269      * @see #getKeywordsFilter
 270      * @see #getPriorStatusFilter
 271      * @see #getRelevantTestFilter
 272      */
 273     TestFilter[] getFilters();
 274 
 275     /**
 276      * Determine whether all the configuration values are valid.
 277      * If so, the result will be true; if not, the result will be false,
 278      * and getErrorMessage will provide details about at least one of the
 279      * invalid values.
 280      *
 281      * @return true if and only if all the configuration values are valid
 282      * @see #getErrorMessage
 283      */
 284     boolean isValid();
 285 
 286     /**
 287      * If there is an error in any of the configuration values,
 288      * as indicated by isValid, this method will provide a detail
 289      * message about one or more of the invalid values.
 290      * The result is undefined if isValid is true.
 291      *
 292      * @return a detail message about one or more invalid values
 293      * @see #isValid
 294      */
 295     String getErrorMessage();

 301      * defining the tests and folders of tests to be run.

 303     interface TestsParameters {
 304         /**
 305          * Get the paths identifying the tests or folders of tests within
 306          * the test suite to be run.
 307          *
 308          * @return an array of paths identifying the tests to be run
 309          * @see Parameters#getTests
 310          */
 311         String[] getTests();
 312     }

 316     /**
 317      * An interface providing abstract access to a set of paths
 318      * defining the tests and folders of tests to be run.
 319      */
 320     interface MutableTestsParameters extends TestsParameters {
 321         /**
 322          * A constant used to indicate that all tests in the test
 323          * suite should be run.
 324          */
 325         int ALL_TESTS = 1;
 326         /**
 327          * A constant used to indicate that specified tests in the test
 328          * suite should be run.
 329          */
 330         int SPECIFIED_TESTS = 2;
 331 
 332         /**
 333          * Specify the tests to be executed.
 334          *
 335          * @param tests If null, set the tests mode to ALL_TESTS;
 336          *              if not null, set the tests mode to SPECIFIED_TESTS, and set
 337          *              the specified tests.
 338          * @see #getTests
 339          */
 340         void setTests(String... tests);
 341 
 342         /**
 343          * Get the current mode for how the tests are specified.
 344          *
 345          * @return ALL_TESTS if all tests are to be run, irrespective of
 346          * the selected tests, or SPECIFIED_TESTS if  a set of specified
 347          * tests are to be run.
 348          * @see #setTestsMode
 349          * @see #ALL_TESTS
 350          * @see #SPECIFIED_TESTS
 351          */
 352         int getTestsMode();
 353 
 354         /**
 355          * Set the current mode for how the tests are specified.
 356          *
 357          * @param mode use ALL_TESTS if all tests are to be run, irrespective of
 358          *             the selected tests, or SPECIFIED_TESTS if  a set of specified
 359          *             tests are to be run.
 360          * @throws IllegalArgumentException if neither ALL_TESTS or SPECIFIED_TESTS
 361          *                                  is given
 362          * @see #getTestsMode
 363          * @see #ALL_TESTS
 364          * @see #SPECIFIED_TESTS
 365          */
 366         void setTestsMode(int mode);
 367 
 368         /**
 369          * Get the set of specified tests to be used as the set of tests to
 370          * be run when the mode is set to SPECIFIED_TESTS. When the mode is
 371          * set to ALL_TESTS, the specified tests are remembered, but not used,
 372          * until the mode is set back to SPECIFIED_TESTS.
 373          *
 374          * @return an array of specified tests to be used as the set of tests to
 375          * be run when the mode is set to SPECIFIED_TESTS.
 376          * @see #setSpecifiedTests
 377          * @see #getTests
 378          */
 379         String[] getSpecifiedTests();
 380 
 381         /**
 382          * Set the specified tests to be be run when the mode is set to SPECIFIED_TESTS.
 383          * When the mode is set to ALL_TESTS, these tests are remembered, but not used,
 384          * until the mode is set back to SPECIFIED_TESTS.
 385          *
 386          * @param tests an array of paths identifying the tests to be run
 387          * @see #getSpecifiedTests
 388          * @see #getTests
 389          */
 390         void setSpecifiedTests(String... tests);
 391     }

 414     /**
 415      * An interface providing abstract access to an exclude list,
 416      * as defined by a set of files, defining tests to be excluded
 417      * from the test run.
 418      */
 419     interface MutableExcludeListParameters extends ExcludeListParameters {
 420         /**
 421          * A constant used to indicate that no exclude list is required.
 422          */
 423         int NO_EXCLUDE_LIST = 1;
 424         /**
 425          * A constant used to indicate that the default exclude list
 426          * (if any) for the test suite should be used.
 427          *
 428          * @see TestSuite#hasInitialExcludeList
 429          */
 430         int INITIAL_EXCLUDE_LIST = 2;
 431         /**
 432          * A constant used to indicate that the latest exclude list
 433          * (if any) for the test suite should be used.
 434          *
 435          * @see TestSuite#hasLatestExcludeList
 436          */
 437         int LATEST_EXCLUDE_LIST = 3;
 438         /**
 439          * A constant used to indicate that a client-supplied set
 440          * of exclude files should be used.
 441          */
 442         int CUSTOM_EXCLUDE_LIST = 4;
 443         /**
 444          * A constant used to indicate that the website used to
 445          * supply the latest exclude list should be checked every
 446          * so many days, to see if a newer version is available.
 447          */
 448         int CHECK_EVERY_X_DAYS = 1;
 449         /**
 450          * A constant used to indicate that the website used to
 451          * supply the latest exclude list should be checked on every
 452          * test run to see if a newer version is available.
 453          */
 454         int CHECK_EVERY_RUN = 2;
 455 
 456         /**
 457          * Get the set of files which define the exclude list. The files
 458          * are all returned as absolute files.
 459          *
 460          * @return the set of files which define the exclude list
 461          * @see #getExcludeFiles
 462          * @see #setExcludeFiles
 463          */
 464         File[] getExcludeFiles();
 465 
 466         /**
 467          * Set the set of files used to define the exclude list.
 468          *
 469          * @param files If null, the exclude mode will be set to NO_EXCLUDE_LIST;
 470          *              if not null, the exclude mode will be set to CUSTOM_EXCLUDE_LIST
 471          *              and the custom exclude files will be set to this value
 472          * @see #getExcludeFiles
 473          */
 474         void setExcludeFiles(File... files);
 475 
 476         /**
 477          * Get the current exclude list mode.
 478          *
 479          * @return A value indicating the current exclude list mode
 480          * @see #setExcludeMode
 481          * @see #NO_EXCLUDE_LIST
 482          * @see #INITIAL_EXCLUDE_LIST
 483          * @see #LATEST_EXCLUDE_LIST
 484          * @see #CUSTOM_EXCLUDE_LIST
 485          */
 486         int getExcludeMode();
 487 
 488         /**
 489          * Set the current exclude list mode.
 490          *
 491          * @param mode A value indicating the desired exclude list mode
 492          * @see #getExcludeMode
 493          * @see #NO_EXCLUDE_LIST
 494          * @see #INITIAL_EXCLUDE_LIST
 495          * @see #LATEST_EXCLUDE_LIST
 496          * @see #CUSTOM_EXCLUDE_LIST
 497          */
 498         void setExcludeMode(int mode);
 499 
 500         /**
 501          * Get the files used to define the exclude list when the
 502          * exclude list mode is set to CUSTOM_EXCLUDE_LIST.
 503          *
 504          * @return the files used to define a custom exclude list
 505          * @see #setCustomExcludeFiles
 506          */
 507         File[] getCustomExcludeFiles();
 508 
 509         /**
 510          * Set the files used to define the exclude list when the
 511          * exclude list mode is set to CUSTOM_EXCLUDE_LIST.
 512          *
 513          * @param files the files used to define a custom exclude list
 514          * @see #getCustomExcludeFiles
 515          */
 516         void setCustomExcludeFiles(File... files);
 517 
 518         /**
 519          * Check if the automatic check for newer exclude lists
 520          * is enabled when the exclude list mode is set to LATEST_EXCLUDE_LIST.
 521          *
 522          * @return true if the automatic check is enabled
 523          * @see #setLatestExcludeAutoCheckEnabled
 524          */
 525         boolean isLatestExcludeAutoCheckEnabled();
 526 
 527         /**
 528          * Specify if the automatic check for newer exclude lists
 529          * is enabled when the exclude list mode is set to LATEST_EXCLUDE_LIST.
 530          *
 531          * @param b whether or not the automatic check is enabled
 532          * @see #isLatestExcludeAutoCheckEnabled
 533          */
 534         void setLatestExcludeAutoCheckEnabled(boolean b);
 535 
 536         /**
 537          * Get the mode which defines how often to automatically check
 538          * for updated exclude lists, when the exclude list mode is set
 539          * to LATEST_EXCLUDE_LIST, and the automatic check is enabled.
 540          *
 541          * @return a value indicating how often to check for the
 542          * availability of a newer exclude list
 543          * @see #setLatestExcludeAutoCheckMode
 544          * @see #CHECK_EVERY_X_DAYS
 545          * @see #CHECK_EVERY_RUN
 546          */
 547         int getLatestExcludeAutoCheckMode();
 548 
 549         /**
 550          * Set the mode which defines how often to automatically check
 551          * for updated exclude lists, when the exclude list mode is set
 552          * to LATEST_EXCLUDE_LIST, and the automatic check is enabled.
 553          *
 554          * @param mode a value indicating how often to check for the
 555          *             availability of a newer exclude list
 556          * @see #getLatestExcludeAutoCheckMode
 557          * @see #CHECK_EVERY_X_DAYS
 558          * @see #CHECK_EVERY_RUN
 559          */
 560         void setLatestExcludeAutoCheckMode(int mode);
 561 
 562         /**
 563          * Get the interval, in days, to be used when automatically
 564          * checking for exclude list updates and the auto check mode
 565          * is set to CHECK_EVERY_X_DAYS.
 566          *
 567          * @return the interval, in days, between checks
 568          * @see #setLatestExcludeAutoCheckInterval
 569          */
 570         int getLatestExcludeAutoCheckInterval();
 571 
 572         /**
 573          * Set the interval, in days, to be used when automatically
 574          * checking for exclude list updates and the auto check mode
 575          * is set to CHECK_EVERY_X_DAYS.
 576          *
 577          * @param days the number of days to wait between checks
 578          * @see #getLatestExcludeAutoCheckInterval
 579          */
 580         void setLatestExcludeAutoCheckInterval(int days);
 581     }

 602     /**
 603      * An interface providing abstract access to a keywords object
 604      * which can be used to select tests according to their keywords.
 605      */
 606     interface MutableKeywordsParameters extends KeywordsParameters {
 607         /**
 608          * A constant used to indicate that no keyword filtering
 609          * should be used.
 610          */
 611         int NO_KEYWORDS = 1;
 612         /**
 613          * A constant used to indicate that only tests matching the
 614          * specified keywords should be selected.
 615          */
 616         int MATCH_KEYWORDS = 2;
 617         /**
 618          * A constant used to indicate that tests that match any of
 619          * the given keywords should be selected.
 620          */
 621         int ANY_OF = 1;
 622         /**
 623          * A constant used to indicate that tests that match all of
 624          * the given keywords should be selected.
 625          */
 626         int ALL_OF = 2;
 627         /**
 628          * A constant used to indicate that tests that match the
 629          * given keyword expression should be selected.
 630          */
 631         int EXPR = 3;
 632 
 633         /**
 634          * Set the details of the keywords to be used, if any,
 635          * to filter tests for execution.
 636          *
 637          * @param mode  the value of the match keywords mode to be set if
 638          *              {@code value} is not null
 639          * @param value if null, the keywords mode will be set to NO_KEYWORDS;
 640          *              if not null, the keywords mode will be set to MATCH_KEYWORDS, the
 641          *              match keywords mode will be set to {@code mode},
 642          *              and the match keywords value will be set to this value
 643          * @see #getKeywords
 644          * @see #NO_KEYWORDS
 645          * @see #MATCH_KEYWORDS
 646          */
 647         void setKeywords(int mode, String value);
 648 
 649         /**
 650          * Get the current keywords mode.
 651          *
 652          * @return NO_KEYWORDS if no keyword filtering will be used to
 653          * select tests for execution, or MATCH_KEYWORDS if  keywords
 654          * will be filtered according to the match mode and match value.
 655          * @see #setKeywordsMode
 656          * @see #NO_KEYWORDS
 657          * @see #MATCH_KEYWORDS
 658          */
 659         int getKeywordsMode();
 660 
 661         /**
 662          * Set the current keywords mode.
 663          *
 664          * @param mode set to NO_KEYWORDS if no keyword filtering will be
 665          *             used to select tests for execution, or MATCH_KEYWORDS if  keywords
 666          *             will be filtered according to the match mode and match value.
 667          * @see #getKeywordsMode
 668          * @see #NO_KEYWORDS
 669          * @see #MATCH_KEYWORDS
 670          */
 671         void setKeywordsMode(int mode);
 672 
 673         /**
 674          * Get a value which indicates how to interpret the match value,
 675          * when the keywords mode is set to MATCH_KEYWORDS.
 676          *
 677          * @return a value which indicates how to interpret the match value,
 678          * when the keywords mode is set to MATCH_KEYWORDS
 679          * @see #setMatchKeywords
 680          * @see #ANY_OF
 681          * @see #ALL_OF
 682          * @see #EXPR
 683          */
 684         int getMatchKeywordsMode();
 685 
 686         /**
 687          * Get a value that identifies which tests are to be selected,
 688          * when the keywords mode is set to MATCH_KEYWORDS.
 689          *
 690          * @return a value that identifies which tests are to be selected,
 691          * when the keywords mode is set to MATCH_KEYWORDS
 692          * @see #setMatchKeywords
 693          * @see #ANY_OF
 694          * @see #ALL_OF
 695          * @see #EXPR
 696          */
 697         String getMatchKeywordsValue();
 698 
 699         /**
 700          * Set how to match a tests keywords when the keywords mode
 701          * is set to MATCH_KEYWORDS.
 702          *
 703          * @param mode  A value indicating how to interpret {@code value}
 704          * @param value If {@code mode} is set to ANY_OF or ALL_OF,
 705          *              this value should give a white-space separate list of keywords
 706          *              to be matched; if {@code mode} is set to EXPR, this value
 707          *              should be a boolean expression using terminals, &amp; (and), | (or),
 708          *              ! (negation) and parentheses, where the terminals are true if
 709          *              a test description contains that name as one of its keywords.
 710          * @see #getMatchKeywordsMode
 711          * @see #getMatchKeywordsValue
 712          * @see #ANY_OF
 713          * @see #ALL_OF
 714          * @see #EXPR
 715          */
 716         void setMatchKeywords(int mode, String value);
 717 
 718     }

 722     /**
 723      * An interface providing basic abstract access to a set of booleans
 724      * which can be used to select tests according to their prior execution
