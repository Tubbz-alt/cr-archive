  34 /**
  35  * Configuration parameters for a test run. Methods are provided
  36  * to access the parameters, and to access objects which contain
  37  * the permanent representation of the parameters, which is otherwise
  38  * undefined. Different representations include implementations
  39  * based on configuration interviews, and on simple files. A default
  40  * implementation, based on configuration interviews, is available
  41  * to simplify the task of providing configuration parameters in
  42  * almost all cases.
  43  *
  44  * @since 3.0.2
  45  */
  46 public interface Parameters {
  47     /**
  48      * Get the test suite for which these parameters apply.
  49      *
  50      * @return the test suite for which these parameters apply.
  51      * @see #setTestSuite
  52      */
  53     TestSuite getTestSuite();
  54 
  55     /**
  56      * Set the test suite for which these parameters apply.
  57      *
  58      * @param ts the test suite for which these parameters apply
  59      * @see #getTestSuite
  60      */
  61     void setTestSuite(TestSuite ts);
  62 
  63     /**
  64      * Get the work directory in which to store the results of the test run.
  65      *
  66      * @return the work directory in which to store the results of the test run.
  67      * @see #setWorkDirectory
  68      */
  69     WorkDirectory getWorkDirectory();
  70 
  71     /**
  72      * Set the work directory for which these parameters apply.
  73      *
  74      * @param wd the work directory for which these parameters apply
  75      * @see #getWorkDirectory
  76      */
  77     void setWorkDirectory(WorkDirectory wd);
  78 
  79     /**
  80      * Get the paths identifying the tests or folders of tests within
  81      * the test suite to be run.
  82      *
  83      * @return an array of paths identifying the tests to be run
  84      * @see Parameters.TestsParameters#getTests
  85      */
  86     String[] getTests();
  87 
  88     /**
  89      * Get an object which provides access to the paths identifying the tests
  90      * or folders of tests to be run.
  91      *
  92      * @return an object which provides access to the paths identifying tests to be run.
  93      */
  94     TestsParameters getTestsParameters();
  95 
  96     /**
  97      * Get an exclude list which identifies tests or test cases
  98      * to be excluded from the test run.
  99      *
 100      * @return an exclude list  identifying tests or test cases
 101      * to be excluded from the test run.
 102      * @see #getExcludeListFilter
 103      * @see Parameters.ExcludeListParameters#getExcludeList
 104      */
 105     ExcludeList getExcludeList();
 106 
 107     /**
 108      * Get an object which provides access to the exclude list which
 109      * identifies tests or test cases to be excluded from the test run.
 110      *
 111      * @return an object which provides access to the exclude list
 112      * identifying tests or test cases to be excluded from the test run.
 113      */
 114     ExcludeListParameters getExcludeListParameters();
 115 
 116     /**
 117      * Get a keywords object which identifies tests to be run according
 118      * to their keywords.
 119      *
 120      * @return a keywords object which identifies tests to be run according
 121      * to their keywords.
 122      * @see Parameters.KeywordsParameters#getKeywords
 123      */
 124     Keywords getKeywords();
 125 
 126     /**
 127      * Get an object which provides access to the keywords object which
 128      * identifies tests to be run according to their keywords.
 129      *
 130      * @return an object which provides access to the keywords object
 131      * which identifies tests to be run according to their keywords.
 132      */
 133     KeywordsParameters getKeywordsParameters();
 134 
 135     /**
 136      * Get an array of booleans which identify tests to be run according to their
 137      * prior execution status. The array can be indexed by the constants
 138      * Status.PASSED, Status.FAILED, Status.ERROR, and Status.NOT_RUN.
 139      * For each of those values, if the corresponding boolean in the array
 140      * is true, a test will be selected if its status matches the index.
 141      * If the array is null, all tests will be selected.
 142      *
 143      * @return an array of booleans which identifying tests to be run according
 144      * to their prior execution status, or null if no such criteria is required.
 145      * @see Parameters.PriorStatusParameters#getPriorStatusValues
 146      */
 147     boolean[] getPriorStatusValues();
 148 
 149     /**
 150      * Get an object which provides access to an array of booleans which
 151      * identify tests to be run according to their prior execution status.
 152      *
 153      * @return an object which provides access to an array of booleans which
 154      * identify tests to be run according to their prior execution status,
 155      * or null if no such selection criteria is required.
 156      */
 157     PriorStatusParameters getPriorStatusParameters();
 158 
 159     /**
 160      * Get the environment of test-suite-specific configuration values,
 161      * to be passed to the script used to run each test.
 162      *
 163      * @return an environment to be passed to the script used to run
 164      * each test.
 165      * @see Parameters.EnvParameters#getEnv
 166      */
 167     TestEnvironment getEnv();
 168 
 169     /**
 170      * Get an object which provides access to the environment of
 171      * test-suite-specific configuration values to be used when each
 172      * test is run.
 173      *
 174      * @return an object which provides access to the environment to
 175      * be used when each test is run.
 176      */
 177     EnvParameters getEnvParameters();
 178 
 179     /**
 180      * Get an integer specifying the maximum number of tests that may
 181      * be run in parallel.
 182      *
 183      * @return an integer specifying the maximum number of tests that
 184      * may be run in parallel
 185      * @see Parameters.ConcurrencyParameters#getConcurrency
 186      */
 187     int getConcurrency();
 188 
 189     /**
 190      * Get an object which provides access to the integer specifying
 191      * the maximum number of tests that may be run in parallel.
 192      *
 193      * @return an object which provides access to the integer specifying
 194      * the maximum number of tests that may be run in parallel.
 195      */
 196     ConcurrencyParameters getConcurrencyParameters();
 197 
 198     /**
 199      * Get an integer specifying a scale factor to be applied to the
 200      * standard timeout for the test.
 201      *
 202      * @return an integer specifying a scale factor to be applied to
 203      * the standard timeout for each test.
 204      * @see Parameters.TimeoutFactorParameters#getTimeoutFactor
 205      */
 206     float getTimeoutFactor();
 207 
 208     /**
 209      * Get an object which provides access to the integer specifying
 210      * a scale factor to be applied to the standard timeout for the test.
 211      *
 212      * @return an object which provides access to the integer specifying
 213      * a scale factor to be applied to the standard timeout for each test.
 214      */
 215     TimeoutFactorParameters getTimeoutFactorParameters();
 216 
 217     /**
 218      * Get a filter which will filter tests according to the result
 219      * of getExcludeList(). If the result of getExcludeList is null
 220      * or an empty exclude list, the result of this method will also be null.
 221      *
 222      * @return a filter which will filter tests according to the result
 223      * of getExcludeList().
 224      * @see #getExcludeList
 225      */
 226     TestFilter getExcludeListFilter();
 227 
 228     /**
 229      * Get a filter which will filter tests according to the result
 230      * of getKeywords(). If the result of getKeywords is null,
 231      * the result of this method will also be null.
 232      *
 233      * @return a filter which will filter tests according to the result
 234      * of getKeywords().
 235      * @see #getKeywords
 236      */
 237     TestFilter getKeywordsFilter();
 238 
 239     /**
 240      * Get a filter which will filter tests according to the result
 241      * of getPriorStatusValus(). If the result of getPriorStatusValues is null,
 242      * the result of this method will also be null.
 243      *
 244      * @return a filter which will filter tests according to the result
 245      * of getPriorStatusValues().
 246      * @see #getPriorStatusValues
 247      */
 248     TestFilter getPriorStatusFilter();
 249 
 250     /**
 251      * Get test-suite specific filters which will filter tests according
 252      * to test-suite-specific criterias, as perhaps determined by
 253      * a configuration interview. For example, if the platform being tested
 254      * does not support some optional feature, the tests for that feature
 255      * could be automatically filtered out. If no such filter is required,
 256      * empty collection can be returned.
 257      *
 258      * @return test-suite-specific filters, or empty collection if no such filter is
 259      * required.
 260      */
 261     TestFilter getRelevantTestFilter();
 262 
 263     /**
 264      * Get test-suite specific filters which will filter tests according
 265      * to test-suite-specific criterias, as perhaps determined by
 266      * a configuration interview. For example, if the platform being tested
 267      * does not support some optional feature, the tests for that feature
 268      * could be automatically filtered out. If no such filter is required,
 269      * empty collection can be returned.
 270      *
 271      * @return test-suite-specific filters, or empty collection if no such filter is
 272      * required.
 273      */
 274     default List<TestFilter> getAllRelevantFiltersInTheSuite() {
 275         List<TestFilter> result = new ArrayList<>();
 276         TestFilter relevantTestFilter = getRelevantTestFilter();
 277         if (relevantTestFilter != null) {
 278             result.add(relevantTestFilter);
 279         }
 280         result.addAll(getTestSuite().createTestFilters(getEnv()));
 281         return result;
 282     }
 283 
 284     /**
 285      * Get an array of the non-null filters returned from
 286      * getExcludeListFilter, getKeywordsFilter, getPriorStatusFilter,
 287      * and getRelevantTestFilter.
 288      *
 289      * @return an array of the non-null filters returned by the
 290      * various getXXXFilter methods.
 291      * @see #getExcludeListFilter
 292      * @see #getKeywordsFilter
 293      * @see #getPriorStatusFilter
 294      * @see #getRelevantTestFilters
 295      */
 296     TestFilter[] getFilters();
 297 
 298     /**
 299      * Determine whether all the configuration values are valid.
 300      * If so, the result will be true; if not, the result will be false,
 301      * and getErrorMessage will provide details about at least one of the
 302      * invalid values.
 303      *
 304      * @return true if and only if all the configuration values are valid
 305      * @see #getErrorMessage
 306      */
 307     boolean isValid();
 308 
 309     /**
 310      * If there is an error in any of the configuration values,
 311      * as indicated by isValid, this method will provide a detail
 312      * message about one or more of the invalid values.
 313      * The result is undefined if isValid is true.
 314      *
 315      * @return a detail message about one or more invalid values
 316      * @see #isValid
 317      */
 318     String getErrorMessage();

 327         /**
 328          * Get the paths identifying the tests or folders of tests within
 329          * the test suite to be run.
 330          *
 331          * @return an array of paths identifying the tests to be run
 332          * @see Parameters#getTests
 333          */
 334         String[] getTests();
 335     }

 339     /**
 340      * An interface providing abstract access to a set of paths
 341      * defining the tests and folders of tests to be run.
 342      */
 343     interface MutableTestsParameters extends TestsParameters {
 344         /**
 345          * A constant used to indicate that all tests in the test
 346          * suite should be run.
 347          */
 348         int ALL_TESTS = 1;
 349         /**
 350          * A constant used to indicate that specified tests in the test
 351          * suite should be run.
 352          */
 353         int SPECIFIED_TESTS = 2;
 354 
 355         /**
 356          * Specify the tests to be executed.
 357          *
 358          * @param tests If null, set the tests mode to ALL_TESTS;
 359          *              if not null, set the tests mode to SPECIFIED_TESTS, and set
 360          *              the specified tests.
 361          * @see #getTests
 362          */
 363         void setTests(String... tests);
 364 
 365         /**
 366          * Get the current mode for how the tests are specified.
 367          *
 368          * @return ALL_TESTS if all tests are to be run, irrespective of
 369          * the selected tests, or SPECIFIED_TESTS if  a set of specified
 370          * tests are to be run.
 371          * @see #setTestsMode
 372          * @see #ALL_TESTS
 373          * @see #SPECIFIED_TESTS
 374          */
 375         int getTestsMode();
 376 
 377         /**
 378          * Set the current mode for how the tests are specified.
 379          *
 380          * @param mode use ALL_TESTS if all tests are to be run, irrespective of
 381          *             the selected tests, or SPECIFIED_TESTS if  a set of specified
 382          *             tests are to be run.
 383          * @throws IllegalArgumentException if neither ALL_TESTS or SPECIFIED_TESTS
 384          *                                  is given
 385          * @see #getTestsMode
 386          * @see #ALL_TESTS
 387          * @see #SPECIFIED_TESTS
 388          */
 389         void setTestsMode(int mode);
 390 
 391         /**
 392          * Get the set of specified tests to be used as the set of tests to
 393          * be run when the mode is set to SPECIFIED_TESTS. When the mode is
 394          * set to ALL_TESTS, the specified tests are remembered, but not used,
 395          * until the mode is set back to SPECIFIED_TESTS.
 396          *
 397          * @return an array of specified tests to be used as the set of tests to
 398          * be run when the mode is set to SPECIFIED_TESTS.
 399          * @see #setSpecifiedTests
 400          * @see #getTests
 401          */
 402         String[] getSpecifiedTests();
 403 
 404         /**
 405          * Set the specified tests to be be run when the mode is set to SPECIFIED_TESTS.
 406          * When the mode is set to ALL_TESTS, these tests are remembered, but not used,
 407          * until the mode is set back to SPECIFIED_TESTS.
 408          *
 409          * @param tests an array of paths identifying the tests to be run
 410          * @see #getSpecifiedTests
 411          * @see #getTests
 412          */
 413         void setSpecifiedTests(String... tests);
 414     }

 437     /**
 438      * An interface providing abstract access to an exclude list,
 439      * as defined by a set of files, defining tests to be excluded
 440      * from the test run.
 441      */
 442     interface MutableExcludeListParameters extends ExcludeListParameters {
 443         /**
 444          * A constant used to indicate that no exclude list is required.
 445          */
 446         int NO_EXCLUDE_LIST = 1;
 447         /**
 448          * A constant used to indicate that the default exclude list
 449          * (if any) for the test suite should be used.
 450          *
 451          * @see TestSuite#hasInitialExcludeList
 452          */
 453         int INITIAL_EXCLUDE_LIST = 2;
 454         /**
 455          * A constant used to indicate that the latest exclude list
 456          * (if any) for the test suite should be used.
 457          *
 458          * @see TestSuite#hasLatestExcludeList
 459          */
 460         int LATEST_EXCLUDE_LIST = 3;
 461         /**
 462          * A constant used to indicate that a client-supplied set
 463          * of exclude files should be used.
 464          */
 465         int CUSTOM_EXCLUDE_LIST = 4;
 466         /**
 467          * A constant used to indicate that the website used to
 468          * supply the latest exclude list should be checked every
 469          * so many days, to see if a newer version is available.
 470          */
 471         int CHECK_EVERY_X_DAYS = 1;
 472         /**
 473          * A constant used to indicate that the website used to
 474          * supply the latest exclude list should be checked on every
 475          * test run to see if a newer version is available.
 476          */
 477         int CHECK_EVERY_RUN = 2;
 478 
 479         /**
 480          * Get the set of files which define the exclude list. The files
 481          * are all returned as absolute files.
 482          *
 483          * @return the set of files which define the exclude list
 484          * @see #getExcludeFiles
 485          * @see #setExcludeFiles
 486          */
 487         File[] getExcludeFiles();
 488 
 489         /**
 490          * Set the set of files used to define the exclude list.
 491          *
 492          * @param files If null, the exclude mode will be set to NO_EXCLUDE_LIST;
 493          *              if not null, the exclude mode will be set to CUSTOM_EXCLUDE_LIST
 494          *              and the custom exclude files will be set to this value
 495          * @see #getExcludeFiles
 496          */
 497         void setExcludeFiles(File... files);
 498 
 499         /**
 500          * Get the current exclude list mode.
 501          *
 502          * @return A value indicating the current exclude list mode
 503          * @see #setExcludeMode
 504          * @see #NO_EXCLUDE_LIST
 505          * @see #INITIAL_EXCLUDE_LIST
 506          * @see #LATEST_EXCLUDE_LIST
 507          * @see #CUSTOM_EXCLUDE_LIST
 508          */
 509         int getExcludeMode();
 510 
 511         /**
 512          * Set the current exclude list mode.
 513          *
 514          * @param mode A value indicating the desired exclude list mode
 515          * @see #getExcludeMode
 516          * @see #NO_EXCLUDE_LIST
 517          * @see #INITIAL_EXCLUDE_LIST
 518          * @see #LATEST_EXCLUDE_LIST
 519          * @see #CUSTOM_EXCLUDE_LIST
 520          */
 521         void setExcludeMode(int mode);
 522 
 523         /**
 524          * Get the files used to define the exclude list when the
 525          * exclude list mode is set to CUSTOM_EXCLUDE_LIST.
 526          *
 527          * @return the files used to define a custom exclude list
 528          * @see #setCustomExcludeFiles
 529          */
 530         File[] getCustomExcludeFiles();
 531 
 532         /**
 533          * Set the files used to define the exclude list when the
 534          * exclude list mode is set to CUSTOM_EXCLUDE_LIST.
 535          *
 536          * @param files the files used to define a custom exclude list
 537          * @see #getCustomExcludeFiles
 538          */
 539         void setCustomExcludeFiles(File... files);
 540 
 541         /**
 542          * Check if the automatic check for newer exclude lists
 543          * is enabled when the exclude list mode is set to LATEST_EXCLUDE_LIST.
 544          *
 545          * @return true if the automatic check is enabled
 546          * @see #setLatestExcludeAutoCheckEnabled
 547          */
 548         boolean isLatestExcludeAutoCheckEnabled();
 549 
 550         /**
 551          * Specify if the automatic check for newer exclude lists
 552          * is enabled when the exclude list mode is set to LATEST_EXCLUDE_LIST.
 553          *
 554          * @param b whether or not the automatic check is enabled
 555          * @see #isLatestExcludeAutoCheckEnabled
 556          */
 557         void setLatestExcludeAutoCheckEnabled(boolean b);
 558 
 559         /**
 560          * Get the mode which defines how often to automatically check
 561          * for updated exclude lists, when the exclude list mode is set
 562          * to LATEST_EXCLUDE_LIST, and the automatic check is enabled.
 563          *
 564          * @return a value indicating how often to check for the
 565          * availability of a newer exclude list
 566          * @see #setLatestExcludeAutoCheckMode
 567          * @see #CHECK_EVERY_X_DAYS
 568          * @see #CHECK_EVERY_RUN
 569          */
 570         int getLatestExcludeAutoCheckMode();
 571 
 572         /**
 573          * Set the mode which defines how often to automatically check
 574          * for updated exclude lists, when the exclude list mode is set
 575          * to LATEST_EXCLUDE_LIST, and the automatic check is enabled.
 576          *
 577          * @param mode a value indicating how often to check for the
 578          *             availability of a newer exclude list
 579          * @see #getLatestExcludeAutoCheckMode
 580          * @see #CHECK_EVERY_X_DAYS
 581          * @see #CHECK_EVERY_RUN
 582          */
 583         void setLatestExcludeAutoCheckMode(int mode);
 584 
 585         /**
 586          * Get the interval, in days, to be used when automatically
 587          * checking for exclude list updates and the auto check mode
 588          * is set to CHECK_EVERY_X_DAYS.
 589          *
 590          * @return the interval, in days, between checks
 591          * @see #setLatestExcludeAutoCheckInterval
 592          */
 593         int getLatestExcludeAutoCheckInterval();
 594 
 595         /**
 596          * Set the interval, in days, to be used when automatically
 597          * checking for exclude list updates and the auto check mode
 598          * is set to CHECK_EVERY_X_DAYS.
 599          *
 600          * @param days the number of days to wait between checks
 601          * @see #getLatestExcludeAutoCheckInterval
 602          */
 603         void setLatestExcludeAutoCheckInterval(int days);
 604     }

 625     /**
 626      * An interface providing abstract access to a keywords object
 627      * which can be used to select tests according to their keywords.
 628      */
 629     interface MutableKeywordsParameters extends KeywordsParameters {
 630         /**
 631          * A constant used to indicate that no keyword filtering
 632          * should be used.
 633          */
 634         int NO_KEYWORDS = 1;
 635         /**
 636          * A constant used to indicate that only tests matching the
 637          * specified keywords should be selected.
 638          */
 639         int MATCH_KEYWORDS = 2;
 640         /**
 641          * A constant used to indicate that tests that match any of
 642          * the given keywords should be selected.
 643          */
 644         int ANY_OF = 1;
 645         /**
 646          * A constant used to indicate that tests that match all of
 647          * the given keywords should be selected.
 648          */
 649         int ALL_OF = 2;
 650         /**
 651          * A constant used to indicate that tests that match the
 652          * given keyword expression should be selected.
 653          */
 654         int EXPR = 3;
 655 
 656         /**
 657          * Set the details of the keywords to be used, if any,
 658          * to filter tests for execution.
 659          *
 660          * @param mode  the value of the match keywords mode to be set if
 661          *              {@code value} is not null
 662          * @param value if null, the keywords mode will be set to NO_KEYWORDS;
 663          *              if not null, the keywords mode will be set to MATCH_KEYWORDS, the
 664          *              match keywords mode will be set to {@code mode},
 665          *              and the match keywords value will be set to this value
 666          * @see #getKeywords
 667          * @see #NO_KEYWORDS
 668          * @see #MATCH_KEYWORDS
 669          */
 670         void setKeywords(int mode, String value);
 671 
 672         /**
 673          * Get the current keywords mode.
 674          *
 675          * @return NO_KEYWORDS if no keyword filtering will be used to
 676          * select tests for execution, or MATCH_KEYWORDS if  keywords
 677          * will be filtered according to the match mode and match value.
 678          * @see #setKeywordsMode
 679          * @see #NO_KEYWORDS
 680          * @see #MATCH_KEYWORDS
 681          */
 682         int getKeywordsMode();
 683 
 684         /**
 685          * Set the current keywords mode.
 686          *
 687          * @param mode set to NO_KEYWORDS if no keyword filtering will be
 688          *             used to select tests for execution, or MATCH_KEYWORDS if  keywords
 689          *             will be filtered according to the match mode and match value.
 690          * @see #getKeywordsMode
 691          * @see #NO_KEYWORDS
 692          * @see #MATCH_KEYWORDS
 693          */
 694         void setKeywordsMode(int mode);
 695 
 696         /**
 697          * Get a value which indicates how to interpret the match value,
 698          * when the keywords mode is set to MATCH_KEYWORDS.
 699          *
 700          * @return a value which indicates how to interpret the match value,
 701          * when the keywords mode is set to MATCH_KEYWORDS
 702          * @see #setMatchKeywords
 703          * @see #ANY_OF
 704          * @see #ALL_OF
 705          * @see #EXPR
 706          */
 707         int getMatchKeywordsMode();
 708 
 709         /**
 710          * Get a value that identifies which tests are to be selected,
 711          * when the keywords mode is set to MATCH_KEYWORDS.
 712          *
 713          * @return a value that identifies which tests are to be selected,
 714          * when the keywords mode is set to MATCH_KEYWORDS
 715          * @see #setMatchKeywords
 716          * @see #ANY_OF
 717          * @see #ALL_OF
 718          * @see #EXPR
 719          */
 720         String getMatchKeywordsValue();
 721 
 722         /**
 723          * Set how to match a tests keywords when the keywords mode
 724          * is set to MATCH_KEYWORDS.
 725          *
 726          * @param mode  A value indicating how to interpret {@code value}
