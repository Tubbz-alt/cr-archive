33 var strictParser = Parser.create("-strict");

35 
36 var withStat = <<EOF
37 
38 with({}) {}
39 EOF;
40 
41 strictParser.parse("with_stat.js", withStat, print);
42 parser.parse("with_stat1.js", withStat, print);
