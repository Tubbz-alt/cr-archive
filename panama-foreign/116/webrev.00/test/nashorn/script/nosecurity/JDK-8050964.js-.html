24 /**
25  * JDK-8050964: OptimisticTypesPersistence.java should use java.util.Date instead of java.sql.Date
26  *
27  * Make sure that nashorn.jar has only 'compact1' dependency.
28  *
29  * @test
30  * @option -scripting
31  * @run
32  */
33 
34 // assume that this script is run with "nashorn.jar" System
35 // property set to relative path of nashorn.jar from the current
36 // directory of test execution.
37 
38 if (typeof fail != 'function') {
39     fail = print;
40 }
41 
42 var System = java.lang.System;
43 var File = java.io.File;
44 var nashornJar = new File(System.getProperty("nashorn.jar"));
45 if (! nashornJar.isAbsolute()) {
46     nashornJar = new File(".", nashornJar);
47 }
48 
49 var os = System.getProperty("os.name"),
50     win = os.startsWith("Windows"),
51     jdepsName = "jdeps" + (win ? ".exe" : "");
52 
53 var javahome = System.getProperty("java.home");
54 var jdepsPath = javahome + "/../bin/".replace(/\//g, File.separator) + jdepsName;
55 if (! new File(jdepsPath).isFile()) {
56     jdepsPath = javahome + "/bin/".replace(/\//g, File.separator) + jdepsName;
57 }
58 
59 // run jdep on nashorn.jar - only summary but print profile info
60 $ENV.PWD=System.getProperty("user.dir") // to avoid RE on Cygwin
61 `${jdepsPath} -s -P ${nashornJar.absolutePath}`
62 
63 // check for "(compact1)" in output from jdep tool
64 if (! /(compact1)/.test($OUT)) {
65     fail("non-compact1 dependency: " + $OUT);
66 }
