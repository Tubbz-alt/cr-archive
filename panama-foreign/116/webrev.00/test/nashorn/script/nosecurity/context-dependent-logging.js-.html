26 /**
27  * Test that logging configuration is per engine, rather than per process.
28  *
29  * @test
30  * @bug 8036977
31  * @run/ignore-std-error
32  * @fork
33  * @option -scripting
34  */
35 
36 // To test, start another engine (testEngine) with a time logger and ensure the
37 // logger exists.
38 
39 var NashornFactory = new (Java.type('jdk.nashorn.api.scripting.NashornScriptEngineFactory'))(),
40     testEngine     = NashornFactory.getScriptEngine("-scripting", "--log=time")
41 
42 if (!testEngine.eval('$OPTIONS._loggers.time')) {
43     throw 'fresh testEngine does not have time logger'
44 }
45 
46 // To test further, have the testEngine start yet another engine (e) without
47 // time logging, but with compiler logging. Check the logging is as configured,
48 // and verify the testEngine still has time logging, but no compiler logging.
49 
50 var script = <<EOS
51     var F = new (Java.type('jdk.nashorn.api.scripting.NashornScriptEngineFactory'))(),
52         e = F.getScriptEngine('-scripting', '--log=compiler')
53     if (!e.eval('$OPTIONS._loggers.compiler')) {
54         throw 'e does not have compiler logger'
55     }
56     if (e.eval('$OPTIONS._loggers.time')) {
57         throw 'e has time logger'
58     }
59 EOS
60 
61 testEngine.eval(script)
62 
63 if (!testEngine.eval('$OPTIONS._loggers.time')) {
64     throw 'after-test testEngine does not have time logger'
65 }
66 if (testEngine.eval('$OPTIONS._loggers.compiler')) {
67     throw 'after-test testEngine has compiler logger'
68 }
