24 /**
25  * JDK-8043235: Type-based optimizations interfere with continuation methods
26  *
27  * @test
28  * @run
29  */
30 
31 function g() {
32   return "Hello World!"
33 }
34 
35 function f1() {
36     var c;
37     var paused = false;
38     // If we didn't disable nullCheck specialization for expressions
39     // containing a deoptimization site, we'd get an AssertionError
40     while (!paused && (null !== (c = g()))) {
41       print(c);
42       paused = true;
43     }
44 }
45 
46 function f2() {
47     var c;
48     var paused = false;
49     // If we didn't disable undefinedCheck specialization for expressions
50     // containing a deoptimization site, we'd get an AssertionError
51     while (!paused && (undefined !== (c = g()))) {
52       print(c);
53       paused = true;
54     }
55 }
56 
57 f1();
58 f2();
