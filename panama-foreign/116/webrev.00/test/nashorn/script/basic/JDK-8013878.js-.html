25  * JDK-8013878: ClassCastException in Regex
26  *
27  * @test
28  * @run
29  */

31 var re = /(a)(b)(c)/;
32 var str = 'abc';
33 
34 print(re.exec(str).length);
35 print(re.exec(str).concat(['d', 'e', 'f']));
36 print(re.exec(str).join('-'));
37 print(re.exec(str).push('d'));
38 print(re.exec(str).pop());
39 print(re.exec(str).reverse());
40 print(re.exec(str).shift());
41 print(re.exec(str).sort());
42 print(re.exec(str).slice(1));
43 print(re.exec(str).splice(1, 2, 'foo'));
44 print(re.exec(str).unshift('x'));
45 print(re.exec(str).indexOf('a'));
46 print(re.exec(str).lastIndexOf('a'));
47 print(re.exec(str).every(function(a) {return a.length;}));
48 print(re.exec(str).some(function(a) {return a.length;}));
49 print(re.exec(str).filter(function(a) {return a.length;}));
50 print(re.exec(str).forEach(function(a) {print(a)}));
51 print(re.exec(str).map(function(a) {return a.length;}));
52 print(re.exec(str).reduce(function(a, b) {return a + b}));
53 print(re.exec(str).reduceRight(function(a, b) {return a + b}));
