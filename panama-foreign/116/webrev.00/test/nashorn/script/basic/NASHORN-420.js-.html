24 /**
25  * NASHORN-420 : When an existing array element attributes are changed by defineProperty, element value becomes undefined
26  *
27  * @test
28  * @run
29  */
30 
31 var arrObj = [100];
32 Object.defineProperty(arrObj, "0", {
33     writable: false,
34     enumerable: false,
35     configurable: false
36 });
37 
38 var desc = Object.getOwnPropertyDescriptor(arrObj, "0");
39 if (desc.value !== arrObj[0]) {
40     fail("#1 desc.value !== arrObj[0]");
41 }
42 
43 if (desc.value !== 100) {
44     fail("#2 desc.value !== 100");
45 }
