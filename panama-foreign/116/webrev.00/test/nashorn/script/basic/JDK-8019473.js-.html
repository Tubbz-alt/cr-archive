31 function checkNoError(code) {
32     try {
33         Function(code);
34     } catch (e) {
35         print("no error expected for: " + code + " , got " + e);
36     }
37 }
38 
39 // implicit newlines at EOF should be accepted
40 checkNoError("for(;;) continue")
41 checkNoError("return")
42 checkNoError("yield")
43 checkNoError("for(;;) break")
44 
45 function checkError(code) {
46     try {
47         eval(code);
48         print("SyntaxError expected for: " + code);
49     } catch (e) {
50         if (! (e instanceof SyntaxError)) {
51             fail("SyntaxError expected, got " + e);
52         }
53     }
54 }
55 
56 checkError("function f() { case0: }");
57 checkError("function f() { if(0) }");
58 checkError("function f() { if(0); else }");
59 checkError("function f() { while(0) }");
60 
61 // comma expression as closure expression
62 checkError("function sq(x) x, x*x");
