24 /*
25  * NASHORN-444 : regex returns different result on second invocation.
26  *
27  * @test
28  * @run
29  */
30 
31 var formatRegExp = /%[sdj%]/g;
32 var format = function(f) {
33   if (typeof f !== 'string') {
34     var objects = [];
35     for (var i = 0; i < arguments.length; i++) {
36       objects.push(inspect(arguments[i]));
37     }
38     return objects.join(' ');
39   }
40 
41   var i = 1;
42   var args = arguments;
43   var len = args.length;
44   if (formatRegExp.lastIndex !== 0) throw "expected lastIndex === 0, actual: " + formatRegExp.lastIndex;
45   var str = String(f).replace(formatRegExp, function(x) {
46     if (x === '%%') return '%';
47     if (i >= len) return x;
48     switch (x) {
49       case '%s': return String(args[i++]);
50       case '%d': return Number(args[i++]);
51       case '%j': return JSON.stringify(args[i++]);
52       default:
53         return x;
54     }
55   });
56   print(i + ' '+len);
57   if (formatRegExp.lastIndex !== 0) throw "expected lastIndex === 0, actual: " + formatRegExp.lastIndex;
58   return str;
59 };
60 
61 var obj = function() {
62     return {
63         a: 1,
64         b: 2
65     };
66 };
67 
68 var o = obj();
69 print(format('obj %j', o));
70 print(format('obj %j', o));
