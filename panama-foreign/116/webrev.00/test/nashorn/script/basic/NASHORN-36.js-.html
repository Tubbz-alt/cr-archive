31 // refer to ECMA section 7.3 Line Terminators
32 // Table 3 Line Terminator Characters.
33 
34 // Because line comment ends with any of line terminators
35 // the following evals should set global 'x' with values.
36 
37 eval("// line comment\u000A x = 1;");
38 print(x); // should print 1
39 
40 eval("// line comment\u000D x = 2;");
41 print(x); // should print 2
42 
43 eval("// line comment\u2028 x = 3;");
44 print(x); // should print 3
45 
46 eval("// line comment\u2029 x = 4;");
47 print(x); // should print 4
