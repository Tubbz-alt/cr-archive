24 /**
25  * JDK-8087211: Indirect evals should be strict with -strict option
26  *
27  * @test
28  * @run
29  * @option -strict
30  */
31 
32 var global = this;
33 
34 try {
35     // indirect eval call.
36     global.eval("x = 34;");
37     throw new Error("should have thrown ReferenceError");
38 } catch (e if e instanceof ReferenceError) {
39 }
40 
41 
42 function teststrict() {
43     "use strict";
44     // strict caller, indirect eval.
45     global.eval('public = 1;');
46 }
47 
48 try {
49     teststrict();
50     throw new Error("should have thrown SyntaxError");
51 } catch (e if e instanceof SyntaxError) {
52 }
53 
54 function testnonstrict() {
55     // non strict caller, indirect eval.
56     global.eval('public = 1;');
57 }
58 
59 try {
60     testnonstrict();
61     throw new Error("should have thrown SyntaxError");
62 } catch (e if e instanceof SyntaxError) {
63 }
