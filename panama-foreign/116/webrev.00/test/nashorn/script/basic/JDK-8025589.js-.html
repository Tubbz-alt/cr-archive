24 /**
25  * JDK-8025589:  Array.prototype.shift should only copy defined elements in generic mode
26  *
27  * @test
28  * @run
29  */
30 
31 var s = {length: 4, 2: 1};
32 Array.prototype.shift.call(s);
33 
34 if (s.length != 3) {
35     fail("s.length != 3");
36 }
37 if (0 in s) {
38     fail("0 in s");
39 }
40 if (s.hasOwnProperty(0)) {
41     fail("s.hasOwnProperty(0)");
42 }
43 if (s[1] !== 1) {
44     fail("s[1] !== 1");
45 }
46 if (2 in s) {
47     fail("2 in s");
48 }
49 if (s.hasOwnProperty(2)) {
50     fail("s.hasOwnProperty(2)");
51 }
