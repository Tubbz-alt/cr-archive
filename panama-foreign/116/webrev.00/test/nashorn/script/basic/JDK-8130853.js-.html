51 function checkLoad(code) {
52     try {
53         load({ name: "test", script: code });
54         throw new Error("should have thrown TypeError for load: " + code);
55     } catch (e) {
56         if (! (e instanceof TypeError)) {
57             throw e;
58         }
59     }
60 }

63 checkLoad("function f() {}");
