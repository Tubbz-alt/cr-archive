24 /**
25  * JDK-8008197: Cross script engine function calls do not work as expected
26  *
27  * @test
28  * @run
29  */
30 
31 
32 var m = new javax.script.ScriptEngineManager();
33 var e = m.getEngineByName("nashorn");
34 
35 var obj = {
36     func: function(str) {
37         return /hello/.exec(str);
38     }
39 };
40 
41 // set our object as object to the engine created
42 e.put("obj", obj);
43 
44 // try to invoke method from the other engine
45 if (e.eval("obj.func('hello')") == null) {
46     fail("FAILED: #1 obj.func('hello') returns null");
47 }
48 
49 // define an object in the engine
50  e.eval("var foo = { callMe: function(callback) { return callback() } }");
51 
52 // try to invoke a script method from here but callback is from this engine
53 var res = e.invokeMethod(e.get("foo"), "callMe", function() {
54     return /nashorn/;
55 });
56 
57 if (! res.exec("nashorn")) {
58     fail("FAILED: #2 /nashorn/ does not match 'nashorn'");
59 }
60 
61 // invoke a script method from here with callback from this engine.
62 // This uses JSObject.call interface
63 res = e.get("foo").callMe(function() {
64     return /ecmascript/;
65 });
66 
67 if (! res.exec("ecmascript")) {
68     fail("FAILED: #3 /ecmascript/ does not match 'ecmascript'");
69 }
