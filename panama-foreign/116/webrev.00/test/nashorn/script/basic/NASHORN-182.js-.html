24 /**
25  * NASHORN-182 :  Array.prototype.reduce does not exit early when a getter function throws error.
26  *
27  * @test
28  * @run
29  */
30 
31 function callback(prev, cur, idx, obj) {
32     fail("callback with " + idx);
33 }
34 
35 // array-like object
36 var obj = { 1: 1, 2: 2, length: 3 };
37 
38 Object.defineProperty(obj, "0", {
39     get: function () {
40         throw new RangeError("exception from getter for '0'");
41     },
42  });
43 
44 obj.reduce = Array.prototype.reduce;
45 
46 try {
47     obj.reduce(callback);
48     fail("should have thrown RangeError");
49 } catch (ex) {
50     if (! (ex instanceof RangeError)) {
51         fail("RangeException expected, got " + ex);
52     }
53 }
