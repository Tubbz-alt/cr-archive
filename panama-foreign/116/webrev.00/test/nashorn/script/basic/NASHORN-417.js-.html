24 /**
25  * NASHORN-417 : Accessor descriptor property with undefined getter and setter is reported as data descriptor property
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = {};
32 Object.defineProperty(obj, "foo", {
33     get: undefined, set: undefined });
34 
35 var desc = Object.getOwnPropertyDescriptor(obj, "foo");
36 if (! ('get' in desc)) {
37     fail("#1 desc does not have 'get'");
38 }
39 
40 if (! ('set' in desc)) {
41     fail("#2 desc does not have 'set'");
42 }
43 
44 if ('value' in desc) {
45     fail("#3 desc has 'value'");
46 }
