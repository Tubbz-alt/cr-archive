24 /**
25  * There was a bug in the old Lower that didn't fully propagate type information
26  * by way of assignment. 'q' in the example below would have finished a double
27  * even though it can get an object value through the assignment 'q = l'
28  *
29  * Furthermore, this caused type coercion to be done at q = l, and not a q = q * 2,
30  * which is a bug. This test ensures it happens in the correct order
31  *
32  * @test
33  * @run
34  */
35 
36 function createObject() {
37     var obj = { valueOf: function() { print("toNumber coercion"); return 17; }}
38     return obj;
39 }
40 
41 function f() {
42     var l = 1.2; //number
43     var q = 2.3; //number
44     for (var i = 0; i < 2; i++) {
45     q = l; // q = toNumber(l), no coercion here
46     print("assignment done");
47     q = q * 2; // q = q * 2, coercion here
48     print("multiplication done");
49     l = createObject();
50     }
51 }
52 
53 f();
