24 /**
25  * JDK-8015959: Can't call foreign constructor
26  *
27  * @test
28  * @run
29  */
30 
31 function check(global) {
32     var obj = new global.Point(344, 12);
33     print("obj.x " + obj.x);
34     print("obj.y " + obj.y);
35     print("obj instanceof global.Point? " + (obj instanceof global.Point))
36 
37     var P = global.Point;
38     var p = new P(343, 54);
39     print("p.x " + p.x);
40     print("p.y " + p.y);
41     print("p instanceof P? " + (p instanceof P))
42 }
43 
44 print("check with loadWithNewGlobal");
45 check(loadWithNewGlobal({
46    name: "myscript",
47    script: "function Point(x, y) { this.x = x; this.y = y }; this"
48 }));
49 
50 print("check with script engine");
51 var m = new javax.script.ScriptEngineManager();
52 var e = m.getEngineByName('nashorn');
53 check(e.eval("function Point(x, y) { this.x = x; this.y = y }; this"));
