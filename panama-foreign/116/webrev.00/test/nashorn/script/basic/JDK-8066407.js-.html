24 /**
25  * JDK-8066407: Function with same body not reparsed after SyntaxError
26  *
27  * @test
28  * @run
29  */
30 
31 function testFunction(body) {
32     try {
33         Function(body);
34         Assert.fail("Should have thrown");
35     } catch (e) {
36         Assert.assertEquals(e.name, "SyntaxError");
37         count++;
38     }
39 
40 }
41 
42 var count = 0;
43 
44 testFunction("/a/r");
45 testFunction("/a/r");
46 testFunction("/a/r");
47 
48 Assert.assertTrue(count === 3);
