24 /**
25  * NASHORN-421 : Many Array.prototype functions miss ToObject conversion of 'self' as required by specification
26  *
27  * @test
28  * @run
29  */
30 
31 function checkUndefined(name) {
32    try {
33        Array.prototype[name].call(undefined);
34        fail( name + " - no TypeError for undefined 'this'");
35        return false;
36     } catch (e) {
37        if (! (e instanceof TypeError)) {
38            throw e;
39        }
40     }
41 }
42 
43 function checkNull(name) {
44    try {
45        Array.prototype[name].call(null);
46        fail(name + " - no TypeError for null 'this'");
47        return false;
48     } catch (e) {
49        if (! (e instanceof TypeError)) {
50            throw e;
51        }
52     }
53 }
54 
55 var methods = [ "concat", "join", "pop", "push", "reverse",
56     "shift", "unshift", "slice", "sort", "splice",
57     "indexOf", "lastIndexOf", "every", "some", "forEach",
58     "map", "filter", "reduce", "reduceRight" ];
59 
60 for (var m in methods) {
61     checkUndefined(methods[m]);
62     checkNull(methods[m]);
63 }
