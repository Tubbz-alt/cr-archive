24 /**
25  * JDK-8066214: Fuzzing bug: Object.prototype.toLocaleString(0)
26  *
27  * @test
28  * @run
29  */
30 
31 function test(func) {
32     print(func.call(0));
33     print(func.call("abc"));
34     print(func.call(true));
35     try {
36         print(func.call(undefined));
37     } catch (e) {
38         print(e);
39     }
40     try {
41         print(func.call(null));
42     } catch (e) {
43         print(e);
44     }
45 }
46 
47 test(Object.prototype.toLocaleString);
48 test(Object.prototype.toString);
49 test(Object.prototype.valueOf);
