24 /**
25  * JDK-8058422: Users should be able to overwrite "context" and "engine" variables
26  *
27  * @test
28  * @run
29  */
30 
31 var m = new javax.script.ScriptEngineManager();
32 var e = m.getEngineByName("nashorn");
33 e.put("foo", "hello");
34 var obj = e.eval("context.getAttribute('foo')");
35 if (obj != "hello") {
36     fail("Expected 'obj' to be 'hello'");
37 }
38 
39 e.put("context", "bar");
40 if (e.eval("context") != "bar") {
41     fail("Expected 'context' to be 'bar'");
42 }
43 
44 if (e.eval("foo") != "hello") {
45     fail("Expected 'foo' to be 'hello'");
46 }
47 
48 if (e.eval("engine") != e) {
49     fail("'engine' is not evaluaed to current engine");
50 }
51 
52 e.put("engine", "foobar");
53 if (e.eval("engine") != "foobar") {
54     fail("'engine' is not evalued to 'foobar'");
55 }
