24 /**
25  * NASHORN-183 :  Array.prototype.reduce does not work for array-like objects
26  *
27  * @test
28  * @run
29  */
30 
31 function callback(acc, newVal) {
32     return acc + newVal;
33 }
34 
35 function func() {
36     return Array.prototype.reduce.call(arguments, callback);
37 }
38 
39 var val = func(232, 334, 454);
40 if (val != (232 + 334 + 454)) {
41     fail("expecting " + (232 + 334 + 454));
42 }
43 
44 var arr = [3, 4, 5, 6];
45 var obj = Object.create(arr);
46 val = Array.prototype.reduce.call(obj, callback);
47 
48 var val2 = 0;
49 for (i in arr) {
50     val2 += arr[i];
51 }
52 
53 if (val != val2) {
54     fail("val != val2, val2 is " + val2);
55 }
