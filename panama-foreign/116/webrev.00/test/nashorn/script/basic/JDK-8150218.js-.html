40     Java.to([f], Java.type("java.lang.Runnable[]"))[0].run()

43     Java.to([[[f]]], Java.type("java.lang.Runnable[][][]"))[0][0][0].run()
44 
45     var rt = new (Java.type("jdk.dynalink.test.ArrayRunnableTest"))
46     var n = "runnables"
47 
48     // Explicit array passed to vararg setter chosen at run time
49     rt[n] = [f]

51 
52     // Automatically packed one-element array passed to vararg setter chosen at run time
53     rt[n] = f
54     rt.firstRunnable.run()
55 
56     // Explicit array passed to vararg method overload chosen at run time
57     rt.setRunnablesOverloaded([f])
58     rt.firstRunnable.run()

61     rt.setRunnablesOverloaded(f)
