24 /**
25  * NASHORN-245 : regression : StackOverflowError when accessing inherited accessor property
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = {};
32 Object.defineProperty(obj, "name", {
33     get: function () {
34         return 32;
35     }
36 });
37 
38 var obj2 = Object.create(obj);
39 
40 var obj3 = Object.create(obj2);
41 
42 // StackOverflowError on obj3.name
43 if (obj3.name !== 32) {
44     fail("obj3.name !== 32");
45 }
