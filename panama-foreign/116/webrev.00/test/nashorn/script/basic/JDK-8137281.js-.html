24 /**
25  * JDK-8137281: OutOfMemoryError with large numeric keys in JSON.parse
26  *
27  * @test
28  * @run
29  */
30 
31 function createObject(startKey, level1, level2) {
32     var root = {};
33     var key = startKey;
34     for (var i = 0; i < level1; i++) {
35         var child = {};
36         for (var j = 0; j < level2; j++) {
37             child[key++] = {};
38         }
39         root[key++] = child;
40     }
41     return root;
42 }
43 
44 JSON.parse(JSON.stringify(createObject(500000, 20, 20)));
45 JSON.parse(JSON.stringify(createObject(1000000, 20, 20)));
46 JSON.parse(JSON.stringify(createObject(2000000, 20, 20)));
47 JSON.parse(JSON.stringify(createObject(4000000, 20, 20)));
48 JSON.parse(JSON.stringify(createObject(8000000, 20, 20)));
49 JSON.parse(JSON.stringify(createObject(16000000, 20, 20)));
