24 /**
25  * JDK-8041995: optimistic object property maps were only updated if the outermost program
26  * point in a property setter failed, not an inner one, which is wrong.
27  *
28  * @test
29  * @run
30  */
31 
32 function xyzzy() {
33     return 17.4711;
34 }
35 var obj = {
36     z: -xyzzy()
37 };
38 print(obj.z);
39 
40 function phlug() {
41     var obj = {
42     4: -Infinity,
43      5: Infinity,
44     length: 5 - Math.pow(2, 32)
45     };
46 
47     return Array.prototype.lastIndexOf.call(obj, -Infinity) === 4;
48 }
49 
50 var d = new Date;
51 print(phlug());
52 var d2 = new Date - d;
53 print(d2 < 5000); // if this takes more than five seconds we have read the double length as an int
54

60     length4: 2 - Math.pow(2, 32),
61     length5: 1 - Math.pow(2, 32),
62     length6: Math.pow(2, 32)
63     };
64     for (var i in obj) {

66     }
