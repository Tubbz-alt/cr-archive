24 /**
25  * Tests for java.util.Map behavior in Nashorn
26  *
27  * @test
28  * @run
29  */
30 var m = new (Java.type("java.util.LinkedHashMap"));
31 print("m.class.name=" + Java.typeName(m.class)) // Has "class" property like any POJO
32 
33 var empty_key = "empty"
34 
35 print("m = " + m)
36 print("m.empty = " + m.empty) // prints "true"
37 print("m['empty'] = " + m['empty']) // prints "true"; item not found, default to property
38 print("m[empty_key] = " + m[empty_key]) // prints "true"; item not found, default to property
39 
40 m.put("empty", "foo")
41 
42 print("m = " + m)
43 print("m.empty = " + m.empty) // prints "false"
44 print("m['empty'] = " + m['empty'])
45 print("m[empty_key] = " + m[empty_key]) // prints "foo"
46 
47 print("m.bwah = " + m.bwah) // prints "undefined"
48 print("m['bwah'] = " + m['bwah']) // prints "undefined"
49 
50 m.put("twonk", "ding")
51 print("m.twonk = " + m.twonk) // prints "ding"
52 print("m['twonk'] = " + m['twonk']) // prints "ding"
53 
54 print("m.size()=" + m.size())
55 
56 print("--for each begin--")
57 for each (i in m.keySet()) {
58   print(i)
59 }
60 print("--for each end--")
