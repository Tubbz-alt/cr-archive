24 /**
25  * JDK-8075448: nashorn parser API returns init variable tree object of a for
26  * loop after for loop statement tree object 
27  *
28  * @test
29  * @option -scripting
30  * @run
31  */
32 
33 var Parser = Java.type("jdk.nashorn.api.tree.Parser");
34 var ForLoopTree = Java.type("jdk.nashorn.api.tree.ForLoopTree");
35 var VariableTree = Java.type("jdk.nashorn.api.tree.VariableTree");
36 var parser = Parser.create();
37 
38 var code = <<EOF
39 for (var i = 0; i < 10; i++)
40     print("hello");
41 EOF;
42 
43 var ast = parser.parse("test.js", code, print);
44 var stats = ast.sourceElements;
45 
46 Assert.assertTrue(stats[0] instanceof VariableTree);
47 Assert.assertEquals(stats[0].binding.name, "i");
48 Assert.assertTrue(stats[1] instanceof ForLoopTree);
