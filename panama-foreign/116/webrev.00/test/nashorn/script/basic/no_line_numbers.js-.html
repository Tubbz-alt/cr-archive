 73 function terminal() {
 74     print("r = "+r);
 75     print("t = "+t);
 76     for (;;) {
 77     var r;
 78     return;
 79     var t;
 80     print("THIS SHOULD NEVER BE PRINTED!");
 81     }
 82     print("NEITHER SHOULD THIS");
 83 }
