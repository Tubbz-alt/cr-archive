24 /**
25  * JDK-8019783: typeof does not work properly for java methods and foreign objects
26  *
27  * @test
28  * @run
29  */
30 
31 function printTypeof(str) {
32     print("typeof(" + str + ") =  " + eval('typeof ' + str));
33 }
34 
35 // Java methods
36 printTypeof("java.lang.System.exit");
37 printTypeof("java.lang.System['exit']");
38 // full signature
39 printTypeof("java.lang.System['exit(int)']");
40 // overloaded method
41 printTypeof("java.security.AccessController.doPrivileged");
42 printTypeof("java.security.AccessController['doPrivileged']");
43 
44 // foreign objects
45 var global = loadWithNewGlobal({ name: "t", script: "this" });
46 print("typeof(global.Object) = " + (typeof global.Object));
47 print("typeof(new global.Object()) = " + (typeof (new global.Object())));
48 
49 // foreign engine objects
50 var m = new javax.script.ScriptEngineManager();
51 var engine = m.getEngineByName("nashorn");
52 var engineGlobal = engine.eval("this");
53 
54 print("typeof(engineGlobal.Object) = " + (typeof engineGlobal.Object));
55 print("typeof(new engineGlobal.Object()) = " + (typeof (new engineGlobal.Object())));
