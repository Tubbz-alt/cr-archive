31 'use strict';
32 
33 var a = {
34             _x : "default",
35             get x() {
36                 return this._x;
37             },
38             set x(x) {
39                 this._x = x;
40             }
41         };
42 
43 var b = {
44             _x : "default",
45             get x() {
46                 return this._x;
47             }
48         };
49 
50 
51 var c = {
52             _x : "default",
53             set x(x) {
54                 this._x = x;
55             }
56         };
57 
58 print(a._x);
59 print(a.x);
60 a.x = "new value";
61 print(a._x);
62 print(a.x);
63 
64 print(b._x);
65 print(b.x);
66 try {
67     b.x = "new value";
68 } catch (e) {
69     print(e);
70 }
71 print(b._x);
72 print(b.x);
73 
74 print(c._x);
75 print(c.x);
76 c.x = "new value";
77 print(c._x);
78 print(c.x);
