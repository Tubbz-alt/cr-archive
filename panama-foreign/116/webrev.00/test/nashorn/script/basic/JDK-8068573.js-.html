24 /**
25  * JDK-8068573: POJO setter using [] syntax throws an exception
26  *
27  * @test
28  * @run
29  */
30 
31 // Invoke a setter using []. It's important that the setter returns void.
32 var pb = new (Java.type("jdk.nashorn.test.models.PropertyBind"))
33 var n = "writeOnly";
34 pb[n] = 2;
35 Assert.assertEquals(pb.peekWriteOnly(), 2);
36 
37 // Invoke an overloaded setter using []. It's important that one of the 
38 // overloads returns void.
39 var os = new (Java.type("jdk.nashorn.test.models.OverloadedSetter"))
40 var n2 = "color";
41 os[n2] = 3; // exercise int overload
42 Assert.assertEquals(os.peekColor(), "3");
43 os[n2] = "blue";  // exercise string overload
44 Assert.assertEquals(os.peekColor(), "blue");
45 for each(var x in [42, "42"]) {
46   os[n2] = x; // exercise both overloads in the same call site
47   Assert.assertEquals(os.peekColor(), "42");
48 }
49 
50 // Invoke an overloaded method using [], repeatedly in the same call 
51 // site. It's important that one of the overloads returns void.
52 var n3="foo";
53 var param=["xyz", 1, "zyx", 2];
54 var expected=["boo", void 0, "boo", void 0];
55 for(var i in param) {
56   Assert.assertEquals(os[n3](param[i]), expected[i]);
57 }
