24 /**
25  * JDK-8025434: RegExp lastIndex can exceed int range
26  *
27  * @test
28  * @run
29  */
30 
31 var r = /a/g;
32 
33 r.lastIndex = 0x100000000;
34 if (r.test("a")) {
35     throw new Error("Expected no match");
36 }
37 
38 r.lastIndex = 0x100000000000000;
39 if (r.test("a")) {
40     throw new Error("Expected no match");
41 }
42 
43 r.lastIndex = -0x100000000;
44 if (r.test("a")) {
45     throw new Error("Expected match");
46 }
47 
48 r.lastIndex = -0x100000000000000;
49 if (r.test("a")) {
50     throw new Error("Expected no match");
51 }
52 
53 r.lastIndex = 1;
54 if (r.test("a")) {
55     throw new Error("Expected no match");
56 }
57 
58 r.lastIndex = -1;
59 if (r.test("a")) {
60     throw new Error("Expected no match");
61 }
62 
63 r.lastIndex = 0;
64 if (!r.test("a")) {
65     throw new Error("Expected match");
66 }
