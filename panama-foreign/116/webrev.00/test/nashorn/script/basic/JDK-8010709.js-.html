24 /**
25  * JDK-8010709  org on the top level doesn't resolve
26  *
27  * @test
28  * @run
29  */
30 
31 function check(pkgName) {
32     if (typeof this[pkgName] != 'object') {
33         fail(pkgName + " not defined");
34     }
35 
36     if (String(this[pkgName]) != '[JavaPackage ' + pkgName + ']') {
37         fail(pkgName + " is not a JavaPackage");
38     }
39 }
40 
41 check("com");
42 check("edu");
43 check("java");
44 check("javafx");
45 check("javax");
46 check("org");
