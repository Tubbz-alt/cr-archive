24 /**
25  * JDK-8015346: JSON parsing issues with escaped strings, octal, decimal numbers *
26  * @test
27  * @run
28  */
29 
30 function checkJSON(str) {
31     try {
32         JSON.parse(str);
33         fail("should have thrown SyntaxError for JSON.parse on " + str);
34     } catch (e) {
35         if (! (e instanceof SyntaxError)) {
36             fail("Expected SyntaxError, but got " + e);
37         }
38     }
39 }
40 
41 // invalid escape in a string
42 checkJSON('"\\a"')
43 
44 // invalid floating point number patterns
45 checkJSON("1.")
46 checkJSON(".8")
47 checkJSON("2.3e+")
48 checkJSON("0.3E+")
49 
50 // octal, hexadecimal not allowed
51 checkJSON("08")
52 checkJSON("06")
53 checkJSON('0x3')
