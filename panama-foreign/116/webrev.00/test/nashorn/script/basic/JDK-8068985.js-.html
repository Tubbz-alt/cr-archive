24 /**
25  * JDK-8068985: Wrong 'this' bound to eval call within a function when caller's 'this' is a Java object
26  *
27  * @test
28  * @run
29  */
30 
31 function func(arg) {
32   (function() { print(eval('this')); }).call(arg);
33 }
34 
35 // primitives
36 func(undefined);
37 func(null);
38 func(34.23);
39 func("hello");
40 func(false);
41 
42 // script objects
43 func(this);
44 func({});
45 func({ toString: function() { return "foo" } });
46 
47 // java objects
48 func(new java.util.Vector());
49 var m = new java.util.HashMap();
50 m.put("foo", "bar");
51 func(m);
