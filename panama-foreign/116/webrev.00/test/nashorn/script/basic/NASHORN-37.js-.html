24 /**
25  * NASHORN-37 :  object and array properties defined with special keys can be accessed be by special or string keys
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = {};
32 
33 obj[true] = "hello";
34 print(obj["true"]);
35 print(obj[true]);
36 
37 obj["false"] = 33;
38 print(obj[false]);
39 print(obj['false']);
40 
41 obj[null] = "foo";
42 print(obj["null"]);
43 print(obj[null]);
44 
45 obj[undefined] = "bar";
46 print(obj["undefined"]);
47 print(obj["undefined"]);
48 
49 obj[33] = 343;
50 print(obj[33]);
51 print(obj['33']);
52 
53 obj['2'] = 3.14;
54 print(obj[2]);
55 print(obj['2']);
56 
57 var key = {
58     toString: function() {
59         print("key.toString called");
60         return 'key';
61     }
62 };
63 
64 obj[key] = 'value';
65 print(obj[key]);
66 print(obj['key']);
67 
68 var array = [ "foo", "bar" ];
69 
70 print(array[0]);
71 print(array["0"]);
72 
73 print(array[1]);
74 print(array["1"]);
