24 /**
25  * JDK-8023368: Instance __proto__ property should exist and be writable.
26  *
27  * @test
28  * @run
29  */
30 
31 // check Object.setPrototypeOf extension rather than using __proto__
32 
33 // function to force same callsites
34 function check(obj) {
35     print(obj.func());
36     print(obj.x);
37     print(obj.toString());
38 }
39 
40 function Func() {
41 }
42 
43 Func.prototype.func = function() {
44     return "Func.prototype.func";
45 }
46 
47 Func.prototype.x = "hello";
48 
49 var obj = new Func();
50 var obj2 = Object.create(obj);
51 
52 // check direct and indirect __proto__ change
53 check(obj);
54 check(obj2);
55 Object.setPrototypeOf(obj, {
56    func: function() {
57         return "obj.__proto__.func @ " + __LINE__;
58    },
59    x: 344
60 });
61 
62 check(obj);
63 check(obj2);
64 
65 // check indirect (1 and 2 levels) __proto__ function change
66 Object.setPrototypeOf(Object.getPrototypeOf(obj), {
67     toString: function() {
68         return "new object.toString";
69     }
70 });
71 
72 check(obj);
73 check(obj2);
