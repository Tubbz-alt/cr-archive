24 /**
25  * JDK-8055796: JSObject and browser JSObject linkers should provide fallback to call underlying Java methods directly
26  *
27  * @test
28  * @option -scripting
29  * @run
30  */
31 
32 function main() {
33     var JSObject;
34     try {
35         JSObject = Java.type("netscape.javascript.JSObject");
36     } catch (e) {
37         if (e instanceof java.lang.ClassNotFoundException) {
38             // pass vacuously by emitting the .EXPECTED file content
39             var str = readFully(__DIR__ + "JDK-8055796_2.js.EXPECTED");
40             print(str.substring(0, str.length - 1));
41             return;
42         } else {
43             fail("unexpected exception on JSObject", e);
44         }
45     }
46     test(JSObject);
47 }
48 
49 function test(JSObject) {
50     var bjsobj = new (Java.extend(JSObject))() {
51         getMember: function(name) {
52             if (name == "func") {
53                 return function(arg) {
54                     print("func called with " + arg);
55                 }
56             }
57             return name.toUpperCase();
58         },
59 
60         getSlot: function(index) {
61             return index*index;
62         },
63 
64         setMember: function(name, value) {
65             print(name + " set to " + value);
66         },
67 
68         setSlot: function(index, value) {
69             print("[" + index + "] set to " + value);
70         }
71     };
72 
73     print("getMember('foo') =", bjsobj['getMember(String)']('foo'));
74     print("getSlot(6) =", bjsobj['getSlot(int)'](6));
75     bjsobj['setMember(String, Object)']('bar', 'hello');
76     bjsobj['setSlot(int, Object)'](10, 42);
77 }
78 
79 main();
