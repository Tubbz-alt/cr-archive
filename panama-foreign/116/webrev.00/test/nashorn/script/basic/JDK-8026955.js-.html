24 /**
25  * JDK-8026955: for-in should convert primitive values to object
26  *
27  * @test
28  * @run
29  */
30 
31 Object.prototype[4] = "world";
32 String.prototype[3] = "hello";
33 Number.prototype[3] = "hello";
34 Boolean.prototype[3] = "hello";
35 
36 function testForIn(x) {
37     for (var i in x) {
38         print(i, x[i]);
39     }
40     for each (var i in x) {
41         print(i);
42     }
43 }
44 
45 testForIn("abc");
46 testForIn(false);
47 testForIn(3);
48 testForIn(null);
49 testForIn();
50 testForIn(String.prototype);
