31 "use strict"
32 
33 function printArray(a,n) {
34     print("PRINT_ARRAY CALLED: length = " + a.length);
35     print();
36 
37     print("INDEXED");
38     for (var x = 0; x<n; x++) {
39 	print("\t" + x + ":"+a[x]);
40     }
41     print("KEYS");
42     for (var key in a) {
43 	print("\t" + key + ";" + a[key]);
44     }
45 }
46 
47 var b = [1,2,3];
48 
49 Object.defineProperty(b, "length", { writable: false });
50 var high = 8;
51 b[high] = high;
52 
53 printArray(b, high + 5);
54 
55 var c = [1,2,3];
56 c[high] = high;
57 print();
58 print("element[" + high + "]: " + c.length + " " + c[high]);
59 print("Resetting length");
60 c.length = 3;
61 print("element[" + high + "]: " + c.length + " " + c[high]);
62 print();
63 
64 printArray(c, high + 5);
65 print();
