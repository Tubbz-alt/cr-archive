24 /**
25  * NASHORN-397 : typeof on certain member access expressions computes to undefined wrongly
26  *
27  * @test
28  * @run
29  */
30 
31 Object.defineProperty(Number.prototype, 'x',
32     { get : function() { return 42; } });
33 
34 if (typeof (5).x !== 'number') {
35     fail("typeof(5).x is not 'number'");
36 }
37 
38 if (typeof (java.net.Proxy.NO_PROXY) != 'object') {
39     fail("typeof java.net.Proxy.NO_PROXY is not 'object'");
40 }
41 
42 if (typeof (java.lang.Math.PI) != 'number') {
43     fail("typeof java.lang.Math.PI is not 'number'");
44 }
45 
46 if (typeof (java.io.File.separator) != 'string') {
