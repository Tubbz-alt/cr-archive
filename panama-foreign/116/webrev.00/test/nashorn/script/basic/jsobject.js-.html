24 /**
25  * JSObject tests
26  *
27  * @test
28  * @run
29  */

31 var m = new javax.script.ScriptEngineManager();
32 var e = m.getEngineByName("nashorn");
33 
34 e.eval("obj = { foo:'hello', 0: 'world', func: function(x) { return x.toUpperCase() } } ");
35 var obj = e.get("obj");
36 
37 
38 // try various getters
39 if (obj.foo != 'hello') {
40     fail("obj.foo does have expected value");
41 }
42 
43 function checkPropGetter(obj, prop, expected) {
44     if (obj[prop] != expected) {
45         fail(prop + " does not have value: " + expected);
46     }
47 }
48 
49 checkPropGetter(obj, "foo", "hello");
50 checkPropGetter(obj, 0, "world");
51 checkPropGetter(obj, "0", "world");
52 
53 // try various setters
54 
55 obj.foo = "HELLO";
56 if (obj.foo != "HELLO") {
57     fail("obj.foo set does not work as expected");
58 }
59 
60 function checkPropSetter(obj, prop, newValue) {
61     obj[prop] = newValue;
62     checkPropGetter(obj, prop, newValue);
63 }
64 
65 checkPropSetter(obj, "foo", "NASHORN");
66 checkPropSetter(obj, 0, "ECMASCRIPT");
67 checkPropSetter(obj, "0", "CHANGED");
68 
69 function callFunc(input, expected) {
70    if (obj.func(input) != expected) {
71        fail("obj.func(..) does not work as expected");
72    }
73 }
74 
75 callFunc("nashorn", "NASHORN");
76 callFunc("javascript", "JAVASCRIPT");
77 callFunc("hello", "HELLO");
78 
79 var Func = obj.func;
80 
81 function callWithoutObject(input, expected) {
82    if (Func(input) != expected) {
83        fail("obj.func(..) does not work as expected");
84    }
85 }
86 
87 callWithoutObject("nashorn", "NASHORN");
88 callWithoutObject("javascript", "JAVASCRIPT");
89 callWithoutObject("hello", "HELLO");
