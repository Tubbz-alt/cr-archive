24 /**
25  * JDK-8058179: Global constants get in the way of self-modifying properties
26  *
27  * @test
28  * @run
29  */
30 
31 var global = this;
32 
33 Object.defineProperty(global, "value", {
34     get: function() {
35         print("getting value");
36         global["value"] = "value 2";
37         return "value 1";
38     },
39     set: function(value) {
40         print("setting value: " + value);
41         delete global["value"];
42         global["value"] = value;
43     },
44     configurable: true
45 });
46 
47 print(value);
48 print(value);
