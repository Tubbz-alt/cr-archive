24 /**
25  * Check important prototype chain properties of Function and Object.
26  *
27  * @test
28  * @run
29  */
30 
31 print(typeof(Object) == 'function');
32 print(typeof(Object.prototype) == 'object');
33 print(typeof(Function) == 'function');
34 // Function's prototype is a function!
35 print(typeof(Function.prototype) == 'function');
36 
37 print(Function.constructor === Function);
38 print(Function.prototype === Object.getPrototypeOf(Function));
39 print(Object.getPrototypeOf(Function.prototype) === Object.prototype);
40 
41 print(Object.constructor === Function);
42 print(Object.getPrototypeOf(Object.prototype) === null);
43 print(Object.getPrototypeOf(Object) === Function.prototype);
44 
45 // check one function from Function.prototype
46 var applyFunc = Function.prototype.apply;
47 print(applyFunc.constructor === Function);
48 print(Object.getPrototypeOf(applyFunc) === Function.prototype);
49 print(applyFunc.prototype === undefined);
50 
51 // check one function from Object.prototype
52 var toStringFunc = Object.prototype.toString;
53 print(toStringFunc.constructor === Function);
54 print(Object.getPrototypeOf(toStringFunc) === Function.prototype);
55 print(toStringFunc.prototype === undefined);
