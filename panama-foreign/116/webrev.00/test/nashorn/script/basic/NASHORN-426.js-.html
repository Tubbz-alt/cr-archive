24 /**
25  * NASHORN-426 : User accessor property modification does not work as expected
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = {};
32 
33 function getter() {
34    return 12;
35 }
36 
37 var setterCalled = false;
38 function setter(value) {
39     setterCalled = true;
40 }
41 
42 Object.defineProperty(obj, "foo", {
43     get: getter,
44     configurable: true
45 });
46 
47 Object.defineProperty(obj, "foo", {
48     set: setter
49 });
50 
51 obj.foo = 44;
52 if (! setterCalled) {
53     fail("setter not called");
54 }
