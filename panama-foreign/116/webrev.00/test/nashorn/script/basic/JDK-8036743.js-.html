24 /**
25  * JDK-8036743: need ArrayBuffer constructor with specified data
26  *
27  * @test
28  * @run
29  */
30 
31 var ByteArray = Java.type("byte[]");
32 var ByteBuffer = Java.type("java.nio.ByteBuffer");
33 
34 var ba = new ByteArray(4);
35 // use constructor that accepts ByteBuffer as first arg.
36 var abuf = new ArrayBuffer(ByteBuffer.wrap(ba));
37 print("abuf.byteLength = " + abuf.byteLength);
38 var view = new DataView(abuf);
39 
40 function setAndPrint(value, endian) {
41     view.setInt32(0, value, endian);
42     print("Little endian? " + endian);
43     print("view[0] = " + view.getInt32(0, endian));
44     print("ba[0] = " + ba[0]);
45     print("ba[1] = " + ba[1]);
46     print("ba[2] = " + ba[2]);
47     print("ba[3] = " + ba[3]);
48 }
49 
50 setAndPrint(42, true);
51 setAndPrint(42, false);
52 setAndPrint(java.lang.Byte.MAX_VALUE, true);
53 setAndPrint(java.lang.Byte.MAX_VALUE, false);
54 setAndPrint(java.lang.Short.MAX_VALUE, true);
55 setAndPrint(java.lang.Short.MAX_VALUE, false);
56 setAndPrint(java.lang.Integer.MAX_VALUE, true);
57 setAndPrint(java.lang.Integer.MAX_VALUE, false);
