24 /**
25  * JDK-8026302: source representation of getter and setter methods is wrong
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = {
32     get "foo"() {},
33     set "foo"(x) {},
34     get bar() {},
35     set bar(x) {},
36     get ":"() {},
37     set ":"(x) {},
38     get 12() {},
39     set 12(x) {},
40     get 1.8e-8() {},
41     set 1.8e-8(x) {}
42 }
43 
44 for (var prop in obj) {
45     var desc = Object.getOwnPropertyDescriptor(obj, prop);
46     print(desc.get);
47     print(desc.set);
48 }
