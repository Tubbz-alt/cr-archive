24 /**
25  * JDK-8011974: Comparator function returning negative and positive Infinity does not work as expected with Array.prototype.sort
26  *
27  * @test
28  * @run
29  */
30 
31 function compare(x, y) {
32     return x < y? -Infinity : (x > y? Infinity: 0)
33 }
34 
35 var sorted =  [5, 4, 3, 2, 1].sort(compare);
36 
37 if (sorted + '' != "1,2,3,4,5") {
38     fail("Array.prototype.sort does not work when compare returns +/-Infinity");
39 }
