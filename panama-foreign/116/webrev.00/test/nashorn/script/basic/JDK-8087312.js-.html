24 /**
25  * JDK-8087312: PropertyMapWrapper.equals should compare className
26  *
27  * @test
28  * @run
29  * @fork
30  * @option -Dnashorn.debug=true
31  */
32 
33 function createObject(type) {
34     // we want to make sure two different object literals with the same keys and types share the same property map.
35     if (type) {
36         return {
37             a: "a",
38             b: 1,
39             c: 0.1
40         }
41     } else {
42         return {
43             a: "x",
44             b: 10,
45             c: 3.4
46         }
47     }
48 }
49 
50 var o1 = createObject(false);
51 var o2 = createObject(true);
52 Assert.assertTrue(Debug.map(o1) === Debug.map(o2));
