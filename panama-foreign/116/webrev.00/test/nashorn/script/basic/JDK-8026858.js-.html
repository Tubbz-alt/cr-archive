24 /**
25  * JDK-8026858: Array length does not handle defined properties correctly
26  *
27  * @test
28  * @run
29  */
30 
31 var arr = [];
32 
33 Object.defineProperty(arr, "3", {value: 1 /* configurable: false */});
34 
35 if (arr[3] != 1) {
36     throw new Error("arr[3] not defined");
37 }
38 
39 if (arr.length !== 4) {
40     throw new Error("Array length not updated to 4");
41 }
42 
43 Object.defineProperty(arr, "5", {value: 1, configurable: true});
44 
45 if (arr[5] != 1) {
46     throw new Error("arr[5] not defined");
47 }
48 
49 if (arr.length !== 6) {
50     throw new Error("Array length not updated to 4");
51 }
52 
53 arr.length = 0;
54 
55 if (5 in arr) {
56     throw new Error("configurable element was not deleted");
57 }
58 
59 if (arr[3] != 1) {
60     throw new Error("non-configurable element was deleted");
61 }
62 
63 if (arr.length !== 4) {
64     throw new Error("Array length not set");
65 }
