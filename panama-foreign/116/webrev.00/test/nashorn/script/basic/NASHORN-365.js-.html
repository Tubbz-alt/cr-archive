24 /**
25  * NASHORN-365 : JSON.parse should complain about junk at the end of the parsed string
26  *
27  * @test
28  * @run
29  */
30 
31 function check(str) {
32     try {
33         JSON.parse(str);
34         fail("should have got SyntaxError for " + str);
35     } catch (e) {
36         if (! (e instanceof SyntaxError)) {
37             fail("SyntaxError expected got " + e);
38         }
39     }
40 }
41 
42 check("34\t454");
43 check("34  55");
44 check("{ 'foo': 34 } 343");
45 check("[344, 55] 454");
