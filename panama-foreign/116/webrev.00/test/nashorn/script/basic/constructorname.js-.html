25  * Check obj.constructor property for user-defined constructor as well
26  * as built-in constructors.
27  *
28  * @test
29  * @run
30  */
31 
32 function MyConstructor() {}
33 var myObj = new MyConstructor();
34 print(myObj.constructor.name);
35 
36 myObj = {};
37 print(myObj.constructor.name);
38 
39 myObj = new Boolean(true);
40 print(myObj.constructor.name);
41 
42 myObj = new Number(3);
43 print(myObj.constructor.name);
44 
45 myObj = new String("hello");
46 print(myObj.constructor.name);
47 
48 myObj = new Array();
49 print(myObj.constructor.name);
50 
51 myObj = /javascript/;
52 print(myObj.constructor.name);
