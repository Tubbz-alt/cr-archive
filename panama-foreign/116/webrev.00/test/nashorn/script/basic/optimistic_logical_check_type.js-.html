24 /**
25  * @test
26  * @bug 8036987, 8037572
27  * @summary Implement tests that checks static types in the compiled code
28  * @run
29  */
30 
31 var inspect = Java.type("jdk.nashorn.test.tools.StaticTypeInspector").inspect
32 var a = 3, b = true, c = 0;
33 var x = { a: 2, b: undefined, c: true}
34 
35 // Testing logical operators
36 //-- Global variable
37 print(inspect("cat" && "dog", "object AND object"))
38 print(inspect(b && a, "true AND non-falsey global int"))
39 print(inspect(a && b, "non-falsey global int AND true"))
40 print(inspect(x && b, "non-falsey object AND true"))
41 print(inspect(b && x, "true AND non-falsey object"))
42 print(inspect("cat" || "dog", "object OR object"))
43 print(inspect(b || a, "true OR non-falsey global int"))
44 print(inspect(a || b, "non-falsey global int OR true"))
45 print(inspect(x || b, "non-falsey object OR true"))
46 print(inspect(b || x, "true OR non-falsey object"))
47 print(inspect(!x.c || b, "false OR true"))
48 print(inspect(c && b, "falsey global int AND true"))
49 print(inspect(c || x.b, "falsey global int OR undefined"))
50 print(inspect(!c || x.b, "logical not falsey global int OR undefined"))
51 print(inspect(!b || x.b, "false OR undefined"))
52 print(inspect(!b || c, "false OR falsey global int"))
53 print(inspect(!c || c, "logical not falsey global int OR falsey global int"))
54  //--Local variable
55 print(inspect(x.b && a, "local undefined AND non-falsey global int"))
56 print(inspect(b && x.b, "true AND local undefined"))
57 print(inspect(x.b && x.a, "local undefined AND non-falsey local int"))
58 print(inspect(x.b || b, "local undefined OR true"))
59 print(inspect(b || x.b, "true OR local undefined"))
60 print(inspect(x.a && x.c, "non-falsey local int AND true"))
61 print(inspect(x.c && x.a, "true AND non-falsey local int"))
62 print(inspect(x.c && !!x.a, "true AND double logical not non-falsey local int "))
63 print(inspect(!x.c && x.a, "false AND non-falsey local int"))
64 print(inspect(x.a || x.c, "non-falsey local int OR true"))
65 print(inspect(!x.c || x.c, "false OR true"))
