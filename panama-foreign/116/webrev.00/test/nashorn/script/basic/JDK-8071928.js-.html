24 /**
25  * JDK-8071928: Instance properties with getters returning wrong values
26  *
27  * @test
28  * @run
29  */
30 
31 
32 var types = {};
33 
34 function Type() {}
35 
36 Type.prototype.getName = function() {
37     return this._name;
38 };
39 
40 function defineType(init) {
41     return Object.create(Type.prototype, {
42         _name: { get: function() { return init.name; } }
43     });
44 }
45 
46 types.A = defineType({ name: 'A' });
47 types.B = defineType({ name: 'B' });
48 types.C = defineType({ name: 'C' });
49 types.D = defineType({ name: 'D' });
50 
51 var keys = Object.keys(types);
52 for (var i = 0; i < keys.length; i++) {
53     var t = types[keys[i]];
54     if (t.getName() != keys[i]) {
55         throw 'wrong name for ' + keys[i] + ': ' + t.getName();
56     }
57 }
