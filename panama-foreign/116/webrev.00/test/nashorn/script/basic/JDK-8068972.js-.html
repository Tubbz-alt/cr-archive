24 /**
25  * JDK-8068972: Array.splice should follow the ES6 specification
26  *
27  * @test
28  * @run
29  */
30 
31 
32 function assertEqualArrays(a, b) {
33     Assert.assertTrue(Array.isArray(a));
34     Assert.assertTrue(Array.isArray(b));
35     Assert.assertTrue(a.length === b.length);
36     Assert.assertTrue(a.every(function(v, j) {
37         return v === b[j];
38     }));
39 }
40 
41 var array = [1, 2, 3, 4, 5, 6, 7];
42 
43 var result = array.splice();
44 assertEqualArrays(array, [1, 2, 3, 4, 5, 6, 7]);
45 assertEqualArrays(result, []);
46 
47 result = array.splice(4);
48 assertEqualArrays(array, [1, 2, 3, 4]);
49 assertEqualArrays(result, [5, 6, 7]);
50 
51 result = array.splice(1, 2, -2, -3);
52 assertEqualArrays(array, [1, -2, -3, 4]);
53 assertEqualArrays(result, [2, 3]);
