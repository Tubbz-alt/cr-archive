24 /**
25  * NASHORN-428 : Redefining a non-configurable property with the "same value" is permitted, but -0.0 and +0.0 are not "same value"
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = {};
32 
33 Object.defineProperty(obj, "0", {
34     value: -0,
35     configurable: false
36 });
37 
38 try {
39     Object.defineProperties(obj, {
40         "0": { value: +0 }
41     });
42     fail("#1 expected TypeError");
43 } catch (e) {
44     if (! (e instanceof TypeError)) {
45         fail("#2 expected TypeError, got " + e);
46     }
47 }
48 
49 // -0 should be fine.. should not throw any TypeError
50 Object.defineProperties(obj, {
51     "0": { value: -0 }
52 });
