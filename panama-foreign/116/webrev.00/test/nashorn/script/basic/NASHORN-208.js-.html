31 var inspect = function(x) { return x; }
32 
33 var formatRegExp = /%[sdj%]/g;
34 
35 var format = function(f) {
36   if (typeof f !== 'string') {
37     var objects = [];
38     for (var i = 0; i < arguments.length; i++) {
39       objects.push(inspect(arguments[i]));
40     }
41     return objects.join(' ');
42   }
43 
44   var i = 1;
45   var args = arguments;
46   var len = args.length;
47   var str = String(f).replace(formatRegExp, function(x) {
48     if (i >= len) return x;
49     switch (x) {
50       case '%s': return String(args[i++]);
51       case '%d': return Number(args[i++]);
52       case '%j': return JSON.stringify(args[i++]);
53       case '%%': return '%';
54       default:
55         return x;
56     }
57   });
58   for (var x = args[i]; i < len; x = args[++i]) {
59     if (x === null || typeof x !== 'object') {
60       str += ' ' + x;
61     } else {
62       str += ' ' + inspect(x);
63     }
64   }
65   return str;
66 };
67 
68 var s = format('%s %s', 'foo', 'bar', 'hop');
69 print(s);
