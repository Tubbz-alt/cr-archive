24 /**
25  * NASHORN-664 : Can not set nashorn extension properties such as fileName, lineNumber and columnNumber of Error objects
26  *
27  * @test
28  * @run
29  */
30 
31 try {
32     throw new Error();
33 } catch (e) {
34     e.fileName = "foo";
35     if (e.fileName !== 'foo') {
36         fail("can not set e.fileName");
37     }
38     e.lineNumber = 100;
39     if (e.lineNumber !== 100) {
40         fail("can not set e.lineNumber");
41     }
42     e.columnNumber = 33;
43     if (e.columnNumber !== 33) {
44         fail("can not set e.columnNumber");
45     }
46     e.stack = "mystack";
47     if (e.stack !== 'mystack') {
48         fail("can not set e.stack");
49     }
50 }
