24 /**
25  * JDK-8011357: Array.prototype.slice and Array.prototype.splice should not call user defined valueOf of start, end arguments more than once
26  *
27  * @test
28  * @run
29  */
30 
31 var startValueOf = 0;
32 var endValueOf = 0;
33 
34 [].slice(
35     {
36         valueOf: function() {
37            startValueOf++;
38         }
39     },
40     {
41         valueOf: function() {
42             endValueOf++;
43         }
44     }
45 );
46 
47 if (startValueOf !== 1) {
48     fail("Array.prototype.slice should call valueOf on start arg once");
49 }
50 
51 if (endValueOf !== 1) {
52     fail("Array.prototype.slice should call valueOf on end arg once");
53 }
54 
55 startValueOf = 0;
56 
57 [].splice(
58     {
59         valueOf: function() {
60            startValueOf++;
61         }
62     }
63 );
64 
65 if (startValueOf !== 1) {
66     fail("Array.prototype.splice should call valueOf on start arg once");
67 }
