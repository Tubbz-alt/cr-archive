24 /**
25  * JDK-8031359: Invocable.getInterface() works incorrectly if interface has default methods
26  *
27  * @test
28  * @run
29  */
30 
31 var func = new java.util.function.Function() {
32     apply: function(arg) {
33         print("func called with " + arg);
34         return arg.toUpperCase();
35     }
36 };
37 
38 // Function.andThen is a default method
39 func.andThen(func)("hello");
40 
41 // Function.compose is another default method
42 func.compose(new java.util.function.Function() {
43     apply: function(arg) {
44         print("compose called with " + arg);
45         return arg.charAt(0);
46     }
47 })("hello");
48 
49 var func2 = new java.util.function.Function() {
50     apply: function(arg) {
51         print("I am func2: " + arg);
52         return arg;
53     },
54 
55     andThen: function(func) {
56         print("This is my andThen!");
57         return func;
58     }
59 };
60 
61 func2.apply("hello");
62 func2.andThen(func);
