24 /**
25  * NASHORN-406 : Property descriptor properties should be enumerable
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = {
32     foo : 10,
33     get bar() { return 32; },
34     set bar(x) {},
35 };
36 
37 function checkData() {
38    var desc = Object.getOwnPropertyDescriptor(obj, "foo");
39    var enumSeen = false, writeSeen = false,
40        configSeen = false, valueSeen = false;
41    for (i in desc) {
42        switch(i) {
43            case 'enumerable':
44                enumSeen = true; break;
45            case 'writable':
46                writeSeen = true; break;
47            case 'configurable':
48                configSeen = true; break;
49            case 'value':
50                valueSeen = true; break;
51        }
52    }
53 
54    return enumSeen && writeSeen && configSeen && valueSeen;
55 }
56 
57 if (!checkData()) {
58     fail("data descriptor check failed");
59 }
60 
61 function checkAccessor() {
62    var desc = Object.getOwnPropertyDescriptor(obj, "bar");
63    var enumSeen = false, getterSeen = false,
64        configSeen = false, setterSeen = false;
65    for (i in desc) {
66        switch(i) {
67            case 'enumerable':
68                enumSeen = true; break;
69            case 'configurable':
70                configSeen = true; break;
71            case 'get':
72                getterSeen = true; break;
73            case 'set':
74                setterSeen = true; break;
75         }
76    }
77 
78    return enumSeen && configSeen && getterSeen && setterSeen;
79 }
80 
81 if (!checkAccessor()) {
82     fail("accessor descriptor check failed");
83 }
