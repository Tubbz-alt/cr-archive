32 var semver = "\\s*[v=]*\\s*([0-9]+)" // major
33         + "\\.([0-9]+)" // minor
34         + "\\.([0-9]+)" // patch
35         + "(-[0-9]+-?)?" // build
36         + "([a-zA-Z-+][a-zA-Z0-9-\.:]*)?" // tag
37     , exprComparator = "^((<|>)?=?)\s*("+semver+")$|^$";
38 var validComparator = new RegExp("^"+exprComparator+"$");
