24 /**
25  * JDK-8073846: Javascript for-in loop returned extra keys
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = {};
32 
33 var list = [
34     '2100000',
35     '420000',
36     '430000'
37 ];
38 
39 for (var i = 0; i < list.length; i++) {
40     if (obj[list[i]]) print("duplicate: " + list[i]);
41     obj[list[i]] = 'obj' + list[i]
42 }
43 
44 var count = 0;
45 for (var a in obj) {
46     count++;
47     if ('obj' + a !== obj[a]) {
48         throw 'wrong key or value: ' + a + ', ' + obj[a];
49     }
50 }
51 
52 if (count !== 3) {
53     throw 'wrong entry count: ' + count;
54 }
