24 /**
25  * NASHORN-114 :  String.fromCharCode should use ToUInt16 as per the spec.
26  *
27  * @test
28  * @run
29  */
30 
31 if (String.fromCharCode(4294967294).charCodeAt(0) !== 65534) {
32     fail("String.fromCharCode(4294967294).charCodeAt(0) !== 65534");
33 }
34 
35 if (String.fromCharCode(Number.POSITIVE_INFINITY).charCodeAt(0) !== 0) {
36     fail("String.fromCharCode(Number.POSITIVE_INFINITY).charCodeAt(0) !== 0");
37 }
38 
39 if (String.fromCharCode(0x0000).charCodeAt(0) !== 0) {
40     fail("String.fromCharCode(0x0000).charCodeAt(0) !== 0");
41 }
42 
43 if (String.fromCharCode(65, 66, 67, 68) !== "ABCD") {
44     fail("String.fromCharCode(65, 66, 67, 68) !== 'ABCD'");
