24 /**
25  * NASHORN-494 :  in strict mode \0 is not allowed as escape sequence
26  *
27  * @test
28  * @run
29  */
30 
31 'use strict';
32 
33 function expectNoError(str) {
34     try {
35         eval(str);
36     } catch (e) {
37         fail("failed", e);
38     }
39 }
40 
41 function expectError(str) {
42     try {
43         eval(str);
44         fail("should have thrown SyntaxError for : " + str);
45     } catch (e) {
46         if (! (e instanceof SyntaxError)) {
47             fail("expected SyntaxError, got " + e);
48         }
49     }
50 }
51 
52 expectNoError("var x = '\\0'");
53 expectNoError("var x = '\\0xyz'");
54 expectError("var x = '\\01'");
55 expectError("var x = '\\08'");
56 expectError("var x = '\\7'");
57 expectError("var x = '\\563'");
