31 // Note: we're using Function("code") instead of (function(){ code }) so that
32 // we don't trigger parser API validation in JDK-8008448 tests. The test code
33 // encapsulated in functions below can't be correctly handled by the parser API
34 // currently, as it contains parser-generated REFERENCE_ERROR runtime nodes.
35 try {
36     Function("L: {this = x;break L}")();
37 } catch (e) {
38    print("threw ReferenceError: " + (e instanceof ReferenceError));
39 }
40 try {
41     Function("L:with(this--)break L;")();
42 } catch (e) {
43    print("threw ReferenceError: " + (e instanceof ReferenceError));
44 }
45 Function("L:with(Object in Object)break L;")();
46 print("SUCCESS");
