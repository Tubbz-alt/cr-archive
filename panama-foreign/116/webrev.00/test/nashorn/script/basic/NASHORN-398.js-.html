24 /**
25  * NASHORN-398 : Should not be able to set non-existing indexed property if object is not extensible
26  *
27  * @test
28  * @run
29  */
30 
31 function funcNonStrict(array) {
32     var obj = array? [] : {};
33     Object.preventExtensions(obj);
34     obj[0] = 12;
35     if (obj[0] === 12) {
36         fail("#1 obj[0] has been set");
37     }
38 
39     if (obj.hasOwnProperty("0")) {
40         fail("#2 has property '0'");
41     }
42 }
43 
44 funcNonStrict(true);
45 funcNonStrict(false);
46 
47 function funcStrict(array) {
48     'use strict';
49 
50     var obj = array? [] : {};
51     Object.preventExtensions(obj);
52     try {
53         obj[0] = 12;
54         fail("#3 should have thrown TypeError");
55     } catch (e) {
56         if (! (e instanceof TypeError)) {
57             fail("#4 TypeError expected, got " + e);
58         }
59     }
60 
61     if (obj[0] === 12) {
62         fail("#5 obj[0] has been set");
63     }
64 
65     if (obj.hasOwnProperty("0")) {
66         fail("has property '0'");
67     }
68 }
69 
70 funcStrict(true);
71 funcStrict(false);
