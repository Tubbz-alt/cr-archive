24 /**
25  * JDK-8129410: Java adapters with class-level overrides should preserve variable arity constructors
26  *
27  * @test
28  * @run
29  */
30 
31 var VarArgConstructor = Java.type("jdk.nashorn.test.models.VarArgConstructor");
32 var VarArgConstructorExtended = Java.extend(VarArgConstructor, {});
33 
34 // If the fix didn't work we wouldn't even get past the constructor invocation
35 // as it'd complain there's no matching arity constructor.
36 var newExtended = new VarArgConstructorExtended(1, true, "a", "b");
37 
38 // Assert the expected constructor was invoked.
39 Assert.assertEquals("vararg", newExtended.indicator);
