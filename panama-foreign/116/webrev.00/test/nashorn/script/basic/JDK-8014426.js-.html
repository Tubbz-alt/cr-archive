24 /**
25  * Ensure catchall exceptions from finally inlining are rethrown as is
26  *
27  * @test
28  * @run
29  */
30 
31 function runScriptEngine() {
32     var fac    = new Packages.jdk.nashorn.api.scripting.NashornScriptEngineFactory();
33     var engine = fac.getScriptEngine();
34     engine.eval(
35 "try {\n\
36   doIt();\n\
37 } finally { \n\
38   var x = 17;\n\
39 }\n\
40 function doIt() {\n\
41   throw new TypeError('en stor graa noshoerning!');\n\
42 }\n");
43 }
44 
45 try {
46     runScriptEngine();
47 } catch(e) {
48     print(e);
49 }
