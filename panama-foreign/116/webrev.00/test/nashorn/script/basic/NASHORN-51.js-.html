24 /**
25  * NASHORN-51 :  Post/pre increment and decrement operators should only accept "left hand side" expression.
26  *
27  * @test
28  * @run
29  */
30 
31 var literals = [1, 0, 3.14, true, false, null];
32 
33 for (i in literals) {
34     try {
35         eval(literals[i] + "++");
36         print("ERROR!! post increment : " + literals[i]);
37     } catch (e) {
38         printError(e);
39     }
40 
41     try {
42         eval(literals[i] + "--");
43         print("ERROR!! post decrement : " + literals[i]);
44     } catch (e) {
45         printError(e);
46     }
47 
48     try {
49         eval("++" + literals[i]);
50         print("ERROR!! pre increment : " + literals[i]);
51     } catch (e) {
52         printError(e);
53     }
54 
55     try {
56         eval("--" + literals[i]);
57         print("ERROR!! pre decrement : " + literals[i]);
58     } catch (e) {
59         printError(e);
60     }
61 }
