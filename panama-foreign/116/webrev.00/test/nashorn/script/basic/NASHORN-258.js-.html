24 /**
25  * NASHORN-258 : Broken slot assignments to non constant members of multidimensional arrays in OP=
26  *
27  * @test
28  * @run
29  */

31 function test3(a) {
32     for (i = 0; i < a.length ; i++) {
33     for (j = 0; j < a[i].length ; j++) {
34         for (k = 0; k < a[i][j].length ; k++) {
35         a[i][j][k] *= 8;
36         }
37     }
38     }
39 }

41 function test3local(a) {
42     for (var i = 0; i < a.length ; i++) {
43     for (var j = 0; j < a[i].length ; j++) {
44         for (var k = 0; k < a[i][j].length ; k++) {
45         a[i][j][k] *= 8;
46         }
47     }
48     }
49 }

53           [[1,1,1],[1,1,1],[1,1,1]] ];
