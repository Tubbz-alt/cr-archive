24 /**
25  * NASHORN-192 :  User defined property setter or getter with extra arguments or lesser argument fails by throwing exception
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = {};
32 Object.defineProperty(obj, "foo", {
33     get: function(obj) {
34         return 'hello';
35     }
36 });
37 
38 try {
39     if (obj.foo !== 'hello') {
40         fail("getter with extra argument does not work");
41     }
42 } catch (e) {
43     fail("failed to get with " + e);
44 }
45 
46 Object.defineProperty(obj, "bar", {
47     set: function() {
48         // do nothing
49     }
50 });
51 
52 try {
53     obj.bar = 33;
54 } catch (e) {
55     fail("failed to set with " + e);
56 }
57 
58 Object.defineProperty(obj, "prop", {
59     set: function(obj1, obj2, obj3) {
60         this.val = obj1;
61     }
62 });
63 
64 try {
65     obj.prop = 33;
66     if (obj.val !== 33) {
67         fail("set with extra argument does not work");
68     }
69 } catch (e) {
70     fail("failed to set with " + e);
71 }
