25  * JDK-8024180: Incorrect handling of expression and parent scope in 'with' statements
26  *
27  * @test
28  * @run
29  */
30 
31 var p =  { };
32 var obj = Object.create(p);
33 
34 var x = "global x";
35 
36 // adding property to __proto__ of 'with' expression should
37 // reflect as a variable immediately.
38 function func() {
39     with(obj) {
40         for (i = 0; i < 2; i++) {
41             print(x);
42             if (i == 0) {
43                 p.x = "p.x";
44             }
45         }
46     }
47 }
