31 var colors = new java.util.ArrayList()
32 colors.add("red")
33 colors.add("purple")
34 colors.add("pink")
35 
36 for(var index in colors) {
37     print("colors[" + index + "]=" + colors[index])
38 }
39 
40 for each(var color in colors) {
41     print(color)
42 }

44 var capitals = new java.util.LinkedHashMap()
45 capitals.Sweden = "Stockholm"
46 capitals.Hungary = "Budapest"
47 capitals.Croatia = "Zagreb"
48 
49 for(var key in capitals) {
50     print("capital of " + key + " is " + capitals[key])
51 }
52 
53 for each(var capital in capitals) {
54     print(capital)
