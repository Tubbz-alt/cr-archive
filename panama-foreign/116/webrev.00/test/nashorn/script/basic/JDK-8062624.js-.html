24 /**
25  * JDK-8062624: java.lang.String methods not available on concatenated strings
26  *
27  * @test
28  * @run
29  */
30 
31 function testStringMethods(s) {
32     print(s.startsWith("f"));
33     print(s.endsWith("r"));
34     print(Java.from(s.getBytes()));
35     print(Java.from(s.bytes));
36 }
37 
38 var s = "f";
39 testStringMethods(s);
40 s = s + "oo";
41 testStringMethods(s);
42 testStringMethods("abc");
43 s += "bar";
44 s = "baz" + s;
45 testStringMethods(s);
