24 /*
25  * Make sure that basic object method call re-linking works properly.
26  *
27  * @test
28  * @run
29  */
30 
31 var obj1 = {
32    func: function() { print("obj1.func called"); }
33 };
34 
35 var obj2 = {
36    func: function() { print("obj2.func called"); }
37 };
38 
39 var obj3 = {
40    __noSuchMethod__: function(name) {
41        print("no such method: " + name);
42    }
43 };
44 
45 
46 // have 'func' on prototype
47 var obj4 = Object.create({
48     func: function() {
49         print("obj4's prototype func called");
50     }
51 });
52 
53 function MyConstructor() {
54 }
55 
56 MyConstructor.prototype.func = function() {
57     print("MyConstructor.prototype.func");
58 }
59 
60 var obj5 = new MyConstructor();
61 var obj6 = new MyConstructor();
62 
63 var arr = [ obj1, obj2, obj3, obj4, obj5, obj6,
64             obj1, obj2, obj3, obj4, obj5, obj6 ];
65 
66 var myObj;
67 for (i in arr) {
68     if (i == 8) {
69         obj3.func = function() {
70             print("new obj3.func called");
71         }
72         obj4.func = function() {
73             print("new obj4.func called");
74         }
75         MyConstructor.prototype.func = function() {
76             print("all new MyConstructor.prototype.func");
77         }
78     }
79     myObj = arr[i];
80     myObj.func();
81 }
