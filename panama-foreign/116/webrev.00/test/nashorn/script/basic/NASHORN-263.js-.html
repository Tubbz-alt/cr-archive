24 /**
25  * NASHORN-263 : delete of function param or caller scope var should throw SyntaxError in strict mode
26  *
27  * @test
28  * @run
29  */
30 
31 function func() {
32     "use strict";
33     var obj = new Boolean(false);
34 
35     try {
36         eval("delete obj;");
37         fail("#1 should have thrown SyntaxError");
38     } catch (e) {
39         if (! (e instanceof SyntaxError)) {
40             fail("#2 SyntaxError expected got " + e);
41         }
42     }
43 }
44 
45 func();
46 
47 function func2(y) {
48     'use strict';
49     try {
50         eval("delete y");
51         fail("#3 should have thrown SyntaxError");
52     } catch (e) {
53         if (! (e instanceof SyntaxError)) {
54             fail("#4 SyntaxError expected got " + e);
55         }
56     }
57 }
58 
59 func2(2);
