24 /*
25  * NASHORN-609: ASSIGN_SHR result is converted to INT (should be UINT32).
26  *
27  * @test
28  * @run
29  */
30 function test() {
31     var shr = -1;
32     print(shr >>>= 0, shr);
33     var shr2 = 0xffffffff;
34     print(shr2 >>>= 0, shr2);
35     var shr3 = "-1";
36     print(shr3 >>>= 0, shr3);
37     var shr4 = -1.5;
38     print(shr4 >>>= 0, shr4);
39     var shr5 = 0xffffffffff;
40     print(shr5 >>>= 0, shr5);
41 }
42 test();
