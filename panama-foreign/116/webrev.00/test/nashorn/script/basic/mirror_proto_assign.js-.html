24 /**
25  * JDK-8023368: Instance __proto__ property should exist and be writable.
26  *
27  * @test
28  * @run
29  */
30 
31 // check that Object.setPrototypeOf works for mirror objects as well.
32 
33 var global = loadWithNewGlobal({
34     name: "test",
35     script: "var obj = {}; this"
36 });
37 
38 var proto = global.eval("({ foo: 323 })");
39 
40 Object.setPrototypeOf(global.obj, proto);
41 
42 function func(obj) {
43     // check proto inherited value
44     print("obj.foo = " + obj.foo);
45 }
46 
47 func(global.obj);
48 
49 var newProto = global.eval("({ foo: 'hello' })");
50 Object.setPrototypeOf(global.obj, newProto);
51 
52 func(global.obj);
