24 /**
25  * SAM interface implemented by a script function.
26  *
27  * @test
28  * @run
29  */
30 
31 // we can pass a script function when an interface with single
32 // abstract method is required. Here we pass script function for
33 // a java.lang.Runnable.
34 
35 var t1 = new java.lang.Thread(function() {
36     print("thread t1.run");
37 });
38 
39 // make sure that we can pass interface implementation as well.
40 var t2 = new java.lang.Thread(new java.lang.Runnable() {
41    run: function() { print("thread t2.run"); }
42 });
43 
44 t1.start();
45 t1.join();
46 
47 t2.start();
48 t2.join();
