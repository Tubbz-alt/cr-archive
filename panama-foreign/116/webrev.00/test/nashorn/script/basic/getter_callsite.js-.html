24 /**
25  * Check that getter property is properly invoked by a callsite
26  *
27  * @test
28  * @run
29  */
30 
31 function newObj(a) {
32   return { get foo() { return a }};
33 }
34 
35 function printFoo(obj) {
36     print("obj.foo is " + obj.foo);
37 }
38 
39 var obj1 = newObj(42);
40 var obj2 = newObj('hello');
41 
42 printFoo(obj1);
43 printFoo(obj2);
44 
45 var obj3 = Object.create(newObj('world'));
46 printFoo(obj3);
47 
48 Object.defineProperty(Object.getPrototypeOf(obj3),
49    "foo", { get: function() { return 3.1415; } });
50 
51 printFoo(obj3);
