24 /**
25  * NASHORN-488 : Function.prototype does not have 'prototype' property
26  *
27  * @test
28  * @run
29  */
30 
31 var desc = Object.getOwnPropertyDescriptor(Function.prototype, "prototype");
32 if (desc !== undefined) {
33     fail("Function.prototype has 'prototype'");
34 }
35 
36 Object.defineProperty(Function.prototype, "prototype", {
37     set: undefined, get: function() { return 32; },
38     configurable: true
39 });
40 
41 if (Function.prototype.prototype != 32) {
42     fail("Function.prototype.prototype != 32");
43 }
44 
45 if (! (delete Function.prototype.prototype)) {
46     fail("can't delete Function.prototype.prototype");
47 }
