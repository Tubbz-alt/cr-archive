24 /**
25  * Check that the expression ("property" in obj) works for own property as
26  * well as properties 'inherited' from proto chain.
27  *
28  * @test
29  * @run
30  */
31 
32 function MyCons() {
33     this.foo = "hello";
34 }
35 
36 MyCons.prototype.func = function() { print("func " + this); }
37 MyCons.prototype.num = 1729;
38 
39 var myObj = new MyCons();
40 print("foo? " + ("foo" in myObj));
41 print("func? " + ("func" in myObj));
42 print("num? " + ("num" in myObj));
43 // From Object.prototype
44 print("toString?" + ("toString" in myObj));
45 
46 MyCons.prototype.sun = "a star";
47 MyCons.prototype.earth = "a planet";
48 
49 print("foo? " + ("foo" in myObj));
50 print("func? " + ("func" in myObj));
51 print("num? " + ("num" in myObj));
52 print("sun? " + ("sun" in myObj));
53 print("earth? " + ("earth" in myObj));
54 print("toString? " + ("toString" in myObj));
