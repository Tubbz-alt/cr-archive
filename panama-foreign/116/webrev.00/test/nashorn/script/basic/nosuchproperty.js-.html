24 /**
25  * Check __noSuchProperty__
26  *
27  * @test
28  * @run
29  */
30 
31 function func(o) {
32    return o.foo;
33 }
34 
35 var obj = {
36     __noSuchProperty__: function(name) {
37         print("obj.__noSuchProperty__ for " + name);
38     }
39 };
40 
41 func(obj);
42 
43 obj.__noSuchProperty__ = function(name) {
44     print("new obj.__noSuchProperty__ for " + name);
45 }
46 
47 func(obj);
48 
49 // try inherited __noSuchProperty__ now.
50 var proto = {
51     __noSuchProperty__ : function(name) {
52         print("proto.__noSuchProperty__ for " + name);
53     }
54 };
55 
56 var obj2 = Object.create(proto);
57 func(obj2);
58 
59 proto.__noSuchProperty__ = function(name) {
60    print("new proto.__noSuchProperty__ for " + name);
61 }
62 
63 func(obj2);
