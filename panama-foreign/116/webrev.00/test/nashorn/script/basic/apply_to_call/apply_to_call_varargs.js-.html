24 /**
25  * apply_to_call_varars - make sure that apply to call transform works
26  * even when supplying too few arguments
27  *
28  * @test
29  * @run
30  */
31 
32 var Class = {
33   create: function() {
34     return function() { //vararg
35         this.initialize.apply(this, arguments);
36     }
37   }
38 };
39 
40 Color = Class.create();
41 Color.prototype = {
42     red: 0, green: 0, blue: 0,
43     initialize: function(r) {
44     this.red = r;
45     this.green = 255;
46     this.blue = 255;
47     },
48     toString: function() {
49     print("[red=" + this.red + ", green=" + this.green + ", blue=" + this.blue + "]");
50     }
51 };
52 
53 var colors = new Array(16);
54 function run() {
55     for (var i = 0; i < colors.length; i++) {
56     colors[i&0xf] = (new Color(i));
57     }
58 }
59 
60 run();
61 for (var i = 0; i < colors.length; i++) {
62     print(colors[i]);
63 }
64 
65 print("Swapping out call");
66 Function.prototype.call = function() {
67     throw "This should not happen, apply should be called instead";
68 };
69 
70 run();
71 for (var i = 0; i < colors.length; i++) {
72     print(colors[i]);
73 }
74 
75 print("All done!");
