24 /**
25  * apply_to_call3.js - do one apply to call specialization, then override, apply and make sure it reverts (i.e. stops
26  * calling call). Then apply should break and throw exception
27  *
28  * @test
29  * @run
30  */
31 
32 print("start");
33 
34 var x = {
35     a : 0,
36     b : 0,
37     c : 0,
38     initialize : function(x,y,z) {
39     this.a = x;
40     this.b = y;
41     this.c = z;
42     }
43 };
44 
45 function test() {
46     x.initialize.apply(x, arguments);
47 }
48 
49 test(4711,23,17);
50 print(x.a);
51 print(x.b);
52 print(x.c);
53 
54 print("Overwriting apply now");
55 
56 Function.prototype.apply = function() {
57     throw "This should be thrown, as the test call transform is no longer valid";
58 };
59 
60 try {
61     test(1,2,3);
62 } catch (e) {
63     print(e);
64 }
65 print(x.a);
66 print(x.b);
67 print(x.c);
