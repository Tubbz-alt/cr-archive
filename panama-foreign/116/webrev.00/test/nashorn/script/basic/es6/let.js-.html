32 "use strict";
33 
34 let a = 2;
35 let c = 2;
36 print(a, c);
37 
38 function f(x) {
39     let a = 5;
40     const c = 10;
41     print(a, c);
42     if (x) {
43         let a = 42;
44         const c = 43;
45         print(a, c);
46     }
47     print(a, c);
48 
49     function inner() {
50         (function() {
51             print(a, c);
52         })();
53     }
54     inner();
55 }
56 
57 f(true);
58 f(false);
59 
60 (function() {
61     (function() {
62         print(a, c);
63     })();
64 })();

67     print(a, c);
