24 /**
25  * JDK-8168373: don't emit conversions for symbols outside their lexical scope
26  *
27  * @test
28  * @run
29  * @option --language=es6
30  */
31 
32 function p() { return false } // "predicate"
33 function r(x) { return x } // "read"
34 
35 (function() {
36   try { // Try creates control flow edges from assignments into catch blocks.
37     // Lexically scoped, never read int variable (undefined at catch block) but still with a cf edge into catch block.
38     // Since it's never read, it's not written either (Nashorn optimizes some dead writes).
39     let x = 0;
40     if (p()) { throw {}; } // We need `p()` so this block doesn't get optimized away, for possibility of a `throw`
41     x = 0.0; // change the type of x to double
42     r(x); // read x otherwise it's optimized away
43   } catch (e) {} // under the bug, "throw" will try to widen unwritten int x to double for here and cause a verifier error
44 })()
