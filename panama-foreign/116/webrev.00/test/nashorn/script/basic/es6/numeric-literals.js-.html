24 /**
25  * JDK-8134873: ECMAScript 6 Numeric Literals
26  *
27  * @test
28  * @option --language=es6
29  */
30 
31 function assertEquals(expected, actual) {
32   if (expected !== actual) {
33     throw new Error("expected: " + expected + ", actual: " + actual);
34   }
35 }
36 
37 assertEquals(0b0, 0);
38 assertEquals(0B0, 0);
39 assertEquals(0b01, 1);
40 assertEquals(0B10, 2);
41 assertEquals(0b11111111, 255);
42 assertEquals(0b11111111111111111111111111111111, 4294967295);
43 
44 assertEquals(0o0, 0);
45 assertEquals(0O0, 0);
46 assertEquals(0o01, 1);
47 assertEquals(0O10, 8);
48 assertEquals(0o777, 511);
