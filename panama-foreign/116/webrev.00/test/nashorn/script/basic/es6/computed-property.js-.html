24 /**
25  * JDK-8164467: ES6 computed properties are implemented wrongly
26  *
27  * @test
28  * @run
29  * @option --language=es6
30  */
31 
32 function f(x) {
33     return x;
34 }
35 
36 var d = 'd';
37 var n = 3;
38 var s1 = Symbol();
39 var s2 = Symbol();
40 
41 var object = {
42     a: 'a',
43     ['b']: 'b',
44     [f('c')]: 'c',
45     [d]: d,
46     [1]: 1,
47     [f(2)]: 2,
48     [n]: 3,
49     [s1]: s1,
50     [f(s2)]: s2
51 };
52 
53 
54 Assert.assertEquals(object.a, 'a');
55 Assert.assertEquals(object.b, 'b');
56 Assert.assertEquals(object.c, 'c');
57 Assert.assertEquals(object.d, 'd');
58 Assert.assertEquals(object[1], 1);
59 Assert.assertEquals(object[2], 2);
60 Assert.assertEquals(object[3], 3);
61 Assert.assertEquals(object[s1], s1);
62 Assert.assertEquals(object[s2], s2);
63 
64 for (var s of ['a', 'b', 'c', 'd', 1, 2, 3, s1, s2]) {
65     Assert.assertEquals(object[s], s);
66 }
