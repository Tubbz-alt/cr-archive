24 /**
25  * JDK-8164467: ES6 computed properties are implemented wrongly
26  *
27  * @test
28  * @run
29  * @option --language=es6
30  */
31 
32 function f(x) {
33     return x;
34 }
35 
36 var object = {
37     [1.5]: 'a',
38     get [f(1e+40)]() { return 'b' },
39     [0.0001]: 'c',
40     [-0]: 'd',
41     [Infinity]: 'e',
42     [-Infinity]: 'f',
43     [NaN]: 'g'
44 
45 };
46 
47 
48 Assert.assertEquals(object['1.5'], 'a');
49 Assert.assertEquals(object['1e+40'], 'b');
50 Assert.assertEquals(object['0.0001'], 'c');
51 Assert.assertEquals(object[0], 'd');
52 Assert.assertEquals(object[Infinity], 'e');
53 Assert.assertEquals(object[-Infinity], 'f');
54 Assert.assertEquals(object[NaN], 'g');
