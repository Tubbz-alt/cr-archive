24 /**
25  * NASHORN-174 :  String.prototype.split does not work properly when separator is a string
26  *
27  * @test
28  * @run
29  */
30 
31 var nums = "24.45.8.78".split('.');
32 
33 if (nums.length != 4) {
34     fail("#1: split result expected to be of length 4");
35 }
36 
37 function check(index, value) {
38     if (nums[index] != value) {
39         fail("expected value @ " + index + " is " + value);
40     }
41 }
42 
43 check(0, '24');
44 check(1, '45');
45 check(2, '8');
46 check(3, '78');
