24 /**
25  * JDK-8155025: 0.001.toFixed(2) should return "0.00" not "0"
26  *
27  * @test
28  * @run
29  */
30 
31 for (var i = 0, zeros=""; i <= 9; i++, zeros += "0") {
32     var n = Number("0." + zeros + "1");
33     for (var j = 1; j <= i + 3; j++) {
34         print(n + ".toFixed(" + j + ")=" + n.toFixed(j));
35     }
36 }
