24 /**
25  * JDK-8015345: Function("}),print('test'),({") should throw SyntaxError
26  *
27  * @test
28  * @run
29  */
30 
31 function checkFunction(code) {
32     try {
33         Function(code);
34         fail("should have thrown SyntaxError for :" + code);
35     } catch (e) {
36         if (! (e instanceof SyntaxError)) {
37             fail("SyntaxError expected, but got " + e);
38         }
39         print(e);
40     }
41 }
42 
43 // invalid body
44 checkFunction("}),print('test'),({");
45 
46 // invalid param list
47 checkFunction("x**y", "print('x')");
48 
49 // invalid param identifier
50 checkFunction("in", "print('hello')");
51 //checkFunction("<>", "print('hello')")
52 
53 // invalid param list and body
54 checkFunction("x--y", ")");
55 
56 // check few valid cases as well
57 var f = Function("x", "return x*x");
58 print(f(10))
59 
60 f = Function("x", "y", "return x+y");
61 print(f(33, 22));
62 
63 f = Function("x,y", "return x/y");
64 print(f(24, 2));
