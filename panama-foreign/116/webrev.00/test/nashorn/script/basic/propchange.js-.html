24 /**
25  * Change attributes of an existing property
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = { x: 343 };
32 
33 for(i in obj) {
34   print(i);
35 }
36 
37 Object.defineProperty(obj, "x", {
38    enumerable: false });
39 
40 // not-enumerable anymore
41 for(i in obj) {
42   print(i);
43 }
44 
45 print(obj.x);
46 obj.x = 42;
47 print(obj.x);
48 
49 // make non-writable
50 Object.defineProperty(obj, "x", {
51     writable: false
52 });
53 
54 obj.x = "hello";
55 print(obj.x);
56 
57 print("done");
