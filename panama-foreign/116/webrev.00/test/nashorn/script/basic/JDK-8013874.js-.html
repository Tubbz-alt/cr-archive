24 /**
25  * JDK-8013874: Function argument's prototype seem cached and wrongly reused
26  *
27  * @test
28  * @run
29  */
30 
31 function deepEqual(actual, expected) {
32     print("deepEqual: " + (actual.prop === expected.prop) + ", prop: " + expected.prop);
33 }
34 
35 var proto = {};
36 var other = {
37     prop: 0
38 };
39 
40 function NameBuilder(first, last) {
41     this.first = first;
42     this.last = last;
43     return this;
44 }
45 
46 NameBuilder.prototype = proto;
47 
48 function NameBuilder2(first, last) {
49     this.first = first;
50     this.last = last;
51     return this;
52 }
53 
54 NameBuilder2.prototype = proto;
55 
56 var nb1 = new NameBuilder('Ryan', 'Dahl');
57 var nb2 = new NameBuilder2('Ryan', 'Dahl');
58 
59 print("In loader, nb1.prop === nb2.prop " + (nb1.prop === nb2.prop));
60 deepEqual(nb1, nb2);
61 
62 NameBuilder2.prototype = other;
63 nb2 = new NameBuilder2('Ryan', 'Dahl');
64 
65 print("In loader, nb1.prop === nb2.prop " + (nb1.prop === nb2.prop));
66 deepEqual(nb1, nb2);
