24 /**
25  * NASHORN-381 : JSAdapter override mechanism should not search properties of it's own proto
26  *
27  * @test
28  * @run
29  */
30 
31 var toStringAccessed = false;
32 
33 var obj = new JSAdapter({}) {
34     __get__: function(name) {
35         if (name == 'toString') {
36            toStringAccessed = true;
37         }
38     }
39 };
40 
41 if (obj.toString === Object.prototype.toString) {
42     fail("JSAdapter uses Object.prototype properties");
43 }
44 
45 if (! toStringAccessed) {
46     fail("JSAdapter does not use __get__ for toString access");
47 }
