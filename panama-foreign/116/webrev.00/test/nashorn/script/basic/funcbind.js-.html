24 /**
25  * Test the functionality of Function.prototype.bind.
26  *
27  * @test
28  * @run
29  */
30 
31 var global = this;
32 
33 function foo() {
34     print("global foo");
35 }
36 
37 function func(x, y) {
38     print(this == global);
39     print(x);
40     print(y);
41     this.foo();
42 }
43 
44 print("func.bind.length = " + func.bind.length);
45 
46 var f = func.bind(this, 434);
47 print(f.length);
48 f("hello");
49 
50 // bind with a different 'this' and different number of args
51 f = func.bind({ foo: function() { print("foo member"); } });
52 print(f.length);
53 f("world", 42);
