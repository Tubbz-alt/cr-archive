24 /**
25  * JDK-8015958: DataView constructor is not defined
26  *
27  * @test
28  * @run
29  */
30 
31 // basic DataView constructor checks.
32 
33 // check ArrayBufferView property values of DataView instance
34 function check(dv, buf, offset, length) {
35     if (dv.buffer !== buf) {
36         fail("DataView.buffer is wrong");
37     }
38 
39     if (dv.byteOffset != offset) {
40         fail("DataView.byteOffset = " + dv.byteOffset + ", expected " + offset);
41     }
42 
43     if (dv.byteLength != length) {
44         fail("DataView.byteLength = " + dv.byteLength + ", expected " + length);
45     }
46 }
47 
48 var buffer = new ArrayBuffer(12);
49 check(new DataView(buffer), buffer, 0, 12);
50 check(new DataView(buffer, 2), buffer, 2, 10);
51 check(new DataView(buffer, 4, 8), buffer, 4, 8);
52 
53 // make sure expected error is thrown
54 function checkError(callback, ErrorType) {
55     try {
56         callback();
57         fail("Should have thrown " + ErrorType.name);
58     } catch (e) {
59         if (! (e instanceof ErrorType)) {
60             print("Expected " + ErrorType.name + " got " + e);
61             e.printStackTrace()
62         }
63     }
64 }
65 
66 // non ArrayBuffer as first arg
67 checkError(function() { new DataView(344) }, TypeError);
68 
69 // illegal offset/length values
70 checkError(function() { new DataView(buffer, -1) }, RangeError);
71 checkError(function() { new DataView(buffer, 15) }, RangeError);
72 checkError(function() { new DataView(buffer, 1, 32) }, RangeError);
