24 /**
25  * NASHORN-241 : regression: "new 1" does not throw TypeError
26  *
27  * @test
28  * @run
29  */
30 
31 try {
32     new 1;
33     fail('new "1" should have thrown TypeError');
34 } catch (e) {
35     if (! (e instanceof TypeError)) {
36         fail('expected TypeError, got ' + e, e);
37     }
38 }
39 
40 try {
41     var x = "1";
42     new x;
43     fail('var x = "1"; new x should have thrown TypeError');
44 } catch (e) {
45     if (! (e instanceof TypeError)) {
46         fail('var x = "1"; new x - expected TypeError, got ' + e, e);
47     }
48 }
49 
50 try {
51     var x = "1";
52     new x();
53     fail('var x = "1"; new x() should have thrown TypeError');
54 } catch (e) {
55     if (! (e instanceof TypeError)) {
56         fail('var x = "1"; new x() - expected TypeError, got ' + e);
57     }
58 }
