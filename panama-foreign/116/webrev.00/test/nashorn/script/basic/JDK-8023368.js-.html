24 /**
25  * JDK-8023368: Instance __proto__ property should exist and be writable.
26  *
27  * @test
28  * @run
29  */
30 
31 // function to force same callsites
32 function check(obj) {
33     print(obj.func());
34     print(obj.x);
35     print(obj.toString());
36 }
37 
38 function Func() {
39 }
40 
41 Func.prototype.func = function() {
42     return "Func.prototype.func";
43 }
44 
45 Func.prototype.x = "hello";
46 
47 var obj = new Func();
48 var obj2 = Object.create(obj);
49 
50 // check direct and indirect __proto__ change
51 check(obj);
52 check(obj2);
53 obj.__proto__ = {
54    func: function() {
55         return "obj.__proto__.func @ " + __LINE__;
56    },
57    x: 344
58 };
59 
60 check(obj);
61 check(obj2);
62 
63 // check indirect (1 and 2 levels) __proto__ function change
64 obj.__proto__.__proto__ = {
65     toString: function() {
66         return "new object.toString";
67     }
68 };
69 
70 check(obj);
71 check(obj2);
