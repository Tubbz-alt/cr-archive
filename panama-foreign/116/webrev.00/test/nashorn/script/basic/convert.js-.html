24 /**
25  * Tests for convert method of ScriptUtils.
26  *
27  * @test
28  * @run
29  */
30 
31 var ScriptUtils = Java.type("jdk.nashorn.api.scripting.ScriptUtils");
32 obj = { valueOf: function() { print("hello"); return 43.3; } };
33 
34 // object to double
35 print(ScriptUtils.convert(obj, java.lang.Number.class));
36 
37 // array to List
38 var arr = [3, 44, 23, 33];
39 var list = ScriptUtils.convert(arr, java.util.List.class);
40 print(list instanceof java.util.List)
41 print(list);
42 
43 // object to Map
44 obj = { foo: 333, bar: 'hello'};
45 var map = ScriptUtils.wrap(obj);
46 print(map instanceof java.util.Map);
47 for (m in map) {
48    print(m + " " + map[m]);
49 }
50 
51 // object to String
52 obj = { toString: function() { print("in toString"); return "foo" } };
53 print(ScriptUtils.convert(obj, java.lang.String.class));
54 
55 // array to Java array
56 var jarr = ScriptUtils.convert(arr, Java.type("int[]"));
57 print(jarr instanceof Java.type("int[]"));
58 for (i in jarr) {
59     print(jarr[i]);
60 }
