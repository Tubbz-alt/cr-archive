31 function f(x) {
32   return (function inner() {
33       var y; (function dummy() { return y })() // force own scope for the inner function
34       with({}) { // 'with' block turns off fast scopes
35           return x
36       }
37   })();
38 }
39 print(f(1));
40 print(f(2));
