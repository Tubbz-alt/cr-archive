26 /**
27  * JDK-8195829 : Parsing a nameless ES6 class results in a thrown NullPointerException.
28  *
29  * @test
30  * @run
31  */
32 
33 var Parser = Java.type("jdk.nashorn.api.tree.Parser");
34 var SimpleTreeVisitor = Java.type("jdk.nashorn.api.tree.SimpleTreeVisitorES6");
35 var parser = Parser.create("--language=es6"); 
36 var ast = parser.parse("test.js", "(class {});", print);
37 var reachedClassExpr = false;
38 ast.accept(new (Java.extend(SimpleTreeVisitor)) {
39     visitClassExpression: function(node, extra) {
40         reachedClassExpr = true;
41         Assert.assertTrue(node.name === null);
42     }
43 }, null);
44 
45 Assert.assertTrue(reachedClassExpr);
