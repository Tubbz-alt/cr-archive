24 /**
25  * JDK-8015349: "abc".lastIndexOf("a",-1) should evaluate to 0 and not -1
26  *
27  * @test
28  * @run
29  */
30 
31 function printEval(code) {
32     print(code + " = " + eval(code));
33 }
34 
35 printEval("'abc'.lastIndexOf('a', 4)");
36 printEval("'abc'.lastIndexOf('b', Infinity)");
37 printEval("'abc'.lastIndexOf('a', -1)");
38 printEval("'abc'.lastIndexOf('a', -Infinity)");
39 printEval("'oracle'.lastIndexOf('u')");
40 printEval("'hello'.lastIndexOf('l')");
41 printEval("'hello'.lastIndexOf('l', 2)");
42 printEval("'hello'.lastIndexOf('l', 3)");
43 printEval("'hello'.lastIndexOf('l', 1)");
