24 /**
25  * NASHORN-376 : using 'this' keyword as operand for ++ or -- or as L-value of assignment operator should result in error
26  *
27  * @test
28  * @run
29  */
30 
31 function check(code) {
32     try {
33         eval(code);
34         fail("should have result in error " + code);
35     } catch (e) {
36         if (! (e instanceof ReferenceError)) {
37             fail("ReferenceError expected for '" + code + "' , got " + e);
38         }
39     }
40 }
41 
42 check("this = 4;");
43 check("this++");
44 check("++this");
45 check("this--");
46 check("--this");
