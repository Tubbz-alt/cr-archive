24 /**
25  * JDK-8012457: Function.prototype.apply should accept any array-like argument for function arguments
26  *
27  * @test
28  * @run
29  */
30 
31 // no exception for these
32 Function().apply(null, {length: null})
33 Function().apply(null, {length: 0.1})
34 
35 // getter should be called
36 var getter_0_called = false;
37 
38 Function().apply(null,
39     Object.defineProperty([],"0",
40         {  get: function(){ getter_0_called = true; return 0 }
41     })
42 );
43 
44 if (! getter_0_called) {
45     fail("getter for '0' of arguments array not called");
46 }
