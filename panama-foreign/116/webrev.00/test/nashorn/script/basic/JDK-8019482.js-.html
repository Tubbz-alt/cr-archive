24 /**
25  * JDK-8019482: Number("0x0.0p0") should evaluate to NaN
26  *
27  * @test
28  * @run
29  */
30 
31 function checkHexLiteral(str) {
32     if (! isNaN(Number(str))) {
33         fail("Number(" + str + ") is not NaN");
34     }
35 }
36 
37 checkHexLiteral("0x0.0");
38 checkHexLiteral("0x0.0p");
39 checkHexLiteral("0x12tu");
40 checkHexLiteral("0x12.2e22");
41 checkHexLiteral("0xtu");
