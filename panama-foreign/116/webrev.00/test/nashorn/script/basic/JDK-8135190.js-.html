31 // Make sure huge object literals are parsed correctly and don't throw
32 // (using buildObject -> JSON.stringify -> eval -> testObject)
33 
34 function buildObject(n, d) {
35     if (n < 2) {
36         return {name: "property", type: "identifier"};
37     }
38     var obj = {};
39     for (var i = 0; i < n; i++) {
40         obj["expr" + i] = buildObject(Math.floor(n / d), d);
41     }
42     return obj;
43 }
44 
45 function testObject(obj, n, d) {
46     var keys = Object.keys(obj);
47     if (n < 2) {
48         Assert.assertTrue(keys.length === 2);
49         Assert.assertTrue(keys[0] === "name");
50         Assert.assertTrue(keys[1] === "type");
51     } else {
52         Assert.assertTrue(keys.length === n);
53         for (var i = 0; i < n; i++) {
54             Assert.assertTrue(keys[i] === "expr" + i);
55         }
56     }
57     if (n >= 2) {
58         for (var k in keys) {
59             testObject(obj[keys[k]], Math.floor(n / d), d)
60         }
61     }
62 }
63 
64 var fieldObject = (eval("(" + JSON.stringify(buildObject(25, 2)) + ")"));
65 testObject(fieldObject, 25, 2);
66 var spillObject = (eval("(" + JSON.stringify(buildObject(1000, 100)) + ")"));
67 testObject(spillObject, 1000, 100);
