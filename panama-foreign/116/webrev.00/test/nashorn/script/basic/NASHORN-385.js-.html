24 /**
25  * NASHORN-385 : JSAdapter throws AssertionError when adaptee does not define hook methods like __get__, __put__ etc.
26  *
27  * @test
28  * @run
29  */
30 
31 var proto = {};
32 var o = new JSAdapter(proto);
33 
34 function checkGetter() {
35     print("o.foo => " + o.foo);
36 }
37 
38 function checkSetter() {
39     o.bar = 44;
40 }
41 
42 function checkCall() {
43     try {
44         o.func();
45     } catch (e) {
46         print(e);
47     }
48 }
49 
50 checkGetter();
51 checkSetter();
52 checkCall();
53 
54 proto.__get__ = function(name) {
55     print("in __get__: " + name);
56     return name;
57 };
58 
59 proto.__put__ = function(name) {
60     print("in __put__: " + name);
61 }
62 
63 proto.__call__ = function(name) {
64     print("in __call__: " + name);
65 }
66 
67 checkGetter();
68 checkSetter();
69 checkCall();
