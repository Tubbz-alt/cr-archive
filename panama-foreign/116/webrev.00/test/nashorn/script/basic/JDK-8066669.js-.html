31 // Make sure index access on Java objects is working as expected.
32 var map = new java.util.LinkedHashMap();
33 
34 map["foo"] = "bar";
35 map[1] = 2;
36 map[false] = true;
37 map[null] = 0;
38 map["a"] = null;
39 
40 print(map);
41 
42 var keys =  map.keySet().iterator();
43 
44 while(keys.hasNext()) {
45     var key = keys.next();
46     print(typeof key, key);
47 }
48 
49 print(typeof map["foo"], map["foo"]);
50 print(typeof map[1], map[1]);
51 print(typeof map[false], map[false]);
52 print(typeof map[null], map[null]);
53 print(typeof map["a"], map["a"]);
54 
55 print("map.foo=" + map.foo);
56 print("map.false=" + map.false);
57 print("map.null=" + map.null);
58 print("map.a=" + map.a);
59 
60 map.foo = "baz";
61 print(map);
