24 /**
25  * NASHORN-473 : Java primitive arrays can not be iterated with for.. in and for each .. in constructs
26  *
27  * @test
28  * @run
29  */
30 
31 var boolArr = new (Java.type("boolean[]"))(2);
32 boolArr[0] = true;
33 boolArr[1] = false;
34 
35 for (i in boolArr) {
36     print(i + " -> " + boolArr[i]);
37 }
38 
39 for each (i in boolArr) {
40     print(i);
41 }
