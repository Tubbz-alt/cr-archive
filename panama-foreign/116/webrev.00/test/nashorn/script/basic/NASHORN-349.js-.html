24 /**
25  * Remove lower optimization to transform obj['foo'] as obj.foo
26  *
27  * @test
28  * @run
29  */
30 
31 var m = new java.util.HashMap();
32 m.put("foo", "hello");
33 
34 // map allows keys to be accessed using indexed access
35 if (m['foo'] !== 'hello') {
36     fail("m['foo'] !== 'hello'");
37 }
38 
39 var prop = 'foo';
40 if (m[prop] !== 'hello') {
41     fail("m['foo'] !== 'hello'");
42 }
43 
44 // map keys are accessible as properties when there's no matching bean property.
45 if (m.foo !== 'hello') {
46     fail("m.foo !== 'hello'");
47 }
