24 /**
25  * Various kinds of "new" usage.
26  *
27  * @test
28  * @run
29  */
30 
31 var File = java.io.File;
32 print(! new File(".").toString().isEmpty());
33 
34 var obj = {
35    foo : function (x) {
36        this.bar = x;
37    }
38 }
39 
40 print(new obj.foo(23).bar);
41 
42 function func() {
43     this.foo = 42;
44 }
45 
46 print(new func().foo);
47 
48 print ((new function() { this.x = "hello" }).x);
49 
50 var abc = {
51    bar: function() {
52       return { x : "hello world" };
53    }
54 };
55 
56 print(new abc.bar().x);
57 
58 function func2() {
59     return {
60         foo: function() {
61             print("foo");
62         }
63     };
64 };
65 
66 print(new func2().foo());
