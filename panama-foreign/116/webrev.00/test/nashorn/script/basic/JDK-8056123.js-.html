25  * JDK-8056123: Anonymous function statements leak internal function names into global scope
26  *
27  * @test
28  * @run
29  */
30 
31 // do *not* introduce newlines before this line!
32 function () { // line 32
33   print("hello function!");
34 }
35 
36 if (typeof this["L:32"] != 'undefined') {
37    fail("anonymous function statement leaks name in global");
38 }
39 
40 var keys = Object.keys(this);
41 for (var i in keys) {
42     if (keys[i].contains("L:")) {
43         fail("found " + keys[i] + " in global scope");
44     }
