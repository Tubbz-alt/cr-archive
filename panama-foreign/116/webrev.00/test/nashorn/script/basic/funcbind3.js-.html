24 /**
25  * Test the assumptions about bound function prototypes
26  *
27  * @test
28  * @run
29  */
30 
31 function printAssumption(__x__) {
32   print(__x__ + ": " + eval(__x__));
33 }
34 
35 function f() { }
36 
37 var b = f.bind(null)
38 
39 var x = new f()
40 var y = new b()
41 
42 printAssumption("x instanceof f")
43 printAssumption("x instanceof b")
44 printAssumption("y instanceof f")
45 printAssumption("y instanceof b")
46 
47 print("\nChanging prototype\n");
48 
49 f.prototype=new Object()
50 
51 printAssumption("x instanceof f")
52 printAssumption("x instanceof b")
53 printAssumption("y instanceof f")
54 printAssumption("y instanceof b")
55 
56 print("\Bound function prototype\n");
57 
58 printAssumption("f.hasOwnProperty('prototype')")
59 printAssumption("b.hasOwnProperty('prototype')")
