24 /**
25  * NASHORN-418 : When a data descriptor property is changed to an accessor descriptor property, new configurable, enumerable flags are not set
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = { foo: 33 };
32 
33 var oldDesc = Object.getOwnPropertyDescriptor(obj, "foo");
34 if (! oldDesc.enumerable) {
35     fail("#1 'foo' is not enumerable");
36 }
37 
38 if (! oldDesc.configurable) {
39     fail("#2 'foo' is not configurable");
40 }
41 
42 Object.defineProperty(obj, "foo", {
43     enumerable: false, configurable: false,
44     get: function() { return 'hello' } });
45 
46 var newDesc = Object.getOwnPropertyDescriptor(obj, "foo");
47 if (newDesc.enumerable) {
48     fail("#3 'foo' is enumerable");
49 }
50 
51 if (newDesc.configurable) {
52     fail("#4 'foo' is configurable");
53 }
54 
55 var obj2 = {};
56 
57 Object.defineProperty(obj2, "foo", {
58     value : 34, writable: false, configurable: true });
59 
60 var setterCalled = true;
61 // descriptor type change, writable is not inherited
62 Object.defineProperty(obj2, "foo", {
63   get: undefined, set: function(x) { setterCalled = true } });
64 
65 // can still attempt to write
66 obj2.foo = 44;
67 if (! setterCalled) {
68     fail("#5 obj2.foo setter not called");
69 }
70 
71 var obj3 = {};
72 
73 Object.defineProperty(obj3, "foo", {
74   get: undefined, set: function(x) { }, configurable: true });
75 
76 Object.defineProperty(obj3, "foo", { value: 33 });
77 
78 var desc = Object.getOwnPropertyDescriptor(obj3, "foo");
79 if (desc.writable) {
80     fail("#6 obj3.foo is writable");
81 }
