24 /**
25  * NASHORN-168 :  Object.prototype.toString does not work as per the spec.
26  *
27  * @test
28  * @run
29  */
30 
31 function printToString(obj) {
32     print(Object.prototype.toString.call(obj));
33 }
34 
35 printToString(null);
36 printToString(undefined);
37 (function () {
38     printToString(arguments);
39 })();
40 printToString([]);
41 printToString(new Boolean(false));
42 printToString(new Date(0));
43 printToString(Object);
44 printToString(JSON);
45 printToString(Math);
46 printToString(new Number(0));
47 printToString({});
48 printToString(/js/);
49 printToString(new String(''));
50 
51 // check the primitives too
52 printToString(false);
53 printToString(3.14);
54 printToString('js');
