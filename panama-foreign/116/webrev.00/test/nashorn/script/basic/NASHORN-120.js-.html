24 /**
25  * NASHORN-120 :  String.prototype.split does not convert the input arguments in the order specified
26  *
27  * @test
28  * @run
29  */
30 
31 var obj1 = {
32     toString: function() {
33         throw "inside obj1.toString";
34     }
35 };
36 
37 var obj2 = {
38     valueOf: function() {
39         throw "inside obj2.valueOf";
40     }
41 };
42 
43 try {
44     "hello".split(obj1, obj2);
45     fail('#1: split not throwing exception on input conversion');
46 } catch (e) {
47     if (e !== "inside obj2.valueOf") {
48         fail('#2: Exception === "obj2.valueOf" got: '+e);
49     }
50 }
51 
52 try {
53     "hello".split(obj1, 10);
54     fail('#3: split not throwing exception on input conversion');
55 } catch (e) {
56     if (e !== "inside obj1.toString") {
57         fail('#4: Exception === "obj1.toString" got: '+e);
58     }
59 }
