24 /**
25  * JDK-8006570 : this-value for non-strict functions should be converted to object
26  *
27  * @test
28  * @run
29  */
30 
31 var strict, nonstrict;
32 
33 nonstrict = Object.prototype.nonstrict = function nonstrict() {
34     print(typeof this, this instanceof Object);
35 };
36 
37 (function() {
38     "use strict";
39     strict = Object.prototype.strict = function strict() {
40         print(typeof this, this instanceof Object);
41     };
42 })();
43 
44 "foo".nonstrict();
45 (1).nonstrict();
46 true.nonstrict();
47 nonstrict();
48 nonstrict.call(null);
49 nonstrict.call("foo");
50 nonstrict.call(1);
51 nonstrict.call(true);
52 
53 "foo".strict();
54 (1).strict();
55 true.strict();
56 strict();
57 strict.call(null);
58 strict.call("foo");
59 strict.call(1);
60 strict.call(true);
