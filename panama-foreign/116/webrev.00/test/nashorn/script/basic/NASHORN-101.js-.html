24 /**
25  * NASHORN-101:  switch statement should use === rather than == to compare case values against switch expression.
26  *
27  * @test
28  * @run
29  */
30 
31 function func(x) {
32     switch(x) {
33         case 0:
34             print("x === 0"); break;
35         case 1:
36             print("x === 1"); break;
37         case true:
38             print("x === true"); break;
39         case false:
40             print("x === false"); break;
41         case null:
42             print("x === null"); break;
43         case undefined:
44             print("x === undefined"); break;
45         default:
46             print("default case x " + x); break;
47     }
48 }
49 
50 func(0);
51 func(1);
52 func(2);
53 func("hello");
54 func(null);
55 func(undefined);
56 func(true);
57 func(false);
58 func({});
