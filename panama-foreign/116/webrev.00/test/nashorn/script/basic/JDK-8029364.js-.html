24 /**
25  * JDK-8029364: NashornException to expose thrown object
26  *
27  * @test
28  * @run
29  */
30 
31 var m = new javax.script.ScriptEngineManager();
32 var e = m.getEngineByName("nashorn");
33 var g = e.eval("this");
34 try {
35     e.eval("var e = new Error('foo'); e.bar = 33; throw e");
36 } catch (se) {
37     // ScriptException instance's cause is a NashornException
38     print(se.getClass());
39     var cause = se.cause;
40     print(cause.getClass());
41     // NashornException instance has 'ecmaError' bean getter
42     print(cause.ecmaError);
43     // access to underlying ECMA Error object
44     print(cause.ecmaError instanceof g.Error);
45     print(cause.ecmaError.name);
46     print(cause.ecmaError.message);
47     print(cause.ecmaError.bar);
48 }
