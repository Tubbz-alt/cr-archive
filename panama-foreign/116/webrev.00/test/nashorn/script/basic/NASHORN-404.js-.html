24 /**
25  * NASHORN-404 : Error.prototype and subtypes do not have any enumerable properties
26  *
27  * @test
28  * @run
29  */
30 
31 function check(name) {
32     var obj = this[name].prototype;
33     for (i in obj) {
34         fail("enumerable property: " + name  + ".prototype." + i);
35     }
36 }
37 
38 check("Error");
39 check("EvalError");
40 check("RangeError");
41 check("ReferenceError");
42 check("SyntaxError");
43 check("URIError");
