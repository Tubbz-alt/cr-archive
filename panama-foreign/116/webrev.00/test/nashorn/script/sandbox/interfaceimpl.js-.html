24 /**
25  * Check that user defined interface can be implemented.
26  *
27  * @test
28  * @run
29  * @security
30  */
31 
32 var Window = Java.type("jdk.nashorn.api.scripting.test.Window");
33 var WindowEventHandler = Java.type("jdk.nashorn.api.scripting.test.WindowEventHandler");
34 
35 var w = new Window();
36 
37 var loadedFuncReached = false;
38 // try function to SAM converter
39 w.onload = function() {
40     loadedFuncReached = true;
41     return true;
42 }
43 
44 w.onload.loaded();
45 if (! loadedFuncReached) {
46     fail("Interface method impl. not called");
47 }
48 
49 // reset
50 loadedFuncReached = false;
51 
52 // try direct interface implementation
53 w.onload = new WindowEventHandler() {
54     loaded: function() {
55         loadedFuncReached = true;
56         return true;
57     }
58 };
59 
60 w.onload.loaded();
61 if (! loadedFuncReached) {
62     fail("Interface method impl. not called");
63 }
