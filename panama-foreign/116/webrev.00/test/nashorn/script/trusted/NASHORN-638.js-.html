24 /**
25  * NASHORN-638 : Callsite tracing and profiling are broken
26  *
27  * @test
28  * @run
29  */
30 
31 /*
32  * creates new script engine initialized with given options and
33  * runs given code on it. Returns standard output captured.
34  */
35 
36 function runScriptEngine(opts, code) {
37     var imports = new JavaImporter(
38         Packages.jdk.nashorn.api.scripting,
39         java.io, java.lang, java.util);
40 
41     with(imports) {
42         var fac = new NashornScriptEngineFactory();
43         // get current System.err
44         var oldErr = System.err;
45         var baos = new ByteArrayOutputStream();
46         var newErr = new PrintStream(baos);
47         try {
48             // set new standard err
49             System.setErr(newErr);
50             var engine = fac.getScriptEngine(Java.to(opts, "java.lang.String[]"));
51             engine.eval(code);
52             newErr.flush();
53             return new java.lang.String(baos.toByteArray());
54         } finally {
55             // restore System.err to old value
56             System.setErr(oldErr);
57         }
58     }
59 }
60 
61 var str = runScriptEngine([ "-tcs=all" ], "new Date");
62 print("hello, world!");
63 
64 if (str.indexOf(" ENTER ") == -1) {
65     fail("expected 'ENTER' in trace mode output");
66 }
67 
68 if (str.indexOf(" EXIT ") == -1) {
69     fail("expected 'EXIT' in trace mode output");
70 }
