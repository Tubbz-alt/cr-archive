32 // check that builtin objects share property map
33 
34 assertSameMap(new Boolean(true), new Boolean(false));
35 assertSameMap(new Number(3), new Number(Math.PI));
36 assertSameMap(new String('hello'), new String('world'));
37 assertSameMap(new Object(), new Object());
38 assertSameMap(/hello/, /world/);
39 // try w/without regexp flags
40 assertSameMap(/hello/i, /world/g);
41 assertSameMap(new Date(), new Date());
42 assertSameMap(new Date(2000, 1, 1), new Date(1972, 5, 6));
43 assertSameMap(Function(), Function());
44 assertSameMap(Function("x", "return x"), Function("x", "return x*x"));
45 assertSameMap(new Error(), new Error());
46 assertSameMap(new Error('foo'), new Error('bar'));
47 assertSameMap(new EvalError(), new EvalError());
48 assertSameMap(new EvalError('foo'), new EvalError('bar'));
49 assertSameMap(new RangeError(), new RangeError());
50 assertSameMap(new RangeError('foo'), new RangeError('bar'));
51 assertSameMap(new ReferenceError(), new ReferenceError());
52 assertSameMap(new ReferenceError('foo'), new ReferenceError('bar'));
53 assertSameMap(new SyntaxError(), new SyntaxError());
54 assertSameMap(new SyntaxError('foo'), new SyntaxError('bar'));
55 assertSameMap(new TypeError(), new TypeError());
56 assertSameMap(new TypeError('foo'), new TypeError('bar'));
57 assertSameMap(new URIError(), new URIError());
58 assertSameMap(new URIError('foo'), new URIError('bar'));
