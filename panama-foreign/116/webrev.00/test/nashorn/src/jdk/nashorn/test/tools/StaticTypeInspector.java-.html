25 package jdk.nashorn.test.tools;
26 
27 import jdk.nashorn.internal.runtime.Undefined;
28 
29 @SuppressWarnings("javadoc")
30 public class StaticTypeInspector {
31 
32     public static String inspect(final boolean x, final String w) {
33         return w + ": boolean";
34     }
35 
36     public static String inspect(final int x, final String w) {
37         return w + ": int";
38     }
39 
40     public static String inspect(final long x, final String w) {
41         return w + ": long";
42     }
43 
44     public static String inspect(final double x, final String w) {
45         return w + ": double";
46     }
47 
48     public static String inspect(final Undefined x, final String w) {
49         return w + ": undefined";
50     }
51 
52     public static String inspect(final Object x, final String w) {
53         return w + ": object";
54     }
55 }
