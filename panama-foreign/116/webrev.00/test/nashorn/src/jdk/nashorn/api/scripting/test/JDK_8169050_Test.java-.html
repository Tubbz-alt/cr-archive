25 package jdk.nashorn.api.scripting.test;
26 
27 import javax.script.ScriptEngine;
28 import javax.script.ScriptEngineManager;
29 import javax.script.ScriptException;
30 import org.testng.annotations.BeforeClass;
31 import org.testng.annotations.Test;
32 import static org.testng.Assert.*;
33 
34 /**
35  * @bug 8169050
36  * @summary underscore_linker.js sample fails after dynalink changes for JDK-8168005
37  */
38 public class JDK_8169050_Test {
39     private ScriptEngine engine;
40 
41     @BeforeClass
42     public void setupTest() {
43         engine = new ScriptEngineManager().getEngineByName("js");
44     }
45 
46     @Test
47     public void testUndersoreName() throws ScriptException {
48         engine.eval("var S = java.util.stream.Stream, v = 0;");
49         // The underscore name 'for_each' exercises pluggable dynalink linker
50         engine.eval("S.of(4, 5, 9).for_each(function(x) { v += x })");
51         assertEquals(18, ((Number)engine.get("v")).intValue());
52     }
53 }
