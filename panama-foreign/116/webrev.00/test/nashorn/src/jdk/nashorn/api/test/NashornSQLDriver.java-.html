26 package jdk.nashorn.api.test;
27 
28 import java.sql.Connection;
29 import java.sql.Driver;
30 import java.sql.DriverManager;
31 import java.sql.DriverPropertyInfo;
32 import java.sql.SQLException;
33 import java.sql.SQLFeatureNotSupportedException;
34 import java.util.Properties;
35 import java.util.logging.Logger;
36 
37 /**
38  * A dummy SQL driver for testing purpose.
39  */
40 public final class NashornSQLDriver implements Driver {
41     static {
42         try {
43             DriverManager.registerDriver(new NashornSQLDriver(), null);
44         } catch (final SQLException se) {
45             throw new RuntimeException(se);
46         }
47     }
48 
49     @Override
50     public boolean acceptsURL(final String url) {
51         return url.startsWith("jdbc:nashorn:");
52     }
53 
54     @Override
55     public Connection connect(final String url, final Properties info) {
56         throw new UnsupportedOperationException("I am a dummy!!");
57     }
58 
59     @Override
60     public int getMajorVersion() {
61         return -1;
62     }
63 
64     @Override
65     public int getMinorVersion() {
66         return -1;
67     }
68 
69     @Override
70     public DriverPropertyInfo[] getPropertyInfo(final String url, final Properties info) {
71         return new DriverPropertyInfo[0];
72     }
73 
74     @Override
75     public boolean jdbcCompliant() {
76         // no way!
77         return false;
78     }
79 
80     @Override
81     public Logger getParentLogger() throws SQLFeatureNotSupportedException {
82         throw new SQLFeatureNotSupportedException();
83     }
84 }
