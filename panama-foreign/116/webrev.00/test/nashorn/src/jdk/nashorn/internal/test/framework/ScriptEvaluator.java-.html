26 package jdk.nashorn.internal.test.framework;
27 
28 import java.io.IOException;
29 import java.io.OutputStream;
30 
31 /**
32  * Support for script evaluation.
33  */
34 public interface ScriptEvaluator {
35     /**
36      * Runs one or more scripts files with given output and error stream set in the context
37      *
38      * @param out OutputStream for output from script evaluation
39      * @param err OutputStream for errors from script evaluation
40      * @param args one or more script arguments
41      * @return success or error code - the codes used are same as used by Shell class
42      * @throws IOException throws if there are issues with stream handling
43      */
44     public int run(OutputStream out, OutputStream err, String[] args) throws IOException;
45 }
