26 package jdk.nashorn.internal.runtime.test;
27 
28 import jdk.nashorn.api.scripting.NashornScriptEngine;
29 import jdk.nashorn.api.scripting.NashornScriptEngineFactory;
30 import javax.script.ScriptException;
31 import static org.testng.Assert.assertEquals;
32 import org.testng.annotations.Test;
33 
34 /**
35  * @test
36  * @bug 8187362
37  * @summary Nashorn unsigned right shift operator unexpectedly returns floating-point
38  * @modules jdk.scripting.nashorn/jdk.nashorn.internal.runtime
39  * @run testng jdk.nashorn.internal.runtime.test.JDK_8187362_Test
40  */
41 public class JDK_8187362_Test {
42 
43     @Test
44     public void testOperator() throws ScriptException {
45         NashornScriptEngine engine = (NashornScriptEngine) new NashornScriptEngineFactory().getScriptEngine();
46         assertEquals(engine.eval("100 >>> 1"), 50);
47         assertEquals(engine.eval("100 >> 1"), 50);
48     }
49 }
