26 package jdk.dynalink.test;
27 
28 import java.util.List;
29 import jdk.dynalink.CallSiteDescriptor;
30 import jdk.dynalink.linker.GuardingDynamicLinker;
31 import jdk.dynalink.linker.GuardingDynamicLinkerExporter;
32 import jdk.dynalink.linker.LinkRequest;
33 import jdk.dynalink.linker.LinkerServices;
34 
35 /**
36  * A trusted linker exporter (build file gives appropriate permission to the jar containing this class!).
37  */
38 public final class TrustedGuardingDynamicLinkerExporter extends GuardingDynamicLinkerExporter {
39 
40     private static final ThreadLocal<CallSiteDescriptor> lastDescriptor = new ThreadLocal<>();
41     private static boolean enabled = false;
42 
43     public static void enable() {
44         reset(true);
45     }
46 
47     public static void disable() {
48         reset(false);
49     }
50     public static boolean isLastCallSiteDescriptor(final CallSiteDescriptor desc) {
51         return lastDescriptor.get() == desc;
52     }
53 
54     private static void reset(final boolean enable) {
55         lastDescriptor.set(null);
56         enabled = enable;
57     }
58 
59     @Override
60     public List<GuardingDynamicLinker> get() {
61         return List.of(((GuardingDynamicLinker) (final LinkRequest linkRequest, final LinkerServices linkerServices) -> {
62             if (enabled) {
63                 lastDescriptor.set(linkRequest.getCallSiteDescriptor());
64             }
65             return null;
66         }));
67     }
68 }
