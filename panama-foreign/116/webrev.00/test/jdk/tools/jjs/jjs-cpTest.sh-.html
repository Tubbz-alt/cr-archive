27 # @test
28 # @bug 8145750
29 # @summary jjs fails to run simple scripts with security manager turned on
30 # @run shell jjs-cpTest.sh
31 # Tests -cp/-classpath option to set the classpath for jjs
32 
33 . ${TESTSRC-.}/common.sh
34 
35 setup
36 
37 rm -f Hello.class
38 ${JAVAC} ${TESTSRC}/Hello.java -d .
39 
40 # we check whether classpath setting for app classes
41 # work with jjs. Script should be able to
42 # access Java class "Hello".
43 
44 ${JJS} -J-Djava.security.manager -cp . ${TESTSRC}/classpath.js
45 
46 if [ $? -ne 0 ]; then
47    exit 1
48 fi
49 
50 # -classpath and -cp are synonyms
51 
52 ${JJS} -J-Djava.security.manager -classpath . ${TESTSRC}/classpath.js
53 
54 if [ $? -ne 0 ]; then
55    exit 1
56 fi
57 
58 rm -f Hello.class
59 echo "Passed"
60 exit 0
