 24 import java.nio.file.Path;
 25 import java.util.List;
 26 
 27 import jdk.jpackage.test.TKit;
 28 import jdk.jpackage.test.Executor;
 29 import jdk.jpackage.test.Executor.Result;
 30 
 31 public class SigningBase {
 32 
 33     public static String DEV_NAME = "jpackage.openjdk.java.net";
 34     public static String APP_CERT
 35             = "Developer ID Application: " + DEV_NAME;
 36     public static String INSTALLER_CERT
 37             = "Developer ID Installer: " + DEV_NAME;
 38     public static String KEYCHAIN = "jpackagerTest.keychain";
 39 
 40     private static void checkString(List<String> result, String lookupString) {
 41         TKit.assertTextStream(lookupString).predicate(
 42                 (line, what) -> line.trim().equals(what)).apply(result.stream());
 43     }
 44 
 45     private static List<String> codesignResult(Path target, boolean signed) {
 46         int exitCode = signed ? 0 : 1;
 47         List<String> result = new Executor()
 48                 .setExecutable("codesign")
 49                 .addArguments("--verify", "--deep", "--strict", "--verbose=2",
 50                         target.toString())
 51                 .saveOutput()
 52                 .execute(exitCode).getOutput();
 53 
 54         return result;
 55     }
 56 
 57     private static void verifyCodesignResult(List<String> result, Path target,
 58             boolean signed) {
 59         result.stream().forEachOrdered(TKit::trace);
 60         if (signed) {
 61             String lookupString = target.toString() + ": valid on disk";
 62             checkString(result, lookupString);
 63             lookupString = target.toString() + ": satisfies its Designated Requirement";
 64             checkString(result, lookupString);
 65         } else {
 66             String lookupString = target.toString()
 67                     + ": code object is not signed at all";
 68             checkString(result, lookupString);
 69         }
 70     }
 71 
 72     private static Result spctlResult(Path target, String type) {
 73         Result result = new Executor()
 74                 .setExecutable("/usr/sbin/spctl")
 75                 .addArguments("-vvv", "--assess", "--type", type,
 76                         target.toString())
 77                 .saveOutput()
 78                 .executeWithoutExitCodeCheck();
 79 
 80         // allow exit code 3 for not being notarized
 81         if (result.getExitCode() != 3) {
 82             result.assertExitCodeIsZero();
 83         }
 84         return result;
 85     }
 86 
 87     private static void verifySpctlResult(List<String> output, Path target,
 88             String type, int exitCode) {
 89         output.stream().forEachOrdered(TKit::trace);
 90         String lookupString;
 91 
 92         if (exitCode == 0) {
 93             lookupString = target.toString() + ": accepted";
 94             checkString(output, lookupString);
 95             lookupString = "source=" + DEV_NAME;
 96             checkString(output, lookupString);
 97         } else if (exitCode == 3) {
 98             // allow failure purely for not being notarized
 99             lookupString = target.toString() + ": rejected";
100             checkString(output, lookupString);
101         }
102 
103         if (type.equals("install")) {
104             lookupString = "origin=" + INSTALLER_CERT;
105         } else {
106             lookupString = "origin=" + APP_CERT;
107         }
108         checkString(output, lookupString);
109     }
110 
111     private static List<String> pkgutilResult(Path target) {
112         List<String> result = new Executor()
113                 .setExecutable("/usr/sbin/pkgutil")
114                 .addArguments("--check-signature",
115                         target.toString())
116                 .executeAndGetOutput();
117 
118         return result;
119     }
120 
121     private static void verifyPkgutilResult(List<String> result) {
122         result.stream().forEachOrdered(TKit::trace);
123         String lookupString = "Status: signed by a certificate trusted for current user";
124         checkString(result, lookupString);
125         lookupString = "1. " + INSTALLER_CERT;
126         checkString(result, lookupString);
127     }
128 
129     public static void verifyCodesign(Path target, boolean signed) {
130         List<String> result = codesignResult(target, signed);
131         verifyCodesignResult(result, target, signed);
132     }
133 
134     public static void verifySpctl(Path target, String type) {
135         Result result = spctlResult(target, type);
136         List<String> output = result.getOutput();
137 
138         verifySpctlResult(output, target, type, result.getExitCode());
139     }
140 
141     public static void verifyPkgutil(Path target) {
142         List<String> result = pkgutilResult(target);
143         verifyPkgutilResult(result);
144     }
145 
146 }
