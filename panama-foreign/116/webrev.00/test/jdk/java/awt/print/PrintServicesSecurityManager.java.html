24 import java.util.Arrays;
25 
26 import javax.print.PrintServiceLookup;
27 
28 /*
29  * @test
30  * @bug 8241829
31  */
32 public final class PrintServicesSecurityManager {
33 
34     public static void main(String[] args) throws InterruptedException {
35         System.setSecurityManager(new SecurityManager());
36         test();
37         Thread.sleep(3000); // to be sure the pooling thread started
38         test();
39     }
40 
41     private static void test() {
42         Object[] services = PrintServiceLookup.lookupPrintServices(null, null);
43         if (services.length != 0) {
44             System.err.println("services = " + Arrays.toString(services));
45             throw new RuntimeException("The array of Services must be empty");
46         }
47     }
48 }
