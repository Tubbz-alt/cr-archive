24 /*
25  * @test
26  * @library /test/lib
27  * @modules jdk.jartool/sun.tools.jar
28  * @build jdk.test.lib.apps.LingeredApp
29  * @run main/othervm/timeout=360 TestJps
30  */
31 
32  /*
33   * Notes:
34   *   @modules tag is ignored in driver mode, so need main/othervm
35   *
36   *   LingeredApp is pre-built separately to have jdk.test.lib classes ready
37   *   before the test is compiled (see JDK-8242282).
38   *
39   *   Launching the process with relative path to an app jar file is not tested
40   *
41   *   This test resides in default package, so correct appearance
42   *   of the full package name actually is not tested.
43   */
44 
45 import jdk.test.lib.apps.LingeredApp;
46 
47 public class TestJps {
48 
49     public static void testJps(boolean useJar) throws Throwable {
50         LingeredAppForJps app = new LingeredAppForJps();
51         if (useJar) {
52             app.buildJar();
53         }
54         try {
55             LingeredApp.startApp(app, JpsHelper.getVmArgs());
56             JpsHelper.runJpsVariants(app.getPid(),
57                     app.getProcessName(), app.getFullProcessName(), app.getLockFileName());
58         } finally {
59             LingeredApp.stopApp(app);
60         }
61     }
62 
63     public static void main(String[] args) throws Throwable {
64         testJps(false);
65         testJps(true);
66     }
67 }
