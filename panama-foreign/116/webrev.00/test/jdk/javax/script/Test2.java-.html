31 import java.io.File;
32 import java.io.FileReader;
33 import javax.script.ScriptEngine;
34 import javax.script.ScriptEngineManager;
35 
36 public class Test2 {
37         public static class Testobj {
38                 private String val;
39                 public Testobj(String s) {
40                         val = s;
41                 }
42                 public void setVal(String v) {
43                         val = v;
44                 }
45                 public String getVal() {
46                         return val;
47                 }
48                 public String toString() {
49                         return "Testobj containing " + val;
50                 }
51         }
52 
53         public static void main(String[] args) throws Exception {
54             System.out.println("\nTest2\n");
55             ScriptEngineManager m = new ScriptEngineManager();
56             ScriptEngine eng = Helper.getJsEngine(m);
57             if (eng == null) {
58                      System.out.println("Warning: No js engine found; test vacuously passes.");
59                      return;
60             }
61             eng.put("Testobj", new Testobj("Hello World"));
62             eng.eval(new FileReader(
63                     new File(System.getProperty("test.src", "."), "Test2.js")));
64         }
65 }
