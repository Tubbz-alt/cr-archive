31 import java.io.File;
32 import java.io.FileReader;
33 import java.io.Reader;
34 import javax.script.ScriptEngine;
35 import javax.script.ScriptEngineManager;
36 
37 public class Test7 {
38         public static void main(String[] args) throws Exception {
39             System.out.println("\nTest7\n");
40             File file =
41                 new File(System.getProperty("test.src", "."), "Test7.js");
42             Reader r = new FileReader(file);
43             ScriptEngineManager m = new ScriptEngineManager();
44             ScriptEngine eng = Helper.getJsEngine(m);
45             if (eng == null) {
46                 System.out.println("Warning: No js engine found; test vacuously passes.");
47                 return;
48             }
49             eng.put("filename", file.getAbsolutePath());
50             eng.eval(r);
51             String str = (String)eng.get("firstLine");
52             // do not change first line in Test7.js -- we check it here!
53             if (!str.equals("//this is the first line of Test7.js")) {
54                 throw new RuntimeException("unexpected first line");
55             }
56         }
57 }
