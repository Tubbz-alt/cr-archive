30 import java.io.File;
31 import java.io.FileReader;
32 import javax.script.Invocable;
33 import javax.script.ScriptEngine;
34 import javax.script.ScriptEngineManager;
35 
36 public class Test8 {
37         public static void main(String[] args) throws Exception {
38             System.out.println("\nTest8\n");
39             ScriptEngineManager m = new ScriptEngineManager();
40             ScriptEngine e  = Helper.getJsEngine(m);
41             if (e == null) {
42                 System.out.println("Warning: No js engine found; test vacuously passes.");
43                 return;
44             }
45             e.eval(new FileReader(
46                 new File(System.getProperty("test.src", "."), "Test8.js")));
47             Invocable inv = (Invocable)e;
48             inv.invokeFunction("main", "Mustang");
49             // use method of a specific script object
50             Object scriptObj = e.get("scriptObj");
51             inv.invokeMethod(scriptObj, "main", "Mustang");
52         }
53 }
