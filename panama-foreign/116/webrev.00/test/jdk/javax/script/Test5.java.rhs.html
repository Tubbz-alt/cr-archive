27  * @summary Tests engine, global scopes and scope hiding.

30 import java.io.File;
31 import java.io.FileReader;
32 import java.io.Reader;
33 import javax.script.Bindings;
34 import javax.script.ScriptContext;
35 import javax.script.ScriptEngine;
36 import javax.script.ScriptEngineManager;
37 import javax.script.SimpleBindings;
38 import javax.script.SimpleScriptContext;
39 
40 public class Test5 {
41         public static void main(String[] args) throws Exception {
42                 System.out.println("\nTest5\n");
43                 ScriptEngineManager m = new ScriptEngineManager();
44                 ScriptEngine engine = Helper.getJsEngine(m);
45                 if (engine == null) {
46                     System.out.println("Warning: No js engine found; test vacuously passes.");
47                     return;
48                 }
49                 Bindings g = new SimpleBindings();
50                 Bindings e = new SimpleBindings();
51                 g.put("key", "value in global");
52                 e.put("key", "value in engine");
53                 ScriptContext ctxt = new SimpleScriptContext();
54                 ctxt.setBindings(e, ScriptContext.ENGINE_SCOPE);
55                 System.out.println("engine scope only");
56                 e.put("count", new Integer(1));
57 
58                 try (Reader reader = new FileReader(
59                     new File(System.getProperty("test.src", "."), "Test5.js"))) {
60                     engine.eval(reader,ctxt);
61                 }
62 
63                 System.out.println("both scopes");
64                 ctxt.setBindings(g, ScriptContext.GLOBAL_SCOPE);
65                 e.put("count", new Integer(2));
66                 try (Reader reader = new FileReader(
67                     new File(System.getProperty("test.src", "."), "Test5.js"))) {
68                     engine.eval(reader,ctxt);
69                 }
70                 System.out.println("only global");
71                 e.put("count", new Integer(3));
72                 ctxt.removeAttribute("key", ScriptContext.ENGINE_SCOPE);
73                 try (Reader reader = new FileReader(
74                     new File(System.getProperty("test.src", "."), "Test5.js"))) {
75                     engine.eval(reader,ctxt);
76                 }
77         }
78 }
