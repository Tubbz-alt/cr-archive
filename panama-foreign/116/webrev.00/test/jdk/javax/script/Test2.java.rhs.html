27  * @summary Test exposing a Java object to script

28  */

30 import java.io.File;
31 import java.io.FileReader;
32 import javax.script.ScriptEngine;
33 import javax.script.ScriptEngineManager;
34 
35 public class Test2 {
36         public static class Testobj {
37                 private String val;
38                 public Testobj(String s) {
39                         val = s;
40                 }
41                 public void setVal(String v) {
42                         val = v;
43                 }
44                 public String getVal() {
45                         return val;
46                 }
47                 public String toString() {
48                         return "Testobj containing " + val;
49                 }
50         }
51 
52         public static void main(String[] args) throws Exception {
53             System.out.println("\nTest2\n");
54             ScriptEngineManager m = new ScriptEngineManager();
55             ScriptEngine eng = Helper.getJsEngine(m);
56             if (eng == null) {
57                      System.out.println("Warning: No js engine found; test vacuously passes.");
58                      return;
59             }
60             eng.put("Testobj", new Testobj("Hello World"));
61             eng.eval(new FileReader(
62                     new File(System.getProperty("test.src", "."), "Test2.js")));
63         }
64 }
