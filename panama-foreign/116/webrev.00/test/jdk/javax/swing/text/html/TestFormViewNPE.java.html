34 /*
35  * @test
36  * @key headful
37  * @bug 8240877
38  * @summary Verify NPE at FormView.appendBuffer with null option values
39  */
40 public class TestFormViewNPE {
41 
42     private static JEditorPane html;
43     private static JFrame frame;
44     private final static int width = 200;
45     private final static int height = 100;
46 
47     private static void setup() {
48         html = new JEditorPane("text/html",
49                 "<html><body><form action=\"http://localhost.cgi\">"
50                         + "<select name=select id=\"myCourses\" autofocus> \n" +
51                         "            <option ></option> \n" +
52                         "        </select> "
53                         + "<input type=submit name=submit value=\"submit\"/>"
54                         + "</form></body></html>");
55         frame = new JFrame();
56         frame.setLayout(new BorderLayout());
57         frame.add(html, BorderLayout.CENTER);
58         frame.setSize(width, height);
59         frame.setVisible(true);
60     }
61 
62     public static void main(String[] args) throws Exception {
63         final Robot robot = new Robot();
64         robot.setAutoDelay(100);
65 
66         try {
67             SwingUtilities.invokeAndWait(() -> setup());
68             robot.waitForIdle();
69             robot.delay(500);
70             robot.mouseMove(width/2, height/2);
71             robot.delay(500);
72             robot.mousePress(InputEvent.BUTTON1_DOWN_MASK);
73             robot.mouseRelease(InputEvent.BUTTON1_DOWN_MASK);
74         } finally {
75             SwingUtilities.invokeAndWait(() -> frame.dispose());
76         }
77     }
78 }
