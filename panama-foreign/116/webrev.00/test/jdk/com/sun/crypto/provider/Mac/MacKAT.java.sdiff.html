 45             return "(null)";

 48         for (int i = 0; i < b.length; i++) {

 50             if (i != 0) {
 51                 sb.append(':');
 52             }
 53             sb.append(hexDigits[k >>> 4]);
 54             sb.append(hexDigits[k & 0xf]);
 55         }
 56         return sb.toString();

 59     public static byte[] parse(String s) {
 60         try {
 61             int n = s.length();
 62             ByteArrayOutputStream out = new ByteArrayOutputStream(n / 3);






 63             StringReader r = new StringReader(s);
 64             while (true) {
 65                 int b1 = nextNibble(r);
 66                 if (b1 < 0) {
 67                     break;
 68                 }
 69                 int b2 = nextNibble(r);
 70                 if (b2 < 0) {
 71                     throw new RuntimeException("Invalid string " + s);
 72                 }
 73                 int b = (b1 << 4) | b2;
 74                 out.write(b);
 75             }
 76             return out.toByteArray();
 77         } catch (IOException e) {
 78             throw new RuntimeException(e);
 79         }

 63     public static byte[] parse(String s) {
 64         try {
 65             int n = s.length();
 66             ByteArrayOutputStream out;
 67             if (s.indexOf(SEP) == -1) {
 68                 out = new ByteArrayOutputStream(n / 2);
 69             } else {
 70                 out = new ByteArrayOutputStream(n / 3);
 71             }
 72 
 73             StringReader r = new StringReader(s);
 74             while (true) {
 75                 int b1 = nextNibble(r);
 76                 if (b1 < 0) {
 77                     break;
 78                 }
 79                 int b2 = nextNibble(r);
 80                 if (b2 < 0) {
 81                     throw new RuntimeException("Invalid string " + s);
 82                 }
 83                 int b = (b1 << 4) | b2;
 84                 out.write(b);
 85             }
 86             return out.toByteArray();
 87         } catch (IOException e) {
 88             throw new RuntimeException(e);
 89         }
