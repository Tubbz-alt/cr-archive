25 public class JimageClassProtDomain {
26     public static void main(String args[]) throws Throwable {
27         // Test ProtectionDomain for boot/app/ext module classes from the "modules" jimage.
28         // The following classes are archived. See runtime/AppCDS/ProtectionDomain.java.
29         //     java.util.Dictionary (testcase 0),
30         //     sun.tools.javac.Main (testcase 1),
31         //     jdk.nio.zipfs.ZipInfo (testcase 2),
32         //     java.net.URL (testcase 3),

33         //     com.sun.jndi.dns.DnsName (testcase 5)
34         String testcases[][] =
35             {{"Loading shared boot module class first",
36               "java.util.Dictionary", "java.util.ServiceConfigurationError"},
37 
38              {"Loading shared app module class first",
39               "com.sun.tools.javac.Main", "com.sun.tools.javac.code.Symbol"},
40 
41              {"Loading shared ext module class first",
42               "jdk.nio.zipfs.ZipInfo", "jdk.nio.zipfs.ZipPath"},
43 
44              {"Loading non-shared boot module class first",
45               "java.net.HttpCookie", "java.net.URL"},
46 
47              {"Loading non-shared app module class first",
48               "com.sun.tools.sjavac.Util", "com.sun.tools.sjavac.Main"},
49 
50              {"Loading non-shared ext module class first",
51               "com.sun.jndi.dns.Resolver", "com.sun.jndi.dns.DnsName"}};
52         for (int i = 0; i < testcases.length; i++) {
53             System.out.println("Testcase " + i + ": " + testcases[i][0]);
54             JimageClassProtDomain.testProtectionDomain(testcases[i][1], testcases[i][2]);
55         }
56     }
57 
58     private static void testProtectionDomain(String shared, String nonShared)
59               throws Throwable {
60         Class c1 = Class.forName(shared);
61         Class c2 = Class.forName(nonShared);
62         if (c1.getProtectionDomain() != c2.getProtectionDomain()) {
63             System.out.println(c1.getProtectionDomain());
64             System.out.println(c1.getProtectionDomain().getCodeSource());
65             System.out.println(c2.getProtectionDomain());
66             System.out.println(c2.getProtectionDomain().getCodeSource());
67             throw new RuntimeException("Failed: Protection Domains do not match!");
68         } else {
69             System.out.println(c1.getProtectionDomain());
70             System.out.println(c1.getProtectionDomain().getCodeSource());
71             System.out.println(c2.getProtectionDomain());
72             System.out.println(c2.getProtectionDomain().getCodeSource());
73             System.out.println("Passed: Protection Domains match.");
74         }
75     }
76 }
