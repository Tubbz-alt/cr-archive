25 public class JimageClassPackage {
26     public static void main(String args[]) throws Throwable {
27         // Test Package for boot/app/ext module classes from the "modules" jimage.
28         // The following classes are archived. See runtime/AppCDS/Package.java.
29         //     java.util.Dictionary (testcase 0),
30         //     sun.tools.javac.Main (testcase 1),
31         //     jdk.nio.zipfs.ZipInfo (testcase 2),
32         //     java.net.URL (testcase 3),
33         //     com.sun.jndi.dns.DnsName (testcase 4)

34         String testcases[][] =
35             {{"Loading shared boot module class first", "java.util",
36               "java.util.Dictionary", "java.util.ServiceConfigurationError"},
37 
38              {"Loading shared app module class first", "sun.tools.javac",
39               "sun.tools.javac.Main", "sun.tools.javac.BatchParser"},
40 
41              {"Loading shared ext module class first", "jdk.nio.zipfs",
42               "jdk.nio.zipfs.ZipInfo", "jdk.nio.zipfs.ZipPath"},
43 
44              {"Loading non-shared boot module class first", "java.net",
45               "java.net.HttpCookie", "java.net.URL"},
46 



47              {"Loading non-shared ext module class first", "com.sun.jndi.dns",
48               "com.sun.jndi.dns.Resolver", "com.sun.jndi.dns.DnsName"}};
49 
50         JimageClassPackage test = new JimageClassPackage();
51         for (int i = 0; i < testcases.length; i++) {
52             System.out.println("Testcase " + i + ": " + testcases[i][0]);
53             test.testPackage(testcases[i][1], testcases[i][2], testcases[i][3]);
54         }
55     }
56 
57     private void testPackage (String pkg,
58                               String shared,
59                               String nonShared) throws Throwable {
60         Class c1 = Class.forName(shared);
61         ClassLoader cl = c1.getClassLoader();
62         Package pkg_from_loader;
63         if (cl != null) {
64             pkg_from_loader = cl.getDefinedPackage(pkg);
65         } else {
66             pkg_from_loader = Package.getPackage(pkg);
67         }
68 
69         Package pkg_from_shared_class = c1.getPackage();
70 
71         Class c2 = Class.forName(nonShared);
72         Package pkg_from_nonshared_class = c2.getPackage();
73 
74         if (pkg_from_loader != null &&
75             pkg_from_shared_class != null &&
76             pkg_from_loader == pkg_from_shared_class &&
77             pkg_from_shared_class == pkg_from_nonshared_class &&
78             pkg_from_shared_class.getName().equals(pkg)) {
79             System.out.println("Expected package: " + pkg_from_shared_class.toString());
80         } else {
81             System.out.println("Unexpected package" + pkg_from_shared_class);
82             System.exit(1);
83         }
84         if (pkg_from_shared_class.isSealed()) {
85             System.out.println("Package is sealed");
86         } else {
87             System.out.println("Package is not sealed");
88             System.exit(1);
89         }
90     }
91 }
