25 /*
26  * @test
27  * @summary AppCDS handling of protection domain.
28  * @requires vm.cds
29  * @library /test/lib
30  * @compile test-classes/ProtDomain.java
31  * @compile test-classes/ProtDomainB.java
32  * @compile test-classes/JimageClassProtDomain.java
33  * @run driver ProtectionDomain
34  */
35 
36 import jdk.test.lib.process.OutputAnalyzer;
37 
38 public class ProtectionDomain {
39   public static void main(String[] args) throws Exception {
40     JarBuilder.build("prot_domain", "ProtDomain", "ProtDomainB", "ProtDomainOther",
41                      "ProtDomainBOther", "JimageClassProtDomain");
42 
43     String appJar = TestCommon.getTestJar("prot_domain.jar");
44     TestCommon.testDump(appJar,
45          TestCommon.list("ProtDomain",
46                          "ProtDomainBOther",
47                          "java/util/Dictionary",
48                          "com/sun/tools/javac/Main",
49                          "jdk/nio/zipfs/ZipInfo",
50                          "java/net/URL",
51                          "com/sun/jndi/dns/DnsName"));
52 
53     // First class is loaded from CDS, second class is loaded from JAR
54     TestCommon.run("-cp", appJar, "ProtDomain")
55         .assertNormalExit("Protection Domains match");
56 
57     // First class is loaded from JAR, second class is loaded from CDS
58     TestCommon.run("-cp", appJar, "ProtDomainB")
59         .assertNormalExit("Protection Domains match");
60 
61     // Test ProtectionDomain for application and extension module classes from the
62     // "modules" jimage
63     TestCommon.run("-cp", appJar, "JimageClassProtDomain")
64         .assertNormalExit(output -> output.shouldNotContain(
65                           "Failed: Protection Domains do not match"));
66   }
67 }
