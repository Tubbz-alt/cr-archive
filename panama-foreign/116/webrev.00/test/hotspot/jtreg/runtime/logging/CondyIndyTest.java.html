24 /*
25  * @test
26  * @summary Test -Xlog:methodhandles with a test that contains both a condy and indy.
27  * @modules java.base/jdk.internal.misc
28  * @library /test/lib
29  * @compile CondyIndyMathOperation.jasm
30  * @compile CondyIndy.jasm
31  * @run driver CondyIndyTest
32  */
33 
34 import jdk.test.lib.process.OutputAnalyzer;
35 import jdk.test.lib.process.ProcessTools;
36 import jdk.test.lib.compiler.InMemoryJavaCompiler;
37 
38 public class CondyIndyTest {
39     public static void main(String... args) throws Exception {
40 
41         // (1) methodhandles should turn on, no indy, no condy
42         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-Xlog:methodhandles",
43                                                                   "CondyIndy");
44         OutputAnalyzer o = new OutputAnalyzer(pb.start());
45         o.shouldContain("[info][methodhandles");
46         o.shouldNotContain("[debug][methodhandles,indy");
47         o.shouldNotContain("[debug][methodhandles,condy");
48 
49         // (2) methodhandles+condy=debug only
50         pb = ProcessTools.createJavaProcessBuilder("-Xlog:methodhandles+condy=debug",
51                                                    "CondyIndy");
52         o = new OutputAnalyzer(pb.start());
53         o.shouldNotContain("[info ][methodhandles");
54         o.shouldNotContain("[debug][methodhandles,indy");
55         o.shouldContain("[debug][methodhandles,condy");
56 
57         // (3) methodhandles+indy=debug only
58         pb = ProcessTools.createJavaProcessBuilder("-Xlog:methodhandles+indy=debug",
59                                                    "CondyIndy");
60         o = new OutputAnalyzer(pb.start());
61         o.shouldNotContain("[info ][methodhandles");
62         o.shouldContain("[debug][methodhandles,indy");
63         o.shouldNotContain("[debug][methodhandles,condy");
64 
65         // (4) methodhandles, condy, indy all on
66         pb = ProcessTools.createJavaProcessBuilder("-Xlog:methodhandles=info",
67                                                    "-Xlog:methodhandles+condy=debug",
68                                                    "-Xlog:methodhandles+indy=debug",
69                                                    "CondyIndy");
70         o = new OutputAnalyzer(pb.start());
71         o.shouldContain("[info ][methodhandles");
72         o.shouldContain("[debug][methodhandles,indy");
73         o.shouldContain("[debug][methodhandles,condy");
74     };
75 }
