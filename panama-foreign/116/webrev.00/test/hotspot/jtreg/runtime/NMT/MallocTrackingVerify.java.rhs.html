 24 /*
 25  * @test
 26  * @bug 8054836
 27  * @summary Test to verify correctness of malloc tracking
 28  * @key nmt jcmd randomness
 29  * @library /test/lib
 30  * @modules java.base/jdk.internal.misc
 31  *          java.management
 32  * @build sun.hotspot.WhiteBox
 33  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
 34  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -XX:NativeMemoryTracking=detail MallocTrackingVerify
 35  *
 36  */
 37 
 38 import java.util.ArrayList;
 39 import java.util.Random;
 40 
 41 import jdk.test.lib.process.ProcessTools;
 42 import jdk.test.lib.process.OutputAnalyzer;
 43 import jdk.test.lib.JDKToolFinder;
 44 import jdk.test.lib.Utils;
 45 
 46 import sun.hotspot.WhiteBox;
 47 
 48 public class MallocTrackingVerify {
 49     private static int MAX_ALLOC = 4 * 1024;
 50 
 51     static ArrayList<MallocMemory> mallocd_memory = new ArrayList<MallocMemory>();
 52     static long mallocd_total = 0;
 53     public static WhiteBox wb = WhiteBox.getWhiteBox();
 54 
 55     public static void main(String args[]) throws Exception {
 56         OutputAnalyzer output;
 57 
 58         // Grab my own PID
 59         String pid = Long.toString(ProcessTools.getProcessId());
 60         ProcessBuilder pb = new ProcessBuilder();
 61 
 62         Random random = Utils.getRandomInstance();
 63         // Allocate small amounts of memory with random pseudo call stack
 64         while (mallocd_total < MAX_ALLOC) {
 65             int size = random.nextInt(31) + 1;
 66             long addr = wb.NMTMallocWithPseudoStack(size, random.nextInt());
 67             if (addr != 0) {
 68                 MallocMemory mem = new MallocMemory(addr, size);
 69                 mallocd_memory.add(mem);
 70                 mallocd_total += size;
 71             } else {
 72                 System.out.println("Out of malloc memory");
 73                 break;
 74             }
 75         }
 76 
 77         pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid, "VM.native_memory", "summary" });
 78         output = new OutputAnalyzer(pb.start());
 79         output.shouldContain("Test (reserved=4KB, committed=4KB)");
 80 
 81         // Free
 82         for (MallocMemory mem : mallocd_memory) {
 83             wb.NMTFree(mem.addr());
 84         }
 85 
 86         // Run 'jcmd <pid> VM.native_memory summary', check for expected output
 87         pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid,
 88                 "VM.native_memory", "summary" });
 89         output = new OutputAnalyzer(pb.start());
 90         output.shouldNotContain("Test (reserved=");
 91     }
 92 
 93     static class MallocMemory {
 94         private long addr;
 95         private int size;
 96 
 97         MallocMemory(long addr, int size) {
 98             this.addr = addr;
 99             this.size = size;
100         }
101 
102         long addr() {
103             return this.addr;
104         }
105 
106         int size() {
107             return this.size;
108         }
109     }
110 }
