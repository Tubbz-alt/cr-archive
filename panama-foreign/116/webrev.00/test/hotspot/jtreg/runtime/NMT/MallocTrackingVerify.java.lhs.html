 24 /*
 25  * @test
 26  * @bug 8054836
 27  * @summary Test to verify correctness of malloc tracking
 28  * @key nmt jcmd
 29  * @library /test/lib
 30  * @modules java.base/jdk.internal.misc
 31  *          java.management
 32  * @build sun.hotspot.WhiteBox
 33  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
 34  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -XX:NativeMemoryTracking=detail MallocTrackingVerify
 35  *
 36  */
 37 
 38 import java.util.ArrayList;
 39 import java.util.Random;
 40 
 41 import jdk.test.lib.process.ProcessTools;
 42 import jdk.test.lib.process.OutputAnalyzer;
 43 import jdk.test.lib.JDKToolFinder;

 44 
 45 import sun.hotspot.WhiteBox;
 46 
 47 public class MallocTrackingVerify {
 48     private static int MAX_ALLOC = 4 * 1024;
 49 
 50     static ArrayList<MallocMemory> mallocd_memory = new ArrayList<MallocMemory>();
 51     static long mallocd_total = 0;
 52     public static WhiteBox wb = WhiteBox.getWhiteBox();
 53 
 54     public static void main(String args[]) throws Exception {
 55         OutputAnalyzer output;
 56 
 57         // Grab my own PID
 58         String pid = Long.toString(ProcessTools.getProcessId());
 59         ProcessBuilder pb = new ProcessBuilder();
 60 
 61         Random random = new Random();
 62         // Allocate small amounts of memory with random pseudo call stack
 63         while (mallocd_total < MAX_ALLOC) {
 64             int size = random.nextInt(31) + 1;
 65             long addr = wb.NMTMallocWithPseudoStack(size, random.nextInt());
 66             if (addr != 0) {
 67                 MallocMemory mem = new MallocMemory(addr, size);
 68                 mallocd_memory.add(mem);
 69                 mallocd_total += size;
 70             } else {
 71                 System.out.println("Out of malloc memory");
 72                 break;
 73             }
 74         }
 75 
 76         pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid, "VM.native_memory", "summary" });
 77         output = new OutputAnalyzer(pb.start());
 78         output.shouldContain("Test (reserved=4KB, committed=4KB)");
 79 
 80         // Free
 81         for (MallocMemory mem : mallocd_memory) {
 82             wb.NMTFree(mem.addr());
 83         }
 84 
 85         // Run 'jcmd <pid> VM.native_memory summary', check for expected output
 86         pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid,
 87                 "VM.native_memory", "summary" });
 88         output = new OutputAnalyzer(pb.start());
 89         output.shouldNotContain("Test (reserved=");
 90     }
 91 
 92     static class MallocMemory {
 93         private long addr;
 94         private int size;
 95 
 96         MallocMemory(long addr, int size) {
 97             this.addr = addr;
 98             this.size = size;
 99         }
100 
101         long addr() {
102             return this.addr;
103         }
104 
105         int size() {
106             return this.size;
107         }
108     }
109 }
