 24 import java.util.ArrayList;
 25 import java.util.List;
 26 import java.util.regex.Matcher;
 27 import java.util.regex.Pattern;
 28 
 29 import jdk.test.lib.JDKToolLauncher;
 30 import jdk.test.lib.SA.SATestUtils;
 31 import jdk.test.lib.Utils;
 32 import jdk.test.lib.apps.LingeredApp;
 33 import jdk.test.lib.process.OutputAnalyzer;
 34 
 35 /**
 36  * @test
 37  * @key randomness
 38  * @bug 8208091
 39  * @requires (os.family == "linux") & (vm.hasSA)
 40  * @library /test/lib
 41  * @run main/othervm TestJhsdbJstackMixed
 42  */
 43 public class TestJhsdbJstackMixed {
 44 
 45     private static final int MAX_ITERATIONS = 20;
 46     private static final String NATIVE_FUNCTION_NAME = "fib";
 47     private static final String LINE_MATCHER_STR = ".*" + NATIVE_FUNCTION_NAME
 48             + ".*";
 49     private static final Pattern LINE_PATTERN = Pattern
 50             .compile(LINE_MATCHER_STR);
 51     private static final String HEX_STR_PATTERN = "0x([a-fA-F0-9]+)";
 52     private static final String FIB_SPLIT_PATTERN = NATIVE_FUNCTION_NAME
 53             + "\\s+\\+";
 54     private static final Pattern HEX_PATTERN = Pattern.compile(HEX_STR_PATTERN);
 55     private static final int ADDRESS_ALIGNMENT_X86 = 4;
 56 
 57     /*
 58      * UnmappedAddressException will be thrown iff:
 59      * - The JNI code is being compiled with -fomit-frame-pointer AND
 60      * - The JNI code is currently executing at address A = pc() + offset
 61      *   where A % ADDRESS_SIZE == 0.
 62      *
 63      * In the below example we have: pc() == f6401546, offset == 56,
 64      * ADDRESS_SIZE == 4. Thus, A == F640159C which satisfies this condition.
 65      *
 66      * "NoFramePointerJNIFib" #11 prio=5 tid=0xa357bc00 nid=0x6de9 runnable [0xa365b000]
 67      *    java.lang.Thread.State: RUNNABLE
 68      *    JavaThread state: _thread_in_native
 69      * 0xf6401546 fib + 0x56
 70      */
 71     private static boolean isFibAndAlignedAddress(List<String> lines) {
 72         List<String> fibLines = findFibLines(lines);
 73         System.out.println("DEBUG: " + fibLines);
 74         // we're only interested in the first matched line.
 75         if (fibLines.size() >= 1) {
 76             String line = fibLines.get(0);
 77             return isMatchLine(line);
 78         }
 79         return false;
 80     }
 81 
 82     private static boolean isMatchLine(String line) {
 83         String[] tokens = line.split(FIB_SPLIT_PATTERN);
 84         if (tokens.length != 2) {
 85             return false; // NOT exactly two tokens, ignore.
 86         }
 87         String pcRaw = tokens[0].trim();
 88         String offsetRaw = tokens[1].trim();
 89         Matcher matcher = HEX_PATTERN.matcher(pcRaw);
 90         long pcVal = 3;
 91         boolean pcMatched = matcher.matches();
 92         if (pcMatched) {
 93             String pc = matcher.group(1);
 94             pcVal = Long.parseUnsignedLong(pc, 16);
 95         }
 96         matcher = HEX_PATTERN.matcher(offsetRaw);
 97         long offsetVal = 0;
 98         boolean offsetMatched = matcher.matches();
 99         if (offsetMatched) {
100             String offset = matcher.group(1);
101             offsetVal = Long.parseUnsignedLong(offset, 16);
102         }
103         if (offsetMatched && pcMatched
104                 && (pcVal + offsetVal) % ADDRESS_ALIGNMENT_X86 == 0) {
105             return true;
106         }
107         return false;
108     }
109 
110     private static List<String> findFibLines(List<String> lines) {
111         boolean startReached = false;
112         boolean endReached = false;
113         List<String> interestingLines = new ArrayList<>();
114         for (String line : lines) {
115             if (line.contains(LingeredAppWithNativeMethod.THREAD_NAME)) {
116                 startReached = true;
117             }
118             if (startReached && line.contains("-------")) {
119                 endReached = true;
120             }
121             if (startReached && !endReached) {
122                 Matcher matcher = LINE_PATTERN.matcher(line);
123                 if (matcher.matches()) {
124                     interestingLines.add(line);
125                 }
126             }
127         }
128         return interestingLines;
129     }
130 
131     private static void runJstackMixedInLoop(LingeredApp app) throws Exception {
132         for (int i = 0; i < MAX_ITERATIONS; i++) {
133             JDKToolLauncher launcher = JDKToolLauncher
134                     .createUsingTestJDK("jhsdb");
135             launcher.addToolArg("jstack");
136             launcher.addToolArg("--mixed");
137             launcher.addToolArg("--pid");
138             launcher.addToolArg(Long.toString(app.getPid()));
139 
140             ProcessBuilder pb = SATestUtils.createProcessBuilder(launcher);
141             Process jhsdb = pb.start();
142             OutputAnalyzer out = new OutputAnalyzer(jhsdb);
143 
144             jhsdb.waitFor();
145 
146             System.out.println(out.getStdout());
147             System.err.println(out.getStderr());
148 
149             out.shouldContain(LingeredAppWithNativeMethod.THREAD_NAME);
150             if (isFibAndAlignedAddress(out.asLines())) {
151                 System.out.println("DEBUG: Test triggered interesting condition.");
152                 out.shouldNotContain("sun.jvm.hotspot.debugger.UnmappedAddressException:");
153                 System.out.println("DEBUG: Test PASSED.");
154                 return; // If we've reached here, all is well.
155             }
156             System.out.println("DEBUG: Iteration: " + (i + 1)
157                                  + " - Test didn't trigger interesting condition.");
158             out.shouldNotContain("sun.jvm.hotspot.debugger.UnmappedAddressException:");
159         }
160         System.out.println("DEBUG: Test didn't trigger interesting condition " +
161                              "but no UnmappedAddressException was thrown. PASS!");
162     }
163 
164     public static void main(String... args) throws Exception {
165         SATestUtils.skipIfCannotAttach(); // throws SkippedException if attach not expected to work.
166         LingeredApp app = null;
167 
168         try {
169             // Needed for LingeredApp to be able to resolve native library.
170             String libPath = System.getProperty("java.library.path");
171             String[] vmArgs = (libPath != null)
172                 ? Utils.prependTestJavaOpts("-Djava.library.path=" + libPath)
173                 : Utils.getTestJavaOpts();
174 
175             app = new LingeredAppWithNativeMethod();
176             LingeredApp.startAppExactJvmOpts(app, vmArgs);
177             System.out.println("Started LingeredApp with pid " + app.getPid());
178             runJstackMixedInLoop(app);
179             System.out.println("Test Completed");
180         } catch (Throwable e) {
181             e.printStackTrace();
182             throw e;
183         } finally {
184             LingeredApp.stopApp(app);
185         }
186     }
187 }
