24 import java.util.HashMap;
25 import java.util.List;
26 import java.util.Map;
27 
28 import jdk.test.lib.apps.LingeredApp;
29 import jtreg.SkippedException;
30 
31 /**
32  * @test
33  * @bug 8190198
34  * @summary Test clhsdb pstack command
35  * @requires vm.hasSA & os.family != "mac"
36  * @library /test/lib
37  * @run main/othervm ClhsdbPstack
38  */
39 
40 public class ClhsdbPstack {
41 
42     public static void main(String[] args) throws Exception {
43         System.out.println("Starting ClhsdbPstack test");
44 
45         LingeredApp theApp = null;
46         try {
47             ClhsdbLauncher test = new ClhsdbLauncher();
48             theApp = LingeredApp.startApp();
49             System.out.println("Started LingeredApp with pid " + theApp.getPid());
50 
51             List<String> cmds = List.of("pstack -v");
52 
53             Map<String, List<String>> expStrMap = new HashMap<>();
54             expStrMap.put("pstack -v", List.of(
55                     "No deadlocks found", "Common-Cleaner",
56                     "Signal Dispatcher", "CompilerThread",
57                     "Sweeper thread", "Service Thread",
58                     "Reference Handler", "Finalizer", "main"));
59 
60             test.run(theApp.getPid(), cmds, expStrMap, null);
61         } catch (SkippedException se) {
62             throw se;
63         } catch (Exception ex) {
64             throw new RuntimeException("Test ERROR " + ex, ex);
65         } finally {
66             LingeredApp.stopApp(theApp);
67         }
68         System.out.println("Test PASSED");
69     }
70 }
