 23 package org.openjdk.bench.java.util.regex;
 24 
 25 import org.openjdk.jmh.annotations.Benchmark;
 26 import org.openjdk.jmh.annotations.BenchmarkMode;
 27 import org.openjdk.jmh.annotations.Measurement;
 28 import org.openjdk.jmh.annotations.Mode;
 29 import org.openjdk.jmh.annotations.OutputTimeUnit;
 30 import org.openjdk.jmh.annotations.Param;
 31 import org.openjdk.jmh.annotations.Scope;
 32 import org.openjdk.jmh.annotations.Setup;
 33 import org.openjdk.jmh.annotations.State;
 34 import org.openjdk.jmh.annotations.Warmup;
 35 
 36 import java.util.concurrent.TimeUnit;
 37 import java.util.regex.Pattern;
 38 import java.util.regex.PatternSyntaxException;
 39 
 40 @BenchmarkMode(Mode.AverageTime)
 41 @OutputTimeUnit(TimeUnit.NANOSECONDS)
 42 @State(Scope.Thread)
 43 public class PatternBench {
 44 
 45     public String fileTestString;
 46     public String flagsString;
 47 
 48     public Pattern graphemePattern;
 49     public Pattern graphemeBoundaryPattern;
 50     public Pattern jmodPattern;
 51     public Pattern jmodCanonicalPattern;
 52 
 53     public String charPatternRegex;
 54     public String[] charPatternStrings;
 55     public Pattern charPattern;
 56 
 57     @Setup
 58     public void setup() {
 59         flagsString = "\ud83c\udde6\ud83c\uddec\ud83c\uddec\ud83c\udde6\ud83c\uddfa\ud83c\uddf8\ud83c\uddeb\ud83c\uddf7";
 60         fileTestString = "META-INF/providers/org.openjdk.foo_hotspot_nodes_PluginFactory_EndLockScopeNode";
 61         graphemePattern = Pattern.compile("\\X");
 62         graphemeBoundaryPattern = Pattern.compile("\\b{g}");
 63 
 64         String jmodRegex = "^.*(?:(?:_the\\.[^/]*)|(?:_[^/]*\\.marker)|(?:[^/]*\\.diz)|(?:[^/]*\\.debuginfo)|(?:[^/]*\\.dSYM/.*)|(?:[^/]*\\.dSYM)|(?:[^/]*\\.pdb)|(?:[^/]*\\.map))$";
 65         jmodCanonicalPattern = Pattern.compile(jmodRegex, Pattern.CANON_EQ);
 66         jmodPattern = Pattern.compile(jmodRegex);
 67 
 68         charPatternRegex = "[ a-zA-Z]*foo[ a-zA-Z0-9]*bar[ a-z]*";
 69         charPatternStrings = new String[] {"avaaafooddddddbariiii", "lorem ipsum dolor foo bar", "fpp brr lorem ipsum dolor foo bar %", "lorem ipsum dolor foo bar lorem ipsum dolor foo bar lorem ipsum dolor foo bar /"};
 70         charPattern = Pattern.compile(charPatternRegex);
 71     }
 72 
 73     @Benchmark
 74     @Warmup(iterations = 3)
 75     @Measurement(iterations = 3)
 76     public long longStringGraphemeMatches() {
 77         return graphemePattern.matcher(flagsString.repeat(3)).results().count();
 78     }
 79 
 80     @Benchmark
 81     @Warmup(iterations = 3)
 82     @Measurement(iterations = 3)
 83     public int splitFlags() {
 84         return graphemeBoundaryPattern.split(flagsString).length;
 85     }
 86 
 87     @Benchmark
 88     @Warmup(iterations = 3)
 89     @Measurement(iterations = 3)
 90     public boolean canonicalJmodMatch() {
 91         return jmodCanonicalPattern.matcher(fileTestString).matches();
 92     }
 93 
 94     @Benchmark
 95     @Warmup(iterations = 3)
 96     @Measurement(iterations = 3)
 97     public boolean normalJmodMatch() {
 98         return jmodPattern.matcher(fileTestString).matches();
 99     }
100 
101     @Benchmark
102     @Warmup(iterations = 3)
103     @Measurement(iterations = 3)
104     public boolean charPatternMatch() {
105         return charPattern.matcher(charPatternStrings[0]).matches()
106                 && charPattern.matcher(charPatternStrings[1]).matches()
107                 && charPattern.matcher(charPatternStrings[2]).matches();
108     }
109 
110     @Benchmark
111     @Warmup(iterations = 3)
112     @Measurement(iterations = 3)
113     public boolean charPatternMatchWithCompile() {
114         Pattern p = Pattern.compile(charPatternRegex);
115         return p.matcher(charPatternStrings[0]).matches()
116                 && p.matcher(charPatternStrings[1]).matches()
117                 && p.matcher(charPatternStrings[2]).matches();
118     }
119 
120     @Benchmark
121     @Warmup(iterations = 3)
122     @Measurement(iterations = 3)
123     public Pattern charPatternCompile() {
124         return Pattern.compile(charPatternRegex);
125     }
126 }
