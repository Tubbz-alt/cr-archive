 24 /*
 25  * @test
 26  * @bug 8241625 8241895
 27  * @summary test the lists generated by the doclet
 28  * @library  /tools/lib ../../lib
 29  * @modules  jdk.javadoc/jdk.javadoc.internal.tool
 30  * @build    toolbox.ToolBox javadoc.tester.*
 31  * @run main TestLists
 32  */
 33 
 34 import java.io.IOException;
 35 import java.nio.file.Path;
 36 
 37 import javadoc.tester.JavadocTester;
 38 import toolbox.ToolBox;
 39 
 40 public class TestLists extends JavadocTester {
 41 
 42     public static void main(String... args) throws Exception {
 43         TestLists tester = new TestLists();
 44         tester.runTests(m -> new Object[]{Path.of(m.getName())});
 45     }
 46 
 47     private final ToolBox tb = new ToolBox();
 48 
 49     @Test
 50     public void testMemberLists(Path base) throws IOException {
 51         Path src = base.resolve("src");
 52         tb.writeJavaFiles(src,
 53                 "package p; public class C {\n"
 54                 + "  public C() { }\n"
 55                 + "  public C(int i) { }\n"
 56                 + "  public int f1;\n"
 57                 + "  public int f2;\n"
 58                 + "  public void m1() { }\n"
 59                 + "  public void m2() { }\n"
 60                 + "}\n",
 61                 "package p; public enum E { E1, E2 }\n",
 62                 "package p; public @interface A { int value(); }\n"
 63         );
 64 
 65         javadoc("-d", base.resolve("out").toString(),
 66                 "-sourcepath", src.toString(),
 67                 "p");
 68         checkExit(Exit.OK);
 69 
 70         checkOutput("p/C.html", true,
 71                 "<h2>Field Details</h2>\n"
 72                 + "<ul class=\"member-list\">\n"
 73                 + "<li>\n"
 74                 + "<section class=\"detail\" id=\"f1\">\n"
 75                 + "<h3>f1</h3>\n",
 76                 "<h2>Constructor Details</h2>\n"
 77                 + "<ul class=\"member-list\">\n"
 78                 + "<li>\n"
 79                 + "<section class=\"detail\" id=\"&lt;init&gt;()\">\n"
 80                 + "<h3>C</h3>",
 81                 "<section class=\"method-details\" id=\"method.detail\">\n"
 82                 + "<h2>Method Details</h2>\n"
 83                 + "<ul class=\"member-list\">\n"
 84                 + "<li>\n"
 85                 + "<section class=\"detail\" id=\"m1()\">\n"
 86                 + "<h3>m1</h3>\n");
 87 
 88         checkOutput("p/E.html", true,
 89                 "<h2>Enum Constant Details</h2>\n"
 90                 + "<ul class=\"member-list\">\n"
 91                 + "<li>\n"
 92                 + "<section class=\"detail\" id=\"E1\">\n");
 93 
 94         checkOutput("p/A.html", true,
 95                 "<h2>Element Details</h2>\n"
 96                 + "<ul class=\"member-list\">\n"
 97                 + "<li>\n"
 98                 + "<section class=\"detail\" id=\"value()\">");
 99     }
100 
101     @Test
102     public void testDetailLists(Path base) throws IOException {
103         Path src = base.resolve("src");
104         tb.writeJavaFiles(src,
105                 "package p; public class C {\n"
106                         + "  public C() { }\n"
107                         + "  public C(int i) { }\n"
108                         + "  public int f1;\n"
109                         + "  public int f2;\n"
110                         + "  public void m1() { }\n"
111                         + "  public void m2() { }\n"
112                         + "}\n",
113                 "package p; public enum E { E1, E2 }\n",
114                 "package p; public @interface A { int value(); }\n"
115         );
116 
117         javadoc("-d", base.resolve("out").toString(),
118                 "-sourcepath", src.toString(),
119                 "p");
120         checkExit(Exit.OK);
121 
122         checkOutput("p/C.html", true,
123                 "<section class=\"details\">\n"
124                 + "<ul class=\"details-list\">\n"
125                 + "<!-- ============ FIELD DETAIL =========== -->\n"
126                 + "<li>\n"
127                 + "<section class=\"field-details\" id=\"field.detail\">\n"
128                 + "<h2>Field Details</h2>\n"
129                 + "<ul class=\"member-list\">",
130                 "</ul>\n"
131                 + "</section>\n"
132                 + "</li>\n"
133                 + "<!-- ========= CONSTRUCTOR DETAIL ======== -->\n"
134                 + "<li>\n"
135                 + "<section class=\"constructor-details\" id=\"constructor.detail\">\n"
136                 + "<h2>Constructor Details</h2>\n"
137                 + "<ul class=\"member-list\">");
138 
139         checkOutput("p/E.html", true,
140                 "<section class=\"details\">\n"
141                 + "<ul class=\"details-list\">\n"
142                 + "<!-- ============ ENUM CONSTANT DETAIL =========== -->\n"
143                 + "<li>\n"
144                 + "<section class=\"constant-details\" id=\"enum.constant.detail\">\n"
145                 + "<h2>Enum Constant Details</h2>\n"
146                 + "<ul class=\"member-list\">");
147 
148         checkOutput("p/A.html", true,
149                 "<ul class=\"details-list\">\n"
150                 + "<!-- ============ ANNOTATION TYPE MEMBER DETAIL =========== -->\n"
151                 + "<li>\n"
152                 + "<section class=\"member-details\">\n"
153                 + "<h2>Element Details</h2>\n"
154                 + "<ul class=\"member-list\">");
155     }
