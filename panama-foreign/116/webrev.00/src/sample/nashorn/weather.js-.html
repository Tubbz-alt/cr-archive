41

46 
47 // pull out humidity as array
48 var humidity = weather.list.map(function(curVal) {
49     return curVal.humidity;
50 })
51 
52 // Stream API to print stat
53 print("Humidity");
54 print(Arrays["stream(int[])"](humidity).summaryStatistics());
55 
56 // pull maximum day time temperature
57 var temp = weather.list.map(function(curVal) {
58     return curVal.temp.max;
59 });
60 
61 // Stream API to print stat
62 print("Max Temperature");
63 print(Arrays["stream(double[])"](temp).summaryStatistics());
