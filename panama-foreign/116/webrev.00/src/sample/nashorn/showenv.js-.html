39 // This script displays environment entries as a HTML table.
40 // Demonstrates heredoc to generate HTML content and display
41 // using JavaFX WebView.
42 
43 // JavaFX classes used
44 var Scene     = Java.type("javafx.scene.Scene");
45 var WebView   = Java.type("javafx.scene.web.WebView");
46 
47 // JavaFX start method
48 function start(stage) {
49     stage.title = "Your Environment";
50     var wv = new WebView();
51     var envrows = "";
52     for (var i in $ENV) {
53         envrows += <<TBL
54 <tr>
55 <td>
56 ${i}
57 </td>
58 <td>
59 ${$ENV[i]}
60 </td>
61 </tr>
62 TBL
63     }
64 
65     wv.engine.loadContent(<<EOF
66 <html>
67 <head>
68 <title>
69 Your Environment
70 </title>
71 </head>
72 <body>
73 <h1>Your Environment</h1>
74 <table border="1">
75 ${envrows}
76 </table>
77 </body>
78 </html>
79 EOF, "text/html");
80     stage.scene = new Scene(wv, 750, 500);
81     stage.show();
82 }
