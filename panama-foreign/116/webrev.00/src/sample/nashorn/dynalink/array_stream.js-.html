34 // This script depends on array stream dynalink linker
35 // to work as expected. Without that linker in jjs classpath,
36 // this script will fail to run.
37 
38 // Object[] and then Stream
39 var s = Java.to(["hello", "world"]).stream
40 s.map(function(s) s.toUpperCase()).forEach(print)
41 
42 // IntStream
43 var is = Java.to([3, 56, 4, 23], "int[]").stream
44 print(is.map(function(x) x*x).sum())
45 
46 // DoubleStream
47 var DoubleArray = Java.type("double[]")
48 var arr = new DoubleArray(100)
49 for (var i = 0; i < arr.length; i++)
50     arr[i] = Math.random()
51 
52 print(arr.stream.summaryStatistics())
