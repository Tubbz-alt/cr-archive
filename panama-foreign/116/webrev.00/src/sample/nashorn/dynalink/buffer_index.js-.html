34 // This script depends on buffer indexing dynalink linker. Without that
35 // linker in classpath, this script will fail to run properly.
36 
37 function str(buf) {
38     var s = ''
39     for (var i = 0; i < buf.length; i++)
40         s += buf[i] + ","
41     return s
42 }
43 
44 var ByteBuffer = Java.type("java.nio.ByteBuffer")
45 var bb = ByteBuffer.allocate(10)
46 for (var i = 0; i < bb.length; i++)
47     bb[i] = i*i
48 print(str(bb))
49 
50 var CharBuffer = Java.type("java.nio.CharBuffer")
51 var cb = CharBuffer.wrap("hello world")
52 print(str(cb))
53 
54 var RandomAccessFile = Java.type("java.io.RandomAccessFile")
55 var raf = new RandomAccessFile("buffer_index.js", "r")
56 var chan = raf.getChannel()
57 var fileSize = chan.size()
58 var buf = ByteBuffer.allocate(fileSize)
59 chan.read(buf)
60 chan.close()
61 
62 var str = ''
63 for (var i = 0; i < buf.length; i++)
64     str += String.fromCharCode(buf[i])
65 print(str)
