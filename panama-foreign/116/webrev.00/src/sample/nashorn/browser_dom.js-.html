39 // JavaFX classes used
40 var ChangeListener = Java.type("javafx.beans.value.ChangeListener");
41 var Scene     = Java.type("javafx.scene.Scene");
42 var WebView   = Java.type("javafx.scene.web.WebView");
43 
44 // JavaFX start method
45 function start(stage) {
46     stage.title = "Web View";
47     var wv = new WebView();
48     wv.engine.loadContent(<<EOF
49 <html>
50 <head>
51 <title>
52 This is the title
53 </title>
54 <script>
55 // click count for OK button
56 var okCount = 0;
57 </script>
58 </head>
59 <body>
60 Button from the input html<br>
61 <button type="button" onclick="okCount++">OK</button><br>
62 </body>
63 </html>
64 EOF, "text/html");
65 
66     // attach onload handler
67     wv.engine.loadWorker.stateProperty().addListener(
68         new ChangeListener() {
69             changed: function() {
70                // DOM document element
71                var document = wv.engine.document;
72                // DOM manipulation
73                var btn = document.createElement("button");
74                var n = 0;
75                // attach a button handler - nashorn function!
76                btn.onclick = function() {
77                    n++; print("You clicked " + n + " time(s)");
78                    print("you clicked OK " + wv.engine.executeScript("okCount"));
79                };
80                // attach text to button
81                var t = document.createTextNode("Click Me!"); 
82                btn.appendChild(t);
83                // attach button to the document
84                document.body.appendChild(btn); 
85            }
86         }
87     );
88     stage.scene = new Scene(wv, 750, 500);
89     stage.show();
90 }
