39 // This script displays System properties as a HTML table.
40 // Demonstrates heredoc to generate HTML content and display
41 // using JavaFX WebView.
42 
43 // JavaFX, Java classes used
44 var Scene     = Java.type("javafx.scene.Scene");
45 var System    = Java.type("java.lang.System");
46 var WebView   = Java.type("javafx.scene.web.WebView");
47 
48 // JavaFX start method
49 function start(stage) {
50     stage.title = "Your System Properties";
51     var wv = new WebView();
52     var sysproprows = "";
53     var sysprops = System.properties;
54     for (var i in sysprops) {
55         sysproprows += <<TBL
56 <tr>
57 <td>
58 ${i}
59 </td>
60 <td>
61 ${sysprops[i]}
62 </td>
63 </tr>
64 TBL
65     }
66 
67     wv.engine.loadContent(<<EOF
68 <html>
69 <head>
70 <title>
71 Your System Properties
72 </title>
73 </head>
74 <body>
75 <h1>Your System Properties</h1>
76 <table border="1">
77 ${sysproprows}
78 </table>
79 </body>
80 </html>
81 EOF, "text/html");
82     stage.scene = new Scene(wv, 750, 500);
83     stage.show();
84 }
