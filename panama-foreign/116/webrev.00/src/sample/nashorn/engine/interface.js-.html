34 // Example demonstrating how to implement a Java interface
35 // whose methods are backed by global script functions
36 
37 var ScriptEngineManager = Java.type("javax.script.ScriptEngineManager");
38 // create manager
39 var manager = new ScriptEngineManager();
40 // create engine
41 var engine = manager.getEngineByName("js");
42 
43 // eval code - too many script escapes?
44 // use heredoc !
45 engine.eval(<<CODE
46 function run() {
47     print("run global function called");
48 }
49 CODE);
50 
51 // create Java interface object whose methods are
52 // implemented by script functions. This is from
53 // javax.script.Invocable. But we are in JS world,
54 // don't worry about types :)
55 
56 var Runnable = Java.type("java.lang.Runnable");
57 var r = engine.getInterface(Runnable.class);
58 print(r.class);
59 
60 r.run();
