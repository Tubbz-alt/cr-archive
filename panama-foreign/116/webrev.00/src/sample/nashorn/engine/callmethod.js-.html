34 // simple example demonstrating calling a script object
35 // method from script engine user code
36 
37 var ScriptEngineManager = Java.type("javax.script.ScriptEngineManager");
38 // create manager
39 var manager = new ScriptEngineManager();
40 // create engine
41 var engine = manager.getEngineByName("js");
42 
43 // eval code - too many script escapes?
44 // use heredoc !
45 engine.eval(<<CODE
46     var obj = {
47         func: function() {
48             print("I am func of " + this);
49         },
50 
51         toString: function() {
52             return "Object 'obj'";
53         }
54    };
55 CODE);
56 
57 // invoke methods of an object in script world
58 // from javax.script.Invocable interface. But, hey,
59 // calling code is JavaScript and don't worry about types :)
60 
61 // get that script object on which to call a method
62 var scriptObj = engine.get("obj");
63 // call 'func' method on object 'obj'
64 engine.invokeMethod(scriptObj, "func");
