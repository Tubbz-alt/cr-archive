34 if (! $OPTIONS._fx) {
35     print("Usage: jjs -fx datepick.js");
36     exit(1);
37 }

39 load("fx:controls.js");
40 var Scene = Java.type("javafx.scene.Scene");
41 
42 function start(stage) {
43     var picker = new DatePicker();
44     picker.onAction = function(evt) {
45         var ld = Object.bindProperties({}, picker.value);
46         print(JSON.stringify(ld));
47     }
48 
49     var scene = new Scene(picker, 100, 30);
50     stage.scene = scene;
51     stage.show();
52 }
