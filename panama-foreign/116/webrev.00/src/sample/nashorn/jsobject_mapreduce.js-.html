34 // Many Array.prototype functions such as map,
35 // filter, reduce, reduceRight, every, some are generic.
36 // These functions accept ECMAScript array as well as
37 // many array-like objects including JSObjects.
38 // See also http://en.wikipedia.org/wiki/MapReduce
39 
40 `javac BufferArray.java`;
41 
42 var BufferArray = Java.type("BufferArray");
43 var buf = new BufferArray(10);
44 
45 var map = Array.prototype.map;
46 var filter = Array.prototype.filter;
47 var reduce = Array.prototype.reduce;
48 
49 // make random list of numbers
50 for (var i = 0; i < 10; i++)
51     buf[i] = Math.random();
52 
53 var forEach = Array.prototype.forEach;
54 // print numbers in the list
55 forEach.call(buf, function(x) print(x));
56 
57 // print sum of squares of the random numbers
58 print("Square sum:",
59     reduce.call(
60         map.call(buf, function(x) x*x),
61         function(x, y) x + y)
62 );
