33 
34 // Yet another FizzBuzz impl. using Java 8 lambda and stream
35 // but using Nashorn. This is ECMAScript port of @stuartmarks'
36 // Java implementation
37 
38 var IntStream = Java.type("java.util.stream.IntStream");
39 
40 function ifmod(m, r, f) {
41     return function(i) { return i % m == 0? r : f(i); }
42 }
43 
44 // pass script function for lambda
45 IntStream.rangeClosed(1, 100).
46    mapToObj(
47       ifmod(15, "FizzBuzz", ifmod(5, "Buzz", ifmod(3, "Fizz", String))))
48       .forEach(print);
