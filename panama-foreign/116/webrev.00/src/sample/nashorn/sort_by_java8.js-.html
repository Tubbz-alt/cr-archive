32 // Simple sorting with Java8 APIs
33 
34 // Separation key-extraction from key ordering
35 
36 // Simple demo for Comparator.comparing
37 // http://docs.oracle.com/javase/8/docs/api/java/util/Comparator.html#comparing-java.util.function.Function-
38 
39 // data to be sorted
40 var cards = [
41    { name: "hello", email: "foo@hello.com" },
42    { name: "world", email: "bar@world.com" },
43    { name: "see", email: "see@dontsee.com" },
44 ];
45 
46 var Collections = java.util.Collections;
47 var Comparator = java.util.Comparator;
48 
49 // sort records name in reverse order of names
50 Collections.sort(cards,
51   Comparator.comparing(function(a) a.name).reversed());
52 
53 print(JSON.stringify(cards))
54 
55 // sort records by email
56 Collections.sort(cards,
57   Comparator.comparing(function(a) a.email));
58 
59 print(JSON.stringify(cards))
