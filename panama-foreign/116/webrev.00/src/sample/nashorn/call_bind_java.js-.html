32 // Sample demonstrating calling Java methods like normal
33 // functions. Bound instance methods, static methods can be
34 // called like normal script functions.
35 
36 // Java types used
37 var Files = Java.type("java.nio.file.Files");
38 var FileSystems = Java.type("java.nio.file.FileSystems");
39 var System = Java.type("java.lang.System");
40 
41 var fs = FileSystems.default;
42 var bind = Function.prototype.bind;
43 
44 // Java methods as "functions"
45 
46 // Java method with bound "this"
47 var Path = bind.call(fs.getPath, fs);
48 // Java static method
49 var Property = System.getProperty;
50 
51 // call Java static methods and bound instance methods
52 // like normal script functions.
53 
54 var root = Path("/");
55 // print URI for root dir
56 print(root.toUri());
57 var home = Path(Property("user.home"));
58 // list home directory
59 Files.walk(home).forEach(print);
