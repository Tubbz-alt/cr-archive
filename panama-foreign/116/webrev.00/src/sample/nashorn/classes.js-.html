34 // print all Java classes of the given package and its subpackages.
35 var pkg = arguments.length > 0? arguments[0] : "java.lang";
36 
37 with (new JavaImporter(javax.tools, java.util)) {
38     var compiler = ToolProvider.systemJavaCompiler;
39     var fm = compiler.getStandardFileManager(null, null, null);
40     var kinds = EnumSet.of(JavaFileObject.Kind.CLASS);
41     var loc = StandardLocation.PLATFORM_CLASS_PATH;
42     var itr = fm.list(loc, pkg, kinds, true).iterator();
43     while(itr.hasNext()) {
44         print(fm.inferBinaryName(loc, itr.next()));
45     }
46     fm.close();
47 }
