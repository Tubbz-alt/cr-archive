34 // The $EXEC builtin function can be used to run external commands:
35 $EXEC("ls")
36 $EXEC("ls -la")
37 
38 // It can also be given a string to use as stdin:
39 $EXEC("cat", "Hello, world!")
40 
41 // Arguments can be passed as an array of strings
42 $EXEC(["ls","-l","-a"]);
43 
44 // Output of running external commands is returned from $EXEC:
45 print($EXEC("ls"))
46 
47 // apply on $EXEC
48 print($EXEC.apply(this, ["ls"]));
