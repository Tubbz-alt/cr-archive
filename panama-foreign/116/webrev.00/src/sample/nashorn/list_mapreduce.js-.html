32 // Usage: jjs list_mapreduce.js
33 
34 // Many Array.prototype functions such as map,
35 // filter, reduce, reduceRight, every, some are generic.
36 // These functions accept ECMAScript array as well as
37 // many array-like objects including java.util.ArrayLists.
38 // So, you can do map/filter/reduce with Java streams or
39 // you can also use Array.prototype functions as below.
40 // See also http://en.wikipedia.org/wiki/MapReduce
41 
42 var ArrayList = Java.type("java.util.ArrayList");
43 var list = new ArrayList();
44 list.add("nashorn");
45 list.add("ecmascript");
46 list.add("javascript");
47 list.add("js");
48 list.add("scheme");
49 
50 var map = Array.prototype.map;
51 var filter = Array.prototype.filter;
52 var reduce = Array.prototype.reduce;
53 
54 // sum of word lengths
55 print("Sum word length:",
56     reduce.call(
57         map.call(list, function(x) x.length),
58         function(x, y) x + y)
59 );
60 
61 // filter use to filter out "j*" and concatenate rest with ":"
62 // after uppercasing all strings
63 print(
64     reduce.call(
65         map.call(
66             filter.call(list, function(x) !x.startsWith("j")),
67             function(x) x.toUpperCase()),
68         function(x, y) x + ":" + y)
69 );
70 
71 // another list example involving numbers
72 list.clear();
73 // make random list of numbers
74 for (var i = 0; i < 10; i++)
75     list.add(Math.random());
76 
77 var forEach = Array.prototype.forEach;
78 // print numbers in the list
79 forEach.call(list, function(x) print(x));
80 
81 // print sum of squares of the random numbers
82 print("Square sum:",
83     reduce.call(
84         map.call(list, function(x) x*x),
85         function(x, y) x + y)
86 );
