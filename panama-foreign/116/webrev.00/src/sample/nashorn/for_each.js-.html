32 // nashorn supports for..each extension supported
33 // by Mozilla. See also
34 // https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Statements/for_each...in
35 
36 var strs = [ "hello", "world" ];
37 for each (str in strs)
38     print(str);
39 
40 // create a java int[] object
41 var JArray = Java.type("int[]");
42 var arr = new JArray(10);
43 
44 // store squares as values
45 for (i in arr)
46     arr[i] = i*i;
47 
48 // for .. each on java arrays
49 print("squares");
50 for each (i in arr)
51     print(i);
52 
53 var System = Java.type("java.lang.System");
54 
55 // for..each on java Iterables
56 // print System properties as name = value pairs
57 print("System properties");
58 for each (p in System.properties.entrySet()) {
59     print(p.key, "=", p.value);
60 }
61 
62 // print process environment vars as name = value pairs
63 print("Process environment");
64 for each (e in System.env.entrySet()) {
65     print(e.key, "=", e.value);
66 }
