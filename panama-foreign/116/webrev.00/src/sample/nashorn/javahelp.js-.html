32 // script helpers to print meta info on Java instances and classes
33 
34 // print instance methods info on a Java object or static methods info of a Java class
35 function methods(jobj) {
36     if (! Java.isJavaObject(jobj)) {
37         throw new TypeError("not a Java object");
38     }
39 
40     var isStatic = Java.isType(jobj);
41     var obj = Object.bindProperties({}, jobj);
42     for each (var i in obj) {
43         if (Java.isJavaMethod(i)) {
44             var str = String(i);
45             var idx = str.indexOf(' ');
46             var overloaded = str.substring(0, idx).endsWith("OverloadedDynamicMethod");
47             var lastIdx = isStatic? str.lastIndexOf('] on') : str.lastIndexOf(']');
48             print(str.substring(idx + 1, lastIdx) + (overloaded? "*" : ""))
49         }
50     }
51 }
52 
53 // print instance field names of a Java object or static field names of a Java class
54 function fields(jobj) {
55     if (! Java.isJavaObject(jobj)) {
56         throw new TypeError("not a Java object");
57     }
58 
59     var obj = Object.bindProperties({}, jobj);
60     for (var i in obj) {
61         if (! Java.isJavaMethod(obj[i])) {
62             print(i);
63         }
64     }
65 }
