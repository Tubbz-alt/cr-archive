32 import javax.script.*;
33 import jdk.nashorn.api.scripting.*;
34 
35 // Simple nashorn demo that evals a script with arbitrary script
36 // object bound as "this" for the evaluated script.
37 
38 public class EvalWithArbitraryThis {
39     public static void main(String[] args) throws Exception {
40         ScriptEngineManager m = new ScriptEngineManager();
41         ScriptEngine e = m.getEngineByName("nashorn");
42         Object sobj = e.eval("( { foo: 343, bar: 'hello' } )");
43 
44         // "this" bound to sobj in this eval.
45         // so it prints sobj.foo and sobj.bar.
46         ((ScriptObjectMirror)sobj).eval("print(this.foo); print(this.bar)");
47     }
48 }
