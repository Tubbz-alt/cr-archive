 60 /**
 61  * WinMsiBundler
 62  *
 63  * Produces .msi installer from application image. Uses WiX Toolkit to build
 64  * .msi installer.
 65  * <p>
 66  * {@link #execute} method creates a number of source files with the description
 67  * of installer to be processed by WiX tools. Generated source files are stored
 68  * in "config" subdirectory next to "app" subdirectory in the root work
 69  * directory. The following WiX source files are generated:
 70  * <ul>
 71  * <li>main.wxs. Main source file with the installer description
 72  * <li>bundle.wxf. Source file with application and Java run-time directory tree
 73  * description.
 74  * </ul>
 75  * <p>
 76  * main.wxs file is a copy of main.wxs resource from
 77  * jdk.incubator.jpackage.internal.resources package. It is parametrized with the
 78  * following WiX variables:
 79  * <ul>
 80  * <li>JpAppName. Name of the application. Set to the value of --name command
 81  * line option
 82  * <li>JpAppVersion. Version of the application. Set to the value of
 83  * --app-version command line option
 84  * <li>JpAppVendor. Vendor of the application. Set to the value of --vendor
 85  * command line option
 86  * <li>JpAppDescription. Description of the application. Set to the value of
 87  * --description command line option
 88  * <li>JpProductCode. Set to product code UUID of the application. Random value
 89  * generated by jpackage every time {@link #execute} method is called
 90  * <li>JpProductUpgradeCode. Set to upgrade code UUID of the application. Random
 91  * value generated by jpackage every time {@link #execute} method is called if
 92  * --win-upgrade-uuid command line option is not specified. Otherwise this
 93  * variable is set to the value of --win-upgrade-uuid command line option
 94  * <li>JpAllowDowngrades. Set to "yes" if --win-upgrade-uuid command line option
 95  * was specified. Undefined otherwise
 96  * <li>JpLicenseRtf. Set to the value of --license-file command line option.
 97  * Undefined is --license-file command line option was not specified
 98  * <li>JpInstallDirChooser. Set to "yes" if --win-dir-chooser command line
 99  * option was specified. Undefined otherwise
100  * <li>JpConfigDir. Absolute path to the directory with generated WiX source
101  * files.
102  * <li>JpIsSystemWide. Set to "yes" if --win-per-user-install command line
103  * option was not specified. Undefined otherwise
104  * </ul>
105  */

204     private static UUID getUpgradeCode(Map<String, ? super Object> params) {
205         String upgradeCode = UPGRADE_UUID.fetchFrom(params);
206         if (upgradeCode != null) {
207             return UUID.fromString(upgradeCode);
208         }
209         return createNameUUID("UpgradeCode", params, List.of(VENDOR, APP_NAME));
210     }

212     private static UUID getProductCode(Map<String, ? super Object> params) {
213         return createNameUUID("ProductCode", params, List.of(VENDOR, APP_NAME,
214                 VERSION));

217     private static UUID createNameUUID(String prefix,
218             Map<String, ? super Object> params,
219             List<StandardBundlerParam<String>> components) {
220         String key = Stream.concat(Stream.of(prefix), components.stream().map(
221                 c -> c.fetchFrom(params))).collect(Collectors.joining("/"));
222         return UUID.nameUUIDFromBytes(key.getBytes(StandardCharsets.UTF_8));
223     }

225     @Override
226     public boolean validate(Map<String, ? super Object> params)
227             throws ConfigException {
228         try {
229             if (wixToolset == null) {
230                 wixToolset = WixTool.toolset();
231             }
232 
233             try {
234                 getUpgradeCode(params);
235             } catch (IllegalArgumentException ex) {
236                 throw new ConfigException(ex);
237             }
238 
239             for (var toolInfo: wixToolset.values()) {
240                 Log.verbose(MessageFormat.format(I18N.getString(
241                         "message.tool-version"), toolInfo.path.getFileName(),
242                         toolInfo.version));
243             }
244 
245             wixSourcesBuilder.setWixVersion(wixToolset.get(WixTool.Light).version);
246 
247             wixSourcesBuilder.logWixFeatures();
248 
249             /********* validate bundle parameters *************/
250 
251             String version = PRODUCT_VERSION.fetchFrom(params);
252             if (!isVersionStringValid(version)) {
253                 throw new ConfigException(
254                         MessageFormat.format(I18N.getString(
255                                 "error.version-string-wrong-format"), version),
256                         MessageFormat.format(I18N.getString(
257                                 "error.version-string-wrong-format.advice"),
258                                 PRODUCT_VERSION.getID()));
259             }
260 
261             // only one mime type per association, at least one file extension
262             List<Map<String, ? super Object>> associations =
263                     FILE_ASSOCIATIONS.fetchFrom(params);
264             if (associations != null) {
265                 for (int i = 0; i < associations.size(); i++) {
266                     Map<String, ? super Object> assoc = associations.get(i);
267                     List<String> mimes = FA_CONTENT_TYPE.fetchFrom(assoc);
268                     if (mimes.size() > 1) {
269                         throw new ConfigException(MessageFormat.format(
270                                 I18N.getString("error.too-many-content-types-for-file-association"), i),
271                                 I18N.getString("error.too-many-content-types-for-file-association.advice"));
272                     }
273                 }
274             }
275 
276             return true;
277         } catch (RuntimeException re) {
278             if (re.getCause() instanceof ConfigException) {
279                 throw (ConfigException) re.getCause();
280             } else {
281                 throw new ConfigException(re);
282             }
283         }
284     }

286     // https://msdn.microsoft.com/en-us/library/aa370859%28v=VS.85%29.aspx
287     // The format of the string is as follows:
288     //     major.minor.build
289     // The first field is the major version and has a maximum value of 255.
290     // The second field is the minor version and has a maximum value of 255.
291     // The third field is called the build version or the update version and
292     // has a maximum value of 65,535.
293     static boolean isVersionStringValid(String v) {
294         if (v == null) {
295             return true;
296         }
297 
298         String p[] = v.split("\\.");
299         if (p.length > 3) {
300             Log.verbose(I18N.getString(
301                     "message.version-string-too-many-components"));
302             return false;
303         }
304 
305         try {
306             int val = Integer.parseInt(p[0]);
307             if (val < 0 || val > 255) {
308                 Log.verbose(I18N.getString(
309                         "error.version-string-major-out-of-range"));
310                 return false;
311             }
312             if (p.length > 1) {
313                 val = Integer.parseInt(p[1]);
314                 if (val < 0 || val > 255) {
315                     Log.verbose(I18N.getString(
316                             "error.version-string-minor-out-of-range"));
317                     return false;
318                 }
319             }
320             if (p.length > 2) {
321                 val = Integer.parseInt(p[2]);
322                 if (val < 0 || val > 65535) {
323                     Log.verbose(I18N.getString(
324                             "error.version-string-build-out-of-range"));
325                     return false;
326                 }
327             }
328         } catch (NumberFormatException ne) {
329             Log.verbose(I18N.getString("error.version-string-part-not-number"));
330             Log.verbose(ne);
331             return false;
332         }
333 
334         return true;
335     }

337     private void prepareProto(Map<String, ? super Object> params)
338                 throws PackagerException, IOException {
339         File appImage = StandardBundlerParam.getPredefinedAppImage(params);
340         File appDir = null;
341 
342         // we either have an application image or need to build one
343         if (appImage != null) {
344             appDir = new File(MSI_IMAGE_DIR.fetchFrom(params),
345                     APP_NAME.fetchFrom(params));
346             // copy everything from appImage dir into appDir/name
347             IOUtils.copyRecursive(appImage.toPath(), appDir.toPath());
348         } else {
349             appDir = APP_BUNDLER.fetchFrom(params).doBundle(params,
350                     MSI_IMAGE_DIR.fetchFrom(params), true);
351         }
352 
353         // Configure installer icon
354         if (StandardBundlerParam.isRuntimeInstaller(params)) {
355             // Use icon from java launcher.
356             // Assume java.exe exists in Java Runtime being packed.
357             // Ignore custom icon if any as we don't want to copy anything in
358             // Java Runtime image.
359             installerIcon = ApplicationLayout.javaRuntime()
360                     .runtimeDirectory()
361                     .resolve(Path.of("bin", "java.exe"));
362         } else {
363             installerIcon = ApplicationLayout.windowsAppImage()
364                     .launchersDirectory()
365                     .resolve(APP_NAME.fetchFrom(params) + ".exe");
366         }
367 
368         params.put(WIN_APP_IMAGE.getID(), appDir);
369 
370         String licenseFile = LICENSE_FILE.fetchFrom(params);
371         if (licenseFile != null) {
372             // need to copy license file to the working directory
373             // and convert to rtf if needed
374             File lfile = new File(licenseFile);
375             File destFile = new File(CONFIG_ROOT.fetchFrom(params),
376                     lfile.getName());
377 
378             IOUtils.copyFile(lfile, destFile);
379             destFile.setWritable(true);
380             ensureByMutationFileIsRTF(destFile);
381         }
382     }

384     public File bundle(Map<String, ? super Object> params, File outdir)
385             throws PackagerException {
386 
387         IOUtils.writableOutputDir(outdir.toPath());
388 
389         Path imageDir = MSI_IMAGE_DIR.fetchFrom(params).toPath();
390         try {
391             Files.createDirectories(imageDir);
392 
393             prepareProto(params);
394 
395             wixSourcesBuilder
396             .initFromParams(WIN_APP_IMAGE.fetchFrom(params).toPath(), params)
397             .createMainFragment(CONFIG_ROOT.fetchFrom(params).toPath().resolve(
398                     "bundle.wxf"));
399 
400             Map<String, String> wixVars = prepareMainProjectFile(params);
401 
402             new ScriptRunner()
403             .setDirectory(imageDir)
404             .setResourceCategoryId("resource.post-app-image-script")
405             .setScriptNameSuffix("post-image")
406             .setEnvironmentVariable("JpAppImageDir", imageDir.toAbsolutePath().toString())
407             .run(params);
408 
409             return buildMSI(params, wixVars, outdir);
410         } catch (IOException ex) {
411             Log.verbose(ex);
412             throw new PackagerException(ex);
413         }
414     }

416     Map<String, String> prepareMainProjectFile(
417             Map<String, ? super Object> params) throws IOException {
418         Map<String, String> data = new HashMap<>();
419 
420         final UUID productCode = getProductCode(params);
421         final UUID upgradeCode = getUpgradeCode(params);
422 
423         data.put("JpProductCode", productCode.toString());
424         data.put("JpProductUpgradeCode", upgradeCode.toString());
425 
426         Log.verbose(MessageFormat.format(I18N.getString("message.product-code"),
427                 productCode));
428         Log.verbose(MessageFormat.format(I18N.getString("message.upgrade-code"),
429                 upgradeCode));
430 
431         data.put("JpAllowUpgrades", "yes");
432         data.put("JpAllowDowngrades", "yes");
433 
434         data.put("JpAppName", APP_NAME.fetchFrom(params));
435         data.put("JpAppDescription", DESCRIPTION.fetchFrom(params));
436         data.put("JpAppVendor", VENDOR.fetchFrom(params));
437         data.put("JpAppVersion", PRODUCT_VERSION.fetchFrom(params));
438         data.put("JpIcon", installerIcon.toString());
439 
440         final Path configDir = CONFIG_ROOT.fetchFrom(params).toPath();
441 
442         data.put("JpConfigDir", configDir.toAbsolutePath().toString());
443 
444         if (MSI_SYSTEM_WIDE.fetchFrom(params)) {
445             data.put("JpIsSystemWide", "yes");
446         }
447 
448         String licenseFile = LICENSE_FILE.fetchFrom(params);
449         if (licenseFile != null) {
450             String lname = new File(licenseFile).getName();
451             File destFile = new File(CONFIG_ROOT.fetchFrom(params), lname);
452             data.put("JpLicenseRtf", destFile.getAbsolutePath());
453         }
454 
455         // Copy CA dll to include with installer
456         if (INSTALLDIR_CHOOSER.fetchFrom(params)) {
457             data.put("JpInstallDirChooser", "yes");
458             String fname = "wixhelper.dll";
459             try (InputStream is = OverridableResource.readDefault(fname)) {
460                 Files.copy(is, Paths.get(
461                         CONFIG_ROOT.fetchFrom(params).getAbsolutePath(),
462                         fname));
463             }
464         }
465 
466         // Copy l10n files.
467         for (String loc : Arrays.asList("en", "ja", "zh_CN")) {
468             String fname = "MsiInstallerStrings_" + loc + ".wxl";
469             try (InputStream is = OverridableResource.readDefault(fname)) {
470                 Files.copy(is, Paths.get(
471                         CONFIG_ROOT.fetchFrom(params).getAbsolutePath(),
472                         fname));
473             }
474         }
475 
476         createResource("main.wxs", params)
477                 .setCategory(I18N.getString("resource.main-wix-file"))
478                 .saveToFile(configDir.resolve("main.wxs"));
479 
480         createResource("overrides.wxi", params)
481                 .setCategory(I18N.getString("resource.overrides-wix-file"))
482                 .saveToFile(configDir.resolve("overrides.wxi"));
483 
484         return data;
485     }

487     private File buildMSI(Map<String, ? super Object> params,
488             Map<String, String> wixVars, File outdir)
489             throws IOException {
490 
491         File msiOut = new File(
492                 outdir, INSTALLER_FILE_NAME.fetchFrom(params) + ".msi");
493 
494         Log.verbose(MessageFormat.format(I18N.getString(
495                 "message.preparing-msi-config"), msiOut.getAbsolutePath()));
496 
497         WixPipeline wixPipeline = new WixPipeline()
498         .setToolset(wixToolset.entrySet().stream().collect(
499                 Collectors.toMap(
500                         entry -> entry.getKey(),
501                         entry -> entry.getValue().path)))
502         .setWixObjDir(TEMP_ROOT.fetchFrom(params).toPath().resolve("wixobj"))
503         .setWorkDir(WIN_APP_IMAGE.fetchFrom(params).toPath())
504         .addSource(CONFIG_ROOT.fetchFrom(params).toPath().resolve("main.wxs"), wixVars)
505         .addSource(CONFIG_ROOT.fetchFrom(params).toPath().resolve("bundle.wxf"), null);
506 
507         Log.verbose(MessageFormat.format(I18N.getString(
508                 "message.generating-msi"), msiOut.getAbsolutePath()));
509 
510         boolean enableLicenseUI = (LICENSE_FILE.fetchFrom(params) != null);
511         boolean enableInstalldirUI = INSTALLDIR_CHOOSER.fetchFrom(params);
512 
513         if (!MSI_SYSTEM_WIDE.fetchFrom(params)) {
514             wixPipeline.addLightOptions("-sice:ICE91");
515         }
516         if (enableLicenseUI || enableInstalldirUI) {
517             wixPipeline.addLightOptions("-ext", "WixUIExtension");
518         }
519 
520         wixPipeline.addLightOptions("-loc",
521                 CONFIG_ROOT.fetchFrom(params).toPath().resolve(I18N.getString(
522                         "resource.wxl-file-name")).toAbsolutePath().toString());
523 
524         // Only needed if we using CA dll, so Wix can find it
525         if (enableInstalldirUI) {
526             wixPipeline.addLightOptions("-b", CONFIG_ROOT.fetchFrom(params).getAbsolutePath());
527         }
528 
529         wixPipeline.buildMsi(msiOut.toPath().toAbsolutePath());
530 
531         return msiOut;
532     }

534     public static void ensureByMutationFileIsRTF(File f) {
535         if (f == null || !f.isFile()) return;
536 
537         try {
538             boolean existingLicenseIsRTF = false;
539 
540             try (FileInputStream fin = new FileInputStream(f)) {
541                 byte[] firstBits = new byte[7];
542 
543                 if (fin.read(firstBits) == firstBits.length) {
544                     String header = new String(firstBits);
545                     existingLicenseIsRTF = "{\\rtf1\\".equals(header);
546                 }
547             }
548 
549             if (!existingLicenseIsRTF) {
550                 List<String> oldLicense = Files.readAllLines(f.toPath());
551                 try (Writer w = Files.newBufferedWriter(
552                         f.toPath(), Charset.forName("Windows-1252"))) {
553                     w.write("{\\rtf1\\ansi\\ansicpg1252\\deff0\\deflang1033"
554                             + "{\\fonttbl{\\f0\\fnil\\fcharset0 Arial;}}\n"
555                             + "\\viewkind4\\uc1\\pard\\sa200\\sl276"
556                             + "\\slmult1\\lang9\\fs20 ");
557                     oldLicense.forEach(l -> {
558                         try {
559                             for (char c : l.toCharArray()) {
560                                 // 0x00 <= ch < 0x20 Escaped (\'hh)
561                                 // 0x20 <= ch < 0x80 Raw(non - escaped) char
562                                 // 0x80 <= ch <= 0xFF Escaped(\ 'hh)
563                                 // 0x5C, 0x7B, 0x7D (special RTF characters
564                                 // \,{,})Escaped(\'hh)
565                                 // ch > 0xff Escaped (\\ud###?)
566                                 if (c < 0x10) {
567                                     w.write("\\'0");
568                                     w.write(Integer.toHexString(c));
569                                 } else if (c > 0xff) {
570                                     w.write("\\ud");
571                                     w.write(Integer.toString(c));
572                                     // \\uc1 is in the header and in effect
573                                     // so we trail with a replacement char if
574                                     // the font lacks that character - '?'
575                                     w.write("?");
576                                 } else if ((c < 0x20) || (c >= 0x80) ||
577                                         (c == 0x5C) || (c == 0x7B) ||
578                                         (c == 0x7D)) {
579                                     w.write("\\'");
580                                     w.write(Integer.toHexString(c));
581                                 } else {
582                                     w.write(c);
583                                 }
584                             }
585                             // blank lines are interpreted as paragraph breaks
586                             if (l.length() < 1) {
587                                 w.write("\\par");
588                             } else {
589                                 w.write(" ");
590                             }
591                             w.write("\r\n");
592                         } catch (IOException e) {
593                             Log.verbose(e);
594                         }
595                     });
596                     w.write("}\r\n");
597                 }
598             }
599         } catch (IOException e) {
600             Log.verbose(e);
601         }
602 
603     }
