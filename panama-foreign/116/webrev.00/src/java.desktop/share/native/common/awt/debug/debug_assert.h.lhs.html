26 #if !defined(_DASSERT_H)
27 #define _DASSERT_H
28 
29 #if defined(__cplusplus)
30 extern "C" {
31 #endif
32 
33 #include "debug_util.h"
34 
35 /* Use THIS_FILE when it is available. */
36 #ifndef THIS_FILE
37     #define THIS_FILE __FILE__
38 #endif
39 
40 #if defined(DEBUG)
41 
42 #define DASSERT(_expr) \
43         if ( !(_expr) ) { \
44             DAssert_Impl( #_expr, THIS_FILE, __LINE__); \
45         } else { \
46         }
47 
48 #define DASSERTMSG(_expr, _msg) \
49         if ( !(_expr) ) { \
50             DAssert_Impl( (_msg), THIS_FILE, __LINE__); \
51         } else { \
52         }
53 
54 /* prototype for assert function */
55 typedef void (*DASSERT_CALLBACK)(const char * msg, const char * file, int line);
56 
57 /* JNIEXPORT because this function is also called from libawt_xawt */
58 JNIEXPORT void JNICALL DAssert_Impl(const char * msg, const char * file, int line);
59 extern void DAssert_SetCallback( DASSERT_CALLBACK pfn );
60 
61 #else /* DEBUG not defined */
62 
63 #define DASSERT(_expr)
64 #define DASSERTMSG(_expr, _msg)
65 
66 #endif /* if defined(DEBUG) */
67 
68 #if defined(__cplusplus)
69 } /* extern "C" */
70 #endif
71 
72 #endif /* _DASSERT_H */
