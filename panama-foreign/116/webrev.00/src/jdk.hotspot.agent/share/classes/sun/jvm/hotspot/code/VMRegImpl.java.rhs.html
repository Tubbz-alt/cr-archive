25 package sun.jvm.hotspot.code;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.*;
33 import sun.jvm.hotspot.utilities.Observable;
34 import sun.jvm.hotspot.utilities.Observer;
35 
36 public class VMRegImpl {
37 
38   private static VMReg stack0;
39   private static int stack0Val;
40   private static Address stack0Addr;
41   private static AddressField regNameField;
42   private static int stackSlotSize;
43 
44   static {
45     VM.registerVMInitializedObserver(new Observer() {
46         public void update(Observable o, Object data) {
47           initialize(VM.getVM().getTypeDataBase());
48         }
49       });
50   }
51 
52   private static void initialize(TypeDataBase db) {
53     Type type = db.lookupType("VMRegImpl");
54     AddressField stack0Field = type.getAddressField("stack0");
55     stack0Addr = stack0Field.getValue();
56     stack0Val = (int) stack0Addr.hashCode();
57     stack0 = new VMReg(stack0Val);
58     regNameField = type.getAddressField("regName[0]");
59     stackSlotSize = db.lookupIntConstant("VMRegImpl::stack_slot_size");
60   }
61 
62   public static VMReg getStack0() {
63     return stack0;
64   }
65 
66   public static String getRegisterName(int index) {
67     if (Assert.ASSERTS_ENABLED) {
68       Assert.that(index >= 0 && index < stack0Val, "invalid index : " + index);
69     }
70     Address regName = regNameField.getStaticFieldAddress();
71     long addrSize = VM.getVM().getAddressSize();
72     return CStringUtilities.getString(regName.getAddressAt(index * addrSize));
73   }
74 
75   public static int getStackSlotSize() {
76     return stackSlotSize;
77   }
78 }
