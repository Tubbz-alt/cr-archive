25 package sun.jvm.hotspot.utilities;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 import sun.jvm.hotspot.oops.Oop;
30 import sun.jvm.hotspot.oops.ObjectHeap;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.runtime.*;


33 
34 public class BasicHashtableEntry extends VMObject {
35   static {
36     VM.registerVMInitializedObserver(new Observer() {
37         public void update(Observable o, Object data) {
38           initialize(VM.getVM().getTypeDataBase());
39         }
40       });
41   }
42 
43   private static synchronized void initialize(TypeDataBase db) {
44     Type type = db.lookupType("BasicHashtableEntry<mtInternal>");
45     hashField      = type.getCIntegerField("_hash");
46     nextField      = type.getAddressField("_next");
47   }
48 
49   // Fields
50   private static CIntegerField hashField;
51   private static AddressField  nextField;
52 
53   // Accessors
54   public long hash() {
55     return hashField.getValue(addr) & 0xFFFFFFFFL;
56   }
57 
58   private long nextAddressValue() {
59     Debugger dbg = VM.getVM().getDebugger();
60     Address nextValue = nextField.getValue(addr);
61     return (nextValue != null) ? dbg.getAddressValue(nextValue) : 0L;
62   }
63 
64   public boolean isShared() {
65     return (nextAddressValue() & 1L) != 0;
66   }
67 
68   public BasicHashtableEntry next() {
69     Address nextValue = nextField.getValue(addr);
70     Address next = (nextValue != null)? nextValue.andWithMask(-2L) : null;
71     // using this.getClass so that return type will be as expected in
72     // subclass context. But, because we can't use covariant return type
73     // caller has to use this next and cast the result to correct type.
74     return (BasicHashtableEntry) VMObjectFactory.newObject(this.getClass(), next);
75   }
76 
77   public BasicHashtableEntry(Address addr) {
78     super(addr);
79   }
80 }
