 56     // FIXME: could keep sorted list of these to be able to do binary

 58     if (pc == null) {
 59       return null;
 60     }
 61     List objs = getLoadObjectList();
 62     for (Iterator iter = objs.iterator(); iter.hasNext(); ) {
 63       LoadObject obj = (LoadObject) iter.next();
 64       if (AddressOps.lte(obj.getBase(), pc) && (pc.minus(obj.getBase()) < obj.getSize())) {
