25 package sun.jvm.hotspot.oops;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.interpreter.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.types.*;
33 import sun.jvm.hotspot.utilities.*;
34 import sun.jvm.hotspot.utilities.Observable;
35 import sun.jvm.hotspot.utilities.Observer;
36 
37 public class CheckedExceptionElement {
38   static {
39     VM.registerVMInitializedObserver(new Observer() {
40         public void update(Observable o, Object data) {
41           initialize(VM.getVM().getTypeDataBase());
42         }
43       });
44   }
45 
46   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
47     Type type            = db.lookupType("CheckedExceptionElement");
48     offsetOfClassCPIndex = type.getCIntegerField("class_cp_index").getOffset();
49   }
50 
51   private static long offsetOfClassCPIndex;
52 
53   private Address   handle;
54   private long      offset;
55 
56   public CheckedExceptionElement(Address handle, long offset) {
57     this.handle = handle;
58     this.offset = offset;
59   }
60 
61   public int getClassCPIndex() {
62     return (int) handle.getCIntegerAt(offset + offsetOfClassCPIndex, 2, true);
63   }
64 }
