 25 package sun.jvm.hotspot.debugger.cdbg.basic;
 26 
 27 import java.util.*;
 28 import sun.jvm.hotspot.debugger.*;
 29 import sun.jvm.hotspot.debugger.cdbg.*;
 30 import sun.jvm.hotspot.utilities.AddressOps;
 31 
 32 public class BasicLineNumberMapping {
 33   private List<BasicLineNumberInfo> infoList;
 34 
 35   public BasicLineNumberMapping() {
 36   }
 37 
 38   /** Add line number information for the given PC. The end PC may be
 39       a very loose approximation (i.e., the end of the given DLL) if
 40       that information is not available in the debug information.
 41       recomputeEndPCs() will recompute them if needed. */
 42   public void addLineNumberInfo(BasicLineNumberInfo info) {
 43     if (infoList == null) {
 44       infoList = new ArrayList<>();
 45     }
 46     infoList.add(info);
 47   }
 48 
 49   /** Sort the line number information by increasing starting program
 50       counter. This must be done before any queries are made. */
 51   public void sort() {
 52     if (infoList == null) return;
 53     Collections.sort(infoList, new Comparator<>() {
 54         public int compare(BasicLineNumberInfo l1, BasicLineNumberInfo l2) {


 55           Address a1 = l1.getStartPC();
 56           Address a2 = l2.getStartPC();
 57           if (AddressOps.lt(a1, a2)) { return -1; }
 58           if (AddressOps.gt(a1, a2)) { return 1; }
 59           return 0;
 60         }
 61       });
 62   }
 63 
 64   /** Recomputes the ending PCs of each interval based on the starting
 65       PC of the next one. If this needs to be called, must be called
 66       after sort(). */
 67   public void recomputeEndPCs() {
 68     if (infoList == null) return;
 69     for (int i = 0; i < infoList.size() - 1; i++) {
 70       BasicLineNumberInfo i1 = get(i);
 71       BasicLineNumberInfo i2 = get(i + 1);
 72       i1.setEndPC(i2.getStartPC());
 73     }
 74   }
 75 
 76   public BasicLineNumberInfo lineNumberForPC(Address pc) throws DebuggerException {
 77     if (infoList == null) return null;
 78     return searchLineNumbers(pc, 0, infoList.size() - 1);
 79   }
 80 
 81   public void iterate(LineNumberVisitor v) {
 82     if (infoList == null) return;
 83     for (int i = 0; i < infoList.size(); i++) {
 84       v.doLineNumber(get(i));
 85     }
 86   }
 87 
 88   //----------------------------------------------------------------------
 89   // Internals only below this point
 90   //
 91 
 92   private BasicLineNumberInfo get(int i) {
 93     return (BasicLineNumberInfo) infoList.get(i);
 94   }
 95 
 96   private BasicLineNumberInfo searchLineNumbers(Address addr, int lowIdx, int highIdx) {
 97     if (highIdx < lowIdx) return null;
 98     if (lowIdx == highIdx) {
 99       // Base case: see whether start PC is less than or equal to addr
100       if (check(addr, lowIdx)) {
101         return get(lowIdx);
102       } else {
103         return null;
104       }
105     } else if (lowIdx == highIdx - 1) {
106       if (check(addr, lowIdx)) {
107         return get(lowIdx);
108       } else if (check(addr, highIdx)) {
109         return get(highIdx);
110       } else {
111         return null;
112       }
113     }
114     int midIdx = (lowIdx + highIdx) >> 1;
115     BasicLineNumberInfo info = get(midIdx);
116     if (AddressOps.lt(addr, info.getStartPC())) {
117       // Always move search down
118       return searchLineNumbers(addr, lowIdx, midIdx);
119     } else if (AddressOps.equal(addr, info.getStartPC())) {
120       return info;
121     } else {
122       // Move search up
123       return searchLineNumbers(addr, midIdx, highIdx);
124     }
125   }
126 
127   private boolean check(Address addr, int idx) {
128     BasicLineNumberInfo info = get(idx);
129     if (AddressOps.lte(info.getStartPC(), addr)) {
130       return true;
131     } else {
132       return false;
133     }
134   }
135 }
