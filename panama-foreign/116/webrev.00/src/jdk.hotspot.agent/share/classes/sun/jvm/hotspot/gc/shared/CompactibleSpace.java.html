25 package sun.jvm.hotspot.gc.shared;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.Observable;
33 import sun.jvm.hotspot.utilities.Observer;
34 
35 /** A space that supports compaction operations.  This is usually, but
36     not necessarily, a space that is normally contiguous.  But, for
37     example, a free-list-based space whose normal collection is a
38     mark-sweep without compaction could still support compaction in
39     full GC's. */
40 
41 public abstract class CompactibleSpace extends Space {
42   private static AddressField compactionTopField;
43 
44   static {
45     VM.registerVMInitializedObserver(new Observer() {
46         public void update(Observable o, Object data) {
47           initialize(VM.getVM().getTypeDataBase());
48         }
49       });
50   }
51 
52   private static synchronized void initialize(TypeDataBase db) {
53     Type type = db.lookupType("CompactibleSpace");
54 
55     compactionTopField = type.getAddressField("_compaction_top");
56   }
57 
58   public CompactibleSpace(Address addr) {
59     super(addr);
60   }
61 
62   /** May be used temporarily during a compaction phase. */
63   public Address compactionTop() {
64     return compactionTopField.getValue(addr);
65   }
66 }
