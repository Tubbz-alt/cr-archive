 25 package sun.jvm.hotspot.gc.serial;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 
 30 import sun.jvm.hotspot.debugger.*;
 31 import sun.jvm.hotspot.gc.shared.*;
 32 import sun.jvm.hotspot.runtime.*;
 33 import sun.jvm.hotspot.types.*;
 34 import sun.jvm.hotspot.utilities.Observable;
 35 import sun.jvm.hotspot.utilities.Observer;
 36 
 37 /** DefNewGeneration is a young generation containing eden, from- and
 38     to-space. */
 39 
 40 public class DefNewGeneration extends Generation {
 41   protected static AddressField edenSpaceField;
 42   protected static AddressField fromSpaceField;
 43   protected static AddressField toSpaceField;
 44 
 45   static {
 46     VM.registerVMInitializedObserver(new Observer() {
 47         public void update(Observable o, Object data) {
 48           initialize(VM.getVM().getTypeDataBase());
 49         }
 50       });
 51   }
 52 
 53   private static synchronized void initialize(TypeDataBase db) {
 54     Type type = db.lookupType("DefNewGeneration");
 55 
 56     edenSpaceField = type.getAddressField("_eden_space");
 57     fromSpaceField = type.getAddressField("_from_space");
 58     toSpaceField   = type.getAddressField("_to_space");
 59   }
 60 
 61   public DefNewGeneration(Address addr) {
 62     super(addr);
 63   }
 64 
 65   public Generation.Name kind() {
 66     return Generation.Name.DEF_NEW;
 67   }
 68 
 69   // Accessing spaces
 70   public ContiguousSpace eden() {
 71     return (ContiguousSpace) VMObjectFactory.newObject(ContiguousSpace.class, edenSpaceField.getValue(addr));
 72   }
 73 
 74   public ContiguousSpace from() {
 75     return (ContiguousSpace) VMObjectFactory.newObject(ContiguousSpace.class, fromSpaceField.getValue(addr));
 76   }
 77 
 78   public ContiguousSpace to() {
 79     return (ContiguousSpace) VMObjectFactory.newObject(ContiguousSpace.class, toSpaceField.getValue(addr));
 80   }
 81 
 82   public long capacity()            { return eden().capacity() + from().capacity(); /* to() is only used during scavenge */ }
 83   public long used()                { return eden().used()     + from().used();     /* to() is only used during scavenge */ }
 84   public long free()                { return eden().free()     + from().free();     /* to() is only used during scavenge */ }
 85   public long contiguousAvailable() { return eden().free(); }
 86 
 87   public String name() {
 88     return "default new generation";
 89   }
 90 
 91   public void spaceIterate(SpaceClosure blk, boolean usedOnly) {
 92     blk.doSpace(eden());
 93     blk.doSpace(from());
 94     if (!usedOnly) {
 95       blk.doSpace(to());
 96     }
 97   }
 98 
 99   public void liveRegionsIterate(LiveRegionsClosure closure) {
100     closure.doLiveRegions(eden());
101     closure.doLiveRegions(from());
102   }
103 
104   public void printOn(PrintStream tty) {
105     tty.print("  eden");
106     eden().printOn(tty);
107     tty.print("\n  from");
108     from().printOn(tty);
109     tty.print("\n  to  ");
110     to().printOn(tty);
111   }
112 }
