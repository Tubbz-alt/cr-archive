25 package sun.jvm.hotspot.gc.parallel;
26 
27 import java.io.*;
28 import java.util.*;
29 
30 import sun.jvm.hotspot.debugger.*;
31 import sun.jvm.hotspot.memory.*;
32 import sun.jvm.hotspot.runtime.*;
33 import sun.jvm.hotspot.types.*;
34 
35 public class PSOldGen extends VMObject {
36    static {
37       VM.registerVMInitializedObserver(new Observer() {
38          public void update(Observable o, Object data) {
39             initialize(VM.getVM().getTypeDataBase());
40          }
41       });
42    }
43 
44    private static synchronized void initialize(TypeDataBase db) {
45       Type type = db.lookupType("PSOldGen");
46       objectSpaceField = type.getAddressField("_object_space");
47    }
48 
49    public PSOldGen(Address addr) {
50       super(addr);
51    }
52 
53    // Fields
54    private static AddressField objectSpaceField;
55 
56    // Accessors
57    public MutableSpace objectSpace() {
58       return (MutableSpace) VMObjectFactory.newObject(MutableSpace.class, objectSpaceField.getValue(addr));
59    }
60 
61    public long capacity() {
62       return objectSpace().capacity();
63    }
64 
65    public long used() {
66       return objectSpace().used();
67    }
68 
69    public boolean isIn(Address a) {
70       return objectSpace().contains(a);
71    }
72 
73    public void printOn(PrintStream tty) {
74       tty.print("PSOldGen [ ");
75       objectSpace().printOn(tty);
76       tty.print(" ] ");
77    }
78 }
