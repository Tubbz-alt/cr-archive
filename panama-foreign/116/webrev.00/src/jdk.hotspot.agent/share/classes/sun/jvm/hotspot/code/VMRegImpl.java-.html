25 package sun.jvm.hotspot.code;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.*;
33 
34 public class VMRegImpl {
35 
36   private static VMReg stack0;
37   private static int stack0Val;
38   private static Address stack0Addr;
39   private static AddressField regNameField;
40   private static int stackSlotSize;
41 
42   static {
43     VM.registerVMInitializedObserver(new Observer() {
44         public void update(Observable o, Object data) {
45           initialize(VM.getVM().getTypeDataBase());
46         }
47       });
48   }
49 
50   private static void initialize(TypeDataBase db) {
51     Type type = db.lookupType("VMRegImpl");
52     AddressField stack0Field = type.getAddressField("stack0");
53     stack0Addr = stack0Field.getValue();
54     stack0Val = (int) stack0Addr.hashCode();
55     stack0 = new VMReg(stack0Val);
56     regNameField = type.getAddressField("regName[0]");
57     stackSlotSize = db.lookupIntConstant("VMRegImpl::stack_slot_size");
58   }
59 
60   public static VMReg getStack0() {
61     return stack0;
62   }
63 
64   public static String getRegisterName(int index) {
65     if (Assert.ASSERTS_ENABLED) {
66       Assert.that(index >= 0 && index < stack0Val, "invalid index : " + index);
67     }
68     Address regName = regNameField.getStaticFieldAddress();
69     long addrSize = VM.getVM().getAddressSize();
70     return CStringUtilities.getString(regName.getAddressAt(index * addrSize));
71   }
72 
73   public static int getStackSlotSize() {
74     return stackSlotSize;
75   }
76 }
