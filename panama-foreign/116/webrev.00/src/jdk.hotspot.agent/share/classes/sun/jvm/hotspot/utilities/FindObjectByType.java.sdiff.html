25 package sun.jvm.hotspot.utilities;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.oops.*;
30 
31 public class FindObjectByType implements HeapVisitor {
32   private Klass type;
33   private List results = new ArrayList();
34 
35   public FindObjectByType(Klass type) {
36     this.type = type;
37   }
38 
39   /** Returns a List of Oops */
40   public List getResults() {
41     return results;
42   }
43 
44   public void prologue(long size) {}
45   public void epilogue()          {}
46 
47   public boolean doObj(Oop obj) {
48     if (obj.getKlass().equals(type)) {
49       results.add(obj);
50     }
51         return false;
52   }
53 }
