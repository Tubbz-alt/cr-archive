25 package sun.jvm.hotspot.oops;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 
33 public class BreakpointInfo extends VMObject {
34   private static CIntegerField origBytecodeField;
35   private static CIntegerField bciField;
36   private static CIntegerField nameIndexField;
37   private static CIntegerField signatureIndexField;
38   private static AddressField  nextField;
39 
40   static {
41     VM.registerVMInitializedObserver(new Observer() {
42         public void update(Observable o, Object data) {
43           initialize(VM.getVM().getTypeDataBase());
44         }
45       });
46   }
47 
48   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
49     if (!VM.getVM().isJvmtiSupported()) {
50       // no BreakpointInfo support without JVMTI
51       return;
52     }
53 
54     Type type                  = db.lookupType("BreakpointInfo");
55 
56     origBytecodeField   = type.getCIntegerField("_orig_bytecode");
57     bciField            = type.getCIntegerField("_bci");
58     nameIndexField      = type.getCIntegerField("_name_index");
59     signatureIndexField = type.getCIntegerField("_signature_index");
60     nextField           = type.getAddressField ("_next");
61   }
62 
63   public BreakpointInfo(Address addr) {
64     super(addr);
65   }
66 
67   public int  getOrigBytecode()   { return (int) origBytecodeField.getValue(addr);   }
68   public int  getBCI()            { return (int) bciField.getValue(addr);            }
69   public long getNameIndex()      { return nameIndexField.getValue(addr);            }
70   public long getSignatureIndex() { return signatureIndexField.getValue(addr);       }
71   public BreakpointInfo getNext() {
72     return (BreakpointInfo) VMObjectFactory.newObject(BreakpointInfo.class, nextField.getValue(addr));
73   }
74 
75   public boolean match(Method m, int bci) {
76     return (bci == getBCI() && match(m));
77   }
78 
79   public boolean match(Method m) {
80     return (getNameIndex() == m.getNameIndex() &&
81             getSignatureIndex() == m.getSignatureIndex());
82   }
83 }
