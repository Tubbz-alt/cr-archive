25 package sun.jvm.hotspot.opto;
26 
27 import java.util.*;
28 import java.io.*;
29 import sun.jvm.hotspot.code.*;
30 import sun.jvm.hotspot.debugger.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.oops.*;
33 import sun.jvm.hotspot.ci.*;
34 import sun.jvm.hotspot.types.*;


35 
36 public class CallJavaNode extends CallNode {
37   static {
38     VM.registerVMInitializedObserver(new Observer() {
39         public void update(Observable o, Object data) {
40           initialize(VM.getVM().getTypeDataBase());
41         }
42       });
43   }
44 
45   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
46     Type type      = db.lookupType("CallJavaNode");
47     methodField = type.getAddressField("_method");
48   }
49 
50   private static AddressField methodField;
51 
52   public CallJavaNode(Address addr) {
53     super(addr);
54   }
55 
56   public ciMethod method() {
57     return (ciMethod) ciObjectFactory.getMetadata(methodField.getValue(getAddress()));
58   }
59 
60   public void dumpSpec(PrintStream out) {
61     if (method() !=  null) {
62       out.print(" " + method().method().externalNameAndSignature());
63     }
64     super.dumpSpec(out);
65   }
66 }
