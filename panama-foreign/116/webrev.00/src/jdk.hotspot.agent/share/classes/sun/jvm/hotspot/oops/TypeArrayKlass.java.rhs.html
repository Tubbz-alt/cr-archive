 25 package sun.jvm.hotspot.oops;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 import sun.jvm.hotspot.debugger.*;
 30 import sun.jvm.hotspot.runtime.*;
 31 import sun.jvm.hotspot.types.*;
 32 import sun.jvm.hotspot.utilities.*;
 33 import sun.jvm.hotspot.utilities.Observable;
 34 import sun.jvm.hotspot.utilities.Observer;
 35 
 36 // TypeArrayKlass is a proxy for TypeArrayKlass in the JVM
 37 
 38 public class TypeArrayKlass extends ArrayKlass {
 39   static {
 40     VM.registerVMInitializedObserver(new Observer() {
 41         public void update(Observable o, Object data) {
 42           initialize(VM.getVM().getTypeDataBase());
 43         }
 44       });
 45   }
 46 
 47   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
 48     Type t             = db.lookupType("TypeArrayKlass");
 49     maxLength          = new IntField(t.getJIntField("_max_length"), 0);
 50   }
 51 
 52   public TypeArrayKlass(Address addr) {
 53     super(addr);
 54   }
 55 
 56   private static IntField  maxLength;
 57 
 58   public long getMaxLength()          { return  maxLength.getValue(this); }
 59 
 60   public static final int T_BOOLEAN = 4;
 61   public static final int T_CHAR    = 5;
 62   public static final int T_FLOAT   = 6;
 63   public static final int T_DOUBLE  = 7;
 64   public static final int T_BYTE    = 8;
 65   public static final int T_SHORT   = 9;
 66   public static final int T_INT     = 10;
 67   public static final int T_LONG    = 11;
 68 
 69   public String getTypeName() {
 70     switch ((int) getElementType()) {
 71       case T_BOOLEAN: return "[Z";
 72       case T_CHAR:    return "[C";
 73       case T_FLOAT:   return "[F";
 74       case T_DOUBLE:  return "[D";
 75       case T_BYTE:    return "[B";
 76       case T_SHORT:   return "[S";
 77       case T_INT:     return "[I";
 78       case T_LONG:    return "[J";
 79     }
 80     return "Unknown TypeArray";
 81   }
 82 
 83   public String getElementTypeName() {
 84     switch ((int) getElementType()) {
 85       case T_BOOLEAN: return "boolean";
 86       case T_CHAR:    return "char";
 87       case T_FLOAT:   return "float";
 88       case T_DOUBLE:  return "double";
 89       case T_BYTE:    return "byte";
 90       case T_SHORT:   return "short";
 91       case T_INT:     return "int";
 92       case T_LONG:    return "long";
 93     }
 94     throw new RuntimeException("should not reach here");
 95   }
 96 
 97   public void printValueOn(PrintStream tty) {
 98     tty.print("TypeArrayKlass for " + getTypeName());
 99   }
100 
101   public void iterateFields(MetadataVisitor visitor) {
102     super.iterateFields(visitor);
103       visitor.doInt(maxLength, true);
104     }
105 
106   public Klass arrayKlassImpl(boolean orNull, int n) {
107     int dimension = (int) getDimension();
108     if (Assert.ASSERTS_ENABLED) {
109       Assert.that(dimension <= n, "check order of chain");
110     }
111     if (dimension == n)
112       return this;
113     ObjArrayKlass ak = (ObjArrayKlass) getHigherDimension();
114     if (ak == null) {
115       if (orNull) return null;
116       // FIXME: would need to change in reflective system to actually
117       // allocate klass
118       throw new RuntimeException("Can not allocate array klasses in debugging system");
119     }
120     if (orNull) {
121       return ak.arrayKlassOrNull(n);
122     }
123     return ak.arrayKlass(n);
124   }
125 
126   public Klass arrayKlassImpl(boolean orNull) {
127     return arrayKlassImpl(orNull, (int) (getDimension() + 1));
128   }
129 }
