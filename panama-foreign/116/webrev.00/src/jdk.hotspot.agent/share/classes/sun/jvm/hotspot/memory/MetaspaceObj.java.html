25 package sun.jvm.hotspot.memory;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 import sun.jvm.hotspot.runtime.*;
30 import sun.jvm.hotspot.types.*;
31 import sun.jvm.hotspot.utilities.Observable;
32 import sun.jvm.hotspot.utilities.Observer;
33 
34 public class MetaspaceObj {
35   private static Address sharedMetaspaceBaseAddr;
36   private static Address sharedMetaspaceTopAddr;
37 
38   static {
39     VM.registerVMInitializedObserver(new Observer() {
40         public void update(Observable o, Object data) {
41           initialize(VM.getVM().getTypeDataBase());
42         }
43       });
44   }
45 
46   private static synchronized void initialize(TypeDataBase db) {
47     Type type = db.lookupType("MetaspaceObj");
48     sharedMetaspaceBaseAddr = type.getAddressField("_shared_metaspace_base").getStaticFieldAddress();
49     sharedMetaspaceTopAddr  = type.getAddressField("_shared_metaspace_top").getStaticFieldAddress();
50   }
51 
52   public static boolean isShared(Address addr) {
53     Address base = sharedMetaspaceBaseAddr.getAddressAt(0);
54     Address top  = sharedMetaspaceTopAddr. getAddressAt(0);
55 
56     return base.lessThanOrEqual(addr) && addr.lessThan(top);
57   }
58 }
