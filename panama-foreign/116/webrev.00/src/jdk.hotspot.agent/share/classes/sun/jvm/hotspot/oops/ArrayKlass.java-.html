 25 package sun.jvm.hotspot.oops;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 import sun.jvm.hotspot.utilities.*;
 30 import sun.jvm.hotspot.debugger.*;
 31 import sun.jvm.hotspot.memory.*;
 32 import sun.jvm.hotspot.runtime.*;
 33 import sun.jvm.hotspot.types.*;
 34 
 35 // ArrayKlass is the abstract class for all array classes
 36 
 37 public class ArrayKlass extends Klass {
 38   static {
 39     VM.registerVMInitializedObserver(new Observer() {
 40         public void update(Observable o, Object data) {
 41           initialize(VM.getVM().getTypeDataBase());
 42         }
 43       });
 44   }
 45 
 46   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
 47     Type type          = db.lookupType("ArrayKlass");
 48     dimension          = new CIntField(type.getCIntegerField("_dimension"), 0);
 49     higherDimension    = new MetadataField(type.getAddressField("_higher_dimension"), 0);
 50     lowerDimension     = new MetadataField(type.getAddressField("_lower_dimension"), 0);
 51     javaLangCloneableName = null;
 52     javaLangObjectName = null;
 53     javaIoSerializableName = null;
 54   }
 55 
 56   public ArrayKlass(Address addr) {
 57     super(addr);
 58   }
 59 
 60   public boolean isArrayKlass()     { return true; }
 61   private static CIntField dimension;
 62   private static MetadataField  higherDimension;
 63   private static MetadataField  lowerDimension;
 64 
 65   public Klass getJavaSuper() {
 66     SystemDictionary sysDict = VM.getVM().getSystemDictionary();
 67     return sysDict.getObjectKlass();
 68   }
 69 
 70   public long  getDimension()       { return         dimension.getValue(this); }
 71   public Klass getHigherDimension() { return (Klass) higherDimension.getValue(this); }
 72   public Klass getLowerDimension()  { return (Klass) lowerDimension.getValue(this); }
 73 
 74   // constant class names - javaLangCloneable, javaIoSerializable, javaLangObject
 75   // Initialized lazily to avoid initialization ordering dependencies between ArrayKlass and String
 76   private static String javaLangCloneableName;
 77   private static String javaLangObjectName;
 78   private static String javaIoSerializableName;
 79   private static String javaLangCloneableName() {
 80     if (javaLangCloneableName == null) {
 81       javaLangCloneableName = "java/lang/Cloneable";
 82     }
 83     return javaLangCloneableName;
 84   }
 85 
 86   private static String javaLangObjectName() {
 87     if (javaLangObjectName == null) {
 88       javaLangObjectName = "java/lang/Object";
 89     }
 90     return javaLangObjectName;
 91   }
 92 
 93   private static String javaIoSerializableName() {
 94     if (javaIoSerializableName == null) {
 95       javaIoSerializableName = "java/io/Serializable";
 96     }
 97     return javaIoSerializableName;
 98   }
 99 
100   public int getClassStatus() {
101      return JVMDIClassStatus.VERIFIED | JVMDIClassStatus.PREPARED | JVMDIClassStatus.INITIALIZED;
102   }
103 
104   public long computeModifierFlags() {
105      return JVM_ACC_ABSTRACT | JVM_ACC_FINAL | JVM_ACC_PUBLIC;
106   }
107 
108   public long getArrayHeaderInBytes() {
109     return Bits.maskBits(getLayoutHelper() >> LH_HEADER_SIZE_SHIFT, 0xFF);
110   }
111 
112   public int getLog2ElementSize() {
113     return Bits.maskBits(getLayoutHelper() >> LH_LOG2_ELEMENT_SIZE_SHIFT, 0xFF);
114   }
115 
116   public int getElementType() {
117     return Bits.maskBits(getLayoutHelper() >> LH_ELEMENT_TYPE_SHIFT, 0xFF);
118   }
119 
120   boolean computeSubtypeOf(Klass k) {
121     // An array is a subtype of Serializable, Clonable, and Object
122     Symbol name = k.getName();
123     if (name != null && (name.equals(javaIoSerializableName()) ||
124                          name.equals(javaLangCloneableName()) ||
125                          name.equals(javaLangObjectName()))) {
126       return true;
127     } else {
128       return false;
129     }
130   }
131 
132   public void printValueOn(PrintStream tty) {
133     tty.print("ArrayKlass");
134   }
135 
136   public void iterateFields(MetadataVisitor visitor) {
137     super.iterateFields(visitor);
138       visitor.doCInt(dimension, true);
139     visitor.doMetadata(higherDimension, true);
140     visitor.doMetadata(lowerDimension, true);
141     }
142   }
