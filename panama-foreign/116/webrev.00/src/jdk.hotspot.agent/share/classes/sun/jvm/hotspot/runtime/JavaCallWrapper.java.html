25 package sun.jvm.hotspot.runtime;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 import sun.jvm.hotspot.types.*;
30 import sun.jvm.hotspot.utilities.Observable;
31 import sun.jvm.hotspot.utilities.Observer;
32 
33 /** <P> A JavaCallWrapper is constructed before each JavaCall and
34     destroyed after the call.  Its purpose is to allocate/deallocate a
35     new handle block and to save/restore the last Java fp/sp. A
36     pointer to the JavaCallWrapper is stored on the stack. </P>
37 
38     <P> NOTE: this port is very minimal and is only designed to get
39     frame traversal working. FIXME: we will have to add platform-
40     specific stuff later and therefore a factory for these
41     objects. </P> */
42 
43 public class JavaCallWrapper extends VMObject {
44   protected static AddressField  anchorField;
45   private static AddressField  lastJavaSPField;
46   private static AddressField  lastJavaPCField;
47 
48   static {
49     VM.registerVMInitializedObserver(new Observer() {
50         public void update(Observable o, Object data) {
51           initialize(VM.getVM().getTypeDataBase());
52         }
53       });
54   }
55 
56   private static synchronized void initialize(TypeDataBase db) {
57     Type type = db.lookupType("JavaCallWrapper");
58     Type anchorType = db.lookupType("JavaFrameAnchor");
59 
60     anchorField      = type.getAddressField("_anchor");
61     lastJavaSPField  = anchorType.getAddressField("_last_Java_sp");
62     lastJavaPCField  = anchorType.getAddressField("_last_Java_pc");
63   }
64 
65   public JavaCallWrapper(Address addr) {
66     super(addr);
67   }
68 
69   public Address getLastJavaSP() {
70     return lastJavaSPField.getValue(addr.addOffsetTo(anchorField.getOffset()));
71   }
72 
73   public Address getLastJavaPC() {
74     return lastJavaPCField.getValue(addr.addOffsetTo(anchorField.getOffset()));
75   }
76 
77 }
