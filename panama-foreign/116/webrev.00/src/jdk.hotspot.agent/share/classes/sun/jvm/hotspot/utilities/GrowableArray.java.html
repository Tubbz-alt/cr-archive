25 package sun.jvm.hotspot.utilities;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 import sun.jvm.hotspot.runtime.*;
30 import sun.jvm.hotspot.oops.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.Observable;
33 import sun.jvm.hotspot.utilities.Observer;
34 
35 public class GrowableArray<T> extends GenericGrowableArray {
36   static {
37     VM.registerVMInitializedObserver(new Observer() {
38         public void update(Observable o, Object data) {
39           initialize(VM.getVM().getTypeDataBase());
40         }
41       });
42   }
43 
44   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
45     Type type      = db.lookupType("GrowableArray<int>");
46     dataField = type.getAddressField("_data");
47   }
48 
49   private static AddressField dataField;
50 
51   private InstanceConstructor<T> virtualConstructor;
52 
53   public static <S> GrowableArray<S> create(Address addr, InstanceConstructor<S> v) {
54     if (addr == null) return null;
55     return new GrowableArray<S>(addr, v);
56   }
57 
58   public T at(int i) {
59     if (i < 0 || i >= length()) throw new ArrayIndexOutOfBoundsException(i);
60     Address data = dataField.getValue(getAddress());
61     Address addr = data.getAddressAt(i * VM.getVM().getAddressSize());
62     if (addr == null) return null;
63     return (T) virtualConstructor.instantiateWrapperFor(addr);
64   }
65 
66   private GrowableArray(Address addr, InstanceConstructor<T> v) {
67     super(addr);
68     virtualConstructor = v;
69   }
70   public Address getData() {
71     return dataField.getValue(getAddress());
72   }
73 }
