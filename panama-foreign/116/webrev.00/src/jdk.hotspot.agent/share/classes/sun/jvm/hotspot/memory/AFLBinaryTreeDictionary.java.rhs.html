26 package sun.jvm.hotspot.memory;
27 
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.types.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.utilities.Observable;
33 import sun.jvm.hotspot.utilities.Observer;
34 
35 public class AFLBinaryTreeDictionary extends VMObject {
36    static {
37       VM.registerVMInitializedObserver(new Observer() {
38          public void update(Observable o, Object data) {
39             initialize(VM.getVM().getTypeDataBase());
40          }
41       });
42    }
43 
44    private static synchronized void initialize(TypeDataBase db) {
45       Type type = db.lookupType("AFLBinaryTreeDictionary");
46       totalSizeField = type.getCIntegerField("_total_size");
47    }
48 
49    // Fields
50    private static CIntegerField totalSizeField;
51 
52    // Accessors
53    public long size() {
54       return totalSizeField.getValue(addr);
55    }
56 
57    // Constructor
58    public AFLBinaryTreeDictionary(Address addr) {
59       super(addr);
60    }
61 }
