25 package sun.jvm.hotspot.utilities.soql;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.oops.*;
29 
30 /**
31    This is JavaScript wrapper for Java ArrayKlass.
32 */
33 
34 public abstract class JSJavaArrayKlass extends JSJavaKlass {
35    public JSJavaArrayKlass(ArrayKlass kls, JSJavaFactory fac) {
36       super(kls, fac);
37    }
38 
39    public final ArrayKlass getArrayKlass() {
40       return (ArrayKlass) getKlass();
41    }
42 
43    public Object getMetaClassFieldValue(String name) {
44       if (name.equals("dimension")) {
45          return new Long(getArrayKlass().getDimension());
46       } else {
47          return super.getMetaClassFieldValue(name);
48       }
49    }
50 
51    public boolean hasMetaClassField(String name) {
52       if (name.equals("dimension")) {
53          return true;
54       } else {
55          return super.hasMetaClassField(name);
56       }
57    }
58 
59    public boolean isArray() {
60       return true;
61    }
62 
63    public String[] getMetaClassFieldNames() {
64        String[] superFields = super.getMetaClassFieldNames();
65        String[] res = new String[superFields.length + 1];
66        System.arraycopy(superFields, 0, res, 0, superFields.length);
67        res[superFields.length] = "dimension";
68        return res;
69    }
70 
71    public abstract Object getFieldValue(int index, Array array);
72 }
