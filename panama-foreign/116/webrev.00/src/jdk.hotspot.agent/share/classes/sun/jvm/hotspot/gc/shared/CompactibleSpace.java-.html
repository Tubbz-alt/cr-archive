25 package sun.jvm.hotspot.gc.shared;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 
33 /** A space that supports compaction operations.  This is usually, but
34     not necessarily, a space that is normally contiguous.  But, for
35     example, a free-list-based space whose normal collection is a
36     mark-sweep without compaction could still support compaction in
37     full GC's. */
38 
39 public abstract class CompactibleSpace extends Space {
40   private static AddressField compactionTopField;
41 
42   static {
43     VM.registerVMInitializedObserver(new Observer() {
44         public void update(Observable o, Object data) {
45           initialize(VM.getVM().getTypeDataBase());
46         }
47       });
48   }
49 
50   private static synchronized void initialize(TypeDataBase db) {
51     Type type = db.lookupType("CompactibleSpace");
52 
53     compactionTopField = type.getAddressField("_compaction_top");
54   }
55 
56   public CompactibleSpace(Address addr) {
57     super(addr);
58   }
59 
60   /** May be used temporarily during a compaction phase. */
61   public Address compactionTop() {
62     return compactionTopField.getValue(addr);
63   }
64 }
