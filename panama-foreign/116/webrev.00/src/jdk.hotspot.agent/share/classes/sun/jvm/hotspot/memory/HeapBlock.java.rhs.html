25 package sun.jvm.hotspot.memory;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 import sun.jvm.hotspot.runtime.*;
30 import sun.jvm.hotspot.types.*;
31 import sun.jvm.hotspot.utilities.Observable;
32 import sun.jvm.hotspot.utilities.Observer;
33 
34 public class HeapBlock extends VMObject {
35   private static long          heapBlockSize;
36   private static Field         headerField;
37   // Fields for class Header
38   private static CIntegerField headerLengthField;
39   private static CIntegerField headerUsedField;
40 
41   static {
42     VM.registerVMInitializedObserver(new Observer() {
43         public void update(Observable o, Object data) {
44           initialize(VM.getVM().getTypeDataBase());
45         }
46       });
47   }
48 
49   private static void initialize(TypeDataBase db) {
50     // HeapBlock
51     Type type = db.lookupType("HeapBlock");
52     heapBlockSize     = type.getSize();
53     headerField       = type.getField("_header");
54 
55     // Header
56     type = db.lookupType("HeapBlock::Header");
57     headerLengthField = type.getCIntegerField("_length");
58     headerUsedField   = type.getCIntegerField("_used");
59   }
60 
61   public HeapBlock(Address addr) {
62     super(addr);
63   }
64 
65   public long getLength() {
66     return getHeader().getLength();
67   }
68 
69   public boolean isFree() {
70     return getHeader().isFree();
71   }
72 
73   public Address getAllocatedSpace() {
74     return addr.addOffsetTo(heapBlockSize);
75   }
76 
77   //--------------------------------------------------------------------------------
78   // Internals only below this point
79   //
80 
81   public static class Header extends VMObject {
82     public Header(Address addr) {
83       super(addr);
84     }
85 
86     public long getLength() {
87       return headerLengthField.getValue(addr);
88     }
89 
90     public boolean isFree() {
91       return (headerUsedField.getValue(addr) == 0);
92     }
93   }
94 
95   private Header getHeader() {
96     return (Header) VMObjectFactory.newObject(HeapBlock.Header.class, addr.addOffsetTo(headerField.getOffset()));
97   }
98 }
