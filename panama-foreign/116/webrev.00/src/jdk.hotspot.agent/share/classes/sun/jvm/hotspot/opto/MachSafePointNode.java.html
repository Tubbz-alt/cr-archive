25 package sun.jvm.hotspot.opto;
26 
27 import java.util.*;
28 import java.io.PrintStream;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.oops.*;
32 import sun.jvm.hotspot.types.*;
33 import sun.jvm.hotspot.utilities.Observable;
34 import sun.jvm.hotspot.utilities.Observer;
35 
36 public class MachSafePointNode extends MachReturnNode {
37   static {
38     VM.registerVMInitializedObserver(new Observer() {
39         public void update(Observable o, Object data) {
40           initialize(VM.getVM().getTypeDataBase());
41         }
42       });
43   }
44 
45   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
46     Type type      = db.lookupType("MachSafePointNode");
47     jvmsField = type.getAddressField("_jvms");
48     jvmadjField = new CIntField(type.getCIntegerField("_jvmadj"), 0);
49   }
50 
51   private static AddressField jvmsField;
52   private static CIntField jvmadjField;
53 
54   public MachSafePointNode(Address addr) {
55     super(addr);
56   }
57 
58   public JVMState jvms() {
59     return JVMState.create(jvmsField.getValue(getAddress()));
60   }
61 
62   public void dumpSpec(PrintStream out) {
63     try {
64       JVMState jvms = jvms();
65       if (jvms != null) out.print(" !");
66       if (jvms == null) out.print("empty jvms");
67       while (jvms != null) {
68         Method m = jvms.method().method();
69         int bci = jvms.bci();
70         out.print(" " + m.getMethodHolder().getName().asString().replace('/', '.') + "::" + m.getName().asString() + " @ bci:" + bci);
71         jvms = jvms.caller();
72       }
73     } catch (Exception e) {
74       out.print(e);
75     }
76   }
77 }
