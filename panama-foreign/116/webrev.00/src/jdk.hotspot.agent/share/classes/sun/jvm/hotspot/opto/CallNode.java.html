25 package sun.jvm.hotspot.opto;
26 
27 import java.io.PrintStream;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.oops.*;
32 import sun.jvm.hotspot.types.*;
33 import sun.jvm.hotspot.utilities.Observable;
34 import sun.jvm.hotspot.utilities.Observer;
35 
36 public class CallNode extends SafePointNode {
37   static {
38     VM.registerVMInitializedObserver(new Observer() {
39         public void update(Observable o, Object data) {
40           initialize(VM.getVM().getTypeDataBase());
41         }
42       });
43   }
44 
45   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
46     Type type      = db.lookupType("CallNode");
47     entryPointField = type.getAddressField("_entry_point");
48   }
49 
50   private static AddressField entryPointField;
51 
52   public Address entryPoint() {
53     return entryPointField.getValue(getAddress());
54   }
55 
56   public CallNode(Address addr) {
57     super(addr);
58   }
59 
60   public void dumpSpec(PrintStream out) {
61     out.print(" ");
62     // tf()->dumpOn(st);
63     // if (_cnt != countUnknown)  st->print(" C=%f",_cnt);
64     JVMState jvms = jvms();
65     if (jvms != null)  jvms.dumpSpec(out);
66   }
67 }
