 25 package sun.jvm.hotspot.memory;
 26 
 27 import java.util.*;
 28 import sun.jvm.hotspot.code.*;
 29 import sun.jvm.hotspot.debugger.*;
 30 import sun.jvm.hotspot.runtime.*;
 31 import sun.jvm.hotspot.types.*;
 32 
 33 public class CodeHeap extends VMObject {
 34   private static Field         memoryField;
 35   private static Field         segmapField;
 36   //  private static CIntegerField numberOfCommittedSegmentsField;
 37   //  private static CIntegerField numberOfReservedSegmentsField;
 38   //  private static CIntegerField segmentSizeField;
 39   private static CIntegerField log2SegmentSizeField;
 40   //  private static CIntegerField nextSegmentField;
 41   //  private static AddressField  freelistField;
 42   //  private static CIntegerField freeSegmentsField;
 43 
 44   private VirtualSpace memory;
 45   private VirtualSpace segmentMap;
 46   private int log2SegmentSize;
 47 
 48   static {
 49     VM.registerVMInitializedObserver(new Observer() {
 50         public void update(Observable o, Object data) {
 51           initialize(VM.getVM().getTypeDataBase());
 52         }
 53       });
 54   }
 55 
 56   private static void initialize(TypeDataBase db) {
 57     Type type = db.lookupType("CodeHeap");
 58 
 59     memoryField          = type.getField("_memory");
 60     segmapField          = type.getField("_segmap");
 61     log2SegmentSizeField = type.getCIntegerField("_log2_segment_size");
 62 
 63   }
 64 
 65   public CodeHeap(Address addr) {
 66     super(addr);
 67     log2SegmentSize = (int) log2SegmentSizeField.getValue(addr);
 68     segmentMap = new VirtualSpace(addr.addOffsetTo(segmapField.getOffset()));
 69     memory = new VirtualSpace(addr.addOffsetTo(memoryField.getOffset()));
 70   }
 71 
 72   public Address begin() {
 73     return getMemory().low();
 74   }
 75 
 76   public Address end() {
 77     return getMemory().high();
 78   }
 79 
 80   public boolean contains(Address p) {
 81     return (begin().lessThanOrEqual(p) && end().greaterThan(p));
 82   }
 83 
 84   /** Returns the start of the block containing p or null */
 85   public Address findStart(Address p) {
 86     if (!contains(p)) return null;
 87     HeapBlock h = blockStart(p);
 88     if (h == null || h.isFree()) {
 89       return null;
 90     }
 91     return h.getAllocatedSpace();
 92   }
 93 
 94   private Address nextBlock(Address ptr) {
 95     Address base = blockBase(ptr);
 96     if (base == null) {
 97       return null;
 98     }
 99     HeapBlock block = getBlockAt(base);
100     return base.addOffsetTo(block.getLength() * (1 << getLog2SegmentSize()));
101   }
102 
103   public void iterate(CodeCacheVisitor visitor, CodeCache cache) {
104     CodeBlob lastBlob = null;
105     Address ptr = begin();
106     while (ptr != null && ptr.lessThan(end())) {
107       try {
108         // Use findStart to get a pointer inside blob other findBlob asserts
109         CodeBlob blob = cache.createCodeBlobWrapper(findStart(ptr));
110         if (blob != null) {
111           visitor.visit(blob);
112           if (blob == lastBlob) {
113             throw new InternalError("saw same blob twice");
114           }
115           lastBlob = blob;
116         }
117       } catch (RuntimeException e) {
118         e.printStackTrace();
119       }
120       Address next = nextBlock(ptr);
121       if (next != null && next.lessThan(ptr)) {
122         throw new InternalError("pointer moved backwards");
123       }
124       ptr = next;
125     }
126   }
127 
128   //--------------------------------------------------------------------------------
129   // Internals only below this point
130   //
131 
132   private VirtualSpace getMemory() {
133     return memory;
134   }
135 
136   private VirtualSpace getSegmentMap() {
137     return segmentMap;
138   }
139 
140   private long segmentFor(Address p) {
141     return p.minus(getMemory().low()) >> getLog2SegmentSize();
142   }
143 
144   private int getLog2SegmentSize() {
145     return log2SegmentSize;
146   }
147 
148   private HeapBlock getBlockAt(Address addr) {
149     return (HeapBlock) VMObjectFactory.newObject(HeapBlock.class, addr);
150   }
151 
152 
153   private HeapBlock blockStart(Address p) {
154     Address base = blockBase(p);
155     if (base == null) return null;
156     return getBlockAt(base);
157   }
158 
159   private Address blockBase(Address p) {
160     long i = segmentFor(p);
161     Address b = getSegmentMap().low();
162     if (b.getCIntegerAt(i, 1, true) == 0xFF) {
163       return null;
164     }
165     while (b.getCIntegerAt(i, 1, true) > 0) {
166       i -= b.getCIntegerAt(i, 1, true);
167     }
168     return getMemory().low().addOffsetTo(i << getLog2SegmentSize());
169   }
