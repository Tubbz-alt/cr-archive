 25 package sun.jvm.hotspot.runtime;
 26 
 27 import java.util.Observer;
 28 import sun.jvm.hotspot.types.TypeDataBase;
 29 
 30 
 31 /** Encapsulates the BasicTypeSize enum in globalDefinitions.hpp in
 32     the VM. */
 33 
 34 public class BasicTypeSize {
 35   private static int tBooleanSize;
 36   private static int tCharSize;
 37   private static int tFloatSize;
 38   private static int tDoubleSize;
 39   private static int tByteSize;
 40   private static int tShortSize;
 41   private static int tIntSize;
 42   private static int tLongSize;
 43   private static int tObjectSize;
 44   private static int tArraySize;
 45   private static int tNarrowOopSize;
 46   private static int tNarrowKlassSize;
 47   private static int tVoidSize;
 48 
 49   static {
 50     VM.registerVMInitializedObserver(
 51         (o, d) -> initialize(VM.getVM().getTypeDataBase()));
 52   }
 53 
 54   private static synchronized void initialize(TypeDataBase db) {
 55     tBooleanSize     = db.lookupIntConstant("T_BOOLEAN_size").intValue();
 56     tCharSize        = db.lookupIntConstant("T_INT_size").intValue();
 57     tFloatSize       = db.lookupIntConstant("T_FLOAT_size").intValue();
 58     tDoubleSize      = db.lookupIntConstant("T_DOUBLE_size").intValue();
 59     tByteSize        = db.lookupIntConstant("T_BYTE_size").intValue();
 60     tShortSize       = db.lookupIntConstant("T_SHORT_size").intValue();
 61     tIntSize         = db.lookupIntConstant("T_INT_size").intValue();
 62     tLongSize        = db.lookupIntConstant("T_LONG_size").intValue();
 63     tObjectSize      = db.lookupIntConstant("T_OBJECT_size").intValue();
 64     tArraySize       = db.lookupIntConstant("T_ARRAY_size").intValue();
 65     tNarrowOopSize   = db.lookupIntConstant("T_NARROWOOP_size").intValue();
 66     tNarrowKlassSize = db.lookupIntConstant("T_NARROWKLASS_size").intValue();
 67     tVoidSize        = db.lookupIntConstant("T_VOID_size").intValue();
 68   }
 69 
 70   public static int getTBooleanSize() {
 71     return tBooleanSize;
 72   }
 73 
 74   public static int getTCharSize() {
 75     return tCharSize;
 76   }
 77 
 78   public static int getTFloatSize() {
 79     return tFloatSize;
 80   }
 81 
 82   public static int getTDoubleSize() {
 83     return tDoubleSize;
 84   }
 85 
 86   public static int getTByteSize() {
 87     return tByteSize;
 88   }
 89 
 90   public static int getTShortSize() {
 91     return tShortSize;
 92   }
 93 
 94   public static int getTIntSize() {
 95     return tIntSize;
 96   }
 97 
 98   public static int getTLongSize() {
 99     return tLongSize;
100   }
101 
102   public static int getTObjectSize() {
103     return tObjectSize;
104   }
105 
106   public static int getTArraySize() {
107     return tArraySize;
108   }
109 
110   public static int getTNarrowOopSize() {
111     return tNarrowOopSize;
112   }
113 
114   public static int getTNarrowKlassSize() {
115     return tNarrowKlassSize;
116   }
117 
118   public static int getTVoidSize() {
119     return tVoidSize;
120   }
121 }
