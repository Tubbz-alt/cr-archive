25 package sun.jvm.hotspot.ci;

27 import java.io.PrintStream;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.oops.*;
32 import sun.jvm.hotspot.types.*;
33 import sun.jvm.hotspot.utilities.Observable;
34 import sun.jvm.hotspot.utilities.Observer;
35 
36 public class ciObjArrayKlass extends ciArrayKlass {
37   static {
38     VM.registerVMInitializedObserver(new Observer() {
39         public void update(Observable o, Object data) {
40           initialize(VM.getVM().getTypeDataBase());
41         }
42       });
43   }
44 
45   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
46     Type type      = db.lookupType("ciObjArrayKlass");
47     elementKlassField = type.getAddressField("_element_klass");
48     baseElementKlassField = type.getAddressField("_base_element_klass");
49   }
50 
51   private static AddressField elementKlassField;
52   private static AddressField baseElementKlassField;
53 
54   public ciObjArrayKlass(Address addr) {
55     super(addr);
56   }
57 }
