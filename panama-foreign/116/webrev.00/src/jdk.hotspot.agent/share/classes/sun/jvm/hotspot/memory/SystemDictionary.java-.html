 25 package sun.jvm.hotspot.memory;
 26 
 27 import java.util.*;
 28 import sun.jvm.hotspot.debugger.*;
 29 import sun.jvm.hotspot.classfile.*;
 30 import sun.jvm.hotspot.oops.*;
 31 import sun.jvm.hotspot.runtime.*;
 32 import sun.jvm.hotspot.types.*;
 33 
 34 public class SystemDictionary {
 35   private static sun.jvm.hotspot.types.OopField javaSystemLoaderField;
 36 
 37   private static AddressField objectKlassField;
 38   private static AddressField classLoaderKlassField;
 39   private static AddressField stringKlassField;
 40   private static AddressField systemKlassField;
 41   private static AddressField threadKlassField;
 42   private static AddressField threadGroupKlassField;
 43   private static AddressField methodHandleKlassField;
 44 
 45   static {
 46     VM.registerVMInitializedObserver(new Observer() {
 47         public void update(Observable o, Object data) {
 48           initialize(VM.getVM().getTypeDataBase());
 49         }
 50       });
 51   }
 52 
 53   private static synchronized void initialize(TypeDataBase db) {
 54     Type type = db.lookupType("SystemDictionary");
 55 
 56     javaSystemLoaderField = type.getOopField("_java_system_loader");
 57 
 58     objectKlassField = type.getAddressField(WK_KLASS("Object_klass"));
 59     classLoaderKlassField = type.getAddressField(WK_KLASS("ClassLoader_klass"));
 60     stringKlassField = type.getAddressField(WK_KLASS("String_klass"));
 61     systemKlassField = type.getAddressField(WK_KLASS("System_klass"));
 62     threadKlassField = type.getAddressField(WK_KLASS("Thread_klass"));
 63     threadGroupKlassField = type.getAddressField(WK_KLASS("ThreadGroup_klass"));
 64     methodHandleKlassField = type.getAddressField(WK_KLASS("MethodHandle_klass"));
 65   }
 66 
 67   // This WK functions must follow the definitions in systemDictionary.hpp:
 68   private static String WK_KLASS(String name) {
 69       //#define WK_KLASS(name) _well_known_klasses[SystemDictionary::WK_KLASS_ENUM_NAME(name)]
 70       return ("_well_known_klasses[SystemDictionary::"+WK_KLASS_ENUM_NAME(name)+"]");
 71   }
 72   private static String WK_KLASS_ENUM_NAME(String kname) {
 73       //#define WK_KLASS_ENUM_NAME(kname)    kname##_knum
 74       return (kname+"_knum");
 75   }
 76 
 77   // few well known classes -- not all are added here.
 78   // add more if needed.
 79   public static InstanceKlass getThreadKlass() {
 80     return (InstanceKlass)Metadata.instantiateWrapperFor(threadKlassField.getValue());
 81   }
 82 
 83   public static InstanceKlass getThreadGroupKlass() {
 84     return (InstanceKlass)Metadata.instantiateWrapperFor(threadGroupKlassField.getValue());
 85   }
 86 
 87   public static InstanceKlass getObjectKlass() {
 88     return (InstanceKlass)Metadata.instantiateWrapperFor(objectKlassField.getValue());
 89   }
 90 
 91   public static InstanceKlass getStringKlass() {
 92     return (InstanceKlass)Metadata.instantiateWrapperFor(stringKlassField.getValue());
 93   }
 94 
 95   public static InstanceKlass getClassLoaderKlass() {
 96     return (InstanceKlass)Metadata.instantiateWrapperFor(classLoaderKlassField.getValue());
 97   }
 98 
 99   public static InstanceKlass getSystemKlass() {
100     return (InstanceKlass)Metadata.instantiateWrapperFor(systemKlassField.getValue());
101   }
102 
103   public static InstanceKlass getMethodHandleKlass() {
104     return (InstanceKlass)Metadata.instantiateWrapperFor(methodHandleKlassField.getValue());
105   }
106 
107   public InstanceKlass getAbstractOwnableSynchronizerKlass() {
108     ClassLoaderDataGraph cldg = VM.getVM().getClassLoaderDataGraph();
109     return (InstanceKlass) cldg.find("java/util/concurrent/locks/AbstractOwnableSynchronizer");
110   }
111 
112   public static Oop javaSystemLoader() {
113     return newOop(javaSystemLoaderField.getValue());
114   }
115 
116   private static Oop newOop(OopHandle handle) {
117     return VM.getVM().getObjectHeap().newOop(handle);
118   }
119 }
