25 package sun.jvm.hotspot.debugger.cdbg;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 
30 // a comparator used to sort LoadObjects by base address
31 
32 public class LoadObjectComparator implements Comparator<LoadObject> {
33    public int compare(LoadObject lo1, LoadObject lo2) {


34       Address base1 = lo1.getBase();
35       Address base2 = lo2.getBase();
36       long diff = base1.minus(base2);
37       return (diff == 0)? 0 : ((diff < 0)? -1 : +1);
38    }
39 
40    public boolean equals(Object o) {
41       if (o == null || !(o instanceof LoadObjectComparator)) {
42          return false;
43       }
44       return true;
45    }
46 }
