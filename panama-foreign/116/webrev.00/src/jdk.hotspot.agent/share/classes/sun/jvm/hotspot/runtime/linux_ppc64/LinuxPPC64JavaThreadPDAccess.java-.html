 25 package sun.jvm.hotspot.runtime.linux_ppc64;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 
 30 import sun.jvm.hotspot.debugger.*;
 31 import sun.jvm.hotspot.debugger.ppc64.*;
 32 import sun.jvm.hotspot.runtime.*;
 33 import sun.jvm.hotspot.runtime.ppc64.*;
 34 import sun.jvm.hotspot.types.*;
 35 import sun.jvm.hotspot.utilities.*;
 36 
 37 public class LinuxPPC64JavaThreadPDAccess implements JavaThreadPDAccess {
 38   private static AddressField  osThreadField;
 39 
 40   // Field from OSThread
 41   private static CIntegerField osThreadThreadIDField;
 42 
 43   // This is currently unneeded but is being kept in case we change
 44   // the currentFrameGuess algorithm
 45   private static final long GUESS_SCAN_RANGE = 128 * 1024;
 46 
 47   static {
 48     VM.registerVMInitializedObserver(new Observer() {
 49         public void update(Observable o, Object data) {
 50           initialize(VM.getVM().getTypeDataBase());
 51         }
 52       });
 53   }
 54 
 55   private static synchronized void initialize(TypeDataBase db) {
 56     Type type = db.lookupType("JavaThread");
 57     osThreadField = type.getAddressField("_osthread");
 58 
 59     Type osThreadType = db.lookupType("OSThread");
 60     osThreadThreadIDField = osThreadType.getCIntegerField("_thread_id");
 61   }
 62 
 63   public Address getLastJavaFP(Address addr) {
 64     return null;
 65   }
 66 
 67   public Address getLastJavaPC(Address addr) {
 68     return null;
 69   }
 70 
 71   public Address getBaseOfStackPointer(Address addr) {
 72     return null;
 73   }
 74 
 75   public Frame getLastFramePD(JavaThread thread, Address addr) {
 76     Address fp = thread.getLastJavaFP();
 77     if (fp == null) {
 78       return null; // no information
 79     }
 80     return new PPC64Frame(thread.getLastJavaSP(), fp);
 81   }
 82 
 83   public RegisterMap newRegisterMap(JavaThread thread, boolean updateMap) {
 84     return new PPC64RegisterMap(thread, updateMap);
 85   }
 86 
 87   public Frame getCurrentFrameGuess(JavaThread thread, Address addr) {
 88     ThreadProxy t = getThreadProxy(addr);
 89     PPC64ThreadContext context = (PPC64ThreadContext) t.getContext();
 90     PPC64CurrentFrameGuess guesser = new PPC64CurrentFrameGuess(context, thread);
 91     if (!guesser.run(GUESS_SCAN_RANGE)) {
 92       return null;
 93     }
 94     if (guesser.getPC() == null) {
 95       return new PPC64Frame(guesser.getSP(), guesser.getFP());
 96     } else {
 97       return new PPC64Frame(guesser.getSP(), guesser.getFP(), guesser.getPC());
 98     }
 99   }
100 
101   public void printThreadIDOn(Address addr, PrintStream tty) {
102     tty.print(getThreadProxy(addr));
103   }
104 
105   public void printInfoOn(Address threadAddr, PrintStream tty) {
106     tty.print("Thread id: ");
107     printThreadIDOn(threadAddr, tty);
108     // tty.println("\nPostJavaState: " + getPostJavaState(threadAddr));
109   }
110 
111   public Address getLastSP(Address addr) {
112     ThreadProxy t = getThreadProxy(addr);
113     PPC64ThreadContext context = (PPC64ThreadContext) t.getContext();
114     return context.getRegisterAsAddress(PPC64ThreadContext.SP);
115   }
116 
117   public Address getLastFP(Address addr) {
118     return getLastSP(addr).getAddressAt(0);
119   }
120 
121   public ThreadProxy getThreadProxy(Address addr) {
122     // Addr is the address of the JavaThread.
123     // Fetch the OSThread (for now and for simplicity, not making a
124     // separate "OSThread" class in this package)
125     Address osThreadAddr = osThreadField.getValue(addr);
126     // Get the address of the _thread_id from the OSThread
127     Address threadIdAddr = osThreadAddr.addOffsetTo(osThreadThreadIDField.getOffset());
128 
129     JVMDebugger debugger = VM.getVM().getDebugger();
130     return debugger.getThreadForIdentifierAddress(threadIdAddr);
131   }
132 }
