25 package sun.jvm.hotspot.utilities;
26 
27 import sun.jvm.hotspot.debugger.Address;
28 import sun.jvm.hotspot.oops.Metadata;
29 import sun.jvm.hotspot.oops.Klass;
30 import sun.jvm.hotspot.runtime.VM;
31 import sun.jvm.hotspot.types.Type;
32 import sun.jvm.hotspot.types.TypeDataBase;
33 import sun.jvm.hotspot.types.WrongTypeException;
34 import sun.jvm.hotspot.utilities.Observable;
35 import sun.jvm.hotspot.utilities.Observer;
36 
37 public class KlassArray extends GenericArray {
38   static {
39     VM.registerVMInitializedObserver(new Observer() {
40       public void update(Observable o, Object data) {
41         initialize(VM.getVM().getTypeDataBase());
42       }
43     });
44   }
45 
46   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
47     elemType = db.lookupType("Klass*");
48 
49     Type type = db.lookupType("Array<Klass*>");
50     dataFieldOffset = type.getAddressField("_data").getOffset();
51   }
52 
53   private static long dataFieldOffset;
54   protected static Type elemType;
55 
56   public KlassArray(Address addr) {
57     super(addr, dataFieldOffset);
58   }
59 
60   public Klass getAt(int i) {
61     return (Klass) Metadata.instantiateWrapperFor(getAddressAt(i));
62   }
63 
64   public Type getElemType() {
65     return elemType;
66   }
67 }
