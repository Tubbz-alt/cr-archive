25 package sun.jvm.hotspot.gc.shared;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 
33 public class GenerationSpec extends VMObject {
34   private static CIntegerField nameField;
35   private static CIntegerField initSizeField;
36   private static CIntegerField maxSizeField;
37 
38   static {
39     VM.registerVMInitializedObserver(new Observer() {
40         public void update(Observable o, Object data) {
41           initialize(VM.getVM().getTypeDataBase());
42         }
43       });
44   }
45 
46   private static synchronized void initialize(TypeDataBase db) {
47     Type type = db.lookupType("GenerationSpec");
48 
49     nameField       = type.getCIntegerField("_name");
50     initSizeField = type.getCIntegerField("_init_size");
51     maxSizeField  = type.getCIntegerField("_max_size");
52   }
53 
54   public GenerationSpec(Address addr) {
55     super(addr);
56   }
57 
58   public Generation.Name name() {
59     return Generation.nameForEnum((int)nameField.getValue(addr));
60   }
61 
62   public long initSize() {
63     return initSizeField.getValue(addr);
64   }
65 
66   public long maxSize() {
67     return maxSizeField.getValue(addr);
68   }
69 }
