 25 package sun.jvm.hotspot.ci;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 import sun.jvm.hotspot.debugger.*;
 30 import sun.jvm.hotspot.code.*;
 31 import sun.jvm.hotspot.runtime.*;
 32 import sun.jvm.hotspot.oops.*;
 33 import sun.jvm.hotspot.types.*;
 34 import sun.jvm.hotspot.utilities.Observable;
 35 import sun.jvm.hotspot.utilities.Observer;
 36 
 37 public class ciMethod extends ciMetadata {
 38   static {
 39     VM.registerVMInitializedObserver(new Observer() {
 40         public void update(Observable o, Object data) {
 41           initialize(VM.getVM().getTypeDataBase());
 42         }
 43       });
 44   }
 45 
 46   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
 47     Type type      = db.lookupType("ciMethod");
 48     interpreterThrowoutCountField = new CIntField(type.getCIntegerField("_interpreter_throwout_count"), 0);
 49     interpreterInvocationCountField = new CIntField(type.getCIntegerField("_interpreter_invocation_count"), 0);
 50     try {
 51         // XXX
 52         instructionsSizeField = new CIntField(type.getCIntegerField("_instructions_size"), 0);
 53     } catch (Exception e) {
 54     }
 55   }
 56 
 57   private static CIntField interpreterThrowoutCountField;
 58   private static CIntField interpreterInvocationCountField;
 59   private static CIntField instructionsSizeField;
 60 
 61   public ciMethod(Address addr) {
 62     super(addr);
 63   }
 64 
 65   public Method method() {
 66     return (Method)getMetadata();
 67   }
 68 
 69   public int interpreterThrowoutCount() {
 70     return (int) interpreterThrowoutCountField.getValue(getAddress());
 71   }
 72 
 73   public int interpreterInvocationCount() {
 74     return (int) interpreterInvocationCountField.getValue(getAddress());
 75   }
 76 
 77   public int instructionsSize() {
 78     if (instructionsSizeField == null) {
 79       // XXX
 80       Method method = method();
 81       NMethod nm = method.getNativeMethod();
 82       if (nm != null) return (int)nm.codeEnd().minus(nm.getVerifiedEntryPoint());
 83       return 0;
 84     }
 85     return (int) instructionsSizeField.getValue(getAddress());
 86   }
 87 
 88   public void printShortName(PrintStream st) {
 89     Method method = method();
 90     st.printf(" %s::%s", method.getMethodHolder().getName().asString().replace('/', '.'),
 91               method.getName().asString());
 92   }
 93 
 94   public void dumpReplayData(PrintStream out) {
 95     Method method = (Method)getMetadata();
 96     NMethod nm = method.getNativeMethod();
 97     out.println("ciMethod " +
 98                 nameAsAscii() + " " +
 99                 method.getInvocationCount() + " " +
100                 method.getBackedgeCount() + " " +
101                 interpreterInvocationCount() + " " +
102                 interpreterThrowoutCount() + " " +
103                 instructionsSize());
104   }
105 
106   public void printValueOn(PrintStream tty) {
107     tty.print("ciMethod " + method().getName().asString() + method().getSignature().asString() + "@" + getAddress());
108   }
109 
110   public String nameAsAscii() {
111     Method method = (Method)getMetadata();
112     return method.nameAsAscii();
113   }
114 }
