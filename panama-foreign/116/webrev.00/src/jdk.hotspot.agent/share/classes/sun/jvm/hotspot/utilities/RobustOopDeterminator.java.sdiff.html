35 /** This class determines to the best of its ability, and in a
36     reasonably robust fashion, whether a given pointer is an intact
37     oop or not. It does this by checking the integrity of the
38     metaclass hierarchy. This is only intended for use in the
39     debugging system. It may provide more resilience to unexpected VM
40     states than the ObjectHeap code. */
41 
42 public class RobustOopDeterminator {
43   private static AddressField klassField;
44 
45   static {
46     VM.registerVMInitializedObserver(new Observer() {
47         public void update(Observable o, Object data) {
48           initialize(VM.getVM().getTypeDataBase());
49         }
50       });
51   }
52 
53   private static void initialize(TypeDataBase db) {

57       klassField = type.getAddressField("_metadata._compressed_klass");
58     } else {
59       klassField = type.getAddressField("_metadata._klass");
60     }
61   }
62 
63   public static boolean oopLooksValid(OopHandle oop) {
64     if (oop == null) {
65       return false;
66     }
67     if (!VM.getVM().getUniverse().isIn(oop)) {
68       return false;
69     }
70     try {
71       // Try to instantiate the Klass
72       if (VM.getVM().isCompressedKlassPointersEnabled()) {
73         Metadata.instantiateWrapperFor(oop.getCompKlassAddressAt(klassField.getOffset()));
74       } else {
75         Metadata.instantiateWrapperFor(klassField.getValue(oop));
76       }
77           return true;
78         }
79     catch (AddressException e) {

80       return false;
81     }
82   }
83 }
