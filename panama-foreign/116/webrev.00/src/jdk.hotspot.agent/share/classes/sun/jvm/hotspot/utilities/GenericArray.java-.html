 25 package sun.jvm.hotspot.utilities;
 26 
 27 import java.util.Observable;
 28 import java.util.Observer;
 29 
 30 import sun.jvm.hotspot.debugger.Address;
 31 import sun.jvm.hotspot.oops.ArrayKlass;
 32 import sun.jvm.hotspot.oops.CIntField;
 33 import sun.jvm.hotspot.oops.Oop;
 34 import sun.jvm.hotspot.runtime.VM;
 35 import sun.jvm.hotspot.runtime.VMObject;
 36 import sun.jvm.hotspot.types.AddressField;
 37 import sun.jvm.hotspot.types.Type;
 38 import sun.jvm.hotspot.types.TypeDataBase;
 39 import sun.jvm.hotspot.types.WrongTypeException;
 40 
 41 /**
 42  * The base class for the mirrors of the Array<T> C++ classes.
 43  */
 44 public abstract class GenericArray extends VMObject {
 45   static {
 46     VM.registerVMInitializedObserver(new Observer() {
 47       public void update(Observable o, Object data) {
 48         initialize(VM.getVM().getTypeDataBase());
 49       }
 50     });
 51   }
 52 
 53   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
 54     // Array<int> is arbitrarily chosen to get the fields in Array<T>.
 55     Type type = db.lookupType("Array<int>");
 56     lengthField = new CIntField(type.getCIntegerField("_length"), 0);
 57   }
 58 
 59   private static long sizeOfArray;
 60   private static CIntField lengthField;
 61 
 62   private long dataFieldOffset;
 63 
 64   public GenericArray(Address addr, long dataOffset) {
 65     super(addr);
 66     dataFieldOffset = dataOffset;
 67   }
 68 
 69   public int length() {
 70     return (int)lengthField.getValue(this);
 71   }
 72 
 73   // for compatibility with TypeArray
 74   public int getLength() {
 75     return length();
 76   }
 77 
 78   /**
 79    * Gets the element at the given index.
 80    */
 81   protected long getIntegerAt(int index) {
 82     if (index < 0 || index >= length()) throw new ArrayIndexOutOfBoundsException(index + " " + length());
 83 
 84     Type elemType = getElemType();
 85     if (!getElemType().isCIntegerType()) throw new RuntimeException("elemType must be of CInteger type");
 86 
 87     Address data = getAddress().addOffsetTo(dataFieldOffset);
 88     long elemSize = elemType.getSize();
 89 
 90     return data.getCIntegerAt(index * elemSize, elemSize, false);
 91   }
 92 
 93   protected Address getAddressAt(int index) {
 94     if (index < 0 || index >= length()) throw new ArrayIndexOutOfBoundsException(index);
 95 
 96     Type elemType = getElemType();
 97     if (getElemType().isCIntegerType()) throw new RuntimeException("elemType must not be of CInteger type");
 98 
 99     Address data = getAddress().addOffsetTo(dataFieldOffset);
100     long elemSize = elemType.getSize();
101 
102     return data.getAddressAt(index * elemSize);
103   }
104 
105   private long byteSizeof(int length) { return sizeOfArray + length * getElemType().getSize(); }
106 
107   public long getSize() {
108     return VM.getVM().alignUp(byteSizeof(length()), VM.getVM().getBytesPerWord()) / VM.getVM().getBytesPerWord();
109   }
110 
111   /**
112    * The element type of this array.
113    */
114   public abstract Type getElemType();
115 }
