 25 package sun.jvm.hotspot.oops;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 import sun.jvm.hotspot.debugger.*;
 30 import sun.jvm.hotspot.runtime.*;
 31 import sun.jvm.hotspot.types.*;
 32 import sun.jvm.hotspot.utilities.Observable;
 33 import sun.jvm.hotspot.utilities.Observer;
 34 
 35 // A TypeArray is an array containing basic types (non oop elements).
 36 // It is used for arrays of {characters, singles, doubles, bytes, shorts, integers, longs}
 37 
 38 public class TypeArray extends Array {
 39   static {
 40     VM.registerVMInitializedObserver(new Observer() {
 41         public void update(Observable o, Object data) {
 42           initialize(VM.getVM().getTypeDataBase());
 43         }
 44       });
 45   }
 46 
 47   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
 48     Type type = db.lookupType("typeArrayOop");
 49   }
 50 
 51   TypeArray(OopHandle handle, ObjectHeap heap) {
 52     super(handle, heap);
 53   }
 54 
 55   public boolean isTypeArray()         { return true; }
 56 
 57   public byte getByteAt(long index) {
 58     if (index < 0 || index >= getLength()) {
 59       throw new ArrayIndexOutOfBoundsException(index + " " + getLength());
 60     }
 61     long offset = baseOffsetInBytes(BasicType.T_BYTE) + index * getHeap().getByteSize();
 62     return getHandle().getJByteAt(offset);
 63   }
 64 
 65   public boolean getBooleanAt(long index) {
 66     long offset = baseOffsetInBytes(BasicType.T_BOOLEAN) + index * getHeap().getBooleanSize();
 67     return getHandle().getJBooleanAt(offset);
 68   }
 69 
 70   public char getCharAt(long index) {
 71     long offset = baseOffsetInBytes(BasicType.T_CHAR) + index * getHeap().getCharSize();
 72     return getHandle().getJCharAt(offset);
 73   }
 74 
 75   public int getIntAt(long index) {
 76     long offset = baseOffsetInBytes(BasicType.T_INT) + index * getHeap().getIntSize();
 77     return getHandle().getJIntAt(offset);
 78   }
 79 
 80   public short getShortAt(long index) {
 81     long offset = baseOffsetInBytes(BasicType.T_SHORT) + index * getHeap().getShortSize();
 82     return getHandle().getJShortAt(offset);
 83   }
 84 
 85   public long getLongAt(long index) {
 86     long offset = baseOffsetInBytes(BasicType.T_LONG) + index * getHeap().getLongSize();
 87     return getHandle().getJLongAt(offset);
 88   }
 89 
 90   public float getFloatAt(long index) {
 91     long offset = baseOffsetInBytes(BasicType.T_FLOAT) + index * getHeap().getFloatSize();
 92     return getHandle().getJFloatAt(offset);
 93   }
 94 
 95   public double getDoubleAt(long index) {
 96     long offset = baseOffsetInBytes(BasicType.T_DOUBLE) + index * getHeap().getDoubleSize();
 97     return getHandle().getJDoubleAt(offset);
 98   }
 99 
100   public void printValueOn(PrintStream tty) {
101     TypeArrayKlass klass = (TypeArrayKlass) getKlass();
102     tty.print(klass.getTypeName());
103   }
104 
105   public void iterateFields(OopVisitor visitor, boolean doVMFields) {
106     super.iterateFields(visitor, doVMFields);
107     TypeArrayKlass klass = (TypeArrayKlass) getKlass();
108     int length = (int) getLength();
109     int type   = (int) klass.getElementType();
110     for (int index = 0; index < length; index++) {
111       FieldIdentifier id = new IndexableFieldIdentifier(index);
112       switch (type) {
113       case TypeArrayKlass.T_BOOLEAN: {
114         long offset = baseOffsetInBytes(BasicType.T_BOOLEAN) + index * getHeap().getBooleanSize();
115         visitor.doBoolean(new BooleanField(id, offset, false), false);
116         break;
117       }
118       case TypeArrayKlass.T_CHAR: {
119         long offset = baseOffsetInBytes(BasicType.T_CHAR) + index * getHeap().getCharSize();
120         visitor.doChar(new CharField(id, offset, false), false);
121         break;
122       }
123       case TypeArrayKlass.T_FLOAT: {
124         long offset = baseOffsetInBytes(BasicType.T_FLOAT) + index * getHeap().getFloatSize();
125         visitor.doFloat(new FloatField(id, offset, false), false);
126         break;
127       }
128       case TypeArrayKlass.T_DOUBLE: {
129         long offset = baseOffsetInBytes(BasicType.T_DOUBLE) + index * getHeap().getDoubleSize();
130         visitor.doDouble(new DoubleField(id, offset, false), false);
131         break;
132       }
133       case TypeArrayKlass.T_BYTE: {
134         long offset = baseOffsetInBytes(BasicType.T_BYTE) + index * getHeap().getByteSize();
135         visitor.doByte(new ByteField(id, offset, false), false);
136         break;
137       }
138       case TypeArrayKlass.T_SHORT: {
139         long offset = baseOffsetInBytes(BasicType.T_SHORT) + index * getHeap().getShortSize();
140         visitor.doShort(new ShortField(id, offset, false), false);
141         break;
142       }
143       case TypeArrayKlass.T_INT: {
144         long offset = baseOffsetInBytes(BasicType.T_INT) + index * getHeap().getIntSize();
145         visitor.doInt(new IntField(id, offset, false), false);
146         break;
147       }
148       case TypeArrayKlass.T_LONG: {
149         long offset = baseOffsetInBytes(BasicType.T_LONG) + index * getHeap().getLongSize();
150         visitor.doLong(new LongField(id, offset, false), false);
151         break;
152       }
153       }
154     }
155   }
156 }
