25 package sun.jvm.hotspot.oops;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.*;
33 
34 public class MethodCounters extends Metadata {
35   public MethodCounters(Address addr) {
36     super(addr);
37   }
38 
39   static {
40     VM.registerVMInitializedObserver(new Observer() {
41         public void update(Observable o, Object data) {
42           initialize(VM.getVM().getTypeDataBase());
43         }
44       });
45   }
46 
47   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
48     Type type      = db.lookupType("MethodCounters");
49 
50     if (VM.getVM().isServerCompiler()) {
51       interpreterInvocationCountField = new CIntField(type.getCIntegerField("_interpreter_invocation_count"), 0);
52       interpreterThrowoutCountField = new CIntField(type.getCIntegerField("_interpreter_throwout_count"), 0);
53     }
54     if (!VM.getVM().isCore()) {
55       invocationCounter        = new CIntField(type.getCIntegerField("_invocation_counter"), 0);
56       backedgeCounter          = new CIntField(type.getCIntegerField("_backedge_counter"), 0);
57     }
58   }
59 
60   private static CIntField interpreterInvocationCountField;
61   private static CIntField interpreterThrowoutCountField;
62   private static CIntField invocationCounter;
63   private static CIntField backedgeCounter;
64 
65   public int interpreterInvocationCount() {
66       if (interpreterInvocationCountField != null) {
67         return (int) interpreterInvocationCountField.getValue(this);
68       } else {
69         return 0;
70       }
71   }
72 
73   public int interpreterThrowoutCount() {
74       if (interpreterThrowoutCountField != null) {
75         return (int) interpreterThrowoutCountField.getValue(this);
76       } else {
77         return 0;
78       }
79   }
80   public long getInvocationCounter() {
81     if (Assert.ASSERTS_ENABLED) {
82       Assert.that(!VM.getVM().isCore(), "must not be used in core build");
83     }
84     return invocationCounter.getValue(this);
85   }
86   public long getBackedgeCounter() {
87     if (Assert.ASSERTS_ENABLED) {
88       Assert.that(!VM.getVM().isCore(), "must not be used in core build");
89     }
90     return backedgeCounter.getValue(this);
91   }
92 
93   public void printValueOn(PrintStream tty) {
94   }
95 }
