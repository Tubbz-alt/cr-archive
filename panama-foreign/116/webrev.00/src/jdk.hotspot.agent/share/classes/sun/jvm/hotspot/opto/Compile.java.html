 25 package sun.jvm.hotspot.opto;
 26 
 27 import java.util.*;
 28 import java.io.PrintStream;
 29 import sun.jvm.hotspot.ci.*;
 30 import sun.jvm.hotspot.debugger.*;
 31 import sun.jvm.hotspot.runtime.*;
 32 import sun.jvm.hotspot.oops.*;
 33 import sun.jvm.hotspot.types.*;
 34 import sun.jvm.hotspot.utilities.Observable;
 35 import sun.jvm.hotspot.utilities.Observer;
 36 
 37 public class Compile extends VMObject {
 38   static {
 39     VM.registerVMInitializedObserver(new Observer() {
 40         public void update(Observable o, Object data) {
 41           initialize(VM.getVM().getTypeDataBase());
 42         }
 43       });
 44   }
 45 
 46   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
 47     Type type      = db.lookupType("Compile");
 48     rootField = type.getAddressField("_root");
 49     uniqueField = new CIntField(type.getCIntegerField("_unique"), 0);
 50     entryBciField = new CIntField(type.getCIntegerField("_entry_bci"), 0);
 51     topField = type.getAddressField("_top");
 52     cfgField = type.getAddressField("_cfg");
 53     regallocField = type.getAddressField("_regalloc");
 54     methodField = type.getAddressField("_method");
 55     iltField = type.getAddressField("_ilt");
 56   }
 57 
 58   private static AddressField rootField;
 59   private static CIntField uniqueField;
 60   private static CIntField entryBciField;
 61   private static AddressField topField;
 62   private static AddressField cfgField;
 63   private static AddressField regallocField;
 64   private static AddressField methodField;
 65   private static AddressField iltField;
 66 
 67   public Compile(Address addr) {
 68     super(addr);
 69   }
 70 
 71   public Node root() {
 72     return new RootNode(rootField.getValue(this.getAddress()));
 73   }
 74 
 75   public int entryBci() {
 76     return (int)entryBciField.getValue(getAddress());
 77   }
 78 
 79   public ciMethod method() {
 80     return (ciMethod) ciObjectFactory.getMetadata(methodField.getValue(getAddress()));
 81   }
 82 
 83   public PhaseCFG cfg() {
 84     Address a = cfgField.getValue(this.getAddress());
 85     if (a != null) {
 86       return new PhaseCFG(a);
 87     }
 88     return null;
 89   }
 90 
 91   public InlineTree ilt() {
 92     Address a = iltField.getValue(this.getAddress());
 93     if (a != null) {
 94       return new InlineTree(a);
 95     }
 96     return null;
 97   }
 98 
 99   public void dumpInlineData(PrintStream out) {
100     InlineTree inlTree = ilt();
101     if (inlTree != null) {
102       out.print(" inline " + inlTree.count());
103       inlTree.dumpReplayData(out);
104     }
105   }
106 
107 }
