25 package sun.jvm.hotspot.gc.shared;
26 
27 import java.io.*;
28 import java.util.*;
29 
30 import sun.jvm.hotspot.debugger.*;
31 import sun.jvm.hotspot.memory.*;
32 import sun.jvm.hotspot.runtime.*;
33 import sun.jvm.hotspot.types.*;
34 
35 public class ContiguousSpace extends CompactibleSpace implements LiveRegionsProvider {
36   private static AddressField topField;
37 
38   static {
39     VM.registerVMInitializedObserver(new Observer() {
40         public void update(Observable o, Object data) {
41           initialize(VM.getVM().getTypeDataBase());
42         }
43       });
44   }
45 
46   private static synchronized void initialize(TypeDataBase db) {
47     Type type = db.lookupType("ContiguousSpace");
48 
49     topField = type.getAddressField("_top");
50   }
51 
52   public ContiguousSpace(Address addr) {
53     super(addr);
54   }
55 
56   public Address top() {
57     return topField.getValue(addr);
58   }
59 
60   /** In bytes */
61   public long capacity() {
62     return end().minus(bottom());
63   }
64 
65   /** In bytes */
66   public long used() {
67     return top().minus(bottom());
68   }
69 
70   /** In bytes */
71   public long free() {
72     return end().minus(top());
73   }
74 
75   /** In a contiguous space we have a more obvious bound on what parts
76       contain objects. */
77   public MemRegion usedRegion() {
78     return new MemRegion(bottom(), top());
79   }
80 
81   /** Returns regions of Space where live objects live */
82   public List<MemRegion> getLiveRegions() {
83     List<MemRegion> res = new ArrayList<>();
84     res.add(new MemRegion(bottom(), top()));
85     return res;
86   }
87 
88   /** Testers */
89   public boolean contains(Address p) {
90     return (bottom().lessThanOrEqual(p) && top().greaterThan(p));
91   }
92 
93   public void printOn(PrintStream tty) {
94     tty.print(" [" + bottom() + "," +
95                 top() + "," + end() + ")");
96     super.printOn(tty);
97   }
98 }
