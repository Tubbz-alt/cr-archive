 25 package sun.jvm.hotspot.gc.parallel;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 
 30 import sun.jvm.hotspot.debugger.*;
 31 import sun.jvm.hotspot.memory.*;
 32 import sun.jvm.hotspot.runtime.*;
 33 import sun.jvm.hotspot.types.*;
 34 
 35 public class PSYoungGen extends VMObject {
 36    static {
 37       VM.registerVMInitializedObserver(new Observer() {
 38          public void update(Observable o, Object data) {
 39             initialize(VM.getVM().getTypeDataBase());
 40          }
 41       });
 42    }
 43 
 44    private static synchronized void initialize(TypeDataBase db) {
 45       Type type = db.lookupType("PSYoungGen");
 46       edenSpaceField = type.getAddressField("_eden_space");
 47       fromSpaceField = type.getAddressField("_from_space");
 48       toSpaceField = type.getAddressField("_to_space");
 49    }
 50 
 51    public PSYoungGen(Address addr) {
 52       super(addr);
 53    }
 54 
 55    // Fields
 56    private static AddressField edenSpaceField;
 57    private static AddressField fromSpaceField;
 58    private static AddressField toSpaceField;
 59 
 60    // Accessors
 61    public MutableSpace edenSpace() {
 62       return (MutableSpace) VMObjectFactory.newObject(MutableSpace.class, edenSpaceField.getValue(addr));
 63    }
 64 
 65    public MutableSpace fromSpace() {
 66       return (MutableSpace) VMObjectFactory.newObject(MutableSpace.class, fromSpaceField.getValue(addr));
 67    }
 68 
 69    public MutableSpace toSpace() {
 70       return (MutableSpace) VMObjectFactory.newObject(MutableSpace.class, toSpaceField.getValue(addr));
 71    }
 72 
 73    public long capacity() {
 74       return edenSpace().capacity() + fromSpace().capacity();
 75    }
 76 
 77    public long used() {
 78       return edenSpace().used() + fromSpace().used();
 79    }
 80 
 81    public boolean isIn(Address a) {
 82       if (edenSpace().contains(a)) {
 83          return true;
 84       }
 85 
 86       if (fromSpace().contains(a)) {
 87          return true;
 88       }
 89       return false;
 90    }
 91 
 92    public void printOn(PrintStream tty) {
 93       tty.print("PSYoungGen [ ");
 94       tty.print("eden = ");
 95       edenSpace().printOn(tty);
 96       tty.print(", from = ");
 97       fromSpace().printOn(tty);
 98       tty.print(", to = ");
 99       toSpace().printOn(tty);
100       tty.print(" ] ");
101    }
102 }
