25 package sun.jvm.hotspot.ci;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.oops.*;
32 import sun.jvm.hotspot.types.*;
33 
34 public class ciMetadata extends ciBaseObject {
35   static {
36     VM.registerVMInitializedObserver(new Observer() {
37         public void update(Observable o, Object data) {
38           initialize(VM.getVM().getTypeDataBase());
39         }
40       });
41   }
42 
43   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
44     Type type      = db.lookupType("ciMetadata");
45     metadataField = new MetadataField(type.getAddressField("_metadata"), 0);
46   }
47 
48   private static MetadataField metadataField;
49 
50   public Metadata getMetadata() {
51     return metadataField.getValue(getAddress());
52   }
53 
54   public ciMetadata(Address addr) {
55     super(addr);
56   }
57 
58   public void printOn(PrintStream out) {
59     getMetadata().printValueOn(out);
60   }
61 
62   public String toString() {
63     return getMetadata().toString();
64   }
65 }
