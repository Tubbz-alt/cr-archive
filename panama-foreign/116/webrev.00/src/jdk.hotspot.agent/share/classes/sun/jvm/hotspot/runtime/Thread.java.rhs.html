 25 package sun.jvm.hotspot.runtime;
 26 
 27 import java.util.*;
 28 import sun.jvm.hotspot.debugger.*;
 29 import sun.jvm.hotspot.types.*;
 30 import sun.jvm.hotspot.utilities.Observable;
 31 import sun.jvm.hotspot.utilities.Observer;
 32 
 33 public class Thread extends VMObject {
 34   private static long tlabFieldOffset;
 35 
 36   private static CIntegerField suspendFlagsField;
 37   // Thread::SuspendFlags enum constants
 38   private static int EXTERNAL_SUSPEND;
 39   private static int EXT_SUSPENDED;
 40   private static int HAS_ASYNC_EXCEPTION;
 41 
 42   private static AddressField activeHandlesField;
 43   private static AddressField currentPendingMonitorField;
 44   private static AddressField currentWaitingMonitorField;
 45 
 46   private static JLongField allocatedBytesField;
 47 
 48   static {
 49     VM.registerVMInitializedObserver(new Observer() {
 50         public void update(Observable o, Object data) {
 51           initialize(VM.getVM().getTypeDataBase());
 52         }
 53       });
 54   }
 55 
 56   private static synchronized void initialize(TypeDataBase db) {
 57     Type type = db.lookupType("Thread");
 58 
 59     suspendFlagsField = type.getCIntegerField("_suspend_flags");
 60     EXTERNAL_SUSPEND = db.lookupIntConstant("Thread::_external_suspend").intValue();
 61     EXT_SUSPENDED = db.lookupIntConstant("Thread::_ext_suspended").intValue();
 62     HAS_ASYNC_EXCEPTION = db.lookupIntConstant("Thread::_has_async_exception").intValue();
 63 
 64     tlabFieldOffset    = type.getField("_tlab").getOffset();
 65     activeHandlesField = type.getAddressField("_active_handles");
 66     currentPendingMonitorField = type.getAddressField("_current_pending_monitor");
 67     currentWaitingMonitorField = type.getAddressField("_current_waiting_monitor");
 68     allocatedBytesField = type.getJLongField("_allocated_bytes");
 69   }
 70 
 71   public Thread(Address addr) {
 72     super(addr);
 73   }
 74 
 75   public int suspendFlags() {
 76     return (int) suspendFlagsField.getValue(addr);
 77   }
 78 
 79   public boolean isExternalSuspend() {
 80     return (suspendFlags() & EXTERNAL_SUSPEND) != 0;
 81   }
 82 
 83   public boolean isExtSuspended() {
 84     return (suspendFlags() & EXT_SUSPENDED) != 0;
 85   }
 86 
 87   public boolean isBeingExtSuspended() {
 88     return isExtSuspended() || isExternalSuspend();
 89   }
 90 
 91   // historical usage: checked for VM or external suspension
 92   public boolean isAnySuspended() {
 93     return isExtSuspended();
 94   }
 95 
 96   public boolean hasAsyncException() {
 97     return (suspendFlags() & HAS_ASYNC_EXCEPTION) != 0;
 98   }
 99 
100   public ThreadLocalAllocBuffer tlab() {
101     return new ThreadLocalAllocBuffer(addr.addOffsetTo(tlabFieldOffset));
102   }
103 
104   public JNIHandleBlock activeHandles() {
105     Address a = activeHandlesField.getAddress(addr);
106     if (a == null) {
107       return null;
108     }
109     return new JNIHandleBlock(a);
110   }
111 
112   public long allocatedBytes() {
113     return allocatedBytesField.getValue(addr);
114   }
115 
116   public boolean   isVMThread()                  { return false; }
117   public boolean   isJavaThread()                { return false; }
118   public boolean   isCompilerThread()            { return false; }
119   public boolean   isCodeCacheSweeperThread()    { return false; }
120   public boolean   isHiddenFromExternalView()    { return false; }
121   public boolean   isJvmtiAgentThread()          { return false; }
122   public boolean   isWatcherThread()             { return false; }
123   public boolean   isServiceThread()             { return false; }
124 
125   /** Memory operations */
126   public void oopsDo(AddressVisitor oopVisitor) {
127     // FIXME: Empty for now; will later traverse JNI handles and
128     // pending exception
129   }
130 
131   public ObjectMonitor getCurrentPendingMonitor() {
132     Address monitorAddr = currentPendingMonitorField.getValue(addr);
133     if (monitorAddr == null) {
134       return null;
135     }
136     return new ObjectMonitor(monitorAddr);
137   }
138 
139   public ObjectMonitor getCurrentWaitingMonitor() {
140     Address monitorAddr = currentWaitingMonitorField.getValue(addr);
141     if (monitorAddr == null) {
142       return null;
143     }
144     return new ObjectMonitor(monitorAddr);
145   }
146 
147   public boolean isLockOwned(Address lock) {
148     if (isInStack(lock)) return true;
149     return false;
150   }
151 
152   public boolean isInStack(Address a) {
153     // In the Serviceability Agent we need access to the thread's
154     // stack pointer to be able to answer this question. Since it is
155     // only a debugging system at the moment we need access to the
156     // underlying thread, which is only present for Java threads; see
157     // JavaThread.java.
158     return false;
159   }
160 
161   /** Assistance for ObjectMonitor implementation */
162   Address threadObjectAddress() { return addr; }
163 }
