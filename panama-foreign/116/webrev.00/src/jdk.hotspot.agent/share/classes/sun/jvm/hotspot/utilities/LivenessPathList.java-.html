25 package sun.jvm.hotspot.utilities;
26 
27 import java.util.*;
28 
29 /** Simple container class for LivenessPaths */
30 
31 public class LivenessPathList {
32   public LivenessPathList() {
33     list = new ArrayList();
34   }
35 
36   public int size() {
37     return list.size();
38   }
39 
40   public LivenessPath get(int i) {
41     return (LivenessPath) list.get(i);
42   }
43 
44   void add(LivenessPath path) {
45     list.add(path);
46   }
47 
48   void remove(LivenessPath path) {
49     list.remove(path);
50   }
51 
52   private ArrayList list;
53 }
