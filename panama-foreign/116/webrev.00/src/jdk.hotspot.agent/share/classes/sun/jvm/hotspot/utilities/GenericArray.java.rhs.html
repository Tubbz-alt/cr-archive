 25 package sun.jvm.hotspot.utilities;
 26 



 27 import sun.jvm.hotspot.debugger.Address;
 28 import sun.jvm.hotspot.oops.ArrayKlass;
 29 import sun.jvm.hotspot.oops.CIntField;
 30 import sun.jvm.hotspot.oops.Oop;
 31 import sun.jvm.hotspot.runtime.VM;
 32 import sun.jvm.hotspot.runtime.VMObject;
 33 import sun.jvm.hotspot.types.AddressField;
 34 import sun.jvm.hotspot.types.Type;
 35 import sun.jvm.hotspot.types.TypeDataBase;
 36 import sun.jvm.hotspot.types.WrongTypeException;
 37 import sun.jvm.hotspot.utilities.Observable;
 38 import sun.jvm.hotspot.utilities.Observer;
 39 
 40 /**
 41  * The base class for the mirrors of the Array<T> C++ classes.
 42  */
 43 public abstract class GenericArray extends VMObject {
 44   static {
 45     VM.registerVMInitializedObserver(new Observer() {
 46       public void update(Observable o, Object data) {
 47         initialize(VM.getVM().getTypeDataBase());
 48       }
 49     });
 50   }
 51 
 52   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
 53     // Array<int> is arbitrarily chosen to get the fields in Array<T>.
 54     Type type = db.lookupType("Array<int>");
 55     lengthField = new CIntField(type.getCIntegerField("_length"), 0);
 56   }
 57 
 58   private static long sizeOfArray;
 59   private static CIntField lengthField;
 60 
 61   private long dataFieldOffset;
 62 
 63   public GenericArray(Address addr, long dataOffset) {
 64     super(addr);
 65     dataFieldOffset = dataOffset;
 66   }
 67 
 68   public int length() {
 69     return (int)lengthField.getValue(this);
 70   }
 71 
 72   // for compatibility with TypeArray
 73   public int getLength() {
 74     return length();
 75   }
 76 
 77   /**
 78    * Gets the element at the given index.
 79    */
 80   protected long getIntegerAt(int index) {
 81     if (index < 0 || index >= length()) throw new ArrayIndexOutOfBoundsException(index + " " + length());
 82 
 83     Type elemType = getElemType();
 84     if (!getElemType().isCIntegerType()) throw new RuntimeException("elemType must be of CInteger type");
 85 
 86     Address data = getAddress().addOffsetTo(dataFieldOffset);
 87     long elemSize = elemType.getSize();
 88 
 89     return data.getCIntegerAt(index * elemSize, elemSize, false);
 90   }
 91 
 92   protected Address getAddressAt(int index) {
 93     if (index < 0 || index >= length()) throw new ArrayIndexOutOfBoundsException(index);
 94 
 95     Type elemType = getElemType();
 96     if (getElemType().isCIntegerType()) throw new RuntimeException("elemType must not be of CInteger type");
 97 
 98     Address data = getAddress().addOffsetTo(dataFieldOffset);
 99     long elemSize = elemType.getSize();
100 
101     return data.getAddressAt(index * elemSize);
102   }
103 
104   private long byteSizeof(int length) { return sizeOfArray + length * getElemType().getSize(); }
105 
106   public long getSize() {
107     return VM.getVM().alignUp(byteSizeof(length()), VM.getVM().getBytesPerWord()) / VM.getVM().getBytesPerWord();
108   }
109 
110   /**
111    * The element type of this array.
112    */
113   public abstract Type getElemType();
114 }
