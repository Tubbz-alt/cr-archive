25 package sun.jvm.hotspot.oops;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.Observable;
33 import sun.jvm.hotspot.utilities.Observer;
34 
35 public class BreakpointInfo extends VMObject {
36   private static CIntegerField origBytecodeField;
37   private static CIntegerField bciField;
38   private static CIntegerField nameIndexField;
39   private static CIntegerField signatureIndexField;
40   private static AddressField  nextField;
41 
42   static {
43     VM.registerVMInitializedObserver(new Observer() {
44         public void update(Observable o, Object data) {
45           initialize(VM.getVM().getTypeDataBase());
46         }
47       });
48   }
49 
50   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
51     if (!VM.getVM().isJvmtiSupported()) {
52       // no BreakpointInfo support without JVMTI
53       return;
54     }
55 
56     Type type                  = db.lookupType("BreakpointInfo");
57 
58     origBytecodeField   = type.getCIntegerField("_orig_bytecode");
59     bciField            = type.getCIntegerField("_bci");
60     nameIndexField      = type.getCIntegerField("_name_index");
61     signatureIndexField = type.getCIntegerField("_signature_index");
62     nextField           = type.getAddressField ("_next");
63   }
64 
65   public BreakpointInfo(Address addr) {
66     super(addr);
67   }
68 
69   public int  getOrigBytecode()   { return (int) origBytecodeField.getValue(addr);   }
70   public int  getBCI()            { return (int) bciField.getValue(addr);            }
71   public long getNameIndex()      { return nameIndexField.getValue(addr);            }
72   public long getSignatureIndex() { return signatureIndexField.getValue(addr);       }
73   public BreakpointInfo getNext() {
74     return (BreakpointInfo) VMObjectFactory.newObject(BreakpointInfo.class, nextField.getValue(addr));
75   }
76 
77   public boolean match(Method m, int bci) {
78     return (bci == getBCI() && match(m));
79   }
80 
81   public boolean match(Method m) {
82     return (getNameIndex() == m.getNameIndex() &&
83             getSignatureIndex() == m.getSignatureIndex());
84   }
85 }
