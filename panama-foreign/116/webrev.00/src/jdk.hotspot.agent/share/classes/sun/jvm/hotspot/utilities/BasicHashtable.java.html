25 package sun.jvm.hotspot.utilities;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 import sun.jvm.hotspot.types.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.utilities.Observable;
32 import sun.jvm.hotspot.utilities.Observer;
33 
34 // Superclass for symbol and string tables.
35 
36 public class BasicHashtable extends VMObject {
37   static {
38     VM.registerVMInitializedObserver(new Observer() {
39         public void update(Observable o, Object data) {
40           initialize(VM.getVM().getTypeDataBase());
41         }
42       });
43   }
44 
45   private static synchronized void initialize(TypeDataBase db) {
46     Type type = db.lookupType("BasicHashtable<mtInternal>");
47     tableSizeField = type.getCIntegerField("_table_size");
48     bucketsField   = type.getAddressField("_buckets");
49     bucketSize = db.lookupType("HashtableBucket<mtInternal>").getSize();
50   }
51 
52   // Fields
53   private static CIntegerField tableSizeField;
54   private static AddressField  bucketsField;
55   private static long bucketSize;
56 
57   // Accessors
58   protected int tableSize() {
59     return (int) tableSizeField.getValue(addr);
60   }
61 
62   protected BasicHashtableEntry bucket(int i) {
63     if (Assert.ASSERTS_ENABLED) {
64        Assert.that(i >= 0 && i < tableSize(), "Invalid bucket id");
65     }
66     Address tmp = bucketsField.getValue(addr);
67     tmp = tmp.addOffsetTo(i * bucketSize);
68     HashtableBucket bucket = VMObjectFactory.newObject(
69                                               HashtableBucket.class, tmp);
70     return bucket.getEntry(getHashtableEntryClass());
71   }
72 
73   // derived class may return Class<? extends BasicHashtableEntry>
74   protected Class<? extends BasicHashtableEntry> getHashtableEntryClass() {
75     return BasicHashtableEntry.class;
76   }
77 
78   public BasicHashtable(Address addr) {
79     super(addr);
80   }
81 }
