25 package sun.jvm.hotspot.opto;
26 
27 import java.util.*;
28 import java.io.PrintStream;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.oops.*;
32 import sun.jvm.hotspot.types.*;
33 import sun.jvm.hotspot.utilities.Observable;
34 import sun.jvm.hotspot.utilities.Observer;
35 
36 public class MachIfNode extends MachNode {
37   static {
38     VM.registerVMInitializedObserver(new Observer() {
39         public void update(Observable o, Object data) {
40           initialize(VM.getVM().getTypeDataBase());
41         }
42       });
43   }
44 
45   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
46     Type type      = db.lookupType("MachIfNode");
47     probField = type.getJFloatField("_prob");
48     fcntField = type.getJFloatField("_fcnt");
49   }
50 
51   private static JFloatField probField;
52   private static JFloatField fcntField;
53 
54   float prob() {
55     return probField.getValue(getAddress());
56   }
57 
58   float cnt() {
59     return fcntField.getValue(getAddress());
60   }
61 
62   public MachIfNode(Address addr) {
63     super(addr);
64   }
65 
66   public void dumpSpec(PrintStream out) {
67     out.print("P=" + prob() + ", C=" + cnt());
68   }
69 }
