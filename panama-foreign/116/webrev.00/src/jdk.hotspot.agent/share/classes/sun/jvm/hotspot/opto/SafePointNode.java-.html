25 package sun.jvm.hotspot.opto;
26 
27 import java.io.PrintStream;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.oops.*;
32 import sun.jvm.hotspot.types.*;
33 
34 public class SafePointNode extends MultiNode {
35   static {
36     VM.registerVMInitializedObserver(new Observer() {
37         public void update(Observable o, Object data) {
38           initialize(VM.getVM().getTypeDataBase());
39         }
40       });
41   }
42 
43   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
44     Type type      = db.lookupType("SafePointNode");
45     jvmsField = type.getAddressField("_jvms");
46   }
47 
48   private static AddressField jvmsField;
49 
50   public SafePointNode(Address addr) {
51     super(addr);
52   }
53 
54   public JVMState jvms() {
55     return JVMState.create(jvmsField.getValue(getAddress()));
56   }
57 
58   public void dumpSpec(PrintStream out) {
59     JVMState jvms = jvms();
60     if (jvms != null) out.print(" !");
61     while (jvms != null) {
62       Method m = jvms.method().method();
63       int bci = jvms.bci();
64       out.print(" " + m.getMethodHolder().getName().asString().replace('/', '.') + "::" + m.getName().asString() + " @ bci:" + bci);
65       jvms = jvms.caller();
66     }
67   }
68 }
