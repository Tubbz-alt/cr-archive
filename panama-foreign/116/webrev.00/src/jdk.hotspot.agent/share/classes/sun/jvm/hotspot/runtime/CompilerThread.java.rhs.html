25 package sun.jvm.hotspot.runtime;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.types.*;
31 import sun.jvm.hotspot.ci.*;
32 import sun.jvm.hotspot.utilities.Observable;
33 import sun.jvm.hotspot.utilities.Observer;
34 
35 public class CompilerThread extends JavaThread {
36   static {
37     VM.registerVMInitializedObserver(new Observer() {
38         public void update(Observable o, Object data) {
39           initialize(VM.getVM().getTypeDataBase());
40         }
41       });
42   }
43 
44   private static AddressField envField;
45 
46   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
47     Type type      = db.lookupType("CompilerThread");
48 
49     envField = type.getAddressField("_env");
50   }
51 
52   private ciEnv _env;
53 
54   public synchronized ciEnv env() {
55     if (_env == null) {
56       Address v = envField.getValue(this.getAddress());
57       if (v != null) {
58         _env = new ciEnv(v);
59       }
60     }
61     return _env;
62   }
63 
64   public CompilerThread(Address addr) {
65     super(addr);
66   }
67 
68   public boolean isJavaThread() { return false; }
69   public boolean isHiddenFromExternalView() { return true; }
70   public boolean isCompilerThread() { return true; }
71 
72 }
