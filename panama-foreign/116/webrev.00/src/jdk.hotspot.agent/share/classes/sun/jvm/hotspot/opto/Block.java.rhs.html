25 package sun.jvm.hotspot.opto;
26 
27 import java.util.*;
28 import java.io.PrintStream;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.oops.*;
32 import sun.jvm.hotspot.types.*;
33 import sun.jvm.hotspot.utilities.Observable;
34 import sun.jvm.hotspot.utilities.Observer;
35 
36 public class Block extends VMObject {
37   static {
38     VM.registerVMInitializedObserver(new Observer() {
39         public void update(Observable o, Object data) {
40           initialize(VM.getVM().getTypeDataBase());
41         }
42       });
43   }
44 
45   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
46     Type type      = db.lookupType("Block");
47     nodesField = type.getAddressField("_nodes");
48     succsField = type.getAddressField("_succs");
49     numSuccsField = new CIntField(type.getCIntegerField("_num_succs"), 0);
50     preOrderField = new CIntField(type.getCIntegerField("_pre_order"), 0);
51     domDepthField = new CIntField(type.getCIntegerField("_dom_depth"), 0);
52     idomField = type.getAddressField("_idom");
53     freqField = type.getJDoubleField("_freq");
54   }
55 
56   private static AddressField nodesField;
57   private static AddressField succsField;
58   private static CIntField numSuccsField;
59   private static CIntField preOrderField;
60   private static CIntField domDepthField;
61   private static AddressField idomField;
62   private static JDoubleField freqField;
63 
64   public Block(Address addr) {
65     super(addr);
66   }
67 
68   public int preOrder() {
69     return (int)preOrderField.getValue(getAddress());
70   }
71 
72   public double freq() {
73     return (double)freqField.getValue(getAddress());
74   }
75 
76   public Node_List nodes() {
77     return new Node_List(getAddress().addOffsetTo(nodesField.getOffset()));
78   }
79 
80   public void dump(PrintStream out) {
81     out.print("B" + preOrder());
82     out.print(" Freq: " + freq());
83     out.println();
84     Node_List nl = nodes();
85     int cnt = nl.size();
86     for( int i=0; i<cnt; i++ )
87       nl.at(i).dump(out);
88     out.print("\n");
89   }
90 }
