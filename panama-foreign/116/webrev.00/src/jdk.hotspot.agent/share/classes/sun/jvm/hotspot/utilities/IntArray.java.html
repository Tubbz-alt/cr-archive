25 package sun.jvm.hotspot.utilities;
26 
27 import sun.jvm.hotspot.debugger.Address;
28 import sun.jvm.hotspot.runtime.VM;
29 import sun.jvm.hotspot.types.Type;
30 import sun.jvm.hotspot.types.TypeDataBase;
31 import sun.jvm.hotspot.types.WrongTypeException;
32 import sun.jvm.hotspot.utilities.Observable;
33 import sun.jvm.hotspot.utilities.Observer;
34 
35 public class IntArray extends GenericArray {
36   static {
37     VM.registerVMInitializedObserver(new Observer() {
38       public void update(Observable o, Object data) {
39         initialize(VM.getVM().getTypeDataBase());
40       }
41     });
42   }
43 
44   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
45     elemType = db.lookupType("int");
46 
47     Type type = db.lookupType("Array<int>");
48     dataFieldOffset = type.getAddressField("_data").getOffset();
49   }
50 
51   private static long dataFieldOffset;
52   protected static Type elemType;
53 
54   public IntArray(Address addr) {
55     super(addr, dataFieldOffset);
56   }
57 
58   public int at(int i) {
59     return (int)getIntegerAt(i);
60   }
61 
62   public Type getElemType() {
63     return elemType;
64   }
65 }
