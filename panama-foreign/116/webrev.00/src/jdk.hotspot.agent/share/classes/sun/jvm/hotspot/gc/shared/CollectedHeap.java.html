 25 package sun.jvm.hotspot.gc.shared;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 
 30 import sun.jvm.hotspot.debugger.*;
 31 import sun.jvm.hotspot.gc.shared.*;
 32 import sun.jvm.hotspot.memory.*;
 33 import sun.jvm.hotspot.runtime.*;
 34 import sun.jvm.hotspot.types.*;
 35 import sun.jvm.hotspot.utilities.BitMapInterface;
 36 import sun.jvm.hotspot.utilities.BitMapSegmented;
 37 import sun.jvm.hotspot.utilities.Observable;
 38 import sun.jvm.hotspot.utilities.Observer;
 39 
 40 public abstract class CollectedHeap extends VMObject {
 41   private static long         reservedFieldOffset;
 42 
 43   static {
 44     VM.registerVMInitializedObserver(new Observer() {
 45         public void update(Observable o, Object data) {
 46           initialize(VM.getVM().getTypeDataBase());
 47         }
 48       });
 49   }
 50 
 51   private static synchronized void initialize(TypeDataBase db) {
 52     Type type = db.lookupType("CollectedHeap");
 53 
 54     reservedFieldOffset = type.getField("_reserved").getOffset();
 55   }
 56 
 57   public CollectedHeap(Address addr) {
 58     super(addr);
 59   }
 60 
 61   /** Returns the lowest address of the heap. */
 62   public Address start() {
 63     return reservedRegion().start();
 64   }
 65 
 66   public abstract long capacity();
 67   public abstract long used();
 68 
 69   public MemRegion reservedRegion() {
 70     return new MemRegion(addr.addOffsetTo(reservedFieldOffset));
 71   }
 72 
 73   public boolean isIn(Address a) {
 74     return isInReserved(a);
 75   }
 76 
 77   public boolean isInReserved(Address a) {
 78     return reservedRegion().contains(a);
 79   }
 80 
 81   public abstract CollectedHeapName kind();
 82 
 83   public abstract void liveRegionsIterate(LiveRegionsClosure closure);
 84 
 85   public String oopAddressDescription(OopHandle handle) {
 86       return handle.toString();
 87   }
 88 
 89   public OopHandle oop_load_at(OopHandle handle, long offset) {
 90       return handle.getOopHandleAt(offset);
 91   }
 92 
 93   public OopHandle oop_load_in_native(Address addr) {
 94       return addr.getOopHandleAt(0);
 95   }
 96 
 97   public void print() { printOn(System.out); }
 98   public void printOn(PrintStream tty) {
 99     MemRegion mr = reservedRegion();
100     tty.println("unknown subtype of CollectedHeap @ " + getAddress() + " (" +
101                 mr.start() + "," + mr.end() + ")");
102   }
103 
104   public BitMapInterface createBitMap(long bits) {
105     return new BitMapSegmented(bits);
106   }
107 }
