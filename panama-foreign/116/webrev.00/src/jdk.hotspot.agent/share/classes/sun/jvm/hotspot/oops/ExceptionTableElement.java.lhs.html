25 package sun.jvm.hotspot.oops;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.interpreter.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.types.*;
33 import sun.jvm.hotspot.utilities.*;


34 
35 public class ExceptionTableElement {
36   static {
37     VM.registerVMInitializedObserver(new Observer() {
38         public void update(Observable o, Object data) {
39           initialize(VM.getVM().getTypeDataBase());
40         }
41       });
42   }
43 
44   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
45     Type type            = db.lookupType("ExceptionTableElement");
46     offsetOfStartPC      = type.getCIntegerField("start_pc").getOffset();
47     offsetOfEndPC        = type.getCIntegerField("end_pc").getOffset();
48     offsetOfHandlerPC    = type.getCIntegerField("handler_pc").getOffset();
49     offsetOfCatchTypeIndex = type.getCIntegerField("catch_type_index").getOffset();
50   }
51 
52   private static long offsetOfStartPC;
53   private static long offsetOfEndPC;
54   private static long offsetOfHandlerPC;
55   private static long offsetOfCatchTypeIndex;
56 
57   private Address handle;
58   private long      offset;
59 
60   public ExceptionTableElement(Address handle, long offset) {
61     this.handle = handle;
62     this.offset = offset;
63   }
64 
65   public int getStartPC() {
66     return (int) handle.getCIntegerAt(offset + offsetOfStartPC, 2, true);
67   }
68 
69   public int getEndPC() {
70     return (int) handle.getCIntegerAt(offset + offsetOfEndPC, 2, true);
71   }
72 
73   public int getHandlerPC() {
74     return (int) handle.getCIntegerAt(offset + offsetOfHandlerPC, 2, true);
75   }
76 
77   public int getCatchTypeIndex() {
78     return (int) handle.getCIntegerAt(offset + offsetOfCatchTypeIndex, 2, true);
79   }
80 }
