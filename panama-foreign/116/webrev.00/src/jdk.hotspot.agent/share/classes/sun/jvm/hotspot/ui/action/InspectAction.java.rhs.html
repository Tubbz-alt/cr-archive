25 package sun.jvm.hotspot.ui.action;
26 
27 import javax.swing.Action;
28 
29 import com.sun.java.swing.action.ActionManager;
30 import com.sun.java.swing.action.DelegateAction;
31 
32 /**
33  * Callback action for Inspecting the Object Type
34  */
35 public class InspectAction extends DelegateAction {
36 
37     public static final String VALUE_COMMAND = "inspect-command";
38     public static final String VALUE_NAME = "Inspect Thread...";
39     public static final String VALUE_SMALL_ICON = "general/ZoomIn16.gif";
40     public static final String VALUE_LARGE_ICON = "general/ZoomIn24.gif";
41     public static final Integer VALUE_MNEMONIC = (int) 'T';
42     public static final String VALUE_SHORT_DESCRIPTION = "Inspect selected thread";
43     public static final String VALUE_LONG_DESCRIPTION = "Open an inspector on the selected thread";
44 
45     public InspectAction() {
46         super(VALUE_NAME, ActionManager.getIcon(VALUE_SMALL_ICON));
47 
48         putValue(Action.ACTION_COMMAND_KEY, VALUE_COMMAND);
49         putValue(Action.SHORT_DESCRIPTION, VALUE_SHORT_DESCRIPTION);
50         putValue(Action.LONG_DESCRIPTION, VALUE_LONG_DESCRIPTION);
51         putValue(Action.MNEMONIC_KEY, VALUE_MNEMONIC);
52     }
53 }
