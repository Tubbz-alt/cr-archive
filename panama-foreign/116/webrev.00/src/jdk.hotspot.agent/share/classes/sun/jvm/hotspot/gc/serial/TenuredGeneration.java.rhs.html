25 package sun.jvm.hotspot.gc.serial;
26 
27 import java.io.*;
28 import java.util.*;
29 
30 import sun.jvm.hotspot.debugger.*;
31 import sun.jvm.hotspot.gc.shared.*;
32 import sun.jvm.hotspot.runtime.*;
33 import sun.jvm.hotspot.types.*;
34 import sun.jvm.hotspot.utilities.Observable;
35 import sun.jvm.hotspot.utilities.Observer;
36 
37 /** <P> TenuredGeneration models a heap of old objects contained
38     in a single contiguous space. </P>
39 
40     <P> Garbage collection is performed using mark-compact. </P> */
41 
42 public class TenuredGeneration extends CardGeneration {
43   private static AddressField theSpaceField;
44 
45   static {
46     VM.registerVMInitializedObserver(new Observer() {
47         public void update(Observable o, Object data) {
48           initialize(VM.getVM().getTypeDataBase());
49         }
50       });
51   }
52 
53   private static synchronized void initialize(TypeDataBase db) {
54     Type type = db.lookupType("TenuredGeneration");
55 
56     theSpaceField = type.getAddressField("_the_space");
57   }
58 
59   public TenuredGeneration(Address addr) {
60     super(addr);
61   }
62 
63   public ContiguousSpace theSpace() {
64     return (ContiguousSpace) VMObjectFactory.newObject(ContiguousSpace.class, theSpaceField.getValue(addr));
65   }
66 
67   public boolean isIn(Address p) {
68     return theSpace().contains(p);
69   }
70 
71   /** Space queries */
72   public long capacity()            { return theSpace().capacity();                                }
73   public long used()                { return theSpace().used();                                    }
74   public long free()                { return theSpace().free();                                    }
75   public long contiguousAvailable() { return theSpace().free() + virtualSpace().uncommittedSize(); }
76 
77   public void spaceIterate(SpaceClosure blk, boolean usedOnly) {
78     blk.doSpace(theSpace());
79   }
80 
81   public void liveRegionsIterate(LiveRegionsClosure closure) {
82     closure.doLiveRegions(theSpace());
83   }
84 
85   public void printOn(PrintStream tty) {
86     tty.print("  old ");
87     theSpace().printOn(tty);
88   }
89 
90   public Generation.Name kind() {
91     return Generation.Name.MARK_SWEEP_COMPACT;
92   }
93 
94   public String name() {
95     return "tenured generation";
96   }
97 }
