25 package sun.jvm.hotspot.runtime;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.*;
33 import sun.jvm.hotspot.utilities.Observable;
34 import sun.jvm.hotspot.utilities.Observer;
35 
36 /** This class represent VM's Arguments class -- command line args, flags etc.*/
37 public class Arguments {
38     static {
39         VM.registerVMInitializedObserver(new Observer() {
40             public void update(Observable o, Object data) {
41                 initialize(VM.getVM().getTypeDataBase());
42             }
43         });
44     }
45 
46     public static String getJVMFlags() {
47         return buildString(jvmFlagsArrayField, numJvmFlags);
48     }
49 
50     public static String getJVMArgs() {
51         return buildString(jvmArgsArrayField, numJvmArgs);
52     }
53 
54     public static String getJavaCommand() {
55         return getString(javaCommandField);
56     }
57 
58     // Internals only below this point
59 
60     // Fields
61     private static AddressField jvmFlagsArrayField;
62     private static AddressField jvmArgsArrayField;
63     private static AddressField javaCommandField;
64     private static long numJvmFlags;
65     private static long numJvmArgs;
66 
67     private static synchronized void initialize(TypeDataBase db) {
68         Type argumentsType = db.lookupType("Arguments");
69         jvmFlagsArrayField = argumentsType.getAddressField("_jvm_flags_array");
70         jvmArgsArrayField = argumentsType.getAddressField("_jvm_args_array");
71         javaCommandField = argumentsType.getAddressField("_java_command");
72 
73         numJvmArgs = argumentsType.getCIntegerField("_num_jvm_args").getValue();
74         numJvmFlags = argumentsType.getCIntegerField("_num_jvm_flags").getValue();
75     }
76 
77     private static String buildString(AddressField arrayField, long count) {
78         StringBuilder sb = new StringBuilder();
79         if (count > 0) {
80             sb.append(getStringAt(arrayField, 0));
81             for (long i = 1; i < count; i++) {
82                 sb.append(" ");
83                 sb.append(getStringAt(arrayField, i));
84             }
85         }
86         return sb.toString();
87     }
88 
89     private static String getString(AddressField field) {
90         Address addr = field.getAddress();
91         return CStringUtilities.getString(addr);
92     }
93 
94     private static String getStringAt(AddressField field, long index) {
95         Address addr = field.getAddress();
96         return CStringUtilities.getString(addr.getAddressAt(index * VM.getVM().getAddressSize()));
97     }
98 }
