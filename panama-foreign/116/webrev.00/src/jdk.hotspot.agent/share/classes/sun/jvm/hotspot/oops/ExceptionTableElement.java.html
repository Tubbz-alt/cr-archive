25 package sun.jvm.hotspot.oops;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.interpreter.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.types.*;
33 import sun.jvm.hotspot.utilities.*;
34 import sun.jvm.hotspot.utilities.Observable;
35 import sun.jvm.hotspot.utilities.Observer;
36 
37 public class ExceptionTableElement {
38   static {
39     VM.registerVMInitializedObserver(new Observer() {
40         public void update(Observable o, Object data) {
41           initialize(VM.getVM().getTypeDataBase());
42         }
43       });
44   }
45 
46   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
47     Type type            = db.lookupType("ExceptionTableElement");
48     offsetOfStartPC      = type.getCIntegerField("start_pc").getOffset();
49     offsetOfEndPC        = type.getCIntegerField("end_pc").getOffset();
50     offsetOfHandlerPC    = type.getCIntegerField("handler_pc").getOffset();
51     offsetOfCatchTypeIndex = type.getCIntegerField("catch_type_index").getOffset();
52   }
53 
54   private static long offsetOfStartPC;
55   private static long offsetOfEndPC;
56   private static long offsetOfHandlerPC;
57   private static long offsetOfCatchTypeIndex;
58 
59   private Address handle;
60   private long      offset;
61 
62   public ExceptionTableElement(Address handle, long offset) {
63     this.handle = handle;
64     this.offset = offset;
65   }
66 
67   public int getStartPC() {
68     return (int) handle.getCIntegerAt(offset + offsetOfStartPC, 2, true);
69   }
70 
71   public int getEndPC() {
72     return (int) handle.getCIntegerAt(offset + offsetOfEndPC, 2, true);
73   }
74 
75   public int getHandlerPC() {
76     return (int) handle.getCIntegerAt(offset + offsetOfHandlerPC, 2, true);
77   }
78 
79   public int getCatchTypeIndex() {
80     return (int) handle.getCIntegerAt(offset + offsetOfCatchTypeIndex, 2, true);
81   }
82 }
