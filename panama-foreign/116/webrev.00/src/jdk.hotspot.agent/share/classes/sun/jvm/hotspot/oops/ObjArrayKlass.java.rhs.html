 25 package sun.jvm.hotspot.oops;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 import sun.jvm.hotspot.debugger.*;
 30 import sun.jvm.hotspot.runtime.*;
 31 import sun.jvm.hotspot.types.*;
 32 import sun.jvm.hotspot.utilities.*;
 33 import sun.jvm.hotspot.utilities.Observable;
 34 import sun.jvm.hotspot.utilities.Observer;
 35 
 36 // ObjArrayKlass is the klass for ObjArrays
 37 
 38 public class ObjArrayKlass extends ArrayKlass {
 39   static {
 40     VM.registerVMInitializedObserver(new Observer() {
 41         public void update(Observable o, Object data) {
 42           initialize(VM.getVM().getTypeDataBase());
 43         }
 44       });
 45   }
 46 
 47   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
 48     Type type = db.lookupType("ObjArrayKlass");
 49     elementKlass = new MetadataField(type.getAddressField("_element_klass"), 0);
 50     bottomKlass  = new MetadataField(type.getAddressField("_bottom_klass"), 0);
 51   }
 52 
 53   public ObjArrayKlass(Address addr) {
 54     super(addr);
 55   }
 56 
 57   private static MetadataField elementKlass;
 58   private static MetadataField bottomKlass;
 59 
 60   public Klass getElementKlass() { return (Klass) elementKlass.getValue(this); }
 61   public Klass getBottomKlass()  { return (Klass) bottomKlass.getValue(this); }
 62 
 63   public long computeModifierFlags() {
 64     long elementFlags = getElementKlass().computeModifierFlags();
 65     long arrayFlags = 0L;
 66     if ((elementFlags & (JVM_ACC_PUBLIC | JVM_ACC_PROTECTED)) != 0) {
 67        // The array type is public if the component type is public or protected
 68        arrayFlags = JVM_ACC_ABSTRACT | JVM_ACC_FINAL | JVM_ACC_PUBLIC;
 69     } else {
 70        // The array type is private if the component type is private
 71        arrayFlags = JVM_ACC_ABSTRACT | JVM_ACC_FINAL;
 72     }
 73     return arrayFlags;
 74   }
 75 
 76   public void iterateFields(MetadataVisitor visitor) {
 77     super.iterateFields(visitor);
 78     visitor.doMetadata(elementKlass, true);
 79     visitor.doMetadata(bottomKlass, true);
 80   }
 81 
 82   public Klass arrayKlassImpl(boolean orNull, int n) {
 83     if (Assert.ASSERTS_ENABLED) {
 84       Assert.that(getDimension() <= n, "check order of chain");
 85     }
 86     int dimension = (int) getDimension();
 87     if (dimension == n) {
 88       return this;
 89     }
 90     ObjArrayKlass ak = (ObjArrayKlass) getHigherDimension();
 91     if (ak == null) {
 92       if (orNull) return null;
 93       // FIXME: would need to change in reflective system to actually
 94       // allocate klass
 95       throw new RuntimeException("Can not allocate array klasses in debugging system");
 96     }
 97 
 98     if (orNull) {
 99       return ak.arrayKlassOrNull(n);
100     }
101     return ak.arrayKlass(n);
102   }
103 
104   public Klass arrayKlassImpl(boolean orNull) {
105     return arrayKlassImpl(orNull, (int) (getDimension() + 1));
106   }
107 
108   public void printValueOn(PrintStream tty) {
109     tty.print("ObjArrayKlass for ");
110     getElementKlass().printValueOn(tty);
111   }
112 };
