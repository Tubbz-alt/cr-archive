25 package sun.jvm.hotspot.gc.shared;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.gc.serial.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.types.*;
33 
34 /** Factory containing a VirtualConstructor suitable for instantiating
35     wrapper objects for all types of generations */
36 
37 public class GenerationFactory {
38   private static VirtualConstructor ctor;
39 
40   static {
41     VM.registerVMInitializedObserver(new Observer() {
42         public void update(Observable o, Object data) {
43           initialize(VM.getVM().getTypeDataBase());
44         }
45       });
46   }
47 
48   private static synchronized void initialize(TypeDataBase db) {
49     ctor = new VirtualConstructor(db);
50 
51     ctor.addMapping("DefNewGeneration", DefNewGeneration.class);
52     ctor.addMapping("TenuredGeneration", TenuredGeneration.class);
53   }
54 
55   public static Generation newObject(Address addr) {
56       try {
57           return (Generation) ctor.instantiateWrapperFor(addr);
58       } catch (WrongTypeException e) {
59           return new Generation(addr) {
60                   public String name() {
61                       return "unknown generation type";
62                   }
63                   public void spaceIterate(SpaceClosure blk, boolean usedOnly) {
64                   }
65                   public void liveRegionsIterate(LiveRegionsClosure closure) {
66                   }
67                   public void printOn(java.io.PrintStream tty) {
68                       tty.println("unknown subtype of Generation @ " + getAddress() + " (" +
69                                   virtualSpace().low() + "," + virtualSpace().high() + ")");
70                   }
71                   public long used() {
72                       return 0;
73                   }
74                   public long free() {
75                       return 0;
76                   }
77                   public long capacity() {
78                       return 0;
79                   }
80                   public long contiguousAvailable() {
81                       return 0;
82                   }
83 
84               };
85       }
86   }
87 }
