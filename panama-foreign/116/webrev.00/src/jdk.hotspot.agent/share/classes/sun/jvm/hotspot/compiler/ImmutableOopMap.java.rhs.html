25 package sun.jvm.hotspot.compiler;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.code.*;
30 import sun.jvm.hotspot.debugger.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.types.*;
33 import sun.jvm.hotspot.utilities.Observable;
34 import sun.jvm.hotspot.utilities.Observer;
35 
36 public class ImmutableOopMap extends VMObject {
37   private static CIntegerField countField;
38   private static long classSize;
39 
40   static {
41     VM.registerVMInitializedObserver(new Observer() {
42         public void update(Observable o, Object data) {
43           initialize(VM.getVM().getTypeDataBase());
44         }
45       });
46   }
47 
48   private static void initialize(TypeDataBase db) {
49     Type type = db.lookupType("ImmutableOopMap");
50     countField = type.getCIntegerField("_count");
51     classSize = type.getSize();
52   }
53 
54   public ImmutableOopMap(Address addr) {
55     super(addr);
56   }
57 
58   //--------------------------------------------------------------------------------
59   // Internals only below this point
60   //
61 
62   long getCount() {
63     return countField.getValue(addr);
64   }
65 
66   public Address getData() {
67     return addr.addOffsetTo(classSize);
68   }
69 }
