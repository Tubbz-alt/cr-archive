 25 package sun.jvm.hotspot.runtime;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 import sun.jvm.hotspot.debugger.*;
 30 import sun.jvm.hotspot.oops.*;
 31 import sun.jvm.hotspot.types.*;
 32 import sun.jvm.hotspot.utilities.Observable;
 33 import sun.jvm.hotspot.utilities.Observer;
 34 
 35 /** <P> ThreadLocalAllocBuffer: a descriptor for thread-local storage
 36     used by the threads for allocation. </P>
 37 
 38     <P> It is thread-private at any time, but maybe multiplexed over
 39     time across multiple threads. </P> */
 40 
 41 public class ThreadLocalAllocBuffer extends VMObject {
 42   private static AddressField  startField;
 43   private static AddressField  topField;
 44   private static AddressField  endField;
 45   private static CIntegerField desired_sizeField;
 46 
 47   static {
 48     VM.registerVMInitializedObserver(new Observer() {
 49         public void update(Observable o, Object data) {
 50           initialize(VM.getVM().getTypeDataBase());
 51         }
 52       });
 53   }
 54 
 55   private static synchronized void initialize(TypeDataBase db) {
 56     Type type = db.lookupType("ThreadLocalAllocBuffer");
 57 
 58     startField         = type.getAddressField("_start");
 59     topField           = type.getAddressField("_top");
 60     endField           = type.getAddressField("_end");
 61     desired_sizeField          = type.getCIntegerField("_desired_size");
 62   }
 63 
 64   public ThreadLocalAllocBuffer(Address addr) {
 65     super(addr);
 66   }
 67 
 68   public Address start()    { return startField.getValue(addr); }
 69   public Address end()      { return   endField.getValue(addr); }
 70   public Address top()      { return   topField.getValue(addr); }
 71   public Address hardEnd()  { return end().addOffsetTo(alignmentReserve()); }
 72 
 73   private long alignmentReserve() {
 74     return Oop.alignObjectSize(endReserve());
 75   }
 76 
 77   private long endReserve() {
 78     long minFillerArraySize = Array.baseOffsetInBytes(BasicType.T_INT);
 79     long reserveForAllocationPrefetch = VM.getVM().getReserveForAllocationPrefetch();
 80     long heapWordSize = VM.getVM().getHeapWordSize();
 81 
 82     return Math.max(minFillerArraySize, reserveForAllocationPrefetch * heapWordSize);
 83   }
 84 
 85   /** Support for iteration over heap -- not sure how this will
 86       interact with GC in reflective system, but necessary for the
 87       debugging mechanism */
 88   public OopHandle startAsOopHandle() {
 89     return startField.getOopHandle(addr);
 90   }
 91 
 92   /** Support for iteration over heap -- not sure how this will
 93       interact with GC in reflective system, but necessary for the
 94       debugging mechanism */
 95   public OopHandle nextOopHandle(OopHandle handle, long size) {
 96     return handle.addOffsetToAsOopHandle(size);
 97   }
 98 
 99   //  public int     size();           // FIXME: must adjust this to be in bytes
100   //  public int     availableSize();  // FIXME: must adjust this to be in bytes
101   public void print() {
102     printOn(System.out);
103   }
104 
105   public boolean contains(Address p) {
106     if (top() == null) {
107       return false;
108     }
109     return (start().lessThanOrEqual(p) && top().greaterThan(p));
110   }
111 
112   public void printOn(PrintStream tty) {
113     tty.println(" [" + start() + "," +
114                 top() + "," + end() + ",{" + hardEnd() + "})");
115   }
116 }
