 25 package sun.jvm.hotspot.gc.parallel;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 
 30 import sun.jvm.hotspot.debugger.*;
 31 import sun.jvm.hotspot.memory.*;
 32 import sun.jvm.hotspot.runtime.*;
 33 import sun.jvm.hotspot.types.*;
 34 import sun.jvm.hotspot.utilities.Observable;
 35 import sun.jvm.hotspot.utilities.Observer;
 36 
 37 public class PSYoungGen extends VMObject {
 38    static {
 39       VM.registerVMInitializedObserver(new Observer() {
 40          public void update(Observable o, Object data) {
 41             initialize(VM.getVM().getTypeDataBase());
 42          }
 43       });
 44    }
 45 
 46    private static synchronized void initialize(TypeDataBase db) {
 47       Type type = db.lookupType("PSYoungGen");
 48       edenSpaceField = type.getAddressField("_eden_space");
 49       fromSpaceField = type.getAddressField("_from_space");
 50       toSpaceField = type.getAddressField("_to_space");
 51    }
 52 
 53    public PSYoungGen(Address addr) {
 54       super(addr);
 55    }
 56 
 57    // Fields
 58    private static AddressField edenSpaceField;
 59    private static AddressField fromSpaceField;
 60    private static AddressField toSpaceField;
 61 
 62    // Accessors
 63    public MutableSpace edenSpace() {
 64       return (MutableSpace) VMObjectFactory.newObject(MutableSpace.class, edenSpaceField.getValue(addr));
 65    }
 66 
 67    public MutableSpace fromSpace() {
 68       return (MutableSpace) VMObjectFactory.newObject(MutableSpace.class, fromSpaceField.getValue(addr));
 69    }
 70 
 71    public MutableSpace toSpace() {
 72       return (MutableSpace) VMObjectFactory.newObject(MutableSpace.class, toSpaceField.getValue(addr));
 73    }
 74 
 75    public long capacity() {
 76       return edenSpace().capacity() + fromSpace().capacity();
 77    }
 78 
 79    public long used() {
 80       return edenSpace().used() + fromSpace().used();
 81    }
 82 
 83    public boolean isIn(Address a) {
 84       if (edenSpace().contains(a)) {
 85          return true;
 86       }
 87 
 88       if (fromSpace().contains(a)) {
 89          return true;
 90       }
 91       return false;
 92    }
 93 
 94    public void printOn(PrintStream tty) {
 95       tty.print("PSYoungGen [ ");
 96       tty.print("eden = ");
 97       edenSpace().printOn(tty);
 98       tty.print(", from = ");
 99       fromSpace().printOn(tty);
100       tty.print(", to = ");
101       toSpace().printOn(tty);
102       tty.print(" ] ");
103    }
104 }
