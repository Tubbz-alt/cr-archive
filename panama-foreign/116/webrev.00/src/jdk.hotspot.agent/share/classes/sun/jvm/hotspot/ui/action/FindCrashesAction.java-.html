25 package sun.jvm.hotspot.ui.action;
26 
27 import javax.swing.Action;
28 
29 import com.sun.java.swing.action.ActionManager;
30 import com.sun.java.swing.action.DelegateAction;
31 
32 /**
33  * Callback action for FindCrashesing the Object Type
34  */
35 public class FindCrashesAction extends DelegateAction {
36 
37     public static final String VALUE_COMMAND = "find-crashes-command";
38     public static final String VALUE_NAME = "Find Crashes...";
39     // XXX - These icons are temporary. Should use a lightning bolt with a search arrow.
40     public static final String VALUE_SMALL_ICON = "general/Delete16.gif";
41     public static final String VALUE_LARGE_ICON = "general/Delete24.gif";
42     public static final Integer VALUE_MNEMONIC = new Integer('H');
43     public static final String VALUE_SHORT_DESCRIPTION = "Find Crashes";
44     public static final String VALUE_LONG_DESCRIPTION = "Searches the threads for potential crashes";
45 
46     public FindCrashesAction() {
47         super(VALUE_NAME, ActionManager.getIcon(VALUE_SMALL_ICON));
48 
49         putValue(Action.ACTION_COMMAND_KEY, VALUE_COMMAND);
50         putValue(Action.SHORT_DESCRIPTION, VALUE_SHORT_DESCRIPTION);
51         putValue(Action.LONG_DESCRIPTION, VALUE_LONG_DESCRIPTION);
52         putValue(Action.MNEMONIC_KEY, VALUE_MNEMONIC);
53     }
54 }
