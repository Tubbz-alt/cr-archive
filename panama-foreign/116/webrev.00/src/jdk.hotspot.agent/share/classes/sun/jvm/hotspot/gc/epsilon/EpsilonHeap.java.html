25 package sun.jvm.hotspot.gc.epsilon;
26 
27 import java.io.*;
28 import java.util.*;
29 
30 import sun.jvm.hotspot.gc.shared.*;
31 import sun.jvm.hotspot.code.*;
32 import sun.jvm.hotspot.debugger.*;
33 import sun.jvm.hotspot.runtime.*;
34 import sun.jvm.hotspot.types.*;
35 import sun.jvm.hotspot.memory.*;
36 import sun.jvm.hotspot.utilities.Observable;
37 import sun.jvm.hotspot.utilities.Observer;
38 
39 public class EpsilonHeap extends CollectedHeap {
40 
41   private static AddressField spaceField;
42   private static Field virtualSpaceField;
43   private ContiguousSpace space;
44   private VirtualSpace virtualSpace;
45 
46   static {
47     VM.registerVMInitializedObserver(new Observer() {
48         public void update(Observable o, Object data) {
49           initialize(VM.getVM().getTypeDataBase());
50         }
51       });
52   }
53 
54   private static void initialize(TypeDataBase db) {
55     Type type = db.lookupType("EpsilonHeap");
56     spaceField = type.getAddressField("_space");
57     virtualSpaceField = type.getField("_virtual_space");
58   }
59 
60   public EpsilonHeap(Address addr) {
61     super(addr);
62     space = new ContiguousSpace(spaceField.getValue(addr));
63     virtualSpace = new VirtualSpace(addr.addOffsetTo(virtualSpaceField.getOffset()));
64   }
65 
66   @Override
67   public CollectedHeapName kind() {
68     return CollectedHeapName.EPSILON;
69   }
70 
71   @Override
72   public long capacity() {
73     return space.capacity();
74   }
75 
76   @Override
77   public long used() {
78     return space.used();
79   }
80 
81   public ContiguousSpace space() {
82     return space;
83   }
84 
85   @Override
86   public void liveRegionsIterate(LiveRegionsClosure closure) {
87     closure.doLiveRegions(space());
88   }
89 
90   @Override
91   public void printOn(PrintStream tty) {
92      MemRegion mr = reservedRegion();
93      tty.println("Epsilon heap");
94      tty.println(" reserved:  [" + mr.start() + ", " + mr.end() + "]");
95      tty.println(" committed: [" + virtualSpace.low() + ", " + virtualSpace.high() + "]");
96      tty.println(" used:      [" + space.bottom() + ", " + space.top() + "]");
97   }
98 
99 }
