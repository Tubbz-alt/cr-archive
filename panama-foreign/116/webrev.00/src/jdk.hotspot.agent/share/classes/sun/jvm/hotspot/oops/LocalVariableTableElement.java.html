25 package sun.jvm.hotspot.oops;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.interpreter.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.types.*;
33 import sun.jvm.hotspot.utilities.*;
34 import sun.jvm.hotspot.utilities.Observable;
35 import sun.jvm.hotspot.utilities.Observer;
36 
37 public class LocalVariableTableElement {
38   static {
39     VM.registerVMInitializedObserver(new Observer() {
40         public void update(Observable o, Object data) {
41           initialize(VM.getVM().getTypeDataBase());
42         }
43       });
44   }
45 
46   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
47     Type type                 = db.lookupType("LocalVariableTableElement");
48     offsetOfStartBCI          = type.getCIntegerField("start_bci").getOffset();
49     offsetOfLength            = type.getCIntegerField("length").getOffset();
50     offsetOfNameCPIndex       = type.getCIntegerField("name_cp_index").getOffset();
51     offsetOfDescriptorCPIndex = type.getCIntegerField("descriptor_cp_index").getOffset();
52     offsetOfSignatureCPIndex  = type.getCIntegerField("signature_cp_index").getOffset();
53     offsetOfSlot              = type.getCIntegerField("slot").getOffset();
54   }
55 
56   private static long offsetOfStartBCI;
57   private static long offsetOfLength;
58   private static long offsetOfNameCPIndex;
59   private static long offsetOfDescriptorCPIndex;
60   private static long offsetOfSignatureCPIndex;
61   private static long offsetOfSlot;
62 
63   private Address   handle;
64   private long      offset;
65 
66   public LocalVariableTableElement(Address handle, long offset) {
67     this.handle = handle;
68     this.offset = offset;
69   }
70 
71   public int getStartBCI() {
72     return (int) handle.getCIntegerAt(offset + offsetOfStartBCI, 2, true);
73   }
74 
75   public int getLength() {
76     return (int) handle.getCIntegerAt(offset + offsetOfLength, 2, true);
77   }
78 
79   public int getNameCPIndex() {
80     return (int) handle.getCIntegerAt(offset + offsetOfNameCPIndex, 2, true);
81   }
82 
83   public int getDescriptorCPIndex() {
84     return (int) handle.getCIntegerAt(offset + offsetOfDescriptorCPIndex, 2, true);
85   }
86 
87   public int getSignatureCPIndex() {
88     return (int) handle.getCIntegerAt(offset + offsetOfSignatureCPIndex, 2, true);
89   }
90 
91   public int getSlot() {
92     return (int) handle.getCIntegerAt(offset + offsetOfSlot, 2, true);
93   }
94 }
