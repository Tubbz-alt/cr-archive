 50 /**
 51  * This base class encapsulates many of the events that are fired from
 52  * the various panels in this directory so they can easily be plugged
 53  * in to different containing frameworks (HSDB).
 54  */
 55 public class SAPanel extends JPanel {
 56     protected List<SAListener> listeners = new ArrayList<>();
 57 
 58 
 59     public SAPanel() {
 60     }
 61 
 62     public void addPanelListener(SAListener listener) {
 63         listeners.add(listener);
 64     }
 65 
 66     public void removePanelListener(SAListener listener) {
 67         listeners.remove(listener);
 68     }
 69 
 70     public void showThreadOopInspector(JavaThread t) {
 71         for (Iterator iter = listeners.iterator(); iter.hasNext(); ) {
 72             SAListener listener = (SAListener) iter.next();
 73             listener.showThreadOopInspector(t);
 74         }
 75     }
 76 
 77     public void showInspector(Oop oop) {
 78         showInspector(new OopTreeNodeAdapter(oop, null));
 79     }
 80 
 81     public void showInspector(SimpleTreeNode node) {
 82         for (Iterator iter = listeners.iterator(); iter.hasNext(); ) {
 83             SAListener listener = (SAListener) iter.next();
 84             listener.showInspector(node);
 85         }
 86     }
 87 
 88     public void showThreadStackMemory(JavaThread t) {
 89         for (Iterator iter = listeners.iterator(); iter.hasNext(); ) {
 90             SAListener listener = (SAListener) iter.next();
 91             listener.showThreadStackMemory(t);
 92         }
 93     }
 94 
 95     public void showJavaStackTrace(JavaThread t) {
 96         for (Iterator iter = listeners.iterator(); iter.hasNext(); ) {
 97             SAListener listener = (SAListener) iter.next();
 98             listener.showJavaStackTrace(t);
 99         }
100     }
101 
102     public void showThreadInfo(JavaThread t) {
103         for (Iterator iter = listeners.iterator(); iter.hasNext(); ) {
104             SAListener listener = (SAListener) iter.next();
105             listener.showThreadInfo(t);
106         }
107     }
108 
109     public void showCodeViewer(Address address) {
110         for (Iterator iter = listeners.iterator(); iter.hasNext(); ) {
111             SAListener listener = (SAListener) iter.next();
112             listener.showCodeViewer(address);
113         }
114     }
115 
116 
117 }
