25 package sun.jvm.hotspot.compiler;
26 
27 import sun.jvm.hotspot.debugger.Address;
28 import sun.jvm.hotspot.runtime.VM;
29 import sun.jvm.hotspot.types.CIntegerField;
30 import sun.jvm.hotspot.types.Type;
31 import sun.jvm.hotspot.types.TypeDataBase;
32 
33 import java.util.Observable;
34 import java.util.Observer;
35 
36 public class ImmutableOopMapPair {
37   private static CIntegerField pcOffsetField;
38   private static CIntegerField oopmapOffsetField;
39   private static long classSize;
40 
41   static {
42     VM.registerVMInitializedObserver(new Observer() {
43       public void update(Observable o, Object data) {
44         initialize(VM.getVM().getTypeDataBase());
45       }
46     });
47   }
48 
49   private final Address address;
50 
51   public ImmutableOopMapPair(Address address) {
52     this.address = address;
53   }
54 
55   public static long classSize() {
56     return classSize;
57   }
58 
59   public int getPC() {
60     return (int) pcOffsetField.getValue(address);
61   }
62 
63   public int getOffset() {
64     return (int) oopmapOffsetField.getValue(address);
65   }
66 
67   private static void initialize(TypeDataBase db) {
68     Type type = db.lookupType("ImmutableOopMapPair");
69 
70     pcOffsetField = type.getCIntegerField("_pc_offset");
71     oopmapOffsetField = type.getCIntegerField("_oopmap_offset");
72     classSize = type.getSize();
73   }
74 
75   public String toString() {
76     return "Pair{pc_offset = " + getPC() + ", data_offset = " + getOffset() + "}";
77   }
78 }
