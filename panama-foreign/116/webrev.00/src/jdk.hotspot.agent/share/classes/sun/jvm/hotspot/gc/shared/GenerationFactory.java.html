25 package sun.jvm.hotspot.gc.shared;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.gc.serial.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.types.*;
33 import sun.jvm.hotspot.utilities.Observable;
34 import sun.jvm.hotspot.utilities.Observer;
35 
36 /** Factory containing a VirtualConstructor suitable for instantiating
37     wrapper objects for all types of generations */
38 
39 public class GenerationFactory {
40   private static VirtualConstructor ctor;
41 
42   static {
43     VM.registerVMInitializedObserver(new Observer() {
44         public void update(Observable o, Object data) {
45           initialize(VM.getVM().getTypeDataBase());
46         }
47       });
48   }
49 
50   private static synchronized void initialize(TypeDataBase db) {
51     ctor = new VirtualConstructor(db);
52 
53     ctor.addMapping("DefNewGeneration", DefNewGeneration.class);
54     ctor.addMapping("TenuredGeneration", TenuredGeneration.class);
55   }
56 
57   public static Generation newObject(Address addr) {
58       try {
59           return (Generation) ctor.instantiateWrapperFor(addr);
60       } catch (WrongTypeException e) {
61           return new Generation(addr) {
62                   public String name() {
63                       return "unknown generation type";
64                   }
65                   public void spaceIterate(SpaceClosure blk, boolean usedOnly) {
66                   }
67                   public void liveRegionsIterate(LiveRegionsClosure closure) {
68                   }
69                   public void printOn(java.io.PrintStream tty) {
70                       tty.println("unknown subtype of Generation @ " + getAddress() + " (" +
71                                   virtualSpace().low() + "," + virtualSpace().high() + ")");
72                   }
73                   public long used() {
74                       return 0;
75                   }
76                   public long free() {
77                       return 0;
78                   }
79                   public long capacity() {
80                       return 0;
81                   }
82                   public long contiguousAvailable() {
83                       return 0;
84                   }
85 
86               };
87       }
88   }
89 }
