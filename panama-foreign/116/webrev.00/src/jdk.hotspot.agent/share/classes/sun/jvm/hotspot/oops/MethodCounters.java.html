25 package sun.jvm.hotspot.oops;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.*;
33 import sun.jvm.hotspot.utilities.Observable;
34 import sun.jvm.hotspot.utilities.Observer;
35 
36 public class MethodCounters extends Metadata {
37   public MethodCounters(Address addr) {
38     super(addr);
39   }
40 
41   static {
42     VM.registerVMInitializedObserver(new Observer() {
43         public void update(Observable o, Object data) {
44           initialize(VM.getVM().getTypeDataBase());
45         }
46       });
47   }
48 
49   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
50     Type type      = db.lookupType("MethodCounters");
51 
52     if (VM.getVM().isServerCompiler()) {
53       interpreterInvocationCountField = new CIntField(type.getCIntegerField("_interpreter_invocation_count"), 0);
54       interpreterThrowoutCountField = new CIntField(type.getCIntegerField("_interpreter_throwout_count"), 0);
55     }
56     if (!VM.getVM().isCore()) {
57       invocationCounter        = new CIntField(type.getCIntegerField("_invocation_counter"), 0);
58       backedgeCounter          = new CIntField(type.getCIntegerField("_backedge_counter"), 0);
59     }
60   }
61 
62   private static CIntField interpreterInvocationCountField;
63   private static CIntField interpreterThrowoutCountField;
64   private static CIntField invocationCounter;
65   private static CIntField backedgeCounter;
66 
67   public int interpreterInvocationCount() {
68       if (interpreterInvocationCountField != null) {
69         return (int) interpreterInvocationCountField.getValue(this);
70       } else {
71         return 0;
72       }
73   }
74 
75   public int interpreterThrowoutCount() {
76       if (interpreterThrowoutCountField != null) {
77         return (int) interpreterThrowoutCountField.getValue(this);
78       } else {
79         return 0;
80       }
81   }
82   public long getInvocationCounter() {
83     if (Assert.ASSERTS_ENABLED) {
84       Assert.that(!VM.getVM().isCore(), "must not be used in core build");
85     }
86     return invocationCounter.getValue(this);
87   }
88   public long getBackedgeCounter() {
89     if (Assert.ASSERTS_ENABLED) {
90       Assert.that(!VM.getVM().isCore(), "must not be used in core build");
91     }
92     return backedgeCounter.getValue(this);
93   }
94 
95   public void printValueOn(PrintStream tty) {
96   }
97 }
