31 import sun.jvm.hotspot.debugger.Address;

40 // Mirror class for HeapRegionSetBase. Represents a group of regions.
41 
42 public class HeapRegionSetBase extends VMObject {
43 
44     // uint _length
45     static private CIntegerField lengthField;
46 
47     static {
48         VM.registerVMInitializedObserver(new Observer() {
49                 public void update(Observable o, Object data) {
50                     initialize(VM.getVM().getTypeDataBase());
51                 }
52             });
53     }
54 
55     static private synchronized void initialize(TypeDataBase db) {
56         Type type = db.lookupType("HeapRegionSetBase");
57 
58         lengthField = type.getCIntegerField("_length");
59     }
60 
61     public long length() {
62         return lengthField.getValue(addr);
63     }
64 
65     public HeapRegionSetBase(Address addr) {
66         super(addr);
67     }
68 }
