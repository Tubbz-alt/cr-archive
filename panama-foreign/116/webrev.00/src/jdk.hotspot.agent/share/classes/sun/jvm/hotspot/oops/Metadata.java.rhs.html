 25 package sun.jvm.hotspot.oops;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 import sun.jvm.hotspot.memory.*;
 30 import sun.jvm.hotspot.utilities.*;
 31 import sun.jvm.hotspot.debugger.*;
 32 import sun.jvm.hotspot.runtime.*;
 33 import sun.jvm.hotspot.types.*;
 34 import sun.jvm.hotspot.utilities.Observable;
 35 import sun.jvm.hotspot.utilities.Observer;
 36 
 37 abstract public class Metadata extends VMObject {
 38   static {
 39     VM.registerVMInitializedObserver(new Observer() {
 40         public void update(Observable o, Object data) {
 41           initialize(VM.getVM().getTypeDataBase());
 42         }
 43       });
 44   }
 45 
 46   public Metadata(Address addr) {
 47     super(addr);
 48   }
 49 
 50   public static long alignSize(long size) {
 51     // natural word size.
 52     return VM.getVM().alignUp(size, VM.getVM().getBytesPerWord());
 53   }
 54 
 55   private static VirtualBaseConstructor<Metadata> metadataConstructor;
 56 
 57   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
 58     metadataConstructor = new VirtualBaseConstructor<Metadata>(db, db.lookupType("Metadata"), null, null);
 59     // Define an explicit mapping since the C++ and Java type names don't match.
 60     metadataConstructor.addMapping("Metadata", Metadata.class);
 61     metadataConstructor.addMapping("Klass", Klass.class);
 62     metadataConstructor.addMapping("InstanceKlass", InstanceKlass.class);
 63     metadataConstructor.addMapping("InstanceMirrorKlass", InstanceMirrorKlass.class);
 64     metadataConstructor.addMapping("InstanceRefKlass", InstanceRefKlass.class);
 65     metadataConstructor.addMapping("InstanceClassLoaderKlass", InstanceClassLoaderKlass.class);
 66     metadataConstructor.addMapping("TypeArrayKlass", TypeArrayKlass.class);
 67     metadataConstructor.addMapping("ObjArrayKlass", ObjArrayKlass.class);
 68     metadataConstructor.addMapping("Method", Method.class);
 69     metadataConstructor.addMapping("MethodData", MethodData.class);
 70     metadataConstructor.addMapping("ConstMethod", ConstMethod.class);
 71     metadataConstructor.addMapping("ConstantPool", ConstantPool.class);
 72     metadataConstructor.addMapping("ConstantPoolCache", ConstantPoolCache.class);
 73   }
 74 
 75   public static Metadata instantiateWrapperFor(Address addr) {
 76     return metadataConstructor.instantiateWrapperFor(addr);
 77   }
 78 
 79   public void iterate(MetadataVisitor visitor) {
 80     visitor.setObj(this);
 81     visitor.prologue();
 82     iterateFields(visitor);
 83     visitor.epilogue();
 84   }
 85 
 86   void iterateFields(MetadataVisitor visitor) {
 87   }
 88 
 89   abstract public void printValueOn(PrintStream tty);
 90   public void dumpReplayData(PrintStream out) {
 91       out.println("# Unknown Metadata");
 92   }
 93 
 94   public boolean isShared() {
 95     VM vm = VM.getVM();
 96     if (vm.isSharingEnabled()) {
 97       return MetaspaceObj.isShared(getAddress());
 98     }
 99     return false;
100   }
101 }
