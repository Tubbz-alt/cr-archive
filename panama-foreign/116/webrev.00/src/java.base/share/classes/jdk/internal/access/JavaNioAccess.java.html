26 package jdk.internal.access;
27 
28 import jdk.internal.access.foreign.MemorySegmentProxy;
29 import jdk.internal.access.foreign.UnmapperProxy;
30 import jdk.internal.misc.VM.BufferPool;
31 
32 import java.nio.Buffer;
33 import java.nio.ByteBuffer;
34 
35 public interface JavaNioAccess {
36 
37     /**
38      * Used by {@code jdk.internal.misc.VM}.
39      */
40     BufferPool getDirectBufferPool();
41 
42     /**
43      * Constructs a direct ByteBuffer referring to the block of memory starting
44      * at the given memory address and extending {@code cap} bytes.
45      * The {@code ob} parameter is an arbitrary object that is attached
46      * to the resulting buffer.
47      * Used by {@code jdk.internal.foreignMemorySegmentImpl}.
48      */
49     ByteBuffer newDirectByteBuffer(long addr, int cap, Object obj, MemorySegmentProxy segment);
50 
51     /**
52      * Constructs a mapped ByteBuffer referring to the block of memory starting
53      * at the given memory address and extending {@code cap} bytes.
54      * The {@code ob} parameter is an arbitrary object that is attached
55      * to the resulting buffer. The {@code sync} and {@code fd} parameters of the mapped
56      * buffer are derived from the {@code UnmapperProxy}.
57      * Used by {@code jdk.internal.foreignMemorySegmentImpl}.
58      */
59     ByteBuffer newMappedByteBuffer(UnmapperProxy unmapperProxy, long addr, int cap, Object obj, MemorySegmentProxy segment);
60 
61     /**
62      * Constructs an heap ByteBuffer with given backing array, offset, capacity and segment.
63      * Used by {@code jdk.internal.foreignMemorySegmentImpl}.
64      */
65     ByteBuffer newHeapByteBuffer(byte[] hb, int offset, int capacity, MemorySegmentProxy segment);
66 
67     /**
68      * Used by {@code jdk.internal.foreign.Utils}.
69      */
70     Object getBufferBase(ByteBuffer bb);
71 
72     /**
73      * Used by {@code jdk.internal.foreign.Utils}.
74      */
75     long getBufferAddress(ByteBuffer bb);
76 
77     /**
78      * Used by {@code jdk.internal.foreign.Utils}.
79      */
80     UnmapperProxy unmapper(ByteBuffer bb);
81 
82     /**
83      * Used by {@code jdk.internal.foreign.AbstractMemorySegmentImpl} and byte buffer var handle views.
84      */
85     MemorySegmentProxy bufferSegment(Buffer buffer);
86 }
