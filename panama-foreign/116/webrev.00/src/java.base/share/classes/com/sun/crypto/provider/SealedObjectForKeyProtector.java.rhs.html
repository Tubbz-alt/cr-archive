 26 package com.sun.crypto.provider;
 27 
 28 import jdk.internal.access.SharedSecrets;
 29 
 30 import java.io.*;
 31 import java.security.*;
 32 import javax.crypto.*;
 33 
 34 final class SealedObjectForKeyProtector extends SealedObject {
 35 
 36     @java.io.Serial
 37     static final long serialVersionUID = -3650226485480866989L;
 38 
 39     /**
 40      * The InputStreamFilter for a Key object inside this SealedObject. It can
 41      * be either provided as a {@link Security} property or a system property
 42      * (when provided as latter, it shadows the former). If the result of this
 43      * filter is {@link java.io.ObjectInputFilter.Status.UNDECIDED}, the system
 44      * level filter defined by jdk.serialFilter will be consulted. The value
 45      * of this property uses the same format of jdk.serialFilter.
 46      */
 47     private static final String KEY_SERIAL_FILTER = "jceks.key.serialFilter";
 48 
 49     SealedObjectForKeyProtector(Serializable object, Cipher c)
 50             throws IOException, IllegalBlockSizeException {
 51         super(object, c);
 52     }
 53 
 54     SealedObjectForKeyProtector(SealedObject so) {
 55         super(so);
 56     }
 57 
 58     AlgorithmParameters getParameters() {
 59         AlgorithmParameters params = null;
 60         if (super.encodedParams != null) {
 61             try {
 62                 params = AlgorithmParameters.getInstance("PBE",
 63                     SunJCE.getInstance());
 64                 params.init(super.encodedParams);
 65             } catch (NoSuchAlgorithmException nsae) {
 66                 throw new RuntimeException(
 67                     "SunJCE provider is not configured properly");
 68             } catch (IOException io) {
 69                 throw new RuntimeException("Parameter failure: "+
 70                     io.getMessage());
 71             }
 72         }
 73         return params;
 74     }
 75 
 76     final Key getKey(Cipher c, int maxLength)
 77             throws IOException, ClassNotFoundException, IllegalBlockSizeException,
 78             BadPaddingException {
 79 
 80         try (ObjectInputStream ois = SharedSecrets.getJavaxCryptoSealedObjectAccess()
 81                 .getExtObjectInputStream(this, c)) {
 82             AccessController.doPrivileged(
 83                     (PrivilegedAction<Void>) () -> {
 84                         ois.setObjectInputFilter(new DeserializationChecker(maxLength));
 85                         return null;
 86                     });
 87             try {
 88                 @SuppressWarnings("unchecked")
 89                 Key t = (Key) ois.readObject();
 90                 return t;
 91             } catch (InvalidClassException ice) {
 92                 String msg = ice.getMessage();
 93                 if (msg.contains("REJECTED")) {
 94                     throw new IOException("Rejected by the"
 95                             + " jceks.key.serialFilter or jdk.serialFilter"
 96                             + " property", ice);
 97                 } else {
 98                     throw ice;
 99                 }
100             }
101         }
102     }
103 
104     /**
105      * The filter for the content of a SealedObjectForKeyProtector.
106      *
107      * First, the jceks.key.serialFilter will be consulted. If the result
108      * is UNDECIDED, the system level jdk.serialFilter will be consulted.
109      */
110     private static class DeserializationChecker implements ObjectInputFilter {
111 
112         private static final ObjectInputFilter OWN_FILTER;
113 
114         static {
115             String prop = AccessController.doPrivileged(
116                     (PrivilegedAction<String>) () -> {
117                         String tmp = System.getProperty(KEY_SERIAL_FILTER);
118                         if (tmp != null) {
119                             return tmp;
120                         } else {
121                             return Security.getProperty(KEY_SERIAL_FILTER);
122                         }
123                     });
124             OWN_FILTER = prop == null
125                     ? null
126                     : ObjectInputFilter.Config.createFilter(prop);
127         }
128 
129         // Maximum possible length of anything inside
130         private final int maxLength;
131 
132         private DeserializationChecker(int maxLength) {
133             this.maxLength = maxLength;
134         }
135 
136         @Override
137         public ObjectInputFilter.Status checkInput(
138                 ObjectInputFilter.FilterInfo info) {
139 
140             if (info.arrayLength() > maxLength) {
141                 return Status.REJECTED;
142             }
143 
144             if (info.serialClass() == Object.class) {
145                 return Status.UNDECIDED;
146             }
147 
148             if (OWN_FILTER != null) {
149                 Status result = OWN_FILTER.checkInput(info);
150                 if (result != Status.UNDECIDED) {
151                     return result;
152                 }
153             }
154 
155             ObjectInputFilter defaultFilter =
156                     ObjectInputFilter.Config.getSerialFilter();
157             if (defaultFilter != null) {
158                 return defaultFilter.checkInput(info);
159             }
160 
161             return Status.UNDECIDED;
162         }
163     }
164 }
