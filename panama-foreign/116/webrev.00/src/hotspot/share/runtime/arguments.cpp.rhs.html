 119 bool PathString::set_value(const char *value) {
 120   if (_value != NULL) {
 121     FreeHeap(_value);
 122   }
 123   _value = AllocateHeap(strlen(value)+1, mtArguments);
 124   assert(_value != NULL, "Unable to allocate space for new path value");
 125   if (_value != NULL) {
 126     strcpy(_value, value);
 127   } else {
 128     // not able to allocate
 129     return false;
 130   }
 131   return true;
 132 }
 133 
 134 void PathString::append_value(const char *value) {
 135   char *sp;
 136   size_t len = 0;
 137   if (value != NULL) {
 138     len = strlen(value);
 139     if (_value != NULL) {
 140       len += strlen(_value);
 141     }
 142     sp = AllocateHeap(len+2, mtArguments);
 143     assert(sp != NULL, "Unable to allocate space for new append path value");
 144     if (sp != NULL) {
 145       if (_value != NULL) {
 146         strcpy(sp, _value);
 147         strcat(sp, os::path_separator());
 148         strcat(sp, value);
 149         FreeHeap(_value);
 150       } else {
 151         strcpy(sp, value);
 152       }
 153       _value = sp;
 154     }
 155   }
 156 }

 224 // Check if head of 'option' matches 'name', and sets 'tail' to the remaining
 225 // part of the option string.
 226 static bool match_option(const JavaVMOption *option, const char* name,
 227                          const char** tail) {
 228   size_t len = strlen(name);
 229   if (strncmp(option->optionString, name, len) == 0) {
 230     *tail = option->optionString + len;
 231     return true;
 232   } else {
 233     return false;
 234   }
 235 }
 236 
 237 // Check if 'option' matches 'name'. No "tail" is allowed.
 238 static bool match_option(const JavaVMOption *option, const char* name) {
 239   const char* tail = NULL;
 240   bool result = match_option(option, name, &tail);
 241   if (tail != NULL && *tail == '\0') {
 242     return result;
 243   } else {
 244     return false;
 245   }
 246 }
 247 
 248 // Return true if any of the strings in null-terminated array 'names' matches.
 249 // If tail_allowed is true, then the tail must begin with a colon; otherwise,
 250 // the option must match exactly.
 251 static bool match_option(const JavaVMOption* option, const char** names, const char** tail,
 252   bool tail_allowed) {
 253   for (/* empty */; *names != NULL; ++names) {
 254   if (match_option(option, *names, tail)) {
 255       if (**tail == '\0' || (tail_allowed && **tail == ':')) {
 256         return true;
 257       }
 258     }
 259   }
 260   return false;
 261 }

 329 // Return TRUE if option matches 'property', or 'property=', or 'property.'.
 330 static bool matches_property_suffix(const char* option, const char* property, size_t len) {
 331   return ((strncmp(option, property, len) == 0) &&
 332           (option[len] == '=' || option[len] == '.' || option[len] == '\0'));
 333 }

 335 // Return true if property starts with "jdk.module." and its ensuing chars match
 336 // any of the reserved module properties.
 337 // property should be passed without the leading "-D".
 338 bool Arguments::is_internal_module_property(const char* property) {
 339   assert((strncmp(property, "-D", 2) != 0), "Unexpected leading -D");
 340   if  (strncmp(property, MODULE_PROPERTY_PREFIX, MODULE_PROPERTY_PREFIX_LEN) == 0) {
 341     const char* property_suffix = property + MODULE_PROPERTY_PREFIX_LEN;
 342     if (matches_property_suffix(property_suffix, ADDEXPORTS, ADDEXPORTS_LEN) ||
 343         matches_property_suffix(property_suffix, ADDREADS, ADDREADS_LEN) ||
 344         matches_property_suffix(property_suffix, ADDOPENS, ADDOPENS_LEN) ||
 345         matches_property_suffix(property_suffix, PATCH, PATCH_LEN) ||
 346         matches_property_suffix(property_suffix, ADDMODS, ADDMODS_LEN) ||
 347         matches_property_suffix(property_suffix, LIMITMODS, LIMITMODS_LEN) ||
 348         matches_property_suffix(property_suffix, PATH, PATH_LEN) ||
 349         matches_property_suffix(property_suffix, UPGRADE_PATH, UPGRADE_PATH_LEN)) {
 350       return true;
 351     }
 352   }
 353   return false;
 354 }

 356 // Process java launcher properties.
 357 void Arguments::process_sun_java_launcher_properties(JavaVMInitArgs* args) {
 358   // See if sun.java.launcher or sun.java.launcher.is_altjvm is defined.
 359   // Must do this before setting up other system properties,
 360   // as some of them may depend on launcher type.
 361   for (int index = 0; index < args->nOptions; index++) {
 362     const JavaVMOption* option = args->options + index;
 363     const char* tail;
 364 
 365     if (match_option(option, "-Dsun.java.launcher=", &tail)) {
 366       process_java_launcher_argument(tail, option->extraInfo);
 367       continue;
 368     }
 369     if (match_option(option, "-Dsun.java.launcher.is_altjvm=", &tail)) {
 370       if (strcmp(tail, "true") == 0) {
 371         _sun_java_launcher_is_altjvm = true;
 372       }
 373       continue;
 374     }
 375   }
 376 }

 378 // Initialize system properties key and value.
 379 void Arguments::init_system_properties() {
 380 
 381   // Set up _system_boot_class_path which is not a property but
 382   // relies heavily on argument processing and the jdk.boot.class.path.append
 383   // property. It is used to store the underlying system boot class path.

 390   PropertyList_add(&_system_properties, new SystemProperty("jdk.debug", VM_Version::jdk_debug_level(),  false));
 391 
 392   // Initialize the vm.info now, but it will need updating after argument parsing.
 393   _vm_info = new SystemProperty("java.vm.info", VM_Version::vm_info_string(), true);
 394 
 395   // Following are JVMTI agent writable properties.
 396   // Properties values are set to NULL and they are
 397   // os specific they are initialized in os::init_system_properties_values().
 398   _sun_boot_library_path = new SystemProperty("sun.boot.library.path", NULL,  true);
 399   _java_library_path = new SystemProperty("java.library.path", NULL,  true);
 400   _java_home =  new SystemProperty("java.home", NULL,  true);
 401   _java_class_path = new SystemProperty("java.class.path", "",  true);
 402   // jdk.boot.class.path.append is a non-writeable, internal property.
 403   // It can only be set by either:
 404   //    - -Xbootclasspath/a:
 405   //    - AddToBootstrapClassLoaderSearch during JVMTI OnLoad phase
 406   _jdk_boot_class_path_append = new SystemProperty("jdk.boot.class.path.append", "", false, true);
 407 
 408   // Add to System Property list.
 409   PropertyList_add(&_system_properties, _sun_boot_library_path);
 410   PropertyList_add(&_system_properties, _java_library_path);
 411   PropertyList_add(&_system_properties, _java_home);
 412   PropertyList_add(&_system_properties, _java_class_path);
 413   PropertyList_add(&_system_properties, _jdk_boot_class_path_append);
 414   PropertyList_add(&_system_properties, _vm_info);
 415 
 416   // Set OS specific system properties values
 417   os::init_system_properties_values();
 418 }

 437 /*
 438  *  -XX argument processing:
 439  *
 440  *  -XX arguments are defined in several places, such as:
 441  *      globals.hpp, globals_<cpu>.hpp, globals_<os>.hpp, <compiler>_globals.hpp, or <gc>_globals.hpp.
 442  *  -XX arguments are parsed in parse_argument().
 443  *  -XX argument bounds checking is done in check_vm_args_consistency().
 444  *
 445  * Over time -XX arguments may change. There are mechanisms to handle common cases:
 446  *
 447  *      ALIASED: An option that is simply another name for another option. This is often
 448  *               part of the process of deprecating a flag, but not all aliases need
 449  *               to be deprecated.
 450  *
 451  *               Create an alias for an option by adding the old and new option names to the
 452  *               "aliased_jvm_flags" table. Delete the old variable from globals.hpp (etc).
 453  *
 454  *   DEPRECATED: An option that is supported, but a warning is printed to let the user know that
 455  *               support may be removed in the future. Both regular and aliased options may be
 456  *               deprecated.
 457  *
 458  *               Add a deprecation warning for an option (or alias) by adding an entry in the
 459  *               "special_jvm_flags" table and setting the "deprecated_in" field.
 460  *               Often an option "deprecated" in one major release will
 461  *               be made "obsolete" in the next. In this case the entry should also have its
 462  *               "obsolete_in" field set.
 463  *
 464  *     OBSOLETE: An option that has been removed (and deleted from globals.hpp), but is still accepted
 465  *               on the command line. A warning is printed to let the user know that option might not
 466  *               be accepted in the future.
 467  *
 468  *               Add an obsolete warning for an option by adding an entry in the "special_jvm_flags"
 469  *               table and setting the "obsolete_in" field.
 470  *
 471  *      EXPIRED: A deprecated or obsolete option that has an "accept_until" version less than or equal
 472  *               to the current JDK version. The system will flatly refuse to admit the existence of
 473  *               the flag. This allows a flag to die automatically over JDK releases.
 474  *
 475  *               Note that manual cleanup of expired options should be done at major JDK version upgrades:
 476  *                  - Newly expired options should be removed from the special_jvm_flags and aliased_jvm_flags tables.
 477  *                  - Newly obsolete or expired deprecated options should have their global variable
 478  *                    definitions removed (from globals.hpp, etc) and related implementations removed.
 479  *
 480  * Recommended approach for removing options:
 481  *
 482  * To remove options commonly used by customers (e.g. product -XX options), use
 483  * the 3-step model adding major release numbers to the deprecate, obsolete and expire columns.
 484  *
 485  * To remove internal options (e.g. diagnostic, experimental, develop options), use
 486  * a 2-step model adding major release numbers to the obsolete and expire columns.
 487  *
 488  * To change the name of an option, use the alias table as well as a 2-step
 489  * model adding major release numbers to the deprecate and expire columns.
 490  * Think twice about aliasing commonly used customer options.
 491  *
 492  * There are times when it is appropriate to leave a future release number as undefined.
 493  *
 494  * Tests:  Aliases should be tested in VMAliasOptions.java.
 495  *         Deprecated options should be tested in VMDeprecatedOptions.java.
 496  */

 712 // Verifies the correctness of the entries in the special_jvm_flags table.
 713 // If there is a semantic error (i.e. a bug in the table) such as the obsoletion
 714 // version being earlier than the deprecation version, then a warning is issued
 715 // and verification fails - by returning false. If it is detected that the table
 716 // is out of date, with respect to the current version, then ideally a warning is
 717 // issued but verification does not fail. This allows the VM to operate when the
 718 // version is first updated, without needing to update all the impacted flags at
 719 // the same time. In practice we can't issue the warning immediately when the version
 720 // is updated as it occurs for every test and some tests are not prepared to handle
 721 // unexpected output - see 8196739. Instead we only check if the table is up-to-date
 722 // if the check_globals flag is true, and in addition allow a grace period and only
 723 // check for stale flags when we hit build 25 (which is far enough into the 6 month
 724 // release cycle that all flag updates should have been processed, whilst still
 725 // leaving time to make the change before RDP2).
 726 // We use a gtest to call this, passing true, so that we can detect stale flags before
 727 // the end of the release cycle.

 733   for (size_t i = 0; special_jvm_flags[i].name != NULL; i++) {
 734     const SpecialFlag& flag = special_jvm_flags[i];
 735     if (lookup_special_flag(flag.name, i)) {
 736       warning("Duplicate special flag declaration \"%s\"", flag.name);
 737       success = false;
 738     }
 739     if (flag.deprecated_in.is_undefined() &&
 740         flag.obsolete_in.is_undefined()) {
 741       warning("Special flag entry \"%s\" must declare version deprecated and/or obsoleted in.", flag.name);
 742       success = false;
 743     }
 744 
 745     if (!flag.deprecated_in.is_undefined()) {
 746       if (!version_less_than(flag.deprecated_in, flag.obsolete_in)) {
 747         warning("Special flag entry \"%s\" must be deprecated before obsoleted.", flag.name);
 748         success = false;
 749       }
 750 
 751       if (!version_less_than(flag.deprecated_in, flag.expired_in)) {
 752         warning("Special flag entry \"%s\" must be deprecated before expired.", flag.name);
 753         success = false;
 754       }
 755     }
 756 
 757     if (!flag.obsolete_in.is_undefined()) {
 758       if (!version_less_than(flag.obsolete_in, flag.expired_in)) {
 759         warning("Special flag entry \"%s\" must be obsoleted before expired.", flag.name);
 760         success = false;
 761       }
 762 
 763       // if flag has become obsolete it should not have a "globals" flag defined anymore.
 764       if (check_globals && VM_Version::vm_build_number() >= SPECIAL_FLAG_VALIDATION_BUILD &&
 765           !version_less_than(JDK_Version::current(), flag.obsolete_in)) {
 766         if (JVMFlag::find_declared_flag(flag.name) != NULL) {
 767           warning("Global variable for obsolete special flag entry \"%s\" should be removed", flag.name);
 768           success = false;
 769         }
 770       }
 771 
 772     } else if (!flag.expired_in.is_undefined()) {
 773       warning("Special flag entry \"%s\" must be explicitly obsoleted before expired.", flag.name);
 774       success = false;
 775     }
 776 
 777     if (!flag.expired_in.is_undefined()) {
 778       // if flag has become expired it should not have a "globals" flag defined anymore.
 779       if (check_globals && VM_Version::vm_build_number() >= SPECIAL_FLAG_VALIDATION_BUILD &&
 780           !version_less_than(JDK_Version::current(), flag.expired_in)) {
 781         if (JVMFlag::find_declared_flag(flag.name) != NULL) {
 782           warning("Global variable for expired flag entry \"%s\" should be removed", flag.name);
 783           success = false;
 784         }
 785       }
 786     }
 787   }
 788   return success;

 792 // Parses a size specification string.
 793 bool Arguments::atojulong(const char *s, julong* result) {
 794   julong n = 0;
 795 
 796   // First char must be a digit. Don't allow negative numbers or leading spaces.
 797   if (!isdigit(*s)) {
 798     return false;
 799   }
 800 
 801   bool is_hex = (s[0] == '0' && (s[1] == 'x' || s[1] == 'X'));
 802   char* remainder;
 803   errno = 0;
 804   n = strtoull(s, &remainder, (is_hex ? 16 : 10));
 805   if (errno != 0) {
 806     return false;
 807   }
 808 
 809   // Fail if no number was read at all or if the remainder contains more than a single non-digit character.
 810   if (remainder == s || strlen(remainder) > 1) {
 811     return false;
 812   }
 813 
 814   switch (*remainder) {
 815     case 'T': case 't':
 816       *result = n * G * K;
 817       // Check for overflow.
 818       if (*result/((julong)G * K) != n) return false;
 819       return true;
 820     case 'G': case 'g':
 821       *result = n * G;
 822       if (*result/G != n) return false;
 823       return true;
 824     case 'M': case 'm':
 825       *result = n * M;
 826       if (*result/M != n) return false;
 827       return true;
 828     case 'K': case 'k':
 829       *result = n * K;
 830       if (*result/K != n) return false;
 831       return true;
 832     case '\0':
 833       *result = n;
 834       return true;
 835     default:
 836       return false;
 837   }
 838 }

 954     value = new_value;

 967   FREE_C_HEAP_ARRAY(char, value);
 968   // JVMFlag made its own copy, so I must delete my own temp. buffer.
 969   FREE_C_HEAP_ARRAY(char, free_this_too);
 970   return true;

1260 bool Arguments::process_argument(const char* arg,
1261                                  jboolean ignore_unrecognized,
1262                                  JVMFlag::Flags origin) {
1263   JDK_Version since = JDK_Version();
1264 
1265   if (parse_argument(arg, origin)) {
1266     return true;
1267   }
1268 
1269   // Determine if the flag has '+', '-', or '=' characters.
1270   bool has_plus_minus = (*arg == '+' || *arg == '-');
1271   const char* const argname = has_plus_minus ? arg + 1 : arg;
1272 
1273   size_t arg_len;
1274   const char* equal_sign = strchr(argname, '=');
1275   if (equal_sign == NULL) {
1276     arg_len = strlen(argname);
1277   } else {
1278     arg_len = equal_sign - argname;
1279   }
1280 
1281   // Only make the obsolete check for valid arguments.
1282   if (arg_len <= BUFLEN) {
1283     // Construct a string which consists only of the argument name without '+', '-', or '='.
1284     char stripped_argname[BUFLEN+1]; // +1 for '\0'
1285     jio_snprintf(stripped_argname, arg_len+1, "%s", argname); // +1 for '\0'
1286     if (is_obsolete_flag(stripped_argname, &since)) {
1287       char version[256];
1288       since.to_string(version, sizeof(version));
1289       warning("Ignoring option %s; support was removed in %s", stripped_argname, version);
1290       return true;
1291     }
1292 #ifndef PRODUCT
1293     else {
1294       const char* replacement;
1295       if ((replacement = removed_develop_logging_flag_name(stripped_argname)) != NULL){
1296         log_warning(arguments)("%s has been removed. Please use %s instead.",
1297                                stripped_argname,
1298                                replacement);
1299         return false;
1300       }
1301     }
1302 #endif //PRODUCT
1303   }
1304 
1305   // For locked flags, report a custom error message if available.
1306   // Otherwise, report the standard unrecognized VM option.
1307   const JVMFlag* found_flag = JVMFlag::find_declared_flag((const char*)argname, arg_len);
1308   if (found_flag != NULL) {
1309     char locked_message_buf[BUFLEN];
1310     JVMFlag::MsgType msg_type = found_flag->get_locked_message(locked_message_buf, BUFLEN);
1311     if (strlen(locked_message_buf) == 0) {
1312       if (found_flag->is_bool() && !has_plus_minus) {
1313         jio_fprintf(defaultStream::error_stream(),
1314           "Missing +/- setting for VM option '%s'\n", argname);
1315       } else if (!found_flag->is_bool() && has_plus_minus) {
1316         jio_fprintf(defaultStream::error_stream(),
1317           "Unexpected +/- setting in VM option '%s'\n", argname);
1318       } else {
1319         jio_fprintf(defaultStream::error_stream(),
1320           "Improperly specified VM option '%s'\n", argname);
1321       }
1322     } else {
1323 #ifdef PRODUCT
1324       bool mismatched = ((msg_type == JVMFlag::NOTPRODUCT_FLAG_BUT_PRODUCT_BUILD) ||
1325                          (msg_type == JVMFlag::DEVELOPER_FLAG_BUT_PRODUCT_BUILD));
1326       if (ignore_unrecognized && mismatched) {
1327         return true;
1328       }
1329 #endif
1330       jio_fprintf(defaultStream::error_stream(), "%s", locked_message_buf);
1331     }
1332   } else {
1333     if (ignore_unrecognized) {
1334       return true;
1335     }
1336     jio_fprintf(defaultStream::error_stream(),
1337                 "Unrecognized VM option '%s'\n", argname);
1338     JVMFlag* fuzzy_matched = JVMFlag::fuzzy_match((const char*)argname, arg_len, true);
1339     if (fuzzy_matched != NULL) {
1340       jio_fprintf(defaultStream::error_stream(),
1341                   "Did you mean '%s%s%s'? ",
1342                   (fuzzy_matched->is_bool()) ? "(+/-)" : "",
1343                   fuzzy_matched->_name,
1344                   (fuzzy_matched->is_bool()) ? "" : "=<value>");
1345     }
1346   }
1347 
1348   // allow for commandline "commenting out" options like -XX:#+Verbose
1349   return arg[0] == '#';
1350 }

1352 bool Arguments::process_settings_file(const char* file_name, bool should_exist, jboolean ignore_unrecognized) {
1353   FILE* stream = fopen(file_name, "rb");
1354   if (stream == NULL) {
1355     if (should_exist) {
1356       jio_fprintf(defaultStream::error_stream(),
1357                   "Could not open settings file %s\n", file_name);
1358       return false;
1359     } else {
1360       return true;
1361     }
1362   }
1363 
1364   char token[1024];
1365   int  pos = 0;
1366 
1367   bool in_white_space = true;
1368   bool in_comment     = false;
1369   bool in_quote       = false;
1370   char quote_c        = 0;
1371   bool result         = true;
1372 
1373   int c = getc(stream);
1374   while(c != EOF && pos < (int)(sizeof(token)-1)) {
1375     if (in_white_space) {
1376       if (in_comment) {
1377         if (c == '\n') in_comment = false;
1378       } else {
1379         if (c == '#') in_comment = true;
1380         else if (!isspace(c)) {
1381           in_white_space = false;
1382           token[pos++] = c;
1383         }
1384       }
1385     } else {
1386       if (c == '\n' || (!in_quote && isspace(c))) {
1387         // token ends at newline, or at unquoted whitespace
1388         // this allows a way to include spaces in string-valued options
1389         token[pos] = '\0';
1390         logOption(token);
1391         result &= process_argument(token, ignore_unrecognized, JVMFlag::CONFIG_FILE);
1392         build_jvm_flags(token);
1393         pos = 0;
1394         in_white_space = true;
1395         in_quote = false;
1396       } else if (!in_quote && (c == '\'' || c == '"')) {
1397         in_quote = true;
1398         quote_c = c;
1399       } else if (in_quote && (c == quote_c)) {
1400         in_quote = false;
1401       } else {
1402         token[pos++] = c;
1403       }
1404     }
1405     c = getc(stream);
1406   }
1407   if (pos > 0) {
1408     token[pos] = '\0';
1409     result &= process_argument(token, ignore_unrecognized, JVMFlag::CONFIG_FILE);
1410     build_jvm_flags(token);
1411   }
1412   fclose(stream);
1413   return result;
1414 }

1423 bool Arguments::add_property(const char* prop, PropertyWriteable writeable, PropertyInternal internal) {
1424   const char* eq = strchr(prop, '=');
1425   const char* key;
1426   const char* value = "";
1427 
1428   if (eq == NULL) {
1429     // property doesn't have a value, thus use passed string
1430     key = prop;
1431   } else {
1432     // property have a value, thus extract it and save to the
1433     // allocated string
1434     size_t key_len = eq - prop;
1435     char* tmp_key = AllocateHeap(key_len + 1, mtArguments);
1436 
1437     jio_snprintf(tmp_key, key_len + 1, "%s", prop);
1438     key = tmp_key;
1439 
1440     value = &prop[key_len + 1];
1441   }
1442 
1443   if (strcmp(key, "java.compiler") == 0) {
1444     process_java_compiler_argument(value);
1445     // Record value in Arguments, but let it get passed to Java.
1446   } else if (strcmp(key, "sun.java.launcher.is_altjvm") == 0) {
1447     // sun.java.launcher.is_altjvm property is
1448     // private and is processed in process_sun_java_launcher_properties();
1449     // the sun.java.launcher property is passed on to the java application
1450   } else if (strcmp(key, "sun.boot.library.path") == 0) {
1451     // append is true, writable is true, internal is false
1452     PropertyList_unique_add(&_system_properties, key, value, AppendProperty,
1453                             WriteableProperty, ExternalProperty);
1454   } else {
1455     if (strcmp(key, "sun.java.command") == 0) {
1456       char *old_java_command = _java_command;
1457       _java_command = os::strdup_check_oom(value, mtArguments);
1458       if (old_java_command != NULL) {
1459         os::free(old_java_command);
1460       }
1461     } else if (strcmp(key, "java.vendor.url.bug") == 0) {
1462       // If this property is set on the command line then its value will be
1463       // displayed in VM error logs as the URL at which to submit such logs.
1464       // Normally the URL displayed in error logs is different from the value
1465       // of this system property, so a different property should have been
1466       // used here, but we leave this as-is in case someone depends upon it.
1467       const char* old_java_vendor_url_bug = _java_vendor_url_bug;
1468       // save it in _java_vendor_url_bug, so JVM fatal error handler can access
1469       // its value without going through the property list or making a Java call.
1470       _java_vendor_url_bug = os::strdup_check_oom(value, mtArguments);
1471       if (old_java_vendor_url_bug != NULL) {
1472         os::free((void *)old_java_vendor_url_bug);
1473       }
1474     }
1475 
1476     // Create new property and add at the end of the list
1477     PropertyList_unique_add(&_system_properties, key, value, AddProperty, writeable, internal);
1478   }
1479 
1480   if (key != prop) {
1481     // SystemProperty copy passed value, thus free previously allocated
1482     // memory
1483     FreeHeap((void *)key);
1484   }
1485 
1486   return true;
1487 }

1637 size_t Arguments::max_heap_for_compressed_oops() {
1638   // Avoid sign flip.
1639   assert(OopEncodingHeapMax > (uint64_t)os::vm_page_size(), "Unusual page size");

1641   // keeping alignment constraints of the heap. To guarantee the latter, as the
1642   // NULL page is located before the heap, we pad the NULL page to the conservative
1643   // maximum alignment that the GC may ever impose upon the heap.
1644   size_t displacement_due_to_null_page = align_up((size_t)os::vm_page_size(),
1645                                                   _conservative_max_heap_alignment);
1646 
1647   LP64_ONLY(return OopEncodingHeapMax - displacement_due_to_null_page);
1648   NOT_LP64(ShouldNotReachHere(); return 0);
1649 }

1745 static const size_t DefaultHeapBaseMinAddress = HeapBaseMinAddress;
1746 
1747 void Arguments::set_heap_size() {
1748   julong phys_mem;
1749 
1750   // If the user specified one of these options, they
1751   // want specific memory sizing so do not limit memory
1752   // based on compressed oops addressability.
1753   // Also, memory limits will be calculated based on
1754   // available os physical memory, not our MaxRAM limit,
1755   // unless MaxRAM is also specified.
1756   bool override_coop_limit = (!FLAG_IS_DEFAULT(MaxRAMPercentage) ||
1757                            !FLAG_IS_DEFAULT(MaxRAMFraction) ||
1758                            !FLAG_IS_DEFAULT(MinRAMPercentage) ||
1759                            !FLAG_IS_DEFAULT(MinRAMFraction) ||
1760                            !FLAG_IS_DEFAULT(InitialRAMPercentage) ||
1761                            !FLAG_IS_DEFAULT(InitialRAMFraction) ||
1762                            !FLAG_IS_DEFAULT(MaxRAM));
1763   if (override_coop_limit) {
1764     if (FLAG_IS_DEFAULT(MaxRAM)) {
1765       phys_mem = os::physical_memory();
1766       FLAG_SET_ERGO(MaxRAM, (uint64_t)phys_mem);
1767     } else {
1768       phys_mem = (julong)MaxRAM;
1769     }
1770   } else {
1771     phys_mem = FLAG_IS_DEFAULT(MaxRAM) ? MIN2(os::physical_memory(), (julong)MaxRAM)
1772                                        : (julong)MaxRAM;
1773   }
1774 
1775 
1776   // Convert deprecated flags
1777   if (FLAG_IS_DEFAULT(MaxRAMPercentage) &&
1778       !FLAG_IS_DEFAULT(MaxRAMFraction))
1779     MaxRAMPercentage = 100.0 / MaxRAMFraction;
1780 
1781   if (FLAG_IS_DEFAULT(MinRAMPercentage) &&
1782       !FLAG_IS_DEFAULT(MinRAMFraction))
1783     MinRAMPercentage = 100.0 / MinRAMFraction;
1784 
1785   if (FLAG_IS_DEFAULT(InitialRAMPercentage) &&
1786       !FLAG_IS_DEFAULT(InitialRAMFraction))
1787     InitialRAMPercentage = 100.0 / InitialRAMFraction;
1788 
1789   // If the maximum heap size has not been set with -Xmx,
1790   // then set it as fraction of the size of physical memory,
1791   // respecting the maximum and minimum sizes of the heap.
1792   if (FLAG_IS_DEFAULT(MaxHeapSize)) {
1793     julong reasonable_max = (julong)((phys_mem * MaxRAMPercentage) / 100);
1794     const julong reasonable_min = (julong)((phys_mem * MinRAMPercentage) / 100);
1795     if (reasonable_min < MaxHeapSize) {
1796       // Small physical memory, so use a minimum fraction of it for the heap
1797       reasonable_max = reasonable_min;
1798     } else {
1799       // Not-small physical memory, so require a heap at least
1800       // as large as MaxHeapSize
1801       reasonable_max = MAX2(reasonable_max, (julong)MaxHeapSize);
1802     }
1803 
1804     if (!FLAG_IS_DEFAULT(ErgoHeapSizeLimit) && ErgoHeapSizeLimit != 0) {
1805       // Limit the heap size to ErgoHeapSizeLimit
1806       reasonable_max = MIN2(reasonable_max, (julong)ErgoHeapSizeLimit);
1807     }
1808 
1809 #ifdef _LP64
1810     if (UseCompressedOops) {
1811       // Limit the heap size to the maximum possible when using compressed oops
1812       julong max_coop_heap = (julong)max_heap_for_compressed_oops();
1813 
1814       // HeapBaseMinAddress can be greater than default but not less than.
1815       if (!FLAG_IS_DEFAULT(HeapBaseMinAddress)) {
1816         if (HeapBaseMinAddress < DefaultHeapBaseMinAddress) {
1817           // matches compressed oops printing flags
1818           log_debug(gc, heap, coops)("HeapBaseMinAddress must be at least " SIZE_FORMAT
1819                                      " (" SIZE_FORMAT "G) which is greater than value given " SIZE_FORMAT,
1820                                      DefaultHeapBaseMinAddress,
1821                                      DefaultHeapBaseMinAddress/G,
1822                                      HeapBaseMinAddress);
1823           FLAG_SET_ERGO(HeapBaseMinAddress, DefaultHeapBaseMinAddress);
1824         }
1825       }
1826 
1827       if (HeapBaseMinAddress + MaxHeapSize < max_coop_heap) {
1828         // Heap should be above HeapBaseMinAddress to get zero based compressed oops
1829         // but it should be not less than default MaxHeapSize.
1830         max_coop_heap -= HeapBaseMinAddress;
1831       }
1832 
1833       // If user specified flags prioritizing os physical
1834       // memory limits, then disable compressed oops if
1835       // limits exceed max_coop_heap and UseCompressedOops
1836       // was not specified.
1837       if (reasonable_max > max_coop_heap) {
1838         if (FLAG_IS_ERGO(UseCompressedOops) && override_coop_limit) {
1839           log_info(cds)("UseCompressedOops and UseCompressedClassPointers have been disabled due to"
1840             " max heap " SIZE_FORMAT " > compressed oop heap " SIZE_FORMAT ". "
1841             "Please check the setting of MaxRAMPercentage %5.2f."
1842             ,(size_t)reasonable_max, (size_t)max_coop_heap, MaxRAMPercentage);
1843           FLAG_SET_ERGO(UseCompressedOops, false);
1844           FLAG_SET_ERGO(UseCompressedClassPointers, false);
1845         } else {
1846           reasonable_max = MIN2(reasonable_max, max_coop_heap);
1847         }
1848       }
1849     }
1850 #endif // _LP64
1851 
1852     reasonable_max = limit_by_allocatable_memory(reasonable_max);
1853 
1854     if (!FLAG_IS_DEFAULT(InitialHeapSize)) {
1855       // An initial heap size was specified on the command line,
1856       // so be sure that the maximum size is consistent.  Done
1857       // after call to limit_by_allocatable_memory because that
1858       // method might reduce the allocation size.
1859       reasonable_max = MAX2(reasonable_max, (julong)InitialHeapSize);
1860     } else if (!FLAG_IS_DEFAULT(MinHeapSize)) {
1861       reasonable_max = MAX2(reasonable_max, (julong)MinHeapSize);
1862     }
1863 
1864     log_trace(gc, heap)("  Maximum heap size " SIZE_FORMAT, (size_t) reasonable_max);
1865     FLAG_SET_ERGO(MaxHeapSize, (size_t)reasonable_max);
1866   }
1867 
1868   // If the minimum or initial heap_size have not been set or requested to be set
1869   // ergonomically, set them accordingly.
1870   if (InitialHeapSize == 0 || MinHeapSize == 0) {
1871     julong reasonable_minimum = (julong)(OldSize + NewSize);
1872 
1873     reasonable_minimum = MIN2(reasonable_minimum, (julong)MaxHeapSize);
1874 
1875     reasonable_minimum = limit_by_allocatable_memory(reasonable_minimum);
1876 
1877     if (InitialHeapSize == 0) {
1878       julong reasonable_initial = (julong)((phys_mem * InitialRAMPercentage) / 100);
1879 
1880       reasonable_initial = MAX3(reasonable_initial, reasonable_minimum, (julong)MinHeapSize);
1881       reasonable_initial = MIN2(reasonable_initial, (julong)MaxHeapSize);
1882 
1883       reasonable_initial = limit_by_allocatable_memory(reasonable_initial);
1884 
1885       FLAG_SET_ERGO(InitialHeapSize, (size_t)reasonable_initial);
1886       log_trace(gc, heap)("  Initial heap size " SIZE_FORMAT, InitialHeapSize);
1887     }
1888     // If the minimum heap size has not been set (via -Xms or -XX:MinHeapSize),
1889     // synchronize with InitialHeapSize to avoid errors with the default value.
1890     if (MinHeapSize == 0) {
1891       FLAG_SET_ERGO(MinHeapSize, MIN2((size_t)reasonable_minimum, InitialHeapSize));
1892       log_trace(gc, heap)("  Minimum heap size " SIZE_FORMAT, MinHeapSize);
1893     }
1894   }
1895 }

1897 // This option inspects the machine and attempts to set various
1898 // parameters to be optimal for long-running, memory allocation
1899 // intensive jobs.  It is intended for machines with large
1900 // amounts of cpu and memory.
1901 jint Arguments::set_aggressive_heap_flags() {
1902   // initHeapSize is needed since _initial_heap_size is 4 bytes on a 32 bit
1903   // VM, but we may not be able to represent the total physical memory
1904   // available (like having 8gb of memory on a box but using a 32bit VM).
1905   // Thus, we need to make sure we're using a julong for intermediate
1906   // calculations.
1907   julong initHeapSize;
1908   julong total_memory = os::physical_memory();
1909 
1910   if (total_memory < (julong) 256 * M) {
1911     jio_fprintf(defaultStream::error_stream(),
1912             "You need at least 256mb of memory to use -XX:+AggressiveHeap\n");
1913     vm_exit(1);
1914   }
1915 
1916   // The heap size is half of available memory, or (at most)
1917   // all of possible memory less 160mb (leaving room for the OS
1918   // when using ISM).  This is the maximum; because adaptive sizing
1919   // is turned on below, the actual space used may be smaller.
1920 
1921   initHeapSize = MIN2(total_memory / (julong) 2,
1922           total_memory - (julong) 160 * M);
1923 
1924   initHeapSize = limit_by_allocatable_memory(initHeapSize);
1925 
1926   if (FLAG_IS_DEFAULT(MaxHeapSize)) {
1927     if (FLAG_SET_CMDLINE(MaxHeapSize, initHeapSize) != JVMFlag::SUCCESS) {
1928       return JNI_EINVAL;
1929     }
1930     if (FLAG_SET_CMDLINE(InitialHeapSize, initHeapSize) != JVMFlag::SUCCESS) {
1931       return JNI_EINVAL;
1932     }
1933     if (FLAG_SET_CMDLINE(MinHeapSize, initHeapSize) != JVMFlag::SUCCESS) {
1934       return JNI_EINVAL;
1935     }
1936   }
1937   if (FLAG_IS_DEFAULT(NewSize)) {
1938     // Make the young generation 3/8ths of the total heap.
1939     if (FLAG_SET_CMDLINE(NewSize,
1940             ((julong) MaxHeapSize / (julong) 8) * (julong) 3) != JVMFlag::SUCCESS) {
1941       return JNI_EINVAL;
1942     }
1943     if (FLAG_SET_CMDLINE(MaxNewSize, NewSize) != JVMFlag::SUCCESS) {
1944       return JNI_EINVAL;
1945     }
1946   }
1947 
1948 #if !defined(_ALLBSD_SOURCE) && !defined(AIX)  // UseLargePages is not yet supported on BSD and AIX.
1949   FLAG_SET_DEFAULT(UseLargePages, true);
1950 #endif
1951 
1952   // Increase some data structure sizes for efficiency
1953   if (FLAG_SET_CMDLINE(BaseFootPrintEstimate, MaxHeapSize) != JVMFlag::SUCCESS) {
1954     return JNI_EINVAL;
1955   }
1956   if (FLAG_SET_CMDLINE(ResizeTLAB, false) != JVMFlag::SUCCESS) {
1957     return JNI_EINVAL;
1958   }
1959   if (FLAG_SET_CMDLINE(TLABSize, 256 * K) != JVMFlag::SUCCESS) {
1960     return JNI_EINVAL;
1961   }
1962 
1963   // See the OldPLABSize comment below, but replace 'after promotion'
1964   // with 'after copying'.  YoungPLABSize is the size of the survivor
1965   // space per-gc-thread buffers.  The default is 4kw.
1966   if (FLAG_SET_CMDLINE(YoungPLABSize, 256 * K) != JVMFlag::SUCCESS) { // Note: this is in words
1967     return JNI_EINVAL;
1968   }
1969 
1970   // OldPLABSize is the size of the buffers in the old gen that
1971   // UseParallelGC uses to promote live data that doesn't fit in the
1972   // survivor spaces.  At any given time, there's one for each gc thread.
1973   // The default size is 1kw. These buffers are rarely used, since the
1974   // survivor spaces are usually big enough.  For specjbb, however, there
1975   // are occasions when there's lots of live data in the young gen
1976   // and we end up promoting some of it.  We don't have a definite
1977   // explanation for why bumping OldPLABSize helps, but the theory
1978   // is that a bigger PLAB results in retaining something like the
1979   // original allocation order after promotion, which improves mutator
1980   // locality.  A minor effect may be that larger PLABs reduce the
1981   // number of PLAB allocation events during gc.  The value of 8kw
1982   // was arrived at by experimenting with specjbb.
1983   if (FLAG_SET_CMDLINE(OldPLABSize, 8 * K) != JVMFlag::SUCCESS) { // Note: this is in words
1984     return JNI_EINVAL;
1985   }
1986 
1987   // Enable parallel GC and adaptive generation sizing
1988   if (FLAG_SET_CMDLINE(UseParallelGC, true) != JVMFlag::SUCCESS) {
1989     return JNI_EINVAL;
1990   }
1991 
1992   // Encourage steady state memory management
1993   if (FLAG_SET_CMDLINE(ThresholdTolerance, 100) != JVMFlag::SUCCESS) {
1994     return JNI_EINVAL;
1995   }
1996 
1997   // This appears to improve mutator locality
1998   if (FLAG_SET_CMDLINE(ScavengeBeforeFullGC, false) != JVMFlag::SUCCESS) {
1999     return JNI_EINVAL;
2000   }
2001 
2002   return JNI_OK;
2003 }

2360 // Parse -Xss memory string parameter and convert to ThreadStackSize in K.
2361 jint Arguments::parse_xss(const JavaVMOption* option, const char* tail, intx* out_ThreadStackSize) {
2362   // The min and max sizes match the values in globals.hpp, but scaled
2363   // with K. The values have been chosen so that alignment with page
2364   // size doesn't change the max value, which makes the conversions
2365   // back and forth between Xss value and ThreadStackSize value easier.
2366   // The values have also been chosen to fit inside a 32-bit signed type.
2367   const julong min_ThreadStackSize = 0;
2368   const julong max_ThreadStackSize = 1 * M;
2369 
2370   const julong min_size = min_ThreadStackSize * K;
2371   const julong max_size = max_ThreadStackSize * K;
2372 
2373   assert(is_aligned(max_size, os::vm_page_size()), "Implementation assumption");
2374 
2375   julong size = 0;
2376   ArgsRange errcode = parse_memory_size(tail, &size, min_size, max_size);
2377   if (errcode != arg_in_range) {
2378     bool silent = (option == NULL); // Allow testing to silence error messages
2379     if (!silent) {
2380       jio_fprintf(defaultStream::error_stream(),
2381                   "Invalid thread stack size: %s\n", option->optionString);
2382       describe_range_error(errcode);
2383     }
2384     return JNI_EINVAL;
2385   }
2386 
2387   // Internally track ThreadStackSize in units of 1024 bytes.
2388   const julong size_aligned = align_up(size, K);
2389   assert(size <= size_aligned,
2390          "Overflow: " JULONG_FORMAT " " JULONG_FORMAT,
2391          size, size_aligned);
2392 
2393   const julong size_in_K = size_aligned / K;
2394   assert(size_in_K < (julong)max_intx,
2395          "size_in_K doesn't fit in the type of ThreadStackSize: " JULONG_FORMAT,
2396          size_in_K);
2397 
2398   // Check that code expanding ThreadStackSize to a page aligned number of bytes won't overflow.
2399   const julong max_expanded = align_up(size_in_K * K, os::vm_page_size());
2400   assert(max_expanded < max_uintx && max_expanded >= size_in_K,
2401          "Expansion overflowed: " JULONG_FORMAT " " JULONG_FORMAT,
2402          max_expanded, size_in_K);
2403 
2404   *out_ThreadStackSize = (intx)size_in_K;
2405 
2406   return JNI_OK;
2407 }

3071 // Remove all empty paths from the app classpath (if IgnoreEmptyClassPaths is enabled)
3072 //
3073 // This is necessary because some apps like to specify classpath like -cp foo.jar:${XYZ}:bar.jar
3074 // in their start-up scripts. If XYZ is empty, the classpath will look like "-cp foo.jar::bar.jar".
3075 // Java treats such empty paths as if the user specified "-cp foo.jar:.:bar.jar". I.e., an empty
3076 // path is treated as the current directory.
3077 //
3078 // This causes problems with CDS, which requires that all directories specified in the classpath
3079 // must be empty. In most cases, applications do NOT want to load classes from the current
3080 // directory anyway. Adding -XX:+IgnoreEmptyClassPaths will make these applications' start-up
3081 // scripts compatible with CDS.
3082 void Arguments::fix_appclasspath() {
3083   if (IgnoreEmptyClassPaths) {
3084     const char separator = *os::path_separator();
3085     const char* src = _java_class_path->value();
3086 
3087     // skip over all the leading empty paths
3088     while (*src == separator) {
3089       src ++;
3090     }
3091 
3092     char* copy = os::strdup_check_oom(src, mtArguments);
3093 
3094     // trim all trailing empty paths
3095     for (char* tail = copy + strlen(copy) - 1; tail >= copy && *tail == separator; tail--) {
3096       *tail = '\0';
3097     }
3098 
3099     char from[3] = {separator, separator, '\0'};
3100     char to  [2] = {separator, '\0'};
3101     while (StringUtils::replace_no_expand(copy, from, to) > 0) {
3102       // Keep replacing "::" -> ":" until we have no more "::" (non-windows)
3103       // Keep replacing ";;" -> ";" until we have no more ";;" (windows)
3104     }
3105 
3106     _java_class_path->set_writeable_value(copy);
3107     FreeHeap(copy); // a copy was made by set_value, so don't need this anymore
3108   }

3238 // deleted on the destruction of the ScopedVMInitArgs object.
3239 class ScopedVMInitArgs : public StackObj {
3240  private:

3253     _is_set = false;
3254     _vm_options_file_arg = NULL;
3255   }
3256 
3257   // Populates the JavaVMInitArgs object represented by this
3258   // ScopedVMInitArgs object with the arguments in options.  The
3259   // allocated memory is deleted by the destructor.  If this method
3260   // returns anything other than JNI_OK, then this object is in a
3261   // partially constructed state, and should be abandoned.
3262   jint set_args(GrowableArray<JavaVMOption>* options) {
3263     _is_set = true;
3264     JavaVMOption* options_arr = NEW_C_HEAP_ARRAY_RETURN_NULL(
3265         JavaVMOption, options->length(), mtArguments);
3266     if (options_arr == NULL) {

3276         _args.nOptions = i;
3277         return JNI_ENOMEM;
3278       }

3306     }

3371 jint Arguments::parse_vm_options_file(const char* file_name, ScopedVMInitArgs* vm_args) {
3372   // read file into buffer
3373   int fd = ::open(file_name, O_RDONLY);
3374   if (fd < 0) {
3375     jio_fprintf(defaultStream::error_stream(),
3376                 "Could not open options file '%s'\n",
3377                 file_name);
3378     return JNI_ERR;
3379   }
3380 
3381   struct stat stbuf;
3382   int retcode = os::stat(file_name, &stbuf);
3383   if (retcode != 0) {
3384     jio_fprintf(defaultStream::error_stream(),
3385                 "Could not stat options file '%s'\n",
3386                 file_name);
3387     os::close(fd);
3388     return JNI_ERR;
3389   }
3390 
3391   if (stbuf.st_size == 0) {
3392     // tell caller there is no option data and that is ok
3393     os::close(fd);
3394     return JNI_OK;
3395   }
3396 
3397   // '+ 1' for NULL termination even with max bytes
3398   size_t bytes_alloc = stbuf.st_size + 1;
3399 
3400   char *buf = NEW_C_HEAP_ARRAY_RETURN_NULL(char, bytes_alloc, mtArguments);
3401   if (NULL == buf) {
3402     jio_fprintf(defaultStream::error_stream(),
3403                 "Could not allocate read buffer for options file parse\n");
3404     os::close(fd);
3405     return JNI_ENOMEM;
3406   }
3407 
3408   memset(buf, 0, bytes_alloc);
3409 
3410   // Fill buffer
3411   ssize_t bytes_read = os::read(fd, (void *)buf, (unsigned)bytes_alloc);
3412   os::close(fd);
3413   if (bytes_read < 0) {
3414     FREE_C_HEAP_ARRAY(char, buf);
3415     jio_fprintf(defaultStream::error_stream(),
3416                 "Could not read options file '%s'\n", file_name);
3417     return JNI_ERR;
3418   }
3419 
3420   if (bytes_read == 0) {
3421     // tell caller there is no option data and that is ok
3422     FREE_C_HEAP_ARRAY(char, buf);
3423     return JNI_OK;
3424   }
3425 
3426   retcode = parse_options_buffer(file_name, buf, bytes_read, vm_args);
3427 
3428   FREE_C_HEAP_ARRAY(char, buf);
3429   return retcode;

3432 jint Arguments::parse_options_buffer(const char* name, char* buffer, const size_t buf_len, ScopedVMInitArgs* vm_args) {
3433   GrowableArray<JavaVMOption> *options = new (ResourceObj::C_HEAP, mtArguments) GrowableArray<JavaVMOption>(2, true);    // Construct option array
3434 
3435   // some pointers to help with parsing
3436   char *buffer_end = buffer + buf_len;
3437   char *opt_hd = buffer;
3438   char *wrt = buffer;
3439   char *rd = buffer;
3440 
3441   // parse all options
3442   while (rd < buffer_end) {
3443     // skip leading white space from the input string
3444     while (rd < buffer_end && isspace(*rd)) {
3445       rd++;
3446     }
3447 
3448     if (rd >= buffer_end) {
3449       break;
3450     }
3451 
3452     // Remember this is where we found the head of the token.
3453     opt_hd = wrt;
3454 
3455     // Tokens are strings of non white space characters separated
3456     // by one or more white spaces.
3457     while (rd < buffer_end && !isspace(*rd)) {
3458       if (*rd == '\'' || *rd == '"') {      // handle a quoted string
3459         int quote = *rd;                    // matching quote to look for
3460         rd++;                               // don't copy open quote
3461         while (rd < buffer_end && *rd != quote) {
3462                                             // include everything (even spaces)
3463                                             // up until the close quote
3464           *wrt++ = *rd++;                   // copy to option string
3465         }
3466 
3467         if (rd < buffer_end) {
3468           rd++;                             // don't copy close quote
3469         } else {
3470                                             // did not see closing quote
3471           jio_fprintf(defaultStream::error_stream(),
3472                       "Unmatched quote in %s\n", name);
3473           delete options;
3474           return JNI_ERR;
3475         }
3476       } else {
3477         *wrt++ = *rd++;                     // copy to option string
3478       }
3479     }
3480 
3481     // steal a white space character and set it to NULL
3482     *wrt++ = '\0';
3483     // We now have a complete token
3484 
3485     JavaVMOption option;
3486     option.optionString = opt_hd;
3487     option.extraInfo = NULL;
3488 
3489     options->append(option);                // Fill in option
3490 
3491     rd++;  // Advance to next character
3492   }
3493 
3494   // Fill out JavaVMInitArgs structure.
3495   jint status = vm_args->set_args(options);
3496 
3497   delete options;
3498   return status;
3499 }

3715 jint Arguments::match_special_option_and_act(const JavaVMInitArgs* args,
3716                                              ScopedVMInitArgs* args_out) {
3717   // Remaining part of option string
3718   const char* tail;
3719   ScopedVMInitArgs vm_options_file_args(args_out->container_name());
3720 
3721   for (int index = 0; index < args->nOptions; index++) {
3722     const JavaVMOption* option = args->options + index;
3723     if (match_option(option, "-XX:Flags=", &tail)) {
3724       Arguments::set_jvm_flags_file(tail);
3725       continue;
3726     }
3727     if (match_option(option, "-XX:VMOptionsFile=", &tail)) {
3728       if (vm_options_file_args.found_vm_options_file_arg()) {
3729         jio_fprintf(defaultStream::error_stream(),
3730                     "The option '%s' is already specified in the options "
3731                     "container '%s' so the specification of '%s' in the "
3732                     "same options container is an error.\n",
3733                     vm_options_file_args.vm_options_file_arg(),
3734                     vm_options_file_args.container_name(),
3735                     option->optionString);
3736         return JNI_EINVAL;
3737       }
3738       vm_options_file_args.set_vm_options_file_arg(option->optionString);
3739       // If there's a VMOptionsFile, parse that
3740       jint code = insert_vm_options_file(args, tail, index,
3741                                          &vm_options_file_args, args_out);
3742       if (code != JNI_OK) {
3743         return code;
3744       }
3745       args_out->set_vm_options_file_arg(vm_options_file_args.vm_options_file_arg());
3746       if (args_out->is_set()) {
3747         // The VMOptions file inserted some options so switch 'args'
3748         // to the new set of options, and continue processing which
3749         // preserves "last option wins" semantics.
3750         args = args_out->get();
3751         // The first option from the VMOptionsFile replaces the
3752         // current option.  So we back track to process the
3753         // replacement option.
3754         index--;
3755       }
3756       continue;
3757     }
3758     if (match_option(option, "-XX:+PrintVMOptions")) {
3759       PrintVMOptions = true;
3760       continue;
3761     }
3762     if (match_option(option, "-XX:-PrintVMOptions")) {
3763       PrintVMOptions = false;
3764       continue;
3765     }
3766     if (match_option(option, "-XX:+IgnoreUnrecognizedVMOptions")) {
3767       IgnoreUnrecognizedVMOptions = true;
3768       continue;
3769     }
3770     if (match_option(option, "-XX:-IgnoreUnrecognizedVMOptions")) {
3771       IgnoreUnrecognizedVMOptions = false;
3772       continue;
3773     }
3774     if (match_option(option, "-XX:+PrintFlagsInitial")) {
3775       JVMFlag::printFlags(tty, false);
3776       vm_exit(0);
3777     }
3778     if (match_option(option, "-XX:NativeMemoryTracking", &tail)) {
3779 #if INCLUDE_NMT
3780       // The launcher did not setup nmt environment variable properly.
3781       if (!MemTracker::check_launcher_nmt_support(tail)) {
3782         warning("Native Memory Tracking did not setup properly, using wrong launcher?");
3783       }
3784 
3785       // Verify if nmt option is valid.
3786       if (MemTracker::verify_nmt_option()) {
3787         // Late initialization, still in single-threaded mode.
3788         if (MemTracker::tracking_level() >= NMT_summary) {
3789           MemTracker::init();
3790         }
3791       } else {
3792         vm_exit_during_initialization("Syntax error, expecting -XX:NativeMemoryTracking=[off|summary|detail]", NULL);

4302 // Copies src into buf, replacing "%%" with "%" and "%p" with pid
4303 // Returns true if all of the source pointed by src has been copied over to
4304 // the destination buffer pointed by buf. Otherwise, returns false.
4305 // Notes:
4306 // 1. If the length (buflen) of the destination buffer excluding the
4307 // NULL terminator character is not long enough for holding the expanded
4308 // pid characters, it also returns false instead of returning the partially
4309 // expanded one.
4310 // 2. The passed in "buflen" should be large enough to hold the null terminator.
4311 bool Arguments::copy_expand_pid(const char* src, size_t srclen,
4312                                 char* buf, size_t buflen) {
4313   const char* p = src;
4314   char* b = buf;
4315   const char* src_end = &src[srclen];
4316   char* buf_end = &buf[buflen - 1];
4317 
4318   while (p < src_end && b < buf_end) {
4319     if (*p == '%') {
4320       switch (*(++p)) {
4321       case '%':         // "%%" ==> "%"
4322         *b++ = *p++;
4323         break;
4324       case 'p':  {       //  "%p" ==> current process id
4325         // buf_end points to the character before the last character so
4326         // that we could write '\0' to the end of the buffer.
4327         size_t buf_sz = buf_end - b + 1;
4328         int ret = jio_snprintf(b, buf_sz, "%d", os::current_process_id());
4329 
4330         // if jio_snprintf fails or the buffer is not long enough to hold
4331         // the expanded pid, returns false.
4332         if (ret < 0 || ret >= (int)buf_sz) {
4333           return false;
4334         } else {
4335           b += ret;
4336           assert(*b == '\0', "fail in copy_expand_pid");
4337           if (p == src_end && b == buf_end + 1) {
4338             // reach the end of the buffer.
4339             return true;
4340           }
4341         }
4342         p++;
4343         break;
4344       }
4345       default :
4346         *b++ = '%';
4347       }
4348     } else {
4349       *b++ = *p++;
4350     }
4351   }
4352   *b = '\0';
4353   return (p == src_end); // return false if not all of the source was copied
4354 }
