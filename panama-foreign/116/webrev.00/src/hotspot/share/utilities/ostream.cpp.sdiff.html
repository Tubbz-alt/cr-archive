 540 long fileStream::fileSize() {
 541   long size = -1;
 542   if (_file != NULL) {
 543     long pos = ::ftell(_file);
 544     if (pos < 0) return pos;
 545     if (::fseek(_file, 0, SEEK_END) == 0) {
 546       size = ::ftell(_file);
 547     }
 548     ::fseek(_file, pos, SEEK_SET);
 549   }
 550   return size;
 551 }
