 162 // Simple helper to see if the caller of a runtime stub which
 163 // entered the VM has been deoptimized
 164 
 165 static bool caller_is_deopted() {
 166   JavaThread* thread = JavaThread::current();
 167   RegisterMap reg_map(thread, false);
 168   frame runtime_frame = thread->last_frame();
 169   frame caller_frame = runtime_frame.sender(&reg_map);
 170   assert(caller_frame.is_compiled_frame(), "must be compiled");
 171   return caller_frame.is_deoptimized_frame();
 172 }
 173 
 174 // Stress deoptimization
 175 static void deopt_caller() {
 176   if ( !caller_is_deopted()) {
 177     JavaThread* thread = JavaThread::current();
 178     RegisterMap reg_map(thread, false);
 179     frame runtime_frame = thread->last_frame();
 180     frame caller_frame = runtime_frame.sender(&reg_map);
 181     Deoptimization::deoptimize_frame(thread, caller_frame.id());
 182     assert(caller_is_deopted(), "Must be deoptimized");
 183   }
 184 }

 196 CodeBlob* Runtime1::generate_blob(BufferBlob* buffer_blob, int stub_id, const char* name, bool expect_oop_map, StubAssemblerCodeGenClosure* cl) {
 197   ResourceMark rm;
 198   // create code buffer for code storage
 199   CodeBuffer code(buffer_blob);
 200 
 201   OopMapSet* oop_maps;
 202   int frame_size;
 203   bool must_gc_arguments;
 204 
 205   Compilation::setup_code_buffer(&code, 0);
 206 
 207   // create assembler for code generation
 208   StubAssembler* sasm = new StubAssembler(&code, name, stub_id);
 209   // generate code for runtime stub
 210   oop_maps = cl->generate_code(sasm);
 211   assert(oop_maps == NULL || sasm->frame_size() != no_frame_size,
 212          "if stub has an oop map it must have a valid frame size");
 213   assert(!expect_oop_map || oop_maps != NULL, "must have an oopmap");
 214 
 215   // align so printing shows nop's instead of random code at the end (SimpleStubs are aligned)
 216   sasm->align(BytesPerWord);
 217   // make sure all code is in code buffer
 218   sasm->flush();
 219 
 220   frame_size = sasm->frame_size();
 221   must_gc_arguments = sasm->must_gc_arguments();
 222   // create blob - distinguish a few special cases
 223   CodeBlob* blob = RuntimeStub::new_runtime_stub(name,
 224                                                  &code,
 225                                                  CodeOffsets::frame_never_safe,
 226                                                  frame_size,
 227                                                  oop_maps,
 228                                                  must_gc_arguments);
 229   assert(blob != NULL, "blob must exist");
 230   return blob;
 231 }

 233 void Runtime1::generate_blob_for(BufferBlob* buffer_blob, StubID id) {
 234   assert(0 <= id && id < number_of_ids, "illegal stub id");
 235   bool expect_oop_map = true;
 236 #ifdef ASSERT
 237   // Make sure that stubs that need oopmaps have them
 238   switch (id) {
 239     // These stubs don't need to have an oopmap
 240   case dtrace_object_alloc_id:
 241   case slow_subtype_check_id:
 242   case fpu2long_stub_id:
 243   case unwind_exception_id:
 244   case counter_overflow_id:
 245 #if defined(SPARC) || defined(PPC32)
 246   case handle_exception_nofpu_id:  // Unused on sparc
 247 #endif
 248     expect_oop_map = false;
 249     break;
 250   default:
 251     break;
 252   }
 253 #endif
 254   StubIDStubAssemblerCodeGenClosure cl(id);
 255   CodeBlob* blob = generate_blob(buffer_blob, id, name_for(id), expect_oop_map, &cl);
 256   // install blob
 257   _blobs[id] = blob;
 258 }

 260 void Runtime1::initialize(BufferBlob* blob) {
 261   // platform-dependent initialization
 262   initialize_pd();
 263   // generate stubs
 264   for (int id = 0; id < number_of_ids; id++) generate_blob_for(blob, (StubID)id);
 265   // printing
 266 #ifndef PRODUCT
 267   if (PrintSimpleStubs) {
 268     ResourceMark rm;
 269     for (int id = 0; id < number_of_ids; id++) {
 270       _blobs[id]->print();
 271       if (_blobs[id]->oop_maps() != NULL) {
 272         _blobs[id]->oop_maps()->print();
 273       }
 274     }
 275   }
 276 #endif
 277   BarrierSetC1* bs = BarrierSet::barrier_set()->barrier_set_c1();
 278   bs->generate_c1_runtime_stubs(blob);
 279 }

 429 // counter_overflow() is called from within C1-compiled methods. The enclosing method is the method
 430 // associated with the top activation record. The inlinee (that is possibly included in the enclosing
 431 // method) method oop is passed as an argument. In order to do that it is embedded in the code as
 432 // a constant.
 433 static nmethod* counter_overflow_helper(JavaThread* THREAD, int branch_bci, Method* m) {
 434   nmethod* osr_nm = NULL;
 435   methodHandle method(THREAD, m);
 436 
 437   RegisterMap map(THREAD, false);
 438   frame fr =  THREAD->last_frame().sender(&map);
 439   nmethod* nm = (nmethod*) fr.cb();
 440   assert(nm!= NULL && nm->is_nmethod(), "Sanity check");
 441   methodHandle enclosing_method(THREAD, nm->method());
 442 
 443   CompLevel level = (CompLevel)nm->comp_level();
 444   int bci = InvocationEntryBci;
 445   if (branch_bci != InvocationEntryBci) {
 446     // Compute destination bci
 447     address pc = method()->code_base() + branch_bci;
 448     Bytecodes::Code branch = Bytecodes::code_at(method(), pc);
 449     int offset = 0;
 450     switch (branch) {
 451       case Bytecodes::_if_icmplt: case Bytecodes::_iflt:
 452       case Bytecodes::_if_icmpgt: case Bytecodes::_ifgt:
 453       case Bytecodes::_if_icmple: case Bytecodes::_ifle:
 454       case Bytecodes::_if_icmpge: case Bytecodes::_ifge:
 455       case Bytecodes::_if_icmpeq: case Bytecodes::_if_acmpeq: case Bytecodes::_ifeq:
 456       case Bytecodes::_if_icmpne: case Bytecodes::_if_acmpne: case Bytecodes::_ifne:
 457       case Bytecodes::_ifnull: case Bytecodes::_ifnonnull: case Bytecodes::_goto:
 458         offset = (int16_t)Bytes::get_Java_u2(pc + 1);
 459         break;
 460       case Bytecodes::_goto_w:
 461         offset = Bytes::get_Java_u4(pc + 1);
 462         break;
 463       default: ;
 464     }
 465     bci = branch_bci + offset;
 466   }
 467   assert(!HAS_PENDING_EXCEPTION, "Should not have any exceptions pending");
 468   osr_nm = CompilationPolicy::policy()->event(enclosing_method, method, branch_bci, bci, level, nm, THREAD);
 469   assert(!HAS_PENDING_EXCEPTION, "Event handler should not throw any exceptions");
 470   return osr_nm;
 471 }
 472 
 473 JRT_BLOCK_ENTRY(address, Runtime1::counter_overflow(JavaThread* thread, int bci, Method* method))
 474   nmethod* osr_nm;
 475   JRT_BLOCK
 476     osr_nm = counter_overflow_helper(thread, bci, method);
 477     if (osr_nm != NULL) {
 478       RegisterMap map(thread, false);
 479       frame fr =  thread->last_frame().sender(&map);
 480       Deoptimization::deoptimize_frame(thread, fr.id());
 481     }
 482   JRT_BLOCK_END
 483   return NULL;
 484 JRT_END

 486 extern void vm_exit(int code);
 487 
 488 // Enter this method from compiled code handler below. This is where we transition
 489 // to VM mode. This is done as a helper routine so that the method called directly
 490 // from compiled code does not have to transition to VM. This allows the entry
 491 // method to see if the nmethod that we have just looked up a handler for has
 492 // been deoptimized while we were in the vm. This simplifies the assembly code
 493 // cpu directories.
 494 //
 495 // We are entering here from exception stub (via the entry method below)
 496 // If there is a compiled exception handler in this method, we will continue there;
 497 // otherwise we will unwind the stack and continue at the caller of top frame method
 498 // Note: we enter in Java using a special JRT wrapper. This wrapper allows us to
 499 // control the area where we can allow a safepoint. After we exit the safepoint area we can
 500 // check to see if the handler we are going to return is now in a nmethod that has
 501 // been deoptimized. If that is the case we return the deopt blob
 502 // unpack_with_exception entry instead. This makes life for the exception blob easier
 503 // because making that same check and diverting is painful from assembly language.
 504 JRT_ENTRY_NO_ASYNC(static address, exception_handler_for_pc_helper(JavaThread* thread, oopDesc* ex, address pc, nmethod*& nm))
 505   // Reset method handle flag.
 506   thread->set_is_method_handle_return(false);
 507 
 508   Handle exception(thread, ex);
 509   nm = CodeCache::find_nmethod(pc);
 510   assert(nm != NULL, "this is not an nmethod");
 511   // Adjust the pc as needed/
 512   if (nm->is_deopt_pc(pc)) {
 513     RegisterMap map(thread, false);
 514     frame exception_frame = thread->last_frame().sender(&map);
 515     // if the frame isn't deopted then pc must not correspond to the caller of last_frame
 516     assert(exception_frame.is_deoptimized_frame(), "must be deopted");
 517     pc = exception_frame.pc();
 518   }
 519 #ifdef ASSERT
 520   assert(exception.not_null(), "NULL exceptions should be handled by throw_exception");
 521   // Check that exception is a subclass of Throwable, otherwise we have a VerifyError
 522   if (!(exception->is_a(SystemDictionary::Throwable_klass()))) {
 523     if (ExitVMOnVerifyError) vm_exit(-1);
 524     ShouldNotReachHere();
 525   }
 526 #endif
 527 
 528   // Check the stack guard pages and reenable them if necessary and there is
 529   // enough space on the stack to do so.  Use fast exceptions only if the guard
 530   // pages are enabled.
 531   bool guard_pages_enabled = thread->stack_guards_enabled();
 532   if (!guard_pages_enabled) guard_pages_enabled = thread->reguard_stack();
 533 
 534   if (JvmtiExport::can_post_on_exceptions()) {
 535     // To ensure correct notification of exception catches and throws
 536     // we have to deoptimize here.  If we attempted to notify the
 537     // catches and throws during this exception lookup it's possible
 538     // we could deoptimize on the way out of the VM and end back in
 539     // the interpreter at the throw site.  This would result in double
 540     // notifications since the interpreter would also notify about
 541     // these same catches and throws as it unwound the frame.
 542 
 543     RegisterMap reg_map(thread);
 544     frame stub_frame = thread->last_frame();
 545     frame caller_frame = stub_frame.sender(&reg_map);
 546 
 547     // We don't really want to deoptimize the nmethod itself since we
 548     // can actually continue in the exception handler ourselves but I
 549     // don't see an easy way to have the desired effect.
 550     Deoptimization::deoptimize_frame(thread, caller_frame.id());
 551     assert(caller_is_deopted(), "Must be deoptimized");
 552 
 553     return SharedRuntime::deopt_blob()->unpack_with_exception_in_tls();
 554   }
 555 
 556   // ExceptionCache is used only for exceptions at call sites and not for implicit exceptions
 557   if (guard_pages_enabled) {
 558     address fast_continuation = nm->handler_for_exception_and_pc(exception, pc);
 559     if (fast_continuation != NULL) {
 560       // Set flag if return address is a method handle call site.
 561       thread->set_is_method_handle_return(nm->is_method_handle_return(pc));
 562       return fast_continuation;
 563     }
 564   }

 567   // the current method.  Otherwise (guard pages disabled), force an unwind and
 568   // skip the exception cache update (i.e., just leave continuation==NULL).
 569   address continuation = NULL;
 570   if (guard_pages_enabled) {
 571 
 572     // New exception handling mechanism can support inlined methods
 573     // with exception handlers since the mappings are from PC to PC
 574 
 575     // debugging support
 576     // tracing
 577     if (log_is_enabled(Info, exceptions)) {
 578       ResourceMark rm;
 579       stringStream tempst;
 580       assert(nm->method() != NULL, "Unexpected NULL method()");
 581       tempst.print("compiled method <%s>\n"
 582                    " at PC" INTPTR_FORMAT " for thread " INTPTR_FORMAT,
 583                    nm->method()->print_value_string(), p2i(pc), p2i(thread));
 584       Exceptions::log_exception(exception, tempst.as_string());
 585     }
 586     // for AbortVMOnException flag
 587     Exceptions::debug_check_abort(exception);
 588 
 589     // Clear out the exception oop and pc since looking up an
 590     // exception handler can cause class loading, which might throw an
 591     // exception and those fields are expected to be clear during
 592     // normal bytecode execution.
 593     thread->clear_exception_oop_and_pc();
 594 
 595     bool recursive_exception = false;
 596     continuation = SharedRuntime::compute_compiled_exc_handler(nm, pc, exception, false, false, recursive_exception);
 597     // If an exception was thrown during exception dispatch, the exception oop may have changed
 598     thread->set_exception_oop(exception());
 599     thread->set_exception_pc(pc);
 600 
 601     // the exception cache is used only by non-implicit exceptions
 602     // Update the exception cache only when there didn't happen
 603     // another exception during the computation of the compiled
 604     // exception handler. Checking for exception oop equality is not
 605     // sufficient because some exceptions are pre-allocated and reused.
 606     if (continuation != NULL && !recursive_exception) {
 607       nm->add_handler_for_exception_and_pc(exception, pc, continuation);
 608     }
 609   }
 610 
 611   thread->set_vm_result(exception());
 612   // Set flag if return address is a method handle call site.
 613   thread->set_is_method_handle_return(nm->is_method_handle_return(pc));
 614 
 615   if (log_is_enabled(Info, exceptions)) {
 616     ResourceMark rm;
 617     log_info(exceptions)("Thread " PTR_FORMAT " continuing at PC " PTR_FORMAT
 618                          " for exception thrown at PC " PTR_FORMAT,
 619                          p2i(thread), p2i(continuation), p2i(pc));
 620   }
 621 
 622   return continuation;
 623 JRT_END

 774 //
 775 // This routine patches sites where a class wasn't loaded or
 776 // initialized at the time the code was generated.  It handles
 777 // references to classes, fields and forcing of initialization.  Most
 778 // of the cases are straightforward and involving simply forcing
 779 // resolution of a class, rewriting the instruction stream with the
 780 // needed constant and replacing the call in this function with the
 781 // patched code.  The case for static field is more complicated since
 782 // the thread which is in the process of initializing a class can
 783 // access it's static fields but other threads can't so the code
 784 // either has to deoptimize when this case is detected or execute a
 785 // check that the current thread is the initializing thread.  The
 786 // current
 787 //
 788 // Patches basically look like this:
 789 //
 790 //
 791 // patch_site: jmp patch stub     ;; will be patched
 792 // continue:   ...
 793 //             ...
 794 //             ...
 795 //             ...
 796 //
 797 // They have a stub which looks like this:
 798 //
 799 //             ;; patch body
 800 //             movl <const>, reg           (for class constants)
 801 //        <or> movl [reg1 + <const>], reg  (for field offsets)
 802 //        <or> movl reg, [reg1 + <const>]  (for field offsets)
 803 //             <being_init offset> <bytes to copy> <bytes to skip>
 804 // patch_stub: call Runtime1::patch_code (through a runtime stub)
 805 //             jmp patch_site
 806 //
 807 //
 808 // A normal patch is done by rewriting the patch body, usually a move,
 809 // and then copying it into place over top of the jmp instruction
 810 // being careful to flush caches and doing it in an MP-safe way.  The
 811 // constants following the patch body are used to find various pieces
 812 // of the patch relative to the call site for Runtime1::patch_code.
 813 // The case for getstatic and putstatic is more complicated because
 814 // getstatic and putstatic have special semantics when executing while
 815 // the class is being initialized.  getstatic/putstatic on a class
 816 // which is being_initialized may be executed by the initializing
 817 // thread but other threads have to block when they execute it.  This
 818 // is accomplished in compiled code by executing a test of the current
 819 // thread against the initializing thread of the class.  It's emitted
 820 // as boilerplate in their stub which allows the patched code to be
 821 // executed before it's copied back into the main body of the nmethod.
 822 //
 823 // being_init: get_thread(<tmp reg>
 824 //             cmpl [reg1 + <init_thread_offset>], <tmp reg>
 825 //             jne patch_stub
 826 //             movl [reg1 + <const>], reg  (for field offsets)  <or>
 827 //             movl reg, [reg1 + <const>]  (for field offsets)
 828 //             jmp continue
 829 //             <being_init offset> <bytes to copy> <bytes to skip>
 830 // patch_stub: jmp Runtim1::patch_code (through a runtime stub)
 831 //             jmp patch_site
 832 //
 833 // If the class is being initialized the patch body is rewritten and
 834 // the patch site is rewritten to jump to being_init, instead of
 835 // patch_stub.  Whenever this code is executed it checks the current
 836 // thread against the intializing thread so other threads will enter
 837 // the runtime and end up blocked waiting the class to finish
 838 // initializing inside the calls to resolve_field below.  The
 839 // initializing class will continue on it's way.  Once the class is
 840 // fully_initialized, the intializing_thread of the class becomes
 841 // NULL, so the next thread to execute this code will fail the test,
 842 // call into patch_code and complete the patching process by copying
 843 // the patch body back into the main part of the nmethod and resume
 844 // executing.
 845 
 846 // NB:
 847 //
 848 // Patchable instruction sequences inherently exhibit race conditions,
 849 // where thread A is patching an instruction at the same time thread B
 850 // is executing it.  The algorithms we use ensure that any observation
 851 // that B can make on any intermediate states during A's patching will
 852 // always end up with a correct outcome.  This is easiest if there are
 853 // few or no intermediate states.  (Some inline caches have two
 854 // related instructions that must be patched in tandem.  For those,
 855 // intermediate states seem to be unavoidable, but we will get the
 856 // right answer from all possible observation orders.)
 857 //
 858 // When patching the entry instruction at the head of a method, or a
 859 // linkable call instruction inside of a method, we try very hard to
 860 // use a patch sequence which executes as a single memory transaction.
 861 // This means, in practice, that when thread A patches an instruction,
 862 // it should patch a 32-bit or 64-bit word that somehow overlaps the
 863 // instruction or is contained in it.  We believe that memory hardware
 864 // will never break up such a word write, if it is naturally aligned
 865 // for the word being written.  We also know that some CPUs work very
 866 // hard to create atomic updates even of naturally unaligned words,
 867 // but we don't want to bet the farm on this always working.
 868 //
 869 // Therefore, if there is any chance of a race condition, we try to
 870 // patch only naturally aligned words, as single, full-word writes.
 871 
 872 JRT_ENTRY(void, Runtime1::patch_code(JavaThread* thread, Runtime1::StubID stub_id ))
 873   NOT_PRODUCT(_patch_code_slowcase_cnt++;)
 874 
 875   ResourceMark rm(thread);
 876   RegisterMap reg_map(thread, false);
 877   frame runtime_frame = thread->last_frame();
 878   frame caller_frame = runtime_frame.sender(&reg_map);
 879 
 880   // last java frame on stack
 881   vframeStream vfst(thread, true);
 882   assert(!vfst.at_end(), "Java frame must exist");
 883 
 884   methodHandle caller_method(THREAD, vfst.method());
 885   // Note that caller_method->code() may not be same as caller_code because of OSR's
 886   // Note also that in the presence of inlining it is not guaranteed
 887   // that caller_method() == caller_code->method()
 888 
 889   int bci = vfst.bci();
 890   Bytecodes::Code code = caller_method()->java_code_at(bci);
 891 
 892   // this is used by assertions in the access_field_patching_id
 893   BasicType patch_field_type = T_ILLEGAL;
 894   bool deoptimize_for_volatile = false;
 895   bool deoptimize_for_atomic = false;
 896   int patch_field_offset = -1;
 897   Klass* init_klass = NULL; // klass needed by load_klass_patching code
 898   Klass* load_klass = NULL; // klass needed by load_klass_patching code
 899   Handle mirror(THREAD, NULL);                    // oop needed by load_mirror_patching code
 900   Handle appendix(THREAD, NULL);                  // oop needed by appendix_patching code
 901   bool load_klass_or_mirror_patch_id =
 902     (stub_id == Runtime1::load_klass_patching_id || stub_id == Runtime1::load_mirror_patching_id);
 903 
 904   if (stub_id == Runtime1::access_field_patching_id) {
 905 
 906     Bytecode_field field_access(caller_method, bci);
 907     fieldDescriptor result; // initialize class if needed
 908     Bytecodes::Code code = field_access.code();
 909     constantPoolHandle constants(THREAD, caller_method->constants());
 910     LinkResolver::resolve_field_access(result, constants, field_access.index(), caller_method, Bytecodes::java_code(code), CHECK);
 911     patch_field_offset = result.offset();
 912 
 913     // If we're patching a field which is volatile then at compile it
 914     // must not have been know to be volatile, so the generated code
 915     // isn't correct for a volatile reference.  The nmethod has to be
 916     // deoptimized so that the code can be regenerated correctly.
 917     // This check is only needed for access_field_patching since this
 918     // is the path for patching field offsets.  load_klass is only
 919     // used for patching references to oops which don't need special
 920     // handling in the volatile case.
 921 
 922     deoptimize_for_volatile = result.access_flags().is_volatile();
 923 
 924     // If we are patching a field which should be atomic, then
 925     // the generated code is not correct either, force deoptimizing.
 926     // We need to only cover T_LONG and T_DOUBLE fields, as we can
 927     // break access atomicity only for them.
 928 
 929     // Strictly speaking, the deoptimization on 64-bit platforms
 930     // is unnecessary, and T_LONG stores on 32-bit platforms need
 931     // to be handled by special patching code when AlwaysAtomicAccesses
 932     // becomes product feature. At this point, we are still going
 933     // for the deoptimization for consistency against volatile
 934     // accesses.
 935 
 936     patch_field_type = result.field_type();
 937     deoptimize_for_atomic = (AlwaysAtomicAccesses && (patch_field_type == T_DOUBLE || patch_field_type == T_LONG));
 938 
 939   } else if (load_klass_or_mirror_patch_id) {
 940     Klass* k = NULL;
 941     switch (code) {
 942       case Bytecodes::_putstatic:
 943       case Bytecodes::_getstatic:
 944         { Klass* klass = resolve_field_return_klass(caller_method, bci, CHECK);
 945           init_klass = klass;
 946           mirror = Handle(THREAD, klass->java_mirror());
 947         }
 948         break;
 949       case Bytecodes::_new:
 950         { Bytecode_new bnew(caller_method(), caller_method->bcp_from(bci));
 951           k = caller_method->constants()->klass_at(bnew.index(), CHECK);
 952         }
 953         break;
 954       case Bytecodes::_multianewarray:
 955         { Bytecode_multianewarray mna(caller_method(), caller_method->bcp_from(bci));
 956           k = caller_method->constants()->klass_at(mna.index(), CHECK);
 957         }
 958         break;
 959       case Bytecodes::_instanceof:
 960         { Bytecode_instanceof io(caller_method(), caller_method->bcp_from(bci));
 961           k = caller_method->constants()->klass_at(io.index(), CHECK);
 962         }
 963         break;
 964       case Bytecodes::_checkcast:
 965         { Bytecode_checkcast cc(caller_method(), caller_method->bcp_from(bci));
 966           k = caller_method->constants()->klass_at(cc.index(), CHECK);
 967         }
 968         break;
 969       case Bytecodes::_anewarray:
 970         { Bytecode_anewarray anew(caller_method(), caller_method->bcp_from(bci));
 971           Klass* ek = caller_method->constants()->klass_at(anew.index(), CHECK);
 972           k = ek->array_klass(CHECK);
 973         }
 974         break;
 975       case Bytecodes::_ldc:
 976       case Bytecodes::_ldc_w:
 977         {
 978           Bytecode_loadconstant cc(caller_method, bci);
 979           oop m = cc.resolve_constant(CHECK);
 980           mirror = Handle(THREAD, m);
 981         }
 982         break;
 983       default: fatal("unexpected bytecode for load_klass_or_mirror_patch_id");
 984     }
 985     load_klass = k;
 986   } else if (stub_id == load_appendix_patching_id) {
 987     Bytecode_invoke bytecode(caller_method, bci);
 988     Bytecodes::Code bc = bytecode.invoke_code();
 989 
 990     CallInfo info;
 991     constantPoolHandle pool(thread, caller_method->constants());
 992     int index = bytecode.index();
 993     LinkResolver::resolve_invoke(info, Handle(), pool, index, bc, CHECK);
 994     switch (bc) {
 995       case Bytecodes::_invokehandle: {
 996         int cache_index = ConstantPool::decode_cpcache_index(index, true);
 997         assert(cache_index >= 0 && cache_index < pool->cache()->length(), "unexpected cache index");
 998         ConstantPoolCacheEntry* cpce = pool->cache()->entry_at(cache_index);
 999         cpce->set_method_handle(pool, info);
1000         appendix = Handle(THREAD, cpce->appendix_if_resolved(pool)); // just in case somebody already resolved the entry
1001         break;
1002       }
1003       case Bytecodes::_invokedynamic: {
1004         ConstantPoolCacheEntry* cpce = pool->invokedynamic_cp_cache_entry_at(index);
1005         cpce->set_dynamic_call(pool, info);
1006         appendix = Handle(THREAD, cpce->appendix_if_resolved(pool)); // just in case somebody already resolved the entry
1007         break;
1008       }
1009       default: fatal("unexpected bytecode for load_appendix_patching_id");
1010     }
1011   } else {
1012     ShouldNotReachHere();
1013   }
1014 
1015   if (deoptimize_for_volatile || deoptimize_for_atomic) {
1016     // At compile time we assumed the field wasn't volatile/atomic but after
1017     // loading it turns out it was volatile/atomic so we have to throw the
1018     // compiled code out and let it be regenerated.
1019     if (TracePatching) {
1020       if (deoptimize_for_volatile) {
1021         tty->print_cr("Deoptimizing for patching volatile field reference");
1022       }
1023       if (deoptimize_for_atomic) {
1024         tty->print_cr("Deoptimizing for patching atomic field reference");
1025       }
1026     }
1027 
1028     // It's possible the nmethod was invalidated in the last
1029     // safepoint, but if it's still alive then make it not_entrant.
1030     nmethod* nm = CodeCache::find_nmethod(caller_frame.pc());
1031     if (nm != NULL) {
1032       nm->make_not_entrant();
1033     }
1034 
1035     Deoptimization::deoptimize_frame(thread, caller_frame.id());
1036 
1037     // Return to the now deoptimized frame.
1038   }
1039 
1040   // Now copy code back
1041 
1042   {
1043     MutexLocker ml_patch (THREAD, Patching_lock, Mutex::_no_safepoint_check_flag);
1044     //
1045     // Deoptimization may have happened while we waited for the lock.
1046     // In that case we don't bother to do any patching we just return
1047     // and let the deopt happen
1048     if (!caller_is_deopted()) {
1049       NativeGeneralJump* jump = nativeGeneralJump_at(caller_frame.pc());
1050       address instr_pc = jump->jump_destination();
1051       NativeInstruction* ni = nativeInstruction_at(instr_pc);
1052       if (ni->is_jump() ) {
1053         // the jump has not been patched yet
1054         // The jump destination is slow case and therefore not part of the stubs
1055         // (stubs are only for StaticCalls)
1056 
1057         // format of buffer
1058         //    ....
1059         //    instr byte 0     <-- copy_buff
1060         //    instr byte 1
1061         //    ..
1062         //    instr byte n-1
1063         //      n
1064         //    ....             <-- call destination
1065 
1066         address stub_location = caller_frame.pc() + PatchingStub::patch_info_offset();
1067         unsigned char* byte_count = (unsigned char*) (stub_location - 1);
1068         unsigned char* byte_skip = (unsigned char*) (stub_location - 2);
1069         unsigned char* being_initialized_entry_offset = (unsigned char*) (stub_location - 3);
1070         address copy_buff = stub_location - *byte_skip - *byte_count;
1071         address being_initialized_entry = stub_location - *being_initialized_entry_offset;
1072         if (TracePatching) {
1073           ttyLocker ttyl;
1074           tty->print_cr(" Patching %s at bci %d at address " INTPTR_FORMAT "  (%s)", Bytecodes::name(code), bci,
1075                         p2i(instr_pc), (stub_id == Runtime1::access_field_patching_id) ? "field" : "klass");
1076           nmethod* caller_code = CodeCache::find_nmethod(caller_frame.pc());
1077           assert(caller_code != NULL, "nmethod not found");
1078 
1079           // NOTE we use pc() not original_pc() because we already know they are
1080           // identical otherwise we'd have never entered this block of code
1081 
1082           const ImmutableOopMap* map = caller_code->oop_map_for_return_address(caller_frame.pc());
1083           assert(map != NULL, "null check");
1084           map->print();
1085           tty->cr();
1086 
1087           Disassembler::decode(copy_buff, copy_buff + *byte_count, tty);
1088         }
1089         // depending on the code below, do_patch says whether to copy the patch body back into the nmethod
1090         bool do_patch = true;
1091         if (stub_id == Runtime1::access_field_patching_id) {
1092           // The offset may not be correct if the class was not loaded at code generation time.
1093           // Set it now.
1094           NativeMovRegMem* n_move = nativeMovRegMem_at(copy_buff);
1095           assert(n_move->offset() == 0 || (n_move->offset() == 4 && (patch_field_type == T_DOUBLE || patch_field_type == T_LONG)), "illegal offset for type");
1096           assert(patch_field_offset >= 0, "illegal offset");
1097           n_move->add_offset_in_bytes(patch_field_offset);
1098         } else if (load_klass_or_mirror_patch_id) {
1099           // If a getstatic or putstatic is referencing a klass which
1100           // isn't fully initialized, the patch body isn't copied into
1101           // place until initialization is complete.  In this case the
1102           // patch site is setup so that any threads besides the
1103           // initializing thread are forced to come into the VM and
1104           // block.
1105           do_patch = (code != Bytecodes::_getstatic && code != Bytecodes::_putstatic) ||
1106                      InstanceKlass::cast(init_klass)->is_initialized();
1107           NativeGeneralJump* jump = nativeGeneralJump_at(instr_pc);
1108           if (jump->jump_destination() == being_initialized_entry) {
1109             assert(do_patch == true, "initialization must be complete at this point");
1110           } else {
1111             // patch the instruction <move reg, klass>
1112             NativeMovConstReg* n_copy = nativeMovConstReg_at(copy_buff);
1113 
1114             assert(n_copy->data() == 0 ||
1115                    n_copy->data() == (intptr_t)Universe::non_oop_word(),
1116                    "illegal init value");
1117             if (stub_id == Runtime1::load_klass_patching_id) {
1118               assert(load_klass != NULL, "klass not set");
1119               n_copy->set_data((intx) (load_klass));
1120             } else {
1121               assert(mirror() != NULL, "klass not set");
1122               // Don't need a G1 pre-barrier here since we assert above that data isn't an oop.
1123               n_copy->set_data(cast_from_oop<intx>(mirror()));
1124             }
1125 
1126             if (TracePatching) {
1127               Disassembler::decode(copy_buff, copy_buff + *byte_count, tty);
1128             }
1129           }
1130         } else if (stub_id == Runtime1::load_appendix_patching_id) {
1131           NativeMovConstReg* n_copy = nativeMovConstReg_at(copy_buff);
1132           assert(n_copy->data() == 0 ||
1133                  n_copy->data() == (intptr_t)Universe::non_oop_word(),
1134                  "illegal init value");
1135           n_copy->set_data(cast_from_oop<intx>(appendix()));
1136 
1137           if (TracePatching) {
1138             Disassembler::decode(copy_buff, copy_buff + *byte_count, tty);
1139           }
1140         } else {
1141           ShouldNotReachHere();
1142         }
1143 
1144 #if defined(SPARC) || defined(PPC32)
1145         if (load_klass_or_mirror_patch_id ||
1146             stub_id == Runtime1::load_appendix_patching_id) {
1147           // Update the location in the nmethod with the proper
1148           // metadata.  When the code was generated, a NULL was stuffed
1149           // in the metadata table and that table needs to be update to
1150           // have the right value.  On intel the value is kept
1151           // directly in the instruction instead of in the metadata
1152           // table, so set_data above effectively updated the value.
1153           nmethod* nm = CodeCache::find_nmethod(instr_pc);
1154           assert(nm != NULL, "invalid nmethod_pc");
1155           RelocIterator mds(nm, copy_buff, copy_buff + 1);
1156           bool found = false;
1157           while (mds.next() && !found) {
1158             if (mds.type() == relocInfo::oop_type) {
1159               assert(stub_id == Runtime1::load_mirror_patching_id ||
1160                      stub_id == Runtime1::load_appendix_patching_id, "wrong stub id");
1161               oop_Relocation* r = mds.oop_reloc();
1162               oop* oop_adr = r->oop_addr();
1163               *oop_adr = stub_id == Runtime1::load_mirror_patching_id ? mirror() : appendix();
1164               r->fix_oop_relocation();
1165               found = true;
1166             } else if (mds.type() == relocInfo::metadata_type) {
1167               assert(stub_id == Runtime1::load_klass_patching_id, "wrong stub id");
1168               metadata_Relocation* r = mds.metadata_reloc();
1169               Metadata** metadata_adr = r->metadata_addr();
1170               *metadata_adr = load_klass;
1171               r->fix_metadata_relocation();
1172               found = true;
1173             }
1174           }
1175           assert(found, "the metadata must exist!");
1176         }
1177 #endif
1178         if (do_patch) {
1179           // replace instructions
1180           // first replace the tail, then the call
1181 #ifdef ARM
1182           if((load_klass_or_mirror_patch_id ||
1183               stub_id == Runtime1::load_appendix_patching_id) &&
1184               nativeMovConstReg_at(copy_buff)->is_pc_relative()) {
1185             nmethod* nm = CodeCache::find_nmethod(instr_pc);
1186             address addr = NULL;
1187             assert(nm != NULL, "invalid nmethod_pc");
1188             RelocIterator mds(nm, copy_buff, copy_buff + 1);
1189             while (mds.next()) {
1190               if (mds.type() == relocInfo::oop_type) {
1191                 assert(stub_id == Runtime1::load_mirror_patching_id ||
1192                        stub_id == Runtime1::load_appendix_patching_id, "wrong stub id");
1193                 oop_Relocation* r = mds.oop_reloc();
1194                 addr = (address)r->oop_addr();
1195                 break;
1196               } else if (mds.type() == relocInfo::metadata_type) {
1197                 assert(stub_id == Runtime1::load_klass_patching_id, "wrong stub id");
1198                 metadata_Relocation* r = mds.metadata_reloc();
1199                 addr = (address)r->metadata_addr();
1200                 break;
1201               }
1202             }
1203             assert(addr != NULL, "metadata relocation must exist");
1204             copy_buff -= *byte_count;
1205             NativeMovConstReg* n_copy2 = nativeMovConstReg_at(copy_buff);
1206             n_copy2->set_pc_relative_offset(addr, instr_pc);
1207           }
1208 #endif
1209 
1210           for (int i = NativeGeneralJump::instruction_size; i < *byte_count; i++) {
1211             address ptr = copy_buff + i;
1212             int a_byte = (*ptr) & 0xFF;
1213             address dst = instr_pc + i;
1214             *(unsigned char*)dst = (unsigned char) a_byte;
1215           }
1216           ICache::invalidate_range(instr_pc, *byte_count);
1217           NativeGeneralJump::replace_mt_safe(instr_pc, copy_buff);
1218 
1219           if (load_klass_or_mirror_patch_id ||
1220               stub_id == Runtime1::load_appendix_patching_id) {
1221             relocInfo::relocType rtype =
1222               (stub_id == Runtime1::load_klass_patching_id) ?
1223                                    relocInfo::metadata_type :
1224                                    relocInfo::oop_type;
1225             // update relocInfo to metadata
1226             nmethod* nm = CodeCache::find_nmethod(instr_pc);
1227             assert(nm != NULL, "invalid nmethod_pc");
1228 
1229             // The old patch site is now a move instruction so update
1230             // the reloc info so that it will get updated during
1231             // future GCs.
1232             RelocIterator iter(nm, (address)instr_pc, (address)(instr_pc + 1));
1233             relocInfo::change_reloc_info_for_address(&iter, (address) instr_pc,
1234                                                      relocInfo::none, rtype);
1235 #ifdef SPARC
1236             // Sparc takes two relocations for an metadata so update the second one.
1237             address instr_pc2 = instr_pc + NativeMovConstReg::add_offset;
1238             RelocIterator iter2(nm, instr_pc2, instr_pc2 + 1);
1239             relocInfo::change_reloc_info_for_address(&iter2, (address) instr_pc2,
1240                                                      relocInfo::none, rtype);
1241 #endif
1242 #ifdef PPC32
1243           { address instr_pc2 = instr_pc + NativeMovConstReg::lo_offset;
1244             RelocIterator iter2(nm, instr_pc2, instr_pc2 + 1);
1245             relocInfo::change_reloc_info_for_address(&iter2, (address) instr_pc2,
1246                                                      relocInfo::none, rtype);
1247           }
1248 #endif
1249           }
1250 
1251         } else {
1252           ICache::invalidate_range(copy_buff, *byte_count);
1253           NativeGeneralJump::insert_unconditional(instr_pc, being_initialized_entry);
1254         }
1255       }
1256     }
1257   }
1258 
1259   // If we are patching in a non-perm oop, make sure the nmethod
1260   // is on the right list.
1261   {
1262     MutexLocker ml_code (THREAD, CodeCache_lock, Mutex::_no_safepoint_check_flag);
1263     nmethod* nm = CodeCache::find_nmethod(caller_frame.pc());
1264     guarantee(nm != NULL, "only nmethods can contain non-perm oops");
1265 
1266     // Since we've patched some oops in the nmethod,
1267     // (re)register it with the heap.
1268     Universe::heap()->register_nmethod(nm);
1269   }
