30 /**
31  * A tree node for <a href="http://www.ecma-international.org/ecma-262/6.0/#sec-template-literals">template literal strings</a>.
32  *
33  * For example:
34  * <pre>
35  * `This is a String with ${computed} values in it`
36  * </pre>
37  *
38  * @deprecated Nashorn JavaScript script engine and APIs, and the jjs tool
39  * are deprecated with the intent to remove them in a future release.
40  *
41  * @since 9
42  *
43  */
44 @Deprecated(since="11", forRemoval=true)
45 public interface TemplateLiteralTree extends ExpressionTree {
46     /**
47      * Returns the list of expressions in this template string
48      *
49      * @return the list of expressions in this template string
50      */
51     List<? extends ExpressionTree> getExpressions();
52 }
