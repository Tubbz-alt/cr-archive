26 package jdk.nashorn.internal.ir;
27 
28 import jdk.nashorn.internal.codegen.Label;
29 
30 /**
31  * This class represents a node from which control flow can execute
32  * a {@code break} statement
33  */
34 public interface BreakableNode extends LexicalContextNode, JoinPredecessor, Labels {
35     /**
36      * Ensure that any labels in this breakable node are unique so
37      * that new jumps won't go to old parts of the tree. Used for
38      * example for cloning finally blocks
39      *
40      * @param lc the lexical context
41      * @return node after labels have been made unique
42      */
43     public abstract Node ensureUniqueLabels(final LexicalContext lc);
44 
45     /**
46      * Check whether this can be broken out from without using a label,
47      * e.g. everything but Blocks, basically
48      * @return true if breakable without label
49      */
50     public boolean isBreakableWithoutLabel();
51 
52     /**
53      * Return the break label, i.e. the location to go to on break.
54      * @return the break label
55      */
56     public Label getBreakLabel();
57 
58 }
