26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.ir.BinaryNode;
29 
30 final class CompoundAssignmentTreeImpl extends ExpressionTreeImpl implements CompoundAssignmentTree {
31 
32     private final ExpressionTree var, expr;
33     private final Kind kind;
34 
35     CompoundAssignmentTreeImpl(final BinaryNode node, final ExpressionTree left, final ExpressionTree right) {
36         super(node);
37         assert node.isAssignment() : "not an assignment node";
38         this.var = left;
39         this.expr = right;
40         this.kind = getOperator(node.tokenType());
41         assert kind != Tree.Kind.ASSIGNMENT : "compound assignment expected";
42     }
43 
44     @Override
45     public Kind getKind() {
46         return kind;
47     }
48 
49     @Override
50     public ExpressionTree getVariable() {
51         return var;
52     }
53 
54     @Override
55     public ExpressionTree getExpression() {
56         return expr;
57     }
58 
59     @Override
60     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
61         return visitor.visitCompoundAssignment(this, data);
62     }
63 }
