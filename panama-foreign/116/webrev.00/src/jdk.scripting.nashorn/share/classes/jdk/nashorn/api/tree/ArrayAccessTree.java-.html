26 package jdk.nashorn.api.tree;

28 /**
29  * A tree node for an array access expression.
30  *
31  * For example:
32  * <pre>
33  *   <em>expression</em> [ <em>index</em> ]
34  * </pre>
35  *
36  * @deprecated Nashorn JavaScript script engine and APIs, and the jjs tool
37  * are deprecated with the intent to remove them in a future release.
38  *
39  * @since 9
40  */
41 @Deprecated(since="11", forRemoval=true)
42 public interface ArrayAccessTree extends ExpressionTree {
43     /**
44      * Returns the array that is accessed.
45      *
46      * @return the array that is accessed
47      */
48     ExpressionTree getExpression();
49 
50     /**
51      * Returns the index of the array element accessed.
52      *
53      * @return the index expression
54      */
55     ExpressionTree getIndex();
56 }
