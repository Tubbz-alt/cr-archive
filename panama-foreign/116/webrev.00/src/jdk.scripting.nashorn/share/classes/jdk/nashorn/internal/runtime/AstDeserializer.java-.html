25 package jdk.nashorn.internal.runtime;
26 
27 import java.io.ByteArrayInputStream;
28 import java.io.IOException;
29 import java.io.ObjectInputStream;
30 import java.security.AccessController;
31 import java.security.PrivilegedAction;
32 import java.util.zip.InflaterInputStream;
33 import jdk.nashorn.internal.ir.FunctionNode;
34 
35 /**
36  * This static utility class performs deserialization of FunctionNode ASTs from a byte array.
37  * The format is a standard Java serialization stream, deflated.
38  */
39 final class AstDeserializer {
40     static FunctionNode deserialize(final byte[] serializedAst) {
41         return AccessController.doPrivileged(new PrivilegedAction<FunctionNode>() {
42             @Override
43             public FunctionNode run() {
44                 try {
45                     return (FunctionNode)new ObjectInputStream(new InflaterInputStream(
46                         new ByteArrayInputStream(serializedAst))).readObject();
47                 } catch (final ClassNotFoundException | IOException e) {
48                     // This is internal, can't happen
49                     throw new AssertionError("Unexpected exception deserializing function", e);
50                 }
51             }
52         });
53     }
54 }
