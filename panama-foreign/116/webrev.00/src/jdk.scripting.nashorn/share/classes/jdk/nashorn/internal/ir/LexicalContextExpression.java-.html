26 package jdk.nashorn.internal.ir;
27 
28 import jdk.nashorn.internal.ir.visitor.NodeVisitor;
29 
30 abstract class LexicalContextExpression extends Expression implements LexicalContextNode {
31     private static final long serialVersionUID = 1L;
32 
33     LexicalContextExpression(final LexicalContextExpression expr) {
34         super(expr);
35     }
36 
37     LexicalContextExpression(final long token, final int start, final int finish) {
38         super(token, start, finish);
39     }
40 
41     LexicalContextExpression(final long token, final int finish) {
42         super(token, finish);
43     }
44 
45     @Override
46     public Node accept(final NodeVisitor<? extends LexicalContext> visitor) {
47         return Acceptor.accept(this, visitor);
48     }
49 }
