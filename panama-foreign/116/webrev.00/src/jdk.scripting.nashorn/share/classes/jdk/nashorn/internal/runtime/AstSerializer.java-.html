25 package jdk.nashorn.internal.runtime;
26 
27 import java.io.ByteArrayOutputStream;
28 import java.io.IOException;
29 import java.io.ObjectOutputStream;
30 import java.util.zip.Deflater;
31 import java.util.zip.DeflaterOutputStream;
32 import jdk.nashorn.internal.ir.FunctionNode;
33 import jdk.nashorn.internal.runtime.options.Options;
34 
35 /**
36  * This static utility class performs serialization of FunctionNode ASTs to a byte array.
37  * The format is a standard Java serialization stream, deflated.
38  */
39 final class AstSerializer {
40     // Experimentally, we concluded that compression level 4 gives a good tradeoff between serialization speed
41     // and size.
42     private static final int COMPRESSION_LEVEL = Options.getIntProperty("nashorn.serialize.compression", 4);
43     static byte[] serialize(final FunctionNode fn) {
44         final ByteArrayOutputStream out = new ByteArrayOutputStream();
45         final Deflater deflater = new Deflater(COMPRESSION_LEVEL);
46         try (final ObjectOutputStream oout = new ObjectOutputStream(new DeflaterOutputStream(out, deflater))) {
47             oout.writeObject(fn);
48         } catch (final IOException e) {
49             throw new AssertionError("Unexpected exception serializing function", e);
50         } finally {
51             deflater.end();
52         }
53         return out.toByteArray();
54     }
55 }
