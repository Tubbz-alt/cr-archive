25 package jdk.nashorn.api.tree;
26 
27 import java.util.List;
28 
29 /**
30  * A tree node that represents a <a href="http://www.ecma-international.org/ecma-262/6.0/#sec-class-definitions">class declaration</a>.
31  *
32  * @deprecated Nashorn JavaScript script engine and APIs, and the jjs tool
33  * are deprecated with the intent to remove them in a future release.
34  *
35  * @since 9
36  */
37 @Deprecated(since="11", forRemoval=true)
38 public interface ClassDeclarationTree extends StatementTree {
39 
40     /**
41      * Class identifier.
42      *
43      * @return the class identifier
44      */
45     IdentifierTree getName();
46 
47     /**
48      * The expression of the {@code extends} clause. Optional.
49      *
50      * @return the class heritage
51      */
52     ExpressionTree getClassHeritage();
53 
54     /**
55      * Get the constructor method definition.
56      *
57      * @return the constructor
58      */
59     PropertyTree getConstructor();
60 
61     /**
62      * Get other property definitions except for the constructor.
63      *
64      * @return the class elements
65      */
66     List<? extends PropertyTree> getClassElements();
67 }
