20 package jdk.nashorn.internal.runtime.regexp.joni;
21 
22 final class BitStatus {
23     public static final int BIT_STATUS_BITS_NUM = 4 * 8;
24 
25     public static int bsClear() {
26         return 0;
27     }
28 
29     public static int bsAll() {
30         return -1;
31     }
32 
33     public static boolean bsAt(final int stats, final int n) {
34         return (n < BIT_STATUS_BITS_NUM ? stats & (1 << n) : (stats & 1)) != 0;
35     }
36 
37     public static int bsOnAt(final int statsp, final int n) {
38         int stats = statsp;
39         if (n < BIT_STATUS_BITS_NUM) {
40             stats |= (1 << n);
41         } else {
42             stats |= 1;
43         }
44         return stats;
45     }
46 
47     public static int bsOnOff(final int v, final int f, final boolean negative) {
48         return negative ? (v & ~f) : (v | f);
49     }
50 }
