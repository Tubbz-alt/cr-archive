26 package jdk.nashorn.api.tree;
27 
28 import java.util.List;
29 import jdk.nashorn.internal.ir.LiteralNode;
30 
31 final class ArrayLiteralTreeImpl extends ExpressionTreeImpl
32     implements ArrayLiteralTree {
33     private final List<? extends ExpressionTree> elements;
34     ArrayLiteralTreeImpl(final LiteralNode<?> node, final List<? extends ExpressionTree> elements) {
35         super(node);
36         this.elements = elements;
37     }
38 
39     @Override
40     public Tree.Kind getKind() {
41         return Tree.Kind.ARRAY_LITERAL;
42     }
43 
44     @Override
45     public List<? extends ExpressionTree> getElements() {
46         return elements;
47     }
48 
49     @Override
50     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
51         return visitor.visitArrayLiteral(this, data);
52     }
53 }
