26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.ir.TernaryNode;
29 
30 final class ConditionalExpressionTreeImpl extends ExpressionTreeImpl implements ConditionalExpressionTree {
31     private final ExpressionTree condExpr, trueExpr, falseExpr;
32 
33     ConditionalExpressionTreeImpl(final TernaryNode node,
34             final ExpressionTree condExpr, final ExpressionTree trueExpr,
35             final ExpressionTree falseExpr) {
36         super(node);
37         this.condExpr = condExpr;
38         this.trueExpr = trueExpr;
39         this.falseExpr = falseExpr;
40 
41     }
42 
43     @Override
44     public Kind getKind() {
45         return Kind.CONDITIONAL_EXPRESSION;
46     }
47 
48     @Override
49     public ExpressionTree getCondition() {
50         return condExpr;
51     }
52 
53     @Override
54     public ExpressionTree getTrueExpression() {
55         return trueExpr;
56     }
57 
58     @Override
59     public ExpressionTree getFalseExpression() {
60         return falseExpr;
61     }
62 
63     @Override
64     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
65         return visitor.visitConditionalExpression(this, data);
66     }
67 }
