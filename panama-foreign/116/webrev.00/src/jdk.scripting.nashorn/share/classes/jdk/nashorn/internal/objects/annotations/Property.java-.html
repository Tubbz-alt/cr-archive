26 package jdk.nashorn.internal.objects.annotations;
27 
28 import static jdk.nashorn.internal.objects.annotations.Attribute.DEFAULT_ATTRIBUTES;
29 
30 import java.lang.annotation.ElementType;
31 import java.lang.annotation.Retention;
32 import java.lang.annotation.RetentionPolicy;
33 import java.lang.annotation.Target;
34 
35 /**
36  * Annotation to specify a JavaScript "data" property.
37  */
38 @Retention(RetentionPolicy.RUNTIME)
39 @Target(ElementType.FIELD)
40 public @interface Property {
41     /**
42      * Name of the script property. If empty, the name is inferred.
43      * @return name
44      */
45     public String name() default "";
46 
47     /**
48      * Attribute flags for this function.
49      * @return attribute
50      */
51     public int attributes() default DEFAULT_ATTRIBUTES;
52 
53     /**
54      * Initialize this property with the object of given class.
55      * @return class
56      */
57     public String clazz() default "";
58 
59     /**
60      * Where this property lives?
61      * @return location of property
62      */
63     public Where where() default Where.INSTANCE;
64 }
