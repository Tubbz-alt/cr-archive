27 package jdk.nashorn.api.tree;
28 
29 import jdk.nashorn.internal.ir.IdentNode;
30 import jdk.nashorn.internal.ir.Module;
31 
32 final class IdentifierTreeImpl extends ExpressionTreeImpl implements IdentifierTree {
33     private final String name;
34 
35     IdentifierTreeImpl(final IdentNode node) {
36         super(node);
37         this.name = node.getName();
38     }
39 
40     @Override
41     public Kind getKind() {
42         return Kind.IDENTIFIER;
43     }
44 
45     @Override
46     public String getName() {
47         return name;
48     }
49 
50     @Override
51     public boolean isRestParameter() {
52         return ((IdentNode)node).isRestParameter();
53     }
54 
55     @Override
56     public boolean isSuper() {
57         final IdentNode ident = (IdentNode)node;
58         return ident.isDirectSuper() || "super".equals(ident.getName());
59     }
60 
61     @Override
62     public boolean isThis() {
63         return "this".equals(((IdentNode)node).getName());
64     }
65 
66     @Override
67     public boolean isStar() {
68         return Module.STAR_NAME.equals(((IdentNode)node).getName());
69     }
70 
71     @Override
72     public boolean isDefault() {
73         return Module.DEFAULT_NAME.equals(((IdentNode)node).getName());
74     }
75 
76     @Override
77     public boolean isStarDefaultStar() {
78         return Module.DEFAULT_EXPORT_BINDING_NAME.equals(((IdentNode)node).getName());
79     }
80 
81     @Override
82     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
83         return visitor.visitIdentifier(this, data);
84     }
85 }
