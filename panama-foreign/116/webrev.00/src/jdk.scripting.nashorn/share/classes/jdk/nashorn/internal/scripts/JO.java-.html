26 package jdk.nashorn.internal.scripts;
27 
28 import jdk.nashorn.internal.runtime.PropertyMap;
29 import jdk.nashorn.internal.runtime.ScriptObject;
30 
31 /**
32  * Empty object class for object-only fields.
33  */
34 public class JO extends ScriptObject {
35 
36     private static final PropertyMap map$ = PropertyMap.newMap(JO.class);
37 
38     /**
39      * Returns the initial property map to be used.
40      * @return the initial property map.
41      */
42     public static PropertyMap getInitialMap() {
43         return map$;
44     }
45 
46     /**
47      * Constructor given an initial property map
48      *
49      * @param map the property map
50      */
51     public JO(final PropertyMap map) {
52         super(map);
53     }
54 
55     /**
56      * Constructor given an initial prototype and the default initial property map.
57      *
58      * @param proto the prototype object
59      */
60     public JO(final ScriptObject proto) {
61         super(proto, getInitialMap());
62     }
63 
64     /**
65      * Constructor that takes a pre-initialized spill pool. Used by
66      * {@link jdk.nashorn.internal.codegen.SpillObjectCreator} and
67      * {@link jdk.nashorn.internal.parser.JSONParser} for initializing object literals
68      *
69      * @param map            property map
70      * @param primitiveSpill primitive spill pool
71      * @param objectSpill    reference spill pool
72      */
73     public JO(final PropertyMap map, final long[] primitiveSpill, final Object[] objectSpill) {
74         super(map, primitiveSpill, objectSpill);
75     }
76 
77     /**
78      * A method handle of this method is passed to the ScriptFunction constructor.
79      *
80      * @param map  the property map to use for allocatorMap
81      *
82      * @return newly allocated ScriptObject
83      */
84     public static ScriptObject allocate(final PropertyMap map) {
85         return new JO(map);
86     }
87 }
