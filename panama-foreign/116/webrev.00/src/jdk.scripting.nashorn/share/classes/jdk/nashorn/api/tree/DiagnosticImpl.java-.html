26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.parser.Token;
29 import jdk.nashorn.internal.runtime.ParserException;
30 
31 final class DiagnosticImpl implements Diagnostic {
32     private final ParserException exp;
33     private final Kind kind;
34 
35     DiagnosticImpl(final ParserException exp, final Kind kind) {
36         this.exp = exp;
37         this.kind = kind;
38     }
39 
40     @Override
41     public Kind getKind() {
42         return kind;
43     }
44 
45     @Override
46     public long getPosition() {
47         return exp.getPosition();
48     }
49 
50     @Override
51     public String getFileName() {
52         return exp.getFileName();
53     }
54 
55     @Override
56     public long getLineNumber() {
57         return exp.getLineNumber();
58     }
59 
60     @Override
61     public long getColumnNumber() {
62         return exp.getColumnNumber();
63     }
64 
65     @Override
66     public String getCode() {
67         final long token = exp.getToken();
68         return (token < 0)? null : Token.toString(null, token, true);
69     }
70 
71     @Override
72     public String getMessage() {
73         return exp.getMessage();
74     }
75 
76     @Override
77     public String toString() {
78         return getMessage();
79     }
80 }
