20 package jdk.nashorn.internal.runtime.regexp.joni;
21 
22 import jdk.nashorn.internal.runtime.regexp.joni.ast.CClassNode;
23 
24 final class ApplyCaseFold {
25 
26     // i_apply_case_fold
27     public static void apply(final int from, final int to, final Object o) {
28         final ApplyCaseFoldArg arg = (ApplyCaseFoldArg)o;
29 
30         final ScanEnvironment env = arg.env;
31         final CClassNode cc = arg.cc;
32         final BitSet bs = cc.bs;
33 
34         final boolean inCC = cc.isCodeInCC(from);
35 
36         if (Config.CASE_FOLD_IS_APPLIED_INSIDE_NEGATIVE_CCLASS) {
37             if ((inCC && !cc.isNot()) || (!inCC && cc.isNot())) {
38                 if (to >= BitSet.SINGLE_BYTE_SIZE) {
39                     cc.addCodeRange(env, to, to);
40                 } else {
41                     /* /(?i:[^A-C])/.match("a") ==> fail. */
42                     bs.set(to);
43                 }
44             }
45         } else {
46             if (inCC) {
47                 if (to >= BitSet.SINGLE_BYTE_SIZE) {
48                     if (cc.isNot()) {
49                         cc.clearNotFlag();
50                     }
51                     cc.addCodeRange(env, to, to);
52                 } else {
53                     if (cc.isNot()) {
54                         bs.clear(to);
55                     } else {
56                         bs.set(to);
57                     }
58                 }
59             }
60         } // CASE_FOLD_IS_APPLIED_INSIDE_NEGATIVE_CCLASS
61 
62     }
63 
64     static final ApplyCaseFold INSTANCE = new ApplyCaseFold();
65 }
