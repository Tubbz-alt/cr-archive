26 package jdk.nashorn.api.tree;
27 
28 /**
29  * To represent property setting in an object literal tree.
30  *
31  * @deprecated Nashorn JavaScript script engine and APIs, and the jjs tool
32  * are deprecated with the intent to remove them in a future release.
33  *
34  * @since 9
35  */
36 @Deprecated(since="11", forRemoval=true)
37 public interface PropertyTree extends Tree {
38     /**
39      * Returns the name of this property.
40      *
41      * @return the name of the property
42      */
43     public ExpressionTree getKey();
44 
45     /**
46      * Returns the value of this property. This is null for accessor properties.
47      *
48      * @return the value of the property
49      */
50     public ExpressionTree getValue();
51 
52     /**
53      * Returns the setter function of this property if this
54      * is an accessor property. This is null for data properties.
55      *
56      * @return the setter function of the property
57      */
58     public FunctionExpressionTree getGetter();
59 
60     /**
61      * Returns the getter function of this property if this
62      * is an accessor property. This is null for data properties.
63      *
64      * @return the getter function of the property
65      */
66     public FunctionExpressionTree getSetter();
67 
68     /**
69      * Is this a class static property?
70      *
71      * @return true if this is a static property
72      */
73     public boolean isStatic();
74 
75     /**
76      * Is this a computed property?
77      *
78      * @return true if this is a computed property
79      */
80     public boolean isComputed();
81 }
