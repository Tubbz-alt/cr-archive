26 package jdk.nashorn.internal.runtime.arrays;
27 
28 import static jdk.nashorn.internal.runtime.ECMAErrors.typeError;
29 import jdk.nashorn.internal.objects.Global;
30 import jdk.nashorn.internal.runtime.PropertyDescriptor;
31 import jdk.nashorn.internal.runtime.ScriptRuntime;
32 
33 /**
34  * ArrayData after the array has been frozen by Object.freeze call.
35  */
36 final class FrozenArrayFilter extends SealedArrayFilter {
37     FrozenArrayFilter(final ArrayData underlying) {
38         super(underlying);
39     }
40 
41     @Override
42     public ArrayData copy() {
43         return this;
44     }
45 
46     @Override
47     public PropertyDescriptor getDescriptor(final Global global, final int index) {
48         return global.newDataDescriptor(getObject(index), false, true, false);
49     }
50 
51     @Override
52     public ArrayData set(final int index, final int value, final boolean strict) {
53         if (strict) {
54             throw typeError("cant.set.property", Integer.toString(index), "frozen array");
55         }
56         return this;
57     }
58 
59     @Override
60     public ArrayData set(final int index, final double value, final boolean strict) {
61         if (strict) {
62             throw typeError("cant.set.property", Integer.toString(index), "frozen array");
63         }
64         return this;
65     }
66 
67     @Override
68     public ArrayData set(final int index, final Object value, final boolean strict) {
69         if (strict) {
70             throw typeError("cant.set.property", Integer.toString(index), "frozen array");
71         }
72         return this;
73     }
74 
75     @Override
76     public ArrayData push(final boolean strict, final Object... items) {
77         return this; //nop
78     }
79 
80     @Override
81     public Object pop() {
82         final int len = (int)underlying.length();
83         return len == 0 ? ScriptRuntime.UNDEFINED : underlying.getObject(len - 1);
84     }
85 }
