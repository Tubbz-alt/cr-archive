26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.ir.VarNode;
29 
30 final class VariableTreeImpl extends StatementTreeImpl implements VariableTree {
31     private final IdentifierTree ident;
32     private final ExpressionTree init;
33 
34     VariableTreeImpl(final VarNode node, final IdentifierTree ident, final ExpressionTree init) {
35         super(node);
36         this.ident = ident;
37         this.init = init;
38     }
39 
40     @Override
41     public Kind getKind() {
42         return Kind.VARIABLE;
43     }
44 
45     @Override
46     public ExpressionTree getBinding() {
47         return ident;
48     }
49 
50     @Override
51     public ExpressionTree getInitializer() {
52         return init;
53     }
54 
55     @Override
56     public boolean isConst() {
57         return ((VarNode)node).isConst();
58     }
59 
60     @Override
61     public boolean isLet() {
62         return ((VarNode)node).isLet();
63     }
64 
65     @Override
66     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
67         return visitor.visitVariable(this, data);
68     }
69 }
