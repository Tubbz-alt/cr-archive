26 package jdk.nashorn.internal.codegen.types;
27 
28 /**
29  * This class represents a numeric type that can be used for bit operations.
30  */
31 public abstract class BitwiseType extends NumericType implements BytecodeBitwiseOps {
32     private static final long serialVersionUID = 1L;
33 
34     /**
35      * Constructor
36      *
37      * @param name   name of type
38      * @param clazz  Java class used to represent type
39      * @param weight weight of type
40      * @param slots  number of bytecode slots this type takes up
41      */
42     protected BitwiseType(final String name, final Class<?> clazz, final int weight, final int slots) {
43         super(name, clazz, weight, slots);
44     }
45 }
