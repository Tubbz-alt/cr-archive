26 package jdk.nashorn.internal.ir;
27 
28 import jdk.nashorn.internal.ir.annotations.Ignore;
29 import jdk.nashorn.internal.ir.visitor.NodeVisitor;
30 
31 /**
32  * Synthetic AST node that represents return from a split fragment of a split function for control flow reasons (break
33  * or continue into a target outside the current fragment). It has no JavaScript source representation and only occurs
34  * in synthetic functions created by the split-into-functions transformation. It is different from a return node in
35  * that the return value is irrelevant, and doesn't affect the function's return type calculation.
36  */
37 public final class SplitReturn extends Statement {
38     private static final long serialVersionUID = 1L;
39 
40     /** The sole instance of this AST node. */
41     @Ignore
42     public static final SplitReturn INSTANCE = new SplitReturn();
43 
44     private SplitReturn() {
45         super(NO_LINE_NUMBER, NO_TOKEN, NO_FINISH);
46     }
47 
48     @Override
49     public boolean isTerminal() {
50         return true;
51     }
52 
53     @Override
54     public Node accept(final NodeVisitor<? extends LexicalContext> visitor) {
55         return visitor.enterSplitReturn(this) ? visitor.leaveSplitReturn(this) : this;
56     }
57 
58     @Override
59     public void toString(final StringBuilder sb, final boolean printType) {
60         sb.append(":splitreturn;");
61     }
62 
63     private Object readResolve() {
64         return INSTANCE;
65     }
66 }
