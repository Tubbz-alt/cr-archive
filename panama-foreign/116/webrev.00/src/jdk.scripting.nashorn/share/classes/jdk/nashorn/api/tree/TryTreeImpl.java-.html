26 package jdk.nashorn.api.tree;
27 
28 import java.util.List;
29 import jdk.nashorn.internal.ir.TryNode;
30 
31 final class TryTreeImpl extends StatementTreeImpl implements TryTree {
32     private final BlockTree block;
33     private final List<? extends CatchTree> catches;
34     private final BlockTree finallyBlock;
35     TryTreeImpl(final TryNode node,
36             final BlockTree block,
37             final List<? extends CatchTree> catches,
38             final BlockTree finallyBlock) {
39         super(node);
40         this.block = block;
41         this.catches = catches;
42         this.finallyBlock = finallyBlock;
43     }
44 
45     @Override
46     public Kind getKind() {
47         return Kind.TRY;
48     }
49 
50     @Override
51     public BlockTree getBlock() {
52         return block;
53     }
54 
55     @Override
56     public List<? extends CatchTree> getCatches() {
57         return catches;
58     }
59 
60     @Override
61     public BlockTree getFinallyBlock() {
62         return finallyBlock;
63     }
64 
65     @Override
66     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
67         return visitor.visitTry(this, data);
68     }
69 }
