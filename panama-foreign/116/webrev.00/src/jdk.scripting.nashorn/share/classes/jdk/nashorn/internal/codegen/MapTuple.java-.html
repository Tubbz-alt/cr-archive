26 package jdk.nashorn.internal.codegen;
27 
28 import jdk.nashorn.internal.codegen.types.Type;
29 import jdk.nashorn.internal.ir.Symbol;
30 
31 /**
32  * A tuple of values used for map creation
33  * @param <T> value type
34  */
35 class MapTuple<T> {
36     final String key;
37     final Symbol symbol;
38     final Type   type;
39     final T      value;
40 
41     MapTuple(final String key, final Symbol symbol, final Type type) {
42         this(key, symbol, type, null);
43     }
44 
45     MapTuple(final String key, final Symbol symbol, final Type type, final T value) {
46         this.key    = key;
47         this.symbol = symbol;
48         this.type   = type;
49         this.value  = value;
50     }
51 
52     public Class<?> getValueType() {
53         return null; //until proven otherwise we are undefined, see NASHORN-592 int.class;
54     }
55 
56     boolean isPrimitive() {
57         return getValueType() != null && getValueType().isPrimitive() && getValueType() != boolean.class;
58     }
59 
60     @Override
61     public String toString() {
62         return "[key=" + key + ", symbol=" + symbol + ", value=" + value + " (" + (value == null ? "null" : value.getClass().getSimpleName()) +")]";
63     }
64 }
