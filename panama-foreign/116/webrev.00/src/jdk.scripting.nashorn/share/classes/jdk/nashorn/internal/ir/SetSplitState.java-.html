26 package jdk.nashorn.internal.ir;
27 
28 import jdk.nashorn.internal.codegen.CompilerConstants;
29 import jdk.nashorn.internal.ir.visitor.NodeVisitor;
30 import jdk.nashorn.internal.runtime.Scope;
31 
32 /**
33  * Synthetic AST node that represents loading of the scope object and invocation of the {@link Scope#setSplitState(int)}
34  * method on it. It has no JavaScript source representation and only occurs in synthetic functions created by
35  * the split-into-functions transformation.
36  */
37 public final class SetSplitState extends Statement {
38     private static final long serialVersionUID = 1L;
39 
40     private final int state;
41 
42     /**
43      * Creates a new split state setter
44      * @param state the state to set
45      * @param lineNumber the line number where it is inserted
46      */
47     public SetSplitState(final int state, final int lineNumber) {
48         super(lineNumber, NO_TOKEN, NO_FINISH);
49         this.state = state;
50     }
51 
52     /**
53      * Returns the state this setter sets.
54      * @return the state this setter sets.
55      */
56     public int getState() {
57         return state;
58     }
59 
60     @Override
61     public Node accept(final NodeVisitor<? extends LexicalContext> visitor) {
62         return visitor.enterSetSplitState(this) ? visitor.leaveSetSplitState(this) : this;
63     }
64 
65     @Override
66     public void toString(final StringBuilder sb, final boolean printType) {
67         sb.append(CompilerConstants.SCOPE.symbolName()).append('.').append(Scope.SET_SPLIT_STATE.name())
68         .append('(').append(state).append(");");
69     }
70 }
