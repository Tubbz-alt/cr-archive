26 package jdk.nashorn.internal.runtime.arrays;
27 
28 import static jdk.nashorn.internal.runtime.ECMAErrors.typeError;
29 import jdk.nashorn.internal.objects.Global;
30 import jdk.nashorn.internal.runtime.ScriptRuntime;
31 
32 /**
33  * Filter class that wrap arrays that have been tagged non extensible
34  */
35 final class NonExtensibleArrayFilter extends ArrayFilter {
36 
37     /**
38      * Constructor
39      * @param underlying array
40      */
41     NonExtensibleArrayFilter(final ArrayData underlying) {
42         super(underlying);
43     }
44 
45     @Override
46     public ArrayData copy() {
47         return new NonExtensibleArrayFilter(underlying.copy());
48     }
49 
50     @Override
51     public ArrayData slice(final long from, final long to) {
52         return new NonExtensibleArrayFilter(underlying.slice(from, to));
53     }
54 
55     private ArrayData extensionCheck(final boolean strict, final int index) {
56         if (!strict) {
57             return this;
58         }
59         throw typeError(Global.instance(), "object.non.extensible", String.valueOf(index), ScriptRuntime.safeToString(this));
60     }
61 
62     @Override
63     public ArrayData set(final int index, final Object value, final boolean strict) {
64         if (has(index)) {
65             return underlying.set(index, value, strict);
66         }
67         return extensionCheck(strict, index);
68     }
69 
70     @Override
71     public ArrayData set(final int index, final int value, final boolean strict) {
72         if (has(index)) {
73             return underlying.set(index, value, strict);
74         }
75         return extensionCheck(strict, index);
76     }
77 
78     @Override
79     public ArrayData set(final int index, final double value, final boolean strict) {
80         if (has(index)) {
81             return underlying.set(index, value, strict);
82         }
83         return extensionCheck(strict, index);
84     }
85 }
