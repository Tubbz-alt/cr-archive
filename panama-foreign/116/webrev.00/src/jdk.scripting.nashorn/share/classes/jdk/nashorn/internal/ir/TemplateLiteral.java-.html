26 package jdk.nashorn.internal.ir;
27 
28 import java.util.Collections;
29 import java.util.List;
30 import jdk.nashorn.internal.codegen.types.Type;
31 import jdk.nashorn.internal.ir.visitor.NodeVisitor;
32 
33 /**
34  * Represents ES6 template string expression. Note that this Node class is used
35  * only in "parse only" mode. In evaluation mode, Parser directly folds template
36  * literal as string concatenation. Parser API uses this node to represent ES6
37  * template literals "as is" rather than as a String concatenation.
38  */
39 public final class TemplateLiteral extends Expression {
40     private static final long serialVersionUID = 1L;
41     private final List<Expression> exprs;
42 
43     public TemplateLiteral(final List<Expression> exprs) {
44         super(exprs.get(0).getToken(), exprs.get(exprs.size() - 1).finish);
45         this.exprs = exprs;
46     }
47 
48     @Override
49     public Type getType() {
50         return Type.STRING;
51     }
52 
53     @Override
54     public Node accept(final NodeVisitor<? extends LexicalContext> visitor) {
55         if (visitor.enterTemplateLiteral(this)) {
56             return visitor.leaveTemplateLiteral(this);
57         }
58 
59         return this;
60     }
61 
62     @Override
63     public void toString(final StringBuilder sb, final boolean printType) {
64         for (final Expression expr : exprs) {
65             sb.append(expr);
66         }
67     }
68 
69     /**
70      * The list of expressions that are part of this template literal.
71      *
72      * @return the list of expressions that are part of this template literal.
73      */
74     public List<Expression> getExpressions() {
75         return Collections.unmodifiableList(exprs);
76     }
77 }
