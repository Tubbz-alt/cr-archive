26 package jdk.nashorn.internal.runtime.arrays;
27 
28 import java.lang.reflect.Array;
29 
30 /**
31   * Iterator over a Java List.
32  */
33 class JavaArrayIterator extends ArrayLikeIterator<Object> {
34 
35     /** Array to iterate over */
36     protected final Object array;
37 
38     /** length of array */
39     protected final long length;
40 
41     /**
42      * Constructor
43      * @param array array to iterate over
44      * @param includeUndefined should undefined elements be included in iteration
45      */
46     protected JavaArrayIterator(final Object array, final boolean includeUndefined) {
47         super(includeUndefined);
48         assert array.getClass().isArray() : "expecting Java array object";
49         this.array = array;
50         this.length = Array.getLength(array);
51     }
52 
53     /**
54      * Is the current index still inside the array
55      * @return true if inside the array
56      */
57     protected boolean indexInArray() {
58         return index < length;
59     }
60 
61     @Override
62     public Object next() {
63         return Array.get(array, (int)bumpIndex());
64     }
65 
66     @Override
67     public long getLength() {
68         return length;
69     }
70 
71     @Override
72     public boolean hasNext() {
73         return indexInArray();
74     }
75 
76     @Override
77     public void remove() {
78         throw new UnsupportedOperationException("remove");
79     }
80 }
