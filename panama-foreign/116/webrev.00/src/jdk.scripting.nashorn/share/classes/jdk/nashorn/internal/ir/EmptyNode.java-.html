26 package jdk.nashorn.internal.ir;
27 
28 import jdk.nashorn.internal.ir.annotations.Immutable;
29 import jdk.nashorn.internal.ir.visitor.NodeVisitor;
30 
31 /**
32  * IR representation for an empty statement.
33  */
34 @Immutable
35 public final class EmptyNode extends Statement {
36     private static final long serialVersionUID = 1L;
37 
38     /**
39      * Constructor
40      *
41      * @param node node to wrap
42      */
43     public EmptyNode(final Statement node) {
44         super(node);
45     }
46 
47     /**
48      * Constructor
49      *
50      * @param lineNumber line number
51      * @param token      token
52      * @param finish     finish
53      */
54     public EmptyNode(final int lineNumber, final long token, final int finish) {
55         super(lineNumber, token, finish);
56     }
57 
58 
59     @Override
60     public Node accept(final NodeVisitor<? extends LexicalContext> visitor) {
61         if (visitor.enterEmptyNode(this)) {
62             return visitor.leaveEmptyNode(this);
63         }
64         return this;
65     }
66 
67     @Override
68     public void toString(final StringBuilder sb, final boolean printTypes) {
69         sb.append(';');
70     }
71 
72 }
