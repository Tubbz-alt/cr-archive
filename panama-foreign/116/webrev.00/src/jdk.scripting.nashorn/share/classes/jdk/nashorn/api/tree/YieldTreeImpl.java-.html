25 package jdk.nashorn.api.tree;
26 
27 import jdk.nashorn.internal.ir.Expression;
28 import jdk.nashorn.internal.ir.UnaryNode;
29 import jdk.nashorn.internal.parser.TokenType;
30 
31 final class YieldTreeImpl extends ExpressionTreeImpl
32         implements YieldTree {
33 
34     private final ExpressionTree expr;
35 
36     YieldTreeImpl(final Expression exprNode, final ExpressionTree expr) {
37         super(exprNode);
38         this.expr = expr;
39     }
40 
41     @Override
42     public Kind getKind() {
43         return Kind.YIELD;
44     }
45 
46     @Override
47     public ExpressionTree getExpression() {
48         return expr;
49     }
50 
51     @Override
52     public boolean isStar() {
53         return ((UnaryNode) node).isTokenType(TokenType.YIELD_STAR);
54     }
55 
56     @Override
57     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
58         return visitor.visitYield(this, data);
59     }
60 }
