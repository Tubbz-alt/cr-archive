25 package jdk.nashorn.api.tree;
26 
27 import jdk.nashorn.internal.ir.ForNode;
28 
29 final class ForOfLoopTreeImpl extends StatementTreeImpl implements ForOfLoopTree {
30     private final ExpressionTree lhsExpr;
31     private final ExpressionTree expr;
32     private final StatementTree stat;
33 
34     ForOfLoopTreeImpl(final ForNode node,
35             final ExpressionTree lhsExpr,
36             final ExpressionTree expr,
37             final StatementTree stat) {
38         super(node);
39         assert node.isForIn() : "for ..in expected";
40         this.lhsExpr = lhsExpr;
41         this.expr = expr;
42         this.stat = stat;
43     }
44 
45     @Override
46     public Kind getKind() {
47         return Kind.FOR_IN_LOOP;
48     }
49 
50     @Override
51     public ExpressionTree getVariable() {
52         return lhsExpr;
53     }
54 
55     @Override
56     public ExpressionTree getExpression() {
57         return expr;
58     }
59 
60     @Override
61     public StatementTree getStatement() {
62         return stat;
63     }
64 
65     @Override
66     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
67         return visitor.visitForOfLoop(this, data);
68     }
69 }
