26 package jdk.nashorn.api.tree;
27 
28 import java.util.List;
29 import jdk.nashorn.internal.ir.VarNode;
30 
31 final class ClassDeclarationTreeImpl extends StatementTreeImpl implements ClassDeclarationTree {
32 
33     private final IdentifierTree name;
34     private final ExpressionTree classHeritage;
35     private final PropertyTree constructor;
36     private final List<? extends PropertyTree> classElements;
37 
38     ClassDeclarationTreeImpl(final VarNode node, final IdentifierTree name,
39             final ExpressionTree classHeritage, final PropertyTree constructor,
40             final List<? extends PropertyTree> classElements) {
41         super(node);
42         this.name = name;
43         this.classHeritage = classHeritage;
44         this.constructor = constructor;
45         this.classElements = classElements;
46     }
47 
48     @Override
49     public Tree.Kind getKind() {
50         return Tree.Kind.CLASS;
51     }
52 
53     @Override
54     public IdentifierTree getName() {
55         return name;
56     }
57 
58     @Override
59     public ExpressionTree getClassHeritage() {
60         return classHeritage;
61     }
62 
63     @Override
64     public PropertyTree getConstructor() {
65         return constructor;
66     }
67 
68     @Override
69     public List<? extends PropertyTree> getClassElements() {
70         return classElements;
71     }
72 
73     @Override
74     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
75         return visitor.visitClassDeclaration(this, data);
76     }
77 }
