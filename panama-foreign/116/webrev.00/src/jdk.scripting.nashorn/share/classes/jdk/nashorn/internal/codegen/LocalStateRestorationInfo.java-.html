25 package jdk.nashorn.internal.codegen;
26 
27 import jdk.nashorn.internal.codegen.types.Type;
28 
29 /**
30  * Encapsulates the information for restoring the local state when continuing execution after a rewrite triggered by
31  * an optimistic assumption failure. An instance of this class is specific to a program point.
32  *
33  */
34 public class LocalStateRestorationInfo {
35     private final Type[] localVariableTypes;
36     private final int[] stackLoads;
37 
38     LocalStateRestorationInfo(final Type[] localVariableTypes, final int[] stackLoads) {
39         this.localVariableTypes = localVariableTypes;
40         this.stackLoads = stackLoads;
41     }
42 
43     /**
44      * Returns the types of the local variables at the continuation of a program point.
45      * @return the types of the local variables at the continuation of a program point.
46      */
47     public Type[] getLocalVariableTypes() {
48         return localVariableTypes.clone();
49     }
50 
51     /**
52      * Returns the indices of local variables that need to be loaded on stack before jumping to the continuation of the
53      * program point.
54      * @return the indices of local variables that need to be loaded on stack.
55      */
56     public int[] getStackLoads() {
57         return stackLoads.clone();
58     }
59 }
