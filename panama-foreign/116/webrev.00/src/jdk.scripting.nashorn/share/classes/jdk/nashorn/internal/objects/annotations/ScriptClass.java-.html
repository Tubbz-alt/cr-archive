26 package jdk.nashorn.internal.objects.annotations;
27 
28 import java.lang.annotation.ElementType;
29 import java.lang.annotation.Retention;
30 import java.lang.annotation.RetentionPolicy;
31 import java.lang.annotation.Target;
32 
33 /**
34  * Annotation to specify that the annotated Java class is a JavaScript "class".
35  */
36 @Retention(RetentionPolicy.RUNTIME)
37 @Target(ElementType.TYPE)
38 public @interface ScriptClass {
39     /**
40      * @return the name of the script class. By default, the name is derived
41      *         from the Java class name.
42      */
43     public String value() default "";
44 }
