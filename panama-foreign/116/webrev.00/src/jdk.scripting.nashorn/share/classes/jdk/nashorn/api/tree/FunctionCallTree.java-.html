26 package jdk.nashorn.api.tree;
27 
28 import java.util.List;
29 
30 /**
31  * A tree node for a function call expression.
32  *
33  * For example:
34  * <pre>
35  *   <em>identifier</em> ( <em>arguments</em> )
36  *
37  *   this . <em>identifier</em> ( <em>arguments</em> )
38  * </pre>
39  *
40  * @deprecated Nashorn JavaScript script engine and APIs, and the jjs tool
41  * are deprecated with the intent to remove them in a future release.
42  *
43  * @since 9
44  */
45 @Deprecated(since="11", forRemoval=true)
46 public interface FunctionCallTree extends ExpressionTree {
47     /**
48      * Returns the function being called.
49      *
50      * @return the function being called
51      */
52     ExpressionTree getFunctionSelect();
53 
54     /**
55      * Returns the list of arguments being passed to this function call.
56      *
57      * @return the list of argument expressions
58      */
59     List<? extends ExpressionTree> getArguments();
60 }
