26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.ir.ForNode;
29 
30 final class ForLoopTreeImpl extends StatementTreeImpl implements ForLoopTree {
31     private final ExpressionTree init;
32     private final ExpressionTree cond;
33     private final ExpressionTree update;
34     private final StatementTree stat;
35 
36     ForLoopTreeImpl(final ForNode node,
37         final ExpressionTree init,
38         final ExpressionTree cond,
39         final ExpressionTree update,
40         final StatementTree  stat) {
41         super(node);
42         assert !node.isForIn() : "for statement expected";
43         this.init = init;
44         this.cond = cond;
45         this.update = update;
46         this.stat = stat;
47     }
48 
49     @Override
50     public Kind getKind() {
51         return Kind.FOR_LOOP;
52     }
53 
54     @Override
55     public ExpressionTree getInitializer() {
56         return init;
57     }
58 
59     @Override
60     public ExpressionTree getCondition() {
61         return cond;
62     }
63 
64     @Override
65     public ExpressionTree getUpdate() {
66         return update;
67     }
68 
69     @Override
70     public StatementTree getStatement() {
71         return stat;
72     }
73 
74     @Override
75     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
76         return visitor.visitForLoop(this, data);
77     }
78 }
