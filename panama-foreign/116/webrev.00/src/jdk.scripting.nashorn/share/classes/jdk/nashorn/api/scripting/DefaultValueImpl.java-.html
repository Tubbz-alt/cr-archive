26 package jdk.nashorn.api.scripting;
27 
28 import jdk.nashorn.internal.runtime.JSType;
29 
30 /**
31  * Default implementation of {@link JSObject#getDefaultValue(Class)}. Isolated into a separate class mostly so
32  * that we can have private static instances of function name arrays, something we couldn't declare without it
33  * being visible in {@link JSObject} interface.
34  */
35 class DefaultValueImpl {
36     private static final String[] DEFAULT_VALUE_FNS_NUMBER = new String[] { "valueOf", "toString" };
37     private static final String[] DEFAULT_VALUE_FNS_STRING = new String[] { "toString", "valueOf" };
38 
39     static Object getDefaultValue(final JSObject jsobj, final Class<?> hint) throws UnsupportedOperationException {
40         final boolean isNumber = hint == null || hint == Number.class;
41         for(final String methodName: isNumber ? DEFAULT_VALUE_FNS_NUMBER : DEFAULT_VALUE_FNS_STRING) {
42             final Object objMember = jsobj.getMember(methodName);
43             if (objMember instanceof JSObject) {
44                 final JSObject member = (JSObject)objMember;
45                 if (member.isFunction()) {
46                     final Object value = member.call(jsobj);
47                     if (JSType.isPrimitive(value)) {
48                         return value;
49                     }
50                 }
51             }
52         }
53         throw new UnsupportedOperationException(isNumber ? "cannot.get.default.number" : "cannot.get.default.string");
54     }
55 }
