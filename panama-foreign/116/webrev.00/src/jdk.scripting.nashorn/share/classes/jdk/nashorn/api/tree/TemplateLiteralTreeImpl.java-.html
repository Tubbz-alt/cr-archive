26 package jdk.nashorn.api.tree;
27 
28 import java.util.List;
29 import jdk.nashorn.internal.ir.Expression;
30 
31 final class TemplateLiteralTreeImpl extends ExpressionTreeImpl
32         implements TemplateLiteralTree {
33 
34     private final List<? extends ExpressionTree> expressions;
35 
36     TemplateLiteralTreeImpl(final Expression node, final List<? extends ExpressionTree> expressions) {
37         super(node);
38         this.expressions = expressions;
39     }
40 
41     @Override
42     public Kind getKind() {
43         return Kind.TEMPLATE_LITERAL;
44     }
45 
46     @Override
47     public List<? extends ExpressionTree> getExpressions() {
48         return expressions;
49     }
50 
51 
52     @Override
53     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
54         return visitor.visitTemplateLiteral(this, data);
55     }
56 }
