 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import java.util.*;
 29 
 30 import javax.lang.model.element.Element;
 31 import javax.lang.model.element.TypeElement;
 32 
 33 import com.sun.source.doctree.DocTree;
 34 
 35 /**
 36  * The interface for writing member summary output.
 37  *
 38  *  <p><b>This is NOT part of any supported API.
 39  *  If you write code that depends on this, you do so at your own risk.
 40  *  This code and its internal interfaces are subject to change or
 41  *  deletion without notice.</b>
 42  */
 43 
 44 public interface MemberSummaryWriter {
 45 
 46     /**
 47      * Get the member summary header for the given class.
 48      *
 49      * @param typeElement the class the summary belongs to
 50      * @param memberSummaryTree the content tree to which the member summary will be added
 51      * @return a content tree for the member summary header
 52      */
 53     Content getMemberSummaryHeader(TypeElement typeElement, Content memberSummaryTree);
 54 
 55     /**
 56      * Get the summary table for the given class.
 57      *
 58      * @param typeElement the class the summary table belongs to
 59      * @return a content tree for the member summary table
 60      */
 61     Content getSummaryTableTree(TypeElement typeElement);
 62 
 63     /**
 64      * Add the member summary for the given class and member.
 65      *
 66      * @param typeElement the class the summary belongs to
 67      * @param member the member that is documented
 68      * @param firstSentenceTags the tags for the sentence being documented
 69      */
 70     void addMemberSummary(TypeElement typeElement, Element member,
 71                           List<? extends DocTree> firstSentenceTags);
 72 
 73     /**
 74      * Get the inherited member summary header for the given class.
 75      *
 76      * @param typeElement the class the summary belongs to
 77      * @return a content tree containing the inherited summary header
 78      */
 79     Content getInheritedSummaryHeader(TypeElement typeElement);
 80 
 81     /**
 82      * Add the inherited member summary for the given class and member.
 83      *
 84      * @param typeElement the class the inherited member belongs to
 85      * @param member the inherited member that is being documented
 86      * @param isFirst true if this is the first member in the list
 87      * @param isLast true if this is the last member in the list
 88      * @param linksTree the content tree to which the links will be added
 89      */
 90     void addInheritedMemberSummary(TypeElement typeElement,
 91                                    Element member, boolean isFirst, boolean isLast,
 92                                    Content linksTree);
 93 
 94     /**
 95      * Get inherited summary links.
 96      *
 97      * @return a content tree containing the inherited summary links
 98      */
 99     Content getInheritedSummaryLinksTree();
100 
101     /**
102      * Adds the given summary to the list of summaries.
103      *
104      * @param summariesList the list of summaries
105      * @param content       the summary
106      */
107     void addSummary(Content summariesList, Content content);
108 
109     /**
110      * Get the member tree.
111      *
112      * @param memberTree the content tree representing the member
113      * @return a content tree for the member
114      */
115     Content getMemberTree(Content memberTree);
116 }
