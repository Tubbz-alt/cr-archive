24 <project name="nashorntask" default="all" basedir=".">
25   <target name="init">
26     <loadproperties srcFile="project.properties"/>
27   </target>
28 
29   <target name="prepare" depends="init">
30     <mkdir dir="${nashorntask.build.classes.dir}"/>
31     <mkdir dir="${nashorntask.dist.dir}"/>
32   </target>
33 
34   <target name="clean" depends="init">
35     <delete dir="${nashorntask.build.dir}"/>
36     <delete dir="${nashorntask.dist.dir}"/>
37   </target>
38 
39   <target name="compile" depends="prepare" description="Compiles the nashorn ant tag sources">
40     <javac srcdir="${src.dir}"
41            destdir="${nashorntask.build.classes.dir}"
42            debug="${javac.debug}"
43            includeantruntime="true">
44       <compilerarg value="-Xlint:unchecked"/>
45       <compilerarg value="-Xlint:deprecation"/>
46     </javac>
47   </target>
48 
49   <target name="jar" depends="compile" description="Creates nashorntask.jar">
50     <jar jarfile="${nashorntask.dist.jar}" basedir="${nashorntask.build.classes.dir}"/>
51   </target>
52 
53   <target name="dist" depends="jar"/>
54 
55   <target name="all" depends="dist" 
56     description="Builds sources and generates nashorntask.jar"/>
57 </project>
