 66 ###############################################################################
 67 #
 68 # Initialization / Boot-strapping
 69 #
 70 # The bootstrapping process needs to solve the "chicken or the egg" problem,
 71 # thus it jumps back and forth, each time gaining something needed later on.
 72 #
 73 ###############################################################################
 74 
 75 # If we are requested to print additional help, do that and then exit.
 76 # This must be the very first call.
 77 HELP_PRINT_ADDITIONAL_HELP_AND_EXIT
 78 
 79 # Basic initialization that must happen first of all in the normal process.
 80 BASIC_INIT
 81 BASIC_SETUP_FUNDAMENTAL_TOOLS
 82 
 83 # Now we can determine OpenJDK build and target platforms. This is required to
 84 # have early on.
 85 PLATFORM_SETUP_OPENJDK_BUILD_AND_TARGET
 86 
 87 # Continue setting up basic stuff. Most remaining code require fundamental tools.
 88 BASIC_SETUP_PATHS
 89 
 90 # Check if it's a pure open build or if custom sources are to be used.
 91 JDKOPT_SETUP_OPEN_OR_CUSTOM
 92 
 93 # These are needed to be able to create a configuration name (and thus the output directory)
 94 JDKOPT_SETUP_DEBUG_LEVEL
 95 HOTSPOT_SETUP_JVM_VARIANTS
 96 
 97 # With basic setup done, call the custom early hook.
 98 CUSTOM_EARLY_HOOK
 99 
100 # This only needs debug level to be setup
101 JDKOPT_ALLOW_ABSOLUTE_PATHS_IN_OUTPUT
102 
103 # Check if we have devkits, extra paths or sysroot set.
104 BASIC_SETUP_DEVKIT
105 
106 # To properly create a configuration name, we need to have the OpenJDK target
107 # and options (variants and debug level) parsed.
108 BASIC_SETUP_OUTPUT_DIR
109 
110 # Must be done before we can call HELP_MSG_MISSING_DEPENDENCY.
111 HELP_SETUP_DEPENDENCY_HELP
112 
113 # Setup tools that requires more complex handling, or that is not needed by the configure script.
114 BASIC_SETUP_COMPLEX_TOOLS
115 BASIC_CHECK_BASH_OPTIONS
116 
117 # Check if pkg-config is available.
118 PKG_PROG_PKG_CONFIG
119 
120 # After basic tools have been setup, we can check build os specific details.
121 PLATFORM_SETUP_OPENJDK_BUILD_OS_VERSION
122 
123 # Misc basic settings
124 BASIC_SETUP_DEFAULT_MAKE_TARGET
125 BASIC_SETUP_DEFAULT_LOG
126 
127 ###############################################################################
128 #
129 # Determine OpenJDK variants, options and version numbers.
130 #
131 ###############################################################################
132 
133 # We need build & target for this.
134 JDKOPT_SETUP_JDK_OPTIONS
135 JDKOPT_SETUP_JLINK_OPTIONS
136 JDKVER_SETUP_JDK_VERSION_NUMBERS
137 
138 ###############################################################################
139 #
140 # Setup BootJDK, used to bootstrap the build.

143 
144 BOOTJDK_SETUP_BOOT_JDK
145 BOOTJDK_SETUP_BUILD_JDK
146 
147 ###############################################################################
148 #
149 # Configure the sources to use. We can add or override individual directories.
150 #
151 ###############################################################################
152 
153 SRCDIRS_SETUP_DIRS
154 SRCDIRS_SETUP_IMPORT_MODULES
155 
156 ###############################################################################
157 #
158 # Setup the toolchain (compilers etc), i.e. tools used to compile and process
159 # native code.
160 #
161 ###############################################################################
162 
163 # See if we are doing a complete static build or not
164 JDKOPT_SETUP_STATIC_BUILD
165 
166 # First determine the toolchain type (compiler family)
167 TOOLCHAIN_DETERMINE_TOOLCHAIN_TYPE
168 
169 # The global flags are needed for configure to be able to run the compilers
170 # correctly.
171 FLAGS_PRE_TOOLCHAIN
172 
173 # Then detect the actual binaries needed
174 TOOLCHAIN_PRE_DETECTION
175 TOOLCHAIN_DETECT_TOOLCHAIN_CORE
176 TOOLCHAIN_DETECT_TOOLCHAIN_EXTRA
177 TOOLCHAIN_POST_DETECTION
178 
179 # Finally do some processing after the detection phase
180 TOOLCHAIN_SETUP_BUILD_COMPILERS
181 TOOLCHAIN_MISC_CHECKS
182 
183 # Setup the JTReg Regression Test Harness.
184 TOOLCHAIN_SETUP_JTREG
185 
186 # Setup the Java Microbenchmark Harness (JMH)
187 LIB_TESTS_SETUP_JMH
188 
189 # Setup Jib dependency tool
190 TOOLCHAIN_SETUP_JIB
191 
192 # After toolchain setup, we need to process some flags to be able to continue.
193 FLAGS_POST_TOOLCHAIN
194 
195 # Now we can test some aspects on the target using configure macros.
196 PLATFORM_SETUP_OPENJDK_TARGET_BITS
197 PLATFORM_SETUP_OPENJDK_TARGET_ENDIANNESS
198 
199 # Configure flags for the tools
200 FLAGS_SETUP_FLAGS
201 
202 # Setup debug symbols (need objcopy from the toolchain for that)
203 JDKOPT_SETUP_DEBUG_SYMBOLS
204 JDKOPT_SETUP_CODE_COVERAGE
205 
206 # AddressSanitizer
207 JDKOPT_SETUP_ADDRESS_SANITIZER
208

210 #
211 # Check dependencies for external and internal libraries.
212 #
213 ###############################################################################
214 
215 # After we have toolchain, we can compile fixpath. It's needed by the lib checks.

217 
218 LIB_DETERMINE_DEPENDENCIES
219 LIB_SETUP_LIBRARIES
220

222 #

227 JVM_FEATURES_PARSE_OPTIONS

229 
230 HOTSPOT_ENABLE_DISABLE_GTEST
231 HOTSPOT_SETUP_MISC

240 JDKOPT_ENABLE_DISABLE_FAILURE_HANDLER
241 JDKOPT_ENABLE_DISABLE_GENERATE_CLASSLIST
242 JDKOPT_EXCLUDE_TRANSLATIONS
243 JDKOPT_ENABLE_DISABLE_MANPAGES
244 JDKOPT_ENABLE_DISABLE_CDS_ARCHIVE

255 BPERF_SETUP_BUILD_JOBS
256 BPERF_SETUP_TEST_JOBS
257 
258 # Setup arguments for the boot jdk (after cores and memory have been setup)
259 BOOTJDK_SETUP_BOOT_JDK_ARGUMENTS
260

263

265 BPERF_SETUP_ICECC
