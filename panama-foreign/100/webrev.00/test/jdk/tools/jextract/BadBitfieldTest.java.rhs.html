33 /*
34  * Not running on Windows since MSVC will not cross storage unit boundaries.
35  * Resulting struct on SysV is 16 bytes, but 24 on MSx64
36  *
37  * MSVC: (/d1reportSingleClassLayoutFoo)
38  * class Foo    size(24):
39  *      +---
40  *  0.    | a (bitstart=0,nbits=45)
41  *  8.    | b (bitstart=0,nbits=24)
42  *  8.    | c (bitstart=24,nbits=1)
43  * 16.    | d (bitstart=0,nbits=58)
44  *      +---
45  *
46  * SysV: (PAHole)
47  * struct Foo {
48  *     long long int a:45;                0:19   8
49  *     long long int b:24;                0:251  8
50  *     XXX 251 bits hole, try to pack
51  *     long long int c:1;                 8:58   8
52  *     long long int d:58;                8: 0   8
53  *
54  *     size: 16, cachelines: 1, members: 4
55  *     bit holes: 1, sum bit holes: 251 bits
56  *     bit_padding: 5 bits
57  *     last cacheline: 16 bytes
58  * };
59  *
60  */
61 
62 import org.testng.annotations.Test;
63 
64 public class BadBitfieldTest extends JextractToolRunner {
65     @Test
66     public void testBadBitfield() {
67         run("-d", getOutputFilePath("badBitfieldsGen").toString(),
68                 getInputFilePath("badBitfields.h").toString())
69             .checkFailure()
70             .checkContainsOutput("Crossing storage unit boundaries");
71     }
72 }
