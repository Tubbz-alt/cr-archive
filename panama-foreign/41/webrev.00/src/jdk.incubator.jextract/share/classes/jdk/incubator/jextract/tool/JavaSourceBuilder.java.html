 47 /**
 48  * A helper class to generate header interface class in source form.
 49  * After aggregating various constituents of a .java source, build
 50  * method is called to get overall generated source string.
 51  */
 52 class JavaSourceBuilder {
 53     private static final String ABI = SystemABI.getInstance().name();
 54     // buffer
 55     protected StringBuffer sb;
 56     // current line alignment (number of 4-spaces)
 57     protected int align;
 58 
 59     JavaSourceBuilder(int align) {
 60         this.align = align;
 61         this.sb = new StringBuffer();
 62     }
 63 
 64     JavaSourceBuilder() {
 65         this(0);
 66     }
 67 
 68     protected int align() {
 69         return align;
 70     }
 71 
 72     final String PUB_CLS_MODS = "public final ";
 73     final String PUB_MODS = "public static final ";
 74     final String PRI_MODS = "private static final ";
 75 
 76     protected void addPackagePrefix(String pkgName) {
 77         assert pkgName.indexOf('/') == -1 : "package name invalid: " + pkgName;
 78         sb.append("// Generated by jextract\n\n");
 79         if (!pkgName.isEmpty()) {
 80             sb.append("package ");
 81             sb.append(pkgName);
 82             sb.append(";\n\n");
 83         }
 84         addImportSection();
 85     }
 86 
 87     protected void addImportSection() {
 88         sb.append("import java.lang.invoke.MethodHandle;\n");
 89         sb.append("import java.lang.invoke.VarHandle;\n");
 90         sb.append("import jdk.incubator.foreign.*;\n");
 91         sb.append("import jdk.incubator.foreign.MemoryLayout.PathElement;\n");
 92         sb.append("import static ");
 93         sb.append(HandleSourceFactory.C_LANG_CONSTANTS_HOLDER);
 94         sb.append(".*;\n");
 95     }

 97     protected void classBegin(String name) {
 98         indent();
 99         sb.append(PUB_CLS_MODS + "class ");
100         sb.append(name);
101         sb.append(" {\n\n");
102     }
103 
104     protected void classEnd() {
105         indent();
106         sb.append("}\n\n");
107     }

109     protected void addLibraries(String[] libraryNames) {
110         incrAlign();
111         indent();
112         sb.append(PRI_MODS + "LibraryLookup[] LIBRARIES = RuntimeHelper.libraries(");
113         sb.append(stringArray(libraryNames) + ");\n");
114         decrAlign();
115     }

117     private String stringArray(String[] elements) {
118         return Stream.of(elements)
119                 .map(n -> "\"" + n + "\"")
120                 .collect(Collectors.joining(",", "new String[] {", "}"));
121     }
122 
123     protected void addLayout(String elementName, MemoryLayout layout) {
124         incrAlign();
125         indent();
126         sb.append(PUB_MODS + "MemoryLayout " + javaSafeIdentifier(elementName) + "$LAYOUT = ");
127         addLayout(layout);
128         sb.append(";\n");
129         decrAlign();
130     }
131 
132     private void addLayout(MemoryLayout l) {
133         if (l instanceof ValueLayout) {
134             SystemABI.Type type = l.abiType().orElseThrow(()->new AssertionError("Should not get here: " + l));
135             sb.append(switch (type) {
136                 case BOOL -> "C_BOOL";
137                 case SIGNED_CHAR -> "C_SCHAR";
138                 case UNSIGNED_CHAR -> "C_UCHAR";
139                 case CHAR -> "C_CHAR";
140                 case SHORT -> "C_SHORT";
141                 case UNSIGNED_SHORT -> "C_USHORT";
142                 case INT -> "C_INT";
143                 case UNSIGNED_INT -> "C_UINT";
144                 case LONG -> "C_LONG";
145                 case UNSIGNED_LONG -> "C_ULONG";
146                 case LONG_LONG -> "C_LONGLONG";
147                 case UNSIGNED_LONG_LONG -> "C_ULONGLONG";
148                 case FLOAT -> "C_FLOAT";
149                 case DOUBLE -> "C_DOUBLE";
150                 case LONG_DOUBLE -> "C_LONGDOUBLE";
151                 case POINTER -> "C_POINTER";
152                 default -> { throw new RuntimeException("should not reach here: " + type); }
153             });
154         } else if (l instanceof SequenceLayout) {
155             sb.append("MemoryLayout.ofSequence(");
156             if (((SequenceLayout) l).elementCount().isPresent()) {
157                 sb.append(((SequenceLayout) l).elementCount().getAsLong() + ", ");
158             }
159             addLayout(((SequenceLayout) l).elementLayout());
160             sb.append(")");
161         } else if (l instanceof GroupLayout) {
162             SystemABI.Type type = l.abiType().orElse(null);
163             if (type == SystemABI.Type.COMPLEX_LONG_DOUBLE) {
164                 if (!ABI.equals(SystemABI.ABI_SYSV)) {
165                     throw new RuntimeException("complex long double is supported only for SysV ABI");
166                 } else {
167                     sb.append("C_COMPLEX_LONGDOUBLE");
168                 }
169             } else {
170                 if (((GroupLayout) l).isStruct()) {
171                     sb.append("MemoryLayout.ofStruct(\n");
172                 } else {
173                     sb.append("MemoryLayout.ofUnion(\n");
174                 }
175                 incrAlign();
176                 String delim = "";
177                 for (MemoryLayout e : ((GroupLayout) l).memberLayouts()) {
178                     sb.append(delim);
179                     indent();
180                     addLayout(e);
181                     delim = ",\n";
182                 }
183                 sb.append("\n");
184                 decrAlign();
185                 indent();
186                 sb.append(")");
187             }
188         } else {
189             //padding
190             sb.append("MemoryLayout.ofPaddingBits(" + l.bitSize() + ")");
191         }
192         if (l.name().isPresent()) {
193             sb.append(".withName(\"" +  l.name().get() + "\")");
194         }
195     }

197     protected void addVarHandle(String name, Class<?> type, String parentName) {
198         incrAlign();
199         indent();
200         parentName = parentName != null? javaSafeIdentifier(parentName) : parentName;
201         name = javaSafeIdentifier(name);
202         String vhName = parentName != null ?
203                 parentName + "$" + name : name;
204         sb.append(PUB_MODS + "VarHandle " + vhName + " = ");
205         if (parentName != null) {
206             addHandlePath(type, parentName, name);
207         } else {
208             addHandlePath(type, name);
209         }
210         sb.append(";\n");
211         decrAlign();
212     }
213 
214     protected void addHandlePath(Class<?> type, String strName, String fieldName) {
215         String ty = type.getName();
216         if (ty.contains("MemoryAddress")) {
217             ty = "long";
218         }
219         sb.append(strName + "$LAYOUT.varHandle(" + ty + ".class, ");
220         sb.append("PathElement.groupElement(\"" + fieldName +"\")");
221         sb.append(")");
222     }
223 
224     protected void addHandlePath(Class<?> type, String varName) {
225         String ty = type.getName();
226         if (ty.contains("MemoryAddress")) {
227             ty = "long";
228         }
229         sb.append(varName + "$LAYOUT.varHandle(" + ty + ".class)");
230     }

232     protected void addMethodHandle(Declaration.Function funcTree, MethodType mtype, FunctionDescriptor desc) {
233         incrAlign();
234         indent();
235         sb.append(PUB_MODS + "MethodHandle " + javaSafeIdentifier(funcTree.name()) + " = ");
236         sb.append("RuntimeHelper.downcallHandle(\n");
237         incrAlign();
238         indent();
239         sb.append("LIBRARIES, \"" + funcTree.name() + "\"");
240         sb.append(",\n");
241         indent();
242         sb.append("\"" + mtype.toMethodDescriptorString() + "\",\n");
243         indent();
244         addFunction(desc);
245         sb.append(",\n");
246         indent();
247         sb.append(funcTree.type().varargs());
248         decrAlign();
249         sb.append("\n");
250         indent();
251         sb.append(");\n");
252         decrAlign();
253     }
254 
255     protected void addAddressLookup(String name) {
256         sb.append("RuntimeHelper.lookupGlobalVariable(LIBRARIES, \"" + name + "\")");
257     }

259     private void addFunction(FunctionDescriptor f) {
260         final boolean noArgs = f.argumentLayouts().isEmpty();
261         if (f.returnLayout().isPresent()) {
262             sb.append("FunctionDescriptor.of(");
263             addLayout(f.returnLayout().get());
264             if (!noArgs) {
265                 sb.append(", ");
266             }
267         } else {
268             sb.append("FunctionDescriptor.ofVoid(");
269         }
270         if (!noArgs) {
271             sb.append("\n");
272             incrAlign();
273             String delim = "";
274             for (MemoryLayout e : f.argumentLayouts()) {
275                 sb.append(delim);
276                 indent();
277                 addLayout(e);
278                 delim = ",\n";
279             }
280             sb.append("\n");
281             decrAlign();
282             indent();
283         }
284         sb.append(")");
285     }
286 
287     protected void addAddress(String name) {
288         incrAlign();
289         indent();
290         sb.append(PUB_MODS + "MemoryAddress " + javaSafeIdentifier(name) + "$ADDR" + " = ");
291         addAddressLookup(name);
292         sb.append(";\n");
293         decrAlign();
294     }

296     protected void addConstant(String name, Class<?> type, Object value) {
297         incrAlign();
298         indent();
299         if (type == MemoryAddress.class || type == MemorySegment.class) {
300             //todo, skip for now (address constants and string constants)
301         } else {
302             sb.append(PUB_MODS + type.getName() + " " + javaSafeIdentifier(name));
303             sb.append(" = ");
304             if (type == float.class) {
305                 sb.append(value);
306                 sb.append("f");
307             } else if (type == long.class) {
308                 sb.append(value);
309                 sb.append("L");
310             } else if (type == double.class) {
311                 sb.append(value);
312                 sb.append("d");
313             } else {
314                 sb.append("(" + type.getName() + ")");
315                 sb.append(value + "L");
316             }
317             sb.append(";\n");
318         }
319 
320         decrAlign();
321     }

323     static int funcIntfCounter = 0;
324 
325     protected void addUpcallFactory(FunctionDescriptor desc) {
326         String fnName = "FI" + funcIntfCounter++;
327         incrAlign();
328         indent();
329         sb.append(PRI_MODS + "FunctionDescriptor " + fnName + "$DESC = ");
330         addFunction(desc);
331         sb.append(";\n");
332         indent();
333         sb.append(PUB_MODS + "MemoryAddress " + fnName + "$make(MethodHandle handle) {\n");
334         incrAlign();
335         indent();
336         sb.append("return RuntimeHelper.upcallStub(handle, " + fnName + "$DESC);\n");
337         decrAlign();
338         indent();
339         sb.append("}\n");
340         decrAlign();
341     }

343     protected void addStaticFunctionWrapper(Declaration.Function f, MethodType mtype) {
344         incrAlign();
345         indent();
346         sb.append(PUB_MODS + mtype.returnType().getName() + " " + javaSafeIdentifier(f.name()) + " (");
347         String delim = "";
348         List<String> pNames = new ArrayList<>();
349         final int numParams = f.parameters().size();
350         for (int i = 0 ; i < numParams; i++) {
351             String pName = f.parameters().get(i).name();
352             if (pName.isEmpty()) {
353                 pName = "x" + i;
354             }
355             pName = javaSafeIdentifier(pName);
356             pNames.add(pName);
357             sb.append(delim + mtype.parameterType(i).getName() + " " + pName);
358             delim = ", ";
359         }
360         if (f.type().varargs()) {
361             String lastArg = "x" + numParams;
362             if (numParams > 0) {
363                 sb.append(", ");
364             }
365             sb.append("Object... " + lastArg);
366             pNames.add(lastArg);
367         }
368         sb.append(") {\n");
369         incrAlign();
370         indent();
371         sb.append("try {\n");
372         incrAlign();
373         indent();
374         if (!mtype.returnType().equals(void.class)) {
375             sb.append("return (" + mtype.returnType().getName() + ")");
376         }
377         sb.append(f.name() + ".invokeExact(" + String.join(", ", pNames) + ");\n");
378         decrAlign();
379         indent();
380         sb.append("} catch (Throwable ex) {\n");
381         incrAlign();
382         indent();
383         sb.append("throw new AssertionError(ex);\n");
384         decrAlign();
385         indent();
386         sb.append("}\n");
387         decrAlign();
388         indent();
389         sb.append("}\n");
390         decrAlign();
391     }

393     void addDescriptor(String name, FunctionDescriptor desc) {
394         incrAlign();
395         indent();
396         sb.append(PRI_MODS + "FunctionDescriptor " + name + "$DESC = ");
397         addFunction(desc);
398         sb.append(";\n");
399         decrAlign();
400         indent();
401     }

403     void addFunctionalInterface(String name, MethodType mtype) {
404         incrAlign();
405         indent();
406         sb.append("public interface " + javaSafeIdentifier(name) + " {\n");
407         incrAlign();
408         indent();
409         sb.append(mtype.returnType().getName() + " apply(");
410         String delim = "";
411         for (int i = 0 ; i < mtype.parameterCount() ; i++) {
412             sb.append(delim + mtype.parameterType(i).getName() + " x" + i);
413             delim = ", ";
414         }
415         sb.append(");\n");
416         decrAlign();
417         indent();
418         sb.append("}\n");
419         decrAlign();
420         indent();
421     }

423     protected void addFunctionalFactory(String name, MethodType mtype) {
424         incrAlign();
425         indent();
426         sb.append(PUB_MODS + "MemoryAddress " + name + "$make(" + name + " fi) {\n");
427         incrAlign();
428         indent();
429         sb.append("return RuntimeHelper.upcallStub(" + name + ".class, fi, " + name + "$DESC, " +
430                 "\"" + mtype.toMethodDescriptorString() + "\");\n");
431         decrAlign();
432         indent();
433         sb.append("}\n");
434         decrAlign();
435     }

437     void addGetter(String name, Class<?> type, Declaration parent) {
438         incrAlign();
439         indent();
440         name = javaSafeIdentifier(name);
441         String vhName = (parent != null ? (javaSafeIdentifier(parent.name()) + "$") : "") + name;
442         String param = parent != null ? (MemorySegment.class.getName() + " seg") : "";
443         sb.append(PUB_MODS + type.getName() + " " + vhName + "$get(" + param + ") {\n");
444         incrAlign();
445         indent();
446         String vhParam = parent != null ?
447                 "seg.baseAddress()" : name + "$ADDR";
448         sb.append("return (" + type.getName() + ")" + vhName + ".get(" + vhParam + ");\n");
449         decrAlign();
450         indent();
451         sb.append("}\n");
452         decrAlign();
453     }

455     void addSetter(String name, Class<?> type, Declaration parent) {
456         incrAlign();
457         indent();
458         name = javaSafeIdentifier(name);
459         String vhName = (parent != null ? (javaSafeIdentifier(parent.name()) + "$") : "") + name;
460         String param = parent != null ? (MemorySegment.class.getName() + " seg, ") : "";
461         sb.append(PUB_MODS + "void " + vhName + "$set(" + param + type.getName() + " x) {\n");
462         incrAlign();
463         indent();
464         String vhParam = parent != null ?
465                 "seg.baseAddress()" : name + "$ADDR";
466         sb.append(vhName + ".set(" + vhParam + ", x);\n");
467         decrAlign();
468         indent();
469         sb.append("}\n");
470         decrAlign();
471     }
472 
473     protected String build() {
474         String res = sb.toString();
475         this.sb = null;
476         return res.toString();
477     }
478 
479     protected void indent() {
480         for (int i = 0; i < align; i++) {
481             sb.append("    ");
482         }
483     }
484 
485     protected void incrAlign() {
486         align++;
487     }
488 
489     protected void decrAlign() {
490         align--;
491     }
492 
493     protected final String javaSafeIdentifier(String name) {
494         // We never get the problem of Java non-identifiers (like 123, ab-xy) as
495         // C identifiers. But we may have a java keyword used as a C identifier.
496         assert SourceVersion.isIdentifier(name);
497 
498         return SourceVersion.isKeyword(name)? (name + "_") : name;
499     }
