 46 /**
 47  * A helper class to generate header interface class in source form.
 48  * After aggregating various constituents of a .java source, build
 49  * method is called to get overall generated source string.
 50  */
 51 class JavaSourceBuilder {
 52     private static final String ABI = SystemABI.getInstance().name();
 53     // buffer
 54     protected StringBuffer sb;
 55     // current line alignment (number of 4-spaces)
 56     protected int align;
 57 
 58     JavaSourceBuilder(int align) {
 59         this.align = align;
 60         this.sb = new StringBuffer();
 61     }
 62 
 63     JavaSourceBuilder() {
 64         this(0);
 65     }
 66 
 67     protected int align() {
 68         return align;
 69     }
 70 
 71     final String PUB_CLS_MODS = "public final ";
 72     final String PUB_MODS = "public static final ";
 73     final String PRI_MODS = "private static final ";
 74 
 75     protected void addPackagePrefix(String pkgName) {
 76         assert pkgName.indexOf('/') == -1 : "package name invalid: " + pkgName;
 77         sb.append("// Generated by jextract\n\n");
 78         if (!pkgName.isEmpty()) {
 79             sb.append("package ");
 80             sb.append(pkgName);
 81             sb.append(";\n\n");
 82         }
 83         addImportSection();
 84     }
 85 
 86     protected void addImportSection() {
 87         sb.append("import java.lang.invoke.MethodHandle;\n");
 88         sb.append("import java.lang.invoke.VarHandle;\n");
 89         sb.append("import jdk.incubator.foreign.*;\n");
 90         sb.append("import jdk.incubator.foreign.MemoryLayout.PathElement;\n");
 91         sb.append("import static ");
 92         sb.append(HandleSourceFactory.C_LANG_CONSTANTS_HOLDER);
 93         sb.append(".*;\n");
 94     }
 95 
 96     protected void classBegin(String name) {
 97         indent();
 98         sb.append(PUB_CLS_MODS + "class ");
 99         sb.append(name);
100         sb.append(" {\n\n");
101     }
102 
103     protected void classEnd() {
104         indent();
105         sb.append("}\n\n");
106     }

108     protected void addLibraries(String[] libraryNames) {
109         incrAlign();
110         indent();
111         sb.append(PRI_MODS + "LibraryLookup[] LIBRARIES = RuntimeHelper.libraries(");
112         sb.append(stringArray(libraryNames) + ");\n");
113         decrAlign();
114     }

116     private String stringArray(String[] elements) {
117         return Stream.of(elements)
118                 .map(n -> "\"" + n + "\"")
119                 .collect(Collectors.joining(",", "new String[] {", "}"));
120     }
121 
122     protected void addLayout(String elementName, MemoryLayout layout) {
123         incrAlign();
124         indent();
125         sb.append(PUB_MODS + "MemoryLayout " + elementName + "$LAYOUT = ");
126         addLayout(layout);
127         sb.append(";\n");
128         decrAlign();
129     }
130 
131     private void addLayout(MemoryLayout l) {
132         if (l instanceof ValueLayout) {
133             SystemABI.Type type = l.abiType().orElseThrow(()->new AssertionError("Should not get here: " + l));
134             sb.append(switch (type) {
135                 case BOOL -> "C_BOOL";
136                 case SIGNED_CHAR -> "C_SCHAR";
137                 case UNSIGNED_CHAR -> "C_UCHAR";
138                 case CHAR -> "C_CHAR";
139                 case SHORT -> "C_SHORT";
140                 case UNSIGNED_SHORT -> "C_USHORT";
141                 case INT -> "C_INT";
142                 case UNSIGNED_INT -> "C_UINT";
143                 case LONG -> "C_LONG";
144                 case UNSIGNED_LONG -> "C_ULONG";
145                 case LONG_LONG -> "C_LONGLONG";
146                 case UNSIGNED_LONG_LONG -> "C_ULONGLONG";
147                 case FLOAT -> "C_FLOAT";
148                 case DOUBLE -> "C_DOUBLE";
149                 case LONG_DOUBLE -> "C_LONGDOUBLE";
150                 case POINTER -> "C_POINTER";
151                 default -> { throw new RuntimeException("should not reach here: " + type); }
152             });
153         } else if (l instanceof SequenceLayout) {
154             sb.append("MemoryLayout.ofSequence(");
155             if (((SequenceLayout) l).elementCount().isPresent()) {
156                 sb.append(((SequenceLayout) l).elementCount().getAsLong() + ", ");
157             }
158             addLayout(((SequenceLayout) l).elementLayout());
159             sb.append(")");
160         } else if (l instanceof GroupLayout) {
161             SystemABI.Type type = l.abiType().orElse(null);
162             if (type == SystemABI.Type.COMPLEX_LONG_DOUBLE) {
163                 if (!ABI.equals(SystemABI.ABI_SYSV)) {
164                     throw new RuntimeException("complex long double is supported only for SysV ABI");
165                 } else {
166                     sb.append("C_COMPLEX_LONGDOUBLE");
167                 }
168             } else {
169                 if (((GroupLayout) l).isStruct()) {
170                     sb.append("MemoryLayout.ofStruct(\n");
171                 } else {
172                     sb.append("MemoryLayout.ofUnion(\n");
173                 }
174                 incrAlign();
175                 String delim = "";
176                 for (MemoryLayout e : ((GroupLayout) l).memberLayouts()) {
177                     sb.append(delim);
178                     indent();
179                     addLayout(e);
180                     delim = ",\n";
181                 }
182                 sb.append("\n");
183                 decrAlign();
184                 indent();
185                 sb.append(")");
186             }
187         } else {
188             //padding
189             sb.append("MemoryLayout.ofPaddingBits(" + l.bitSize() + ")");
190         }
191         if (l.name().isPresent()) {
192             sb.append(".withName(\"" +  l.name().get() + "\")");
193         }
194     }

196     protected void addVarHandle(String name, Class<?> type, String parentName) {
197         incrAlign();
198         indent();
199         String vhName = parentName != null ?
200                 parentName + "$" + name :
201                 name;
202         sb.append(PUB_MODS + "VarHandle " + vhName + " = ");
203         if (parentName != null) {
204             addHandlePath(type, parentName, name);
205         } else {
206             addHandlePath(type, name);
207         }
208         sb.append(";\n");
209         decrAlign();
210     }
211 
212     protected void addHandlePath(Class<?> type, String strName, String fieldName) {
213         String ty = type.getName();
214         if (ty.contains("MemoryAddress")) {
215             ty = "long";
216         }
217         sb.append(strName + "$LAYOUT.varHandle(" + ty + ".class, ");
218         sb.append("PathElement.groupElement(\"" + fieldName +"\")");
219         sb.append(")");
220     }
221 
222     protected void addHandlePath(Class<?> type, String varName) {
223         String ty = type.getName();
224         if (ty.contains("MemoryAddress")) {
225             ty = "long";
226         }
227         sb.append(varName + "$LAYOUT.varHandle(" + ty + ".class)");
228     }

230     protected void addMethodHandle(Declaration.Function funcTree, MethodType mtype, FunctionDescriptor desc) {
231         incrAlign();
232         indent();
233         sb.append(PUB_MODS + "MethodHandle " + funcTree.name() + " = ");
234         sb.append("RuntimeHelper.downcallHandle(\n");
235         incrAlign();
236         indent();
237         sb.append("LIBRARIES, \"" + funcTree.name() + "\"");
238         sb.append(",\n");
239         indent();
240         sb.append("\"" + mtype.toMethodDescriptorString() + "\",\n");
241         indent();
242         addFunction(desc);
243         sb.append(",\n");
244         indent();
245         sb.append(funcTree.type().varargs());
246         decrAlign();
247         sb.append("\n");
248         indent();
249         sb.append(");\n");
250         decrAlign();
251     }
252 
253     protected void addAddressLookup(String name) {
254         sb.append("RuntimeHelper.lookupGlobalVariable(LIBRARIES, \"" + name + "\")");
255     }

257     private void addFunction(FunctionDescriptor f) {
258         final boolean noArgs = f.argumentLayouts().isEmpty();
259         if (f.returnLayout().isPresent()) {
260             sb.append("FunctionDescriptor.of(");
261             addLayout(f.returnLayout().get());
262             if (!noArgs) {
263                 sb.append(", ");
264             }
265         } else {
266             sb.append("FunctionDescriptor.ofVoid(");
267         }
268         if (!noArgs) {
269             sb.append("\n");
270             incrAlign();
271             String delim = "";
272             for (MemoryLayout e : f.argumentLayouts()) {
273                 sb.append(delim);
274                 indent();
275                 addLayout(e);
276                 delim = ",\n";
277             }
278             sb.append("\n");
279             decrAlign();
280             indent();
281         }
282         sb.append(")");
283     }

285     protected void addAddress(String name) {
286         incrAlign();
287         indent();
288         sb.append(PUB_MODS + "MemoryAddress " + name + "$ADDR" + " = ");
289         addAddressLookup(name);
290         sb.append(";\n");
291         decrAlign();
292     }

294     protected void addConstant(String name, Class<?> type, Object value) {
295         incrAlign();
296         indent();
297         if (type == MemoryAddress.class || type == MemorySegment.class) {
298             //todo, skip for now (address constants and string constants)
299         } else {
300             sb.append(PUB_MODS + type.getName() + " " + name);
301             sb.append(" = ");
302             if (type == float.class) {
303                 sb.append(value);
304                 sb.append("f");
305             } else if (type == long.class) {
306                 sb.append(value);
307                 sb.append("L");
308             } else if (type == double.class) {
309                 sb.append(value);
310                 sb.append("d");
311             } else {
312                 sb.append("(" + type.getName() + ")");
313                 sb.append(value + "L");
314             }
315             sb.append(";\n");
316         }
317 
318         decrAlign();
319     }

321     static int funcIntfCounter = 0;
322 
323     protected void addUpcallFactory(FunctionDescriptor desc) {
324         String fnName = "FI" + funcIntfCounter++;
325         incrAlign();
326         indent();
327         sb.append(PRI_MODS + "FunctionDescriptor " + fnName + "$DESC = ");
328         addFunction(desc);
329         sb.append(";\n");
330         indent();
331         sb.append(PUB_MODS + "MemoryAddress " + fnName + "$make(MethodHandle handle) {\n");
332         incrAlign();
333         indent();
334         sb.append("return RuntimeHelper.upcallStub(handle, " + fnName + "$DESC);\n");
335         decrAlign();
336         indent();
337         sb.append("}\n");
338         decrAlign();
339     }

341     protected void addStaticFunctionWrapper(Declaration.Function f, MethodType mtype) {
342         incrAlign();
343         indent();
344         sb.append(PUB_MODS + mtype.returnType().getName() + " " + f.name() + " (");
345         String delim = "";
346         List<String> pNames = new ArrayList<>();
347         final int numParams = f.parameters().size();
348         for (int i = 0 ; i < numParams; i++) {
349             String pName = f.parameters().get(i).name();
350             if (pName.isEmpty()) {
351                 pName = "x" + i;
352             }
353             pNames.add(pName);
354             sb.append(delim + mtype.parameterType(i).getName() + " " + pName);
355             delim = ", ";
356         }
357         if (f.type().varargs()) {
358             String lastArg = "x" + numParams;
359             if (numParams > 0) {
360                 sb.append(", ");
361             }
362             sb.append("Object... " + lastArg);
363             pNames.add(lastArg);
364         }
365         sb.append(") {\n");
366         incrAlign();
367         indent();
368         sb.append("try {\n");
369         incrAlign();
370         indent();
371         if (!mtype.returnType().equals(void.class)) {
372             sb.append("return (" + mtype.returnType().getName() + ")");
373         }
374         sb.append(f.name() + ".invokeExact(" + String.join(", ", pNames) + ");\n");
375         decrAlign();
376         indent();
377         sb.append("} catch (Throwable ex) {\n");
378         incrAlign();
379         indent();
380         sb.append("throw new AssertionError(ex);\n");
381         decrAlign();
382         indent();
383         sb.append("}\n");
384         decrAlign();
385         indent();
386         sb.append("}\n");
387         decrAlign();
388     }

390     void addDescriptor(String name, FunctionDescriptor desc) {
391         incrAlign();
392         indent();
393         sb.append(PRI_MODS + "FunctionDescriptor " + name + "$DESC = ");
394         addFunction(desc);
395         sb.append(";\n");
396         decrAlign();
397         indent();
398     }

400     void addFunctionalInterface(String name, MethodType mtype) {
401         incrAlign();
402         indent();
403         sb.append("public interface " + name + " {\n");
404         incrAlign();
405         indent();
406         sb.append(mtype.returnType().getName() + " apply(");
407         String delim = "";
408         for (int i = 0 ; i < mtype.parameterCount() ; i++) {
409             sb.append(delim + mtype.parameterType(i).getName() + " x" + i);
410             delim = ", ";
411         }
412         sb.append(");\n");
413         decrAlign();
414         indent();
415         sb.append("}\n");
416         decrAlign();
417         indent();
418     }

420     protected void addFunctionalFactory(String name, MethodType mtype) {
421         incrAlign();
422         indent();
423         sb.append(PUB_MODS + "MemoryAddress " + name + "$make(" + name + " fi) {\n");
424         incrAlign();
425         indent();
426         sb.append("return RuntimeHelper.upcallStub(" + name + ".class, fi, " + name + "$DESC, " +
427                 "\"" + mtype.toMethodDescriptorString() + "\");\n");
428         decrAlign();
429         indent();
430         sb.append("}\n");
431         decrAlign();
432     }

434     void addGetter(String name, Class<?> type, Declaration parent) {
435         incrAlign();
436         indent();
437         String vhName = (parent != null ? (parent.name() + "$") : "") + name;
438         String param = parent != null ? (MemorySegment.class.getName() + " seg") : "";
439         sb.append(PUB_MODS + type.getName() + " " + vhName + "$get(" + param + ") {\n");
440         incrAlign();
441         indent();
442         String vhParam = parent != null ?
443                 "seg.baseAddress()" : name + "$ADDR";
444         sb.append("return (" + type.getName() + ")" + vhName + ".get(" + vhParam + ");\n");
445         decrAlign();
446         indent();
447         sb.append("}\n");
448         decrAlign();
449     }
450 
451     void addSetter(String name, Class<?> type, Declaration parent) {
452         incrAlign();
453         indent();
454         String vhName = (parent != null ? (parent.name() + "$") : "") + name;
455         String param = parent != null ? (MemorySegment.class.getName() + " seg, ") : "";
456         sb.append(PUB_MODS + "void " + vhName + "$set(" + param + type.getName() + " x) {\n");
457         incrAlign();
458         indent();
459         String vhParam = parent != null ?
460                 "seg.baseAddress()" : name + "$ADDR";
461         sb.append(vhName + ".set(" + vhParam + ", x);\n");
462         decrAlign();
463         indent();
464         sb.append("}\n");
465         decrAlign();
466     }
467 
468     protected String build() {
469         String res = sb.toString();
470         this.sb = null;
471         return res.toString();
472     }
473 
474     protected void indent() {
475         for (int i = 0; i < align; i++) {
476             sb.append("    ");
477         }
478     }
479 
480     protected void incrAlign() {
481         align++;
482     }
483 
484     protected void decrAlign() {
485         align--;
486     }
