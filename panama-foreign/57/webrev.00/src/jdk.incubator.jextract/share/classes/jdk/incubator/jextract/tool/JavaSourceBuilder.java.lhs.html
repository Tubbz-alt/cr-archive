 47 /**
 48  * A helper class to generate header interface class in source form.
 49  * After aggregating various constituents of a .java source, build
 50  * method is called to get overall generated source string.
 51  */
 52 class JavaSourceBuilder {
 53     private static final String ABI = SystemABI.getInstance().name();
 54     // buffer
 55     protected StringBuffer sb;
 56     // current line alignment (number of 4-spaces)
 57     protected int align;
 58 
 59     JavaSourceBuilder(int align) {
 60         this.align = align;
 61         this.sb = new StringBuffer();
 62     }
 63 
 64     JavaSourceBuilder() {
 65         this(0);
 66     }
 67 
 68     protected int align() {
 69         return align;
 70     }

 72     final String PUB_CLS_MODS = "public final ";
 73     final String PUB_MODS = "public static final ";
 74     final String PRI_MODS = "private static final ";
 75 
 76     protected void addPackagePrefix(String pkgName) {
 77         assert pkgName.indexOf('/') == -1 : "package name invalid: " + pkgName;
 78         sb.append("// Generated by jextract\n\n");
 79         if (!pkgName.isEmpty()) {
 80             sb.append("package ");
 81             sb.append(pkgName);
 82             sb.append(";\n\n");
 83         }
 84         addImportSection();
 85     }
 86 
 87     protected void addImportSection() {
 88         sb.append("import java.lang.invoke.MethodHandle;\n");
 89         sb.append("import java.lang.invoke.VarHandle;\n");
 90         sb.append("import jdk.incubator.foreign.*;\n");
 91         sb.append("import jdk.incubator.foreign.MemoryLayout.PathElement;\n");
 92         sb.append("import static ");
 93         sb.append(HandleSourceFactory.C_LANG_CONSTANTS_HOLDER);
 94         sb.append(".*;\n");
 95     }

 97     protected void classBegin(String name) {
 98         indent();
 99         sb.append(PUB_CLS_MODS + "class ");
100         sb.append(name);
101         sb.append(" {\n\n");
102     }

104     protected void classEnd() {
105         indent();
106         sb.append("}\n\n");
107     }

109     protected void addLibraries(String[] libraryNames) {
110         incrAlign();
111         indent();
112         sb.append(PRI_MODS + "LibraryLookup[] LIBRARIES = RuntimeHelper.libraries(");
113         sb.append(stringArray(libraryNames) + ");\n");
114         decrAlign();
115     }

117     private String stringArray(String[] elements) {
118         return Stream.of(elements)
119                 .map(n -> "\"" + n + "\"")
120                 .collect(Collectors.joining(",", "new String[] {", "}"));
121     }

123     protected void addLayout(String elementName, MemoryLayout layout) {
124         incrAlign();
125         indent();
126         sb.append(PUB_MODS + "MemoryLayout " + javaSafeIdentifier(elementName) + "$LAYOUT = ");
127         addLayout(layout);
128         sb.append(";\n");
129         decrAlign();
130     }
131 
132     private void addLayout(MemoryLayout l) {
133         if (l instanceof ValueLayout) {
134             SystemABI.Type type = l.abiType().orElseThrow(()->new AssertionError("Should not get here: " + l));
135             sb.append(switch (type) {
136                 case BOOL -> "C_BOOL";
137                 case SIGNED_CHAR -> "C_SCHAR";
138                 case UNSIGNED_CHAR -> "C_UCHAR";
139                 case CHAR -> "C_CHAR";
140                 case SHORT -> "C_SHORT";
141                 case UNSIGNED_SHORT -> "C_USHORT";
142                 case INT -> "C_INT";
143                 case UNSIGNED_INT -> "C_UINT";
144                 case LONG -> "C_LONG";
145                 case UNSIGNED_LONG -> "C_ULONG";
146                 case LONG_LONG -> "C_LONGLONG";
147                 case UNSIGNED_LONG_LONG -> "C_ULONGLONG";
148                 case FLOAT -> "C_FLOAT";
149                 case DOUBLE -> "C_DOUBLE";
150                 case LONG_DOUBLE -> "C_LONGDOUBLE";
151                 case POINTER -> "C_POINTER";
152                 default -> { throw new RuntimeException("should not reach here: " + type); }
153             });
154         } else if (l instanceof SequenceLayout) {
155             sb.append("MemoryLayout.ofSequence(");
156             if (((SequenceLayout) l).elementCount().isPresent()) {
157                 sb.append(((SequenceLayout) l).elementCount().getAsLong() + ", ");
158             }
159             addLayout(((SequenceLayout) l).elementLayout());
160             sb.append(")");
161         } else if (l instanceof GroupLayout) {
162             SystemABI.Type type = l.abiType().orElse(null);
163             if (type == SystemABI.Type.COMPLEX_LONG_DOUBLE) {
164                 if (!ABI.equals(SystemABI.ABI_SYSV)) {
165                     throw new RuntimeException("complex long double is supported only for SysV ABI");
166                 } else {
167                     sb.append("C_COMPLEX_LONGDOUBLE");
168                 }
169             } else {
170                 if (((GroupLayout) l).isStruct()) {
171                     sb.append("MemoryLayout.ofStruct(\n");
172                 } else {
173                     sb.append("MemoryLayout.ofUnion(\n");
174                 }
175                 incrAlign();
176                 String delim = "";
177                 for (MemoryLayout e : ((GroupLayout) l).memberLayouts()) {
178                     sb.append(delim);
179                     indent();
180                     addLayout(e);
181                     delim = ",\n";
182                 }
183                 sb.append("\n");
184                 decrAlign();
185                 indent();
186                 sb.append(")");
187             }
188         } else {
189             //padding
190             sb.append("MemoryLayout.ofPaddingBits(" + l.bitSize() + ")");
191         }
192         if (l.name().isPresent()) {
193             sb.append(".withName(\"" +  l.name().get() + "\")");
194         }
195     }

197     protected void addVarHandle(String name, Class<?> type, String parentName) {
198         incrAlign();
199         indent();
200         parentName = parentName != null? javaSafeIdentifier(parentName) : parentName;
201         name = javaSafeIdentifier(name);
202         String vhName = parentName != null ?
203                 parentName + "$" + name : name;
204         sb.append(PUB_MODS + "VarHandle " + vhName + " = ");
205         if (parentName != null) {
206             addHandlePath(type, parentName, name);
207         } else {
208             addHandlePath(type, name);
209         }
210         sb.append(";\n");
211         decrAlign();
212     }

214     protected void addHandlePath(Class<?> type, String strName, String fieldName) {
215         String ty = type.getName();
216         if (ty.contains("MemoryAddress")) {
217             ty = "long";
218         }
219         sb.append(strName + "$LAYOUT.varHandle(" + ty + ".class, ");
220         sb.append("PathElement.groupElement(\"" + fieldName +"\")");
221         sb.append(")");
222     }
223 
224     protected void addHandlePath(Class<?> type, String varName) {
225         String ty = type.getName();
226         if (ty.contains("MemoryAddress")) {
227             ty = "long";
228         }
229         sb.append(varName + "$LAYOUT.varHandle(" + ty + ".class)");
230     }

232     protected void addMethodHandle(Declaration.Function funcTree, MethodType mtype, FunctionDescriptor desc) {
233         incrAlign();
234         indent();
235         sb.append(PUB_MODS + "MethodHandle " + javaSafeIdentifier(funcTree.name()) + " = ");
236         sb.append("RuntimeHelper.downcallHandle(\n");
237         incrAlign();
238         indent();
239         sb.append("LIBRARIES, \"" + funcTree.name() + "\"");
240         sb.append(",\n");
241         indent();
242         sb.append("\"" + mtype.toMethodDescriptorString() + "\",\n");
243         indent();
244         addFunction(desc);
245         sb.append(",\n");
246         indent();
247         sb.append(funcTree.type().varargs());
248         decrAlign();
249         sb.append("\n");
250         indent();
251         sb.append(");\n");
252         decrAlign();
253     }

255     protected void addAddressLookup(String name) {
256         sb.append("RuntimeHelper.lookupGlobalVariable(LIBRARIES, \"" + name + "\")");
257     }

259     private void addFunction(FunctionDescriptor f) {
260         final boolean noArgs = f.argumentLayouts().isEmpty();
261         if (f.returnLayout().isPresent()) {
262             sb.append("FunctionDescriptor.of(");
263             addLayout(f.returnLayout().get());
264             if (!noArgs) {
265                 sb.append(", ");
266             }
267         } else {
268             sb.append("FunctionDescriptor.ofVoid(");
269         }
270         if (!noArgs) {
271             sb.append("\n");
272             incrAlign();
273             String delim = "";
274             for (MemoryLayout e : f.argumentLayouts()) {
275                 sb.append(delim);
276                 indent();
277                 addLayout(e);
278                 delim = ",\n";
279             }
280             sb.append("\n");
281             decrAlign();
282             indent();
283         }
284         sb.append(")");
285     }

287     protected void addAddress(String name) {
288         incrAlign();
289         indent();
290         sb.append(PUB_MODS + "MemoryAddress " + javaSafeIdentifier(name) + "$ADDR" + " = ");
291         addAddressLookup(name);
292         sb.append(";\n");
293         decrAlign();

296     protected void addConstant(String name, Class<?> type, Object value) {
297         incrAlign();
298         indent();
299         if (type == MemoryAddress.class || type == MemorySegment.class) {
300             //todo, skip for now (address constants and string constants)
301         } else {
302             sb.append(PUB_MODS + type.getName() + " " + javaSafeIdentifier(name));
303             sb.append(" = ");
304             if (type == float.class) {
305                 float f = ((Number)value).floatValue();
306                 if (Float.isNaN(f)) {
307                     sb.append("Float.NaN");
308                 } else if (Float.isInfinite(f)) {
309                     sb.append(f > 0? "Float.POSITIVE_INFINITY" : "Float.NEGATIVE_INFINITY");
310                 } else {
311                     sb.append(value);
312                     sb.append("f");
313                 }
314             } else if (type == long.class) {
315                 sb.append(value);
316                 sb.append("L");
317             } else if (type == double.class) {
318                 double d = ((Number)value).doubleValue();
319                 if (Double.isNaN(d)) {
320                     sb.append("Double.NaN");
321                 } else if (Double.isInfinite(d)) {
322                     sb.append(d > 0? "Double.POSITIVE_INFINITY" : "Double.NEGATIVE_INFINITY");
323                 } else {
324                    sb.append(value);
325                    sb.append("d");
326                 }
327             } else {
328                 sb.append("(" + type.getName() + ")");
329                 sb.append(value + "L");
330             }
331             sb.append(";\n");
332         }
333 
334         decrAlign();
335     }

339     protected void addUpcallFactory(FunctionDescriptor desc) {
340         String fnName = "FI" + funcIntfCounter++;
341         incrAlign();
342         indent();
343         sb.append(PRI_MODS + "FunctionDescriptor " + fnName + "$DESC = ");
344         addFunction(desc);
345         sb.append(";\n");
346         indent();
347         sb.append(PUB_MODS + "MemoryAddress " + fnName + "$make(MethodHandle handle) {\n");
348         incrAlign();
349         indent();
350         sb.append("return RuntimeHelper.upcallStub(handle, " + fnName + "$DESC);\n");
351         decrAlign();
352         indent();
353         sb.append("}\n");
354         decrAlign();
355     }

357     protected void addStaticFunctionWrapper(Declaration.Function f, MethodType mtype) {
358         incrAlign();
359         indent();
360         sb.append(PUB_MODS + mtype.returnType().getName() + " " + javaSafeIdentifier(f.name()) + " (");
361         String delim = "";
362         List<String> pNames = new ArrayList<>();
363         final int numParams = f.parameters().size();
364         for (int i = 0 ; i < numParams; i++) {
365             String pName = f.parameters().get(i).name();
366             if (pName.isEmpty()) {
367                 pName = "x" + i;
368             }
369             pName = javaSafeIdentifier(pName);
370             pNames.add(pName);
371             sb.append(delim + mtype.parameterType(i).getName() + " " + pName);
372             delim = ", ";
373         }
374         if (f.type().varargs()) {
375             String lastArg = "x" + numParams;
376             if (numParams > 0) {
377                 sb.append(", ");
378             }
379             sb.append("Object... " + lastArg);
380             pNames.add(lastArg);
381         }
382         sb.append(") {\n");
383         incrAlign();
384         indent();
385         sb.append("try {\n");
386         incrAlign();
387         indent();
388         if (!mtype.returnType().equals(void.class)) {
389             sb.append("return (" + mtype.returnType().getName() + ")");
390         }
391         sb.append(f.name() + ".invokeExact(" + String.join(", ", pNames) + ");\n");
392         decrAlign();
393         indent();
394         sb.append("} catch (Throwable ex) {\n");
395         incrAlign();
396         indent();
397         sb.append("throw new AssertionError(ex);\n");
398         decrAlign();
399         indent();
400         sb.append("}\n");
401         decrAlign();
402         indent();
403         sb.append("}\n");
404         decrAlign();
405     }

407     void addDescriptor(String name, FunctionDescriptor desc) {
408         incrAlign();
409         indent();
410         sb.append(PRI_MODS + "FunctionDescriptor " + name + "$DESC = ");

417     void addFunctionalInterface(String name, MethodType mtype) {
418         incrAlign();
419         indent();
420         sb.append("public interface " + javaSafeIdentifier(name) + " {\n");
421         incrAlign();
422         indent();
423         sb.append(mtype.returnType().getName() + " apply(");
424         String delim = "";
425         for (int i = 0 ; i < mtype.parameterCount() ; i++) {
426             sb.append(delim + mtype.parameterType(i).getName() + " x" + i);
427             delim = ", ";
428         }
429         sb.append(");\n");
430         decrAlign();
431         indent();
432         sb.append("}\n");
433         decrAlign();
434         indent();
435     }

437     protected void addFunctionalFactory(String name, MethodType mtype) {
438         incrAlign();
439         indent();
440         sb.append(PUB_MODS + "MemoryAddress " + name + "$make(" + name + " fi) {\n");
441         incrAlign();
442         indent();
443         sb.append("return RuntimeHelper.upcallStub(" + name + ".class, fi, " + name + "$DESC, " +
444                 "\"" + mtype.toMethodDescriptorString() + "\");\n");
445         decrAlign();
446         indent();
447         sb.append("}\n");
448         decrAlign();

451     void addGetter(String name, Class<?> type, String parent) {
452         incrAlign();
453         indent();
454         name = javaSafeIdentifier(name);
455         String vhName = (parent != null ? (javaSafeIdentifier(parent) + "$") : "") + name;
456         String param = parent != null ? (MemorySegment.class.getName() + " seg") : "";
457         sb.append(PUB_MODS + type.getName() + " " + vhName + "$get(" + param + ") {\n");
458         incrAlign();
459         indent();
460         String vhParam = parent != null ?
461                 "seg.baseAddress()" : name + "$ADDR";
462         sb.append("return (" + type.getName() + ")" + vhName + ".get(" + vhParam + ");\n");
463         decrAlign();
464         indent();
465         sb.append("}\n");
466         decrAlign();
467     }

469     void addSetter(String name, Class<?> type, String parent) {
470         incrAlign();
471         indent();
472         name = javaSafeIdentifier(name);
473         String vhName = (parent != null ? (javaSafeIdentifier(parent) + "$") : "") + name;
474         String param = parent != null ? (MemorySegment.class.getName() + " seg, ") : "";
475         sb.append(PUB_MODS + "void " + vhName + "$set(" + param + type.getName() + " x) {\n");
476         incrAlign();
477         indent();
478         String vhParam = parent != null ?
479                 "seg.baseAddress()" : name + "$ADDR";
480         sb.append(vhName + ".set(" + vhParam + ", x);\n");
481         decrAlign();
482         indent();
483         sb.append("}\n");
484         decrAlign();
485     }

487     protected String build() {
488         String res = sb.toString();
489         this.sb = null;
490         return res.toString();
491     }
492 
493     protected void indent() {
494         for (int i = 0; i < align; i++) {
495             sb.append("    ");
496         }
497     }
498 
499     protected void incrAlign() {
500         align++;
501     }
502 
503     protected void decrAlign() {
504         align--;
505     }
506 
507     protected final String javaSafeIdentifier(String name) {
508         // We never get the problem of Java non-identifiers (like 123, ab-xy) as
509         // C identifiers. But we may have a java keyword used as a C identifier.
510         assert SourceVersion.isIdentifier(name);
511 
512         return SourceVersion.isKeyword(name)? (name + "_") : name;
513     }
