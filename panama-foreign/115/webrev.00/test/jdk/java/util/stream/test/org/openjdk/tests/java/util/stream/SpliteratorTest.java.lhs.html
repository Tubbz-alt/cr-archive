62         SpliteratorTestHelper.testDoubleSpliterator(supplier);
63     }
64

66     public void testSegmentSpliterator(String name, SequenceLayout layout, SpliteratorTestHelper.ContentAsserter<MemorySegment> contentAsserter) {

70                     SegmentTestDataProvider::segmentCopier, contentAsserter);
71         }
72     }
