106         private Object invoke(Object[] args) throws Throwable {
107             // one trailing Object[]
108             int nNamedArgs = function.argumentLayouts().size();
109             assert(args.length == nNamedArgs + 1);
110             // The last argument is the array of vararg collector
111             Object[] unnamedArgs = (Object[]) args[args.length - 1];
112 
113             int argsCount = nNamedArgs + unnamedArgs.length;
114             Class<?>[] argTypes = new Class<?>[argsCount];
115             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];
116 
117             int pos = 0;
118             for (pos = 0; pos < nNamedArgs; pos++) {
119                 argTypes[pos] = varargs.parameterType(pos);
120                 argLayouts[pos] = function.argumentLayouts().get(pos);
121             }
122 
123             assert pos == nNamedArgs;
124             for (Object o: unnamedArgs) {
125                 argTypes[pos] = normalize(o.getClass());
126                 argLayouts[pos] = variadicLayout(argTypes[pos]);
127                 pos++;
128             }
129             assert pos == argsCount;
130 
131             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);
132             FunctionDescriptor f = (function.returnLayout().isEmpty()) ?
133                     FunctionDescriptor.ofVoid(argLayouts) :
134                     FunctionDescriptor.of(function.returnLayout().get(), argLayouts);
135             MethodHandle mh = ABI.downcallHandle(symbol, mt, f);
136             // flatten argument list so that it can be passed to an asSpreader MH
137             Object[] allArgs = new Object[nNamedArgs + unnamedArgs.length];
138             System.arraycopy(args, 0, allArgs, 0, nNamedArgs);
139             System.arraycopy(unnamedArgs, 0, allArgs, nNamedArgs, unnamedArgs.length);
140 
141             return mh.asSpreader(Object[].class, argsCount).invoke(allArgs);
142         }

168         private Class<?> normalize(Class<?> c) {
169             c = unboxIfNeeded(c);
170             if (c.isPrimitive()) {

172             }

182         private MemoryLayout variadicLayout(Class<?> c) {
183             if (c == char.class || c == byte.class || c == short.class || c == int.class || c == long.class) {
