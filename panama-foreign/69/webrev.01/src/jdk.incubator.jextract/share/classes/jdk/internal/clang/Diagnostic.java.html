26 package jdk.internal.clang;
27 
28 import jdk.incubator.foreign.MemoryAddress;
29 import jdk.internal.clang.libclang.Index_h;
30 
31 public class Diagnostic {
32     final MemoryAddress ptr;
33 
34     // Various Diagnostic severity levels - from Clang enum CXDiagnosticSeverity
35 
36     /**
37      * A diagnostic that has been suppressed, e.g., by a command-line
38      * option.
39      */
40     public static final int CXDiagnostic_Ignored = 0;
41 
42     /**
43      * This diagnostic is a note that should be attached to the
44      * previous (non-note) diagnostic.
45      */
46     public static final int CXDiagnostic_Note    = 1;
47 
48     /**
49      * This diagnostic indicates suspicious code that may not be
50      * wrong.
51      */
52     public static final int CXDiagnostic_Warning = 2;
53 
54     /**
55      * This diagnostic indicates that the code is ill-formed.
56      */
57     public static final int CXDiagnostic_Error   = 3;
58 
59     /**
60      * This diagnostic indicates that the code is ill-formed such
61      * that future parser recovery is unlikely to produce useful
62      * results.
63      */
64     public static final int CXDiagnostic_Fatal   = 4;
65 
66     Diagnostic(MemoryAddress ptr) {
67         this.ptr = ptr;
68     }
69 
70     public int severity() {
71         return Index_h.clang_getDiagnosticSeverity(ptr);
72     }
73 
74     public SourceLocation location() {
75         return new SourceLocation(Index_h.clang_getDiagnosticLocation(ptr));
76     }
77 
78     public String spelling() {
79         return LibClang.CXStrToString(
80                 Index_h.clang_getDiagnosticSpelling(ptr));
81     }
82 
83     public void dispose() {
84         Index_h.clang_disposeDiagnostic(ptr);
85     }
86 
87     @Override
88     public String toString() {
89         return LibClang.CXStrToString(
90                 Index_h.clang_formatDiagnostic(ptr,
91                     Index_h.clang_defaultDiagnosticDisplayOptions()));
92     }
93 }
