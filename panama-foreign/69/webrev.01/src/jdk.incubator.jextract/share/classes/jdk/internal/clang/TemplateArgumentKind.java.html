27 package jdk.internal.clang;
28 
29 import java.util.HashMap;
30 import java.util.Map;
31 import java.util.NoSuchElementException;
32 
33 /**
34  * Describes the kind of a template argument.
35  */
36 public enum TemplateArgumentKind {
37     Null(0),
38     Type(1),
39     Declaration(2),
40     NullPtr(3),
41     Integral(4),
42     Template(5),
43     TemplateExpansion(6),
44     Expression(7),
45     Pack(8),
46     /* Indicates an error case, preventing the kind from being deduced. */
47     Invalid(9);
48 
49     private final int value;
50 
51     TemplateArgumentKind(int value) {
52         this.value = value;
53     }
54 
55     public int value() {
56         return value;
57     }
58 
59     private final static Map<Integer, TemplateArgumentKind> lookup;
60 
61     static {
62         lookup = new HashMap<>();
63         for (TemplateArgumentKind e: TemplateArgumentKind.values()) {
64             lookup.put(e.value(), e);
65         }
66     }
67 
68     public final static TemplateArgumentKind valueOf(int value) {
69         TemplateArgumentKind x = lookup.get(value);
70         if (null == x) {
71             throw new NoSuchElementException("Invalid TemplateArgumentKind kind value: " + value);
72         }
73         return x;
74     }
75 }
