27 package jdk.internal.clang;
28 
29 import java.util.HashMap;
30 import java.util.Map;
31 import java.util.NoSuchElementException;
32 
33 public enum RefQualifierKind {
34     /** No ref-qualifier was provided. */
35     CXRefQualifier_None(0),
36     /** An lvalue ref-qualifier was provided (\c &). */
37     CXRefQualifier_LValue(1),
38     /** An rvalue ref-qualifier was provided (\c &&). */
39     CXRefQualifier_RValue(2);
40 
41     private final int value;
42 
43     RefQualifierKind(int value) {
44         this.value = value;
45     }
46 
47     public int value() {
48         return value;
49     }
50 
51     private final static Map<Integer, RefQualifierKind> lookup;
52 
53     static {
54         lookup = new HashMap<>();
55         for (RefQualifierKind e: RefQualifierKind.values()) {
56             lookup.put(e.value(), e);
57         }
58     }
59 
60     public final static RefQualifierKind valueOf(int value) {
61         RefQualifierKind x = lookup.get(value);
62         if (null == x) {
63             throw new NoSuchElementException("Invalid RefQualifierKind kind value: " + value);
64         }
65         return x;
66     }
67 }
