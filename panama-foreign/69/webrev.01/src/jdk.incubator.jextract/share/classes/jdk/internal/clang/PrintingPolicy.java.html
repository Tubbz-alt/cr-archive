27 package jdk.internal.clang;
28 
29 import jdk.incubator.foreign.MemoryAddress;
30 import jdk.internal.clang.libclang.Index_h;
31 
32 public final class PrintingPolicy implements AutoCloseable {
33     private MemoryAddress policy;
34 
35     PrintingPolicy(MemoryAddress policy) {
36         this.policy = policy;
37     }
38 
39     MemoryAddress ptr() {
40         return policy;
41     }
42 
43     @Override
44     public void close() {
45         dispose();
46     }
47 
48     public void dispose() {
49         if (policy != MemoryAddress.NULL) {
50             Index_h.clang_PrintingPolicy_dispose(policy);
51             policy = MemoryAddress.NULL;
52         }
53     }
54 
55     public boolean getProperty(PrintingPolicyProperty prop) {
56         return Index_h.clang_PrintingPolicy_getProperty(policy, prop.value()) != 0;
57     }
58 
59     public void setProperty(PrintingPolicyProperty prop, boolean value) {
60         Index_h.clang_PrintingPolicy_setProperty(policy, prop.value(), value? 1 : 0);
61     }
62 }
