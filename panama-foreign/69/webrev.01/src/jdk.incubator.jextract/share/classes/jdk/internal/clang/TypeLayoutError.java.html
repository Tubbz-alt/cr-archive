27 package jdk.internal.clang;
28 
29 import java.util.HashMap;
30 import java.util.Map;
31 import java.util.NoSuchElementException;
32 
33 public class TypeLayoutError extends IllegalStateException {
34 
35     private static final long serialVersionUID = 0L;
36 
37     private final Kind kind;
38 
39     public TypeLayoutError(long value, String message) {
40         super(Kind.valueOf(value) + ". " + message);
41         this.kind = Kind.valueOf(value);
42     }
43 
44     public Kind kind() {
45         return kind;
46     }
47 
48     public static boolean isError(long value) {
49         return Kind.isError(value);
50     }
51 
52     public enum Kind {
53         Invalid(-1),
54         Incomplete(-2),
55         Dependent(-3),
56         NotConstantSize(-4),
57         InvalidFieldName(-5);
58 
59         private final long value;
60 
61         Kind(long value) {
62             this.value = value;
63         }
64 
65         private final static Map<Long, Kind> lookup;
66 
67         static {
68             lookup = new HashMap<>();
69             for (Kind e: Kind.values()) {
70                 lookup.put(e.value, e);
71             }
72         }
73 
74         public final static Kind valueOf(long value) {
75             Kind x = lookup.get(value);
76             if (null == x) {
77                 throw new NoSuchElementException("TypeLayoutError = " + value);
78             }
79             return x;
80         }
81 
82         public static boolean isError(long value) {
83             return lookup.containsKey(value);
84         }
85     }
86 }
