27 package jdk.internal.clang;
28 
29 import java.util.HashMap;
30 import java.util.Map;
31 import java.util.NoSuchElementException;
32 
33 /**
34  * Represents the C++ access control level of a member or of a base class.
35  */
36 public enum AccessSpecifier {
37     InvalidAccessSpecifier(0),
38     Public(1),
39     Protected(2),
40     Private(3);
41 
42     private final int value;
43 
44     AccessSpecifier(int value) {
45         this.value = value;
46     }
47 
48     public int value() {
49         return value;
50     }
51 
52     private final static Map<Integer, AccessSpecifier> lookup;
53 
54     static {
55         lookup = new HashMap<>();
56         for (AccessSpecifier e: AccessSpecifier.values()) {
57             lookup.put(e.value(), e);
58         }
59     }
60 
61     public final static AccessSpecifier valueOf(int value) {
62         AccessSpecifier x = lookup.get(value);
63         if (null == x) {
64             throw new NoSuchElementException("Invalid AccessSpecifier kind value: " + value);
65         }
66         return x;
67     }
68 }
