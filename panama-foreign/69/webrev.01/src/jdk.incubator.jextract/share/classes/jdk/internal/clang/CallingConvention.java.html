26 package jdk.internal.clang;
27 
28 import java.util.HashMap;
29 import java.util.Map;
30 import java.util.NoSuchElementException;
31 
32 public enum CallingConvention {
33 
34     Default(0),
35     C(1),
36     X86StdCall(2),
37     X86FastCall(3),
38     X86ThisCall(4),
39     X86Pascal(5),
40     AAPCS(6),
41     AAPCS_VFP(7),
42     PnaclCall(8),
43     IntelOclBicc(9),
44     X86_64Win64(10),
45     X86_64SysV(11),
46     Invalid(100),
47     Unexposed(200);
48 
49     private final int value;
50 
51     CallingConvention(int value) {
52         this.value = value;
53     }
54 
55     public int value() {
56         return value;
57     }
58 
59     private final static Map<Integer, CallingConvention> lookup;
60 
61     static {
62         lookup = new HashMap<>();
63         for (CallingConvention e: CallingConvention.values()) {
64             lookup.put(e.value(), e);
65         }
66     }
67 
68     public final static CallingConvention valueOf(int value) {
69         CallingConvention x = lookup.get(value);
70         if (null == x) {
71             throw new NoSuchElementException();
72         }
73         return x;
74     }
75 }
