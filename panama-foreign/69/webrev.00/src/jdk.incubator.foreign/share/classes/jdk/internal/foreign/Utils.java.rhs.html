 64     private static final Unsafe unsafe = Unsafe.getUnsafe();

 66     private static final MethodHandle ADDRESS_FILTER;
 67     private static final MethodHandle LONG_TO_ADDRESS;
 68     private static final MethodHandle ADDRESS_TO_LONG;

 70     static {
 71         try {
 72             ADDRESS_FILTER = MethodHandles.lookup().findStatic(Utils.class, "filterAddress",
 73                     MethodType.methodType(MemoryAddressProxy.class, MemoryAddress.class));
 74             LONG_TO_ADDRESS = MethodHandles.lookup().findStatic(Utils.class, "longToAddress",
 75                     MethodType.methodType(MemoryAddress.class, long.class));
 76             ADDRESS_TO_LONG = MethodHandles.lookup().findStatic(Utils.class, "addressToLong",
 77                     MethodType.methodType(long.class, MemoryAddress.class));
 78         } catch (Throwable ex) {
 79             throw new ExceptionInInitializerError(ex);
 80         }
 81     }

 83     // The maximum alignment supported by malloc - typically 16 on
 84     // 64-bit platforms and 8 on 32-bit platforms.
 85     private final static long MAX_ALIGN = Unsafe.ADDRESS_SIZE == 4 ? 8 : 16;
 86 
 87     // the memory address var handle assumes that addresses have same size as a Java long
 88     private final static long POINTER_SIZE = 8;
 89 
 90     private static final JavaNioAccess javaNioAccess = SharedSecrets.getJavaNioAccess();
 91     private static final JavaLangInvokeAccess javaLangInvokeAccess = SharedSecrets.getJavaLangInvokeAccess();

 93     private static final boolean skipZeroMemory = GetBooleanAction.privilegedGetProperty("jdk.internal.foreign.skipZeroMemory");
 94 
 95     public static long alignUp(long n, long alignment) {
 96         return (n + alignment - 1) & -alignment;
 97     }
 98 
 99     public static long bitsToBytesOrThrow(long bits, Supplier<RuntimeException> exFactory) {
100         if (bits % 8 == 0) {
101             return bits / 8;
102         } else {
103             throw exFactory.get();
104         }
105     }

107     static final Class<?> PADDING_CLASS;
108 
109     static {
110         try {
111             PADDING_CLASS = Class.forName("jdk.incubator.foreign.PaddingLayout");
112         } catch (ReflectiveOperationException ex) {
113             throw new IllegalStateException(ex);
114         }
115     }

117     public static boolean isPadding(MemoryLayout layout) {
118         return layout.getClass() == PADDING_CLASS;

121     public static MemoryAddress resizeNativeAddress(MemoryAddress base, long byteSize) {
122         return new MemoryAddressImpl((MemorySegmentImpl)Utils.makeNativeSegmentUnchecked(base, byteSize), 0);
123     }

125     public static void checkCarrier(Class<?> carrier) {
126         if (carrier == void.class || carrier == boolean.class ||
127                 (!carrier.isPrimitive() && !isAddress(carrier))) {
128             throw new IllegalArgumentException("Illegal carrier: " + carrier.getSimpleName());
129         }
130     }
131 
132     public static long carrierSize(Class<?> carrier) {
133         if (isAddress(carrier)) {
134             return POINTER_SIZE;
135         }
136         long bitsAlignment = Math.max(8, Wrapper.forPrimitiveType(carrier).bitWidth());
137         return Utils.bitsToBytesOrThrow(bitsAlignment, IllegalStateException::new);
138     }
139 
140     public static boolean isAddress(Class<?> carrier) {
141         return MemoryAddress.class == carrier ||
142                 MemoryAddressProxy.class == carrier;
143     }
144 
145     public static Class<?> adjustCarrier(Class<?> carrier) {
146         return carrier == MemoryAddress.class ?
147                 MemoryAddressProxy.class : carrier;
148     }

152     public static MemorySegment makeNativeSegment(long bytesSize, long alignmentBytes) {
153         long alignedSize = bytesSize;
154 
155         if (alignmentBytes > MAX_ALIGN) {
156             alignedSize = bytesSize + (alignmentBytes - 1);
157         }
158 
159         long buf = unsafe.allocateMemory(alignedSize);
160         if (!skipZeroMemory) {
161             unsafe.setMemory(buf, alignedSize, (byte)0);
162         }
163         long alignedBuf = Utils.alignUp(buf, alignmentBytes);
164         MemoryScope scope = new MemoryScope(null, () -> unsafe.freeMemory(buf));
165         MemorySegment segment = new MemorySegmentImpl(buf, null, alignedSize, Thread.currentThread(), scope);
166         if (alignedBuf != buf) {
167             long delta = alignedBuf - buf;
168             segment = segment.asSlice(delta, bytesSize);
169         }
170         return segment;
171     }

173     public static MemorySegment makeNativeSegmentUnchecked(MemoryAddress base, long bytesSize) {
174         if (((MemorySegmentImpl)base.segment()).base != null) {
175             throw new IllegalArgumentException("Not a native address: " + base);
176         }
177         return makeNativeSegmentUnchecked(((MemoryAddressImpl)base).unsafeGetOffset(), bytesSize);
178     }

180     public static MemorySegment makeNativeSegmentUnchecked(long min, long bytesSize) {
181         MemoryScope scope = new MemoryScope(null, null);
182         return new MemorySegmentImpl(min, null, bytesSize, Thread.currentThread(), scope);
183     }

213     private static MemorySegment makeArraySegment(Object arr, int size, int base, int scale) {
214         MemoryScope scope = new MemoryScope(null, null);
215         return new MemorySegmentImpl(base, arr, size * scale, Thread.currentThread(), scope);
216     }
217 
218     public static MemorySegment makeBufferSegment(ByteBuffer bb) {
219         long bbAddress = javaNioAccess.getBufferAddress(bb);
220         Object base = javaNioAccess.getBufferBase(bb);
221 
222         int pos = bb.position();
223         int limit = bb.limit();
224 
225         MemoryScope bufferScope = new MemoryScope(bb, null);
226         return new MemorySegmentImpl(bbAddress + pos, base, limit - pos, Thread.currentThread(), bufferScope);
227     }

229     // create and map a file into a fresh segment
230     public static MemorySegment makeMappedSegment(Path path, long bytesSize, FileChannel.MapMode mapMode) throws IOException {
231         if (bytesSize <= 0) throw new IllegalArgumentException("Requested bytes size must be > 0.");
232         try (FileChannelImpl channelImpl = (FileChannelImpl)FileChannel.open(path, openOptions(mapMode))) {
233             UnmapperProxy unmapperProxy = channelImpl.mapInternal(mapMode, 0L, bytesSize);
234             MemoryScope scope = new MemoryScope(null, unmapperProxy::unmap);
235             return new MemorySegmentImpl(unmapperProxy.address(), null, bytesSize, Thread.currentThread(), scope);
236         }
237     }
238 
239     private static OpenOption[] openOptions(FileChannel.MapMode mapMode) {
240         if (mapMode == FileChannel.MapMode.READ_ONLY) {
241             return new OpenOption[] { StandardOpenOption.READ };
242         } else if (mapMode == FileChannel.MapMode.READ_WRITE || mapMode == FileChannel.MapMode.PRIVATE) {
243             return new OpenOption[] { StandardOpenOption.READ, StandardOpenOption.WRITE };
244         } else {
245             throw new UnsupportedOperationException("Unsupported map mode: " + mapMode);
246         }
247     }

249     public static VarHandle fixUpVarHandle(VarHandle handle) {
250         // This adaptation is required, otherwise the memory access var handle will have type MemoryAddressProxy,
251         // and not MemoryAddress (which the user expects), which causes performance issues with asType() adaptations.
252         handle = MethodHandles.filterCoordinates(handle, 0, ADDRESS_FILTER);
253         return (javaLangInvokeAccess.memoryAddressCarrier(handle) == MemoryAddressProxy.class) ?
254                 MethodHandles.filterValue(handle, ADDRESS_TO_LONG, LONG_TO_ADDRESS) :
255                 handle;
256     }

258     private static MemoryAddressProxy filterAddress(MemoryAddress addr) {
259         return (MemoryAddressImpl)addr;
260     }

262     private static MemoryAddress longToAddress(long value) {
263         return MemoryAddress.ofLong(value);
264     }
265 
266     private static long addressToLong(MemoryAddress value) {
267         return ((MemoryAddressImpl)value).unsafeGetOffset();
268     }
