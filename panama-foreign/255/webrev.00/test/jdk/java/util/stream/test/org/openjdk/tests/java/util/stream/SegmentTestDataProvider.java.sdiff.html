 50     static boolean compareSegmentsByte(Collection<MemorySegment> segments1, Collection<MemorySegment> segments2, boolean isOrdered) {
 51         Function<MemorySegment, Byte> mapper = segment -> (byte)BYTE_HANDLE.get(segment.baseAddress());
 52         List<Byte> list1 = segments1.stream()
 53                 .map(mapper)
 54                 .collect(Collectors.toList());
 55         List<Byte> list2 = segments2.stream()
 56                 .map(mapper)
 57                 .collect(Collectors.toList());
 58         return list1.equals(list2);
 59     }
