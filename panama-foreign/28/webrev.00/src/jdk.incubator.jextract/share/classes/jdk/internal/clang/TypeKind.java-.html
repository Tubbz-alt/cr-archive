 34     Invalid(0),
 35     Unexposed(1),
 36     Void(2),
 37     Bool(3),
 38     Char_U(4),
 39     UChar(5),
 40     Char16(6),
 41     Char32(7),
 42     UShort(8),
 43     UInt(9),
 44     ULong(10),
 45     ULongLong(11),
 46     UInt128(12),
 47     Char_S(13),
 48     SChar(14),
 49     WChar(15),
 50     Short(16),
 51     Int(17),
 52     Long(18),
 53     LongLong(19),
 54     Int128(20),
 55     Float(21),
 56     Double(22),
 57     LongDouble(23),
 58     NullPtr(24),
 59     Overload(25),
 60     Dependent(26),
 61     ObjCId(27),
 62     ObjCClass(28),
 63     ObjCSel(29),
 64     Float128(30),
 65     Half(31),
 66     Complex(100),
 67     Pointer(101),
 68     BlockPointer(102),
 69     LValueReference(103),
 70     RValueReference(104),
 71     Record(105),
 72     Enum(106),
 73     Typedef(107),
 74     ObjCInterface(108),
 75     ObjCObjectPointer(109),
 76     FunctionNoProto(110),
 77     FunctionProto(111),
 78     ConstantArray(112),
 79     Vector(113),
 80     IncompleteArray(114),
 81     VariableArray(115),
 82     DependentSizedArray(116),
 83     MemberPointer(117),
 84     Auto(118),
 85     Elaborated(119),
 86     Atomic(177);
 87 
 88     private final int value;
 89 
 90     TypeKind(int value) {
 91         this.value = value;
 92     }
 93 
 94     public int value() {
 95         return value;
 96     }
 97 
 98     private final static Map<Integer, TypeKind> lookup;
 99 
100     static {
101         lookup = new HashMap<>();
102         for (TypeKind e: TypeKind.values()) {
103             lookup.put(e.value(), e);
104         }
105     }
106 
107     public final static TypeKind valueOf(int value) {
108         TypeKind x = lookup.get(value);
109         if (null == x) {
110             throw new NoSuchElementException("kind = " + value);
111         }
112         return x;
113     }
114 }
