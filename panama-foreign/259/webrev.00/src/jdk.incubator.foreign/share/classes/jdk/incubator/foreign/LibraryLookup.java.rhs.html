 26 package jdk.incubator.foreign;
 27 
 28 import jdk.internal.foreign.LibrariesHelper;
 29 
 30 import java.io.File;
 31 import java.lang.invoke.MethodType;
 32 import java.util.Objects;
 33 
 34 /**
 35  * A native library lookup. Exposes lookup operation for searching symbols, see {@link LibraryLookup#lookup(String)}.
 36  * A given native library remains loaded as long as there is at least one <em>live</em> library lookup instance referring
 37  * to it.
 38  * <p>
 39  * All symbol instances (see {@link LibraryLookup.Symbol}) generated by a given library lookup object contain a strong reference
 40  * to said lookup object, therefore preventing library unloading; in turn method handle instances obtained from
 41  * {@link ForeignLinker#downcallHandle(Addressable, MethodType, FunctionDescriptor)}) also maintain a strong reference
 42  * to the addressable parameter used for their construction. This means that there is always a strong reachability chain
 43  * from a native method handle to a lookup object (the one that was used to lookup the native library symbol the method handle
 44  * refers to); this is useful to prevent situations where a native library is unloaded in the middle of a native call.
 45  * <p>
 46  * To allow for a library to be unloaded, a client will have to discard any strong references it
 47  * maintains, directly, or indirectly to a lookup object associated with given library.
 48  */
 49 public interface LibraryLookup {
 50 
 51     /**
 52      * A symbol retrieved during a library lookup. A lookup symbol has a <em>name</em> and can be projected
 53      * into a memory address (see {@link #name()} and {@link #address()}, respectively).
 54      *
 55      * @apiNote In the future, if the Java language permits, {@link Symbol}
 56      * may become a {@code sealed} interface, which would prohibit subclassing except by
 57      * explicitly permitted types.
 58      *
 59      * @implSpec
 60      * Implementations of this interface are immutable, thread-safe and <a href="{@docRoot}/java.base/java/lang/doc-files/ValueBased.html">value-based</a>.
 61      */
 62     interface Symbol extends Addressable {
 63         /**
 64          * The name of this lookup symbol.
 65          * @return the name of this lookup symbol.
 66          */
 67         String name();
 68 
 69         /**
 70          * The memory address of this lookup symbol.
 71          * @return the memory address of this lookup symbol.
 72          */
 73         @Override
 74         MemoryAddress address();
 75     }
 76 
 77     /**
 78      * Lookups a symbol with given name in this library. The returned symbol maintains a strong reference to this lookup object.

 79      * @param name the symbol name.
 80      * @return the library symbol (if any).
 81      * @throws NoSuchMethodException if no symbol with given name could be found.
 82      */
 83     Symbol lookup(String name) throws NoSuchMethodException;
 84 
 85     /**
 86      * Obtain a default library lookup object.
 87      * @return the default library lookup object.
 88      */
 89     static LibraryLookup ofDefault() {
 90         SecurityManager security = System.getSecurityManager();
 91         if (security != null) {
 92             security.checkPermission(new RuntimePermission("java.foreign.getDefaultLibrary"));
 93         }
 94         return LibrariesHelper.getDefaultLibrary();
 95     }
 96 
 97     /**
 98      * Obtain a library lookup object corresponding to a library identified by given path.
 99      * @param path the library path.
100      * @return a library lookup object for given path.
101      */
102     static LibraryLookup ofPath(String path) {
103         Objects.requireNonNull(path);
104         SecurityManager security = System.getSecurityManager();
105         if (security != null) {
106             security.checkLink(path);
107         }
108         if (!(new File(path).isAbsolute())) {
109             throw new UnsatisfiedLinkError(
110                     "Expecting an absolute path of the library: " + path);
111         }
112         return LibrariesHelper.load(path);
113     }
114 
115     /**
116      * Obtain a library lookup object corresponding to a library identified by given library name.
117      * @param libName the library name.
118      * @return a library lookup object for given library name.
119      */
120     static LibraryLookup ofLibrary(String libName) {
121         Objects.requireNonNull(libName);
122         SecurityManager security = System.getSecurityManager();
123         if (security != null) {
124             security.checkLink(libName);
125         }
126         if (libName.indexOf(File.separatorChar) != -1) {
127             throw new UnsatisfiedLinkError(
128                     "Directory separator should not appear in library name: " + libName);
129         }
130         return LibrariesHelper.loadLibrary(libName);
131     }
132 }
