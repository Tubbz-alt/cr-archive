24 import org.testng.annotations.Test;
25 
26 import java.nio.file.Path;

28 /*
29  * @test
30  * @library .. /test/lib
31  * @modules jdk.incubator.jextract
32  * @build JextractToolRunner
33  * @bug 8249301
34  * @summary jextract fails with CCE when encountering a typedefdecl whose type is FunctionProto
35  * @run testng/othervm -Dforeign.restricted=permit TestTypedefIsFunctionProto
36  */
37 public class TestTypedefIsFunctionProto extends JextractToolRunner {
38     @Test
39     public void testVoidTypedef() {
40         Path outputPath = getOutputFilePath("outputTDIFP");
41         Path headerFile = getInputFilePath("funcproto.h");
42         run("-d", outputPath.toString(), headerFile.toString()).checkSuccess();
43         // nothing is generated that we can check, so we just check that jextract ran successfully
44         deleteDir(outputPath);
45     }
46 }
