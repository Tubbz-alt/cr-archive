109         private Object invoke(Object[] args) throws Throwable {
110             // one trailing Object[]
111             int nNamedArgs = function.argumentLayouts().size();
112             assert(args.length == nNamedArgs + 1);
113             // The last argument is the array of vararg collector
114             Object[] unnamedArgs = (Object[]) args[args.length - 1];
115 
116             int argsCount = nNamedArgs + unnamedArgs.length;
117             Class<?>[] argTypes = new Class<?>[argsCount];
118             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];
119 
120             int pos = 0;
121             for (pos = 0; pos < nNamedArgs; pos++) {
122                 argTypes[pos] = varargs.parameterType(pos);
123                 argLayouts[pos] = function.argumentLayouts().get(pos);
124             }
125 
126             assert pos == nNamedArgs;
127             for (Object o: unnamedArgs) {
128                 argTypes[pos] = normalize(o.getClass());
129                 argLayouts[pos] = variadicLayout(argTypes[pos]);
130                 pos++;
131             }
132             assert pos == argsCount;
133 
134             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);
135             FunctionDescriptor f = (function.returnLayout().isEmpty()) ?
136                     FunctionDescriptor.ofVoid(argLayouts) :
137                     FunctionDescriptor.of(function.returnLayout().get(), argLayouts);
138             MethodHandle mh = ABI.downcallHandle(symbol, mt, f);
139             // flatten argument list so that it can be passed to an asSpreader MH
140             Object[] allArgs = new Object[nNamedArgs + unnamedArgs.length];
141             System.arraycopy(args, 0, allArgs, 0, nNamedArgs);
142             System.arraycopy(unnamedArgs, 0, allArgs, nNamedArgs, unnamedArgs.length);
143 
144             return mh.asSpreader(Object[].class, argsCount).invoke(allArgs);
145         }

147         private static Class<?> unboxIfNeeded(Class<?> clazz) {
148             if (clazz == Boolean.class) {
149                 return boolean.class;
150             } else if (clazz == Void.class) {
151                 return void.class;
152             } else if (clazz == Byte.class) {
153                 return byte.class;
154             } else if (clazz == Character.class) {
155                 return char.class;

158             } else if (clazz == Integer.class) {

162             } else if (clazz == Float.class) {

171         private Class<?> normalize(Class<?> c) {
172             c = unboxIfNeeded(c);
173             if (c.isPrimitive()) {
174                 return c;
175             }
176             if (MemoryAddress.class.isAssignableFrom(c)) {
177                 return MemoryAddress.class;
178             }
179             if (MemorySegment.class.isAssignableFrom(c)) {
180                 return MemorySegment.class;
181             }
182             throw new IllegalArgumentException("Invalid type for ABI: " + c.getTypeName());
183         }

185         private MemoryLayout variadicLayout(Class<?> c) {
186             if (c == char.class || c == byte.class || c == short.class || c == int.class || c == long.class) {
187                 //it is ok to approximate with a machine word here; numerics arguments in a prototype-less
188                 //function call are always rounded up to a register size anyway.
189                 return C_LONGLONG;
190             } else if (c == float.class || c == double.class) {
191                 return C_DOUBLE;
192             } else if (MemoryAddress.class.isAssignableFrom(c)) {
193                 return C_POINTER;
194             } else {
195                 throw new IllegalArgumentException("Unhandled variadic argument class: " + c);
196             }
