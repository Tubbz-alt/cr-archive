 46 /**
 47  * A helper class to generate header interface class in source form.
 48  * After aggregating various constituents of a .java source, build
 49  * method is called to get overall generated source string.
 50  */
 51 class JavaSourceBuilder {
 52     private static final String ABI = SystemABI.getInstance().name();
 53     // buffer
 54     protected StringBuffer sb;
 55     // current line alignment (number of 4-spaces)
 56     protected int align;
 57 
 58     JavaSourceBuilder(int align) {
 59         this.align = align;
 60         this.sb = new StringBuffer();
 61     }
 62 
 63     JavaSourceBuilder() {
 64         this(0);
 65     }
 66 
 67     protected int align() {
 68         return align;
 69     }
 70 
 71     final String PUB_CLS_MODS = "public final ";
 72     final String PUB_MODS = "public static final ";
 73     final String PRI_MODS = "private static final ";
 74 
 75     protected void addPackagePrefix(String pkgName) {
 76         assert pkgName.indexOf('/') == -1 : "package name invalid: " + pkgName;
 77         sb.append("// Generated by jextract\n\n");
 78         if (!pkgName.isEmpty()) {
 79             sb.append("package ");
 80             sb.append(pkgName);
 81             sb.append(";\n\n");
 82         }
 83         addImportSection();
 84     }
 85 
 86     protected void addImportSection() {
 87         sb.append("import java.lang.invoke.MethodHandle;\n");
 88         sb.append("import java.lang.invoke.VarHandle;\n");
 89         sb.append("import jdk.incubator.foreign.*;\n");
 90         sb.append("import jdk.incubator.foreign.MemoryLayout.PathElement;\n");
 91         sb.append("import static ");
 92         sb.append(HandleSourceFactory.C_LANG_CONSTANTS_HOLDER);
 93         sb.append(".*;\n");
 94     }

 96     protected void classBegin(String name) {
 97         indent();
 98         sb.append(PUB_CLS_MODS + "class ");
 99         sb.append(name);
100         sb.append(" {\n\n");
101     }
102 
103     protected void classEnd() {
104         indent();
105         sb.append("}\n\n");
106     }

108     protected void addLibraries(String[] libraryNames) {
109         incrAlign();
110         indent();
111         sb.append(PRI_MODS + "LibraryLookup[] LIBRARIES = RuntimeHelper.libraries(");
112         sb.append(stringArray(libraryNames) + ");\n");
113         decrAlign();
114     }

116     private String stringArray(String[] elements) {
117         return Stream.of(elements)
118                 .map(n -> "\"" + n + "\"")
119                 .collect(Collectors.joining(",", "new String[] {", "}"));
120     }
121 
122     protected void addLayout(String elementName, MemoryLayout layout) {
123         incrAlign();
124         indent();
125         sb.append(PUB_MODS + "MemoryLayout " + elementName + "$LAYOUT = ");
126         addLayout(layout);
127         sb.append(";\n");
128         decrAlign();
129     }
130 
131     private void addLayout(MemoryLayout l) {
132         if (l instanceof ValueLayout) {
133             SystemABI.Type type = l.abiType().orElseThrow(()->new AssertionError("Should not get here: " + l));
134             sb.append(switch (type) {
135                 case BOOL -> "C_BOOL";
136                 case SIGNED_CHAR -> "C_SCHAR";
137                 case UNSIGNED_CHAR -> "C_UCHAR";
138                 case CHAR -> "C_CHAR";
139                 case SHORT -> "C_SHORT";
140                 case UNSIGNED_SHORT -> "C_USHORT";
141                 case INT -> "C_INT";
142                 case UNSIGNED_INT -> "C_UINT";
143                 case LONG -> "C_LONG";
144                 case UNSIGNED_LONG -> "C_ULONG";
145                 case LONG_LONG -> "C_LONGLONG";
146                 case UNSIGNED_LONG_LONG -> "C_ULONGLONG";
147                 case FLOAT -> "C_FLOAT";
148                 case DOUBLE -> "C_DOUBLE";
149                 case LONG_DOUBLE -> "C_LONGDOUBLE";
150                 case POINTER -> "C_POINTER";
151                 default -> { throw new RuntimeException("should not reach here: " + type); }
152             });
153         } else if (l instanceof SequenceLayout) {
154             sb.append("MemoryLayout.ofSequence(");
155             if (((SequenceLayout) l).elementCount().isPresent()) {
156                 sb.append(((SequenceLayout) l).elementCount().getAsLong() + ", ");
157             }
158             addLayout(((SequenceLayout) l).elementLayout());
159             sb.append(")");
160         } else if (l instanceof GroupLayout) {
161             SystemABI.Type type = l.abiType().orElse(null);
162             if (type == SystemABI.Type.COMPLEX_LONG_DOUBLE) {
163                 if (!ABI.equals(SystemABI.ABI_SYSV)) {
164                     throw new RuntimeException("complex long double is supported only for SysV ABI");
165                 } else {
166                     sb.append("C_COMPLEX_LONGDOUBLE");
167                 }
168             } else {
169                 if (((GroupLayout) l).isStruct()) {
170                     sb.append("MemoryLayout.ofStruct(\n");
171                 } else {
172                     sb.append("MemoryLayout.ofUnion(\n");
173                 }
174                 incrAlign();
175                 String delim = "";
176                 for (MemoryLayout e : ((GroupLayout) l).memberLayouts()) {
177                     sb.append(delim);
178                     indent();
179                     addLayout(e);
180                     delim = ",\n";
181                 }
182                 sb.append("\n");
183                 decrAlign();
184                 indent();
185                 sb.append(")");
186             }
187         } else {
188             //padding
189             sb.append("MemoryLayout.ofPaddingBits(" + l.bitSize() + ")");
190         }
191         if (l.name().isPresent()) {
192             sb.append(".withName(\"" +  l.name().get() + "\")");
193         }
194     }

196     protected void addVarHandle(String name, Class<?> type, String parentName) {
197         incrAlign();
198         indent();
199         String vhName = parentName != null ?
200                 parentName + "$" + name :
201                 name;
202         sb.append(PUB_MODS + "VarHandle " + vhName + " = ");
203         if (parentName != null) {
204             addHandlePath(type, parentName, name);
205         } else {
206             addHandlePath(type, name);
207         }
208         sb.append(";\n");
209         decrAlign();
210     }
211 
212     protected void addHandlePath(Class<?> type, String strName, String fieldName) {
213         String ty = type.getName();
214         if (ty.contains("MemoryAddress")) {
215             ty = "long";
216         }
217         sb.append(strName + "$LAYOUT.varHandle(" + ty + ".class, ");
218         sb.append("PathElement.groupElement(\"" + fieldName +"\")");
219         sb.append(")");
220     }
221 
222     protected void addHandlePath(Class<?> type, String varName) {
223         String ty = type.getName();
224         if (ty.contains("MemoryAddress")) {
225             ty = "long";
226         }
227         sb.append(varName + "$LAYOUT.varHandle(" + ty + ".class)");
228     }

230     protected void addMethodHandle(Declaration.Function funcTree, MethodType mtype, FunctionDescriptor desc) {
231         incrAlign();
232         indent();
233         sb.append(PUB_MODS + "MethodHandle " + funcTree.name() + " = ");
234         sb.append("RuntimeHelper.downcallHandle(\n");
235         incrAlign();
236         indent();
237         sb.append("LIBRARIES, \"" + funcTree.name() + "\"");
238         sb.append(",\n");
239         indent();
240         sb.append("\"" + mtype.toMethodDescriptorString() + "\",\n");
241         indent();
242         addFunction(desc);
243         sb.append(",\n");
244         indent();
245         sb.append(funcTree.type().varargs());
246         decrAlign();
247         sb.append("\n");
248         indent();
249         sb.append(");\n");
250         decrAlign();
251     }
252 
253     protected void addAddressLookup(String name) {
254         sb.append("RuntimeHelper.lookupGlobalVariable(LIBRARIES, \"" + name + "\")");
255     }

257     private void addFunction(FunctionDescriptor f) {
258         if (f.returnLayout().isPresent()) {
259             sb.append("FunctionDescriptor.of(");
260             addLayout(f.returnLayout().get());
261             sb.append(", ");
262         } else {
263             sb.append("FunctionDescriptor.ofVoid(");
264         }
265         if (!f.argumentLayouts().isEmpty()) {
266             sb.append("\n");
267             incrAlign();
268             String delim = "";
269             for (MemoryLayout e : f.argumentLayouts()) {
270                 sb.append(delim);
271                 indent();
272                 addLayout(e);
273                 delim = ",\n";
274             }
275             sb.append("\n");
276             decrAlign();
277             indent();
278         }
279         sb.append(")");
280     }

282     protected void addAddress(String name) {
283         incrAlign();
284         indent();
285         sb.append(PUB_MODS + "MemoryAddress " + name + "$ADDR" + " = ");
286         addAddressLookup(name);
287         sb.append(";\n");
288         decrAlign();
289     }

291     protected void addConstant(String name, Class<?> type, Object value) {
292         incrAlign();
293         indent();
294         if (type == MemoryAddress.class || type == MemorySegment.class) {
295             //todo, skip for now (address constants and string constants)
296         } else {
297             sb.append(PUB_MODS + type.getName() + " " + name);
298             sb.append(" = ");
299             if (type == float.class) {
300                 sb.append(value);
301                 sb.append("f");
302             } else if (type == long.class) {
303                 sb.append(value);
304                 sb.append("L");
305             } else if (type == double.class) {
306                 sb.append(value);
307                 sb.append("d");
308             } else {
309                 sb.append("(" + type.getName() + ")");
310                 sb.append(value + "L");
311             }
312             sb.append(";\n");
313         }
314 
315         decrAlign();
316     }

318     static int funcIntfCounter = 0;

320     protected void addUpcallFactory(FunctionDescriptor desc) {
321         String fnName = "FI" + funcIntfCounter++;
322         incrAlign();
323         indent();
324         sb.append(PRI_MODS + "FunctionDescriptor " + fnName + "$DESC = ");
325         addFunction(desc);
326         sb.append(";\n");
327         indent();
328         sb.append(PUB_MODS + "MemoryAddress " + fnName + "$make(MethodHandle handle) {\n");
329         incrAlign();
330         indent();
331         sb.append("return RuntimeHelper.upcallStub(handle, " + fnName + "$DESC);\n");
332         decrAlign();
333         indent();
334         sb.append("}\n");
335         decrAlign();
336     }

338     protected void addStaticFunctionWrapper(Declaration.Function f, MethodType mtype) {
339         incrAlign();
340         indent();
341         sb.append(PUB_MODS + mtype.returnType().getName() + " " + f.name() + " (");
342         String delim = "";
343         List<String> pNames = new ArrayList<>();
344         final int numParams = f.parameters().size();
345         for (int i = 0 ; i < numParams; i++) {
346             String pName = f.parameters().get(i).name();
347             if (pName.isEmpty()) {
348                 pName = "x" + i;
349             }
350             pNames.add(pName);
351             sb.append(delim + mtype.parameterType(i).getName() + " " + pName);
352             delim = ", ";
353         }
354         if (f.type().varargs()) {
355             String lastArg = "x" + numParams;
356             if (numParams > 0) {
357                 sb.append(", ");
358             }
359             sb.append("Object... " + lastArg);
360             pNames.add(lastArg);
361         }
362         sb.append(") {\n");
363         incrAlign();
364         indent();
365         sb.append("try {\n");
366         incrAlign();
367         indent();
368         if (!mtype.returnType().equals(void.class)) {
369             sb.append("return (" + mtype.returnType().getName() + ")");
370         }
371         sb.append(f.name() + ".invokeExact(" + String.join(", ", pNames) + ");\n");
372         decrAlign();
373         indent();
374         sb.append("} catch (Throwable ex) {\n");
375         incrAlign();
376         indent();
377         sb.append("throw new AssertionError(ex);\n");
378         decrAlign();
379         indent();
380         sb.append("}\n");
381         decrAlign();
382         indent();
383         sb.append("}\n");
384         decrAlign();
385     }

387     void addDescriptor(String name, FunctionDescriptor desc) {
388         incrAlign();
389         indent();
390         sb.append(PRI_MODS + "FunctionDescriptor " + name + "$DESC = ");
391         addFunction(desc);
392         sb.append(";\n");
393         decrAlign();
394         indent();
395     }

397     void addFunctionalInterface(String name, MethodType mtype) {
398         incrAlign();
399         indent();
400         sb.append("public interface " + name + " {\n");
401         incrAlign();
402         indent();
403         sb.append(mtype.returnType().getName() + " apply(");
404         String delim = "";
405         for (int i = 0 ; i < mtype.parameterCount() ; i++) {
406             sb.append(delim + mtype.parameterType(i).getName() + " x" + i);
407             delim = ", ";
408         }
409         sb.append(");\n");
410         decrAlign();
411         indent();
412         sb.append("}\n");
413         decrAlign();
414         indent();
415     }

417     protected void addFunctionalFactory(String name, MethodType mtype) {
418         incrAlign();
419         indent();
420         sb.append(PUB_MODS + "MemoryAddress " + name + "$make(" + name + " fi) {\n");
421         incrAlign();
422         indent();
423         sb.append("return RuntimeHelper.upcallStub(" + name + ".class, fi, " + name + "$DESC, " +
424                 "\"" + mtype.toMethodDescriptorString() + "\");\n");
425         decrAlign();
426         indent();
427         sb.append("}\n");
428         decrAlign();

431     void addGetter(String name, Class<?> type, Declaration parent) {
432         incrAlign();
433         indent();
434         String vhName = (parent != null ? (parent.name() + "$") : "") + name;
435         String param = parent != null ? (MemorySegment.class.getName() + " seg") : "";
436         sb.append(PUB_MODS + type.getName() + " " + vhName + "$get(" + param + ") {\n");
437         incrAlign();
438         indent();
439         String vhParam = parent != null ?
440                 "seg.baseAddress()" : name + "$ADDR";
441         sb.append("return (" + type.getName() + ")" + vhName + ".get(" + vhParam + ");\n");
442         decrAlign();
443         indent();
444         sb.append("}\n");
445         decrAlign();
446     }
447 
448     void addSetter(String name, Class<?> type, Declaration parent) {
449         incrAlign();
450         indent();
451         String vhName = (parent != null ? (parent.name() + "$") : "") + name;
452         String param = parent != null ? (MemorySegment.class.getName() + " seg, ") : "";
453         sb.append(PUB_MODS + "void " + vhName + "$set(" + param + type.getName() + " x) {\n");
454         incrAlign();
455         indent();
456         String vhParam = parent != null ?
457                 "seg.baseAddress()" : name + "$ADDR";
458         sb.append(vhName + ".set(" + vhParam + ", x);\n");
459         decrAlign();
460         indent();
461         sb.append("}\n");
462         decrAlign();
463     }
464 
465     protected String build() {
466         String res = sb.toString();
467         this.sb = null;
468         return res.toString();
469     }
470 
471     protected void indent() {
472         for (int i = 0; i < align; i++) {
473             sb.append("    ");
474         }
475     }
476 
477     protected void incrAlign() {
478         align++;
479     }
480 
481     protected void decrAlign() {
482         align--;
483     }
