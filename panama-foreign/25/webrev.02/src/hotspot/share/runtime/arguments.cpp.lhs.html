 118 bool PathString::set_value(const char *value) {
 119   if (_value != NULL) {
 120     FreeHeap(_value);
 121   }
 122   _value = AllocateHeap(strlen(value)+1, mtArguments);
 123   assert(_value != NULL, "Unable to allocate space for new path value");
 124   if (_value != NULL) {
 125     strcpy(_value, value);
 126   } else {
 127     // not able to allocate
 128     return false;
 129   }
 130   return true;
 131 }
 132 
 133 void PathString::append_value(const char *value) {
 134   char *sp;
 135   size_t len = 0;
 136   if (value != NULL) {
 137     len = strlen(value);
 138     if (_value != NULL) {
 139       len += strlen(_value);
 140     }
 141     sp = AllocateHeap(len+2, mtArguments);
 142     assert(sp != NULL, "Unable to allocate space for new append path value");
 143     if (sp != NULL) {
 144       if (_value != NULL) {
 145         strcpy(sp, _value);
 146         strcat(sp, os::path_separator());
 147         strcat(sp, value);
 148         FreeHeap(_value);
 149       } else {
 150         strcpy(sp, value);
 151       }
 152       _value = sp;
 153     }
 154   }
 155 }

 223 // Check if head of 'option' matches 'name', and sets 'tail' to the remaining
 224 // part of the option string.
 225 static bool match_option(const JavaVMOption *option, const char* name,
 226                          const char** tail) {
 227   size_t len = strlen(name);
 228   if (strncmp(option->optionString, name, len) == 0) {
 229     *tail = option->optionString + len;
 230     return true;
 231   } else {
 232     return false;
 233   }
 234 }
 235 
 236 // Check if 'option' matches 'name'. No "tail" is allowed.
 237 static bool match_option(const JavaVMOption *option, const char* name) {
 238   const char* tail = NULL;
 239   bool result = match_option(option, name, &tail);
 240   if (tail != NULL && *tail == '\0') {
 241     return result;
 242   } else {
 243     return false;
 244   }
 245 }
 246 
 247 // Return true if any of the strings in null-terminated array 'names' matches.
 248 // If tail_allowed is true, then the tail must begin with a colon; otherwise,
 249 // the option must match exactly.
 250 static bool match_option(const JavaVMOption* option, const char** names, const char** tail,
 251   bool tail_allowed) {
 252   for (/* empty */; *names != NULL; ++names) {
 253   if (match_option(option, *names, tail)) {
 254       if (**tail == '\0' || (tail_allowed && **tail == ':')) {
 255         return true;
 256       }
 257     }
 258   }
 259   return false;
 260 }

 328 // Return TRUE if option matches 'property', or 'property=', or 'property.'.
 329 static bool matches_property_suffix(const char* option, const char* property, size_t len) {
 330   return ((strncmp(option, property, len) == 0) &&
 331           (option[len] == '=' || option[len] == '.' || option[len] == '\0'));
 332 }

 334 // Return true if property starts with "jdk.module." and its ensuing chars match
 335 // any of the reserved module properties.
 336 // property should be passed without the leading "-D".
 337 bool Arguments::is_internal_module_property(const char* property) {
 338   assert((strncmp(property, "-D", 2) != 0), "Unexpected leading -D");
 339   if  (strncmp(property, MODULE_PROPERTY_PREFIX, MODULE_PROPERTY_PREFIX_LEN) == 0) {
 340     const char* property_suffix = property + MODULE_PROPERTY_PREFIX_LEN;
 341     if (matches_property_suffix(property_suffix, ADDEXPORTS, ADDEXPORTS_LEN) ||
 342         matches_property_suffix(property_suffix, ADDREADS, ADDREADS_LEN) ||
 343         matches_property_suffix(property_suffix, ADDOPENS, ADDOPENS_LEN) ||
 344         matches_property_suffix(property_suffix, PATCH, PATCH_LEN) ||
 345         matches_property_suffix(property_suffix, ADDMODS, ADDMODS_LEN) ||
 346         matches_property_suffix(property_suffix, LIMITMODS, LIMITMODS_LEN) ||
 347         matches_property_suffix(property_suffix, PATH, PATH_LEN) ||
 348         matches_property_suffix(property_suffix, UPGRADE_PATH, UPGRADE_PATH_LEN)) {
 349       return true;
 350     }
 351   }
 352   return false;
 353 }

 355 // Process java launcher properties.
 356 void Arguments::process_sun_java_launcher_properties(JavaVMInitArgs* args) {
 357   // See if sun.java.launcher or sun.java.launcher.is_altjvm is defined.
 358   // Must do this before setting up other system properties,
 359   // as some of them may depend on launcher type.
 360   for (int index = 0; index < args->nOptions; index++) {
 361     const JavaVMOption* option = args->options + index;
 362     const char* tail;
 363 
 364     if (match_option(option, "-Dsun.java.launcher=", &tail)) {
 365       process_java_launcher_argument(tail, option->extraInfo);
 366       continue;
 367     }
 368     if (match_option(option, "-Dsun.java.launcher.is_altjvm=", &tail)) {
 369       if (strcmp(tail, "true") == 0) {
 370         _sun_java_launcher_is_altjvm = true;
 371       }
 372       continue;
 373     }
 374   }
 375 }

 377 // Initialize system properties key and value.
 378 void Arguments::init_system_properties() {
 379 
 380   // Set up _system_boot_class_path which is not a property but
 381   // relies heavily on argument processing and the jdk.boot.class.path.append
 382   // property. It is used to store the underlying system boot class path.
 383   _system_boot_class_path = new PathString(NULL);
 384 
 385   PropertyList_add(&_system_properties, new SystemProperty("java.vm.specification.name",
 386                                                            "Java Virtual Machine Specification",  false));
 387   PropertyList_add(&_system_properties, new SystemProperty("java.vm.version", VM_Version::vm_release(),  false));
 388   PropertyList_add(&_system_properties, new SystemProperty("java.vm.name", VM_Version::vm_name(),  false));
 389   PropertyList_add(&_system_properties, new SystemProperty("jdk.debug", VM_Version::jdk_debug_level(),  false));
 390 
 391   // Initialize the vm.info now, but it will need updating after argument parsing.
 392   _vm_info = new SystemProperty("java.vm.info", VM_Version::vm_info_string(), true);
 393 
 394   // Following are JVMTI agent writable properties.
 395   // Properties values are set to NULL and they are
 396   // os specific they are initialized in os::init_system_properties_values().
 397   _sun_boot_library_path = new SystemProperty("sun.boot.library.path", NULL,  true);
 398   _java_library_path = new SystemProperty("java.library.path", NULL,  true);
 399   _java_home =  new SystemProperty("java.home", NULL,  true);
 400   _java_class_path = new SystemProperty("java.class.path", "",  true);
 401   // jdk.boot.class.path.append is a non-writeable, internal property.
 402   // It can only be set by either:
 403   //    - -Xbootclasspath/a:
 404   //    - AddToBootstrapClassLoaderSearch during JVMTI OnLoad phase
 405   _jdk_boot_class_path_append = new SystemProperty("jdk.boot.class.path.append", "", false, true);
 406 
 407   // Add to System Property list.
 408   PropertyList_add(&_system_properties, _sun_boot_library_path);
 409   PropertyList_add(&_system_properties, _java_library_path);
 410   PropertyList_add(&_system_properties, _java_home);
 411   PropertyList_add(&_system_properties, _java_class_path);
 412   PropertyList_add(&_system_properties, _jdk_boot_class_path_append);
 413   PropertyList_add(&_system_properties, _vm_info);
 414 
 415   // Set OS specific system properties values
 416   os::init_system_properties_values();
 417 }

 436 /*
 437  *  -XX argument processing:
 438  *
 439  *  -XX arguments are defined in several places, such as:
 440  *      globals.hpp, globals_<cpu>.hpp, globals_<os>.hpp, <compiler>_globals.hpp, or <gc>_globals.hpp.
 441  *  -XX arguments are parsed in parse_argument().
 442  *  -XX argument bounds checking is done in check_vm_args_consistency().
 443  *
 444  * Over time -XX arguments may change. There are mechanisms to handle common cases:
 445  *
 446  *      ALIASED: An option that is simply another name for another option. This is often
 447  *               part of the process of deprecating a flag, but not all aliases need
 448  *               to be deprecated.
 449  *
 450  *               Create an alias for an option by adding the old and new option names to the
 451  *               "aliased_jvm_flags" table. Delete the old variable from globals.hpp (etc).
 452  *
 453  *   DEPRECATED: An option that is supported, but a warning is printed to let the user know that
 454  *               support may be removed in the future. Both regular and aliased options may be
 455  *               deprecated.
 456  *
 457  *               Add a deprecation warning for an option (or alias) by adding an entry in the
 458  *               "special_jvm_flags" table and setting the "deprecated_in" field.
 459  *               Often an option "deprecated" in one major release will
 460  *               be made "obsolete" in the next. In this case the entry should also have its
 461  *               "obsolete_in" field set.
 462  *
 463  *     OBSOLETE: An option that has been removed (and deleted from globals.hpp), but is still accepted
 464  *               on the command line. A warning is printed to let the user know that option might not
 465  *               be accepted in the future.
 466  *
 467  *               Add an obsolete warning for an option by adding an entry in the "special_jvm_flags"
 468  *               table and setting the "obsolete_in" field.
 469  *
 470  *      EXPIRED: A deprecated or obsolete option that has an "accept_until" version less than or equal
 471  *               to the current JDK version. The system will flatly refuse to admit the existence of
 472  *               the flag. This allows a flag to die automatically over JDK releases.
 473  *
 474  *               Note that manual cleanup of expired options should be done at major JDK version upgrades:
 475  *                  - Newly expired options should be removed from the special_jvm_flags and aliased_jvm_flags tables.
 476  *                  - Newly obsolete or expired deprecated options should have their global variable
 477  *                    definitions removed (from globals.hpp, etc) and related implementations removed.
 478  *
 479  * Recommended approach for removing options:
 480  *
 481  * To remove options commonly used by customers (e.g. product -XX options), use
 482  * the 3-step model adding major release numbers to the deprecate, obsolete and expire columns.
 483  *
 484  * To remove internal options (e.g. diagnostic, experimental, develop options), use
 485  * a 2-step model adding major release numbers to the obsolete and expire columns.
 486  *
 487  * To change the name of an option, use the alias table as well as a 2-step
 488  * model adding major release numbers to the deprecate and expire columns.
 489  * Think twice about aliasing commonly used customer options.
 490  *
 491  * There are times when it is appropriate to leave a future release number as undefined.
 492  *
 493  * Tests:  Aliases should be tested in VMAliasOptions.java.
 494  *         Deprecated options should be tested in VMDeprecatedOptions.java.
 495  */

 706 // Verifies the correctness of the entries in the special_jvm_flags table.
 707 // If there is a semantic error (i.e. a bug in the table) such as the obsoletion
 708 // version being earlier than the deprecation version, then a warning is issued
 709 // and verification fails - by returning false. If it is detected that the table
 710 // is out of date, with respect to the current version, then ideally a warning is
 711 // issued but verification does not fail. This allows the VM to operate when the
 712 // version is first updated, without needing to update all the impacted flags at
 713 // the same time. In practice we can't issue the warning immediately when the version
 714 // is updated as it occurs for every test and some tests are not prepared to handle
 715 // unexpected output - see 8196739. Instead we only check if the table is up-to-date
 716 // if the check_globals flag is true, and in addition allow a grace period and only
 717 // check for stale flags when we hit build 20 (which is far enough into the 6 month
 718 // release cycle that all flag updates should have been processed, whilst still
 719 // leaving time to make the change before RDP2).
 720 // We use a gtest to call this, passing true, so that we can detect stale flags before
 721 // the end of the release cycle.

 726   bool success = true;
 727   for (size_t i = 0; special_jvm_flags[i].name != NULL; i++) {
 728     const SpecialFlag& flag = special_jvm_flags[i];
 729     if (lookup_special_flag(flag.name, i)) {
 730       warning("Duplicate special flag declaration \"%s\"", flag.name);
 731       success = false;
 732     }
 733     if (flag.deprecated_in.is_undefined() &&
 734         flag.obsolete_in.is_undefined()) {
 735       warning("Special flag entry \"%s\" must declare version deprecated and/or obsoleted in.", flag.name);
 736       success = false;
 737     }
 738 
 739     if (!flag.deprecated_in.is_undefined()) {
 740       if (!version_less_than(flag.deprecated_in, flag.obsolete_in)) {
 741         warning("Special flag entry \"%s\" must be deprecated before obsoleted.", flag.name);
 742         success = false;
 743       }
 744 
 745       if (!version_less_than(flag.deprecated_in, flag.expired_in)) {
 746         warning("Special flag entry \"%s\" must be deprecated before expired.", flag.name);
 747         success = false;
 748       }
 749     }
 750 
 751     if (!flag.obsolete_in.is_undefined()) {
 752       if (!version_less_than(flag.obsolete_in, flag.expired_in)) {
 753         warning("Special flag entry \"%s\" must be obsoleted before expired.", flag.name);
 754         success = false;
 755       }
 756 
 757       // if flag has become obsolete it should not have a "globals" flag defined anymore.
 758       if (check_globals && VM_Version::vm_build_number() >= SPECIAL_FLAG_VALIDATION_BUILD &&
 759           !version_less_than(JDK_Version::current(), flag.obsolete_in)) {
 760         if (JVMFlag::find_declared_flag(flag.name) != NULL) {
 761           warning("Global variable for obsolete special flag entry \"%s\" should be removed", flag.name);
 762           success = false;
 763         }
 764       }
 765 
 766     } else if (!flag.expired_in.is_undefined()) {
 767       warning("Special flag entry \"%s\" must be explicitly obsoleted before expired.", flag.name);
 768       success = false;
 769     }
 770 
 771     if (!flag.expired_in.is_undefined()) {
 772       // if flag has become expired it should not have a "globals" flag defined anymore.
 773       if (check_globals && VM_Version::vm_build_number() >= SPECIAL_FLAG_VALIDATION_BUILD &&
 774           !version_less_than(JDK_Version::current(), flag.expired_in)) {
 775         if (JVMFlag::find_declared_flag(flag.name) != NULL) {
 776           warning("Global variable for expired flag entry \"%s\" should be removed", flag.name);
 777           success = false;
 778         }
 779       }
 780     }
 781   }
 782   return success;
 783 }

 786 // Parses a size specification string.
 787 bool Arguments::atojulong(const char *s, julong* result) {
 788   julong n = 0;
 789 
 790   // First char must be a digit. Don't allow negative numbers or leading spaces.
 791   if (!isdigit(*s)) {
 792     return false;
 793   }
 794 
 795   bool is_hex = (s[0] == '0' && (s[1] == 'x' || s[1] == 'X'));
 796   char* remainder;
 797   errno = 0;
 798   n = strtoull(s, &remainder, (is_hex ? 16 : 10));
 799   if (errno != 0) {
 800     return false;
 801   }
 802 
 803   // Fail if no number was read at all or if the remainder contains more than a single non-digit character.
 804   if (remainder == s || strlen(remainder) > 1) {
 805     return false;
 806   }
 807 
 808   switch (*remainder) {
 809     case 'T': case 't':
 810       *result = n * G * K;
 811       // Check for overflow.
 812       if (*result/((julong)G * K) != n) return false;
 813       return true;
 814     case 'G': case 'g':
 815       *result = n * G;
 816       if (*result/G != n) return false;
 817       return true;
 818     case 'M': case 'm':
 819       *result = n * M;
 820       if (*result/M != n) return false;
 821       return true;
 822     case 'K': case 'k':
 823       *result = n * K;
 824       if (*result/K != n) return false;
 825       return true;
 826     case '\0':
 827       *result = n;
 828       return true;
 829     default:
 830       return false;
 831   }
 832 }

 948     value = new_value;
 949   } else if (new_len == 0) {

 952      size_t length = old_len + 1 + new_len + 1;

 961   FREE_C_HEAP_ARRAY(char, value);
 962   // JVMFlag made its own copy, so I must delete my own temp. buffer.
 963   FREE_C_HEAP_ARRAY(char, free_this_too);
 964   return true;

1254 bool Arguments::process_argument(const char* arg,
1255                                  jboolean ignore_unrecognized,
1256                                  JVMFlag::Flags origin) {
1257   JDK_Version since = JDK_Version();
1258 
1259   if (parse_argument(arg, origin)) {
1260     return true;
1261   }
1262 
1263   // Determine if the flag has '+', '-', or '=' characters.
1264   bool has_plus_minus = (*arg == '+' || *arg == '-');
1265   const char* const argname = has_plus_minus ? arg + 1 : arg;
1266 
1267   size_t arg_len;
1268   const char* equal_sign = strchr(argname, '=');
1269   if (equal_sign == NULL) {
1270     arg_len = strlen(argname);
1271   } else {
1272     arg_len = equal_sign - argname;
1273   }
1274 
1275   // Only make the obsolete check for valid arguments.
1276   if (arg_len <= BUFLEN) {
1277     // Construct a string which consists only of the argument name without '+', '-', or '='.
1278     char stripped_argname[BUFLEN+1]; // +1 for '\0'
1279     jio_snprintf(stripped_argname, arg_len+1, "%s", argname); // +1 for '\0'
1280     if (is_obsolete_flag(stripped_argname, &since)) {
1281       char version[256];
1282       since.to_string(version, sizeof(version));
1283       warning("Ignoring option %s; support was removed in %s", stripped_argname, version);
1284       return true;
1285     }
1286 #ifndef PRODUCT
1287     else {
1288       const char* replacement;
1289       if ((replacement = removed_develop_logging_flag_name(stripped_argname)) != NULL){
1290         log_warning(arguments)("%s has been removed. Please use %s instead.",
1291                                stripped_argname,
1292                                replacement);
1293         return false;
1294       }
1295     }
1296 #endif //PRODUCT
1297   }
1298 
1299   // For locked flags, report a custom error message if available.
1300   // Otherwise, report the standard unrecognized VM option.
1301   const JVMFlag* found_flag = JVMFlag::find_declared_flag((const char*)argname, arg_len);
1302   if (found_flag != NULL) {
1303     char locked_message_buf[BUFLEN];
1304     JVMFlag::MsgType msg_type = found_flag->get_locked_message(locked_message_buf, BUFLEN);
1305     if (strlen(locked_message_buf) == 0) {
1306       if (found_flag->is_bool() && !has_plus_minus) {
1307         jio_fprintf(defaultStream::error_stream(),
1308           "Missing +/- setting for VM option '%s'\n", argname);
1309       } else if (!found_flag->is_bool() && has_plus_minus) {
1310         jio_fprintf(defaultStream::error_stream(),
1311           "Unexpected +/- setting in VM option '%s'\n", argname);
1312       } else {
1313         jio_fprintf(defaultStream::error_stream(),
1314           "Improperly specified VM option '%s'\n", argname);
1315       }
1316     } else {
1317 #ifdef PRODUCT
1318       bool mismatched = ((msg_type == JVMFlag::NOTPRODUCT_FLAG_BUT_PRODUCT_BUILD) ||
1319                          (msg_type == JVMFlag::DEVELOPER_FLAG_BUT_PRODUCT_BUILD));
1320       if (ignore_unrecognized && mismatched) {
1321         return true;
1322       }
1323 #endif
1324       jio_fprintf(defaultStream::error_stream(), "%s", locked_message_buf);
1325     }
1326   } else {
1327     if (ignore_unrecognized) {
1328       return true;
1329     }
1330     jio_fprintf(defaultStream::error_stream(),
1331                 "Unrecognized VM option '%s'\n", argname);
1332     JVMFlag* fuzzy_matched = JVMFlag::fuzzy_match((const char*)argname, arg_len, true);
1333     if (fuzzy_matched != NULL) {
1334       jio_fprintf(defaultStream::error_stream(),
1335                   "Did you mean '%s%s%s'? ",
1336                   (fuzzy_matched->is_bool()) ? "(+/-)" : "",
1337                   fuzzy_matched->_name,
1338                   (fuzzy_matched->is_bool()) ? "" : "=<value>");
1339     }
1340   }
1341 
1342   // allow for commandline "commenting out" options like -XX:#+Verbose
1343   return arg[0] == '#';
1344 }

1346 bool Arguments::process_settings_file(const char* file_name, bool should_exist, jboolean ignore_unrecognized) {
1347   FILE* stream = fopen(file_name, "rb");
1348   if (stream == NULL) {
1349     if (should_exist) {
1350       jio_fprintf(defaultStream::error_stream(),
1351                   "Could not open settings file %s\n", file_name);
1352       return false;
1353     } else {
1354       return true;
1355     }
1356   }
1357 
1358   char token[1024];
1359   int  pos = 0;
1360 
1361   bool in_white_space = true;
1362   bool in_comment     = false;
1363   bool in_quote       = false;
1364   char quote_c        = 0;
1365   bool result         = true;
1366 
1367   int c = getc(stream);
1368   while(c != EOF && pos < (int)(sizeof(token)-1)) {
1369     if (in_white_space) {
1370       if (in_comment) {
1371         if (c == '\n') in_comment = false;
1372       } else {
1373         if (c == '#') in_comment = true;
1374         else if (!isspace(c)) {
1375           in_white_space = false;
1376           token[pos++] = c;
1377         }
1378       }
1379     } else {
1380       if (c == '\n' || (!in_quote && isspace(c))) {
1381         // token ends at newline, or at unquoted whitespace
1382         // this allows a way to include spaces in string-valued options
1383         token[pos] = '\0';
1384         logOption(token);
1385         result &= process_argument(token, ignore_unrecognized, JVMFlag::CONFIG_FILE);
1386         build_jvm_flags(token);
1387         pos = 0;
1388         in_white_space = true;
1389         in_quote = false;
1390       } else if (!in_quote && (c == '\'' || c == '"')) {
1391         in_quote = true;
1392         quote_c = c;
1393       } else if (in_quote && (c == quote_c)) {
1394         in_quote = false;
1395       } else {
1396         token[pos++] = c;
1397       }
1398     }
1399     c = getc(stream);
1400   }
1401   if (pos > 0) {
1402     token[pos] = '\0';
1403     result &= process_argument(token, ignore_unrecognized, JVMFlag::CONFIG_FILE);
1404     build_jvm_flags(token);
1405   }
1406   fclose(stream);
1407   return result;
1408 }

1417 bool Arguments::add_property(const char* prop, PropertyWriteable writeable, PropertyInternal internal) {
1418   const char* eq = strchr(prop, '=');
1419   const char* key;
1420   const char* value = "";
1421 
1422   if (eq == NULL) {
1423     // property doesn't have a value, thus use passed string
1424     key = prop;
1425   } else {
1426     // property have a value, thus extract it and save to the
1427     // allocated string
1428     size_t key_len = eq - prop;
1429     char* tmp_key = AllocateHeap(key_len + 1, mtArguments);
1430 
1431     jio_snprintf(tmp_key, key_len + 1, "%s", prop);
1432     key = tmp_key;
1433 
1434     value = &prop[key_len + 1];
1435   }
1436 
1437   if (strcmp(key, "java.compiler") == 0) {
1438     process_java_compiler_argument(value);
1439     // Record value in Arguments, but let it get passed to Java.
1440   } else if (strcmp(key, "sun.java.launcher.is_altjvm") == 0) {
1441     // sun.java.launcher.is_altjvm property is
1442     // private and is processed in process_sun_java_launcher_properties();
1443     // the sun.java.launcher property is passed on to the java application
1444   } else if (strcmp(key, "sun.boot.library.path") == 0) {
1445     // append is true, writable is true, internal is false
1446     PropertyList_unique_add(&_system_properties, key, value, AppendProperty,
1447                             WriteableProperty, ExternalProperty);
1448   } else {
1449     if (strcmp(key, "sun.java.command") == 0) {
1450       char *old_java_command = _java_command;
1451       _java_command = os::strdup_check_oom(value, mtArguments);
1452       if (old_java_command != NULL) {
1453         os::free(old_java_command);
1454       }
1455     } else if (strcmp(key, "java.vendor.url.bug") == 0) {
1456       // If this property is set on the command line then its value will be
1457       // displayed in VM error logs as the URL at which to submit such logs.
1458       // Normally the URL displayed in error logs is different from the value
1459       // of this system property, so a different property should have been
1460       // used here, but we leave this as-is in case someone depends upon it.
1461       const char* old_java_vendor_url_bug = _java_vendor_url_bug;
1462       // save it in _java_vendor_url_bug, so JVM fatal error handler can access
1463       // its value without going through the property list or making a Java call.

1465       if (old_java_vendor_url_bug != NULL) {
1466         os::free((void *)old_java_vendor_url_bug);
1467       }
1468     }
1469 
1470     // Create new property and add at the end of the list
1471     PropertyList_unique_add(&_system_properties, key, value, AddProperty, writeable, internal);
1472   }
1473 
1474   if (key != prop) {
1475     // SystemProperty copy passed value, thus free previously allocated
1476     // memory
1477     FreeHeap((void *)key);
1478   }
1479

1631 size_t Arguments::max_heap_for_compressed_oops() {
1632   // Avoid sign flip.
1633   assert(OopEncodingHeapMax > (uint64_t)os::vm_page_size(), "Unusual page size");

1635   // keeping alignment constraints of the heap. To guarantee the latter, as the
1636   // NULL page is located before the heap, we pad the NULL page to the conservative
1637   // maximum alignment that the GC may ever impose upon the heap.

1639                                                   _conservative_max_heap_alignment);
1640 
1641   LP64_ONLY(return OopEncodingHeapMax - displacement_due_to_null_page);
1642   NOT_LP64(ShouldNotReachHere(); return 0);
1643 }

1739 static const size_t DefaultHeapBaseMinAddress = HeapBaseMinAddress;
1740 
1741 void Arguments::set_heap_size() {
1742   julong phys_mem;
1743 
1744   // If the user specified one of these options, they
1745   // want specific memory sizing so do not limit memory
1746   // based on compressed oops addressability.
1747   // Also, memory limits will be calculated based on
1748   // available os physical memory, not our MaxRAM limit,
1749   // unless MaxRAM is also specified.
1750   bool override_coop_limit = (!FLAG_IS_DEFAULT(MaxRAMPercentage) ||
1751                            !FLAG_IS_DEFAULT(MaxRAMFraction) ||
1752                            !FLAG_IS_DEFAULT(MinRAMPercentage) ||
1753                            !FLAG_IS_DEFAULT(MinRAMFraction) ||
1754                            !FLAG_IS_DEFAULT(InitialRAMPercentage) ||
1755                            !FLAG_IS_DEFAULT(InitialRAMFraction) ||
1756                            !FLAG_IS_DEFAULT(MaxRAM));
1757   if (override_coop_limit) {
1758     if (FLAG_IS_DEFAULT(MaxRAM)) {
1759       phys_mem = os::physical_memory();
1760       FLAG_SET_ERGO(MaxRAM, (uint64_t)phys_mem);
1761     } else {
1762       phys_mem = (julong)MaxRAM;
1763     }
1764   } else {
1765     phys_mem = FLAG_IS_DEFAULT(MaxRAM) ? MIN2(os::physical_memory(), (julong)MaxRAM)
1766                                        : (julong)MaxRAM;
1767   }
1768 
1769 
1770   // Convert deprecated flags
1771   if (FLAG_IS_DEFAULT(MaxRAMPercentage) &&
1772       !FLAG_IS_DEFAULT(MaxRAMFraction))
1773     MaxRAMPercentage = 100.0 / MaxRAMFraction;
1774 
1775   if (FLAG_IS_DEFAULT(MinRAMPercentage) &&
1776       !FLAG_IS_DEFAULT(MinRAMFraction))
1777     MinRAMPercentage = 100.0 / MinRAMFraction;
1778 
1779   if (FLAG_IS_DEFAULT(InitialRAMPercentage) &&
1780       !FLAG_IS_DEFAULT(InitialRAMFraction))
1781     InitialRAMPercentage = 100.0 / InitialRAMFraction;
1782 
1783   // If the maximum heap size has not been set with -Xmx,
1784   // then set it as fraction of the size of physical memory,
1785   // respecting the maximum and minimum sizes of the heap.
1786   if (FLAG_IS_DEFAULT(MaxHeapSize)) {
1787     julong reasonable_max = (julong)((phys_mem * MaxRAMPercentage) / 100);
1788     const julong reasonable_min = (julong)((phys_mem * MinRAMPercentage) / 100);
1789     if (reasonable_min < MaxHeapSize) {
1790       // Small physical memory, so use a minimum fraction of it for the heap
1791       reasonable_max = reasonable_min;
1792     } else {
1793       // Not-small physical memory, so require a heap at least
1794       // as large as MaxHeapSize
1795       reasonable_max = MAX2(reasonable_max, (julong)MaxHeapSize);
1796     }
1797 
1798     if (!FLAG_IS_DEFAULT(ErgoHeapSizeLimit) && ErgoHeapSizeLimit != 0) {
1799       // Limit the heap size to ErgoHeapSizeLimit
1800       reasonable_max = MIN2(reasonable_max, (julong)ErgoHeapSizeLimit);
1801     }
1802 
1803 #ifdef _LP64
1804     if (UseCompressedOops) {
1805       // Limit the heap size to the maximum possible when using compressed oops
1806       julong max_coop_heap = (julong)max_heap_for_compressed_oops();
1807 
1808       // HeapBaseMinAddress can be greater than default but not less than.
1809       if (!FLAG_IS_DEFAULT(HeapBaseMinAddress)) {
1810         if (HeapBaseMinAddress < DefaultHeapBaseMinAddress) {
1811           // matches compressed oops printing flags
1812           log_debug(gc, heap, coops)("HeapBaseMinAddress must be at least " SIZE_FORMAT
1813                                      " (" SIZE_FORMAT "G) which is greater than value given " SIZE_FORMAT,
1814                                      DefaultHeapBaseMinAddress,
1815                                      DefaultHeapBaseMinAddress/G,
1816                                      HeapBaseMinAddress);
1817           FLAG_SET_ERGO(HeapBaseMinAddress, DefaultHeapBaseMinAddress);
1818         }
1819       }
1820 
1821       if (HeapBaseMinAddress + MaxHeapSize < max_coop_heap) {
1822         // Heap should be above HeapBaseMinAddress to get zero based compressed oops
1823         // but it should be not less than default MaxHeapSize.
1824         max_coop_heap -= HeapBaseMinAddress;
1825       }
1826 
1827       // If user specified flags prioritizing os physical
1828       // memory limits, then disable compressed oops if
1829       // limits exceed max_coop_heap and UseCompressedOops
1830       // was not specified.
1831       if (reasonable_max > max_coop_heap) {
1832         if (FLAG_IS_ERGO(UseCompressedOops) && override_coop_limit) {
1833           log_info(cds)("UseCompressedOops and UseCompressedClassPointers have been disabled due to"
1834             " max heap " SIZE_FORMAT " > compressed oop heap " SIZE_FORMAT ". "
1835             "Please check the setting of MaxRAMPercentage %5.2f."
1836             ,(size_t)reasonable_max, (size_t)max_coop_heap, MaxRAMPercentage);
1837           FLAG_SET_ERGO(UseCompressedOops, false);
1838           FLAG_SET_ERGO(UseCompressedClassPointers, false);
1839         } else {
1840           reasonable_max = MIN2(reasonable_max, max_coop_heap);
1841         }
1842       }
1843     }
1844 #endif // _LP64
1845 
1846     reasonable_max = limit_by_allocatable_memory(reasonable_max);
1847 
1848     if (!FLAG_IS_DEFAULT(InitialHeapSize)) {
1849       // An initial heap size was specified on the command line,
1850       // so be sure that the maximum size is consistent.  Done
1851       // after call to limit_by_allocatable_memory because that
1852       // method might reduce the allocation size.
1853       reasonable_max = MAX2(reasonable_max, (julong)InitialHeapSize);
1854     } else if (!FLAG_IS_DEFAULT(MinHeapSize)) {
1855       reasonable_max = MAX2(reasonable_max, (julong)MinHeapSize);
1856     }
1857 
1858     log_trace(gc, heap)("  Maximum heap size " SIZE_FORMAT, (size_t) reasonable_max);
1859     FLAG_SET_ERGO(MaxHeapSize, (size_t)reasonable_max);
1860   }
1861 
1862   // If the minimum or initial heap_size have not been set or requested to be set
1863   // ergonomically, set them accordingly.
1864   if (InitialHeapSize == 0 || MinHeapSize == 0) {
1865     julong reasonable_minimum = (julong)(OldSize + NewSize);
1866 
1867     reasonable_minimum = MIN2(reasonable_minimum, (julong)MaxHeapSize);
1868 
1869     reasonable_minimum = limit_by_allocatable_memory(reasonable_minimum);
1870 
1871     if (InitialHeapSize == 0) {
1872       julong reasonable_initial = (julong)((phys_mem * InitialRAMPercentage) / 100);
1873 
1874       reasonable_initial = MAX3(reasonable_initial, reasonable_minimum, (julong)MinHeapSize);
1875       reasonable_initial = MIN2(reasonable_initial, (julong)MaxHeapSize);
1876 
1877       reasonable_initial = limit_by_allocatable_memory(reasonable_initial);
1878 
1879       FLAG_SET_ERGO(InitialHeapSize, (size_t)reasonable_initial);
1880       log_trace(gc, heap)("  Initial heap size " SIZE_FORMAT, InitialHeapSize);
1881     }
1882     // If the minimum heap size has not been set (via -Xms or -XX:MinHeapSize),
1883     // synchronize with InitialHeapSize to avoid errors with the default value.
1884     if (MinHeapSize == 0) {
1885       FLAG_SET_ERGO(MinHeapSize, MIN2((size_t)reasonable_minimum, InitialHeapSize));
1886       log_trace(gc, heap)("  Minimum heap size " SIZE_FORMAT, MinHeapSize);
1887     }
1888   }
1889 }

1891 // This option inspects the machine and attempts to set various
1892 // parameters to be optimal for long-running, memory allocation
1893 // intensive jobs.  It is intended for machines with large
1894 // amounts of cpu and memory.
1895 jint Arguments::set_aggressive_heap_flags() {
1896   // initHeapSize is needed since _initial_heap_size is 4 bytes on a 32 bit
1897   // VM, but we may not be able to represent the total physical memory
1898   // available (like having 8gb of memory on a box but using a 32bit VM).
1899   // Thus, we need to make sure we're using a julong for intermediate
1900   // calculations.
1901   julong initHeapSize;
1902   julong total_memory = os::physical_memory();
1903 
1904   if (total_memory < (julong) 256 * M) {
1905     jio_fprintf(defaultStream::error_stream(),
1906             "You need at least 256mb of memory to use -XX:+AggressiveHeap\n");
1907     vm_exit(1);
1908   }
1909 
1910   // The heap size is half of available memory, or (at most)
1911   // all of possible memory less 160mb (leaving room for the OS
1912   // when using ISM).  This is the maximum; because adaptive sizing
1913   // is turned on below, the actual space used may be smaller.
1914 
1915   initHeapSize = MIN2(total_memory / (julong) 2,
1916           total_memory - (julong) 160 * M);
1917 
1918   initHeapSize = limit_by_allocatable_memory(initHeapSize);
1919 
1920   if (FLAG_IS_DEFAULT(MaxHeapSize)) {
1921     if (FLAG_SET_CMDLINE(MaxHeapSize, initHeapSize) != JVMFlag::SUCCESS) {
1922       return JNI_EINVAL;
1923     }
1924     if (FLAG_SET_CMDLINE(InitialHeapSize, initHeapSize) != JVMFlag::SUCCESS) {
1925       return JNI_EINVAL;
1926     }
1927     if (FLAG_SET_CMDLINE(MinHeapSize, initHeapSize) != JVMFlag::SUCCESS) {
1928       return JNI_EINVAL;
1929     }
1930   }
1931   if (FLAG_IS_DEFAULT(NewSize)) {
1932     // Make the young generation 3/8ths of the total heap.
1933     if (FLAG_SET_CMDLINE(NewSize,
1934             ((julong) MaxHeapSize / (julong) 8) * (julong) 3) != JVMFlag::SUCCESS) {
1935       return JNI_EINVAL;
1936     }
1937     if (FLAG_SET_CMDLINE(MaxNewSize, NewSize) != JVMFlag::SUCCESS) {
1938       return JNI_EINVAL;
1939     }
1940   }
1941 
1942 #if !defined(_ALLBSD_SOURCE) && !defined(AIX)  // UseLargePages is not yet supported on BSD and AIX.
1943   FLAG_SET_DEFAULT(UseLargePages, true);
1944 #endif
1945 
1946   // Increase some data structure sizes for efficiency
1947   if (FLAG_SET_CMDLINE(BaseFootPrintEstimate, MaxHeapSize) != JVMFlag::SUCCESS) {
1948     return JNI_EINVAL;
1949   }
1950   if (FLAG_SET_CMDLINE(ResizeTLAB, false) != JVMFlag::SUCCESS) {
1951     return JNI_EINVAL;
1952   }
1953   if (FLAG_SET_CMDLINE(TLABSize, 256 * K) != JVMFlag::SUCCESS) {
1954     return JNI_EINVAL;
1955   }
1956 
1957   // See the OldPLABSize comment below, but replace 'after promotion'
1958   // with 'after copying'.  YoungPLABSize is the size of the survivor
1959   // space per-gc-thread buffers.  The default is 4kw.
1960   if (FLAG_SET_CMDLINE(YoungPLABSize, 256 * K) != JVMFlag::SUCCESS) { // Note: this is in words
1961     return JNI_EINVAL;
1962   }
1963 
1964   // OldPLABSize is the size of the buffers in the old gen that
1965   // UseParallelGC uses to promote live data that doesn't fit in the
1966   // survivor spaces.  At any given time, there's one for each gc thread.
1967   // The default size is 1kw. These buffers are rarely used, since the
1968   // survivor spaces are usually big enough.  For specjbb, however, there
1969   // are occasions when there's lots of live data in the young gen
1970   // and we end up promoting some of it.  We don't have a definite
1971   // explanation for why bumping OldPLABSize helps, but the theory
1972   // is that a bigger PLAB results in retaining something like the
1973   // original allocation order after promotion, which improves mutator
1974   // locality.  A minor effect may be that larger PLABs reduce the
1975   // number of PLAB allocation events during gc.  The value of 8kw
1976   // was arrived at by experimenting with specjbb.
1977   if (FLAG_SET_CMDLINE(OldPLABSize, 8 * K) != JVMFlag::SUCCESS) { // Note: this is in words
1978     return JNI_EINVAL;
1979   }
1980 
1981   // Enable parallel GC and adaptive generation sizing
1982   if (FLAG_SET_CMDLINE(UseParallelGC, true) != JVMFlag::SUCCESS) {
1983     return JNI_EINVAL;
1984   }
1985 
1986   // Encourage steady state memory management
1987   if (FLAG_SET_CMDLINE(ThresholdTolerance, 100) != JVMFlag::SUCCESS) {
1988     return JNI_EINVAL;
1989   }
1990 
1991   // This appears to improve mutator locality
1992   if (FLAG_SET_CMDLINE(ScavengeBeforeFullGC, false) != JVMFlag::SUCCESS) {
1993     return JNI_EINVAL;
1994   }
1995 
1996   return JNI_OK;
1997 }

2354 // Parse -Xss memory string parameter and convert to ThreadStackSize in K.
2355 jint Arguments::parse_xss(const JavaVMOption* option, const char* tail, intx* out_ThreadStackSize) {
2356   // The min and max sizes match the values in globals.hpp, but scaled
2357   // with K. The values have been chosen so that alignment with page
2358   // size doesn't change the max value, which makes the conversions
2359   // back and forth between Xss value and ThreadStackSize value easier.
2360   // The values have also been chosen to fit inside a 32-bit signed type.
2361   const julong min_ThreadStackSize = 0;
2362   const julong max_ThreadStackSize = 1 * M;
2363 
2364   const julong min_size = min_ThreadStackSize * K;
2365   const julong max_size = max_ThreadStackSize * K;
2366 
2367   assert(is_aligned(max_size, os::vm_page_size()), "Implementation assumption");
2368 
2369   julong size = 0;
2370   ArgsRange errcode = parse_memory_size(tail, &size, min_size, max_size);
2371   if (errcode != arg_in_range) {
2372     bool silent = (option == NULL); // Allow testing to silence error messages
2373     if (!silent) {
2374       jio_fprintf(defaultStream::error_stream(),
2375                   "Invalid thread stack size: %s\n", option->optionString);
2376       describe_range_error(errcode);
2377     }
2378     return JNI_EINVAL;
2379   }
2380 
2381   // Internally track ThreadStackSize in units of 1024 bytes.
2382   const julong size_aligned = align_up(size, K);
2383   assert(size <= size_aligned,
2384          "Overflow: " JULONG_FORMAT " " JULONG_FORMAT,
2385          size, size_aligned);
2386 
2387   const julong size_in_K = size_aligned / K;
2388   assert(size_in_K < (julong)max_intx,
2389          "size_in_K doesn't fit in the type of ThreadStackSize: " JULONG_FORMAT,
2390          size_in_K);
2391 
2392   // Check that code expanding ThreadStackSize to a page aligned number of bytes won't overflow.
2393   const julong max_expanded = align_up(size_in_K * K, os::vm_page_size());
2394   assert(max_expanded < max_uintx && max_expanded >= size_in_K,
2395          "Expansion overflowed: " JULONG_FORMAT " " JULONG_FORMAT,
2396          max_expanded, size_in_K);
2397 
2398   *out_ThreadStackSize = (intx)size_in_K;
2399 
2400   return JNI_OK;
2401 }

3065 // Remove all empty paths from the app classpath (if IgnoreEmptyClassPaths is enabled)
3066 //
3067 // This is necessary because some apps like to specify classpath like -cp foo.jar:${XYZ}:bar.jar
3068 // in their start-up scripts. If XYZ is empty, the classpath will look like "-cp foo.jar::bar.jar".
3069 // Java treats such empty paths as if the user specified "-cp foo.jar:.:bar.jar". I.e., an empty
3070 // path is treated as the current directory.
3071 //
3072 // This causes problems with CDS, which requires that all directories specified in the classpath
3073 // must be empty. In most cases, applications do NOT want to load classes from the current
3074 // directory anyway. Adding -XX:+IgnoreEmptyClassPaths will make these applications' start-up
3075 // scripts compatible with CDS.
3076 void Arguments::fix_appclasspath() {
3077   if (IgnoreEmptyClassPaths) {
3078     const char separator = *os::path_separator();
3079     const char* src = _java_class_path->value();
3080 
3081     // skip over all the leading empty paths
3082     while (*src == separator) {
3083       src ++;
3084     }
3085 
3086     char* copy = os::strdup_check_oom(src, mtArguments);
3087 
3088     // trim all trailing empty paths
3089     for (char* tail = copy + strlen(copy) - 1; tail >= copy && *tail == separator; tail--) {
3090       *tail = '\0';
3091     }
3092 
3093     char from[3] = {separator, separator, '\0'};
3094     char to  [2] = {separator, '\0'};
3095     while (StringUtils::replace_no_expand(copy, from, to) > 0) {
3096       // Keep replacing "::" -> ":" until we have no more "::" (non-windows)
3097       // Keep replacing ";;" -> ";" until we have no more ";;" (windows)
3098     }
3099 
3100     _java_class_path->set_writeable_value(copy);
3101     FreeHeap(copy); // a copy was made by set_value, so don't need this anymore
3102   }
3103 }

3426 jint Arguments::parse_options_buffer(const char* name, char* buffer, const size_t buf_len, ScopedVMInitArgs* vm_args) {
3427   GrowableArray<JavaVMOption> *options = new (ResourceObj::C_HEAP, mtArguments) GrowableArray<JavaVMOption>(2, true);    // Construct option array
3428 
3429   // some pointers to help with parsing
3430   char *buffer_end = buffer + buf_len;
3431   char *opt_hd = buffer;
3432   char *wrt = buffer;
3433   char *rd = buffer;
3434 
3435   // parse all options
3436   while (rd < buffer_end) {
3437     // skip leading white space from the input string
3438     while (rd < buffer_end && isspace(*rd)) {
3439       rd++;
3440     }
3441 
3442     if (rd >= buffer_end) {
3443       break;
3444     }
3445 
3446     // Remember this is where we found the head of the token.
3447     opt_hd = wrt;
3448 
3449     // Tokens are strings of non white space characters separated
3450     // by one or more white spaces.
3451     while (rd < buffer_end && !isspace(*rd)) {
3452       if (*rd == '\'' || *rd == '"') {      // handle a quoted string
3453         int quote = *rd;                    // matching quote to look for
3454         rd++;                               // don't copy open quote
3455         while (rd < buffer_end && *rd != quote) {
3456                                             // include everything (even spaces)
3457                                             // up until the close quote
3458           *wrt++ = *rd++;                   // copy to option string
3459         }
3460 
3461         if (rd < buffer_end) {
3462           rd++;                             // don't copy close quote
3463         } else {
3464                                             // did not see closing quote
3465           jio_fprintf(defaultStream::error_stream(),
3466                       "Unmatched quote in %s\n", name);
3467           delete options;
3468           return JNI_ERR;
3469         }
3470       } else {
3471         *wrt++ = *rd++;                     // copy to option string
3472       }
3473     }
3474 
3475     // steal a white space character and set it to NULL
3476     *wrt++ = '\0';
3477     // We now have a complete token
3478 
3479     JavaVMOption option;
3480     option.optionString = opt_hd;
3481     option.extraInfo = NULL;
3482 
3483     options->append(option);                // Fill in option
3484 
3485     rd++;  // Advance to next character
3486   }
3487 
3488   // Fill out JavaVMInitArgs structure.
3489   jint status = vm_args->set_args(options);
3490 
3491   delete options;
3492   return status;
3493 }

3709 jint Arguments::match_special_option_and_act(const JavaVMInitArgs* args,
3710                                              ScopedVMInitArgs* args_out) {
3711   // Remaining part of option string
3712   const char* tail;
3713   ScopedVMInitArgs vm_options_file_args(args_out->container_name());
3714 
3715   for (int index = 0; index < args->nOptions; index++) {
3716     const JavaVMOption* option = args->options + index;
3717     if (match_option(option, "-XX:Flags=", &tail)) {
3718       Arguments::set_jvm_flags_file(tail);
3719       continue;
3720     }
3721     if (match_option(option, "-XX:VMOptionsFile=", &tail)) {
3722       if (vm_options_file_args.found_vm_options_file_arg()) {
3723         jio_fprintf(defaultStream::error_stream(),
3724                     "The option '%s' is already specified in the options "
3725                     "container '%s' so the specification of '%s' in the "
3726                     "same options container is an error.\n",
3727                     vm_options_file_args.vm_options_file_arg(),
3728                     vm_options_file_args.container_name(),
3729                     option->optionString);
3730         return JNI_EINVAL;
3731       }
3732       vm_options_file_args.set_vm_options_file_arg(option->optionString);
3733       // If there's a VMOptionsFile, parse that
3734       jint code = insert_vm_options_file(args, tail, index,
3735                                          &vm_options_file_args, args_out);
3736       if (code != JNI_OK) {
3737         return code;
3738       }
3739       args_out->set_vm_options_file_arg(vm_options_file_args.vm_options_file_arg());
3740       if (args_out->is_set()) {
3741         // The VMOptions file inserted some options so switch 'args'
3742         // to the new set of options, and continue processing which
3743         // preserves "last option wins" semantics.
3744         args = args_out->get();
3745         // The first option from the VMOptionsFile replaces the
3746         // current option.  So we back track to process the
3747         // replacement option.
3748         index--;
3749       }
3750       continue;
3751     }
3752     if (match_option(option, "-XX:+PrintVMOptions")) {
3753       PrintVMOptions = true;
3754       continue;
3755     }
3756     if (match_option(option, "-XX:-PrintVMOptions")) {
3757       PrintVMOptions = false;
3758       continue;
3759     }
3760     if (match_option(option, "-XX:+IgnoreUnrecognizedVMOptions")) {
3761       IgnoreUnrecognizedVMOptions = true;
3762       continue;
3763     }
3764     if (match_option(option, "-XX:-IgnoreUnrecognizedVMOptions")) {
3765       IgnoreUnrecognizedVMOptions = false;
3766       continue;
3767     }
3768     if (match_option(option, "-XX:+PrintFlagsInitial")) {
3769       JVMFlag::printFlags(tty, false);
3770       vm_exit(0);
3771     }
3772     if (match_option(option, "-XX:NativeMemoryTracking", &tail)) {
3773 #if INCLUDE_NMT
3774       // The launcher did not setup nmt environment variable properly.
3775       if (!MemTracker::check_launcher_nmt_support(tail)) {
3776         warning("Native Memory Tracking did not setup properly, using wrong launcher?");
3777       }
3778 
3779       // Verify if nmt option is valid.
3780       if (MemTracker::verify_nmt_option()) {
3781         // Late initialization, still in single-threaded mode.
3782         if (MemTracker::tracking_level() >= NMT_summary) {
3783           MemTracker::init();
3784         }
3785       } else {
3786         vm_exit_during_initialization("Syntax error, expecting -XX:NativeMemoryTracking=[off|summary|detail]", NULL);
3787       }
3788       continue;
3789 #else
3790       jio_fprintf(defaultStream::error_stream(),
3791         "Native Memory Tracking is not supported in this VM\n");
3792       return JNI_ERR;

4296 // Copies src into buf, replacing "%%" with "%" and "%p" with pid
4297 // Returns true if all of the source pointed by src has been copied over to
4298 // the destination buffer pointed by buf. Otherwise, returns false.
4299 // Notes:
4300 // 1. If the length (buflen) of the destination buffer excluding the
4301 // NULL terminator character is not long enough for holding the expanded
4302 // pid characters, it also returns false instead of returning the partially
4303 // expanded one.
4304 // 2. The passed in "buflen" should be large enough to hold the null terminator.
4305 bool Arguments::copy_expand_pid(const char* src, size_t srclen,
4306                                 char* buf, size_t buflen) {
4307   const char* p = src;
4308   char* b = buf;
4309   const char* src_end = &src[srclen];
4310   char* buf_end = &buf[buflen - 1];
4311 
4312   while (p < src_end && b < buf_end) {
4313     if (*p == '%') {
4314       switch (*(++p)) {
4315       case '%':         // "%%" ==> "%"
4316         *b++ = *p++;
4317         break;
4318       case 'p':  {       //  "%p" ==> current process id
4319         // buf_end points to the character before the last character so
4320         // that we could write '\0' to the end of the buffer.
4321         size_t buf_sz = buf_end - b + 1;
4322         int ret = jio_snprintf(b, buf_sz, "%d", os::current_process_id());
4323 
4324         // if jio_snprintf fails or the buffer is not long enough to hold
4325         // the expanded pid, returns false.
4326         if (ret < 0 || ret >= (int)buf_sz) {
4327           return false;
4328         } else {
4329           b += ret;
4330           assert(*b == '\0', "fail in copy_expand_pid");
4331           if (p == src_end && b == buf_end + 1) {
4332             // reach the end of the buffer.
4333             return true;
4334           }
4335         }
4336         p++;
4337         break;
4338       }
4339       default :
4340         *b++ = '%';
4341       }
4342     } else {
4343       *b++ = *p++;
4344     }
4345   }
4346   *b = '\0';
4347   return (p == src_end); // return false if not all of the source was copied
4348 }
