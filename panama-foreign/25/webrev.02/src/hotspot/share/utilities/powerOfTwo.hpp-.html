 36 // Power of two convenience library.
 37 
 38 // Round down to the closest power of two greater to or equal to the given
 39 // value.
 40 
 41 // Signed version: 0 is an invalid input, negative values are invalid
 42 template <typename T>
 43 inline typename EnableIf<IsSigned<T>::value, T>::type round_down_power_of_2(T value) {
 44   STATIC_ASSERT(IsIntegral<T>::value);
 45   assert(value > 0, "Invalid value");
 46   uint32_t lz = count_leading_zeros(value);
 47   assert(lz < sizeof(T) * BitsPerByte, "Sanity");
 48   return T(1) << (sizeof(T) * BitsPerByte - 1 - lz);
 49 }
 50 
 51 // Unsigned version: 0 is an invalid input
 52 template <typename T>
 53 inline typename EnableIf<!IsSigned<T>::value, T>::type round_down_power_of_2(T value) {
 54   STATIC_ASSERT(IsIntegral<T>::value);
 55   assert(value != 0, "Invalid value");
 56   uint32_t lz = count_leading_zeros(value);
 57   assert(lz < sizeof(T) * BitsPerByte, "Sanity");
 58   return T(1) << (sizeof(T) * BitsPerByte - 1 - lz);
 59 }
 60 
 61 // Round up to the closest power of two greater to or equal to
 62 // the given value.
 63 
 64 // Signed version: 0 is an invalid input, negative values are invalid,
 65 // overflows with assert if value is larger than 2^30 or 2^62 for 32- and
 66 // 64-bit integers, respectively
 67 template <typename T>
 68 inline typename EnableIf<IsSigned<T>::value, T>::type round_up_power_of_2(T value) {
 69   STATIC_ASSERT(IsIntegral<T>::value);
 70   STATIC_ASSERT(IsSigned<T>::value);
 71   assert(value > 0, "Invalid value");
 72   if (is_power_of_2_t(value)) {
 73     return value;
 74   }
 75   uint32_t lz = count_leading_zeros(value);
 76   assert(lz < sizeof(T) * BitsPerByte, "Sanity");
 77   assert(lz > 1, "Will overflow");
 78   return T(1) << (sizeof(T) * BitsPerByte - lz);
 79 }
 80 
 81 // Unsigned version: 0 is an invalid input, overflows with assert if value
 82 // is larger than 2^31 or 2^63 for 32- and 64-bit integers, respectively
 83 template <typename T>
 84 inline typename EnableIf<!IsSigned<T>::value, T>::type round_up_power_of_2(T value) {
 85   STATIC_ASSERT(IsIntegral<T>::value);
 86   STATIC_ASSERT(!IsSigned<T>::value);
 87   assert(value != 0, "Invalid value");
 88   if (is_power_of_2_t(value)) {
 89     return value;
 90   }
 91   uint32_t lz = count_leading_zeros(value);
 92   assert(lz < sizeof(T) * BitsPerByte, "Sanity");
 93   assert(lz > 0, "Will overflow");
 94   return T(1) << (sizeof(T) * BitsPerByte - lz);
 95 }
 96 
 97 // Helper function to get the maximum positive value. Implemented here
 98 // since using std::numeric_limits<T>::max() seems problematic on some
 99 // platforms.
100 
101 template <typename T> T max_value() {
102   if (IsSigned<T>::value) {
103     // Highest positive power of two expressible in the type
104     uint64_t val = static_cast<T>(1) << (sizeof(T) * BitsPerByte - 2);
105     // Fill lower bits with ones
106     val |= val >> 1;
107     val |= val >> 2;
108     val |= val >> 4;
109     if (sizeof(T) >= 2)  val |= val >> 8;
110     if (sizeof(T) >= 4)  val |= val >> 16;
111     if (sizeof(T) == 8)  val |= val >> 32;
112     return (T)val;
113   } else {
114     return ~(static_cast<T>(0));
115   }
116 }
117 
118 // Calculate the next power of two greater than the given value.
119 
120 // Accepts 0 (returns 1), overflows with assert if value is larger than
121 // or equal to 2^31 (signed: 2^30) or 2^63 (signed: 2^62), for 32-
122 // and 64-bit integers, respectively
123 template <typename T>
124 inline T next_power_of_2(T value)  {
125   assert(value != max_value<T>(), "Overflow");
126   return round_up_power_of_2(value + 1);
127 }
