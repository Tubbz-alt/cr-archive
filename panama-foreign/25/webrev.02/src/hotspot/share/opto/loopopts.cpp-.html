  46 //------------------------------split_thru_phi---------------------------------
  47 // Split Node 'n' through merge point if there is enough win.
  48 Node *PhaseIdealLoop::split_thru_phi( Node *n, Node *region, int policy ) {
  49   if (n->Opcode() == Op_ConvI2L && n->bottom_type() != TypeLong::LONG) {
  50     // ConvI2L may have type information on it which is unsafe to push up
  51     // so disable this for now
  52     return NULL;
  53   }
  54 
  55   // Splitting range check CastIIs through a loop induction Phi can
  56   // cause new Phis to be created that are left unrelated to the loop
  57   // induction Phi and prevent optimizations (vectorization)
  58   if (n->Opcode() == Op_CastII && n->as_CastII()->has_range_check() &&
  59       region->is_CountedLoop() && n->in(1) == region->as_CountedLoop()->phi()) {
  60     return NULL;
  61   }
  62 
  63   int wins = 0;
  64   assert(!n->is_CFG(), "");
  65   assert(region->is_Region(), "");
  66 
  67   const Type* type = n->bottom_type();
  68   const TypeOopPtr *t_oop = _igvn.type(n)->isa_oopptr();
  69   Node *phi;
  70   if (t_oop != NULL && t_oop->is_known_instance_field()) {
  71     int iid    = t_oop->instance_id();
  72     int index  = C->get_alias_index(t_oop);
  73     int offset = t_oop->offset();
  74     phi = new PhiNode(region, type, NULL, iid, index, offset);
  75   } else {
  76     phi = PhiNode::make_blank(region, n);
  77   }
  78   uint old_unique = C->unique();
  79   for (uint i = 1; i < region->req(); i++) {
  80     Node *x;
  81     Node* the_clone = NULL;
  82     if (region->in(i) == C->top()) {
  83       x = C->top();             // Dead path?  Use a dead data op
  84     } else {
  85       x = n->clone();           // Else clone up the data op
  86       the_clone = x;            // Remember for possible deletion.
  87       // Alter data node to use pre-phi inputs
  88       if (n->in(0) == region)
  89         x->set_req( 0, region->in(i) );
  90       for (uint j = 1; j < n->req(); j++) {
  91         Node *in = n->in(j);
  92         if (in->is_Phi() && in->in(0) == region)
  93           x->set_req( j, in->in(i) ); // Use pre-Phi input for the clone
  94       }
  95     }
  96     // Check for a 'win' on some paths
  97     const Type *t = x->Value(&_igvn);
  98 
  99     bool singleton = t->singleton();
 100 
 101     // A TOP singleton indicates that there are no possible values incoming
 102     // along a particular edge. In most cases, this is OK, and the Phi will
 103     // be eliminated later in an Ideal call. However, we can't allow this to
 104     // happen if the singleton occurs on loop entry, as the elimination of
 105     // the PhiNode may cause the resulting node to migrate back to a previous
 106     // loop iteration.
 107     if (singleton && t == Type::TOP) {
 108       // Is_Loop() == false does not confirm the absence of a loop (e.g., an
 109       // irreducible loop may not be indicated by an affirmative is_Loop());
 110       // therefore, the only top we can split thru a phi is on a backedge of
 111       // a loop.
 112       singleton &= region->is_Loop() && (i != LoopNode::EntryControl);
 113     }
 114 
 115     if (singleton) {
 116       wins++;
 117       x = ((PhaseGVN&)_igvn).makecon(t);
 118     } else {
 119       // We now call Identity to try to simplify the cloned node.
 120       // Note that some Identity methods call phase->type(this).
 121       // Make sure that the type array is big enough for
 122       // our new node, even though we may throw the node away.
 123       // (Note: This tweaking with igvn only works because x is a new node.)
 124       _igvn.set_type(x, t);
 125       // If x is a TypeNode, capture any more-precise type permanently into Node
 126       // otherwise it will be not updated during igvn->transform since
 127       // igvn->type(x) is set to x->Value() already.
 128       x->raise_bottom_type(t);
 129       Node* y = x->Identity(&_igvn);
 130       if (y != x) {
 131         wins++;
 132         x = y;
 133       } else {
 134         y = _igvn.hash_find(x);
 135         if (y) {
 136           wins++;
 137           x = y;
 138         } else {
 139           // Else x is a new node we are keeping
 140           // We do not need register_new_node_with_optimizer
 141           // because set_type has already been called.
 142           _igvn._worklist.push(x);
 143         }
 144       }
 145     }
 146     if (x != the_clone && the_clone != NULL)
 147       _igvn.remove_dead_node(the_clone);
 148     phi->set_req( i, x );
 149   }
 150   // Too few wins?
 151   if (wins <= policy) {
 152     _igvn.remove_dead_node(phi);
 153     return NULL;
 154   }
 155 
 156   // Record Phi
 157   register_new_node( phi, region );
 158 
 159   for (uint i2 = 1; i2 < phi->req(); i2++) {
 160     Node *x = phi->in(i2);
 161     // If we commoned up the cloned 'x' with another existing Node,
 162     // the existing Node picks up a new use.  We need to make the
 163     // existing Node occur higher up so it dominates its uses.
 164     Node *old_ctrl;
 165     IdealLoopTree *old_loop;
 166 
 167     if (x->is_Con()) {
 168       // Constant's control is always root.
 169       set_ctrl(x, C->root());
 170       continue;
 171     }
 172     // The occasional new node
 173     if (x->_idx >= old_unique) {     // Found a new, unplaced node?
 174       old_ctrl = NULL;
 175       old_loop = NULL;               // Not in any prior loop
 176     } else {
 177       old_ctrl = get_ctrl(x);
 178       old_loop = get_loop(old_ctrl); // Get prior loop
 179     }
 180     // New late point must dominate new use
 181     Node *new_ctrl = dom_lca(old_ctrl, region->in(i2));
 182     if (new_ctrl == old_ctrl) // Nothing is changed
 183       continue;
 184 
 185     IdealLoopTree *new_loop = get_loop(new_ctrl);
 186 
 187     // Don't move x into a loop if its uses are
 188     // outside of loop. Otherwise x will be cloned
 189     // for each use outside of this loop.
 190     IdealLoopTree *use_loop = get_loop(region);
 191     if (!new_loop->is_member(use_loop) &&
 192         (old_loop == NULL || !new_loop->is_member(old_loop))) {
 193       // Take early control, later control will be recalculated
 194       // during next iteration of loop optimizations.
 195       new_ctrl = get_early_ctrl(x);
 196       new_loop = get_loop(new_ctrl);
 197     }
 198     // Set new location
 199     set_ctrl(x, new_ctrl);
 200     // If changing loop bodies, see if we need to collect into new body
 201     if (old_loop != new_loop) {
 202       if (old_loop && !old_loop->_child)
 203         old_loop->_body.yank(x);
 204       if (!new_loop->_child)
 205         new_loop->_body.push(x);  // Collect body info
 206     }
 207   }
 208 
 209   return phi;
 210 }

 212 //------------------------------dominated_by------------------------------------
 213 // Replace the dominated test with an obvious true or false.  Place it on the
 214 // IGVN worklist for later cleanup.  Move control-dependent data Nodes on the
 215 // live path up to the dominating control.
 216 void PhaseIdealLoop::dominated_by( Node *prevdom, Node *iff, bool flip, bool exclude_loop_predicate ) {
 217   if (VerifyLoopOptimizations && PrintOpto) { tty->print_cr("dominating test"); }
 218 
 219   // prevdom is the dominating projection of the dominating test.
 220   assert( iff->is_If(), "" );
 221   assert(iff->Opcode() == Op_If || iff->Opcode() == Op_CountedLoopEnd || iff->Opcode() == Op_RangeCheck, "Check this code when new subtype is added");
 222   int pop = prevdom->Opcode();
 223   assert( pop == Op_IfFalse || pop == Op_IfTrue, "" );
 224   if (flip) {
 225     if (pop == Op_IfTrue)
 226       pop = Op_IfFalse;
 227     else
 228       pop = Op_IfTrue;
 229   }
 230   // 'con' is set to true or false to kill the dominated test.
 231   Node *con = _igvn.makecon(pop == Op_IfTrue ? TypeInt::ONE : TypeInt::ZERO);
 232   set_ctrl(con, C->root()); // Constant gets a new use
 233   // Hack the dominated test
 234   _igvn.replace_input_of(iff, 1, con);
 235 
 236   // If I dont have a reachable TRUE and FALSE path following the IfNode then
 237   // I can assume this path reaches an infinite loop.  In this case it's not
 238   // important to optimize the data Nodes - either the whole compilation will
 239   // be tossed or this path (and all data Nodes) will go dead.
 240   if (iff->outcnt() != 2) return;
 241 
 242   // Make control-dependent data Nodes on the live path (path that will remain
 243   // once the dominated IF is removed) become control-dependent on the
 244   // dominating projection.
 245   Node* dp = iff->as_If()->proj_out_or_null(pop == Op_IfTrue);
 246 
 247   // Loop predicates may have depending checks which should not
 248   // be skipped. For example, range check predicate has two checks
 249   // for lower and upper bounds.
 250   if (dp == NULL)
 251     return;
 252 
 253   ProjNode* dp_proj  = dp->as_Proj();
 254   ProjNode* unc_proj = iff->as_If()->proj_out(1 - dp_proj->_con)->as_Proj();
 255   if (exclude_loop_predicate &&
 256       (unc_proj->is_uncommon_trap_proj(Deoptimization::Reason_predicate) != NULL ||
 257        unc_proj->is_uncommon_trap_proj(Deoptimization::Reason_profile_predicate) != NULL ||
 258        unc_proj->is_uncommon_trap_proj(Deoptimization::Reason_range_check) != NULL)) {
 259     // If this is a range check (IfNode::is_range_check), do not
 260     // reorder because Compile::allow_range_check_smearing might have
 261     // changed the check.
 262     return; // Let IGVN transformation change control dependence.
 263   }
 264 
 265   IdealLoopTree *old_loop = get_loop(dp);
 266 
 267   for (DUIterator_Fast imax, i = dp->fast_outs(imax); i < imax; i++) {
 268     Node* cd = dp->fast_out(i); // Control-dependent node
 269     if (cd->depends_only_on_test()) {
 270       assert(cd->in(0) == dp, "");
 271       _igvn.replace_input_of(cd, 0, prevdom);
 272       set_early_ctrl(cd);
 273       IdealLoopTree *new_loop = get_loop(get_ctrl(cd));
 274       if (old_loop != new_loop) {
 275         if (!old_loop->_child) old_loop->_body.yank(cd);
 276         if (!new_loop->_child) new_loop->_body.push(cd);
 277       }
 278       --i;
 279       --imax;
 280     }
 281   }
 282 }

 284 //------------------------------has_local_phi_input----------------------------
 285 // Return TRUE if 'n' has Phi inputs from its local block and no other
 286 // block-local inputs (all non-local-phi inputs come from earlier blocks)
 287 Node *PhaseIdealLoop::has_local_phi_input( Node *n ) {
 288   Node *n_ctrl = get_ctrl(n);
 289   // See if some inputs come from a Phi in this block, or from before
 290   // this block.
 291   uint i;
 292   for( i = 1; i < n->req(); i++ ) {
 293     Node *phi = n->in(i);
 294     if( phi->is_Phi() && phi->in(0) == n_ctrl )
 295       break;
 296   }
 297   if( i >= n->req() )
 298     return NULL;                // No Phi inputs; nowhere to clone thru
 299 
 300   // Check for inputs created between 'n' and the Phi input.  These
 301   // must split as well; they have already been given the chance
 302   // (courtesy of a post-order visit) and since they did not we must
 303   // recover the 'cost' of splitting them by being very profitable
 304   // when splitting 'n'.  Since this is unlikely we simply give up.
 305   for( i = 1; i < n->req(); i++ ) {
 306     Node *m = n->in(i);
 307     if( get_ctrl(m) == n_ctrl && !m->is_Phi() ) {
 308       // We allow the special case of AddP's with no local inputs.
 309       // This allows us to split-up address expressions.
 310       if (m->is_AddP() &&
 311           get_ctrl(m->in(2)) != n_ctrl &&
 312           get_ctrl(m->in(3)) != n_ctrl) {
 313         // Move the AddP up to dominating point
 314         Node* c = find_non_split_ctrl(idom(n_ctrl));
 315         if (c->is_OuterStripMinedLoop()) {
 316           c->as_Loop()->verify_strip_mined(1);
 317           c = c->in(LoopNode::EntryControl);
 318         }
 319         set_ctrl_and_loop(m, c);
 320         continue;
 321       }
 322       return NULL;
 323     }
 324     assert(n->is_Phi() || m->is_Phi() || is_dominator(get_ctrl(m), n_ctrl), "m has strange control");
 325   }
 326 
 327   return n_ctrl;
 328 }

 330 //------------------------------remix_address_expressions----------------------
 331 // Rework addressing expressions to get the most loop-invariant stuff
 332 // moved out.  We'd like to do all associative operators, but it's especially
 333 // important (common) to do address expressions.
 334 Node *PhaseIdealLoop::remix_address_expressions( Node *n ) {
 335   if (!has_ctrl(n))  return NULL;
 336   Node *n_ctrl = get_ctrl(n);
 337   IdealLoopTree *n_loop = get_loop(n_ctrl);
 338 
 339   // See if 'n' mixes loop-varying and loop-invariant inputs and
 340   // itself is loop-varying.
 341 
 342   // Only interested in binary ops (and AddP)
 343   if( n->req() < 3 || n->req() > 4 ) return NULL;
 344 
 345   Node *n1_ctrl = get_ctrl(n->in(                    1));
 346   Node *n2_ctrl = get_ctrl(n->in(                    2));
 347   Node *n3_ctrl = get_ctrl(n->in(n->req() == 3 ? 2 : 3));
 348   IdealLoopTree *n1_loop = get_loop( n1_ctrl );
 349   IdealLoopTree *n2_loop = get_loop( n2_ctrl );
 350   IdealLoopTree *n3_loop = get_loop( n3_ctrl );
 351 
 352   // Does one of my inputs spin in a tighter loop than self?
 353   if( (n_loop->is_member( n1_loop ) && n_loop != n1_loop) ||
 354       (n_loop->is_member( n2_loop ) && n_loop != n2_loop) ||
 355       (n_loop->is_member( n3_loop ) && n_loop != n3_loop) )
 356     return NULL;                // Leave well enough alone
 357 
 358   // Is at least one of my inputs loop-invariant?
 359   if( n1_loop == n_loop &&
 360       n2_loop == n_loop &&
 361       n3_loop == n_loop )
 362     return NULL;                // No loop-invariant inputs
 363 
 364 
 365   int n_op = n->Opcode();
 366 
 367   // Replace expressions like ((V+I) << 2) with (V<<2 + I<<2).
 368   if( n_op == Op_LShiftI ) {
 369     // Scale is loop invariant
 370     Node *scale = n->in(2);
 371     Node *scale_ctrl = get_ctrl(scale);
 372     IdealLoopTree *scale_loop = get_loop(scale_ctrl );
 373     if( n_loop == scale_loop || !scale_loop->is_member( n_loop ) )
 374       return NULL;
 375     const TypeInt *scale_t = scale->bottom_type()->isa_int();
 376     if( scale_t && scale_t->is_con() && scale_t->get_con() >= 16 )
 377       return NULL;              // Dont bother with byte/short masking
 378     // Add must vary with loop (else shift would be loop-invariant)
 379     Node *add = n->in(1);
 380     Node *add_ctrl = get_ctrl(add);
 381     IdealLoopTree *add_loop = get_loop(add_ctrl);
 382     //assert( n_loop == add_loop, "" );
 383     if( n_loop != add_loop ) return NULL;  // happens w/ evil ZKM loops
 384 
 385     // Convert I-V into I+ (0-V); same for V-I
 386     if( add->Opcode() == Op_SubI &&
 387         _igvn.type( add->in(1) ) != TypeInt::ZERO ) {
 388       Node *zero = _igvn.intcon(0);
 389       set_ctrl(zero, C->root());
 390       Node *neg = new SubINode( _igvn.intcon(0), add->in(2) );
 391       register_new_node( neg, get_ctrl(add->in(2) ) );
 392       add = new AddINode( add->in(1), neg );
 393       register_new_node( add, add_ctrl );
 394     }
 395     if( add->Opcode() != Op_AddI ) return NULL;
 396     // See if one add input is loop invariant
 397     Node *add_var = add->in(1);
 398     Node *add_var_ctrl = get_ctrl(add_var);
 399     IdealLoopTree *add_var_loop = get_loop(add_var_ctrl );
 400     Node *add_invar = add->in(2);
 401     Node *add_invar_ctrl = get_ctrl(add_invar);
 402     IdealLoopTree *add_invar_loop = get_loop(add_invar_ctrl );
 403     if( add_var_loop == n_loop ) {
 404     } else if( add_invar_loop == n_loop ) {
 405       // Swap to find the invariant part
 406       add_invar = add_var;
 407       add_invar_ctrl = add_var_ctrl;
 408       add_invar_loop = add_var_loop;
 409       add_var = add->in(2);
 410       Node *add_var_ctrl = get_ctrl(add_var);
 411       IdealLoopTree *add_var_loop = get_loop(add_var_ctrl );
 412     } else                      // Else neither input is loop invariant
 413       return NULL;
 414     if( n_loop == add_invar_loop || !add_invar_loop->is_member( n_loop ) )
 415       return NULL;              // No invariant part of the add?
 416 
 417     // Yes!  Reshape address expression!
 418     Node *inv_scale = new LShiftINode( add_invar, scale );
 419     Node *inv_scale_ctrl =
 420       dom_depth(add_invar_ctrl) > dom_depth(scale_ctrl) ?
 421       add_invar_ctrl : scale_ctrl;
 422     register_new_node( inv_scale, inv_scale_ctrl );
 423     Node *var_scale = new LShiftINode( add_var, scale );
 424     register_new_node( var_scale, n_ctrl );
 425     Node *var_add = new AddINode( var_scale, inv_scale );
 426     register_new_node( var_add, n_ctrl );
 427     _igvn.replace_node( n, var_add );
 428     return var_add;
 429   }
 430 
 431   // Replace (I+V) with (V+I)
 432   if( n_op == Op_AddI ||
 433       n_op == Op_AddL ||
 434       n_op == Op_AddF ||
 435       n_op == Op_AddD ||
 436       n_op == Op_MulI ||
 437       n_op == Op_MulL ||
 438       n_op == Op_MulF ||
 439       n_op == Op_MulD ) {
 440     if( n2_loop == n_loop ) {
 441       assert( n1_loop != n_loop, "" );
 442       n->swap_edges(1, 2);
 443     }
 444   }
 445 
 446   // Replace ((I1 +p V) +p I2) with ((I1 +p I2) +p V),
 447   // but not if I2 is a constant.
 448   if( n_op == Op_AddP ) {
 449     if( n2_loop == n_loop && n3_loop != n_loop ) {
 450       if( n->in(2)->Opcode() == Op_AddP && !n->in(3)->is_Con() ) {
 451         Node *n22_ctrl = get_ctrl(n->in(2)->in(2));
 452         Node *n23_ctrl = get_ctrl(n->in(2)->in(3));
 453         IdealLoopTree *n22loop = get_loop( n22_ctrl );
 454         IdealLoopTree *n23_loop = get_loop( n23_ctrl );
 455         if( n22loop != n_loop && n22loop->is_member(n_loop) &&
 456             n23_loop == n_loop ) {
 457           Node *add1 = new AddPNode( n->in(1), n->in(2)->in(2), n->in(3) );
 458           // Stuff new AddP in the loop preheader
 459           register_new_node( add1, n_loop->_head->in(LoopNode::EntryControl) );
 460           Node *add2 = new AddPNode( n->in(1), add1, n->in(2)->in(3) );
 461           register_new_node( add2, n_ctrl );
 462           _igvn.replace_node( n, add2 );
 463           return add2;
 464         }
 465       }
 466     }
 467 
 468     // Replace (I1 +p (I2 + V)) with ((I1 +p I2) +p V)
 469     if (n2_loop != n_loop && n3_loop == n_loop) {
 470       if (n->in(3)->Opcode() == Op_AddX) {
 471         Node *V = n->in(3)->in(1);
 472         Node *I = n->in(3)->in(2);
 473         if (is_member(n_loop,get_ctrl(V))) {
 474         } else {
 475           Node *tmp = V; V = I; I = tmp;
 476         }
 477         if (!is_member(n_loop,get_ctrl(I))) {
 478           Node *add1 = new AddPNode(n->in(1), n->in(2), I);
 479           // Stuff new AddP in the loop preheader
 480           register_new_node(add1, n_loop->_head->in(LoopNode::EntryControl));
 481           Node *add2 = new AddPNode(n->in(1), add1, V);
 482           register_new_node(add2, n_ctrl);
 483           _igvn.replace_node(n, add2);
 484           return add2;
 485         }
 486       }
 487     }
 488   }
 489 
 490   return NULL;
 491 }

 493 // Optimize ((in1[2*i] * in2[2*i]) + (in1[2*i+1] * in2[2*i+1]))
 494 Node *PhaseIdealLoop::convert_add_to_muladd(Node* n) {
 495   assert(n->Opcode() == Op_AddI, "sanity");
 496   Node * nn = NULL;
 497   Node * in1 = n->in(1);
 498   Node * in2 = n->in(2);
 499   if (in1->Opcode() == Op_MulI && in2->Opcode() == Op_MulI) {
 500     IdealLoopTree* loop_n = get_loop(get_ctrl(n));
 501     if (loop_n->is_counted() &&
 502         loop_n->_head->as_Loop()->is_valid_counted_loop() &&
 503         Matcher::match_rule_supported(Op_MulAddVS2VI) &&
 504         Matcher::match_rule_supported(Op_MulAddS2I)) {
 505       Node* mul_in1 = in1->in(1);
 506       Node* mul_in2 = in1->in(2);
 507       Node* mul_in3 = in2->in(1);
 508       Node* mul_in4 = in2->in(2);
 509       if (mul_in1->Opcode() == Op_LoadS &&
 510           mul_in2->Opcode() == Op_LoadS &&
 511           mul_in3->Opcode() == Op_LoadS &&
 512           mul_in4->Opcode() == Op_LoadS) {
 513         IdealLoopTree* loop1 = get_loop(get_ctrl(mul_in1));
 514         IdealLoopTree* loop2 = get_loop(get_ctrl(mul_in2));
 515         IdealLoopTree* loop3 = get_loop(get_ctrl(mul_in3));
 516         IdealLoopTree* loop4 = get_loop(get_ctrl(mul_in4));
 517         IdealLoopTree* loop5 = get_loop(get_ctrl(in1));
 518         IdealLoopTree* loop6 = get_loop(get_ctrl(in2));
 519         // All nodes should be in the same counted loop.
 520         if (loop_n == loop1 && loop_n == loop2 && loop_n == loop3 &&
 521             loop_n == loop4 && loop_n == loop5 && loop_n == loop6) {
 522           Node* adr1 = mul_in1->in(MemNode::Address);
 523           Node* adr2 = mul_in2->in(MemNode::Address);
 524           Node* adr3 = mul_in3->in(MemNode::Address);
 525           Node* adr4 = mul_in4->in(MemNode::Address);
 526           if (adr1->is_AddP() && adr2->is_AddP() && adr3->is_AddP() && adr4->is_AddP()) {
 527             if ((adr1->in(AddPNode::Base) == adr3->in(AddPNode::Base)) &&
 528                 (adr2->in(AddPNode::Base) == adr4->in(AddPNode::Base))) {
 529               nn = new MulAddS2INode(mul_in1, mul_in2, mul_in3, mul_in4);
 530               register_new_node(nn, get_ctrl(n));
 531               _igvn.replace_node(n, nn);
 532               return nn;
 533             } else if ((adr1->in(AddPNode::Base) == adr4->in(AddPNode::Base)) &&
 534                        (adr2->in(AddPNode::Base) == adr3->in(AddPNode::Base))) {
 535               nn = new MulAddS2INode(mul_in1, mul_in2, mul_in4, mul_in3);
 536               register_new_node(nn, get_ctrl(n));
 537               _igvn.replace_node(n, nn);
 538               return nn;
 539             }
 540           }
 541         }
 542       }
 543     }
 544   }
 545   return nn;

 548 //------------------------------conditional_move-------------------------------
 549 // Attempt to replace a Phi with a conditional move.  We have some pretty
 550 // strict profitability requirements.  All Phis at the merge point must
 551 // be converted, so we can remove the control flow.  We need to limit the
 552 // number of c-moves to a small handful.  All code that was in the side-arms
 553 // of the CFG diamond is now speculatively executed.  This code has to be
 554 // "cheap enough".  We are pretty much limited to CFG diamonds that merge
 555 // 1 or 2 items with a total of 1 or 2 ops executed speculatively.
 556 Node *PhaseIdealLoop::conditional_move( Node *region ) {
 557 
 558   assert(region->is_Region(), "sanity check");
 559   if (region->req() != 3) return NULL;
 560 
 561   // Check for CFG diamond
 562   Node *lp = region->in(1);
 563   Node *rp = region->in(2);
 564   if (!lp || !rp) return NULL;
 565   Node *lp_c = lp->in(0);
 566   if (lp_c == NULL || lp_c != rp->in(0) || !lp_c->is_If()) return NULL;
 567   IfNode *iff = lp_c->as_If();
 568 
 569   // Check for ops pinned in an arm of the diamond.
 570   // Can't remove the control flow in this case
 571   if (lp->outcnt() > 1) return NULL;
 572   if (rp->outcnt() > 1) return NULL;
 573 
 574   IdealLoopTree* r_loop = get_loop(region);
 575   assert(r_loop == get_loop(iff), "sanity");
 576   // Always convert to CMOVE if all results are used only outside this loop.
 577   bool used_inside_loop = (r_loop == _ltree_root);
 578 
 579   // Check profitability
 580   int cost = 0;
 581   int phis = 0;
 582   for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
 583     Node *out = region->fast_out(i);
 584     if (!out->is_Phi()) continue; // Ignore other control edges, etc
 585     phis++;
 586     PhiNode* phi = out->as_Phi();
 587     BasicType bt = phi->type()->basic_type();
 588     switch (bt) {
 589     case T_DOUBLE:
 590     case T_FLOAT:
 591       if (C->use_cmove()) {
 592         continue; //TODO: maybe we want to add some cost
 593       }
 594       cost += Matcher::float_cmove_cost(); // Could be very expensive
 595       break;
 596     case T_LONG: {
 597       cost += Matcher::long_cmove_cost(); // May encodes as 2 CMOV's
 598     }
 599     case T_INT:                 // These all CMOV fine
 600     case T_ADDRESS: {           // (RawPtr)
 601       cost++;
 602       break;
 603     }
 604     case T_NARROWOOP: // Fall through
 605     case T_OBJECT: {            // Base oops are OK, but not derived oops
 606       const TypeOopPtr *tp = phi->type()->make_ptr()->isa_oopptr();
 607       // Derived pointers are Bad (tm): what's the Base (for GC purposes) of a
 608       // CMOVE'd derived pointer?  It's a CMOVE'd derived base.  Thus
 609       // CMOVE'ing a derived pointer requires we also CMOVE the base.  If we
 610       // have a Phi for the base here that we convert to a CMOVE all is well
 611       // and good.  But if the base is dead, we'll not make a CMOVE.  Later
 612       // the allocator will have to produce a base by creating a CMOVE of the
 613       // relevant bases.  This puts the allocator in the business of
 614       // manufacturing expensive instructions, generally a bad plan.
 615       // Just Say No to Conditionally-Moved Derived Pointers.
 616       if (tp && tp->offset() != 0)
 617         return NULL;
 618       cost++;
 619       break;
 620     }
 621     default:
 622       return NULL;              // In particular, can't do memory or I/O
 623     }
 624     // Add in cost any speculative ops
 625     for (uint j = 1; j < region->req(); j++) {
 626       Node *proj = region->in(j);
 627       Node *inp = phi->in(j);
 628       if (get_ctrl(inp) == proj) { // Found local op
 629         cost++;
 630         // Check for a chain of dependent ops; these will all become
 631         // speculative in a CMOV.
 632         for (uint k = 1; k < inp->req(); k++)
 633           if (get_ctrl(inp->in(k)) == proj)
 634             cost += ConditionalMoveLimit; // Too much speculative goo
 635       }
 636     }
 637     // See if the Phi is used by a Cmp or Narrow oop Decode/Encode.
 638     // This will likely Split-If, a higher-payoff operation.
 639     for (DUIterator_Fast kmax, k = phi->fast_outs(kmax); k < kmax; k++) {
 640       Node* use = phi->fast_out(k);
 641       if (use->is_Cmp() || use->is_DecodeNarrowPtr() || use->is_EncodeNarrowPtr())
 642         cost += ConditionalMoveLimit;
 643       // Is there a use inside the loop?
 644       // Note: check only basic types since CMoveP is pinned.
 645       if (!used_inside_loop && is_java_primitive(bt)) {
 646         IdealLoopTree* u_loop = get_loop(has_ctrl(use) ? get_ctrl(use) : use);
 647         if (r_loop == u_loop || r_loop->is_member(u_loop)) {
 648           used_inside_loop = true;
 649         }
 650       }
 651     }
 652   }//for
 653   Node* bol = iff->in(1);
 654   if (bol->Opcode() == Op_Opaque4) {
 655     return NULL; // Ignore loop predicate checks (the Opaque4 ensures they will go away)
 656   }
 657   assert(bol->Opcode() == Op_Bool, "Unexpected node");
 658   int cmp_op = bol->in(1)->Opcode();
 659   // It is expensive to generate flags from a float compare.
 660   // Avoid duplicated float compare.
 661   if (phis > 1 && (cmp_op == Op_CmpF || cmp_op == Op_CmpD)) return NULL;
 662 
 663   float infrequent_prob = PROB_UNLIKELY_MAG(3);
 664   // Ignore cost and blocks frequency if CMOVE can be moved outside the loop.
 665   if (used_inside_loop) {
 666     if (cost >= ConditionalMoveLimit) return NULL; // Too much goo
 667 
 668     // BlockLayoutByFrequency optimization moves infrequent branch
 669     // from hot path. No point in CMOV'ing in such case (110 is used
 670     // instead of 100 to take into account not exactness of float value).
 671     if (BlockLayoutByFrequency) {
 672       infrequent_prob = MAX2(infrequent_prob, (float)BlockLayoutMinDiamondPercentage/110.0f);
 673     }
 674   }
 675   // Check for highly predictable branch.  No point in CMOV'ing if
 676   // we are going to predict accurately all the time.
 677   if (C->use_cmove() && (cmp_op == Op_CmpF || cmp_op == Op_CmpD)) {
 678     //keep going
 679   } else if (iff->_prob < infrequent_prob ||
 680       iff->_prob > (1.0f - infrequent_prob))
 681     return NULL;
 682 
 683   // --------------
 684   // Now replace all Phis with CMOV's
 685   Node *cmov_ctrl = iff->in(0);
 686   uint flip = (lp->Opcode() == Op_IfTrue);
 687   Node_List wq;
 688   while (1) {
 689     PhiNode* phi = NULL;
 690     for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
 691       Node *out = region->fast_out(i);
 692       if (out->is_Phi()) {
 693         phi = out->as_Phi();
 694         break;
 695       }
 696     }
 697     if (phi == NULL)  break;
 698     if (PrintOpto && VerifyLoopOptimizations) { tty->print_cr("CMOV"); }
 699     // Move speculative ops
 700     wq.push(phi);
 701     while (wq.size() > 0) {
 702       Node *n = wq.pop();
 703       for (uint j = 1; j < n->req(); j++) {
 704         Node* m = n->in(j);
 705         if (m != NULL && !is_dominator(get_ctrl(m), cmov_ctrl)) {
 706 #ifndef PRODUCT
 707           if (PrintOpto && VerifyLoopOptimizations) {
 708             tty->print("  speculate: ");
 709             m->dump();
 710           }
 711 #endif
 712           set_ctrl(m, cmov_ctrl);
 713           wq.push(m);
 714         }
 715       }
 716     }
 717     Node *cmov = CMoveNode::make(cmov_ctrl, iff->in(1), phi->in(1+flip), phi->in(2-flip), _igvn.type(phi));
 718     register_new_node( cmov, cmov_ctrl );
 719     _igvn.replace_node( phi, cmov );
 720 #ifndef PRODUCT
 721     if (TraceLoopOpts) {
 722       tty->print("CMOV  ");
 723       r_loop->dump_head();
 724       if (Verbose) {
 725         bol->in(1)->dump(1);
 726         cmov->dump(1);
 727       }
 728     }
 729     if (VerifyLoopOptimizations) verify();
 730 #endif
 731   }
 732 
 733   // The useless CFG diamond will fold up later; see the optimization in
 734   // RegionNode::Ideal.
 735   _igvn._worklist.push(region);
 736 
 737   return iff->in(1);
 738 }

 741   for (DUIterator_Fast imax, i = m->fast_outs(imax); i < imax; i++) {

 745         u = ((NeverBranchNode*)u)->proj_out(0);
 746         enqueue_cfg_uses(u, wq);
 747       } else {
 748         wq.push(u);
 749       }
 750     }
 751   }

 754 // Try moving a store out of a loop, right before the loop
 755 Node* PhaseIdealLoop::try_move_store_before_loop(Node* n, Node *n_ctrl) {
 756   // Store has to be first in the loop body
 757   IdealLoopTree *n_loop = get_loop(n_ctrl);
 758   if (n->is_Store() && n_loop != _ltree_root &&
 759       n_loop->is_loop() && n_loop->_head->is_Loop() &&
 760       n->in(0) != NULL) {
 761     Node* address = n->in(MemNode::Address);
 762     Node* value = n->in(MemNode::ValueIn);
 763     Node* mem = n->in(MemNode::Memory);
 764     IdealLoopTree* address_loop = get_loop(get_ctrl(address));
 765     IdealLoopTree* value_loop = get_loop(get_ctrl(value));
 766 
 767     // - address and value must be loop invariant
 768     // - memory must be a memory Phi for the loop
 769     // - Store must be the only store on this memory slice in the
 770     // loop: if there's another store following this one then value
 771     // written at iteration i by the second store could be overwritten
 772     // at iteration i+n by the first store: it's not safe to move the
 773     // first store out of the loop
 774     // - nothing must observe the memory Phi: it guarantees no read
 775     // before the store, we are also guaranteed the store post
 776     // dominates the loop head (ignoring a possible early
 777     // exit). Otherwise there would be extra Phi involved between the
 778     // loop's Phi and the store.
 779     // - there must be no early exit from the loop before the Store
 780     // (such an exit most of the time would be an extra use of the
 781     // memory Phi but sometimes is a bottom memory Phi that takes the
 782     // store as input).
 783 
 784     if (!n_loop->is_member(address_loop) &&
 785         !n_loop->is_member(value_loop) &&
 786         mem->is_Phi() && mem->in(0) == n_loop->_head &&
 787         mem->outcnt() == 1 &&
 788         mem->in(LoopNode::LoopBackControl) == n) {
 789 
 790       assert(n_loop->_tail != NULL, "need a tail");
 791       assert(is_dominator(n_ctrl, n_loop->_tail), "store control must not be in a branch in the loop");
 792 
 793       // Verify that there's no early exit of the loop before the store.
 794       bool ctrl_ok = false;
 795       {
 796         // Follow control from loop head until n, we exit the loop or
 797         // we reach the tail
 798         ResourceMark rm;
 799         Unique_Node_List wq;
 800         wq.push(n_loop->_head);
 801 
 802         for (uint next = 0; next < wq.size(); ++next) {
 803           Node *m = wq.at(next);
 804           if (m == n->in(0)) {
 805             ctrl_ok = true;
 806             continue;
 807           }
 808           assert(!has_ctrl(m), "should be CFG");
 809           if (!n_loop->is_member(get_loop(m)) || m == n_loop->_tail) {
 810             ctrl_ok = false;
 811             break;
 812           }
 813           enqueue_cfg_uses(m, wq);
 814           if (wq.size() > 10) {
 815             ctrl_ok = false;
 816             break;
 817           }
 818         }
 819       }
 820       if (ctrl_ok) {
 821         // move the Store
 822         _igvn.replace_input_of(mem, LoopNode::LoopBackControl, mem);
 823         _igvn.replace_input_of(n, 0, n_loop->_head->as_Loop()->skip_strip_mined()->in(LoopNode::EntryControl));
 824         _igvn.replace_input_of(n, MemNode::Memory, mem->in(LoopNode::EntryControl));
 825         // Disconnect the phi now. An empty phi can confuse other
 826         // optimizations in this pass of loop opts.
 827         _igvn.replace_node(mem, mem->in(LoopNode::EntryControl));
 828         n_loop->_body.yank(mem);
 829 
 830         set_ctrl_and_loop(n, n->in(0));
 831 
 832         return n;
 833       }
 834     }
 835   }
 836   return NULL;
 837 }

 839 // Try moving a store out of a loop, right after the loop
 840 void PhaseIdealLoop::try_move_store_after_loop(Node* n) {
 841   if (n->is_Store() && n->in(0) != NULL) {
 842     Node *n_ctrl = get_ctrl(n);
 843     IdealLoopTree *n_loop = get_loop(n_ctrl);
 844     // Store must be in a loop
 845     if (n_loop != _ltree_root && !n_loop->_irreducible) {
 846       Node* address = n->in(MemNode::Address);
 847       Node* value = n->in(MemNode::ValueIn);
 848       IdealLoopTree* address_loop = get_loop(get_ctrl(address));
 849       // address must be loop invariant
 850       if (!n_loop->is_member(address_loop)) {
 851         // Store must be last on this memory slice in the loop and
 852         // nothing in the loop must observe it
 853         Node* phi = NULL;
 854         for (DUIterator_Fast imax, i = n->fast_outs(imax); i < imax; i++) {
 855           Node* u = n->fast_out(i);
 856           if (has_ctrl(u)) { // control use?
 857             IdealLoopTree *u_loop = get_loop(get_ctrl(u));
 858             if (!n_loop->is_member(u_loop)) {
 859               continue;
 860             }
 861             if (u->is_Phi() && u->in(0) == n_loop->_head) {
 862               assert(_igvn.type(u) == Type::MEMORY, "bad phi");
 863               // multiple phis on the same slice are possible
 864               if (phi != NULL) {
 865                 return;
 866               }
 867               phi = u;
 868               continue;
 869             }
 870           }
 871           return;
 872         }
 873         if (phi != NULL) {
 874           // Nothing in the loop before the store (next iteration)
 875           // must observe the stored value
 876           bool mem_ok = true;
 877           {
 878             ResourceMark rm;
 879             Unique_Node_List wq;
 880             wq.push(phi);
 881             for (uint next = 0; next < wq.size() && mem_ok; ++next) {
 882               Node *m = wq.at(next);
 883               for (DUIterator_Fast imax, i = m->fast_outs(imax); i < imax && mem_ok; i++) {
 884                 Node* u = m->fast_out(i);
 885                 if (u->is_Store() || u->is_Phi()) {
 886                   if (u != n) {
 887                     wq.push(u);
 888                     mem_ok = (wq.size() <= 10);
 889                   }
 890                 } else {
 891                   mem_ok = false;
 892                   break;
 893                 }
 894               }
 895             }
 896           }
 897           if (mem_ok) {
 898             // Move the store out of the loop if the LCA of all
 899             // users (except for the phi) is outside the loop.
 900             Node* hook = new Node(1);
 901             hook->init_req(0, n_ctrl); // Add an input to prevent hook from being dead
 902             _igvn.rehash_node_delayed(phi);
 903             int count = phi->replace_edge(n, hook);
 904             assert(count > 0, "inconsistent phi");
 905 
 906             // Compute latest point this store can go
 907             Node* lca = get_late_ctrl(n, get_ctrl(n));
 908             if (n_loop->is_member(get_loop(lca))) {
 909               // LCA is in the loop - bail out
 910               _igvn.replace_node(hook, n);
 911               return;
 912             }
 913 #ifdef ASSERT
 914             if (n_loop->_head->is_Loop() && n_loop->_head->as_Loop()->is_strip_mined()) {
 915               assert(n_loop->_head->Opcode() == Op_CountedLoop, "outer loop is a strip mined");
 916               n_loop->_head->as_Loop()->verify_strip_mined(1);
 917               Node* outer = n_loop->_head->as_CountedLoop()->outer_loop();
 918               IdealLoopTree* outer_loop = get_loop(outer);
 919               assert(n_loop->_parent == outer_loop, "broken loop tree");
 920               assert(get_loop(lca) == outer_loop, "safepoint in outer loop consume all memory state");
 921             }
 922 #endif
 923 
 924             // Move store out of the loop
 925             _igvn.replace_node(hook, n->in(MemNode::Memory));
 926             _igvn.replace_input_of(n, 0, lca);
 927             set_ctrl_and_loop(n, lca);
 928 
 929             // Disconnect the phi now. An empty phi can confuse other
 930             // optimizations in this pass of loop opts..
 931             if (phi->in(LoopNode::LoopBackControl) == phi) {
 932               _igvn.replace_node(phi, phi->in(LoopNode::EntryControl));
 933               n_loop->_body.yank(phi);
 934             }
 935           }
 936         }
 937       }
 938     }
 939   }
 940 }

 942 //------------------------------split_if_with_blocks_pre-----------------------
 943 // Do the real work in a non-recursive function.  Data nodes want to be
 944 // cloned in the pre-order so they can feed each other nicely.
 945 Node *PhaseIdealLoop::split_if_with_blocks_pre( Node *n ) {
 946   // Cloning these guys is unlikely to win
 947   int n_op = n->Opcode();
 948   if( n_op == Op_MergeMem ) return n;
 949   if( n->is_Proj() ) return n;
 950   // Do not clone-up CmpFXXX variations, as these are always
 951   // followed by a CmpI
 952   if( n->is_Cmp() ) return n;
 953   // Attempt to use a conditional move instead of a phi/branch
 954   if( ConditionalMoveLimit > 0 && n_op == Op_Region ) {
 955     Node *cmov = conditional_move( n );
 956     if( cmov ) return cmov;
 957   }
 958   if( n->is_CFG() || n->is_LoadStore() )
 959     return n;
 960   if( n_op == Op_Opaque1 ||     // Opaque nodes cannot be mod'd
 961       n_op == Op_Opaque2 ) {
 962     if( !C->major_progress() )   // If chance of no more loop opts...
 963       _igvn._worklist.push(n);  // maybe we'll remove them
 964     return n;
 965   }
 966 
 967   if( n->is_Con() ) return n;   // No cloning for Con nodes
 968 
 969   Node *n_ctrl = get_ctrl(n);
 970   if( !n_ctrl ) return n;       // Dead node
 971 
 972   Node* res = try_move_store_before_loop(n, n_ctrl);
 973   if (res != NULL) {
 974     return n;
 975   }
 976 
 977   // Attempt to remix address expressions for loop invariants
 978   Node *m = remix_address_expressions( n );
 979   if( m ) return m;
 980 
 981   if (n_op == Op_AddI) {
 982     Node *nn = convert_add_to_muladd( n );
 983     if ( nn ) return nn;
 984   }
 985 
 986   if (n->is_ConstraintCast()) {
 987     Node* dom_cast = n->as_ConstraintCast()->dominating_cast(&_igvn, this);
 988     // ConstraintCastNode::dominating_cast() uses node control input to determine domination.
 989     // Node control inputs don't necessarily agree with loop control info (due to
 990     // transformations happened in between), thus additional dominance check is needed
 991     // to keep loop info valid.
 992     if (dom_cast != NULL && is_dominator(get_ctrl(dom_cast), get_ctrl(n))) {
 993       _igvn.replace_node(n, dom_cast);
 994       return dom_cast;
 995     }
 996   }
 997 
 998   // Determine if the Node has inputs from some local Phi.
 999   // Returns the block to clone thru.
1000   Node *n_blk = has_local_phi_input( n );
1001   if( !n_blk ) return n;
1002 
1003   // Do not clone the trip counter through on a CountedLoop
1004   // (messes up the canonical shape).
1005   if( n_blk->is_CountedLoop() && n->Opcode() == Op_AddI ) return n;
1006 
1007   // Check for having no control input; not pinned.  Allow
1008   // dominating control.
1009   if (n->in(0)) {
1010     Node *dom = idom(n_blk);
1011     if (dom_lca(n->in(0), dom) != n->in(0)) {
1012       return n;
1013     }
1014   }
1015   // Policy: when is it profitable.  You must get more wins than
1016   // policy before it is considered profitable.  Policy is usually 0,
1017   // so 1 win is considered profitable.  Big merges will require big
1018   // cloning, so get a larger policy.
1019   int policy = n_blk->req() >> 2;
1020 
1021   // If the loop is a candidate for range check elimination,
1022   // delay splitting through it's phi until a later loop optimization
1023   if (n_blk->is_CountedLoop()) {
1024     IdealLoopTree *lp = get_loop(n_blk);
1025     if (lp && lp->_rce_candidate) {
1026       return n;
1027     }
1028   }
1029 
1030   if (must_throttle_split_if()) return n;
1031 
1032   // Split 'n' through the merge point if it is profitable
1033   Node *phi = split_thru_phi( n, n_blk, policy );
1034   if (!phi) return n;
1035 
1036   // Found a Phi to split thru!
1037   // Replace 'n' with the new phi
1038   _igvn.replace_node( n, phi );
1039   // Moved a load around the loop, 'en-registering' something.
1040   if (n_blk->is_Loop() && n->is_Load() &&
1041       !phi->in(LoopNode::LoopBackControl)->is_Load())
1042     C->set_major_progress();
1043 
1044   return phi;
1045 }

1047 static bool merge_point_too_heavy(Compile* C, Node* region) {
1048   // Bail out if the region and its phis have too many users.
1049   int weight = 0;
1050   for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
1051     weight += region->fast_out(i)->outcnt();
1052   }
1053   int nodes_left = C->max_node_limit() - C->live_nodes();
1054   if (weight * 8 > nodes_left) {
1055     if (PrintOpto) {
1056       tty->print_cr("*** Split-if bails out:  %d nodes, region weight %d", C->unique(), weight);
1057     }
1058     return true;
1059   } else {
1060     return false;
1061   }
1062 }

1064 static bool merge_point_safe(Node* region) {
1065   // 4799512: Stop split_if_with_blocks from splitting a block with a ConvI2LNode
1066   // having a PhiNode input. This sidesteps the dangerous case where the split

1139 bool PhaseIdealLoop::can_split_if(Node* n_ctrl) {
1140   if (must_throttle_split_if()) {
1141     return false;
1142   }
1143 
1144   // Do not do 'split-if' if irreducible loops are present.
1145   if (_has_irreducible_loops) {
1146     return false;
1147   }
1148 
1149   if (merge_point_too_heavy(C, n_ctrl)) {
1150     return false;
1151   }
1152 
1153   // Do not do 'split-if' if some paths are dead.  First do dead code
1154   // elimination and then see if its still profitable.
1155   for (uint i = 1; i < n_ctrl->req(); i++) {
1156     if (n_ctrl->in(i) == C->top()) {
1157       return false;
1158     }
1159   }
1160 
1161   // If trying to do a 'Split-If' at the loop head, it is only
1162   // profitable if the cmp folds up on BOTH paths.  Otherwise we
1163   // risk peeling a loop forever.
1164 
1165   // CNC - Disabled for now.  Requires careful handling of loop
1166   // body selection for the cloned code.  Also, make sure we check
1167   // for any input path not being in the same loop as n_ctrl.  For
1168   // irreducible loops we cannot check for 'n_ctrl->is_Loop()'
1169   // because the alternative loop entry points won't be converted
1170   // into LoopNodes.
1171   IdealLoopTree *n_loop = get_loop(n_ctrl);
1172   for (uint j = 1; j < n_ctrl->req(); j++) {
1173     if (get_loop(n_ctrl->in(j)) != n_loop) {
1174       return false;
1175     }
1176   }
1177 
1178   // Check for safety of the merge point.
1179   if (!merge_point_safe(n_ctrl)) {
1180     return false;
1181   }
1182 
1183   return true;
1184 }

1202 //------------------------------split_if_with_blocks_post----------------------
1203 // Do the real work in a non-recursive function.  CFG hackery wants to be
1204 // in the post-order, so it can dirty the I-DOM info and not use the dirtied
1205 // info.
1206 void PhaseIdealLoop::split_if_with_blocks_post(Node *n) {
1207 
1208   // Cloning Cmp through Phi's involves the split-if transform.
1209   // FastLock is not used by an If
1210   if (n->is_Cmp() && !n->is_FastLock()) {
1211     Node *n_ctrl = get_ctrl(n);
1212     // Determine if the Node has inputs from some local Phi.
1213     // Returns the block to clone thru.
1214     Node *n_blk = has_local_phi_input(n);
1215     if (n_blk != n_ctrl) {
1216       return;
1217     }
1218 
1219     if (!can_split_if(n_ctrl)) {
1220       return;
1221     }
1222 
1223     if (n->outcnt() != 1) {
1224       return; // Multiple bool's from 1 compare?
1225     }
1226     Node *bol = n->unique_out();
1227     assert(bol->is_Bool(), "expect a bool here");
1228     if (bol->outcnt() != 1) {
1229       return;// Multiple branches from 1 compare?
1230     }
1231     Node *iff = bol->unique_out();
1232 
1233     // Check some safety conditions
1234     if (iff->is_If()) {        // Classic split-if?
1235       if (iff->in(0) != n_ctrl) {
1236         return; // Compare must be in same blk as if
1237       }
1238     } else if (iff->is_CMove()) { // Trying to split-up a CMOVE
1239       // Can't split CMove with different control edge.
1240       if (iff->in(0) != NULL && iff->in(0) != n_ctrl ) {
1241         return;
1242       }
1243       if (get_ctrl(iff->in(2)) == n_ctrl ||
1244           get_ctrl(iff->in(3)) == n_ctrl) {
1245         return;                 // Inputs not yet split-up
1246       }
1247       if (get_loop(n_ctrl) != get_loop(get_ctrl(iff))) {
1248         return;                 // Loop-invar test gates loop-varying CMOVE
1249       }
1250     } else {
1251       return;  // some other kind of node, such as an Allocate
1252     }
1253 
1254     // When is split-if profitable?  Every 'win' on means some control flow
1255     // goes dead, so it's almost always a win.
1256     int policy = 0;
1257     // Split compare 'n' through the merge point if it is profitable
1258     Node *phi = split_thru_phi( n, n_ctrl, policy);
1259     if (!phi) {
1260       return;
1261     }
1262 
1263     // Found a Phi to split thru!
1264     // Replace 'n' with the new phi
1265     _igvn.replace_node(n, phi);
1266 
1267     // Now split the bool up thru the phi
1268     Node *bolphi = split_thru_phi(bol, n_ctrl, -1);
1269     guarantee(bolphi != NULL, "null boolean phi node");
1270 
1271     _igvn.replace_node(bol, bolphi);
1272     assert(iff->in(1) == bolphi, "");
1273 
1274     if (bolphi->Value(&_igvn)->singleton()) {
1275       return;
1276     }
1277 
1278     // Conditional-move?  Must split up now
1279     if (!iff->is_If()) {
1280       Node *cmovphi = split_thru_phi(iff, n_ctrl, -1);
1281       _igvn.replace_node(iff, cmovphi);
1282       return;
1283     }
1284 
1285     // Now split the IF
1286     do_split_if(iff);
1287     return;
1288   }
1289 
1290   // Two identical ifs back to back can be merged
1291   if (identical_backtoback_ifs(n) && can_split_if(n->in(0))) {
1292     Node *n_ctrl = n->in(0);
1293     PhiNode* bolphi = PhiNode::make_blank(n_ctrl, n->in(1));
1294     IfNode* dom_if = idom(n_ctrl)->as_If();
1295     Node* proj_true = dom_if->proj_out(1);
1296     Node* proj_false = dom_if->proj_out(0);
1297     Node* con_true = _igvn.makecon(TypeInt::ONE);
1298     Node* con_false = _igvn.makecon(TypeInt::ZERO);
1299 
1300     for (uint i = 1; i < n_ctrl->req(); i++) {
1301       if (is_dominator(proj_true, n_ctrl->in(i))) {
1302         bolphi->init_req(i, con_true);
1303       } else {
1304         assert(is_dominator(proj_false, n_ctrl->in(i)), "bad if");
1305         bolphi->init_req(i, con_false);
1306       }
1307     }
1308     register_new_node(bolphi, n_ctrl);
1309     _igvn.replace_input_of(n, 1, bolphi);
1310 
1311     // Now split the IF
1312     do_split_if(n);
1313     return;
1314   }
1315 
1316   // Check for an IF ready to split; one that has its
1317   // condition codes input coming from a Phi at the block start.
1318   int n_op = n->Opcode();
1319 
1320   // Check for an IF being dominated by another IF same test
1321   if (n_op == Op_If ||
1322       n_op == Op_RangeCheck) {
1323     Node *bol = n->in(1);
1324     uint max = bol->outcnt();
1325     // Check for same test used more than once?
1326     if (max > 1 && bol->is_Bool()) {
1327       // Search up IDOMs to see if this IF is dominated.
1328       Node *cutoff = get_ctrl(bol);
1329 
1330       // Now search up IDOMs till cutoff, looking for a dominating test
1331       Node *prevdom = n;
1332       Node *dom = idom(prevdom);
1333       while (dom != cutoff) {
1334         if (dom->req() > 1 && dom->in(1) == bol && prevdom->in(0) == dom) {
1335           // It's invalid to move control dependent data nodes in the inner
1336           // strip-mined loop, because:
1337           //  1) break validation of LoopNode::verify_strip_mined()
1338           //  2) move code with side-effect in strip-mined loop
1339           // Move to the exit of outer strip-mined loop in that case.
1340           Node* out_le = is_inner_of_stripmined_loop(dom);
1341           if (out_le != NULL) {
1342             prevdom = out_le;
1343           }
1344           // Replace the dominated test with an obvious true or false.
1345           // Place it on the IGVN worklist for later cleanup.
1346           C->set_major_progress();
1347           dominated_by(prevdom, n, false, true);
1348 #ifndef PRODUCT
1349           if( VerifyLoopOptimizations ) verify();
1350 #endif
1351           return;
1352         }
1353         prevdom = dom;
1354         dom = idom(prevdom);
1355       }
1356     }
1357   }
1358 
1359   // See if a shared loop-varying computation has no loop-varying uses.
1360   // Happens if something is only used for JVM state in uncommon trap exits,
1361   // like various versions of induction variable+offset.  Clone the
1362   // computation per usage to allow it to sink out of the loop.
1363   if (has_ctrl(n) && !n->in(0)) {// n not dead and has no control edge (can float about)
1364     Node *n_ctrl = get_ctrl(n);
1365     IdealLoopTree *n_loop = get_loop(n_ctrl);
1366     if( n_loop != _ltree_root ) {
1367       DUIterator_Fast imax, i = n->fast_outs(imax);
1368       for (; i < imax; i++) {
1369         Node* u = n->fast_out(i);
1370         if( !has_ctrl(u) )     break; // Found control user
1371         IdealLoopTree *u_loop = get_loop(get_ctrl(u));
1372         if( u_loop == n_loop ) break; // Found loop-varying use
1373         if( n_loop->is_member( u_loop ) ) break; // Found use in inner loop
1374         if( u->Opcode() == Op_Opaque1 ) break; // Found loop limit, bugfix for 4677003
1375       }
1376       bool did_break = (i < imax);  // Did we break out of the previous loop?
1377       if (!did_break && n->outcnt() > 1) { // All uses in outer loops!
1378         Node *late_load_ctrl = NULL;
1379         if (n->is_Load()) {
1380           // If n is a load, get and save the result from get_late_ctrl(),
1381           // to be later used in calculating the control for n's clones.
1382           clear_dom_lca_tags();
1383           late_load_ctrl = get_late_ctrl(n, n_ctrl);
1384         }
1385         // If n is a load, and the late control is the same as the current
1386         // control, then the cloning of n is a pointless exercise, because
1387         // GVN will ensure that we end up where we started.
1388         if (!n->is_Load() || late_load_ctrl != n_ctrl) {
1389           for (DUIterator_Last jmin, j = n->last_outs(jmin); j >= jmin; ) {
1390             Node *u = n->last_out(j); // Clone private computation per use
1391             _igvn.rehash_node_delayed(u);
1392             Node *x = n->clone(); // Clone computation
1393             Node *x_ctrl = NULL;
1394             if( u->is_Phi() ) {
1395               // Replace all uses of normal nodes.  Replace Phi uses
1396               // individually, so the separate Nodes can sink down
1397               // different paths.
1398               uint k = 1;
1399               while( u->in(k) != n ) k++;
1400               u->set_req( k, x );
1401               // x goes next to Phi input path
1402               x_ctrl = u->in(0)->in(k);
1403               --j;
1404             } else {              // Normal use
1405               // Replace all uses
1406               for( uint k = 0; k < u->req(); k++ ) {
1407                 if( u->in(k) == n ) {
1408                   u->set_req( k, x );
1409                   --j;
1410                 }
1411               }
1412               x_ctrl = get_ctrl(u);
1413             }
1414 
1415             // Find control for 'x' next to use but not inside inner loops.
1416             // For inner loop uses get the preheader area.
1417             x_ctrl = place_near_use(x_ctrl);
1418 
1419             if (n->is_Load()) {
1420               // For loads, add a control edge to a CFG node outside of the loop
1421               // to force them to not combine and return back inside the loop
1422               // during GVN optimization (4641526).
1423               //
1424               // Because we are setting the actual control input, factor in
1425               // the result from get_late_ctrl() so we respect any
1426               // anti-dependences. (6233005).
1427               x_ctrl = dom_lca(late_load_ctrl, x_ctrl);
1428 
1429               // Don't allow the control input to be a CFG splitting node.
1430               // Such nodes should only have ProjNodes as outs, e.g. IfNode
1431               // should only have IfTrueNode and IfFalseNode (4985384).
1432               x_ctrl = find_non_split_ctrl(x_ctrl);
1433 
1434               IdealLoopTree* x_loop = get_loop(x_ctrl);
1435               Node* x_head = x_loop->_head;
1436               if (x_head->is_Loop() && (x_head->is_OuterStripMinedLoop() || x_head->as_Loop()->is_strip_mined()) && is_dominator(n_ctrl, x_head)) {
1437                 // Anti dependence analysis is sometimes too
1438                 // conservative: a store in the outer strip mined loop
1439                 // can prevent a load from floating out of the outer
1440                 // strip mined loop but the load may not be referenced
1441                 // from the safepoint: loop strip mining verification
1442                 // code reports a problem in that case. Make sure the
1443                 // load is not moved in the outer strip mined loop in
1444                 // that case.
1445                 x_ctrl = x_head->as_Loop()->skip_strip_mined()->in(LoopNode::EntryControl);
1446               }
1447               assert(dom_depth(n_ctrl) <= dom_depth(x_ctrl), "n is later than its clone");
1448 
1449               x->set_req(0, x_ctrl);
1450             }
1451             register_new_node(x, x_ctrl);
1452 
1453             // Some institutional knowledge is needed here: 'x' is
1454             // yanked because if the optimizer runs GVN on it all the
1455             // cloned x's will common up and undo this optimization and
1456             // be forced back in the loop.
1457             // I tried setting control edges on the x's to force them to
1458             // not combine, but the matching gets worried when it tries
1459             // to fold a StoreP and an AddP together (as part of an
1460             // address expression) and the AddP and StoreP have
1461             // different controls.
1462             if (!x->is_Load() && !x->is_DecodeNarrowPtr()) _igvn._worklist.yank(x);
1463           }
1464           _igvn.remove_dead_node(n);
1465         }
1466       }
1467     }
1468   }
1469 
1470   try_move_store_after_loop(n);
1471 
1472   // Check for Opaque2's who's loop has disappeared - who's input is in the
1473   // same loop nest as their output.  Remove 'em, they are no longer useful.
1474   if( n_op == Op_Opaque2 &&
1475       n->in(1) != NULL &&
1476       get_loop(get_ctrl(n)) == get_loop(get_ctrl(n->in(1))) ) {
1477     _igvn.replace_node( n, n->in(1) );
1478   }
1479 }

1481 //------------------------------split_if_with_blocks---------------------------
1482 // Check for aggressive application of 'split-if' optimization,
1483 // using basic block level info.
1484 void PhaseIdealLoop::split_if_with_blocks(VectorSet &visited, Node_Stack &nstack) {
1485   Node* root = C->root();
1486   visited.set(root->_idx); // first, mark root as visited
1487   // Do pre-visit work for root
1488   Node* n   = split_if_with_blocks_pre(root);
1489   uint  cnt = n->outcnt();
1490   uint  i   = 0;
1491 
1492   while (true) {
1493     // Visit all children
1494     if (i < cnt) {
1495       Node* use = n->raw_out(i);
1496       ++i;
1497       if (use->outcnt() != 0 && !visited.test_set(use->_idx)) {
1498         // Now do pre-visit work for this use
1499         use = split_if_with_blocks_pre(use);
1500         nstack.push(n, i); // Save parent and next use's index.
1501         n   = use;         // Process all children of current use.
1502         cnt = use->outcnt();
1503         i   = 0;
1504       }
1505     }
1506     else {
1507       // All of n's children have been processed, complete post-processing.
1508       if (cnt != 0 && !n->is_Con()) {
1509         assert(has_node(n), "no dead nodes");
1510         split_if_with_blocks_post(n);
1511       }
1512       if (must_throttle_split_if()) {
1513         nstack.clear();
1514       }
1515       if (nstack.is_empty()) {
1516         // Finished all nodes on stack.
1517         break;
1518       }
1519       // Get saved parent node and next use's index. Visit the rest of uses.
1520       n   = nstack.node();
1521       cnt = n->outcnt();
1522       i   = nstack.index();
1523       nstack.pop();
1524     }
1525   }
1526 }

1535 // Passed in a Phi merging (recursively) some nearly equivalent Bool/Cmps.
1536 // "Nearly" because all Nodes have been cloned from the original in the loop,

1542   uint i;
1543   for (i = 1; i < phi->req(); i++) {
1544     Node *b = phi->in(i);
1545     if (b->is_Phi()) {
1546       _igvn.replace_input_of(phi, i, clone_iff(b->as_Phi(), loop));
1547     } else {
1548       assert(b->is_Bool() || b->Opcode() == Op_Opaque4, "");
1549     }
1550   }
1551 
1552   Node* n = phi->in(1);
1553   Node* sample_opaque = NULL;
1554   Node *sample_bool = NULL;
1555   if (n->Opcode() == Op_Opaque4) {
1556     sample_opaque = n;

1558     assert(sample_bool->is_Bool(), "wrong type");
1559   } else {
1560     sample_bool = n;
1561   }
1562   Node *sample_cmp = sample_bool->in(1);
1563

1566   PhiNode *phi2 = new PhiNode(phi->in(0), Type::TOP);
1567   for (i = 1; i < phi->req(); i++) {

1569     Node *n2 = sample_opaque == NULL ? phi->in(i)->in(1)->in(2) : phi->in(i)->in(1)->in(1)->in(2);

1571     phi2->set_req(i, n2);
1572     phi1->set_type(phi1->type()->meet_speculative(n1->bottom_type()));
1573     phi2->set_type(phi2->type()->meet_speculative(n2->bottom_type()));
1574   }

1576   // Register with optimizer
1577   Node *hit1 = _igvn.hash_find_insert(phi1);
1578   if (hit1) {                   // Hit, toss just made Phi
1579     _igvn.remove_dead_node(phi1); // Remove new phi
1580     assert(hit1->is_Phi(), "" );

1582   } else {                      // Miss
1583     _igvn.register_new_node_with_optimizer(phi1);
1584   }
1585   Node *hit2 = _igvn.hash_find_insert(phi2);
1586   if (hit2) {                   // Hit, toss just made Phi
1587     _igvn.remove_dead_node(phi2); // Remove new phi
1588     assert(hit2->is_Phi(), "" );

1590   } else {                      // Miss

1592   }

1594   set_ctrl(phi1, phi->in(0));
1595   set_ctrl(phi2, phi->in(0));
1596   // Make a new Cmp
1597   Node *cmp = sample_cmp->clone();
1598   cmp->set_req(1, phi1);
1599   cmp->set_req(2, phi2);
1600   _igvn.register_new_node_with_optimizer(cmp);
1601   set_ctrl(cmp, phi->in(0));

1606   _igvn.register_new_node_with_optimizer(b);
1607   set_ctrl(b, phi->in(0));
1608 
1609   if (sample_opaque != NULL) {

1611     opaque->set_req(1, b);
1612     _igvn.register_new_node_with_optimizer(opaque);
1613     set_ctrl(opaque, phi->in(0));
1614     return opaque;
1615   }
1616

1618   return b;

1622 // Passed in a Phi merging (recursively) some nearly equivalent Bool/Cmps.
1623 // "Nearly" because all Nodes have been cloned from the original in the loop,
1624 // but the fall-in edges to the Cmp are different.  Clone bool/Cmp pairs
1625 // through the Phi recursively, and return a Bool.
1626 CmpNode *PhaseIdealLoop::clone_bool( PhiNode *phi, IdealLoopTree *loop ) {
1627   uint i;
1628   // Convert this Phi into a Phi merging Bools
1629   for( i = 1; i < phi->req(); i++ ) {
1630     Node *b = phi->in(i);
1631     if( b->is_Phi() ) {
1632       _igvn.replace_input_of(phi, i, clone_bool( b->as_Phi(), loop ));
1633     } else {
1634       assert( b->is_Cmp() || b->is_top(), "inputs are all Cmp or TOP" );
1635     }
1636   }
1637 
1638   Node *sample_cmp = phi->in(1);
1639 
1640   // Make Phis to merge the Cmp's inputs.
1641   PhiNode *phi1 = new PhiNode( phi->in(0), Type::TOP );
1642   PhiNode *phi2 = new PhiNode( phi->in(0), Type::TOP );
1643   for( uint j = 1; j < phi->req(); j++ ) {
1644     Node *cmp_top = phi->in(j); // Inputs are all Cmp or TOP
1645     Node *n1, *n2;
1646     if( cmp_top->is_Cmp() ) {
1647       n1 = cmp_top->in(1);
1648       n2 = cmp_top->in(2);
1649     } else {
1650       n1 = n2 = cmp_top;
1651     }
1652     phi1->set_req( j, n1 );
1653     phi2->set_req( j, n2 );
1654     phi1->set_type(phi1->type()->meet_speculative(n1->bottom_type()));
1655     phi2->set_type(phi2->type()->meet_speculative(n2->bottom_type()));
1656   }
1657 
1658   // See if these Phis have been made before.
1659   // Register with optimizer
1660   Node *hit1 = _igvn.hash_find_insert(phi1);
1661   if( hit1 ) {                  // Hit, toss just made Phi
1662     _igvn.remove_dead_node(phi1); // Remove new phi
1663     assert( hit1->is_Phi(), "" );
1664     phi1 = (PhiNode*)hit1;      // Use existing phi
1665   } else {                      // Miss
1666     _igvn.register_new_node_with_optimizer(phi1);
1667   }
1668   Node *hit2 = _igvn.hash_find_insert(phi2);
1669   if( hit2 ) {                  // Hit, toss just made Phi
1670     _igvn.remove_dead_node(phi2); // Remove new phi
1671     assert( hit2->is_Phi(), "" );
1672     phi2 = (PhiNode*)hit2;      // Use existing phi
1673   } else {                      // Miss
1674     _igvn.register_new_node_with_optimizer(phi2);
1675   }
1676   // Register Phis with loop/block info
1677   set_ctrl(phi1, phi->in(0));
1678   set_ctrl(phi2, phi->in(0));
1679   // Make a new Cmp
1680   Node *cmp = sample_cmp->clone();
1681   cmp->set_req( 1, phi1 );
1682   cmp->set_req( 2, phi2 );
1683   _igvn.register_new_node_with_optimizer(cmp);
1684   set_ctrl(cmp, phi->in(0));
1685 
1686   assert( cmp->is_Cmp(), "" );
1687   return (CmpNode*)cmp;
1688 }

1701 void PhaseIdealLoop::clone_loop_handle_data_uses(Node* old, Node_List &old_new,
1702                                                  IdealLoopTree* loop, IdealLoopTree* outer_loop,
1703                                                  Node_List*& split_if_set, Node_List*& split_bool_set,
1704                                                  Node_List*& split_cex_set, Node_List& worklist,
1705                                                  uint new_counter, CloneLoopMode mode) {
1706   Node* nnn = old_new[old->_idx];
1707   // Copy uses to a worklist, so I can munge the def-use info
1708   // with impunity.
1709   for (DUIterator_Fast jmax, j = old->fast_outs(jmax); j < jmax; j++)
1710     worklist.push(old->fast_out(j));
1711 
1712   while( worklist.size() ) {
1713     Node *use = worklist.pop();
1714     if (!has_node(use))  continue; // Ignore dead nodes
1715     if (use->in(0) == C->top())  continue;
1716     IdealLoopTree *use_loop = get_loop( has_ctrl(use) ? get_ctrl(use) : use );
1717     // Check for data-use outside of loop - at least one of OLD or USE
1718     // must not be a CFG node.
1719 #ifdef ASSERT
1720     if (loop->_head->as_Loop()->is_strip_mined() && outer_loop->is_member(use_loop) && !loop->is_member(use_loop) && old_new[use->_idx] == NULL) {
1721       Node* sfpt = loop->_head->as_CountedLoop()->outer_safepoint();
1722       assert(mode != IgnoreStripMined, "incorrect cloning mode");
1723       assert((mode == ControlAroundStripMined && use == sfpt) || !use->is_reachable_from_root(), "missed a node");
1724     }
1725 #endif
1726     if (!loop->is_member(use_loop) && !outer_loop->is_member(use_loop) && (!old->is_CFG() || !use->is_CFG())) {
1727 
1728       // If the Data use is an IF, that means we have an IF outside of the
1729       // loop that is switching on a condition that is set inside of the
1730       // loop.  Happens if people set a loop-exit flag; then test the flag
1731       // in the loop to break the loop, then test is again outside of the
1732       // loop to determine which way the loop exited.
1733       // Loop predicate If node connects to Bool node through Opaque1 node.
1734       if (use->is_If() || use->is_CMove() || C->is_predicate_opaq(use) || use->Opcode() == Op_Opaque4) {
1735         // Since this code is highly unlikely, we lazily build the worklist
1736         // of such Nodes to go split.
1737         if (!split_if_set) {
1738           ResourceArea *area = Thread::current()->resource_area();
1739           split_if_set = new Node_List(area);
1740         }
1741         split_if_set->push(use);
1742       }
1743       if (use->is_Bool()) {
1744         if (!split_bool_set) {
1745           ResourceArea *area = Thread::current()->resource_area();
1746           split_bool_set = new Node_List(area);
1747         }
1748         split_bool_set->push(use);
1749       }
1750       if (use->Opcode() == Op_CreateEx) {
1751         if (!split_cex_set) {
1752           ResourceArea *area = Thread::current()->resource_area();
1753           split_cex_set = new Node_List(area);
1754         }
1755         split_cex_set->push(use);
1756       }
1757 
1758 
1759       // Get "block" use is in
1760       uint idx = 0;
1761       while( use->in(idx) != old ) idx++;
1762       Node *prev = use->is_CFG() ? use : get_ctrl(use);
1763       assert(!loop->is_member(get_loop(prev)) && !outer_loop->is_member(get_loop(prev)), "" );
1764       Node *cfg = prev->_idx >= new_counter
1765         ? prev->in(2)
1766         : idom(prev);
1767       if( use->is_Phi() )     // Phi use is in prior block
1768         cfg = prev->in(idx);  // NOT in block of Phi itself
1769       if (cfg->is_top()) {    // Use is dead?
1770         _igvn.replace_input_of(use, idx, C->top());
1771         continue;
1772       }
1773 
1774       // If use is referenced through control edge... (idx == 0)
1775       if (mode == IgnoreStripMined && idx == 0) {
1776         LoopNode *head = loop->_head->as_Loop();
1777         if (head->is_strip_mined() && is_dominator(head->outer_loop_exit(), prev)) {
1778           // That node is outside the inner loop, leave it outside the
1779           // outer loop as well to not confuse verification code.
1780           assert(!loop->_parent->is_member(use_loop), "should be out of the outer loop");
1781           _igvn.replace_input_of(use, 0, head->outer_loop_exit());
1782           continue;
1783         }
1784       }
1785 
1786       while(!outer_loop->is_member(get_loop(cfg))) {
1787         prev = cfg;
1788         cfg = cfg->_idx >= new_counter ? cfg->in(2) : idom(cfg);
1789       }
1790       // If the use occurs after merging several exits from the loop, then
1791       // old value must have dominated all those exits.  Since the same old
1792       // value was used on all those exits we did not need a Phi at this
1793       // merge point.  NOW we do need a Phi here.  Each loop exit value
1794       // is now merged with the peeled body exit; each exit gets its own
1795       // private Phi and those Phis need to be merged here.
1796       Node *phi;
1797       if( prev->is_Region() ) {
1798         if( idx == 0 ) {      // Updating control edge?
1799           phi = prev;         // Just use existing control
1800         } else {              // Else need a new Phi
1801           phi = PhiNode::make( prev, old );
1802           // Now recursively fix up the new uses of old!
1803           for( uint i = 1; i < prev->req(); i++ ) {
1804             worklist.push(phi); // Onto worklist once for each 'old' input
1805           }
1806         }
1807       } else {
1808         // Get new RegionNode merging old and new loop exits
1809         prev = old_new[prev->_idx];
1810         assert( prev, "just made this in step 7" );
1811         if( idx == 0) {      // Updating control edge?
1812           phi = prev;         // Just use existing control
1813         } else {              // Else need a new Phi
1814           // Make a new Phi merging data values properly
1815           phi = PhiNode::make( prev, old );
1816           phi->set_req( 1, nnn );
1817         }
1818       }
1819       // If inserting a new Phi, check for prior hits
1820       if( idx != 0 ) {
1821         Node *hit = _igvn.hash_find_insert(phi);
1822         if( hit == NULL ) {
1823           _igvn.register_new_node_with_optimizer(phi); // Register new phi
1824         } else {                                      // or
1825           // Remove the new phi from the graph and use the hit
1826           _igvn.remove_dead_node(phi);
1827           phi = hit;                                  // Use existing phi
1828         }
1829         set_ctrl(phi, prev);
1830       }
1831       // Make 'use' use the Phi instead of the old loop body exit value
1832       _igvn.replace_input_of(use, idx, phi);
1833       if( use->_idx >= new_counter ) { // If updating new phis
1834         // Not needed for correctness, but prevents a weak assert
1835         // in AddPNode from tripping (when we end up with different
1836         // base & derived Phis that will become the same after
1837         // IGVN does CSE).
1838         Node *hit = _igvn.hash_find_insert(use);
1839         if( hit )             // Go ahead and re-hash for hits.
1840           _igvn.replace_node( use, hit );
1841       }
1842 
1843       // If 'use' was in the loop-exit block, it now needs to be sunk
1844       // below the post-loop merge point.
1845       sink_use( use, prev );
1846     }
1847   }
1848 }

1918       _igvn.register_new_node_with_optimizer(new_le);

1993 //------------------------------clone_loop-------------------------------------
1994 //
1995 //                   C L O N E   A   L O O P   B O D Y
1996 //
1997 // This is the basic building block of the loop optimizations.  It clones an
1998 // entire loop body.  It makes an old_new loop body mapping; with this mapping
1999 // you can find the new-loop equivalent to an old-loop node.  All new-loop
2000 // nodes are exactly equal to their old-loop counterparts, all edges are the
2001 // same.  All exits from the old-loop now have a RegionNode that merges the
2002 // equivalent new-loop path.  This is true even for the normal "loop-exit"
2003 // condition.  All uses of loop-invariant old-loop values now come from (one
2004 // or more) Phis that merge their new-loop equivalents.
2005 //
2006 // This operation leaves the graph in an illegal state: there are two valid
2007 // control edges coming from the loop pre-header to both loop bodies.  I'll
2008 // definitely have to hack the graph after running this transform.
2009 //
2010 // From this building block I will further edit edges to perform loop peeling
2011 // or loop unrolling or iteration splitting (Range-Check-Elimination), etc.
2012 //
2013 // Parameter side_by_size_idom:
2014 //   When side_by_size_idom is NULL, the dominator tree is constructed for
2015 //      the clone loop to dominate the original.  Used in construction of
2016 //      pre-main-post loop sequence.
2017 //   When nonnull, the clone and original are side-by-side, both are
2018 //      dominated by the side_by_side_idom node.  Used in construction of
2019 //      unswitched loops.
2020 void PhaseIdealLoop::clone_loop( IdealLoopTree *loop, Node_List &old_new, int dd,
2021                                 CloneLoopMode mode, Node* side_by_side_idom) {
2022 
2023   LoopNode* head = loop->_head->as_Loop();
2024   head->verify_strip_mined(1);
2025 
2026   if (C->do_vector_loop() && PrintOpto) {
2027     const char* mname = C->method()->name()->as_quoted_ascii();
2028     if (mname != NULL) {
2029       tty->print("PhaseIdealLoop::clone_loop: for vectorize method %s\n", mname);
2030     }
2031   }
2032 
2033   CloneMap& cm = C->clone_map();
2034   Dict* dict = cm.dict();
2035   if (C->do_vector_loop()) {
2036     cm.set_clone_idx(cm.max_gen()+1);
2037 #ifndef PRODUCT
2038     if (PrintOpto) {
2039       tty->print_cr("PhaseIdealLoop::clone_loop: _clone_idx %d", cm.clone_idx());
2040       loop->dump_head();
2041     }
2042 #endif
2043   }
2044 
2045   // Step 1: Clone the loop body.  Make the old->new mapping.
2046   uint i;
2047   for( i = 0; i < loop->_body.size(); i++ ) {
2048     Node *old = loop->_body.at(i);
2049     Node *nnn = old->clone();
2050     old_new.map( old->_idx, nnn );
2051     if (C->do_vector_loop()) {
2052       cm.verify_insert_and_clone(old, nnn, cm.clone_idx());
2053     }
2054     _igvn.register_new_node_with_optimizer(nnn);
2055   }
2056 
2057   IdealLoopTree* outer_loop = (head->is_strip_mined() && mode != IgnoreStripMined) ? get_loop(head->as_CountedLoop()->outer_loop()) : loop;
2058 
2059   // Step 2: Fix the edges in the new body.  If the old input is outside the
2060   // loop use it.  If the old input is INside the loop, use the corresponding
2061   // new node instead.
2062   for( i = 0; i < loop->_body.size(); i++ ) {
2063     Node *old = loop->_body.at(i);
2064     Node *nnn = old_new[old->_idx];
2065     // Fix CFG/Loop controlling the new node
2066     if (has_ctrl(old)) {
2067       set_ctrl(nnn, old_new[get_ctrl(old)->_idx]);
2068     } else {
2069       set_loop(nnn, outer_loop->_parent);
2070       if (old->outcnt() > 0) {
2071         set_idom( nnn, old_new[idom(old)->_idx], dd );
2072       }
2073     }
2074     // Correct edges to the new node
2075     for( uint j = 0; j < nnn->req(); j++ ) {
2076         Node *n = nnn->in(j);
2077         if( n ) {
2078           IdealLoopTree *old_in_loop = get_loop( has_ctrl(n) ? get_ctrl(n) : n );
2079           if( loop->is_member( old_in_loop ) )
2080             nnn->set_req(j, old_new[n->_idx]);
2081         }
2082     }
2083     _igvn.hash_find_insert(nnn);
2084   }
2085 
2086   ResourceArea *area = Thread::current()->resource_area();
2087   Node_List extra_data_nodes(area); // data nodes in the outer strip mined loop
2088   clone_outer_loop(head, mode, loop, outer_loop, dd, old_new, extra_data_nodes);
2089 
2090   // Step 3: Now fix control uses.  Loop varying control uses have already
2091   // been fixed up (as part of all input edges in Step 2).  Loop invariant
2092   // control uses must be either an IfFalse or an IfTrue.  Make a merge
2093   // point to merge the old and new IfFalse/IfTrue nodes; make the use
2094   // refer to this.
2095   Node_List worklist(area);
2096   uint new_counter = C->unique();
2097   for( i = 0; i < loop->_body.size(); i++ ) {
2098     Node* old = loop->_body.at(i);
2099     if( !old->is_CFG() ) continue;
2100 
2101     // Copy uses to a worklist, so I can munge the def-use info
2102     // with impunity.
2103     for (DUIterator_Fast jmax, j = old->fast_outs(jmax); j < jmax; j++)
2104       worklist.push(old->fast_out(j));
2105 
2106     while( worklist.size() ) {  // Visit all uses
2107       Node *use = worklist.pop();

2244 //---------------------- stride_of_possible_iv -------------------------------------
2245 // Looks for an iff/bool/comp with one operand of the compare
2246 // being a cycle involving an add and a phi,
2247 // with an optional truncation (left-shift followed by a right-shift)
2248 // of the add. Returns zero if not an iv.
2249 int PhaseIdealLoop::stride_of_possible_iv(Node* iff) {
2250   Node* trunc1 = NULL;
2251   Node* trunc2 = NULL;
2252   const TypeInt* ttype = NULL;
2253   if (!iff->is_If() || iff->in(1) == NULL || !iff->in(1)->is_Bool()) {
2254     return 0;
2255   }
2256   BoolNode* bl = iff->in(1)->as_Bool();
2257   Node* cmp = bl->in(1);
2258   if (!cmp || (cmp->Opcode() != Op_CmpI && cmp->Opcode() != Op_CmpU)) {
2259     return 0;
2260   }
2261   // Must have an invariant operand
2262   if (is_member(get_loop(iff), get_ctrl(cmp->in(2)))) {
2263     return 0;
2264   }
2265   Node* add2 = NULL;
2266   Node* cmp1 = cmp->in(1);
2267   if (cmp1->is_Phi()) {
2268     // (If (Bool (CmpX phi:(Phi ...(Optional-trunc(AddI phi add2))) )))
2269     Node* phi = cmp1;
2270     for (uint i = 1; i < phi->req(); i++) {
2271       Node* in = phi->in(i);
2272       Node* add = CountedLoopNode::match_incr_with_optional_truncation(in,
2273                                 &trunc1, &trunc2, &ttype);
2274       if (add && add->in(1) == phi) {
2275         add2 = add->in(2);
2276         break;
2277       }
2278     }
2279   } else {
2280     // (If (Bool (CmpX addtrunc:(Optional-trunc((AddI (Phi ...addtrunc...) add2)) )))
2281     Node* addtrunc = cmp1;
2282     Node* add = CountedLoopNode::match_incr_with_optional_truncation(addtrunc,
2283                                 &trunc1, &trunc2, &ttype);
2284     if (add && add->in(1)->is_Phi()) {
2285       Node* phi = add->in(1);
2286       for (uint i = 1; i < phi->req(); i++) {
2287         if (phi->in(i) == addtrunc) {
2288           add2 = add->in(2);
2289           break;
2290         }
2291       }
2292     }
2293   }
2294   if (add2 != NULL) {
2295     const TypeInt* add2t = _igvn.type(add2)->is_int();
2296     if (add2t->is_con()) {
2297       return add2t->get_con();
2298     }
2299   }
2300   return 0;

2468 //------------------------------ insert_cmpi_loop_exit -------------------------------------
2469 // Clone a signed compare loop exit from an unsigned compare and
2470 // insert it before the unsigned cmp on the stay-in-loop path.
2471 // All new nodes inserted in the dominator tree between the original
2472 // if and it's projections.  The original if test is replaced with
2473 // a constant to force the stay-in-loop path.
2474 //
2475 // This is done to make sure that the original if and it's projections
2476 // still dominate the same set of control nodes, that the ctrl() relation
2477 // from data nodes to them is preserved, and that their loop nesting is
2478 // preserved.
2479 //
2480 // before
2481 //          if(i <u limit)    unsigned compare loop exit
2482 //         /       |
2483 //        v        v
2484 //   exit-proj   stay-in-loop-proj

2486 // after
2487 //          if(stay-in-loop-const)  original if
2488 //         /       |
2489 //        /        v
2490 //       /  if(i <  limit)    new signed test
2491 //      /  /       |
2492 //     /  /        v
2493 //    /  /  if(i <u limit)    new cloned unsigned test
2494 //   /  /   /      |
2495 //   v  v  v       |
2496 //    region       |
2497 //        |        |
2498 //      dum-if     |
2499 //     /  |        |
2500 // ether  |        |
2501 //        v        v
2502 //   exit-proj   stay-in-loop-proj

2504 IfNode* PhaseIdealLoop::insert_cmpi_loop_exit(IfNode* if_cmpu, IdealLoopTree *loop) {

2511   if (cmpu->Opcode() != Op_CmpU) return NULL;

2513   if (stride == 0) return NULL;

2515   Node* lp_proj = stay_in_loop(if_cmpu, loop);
2516   guarantee(lp_proj != NULL, "null loop node");
2517 
2518   ProjNode* lp_continue = lp_proj->as_Proj();

2560 //------------------------------ scheduled_nodelist -------------------------------------
2561 // Create a post order schedule of nodes that are in the
2562 // "member" set.  The list is returned in "sched".
2563 // The first node in "sched" is the loop head, followed by
2564 // nodes which have no inputs in the "member" set, and then
2565 // followed by the nodes that have an immediate input dependence
2566 // on a node in "sched".
2567 void PhaseIdealLoop::scheduled_nodelist( IdealLoopTree *loop, VectorSet& member, Node_List &sched ) {
2568 
2569   assert(member.test(loop->_head->_idx), "loop head must be in member set");
2570   Arena *a = Thread::current()->resource_area();
2571   VectorSet visited(a);
2572   Node_Stack nstack(a, loop->_body.size());
2573 
2574   Node* n  = loop->_head;  // top of stack is cached in "n"
2575   uint idx = 0;
2576   visited.set(n->_idx);
2577 
2578   // Initially push all with no inputs from within member set
2579   for(uint i = 0; i < loop->_body.size(); i++ ) {
2580     Node *elt = loop->_body.at(i);
2581     if (member.test(elt->_idx)) {
2582       bool found = false;
2583       for (uint j = 0; j < elt->req(); j++) {
2584         Node* def = elt->in(j);
2585         if (def && member.test(def->_idx) && def != elt) {
2586           found = true;
2587           break;
2588         }
2589       }
2590       if (!found && elt != loop->_head) {
2591         nstack.push(n, idx);
2592         n = elt;
2593         assert(!visited.test(n->_idx), "not seen yet");
2594         visited.set(n->_idx);
2595       }
2596     }
2597   }
2598 
2599   // traverse out's that are in the member set
2600   while (true) {
2601     if (idx < n->outcnt()) {
2602       Node* use = n->raw_out(idx);
2603       idx++;
2604       if (!visited.test_set(use->_idx)) {
2605         if (member.test(use->_idx)) {
2606           nstack.push(n, idx);
2607           n = use;
2608           idx = 0;
2609         }
2610       }
2611     } else {
2612       // All outputs processed
2613       sched.push(n);
2614       if (nstack.is_empty()) break;
2615       n   = nstack.node();
2616       idx = nstack.index();
2617       nstack.pop();
2618     }
2619   }
2620 }

3247           new_phi_cnt++;
3248         }
3249       }
3250     }
3251     if (incr) i++;
3252   }

3254   estimate += cloned_for_outside_use + new_phi_cnt;
3255   bool exceed_node_budget = !may_require_nodes(estimate);
3256   bool exceed_phi_limit = new_phi_cnt > old_phi_cnt + PartialPeelNewPhiDelta;
3257 
3258   if (exceed_node_budget || exceed_phi_limit) {
3259 #ifndef PRODUCT

3262                     new_phi_cnt, old_phi_cnt, new_peel_if != NULL?'T':'F');
3263     }
3264 #endif
3265     if (new_peel_if != NULL) {
3266       remove_cmpi_loop_exit(new_peel_if, loop);
3267     }
3268     // Inhibit more partial peeling on this loop
3269     assert(!head->is_partial_peel_loop(), "not partial peeled");
3270     head->mark_partial_peel_failed();
3271     if (cloned_for_outside_use > 0) {
3272       // Terminate this round of loop opts because
3273       // the graph outside this loop was changed.
3274       C->set_major_progress();
3275       return true;
3276     }
3277     return false;
3278   }

3283   // the nodes being moved (sinked) from the peel region.

3288   _igvn.replace_input_of(first_not_peeled, 0, new_head);

3291   not_peel.set(new_head->_idx);
3292   set_idom(new_head, last_peel, dom_depth(first_not_peeled));
3293   set_idom(first_not_peeled, new_head, dom_depth(first_not_peeled));

3295   while (sink_list.size() > 0) {

3297     set_ctrl(n, new_head);
3298   }

3300   assert(is_valid_loop_partition(loop, peel, peel_list, not_peel), "bad partition");
3301 
3302   clone_loop(loop, old_new, dd, IgnoreStripMined);
3303 
3304   const uint clone_exit_idx = 1;
3305   const uint orig_exit_idx  = 2;
3306   assert(is_valid_clone_loop_form(loop, peel_list, orig_exit_idx, clone_exit_idx), "bad clone loop");
3307 
3308   Node* head_clone             = old_new[head->_idx];

3314   for (uint i = 0; i < peel_list.size(); i++) {

3332             // or "use" is in the entry boundary (a phi) of the peel set

3335

3339                 Node* use_clone = old_new[use->_idx];

3342               }
3343             } else {
3344               assert(is_valid_clone_loop_exit_use(loop, use, orig_exit_idx), "clone loop format");

3346               Node* lp_if = use_c->in(orig_exit_idx)->in(0);
3347               if (not_peel.test(lp_if->_idx)) {

3359 
3360   // Redirect control to the new loop head if a cloned node in
3361   // the not_peeled region has control that points into the peeled region.
3362   // This necessary because the cloned peeled region will be outside
3363   // the loop.
3364   //                            from    to
3365   //          cloned-peeled    <---+

3367   //          cloned-not_peeled  in(0)    in(0)
3368   //          orig-peeled

3371     Node *n = loop->_body.at(i);
3372     if (!n->is_CFG()           && n->in(0) != NULL        &&
3373         not_peel.test(n->_idx) && peel.test(n->in(0)->_idx)) {
3374       Node* n_clone = old_new[n->_idx];
3375       _igvn.replace_input_of(n_clone, 0, new_head_clone);
3376     }
3377   }

3380   _igvn.replace_input_of(new_head_clone, LoopNode::LoopBackControl, last_peel);
3381

3383   _igvn.rehash_node_delayed(new_head);                     // Multiple edge updates:

3385   new_head->set_req(LoopNode::LoopBackControl, C->top());  //   multiple replace_input_of calls

3389   // clone head's back-branch

3391   head->set_req(LoopNode::EntryControl,    head_clone->in(LoopNode::LoopBackControl));
3392   head->set_req(LoopNode::LoopBackControl, C->top());

3396   for (DUIterator_Fast kmax, k = head->fast_outs(kmax); k < kmax; k++) {
3397     Node* use = head->fast_out(k);
3398     if (use->is_Phi() && use->outcnt() > 0) {
3399       Node* use_clone = old_new[use->_idx];
3400       _igvn.rehash_node_delayed(use); // Multiple edge updates
3401       use->set_req(LoopNode::EntryControl,    use_clone->in(LoopNode::LoopBackControl));
3402       use->set_req(LoopNode::LoopBackControl, C->top());
3403       _igvn.replace_input_of(use_clone, LoopNode::LoopBackControl, C->top());
3404     }
3405   }
3406 
3407   // Step 4: update dominator tree and dominator depth
3408 
3409   set_idom(head, orig_tail_clone, dd);
3410   recompute_dom_depth();
3411 
3412   // Inhibit more partial peeling on this loop
3413   new_head_clone->set_partial_peel_loop();
3414   C->set_major_progress();
3415   loop->record_for_igvn();
3416 
3417 #ifndef PRODUCT
3418   if (TracePartialPeeling) {
3419     tty->print_cr("\nafter partial peel one iteration");
3420     Node_List wl(area);
3421     Node* t = last_peel;
3422     while (true) {
3423       wl.push(t);
3424       if (t == head_clone) break;
3425       t = idom(t);
3426     }
3427     while (wl.size() > 0) {

3430       else if (tt == new_head_clone) tty->print_cr("new head");
3431       else if (tt == head_clone) tty->print_cr("clone head");
3432       tt->dump();
3433     }
3434   }
3435 #endif

3439 //------------------------------reorg_offsets----------------------------------
3440 // Reorganize offset computations to lower register pressure.  Mostly
3441 // prevent loop-fallout uses of the pre-incremented trip counter (which are
3442 // then alive with the post-incremented trip counter forcing an extra
3443 // register move)
3444 void PhaseIdealLoop::reorg_offsets(IdealLoopTree *loop) {
3445   // Perform it only for canonical counted loops.
3446   // Loop's shape could be messed up by iteration_split_impl.
3447   if (!loop->_head->is_CountedLoop())
3448     return;
3449   if (!loop->_head->as_Loop()->is_valid_counted_loop())
3450     return;
3451 
3452   CountedLoopNode *cl = loop->_head->as_CountedLoop();
3453   CountedLoopEndNode *cle = cl->loopexit();
3454   Node *exit = cle->proj_out(false);
3455   Node *phi = cl->phi();
3456 
3457   // Check for the special case when using the pre-incremented trip-counter on
3458   // the fall-out  path (forces the pre-incremented  and post-incremented trip
3459   // counter to be live  at the same time).  Fix this by  adjusting to use the
3460   // post-increment trip counter.
3461 
3462   bool progress = true;
3463   while (progress) {
3464     progress = false;
3465     for (DUIterator_Fast imax, i = phi->fast_outs(imax); i < imax; i++) {
3466       Node* use = phi->fast_out(i);   // User of trip-counter
3467       if (!has_ctrl(use))  continue;
3468       Node *u_ctrl = get_ctrl(use);
3469       if (use->is_Phi()) {
3470         u_ctrl = NULL;
3471         for (uint j = 1; j < use->req(); j++)
3472           if (use->in(j) == phi)
3473             u_ctrl = dom_lca(u_ctrl, use->in(0)->in(j));
3474       }
3475       IdealLoopTree *u_loop = get_loop(u_ctrl);
3476       // Look for loop-invariant use
3477       if (u_loop == loop) continue;
3478       if (loop->is_member(u_loop)) continue;
3479       // Check that use is live out the bottom.  Assuming the trip-counter
3480       // update is right at the bottom, uses of of the loop middle are ok.
3481       if (dom_lca(exit, u_ctrl) != exit) continue;
3482       // Hit!  Refactor use to use the post-incremented tripcounter.
3483       // Compute a post-increment tripcounter.
3484       Node* c = exit;
3485       if (cl->is_strip_mined()) {
3486         IdealLoopTree* outer_loop = get_loop(cl->outer_loop());
3487         if (!outer_loop->is_member(u_loop)) {
3488           c = cl->outer_loop_exit();
3489         }
3490       }
3491       Node *opaq = new Opaque2Node(C, cle->incr());
3492       register_new_node(opaq, c);
3493       Node *neg_stride = _igvn.intcon(-cle->stride_con());
3494       set_ctrl(neg_stride, C->root());
3495       Node *post = new AddINode(opaq, neg_stride);
3496       register_new_node(post, c);
3497       _igvn.rehash_node_delayed(use);
3498       for (uint j = 1; j < use->req(); j++) {
3499         if (use->in(j) == phi)
3500           use->set_req(j, post);
3501       }
3502       // Since DU info changed, rerun loop
3503       progress = true;
3504       break;
3505     }
3506   }
