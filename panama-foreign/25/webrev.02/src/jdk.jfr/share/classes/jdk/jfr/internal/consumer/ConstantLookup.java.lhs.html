24 package jdk.jfr.internal.consumer;
25 
26 import jdk.jfr.internal.Type;
27 
28 final class ConstantLookup {
29     private final Type type;
30     private ConstantMap current;
31     private ConstantMap previous = ConstantMap.EMPTY;
32 
33     ConstantLookup(ConstantMap current, Type type) {
34         this.current = current;
35         this.type = type;
36     }
37 
38     public Type getType() {
39         return type;
40     }
41 
42     public ConstantMap getLatestPool() {
43         return current;
44     }
45 
46     public void newPool() {
47         previous = current;
48         current = new ConstantMap(current.factory, current.name);
49     }
50 
51     public Object getPreviousResolved(long key) {
52         return previous.getResolved(key);
53     }
54 
55     public Object getCurrentResolved(long key) {
56         return current.getResolved(key);
57     }
58 
59     public Object getCurrent(long key) {
60         return current.get(key);
61     }
62 }
