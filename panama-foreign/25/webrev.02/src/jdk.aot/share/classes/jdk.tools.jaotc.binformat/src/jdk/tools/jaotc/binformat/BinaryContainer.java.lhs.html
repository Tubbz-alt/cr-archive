 55  * ELFContainer for Linux and Solaris operating systems or MachOContainer for Mac OS or PEContainer

 60  * contents of a binary as the first step to create a binary representation of code generated by a

283     /**
284      * Allocates a {@code BinaryContainer} object whose content will be generated in a file with the
285      * prefix {@code prefix}. It also initializes internal code container, symbol table and
286      * relocation tables.
287      *
288      * @param graalOptions
289      */
290     public BinaryContainer(OptionValues graalOptions, GraalHotSpotVMConfig graalHotSpotVMConfig, GraphBuilderConfiguration graphBuilderConfig, int gc, String jvmVersion) {
291         this.graalOptions = graalOptions;
292 
293         this.codeSegmentSize = graalHotSpotVMConfig.codeSegmentSize;
294         if (codeSegmentSize < 1 || codeSegmentSize > 1024) {
295             throw new InternalError("codeSegmentSize is not in range [1, 1024] bytes: (" + codeSegmentSize + "), update JPECoffRelocObject");
296         }
297         if ((codeSegmentSize & (codeSegmentSize - 1)) != 0) {
298             throw new InternalError("codeSegmentSize is not power of 2: (" + codeSegmentSize + "), update JPECoffRelocObject");
299         }
300 
301         this.codeEntryAlignment = graalHotSpotVMConfig.codeEntryAlignment;
302 
303         this.threadLocalHandshakes = graalHotSpotVMConfig.threadLocalHandshakes;
304 
305         // Section unique name is limited to 8 characters due to limitation on Windows.
306         // Name could be longer but only first 8 characters are stored on Windows.
307 
308         // read only, code
309         codeContainer = new CodeContainer(".text", this);
310 
311         // read only, info
312         headerContainer = new HeaderContainer(jvmVersion, new ReadOnlyDataContainer(".header", this));
313         configContainer = new ReadOnlyDataContainer(".config", this);
314         metaspaceNamesContainer = new ReadOnlyDataContainer(".meta.names", this);
315         methodsOffsetsContainer = new ReadOnlyDataContainer(".meth.offsets", this);
316         klassesOffsetsContainer = new ReadOnlyDataContainer(".kls.offsets", this);
317         klassesDependenciesContainer = new ReadOnlyDataContainer(".kls.dependencies", this);
318 
319         stubsOffsetsContainer = new ReadOnlyDataContainer(".stubs.offsets", this);
320         codeSegmentsContainer = new ReadOnlyDataContainer(".code.segments", this);
321         constantDataContainer = new ReadOnlyDataContainer(".meth.constdata", this);
322         methodMetadataContainer = new ReadOnlyDataContainer(".meth.metadata", this);
323 
324         // writable sections
325         oopGotContainer = new ByteContainer(".oop.got", this);
326         klassesGotContainer = new ByteContainer(".kls.got", this);
327         countersGotContainer = new ByteContainer(".cnt.got", this);
328         metadataGotContainer = new ByteContainer(".meta.got", this);
329         methodStateContainer = new ByteContainer(".meth.state", this);
330         extLinkageGOTContainer = new ByteContainer(".got.linkage", this);
331 
332         addGlobalSymbols();
333 
334         recordConfiguration(graalHotSpotVMConfig, graphBuilderConfig, gc);
335     }

337     private void recordConfiguration(GraalHotSpotVMConfig graalHotSpotVMConfig, GraphBuilderConfiguration graphBuilderConfig, int gc) {
338         // @Checkstyle: stop
339         // @formatter:off
340         boolean[] booleanFlags = { graalHotSpotVMConfig.cAssertions, // Debug VM
341                                    graalHotSpotVMConfig.useCompressedOops,
342                                    graalHotSpotVMConfig.useCompressedClassPointers,
343                                    graalHotSpotVMConfig.useTLAB,
344                                    graalHotSpotVMConfig.useBiasedLocking,
345                                    TieredAOT.getValue(graalOptions),
346                                    graalHotSpotVMConfig.enableContended,
347                                    graalHotSpotVMConfig.restrictContended,
348                                    graphBuilderConfig.omitAssertions(),
349         };
350 
351         int[] intFlags         = { graalHotSpotVMConfig.getOopEncoding().getShift(),
352                                    graalHotSpotVMConfig.getKlassEncoding().getShift(),
353                                    graalHotSpotVMConfig.contendedPaddingWidth,
354                                    1 << graalHotSpotVMConfig.logMinObjAlignment(),
355                                    graalHotSpotVMConfig.codeSegmentSize,
356                                    gc
357         };
358         // @formatter:on
359         // @Checkstyle: resume
360 
361         if (JavaVersionUtil.JAVA_SPEC < 14) {
362             // See JDK-8220049. Thread local handshakes are on by default since JDK14, the command line option has been removed.
363             booleanFlags = Arrays.copyOf(booleanFlags, booleanFlags.length + 1);
364             booleanFlags[booleanFlags.length - 1] = graalHotSpotVMConfig.threadLocalHandshakes;
365         }
366 
367         byte[] booleanFlagsAsBytes = flagsToByteArray(booleanFlags);
















368         int size0 = configContainer.getByteStreamSize();
369 
370         // @formatter:off
371         int computedSize = booleanFlagsAsBytes.length * Byte.BYTES    + // size of boolean flags
372                            intFlags.length            * Integer.BYTES + // size of int flags
373                            Integer.BYTES;                               // size of the "computedSize"
374 
375         configContainer.appendInt(computedSize).
376                         appendInts(intFlags).
377                         appendBytes(booleanFlagsAsBytes);
378         // @formatter:on
379 
380         int size = configContainer.getByteStreamSize() - size0;
381         assert size == computedSize;
382     }

384     private static byte[] flagsToByteArray(boolean[] flags) {
385         byte[] byteArray = new byte[flags.length];
386         for (int i = 0; i < flags.length; ++i) {
387             byteArray[i] = boolToByte(flags[i]);
388         }
389         return byteArray;
390     }
391 
392     private static byte boolToByte(boolean flag) {
393         return (byte) (flag ? 1 : 0);
394     }

537      * Create a platform-specific binary file representing the content of the
538      * {@code BinaryContainer} object.
539      *
540      * This method is called after creating and performing any necessary changes to the contents of
541      * code stream, symbol tables and relocation tables is completely finalized
542      *
543      * @param outputFileName name of output file
544      *
545      * @throws IOException in case of file creation failure
546      */
547     public void createBinary(String outputFileName) throws IOException {
548         String osName = System.getProperty("os.name");
549         switch (osName) {
550             case "Linux":
551             case "SunOS":
552                 JELFRelocObject elfobj = JELFRelocObject.newInstance(this, outputFileName);
553                 elfobj.createELFRelocObject(relocationTable, symbolTable.values());
554                 break;
555             case "Mac OS X":
556                 JMachORelocObject machobj = new JMachORelocObject(this, outputFileName);
557                 machobj.createMachORelocObject(relocationTable, symbolTable.values());
558                 break;
559             default:
560                 if (osName.startsWith("Windows")) {
561                     JPECoffRelocObject pecoffobj = new JPECoffRelocObject(this, outputFileName);
562                     pecoffobj.createPECoffRelocObject(relocationTable, symbolTable.values());
563                     break;
564                 } else {
565                     throw new InternalError("Unsupported platform: " + osName);
566                 }
567         }
568     }

596      * Add relocation entry for {@code symName}. Multiple relocation entries for a given symbol may
597      * exist.
598      *
599      * @param info relocation information to be added

601     public void addRelocation(Relocation info) {
602         // System.out.println("# Relocation [" + info.getSymbol() + "] [" + info.getOffset() + "] ["
603         // +
604         // info.getSection().getContainerName() + "] [" + info.getSymbol().getName() + "] [" +

618         }

681     public int addMetaspaceName(String name) {
682         Integer value = metaspaceNames.get(name);
683         if (value != null) {
684             return value.intValue();
685         }
686         // Get the current length of the stubsNameContainer
687         // align on 8-byte boundary
688         int nameOffset = alignUp(metaspaceNamesContainer, 8);
689 
690         try {
691             // Add the name of the symbol to the .stubs.names section
692             // Modify them to sequence of utf8 strings with length:
693             // "<u2_size>Ljava/lang/ThreadGroup;<u2_size>addUnstarted<u2_size>()V"
694             ByteArrayOutputStream bout = new ByteArrayOutputStream();
695             DataOutputStream out = new DataOutputStream(bout);
696             int len = name.length();
697             if (name.startsWith("Stub")) { // Stub
698                 out.writeUTF(name);
699             } else { // Method or Klass
700                 int parenthesesIndex = name.lastIndexOf('(', len - 1);
701                 if (parenthesesIndex > 0) {  // Method name
702                     int dotIndex = name.lastIndexOf('.', parenthesesIndex - 1);
703                     assert dotIndex > 0 : "method's full name should have '.' : " + name;
704                     String klassName = name.substring(0, dotIndex);
705                     out.writeUTF(klassName);
706                     String methodName = name.substring(dotIndex + 1, parenthesesIndex);
707                     out.writeUTF(methodName);
708                     String signature = name.substring(parenthesesIndex, len);
709                     out.writeUTF(signature);
710                 } else {
711                     out.writeUTF(name); // Klass
712                 }
713             }
714             out.writeShort(0); // Terminate by 0.
715             byte[] b = bout.toByteArray();
716             metaspaceNamesContainer.appendBytes(b, 0, b.length);
717 
718             metaspaceNames.put(name, nameOffset);
719             return nameOffset;
720         } catch (IOException e) {
721             throw new InternalError("Failed to append bytes to stubs sections", e);
722         }
723     }

817     public static int alignUp(ByteContainer container, int alignment) {
818         if (Integer.bitCount(alignment) != 1) {
819             throw new IllegalArgumentException("Must be a power of 2");
820         }
821         int offset = container.getByteStreamSize();
822         int aligned = (offset + (alignment - 1)) & -alignment;
823         if (aligned < offset || (aligned & (alignment - 1)) != 0) {
824             throw new RuntimeException("Error aligning: " + offset + " -> " + aligned);
825         }
826         if (aligned != offset) {
827             int nullArraySz = aligned - offset;
828             byte[] nullArray = new byte[nullArraySz];
829             container.appendBytes(nullArray, 0, nullArraySz);
830             offset = aligned;
831         }
832         return offset;
833     }

835     public void addCodeSegments(int start, int end) {
836         assert (start % codeSegmentSize) == 0 : "not aligned code";
837         int currentOffset = codeSegmentsContainer.getByteStreamSize();
838         int offset = start / codeSegmentSize;
839         int emptySize = offset - currentOffset;
840         // add empty segments if needed
841         if (emptySize > 0) {
842             byte[] emptyArray = new byte[emptySize];
843             for (int i = 0; i < emptySize; i++) {
844                 emptyArray[i] = (byte) 0xff;
845             }
846             appendBytes(codeSegmentsContainer, emptyArray, 0, emptySize);
847         }
848         int alignedEnd = (end + (codeSegmentSize - 1)) & -codeSegmentSize;
849         int segmentsCount = (alignedEnd / codeSegmentSize) - offset;
850         byte[] segments = new byte[segmentsCount];
851         int idx = 0;
852         for (int i = 0; i < segmentsCount; i++) {
853             segments[i] = (byte) idx;
854             idx = (idx == 0xfe) ? 1 : (idx + 1);
855         }
856         appendBytes(codeSegmentsContainer, segments, 0, segmentsCount);
857     }
