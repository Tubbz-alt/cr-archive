41 /**
42  * Intrinsic for closing a {@linkplain BeginLockScopeNode scope} binding a stack-based lock with an
43  * object.
44  */
45 @NodeInfo(allowedUsageTypes = Memory, cycles = CYCLES_0, size = SIZE_0)
46 public final class EndLockScopeNode extends AbstractMemoryCheckpoint implements LIRLowerable, MonitorExit, SingleMemoryKill {
47     public static final NodeClass<EndLockScopeNode> TYPE = NodeClass.create(EndLockScopeNode.class);
48 
49     public EndLockScopeNode() {
50         super(TYPE, StampFactory.forVoid());
51     }
52 
53     @Override
54     public boolean hasSideEffect() {
55         return false;
56     }
57 
58     @Override
59     public LocationIdentity getKilledLocationIdentity() {
60         return LocationIdentity.any();
61     }
62 
63     @Override
64     public void generate(NodeLIRBuilderTool gen) {
65     }
66 
67     @NodeIntrinsic
68     public static native void endLockScope();
69 }
