25 package org.graalvm.compiler.nodes.memory;
26 
27 import jdk.internal.vm.compiler.word.LocationIdentity;
28 
29 /**
30  * This interface marks nodes that access some memory location, and that have an edge to the last
31  * node that kills this location.
32  */
33 public interface MemoryAccess {
34 
35     LocationIdentity getLocationIdentity();
36 
37     /**
38      *
39      * @return a {@linkplain MemoryNode} that represents the last memory state in the memory graph
40      *         for the {@linkplain LocationIdentity} returned by
41      *         {@linkplain MemoryAccess#getLocationIdentity()}
42      */
43     MemoryNode getLastLocationAccess();
44 
45     /**
46      * @param lla the {@link MemoryNode} that represents the last kill of the
47      *            {@linkplain LocationIdentity} returned by
48      *            {@linkplain MemoryAccess#getLocationIdentity()}
49      */
50     void setLastLocationAccess(MemoryNode lla);
51 
52 }
