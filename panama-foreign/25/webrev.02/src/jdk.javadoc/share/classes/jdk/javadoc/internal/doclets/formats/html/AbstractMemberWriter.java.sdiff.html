201     protected CharSequence makeSpace(int len) {
202         if (len <= 0) {
203             return "";
204         }
205         StringBuilder sb = new StringBuilder(len);
206         for (int i = 0; i < len; i++) {
207             sb.append(' ');
208         }
209         return sb;
210     }
