24 package jdk.test.lib.artifacts;
25 

26 import java.nio.file.Path;
27 import java.nio.file.Paths;
28 
29 public class DefaultArtifactManager implements ArtifactManager {
30     @Override
31     public Path resolve(Artifact artifact) throws ArtifactResolverException {
32         return resolve(artifact.name());
33     }
34 
35     public Path resolve(String name) throws ArtifactResolverException {
36         String location = System.getProperty(artifactProperty(name));
37         if (location == null) {
38             throw new ArtifactResolverException("Couldn't automatically resolve dependency for " + name + "\n" +

39                     "Please specify the location using " + artifactProperty(name));
40         }
41         return Paths.get(location);
42     }

43     private static String artifactProperty(String name) {
44         return "jdk.test.lib.artifacts." + name;
45     }
46 }
