195 ################################################################################
196 # libstlport.so.1 is needed for running gtest on Solaris. Find it to
197 # redistribute it in the test image.
198 ################################################################################
199 AC_DEFUN_ONCE([LIB_SETUP_SOLARIS_STLPORT],
200 [
201   if test "$OPENJDK_TARGET_OS" = "solaris" && test "x$BUILD_GTEST" = "xtrue"; then
202     # Find the root of the Solaris Studio installation from the compiler path
203     SOLARIS_STUDIO_DIR="$(dirname $CC)/.."
204     STLPORT_LIB="$SOLARIS_STUDIO_DIR/lib/stlport4$OPENJDK_TARGET_CPU_ISADIR/libstlport.so.1"
205     AC_MSG_CHECKING([for libstlport.so.1])
206     if ! test -f "$STLPORT_LIB" && test "x$OPENJDK_TARGET_CPU_ISADIR" = "x/sparcv9"; then
207       # SS12u3 has libstlport under 'stlport4/v9' instead of 'stlport4/sparcv9'
208       STLPORT_LIB="$SOLARIS_STUDIO_DIR/lib/stlport4/v9/libstlport.so.1"
209     fi
210     if test -f "$STLPORT_LIB"; then
211       AC_MSG_RESULT([yes, $STLPORT_LIB])
212       BASIC_FIXUP_PATH([STLPORT_LIB])
213     else
214       AC_MSG_RESULT([no, not found at $STLPORT_LIB])
215       AC_MSG_ERROR([Failed to find libstlport.so.1, cannot build Hotspot gtests])
216     fi
217     AC_SUBST(STLPORT_LIB)
218   fi
219 ])
