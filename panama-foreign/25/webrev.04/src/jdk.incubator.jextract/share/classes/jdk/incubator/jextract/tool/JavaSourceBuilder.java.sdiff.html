126     private void addLayout(MemoryLayout l) {
127         if (l instanceof ValueLayout) {
128             boolean found = false;
129             for (Field fs : MemoryLayouts.SysV.class.getDeclaredFields()) {
130                 try {
131                     MemoryLayout constant = (MemoryLayout)fs.get(null);
132                     if (l.name().isPresent()) {
133                         constant = constant.withName(l.name().get());
134                     }
135                     if (constant.equals(l)) {
136                         found = true;
137                         sb.append("MemoryLayouts.SysV." + fs.getName());
138                         break;
139                     }
140                 } catch (ReflectiveOperationException ex) {
141                     throw new AssertionError(ex);
142                 }
143             }
144             if (!found) {
145                 throw new AssertionError("Should not get here: " + l);
146             }

147         } else if (l instanceof SequenceLayout) {
148             sb.append("MemoryLayout.ofSequence(");
149             if (((SequenceLayout) l).elementCount().isPresent()) {
150                 sb.append(((SequenceLayout) l).elementCount().getAsLong() + ", ");
151             }
152             addLayout(((SequenceLayout) l).elementLayout());
153             sb.append(")");
154         } else if (l instanceof GroupLayout) {
155             if (((GroupLayout) l).isStruct()) {
156                 sb.append("MemoryLayout.ofStruct(\n");





157             } else {
158                 sb.append("MemoryLayout.ofUnion(\n");
159             }
160             incrAlign();
161             String delim = "";
162             for (MemoryLayout e : ((GroupLayout) l).memberLayouts()) {
163                 sb.append(delim);









164                 indent();
165                 addLayout(e);
166                 delim = ",\n";
167             }
168             sb.append("\n");
169             decrAlign();
170             indent();
171             sb.append(")");
172         } else {
173             //padding
174             sb.append("MemoryLayout.ofPaddingBits(" + l.bitSize() + ")");
175         }
176         if (l.name().isPresent()) {
177             sb.append(".withName(\"" +  l.name().get() + "\")");
178         }
179     }

131     private void addLayout(MemoryLayout l) {
132         if (l instanceof ValueLayout) {
133             SystemABI.Type type = l.abiType().orElseThrow(()->new AssertionError("Should not get here: " + l));
134             sb.append(switch (type) {
135                 case BOOL -> "C_BOOL";
136                 case SIGNED_CHAR -> "C_SCHAR";
137                 case UNSIGNED_CHAR -> "C_UCHAR";
138                 case CHAR -> "C_CHAR";
139                 case SHORT -> "C_SHORT";
140                 case UNSIGNED_SHORT -> "C_USHORT";
141                 case INT -> "C_INT";
142                 case UNSIGNED_INT -> "C_UINT";
143                 case LONG -> "C_LONG";
144                 case UNSIGNED_LONG -> "C_ULONG";
145                 case LONG_LONG -> "C_LONGLONG";
146                 case UNSIGNED_LONG_LONG -> "C_ULONGLONG";
147                 case FLOAT -> "C_FLOAT";
148                 case DOUBLE -> "C_DOUBLE";
149                 case LONG_DOUBLE -> "C_LONGDOUBLE";
150                 case POINTER -> "C_POINTER";
151                 default -> { throw new RuntimeException("should not reach here: " + type); }
152             });
153         } else if (l instanceof SequenceLayout) {
154             sb.append("MemoryLayout.ofSequence(");
155             if (((SequenceLayout) l).elementCount().isPresent()) {
156                 sb.append(((SequenceLayout) l).elementCount().getAsLong() + ", ");
157             }
158             addLayout(((SequenceLayout) l).elementLayout());
159             sb.append(")");
160         } else if (l instanceof GroupLayout) {
161             SystemABI.Type type = l.abiType().orElse(null);
162             if (type == SystemABI.Type.COMPLEX_LONG_DOUBLE) {
163                 if (!ABI.equals(SystemABI.ABI_SYSV)) {
164                     throw new RuntimeException("complex long double is supported only for SysV ABI");
165                 } else {
166                     sb.append("C_COMPLEX_LONGDOUBLE");
167                 }
168             } else {
169                 if (((GroupLayout) l).isStruct()) {
170                     sb.append("MemoryLayout.ofStruct(\n");
171                 } else {
172                     sb.append("MemoryLayout.ofUnion(\n");
173                 }
174                 incrAlign();
175                 String delim = "";
176                 for (MemoryLayout e : ((GroupLayout) l).memberLayouts()) {
177                     sb.append(delim);
178                     indent();
179                     addLayout(e);
180                     delim = ",\n";
181                 }
182                 sb.append("\n");
183                 decrAlign();
184                 indent();
185                 sb.append(")");

186             }




187         } else {
188             //padding
189             sb.append("MemoryLayout.ofPaddingBits(" + l.bitSize() + ")");
190         }
191         if (l.name().isPresent()) {
192             sb.append(".withName(\"" +  l.name().get() + "\")");
193         }
194     }
