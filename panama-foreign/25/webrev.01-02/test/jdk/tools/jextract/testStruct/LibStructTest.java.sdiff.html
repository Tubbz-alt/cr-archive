46     @Test
47     public void testFieldTypes() {
48          GroupLayout g = (GroupLayout)AllTypes$LAYOUT;
49          int fieldCount = 0;
50          for (var ml : g.memberLayouts()) {
51              var type = ml.abiType().orElse(null);
52              if (type == null) {
53                  // ignore paddings
54                  continue;
55              }
56              switch (ml.name().get()) {
57                  case "sc":
58                      assertEquals(type, Type.SIGNED_CHAR);
59                      break;
60                  case "uc":
61                      assertEquals(type, Type.UNSIGNED_CHAR);
62                      break;
63                  case "s":
64                      assertEquals(type, Type.SHORT);
65                      break;
66                  case "us":
67                      assertEquals(type, Type.UNSIGNED_SHORT);
68                      break;
69                  case "i":
70                      assertEquals(type, Type.INT);
71                      break;
72                  case "ui":
73                      assertEquals(type, Type.UNSIGNED_INT);
74                      break;
75                  case "l":
76                      assertEquals(type, Type.LONG);
77                      break;
78                  case "ul":
79                      assertEquals(type, Type.UNSIGNED_LONG);
80                      break;
81                  case "ll":
82                      assertEquals(type, Type.LONG_LONG);
83                      break;
84                  case "ull":
85                      assertEquals(type, Type.UNSIGNED_LONG_LONG);
86                      break;
87              }
88              fieldCount++;
89          }
90          assertEquals(fieldCount, 10);
91     }

47     private static void checkFieldABIType(GroupLayout group, String fieldName, Type expected) {
48         assertEquals(group.select(PathElement.groupElement(fieldName)).abiType().orElseThrow(), expected);
49     }
50 
51     @Test
52     public void testFieldTypes() {
53         GroupLayout g = (GroupLayout)AllTypes$LAYOUT;
54         checkFieldABIType(g, "sc",  Type.SIGNED_CHAR);
55         checkFieldABIType(g, "uc",  Type.UNSIGNED_CHAR);
56         checkFieldABIType(g, "s",   Type.SHORT);
57         checkFieldABIType(g, "us",  Type.UNSIGNED_SHORT);
58         checkFieldABIType(g, "i",   Type.INT);
59         checkFieldABIType(g, "ui",  Type.UNSIGNED_INT);
60         checkFieldABIType(g, "l",   Type.LONG);
61         checkFieldABIType(g, "ul",  Type.UNSIGNED_LONG);
62         checkFieldABIType(g, "ll",  Type.LONG_LONG);
63         checkFieldABIType(g, "ull", Type.UNSIGNED_LONG_LONG);
64         checkFieldABIType(g, "f", Type.FLOAT);
65         checkFieldABIType(g, "d", Type.DOUBLE);
66         checkFieldABIType(g, "ld", Type.LONG_DOUBLE);
