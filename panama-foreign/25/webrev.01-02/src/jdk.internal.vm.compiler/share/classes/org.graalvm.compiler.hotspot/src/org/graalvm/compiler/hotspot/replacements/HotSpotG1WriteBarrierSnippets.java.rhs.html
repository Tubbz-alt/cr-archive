189     public static class Templates extends AbstractTemplates {
190         private final SnippetInfo g1PreWriteBarrier;
191         private final SnippetInfo g1ReferentReadBarrier;
192         private final SnippetInfo g1PostWriteBarrier;
193         private final SnippetInfo g1ArrayRangePreWriteBarrier;
194         private final SnippetInfo g1ArrayRangePostWriteBarrier;
195 
196         private final G1WriteBarrierLowerer lowerer;
197 
198         public Templates(OptionValues options, Iterable<DebugHandlersFactory> factories, Group.Factory factory, HotSpotProviders providers, TargetDescription target, GraalHotSpotVMConfig config) {
199             super(options, factories, providers, providers.getSnippetReflection(), target);
200             this.lowerer = new HotspotG1WriteBarrierLowerer(config, factory);
201 
202             HotSpotG1WriteBarrierSnippets receiver = new HotSpotG1WriteBarrierSnippets(config, providers.getRegisters());
203             g1PreWriteBarrier = snippet(G1WriteBarrierSnippets.class, "g1PreWriteBarrier", null, receiver, GC_INDEX_LOCATION, GC_LOG_LOCATION, SATB_QUEUE_MARKING_LOCATION, SATB_QUEUE_INDEX_LOCATION,
204                             SATB_QUEUE_BUFFER_LOCATION);
205             g1ReferentReadBarrier = snippet(G1WriteBarrierSnippets.class, "g1ReferentReadBarrier", null, receiver, GC_INDEX_LOCATION, GC_LOG_LOCATION, SATB_QUEUE_MARKING_LOCATION,
206                             SATB_QUEUE_INDEX_LOCATION, SATB_QUEUE_BUFFER_LOCATION);

207             g1PostWriteBarrier = snippet(G1WriteBarrierSnippets.class, "g1PostWriteBarrier", null, receiver, GC_CARD_LOCATION, GC_INDEX_LOCATION, GC_LOG_LOCATION, CARD_QUEUE_INDEX_LOCATION,
208                             CARD_QUEUE_BUFFER_LOCATION);
209             g1ArrayRangePreWriteBarrier = snippet(G1WriteBarrierSnippets.class, "g1ArrayRangePreWriteBarrier", null, receiver, GC_INDEX_LOCATION, GC_LOG_LOCATION, SATB_QUEUE_MARKING_LOCATION,
210                             SATB_QUEUE_INDEX_LOCATION, SATB_QUEUE_BUFFER_LOCATION);
211             g1ArrayRangePostWriteBarrier = snippet(G1WriteBarrierSnippets.class, "g1ArrayRangePostWriteBarrier", null, receiver, GC_CARD_LOCATION, GC_INDEX_LOCATION, GC_LOG_LOCATION,
212                             CARD_QUEUE_INDEX_LOCATION, CARD_QUEUE_BUFFER_LOCATION);
213         }

236     static final class HotspotG1WriteBarrierLowerer extends G1WriteBarrierLowerer {
237         private final CompressEncoding oopEncoding;
238 
239         HotspotG1WriteBarrierLowerer(GraalHotSpotVMConfig config, Factory factory) {
240             super(factory);
241             oopEncoding = config.useCompressedOops ? config.getOopEncoding() : null;
242         }
243 
244         @Override
245         public ValueNode uncompress(ValueNode expected) {
246             assert oopEncoding != null;
247             return HotSpotCompressionNode.uncompress(expected, oopEncoding);
248         }
249     }
