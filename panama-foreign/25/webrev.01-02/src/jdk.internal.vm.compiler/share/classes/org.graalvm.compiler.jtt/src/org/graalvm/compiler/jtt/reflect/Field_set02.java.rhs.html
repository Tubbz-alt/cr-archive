 34 public class Field_set02 extends JTTTest {
 35 
 36     private static class TestClass {
 37         public byte byteField;
 38         public short shortField;
 39         public char charField;
 40         public int intField;
 41         public long longField;
 42         public float floatField;
 43         public double doubleField;
 44         public boolean booleanField;
 45     }
 46 
 47     private static final TestClass object = new TestClass();
 48 
 49     public static boolean test(int arg) throws NoSuchFieldException, IllegalAccessException {
 50         if (arg == 0) {
 51             TestClass.class.getField("byteField").set(object, Byte.valueOf((byte) 11));
 52             return object.byteField == 11;
 53         } else if (arg == 1) {
 54             TestClass.class.getField("shortField").set(object, Short.valueOf((short) 12));
 55             return object.shortField == 12;
 56         } else if (arg == 2) {
 57             TestClass.class.getField("charField").set(object, Character.valueOf((char) 13));
 58             return object.charField == 13;
 59         } else if (arg == 3) {
 60             TestClass.class.getField("intField").set(object, Integer.valueOf(14));
 61             return object.intField == 14;
 62         } else if (arg == 4) {
 63             TestClass.class.getField("longField").set(object, Long.valueOf(15L));
 64             return object.longField == 15;
 65         } else if (arg == 5) {
 66             TestClass.class.getField("floatField").set(object, Float.valueOf(16));
 67             return object.floatField == 16;
 68         } else if (arg == 6) {
 69             TestClass.class.getField("doubleField").set(object, Double.valueOf(17));
 70             return object.doubleField == 17;
 71         } else if (arg == 7) {
 72             TestClass.class.getField("booleanField").set(object, true);
 73             return object.booleanField == true;
 74         }
 75         return false;
 76     }
 77 
 78     @Test
 79     public void run0() throws Throwable {
 80         try {
 81             runTest("test", 0);
 82         } catch (AssertionError e) {
 83             System.err.println(e);
 84             System.err.println("object.byteField == " + object.byteField);
 85             runTest(new OptionValues(getInitialOptions(), DebugOptions.Dump, ":2"), "test", 0);
 86         }
 87     }
