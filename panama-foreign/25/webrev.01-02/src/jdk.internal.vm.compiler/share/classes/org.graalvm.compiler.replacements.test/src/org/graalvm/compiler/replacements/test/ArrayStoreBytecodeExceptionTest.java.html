36 @RunWith(Parameterized.class)

48         Object[] objects = {"string", 42, new int[0], new String[0], new double[0][]};
49 
50         ArrayList<Object[]> ret = new ArrayList<>(objects.length);
51         for (Object o : objects) {
52             ret.add(new Object[]{o, o.getClass()});
53         }
54         return ret;
55     }
56 
57     @Test
58     public void testArrayStoreException() {
59         test("arrayStoreSnippet", new ArrayStoreBytecodeExceptionTest[1], object);
60     }
