 47 /**
 48  * A helper class to generate header interface class in source form.
 49  * After aggregating various constituents of a .java source, build
 50  * method is called to get overall generated source string.
 51  */
 52 class JavaSourceBuilder {
 53     private static final String ABI = SystemABI.getInstance().name();
 54     // buffer
 55     protected StringBuffer sb;
 56     // current line alignment (number of 4-spaces)
 57     protected int align;
 58 
 59     JavaSourceBuilder(int align) {
 60         this.align = align;
 61         this.sb = new StringBuffer();
 62     }
 63 
 64     JavaSourceBuilder() {
 65         this(0);
 66     }
 67 
 68     protected int align() {
 69         return align;
 70     }
 71 
 72     final String PUB_CLS_MODS = "public final ";
 73     final String PUB_MODS = "public static final ";
 74     final String PRI_MODS = "private static final ";
 75 
 76     protected void addPackagePrefix(String pkgName) {
 77         assert pkgName.indexOf('/') == -1 : "package name invalid: " + pkgName;
 78         sb.append("// Generated by jextract\n\n");
 79         if (!pkgName.isEmpty()) {
 80             sb.append("package ");
 81             sb.append(pkgName);
 82             sb.append(";\n\n");
 83         }
 84         addImportSection();
 85     }
 86 
 87     protected void addImportSection() {
 88         sb.append("import java.lang.invoke.MethodHandle;\n");
 89         sb.append("import java.lang.invoke.VarHandle;\n");
 90         sb.append("import jdk.incubator.foreign.*;\n");
 91         sb.append("import jdk.incubator.foreign.MemoryLayout.PathElement;\n");
 92         sb.append("import static ");
 93         sb.append(HandleSourceFactory.C_LANG_CONSTANTS_HOLDER);
 94         sb.append(".*;\n");
 95     }
 96 
 97     protected void classBegin(String name) {
 98         indent();
 99         sb.append(PUB_CLS_MODS + "class ");
100         sb.append(name);
101         sb.append(" {\n\n");
102     }
103 
104     protected void classEnd() {
105         indent();
106         sb.append("}\n\n");
107     }

109     protected void addLibraries(String[] libraryNames) {
110         incrAlign();
111         indent();
112         sb.append(PRI_MODS + "LibraryLookup[] LIBRARIES = RuntimeHelper.libraries(");
113         sb.append(stringArray(libraryNames) + ");\n");
114         decrAlign();
115     }

117     private String stringArray(String[] elements) {
118         return Stream.of(elements)
119                 .map(n -> "\"" + n + "\"")
120                 .collect(Collectors.joining(",", "new String[] {", "}"));
121     }
122 
123     protected void addLayout(String elementName, MemoryLayout layout) {
124         incrAlign();
125         indent();
126         sb.append(PUB_MODS + "MemoryLayout " + elementName + "$LAYOUT = ");
127         addLayout(layout);
128         sb.append(";\n");
129         decrAlign();
130     }
131 
132     private void addLayout(MemoryLayout l) {
133         if (l instanceof ValueLayout) {
134             SystemABI.Type type = l.abiType().orElseThrow(()->new AssertionError("Should not get here: " + l));
135             sb.append(switch (type) {
136                 case BOOL -> "C_BOOL";
137                 case SIGNED_CHAR -> "C_SCHAR";
138                 case UNSIGNED_CHAR -> "C_UCHAR";
139                 case CHAR -> "C_CHAR";
140                 case SHORT -> "C_SHORT";
141                 case UNSIGNED_SHORT -> "C_USHORT";
142                 case INT -> "C_INT";
143                 case UNSIGNED_INT -> "C_UINT";
144                 case LONG -> "C_LONG";
145                 case UNSIGNED_LONG -> "C_ULONG";
146                 case LONG_LONG -> "C_LONGLONG";
147                 case UNSIGNED_LONG_LONG -> "C_ULONGLONG";
148                 case FLOAT -> "C_FLOAT";
149                 case DOUBLE -> "C_DOUBLE";
150                 case LONG_DOUBLE -> "C_LONGDOUBLE";
151                 case POINTER -> "C_POINTER";
152                 default -> { throw new RuntimeException("should not reach here: " + type); }
153             });
154         } else if (l instanceof SequenceLayout) {
155             sb.append("MemoryLayout.ofSequence(");
156             if (((SequenceLayout) l).elementCount().isPresent()) {
157                 sb.append(((SequenceLayout) l).elementCount().getAsLong() + ", ");
158             }
159             addLayout(((SequenceLayout) l).elementLayout());
160             sb.append(")");
161         } else if (l instanceof GroupLayout) {
162             SystemABI.Type type = l.abiType().orElse(null);
163             if (type == SystemABI.Type.COMPLEX_LONG_DOUBLE) {
164                 if (ABI != SystemABI.ABI_SYSV) {
165                     throw new RuntimeException("complex long double is supported only for SysV ABI");
166                 } else {
167                     sb.append("C_COMPLEX_LONGDOUBLE");
168                 }
169             } else {
170                 if (((GroupLayout) l).isStruct()) {
171                     sb.append("MemoryLayout.ofStruct(\n");
172                 } else {
173                     sb.append("MemoryLayout.ofUnion(\n");
174                 }
175                 incrAlign();
176                 String delim = "";
177                 for (MemoryLayout e : ((GroupLayout) l).memberLayouts()) {
178                     sb.append(delim);
179                     indent();
180                     addLayout(e);
181                     delim = ",\n";
182                 }
183                 sb.append("\n");
184                 decrAlign();
185                 indent();
186                 sb.append(")");
187             }
188         } else {
189             //padding
190             sb.append("MemoryLayout.ofPaddingBits(" + l.bitSize() + ")");
191         }
192         if (l.name().isPresent()) {
193             sb.append(".withName(\"" +  l.name().get() + "\")");
194         }
195     }

197     protected void addVarHandle(String name, Class<?> type, String parentName) {
198         incrAlign();
199         indent();
200         String vhName = parentName != null ?
201                 parentName + "$" + name :
202                 name;
203         sb.append(PUB_MODS + "VarHandle " + vhName + " = ");
204         if (parentName != null) {
205             addHandlePath(type, parentName, name);
206         } else {
207             addHandlePath(type, name);
208         }
209         sb.append(";\n");
210         decrAlign();
211     }
212 
213     protected void addHandlePath(Class<?> type, String strName, String fieldName) {
214         String ty = type.getName();
215         if (ty.contains("MemoryAddress")) {
216             ty = "long";
217         }
218         sb.append(strName + "$LAYOUT.varHandle(" + ty + ".class, ");
219         sb.append("PathElement.groupElement(\"" + fieldName +"\")");
220         sb.append(")");
221     }
222 
223     protected void addHandlePath(Class<?> type, String varName) {
224         String ty = type.getName();
225         if (ty.contains("MemoryAddress")) {
226             ty = "long";
227         }
228         sb.append(varName + "$LAYOUT.varHandle(" + ty + ".class)");
229     }

231     protected void addMethodHandle(Declaration.Function funcTree, MethodType mtype, FunctionDescriptor desc) {
232         incrAlign();
233         indent();
234         sb.append(PUB_MODS + "MethodHandle " + funcTree.name() + " = ");
235         sb.append("RuntimeHelper.downcallHandle(\n");
236         incrAlign();
237         indent();
238         sb.append("LIBRARIES, \"" + funcTree.name() + "\"");
239         sb.append(",\n");
240         indent();
241         sb.append("\"" + mtype.toMethodDescriptorString() + "\",\n");
242         indent();
243         addFunction(desc);
244         sb.append(",\n");
245         indent();
246         sb.append(funcTree.type().varargs());
247         decrAlign();
248         sb.append("\n");
249         indent();
250         sb.append(");\n");
251         decrAlign();
252     }
253 
254     protected void addAddressLookup(String name) {
255         sb.append("RuntimeHelper.lookupGlobalVariable(LIBRARIES, \"" + name + "\")");
256     }

258     private void addFunction(FunctionDescriptor f) {
259         if (f.returnLayout().isPresent()) {
260             sb.append("FunctionDescriptor.of(");
261             addLayout(f.returnLayout().get());
262             sb.append(", ");
263         } else {
264             sb.append("FunctionDescriptor.ofVoid(");
265         }
266         if (!f.argumentLayouts().isEmpty()) {
267             sb.append("\n");
268             incrAlign();
269             String delim = "";
270             for (MemoryLayout e : f.argumentLayouts()) {
271                 sb.append(delim);
272                 indent();
273                 addLayout(e);
274                 delim = ",\n";
275             }
276             sb.append("\n");
277             decrAlign();
278             indent();
279         }
280         sb.append(")");
281     }
282 
283     protected void addAddress(String name) {
284         incrAlign();
285         indent();
286         sb.append(PUB_MODS + "MemoryAddress " + name + "$ADDR" + " = ");
287         addAddressLookup(name);
288         sb.append(";\n");
289         decrAlign();
290     }

292     protected void addConstant(String name, Class<?> type, Object value) {
293         incrAlign();
294         indent();
295         if (type == MemoryAddress.class || type == MemorySegment.class) {
296             //todo, skip for now (address constants and string constants)
297         } else {
298             sb.append(PUB_MODS + type.getName() + " " + name);
299             sb.append(" = ");
300             if (type == float.class) {
301                 sb.append(value);
302                 sb.append("f");
303             } else if (type == long.class) {
304                 sb.append(value);
305                 sb.append("L");
306             } else if (type == double.class) {
307                 sb.append(value);
308                 sb.append("d");
309             } else {
310                 sb.append("(" + type.getName() + ")");
311                 sb.append(value + "L");
312             }
313             sb.append(";\n");
314         }
315 
316         decrAlign();
317     }

319     static int funcIntfCounter = 0;
320 
321     protected void addUpcallFactory(FunctionDescriptor desc) {
322         String fnName = "FI" + funcIntfCounter++;
323         incrAlign();
324         indent();
325         sb.append(PRI_MODS + "FunctionDescriptor " + fnName + "$DESC = ");
326         addFunction(desc);
327         sb.append(";\n");
328         indent();
329         sb.append(PUB_MODS + "MemoryAddress " + fnName + "$make(MethodHandle handle) {\n");
330         incrAlign();
331         indent();
332         sb.append("return RuntimeHelper.upcallStub(handle, " + fnName + "$DESC);\n");
333         decrAlign();
334         indent();
335         sb.append("}\n");
336         decrAlign();
337     }

339     protected void addStaticFunctionWrapper(Declaration.Function f, MethodType mtype) {
340         incrAlign();
341         indent();
342         sb.append(PUB_MODS + mtype.returnType().getName() + " " + f.name() + " (");
343         String delim = "";
344         List<String> pNames = new ArrayList<>();
345         final int numParams = f.parameters().size();
346         for (int i = 0 ; i < numParams; i++) {
347             String pName = f.parameters().get(i).name();
348             if (pName.isEmpty()) {
349                 pName = "x" + i;
350             }
351             pNames.add(pName);
352             sb.append(delim + mtype.parameterType(i).getName() + " " + pName);
353             delim = ", ";
354         }
355         if (f.type().varargs()) {
356             String lastArg = "x" + numParams;
357             if (numParams > 0) {
358                 sb.append(", ");
359             }
360             sb.append("Object... " + lastArg);
361             pNames.add(lastArg);
362         }
363         sb.append(") {\n");
364         incrAlign();
365         indent();
366         sb.append("try {\n");
367         incrAlign();
368         indent();
369         if (!mtype.returnType().equals(void.class)) {
370             sb.append("return (" + mtype.returnType().getName() + ")");
371         }
372         sb.append(f.name() + ".invokeExact(" + String.join(", ", pNames) + ");\n");
373         decrAlign();
374         indent();
375         sb.append("} catch (Throwable ex) {\n");
376         incrAlign();
377         indent();
378         sb.append("throw new AssertionError(ex);\n");
379         decrAlign();
380         indent();
381         sb.append("}\n");
382         decrAlign();
383         indent();
384         sb.append("}\n");
385         decrAlign();
386     }
387 
388     void addDescriptor(String name, FunctionDescriptor desc) {
389         incrAlign();
390         indent();
391         sb.append(PRI_MODS + "FunctionDescriptor " + name + "$DESC = ");
392         addFunction(desc);
393         sb.append(";\n");
394         decrAlign();
395         indent();
396     }

398     void addFunctionalInterface(String name, MethodType mtype) {
399         incrAlign();
400         indent();
401         sb.append("public interface " + name + " {\n");
402         incrAlign();
403         indent();
404         sb.append(mtype.returnType().getName() + " apply(");
405         String delim = "";
406         for (int i = 0 ; i < mtype.parameterCount() ; i++) {
407             sb.append(delim + mtype.parameterType(i).getName() + " x" + i);
408             delim = ", ";
409         }
410         sb.append(");\n");
411         decrAlign();
412         indent();
413         sb.append("}\n");
414         decrAlign();
415         indent();
416     }

418     protected void addFunctionalFactory(String name, MethodType mtype) {
419         incrAlign();
420         indent();
421         sb.append(PUB_MODS + "MemoryAddress " + name + "$make(" + name + " fi) {\n");
422         incrAlign();
423         indent();
424         sb.append("return RuntimeHelper.upcallStub(" + name + ".class, fi, " + name + "$DESC, " +
425                 "\"" + mtype.toMethodDescriptorString() + "\");\n");
426         decrAlign();
427         indent();
428         sb.append("}\n");
429         decrAlign();
430     }

432     void addGetter(String name, Class<?> type, Declaration parent) {
433         incrAlign();
434         indent();
435         String vhName = (parent != null ? (parent.name() + "$") : "") + name;
436         String param = parent != null ? (MemorySegment.class.getName() + " seg") : "";
437         sb.append(PUB_MODS + type.getName() + " " + vhName + "$get(" + param + ") {\n");
438         incrAlign();
439         indent();
440         String vhParam = parent != null ?
441                 "seg.baseAddress()" : name + "$ADDR";
442         sb.append("return (" + type.getName() + ")" + vhName + ".get(" + vhParam + ");\n");
443         decrAlign();
444         indent();
445         sb.append("}\n");
446         decrAlign();
447     }
448 
449     void addSetter(String name, Class<?> type, Declaration parent) {
450         incrAlign();
451         indent();
452         String vhName = (parent != null ? (parent.name() + "$") : "") + name;
453         String param = parent != null ? (MemorySegment.class.getName() + " seg, ") : "";
454         sb.append(PUB_MODS + "void " + vhName + "$set(" + param + type.getName() + " x) {\n");
455         incrAlign();
456         indent();
457         String vhParam = parent != null ?
458                 "seg.baseAddress()" : name + "$ADDR";
459         sb.append(vhName + ".set(" + vhParam + ", x);\n");
460         decrAlign();
461         indent();
462         sb.append("}\n");
463         decrAlign();
464     }
465 
466     protected String build() {
467         String res = sb.toString();
468         this.sb = null;
469         return res.toString();
470     }
471 
472     protected void indent() {
473         for (int i = 0; i < align; i++) {
474             sb.append("    ");
475         }
476     }
477 
478     protected void incrAlign() {
479         align++;
480     }
481 
482     protected void decrAlign() {
483         align--;
484     }
