23 #include <math.h>
24 
25 #include "points.h"
26 
27 #ifdef _WIN64
28 #define EXPORT __declspec(dllexport)
29 #else
30 #define EXPORT
31 #endif
32 
33 EXPORT double distance(Point p1, Point p2) {
34     int xDist = abs(p1.x - p2.x);
35     int yDist = abs(p1.y - p2.y);
36     return sqrt((xDist * xDist) + (yDist * yDist));
37 }
38 
39 EXPORT double distance_ptrs(Point* p1, Point* p2) {
40     return distance(*p1, *p2);
41 }
