 23 package org.openjdk.bench.java.util.regex;
 24 
 25 import org.openjdk.jmh.annotations.Benchmark;
 26 import org.openjdk.jmh.annotations.BenchmarkMode;
 27 import org.openjdk.jmh.annotations.Measurement;
 28 import org.openjdk.jmh.annotations.Mode;
 29 import org.openjdk.jmh.annotations.OutputTimeUnit;
 30 import org.openjdk.jmh.annotations.Param;
 31 import org.openjdk.jmh.annotations.Scope;
 32 import org.openjdk.jmh.annotations.Setup;
 33 import org.openjdk.jmh.annotations.State;
 34 import org.openjdk.jmh.annotations.Warmup;
 35 
 36 import java.util.concurrent.TimeUnit;
 37 import java.util.regex.Pattern;
 38 import java.util.regex.PatternSyntaxException;
 39 
 40 @BenchmarkMode(Mode.AverageTime)
 41 @OutputTimeUnit(TimeUnit.NANOSECONDS)
 42 @State(Scope.Thread)
 43 public class PatternBench {
 44 
 45     public String fileTestString;
 46     public String flagsString;
 47 
 48     public Pattern graphemePattern;
 49     public Pattern jmodPattern;
 50     public Pattern jmodCanonicalPattern;
 51 
 52     public String charPatternRegex;
 53     public String[] charPatternStrings;
 54     public Pattern charPattern;
 55 
 56     @Setup
 57     public void setup() {
 58         flagsString = "\ud83c\udde6\ud83c\uddec\ud83c\uddec\ud83c\udde6\ud83c\uddfa\ud83c\uddf8\ud83c\uddeb\ud83c\uddf7";
 59         fileTestString = "META-INF/providers/org.openjdk.foo_hotspot_nodes_PluginFactory_EndLockScopeNode";
 60         graphemePattern = Pattern.compile("\\b{g}");
 61 
 62         String jmodRegex = "^.*(?:(?:_the\\.[^/]*)|(?:_[^/]*\\.marker)|(?:[^/]*\\.diz)|(?:[^/]*\\.debuginfo)|(?:[^/]*\\.dSYM/.*)|(?:[^/]*\\.dSYM)|(?:[^/]*\\.pdb)|(?:[^/]*\\.map))$";
 63         jmodCanonicalPattern = Pattern.compile(jmodRegex, Pattern.CANON_EQ);
 64         jmodPattern = Pattern.compile(jmodRegex);
 65 
 66         charPatternRegex = "[ a-zA-Z]*foo[ a-zA-Z0-9]*bar[ a-z]*";
 67         charPatternStrings = new String[] {"avaaafooddddddbariiii", "lorem ipsum dolor foo bar", "fpp brr lorem ipsum dolor foo bar %", "lorem ipsum dolor foo bar lorem ipsum dolor foo bar lorem ipsum dolor foo bar /"};
 68         charPattern = Pattern.compile(charPatternRegex);
 69     }
 70 
 71     @Benchmark
 72     @Warmup(iterations = 3)
 73     @Measurement(iterations = 3)
 74     public int splitFlags() {
 75         return graphemePattern.split(flagsString).length;
 76     }
 77 
 78     @Benchmark
 79     @Warmup(iterations = 3)
 80     @Measurement(iterations = 3)
 81     public boolean canonicalJmodMatch() {
 82         return jmodCanonicalPattern.matcher(fileTestString).matches();
 83     }
 84 
 85     @Benchmark
 86     @Warmup(iterations = 3)
 87     @Measurement(iterations = 3)
 88     public boolean normalJmodMatch() {
 89         return jmodPattern.matcher(fileTestString).matches();
 90     }
 91 
 92     @Benchmark
 93     @Warmup(iterations = 3)
 94     @Measurement(iterations = 3)
 95     public boolean charPatternMatch() {
 96         return charPattern.matcher(charPatternStrings[0]).matches()
 97                 && charPattern.matcher(charPatternStrings[1]).matches()
 98                 && charPattern.matcher(charPatternStrings[2]).matches();
 99     }
100 
101     @Benchmark
102     @Warmup(iterations = 3)
103     @Measurement(iterations = 3)
104     public boolean charPatternMatchWithCompile() {
105         Pattern p = Pattern.compile(charPatternRegex);
106         return p.matcher(charPatternStrings[0]).matches()
107                 && p.matcher(charPatternStrings[1]).matches()
108                 && p.matcher(charPatternStrings[2]).matches();
109     }
110 
111     @Benchmark
112     @Warmup(iterations = 3)
113     @Measurement(iterations = 3)
114     public Pattern charPatternCompile() {
115         return Pattern.compile(charPatternRegex);
116     }
117 }
