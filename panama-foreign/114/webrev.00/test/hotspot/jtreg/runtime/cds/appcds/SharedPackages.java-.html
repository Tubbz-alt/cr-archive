25 /*
26  * @test
27  * @summary AppCDS handling of package.
28  * @requires vm.cds
29  * @library /test/lib
30  * @compile test-classes/PackageTest.java
31  * @compile test-classes/JimageClassPackage.java
32  * @run driver SharedPackages
33  */
34 
35 import jdk.test.lib.process.OutputAnalyzer;
36 
37 public class SharedPackages {
38     public static void main(String[] args) throws Exception {
39         JarBuilder.build("pkg", "p/PackageTest", "JimageClassPackage");
40 
41         String appJar = TestCommon.getTestJar("pkg.jar");
42         TestCommon.testDump(appJar, TestCommon.list("p/PackageTest",
43                                                     "java/util/Dictionary",
44                                                     "sun/tools/javac/Main",
45                                                     "jdk/nio/zipfs/ZipInfo",
46                                                     "java/net/URL",
47                                                     "sun/rmi/rmic/Main",
48                                                     "com/sun/jndi/dns/DnsName"));
49 
50         OutputAnalyzer output;
51 
52         // Test 1: shared class from Jar on the -cp
53         output = TestCommon.exec(appJar, "-verbose:class", "p.PackageTest");
54         TestCommon.checkExec(output, "Expected package");
55         if (!TestCommon.isUnableToMap(output))
56             output.shouldContain("Package is not sealed");
57 
58         // Test 2: shared classes from "modules" jimage
59         output = TestCommon.exec(appJar, "-verbose:class",
60                                  "JimageClassPackage");
61         if (!TestCommon.isUnableToMap(output)) {
62             output.shouldNotContain("Unexpected package");
63             output.shouldNotContain("Package is not sealed");
64         }
65 
66         // Test 3: shared class from Jar on the -Xbootclasspath/a
67         TestCommon.dump(
68             appJar, TestCommon.list("p/PackageTest"), "-Xbootclasspath/a:" + appJar);
69         output = TestCommon.exec(appJar, "-Xbootclasspath/a:" + appJar, "p.PackageTest");
70         if (!TestCommon.isUnableToMap(output)) {
71             output.shouldNotContain("Unexpected package");
72             output.shouldContain("Package is not sealed");
73         }
74     }
75 }
