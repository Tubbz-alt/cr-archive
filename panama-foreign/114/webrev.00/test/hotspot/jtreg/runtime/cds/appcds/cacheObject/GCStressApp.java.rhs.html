25 import java.io.*;
26 import java.util.*;
27 import jdk.test.lib.Utils;
28 import sun.hotspot.WhiteBox;
29 
30 // All strings in archived classes are shared
31 public class GCStressApp {
32     static WhiteBox wb = WhiteBox.getWhiteBox();
33     static int[] arr;
34 
35     static String get_shared_string() {
36         String shared_str = "GCStressApp_shared_string";
37         return shared_str;
38     }
39 
40     static String get_shared_string1() {
41         String shared_str1 = "GCStressApp_shared_string1";
42         return shared_str1;
43     }
44 
45     static void allocAlot() {
46         try {
47             Random random = Utils.getRandomInstance();
48             for (int i = 0; i < 1024 * 1024; i++) {
49                 int len = random.nextInt(10000);
50                 arr = new int[len];
51             }
52         } catch (java.lang.OutOfMemoryError e) { }
53     }
54 
55     static void runGC() {
56         wb.fullGC();
57     }
58 
59     public static void main(String args[]) throws Exception {
60         if (!wb.isSharedClass(GCStressApp.class)) {
61            System.out.println("GCStressApp is not shared. Possibly there was a mapping failure.");
62            return;
63         }
64 
65         if (wb.areSharedStringsIgnored()) {
66           System.out.println("Shared strings are ignored.");
67           return;
68         }
69 
70         Object refs = wb.getResolvedReferences(GCStressApp.class);
71         if (wb.isShared(refs)) {
72             String shared_str = get_shared_string();
73             String shared_str1 = get_shared_string1();
74 
75             if (!wb.isShared(shared_str)) {
76                 throw new RuntimeException("FAILED. GCStressApp_shared_string is not shared");
77             }
78 
79             if (!wb.isShared(shared_str1)) {
80                 throw new RuntimeException("FAILED. GCStressApp_shared_string1 is not shared");
81             }
82 
83             allocAlot();
84             runGC();
85             runGC();
86             runGC();
87 
88             System.out.println("Passed");
89         } else {
90             System.out.println(
91                 "No cached resolved references. Open archive heap data is not used.");
92         }
93     }
94 }
