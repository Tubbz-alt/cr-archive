 24 package nsk.jdi.Argument.value;
 25 
 26 import nsk.share.*;
 27 import nsk.share.jpda.*;
 28 import nsk.share.jdi.*;
 29 
 30 import com.sun.jdi.*;
 31 import com.sun.jdi.connect.*;
 32 import java.io.*;
 33 import javax.naming.directory.Attribute;
 34 import java.util.*;
 35 


 36 /**
 37  * Test for the control of
 38  *
 39  *      Interface:      com.sun.jdi.connect.Connector.Argument
 40  *      Method:         public java.lang.String value()
 41  *      Assertion:      "Returns the current value of the argument."
 42  *
 43  *
 44  *      Comments:       The test aims on the concrete Sun's JDI
 45  *                      reference implementations. It uses
 46  *                      com.sun.jdi.RawCommandLineLaunch connector and its
 47  *                      "command" argument.
 48  *                      The test sets up the new "command" argument
 49  *                      value and then checks that new value remains previously
 50  *                      after connection establishing with debugee VM and
 51  *                      after debugee VM finishing.
 52  *
 53  *      Warning:        Temporarily the test is prepared only for
 54  *                      Sparc.Solaris.dt_socket-transport of RawCommandLineLaunch
 55  *                      connector
 56  */
 57 
 58 public class value004 {
 59     private static Log log;
 60 
 61     public static void main( String argv[] ) {
 62         System.exit(run(argv, System.out)+95); // JCK-compatible exit status
 63     }
 64 
 65     public static int run(String argv[],PrintStream out) {
 66         ArgumentHandler argHandler = new ArgumentHandler(argv);
 67         log = new Log(out, argHandler);
 68         VirtualMachineManager vmm = Bootstrap.virtualMachineManager();
 69 
 70         String address = argHandler.getTransportPort();
 71 
 72         List lcl = vmm.launchingConnectors();
 73         if (lcl.size() > 0) {
 74             log.display("Number of all known JDI launching connectors: "
 75                         + lcl.size());
 76         } else {
 77             log.complain("FAILURE: no JDI launching connectors found!");
 78             return 2;
 79         }
 80 

 81         Iterator lci = lcl.iterator();
 82         for (int i = 1; lci.hasNext(); i++) {
 83             Connector c = (Connector) lci.next();
 84             if (c.name().compareTo("com.sun.jdi.RawCommandLineLaunch") != 0) {

 85                 continue;
 86             }
 87             log.display("Connector's transport is: " + c.transport().name());
 88             if (c.transport().name().compareTo("dt_socket") != 0) {
 89                 log.display("WARNING: Temporarily the test is prepared only "
 90                           + "for dt_socket-transport of "
 91                           + "RawCommandLineLaunch connector!");
 92                 return 0;
 93             }
 94             if (System.getProperties().getProperty("os.arch")
 95                 .compareTo("sparc") != 0 ||
 96                 System.getProperties().getProperty("os.name")
 97                 .compareTo("SunOS") != 0) {
 98                 log.display("WARNING: Temporarily the test is prepared "
 99                           + "only for Sparc.Solaris-transport of "
100                           + "RawCommandLineLaunch connector!");
101                 return 0;
102             }
103             Map<java.lang.String,? extends com.sun.jdi.connect.Connector.Argument> cdfltArgmnts = c.defaultArguments();
104             int ksz = cdfltArgmnts.size();
105             String av[] = new String[ksz + 1];
106             Set ks = cdfltArgmnts.keySet();
107             if (ks.isEmpty()) {
108                 log.complain("FAILURE: empty default arguments set is "
109                            + "found for " + c.name() + " connector!");
110                 return 2;
111             }
112 
113             log.display("Looking over " + c.name() + " connector arguments: ");
114 
115             boolean flg = false;
116             Iterator argi = ks.iterator();
117             String ovl = null;
118             String nvl = null;
119             for (int j = 1; argi.hasNext(); j++) {
120                 String argkey = (String) argi.next();
121                 Connector.Argument argval =
122                     (Connector.Argument)cdfltArgmnts.get((Object) argkey);
123                 log.display("Connector.Argument argval = "+ argval);
124                 if (argkey.compareTo("command") != 0 &&
125                     argkey.compareTo("address") != 0) {
126                     continue;
127                 }
128                 if (argkey.compareTo("address") == 0) {
129                     if (argval.isValid(address)) {
130                         argval.setValue(address);
131                     } else {
132                         log.complain("FAILURE: Can't set up new value "
133                                    + "for address-argument");
134                         return 2;
135                     }
136                     continue;
137                 }
138                 flg = true;
139                 ovl = argHandler.getLaunchExecPath() + " "
140                      + "-Xdebug -Xnoagent -Djava.compiler=NONE "
141                      + "-Xrunjdwp:transport=dt_socket,suspend=y,"
142                      + "address=" + address + " nsk.jdi.Argument.value.value004a";
143                 if (argval.isValid(ovl)) {
144                     argval.setValue(ovl);
145                 } else {
146                     log.complain("FAILURE: Can't set up new value for "
147                                + "command-argument");
148                     return 2;
149                 }
150 
151                 nvl = argval.value();
152                 if (nvl.compareTo(ovl) != 0) {
153                     log.complain("FAILURE: Can't set up argument value!");
154                     return 2;
155                 }
156                 log.display("Changed " + argval.name() + " argument's "
157                           + "value is: " + nvl);
158             };
159 
160             Binder binder = new Binder(argHandler, log);
161             Debugee debugee = null;
162 
163             try {
164                 if (flg) {
165                     flg = false;
166                     VirtualMachine vm =
167                         ((LaunchingConnector)c).launch(cdfltArgmnts);
168                     log.display("VM = (" + vm + ")");
169                     debugee = binder.enwrapDebugee(vm, vm.process());
170 
171                     if (((Connector.Argument) cdfltArgmnts
172                         .get((Object) "command")).value()
173                         .compareTo(ovl) != 0) {
174                         log.complain("FAILURE: Current 'command' argument "
175                                    + "value is not coinsides with the "
176                                    + "last setted up value.");
177                         return 2;
178                     }
179 
180                     debugee.resume();
181                 }
182             } catch ( java.io.IOException exc) {
183                 log.complain("FAILURE: Unable to launch, so "
184                            + "java.io.IOException is arisen.");
185                 log.complain(exc.getMessage());
186                 return 2;
187             } catch ( com.sun.jdi.connect.IllegalConnectorArgumentsException
188                 exc) {
189                 log.complain("FAILURE: One of the connector arguments "
190                            + "is invalid, so "
191                            + "IllegalConnectorArgumentsException is arisen.");
192                 log.complain(exc.getMessage());
193                 return 2;
194             } catch ( com.sun.jdi.connect.VMStartException exc) {
195                 log.complain("FAILURE: VM was terminated with error before "
196                            + "a connection could be established, so "
197                            + "VMStartException is arisen.");
198                 log.complain(exc.getMessage());
199                 log.complain(Binder.readVMStartExceptionOutput(exc, log.getOutStream()));
200                 return 2;
201             } finally {
202                 if (debugee != null) {
203                     try {
204                         debugee.dispose();
205                     } catch (VMDisconnectedException ignore) {
206                     }
207 
208                     int extcd = debugee.waitFor();
209                     if (extcd != 95) {
210                         log.complain("FAILURE: Launching VM crushes "
211                                    + "with " + extcd + " exit code.");
212                         return 2;
213                     }
214                 }
215             }
216         };




217         log.display("Test PASSED!");
218         return 0;
219     }
220 }
