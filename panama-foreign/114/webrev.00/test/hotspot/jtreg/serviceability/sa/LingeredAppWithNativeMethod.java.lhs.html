25 import jdk.test.lib.apps.LingeredApp;

26 
27 public class LingeredAppWithNativeMethod extends LingeredApp {
28 
29     public static final String THREAD_NAME = "NoFramePointerJNIFib";
30     private static final int UPPER_BOUND = 55;
31     private static final int LOWER_BOUND = 40;

32 
33     static {
34         // JNI library compiled with no frame pointer info
35         System.loadLibrary("NoFramePointer");
36     }
37 
38     public void callNative() {
39         // Call JNI code which does something compute
40         // intensive: fibonacci
41         // That is to ensure that the native bits run when
42         // jstack --mixed info is to be gathered.
43         // Results of fibonacci calculation from JNI are
44         // reported via callback(). That's where the process
45         // of calculating fibonacci restarts.
46         int num = (int) (Math.random() * UPPER_BOUND);
47         while (num < LOWER_BOUND) {
48             num = (int) (Math.random() * UPPER_BOUND);
49         }
50         System.out.print("fib(" + num + ") = ");
51         callJNI(this, num);
52     }
53 
54     // Called from JNI library libNoFramePointer
55     private void callback(long val) {
56         System.out.println(val);
57         // Call native again so as to increase chances of
58         // being currently in JNI code when jstack --mixed
59         // runs.
60         callNative();
61     }
62 
63     public static native void callJNI(Object target, int num);
64 
65     public static void main(String[] args) {
66         LingeredAppWithNativeMethod app = new LingeredAppWithNativeMethod();
67         Thread fibonacci = new Thread(() -> {
68             app.callNative();
69         });
70         fibonacci.setName(THREAD_NAME);
71         fibonacci.start();
72         LingeredApp.main(args);
73     }
74 }
