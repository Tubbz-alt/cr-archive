24 import javax.script.*;
25 
26 /*
27  * If the JDK being tested is <b>not</b> a Sun product JDK and a js
28  * engine is not present, return an exit code of 2 to indicate that
29  * the jrunscript tests which assume a js engine can be vacuously
30  * passed.
31  */
32 public class CheckEngine {
33     public static void main(String... args) {
34         int exitCode = 0;
35         ScriptEngine engine =
36             (new ScriptEngineManager()).getEngineByName("nashorn");
37 
38         if (engine == null &&
39             !(System.getProperty("java.runtime.name").startsWith("Java(TM)"))) {
40             exitCode = 2;
41         }
42 
43         System.exit(exitCode);
44     }
45 }
