24 /*
25  * @test
26  * @bug 6249843 6705893
27  * @summary Test basic script compilation. Value eval'ed from
28  * compiled and interpreted scripts should be same.
29  */
30 
31 import java.io.File;
32 import java.io.FileReader;
33 import java.io.Reader;
34 import javax.script.Compilable;
35 import javax.script.CompiledScript;
36 import javax.script.ScriptEngine;
37 import javax.script.ScriptEngineManager;
38 
39 public class Test6 {
40         public static void main(String[] args) throws Exception {
41             System.out.println("\nTest6\n");
42             ScriptEngineManager m = new ScriptEngineManager();
43             ScriptEngine engine = Helper.getJsEngine(m);
44             if (engine == null) {
45                 System.out.println("Warning: No js engine found; test vacuously passes.");
46                 return;
47             }
48 
49             try (Reader reader = new FileReader(
50                 new File(System.getProperty("test.src", "."), "Test6.js"))) {
51                 engine.eval(reader);
52             }
53             Object res = engine.get("res");
54 
55             CompiledScript scr = null;
56             try (Reader reader = new FileReader(
57                 new File(System.getProperty("test.src", "."), "Test6.js"))) {
58                 scr = ((Compilable)engine).compile(reader);
59             }
60 
61             if (scr == null) {
62                 throw new RuntimeException("compilation failed!");
63             }
64 
65             scr.eval();
66             Object res1 = engine.get("res");
67             if (! res.equals(res1)) {
68                 throw new RuntimeException("values not equal");
69             }
70         }
71 }
