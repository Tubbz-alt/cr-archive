30 import java.io.File;
31 import java.io.FileReader;
32 import java.io.Reader;
33 import javax.script.ScriptEngine;
34 import javax.script.ScriptEngineManager;
35 
36 public class Test7 {
37         public static void main(String[] args) throws Exception {
38             System.out.println("\nTest7\n");
39             File file =
40                 new File(System.getProperty("test.src", "."), "Test7.js");
41             Reader r = new FileReader(file);
42             ScriptEngineManager m = new ScriptEngineManager();
43             ScriptEngine eng = Helper.getJsEngine(m);
44             if (eng == null) {
45                 System.out.println("Warning: No js engine found; test vacuously passes.");
46                 return;
47             }
48             eng.put("filename", file.getAbsolutePath());
49             eng.eval(r);
50             String str = (String)eng.get("firstLine");
51             // do not change first line in Test7.js -- we check it here!
52             if (!str.equals("//this is the first line of Test7.js")) {
53                 throw new RuntimeException("unexpected first line");
54             }
55         }
56 }
