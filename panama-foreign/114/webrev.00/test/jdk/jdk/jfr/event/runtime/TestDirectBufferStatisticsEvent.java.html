26 package jdk.jfr.event.runtime;
27 
28 import java.nio.ByteBuffer;
29 import java.util.List;
30 
31 import jdk.internal.misc.VM;
32 
33 import jdk.jfr.Recording;
34 import jdk.jfr.consumer.RecordedEvent;
35 import jdk.test.lib.Asserts;
36 import jdk.test.lib.jfr.EventNames;
37 import jdk.test.lib.jfr.Events;
38 
39 /**
40  * @test
41  * @key jfr
42  * @requires vm.hasJFR
43  * @library /test/lib
44  * @modules java.base/jdk.internal.misc
45  * @run main/othervm -XX:MaxDirectMemorySize=128m jdk.jfr.event.runtime.TestDirectBufferStatisticsEvent
46  * @run main/othervm jdk.jfr.event.runtime.TestDirectBufferStatisticsEvent
47  */
48 public class TestDirectBufferStatisticsEvent {
49 
50     private static final String EVENT_PATH = EventNames.DirectBufferStatistics;
51 
52     public static void main(String[] args) throws Throwable {
53         try (Recording recording = new Recording()) {
54             recording.enable(EVENT_PATH);
55             recording.start();
56             int rounds = 16;
57             int size = 1 * 1024 * 1024; // 1M
58             for (int i = 0; i < rounds; i++) {
59                 ByteBuffer.allocateDirect(size);
60             }
61             recording.stop();
62 
63             List<RecordedEvent> events = Events.fromRecording(recording);
64             Events.hasEvents(events);
65 
66             long count = 0;
67             long totalCapacity = 0;
68             long memoryUsed = 0;
69 
70             for (RecordedEvent event : events) {
71                 System.out.println(event);
72                 Asserts.assertTrue(Events.isEventType(event, EVENT_PATH), "Wrong event type");
73                 count = Math.max(count, Events.assertField(event, "count").getValue());
74                 totalCapacity = Math.max(totalCapacity, Events.assertField(event, "totalCapacity").getValue());
75                 memoryUsed = Math.max(memoryUsed, Events.assertField(event, "memoryUsed").getValue());
76                 Asserts.assertEquals(VM.maxDirectMemory(), Events.assertField(event, "maxCapacity").getValue());
77             }
78 
79             Asserts.assertGreaterThanOrEqual(count, (long)rounds, "Too few count in statistics event");
80             Asserts.assertGreaterThanOrEqual(totalCapacity, (long)(rounds * size), "Too few totalCapacity in statistics event");
81             Asserts.assertGreaterThanOrEqual(memoryUsed, totalCapacity, "Too few memoryUsed in statistics event");
82         }
83     }
84 }
