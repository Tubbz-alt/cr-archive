 33 /**
 34  * Test --file-associations parameter. Output of the test should be
 35  * fileassociationstest*.* installer. The output installer should provide the
 36  * same functionality as the default installer (see description of the default
 37  * installer in SimplePackageTest.java) plus configure file associations. After
 38  * installation files with ".jptest1" and ".jptest2" suffixes should be
 39  * associated with the test app.
 40  *
 41  * Suggested test scenario is to create empty file with ".jptest1" suffix,
 42  * double click on it and make sure that test application was launched in
 43  * response to double click event with the path to test .jptest1 file on the
 44  * commend line. The same applies to ".jptest2" suffix.
 45  *
 46  * On Linux use "echo > foo.jptest1" and not "touch foo.jptest1" to create test
 47  * file as empty files are always interpreted as plain text and will not be
 48  * opened with the test app. This is a known bug.
 49  *
 50  * Icon associated with the main launcher should be associated with files with
 51  * ".jptest1" suffix. Different icon should be associated with files with with
 52  * ".jptest2" suffix. Icon for files with ".jptest1" suffix is platform specific
 53  * and is one of 'icon.*' files in test/jdk/tools/jpackage/resources directory.
 54  */
 55

 69 /*
 70  * @test
 71  * @summary jpackage with --file-associations
 72  * @library ../helpers
 73  * @key jpackagePlatformPackage
 74  * @requires jpackage.test.SQETest != null
 75  * @build jdk.jpackage.test.*
 76  * @modules jdk.incubator.jpackage/jdk.incubator.jpackage.internal
 77  * @compile FileAssociationsTest.java
 78  * @run main/othervm/timeout=360 -Xmx512m jdk.jpackage.test.Main
 79  *  --jpt-run=FileAssociationsTest.test
 80  */
 81 
 82 public class FileAssociationsTest {
 83     @Test
 84     public static void test() {
 85         PackageTest packageTest = new PackageTest();
 86 
 87         // Not supported
 88         packageTest.excludeTypes(PackageType.MAC_DMG);
 89 
 90         new FileAssociations("jptest1").applyTo(packageTest);
 91 
 92         Path icon = TKit.TEST_SRC_ROOT.resolve(Path.of("resources", "icon"
 93                 + TKit.ICON_SUFFIX));
 94 
 95         icon = TKit.createRelativePathCopy(icon);
 96 
 97         new FileAssociations("jptest2")
 98                 .setFilename("fa2")
 99                 .setIcon(icon)
100                 .applyTo(packageTest);
101 
102         packageTest.run();
103     }
104 
105     @Test
106     public static void testNoMime() {
107         final Path propFile = TKit.workDir().resolve("fa.properties");
108 
109         PackageTest packageTest = new PackageTest().excludeTypes(PackageType.MAC);
110 
111         packageTest.configureHelloApp().addRunOnceInitializer(() -> {
112             TKit.createPropertiesFile(propFile, Map.of(
113                 "extension", "foo",
114                 "description", "bar"
115             ));
116         }).addInitializer(cmd -> {
117             cmd.addArguments("--file-associations", propFile).saveConsoleOutput(true);
118         }).setExpectedExitCode(1).addBundleVerifier((cmd, result) -> {
119            TKit.assertTextStream(
120                    "No MIME types were specified for File Association number 1")
121                    .apply(result.getOutput().stream());
122            TKit.assertTextStream(
123                    "Advice to fix: Specify MIME type for File Association number 1")
124                    .apply(result.getOutput().stream());
125         }).run();
126     }
127 
128     @Test
129     public static void testTooManyMimes() {
130         final Path propFile = TKit.workDir().resolve("fa.properties");
131 
132         PackageTest packageTest = new PackageTest().excludeTypes(PackageType.MAC);
133 
134         packageTest.configureHelloApp().addRunOnceInitializer(() -> {
135             TKit.createPropertiesFile(propFile, Map.of(
136                 "mime-type", "application/x-jpackage-foo, application/x-jpackage-bar",
137                 "extension", "foo",
138                 "description", "bar"
139             ));
140         }).addInitializer(cmd -> {
141             cmd.addArguments("--file-associations", propFile).saveConsoleOutput(true);
142         }).setExpectedExitCode(1).addBundleVerifier((cmd, result) -> {
143            TKit.assertTextStream(
144                    "More than one MIME types was specified for File Association number 1")
145                    .apply(result.getOutput().stream());
146            TKit.assertTextStream(
147                    "Advice to fix: Specify only one MIME type for File Association number 1")
148                    .apply(result.getOutput().stream());
149         }).run();
150     }
