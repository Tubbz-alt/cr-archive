24 /**
25  * Simple benchmark to measure push/pop specialized method performance
26  */
27 
28 var a = [];
29 
30 var RESULT = 15;
31 
32 function bench() {
33     var sum = 0;
34     for (var i=0;i<10;i++) {
35 	a.push(i);
36     }
37     for (var i=0;i<10;i++) {
38 	sum |= a.pop();
39     }
40     return sum;
41 }
42 
43 function runbench() {
44     var sum = 0;
45     for (var iters = 0; iters<1e8; iters++) {
46 	sum |= bench();
47     }
48     return sum;
49 }
50 
51 var d = new Date;
52 var res = runbench();
53 print((new Date - d) + " ms");
54 print();
55 if (res != RESULT) {
56     print("ERROR: Wrong result - should be " + RESULT);
57 } else {
58     print("Verified OK - result is correct");
59 }
