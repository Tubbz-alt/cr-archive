24 /**
25  * Simple benchmark to measure charCodeAt specialized method performance
26  */
27 
28 var str = "sghjkdsfkjghsdfjkfkjdfkjdfjkdfjkfdjkfdkfldjfhdfpkjdhafgksdjfgldfgjldfkjgdlfjgldkfjgkldfj";
29 var RESULT1 = 9187;
30 var RESULT2 = 1496;
31 
32 function f() {
33     var len = str.length;
34     var c = 0;
35     for (var i = 0; i < len; i++) {
36 	c += str.charCodeAt(i);
37     }
38     return c;
39 }
40 
41 function bench(res) {
42     var d = new Date;
43     var sum = 0;
44     for (var i = 0; i < 1e6; i++) {
45 	sum |= f();
46     }
47     if (sum == res) {
48 	print("Verified OK"); 
49     } else {
50 	print("Verification failed " + sum + " should be " + res);
51     }
52     print((new Date - d) + " ms");
53 }
54 
55 bench(RESULT1);
56 
57 print("Replacing charCodeAt... ");
58 
59 String.prototype.charCodeAt = function() { return 17; }
60 
61 bench(RESULT2);
62 
63 print("Done");
