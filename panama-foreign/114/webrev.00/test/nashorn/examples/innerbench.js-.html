32 function bench() {
33    var N = 10000000;
34 
35    function empty() {}
36 
37    function outer1() {
38        function inner() {}
39    }
40 
41    function outer2() {
42        function inner() {}
43        inner();
44    }
45 
46    for (var i = 0, d = Date.now(); i < N; i++) {
47       empty();
48    }
49    print("empty :", Date.now() - d);
50    for (var i = 0, d = Date.now(); i < N; i++) {
51       outer1();
52    }
53    print("outer1:", Date.now() - d);
54    for (var i = 0, d = Date.now(); i < N; i++) {
55       outer2();
56    }
57    print("outer2:", Date.now() - d);
58 }
59 bench();
