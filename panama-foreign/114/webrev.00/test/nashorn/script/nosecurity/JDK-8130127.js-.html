24 /**
25  * JDK-8130127: streamline input parameter of Nashorn scripting $EXEC function
26  *
27  * Test different variants of stdin passing to $EXEC.
28  *
29  * @test
30  * @option -scripting
31  * @run
32  */
33 
34 var File = java.io.File,
35     sep = File.separator,
36     System = java.lang.System,
37     os = System.getProperty("os.name"),
38     win = os.startsWith("Windows"),
39     jjsName = "jjs" + (win ? ".exe" : ""),
40     javaHome = System.getProperty("java.home")
41 
42 var jjs = javaHome + "/../bin/".replace(/\//g, sep) + jjsName
43 if (!new File(jjs).isFile()) {
44     jjs = javaHome + "/bin/".replace(/\//g, sep) + jjsName
45 }
46 
47 var jjsCmd = jjs + " readprint.js"
48 
49 print($EXEC(jjsCmd))
50 print($EXEC(jjsCmd, null))
51 print($EXEC(jjsCmd, undefined))
52 print($EXEC(jjsCmd, ""))
53 
54 print($EXEC(jjs, "print('hello')\n"))
