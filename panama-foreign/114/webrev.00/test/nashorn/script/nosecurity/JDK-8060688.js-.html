24 /**
25  * JDK-8060688: Nashorn: Generated script class name fails --verify-code for names with special chars
26  *
27  * @test
28  * @run
29  */
30 
31 var NashornEngineFactory = Java.type("jdk.nashorn.api.scripting.NashornScriptEngineFactory");
32 var ScriptEngine = Java.type("javax.script.ScriptEngine");
33 var ScriptContext = Java.type("javax.script.ScriptContext");
34 
35 var factory = new NashornEngineFactory();
36 
37 var e = factory.getScriptEngine("--verify-code");
38 
39 function evalAndCheck(code) {
40     try {
41         e.eval(code);
42     } catch (exp) {
43         exp.printStackTrace();
44     }
45 }
46 
47 // check default name
48 evalAndCheck("var a = 3");
49 // check few names with special chars
50 var scontext = e.context;
51 scontext.setAttribute(ScriptEngine.FILENAME, "<myscript>", ScriptContext.ENGINE_SCOPE);
52 evalAndCheck("var h = 'hello'");
53 scontext.setAttribute(ScriptEngine.FILENAME, "[myscript]", ScriptContext.ENGINE_SCOPE);
54 evalAndCheck("var foo = 'world'");
55 scontext.setAttribute(ScriptEngine.FILENAME, ";/\\$.", ScriptContext.ENGINE_SCOPE);
56 evalAndCheck("var foo = 'helloworld'");
