32 /**
33  * JDK-8006191 - `cmd` -> exec("cmd") in script mode
34  *
35  * @test
36  * @option -scripting
37  * @argument ArgumentFromCommandLine
38  * @run
39  */

41 #!/usr/bin/jjs
42 
43 $ENV.PWD = ".";
44 print($ENV.PWD);
45 
46 var files = `ls`.trim().split("\n");
47 for (var i in files) {
48     var file = files[i];
49     if (file.contains("README")) {
50         print(file);
51     }
52 }
53 
54 var result = $EXEC("cat", <<EOD);
55 This is a bunch of stuff
56 that I want written out
57 including ${$ARG[0]}
58 EOD
59 print(result);
60 print($OUT);
61 
62 var arg = "-Q";
63 `ls ${arg}`;
64 print($ERR);
65 print($EXIT);
