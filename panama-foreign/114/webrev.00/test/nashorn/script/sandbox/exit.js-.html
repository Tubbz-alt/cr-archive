24 /**
25  * Try to call System.exit, quit and exit.
26  *
27  * @test
28  * @security
29  */
30 
31 function check(e) {
32     if (e instanceof java.lang.SecurityException) {
33         print(e);
34     } else {
35         fail("expected SecurityException, got " + e);
36     }
37 }
38 
39 try {
40     java.lang.System.exit(0);
41     // will not reach here regardless of outcome..
42 } catch (e) {
43     check(e);
44 }
45 
46 try {
47     quit();
48     // will not reach here regardless of outcome..
49 } catch (e) {
50     check(e);
51 }
52 
53 try {
54     exit(0);
55     // will not reach here regardless of outcome..
56 } catch (e) {
57     check(e);
58 }
59 
60 print("Success, didn't exit!");
