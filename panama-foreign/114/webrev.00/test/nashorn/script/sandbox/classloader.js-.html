24 /**
25  * Try to extend ClassLoader.
26  *
27  * @test
28  * @security
29  * @bug JDK-8032954: Nashorn: extend Java.extend
30  */
31 
32 try {
33     var l = new (Java.extend(java.lang.ClassLoader))({});
34     fail("should have thrown SecurityException");
35 } catch (e) {
36     if (e instanceof java.lang.SecurityException) {
37         print(e);
38     } else {
39         fail("expected SecurityException, got " + e);
40     }
41 }
42 
43 try {
44     Java.extend(Java.type('java.lang.ClassLoader'));
45     fail("should have thrown SecurityException");
46 } catch (e) {
47     if (e instanceof java.lang.SecurityException) {
48         print(e);
49     } else {
50         fail("expected SecurityException, got " + e);
51     }
52 }
53

55     Java.extend(Java.type("javax.management.loading.MLet"));

59         print(e);
60     } else {

62     }
