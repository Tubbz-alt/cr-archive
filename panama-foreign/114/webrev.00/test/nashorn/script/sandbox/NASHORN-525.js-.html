24 /**
25  * NASHORN-525 : nashorn misses security access checks
26  *
27  * @test
28  * @run
29  */
30 
31 function check(code) {
32     try {
33         eval(code);
34         fail("SecurityException expected for : " + code);
35     } catch (e) {
36         if (! (e instanceof java.lang.SecurityException)) {
37             fail("SecurityException expected, but got " + e);
38         }
39     }
40 }
41 
42 // if security manager is absent, pass the test vacuously.
43 if (java.lang.System.getSecurityManager() != null) {
44     // try accessing class from 'sun.*' packages
45     check("Packages.sun.misc.Unsafe");
46     check("Java.type('sun.misc.Unsafe')");
47 
48     // TODO this works in Java8 but not in Java8, disabling for now
49     check("java.lang.Class.forName('sun.misc.Unsafe')");
50 
51     // try System.exit and System.loadLibrary
52     check("java.lang.System.exit(0)");
53     check("java.lang.System.loadLibrary('foo')");
54 }
