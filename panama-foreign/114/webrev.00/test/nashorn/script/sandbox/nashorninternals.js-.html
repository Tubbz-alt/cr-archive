25 /**
26  * Test to check that nashorn "internal" classes in codegen, parser, ir
27  * packages cannot * be accessed from sandbox scripts.
28  *
29  * @test
30  * @run
31  * @security
32  */
33 
34 function checkClass(name) {
35     try {
36         Java.type(name);
37         fail("should have thrown exception for: " + name);
38     } catch (e) {
39         if (! (e instanceof java.lang.SecurityException)) {
40             fail("Expected SecurityException, but got " + e);
41         }
42     }
43 }
44
