32 // add/delete property to proto (direct/indirect) should
33 // not affect the property map of the objects
34 
35 var proto2 = { foo: 334 }
36 var proto  = Object.create(proto2);
37 proto.bar = "hello";
38 
39 var obj1 = Object.create(proto);
40 var obj2 = Object.create(proto);
41 
42 assertSameMap(obj1, obj2);
43 
44 proto.newX = 'world';
45 assertSameMap(obj1, obj2);
46 
47 delete proto.newX;
48 assertSameMap(obj1, obj2);
49 
50 proto2.newX = "foo";
51 assertSameMap(obj1, obj2);
52 
53 delete proto2.newX;
54 assertSameMap(obj1, obj2);
