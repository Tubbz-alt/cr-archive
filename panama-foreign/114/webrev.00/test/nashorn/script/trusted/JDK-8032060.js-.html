24 /**
25  * JDK-8032060: PropertyMap of Error objects is not stable
26  *
27  * @test
28  * @option -Dnashorn.debug=true
29  * @fork
30  * @run
31  */
32 
33 function checkMap(e1, e2) {
34     if (! Debug.identical(Debug.map(e1), Debug.map(e2))) {
35         fail("e1 and e2 have different maps");
36     }
37 
38     var m1, m2;
39 
40     try {
41         throw e1
42     } catch (e) {
43         m1 = Debug.map(e)
44     }
45 
46     try {
47         throw e2
48     } catch (e) {
49         m2 = Debug.map(e)
50     }
51 
52     if (! Debug.identical(m1, m2)) {
53         fail("e1 and e2 have different maps after being thrown");
54     }
55 }
56 
57 checkMap(new Error(), new Error());
58 checkMap(new EvalError(), new EvalError());
59 checkMap(new RangeError(), new RangeError());
60 checkMap(new ReferenceError(), new ReferenceError());
61 checkMap(new SyntaxError(), new SyntaxError());
62 checkMap(new TypeError(), new TypeError());
63 checkMap(new URIError(), new URIError());
64 
65 // now try with message param
66 checkMap(new Error("x"), new Error("y"));
67 checkMap(new EvalError("x"), new EvalError("y"));
68 checkMap(new RangeError("x"), new RangeError("y"));
69 checkMap(new ReferenceError("x"), new ReferenceError("y"));
70 checkMap(new SyntaxError("x"), new SyntaxError("y"));
71 checkMap(new TypeError("x"), new TypeError("y"));
72 checkMap(new URIError("x"), new URIError("y"));
