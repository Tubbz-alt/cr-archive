24 /**
25  * JDK-8020809: Java adapter should not allow overriding of caller sensitive methods
26  *
27  * @test
28  * @run
29  */
30 
31 // This test runs as trusted as we need the
32 // "enableContextClassLoaderOverride" runtime permission.
33 
34 var T = Java.extend(java.lang.Thread, {
35     getContextClassLoader: function() {
36         // Since getContextClassLoader is caller sensitive, this won't
37         // actually act as an override; it will be ignored. If we could
38         // invoke it, it would be an error.
39         throw new Error()
40     }
41 })
42 
43 // Retrieve the context ClassLoader on our Thread adapter, ensure the
44 // method was not, in fact, overridden.
45 var cl = (new T()).contextClassLoader
46 
47 print("cl is class loader: "  + (cl instanceof java.lang.ClassLoader))
