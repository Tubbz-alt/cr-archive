24 /**
25  * JDK-8011209: Object.getOwnPropertyDescriptor(function(){"use strict"},"caller").get.length is not 0
26  *
27  * @test
28  * @run
29  */
30 
31 var callerPropDesc = Object.getOwnPropertyDescriptor(function(){"use strict"},"caller");
32 
33 var getterLen = callerPropDesc.get.length;
34 if (getterLen != 0) {
35     fail("caller's get.length != 0");
36 }
37 
38 var setterLen = callerPropDesc.set.length;
39 if (setterLen != 0) {
40     fail("caller's set.length != 1");
41 }
42 
43 var argumentsPropDesc = Object.getOwnPropertyDescriptor(function(){"use strict"},"arguments");
44 
45 getterLen = argumentsPropDesc.get.length;
46 if (getterLen != 0) {
47     fail("arguments's get.length != 0");
48 }
49 
50 setterLen = argumentsPropDesc.set.length;
51 if (setterLen != 0) {
52     fail("arguments's set.length != 1");
53 }
54 
55 var strictArgs = (function() { 'use strict'; return arguments; })();
56 callerPropDesc = Object.getOwnPropertyDescriptor(strictArgs,"caller");
57 getterLen = callerPropDesc.get.length;
58 if (getterLen != 0) {
59     fail("argument.caller's get.length != 0");
60 }
61 
62 setterLen = callerPropDesc.set.length;
63 if (setterLen != 0) {
64     fail("argument.caller's set.length != 1");
65 }
66 
67 calleePropDesc = Object.getOwnPropertyDescriptor(strictArgs,"callee");
68 getterLen = calleePropDesc.get.length;
69 if (getterLen != 0) {
70     fail("argument.callee's get.length != 0");
71 }
72 
73 setterLen = calleePropDesc.set.length;
74 if (setterLen != 0) {
75     fail("argument.callee's set.length != 1");
76 }
