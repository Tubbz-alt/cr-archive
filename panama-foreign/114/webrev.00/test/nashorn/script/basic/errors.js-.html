47 var err = new Error("my error");
48 try {
49     throw err;
50 } catch (e) {
51     print(e instanceof Error);
52     print(e.message);
53     print(e.name);
54     var ne = e.nashornException;
55     if (ne != undefined) {
56         if (ne.fileName !== __FILE__) {
57             fail("incorrect filename in error");
58         }
59         print("thrown @ " + ne.lineNumber);
60     }
61 }

63 // try to print undefined global var..
64 try {
65     print(foo);
66 } catch (e) {
67     print(e instanceof ReferenceError);
68     print(e.name);
69     print(e.message);

72 // try to call something that is undefined

74     Object.foo_method();
