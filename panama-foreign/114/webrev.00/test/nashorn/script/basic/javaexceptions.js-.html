24 /**
25  * Basic checks for throwing and catching java exceptions from script.
26  *
27  * @test
28  * @run
29  */
30 
31 try {
32     new java.io.FileInputStream("non_existent_file");
33 } catch (e) {
34     print(e instanceof java.io.FileNotFoundException || e instanceof java.lang.SecurityException);
35 }
36 
37 try {
38     new java.net.URL("invalid_url");
39 } catch (e) {
40     print(e instanceof java.net.MalformedURLException);
41     print(e);
42 }
43 
44 try {
45     var obj = new java.lang.Object();
46     obj.wait();
47 } catch (e) {
48     print(e instanceof java.lang.IllegalMonitorStateException);
49     print(e);
50 }
51 
52 // directly throw a Java exception
53 try {
54     throw new java.io.IOException("I/O failed");
55 } catch (e) {
56     print(e instanceof java.io.IOException);
57     print(e);
58 }
