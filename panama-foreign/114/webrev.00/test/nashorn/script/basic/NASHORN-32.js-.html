24 /**
25  * NASHORN-32: should not be able to read, write, call properties from null literal.
26  *
27  * @test
28  * @run
29  */
30 
31 try {
32     print(null["foo"]);
33 } catch (e) {
34     print(e instanceof TypeError);
35     print(e);
36 }
37 
38 try {
39     print(null.foo);
40 } catch (e) {
41     print(e instanceof TypeError);
42     print(e);
43 }
44 
45 try {
46     null["foo"] = 55;
47 } catch (e) {
48     print(e instanceof TypeError);
49     print(e);
50 }
51 
52 try {
53     null.foo = 55;
54 } catch (e) {
55     print(e instanceof TypeError);
56     print(e);
57 }
58 
59 try {
60     print(null.foo());
61 } catch (e) {
62     print(e instanceof TypeError);
63     print(e);
64 }
65 
66 try {
67     print(new null());
68 } catch (e) {
69     print(e instanceof TypeError);
70     print(e);
