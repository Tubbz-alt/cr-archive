25 /**
26  * NASHORN-99 :  Passing less number of arguments to a native varargs function results in IllegalArgumentException.
27  *
28  * @test
29  * @run
30  */
31 
32 function func() {
33 }
34 
35 try {
36     func.call();
37 } catch (e) {
38     fail("func.call() failed", e);
39 }
40 
41 var obj = Object.create(Function.prototype);
42 
43 if (typeof obj.call !== "function") {
44     fail('#1: call method not found!');
45 }
46 
47 try {
48     obj.call();
49     fail('#2: No [[Call]] property, a TypeError expected');
50 } catch (e) {
51     if (!(e instanceof TypeError)) {
52         fail('#3: TypeError expected but got ' + e);
53     }
54 }
