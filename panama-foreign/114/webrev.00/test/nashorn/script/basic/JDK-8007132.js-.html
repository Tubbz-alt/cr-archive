25 /**
26  * JDK-8007232: Java objects returned from constructor functions are lost
27  *
28  * @test
29  * @run
30  */
31 
32 function Func() {
33     return new java.util.HashMap();
34 }
35 
36 var f = new Func();
37 if (! (f instanceof java.util.Map)) {
38     fail("instance of java.util.Map expected")
39 }
40 
41 function check(obj) {
42     if (typeof obj != 'object') {
43         fail("expected 'object' but got " + (typeof obj));
44     }
45 }
46 
47 // check primitive returning constructors
48 check(new (function() { return 22; }));
49 check(new (function() { return false }));
50 check(new (function() { return "hello" }));
51 
52 // check null and undefined returning constructors
53 check(new (function() { return null; }));
54 check(new (function() { return undefined; }));
