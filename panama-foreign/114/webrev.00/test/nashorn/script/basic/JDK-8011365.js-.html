24 /**
25  * JDK-8011365: Array.prototype.join and Array.prototype.toString do not throw TypeError on null, undefined
26  *
27  * @test
28  * @run
29  */
30 
31 try {
32     Array.prototype.join.call(null, { toString:function() { throw 2 } });
33     fail("should have thrown TypeError");
34 } catch (e) {
35     if (! (e instanceof TypeError)) {
36         fail("TypeError expected, got " + e);
37     }
38 }
39 
40 // check all Array.prototype functions to be sure
41 var names = Object.getOwnPropertyNames(Array.prototype);
42 
43 for (var n in names) {
44     var funcName = names[n];
45     // ignore constructor
46     if (funcName == "constructor") {
47         continue;
48     }
49 
50     var prop = Array.prototype[funcName];
51     if (prop instanceof Function) {
52         // try 'null' this
53         try {
54             prop.call(null);
55             fail(funcName + " does not throw TypeError on 'null' this");
56         } catch (e) {
57             if (! (e instanceof TypeError)) {
58                 fail("TypeError expected from " + funcName + ", got " + e);
59             }
60         }
61 
62         // try 'undefined' this
63         try {
64             prop.call(undefined);
65             fail(funcName + " does not throw TypeError on 'undefined' this");
66         } catch (e) {
67             if (! (e instanceof TypeError)) {
68                 fail("TypeError expected from " + funcName + ", got " + e);
69             }
70         }
71     }
72 }
