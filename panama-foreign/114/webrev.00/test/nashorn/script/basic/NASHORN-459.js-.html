24 /**
25  * NASHORN-459 : Cannot delete configurable accessor properties of global object
26  *
27  * @test
28  * @run
29  */
30 
31 var global = this;
32 Object.defineProperty(global, "0", {
33     get: function() { return 43; },
34     set: function(x) { print('in setter ' + x); },
35     configurable: true
36 });
37 
38 if ((delete global[0]) !== true) {
39     fail("can not delete configurable property '0' of global");
40 }
41 
42 Object.defineProperty(global, "foo", {
43     get: function() { return 33; },
44     set: function(x) { print('setter foo'); },
45     configurable: true
46 });
47 
48 if ((delete global.foo) !== true) {
49     fail("can not delete configurable property 'foo' of global");
50 }
51 
52 Object.defineProperty(global, "1", {
53     get: function() { return 1 },
54     set: undefined,
55     configurable: true
56 });
57 
58 Object.defineProperty(global, "1", {
59     get: function() { return 11 },
60     set: undefined
61 });
62 
63 if ((delete global[1]) !== true) {
64     fail("can not delete configurable property '1' of global");
65 }
