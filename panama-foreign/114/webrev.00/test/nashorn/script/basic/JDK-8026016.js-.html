31 function accessMegamorphic() {
32     for (var i = 0; i < 26; i++) {
33         var o = {};
34         o[String.fromCharCode(i + 97)] = 1;
35         o._;
36     }
37 }

39 function invokeMegamorphic() {
40     for (var i = 0; i < 26; i++) {
41         var o = {};
42         o[String.fromCharCode(i + 97)] = 1;
43         try {
44             o._(i);
45         } catch (e) {
46             print(e);
47         }
48     }
49 }
