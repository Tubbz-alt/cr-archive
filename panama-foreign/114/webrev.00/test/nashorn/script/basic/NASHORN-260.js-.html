24 /**
25  * NASHORN-260 : builtin constructor (Array, RegExp) inlining is broken inside 'with' statements
26  *
27  * @test
28  * @run
29  */
30 
31 var globalArray = Array;
32 var reachedWithArray = false;
33 with ({ Array: function func() { reachedWithArray = true; } }) {
34     var v = new Array();
35 
36     if (! reachedWithArray) {
37         fail("didn't reach with scope's Array func");
38     }
39 
40     if (v instanceof globalArray) {
41         fail("should not have been global's Array");
42     }
43 }
44 
45 var reached = false;
46 with ({ Array: function () { reached = true } }) {
47    with({}) { }
48 
49   var f = new Array();
50 }
51 
52 if (! reached) {
53     fail("not calling with's Array");
54 }
