24 /**
25  * NASHORN-261 : All 'eval' calls assumed to be global builtin eval calls
26  *
27  * @test
28  * @run
29  */
30 
31 // call eval with no args
32 if (eval() !== undefined) {
33     fail("eval with no arg should return undefined");
34 }
35 
36 // call eval with extra (ignored) args
37 function func() {
38     'use strict';
39 
40     try {
41         // we pass hidden args after first arg which is code. Make sure
42         // those stay intact even when user passes extra args.
43         // Example: strict mode flag is passed as hidden arg...
44 
45         eval("eval = 3", "hello", "world", "nashorn");
46         fail("SyntaxError expected!");
47     } catch (e) {
48         if (! (e instanceof SyntaxError)) {
49             fail("SyntaxError expected, got " + e);
50         }
51     }
52 }
53 
54 func();
55 
56 // try calling 'eval' -- but from with scope rather than builtin one
57 with ( {
58     eval: function() {
59         if (arguments.length != 1) {
60             fail("arguments.length !== 1, it is " + arguments.length);
61             for (i in arguments) { print(arguments[i]); }
62         }
63     }
64 }) {
65 
66     eval("hello");
67 }
68 
69 // finally, overwrite builtin 'eval'
70 var reached = false;
71 eval = function() {
72     reached = true;
73     for (i in arguments) {
74         print(arguments[i]);
75     }
76 }
77 
78 // pass no args to our overwritten eval
79 // the new eval should not print anything (no hidden args passed)
80 eval();
81 
82 // make sure our modified eval was called
83 if (! reached) {
84     fail("modified eval was not called");
85 }
