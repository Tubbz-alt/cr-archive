24 /**
25  * JDK-8166902: Nested object literal property maps not reset in optimistic recompilation
26  *
27  * @test
28  * @run
29  */
30 
31 var o = {
32     a: "A",
33     b: "B"
34 };
35 
36 var m = {
37     x: { z: o.a },
38     y: o.b
39 };
40 
41 Assert.assertEquals(m.x.z, "A");
42 Assert.assertEquals(m.y, "B");
