24 /**
25  * Make sure that there are no-enumerable items on standard objects.
26  *
27  * @test
28  * @run
29  */
30 
31 function enumerate(obj, name) {
32     var count = 0;
33     for (i in obj) {
34         print(i);
35         count++;
36     }
37     print("Enumerable items in " + name + " = " + count);
38 }
39 
40 enumerate(Array.prototype, "Array.prototype");
41 enumerate(Array, "Array");
42 enumerate(Boolean.prototype, "Boolean.prototype");
43 enumerate(Boolean, "Boolean");
44 enumerate(Date.prototype, "Date.prototype");
45 enumerate(Date, "Date");
46 enumerate(Function.prototype, "Function.prototype");
47 enumerate(Function, "Function");
48 enumerate(JSON, "JSON");
49 enumerate(Math, "Math");

51 enumerate(Number, "Number");
52 enumerate(Object.prototype, "Object.prototype");
53 enumerate(Object, "Object");
