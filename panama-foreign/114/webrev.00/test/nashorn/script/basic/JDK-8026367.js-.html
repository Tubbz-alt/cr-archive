24 /**
25  * JDK-8026367: Add a sync keyword to mozilla_compat
26  *
27  * @test
28  * @run
29  */
30 
31 if (typeof sync === "undefined") {
32     load("nashorn:mozilla_compat.js");
33 }
34 
35 var obj = {
36     count: 0,
37     // Sync called with one argument will synchronize on this-object of invocation
38     inc: sync(function(d) {
39         this.count += d;
40         Assert.assertTrue(java.lang.Thread.holdsLock(this));
41     }),
42     // Pass explicit object to synchronize on as second argument
43     dec: sync(function(d) {
44         this.count -= d;
45         Assert.assertTrue(java.lang.Thread.holdsLock(obj));
46     }, obj)
47 };
48 
49 var t1 = new java.lang.Thread(function() {
50     for (var i = 0; i < 100000; i++) obj.inc(1);
51 });
52 var t2 = new java.lang.Thread(function() {
53     for (var i = 0; i < 100000; i++) obj.dec(1);
54 });
55 
56 t1.start();
57 t2.start();
58 t1.join();
59 t2.join();
60 
61 if (obj.count !== 0) {
62     throw new Error("Expected count == 0, got " + obj.count);
63 }
