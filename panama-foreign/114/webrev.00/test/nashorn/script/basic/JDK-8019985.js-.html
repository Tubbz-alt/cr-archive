24 /**
25  * JDK-8019985: Date.parse("2000-01-01T00:00:00.Z") should return NaN
26  *
27  * @test
28  * @run
29  */
30 
31 function testFail(str) {
32     if (!isNaN(Date.parse(str))) {
33         throw new Error("Parsed invalid date string: " + str);
34     }
35 }
36 
37 function testOk(str) {
38     if (isNaN(Date.parse(str))) {
39         throw new Error("Failed to parse valid date string: " + str);
40     }
41 }
42 
43 testFail("2000-01-01T00:00:00.Z");
44 testFail("2000-01-01T00:00:Z");
45 testFail("2000-01-01T00:Z");
46 testFail("2000-01-01T00Z");
47 testOk("2000-01-01T00:00:00.000Z");
48 testOk("2000-01-01T00:00:00.0Z");
49 testOk("2000-01-01T00:00:00Z");
50 testOk("2000-01-01T00:00Z");
