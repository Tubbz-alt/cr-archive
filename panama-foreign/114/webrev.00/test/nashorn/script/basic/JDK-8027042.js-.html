32 // var with getter side effect
33 Object.defineProperty(this, "a1", { get: function() {print("get a"); return 1; }});
34 
35 // var with both getter and conversion side effect
36 Object.defineProperty(this, "b1", { get: function() {print("get b"); return {valueOf: function() { print("conv b"); return 10; }}; }});
37 
38 (function() {
39     // var with toPrimitive conversion side effect
40     var c1 = {valueOf: function() { print("conv c"); return 100; }};
41     print("start");
42     print(b1 + (c1 + a1));
43     print("done with first");
44     print(b1 + (c1 + b1));
45     print(b1 + (a1 + b1));
46     print(b1 + (b1 + c1));
47     print(b1 + (b1 + c1));
48     print(b1 + (c1 + (a1 - b1)));
49     print(b1 + (c1 + (c1 - b1)));
50     print(b1 + (c1 + (b1 - c1)));
51     print(b1 + (b1 + (a1 ? 2 : 3)));
52     print(b1 + (b1 + (b1 ? 2 : 3)));
53     print(b1 + (b1 + (c1 ? 2 : 3)));
54     print(b1 + ((-c1) + (-a1)));
55     print(b1 + ((-c1) + (-b1)));
56     print(b1 + ((-c1) + (-c1)));
57     try { print(b1 + new a1); } catch (e) {}
58     try { print(b1 + new b1); } catch (e) {}
59     try { print(b1 + new c1); } catch (e) {}
60 })();
