24 /**
25  * JDK-8201466: Nashorn: defineProperty setters/getters on prototype object ignored with numeric property names
26  *
27  * @test
28  * @run
29  */
30 
31 var z = {}, s = null, g = null;
32 
33 Object.defineProperty(z,'0', {
34     get:function(){ g = 2; return 1;},
35     set:function(v){ s = v;}
36 });
37 
38 var x = Object.create(z);
39 x[0] = 3;
40 
41 Assert.assertTrue(x[0] === 1);
42 Assert.assertTrue(g === 2);
43 Assert.assertTrue(s === 3);
