24 /**
25  * JDK-8010924:  Dealing with undefined property gets you a fatal stack
26  *
27  * @test
28  * @run
29  * @option -scripting
30  */
31 
32 load("nashorn:mozilla_compat.js");
33 
34 if (this.non_existent_foo !== undefined) {
35     fail("this.non_existent_foo is defined!");
36 }
37 
38 try {
39     non_existent_foo;
40     fail("should have thrown ReferenceError");
41 } catch (e) {
42     if (! (e instanceof ReferenceError)) {
43         fail("ReferenceError expected, got " + e);
44     }
45 }
46 
47 // try the same via script engine
48 
49 var ScriptEngineManager = Java.type("javax.script.ScriptEngineManager");
50 var engine = new ScriptEngineManager().getEngineByName("nashorn");
51 
52 engine.eval("load('nashorn:mozilla_compat.js')");
53 
54 if (! engine.eval("this.non_existent_foo === undefined")) {
55     fail("this.non_existent_foo is not undefined");
56 }
57 
58 engine.eval(<<EOF
59     try {
60         non_existent_foo;
61         throw new Error("should have thrown ReferenceError");
62     } catch (e) {
63         if (! (e instanceof ReferenceError)) {
64             throw new Error("ReferenceError expected, got " + e);
65         }
66     }
67 EOF);
