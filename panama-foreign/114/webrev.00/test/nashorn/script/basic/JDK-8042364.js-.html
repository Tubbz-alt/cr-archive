31 // check for Object.prototype.__proto__ accessor property
32 print("Object.prototype has __proto__?",
33     Object.prototype.hasOwnProperty("__proto__"))
34 
35 var desc = Object.getOwnPropertyDescriptor(Object.prototype, "__proto__")
36 print("descriptor");
37 print(JSON.stringify(desc))
38 print("getter", desc.get)
39 print("setter", desc.set)
40 
41 // no computed "__proto__" name, only identifier!
42 var p = {}
43 var obj = {
44     "__proto__" : p
45 }
46 
47 if (Object.getPrototypeOf(obj) === p) {
48     fail("obj has wrong __proto__, allows computed __proto__!")
49 }
50 
51 if (obj.__proto__ !== p) {
52     fail("__proto__ not created as normal property!")
53 }
54 
55 if (Object.getPrototypeOf(obj) !== Object.prototype) {
56     fail("obj has wrong __proto__")
57 }
58 
59 var obj2 = {
60     __proto__: p
61 }
62 
63 if (Object.getPrototypeOf(obj2) !== p) {
64     fail("can't set __proto__ in object literal")
