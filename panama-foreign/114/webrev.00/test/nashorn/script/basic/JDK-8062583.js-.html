24 /**
25  * JDK-8062583: Throwing object with error prototype causes error proto to be caught
26  *
27  * @test
28  * @run
29  */
30 
31 function CustomError() {
32     this.name = "CustomError";
33 }
34 
35 CustomError.prototype = new Error();
36 
37 var c1 = new CustomError();
38 
39 try {
40     throw c1;
41 } catch (e) {
42     print(e === c1);
43     print(e === CustomError.prototype);
44     print(e.stack.replace(/\\/g, '/'));
45     print(e.nashornException.toString().replace(/\\/g, '/'));
46 }
47 
48 var c2 = new CustomError();
49 Error.captureStackTrace(c2);
50 print(c2.stack.replace(/\\/g, '/'));
51 print(c2.nashornException.toString().replace(/\\/g, '/'));
