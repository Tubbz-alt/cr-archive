24 /**
25  * JDK-8014785: Ability to extend global instance by binding properties of another object
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = { x: 34, y: 100 };
32 var foo = {}
33 
34 // bind properties of "obj" to "foo" obj
35 Object.bindProperties(foo, obj);
36 
37 // now we can access/write on foo properties
38 print("foo.x = " + foo.x); // prints obj.x which is 34
39 
40 // update obj.x via foo.x
41 foo.x = "hello";
42 print("obj.x = " + obj.x); // prints "hello" now
43 
44 obj.x = 42;   // foo.x also becomes 42
45 print("obj.x = " + obj.x); // prints 42
46 print("foo.x = " + foo.x); // prints 42
47 
48 // now bind a mirror object to an object
49 var obj = loadWithNewGlobal({
50   name: "test",
51   script: "obj = { x: 33, y: 'hello' }"
52 });
53 
54 Object.bindProperties(this, obj);
55 print("x = " + x); // prints 33
56 print("y = " + y); // prints "hello"
57 
58 x = Math.PI;               // changes obj.x to Math.PI
59 print("obj.x = " +obj.x);  // prints Math.PI
60 
61 obj.y = 32;
62 print("y = " + y);  // should print 32
