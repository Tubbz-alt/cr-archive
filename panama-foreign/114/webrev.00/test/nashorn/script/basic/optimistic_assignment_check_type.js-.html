24 /**
25  * @test
26  * @bug 8036987, 8037572
27  * @summary Implement tests that checks static types in the compiled code
28  * @option --optimistic-types=true
29  * @run
30  */
31 
32 var inspect = Java.type("jdk.nashorn.test.tools.StaticTypeInspector").inspect
33 var a = b = 3;
34 var c;
35 var x = { a: 2, b:1, c: 7, d: -1, e: 1}
36 var y = { a: undefined, b: undefined}
37 
38 // Testing assignment operators
39 //-- Global variable
40 print(inspect(a=c, "global int assignment to global variable"))
41 print(inspect(a=b, "undefined assignment to global int"))
42 print(inspect(a=y.a, "global int assignment to undefined"))
43 print(inspect(a+=b, "undefined addition assignment to global int"))
44 print(inspect(b=b+b, "global int addition global int"))
45 print(inspect(b+= y.a, "global int addition assignment undefined"))
46 //--Local variable
47 print(inspect(x.a+= y.a, "local int addition assignment local undefined"))
48 print(inspect(x.b=y.a, "local int assignment to undefined"))
49 print(inspect(y.a+=y.a, "local undefined addition assignment local undefined"))
50 print(inspect(x.c-=x.d, "local int substraction assignment local int"))
51 print(inspect(x.c*=x.d, "local int multiplication assignment local int"))
52 print(inspect(x.c/=x.d, "local int division assignment local int"))
53 print(inspect(y.b=x.c, "local undefined assignment to local int"))
54 print(inspect(y.c=x.c, "local boolean assignment to local int"))
