24 /**
25  * NASHORN-443 : Array.prototype.lastIndexOf does not handle the cases where 'fromIndex' is undefined
26  *
27  * @test
28  * @run
29  */
30 
31 var arr = [ 'hello', 'world', 'hello', undefined ];
32 
33 if (arr.lastIndexOf('hello') !== 2) {
34     fail("#1 lastIndexOf fails when fromIndex is not passed");
35 }
36 
37 if (arr.lastIndexOf('world', undefined) !== -1) {
38     fail("#2 lastIndexOf fails when fromIndex is undefined");
39 }
40 
41 if (arr.lastIndexOf('hello', undefined) !== 0) {
42     fail("#3 lastIndexOf fails when fromIndex is undefined");
43 }
44 
45 if (arr.lastIndexOf() !== 3) {
46     fail("#4 lastIndexOf fails when searchElement is not passed");
47 }
