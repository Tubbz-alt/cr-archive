24 /**
25  * JDK-8011421: When using Object.defineProperty on arrays, PropertyDescriptor's property accessors are invoked multiple times
26  *
27  * @test
28  * @run
29  */
30 
31 var configurableGetterCalled = 0;
32 
33 // create a property descriptor object with "configurable"
34 // property with a user defined getter
35 var propDesc = Object.defineProperty({},
36     "configurable",
37     {
38          get: function() {
39             configurableGetterCalled++;
40             return false
41          }
42     }
43 );
44 
45 // make array length non-configurable
46 Object.defineProperty([], "length", propDesc);
47 
48 // above should have called "configurable" getter only once
49 if (configurableGetterCalled !== 1) {
50     fail("defineProperty on array should call propDesc getters only once");
51 }
