24 /**
25  * NASHORN-437 : Array.prototype.map should return an array whose length should be same as input array length
26  *
27  * @test
28  * @run
29  */
30 
31 var callbackCount = 0;
32 
33 function func(kVal, k, arr) {
34     arr.length = 2;
35     callbackCount++;
36     return 42;
37 }
38 
39 var arr = [ 1, 2, 3, 4];
40 var res = arr.map(func);
41 if (res.length !== 4) {
42    fail("map's result does not have 4 elements");
43 }
44 
45 if (callbackCount !== 2) {
46     fail("callback should be called only twice");
47 }
48 
49 if (res[2] !== undefined || res[3] !== undefined) {
50     fail("last two elements are not undefined");
51 }
