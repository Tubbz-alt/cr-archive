24 /**
25  * Basic checks for Date.parse function.
26  *
27  * @test
28  * @option -timezone=Asia/Calcutta
29  * @run
30  */
31 
32 // ISO format
33 var d = new Date(Date.parse("1972-06-16T00:00:00.000Z"));
34 print(d.toString());
35 print(d.toUTCString());
36 
37 // simple Date
38 d = new Date(Date.parse("2009-01-01"));
39 print(d.toString());
40 print(d.toUTCString());
41 
42 // simple date and make sure we can parse back toString, toISOString
43 // and toUTCString output ..
44 d = new Date(2011, 4, 11);
45 d = new Date(Date.parse(d.toString()));
46 print(d.toString());
47 print(d.toUTCString());
48 
49 d = new Date(Date.parse(d.toISOString()));
50 print(d.toString());
51 print(d.toUTCString());
52 
53 d = new Date(Date.parse(d.toUTCString()));
54 print(d.toString());
55 print(d.toUTCString());
