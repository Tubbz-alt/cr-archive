24 /**
25  * NASHORN-378 : Redefining a configurable accessor property to be a data property on a non-extensible object fails with NPE
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = {};
32 Object.defineProperty(obj, "foo",
33     { get: function() { return 42; },
34       configurable: true});
35 
36 if (obj.foo !== 42) {
37     fail("#0 obj.foo !== 42");
38 }
39 
40 Object.preventExtensions(obj);
41 Object.defineProperty(obj, "foo", { value: "hello"});
42 
43 var desc = Object.getOwnPropertyDescriptor(obj, "foo");
44 
45 if (obj.foo !== "hello") {
46     fail("#1 obj.foo !== 'hello'");
47 }
48 
49 if (desc.get !== undefined) {
50     fail("#2 getter found on a date property");
51 }
52 
53 if (desc.set !== undefined) {
54     fail("#3 setter found on a date property");
55 }
56 
57 if (desc.value !== 'hello') {
58     fail("#4 value property is wrong");
59 }
