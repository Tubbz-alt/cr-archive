24 /**
25  * NASHORN-665 : delete on string index should return false
26  *
27  * @test
28  * @run
29  */
30 
31 function func(str) {
32    'use strict';
33 
34    try {
35        delete str[0];
36        fail("expected TypeError");
37    } catch (e) {
38        if (! (e instanceof TypeError)) {
39            fail("expected TypeError, got " + e);
40        }
41    }
42 }
43 
44 func(new String("hello"));
45 func("hello");
46 
47 var str = new String("hello");
48 if (delete str[0] != false) {
49     fail("delete str[0] does not result in false");
50 }
51 
52 str = "hello";
53 if (delete str[0] != false) {
54     fail("delete str[0] does not result in false");
55 }
