24 /**
25  * NASHORN-226 : Error.prototype.toString does not confirm to spec.
26  *
27  * @test
28  * @run
29  */
30 
31 var e = new Error();
32 if (e.toString() !== "Error") {
33     fail("#1 expected 'Error', got " + e.toString());
34 }
35 
36 e.message = "";
37 if (e.toString() !== "Error") {
38     fail("#2 expected 'Error', got " + e.toString());
39 }
40 
41 e.message = { toString: function() { return '' } };
42 if (e.toString() !== "Error") {
43     fail("#3 expected 'Error', got " + e.toString());
44 }
45 
46 e.name = "MyError";
47 e.message = "";
48 if (e.toString() !== "MyError") {
49     fail("#4 expected 'MyError', got " + e.toString());
50 }
