24 /**
25  * JDK-8068901: Surprising behavior with more than one functional interface on a class
26  *
27  * @test
28  * @run
29  */
30 
31 var Consumer = java.util.function.Consumer;
32 var JFunction = java.util.function.Function;
33 
34 var fc = new (Java.extend(JFunction, Consumer))({
35     apply: function(x) { print("fc invoked as a function") },
36     accept: function(x) { print("fc invoked as a consumer") }
37 });
38 
39 var c = new Consumer(function(x) { print("c invoked as a consumer") });
40 
41 var cf = new (Java.extend(Consumer, JFunction))({
42     apply: function(x) { print("cf invoked as a function") },
43     accept: function(x) { print("cf invoked as a consumer") }
44 });
45 
46 var f = new JFunction(function(x) { print("f invoked as a function") });
47 
48 for each(x in [fc, c, fc, cf, f, cf, c, fc, f, cf]) { x(null); }
