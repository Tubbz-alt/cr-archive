24 /**
25  * JDK-8024255: When a keyword is used as object property name, the property can not be deleted
26  *
27  * @test
28  * @run
29  */
30 
31 function check(obj, name) {
32     var desc = Object.getOwnPropertyDescriptor(obj, name);
33     if (! desc.configurable) {
34         fail("Property " + name + " is not configurable");
35     }
36 
37     if (! (delete obj[name])) {
38         fail("Property " + name + " can not be deleted");
39     }
40 }
41 
42 var obj = {
43     default: 344,
44     in: 'hello',
45     if: false,
46     class: 4.223
47 }
48 
49 for (var p in obj) {
50     check(obj, p);
51 }
