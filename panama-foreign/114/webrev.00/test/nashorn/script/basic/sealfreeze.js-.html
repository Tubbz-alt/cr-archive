24 /**
25  * Basic checks for Object.seal and Object.freeze.
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = { num: Math.E };
32 
33 print("sealed? " + Object.isSealed(obj));
34 print("frozen? " + Object.isFrozen(obj));
35 
36 Object.seal(obj);
37 print("sealed? " + Object.isSealed(obj));
38 
39 // delete should fail - because we have sealed
40 print("deleted num? " + (delete obj.num));
41 print("num = " + obj.num);
42 
43 // assignment should succeed
44 print("assign PI to num");
45 obj.num = Math.PI;
46 print("num = " + obj.num);
47 
48 // assignment should fail -- because we freeze
49 Object.freeze(obj);
50 print("frozen? " + Object.isFrozen(obj));
51 
52 print("assign 1729 to num");
53 obj.num = 1729;
54 print("num = " + obj.num);
