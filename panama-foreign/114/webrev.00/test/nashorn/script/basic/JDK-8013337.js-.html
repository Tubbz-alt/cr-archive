24 /**
25  * JDK-8013337: Issues with Date.prototype's get, set functions
26  *
27  * @test
28  * @option -timezone=Asia/Calcutta
29  * @run
30  */
31 
32 function check(str) {
33     print(str + " = " + eval(str));
34 }
35 
36 check('new Date(NaN).setFullYear(NaN)');
37 check('new Date(0).setYear(70)');
38 check('new Date(0).setYear(NaN)');
39 check('new Date(NaN).setYear(70)');
40 check('new Date(NaN).getTimezoneOffset()');
41 
42 function checkGetterCalled(func) {
43     var getterCalled = false;
44 
45     new Date(NaN)[func]( { valueOf: function() { getterCalled = true } } );
46 
47     if (getterCalled) {
48        print("Date.prototype." + func + " calls valueOf on arg");
49     }
50 }
51 
52 checkGetterCalled("setMilliseconds");
53 checkGetterCalled("setUTCMilliseconds");
54 checkGetterCalled("setSeconds");
55 checkGetterCalled("setUTCSeconds");
56 checkGetterCalled("setMinutes");
57 checkGetterCalled("setUTCMinutes");
58 checkGetterCalled("setHours");
59 checkGetterCalled("setUTCHours");
60 checkGetterCalled("setDate");
61 checkGetterCalled("setUTCDate");
62 checkGetterCalled("setMonth");
63 checkGetterCalled("setUTCMonth");
64

66     Date.prototype.setTime.call({}, { valueOf: function() { throw "err" } })
67 } catch (e) {
68     if (! (e instanceof TypeError)) {
69         fail("TypeError expected, got " + e);
70     }
