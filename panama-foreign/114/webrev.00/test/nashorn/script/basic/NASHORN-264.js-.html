24 /**
25  * NASHORN-264 : If the value of user getter property is used as function when it is not, we get ClassCastException instead of TypeError
26  *
27  * @test
28  * @run
29  */
30 
31 function func() {
32     var obj = {};
33     Object.defineProperty(obj, "bar", {
34         get: function()  { return 'hello'; }
35     });
36 
37     try {
38         obj.bar();
39         fail("should have thrown TypeError, obj.bar is not a function");
40     } catch(e) {
41         if (! (e instanceof TypeError)) {
42             fail("expected TypeError got " + e);
43         }
44     }
45 }
46 
47 func();
