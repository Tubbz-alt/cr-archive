24 /**
25  * JDK-8025520: Array.prototype.slice should only copy defined elements
26  *
27  * @test
28  * @run
29  */
30 
31 var s = Array.prototype.slice.call({length: 6, 3: 1}, 2, 5);
32 
33 if (s.length != 3) {
34     fail("s.length != 3");
35 }
36 if (0 in s) {
37     fail("0 in s");
38 }
39 if (s.hasOwnProperty(0)) {
40     fail("s.hasOwnProperty(0)");
41 }
42 if (s[1] !== 1) {
43     fail("s[1] !== 1");
44 }
45 if (2 in s) {
46     fail("2 in s");
47 }
48 if (s.hasOwnProperty(2)) {
49     fail("s.hasOwnProperty(2)");
50 }
