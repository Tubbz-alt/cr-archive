24 /**
25  * JDK-8024619: JDBC java.sql.DriverManager is not usable from JS script
26  *
27  * @test
28  * @run
29  */
30 
31 var DriverManager = Java.type("java.sql.DriverManager");
32 var e = DriverManager.getDrivers();
33 
34 var driverFound = false;
35 // check for Nashorn SQL driver
36 while (e.hasMoreElements()) {
37     var driver = e.nextElement();
38     if (driver.acceptsURL("jdbc:nashorn:")) {
39         driverFound = true;
40         break;
41     }
42 }
43 
44 if (! driverFound) {
45     fail("Nashorn JDBC Driver not found!");
46 }
