25 /*
26  * NASHORN-188 : Nashorn uses locale for Number.prototype.toPrecision()
27  *
28  * @test
29  * @run
30  */
31 
32 function format(value) {
33     print(value.toFixed());
34     print(value.toFixed(1));
35     print(value.toFixed(2));
36     print(value.toFixed(3));
37     print(value.toFixed(10));
38 
39     print(value.toExponential());
40     print(value.toExponential(1));
41     print(value.toExponential(2));
42     print(value.toExponential(3));
43     print(value.toExponential(10));
44 
45     print(value.toPrecision());
46     print(value.toPrecision(1));
47     print(value.toPrecision(2));
48     print(value.toPrecision(3));
49     print(value.toPrecision(10));
50 
51     print();
52 }
53 
54 format(0);
55 format(1/10);
56 format(1/12);
57 format(1);
58 format(123456789000);
59 format(123456.123456);
60 format(-1/10);
61 format(-1/12);
62 format(-1);
63 format(-123456789000);
64 format(-123456.123456);
65 format(NaN);
66 format(Infinity);
