24 /**
25  * JDK-8010731: Nashorn exposes internal symbols such as __callee__, __scope__ to scripts
26  *
27  * @test
28  * @run
29  */
30 
31 function checkCallee() {
32      var x = arguments[0]; // force __callee__ (renamed as :callee)
33 
34      print(__callee__);
35 }
36 
37 try {
38     checkCallee();
39     fail("Should have thrown ReferenceError for __callee__");
40 } catch (e) {
41     if (! (e instanceof ReferenceError)) {
42         fail("ReferenceError expected, got " + e);
43     }
44 }
45 
46 function checkScope() {
47     var x = 334;
48 
49     function inner() {
50         var y = x * x;  // force __scope__ (renamed as :scope")
51         print(__scope__);
52     }
53 
54     inner();
55 }
56 
57 try {
58     checkScope();
59     fail("Should have thrown ReferenceError for __scope__");
60 } catch (e) {
61     if (! (e instanceof ReferenceError)) {
62         fail("ReferenceError expected, got " + e);
63     }
64 }
