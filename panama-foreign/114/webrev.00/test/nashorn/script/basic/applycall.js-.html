24 /**
25  * Verify "length" property of functions is arity.
26  * Verify that function can be called by apply and call.
27  *
28  * @test
29  * @run
30  */
31 
32 function func(x, y) { print(x); print(y); }
33 
34 print(func.call);
35 print("func.call.length = " + func.call.length);
36 
37 func.call(this, "hello, ", "world");
38 func.apply(this, [ "hello, " , "world" ]);
39 func("hello, ", "world");
40 
41 // extension: you can pass java List to apply
42 
43 var list = new java.util.ArrayList();
44 list.add("invokedynamic");
45 list.add(" is great!");
46 func.apply(this, list);
47 
48 print("func.apply.length = " + func.apply.length);
49 
50 function func2() {
51     print("I am func2");
52 }
53 
54 func2.apply(this);
55 
56 func.apply(this);
