26  *
27  * @test
28  * @option --locale=tr-TR
29  * @run
30  */
31 
32 if ("i".toUpperCase() != "I") {
33     fail("'i'.toUpperCase() is not 'I'");
34 }
35 
36 if ("i".toUpperCase() == "i".toLocaleUpperCase()) {
37     fail("'i'.toUpperCase() == 'i'.toLocaleUpperCase()");
38 }
39 
40 if ("I".toLowerCase() != "i") {
41     fail("'I'.toLowerCase() is not 'i'");
42 }
43 
44 if ("I".toLowerCase() == "I".toLocaleLowerCase()) {
45     fail("'i'.toLowerCase() == 'i'.toLocaleLowerCase()");
