24 /**
25  * JDK-8181191: getUint32 returning Long
26  *
27  * @test
28  * @run
29  */
30 
31 
32 function uint32(x) {
33     var buffer = new ArrayBuffer(16);
34     var dataview = new DataView(buffer);
35     dataview.setUint32(0, x);
36     return dataview.getUint32(0);
37 }
38 
39 Assert.assertTrue(typeof uint32(0x7f) === 'number');
40 Assert.assertTrue(typeof uint32(0x80) === 'number');
41 Assert.assertTrue(typeof uint32(0xffffffff) === 'number');
42 Assert.assertTrue(typeof uint32(0x100000000) === 'number');
43 
44 Assert.assertTrue(uint32(0x7f) === 0x7f);
45 Assert.assertTrue(uint32(0x80) === 0x80);
46 Assert.assertTrue(uint32(0xffffffff) === 0xffffffff);
47 Assert.assertTrue(uint32(0x100000000) === 0x0);
48 
49 Assert.assertTrue(uint32(0x7f) === uint32(0x7f));
50 Assert.assertTrue(uint32(0x80) === uint32(0x80));
51 Assert.assertTrue(uint32(0xffffffff) === uint32(0xffffffff));
52 Assert.assertTrue(uint32(0x100000000) === uint32(0x100000000));
