25  * Check that trying to operate on results in exception.
26  *
27  * @test
28  * @run
29  */
30 
31 var x;
32 
33 print("x undefined? " + (x === undefined));
34 
35 try {
36     print(x.foo);
37 } catch (e) {
38     print(e);
39 }
40 
41 try {
42     x.bar = 34;
43 } catch (e) {
44     print(e);
45 }
46 
47 try {
48     print(x[1]);
49 } catch (e) {
50     print(e);
51 }
52 
53 try {
54     x[0] = 12;
55 } catch (e) {
56     print(e);
57 }
58

60     x.func();

64

66     delete x[0];

78     print(Object.getPrototypeOf(x));

80     print(e);
