24 /**
25  * JDK-8019947: inherited property invalidation does not work with two globals in same context
26  *
27  * @test
28  * @option -scripting
29  * @run
30  */
31 
32 function func(arr) {
33     try {
34        print(arr.toString());
35     } catch (e) {
36        print(e.stack);
37     }
38 }
39 
40 var arr = ["hello", "world"]
41 
42 func(arr);
43 
44 var global = loadWithNewGlobal({
45    name: "t",
46    script: <<EOF
47 
48 
49 function func(arr) {
50    try {
51        print(arr.toString());
52    } catch (e) {
53        print(e.stack);
54    }
55 }
56 
57 var arr = [1, 2, 3];
58 
59 func(arr);
60 
61 delete Array.prototype.toString;
62 
63 // Object.prototype.toString should be visible
64 // after Array.prototype.toString is deleted.
65 func(arr);
66 this;
67 EOF
68 });
