24 /**
25  * NASHORN-266 : Strict mode functions see scope call 'this' as undefined
26  *
27  * @test
28  * @run
29  */
30 
31 
32 function func() {
33     'use strict';
34     return this;
35 }
36 
37 if (func() !== undefined) {
38     fail("#1 'this' should be undefined");
39 }
40 
41 if (this.func() !== this) {
42     fail("#2 'this' should be global");
43 }
44 
45 function func2() {
46     return this;
47 }
48 
49 if (func2() === undefined) {
50     fail("#3 'this' should not be undefined");
51 }
52 
53 if (this.func2() !== this) {
54     fail("#4 'this' should be global");
55 }
56 
57 // strict calling non-strict
58 function func3() {
59     'use strict';
60 
61     if (func2() === undefined) {
62         fail("#5 'this' should not be undefined");
63     }
64 }
65 
66 func3();
