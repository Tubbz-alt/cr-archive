24 /**
25  * JDK-8009553: Object.create(Array.prototype) doesn't respect reset length
26  *
27  * @test
28  * @run
29  */
30 
31 var desc = Object.getOwnPropertyDescriptor(Array.prototype, "length");
32 if (! desc.writable) {
33     fail("Array.prototype.length is not writable");
34 }
35 
36 if (desc.value != 0) {
37     fail("Array.prototype.length != 0");
38 }
39 
40 var array = Object.create(Array.prototype);
41 if (array.hasOwnProperty("length")) {
42     fail("array has 'length'");
43 }
44 
45 if (array.length != 0) {
46     fail("array.length != 0");
47 }
48 
49 array.length = 4;
50 if (! array.hasOwnProperty("length")) {
51     fail("array does not have 'length'");
52 }
53 
54 if (array.length != 4) {
55     fail("array.length != 4");
56 }
