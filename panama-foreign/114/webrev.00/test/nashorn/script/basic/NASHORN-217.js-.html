24 /**
25  * NASHORN-217 :  getter and setter name in object literal could be string, number, reserved word as well
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = {
32    get class() { return 44; },
33    get in() { return 'hello' },
34    get typeof() { return false; },
35    set class(x) { print('class ' + x); },
36    set in(x) { print('in ' + x); },
37    set typeof(x) { print('typeof ' + x); },
38    get 14() { return 14; },
39    set 14(x) { print("14 " + x); },
40    get 'foo prop'() { return 'foo prop value' },
41    set 'foo prop'(x) { print('foo prop ' + x); }
42 };
43 
44 print('obj.class ' + obj.class);
45 print('obj.in ' + obj.in);
46 print('obj.typeof ' + obj.typeof);
47 print('obj[14] ' + obj[14]);
48 print("obj['foo prop'] " + obj['foo prop']);
49 
50 obj.class = 3;
51 obj.in = 'world';
52 obj.typeof = 34;
53 obj[14] = 34;
54 obj['foo prop'] = 'new foo prop';
