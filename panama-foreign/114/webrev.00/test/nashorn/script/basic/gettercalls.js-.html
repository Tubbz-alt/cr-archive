24 /**
25  * A getter method returns function typed value. When this property is called,
26  * we should call getter every time and then call the resulting function.
27  *
28  * @test
29  * @run
30  */
31 
32 var count = 0;
33 
34 var obj = {
35   get func() {
36      print("getter for func");
37      return function() {
38         return count++;
39      }
40   }
41 };
42 
43 for (i = 0; i < 10; i++) {
44    print(obj.func());
45 }
46 
47 print("count = " + count);
