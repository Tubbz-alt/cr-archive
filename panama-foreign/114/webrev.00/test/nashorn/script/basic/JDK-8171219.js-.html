24 /**
25  * JDK-8171219: Missing checks in sparse array shift() implementation
26  *
27  * @test
28  * @run
29  */
30 
31 var a = [];
32 a[1048577] = 1;
33 a.shift();
34 a[1] = 2;
35 a.shift();
36 var ka = Object.keys(a);
37 Assert.assertTrue(ka.length === 2);
38 Assert.assertTrue(ka[0] === '0');
39 Assert.assertTrue(ka[1] === '1048575');
40 Assert.assertTrue(a.length === 1048576);
41 Assert.assertTrue(a[0] === 2);
42 Assert.assertTrue(a[1048575] = 1);
43 
44 var b = [];
45 b[1048577] = 1;
46 b.unshift(2);
47 b.shift();
48 b[1] = 3;
49 b.shift();
50 var kb = Object.keys(b);
51 Assert.assertTrue(kb.length === 2);
52 Assert.assertTrue(kb[0] === '0');
53 Assert.assertTrue(kb[1] === '1048576');
54 Assert.assertTrue(b.length === 1048577);
55 Assert.assertTrue(b[0] === 3);
56 Assert.assertTrue(b[1048576] = 1);
