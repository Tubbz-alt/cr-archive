24 /**
25  * NASHORN-212 :  Object literal property redefinition restrictions are not implemented
26  *
27  * @test
28  * @run
29  */
30 
31 try {
32     eval("obj = { x : 3, get x() { return 42; }");
33     fail("data property redefined as accessor");
34 } catch (e) {
35     if (! (e instanceof SyntaxError)) {
36         fail("SyntaxError expected got " + e);
37     }
38 }
39 
40 try {
41     eval("obj = { get x() { return 42; }, x: 'hello'");
42     fail("accessor property redefined as data");
43 } catch (e) {
44     if (! (e instanceof SyntaxError)) {
45         fail("SyntaxError expected got " + e);
46     }
47 }
48 
49 try {
50     eval("obj = { get x() { return 42; }, get x() { return 'hello' }");
51     fail("accessor property redefined with different getter");
52 } catch (e) {
53     if (! (e instanceof SyntaxError)) {
54         fail("SyntaxError expected got " + e);
55     }
56 }
57 
58 try {
59     eval("obj = { set x(val) { }, set x(val) { }");
60     fail("accessor property redefined with different setter");
61 } catch (e) {
62     if (! (e instanceof SyntaxError)) {
63         fail("SyntaxError expected got " + e);
64     }
65 }
66 
67 // data redefinition is fine
68 obj = { x: 44, x: 'hello' };
69 
70 // can define getter and setter
71 obj = { get x() { return 3; }, set x(val) {} };
