24 /**
25  * JDK-8011237: Object.isExtensible(Object.getOwnPropertyDescriptor(function(){"use strict"},"caller").get) should be false
26  *
27  * @test
28  * @run
29  */
30 
31 // ECMA Section 13.2.3 The [[ThrowTypeError]] Function Object
32 // 11. Set the [[Extensible]] internal property of F to false
33 
34 var strictFunc = (function() { 'use strict' });
35 var strictFuncCallerDesc = Object.getOwnPropertyDescriptor(strictFunc, "caller")
36 var isExtensible = Object.isExtensible(strictFuncCallerDesc.get);
37 if (isExtensible) {
38     fail("strict function caller's getter is extensible!");
39 }
