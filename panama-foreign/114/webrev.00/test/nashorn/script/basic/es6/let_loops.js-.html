34 function tryIt (code) {
35     try {
36         eval(code)
37     } catch (e) {
38         print(e)
39     }
40 }
41 
42 tryIt(<<CODE
43       let a = 2;
44       do {
45         a--;
46         let b = a;
47       } while (a > 0);
48       print(a)
49       print(b)
50 CODE)
51 
52 tryIt(<<CODE
53        let a = 2
54        while(a > 0) {
55             a--
56             let b = a
57        }
58        print(a)
59        print(b)
60 CODE)
61 
62 tryIt(<<CODE
63        let a = 2
64        while(a > 0) {
65             a--
66             const b = a
67        }
68        print(a)
69        print(b)
70 CODE)
71 
72 tryIt(<<CODE
73        let a = 2;

80 CODE)
