24 /**
25  * JDK-8151811: Const declarations do not work in for..in loops
26  *
27  * @test
28  * @run
29  * @option --language=es6
30  */
31 
32 let array = ["a", "b", "c"];
33 let count = 0;
34 
35 for (const i in array) {
36     try {
37         eval("i = 5");
38         fail("const assignment should have thrown")
39     } catch (e) {
40         Assert.assertTrue(e instanceof TypeError);
41     }
42     Assert.assertTrue(i == count++);
43 }
44 
45 let funcs = [];
46 
47 for (const i in array) {
48     try {
49         eval("i = 5");
50         fail("const assignment should have thrown")
51     } catch (e) {
52         Assert.assertTrue(e instanceof TypeError);
53     }
54     for (const j in array) {
55         for (const k in array) {
56             funcs.push(function () {
57                 return array[i] + array[j] + array[k];
58             });
59         }
60     }
61 }
62 
63 Assert.assertEquals(funcs.length, 3 * 3 * 3);
64 count = 0;
65 
66 for (let i = 0; i < 3; i++) {
67     for (let j = 0; j < 3; j++) {
68         for (let k = 0; k < 3; k++) {
69             Assert.assertEquals(funcs[count++](), array[i] + array[j] + array[k]);
70         }
71     }
72 }
