24 /**
25  * JDK-8156743: ES6 for..of should work for Java Maps and Sets
26  *
27  * @test
28  * @run
29  * @option --language=es6
30  */
31 
32 var map = new Map([
33     [1, 'one'],
34     [2, 'two'],
35     [3, 'three']
36 ]);
37 
38 var set = new Set(['red', 'green', 'blue']);
39 
40 var HashMap = Java.type("java.util.HashMap");
41 var jmap = new HashMap();
42 jmap.put(1, 'one');
43 jmap.put(2, 'two');
44 jmap.put(3, 'three');
45 
46 var HashSet = Java.type("java.util.HashSet");
47 var jset = new HashSet();
48 jset.add('red');
49 jset.add('green');
50 jset.add('blue');
51 
52 for(var keyvalue of map){
53     print(keyvalue[0],keyvalue[1]);
54 }
55 
56 for(var keyvalue of jmap){
57     print(keyvalue[0],keyvalue[1]);
58 }
59 
60 for(var keyvalue of map){
61     print(keyvalue);
62 }
63 
64 for(var keyvalue of jmap){
65     print(keyvalue);
66 }
67 
68 for(var element of set){
69     print(element);
70 }
71 
72 for(var element of jset){
73     print(element);
74 }
