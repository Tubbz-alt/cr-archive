33 function tryIt (code) {
34     try {
35         eval(code)
36     } catch (e) {
37         print(String(e).replace(/\\/g, "/"))
38     }
39 }
40 
41 tryIt(<<CODE
42     "use strict";
43     let x = 2;
44     const x = function (a,b,c) {};
45 CODE)
46 
47 tryIt(<<CODE
48     "use strict";
49     let x = {};
50     var x = 2;
51 CODE)
52 
53 tryIt(<<CODE
54     "use strict";
55     var x = 2;
56     let x = undefined;
57 CODE)
58 
59 tryIt(<<CODE
60     "use strict";
61     const x = function (){};
62     let x = {};
63 CODE)
64

67     "use strict";
68     let a = 2;
69     function a () {};
70 CODE)
