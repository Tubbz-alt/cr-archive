33 function tryIt (code) {
34      try {
35          eval(code)
36      } catch (e) {
37          print(String(e).replace(/\\/g, "/"))
38      }
39 }
40 
41 tryIt(<<CODE
42     let a = 23
43     if (true) {
44         a--
45         let a = 43;
46     }
47 CODE)
48 
49 tryIt(<<CODE
50     const a = 23
51     if (true) {
52         a--
53         const a = 43;
54     }
55 CODE)
