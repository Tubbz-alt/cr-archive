25 /**
26  * NASHORN-14 :  Function.apply and call do not work with varargs in callee.
27  *
28  * @test
29  * @run
30  */
31 
32 
33 function callback() {
34     print('callback with args ' + Array.prototype.join.apply(arguments))
35 }
36 
37 var cb = callback;
38 cb.call(this, 's', 10, 13.1, true, false, null, {}, [], [1,2,{}]);
39 cb.apply(this, ['s', 10, 13.1, true, false, null, {}, [], [1,2,{}]]);
