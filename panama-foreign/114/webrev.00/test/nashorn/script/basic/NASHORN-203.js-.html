24 /**
25  * NASHORN-203:  Date.prototype.toISOString should throw RangeError on invalid date.
26  *
27  * @test
28  * @run
29  */
30 
31 var date = new Date(-Infinity, Infinity);
32 
33 try {
34     date.toISOString();
35     fail("toISOString should have thrown RangeError");
36 } catch (ex) {
37     if (! (ex instanceof RangeError)) {
38         fail("expected RangeError but got " + ex);
39     }
40 }
