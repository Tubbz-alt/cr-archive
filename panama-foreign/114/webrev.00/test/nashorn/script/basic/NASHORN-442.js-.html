24 /**
25  * NASHORN-442 : for loop for object properties repeats property names.
26  *
27  * @test
28  * @run
29  */
30 
31 var proto = { foo : 33 };
32 var obj = Object.create(proto);
33 obj.foo = 'hello';
34 
35 var fooSeen = false;
36 var count = 0;
37 for (i in obj) {
38     count++;
39     if (i === "foo") {
40         if (fooSeen) {
41             fail("foo repeated in for loop");
42         }
43         fooSeen = true;
44     }
45 }
46 
47 if (count !== 1) {
48     fail("only one property expected in for loop");
49 }
