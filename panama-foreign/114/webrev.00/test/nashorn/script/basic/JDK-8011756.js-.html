32 // Invalid control letters should be escaped:
33 print(/\cı/.test("\x09"));
34 print(/\cı/.test("\\cı"));
35 
36 print(/\cſ/.test("\x13"));
37 print(/\cſ/.test("\\cſ"));
38 
39 print(/[\cſ]/.test("\x13"));
40 print(/[\cſ]/.test("\\"));
41 print(/[\cſ]/.test("c"));
42 print(/[\cſ]/.test("ſ"));
43 
44 print(/[\c#]/.test("\\"));
45 print(/[\c#]/.test("c"));
46 print(/[\c#]/.test("#"));
47

54 print(/[\c5]/.test("\x15"));
55 print(/[\c6]/.test("\x16"));
56 print(/[\c7]/.test("\x17"));
57 print(/[\c8]/.test("\x18"));
58 print(/[\c9]/.test("\x19"));
