24 /**
25  * JDK-8044750: megamorphic getter for scope objects does not call __noSuchProperty__ hook
26  *
27  * @test
28  * @option --unstable-relink-threshold=16
29  * @run
30  */
31 
32 __noSuchProperty__ = function(name) {
33     return 1;
34 }
35 
36 function func(obj) {
37     with(obj) {
38         // this "foo" getter site becomes megamorphic
39         // due to different 'with' scope objects.
40         foo;
41     }
42 }
43 
44 var LIMIT = 20; // should be more than megamorphic threshold set via @option
45 
46 for (var i = 0; i < LIMIT; i++) {
47     var obj = {};
48     obj.foo = i;
49     obj[i] = i;
50     func(obj);
51 }
52 
53 // pass a 'with' scope object that does not have 'foo'.
54 // callsite inside func should see __noSuchProperty__
55 // hook on global scope object.
56 func({});
57 
58 function checkFoo() {
59     with({}) {
60         try {
61             foo;
62             return true;
63         } catch (e) {
64             return false;
65         }
66     }
67 }
68 
69 var oldNoSuchProperty = this.__noSuchProperty__;
70 delete this.__noSuchProperty__;
71 
72 // keep deleting/restorting __noSuchProperty__ alternatively
73 // to make "foo" access in checkFoo function megamorphic!
74 
75 for (var i = 0; i < LIMIT; i++) {
76     // no __noSuchProperty__ and 'with' scope object has no 'foo'
77     delete __noSuchProperty__;
78     Assert.assertFalse(checkFoo(), "Expected false in iteration " + i);
79 
80     // __noSuchProperty__ is exists but 'with' scope object has no 'foo'
81     this.__noSuchProperty__ = oldNoSuchProperty;
82     Assert.assertTrue(checkFoo(), "Expected true in iteration " + i);
83 }
