24 /**
25  * JDK-8135178: importPackage not working even with load "Mozilla compatibility script"
26  *
27  * @test
28  * @run
29  */
30 
31 var ScriptContext = javax.script.ScriptContext;
32 var manager = new javax.script.ScriptEngineManager();
33 
34 var engine1 = manager.getEngineByName("nashorn");
35 engine1.eval("load('nashorn:mozilla_compat.js')");
36 manager.setBindings(engine1.getBindings(ScriptContext.ENGINE_SCOPE));
37 
38 var engine2 = manager.getEngineByName("nashorn");
39 engine2.eval("load('nashorn:mozilla_compat.js');");
40 engine2.eval("importPackage(java.util);");
41 
42 engine2.eval("var al = new ArrayList()");
43 Assert.assertTrue(engine2.eval("al instanceof java.util.ArrayList"));
44 Assert.assertTrue(engine2.get("al") instanceof java.util.ArrayList);
45 
46 engine2.eval("var hm = new HashMap()");
47 Assert.assertTrue(engine2.eval("hm instanceof java.util.HashMap"));
48 Assert.assertTrue(engine2.get("hm") instanceof java.util.HashMap);
