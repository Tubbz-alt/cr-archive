24 /**
25  * NASHORN-247 : Object gets stale value when inherited accessor property changes to a data property
26  *
27  * @test
28  * @run
29  */
30 
31 var data = "foo1";
32 var obj = {
33   get foo() { return data; },
34   set foo(value) { data = value; }
35 };
36 
37 var obj2 = Object.create(obj);
38 
39 function checkObjFoo(expected) {
40     if (obj2.foo !== expected) {
41         fail("obj2.foo has to be " + expected);
42     }
43 }
44 
45 checkObjFoo("foo1");
46 
47 obj2.foo = "foo2";
48 checkObjFoo("foo2");
49 
50 // change the property to data property
51 Object.defineProperty(obj, "foo", {
52    value: "foo3"
53 });
54 checkObjFoo("foo3");
