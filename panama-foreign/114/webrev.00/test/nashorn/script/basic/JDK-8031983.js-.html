24 /**
25  * JDK-8031983: Error objects should capture stack at the constructor
26  *
27  * @test
28  * @run
29  */
30 
31 var e = new Error();
32 print("hello");
33 
34 try {
35     throw e;
36 } catch (e) {
37     print(e.lineNumber);
38     print(e.stack.replace(/\\/g, '/'));
39 }
40 
41 Error.captureStackTrace(e);
42 try {
43     throw e;
44 } catch (e) {
45     print(e.lineNumber);
46     print(e.stack.replace(/\\/g, '/'));
47 }
48 
49 var obj = {};
50 Error.captureStackTrace(obj);
51 try {
52     throw obj;
53 } catch (e) {
54     print(e.stack.replace(/\\/g, '/'));
55 }
