24 /**
25  * NASHORN-490 : Inherited index properties is problematic for Array.prototype.pop function
26  *
27  * @test
28  * @run
29  */
30 
31 Array.prototype[1] = 1;
32 var x = [0];
33 x.length = 2;
34 
35 var obj = x.pop();
36 if (obj !== 1) {
37   fail('Array.prototype[1] = 1; x=[0]; x.length = 2; x.pop() == 1. Actual: ' + (obj));
38 }
39 
40 delete Array.prototype[1];
41 
42 Object.prototype[1] = 1;
43 var y = [0];
44 y.length = 2;
45 var obj = y.pop();
46 if (obj !== 1) {
47   fail('Object.prototype[1] = 1; y=[0]; y.length = 2; y.pop() == 1. Actual: ' + (obj));
48 }
