24 /**
25  * JDK-8023026: Array.prototype iterator functions like forEach, reduce should work for Java arrays, lists
26  *
27  * @test
28  * @run
29  */
30 
31 function checkIterations(obj) {
32     if (typeof obj.getClass == 'function') {
33         print("iterating on an object of " + obj.getClass());
34     } else {
35         print("iterating on " + String(obj));
36     }
37 
38     Array.prototype.forEach.call(obj,
39         function(x) { print("forEach " + x); });
40 
41     print("left sum " + Array.prototype.reduce.call(obj,
42         function(x, y) { print("reduce", x, y); return x + y; }));
43 
44     print("right sum " + Array.prototype.reduceRight.call(obj,
45         function(x, y) { print("reduceRight", x, y); return x + y; }));
46 
47     print("squared " + Array.prototype.map.call(obj,
48         function(x) x*x));
49 }
50 
51 var array = new (Java.type("int[]"))(4);
52 for (var i in array) {
53     array[i] = i;
54 }
55 
56 checkIterations(array);
57 
58 var list = new java.util.ArrayList();
59 list.add(1);
60 list.add(3);
61 list.add(5);
62 list.add(7);
63 
64 checkIterations(list);
