24 /**
25  * JDK-8024180: Incorrect handling of expression and parent scope in 'with' statements
26  *
27  * @test
28  * @run
29  */
30 
31 // global variable is shadowed by with 'expression' property
32 var  user = { name: 'foo' };
33 
34 function func(locals) {
35     with (locals) {
36         print(user.name);
37     }
38 }
39 
40 // global user.name 'foo' printed
41 func({});
42 
43 // local user.name 'toto' printed
44 func({ user: { name: 'toto' } });
