24 /**
25  * NASHORN-86 :  right hand side of instanceof operator should be objects that have [[HasInstance]]
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = {};
32 
33 try {
34     print(obj instanceof true);
35     fail("#1 TypeError expected");
36 } catch (e) {
37     if (! (e instanceof TypeError)) {
38         fail("#1 TypeError expected");
39     }
40 }
41 
42 try {
43     print(obj instanceof 3.14);
44     fail("#2 TypeError expected");
45 } catch (e) {
46     if (! (e instanceof TypeError)) {
47         fail("#2 TypeError expected");
48     }
49 }
50 
51 try {
52     print(obj instanceof "hello");
53     fail("#3 TypeError expected");
54 } catch (e) {
55     if (! (e instanceof TypeError)) {
56         fail("#3 TypeError expected");
57     }
58 }
59 try {
60     var foo = {};
61     print(obj instanceof foo);
62     fail("#4 TypeError expected");
63 } catch (e) {
64     if (! (e instanceof TypeError)) {
65         fail("#4 TypeError expected");
66     }
