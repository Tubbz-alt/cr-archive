 24 /**
 25  * Test to check JavaAdapter constructor.
 26  *
 27  * @test
 28  * @run
 29  */
 30 
 31 try {
 32     load("nashorn:mozilla_compat.js");
 33 } catch (e) {
 34 }
 35 
 36 // try various JavaAdapter cases
 37 
 38 // Single interface
 39 var runReached = false;
 40 var r = new JavaAdapter(java.lang.Runnable) {
 41     run: function() {
 42         runReached = true;
 43     }
 44 };
 45 
 46 r.run();
 47 if (! runReached) {
 48     fail("run was not called");
 49 }
 50 
 51 if (! (r instanceof java.lang.Runnable)) {
 52     fail("r is not a Runnable");
 53 }
 54 
 55 // Multiple intefaces
 56 var runReached = false;
 57 var actionPerformedReached = false;
 58 
 59 var obj = new JavaAdapter(java.awt.event.ActionListener, java.lang.Runnable) {
 60     actionPerformed : function(e) {
 61         actionPerformedReached = true;
 62     },
 63 
 64     run: function() {
 65         runReached = true;
 66     }
 67 };
 68 
 69 obj.actionPerformed(null);
 70 if (! actionPerformedReached) {
 71     fail("actionPerformed was not called");
 72 }
 73 
 74 obj.run();
 75 if (! runReached) {
 76     fail("run was not called");
 77 }
 78 
 79 if (! (obj instanceof java.lang.Runnable)) {
 80     fail("obj is not a Runnable");
 81 }
 82 
 83 if (! (obj instanceof java.awt.event.ActionListener)) {
 84     fail("obj is not an ActionListener");
 85 }
 86 
 87 // Single class
 88 var obj = new JavaAdapter(java.lang.Object) {
 89     toString: function() { return "I am an Object"; }
 90 };
 91 
 92 if (! (obj instanceof java.lang.Object)) {
 93     fail("obj is not an instance of java.lang.Object");
 94 }
 95 
 96 if (obj.toString() != "I am an Object") {
 97     fail("Object.toString did not get called");
 98 }
 99 
100 // Single class and single interface
101 var runReached = false;
102 var obj = new JavaAdapter(java.lang.Object, java.lang.Runnable) {
103     run: function() {
104         runReached = true;
105     },
106 
107     hashCode: function() {
108         return 12;
109     }
110 };
111 
112 obj.run();
113 if (! runReached) {
114     fail("run was not called");
115 }
116 
117 if (obj.hashCode() != 12) {
118     fail("hashCode does not return 12");
119 }
