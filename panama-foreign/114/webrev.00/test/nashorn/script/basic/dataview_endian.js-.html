24 /**
25  * JDK-8015958: DataView constructor is not defined
26  *
27  * @test
28  * @run
29  */
30 
31 // set/get endianess checks
32 
33 var buffer = new ArrayBuffer(4);
34 var dv = new DataView(buffer);
35 
36 // write (default) big endian, read big/little endian
37 dv.setUint16(0, 0xABCD);
38 Assert.assertEquals(dv.getUint16(0), 0xABCD);
39 Assert.assertEquals(dv.getUint16(0, false), 0xABCD);
40 Assert.assertEquals(dv.getUint16(0, true), 0xCDAB);
41 
42 // write little endian, read big/little endian
43 dv.setUint16(0, 0xABCD, true);
44 Assert.assertEquals(dv.getUint16(0), 0xCDAB);
45 Assert.assertEquals(dv.getUint16(0, false), 0xCDAB);
46 Assert.assertEquals(dv.getUint16(0, true), 0xABCD);
47 
48 // write explicit big endian, read big/little endian
49 dv.setUint16(0, 0xABCD, false);
50 Assert.assertEquals(dv.getUint16(0), 0xABCD);
51 Assert.assertEquals(dv.getUint16(0, false), 0xABCD);
52 Assert.assertEquals(dv.getUint16(0, true), 0xCDAB);
53 
54 // write (default) big endian, read big/little endian
55 dv.setUint32(0, 0xABCDEF89);
56 Assert.assertEquals(dv.getUint32(0), 0xABCDEF89);
57 Assert.assertEquals(dv.getUint32(0, false), 0xABCDEF89);
58 Assert.assertEquals(dv.getUint32(0, true), 0x89EFCDAB);
59 
60 // write little endian, read big/little endian
61 dv.setUint32(0, 0xABCDEF89, true);
62 Assert.assertEquals(dv.getUint32(0), 0x89EFCDAB);
63 Assert.assertEquals(dv.getUint32(0, false), 0x89EFCDAB);
64 Assert.assertEquals(dv.getUint32(0, true), 0xABCDEF89);
65 
66 // write explicit big endian, read big/little endian
67 dv.setUint32(0, 0xABCDEF89, false);
68 Assert.assertEquals(dv.getUint32(0), 0xABCDEF89);
69 Assert.assertEquals(dv.getUint32(0, false), 0xABCDEF89);
70 Assert.assertEquals(dv.getUint32(0, true), 0x89EFCDAB);
