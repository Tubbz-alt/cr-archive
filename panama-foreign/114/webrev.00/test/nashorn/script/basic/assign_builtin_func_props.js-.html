25 /**
26  * Check that we can assign to all properties from builtin functions and their
27  * prototypes. This is to make sure all such properties are writable.
28  *
29  * @test
30  * @run
31  */
32 
33 (function() {
34     var PropNamesGetter = Object.getOwnPropertyNames;
35     var ObjectType = Object;
36 
37     function assignAll(obj) {
38         if (! (obj instanceof ObjectType)) {
39             return;
40         }
41         var props = PropNamesGetter(obj);
42         for (var p in props) {
43             var value = obj[props[p]];
44             obj[props[p]] = value;
45         }
46     }
47 
48     var globalProps = PropNamesGetter(this);
49     for (var i in globalProps) {
50         var prop = globalProps[i];
51         if (typeof this[prop] == 'function') {
52             assignAll(this[prop].prototype);
53             assignAll(this[prop]);
54         }
55     }
56 })();
