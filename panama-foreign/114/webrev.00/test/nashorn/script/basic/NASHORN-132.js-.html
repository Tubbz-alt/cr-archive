24 /**
25  * NASHORN-132 : Implementation of __FILE__ and __LINE__ in lower is incorrect.
26  *
27  * @test
28  * @run
29  */
30 
31 function printFile() {
32     var idx = __FILE__.lastIndexOf(java.io.File.separator);
33     if (idx < 0) {
34         // separator is "/" when running under ant on windows
35         idx = __FILE__.lastIndexOf("/");
36     }
37     var file = (idx != -1)? __FILE__.substring(idx + 1) : __FILE__;
38     print("file: " + file);
39 }
40 
41 // assignment to global read-only __FILE__ ignored
42 __FILE__ = 454;
43 printFile();
44 
45 // assignment to global read-only __LINE__ ignored
46 __LINE__ = "hello";
47 print("line: " + __LINE__);
48 
49 var obj = { __FILE__: "obj.__FILE__", __LINE__: "obj.__LINE__" };
50 
51 // obj.__FILE__ is different from global __FILE__
52 print(obj.__FILE__);
53 printFile();
54 
55 // obj.__LINE__ is different from global __LINE__
56 print(obj.__LINE__);
57 print("line: " + __LINE__);
58 
59 function func() {
60    // can have local variable of that name
61    var __FILE__ = "local __FILE__ value";
62    print(__FILE__);
63 
64    var __LINE__ = "local __LINE__ value";
65    print(__LINE__);
66 }
67 
68 func();
69 printFile();
70 print("line: " + __LINE__);
