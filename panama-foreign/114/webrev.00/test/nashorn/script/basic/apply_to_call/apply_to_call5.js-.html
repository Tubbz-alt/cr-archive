 24 /**
 25  * apply_to_call5.js - do one apply to call specialization, then override, apply and make sure it reverts (i.e. stops
 26  * calling call)
 27  *
 28  * @test
 29  * @run
 30  */
 31 
 32 print("start");
 33 
 34 var x = {
 35     a : 0,
 36     b : 0,
 37     c : 0,
 38     initialize : function(x,y,z) {
 39 	this.a = x;
 40 	this.b = y;
 41 	this.c = z;
 42     }
 43 };
 44 
 45 function test() {
 46     x.initialize.apply(x, arguments);
 47 }
 48 
 49 test(4711,23,17);
 50 print(x.a);
 51 print(x.b);
 52 print(x.c);
 53 
 54 print("Overwriting apply now");
 55 x.initialize.apply = function() { print("New function for apply - not a property"); }
 56   
 57 test(4712);
 58 print(x.a);
