26  *
27  * @test

30 function loop1() {
31     print("loop1");
32     for (var i = 0; i < 5; i++) {
33     print(i);
34     continue;
35     }
36     print(i);
37     print("done1");
38 }
39 
40 function loop2() {
41     print("loop2");
42     for (var i = 0; i < 5; i++) {
43     print(i);
44     break;
45     }
46     print(i);
47     print("done2");
48 }
49
