24 /**
25  * NASHORN-190 :  Object property keys should be Strings
26  *
27  * @test
28  * @run
29  */
30 
31 var obj = {
32     20: 'world', "2.3": 'hello'
33 };
34 
35 obj[true] = 30;
36 obj[null] = 'java';
37 obj[undefined] = 'js';
38 
39 if (obj[20] !== 'world') {
40     fail("obj[20] !== 'world'");
41 }
42 
43 if (obj['20'] !== 'world') {
44     fail("obj['20'] !== 'world'");
45 }
46 
47 if (obj[2.3] !== 'hello') {
48     fail("obj[2.3] !== 'hello'");
49 }
50 
51 if (obj['2.3'] !== 'hello') {
52     fail("obj['2.3'] !== 'hello'");
53 }
54 
55 if (obj[true] !== 30) {
56     fail("obj[true] !== 30");
57 }
58 
59 if (obj['true'] !== 30) {
60     fail("obj['true'] !== 30");
61 }
62 
63 if (obj[null] !== 'java') {
64     fail("obj[null] !== 'java'");
65 }
66 
67 if (obj['null'] !== 'java') {
68     fail("obj['null'] !== 'java'");
69 }
70 
71 if (obj[undefined] !== 'js') {
72     fail("obj[undefined] !== 'js'");
73 }
74 
75 if (obj['undefined'] !== 'js') {
76     fail("obj['undefined'] !== 'js'");
77 }
