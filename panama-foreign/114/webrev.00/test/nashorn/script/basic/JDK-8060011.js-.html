24 /**
25  * JDK-8060011: Concatenating an array and converting it to Java gives wrong result
26  *
27  * @test
28  * @run
29  */
30 
31 
32 function compareAsJavaArrays(a1, a2) {
33     var ja1 = Java.to(a1);
34     var ja2 = Java.to(a2);
35     if (ja1.length !== ja2.length) {
36         throw "different length";
37     }
38     for (var i = 0; i < ja1.length; i++) {
39         if (ja1[i] !== ja2[i]) {
40             throw "different element at " + i;
41         }
42     }
43     if (java.util.Arrays.toString(ja1) !== java.util.Arrays.toString(ja2)) {
44         throw "different string representation";
45     }
46 }
47 
48 compareAsJavaArrays([0, 1, 2, 3],
49                     [0].concat([1, 2, 3]));
50 compareAsJavaArrays([1000000000, 2000000000, 3000000000, 4000000000],
51                     [1000000000].concat([2000000000, 3000000000, 4000000000]));
52 compareAsJavaArrays([0.5, 1.5, 2.5, 3.5],
53                     [0.5].concat([1.5, 2.5, 3.5]));
54 compareAsJavaArrays(["0", "1", "2", "3"],
55                     ["0"].concat(["1", "2", "3"]));
