24 /**
25  * Inherited non-writable properties can not be overridden.
26  *
27  * @test
28  * @run
29  */
30 
31 var p = {};
32 Object.defineProperty(p, "foo", { writable: false, value: 12 });
33 
34 var o = Object.create(p);
35 o.foo = 44;
36 if (o.foo !== 12) {
37     fail("o.foo can be overridden");
38 }
