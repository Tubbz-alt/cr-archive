31 var locale = java.util.Locale.ENGLISH;
32 var prop = 'ISO3Language';
33 var prop1 = 'ISO3';
34 var prop2 = prop1 + 'Language';
35 var prop3 = String(prop2);
36 
37 function checkLang(obj) {
38     if (obj != "eng") {
39         throw new Error("FAILED: expected 'eng', got " + obj);
40     }
41 }
42 
43 checkLang(locale.ISO3Language);
44 checkLang(locale['ISO3Language']);
45 checkLang(locale[prop]);
46 checkLang(locale[prop1 + 'Language']);
47 checkLang(locale[prop2]);
48 checkLang(locale[prop3]);
49 checkLang(locale[String(prop1 + 'Language')]);
