24 /**
25  * NASHORN-269 : Global constructor inlining breaks when a builtin constructor is deleted
26  *
27  * @test
28  * @run
29  */
30 
31 delete Array;
32 
33 try {
34     print(new Array(3));
35     fail("#1 should have thrown ReferenceError");
36 } catch (e) {
37     if (! (e instanceof ReferenceError)) {
38         fail("#2 expected ReferenceError, got " + e);
39     }
40 }
41 
42 Object.defineProperty(this, "Array", {
43     get: function() { return 63; }
44 });
45 
46 try {
47     print(new Array(2));
48     fail("#3 should have thrown TypeError");
49 } catch (e) {
50     if (! (e instanceof TypeError)) {
51         fail("#4 expected TypeError, got " + e);
52     }
53 }
54 
55 Object.defineProperty(this, "RegExp", {
56     get: function() { return 23; }
57 });
58 
59 
60 try {
61     print(new RegExp("abc"));
62     fail("#5 should have thrown TypeError");
63 } catch (e) {
64     if (! (e instanceof TypeError)) {
65         fail("#6 expected TypeError, got " + e);
66     }
67 }
