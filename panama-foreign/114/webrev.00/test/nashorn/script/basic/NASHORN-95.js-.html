24 /**
25  * NASHORN-95 :  function length is always -1 when it uses "arguments" regardless of actual formals specified.
26  *
27  *
28  * @test
29  * @run
30  */
31 
32 function func1() {
33     print(arguments);
34 }
35 
36 function func2(x, y) {
37     print(arguments);
38 }
39 
40 function func3(x, y, z) {
41     print(arguments);
42 }
43 
44 // more than arglimit number of arguments - so uses varargs in impl..
45 function func4(a, b, c, d, e, f, g, h, i, j) {}
46 
47 print("func1.length = " + func1.length);
48 print("func2.length = " + func2.length);
49 print("func3.length = " + func3.length);
50 print("func4.length = " + func4.length);
