24 /**
25  * Example of an add function that gets specialized to doubles
26  * if run with --optimize flag set
27  */
28 
29 function add(a,b) {
30     return a + b;
31 }
32 
33 
34 function bench() {
35     var sum = 1;
36     for (var x = 0 ; x < 5e7 ; x++) {
37     sum *= add(x, 17);
38     }
39     return sum;
40 }
41 
42 var d = new Date;
43 print(bench());
44 d = new Date - d;
45 print("took " + d + " ms");
