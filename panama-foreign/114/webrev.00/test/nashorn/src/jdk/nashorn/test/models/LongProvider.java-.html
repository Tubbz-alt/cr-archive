26 package jdk.nashorn.test.models;
27 
28 import java.util.Arrays;
29 
30 /**
31  * A class that returns and receives longs.
32  */
33 @SuppressWarnings("javadoc")
34 public class LongProvider {
35 
36     final static long[][] arrays = {
37             {1L, 2L, 3L},
38             {1L, 1L << 30, 1L << 50, 4L},
39             {1L, Long.MAX_VALUE, Long.MIN_VALUE, 4L}
40     };
41 
42     public static long getLong(final String str) {
43         final long l = Long.parseLong(str);
44         checkLong(l, str);
45         return l;
46     }
47 
48     public static long[] getLongArray(final int n) {
49         return arrays[n];
50     }
51 
52     public static void checkLong(final long value, final String str) {
53         if (!Long.toString(value).equals(str)) {
54             throw new RuntimeException("Wrong value. Expected " + str + ", got " + value);
55         }
56     }
57 
58     public static void checkLongArray(final long[] array, final int n) {
59         if (!Arrays.equals(array, arrays[n])) {
60             throw new RuntimeException("Arrays don't match: " + Arrays.toString(array) + ", " + Arrays.toString(arrays[n]));
61         }
62     }
63 }
