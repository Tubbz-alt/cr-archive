25 package jdk.nashorn.api.scripting.test;
26 
27 import java.util.Arrays;
28 import javax.script.ScriptEngine;
29 import javax.script.ScriptEngineManager;
30 import javax.script.ScriptException;
31 import jdk.nashorn.api.scripting.AbstractJSObject;
32 
33 import org.testng.annotations.BeforeClass;
34 import org.testng.annotations.Test;
35 
36 import static org.testng.Assert.*;
37 
38 /**
39  * @bug 8148140
40  * @summary arguments are handled differently in apply for JS functions and AbstractJSObjects
41  */
42 public class JDK_8148140_Test {
43 
44     ScriptEngine engine;
45 
46     static final String RESULT = "[1, 2, 3]";
47 
48     @BeforeClass
49     public void setupTest() {
50         engine = new ScriptEngineManager().getEngineByName("js");
51         engine.put("f", new AbstractJSObject() {
52             @Override
53             public boolean isFunction() {
54                 return true;
55             }
56             @Override
57             public Object call(final Object thiz, final Object... args) {
58                 return Arrays.deepToString(args);
59             }
60         });
61     }
62 
63     @Test
64     public void testCallF() throws ScriptException {
65         assertEquals(RESULT, engine.eval("f(1,2,3)"));
66     }
67 
68     @Test
69     public void testApplyF() throws ScriptException {
70         assertEquals(RESULT, engine.eval("Function.prototype.apply.call(f, null, [1,2,3])"));
71     }
72 
73 }
