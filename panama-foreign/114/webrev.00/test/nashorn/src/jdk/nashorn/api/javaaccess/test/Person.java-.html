26 package jdk.nashorn.api.javaaccess.test;
27 
28 @SuppressWarnings("javadoc")
29 public class Person {
30 
31     public int id = 0;
32 
33     public Person() {
34     }
35 
36     public Person(final int code) {
37         this.id = code;
38     }
39 
40     @Override
41     public boolean equals(final Object obj) {
42         if (obj != null && obj instanceof Person) {
43             final Person o = (Person)obj;
44             return this.id == o.id;
45         }
46         return false;
47     }
48 
49     @Override
50     public int hashCode() {
51         return id;
52     }
53 
54     @Override
55     public String toString() {
56         return "Person(" + id + ")";
57     }
58 
59 }
