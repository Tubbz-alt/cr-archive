28 import static org.testng.Assert.assertEquals;
29 import static org.testng.Assert.assertNull;
30 
31 import java.util.Collection;
32 import java.util.Queue;
33 import javax.script.ScriptEngine;
34 import javax.script.ScriptException;
35 import jdk.nashorn.api.scripting.NashornScriptEngineFactory;
36 import jdk.nashorn.test.models.JDK_8081015_TestModel;
37 import org.testng.annotations.Test;
38 
39 /**
40  * @bug 8081015
41  * @summary Test that native arrays get converted to {@link Queue} and {@link Collection}.
42  */
43 @SuppressWarnings("javadoc")
44 public class JDK_8081015_Test {
45     @Test
46     public void testConvertToCollection() throws ScriptException {
47         test("receiveCollection");
48     }
49 
50     @Test
51     public void testConvertToDeque() throws ScriptException {
52         test("receiveDeque");
53     }
54 
55     @Test
56     public void testConvertToList() throws ScriptException {
57         test("receiveList");
58     }
59 
60     @Test
61     public void testConvertToQueue() throws ScriptException {
62         test("receiveQueue");
63     }
64 
65     private static void test(final String methodName) throws ScriptException {
66         final ScriptEngine engine = new NashornScriptEngineFactory().getScriptEngine();
67         final JDK_8081015_TestModel model = new JDK_8081015_TestModel();
68         engine.put("test", model);
69 
70         assertNull(model.getLastInvoked());
71         engine.eval("test." + methodName + "([1, 2, 3.3, 'foo'])");
72         assertEquals(model.getLastInvoked(), methodName );
73     }
74 }
