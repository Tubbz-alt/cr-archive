26 package jdk.nashorn.api.scripting.test;
27 
28 import org.testng.annotations.BeforeClass;
29 import org.testng.annotations.Test;
30 
31 import javax.script.*;
32 
33 import static org.testng.Assert.assertEquals;
34 
35 /**
36  * @bug 8182996
37  * @summary Incorrect mapping Long type to JavaScript equivalent
38  */
39 @SuppressWarnings("javadoc")
40 public class JDK_8182996_Test {
41 
42     private ScriptEngine engine;
43     Bindings bindings;
44 
45 
46     @BeforeClass
47     public void setupTest() {
48         engine = new ScriptEngineManager().getEngineByName("js");
49         bindings = engine.getBindings(ScriptContext.ENGINE_SCOPE);
50 
51         bindings.put("long1", Long.valueOf(1L));
52         bindings.put("long2", Long.valueOf(2L));
53         bindings.put("long2", Long.valueOf(3L));
54     }
55 
56     @Test
57     public void testType() throws ScriptException {
58         assertEquals(engine.eval("typeof long1"), "object");
59         assertEquals(engine.eval("typeof long2"), "object");
60     }
61 
62     @Test
63     public void testValue() throws ScriptException {
64         assertEquals(engine.eval("long1"), Long.valueOf(1));
65         assertEquals(engine.eval("long2"), Long.valueOf(3));
66         assertEquals(bindings.get("long1"), Long.valueOf(1));
67         assertEquals(bindings.get("long2"), Long.valueOf(3));
68     }
69 
70 }
