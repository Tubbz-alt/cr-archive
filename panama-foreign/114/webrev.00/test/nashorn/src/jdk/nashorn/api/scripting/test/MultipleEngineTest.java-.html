26 package jdk.nashorn.api.scripting.test;
27 
28 import javax.script.ScriptEngine;
29 import javax.script.ScriptEngineManager;
30 import javax.script.ScriptException;
31 import org.testng.annotations.Test;
32 
33 /**
34  * Test that we can create multiple, independent script engines and use those
35  * independently.
36  *
37  * @test
38  * @run testng jdk.nashorn.api.scripting.test.MultipleEngineTest
39  */
40 @SuppressWarnings("javadoc")
41 public class MultipleEngineTest {
42     @Test
43     public void createAndUseManyEngine() throws ScriptException {
44         final ScriptEngineManager m = new ScriptEngineManager();
45 
46         final ScriptEngine e1 = m.getEngineByName("nashorn");
47         e1.eval("var  x = 33; print(x);");
48 
49         final ScriptEngine e2 = m.getEngineByName("nashorn");
50         e2.eval("try { print(x) } catch(e) { print(e); }");
51     }
52 }
