26 package jdk.nashorn.internal.test.framework;
27 
28 import java.util.Set;
29 import org.testng.Assert;
30 import org.testng.ITest;
31 import org.testng.annotations.Test;
32 
33 /**
34  * Test case used by JSCompilerTest to complain if test files are marked as
35  * neither test nor subtest.
36  */
37 @SuppressWarnings("javadoc")
38 public final class OrphanTestFinder implements ITest {
39     private final Set<String> orphanFiles;
40 
41     public OrphanTestFinder(final Set<String> orphanFiles) {
42         this.orphanFiles = orphanFiles;
43     }
44 
45     @Override
46     public String getTestName() {
47         return getClass().getName();
48     }
49 
50     @Test
51     public void test() {
52         if (orphanFiles == null || orphanFiles.isEmpty()) {
53             return;
54         }
55 
56         final StringBuilder sb = new StringBuilder();
57         final String NL = System.getProperty("line.separator");
58         sb.append(orphanFiles.size());
59         sb.append(" files found with neither @test nor @subtest: ");
60         sb.append(NL);
61         for (final String s : orphanFiles) {
62             sb.append("  ");
63             sb.append(s);
64             sb.append(NL);
65         }
66         Assert.fail(sb.toString());
67     }
68 }
