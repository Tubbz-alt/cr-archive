26 package jdk.nashorn.internal.test.framework;
27 
28 import java.util.Collections;
29 import java.util.Comparator;
30 import java.util.List;
31 import org.testng.IMethodInstance;
32 import org.testng.IMethodInterceptor;
33 import org.testng.ITest;
34 import org.testng.ITestContext;
35 
36 /**
37  * TestNG method interceptor to make sure that the tests are run in the test name order.
38  */
39 public final class TestReorderInterceptor implements IMethodInterceptor {
40     @Override
41     public List<IMethodInstance> intercept(final List<IMethodInstance> methods, final ITestContext context) {
42         Collections.sort(methods, new Comparator<IMethodInstance>() {
43             @Override
44             public int compare(final IMethodInstance mi1, final IMethodInstance mi2) {
45                 // get test instances to order the tests.
46                 final Object o1 = mi1.getInstance();
47                 final Object o2 = mi2.getInstance();
48                 if (o1 instanceof ITest && o2 instanceof ITest) {
49                     return ((ITest)o1).getTestName().compareTo(((ITest)o2).getTestName());
50                 } else if (o1 instanceof ITest) {
51                     return 1;
52                 } else if (o2 instanceof ITest) {
53                     return -1;
54                 }
55                 // something else, don't care about the order
56                 return 0;
57             }
58         });
59 
60         return methods;
61     }
62 }
