26 package jdk.nashorn.internal.runtime.test;
27 
28 import java.util.Iterator;
29 import jdk.nashorn.internal.runtime.PropertyMap;
30 import jdk.nashorn.internal.runtime.ScriptObject;
31 import org.testng.Assert;
32 import org.testng.annotations.Test;
33 
34 /**
35  * Tests for PropertyMap functionality
36  *
37  * @test
38  * @modules jdk.scripting.nashorn/jdk.nashorn.internal.runtime
39  * @run testng jdk.nashorn.internal.runtime.test.PropertyMapTest
40  */
41 @SuppressWarnings("javadoc")
42 public class PropertyMapTest {
43 
44     @Test
45     public void propertyMapIteratorTest() {
46         final ScriptObject scriptObject = new ScriptObject(PropertyMap.newMap()) {};
47         Assert.assertFalse(scriptObject.getMap().iterator().hasNext());
48 
49         scriptObject.set("a", "a", 0);
50         scriptObject.set("b", 3, 0);
51         // 3 is a valid array key not stored in property map
52         scriptObject.set(3, 1, 0);
53         scriptObject.set(6.5, 1.3, 0);
54         final Iterator<Object> iterator = scriptObject.getMap().iterator();
55 
56         Assert.assertTrue(iterator.hasNext());
57         Assert.assertEquals(iterator.next(), "a");
58         Assert.assertTrue(iterator.hasNext());
59         Assert.assertEquals(iterator.next(), "b");
60         Assert.assertTrue(iterator.hasNext());
61         Assert.assertEquals(iterator.next(), "6.5");
62         Assert.assertFalse(iterator.hasNext());
63     }
64 
65 }
