26 package jdk.nashorn.internal.runtime.test;
27 
28 import jdk.nashorn.api.scripting.AbstractJSObject;
29 import jdk.nashorn.api.scripting.NashornScriptEngine;
30 import jdk.nashorn.api.scripting.NashornScriptEngineFactory;
31 import jdk.nashorn.api.scripting.ScriptObjectMirror;
32 import static org.testng.Assert.assertEquals;
33 import org.testng.annotations.Test;
34 import javax.script.ScriptException;
35 
36 
37 /**
38  * @test
39  * @bug 8186807
40  * @summary JSObject gets ScriptFunction when ScriptObjectMirror is expected
41  * @modules jdk.scripting.nashorn/jdk.nashorn.internal.runtime
42  * @run testng/othervm -Dnashorn.unstable.relink.threshold=1 jdk.nashorn.internal.runtime.test.JDK_8186807_Test
43  */
44 
45 public class JDK_8186807_Test {
46 
47     @Test
48     public void testScript() throws ScriptException {
49         NashornScriptEngine engine = (NashornScriptEngine) new NashornScriptEngineFactory().getScriptEngine();
50         engine.put("a", new Func());
51         engine.eval("var Assert = Java.type('org.testng.Assert')\n" +
52                 "var fn=function() {return 2;}\n" +
53                 "var arr = [ a, fn, a ];\n" +
54                 "var result;\n" +
55                 "for (i in arr) {\n" +
56                 "    result = Function.prototype.apply.apply(arr[i],[null,[fn]]);\n" +
57                 "    if(i==0 || i==2)\n" +
58                 "        Assert.assertTrue(result==1);\n" +
59                 "    if(i==1)\n" +
60                 "        Assert.assertTrue(result==2);\n" +
61                 "}");
62     }
63 
64     static class Func extends AbstractJSObject {
65         @Override
66         public boolean isFunction() {
67             return true;
68         }
69 
70         @Override
71         public Object call(Object thiz, Object... args) {
72             assertEquals(args[0].getClass(), ScriptObjectMirror.class);
73             return 1;
74         }
75     }
76 }
