26 package jdk.nashorn.api.scripting.test;
27 
28 import javax.script.ScriptEngineFactory;
29 import jdk.nashorn.api.scripting.NashornScriptEngineFactory;
30 import org.testng.Assert;
31 import org.testng.annotations.Test;
32 
33 /**
34  * @test
35  * @run testng jdk.nashorn.api.scripting.test.JDK_8068741_Test
36  * @bug 8068741
37  * @summary javax.script.ScriptEngineFactory.getMethodCallSyntax() spec allows null passed as an object
38  */
39 public class JDK_8068741_Test {
40     @Test
41     public void testGetMethodCallSyntax() {
42         ScriptEngineFactory fac = new NashornScriptEngineFactory();
43         checkThrowsNPE(() -> fac.getMethodCallSyntax(null, "foo"));
44         checkThrowsNPE(() -> fac.getMethodCallSyntax("obj", null));
45         checkThrowsNPE(() -> fac.getMethodCallSyntax("obj", "foo", (String[])null));
46         checkThrowsNPE(() -> fac.getMethodCallSyntax("obj", "foo", null, "xyz"));
47         checkThrowsNPE(() -> fac.getMethodCallSyntax("obj", "foo", "xyz", null));
48     }
49 
50     private void checkThrowsNPE(Runnable r) {
51         boolean gotNPE = false;
52         try {
53             r.run();
54         } catch (NullPointerException npe) {
55             gotNPE = true;
56             System.err.println("Got NPE as expected: " + npe);
57         }
58         Assert.assertTrue(gotNPE);
59     }
60 }
