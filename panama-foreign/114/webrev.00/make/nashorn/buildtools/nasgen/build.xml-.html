24 <project name="nasgen" default="all" basedir=".">
25   <target name="init">
26     <loadproperties srcFile="project.properties"/>
27   </target>
28 
29   <target name="prepare" depends="init">
30     <mkdir dir="${nasgen.build.classes.dir}"/>
31     <mkdir dir="${nasgen.dist.dir}"/>
32     <mkdir dir="${nasgen.dist.dir}/lib"/>
33   </target>
34 
35   <target name="clean" depends="init">
36     <delete dir="${nasgen.build.dir}"/>
37     <delete dir="${nasgen.dist.dir}"/>
38   </target>
39 
40   <target name="compile" depends="prepare" description="Compiles the nasgen sources">
41     <javac srcdir="${src.dir}"
42            destdir="${nasgen.build.classes.dir}"
43            debug="${javac.debug}"
44            includeantruntime="false" fork="true">
45       <compilerarg line="${nasgen.module.imports}"/>
46       <compilerarg value="-Xlint:unchecked"/>
47       <compilerarg value="-Xlint:deprecation"/>
48       <compilerarg value="-XDignore.symbol.file"/>
49     </javac>
50   </target>
51 
52   <target name="jar" depends="compile" description="Creates nasgen.jar">
53     <jar jarfile="${nasgen.dist.jar}" basedir="${nasgen.build.classes.dir}" manifest="${meta.inf.dir}/MANIFEST.MF"/>
54   </target>
55 
56   <target name="dist" depends="jar"/>
57 
58   <target name="all" depends="dist" 
59     description="Builds sources and generates nasgen.jar"/>
60 </project>
