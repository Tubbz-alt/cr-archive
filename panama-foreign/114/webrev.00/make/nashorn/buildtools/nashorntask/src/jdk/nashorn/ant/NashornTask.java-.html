26 package jdk.nashorn.ant;
27 
28 import javax.script.ScriptEngine;
29 import javax.script.ScriptEngineManager;
30 import javax.script.ScriptException;
31 import org.apache.tools.ant.Project;
32 import org.apache.tools.ant.Task;
33 import org.apache.tools.ant.BuildException;
34 
35 /**
36  * This class implements an ant task to evaluate nashorn scripts
37  * from ant projects.
38  */
39 public final class NashornTask extends Task {
40     // Underlying nashorn script engine
41     private final ScriptEngine engine;
42     // the current ant project
43     private Project project;
44     // the script evaluated by this task
45     private String script;
46 
47     public NashornTask() {
48         final ScriptEngineManager m = new ScriptEngineManager();
49         this.engine = m.getEngineByName("nashorn");
50     }
51 
52     @Override
53     public void setProject(Project proj) {
54         this.project = proj;
55     }
56 
57     // set the script to be evaluated
58     public void addText(String text) {
59         this.script = text;
60     }
61 
62     @Override
63     public void execute() {
64         // expose project as "project" variable
65         engine.put("project", project);
66         // expose this task as "self" variable
67         engine.put("self", this);
68 
69         // evaluate specified script
70         try {
71             engine.eval(script);
72         } catch (final ScriptException se) {
73             throw new BuildException(se);
74         }
75     }
76 }
