32 // Simple example demonstrating how to expose 'function's
33 // from embedding code. Any lambda object exposed to engine
34 // can be called as 'function' in script.
35 
36 var ScriptEngineManager = Java.type("javax.script.ScriptEngineManager");
37 // create manager
38 var manager = new ScriptEngineManager();
39 // create engine
40 var engine = manager.getEngineByName("js");
41 
42 // Any lambda (@FunctionalInterface annotated type) object can be
43 // be exposed from script embedding code. Script can call
44 // it as a function
45 engine.put("upper", new java.util.function.Function() {
46      apply: function(x) x.toUpperCase()
47 });
48 
49 print(engine.eval("upper('hello')"));
