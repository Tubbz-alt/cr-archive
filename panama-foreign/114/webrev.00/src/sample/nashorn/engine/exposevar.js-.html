32 // Example showing how to expose a script global var from caller
33 
34 var ScriptEngineManager = Java.type("javax.script.ScriptEngineManager");
35 // create manager
36 var manager = new ScriptEngineManager();
37 // create engine
38 var engine = manager.getEngineByName("js");
39 
40 // expose variable to engine
41 engine.put("name", "Nashorn");
42 
43 // access it from script
44 engine.eval("print('Hello, ' + name)");
