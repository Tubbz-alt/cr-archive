33 // MissingMethodExample class extends ArrayList and
34 // implements MissingMethodHandler interface
35 var MME = Java.type("MissingMethodExample")
36 
37 var obj = new MME()
38 
39 // can call any ArrayList method
40 obj.add("hello")
41 obj.add("world")
42 print(obj.size())
43 print(obj)
44 
45 // "foo", "bar" are 'missing' methods
46 // these will result in doesNotUnderstand calls
47 obj.foo(1, "www", true)
48 obj.bar("http", "dynalink")
49 
50 // another valid ArrayList method call
51 obj.forEach(print)
