32 /*
33  * This sample demonstrates the use of es6 tagged template literal to
34  * create a java object. A XML DOM Document object is created from
35  * String using es6 tagged template literal.
36  *
37  * Usage:
38  *    jjs --language=es6 dom_tagged_literal.js
39  */
40 
41 // Java types used
42 const BAIS = Java.type("java.io.ByteArrayInputStream")
43 const DocBuilderFac = Java.type("javax.xml.parsers.DocumentBuilderFactory")
44 const DOMSource = Java.type("javax.xml.transform.dom.DOMSource")
45 const StreamResult = Java.type("javax.xml.transform.stream.StreamResult")
46 const StringWriter = Java.type("java.io.StringWriter")
47 const TransformerFactory = Java.type("javax.xml.transform.TransformerFactory")
48 
49 function DOM(str) {
50     var docBuilder = DocBuilderFac.newInstance().newDocumentBuilder()
51     docBuilder.validating = false
52     return docBuilder["parse(java.io.InputStream)"](new BAIS(String(str).bytes))
53 }
54 
55 // es6 tagged template literal to create DOM from
56 // multi-line XML string
57 
58 const dom = DOM`
59 <foo>
60   <bar title="hello">world</bar>
61 </foo>`
62 
63 // access DOM elements
64 const foo = dom.documentElement
65 print(foo.tagName)
66 const bar = foo.getElementsByTagName("bar").item(0)
67 print(bar.tagName)
68 print(bar.getAttribute("title"))
69 
70 // modify DOM
71 foo.setAttribute("name", "nashorn")
72 foo.appendChild(dom.createElement("test"))
73 
74 // serialize DOM to XML string
75 function domToXML(d) {
76     const transformer = TransformerFactory.newInstance().newTransformer()
77     const res = new StreamResult(new StringWriter())
78     transformer.transform(new DOMSource(d), res)
79     return res.writer.toString()
80 }
81 
82 // serialize DOM to a String & print
83 print(domToXML(dom))
