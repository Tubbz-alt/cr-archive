32 // Java Map keys as properties
33 
34 // Demonstrating Java Map key/value can be accessed
35 // as property/value from script.
36 
37 var HashMap = Java.type("java.util.HashMap");
38 var map = new HashMap();
39 
40 // map key-value access by java get/put method calls
41 map.put('js', 'nashorn');
42 print(map.get('js'));
43 
44 // access keys of map as properties
45 print(map['js']);
46 print(map.js);
47 
48 // also assign new key-value pair
49 // as 'property-value'
50 map['language'] = 'java';
51 print(map.get("language"));
52 print(map.language);
53 print(map['language']);
54 
55 map.answer = 42;
56 print(map.get("answer"));
57 print(map.answer);
58 print(map['answer']);
