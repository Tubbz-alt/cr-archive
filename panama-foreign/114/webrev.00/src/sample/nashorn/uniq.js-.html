32 // Usage: jjs uniq.js
33 // or: jjs uniq.js -- <file>
34 
35 // omit repeated lines and print unique lines
36 
37 var BufferedReader = Java.type("java.io.BufferedReader");
38 var FileReader = Java.type("java.io.FileReader");
39 var InputStreamReader = Java.type("java.io.InputStreamReader");
40 var System = Java.type("java.lang.System");
41 
42 // use object as set - but insertion order preserved
43 var uniqueLines = {};
44 var reader = arguments.length > 0 ?
45     new FileReader(arguments[0])  :
46     new InputStreamReader(System.in);
47 reader = new BufferedReader(reader);
48 
49 // add unique lines
50 reader.lines().forEach(function(line) {
51     uniqueLines[line] = true;
52 })
53 
54 for (line in uniqueLines) {
55     print(line);
56 }
