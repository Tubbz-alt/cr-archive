32 // JavaImporter along with 'with' statement helps in
33 // localized Java class references
34 
35 function readTextFromURL(url) {
36 
37     // equivalent to
38     //
39     //    import java.io.*;
40     //    import java.net.*;
41     //    import java.lang.StringBuffer;
42     //
43     // only inside the 'with' statement
44     with (new JavaImporter(java.io,
45         java.net,
46         java.lang.StringBuilder)) {
47         var buf = new StringBuilder();
48         var u = new URL(url);
49         var reader = new BufferedReader(
50             new InputStreamReader(u.openStream()));
51         var line = null;
52         try {
53             while ((line = reader.readLine()) != null)
54                 buf.append(line).append('\n');
55         } finally {
56             reader.close();
57         }
58 
59         return buf.toString();
60     }
61 }
62 
63 print(readTextFromURL("https://twitter.com/"));
