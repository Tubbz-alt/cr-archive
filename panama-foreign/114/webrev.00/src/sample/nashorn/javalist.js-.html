32 // Java List elements accessed/modified via
33 // array element access/update syntax
34 
35 var ArrayList = Java.type("java.util.ArrayList");
36 var list = new ArrayList();
37 
38 // add elements to list by List's add method calls
39 list.add("js");
40 list.add("ecmascript");
41 list.add("nashorn");
42 
43 // get by List's get(int) method
44 print(list[0]);
45 print(list[1]);
46 print(list[2]);
47 
48 // access list elements by indexed access as well
49 print(list[0]);
50 print(list[1]);
51 print(list[2]);
52 
53 // assign to list elements by index as well
54 list[0] = list[0].toUpperCase();
55 list[1] = list[1].toUpperCase();
56 list[2] = list[2].toUpperCase();
57 
58 print(list.get(0));
59 print(list.get(1));
60 print(list.get(2));
61 print(list[0]);
62 print(list[1]);
63 print(list[2]);
