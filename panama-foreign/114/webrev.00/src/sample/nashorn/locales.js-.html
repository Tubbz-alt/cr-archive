32 // Simple program that lists available locals. This is ECMAScript
33 // port of Java example by @brunoborges
34 
35 // Java classes used
36 var Arrays = Java.type("java.util.Arrays");
37 var Collectors = Java.type("java.util.stream.Collectors");
38 var JString = Java.type("java.lang.String");
39 var Locale = Java.type("java.util.Locale");
40 
41 var formatStr = "Country : %s \t\t\t\t:\t Country Code : %s";
42 
43 // Nashorn allows script functions to be passed
44 // whereever Java8 lambdas are expected.
45 
46 // Nashorn also supports "expression closures" supported by
47 // Mozilla JavaScript 1.8 version. See also
48 // https://developer.mozilla.org/en-US/docs/Web/JavaScript/New_in_JavaScript/1.8
49 
50 // The following prints locales in (country) display name order
51 var list = Arrays.asList(Locale.getISOCountries())
52     .stream()
53     .map(function(x) new Locale("", x))
54     .sorted(function(c0, c1) c0.displayCountry.compareTo(c1.displayCountry))
55     .map(function(l) JString.format(formatStr, l.displayCountry, l.country))
56     .collect(Collectors.toList());
57 
58 list.forEach(print);
