32 // how to pass arbitrary "this" object for eval'ed script?
33 
34 var obj = { foo: 44 };
35 
36 // the following won't work. eval.apply is indirect eval call
37 // and so 'this' will be global object always! So, this
38 // line will print 'undefined'
39 eval.apply(obj, [ " print(this.foo)" ]);
40 
41 obj.myEval = eval;
42 
43 // still won't work - still indirect 'eval' call!
44 // still undefined is printed!
45 obj.myEval("print(this.foo)");
46 
47 function func(code) {
48     eval(code);
49 }
50 
51 // eval called inside func and so get's func's "this" as it's "this"!
52 // So, 44 is printed
53 
54 func.apply(obj, [ "print(this.foo)" ]);
