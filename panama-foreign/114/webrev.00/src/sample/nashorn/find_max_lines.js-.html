32 // Find the file with maximum number of lines
33 
34 var Files = Java.type("java.nio.file.Files");
35 var Integer = Java.type("java.lang.Integer");
36 var Paths = Java.type("java.nio.file.Paths");
37 
38 var file = arguments.length == 0? "." : arguments[0];
39 var ext = arguments.length > 1? arguments[1] : ".java";
40 var path = Paths.get(file);
41 
42 // return number of lines in given Path (that represents a file)
43 function lines(p) {
44     var strm = Files.lines(p);
45     try {
46         return strm.count();
47     } finally {
48         strm.close();
49     }
50 }
51 
52 // walk files, map to file and lines, find the max    
53 var obj = Files.find(path, Integer.MAX_VALUE, 
54     function(p, a) !p.toFile().isDirectory() && p.toString().endsWith(ext)).
55 map(function(p) ({ path : p, lines: lines(p) })).
56 max(function(x, y) x.lines - y.lines).get();
57 
58 // print path and lines of the max line file
59 print(obj.path, obj.lines);
