32 import javax.script.*;
33 import java.util.function.*;
34 
35 // example demonstrating that arbitrary Lambda can be called as function from script
36 
37 public class LambdaAsFunc {
38     public static void main(String[] args) throws Exception {
39         ScriptEngineManager m = new ScriptEngineManager();
40         ScriptEngine e = m.getEngineByName("nashorn");
41 
42         // expose a lambda as top-level script function
43         e.put("upper", (Function<String, String>) String::toUpperCase);
44 
45         // call lambda as though it is a normal script function
46         System.out.println(e.eval("upper('hello')"));
47     }
48 }
