32 // delete of scope vars is reported as error in strict mode.
33 // This script finds scripts that such deletes. Scripts in
34 // the specified directory are scanned. If no directory is 
35 // specified, the current directory is scanned.
36 
37 if (arguments.length == 0) {
38     arguments[0] = ".";
39 }
40 
41 var File = Java.type("java.io.File");
42 var file = new File(arguments[0]);
43 if (!file.exists()) {
44     print(arguments[0] + " is neither a directory nor a file");
45     exit(1);
46 }
47 
48 var Files = Java.type("java.nio.file.Files");
49 var IdentifierTree = Java.type("jdk.nashorn.api.tree.IdentifierTree");
50 var Parser = Java.type("jdk.nashorn.api.tree.Parser");
51 var SimpleTreeVisitor = Java.type("jdk.nashorn.api.tree.SimpleTreeVisitorES5_1");
52 var Tree = Java.type("jdk.nashorn.api.tree.Tree");
53 
54 var parser = Parser.create("-scripting", "--const-as-var");
55 
56 function checkFile(file) {
57     // print("checking " + file);
58     var ast = parser.parse(file, print);
59     if (!ast) {
60         return;
61     }
62 
63     // locate __proto__ usage and warn
64     ast.accept(visitor = new (Java.extend(SimpleTreeVisitor)) {
65         lineMap: null,
66 
67         printWarning: function(node, varName) {
68             var pos = node.startPosition;
69             var line = this.lineMap.getLineNumber(pos);
70             var column = this.lineMap.getColumnNumber(pos);
71             print("WARNING: delete " + varName + " in " + file + " @ " + line + ":" + column);
72         },
73 
74         visitCompilationUnit: function(node, extra) {
75             this.lineMap = node.lineMap;
76             Java.super(visitor).visitCompilationUnit(node, extra);
77         },
78 
79         visitUnary: function(node, extra) {
80             if (node.kind == Tree.Kind.DELETE &&
81                 node.expression instanceof IdentifierTree) {
82                 this.printWarning(node, node.expression.name);
83             } 
84             Java.super(visitor).visitUnary(node, extra);
85         },
86 
87     }, null);
88 }
89 
90 if (file.isDirectory()) {
91     Files.walk(file.toPath())
92         .filter(function(p) Files.isRegularFile(p))
93         .filter(function(p) p.toFile().name.endsWith('.js'))
94         .forEach(checkFile);
95 } else {
96     checkFile(file);
97 }
