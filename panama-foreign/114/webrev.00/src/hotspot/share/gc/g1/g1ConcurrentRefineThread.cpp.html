 36 G1ConcurrentRefineThread::G1ConcurrentRefineThread(G1ConcurrentRefine* cr, uint worker_id) :
 37   ConcurrentGCThread(),
 38   _vtime_start(0.0),
 39   _vtime_accum(0.0),
 40   _refinement_stats(new G1ConcurrentRefineStats()),
 41   _worker_id(worker_id),
 42   _notifier(new Semaphore(0)),
 43   _should_notify(true),
 44   _cr(cr)
 45 {
 46   // set name
 47   set_name("G1 Refine#%d", worker_id);
 48   create_and_start();
 49 }
 50 
 51 G1ConcurrentRefineThread::~G1ConcurrentRefineThread() {
 52   delete _refinement_stats;
 53   delete _notifier;
 54 }
 55 
 56 void G1ConcurrentRefineThread::wait_for_completed_buffers() {
 57   assert(this == Thread::current(), "precondition");
 58   while (Atomic::load_acquire(&_should_notify)) {
 59     _notifier->wait();
 60   }
 61 }
 62 
 63 void G1ConcurrentRefineThread::activate() {
 64   assert(this != Thread::current(), "precondition");
 65   // Notify iff transitioning from needing activation to not.  This helps
 66   // keep the semaphore count bounded and minimizes the work done by
 67   // activators when the thread is already active.
 68   if (Atomic::load_acquire(&_should_notify) &&
 69       Atomic::cmpxchg(&_should_notify, true, false)) {
 70     _notifier->signal();
 71   }
 72 }
 73 
 74 bool G1ConcurrentRefineThread::maybe_deactivate(bool more_work) {
 75   assert(this == Thread::current(), "precondition");
 76 
 77   if (more_work) {
 78     // Suppress unnecessary notifications.
 79     Atomic::release_store(&_should_notify, false);
 80     return false;
 81   } else if (Atomic::load_acquire(&_should_notify)) {
 82     // Deactivate if no notifications since enabled (see below).
 83     return true;
 84   } else {
 85     // Try for more refinement work with notifications enabled, to close
 86     // race; there could be a plethora of suppressed activation attempts
 87     // after we found no work but before we enable notifications here
 88     // (so there could be lots of work for this thread to do), followed
 89     // by a long time without activation after enabling notifications.
 90     // But first, clear any pending signals to prevent accumulation.
 91     while (_notifier->trywait()) {}
 92     Atomic::release_store(&_should_notify, true);
 93     return false;
 94   }
 95 }

 97 void G1ConcurrentRefineThread::run_service() {
 98   _vtime_start = os::elapsedVTime();
 99 
100   while (!should_terminate()) {
101     // Wait for work
102     wait_for_completed_buffers();
103     if (should_terminate()) {
104       break;
105     }
106 
107     log_debug(gc, refine)("Activated worker %d, on threshold: " SIZE_FORMAT ", current: " SIZE_FORMAT,
108                           _worker_id, _cr->activation_threshold(_worker_id),
109                           G1BarrierSet::dirty_card_queue_set().num_cards());
110 
111     // For logging.
112     G1ConcurrentRefineStats start_stats = *_refinement_stats;
113     G1ConcurrentRefineStats total_stats; // Accumulate over activation.
114 
115     {
116       SuspendibleThreadSetJoiner sts_join;
117 
118       while (!should_terminate()) {
119         if (sts_join.should_yield()) {
120           // Accumulate changed stats before possible GC that resets stats.
121           total_stats += *_refinement_stats - start_stats;
122           sts_join.yield();
123           // Reinitialize baseline stats after safepoint.
124           start_stats = *_refinement_stats;
125           continue;             // Re-check for termination after yield delay.
126         }
127 
128         bool more_work = _cr->do_refinement_step(_worker_id, _refinement_stats);
129         if (maybe_deactivate(more_work)) break;
130       }
131     }
132 
133     total_stats += *_refinement_stats - start_stats;
134     log_debug(gc, refine)("Deactivated worker %d, off threshold: " SIZE_FORMAT
135                           ", current: " SIZE_FORMAT
136                           ", refined cards: " SIZE_FORMAT,
137                           _worker_id, _cr->deactivation_threshold(_worker_id),
138                           G1BarrierSet::dirty_card_queue_set().num_cards(),
139                           total_stats.refined_cards());
140 
141     if (os::supports_vtime()) {
142       _vtime_accum = (os::elapsedVTime() - _vtime_start);
143     } else {
144       _vtime_accum = 0.0;
145     }
146   }
147 
148   log_debug(gc, refine)("Stopping %d", _worker_id);
149 }
150 
151 void G1ConcurrentRefineThread::stop_service() {
152   activate();
153 }
