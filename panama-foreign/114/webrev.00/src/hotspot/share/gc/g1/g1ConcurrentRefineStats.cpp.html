25 #include "precompiled.hpp"
26 #include "gc/g1/g1ConcurrentRefineStats.hpp"
27 
28 G1ConcurrentRefineStats::G1ConcurrentRefineStats() :
29   _refinement_time(),
30   _refined_cards(0),
31   _precleaned_cards(0),
32   _dirtied_cards(0)
33 {}
34 
35 G1ConcurrentRefineStats&
36 G1ConcurrentRefineStats::operator+=(const G1ConcurrentRefineStats& other) {
37   _refinement_time += other._refinement_time;
38   _refined_cards += other._refined_cards;
39   _precleaned_cards += other._precleaned_cards;
40   _dirtied_cards += other._dirtied_cards;
41   return *this;
42 }
43 
44 template<typename T>
45 static T clipped_sub(T x, T y) {
46   return (x < y) ? T() : (x - y);
47 }
48 
49 G1ConcurrentRefineStats&
50 G1ConcurrentRefineStats::operator-=(const G1ConcurrentRefineStats& other) {
51   _refinement_time = clipped_sub(_refinement_time, other._refinement_time);
52   _refined_cards = clipped_sub(_refined_cards, other._refined_cards);
53   _precleaned_cards = clipped_sub(_precleaned_cards, other._precleaned_cards);
54   _dirtied_cards = clipped_sub(_dirtied_cards, other._dirtied_cards);
55   return *this;
56 }
57 
58 void G1ConcurrentRefineStats::reset() {
59   *this = G1ConcurrentRefineStats();
60 }
