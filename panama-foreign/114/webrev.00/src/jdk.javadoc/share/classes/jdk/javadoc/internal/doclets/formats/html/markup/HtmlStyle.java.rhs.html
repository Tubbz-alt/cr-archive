370     private final String cssName;
371 
372     HtmlStyle() {
373         cssName = Pattern.compile("\\p{Upper}")
374                 .matcher(toString())
375                 .replaceAll(mr -> "-" + mr.group().toLowerCase(Locale.US));
376     }
377 
378     HtmlStyle(String cssName) {
379         this.cssName = cssName;
380     }
381 
382     /**
383      * Returns the CSS class name associated with this style.
384      * @return the CSS class name
385      */
386     public String cssName() {
387         return cssName;
388     }
