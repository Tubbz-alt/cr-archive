 51 /***************************************************************/
 52 void mlib_v_ImageLookUp_U16_U8_124_D1(const mlib_u16 *src,
 53                                       mlib_u8        *dst,
 54                                       mlib_s32       xsize,
 55                                       const mlib_u8  *table0,
 56                                       const mlib_u8  *table1,
 57                                       const mlib_u8  *table2,
 58                                       const mlib_u8  *table3)
 59 {
 60   mlib_u16 *sp;                        /* pointer to source data */
 61   mlib_s32 s0, s1, s2, s3;             /* source data */
 62   mlib_s32 s4, s5, s6, s7;             /* source data */
 63   mlib_u8 *dl;                         /* pointer to start of destination */
 64   mlib_u8 *dend;                       /* pointer to end of destination */
 65   mlib_d64 *dp;                        /* aligned pointer to destination */
 66   mlib_d64 t0, t1, t2;                 /* destination data */
 67   mlib_d64 t3, t4, t5;                 /* destination data */
 68   mlib_d64 t6, t7, acc;                /* destination data */
 69   mlib_s32 emask;                      /* edge mask */
 70   mlib_s32 i, num;                     /* loop variable */
 71 
 72   dl = dst;
 73   dp = (mlib_d64 *) dl;
 74   dend = dl + xsize - 1;
 75   sp = (void *)src;
 76 
 77   vis_alignaddr((void *)0, 7);
 78 
 79   if (xsize >= 8) {
 80 
 81     s0 = sp[0];
 82     s1 = sp[1];
 83     s2 = sp[2];
 84     s3 = sp[3];
 85     s4 = sp[4];
 86     s5 = sp[5];
 87     s6 = sp[6];
 88     s7 = sp[7];
 89     sp += 8;
 90 
 91 #pragma pipeloop(0)
 92     for (i = 0; i <= xsize - 16; i += 8, sp += 8) {
 93       t7 = VIS_LD_U8_I(table3, s7);
 94       t6 = VIS_LD_U8_I(table2, s6);
 95       t5 = VIS_LD_U8_I(table1, s5);
 96       t4 = VIS_LD_U8_I(table0, s4);
 97       t3 = VIS_LD_U8_I(table3, s3);
 98       t2 = VIS_LD_U8_I(table2, s2);
 99       t1 = VIS_LD_U8_I(table1, s1);
100       t0 = VIS_LD_U8_I(table0, s0);
101       acc = vis_faligndata(t7, acc);
102       acc = vis_faligndata(t6, acc);
103       acc = vis_faligndata(t5, acc);
104       acc = vis_faligndata(t4, acc);
105       acc = vis_faligndata(t3, acc);
106       acc = vis_faligndata(t2, acc);
107       acc = vis_faligndata(t1, acc);
108       acc = vis_faligndata(t0, acc);
109       s0 = sp[0];
110       s1 = sp[1];
111       s2 = sp[2];
112       s3 = sp[3];
113       s4 = sp[4];
114       s5 = sp[5];
115       s6 = sp[6];
116       s7 = sp[7];
117       *dp++ = acc;
118     }
119 
120     t7 = VIS_LD_U8_I(table3, s7);
121     t6 = VIS_LD_U8_I(table2, s6);
122     t5 = VIS_LD_U8_I(table1, s5);
123     t4 = VIS_LD_U8_I(table0, s4);
124     t3 = VIS_LD_U8_I(table3, s3);
125     t2 = VIS_LD_U8_I(table2, s2);
126     t1 = VIS_LD_U8_I(table1, s1);
127     t0 = VIS_LD_U8_I(table0, s0);
128     acc = vis_faligndata(t7, acc);
129     acc = vis_faligndata(t6, acc);
130     acc = vis_faligndata(t5, acc);
131     acc = vis_faligndata(t4, acc);
132     acc = vis_faligndata(t3, acc);
133     acc = vis_faligndata(t2, acc);
134     acc = vis_faligndata(t1, acc);
135     acc = vis_faligndata(t0, acc);
136     *dp++ = acc;
137   }
138 
139   if ((mlib_addr) dp <= (mlib_addr) dend) {
140 
141     num = (mlib_addr) dend - (mlib_addr) dp;
142     sp += num;
143     num++;
144 
145     if ((num & 3) == 1) {
146       s0 = (mlib_s32) * sp;
147       sp--;
148 
149       t0 = VIS_LD_U8_I(table0, s0);
150       acc = vis_faligndata(t0, acc);
151       num--;
152     }
153     else if ((num & 3) == 2) {
154       s0 = (mlib_s32) * sp;
155       sp--;
156 
157       t0 = VIS_LD_U8_I(table1, s0);
158       acc = vis_faligndata(t0, acc);
159 
160       s0 = (mlib_s32) * sp;
161       sp--;
162 
163       t0 = VIS_LD_U8_I(table0, s0);
164       acc = vis_faligndata(t0, acc);
165       num -= 2;
166     }
167     else if ((num & 3) == 3) {
168       s0 = (mlib_s32) * sp;
169       sp--;
170 
171       t0 = VIS_LD_U8_I(table2, s0);
172       acc = vis_faligndata(t0, acc);
173 
174       s0 = (mlib_s32) * sp;
175       sp--;
176 
177       t0 = VIS_LD_U8_I(table1, s0);
178       acc = vis_faligndata(t0, acc);
179 
180       s0 = (mlib_s32) * sp;
181       sp--;
182 
183       t0 = VIS_LD_U8_I(table0, s0);
184       acc = vis_faligndata(t0, acc);
185       num -= 3;
186     }
187 
188     if (num != 0) {
189       s0 = (mlib_s32) * sp;
190       sp--;
191 
192       t0 = VIS_LD_U8_I(table3, s0);
193       acc = vis_faligndata(t0, acc);
194 
195       s0 = (mlib_s32) * sp;
196       sp--;
197 
198       t0 = VIS_LD_U8_I(table2, s0);
199       acc = vis_faligndata(t0, acc);
200 
201       s0 = (mlib_s32) * sp;
202       sp--;
203 
204       t0 = VIS_LD_U8_I(table1, s0);
205       acc = vis_faligndata(t0, acc);
206 
207       s0 = (mlib_s32) * sp;
208       sp--;
209 
210       t0 = VIS_LD_U8_I(table0, s0);
211       acc = vis_faligndata(t0, acc);
212     }
213 
214     emask = vis_edge8(dp, dend);
215     vis_pst_8(acc, dp, emask);
216   }
217 }

219 /***************************************************************/
220 void mlib_v_ImageLookUp_U16_U8_1(const mlib_u16 *src,
221                                  mlib_s32       slb,
222                                  mlib_u8        *dst,
223                                  mlib_s32       dlb,
224                                  mlib_s32       xsize,
225                                  mlib_s32       ysize,
226                                  const mlib_u8  **table)
227 {
228   mlib_u16 *sl;
229   mlib_u8 *dl;
230   const mlib_u8 *tab = &table[0][0];
231   mlib_s32 j, i;
232 
233   sl = (void *)src;
234   dl = dst;
235 
236   /* row loop */
237   for (j = 0; j < ysize; j++) {
238     mlib_u16 *sp = sl;
239     mlib_u8 *dp = dl;
240     mlib_s32 off, size = xsize;
241 
242     off = (8 - ((mlib_addr) dp & 7)) & 7;
243 
244     off = (off < size) ? off : size;
245 
246     for (i = 0; i < off; i++, sp++) {
247       *dp++ = tab[sp[0]];
248       size--;
249     }
250 
251     if (size > 0) {
252       mlib_v_ImageLookUp_U16_U8_124_D1(sp, dp, size, tab, tab, tab, tab);
253     }
254 
255     sl = (mlib_u16 *) ((mlib_u8 *) sl + slb);
256     dl = (mlib_u8 *) ((mlib_u8 *) dl + dlb);
257   }
258 }

260 /***************************************************************/
261 void mlib_v_ImageLookUp_U16_U8_2(const mlib_u16 *src,
262                                  mlib_s32       slb,
263                                  mlib_u8        *dst,
264                                  mlib_s32       dlb,
265                                  mlib_s32       xsize,
266                                  mlib_s32       ysize,
267                                  const mlib_u8  **table)
268 {
269   mlib_u16 *sl;
270   mlib_u8 *dl;
271   const mlib_u8 *tab;
272   mlib_s32 j, i;
273 
274   sl = (void *)src;
275   dl = dst;
276 
277   /* row loop */
278   for (j = 0; j < ysize; j++) {
279     mlib_u16 *sp = sl;
280     mlib_u8 *dp = dl;
281     mlib_s32 off, size = xsize * 2;
282     const mlib_u8 *tab0 = &table[0][0];
283     const mlib_u8 *tab1 = &table[1][0];
284 
285     off = (8 - ((mlib_addr) dp & 7)) & 7;
286 
287     off = (off < size) ? off : size;
288 
289     for (i = 0; i < off - 1; i += 2, sp += 2) {
290       *dp++ = tab0[sp[0]];
291       *dp++ = tab1[sp[1]];
292       size -= 2;
293     }
294 
295     if ((off & 1) != 0) {
296       *dp++ = tab0[sp[0]];
297       size--;
298       sp++;
299       tab = tab0;
300       tab0 = tab1;
301       tab1 = tab;
302     }
303 
304     if (size > 0) {
305       mlib_v_ImageLookUp_U16_U8_124_D1(sp, dp, size, tab0, tab1, tab0, tab1);
306     }
307 
308     sl = (mlib_u16 *) ((mlib_u8 *) sl + slb);
309     dl = (mlib_u8 *) ((mlib_u8 *) dl + dlb);
310   }
311 }

345       *dp++ = tab0[sp[0]];

363     }

370       tab = tab1;

381       tab3 = tab2;
