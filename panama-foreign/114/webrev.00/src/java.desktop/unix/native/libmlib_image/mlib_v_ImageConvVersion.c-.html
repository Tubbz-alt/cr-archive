28 /*
29  * FUNCTION
30  *      mlib_ImageConvVersion - Get Conv* funtions version
31  *      0  - "C" version
32  *      1  - "VIS" version
33  *      2  - "i386" version
34  *      3  - "MMX" version
35  *
36  * SYNOPSIS
37  *      mlib_s32 mlib_ImageConvVersion(mlib_s32  m,
38  *                                     mlib_s32  n,
39  *                                     mlib_s32  scale,
40  *                                     mlib_type type)
41  *
42  */
43 
44 #include "mlib_image.h"
45 
46 /***************************************************************/
47 #define MAX_U8   8
48 #define MAX_S16 32
49 
50 /***************************************************************/
51 mlib_s32 mlib_ImageConvVersion(mlib_s32  m,
52                                mlib_s32  n,
53                                mlib_s32  scale,
54                                mlib_type type)
55 {
56   mlib_d64 dscale = 1.0 / (1 << scale); /* 16 < scale <= 31 */
57 
58   if (type == MLIB_BYTE) {
59     if ((m * n * dscale * 32768.0 * 512.0) > MAX_U8)
60       return 0;
61     return 1;
62   }
63   else if ((type == MLIB_USHORT) || (type == MLIB_SHORT)) {
64 
65     if ((m != 2) || (n != 2))
66       return 0;
67 
68     if ((m * n * dscale * 32768.0 * 32768.0) > MAX_S16)
69       return 0;
70     return 1;
71   }
72   else
73     return 0;
74 }
