26 #if !defined(_DASSERT_H)
27 #define _DASSERT_H
28 
29 #if defined(__cplusplus)
30 extern "C" {
31 #endif
32 
33 #include "debug_util.h"
34 





35 #if defined(DEBUG)
36 
37 #define DASSERT(_expr) \
38         if ( !(_expr) ) { \
39             DAssert_Impl( #_expr, __FILE__, __LINE__); \
40         } else { \
41         }
42 
43 #define DASSERTMSG(_expr, _msg) \
44         if ( !(_expr) ) { \
45             DAssert_Impl( (_msg), __FILE__, __LINE__); \
46         } else { \
47         }
48 
49 /* prototype for assert function */
50 typedef void (*DASSERT_CALLBACK)(const char * msg, const char * file, int line);
51 
52 /* JNIEXPORT because this function is also called from libawt_xawt */
53 JNIEXPORT void JNICALL DAssert_Impl(const char * msg, const char * file, int line);
54 extern void DAssert_SetCallback( DASSERT_CALLBACK pfn );
55 
56 #else /* DEBUG not defined */
57 
58 #define DASSERT(_expr)
59 #define DASSERTMSG(_expr, _msg)
60 
61 #endif /* if defined(DEBUG) */
62 
63 #if defined(__cplusplus)
64 } /* extern "C" */
65 #endif
66 
67 #endif /* _DASSERT_H */
