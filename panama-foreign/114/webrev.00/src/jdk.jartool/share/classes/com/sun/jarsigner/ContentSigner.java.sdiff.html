26 package com.sun.jarsigner;
27 
28 import java.io.IOException;
29 import java.security.NoSuchAlgorithmException;
30 import java.security.cert.CertificateException;
31 
32 /**
33  * This class defines a content signing service.
34  * Implementations must be instantiable using a zero-argument constructor.
35  *
36  * @since 1.5
37  * @author Vincent Ryan
38  * @deprecated This class has been deprecated.
39  */
40 
41 @Deprecated(since="9")
42 public abstract class ContentSigner {
43 
44     /**
45      * Generates a PKCS #7 signed data message.
46      * This method is used when the signature has already been generated.
47      * The signature, the signer's details, and optionally a signature
48      * timestamp and the content that was signed, are all packaged into a
49      * signed data message.
50      *
51      * @param parameters The non-null input parameters.
52      * @param omitContent true if the content should be omitted from the
53      *         signed data message. Otherwise the content is included.
54      * @param applyTimestamp true if the signature should be timestamped.
55      *         Otherwise timestamping is not performed.
56      * @return A PKCS #7 signed data message.
57      * @throws NoSuchAlgorithmException The exception is thrown if the signature
58      *         algorithm is unrecognised.
59      * @throws CertificateException The exception is thrown if an error occurs
60      *         while processing the signer's certificate or the TSA's
61      *         certificate.
62      * @throws IOException The exception is thrown if an error occurs while
63      *         generating the signature timestamp or while generating the signed
64      *         data message.
65      * @throws NullPointerException The exception is thrown if parameters is
66      *         null.
67      */

68     public abstract byte[] generateSignedData(
69         ContentSignerParameters parameters, boolean omitContent,
70         boolean applyTimestamp)
71             throws NoSuchAlgorithmException, CertificateException, IOException;
72 }
