26 package jdk.nashorn.internal.ir;
27 
28 import jdk.nashorn.internal.ir.annotations.Immutable;
29 import jdk.nashorn.internal.ir.visitor.NodeVisitor;
30 
31 /**
32  * IR representation for a debugger statement.
33  */
34 @Immutable
35 public final class DebuggerNode extends Statement {
36     private static final long serialVersionUID = 1L;
37 
38     /**
39      * Constructor
40      *
41      * @param lineNumber line number
42      * @param token      token
43      * @param finish     finish
44      */
45     public DebuggerNode(final int lineNumber, final long token, final int finish) {
46         super(lineNumber, token, finish);
47     }
48 
49     @Override
50     public Node accept(final NodeVisitor<? extends LexicalContext> visitor) {
51         if (visitor.enterDebuggerNode(this)) {
52             return visitor.leaveDebuggerNode(this);
53         }
54 
55         return this;
56     }
57 
58     @Override
59     public void toString(final StringBuilder sb, final boolean printType) {
60         sb.append("debugger");
61     }
62 }
