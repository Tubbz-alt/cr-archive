20 package jdk.nashorn.internal.runtime.regexp.joni.constants;
21 
22 @SuppressWarnings("javadoc")
23 public interface StackType {
24     /** stack **/
25     final int INVALID_STACK_INDEX           = -1;
26 
27     /* stack type */
28     /* used by normal-POP */
29     final int ALT                           = 0x0001;
30     final int LOOK_BEHIND_NOT               = 0x0002;
31     final int POS_NOT                       = 0x0003;
32     /* handled by normal-POP */
33     final int MEM_START                     = 0x0100;
34     final int MEM_END                       = 0x8200;
35     final int REPEAT_INC                    = 0x0300;
36     final int STATE_CHECK_MARK              = 0x1000;
37     /* avoided by normal-POP */
38     final int NULL_CHECK_START              = 0x3000;
39     final int NULL_CHECK_END                = 0x5000;  /* for recursive call */
40     final int MEM_END_MARK                  = 0x8400;
41     final int POS                           = 0x0500;  /* used when POP-POS */
42     final int STOP_BT                       = 0x0600;  /* mark for "(?>...)" */
43     final int REPEAT                        = 0x0700;
44     final int CALL_FRAME                    = 0x0800;
45     final int RETURN                        = 0x0900;
46     final int VOID                          = 0x0a00;  /* for fill a blank */
47 
48     /* stack type check mask */
49     final int MASK_POP_USED                 = 0x00ff;
50     final int MASK_TO_VOID_TARGET           = 0x10ff;
51     final int MASK_MEM_END_OR_MARK          = 0x8000;  /* MEM_END or MEM_END_MARK */
52 }
