25 package jdk.nashorn.internal.parser;
26 
27 import java.util.List;
28 import jdk.nashorn.internal.ir.Statement;
29 
30 /**
31  * Used for keeping state when needed in the parser.
32  */
33 interface ParserContextNode {
34     /**
35      * @return The flags for this node
36      */
37     public int getFlags();
38 
39     /**
40      * @param flag The flag to set
41      * @return All current flags after update
42      */
43     public int setFlag(final int flag);
44 
45     /**
46      * @return The list of statements that belongs to this node
47      */
48     public List<Statement> getStatements();
49 
50     /**
51      * @param statements The statement list
52      */
53     public void setStatements(final List<Statement> statements);
54 
55     /**
56      * Adds a statement at the end of the statement list
57      * @param statement The statement to add
58      */
59     public void appendStatement(final Statement statement);
60 
61     /**
62      * Adds a statement at the beginning of the statement list
63      * @param statement The statement to add
64      */
65     public void prependStatement(final Statement statement);
66 
67 }
