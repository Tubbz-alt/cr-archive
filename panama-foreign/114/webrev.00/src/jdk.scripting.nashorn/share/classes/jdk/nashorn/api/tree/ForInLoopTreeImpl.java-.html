26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.ir.ForNode;
29 
30 final class ForInLoopTreeImpl extends StatementTreeImpl implements ForInLoopTree {
31     private final ExpressionTree lhsExpr;
32     private final ExpressionTree expr;
33     private final StatementTree stat;
34     private final boolean forEach;
35 
36     ForInLoopTreeImpl(final ForNode node,
37             final ExpressionTree lhsExpr,
38             final ExpressionTree expr,
39             final StatementTree stat) {
40         super(node);
41         assert node.isForIn() : "for ..in expected";
42         this.lhsExpr = lhsExpr;
43         this.expr = expr;
44         this.stat = stat;
45         this.forEach = node.isForEach();
46     }
47 
48     @Override
49     public Kind getKind() {
50         return Kind.FOR_IN_LOOP;
51     }
52 
53     @Override
54     public ExpressionTree getVariable() {
55         return lhsExpr;
56     }
57 
58     @Override
59     public ExpressionTree getExpression() {
60         return expr;
61     }
62 
63     @Override
64     public StatementTree getStatement() {
65         return stat;
66     }
67 
68     @Override
69     public boolean isForEach() {
70         return forEach;
71     }
72 
73     @Override
74     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
75         return visitor.visitForInLoop(this, data);
76     }
77 }
