26 package jdk.nashorn.api.tree;
27 
28 import java.util.List;
29 import jdk.nashorn.internal.ir.FunctionNode;
30 import jdk.nashorn.internal.ir.VarNode;
31 
32 final class FunctionDeclarationTreeImpl extends StatementTreeImpl
33     implements FunctionDeclarationTree {
34     private final FunctionNode funcNode;
35     private final IdentifierTree funcName;
36     private final List<? extends ExpressionTree> params;
37     private final BlockTree body;
38 
39     FunctionDeclarationTreeImpl(final VarNode node,
40             final List<? extends ExpressionTree> params,
41             final BlockTree body) {
42         super(node);
43         assert node.getInit() instanceof FunctionNode : "function expected";
44         funcNode = (FunctionNode)node.getInit();
45         assert funcNode.isDeclared() : "function declaration expected";
46         funcName = funcNode.isAnonymous()? null : new IdentifierTreeImpl(node.getName());
47         this.params = params;
48         this.body = body;
49     }
50 
51     @Override
52     public Kind getKind() {
53         return Kind.FUNCTION;
54     }
55 
56     @Override
57     public IdentifierTree getName() {
58         return funcName;
59     }
60 
61     @Override
62     public List<? extends ExpressionTree> getParameters() {
63         return params;
64     }
65 
66     @Override
67     public BlockTree getBody() {
68         return body;
69     }
70 
71     @Override
72     public boolean isStrict() {
73         return funcNode.isStrict();
74     }
75 
76     @Override
77     public boolean isGenerator() {
78         return funcNode.getKind() == FunctionNode.Kind.GENERATOR;
79     }
80 
81     @Override
82     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
83         return visitor.visitFunctionDeclaration(this, data);
84     }
85 }
