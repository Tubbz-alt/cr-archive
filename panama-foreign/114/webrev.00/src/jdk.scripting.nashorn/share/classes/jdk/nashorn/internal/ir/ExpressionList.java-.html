26 package jdk.nashorn.internal.ir;
27 
28 import java.util.Collections;
29 import java.util.List;
30 
31 import jdk.nashorn.internal.codegen.types.Type;
32 import jdk.nashorn.internal.ir.visitor.NodeVisitor;
33 
34 /**
35  * IR for CoverParenthesizedExpressionAndArrowParameterList, used only during parsing.
36  */
37 public final class ExpressionList extends Expression {
38     private static final long serialVersionUID = 1L;
39 
40     private final List<Expression> expressions;
41 
42     /**
43      * Constructor.
44      *
45      * @param token token
46      * @param finish finish
47      * @param expressions expression
48      */
49     public ExpressionList(final long token, final int finish, final List<Expression> expressions) {
50         super(token, finish);
51         this.expressions = expressions;
52     }
53 
54     /**
55      * Get the list of expressions.
56      *
57      * @return the list of expressions
58      */
59     public List<Expression> getExpressions() {
60         return Collections.unmodifiableList(expressions);
61     }
62 
63     @Override
64     public Node accept(final NodeVisitor<? extends LexicalContext> visitor) {
65         throw new UnsupportedOperationException();
66     }
67 
68     @Override
69     public Type getType() {
70         return null;
71     }
72 
73     @Override
74     public void toString(final StringBuilder sb, final boolean printType) {
75         sb.append("(");
76         boolean first = true;
77         for (final Expression expression : expressions) {
78             if (first) {
79                 first = false;
80             } else {
81                 sb.append(", ");
82             }
83             expression.toString(sb, printType);
84         }
85         sb.append(")");
86     }
87 }
