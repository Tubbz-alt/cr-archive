20 package jdk.nashorn.internal.runtime.regexp.joni;
21 
22 @SuppressWarnings("javadoc")
23 public final class Region {
24     static final int REGION_NOTPOS = -1;
25 
26     public final int numRegs;
27     public final int[]beg;
28     public final int[]end;
29 
30     public Region(final int num) {
31         this.numRegs = num;
32         this.beg = new int[num];
33         this.end = new int[num];
34     }
35 
36     @Override
37     public String toString() {
38         final StringBuilder sb = new StringBuilder();
39         sb.append("Region: \n");
40         for (int i=0; i<beg.length; i++) {
41             sb.append(" ").append(i).append(": (").append(beg[i]).append("-").append(end[i]).append(")");
42         }
43         return sb.toString();
44     }
45 
46     void clear() {
47         for (int i=0; i<beg.length; i++) {
48             beg[i] = end[i] = REGION_NOTPOS;
49         }
50     }
51 }
