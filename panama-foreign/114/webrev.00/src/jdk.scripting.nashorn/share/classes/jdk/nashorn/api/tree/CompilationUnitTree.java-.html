26 package jdk.nashorn.api.tree;
27 
28 import java.util.List;
29 
30 /**
31  * Represents the abstract syntax tree for compilation units (source
32  * files)
33  *
34  * @deprecated Nashorn JavaScript script engine and APIs, and the jjs tool
35  * are deprecated with the intent to remove them in a future release.
36  *
37  * @since 9
38  */
39 @Deprecated(since="11", forRemoval=true)
40 public interface CompilationUnitTree extends Tree {
41     /**
42      * Return the list of source elements in this compilation unit.
43      *
44      * @return the list of source elements in this compilation unit
45      */
46     List<? extends Tree> getSourceElements();
47 
48     /**
49      * Return the source name of this script compilation unit.
50      *
51      * @return the source name of this script compilation unit
52      */
53     String getSourceName();
54 
55     /**
56      * Returns if this is a ECMAScript "strict" compilation unit or not.
57      *
58      * @return true if this compilation unit is declared "strict"
59      */
60     boolean isStrict();
61 
62     /**
63      * Returns the line map for this compilation unit, if available.
64      * Returns null if the line map is not available.
65      *
66      * @return the line map for this compilation unit
67      */
68     LineMap getLineMap();
69 
70     /**
71      * Return the {@link ModuleTree} associated with this compilation unit. This is null,
72      * if there is no module information from this compilation unit.
73      *
74      * @return the Module info or null
75      */
76     ModuleTree getModule();
77 }
