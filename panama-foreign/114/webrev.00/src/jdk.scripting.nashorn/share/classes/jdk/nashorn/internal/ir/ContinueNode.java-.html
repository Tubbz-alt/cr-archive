26 package jdk.nashorn.internal.ir;
27 
28 import jdk.nashorn.internal.codegen.Label;
29 import jdk.nashorn.internal.ir.annotations.Immutable;
30 import jdk.nashorn.internal.ir.visitor.NodeVisitor;
31 
32 /**
33  * IR representation for CONTINUE statements.
34  */
35 @Immutable
36 public class ContinueNode extends JumpStatement {
37     private static final long serialVersionUID = 1L;
38 
39     /**
40      * Constructor
41      *
42      * @param lineNumber line number
43      * @param token      token
44      * @param finish     finish
45      * @param labelName  label name for continue or null if none
46      */
47     public ContinueNode(final int lineNumber, final long token, final int finish, final String labelName) {
48         super(lineNumber, token, finish, labelName);
49     }
50 
51     private ContinueNode(final ContinueNode continueNode, final LocalVariableConversion conversion) {
52         super(continueNode, conversion);
53     }
54 
55     @Override
56     public Node accept(final NodeVisitor<? extends LexicalContext> visitor) {
57         if (visitor.enterContinueNode(this)) {
58             return visitor.leaveContinueNode(this);
59         }
60 
61         return this;
62     }
63 
64     @Override
65     JumpStatement createNewJumpStatement(final LocalVariableConversion conversion) {
66         return new ContinueNode(this, conversion);
67     }
68 
69     @Override
70     String getStatementName() {
71         return "continue";
72     }
73 
74 
75     @Override
76     public BreakableNode getTarget(final LexicalContext lc) {
77         return lc.getContinueTo(getLabelName());
78     }
79 
80     @Override
81     Label getTargetLabel(final BreakableNode target) {
82         return ((LoopNode)target).getContinueLabel();
83     }
84 }
