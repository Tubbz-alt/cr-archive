26 package jdk.nashorn.internal.runtime;
27 
28 import java.util.MissingResourceException;
29 import java.util.ResourceBundle;
30 
31 /**
32  * Class to handle version strings for Nashorn.
33  */
34 public final class Version {
35     // Don't create me!
36     private Version() {
37     }
38 
39     /**
40      * The current version number as a string.
41      * @return version string
42      */
43     public static String version() {
44         return version("version_short");  // E.g. "9-internal" or "9.1.2"
45     }
46 
47     /**
48      * The current full version number as a string.
49      * @return full version string
50      */
51     public static String fullVersion() {
52         return version("version_string"); // E.g. "9.1.2.3-ea-4+nashorn-testing"
53     }
54 
55     private static final String   VERSION_RB_NAME = "jdk.nashorn.internal.runtime.resources.version";
56     private static ResourceBundle versionRB;
57 
58     private static String version(final String key) {
59         if (versionRB == null) {
60             try {
61                 versionRB = ResourceBundle.getBundle(VERSION_RB_NAME);
62             } catch (final MissingResourceException e) {
63                 return "version not available";
64             }
65         }
66         try {
67             return versionRB.getString(key);
68         }
69         catch (final MissingResourceException e) {
70             return "version not available";
71         }
72     }
73 }
