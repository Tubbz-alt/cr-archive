26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.ir.LiteralNode;
29 
30 final class LiteralTreeImpl extends ExpressionTreeImpl implements LiteralTree {
31     private final Object value;
32     private final Kind kind;
33     LiteralTreeImpl(final LiteralNode<?> node) {
34         super(node);
35         this.kind = literalKind(node);
36         this.value = node.getValue();
37     }
38 
39     @Override
40     public Kind getKind() {
41         return kind;
42     }
43 
44     @Override
45     public Object getValue() {
46         return value;
47     }
48 
49     private static Kind literalKind(final LiteralNode<?> node) {
50         if (node.isBoolean()) {
51             return Kind.BOOLEAN_LITERAL;
52         } else if (node.isNumeric()) {
53             return Kind.NUMBER_LITERAL;
54         } else if (node.isString()) {
55             return Kind.STRING_LITERAL;
56         } else if (node.isNull()) {
57             return Kind.NULL_LITERAL;
58         } else {
59             throw new AssertionError("should not reach here: " + node.getValue());
60         }
61     }
62 
63     @Override
64     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
65         return visitor.visitLiteral(this, data);
66     }
67 }
