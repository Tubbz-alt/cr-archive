26 package jdk.nashorn.internal.runtime.arrays;
27 
28 import java.util.NoSuchElementException;
29 
30 /**
31  * Dummy array iterator that has no elements
32  */
33 final class EmptyArrayLikeIterator extends ArrayLikeIterator<Object> {
34 
35     EmptyArrayLikeIterator() {
36         super(false);
37     }
38 
39     @Override
40     public boolean hasNext() {
41         return false;
42     }
43 
44     @Override
45     public Object next() {
46         throw new NoSuchElementException();
47     }
48 
49     @Override
50     public long getLength() {
51         return 0;
52     }
53 }
