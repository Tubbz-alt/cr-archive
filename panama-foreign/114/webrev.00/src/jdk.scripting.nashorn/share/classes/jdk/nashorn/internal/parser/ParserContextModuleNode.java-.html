26 package jdk.nashorn.internal.parser;
27 
28 import java.util.ArrayList;
29 import java.util.List;
30 
31 import jdk.nashorn.internal.ir.IdentNode;
32 import jdk.nashorn.internal.ir.Module;
33 import jdk.nashorn.internal.ir.Module.ExportEntry;
34 import jdk.nashorn.internal.ir.Module.ImportEntry;
35 
36 /**
37  * ParserContextNode that represents a module.
38  */
39 class ParserContextModuleNode extends ParserContextBaseNode {
40 
41     /** Module name. */
42     private final String name;
43 
44     private final List<String> requestedModules = new ArrayList<>();
45     private final List<ImportEntry> importEntries = new ArrayList<>();
46     private final List<ExportEntry> localExportEntries = new ArrayList<>();
47     private final List<ExportEntry> indirectExportEntries = new ArrayList<>();
48     private final List<ExportEntry> starExportEntries = new ArrayList<>();
49 
50     /**
51      * Constructor.
52      *
53      * @param name name of the module
54      */
55     ParserContextModuleNode(final String name) {
56         this.name = name;
57     }
58 
59     /**
60      * Returns the name of the module.
61      *
62      * @return name of the module
63      */
64     public String getModuleName() {
65         return name;
66     }
67 
68     public void addModuleRequest(final IdentNode moduleRequest) {
69         requestedModules.add(moduleRequest.getName());
70     }
71 
72     public void addImportEntry(final ImportEntry importEntry) {
73         importEntries.add(importEntry);
74     }
75 
76     public void addLocalExportEntry(final ExportEntry exportEntry) {
77         localExportEntries.add(exportEntry);
78     }
79 
80     public void addIndirectExportEntry(final ExportEntry exportEntry) {
81         indirectExportEntries.add(exportEntry);
82     }
83 
84     public void addStarExportEntry(final ExportEntry exportEntry) {
85         starExportEntries.add(exportEntry);
86     }
87 
88     public Module createModule() {
89         return new Module(requestedModules, importEntries, localExportEntries, indirectExportEntries, starExportEntries);
90     }
91 }
