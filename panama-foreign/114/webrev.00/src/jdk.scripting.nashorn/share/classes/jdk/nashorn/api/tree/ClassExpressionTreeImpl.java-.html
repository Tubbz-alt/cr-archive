25 package jdk.nashorn.api.tree;
26 
27 import java.util.List;
28 import jdk.nashorn.internal.ir.ClassNode;
29 
30 final class ClassExpressionTreeImpl extends ExpressionTreeImpl implements ClassExpressionTree {
31 
32     private final IdentifierTree name;
33     private final ExpressionTree classHeritage;
34     private final PropertyTree constructor;
35     private final List<? extends PropertyTree> classElements;
36 
37     ClassExpressionTreeImpl(final ClassNode cn, final IdentifierTree name,
38             final ExpressionTree classHeritage, final PropertyTree constructor,
39             final List<? extends PropertyTree> classElements) {
40         super(cn);
41         this.name = name;
42         this.classHeritage = classHeritage;
43         this.constructor = constructor;
44         this.classElements = classElements;
45     }
46 
47     @Override
48     public Kind getKind() {
49         return Kind.CLASS_EXPRESSION;
50     }
51 
52     @Override
53     public IdentifierTree getName() {
54         return name;
55     }
56 
57     @Override
58     public ExpressionTree getClassHeritage() {
59         return classHeritage;
60     }
61 
62     @Override
63     public PropertyTree getConstructor() {
64         return constructor;
65     }
66 
67     @Override
68     public List<? extends PropertyTree> getClassElements() {
69         return classElements;
70     }
71 
72     @Override
73     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
74         return visitor.visitClassExpression(this, data);
75     }
76 }
