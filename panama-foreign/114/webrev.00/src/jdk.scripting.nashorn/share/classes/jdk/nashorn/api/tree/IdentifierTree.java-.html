26 package jdk.nashorn.api.tree;
27 
28 /**
29  * A tree node for an identifier expression.
30  *
31  * For example:
32  * <pre>
33  *   <em>name</em>
34  * </pre>
35  *
36  * @deprecated Nashorn JavaScript script engine and APIs, and the jjs tool
37  * are deprecated with the intent to remove them in a future release.
38  *
39  * @since 9
40  */
41 @Deprecated(since="11", forRemoval=true)
42 public interface IdentifierTree extends ExpressionTree {
43     /**
44      * Returns the name of this identifier.
45      *
46      * @return the name of this identifier
47      */
48     String getName();
49 
50     /**
51      * Is this a rest parameter for a function or rest elements of an array?
52      *
53      * @return true if this is a rest parameter
54      */
55     boolean isRestParameter();
56 
57     /**
58      * Is this super identifier?
59      *
60      * @return true if this is super identifier
61      */
62     boolean isSuper();
63 
64     /**
65      * Is this 'this' identifier?
66      *
67      * @return true if this is 'this' identifier
68      */
69     boolean isThis();
70 
71     /**
72      * Is this "*" used in module export entry?
73      *
74      * @return true if this "*" used in module export entry?
75      */
76     boolean isStar();
77 
78     /**
79      * Is this "default" used in module export entry?
80      *
81      * @return true if this 'default' used in module export entry?
82      */
83     boolean isDefault();
84 
85     /**
86      * Is this "*default*" used in module export entry?
87      *
88      * @return true if this '*default*' used in module export entry?
89      */
90     boolean isStarDefaultStar();
91 }
