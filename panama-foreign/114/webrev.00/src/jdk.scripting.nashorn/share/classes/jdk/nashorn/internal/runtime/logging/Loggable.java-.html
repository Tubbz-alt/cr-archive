25 package jdk.nashorn.internal.runtime.logging;
26 
27 import jdk.nashorn.internal.runtime.Context;
28 
29 /**
30  * Interface implemented by classes that are loggable.
31  * Their instances will provide functionality for initializing
32  * a logger (usually by asking Global for it, with a reference
33  * to this.getClass()) and a method to return the logger in
34  * use
35  *
36  * Typically a class implementing this interface also has the
37  * Logger annotation
38  *
39  * @see Logger
40  */
41 public interface Loggable {
42     /**
43      * Initialize a logger, by asking Context to get or create it
44      * and then keep it in a table by name
45      *
46      * @param context context
47      * @return the initialized logger
48      */
49     public DebugLogger initLogger(final Context context);
50 
51     /**
52      * Return the logger in use
53      * @return logger
54      */
55     public DebugLogger getLogger();
56 }
