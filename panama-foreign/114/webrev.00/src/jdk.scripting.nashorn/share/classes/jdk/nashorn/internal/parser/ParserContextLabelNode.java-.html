26 package jdk.nashorn.internal.parser;
27 
28 /**
29  * ParserContextNode that represents a LabelNode
30  */
31 class ParserContextLabelNode extends ParserContextBaseNode {
32 
33     /** Name for label */
34     private final String name;
35 
36     /**
37      * Constructor
38      *
39      * @param name The name of the label
40      */
41     public ParserContextLabelNode(final String name) {
42         this.name = name;
43     }
44 
45     /**
46      * Returns the name of the label
47      * @return name of label
48      */
49     public String getLabelName() {
50         return name;
51     }
52 }
