26 package jdk.nashorn.internal.runtime.arrays;
27 
28 import java.util.NoSuchElementException;
29 import jdk.nashorn.api.scripting.JSObject;
30 import jdk.nashorn.internal.runtime.JSType;
31 
32 /**
33  * Iterator over a ScriptObjectMirror
34  */
35 class JSObjectIterator extends ArrayLikeIterator<Object> {
36 
37     protected final JSObject obj;
38     private final long length;
39 
40     JSObjectIterator(final JSObject obj, final boolean includeUndefined) {
41         super(includeUndefined);
42         this.obj    = obj;
43         this.length = JSType.toUint32(obj.hasMember("length")? obj.getMember("length") : 0);
44         this.index  = 0;
45     }
46 
47     protected boolean indexInArray() {
48         return index < length;
49     }
50 
51     @Override
52     public long getLength() {
53         return length;
54     }
55 
56     @Override
57     public boolean hasNext() {
58         if (length == 0L) {
59             return false; //return empty string if toUint32(length) == 0
60         }
61 
62         while (indexInArray()) {
63             if (obj.hasSlot((int)index) || includeUndefined) {
64                 break;
65             }
66             bumpIndex();
67         }
68 
69         return indexInArray();
70     }
71 
72     @Override
73     public Object next() {
74         if (indexInArray()) {
75             return obj.getSlot((int)bumpIndex());
76         }
77 
78         throw new NoSuchElementException();
79     }
80 }
