20 package jdk.nashorn.internal.runtime.regexp.joni.ast;
21 
22 import jdk.nashorn.internal.runtime.regexp.joni.ScanEnvironment;
23 
24 @SuppressWarnings("javadoc")
25 public final class BackRefNode extends StateNode {
26     public final int backRef;
27 
28     public BackRefNode(final int backRef, final ScanEnvironment env) {
29         this.backRef = backRef;
30 
31         if (backRef <= env.numMem && env.memNodes[backRef] == null) {
32             setRecursion(); /* /...(\1).../ */
33         }
34     }
35 
36     @Override
37     public int getType() {
38         return BREF;
39     }
40 
41     @Override
42     public String getName() {
43         return "Back Ref";
44     }
45 
46     @Override
47     public String toString(final int level) {
48         final StringBuilder value = new StringBuilder(super.toString(level));
49         value.append("\n  back: ").append(backRef);
50         return value.toString();
51     }
52 }
