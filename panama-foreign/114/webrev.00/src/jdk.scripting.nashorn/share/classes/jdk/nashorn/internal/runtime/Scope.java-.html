 26 package jdk.nashorn.internal.runtime;
 27 
 28 import static jdk.nashorn.internal.codegen.CompilerConstants.virtualCallNoLookup;
 29 
 30 import java.util.concurrent.atomic.LongAdder;
 31 import jdk.nashorn.internal.codegen.CompilerConstants;
 32 
 33 /**
 34  * A {@link ScriptObject} subclass for objects that act as scope.
 35  */
 36 public class Scope extends ScriptObject {
 37 
 38     /* This is used to store return state of split functions. */
 39     private int splitState = -1;
 40 
 41     /** This is updated only in debug mode - counts number of {@code ScriptObject} instances created that are scope */
 42     private static final LongAdder count = Context.DEBUG ? new LongAdder() : null;
 43 
 44     /** Method handle that points to {@link Scope#getSplitState}. */
 45     public static final CompilerConstants.Call GET_SPLIT_STATE = virtualCallNoLookup(Scope.class, "getSplitState", int.class);
 46     /** Method handle that points to {@link Scope#setSplitState(int)}. */
 47     public static final CompilerConstants.Call SET_SPLIT_STATE = virtualCallNoLookup(Scope.class, "setSplitState", void.class, int.class);
 48 
 49     /**
 50      * Constructor
 51      *
 52      * @param map initial property map
 53      */
 54     public Scope(final PropertyMap map) {
 55         super(map);
 56         incrementCount();
 57     }
 58 
 59     /**
 60      * Constructor
 61      *
 62      * @param proto parent scope
 63      * @param map   initial property map
 64      */
 65     public Scope(final ScriptObject proto, final PropertyMap map) {
 66         super(proto, map);
 67         incrementCount();
 68     }
 69 
 70     /**
 71      * Constructor
 72      *
 73      * @param map            property map
 74      * @param primitiveSpill primitive spill array
 75      * @param objectSpill    reference spill array
 76      */
 77     public Scope(final PropertyMap map, final long[] primitiveSpill, final Object[] objectSpill) {
 78         super(map, primitiveSpill, objectSpill);
 79         incrementCount();
 80     }
 81 
 82     @Override
 83     public boolean isScope() {
 84         return true;
 85     }
 86 
 87     @Override
 88     boolean hasWithScope() {
 89         for (ScriptObject obj = this; obj != null; obj = obj.getProto()) {
 90             if (obj instanceof WithObject) {
 91                 return true;
 92             }
 93         }
 94         return false;
 95     }
 96 
 97     /**
 98      * Get the scope's split method state.
 99      *
100      * @return current split state
101      */
102     public int getSplitState() {
103         return splitState;
104     }
105 
106     /**
107      * Set the scope's split method state.
108      *
109      * @param state current split state
110      */
111     public void setSplitState(final int state) {
112         splitState = state;
113     }
114 
115     /**
116      * Get number of {@code Scope} instances created. If not running in debug
117      * mode this is always 0.
118      *
119      * @return number of scope ScriptObjects created
120      */
121     public static long getScopeCount() {
122         return count != null ? count.sum() : 0;
123     }
124 
125     private static void incrementCount() {
126         if (Context.DEBUG) {
127             count.increment();
128         }
129     }
130 }
