26 package jdk.nashorn.internal.ir;
27 
28 /**
29  * Can a node be an assignment under certain circumstances?
30  * Then it should implement this interface
31  *
32  * @param <D> the destination type
33  */
34 public interface Assignment<D extends Expression> {
35 
36     /**
37      * Get assignment destination
38      *
39      * @return get the assignment destination node
40      */
41     public D getAssignmentDest();
42 
43     /**
44      * Get the assignment source
45      *
46      * @return get the assignment source node
47      */
48     public Expression getAssignmentSource();
49 
50     /**
51      * Set assignment destination node.
52      * @param n the assignment destination node.
53      * @return a node equivalent to this one except for the requested change.
54      */
55     public Node setAssignmentDest(D n);
56 }
