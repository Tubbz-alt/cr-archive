26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.ir.UnaryNode;
29 
30 class UnaryTreeImpl extends ExpressionTreeImpl implements UnaryTree {
31     private final ExpressionTree expr;
32     private final Kind kind;
33     UnaryTreeImpl(final UnaryNode node, final ExpressionTree expr) {
34         super(node);
35         this.expr = expr;
36         this.kind = getOperator(node.tokenType());
37     }
38 
39     @Override
40     public Kind getKind() {
41         return kind;
42     }
43 
44     @Override
45     public ExpressionTree getExpression() {
46         return expr;
47     }
48 
49     @Override
50     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
51         return visitor.visitUnary(this, data);
52     }
53 }
