20 package jdk.nashorn.internal.runtime.regexp.joni.constants;
21 
22 @SuppressWarnings("javadoc")
23 public interface AnchorType {
24     final int BEGIN_BUF         = (1<<0);
25     final int BEGIN_LINE        = (1<<1);
26     final int BEGIN_POSITION    = (1<<2);
27     final int END_BUF           = (1<<3);
28     final int SEMI_END_BUF      = (1<<4);
29     final int END_LINE          = (1<<5);
30 
31     final int WORD_BOUND        = (1<<6);
32     final int NOT_WORD_BOUND    = (1<<7);
33     final int WORD_BEGIN        = (1<<8);
34     final int WORD_END          = (1<<9);
35     final int PREC_READ         = (1<<10);
36     final int PREC_READ_NOT     = (1<<11);
37     final int LOOK_BEHIND       = (1<<12);
38     final int LOOK_BEHIND_NOT   = (1<<13);
39 
40     final int ANYCHAR_STAR      = (1<<14);   /* ".*" optimize info */
41     final int ANYCHAR_STAR_ML   = (1<<15);   /* ".*" optimize info (multi-line) */
42 
43     final int ANYCHAR_STAR_MASK = (ANYCHAR_STAR | ANYCHAR_STAR_ML);
44     final int END_BUF_MASK      = (END_BUF | SEMI_END_BUF);
45 
46     final int ALLOWED_IN_LB =     ( LOOK_BEHIND |
47                                     BEGIN_LINE |
48                                     END_LINE |
49                                     BEGIN_BUF |
50                                     BEGIN_POSITION );
51 
52     final int ALLOWED_IN_LB_NOT = ( LOOK_BEHIND |
53                                     LOOK_BEHIND_NOT |
54                                     BEGIN_LINE |
55                                     END_LINE |
56                                     BEGIN_BUF |
57                                     BEGIN_POSITION );
58 
59 }
