26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.ir.LabelNode;
29 
30 final class LabeledStatementTreeImpl extends StatementTreeImpl
31     implements LabeledStatementTree {
32     private final String name;
33     private final StatementTree stat;
34 
35     LabeledStatementTreeImpl(final LabelNode node, final StatementTree stat) {
36         super(node);
37         this.name = node.getLabelName();
38         this.stat = stat;
39     }
40 
41     @Override
42     public Kind getKind() {
43         return Kind.LABELED_STATEMENT;
44     }
45 
46     @Override
47     public String getLabel() {
48         return name;
49     }
50 
51     @Override
52     public StatementTree getStatement() {
53         return stat;
54     }
55 
56     @Override
57     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
58         return visitor.visitLabeledStatement(this, data);
59     }
60 }
