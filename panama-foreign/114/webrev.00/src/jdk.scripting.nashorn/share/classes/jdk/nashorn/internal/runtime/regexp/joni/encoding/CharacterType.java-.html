20 package jdk.nashorn.internal.runtime.regexp.joni.encoding;
21 
22 @SuppressWarnings("javadoc")
23 public interface CharacterType {
24 
25     final int NEWLINE   = 0;
26     final int ALPHA     = 1;
27     final int BLANK     = 2;
28     final int CNTRL     = 3;
29     final int DIGIT     = 4;
30     final int GRAPH     = 5;
31     final int LOWER     = 6;
32     final int PRINT     = 7;
33     final int PUNCT     = 8;
34     final int SPACE     = 9;
35     final int UPPER     = 10;
36     final int XDIGIT    = 11;
37     final int WORD      = 12;
38     final int ALNUM     = 13;      /* alpha || digit */
39     final int ASCII     = 14;
40 
41     final int SPECIAL_MASK = 256;
42     final int S = SPECIAL_MASK | SPACE;
43     final int D = SPECIAL_MASK | DIGIT;
44     final int W = SPECIAL_MASK | WORD;
45 
46     final int LETTER_MASK = (1 << Character.UPPERCASE_LETTER)
47                           | (1 << Character.LOWERCASE_LETTER)
48                           | (1 << Character.TITLECASE_LETTER)
49                           | (1 << Character.MODIFIER_LETTER)
50                           | (1 << Character.OTHER_LETTER);
51     final int ALPHA_MASK = LETTER_MASK
52                           | (1 << Character.COMBINING_SPACING_MARK)
53                           | (1 << Character.NON_SPACING_MARK)
54                           | (1 << Character.ENCLOSING_MARK);
55     final int ALNUM_MASK = ALPHA_MASK
56                           | (1 << Character.DECIMAL_DIGIT_NUMBER);
57     final int WORD_MASK = ALNUM_MASK
58                           | (1 << Character.CONNECTOR_PUNCTUATION);
59     final int PUNCT_MASK =  (1 << Character.CONNECTOR_PUNCTUATION)
60                           | (1 << Character.DASH_PUNCTUATION)
61                           | (1 << Character.END_PUNCTUATION)
62                           | (1 << Character.FINAL_QUOTE_PUNCTUATION)
63                           | (1 << Character.INITIAL_QUOTE_PUNCTUATION)
64                           | (1 << Character.OTHER_PUNCTUATION)
65                           | (1 << Character.START_PUNCTUATION);
66     final int CNTRL_MASK =  (1 << Character.CONTROL)
67                           | (1 << Character.FORMAT)
68                           | (1 << Character.PRIVATE_USE)
69                           | (1 << Character.SURROGATE);
70     final int SPACE_MASK =  (1 << Character.SPACE_SEPARATOR)
71                           | (1 << Character.LINE_SEPARATOR)        // 0x2028
72                           | (1 << Character.PARAGRAPH_SEPARATOR);  // 0x2029
73     final int GRAPH_MASK = SPACE_MASK
74                           | (1 << Character.CONTROL)
75                           | (1 << Character.SURROGATE);
76     final int PRINT_MASK =  (1 << Character.CONTROL)
77                           | (1 << Character.SURROGATE);
78 
79 
80 }
