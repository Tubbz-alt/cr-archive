26 package jdk.nashorn.internal.objects.annotations;
27 
28 /**
29  * Attributes for JavaScript properties. The negative logic "NOT_xxx" is because the
30  * common case is to be writable, enumerable and configurable
31  */
32 
33 public interface Attribute {
34     /** Flag for non-writable properties */
35     public static final int NOT_WRITABLE     = jdk.nashorn.internal.runtime.Property.NOT_WRITABLE;
36 
37     /** Flag for non-enumerable properties */
38     public static final int NOT_ENUMERABLE   = jdk.nashorn.internal.runtime.Property.NOT_ENUMERABLE;
39 
40     /** Flag for non-configurable properties */
41     public static final int NOT_CONFIGURABLE = jdk.nashorn.internal.runtime.Property.NOT_CONFIGURABLE;
42 
43     /**
44      * Flag for accessor (getter/setter) properties as opposed to data properties.
45      *
46      * <p>This allows nasgen-created properties to behave like user-accessors. it should only be used for
47      * properties that are explicitly specified as accessor properties in the ECMAScript specification
48      * such as Map.prototype.size in ES6, not value properties that happen to be implemented by getter/setter
49      * such as the "length" properties of String or Array objects.</p>
50      */
51     public static final int IS_ACCESSOR = jdk.nashorn.internal.runtime.Property.IS_ACCESSOR_PROPERTY;
52 
53     /** Read-only, non-configurable property */
54     public static final int CONSTANT = NOT_WRITABLE | NOT_CONFIGURABLE;
55 
56     /** Non-enumerable, read-only, non-configurable property */
57     public static final int NON_ENUMERABLE_CONSTANT = NOT_ENUMERABLE | CONSTANT;
58 
59     /** By default properties are writable, enumerable and configurable */
60     public static final int DEFAULT_ATTRIBUTES = 0;
61 }
