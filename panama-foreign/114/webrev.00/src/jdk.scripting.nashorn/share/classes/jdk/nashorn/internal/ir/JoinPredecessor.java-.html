26 package jdk.nashorn.internal.ir;
27 
28 /**
29  * Interface implemented by AST nodes that either can occur as predecessors of a control flow join, or contain a control
30  * flow join themselves. JoinPredecessor only provides a getter and setter for a {@link LocalVariableConversion}; the
31  * semantics of control flow for a particular node implementing the interface are shared between
32  * {@code LocalVariableTypesCalculator} that creates the conversions, and {@code CodeGenerator} that uses them.
33  */
34 public interface JoinPredecessor {
35     /**
36      * Set the local variable conversions needed to unify their types at a control flow join point.
37      * @param lc the current lexical context
38      * @param conversion the conversions.
39      * @return this node or a different node representing the change.
40      */
41     public JoinPredecessor setLocalVariableConversion(LexicalContext lc, LocalVariableConversion conversion);
42 
43     /**
44      * Returns the local variable conversions needed to unify their types at a control flow join point.
45      * @return the local variable conversions needed to unify their types at a control flow join point. Can be null.
46      * Can contain {@link LocalVariableConversion#isLive() dead conversions}.
47      */
48     public LocalVariableConversion getLocalVariableConversion();
49 }
