26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.runtime.Source;
29 
30 final class LineMapImpl implements LineMap {
31     private final Source source;
32 
33     LineMapImpl(final Source source) {
34         this.source = source;
35     }
36 
37     @Override
38     public long getLineNumber(final long pos) {
39         return source.getLine((int)pos);
40     }
41 
42     @Override
43     public long getColumnNumber(final long pos) {
44         return source.getColumn((int)pos);
45     }
46 }
