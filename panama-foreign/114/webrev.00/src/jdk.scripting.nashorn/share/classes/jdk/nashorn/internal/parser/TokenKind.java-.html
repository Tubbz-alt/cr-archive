26 package jdk.nashorn.internal.parser;
27 
28 /**
29  * Classification of token types.
30  */
31 public enum TokenKind {
32     /** Error, EOF, EOL...*/
33     SPECIAL,
34     /** Unary operators. */
35     UNARY,
36     /** Binary operators. */
37     BINARY,
38     /** [] () {} */
39     BRACKET,
40     /** String recognized as a keyword. */
41     KEYWORD,
42     /** Literal constant. */
43     LITERAL,
44     /** IR only token. */
45     IR,
46     /** Token reserved for future usage. */
47     FUTURE,
48     /** Token reserved for future in strict mode. */
49     FUTURESTRICT
50 }
