25 package jdk.nashorn.api.tree;
26 
27 import jdk.nashorn.internal.ir.UnaryNode;
28 import jdk.nashorn.internal.parser.TokenType;
29 
30 final class NewTreeImpl extends ExpressionTreeImpl implements NewTree {
31     private final ExpressionTree constrExpr;
32 
33     NewTreeImpl(final UnaryNode node, final ExpressionTree constrExpr) {
34         super(node);
35         assert (node.isTokenType(TokenType.NEW)) : "new expected";
36         this.constrExpr = constrExpr;
37     }
38 
39     @Override
40     public Kind getKind() {
41         return Kind.NEW;
42     }
43 
44     @Override
45     public ExpressionTree getConstructorExpression() {
46         return constrExpr;
47     }
48 
49     @Override
50     public <R, D> R accept(final TreeVisitor<R, D> visitor, final D data) {
51         return visitor.visitNew(this, data);
52     }
53 }
