26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.ir.LiteralNode;
29 import jdk.nashorn.internal.parser.Lexer;
30 
31 final class RegExpLiteralTreeImpl extends ExpressionTreeImpl
32     implements RegExpLiteralTree {
33     private final String pattern;
34     private final String options;
35     RegExpLiteralTreeImpl(final LiteralNode<?> node) {
36         super(node);
37         assert node.getValue() instanceof Lexer.RegexToken : "regexp expected";
38         final Lexer.RegexToken regex = (Lexer.RegexToken) node.getValue();
39         this.pattern = regex.getExpression();
40         this.options = regex.getOptions();
41     }
42 
43     @Override
44     public Kind getKind() {
45         return Kind.REGEXP_LITERAL;
46     }
47 
48     @Override
49     public String getPattern() {
50         return pattern;
51     }
52 
53     @Override
54     public String getOptions() {
55         return options;
56     }
57 
58     @Override
59     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
60         return visitor.visitRegExpLiteral(this, data);
61     }
62 }
