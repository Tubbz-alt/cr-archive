26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.ir.ExpressionStatement;
29 import jdk.nashorn.internal.parser.TokenType;
30 
31 // This implementation of VariableTree represents a destructuring declaration
32 final class DestructuringDeclTreeImpl extends StatementTreeImpl
33         implements VariableTree {
34 
35     private final TokenType declType;
36     private final ExpressionTree lhs;
37     private final ExpressionTree init;
38 
39     DestructuringDeclTreeImpl(final ExpressionStatement exprStat, final ExpressionTree lhs, final ExpressionTree init) {
40         super(exprStat);
41         assert exprStat.destructuringDeclarationType() != null : "expecting a destructuring decl. statement";
42 
43         this.declType = exprStat.destructuringDeclarationType();
44         this.lhs = lhs;
45         this.init = init;
46     }
47 
48     @Override
49     public Kind getKind() {
50         return Kind.VARIABLE;
51     }
52 
53     @Override
54     public ExpressionTree getBinding() {
55         return lhs;
56     }
57 
58     @Override
59     public ExpressionTree getInitializer() {
60         return init;
61     }
62 
63     @Override
64     public boolean isConst() {
65         return declType == TokenType.CONST;
66     }
67 
68     @Override
69     public boolean isLet() {
70         return declType == TokenType.LET;
71     }
72 
73     @Override
74     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
75         return visitor.visitVariable(this, data);
76     }
77 }
