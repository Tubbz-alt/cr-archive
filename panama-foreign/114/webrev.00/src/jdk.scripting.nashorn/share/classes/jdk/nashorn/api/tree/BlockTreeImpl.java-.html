26 package jdk.nashorn.api.tree;
27 
28 import java.util.List;
29 import jdk.nashorn.internal.ir.Block;
30 import jdk.nashorn.internal.ir.BlockStatement;
31 
32 final class BlockTreeImpl extends StatementTreeImpl implements BlockTree {
33     private final List<? extends StatementTree> statements;
34 
35     BlockTreeImpl(final BlockStatement node, final List<? extends StatementTree> statements) {
36         super(node);
37         this.statements = statements;
38     }
39 
40     BlockTreeImpl(final Block node, final List<? extends StatementTree> statements) {
41         super(node);
42         this.statements = statements;
43     }
44 
45     @Override
46     public Kind getKind() {
47         return Kind.BLOCK;
48     }
49 
50     @Override
51     public List<? extends StatementTree> getStatements() {
52         return statements;
53     }
54 
55     @Override
56     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
57         return visitor.visitBlock(this, data);
58     }
59 }
