26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.ir.AccessNode;
29 
30 final class MemberSelectTreeImpl extends ExpressionTreeImpl
31     implements MemberSelectTree {
32     private final String ident;
33     private final ExpressionTree expr;
34     MemberSelectTreeImpl(final AccessNode node, final ExpressionTree expr) {
35         super(node);
36         this.ident = node.getProperty();
37         this.expr = expr;
38     }
39 
40     @Override
41     public Kind getKind() {
42         return Kind.MEMBER_SELECT;
43     }
44 
45     @Override
46     public ExpressionTree getExpression() {
47         return expr;
48     }
49 
50     @Override
51     public String getIdentifier() {
52         return ident;
53     }
54 
55     @Override
56     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
57         return visitor.visitMemberSelect(this, data);
58     }
59 }
