26 package jdk.nashorn.internal.runtime.arrays;
27 
28 import java.util.List;
29 
30 /**
31   * Iterator over a Java List.
32  */
33 class JavaListIterator extends ArrayLikeIterator<Object> {
34 
35     /** {@link java.util.List} to iterate over */
36     protected final List<?> list;
37 
38     /** length of array */
39     protected final long length;
40 
41     /**
42      * Constructor
43      * @param list list to iterate over
44      * @param includeUndefined should undefined elements be included in iteration
45      */
46     protected JavaListIterator(final List<?> list, final boolean includeUndefined) {
47         super(includeUndefined);
48         this.list = list;
49         this.length = list.size();
50     }
51 
52     /**
53      * Is the current index still inside the array
54      * @return true if inside the array
55      */
56     protected boolean indexInArray() {
57         return index < length;
58     }
59 
60     @Override
61     public Object next() {
62         return list.get((int)bumpIndex());
63     }
64 
65     @Override
66     public long getLength() {
67         return length;
68     }
69 
70     @Override
71     public boolean hasNext() {
72         return indexInArray();
73     }
74 
75     @Override
76     public void remove() {
77         list.remove(index);
78     }
79 }
