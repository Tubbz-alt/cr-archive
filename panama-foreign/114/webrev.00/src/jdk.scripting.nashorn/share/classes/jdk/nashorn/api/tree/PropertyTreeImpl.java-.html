26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.ir.PropertyNode;
29 
30 final class PropertyTreeImpl extends TreeImpl implements PropertyTree  {
31     private final ExpressionTree key;
32     private final ExpressionTree value;
33     private final FunctionExpressionTree getter;
34     private final FunctionExpressionTree setter;
35     private final boolean isStatic, isComputed;
36 
37     PropertyTreeImpl(final PropertyNode node,
38             final ExpressionTree key,
39             final ExpressionTree value,
40             final FunctionExpressionTree getter,
41             final FunctionExpressionTree setter) {
42         super(node);
43         this.key    = key;
44         this.value  = value;
45         this.getter = getter;
46         this.setter = setter;
47         this.isStatic = node.isStatic();
48         this.isComputed = node.isComputed();
49     }
50 
51     @Override
52     public Kind getKind() {
53         return Kind.PROPERTY;
54     }
55 
56     @Override
57     public ExpressionTree getKey() {
58         return key;
59     }
60 
61     @Override
62     public ExpressionTree getValue() {
63         return value;
64     }
65 
66     @Override
67     public FunctionExpressionTree getGetter() {
68         return getter;
69     }
70 
71     @Override
72     public FunctionExpressionTree getSetter() {
73         return setter;
74     }
75 
76     @Override
77     public boolean isStatic() {
78         return isStatic;
79     }
80 
81     @Override
82     public boolean isComputed() {
83         return isComputed;
84     }
85 
86     @Override
87     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
88         return visitor.visitProperty(this, data);
89     }
90 }
