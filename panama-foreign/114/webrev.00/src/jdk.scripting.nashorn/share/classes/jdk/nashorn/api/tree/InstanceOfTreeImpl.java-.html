26 package jdk.nashorn.api.tree;
27 
28 import jdk.nashorn.internal.ir.BinaryNode;
29 import jdk.nashorn.internal.parser.TokenType;
30 
31 final class InstanceOfTreeImpl extends BinaryTreeImpl implements InstanceOfTree {
32     InstanceOfTreeImpl(final BinaryNode node,
33             final ExpressionTree expr,
34             final ExpressionTree type) {
35         super(node, expr, type);
36         assert node.isTokenType(TokenType.INSTANCEOF) : "instanceof expected";
37     }
38 
39     @Override
40     public Kind getKind() {
41         return Kind.INSTANCE_OF;
42     }
43 
44     @Override
45     public ExpressionTree getExpression() {
46         return getLeftOperand();
47     }
48 
49     @Override
50     public Tree getType() {
51         return getRightOperand();
52     }
53 
54     @Override
55     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
56         return visitor.visitInstanceOf(this, data);
57     }
58 }
