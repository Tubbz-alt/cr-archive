26 package jdk.nashorn.api.tree;
27 
28 import java.util.List;
29 import jdk.nashorn.internal.ir.FunctionNode;
30 
31 final class CompilationUnitTreeImpl extends TreeImpl
32     implements CompilationUnitTree {
33     private final FunctionNode funcNode;
34     private final List<? extends Tree> elements;
35     private final ModuleTree module;
36 
37     CompilationUnitTreeImpl(final FunctionNode node,
38             final List<? extends Tree> elements,
39             final ModuleTree module) {
40         super(node);
41         this.funcNode = node;
42         assert funcNode.getKind() == FunctionNode.Kind.SCRIPT ||
43                 funcNode.getKind() == FunctionNode.Kind.MODULE :
44                 "script or module function expected";
45         this.elements = elements;
46         this.module = module;
47     }
48 
49     @Override
50     public Tree.Kind getKind() {
51         return Tree.Kind.COMPILATION_UNIT;
52     }
53 
54     @Override
55     public List<? extends Tree> getSourceElements() {
56         return elements;
57     }
58 
59     @Override
60     public String getSourceName() {
61         return funcNode.getSourceName();
62     }
63 
64     @Override
65     public boolean isStrict() {
66         return funcNode.isStrict();
67     }
68 
69     @Override
70     public LineMap getLineMap() {
71         return new LineMapImpl(funcNode.getSource());
72     }
73 
74     @Override
75     public ModuleTree getModule() {
76         return module;
77     }
78 
79     @Override
80     public <R,D> R accept(final TreeVisitor<R,D> visitor, final D data) {
81         return visitor.visitCompilationUnit(this, data);
82     }
83 }
