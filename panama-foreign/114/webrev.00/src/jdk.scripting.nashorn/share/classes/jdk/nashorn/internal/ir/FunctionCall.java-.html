26 package jdk.nashorn.internal.ir;
27 
28 import jdk.nashorn.internal.runtime.linker.NashornCallSiteDescriptor;
29 
30 /**
31  * Interface used by AccessNodes, IndexNodes and IdentNodes to signal that when evaluated, their value will be treated
32  * as a function and immediately invoked, e.g. {@code foo()}, {@code foo.bar()} or {@code foo[bar]()}. Used to customize
33  * the priority of composite dynamic operations when emitting {@code INVOKEDYNAMIC} instructions that implement them,
34  * namely prioritize {@code getMethod} over {@code getElem} or {@code getProp}. An access or ident node with isFunction
35  * set to true will be emitted as {@link NashornCallSiteDescriptor#GET_METHOD_PROPERTY} while one with it set to false will be emitted
36  * as {@link NashornCallSiteDescriptor#GET_PROPERTY}. Similarly, an index node with isFunction set to true will be emitted as
37  * {@link NashornCallSiteDescriptor#GET_METHOD_ELEMENT} while the one set to false will be emitted as {@link NashornCallSiteDescriptor#GET_ELEMENT}.
38  */
39 public interface FunctionCall {
40     /**
41      * Returns true if the value of this expression will be treated as a function and immediately invoked.
42      * @return true if the value of this expression will be treated as a function and immediately invoked.
43      */
44     public boolean isFunction();
45 }
