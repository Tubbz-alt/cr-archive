 26 package sun.jvm.hotspot.runtime.linux_aarch64;
 27 
 28 import java.io.*;
 29 import java.util.*;
 30 import sun.jvm.hotspot.debugger.*;
 31 import sun.jvm.hotspot.debugger.aarch64.*;
 32 import sun.jvm.hotspot.runtime.*;
 33 import sun.jvm.hotspot.runtime.aarch64.*;
 34 import sun.jvm.hotspot.types.*;
 35 import sun.jvm.hotspot.utilities.*;
 36 import sun.jvm.hotspot.utilities.Observable;
 37 import sun.jvm.hotspot.utilities.Observer;
 38 
 39 public class LinuxAARCH64JavaThreadPDAccess implements JavaThreadPDAccess {
 40   private static AddressField  lastJavaFPField;
 41   private static AddressField  osThreadField;
 42 
 43   // Field from OSThread
 44   private static CIntegerField osThreadThreadIDField;
 45 
 46   // This is currently unneeded but is being kept in case we change
 47   // the currentFrameGuess algorithm
 48   private static final long GUESS_SCAN_RANGE = 128 * 1024;
 49 
 50   static {
 51     VM.registerVMInitializedObserver(new Observer() {
 52         public void update(Observable o, Object data) {
 53           initialize(VM.getVM().getTypeDataBase());
 54         }
 55       });
 56   }
 57 
 58   private static synchronized void initialize(TypeDataBase db) {
 59     Type type = db.lookupType("JavaThread");
 60     osThreadField           = type.getAddressField("_osthread");
 61 
 62     Type anchorType = db.lookupType("JavaFrameAnchor");
 63     lastJavaFPField         = anchorType.getAddressField("_last_Java_fp");
 64 
 65     Type osThreadType = db.lookupType("OSThread");
 66     osThreadThreadIDField   = osThreadType.getCIntegerField("_thread_id");
 67   }
 68 
 69   public Address getLastJavaFP(Address addr) {
 70     return lastJavaFPField.getValue(addr.addOffsetTo(sun.jvm.hotspot.runtime.JavaThread.getAnchorField().getOffset()));
 71   }
 72 
 73   public Address getLastJavaPC(Address addr) {
 74     return null;
 75   }
 76 
 77   public Address getBaseOfStackPointer(Address addr) {
 78     return null;
 79   }
 80 
 81   public Frame getLastFramePD(JavaThread thread, Address addr) {
 82     Address fp = thread.getLastJavaFP();
 83     if (fp == null) {
 84       return null; // no information
 85     }
 86     return new AARCH64Frame(thread.getLastJavaSP(), fp);
 87   }
 88 
 89   public RegisterMap newRegisterMap(JavaThread thread, boolean updateMap) {
 90     return new AARCH64RegisterMap(thread, updateMap);
 91   }
 92 
 93   public Frame getCurrentFrameGuess(JavaThread thread, Address addr) {
 94     ThreadProxy t = getThreadProxy(addr);
 95     AARCH64ThreadContext context = (AARCH64ThreadContext) t.getContext();
 96     AARCH64CurrentFrameGuess guesser = new AARCH64CurrentFrameGuess(context, thread);
 97     if (!guesser.run(GUESS_SCAN_RANGE)) {
 98       return null;
 99     }
100     if (guesser.getPC() == null) {
101       return new AARCH64Frame(guesser.getSP(), guesser.getFP());
102     } else {
103       return new AARCH64Frame(guesser.getSP(), guesser.getFP(), guesser.getPC());
104     }
105   }
106 
107   public void printThreadIDOn(Address addr, PrintStream tty) {
108     tty.print(getThreadProxy(addr));
109   }
110 
111   public void printInfoOn(Address threadAddr, PrintStream tty) {
112     tty.print("Thread id: ");
113     printThreadIDOn(threadAddr, tty);
114 //    tty.println("\nPostJavaState: " + getPostJavaState(threadAddr));
115   }
116 
117   public Address getLastSP(Address addr) {
118     ThreadProxy t = getThreadProxy(addr);
119     AARCH64ThreadContext context = (AARCH64ThreadContext) t.getContext();
120     return context.getRegisterAsAddress(AARCH64ThreadContext.SP);
121   }
122 
123   public ThreadProxy getThreadProxy(Address addr) {
124     // Addr is the address of the JavaThread.
125     // Fetch the OSThread (for now and for simplicity, not making a
126     // separate "OSThread" class in this package)
127     Address osThreadAddr = osThreadField.getValue(addr);
128     // Get the address of the _thread_id from the OSThread
129     Address threadIdAddr = osThreadAddr.addOffsetTo(osThreadThreadIDField.getOffset());
130 
131     JVMDebugger debugger = VM.getVM().getDebugger();
132     return debugger.getThreadForIdentifierAddress(threadIdAddr);
133   }
134 }
