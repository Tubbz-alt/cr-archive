25 package sun.jvm.hotspot.utilities;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 import sun.jvm.hotspot.types.*;
30 import sun.jvm.hotspot.oops.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.utilities.Observable;
33 import sun.jvm.hotspot.utilities.Observer;
34 
35 public class Hashtable extends BasicHashtable {
36   static {
37     VM.registerVMInitializedObserver(new Observer() {
38         public void update(Observable o, Object data) {
39           initialize(VM.getVM().getTypeDataBase());
40         }
41       });
42   }
43 
44   private static synchronized void initialize(TypeDataBase db) {
45     // just to confirm that type exists
46     Type type = db.lookupType("IntptrHashtable");
47   }
48 
49   protected Class<? extends HashtableEntry> getHashtableEntryClass() {
50     return HashtableEntry.class;
51   }
52 
53   public long computeHash(Symbol name) {
54     return name.identityHash();
55   }
56 
57   public int hashToIndex(long fullHash) {
58     return (int) (fullHash % tableSize());
59   }
60 
61   public Hashtable(Address addr) {
62     super(addr);
63   }
64 
65   // VM's Hashtable::hash_symbol
66   protected static long hashSymbol(byte[] buf) {
67     long h = 0;
68     int s = 0;
69     int len = buf.length;
70     while (len-- > 0) {
71       h = 31*h + (0xFFL & buf[s]);
72       s++;
73     }
74     return h & 0xFFFFFFFFL;
75   }
76 }
