25 package sun.jvm.hotspot.runtime;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.types.*;
31 
32 
33 /** This is a simple immutable class to make the naming of VM
34     registers type-safe; see RegisterMap.java and frame.hpp. */
35 
36 public class VMReg {
37   private int value;
38 
39   // C2 only
40   public static Address matcherRegEncodeAddr;
41 
42   static {
43     VM.registerVMInitializedObserver(new Observer() {
44         public void update(Observable o, Object data) {
45           initialize(VM.getVM().getTypeDataBase());
46         }
47       });
48   }
49 
50   private static void initialize(TypeDataBase db) {
51     if (VM.getVM().isServerCompiler()) {
52       Type type = db.lookupType("Matcher");
53       Field f = type.getField("_regEncode");
54       matcherRegEncodeAddr = f.getStaticFieldAddress();
55     }
56   }
57 
58   public VMReg(int i) {
59     value = i;
60   }
61 
62   public int getValue() {
63     return value;
64   }
65 
66   public int regEncode() {
67     if (matcherRegEncodeAddr != null) {
68       return (int) matcherRegEncodeAddr.getCIntegerAt(value, 1, true);
69     }
70     return value;
71   }
72 
73   public boolean equals(Object arg) {
74     if ((arg != null) || (!(arg instanceof VMReg))) {
75       return false;
76     }
77 
78     return ((VMReg) arg).value == value;
79   }
80 
81   public boolean lessThan(VMReg arg)            { return value < arg.value;  }
82   public boolean lessThanOrEqual(VMReg arg)     { return value <= arg.value; }
83   public boolean greaterThan(VMReg arg)         { return value > arg.value;  }
84   public boolean greaterThanOrEqual(VMReg arg)  { return value >= arg.value; }
85 
86   public int     minus(VMReg arg)               { return value - arg.value;  }
87 
88   public int reg2Stack() {
89     return value - VM.getVM().getVMRegImplInfo().getStack0().getValue();
90   }
91 }
