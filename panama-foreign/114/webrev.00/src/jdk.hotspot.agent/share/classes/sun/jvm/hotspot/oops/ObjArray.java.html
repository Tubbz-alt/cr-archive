25 package sun.jvm.hotspot.oops;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.Observable;
33 import sun.jvm.hotspot.utilities.Observer;
34 
35 // An ObjArray is an array containing oops
36 
37 public class ObjArray extends Array {
38   static {
39     VM.registerVMInitializedObserver(new Observer() {
40         public void update(Observable o, Object data) {
41           initialize(VM.getVM().getTypeDataBase());
42         }
43       });
44   }
45 
46   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
47     Type type   = db.lookupType("objArrayOopDesc");
48     elementSize = VM.getVM().getHeapOopSize();
49   }
50 
51   ObjArray(OopHandle handle, ObjectHeap heap) {
52     super(handle, heap);
53   }
54 
55   public boolean isObjArray()          { return true; }
56 
57   private static long elementSize;
58 
59   public OopHandle getOopHandleAt(long index) {
60     long offset = baseOffsetInBytes(BasicType.T_OBJECT) + (index * elementSize);
61     if (VM.getVM().isCompressedOopsEnabled()) {
62       return getHandle().getCompOopHandleAt(offset);
63     } else {
64       return getHandle().getOopHandleAt(offset);
65     }
66   }
67 
68   public Oop getObjAt(long index) {
69       return getHeap().newOop(getOopHandleAt(index));
70   }
71 
72   public void printValueOn(PrintStream tty) {
73     tty.print("ObjArray");
74   }
75 
76   public void iterateFields(OopVisitor visitor, boolean doVMFields) {
77     super.iterateFields(visitor, doVMFields);
78     int length = (int) getLength();
79     long baseOffset = baseOffsetInBytes(BasicType.T_OBJECT);
80     for (int index = 0; index < length; index++) {
81       long offset = baseOffset + (index * elementSize);
82       OopField field;
83       if (VM.getVM().isCompressedOopsEnabled()) {
84         field = new NarrowOopField(new IndexableFieldIdentifier(index), offset, false);
85       } else {
86         field = new OopField(new IndexableFieldIdentifier(index), offset, false);
87       }
88       visitor.doOop(field, false);
89     }
90   }
91 }
