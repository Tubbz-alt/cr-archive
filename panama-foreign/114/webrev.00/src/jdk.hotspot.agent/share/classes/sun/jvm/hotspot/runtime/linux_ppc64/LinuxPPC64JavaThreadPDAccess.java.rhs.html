 25 package sun.jvm.hotspot.runtime.linux_ppc64;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 
 30 import sun.jvm.hotspot.debugger.*;
 31 import sun.jvm.hotspot.debugger.ppc64.*;
 32 import sun.jvm.hotspot.runtime.*;
 33 import sun.jvm.hotspot.runtime.ppc64.*;
 34 import sun.jvm.hotspot.types.*;
 35 import sun.jvm.hotspot.utilities.*;
 36 import sun.jvm.hotspot.utilities.Observable;
 37 import sun.jvm.hotspot.utilities.Observer;
 38 
 39 public class LinuxPPC64JavaThreadPDAccess implements JavaThreadPDAccess {
 40   private static AddressField  osThreadField;
 41 
 42   // Field from OSThread
 43   private static CIntegerField osThreadThreadIDField;
 44 
 45   // This is currently unneeded but is being kept in case we change
 46   // the currentFrameGuess algorithm
 47   private static final long GUESS_SCAN_RANGE = 128 * 1024;
 48 
 49   static {
 50     VM.registerVMInitializedObserver(new Observer() {
 51         public void update(Observable o, Object data) {
 52           initialize(VM.getVM().getTypeDataBase());
 53         }
 54       });
 55   }
 56 
 57   private static synchronized void initialize(TypeDataBase db) {
 58     Type type = db.lookupType("JavaThread");
 59     osThreadField = type.getAddressField("_osthread");
 60 
 61     Type osThreadType = db.lookupType("OSThread");
 62     osThreadThreadIDField = osThreadType.getCIntegerField("_thread_id");
 63   }
 64 
 65   public Address getLastJavaFP(Address addr) {
 66     return null;
 67   }
 68 
 69   public Address getLastJavaPC(Address addr) {
 70     return null;
 71   }
 72 
 73   public Address getBaseOfStackPointer(Address addr) {
 74     return null;
 75   }
 76 
 77   public Frame getLastFramePD(JavaThread thread, Address addr) {
 78     Address fp = thread.getLastJavaFP();
 79     if (fp == null) {
 80       return null; // no information
 81     }
 82     return new PPC64Frame(thread.getLastJavaSP(), fp);
 83   }
 84 
 85   public RegisterMap newRegisterMap(JavaThread thread, boolean updateMap) {
 86     return new PPC64RegisterMap(thread, updateMap);
 87   }
 88 
 89   public Frame getCurrentFrameGuess(JavaThread thread, Address addr) {
 90     ThreadProxy t = getThreadProxy(addr);
 91     PPC64ThreadContext context = (PPC64ThreadContext) t.getContext();
 92     PPC64CurrentFrameGuess guesser = new PPC64CurrentFrameGuess(context, thread);
 93     if (!guesser.run(GUESS_SCAN_RANGE)) {
 94       return null;
 95     }
 96     if (guesser.getPC() == null) {
 97       return new PPC64Frame(guesser.getSP(), guesser.getFP());
 98     } else {
 99       return new PPC64Frame(guesser.getSP(), guesser.getFP(), guesser.getPC());
100     }
101   }
102 
103   public void printThreadIDOn(Address addr, PrintStream tty) {
104     tty.print(getThreadProxy(addr));
105   }
106 
107   public void printInfoOn(Address threadAddr, PrintStream tty) {
108     tty.print("Thread id: ");
109     printThreadIDOn(threadAddr, tty);
110     // tty.println("\nPostJavaState: " + getPostJavaState(threadAddr));
111   }
112 
113   public Address getLastSP(Address addr) {
114     ThreadProxy t = getThreadProxy(addr);
115     PPC64ThreadContext context = (PPC64ThreadContext) t.getContext();
116     return context.getRegisterAsAddress(PPC64ThreadContext.SP);
117   }
118 
119   public Address getLastFP(Address addr) {
120     return getLastSP(addr).getAddressAt(0);
121   }
122 
123   public ThreadProxy getThreadProxy(Address addr) {
124     // Addr is the address of the JavaThread.
125     // Fetch the OSThread (for now and for simplicity, not making a
126     // separate "OSThread" class in this package)
127     Address osThreadAddr = osThreadField.getValue(addr);
128     // Get the address of the _thread_id from the OSThread
129     Address threadIdAddr = osThreadAddr.addOffsetTo(osThreadThreadIDField.getOffset());
130 
131     JVMDebugger debugger = VM.getVM().getDebugger();
132     return debugger.getThreadForIdentifierAddress(threadIdAddr);
133   }
134 }
