25 package sun.jvm.hotspot.runtime;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.types.*;


31 
32 public class BasicObjectLock extends VMObject {
33   static {
34     VM.registerVMInitializedObserver(new Observer() {
35         public void update(Observable o, Object data) {
36           initialize(VM.getVM().getTypeDataBase());
37         }
38       });
39   }
40 
41   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
42     Type type  = db.lookupType("BasicObjectLock");
43     lockField  = type.getField("_lock");
44     objField   = type.getOopField("_obj");
45     size       = (int) type.getSize();
46   }
47 
48   private static sun.jvm.hotspot.types.Field    lockField;
49   private static sun.jvm.hotspot.types.OopField objField;
50   private static int        size;
51 
52   public BasicObjectLock(Address addr) {
53     super(addr);
54   }
55 
56   public OopHandle obj()  { return objField.getValue(addr); }
57   public BasicLock lock() { return new BasicLock(addr.addOffsetTo(lockField.getOffset())); }
58 
59   /** Note: Use frame::interpreter_frame_monitor_size() for the size
60       of BasicObjectLocks in interpreter activation frames since it
61       includes machine-specific padding. This routine returns a size
62       in BYTES in this system! */
63   public static int size() { return size; }
64 
65   /** Helper routine for Frames (also probably needed for iteration) */
66   public Address address() { return addr; }
67 }
