25 package sun.jvm.hotspot.runtime;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.oops.*;
31 import sun.jvm.hotspot.types.*;


32 
33 public class BasicLock extends VMObject {
34   static {
35     VM.registerVMInitializedObserver(new Observer() {
36         public void update(Observable o, Object data) {
37           initialize(VM.getVM().getTypeDataBase());
38         }
39       });
40   }
41 
42   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
43     Type type  = db.lookupType("BasicLock");
44     displacedHeaderField = type.getCIntegerField("_displaced_header");
45   }
46 
47   private static CIntegerField displacedHeaderField;
48 
49   public BasicLock(Address addr) {
50     super(addr);
51   }
52 
53   public Mark displacedHeader() {
54     return new Mark(addr.addOffsetTo(displacedHeaderField.getOffset()));
55   }
56 }
