25 package sun.jvm.hotspot.opto;
26 
27 import java.util.*;
28 import java.io.PrintStream;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.oops.*;
32 import sun.jvm.hotspot.types.*;


33 
34 public class MachSafePointNode extends MachReturnNode {
35   static {
36     VM.registerVMInitializedObserver(new Observer() {
37         public void update(Observable o, Object data) {
38           initialize(VM.getVM().getTypeDataBase());
39         }
40       });
41   }
42 
43   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
44     Type type      = db.lookupType("MachSafePointNode");
45     jvmsField = type.getAddressField("_jvms");
46     jvmadjField = new CIntField(type.getCIntegerField("_jvmadj"), 0);
47   }
48 
49   private static AddressField jvmsField;
50   private static CIntField jvmadjField;
51 
52   public MachSafePointNode(Address addr) {
53     super(addr);
54   }
55 
56   public JVMState jvms() {
57     return JVMState.create(jvmsField.getValue(getAddress()));
58   }
59 
60   public void dumpSpec(PrintStream out) {
61     try {
62       JVMState jvms = jvms();
63       if (jvms != null) out.print(" !");
64       if (jvms == null) out.print("empty jvms");
65       while (jvms != null) {
66         Method m = jvms.method().method();
67         int bci = jvms.bci();
68         out.print(" " + m.getMethodHolder().getName().asString().replace('/', '.') + "::" + m.getName().asString() + " @ bci:" + bci);
69         jvms = jvms.caller();
70       }
71     } catch (Exception e) {
72       out.print(e);
73     }
74   }
75 }
