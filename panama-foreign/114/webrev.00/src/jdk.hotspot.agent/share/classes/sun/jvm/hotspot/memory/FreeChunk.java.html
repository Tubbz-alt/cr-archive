25 package sun.jvm.hotspot.memory;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 import sun.jvm.hotspot.types.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.oops.*;
32 import sun.jvm.hotspot.utilities.Observable;
33 import sun.jvm.hotspot.utilities.Observer;
34 
35 public class FreeChunk extends VMObject {
36    static {
37       VM.registerVMInitializedObserver(new Observer() {
38          public void update(Observable o, Object data) {
39             initialize(VM.getVM().getTypeDataBase());
40          }
41       });
42    }
43 
44    private static synchronized void initialize(TypeDataBase db) {
45       Type type = db.lookupType("FreeChunk");
46       nextField = type.getAddressField("_next");
47       prevField = type.getAddressField("_prev");
48       sizeField = type.getAddressField("_size");
49    }
50 
51    // Fields
52    private static AddressField nextField;
53    private static AddressField prevField;
54    private static AddressField sizeField;
55 
56    // Accessors
57    public FreeChunk next() {
58       return (FreeChunk) VMObjectFactory.newObject(FreeChunk.class, nextField.getValue(addr));
59    }
60 
61    public FreeChunk prev() {
62       Address prev = prevField.getValue(addr).andWithMask(~0x3);
63       return (FreeChunk) VMObjectFactory.newObject(FreeChunk.class, prev);
64    }
65 
66    public long size() {
67       if (VM.getVM().isCompressedOopsEnabled()) {
68         Mark mark = new Mark(addr.addOffsetTo(sizeField.getOffset()));
69         return mark.getSize();
70       } else {
71         Address size = sizeField.getValue(addr);
72         Debugger dbg = VM.getVM().getDebugger();
73         return dbg.getAddressValue(size);
74       }
75    }
76 
77    public FreeChunk(Address addr) {
78       super(addr);
79    }
80 
81    public static boolean indicatesFreeChunk(Address cur) {
82       FreeChunk f = new FreeChunk(cur);
83       return f.isFree();
84    }
85 
86    public boolean isFree() {
87       if (VM.getVM().isCompressedOopsEnabled()) {
88         Mark mark = new Mark(addr.addOffsetTo(sizeField.getOffset()));
89         return mark.isCmsFreeChunk();
90       } else {
91         Address prev = prevField.getValue(addr);
92         Debugger dbg = VM.getVM().getDebugger();
93         long word = dbg.getAddressValue(prev);
94         return (word & 0x1L) == 0x1L;
95       }
96    }
97 }
