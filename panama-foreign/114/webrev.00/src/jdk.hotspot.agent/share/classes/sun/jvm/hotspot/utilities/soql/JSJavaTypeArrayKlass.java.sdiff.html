71    public Object getFieldValue(int index, Array array) {
72       TypeArray typeArr = (TypeArray) array;
73       int type = (int) getTypeArrayKlass().getElementType();
74       switch (type) {
75          case TypeArrayKlass.T_BOOLEAN:
