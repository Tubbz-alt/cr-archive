25 package sun.jvm.hotspot.opto;
26 
27 import java.util.*;
28 import java.io.PrintStream;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.oops.*;
32 import sun.jvm.hotspot.types.*;
33 
34 public class Block extends VMObject {
35   static {
36     VM.registerVMInitializedObserver(new Observer() {
37         public void update(Observable o, Object data) {
38           initialize(VM.getVM().getTypeDataBase());
39         }
40       });
41   }
42 
43   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
44     Type type      = db.lookupType("Block");
45     nodesField = type.getAddressField("_nodes");
46     succsField = type.getAddressField("_succs");
47     numSuccsField = new CIntField(type.getCIntegerField("_num_succs"), 0);
48     preOrderField = new CIntField(type.getCIntegerField("_pre_order"), 0);
49     domDepthField = new CIntField(type.getCIntegerField("_dom_depth"), 0);
50     idomField = type.getAddressField("_idom");
51     freqField = type.getJDoubleField("_freq");
52   }
53 
54   private static AddressField nodesField;
55   private static AddressField succsField;
56   private static CIntField numSuccsField;
57   private static CIntField preOrderField;
58   private static CIntField domDepthField;
59   private static AddressField idomField;
60   private static JDoubleField freqField;
61 
62   public Block(Address addr) {
63     super(addr);
64   }
65 
66   public int preOrder() {
67     return (int)preOrderField.getValue(getAddress());
68   }
69 
70   public double freq() {
71     return (double)freqField.getValue(getAddress());
72   }
73 
74   public Node_List nodes() {
75     return new Node_List(getAddress().addOffsetTo(nodesField.getOffset()));
76   }
77 
78   public void dump(PrintStream out) {
79     out.print("B" + preOrder());
80     out.print(" Freq: " + freq());
81     out.println();
82     Node_List nl = nodes();
83     int cnt = nl.size();
84     for( int i=0; i<cnt; i++ )
85       nl.at(i).dump(out);
86     out.print("\n");
87   }
88 }
