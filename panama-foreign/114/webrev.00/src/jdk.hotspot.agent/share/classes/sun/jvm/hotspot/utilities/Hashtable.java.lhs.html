25 package sun.jvm.hotspot.utilities;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 import sun.jvm.hotspot.types.*;
30 import sun.jvm.hotspot.oops.*;
31 import sun.jvm.hotspot.runtime.*;


32 
33 public class Hashtable extends BasicHashtable {
34   static {
35     VM.registerVMInitializedObserver(new Observer() {
36         public void update(Observable o, Object data) {
37           initialize(VM.getVM().getTypeDataBase());
38         }
39       });
40   }
41 
42   private static synchronized void initialize(TypeDataBase db) {
43     // just to confirm that type exists
44     Type type = db.lookupType("IntptrHashtable");
45   }
46 
47   // derived class may return Class<? extends HashtableEntry>
48   protected Class getHashtableEntryClass() {
49     return HashtableEntry.class;
50   }
51 
52   public long computeHash(Symbol name) {
53     return name.identityHash();
54   }
55 
56   public int hashToIndex(long fullHash) {
57     return (int) (fullHash % tableSize());
58   }
59 
60   public Hashtable(Address addr) {
61     super(addr);
62   }
63 
64   // VM's Hashtable::hash_symbol
65   protected static long hashSymbol(byte[] buf) {
66     long h = 0;
67     int s = 0;
68     int len = buf.length;
69     while (len-- > 0) {
70       h = 31*h + (0xFFL & buf[s]);
71       s++;
72     }
73     return h & 0xFFFFFFFFL;
74   }
75 }
