 25 package sun.jvm.hotspot.gc.parallel;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 
 30 import sun.jvm.hotspot.debugger.*;
 31 import sun.jvm.hotspot.gc.shared.*;
 32 import sun.jvm.hotspot.memory.MemRegion;
 33 import sun.jvm.hotspot.runtime.*;
 34 import sun.jvm.hotspot.types.*;
 35 import sun.jvm.hotspot.utilities.Observable;
 36 import sun.jvm.hotspot.utilities.Observer;
 37 
 38 public class ParallelScavengeHeap extends CollectedHeap {
 39    static {
 40       VM.registerVMInitializedObserver(new Observer() {
 41          public void update(Observable o, Object data) {
 42             initialize(VM.getVM().getTypeDataBase());
 43          }
 44       });
 45    }
 46 
 47    private static synchronized void initialize(TypeDataBase db) {
 48       Type type = db.lookupType("ParallelScavengeHeap");
 49       youngGenField = type.getAddressField("_young_gen");
 50       oldGenField    = type.getAddressField("_old_gen");
 51    }
 52 
 53    public ParallelScavengeHeap(Address addr) {
 54       super(addr);
 55    }
 56 
 57    // Fields
 58    private static AddressField youngGenField;
 59    private static AddressField oldGenField;
 60 
 61    // Accessors
 62    public PSYoungGen youngGen() {
 63       return (PSYoungGen) VMObjectFactory.newObject(PSYoungGen.class, youngGenField.getValue());
 64    }
 65 
 66    public PSOldGen oldGen() {
 67       return (PSOldGen) VMObjectFactory.newObject(PSOldGen.class, oldGenField.getValue());
 68    }
 69 
 70    public long capacity() {
 71       return youngGen().capacity() + oldGen().capacity();
 72    }
 73 
 74    public long used() {
 75       return youngGen().used() + oldGen().used();
 76    }
 77 
 78    public boolean isIn(Address a) {
 79       if (youngGen().isIn(a)) {
 80          return true;
 81       }
 82 
 83       if (oldGen().isIn(a)) {
 84          return true;
 85       }
 86 
 87       return false;
 88    }
 89 
 90    public CollectedHeapName kind() {
 91       return CollectedHeapName.PARALLEL;
 92    }
 93 
 94    // Simple wrapper to provide toString() usable for debugging.
 95    private class LiveRegionProviderImpl implements LiveRegionsProvider {
 96       private String name;
 97       private MutableSpace space;
 98 
 99       public LiveRegionProviderImpl(String name, MutableSpace space) {
100          this.name = name;
101          this.space = space;
102       }
103 
104       @Override
105       public List<MemRegion> getLiveRegions() {
106          return space.getLiveRegions();
107       }
108       @Override
109       public String toString() {
110          return name;
111       }
112    }
113 
114    public void liveRegionsIterate(LiveRegionsClosure closure) {
115       // Add eden space
116       closure.doLiveRegions(new LiveRegionProviderImpl("eden", youngGen().edenSpace()));
117       // Add from-space but not to-space
118       closure.doLiveRegions(new LiveRegionProviderImpl("from", youngGen().fromSpace()));
119 
120       closure.doLiveRegions(new LiveRegionProviderImpl("old ", oldGen().objectSpace()));
121    }
122 
123    public void printOn(PrintStream tty) {
124       tty.print("ParallelScavengeHeap [ ");
125       youngGen().printOn(tty);
126       oldGen().printOn(tty);
127       tty.print(" ] ");
128    }
129 }
