25 package sun.jvm.hotspot.oops;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.memory.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.types.AddressField;
33 import sun.jvm.hotspot.types.Type;
34 import sun.jvm.hotspot.types.TypeDataBase;
35 import sun.jvm.hotspot.utilities.*;
36 import sun.jvm.hotspot.utilities.Observable;
37 import sun.jvm.hotspot.utilities.Observer;
38 
39 /** A utility class encapsulating useful oop operations */
40 
41 // initialize fields for java.lang.Class
42 public class java_lang_Class {
43 
44   // java.lang.Class fields
45   static int klassOffset;
46   static int arrayKlassOffset;
47   static IntField oopSizeField;
48 
49   static {
50     VM.registerVMInitializedObserver(new Observer() {
51         public void update(Observable o, Object data) {
52           initialize(VM.getVM().getTypeDataBase());
53         }
54       });
55   }
56 
57   private static synchronized void initialize(TypeDataBase db) {
58     // klass and oop_size are HotSpot magic fields and hence we can't
59     // find them from InstanceKlass for java.lang.Class.
60     Type jlc = db.lookupType("java_lang_Class");
61     klassOffset = (int) jlc.getCIntegerField("_klass_offset").getValue();
62     arrayKlassOffset = (int) jlc.getCIntegerField("_array_klass_offset").getValue();
63     int oopSizeOffset = (int) jlc.getCIntegerField("_oop_size_offset").getValue();
64     oopSizeField = new IntField(new NamedFieldIdentifier("oop_size"), oopSizeOffset, true);
65   }
66 
67   /** get Klass* field at offset hc_klass_offset from a java.lang.Class object */
68   public static Klass asKlass(Oop aClass) {
69     return (Klass)Metadata.instantiateWrapperFor(aClass.getHandle().getAddressAt(klassOffset));
70   }
71 
72   /** get oop_size field at offset oop_size_offset from a java.lang.Class object */
73   public static long getOopSize(Oop aClass) {
74     return java_lang_Class.oopSizeField.getValue(aClass);
75   }
76 
77   /**
78    * Returns the Java name for this Java mirror
79    */
80   public static String asExternalName(Oop aClass) {
81     Klass k = java_lang_Class.asKlass(aClass);
82     if (k == null) { // primitive array
83       BasicType type = BasicType.T_VOID;
84       ArrayKlass ak = (ArrayKlass)Metadata.instantiateWrapperFor(
85                              aClass.getHandle().getAddressAt(arrayKlassOffset));
86       if (ak != null) {
87         type = BasicType.intToBasicType(ak.getElementType());
88       }
89       return type.getName();
90     } else {
91       return k.getName().asString();
92     }
93   }
94 
95 }
