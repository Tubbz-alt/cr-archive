25 package sun.jvm.hotspot.oops;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.memory.*;
30 import sun.jvm.hotspot.utilities.*;
31 import sun.jvm.hotspot.debugger.*;
32 import sun.jvm.hotspot.runtime.*;
33 import sun.jvm.hotspot.types.*;
34 
35 abstract public class Metadata extends VMObject {
36   static {
37     VM.registerVMInitializedObserver(new Observer() {
38         public void update(Observable o, Object data) {
39           initialize(VM.getVM().getTypeDataBase());
40         }
41       });
42   }
43 
44   public Metadata(Address addr) {
45     super(addr);
46   }
47 
48   public static long alignSize(long size) {
49     // natural word size.
50     return VM.getVM().alignUp(size, VM.getVM().getBytesPerWord());
51   }
52 
53   private static VirtualBaseConstructor<Metadata> metadataConstructor;
54 
55   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
56     metadataConstructor = new VirtualBaseConstructor<Metadata>(db, db.lookupType("Metadata"), null, null);
57     // Define an explicit mapping since the C++ and Java type names don't match.
58     metadataConstructor.addMapping("Metadata", Metadata.class);
59     metadataConstructor.addMapping("Klass", Klass.class);
60     metadataConstructor.addMapping("InstanceKlass", InstanceKlass.class);
61     metadataConstructor.addMapping("InstanceMirrorKlass", InstanceMirrorKlass.class);
62     metadataConstructor.addMapping("InstanceRefKlass", InstanceRefKlass.class);
63     metadataConstructor.addMapping("InstanceClassLoaderKlass", InstanceClassLoaderKlass.class);
64     metadataConstructor.addMapping("TypeArrayKlass", TypeArrayKlass.class);
65     metadataConstructor.addMapping("ObjArrayKlass", ObjArrayKlass.class);
66     metadataConstructor.addMapping("Method", Method.class);
67     metadataConstructor.addMapping("MethodData", MethodData.class);
68     metadataConstructor.addMapping("ConstMethod", ConstMethod.class);
69     metadataConstructor.addMapping("ConstantPool", ConstantPool.class);
70     metadataConstructor.addMapping("ConstantPoolCache", ConstantPoolCache.class);
71   }
72 
73   public static Metadata instantiateWrapperFor(Address addr) {
74     return metadataConstructor.instantiateWrapperFor(addr);
75   }
76 
77   public void iterate(MetadataVisitor visitor) {
78     visitor.setObj(this);
79     visitor.prologue();
80     iterateFields(visitor);
81     visitor.epilogue();
82   }
83 
84   void iterateFields(MetadataVisitor visitor) {
85   }
86 
87   abstract public void printValueOn(PrintStream tty);
88   public void dumpReplayData(PrintStream out) {
89       out.println("# Unknown Metadata");
90   }
91 
92   public boolean isShared() {
93     VM vm = VM.getVM();
94     if (vm.isSharingEnabled()) {
95       return MetaspaceObj.isShared(getAddress());
96     }
97     return false;
98   }
99 }
