25 package sun.jvm.hotspot.interpreter;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.code.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.Observable;
33 import sun.jvm.hotspot.utilities.Observer;
34 
35 public class Interpreter {
36   private static AddressField codeField;
37 
38   static {
39     VM.registerVMInitializedObserver(new Observer() {
40         public void update(Observable o, Object data) {
41           initialize(VM.getVM().getTypeDataBase());
42         }
43       });
44   }
45 
46   private static synchronized void initialize(TypeDataBase db) {
47     Type type = db.lookupType("AbstractInterpreter");
48 
49     codeField     = type.getAddressField("_code");
50   }
51 
52   public Interpreter() {
53   }
54 
55   public StubQueue getCode() {
56     Address code = codeField.getValue();
57     if (code == null) return null;
58     return new StubQueue(code, InterpreterCodelet.class);
59   }
60 
61   public boolean contains(Address pc) {
62     return getCode().contains(pc);
63   }
64 
65   /** Debugging/printing */
66   public InterpreterCodelet getCodeletContaining(Address pc) {
67     return (InterpreterCodelet) getCode().getStubContaining(pc);
68   }
69 }
