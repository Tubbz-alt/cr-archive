25 package sun.jvm.hotspot.opto;
26 
27 import java.util.*;
28 import java.io.*;
29 import sun.jvm.hotspot.code.*;
30 import sun.jvm.hotspot.debugger.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.oops.*;
33 import sun.jvm.hotspot.ci.*;
34 import sun.jvm.hotspot.types.*;
35 import sun.jvm.hotspot.utilities.Observable;
36 import sun.jvm.hotspot.utilities.Observer;
37 
38 public class CallJavaNode extends CallNode {
39   static {
40     VM.registerVMInitializedObserver(new Observer() {
41         public void update(Observable o, Object data) {
42           initialize(VM.getVM().getTypeDataBase());
43         }
44       });
45   }
46 
47   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
48     Type type      = db.lookupType("CallJavaNode");
49     methodField = type.getAddressField("_method");
50   }
51 
52   private static AddressField methodField;
53 
54   public CallJavaNode(Address addr) {
55     super(addr);
56   }
57 
58   public ciMethod method() {
59     return (ciMethod) ciObjectFactory.getMetadata(methodField.getValue(getAddress()));
60   }
61 
62   public void dumpSpec(PrintStream out) {
63     if (method() !=  null) {
64       out.print(" " + method().method().externalNameAndSignature());
65     }
66     super.dumpSpec(out);
67   }
68 }
