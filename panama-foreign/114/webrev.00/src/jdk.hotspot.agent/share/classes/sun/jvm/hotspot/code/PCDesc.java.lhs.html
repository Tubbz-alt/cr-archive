 25 package sun.jvm.hotspot.code;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 import sun.jvm.hotspot.debugger.*;
 30 import sun.jvm.hotspot.runtime.*;
 31 import sun.jvm.hotspot.types.*;


 32 
 33 /** PcDescs map a physical PC (given as offset from start of nmethod)
 34     to the corresponding source scope and byte code index. */
 35 
 36 public class PCDesc extends VMObject {
 37   private static CIntegerField pcOffsetField;
 38   private static CIntegerField scopeDecodeOffsetField;
 39   private static CIntegerField objDecodeOffsetField;
 40   private static CIntegerField pcFlagsField;
 41   private static int reexecuteMask;
 42   private static int isMethodHandleInvokeMask;
 43   private static int returnOopMask;
 44 
 45   static {
 46     VM.registerVMInitializedObserver(new Observer() {
 47         public void update(Observable o, Object data) {
 48           initialize(VM.getVM().getTypeDataBase());
 49         }
 50       });
 51   }
 52 
 53   private static void initialize(TypeDataBase db) {
 54     Type type = db.lookupType("PcDesc");
 55 
 56     pcOffsetField          = type.getCIntegerField("_pc_offset");
 57     scopeDecodeOffsetField = type.getCIntegerField("_scope_decode_offset");
 58     objDecodeOffsetField   = type.getCIntegerField("_obj_decode_offset");
 59     pcFlagsField           = type.getCIntegerField("_flags");
 60 
 61     reexecuteMask            = db.lookupIntConstant("PcDesc::PCDESC_reexecute");
 62     isMethodHandleInvokeMask = db.lookupIntConstant("PcDesc::PCDESC_is_method_handle_invoke");
 63     returnOopMask            = db.lookupIntConstant("PcDesc::PCDESC_return_oop");
 64   }
 65 
 66   public PCDesc(Address addr) {
 67     super(addr);
 68   }
 69 
 70   // FIXME: add additional constructor probably needed for ScopeDesc::sender()
 71 
 72   public int getPCOffset() {
 73     return (int) pcOffsetField.getValue(addr);
 74   }
 75 
 76   public int getScopeDecodeOffset() {
 77     return ((int) scopeDecodeOffsetField.getValue(addr));
 78   }
 79 
 80   public int getObjDecodeOffset() {
 81     return ((int) objDecodeOffsetField.getValue(addr));
 82   }
 83 
 84   public Address getRealPC(NMethod code) {
 85     return code.codeBegin().addOffsetTo(getPCOffset());
 86   }
 87 
 88 
 89   public boolean getReexecute() {
 90     int flags = (int)pcFlagsField.getValue(addr);
 91     return (flags & reexecuteMask) != 0;
 92   }
 93 
 94   public boolean isMethodHandleInvoke() {
 95     int flags = (int)pcFlagsField.getValue(addr);
 96     return (flags & isMethodHandleInvokeMask) != 0;
 97   }
 98 
 99   public void print(NMethod code) {
100     printOn(System.out, code);
101   }
102 
103   public void printOn(PrintStream tty, NMethod code) {
104     tty.println("PCDesc(" + getRealPC(code) + "):");
105     for (ScopeDesc sd = code.getScopeDescAt(getRealPC(code));
106          sd != null;
107          sd = sd.sender()) {
108       tty.print(" ");
109       sd.getMethod().printValueOn(tty);
110       tty.print("  @" + sd.getBCI());
111       tty.print("  reexecute=" + sd.getReexecute());
112       tty.println();
113     }
114   }
115 }
