23 package sun.jvm.hotspot.code;
24 
25 import java.util.*;
26 
27 import sun.jvm.hotspot.debugger.Address;
28 import sun.jvm.hotspot.oops.*;
29 import sun.jvm.hotspot.runtime.*;
30 import sun.jvm.hotspot.types.*;
31 import sun.jvm.hotspot.utilities.*;
32 
33 public abstract class CompiledMethod extends CodeBlob {
34   private static AddressField  methodField;
35   private static AddressField  deoptHandlerBeginField;
36   private static AddressField  deoptMhHandlerBeginField;
37   private static AddressField  scopesDataBeginField;
38 
39   static {
40     VM.registerVMInitializedObserver(new Observer() {
41         public void update(Observable o, Object data) {
42           initialize(VM.getVM().getTypeDataBase());
43         }
44       });
45   }
46 
47   private static void initialize(TypeDataBase db) {
48     Type type = db.lookupType("CompiledMethod");
49 
50     methodField                 = type.getAddressField("_method");
51     deoptHandlerBeginField      = type.getAddressField("_deopt_handler_begin");
52     deoptMhHandlerBeginField    = type.getAddressField("_deopt_mh_handler_begin");
53     scopesDataBeginField        = type.getAddressField("_scopes_data_begin");
54   }
55 
56   public CompiledMethod(Address addr) {
57     super(addr);
58   }
59 
60   public Method getMethod() {
61     return (Method)Metadata.instantiateWrapperFor(methodField.getValue(addr));
62   }
63 
64   public Address deoptHandlerBegin()    { return deoptHandlerBeginField.getValue(addr);              }
65   public Address deoptMhHandlerBegin()  { return deoptMhHandlerBeginField.getValue(addr);            }
66   public Address scopesDataBegin()      { return scopesDataBeginField.getValue(addr);                }
67 
68   public static int getMethodOffset()                { return (int) methodField.getOffset();                }
69 
70   @Override
71   public boolean isCompiled() {
72     return true;
73   }
74 }
