25 package sun.jvm.hotspot.runtime;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.*;
33 
34 /** This class represent VM's Arguments class -- command line args, flags etc.*/
35 public class Arguments {
36     static {
37         VM.registerVMInitializedObserver(new Observer() {
38             public void update(Observable o, Object data) {
39                 initialize(VM.getVM().getTypeDataBase());
40             }
41         });
42     }
43 
44     public static String getJVMFlags() {
45         return buildString(jvmFlagsArrayField, numJvmFlags);
46     }
47 
48     public static String getJVMArgs() {
49         return buildString(jvmArgsArrayField, numJvmArgs);
50     }
51 
52     public static String getJavaCommand() {
53         return getString(javaCommandField);
54     }
55 
56     // Internals only below this point
57 
58     // Fields
59     private static AddressField jvmFlagsArrayField;
60     private static AddressField jvmArgsArrayField;
61     private static AddressField javaCommandField;
62     private static long numJvmFlags;
63     private static long numJvmArgs;
64 
65     private static synchronized void initialize(TypeDataBase db) {
66         Type argumentsType = db.lookupType("Arguments");
67         jvmFlagsArrayField = argumentsType.getAddressField("_jvm_flags_array");
68         jvmArgsArrayField = argumentsType.getAddressField("_jvm_args_array");
69         javaCommandField = argumentsType.getAddressField("_java_command");
70 
71         numJvmArgs = argumentsType.getCIntegerField("_num_jvm_args").getValue();
72         numJvmFlags = argumentsType.getCIntegerField("_num_jvm_flags").getValue();
73     }
74 
75     private static String buildString(AddressField arrayField, long count) {
76         StringBuilder sb = new StringBuilder();
77         if (count > 0) {
78             sb.append(getStringAt(arrayField, 0));
79             for (long i = 1; i < count; i++) {
80                 sb.append(" ");
81                 sb.append(getStringAt(arrayField, i));
82             }
83         }
84         return sb.toString();
85     }
86 
87     private static String getString(AddressField field) {
88         Address addr = field.getAddress();
89         return CStringUtilities.getString(addr);
90     }
91 
92     private static String getStringAt(AddressField field, long index) {
93         Address addr = field.getAddress();
94         return CStringUtilities.getString(addr.getAddressAt(index * VM.getVM().getAddressSize()));
95     }
96 }
