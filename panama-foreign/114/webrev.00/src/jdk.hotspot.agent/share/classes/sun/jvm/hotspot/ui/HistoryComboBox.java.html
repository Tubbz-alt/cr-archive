25 package sun.jvm.hotspot.ui;
26 
27 
28 import java.awt.event.*;
29 import javax.swing.*;
30 import javax.swing.event.*;
31 
32 /** Provides an editable text field with history. */
33 
34 public class HistoryComboBox extends JComboBox<String> {
35   static final int HISTORY_LENGTH = 15;
36 
37   public HistoryComboBox() {
38     setEditable(true);
39     addActionListener(new ActionListener() {
40         public void actionPerformed(ActionEvent e) {
41           Object text = getSelectedItem();
42           if (text != null) {
43             setText((String)text);
44           }
45         }
46       });
47   }
48 
49   public String getText() {
50     Object text = getSelectedItem();
51     if (text == null) {
52       return "";
53     }
54     return (String)text;
55   }
56 
57   public void setText(String text) {
58     removeItem(text);
59     insertItemAt(text, 0);
60     setSelectedItem(text);
61     int length = getModel().getSize();
62     while (length > HISTORY_LENGTH) {
63       removeItemAt(--length);
64     }
65   }
66 }
