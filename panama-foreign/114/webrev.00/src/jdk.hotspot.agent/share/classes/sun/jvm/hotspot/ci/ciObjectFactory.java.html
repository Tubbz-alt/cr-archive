25 package sun.jvm.hotspot.ci;
26 
27 import java.lang.reflect.Constructor;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.utilities.*;
32 import sun.jvm.hotspot.oops.*;
33 import sun.jvm.hotspot.types.*;
34 import sun.jvm.hotspot.utilities.Observable;
35 import sun.jvm.hotspot.utilities.Observer;
36 
37 public class ciObjectFactory extends VMObject {
38   static {
39     VM.registerVMInitializedObserver(new Observer() {
40         public void update(Observable o, Object data) {
41           initialize(VM.getVM().getTypeDataBase());
42         }
43       });
44   }
45 
46   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
47     Type type      = db.lookupType("ciObjectFactory");
48     unloadedMethodsField = type.getAddressField("_unloaded_methods");
49     ciMetadataField = type.getAddressField("_ci_metadata");
50     symbolsField = type.getAddressField("_symbols");
51 
52     ciObjectConstructor = new VirtualBaseConstructor<ciObject>(db, db.lookupType("ciObject"), "sun.jvm.hotspot.ci", ciObject.class);
53     ciMetadataConstructor = new VirtualBaseConstructor<ciMetadata>(db, db.lookupType("ciMetadata"), "sun.jvm.hotspot.ci", ciMetadata.class);
54     ciSymbolConstructor = new VirtualBaseConstructor<ciSymbol>(db, db.lookupType("ciSymbol"), "sun.jvm.hotspot.ci", ciSymbol.class);
55   }
56 
57   private static AddressField unloadedMethodsField;
58   private static AddressField ciMetadataField;
59   private static AddressField symbolsField;
60 
61   private static VirtualBaseConstructor<ciObject> ciObjectConstructor;
62   private static VirtualBaseConstructor<ciMetadata> ciMetadataConstructor;
63   private static VirtualBaseConstructor<ciSymbol> ciSymbolConstructor;
64 
65   public static ciObject get(Address addr) {
66     if (addr == null) return null;
67 
68     return ciObjectConstructor.instantiateWrapperFor(addr);
69   }
70 
71   public static ciMetadata getMetadata(Address addr) {
72     if (addr == null) return null;
73 
74     return ciMetadataConstructor.instantiateWrapperFor(addr);
75   }
76 
77   public GrowableArray<ciMetadata> objects() {
78     return GrowableArray.create(ciMetadataField.getValue(getAddress()), ciMetadataConstructor);
79   }
80 
81   public GrowableArray<ciSymbol> symbols() {
82     return GrowableArray.create(symbolsField.getValue(getAddress()), ciSymbolConstructor);
83   }
84 
85   public ciObjectFactory(Address addr) {
86     super(addr);
87   }
88 }
