 25 package sun.jvm.hotspot.gc.shared;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 
 30 import sun.jvm.hotspot.debugger.*;
 31 import sun.jvm.hotspot.memory.*;
 32 import sun.jvm.hotspot.runtime.*;
 33 import sun.jvm.hotspot.types.*;
 34 import sun.jvm.hotspot.utilities.Observable;
 35 import sun.jvm.hotspot.utilities.Observer;
 36 
 37 public class ContiguousSpace extends CompactibleSpace implements LiveRegionsProvider {
 38   private static AddressField topField;
 39 
 40   static {
 41     VM.registerVMInitializedObserver(new Observer() {
 42         public void update(Observable o, Object data) {
 43           initialize(VM.getVM().getTypeDataBase());
 44         }
 45       });
 46   }
 47 
 48   private static synchronized void initialize(TypeDataBase db) {
 49     Type type = db.lookupType("ContiguousSpace");
 50 
 51     topField = type.getAddressField("_top");
 52   }
 53 
 54   public ContiguousSpace(Address addr) {
 55     super(addr);
 56   }
 57 
 58   public Address top() {
 59     return topField.getValue(addr);
 60   }
 61 
 62   /** In bytes */
 63   public long capacity() {
 64     return end().minus(bottom());
 65   }
 66 
 67   /** In bytes */
 68   public long used() {
 69     return top().minus(bottom());
 70   }
 71 
 72   /** In bytes */
 73   public long free() {
 74     return end().minus(top());
 75   }
 76 
 77   /** In a contiguous space we have a more obvious bound on what parts
 78       contain objects. */
 79   public MemRegion usedRegion() {
 80     return new MemRegion(bottom(), top());
 81   }
 82 
 83   /** Returns regions of Space where live objects live */
 84   public List<MemRegion> getLiveRegions() {
 85     List<MemRegion> res = new ArrayList<>();
 86     res.add(new MemRegion(bottom(), top()));
 87     return res;
 88   }
 89 
 90   /** Testers */
 91   public boolean contains(Address p) {
 92     return (bottom().lessThanOrEqual(p) && top().greaterThan(p));
 93   }
 94 
 95   public void printOn(PrintStream tty) {
 96     tty.print(" [" + bottom() + "," +
 97                 top() + "," + end() + ")");
 98     super.printOn(tty);
 99   }
100 }
