25 package sun.jvm.hotspot.memory;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 import sun.jvm.hotspot.runtime.*;
30 import sun.jvm.hotspot.types.*;
31 import sun.jvm.hotspot.utilities.Observable;
32 import sun.jvm.hotspot.utilities.Observer;
33 
34 public class VirtualSpace extends VMObject {
35   private static AddressField lowField;
36   private static AddressField highField;
37   private static AddressField lowBoundaryField;
38   private static AddressField highBoundaryField;
39 
40   static {
41     VM.registerVMInitializedObserver(new Observer() {
42         public void update(Observable o, Object data) {
43           initialize(VM.getVM().getTypeDataBase());
44         }
45       });
46   }
47 
48   private static synchronized void initialize(TypeDataBase db) {
49     Type type = db.lookupType("VirtualSpace");
50 
51     lowField          = type.getAddressField("_low");
52     highField         = type.getAddressField("_high");
53     lowBoundaryField  = type.getAddressField("_low_boundary");
54     highBoundaryField = type.getAddressField("_high_boundary");
55   }
56 
57   public VirtualSpace(Address addr) {
58     super(addr);
59   }
60 
61   public Address low()                          { return lowField.getValue(addr);          }
62   public Address high()                         { return highField.getValue(addr);         }
63   public Address lowBoundary()                  { return lowBoundaryField.getValue(addr);  }
64   public Address highBoundary()                 { return highBoundaryField.getValue(addr); }
65 
66   /** Testers (all sizes are byte sizes) */
67   public long committedSize()                   { return high().minus(low());                                    }
68   public long reservedSize()                    { return highBoundary().minus(lowBoundary());                    }
69   public long uncommittedSize()                 { return reservedSize() - committedSize();                       }
70   public boolean contains(Address addr)         { return (low().lessThanOrEqual(addr) && addr.lessThan(high())); }
71 }
