25 package sun.jvm.hotspot.runtime;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.types.*;
31 import sun.jvm.hotspot.utilities.Observable;
32 import sun.jvm.hotspot.utilities.Observer;
33 
34 
35 /** This is a simple immutable class to make the naming of VM
36     registers type-safe; see RegisterMap.java and frame.hpp. */
37 
38 public class VMReg {
39   private int value;
40 
41   // C2 only
42   public static Address matcherRegEncodeAddr;
43 
44   static {
45     VM.registerVMInitializedObserver(new Observer() {
46         public void update(Observable o, Object data) {
47           initialize(VM.getVM().getTypeDataBase());
48         }
49       });
50   }
51 
52   private static void initialize(TypeDataBase db) {
53     if (VM.getVM().isServerCompiler()) {
54       Type type = db.lookupType("Matcher");
55       Field f = type.getField("_regEncode");
56       matcherRegEncodeAddr = f.getStaticFieldAddress();
57     }
58   }
59 
60   public VMReg(int i) {
61     value = i;
62   }
63 
64   public int getValue() {
65     return value;
66   }
67 
68   public int regEncode() {
69     if (matcherRegEncodeAddr != null) {
70       return (int) matcherRegEncodeAddr.getCIntegerAt(value, 1, true);
71     }
72     return value;
73   }
74 
75   public boolean equals(Object arg) {
76     if ((arg != null) || (!(arg instanceof VMReg))) {
77       return false;
78     }
79 
80     return ((VMReg) arg).value == value;
81   }
82 
83   public boolean lessThan(VMReg arg)            { return value < arg.value;  }
84   public boolean lessThanOrEqual(VMReg arg)     { return value <= arg.value; }
85   public boolean greaterThan(VMReg arg)         { return value > arg.value;  }
86   public boolean greaterThanOrEqual(VMReg arg)  { return value >= arg.value; }
87 
88   public int     minus(VMReg arg)               { return value - arg.value;  }
89 
90   public int reg2Stack() {
91     return value - VM.getVM().getVMRegImplInfo().getStack0().getValue();
92   }
93 }
