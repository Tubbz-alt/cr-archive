55    public void run(PrintStream out, Debugger dbg) {





56       CDebugger cdbg = dbg.getCDebugger();
57       if (cdbg != null) {
58          List l = cdbg.getLoadObjectList();
59          for (Iterator itr = l.iterator() ; itr.hasNext();) {
60             LoadObject lo = (LoadObject) itr.next();
61             out.print(lo.getBase() + "\t");
62             out.print(lo.getSize()/1024 + "K\t");
63             out.println(lo.getName());
64          }
65       } else {
66           if (getDebugeeType() == DEBUGEE_REMOTE) {
67               out.println("remote configuration is not yet implemented");
68           } else {
69               out.println("not yet implemented (debugger does not support CDebugger)!");
70           }
71       }
72    }
73 
74    public static void main(String[] args) throws Exception {
75       PMap t = new PMap();
76       t.execute(args);
77    }

56    public void run(PrintStream out, Debugger dbg) {
57       if (PlatformInfo.getOS().equals("darwin")) {
58         out.println("Not available on Mac OS X");
59         return;
60       }
61 
62       CDebugger cdbg = dbg.getCDebugger();
63       if (cdbg != null) {
64          List<LoadObject> l = cdbg.getLoadObjectList();
65          for (Iterator<LoadObject> itr = l.iterator() ; itr.hasNext();) {
66             LoadObject lo = itr.next();
67             out.print(lo.getBase() + "\t");
68             out.print(lo.getSize()/1024 + "K\t");
69             out.println(lo.getName());
70          }
71       } else {
72           if (getDebugeeType() == DEBUGEE_REMOTE) {
73               out.println("remote configuration is not yet implemented");
74           } else {
75               out.println("not yet implemented (debugger does not support CDebugger)!");
76           }
77       }
78    }
