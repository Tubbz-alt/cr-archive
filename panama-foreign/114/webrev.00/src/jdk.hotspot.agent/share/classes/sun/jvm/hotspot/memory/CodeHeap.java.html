 25 package sun.jvm.hotspot.memory;
 26 
 27 import java.util.*;
 28 import sun.jvm.hotspot.code.*;
 29 import sun.jvm.hotspot.debugger.*;
 30 import sun.jvm.hotspot.runtime.*;
 31 import sun.jvm.hotspot.types.*;
 32 import sun.jvm.hotspot.utilities.Observable;
 33 import sun.jvm.hotspot.utilities.Observer;
 34 
 35 public class CodeHeap extends VMObject {
 36   private static Field         memoryField;
 37   private static Field         segmapField;
 38   //  private static CIntegerField numberOfCommittedSegmentsField;
 39   //  private static CIntegerField numberOfReservedSegmentsField;
 40   //  private static CIntegerField segmentSizeField;
 41   private static CIntegerField log2SegmentSizeField;
 42   //  private static CIntegerField nextSegmentField;
 43   //  private static AddressField  freelistField;
 44   //  private static CIntegerField freeSegmentsField;
 45 
 46   private VirtualSpace memory;
 47   private VirtualSpace segmentMap;
 48   private int log2SegmentSize;
 49 
 50   static {
 51     VM.registerVMInitializedObserver(new Observer() {
 52         public void update(Observable o, Object data) {
 53           initialize(VM.getVM().getTypeDataBase());
 54         }
 55       });
 56   }
 57 
 58   private static void initialize(TypeDataBase db) {
 59     Type type = db.lookupType("CodeHeap");
 60 
 61     memoryField          = type.getField("_memory");
 62     segmapField          = type.getField("_segmap");
 63     log2SegmentSizeField = type.getCIntegerField("_log2_segment_size");
 64 
 65   }
 66 
 67   public CodeHeap(Address addr) {
 68     super(addr);
 69     log2SegmentSize = (int) log2SegmentSizeField.getValue(addr);
 70     segmentMap = new VirtualSpace(addr.addOffsetTo(segmapField.getOffset()));
 71     memory = new VirtualSpace(addr.addOffsetTo(memoryField.getOffset()));
 72   }
 73 
 74   public Address begin() {
 75     return getMemory().low();
 76   }
 77 
 78   public Address end() {
 79     return getMemory().high();
 80   }
 81 
 82   public boolean contains(Address p) {
 83     return (begin().lessThanOrEqual(p) && end().greaterThan(p));
 84   }
 85 
 86   /** Returns the start of the block containing p or null */
 87   public Address findStart(Address p) {
 88     if (!contains(p)) return null;
 89     HeapBlock h = blockStart(p);
 90     if (h == null || h.isFree()) {
 91       return null;
 92     }
 93     return h.getAllocatedSpace();
 94   }
 95 
 96   private Address nextBlock(Address ptr) {
 97     Address base = blockBase(ptr);
 98     if (base == null) {
 99       return null;
100     }
101     HeapBlock block = getBlockAt(base);
102     return base.addOffsetTo(block.getLength() * (1 << getLog2SegmentSize()));
103   }
104 
105   public void iterate(CodeCacheVisitor visitor, CodeCache cache) {
106     CodeBlob lastBlob = null;
107     Address ptr = begin();
108     while (ptr != null && ptr.lessThan(end())) {
109       try {
110         // Use findStart to get a pointer inside blob other findBlob asserts
111         CodeBlob blob = cache.createCodeBlobWrapper(findStart(ptr));
112         if (blob != null) {
113           visitor.visit(blob);
114           if (blob == lastBlob) {
115             throw new InternalError("saw same blob twice");
116           }
117           lastBlob = blob;
118         }
119       } catch (RuntimeException e) {
120         e.printStackTrace();
121       }
122       Address next = nextBlock(ptr);
123       if (next != null && next.lessThan(ptr)) {
124         throw new InternalError("pointer moved backwards");
125       }
126       ptr = next;
127     }
128   }
129 
130   //--------------------------------------------------------------------------------
131   // Internals only below this point
132   //
133 
134   private VirtualSpace getMemory() {
135     return memory;
136   }
137 
138   private VirtualSpace getSegmentMap() {
139     return segmentMap;
140   }
141 
142   private long segmentFor(Address p) {
143     return p.minus(getMemory().low()) >> getLog2SegmentSize();
144   }
145 
146   private int getLog2SegmentSize() {
147     return log2SegmentSize;
148   }
149 
150   private HeapBlock getBlockAt(Address addr) {
151     return (HeapBlock) VMObjectFactory.newObject(HeapBlock.class, addr);
152   }
153 
154 
155   private HeapBlock blockStart(Address p) {
156     Address base = blockBase(p);
157     if (base == null) return null;
158     return getBlockAt(base);
159   }
160 
161   private Address blockBase(Address p) {
162     long i = segmentFor(p);
163     Address b = getSegmentMap().low();
164     if (b.getCIntegerAt(i, 1, true) == 0xFF) {
165       return null;
166     }
167     while (b.getCIntegerAt(i, 1, true) > 0) {
168       i -= b.getCIntegerAt(i, 1, true);
169     }
170     return getMemory().low().addOffsetTo(i << getLog2SegmentSize());
171   }
