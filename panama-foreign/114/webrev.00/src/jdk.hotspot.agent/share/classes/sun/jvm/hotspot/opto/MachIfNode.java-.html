25 package sun.jvm.hotspot.opto;
26 
27 import java.util.*;
28 import java.io.PrintStream;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.oops.*;
32 import sun.jvm.hotspot.types.*;
33 
34 public class MachIfNode extends MachNode {
35   static {
36     VM.registerVMInitializedObserver(new Observer() {
37         public void update(Observable o, Object data) {
38           initialize(VM.getVM().getTypeDataBase());
39         }
40       });
41   }
42 
43   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
44     Type type      = db.lookupType("MachIfNode");
45     probField = type.getJFloatField("_prob");
46     fcntField = type.getJFloatField("_fcnt");
47   }
48 
49   private static JFloatField probField;
50   private static JFloatField fcntField;
51 
52   float prob() {
53     return probField.getValue(getAddress());
54   }
55 
56   float cnt() {
57     return fcntField.getValue(getAddress());
58   }
59 
60   public MachIfNode(Address addr) {
61     super(addr);
62   }
63 
64   public void dumpSpec(PrintStream out) {
65     out.print("P=" + prob() + ", C=" + cnt());
66   }
67 }
