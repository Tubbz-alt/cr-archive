25 package sun.jvm.hotspot.memory;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 import sun.jvm.hotspot.runtime.*;
30 import sun.jvm.hotspot.types.*;


31 
32 public class HeapBlock extends VMObject {
33   private static long          heapBlockSize;
34   private static Field         headerField;
35   // Fields for class Header
36   private static CIntegerField headerLengthField;
37   private static CIntegerField headerUsedField;
38 
39   static {
40     VM.registerVMInitializedObserver(new Observer() {
41         public void update(Observable o, Object data) {
42           initialize(VM.getVM().getTypeDataBase());
43         }
44       });
45   }
46 
47   private static void initialize(TypeDataBase db) {
48     // HeapBlock
49     Type type = db.lookupType("HeapBlock");
50     heapBlockSize     = type.getSize();
51     headerField       = type.getField("_header");
52 
53     // Header
54     type = db.lookupType("HeapBlock::Header");
55     headerLengthField = type.getCIntegerField("_length");
56     headerUsedField   = type.getCIntegerField("_used");
57   }
58 
59   public HeapBlock(Address addr) {
60     super(addr);
61   }
62 
63   public long getLength() {
64     return getHeader().getLength();
65   }
66 
67   public boolean isFree() {
68     return getHeader().isFree();
69   }
70 
71   public Address getAllocatedSpace() {
72     return addr.addOffsetTo(heapBlockSize);
73   }
74 
75   //--------------------------------------------------------------------------------
76   // Internals only below this point
77   //
78 
79   public static class Header extends VMObject {
80     public Header(Address addr) {
81       super(addr);
82     }
83 
84     public long getLength() {
85       return headerLengthField.getValue(addr);
86     }
87 
88     public boolean isFree() {
89       return (headerUsedField.getValue(addr) == 0);
90     }
91   }
92 
93   private Header getHeader() {
94     return (Header) VMObjectFactory.newObject(HeapBlock.Header.class, addr.addOffsetTo(headerField.getOffset()));
95   }
96 }
