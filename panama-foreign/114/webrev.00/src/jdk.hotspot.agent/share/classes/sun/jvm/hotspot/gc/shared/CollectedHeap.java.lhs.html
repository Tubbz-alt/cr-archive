 25 package sun.jvm.hotspot.gc.shared;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 
 30 import sun.jvm.hotspot.debugger.*;
 31 import sun.jvm.hotspot.gc.shared.*;
 32 import sun.jvm.hotspot.memory.*;
 33 import sun.jvm.hotspot.runtime.*;
 34 import sun.jvm.hotspot.types.*;
 35 import sun.jvm.hotspot.utilities.BitMapInterface;
 36 import sun.jvm.hotspot.utilities.BitMapSegmented;


 37 
 38 public abstract class CollectedHeap extends VMObject {
 39   private static long         reservedFieldOffset;
 40 
 41   static {
 42     VM.registerVMInitializedObserver(new Observer() {
 43         public void update(Observable o, Object data) {
 44           initialize(VM.getVM().getTypeDataBase());
 45         }
 46       });
 47   }
 48 
 49   private static synchronized void initialize(TypeDataBase db) {
 50     Type type = db.lookupType("CollectedHeap");
 51 
 52     reservedFieldOffset = type.getField("_reserved").getOffset();
 53   }
 54 
 55   public CollectedHeap(Address addr) {
 56     super(addr);
 57   }
 58 
 59   /** Returns the lowest address of the heap. */
 60   public Address start() {
 61     return reservedRegion().start();
 62   }
 63 
 64   public abstract long capacity();
 65   public abstract long used();
 66 
 67   public MemRegion reservedRegion() {
 68     return new MemRegion(addr.addOffsetTo(reservedFieldOffset));
 69   }
 70 
 71   public boolean isIn(Address a) {
 72     return isInReserved(a);
 73   }
 74 
 75   public boolean isInReserved(Address a) {
 76     return reservedRegion().contains(a);
 77   }
 78 
 79   public abstract CollectedHeapName kind();
 80 
 81   public abstract void liveRegionsIterate(LiveRegionsClosure closure);
 82 
 83   public String oopAddressDescription(OopHandle handle) {
 84       return handle.toString();
 85   }
 86 
 87   public OopHandle oop_load_at(OopHandle handle, long offset) {
 88       return handle.getOopHandleAt(offset);
 89   }
 90 
 91   public OopHandle oop_load_in_native(Address addr) {
 92       return addr.getOopHandleAt(0);
 93   }
 94 
 95   public void print() { printOn(System.out); }
 96   public void printOn(PrintStream tty) {
 97     MemRegion mr = reservedRegion();
 98     tty.println("unknown subtype of CollectedHeap @ " + getAddress() + " (" +
 99                 mr.start() + "," + mr.end() + ")");
100   }
101 
102   public BitMapInterface createBitMap(long bits) {
103     return new BitMapSegmented(bits);
104   }
105 }
