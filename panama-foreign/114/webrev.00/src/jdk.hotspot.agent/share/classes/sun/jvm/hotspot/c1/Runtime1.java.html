25 package sun.jvm.hotspot.c1;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.code.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.Observable;
33 import sun.jvm.hotspot.utilities.Observer;
34 
35 /** Currently a minimal port to get C1 frame traversal working */
36 
37 public class Runtime1 {
38   private static Field         blobsField;
39 
40   static {
41     VM.registerVMInitializedObserver(new Observer() {
42         public void update(Observable o, Object data) {
43           initialize(VM.getVM().getTypeDataBase());
44         }
45       });
46   }
47 
48   private static synchronized void initialize(TypeDataBase db) {
49     Type type = db.lookupType("Runtime1");
50 
51     blobsField = type.getField("_blobs");
52   }
53 
54   public Runtime1() {
55   }
56 
57   /** FIXME: consider making argument "type-safe" in Java port */
58   public Address entryFor(int id) {
59     return blobFor(id).codeBegin();
60   }
61 
62   /** FIXME: consider making argument "type-safe" in Java port */
63   public CodeBlob blobFor(int id) {
64     Address blobAddr = blobsField.getStaticFieldAddress().getAddressAt(id * VM.getVM().getAddressSize());
65     return VM.getVM().getCodeCache().createCodeBlobWrapper(blobAddr);
66   }
67 }
