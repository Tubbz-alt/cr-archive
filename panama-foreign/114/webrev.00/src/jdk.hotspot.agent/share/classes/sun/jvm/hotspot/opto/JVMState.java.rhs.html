 25 package sun.jvm.hotspot.opto;
 26 
 27 import java.io.PrintStream;
 28 import java.util.*;
 29 import sun.jvm.hotspot.debugger.*;
 30 import sun.jvm.hotspot.runtime.*;
 31 import sun.jvm.hotspot.oops.*;
 32 import sun.jvm.hotspot.ci.*;
 33 import sun.jvm.hotspot.types.*;
 34 import sun.jvm.hotspot.utilities.Observable;
 35 import sun.jvm.hotspot.utilities.Observer;
 36 
 37 public class JVMState extends VMObject {
 38   static {
 39     VM.registerVMInitializedObserver(new Observer() {
 40         public void update(Observable o, Object data) {
 41           initialize(VM.getVM().getTypeDataBase());
 42         }
 43       });
 44   }
 45 
 46   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
 47     Type type      = db.lookupType("JVMState");
 48     mapField = type.getAddressField("_map");
 49     methodField = type.getAddressField("_method");
 50     bciField = new CIntField(type.getCIntegerField("_bci"), 0);
 51     spField = new CIntField(type.getCIntegerField("_sp"), 0);
 52     endoffField = new CIntField(type.getCIntegerField("_endoff"), 0);
 53     try {
 54         scloffField = new CIntField(type.getCIntegerField("_scloff"), 0);
 55     } catch (Exception e) {
 56     }
 57     monoffField = new CIntField(type.getCIntegerField("_monoff"), 0);
 58     stkoffField = new CIntField(type.getCIntegerField("_stkoff"), 0);
 59     locoffField = new CIntField(type.getCIntegerField("_locoff"), 0);
 60     depthField = new CIntField(type.getCIntegerField("_depth"), 0);
 61     callerField = type.getAddressField("_caller");
 62   }
 63 
 64   private static AddressField mapField;
 65   private static AddressField methodField;
 66   private static CIntField bciField;
 67   private static CIntField spField;
 68   private static CIntField endoffField;
 69   private static CIntField scloffField;
 70   private static CIntField monoffField;
 71   private static CIntField stkoffField;
 72   private static CIntField locoffField;
 73   private static CIntField depthField;
 74   private static AddressField callerField;
 75 
 76   public static JVMState create(Address addr) {
 77     if (addr == null) return null;
 78     return new JVMState(addr);
 79   }
 80 
 81   public JVMState(Address addr) {
 82     super(addr);
 83   }
 84 
 85   public ciMethod method() {
 86     return (ciMethod) ciObjectFactory.getMetadata(methodField.getValue(getAddress()));
 87   }
 88 
 89   public int bci() {
 90     return (int)bciField.getValue(getAddress());
 91   }
 92 
 93   public int depth() {
 94     return (int)depthField.getValue(getAddress());
 95   }
 96 
 97   public JVMState caller() {
 98     return create(callerField.getValue(getAddress()));
 99   }
100 
101   public void dumpSpec(PrintStream out) {
102     ciMethod m = method();
103     if (m != null) {
104       Method meth = m.method();
105       out.print(" " + meth.getMethodHolder().getName().asString().replace('/', '.') + "::" +
106                 meth.getName().asString() + " @ bci:" + bci());
107     } else {
108       out.print(" runtime stub");
109     }
110     if (caller() != null)  caller().dumpSpec(out);
111   }
112 }
