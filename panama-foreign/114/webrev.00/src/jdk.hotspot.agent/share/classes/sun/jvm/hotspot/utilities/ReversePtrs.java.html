25 package sun.jvm.hotspot.utilities;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.oops.*;
29 
30 /** ReversePtrs hashtable. */
31 
32 public class ReversePtrs  {
33   private HashMap<Oop, ArrayList<LivenessPathElement>> rp;
34 
35   public ReversePtrs() {
36     rp = new HashMap<>();
37   }
38 
39   public void put(LivenessPathElement from, Oop to) {
40     if (to == null) return;
41     ArrayList<LivenessPathElement> al = rp.get(to);
42     if (al == null) al = new ArrayList<>();
43     // Inserting at the beginning is a hack to change the reported
44     // paths from LivenessAnalysis to look more like they used to;
45     // otherwise paths through the Finalizer queue to popular objects
46     // seem to be preferred
47     al.add(0, from);
48     rp.put(to, al);
49   }
50 
51   /** Returns an ArrayList of the incoming references to this Oop if
52       it is alive, and null if it is dead according to the
53       ReversePtrsAnalysis. Currently not all roots are scanned so this
54       result is frequently inaccurate for JVM-internal objects, but is
55       usually correct for Java-level objects. */
56   public ArrayList<LivenessPathElement> get(Oop obj) {
57     return rp.get(obj);
58   }
59 }
