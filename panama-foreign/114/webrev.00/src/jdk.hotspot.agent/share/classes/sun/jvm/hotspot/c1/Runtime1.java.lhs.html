25 package sun.jvm.hotspot.c1;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.code.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;


32 
33 /** Currently a minimal port to get C1 frame traversal working */
34 
35 public class Runtime1 {
36   private static Field         blobsField;
37 
38   static {
39     VM.registerVMInitializedObserver(new Observer() {
40         public void update(Observable o, Object data) {
41           initialize(VM.getVM().getTypeDataBase());
42         }
43       });
44   }
45 
46   private static synchronized void initialize(TypeDataBase db) {
47     Type type = db.lookupType("Runtime1");
48 
49     blobsField = type.getField("_blobs");
50   }
51 
52   public Runtime1() {
53   }
54 
55   /** FIXME: consider making argument "type-safe" in Java port */
56   public Address entryFor(int id) {
57     return blobFor(id).codeBegin();
58   }
59 
60   /** FIXME: consider making argument "type-safe" in Java port */
61   public CodeBlob blobFor(int id) {
62     Address blobAddr = blobsField.getStaticFieldAddress().getAddressAt(id * VM.getVM().getAddressSize());
63     return VM.getVM().getCodeCache().createCodeBlobWrapper(blobAddr);
64   }
65 }
