25 package sun.jvm.hotspot.memory;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 import sun.jvm.hotspot.runtime.*;
30 import sun.jvm.hotspot.types.*;
31 
32 public class MetaspaceObj {
33   private static Address sharedMetaspaceBaseAddr;
34   private static Address sharedMetaspaceTopAddr;
35 
36   static {
37     VM.registerVMInitializedObserver(new Observer() {
38         public void update(Observable o, Object data) {
39           initialize(VM.getVM().getTypeDataBase());
40         }
41       });
42   }
43 
44   private static synchronized void initialize(TypeDataBase db) {
45     Type type = db.lookupType("MetaspaceObj");
46     sharedMetaspaceBaseAddr = type.getAddressField("_shared_metaspace_base").getStaticFieldAddress();
47     sharedMetaspaceTopAddr  = type.getAddressField("_shared_metaspace_top").getStaticFieldAddress();
48   }
49 
50   public static boolean isShared(Address addr) {
51     Address base = sharedMetaspaceBaseAddr.getAddressAt(0);
52     Address top  = sharedMetaspaceTopAddr. getAddressAt(0);
53 
54     return base.lessThanOrEqual(addr) && addr.lessThan(top);
55   }
56 }
