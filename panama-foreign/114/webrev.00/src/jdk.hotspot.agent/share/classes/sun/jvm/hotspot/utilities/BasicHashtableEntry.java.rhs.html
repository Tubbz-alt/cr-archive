25 package sun.jvm.hotspot.utilities;
26 
27 import java.util.*;
28 import sun.jvm.hotspot.debugger.*;
29 import sun.jvm.hotspot.oops.Oop;
30 import sun.jvm.hotspot.oops.ObjectHeap;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.runtime.*;
33 import sun.jvm.hotspot.utilities.Observable;
34 import sun.jvm.hotspot.utilities.Observer;
35 
36 public class BasicHashtableEntry extends VMObject {
37   static {
38     VM.registerVMInitializedObserver(new Observer() {
39         public void update(Observable o, Object data) {
40           initialize(VM.getVM().getTypeDataBase());
41         }
42       });
43   }
44 
45   private static synchronized void initialize(TypeDataBase db) {
46     Type type = db.lookupType("BasicHashtableEntry<mtInternal>");
47     hashField      = type.getCIntegerField("_hash");
48     nextField      = type.getAddressField("_next");
49   }
50 
51   // Fields
52   private static CIntegerField hashField;
53   private static AddressField  nextField;
54 
55   // Accessors
56   public long hash() {
57     return hashField.getValue(addr) & 0xFFFFFFFFL;
58   }
59 
60   private long nextAddressValue() {
61     Debugger dbg = VM.getVM().getDebugger();
62     Address nextValue = nextField.getValue(addr);
63     return (nextValue != null) ? dbg.getAddressValue(nextValue) : 0L;
64   }
65 
66   public boolean isShared() {
67     return (nextAddressValue() & 1L) != 0;
68   }
69 
70   public BasicHashtableEntry next() {
71     Address nextValue = nextField.getValue(addr);
72     Address next = (nextValue != null)? nextValue.andWithMask(-2L) : null;
73     // using this.getClass so that return type will be as expected in
74     // subclass context. But, because we can't use covariant return type
75     // caller has to use this next and cast the result to correct type.
76     return (BasicHashtableEntry) VMObjectFactory.newObject(this.getClass(), next);
77   }
78 
79   public BasicHashtableEntry(Address addr) {
80     super(addr);
81   }
82 }
