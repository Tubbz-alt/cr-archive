 25 package sun.jvm.hotspot.memory;
 26 
 27 import java.util.*;
 28 import sun.jvm.hotspot.debugger.*;
 29 import sun.jvm.hotspot.classfile.*;
 30 import sun.jvm.hotspot.oops.*;
 31 import sun.jvm.hotspot.runtime.*;
 32 import sun.jvm.hotspot.types.*;
 33 import sun.jvm.hotspot.utilities.Observable;
 34 import sun.jvm.hotspot.utilities.Observer;
 35 
 36 public class SystemDictionary {
 37   private static sun.jvm.hotspot.types.OopField javaSystemLoaderField;
 38 
 39   private static AddressField objectKlassField;
 40   private static AddressField classLoaderKlassField;
 41   private static AddressField stringKlassField;
 42   private static AddressField systemKlassField;
 43   private static AddressField threadKlassField;
 44   private static AddressField threadGroupKlassField;
 45   private static AddressField methodHandleKlassField;
 46 
 47   static {
 48     VM.registerVMInitializedObserver(new Observer() {
 49         public void update(Observable o, Object data) {
 50           initialize(VM.getVM().getTypeDataBase());
 51         }
 52       });
 53   }
 54 
 55   private static synchronized void initialize(TypeDataBase db) {
 56     Type type = db.lookupType("SystemDictionary");
 57 
 58     javaSystemLoaderField = type.getOopField("_java_system_loader");
 59 
 60     objectKlassField = type.getAddressField(WK_KLASS("Object_klass"));
 61     classLoaderKlassField = type.getAddressField(WK_KLASS("ClassLoader_klass"));
 62     stringKlassField = type.getAddressField(WK_KLASS("String_klass"));
 63     systemKlassField = type.getAddressField(WK_KLASS("System_klass"));
 64     threadKlassField = type.getAddressField(WK_KLASS("Thread_klass"));
 65     threadGroupKlassField = type.getAddressField(WK_KLASS("ThreadGroup_klass"));
 66     methodHandleKlassField = type.getAddressField(WK_KLASS("MethodHandle_klass"));
 67   }
 68 
 69   // This WK functions must follow the definitions in systemDictionary.hpp:
 70   private static String WK_KLASS(String name) {
 71       //#define WK_KLASS(name) _well_known_klasses[SystemDictionary::WK_KLASS_ENUM_NAME(name)]
 72       return ("_well_known_klasses[SystemDictionary::"+WK_KLASS_ENUM_NAME(name)+"]");
 73   }
 74   private static String WK_KLASS_ENUM_NAME(String kname) {
 75       //#define WK_KLASS_ENUM_NAME(kname)    kname##_knum
 76       return (kname+"_knum");
 77   }
 78 
 79   // few well known classes -- not all are added here.
 80   // add more if needed.
 81   public static InstanceKlass getThreadKlass() {
 82     return (InstanceKlass)Metadata.instantiateWrapperFor(threadKlassField.getValue());
 83   }
 84 
 85   public static InstanceKlass getThreadGroupKlass() {
 86     return (InstanceKlass)Metadata.instantiateWrapperFor(threadGroupKlassField.getValue());
 87   }
 88 
 89   public static InstanceKlass getObjectKlass() {
 90     return (InstanceKlass)Metadata.instantiateWrapperFor(objectKlassField.getValue());
 91   }
 92 
 93   public static InstanceKlass getStringKlass() {
 94     return (InstanceKlass)Metadata.instantiateWrapperFor(stringKlassField.getValue());
 95   }
 96 
 97   public static InstanceKlass getClassLoaderKlass() {
 98     return (InstanceKlass)Metadata.instantiateWrapperFor(classLoaderKlassField.getValue());
 99   }
100 
101   public static InstanceKlass getSystemKlass() {
102     return (InstanceKlass)Metadata.instantiateWrapperFor(systemKlassField.getValue());
103   }
104 
105   public static InstanceKlass getMethodHandleKlass() {
106     return (InstanceKlass)Metadata.instantiateWrapperFor(methodHandleKlassField.getValue());
107   }
108 
109   public InstanceKlass getAbstractOwnableSynchronizerKlass() {
110     ClassLoaderDataGraph cldg = VM.getVM().getClassLoaderDataGraph();
111     return (InstanceKlass) cldg.find("java/util/concurrent/locks/AbstractOwnableSynchronizer");
112   }
113 
114   public static Oop javaSystemLoader() {
115     return newOop(javaSystemLoaderField.getValue());
116   }
117 
118   private static Oop newOop(OopHandle handle) {
119     return VM.getVM().getObjectHeap().newOop(handle);
120   }
121 }
