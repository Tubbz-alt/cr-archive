25 package sun.jvm.hotspot.oops;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.interpreter.*;
31 import sun.jvm.hotspot.runtime.*;
32 import sun.jvm.hotspot.types.*;
33 import sun.jvm.hotspot.utilities.*;


34 
35 public class LocalVariableTableElement {
36   static {
37     VM.registerVMInitializedObserver(new Observer() {
38         public void update(Observable o, Object data) {
39           initialize(VM.getVM().getTypeDataBase());
40         }
41       });
42   }
43 
44   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
45     Type type                 = db.lookupType("LocalVariableTableElement");
46     offsetOfStartBCI          = type.getCIntegerField("start_bci").getOffset();
47     offsetOfLength            = type.getCIntegerField("length").getOffset();
48     offsetOfNameCPIndex       = type.getCIntegerField("name_cp_index").getOffset();
49     offsetOfDescriptorCPIndex = type.getCIntegerField("descriptor_cp_index").getOffset();
50     offsetOfSignatureCPIndex  = type.getCIntegerField("signature_cp_index").getOffset();
51     offsetOfSlot              = type.getCIntegerField("slot").getOffset();
52   }
53 
54   private static long offsetOfStartBCI;
55   private static long offsetOfLength;
56   private static long offsetOfNameCPIndex;
57   private static long offsetOfDescriptorCPIndex;
58   private static long offsetOfSignatureCPIndex;
59   private static long offsetOfSlot;
60 
61   private Address   handle;
62   private long      offset;
63 
64   public LocalVariableTableElement(Address handle, long offset) {
65     this.handle = handle;
66     this.offset = offset;
67   }
68 
69   public int getStartBCI() {
70     return (int) handle.getCIntegerAt(offset + offsetOfStartBCI, 2, true);
71   }
72 
73   public int getLength() {
74     return (int) handle.getCIntegerAt(offset + offsetOfLength, 2, true);
75   }
76 
77   public int getNameCPIndex() {
78     return (int) handle.getCIntegerAt(offset + offsetOfNameCPIndex, 2, true);
79   }
80 
81   public int getDescriptorCPIndex() {
82     return (int) handle.getCIntegerAt(offset + offsetOfDescriptorCPIndex, 2, true);
83   }
84 
85   public int getSignatureCPIndex() {
86     return (int) handle.getCIntegerAt(offset + offsetOfSignatureCPIndex, 2, true);
87   }
88 
89   public int getSlot() {
90     return (int) handle.getCIntegerAt(offset + offsetOfSlot, 2, true);
91   }
92 }
