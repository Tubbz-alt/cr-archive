25 package sun.jvm.hotspot.opto;
26 
27 import java.util.*;
28 import java.io.PrintStream;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.oops.*;
32 import sun.jvm.hotspot.types.*;
33 import sun.jvm.hotspot.utilities.Observable;
34 import sun.jvm.hotspot.utilities.Observer;
35 
36 public class PhaseCFG extends Phase {
37   static {
38     VM.registerVMInitializedObserver(new Observer() {
39         public void update(Observable o, Object data) {
40           initialize(VM.getVM().getTypeDataBase());
41         }
42       });
43   }
44 
45   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
46     Type type      = db.lookupType("PhaseCFG");
47     numBlocksField = new CIntField(type.getCIntegerField("_number_of_blocks"), 0);
48     blocksField = type.getAddressField("_blocks");
49     bbsField = type.getAddressField("_node_to_block_mapping");
50     brootField = type.getAddressField("_root_block");
51   }
52 
53   private static CIntField numBlocksField;
54   private static AddressField blocksField;
55   private static AddressField bbsField;
56   private static AddressField brootField;
57 
58   public PhaseCFG(Address addr) {
59     super(addr);
60   }
61 
62   public void dump(PrintStream out) {
63     int addressSize = (int)VM.getVM().getAddressSize();
64     int numBlocks = (int)numBlocksField.getValue(getAddress());
65     Block_List blocks = new Block_List(getAddress().addOffsetTo(blocksField.getOffset()));
66     int offset = 0;
67     for (int i  = 0; i < numBlocks; i++) {
68       blocks.at(i).dump(out);
69     }
70   }
71 }
