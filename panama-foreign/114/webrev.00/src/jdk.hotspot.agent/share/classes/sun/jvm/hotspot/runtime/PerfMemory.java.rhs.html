 25 package sun.jvm.hotspot.runtime;
 26 
 27 import java.util.*;
 28 import sun.jvm.hotspot.debugger.*;
 29 import sun.jvm.hotspot.oops.*;
 30 import sun.jvm.hotspot.types.*;
 31 import sun.jvm.hotspot.utilities.Observable;
 32 import sun.jvm.hotspot.utilities.Observer;
 33 
 34 public class PerfMemory {
 35     private static AddressField  startField;
 36     private static AddressField  endField;
 37     private static AddressField  topField;
 38     private static CIntegerField capacityField;
 39     private static AddressField  prologueField;
 40     private static CIntegerField initializedField;
 41 
 42     static {
 43         VM.registerVMInitializedObserver(new Observer() {
 44                 public void update(Observable o, Object data) {
 45                     initialize(VM.getVM().getTypeDataBase());
 46                 }
 47             });
 48     }
 49 
 50     private static synchronized void initialize(TypeDataBase db) {
 51         Type type = db.lookupType("PerfMemory");
 52         startField = type.getAddressField("_start");
 53         endField = type.getAddressField("_end");
 54         topField = type.getAddressField("_top");
 55         capacityField = type.getCIntegerField("_capacity");
 56         prologueField = type.getAddressField("_prologue");
 57         initializedField = type.getCIntegerField("_initialized");
 58     }
 59 
 60     // Accessors
 61     public static Address start() {
 62         return startField.getValue();
 63     }
 64 
 65     public static Address end() {
 66         return endField.getValue();
 67     }
 68 
 69     public static Address top() {
 70         return topField.getValue();
 71     }
 72 
 73     public static long capacity() {
 74         return capacityField.getValue();
 75     }
 76 
 77     public static boolean initialized() {
 78         return (initializedField.getValue()) != 0;
 79     }
 80 
 81     public static PerfDataPrologue prologue() {
 82         return (PerfDataPrologue) VMObjectFactory.newObject(
 83                    PerfDataPrologue.class, prologueField.getValue());
 84     }
 85 
 86     public static boolean contains(Address addr) {
 87         return start() != null &&
 88             addr.minus(start()) >= 0 &&
 89             end().minus(addr) > 0;
 90     }
 91 
 92     // an interface supplied to iterate PerfDataEntries
 93     public static interface PerfDataEntryVisitor {
 94         // returns false to stop the iteration
 95         public boolean visit(PerfDataEntry pde);
 96     }
 97 
 98     public static void iterate(PerfDataEntryVisitor visitor) {
 99         PerfDataPrologue header = prologue();
100         int off = header.entryOffset();
101         int num = header.numEntries();
102         Address addr = header.getAddress();
103 
104         for (int i = 0; i < num; i++) {
105             PerfDataEntry pde = (PerfDataEntry) VMObjectFactory.newObject(
106                                PerfDataEntry.class, addr.addOffsetTo(off));
107             off += pde.entryLength();
108             if (visitor.visit(pde) == false) return;
109         }
110     }
111 }
