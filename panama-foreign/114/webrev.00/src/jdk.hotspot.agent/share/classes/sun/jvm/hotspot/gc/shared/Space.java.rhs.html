 25 package sun.jvm.hotspot.gc.shared;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 import sun.jvm.hotspot.debugger.*;
 30 import sun.jvm.hotspot.memory.*;
 31 import sun.jvm.hotspot.runtime.*;
 32 import sun.jvm.hotspot.types.*;
 33 import sun.jvm.hotspot.utilities.Observable;
 34 import sun.jvm.hotspot.utilities.Observer;
 35 
 36 /** <P> A Space describes a heap area. Class Space is an abstract base
 37     class. </P>
 38 
 39     <P> Space supports allocation, size computation and GC support is
 40     provided. </P>
 41 
 42     <P> Invariant: bottom() and end() are on page_size boundaries and: </P>
 43 
 44     <P> bottom() <= top() <= end() </P>
 45 
 46     <P> top() is inclusive and end() is exclusive. </P> */
 47 
 48 public abstract class Space extends VMObject {
 49   private static AddressField bottomField;
 50   private static AddressField endField;
 51 
 52   static {
 53     VM.registerVMInitializedObserver(new Observer() {
 54         public void update(Observable o, Object data) {
 55           initialize(VM.getVM().getTypeDataBase());
 56         }
 57       });
 58   }
 59 
 60   private static synchronized void initialize(TypeDataBase db) {
 61     Type type = db.lookupType("Space");
 62 
 63     bottomField = type.getAddressField("_bottom");
 64     endField    = type.getAddressField("_end");
 65   }
 66 
 67   public Space(Address addr) {
 68     super(addr);
 69   }
 70 
 71   public Address   bottom()       { return bottomField.getValue(addr); }
 72   public Address   end()          { return endField.getValue(addr);    }
 73 
 74   /** Returns a subregion of the space containing all the objects in
 75       the space. */
 76   public MemRegion usedRegion() {
 77     return new MemRegion(bottom(), end());
 78   }
 79 
 80   /** Support for iteration over heap -- not sure how this will
 81       interact with GC in reflective system, but necessary for the
 82       debugging mechanism */
 83   public OopHandle bottomAsOopHandle() {
 84     return bottomField.getOopHandle(addr);
 85   }
 86 
 87   /** Support for iteration over heap -- not sure how this will
 88       interact with GC in reflective system, but necessary for the
 89       debugging mechanism */
 90   public OopHandle nextOopHandle(OopHandle handle, long size) {
 91     return handle.addOffsetToAsOopHandle(size);
 92   }
 93 
 94   /** Returned value is in bytes */
 95   public long capacity() { return end().minus(bottom()); }
 96   /** Returned value is in bytes */
 97   public abstract long used();
 98   /** Returned value is in bytes */
 99   public abstract long free();
100 
101   /** Testers */
102   public boolean contains(Address p) {
103     return (bottom().lessThanOrEqual(p) && end().greaterThan(p));
104   }
105 
106   public void print() { printOn(System.out); }
107   public void printOn(PrintStream tty) {
108     tty.print(" space capacity = ");
109     tty.print(capacity());
110     tty.print(", ");
111     tty.print((double) used() * 100.0/ capacity());
112     tty.print(" used");
113   }
114 }
