 50 /**
 51  * This base class encapsulates many of the events that are fired from
 52  * the various panels in this directory so they can easily be plugged
 53  * in to different containing frameworks (HSDB).
 54  */
 55 public class SAPanel extends JPanel {
 56     protected List listeners = new ArrayList();
 57 
 58 
 59     public SAPanel() {
 60     }
 61 
 62     public void addPanelListener(SAListener listener) {
 63         listeners.add(listener);
 64     }
 65 
 66     public void removePanelListener(SAListener listener) {
 67         listeners.remove(listener);
 68     }
 69 
 70     public void showThreadOopInspector(JavaThread t) {
 71         for (Iterator iter = listeners.iterator(); iter.hasNext(); ) {
 72             SAListener listener = (SAListener) iter.next();
 73             listener.showThreadOopInspector(t);
 74         }
 75     }
