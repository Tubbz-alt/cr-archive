25 package sun.jvm.hotspot.tools.jcore;
26 
27 import sun.jvm.hotspot.oops.*;
28 import java.util.*;
29 
30 public class PackageNameFilter implements ClassFilter
31 {
32     public Object[] pkgList;
33 
34     public PackageNameFilter() {
35         // give comma separated list of package names to include
36         this(System.getProperty("sun.jvm.hotspot.tools.jcore.PackageNameFilter.pkgList"));
37     }
38 
39     public PackageNameFilter(String pattern) {
40         try {
41             StringTokenizer st = new StringTokenizer(pattern, ",");
42             List l = new LinkedList();
43             while (st.hasMoreTokens()) {
44                 l.add(st.nextToken());
45             }
46             pkgList = l.toArray();
47         } catch (Exception exp) {
48            exp.printStackTrace();
49         }
50     }
51 
52     public boolean canInclude(InstanceKlass kls) {
53         if (pkgList == null) {
54             // Dump everything
55             return true;
56         }
57         final int len = pkgList.length;
58         if (len == 0)
59             return true;
60         String klassName = kls.getName().asString().replace('/', '.');
61         for (int i=0; i < len; i++)
62             if (klassName.startsWith((String) pkgList[i] )) return true;
63         return false;
64     }
65 }
