25 package sun.jvm.hotspot.oops;
26 
27 import java.io.*;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.Observable;
33 import sun.jvm.hotspot.utilities.Observer;
34 
35 public class CompiledICHolder extends VMObject {
36   static {
37     VM.registerVMInitializedObserver(new Observer() {
38         public void update(Observable o, Object data) {
39           initialize(VM.getVM().getTypeDataBase());
40         }
41       });
42   }
43 
44   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
45     Type type      = db.lookupType("CompiledICHolder");
46     holderMetadata = new MetadataField(type.getAddressField("_holder_metadata"), 0);
47     holderKlass    = new MetadataField(type.getAddressField("_holder_klass"), 0);
48     headerSize     = type.getSize();
49   }
50 
51   public CompiledICHolder(Address addr) {
52       super(addr);
53   }
54 
55   public boolean isCompiledICHolder()  { return true; }
56 
57   private static long headerSize;
58 
59   // Fields
60   private static MetadataField holderMetadata;
61   private static MetadataField holderKlass;
62 
63   // Accessors for declared fields
64   public Metadata getHolderMetadata() { return (Metadata) holderMetadata.getValue(this); }
65   public Klass    getHolderKlass()    { return (Klass)    holderKlass.getValue(this); }
66 
67   public void printValueOn(PrintStream tty) {
68     tty.print("CompiledICHolder");
69   }
70   }
