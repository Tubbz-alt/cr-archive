25 package sun.jvm.hotspot.gc.shared;
26 
27 import java.util.*;
28 
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.types.*;
32 import sun.jvm.hotspot.utilities.Observable;
33 import sun.jvm.hotspot.utilities.Observer;
34 
35 public class GenerationSpec extends VMObject {
36   private static CIntegerField nameField;
37   private static CIntegerField initSizeField;
38   private static CIntegerField maxSizeField;
39 
40   static {
41     VM.registerVMInitializedObserver(new Observer() {
42         public void update(Observable o, Object data) {
43           initialize(VM.getVM().getTypeDataBase());
44         }
45       });
46   }
47 
48   private static synchronized void initialize(TypeDataBase db) {
49     Type type = db.lookupType("GenerationSpec");
50 
51     nameField       = type.getCIntegerField("_name");
52     initSizeField = type.getCIntegerField("_init_size");
53     maxSizeField  = type.getCIntegerField("_max_size");
54   }
55 
56   public GenerationSpec(Address addr) {
57     super(addr);
58   }
59 
60   public Generation.Name name() {
61     return Generation.nameForEnum((int)nameField.getValue(addr));
62   }
63 
64   public long initSize() {
65     return initSizeField.getValue(addr);
66   }
67 
68   public long maxSize() {
69     return maxSizeField.getValue(addr);
70   }
71 }
