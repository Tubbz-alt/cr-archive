25 package sun.jvm.hotspot.compiler;
26 
27 import java.io.PrintStream;
28 import java.util.*;
29 import sun.jvm.hotspot.debugger.*;
30 import sun.jvm.hotspot.runtime.*;
31 import sun.jvm.hotspot.oops.*;
32 import sun.jvm.hotspot.opto.*;
33 import sun.jvm.hotspot.prims.JvmtiExport;
34 import sun.jvm.hotspot.types.*;
35 import sun.jvm.hotspot.utilities.Observable;
36 import sun.jvm.hotspot.utilities.Observer;
37 
38 public class CompileTask extends VMObject {
39   static {
40     VM.registerVMInitializedObserver(new Observer() {
41         public void update(Observable o, Object data) {
42           initialize(VM.getVM().getTypeDataBase());
43         }
44       });
45   }
46 
47   private static synchronized void initialize(TypeDataBase db) throws WrongTypeException {
48     Type type      = db.lookupType("CompileTask");
49     methodField = type.getAddressField("_method");
50     osrBciField = new CIntField(type.getCIntegerField("_osr_bci"), 0);
51     compLevelField = new CIntField(type.getCIntegerField("_comp_level"), 0);
52   }
53 
54   private static AddressField methodField;
55   private static CIntField osrBciField;
56   private static CIntField compLevelField;
57 
58   public CompileTask(Address addr) {
59     super(addr);
60   }
61 
62   public Method method() {
63     Address oh =  methodField.getValue(getAddress());
64     return (Method)Metadata.instantiateWrapperFor(oh);
65   }
66 
67   public int osrBci() {
68     return (int)osrBciField.getValue(getAddress());
69   }
70 
71   public int compLevel() {
72       return (int)compLevelField.getValue(getAddress());
73   }
74 }
