 25 package sun.jvm.hotspot.runtime;
 26 
 27 import java.util.*;
 28 import sun.jvm.hotspot.debugger.*;
 29 import sun.jvm.hotspot.oops.*;
 30 import sun.jvm.hotspot.types.*;
 31 import sun.jvm.hotspot.utilities.Observable;
 32 import sun.jvm.hotspot.utilities.Observer;
 33 
 34 public class PerfDataPrologue extends VMObject {
 35     private static JIntField  magicField;
 36     private static JByteField byteOrderField;
 37     private static JByteField majorVersionField;
 38     private static JByteField minorVersionField;
 39     private static JByteField accessibleField;
 40     private static JIntField  usedField;
 41     private static JIntField  overflowField;
 42     private static JLongField modTimeStampField;
 43     private static JIntField  entryOffsetField;
 44     private static JIntField  numEntriesField;
 45 
 46     static {
 47         VM.registerVMInitializedObserver(new Observer() {
 48                 public void update(Observable o, Object data) {
 49                     initialize(VM.getVM().getTypeDataBase());
 50                 }
 51             });
 52     }
 53 
 54     private static synchronized void initialize(TypeDataBase db) {
 55         Type type = db.lookupType("PerfDataPrologue");
 56         magicField = type.getJIntField("magic");
 57         byteOrderField = type.getJByteField("byte_order");
 58         majorVersionField = type.getJByteField("major_version");
 59         minorVersionField = type.getJByteField("minor_version");
 60         accessibleField = type.getJByteField("accessible");
 61         usedField = type.getJIntField("used");
 62         overflowField = type.getJIntField("overflow");
 63         modTimeStampField = type.getJLongField("mod_time_stamp");
 64         entryOffsetField = type.getJIntField("entry_offset");
 65         numEntriesField = type.getJIntField("num_entries");
 66     }
 67 
 68     public PerfDataPrologue(Address addr) {
 69         super(addr);
 70     }
 71 
 72     // Accessors
 73     public int magic() {
 74         return (int) magicField.getValue(addr);
 75     }
 76 
 77     public byte byteOrder() {
 78         return (byte) byteOrderField.getValue(addr);
 79     }
 80 
 81     public byte majorVersion() {
 82         return (byte) majorVersionField.getValue(addr);
 83     }
 84 
 85     public boolean accessible() {
 86         return ((byte) accessibleField.getValue(addr)) != (byte)0;
 87     }
 88 
 89     public int used() {
 90         return (int) usedField.getValue(addr);
 91     }
 92 
 93     public int overflow() {
 94         return (int) overflowField.getValue(addr);
 95     }
 96 
 97     public long modTimeStamp() {
 98         return (long) modTimeStampField.getValue(addr);
 99     }
100 
101     public int entryOffset() {
102         return (int) entryOffsetField.getValue(addr);
103     }
104 
105     public int numEntries() {
106         return (int) numEntriesField.getValue(addr);
107     }
108 }
