 25 package sun.jvm.hotspot.runtime.linux_amd64;
 26 
 27 import java.io.*;
 28 import java.util.*;
 29 import sun.jvm.hotspot.debugger.*;
 30 import sun.jvm.hotspot.debugger.amd64.*;
 31 import sun.jvm.hotspot.runtime.*;
 32 import sun.jvm.hotspot.runtime.amd64.*;
 33 import sun.jvm.hotspot.runtime.x86.*;
 34 import sun.jvm.hotspot.types.*;
 35 import sun.jvm.hotspot.utilities.*;
 36 import sun.jvm.hotspot.utilities.Observable;
 37 import sun.jvm.hotspot.utilities.Observer;
 38 
 39 public class LinuxAMD64JavaThreadPDAccess implements JavaThreadPDAccess {
 40   private static AddressField  lastJavaFPField;
 41   private static AddressField  osThreadField;
 42 
 43   // Field from OSThread
 44   private static CIntegerField osThreadThreadIDField;
 45 
 46   // This is currently unneeded but is being kept in case we change
 47   // the currentFrameGuess algorithm
 48   private static final long GUESS_SCAN_RANGE = 128 * 1024;
 49 
 50   static {
 51     VM.registerVMInitializedObserver(new Observer() {
 52         public void update(Observable o, Object data) {
 53           initialize(VM.getVM().getTypeDataBase());
 54         }
 55       });
 56   }
 57 
 58   private static synchronized void initialize(TypeDataBase db) {
 59     Type type = db.lookupType("JavaThread");
 60     osThreadField           = type.getAddressField("_osthread");
 61 
 62     Type anchorType = db.lookupType("JavaFrameAnchor");
 63     lastJavaFPField         = anchorType.getAddressField("_last_Java_fp");
 64 
 65     Type osThreadType = db.lookupType("OSThread");
 66     osThreadThreadIDField   = osThreadType.getCIntegerField("_thread_id");
 67   }
 68 
 69   public    Address getLastJavaFP(Address addr) {
 70     return lastJavaFPField.getValue(addr.addOffsetTo(sun.jvm.hotspot.runtime.JavaThread.getAnchorField().getOffset()));
 71   }
 72 
 73   public    Address getLastJavaPC(Address addr) {
 74     return null;
 75   }
 76 
 77   public    Address getBaseOfStackPointer(Address addr) {
 78     return null;
 79   }
 80 
 81   public    Frame getLastFramePD(JavaThread thread, Address addr) {
 82     Address fp = thread.getLastJavaFP();
 83     if (fp == null) {
 84       return null; // no information
 85     }
 86     return new X86Frame(thread.getLastJavaSP(), fp);
 87   }
 88 
 89   public    RegisterMap newRegisterMap(JavaThread thread, boolean updateMap) {
 90     return new X86RegisterMap(thread, updateMap);
 91   }
 92 
 93   public    Frame getCurrentFrameGuess(JavaThread thread, Address addr) {
 94     ThreadProxy t = getThreadProxy(addr);
 95     AMD64ThreadContext context = (AMD64ThreadContext) t.getContext();
 96     AMD64CurrentFrameGuess guesser = new AMD64CurrentFrameGuess(context, thread);
 97     if (!guesser.run(GUESS_SCAN_RANGE)) {
 98       return null;
 99     }
100     if (guesser.getPC() == null) {
101       return new X86Frame(guesser.getSP(), guesser.getFP());
102     } else if (VM.getVM().getInterpreter().contains(guesser.getPC())) {
103       // pass the value of R13 which contains the bcp for the top level frame
104       Address bcp = context.getRegisterAsAddress(AMD64ThreadContext.R13);
105       return new X86Frame(guesser.getSP(), guesser.getFP(), guesser.getPC(), null, bcp);
106     } else {
107       return new X86Frame(guesser.getSP(), guesser.getFP(), guesser.getPC());
108     }
109   }
110 
111   public    void printThreadIDOn(Address addr, PrintStream tty) {
112     tty.print(getThreadProxy(addr));
113   }
114 
115   public    void printInfoOn(Address threadAddr, PrintStream tty) {
116     tty.print("Thread id: ");
117     printThreadIDOn(threadAddr, tty);
118 //    tty.println("\nPostJavaState: " + getPostJavaState(threadAddr));
119   }
120 
121   public    Address getLastSP(Address addr) {
122     ThreadProxy t = getThreadProxy(addr);
123     AMD64ThreadContext context = (AMD64ThreadContext) t.getContext();
124     return context.getRegisterAsAddress(AMD64ThreadContext.RSP);
125   }
126 
127   public    ThreadProxy getThreadProxy(Address addr) {
128     // Addr is the address of the JavaThread.
129     // Fetch the OSThread (for now and for simplicity, not making a
130     // separate "OSThread" class in this package)
131     Address osThreadAddr = osThreadField.getValue(addr);
132     // Get the address of the _thread_id from the OSThread
133     Address threadIdAddr = osThreadAddr.addOffsetTo(osThreadThreadIDField.getOffset());
134 
135     JVMDebugger debugger = VM.getVM().getDebugger();
136     return debugger.getThreadForIdentifierAddress(threadIdAddr);
137   }
138 }
