 27 package jdk.incubator.foreign;
 28 
 29 import jdk.internal.foreign.InternalForeign;
 30 import jdk.internal.foreign.MemorySegmentImpl;
 31 
 32 /**
 33  * A class containing various methods relating to native interop.
 34  */
 35 public interface Foreign {
 36 
 37     /**
 38      * Get an instance of the {@linkplain Foreign} interface.
 39      * <p>
 40      * Access to this method must be explicitly enabled by setting the {@code jdk.incubator.foreign.Foreign} system
 41      * property. The available values are {@code deny}, {@code warn}, {@code debug}, {@code permit}.
 42      * <ul>
 43      * <li>
 44      * If set to {@code deny} an {@linkplain IllegalAccessError} will be thrown.
 45      * <li>
 46      * If set to {@code warn} a warning message will be written to the standard error stream.
 47      * <li>
 48      * If set to {@code debug} a debug message and a stack trace will be printed to the standard output stream.
 49      * <li>
 50      * If set to {@code permit} no special action will be taken.
 51      * <li>
 52      * If set to any other value an {@linkplain IllegalAccessError} will be thrown.
 53      * </ul>
 54      * <p>
 55      * {@code deny} is the default value.
 56      *
 57      * @return an instance of {@linkplain Foreign}
 58      * @throws IllegalAccessError if {@code jdk.incubator.foreign.Foreign} is set to {@code deny}
 59      */
 60     static Foreign getInstance() throws IllegalAccessError {
 61         return InternalForeign.getInstance();
 62     }
 63 
 64     /**
 65      * Returns a new native memory segment with given base address and size. The returned segment has its own temporal
 66      * bounds, and can therefore be closed; closing such a segment does <em>not</em> result in any resource being
 67      * deallocated.

 68      * <p>
 69      * This method is <em>restricted</em>. Restricted method are unsafe, and, if used incorrectly, their use might crash
 70      * the JVM crash or, worse, silently result in memory corruption. Thus, clients should refrain from depending on
 71      * restricted methods, and use safe and supported functionalities, where possible.















 72      * <p>
 73      * This method allows for making an otherwise in-accessible memory region accessible. However, there
 74      * is no guarantee that this memory is safe to access, or that the given size for the new segment is not too large,
 75      * potentially resulting in out-of-bounds accesses. The developer is trusted to make the judgement that the use of the
 76      * returned memory segment is safe.
 77      *
 78      * @param base the desired base address
 79      * @param byteSize the desired size.
 80      * @return a new native memory segment with given base address and size.
 81      * @throws IllegalArgumentException if {@code base} does not encapsulate a native memory address.

 82      * @throws IllegalAccessError if the permission jkd.incubator.foreign.restrictedMethods is set to 'deny'
 83      */
 84     MemorySegment ofNativeUnchecked(MemoryAddress base, long byteSize) throws IllegalAccessError;
 85 
 86     /**
 87      * Returns a non-confined memory segment that has the same spatial and temporal bounds as the provided segment.
 88      * Since the returned segment can be effectively accessed by multiple threads in an unconstrained fashion,
 89      * this method should be used with care, as it might lead to JVM crashes - e.g. in the case where a thread {@code A}
 90      * closes a segment while another thread {@code B} is accessing it.
 91      * <p>
 92      * This method is <em>restricted</em>. Restricted method are unsafe, and, if used incorrectly, their use might crash
 93      * the JVM crash or, worse, silently result in memory corruption. Thus, clients should refrain from depending on
 94      * restricted methods, and use safe and supported functionalities, where possible.
 95      * @param segment the segment from which an unconfined view is to be created.
 96      * @return a non-confined memory segment that has the same spatial and temporal bounds as the provided segment.
 97      */
 98     MemorySegment asUnconfined(MemorySegment segment);
 99 





100 }
