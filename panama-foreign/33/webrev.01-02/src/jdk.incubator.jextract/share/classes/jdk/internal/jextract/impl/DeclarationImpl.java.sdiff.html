 40 public abstract class DeclarationImpl implements Declaration {
 41 
 42     private final String name;
 43     private final Position pos;
 44     private final Map<String, List<ConstantDesc>> attributes;
 45 
 46     public DeclarationImpl(String name, Position pos, Map<String, List<ConstantDesc>> attrs) {
 47         this.name = name;
 48         this.pos = pos;
 49         this.attributes = attrs;
 50     }
 51 
 52     public String toString() {
 53         return new PrettyPrinter().print(this);
 54     }
 55 
 56     public String name() {
 57         return name;
 58     }
 59 
 60     @Override
 61     public Position pos() {
 62         return pos;
 63     }
 64 
 65     @Override
 66     public Optional<List<ConstantDesc>> getAttribute(String name) {
 67         return Optional.ofNullable(attributes.get(name));
 68     }
 69 
 70     @Override
 71     public Set<String> attributeNames() { return Collections.unmodifiableSet(attributes.keySet()); }

 40 public abstract class DeclarationImpl implements Declaration {
 41 
 42     private final String name;
 43     private final Position pos;
 44     private final Optional<Map<String, List<Constable>>> attributes;
 45 
 46     public DeclarationImpl(String name, Position pos, Map<String, List<Constable>> attrs) {
 47         this.name = name;
 48         this.pos = pos;
 49         this.attributes = Optional.ofNullable(attrs);
 50     }
 51 
 52     public String toString() {
 53         return new PrettyPrinter().print(this);
 54     }
 55 
 56     public String name() {
 57         return name;
 58     }
 59 
 60     @Override
 61     public Position pos() {
 62         return pos;
 63     }
 64 
 65     @Override
 66     public Optional<List<Constable>> getAttribute(String name) {
 67         return attributes.map(attrs -> attrs.get(name));
 68     }
 69 
 70     @Override
 71     public Set<String> attributeNames() { return Collections.unmodifiableSet(
 72             attributes.map(Map::keySet).orElse(Collections.emptySet()));
 73     }
