27 import jdk.incubator.foreign.MemoryAddress;
28 import jdk.internal.foreign.MemoryAddressImpl;
29 import jdk.internal.foreign.MemoryScope;
30 import jdk.internal.foreign.MemorySegmentImpl;
31 
32 public class UpcallStubs {
33 
34     public static MemoryAddress upcallAddress(UpcallHandler handler) {
35         long addr = handler.entryPoint();
36         return MemoryAddress.ofLong(addr);
37     }
38 
39     public static void freeUpcallStub(MemoryAddress address) {
40         MemoryAddressImpl ma = (MemoryAddressImpl) address;
41         if (ma.unsafeGetBase() != null || !freeUpcallStub(ma.unsafeGetOffset())) {
42             throw new IllegalArgumentException("Not a stub address: " + address);
43         }
44     }
45 
46     // natives
47 
48     // returns true if the stub was found (and freed)
49     private static native boolean freeUpcallStub(long addr);
