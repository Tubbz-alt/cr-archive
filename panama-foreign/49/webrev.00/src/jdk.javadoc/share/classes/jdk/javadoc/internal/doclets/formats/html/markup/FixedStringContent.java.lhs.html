25 package jdk.javadoc.internal.doclets.formats.html.markup;
26 
27 import java.io.IOException;
28 import java.io.Writer;
29 
30 import jdk.javadoc.internal.doclets.toolkit.Content;
31 import jdk.javadoc.internal.doclets.toolkit.util.DocletConstants;
32 
33 /**
34  * Class for containing fixed string content for HTML tags of javadoc output.
35  *
36  *  <p><b>This is NOT part of any supported API.
37  *  If you write code that depends on this, you do so at your own risk.
38  *  This code and its internal interfaces are subject to change or
39  *  deletion without notice.</b>
40  */
41 public class FixedStringContent extends Content {
42     private final String string;
43 
44     /**
45      * Constructor to construct FixedStringContent object.
46      *
47      * @param content content for the object
48      */
49     public FixedStringContent(CharSequence content) {
50         string = Entity.escapeHtmlChars(content);
51     }
52 
53     /**
54      * This method is not supported by the class.
55      *
56      * @param content content that needs to be added
57      * @throws UnsupportedOperationException always
58      */
59     @Override
60     public void add(Content content) {
61         throw new UnsupportedOperationException();
62     }
63 
64     /**
65      * Adds content for the StringContent object.  The method escapes
66      * HTML characters for the string content that is added.
67      *
68      * @param strContent string content to be added
69      * @throws UnsupportedOperationException always
70      */
71     @Override
72     public void add(CharSequence strContent) {
73         throw new UnsupportedOperationException();
74     }
75 
76     @Override
77     public boolean isEmpty() {
78         return string.isEmpty();
79     }
80 
81     @Override
82     public int charCount() {
83         return RawHtml.charCount(string);
84     }
85 
86     @Override
87     public String toString() {
88         return string;
89     }
90 
91     @Override
92     public boolean write(Writer out, boolean atNewline) throws IOException {
93         out.write(string);
94         return string.endsWith(DocletConstants.NL);
95     }
96 
97 }
