 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import java.io.IOException;
 29 import java.io.StringWriter;
 30 import java.io.Writer;
 31 import java.util.Objects;
 32 
 33 /**
 34  * A class to create content for javadoc output pages.
 35  *
 36  *  <p><b>This is NOT part of any supported API.
 37  *  If you write code that depends on this, you do so at your own risk.
 38  *  This code and its internal interfaces are subject to change or
 39  *  deletion without notice.</b>
 40  */
 41 public abstract class Content {
 42 
 43     /**
 44      * Returns a string representation of the content.
 45      *
 46      * @return string representation of the content
 47      */
 48     @Override
 49     public String toString() {
 50         StringWriter out = new StringWriter();
 51         try {
 52             write(out, true);
 53         } catch (IOException e) {
 54             // cannot happen from StringWriter
 55             throw new AssertionError(e);
 56         }
 57         return out.toString();
 58     }
 59 
 60     /**
 61      * Adds content to the existing content.

 62      *
 63      * @param content content that needs to be added






 64      */
 65     public abstract void add(Content content);


 66 
 67     /**
 68      * Adds a string content to the existing content.




 69      *
 70      * @param stringContent the string content to be added




 71      */
 72     public abstract void add(CharSequence stringContent);


 73 
 74     /**
 75      * Writes content to a writer.
 76      *
 77      * @param writer the writer
 78      * @param atNewline whether the writer has just written a newline
 79      * @return  whether the writer has just written a newline
 80      * @throws IOException if an error occurs while writing the output
 81      */
 82     public abstract boolean write(Writer writer, boolean atNewline) throws IOException ;
 83 
 84     /**
 85      * Returns true if the content is empty.
 86      *
 87      * @return true if no content to be displayed else return false
 88      */
 89     public abstract boolean isEmpty();
 90 
 91     /**
 92      * Returns true if the content is valid.
 93      *
 94      * @return true if the content is valid else return false
 95      */
 96     public boolean isValid() {
 97         return !isEmpty();
 98     }
 99 
100     /**
101      * Return the number of characters of plain text content in this object
102      * (optional operation.)
103      * @return the number of characters of plain text content in this
104      */
105     public int charCount() {
106         return 0;
107     }
108 
109     /**
110      * Checks for null values.
111      *
112      * @param <T> the type of the item being checked
113      * @param t reference type to check for null values
114      * @return the reference type if not null or else throws a null pointer exception
115      */
116     protected static <T> T nullCheck(T t) {
117         return Objects.requireNonNull(t);
118     }
119 }
