 96     /**
 97      * Details about a package in a module.
 98      * A package may be not exported, or exported to some modules, or exported to all modules.
 99      * A package may be not opened, or opened to some modules, or opened to all modules.
100      * A package that is neither exported or opened to any modules is a concealed package.
101      * An open module opens all its packages to all modules.
102      */
103     class PackageEntry {
104         /**
105          * Summary of package exports:
106          * If null, the package is not exported to any modules;
107          * if empty, the package is exported to all modules;
108          * otherwise, the package is exported to these modules.
109          */
110         Set<ModuleElement> exportedTo;
111 
112         /**
113          * Summary of package opens:
114          * If null, the package is not opened to any modules;
115          * if empty, the package is opened to all modules;
116          * otherwise, the package is opened to these modules.
117          */
118         Set<ModuleElement> openedTo;
119     }

181     /**
182      * Get the module header.
183      *
184      * @param heading the heading for the section
185      */
186     @Override
187     public Content getModuleHeader(String heading) {
188         HtmlTree bodyTree = getBody(getWindowTitle(mdle.getQualifiedName().toString()));
189         Content headerContent = new ContentBuilder();
190         addTop(headerContent);
191         navBar.setDisplaySummaryModuleDescLink(!utils.getFullBody(mdle).isEmpty() && !options.noComment());
192         navBar.setDisplaySummaryModulesLink(display(requires) || display(indirectModules));
193         navBar.setDisplaySummaryPackagesLink(display(packages) || display(indirectPackages)
194                 || display(indirectOpenPackages));
195         navBar.setDisplaySummaryServicesLink(displayServices(uses, usesTrees) || displayServices(provides.keySet(), providesTrees));
196         navBar.setUserHeader(getUserHeaderFooter(true));
197         headerContent.add(navBar.getContent(true));
198         HtmlTree div = new HtmlTree(HtmlTag.DIV);
199         div.setStyle(HtmlStyle.header);
200         Content annotationContent = new HtmlTree(HtmlTag.P);
201         addAnnotationInfo(mdle, annotationContent);
202         div.add(annotationContent);
203         Content label = mdle.isOpen() && (configuration.docEnv.getModuleMode() == ModuleMode.ALL)
204                 ? contents.openModuleLabel : contents.moduleLabel;
205         Content tHeading = HtmlTree.HEADING(Headings.PAGE_TITLE_HEADING, true,
206                 HtmlStyle.title, label);
207         tHeading.add(Entity.NO_BREAK_SPACE);
208         Content moduleHead = new RawHtml(heading);
209         tHeading.add(moduleHead);
210         div.add(tHeading);
211         bodyContents.setHeader(headerContent)
212                 .addMainContent(div);
213         return bodyTree;
214     }

244     /**
245      * Compute the modules data that will be displayed in various tables on the module summary page.
246      */
247     public void computeModulesData() {
248         CommentHelper ch = utils.getCommentHelper(mdle);
249         // Get module dependencies using the module's transitive closure.
250         Map<ModuleElement, String> dependentModules = utils.getDependentModules(mdle);
251         // Add all dependent modules to indirect modules set. We will remove the modules,
252         // listed using the requires directive, from this set to come up with the table of indirect
253         // required modules.
254         dependentModules.forEach((module, mod) -> {
255             if (shouldDocument(module)) {
256                 indirectModules.put(module, new StringContent(mod));
257             }
258         });
259         ElementFilter.requiresIn(mdle.getDirectives()).forEach(directive -> {
260             ModuleElement m = directive.getDependency();
261             if (shouldDocument(m)) {
262                 if (moduleMode == ModuleMode.ALL || directive.isTransitive()) {
263                     requires.put(m, new StringContent(utils.getModifiers(directive)));
264                 } else {
265                     // For api mode, just keep the public requires in dependentModules for display of
266                     // indirect packages in the "Packages" section.
267                     dependentModules.remove(m);
268                 }
269                 indirectModules.remove(m);
270             }
271         });
272 
273         // Get all packages if module is open or if displaying concealed modules
274         for (PackageElement pkg : utils.getModulePackageMap().getOrDefault(mdle, Collections.emptySet())) {
275             if (shouldDocument(pkg) && (mdle.isOpen() || moduleMode == ModuleMode.ALL)) {
276                 PackageEntry e = new PackageEntry();
277                 if (mdle.isOpen()) {
278                     e.openedTo = Collections.emptySet();
279                 }
280                 packages.put(pkg, e);
281             }
282         }
283 
284         // Get all exported packages for the module, using the exports directive for the module.
285         for (ModuleElement.ExportsDirective directive : ElementFilter.exportsIn(mdle.getDirectives())) {
286             PackageElement p = directive.getPackage();
287             if (shouldDocument(p)) {
288                 List<? extends ModuleElement> targetMdles = directive.getTargetModules();
289                 // Include package if in details mode, or exported to all (i.e. targetModules == null)
290                 if (moduleMode == ModuleMode.ALL || targetMdles == null) {
291                     PackageEntry packageEntry = packages.computeIfAbsent(p, pkg -> new PackageEntry());
292                     SortedSet<ModuleElement> mdleList = new TreeSet<>(utils.makeModuleComparator());
293                     if (targetMdles != null) {
294                         mdleList.addAll(targetMdles);
295                     }
296                     packageEntry.exportedTo = mdleList;
297                 }
298             }
299         }
300 
301         // Get all opened packages for the module, using the opens directive for the module.
302         // If it is an open module, there will be no separate opens directives.
303         for (ModuleElement.OpensDirective directive : ElementFilter.opensIn(mdle.getDirectives())) {
304             PackageElement p = directive.getPackage();
305             if (shouldDocument(p)) {
306                 List<? extends ModuleElement> targetMdles = directive.getTargetModules();
307                 // Include package if in details mode, or opened to all (i.e. targetModules == null)
308                 if (moduleMode == ModuleMode.ALL || targetMdles == null) {
309                     PackageEntry packageEntry = packages.computeIfAbsent(p, pkg -> new PackageEntry());
310                     SortedSet<ModuleElement> mdleList = new TreeSet<>(utils.makeModuleComparator());
311                     if (targetMdles != null) {
312                         mdleList.addAll(targetMdles);
313                     }
314                     packageEntry.openedTo = mdleList;
315                 }
316             }
317         }
318 
319         // Get all the exported and opened packages, for the transitive closure of the module, to be displayed in
320         // the indirect packages tables.
321         dependentModules.forEach((module, mod) -> {
322             SortedSet<PackageElement> exportedPackages = new TreeSet<>(utils.makePackageComparator());
323             ElementFilter.exportsIn(module.getDirectives()).forEach(directive -> {
324                 PackageElement pkg = directive.getPackage();
325                 if (shouldDocument(pkg)) {
326                     // Qualified exports are not displayed in API mode
327                     if (moduleMode == ModuleMode.ALL || directive.getTargetModules() == null) {
328                         exportedPackages.add(pkg);
329                     }
330                 }
331             });
332             // If none of the indirect modules have exported packages to be displayed, we should not be
333             // displaying the table and so it should not be added to the map.
334             if (!exportedPackages.isEmpty()) {
335                 indirectPackages.put(module, exportedPackages);
336             }
337             SortedSet<PackageElement> openPackages = new TreeSet<>(utils.makePackageComparator());
338             if (module.isOpen()) {
339                 openPackages.addAll(utils.getModulePackageMap().getOrDefault(module, Collections.emptySet()));
340             } else {
341                 ElementFilter.opensIn(module.getDirectives()).forEach(directive -> {
342                     PackageElement pkg = directive.getPackage();
343                     if (shouldDocument(pkg)) {
344                         // Qualified opens are not displayed in API mode
345                         if (moduleMode == ModuleMode.ALL || directive.getTargetModules() == null) {
346                             openPackages.add(pkg);
347                         }
348                     }
349                 });
350             }
351             // If none of the indirect modules have opened packages to be displayed, we should not be
352             // displaying the table and so it should not be added to the map.
353             if (!openPackages.isEmpty()) {
354                 indirectOpenPackages.put(module, openPackages);
355             }
356         });
357         // Get all the services listed as uses directive.
358         ElementFilter.usesIn(mdle.getDirectives()).forEach(directive -> {
359             TypeElement u = directive.getService();
360             if (shouldDocument(u)) {
361                 uses.add(u);
362             }
363         });
364         // Get all the services and implementations listed as provides directive.
365         ElementFilter.providesIn(mdle.getDirectives()).forEach(directive -> {
366             TypeElement u = directive.getService();
367             if (shouldDocument(u)) {
368                 List<? extends TypeElement> implList = directive.getImplementations();
369                 SortedSet<TypeElement> implSet = new TreeSet<>(utils.makeAllClassesComparator());
370                 implSet.addAll(implList);
371                 provides.put(u, implSet);
372             }
373         });
374         // Generate the map of all services listed using @provides, and the description.
375         utils.getBlockTags(mdle, DocTree.Kind.PROVIDES).forEach(tree -> {
376             TypeElement t = ch.getServiceType(tree);
377             if (t != null) {
378                 providesTrees.put(t, commentTagsToContent(tree, mdle, ch.getDescription(tree), false, true));
379             }
380         });
381         // Generate the map of all services listed using @uses, and the description.
382         utils.getBlockTags(mdle, DocTree.Kind.USES).forEach(tree -> {
383             TypeElement t = ch.getServiceType(tree);
384             if (t != null) {
385                 usesTrees.put(t, commentTagsToContent(tree, mdle, ch.getDescription(tree), false, true));
386             }
387         });
388     }

420     /*
421      * Returns true, in API mode, if at least one type element in
422      * the typeElements set is referenced by a javadoc tag in tagsMap.
423      */
424     private boolean displayServices(Set<TypeElement> typeElements,
425                                     Map<TypeElement, Content> tagsMap) {
426         return typeElements != null &&
427                 typeElements.stream().anyMatch(v -> displayServiceDirective(v, tagsMap));
428     }
429 
430     /*
431      * Returns true, in API mode, if the type element is referenced
432      * from a javadoc tag in tagsMap.
433      */
434     private boolean displayServiceDirective(TypeElement typeElement,
435                                             Map<TypeElement, Content> tagsMap) {
436         return moduleMode == ModuleMode.ALL || tagsMap.containsKey(typeElement);
437     }

486     @Override
487     public void addModulesSummary(Content summaryContentTree) {
488         if (display(requires) || display(indirectModules)) {
489             TableHeader requiresTableHeader =
490                     new TableHeader(contents.modifierLabel, contents.moduleLabel,
491                             contents.descriptionLabel);
492             HtmlTree section = HtmlTree.SECTION(HtmlStyle.modulesSummary)
493                     .setId(SectionName.MODULES.getName());
494             addSummaryHeader(MarkerComments.START_OF_MODULES_SUMMARY, contents.navModules, section);
495             if (display(requires)) {
496                 String text = resources.getText("doclet.Requires_Summary");
497                 String tableSummary = resources.getText("doclet.Member_Table_Summary",
498                         text,
499                         resources.getText("doclet.modules"));
500                 Content caption = getTableCaption(new StringContent(text));
501                 Table table = getTable3(caption, tableSummary, HtmlStyle.requiresSummary,
502                             requiresTableHeader);
503                 addModulesList(requires, table);
504                 section.add(table.toContent());
505             }
506             // Display indirect modules table in both "api" and "all" mode.
507             if (display(indirectModules)) {
508                 String amrText = resources.getText("doclet.Indirect_Requires_Summary");
509                 String amrTableSummary = resources.getText("doclet.Member_Table_Summary",
510                         amrText,
511                         resources.getText("doclet.modules"));
512                 Content amrCaption = getTableCaption(new StringContent(amrText));
513                 Table amrTable = getTable3(amrCaption, amrTableSummary, HtmlStyle.requiresSummary,
514                             requiresTableHeader);
515                 addModulesList(indirectModules, amrTable);
516                 section.add(amrTable.toContent());
517             }
518             summaryContentTree.add(HtmlTree.LI(HtmlStyle.blockList, section));
519         }
520     }

538     @Override
539     public void addPackagesSummary(Content summaryContentTree) {
540         if (display(packages)
541                 || display(indirectPackages) || display(indirectOpenPackages)) {
542             HtmlTree section = HtmlTree.SECTION(HtmlStyle.packagesSummary)
543                     .setId(SectionName.PACKAGES.getName());;
544             addSummaryHeader(MarkerComments.START_OF_PACKAGES_SUMMARY, contents.navPackages, section);
545             if (display(packages)) {
546                 addPackageSummary(section);
547             }
548             TableHeader indirectPackagesHeader =
549                     new TableHeader(contents.fromLabel, contents.packagesLabel);
550             if (display(indirectPackages)) {
551                 String aepText = resources.getText("doclet.Indirect_Exports_Summary");
552                 Table aepTable = getTable2(new StringContent(aepText),
553                         HtmlStyle.packagesSummary, indirectPackagesHeader);
554                 addIndirectPackages(aepTable, indirectPackages);
555                 section.add(aepTable.toContent());
556             }
557             if (display(indirectOpenPackages)) {
558                 String aopText = resources.getText("doclet.Indirect_Opens_Summary");
559                 Table aopTable = getTable2(new StringContent(aopText), HtmlStyle.packagesSummary,
560                         indirectPackagesHeader);
561                 addIndirectPackages(aopTable, indirectOpenPackages);
562                 section.add(aopTable.toContent());
563             }
564             summaryContentTree.add(HtmlTree.LI(HtmlStyle.blockList, section));
565         }

568     /**
569      * Add the package summary for the module.
570      *
571      * @param li
572      */
573     public void addPackageSummary(HtmlTree li) {
574         Table table = new Table(HtmlStyle.packagesSummary)
575                 .setDefaultTab(resources.getText("doclet.All_Packages"))
576                 .addTab(resources.getText("doclet.Exported_Packages_Summary"), this::isExported)
577                 .addTab(resources.getText("doclet.Opened_Packages_Summary"), this::isOpened)
578                 .addTab(resources.getText("doclet.Concealed_Packages_Summary"), this::isConcealed)
579                 .setTabScript(i -> String.format("show(%d);", i));
580 
581         // Determine whether to show the "Exported To" and "Opened To" columns,
582         // based on whether such columns would provide "useful" info.
583         int numExports = 0;
584         int numUnqualifiedExports = 0;
585         int numOpens = 0;
586         int numUnqualifiedOpens = 0;
587 
588         for (PackageEntry e : packages.values()) {
589             if (e.exportedTo != null) {
590                 numExports++;
591                 if (e.exportedTo.isEmpty()) {
592                     numUnqualifiedExports++;
593                 }
594             }
595             if (e.openedTo != null) {
596                 numOpens++;
597                 if (e.openedTo.isEmpty()) {
598                     numUnqualifiedOpens++;
599                 }
600             }
601         }
602 
603         boolean showExportedTo = numExports > 0 && (numOpens > 0   || numUnqualifiedExports < packages.size());
604         boolean showOpenedTo   = numOpens > 0   && (numExports > 0 || numUnqualifiedOpens < packages.size());
605 
606         // Create the table header and column styles.
607         List<Content> colHeaders = new ArrayList<>();
608         List<HtmlStyle> colStyles = new ArrayList<>();
609         colHeaders.add(contents.packageLabel);
610         colStyles.add(HtmlStyle.colFirst);
611 
612         if (showExportedTo) {
613             colHeaders.add(contents.exportedTo);
614             colStyles.add(HtmlStyle.colSecond);
615         }
616 
617         if (showOpenedTo) {
618             colHeaders.add(contents.openedTo);
619             colStyles.add(HtmlStyle.colSecond);
620         }
621 
622         colHeaders.add(contents.descriptionLabel);
623         colStyles.add(HtmlStyle.colLast);
624 
625         table.setHeader(new TableHeader(colHeaders).styles(colStyles))
626                 .setColumnStyles(colStyles);
627 
628         // Add the table rows, based on the "packages" map.
629         for (Map.Entry<PackageElement, PackageEntry> e : packages.entrySet()) {
630             PackageElement pkg = e.getKey();
631             PackageEntry entry = e.getValue();
632             List<Content> row = new ArrayList<>();
633             Content pkgLinkContent = getPackageLink(pkg, new StringContent(utils.getPackageName(pkg)));
634             row.add(pkgLinkContent);
635 
636             if (showExportedTo) {
637                 row.add(getPackageExportOpensTo(entry.exportedTo));
638             }
639             if (showOpenedTo) {
640                 row.add(getPackageExportOpensTo(entry.openedTo));
641             }
642             Content summary = new ContentBuilder();
643             addSummaryComment(pkg, summary);
644             row.add(summary);
645 
646             table.addRow(pkg, row);
647         }
648 
649         li.add(table.toContent());
650         if (table.needsScript()) {
651             mainBodyScript.append(table.getScript());
652         }
653     }

655     private boolean isExported(Element e) {
656         PackageEntry entry = packages.get((PackageElement) e);
657         return (entry != null) && (entry.exportedTo != null);
658     }

665     private boolean isConcealed(Element e) {
666         PackageEntry entry = packages.get((PackageElement) e);

670     private Content getPackageExportOpensTo(Set<ModuleElement> modules) {
671         if (modules == null) {
672             return new StringContent(resources.getText("doclet.None"));
673         } else if (modules.isEmpty()) {
674             return new StringContent(resources.getText("doclet.All_Modules"));
675         } else {
676             Content list = new ContentBuilder();
677             for (ModuleElement m : modules) {
678                 if (!list.isEmpty()) {
679                     list.add(new StringContent(", "));
680                 }
681                 list.add(getModuleLink(m, new StringContent(m.getQualifiedName())));
682             }
683             return list;
684         }

743     /**
744      * Add the uses list for the module.
745      *
746      * @param table the table to which the services used will be added
747      */
748     public void addUsesList(Table table) {
749         Content typeLinkContent;
750         Content description;
751         for (TypeElement t : uses) {
752             if (!displayServiceDirective(t, usesTrees)) {
753                 continue;
754             }
755             typeLinkContent = getLink(new LinkInfoImpl(configuration, LinkInfoImpl.Kind.PACKAGE, t));
756             Content summary = new ContentBuilder();
757             if (display(usesTrees)) {
758                 description = usesTrees.get(t);
759                 if (description != null && !description.isEmpty()) {
760                     summary.add(HtmlTree.DIV(HtmlStyle.block, description));
761                 } else {
762                     addSummaryComment(t, summary);
763                 }
764             } else {
765                 summary.add(Entity.NO_BREAK_SPACE);
766             }
767             table.addRow(typeLinkContent, summary);
768         }
769     }

771     /**
772      * Add the provides list for the module.
773      *
774      * @param table the table to which the services provided will be added
775      */
776     public void addProvidesList(Table table) {
777         SortedSet<TypeElement> implSet;
778         Content description;
779         for (Map.Entry<TypeElement, SortedSet<TypeElement>> entry : provides.entrySet()) {
780             TypeElement srv = entry.getKey();
781             if (!displayServiceDirective(srv, providesTrees)) {
782                 continue;
783             }
784             implSet = entry.getValue();
785             Content srvLinkContent = getLink(new LinkInfoImpl(configuration, LinkInfoImpl.Kind.PACKAGE, srv));
786             Content desc = new ContentBuilder();
787             if (display(providesTrees)) {
788                 description = providesTrees.get(srv);
789                 desc.add((description != null && !description.isEmpty())
790                         ? HtmlTree.DIV(HtmlStyle.block, description)
791                         : Entity.NO_BREAK_SPACE);
792             } else {
793                 desc.add(Entity.NO_BREAK_SPACE);
794                 }
795             // Only display the implementation details in the "all" mode.
796             if (moduleMode == ModuleMode.ALL && !implSet.isEmpty()) {
797                 desc.add(new HtmlTree(HtmlTag.BR));
798                 desc.add("(");
799                 HtmlTree implSpan = HtmlTree.SPAN(HtmlStyle.implementationLabel, contents.implementation);
800                 desc.add(implSpan);
801                 desc.add(Entity.NO_BREAK_SPACE);
802                 String sep = "";
803                 for (TypeElement impl : implSet) {
804                     desc.add(sep);
805                     desc.add(getLink(new LinkInfoImpl(configuration, LinkInfoImpl.Kind.PACKAGE, impl)));
806                     sep = ", ";
807                 }
808                 desc.add(")");
809             }
810             table.addRow(srvLinkContent, desc);
811         }
812     }
