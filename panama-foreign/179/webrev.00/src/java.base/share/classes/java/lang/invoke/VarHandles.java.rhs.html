 56     static VarHandle makeFieldHandle(MemberName f, Class<?> refc, Class<?> type, boolean isWriteAllowedOnFinalFields) {
 57         if (!f.isStatic()) {
 58             long foffset = MethodHandleNatives.objectFieldOffset(f);
 59             if (!type.isPrimitive()) {
 60                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 61                        ? new VarHandleReferences.FieldInstanceReadOnly(refc, foffset, type)
 62                        : new VarHandleReferences.FieldInstanceReadWrite(refc, foffset, type));
 63             }
 64             else if (type == boolean.class) {
 65                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 66                        ? new VarHandleBooleans.FieldInstanceReadOnly(refc, foffset)
 67                        : new VarHandleBooleans.FieldInstanceReadWrite(refc, foffset));
 68             }
 69             else if (type == byte.class) {
 70                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 71                        ? new VarHandleBytes.FieldInstanceReadOnly(refc, foffset)
 72                        : new VarHandleBytes.FieldInstanceReadWrite(refc, foffset));
 73             }
 74             else if (type == short.class) {
 75                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 76                        ? new VarHandleShorts.FieldInstanceReadOnly(refc, foffset)
 77                        : new VarHandleShorts.FieldInstanceReadWrite(refc, foffset));
 78             }
 79             else if (type == char.class) {
 80                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 81                        ? new VarHandleChars.FieldInstanceReadOnly(refc, foffset)
 82                        : new VarHandleChars.FieldInstanceReadWrite(refc, foffset));
 83             }
 84             else if (type == int.class) {
 85                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 86                        ? new VarHandleInts.FieldInstanceReadOnly(refc, foffset)
 87                        : new VarHandleInts.FieldInstanceReadWrite(refc, foffset));
 88             }
 89             else if (type == long.class) {
 90                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 91                        ? new VarHandleLongs.FieldInstanceReadOnly(refc, foffset)
 92                        : new VarHandleLongs.FieldInstanceReadWrite(refc, foffset));
 93             }
 94             else if (type == float.class) {
 95                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 96                        ? new VarHandleFloats.FieldInstanceReadOnly(refc, foffset)
 97                        : new VarHandleFloats.FieldInstanceReadWrite(refc, foffset));
 98             }
 99             else if (type == double.class) {
100                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
101                        ? new VarHandleDoubles.FieldInstanceReadOnly(refc, foffset)
102                        : new VarHandleDoubles.FieldInstanceReadWrite(refc, foffset));
103             }
104             else {
105                 throw new UnsupportedOperationException();
106             }
107         }
108         else {
109             // TODO This is not lazy on first invocation
110             // and might cause some circular initialization issues
111 
112             // Replace with something similar to direct method handles
113             // where a barrier is used then elided after use
114 
115             if (UNSAFE.shouldBeInitialized(refc))
116                 UNSAFE.ensureClassInitialized(refc);
117 
118             Object base = MethodHandleNatives.staticFieldBase(f);
119             long foffset = MethodHandleNatives.staticFieldOffset(f);
120             if (!type.isPrimitive()) {
121                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
122                        ? new VarHandleReferences.FieldStaticReadOnly(base, foffset, type)
123                        : new VarHandleReferences.FieldStaticReadWrite(base, foffset, type));
124             }
125             else if (type == boolean.class) {
126                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
127                        ? new VarHandleBooleans.FieldStaticReadOnly(base, foffset)
128                        : new VarHandleBooleans.FieldStaticReadWrite(base, foffset));
129             }
130             else if (type == byte.class) {
131                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
132                        ? new VarHandleBytes.FieldStaticReadOnly(base, foffset)
133                        : new VarHandleBytes.FieldStaticReadWrite(base, foffset));
134             }
135             else if (type == short.class) {
136                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
137                        ? new VarHandleShorts.FieldStaticReadOnly(base, foffset)
138                        : new VarHandleShorts.FieldStaticReadWrite(base, foffset));
139             }
140             else if (type == char.class) {
141                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
142                        ? new VarHandleChars.FieldStaticReadOnly(base, foffset)
143                        : new VarHandleChars.FieldStaticReadWrite(base, foffset));
144             }
145             else if (type == int.class) {
146                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
147                        ? new VarHandleInts.FieldStaticReadOnly(base, foffset)
148                        : new VarHandleInts.FieldStaticReadWrite(base, foffset));
149             }
150             else if (type == long.class) {
151                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
152                        ? new VarHandleLongs.FieldStaticReadOnly(base, foffset)
153                        : new VarHandleLongs.FieldStaticReadWrite(base, foffset));
154             }
155             else if (type == float.class) {
156                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
157                        ? new VarHandleFloats.FieldStaticReadOnly(base, foffset)
158                        : new VarHandleFloats.FieldStaticReadWrite(base, foffset));
159             }
160             else if (type == double.class) {
161                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
162                        ? new VarHandleDoubles.FieldStaticReadOnly(base, foffset)
163                        : new VarHandleDoubles.FieldStaticReadWrite(base, foffset));
164             }
165             else {
166                 throw new UnsupportedOperationException();
167             }
168         }

171     // Required by instance field handles
172     static Field getFieldFromReceiverAndOffset(Class<?> receiverType,
173                                                long offset,
174                                                Class<?> fieldType) {
175         for (Field f : receiverType.getDeclaredFields()) {
176             if (Modifier.isStatic(f.getModifiers())) continue;
177 
178             if (offset == UNSAFE.objectFieldOffset(f)) {
179                 assert f.getType() == fieldType;
180                 return f;
181             }
182         }
183         throw new InternalError("Field not found at offset");
184     }

203     static VarHandle makeArrayElementHandle(Class<?> arrayClass) {
204         if (!arrayClass.isArray())
205             throw new IllegalArgumentException("not an array: " + arrayClass);
206 
207         Class<?> componentType = arrayClass.getComponentType();
208 
209         int aoffset = UNSAFE.arrayBaseOffset(arrayClass);
210         int ascale = UNSAFE.arrayIndexScale(arrayClass);
211         int ashift = 31 - Integer.numberOfLeadingZeros(ascale);
212 
213         if (!componentType.isPrimitive()) {
214             return maybeAdapt(new VarHandleReferences.Array(aoffset, ashift, arrayClass));
215         }
216         else if (componentType == boolean.class) {
217             return maybeAdapt(new VarHandleBooleans.Array(aoffset, ashift));
218         }
219         else if (componentType == byte.class) {
220             return maybeAdapt(new VarHandleBytes.Array(aoffset, ashift));
221         }
222         else if (componentType == short.class) {
223             return maybeAdapt(new VarHandleShorts.Array(aoffset, ashift));
224         }
225         else if (componentType == char.class) {
226             return maybeAdapt(new VarHandleChars.Array(aoffset, ashift));
227         }
228         else if (componentType == int.class) {
229             return maybeAdapt(new VarHandleInts.Array(aoffset, ashift));
230         }
231         else if (componentType == long.class) {
232             return maybeAdapt(new VarHandleLongs.Array(aoffset, ashift));
233         }
234         else if (componentType == float.class) {
235             return maybeAdapt(new VarHandleFloats.Array(aoffset, ashift));
236         }
237         else if (componentType == double.class) {
238             return maybeAdapt(new VarHandleDoubles.Array(aoffset, ashift));
239         }
240         else {
241             throw new UnsupportedOperationException();
242         }
243     }

303     /**
304      * Creates a memory access VarHandle.
305      *
306      * Resulting VarHandle will take a memory address as first argument,
307      * and a certain number of coordinate {@code long} parameters, depending on the length
308      * of the {@code strides} argument array.
309      *
310      * Coordinates are multiplied with corresponding scale factors ({@code strides}) and added
311      * to a single fixed offset to compute an effective offset from the given MemoryAddress for the access.
312      *
313      * @param carrier the Java carrier type.
314      * @param alignmentMask alignment requirement to be checked upon access. In bytes. Expressed as a mask.
315      * @param byteOrder the byte order.
316      * @param offset a constant offset for the access.
317      * @param strides the scale factors with which to multiply given access coordinates.
318      * @return the created VarHandle.
319      */
320     static VarHandle makeMemoryAddressViewHandle(Class<?> carrier, long alignmentMask,
321                                                  ByteOrder byteOrder, long offset, long[] strides) {
322         if (!carrier.isPrimitive() || carrier == void.class || carrier == boolean.class) {
323             throw new IllegalArgumentException("Invalid carrier: " + carrier.getName());
324         }
325         long size = Wrapper.forPrimitiveType(carrier).bitWidth() / 8;
326         boolean be = byteOrder == ByteOrder.BIG_ENDIAN;
327 
328         Map<Integer, MethodHandle> carrierFactory = ADDRESS_FACTORIES.get(carrier);
329         MethodHandle fac = carrierFactory.computeIfAbsent(strides.length,
330                 dims -> new MemoryAccessVarHandleGenerator(carrier, dims)
331                             .generateHandleFactory());
332 
333         try {
334             return maybeAdapt((VarHandle)fac.invoke(be, size, offset, alignmentMask, strides));
335         } catch (Throwable ex) {
336             throw new IllegalStateException(ex);
337         }
338     }

340     private static VarHandle maybeAdapt(VarHandle target) {
341         if (!VAR_HANDLE_IDENTITY_ADAPT) return target;
342         target = filterValue(target,
343                         MethodHandles.identity(target.varType()), MethodHandles.identity(target.varType()));
344         MethodType mtype = target.accessModeType(VarHandle.AccessMode.GET).dropParameterTypes(0, 1);
345         for (int i = 0 ; i < mtype.parameterCount() ; i++) {
346             target = filterCoordinates(target, i, MethodHandles.identity(mtype.parameterType(i)));
347         }
348         return target;
349     }

351     public static VarHandle filterValue(VarHandle target, MethodHandle filterToTarget, MethodHandle filterFromTarget) {
352         Objects.nonNull(target);
353         Objects.nonNull(filterToTarget);
354         Objects.nonNull(filterFromTarget);
355         //check that from/to filters do not throw checked exceptions
356         noCheckedExceptions(filterToTarget);
357         noCheckedExceptions(filterFromTarget);
358 
359         List<Class<?>> newCoordinates = new ArrayList<>();
360         List<Class<?>> additionalCoordinates = new ArrayList<>();
361         newCoordinates.addAll(target.coordinateTypes());
362 
363         //check that from/to filters have right signatures
364         if (filterFromTarget.type().parameterCount() != filterToTarget.type().parameterCount()) {
365             throw newIllegalArgumentException("filterFromTarget and filterToTarget have different arity", filterFromTarget.type(), filterToTarget.type());
366         } else if (filterFromTarget.type().parameterCount() < 1) {
367             throw newIllegalArgumentException("filterFromTarget filter type has wrong arity", filterFromTarget.type());
368         } else if (filterToTarget.type().parameterCount() < 1) {
369             throw newIllegalArgumentException("filterToTarget filter type has wrong arity", filterFromTarget.type());
370         } else if (filterFromTarget.type().lastParameterType() != filterToTarget.type().returnType() ||
371                 filterToTarget.type().lastParameterType() != filterFromTarget.type().returnType()) {
372             throw newIllegalArgumentException("filterFromTarget and filterToTarget filter types do not match", filterFromTarget.type(), filterToTarget.type());
373         } else if (target.varType() != filterFromTarget.type().lastParameterType()) {
374             throw newIllegalArgumentException("filterFromTarget filter type does not match target var handle type", filterFromTarget.type(), target.varType());
375         } else if (target.varType() != filterToTarget.type().returnType()) {
376             throw newIllegalArgumentException("filterFromTarget filter type does not match target var handle type", filterToTarget.type(), target.varType());
377         } else if (filterFromTarget.type().parameterCount() > 1) {
378             for (int i = 0 ; i < filterFromTarget.type().parameterCount() - 1 ; i++) {
379                 if (filterFromTarget.type().parameterType(i) != filterToTarget.type().parameterType(i)) {
380                     throw newIllegalArgumentException("filterFromTarget and filterToTarget filter types do not match", filterFromTarget.type(), filterToTarget.type());
381                 } else {
382                     newCoordinates.add(filterFromTarget.type().parameterType(i));
383                     additionalCoordinates.add((filterFromTarget.type().parameterType(i)));
384                 }
385             }
386         }
387 
388         return new IndirectVarHandle(target, filterFromTarget.type().returnType(), newCoordinates.toArray(new Class<?>[0]),
389                 (mode, modeHandle) -> {
390                     int lastParameterPos = modeHandle.type().parameterCount() - 1;
391                     return switch (mode.at) {
392                         case GET -> MethodHandles.collectReturnValue(modeHandle, filterFromTarget);
393                         case SET -> MethodHandles.collectArguments(modeHandle, lastParameterPos, filterToTarget);
394                         case GET_AND_UPDATE -> {
395                             MethodHandle adapter = MethodHandles.collectReturnValue(modeHandle, filterFromTarget);
396                             MethodHandle res = MethodHandles.collectArguments(adapter, lastParameterPos, filterToTarget);
397                             if (additionalCoordinates.size() > 0) {
398                                 res = joinDuplicateArgs(res, lastParameterPos,
399                                         lastParameterPos + additionalCoordinates.size() + 1,
400                                         additionalCoordinates.size());
401                             }
402                             yield res;
403                         }
404                         case COMPARE_AND_EXCHANGE -> {
405                             MethodHandle adapter = MethodHandles.collectReturnValue(modeHandle, filterFromTarget);
406                             adapter = MethodHandles.collectArguments(adapter, lastParameterPos, filterToTarget);
407                             if (additionalCoordinates.size() > 0) {
408                                 adapter = joinDuplicateArgs(adapter, lastParameterPos,
409                                         lastParameterPos + additionalCoordinates.size() + 1,
410                                         additionalCoordinates.size());
411                             }
412                             MethodHandle res = MethodHandles.collectArguments(adapter, lastParameterPos - 1, filterToTarget);
413                             if (additionalCoordinates.size() > 0) {
414                                 res = joinDuplicateArgs(res, lastParameterPos - 1,
415                                         lastParameterPos + additionalCoordinates.size(),
416                                         additionalCoordinates.size());
417                             }
418                             yield res;
419                         }
420                         case COMPARE_AND_SET -> {
421                             MethodHandle adapter = MethodHandles.collectArguments(modeHandle, lastParameterPos, filterToTarget);
422                             MethodHandle res = MethodHandles.collectArguments(adapter, lastParameterPos - 1, filterToTarget);
423                             if (additionalCoordinates.size() > 0) {
424                                 res = joinDuplicateArgs(res, lastParameterPos - 1,
425                                         lastParameterPos + additionalCoordinates.size(),
426                                         additionalCoordinates.size());
427                             }
428                             yield res;
429                         }
430                     };
431                 });
432     }

434     private static MethodHandle joinDuplicateArgs(MethodHandle handle, int originalStart, int dropStart, int length) {
435         int[] perms = new int[handle.type().parameterCount()];
436         for (int i = 0 ; i < dropStart; i++) {
437             perms[i] = i;
438         }
439         for (int i = 0 ; i < length ; i++) {
440             perms[dropStart + i] = originalStart + i;
441         }
442         for (int i = dropStart + length ; i < perms.length ; i++) {
443             perms[i] = i - length;
444         }
445         return MethodHandles.permuteArguments(handle,
446                 handle.type().dropParameterTypes(dropStart, dropStart + length),
447                 perms);
448     }

450     public static VarHandle filterCoordinates(VarHandle target, int pos, MethodHandle... filters) {
451         Objects.nonNull(target);
452         Objects.nonNull(filters);
453 
454         List<Class<?>> targetCoordinates = target.coordinateTypes();
455         if (pos < 0 || pos >= targetCoordinates.size()) {
456             throw newIllegalArgumentException("Invalid position " + pos + " for coordinate types", targetCoordinates);
457         } else if (pos + filters.length > targetCoordinates.size()) {
458             throw new IllegalArgumentException("Too many filters");
459         }
460 
461         if (filters.length == 0) return target;
462 
463         List<Class<?>> newCoordinates = new ArrayList<>(targetCoordinates);
464         for (int i = 0 ; i < filters.length ; i++) {
465             noCheckedExceptions(filters[i]);
466             MethodType filterType = filters[i].type();
467             if (filterType.parameterCount() != 1) {
468                 throw newIllegalArgumentException("Invalid filter type " + filterType);
469             } else if (newCoordinates.get(pos + i) != filterType.returnType()) {
470                 throw newIllegalArgumentException("Invalid filter type " + filterType + " for coordinate type " + newCoordinates.get(i));
471             }
472             newCoordinates.set(pos + i, filters[i].type().parameterType(0));
473         }
474 
475         return new IndirectVarHandle(target, target.varType(), newCoordinates.toArray(new Class<?>[0]),
476                 (mode, modeHandle) -> MethodHandles.filterArguments(modeHandle, 1 + pos, filters));
477     }

479     public static VarHandle insertCoordinates(VarHandle target, int pos, Object... values) {
480         Objects.nonNull(target);
481         Objects.nonNull(values);
482 
483         List<Class<?>> targetCoordinates = target.coordinateTypes();
484         if (pos < 0 || pos >= targetCoordinates.size()) {
485             throw newIllegalArgumentException("Invalid position " + pos + " for coordinate types", targetCoordinates);
486         } else if (pos + values.length > targetCoordinates.size()) {
487             throw new IllegalArgumentException("Too many values");
488         }
489 
490         if (values.length == 0) return target;
491 
492         List<Class<?>> newCoordinates = new ArrayList<>(targetCoordinates);
493         for (int i = 0 ; i < values.length ; i++) {
494             Class<?> pt = newCoordinates.get(pos);
495             if (pt.isPrimitive()) {
496                 Wrapper w = Wrapper.forPrimitiveType(pt);
497                 w.convert(values[i], pt);
498             } else {
499                 pt.cast(values[i]);
500             }
501             newCoordinates.remove(pos);
502         }
503 
504         return new IndirectVarHandle(target, target.varType(), newCoordinates.toArray(new Class<?>[0]),
505                 (mode, modeHandle) -> MethodHandles.insertArguments(modeHandle, 1 + pos, values));
506     }

508     public static VarHandle permuteCoordinates(VarHandle target, List<Class<?>> newCoordinates, int... reorder) {
509         Objects.nonNull(target);
510         Objects.nonNull(newCoordinates);
511         Objects.nonNull(reorder);
512 
513         List<Class<?>> targetCoordinates = target.coordinateTypes();
514         MethodHandles.permuteArgumentChecks(reorder,
515                 MethodType.methodType(void.class, newCoordinates),
516                 MethodType.methodType(void.class, targetCoordinates));
517 
518         return new IndirectVarHandle(target, target.varType(), newCoordinates.toArray(new Class<?>[0]),
519                 (mode, modeHandle) ->
520                         MethodHandles.permuteArguments(modeHandle,
521                                 methodTypeFor(mode.at, modeHandle.type(), targetCoordinates, newCoordinates),
522                                 reorderArrayFor(mode.at, newCoordinates, reorder)));
523     }

525     private static int numTrailingArgs(VarHandle.AccessType at) {
526         return switch (at) {
527             case GET -> 0;
528             case GET_AND_UPDATE, SET -> 1;
529             case COMPARE_AND_SET, COMPARE_AND_EXCHANGE -> 2;
530         };
531     }
532 
533     private static int[] reorderArrayFor(VarHandle.AccessType at, List<Class<?>> newCoordinates, int[] reorder) {
534         int numTrailingArgs = numTrailingArgs(at);
535         int[] adjustedReorder = new int[reorder.length + 1 + numTrailingArgs];
536         adjustedReorder[0] = 0;
537         for (int i = 0 ; i < reorder.length ; i++) {
538             adjustedReorder[i + 1] = reorder[i] + 1;
539         }
540         for (int i = 0 ; i < numTrailingArgs ; i++) {
541             adjustedReorder[i + reorder.length + 1] = i + newCoordinates.size() + 1;
542         }
543         return adjustedReorder;
544     }

546     private static MethodType methodTypeFor(VarHandle.AccessType at, MethodType oldType, List<Class<?>> oldCoordinates, List<Class<?>> newCoordinates) {
547         int numTrailingArgs = numTrailingArgs(at);
548         MethodType adjustedType = MethodType.methodType(oldType.returnType(), oldType.parameterType(0));
549         adjustedType = adjustedType.appendParameterTypes(newCoordinates);
550         for (int i = 0 ; i < numTrailingArgs ; i++) {
551             adjustedType = adjustedType.appendParameterTypes(oldType.parameterType(1 + oldCoordinates.size() + i));
552         }
553         return adjustedType;
554     }

556     public static VarHandle collectCoordinates(VarHandle target, int pos, MethodHandle filter) {
557         Objects.nonNull(target);
558         Objects.nonNull(filter);
559         noCheckedExceptions(filter);
560 
561         List<Class<?>> targetCoordinates = target.coordinateTypes();
562         if (pos < 0 || pos >= targetCoordinates.size()) {
563             throw newIllegalArgumentException("Invalid position " + pos + " for coordinate types", targetCoordinates);
564         } else if (filter.type().returnType() == void.class) {
565             throw newIllegalArgumentException("Invalid filter type " + filter.type() + " ; filter cannot be void");
566         } else if (filter.type().returnType() != targetCoordinates.get(pos)) {
567             throw newIllegalArgumentException("Invalid filter type " + filter.type() + " for coordinate type " + targetCoordinates.get(pos));
568         }
569 
570         List<Class<?>> newCoordinates = new ArrayList<>(targetCoordinates);
571         newCoordinates.remove(pos);
572         newCoordinates.addAll(pos, filter.type().parameterList());
573 
574         return new IndirectVarHandle(target, target.varType(), newCoordinates.toArray(new Class<?>[0]),
575                 (mode, modeHandle) -> MethodHandles.collectArguments(modeHandle, 1 + pos, filter));
576     }

578     public static VarHandle dropCoordinates(VarHandle target, int pos, Class<?>... valueTypes) {
579         Objects.nonNull(target);
580         Objects.nonNull(valueTypes);
581 
582         List<Class<?>> targetCoordinates = target.coordinateTypes();
583         if (pos < 0 || pos > targetCoordinates.size()) {
584             throw newIllegalArgumentException("Invalid position " + pos + " for coordinate types", targetCoordinates);
585         }
586 
587         if (valueTypes.length == 0) return target;
588 
589         List<Class<?>> newCoordinates = new ArrayList<>(targetCoordinates);
590         newCoordinates.addAll(pos, List.of(valueTypes));
591 
592         return new IndirectVarHandle(target, target.varType(), newCoordinates.toArray(new Class<?>[0]),
593                 (mode, modeHandle) -> MethodHandles.dropArguments(modeHandle, 1 + pos, valueTypes));
594     }

596     private static void noCheckedExceptions(MethodHandle handle) {
597         if (handle instanceof DirectMethodHandle) {
598             DirectMethodHandle directHandle = (DirectMethodHandle)handle;
599             byte refKind = directHandle.member.getReferenceKind();
600             MethodHandleInfo info = new InfoFromMemberName(
601                     MethodHandles.Lookup.IMPL_LOOKUP,
602                     directHandle.member,
603                     refKind);
604             final Class<?>[] exceptionTypes;
605             if (MethodHandleNatives.refKindIsMethod(refKind)) {
606                 exceptionTypes = info.reflectAs(Method.class, MethodHandles.Lookup.IMPL_LOOKUP)
607                         .getExceptionTypes();
608             } else if (MethodHandleNatives.refKindIsField(refKind)) {
609                 exceptionTypes = null;
610             } else if (MethodHandleNatives.refKindIsConstructor(refKind)) {
611                 exceptionTypes = info.reflectAs(Constructor.class, MethodHandles.Lookup.IMPL_LOOKUP)
612                         .getExceptionTypes();
613             } else {
614                 throw new AssertionError("Cannot get here");
615             }
616             if (exceptionTypes != null) {
617                 if (Stream.of(exceptionTypes).anyMatch(VarHandles::isCheckedException)) {
618                     throw newIllegalArgumentException("Cannot adapt a var handle with a method handle which throws checked exceptions");
619                 }
620             }
621         } else if (handle instanceof DelegatingMethodHandle) {
622             noCheckedExceptions(((DelegatingMethodHandle)handle).getTarget());
623         } else {
624             //bound
625             BoundMethodHandle boundHandle = (BoundMethodHandle)handle;
626             for (int i = 0 ; i < boundHandle.fieldCount() ; i++) {
627                 Object arg = boundHandle.arg(i);
628                 if (arg instanceof MethodHandle){
629                     noCheckedExceptions((MethodHandle) arg);
630                 }
631             }
632         }
633     }

635     private static boolean isCheckedException(Class<?> clazz) {
636         return Throwable.class.isAssignableFrom(clazz) &&
637                 !RuntimeException.class.isAssignableFrom(clazz) &&
638                 !Error.class.isAssignableFrom(clazz);
