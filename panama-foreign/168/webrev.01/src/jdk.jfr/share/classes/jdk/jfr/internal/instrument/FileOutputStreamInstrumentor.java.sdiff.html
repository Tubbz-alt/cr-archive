 26 package jdk.jfr.internal.instrument;
 27 
 28 import java.io.IOException;
 29 
 30 import jdk.jfr.events.FileWriteEvent;

 31 
 32 /**
 33  * See {@link JITracer} for an explanation of this code.
 34  */
 35 @JIInstrumentationTarget("java.io.FileOutputStream")
 36 final class FileOutputStreamInstrumentor {
 37 
 38     private FileOutputStreamInstrumentor() {
 39     }
 40 
 41     private String path;
 42 
 43     @SuppressWarnings("deprecation")
 44     @JIInstrumentationMethod
 45     public void write(int b) throws IOException {
 46         FileWriteEvent event = FileWriteEvent.EVENT.get();
 47         if (!event.isEnabled()) {
 48             write(b);
 49             return;
 50         }


 51         try {
 52             event.begin();
 53             write(b);
 54             event.bytesWritten = 1;
 55         } finally {
 56             event.path = path;
 57             event.commit();
 58             event.reset();

 59         }
 60     }
 61 
 62     @SuppressWarnings("deprecation")
 63     @JIInstrumentationMethod
 64     public void write(byte b[]) throws IOException {
 65         FileWriteEvent event = FileWriteEvent.EVENT.get();
 66         if (!event.isEnabled()) {
 67             write(b);
 68             return;
 69         }


 70         try {
 71             event.begin();
 72             write(b);
 73             event.bytesWritten = b.length;
 74         } finally {
 75             event.path = path;
 76             event.commit();
 77             event.reset();

 78         }
 79     }
 80 
 81     @SuppressWarnings("deprecation")
 82     @JIInstrumentationMethod
 83     public void write(byte b[], int off, int len) throws IOException {
 84         FileWriteEvent event = FileWriteEvent.EVENT.get();
 85         if (!event.isEnabled()) {
 86             write(b, off, len);
 87             return;
 88         }


 89         try {
 90             event.begin();
 91             write(b, off, len);
 92             event.bytesWritten = len;
 93         } finally {
 94             event.path = path;
 95             event.commit();
 96             event.reset();

 97         }
 98     }
 99 }

 26 package jdk.jfr.internal.instrument;
 27 
 28 import java.io.IOException;
 29 
 30 import jdk.jfr.events.Handlers;
 31 import jdk.jfr.internal.handlers.EventHandler;
 32 
 33 /**
 34  * See {@link JITracer} for an explanation of this code.
 35  */
 36 @JIInstrumentationTarget("java.io.FileOutputStream")
 37 final class FileOutputStreamInstrumentor {
 38 
 39     private FileOutputStreamInstrumentor() {
 40     }
 41 
 42     private String path;
 43 
 44     @SuppressWarnings("deprecation")
 45     @JIInstrumentationMethod
 46     public void write(int b) throws IOException {
 47         EventHandler handler = Handlers.FILE_WRITE;
 48         if (!handler.isEnabled()) {
 49             write(b);
 50             return;
 51         }
 52         long bytesWritten = 0;
 53         long start = 0;
 54         try {
 55             start = EventHandler.timestamp();
 56             write(b);
 57             bytesWritten = 1;
 58         } finally {
 59             long duration = EventHandler.timestamp() - start;
 60             if (handler.shouldCommit(duration)) {
 61                 handler.write(start, duration, path, bytesWritten);
 62             }
 63         }
 64     }
 65 
 66     @SuppressWarnings("deprecation")
 67     @JIInstrumentationMethod
 68     public void write(byte b[]) throws IOException {
 69         EventHandler handler = Handlers.FILE_WRITE;
 70         if (!handler.isEnabled()) {
 71             write(b);
 72             return;
 73         }
 74         long bytesWritten = 0;
 75         long start = 0;
 76         try {
 77             start = EventHandler.timestamp();
 78             write(b);
 79             bytesWritten = b.length;
 80         } finally {
 81             long duration = EventHandler.timestamp() - start;
 82             if (handler.shouldCommit(duration)) {
 83                 handler.write(start, duration, path, bytesWritten);
 84             }
 85         }
 86     }
 87 
 88     @SuppressWarnings("deprecation")
 89     @JIInstrumentationMethod
 90     public void write(byte b[], int off, int len) throws IOException {
 91         EventHandler handler = Handlers.FILE_WRITE;
 92         if (!handler.isEnabled()) {
 93             write(b, off, len);
 94             return;
 95         }
 96         long bytesWritten = 0;
 97         long start = 0;
 98         try {
 99             start = EventHandler.timestamp();
100             write(b, off, len);
101             bytesWritten = len;
102         } finally {
103             long duration = EventHandler.timestamp() - start;
104             if (handler.shouldCommit(duration)) {
105                 handler.write(start, duration, path, bytesWritten);
106             }
107         }
108     }
109 }
