 45 final class DCmdConfigure extends AbstractDCmd {
 46     /**
 47      * Execute JFR.configure.
 48      *
 49      * @param repositoryPath the path
 50      * @param dumpPath path to dump to on fatal error (oom)
 51      * @param stackDepth depth of stack traces
 52      * @param globalBufferCount number of global buffers
 53      * @param globalBufferSize size of global buffers
 54      * @param threadBufferSize size of thread buffer for events
 55      * @param maxChunkSize threshold at which a new chunk is created in the disk repository
 56      * @param sampleThreads if thread sampling should be enabled
 57      *
 58      * @return result
 59 
 60      * @throws DCmdException
 61      *             if the dump could not be completed
 62      */
 63     public String execute
 64     (

 65             String repositoryPath,
 66             String dumpPath,
 67             Integer stackDepth,
 68             Long globalBufferCount,
 69             Long globalBufferSize,
 70             Long threadBufferSize,
 71             Long memorySize,
 72             Long maxChunkSize,
 73             Boolean sampleThreads
 74 
 75     ) throws DCmdException {
 76         if (Logger.shouldLog(LogTag.JFR_DCMD, LogLevel.DEBUG)) {
 77             Logger.log(LogTag.JFR_DCMD, LogLevel.DEBUG, "Executing DCmdConfigure: repositorypath=" + repositoryPath +
 78                     ", dumppath=" + dumpPath +
 79                     ", stackdepth=" + stackDepth +
 80                     ", globalbuffercount=" + globalBufferCount +
 81                     ", globalbuffersize=" + globalBufferSize +
 82                     ", thread_buffer_size=" + threadBufferSize +
 83                     ", memorysize=" + memorySize +
 84                     ", maxchunksize=" + maxChunkSize +
 85                     ", samplethreads=" + sampleThreads);
 86         }
 87 
 88 
 89         boolean updated = false;
 90         if (repositoryPath != null) {
 91             try {
 92                 SafePath s = new SafePath(repositoryPath);
 93                 if (FlightRecorder.isInitialized()) {
 94                     PrivateAccess.getInstance().getPlatformRecorder().migrate(s);
 95                 } else {
 96                     Repository.getRepository().setBasePath(s);
 97                 }
 98                 Logger.log(LogTag.JFR, LogLevel.INFO, "Base repository path set to " + repositoryPath);
 99             } catch (Exception e) {
100                 throw new DCmdException("Could not use " + repositoryPath + " as repository. " + e.getMessage(), e);
101             }
102             printRepositoryPath();


103             updated = true;
104         }
105 
106         if (dumpPath != null)  {
107             Options.setDumpPath(new SafePath(dumpPath));
108             Logger.log(LogTag.JFR, LogLevel.INFO, "Emergency dump path set to " + dumpPath);
109             printDumpPath();


110             updated = true;
111         }
112 
113         if (stackDepth != null)  {
114             Options.setStackDepth(stackDepth);
115             Logger.log(LogTag.JFR, LogLevel.INFO, "Stack depth set to " + stackDepth);
116             printStackDepth();


117             updated = true;
118         }
119 
120         if (globalBufferCount != null)  {
121             Options.setGlobalBufferCount(globalBufferCount);
122             Logger.log(LogTag.JFR, LogLevel.INFO, "Global buffer count set to " + globalBufferCount);
123             printGlobalBufferCount();


124             updated = true;
125         }
126 
127         if (globalBufferSize != null)  {
128             Options.setGlobalBufferSize(globalBufferSize);
129             Logger.log(LogTag.JFR, LogLevel.INFO, "Global buffer size set to " + globalBufferSize);
130             printGlobalBufferSize();


131             updated = true;
132         }
133 
134         if (threadBufferSize != null)  {
135             Options.setThreadBufferSize(threadBufferSize);
136             Logger.log(LogTag.JFR, LogLevel.INFO, "Thread buffer size set to " + threadBufferSize);
137             printThreadBufferSize();


138             updated = true;
139         }
140 
141         if (memorySize != null) {
142             Options.setMemorySize(memorySize);
143             Logger.log(LogTag.JFR, LogLevel.INFO, "Memory size set to " + memorySize);
144             printMemorySize();


145             updated = true;
146         }
147 
148         if (maxChunkSize != null)  {
149             Options.setMaxChunkSize(maxChunkSize);
150             Logger.log(LogTag.JFR, LogLevel.INFO, "Max chunk size set to " + maxChunkSize);
151             printMaxChunkSize();


152             updated = true;
153         }
154 
155         if (sampleThreads != null)  {
156             Options.setSampleThreads(sampleThreads);
157             Logger.log(LogTag.JFR, LogLevel.INFO, "Sample threads set to " + sampleThreads);
158             printSampleThreads();


159             updated = true;
160         }
161 


162         if (!updated) {
163             println("Current configuration:");
164             println();
165             printRepositoryPath();
166             printStackDepth();
167             printGlobalBufferCount();
168             printGlobalBufferSize();
169             printThreadBufferSize();
170             printMemorySize();
171             printMaxChunkSize();
172             printSampleThreads();
173         }
174         return getResult();
175     }
176 
177     private void printRepositoryPath() {
178         print("Repository path: ");
179         printPath(Repository.getRepository().getRepositoryPath());
180         println();
181     }
