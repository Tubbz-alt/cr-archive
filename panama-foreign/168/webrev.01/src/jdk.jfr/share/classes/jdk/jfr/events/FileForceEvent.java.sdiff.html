28 import jdk.jfr.Category;
29 import jdk.jfr.Description;
30 import jdk.jfr.Label;
31 import jdk.jfr.Name;
32 import jdk.jfr.internal.Type;
33 
34 @Name(Type.EVENT_NAME_PREFIX + "FileForce")
35 @Label("File Force")
36 @Category("Java Application")
37 @Description("Force updates to be written to file")
38 public final class FileForceEvent extends AbstractJDKEvent {
39 
40     public static final ThreadLocal<FileForceEvent> EVENT =
41         new ThreadLocal<>() {
42             @Override protected FileForceEvent initialValue() {
43                 return new FileForceEvent();
44             }
45         };
46 
47     @Label("Path")
48     @Description("Full path of the file")
49     public String path;
50 
51     @Label("Update Metadata")
52     @Description("Whether the file metadata is updated")
53     public boolean metaData;
54 
55     public void reset() {
56         path = null;
57         metaData = false;
58     }
59 }
