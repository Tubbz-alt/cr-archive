26 package jdk.jfr.internal.instrument;
27 
28 import java.util.concurrent.atomic.AtomicLong;
29 
30 import jdk.jfr.events.ErrorThrownEvent;
31 import jdk.jfr.events.ExceptionThrownEvent;
32 
33 public final class ThrowableTracer {
34 
35     private static AtomicLong numThrowables = new AtomicLong(0);
36 
37     public static void traceError(Error e, String message) {
38         if (e instanceof OutOfMemoryError) {
39             return;
40         }
41         ErrorThrownEvent errorEvent = new ErrorThrownEvent();
42         errorEvent.message = message;
43         errorEvent.thrownClass = e.getClass();
44         errorEvent.commit();
45 
46         ExceptionThrownEvent exceptionEvent = new ExceptionThrownEvent();
47         exceptionEvent.message = message;
48         exceptionEvent.thrownClass = e.getClass();
49         exceptionEvent.commit();
50         numThrowables.incrementAndGet();
51     }
52 
53     public static void traceThrowable(Throwable t, String message) {
54         ExceptionThrownEvent event = new ExceptionThrownEvent();
55         event.message = message;
56         event.thrownClass = t.getClass();
57         event.commit();
58         numThrowables.incrementAndGet();
59     }
60 
61     public static long numThrowables() {
62         return numThrowables.get();
63     }
64 }
