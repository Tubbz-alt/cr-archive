26 package jdk.jfr.events;
27 
28 import jdk.jfr.Category;
29 import jdk.jfr.Description;
30 import jdk.jfr.Label;
31 import jdk.jfr.DataAmount;
32 import jdk.jfr.Name;
33 import jdk.jfr.Timespan;
34 import jdk.jfr.internal.Type;
35 
36 @Name(Type.EVENT_NAME_PREFIX + "SocketRead")
37 @Label("Socket Read")
38 @Category("Java Application")
39 @Description("Reading data from a socket")
40 public final class SocketReadEvent extends AbstractJDKEvent {
41 
42     public static final ThreadLocal<SocketReadEvent> EVENT =
43         new ThreadLocal<>() {
44             @Override protected SocketReadEvent initialValue() {
45                 return new SocketReadEvent();
46             }
47         };
48 
49     @Label("Remote Host")
50     public String host;
51 
52     @Label("Remote Address")
53     public String address;
54 
55     @Label("Remote Port")
56     public int port;
57 
58     @Label("Timeout Value")
59     @Timespan(Timespan.MILLISECONDS)
60     public long timeout;
61 
62     @Label("Bytes Read")
63     @Description("Number of bytes read from the socket")
64     @DataAmount
65     public long bytesRead;
66 
67     @Label("End of Stream")
68     @Description("If end of stream was reached")
69     public boolean endOfStream;
70 
71     public void reset() {
72         host = null;
73         address = null;
74         port = 0;
75         timeout = 0;
76         bytesRead = 0L;
77         endOfStream = false;
78     }
79 }
