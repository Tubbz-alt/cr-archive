26 package jdk.jfr.events;
27 
28 import jdk.jfr.Category;
29 import jdk.jfr.Description;
30 import jdk.jfr.Label;
31 import jdk.jfr.DataAmount;
32 import jdk.jfr.Name;
33 import jdk.jfr.internal.Type;
34 
35 @Name(Type.EVENT_NAME_PREFIX + "FileRead")
36 @Label("File Read")
37 @Category("Java Application")
38 @Description("Reading data from a file")
39 public final class FileReadEvent extends AbstractJDKEvent {
40 
41     public static final ThreadLocal<FileReadEvent> EVENT =
42         new ThreadLocal<>() {
43             @Override protected FileReadEvent initialValue() {
44                 return new FileReadEvent();
45             }
46         };
47 
48     @Label("Path")
49     @Description("Full path of the file")
50     public String path;
51 
52     @Label("Bytes Read")
53     @Description("Number of bytes read from the file (possibly 0)")
54     @DataAmount
55     public long bytesRead;
56 
57     @Label("End of File")
58     @Description("If end of file was reached")
59     public boolean endOfFile;
60 
61     public void reset() {
62         path = null;
63         endOfFile = false;
64         bytesRead = 0;
65     }
66 }
