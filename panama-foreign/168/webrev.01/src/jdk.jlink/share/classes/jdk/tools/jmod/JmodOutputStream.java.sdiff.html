100     public void writeEntry(InputStream in, Entry e) throws IOException {
101         ZipEntry e1 = e.zipEntry();
102         // Only preserve attributes which won't change by
103         // inflating and deflating the entry. See:
104         // sun.tools.jar.Main.update()
105         ZipEntry e2 = new ZipEntry(e1.getName());
106         e2.setMethod(e1.getMethod());
107         e2.setTime(e1.getTime());
108         e2.setComment(e1.getComment());
109         e2.setExtra(e1.getExtra());
110         if (e1.getMethod() == ZipEntry.STORED) {

119     private ZipEntry newEntry(Section section, String path) throws IOException {
120         if (contains(section, path)) {
121             throw new IOException("duplicate entry: " + path + " in section " + section);
122         }
123         String prefix = section.jmodDir();
124         String name = Paths.get(prefix, path).toString()
125                            .replace(File.separatorChar, '/');
126         entries.get(section).add(path);
127         return new ZipEntry(name);
128     }
129 
130     public boolean contains(Section section, String path) {
131         Set<String> set = entries.computeIfAbsent(section, k -> new HashSet<>());
132         return set.contains(path);
133     }
