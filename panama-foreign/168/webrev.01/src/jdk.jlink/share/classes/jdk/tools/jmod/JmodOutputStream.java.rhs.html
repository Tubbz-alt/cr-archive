 26 package jdk.tools.jmod;
 27 
 28 import java.io.BufferedOutputStream;
 29 import java.io.File;
 30 import java.io.IOException;
 31 import java.io.InputStream;
 32 import java.io.OutputStream;
 33 import java.io.UncheckedIOException;
 34 import java.nio.file.Files;
 35 import java.nio.file.Path;
 36 import java.nio.file.Paths;
 37 import java.util.HashMap;
 38 import java.util.HashSet;
 39 import java.util.Map;
 40 import java.util.Set;
 41 import java.util.zip.ZipEntry;
 42 import java.util.zip.ZipOutputStream;
 43 import jdk.internal.jmod.JmodFile;
 44 
 45 import static jdk.internal.jmod.JmodFile.*;
 46 
 47 /**
 48  * Output stream to write to JMOD file
 49  */
 50 class JmodOutputStream extends OutputStream implements AutoCloseable {
 51     private final Map<Section, Set<String>> entries = new HashMap<>();
 52 
 53     /**
 54      * This method creates (or overrides, if exists) the JMOD file,
 55      * returning the the output stream to write to the JMOD file.
 56      */
 57     static JmodOutputStream newOutputStream(Path file) throws IOException {
 58         OutputStream out = Files.newOutputStream(file);
 59         BufferedOutputStream bos = new BufferedOutputStream(out);
 60         return new JmodOutputStream(bos);
 61     }
 62 
 63     private final ZipOutputStream zos;
 64     private JmodOutputStream(OutputStream out) {
 65         this.zos = new ZipOutputStream(out);
 66         try {
 67             JmodFile.writeMagicNumber(out);
 68         } catch (IOException e) {
 69             throw new UncheckedIOException(e);
 70         }
 71     }
 72 
 73     /**
 74      * Writes the input stream to the named entry of the given section.
 75      */
 76     public void writeEntry(InputStream in, Section section, String name)
 77         throws IOException
 78     {
 79         ZipEntry ze = newEntry(section, name);
 80         zos.putNextEntry(ze);
 81         in.transferTo(zos);
 82         zos.closeEntry();
 83     }
 84 
 85     /**
 86      * Writes the given bytes to the named entry of the given section.
 87      */
 88     public void writeEntry(byte[] bytes, Section section, String path)
 89         throws IOException
 90     {
 91         ZipEntry ze = newEntry(section, path);
 92         zos.putNextEntry(ze);
 93         zos.write(bytes);
 94         zos.closeEntry();
 95     }
 96 
 97     /**
 98      * Writes the given entry to the given input stream.
 99      */
100     public void writeEntry(InputStream in, Entry e) throws IOException {
101         ZipEntry e1 = e.zipEntry();
102         // Only preserve attributes which won't change by
103         // inflating and deflating the entry. See:
104         // sun.tools.jar.Main.update()
105         ZipEntry e2 = new ZipEntry(e1.getName());
106         e2.setMethod(e1.getMethod());
107         e2.setTime(e1.getTime());
108         e2.setComment(e1.getComment());
109         e2.setExtra(e1.getExtra());
110         if (e1.getMethod() == ZipEntry.STORED) {
111             e2.setSize(e1.getSize());
112             e2.setCrc(e1.getCrc());
113         }
114         zos.putNextEntry(e2);
115         zos.write(in.readAllBytes());
116         zos.closeEntry();
117     }
118 
119     private ZipEntry newEntry(Section section, String path) throws IOException {
120         if (contains(section, path)) {
121             throw new IOException("duplicate entry: " + path + " in section " + section);
122         }
123         String prefix = section.jmodDir();
124         String name = Paths.get(prefix, path).toString()
125                            .replace(File.separatorChar, '/');
126         entries.get(section).add(path);
127         return new ZipEntry(name);
128     }
129 
130     public boolean contains(Section section, String path) {
131         Set<String> set = entries.computeIfAbsent(section, k -> new HashSet<>());
132         return set.contains(path);
133     }
134 
135     @Override
136     public void write(int b) throws IOException {
137         zos.write(b);
138     }
139 
140     @Override
141     public void close() throws IOException {
142         zos.close();
143     }
144 }
