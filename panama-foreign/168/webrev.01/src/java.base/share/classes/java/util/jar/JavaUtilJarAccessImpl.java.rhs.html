26 package java.util.jar;
27 
28 import java.io.IOException;
29 import java.net.URL;
30 import java.security.CodeSource;
31 import java.util.Enumeration;
32 import java.util.List;
33 import java.util.zip.ZipEntry;
34 import java.util.zip.ZipFile;
35 
36 import jdk.internal.access.JavaUtilJarAccess;
37 
38 class JavaUtilJarAccessImpl implements JavaUtilJarAccess {
39     public boolean jarFileHasClassPathAttribute(JarFile jar) throws IOException {
40         return jar.hasClassPathAttribute();
41     }
42 
43     public CodeSource[] getCodeSources(JarFile jar, URL url) {
44         return jar.getCodeSources(url);
45     }
46 
47     public CodeSource getCodeSource(JarFile jar, URL url, String name) {
48         return jar.getCodeSource(url, name);
49     }
50 
51     public Enumeration<String> entryNames(JarFile jar, CodeSource[] cs) {
52         return jar.entryNames(cs);
53     }
54 
55     public Enumeration<JarEntry> entries2(JarFile jar) {
56         return jar.entries2();
57     }
58 
59     public void setEagerValidation(JarFile jar, boolean eager) {
60         jar.setEagerValidation(eager);
61     }
62 
63     public List<Object> getManifestDigests(JarFile jar) {
64         return jar.getManifestDigests();
65     }
66 
67     public Attributes getTrustedAttributes(Manifest man, String name) {
68         return man.getTrustedAttributes(name);
69     }
70 
71     public void ensureInitialization(JarFile jar) {
72         jar.ensureInitialization();
73     }
74 
75     public boolean isInitializing() {
76         return JarFile.isInitializing();
77     }
78 
79     public JarEntry entryFor(JarFile jar, String name) {
80         return jar.entryFor(name);
81     }
82 }
