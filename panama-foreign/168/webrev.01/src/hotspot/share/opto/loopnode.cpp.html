  68 //------------------------------is_valid_counted_loop-------------------------
  69 bool LoopNode::is_valid_counted_loop() const {
  70   if (is_CountedLoop()) {
  71     CountedLoopNode*    l  = as_CountedLoop();
  72     CountedLoopEndNode* le = l->loopexit_or_null();
  73     if (le != NULL &&
  74         le->proj_out_or_null(1 /* true */) == l->in(LoopNode::LoopBackControl)) {
  75       Node* phi  = l->phi();
  76       Node* exit = le->proj_out_or_null(0 /* false */);
  77       if (exit != NULL && exit->Opcode() == Op_IfFalse &&
  78           phi != NULL && phi->is_Phi() &&
  79           phi->in(LoopNode::LoopBackControl) == l->incr() &&
  80           le->loopnode() == l && le->stride_is_con()) {
  81         return true;
  82       }
  83     }
  84   }
  85   return false;
  86 }

  88 //------------------------------get_early_ctrl---------------------------------
  89 // Compute earliest legal control
  90 Node *PhaseIdealLoop::get_early_ctrl( Node *n ) {
  91   assert( !n->is_Phi() && !n->is_CFG(), "this code only handles data nodes" );
  92   uint i;
  93   Node *early;
  94   if (n->in(0) && !n->is_expensive()) {
  95     early = n->in(0);
  96     if (!early->is_CFG()) // Might be a non-CFG multi-def
  97       early = get_ctrl(early);        // So treat input as a straight data input
  98     i = 1;
  99   } else {
 100     early = get_ctrl(n->in(1));
 101     i = 2;
 102   }
 103   uint e_d = dom_depth(early);
 104   assert( early, "" );
 105   for (; i < n->req(); i++) {
 106     Node *cin = get_ctrl(n->in(i));
 107     assert( cin, "" );
 108     // Keep deepest dominator depth
 109     uint c_d = dom_depth(cin);
 110     if (c_d > e_d) {           // Deeper guy?
 111       early = cin;              // Keep deepest found so far
 112       e_d = c_d;
 113     } else if (c_d == e_d &&    // Same depth?
 114                early != cin) { // If not equal, must use slower algorithm
 115       // If same depth but not equal, one _must_ dominate the other
 116       // and we want the deeper (i.e., dominated) guy.
 117       Node *n1 = early;
 118       Node *n2 = cin;
 119       while (1) {
 120         n1 = idom(n1);          // Walk up until break cycle
 121         n2 = idom(n2);
 122         if (n1 == cin ||        // Walked early up to cin
 123             dom_depth(n2) < c_d)
 124           break;                // early is deeper; keep him
 125         if (n2 == early ||      // Walked cin up to early
 126             dom_depth(n1) < c_d) {
 127           early = cin;          // cin is deeper; keep him
 128           break;
 129         }
 130       }
 131       e_d = dom_depth(early);   // Reset depth register cache
 132     }
 133   }
 134 
 135   // Return earliest legal location
 136   assert(early == find_non_split_ctrl(early), "unexpected early control");
 137 
 138   if (n->is_expensive() && !_verify_only && !_verify_me) {
 139     assert(n->in(0), "should have control input");
 140     early = get_early_ctrl_for_expensive(n, early);
 141   }
 142 
 143   return early;
 144 }

 146 //------------------------------get_early_ctrl_for_expensive---------------------------------
 147 // Move node up the dominator tree as high as legal while still beneficial
 148 Node *PhaseIdealLoop::get_early_ctrl_for_expensive(Node *n, Node* earliest) {
 149   assert(n->in(0) && n->is_expensive(), "expensive node with control input here");
 150   assert(OptimizeExpensiveOps, "optimization off?");
 151 
 152   Node* ctl = n->in(0);
 153   assert(ctl->is_CFG(), "expensive input 0 must be cfg");
 154   uint min_dom_depth = dom_depth(earliest);
 155 #ifdef ASSERT
 156   if (!is_dominator(ctl, earliest) && !is_dominator(earliest, ctl)) {
 157     dump_bad_graph("Bad graph detected in get_early_ctrl_for_expensive", n, earliest, ctl);
 158     assert(false, "Bad graph detected in get_early_ctrl_for_expensive");
 159   }
 160 #endif
 161   if (dom_depth(ctl) < min_dom_depth) {
 162     return earliest;
 163   }
 164 
 165   while (1) {
 166     Node *next = ctl;
 167     // Moving the node out of a loop on the projection of a If
 168     // confuses loop predication. So once we hit a Loop in a If branch
 169     // that doesn't branch to an UNC, we stop. The code that process
 170     // expensive nodes will notice the loop and skip over it to try to
 171     // move the node further up.
 172     if (ctl->is_CountedLoop() && ctl->in(1) != NULL && ctl->in(1)->in(0) != NULL && ctl->in(1)->in(0)->is_If()) {
 173       if (!ctl->in(1)->as_Proj()->is_uncommon_trap_if_pattern(Deoptimization::Reason_none)) {
 174         break;
 175       }
 176       next = idom(ctl->in(1)->in(0));
 177     } else if (ctl->is_Proj()) {
 178       // We only move it up along a projection if the projection is
 179       // the single control projection for its parent: same code path,
 180       // if it's a If with UNC or fallthrough of a call.
 181       Node* parent_ctl = ctl->in(0);
 182       if (parent_ctl == NULL) {
 183         break;
 184       } else if (parent_ctl->is_CountedLoopEnd() && parent_ctl->as_CountedLoopEnd()->loopnode() != NULL) {
 185         next = parent_ctl->as_CountedLoopEnd()->loopnode()->init_control();
 186       } else if (parent_ctl->is_If()) {
 187         if (!ctl->as_Proj()->is_uncommon_trap_if_pattern(Deoptimization::Reason_none)) {
 188           break;
 189         }
 190         assert(idom(ctl) == parent_ctl, "strange");
 191         next = idom(parent_ctl);
 192       } else if (ctl->is_CatchProj()) {
 193         if (ctl->as_Proj()->_con != CatchProjNode::fall_through_index) {
 194           break;
 195         }
 196         assert(parent_ctl->in(0)->in(0)->is_Call(), "strange graph");
 197         next = parent_ctl->in(0)->in(0)->in(0);
 198       } else {
 199         // Check if parent control has a single projection (this
 200         // control is the only possible successor of the parent
 201         // control). If so, we can try to move the node above the
 202         // parent control.
 203         int nb_ctl_proj = 0;
 204         for (DUIterator_Fast imax, i = parent_ctl->fast_outs(imax); i < imax; i++) {
 205           Node *p = parent_ctl->fast_out(i);
 206           if (p->is_Proj() && p->is_CFG()) {
 207             nb_ctl_proj++;
 208             if (nb_ctl_proj > 1) {
 209               break;
 210             }
 211           }
 212         }
 213 
 214         if (nb_ctl_proj > 1) {
 215           break;
 216         }
 217         assert(parent_ctl->is_Start() || parent_ctl->is_MemBar() || parent_ctl->is_Call() ||
 218                BarrierSet::barrier_set()->barrier_set_c2()->is_gc_barrier_node(parent_ctl), "unexpected node");
 219         assert(idom(ctl) == parent_ctl, "strange");
 220         next = idom(parent_ctl);
 221       }
 222     } else {
 223       next = idom(ctl);
 224     }
 225     if (next->is_Root() || next->is_Start() || dom_depth(next) < min_dom_depth) {
 226       break;
 227     }
 228     ctl = next;
 229   }
 230 
 231   if (ctl != n->in(0)) {
 232     _igvn.replace_input_of(n, 0, ctl);
 233     _igvn.hash_insert(n);
 234   }
 235 
 236   return ctl;
 237 }

 249 //------------------------------set_subtree_ctrl-------------------------------
 250 // set missing _ctrl entries on new nodes
 251 void PhaseIdealLoop::set_subtree_ctrl( Node *n ) {
 252   // Already set?  Get out.
 253   if( _nodes[n->_idx] ) return;
 254   // Recursively set _nodes array to indicate where the Node goes
 255   uint i;
 256   for( i = 0; i < n->req(); ++i ) {
 257     Node *m = n->in(i);
 258     if( m && m != C->root() )
 259       set_subtree_ctrl( m );
 260   }
 261 
 262   // Fixup self
 263   set_early_ctrl( n );
 264 }

 266 // Create a skeleton strip mined outer loop: a Loop head before the
 267 // inner strip mined loop, a safepoint and an exit condition guarded
 268 // by an opaque node after the inner strip mined loop with a backedge
 269 // to the loop head. The inner strip mined loop is left as it is. Only
 270 // once loop optimizations are over, do we adjust the inner loop exit
 271 // condition to limit its number of iterations, set the outer loop
 272 // exit condition and add Phis to the outer loop head. Some loop
 273 // optimizations that operate on the inner strip mined loop need to be
 274 // aware of the outer strip mined loop: loop unswitching needs to
 275 // clone the outer loop as well as the inner, unrolling needs to only
 276 // clone the inner loop etc. No optimizations need to change the outer
 277 // strip mined loop as it is only a skeleton.
 278 IdealLoopTree* PhaseIdealLoop::create_outer_strip_mined_loop(BoolNode *test, Node *cmp, Node *init_control,
 279                                                              IdealLoopTree* loop, float cl_prob, float le_fcnt,
 280                                                              Node*& entry_control, Node*& iffalse) {
 281   Node* outer_test = _igvn.intcon(0);
 282   set_ctrl(outer_test, C->root());
 283   Node *orig = iffalse;
 284   iffalse = iffalse->clone();
 285   _igvn.register_new_node_with_optimizer(iffalse);
 286   set_idom(iffalse, idom(orig), dom_depth(orig));
 287 
 288   IfNode *outer_le = new OuterStripMinedLoopEndNode(iffalse, outer_test, cl_prob, le_fcnt);
 289   Node *outer_ift = new IfTrueNode (outer_le);
 290   Node* outer_iff = orig;
 291   _igvn.replace_input_of(outer_iff, 0, outer_le);
 292 
 293   LoopNode *outer_l = new OuterStripMinedLoopNode(C, init_control, outer_ift);
 294   entry_control = outer_l;
 295 
 296   IdealLoopTree* outer_ilt = new IdealLoopTree(this, outer_l, outer_ift);
 297   IdealLoopTree* parent = loop->_parent;
 298   IdealLoopTree* sibling = parent->_child;
 299   if (sibling == loop) {
 300     parent->_child = outer_ilt;
 301   } else {
 302     while (sibling->_next != loop) {
 303       sibling = sibling->_next;
 304     }
 305     sibling->_next = outer_ilt;
 306   }
 307   outer_ilt->_next = loop->_next;
 308   outer_ilt->_parent = parent;
 309   outer_ilt->_child = loop;
 310   outer_ilt->_nest = loop->_nest;
 311   loop->_parent = outer_ilt;
 312   loop->_next = NULL;
 313   loop->_nest++;
 314 
 315   set_loop(iffalse, outer_ilt);
 316   register_control(outer_le, outer_ilt, iffalse);
 317   register_control(outer_ift, outer_ilt, outer_le);
 318   set_idom(outer_iff, outer_le, dom_depth(outer_le));
 319   _igvn.register_new_node_with_optimizer(outer_l);
 320   set_loop(outer_l, outer_ilt);
 321   set_idom(outer_l, init_control, dom_depth(init_control)+1);
 322 
 323   return outer_ilt;
 324 }

 326 void PhaseIdealLoop::insert_loop_limit_check(ProjNode* limit_check_proj, Node* cmp_limit, Node* bol) {
 327   Node* new_predicate_proj = create_new_if_for_predicate(limit_check_proj, NULL,
 328                                                          Deoptimization::Reason_loop_limit_check,
 329                                                          Op_If);
 330   Node* iff = new_predicate_proj->in(0);
 331   assert(iff->Opcode() == Op_If, "bad graph shape");
 332   Node* conv = iff->in(1);
 333   assert(conv->Opcode() == Op_Conv2B, "bad graph shape");
 334   Node* opaq = conv->in(1);
 335   assert(opaq->Opcode() == Op_Opaque1, "bad graph shape");
 336   cmp_limit = _igvn.register_new_node_with_optimizer(cmp_limit);
 337   bol = _igvn.register_new_node_with_optimizer(bol);
 338   set_subtree_ctrl(bol);
 339   _igvn.replace_input_of(iff, 1, bol);
 340 
 341 #ifndef PRODUCT
 342   // report that the loop predication has been actually performed
 343   // for this loop
 344   if (TraceLoopLimitCheck) {
 345     tty->print_cr("Counted Loop Limit Check generated:");
 346     debug_only( bol->dump(2); )
 347   }
 348 #endif
 349 }

 351 //------------------------------is_counted_loop--------------------------------
 352 bool PhaseIdealLoop::is_counted_loop(Node* x, IdealLoopTree*& loop) {
 353   PhaseGVN *gvn = &_igvn;
 354 
 355   // Counted loop head must be a good RegionNode with only 3 not NULL
 356   // control input edges: Self, Entry, LoopBack.
 357   if (x->in(LoopNode::Self) == NULL || x->req() != 3 || loop->_irreducible) {
 358     return false;
 359   }
 360   Node *init_control = x->in(LoopNode::EntryControl);
 361   Node *back_control = x->in(LoopNode::LoopBackControl);
 362   if (init_control == NULL || back_control == NULL)    // Partially dead
 363     return false;
 364   // Must also check for TOP when looking for a dead loop
 365   if (init_control->is_top() || back_control->is_top())
 366     return false;
 367 
 368   // Allow funny placement of Safepoint
 369   if (back_control->Opcode() == Op_SafePoint) {
 370     if (LoopStripMiningIter != 0) {
 371       // Leaving the safepoint on the backedge and creating a
 372       // CountedLoop will confuse optimizations. We can't move the
 373       // safepoint around because its jvm state wouldn't match a new
 374       // location. Give up on that loop.
 375       return false;
 376     }
 377     back_control = back_control->in(TypeFunc::Control);
 378   }
 379 
 380   // Controlling test for loop
 381   Node *iftrue = back_control;
 382   uint iftrue_op = iftrue->Opcode();
 383   if (iftrue_op != Op_IfTrue &&
 384       iftrue_op != Op_IfFalse)
 385     // I have a weird back-control.  Probably the loop-exit test is in
 386     // the middle of the loop and I am looking at some trailing control-flow
 387     // merge point.  To fix this I would have to partially peel the loop.
 388     return false; // Obscure back-control
 389 
 390   // Get boolean guarding loop-back test
 391   Node *iff = iftrue->in(0);
 392   if (get_loop(iff) != loop || !iff->in(1)->is_Bool())
 393     return false;
 394   BoolNode *test = iff->in(1)->as_Bool();
 395   BoolTest::mask bt = test->_test._test;
 396   float cl_prob = iff->as_If()->_prob;
 397   if (iftrue_op == Op_IfFalse) {
 398     bt = BoolTest(bt).negate();
 399     cl_prob = 1.0 - cl_prob;
 400   }
 401   // Get backedge compare
 402   Node *cmp = test->in(1);
 403   int cmp_op = cmp->Opcode();
 404   if (cmp_op != Op_CmpI)
 405     return false;                // Avoid pointer & float compares
 406 
 407   // Find the trip-counter increment & limit.  Limit must be loop invariant.
 408   Node *incr  = cmp->in(1);
 409   Node *limit = cmp->in(2);
 410 
 411   // ---------
 412   // need 'loop()' test to tell if limit is loop invariant
 413   // ---------
 414 
 415   if (!is_member(loop, get_ctrl(incr))) { // Swapped trip counter and limit?
 416     Node *tmp = incr;            // Then reverse order into the CmpI
 417     incr = limit;
 418     limit = tmp;
 419     bt = BoolTest(bt).commute(); // And commute the exit test
 420   }
 421   if (is_member(loop, get_ctrl(limit))) // Limit must be loop-invariant
 422     return false;
 423   if (!is_member(loop, get_ctrl(incr))) // Trip counter must be loop-variant
 424     return false;
 425 
 426   Node* phi_incr = NULL;
 427   // Trip-counter increment must be commutative & associative.
 428   if (incr->Opcode() == Op_CastII) {
 429     incr = incr->in(1);
 430   }
 431   if (incr->is_Phi()) {
 432     if (incr->as_Phi()->region() != x || incr->req() != 3)
 433       return false; // Not simple trip counter expression
 434     phi_incr = incr;
 435     incr = phi_incr->in(LoopNode::LoopBackControl); // Assume incr is on backedge of Phi
 436     if (!is_member(loop, get_ctrl(incr))) // Trip counter must be loop-variant
 437       return false;
 438   }
 439 
 440   Node* trunc1 = NULL;
 441   Node* trunc2 = NULL;
 442   const TypeInt* iv_trunc_t = NULL;
 443   Node* orig_incr = incr;
 444   if (!(incr = CountedLoopNode::match_incr_with_optional_truncation(incr, &trunc1, &trunc2, &iv_trunc_t))) {
 445     return false; // Funny increment opcode
 446   }
 447   assert(incr->Opcode() == Op_AddI, "wrong increment code");
 448 
 449   const TypeInt* limit_t = gvn->type(limit)->is_int();
 450   if (trunc1 != NULL) {
 451     // When there is a truncation, we must be sure that after the truncation
 452     // the trip counter will end up higher than the limit, otherwise we are looking
 453     // at an endless loop. Can happen with range checks.
 454 
 455     // Example:
 456     // int i = 0;
 457     // while (true)
 458     //    sum + = array[i];
 459     //    i++;
 460     //    i = i && 0x7fff;
 461     //  }
 462     //
 463     // If the array is shorter than 0x8000 this exits through a AIOOB
 464     //  - Counted loop transformation is ok
 465     // If the array is longer then this is an endless loop
 466     //  - No transformation can be done.
 467 
 468     const TypeInt* incr_t = gvn->type(orig_incr)->is_int();
 469     if (limit_t->_hi > incr_t->_hi) {
 470       // if the limit can have a higher value than the increment (before the phi)
 471       return false;
 472     }
 473   }
 474 
 475   // Get merge point
 476   Node *xphi = incr->in(1);
 477   Node *stride = incr->in(2);
 478   if (!stride->is_Con()) {     // Oops, swap these
 479     if (!xphi->is_Con())       // Is the other guy a constant?
 480       return false;             // Nope, unknown stride, bail out
 481     Node *tmp = xphi;           // 'incr' is commutative, so ok to swap
 482     xphi = stride;
 483     stride = tmp;
 484   }
 485   if (xphi->Opcode() == Op_CastII) {
 486     xphi = xphi->in(1);
 487   }
 488   // Stride must be constant
 489   int stride_con = stride->get_int();
 490   if (stride_con == 0)
 491     return false; // missed some peephole opt
 492 
 493   if (!xphi->is_Phi())
 494     return false; // Too much math on the trip counter
 495   if (phi_incr != NULL && phi_incr != xphi)
 496     return false;
 497   PhiNode *phi = xphi->as_Phi();
 498 
 499   // Phi must be of loop header; backedge must wrap to increment
 500   if (phi->region() != x)
 501     return false;
 502   if ((trunc1 == NULL && phi->in(LoopNode::LoopBackControl) != incr) ||
 503       (trunc1 != NULL && phi->in(LoopNode::LoopBackControl) != trunc1)) {
 504     return false;
 505   }
 506   Node *init_trip = phi->in(LoopNode::EntryControl);
 507 
 508   // If iv trunc type is smaller than int, check for possible wrap.
 509   if (!TypeInt::INT->higher_equal(iv_trunc_t)) {
 510     assert(trunc1 != NULL, "must have found some truncation");
 511 
 512     // Get a better type for the phi (filtered thru if's)
 513     const TypeInt* phi_ft = filtered_type(phi);
 514 
 515     // Can iv take on a value that will wrap?
 516     //
 517     // Ensure iv's limit is not within "stride" of the wrap value.
 518     //
 519     // Example for "short" type
 520     //    Truncation ensures value is in the range -32768..32767 (iv_trunc_t)
 521     //    If the stride is +10, then the last value of the induction
 522     //    variable before the increment (phi_ft->_hi) must be
 523     //    <= 32767 - 10 and (phi_ft->_lo) must be >= -32768 to
 524     //    ensure no truncation occurs after the increment.
 525 
 526     if (stride_con > 0) {
 527       if (iv_trunc_t->_hi - phi_ft->_hi < stride_con ||
 528           iv_trunc_t->_lo > phi_ft->_lo) {
 529         return false;  // truncation may occur
 530       }
 531     } else if (stride_con < 0) {
 532       if (iv_trunc_t->_lo - phi_ft->_lo > stride_con ||
 533           iv_trunc_t->_hi < phi_ft->_hi) {
 534         return false;  // truncation may occur
 535       }
 536     }
 537     // No possibility of wrap so truncation can be discarded
 538     // Promote iv type to Int
 539   } else {
 540     assert(trunc1 == NULL && trunc2 == NULL, "no truncation for int");
 541   }
 542 
 543   // If the condition is inverted and we will be rolling
 544   // through MININT to MAXINT, then bail out.
 545   if (bt == BoolTest::eq || // Bail out, but this loop trips at most twice!
 546       // Odd stride
 547       (bt == BoolTest::ne && stride_con != 1 && stride_con != -1) ||
 548       // Count down loop rolls through MAXINT
 549       ((bt == BoolTest::le || bt == BoolTest::lt) && stride_con < 0) ||
 550       // Count up loop rolls through MININT
 551       ((bt == BoolTest::ge || bt == BoolTest::gt) && stride_con > 0)) {
 552     return false; // Bail out
 553   }
 554 
 555   const TypeInt* init_t = gvn->type(init_trip)->is_int();
 556 
 557   if (stride_con > 0) {
 558     jlong init_p = (jlong)init_t->_lo + stride_con;
 559     if (init_p > (jlong)max_jint || init_p > (jlong)limit_t->_hi)
 560       return false; // cyclic loop or this loop trips only once
 561   } else {
 562     jlong init_p = (jlong)init_t->_hi + stride_con;
 563     if (init_p < (jlong)min_jint || init_p < (jlong)limit_t->_lo)
 564       return false; // cyclic loop or this loop trips only once
 565   }
 566 
 567   if (phi_incr != NULL && bt != BoolTest::ne) {
 568     // check if there is a possiblity of IV overflowing after the first increment
 569     if (stride_con > 0) {
 570       if (init_t->_hi > max_jint - stride_con) {
 571         return false;
 572       }
 573     } else {
 574       if (init_t->_lo < min_jint - stride_con) {
 575         return false;
 576       }
 577     }
 578   }
 579 
 580   // =================================================
 581   // ---- SUCCESS!   Found A Trip-Counted Loop!  -----
 582   //
 583   assert(x->Opcode() == Op_Loop, "regular loops only");
 584   C->print_method(PHASE_BEFORE_CLOOPS, 3);
 585 
 586   Node *hook = new Node(6);
 587 
 588   // ===================================================
 589   // Generate loop limit check to avoid integer overflow
 590   // in cases like next (cyclic loops):
 591   //
 592   // for (i=0; i <= max_jint; i++) {}
 593   // for (i=0; i <  max_jint; i+=2) {}
 594   //
 595   //
 596   // Limit check predicate depends on the loop test:
 597   //
 598   // for(;i != limit; i++)       --> limit <= (max_jint)
 599   // for(;i <  limit; i+=stride) --> limit <= (max_jint - stride + 1)
 600   // for(;i <= limit; i+=stride) --> limit <= (max_jint - stride    )
 601   //
 602 
 603   // Check if limit is excluded to do more precise int overflow check.
 604   bool incl_limit = (bt == BoolTest::le || bt == BoolTest::ge);
 605   int stride_m  = stride_con - (incl_limit ? 0 : (stride_con > 0 ? 1 : -1));
 606 
 607   // If compare points directly to the phi we need to adjust
 608   // the compare so that it points to the incr. Limit have
 609   // to be adjusted to keep trip count the same and the
 610   // adjusted limit should be checked for int overflow.
 611   if (phi_incr != NULL) {
 612     stride_m  += stride_con;
 613   }
 614 
 615   if (limit->is_Con()) {
 616     int limit_con = limit->get_int();
 617     if ((stride_con > 0 && limit_con > (max_jint - stride_m)) ||
 618         (stride_con < 0 && limit_con < (min_jint - stride_m))) {
 619       // Bailout: it could be integer overflow.
 620       return false;
 621     }
 622   } else if ((stride_con > 0 && limit_t->_hi <= (max_jint - stride_m)) ||
 623              (stride_con < 0 && limit_t->_lo >= (min_jint - stride_m))) {
 624       // Limit's type may satisfy the condition, for example,
 625       // when it is an array length.
 626   } else {
 627     // Generate loop's limit check.
 628     // Loop limit check predicate should be near the loop.
 629     ProjNode *limit_check_proj = find_predicate_insertion_point(init_control, Deoptimization::Reason_loop_limit_check);
 630     if (!limit_check_proj) {
 631       // The limit check predicate is not generated if this method trapped here before.
 632 #ifdef ASSERT
 633       if (TraceLoopLimitCheck) {
 634         tty->print("missing loop limit check:");
 635         loop->dump_head();
 636         x->dump(1);
 637       }
 638 #endif
 639       return false;
 640     }
 641 
 642     IfNode* check_iff = limit_check_proj->in(0)->as_If();
 643 
 644     if (!is_dominator(get_ctrl(limit), check_iff->in(0))) {
 645       return false;
 646     }
 647 
 648     Node* cmp_limit;
 649     Node* bol;
 650 
 651     if (stride_con > 0) {
 652       cmp_limit = new CmpINode(limit, _igvn.intcon(max_jint - stride_m));
 653       bol = new BoolNode(cmp_limit, BoolTest::le);
 654     } else {
 655       cmp_limit = new CmpINode(limit, _igvn.intcon(min_jint - stride_m));
 656       bol = new BoolNode(cmp_limit, BoolTest::ge);
 657     }
 658 
 659     insert_loop_limit_check(limit_check_proj, cmp_limit, bol);
 660   }
 661 
 662   // Now we need to canonicalize loop condition.
 663   if (bt == BoolTest::ne) {
 664     assert(stride_con == 1 || stride_con == -1, "simple increment only");
 665     if (stride_con > 0 && init_t->_hi < limit_t->_lo) {
 666       // 'ne' can be replaced with 'lt' only when init < limit.
 667       bt = BoolTest::lt;
 668     } else if (stride_con < 0 && init_t->_lo > limit_t->_hi) {
 669       // 'ne' can be replaced with 'gt' only when init > limit.
 670       bt = BoolTest::gt;
 671     } else {
 672       ProjNode *limit_check_proj = find_predicate_insertion_point(init_control, Deoptimization::Reason_loop_limit_check);
 673       if (!limit_check_proj) {
 674         // The limit check predicate is not generated if this method trapped here before.
 675 #ifdef ASSERT
 676         if (TraceLoopLimitCheck) {
 677           tty->print("missing loop limit check:");
 678           loop->dump_head();
 679           x->dump(1);
 680         }
 681 #endif
 682         return false;
 683       }
 684       IfNode* check_iff = limit_check_proj->in(0)->as_If();
 685 
 686       if (!is_dominator(get_ctrl(limit), check_iff->in(0)) ||
 687           !is_dominator(get_ctrl(init_trip), check_iff->in(0))) {
 688         return false;
 689       }
 690 
 691       Node* cmp_limit;
 692       Node* bol;
 693 
 694       if (stride_con > 0) {
 695         cmp_limit = new CmpINode(init_trip, limit);
 696         bol = new BoolNode(cmp_limit, BoolTest::lt);
 697       } else {
 698         cmp_limit = new CmpINode(init_trip, limit);
 699         bol = new BoolNode(cmp_limit, BoolTest::gt);
 700       }
 701 
 702       insert_loop_limit_check(limit_check_proj, cmp_limit, bol);
 703 
 704       if (stride_con > 0) {
 705         // 'ne' can be replaced with 'lt' only when init < limit.
 706         bt = BoolTest::lt;
 707       } else if (stride_con < 0) {
 708         // 'ne' can be replaced with 'gt' only when init > limit.
 709         bt = BoolTest::gt;
 710       }
 711     }
 712   }
 713 
 714   if (phi_incr != NULL) {
 715     // If compare points directly to the phi we need to adjust
 716     // the compare so that it points to the incr. Limit have
 717     // to be adjusted to keep trip count the same and we
 718     // should avoid int overflow.
 719     //
 720     //   i = init; do {} while(i++ < limit);
 721     // is converted to
 722     //   i = init; do {} while(++i < limit+1);
 723     //
 724     limit = gvn->transform(new AddINode(limit, stride));
 725   }
 726 
 727   if (incl_limit) {
 728     // The limit check guaranties that 'limit <= (max_jint - stride)' so
 729     // we can convert 'i <= limit' to 'i < limit+1' since stride != 0.
 730     //
 731     Node* one = (stride_con > 0) ? gvn->intcon( 1) : gvn->intcon(-1);
 732     limit = gvn->transform(new AddINode(limit, one));
 733     if (bt == BoolTest::le)
 734       bt = BoolTest::lt;
 735     else if (bt == BoolTest::ge)
 736       bt = BoolTest::gt;
 737     else
 738       ShouldNotReachHere();
 739   }
 740   set_subtree_ctrl( limit );
 741 
 742   if (LoopStripMiningIter == 0) {
 743     // Check for SafePoint on backedge and remove
 744     Node *sfpt = x->in(LoopNode::LoopBackControl);
 745     if (sfpt->Opcode() == Op_SafePoint && is_deleteable_safept(sfpt)) {
 746       lazy_replace( sfpt, iftrue );
 747       if (loop->_safepts != NULL) {
 748         loop->_safepts->yank(sfpt);
 749       }
 750       loop->_tail = iftrue;
 751     }
 752   }
 753 
 754   // Build a canonical trip test.
 755   // Clone code, as old values may be in use.
 756   incr = incr->clone();
 757   incr->set_req(1,phi);
 758   incr->set_req(2,stride);
 759   incr = _igvn.register_new_node_with_optimizer(incr);
 760   set_early_ctrl( incr );
 761   _igvn.rehash_node_delayed(phi);
 762   phi->set_req_X( LoopNode::LoopBackControl, incr, &_igvn );
 763 
 764   // If phi type is more restrictive than Int, raise to
 765   // Int to prevent (almost) infinite recursion in igvn
 766   // which can only handle integer types for constants or minint..maxint.
 767   if (!TypeInt::INT->higher_equal(phi->bottom_type())) {
 768     Node* nphi = PhiNode::make(phi->in(0), phi->in(LoopNode::EntryControl), TypeInt::INT);
 769     nphi->set_req(LoopNode::LoopBackControl, phi->in(LoopNode::LoopBackControl));
 770     nphi = _igvn.register_new_node_with_optimizer(nphi);
 771     set_ctrl(nphi, get_ctrl(phi));
 772     _igvn.replace_node(phi, nphi);
 773     phi = nphi->as_Phi();
 774   }
 775   cmp = cmp->clone();
 776   cmp->set_req(1,incr);
 777   cmp->set_req(2,limit);
 778   cmp = _igvn.register_new_node_with_optimizer(cmp);
 779   set_ctrl(cmp, iff->in(0));
 780 
 781   test = test->clone()->as_Bool();
 782   (*(BoolTest*)&test->_test)._test = bt;
 783   test->set_req(1,cmp);
 784   _igvn.register_new_node_with_optimizer(test);
 785   set_ctrl(test, iff->in(0));
 786 
 787   // Replace the old IfNode with a new LoopEndNode
 788   Node *lex = _igvn.register_new_node_with_optimizer(new CountedLoopEndNode( iff->in(0), test, cl_prob, iff->as_If()->_fcnt ));
 789   IfNode *le = lex->as_If();
 790   uint dd = dom_depth(iff);
 791   set_idom(le, le->in(0), dd); // Update dominance for loop exit
 792   set_loop(le, loop);
 793 
 794   // Get the loop-exit control
 795   Node *iffalse = iff->as_If()->proj_out(!(iftrue_op == Op_IfTrue));
 796 
 797   // Need to swap loop-exit and loop-back control?
 798   if (iftrue_op == Op_IfFalse) {
 799     Node *ift2=_igvn.register_new_node_with_optimizer(new IfTrueNode (le));
 800     Node *iff2=_igvn.register_new_node_with_optimizer(new IfFalseNode(le));
 801 
 802     loop->_tail = back_control = ift2;
 803     set_loop(ift2, loop);
 804     set_loop(iff2, get_loop(iffalse));
 805 
 806     // Lazy update of 'get_ctrl' mechanism.
 807     lazy_replace(iffalse, iff2);
 808     lazy_replace(iftrue,  ift2);
 809 
 810     // Swap names
 811     iffalse = iff2;
 812     iftrue  = ift2;
 813   } else {
 814     _igvn.rehash_node_delayed(iffalse);
 815     _igvn.rehash_node_delayed(iftrue);
 816     iffalse->set_req_X( 0, le, &_igvn );
 817     iftrue ->set_req_X( 0, le, &_igvn );
 818   }
 819 
 820   set_idom(iftrue,  le, dd+1);
 821   set_idom(iffalse, le, dd+1);
 822   assert(iff->outcnt() == 0, "should be dead now");
 823   lazy_replace( iff, le ); // fix 'get_ctrl'
 824 
 825   Node *sfpt2 = le->in(0);
 826 
 827   Node* entry_control = init_control;
 828   bool strip_mine_loop = LoopStripMiningIter > 1 && loop->_child == NULL &&
 829     sfpt2->Opcode() == Op_SafePoint && !loop->_has_call;
 830   IdealLoopTree* outer_ilt = NULL;
 831   if (strip_mine_loop) {
 832     outer_ilt = create_outer_strip_mined_loop(test, cmp, init_control, loop,
 833                                               cl_prob, le->_fcnt, entry_control,
 834                                               iffalse);
 835   }
 836 
 837   // Now setup a new CountedLoopNode to replace the existing LoopNode
 838   CountedLoopNode *l = new CountedLoopNode(entry_control, back_control);
 839   l->set_unswitch_count(x->as_Loop()->unswitch_count()); // Preserve
 840   // The following assert is approximately true, and defines the intention
 841   // of can_be_counted_loop.  It fails, however, because phase->type
 842   // is not yet initialized for this loop and its parts.
 843   //assert(l->can_be_counted_loop(this), "sanity");
 844   _igvn.register_new_node_with_optimizer(l);
 845   set_loop(l, loop);
 846   loop->_head = l;
 847   // Fix all data nodes placed at the old loop head.
 848   // Uses the lazy-update mechanism of 'get_ctrl'.
 849   lazy_replace( x, l );
 850   set_idom(l, entry_control, dom_depth(entry_control) + 1);
 851 
 852   if (LoopStripMiningIter == 0 || strip_mine_loop) {
 853     // Check for immediately preceding SafePoint and remove
 854     if (sfpt2->Opcode() == Op_SafePoint && (LoopStripMiningIter != 0 || is_deleteable_safept(sfpt2))) {
 855       if (strip_mine_loop) {
 856         Node* outer_le = outer_ilt->_tail->in(0);
 857         Node* sfpt = sfpt2->clone();
 858         sfpt->set_req(0, iffalse);
 859         outer_le->set_req(0, sfpt);
 860         register_control(sfpt, outer_ilt, iffalse);
 861         set_idom(outer_le, sfpt, dom_depth(sfpt));
 862       }
 863       lazy_replace( sfpt2, sfpt2->in(TypeFunc::Control));
 864       if (loop->_safepts != NULL) {
 865         loop->_safepts->yank(sfpt2);
 866       }
 867     }
 868   }
 869 
 870   // Free up intermediate goo
 871   _igvn.remove_dead_node(hook);
 872 
 873 #ifdef ASSERT
 874   assert(l->is_valid_counted_loop(), "counted loop shape is messed up");
 875   assert(l == loop->_head && l->phi() == phi && l->loopexit_or_null() == lex, "" );
 876 #endif
 877 #ifndef PRODUCT
 878   if (TraceLoopOpts) {
 879     tty->print("Counted      ");
 880     loop->dump_head();
 881   }
 882 #endif
 883 
 884   C->print_method(PHASE_AFTER_CLOOPS, 3);
 885 
 886   // Capture bounds of the loop in the induction variable Phi before
 887   // subsequent transformation (iteration splitting) obscures the
 888   // bounds
 889   l->phi()->as_Phi()->set_type(l->phi()->Value(&_igvn));
 890 
 891   if (strip_mine_loop) {
 892     l->mark_strip_mined();
 893     l->verify_strip_mined(1);
 894     outer_ilt->_head->as_Loop()->verify_strip_mined(1);
 895     loop = outer_ilt;
 896   }
 897 
 898   return true;
 899 }

 901 //----------------------exact_limit-------------------------------------------
 902 Node* PhaseIdealLoop::exact_limit( IdealLoopTree *loop ) {
 903   assert(loop->_head->is_CountedLoop(), "");
 904   CountedLoopNode *cl = loop->_head->as_CountedLoop();
 905   assert(cl->is_valid_counted_loop(), "");
 906 
 907   if (ABS(cl->stride_con()) == 1 ||
 908       cl->limit()->Opcode() == Op_LoopLimit) {
 909     // Old code has exact limit (it could be incorrect in case of int overflow).
 910     // Loop limit is exact with stride == 1. And loop may already have exact limit.
 911     return cl->limit();
 912   }
 913   Node *limit = NULL;
 914 #ifdef ASSERT
 915   BoolTest::mask bt = cl->loopexit()->test_trip();
 916   assert(bt == BoolTest::lt || bt == BoolTest::gt, "canonical test is expected");
 917 #endif
 918   if (cl->has_exact_trip_count()) {
 919     // Simple case: loop has constant boundaries.
 920     // Use jlongs to avoid integer overflow.
 921     int stride_con = cl->stride_con();
 922     jlong  init_con = cl->init_trip()->get_int();
 923     jlong limit_con = cl->limit()->get_int();
 924     julong trip_cnt = cl->trip_count();
 925     jlong final_con = init_con + trip_cnt*stride_con;
 926     int final_int = (int)final_con;
 927     // The final value should be in integer range since the loop
 928     // is counted and the limit was checked for overflow.
 929     assert(final_con == (jlong)final_int, "final value should be integer");
 930     limit = _igvn.intcon(final_int);
 931   } else {
 932     // Create new LoopLimit node to get exact limit (final iv value).
 933     limit = new LoopLimitNode(C, cl->init_trip(), cl->limit(), cl->stride());
 934     register_new_node(limit, cl->in(LoopNode::EntryControl));
 935   }
 936   assert(limit != NULL, "sanity");
 937   return limit;
 938 }

1020         assert(u == outer || u == inner || u->is_Phi(), "nothing between inner and outer loop");

1023       for (DUIterator_Fast imax, i = inner_out->fast_outs(imax); i < imax; i++) {
1024         Node* u = inner_out->fast_out(i);
1025         if (u->is_Store()) {
1026           stores++;
1027         }
1028       }
1029       assert(outer->outcnt() >= phis + 2 && outer->outcnt() <= phis + 2 + stores + 1, "only phis");
1030     }
1031     assert(sfpt->outcnt() == 1, "no data node");
1032     assert(outer_tail->outcnt() == 1 || !has_skeleton, "no data node");
1033   }

1065 //=============================================================================
1066 //------------------------------Value-----------------------------------------
1067 const Type* LoopLimitNode::Value(PhaseGVN* phase) const {
1068   const Type* init_t   = phase->type(in(Init));
1069   const Type* limit_t  = phase->type(in(Limit));
1070   const Type* stride_t = phase->type(in(Stride));
1071   // Either input is TOP ==> the result is TOP
1072   if (init_t   == Type::TOP) return Type::TOP;
1073   if (limit_t  == Type::TOP) return Type::TOP;
1074   if (stride_t == Type::TOP) return Type::TOP;
1075 
1076   int stride_con = stride_t->is_int()->get_con();
1077   if (stride_con == 1)
1078     return NULL;  // Identity
1079 
1080   if (init_t->is_int()->is_con() && limit_t->is_int()->is_con()) {
1081     // Use jlongs to avoid integer overflow.
1082     jlong init_con   =  init_t->is_int()->get_con();
1083     jlong limit_con  = limit_t->is_int()->get_con();
1084     int  stride_m   = stride_con - (stride_con > 0 ? 1 : -1);
1085     jlong trip_count = (limit_con - init_con + stride_m)/stride_con;
1086     jlong final_con  = init_con + stride_con*trip_count;
1087     int final_int = (int)final_con;
1088     // The final value should be in integer range since the loop
1089     // is counted and the limit was checked for overflow.
1090     assert(final_con == (jlong)final_int, "final value should be integer");
1091     return TypeInt::make(final_int);
1092   }
1093 
1094   return bottom_type(); // TypeInt::INT
1095 }

1097 //------------------------------Ideal------------------------------------------
1098 // Return a node which is more "ideal" than the current node.
1099 Node *LoopLimitNode::Ideal(PhaseGVN *phase, bool can_reshape) {
1100   if (phase->type(in(Init))   == Type::TOP ||
1101       phase->type(in(Limit))  == Type::TOP ||
1102       phase->type(in(Stride)) == Type::TOP)
1103     return NULL;  // Dead
1104 
1105   int stride_con = phase->type(in(Stride))->is_int()->get_con();
1106   if (stride_con == 1)
1107     return NULL;  // Identity
1108 
1109   if (in(Init)->is_Con() && in(Limit)->is_Con())
1110     return NULL;  // Value
1111 
1112   // Delay following optimizations until all loop optimizations
1113   // done to keep Ideal graph simple.
1114   if (!can_reshape || phase->C->major_progress())
1115     return NULL;
1116 
1117   const TypeInt* init_t  = phase->type(in(Init) )->is_int();
1118   const TypeInt* limit_t = phase->type(in(Limit))->is_int();
1119   int stride_p;
1120   jlong lim, ini;
1121   julong max;
1122   if (stride_con > 0) {
1123     stride_p = stride_con;
1124     lim = limit_t->_hi;
1125     ini = init_t->_lo;
1126     max = (julong)max_jint;
1127   } else {
1128     stride_p = -stride_con;
1129     lim = init_t->_hi;
1130     ini = limit_t->_lo;
1131     max = (julong)min_jint;
1132   }
1133   julong range = lim - ini + stride_p;
1134   if (range <= max) {
1135     // Convert to integer expression if it is not overflow.
1136     Node* stride_m = phase->intcon(stride_con - (stride_con > 0 ? 1 : -1));
1137     Node *range = phase->transform(new SubINode(in(Limit), in(Init)));
1138     Node *bias  = phase->transform(new AddINode(range, stride_m));
1139     Node *trip  = phase->transform(new DivINode(0, bias, in(Stride)));
1140     Node *span  = phase->transform(new MulINode(trip, in(Stride)));
1141     return new AddINode(span, in(Init)); // exact limit
1142   }
1143 
1144   if (is_power_of_2(stride_p) ||                // divisor is 2^n
1145       !Matcher::has_match_rule(Op_LoopLimit)) { // or no specialized Mach node?
1146     // Convert to long expression to avoid integer overflow
1147     // and let igvn optimizer convert this division.
1148     //
1149     Node*   init   = phase->transform( new ConvI2LNode(in(Init)));
1150     Node*  limit   = phase->transform( new ConvI2LNode(in(Limit)));
1151     Node* stride   = phase->longcon(stride_con);
1152     Node* stride_m = phase->longcon(stride_con - (stride_con > 0 ? 1 : -1));
1153 
1154     Node *range = phase->transform(new SubLNode(limit, init));
1155     Node *bias  = phase->transform(new AddLNode(range, stride_m));
1156     Node *span;
1157     if (stride_con > 0 && is_power_of_2(stride_p)) {
1158       // bias >= 0 if stride >0, so if stride is 2^n we can use &(-stride)
1159       // and avoid generating rounding for division. Zero trip guard should
1160       // guarantee that init < limit but sometimes the guard is missing and
1161       // we can get situation when init > limit. Note, for the empty loop
1162       // optimization zero trip guard is generated explicitly which leaves
1163       // only RCE predicate where exact limit is used and the predicate
1164       // will simply fail forcing recompilation.
1165       Node* neg_stride   = phase->longcon(-stride_con);
1166       span = phase->transform(new AndLNode(bias, neg_stride));
1167     } else {
1168       Node *trip  = phase->transform(new DivLNode(0, bias, stride));
1169       span = phase->transform(new MulLNode(trip, stride));
1170     }
1171     // Convert back to int
1172     Node *span_int = phase->transform(new ConvL2INode(span));
1173     return new AddINode(span_int, in(Init)); // exact limit
1174   }
1175 
1176   return NULL;    // No progress
1177 }

1188 //=============================================================================
1189 //----------------------match_incr_with_optional_truncation--------------------
1190 // Match increment with optional truncation:
1191 // CHAR: (i+1)&0x7fff, BYTE: ((i+1)<<8)>>8, or SHORT: ((i+1)<<16)>>16
1192 // Return NULL for failure. Success returns the increment node.
1193 Node* CountedLoopNode::match_incr_with_optional_truncation(
1194                       Node* expr, Node** trunc1, Node** trunc2, const TypeInt** trunc_type) {
1195   // Quick cutouts:
1196   if (expr == NULL || expr->req() != 3)  return NULL;
1197 
1198   Node *t1 = NULL;
1199   Node *t2 = NULL;
1200   const TypeInt* trunc_t = TypeInt::INT;
1201   Node* n1 = expr;
1202   int   n1op = n1->Opcode();
1203 
1204   // Try to strip (n1 & M) or (n1 << N >> N) from n1.
1205   if (n1op == Op_AndI &&
1206       n1->in(2)->is_Con() &&
1207       n1->in(2)->bottom_type()->is_int()->get_con() == 0x7fff) {
1208     // %%% This check should match any mask of 2**K-1.
1209     t1 = n1;
1210     n1 = t1->in(1);
1211     n1op = n1->Opcode();
1212     trunc_t = TypeInt::CHAR;
1213   } else if (n1op == Op_RShiftI &&
1214              n1->in(1) != NULL &&
1215              n1->in(1)->Opcode() == Op_LShiftI &&
1216              n1->in(2) == n1->in(1)->in(2) &&
1217              n1->in(2)->is_Con()) {
1218     jint shift = n1->in(2)->bottom_type()->is_int()->get_con();
1219     // %%% This check should match any shift in [1..31].
1220     if (shift == 16 || shift == 8) {
1221       t1 = n1;
1222       t2 = t1->in(1);
1223       n1 = t2->in(1);
1224       n1op = n1->Opcode();
1225       if (shift == 16) {
1226         trunc_t = TypeInt::SHORT;
1227       } else if (shift == 8) {
1228         trunc_t = TypeInt::BYTE;
1229       }
1230     }
1231   }
1232 
1233   // If (maybe after stripping) it is an AddI, we won:
1234   if (n1op == Op_AddI) {
1235     *trunc1 = t1;
1236     *trunc2 = t2;
1237     *trunc_type = trunc_t;
1238     return n1;
1239   }
1240 
1241   // failed
1242   return NULL;
1243 }

1358 void OuterStripMinedLoopNode::adjust_strip_mined_loop(PhaseIterGVN* igvn) {
1359   // Look for the outer & inner strip mined loop, reduce number of
1360   // iterations of the inner loop, set exit condition of outer loop,
1361   // construct required phi nodes for outer loop.
1362   CountedLoopNode* inner_cl = unique_ctrl_out()->as_CountedLoop();
1363   assert(inner_cl->is_strip_mined(), "inner loop should be strip mined");
1364   Node* inner_iv_phi = inner_cl->phi();
1365   if (inner_iv_phi == NULL) {
1366     IfNode* outer_le = outer_loop_end();
1367     Node* iff = igvn->transform(new IfNode(outer_le->in(0), outer_le->in(1), outer_le->_prob, outer_le->_fcnt));
1368     igvn->replace_node(outer_le, iff);
1369     inner_cl->clear_strip_mined();
1370     return;
1371   }
1372   CountedLoopEndNode* inner_cle = inner_cl->loopexit();
1373 
1374   int stride = inner_cl->stride_con();
1375   jlong scaled_iters_long = ((jlong)LoopStripMiningIter) * ABS(stride);
1376   int scaled_iters = (int)scaled_iters_long;
1377   int short_scaled_iters = LoopStripMiningIterShortLoop* ABS(stride);
1378   const TypeInt* inner_iv_t = igvn->type(inner_iv_phi)->is_int();
1379   jlong iter_estimate = (jlong)inner_iv_t->_hi - (jlong)inner_iv_t->_lo;
1380   assert(iter_estimate > 0, "broken");
1381   if ((jlong)scaled_iters != scaled_iters_long || iter_estimate <= short_scaled_iters) {
1382     // Remove outer loop and safepoint (too few iterations)
1383     Node* outer_sfpt = outer_safepoint();
1384     Node* outer_out = outer_loop_exit();
1385     igvn->replace_node(outer_out, outer_sfpt->in(0));
1386     igvn->replace_input_of(outer_sfpt, 0, igvn->C->top());
1387     inner_cl->clear_strip_mined();
1388     return;
1389   }
1390   if (iter_estimate <= scaled_iters_long) {
1391     // We would only go through one iteration of
1392     // the outer loop: drop the outer loop but
1393     // keep the safepoint so we don't run for
1394     // too long without a safepoint
1395     IfNode* outer_le = outer_loop_end();
1396     Node* iff = igvn->transform(new IfNode(outer_le->in(0), outer_le->in(1), outer_le->_prob, outer_le->_fcnt));
1397     igvn->replace_node(outer_le, iff);
1398     inner_cl->clear_strip_mined();
1399     return;
1400   }
1401 
1402   Node* cle_tail = inner_cle->proj_out(true);
1403   ResourceMark rm;
1404   Node_List old_new;
1405   if (cle_tail->outcnt() > 1) {
1406     // Look for nodes on backedge of inner loop and clone them
1407     Unique_Node_List backedge_nodes;
1408     for (DUIterator_Fast imax, i = cle_tail->fast_outs(imax); i < imax; i++) {
1409       Node* u = cle_tail->fast_out(i);
1410       if (u != inner_cl) {
1411         assert(!u->is_CFG(), "control flow on the backedge?");
1412         backedge_nodes.push(u);
1413       }
1414     }
1415     uint last = igvn->C->unique();
1416     for (uint next = 0; next < backedge_nodes.size(); next++) {
1417       Node* n = backedge_nodes.at(next);
1418       old_new.map(n->_idx, n->clone());
1419       for (DUIterator_Fast imax, i = n->fast_outs(imax); i < imax; i++) {
1420         Node* u = n->fast_out(i);
1421         assert(!u->is_CFG(), "broken");
1422         if (u->_idx >= last) {
1423           continue;
1424         }
1425         if (!u->is_Phi()) {
1426           backedge_nodes.push(u);
1427         } else {
1428           assert(u->in(0) == inner_cl, "strange phi on the backedge");
1429         }
1430       }
1431     }
1432     // Put the clones on the outer loop backedge
1433     Node* le_tail = outer_loop_tail();
1434     for (uint next = 0; next < backedge_nodes.size(); next++) {
1435       Node *n = old_new[backedge_nodes.at(next)->_idx];
1436       for (uint i = 1; i < n->req(); i++) {
1437         if (n->in(i) != NULL && old_new[n->in(i)->_idx] != NULL) {
1438           n->set_req(i, old_new[n->in(i)->_idx]);
1439         }
1440       }
1441       if (n->in(0) != NULL && n->in(0) == cle_tail) {
1442         n->set_req(0, le_tail);
1443       }
1444       igvn->register_new_node_with_optimizer(n);
1445     }
1446   }
1447 
1448   Node* iv_phi = NULL;
1449   // Make a clone of each phi in the inner loop
1450   // for the outer loop
1451   for (uint i = 0; i < inner_cl->outcnt(); i++) {
1452     Node* u = inner_cl->raw_out(i);
1453     if (u->is_Phi()) {
1454       assert(u->in(0) == inner_cl, "inconsistent");
1455       Node* phi = u->clone();
1456       phi->set_req(0, this);
1457       Node* be = old_new[phi->in(LoopNode::LoopBackControl)->_idx];
1458       if (be != NULL) {
1459         phi->set_req(LoopNode::LoopBackControl, be);
1460       }
1461       phi = igvn->transform(phi);
1462       igvn->replace_input_of(u, LoopNode::EntryControl, phi);
1463       if (u == inner_iv_phi) {
1464         iv_phi = phi;
1465       }
1466     }
1467   }
1468   Node* cle_out = inner_cle->proj_out(false);
1469   if (cle_out->outcnt() > 1) {
1470     // Look for chains of stores that were sunk
1471     // out of the inner loop and are in the outer loop
1472     for (DUIterator_Fast imax, i = cle_out->fast_outs(imax); i < imax; i++) {
1473       Node* u = cle_out->fast_out(i);
1474       if (u->is_Store()) {
1475         Node* first = u;
1476         for(;;) {
1477           Node* next = first->in(MemNode::Memory);
1478           if (!next->is_Store() || next->in(0) != cle_out) {
1479             break;
1480           }
1481           first = next;
1482         }
1483         Node* last = u;
1484         for(;;) {
1485           Node* next = NULL;
1486           for (DUIterator_Fast jmax, j = last->fast_outs(jmax); j < jmax; j++) {
1487             Node* uu = last->fast_out(j);
1488             if (uu->is_Store() && uu->in(0) == cle_out) {
1489               assert(next == NULL, "only one in the outer loop");
1490               next = uu;
1491             }
1492           }
1493           if (next == NULL) {
1494             break;
1495           }
1496           last = next;
1497         }
1498         Node* phi = NULL;
1499         for (DUIterator_Fast jmax, j = fast_outs(jmax); j < jmax; j++) {
1500           Node* uu = fast_out(j);
1501           if (uu->is_Phi()) {
1502             Node* be = uu->in(LoopNode::LoopBackControl);
1503             if (be->is_Store() && old_new[be->_idx] != NULL) {
1504               assert(false, "store on the backedge + sunk stores: unsupported");
1505               // drop outer loop
1506               IfNode* outer_le = outer_loop_end();
1507               Node* iff = igvn->transform(new IfNode(outer_le->in(0), outer_le->in(1), outer_le->_prob, outer_le->_fcnt));
1508               igvn->replace_node(outer_le, iff);
1509               inner_cl->clear_strip_mined();
1510               return;
1511             }
1512             if (be == last || be == first->in(MemNode::Memory)) {
1513               assert(phi == NULL, "only one phi");
1514               phi = uu;
1515             }
1516           }
1517         }
1518 #ifdef ASSERT
1519         for (DUIterator_Fast jmax, j = fast_outs(jmax); j < jmax; j++) {
1520           Node* uu = fast_out(j);
1521           if (uu->is_Phi() && uu->bottom_type() == Type::MEMORY) {
1522             if (uu->adr_type() == igvn->C->get_adr_type(igvn->C->get_alias_index(u->adr_type()))) {
1523               assert(phi == uu, "what's that phi?");
1524             } else if (uu->adr_type() == TypePtr::BOTTOM) {
1525               Node* n = uu->in(LoopNode::LoopBackControl);
1526               uint limit = igvn->C->live_nodes();
1527               uint i = 0;
1528               while (n != uu) {
1529                 i++;
1530                 assert(i < limit, "infinite loop");
1531                 if (n->is_Proj()) {
1532                   n = n->in(0);
1533                 } else if (n->is_SafePoint() || n->is_MemBar()) {
1534                   n = n->in(TypeFunc::Memory);
1535                 } else if (n->is_Phi()) {
1536                   n = n->in(1);
1537                 } else if (n->is_MergeMem()) {
1538                   n = n->as_MergeMem()->memory_at(igvn->C->get_alias_index(u->adr_type()));
1539                 } else if (n->is_Store() || n->is_LoadStore() || n->is_ClearArray()) {
1540                   n = n->in(MemNode::Memory);
1541                 } else {
1542                   n->dump();
1543                   ShouldNotReachHere();
1544                 }
1545               }
1546             }
1547           }
1548         }
1549 #endif
1550         if (phi == NULL) {
1551           // If the an entire chains was sunk, the
1552           // inner loop has no phi for that memory
1553           // slice, create one for the outer loop
1554           phi = PhiNode::make(this, first->in(MemNode::Memory), Type::MEMORY,
1555                               igvn->C->get_adr_type(igvn->C->get_alias_index(u->adr_type())));
1556           phi->set_req(LoopNode::LoopBackControl, last);
1557           phi = igvn->transform(phi);
1558           igvn->replace_input_of(first, MemNode::Memory, phi);
1559         } else {
1560           // Or fix the outer loop fix to include
1561           // that chain of stores.
1562           Node* be = phi->in(LoopNode::LoopBackControl);
1563           assert(!(be->is_Store() && old_new[be->_idx] != NULL), "store on the backedge + sunk stores: unsupported");
1564           if (be == first->in(MemNode::Memory)) {
1565             if (be == phi->in(LoopNode::LoopBackControl)) {
1566               igvn->replace_input_of(phi, LoopNode::LoopBackControl, last);
1567             } else {
1568               igvn->replace_input_of(be, MemNode::Memory, last);
1569             }
1570           } else {
1571 #ifdef ASSERT
1572             if (be == phi->in(LoopNode::LoopBackControl)) {
1573               assert(phi->in(LoopNode::LoopBackControl) == last, "");
1574             } else {
1575               assert(be->in(MemNode::Memory) == last, "");
1576             }
1577 #endif
1578           }
1579         }
1580       }
1581     }
1582   }
1583 
1584   if (iv_phi != NULL) {
1585     // Now adjust the inner loop's exit condition
1586     Node* limit = inner_cl->limit();
1587     Node* sub = NULL;
1588     if (stride > 0) {
1589       sub = igvn->transform(new SubINode(limit, iv_phi));
1590     } else {
1591       sub = igvn->transform(new SubINode(iv_phi, limit));
1592     }
1593     // sub is positive and can be larger than the max signed int
1594     // value. Use an unsigned min.
1595     Node* const_iters = igvn->intcon(scaled_iters);
1596     Node* cmp = igvn->transform(new CmpUNode(sub, const_iters));
1597     Node* bol = igvn->transform(new BoolNode(cmp, BoolTest::lt));
1598     Node* min = igvn->transform(new CMoveINode(bol, const_iters, sub, TypeInt::make(0, scaled_iters, Type::WidenMin)));
1599 
1600     Node* new_limit = NULL;
1601     if (stride > 0) {
1602       new_limit = igvn->transform(new AddINode(min, iv_phi));
1603     } else {
1604       new_limit = igvn->transform(new SubINode(iv_phi, min));
1605     }
1606     Node* inner_cmp = inner_cle->cmp_node();
1607     Node* inner_bol = inner_cle->in(CountedLoopEndNode::TestValue);
1608     Node* outer_bol = inner_bol;
1609     // cmp node for inner loop may be shared
1610     inner_cmp = inner_cmp->clone();
1611     inner_cmp->set_req(2, new_limit);
1612     inner_bol = inner_bol->clone();
1613     inner_bol->set_req(1, igvn->transform(inner_cmp));
1614     igvn->replace_input_of(inner_cle, CountedLoopEndNode::TestValue, igvn->transform(inner_bol));
1615     // Set the outer loop's exit condition too
1616     igvn->replace_input_of(outer_loop_end(), 1, outer_bol);
1617   } else {
1618     assert(false, "should be able to adjust outer loop");
1619     IfNode* outer_le = outer_loop_end();
1620     Node* iff = igvn->transform(new IfNode(outer_le->in(0), outer_le->in(1), outer_le->_prob, outer_le->_fcnt));
1621     igvn->replace_node(outer_le, iff);
1622     inner_cl->clear_strip_mined();
1623   }
1624 }

1691 // Return a possibly more restrictive type for val based on condition control flow of dominators
1692 const TypeInt* PhaseIdealLoop::filtered_type_from_dominators( Node* val, Node *use_ctrl) {

1702     Node* pred = use_ctrl;

1704     while (if_cnt < if_limit) {

1715       }

1718         break;

1721       if (dom_depth(pred) < val_dom_depth) {

1724     }
1725   }

1765 //------------------------------split_fall_in----------------------------------
1766 // Split out multiple fall-in edges from the loop header.  Move them to a
1767 // private RegionNode before the loop.  This becomes the loop landing pad.
1768 void IdealLoopTree::split_fall_in( PhaseIdealLoop *phase, int fall_in_cnt ) {
1769   PhaseIterGVN &igvn = phase->_igvn;
1770   uint i;
1771 
1772   // Make a new RegionNode to be the landing pad.
1773   Node *landing_pad = new RegionNode( fall_in_cnt+1 );
1774   phase->set_loop(landing_pad,_parent);
1775   // Gather all the fall-in control paths into the landing pad
1776   uint icnt = fall_in_cnt;
1777   uint oreq = _head->req();
1778   for( i = oreq-1; i>0; i-- )
1779     if( !phase->is_member( this, _head->in(i) ) )
1780       landing_pad->set_req(icnt--,_head->in(i));
1781 
1782   // Peel off PhiNode edges as well
1783   for (DUIterator_Fast jmax, j = _head->fast_outs(jmax); j < jmax; j++) {
1784     Node *oj = _head->fast_out(j);
1785     if( oj->is_Phi() ) {
1786       PhiNode* old_phi = oj->as_Phi();
1787       assert( old_phi->region() == _head, "" );
1788       igvn.hash_delete(old_phi);   // Yank from hash before hacking edges
1789       Node *p = PhiNode::make_blank(landing_pad, old_phi);
1790       uint icnt = fall_in_cnt;
1791       for( i = oreq-1; i>0; i-- ) {
1792         if( !phase->is_member( this, _head->in(i) ) ) {
1793           p->init_req(icnt--, old_phi->in(i));
1794           // Go ahead and clean out old edges from old phi
1795           old_phi->del_req(i);
1796         }
1797       }
1798       // Search for CSE's here, because ZKM.jar does a lot of
1799       // loop hackery and we need to be a little incremental
1800       // with the CSE to avoid O(N^2) node blow-up.
1801       Node *p2 = igvn.hash_find_insert(p); // Look for a CSE
1802       if( p2 ) {                // Found CSE
1803         p->destruct();          // Recover useless new node
1804         p = p2;                 // Use old node
1805       } else {
1806         igvn.register_new_node_with_optimizer(p, old_phi);
1807       }
1808       // Make old Phi refer to new Phi.
1809       old_phi->add_req(p);
1810       // Check for the special case of making the old phi useless and
1811       // disappear it.  In JavaGrande I have a case where this useless
1812       // Phi is the loop limit and prevents recognizing a CountedLoop
1813       // which in turn prevents removing an empty loop.
1814       Node *id_old_phi = old_phi->Identity(&igvn);
1815       if( id_old_phi != old_phi ) { // Found a simple identity?
1816         // Note that I cannot call 'replace_node' here, because
1817         // that will yank the edge from old_phi to the Region and
1818         // I'm mid-iteration over the Region's uses.
1819         for (DUIterator_Last imin, i = old_phi->last_outs(imin); i >= imin; ) {
1820           Node* use = old_phi->last_out(i);
1821           igvn.rehash_node_delayed(use);
1822           uint uses_found = 0;
1823           for (uint j = 0; j < use->len(); j++) {
1824             if (use->in(j) == old_phi) {
1825               if (j < use->req()) use->set_req (j, id_old_phi);
1826               else                use->set_prec(j, id_old_phi);
1827               uses_found++;
1828             }
1829           }
1830           i -= uses_found;    // we deleted 1 or more copies of this edge
1831         }
1832       }
1833       igvn._worklist.push(old_phi);
1834     }
1835   }
1836   // Finally clean out the fall-in edges from the RegionNode
1837   for( i = oreq-1; i>0; i-- ) {
1838     if( !phase->is_member( this, _head->in(i) ) ) {
1839       _head->del_req(i);
1840     }
1841   }
1842   igvn.rehash_node_delayed(_head);
1843   // Transform landing pad
1844   igvn.register_new_node_with_optimizer(landing_pad, _head);
1845   // Insert landing pad into the header
1846   _head->add_req(landing_pad);
1847 }

1849 //------------------------------split_outer_loop-------------------------------
1850 // Split out the outermost loop from this shared header.
1851 void IdealLoopTree::split_outer_loop( PhaseIdealLoop *phase ) {
1852   PhaseIterGVN &igvn = phase->_igvn;
1853 
1854   // Find index of outermost loop; it should also be my tail.
1855   uint outer_idx = 1;
1856   while( _head->in(outer_idx) != _tail ) outer_idx++;
1857 
1858   // Make a LoopNode for the outermost loop.
1859   Node *ctl = _head->in(LoopNode::EntryControl);
1860   Node *outer = new LoopNode( ctl, _head->in(outer_idx) );
1861   outer = igvn.register_new_node_with_optimizer(outer, _head);
1862   phase->set_created_loop_node();
1863 
1864   // Outermost loop falls into '_head' loop
1865   _head->set_req(LoopNode::EntryControl, outer);
1866   _head->del_req(outer_idx);
1867   // Split all the Phis up between '_head' loop and 'outer' loop.
1868   for (DUIterator_Fast jmax, j = _head->fast_outs(jmax); j < jmax; j++) {
1869     Node *out = _head->fast_out(j);
1870     if( out->is_Phi() ) {
1871       PhiNode *old_phi = out->as_Phi();
1872       assert( old_phi->region() == _head, "" );
1873       Node *phi = PhiNode::make_blank(outer, old_phi);
1874       phi->init_req(LoopNode::EntryControl,    old_phi->in(LoopNode::EntryControl));
1875       phi->init_req(LoopNode::LoopBackControl, old_phi->in(outer_idx));
1876       phi = igvn.register_new_node_with_optimizer(phi, old_phi);
1877       // Make old Phi point to new Phi on the fall-in path
1878       igvn.replace_input_of(old_phi, LoopNode::EntryControl, phi);
1879       old_phi->del_req(outer_idx);
1880     }
1881   }
1882 
1883   // Use the new loop head instead of the old shared one
1884   _head = outer;
1885   phase->set_loop(_head, this);
1886 }

1895 //------------------------------estimate_path_freq-----------------------------
1896 static float estimate_path_freq( Node *n ) {
1897   // Try to extract some path frequency info
1898   IfNode *iff;
1899   for( int i = 0; i < 50; i++ ) { // Skip through a bunch of uncommon tests
1900     uint nop = n->Opcode();
1901     if( nop == Op_SafePoint ) {   // Skip any safepoint
1902       n = n->in(0);
1903       continue;
1904     }
1905     if( nop == Op_CatchProj ) {   // Get count from a prior call
1906       // Assume call does not always throw exceptions: means the call-site
1907       // count is also the frequency of the fall-through path.
1908       assert( n->is_CatchProj(), "" );
1909       if( ((CatchProjNode*)n)->_con != CatchProjNode::fall_through_index )
1910         return 0.0f;            // Assume call exception path is rare
1911       Node *call = n->in(0)->in(0)->in(0);
1912       assert( call->is_Call(), "expect a call here" );
1913       const JVMState *jvms = ((CallNode*)call)->jvms();
1914       ciMethodData* methodData = jvms->method()->method_data();
1915       if (!methodData->is_mature())  return 0.0f; // No call-site data
1916       ciProfileData* data = methodData->bci_to_data(jvms->bci());
1917       if ((data == NULL) || !data->is_CounterData()) {
1918         // no call profile available, try call's control input
1919         n = n->in(0);
1920         continue;
1921       }
1922       return data->as_CounterData()->count()/FreqCountInvocations;
1923     }
1924     // See if there's a gating IF test
1925     Node *n_c = n->in(0);
1926     if( !n_c->is_If() ) break;       // No estimate available
1927     iff = n_c->as_If();
1928     if( iff->_fcnt != COUNT_UNKNOWN )   // Have a valid count?
1929       // Compute how much count comes on this path
1930       return ((nop == Op_IfTrue) ? iff->_prob : 1.0f - iff->_prob) * iff->_fcnt;
1931     // Have no count info.  Skip dull uncommon-trap like branches.
1932     if( (nop == Op_IfTrue  && iff->_prob < PROB_LIKELY_MAG(5)) ||
1933         (nop == Op_IfFalse && iff->_prob > PROB_UNLIKELY_MAG(5)) )
1934       break;
1935     // Skip through never-taken branch; look for a real loop exit.
1936     n = iff->in(0);
1937   }
1938   return 0.0f;                  // No estimate available
1939 }

1941 //------------------------------merge_many_backedges---------------------------
1942 // Merge all the backedges from the shared header into a private Region.
1943 // Feed that region as the one backedge to this loop.
1944 void IdealLoopTree::merge_many_backedges( PhaseIdealLoop *phase ) {
1945   uint i;
1946 
1947   // Scan for the top 2 hottest backedges
1948   float hotcnt = 0.0f;
1949   float warmcnt = 0.0f;
1950   uint hot_idx = 0;
1951   // Loop starts at 2 because slot 1 is the fall-in path
1952   for( i = 2; i < _head->req(); i++ ) {
1953     float cnt = estimate_path_freq(_head->in(i));
1954     if( cnt > hotcnt ) {       // Grab hottest path
1955       warmcnt = hotcnt;
1956       hotcnt = cnt;
1957       hot_idx = i;
1958     } else if( cnt > warmcnt ) { // And 2nd hottest path
1959       warmcnt = cnt;
1960     }
1961   }
1962 
1963   // See if the hottest backedge is worthy of being an inner loop
1964   // by being much hotter than the next hottest backedge.
1965   if( hotcnt <= 0.0001 ||
1966       hotcnt < 2.0*warmcnt ) hot_idx = 0;// No hot backedge
1967 
1968   // Peel out the backedges into a private merge point; peel
1969   // them all except optionally hot_idx.
1970   PhaseIterGVN &igvn = phase->_igvn;
1971 
1972   Node *hot_tail = NULL;
1973   // Make a Region for the merge point
1974   Node *r = new RegionNode(1);
1975   for( i = 2; i < _head->req(); i++ ) {
1976     if( i != hot_idx )
1977       r->add_req( _head->in(i) );
1978     else hot_tail = _head->in(i);
1979   }
1980   igvn.register_new_node_with_optimizer(r, _head);
1981   // Plug region into end of loop _head, followed by hot_tail
1982   while( _head->req() > 3 ) _head->del_req( _head->req()-1 );
1983   igvn.replace_input_of(_head, 2, r);
1984   if( hot_idx ) _head->add_req(hot_tail);
1985 
1986   // Split all the Phis up between '_head' loop and the Region 'r'
1987   for (DUIterator_Fast jmax, j = _head->fast_outs(jmax); j < jmax; j++) {
1988     Node *out = _head->fast_out(j);
1989     if( out->is_Phi() ) {
1990       PhiNode* n = out->as_Phi();
1991       igvn.hash_delete(n);      // Delete from hash before hacking edges
1992       Node *hot_phi = NULL;
1993       Node *phi = new PhiNode(r, n->type(), n->adr_type());
1994       // Check all inputs for the ones to peel out
1995       uint j = 1;
1996       for( uint i = 2; i < n->req(); i++ ) {
1997         if( i != hot_idx )
1998           phi->set_req( j++, n->in(i) );
1999         else hot_phi = n->in(i);
2000       }
2001       // Register the phi but do not transform until whole place transforms
2002       igvn.register_new_node_with_optimizer(phi, n);
2003       // Add the merge phi to the old Phi
2004       while( n->req() > 3 ) n->del_req( n->req()-1 );
2005       igvn.replace_input_of(n, 2, phi);
2006       if( hot_idx ) n->add_req(hot_phi);
2007     }
2008   }
2009 
2010 
2011   // Insert a new IdealLoopTree inserted below me.  Turn it into a clone
2012   // of self loop tree.  Turn self into a loop headed by _head and with
2013   // tail being the new merge point.
2014   IdealLoopTree *ilt = new IdealLoopTree( phase, _head, _tail );
2015   phase->set_loop(_tail,ilt);   // Adjust tail
2016   _tail = r;                    // Self's tail is new merge point
2017   phase->set_loop(r,this);
2018   ilt->_child = _child;         // New guy has my children
2019   _child = ilt;                 // Self has new guy as only child
2020   ilt->_parent = this;          // new guy has self for parent
2021   ilt->_nest = _nest;           // Same nesting depth (for now)
2022 
2023   // Starting with 'ilt', look for child loop trees using the same shared
2024   // header.  Flatten these out; they will no longer be loops in the end.
2025   IdealLoopTree **pilt = &_child;
2026   while( ilt ) {
2027     if( ilt->_head == _head ) {
2028       uint i;
2029       for( i = 2; i < _head->req(); i++ )
2030         if( _head->in(i) == ilt->_tail )
2031           break;                // Still a loop
2032       if( i == _head->req() ) { // No longer a loop
2033         // Flatten ilt.  Hang ilt's "_next" list from the end of
2034         // ilt's '_child' list.  Move the ilt's _child up to replace ilt.
2035         IdealLoopTree **cp = &ilt->_child;
2036         while( *cp ) cp = &(*cp)->_next;   // Find end of child list
2037         *cp = ilt->_next;       // Hang next list at end of child list
2038         *pilt = ilt->_child;    // Move child up to replace ilt
2039         ilt->_head = NULL;      // Flag as a loop UNIONED into parent
2040         ilt = ilt->_child;      // Repeat using new ilt
2041         continue;               // do not advance over ilt->_child
2042       }
2043       assert( ilt->_tail == hot_tail, "expected to only find the hot inner loop here" );
2044       phase->set_loop(_head,ilt);
2045     }
2046     pilt = &ilt->_child;        // Advance to next
2047     ilt = *pilt;
2048   }
2049 
2050   if( _child ) fix_parent( _child, this );
2051 }

2053 //------------------------------beautify_loops---------------------------------
2054 // Split shared headers and insert loop landing pads.
2055 // Insert a LoopNode to replace the RegionNode.
2056 // Return TRUE if loop tree is structurally changed.
2057 bool IdealLoopTree::beautify_loops( PhaseIdealLoop *phase ) {
2058   bool result = false;
2059   // Cache parts in locals for easy
2060   PhaseIterGVN &igvn = phase->_igvn;
2061 
2062   igvn.hash_delete(_head);      // Yank from hash before hacking edges
2063 
2064   // Check for multiple fall-in paths.  Peel off a landing pad if need be.
2065   int fall_in_cnt = 0;
2066   for( uint i = 1; i < _head->req(); i++ )
2067     if( !phase->is_member( this, _head->in(i) ) )
2068       fall_in_cnt++;
2069   assert( fall_in_cnt, "at least 1 fall-in path" );
2070   if( fall_in_cnt > 1 )         // Need a loop landing pad to merge fall-ins
2071     split_fall_in( phase, fall_in_cnt );
2072 
2073   // Swap inputs to the _head and all Phis to move the fall-in edge to
2074   // the left.
2075   fall_in_cnt = 1;
2076   while( phase->is_member( this, _head->in(fall_in_cnt) ) )
2077     fall_in_cnt++;
2078   if( fall_in_cnt > 1 ) {
2079     // Since I am just swapping inputs I do not need to update def-use info
2080     Node *tmp = _head->in(1);
2081     igvn.rehash_node_delayed(_head);
2082     _head->set_req( 1, _head->in(fall_in_cnt) );
2083     _head->set_req( fall_in_cnt, tmp );
2084     // Swap also all Phis
2085     for (DUIterator_Fast imax, i = _head->fast_outs(imax); i < imax; i++) {
2086       Node* phi = _head->fast_out(i);
2087       if( phi->is_Phi() ) {
2088         igvn.rehash_node_delayed(phi); // Yank from hash before hacking edges
2089         tmp = phi->in(1);
2090         phi->set_req( 1, phi->in(fall_in_cnt) );
2091         phi->set_req( fall_in_cnt, tmp );
2092       }
2093     }
2094   }
2095   assert( !phase->is_member( this, _head->in(1) ), "left edge is fall-in" );
2096   assert(  phase->is_member( this, _head->in(2) ), "right edge is loop" );
2097 
2098   // If I am a shared header (multiple backedges), peel off the many
2099   // backedges into a private merge point and use the merge point as
2100   // the one true backedge.
2101   if (_head->req() > 3) {
2102     // Merge the many backedges into a single backedge but leave
2103     // the hottest backedge as separate edge for the following peel.
2104     if (!_irreducible) {
2105       merge_many_backedges( phase );
2106     }
2107 
2108     // When recursively beautify my children, split_fall_in can change
2109     // loop tree structure when I am an irreducible loop. Then the head
2110     // of my children has a req() not bigger than 3. Here we need to set
2111     // result to true to catch that case in order to tell the caller to
2112     // rebuild loop tree. See issue JDK-8244407 for details.
2113     result = true;
2114   }
2115 
2116   // If I have one hot backedge, peel off myself loop.
2117   // I better be the outermost loop.
2118   if (_head->req() > 3 && !_irreducible) {
2119     split_outer_loop( phase );
2120     result = true;
2121 
2122   } else if (!_head->is_Loop() && !_irreducible) {
2123     // Make a new LoopNode to replace the old loop head
2124     Node *l = new LoopNode( _head->in(1), _head->in(2) );
2125     l = igvn.register_new_node_with_optimizer(l, _head);
2126     phase->set_created_loop_node();
2127     // Go ahead and replace _head
2128     phase->_igvn.replace_node( _head, l );
2129     _head = l;
2130     phase->set_loop(_head, this);
2131   }
2132 
2133   // Now recursively beautify nested loops
2134   if( _child ) result |= _child->beautify_loops( phase );
2135   if( _next  ) result |= _next ->beautify_loops( phase );
2136   return result;
2137 }

2171 //------------------------------check_safepts----------------------------
2172 // Given dominators, try to find loops with calls that must always be
2173 // executed (call dominates loop tail).  These loops do not need non-call
2174 // safepoints (ncsfpt).
2175 //
2176 // A complication is that a safepoint in a inner loop may be needed
2177 // by an outer loop. In the following, the inner loop sees it has a
2178 // call (block 3) on every path from the head (block 2) to the
2179 // backedge (arc 3->2).  So it deletes the ncsfpt (non-call safepoint)
2180 // in block 2, _but_ this leaves the outer loop without a safepoint.
2181 //
2182 //          entry  0
2183 //                 |
2184 //                 v
2185 // outer 1,2    +->1
2186 //              |  |
2187 //              |  v
2188 //              |  2<---+  ncsfpt in 2
2189 //              |_/|\   |
2190 //                 | v  |
2191 // inner 2,3      /  3  |  call in 3
2192 //               /   |  |
2193 //              v    +--+
2194 //        exit  4
2195 //
2196 //
2197 // This method creates a list (_required_safept) of ncsfpt nodes that must
2198 // be protected is created for each loop. When a ncsfpt maybe deleted, it
2199 // is first looked for in the lists for the outer loops of the current loop.
2200 //
2201 // The insights into the problem:
2202 //  A) counted loops are okay
2203 //  B) innermost loops are okay (only an inner loop can delete
2204 //     a ncsfpt needed by an outer loop)
2205 //  C) a loop is immune from an inner loop deleting a safepoint
2206 //     if the loop has a call on the idom-path
2207 //  D) a loop is also immune if it has a ncsfpt (non-call safepoint) on the
2208 //     idom-path that is not in a nested loop
2209 //  E) otherwise, an ncsfpt on the idom-path that is nested in an inner
2210 //     loop needs to be prevented from deletion by an inner loop
2211 //
2212 // There are two analyses:
2213 //  1) The first, and cheaper one, scans the loop body from
2214 //     tail to head following the idom (immediate dominator)
2215 //     chain, looking for the cases (C,D,E) above.
2216 //     Since inner loops are scanned before outer loops, there is summary
2217 //     information about inner loops.  Inner loops can be skipped over
2218 //     when the tail of an inner loop is encountered.
2219 //
2220 //  2) The second, invoked if the first fails to find a call or ncsfpt on
2221 //     the idom path (which is rare), scans all predecessor control paths
2222 //     from the tail to the head, terminating a path when a call or sfpt
2223 //     is encountered, to find the ncsfpt's that are closest to the tail.
2224 //
2225 void IdealLoopTree::check_safepts(VectorSet &visited, Node_List &stack) {
2226   // Bottom up traversal
2227   IdealLoopTree* ch = _child;
2228   if (_child) _child->check_safepts(visited, stack);
2229   if (_next)  _next ->check_safepts(visited, stack);
2230 
2231   if (!_head->is_CountedLoop() && !_has_sfpt && _parent != NULL && !_irreducible) {
2232     bool  has_call         = false; // call on dom-path
2233     bool  has_local_ncsfpt = false; // ncsfpt on dom-path at this loop depth
2234     Node* nonlocal_ncsfpt  = NULL;  // ncsfpt on dom-path at a deeper depth
2235     // Scan the dom-path nodes from tail to head
2236     for (Node* n = tail(); n != _head; n = _phase->idom(n)) {
2237       if (n->is_Call() && n->as_Call()->guaranteed_safepoint()) {
2238         has_call = true;
2239         _has_sfpt = 1;          // Then no need for a safept!
2240         break;
2241       } else if (n->Opcode() == Op_SafePoint) {
2242         if (_phase->get_loop(n) == this) {
2243           has_local_ncsfpt = true;
2244           break;
2245         }
2246         if (nonlocal_ncsfpt == NULL) {
2247           nonlocal_ncsfpt = n; // save the one closest to the tail
2248         }
2249       } else {
2250         IdealLoopTree* nlpt = _phase->get_loop(n);
2251         if (this != nlpt) {
2252           // If at an inner loop tail, see if the inner loop has already
2253           // recorded seeing a call on the dom-path (and stop.)  If not,
2254           // jump to the head of the inner loop.
2255           assert(is_member(nlpt), "nested loop");
2256           Node* tail = nlpt->_tail;
2257           if (tail->in(0)->is_If()) tail = tail->in(0);
2258           if (n == tail) {
2259             // If inner loop has call on dom-path, so does outer loop
2260             if (nlpt->_has_sfpt) {
2261               has_call = true;
2262               _has_sfpt = 1;
2263               break;
2264             }
2265             // Skip to head of inner loop
2266             assert(_phase->is_dominator(_head, nlpt->_head), "inner head dominated by outer head");
2267             n = nlpt->_head;
2268           }
2269         }
2270       }
2271     }
2272     // Record safept's that this loop needs preserved when an
2273     // inner loop attempts to delete it's safepoints.
2274     if (_child != NULL && !has_call && !has_local_ncsfpt) {
2275       if (nonlocal_ncsfpt != NULL) {
2276         if (_required_safept == NULL) _required_safept = new Node_List();
2277         _required_safept->push(nonlocal_ncsfpt);
2278       } else {
2279         // Failed to find a suitable safept on the dom-path.  Now use
2280         // an all paths walk from tail to head, looking for safepoints to preserve.
2281         allpaths_check_safepts(visited, stack);
2282       }
2283     }
2284   }
2285 }

2288 // Is safept not required by an outer loop?

2290   assert(sfpt->Opcode() == Op_SafePoint, "");

2296         if (sfpt == sfpts->at(i))

2305 //---------------------------replace_parallel_iv-------------------------------
2306 // Replace parallel induction variable (parallel to trip counter)
2307 void PhaseIdealLoop::replace_parallel_iv(IdealLoopTree *loop) {
2308   assert(loop->_head->is_CountedLoop(), "");
2309   CountedLoopNode *cl = loop->_head->as_CountedLoop();
2310   if (!cl->is_valid_counted_loop())
2311     return;         // skip malformed counted loop
2312   Node *incr = cl->incr();
2313   if (incr == NULL)
2314     return;         // Dead loop?
2315   Node *init = cl->init_trip();
2316   Node *phi  = cl->phi();
2317   int stride_con = cl->stride_con();
2318 
2319   // Visit all children, looking for Phis
2320   for (DUIterator i = cl->outs(); cl->has_out(i); i++) {
2321     Node *out = cl->out(i);
2322     // Look for other phis (secondary IVs). Skip dead ones
2323     if (!out->is_Phi() || out == phi || !has_node(out))
2324       continue;
2325     PhiNode* phi2 = out->as_Phi();
2326     Node *incr2 = phi2->in( LoopNode::LoopBackControl );
2327     // Look for induction variables of the form:  X += constant
2328     if (phi2->region() != loop->_head ||
2329         incr2->req() != 3 ||
2330         incr2->in(1) != phi2 ||
2331         incr2 == incr ||
2332         incr2->Opcode() != Op_AddI ||
2333         !incr2->in(2)->is_Con())
2334       continue;
2335 
2336     // Check for parallel induction variable (parallel to trip counter)
2337     // via an affine function.  In particular, count-down loops with
2338     // count-up array indices are common. We only RCE references off
2339     // the trip-counter, so we need to convert all these to trip-counter
2340     // expressions.
2341     Node *init2 = phi2->in( LoopNode::EntryControl );
2342     int stride_con2 = incr2->in(2)->get_int();
2343 
2344     // The ratio of the two strides cannot be represented as an int
2345     // if stride_con2 is min_int and stride_con is -1.
2346     if (stride_con2 == min_jint && stride_con == -1) {
2347       continue;
2348     }
2349 
2350     // The general case here gets a little tricky.  We want to find the
2351     // GCD of all possible parallel IV's and make a new IV using this
2352     // GCD for the loop.  Then all possible IVs are simple multiples of
2353     // the GCD.  In practice, this will cover very few extra loops.
2354     // Instead we require 'stride_con2' to be a multiple of 'stride_con',
2355     // where +/-1 is the common case, but other integer multiples are
2356     // also easy to handle.
2357     int ratio_con = stride_con2/stride_con;
2358 
2359     if ((ratio_con * stride_con) == stride_con2) { // Check for exact
2360 #ifndef PRODUCT
2361       if (TraceLoopOpts) {
2362         tty->print("Parallel IV: %d ", phi2->_idx);
2363         loop->dump_head();
2364       }
2365 #endif
2366       // Convert to using the trip counter.  The parallel induction
2367       // variable differs from the trip counter by a loop-invariant
2368       // amount, the difference between their respective initial values.
2369       // It is scaled by the 'ratio_con'.
2370       Node* ratio = _igvn.intcon(ratio_con);
2371       set_ctrl(ratio, C->root());
2372       Node* ratio_init = new MulINode(init, ratio);
2373       _igvn.register_new_node_with_optimizer(ratio_init, init);
2374       set_early_ctrl(ratio_init);
2375       Node* diff = new SubINode(init2, ratio_init);
2376       _igvn.register_new_node_with_optimizer(diff, init2);
2377       set_early_ctrl(diff);
2378       Node* ratio_idx = new MulINode(phi, ratio);
2379       _igvn.register_new_node_with_optimizer(ratio_idx, phi);
2380       set_ctrl(ratio_idx, cl);
2381       Node* add = new AddINode(ratio_idx, diff);
2382       _igvn.register_new_node_with_optimizer(add);
2383       set_ctrl(add, cl);
2384       _igvn.replace_node( phi2, add );
2385       // Sometimes an induction variable is unused
2386       if (add->outcnt() == 0) {
2387         _igvn.remove_dead_node(add);
2388       }
2389       --i; // deleted this phi; rescan starting with next position
2390       continue;
2391     }
2392   }
2393 }

2395 void IdealLoopTree::remove_safepoints(PhaseIdealLoop* phase, bool keep_one) {
2396   Node* keep = NULL;
2397   if (keep_one) {
2398     // Look for a safepoint on the idom-path.
2399     for (Node* i = tail(); i != _head; i = phase->idom(i)) {
2400       if (i->Opcode() == Op_SafePoint && phase->get_loop(i) == this) {
2401         keep = i;
2402         break; // Found one
2403       }
2404     }
2405   }
2406 
2407   // Don't remove any safepoints if it is requested to keep a single safepoint and
2408   // no safepoint was found on idom-path. It is not safe to remove any safepoint
2409   // in this case since there's no safepoint dominating all paths in the loop body.
2410   bool prune = !keep_one || keep != NULL;
2411 
2412   // Delete other safepoints in this loop.
2413   Node_List* sfpts = _safepts;
2414   if (prune && sfpts != NULL) {
2415     assert(keep == NULL || keep->Opcode() == Op_SafePoint, "not safepoint");
2416     for (uint i = 0; i < sfpts->size(); i++) {
2417       Node* n = sfpts->at(i);
2418       assert(phase->get_loop(n) == this, "");
2419       if (n != keep && phase->is_deleteable_safept(n)) {
2420         phase->lazy_replace(n, n->in(TypeFunc::Control));
2421       }
2422     }
2423   }
2424 }

2428 void IdealLoopTree::counted_loop( PhaseIdealLoop *phase ) {
2429

2431   if (!_child) {
2432     if (_head->is_Loop()) _head->as_Loop()->set_inner_loop();

2465 // The Estimated Loop Clone Size:
2466 //   CloneFactor * (~112% * BodySize + BC) + CC + FanOutTerm,
2467 // where  BC and  CC are  totally ad-hoc/magic  "body" and "clone" constants,
2468 // respectively, used to ensure that the node usage estimates made are on the
2469 // safe side, for the most part. The FanOutTerm is an attempt to estimate the
2470 // possible additional/excessive nodes generated due to data and control flow
2471 // merging, for edges reaching outside the loop.
2472 uint IdealLoopTree::est_loop_clone_sz(uint factor) const {
2473 
2474   precond(0 < factor && factor < 16);
2475 
2476   uint const bc = 13;
2477   uint const cc = 17;
2478   uint const sz = _body.size() + (_body.size() + 7) / 8;
2479   uint estimate = factor * (sz + bc) + cc;
2480 
2481   assert((estimate - cc) / factor == sz + bc, "overflow");
2482 
2483   return estimate + est_loop_flow_merge_sz();
2484 }

2486 // The Estimated Loop (full-) Unroll Size:
2487 //   UnrollFactor * (~106% * BodySize) + CC + FanOutTerm,
2488 // where CC is a (totally) ad-hoc/magic "clone" constant, used to ensure that
2489 // node usage estimates made are on the safe side, for the most part. This is
2490 // a "light" version of the loop clone size calculation (above), based on the
2491 // assumption that most of the loop-construct overhead will be unraveled when
2492 // (fully) unrolled. Defined for unroll factors larger or equal to one (>=1),
2493 // including an overflow check and returning UINT_MAX in case of an overflow.
2494 uint IdealLoopTree::est_loop_unroll_sz(uint factor) const {
2495 
2496   precond(factor > 0);
2497 
2498   // Take into account that after unroll conjoined heads and tails will fold.
2499   uint const b0 = _body.size() - EMPTY_LOOP_SIZE;
2500   uint const cc = 7;
2501   uint const sz = b0 + (b0 + 15) / 16;
2502   uint estimate = factor * sz + cc;
2503 
2504   if ((estimate - cc) / factor != sz) {
2505     return UINT_MAX;
2506   }
2507 
2508   return estimate + est_loop_flow_merge_sz();
2509 }

2511 // Estimate the growth effect (in nodes) of merging control and data flow when
2512 // cloning a loop body, based on the amount of  control and data flow reaching
2513 // outside of the (current) loop body.
2514 uint IdealLoopTree::est_loop_flow_merge_sz() const {
2515 
2516   uint ctrl_edge_out_cnt = 0;
2517   uint data_edge_out_cnt = 0;
2518 
2519   for (uint i = 0; i < _body.size(); i++) {
2520     Node* node = _body.at(i);
2521     uint outcnt = node->outcnt();
2522 
2523     for (uint k = 0; k < outcnt; k++) {
2524       Node* out = node->raw_out(k);
2525       if (out == NULL) continue;
2526       if (out->is_CFG()) {
2527         if (!is_member(_phase->get_loop(out))) {
2528           ctrl_edge_out_cnt++;
2529         }
2530       } else if (_phase->has_ctrl(out)) {
2531         Node* ctrl = _phase->get_ctrl(out);
2532         assert(ctrl != NULL, "must be");
2533         assert(ctrl->is_CFG(), "must be");
2534         if (!is_member(_phase->get_loop(ctrl))) {
2535           data_edge_out_cnt++;
2536         }
2537       }
2538     }
2539   }
2540   // Use data and control count (x2.0) in estimate iff both are > 0. This is
2541   // a rather pessimistic estimate for the most part, in particular for some
2542   // complex loops, but still not enough to capture all loops.
2543   if (ctrl_edge_out_cnt > 0 && data_edge_out_cnt > 0) {
2544     return 2 * (ctrl_edge_out_cnt + data_edge_out_cnt);
2545   }
2546   return 0;
2547 }

2704 void PhaseIdealLoop::eliminate_useless_predicates() {

2722 //------------------------process_expensive_nodes-----------------------------
2723 // Expensive nodes have their control input set to prevent the GVN
2724 // from commoning them and as a result forcing the resulting node to
2725 // be in a more frequent path. Use CFG information here, to change the
2726 // control inputs so that some expensive nodes can be commoned while
2727 // not executed more frequently.
2728 bool PhaseIdealLoop::process_expensive_nodes() {
2729   assert(OptimizeExpensiveOps, "optimization off?");
2730 
2731   // Sort nodes to bring similar nodes together
2732   C->sort_expensive_nodes();
2733 
2734   bool progress = false;
2735 
2736   for (int i = 0; i < C->expensive_count(); ) {
2737     Node* n = C->expensive_node(i);
2738     int start = i;
2739     // Find nodes similar to n
2740     i++;
2741     for (; i < C->expensive_count() && Compile::cmp_expensive_nodes(n, C->expensive_node(i)) == 0; i++);
2742     int end = i;
2743     // And compare them two by two
2744     for (int j = start; j < end; j++) {
2745       Node* n1 = C->expensive_node(j);
2746       if (is_node_unreachable(n1)) {
2747         continue;
2748       }
2749       for (int k = j+1; k < end; k++) {
2750         Node* n2 = C->expensive_node(k);
2751         if (is_node_unreachable(n2)) {
2752           continue;
2753         }
2754 
2755         assert(n1 != n2, "should be pair of nodes");
2756 
2757         Node* c1 = n1->in(0);
2758         Node* c2 = n2->in(0);
2759 
2760         Node* parent_c1 = c1;
2761         Node* parent_c2 = c2;
2762 
2763         // The call to get_early_ctrl_for_expensive() moves the
2764         // expensive nodes up but stops at loops that are in a if
2765         // branch. See whether we can exit the loop and move above the
2766         // If.
2767         if (c1->is_Loop()) {
2768           parent_c1 = c1->in(1);
2769         }
2770         if (c2->is_Loop()) {
2771           parent_c2 = c2->in(1);
2772         }
2773 
2774         if (parent_c1 == parent_c2) {
2775           _igvn._worklist.push(n1);
2776           _igvn._worklist.push(n2);
2777           continue;
2778         }
2779 
2780         // Look for identical expensive node up the dominator chain.
2781         if (is_dominator(c1, c2)) {
2782           c2 = c1;
2783         } else if (is_dominator(c2, c1)) {
2784           c1 = c2;
2785         } else if (parent_c1->is_Proj() && parent_c1->in(0)->is_If() &&
2786                    parent_c2->is_Proj() && parent_c1->in(0) == parent_c2->in(0)) {
2787           // Both branches have the same expensive node so move it up
2788           // before the if.
2789           c1 = c2 = idom(parent_c1->in(0));
2790         }
2791         // Do the actual moves
2792         if (n1->in(0) != c1) {
2793           _igvn.hash_delete(n1);
2794           n1->set_req(0, c1);
2795           _igvn.hash_insert(n1);
2796           _igvn._worklist.push(n1);
2797           progress = true;
2798         }
2799         if (n2->in(0) != c2) {
2800           _igvn.hash_delete(n2);
2801           n2->set_req(0, c2);
2802           _igvn.hash_insert(n2);
2803           _igvn._worklist.push(n2);
2804           progress = true;
2805         }
2806       }
2807     }
2808   }
2809 
2810   return progress;
2811 }

2816 // Create a PhaseLoop.  Build the ideal Loop tree.  Map each Ideal Node to
2817 // its corresponding LoopNode.  If 'optimize' is true, do some loop cleanups.

2820   bool skip_loop_opts = (mode == LoopOptsNone);
2821 
2822   int old_progress = C->major_progress();
2823   uint orig_worklist_size = _igvn._worklist.size();
2824 
2825   // Reset major-progress flag for the driver's heuristics
2826   C->clear_major_progress();
2827 
2828 #ifndef PRODUCT
2829   // Capture for later assert
2830   uint unique = C->unique();
2831   _loop_invokes++;
2832   _loop_work += unique;
2833 #endif
2834 
2835   // True if the method has at least 1 irreducible loop
2836   _has_irreducible_loops = false;
2837 
2838   _created_loop_node = false;
2839 
2840   Arena *a = Thread::current()->resource_area();
2841   VectorSet visited(a);
2842   // Pre-grow the mapping from Nodes to IdealLoopTrees.
2843   _nodes.map(C->unique(), NULL);
2844   memset(_nodes.adr(), 0, wordSize * C->unique());
2845 
2846   // Pre-build the top-level outermost loop tree entry
2847   _ltree_root = new IdealLoopTree( this, C->root(), C->root() );
2848   // Do not need a safepoint at the top level
2849   _ltree_root->_has_sfpt = 1;
2850 
2851   // Initialize Dominators.
2852   // Checked in clone_loop_predicate() during beautify_loops().
2853   _idom_size = 0;
2854   _idom      = NULL;
2855   _dom_depth = NULL;
2856   _dom_stk   = NULL;
2857 
2858   // Empty pre-order array
2859   allocate_preorders();
2860 
2861   // Build a loop tree on the fly.  Build a mapping from CFG nodes to
2862   // IdealLoopTree entries.  Data nodes are NOT walked.
2863   build_loop_tree();
2864   // Check for bailout, and return
2865   if (C->failing()) {
2866     return;
2867   }
2868 
2869   // No loops after all
2870   if( !_ltree_root->_child && !_verify_only ) C->set_has_loops(false);
2871 
2872   // There should always be an outer loop containing the Root and Return nodes.
2873   // If not, we have a degenerate empty program.  Bail out in this case.
2874   if (!has_node(C->root())) {
2875     if (!_verify_only) {
2876       C->clear_major_progress();
2877       C->record_method_not_compilable("empty program detected during loop optimization");
2878     }
2879     return;
2880   }
2881 
2882   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
2883   // Nothing to do, so get out
2884   bool stop_early = !C->has_loops() && !skip_loop_opts && !do_split_ifs && !_verify_me && !_verify_only &&
2885     !bs->is_gc_specific_loop_opts_pass(mode);
2886   bool do_expensive_nodes = C->should_optimize_expensive_nodes(_igvn);
2887   bool strip_mined_loops_expanded = bs->strip_mined_loops_expanded(mode);
2888   if (stop_early && !do_expensive_nodes) {
2889     _igvn.optimize();           // Cleanup NeverBranches
2890     return;
2891   }
2892 
2893   // Set loop nesting depth
2894   _ltree_root->set_nest( 0 );
2895 
2896   // Split shared headers and insert loop landing pads.
2897   // Do not bother doing this on the Root loop of course.
2898   if( !_verify_me && !_verify_only && _ltree_root->_child ) {
2899     C->print_method(PHASE_BEFORE_BEAUTIFY_LOOPS, 3);
2900     if( _ltree_root->_child->beautify_loops( this ) ) {
2901       // Re-build loop tree!
2902       _ltree_root->_child = NULL;
2903       _nodes.clear();
2904       reallocate_preorders();
2905       build_loop_tree();
2906       // Check for bailout, and return
2907       if (C->failing()) {
2908         return;
2909       }
2910       // Reset loop nesting depth
2911       _ltree_root->set_nest( 0 );
2912 
2913       C->print_method(PHASE_AFTER_BEAUTIFY_LOOPS, 3);
2914     }
2915   }
2916 
2917   // Build Dominators for elision of NULL checks & loop finding.
2918   // Since nodes do not have a slot for immediate dominator, make
2919   // a persistent side array for that info indexed on node->_idx.
2920   _idom_size = C->unique();
2921   _idom      = NEW_RESOURCE_ARRAY( Node*, _idom_size );
2922   _dom_depth = NEW_RESOURCE_ARRAY( uint,  _idom_size );
2923   _dom_stk   = NULL; // Allocated on demand in recompute_dom_depth
2924   memset( _dom_depth, 0, _idom_size * sizeof(uint) );
2925 
2926   Dominators();
2927 
2928   if (!_verify_only) {
2929     // As a side effect, Dominators removed any unreachable CFG paths
2930     // into RegionNodes.  It doesn't do this test against Root, so
2931     // we do it here.
2932     for( uint i = 1; i < C->root()->req(); i++ ) {
2933       if( !_nodes[C->root()->in(i)->_idx] ) {    // Dead path into Root?
2934         _igvn.delete_input_of(C->root(), i);
2935         i--;                      // Rerun same iteration on compressed edges
2936       }
2937     }
2938 
2939     // Given dominators, try to find inner loops with calls that must
2940     // always be executed (call dominates loop tail).  These loops do
2941     // not need a separate safepoint.
2942     Node_List cisstack(a);
2943     _ltree_root->check_safepts(visited, cisstack);
2944   }
2945 
2946   // Walk the DATA nodes and place into loops.  Find earliest control
2947   // node.  For CFG nodes, the _nodes array starts out and remains
2948   // holding the associated IdealLoopTree pointer.  For DATA nodes, the
2949   // _nodes array holds the earliest legal controlling CFG node.
2950 
2951   // Allocate stack with enough space to avoid frequent realloc
2952   int stack_size = (C->live_nodes() >> 1) + 16; // (live_nodes>>1)+16 from Java2D stats
2953   Node_Stack nstack( a, stack_size );
2954 
2955   visited.clear();
2956   Node_List worklist(a);
2957   // Don't need C->root() on worklist since
2958   // it will be processed among C->top() inputs
2959   worklist.push(C->top());
2960   visited.set(C->top()->_idx); // Set C->top() as visited now
2961   build_loop_early( visited, worklist, nstack );
2962 
2963   // Given early legal placement, try finding counted loops.  This placement
2964   // is good enough to discover most loop invariants.
2965   if (!_verify_me && !_verify_only && !strip_mined_loops_expanded) {
2966     _ltree_root->counted_loop( this );
2967   }
2968 
2969   // Find latest loop placement.  Find ideal loop placement.
2970   visited.clear();
2971   init_dom_lca_tags();
2972   // Need C->root() on worklist when processing outs
2973   worklist.push(C->root());
2974   NOT_PRODUCT( C->verify_graph_edges(); )
2975   worklist.push(C->top());
2976   build_loop_late( visited, worklist, nstack );
2977 
2978   if (_verify_only) {
2979     C->restore_major_progress(old_progress);
2980     assert(C->unique() == unique, "verification mode made Nodes? ? ?");
2981     assert(_igvn._worklist.size() == orig_worklist_size, "shouldn't push anything");
2982     return;
2983   }
2984 
2985   // clear out the dead code after build_loop_late
2986   while (_deadlist.size()) {
2987     _igvn.remove_globally_dead_node(_deadlist.pop());
2988   }
2989 
2990   if (stop_early) {
2991     assert(do_expensive_nodes, "why are we here?");
2992     if (process_expensive_nodes()) {
2993       // If we made some progress when processing expensive nodes then
2994       // the IGVN may modify the graph in a way that will allow us to
2995       // make some more progress: we need to try processing expensive
2996       // nodes again.
2997       C->set_major_progress();
2998     }
2999     _igvn.optimize();
3000     return;
3001   }
3002 
3003   // Some parser-inserted loop predicates could never be used by loop
3004   // predication or they were moved away from loop during some optimizations.
3005   // For example, peeling. Eliminate them before next loop optimizations.
3006   eliminate_useless_predicates();
3007 
3008 #ifndef PRODUCT
3009   C->verify_graph_edges();
3010   if (_verify_me) {             // Nested verify pass?
3011     // Check to see if the verify mode is broken
3012     assert(C->unique() == unique, "non-optimize mode made Nodes? ? ?");
3013     return;
3014   }
3015   if (VerifyLoopOptimizations) verify();
3016   if (TraceLoopOpts && C->has_loops()) {
3017     _ltree_root->dump();
3018   }
3019 #endif
3020 
3021   if (skip_loop_opts) {
3022     // restore major progress flag
3023     C->restore_major_progress(old_progress);
3024 
3025     // Cleanup any modified bits
3026     _igvn.optimize();
3027 
3028     if (C->log() != NULL) {
3029       log_loop_tree(_ltree_root, _ltree_root, C->log());
3030     }
3031     return;
3032   }
3033 
3034   if (mode == LoopOptsMaxUnroll) {
3035     for (LoopTreeIterator iter(_ltree_root); !iter.done(); iter.next()) {
3036       IdealLoopTree* lpt = iter.current();
3037       if (lpt->is_innermost() && lpt->_allow_optimizations && !lpt->_has_call && lpt->is_counted()) {
3038         lpt->compute_trip_count(this);
3039         if (!lpt->do_one_iteration_loop(this) &&
3040             !lpt->do_remove_empty_loop(this)) {
3041           AutoNodeBudget node_budget(this);
3042           if (lpt->_head->as_CountedLoop()->is_normal_loop() &&
3043               lpt->policy_maximally_unroll(this)) {
3044             memset( worklist.adr(), 0, worklist.Size()*sizeof(Node*) );
3045             do_maximally_unroll(lpt, worklist);
3046           }
3047         }
3048       }
3049     }
3050 
3051     C->restore_major_progress(old_progress);
3052 
3053     _igvn.optimize();
3054 
3055     if (C->log() != NULL) {
3056       log_loop_tree(_ltree_root, _ltree_root, C->log());
3057     }
3058     return;
3059   }
3060 
3061   if (bs->optimize_loops(this, mode, visited, nstack, worklist)) {
3062     _igvn.optimize();
3063     if (C->log() != NULL) {
3064       log_loop_tree(_ltree_root, _ltree_root, C->log());
3065     }
3066     return;
3067   }
3068 
3069   if (ReassociateInvariants) {
3070     // Reassociate invariants and prep for split_thru_phi
3071     for (LoopTreeIterator iter(_ltree_root); !iter.done(); iter.next()) {
3072       IdealLoopTree* lpt = iter.current();
3073       bool is_counted = lpt->is_counted();
3074       if (!is_counted || !lpt->is_innermost()) continue;
3075 
3076       // check for vectorized loops, any reassociation of invariants was already done
3077       if (is_counted && lpt->_head->as_CountedLoop()->is_unroll_only()) {
3078         continue;
3079       } else {
3080         AutoNodeBudget node_budget(this);
3081         lpt->reassociate_invariants(this);
3082       }
3083       // Because RCE opportunities can be masked by split_thru_phi,
3084       // look for RCE candidates and inhibit split_thru_phi
3085       // on just their loop-phi's for this pass of loop opts
3086       if (SplitIfBlocks && do_split_ifs) {
3087         AutoNodeBudget node_budget(this, AutoNodeBudget::NO_BUDGET_CHECK);
3088         if (lpt->policy_range_check(this)) {
3089           lpt->_rce_candidate = 1; // = true
3090         }
3091       }
3092     }
3093   }
3094 
3095   // Check for aggressive application of split-if and other transforms
3096   // that require basic-block info (like cloning through Phi's)
3097   if( SplitIfBlocks && do_split_ifs ) {
3098     visited.clear();
3099     split_if_with_blocks( visited, nstack);
3100     NOT_PRODUCT( if( VerifyLoopOptimizations ) verify(); );
3101   }
3102 
3103   if (!C->major_progress() && do_expensive_nodes && process_expensive_nodes()) {
3104     C->set_major_progress();
3105   }
3106 
3107   // Perform loop predication before iteration splitting
3108   if (C->has_loops() && !C->major_progress() && (C->predicate_count() > 0)) {
3109     _ltree_root->_child->loop_predication(this);
3110   }
3111 
3112   if (OptimizeFill && UseLoopPredicate && C->has_loops() && !C->major_progress()) {
3113     if (do_intrinsify_fill()) {
3114       C->set_major_progress();
3115     }
3116   }
3117 
3118   // Perform iteration-splitting on inner loops.  Split iterations to avoid
3119   // range checks or one-shot null checks.
3120 
3121   // If split-if's didn't hack the graph too bad (no CFG changes)
3122   // then do loop opts.
3123   if (C->has_loops() && !C->major_progress()) {
3124     memset( worklist.adr(), 0, worklist.Size()*sizeof(Node*) );
3125     _ltree_root->_child->iteration_split( this, worklist );
3126     // No verify after peeling!  GCM has hoisted code out of the loop.
3127     // After peeling, the hoisted code could sink inside the peeled area.
3128     // The peeling code does not try to recompute the best location for
3129     // all the code before the peeled area, so the verify pass will always
3130     // complain about it.
3131   }
3132   // Do verify graph edges in any case
3133   NOT_PRODUCT( C->verify_graph_edges(); );
3134 
3135   if (!do_split_ifs) {
3136     // We saw major progress in Split-If to get here.  We forced a
3137     // pass with unrolling and not split-if, however more split-if's
3138     // might make progress.  If the unrolling didn't make progress
3139     // then the major-progress flag got cleared and we won't try
3140     // another round of Split-If.  In particular the ever-common
3141     // instance-of/check-cast pattern requires at least 2 rounds of
3142     // Split-If to clear out.
3143     C->set_major_progress();
3144   }
3145 
3146   // Repeat loop optimizations if new loops were seen
3147   if (created_loop_node()) {
3148     C->set_major_progress();
3149   }
3150 
3151   // Keep loop predicates and perform optimizations with them
3152   // until no more loop optimizations could be done.
3153   // After that switch predicates off and do more loop optimizations.
3154   if (!C->major_progress() && (C->predicate_count() > 0)) {
3155      C->cleanup_loop_predicates(_igvn);
3156      if (TraceLoopOpts) {
3157        tty->print_cr("PredicatesOff");
3158      }
3159      C->set_major_progress();
3160   }
3161 
3162   // Convert scalar to superword operations at the end of all loop opts.
3163   if (UseSuperWord && C->has_loops() && !C->major_progress()) {
3164     // SuperWord transform
3165     SuperWord sw(this);
3166     for (LoopTreeIterator iter(_ltree_root); !iter.done(); iter.next()) {
3167       IdealLoopTree* lpt = iter.current();
3168       if (lpt->is_counted()) {
3169         CountedLoopNode *cl = lpt->_head->as_CountedLoop();
3170 
3171         if (PostLoopMultiversioning && cl->is_rce_post_loop() && !cl->is_vectorized_loop()) {
3172           // Check that the rce'd post loop is encountered first, multiversion after all
3173           // major main loop optimization are concluded
3174           if (!C->major_progress()) {
3175             IdealLoopTree *lpt_next = lpt->_next;
3176             if (lpt_next && lpt_next->is_counted()) {
3177               CountedLoopNode *cl = lpt_next->_head->as_CountedLoop();
3178               has_range_checks(lpt_next);
3179               if (cl->is_post_loop() && cl->range_checks_present()) {
3180                 if (!cl->is_multiversioned()) {
3181                   if (multi_version_post_loops(lpt, lpt_next) == false) {
3182                     // Cause the rce loop to be optimized away if we fail
3183                     cl->mark_is_multiversioned();
3184                     cl->set_slp_max_unroll(0);
3185                     poison_rce_post_loop(lpt);
3186                   }
3187                 }
3188               }
3189             }
3190             sw.transform_loop(lpt, true);
3191           }
3192         } else if (cl->is_main_loop()) {
3193           sw.transform_loop(lpt, true);
3194         }
3195       }
3196     }
3197   }
3198 
3199   // Cleanup any modified bits
3200   _igvn.optimize();
3201 
3202   // disable assert until issue with split_flow_path is resolved (6742111)
3203   // assert(!_has_irreducible_loops || C->parsed_irreducible_loop() || C->is_osr_compilation(),
3204   //        "shouldn't introduce irreducible loops");
3205 
3206   if (C->log() != NULL) {
3207     log_loop_tree(_ltree_root, _ltree_root, C->log());
3208   }
3209 }

3453 //------------------------------recompute_dom_depth---------------------------------------
3454 // The dominator tree is constructed with only parent pointers.
3455 // This recomputes the depth in the tree by first tagging all
3456 // nodes as "no depth yet" marker.  The next pass then runs up
3457 // the dom tree from each node marked "no depth yet", and computes
3458 // the depth on the way back down.
3459 void PhaseIdealLoop::recompute_dom_depth() {
3460   uint no_depth_marker = C->unique();
3461   uint i;
3462   // Initialize depth to "no depth yet" and realize all lazy updates
3463   for (i = 0; i < _idom_size; i++) {
3464     // Only indices with a _dom_depth has a Node* or NULL (otherwise uninitalized).
3465     if (_dom_depth[i] > 0 && _idom[i] != NULL) {
3466       _dom_depth[i] = no_depth_marker;
3467 
3468       // heal _idom if it has a fwd mapping in _nodes
3469       if (_idom[i]->in(0) == NULL) {
3470         idom(i);
3471       }
3472     }
3473   }
3474   if (_dom_stk == NULL) {
3475     uint init_size = C->live_nodes() / 100; // Guess that 1/100 is a reasonable initial size.
3476     if (init_size < 10) init_size = 10;
3477     _dom_stk = new GrowableArray<uint>(init_size);
3478   }
3479   // Compute new depth for each node.
3480   for (i = 0; i < _idom_size; i++) {
3481     uint j = i;
3482     // Run up the dom tree to find a node with a depth
3483     while (_dom_depth[j] == no_depth_marker) {
3484       _dom_stk->push(j);
3485       j = _idom[j]->_idx;
3486     }
3487     // Compute the depth on the way back down this tree branch
3488     uint dd = _dom_depth[j] + 1;
3489     while (_dom_stk->length() > 0) {
3490       uint j = _dom_stk->pop();
3491       _dom_depth[j] = dd;
3492       dd++;
3493     }
3494   }

3497 //------------------------------sort-------------------------------------------
3498 // Insert 'loop' into the existing loop tree.  'innermost' is a leaf of the
3499 // loop tree, not the root.
3500 IdealLoopTree *PhaseIdealLoop::sort( IdealLoopTree *loop, IdealLoopTree *innermost ) {
3501   if( !innermost ) return loop; // New innermost loop
3502 
3503   int loop_preorder = get_preorder(loop->_head); // Cache pre-order number
3504   assert( loop_preorder, "not yet post-walked loop" );
3505   IdealLoopTree **pp = &innermost;      // Pointer to previous next-pointer
3506   IdealLoopTree *l = *pp;               // Do I go before or after 'l'?
3507 
3508   // Insert at start of list
3509   while( l ) {                  // Insertion sort based on pre-order
3510     if( l == loop ) return innermost; // Already on list!
3511     int l_preorder = get_preorder(l->_head); // Cache pre-order number
3512     assert( l_preorder, "not yet post-walked l" );
3513     // Check header pre-order number to figure proper nesting
3514     if( loop_preorder > l_preorder )
3515       break;                    // End of insertion
3516     // If headers tie (e.g., shared headers) check tail pre-order numbers.
3517     // Since I split shared headers, you'd think this could not happen.
3518     // BUT: I must first do the preorder numbering before I can discover I
3519     // have shared headers, so the split headers all get the same preorder
3520     // number as the RegionNode they split from.
3521     if( loop_preorder == l_preorder &&
3522         get_preorder(loop->_tail) < get_preorder(l->_tail) )
3523       break;                    // Also check for shared headers (same pre#)
3524     pp = &l->_parent;           // Chain up list
3525     l = *pp;
3526   }
3527   // Link into list
3528   // Point predecessor to me
3529   *pp = loop;
3530   // Point me to successor
3531   IdealLoopTree *p = loop->_parent;
3532   loop->_parent = l;            // Point me to successor
3533   if( p ) sort( p, innermost ); // Insert my parents into list as well
3534   return innermost;
3535 }

3537 //------------------------------build_loop_tree--------------------------------
3538 // I use a modified Vick/Tarjan algorithm.  I need pre- and a post- visit
3539 // bits.  The _nodes[] array is mapped by Node index and holds a NULL for
3540 // not-yet-pre-walked, pre-order # for pre-but-not-post-walked and holds the
3541 // tightest enclosing IdealLoopTree for post-walked.
3542 //
3543 // During my forward walk I do a short 1-layer lookahead to see if I can find
3544 // a loop backedge with that doesn't have any work on the backedge.  This
3545 // helps me construct nested loops with shared headers better.
3546 //
3547 // Once I've done the forward recursion, I do the post-work.  For each child
3548 // I check to see if there is a backedge.  Backedges define a loop!  I
3549 // insert an IdealLoopTree at the target of the backedge.
3550 //
3551 // During the post-work I also check to see if I have several children
3552 // belonging to different loops.  If so, then this Node is a decision point
3553 // where control flow can choose to change loop nests.  It is at this
3554 // decision point where I can figure out how loops are nested.  At this
3555 // time I can properly order the different loop nests from my children.
3556 // Note that there may not be any backedges at the decision point!
3557 //
3558 // Since the decision point can be far removed from the backedges, I can't
3559 // order my loops at the time I discover them.  Thus at the decision point
3560 // I need to inspect loop header pre-order numbers to properly nest my
3561 // loops.  This means I need to sort my childrens' loops by pre-order.
3562 // The sort is of size number-of-control-children, which generally limits
3563 // it to size 2 (i.e., I just choose between my 2 target loops).
3564 void PhaseIdealLoop::build_loop_tree() {
3565   // Allocate stack of size C->live_nodes()/2 to avoid frequent realloc
3566   GrowableArray <Node *> bltstack(C->live_nodes() >> 1);
3567   Node *n = C->root();
3568   bltstack.push(n);
3569   int pre_order = 1;
3570   int stack_size;
3571 
3572   while ( ( stack_size = bltstack.length() ) != 0 ) {
3573     n = bltstack.top(); // Leave node on stack
3574     if ( !is_visited(n) ) {
3575       // ---- Pre-pass Work ----
3576       // Pre-walked but not post-walked nodes need a pre_order number.
3577 
3578       set_preorder_visited( n, pre_order ); // set as visited
3579 
3580       // ---- Scan over children ----
3581       // Scan first over control projections that lead to loop headers.
3582       // This helps us find inner-to-outer loops with shared headers better.
3583 
3584       // Scan children's children for loop headers.
3585       for ( int i = n->outcnt() - 1; i >= 0; --i ) {
3586         Node* m = n->raw_out(i);       // Child
3587         if( m->is_CFG() && !is_visited(m) ) { // Only for CFG children
3588           // Scan over children's children to find loop
3589           for (DUIterator_Fast jmax, j = m->fast_outs(jmax); j < jmax; j++) {
3590             Node* l = m->fast_out(j);
3591             if( is_visited(l) &&       // Been visited?
3592                 !is_postvisited(l) &&  // But not post-visited
3593                 get_preorder(l) < pre_order ) { // And smaller pre-order
3594               // Found!  Scan the DFS down this path before doing other paths
3595               bltstack.push(m);
3596               break;
3597             }
3598           }
3599         }
3600       }
3601       pre_order++;
3602     }
3603     else if ( !is_postvisited(n) ) {
3604       // Note: build_loop_tree_impl() adds out edges on rare occasions,
3605       // such as com.sun.rsasign.am::a.
3606       // For non-recursive version, first, process current children.
3607       // On next iteration, check if additional children were added.
3608       for ( int k = n->outcnt() - 1; k >= 0; --k ) {
3609         Node* u = n->raw_out(k);
3610         if ( u->is_CFG() && !is_visited(u) ) {
3611           bltstack.push(u);
3612         }
3613       }
3614       if ( bltstack.length() == stack_size ) {
3615         // There were no additional children, post visit node now
3616         (void)bltstack.pop(); // Remove node from stack
3617         pre_order = build_loop_tree_impl( n, pre_order );
3618         // Check for bailout
3619         if (C->failing()) {
3620           return;
3621         }
3622         // Check to grow _preorders[] array for the case when
3623         // build_loop_tree_impl() adds new nodes.
3624         check_grow_preorders();
3625       }
3626     }
3627     else {
3628       (void)bltstack.pop(); // Remove post-visited node from stack
3629     }
3630   }
3631 }

3633 //------------------------------build_loop_tree_impl---------------------------
3634 int PhaseIdealLoop::build_loop_tree_impl( Node *n, int pre_order ) {
3635   // ---- Post-pass Work ----
3636   // Pre-walked but not post-walked nodes need a pre_order number.
3637 
3638   // Tightest enclosing loop for this Node
3639   IdealLoopTree *innermost = NULL;
3640 
3641   // For all children, see if any edge is a backedge.  If so, make a loop
3642   // for it.  Then find the tightest enclosing loop for the self Node.
3643   for (DUIterator_Fast imax, i = n->fast_outs(imax); i < imax; i++) {
3644     Node* m = n->fast_out(i);   // Child
3645     if( n == m ) continue;      // Ignore control self-cycles
3646     if( !m->is_CFG() ) continue;// Ignore non-CFG edges
3647 
3648     IdealLoopTree *l;           // Child's loop
3649     if( !is_postvisited(m) ) {  // Child visited but not post-visited?
3650       // Found a backedge
3651       assert( get_preorder(m) < pre_order, "should be backedge" );
3652       // Check for the RootNode, which is already a LoopNode and is allowed
3653       // to have multiple "backedges".
3654       if( m == C->root()) {     // Found the root?
3655         l = _ltree_root;        // Root is the outermost LoopNode
3656       } else {                  // Else found a nested loop
3657         // Insert a LoopNode to mark this loop.
3658         l = new IdealLoopTree(this, m, n);
3659       } // End of Else found a nested loop
3660       if( !has_loop(m) )        // If 'm' does not already have a loop set
3661         set_loop(m, l);         // Set loop header to loop now
3662 
3663     } else {                    // Else not a nested loop
3664       if( !_nodes[m->_idx] ) continue; // Dead code has no loop
3665       l = get_loop(m);          // Get previously determined loop
3666       // If successor is header of a loop (nest), move up-loop till it
3667       // is a member of some outer enclosing loop.  Since there are no
3668       // shared headers (I've split them already) I only need to go up
3669       // at most 1 level.
3670       while( l && l->_head == m ) // Successor heads loop?
3671         l = l->_parent;         // Move up 1 for me
3672       // If this loop is not properly parented, then this loop
3673       // has no exit path out, i.e. its an infinite loop.
3674       if( !l ) {
3675         // Make loop "reachable" from root so the CFG is reachable.  Basically
3676         // insert a bogus loop exit that is never taken.  'm', the loop head,
3677         // points to 'n', one (of possibly many) fall-in paths.  There may be
3678         // many backedges as well.
3679 
3680         // Here I set the loop to be the root loop.  I could have, after
3681         // inserting a bogus loop exit, restarted the recursion and found my
3682         // new loop exit.  This would make the infinite loop a first-class
3683         // loop and it would then get properly optimized.  What's the use of
3684         // optimizing an infinite loop?
3685         l = _ltree_root;        // Oops, found infinite loop
3686 
3687         if (!_verify_only) {
3688           // Insert the NeverBranch between 'm' and it's control user.
3689           NeverBranchNode *iff = new NeverBranchNode( m );
3690           _igvn.register_new_node_with_optimizer(iff);
3691           set_loop(iff, l);
3692           Node *if_t = new CProjNode( iff, 0 );
3693           _igvn.register_new_node_with_optimizer(if_t);
3694           set_loop(if_t, l);
3695 
3696           Node* cfg = NULL;       // Find the One True Control User of m
3697           for (DUIterator_Fast jmax, j = m->fast_outs(jmax); j < jmax; j++) {
3698             Node* x = m->fast_out(j);
3699             if (x->is_CFG() && x != m && x != iff)
3700               { cfg = x; break; }
3701           }
3702           assert(cfg != NULL, "must find the control user of m");
3703           uint k = 0;             // Probably cfg->in(0)
3704           while( cfg->in(k) != m ) k++; // But check incase cfg is a Region
3705           cfg->set_req( k, if_t ); // Now point to NeverBranch
3706           _igvn._worklist.push(cfg);
3707 
3708           // Now create the never-taken loop exit
3709           Node *if_f = new CProjNode( iff, 1 );
3710           _igvn.register_new_node_with_optimizer(if_f);
3711           set_loop(if_f, l);
3712           // Find frame ptr for Halt.  Relies on the optimizer
3713           // V-N'ing.  Easier and quicker than searching through
3714           // the program structure.
3715           Node *frame = new ParmNode( C->start(), TypeFunc::FramePtr );
3716           _igvn.register_new_node_with_optimizer(frame);
3717           // Halt & Catch Fire
3718           Node* halt = new HaltNode(if_f, frame, "never-taken loop exit reached");
3719           _igvn.register_new_node_with_optimizer(halt);
3720           set_loop(halt, l);
3721           C->root()->add_req(halt);
3722         }
3723         set_loop(C->root(), _ltree_root);
3724       }
3725     }
3726     // Weeny check for irreducible.  This child was already visited (this
3727     // IS the post-work phase).  Is this child's loop header post-visited
3728     // as well?  If so, then I found another entry into the loop.
3729     if (!_verify_only) {
3730       while( is_postvisited(l->_head) ) {
3731         // found irreducible
3732         l->_irreducible = 1; // = true
3733         l = l->_parent;
3734         _has_irreducible_loops = true;
3735         // Check for bad CFG here to prevent crash, and bailout of compile
3736         if (l == NULL) {
3737           C->record_method_not_compilable("unhandled CFG detected during loop optimization");
3738           return pre_order;
3739         }
3740       }
3741       C->set_has_irreducible_loop(_has_irreducible_loops);
3742     }
3743 
3744     // This Node might be a decision point for loops.  It is only if
3745     // it's children belong to several different loops.  The sort call
3746     // does a trivial amount of work if there is only 1 child or all
3747     // children belong to the same loop.  If however, the children
3748     // belong to different loops, the sort call will properly set the
3749     // _parent pointers to show how the loops nest.
3750     //
3751     // In any case, it returns the tightest enclosing loop.
3752     innermost = sort( l, innermost );
3753   }
3754 
3755   // Def-use info will have some dead stuff; dead stuff will have no
3756   // loop decided on.
3757 
3758   // Am I a loop header?  If so fix up my parent's child and next ptrs.
3759   if( innermost && innermost->_head == n ) {
3760     assert( get_loop(n) == innermost, "" );
3761     IdealLoopTree *p = innermost->_parent;
3762     IdealLoopTree *l = innermost;
3763     while( p && l->_head == n ) {
3764       l->_next = p->_child;     // Put self on parents 'next child'
3765       p->_child = l;            // Make self as first child of parent
3766       l = p;                    // Now walk up the parent chain
3767       p = l->_parent;
3768     }
3769   } else {
3770     // Note that it is possible for a LoopNode to reach here, if the
3771     // backedge has been made unreachable (hence the LoopNode no longer
3772     // denotes a Loop, and will eventually be removed).
3773 
3774     // Record tightest enclosing loop for self.  Mark as post-visited.
3775     set_loop(n, innermost);
3776     // Also record has_call flag early on
3777     if( innermost ) {
3778       if( n->is_Call() && !n->is_CallLeaf() && !n->is_macro() ) {
3779         // Do not count uncommon calls
3780         if( !n->is_CallStaticJava() || !n->as_CallStaticJava()->_name ) {
3781           Node *iff = n->in(0)->in(0);
3782           // No any calls for vectorized loops.
3783           if( UseSuperWord || !iff->is_If() ||
3784               (n->in(0)->Opcode() == Op_IfFalse &&
3785                (1.0 - iff->as_If()->_prob) >= 0.01) ||

3811 // First pass computes the earliest controlling node possible.  This is the
3812 // controlling input with the deepest dominating depth.
3813 void PhaseIdealLoop::build_loop_early( VectorSet &visited, Node_List &worklist, Node_Stack &nstack ) {
3814   while (worklist.size() != 0) {
3815     // Use local variables nstack_top_n & nstack_top_i to cache values
3816     // on nstack's top.
3817     Node *nstack_top_n = worklist.pop();
3818     uint  nstack_top_i = 0;
3819 //while_nstack_nonempty:
3820     while (true) {
3821       // Get parent node and next input's index from stack's top.
3822       Node  *n = nstack_top_n;
3823       uint   i = nstack_top_i;
3824       uint cnt = n->req(); // Count of inputs
3825       if (i == 0) {        // Pre-process the node.
3826         if( has_node(n) &&            // Have either loop or control already?
3827             !has_ctrl(n) ) {          // Have loop picked out already?
3828           // During "merge_many_backedges" we fold up several nested loops
3829           // into a single loop.  This makes the members of the original
3830           // loop bodies pointing to dead loops; they need to move up
3831           // to the new UNION'd larger loop.  I set the _head field of these
3832           // dead loops to NULL and the _parent field points to the owning
3833           // loop.  Shades of UNION-FIND algorithm.
3834           IdealLoopTree *ilt;
3835           while( !(ilt = get_loop(n))->_head ) {
3836             // Normally I would use a set_loop here.  But in this one special
3837             // case, it is legal (and expected) to change what loop a Node
3838             // belongs to.
3839             _nodes.map(n->_idx, (Node*)(ilt->_parent) );
3840           }
3841           // Remove safepoints ONLY if I've already seen I don't need one.
3842           // (the old code here would yank a 2nd safepoint after seeing a
3843           // first one, even though the 1st did not dominate in the loop body
3844           // and thus could be avoided indefinitely)
3845           if( !_verify_only && !_verify_me && ilt->_has_sfpt && n->Opcode() == Op_SafePoint &&
3846               is_deleteable_safept(n)) {

3849             if (ilt->_safepts != NULL) {

3851             }

3853             // only the control input

3855               worklist.push(in);      // Visit this guy later, using worklist
3856             }

3860             //   has_node(n) == true (see the condition above).
3861             i = cnt + 1;
3862           }
3863         }
3864       } // if (i == 0)
3865 
3866       // Visit all inputs
3867       bool done = true;       // Assume all n's inputs will be processed
3868       while (i < cnt) {
3869         Node *in = n->in(i);
3870         ++i;
3871         if (in == NULL) continue;
3872         if (in->pinned() && !in->is_CFG())
3873           set_ctrl(in, in->in(0));
3874         int is_visited = visited.test_set( in->_idx );
3875         if (!has_node(in)) {  // No controlling input yet?
3876           assert( !in->is_CFG(), "CFG Node with no controlling input?" );
3877           assert( !is_visited, "visit only once" );
3878           nstack.push(n, i);  // Save parent node and next input's index.
3879           nstack_top_n = in;  // Process current input now.
3880           nstack_top_i = 0;
3881           done = false;       // Not all n's inputs processed.
3882           break; // continue while_nstack_nonempty;
3883         } else if (!is_visited) {
3884           // This guy has a location picked out for him, but has not yet
3885           // been visited.  Happens to all CFG nodes, for instance.
3886           // Visit him using the worklist instead of recursion, to break
3887           // cycles.  Since he has a location already we do not need to
3888           // find his location before proceeding with the current Node.
3889           worklist.push(in);  // Visit this guy later, using worklist
3890         }
3891       }
3892       if (done) {
3893         // All of n's inputs have been processed, complete post-processing.
3894 
3895         // Compute earliest point this Node can go.
3896         // CFG, Phi, pinned nodes already know their controlling input.
3897         if (!has_node(n)) {
3898           // Record earliest legal location
3899           set_early_ctrl( n );
3900         }
3901         if (nstack.is_empty()) {
3902           // Finished all nodes on stack.
3903           // Process next node on the worklist.
3904           break;
3905         }
3906         // Get saved parent node and next input's index.
3907         nstack_top_n = nstack.node();
3908         nstack_top_i = nstack.index();
3909         nstack.pop();
3910       }
3911     } // while (true)
3912   }

3915 //------------------------------dom_lca_internal--------------------------------
3916 // Pair-wise LCA
3917 Node *PhaseIdealLoop::dom_lca_internal( Node *n1, Node *n2 ) const {
3918   if( !n1 ) return n2;          // Handle NULL original LCA
3919   assert( n1->is_CFG(), "" );
3920   assert( n2->is_CFG(), "" );
3921   // find LCA of all uses
3922   uint d1 = dom_depth(n1);
3923   uint d2 = dom_depth(n2);
3924   while (n1 != n2) {
3925     if (d1 > d2) {
3926       n1 =      idom(n1);
3927       d1 = dom_depth(n1);
3928     } else if (d1 < d2) {
3929       n2 =      idom(n2);
3930       d2 = dom_depth(n2);
3931     } else {
3932       // Here d1 == d2.  Due to edits of the dominator-tree, sections
3933       // of the tree might have the same depth.  These sections have
3934       // to be searched more carefully.
3935 
3936       // Scan up all the n1's with equal depth, looking for n2.
3937       Node *t1 = idom(n1);
3938       while (dom_depth(t1) == d1) {
3939         if (t1 == n2)  return n2;
3940         t1 = idom(t1);
3941       }
3942       // Scan up all the n2's with equal depth, looking for n1.
3943       Node *t2 = idom(n2);
3944       while (dom_depth(t2) == d2) {
3945         if (t2 == n1)  return n1;
3946         t2 = idom(t2);
3947       }
3948       // Move up to a new dominator-depth value as well as up the dom-tree.
3949       n1 = t1;
3950       n2 = t2;
3951       d1 = dom_depth(n1);
3952       d2 = dom_depth(n2);
3953     }
3954   }
3955   return n1;

4143 //------------------------------dom_lca_for_get_late_ctrl_internal-------------
4144 // Pair-wise LCA with tags.
4145 // Tag each index with the node 'tag' currently being processed
4146 // before advancing up the dominator chain using idom().
4147 // Later calls that find a match to 'tag' know that this path has already
4148 // been considered in the current LCA (which is input 'n1' by convention).
4149 // Since get_late_ctrl() is only called once for each node, the tag array
4150 // does not need to be cleared between calls to get_late_ctrl().
4151 // Algorithm trades a larger constant factor for better asymptotic behavior
4152 //
4153 Node *PhaseIdealLoop::dom_lca_for_get_late_ctrl_internal( Node *n1, Node *n2, Node *tag ) {
4154   uint d1 = dom_depth(n1);
4155   uint d2 = dom_depth(n2);
4156 
4157   do {
4158     if (d1 > d2) {
4159       // current lca is deeper than n2
4160       _dom_lca_tags.map(n1->_idx, tag);
4161       n1 =      idom(n1);
4162       d1 = dom_depth(n1);
4163     } else if (d1 < d2) {
4164       // n2 is deeper than current lca
4165       Node *memo = _dom_lca_tags[n2->_idx];
4166       if( memo == tag ) {
4167         return n1;    // Return the current LCA
4168       }
4169       _dom_lca_tags.map(n2->_idx, tag);
4170       n2 =      idom(n2);
4171       d2 = dom_depth(n2);
4172     } else {
4173       // Here d1 == d2.  Due to edits of the dominator-tree, sections
4174       // of the tree might have the same depth.  These sections have
4175       // to be searched more carefully.
4176 
4177       // Scan up all the n1's with equal depth, looking for n2.
4178       _dom_lca_tags.map(n1->_idx, tag);
4179       Node *t1 = idom(n1);
4180       while (dom_depth(t1) == d1) {
4181         if (t1 == n2)  return n2;
4182         _dom_lca_tags.map(t1->_idx, tag);
4183         t1 = idom(t1);
4184       }
4185       // Scan up all the n2's with equal depth, looking for n1.
4186       _dom_lca_tags.map(n2->_idx, tag);
4187       Node *t2 = idom(n2);
4188       while (dom_depth(t2) == d2) {
4189         if (t2 == n1)  return n1;
4190         _dom_lca_tags.map(t2->_idx, tag);
4191         t2 = idom(t2);
4192       }
4193       // Move up to a new dominator-depth value as well as up the dom-tree.
4194       n1 = t1;
4195       n2 = t2;
4196       d1 = dom_depth(n1);
4197       d2 = dom_depth(n2);

4232 //------------------------------build_loop_late--------------------------------
4233 // Put Data nodes into some loop nest, by setting the _nodes[]->loop mapping.
4234 // Second pass finds latest legal placement, and ideal loop placement.
4235 void PhaseIdealLoop::build_loop_late( VectorSet &visited, Node_List &worklist, Node_Stack &nstack ) {
4236   while (worklist.size() != 0) {
4237     Node *n = worklist.pop();
4238     // Only visit once
4239     if (visited.test_set(n->_idx)) continue;
4240     uint cnt = n->outcnt();
4241     uint   i = 0;
4242     while (true) {
4243       assert( _nodes[n->_idx], "no dead nodes" );
4244       // Visit all children
4245       if (i < cnt) {
4246         Node* use = n->raw_out(i);
4247         ++i;
4248         // Check for dead uses.  Aggressively prune such junk.  It might be
4249         // dead in the global sense, but still have local uses so I cannot
4250         // easily call 'remove_dead_node'.
4251         if( _nodes[use->_idx] != NULL || use->is_top() ) { // Not dead?
4252           // Due to cycles, we might not hit the same fixed point in the verify
4253           // pass as we do in the regular pass.  Instead, visit such phis as
4254           // simple uses of the loop head.
4255           if( use->in(0) && (use->is_CFG() || use->is_Phi()) ) {
4256             if( !visited.test(use->_idx) )
4257               worklist.push(use);
4258           } else if( !visited.test_set(use->_idx) ) {
4259             nstack.push(n, i); // Save parent and next use's index.
4260             n   = use;         // Process all children of current use.
4261             cnt = use->outcnt();
4262             i   = 0;
4263           }
4264         } else {
4265           // Do not visit around the backedge of loops via data edges.
4266           // push dead code onto a worklist
4267           _deadlist.push(use);
4268         }
4269       } else {
4270         // All of n's children have been processed, complete post-processing.
4271         build_loop_late_post(n);
4272         if (nstack.is_empty()) {
4273           // Finished all nodes on stack.
4274           // Process next node on the worklist.
4275           break;
4276         }
4277         // Get saved parent node and next use's index. Visit the rest of uses.
4278         n   = nstack.node();
4279         cnt = n->outcnt();
4280         i   = nstack.index();
4281         nstack.pop();
4282       }
4283     }
4284   }
4285 }

4331     _igvn._worklist.push(n);  // Maybe we'll normalize it, if no more loops.
4332   }
4333 
4334 #ifdef ASSERT
4335   if (_verify_only && !n->is_CFG()) {
4336     // Check def-use domination.
4337     compute_lca_of_uses(n, get_ctrl(n), true /* verify */);
4338   }
4339 #endif
4340 
4341   // CFG and pinned nodes already handled
4342   if( n->in(0) ) {
4343     if( n->in(0)->is_top() ) return; // Dead?
4344 
4345     // We'd like +VerifyLoopOptimizations to not believe that Mod's/Loads
4346     // _must_ be pinned (they have to observe their control edge of course).
4347     // Unlike Stores (which modify an unallocable resource, the memory
4348     // state), Mods/Loads can float around.  So free them up.
4349     switch( n->Opcode() ) {
4350     case Op_DivI:
4351     case Op_DivF:
4352     case Op_DivD:
4353     case Op_ModI:
4354     case Op_ModF:
4355     case Op_ModD:
4356     case Op_LoadB:              // Same with Loads; they can sink
4357     case Op_LoadUB:             // during loop optimizations.
4358     case Op_LoadUS:
4359     case Op_LoadD:
4360     case Op_LoadF:
4361     case Op_LoadI:
4362     case Op_LoadKlass:
4363     case Op_LoadNKlass:
4364     case Op_LoadL:
4365     case Op_LoadS:
4366     case Op_LoadP:
4367     case Op_LoadN:
4368     case Op_LoadRange:
4369     case Op_LoadD_unaligned:
4370     case Op_LoadL_unaligned:
4371     case Op_StrComp:            // Does a bunch of load-like effects
4372     case Op_StrEquals:
4373     case Op_StrIndexOf:
4374     case Op_StrIndexOfChar:
4375     case Op_AryEq:
4376     case Op_HasNegatives:
4377       pinned = false;
4378     }
4379     if (n->is_CMove()) {
4380       pinned = false;
4381     }
4382     if( pinned ) {
4383       IdealLoopTree *chosen_loop = get_loop(n->is_CFG() ? n : get_ctrl(n));
4384       if( !chosen_loop->_child )       // Inner loop?
4385         chosen_loop->_body.push(n); // Collect inner loops
4386       return;
4387     }
4388   } else {                      // No slot zero
4389     if( n->is_CFG() ) {         // CFG with no slot 0 is dead
4390       _nodes.map(n->_idx,0);    // No block setting, it's globally dead
4391       return;
4392     }
4393     assert(!n->is_CFG() || n->outcnt() == 0, "");
4394   }
4395 
4396   // Do I have a "safe range" I can select over?
4397   Node *early = get_ctrl(n);// Early location already computed
4398 
4399   // Compute latest point this Node can go
4400   Node *LCA = get_late_ctrl( n, early );
4401   // LCA is NULL due to uses being dead
4402   if( LCA == NULL ) {
4403 #ifdef ASSERT
4404     for (DUIterator i1 = n->outs(); n->has_out(i1); i1++) {
4405       assert( _nodes[n->out(i1)->_idx] == NULL, "all uses must also be dead");
4406     }
4407 #endif
4408     _nodes.map(n->_idx, 0);     // This node is useless
4409     _deadlist.push(n);
4410     return;
4411   }
4412   assert(LCA != NULL && !LCA->is_top(), "no dead nodes");
4413 
4414   Node *legal = LCA;            // Walk 'legal' up the IDOM chain
4415   Node *least = legal;          // Best legal position so far
4416   while( early != legal ) {     // While not at earliest legal
4417 #ifdef ASSERT
4418     if (legal->is_Start() && !early->is_Root()) {
4419       // Bad graph. Print idom path and fail.
4420       dump_bad_graph("Bad graph detected in build_loop_late", n, early, LCA);
4421       assert(false, "Bad graph detected in build_loop_late");
4422     }
4423 #endif
4424     // Find least loop nesting depth
4425     legal = idom(legal);        // Bump up the IDOM tree
4426     // Check for lower nesting depth
4427     if( get_loop(legal)->_nest < get_loop(least)->_nest )
4428       least = legal;
4429   }
4430   assert(early == legal || legal != C->root(), "bad dominance of inputs");
4431 
4432   // Try not to place code on a loop entry projection
4433   // which can inhibit range check elimination.
4434   if (least != early) {
4435     Node* ctrl_out = least->unique_ctrl_out();
4436     if (ctrl_out && ctrl_out->is_Loop() &&
4437         least == ctrl_out->in(LoopNode::EntryControl)) {
4438       // Move the node above predicates as far up as possible so a
4439       // following pass of loop predication doesn't hoist a predicate
4440       // that depends on it above that node.
4441       Node* new_ctrl = least;
4442       for (;;) {
4443         if (!new_ctrl->is_Proj()) {
4444           break;
4445         }
4446         CallStaticJavaNode* call = new_ctrl->as_Proj()->is_uncommon_trap_if_pattern(Deoptimization::Reason_none);
4447         if (call == NULL) {
4448           break;
4449         }
4450         int req = call->uncommon_trap_request();
4451         Deoptimization::DeoptReason trap_reason = Deoptimization::trap_request_reason(req);
4452         if (trap_reason != Deoptimization::Reason_loop_limit_check &&
4453             trap_reason != Deoptimization::Reason_predicate &&
4454             trap_reason != Deoptimization::Reason_profile_predicate) {
4455           break;
4456         }
4457         Node* c = new_ctrl->in(0)->in(0);
4458         if (is_dominator(c, early) && c != early) {
4459           break;
4460         }
4461         new_ctrl = c;
4462       }
4463       least = new_ctrl;
4464     }
4465   }
4466 
4467 #ifdef ASSERT
4468   // If verifying, verify that 'verify_me' has a legal location
4469   // and choose it as our location.
4470   if( _verify_me ) {
4471     Node *v_ctrl = _verify_me->get_ctrl_no_update(n);
4472     Node *legal = LCA;
4473     while( early != legal ) {   // While not at earliest legal
4474       if( legal == v_ctrl ) break;  // Check for prior good location
4475       legal = idom(legal)      ;// Bump up the IDOM tree
4476     }
4477     // Check for prior good location
4478     if( legal == v_ctrl ) least = legal; // Keep prior if found
4479   }
4480 #endif
4481 
4482   // Assign discovered "here or above" point
4483   least = find_non_split_ctrl(least);
4484   verify_strip_mined_scheduling(n, least);
4485   set_ctrl(n, least);
4486 
4487   // Collect inner loop bodies
4488   IdealLoopTree *chosen_loop = get_loop(least);
4489   if( !chosen_loop->_child )   // Inner loop?
4490     chosen_loop->_body.push(n);// Collect inner loops
4491 }

4646 // Collect a R-P-O for the whole CFG.
4647 // Result list is in post-order (scan backwards for RPO)
4648 void PhaseIdealLoop::rpo(Node* start, Node_Stack &stk, VectorSet &visited, Node_List &rpo_list) const {
4649   stk.push(start, 0);
4650   visited.set(start->_idx);
4651 
4652   while (stk.is_nonempty()) {
4653     Node* m   = stk.node();
4654     uint  idx = stk.index();
4655     if (idx < m->outcnt()) {
4656       stk.set_index(idx + 1);
4657       Node* n = m->raw_out(idx);
4658       if (n->is_CFG() && !visited.test_set(n->_idx)) {
4659         stk.push(n, 0);
4660       }
4661     } else {
4662       rpo_list.push(m);
4663       stk.pop();
4664     }
4665   }

4681       _curnt = _curnt->_parent;
