2228 WB_ENTRY(void, WB_CheckThreadObjOfTerminatingThread(JNIEnv* env, jobject wb, jobject target_handle))
2229   oop target_oop = JNIHandles::resolve_non_null(target_handle);
2230   jlong tid = java_lang_Thread::thread_id(target_oop);
2231   JavaThread* target = java_lang_Thread::thread(target_oop);
2232 
2233   // Grab a ThreadsListHandle to protect the target thread whilst terminating
2234   ThreadsListHandle tlh;
2235 
2236   // Look up the target thread by tid to ensure it is present
2237   JavaThread* t = tlh.list()->find_JavaThread_from_java_tid(tid);
2238   if (t == NULL) {
2239     THROW_MSG(vmSymbols::java_lang_RuntimeException(), "Target thread not found in ThreadsList!");
2240   }
2241 
2242   tty->print_cr("WB_CheckThreadObjOfTerminatingThread: target thread is protected");
2243   // Allow target to terminate by boosting priority
2244   java_lang_Thread::set_priority(t->threadObj(), ThreadPriority(NormPriority + 1));
2245 
2246   // Now wait for the target to terminate
2247   while (!target->is_terminated()) {
2248     ThreadBlockInVM tbivm(thread);  // just in case target is involved in a safepoint
2249     os::naked_short_sleep(0);
2250   }
2251 
2252   tty->print_cr("WB_CheckThreadObjOfTerminatingThread: target thread is terminated");
2253 
2254   // Now release the GC inducing thread - we have to re-resolve the external oop that
2255   // was passed in as GC may have occurred and we don't know if we can trust t->threadObj() now.
2256   oop original = JNIHandles::resolve_non_null(target_handle);
2257   java_lang_Thread::set_priority(original, ThreadPriority(NormPriority + 2));
2258 
2259   tty->print_cr("WB_CheckThreadObjOfTerminatingThread: GC has been initiated - checking threadObj:");
2260 
2261   // The Java code should be creating garbage and triggering GC, which would potentially move
2262   // the threadObj oop. If the exiting thread is properly protected then its threadObj should
2263   // remain valid and equal to our initial target_handle. Loop a few times to give GC a chance to
2264   // kick in.
2265   for (int i = 0; i < 5; i++) {
2266     oop original = JNIHandles::resolve_non_null(target_handle);
2267     oop current = t->threadObj();
2268     if (original != current) {
2269       tty->print_cr("WB_CheckThreadObjOfTerminatingThread: failed comparison on iteration %d", i);
2270       THROW_MSG(vmSymbols::java_lang_RuntimeException(), "Target thread oop has changed!");
2271     } else {
2272       tty->print_cr("WB_CheckThreadObjOfTerminatingThread: successful comparison on iteration %d", i);
2273       ThreadBlockInVM tbivm(thread);
2274       os::naked_short_sleep(50);
2275     }
2276   }
2277 WB_END
