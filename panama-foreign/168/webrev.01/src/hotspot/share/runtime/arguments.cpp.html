 119 bool PathString::set_value(const char *value) {
 120   if (_value != NULL) {
 121     FreeHeap(_value);
 122   }
 123   _value = AllocateHeap(strlen(value)+1, mtArguments);
 124   assert(_value != NULL, "Unable to allocate space for new path value");
 125   if (_value != NULL) {
 126     strcpy(_value, value);
 127   } else {
 128     // not able to allocate
 129     return false;
 130   }
 131   return true;
 132 }
 133 
 134 void PathString::append_value(const char *value) {
 135   char *sp;
 136   size_t len = 0;
 137   if (value != NULL) {
 138     len = strlen(value);
 139     if (_value != NULL) {
 140       len += strlen(_value);
 141     }
 142     sp = AllocateHeap(len+2, mtArguments);
 143     assert(sp != NULL, "Unable to allocate space for new append path value");
 144     if (sp != NULL) {
 145       if (_value != NULL) {
 146         strcpy(sp, _value);
 147         strcat(sp, os::path_separator());
 148         strcat(sp, value);
 149         FreeHeap(_value);
 150       } else {
 151         strcpy(sp, value);
 152       }
 153       _value = sp;
 154     }
 155   }
 156 }

 224 // Check if head of 'option' matches 'name', and sets 'tail' to the remaining
 225 // part of the option string.
 226 static bool match_option(const JavaVMOption *option, const char* name,
 227                          const char** tail) {
 228   size_t len = strlen(name);
 229   if (strncmp(option->optionString, name, len) == 0) {
 230     *tail = option->optionString + len;
 231     return true;
 232   } else {
 233     return false;
 234   }
 235 }
 236 
 237 // Check if 'option' matches 'name'. No "tail" is allowed.
 238 static bool match_option(const JavaVMOption *option, const char* name) {
 239   const char* tail = NULL;
 240   bool result = match_option(option, name, &tail);
 241   if (tail != NULL && *tail == '\0') {
 242     return result;
 243   } else {
 244     return false;
 245   }
 246 }
 247 
 248 // Return true if any of the strings in null-terminated array 'names' matches.
 249 // If tail_allowed is true, then the tail must begin with a colon; otherwise,
 250 // the option must match exactly.
 251 static bool match_option(const JavaVMOption* option, const char** names, const char** tail,
 252   bool tail_allowed) {
 253   for (/* empty */; *names != NULL; ++names) {
 254   if (match_option(option, *names, tail)) {
 255       if (**tail == '\0' || (tail_allowed && **tail == ':')) {
 256         return true;
 257       }
 258     }
 259   }
 260   return false;
 261 }

 329 // Return TRUE if option matches 'property', or 'property=', or 'property.'.
 330 static bool matches_property_suffix(const char* option, const char* property, size_t len) {
 331   return ((strncmp(option, property, len) == 0) &&
 332           (option[len] == '=' || option[len] == '.' || option[len] == '\0'));
 333 }

 335 // Return true if property starts with "jdk.module." and its ensuing chars match
 336 // any of the reserved module properties.
 337 // property should be passed without the leading "-D".
 338 bool Arguments::is_internal_module_property(const char* property) {
 339   assert((strncmp(property, "-D", 2) != 0), "Unexpected leading -D");
 340   if  (strncmp(property, MODULE_PROPERTY_PREFIX, MODULE_PROPERTY_PREFIX_LEN) == 0) {
 341     const char* property_suffix = property + MODULE_PROPERTY_PREFIX_LEN;
 342     if (matches_property_suffix(property_suffix, ADDEXPORTS, ADDEXPORTS_LEN) ||
 343         matches_property_suffix(property_suffix, ADDREADS, ADDREADS_LEN) ||
 344         matches_property_suffix(property_suffix, ADDOPENS, ADDOPENS_LEN) ||
 345         matches_property_suffix(property_suffix, PATCH, PATCH_LEN) ||
 346         matches_property_suffix(property_suffix, ADDMODS, ADDMODS_LEN) ||
 347         matches_property_suffix(property_suffix, LIMITMODS, LIMITMODS_LEN) ||
 348         matches_property_suffix(property_suffix, PATH, PATH_LEN) ||
 349         matches_property_suffix(property_suffix, UPGRADE_PATH, UPGRADE_PATH_LEN)) {
 350       return true;
 351     }
 352   }
 353   return false;
 354 }

 356 // Process java launcher properties.
 357 void Arguments::process_sun_java_launcher_properties(JavaVMInitArgs* args) {
 358   // See if sun.java.launcher or sun.java.launcher.is_altjvm is defined.
 359   // Must do this before setting up other system properties,
 360   // as some of them may depend on launcher type.
 361   for (int index = 0; index < args->nOptions; index++) {
 362     const JavaVMOption* option = args->options + index;
 363     const char* tail;
 364 
 365     if (match_option(option, "-Dsun.java.launcher=", &tail)) {
 366       process_java_launcher_argument(tail, option->extraInfo);
 367       continue;
 368     }
 369     if (match_option(option, "-Dsun.java.launcher.is_altjvm=", &tail)) {
 370       if (strcmp(tail, "true") == 0) {
 371         _sun_java_launcher_is_altjvm = true;
 372       }
 373       continue;
 374     }
 375   }
 376 }

 380 
 381   // Set up _system_boot_class_path which is not a property but
 382   // relies heavily on argument processing and the jdk.boot.class.path.append

 390   PropertyList_add(&_system_properties, new SystemProperty("jdk.debug", VM_Version::jdk_debug_level(),  false));
 391 
 392   // Initialize the vm.info now, but it will need updating after argument parsing.
 393   _vm_info = new SystemProperty("java.vm.info", VM_Version::vm_info_string(), true);
 394 
 395   // Following are JVMTI agent writable properties.
 396   // Properties values are set to NULL and they are
 397   // os specific they are initialized in os::init_system_properties_values().
 398   _sun_boot_library_path = new SystemProperty("sun.boot.library.path", NULL,  true);
 399   _java_library_path = new SystemProperty("java.library.path", NULL,  true);
 400   _java_home =  new SystemProperty("java.home", NULL,  true);
 401   _java_class_path = new SystemProperty("java.class.path", "",  true);
 402   // jdk.boot.class.path.append is a non-writeable, internal property.
 403   // It can only be set by either:
 404   //    - -Xbootclasspath/a:
 405   //    - AddToBootstrapClassLoaderSearch during JVMTI OnLoad phase
 406   _jdk_boot_class_path_append = new SystemProperty("jdk.boot.class.path.append", "", false, true);

 408   // Add to System Property list.
 409   PropertyList_add(&_system_properties, _sun_boot_library_path);
 410   PropertyList_add(&_system_properties, _java_library_path);
 411   PropertyList_add(&_system_properties, _java_home);
 412   PropertyList_add(&_system_properties, _java_class_path);
 413   PropertyList_add(&_system_properties, _jdk_boot_class_path_append);
 414   PropertyList_add(&_system_properties, _vm_info);
 415 
 416   // Set OS specific system properties values
 417   os::init_system_properties_values();
 418 }

 437 /*
 438  *  -XX argument processing:
 439  *
 440  *  -XX arguments are defined in several places, such as:
 441  *      globals.hpp, globals_<cpu>.hpp, globals_<os>.hpp, <compiler>_globals.hpp, or <gc>_globals.hpp.
 442  *  -XX arguments are parsed in parse_argument().
 443  *  -XX argument bounds checking is done in check_vm_args_consistency().
 444  *
 445  * Over time -XX arguments may change. There are mechanisms to handle common cases:
 446  *
 447  *      ALIASED: An option that is simply another name for another option. This is often
 448  *               part of the process of deprecating a flag, but not all aliases need
 449  *               to be deprecated.
 450  *
 451  *               Create an alias for an option by adding the old and new option names to the
 452  *               "aliased_jvm_flags" table. Delete the old variable from globals.hpp (etc).
 453  *
 454  *   DEPRECATED: An option that is supported, but a warning is printed to let the user know that
 455  *               support may be removed in the future. Both regular and aliased options may be
 456  *               deprecated.
 457  *
 458  *               Add a deprecation warning for an option (or alias) by adding an entry in the
 459  *               "special_jvm_flags" table and setting the "deprecated_in" field.
 460  *               Often an option "deprecated" in one major release will
 461  *               be made "obsolete" in the next. In this case the entry should also have its
 462  *               "obsolete_in" field set.
 463  *
 464  *     OBSOLETE: An option that has been removed (and deleted from globals.hpp), but is still accepted
 465  *               on the command line. A warning is printed to let the user know that option might not
 466  *               be accepted in the future.
 467  *
 468  *               Add an obsolete warning for an option by adding an entry in the "special_jvm_flags"
 469  *               table and setting the "obsolete_in" field.
 470  *
 471  *      EXPIRED: A deprecated or obsolete option that has an "accept_until" version less than or equal
 472  *               to the current JDK version. The system will flatly refuse to admit the existence of
 473  *               the flag. This allows a flag to die automatically over JDK releases.
 474  *
 475  *               Note that manual cleanup of expired options should be done at major JDK version upgrades:
 476  *                  - Newly expired options should be removed from the special_jvm_flags and aliased_jvm_flags tables.
 477  *                  - Newly obsolete or expired deprecated options should have their global variable
 478  *                    definitions removed (from globals.hpp, etc) and related implementations removed.
 479  *
 480  * Recommended approach for removing options:
 481  *
 482  * To remove options commonly used by customers (e.g. product -XX options), use
 483  * the 3-step model adding major release numbers to the deprecate, obsolete and expire columns.
 484  *
 485  * To remove internal options (e.g. diagnostic, experimental, develop options), use
 486  * a 2-step model adding major release numbers to the obsolete and expire columns.
 487  *
 488  * To change the name of an option, use the alias table as well as a 2-step
 489  * model adding major release numbers to the deprecate and expire columns.
 490  * Think twice about aliasing commonly used customer options.
 491  *
 492  * There are times when it is appropriate to leave a future release number as undefined.
 493  *
 494  * Tests:  Aliases should be tested in VMAliasOptions.java.
 495  *         Deprecated options should be tested in VMDeprecatedOptions.java.
 496  */

 722 // Verifies the correctness of the entries in the special_jvm_flags table.
 723 // If there is a semantic error (i.e. a bug in the table) such as the obsoletion
 724 // version being earlier than the deprecation version, then a warning is issued
 725 // and verification fails - by returning false. If it is detected that the table
 726 // is out of date, with respect to the current version, then ideally a warning is
 727 // issued but verification does not fail. This allows the VM to operate when the
 728 // version is first updated, without needing to update all the impacted flags at
 729 // the same time. In practice we can't issue the warning immediately when the version
 730 // is updated as it occurs for every test and some tests are not prepared to handle
 731 // unexpected output - see 8196739. Instead we only check if the table is up-to-date
 732 // if the check_globals flag is true, and in addition allow a grace period and only
 733 // check for stale flags when we hit build 25 (which is far enough into the 6 month
 734 // release cycle that all flag updates should have been processed, whilst still
 735 // leaving time to make the change before RDP2).
 736 // We use a gtest to call this, passing true, so that we can detect stale flags before
 737 // the end of the release cycle.

 741 bool Arguments::verify_special_jvm_flags(bool check_globals) {
 742   bool success = true;
 743   for (size_t i = 0; special_jvm_flags[i].name != NULL; i++) {
 744     const SpecialFlag& flag = special_jvm_flags[i];
 745     if (lookup_special_flag(flag.name, i)) {
 746       warning("Duplicate special flag declaration \"%s\"", flag.name);
 747       success = false;
 748     }
 749     if (flag.deprecated_in.is_undefined() &&
 750         flag.obsolete_in.is_undefined()) {
 751       warning("Special flag entry \"%s\" must declare version deprecated and/or obsoleted in.", flag.name);
 752       success = false;
 753     }
 754 
 755     if (!flag.deprecated_in.is_undefined()) {
 756       if (!version_less_than(flag.deprecated_in, flag.obsolete_in)) {
 757         warning("Special flag entry \"%s\" must be deprecated before obsoleted.", flag.name);
 758         success = false;
 759       }
 760 
 761       if (!version_less_than(flag.deprecated_in, flag.expired_in)) {
 762         warning("Special flag entry \"%s\" must be deprecated before expired.", flag.name);
 763         success = false;
 764       }
 765     }
 766 
 767     if (!flag.obsolete_in.is_undefined()) {
 768       if (!version_less_than(flag.obsolete_in, flag.expired_in)) {
 769         warning("Special flag entry \"%s\" must be obsoleted before expired.", flag.name);
 770         success = false;
 771       }
 772 
 773       // if flag has become obsolete it should not have a "globals" flag defined anymore.
 774       if (check_globals && VM_Version::vm_build_number() >= SPECIAL_FLAG_VALIDATION_BUILD &&
 775           !version_less_than(JDK_Version::current(), flag.obsolete_in)) {
 776         if (JVMFlag::find_declared_flag(flag.name) != NULL) {
 777           warning("Global variable for obsolete special flag entry \"%s\" should be removed", flag.name);
 778           success = false;
 779         }
 780       }
 781 
 782     } else if (!flag.expired_in.is_undefined()) {
 783       warning("Special flag entry \"%s\" must be explicitly obsoleted before expired.", flag.name);
 784       success = false;
 785     }
 786 
 787     if (!flag.expired_in.is_undefined()) {
 788       // if flag has become expired it should not have a "globals" flag defined anymore.
 789       if (check_globals && VM_Version::vm_build_number() >= SPECIAL_FLAG_VALIDATION_BUILD &&
 790           !version_less_than(JDK_Version::current(), flag.expired_in)) {
 791         if (JVMFlag::find_declared_flag(flag.name) != NULL) {
 792           warning("Global variable for expired flag entry \"%s\" should be removed", flag.name);
 793           success = false;
 794         }
 795       }
 796     }
 797   }
 798   return success;
 799 }

 802 // Parses a size specification string.
 803 bool Arguments::atojulong(const char *s, julong* result) {
 804   julong n = 0;
 805 
 806   // First char must be a digit. Don't allow negative numbers or leading spaces.
 807   if (!isdigit(*s)) {
 808     return false;
 809   }
 810 
 811   bool is_hex = (s[0] == '0' && (s[1] == 'x' || s[1] == 'X'));
 812   char* remainder;
 813   errno = 0;
 814   n = strtoull(s, &remainder, (is_hex ? 16 : 10));
 815   if (errno != 0) {
 816     return false;
 817   }
 818 
 819   // Fail if no number was read at all or if the remainder contains more than a single non-digit character.
 820   if (remainder == s || strlen(remainder) > 1) {
 821     return false;
 822   }
 823 
 824   switch (*remainder) {
 825     case 'T': case 't':
 826       *result = n * G * K;
 827       // Check for overflow.
 828       if (*result/((julong)G * K) != n) return false;
 829       return true;
 830     case 'G': case 'g':
 831       *result = n * G;
 832       if (*result/G != n) return false;
 833       return true;
 834     case 'M': case 'm':
 835       *result = n * M;
 836       if (*result/M != n) return false;
 837       return true;
 838     case 'K': case 'k':
 839       *result = n * K;
 840       if (*result/K != n) return false;
 841       return true;
 842     case '\0':
 843       *result = n;
 844       return true;
 845     default:
 846       return false;
 847   }
 848 }

 961   const char* value;
 962   char* free_this_too = NULL;
 963   if (old_len == 0) {
 964     value = new_value;
 965   } else if (new_len == 0) {
 966     value = old_value;
 967   } else {
 968      size_t length = old_len + 1 + new_len + 1;

 974   }

 976   // JVMFlag always returns a pointer that needs freeing.
 977   FREE_C_HEAP_ARRAY(char, value);
 978   // JVMFlag made its own copy, so I must delete my own temp. buffer.
 979   FREE_C_HEAP_ARRAY(char, free_this_too);
 980   return true;

1057 bool Arguments::parse_argument(const char* arg, JVMFlag::Flags origin) {
1058 
1059   // range of acceptable characters spelled out for portability reasons
1060 #define NAME_RANGE  "[abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789_]"
1061 #define BUFLEN 255
1062   char name[BUFLEN+1];
1063   char dummy;
1064   const char* real_name;
1065   bool warn_if_deprecated = true;
1066 
1067   if (sscanf(arg, "-%" XSTR(BUFLEN) NAME_RANGE "%c", name, &dummy) == 1) {
1068     AliasedLoggingFlag alf = catch_logging_aliases(name, false);
1069     if (alf.alias_name != NULL){
1070       LogConfiguration::configure_stdout(LogLevel::Off, alf.exactMatch, alf.tag0, alf.tag1, alf.tag2, alf.tag3, alf.tag4, alf.tag5);
1071       return true;
1072     }
1073     real_name = handle_aliases_and_deprecation(name, warn_if_deprecated);
1074     if (real_name == NULL) {
1075       return false;
1076     }
1077     JVMFlag* flag = JVMFlag::find_flag(real_name);
1078     return set_bool_flag(flag, false, origin);
1079   }
1080   if (sscanf(arg, "+%" XSTR(BUFLEN) NAME_RANGE "%c", name, &dummy) == 1) {
1081     AliasedLoggingFlag alf = catch_logging_aliases(name, true);
1082     if (alf.alias_name != NULL){
1083       LogConfiguration::configure_stdout(alf.level, alf.exactMatch, alf.tag0, alf.tag1, alf.tag2, alf.tag3, alf.tag4, alf.tag5);
1084       return true;
1085     }
1086     real_name = handle_aliases_and_deprecation(name, warn_if_deprecated);
1087     if (real_name == NULL) {
1088       return false;
1089     }
1090     JVMFlag* flag = JVMFlag::find_flag(real_name);
1091     return set_bool_flag(flag, true, origin);
1092   }
1093 
1094   char punct;
1095   if (sscanf(arg, "%" XSTR(BUFLEN) NAME_RANGE "%c", name, &punct) == 2 && punct == '=') {
1096     const char* value = strchr(arg, '=') + 1;
1097 
1098     // this scanf pattern matches both strings (handled here) and numbers (handled later))
1099     AliasedLoggingFlag alf = catch_logging_aliases(name, true);
1100     if (alf.alias_name != NULL) {
1101       LogConfiguration::configure_stdout(alf.level, alf.exactMatch, alf.tag0, alf.tag1, alf.tag2, alf.tag3, alf.tag4, alf.tag5);
1102       return true;
1103     }
1104     real_name = handle_aliases_and_deprecation(name, warn_if_deprecated);
1105     if (real_name == NULL) {
1106       return false;
1107     }
1108     JVMFlag* flag = JVMFlag::find_flag(real_name);
1109     if (flag != NULL && flag->is_ccstr()) {
1110       if (flag->ccstr_accumulates()) {
1111         return append_to_string_flag(flag, value, origin);
1112       } else {
1113         if (value[0] == '\0') {
1114           value = NULL;
1115         }
1116         return set_string_flag(flag, value, origin);
1117       }
1118     } else {
1119       warn_if_deprecated = false; // if arg is deprecated, we've already done warning...
1120     }
1121   }
1122 
1123   if (sscanf(arg, "%" XSTR(BUFLEN) NAME_RANGE ":%c", name, &punct) == 2 && punct == '=') {
1124     const char* value = strchr(arg, '=') + 1;
1125     // -XX:Foo:=xxx will reset the string flag to the given value.
1126     if (value[0] == '\0') {
1127       value = NULL;
1128     }
1129     real_name = handle_aliases_and_deprecation(name, warn_if_deprecated);
1130     if (real_name == NULL) {
1131       return false;
1132     }
1133     JVMFlag* flag = JVMFlag::find_flag(real_name);
1134     return set_string_flag(flag, value, origin);
1135   }

1137 #define SIGNED_FP_NUMBER_RANGE "[-0123456789.eE+]"
1138 #define SIGNED_NUMBER_RANGE    "[-0123456789]"
1139 #define        NUMBER_RANGE    "[0123456789eE+-]"
1140   char value[BUFLEN + 1];
1141   char value2[BUFLEN + 1];
1142   if (sscanf(arg, "%" XSTR(BUFLEN) NAME_RANGE "=" "%" XSTR(BUFLEN) SIGNED_NUMBER_RANGE "." "%" XSTR(BUFLEN) NUMBER_RANGE "%c", name, value, value2, &dummy) == 3) {
1143     // Looks like a floating-point number -- try again with more lenient format string
1144     if (sscanf(arg, "%" XSTR(BUFLEN) NAME_RANGE "=" "%" XSTR(BUFLEN) SIGNED_FP_NUMBER_RANGE "%c", name, value, &dummy) == 2) {
1145       real_name = handle_aliases_and_deprecation(name, warn_if_deprecated);
1146       if (real_name == NULL) {
1147         return false;
1148       }
1149       JVMFlag* flag = JVMFlag::find_flag(real_name);
1150       return set_fp_numeric_flag(flag, value, origin);
1151     }
1152   }
1153 
1154 #define VALUE_RANGE "[-kmgtxKMGTX0123456789abcdefABCDEF]"
1155   if (sscanf(arg, "%" XSTR(BUFLEN) NAME_RANGE "=" "%" XSTR(BUFLEN) VALUE_RANGE "%c", name, value, &dummy) == 2) {
1156     real_name = handle_aliases_and_deprecation(name, warn_if_deprecated);
1157     if (real_name == NULL) {
1158       return false;
1159     }
1160     JVMFlag* flag = JVMFlag::find_flag(real_name);
1161     return set_numeric_flag(flag, value, origin);
1162   }
1163 
1164   return false;
1165 }

1270 bool Arguments::process_argument(const char* arg,
1271                                  jboolean ignore_unrecognized,
1272                                  JVMFlag::Flags origin) {
1273   JDK_Version since = JDK_Version();
1274 
1275   if (parse_argument(arg, origin)) {
1276     return true;
1277   }
1278 
1279   // Determine if the flag has '+', '-', or '=' characters.
1280   bool has_plus_minus = (*arg == '+' || *arg == '-');
1281   const char* const argname = has_plus_minus ? arg + 1 : arg;
1282 
1283   size_t arg_len;
1284   const char* equal_sign = strchr(argname, '=');
1285   if (equal_sign == NULL) {
1286     arg_len = strlen(argname);
1287   } else {
1288     arg_len = equal_sign - argname;
1289   }
1290 
1291   // Only make the obsolete check for valid arguments.
1292   if (arg_len <= BUFLEN) {
1293     // Construct a string which consists only of the argument name without '+', '-', or '='.
1294     char stripped_argname[BUFLEN+1]; // +1 for '\0'
1295     jio_snprintf(stripped_argname, arg_len+1, "%s", argname); // +1 for '\0'
1296     if (is_obsolete_flag(stripped_argname, &since)) {
1297       char version[256];
1298       since.to_string(version, sizeof(version));
1299       warning("Ignoring option %s; support was removed in %s", stripped_argname, version);
1300       return true;
1301     }
1302 #ifndef PRODUCT
1303     else {
1304       const char* replacement;
1305       if ((replacement = removed_develop_logging_flag_name(stripped_argname)) != NULL){
1306         log_warning(arguments)("%s has been removed. Please use %s instead.",
1307                                stripped_argname,
1308                                replacement);
1309         return false;
1310       }
1311     }
1312 #endif //PRODUCT
1313   }
1314 
1315   // For locked flags, report a custom error message if available.
1316   // Otherwise, report the standard unrecognized VM option.
1317   const JVMFlag* found_flag = JVMFlag::find_declared_flag((const char*)argname, arg_len);
1318   if (found_flag != NULL) {
1319     char locked_message_buf[BUFLEN];
1320     JVMFlag::MsgType msg_type = found_flag->get_locked_message(locked_message_buf, BUFLEN);
1321     if (strlen(locked_message_buf) == 0) {
1322       if (found_flag->is_bool() && !has_plus_minus) {
1323         jio_fprintf(defaultStream::error_stream(),
1324           "Missing +/- setting for VM option '%s'\n", argname);
1325       } else if (!found_flag->is_bool() && has_plus_minus) {
1326         jio_fprintf(defaultStream::error_stream(),
1327           "Unexpected +/- setting in VM option '%s'\n", argname);
1328       } else {
1329         jio_fprintf(defaultStream::error_stream(),
1330           "Improperly specified VM option '%s'\n", argname);
1331       }
1332     } else {
1333 #ifdef PRODUCT
1334       bool mismatched = ((msg_type == JVMFlag::NOTPRODUCT_FLAG_BUT_PRODUCT_BUILD) ||
1335                          (msg_type == JVMFlag::DEVELOPER_FLAG_BUT_PRODUCT_BUILD));
1336       if (ignore_unrecognized && mismatched) {
1337         return true;
1338       }
1339 #endif
1340       jio_fprintf(defaultStream::error_stream(), "%s", locked_message_buf);
1341     }
1342   } else {
1343     if (ignore_unrecognized) {
1344       return true;
1345     }
1346     jio_fprintf(defaultStream::error_stream(),
1347                 "Unrecognized VM option '%s'\n", argname);
1348     JVMFlag* fuzzy_matched = JVMFlag::fuzzy_match((const char*)argname, arg_len, true);
1349     if (fuzzy_matched != NULL) {
1350       jio_fprintf(defaultStream::error_stream(),
1351                   "Did you mean '%s%s%s'? ",
1352                   (fuzzy_matched->is_bool()) ? "(+/-)" : "",
1353                   fuzzy_matched->_name,
1354                   (fuzzy_matched->is_bool()) ? "" : "=<value>");
1355     }
1356   }
1357 
1358   // allow for commandline "commenting out" options like -XX:#+Verbose
1359   return arg[0] == '#';
1360 }

1362 bool Arguments::process_settings_file(const char* file_name, bool should_exist, jboolean ignore_unrecognized) {
1363   FILE* stream = fopen(file_name, "rb");
1364   if (stream == NULL) {
1365     if (should_exist) {
1366       jio_fprintf(defaultStream::error_stream(),
1367                   "Could not open settings file %s\n", file_name);
1368       return false;
1369     } else {
1370       return true;
1371     }
1372   }
1373 
1374   char token[1024];
1375   int  pos = 0;
1376 
1377   bool in_white_space = true;
1378   bool in_comment     = false;
1379   bool in_quote       = false;
1380   char quote_c        = 0;
1381   bool result         = true;
1382 
1383   int c = getc(stream);
1384   while(c != EOF && pos < (int)(sizeof(token)-1)) {
1385     if (in_white_space) {
1386       if (in_comment) {
1387         if (c == '\n') in_comment = false;
1388       } else {
1389         if (c == '#') in_comment = true;
1390         else if (!isspace(c)) {
1391           in_white_space = false;
1392           token[pos++] = c;
1393         }
1394       }
1395     } else {
1396       if (c == '\n' || (!in_quote && isspace(c))) {
1397         // token ends at newline, or at unquoted whitespace
1398         // this allows a way to include spaces in string-valued options
1399         token[pos] = '\0';
1400         logOption(token);
1401         result &= process_argument(token, ignore_unrecognized, JVMFlag::CONFIG_FILE);
1402         build_jvm_flags(token);
1403         pos = 0;
1404         in_white_space = true;
1405         in_quote = false;
1406       } else if (!in_quote && (c == '\'' || c == '"')) {
1407         in_quote = true;
1408         quote_c = c;
1409       } else if (in_quote && (c == quote_c)) {
1410         in_quote = false;
1411       } else {
1412         token[pos++] = c;
1413       }
1414     }
1415     c = getc(stream);
1416   }
1417   if (pos > 0) {
1418     token[pos] = '\0';
1419     result &= process_argument(token, ignore_unrecognized, JVMFlag::CONFIG_FILE);
1420     build_jvm_flags(token);
1421   }
1422   fclose(stream);
1423   return result;
1424 }

1433 bool Arguments::add_property(const char* prop, PropertyWriteable writeable, PropertyInternal internal) {
1434   const char* eq = strchr(prop, '=');
1435   const char* key;
1436   const char* value = "";
1437 
1438   if (eq == NULL) {
1439     // property doesn't have a value, thus use passed string
1440     key = prop;
1441   } else {
1442     // property have a value, thus extract it and save to the
1443     // allocated string
1444     size_t key_len = eq - prop;
1445     char* tmp_key = AllocateHeap(key_len + 1, mtArguments);
1446 
1447     jio_snprintf(tmp_key, key_len + 1, "%s", prop);
1448     key = tmp_key;
1449 
1450     value = &prop[key_len + 1];
1451   }
1452 
1453   if (strcmp(key, "java.compiler") == 0) {
1454     process_java_compiler_argument(value);
1455     // Record value in Arguments, but let it get passed to Java.
1456   } else if (strcmp(key, "sun.java.launcher.is_altjvm") == 0) {
1457     // sun.java.launcher.is_altjvm property is
1458     // private and is processed in process_sun_java_launcher_properties();
1459     // the sun.java.launcher property is passed on to the java application
1460   } else if (strcmp(key, "sun.boot.library.path") == 0) {
1461     // append is true, writable is true, internal is false
1462     PropertyList_unique_add(&_system_properties, key, value, AppendProperty,
1463                             WriteableProperty, ExternalProperty);
1464   } else {
1465     if (strcmp(key, "sun.java.command") == 0) {
1466       char *old_java_command = _java_command;
1467       _java_command = os::strdup_check_oom(value, mtArguments);
1468       if (old_java_command != NULL) {
1469         os::free(old_java_command);
1470       }
1471     } else if (strcmp(key, "java.vendor.url.bug") == 0) {
1472       // If this property is set on the command line then its value will be
1473       // displayed in VM error logs as the URL at which to submit such logs.
1474       // Normally the URL displayed in error logs is different from the value
1475       // of this system property, so a different property should have been
1476       // used here, but we leave this as-is in case someone depends upon it.
1477       const char* old_java_vendor_url_bug = _java_vendor_url_bug;
1478       // save it in _java_vendor_url_bug, so JVM fatal error handler can access
1479       // its value without going through the property list or making a Java call.
1480       _java_vendor_url_bug = os::strdup_check_oom(value, mtArguments);
1481       if (old_java_vendor_url_bug != NULL) {
1482         os::free((void *)old_java_vendor_url_bug);
1483       }
1484     }
1485 
1486     // Create new property and add at the end of the list
1487     PropertyList_unique_add(&_system_properties, key, value, AddProperty, writeable, internal);
1488   }
1489 
1490   if (key != prop) {
1491     // SystemProperty copy passed value, thus free previously allocated
1492     // memory
1493     FreeHeap((void *)key);
1494   }
1495 
1496   return true;
1497 }

1647 size_t Arguments::max_heap_for_compressed_oops() {
1648   // Avoid sign flip.
1649   assert(OopEncodingHeapMax > (uint64_t)os::vm_page_size(), "Unusual page size");

1651   // keeping alignment constraints of the heap. To guarantee the latter, as the
1652   // NULL page is located before the heap, we pad the NULL page to the conservative
1653   // maximum alignment that the GC may ever impose upon the heap.

1655                                                   _conservative_max_heap_alignment);

1762 // Use static initialization to get the default before parsing
1763 static const size_t DefaultHeapBaseMinAddress = HeapBaseMinAddress;
1764 
1765 void Arguments::set_heap_size() {
1766   julong phys_mem;
1767 
1768   // If the user specified one of these options, they
1769   // want specific memory sizing so do not limit memory
1770   // based on compressed oops addressability.
1771   // Also, memory limits will be calculated based on
1772   // available os physical memory, not our MaxRAM limit,
1773   // unless MaxRAM is also specified.
1774   bool override_coop_limit = (!FLAG_IS_DEFAULT(MaxRAMPercentage) ||
1775                            !FLAG_IS_DEFAULT(MaxRAMFraction) ||
1776                            !FLAG_IS_DEFAULT(MinRAMPercentage) ||
1777                            !FLAG_IS_DEFAULT(MinRAMFraction) ||
1778                            !FLAG_IS_DEFAULT(InitialRAMPercentage) ||
1779                            !FLAG_IS_DEFAULT(InitialRAMFraction) ||
1780                            !FLAG_IS_DEFAULT(MaxRAM));
1781   if (override_coop_limit) {
1782     if (FLAG_IS_DEFAULT(MaxRAM)) {
1783       phys_mem = os::physical_memory();
1784       FLAG_SET_ERGO(MaxRAM, (uint64_t)phys_mem);
1785     } else {
1786       phys_mem = (julong)MaxRAM;
1787     }
1788   } else {
1789     phys_mem = FLAG_IS_DEFAULT(MaxRAM) ? MIN2(os::physical_memory(), (julong)MaxRAM)
1790                                        : (julong)MaxRAM;
1791   }
1792 
1793 
1794   // Convert deprecated flags
1795   if (FLAG_IS_DEFAULT(MaxRAMPercentage) &&
1796       !FLAG_IS_DEFAULT(MaxRAMFraction))
1797     MaxRAMPercentage = 100.0 / MaxRAMFraction;
1798 
1799   if (FLAG_IS_DEFAULT(MinRAMPercentage) &&
1800       !FLAG_IS_DEFAULT(MinRAMFraction))
1801     MinRAMPercentage = 100.0 / MinRAMFraction;
1802 
1803   if (FLAG_IS_DEFAULT(InitialRAMPercentage) &&
1804       !FLAG_IS_DEFAULT(InitialRAMFraction))
1805     InitialRAMPercentage = 100.0 / InitialRAMFraction;
1806 
1807   // If the maximum heap size has not been set with -Xmx,
1808   // then set it as fraction of the size of physical memory,
1809   // respecting the maximum and minimum sizes of the heap.
1810   if (FLAG_IS_DEFAULT(MaxHeapSize)) {
1811     julong reasonable_max = (julong)((phys_mem * MaxRAMPercentage) / 100);
1812     const julong reasonable_min = (julong)((phys_mem * MinRAMPercentage) / 100);
1813     if (reasonable_min < MaxHeapSize) {
1814       // Small physical memory, so use a minimum fraction of it for the heap
1815       reasonable_max = reasonable_min;
1816     } else {
1817       // Not-small physical memory, so require a heap at least
1818       // as large as MaxHeapSize
1819       reasonable_max = MAX2(reasonable_max, (julong)MaxHeapSize);
1820     }
1821 
1822     if (!FLAG_IS_DEFAULT(ErgoHeapSizeLimit) && ErgoHeapSizeLimit != 0) {
1823       // Limit the heap size to ErgoHeapSizeLimit
1824       reasonable_max = MIN2(reasonable_max, (julong)ErgoHeapSizeLimit);
1825     }
1826 
1827 #ifdef _LP64
1828     if (UseCompressedOops || UseCompressedClassPointers) {
1829       // HeapBaseMinAddress can be greater than default but not less than.
1830       if (!FLAG_IS_DEFAULT(HeapBaseMinAddress)) {
1831         if (HeapBaseMinAddress < DefaultHeapBaseMinAddress) {
1832           // matches compressed oops printing flags
1833           log_debug(gc, heap, coops)("HeapBaseMinAddress must be at least " SIZE_FORMAT
1834                                      " (" SIZE_FORMAT "G) which is greater than value given " SIZE_FORMAT,
1835                                      DefaultHeapBaseMinAddress,
1836                                      DefaultHeapBaseMinAddress/G,
1837                                      HeapBaseMinAddress);
1838           FLAG_SET_ERGO(HeapBaseMinAddress, DefaultHeapBaseMinAddress);
1839         }
1840       }
1841     }
1842     if (UseCompressedOops) {
1843       // Limit the heap size to the maximum possible when using compressed oops
1844       julong max_coop_heap = (julong)max_heap_for_compressed_oops();
1845 
1846       if (HeapBaseMinAddress + MaxHeapSize < max_coop_heap) {
1847         // Heap should be above HeapBaseMinAddress to get zero based compressed oops
1848         // but it should be not less than default MaxHeapSize.
1849         max_coop_heap -= HeapBaseMinAddress;
1850       }
1851 
1852       // If user specified flags prioritizing os physical
1853       // memory limits, then disable compressed oops if
1854       // limits exceed max_coop_heap and UseCompressedOops
1855       // was not specified.
1856       if (reasonable_max > max_coop_heap) {
1857         if (FLAG_IS_ERGO(UseCompressedOops) && override_coop_limit) {
1858           log_info(cds)("UseCompressedOops and UseCompressedClassPointers have been disabled due to"
1859             " max heap " SIZE_FORMAT " > compressed oop heap " SIZE_FORMAT ". "
1860             "Please check the setting of MaxRAMPercentage %5.2f."
1861             ,(size_t)reasonable_max, (size_t)max_coop_heap, MaxRAMPercentage);
1862           FLAG_SET_ERGO(UseCompressedOops, false);
1863           if (COMPRESSED_CLASS_POINTERS_DEPENDS_ON_COMPRESSED_OOPS) {
1864             FLAG_SET_ERGO(UseCompressedClassPointers, false);
1865           }
1866         } else {
1867           reasonable_max = MIN2(reasonable_max, max_coop_heap);
1868         }
1869       }
1870     }
1871 #endif // _LP64
1872 
1873     reasonable_max = limit_by_allocatable_memory(reasonable_max);
1874 
1875     if (!FLAG_IS_DEFAULT(InitialHeapSize)) {
1876       // An initial heap size was specified on the command line,
1877       // so be sure that the maximum size is consistent.  Done
1878       // after call to limit_by_allocatable_memory because that
1879       // method might reduce the allocation size.
1880       reasonable_max = MAX2(reasonable_max, (julong)InitialHeapSize);
1881     } else if (!FLAG_IS_DEFAULT(MinHeapSize)) {
1882       reasonable_max = MAX2(reasonable_max, (julong)MinHeapSize);
1883     }
1884 
1885     log_trace(gc, heap)("  Maximum heap size " SIZE_FORMAT, (size_t) reasonable_max);
1886     FLAG_SET_ERGO(MaxHeapSize, (size_t)reasonable_max);
1887   }
1888 
1889   // If the minimum or initial heap_size have not been set or requested to be set
1890   // ergonomically, set them accordingly.
1891   if (InitialHeapSize == 0 || MinHeapSize == 0) {
1892     julong reasonable_minimum = (julong)(OldSize + NewSize);
1893 
1894     reasonable_minimum = MIN2(reasonable_minimum, (julong)MaxHeapSize);
1895 
1896     reasonable_minimum = limit_by_allocatable_memory(reasonable_minimum);
1897 
1898     if (InitialHeapSize == 0) {
1899       julong reasonable_initial = (julong)((phys_mem * InitialRAMPercentage) / 100);
1900 
1901       reasonable_initial = MAX3(reasonable_initial, reasonable_minimum, (julong)MinHeapSize);
1902       reasonable_initial = MIN2(reasonable_initial, (julong)MaxHeapSize);
1903 
1904       reasonable_initial = limit_by_allocatable_memory(reasonable_initial);
1905 
1906       FLAG_SET_ERGO(InitialHeapSize, (size_t)reasonable_initial);
1907       log_trace(gc, heap)("  Initial heap size " SIZE_FORMAT, InitialHeapSize);
1908     }
1909     // If the minimum heap size has not been set (via -Xms or -XX:MinHeapSize),
1910     // synchronize with InitialHeapSize to avoid errors with the default value.
1911     if (MinHeapSize == 0) {
1912       FLAG_SET_ERGO(MinHeapSize, MIN2((size_t)reasonable_minimum, InitialHeapSize));
1913       log_trace(gc, heap)("  Minimum heap size " SIZE_FORMAT, MinHeapSize);
1914     }
1915   }
1916 }

1918 // This option inspects the machine and attempts to set various
1919 // parameters to be optimal for long-running, memory allocation
1920 // intensive jobs.  It is intended for machines with large
1921 // amounts of cpu and memory.
1922 jint Arguments::set_aggressive_heap_flags() {
1923   // initHeapSize is needed since _initial_heap_size is 4 bytes on a 32 bit
1924   // VM, but we may not be able to represent the total physical memory
1925   // available (like having 8gb of memory on a box but using a 32bit VM).
1926   // Thus, we need to make sure we're using a julong for intermediate
1927   // calculations.
1928   julong initHeapSize;
1929   julong total_memory = os::physical_memory();
1930 
1931   if (total_memory < (julong) 256 * M) {
1932     jio_fprintf(defaultStream::error_stream(),
1933             "You need at least 256mb of memory to use -XX:+AggressiveHeap\n");
1934     vm_exit(1);
1935   }
1936 
1937   // The heap size is half of available memory, or (at most)
1938   // all of possible memory less 160mb (leaving room for the OS
1939   // when using ISM).  This is the maximum; because adaptive sizing
1940   // is turned on below, the actual space used may be smaller.
1941 
1942   initHeapSize = MIN2(total_memory / (julong) 2,
1943           total_memory - (julong) 160 * M);
1944 
1945   initHeapSize = limit_by_allocatable_memory(initHeapSize);
1946 
1947   if (FLAG_IS_DEFAULT(MaxHeapSize)) {
1948     if (FLAG_SET_CMDLINE(MaxHeapSize, initHeapSize) != JVMFlag::SUCCESS) {
1949       return JNI_EINVAL;
1950     }
1951     if (FLAG_SET_CMDLINE(InitialHeapSize, initHeapSize) != JVMFlag::SUCCESS) {
1952       return JNI_EINVAL;
1953     }
1954     if (FLAG_SET_CMDLINE(MinHeapSize, initHeapSize) != JVMFlag::SUCCESS) {
1955       return JNI_EINVAL;
1956     }
1957   }
1958   if (FLAG_IS_DEFAULT(NewSize)) {
1959     // Make the young generation 3/8ths of the total heap.
1960     if (FLAG_SET_CMDLINE(NewSize,
1961             ((julong) MaxHeapSize / (julong) 8) * (julong) 3) != JVMFlag::SUCCESS) {
1962       return JNI_EINVAL;
1963     }
1964     if (FLAG_SET_CMDLINE(MaxNewSize, NewSize) != JVMFlag::SUCCESS) {
1965       return JNI_EINVAL;
1966     }
1967   }
1968 
1969 #if !defined(_ALLBSD_SOURCE) && !defined(AIX)  // UseLargePages is not yet supported on BSD and AIX.
1970   FLAG_SET_DEFAULT(UseLargePages, true);
1971 #endif
1972 
1973   // Increase some data structure sizes for efficiency
1974   if (FLAG_SET_CMDLINE(BaseFootPrintEstimate, MaxHeapSize) != JVMFlag::SUCCESS) {
1975     return JNI_EINVAL;
1976   }
1977   if (FLAG_SET_CMDLINE(ResizeTLAB, false) != JVMFlag::SUCCESS) {
1978     return JNI_EINVAL;
1979   }
1980   if (FLAG_SET_CMDLINE(TLABSize, 256 * K) != JVMFlag::SUCCESS) {
1981     return JNI_EINVAL;
1982   }
1983 
1984   // See the OldPLABSize comment below, but replace 'after promotion'
1985   // with 'after copying'.  YoungPLABSize is the size of the survivor
1986   // space per-gc-thread buffers.  The default is 4kw.
1987   if (FLAG_SET_CMDLINE(YoungPLABSize, 256 * K) != JVMFlag::SUCCESS) { // Note: this is in words
1988     return JNI_EINVAL;
1989   }
1990 
1991   // OldPLABSize is the size of the buffers in the old gen that
1992   // UseParallelGC uses to promote live data that doesn't fit in the
1993   // survivor spaces.  At any given time, there's one for each gc thread.
1994   // The default size is 1kw. These buffers are rarely used, since the
1995   // survivor spaces are usually big enough.  For specjbb, however, there
1996   // are occasions when there's lots of live data in the young gen
1997   // and we end up promoting some of it.  We don't have a definite
1998   // explanation for why bumping OldPLABSize helps, but the theory
1999   // is that a bigger PLAB results in retaining something like the
2000   // original allocation order after promotion, which improves mutator
2001   // locality.  A minor effect may be that larger PLABs reduce the
2002   // number of PLAB allocation events during gc.  The value of 8kw
2003   // was arrived at by experimenting with specjbb.
2004   if (FLAG_SET_CMDLINE(OldPLABSize, 8 * K) != JVMFlag::SUCCESS) { // Note: this is in words
2005     return JNI_EINVAL;
2006   }
2007 
2008   // Enable parallel GC and adaptive generation sizing
2009   if (FLAG_SET_CMDLINE(UseParallelGC, true) != JVMFlag::SUCCESS) {
2010     return JNI_EINVAL;
2011   }
2012 
2013   // Encourage steady state memory management
2014   if (FLAG_SET_CMDLINE(ThresholdTolerance, 100) != JVMFlag::SUCCESS) {
2015     return JNI_EINVAL;
2016   }
2017 
2018   // This appears to improve mutator locality
2019   if (FLAG_SET_CMDLINE(ScavengeBeforeFullGC, false) != JVMFlag::SUCCESS) {
2020     return JNI_EINVAL;
2021   }
2022 
2023   return JNI_OK;
2024 }

2381 // Parse -Xss memory string parameter and convert to ThreadStackSize in K.
2382 jint Arguments::parse_xss(const JavaVMOption* option, const char* tail, intx* out_ThreadStackSize) {
2383   // The min and max sizes match the values in globals.hpp, but scaled
2384   // with K. The values have been chosen so that alignment with page
2385   // size doesn't change the max value, which makes the conversions
2386   // back and forth between Xss value and ThreadStackSize value easier.
2387   // The values have also been chosen to fit inside a 32-bit signed type.
2388   const julong min_ThreadStackSize = 0;
2389   const julong max_ThreadStackSize = 1 * M;
2390 
2391   const julong min_size = min_ThreadStackSize * K;
2392   const julong max_size = max_ThreadStackSize * K;
2393 
2394   assert(is_aligned(max_size, os::vm_page_size()), "Implementation assumption");
2395 
2396   julong size = 0;
2397   ArgsRange errcode = parse_memory_size(tail, &size, min_size, max_size);
2398   if (errcode != arg_in_range) {
2399     bool silent = (option == NULL); // Allow testing to silence error messages
2400     if (!silent) {
2401       jio_fprintf(defaultStream::error_stream(),
2402                   "Invalid thread stack size: %s\n", option->optionString);
2403       describe_range_error(errcode);
2404     }
2405     return JNI_EINVAL;
2406   }
2407 
2408   // Internally track ThreadStackSize in units of 1024 bytes.
2409   const julong size_aligned = align_up(size, K);
2410   assert(size <= size_aligned,
2411          "Overflow: " JULONG_FORMAT " " JULONG_FORMAT,
2412          size, size_aligned);
2413 
2414   const julong size_in_K = size_aligned / K;
2415   assert(size_in_K < (julong)max_intx,
2416          "size_in_K doesn't fit in the type of ThreadStackSize: " JULONG_FORMAT,
2417          size_in_K);
2418 
2419   // Check that code expanding ThreadStackSize to a page aligned number of bytes won't overflow.
2420   const julong max_expanded = align_up(size_in_K * K, os::vm_page_size());
2421   assert(max_expanded < max_uintx && max_expanded >= size_in_K,
2422          "Expansion overflowed: " JULONG_FORMAT " " JULONG_FORMAT,
2423          max_expanded, size_in_K);
2424 
2425   *out_ThreadStackSize = (intx)size_in_K;
2426 
2427   return JNI_OK;
2428 }

3092 // Remove all empty paths from the app classpath (if IgnoreEmptyClassPaths is enabled)
3093 //
3094 // This is necessary because some apps like to specify classpath like -cp foo.jar:${XYZ}:bar.jar
3095 // in their start-up scripts. If XYZ is empty, the classpath will look like "-cp foo.jar::bar.jar".
3096 // Java treats such empty paths as if the user specified "-cp foo.jar:.:bar.jar". I.e., an empty
3097 // path is treated as the current directory.
3098 //
3099 // This causes problems with CDS, which requires that all directories specified in the classpath
3100 // must be empty. In most cases, applications do NOT want to load classes from the current
3101 // directory anyway. Adding -XX:+IgnoreEmptyClassPaths will make these applications' start-up
3102 // scripts compatible with CDS.
3103 void Arguments::fix_appclasspath() {
3104   if (IgnoreEmptyClassPaths) {
3105     const char separator = *os::path_separator();
3106     const char* src = _java_class_path->value();
3107 
3108     // skip over all the leading empty paths
3109     while (*src == separator) {
3110       src ++;
3111     }
3112 
3113     char* copy = os::strdup_check_oom(src, mtArguments);
3114 
3115     // trim all trailing empty paths
3116     for (char* tail = copy + strlen(copy) - 1; tail >= copy && *tail == separator; tail--) {
3117       *tail = '\0';
3118     }
3119 
3120     char from[3] = {separator, separator, '\0'};
3121     char to  [2] = {separator, '\0'};
3122     while (StringUtils::replace_no_expand(copy, from, to) > 0) {
3123       // Keep replacing "::" -> ":" until we have no more "::" (non-windows)
3124       // Keep replacing ";;" -> ";" until we have no more ";;" (windows)
3125     }
3126 
3127     _java_class_path->set_writeable_value(copy);
3128     FreeHeap(copy); // a copy was made by set_value, so don't need this anymore
3129   }
3130 }

3453 jint Arguments::parse_options_buffer(const char* name, char* buffer, const size_t buf_len, ScopedVMInitArgs* vm_args) {
3454   GrowableArray<JavaVMOption> *options = new (ResourceObj::C_HEAP, mtArguments) GrowableArray<JavaVMOption>(2, true);    // Construct option array
3455 
3456   // some pointers to help with parsing
3457   char *buffer_end = buffer + buf_len;
3458   char *opt_hd = buffer;
3459   char *wrt = buffer;
3460   char *rd = buffer;
3461 
3462   // parse all options
3463   while (rd < buffer_end) {
3464     // skip leading white space from the input string
3465     while (rd < buffer_end && isspace(*rd)) {
3466       rd++;
3467     }
3468 
3469     if (rd >= buffer_end) {
3470       break;
3471     }
3472 
3473     // Remember this is where we found the head of the token.
3474     opt_hd = wrt;
3475 
3476     // Tokens are strings of non white space characters separated
3477     // by one or more white spaces.
3478     while (rd < buffer_end && !isspace(*rd)) {
3479       if (*rd == '\'' || *rd == '"') {      // handle a quoted string
3480         int quote = *rd;                    // matching quote to look for
3481         rd++;                               // don't copy open quote
3482         while (rd < buffer_end && *rd != quote) {
3483                                             // include everything (even spaces)
3484                                             // up until the close quote
3485           *wrt++ = *rd++;                   // copy to option string
3486         }
3487 
3488         if (rd < buffer_end) {
3489           rd++;                             // don't copy close quote
3490         } else {
3491                                             // did not see closing quote
3492           jio_fprintf(defaultStream::error_stream(),
3493                       "Unmatched quote in %s\n", name);
3494           delete options;
3495           return JNI_ERR;
3496         }
3497       } else {
3498         *wrt++ = *rd++;                     // copy to option string
3499       }
3500     }
3501 
3502     // steal a white space character and set it to NULL
3503     *wrt++ = '\0';
3504     // We now have a complete token
3505 
3506     JavaVMOption option;
3507     option.optionString = opt_hd;
3508     option.extraInfo = NULL;
3509 
3510     options->append(option);                // Fill in option
3511 
3512     rd++;  // Advance to next character
3513   }
3514 
3515   // Fill out JavaVMInitArgs structure.
3516   jint status = vm_args->set_args(options);
3517 
3518   delete options;
3519   return status;
3520 }

4318 // Copies src into buf, replacing "%%" with "%" and "%p" with pid
4319 // Returns true if all of the source pointed by src has been copied over to
4320 // the destination buffer pointed by buf. Otherwise, returns false.
4321 // Notes:
4322 // 1. If the length (buflen) of the destination buffer excluding the
4323 // NULL terminator character is not long enough for holding the expanded
4324 // pid characters, it also returns false instead of returning the partially
4325 // expanded one.
4326 // 2. The passed in "buflen" should be large enough to hold the null terminator.
4327 bool Arguments::copy_expand_pid(const char* src, size_t srclen,
4328                                 char* buf, size_t buflen) {
4329   const char* p = src;
4330   char* b = buf;
4331   const char* src_end = &src[srclen];
4332   char* buf_end = &buf[buflen - 1];
4333 
4334   while (p < src_end && b < buf_end) {
4335     if (*p == '%') {
4336       switch (*(++p)) {
4337       case '%':         // "%%" ==> "%"
4338         *b++ = *p++;
4339         break;
4340       case 'p':  {       //  "%p" ==> current process id
4341         // buf_end points to the character before the last character so
4342         // that we could write '\0' to the end of the buffer.
4343         size_t buf_sz = buf_end - b + 1;
4344         int ret = jio_snprintf(b, buf_sz, "%d", os::current_process_id());
4345 
4346         // if jio_snprintf fails or the buffer is not long enough to hold
4347         // the expanded pid, returns false.
4348         if (ret < 0 || ret >= (int)buf_sz) {
4349           return false;
4350         } else {
4351           b += ret;
4352           assert(*b == '\0', "fail in copy_expand_pid");
4353           if (p == src_end && b == buf_end + 1) {
4354             // reach the end of the buffer.
4355             return true;
4356           }
4357         }
4358         p++;
4359         break;
4360       }
4361       default :
4362         *b++ = '%';
4363       }
4364     } else {
4365       *b++ = *p++;
4366     }
4367   }
4368   *b = '\0';
4369   return (p == src_end); // return false if not all of the source was copied
4370 }
