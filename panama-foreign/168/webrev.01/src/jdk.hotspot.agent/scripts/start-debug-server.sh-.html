27 if [ "$1" = "-help" ] ; then
28      echo "Usage: $0 <pid>"
29      echo "       $0 <java executable> <core file>"
30      echo " Start the JDI debug server on <pid> or <core file>"
31      echo " so that it can be debugged from a remote machine."
32      echo " JAVA_HOME must contain the pathname of a J2SE 1.5"
33      echo " installation."
34      exit 0
35 fi
36 
37 if [ ! -x ${JAVA_HOME}/bin/java -o ! -r ${JAVA_HOME}/lib/sa-jdi.jar ] ; 
38 then
39     echo '${JAVA_HOME} does not point to a working J2SE 1.5 installation.'
40     exit 1
41 fi
42 
43 ${JAVA_HOME}/bin/java -classpath ${JAVA_HOME}/lib/sa-jdi.jar sun.jvm.hotspot.jdi.SADebugServer $*
