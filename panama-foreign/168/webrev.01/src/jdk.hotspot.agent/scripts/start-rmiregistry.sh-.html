27 if [ "$1" = "-help" ] ; then
28     echo "usage: $0&"
29     echo " Start the rmi registry with with sa-jdi.jar on the bootclasspath"
30     echo " for use by the debug server."
31     echo " JAVA_HOME must contain the pathname of a J2SE 1.5" 
32     echo " installation." 
33     exit 0
34 fi
35 
36 if [ ! -x ${JAVA_HOME}/bin/rmiregistry -o ! -r ${JAVA_HOME}/lib/sa-jdi.jar ] ; 
37 then
38     echo '${JAVA_HOME} does not point to a working J2SE installation.'
39     exit 1
40 fi
41 
42 ${JAVA_HOME}/bin/rmiregistry -J-Xbootclasspath/p:${JAVA_HOME}/lib/sa-jdi.jar
