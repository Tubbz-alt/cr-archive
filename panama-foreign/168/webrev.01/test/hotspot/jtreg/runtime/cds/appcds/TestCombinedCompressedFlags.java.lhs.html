 24 /**
 25  * @test
 26  * @bug 8232069
 27  * @summary Testing different combination of CompressedOops and CompressedClassPointers
 28  * @requires vm.cds
 29  * @requires vm.gc == "null"
 30  * @requires vm.bits == 64
 31  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
 32  * @compile test-classes/Hello.java
 33  * @modules java.base/jdk.internal.misc
 34  * @run driver TestCombinedCompressedFlags
 35  */
 36 
 37 import jdk.test.lib.process.OutputAnalyzer;
 38 import java.util.List;
 39 import java.util.ArrayList;
 40 
 41 public class TestCombinedCompressedFlags {
 42     public static String HELLO_STRING = "Hello World";
 43     public static String EXEC_ABNORMAL_MSG = "Unable to use shared archive.";
 44     public static final int PASS = 0;
 45     public static final int FAIL = 1;
 46 
 47     static class ConfArg {
 48         public boolean useCompressedOops;            // UseCompressedOops
 49         public boolean useCompressedClassPointers;   // UseCompressedClassPointers
 50         public String  msg;
 51         public int code;
 52         public ConfArg(boolean useCompressedOops, boolean useCompressedClassPointers, String msg, int code) {
 53             this.useCompressedOops = useCompressedOops;
 54             this.useCompressedClassPointers = useCompressedClassPointers;
 55             this.msg  = msg;
 56             this.code = code;
 57         }
 58     }
 59 
 60     static class RunArg {
 61         public ConfArg dumpArg;
 62         public List<ConfArg> execArgs;
 63         public RunArg(ConfArg arg) {
 64             dumpArg = arg;
 65             initExecArgs();
 66         }
 67         private void initExecArgs() {
 68            /* The combinations have four cases. Note COOP off, CCPTR must be off
 69             *          UseCompressedOops   UseCompressedClassPointers  Result
 70             *    1.
 71             *    dump: on                  on
 72             *    test: on                  on                          Pass
 73             *          on                  off                         Fail
 74             *          off                 on                          Fail
 75             *          off                 off                         Fail
 76             *    2.
 77             *    dump: on                  off
 78             *    test: on                  off                         Pass
 79             *          on                  on                          Fail
 80             *          off                 on                          Pass
 81             *          off                 off                         Fail
 82             *    3.
 83             *    dump: off                 on
 84             *    test: off                 on                          Pass
 85             *          off                 off                         Pass
 86             *          on                  on                          Fail
 87             *          on                  off                         Fail
 88             *    4.
 89             *    dump: off                 off
 90             *    test: off                 off                         Pass
 91             *          off                 on                          Pass
 92             *          on                  on                          Fail
 93             *          on                  off                         Fail
 94             **/
 95             execArgs = new ArrayList<ConfArg>();
 96             if (dumpArg.useCompressedOops && dumpArg.useCompressedClassPointers) {
 97                 execArgs
 98                     .add(new ConfArg(true, true, HELLO_STRING, PASS));
 99                 execArgs
100                     .add(new ConfArg(true, false, EXEC_ABNORMAL_MSG, FAIL));
101                 execArgs
102                     .add(new ConfArg(false, true, EXEC_ABNORMAL_MSG, FAIL));
103                 execArgs
104                     .add(new ConfArg(false, false, EXEC_ABNORMAL_MSG, FAIL));
105 
106             }  else if(dumpArg.useCompressedOops && !dumpArg.useCompressedClassPointers) {
107                 execArgs
108                     .add(new ConfArg(true, false, HELLO_STRING, PASS));
109                 execArgs
110                     .add(new ConfArg(true, true, EXEC_ABNORMAL_MSG, FAIL));
111                 execArgs
112                     .add(new ConfArg(false, true, EXEC_ABNORMAL_MSG, FAIL));
113                 execArgs
114                     .add(new ConfArg(false, false, EXEC_ABNORMAL_MSG, FAIL));
115 
116             } else if (!dumpArg.useCompressedOops && dumpArg.useCompressedClassPointers) {
117                 execArgs
118                     .add(new ConfArg(false, false, HELLO_STRING, PASS));
119                 execArgs
120                     .add(new ConfArg(false, true, HELLO_STRING, PASS));
121                 execArgs
122                     .add(new ConfArg(true, true, EXEC_ABNORMAL_MSG, FAIL));
123                 execArgs
124                     .add(new ConfArg(true, false, EXEC_ABNORMAL_MSG, FAIL));
125             } else if (!dumpArg.useCompressedOops && !dumpArg.useCompressedClassPointers) {
126                 execArgs
127                     .add(new ConfArg(false, false, HELLO_STRING, PASS));
128                 execArgs
129                     .add(new ConfArg(false, true, HELLO_STRING, PASS));
130                 execArgs
131                     .add(new ConfArg(true, true, EXEC_ABNORMAL_MSG, FAIL));
132                 execArgs
133                     .add(new ConfArg(true, false, EXEC_ABNORMAL_MSG, FAIL));
134             }
135         }
136     }
137 
138     public static String getCompressedOopsArg(boolean on) {
139         if (on) return "-XX:+UseCompressedOops";
140         else    return "-XX:-UseCompressedOops";
141     }
142 
143     public static String getCompressedClassPointersArg(boolean on) {
144         if (on) return "-XX:+UseCompressedClassPointers";
145         else    return "-XX:-UseCompressedClassPointers";
146     }
147 
148     public static List<RunArg> runList;
149 
150     public static void configureRunArgs() {
151         runList = new ArrayList<RunArg>();
152         runList
153             .add(new RunArg(new ConfArg(true, true, null, PASS)));
154         runList
155             .add(new RunArg(new ConfArg(true, false, null, PASS)));
156         runList
157             .add(new RunArg(new ConfArg(false, true, null, PASS)));
158         runList
159             .add(new RunArg(new ConfArg(false, false, null, PASS)));
160     }
161 
162     public static void main(String[] args) throws Exception {
163         String helloJar = JarBuilder.build("hello", "Hello");
164         configureRunArgs();
165         OutputAnalyzer out;
166         for (RunArg t: runList) {
167             out = TestCommon
168                 .dump(helloJar,
169                       new String[] {"Hello"},
170                       getCompressedOopsArg(t.dumpArg.useCompressedOops),
171                       getCompressedClassPointersArg(t.dumpArg.useCompressedClassPointers),
172                       "-Xlog:cds");
173             out.shouldContain("Dumping shared data to file:");
174             out.shouldHaveExitValue(0);
175 
176             for (ConfArg c : t.execArgs) {
177                 out = TestCommon.exec(helloJar,
178                                       "-cp",
179                                       helloJar,
180                                       "-Xlog:cds",
181                                       getCompressedOopsArg(c.useCompressedOops),
182                                       getCompressedClassPointersArg(c.useCompressedClassPointers),
183                                       "Hello");
184                 out.shouldContain(c.msg);
185                 out.shouldHaveExitValue(c.code);
186             }
187         }
188     }
189 }
