25 /*
26  * @test
27  * @summary When HelloA and HelloB are copied into the dynamic archive, the Symbols
28  *          for their method's names will have a different sorting order. This requires
29  *          that the dumped InstanceKlass to re-sort their "methods" array and re-layout the vtables/itables.
30  *          A regression test for an earlier bug in DynamicArchiveBuilder::relocate_buffer_to_target().
31  * @requires vm.cds
32  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds /test/hotspot/jtreg/runtime/cds/appcds/test-classes
33  *          /test/hotspot/jtreg/runtime/cds/appcds/dynamicArchive/test-classes
34  * @build MethodSortingApp
35  * @build sun.hotspot.WhiteBox
36  * @run driver ClassFileInstaller -jar method_sorting.jar
37  *             MethodSortingApp
38  *             MethodSortingApp$HelloA
39  *             MethodSortingApp$HelloA1
40  *             MethodSortingApp$HelloB
41  *             MethodSortingApp$HelloB1
42  *             MethodSortingApp$InterfaceA
43  *             MethodSortingApp$InterfaceB
44  *             MethodSortingApp$ImplementorA
45  *             MethodSortingApp$ImplementorA1
46  *             MethodSortingApp$ImplementorB
47  *             MethodSortingApp$ImplementorB1
48  * @run driver ClassFileInstaller sun.hotspot.WhiteBox sun.hotspot.WhiteBox$WhiteBoxPermission
49  * @run main/othervm -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:. MethodSorting
50  */
51 
52 public class MethodSorting extends DynamicArchiveTestBase {
53     public static void main(String[] args) throws Exception {
54         runTest(MethodSorting::test);
55     }
56 
57     static void test() throws Exception {
58         String topArchiveName = getNewArchiveName();
59         String appJar = ClassFileInstaller.getJarPath("method_sorting.jar");
60         String mainClass = "MethodSortingApp";
61 
62         dumpAndRun(topArchiveName, "-Xlog:cds+dynamic=debug", "-cp", appJar, mainClass);
63     }
64 }
