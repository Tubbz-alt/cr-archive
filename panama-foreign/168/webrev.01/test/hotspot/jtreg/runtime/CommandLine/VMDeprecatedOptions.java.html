 24 import jdk.test.lib.process.ProcessTools;
 25 import jdk.test.lib.process.OutputAnalyzer;
 26 import jdk.test.lib.cli.*;
 27 
 28 /*
 29  * @test
 30  * @bug 8066821
 31  * @summary Test that various options are deprecated. See deprecated_jvm_flags in arguments.cpp.
 32  * @modules java.base/jdk.internal.misc
 33  * @library /test/lib
 34  * @run driver VMDeprecatedOptions
 35  */
 36 public class VMDeprecatedOptions {
 37 
 38     /**
 39      * each entry is {[0]: option name, [1]: value to set
 40      * (true/false/n/string)}.
 41      */
 42     public static final String[][] DEPRECATED_OPTIONS = {
 43         // deprecated non-alias flags:
 44         {"MaxGCMinorPauseMillis",     "1032"},
 45         {"MaxRAMFraction",            "8"},
 46         {"MinRAMFraction",            "2"},
 47         {"InitialRAMFraction",        "64"},
 48         {"TLABStats",                 "false"},
 49         {"AllowRedefinitionToAddDeleteMethods", "true"},
 50         {"PrintVMQWaitTime",          "true"},
 51         {"UseNewFieldLayout",         "true"},
 52         {"UseBiasedLocking",                    "false"},
 53         {"BiasedLockingStartupDelay",           "0"},
 54         {"BiasedLockingBulkRebiasThreshold",    "20"},
 55         {"BiasedLockingBulkRevokeThreshold",    "40"},
 56         {"BiasedLockingDecayTime",              "25000"},
 57         {"UseOptoBiasInlining",                 "true"},
 58 
 59         // deprecated alias flags (see also aliased_jvm_flags):
 60         {"DefaultMaxRAMFraction", "4"},
 61         {"CreateMinidumpOnCrash", "false"}
 62     };
 63 
 64     static String getDeprecationString(String optionName) {
 65         return "Option " + optionName
 66             + " was deprecated in version [\\S]+ and will likely be removed in a future release";
 67     }
 68 
 69     static void testDeprecated(String[][] optionInfo) throws Throwable {
 70         String optionNames[] = new String[optionInfo.length];
 71         String expectedValues[] = new String[optionInfo.length];
 72         for (int i = 0; i < optionInfo.length; i++) {
 73             optionNames[i] = optionInfo[i][0];
 74             expectedValues[i] = optionInfo[i][1];
 75         }
 76 
 77         OutputAnalyzer output = CommandLineOptionTest.startVMWithOptions(optionNames, expectedValues);
 78 
 79         // check for option deprecation messages:
 80         output.shouldHaveExitValue(0);
 81         for (String[] deprecated : optionInfo) {
 82             String match = getDeprecationString(deprecated[0]);
 83             output.shouldMatch(match);
 84         }
 85     }
 86 
 87     // Deprecated diagnostic command line options need to be preceded on the
 88     // command line by -XX:+UnlockDiagnosticVMOptions.
 89     static void testDeprecatedDiagnostic(String option, String value)  throws Throwable {
 90         String XXoption = CommandLineOptionTest.prepareFlag(option, value);
 91         ProcessBuilder processBuilder = ProcessTools.createJavaProcessBuilder(
 92             CommandLineOptionTest.UNLOCK_DIAGNOSTIC_VM_OPTIONS, XXoption, "-version");
 93         OutputAnalyzer output = new OutputAnalyzer(processBuilder.start());
 94         // check for option deprecation message:
 95         output.shouldHaveExitValue(0);
 96         String match = getDeprecationString(option);
 97         output.shouldMatch(match);
 98     }
 99 
100     // Deprecated experimental command line options need to be preceded on the
101     // command line by -XX:+UnlockExperimentalVMOption.
102     static void testDeprecatedExperimental(String option, String value)  throws Throwable {
103         String XXoption = CommandLineOptionTest.prepareFlag(option, value);
104         ProcessBuilder processBuilder = ProcessTools.createJavaProcessBuilder(
105             CommandLineOptionTest.UNLOCK_EXPERIMENTAL_VM_OPTIONS, XXoption, "-version");
106         OutputAnalyzer output = new OutputAnalyzer(processBuilder.start());
107         // check for option deprecation message:
108         output.shouldHaveExitValue(0);
109         String match = getDeprecationString(option);
110         output.shouldMatch(match);
111     }
112 
113     public static void main(String[] args) throws Throwable {
114         testDeprecated(DEPRECATED_OPTIONS);  // Make sure that each deprecated option is mentioned in the output.
115     }
116 }
