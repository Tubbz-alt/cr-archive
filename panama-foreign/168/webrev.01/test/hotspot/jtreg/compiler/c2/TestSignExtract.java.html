25 /*
26  * @test
27  * @bug 8242429
28  * @summary Better implementation for sign extract.
29  *
30  * @run main/othervm -XX:-TieredCompilation -XX:CompileCommand=dontinline,compiler.c2.TestSignExtract::signExtract*
31  *      compiler.c2.TestSignExtract
32  */
33 package compiler.c2;
34 
35 public class TestSignExtract {
36 
37     private static final long[] LONG_VALUES = {0L, 0xFFFFFFFFL, 0x12L, -1L, -123L, -0x12L, Long.MAX_VALUE, Long.MIN_VALUE};
38     private static final int[] INT_VALUES = {0, 0x1234, -1, -0x12345678, Integer.MAX_VALUE, Integer.MIN_VALUE};
39 
40 
41     private static int signExtractInt1(int x) {
42         return (x >> 1) >>> 31;
43     }
44 
45     private static int signExtractInt2(int x) {
46         return (x >> 32) >>> 31;
47     }
48 
49     private static int signExtractInt3(int x) {
50         return (x >> 31) >>> 31;
51     }
52 
53     private static int signExtractInt4(int x) {
54         return 0 - (x >> 31);
55     }
56 
57     private static long signExtractLong1(long x) {
58         return (x >> 1) >>> 63;
59     }
60 
61     private static long signExtractLong2(long x) {
62         return (x >> 54) >>> 63;
63     }
64 
65     private static long signExtractLong3(long x) {
66         return (x >> 63) >>> 63;
67     }
68 
69     private static long signExtractLong4(long x) {
70         return 0 - (x >> 63);
71     }
72 
73     private static int WARMUP = 5000;
74 
75     public static void main(String[] args) {
76         for (int i = 0; i < WARMUP; i++) {
77             for (int e : INT_VALUES) {
78                 // "(A >> n) >>> 31" => "(A >>> 31)"
79                 assert e >>> 31 == signExtractInt1(e);
80                 assert e >>> 31 == signExtractInt2(e);
81                 assert e >>> 31 == signExtractInt3(e);
82                 // "0 - (A >> 31)" => "(A >>> 31)"
83                 assert e >>> 31 == signExtractInt4(e);
84             }
85 
86             for (long e : LONG_VALUES) {
87                 // "(A >> n) >>> 63" => "(A >>> 63)"
88                 assert e >>> 63 == signExtractLong1(e);
89                 assert e >>> 63 == signExtractLong2(e);
90                 assert e >>> 63 == signExtractLong3(e);
91                 // "0 - (A >> 63)" => "(A >>> 63)"
92                 assert e >>> 63 == signExtractLong4(e);
93             }
94         }
95     }
96 }
