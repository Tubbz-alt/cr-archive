59     /**
60      * Formats the command according to the following pattern:
61      * {@code <command_name> Type: <type> Compiler: <compiler> MethodDescriptor: <method_descriptor> IsValid: <true/false>}
62      * Sample output:
63      * COMPILEONLY Type: OPTION Compiler: C1 MethodDescriptor: *Klass.method* IsValid: true
64      */
65     protected String formatFields() {
66         return command.name() +
67                " Type: " + type +
68                " Compiler: " + compiler +
69                " MethodDescriptor: " + (methodDescriptor == null ? "null" : methodDescriptor.getString()) +
70                " IsValid: " + isValid();
71     }
72 
73     /**
74      * Returns formatted string representation in the form
75      * {@code "(CompileCommand Field1: <field1> Field2: <field2> ...)}
76      * The fields are formatted by {@link #formatFields()}.
77      */
78     public String toString() {
79         return "(CompileCommand " + formatFields() + ")";
80     }
