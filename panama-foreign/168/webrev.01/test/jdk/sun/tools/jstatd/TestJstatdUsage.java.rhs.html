24 import jdk.test.lib.JDKToolLauncher;
25 import jdk.test.lib.Utils;
26 import jdk.test.lib.process.OutputAnalyzer;
27 import jdk.test.lib.process.ProcessTools;
28 
29 /*
30  * @test
31  * @bug 4990825
32  * @library /test/lib
33  * @modules java.management
34  * @run main TestJstatdUsage
35  */
36 public class TestJstatdUsage {
37 
38     public static void main(String[] args) throws Exception {
39         testUsage("-?");
40         testUsage("-h");
41         testUsage("--help");
42     }
43 
44     private static void testUsage(String option) throws Exception {
45         JDKToolLauncher launcher = JDKToolLauncher.createUsingTestJDK("jstatd");
46         launcher.addVMArgs(Utils.getTestJavaOpts());
47         launcher.addToolArg(option);
48         ProcessBuilder processBuilder = new ProcessBuilder(launcher.getCommand());
49         OutputAnalyzer output = ProcessTools.executeProcess(processBuilder);
50 
51         output.shouldContain("usage: jstatd [-nr] [-p port] [-r rmiport] [-n rminame]");
52         output.shouldHaveExitValue(0);
53     }
54 
55 }
