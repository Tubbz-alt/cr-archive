25 import jdk.test.lib.process.OutputAnalyzer;
26 import jdk.test.lib.process.ProcessTools;
27 
28 /*
29  * @test
30  * @bug 4990825
31  * @library /test/lib
32  * @modules java.management
33  * @run main TestJstatdUsage
34  */
35 public class TestJstatdUsage {
36 
37     public static void main(String[] args) throws Exception {
38         testUsage("-?");
39         testUsage("-h");
40         testUsage("--help");
41     }
42 
43     private static void testUsage(String option) throws Exception {
44         JDKToolLauncher launcher = JDKToolLauncher.createUsingTestJDK("jstatd");

45         launcher.addToolArg(option);
46         ProcessBuilder processBuilder = new ProcessBuilder(launcher.getCommand());
47         OutputAnalyzer output = ProcessTools.executeProcess(processBuilder);
48 
49         output.shouldContain("usage: jstatd [-nr] [-p port] [-r rmiport] [-n rminame]");
50         output.shouldHaveExitValue(0);
51     }
52 
53 }
