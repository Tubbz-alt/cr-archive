 26 package jdk.jfr.tool;
 27 
 28 import java.nio.file.Path;
 29 import java.time.OffsetDateTime;
 30 import java.util.Collections;
 31 import java.util.Iterator;
 32 import java.util.List;
 33 
 34 import javax.script.ScriptEngine;
 35 import javax.script.ScriptEngineManager;
 36 
 37 import jdk.jfr.Timespan;
 38 import jdk.jfr.Timestamp;
 39 import jdk.jfr.ValueDescriptor;
 40 import jdk.jfr.consumer.RecordedEvent;
 41 import jdk.jfr.consumer.RecordedObject;
 42 import jdk.jfr.consumer.RecordingFile;
 43 import jdk.nashorn.api.scripting.JSObject;
 44 import jdk.test.lib.Asserts;
 45 import jdk.test.lib.process.OutputAnalyzer;
 46 
 47 /**
 48  * @test
 49  * @key jfr
 50  * @summary Tests print --json
 51  * @requires vm.hasJFR
 52  *
 53  * @library /test/lib /test/jdk
 54  * @modules jdk.scripting.nashorn
 55  *          jdk.jfr
 56  *
 57  * @run main/othervm jdk.jfr.tool.TestPrintJSON
 58  */
 59 public class TestPrintJSON {
 60 
 61     public static void main(String... args) throws Throwable {
 62 
 63         Path recordingFile = ExecuteHelper.createProfilingRecording().toAbsolutePath();
 64 
 65         OutputAnalyzer output = ExecuteHelper.jfr("print", "--json", "--stack-depth", "999", recordingFile.toString());
 66         String json = output.getStdout();
 67 
 68         // Parse JSON using Nashorn
 69         String statement = "var jsonObject = " + json;
 70         ScriptEngineManager factory = new ScriptEngineManager();
 71         ScriptEngine engine = factory.getEngineByName("nashorn");
 72         engine.eval(statement);
 73         JSObject o = (JSObject) engine.get("jsonObject");
 74         JSObject recording = (JSObject) o.getMember("recording");
 75         JSObject jsonEvents = (JSObject) recording.getMember("events");
 76 
 77         List<RecordedEvent> events = RecordingFile.readAllEvents(recordingFile);
 78         Collections.sort(events, (e1, e2) -> e1.getEndTime().compareTo(e2.getEndTime()));
 79         // Verify events are equal
 80         Iterator<RecordedEvent> it = events.iterator();
 81 
 82         for (Object jsonEvent : jsonEvents.values()) {
 83             RecordedEvent recordedEvent = it.next();
 84             String typeName = recordedEvent.getEventType().getName();
 85             Asserts.assertEquals(typeName, ((JSObject) jsonEvent).getMember("type").toString());
 86             assertEquals(jsonEvent, recordedEvent);
 87         }
 88         Asserts.assertFalse(events.size() != jsonEvents.values().size(), "Incorrect number of events");
 89     }
 90 
 91     private static void assertEquals(Object jsonObject, Object jfrObject) throws Exception {
 92         // Check object
 93         if (jfrObject instanceof RecordedObject) {
 94             JSObject values = (JSObject) ((JSObject) jsonObject).getMember("values");
 95             RecordedObject recObject = (RecordedObject) jfrObject;
 96             Asserts.assertEquals(values.values().size(), recObject.getFields().size());
 97             for (ValueDescriptor v : recObject.getFields()) {
 98                 String name = v.getName();
 99                 Object jsonValue = values.getMember(name);
100                 Object expectedValue = recObject.getValue(name);
101                 if (v.getAnnotation(Timestamp.class) != null) {
102                     // Make instant of OffsetDateTime
103                     jsonValue = OffsetDateTime.parse("" + jsonValue).toInstant().toString();

104                     expectedValue = recObject.getInstant(name);
105                 }
106                 if (v.getAnnotation(Timespan.class) != null) {
107                     expectedValue = recObject.getDuration(name);
108                 }
109                 assertEquals(jsonValue, expectedValue);
110                 return;
111             }
112         }
113         // Check array
114         if (jfrObject != null && jfrObject.getClass().isArray()) {
115             Object[] jfrArray = (Object[]) jfrObject;
116             JSObject jsArray = (JSObject) jsonObject;
117             for (int i = 0; i < jfrArray.length; i++) {
118                 assertEquals(jsArray.getSlot(i), jfrArray[i]);
119             }
120             return;
121         }
122         String jsonText = String.valueOf(jsonObject);
123         // Double.NaN / Double.Inifinity is not supported by JSON format,
124         // use null
125         if (jfrObject instanceof Double) {
126             double expected = ((Double) jfrObject);
127             if (Double.isInfinite(expected) || Double.isNaN(expected)) {
128                 Asserts.assertEquals("null", jsonText);
129                 return;
130             }
131             double value = Double.parseDouble(jsonText);
132             Asserts.assertEquals(expected, value);
133             return;
134         }
135         // Float.NaN / Float.Inifinity is not supported by JSON format,
136         // use null
137         if (jfrObject instanceof Float) {
138             float expected = ((Float) jfrObject);
139             if (Float.isInfinite(expected) || Float.isNaN(expected)) {
140                 Asserts.assertEquals("null", jsonText);
141                 return;
142             }
143             float value = Float.parseFloat(jsonText);
144             Asserts.assertEquals(expected, value);
145             return;
146         }
147         if (jfrObject instanceof Integer) {
148             Integer expected = ((Integer) jfrObject);
149             double value = Double.parseDouble(jsonText);
150             Asserts.assertEquals(expected.doubleValue(), value);
151             return;
152         }
153         if (jfrObject instanceof Long) {
154             Long expected = ((Long) jfrObject);
155             double value = Double.parseDouble(jsonText);
156             Asserts.assertEquals(expected.doubleValue(), value);
157             return;
158         }
159 
160         String jfrText = String.valueOf(jfrObject);
161         Asserts.assertEquals(jfrText, jsonText, "Primitive values don't match. JSON = " + jsonText);
162     }
163 }
