 59 public class TestPrintJSON {
 60 
 61     public static void main(String... args) throws Throwable {
 62 
 63         Path recordingFile = ExecuteHelper.createProfilingRecording().toAbsolutePath();
 64 
 65         OutputAnalyzer output = ExecuteHelper.jfr("print", "--json", "--stack-depth", "999", recordingFile.toString());
 66         String json = output.getStdout();
 67 
 68         // Parse JSON using Nashorn
 69         String statement = "var jsonObject = " + json;
 70         ScriptEngineManager factory = new ScriptEngineManager();
 71         ScriptEngine engine = factory.getEngineByName("nashorn");
 72         engine.eval(statement);
 73         JSObject o = (JSObject) engine.get("jsonObject");
 74         JSObject recording = (JSObject) o.getMember("recording");
 75         JSObject jsonEvents = (JSObject) recording.getMember("events");
 76 
 77         List<RecordedEvent> events = RecordingFile.readAllEvents(recordingFile);
 78         Collections.sort(events, (e1, e2) -> e1.getEndTime().compareTo(e2.getEndTime()));
 79         // Verify events are equal
 80         Iterator<RecordedEvent> it = events.iterator();
 81 
 82         for (Object jsonEvent : jsonEvents.values()) {
 83             RecordedEvent recordedEvent = it.next();
 84             String typeName = recordedEvent.getEventType().getName();
 85             Asserts.assertEquals(typeName, ((JSObject) jsonEvent).getMember("type").toString());
 86             assertEquals(jsonEvent, recordedEvent);
 87         }
 88         Asserts.assertFalse(events.size() != jsonEvents.values().size(), "Incorrect number of events");
 89     }
