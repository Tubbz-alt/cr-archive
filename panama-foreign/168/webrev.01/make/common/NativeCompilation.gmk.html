  41 # get the exported symbols from mapfiles and if there
  42 # is no mapfile, get them from the archive
  43 define GetSymbols
  44   $(RM) $$(@D)/$$(basename $$(@F)).symbols; \
  45   if [ ! -z $$($1_MAPFILE) -a -e $$($1_MAPFILE) ]; then \
  46     $(ECHO) "Getting symbols from mapfile $$($1_MAPFILE)"; \
  47     $(AWK) '/global:/','/local:/' $$($1_MAPFILE) | \
  48         $(SED) -e 's/#.*//;s/global://;s/local://;s/\;//;s/^[ 	]*/_/;/^_$$$$/d' | \
  49         $(EGREP) -v "JNI_OnLoad|JNI_OnUnload|Agent_OnLoad|Agent_OnUnload|Agent_OnAttach" > \
  50         $$(@D)/$$(basename $$(@F)).symbols || true; \
  51     $(NM) $$($1_TARGET) | $(GREP)  " T " | \
  52         $(EGREP) "JNI_OnLoad|JNI_OnUnload|Agent_OnLoad|Agent_OnUnload|Agent_OnAttach" | \
  53         $(CUT) -d ' ' -f 3 >>  $$(@D)/$$(basename $$(@F)).symbols || true;\
  54   else \
  55     $(ECHO) "Getting symbols from nm"; \
  56     $(NM) -m $$($1_TARGET) | $(GREP)  "__TEXT" | \
  57         $(EGREP) -v "non-external|private extern|__TEXT,__eh_frame" | \
  58         $(SED) -e  's/.* //' > $$(@D)/$$(basename $$(@F)).symbols; \
  59   fi
  60 endef

  62 ################################################################################
  63 # Creates a recipe that creates a compile_commands.json fragment. Remove any
  64 # occurences of FIXPATH programs from the command to show the actual invocation.
  65 #
  66 # Param 1: Name of file to create
  67 # Param 2: Working directory
  68 # Param 3: Source file
  69 # Param 4: Compile command
  70 ################################################################################
  71 define WriteCompileCommandsFragment
  72   $(call LogInfo, Creating compile commands fragment for $(notdir $3))
  73   $(call MakeDir, $(dir $1))
  74   $(call WriteFile,{ \
  75       "directory": "$(strip $2)"$(COMMA) \
  76       "file": "$(strip $3)"$(COMMA) \
  77       "command": "$(strip $(subst $(DQUOTE),\$(DQUOTE),$(subst \,\\,\
  78         $(subst $(FIXPATH),,$4))))" \
  79     }$(COMMA), \
  80     $1)
  81 endef

  84 # Define a native toolchain configuration that can be used by
  85 # SetupNativeCompilation calls
  86 #
  87 # Parameter 1 is the name of the toolchain definition
  88 #
  89 # Remaining parameters are named arguments:
  90 #   EXTENDS - Optional parent definition to get defaults from
  91 #   CC - The C compiler
  92 #   CXX - The C++ compiler
  93 #   LD - The Linker
  94 #   AR - Static linker
  95 #   AS - Assembler
  96 #   MT - Windows MT tool
  97 #   RC - Windows RC tool
  98 #   OBJCOPY - The objcopy tool for debug symbol handling
  99 #   STRIP - The tool to use for stripping debug symbols
 100 #   SYSROOT_CFLAGS - Compiler flags for using the specific sysroot
 101 #   SYSROOT_LDFLAGS - Linker flags for using the specific sysroot
 102 DefineNativeToolchain = $(NamedParamsMacroTemplate)
 103 define DefineNativeToolchainBody
 104   # If extending another definition, get default values from that,
 105   # otherwise, nothing more needs to be done as variable assignments
 106   # already happened in NamedParamsMacroTemplate.
 107   ifneq ($$($1_EXTENDS), )
 108     $$(call SetIfEmpty, $1_CC, $$($$($1_EXTENDS)_CC))
 109     $$(call SetIfEmpty, $1_CXX, $$($$($1_EXTENDS)_CXX))
 110     $$(call SetIfEmpty, $1_LD, $$($$($1_EXTENDS)_LD))
 111     $$(call SetIfEmpty, $1_AR, $$($$($1_EXTENDS)_AR))
 112     $$(call SetIfEmpty, $1_AS, $$($$($1_EXTENDS)_AS))
 113     $$(call SetIfEmpty, $1_MT, $$($$($1_EXTENDS)_MT))
 114     $$(call SetIfEmpty, $1_RC, $$($$($1_EXTENDS)_RC))
 115     $$(call SetIfEmpty, $1_OBJCOPY, $$($$($1_EXTENDS)_OBJCOPY))
 116     $$(call SetIfEmpty, $1_STRIP, $$($$($1_EXTENDS)_STRIP))
 117     $$(call SetIfEmpty, $1_SYSROOT_CFLAGS, $$($$($1_EXTENDS)_SYSROOT_CFLAGS))
 118     $$(call SetIfEmpty, $1_SYSROOT_LDFLAGS, $$($$($1_EXTENDS)_SYSROOT_LDFLAGS))
 119   endif
 120 endef

 207 ################################################################################
 208 # When absolute paths are not allowed in the output, and the compiler does not
 209 # support any options to avoid it, we need to rewrite compile commands to use
 210 # relative paths. By doing this, the __FILE__ macro will resolve to relative
 211 # paths. The relevant input paths on the command line are the -I flags and the
 212 # path to the source file itself.
 213 #
 214 # The macro MakeCommandRelative is used to rewrite the command line like this:
 215 # 'CD $(WORKSPACE_ROOT) && <cmd>'
 216 # and changes all paths in cmd to be relative to the workspace root. This only
 217 # works properly if the build dir is inside the workspace root. If it's not,
 218 # relative paths are still calculated, but depending on the distance between the
 219 # dirs, paths in the build dir may end up as essentially absolute anyway.
 220 #
 221 # The fix-deps-file macro is used to adjust the contents of the generated make
 222 # dependency files to contain paths compatible with make.
 223 #
 224 ifeq ($(ALLOW_ABSOLUTE_PATHS_IN_OUTPUT)-$(FILE_MACRO_CFLAGS), false-)
 225   # Need to handle -I flags as both '-Ifoo' and '-I foo'.
 226   MakeCommandRelative = \
 227       $(CD) $(WORKSPACE_ROOT) && \
 228       $(foreach o, $1, \
 229         $(if $(filter $(WORKSPACE_ROOT)/% $(OUTPUTDIR)/%, $o), \
 230           $(call RelativePath, $o, $(WORKSPACE_ROOT)) \
 231         , \
 232           $(if $(filter -I$(WORKSPACE_ROOT)/%, $o), \
 233             -I$(call RelativePath, $(patsubst -I%, %, $o), $(WORKSPACE_ROOT)) \
 234           , \
 235             $o \
 236           ) \
 237         ) \
 238       )
 239 
 240   # When compiling with relative paths, the deps file comes out with relative
 241   # paths.
 242   ifeq ($(TOOLCHAIN_TYPE), solstudio)
 243     define fix-deps-file
 244 	$(SED) -e 's|\./|$(WORKSPACE_ROOT)/|g' $1.tmp > $1
 245     endef
 246   else
 247     define fix-deps-file
 248 	$(SED) -e 's|^\([ ]*\)|\1$(WORKSPACE_ROOT)|' $1.tmp > $1
 249     endef
 250   endif
 251 else
 252   # By default the MakeCommandRelative macro does nothing.
 253   MakeCommandRelative = $1
 254 
 255   # Even with absolute paths on the command line, the Solaris studio compiler
 256   # doesn't output the full path to the object file in the generated deps files.
 257   # For other toolchains, no adjustment is needed.
 258   ifeq ($(TOOLCHAIN_TYPE), solstudio)
 259     define fix-deps-file
 260 	$(SED) 's|^$$(@F):|$$@:|' $1.tmp > $1
 261     endef
 262   else
 263     define fix-deps-file
 264 	$(MV) $1.tmp $1
 265     endef
 266   endif
 267 endif

 282 ################################################################################
 283 # Create the recipe needed to compile a single native source file.
 284 #
 285 # Parameter 1 is the name of the rule, based on the name of the library/
 286 # program being build and the name of the source code file, e.g.
 287 # BUILD_LIBFOO_fooMain.cpp.
 288 #
 289 # Remaining parameters are named arguments:
 290 #   FILE - The full path of the source file to compiler
 291 #   BASE - The name of the rule for the entire binary to build ($1)
 292 #
 293 SetupCompileNativeFile = $(NamedParamsMacroTemplate)
 294 define SetupCompileNativeFileBody
 295   $1_FILENAME := $$(notdir $$($1_FILE))
 296 
 297   # The target file to be generated.
 298   $1_OBJ := $$($$($1_BASE)_OBJECT_DIR)/$$(call replace_with_obj_extension, \
 299       $$($1_FILENAME))
 300 
 301   # Generate the corresponding compile_commands.json fragment.
 302   $1_OBJ_JSON = $$(MAKESUPPORT_OUTPUTDIR)/compile-commands/$$(subst /,_,$$(subst \
 303       $$(OUTPUTDIR)/,,$$($1_OBJ))).json
 304   $$($1_BASE)_ALL_OBJS_JSON += $$($1_OBJ_JSON)
 305 
 306   # Only continue if this object file hasn't been processed already. This lets
 307   # the first found source file override any other with the same name.
 308   ifeq ($$($1_OBJ_PROCESSED), )
 309     $1_OBJ_PROCESSED := true
 310     # This is the definite source file to use for $1_FILENAME.
 311     $1_SRC_FILE := $$($1_FILE)
 312 
 313     ifeq ($$($1_OPTIMIZATION), )
 314       $1_OPT_CFLAGS := $$($$($1_BASE)_OPT_CFLAGS)
 315       $1_OPT_CXXFLAGS := $$($$($1_BASE)_OPT_CXXFLAGS)
 316     else
 317       ifeq ($$($1_OPTIMIZATION), NONE)
 318         $1_OPT_CFLAGS := $(C_O_FLAG_NONE)
 319         $1_OPT_CXXFLAGS := $(CXX_O_FLAG_NONE)
 320       else ifeq ($$($1_OPTIMIZATION), LOW)
 321         $1_OPT_CFLAGS := $(C_O_FLAG_NORM)
 322         $1_OPT_CXXFLAGS := $(CXX_O_FLAG_NORM)
 323       else ifeq ($$($1_OPTIMIZATION), HIGH)
 324         $1_OPT_CFLAGS := $(C_O_FLAG_HI)
 325         $1_OPT_CXXFLAGS := $(CXX_O_FLAG_HI)
 326       else ifeq ($$($1_OPTIMIZATION), HIGHEST)
 327         $1_OPT_CFLAGS := $(C_O_FLAG_HIGHEST)
 328         $1_OPT_CXXFLAGS := $(CXX_O_FLAG_HIGHEST)
 329       else ifeq ($$($1_OPTIMIZATION), HIGHEST_JVM)
 330         $1_OPT_CFLAGS := $(C_O_FLAG_HIGHEST_JVM)
 331         $1_OPT_CXXFLAGS := $(CXX_O_FLAG_HIGHEST_JVM)
 332       else ifeq ($$($1_OPTIMIZATION), SIZE)
 333         $1_OPT_CFLAGS := $(C_O_FLAG_SIZE)
 334         $1_OPT_CXXFLAGS := $(CXX_O_FLAG_SIZE)
 335       else
 336         $$(error Unknown value for file OPTIMIZATION: $$($1_OPTIMIZATION))
 337       endif
 338     endif
 339 
 340     ifneq ($$($$($1_BASE)_PRECOMPILED_HEADER), )
 341       ifeq ($$(filter $$($1_FILENAME), $$($$($1_BASE)_PRECOMPILED_HEADER_EXCLUDE)), )
 342         $1_USE_PCH_FLAGS := $$($$($1_BASE)_USE_PCH_FLAGS)
 343       endif
 344     endif
 345 
 346     $1_BASE_CFLAGS :=  $$($$($1_BASE)_CFLAGS) $$($$($1_BASE)_EXTRA_CFLAGS) \
 347         $$($$($1_BASE)_SYSROOT_CFLAGS)
 348     $1_BASE_CXXFLAGS := $$($$($1_BASE)_CXXFLAGS) $$($$($1_BASE)_EXTRA_CXXFLAGS) \
 349         $$($$($1_BASE)_SYSROOT_CFLAGS) $$($1_EXTRA_CXXFLAGS)
 350     $1_BASE_ASFLAGS := $$($$($1_BASE)_ASFLAGS) $$($$($1_BASE)_EXTRA_ASFLAGS)
 351 
 352     ifneq ($$(filter %.c, $$($1_FILENAME)), )
 353       # Compile as a C file
 354       $1_FLAGS := $(CFLAGS_CCACHE) $$($1_USE_PCH_FLAGS) $$($1_BASE_CFLAGS) \
 355           $$($1_OPT_CFLAGS) $$($1_CFLAGS) -c
 356       $1_COMPILER := $$($$($1_BASE)_CC)
 357       $1_DEP_FLAG := $(C_FLAG_DEPS)
 358     else ifneq ($$(filter %.m, $$($1_FILENAME)), )
 359       # Compile as an Objective-C file
 360       $1_FLAGS := -x objective-c $(CFLAGS_CCACHE) $$($1_USE_PCH_FLAGS) \
 361           $$($1_BASE_CFLAGS) $$($1_OPT_CFLAGS) $$($1_CFLAGS) -c
 362       $1_COMPILER := $$($$($1_BASE)_CC)
 363       $1_DEP_FLAG := $(C_FLAG_DEPS)
 364     else ifneq ($$(filter %.s %.S, $$($1_FILENAME)), )
 365       # Compile as assembler file
 366       $1_FLAGS := $$($1_BASE_ASFLAGS)
 367       $1_COMPILER := $(AS)
 368       $1_DEP_FLAG :=
 369     else ifneq ($$(filter %.cpp %.cc %.mm, $$($1_FILENAME)), )
 370       # Compile as a C++ or Objective-C++ file
 371       $1_FLAGS := $(CFLAGS_CCACHE) $$($1_USE_PCH_FLAGS) $$($1_BASE_CXXFLAGS) \
 372           $$($1_OPT_CXXFLAGS) $$($1_CXXFLAGS) -c
 373       $1_COMPILER := $$($$($1_BASE)_CXX)
 374       $1_DEP_FLAG := $(CXX_FLAG_DEPS)
 375     else
 376       $$(error Internal error in NativeCompilation.gmk: no compiler for file $$($1_FILENAME))
 377     endif
 378 
 379     ifeq ($$(filter %.s %.S, $$($1_FILENAME)), )
 380       # And this is the dependency file for this obj file.
 381       $1_DEPS_FILE := $$(patsubst %$(OBJ_SUFFIX),%.d,$$($1_OBJ))
 382       # The dependency target file lists all dependencies as empty targets to
 383       # avoid make error "No rule to make target" for removed files
 384       $1_DEPS_TARGETS_FILE := $$(patsubst %$(OBJ_SUFFIX),%.d.targets,$$($1_OBJ))
 385 
 386       # Only try to load individual dependency information files if the global
 387       # file hasn't been loaded (could happen if make was interrupted).
 388       ifneq ($$($$($1_BASE)_DEPS_FILE_LOADED), true)
 389         # Include previously generated dependency information. (if it exists)
 390         -include $$($1_DEPS_FILE)
 391         -include $$($1_DEPS_TARGETS_FILE)
 392       endif
 393     endif
 394 
 395     ifneq ($$(strip $$($1_CFLAGS) $$($1_CXXFLAGS) $$($1_OPTIMIZATION)), )
 396       $1_VARDEPS := $$($1_CFLAGS) $$($1_CXXFLAGS) $$($1_OPTIMIZATION)
 397       $1_VARDEPS_FILE := $$(call DependOnVariable, $1_VARDEPS, $$($1_OBJ).vardeps)
 398     endif
 399 
 400     $1_OBJ_DEPS := $$($1_SRC_FILE) $$($$($1_BASE)_COMPILE_VARDEPS_FILE) \
 401         $$($$($1_BASE)_EXTRA_DEPS) $$($1_VARDEPS_FILE)
 402     $1_COMPILE_OPTIONS := $$($1_FLAGS) $(CC_OUT_OPTION)$$($1_OBJ) $$($1_SRC_FILE)
 403 
 404     $$($1_OBJ_JSON): $$($1_OBJ_DEPS)
 405 	$$(call WriteCompileCommandsFragment, $$@, $$(PWD), $$($1_SRC_FILE), \
 406 	    $$($1_COMPILER) $$($1_COMPILE_OPTIONS))
 407 
 408     $$($1_OBJ): $$($1_OBJ_DEPS) | $$($$($1_BASE)_BUILD_INFO)
 409 	$$(call LogInfo, Compiling $$($1_FILENAME) (for $$($$($1_BASE)_BASENAME)))
 410 	$$(call MakeDir, $$(@D))
 411         ifneq ($(TOOLCHAIN_TYPE), microsoft)
 412 	  $$(call ExecuteWithLog, $$@, $$(call MakeCommandRelative, \
 413 	      $$($1_COMPILER) $$($1_DEP_FLAG) \
 414 	      $$(addsuffix .tmp, $$($1_DEPS_FILE)) \
 415 	      $$($1_COMPILE_OPTIONS)))
 416           ifneq ($$($1_DEPS_FILE), )
 417 	    $$(call fix-deps-file, $$($1_DEPS_FILE))
 418             # Create a dependency target file from the dependency file.
 419             # Solution suggested by:
 420             # http://make.mad-scientist.net/papers/advanced-auto-dependency-generation/
 421 	    $(SED) $(DEPENDENCY_TARGET_SED_PATTERN) $$($1_DEPS_FILE) \
 422 	        > $$($1_DEPS_TARGETS_FILE)
 423           endif
 424         else
 425           # The Visual Studio compiler lacks a feature for generating make
 426           # dependencies, but by setting -showIncludes, all included files are
 427           # printed. These are filtered out and parsed into make dependences.
 428           #
 429           # Keep as much as possible on one execution line for best performance
 430           # on Windows. No need to save exit code from compilation since
 431           # pipefail is always active on Windows.
 432 	  $$(call ExecuteWithLog, $$@, $$(call MakeCommandRelative, \
 433 	      $$($1_COMPILER) -showIncludes $$($1_COMPILE_OPTIONS))) \
 434 	      | $(TR) -d '\r' | $(GREP) -v -e "^Note: including file:" \
 435 	          -e "^$$($1_FILENAME)$$$$" || test "$$$$?" = "1" ; \
 436 	  $(ECHO) $$@: \\ > $$($1_DEPS_FILE) ; \
 437 	  $(SED) $(WINDOWS_SHOWINCLUDE_SED_PATTERN) $$($1_OBJ).log \
 438 	      | $(SORT) -u >> $$($1_DEPS_FILE) ; \
 439 	  $(ECHO) >> $$($1_DEPS_FILE) ; \
 440 	  $(SED) $(DEPENDENCY_TARGET_SED_PATTERN) $$($1_DEPS_FILE) > $$($1_DEPS_TARGETS_FILE)
 441         endif
 442   endif
 443 endef

 445 # Setup make rules for creating a native binary (a shared library or an
 446 # executable).
 447 #
 448 # Parameter 1 is the name of the rule. This name is used as variable prefix,
 449 # and the targets generated are listed in a variable by that name.
 450 #
 451 # Remaining parameters are named arguments. These include:
 452 #   NAME The base name for the resulting binary, excluding decorations (like *.exe)
 453 #   TYPE Type of binary (EXECUTABLE, LIBRARY or STATIC_LIBRARY). Default is LIBRARY.
 454 #   SUFFIX Override the default suffix for the output file
 455 #   TOOLCHAIN Name of toolchain setup to use. Defaults to TOOLCHAIN_DEFAULT.
 456 #   SRC one or more directory roots to scan for C/C++ files.
 457 #   CFLAGS the compiler flags to be used, used both for C and C++.
 458 #   CXXFLAGS the compiler flags to be used for c++, if set overrides CFLAGS.
 459 #   LDFLAGS the linker flags to be used, used both for C and C++.
 460 #   LIBS the libraries to link to
 461 #   ARFLAGS the archiver flags to be used
 462 #   OBJECT_DIR the directory where we store the object files
 463 #   OUTPUT_DIR the directory where the resulting binary is put
 464 #   SYMBOLS_DIR the directory where the debug symbols are put, defaults to OUTPUT_DIR
 465 #   INCLUDES only pick source from these directories
 466 #   EXCLUDES do not pick source from these directories
 467 #   INCLUDE_FILES only compile exactly these files!
 468 #   EXCLUDE_FILES with these names
 469 #   EXCLUDE_PATTERN exclude files matching any of these substrings
 470 #   EXTRA_FILES List of extra files not in any of the SRC dirs
 471 #   EXTRA_OBJECT_FILES List of extra object files to include when linking
 472 #   EXTRA_DEPS List of extra dependencies to be added to each compiled file
 473 #   VERSIONINFO_RESOURCE Input file for RC. Setting this implies that RC will be run
 474 #   RC_FLAGS flags for RC.
 475 #   EMBED_MANIFEST if true, embed manifest on Windows.
 476 #   MAPFILE mapfile
 477 #   REORDER reorder file
 478 #   USE_MAPFILE_FOR_SYMBOLS if true and this is a STATIC_BUILD, just copy the
 479 #       mapfile for the output symbols file
 480 #   CC the compiler to use, default is $(CC)
 481 #   LD the linker to use, default is $(LD)
 482 #   OPTIMIZATION sets optimization level to NONE, LOW, HIGH, HIGHEST, HIGHEST_JVM, SIZE
 483 #   DISABLED_WARNINGS_<toolchain> Disable the given warnings for the specified toolchain
 484 #   DISABLED_WARNINGS_C_<toolchain> Disable the given warnings for the specified toolchain
 485 #       when compiling C code
 486 #   DISABLED_WARNINGS_CXX_<toolchain> Disable the given warnings for the specified
 487 #       toolchain when compiling C++ code
 488 #   STRIP_SYMBOLS Set to false to override global strip policy and always leave
 489 #       symbols in the binary, if the toolchain allows for it
 490 #   DEBUG_SYMBOLS Set to false to disable generation of debug symbols
 491 #   COPY_DEBUG_SYMBOLS Set to false to override global setting of debug symbol copying
 492 #   ZIP_EXTERNAL_DEBUG_SYMBOLS Set to false to override global setting of debug symbol
 493 #       zipping
 494 #   STRIPFLAGS Optionally change the flags given to the strip command
 495 #   PRECOMPILED_HEADER Header file to use as precompiled header
 496 #   PRECOMPILED_HEADER_EXCLUDE List of source files that should not use PCH
 497 #
 498 # After being called, some variables are exported from this macro, all prefixed
 499 # with parameter 1 followed by a '_':
 500 #   TARGET The library or executable created by the macro
 501 #   TARGET_DEPS All prerequisites for the target calculated by the macro
 502 #   ALL_OBJS All object files
 503 #   IMPORT_LIBRARY The import library created for a shared library on Windows
 504 SetupNativeCompilation = $(NamedParamsMacroTemplate)
 505 define SetupNativeCompilationBody
