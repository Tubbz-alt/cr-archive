 26 package jdk.jfr.api.consumer.security;
 27 
 28 import java.io.File;
 29 import java.io.IOException;
 30 import java.io.PrintWriter;
 31 import java.nio.file.Files;
 32 import java.nio.file.Path;
 33 import java.nio.file.Paths;

 34 import java.time.Instant;
 35 
 36 import jdk.jfr.Event;
 37 import jdk.jfr.Recording;
 38 import jdk.jfr.consumer.EventStream;
 39 import jdk.test.lib.process.OutputAnalyzer;
 40 import jdk.test.lib.process.ProcessTools;
 41 
 42 /**
 43  * @test
 44  * @summary Test that a stream can be opened against a remote repository using
 45  *          only file permission
 46  * @key jfr
 47  * @requires vm.hasJFR
 48  * @library /test/lib
 49  *
 50  * @run main/othervm jdk.jfr.api.consumer.security.TestStreamingRemote
 51  */
 52 public class TestStreamingRemote {
 53 
 54     private static final String SUCCESS = "Success!";
 55 
 56     public static class TestEvent extends Event {
 57     }
 58 
 59     public static class Test {
 60         public static void main(String... args) throws Exception {
 61             Path repo = Paths.get(args[0]);
 62             System.out.println("Repository: " + repo);
 63             try (EventStream es = EventStream.openRepository(repo)) {
 64                 es.setStartTime(Instant.EPOCH);
 65                 es.onEvent(e -> {
 66                     System.out.println(SUCCESS);
 67                     es.close();
 68                 });
 69                 es.start();
 70             }
 71         }
 72     }
 73 
 74     public static void main(String... args) throws Exception {
 75         try (Recording r = new Recording()) {
 76             r.start();
 77             String repository = System.getProperty("jdk.jfr.repository");
 78             Path policy = createPolicyFile(repository);
 79             TestEvent e = new TestEvent();
 80             e.commit();
 81             String[] c = new String[4];
 82             c[0] = "-Djava.security.manager";
 83             c[1] = "-Djava.security.policy=" + escapeBackslashes(policy.toString());
 84             c[2] = Test.class.getName();
 85             c[3] = repository;
 86             OutputAnalyzer oa = ProcessTools.executeTestJvm(c);
 87             oa.shouldContain(SUCCESS);
 88         }
 89     }
 90 
 91     private static Path createPolicyFile(String repository) throws IOException {
 92         Path policy = Paths.get("permission.policy").toAbsolutePath();
 93         try (PrintWriter pw = new PrintWriter(policy.toFile())) {
 94             pw.println("grant {");
 95             // All the files and directories the contained in path
 96             String dir = escapeBackslashes(repository);
 97             String contents = escapeBackslashes(repository + File.separatorChar + "-");
 98             pw.println("  permission java.io.FilePermission \"" + dir + "\", \"read\";");
 99             pw.println("  permission java.io.FilePermission \"" + contents + "\", \"read\";");
100             pw.println("};");
101             pw.println();
102         }
103         System.out.println("Permission file: " + policy);
104         for (String line : Files.readAllLines(policy)) {
105             System.out.println(line);
106         }
107         System.out.println();
108         return policy;
109     }
110 
111     // Needed for Windows
112     private static String escapeBackslashes(String text) {
113         return text.replace("\\", "\\\\");
114     }
115 }
