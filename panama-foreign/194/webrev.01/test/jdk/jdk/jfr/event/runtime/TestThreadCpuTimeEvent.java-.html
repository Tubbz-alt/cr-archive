 26 package jdk.jfr.event.runtime;
 27 
 28 import jdk.jfr.Recording;
 29 import jdk.jfr.consumer.RecordedEvent;
 30 import jdk.jfr.consumer.RecordedThread;
 31 import jdk.test.lib.Asserts;
 32 import jdk.test.lib.jfr.EventNames;
 33 import jdk.test.lib.jfr.Events;

 35 import java.lang.management.ManagementFactory;
 36 import java.time.Duration;
 37 import java.time.Instant;
 38 import java.util.Comparator;
 39 import java.util.Iterator;
 40 import java.util.List;
 41 import java.util.concurrent.BrokenBarrierException;
 42 import java.util.concurrent.CyclicBarrier;
 43 import java.util.stream.Collectors;

 45 /**
 46  * @test
 47  * @key jfr
 48  * @requires vm.hasJFR
 49  * @library /test/lib
 50  * @modules jdk.jfr
 51  *          jdk.management
 52  *
 53  * @run main/othervm jdk.jfr.event.runtime.TestThreadCpuTimeEvent
 54  */
 55 
 56 /**
 57  */
 58 public class TestThreadCpuTimeEvent {
 59 
 60     public static void main(String[] args) throws Throwable {
 61         testSimple();
 62         testEventAtThreadExit();
 63     }
 64 
 65     private static final long eventPeriodMillis = 50;
 66     private static final String cpuConsumerThreadName = "cpuConsumer";
 67 
 68     // The cpu consumer will run for eventPeriodMillis times this factor to ensure that we see some
 69     // events even if the scheduler isn't cooperating.
 70     private static final long cpuConsumerRunFactor = 10;
 71 
 72     // The cpu consumer will run at least this number of loops, even if it takes longer than
 73     // the requested period of time (in case the thread didn't get scheduled within the allotted time).
 74     private static final long cpuConsumerMinCount = 1000000;
 75 
 76     static class CpuConsumingThread extends Thread {
 77 
 78         Duration runTime;
 79         CyclicBarrier barrier;
 80         volatile long counter;
 81 
 82         CpuConsumingThread(Duration runTime, CyclicBarrier barrier, String threadName) {
 83             super(threadName);
 84             this.runTime = runTime;
 85             this.barrier = barrier;
 86         }
 87 
 88         CpuConsumingThread(Duration runTime, CyclicBarrier barrier) {
 89             this(runTime, barrier, cpuConsumerThreadName);
 90         }
 91 
 92         @Override
 93         public void run() {
 94             try {
 95                 while (true) {
 96                     barrier.await();
 97                     Instant start = Instant.now();
 98                     counter = 0;
 99                     while ((Duration.between(start, Instant.now()).compareTo(runTime) < 0) ||
100                             (counter < cpuConsumerMinCount)) {
101                         counter++;
102                     }
103                     barrier.await();
104                 }
105             } catch (BrokenBarrierException e) {
106                 // Another thread has been interrupted - wait for us to be interrupted as well
107                 while (!interrupted()) {
108                     Thread.yield();
109                 }
110             } catch (InterruptedException e) {
111                 // Normal way of stopping the thread
112             }
113         }
114     }
115 
116     // For a given thread, check that accumulated processTime >= cpuTime >= userTime.
117     // This may not hold for a single event instance due to differences in counter resolution
118     static void verifyPerThreadInvariant(List<RecordedEvent> events, String threadName) {
119         List<RecordedEvent> filteredEvents = events.stream()
120                 .filter(e -> e.getThread().getJavaName().equals(threadName))
121                 .sorted(Comparator.comparing(RecordedEvent::getStartTime))
122                 .collect(Collectors.toList());
123 
124         int numCpus = Runtime.getRuntime().availableProcessors();
125         Iterator<RecordedEvent> i = filteredEvents.iterator();
126         while (i.hasNext()) {
127             RecordedEvent event = i.next();
128 
129             Float systemLoad = (Float)event.getValue("system");
130             Float userLoad = (Float)event.getValue("user");
131 
132             Asserts.assertLessThan(systemLoad + userLoad, 1.01f / numCpus); // 100% + rounding errors
133         }
134     }
135 
136     static List<RecordedEvent> generateEvents(int minimumEventCount, CyclicBarrier barrier) throws Throwable {
137         int retryCount = 0;
138 
139         while (true) {
140             Recording recording = new Recording();
141 
142             // Default period is once per chunk
143             recording.enable(EventNames.ThreadCPULoad).withPeriod(Duration.ofMillis(eventPeriodMillis));
144             recording.start();
145 
146             // Run a single pass
147             barrier.await();
148             barrier.await();
149 
150             recording.stop();
151             List<RecordedEvent> events = Events.fromRecording(recording);
152 
153             long numEvents = events.stream()
154                     .filter(e -> e.getThread().getJavaName().equals(cpuConsumerThreadName))
155                     .count();
156 
157             // If the JFR periodicals thread is really starved, we may not get enough events.
158             // In that case, we simply retry the operation.
159             if (numEvents < minimumEventCount) {
160                 System.out.println("Not enough events recorded, trying again...");
161                 if (retryCount++ > 10) {
162                     Asserts.fail("Retry count exceeded");
163                     throw new RuntimeException();
164                 }
165             } else {
166                 return events;
167             }
168         }
169     }
170 
171     static void testSimple() throws Throwable {
172         Duration testRunTime = Duration.ofMillis(eventPeriodMillis * cpuConsumerRunFactor);
173         CyclicBarrier barrier = new CyclicBarrier(2);
174         CpuConsumingThread thread = new CpuConsumingThread(testRunTime, barrier);
175         thread.start();
176 
177         List<RecordedEvent> events = generateEvents(1, barrier);
178         verifyPerThreadInvariant(events, cpuConsumerThreadName);
179 
180         thread.interrupt();
181         thread.join();
182     }
183 
184     static void testEventAtThreadExit() throws Throwable {
185         Recording recording = new Recording();
186 
187         recording.enable(EventNames.ThreadCPULoad).withPeriod(Duration.ofHours(10));
188         recording.start();
189 
190         Duration testRunTime = Duration.ofMillis(eventPeriodMillis * cpuConsumerRunFactor);
191         CyclicBarrier barrier = new CyclicBarrier(2);
192         CpuConsumingThread thread = new CpuConsumingThread(testRunTime, barrier);
193 
194         // Run a single pass
195         thread.start();
196         barrier.await();
197         barrier.await();
198 
199         thread.interrupt();
200         thread.join();
201 
202         recording.stop();
203 
204         List<RecordedEvent> events = Events.fromRecording(recording);
205         verifyPerThreadInvariant(events, cpuConsumerThreadName);
206 
207         int exitingCount = 0;
208         for (RecordedEvent event : events) {
209             RecordedThread eventThread = event.getThread();
210             if (eventThread.getJavaName().equals(cpuConsumerThreadName)) {
211                 exitingCount++;
212             }
213         }
214         Asserts.assertEquals(exitingCount, 1);
215     }
