 25 package jdk.jfr.event.runtime;

 27 import java.io.IOException;
 28 import java.lang.instrument.ClassFileTransformer;
 29 import java.lang.instrument.IllegalClassFormatException;
 30 import java.lang.instrument.Instrumentation;
 31 import java.nio.file.Path;
 32 import java.nio.file.Paths;
 33 import java.security.ProtectionDomain;
 34 import java.util.List;
 35 
 36 import jdk.jfr.Recording;
 37 import jdk.jfr.consumer.RecordedEvent;
 38 import jdk.jfr.consumer.RecordingFile;
 39 import jdk.test.lib.Asserts;
 40 import jdk.test.lib.jfr.EventNames;
 41 import jdk.test.lib.jfr.Events;
 42 
 43 /**
 44  * @test
 45  * @summary Tests the RetransformClasses event by redefining a class in a Java
 46  *          agent
 47  * @key jfr
 48  * @requires vm.hasJFR
 49  * @library /test/lib /test/jdk
 50  * @modules java.instrument
 51  *
 52  * @build jdk.jfr.event.runtime.RedefinableClass
 53  * @build jdk.jfr.event.runtime.Bytes
 54  * @build jdk.jfr.event.runtime.TestRetransformClasses
 55  *
 56  * @run driver jdk.test.lib.util.JavaAgentBuilder
 57  *      jdk.jfr.event.runtime.TestRetransformClasses TestRetransformClasses.jar
 58  *
 59  * @run main/othervm -javaagent:TestRetransformClasses.jar
 60  *      jdk.jfr.event.runtime.TestRetransformClasses
 61  */
 62 public class TestRetransformClasses {
 63     private final static Path DUMP_PATH = Paths.get("dump.jfr");
 64     private final static String TEST_AGENT = "Test Agent";
 65 
 66     public static class TestClassFileTransformer implements ClassFileTransformer {
 67         public byte[] transform(ClassLoader loader, String className, Class<?> classBeingRedefined, ProtectionDomain protectionDomain, byte[] classfileBuffer) throws IllegalClassFormatException {
 68             return Bytes.replaceAll(classfileBuffer, Bytes.WORLD, Bytes.EARTH);
 69         }
 70     }
 71 
 72     // Called when agent is loaded at startup
 73     public static void premain(String agentArgs, Instrumentation instrumentation) throws Exception {
 74         Thread.currentThread().setName(TEST_AGENT);
 75         try (Recording r = new Recording()) {
 76             r.enable(EventNames.RetransformClasses);
 77             r.start();
 78             RedefinableClass.sayHello();
 79             instrumentation.addTransformer(new TestClassFileTransformer());
 80             instrumentation.retransformClasses(RedefinableClass.class);
 81             RedefinableClass.sayHello();
 82             r.stop();
 83             r.dump(DUMP_PATH);
 84         }
 85     }
 86 
 87     public static void main(String[] args) throws Throwable {
 88         List<RecordedEvent> events = RecordingFile.readAllEvents(DUMP_PATH);
 89         Asserts.assertEquals(events.size(), 1, "Expected one RetransformClasses event");
 90         RecordedEvent event = events.get(0);
 91 
 92         System.out.println(event);
 93 
 94         Events.assertField(event, "eventThread.javaName").equal(TEST_AGENT);
 95         Events.assertField(event, "classCount").equal(1);
 96         Events.assertField(event, "redefinitionId").atLeast(1L);
 97         Events.assertField(event, "duration").atLeast(1L);
 98         Events.assertFrame(event, TestRetransformClasses.class, "premain");
 99     }
100 }
