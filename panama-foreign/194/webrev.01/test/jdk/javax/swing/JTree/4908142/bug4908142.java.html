 25 /*
 26  * @test
 27  * @key headful
 28  * @bug 4908142
 29  * @summary JList doesn't handle search function appropriately
 30  * @author Andrey Pikalev
 31  * @library ../../regtesthelpers
 32  * @build Util
 33  * @run main bug4908142
 34  */
 35 import javax.swing.JFrame;
 36 import javax.swing.JScrollPane;
 37 import javax.swing.JTree;
 38 import javax.swing.SwingUtilities;
 39 import javax.swing.tree.DefaultMutableTreeNode;
 40 import java.awt.Robot;
 41 import java.awt.event.KeyEvent;
 42 import java.util.concurrent.Callable;
 43 
 44 public class bug4908142 {
 45 
 46     private static JFrame fr = null;
 47     private static JTree tree = null;
 48 
 49     public static void main(String[] args) throws Exception {
 50 
 51         Robot robot = new Robot();
 52         robot.setAutoDelay(50);
 53 
 54         try {
 55             SwingUtilities.invokeAndWait(new Runnable() {
 56 
 57                 public void run() {
 58                     createAndShowGUI();
 59                 }
 60             });
 61 
 62             robot.waitForIdle();
 63 
 64             SwingUtilities.invokeAndWait(new Runnable() {
 65 
 66                 public void run() {
 67                     tree.requestFocus();
 68                     tree.setSelectionRow(0);
 69                 }
 70             });
 71 
 72             robot.waitForIdle();
 73 
 74 
 75             robot.keyPress(KeyEvent.VK_A);
 76             robot.keyRelease(KeyEvent.VK_A);
 77             robot.keyPress(KeyEvent.VK_A);
 78             robot.keyRelease(KeyEvent.VK_A);
 79             robot.keyPress(KeyEvent.VK_D);
 80             robot.keyRelease(KeyEvent.VK_D);
 81             robot.waitForIdle();
 82 
 83 
 84             String sel = Util.invokeOnEDT(new Callable<String>() {
 85 
 86                 @Override
 87                 public String call() throws Exception {
 88                     return tree.getLastSelectedPathComponent().toString();
 89                 }
 90             });
 91 
 92             if (!"aad".equals(sel)) {
 93                 throw new Error("The selected index should be \"aad\", but not " + sel);
 94             }
 95         } finally {
 96             if (fr != null) {
 97                 SwingUtilities.invokeAndWait(fr::dispose);
 98             }
 99         }
100     }
101 
102     private static void createAndShowGUI() {
103         fr = new JFrame("Test");
104         fr.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
105 
106         String[] data = {"aaa", "aab", "aac", "aad", "ade", "bba"};
107         final DefaultMutableTreeNode root = new DefaultMutableTreeNode(data[0]);
108         for (int i = 1; i < data.length; i++) {
109             DefaultMutableTreeNode node = new DefaultMutableTreeNode(data[i]);
110             root.add(node);
111         }
112 
113         tree = new JTree(root);
114 
115         JScrollPane sp = new JScrollPane(tree);
116         fr.getContentPane().add(sp);
117         fr.setSize(200, 200);
118         fr.setVisible(true);
119     }
120 }
