31 // Test that a method in a sealed class or interface can be overridden.
32 public class OverrideSealedTest {
33 
34     sealed class Rectangle permits Square {
35         public String draw() { return "rectangle"; }
36     }
37 
38     final class Square extends Rectangle {
39         public String draw() { return "square"; }
40     }
41 
42     Rectangle r = new Rectangle();
43     Rectangle rs = new Square();
44     Square s = new Square();
45 
46 
47     public sealed interface Shape permits Circle {
48         default String name() { return "shape"; }
49     }
50 
51     final class Circle implements Shape {
52         public String name() { return "circle"; }
53     }
54 
55     Shape sc = new Circle();
56     Circle c = new Circle();
57 
58 
59     public static void main(String... args) {
60         OverrideSealedTest ost = new OverrideSealedTest();
61         if (ost.r.draw() != "rectangle")
62             throw new RuntimeException("Bad value returned by draw(): " + ost.r.draw());
63         if (ost.rs.draw() != "square")
64             throw new RuntimeException("Bad value returned by draw(): " + ost.rs.draw());
65         if (ost.s.draw() != "square")
66             throw new RuntimeException("Bad value returned by draw(): " + ost.s.draw());
67 
68         if (ost.sc.name() != "circle")
69             throw new RuntimeException("Bad value returned by name(): " + ost.sc.name());
70         if (ost.c.name() != "circle")
71             throw new RuntimeException("Bad value returned by name(): " + ost.c.name());
72     }
73 }
