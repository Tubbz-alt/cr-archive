25 /*
26  * @test
27  * @summary Test how the dynamic archive handles -XX:SharedBaseAddress
28  * @requires vm.cds
29  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds /test/hotspot/jtreg/runtime/cds/appcds/test-classes
30  * @build Hello
31  * @build sun.hotspot.WhiteBox
32  * @run driver ClassFileInstaller -jar hello.jar Hello
33  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
34  * @run main/othervm -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:. SharedBaseAddressOption
35  */
36 
37 import jdk.test.lib.Platform;
38 
39 public class SharedBaseAddressOption extends DynamicArchiveTestBase {
40     static String appJar = ClassFileInstaller.getJarPath("hello.jar");
41     static String mainClass = "Hello";
42 
43     public static void main(String[] args) throws Exception {
44         runTest(SharedBaseAddressOption::testCustomBase);
45     }
46 
47     static void testCustomBase() throws Exception {
48 
49         // (1) -XX:SharedBaseAddress=0 -- the archives will always be relocated at runtime
50         doTest("0");
51 
52         // (2) -XX:SharedBaseAddress=0x810000000
53         if (Platform.is64bit()) {
54             doTest("0x810000000");
55         }
56 
57         // (3) -XX:SharedBaseAddress that's so high that the archive may wrap around 64-bit
58         if (Platform.is64bit()) {
59             doTest("0xfffffffffff00000");
60         }
61     }
62 
63     static void doTest(String sharedBase) throws Exception {
64         String baseArchiveName = getNewArchiveName("base");
65         String topArchiveName = getNewArchiveName("top");
66 
67         TestCommon.dumpBaseArchive(baseArchiveName, "-XX:SharedBaseAddress=" + sharedBase,
68                         "-Xlog:cds=debug",
69                         "-Xlog:cds+reloc=debug",
70                         "-Xlog:cds+dynamic=debug");
71 
72         dump2(baseArchiveName, topArchiveName,
73               "-Xlog:cds=debug",
74               "-Xlog:cds+reloc=debug",
75               "-Xlog:cds+dynamic=debug",
76               "-cp", appJar, mainClass)
77             .assertNormalExit();
78 
79         // a top archive specified in the base archive position
80         run2(baseArchiveName, topArchiveName,
81              "-Xlog:cds=debug",
82              "-Xlog:cds+reloc=debug",
83              "-Xlog:cds+dynamic=debug",
84              "-cp", appJar, mainClass)
85             .assertNormalExit();
86     }
87 }
