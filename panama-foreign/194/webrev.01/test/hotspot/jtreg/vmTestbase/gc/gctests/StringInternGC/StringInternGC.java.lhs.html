38 package gc.gctests.StringInternGC;
39 
40 import nsk.share.test.*;
41 import nsk.share.gc.*;
42 
43 /**
44  * Test that strings returned by String.intern() can be collected.
45  *
46  * Create strings consisting of random characters, call String.intern().
47  * String pool should not overflow.
48  */
49 public class StringInternGC extends ThreadedGCTest {
50         private int maxLength = 1000;
51 
52         private class StringGenerator implements Runnable {
53                 private StringBuffer sb = new StringBuffer();
54 
55                 private String generateString() {
56                         int length = LocalRandom.nextInt(maxLength);
57                         for (int i = 0; i < length; ++i)
58                                 sb.append((char) LocalRandom.nextInt(Integer.MAX_VALUE));
59                         return sb.toString();
60                 }
61 
62 
63                 public void run() {
64                         generateString().intern();
65                 }
66         }
67 
68         protected Runnable createRunnable(int i) {
69                 return new StringGenerator();
70         }
71 
72         public static void main(String[] args) {
73                 GC.runTest(new StringInternGC(), args);
74         }
75 }
