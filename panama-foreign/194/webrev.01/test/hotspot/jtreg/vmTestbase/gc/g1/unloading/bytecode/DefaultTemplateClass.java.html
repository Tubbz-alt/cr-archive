23 package gc.g1.unloading.bytecode;
24 
25 import java.util.Random;
26 
27 /**
28  * This is template class. Loaded classes will derive from it.
29  */
30 public class DefaultTemplateClass {
31 
32     public static void main() {
33         System.out.println("In method of generated class. Random.nextDouble =  " + new Random(42).nextDouble());
34         System.out.println(" Printing bytesToReplace0 bytesToReplace2");
35     }
36 
37     public static long field;
38 
39     public static void methodForCompilation(Object object) {
40         int i = object.hashCode();
41         i = i * 2000 / 1994 + 153;
42         field = i;
43     }
44 
45 }
