38 package gc.gctests.CallGC.CallGC02;
39 
40 import nsk.share.gc.*;
41 import nsk.share.gc.gp.*;
42 import nsk.share.runner.*;
43 
44 /**
45  * This test starts a number of threads that do System.gc() and
46  * System.runFinalization() and checks that there are no crashes.
47  *
48  * There is also a thread that produces garbage.
49  */
50 public class CallGC02 extends ThreadedGCTest implements GarbageProducerAware {
51         private GarbageProducer garbageProducer;
52         private final int objectSize = 100;
53 
54         private class GarbageProduction implements Runnable {
55                 public void run() {
56                         garbageProducer.create(objectSize);
57                 }
58         }
59 
60         protected Runnable createRunnable(int i) {
61                 if (i == 0)
62                         return new GarbageProduction();
63                 else if (i % 2 == 0)
64                         return new GCRunner();
65                 else
66                         return new FinRunner();
67         }
68 
69         public final void setGarbageProducer(GarbageProducer garbageProducer) {
70                 this.garbageProducer = garbageProducer;
71         }
72 
73         public static void main(String[] args) {
74                 GC.runTest(new CallGC02(), args);
75         }
76 }
