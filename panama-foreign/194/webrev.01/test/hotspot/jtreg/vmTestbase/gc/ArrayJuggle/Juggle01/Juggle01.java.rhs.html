38 package gc.ArrayJuggle.Juggle01;
39 
40 import nsk.share.test.*;
41 import nsk.share.gc.*;
42 import nsk.share.gc.gp.*;
43 
44 /**
45  * This test randomly replaces elements of an array with new
46  * objects using given garbage producer and memory strategy.
47  */
48 public class Juggle01 extends ThreadedGCTest implements GarbageProducerAware, MemoryStrategyAware {
49         private GarbageProducer garbageProducer;
50         private MemoryStrategy memoryStrategy;
51         private Object[] array;
52         long objectSize;
53 
54         private class Juggler implements Runnable {
55                 public void run() {
56                         synchronized (this) {
57                                 int index = LocalRandom.nextInt(array.length);
58                                 array[index] = garbageProducer.create(objectSize);
59                         }
60                 }
61         }
62 
63         protected Runnable createRunnable(int i) {
64                 return new Juggler();
65         }
66 
67         public void run() {
68                 log.debug("Garbage producer: " + garbageProducer);
69                 log.debug("Memory strategy: " + memoryStrategy);
70                 long memory = runParams.getTestMemory();
71                 int objectCount = memoryStrategy.getCount(memory);
72                 objectSize = memoryStrategy.getSize(memory);
73                 log.debug("Object count: " + objectCount);
74                 log.debug("Object size: " + objectSize);
75                 array = new Object[objectCount - 1];
76                 super.run();
77         }
78 
79         public void setGarbageProducer(GarbageProducer garbageProducer) {
80                 this.garbageProducer = garbageProducer;
81         }
82 
83         public void setMemoryStrategy(MemoryStrategy memoryStrategy) {
84                 this.memoryStrategy = memoryStrategy;
85         }
86 
87         public static void main(String[] args) {
88                 GC.runTest(new Juggle01(), args);
89         }
90 }
