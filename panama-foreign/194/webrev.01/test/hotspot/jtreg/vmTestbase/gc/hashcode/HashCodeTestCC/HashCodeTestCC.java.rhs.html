44 package gc.hashcode.HashCodeTestCC;
45 
46 import gc.hashcode.HCHelper;
47 import nsk.share.TestFailure;
48 import nsk.share.gc.GC;
49 import nsk.share.gc.GCTestBase;
50 import nsk.share.gc.gp.GarbageUtils;
51 import nsk.share.test.Stresser;
52 
53 /**
54  * Test that verifies external hash codes. This class tests the scenario
55  * with double compaction.
56  */
57 public class HashCodeTestCC extends GCTestBase {
58 
59 
60     /**
61      * Test external hash codes when two compactions have been performed.
62      *
63      * @return Success if all hash codes matches original hash codes
64      */
65     @Override
66     public final void run() {
67         HCHelper hch = new HCHelper(512, 3584, runParams.getSeed(),
68                 0.7, 10240);
69 
70         hch.setupLists();
71         // Remove evac list 1 to free some space
72         hch.clearList(HCHelper.EVAC_LIST_1);
73         Stresser stresser = new Stresser(runParams.getStressOptions());
74         stresser.start(0);
75         GarbageUtils.eatMemory(stresser);
76         if (!stresser.continueExecution()) {
77             return;// we didn't trigger GC, nothing
78         }
79 
80         // Remove evac list 2 to free some more space
81         hch.clearList(HCHelper.EVAC_LIST_2);
82         GarbageUtils.eatMemory(stresser);
83         if (!stresser.continueExecution()) {
84             return;// we didn't trigger GC, nothing
85         }
86 
87         boolean testResult = hch.verifyHashCodes();
88         hch.cleanupLists();
89 
90          if(!testResult) {
91             throw new TestFailure("Some hash codes didn't match");
92         }
93     }
94 
95     public static void main(String[] args) {
96         GC.runTest(new HashCodeTestCC(), args);
97     }
98 }
