 48 /**
 49  * Template to help speed up your client/server tests.
 50  *

162     /*
163      * =============================================
164      * Define the client and server side operations.
165      *
166      * If the client or server is doing some kind of object creation
167      * that the other side depends on, and that thread prematurely
168      * exits, you may experience a hang.  The test harness will
169      * terminate all hung threads after its timeout has expired,
170      * currently 3 minutes by default, but you might try to be
171      * smart about it....
172      */
173 
174     /*
175      * Is the server ready to serve?
176      */
177     private final CountDownLatch serverCondition = new CountDownLatch(1);
178 
179     /*
180      * Is the client ready to handshake?
181      */
182     private final CountDownLatch clientCondition = new CountDownLatch(1);
183 
184     /*
185      * What's the server port?  Use any free port by default
186      */
187     private volatile int serverPort = 0;
188 
189     /*
190      * Define the server side of the test.
191      */
192     private void doServerSide() throws Exception {
193         // kick start the server side service
194         SSLContext context = createServerSSLContext();
195         SSLServerSocketFactory sslssf = context.getServerSocketFactory();
196         SSLServerSocket sslServerSocket =
197                 (SSLServerSocket)sslssf.createServerSocket(serverPort);
198         configureServerSocket(sslServerSocket);
199         serverPort = sslServerSocket.getLocalPort();
200 
201         // Signal the client, the server is ready to accept connection.
202         serverCondition.countDown();
203 
204         // Try to accept a connection in 30 seconds.
205         SSLSocket sslSocket;
206         try {
207             sslServerSocket.setSoTimeout(30000);
208             sslSocket = (SSLSocket)sslServerSocket.accept();
209         } catch (SocketTimeoutException ste) {
210             // Ignore the test case if no connection within 30 seconds.
211             System.out.println(
212                 "No incoming client connection in 30 seconds. " +
213                 "Ignore in server side.");
214             return;
215         } finally {
216             sslServerSocket.close();
217         }
218 
219         // handle the connection
220         try {
221             // Is it the expected client connection?
222             //
223             // Naughty test cases or third party routines may try to
224             // connection to this server port unintentionally.  In
225             // order to mitigate the impact of unexpected client
226             // connections and avoid intermittent failure, it should
227             // be checked that the accepted connection is really linked
228             // to the expected client.
229             boolean clientIsReady =
230                     clientCondition.await(30L, TimeUnit.SECONDS);
231 
232             if (clientIsReady) {
233                 // Run the application in server side.
234                 runServerApplication(sslSocket);
235             } else {    // Otherwise, ignore
236                 // We don't actually care about plain socket connections
237                 // for TLS communication testing generally.  Just ignore
238                 // the test if the accepted connection is not linked to
239                 // the expected client or the client connection timeout
240                 // in 30 seconds.
241                 System.out.println(
242                         "The client is not the expected one or timeout. " +
243                         "Ignore in server side.");
244             }
245         } finally {
246             sslSocket.close();
247         }
248     }
249 
250     /*
251      * Define the client side of the test.
252      */
253     private void doClientSide() throws Exception {
254 
255         // Wait for server to get started.
256         //
257         // The server side takes care of the issue if the server cannot
258         // get started in 90 seconds.  The client side would just ignore
259         // the test case if the serer is not ready.
260         boolean serverIsReady =
261                 serverCondition.await(90L, TimeUnit.SECONDS);
262         if (!serverIsReady) {
263             System.out.println(
264                     "The server is not ready yet in 90 seconds. " +
265                     "Ignore in client side.");
266             return;
267         }
268 
269         if (isCustomizedClientConnection()) {
270             // Signal the server, the client is ready to communicate.
271             clientCondition.countDown();
272 
273             // Run the application in client side.
274             runClientApplication(serverPort);
275 
276             return;
277         }
278 
279         SSLContext context = createClientSSLContext();
280         SSLSocketFactory sslsf = context.getSocketFactory();
281 
282         try (SSLSocket sslSocket = (SSLSocket)sslsf.createSocket()) {
283             try {
284                 sslSocket.connect(
285                         new InetSocketAddress("localhost", serverPort), 15000);
286             } catch (IOException ioe) {
287                 // The server side may be impacted by naughty test cases or
288                 // third party routines, and cannot accept connections.
289                 //
290                 // Just ignore the test if the connection cannot be
291                 // established.
292                 System.out.println(
293                         "Cannot make a connection in 15 seconds. " +
294                         "Ignore in client side.");
295                 return;
296             }
297 
298             // OK, here the client and server get connected.
299 
300             // Signal the server, the client is ready to communicate.
301             clientCondition.countDown();
302 
303             // There is still a chance in theory that the server thread may
304             // wait client-ready timeout and then quit.  The chance should
305             // be really rare so we don't consider it until it becomes a
306             // real problem.
307 
308             // Run the application in client side.
309             runClientApplication(sslSocket);
310         }
311     }

661 
662             Certificate[] trustedCert =
663                     new Certificate[trustedMaterials.length];
664             for (int i = 0; i < trustedMaterials.length; i++) {
665                 String trustedCertStr = trustedMaterials[i];
666 
667                 is = new ByteArrayInputStream(trustedCertStr.getBytes());
668                 try {
669                     trustedCert[i] = cf.generateCertificate(is);
670                 } finally {
671                     is.close();
672                 }
673 
674                 ts.setCertificateEntry("trusted-cert-" + i, trustedCert[i]);
675             }
676         }
677 
678         // Import the key materials.
679         //
680         // Note that certification pathes bigger than one are not supported yet.
681         boolean hasKeyMaterials =
682             (keyMaterialCerts != null) && (keyMaterialCerts.length != 0) &&
683             (keyMaterialKeys != null) && (keyMaterialKeys.length != 0) &&
684             (keyMaterialKeyAlgs != null) && (keyMaterialKeyAlgs.length != 0) &&
685             (keyMaterialCerts.length == keyMaterialKeys.length) &&
686             (keyMaterialCerts.length == keyMaterialKeyAlgs.length);
687         if (hasKeyMaterials) {
688             ks = KeyStore.getInstance("JKS");
689             ks.load(null, null);
690 
691             for (int i = 0; i < keyMaterialCerts.length; i++) {
692                 String keyCertStr = keyMaterialCerts[i];
693 
694                 // generate the private key.
695                 PKCS8EncodedKeySpec priKeySpec = new PKCS8EncodedKeySpec(
696                     Base64.getMimeDecoder().decode(keyMaterialKeys[i]));
697                 KeyFactory kf =
698                     KeyFactory.getInstance(keyMaterialKeyAlgs[i]);
699                 PrivateKey priKey = kf.generatePrivate(priKeySpec);
700 
701                 // generate certificate chain
702                 is = new ByteArrayInputStream(keyCertStr.getBytes());
703                 Certificate keyCert = null;
704                 try {
705                     keyCert = cf.generateCertificate(is);
706                 } finally {
707                     is.close();
708                 }
709 
710                 Certificate[] chain = new Certificate[] { keyCert };
711 
712                 // import the key entry.
713                 ks.setKeyEntry("cert-" + keyMaterialKeyNames[i],
714                         priKey, passphrase, chain);
715             }
716         }
717 
718         // Create an SSLContext object.
719         TrustManagerFactory tmf =
720                 TrustManagerFactory.getInstance(params.tmAlgorithm);
721         tmf.init(ts);
722 
723         SSLContext context = SSLContext.getInstance(params.contextProtocol);
724         if (hasKeyMaterials && ks != null) {
725             KeyManagerFactory kmf =
726                     KeyManagerFactory.getInstance(params.kmAlgorithm);
727             kmf.init(ks, passphrase);
728 
729             context.init(kmf.getKeyManagers(), tmf.getTrustManagers(), null);
730         } else {
731             context.init(null, tmf.getTrustManagers(), null);
732         }
733 
734         return context;
735     }

737     /*
738      * =================================================
739      * Stuffs to boot up the client-server mode testing.
740      */
741     private Thread clientThread = null;
742     private Thread serverThread = null;
743     private volatile Exception serverException = null;
744     private volatile Exception clientException = null;
745 
746     /*
747      * Should we run the client or server in a separate thread?
748      * Both sides can throw exceptions, but do you have a preference
749      * as to which side should be the main thread.
750      */
751     private static final boolean separateServerThread = false;
752 
753     /*
754      * Boot up the testing, used to drive remainder of the test.
755      */
756     private void bootup() throws Exception {
757         Exception startException = null;
758         try {
759             if (separateServerThread) {
760                 startServer(true);
761                 startClient(false);
762             } else {
763                 startClient(true);
764                 startServer(false);
765             }
766         } catch (Exception e) {
767             startException = e;
768         }
769 
770         /*
771          * Wait for other side to close down.
772          */
773         if (separateServerThread) {
774             if (serverThread != null) {
775                 serverThread.join();
776             }
777         } else {
778             if (clientThread != null) {
779                 clientThread.join();
780             }
781         }
782 
783         /*
784          * When we get here, the test is pretty much over.
785          * Which side threw the error?
786          */
787         Exception local;
788         Exception remote;
789 
790         if (separateServerThread) {
791             remote = serverException;
792             local = clientException;
793         } else {
794             remote = clientException;
795             local = serverException;
796         }
797 
798         Exception exception = null;
799 
800         /*
801          * Check various exception conditions.
802          */
803         if ((local != null) && (remote != null)) {
804             // If both failed, return the curthread's exception.
805             local.initCause(remote);
806             exception = local;
807         } else if (local != null) {
808             exception = local;
809         } else if (remote != null) {
810             exception = remote;
811         } else if (startException != null) {
812             exception = startException;
813         }
814 
815         /*
816          * If there was an exception *AND* a startException,
817          * output it.
818          */
819         if (exception != null) {
820             if (exception != startException && startException != null) {
821                 exception.addSuppressed(startException);
822             }
823             throw exception;
824         }
825 
826         // Fall-through: no exception to throw!
827     }
828 
829     private void startServer(boolean newThread) throws Exception {
830         if (newThread) {
831             serverThread = new Thread() {
832                 @Override
833                 public void run() {
834                     try {
835                         doServerSide();
836                     } catch (Exception e) {
837                         /*
838                          * Our server thread just died.
839                          *
840                          * Release the client, if not active already...
841                          */
842                         logException("Server died", e);
843                         serverException = e;
844                     }
845                 }
846             };
847             serverThread.start();
848         } else {
849             try {
850                 doServerSide();
851             } catch (Exception e) {
852                 logException("Server failed", e);
853                 serverException = e;
854             }
855         }
856     }
857 
858     private void startClient(boolean newThread) throws Exception {
859         if (newThread) {
860             clientThread = new Thread() {
861                 @Override
862                 public void run() {
863                     try {
864                         doClientSide();
865                     } catch (Exception e) {
866                         /*
867                          * Our client thread just died.
868                          */
869                         logException("Client died", e);
870                         clientException = e;
871                     }
872                 }
873             };
874             clientThread.start();
875         } else {
876             try {
877                 doClientSide();
878             } catch (Exception e) {
879                 logException("Client failed", e);
880                 clientException = e;
881             }
882         }
883     }

886         System.out.println(prefix + ": " + cause);
887         cause.printStackTrace(System.out);
