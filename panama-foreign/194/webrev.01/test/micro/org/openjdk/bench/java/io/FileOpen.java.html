23 package org.openjdk.bench.java.io;
24 
25 import org.openjdk.jmh.annotations.*;
26 import org.openjdk.jmh.infra.Blackhole;
27 
28 import java.io.File;
29 import java.util.concurrent.TimeUnit;
30 
31 /**
32  * Tests the overheads of I/O API.
33  */
34 @BenchmarkMode(Mode.AverageTime)
35 @OutputTimeUnit(TimeUnit.MICROSECONDS)
36 @State(Scope.Thread)
37 @Warmup(time=2, iterations=5)
38 @Measurement(time=3, iterations=5)
39 @Fork(value=2, jvmArgs="-Xmx1g")
40 public class FileOpen {
41 
42     public String normalFile = "/test/dir/file/name.txt";
43     public String root = "/";
44     public String trailingSlash = "/test/dir/file/name.txt/";
45     public String notNormalizedFile = "/test/dir/file//name.txt";
46 
47     @Benchmark
48     public void mix(Blackhole bh) {
49         bh.consume(new File(normalFile));
50         bh.consume(new File(root));
51         bh.consume(new File(trailingSlash));
52         bh.consume(new File(notNormalizedFile));
53     }
54 
55     @Benchmark
56     public File normalized() {
57         return new File(normalFile);
58     }
59 
60     @Benchmark
61     public File trailingSlash() {
62         return new File(trailingSlash);
63     }
64 
65     @Benchmark
66     public File notNormalized() {
67         return new File(notNormalizedFile);
68     }
69 }
