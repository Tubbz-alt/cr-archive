25 package jdk.incubator.jpackage.internal;
26 
27 import java.math.BigInteger;
28 
29 
30 final class MsiVersion {
31     /**
32      * Parse the given string as Windows MSI Product version.
33      * https://msdn.microsoft.com/en-us/library/aa370859%28v=VS.85%29.aspx The
34      * format of the string is as follows: major.minor.build. The first field is
35      * the major version and has a maximum value of 255. The second field is the
36      * minor version and has a maximum value of 255. The third field is called
37      * the build version or the update version and has a maximum value of
38      * 65,535.
39      * @throws IllegalArgumentException
40      */
41     static DottedVersion of(String value) {
42         DottedVersion ver = new DottedVersion(value);
43 
44         BigInteger[] components = ver.getComponents();
45         if (components.length > 3) {
46             throw new IllegalArgumentException(I18N.getString(
47                     "error.msi-product-version-too-many-components"));
48         }
49 
50         if (BigInteger.valueOf(255).compareTo(components[0]) < 0) {
51             throw new IllegalArgumentException(I18N.getString(
52                     "error.msi-product-version-major-out-of-range"));
53         }
54 
55         if (components.length > 1 && BigInteger.valueOf(255).compareTo(
56                 components[1]) < 0) {
57             throw new IllegalArgumentException(I18N.getString(
58                     "error.msi-product-version-minor-out-of-range"));
59         }
60 
61         if (components.length > 2 && BigInteger.valueOf(65535).compareTo(
62                 components[2]) < 0) {
63             throw new IllegalArgumentException(I18N.getString(
64                     "error.msi-product-version-build-out-of-range"));
65         }
66 
67         return ver;
68     }
69 }
