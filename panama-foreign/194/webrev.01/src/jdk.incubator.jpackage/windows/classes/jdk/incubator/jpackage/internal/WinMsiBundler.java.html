 60 /**
 61  * WinMsiBundler
 62  *
 63  * Produces .msi installer from application image. Uses WiX Toolkit to build
 64  * .msi installer.
 65  * <p>
 66  * {@link #execute} method creates a number of source files with the description
 67  * of installer to be processed by WiX tools. Generated source files are stored
 68  * in "config" subdirectory next to "app" subdirectory in the root work
 69  * directory. The following WiX source files are generated:
 70  * <ul>
 71  * <li>main.wxs. Main source file with the installer description
 72  * <li>bundle.wxf. Source file with application and Java run-time directory tree
 73  * description.
 74  * </ul>
 75  * <p>
 76  * main.wxs file is a copy of main.wxs resource from
 77  * jdk.incubator.jpackage.internal.resources package. It is parametrized with the
 78  * following WiX variables:
 79  * <ul>
 80  * <li>JpAppName. Name of the application. Set to the value of --name command
 81  * line option
 82  * <li>JpAppVersion. Version of the application. Set to the value of
 83  * --app-version command line option
 84  * <li>JpAppVendor. Vendor of the application. Set to the value of --vendor
 85  * command line option
 86  * <li>JpAppDescription. Description of the application. Set to the value of
 87  * --description command line option
 88  * <li>JpProductCode. Set to product code UUID of the application. Random value
 89  * generated by jpackage every time {@link #execute} method is called
 90  * <li>JpProductUpgradeCode. Set to upgrade code UUID of the application. Random
 91  * value generated by jpackage every time {@link #execute} method is called if
 92  * --win-upgrade-uuid command line option is not specified. Otherwise this
 93  * variable is set to the value of --win-upgrade-uuid command line option
 94  * <li>JpAllowDowngrades. Set to "yes" if --win-upgrade-uuid command line option
 95  * was specified. Undefined otherwise
 96  * <li>JpLicenseRtf. Set to the value of --license-file command line option.
 97  * Undefined is --license-file command line option was not specified
 98  * <li>JpInstallDirChooser. Set to "yes" if --win-dir-chooser command line
 99  * option was specified. Undefined otherwise
100  * <li>JpConfigDir. Absolute path to the directory with generated WiX source
101  * files.
102  * <li>JpIsSystemWide. Set to "yes" if --win-per-user-install command line
103  * option was not specified. Undefined otherwise
104  * </ul>
105  */
106 public class WinMsiBundler  extends AbstractBundler {

181     @Override
182     public boolean supported(boolean platformInstaller) {
183         try {
184             if (wixToolset == null) {
185                 wixToolset = WixTool.toolset();
186             }
187             return true;
188         } catch (ConfigException ce) {
189             Log.error(ce.getMessage());
190             if (ce.getAdvice() != null) {
191                 Log.error(ce.getAdvice());
192             }
193         } catch (Exception e) {
194             Log.error(e.getMessage());
195         }
196         return false;

204     private static UUID getUpgradeCode(Map<String, ? super Object> params) {
205         String upgradeCode = UPGRADE_UUID.fetchFrom(params);
206         if (upgradeCode != null) {
207             return UUID.fromString(upgradeCode);
208         }
209         return createNameUUID("UpgradeCode", params, List.of(VENDOR, APP_NAME));
210     }
211 
212     private static UUID getProductCode(Map<String, ? super Object> params) {
213         return createNameUUID("ProductCode", params, List.of(VENDOR, APP_NAME,
214                 VERSION));
215     }
216 
217     private static UUID createNameUUID(String prefix,
218             Map<String, ? super Object> params,
219             List<StandardBundlerParam<String>> components) {
220         String key = Stream.concat(Stream.of(prefix), components.stream().map(
221                 c -> c.fetchFrom(params))).collect(Collectors.joining("/"));
222         return UUID.nameUUIDFromBytes(key.getBytes(StandardCharsets.UTF_8));
223     }

225     @Override
226     public boolean validate(Map<String, ? super Object> params)
227             throws ConfigException {
228         try {
229             if (wixToolset == null) {
230                 wixToolset = WixTool.toolset();
231             }
232 
233             try {
234                 getUpgradeCode(params);
235             } catch (IllegalArgumentException ex) {
236                 throw new ConfigException(ex);
237             }
238 
239             for (var toolInfo: wixToolset.values()) {
240                 Log.verbose(MessageFormat.format(I18N.getString(
241                         "message.tool-version"), toolInfo.path.getFileName(),
242                         toolInfo.version));
243             }
244 
245             wixSourcesBuilder.setWixVersion(wixToolset.get(WixTool.Light).version);
246 
247             wixSourcesBuilder.logWixFeatures();
248 
249             /********* validate bundle parameters *************/
250 
251             try {
252                 String version = PRODUCT_VERSION.fetchFrom(params);
253                 MsiVersion.of(version);
254             } catch (IllegalArgumentException ex) {
255                 throw new ConfigException(ex.getMessage(), I18N.getString(
256                         "error.version-string-wrong-format.advice"), ex);
257             }
258 
259             FileAssociation.verify(FileAssociation.fetchFrom(params));
260 
261             return true;
262         } catch (RuntimeException re) {
263             if (re.getCause() instanceof ConfigException) {
264                 throw (ConfigException) re.getCause();
265             } else {
266                 throw new ConfigException(re);
267             }
268         }
269     }

271     private void prepareProto(Map<String, ? super Object> params)
272                 throws PackagerException, IOException {
273         File appImage = StandardBundlerParam.getPredefinedAppImage(params);
274         File appDir = null;
275 
276         // we either have an application image or need to build one
277         if (appImage != null) {
278             appDir = new File(MSI_IMAGE_DIR.fetchFrom(params),
279                     APP_NAME.fetchFrom(params));
280             // copy everything from appImage dir into appDir/name
281             IOUtils.copyRecursive(appImage.toPath(), appDir.toPath());
282         } else {
283             appDir = APP_BUNDLER.fetchFrom(params).doBundle(params,
284                     MSI_IMAGE_DIR.fetchFrom(params), true);
285         }
286 
287         // Configure installer icon
288         if (StandardBundlerParam.isRuntimeInstaller(params)) {
289             // Use icon from java launcher.
290             // Assume java.exe exists in Java Runtime being packed.
291             // Ignore custom icon if any as we don't want to copy anything in
292             // Java Runtime image.
293             installerIcon = ApplicationLayout.javaRuntime()
294                     .runtimeDirectory()
295                     .resolve(Path.of("bin", "java.exe"));
296         } else {
297             installerIcon = ApplicationLayout.windowsAppImage()
298                     .launchersDirectory()
299                     .resolve(APP_NAME.fetchFrom(params) + ".exe");
300         }
301 
302         params.put(WIN_APP_IMAGE.getID(), appDir);
303 
304         String licenseFile = LICENSE_FILE.fetchFrom(params);
305         if (licenseFile != null) {
306             // need to copy license file to the working directory
307             // and convert to rtf if needed
308             File lfile = new File(licenseFile);
309             File destFile = new File(CONFIG_ROOT.fetchFrom(params),
310                     lfile.getName());
311 
312             IOUtils.copyFile(lfile, destFile);
313             destFile.setWritable(true);
314             ensureByMutationFileIsRTF(destFile);
315         }
316     }

318     public File bundle(Map<String, ? super Object> params, File outdir)
319             throws PackagerException {
320 
321         IOUtils.writableOutputDir(outdir.toPath());
322 
323         Path imageDir = MSI_IMAGE_DIR.fetchFrom(params).toPath();
324         try {
325             Files.createDirectories(imageDir);
326 
327             prepareProto(params);
328 
329             wixSourcesBuilder
330             .initFromParams(WIN_APP_IMAGE.fetchFrom(params).toPath(), params)
331             .createMainFragment(CONFIG_ROOT.fetchFrom(params).toPath().resolve(
332                     "bundle.wxf"));
333 
334             Map<String, String> wixVars = prepareMainProjectFile(params);
335 
336             new ScriptRunner()
337             .setDirectory(imageDir)
338             .setResourceCategoryId("resource.post-app-image-script")
339             .setScriptNameSuffix("post-image")
340             .setEnvironmentVariable("JpAppImageDir", imageDir.toAbsolutePath().toString())
341             .run(params);
342 
343             return buildMSI(params, wixVars, outdir);
344         } catch (IOException ex) {
345             Log.verbose(ex);
346             throw new PackagerException(ex);
347         }
348     }

350     Map<String, String> prepareMainProjectFile(
351             Map<String, ? super Object> params) throws IOException {
352         Map<String, String> data = new HashMap<>();
353 
354         final UUID productCode = getProductCode(params);
355         final UUID upgradeCode = getUpgradeCode(params);
356 
357         data.put("JpProductCode", productCode.toString());
358         data.put("JpProductUpgradeCode", upgradeCode.toString());
359 
360         Log.verbose(MessageFormat.format(I18N.getString("message.product-code"),
361                 productCode));
362         Log.verbose(MessageFormat.format(I18N.getString("message.upgrade-code"),
363                 upgradeCode));
364 
365         data.put("JpAllowUpgrades", "yes");
366         data.put("JpAllowDowngrades", "yes");
367 
368         data.put("JpAppName", APP_NAME.fetchFrom(params));
369         data.put("JpAppDescription", DESCRIPTION.fetchFrom(params));
370         data.put("JpAppVendor", VENDOR.fetchFrom(params));
371         data.put("JpAppVersion", PRODUCT_VERSION.fetchFrom(params));
372         data.put("JpIcon", installerIcon.toString());
373 
374         final Path configDir = CONFIG_ROOT.fetchFrom(params).toPath();
375 
376         data.put("JpConfigDir", configDir.toAbsolutePath().toString());
377 
378         if (MSI_SYSTEM_WIDE.fetchFrom(params)) {
379             data.put("JpIsSystemWide", "yes");
380         }
381 
382         String licenseFile = LICENSE_FILE.fetchFrom(params);
383         if (licenseFile != null) {
384             String lname = new File(licenseFile).getName();
385             File destFile = new File(CONFIG_ROOT.fetchFrom(params), lname);
386             data.put("JpLicenseRtf", destFile.getAbsolutePath());
387         }
388 
389         // Copy CA dll to include with installer
390         if (INSTALLDIR_CHOOSER.fetchFrom(params)) {
391             data.put("JpInstallDirChooser", "yes");
392             String fname = "wixhelper.dll";
393             try (InputStream is = OverridableResource.readDefault(fname)) {
394                 Files.copy(is, Paths.get(
395                         CONFIG_ROOT.fetchFrom(params).getAbsolutePath(),
396                         fname));
397             }
398         }
399 
400         // Copy l10n files.
401         for (String loc : Arrays.asList("en", "ja", "zh_CN")) {
402             String fname = "MsiInstallerStrings_" + loc + ".wxl";
403             try (InputStream is = OverridableResource.readDefault(fname)) {
404                 Files.copy(is, Paths.get(
405                         CONFIG_ROOT.fetchFrom(params).getAbsolutePath(),
406                         fname));
407             }
408         }
409 
410         createResource("main.wxs", params)
411                 .setCategory(I18N.getString("resource.main-wix-file"))
412                 .saveToFile(configDir.resolve("main.wxs"));
413 
414         createResource("overrides.wxi", params)
415                 .setCategory(I18N.getString("resource.overrides-wix-file"))
416                 .saveToFile(configDir.resolve("overrides.wxi"));
417 
418         return data;
419     }

421     private File buildMSI(Map<String, ? super Object> params,
422             Map<String, String> wixVars, File outdir)
423             throws IOException {
424 
425         File msiOut = new File(
426                 outdir, INSTALLER_FILE_NAME.fetchFrom(params) + ".msi");
427 
428         Log.verbose(MessageFormat.format(I18N.getString(
429                 "message.preparing-msi-config"), msiOut.getAbsolutePath()));
430 
431         WixPipeline wixPipeline = new WixPipeline()
432         .setToolset(wixToolset.entrySet().stream().collect(
433                 Collectors.toMap(
434                         entry -> entry.getKey(),
435                         entry -> entry.getValue().path)))
436         .setWixObjDir(TEMP_ROOT.fetchFrom(params).toPath().resolve("wixobj"))
437         .setWorkDir(WIN_APP_IMAGE.fetchFrom(params).toPath())
438         .addSource(CONFIG_ROOT.fetchFrom(params).toPath().resolve("main.wxs"), wixVars)
439         .addSource(CONFIG_ROOT.fetchFrom(params).toPath().resolve("bundle.wxf"), null);
440 
441         Log.verbose(MessageFormat.format(I18N.getString(
442                 "message.generating-msi"), msiOut.getAbsolutePath()));
443 
444         boolean enableLicenseUI = (LICENSE_FILE.fetchFrom(params) != null);
445         boolean enableInstalldirUI = INSTALLDIR_CHOOSER.fetchFrom(params);
446 
447         if (!MSI_SYSTEM_WIDE.fetchFrom(params)) {
448             wixPipeline.addLightOptions("-sice:ICE91");
449         }
450         if (enableLicenseUI || enableInstalldirUI) {
451             wixPipeline.addLightOptions("-ext", "WixUIExtension");
452         }
453 
454         wixPipeline.addLightOptions("-loc",
455                 CONFIG_ROOT.fetchFrom(params).toPath().resolve(I18N.getString(
456                         "resource.wxl-file-name")).toAbsolutePath().toString());
457 
458         // Only needed if we using CA dll, so Wix can find it
459         if (enableInstalldirUI) {
460             wixPipeline.addLightOptions("-b", CONFIG_ROOT.fetchFrom(params).getAbsolutePath());
461         }
462 
463         wixPipeline.buildMsi(msiOut.toPath().toAbsolutePath());
464 
465         return msiOut;
466     }

468     public static void ensureByMutationFileIsRTF(File f) {
469         if (f == null || !f.isFile()) return;
470 
471         try {
472             boolean existingLicenseIsRTF = false;
473 
474             try (FileInputStream fin = new FileInputStream(f)) {
475                 byte[] firstBits = new byte[7];
476 
477                 if (fin.read(firstBits) == firstBits.length) {
478                     String header = new String(firstBits);
479                     existingLicenseIsRTF = "{\\rtf1\\".equals(header);
480                 }
481             }
482 
483             if (!existingLicenseIsRTF) {
484                 List<String> oldLicense = Files.readAllLines(f.toPath());
485                 try (Writer w = Files.newBufferedWriter(
486                         f.toPath(), Charset.forName("Windows-1252"))) {
487                     w.write("{\\rtf1\\ansi\\ansicpg1252\\deff0\\deflang1033"
488                             + "{\\fonttbl{\\f0\\fnil\\fcharset0 Arial;}}\n"
489                             + "\\viewkind4\\uc1\\pard\\sa200\\sl276"
490                             + "\\slmult1\\lang9\\fs20 ");
491                     oldLicense.forEach(l -> {
492                         try {
493                             for (char c : l.toCharArray()) {
494                                 // 0x00 <= ch < 0x20 Escaped (\'hh)
495                                 // 0x20 <= ch < 0x80 Raw(non - escaped) char
496                                 // 0x80 <= ch <= 0xFF Escaped(\ 'hh)
497                                 // 0x5C, 0x7B, 0x7D (special RTF characters
498                                 // \,{,})Escaped(\'hh)
499                                 // ch > 0xff Escaped (\\ud###?)
500                                 if (c < 0x10) {
501                                     w.write("\\'0");
502                                     w.write(Integer.toHexString(c));
503                                 } else if (c > 0xff) {
504                                     w.write("\\ud");
505                                     w.write(Integer.toString(c));
506                                     // \\uc1 is in the header and in effect
507                                     // so we trail with a replacement char if
508                                     // the font lacks that character - '?'
509                                     w.write("?");
510                                 } else if ((c < 0x20) || (c >= 0x80) ||
511                                         (c == 0x5C) || (c == 0x7B) ||
512                                         (c == 0x7D)) {
513                                     w.write("\\'");
514                                     w.write(Integer.toHexString(c));
515                                 } else {
516                                     w.write(c);
517                                 }
518                             }
519                             // blank lines are interpreted as paragraph breaks
520                             if (l.length() < 1) {
521                                 w.write("\\par");
522                             } else {
523                                 w.write(" ");
524                             }
525                             w.write("\r\n");
526                         } catch (IOException e) {
527                             Log.verbose(e);
528                         }
529                     });
530                     w.write("}\r\n");
531                 }
532             }
533         } catch (IOException e) {
534             Log.verbose(e);
535         }
536 
537     }
