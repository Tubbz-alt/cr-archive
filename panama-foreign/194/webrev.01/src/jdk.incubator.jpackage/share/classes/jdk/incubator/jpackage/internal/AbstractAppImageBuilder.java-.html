 44 /*
 45  * AbstractAppImageBuilder
 46  *     This is sub-classed by each of the platform dependent AppImageBuilder
 47  * classes, and contains resource processing code common to all platforms.
 48  */
 49 
 50 public abstract class AbstractAppImageBuilder {
 51 
 52     private static final ResourceBundle I18N = ResourceBundle.getBundle(
 53             "jdk.incubator.jpackage.internal.resources.MainResources");
 54 
 55     private final Path root;
 56 
 57     public AbstractAppImageBuilder(Map<String, Object> unused, Path root) {
 58         this.root = root;
 59     }
 60 
 61     public InputStream getResourceAsStream(String name) {
 62         return ResourceLocator.class.getResourceAsStream(name);
 63     }

 65     public abstract void prepareApplicationFiles(
 66             Map<String, ? super Object> params) throws IOException;
 67     public abstract void prepareJreFiles(
 68             Map<String, ? super Object> params) throws IOException;
 69     public abstract Path getAppDir();
 70     public abstract Path getAppModsDir();
 71 
 72     public Path getRuntimeRoot() {
 73         return this.root;
 74     }
 75 
 76     protected void copyEntry(Path appDir, File srcdir, String fname)
 77             throws IOException {
 78         Path dest = appDir.resolve(fname);
 79         Files.createDirectories(dest.getParent());
 80         File src = new File(srcdir, fname);
 81         if (src.isDirectory()) {
 82             IOUtils.copyRecursive(src.toPath(), dest);
 83         } else {
 84             Files.copy(src.toPath(), dest);
 85         }
 86     }
 87 
 88     public void writeCfgFile(Map<String, ? super Object> params,
 89             File cfgFileName) throws IOException {
 90         cfgFileName.getParentFile().mkdirs();
 91         cfgFileName.delete();
 92         File mainJar = JLinkBundlerHelper.getMainJar(params);
 93         ModFile.ModType mainJarType = ModFile.ModType.Unknown;
 94 
 95         if (mainJar != null) {
 96             mainJarType = new ModFile(mainJar).getModType();
 97         }
 98 
 99         String mainModule = StandardBundlerParam.MODULE.fetchFrom(params);
100 
101         try (PrintStream out = new PrintStream(cfgFileName)) {
102 
103             out.println("[Application]");
104             out.println("app.name=" + APP_NAME.fetchFrom(params));
105             out.println("app.version=" + VERSION.fetchFrom(params));
106             out.println("app.runtime=" + getCfgRuntimeDir());
107             out.println("app.identifier=" + IDENTIFIER.fetchFrom(params));
108             out.println("app.classpath="
109                     + getCfgClassPath(CLASSPATH.fetchFrom(params)));
110 
111             // The main app is required to be a jar, modular or unnamed.
112             if (mainModule != null &&
113                     (mainJarType == ModFile.ModType.Unknown ||
114                     mainJarType == ModFile.ModType.ModularJar)) {
115                 out.println("app.mainmodule=" + mainModule);
116             } else {
117                 String mainClass =
118                         StandardBundlerParam.MAIN_CLASS.fetchFrom(params);
119                 // If the app is contained in an unnamed jar then launch it the
120                 // legacy way and the main class string must be
121                 // of the format com/foo/Main
122                 if (mainJar != null) {
123                     out.println("app.classpath=" + getCfgAppDir()
124                             + mainJar.toPath().getFileName().toString());
125                 }
126                 if (mainClass != null) {
127                     out.println("app.mainclass=" + mainClass);
128                 }
129             }
130 
131             out.println();
132             out.println("[JavaOptions]");
133             List<String> jvmargs = JAVA_OPTIONS.fetchFrom(params);
134             for (String arg : jvmargs) {
135                 out.println("java-options=" + arg);
136             }
137             Path modsDir = getAppModsDir();
138 
139             if (modsDir != null && modsDir.toFile().exists()) {
140                 out.println("java-options=" + "--module-path");
141                 out.println("java-options=" + getCfgAppDir().replace("\\","/") + "mods");
142             }
143 
144             out.println();
145             out.println("[ArgOptions]");
146             List<String> args = ARGUMENTS.fetchFrom(params);
147             for (String arg : args) {
148                 out.println("arguments=" + arg);
149             }
150         }
151     }
152 
153     File getRuntimeImageDir(File runtimeImageTop) {
154         return runtimeImageTop;
155     }
156 
157     protected String getCfgAppDir() {
158         return "$ROOTDIR" + File.separator
159                 + getAppDir().getFileName() + File.separator;
160     }
161 
162     protected String getCfgRuntimeDir() {
163         return "$ROOTDIR" + File.separator + "runtime";
164     }
165 
166     String getCfgClassPath(String classpath) {
167         String cfgAppDir = getCfgAppDir();
168 
169         StringBuilder sb = new StringBuilder();
170         for (String path : classpath.split("[:;]")) {
171             if (path.length() > 0) {
172                 sb.append(cfgAppDir);
173                 sb.append(path);
174                 sb.append(File.pathSeparator);
175             }
176         }
177         if (sb.length() > 0) {
178             sb.deleteCharAt(sb.length() - 1);
179         }
180         return sb.toString();
181     }

183     public static OverridableResource createIconResource(String defaultIconName,
184             BundlerParamInfo<File> iconParam, Map<String, ? super Object> params,
185             Map<String, ? super Object> mainParams) throws IOException {
186 
187         if (mainParams != null) {
188             params = AddLauncherArguments.merge(mainParams, params, ICON.getID(),
189                     iconParam.getID());
190         }
191 
192         final String resourcePublicName = APP_NAME.fetchFrom(params)
193                 + IOUtils.getSuffix(Path.of(defaultIconName));
194 
195         IconType iconType = getLauncherIconType(params);
196         if (iconType == IconType.NoIcon) {
197             return null;
198         }
199 
200         OverridableResource resource = createResource(defaultIconName, params)
201                 .setCategory("icon")
202                 .setExternal(iconParam.fetchFrom(params))
203                 .setPublicName(resourcePublicName);
204 
205         if (iconType == IconType.DefaultOrResourceDirIcon && mainParams != null) {
206             // No icon explicitly configured for this launcher.
207             // Dry-run resource creation to figure out its source.
208             final Path nullPath = null;
209             if (resource.saveToFile(nullPath)
210                     != OverridableResource.Source.ResourceDir) {
211                 // No icon in resource dir for this launcher, inherit icon
212                 // configured for the main launcher.
213                 resource = createIconResource(defaultIconName, iconParam,
214                         mainParams, null).setLogPublicName(resourcePublicName);
215             }
216         }
217 
218         return resource;
219     }
220 
221     private enum IconType { DefaultOrResourceDirIcon, CustomIcon, NoIcon };
222 
223     private static IconType getLauncherIconType(Map<String, ? super Object> params) {
224         File launcherIcon = ICON.fetchFrom(params);
225         if (launcherIcon == null) {
226             return IconType.DefaultOrResourceDirIcon;
227         }
228 
229         if (launcherIcon.getName().isEmpty()) {
230             return IconType.NoIcon;
231         }
232 
233         return IconType.CustomIcon;
234     }
