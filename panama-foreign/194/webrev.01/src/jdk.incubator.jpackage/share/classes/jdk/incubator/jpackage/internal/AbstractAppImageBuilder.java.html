 44 /*
 45  * AbstractAppImageBuilder
 46  *     This is sub-classed by each of the platform dependent AppImageBuilder
 47  * classes, and contains resource processing code common to all platforms.
 48  */
 49 
 50 public abstract class AbstractAppImageBuilder {
 51 
 52     private static final ResourceBundle I18N = ResourceBundle.getBundle(
 53             "jdk.incubator.jpackage.internal.resources.MainResources");
 54 
 55     private final Path root;
 56 
 57     public AbstractAppImageBuilder(Map<String, Object> unused, Path root) {
 58         this.root = root;
 59     }
 60 
 61     public InputStream getResourceAsStream(String name) {
 62         return ResourceLocator.class.getResourceAsStream(name);
 63     }
 64 
 65     public abstract void prepareApplicationFiles(
 66             Map<String, ? super Object> params) throws IOException;
 67     public abstract void prepareJreFiles(
 68             Map<String, ? super Object> params) throws IOException;
 69     public abstract Path getAppDir();
 70     public abstract Path getAppModsDir();
 71 
 72     public Path getRuntimeRoot() {
 73         return this.root;
 74     }
 75 
 76     protected void copyEntry(Path appDir, File srcdir, String fname)
 77             throws IOException {
 78         Path dest = appDir.resolve(fname);
 79         Files.createDirectories(dest.getParent());
 80         File src = new File(srcdir, fname);
 81         if (src.isDirectory()) {
 82             IOUtils.copyRecursive(src.toPath(), dest);
 83         } else {
 84             Files.copy(src.toPath(), dest);
 85         }
 86     }
 87 
 88     public void writeCfgFile(Map<String, ? super Object> params,
 89             File cfgFileName) throws IOException {
 90         cfgFileName.getParentFile().mkdirs();
 91         cfgFileName.delete();
 92         File mainJar = JLinkBundlerHelper.getMainJar(params);
 93         ModFile.ModType mainJarType = ModFile.ModType.Unknown;
 94 
 95         if (mainJar != null) {
 96             mainJarType = new ModFile(mainJar).getModType();
 97         }
 98 
 99         String mainModule = StandardBundlerParam.MODULE.fetchFrom(params);
100 
101         try (PrintStream out = new PrintStream(cfgFileName)) {
102 
103             out.println("[Application]");
104             out.println("app.name=" + APP_NAME.fetchFrom(params));
105             out.println("app.version=" + VERSION.fetchFrom(params));
106             out.println("app.runtime=" + getCfgRuntimeDir());
107             out.println("app.classpath="
108                     + getCfgClassPath(CLASSPATH.fetchFrom(params)));
109 
110             // The main app is required to be a jar, modular or unnamed.
111             if (mainModule != null &&
112                     (mainJarType == ModFile.ModType.Unknown ||
113                     mainJarType == ModFile.ModType.ModularJar)) {
114                 out.println("app.mainmodule=" + mainModule);
115             } else {
116                 String mainClass =
117                         StandardBundlerParam.MAIN_CLASS.fetchFrom(params);
118                 // If the app is contained in an unnamed jar then launch it the
119                 // legacy way and the main class string must be
120                 // of the format com/foo/Main
121                 if (mainJar != null) {
122                     out.println("app.classpath=" + getCfgAppDir()
123                             + mainJar.toPath().getFileName().toString());
124                 }
125                 if (mainClass != null) {
126                     out.println("app.mainclass=" + mainClass);
127                 }
128             }
129 
130             out.println();
131             out.println("[JavaOptions]");
132             List<String> jvmargs = JAVA_OPTIONS.fetchFrom(params);
133             for (String arg : jvmargs) {
134                 out.println("java-options=" + arg);
135             }
136             Path modsDir = getAppModsDir();
137 
138             if (modsDir != null && modsDir.toFile().exists()) {
139                 out.println("java-options=" + "--module-path");
140                 out.println("java-options=" + getCfgAppDir().replace("\\","/") + "mods");
141             }
142 
143             out.println();
144             out.println("[ArgOptions]");
145             List<String> args = ARGUMENTS.fetchFrom(params);
146             for (String arg : args) {
147                 out.println("arguments=" + arg);
148             }
149         }
150     }
151 
152     File getRuntimeImageDir(File runtimeImageTop) {
153         return runtimeImageTop;
154     }
155 
156     protected String getCfgAppDir() {
157         return "$ROOTDIR" + File.separator
158                 + getAppDir().getFileName() + File.separator;
159     }
160 
161     protected String getCfgRuntimeDir() {
162         return "$ROOTDIR" + File.separator + "runtime";
163     }
164 
165     String getCfgClassPath(String classpath) {
166         String cfgAppDir = getCfgAppDir();
167 
168         StringBuilder sb = new StringBuilder();
169         for (String path : classpath.split("[:;]")) {
170             if (path.length() > 0) {
171                 sb.append(cfgAppDir);
172                 sb.append(path);
173                 sb.append(File.pathSeparator);
174             }
175         }
176         if (sb.length() > 0) {
177             sb.deleteCharAt(sb.length() - 1);
178         }
179         return sb.toString();
180     }

182     public static OverridableResource createIconResource(String defaultIconName,
183             BundlerParamInfo<File> iconParam, Map<String, ? super Object> params,
184             Map<String, ? super Object> mainParams) throws IOException {
185 
186         if (mainParams != null) {
187             params = AddLauncherArguments.merge(mainParams, params, ICON.getID(),
188                     iconParam.getID());
189         }
190 
191         final String resourcePublicName = APP_NAME.fetchFrom(params)
192                 + IOUtils.getSuffix(Path.of(defaultIconName));
193 
194         IconType iconType = getLauncherIconType(params);
195         if (iconType == IconType.NoIcon) {
196             return null;
197         }
198 
199         OverridableResource resource = createResource(defaultIconName, params)
200                 .setCategory("icon")
201                 .setExternal(iconParam.fetchFrom(params))
202                 .setPublicName(resourcePublicName);
203 
204         if (iconType == IconType.DefaultOrResourceDirIcon && mainParams != null) {
205             // No icon explicitly configured for this launcher.
206             // Dry-run resource creation to figure out its source.
207             final Path nullPath = null;
208             if (resource.saveToFile(nullPath)
209                     != OverridableResource.Source.ResourceDir) {
210                 // No icon in resource dir for this launcher, inherit icon
211                 // configured for the main launcher.
212                 resource = createIconResource(defaultIconName, iconParam,
213                         mainParams, null).setLogPublicName(resourcePublicName);
214             }
215         }
216 
217         return resource;
218     }
219 
220     private enum IconType { DefaultOrResourceDirIcon, CustomIcon, NoIcon };
221 
222     private static IconType getLauncherIconType(Map<String, ? super Object> params) {
223         File launcherIcon = ICON.fetchFrom(params);
224         if (launcherIcon == null) {
225             return IconType.DefaultOrResourceDirIcon;
226         }
227 
228         if (launcherIcon.getName().isEmpty()) {
229             return IconType.NoIcon;
230         }
231 
232         return IconType.CustomIcon;
233     }
