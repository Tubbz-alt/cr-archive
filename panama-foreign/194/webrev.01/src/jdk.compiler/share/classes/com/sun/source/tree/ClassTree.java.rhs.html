 26 package com.sun.source.tree;
 27 
 28 import java.util.Collections;
 29 import java.util.List;
 30 import javax.lang.model.element.Name;
 31 
 32 /**
 33  * A tree node for a class, interface, enum, record, or annotation
 34  * type declaration.
 35  *
 36  * For example:
 37  * <pre>
 38  *   <em>modifiers</em> class <em>simpleName</em> <em>typeParameters</em>
 39  *       extends <em>extendsClause</em>
 40  *       implements <em>implementsClause</em>
 41  *   {
 42  *       <em>members</em>
 43  *   }
 44  * </pre>
 45  *
 46  * @jls 8.1 Class Declarations
 47  * @jls 8.9 Enum Types
 48  * @jls 8.10 Record Types
 49  * @jls 9.1 Interface Declarations
 50  * @jls 9.6 Annotation Types
 51  *
 52  * @author Peter von der Ah&eacute;
 53  * @author Jonathan Gibbons
 54  * @since 1.6
 55  */
 56 public interface ClassTree extends StatementTree {
 57     /**
 58      * Returns the modifiers, including any annotations,
 59      * for this type declaration.
 60      * @return the modifiers
 61      */
 62     ModifiersTree getModifiers();
 63 
 64     /**
 65      * Returns the simple name of this type declaration.
 66      * @return the simple name
 67      */
 68     Name getSimpleName();
 69 
 70     /**
 71      * Returns any type parameters of this type declaration.
 72      * @return the type parameters
 73      */
 74     List<? extends TypeParameterTree> getTypeParameters();
 75 
 76     /**
 77      * Returns the supertype of this type declaration,
 78      * or {@code null} if none is provided.
 79      * @return the supertype
 80      */
 81     Tree getExtendsClause();
 82 
 83     /**
 84      * Returns the interfaces implemented by this type declaration.
 85      * @return the interfaces
 86      */
 87     List<? extends Tree> getImplementsClause();
 88 
 89     /**
 90      * {@preview Associated with sealed classes, a preview feature of the Java language.
 91      *
 92      *           This method is associated with <i>sealed classes</i>, a preview
 93      *           feature of the Java language. Preview features
 94      *           may be removed in a future release, or upgraded to permanent
 95      *           features of the Java language.}
 96      *
 97      * Returns the subclasses permitted by this type declaration.
 98      *
 99      * @implSpec this implementation returns an empty list
100      *
101      * @return the subclasses
102      *
103      * @since 15
104      */
105     @jdk.internal.PreviewFeature(feature=jdk.internal.PreviewFeature.Feature.SEALED_CLASSES,
106                                              essentialAPI=false)
107     default List<? extends Tree> getPermitsClause() {
108         return List.of();
109     }
110 
111     /**
112      * Returns the members declared in this type declaration.
113      * @return the members
114      */
115     List<? extends Tree> getMembers();
116 }
