26 package jdk.jfr.events;
27 
28 import jdk.internal.misc.VM.BufferPool;
29 
30 import jdk.jfr.*;
31 import jdk.jfr.internal.Type;
32 
33 @Category({ "Java Application", "Statistics" })
34 public abstract class AbstractBufferStatisticsEvent extends AbstractJDKEvent {
35 
36     AbstractBufferStatisticsEvent() {
37         BufferPool bufferPool = getBufferPool();
38 
39         count = bufferPool.getCount();
40         totalCapacity = bufferPool.getTotalCapacity();
41         memoryUsed = bufferPool.getMemoryUsed();
42     }
43 
44     @Label("Count")
45     public long count;
46 
47     @Label("Total Capacity")
48     @DataAmount
49     public long totalCapacity;
50 
51     @Label("Memory Used")
52     @DataAmount
53     public long memoryUsed;
54 
55     abstract BufferPool getBufferPool();







56 }
