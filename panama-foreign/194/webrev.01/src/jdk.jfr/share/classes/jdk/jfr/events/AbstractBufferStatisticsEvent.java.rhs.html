26 package jdk.jfr.events;
27 
28 import jdk.internal.misc.VM.BufferPool;
29 import jdk.internal.misc.VM;
30 import jdk.jfr.*;
31 import jdk.jfr.internal.Type;
32 
33 @Category({ "Java Application", "Statistics" })
34 public abstract class AbstractBufferStatisticsEvent extends AbstractJDKEvent {
35 
36     protected AbstractBufferStatisticsEvent(BufferPool bufferPool) {


37         count = bufferPool.getCount();
38         totalCapacity = bufferPool.getTotalCapacity();
39         memoryUsed = bufferPool.getMemoryUsed();
40     }
41 
42     @Label("Count")
43     final long count;
44 
45     @Label("Total Capacity")
46     @DataAmount
47     final long totalCapacity;
48 
49     @Label("Memory Used")
50     @DataAmount
51     final long memoryUsed;
52 
53     static BufferPool findPoolByName(String name) {
54         for (BufferPool pool : VM.getBufferPools()) {
55             if (pool.getName().equals(name)) {
56                 return pool;
57             }
58         }
59         throw new InternalError("No buffer pool with name " + name);
60     }
61 }
