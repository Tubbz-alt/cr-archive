142     /**
143      * Returns the path for a type element.
144      * For example, if the type element is {@code java.lang.Object},
145      * the path is {@code java/lang/Object.html}.
146      *
147      * @param typeElement the type element
148      * @return the path
149      */
150     public DocPath forClass(TypeElement typeElement) {
151         return (typeElement == null)
152                 ? DocPath.empty
153                 : forPackage(utils.containingPackage(typeElement)).resolve(forName(typeElement));
154     }
155 
156     /**
157      * Returns the path for the simple name of a type element.
158      * For example, if the type element is {@code java.lang.Object},
159      * the path is {@code Object.html}.
160      *
161      * @param typeElement the type element
162      * @return the path
163      */
164     public DocPath forName(TypeElement typeElement) {
165         return (typeElement == null)
166                 ? DocPath.empty
167                 : new DocPath(utils.getSimpleName(typeElement) + ".html");
168     }
169 
170     public static DocPath forModule(ModuleElement mdle) {
171         return mdle == null || mdle.isUnnamed()
172                 ? DocPath.empty
173                 : DocPath.create(mdle.getQualifiedName().toString());
174     }
175 
176     /**
177      * Returns the path for the package of a type element.
178      * For example, if the type element is {@code java.lang.Object},
179      * the path is {@code java/lang}.
180      *
181      * @param typeElement the type element
182      * @return the path
183      */
184     public DocPath forPackage(TypeElement typeElement) {
185         return (typeElement == null)
186                 ? DocPath.empty
187                 : forPackage(utils.containingPackage(typeElement));
188     }
189 
190     /**
191      * Returns the path for a package.
192      * For example, if the package is {@code java.lang},
193      * the path is {@code java/lang}.
194      *
195      * @param pkgElement the package element
196      * @return the path
197      */
198     public DocPath forPackage(PackageElement pkgElement) {
199         if (pkgElement == null || pkgElement.isUnnamed()) {
200             return DocPath.empty;
201         }
202 
203         DocPath pkgPath = DocPath.create(pkgElement.getQualifiedName().toString().replace('.', '/'));
204         ModuleElement mdle = (ModuleElement) pkgElement.getEnclosingElement();
205         return forModule(mdle).resolve(pkgPath);
206     }
207 
208     /**
209      * Returns the inverse path for a package.
210      * For example, if the package is {@code java.lang},
211      * the inverse path is {@code ../..}.
212      *
213      * @param pkgElement the package element
214      * @return the path
215      */
216     public static DocPath forRoot(PackageElement pkgElement) {
217         String name = (pkgElement == null || pkgElement.isUnnamed())
218                 ? ""
219                 : pkgElement.getQualifiedName().toString();
220         return new DocPath(name.replace('.', '/').replaceAll("[^/]+", ".."));
221     }
222 
223     /**
224      * Returns a relative path from one package to another.
225      *
226      * @param from the origin of the relative path
227      * @param to the destination of the relative path
228      * @return the path
229      */
230     public DocPath relativePath(PackageElement from, PackageElement to) {
231         return forRoot(from).resolve(forPackage(to));
232     }

234     /**
235      * The path for the output directory for module documentation files.
236      * @param mdle the module
237      * @return the path
238      */

240         return createModulePath(mdle, "doc-files");
241     }

244      * The path for the file for a module's summary page.
245      * @param mdle the module
246      * @return the path
247      */
248     public DocPath moduleSummary(ModuleElement mdle) {
249         return createModulePath(mdle, "summary.html");
250     }

252     /**
253      * The path for the file for a module's summary page.
254      * @param mdleName the module
255      * @return the path
256      */
257     public DocPath moduleSummary(String mdleName) {
258         return createModulePath(mdleName, "summary.html");
259     }
260 
261     private DocPath createModulePath(ModuleElement mdle, String path) {
262         return DocPath.create(mdle.getQualifiedName() + moduleSeparator + path);
263     }
264 
265     private DocPath createModulePath(String moduleName, String path) {
266         return DocPath.create(moduleName + moduleSeparator + path);
267     }
