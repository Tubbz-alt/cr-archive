719     private final String cssName;
720 
721     HtmlStyle() {
722         cssName = Pattern.compile("\\p{Upper}")
723                 .matcher(toString())
724                 .replaceAll(mr -> "-" + mr.group().toLowerCase(Locale.US));
725     }
726 
727     HtmlStyle(String cssName) {
728         this.cssName = cssName;
729     }
730 
731     /**
732      * Returns the CSS class name associated with this style.
733      * @return the CSS class name
734      */
735     public String cssName() {
736         return cssName;
737     }
