  64 address TemplateInterpreterGenerator::generate_slow_signature_handler() {
  65   address entry = __ pc();
  66 
  67   // callee-save register for saving LR, shared with generate_native_entry
  68   const Register Rsaved_ret_addr = Rtmp_save0;
  69 
  70   __ mov(Rsaved_ret_addr, LR);
  71 
  72   __ mov(R1, Rmethod);
  73   __ mov(R2, Rlocals);
  74   __ mov(R3, SP);
  75 
  76 
  77   // Safer to save R9 (when scratched) since callers may have been
  78   // written assuming R9 survives. This is suboptimal but
  79   // probably not important for this slow case call site.
  80   // Note for R9 saving: slow_signature_handler may copy register
  81   // arguments above the current SP (passed as R3). It is safe for
  82   // call_VM to use push and pop to protect additional values on the
  83   // stack if needed.
  84   __ call_VM(CAST_FROM_FN_PTR(address, InterpreterRuntime::slow_signature_handler), true /* save R9 if needed*/);
  85   __ add(SP, SP, wordSize);     // Skip R0
  86   __ pop(RegisterSet(R1, R3));  // Load arguments passed in registers
  87 #ifdef __ABI_HARD__
  88   // Few alternatives to an always-load-FP-registers approach:
  89   // - parse method signature to detect FP arguments
  90   // - keep a counter/flag on a stack indicationg number of FP arguments in the method.
  91   // The later has been originally implemented and tested but a conditional path could
  92   // eliminate any gain imposed by avoiding 8 double word loads.
  93   __ fldmiad(SP, FloatRegisterSet(D0, 8), writeback);
  94 #endif // __ABI_HARD__
  95 
  96   __ ret(Rsaved_ret_addr);
  97 
  98   return entry;
  99 }

 133 address TemplateInterpreterGenerator::generate_StackOverflowError_handler() {
 134   address entry = __ pc();
 135 
 136   // Note: There should be a minimal interpreter frame set up when stack
 137   // overflow occurs since we check explicitly for it now.
 138   //
 139 #ifdef ASSERT
 140   { Label L;
 141     __ sub(Rtemp, FP, - frame::interpreter_frame_monitor_block_top_offset * wordSize);
 142     __ cmp(SP, Rtemp);  // Rtemp = maximal SP for current FP,
 143                         //  (stack grows negative)
 144     __ b(L, ls); // check if frame is complete
 145     __ stop ("interpreter frame not set up");
 146     __ bind(L);
 147   }
 148 #endif // ASSERT
 149 
 150   // Restore bcp under the assumption that the current frame is still
 151   // interpreted
 152   __ restore_bcp();
 153 
 154   // expression stack must be empty before entering the VM if an exception
 155   // happened
 156   __ empty_expression_stack();
 157 
 158   // throw exception
 159   __ call_VM(noreg, CAST_FROM_FN_PTR(address, InterpreterRuntime::throw_StackOverflowError));
 160 
 161   __ should_not_reach_here();
 162 
 163   return entry;
 164 }

 207 address TemplateInterpreterGenerator::generate_exception_handler_common(const char* name, const char* message, bool pass_oop) {
 208   assert(!pass_oop || message == NULL, "either oop or message but not both");
 209   address entry = __ pc();
 210 
 211   InlinedString Lname(name);
 212   InlinedString Lmessage(message);
 213 
 214   if (pass_oop) {
 215     // object is at TOS
 216     __ pop_ptr(R2);
 217   }
 218 
 219   // expression stack must be empty before entering the VM if an exception happened
 220   __ empty_expression_stack();
 221 
 222   // setup parameters
 223   __ ldr_literal(R1, Lname);
 224 
 225   if (pass_oop) {
 226     __ call_VM(Rexception_obj, CAST_FROM_FN_PTR(address, InterpreterRuntime::create_klass_exception), R1, R2);
 227   } else {
 228     if (message != NULL) {
 229       __ ldr_literal(R2, Lmessage);
 230     } else {
 231       __ mov(R2, 0);
 232     }
 233     __ call_VM(Rexception_obj, CAST_FROM_FN_PTR(address, InterpreterRuntime::create_exception), R1, R2);
 234   }
 235 
 236   // throw exception
 237   __ b(Interpreter::throw_exception_entry());
 238 
 239   __ nop(); // to avoid filling CPU pipeline with invalid instructions
 240   __ nop();
 241   __ bind_literal(Lname);
 242   if (!pass_oop && (message != NULL)) {
 243     __ bind_literal(Lmessage);
 244   }
 245 
 246   return entry;
 247 }

 249 address TemplateInterpreterGenerator::generate_return_entry_for(TosState state, int step, size_t index_size) {
 250   address entry = __ pc();
 251 
 252   __ interp_verify_oop(R0_tos, state, __FILE__, __LINE__);
 253 
 254   // Restore stack bottom in case i2c adjusted stack
 255   __ ldr(SP, Address(FP, frame::interpreter_frame_last_sp_offset * wordSize));
 256   // and NULL it as marker that SP is now tos until next java call
 257   __ mov(Rtemp, (int)NULL_WORD);
 258   __ str(Rtemp, Address(FP, frame::interpreter_frame_last_sp_offset * wordSize));
 259 
 260   __ restore_method();
 261   __ restore_bcp();
 262   __ restore_dispatch();
 263   __ restore_locals();
 264 
 265   const Register Rcache = R2_tmp;
 266   const Register Rindex = R3_tmp;
 267   __ get_cache_and_index_at_bcp(Rcache, Rindex, 1, index_size);
 268 
 269   __ add(Rtemp, Rcache, AsmOperand(Rindex, lsl, LogBytesPerWord));
 270   __ ldrb(Rtemp, Address(Rtemp, ConstantPoolCache::base_offset() + ConstantPoolCacheEntry::flags_offset()));
 271   __ check_stack_top();
 272   __ add(Rstack_top, Rstack_top, AsmOperand(Rtemp, lsl, Interpreter::logStackElementSize));
 273 
 274   __ convert_retval_to_tos(state);
 275 
 276  __ check_and_handle_popframe();
 277  __ check_and_handle_earlyret();
 278 
 279   __ dispatch_next(state, step);
 280 
 281   return entry;
 282 }

 285 address TemplateInterpreterGenerator::generate_deopt_entry_for(TosState state, int step, address continuation) {
 286   address entry = __ pc();
 287 
 288   __ interp_verify_oop(R0_tos, state, __FILE__, __LINE__);
 289 
 290   // The stack is not extended by deopt but we must NULL last_sp as this
 291   // entry is like a "return".
 292   __ mov(Rtemp, 0);
 293   __ str(Rtemp, Address(FP, frame::interpreter_frame_last_sp_offset * wordSize));
 294 
 295   __ restore_method();
 296   __ restore_bcp();
 297   __ restore_dispatch();
 298   __ restore_locals();
 299 
 300   // handle exceptions
 301   { Label L;
 302     __ ldr(Rtemp, Address(Rthread, Thread::pending_exception_offset()));
 303     __ cbz(Rtemp, L);
 304     __ call_VM(noreg, CAST_FROM_FN_PTR(address, InterpreterRuntime::throw_pending_exception));
 305     __ should_not_reach_here();
 306     __ bind(L);
 307   }
 308 
 309   if (continuation == NULL) {
 310     __ dispatch_next(state, step);
 311   } else {
 312     __ jump_to_entry(continuation);
 313   }
 314 
 315   return entry;
 316 }

 318 address TemplateInterpreterGenerator::generate_result_handler_for(BasicType type) {
 319   address entry = __ pc();
 320 
 321   switch (type) {
 322   case T_CHAR    : /* Nothing to do */  break;
 323   case T_BYTE    : /* Nothing to do */  break;
 324   case T_SHORT   : /* Nothing to do */  break;
 325   case T_INT     : /* Nothing to do */  break;
 326   case T_LONG    : /* Nothing to do */  break;
 327   case T_VOID    : /* Nothing to do */  break;
 328   case T_DOUBLE  : /* Nothing to do */  break;
 329   case T_FLOAT   : /* Nothing to do */  break;
 330   case T_BOOLEAN : __ c2bool(R0);       break;
 331   case T_OBJECT  :
 332     __ ldr(R0, Address(FP, frame::interpreter_frame_oop_temp_offset * wordSize));
 333     __ verify_oop(R0);
 334     break;
 335   default        : __ should_not_reach_here(); break;
 336   }
 337 
 338   __ ret();
 339   return entry;
 340 }

 357 // increment invocation count & check for overflow
 358 //
 359 // Note: checking for negative value instead of overflow
 360 //       so we have a 'sticky' overflow test
 361 //
 362 // In: Rmethod.
 363 //
 364 // Uses R0, R1, Rtemp.
 365 //
 366 void TemplateInterpreterGenerator::generate_counter_incr(Label* overflow,
 367                                                  Label* profile_method,
 368                                                  Label* profile_method_continue) {
 369   Label done;
 370   const Register Rcounters = Rtemp;
 371   const Address invocation_counter(Rcounters,
 372                 MethodCounters::invocation_counter_offset() +
 373                 InvocationCounter::counter_offset());
 374 
 375   // Note: In tiered we increment either counters in MethodCounters* or
 376   // in MDO depending if we're profiling or not.
 377   if (TieredCompilation) {
 378     int increment = InvocationCounter::count_increment;
 379     Label no_mdo;
 380     if (ProfileInterpreter) {
 381       // Are we profiling?
 382       __ ldr(R1_tmp, Address(Rmethod, Method::method_data_offset()));
 383       __ cbz(R1_tmp, no_mdo);
 384       // Increment counter in the MDO
 385       const Address mdo_invocation_counter(R1_tmp,
 386                     in_bytes(MethodData::invocation_counter_offset()) +
 387                     in_bytes(InvocationCounter::counter_offset()));
 388       const Address mask(R1_tmp, in_bytes(MethodData::invoke_mask_offset()));
 389       __ increment_mask_and_jump(mdo_invocation_counter, increment, mask, R0_tmp, Rtemp, eq, overflow);
 390       __ b(done);
 391     }
 392     __ bind(no_mdo);
 393     __ get_method_counters(Rmethod, Rcounters, done);
 394     const Address mask(Rcounters, in_bytes(MethodCounters::invoke_mask_offset()));
 395     __ increment_mask_and_jump(invocation_counter, increment, mask, R0_tmp, R1_tmp, eq, overflow);
 396     __ bind(done);
 397   } else { // not TieredCompilation
 398     const Address backedge_counter(Rcounters,
 399                   MethodCounters::backedge_counter_offset() +
 400                   InvocationCounter::counter_offset());
 401 
 402     const Register Ricnt = R0_tmp;  // invocation counter
 403     const Register Rbcnt = R1_tmp;  // backedge counter
 404 
 405     __ get_method_counters(Rmethod, Rcounters, done);
 406 
 407     if (ProfileInterpreter) {
 408       const Register Riic = R1_tmp;
 409       __ ldr_s32(Riic, Address(Rcounters, MethodCounters::interpreter_invocation_counter_offset()));
 410       __ add(Riic, Riic, 1);
 411       __ str_32(Riic, Address(Rcounters, MethodCounters::interpreter_invocation_counter_offset()));
 412     }
 413 
 414     // Update standard invocation counters
 415 
 416     __ ldr_u32(Ricnt, invocation_counter);
 417     __ ldr_u32(Rbcnt, backedge_counter);
 418 
 419     __ add(Ricnt, Ricnt, InvocationCounter::count_increment);
 420 
 421     __ bic(Rbcnt, Rbcnt, ~InvocationCounter::count_mask_value); // mask out the status bits
 422 
 423     __ str_32(Ricnt, invocation_counter);            // save invocation count
 424     __ add(Ricnt, Ricnt, Rbcnt);                     // add both counters
 425 
 426     // profile_method is non-null only for interpreted method so
 427     // profile_method != NULL == !native_call
 428     // BytecodeInterpreter only calls for native so code is elided.
 429 
 430     if (ProfileInterpreter && profile_method != NULL) {
 431       assert(profile_method_continue != NULL, "should be non-null");
 432 
 433       // Test to see if we should create a method data oop
 434       // Reuse R1_tmp as we don't need backedge counters anymore.
 435       Address profile_limit(Rcounters, in_bytes(MethodCounters::interpreter_profile_limit_offset()));
 436       __ ldr_s32(R1_tmp, profile_limit);
 437       __ cmp_32(Ricnt, R1_tmp);
 438       __ b(*profile_method_continue, lt);
 439 
 440       // if no method data exists, go to profile_method
 441       __ test_method_data_pointer(R1_tmp, *profile_method);
 442     }
 443 
 444     Address invoke_limit(Rcounters, in_bytes(MethodCounters::interpreter_invocation_limit_offset()));
 445     __ ldr_s32(R1_tmp, invoke_limit);
 446     __ cmp_32(Ricnt, R1_tmp);
 447     __ b(*overflow, hs);
 448     __ bind(done);
 449   }
 450 }

 453   // InterpreterRuntime::frequency_counter_overflow takes one argument
 454   // indicating if the counter overflow occurs at a backwards branch (non-NULL bcp).
 455   // The call returns the address of the verified entry point for the method or NULL

 461   __ b(do_continue);

 464 void TemplateInterpreterGenerator::generate_stack_overflow_check(void) {
 465   // Check if we've got enough room on the stack for
 466   //  - overhead;
 467   //  - locals;
 468   //  - expression stack.
 469   //
 470   // Registers on entry:
 471   //
 472   // R3 = number of additional locals
 473   // Rthread
 474   // Rmethod
 475   // Registers used: R0, R1, R2, Rtemp.
 476 
 477   const Register Radditional_locals = R3;
 478   const Register RmaxStack = R2;
 479 
 480   // monitor entry size
 481   const int entry_size = frame::interpreter_frame_monitor_size() * wordSize;
 482 
 483   // total overhead size: entry_size + (saved registers, thru expr stack bottom).
 484   // be sure to change this if you add/subtract anything to/from the overhead area
 485   const int overhead_size = (frame::sender_sp_offset - frame::interpreter_frame_initial_sp_offset)*wordSize + entry_size;
 486 
 487   // Pages reserved for VM runtime calls and subsequent Java calls.
 488   const int reserved_pages = JavaThread::stack_shadow_zone_size();
 489 
 490   // Thread::stack_size() includes guard pages, and they should not be touched.
 491   const int guard_pages = JavaThread::stack_guard_zone_size();
 492 
 493   __ ldr(R0, Address(Rthread, Thread::stack_base_offset()));
 494   __ ldr(R1, Address(Rthread, Thread::stack_size_offset()));
 495   __ ldr(Rtemp, Address(Rmethod, Method::const_offset()));
 496   __ ldrh(RmaxStack, Address(Rtemp, ConstMethod::max_stack_offset()));
 497   __ sub_slow(Rtemp, SP, overhead_size + reserved_pages + guard_pages + Method::extra_stack_words());
 498 
 499   // reserve space for additional locals
 500   __ sub(Rtemp, Rtemp, AsmOperand(Radditional_locals, lsl, Interpreter::logStackElementSize));
 501 
 502   // stack size
 503   __ sub(R0, R0, R1);
 504 
 505   // reserve space for expression stack
 506   __ sub(Rtemp, Rtemp, AsmOperand(RmaxStack, lsl, Interpreter::logStackElementSize));
 507 
 508   __ cmp(Rtemp, R0);
 509 
 510   __ mov(SP, Rsender_sp, ls);  // restore SP
 511   __ b(StubRoutines::throw_StackOverflowError_entry(), ls);
 512 }

 515 // Allocate monitor and lock method (asm interpreter)
 516 //
 517 void TemplateInterpreterGenerator::lock_method() {
 518   // synchronize method
 519 
 520   const int entry_size = frame::interpreter_frame_monitor_size() * wordSize;
 521   assert ((entry_size % StackAlignmentInBytes) == 0, "should keep stack alignment");
 522 
 523   #ifdef ASSERT
 524     { Label L;
 525       __ ldr_u32(Rtemp, Address(Rmethod, Method::access_flags_offset()));
 526       __ tbnz(Rtemp, JVM_ACC_SYNCHRONIZED_BIT, L);
 527       __ stop("method doesn't need synchronization");
 528       __ bind(L);
 529     }
 530   #endif // ASSERT
 531 
 532   // get synchronization object
 533   { Label done;
 534     __ ldr_u32(Rtemp, Address(Rmethod, Method::access_flags_offset()));
 535     __ tst(Rtemp, JVM_ACC_STATIC);
 536     __ ldr(R0, Address(Rlocals, Interpreter::local_offset_in_bytes(0)), eq); // get receiver (assume this is frequent case)
 537     __ b(done, eq);
 538     __ load_mirror(R0, Rmethod, Rtemp);
 539     __ bind(done);
 540     __ resolve(IS_NOT_NULL, R0);
 541   }
 542 
 543   // add space for monitor & lock
 544 
 545 
 546   __ sub(Rstack_top, Rstack_top, entry_size);
 547   __ check_stack_top_on_expansion();
 548                                               // add space for a monitor entry
 549   __ str(Rstack_top, Address(FP, frame::interpreter_frame_monitor_block_top_offset * wordSize));
 550                                               // set new monitor block top
 551   __ str(R0, Address(Rstack_top, BasicObjectLock::obj_offset_in_bytes()));
 552                                               // store object
 553   __ mov(R1, Rstack_top);                     // monitor entry address
 554   __ lock_object(R1);
 555 }

 558 //
 559 // Generate a fixed interpreter frame. This is identical setup for interpreted methods
 560 // and for native methods hence the shared code.
 561 
 562 void TemplateInterpreterGenerator::generate_fixed_frame(bool native_call) {
 563   // Generates the following stack layout:
 564   //
 565   // [ expr. stack bottom ]
 566   // [ saved Rbcp         ]
 567   // [ current Rlocals    ]
 568   // [ cache              ]
 569   // [ mdx                ]
 570   // [ Method*            ]
 571   // [ last_sp            ]
 572   // [ sender_sp          ]
 573   // [ saved FP           ] <--- FP
 574   // [ saved LR           ]
 575 
 576   // initialize fixed part of activation frame
 577   __ push(LR);                                        // save return address
 578   __ push(FP);                                        // save FP
 579   __ mov(FP, SP);                                     // establish new FP
 580 
 581   __ push(Rsender_sp);
 582 
 583   __ mov(R0, 0);
 584   __ push(R0);                                        // leave last_sp as null
 585 
 586   // setup Rbcp
 587   if (native_call) {
 588     __ mov(Rbcp, 0);                                  // bcp = 0 for native calls
 589   } else {
 590     __ ldr(Rtemp, Address(Rmethod, Method::const_offset())); // get ConstMethod*
 591     __ add(Rbcp, Rtemp, ConstMethod::codes_offset()); // get codebase
 592   }
 593 
 594   __ push(Rmethod);                                    // save Method*
 595   // Get mirror and store it in the frame as GC root for this Method*
 596   __ load_mirror(Rtemp, Rmethod, Rtemp);
 597   __ push(Rtemp);
 598 
 599   if (ProfileInterpreter) {
 600     __ ldr(Rtemp, Address(Rmethod, Method::method_data_offset()));
 601     __ tst(Rtemp, Rtemp);
 602     __ add(Rtemp, Rtemp, in_bytes(MethodData::data_offset()), ne);
 603     __ push(Rtemp);                                    // set the mdp (method data pointer)
 604   } else {
 605     __ push(R0);
 606   }
 607 
 608   __ ldr(Rtemp, Address(Rmethod, Method::const_offset()));
 609   __ ldr(Rtemp, Address(Rtemp, ConstMethod::constants_offset()));
 610   __ ldr(Rtemp, Address(Rtemp, ConstantPool::cache_offset_in_bytes()));
 611   __ push(Rtemp);                                      // set constant pool cache
 612   __ push(Rlocals);                                    // set locals pointer
 613   __ push(Rbcp);                                       // set bcp
 614   __ push(R0);                                         // reserve word for pointer to expression stack bottom
 615   __ str(SP, Address(SP, 0));                          // set expression stack bottom
 616 }

 621 //------------------------------------------------------------------------------------------------------------------------
 622 // Entry points
 623 //
 624 // Here we generate the various kind of entries into the interpreter.
 625 // The two main entry type are generic bytecode methods and native call method.
 626 // These both come in synchronized and non-synchronized versions but the
 627 // frame layout they create is very similar. The other method entry
 628 // types are really just special purpose entries that are really entry
 629 // and interpretation all in one. These are for trivial methods like
 630 // accessor, empty, or special math methods.
 631 //
 632 // When control flow reaches any of the entry types for the interpreter
 633 // the following holds ->
 634 //
 635 // Arguments:
 636 //
 637 // Rmethod: Method*
 638 // Rthread: thread
 639 // Rsender_sp:  sender sp
 640 // Rparams (SP on 32-bit ARM): pointer to method parameters
 641 //
 642 // LR: return address
 643 //
 644 // Stack layout immediately at entry
 645 //
 646 // [ parameter n        ] <--- Rparams (SP on 32-bit ARM)
 647 //   ...
 648 // [ parameter 1        ]
 649 // [ expression stack   ] (caller's java expression stack)
 650 
 651 // Assuming that we don't go to one of the trivial specialized
 652 // entries the stack will look like below when we are ready to execute
 653 // the first bytecode (or call the native routine). The register usage
 654 // will be as the template based interpreter expects.
 655 //
 656 // local variables follow incoming parameters immediately; i.e.
 657 // the return address is saved at the end of the locals.
 658 //
 659 // [ expr. stack        ] <--- Rstack_top (SP on 32-bit ARM)
 660 // [ monitor entry      ]
 661 //   ...
 662 // [ monitor entry      ]
 663 // [ expr. stack bottom ]
 664 // [ saved Rbcp         ]
 665 // [ current Rlocals    ]
 666 // [ cache              ]
 667 // [ mdx                ]
 668 // [ mirror             ]
 669 // [ Method*            ]
 670 //
 671 // 32-bit ARM:
 672 // [ last_sp            ]
 673 //
 674 // [ sender_sp          ]
 675 // [ saved FP           ] <--- FP
 676 // [ saved LR           ]
 677 // [ optional padding(*)]
 678 // [ local variable m   ]
 679 //   ...
 680 // [ local variable 1   ]
 681 // [ parameter n        ]
 682 //   ...
 683 // [ parameter 1        ] <--- Rlocals
 684 //
 685 
 686 address TemplateInterpreterGenerator::generate_Reference_get_entry(void) {
 687   // Code: _aload_0, _getfield, _areturn
 688   // parameter size = 1
 689   //
 690   // The code that gets generated by this routine is split into 2 parts:
 691   //    1. The "intrinsified" code performing an ON_WEAK_OOP_REF load,
 692   //    2. The slow path - which is an expansion of the regular method entry.
 693   //
 694   // Notes:-
 695   // * An intrinsic is always executed, where an ON_WEAK_OOP_REF load is performed.
 696   // * We may jump to the slow path iff the receiver is null. If the
 697   //   Reference object is null then we no longer perform an ON_WEAK_OOP_REF load
 698   //   Thus we can use the regular method entry code to generate the NPE.
 699   //
 700   // Rmethod: Method*
 701   // Rthread: thread
 702   // Rsender_sp: sender sp, must be preserved for slow path, set SP to it on fast path
 703   // Rparams: parameters
 704 
 705   address entry = __ pc();
 706   Label slow_path;
 707   const Register Rthis = R0;
 708   const Register Rret_addr = Rtmp_save1;
 709   assert_different_registers(Rthis, Rret_addr, Rsender_sp);
 710 
 711   const int referent_offset = java_lang_ref_Reference::referent_offset();
 712 
 713   // Check if local 0 != NULL
 714   // If the receiver is null then it is OK to jump to the slow path.
 715   __ ldr(Rthis, Address(Rparams));
 716   __ cbz(Rthis, slow_path);
 717 
 718   // Preserve LR
 719   __ mov(Rret_addr, LR);
 720 
 721   // Load the value of the referent field.
 722   const Address field_address(Rthis, referent_offset);
 723   __ load_heap_oop(R0, field_address, Rtemp, R1_tmp, R2_tmp, ON_WEAK_OOP_REF);
 724 
 725   // _areturn
 726   __ mov(SP, Rsender_sp);
 727   __ ret(Rret_addr);
 728 
 729   // generate a vanilla interpreter entry as the slow path
 730   __ bind(slow_path);
 731   __ jump_to_entry(Interpreter::entry_for_kind(Interpreter::zerolocals));
 732   return entry;
 733 }

 740 //
 741 // Interpreter stub for calling a native method. (asm interpreter)
 742 // This sets up a somewhat different looking stack for calling the native method
 743 // than the typical interpreter frame setup.
 744 //
 745 
 746 address TemplateInterpreterGenerator::generate_native_entry(bool synchronized) {
 747   // determine code generation flags
 748   bool inc_counter  = UseCompiler || CountCompiledCalls || LogTouchedMethods;
 749 
 750   // Incoming registers:
 751   //
 752   // Rmethod: Method*
 753   // Rthread: thread
 754   // Rsender_sp: sender sp
 755   // Rparams: parameters
 756 
 757   address entry_point = __ pc();
 758 
 759   // Register allocation
 760   const Register Rsize_of_params = R6;
 761   const Register Rsig_handler    = Rtmp_save0;   // R4
 762   const Register Rnative_code    = Rtmp_save1;   // R5
 763   const Register Rresult_handler = R6;
 764 
 765   const Register Rsaved_result_lo = Rtmp_save0;  // R4
 766   const Register Rsaved_result_hi = Rtmp_save1;  // R5
 767   FloatRegister saved_result_fp;
 768 
 769 
 770   __ ldr(Rsize_of_params, Address(Rmethod, Method::const_offset()));
 771   __ ldrh(Rsize_of_params,  Address(Rsize_of_params, ConstMethod::size_of_parameters_offset()));
 772 
 773   // native calls don't need the stack size check since they have no expression stack
 774   // and the arguments are already on the stack and we only add a handful of words
 775   // to the stack
 776 
 777   // compute beginning of parameters (Rlocals)
 778   __ sub(Rlocals, Rparams, wordSize);
 779   __ add(Rlocals, Rlocals, AsmOperand(Rsize_of_params, lsl, Interpreter::logStackElementSize));
 780 
 781   // reserve stack space for oop_temp
 782   __ mov(R0, 0);
 783   __ push(R0);
 784 
 785   generate_fixed_frame(true); // Note: R9 is now saved in the frame
 786 
 787   // make sure method is native & not abstract
 788 #ifdef ASSERT
 789   __ ldr_u32(Rtemp, Address(Rmethod, Method::access_flags_offset()));
 790   {
 791     Label L;
 792     __ tbnz(Rtemp, JVM_ACC_NATIVE_BIT, L);
 793     __ stop("tried to execute non-native method as native");
 794     __ bind(L);
 795   }
 796   { Label L;
 797     __ tbz(Rtemp, JVM_ACC_ABSTRACT_BIT, L);
 798     __ stop("tried to execute abstract method in interpreter");
 799     __ bind(L);
 800   }
 801 #endif
 802 
 803   // increment invocation count & check for overflow
 804   Label invocation_counter_overflow;
 805   if (inc_counter) {
 806     if (synchronized) {
 807       // Avoid unlocking method's monitor in case of exception, as it has not
 808       // been locked yet.
 809       __ set_do_not_unlock_if_synchronized(true, Rtemp);
 810     }
 811     generate_counter_incr(&invocation_counter_overflow, NULL, NULL);
 812   }
 813 
 814   Label continue_after_compile;
 815   __ bind(continue_after_compile);
 816 
 817   if (inc_counter && synchronized) {
 818     __ set_do_not_unlock_if_synchronized(false, Rtemp);
 819   }
 820 
 821   // check for synchronized methods
 822   // Must happen AFTER invocation_counter check and stack overflow check,
 823   // so method is not locked if overflows.
 824   //
 825   if (synchronized) {
 826     lock_method();
 827   } else {
 828     // no synchronization necessary
 829 #ifdef ASSERT
 830       { Label L;
 831         __ ldr_u32(Rtemp, Address(Rmethod, Method::access_flags_offset()));
 832         __ tbz(Rtemp, JVM_ACC_SYNCHRONIZED_BIT, L);
 833         __ stop("method needs synchronization");
 834         __ bind(L);
 835       }
 836 #endif
 837   }
 838 
 839   // start execution
 840 #ifdef ASSERT
 841   { Label L;
 842     __ ldr(Rtemp, Address(FP, frame::interpreter_frame_monitor_block_top_offset * wordSize));
 843     __ cmp(Rtemp, Rstack_top);
 844     __ b(L, eq);
 845     __ stop("broken stack frame setup in interpreter");
 846     __ bind(L);
 847   }
 848 #endif
 849   __ check_extended_sp(Rtemp);
 850 
 851   // jvmti/dtrace support
 852   __ notify_method_entry();
 853 #if R9_IS_SCRATCHED
 854   __ restore_method();
 855 #endif
 856 
 857   {
 858     Label L;
 859     __ ldr(Rsig_handler, Address(Rmethod, Method::signature_handler_offset()));
 860     __ cbnz(Rsig_handler, L);
 861     __ mov(R1, Rmethod);
 862     __ call_VM(noreg, CAST_FROM_FN_PTR(address, InterpreterRuntime::prepare_native_call), R1, true);
 863     __ ldr(Rsig_handler, Address(Rmethod, Method::signature_handler_offset()));
 864     __ bind(L);
 865   }
 866 
 867   {
 868     Label L;
 869     __ ldr(Rnative_code, Address(Rmethod, Method::native_function_offset()));
 870     __ cbnz(Rnative_code, L);
 871     __ mov(R1, Rmethod);
 872     __ call_VM(noreg, CAST_FROM_FN_PTR(address, InterpreterRuntime::prepare_native_call), R1);
 873     __ ldr(Rnative_code, Address(Rmethod, Method::native_function_offset()));
 874     __ bind(L);
 875   }
 876 
 877   // Allocate stack space for arguments
 878 
 879 
 880   // C functions need aligned stack
 881   __ bic(SP, SP, StackAlignmentInBytes - 1);
 882   // Multiply by BytesPerLong instead of BytesPerWord, because calling convention
 883   // may require empty slots due to long alignment, e.g. func(int, jlong, int, jlong)
 884   __ sub(SP, SP, AsmOperand(Rsize_of_params, lsl, LogBytesPerLong));
 885 
 886 #ifdef __ABI_HARD__
 887   // Allocate more stack space to accomodate all GP as well as FP registers:
 888   // 4 * wordSize
 889   // 8 * BytesPerLong
 890   int reg_arguments = align_up((4*wordSize) + (8*BytesPerLong), StackAlignmentInBytes);
 891 #else
 892   // Reserve at least 4 words on the stack for loading
 893   // of parameters passed on registers (R0-R3).
 894   // See generate_slow_signature_handler().
 895   // It is also used for JNIEnv & class additional parameters.
 896   int reg_arguments = 4 * wordSize;
 897 #endif // __ABI_HARD__
 898 
 899   __ sub(SP, SP, reg_arguments);
 900 
 901 
 902   // Note: signature handler blows R4 besides all scratch registers.
 903   // See AbstractInterpreterGenerator::generate_slow_signature_handler().
 904   __ call(Rsig_handler);
 905 #if R9_IS_SCRATCHED
 906   __ restore_method();
 907 #endif
 908   __ mov(Rresult_handler, R0);
 909 
 910   // Pass JNIEnv and mirror for static methods
 911   {
 912     Label L;
 913     __ ldr_u32(Rtemp, Address(Rmethod, Method::access_flags_offset()));
 914     __ add(R0, Rthread, in_bytes(JavaThread::jni_environment_offset()));
 915     __ tbz(Rtemp, JVM_ACC_STATIC_BIT, L);
 916     __ load_mirror(Rtemp, Rmethod, Rtemp);
 917     __ add(R1, FP, frame::interpreter_frame_oop_temp_offset * wordSize);
 918     __ str(Rtemp, Address(R1, 0));
 919     __ bind(L);
 920   }
 921 
 922   __ set_last_Java_frame(SP, FP, true, Rtemp);
 923 
 924   // Changing state to _thread_in_native must be the last thing to do
 925   // before the jump to native code. At this moment stack must be
 926   // safepoint-safe and completely prepared for stack walking.
 927 #ifdef ASSERT
 928   {
 929     Label L;
 930     __ ldr_u32(Rtemp, Address(Rthread, JavaThread::thread_state_offset()));
 931     __ cmp_32(Rtemp, _thread_in_Java);
 932     __ b(L, eq);
 933     __ stop("invalid thread state");
 934     __ bind(L);
 935   }
 936 #endif
 937 
 938   // Force all preceding writes to be observed prior to thread state change
 939   __ membar(MacroAssembler::StoreStore, Rtemp);
 940 
 941   __ mov(Rtemp, _thread_in_native);
 942   __ str(Rtemp, Address(Rthread, JavaThread::thread_state_offset()));
 943 
 944   __ call(Rnative_code);
 945 #if R9_IS_SCRATCHED
 946   __ restore_method();
 947 #endif
 948 
 949   // Set FPSCR/FPCR to a known state
 950   if (AlwaysRestoreFPU) {
 951     __ restore_default_fp_mode();
 952   }
 953 
 954   // Do safepoint check
 955   __ mov(Rtemp, _thread_in_native_trans);
 956   __ str_32(Rtemp, Address(Rthread, JavaThread::thread_state_offset()));
 957 
 958     // Force this write out before the read below
 959   __ membar(MacroAssembler::StoreLoad, Rtemp);
 960 
 961   // Protect the return value in the interleaved code: save it to callee-save registers.
 962   __ mov(Rsaved_result_lo, R0);
 963   __ mov(Rsaved_result_hi, R1);
 964 #ifdef __ABI_HARD__
 965   // preserve native FP result in a callee-saved register
 966   saved_result_fp = D8;
 967   __ fcpyd(saved_result_fp, D0);
 968 #else
 969   saved_result_fp = fnoreg;
 970 #endif // __ABI_HARD__
 971 
 972   {
 973   Label call, skip_call;
 974   __ safepoint_poll(Rtemp, call);
 975   __ ldr_u32(R3, Address(Rthread, JavaThread::suspend_flags_offset()));
 976   __ cmp(R3, 0);
 977   __ b(skip_call, eq);
 978   __ bind(call);
 979   __ mov(R0, Rthread);
 980   __ call(CAST_FROM_FN_PTR(address, JavaThread::check_special_condition_for_native_trans), relocInfo::none);
 981   __ bind(skip_call);
 982 
 983 #if R9_IS_SCRATCHED
 984   __ restore_method();
 985 #endif
 986   }
 987 
 988   // Perform Native->Java thread transition
 989   __ mov(Rtemp, _thread_in_Java);
 990   __ str_32(Rtemp, Address(Rthread, JavaThread::thread_state_offset()));
 991 
 992   // Zero handles and last_java_sp
 993   __ reset_last_Java_frame(Rtemp);
 994   __ ldr(R3, Address(Rthread, JavaThread::active_handles_offset()));
 995   __ str_32(__ zero_register(Rtemp), Address(R3, JNIHandleBlock::top_offset_in_bytes()));
 996   if (CheckJNICalls) {
 997     __ str(__ zero_register(Rtemp), Address(Rthread, JavaThread::pending_jni_exception_check_fn_offset()));
 998   }
 999 
1000   // Unbox oop result, e.g. JNIHandles::resolve result if it's an oop.
1001   {
1002     Label Lnot_oop;
1003     __ mov_slow(Rtemp, AbstractInterpreter::result_handler(T_OBJECT));
1004     __ cmp(Rtemp, Rresult_handler);
1005     __ b(Lnot_oop, ne);
1006     Register value = Rsaved_result_lo;
1007     __ resolve_jobject(value,   // value
1008                        Rtemp,   // tmp1
1009                        R1_tmp); // tmp2
1010     // Store resolved result in frame for GC visibility.
1011     __ str(value, Address(FP, frame::interpreter_frame_oop_temp_offset * wordSize));
1012     __ bind(Lnot_oop);
1013   }
1014 
1015 
1016   // reguard stack if StackOverflow exception happened while in native.
1017   {
1018     __ ldr_u32(Rtemp, Address(Rthread, JavaThread::stack_guard_state_offset()));
1019     __ cmp_32(Rtemp, JavaThread::stack_guard_yellow_reserved_disabled);
1020   __ call(CAST_FROM_FN_PTR(address, SharedRuntime::reguard_yellow_pages), relocInfo::none, eq);
1021 #if R9_IS_SCRATCHED
1022   __ restore_method();
1023 #endif
1024   }
1025 
1026   // check pending exceptions
1027   {
1028     __ ldr(Rtemp, Address(Rthread, Thread::pending_exception_offset()));
1029     __ cmp(Rtemp, 0);
1030     __ mov(Rexception_pc, PC, ne);
1031     __ b(StubRoutines::forward_exception_entry(), ne);
1032   }
1033 
1034   if (synchronized) {
1035     // address of first monitor
1036     __ sub(R1, FP, - (frame::interpreter_frame_monitor_block_bottom_offset - frame::interpreter_frame_monitor_size()) * wordSize);
1037     __ unlock_object(R1);
1038   }
1039 
1040   // jvmti/dtrace support
1041   // Note: This must happen _after_ handling/throwing any exceptions since
1042   //       the exception handler code notifies the runtime of method exits
1043   //       too. If this happens before, method entry/exit notifications are
1044   //       not properly paired (was bug - gri 11/22/99).
1045   __ notify_method_exit(vtos, InterpreterMacroAssembler::NotifyJVMTI, true, Rsaved_result_lo, Rsaved_result_hi, saved_result_fp);
1046 
1047   // Restore the result. Oop result is restored from the stack by the
1048   // result handler.
1049   __ mov(R0, Rsaved_result_lo);
1050   __ mov(R1, Rsaved_result_hi);
1051 
1052 #ifdef __ABI_HARD__
1053   // reload native FP result
1054   __ fcpyd(D0, D8);
1055 #endif // __ABI_HARD__
1056 
1057   __ blx(Rresult_handler);
1058 
1059   // Restore FP/LR, sender_sp and return
1060   __ mov(Rtemp, FP);
1061   __ ldmia(FP, RegisterSet(FP) | RegisterSet(LR));
1062   __ ldr(SP, Address(Rtemp, frame::interpreter_frame_sender_sp_offset * wordSize));
1063 
1064   __ ret();
1065 
1066   if (inc_counter) {
1067     // Handle overflow of counter and compile method
1068     __ bind(invocation_counter_overflow);
1069     generate_counter_overflow(continue_after_compile);
1070   }
1071 
1072   return entry_point;
1073 }

1076 // Generic interpreted method entry to (asm) interpreter
1077 //
1078 address TemplateInterpreterGenerator::generate_normal_entry(bool synchronized) {
1079   // determine code generation flags
1080   bool inc_counter  = UseCompiler || CountCompiledCalls || LogTouchedMethods;
1081 
1082   // Rmethod: Method*
1083   // Rthread: thread
1084   // Rsender_sp: sender sp (could differ from SP if we were called via c2i)
1085   // Rparams: pointer to the last parameter in the stack
1086 
1087   address entry_point = __ pc();
1088 
1089   const Register RconstMethod = R3;
1090 
1091 
1092   __ ldr(RconstMethod, Address(Rmethod, Method::const_offset()));
1093 
1094   __ ldrh(R2, Address(RconstMethod, ConstMethod::size_of_parameters_offset()));
1095   __ ldrh(R3, Address(RconstMethod, ConstMethod::size_of_locals_offset()));
1096 
1097   // setup Rlocals
1098   __ sub(Rlocals, Rparams, wordSize);
1099   __ add(Rlocals, Rlocals, AsmOperand(R2, lsl, Interpreter::logStackElementSize));
1100 
1101   __ sub(R3, R3, R2); // number of additional locals
1102 
1103 
1104   // see if we've got enough room on the stack for locals plus overhead.
1105   generate_stack_overflow_check();
1106 
1107   // allocate space for locals
1108   // explicitly initialize locals
1109 
1110   // Loop is unrolled 4 times
1111   Label loop;
1112   __ mov(R0, 0);
1113   __ bind(loop);
1114 
1115   // #1
1116   __ subs(R3, R3, 1);
1117   __ push(R0, ge);
1118 
1119   // #2
1120   __ subs(R3, R3, 1, ge);
1121   __ push(R0, ge);
1122 
1123   // #3
1124   __ subs(R3, R3, 1, ge);
1125   __ push(R0, ge);
1126 
1127   // #4
1128   __ subs(R3, R3, 1, ge);
1129   __ push(R0, ge);
1130 
1131   __ b(loop, gt);
1132 
1133   // initialize fixed part of activation frame
1134   generate_fixed_frame(false);
1135 
1136   __ restore_dispatch();
1137 
1138   // make sure method is not native & not abstract
1139 #ifdef ASSERT
1140   __ ldr_u32(Rtemp, Address(Rmethod, Method::access_flags_offset()));
1141   {
1142     Label L;
1143     __ tbz(Rtemp, JVM_ACC_NATIVE_BIT, L);
1144     __ stop("tried to execute native method as non-native");
1145     __ bind(L);
1146   }
1147   { Label L;
1148     __ tbz(Rtemp, JVM_ACC_ABSTRACT_BIT, L);
1149     __ stop("tried to execute abstract method in interpreter");
1150     __ bind(L);
1151   }
1152 #endif
1153 
1154   // increment invocation count & check for overflow
1155   Label invocation_counter_overflow;
1156   Label profile_method;
1157   Label profile_method_continue;
1158   if (inc_counter) {
1159     if (synchronized) {
1160       // Avoid unlocking method's monitor in case of exception, as it has not
1161       // been locked yet.
1162       __ set_do_not_unlock_if_synchronized(true, Rtemp);
1163     }
1164     generate_counter_incr(&invocation_counter_overflow, &profile_method, &profile_method_continue);
1165     if (ProfileInterpreter) {
1166       __ bind(profile_method_continue);
1167     }
1168   }
1169   Label continue_after_compile;
1170   __ bind(continue_after_compile);
1171 
1172   if (inc_counter && synchronized) {
1173     __ set_do_not_unlock_if_synchronized(false, Rtemp);
1174   }
1175 #if R9_IS_SCRATCHED
1176   __ restore_method();
1177 #endif
1178 
1179   // check for synchronized methods
1180   // Must happen AFTER invocation_counter check and stack overflow check,
1181   // so method is not locked if overflows.
1182   //
1183   if (synchronized) {
1184     // Allocate monitor and lock method
1185     lock_method();
1186   } else {
1187     // no synchronization necessary
1188 #ifdef ASSERT
1189       { Label L;
1190         __ ldr_u32(Rtemp, Address(Rmethod, Method::access_flags_offset()));
1191         __ tbz(Rtemp, JVM_ACC_SYNCHRONIZED_BIT, L);
1192         __ stop("method needs synchronization");
1193         __ bind(L);
1194       }
1195 #endif
1196   }
1197 
1198   // start execution
1199 #ifdef ASSERT
1200   { Label L;
1201     __ ldr(Rtemp, Address(FP, frame::interpreter_frame_monitor_block_top_offset * wordSize));
1202     __ cmp(Rtemp, Rstack_top);
1203     __ b(L, eq);
1204     __ stop("broken stack frame setup in interpreter");
1205     __ bind(L);
1206   }
1207 #endif
1208   __ check_extended_sp(Rtemp);
1209 
1210   // jvmti support
1211   __ notify_method_entry();
1212 #if R9_IS_SCRATCHED
1213   __ restore_method();
1214 #endif
1215 
1216   __ dispatch_next(vtos);
1217 
1218   // invocation counter overflow
1219   if (inc_counter) {
1220     if (ProfileInterpreter) {
1221       // We have decided to profile this method in the interpreter
1222       __ bind(profile_method);
1223 
1224       __ call_VM(noreg, CAST_FROM_FN_PTR(address, InterpreterRuntime::profile_method));
1225       __ set_method_data_pointer_for_bcp();
1226 
1227       __ b(profile_method_continue);
1228     }
1229 
1230     // Handle overflow of counter and compile method
1231     __ bind(invocation_counter_overflow);
1232     generate_counter_overflow(continue_after_compile);
1233   }
1234 
1235   return entry_point;
1236 }

1241 void TemplateInterpreterGenerator::generate_throw_exception() {
1242   // Entry point in previous activation (i.e., if the caller was interpreted)
1243   Interpreter::_rethrow_exception_entry = __ pc();
1244   // Rexception_obj: exception
1245 
1246   // Clear interpreter_frame_last_sp.
1247   __ mov(Rtemp, 0);
1248   __ str(Rtemp, Address(FP, frame::interpreter_frame_last_sp_offset * wordSize));
1249 
1250 #if R9_IS_SCRATCHED
1251   __ restore_method();
1252 #endif
1253   __ restore_bcp();
1254   __ restore_dispatch();
1255   __ restore_locals();
1256 
1257 
1258   // Entry point for exceptions thrown within interpreter code
1259   Interpreter::_throw_exception_entry = __ pc();
1260 
1261   // expression stack is undefined here
1262   // Rexception_obj: exception
1263   // Rbcp: exception bcp
1264   __ verify_oop(Rexception_obj);
1265 
1266   // expression stack must be empty before entering the VM in case of an exception
1267   __ empty_expression_stack();
1268   // find exception handler address and preserve exception oop
1269   __ mov(R1, Rexception_obj);
1270   __ call_VM(Rexception_obj, CAST_FROM_FN_PTR(address, InterpreterRuntime::exception_handler_for_exception), R1);
1271   // R0: exception handler entry point
1272   // Rexception_obj: preserved exception oop
1273   // Rbcp: bcp for exception handler
1274   __ push_ptr(Rexception_obj);                    // push exception which is now the only value on the stack
1275   __ jump(R0);                                    // jump to exception handler (may be _remove_activation_entry!)
1276 
1277   // If the exception is not handled in the current frame the frame is removed and
1278   // the exception is rethrown (i.e. exception continuation is _rethrow_exception).
1279   //
1280   // Note: At this point the bci is still the bxi for the instruction which caused
1281   //       the exception and the expression stack is empty. Thus, for any VM calls
1282   //       at this point, GC will find a legal oop map (with empty expression stack).
1283 
1284   // In current activation
1285   // tos: exception
1286   // Rbcp: exception bcp
1287 
1288   //
1289   // JVMTI PopFrame support
1290   //
1291    Interpreter::_remove_activation_preserving_args_entry = __ pc();
1292 
1293 
1294   __ empty_expression_stack();
1295 
1296   // Set the popframe_processing bit in _popframe_condition indicating that we are
1297   // currently handling popframe, so that call_VMs that may happen later do not trigger new
1298   // popframe handling cycles.
1299 
1300   __ ldr_s32(Rtemp, Address(Rthread, JavaThread::popframe_condition_offset()));
1301   __ orr(Rtemp, Rtemp, (unsigned)JavaThread::popframe_processing_bit);
1302   __ str_32(Rtemp, Address(Rthread, JavaThread::popframe_condition_offset()));
1303 
1304   {
1305     // Check to see whether we are returning to a deoptimized frame.
1306     // (The PopFrame call ensures that the caller of the popped frame is
1307     // either interpreted or compiled and deoptimizes it if compiled.)
1308     // In this case, we can't call dispatch_next() after the frame is
1309     // popped, but instead must save the incoming arguments and restore
1310     // them after deoptimization has occurred.
1311     //
1312     // Note that we don't compare the return PC against the
1313     // deoptimization blob's unpack entry because of the presence of
1314     // adapter frames in C2.
1315     Label caller_not_deoptimized;
1316     __ ldr(R0, Address(FP, frame::return_addr_offset * wordSize));
1317     __ call_VM_leaf(CAST_FROM_FN_PTR(address, InterpreterRuntime::interpreter_contains), R0);
1318     __ cbnz_32(R0, caller_not_deoptimized);
1319 
1320     // Compute size of arguments for saving when returning to deoptimized caller
1321     __ restore_method();
1322     __ ldr(R0, Address(Rmethod, Method::const_offset()));
1323     __ ldrh(R0, Address(R0, ConstMethod::size_of_parameters_offset()));
1324 
1325     __ logical_shift_left(R1, R0, Interpreter::logStackElementSize);
1326     // Save these arguments
1327     __ restore_locals();
1328     __ sub(R2, Rlocals, R1);
1329     __ add(R2, R2, wordSize);
1330     __ mov(R0, Rthread);
1331     __ call_VM_leaf(CAST_FROM_FN_PTR(address, Deoptimization::popframe_preserve_args), R0, R1, R2);
1332 
1333     __ remove_activation(vtos, LR,
1334                          /* throw_monitor_exception */ false,
1335                          /* install_monitor_exception */ false,
1336                          /* notify_jvmdi */ false);
1337 
1338     // Inform deoptimization that it is responsible for restoring these arguments
1339     __ mov(Rtemp, JavaThread::popframe_force_deopt_reexecution_bit);
1340     __ str_32(Rtemp, Address(Rthread, JavaThread::popframe_condition_offset()));
1341 
1342     // Continue in deoptimization handler
1343     __ ret();
1344 
1345     __ bind(caller_not_deoptimized);
1346   }
1347 
1348   __ remove_activation(vtos, R4,
1349                        /* throw_monitor_exception */ false,
1350                        /* install_monitor_exception */ false,
1351                        /* notify_jvmdi */ false);
1352 
1353   // Finish with popframe handling
1354   // A previous I2C followed by a deoptimization might have moved the
1355   // outgoing arguments further up the stack. PopFrame expects the
1356   // mutations to those outgoing arguments to be preserved and other
1357   // constraints basically require this frame to look exactly as
1358   // though it had previously invoked an interpreted activation with
1359   // no space between the top of the expression stack (current
1360   // last_sp) and the top of stack. Rather than force deopt to
1361   // maintain this kind of invariant all the time we call a small
1362   // fixup routine to move the mutated arguments onto the top of our
1363   // expression stack if necessary.
1364   __ mov(R1, SP);
1365   __ ldr(R2, Address(FP, frame::interpreter_frame_last_sp_offset * wordSize));
1366   // PC must point into interpreter here
1367   __ set_last_Java_frame(SP, FP, true, Rtemp);
1368   __ mov(R0, Rthread);
1369   __ call_VM_leaf(CAST_FROM_FN_PTR(address, InterpreterRuntime::popframe_move_outgoing_args), R0, R1, R2);
1370   __ reset_last_Java_frame(Rtemp);
1371 
1372   // Restore the last_sp and null it out
1373   __ ldr(SP, Address(FP, frame::interpreter_frame_last_sp_offset * wordSize));
1374   __ mov(Rtemp, (int)NULL_WORD);
1375   __ str(Rtemp, Address(FP, frame::interpreter_frame_last_sp_offset * wordSize));
1376 
1377   __ restore_bcp();
1378   __ restore_dispatch();
1379   __ restore_locals();
1380   __ restore_method();
1381 
1382   // The method data pointer was incremented already during
1383   // call profiling. We have to restore the mdp for the current bcp.
1384   if (ProfileInterpreter) {
1385     __ set_method_data_pointer_for_bcp();
1386   }
1387 
1388   // Clear the popframe condition flag
1389   assert(JavaThread::popframe_inactive == 0, "adjust this code");
1390   __ str_32(__ zero_register(Rtemp), Address(Rthread, JavaThread::popframe_condition_offset()));
1391 
1392 #if INCLUDE_JVMTI
1393   {
1394     Label L_done;
1395 
1396     __ ldrb(Rtemp, Address(Rbcp, 0));
1397     __ cmp(Rtemp, Bytecodes::_invokestatic);
1398     __ b(L_done, ne);
1399 
1400     // The member name argument must be restored if _invokestatic is re-executed after a PopFrame call.
1401     // Detect such a case in the InterpreterRuntime function and return the member name argument, or NULL.
1402 
1403     // get local0
1404     __ ldr(R1, Address(Rlocals, 0));
1405     __ mov(R2, Rmethod);
1406     __ mov(R3, Rbcp);
1407     __ call_VM(R0, CAST_FROM_FN_PTR(address, InterpreterRuntime::member_name_arg_or_null), R1, R2, R3);
1408 
1409     __ cbz(R0, L_done);
1410 
1411     __ str(R0, Address(Rstack_top));
1412     __ bind(L_done);
1413   }
1414 #endif // INCLUDE_JVMTI
1415 
1416   __ dispatch_next(vtos);
1417   // end of PopFrame support
1418 
1419   Interpreter::_remove_activation_entry = __ pc();
1420 
1421   // preserve exception over this code sequence
1422   __ pop_ptr(R0_tos);
1423   __ str(R0_tos, Address(Rthread, JavaThread::vm_result_offset()));
1424   // remove the activation (without doing throws on illegalMonitorExceptions)
1425   __ remove_activation(vtos, Rexception_pc, false, true, false);
1426   // restore exception
1427   __ get_vm_result(Rexception_obj, Rtemp);
1428 
1429   // Inbetween activations - previous activation type unknown yet
1430   // compute continuation point - the continuation point expects
1431   // the following registers set up:
1432   //
1433   // Rexception_obj: exception
1434   // Rexception_pc: return address/pc that threw exception
1435   // SP: expression stack of caller
1436   // FP: frame pointer of caller
1437   __ mov(c_rarg0, Rthread);
1438   __ mov(c_rarg1, Rexception_pc);
1439   __ call_VM_leaf(CAST_FROM_FN_PTR(address, SharedRuntime::exception_handler_for_return_address), c_rarg0, c_rarg1);
1440   // Note that an "issuing PC" is actually the next PC after the call
1441 
1442   __ jump(R0);                             // jump to exception handler of caller
1443 }

1553 void TemplateInterpreterGenerator::histogram_bytecode_pair(Template* t) {
1554   const Register Rindex_addr = R2_tmp;
1555   Label Lcontinue;
1556   InlinedAddress Lcounters((address)BytecodePairHistogram::_counters);
1557   InlinedAddress Lindex((address)&BytecodePairHistogram::_index);
1558   const Register Rcounters_addr = R2_tmp;
1559   const Register Rindex = R4_tmp;
1560 
1561   // calculate new index for counter:
1562   // index = (_index >> log2_number_of_codes) | (bytecode << log2_number_of_codes).
1563   // (_index >> log2_number_of_codes) is previous bytecode
1564 
1565   __ ldr_literal(Rindex_addr, Lindex);
1566   __ ldr_s32(Rindex, Address(Rindex_addr));
1567   __ mov_slow(Rtemp, ((int)t->bytecode()) << BytecodePairHistogram::log2_number_of_codes);
1568   __ orr(Rindex, Rtemp, AsmOperand(Rindex, lsr, BytecodePairHistogram::log2_number_of_codes));
1569   __ str_32(Rindex, Address(Rindex_addr));
1570 
1571   // Rindex (R4) contains index of counter
1572 
1573   __ ldr_literal(Rcounters_addr, Lcounters);
1574   __ ldr_s32(Rtemp, Address::indexed_32(Rcounters_addr, Rindex));
1575   __ adds_32(Rtemp, Rtemp, 1);
1576   __ b(Lcontinue, mi);                           // avoid overflow
1577   __ str_32(Rtemp, Address::indexed_32(Rcounters_addr, Rindex));
1578 
1579   __ b(Lcontinue);
1580 
1581   __ bind_literal(Lindex);
1582   __ bind_literal(Lcounters);
1583 
1584   __ bind(Lcontinue);
