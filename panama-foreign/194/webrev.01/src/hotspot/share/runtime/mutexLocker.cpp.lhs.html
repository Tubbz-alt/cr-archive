159 #define MAX_NUM_MUTEX 128
160 static Mutex* _mutex_array[MAX_NUM_MUTEX];
161 static int _num_mutex;
162 
163 #ifdef ASSERT
164 void assert_locked_or_safepoint(const Mutex* lock) {
165   // check if this thread owns the lock (common case)
166   assert(lock != NULL, "Need non-NULL lock");
167   if (lock->owned_by_self()) return;
168   if (SafepointSynchronize::is_at_safepoint()) return;
169   if (!Universe::is_fully_initialized()) return;
170   // see if invoker of VM operation owns it
171   VM_Operation* op = VMThread::vm_operation();
172   if (op != NULL && op->calling_thread() == lock->owner()) return;
173   fatal("must own lock %s", lock->name());
174 }
175 
176 // a weaker assertion than the above
177 void assert_locked_or_safepoint_weak(const Mutex* lock) {
178   assert(lock != NULL, "Need non-NULL lock");
179   if (lock->is_locked()) return;
180   if (SafepointSynchronize::is_at_safepoint()) return;
181   if (!Universe::is_fully_initialized()) return;
182   fatal("must own lock %s", lock->name());
183 }
184 
185 // a stronger assertion than the above
186 void assert_lock_strong(const Mutex* lock) {
187   assert(lock != NULL, "Need non-NULL lock");
188   if (lock->owned_by_self()) return;
189   fatal("must own lock %s", lock->name());
190 }
191 
192 void assert_locked_or_safepoint_or_handshake(const Mutex* lock, const JavaThread* thread) {
193   if (Thread::current() == thread->active_handshaker()) return;
194   assert_locked_or_safepoint(lock);
195 }
196 #endif

198 #define def(var, type, pri, vm_block, safepoint_check_allowed ) {      \
199   var = new type(Mutex::pri, #var, vm_block, Mutex::safepoint_check_allowed); \
200   assert(_num_mutex < MAX_NUM_MUTEX, "increase MAX_NUM_MUTEX");        \
201   _mutex_array[_num_mutex++] = var;                                      \
202 }

352 GCMutexLocker::GCMutexLocker(Mutex* mutex) {
353   if (SafepointSynchronize::is_at_safepoint()) {
354     _locked = false;
355   } else {
356     _mutex = mutex;
357     _locked = true;
358     _mutex->lock();
359   }
360 }

362 // Print all mutexes/monitors that are currently owned by a thread; called
363 // by fatal error handler.
364 void print_owned_locks_on_error(outputStream* st) {
365   st->print("VM Mutex/Monitor currently owned by a thread: ");
366   bool none = true;
367   for (int i = 0; i < _num_mutex; i++) {
368      // see if it has an owner
369      if (_mutex_array[i]->owner() != NULL) {
370        if (none) {
371           // print format used by Mutex::print_on_error()
372           st->print_cr(" ([mutex/lock_event])");
373           none = false;
374        }
375        _mutex_array[i]->print_on_error(st);
376        st->cr();
377      }
378   }
379   if (none) st->print_cr("None");
380 }
