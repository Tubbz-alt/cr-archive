42 inline int compare_traceid(const traceid& lhs, const traceid& rhs) {
43   return lhs > rhs ? 1 : (lhs < rhs) ? -1 : 0;
44 }
45 
46 inline int sort_traceid(traceid* lhs, traceid* rhs) {
47   return compare_traceid(*lhs, *rhs);
48 }
