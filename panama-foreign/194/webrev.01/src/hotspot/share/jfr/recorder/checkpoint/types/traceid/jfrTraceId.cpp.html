 43  // returns updated value
 44 static traceid atomic_inc(traceid volatile* const dest) {
 45   assert(VM_Version::supports_cx8(), "invariant");
 46   traceid compare_value;
 47   traceid exchange_value;
 48   do {
 49     compare_value = *dest;
 50     exchange_value = compare_value + 1;
 51   } while (Atomic::cmpxchg(dest, compare_value, exchange_value) != compare_value);
 52   return exchange_value;
 53 }

 55 static traceid next_class_id() {
 56   static volatile traceid class_id_counter = LAST_TYPE_ID + 1;
 57   return atomic_inc(&class_id_counter) << TRACE_ID_SHIFT;
 58 }
 59 
 60 static traceid next_thread_id() {
 61   static volatile traceid thread_id_counter = 0;
 62   return atomic_inc(&thread_id_counter);
 63 }

 65 static traceid next_module_id() {
 66   static volatile traceid module_id_counter = 1;
 67   return atomic_inc(&module_id_counter) << TRACE_ID_SHIFT;
 68 }

 70 static traceid next_package_id() {
 71   static volatile traceid package_id_counter = 1;
 72   return atomic_inc(&package_id_counter) << TRACE_ID_SHIFT;
 73 }

 75 static traceid next_class_loader_data_id() {
 76   static volatile traceid cld_id_counter = 1;
 77   return atomic_inc(&cld_id_counter) << TRACE_ID_SHIFT;
 78 }

 80 static bool found_jdk_internal_event_klass = false;
 81 static bool found_jdk_jfr_event_klass = false;
 82 
 83 static void check_klass(const Klass* klass) {
 84   assert(klass != NULL, "invariant");
 85   if (found_jdk_internal_event_klass && found_jdk_jfr_event_klass) {
 86     return;
 87   }
 88   static const Symbol* jdk_internal_event_sym = NULL;
 89   if (jdk_internal_event_sym == NULL) {
 90     // setup when loading the first TypeArrayKlass (Universe::genesis) hence single threaded invariant
 91     jdk_internal_event_sym = SymbolTable::new_permanent_symbol("jdk/internal/event/Event");
 92   }
 93   assert(jdk_internal_event_sym != NULL, "invariant");
 94 
 95   static const Symbol* jdk_jfr_event_sym = NULL;
 96   if (jdk_jfr_event_sym == NULL) {
 97     // setup when loading the first TypeArrayKlass (Universe::genesis) hence single threaded invariant
 98     jdk_jfr_event_sym = SymbolTable::new_permanent_symbol("jdk/jfr/Event");
 99   }
100   assert(jdk_jfr_event_sym != NULL, "invariant");
101   const Symbol* const klass_name = klass->name();
102 
103   if (!found_jdk_internal_event_klass) {
104     if (jdk_internal_event_sym == klass_name && klass->class_loader() == NULL) {
105       found_jdk_internal_event_klass = true;
106       JfrTraceId::tag_as_jdk_jfr_event(klass);
107       return;
108     }
109   }
110 
111   if (!found_jdk_jfr_event_klass) {
112     if (jdk_jfr_event_sym == klass_name && klass->class_loader() == NULL) {
113       found_jdk_jfr_event_klass = true;
114       JfrTraceId::tag_as_jdk_jfr_event(klass);
115       return;
116     }
117   }
118 }

120 void JfrTraceId::assign(const Klass* klass) {
121   assert(klass != NULL, "invariant");
122   klass->set_trace_id(next_class_id());
123   check_klass(klass);
124   const Klass* const super = klass->super();
125   if (super == NULL) {
126     return;
127   }
128   if (IS_EVENT_KLASS(super)) {
129     tag_as_jdk_jfr_event_sub(klass);
130   }
131 }

143 void JfrTraceId::assign(const ClassLoaderData* cld) {
144   assert(cld != NULL, "invariant");
145   if (cld->has_class_mirror_holder()) {
146     cld->set_trace_id(0);
147     return;
148   }
149   cld->set_trace_id(next_class_loader_data_id());
150 }

156 // used by CDS / APPCDS as part of "remove_unshareable_info"
157 void JfrTraceId::remove(const Klass* k) {
158   assert(k != NULL, "invariant");
159   // Mask off and store the event flags.
160   // This mechanism will retain the event specific flags
161   // in the archive, allowing for event flag restoration
162   // when renewing the traceid on klass revival.
163   k->set_trace_id(EVENT_KLASS_MASK(k));
164 }
165 
166 // used by CDS / APPCDS as part of "restore_unshareable_info"
167 void JfrTraceId::restore(const Klass* k) {
168   assert(k != NULL, "invariant");
169   if (IS_JDK_JFR_EVENT_KLASS(k)) {
170     found_jdk_jfr_event_klass = true;
171   }
172   const traceid event_flags = k->trace_id();
173   // get a fresh traceid and restore the original event flags
174   k->set_trace_id(next_class_id() | event_flags);
175 }
