25 #include "precompiled.hpp"
26 #include "jfr/recorder/jfrEventSetting.inline.hpp"
27 
28 JfrNativeSettings JfrEventSetting::_jvm_event_settings;
29 
30 bool JfrEventSetting::set_threshold(jlong id, jlong threshold_ticks) {
31   JfrEventId event_id = (JfrEventId)id;
32   assert(bounds_check_event(event_id), "invariant");
33   setting(event_id).threshold_ticks = threshold_ticks;
34   return true;
35 }
36 
37 bool JfrEventSetting::set_cutoff(jlong id, jlong cutoff_ticks) {
38   JfrEventId event_id = (JfrEventId)id;
39   assert(bounds_check_event(event_id), "invariant");
40   setting(event_id).cutoff_ticks = cutoff_ticks;
41   return true;
42 }
43 
44 void JfrEventSetting::set_stacktrace(jlong id, bool enabled) {
45   JfrEventId event_id = (JfrEventId)id;
46   assert(bounds_check_event(event_id), "invariant");
47   setting(event_id).stacktrace = enabled;
48 }
49 
50 void JfrEventSetting::set_enabled(jlong id, bool enabled) {
51   JfrEventId event_id = (JfrEventId)id;
52   assert(bounds_check_event(event_id), "invariant");
53   setting(event_id).enabled = enabled;
54 }
55 





56 #ifdef ASSERT
57 bool JfrEventSetting::bounds_check_event(jlong id) {
58   if ((unsigned)id < NUM_RESERVED_EVENTS || (unsigned)id >= MaxJfrEventId) {



59     return false;
60   }
61   return true;
62 }
63 #endif // ASSERT
