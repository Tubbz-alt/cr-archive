24 #include "precompiled.hpp"

29 ZLargePages::State ZLargePages::_state;
30 
31 void ZLargePages::initialize() {
32   initialize_platform();
33 
34   log_info_p(gc, init)("Memory: " JULONG_FORMAT "M", os::physical_memory() / M);
35   log_info_p(gc, init)("Large Page Support: %s", to_string());
36 }
37 
38 const char* ZLargePages::to_string() {
39   switch (_state) {
40   case Explicit:
41     return "Enabled (Explicit)";
42 
43   case Transparent:
44     return "Enabled (Transparent)";
45 
46   default:
47     return "Disabled";
48   }
49 }
