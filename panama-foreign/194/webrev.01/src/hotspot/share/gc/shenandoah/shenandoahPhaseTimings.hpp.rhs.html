181 class ShenandoahPhaseTimings : public CHeapObj<mtGC> {
182   friend class ShenandoahGCPhase;
183   friend class ShenandoahWorkerTimingsTracker;
184 public:
185 #define SHENANDOAH_PHASE_DECLARE_ENUM(type, title)   type,
186 
187   enum Phase {
188     SHENANDOAH_PHASE_DO(SHENANDOAH_PHASE_DECLARE_ENUM)
189     _num_phases,
190     _invalid_phase = _num_phases
191   };
192 
193   enum ParPhase {
194     SHENANDOAH_PAR_PHASE_DO(,, SHENANDOAH_PHASE_DECLARE_ENUM)
195     _num_par_phases
196   };
197 
198 #undef SHENANDOAH_PHASE_DECLARE_ENUM
199 
200 private:
201   uint                _max_workers;
202   double              _cycle_data[_num_phases];
203   HdrSeq              _global_data[_num_phases];
204   static const char*  _phase_names[_num_phases];
205 
206   ShenandoahWorkerData* _worker_data[_num_phases];
207   ShenandoahCollectorPolicy* _policy;
208 
209   static bool is_worker_phase(Phase phase);
210   static bool is_root_work_phase(Phase phase);
211 
212   ShenandoahWorkerData* worker_data(Phase phase, ParPhase par_phase);
213   Phase worker_par_phase(Phase phase, ParPhase par_phase);
214 
215   void set_cycle_data(Phase phase, double time);
216   static double uninitialized() { return -1; }
217 
218 public:
219   ShenandoahPhaseTimings(uint max_workers);
220 
221   void record_phase_time(Phase phase, double time);
222 
223   void record_workers_start(Phase phase);
224   void record_workers_end(Phase phase);
225 
226   void flush_par_workers_to_cycle();
227   void flush_cycle_to_global();
228 
229   static const char* phase_name(Phase phase) {
230     assert(phase >= 0 && phase < _num_phases, "Out of bound");
231     return _phase_names[phase];
232   }
233 
234   void print_cycle_on(outputStream* out) const;
235   void print_global_on(outputStream* out) const;
236 };

239 private:
240   ShenandoahPhaseTimings*          const _timings;
241   ShenandoahPhaseTimings::Phase    const _phase;
242   ShenandoahPhaseTimings::ParPhase const _par_phase;
243   uint const _worker_id;
244 
245   double _start_time;
246   EventGCPhaseParallel _event;
247 public:
248   ShenandoahWorkerTimingsTracker(ShenandoahPhaseTimings::Phase phase, ShenandoahPhaseTimings::ParPhase par_phase, uint worker_id);
249   ~ShenandoahWorkerTimingsTracker();
250 };
