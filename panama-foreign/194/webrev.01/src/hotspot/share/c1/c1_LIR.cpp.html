  47 LIR_Opr LIR_OprFact::value_type(ValueType* type) {
  48   ValueTag tag = type->tag();
  49   switch (tag) {
  50   case metaDataTag : {
  51     ClassConstant* c = type->as_ClassConstant();
  52     if (c != NULL && !c->value()->is_loaded()) {
  53       return LIR_OprFact::metadataConst(NULL);
  54     } else if (c != NULL) {
  55       return LIR_OprFact::metadataConst(c->value()->constant_encoding());
  56     } else {
  57       MethodConstant* m = type->as_MethodConstant();
  58       assert (m != NULL, "not a class or a method?");
  59       return LIR_OprFact::metadataConst(m->value()->constant_encoding());
  60     }
  61   }
  62   case objectTag : {
  63       return LIR_OprFact::oopConst(type->as_ObjectType()->encoding());
  64     }
  65   case addressTag: return LIR_OprFact::addressConst(type->as_AddressConstant()->value());
  66   case intTag    : return LIR_OprFact::intConst(type->as_IntConstant()->value());
  67   case floatTag  : return LIR_OprFact::floatConst(type->as_FloatConstant()->value());
  68   case longTag   : return LIR_OprFact::longConst(type->as_LongConstant()->value());
  69   case doubleTag : return LIR_OprFact::doubleConst(type->as_DoubleConstant()->value());
  70   default: ShouldNotReachHere(); return LIR_OprFact::intConst(-1);
  71   }
  72 }

  78 LIR_Address::Scale LIR_Address::scale(BasicType type) {
  79   int elem_size = type2aelembytes(type);
  80   switch (elem_size) {
  81   case 1: return LIR_Address::times_1;
  82   case 2: return LIR_Address::times_2;
  83   case 4: return LIR_Address::times_4;
  84   case 8: return LIR_Address::times_8;
  85   }
  86   ShouldNotReachHere();
  87   return LIR_Address::times_1;
  88 }

  92 char LIR_OprDesc::type_char(BasicType t) {
  93   switch (t) {
  94     case T_ARRAY:
  95       t = T_OBJECT;
  96     case T_BOOLEAN:
  97     case T_CHAR:
  98     case T_FLOAT:
  99     case T_DOUBLE:
 100     case T_BYTE:
 101     case T_SHORT:
 102     case T_INT:
 103     case T_LONG:
 104     case T_OBJECT:
 105     case T_ADDRESS:
 106     case T_VOID:
 107       return ::type2char(t);
 108     case T_METADATA:
 109       return 'M';
 110     case T_ILLEGAL:
 111       return '?';
 112 
 113     default:
 114       ShouldNotReachHere();
 115       return '?';
 116   }

 119 #ifndef PRODUCT
 120 void LIR_OprDesc::validate_type() const {
 121 
 122 #ifdef ASSERT
 123   if (!is_pointer() && !is_illegal()) {
 124     OprKind kindfield = kind_field(); // Factored out because of compiler bug, see 8002160
 125     switch (as_BasicType(type_field())) {
 126     case T_LONG:
 127       assert((kindfield == cpu_register || kindfield == stack_value) &&
 128              size_field() == double_size, "must match");
 129       break;
 130     case T_FLOAT:
 131       // FP return values can be also in CPU registers on ARM and PPC32 (softfp ABI)
 132       assert((kindfield == fpu_register || kindfield == stack_value
 133              ARM_ONLY(|| kindfield == cpu_register)
 134              PPC32_ONLY(|| kindfield == cpu_register) ) &&
 135              size_field() == single_size, "must match");
 136       break;
 137     case T_DOUBLE:
 138       // FP return values can be also in CPU registers on ARM and PPC32 (softfp ABI)
 139       assert((kindfield == fpu_register || kindfield == stack_value
 140              ARM_ONLY(|| kindfield == cpu_register)
 141              PPC32_ONLY(|| kindfield == cpu_register) ) &&
 142              size_field() == double_size, "must match");
 143       break;
 144     case T_BOOLEAN:
 145     case T_CHAR:
 146     case T_BYTE:
 147     case T_SHORT:
 148     case T_INT:
 149     case T_ADDRESS:
 150     case T_OBJECT:
 151     case T_METADATA:
 152     case T_ARRAY:
 153       assert((kindfield == cpu_register || kindfield == stack_value) &&
 154              size_field() == single_size, "must match");
 155       break;
 156 
 157     case T_ILLEGAL:
 158       // XXX TKR also means unknown right now
 159       // assert(is_illegal(), "must match");
 160       break;
 161 
 162     default:
 163       ShouldNotReachHere();
 164     }
 165   }
 166 #endif
 167 
 168 }

 184 void LIR_Op2::verify() const {
 185 #ifdef ASSERT
 186   switch (code()) {
 187     case lir_cmove:
 188     case lir_xchg:
 189       break;
 190 
 191     default:
 192       assert(!result_opr()->is_register() || !result_opr()->is_oop_register(),
 193              "can't produce oops from arith");
 194   }
 195 
 196   if (TwoOperandLIRForm) {
 197 
 198 #ifdef ASSERT
 199     bool threeOperandForm = false;
 200 #ifdef S390
 201     // There are 3 operand shifts on S390 (see LIR_Assembler::shift_op()).
 202     threeOperandForm =
 203       code() == lir_shl ||
 204       ((code() == lir_shr || code() == lir_ushr) && (result_opr()->is_double_cpu() || in_opr1()->type() == T_OBJECT));
 205 #endif
 206 #endif
 207 
 208     switch (code()) {
 209     case lir_add:
 210     case lir_sub:
 211     case lir_mul:
 212     case lir_mul_strictfp:
 213     case lir_div:
 214     case lir_div_strictfp:
 215     case lir_rem:
 216     case lir_logic_and:
 217     case lir_logic_or:
 218     case lir_logic_xor:
 219     case lir_shl:
 220     case lir_shr:
 221       assert(in_opr1() == result_opr() || threeOperandForm, "opr1 and result must match");
 222       assert(in_opr1()->is_valid() && in_opr2()->is_valid(), "must be valid");
 223       break;
 224 
 225     // special handling for lir_ushr because of write barriers
 226     case lir_ushr:
 227       assert(in_opr1() == result_opr() || in_opr2()->is_constant() || threeOperandForm, "opr1 and result must match or shift count is constant");
 228       assert(in_opr1()->is_valid() && in_opr2()->is_valid(), "must be valid");
 229       break;
 230 
 231     default:
 232       break;
 233     }
 234   }
 235 #endif
 236 }

 280 void LIR_OpBranch::negate_cond() {
 281   switch (_cond) {
 282     case lir_cond_equal:        _cond = lir_cond_notEqual;     break;
 283     case lir_cond_notEqual:     _cond = lir_cond_equal;        break;
 284     case lir_cond_less:         _cond = lir_cond_greaterEqual; break;
 285     case lir_cond_lessEqual:    _cond = lir_cond_greater;      break;
 286     case lir_cond_greaterEqual: _cond = lir_cond_less;         break;
 287     case lir_cond_greater:      _cond = lir_cond_lessEqual;    break;
 288     default: ShouldNotReachHere();
 289   }

 393 // complete rework of LIR instruction visitor.
 394 // The virtual call for each instruction type is replaced by a big
 395 // switch that adds the operands for each instruction
 396 
 397 void LIR_OpVisitState::visit(LIR_Op* op) {
 398   // copy information from the LIR_Op
 399   reset();
 400   set_op(op);
 401 
 402   switch (op->code()) {
 403 
 404 // LIR_Op0
 405     case lir_backwardbranch_target:    // result and info always invalid
 406     case lir_fpop_raw:                 // result and info always invalid
 407     case lir_breakpoint:               // result and info always invalid
 408     case lir_membar:                   // result and info always invalid
 409     case lir_membar_acquire:           // result and info always invalid
 410     case lir_membar_release:           // result and info always invalid
 411     case lir_membar_loadload:          // result and info always invalid
 412     case lir_membar_storestore:        // result and info always invalid
 413     case lir_membar_loadstore:         // result and info always invalid
 414     case lir_membar_storeload:         // result and info always invalid
 415     case lir_on_spin_wait:
 416     {
 417       assert(op->as_Op0() != NULL, "must be");
 418       assert(op->_info == NULL, "info not used by this instruction");
 419       assert(op->_result->is_illegal(), "not used");
 420       break;
 421     }
 422 
 423     case lir_nop:                      // may have info, result always invalid
 424     case lir_std_entry:                // may have result, info always invalid
 425     case lir_osr_entry:                // may have result, info always invalid
 426     case lir_get_thread:               // may have result, info always invalid
 427     {
 428       assert(op->as_Op0() != NULL, "must be");
 429       if (op->_info != NULL)           do_info(op->_info);
 430       if (op->_result->is_valid())     do_output(op->_result);
 431       break;
 432     }
 433 
 434 
 435 // LIR_OpLabel
 436     case lir_label:                    // result and info always invalid
 437     {
 438       assert(op->as_OpLabel() != NULL, "must be");
 439       assert(op->_info == NULL, "info not used by this instruction");
 440       assert(op->_result->is_illegal(), "not used");
 441       break;
 442     }
 443 
 444 
 445 // LIR_Op1
 446     case lir_fxch:           // input always valid, result and info always invalid
 447     case lir_fld:            // input always valid, result and info always invalid
 448     case lir_push:           // input always valid, result and info always invalid
 449     case lir_pop:            // input always valid, result and info always invalid
 450     case lir_return:         // input always valid, result and info always invalid
 451     case lir_leal:           // input and result always valid, info always invalid
 452     case lir_monaddr:        // input and result always valid, info always invalid
 453     case lir_null_check:     // input and info always valid, result always invalid
 454     case lir_move:           // input and result always valid, may have info
 455     {
 456       assert(op->as_Op1() != NULL, "must be");
 457       LIR_Op1* op1 = (LIR_Op1*)op;
 458 
 459       if (op1->_info)                  do_info(op1->_info);
 460       if (op1->_opr->is_valid())       do_input(op1->_opr);
 461       if (op1->_result->is_valid())    do_output(op1->_result);
 462 
 463       break;
 464     }
 465 
 466     case lir_safepoint:
 467     {
 468       assert(op->as_Op1() != NULL, "must be");
 469       LIR_Op1* op1 = (LIR_Op1*)op;
 470 
 471       assert(op1->_info != NULL, "");  do_info(op1->_info);
 472       if (op1->_opr->is_valid())       do_temp(op1->_opr); // safepoints on SPARC need temporary register
 473       assert(op1->_result->is_illegal(), "safepoint does not produce value");
 474 
 475       break;
 476     }
 477 
 478 // LIR_OpConvert;
 479     case lir_convert:        // input and result always valid, info always invalid
 480     {
 481       assert(op->as_OpConvert() != NULL, "must be");
 482       LIR_OpConvert* opConvert = (LIR_OpConvert*)op;
 483 
 484       assert(opConvert->_info == NULL, "must be");
 485       if (opConvert->_opr->is_valid())       do_input(opConvert->_opr);
 486       if (opConvert->_result->is_valid())    do_output(opConvert->_result);
 487 #ifdef PPC32
 488       if (opConvert->_tmp1->is_valid())      do_temp(opConvert->_tmp1);
 489       if (opConvert->_tmp2->is_valid())      do_temp(opConvert->_tmp2);
 490 #endif
 491       do_stub(opConvert->_stub);
 492 
 493       break;
 494     }
 495 
 496 // LIR_OpBranch;
 497     case lir_branch:                   // may have info, input and result register always invalid
 498     case lir_cond_float_branch:        // may have info, input and result register always invalid
 499     {
 500       assert(op->as_OpBranch() != NULL, "must be");
 501       LIR_OpBranch* opBranch = (LIR_OpBranch*)op;
 502 
 503       if (opBranch->_info != NULL)     do_info(opBranch->_info);
 504       assert(opBranch->_result->is_illegal(), "not used");
 505       if (opBranch->_stub != NULL)     opBranch->stub()->visit(this);
 506 
 507       break;
 508     }
 509 
 510 
 511 // LIR_OpAllocObj
 512     case lir_alloc_object:
 513     {
 514       assert(op->as_OpAllocObj() != NULL, "must be");
 515       LIR_OpAllocObj* opAllocObj = (LIR_OpAllocObj*)op;
 516 
 517       if (opAllocObj->_info)                     do_info(opAllocObj->_info);
 518       if (opAllocObj->_opr->is_valid()) {        do_input(opAllocObj->_opr);
 519                                                  do_temp(opAllocObj->_opr);
 520                                         }
 521       if (opAllocObj->_tmp1->is_valid())         do_temp(opAllocObj->_tmp1);
 522       if (opAllocObj->_tmp2->is_valid())         do_temp(opAllocObj->_tmp2);
 523       if (opAllocObj->_tmp3->is_valid())         do_temp(opAllocObj->_tmp3);
 524       if (opAllocObj->_tmp4->is_valid())         do_temp(opAllocObj->_tmp4);
 525       if (opAllocObj->_result->is_valid())       do_output(opAllocObj->_result);
 526                                                  do_stub(opAllocObj->_stub);
 527       break;
 528     }
 529 
 530 
 531 // LIR_OpRoundFP;
 532     case lir_roundfp: {
 533       assert(op->as_OpRoundFP() != NULL, "must be");
 534       LIR_OpRoundFP* opRoundFP = (LIR_OpRoundFP*)op;
 535 
 536       assert(op->_info == NULL, "info not used by this instruction");
 537       assert(opRoundFP->_tmp->is_illegal(), "not used");
 538       do_input(opRoundFP->_opr);
 539       do_output(opRoundFP->_result);
 540 
 541       break;
 542     }
 543 
 544 
 545 // LIR_Op2
 546     case lir_cmp:
 547     case lir_cmp_l2i:
 548     case lir_ucmp_fd2i:
 549     case lir_cmp_fd2i:
 550     case lir_add:
 551     case lir_sub:
 552     case lir_mul:
 553     case lir_div:
 554     case lir_rem:
 555     case lir_sqrt:
 556     case lir_abs:
 557     case lir_neg:
 558     case lir_logic_and:
 559     case lir_logic_or:
 560     case lir_logic_xor:
 561     case lir_shl:
 562     case lir_shr:
 563     case lir_ushr:
 564     case lir_xadd:
 565     case lir_xchg:
 566     case lir_assert:
 567     {
 568       assert(op->as_Op2() != NULL, "must be");
 569       LIR_Op2* op2 = (LIR_Op2*)op;
 570       assert(op2->_tmp2->is_illegal() && op2->_tmp3->is_illegal() &&
 571              op2->_tmp4->is_illegal() && op2->_tmp5->is_illegal(), "not used");
 572 
 573       if (op2->_info)                     do_info(op2->_info);
 574       if (op2->_opr1->is_valid())         do_input(op2->_opr1);
 575       if (op2->_opr2->is_valid())         do_input(op2->_opr2);
 576       if (op2->_tmp1->is_valid())         do_temp(op2->_tmp1);
 577       if (op2->_result->is_valid())       do_output(op2->_result);
 578       if (op->code() == lir_xchg || op->code() == lir_xadd) {
 579         // on ARM and PPC, return value is loaded first so could
 580         // destroy inputs. On other platforms that implement those
 581         // (x86, sparc), the extra constrainsts are harmless.
 582         if (op2->_opr1->is_valid())       do_temp(op2->_opr1);
 583         if (op2->_opr2->is_valid())       do_temp(op2->_opr2);
 584       }
 585 
 586       break;
 587     }
 588 
 589     // special handling for cmove: right input operand must not be equal
 590     // to the result operand, otherwise the backend fails
 591     case lir_cmove:
 592     {
 593       assert(op->as_Op2() != NULL, "must be");
 594       LIR_Op2* op2 = (LIR_Op2*)op;
 595 
 596       assert(op2->_info == NULL && op2->_tmp1->is_illegal() && op2->_tmp2->is_illegal() &&
 597              op2->_tmp3->is_illegal() && op2->_tmp4->is_illegal() && op2->_tmp5->is_illegal(), "not used");
 598       assert(op2->_opr1->is_valid() && op2->_opr2->is_valid() && op2->_result->is_valid(), "used");
 599 
 600       do_input(op2->_opr1);
 601       do_input(op2->_opr2);
 602       do_temp(op2->_opr2);
 603       do_output(op2->_result);
 604 
 605       break;
 606     }
 607 
 608     // vspecial handling for strict operations: register input operands
 609     // as temp to guarantee that they do not overlap with other
 610     // registers
 611     case lir_mul_strictfp:
 612     case lir_div_strictfp:
 613     {
 614       assert(op->as_Op2() != NULL, "must be");
 615       LIR_Op2* op2 = (LIR_Op2*)op;
 616 
 617       assert(op2->_info == NULL, "not used");
 618       assert(op2->_opr1->is_valid(), "used");
 619       assert(op2->_opr2->is_valid(), "used");
 620       assert(op2->_result->is_valid(), "used");
 621       assert(op2->_tmp2->is_illegal() && op2->_tmp3->is_illegal() &&
 622              op2->_tmp4->is_illegal() && op2->_tmp5->is_illegal(), "not used");
 623 
 624       do_input(op2->_opr1); do_temp(op2->_opr1);
 625       do_input(op2->_opr2); do_temp(op2->_opr2);
 626       if (op2->_tmp1->is_valid()) do_temp(op2->_tmp1);
 627       do_output(op2->_result);
 628 
 629       break;
 630     }
 631 
 632     case lir_throw: {
 633       assert(op->as_Op2() != NULL, "must be");
 634       LIR_Op2* op2 = (LIR_Op2*)op;
 635 
 636       if (op2->_info)                     do_info(op2->_info);
 637       if (op2->_opr1->is_valid())         do_temp(op2->_opr1);
 638       if (op2->_opr2->is_valid())         do_input(op2->_opr2); // exception object is input parameter
 639       assert(op2->_result->is_illegal(), "no result");
 640       assert(op2->_tmp2->is_illegal() && op2->_tmp3->is_illegal() &&
 641              op2->_tmp4->is_illegal() && op2->_tmp5->is_illegal(), "not used");
 642 
 643       break;
 644     }
 645 
 646     case lir_unwind: {
 647       assert(op->as_Op1() != NULL, "must be");
 648       LIR_Op1* op1 = (LIR_Op1*)op;
 649 
 650       assert(op1->_info == NULL, "no info");
 651       assert(op1->_opr->is_valid(), "exception oop");         do_input(op1->_opr);
 652       assert(op1->_result->is_illegal(), "no result");
 653 
 654       break;
 655     }
 656 
 657 // LIR_Op3
 658     case lir_idiv:
 659     case lir_irem: {
 660       assert(op->as_Op3() != NULL, "must be");
 661       LIR_Op3* op3= (LIR_Op3*)op;
 662 
 663       if (op3->_info)                     do_info(op3->_info);
 664       if (op3->_opr1->is_valid())         do_input(op3->_opr1);
 665 
 666       // second operand is input and temp, so ensure that second operand
 667       // and third operand get not the same register
 668       if (op3->_opr2->is_valid())         do_input(op3->_opr2);
 669       if (op3->_opr2->is_valid())         do_temp(op3->_opr2);
 670       if (op3->_opr3->is_valid())         do_temp(op3->_opr3);
 671 
 672       if (op3->_result->is_valid())       do_output(op3->_result);
 673 
 674       break;
 675     }
 676 
 677     case lir_fmad:
 678     case lir_fmaf: {
 679       assert(op->as_Op3() != NULL, "must be");
 680       LIR_Op3* op3= (LIR_Op3*)op;
 681       assert(op3->_info == NULL, "no info");
 682       do_input(op3->_opr1);
 683       do_input(op3->_opr2);
 684       do_input(op3->_opr3);
 685       do_output(op3->_result);
 686       break;
 687     }
 688 
 689 // LIR_OpJavaCall
 690     case lir_static_call:
 691     case lir_optvirtual_call:
 692     case lir_icvirtual_call:
 693     case lir_virtual_call:
 694     case lir_dynamic_call: {
 695       LIR_OpJavaCall* opJavaCall = op->as_OpJavaCall();
 696       assert(opJavaCall != NULL, "must be");
 697 
 698       if (opJavaCall->_receiver->is_valid())     do_input(opJavaCall->_receiver);
 699 
 700       // only visit register parameters
 701       int n = opJavaCall->_arguments->length();
 702       for (int i = opJavaCall->_receiver->is_valid() ? 1 : 0; i < n; i++) {
 703         if (!opJavaCall->_arguments->at(i)->is_pointer()) {
 704           do_input(*opJavaCall->_arguments->adr_at(i));
 705         }
 706       }
 707 
 708       if (opJavaCall->_info)                     do_info(opJavaCall->_info);
 709       if (FrameMap::method_handle_invoke_SP_save_opr() != LIR_OprFact::illegalOpr &&
 710           opJavaCall->is_method_handle_invoke()) {
 711         opJavaCall->_method_handle_invoke_SP_save_opr = FrameMap::method_handle_invoke_SP_save_opr();
 712         do_temp(opJavaCall->_method_handle_invoke_SP_save_opr);
 713       }
 714       do_call();
 715       if (opJavaCall->_result->is_valid())       do_output(opJavaCall->_result);
 716 
 717       break;
 718     }
 719 
 720 
 721 // LIR_OpRTCall
 722     case lir_rtcall: {
 723       assert(op->as_OpRTCall() != NULL, "must be");
 724       LIR_OpRTCall* opRTCall = (LIR_OpRTCall*)op;
 725 
 726       // only visit register parameters
 727       int n = opRTCall->_arguments->length();
 728       for (int i = 0; i < n; i++) {
 729         if (!opRTCall->_arguments->at(i)->is_pointer()) {
 730           do_input(*opRTCall->_arguments->adr_at(i));
 731         }
 732       }
 733       if (opRTCall->_info)                     do_info(opRTCall->_info);
 734       if (opRTCall->_tmp->is_valid())          do_temp(opRTCall->_tmp);
 735       do_call();
 736       if (opRTCall->_result->is_valid())       do_output(opRTCall->_result);
 737 
 738       break;
 739     }
 740 
 741 
 742 // LIR_OpArrayCopy
 743     case lir_arraycopy: {
 744       assert(op->as_OpArrayCopy() != NULL, "must be");
 745       LIR_OpArrayCopy* opArrayCopy = (LIR_OpArrayCopy*)op;
 746 
 747       assert(opArrayCopy->_result->is_illegal(), "unused");
 748       assert(opArrayCopy->_src->is_valid(), "used");          do_input(opArrayCopy->_src);     do_temp(opArrayCopy->_src);
 749       assert(opArrayCopy->_src_pos->is_valid(), "used");      do_input(opArrayCopy->_src_pos); do_temp(opArrayCopy->_src_pos);
 750       assert(opArrayCopy->_dst->is_valid(), "used");          do_input(opArrayCopy->_dst);     do_temp(opArrayCopy->_dst);
 751       assert(opArrayCopy->_dst_pos->is_valid(), "used");      do_input(opArrayCopy->_dst_pos); do_temp(opArrayCopy->_dst_pos);
 752       assert(opArrayCopy->_length->is_valid(), "used");       do_input(opArrayCopy->_length);  do_temp(opArrayCopy->_length);
 753       assert(opArrayCopy->_tmp->is_valid(), "used");          do_temp(opArrayCopy->_tmp);
 754       if (opArrayCopy->_info)                     do_info(opArrayCopy->_info);
 755 
 756       // the implementation of arraycopy always has a call into the runtime
 757       do_call();
 758 
 759       break;
 760     }
 761

 763 // LIR_OpUpdateCRC32
 764     case lir_updatecrc32: {
 765       assert(op->as_OpUpdateCRC32() != NULL, "must be");
 766       LIR_OpUpdateCRC32* opUp = (LIR_OpUpdateCRC32*)op;
 767 
 768       assert(opUp->_crc->is_valid(), "used");          do_input(opUp->_crc);     do_temp(opUp->_crc);
 769       assert(opUp->_val->is_valid(), "used");          do_input(opUp->_val);     do_temp(opUp->_val);
 770       assert(opUp->_result->is_valid(), "used");       do_output(opUp->_result);
 771       assert(opUp->_info == NULL, "no info for LIR_OpUpdateCRC32");
 772 
 773       break;
 774     }
 775 
 776 
 777 // LIR_OpLock
 778     case lir_lock:
 779     case lir_unlock: {
 780       assert(op->as_OpLock() != NULL, "must be");
 781       LIR_OpLock* opLock = (LIR_OpLock*)op;
 782 
 783       if (opLock->_info)                          do_info(opLock->_info);
 784 
 785       // TODO: check if these operands really have to be temp
 786       // (or if input is sufficient). This may have influence on the oop map!
 787       assert(opLock->_lock->is_valid(), "used");  do_temp(opLock->_lock);
 788       assert(opLock->_hdr->is_valid(),  "used");  do_temp(opLock->_hdr);
 789       assert(opLock->_obj->is_valid(),  "used");  do_temp(opLock->_obj);
 790 
 791       if (opLock->_scratch->is_valid())           do_temp(opLock->_scratch);
 792       assert(opLock->_result->is_illegal(), "unused");
 793 
 794       do_stub(opLock->_stub);
 795 
 796       break;
 797     }
 798 
 799 
 800 // LIR_OpDelay
 801     case lir_delay_slot: {
 802       assert(op->as_OpDelay() != NULL, "must be");
 803       LIR_OpDelay* opDelay = (LIR_OpDelay*)op;
 804 
 805       visit(opDelay->delay_op());
 806       break;
 807     }

 809 // LIR_OpTypeCheck
 810     case lir_instanceof:
 811     case lir_checkcast:
 812     case lir_store_check: {
 813       assert(op->as_OpTypeCheck() != NULL, "must be");
 814       LIR_OpTypeCheck* opTypeCheck = (LIR_OpTypeCheck*)op;
 815 
 816       if (opTypeCheck->_info_for_exception)       do_info(opTypeCheck->_info_for_exception);
 817       if (opTypeCheck->_info_for_patch)           do_info(opTypeCheck->_info_for_patch);
 818       if (opTypeCheck->_object->is_valid())       do_input(opTypeCheck->_object);
 819       if (op->code() == lir_store_check && opTypeCheck->_object->is_valid()) {
 820         do_temp(opTypeCheck->_object);
 821       }
 822       if (opTypeCheck->_array->is_valid())        do_input(opTypeCheck->_array);
 823       if (opTypeCheck->_tmp1->is_valid())         do_temp(opTypeCheck->_tmp1);
 824       if (opTypeCheck->_tmp2->is_valid())         do_temp(opTypeCheck->_tmp2);
 825       if (opTypeCheck->_tmp3->is_valid())         do_temp(opTypeCheck->_tmp3);
 826       if (opTypeCheck->_result->is_valid())       do_output(opTypeCheck->_result);
 827                                                   do_stub(opTypeCheck->_stub);
 828       break;
 829     }

 831 // LIR_OpCompareAndSwap
 832     case lir_cas_long:
 833     case lir_cas_obj:
 834     case lir_cas_int: {
 835       assert(op->as_OpCompareAndSwap() != NULL, "must be");
 836       LIR_OpCompareAndSwap* opCompareAndSwap = (LIR_OpCompareAndSwap*)op;
 837 
 838       assert(opCompareAndSwap->_addr->is_valid(),      "used");
 839       assert(opCompareAndSwap->_cmp_value->is_valid(), "used");
 840       assert(opCompareAndSwap->_new_value->is_valid(), "used");
 841       if (opCompareAndSwap->_info)                    do_info(opCompareAndSwap->_info);
 842                                                       do_input(opCompareAndSwap->_addr);
 843                                                       do_temp(opCompareAndSwap->_addr);
 844                                                       do_input(opCompareAndSwap->_cmp_value);
 845                                                       do_temp(opCompareAndSwap->_cmp_value);
 846                                                       do_input(opCompareAndSwap->_new_value);
 847                                                       do_temp(opCompareAndSwap->_new_value);
 848       if (opCompareAndSwap->_tmp1->is_valid())        do_temp(opCompareAndSwap->_tmp1);
 849       if (opCompareAndSwap->_tmp2->is_valid())        do_temp(opCompareAndSwap->_tmp2);
 850       if (opCompareAndSwap->_result->is_valid())      do_output(opCompareAndSwap->_result);
 851 
 852       break;
 853     }
 854 
 855 
 856 // LIR_OpAllocArray;
 857     case lir_alloc_array: {
 858       assert(op->as_OpAllocArray() != NULL, "must be");
 859       LIR_OpAllocArray* opAllocArray = (LIR_OpAllocArray*)op;
 860 
 861       if (opAllocArray->_info)                        do_info(opAllocArray->_info);
 862       if (opAllocArray->_klass->is_valid())           do_input(opAllocArray->_klass); do_temp(opAllocArray->_klass);
 863       if (opAllocArray->_len->is_valid())             do_input(opAllocArray->_len);   do_temp(opAllocArray->_len);
 864       if (opAllocArray->_tmp1->is_valid())            do_temp(opAllocArray->_tmp1);
 865       if (opAllocArray->_tmp2->is_valid())            do_temp(opAllocArray->_tmp2);
 866       if (opAllocArray->_tmp3->is_valid())            do_temp(opAllocArray->_tmp3);
 867       if (opAllocArray->_tmp4->is_valid())            do_temp(opAllocArray->_tmp4);
 868       if (opAllocArray->_result->is_valid())          do_output(opAllocArray->_result);
 869                                                       do_stub(opAllocArray->_stub);
 870       break;
 871     }
 872 
 873 // LIR_OpProfileCall:
 874     case lir_profile_call: {
 875       assert(op->as_OpProfileCall() != NULL, "must be");
 876       LIR_OpProfileCall* opProfileCall = (LIR_OpProfileCall*)op;
 877 
 878       if (opProfileCall->_recv->is_valid())              do_temp(opProfileCall->_recv);
 879       assert(opProfileCall->_mdo->is_valid(), "used");   do_temp(opProfileCall->_mdo);
 880       assert(opProfileCall->_tmp1->is_valid(), "used");  do_temp(opProfileCall->_tmp1);
 881       break;
 882     }
 883 
 884 // LIR_OpProfileType:
 885     case lir_profile_type: {
 886       assert(op->as_OpProfileType() != NULL, "must be");
 887       LIR_OpProfileType* opProfileType = (LIR_OpProfileType*)op;
 888 
 889       do_input(opProfileType->_mdp); do_temp(opProfileType->_mdp);
 890       do_input(opProfileType->_obj);
 891       do_temp(opProfileType->_tmp);
 892       break;
 893     }
 894   default:
 895     op->visit(this);
 896   }

1080 void LIR_List::append(LIR_InsertionBuffer* buffer) {
1081   assert(this == buffer->lir_list(), "wrong lir list");
1082   const int n = _operations.length();
1083 
1084   if (buffer->number_of_ops() > 0) {
1085     // increase size of instructions list
1086     _operations.at_grow(n + buffer->number_of_ops() - 1, NULL);
1087     // insert ops from buffer into instructions list
1088     int op_index = buffer->number_of_ops() - 1;
1089     int ip_index = buffer->number_of_insertion_points() - 1;
1090     int from_index = n - 1;
1091     int to_index = _operations.length() - 1;
1092     for (; ip_index >= 0; ip_index --) {
1093       int index = buffer->index_at(ip_index);
1094       // make room after insertion point
1095       while (index < from_index) {
1096         _operations.at_put(to_index --, _operations.at(from_index --));
1097       }
1098       // insert ops from buffer
1099       for (int i = buffer->count_at(ip_index); i > 0; i --) {
1100         _operations.at_put(to_index --, buffer->op_at(op_index --));
1101       }
1102     }
1103   }
1104 
1105   buffer->finish();
1106 }

1536   if (x->is_set(BlockBegin::exception_entry_flag))         tty->print("ex ");

1545   // print predecessors and successors

2005 void LIR_InsertionBuffer::append(int index, LIR_Op* op) {
2006   assert(_index_and_count.length() % 2 == 0, "must have a count for each index");
2007 
2008   int i = number_of_insertion_points() - 1;
2009   if (i < 0 || index_at(i) < index) {
2010     append_new(index, 1);
2011   } else {
2012     assert(index_at(i) == index, "can append LIR_Ops in ascending order only");
2013     assert(count_at(i) > 0, "check");
2014     set_count_at(i, count_at(i) + 1);
2015   }
2016   _ops.push(op);
2017 
2018   DEBUG_ONLY(verify());
2019 }
