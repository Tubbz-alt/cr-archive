26 package jdk.internal;
27 
28 import java.lang.annotation.*;
29 
30 /**
31  * Indicates the API declaration in question is associated with a
32  * <em>preview feature</em>. See JEP 12: "Preview Language and VM
33  * Features" (http://openjdk.java.net/jeps/12).
34  * @since 14
35  */
36 // Match the meaningful targets of java.lang.Deprecated, omit local
37 // variables and parameter declarations
38 @Target({ElementType.METHOD,
39          ElementType.CONSTRUCTOR,
40          ElementType.FIELD,
41          ElementType.PACKAGE,
42          ElementType.MODULE,
43          ElementType.TYPE})
44  // CLASS retention will hopefully be sufficient for the purposes at hand
45 @Retention(RetentionPolicy.CLASS)
46 // *Not* @Documented
47 public @interface PreviewFeature {
48     /**
49      * Name of the preview feature the annotated API is associated
50      * with.
51      */
52     public Feature feature();
53 
54     public boolean essentialAPI() default false;
55 
56     public enum Feature {
57         PATTERN_MATCHING_IN_INSTANCEOF,
58         // 8242284:
59         // The TEXT_BLOCKS enum constant is not used in the JDK 15 codebase, but
60         // exists to support the bootcycle build of JDK 15. The bootcycle build
61         // of JDK 15 is performed with JDK 14 and the PreviewFeature type from
62         // JDK 15. Since the JDK 14 codebase uses the enum constant, it is
63         // necessary for PreviewFeature in JDK 15 to declare the enum constant.
64         TEXT_BLOCKS,
65         RECORDS,
66         SEALED_CLASSES,
67         ;
68     }
69 }
