 26 package sun.security.rsa;
 27 
 28 import java.io.IOException;
 29 import java.math.BigInteger;
 30 
 31 import java.security.*;
 32 import java.security.spec.*;
 33 import java.security.interfaces.*;
 34 
 35 import sun.security.util.*;
 36 import sun.security.x509.X509Key;

 37 
 38 import sun.security.rsa.RSAUtil.KeyType;
 39 
 40 /**
 41  * RSA public key implementation for "RSA", "RSASSA-PSS" algorithms.
 42  *
 43  * Note: RSA keys must be at least 512 bits long
 44  *
 45  * @see RSAPrivateCrtKeyImpl
 46  * @see RSAPrivateKeyImpl
 47  * @see RSAKeyFactory
 48  *
 49  * @since   1.5
 50  * @author  Andreas Sterbenz
 51  */
 52 public final class RSAPublicKeyImpl extends X509Key implements RSAPublicKey {
 53 
 54     @java.io.Serial
 55     private static final long serialVersionUID = 2644735423591199609L;
 56     private static final BigInteger THREE = BigInteger.valueOf(3);
 57 
 58     private BigInteger n;       // modulus
 59     private BigInteger e;       // public exponent
 60 
 61     private transient KeyType type;
 62 
 63     // optional parameters associated with this RSA key
 64     // specified in the encoding of its AlgorithmId
 65     // must be null for "RSA" keys.
 66     private transient AlgorithmParameterSpec keyParams;

 67 
 68     /**
 69      * Generate a new RSAPublicKey from the specified encoding.
 70      * Used by SunPKCS11 provider.
 71      */
 72     public static RSAPublicKey newKey(byte[] encoded)
 73             throws InvalidKeyException {
 74         return new RSAPublicKeyImpl(encoded);
 75     }
 76 
 77     /**
 78      * Generate a new RSAPublicKey from the specified type and components.
 79      * Used by SunPKCS11 provider.
 80      */
 81     public static RSAPublicKey newKey(KeyType type,
 82             AlgorithmParameterSpec params, BigInteger n, BigInteger e)
 83             throws InvalidKeyException {
 84         return new RSAPublicKeyImpl(type, params, n, e);

 85     }
 86 
 87     /**
 88      * Construct a RSA key from the specified type and components. Used by
 89      * RSAKeyFactory and RSAKeyPairGenerator.
 90      */
 91     RSAPublicKeyImpl(KeyType type, AlgorithmParameterSpec keyParams,
 92             BigInteger n, BigInteger e) throws InvalidKeyException {
 93 
 94         RSAKeyFactory.checkRSAProviderKeyLengths(n.bitLength(), e);
 95         checkExponentRange(n, e);
 96 
 97         this.n = n;
 98         this.e = e;

 99 


100         try {
101             // validate and generate algid encoding
102             algid = RSAUtil.createAlgorithmId(type, keyParams);
103         } catch (ProviderException pe) {
104             throw new InvalidKeyException(pe);
105         }
106 
107         this.type = type;
108         this.keyParams = keyParams;
109 
110         try {
111             // generate the key encoding
112             DerOutputStream out = new DerOutputStream();
113             out.putInteger(n);
114             out.putInteger(e);
115             byte[] keyArray =
116                 new DerValue(DerValue.tag_Sequence,
117                              out.toByteArray()).toByteArray();
118             setKey(new BitArray(keyArray.length*8, keyArray));
119         } catch (IOException exc) {
120             // should never occur
121             throw new InvalidKeyException(exc);
122         }
123     }
124 
125     /**
126      * Construct a key from its encoding. Used by RSAKeyFactory.
127      */
128     RSAPublicKeyImpl(byte[] encoded) throws InvalidKeyException {
129         if (encoded == null || encoded.length == 0) {
130             throw new InvalidKeyException("Missing key encoding");
131         }
132         decode(encoded); // this sets n and e value
133         RSAKeyFactory.checkRSAProviderKeyLengths(n.bitLength(), e);
134         checkExponentRange(n, e);
135 
136         try {
137             // check the validity of oid and params
138             Object[] o = RSAUtil.getTypeAndParamSpec(algid);
139             this.type = (KeyType) o[0];
140             this.keyParams = (AlgorithmParameterSpec) o[1];
141         } catch (ProviderException e) {
142             throw new InvalidKeyException(e);
143         }
144     }
145 
146     // pkg private utility method for checking RSA modulus and public exponent
147     static void checkExponentRange(BigInteger mod, BigInteger exp)
148             throws InvalidKeyException {
149         // the exponent should be smaller than the modulus
150         if (exp.compareTo(mod) >= 0) {
151             throw new InvalidKeyException("exponent is larger than modulus");
152         }
153 
154         // the exponent should be at least 3
155         if (exp.compareTo(THREE) < 0) {
156             throw new InvalidKeyException("exponent is smaller than 3");
157         }
158     }
159 
160     // see JCA doc
161     @Override
162     public String getAlgorithm() {
163         return type.keyAlgo;
164     }
165 
166     // see JCA doc
167     @Override
168     public BigInteger getModulus() {
169         return n;
170     }
171 
172     // see JCA doc
173     @Override
174     public BigInteger getPublicExponent() {
175         return e;
176     }
177 
178     // see JCA doc
179     @Override
180     public AlgorithmParameterSpec getParams() {
181         return keyParams;
182     }
183 
184     /**
185      * Parse the key. Called by X509Key.
186      */
187     protected void parseKeyBits() throws InvalidKeyException {
188         try {
189             DerInputStream in = new DerInputStream(getKey().toByteArray());
190             DerValue derValue = in.getDerValue();
191             if (derValue.tag != DerValue.tag_Sequence) {
192                 throw new IOException("Not a SEQUENCE");
193             }
194             DerInputStream data = derValue.data;
195             n = data.getPositiveBigInteger();
196             e = data.getPositiveBigInteger();
197             if (derValue.data.available() != 0) {
198                 throw new IOException("Extra data available");
199             }
200         } catch (IOException e) {
201             throw new InvalidKeyException("Invalid RSA public key", e);
202         }
203     }
204 
205     // return a string representation of this key for debugging
206     @Override
207     public String toString() {
208         return "Sun " + type.keyAlgo + " public key, " + n.bitLength()
209                + " bits" + "\n  params: " + keyParams + "\n  modulus: " + n
210                + "\n  public exponent: " + e;
211     }
212 
213     @java.io.Serial
214     protected Object writeReplace() throws java.io.ObjectStreamException {
215         return new KeyRep(KeyRep.Type.PUBLIC,
216                         getAlgorithm(),
217                         getFormat(),
218                         getEncoded());
219     }
220 }
