27 package jdk.internal.jextract.impl;
28 
29 import java.nio.file.Path;
30 import java.util.function.Function;
31 import java.util.Objects;
32 import jdk.internal.clang.Cursor;
33 import jdk.internal.clang.SourceLocation;
34 import jdk.incubator.jextract.Position;
35 
36 final class CursorPosition implements Position {
37     private final Cursor cursor;
38     private final Path path;
39     private final int line;
40     private final int column;
41     private final Function<Path, Position> whereFrom;
42 
43     CursorPosition(Cursor cursor, Function<Path, Position> whereFrom) {
44         Objects.requireNonNull(cursor);
45         SourceLocation.Location loc = cursor.getSourceLocation().getFileLocation();
46         this.cursor = cursor;
47         this.path = loc.path();
48         this.line = loc.line();
49         this.column = loc.column();
50         this.whereFrom = whereFrom;
51     }
52 
53     @Override
54     public Path path() { return path; }
55 
56     @Override
57     public int line() { return line; }
58 
59     @Override
60     public int col() { return column; }
61 
62     @Override
63     public Position origin() {
64         return whereFrom.apply(path);
65     }
66 
67     public Cursor cursor() { return cursor; }
68 
69     @Override
70     public String toString() {
71         return PrettyPrinter.position(this);
72     }
73 
74     @Override
75     public int hashCode() {
76         return Objects.hash(this);
77     }
78 
79     @Override
80     public boolean equals(Object o) {
81         if (o instanceof CursorPosition) {
82             CursorPosition other = (CursorPosition) o;
83             return (path().equals(other.path()) && line() == other.line() && col() == other.col()
84                     && origin().equals(other.origin()));
85         }
86         return false;
87     }
88 }
