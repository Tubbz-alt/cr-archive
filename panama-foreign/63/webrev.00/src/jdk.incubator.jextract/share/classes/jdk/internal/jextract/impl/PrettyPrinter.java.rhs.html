 27 package jdk.internal.jextract.impl;
 28 
 29 import java.lang.constant.Constable;
 30 import java.util.Set;
 31 import java.util.stream.Collectors;
 32 import jdk.incubator.foreign.MemoryLayout;
 33 import jdk.incubator.jextract.Declaration;
 34 import jdk.incubator.jextract.Position;
 35 import jdk.incubator.jextract.Type;
 36 
 37 class PrettyPrinter implements Declaration.Visitor<Void, Void> {
 38 
 39     int align = 0;
 40 
 41     void incr() {
 42         align += 4;
 43     }
 44 
 45     void decr() {
 46         align -= 4;
 47     }
 48 
 49     void indent() {
 50         builder.append("                                                                                           ".substring(0, align));
 51     }
 52 
 53     StringBuilder builder = new StringBuilder();
 54 
 55     private void getAttributes(Declaration decl) {
 56         Set<String> attrs = decl.attributeNames();



 57         incr();
 58         if (! attrs.isEmpty()) {
 59             indent();
 60             for (String k: attrs) {
 61                 builder.append("Attr: ");
 62                 builder.append(k);
 63                 builder.append(" -> [");
 64                 builder.append(decl.getAttribute(k).get().stream()
 65                     .map(Constable::toString)
 66                     .collect(Collectors.joining(", ")));
 67                 builder.append("]\n");
 68             }
 69         }
 70         indent();
 71         builder.append("Position: " + decl.pos() + "\n");
 72         decr();
 73     }
 74 
 75     public String print(Declaration decl) {
 76         decl.accept(this, null);
 77         return builder.toString();
 78     }
 79 
 80     @Override
 81     public Void visitScoped(Declaration.Scoped d, Void aVoid) {
 82         indent();
 83         builder.append("Scoped: " + d.kind() + " " + d.name() + d.layout().map(l -> " layout = " + l).orElse(""));
 84         builder.append("\n");
 85         getAttributes(d);
 86         incr();
 87         d.members().forEach(m -> m.accept(this, null));
 88         decr();
 89         return null;
 90     }
 91 
 92     @Override
 93     public Void visitFunction(Declaration.Function d, Void aVoid) {
 94         indent();
 95         builder.append("Function: " + d.name() + " type = " + d.type().accept(typeVisitor, null));
 96         builder.append("\n");
 97         getAttributes(d);
 98         incr();
 99         d.parameters().forEach(m -> m.accept(this, null));
100         decr();
101         return null;
102     }
103 
104     @Override
105     public Void visitVariable(Declaration.Variable d, Void aVoid) {
106         indent();
107         builder.append("Variable: " + d.kind() + " " + d.name() + " type = " + d.type().accept(typeVisitor, null) + ", layout = " + d.layout());
108         builder.append("\n");
109         getAttributes(d);
110         return null;
111     }
112 
113     @Override
114     public Void visitConstant(Declaration.Constant d, Void aVoid) {
115         indent();
116         builder.append("Constant: " + d.name() + " " + d.value() + " type = " + d.type().accept(typeVisitor, null));
117         builder.append("\n");
118         getAttributes(d);
119         return null;
120     }
121 
122     private static Type.Visitor<String, Void> typeVisitor = new Type.Visitor<>() {
123         @Override
124         public String visitPrimitive(Type.Primitive t, Void aVoid) {
125             return t.kind().toString() + t.layout().map(l -> "(layout = " + l + ")").orElse("");
126         }
127 
128         @Override
129         public String visitDelegated(Type.Delegated t, Void aVoid) {
130             switch (t.kind()) {
131                 case TYPEDEF:
132                     return "typedef " + t.name() + " = " + t.type().accept(this, null);
133                 case POINTER:
134                     return "(" + t.type().accept(this, null) + ")*";
135                 default:
136                     return t.kind() + " = " + t.type().accept(this, null);
137             }
138         }
139 
140         @Override
141         public String visitFunction(Type.Function t, Void aVoid) {
142             String res = t.returnType().accept(this, null);
143             String args = t.argumentTypes().stream()
144                     .map(a -> a.accept(this, null))
145                     .collect(Collectors.joining(",", "(", ")"));
146             return res + args;
147         }
148 
149         @Override
150         public String visitDeclared(Type.Declared t, Void aVoid) {
151             return "Declared(" + t.tree().layout().map(MemoryLayout::toString).orElse("") + ")";
152         }
153 
154         @Override
155         public String visitArray(Type.Array t, Void aVoid) {
156             String brackets = String.format("%s[%s]", t.kind() == Type.Array.Kind.VECTOR ? "v" : "",
157                     t.elementCount().isPresent() ? t.elementCount().getAsLong() : "");
158             return t.elementType().accept(this, null) + brackets;
159         }
160 
161         @Override
162         public String visitType(Type t, Void aVoid) {
163             return "Unknown type: " + t.getClass().getName();
164         }
165     };
166 
167     public static String type(Type type) {
168         return type.accept(typeVisitor, null);
169     }
170 
171     public static String position(Position pos) {
172         return String.format("%s:%d:%d@%d",
173                 pos.path() == null ? "N/A" : pos.path().toString(),
174                 pos.line(), pos.col(), pos.depth());
175     }
176 }
