 30 import java.io.ByteArrayOutputStream;
 31 import java.nio.file.Path;
 32 import java.nio.file.Paths;
 33 import java.util.ArrayList;
 34 
 35 import jdk.incubator.foreign.MemoryLayouts;
 36 import jdk.incubator.jextract.Declaration;
 37 import jdk.incubator.jextract.Position;
 38 import jdk.incubator.jextract.Type;
 39 import org.testng.annotations.Test;
 40 import static org.testng.Assert.*;
 41 
 42 public class TestDepth extends JextractApiTestBase {
 43     private final static Type C_INT = Type.primitive(Type.Primitive.Kind.Int, MemoryLayouts.C_INT);
 44     // We need stdint.h for pointer macro, otherwise evaluation failed and Constant declaration is not created
 45     String BUILTIN_INCLUDE = Paths.get(System.getProperty("java.home"), "conf", "jextract").toString();
 46 
 47     private void assertOrigin(Position target, String filename, int depth) {
 48         assertTrue(target.origin().path().endsWith(filename));
 49         assertEquals(target.depth(), depth);
 50     }
 51 
 52     private String getSysRoot() {
 53         try (ByteArrayOutputStream output = new ByteArrayOutputStream()) {
 54             Process xcrun = new ProcessBuilder("xcrun", "--show-sdk-path").start();
 55             xcrun.waitFor();
 56             xcrun.getInputStream().transferTo(output);
 57             String sysroot = output.toString();
 58             // Trim tailing \n
 59             sysroot = sysroot.substring(0, sysroot.length() - 1);
 60             return sysroot;
 61         } catch (Throwable t) {
 62             t.printStackTrace();
 63             return "";
 64         }
 65     }
 66 
 67     @Test
 68     public void parseEmpty() {
 69         Declaration.Scoped root = parse("empty.h");
 70         assertTrue(root.members().isEmpty());
 71     }
 72 
 73     @Test
 74     public void parseMacFramework() {
 75         if (!isMacOSX) {
 76             return;
 77         }
 78 
 79         Declaration.Scoped root = parse("macos.h", "-isysroot", getSysRoot(), "-I", BUILTIN_INCLUDE);
 80         // FIXME: kCFNotFound is not generated
 81         // static const CFIndex kCFNotFound = -1;
 82         Declaration.Constant cf = checkConstant(root, "__COREFOUNDATION_CFBASE__", C_INT, 1L);
 83         assertTrue(cf.pos().path().endsWith("CFBase.h"));
 84         assertOrigin(cf.pos(), "CoreFoundation.h", 3);
 85     }
 86 
 87     @Test
 88     public void parseDepth1() {
 89         Declaration.Scoped root = parse("depth1.h", "-I", BUILTIN_INCLUDE);
 90         Declaration.Function fn = checkFunction(root, "depth1", C_INT);
 91         assertEquals(fn.pos().origin(), Position.NO_POSITION);
 92         assertEquals(fn.pos().depth(), 1);
 93 
 94         fn = checkFunction(root, "depth2", C_INT);
 95         assertOrigin(fn.pos(), "depth1.h", 2);
 96 
 97         Declaration.Variable global = checkGlobal(root, "depth2and3", C_INT);
 98         assertOrigin(global.pos(), "depth1.h", 2);
 99 
100         global = checkGlobal(root, "foo", C_INT);
101         assertTrue(global.pos().path().endsWith("libAsmSymbol.h"));
102         assertOrigin(global.pos(), "depth2.h", 3);
103 
104         Declaration.Constant zero = checkConstant(root, "ZERO", C_INT, 0L);
105         assertTrue(zero.pos().path().endsWith("smoke.h"));
106         assertOrigin(zero.pos(), "depth2and3.h", 3);
107     }
108 }
