107         private Object invoke(Object[] args) throws Throwable {
108             // one trailing Object[]
109             int nNamedArgs = function.argumentLayouts().size();
110             assert(args.length == nNamedArgs + 1);
111             // The last argument is the array of vararg collector
112             Object[] unnamedArgs = (Object[]) args[args.length - 1];
113 
114             int argsCount = nNamedArgs + unnamedArgs.length;
115             Class<?>[] argTypes = new Class<?>[argsCount];
116             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];
117 
118             int pos = 0;
119             for (pos = 0; pos < nNamedArgs; pos++) {
120                 argTypes[pos] = varargs.parameterType(pos);
121                 argLayouts[pos] = function.argumentLayouts().get(pos);
122             }
123 
124             assert pos == nNamedArgs;
125             for (Object o: unnamedArgs) {
126                 argTypes[pos] = normalize(o.getClass());
127                 argLayouts[pos] = variadicLayout(argTypes[pos]);
128                 pos++;
129             }
130             assert pos == argsCount;
131 
132             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);
133             FunctionDescriptor f = (function.returnLayout().isEmpty()) ?
134                     FunctionDescriptor.ofVoid(argLayouts) :
135                     FunctionDescriptor.of(function.returnLayout().get(), argLayouts);
136             MethodHandle mh = Foreign.getInstance().getSystemABI().downcallHandle(symbol, mt, f);
137             // flatten argument list so that it can be passed to an asSpreader MH
138             Object[] allArgs = new Object[nNamedArgs + unnamedArgs.length];
139             System.arraycopy(args, 0, allArgs, 0, nNamedArgs);
140             System.arraycopy(unnamedArgs, 0, allArgs, nNamedArgs, unnamedArgs.length);
141 
142             return mh.asSpreader(Object[].class, argsCount).invoke(allArgs);
143         }

145         private static Class<?> unboxIfNeeded(Class<?> clazz) {
146             if (clazz == Boolean.class) {
147                 return boolean.class;
148             } else if (clazz == Void.class) {
149                 return void.class;
150             } else if (clazz == Byte.class) {
151                 return byte.class;
152             } else if (clazz == Character.class) {
153                 return char.class;
154             } else if (clazz == Short.class) {
155                 return short.class;
156             } else if (clazz == Integer.class) {
157                 return int.class;
158             } else if (clazz == Long.class) {
159                 return long.class;
160             } else if (clazz == Float.class) {
161                 return float.class;
162             } else if (clazz == Double.class) {

169         private Class<?> normalize(Class<?> c) {
170             c = unboxIfNeeded(c);
171             if (c.isPrimitive()) {
172                 return c;
173             }
174             if (MemoryAddress.class.isAssignableFrom(c)) {
175                 return MemoryAddress.class;
176             }
177             if (MemorySegment.class.isAssignableFrom(c)) {
178                 return MemorySegment.class;
179             }
180             throw new IllegalArgumentException("Invalid type for ABI: " + c.getTypeName());
181         }

183         private MemoryLayout variadicLayout(Class<?> c) {
184             if (c == char.class || c == byte.class || c == short.class || c == int.class || c == long.class) {
185                 //it is ok to approximate with a machine word here; numerics arguments in a prototype-less
186                 //function call are always rounded up to a register size anyway.
187                 return C_LONGLONG;
188             } else if (c == float.class || c == double.class) {
189                 return C_DOUBLE;
190             } else if (MemoryAddress.class.isAssignableFrom(c)) {
191                 return C_POINTER;
192             } else {
193                 throw new IllegalArgumentException("Unhandled variadic argument class: " + c);
194             }
195         }
