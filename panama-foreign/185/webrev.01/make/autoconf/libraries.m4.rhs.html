 38 ################################################################################
 39 # Determine which libraries are needed for this configuration
 40 ################################################################################
 41 AC_DEFUN_ONCE([LIB_DETERMINE_DEPENDENCIES],
 42 [
 43   # Check if X11 is needed
 44   if test "x$OPENJDK_TARGET_OS" = xwindows || test "x$OPENJDK_TARGET_OS" = xmacosx; then
 45     # No X11 support on windows or macosx
 46     NEEDS_LIB_X11=false
 47   else
 48     # All other instances need X11, even if building headless only, libawt still
 49     # needs X11 headers.
 50     NEEDS_LIB_X11=true
 51   fi
 52 
 53   # Check if fontconfig is needed
 54   if test "x$OPENJDK_TARGET_OS" = xwindows || test "x$OPENJDK_TARGET_OS" = xmacosx; then
 55     # No fontconfig support on windows or macosx
 56     NEEDS_LIB_FONTCONFIG=false
 57   else
 58     # All other instances need fontconfig, even if building headless only,
 59     # libawt still needs fontconfig headers.
 60     NEEDS_LIB_FONTCONFIG=true
 61   fi
 62 
 63   # Check if cups is needed
 64   if test "x$OPENJDK_TARGET_OS" = xwindows; then
 65     # Windows have a separate print system
 66     NEEDS_LIB_CUPS=false
 67   else
 68     NEEDS_LIB_CUPS=true
 69   fi
 70 
 71   # A custom hook may have set this already
 72   if test "x$NEEDS_LIB_FREETYPE" = "x"; then
 73     NEEDS_LIB_FREETYPE=true
 74   fi
 75 
 76   # Check if alsa is needed
 77   if test "x$OPENJDK_TARGET_OS" = xlinux; then
 78     NEEDS_LIB_ALSA=true
 79   else
 80     NEEDS_LIB_ALSA=false
 81   fi
 82 
 83   # Check if ffi is needed
 84   if HOTSPOT_CHECK_JVM_VARIANT(zero); then
 85     NEEDS_LIB_FFI=true
 86   else
 87     NEEDS_LIB_FFI=false
 88   fi
 89 ])

 91 ################################################################################
 92 # Parse library options, and setup needed libraries
 93 ################################################################################
 94 AC_DEFUN_ONCE([LIB_SETUP_LIBRARIES],
 95 [
 96   LIB_SETUP_STD_LIBS
 97   LIB_SETUP_X11
 98   LIB_SETUP_CUPS
 99   LIB_SETUP_FONTCONFIG
100   LIB_SETUP_FREETYPE
101   LIB_SETUP_ALSA
102   LIB_SETUP_LIBFFI
103   LIB_SETUP_BUNDLED_LIBS
104   LIB_SETUP_MISC_LIBS
105   LIB_TESTS_SETUP_GRAALUNIT
106   LIB_TESTS_SETUP_GTEST
107   LIB_SETUP_LIBCLANG
108 
109   BASIC_JDKLIB_LIBS=""
110   if test "x$TOOLCHAIN_TYPE" != xmicrosoft; then
111     BASIC_JDKLIB_LIBS="-ljava -ljvm"
112   fi
113 
114   # Math library
115   BASIC_JVM_LIBS="$LIBM"
116 
117   # Dynamic loading library
118   if test "x$OPENJDK_TARGET_OS" = xlinux || test "x$OPENJDK_TARGET_OS" = xaix; then
119     BASIC_JVM_LIBS="$BASIC_JVM_LIBS $LIBDL"
120   fi
121 
122   # Threading library
123   if test "x$OPENJDK_TARGET_OS" = xlinux || test "x$OPENJDK_TARGET_OS" = xaix; then
124     BASIC_JVM_LIBS="$BASIC_JVM_LIBS -lpthread"
125   fi
126 
127   # perfstat lib
128   if test "x$OPENJDK_TARGET_OS" = xaix; then
129     BASIC_JVM_LIBS="$BASIC_JVM_LIBS -lperfstat"
130   fi
131 
132   if test "x$OPENJDK_TARGET_OS" = xwindows; then
133     BASIC_JVM_LIBS="$BASIC_JVM_LIBS kernel32.lib user32.lib gdi32.lib winspool.lib \
134         comdlg32.lib advapi32.lib shell32.lib ole32.lib oleaut32.lib uuid.lib \
135         wsock32.lib winmm.lib version.lib psapi.lib"
136   fi
137 
138   JDKLIB_LIBS="$BASIC_JDKLIB_LIBS"
139   JDKEXE_LIBS=""
140   JVM_LIBS="$BASIC_JVM_LIBS"
141   OPENJDK_BUILD_JDKLIB_LIBS="$BASIC_JDKLIB_LIBS"
142   OPENJDK_BUILD_JVM_LIBS="$BASIC_JVM_LIBS"
143 
144   AC_SUBST(JDKLIB_LIBS)
145   AC_SUBST(JDKEXE_LIBS)
146   AC_SUBST(JVM_LIBS)
147   AC_SUBST(OPENJDK_BUILD_JDKLIB_LIBS)
148   AC_SUBST(OPENJDK_BUILD_JVM_LIBS)
149 ])
150 
151 ################################################################################
152 # Setup various libraries, typically small system libraries
153 ################################################################################
154 AC_DEFUN_ONCE([LIB_SETUP_MISC_LIBS],
155 [
156   # Setup libm (the maths library)
157   if test "x$OPENJDK_TARGET_OS" != "xwindows"; then
158     AC_CHECK_LIB(m, cos, [], [
159         AC_MSG_NOTICE([Maths library was not found])
160     ])
161     LIBM="-lm"
162   else
163     LIBM=""
164   fi
165   AC_SUBST(LIBM)
166 
167   # Setup libdl (for dynamic library loading)
168   save_LIBS="$LIBS"
169   LIBS=""
170   AC_CHECK_LIB(dl, dlopen)
171   LIBDL="$LIBS"
172   AC_SUBST(LIBDL)
173   LIBS="$save_LIBS"
174 
175   # Control if libzip can use mmap. Available for purposes of overriding.
176   LIBZIP_CAN_USE_MMAP=true
177   AC_SUBST(LIBZIP_CAN_USE_MMAP)
178 ])
