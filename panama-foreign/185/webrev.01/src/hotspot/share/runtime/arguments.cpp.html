 119 bool PathString::set_value(const char *value) {
 120   if (_value != NULL) {
 121     FreeHeap(_value);
 122   }
 123   _value = AllocateHeap(strlen(value)+1, mtArguments);
 124   assert(_value != NULL, "Unable to allocate space for new path value");
 125   if (_value != NULL) {
 126     strcpy(_value, value);
 127   } else {
 128     // not able to allocate
 129     return false;
 130   }
 131   return true;
 132 }
 133 
 134 void PathString::append_value(const char *value) {
 135   char *sp;
 136   size_t len = 0;
 137   if (value != NULL) {
 138     len = strlen(value);
 139     if (_value != NULL) {
 140       len += strlen(_value);
 141     }
 142     sp = AllocateHeap(len+2, mtArguments);
 143     assert(sp != NULL, "Unable to allocate space for new append path value");
 144     if (sp != NULL) {
 145       if (_value != NULL) {
 146         strcpy(sp, _value);
 147         strcat(sp, os::path_separator());
 148         strcat(sp, value);
 149         FreeHeap(_value);
 150       } else {
 151         strcpy(sp, value);
 152       }
 153       _value = sp;
 154     }
 155   }
 156 }

 224 // Check if head of 'option' matches 'name', and sets 'tail' to the remaining
 225 // part of the option string.
 226 static bool match_option(const JavaVMOption *option, const char* name,
 227                          const char** tail) {
 228   size_t len = strlen(name);
 229   if (strncmp(option->optionString, name, len) == 0) {
 230     *tail = option->optionString + len;
 231     return true;
 232   } else {
 233     return false;
 234   }
 235 }
 236 
 237 // Check if 'option' matches 'name'. No "tail" is allowed.
 238 static bool match_option(const JavaVMOption *option, const char* name) {
 239   const char* tail = NULL;
 240   bool result = match_option(option, name, &tail);
 241   if (tail != NULL && *tail == '\0') {
 242     return result;
 243   } else {
 244     return false;
 245   }
 246 }
 247 
 248 // Return true if any of the strings in null-terminated array 'names' matches.
 249 // If tail_allowed is true, then the tail must begin with a colon; otherwise,
 250 // the option must match exactly.
 251 static bool match_option(const JavaVMOption* option, const char** names, const char** tail,
 252   bool tail_allowed) {
 253   for (/* empty */; *names != NULL; ++names) {
 254   if (match_option(option, *names, tail)) {
 255       if (**tail == '\0' || (tail_allowed && **tail == ':')) {
 256         return true;
 257       }
 258     }
 259   }
 260   return false;
 261 }

 329 // Return TRUE if option matches 'property', or 'property=', or 'property.'.
 330 static bool matches_property_suffix(const char* option, const char* property, size_t len) {
 331   return ((strncmp(option, property, len) == 0) &&
 332           (option[len] == '=' || option[len] == '.' || option[len] == '\0'));
 333 }

 335 // Return true if property starts with "jdk.module." and its ensuing chars match
 336 // any of the reserved module properties.
 337 // property should be passed without the leading "-D".
 338 bool Arguments::is_internal_module_property(const char* property) {
 339   assert((strncmp(property, "-D", 2) != 0), "Unexpected leading -D");
 340   if  (strncmp(property, MODULE_PROPERTY_PREFIX, MODULE_PROPERTY_PREFIX_LEN) == 0) {
 341     const char* property_suffix = property + MODULE_PROPERTY_PREFIX_LEN;
 342     if (matches_property_suffix(property_suffix, ADDEXPORTS, ADDEXPORTS_LEN) ||
 343         matches_property_suffix(property_suffix, ADDREADS, ADDREADS_LEN) ||
 344         matches_property_suffix(property_suffix, ADDOPENS, ADDOPENS_LEN) ||
 345         matches_property_suffix(property_suffix, PATCH, PATCH_LEN) ||
 346         matches_property_suffix(property_suffix, ADDMODS, ADDMODS_LEN) ||
 347         matches_property_suffix(property_suffix, LIMITMODS, LIMITMODS_LEN) ||
 348         matches_property_suffix(property_suffix, PATH, PATH_LEN) ||
 349         matches_property_suffix(property_suffix, UPGRADE_PATH, UPGRADE_PATH_LEN)) {
 350       return true;
 351     }
 352   }
 353   return false;
 354 }

 356 // Process java launcher properties.
 357 void Arguments::process_sun_java_launcher_properties(JavaVMInitArgs* args) {
 358   // See if sun.java.launcher or sun.java.launcher.is_altjvm is defined.
 359   // Must do this before setting up other system properties,
 360   // as some of them may depend on launcher type.
 361   for (int index = 0; index < args->nOptions; index++) {
 362     const JavaVMOption* option = args->options + index;
 363     const char* tail;
 364 
 365     if (match_option(option, "-Dsun.java.launcher=", &tail)) {
 366       process_java_launcher_argument(tail, option->extraInfo);
 367       continue;
 368     }
 369     if (match_option(option, "-Dsun.java.launcher.is_altjvm=", &tail)) {
 370       if (strcmp(tail, "true") == 0) {
 371         _sun_java_launcher_is_altjvm = true;
 372       }
 373       continue;
 374     }
 375   }
 376 }

 378 // Initialize system properties key and value.
 379 void Arguments::init_system_properties() {
 380 
 381   // Set up _system_boot_class_path which is not a property but
 382   // relies heavily on argument processing and the jdk.boot.class.path.append
 383   // property. It is used to store the underlying system boot class path.
 384   _system_boot_class_path = new PathString(NULL);

 386   PropertyList_add(&_system_properties, new SystemProperty("java.vm.specification.name",

 389   PropertyList_add(&_system_properties, new SystemProperty("java.vm.name", VM_Version::vm_name(),  false));
 390   PropertyList_add(&_system_properties, new SystemProperty("jdk.debug", VM_Version::jdk_debug_level(),  false));
 391 
 392   // Initialize the vm.info now, but it will need updating after argument parsing.
 393   _vm_info = new SystemProperty("java.vm.info", VM_Version::vm_info_string(), true);
 394 
 395   // Following are JVMTI agent writable properties.
 396   // Properties values are set to NULL and they are
 397   // os specific they are initialized in os::init_system_properties_values().
 398   _sun_boot_library_path = new SystemProperty("sun.boot.library.path", NULL,  true);
 399   _java_library_path = new SystemProperty("java.library.path", NULL,  true);
 400   _java_home =  new SystemProperty("java.home", NULL,  true);
 401   _java_class_path = new SystemProperty("java.class.path", "",  true);
 402   // jdk.boot.class.path.append is a non-writeable, internal property.
 403   // It can only be set by either:
 404   //    - -Xbootclasspath/a:
 405   //    - AddToBootstrapClassLoaderSearch during JVMTI OnLoad phase
 406   _jdk_boot_class_path_append = new SystemProperty("jdk.boot.class.path.append", "", false, true);
 407 
 408   // Add to System Property list.
 409   PropertyList_add(&_system_properties, _sun_boot_library_path);
 410   PropertyList_add(&_system_properties, _java_library_path);
 411   PropertyList_add(&_system_properties, _java_home);
 412   PropertyList_add(&_system_properties, _java_class_path);
 413   PropertyList_add(&_system_properties, _jdk_boot_class_path_append);
 414   PropertyList_add(&_system_properties, _vm_info);
 415 
 416   // Set OS specific system properties values
 417   os::init_system_properties_values();
 418 }

 437 /*
 438  *  -XX argument processing:
 439  *
 440  *  -XX arguments are defined in several places, such as:
 441  *      globals.hpp, globals_<cpu>.hpp, globals_<os>.hpp, <compiler>_globals.hpp, or <gc>_globals.hpp.
 442  *  -XX arguments are parsed in parse_argument().
 443  *  -XX argument bounds checking is done in check_vm_args_consistency().
 444  *
 445  * Over time -XX arguments may change. There are mechanisms to handle common cases:
 446  *
 447  *      ALIASED: An option that is simply another name for another option. This is often
 448  *               part of the process of deprecating a flag, but not all aliases need
 449  *               to be deprecated.
 450  *
 451  *               Create an alias for an option by adding the old and new option names to the
 452  *               "aliased_jvm_flags" table. Delete the old variable from globals.hpp (etc).
 453  *
 454  *   DEPRECATED: An option that is supported, but a warning is printed to let the user know that
 455  *               support may be removed in the future. Both regular and aliased options may be
 456  *               deprecated.
 457  *
 458  *               Add a deprecation warning for an option (or alias) by adding an entry in the
 459  *               "special_jvm_flags" table and setting the "deprecated_in" field.
 460  *               Often an option "deprecated" in one major release will
 461  *               be made "obsolete" in the next. In this case the entry should also have its
 462  *               "obsolete_in" field set.
 463  *
 464  *     OBSOLETE: An option that has been removed (and deleted from globals.hpp), but is still accepted
 465  *               on the command line. A warning is printed to let the user know that option might not
 466  *               be accepted in the future.
 467  *
 468  *               Add an obsolete warning for an option by adding an entry in the "special_jvm_flags"
 469  *               table and setting the "obsolete_in" field.
 470  *
 471  *      EXPIRED: A deprecated or obsolete option that has an "accept_until" version less than or equal
 472  *               to the current JDK version. The system will flatly refuse to admit the existence of
 473  *               the flag. This allows a flag to die automatically over JDK releases.
 474  *
 475  *               Note that manual cleanup of expired options should be done at major JDK version upgrades:
 476  *                  - Newly expired options should be removed from the special_jvm_flags and aliased_jvm_flags tables.
 477  *                  - Newly obsolete or expired deprecated options should have their global variable
 478  *                    definitions removed (from globals.hpp, etc) and related implementations removed.
 479  *
 480  * Recommended approach for removing options:
 481  *
 482  * To remove options commonly used by customers (e.g. product -XX options), use
 483  * the 3-step model adding major release numbers to the deprecate, obsolete and expire columns.
 484  *
 485  * To remove internal options (e.g. diagnostic, experimental, develop options), use
 486  * a 2-step model adding major release numbers to the obsolete and expire columns.
 487  *
 488  * To change the name of an option, use the alias table as well as a 2-step
 489  * model adding major release numbers to the deprecate and expire columns.
 490  * Think twice about aliasing commonly used customer options.
 491  *
 492  * There are times when it is appropriate to leave a future release number as undefined.
 493  *
 494  * Tests:  Aliases should be tested in VMAliasOptions.java.
 495  *         Deprecated options should be tested in VMDeprecatedOptions.java.
 496  */

 661 bool Arguments::is_obsolete_flag(const char *flag_name, JDK_Version* version) {
 662   assert(version != NULL, "Must provide a version buffer");
 663   SpecialFlag flag;
 664   if (lookup_special_flag(flag_name, flag)) {
 665     if (!flag.obsolete_in.is_undefined()) {
 666       if (!version_less_than(JDK_Version::current(), flag.obsolete_in)) {
 667         *version = flag.obsolete_in;
 668         // This flag may have been marked for obsoletion in this version, but we may not
 669         // have actually removed it yet. Rather than ignoring it as soon as we reach
 670         // this version we allow some time for the removal to happen. So if the flag
 671         // still actually exists we process it as normal, but issue an adjusted warning.
 672         const JVMFlag *real_flag = JVMFlag::find_declared_flag(flag_name);
 673         if (real_flag != NULL) {
 674           char version_str[256];
 675           version->to_string(version_str, sizeof(version_str));
 676           warning("Temporarily processing option %s; support is scheduled for removal in %s",
 677                   flag_name, version_str);
 678           return false;
 679         }
 680         return true;
 681       }
 682     }
 683   }
 684   return false;
 685 }

 736 // Verifies the correctness of the entries in the special_jvm_flags table.
 737 // If there is a semantic error (i.e. a bug in the table) such as the obsoletion
 738 // version being earlier than the deprecation version, then a warning is issued
 739 // and verification fails - by returning false. If it is detected that the table
 740 // is out of date, with respect to the current version, then ideally a warning is
 741 // issued but verification does not fail. This allows the VM to operate when the
 742 // version is first updated, without needing to update all the impacted flags at
 743 // the same time. In practice we can't issue the warning immediately when the version
 744 // is updated as it occurs for every test and some tests are not prepared to handle
 745 // unexpected output - see 8196739. Instead we only check if the table is up-to-date
 746 // if the check_globals flag is true, and in addition allow a grace period and only
 747 // check for stale flags when we hit build 25 (which is far enough into the 6 month
 748 // release cycle that all flag updates should have been processed, whilst still
 749 // leaving time to make the change before RDP2).
 750 // We use a gtest to call this, passing true, so that we can detect stale flags before
 751 // the end of the release cycle.
 752 
 753 static const int SPECIAL_FLAG_VALIDATION_BUILD = 25;

 755 bool Arguments::verify_special_jvm_flags(bool check_globals) {
 756   bool success = true;
 757   for (size_t i = 0; special_jvm_flags[i].name != NULL; i++) {
 758     const SpecialFlag& flag = special_jvm_flags[i];
 759     if (lookup_special_flag(flag.name, i)) {
 760       warning("Duplicate special flag declaration \"%s\"", flag.name);
 761       success = false;
 762     }
 763     if (flag.deprecated_in.is_undefined() &&
 764         flag.obsolete_in.is_undefined()) {
 765       warning("Special flag entry \"%s\" must declare version deprecated and/or obsoleted in.", flag.name);
 766       success = false;
 767     }
 768 
 769     if (!flag.deprecated_in.is_undefined()) {
 770       if (!version_less_than(flag.deprecated_in, flag.obsolete_in)) {
 771         warning("Special flag entry \"%s\" must be deprecated before obsoleted.", flag.name);
 772         success = false;
 773       }
 774 
 775       if (!version_less_than(flag.deprecated_in, flag.expired_in)) {
 776         warning("Special flag entry \"%s\" must be deprecated before expired.", flag.name);
 777         success = false;
 778       }
 779     }
 780 
 781     if (!flag.obsolete_in.is_undefined()) {
 782       if (!version_less_than(flag.obsolete_in, flag.expired_in)) {
 783         warning("Special flag entry \"%s\" must be obsoleted before expired.", flag.name);
 784         success = false;
 785       }
 786 
 787       // if flag has become obsolete it should not have a "globals" flag defined anymore.
 788       if (check_globals && VM_Version::vm_build_number() >= SPECIAL_FLAG_VALIDATION_BUILD &&
 789           !version_less_than(JDK_Version::current(), flag.obsolete_in)) {
 790         if (JVMFlag::find_declared_flag(flag.name) != NULL) {
 791           warning("Global variable for obsolete special flag entry \"%s\" should be removed", flag.name);
 792           success = false;
 793         }
 794       }
 795 
 796     } else if (!flag.expired_in.is_undefined()) {
 797       warning("Special flag entry \"%s\" must be explicitly obsoleted before expired.", flag.name);
 798       success = false;
 799     }
 800 
 801     if (!flag.expired_in.is_undefined()) {
 802       // if flag has become expired it should not have a "globals" flag defined anymore.
 803       if (check_globals && VM_Version::vm_build_number() >= SPECIAL_FLAG_VALIDATION_BUILD &&
 804           !version_less_than(JDK_Version::current(), flag.expired_in)) {
 805         if (JVMFlag::find_declared_flag(flag.name) != NULL) {
 806           warning("Global variable for expired flag entry \"%s\" should be removed", flag.name);
 807           success = false;
 808         }
 809       }
 810     }
 811   }
 812   return success;
 813 }

 816 // Parses a size specification string.
 817 bool Arguments::atojulong(const char *s, julong* result) {
 818   julong n = 0;
 819 
 820   // First char must be a digit. Don't allow negative numbers or leading spaces.
 821   if (!isdigit(*s)) {
 822     return false;
 823   }
 824 
 825   bool is_hex = (s[0] == '0' && (s[1] == 'x' || s[1] == 'X'));
 826   char* remainder;
 827   errno = 0;
 828   n = strtoull(s, &remainder, (is_hex ? 16 : 10));
 829   if (errno != 0) {
 830     return false;
 831   }
 832 
 833   // Fail if no number was read at all or if the remainder contains more than a single non-digit character.
 834   if (remainder == s || strlen(remainder) > 1) {
 835     return false;
 836   }
 837 
 838   switch (*remainder) {
 839     case 'T': case 't':
 840       *result = n * G * K;
 841       // Check for overflow.
 842       if (*result/((julong)G * K) != n) return false;
 843       return true;
 844     case 'G': case 'g':
 845       *result = n * G;
 846       if (*result/G != n) return false;
 847       return true;
 848     case 'M': case 'm':
 849       *result = n * M;
 850       if (*result/M != n) return false;
 851       return true;
 852     case 'K': case 'k':
 853       *result = n * K;
 854       if (*result/K != n) return false;
 855       return true;
 856     case '\0':
 857       *result = n;
 858       return true;
 859     default:
 860       return false;
 861   }
 862 }

 911   julong v;
 912   int int_v;
 913   intx intx_v;
 914   bool is_neg = false;
 915 
 916   if (flag == NULL) {
 917     return false;
 918   }
 919 
 920   // Check the sign first since atojulong() parses only unsigned values.
 921   if (*value == '-') {

 923       return false;
 924     }
 925     value++;
 926     is_neg = true;
 927   }
 928   if (!Arguments::atojulong(value, &v)) {
 929     return false;
 930   }

 970 static bool append_to_string_flag(JVMFlag* flag, const char* new_value, JVMFlag::Flags origin) {
 971   const char* old_value = "";
 972   if (JVMFlag::ccstrAt(flag, &old_value) != JVMFlag::SUCCESS) return false;
 973   size_t old_len = old_value != NULL ? strlen(old_value) : 0;
 974   size_t new_len = strlen(new_value);
 975   const char* value;
 976   char* free_this_too = NULL;
 977   if (old_len == 0) {
 978     value = new_value;
 979   } else if (new_len == 0) {
 980     value = old_value;
 981   } else {
 982      size_t length = old_len + 1 + new_len + 1;
 983      char* buf = NEW_C_HEAP_ARRAY(char, length, mtArguments);
 984     // each new setting adds another LINE to the switch:
 985     jio_snprintf(buf, length, "%s\n%s", old_value, new_value);
 986     value = buf;
 987     free_this_too = buf;
 988   }
 989   (void) JVMFlag::ccstrAtPut(flag, &value, origin);
 990   // JVMFlag always returns a pointer that needs freeing.
 991   FREE_C_HEAP_ARRAY(char, value);
 992   // JVMFlag made its own copy, so I must delete my own temp. buffer.
 993   FREE_C_HEAP_ARRAY(char, free_this_too);
 994   return true;

 998   const char* real_name = real_flag_name(arg);

1009       // as obsoletion must come first.

1019           warning("Option %s was deprecated in version %s and will likely be removed in a future release. Use option %s instead.",
1020                   arg, version, real_name);

1071 bool Arguments::parse_argument(const char* arg, JVMFlag::Flags origin) {
1072 
1073   // range of acceptable characters spelled out for portability reasons
1074 #define NAME_RANGE  "[abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789_]"
1075 #define BUFLEN 255
1076   char name[BUFLEN+1];
1077   char dummy;
1078   const char* real_name;
1079   bool warn_if_deprecated = true;
1080 
1081   if (sscanf(arg, "-%" XSTR(BUFLEN) NAME_RANGE "%c", name, &dummy) == 1) {
1082     AliasedLoggingFlag alf = catch_logging_aliases(name, false);
1083     if (alf.alias_name != NULL){
1084       LogConfiguration::configure_stdout(LogLevel::Off, alf.exactMatch, alf.tag0, alf.tag1, alf.tag2, alf.tag3, alf.tag4, alf.tag5);
1085       return true;
1086     }
1087     real_name = handle_aliases_and_deprecation(name, warn_if_deprecated);
1088     if (real_name == NULL) {
1089       return false;
1090     }
1091     JVMFlag* flag = JVMFlag::find_flag(real_name);
1092     return set_bool_flag(flag, false, origin);
1093   }
1094   if (sscanf(arg, "+%" XSTR(BUFLEN) NAME_RANGE "%c", name, &dummy) == 1) {
1095     AliasedLoggingFlag alf = catch_logging_aliases(name, true);
1096     if (alf.alias_name != NULL){
1097       LogConfiguration::configure_stdout(alf.level, alf.exactMatch, alf.tag0, alf.tag1, alf.tag2, alf.tag3, alf.tag4, alf.tag5);
1098       return true;
1099     }
1100     real_name = handle_aliases_and_deprecation(name, warn_if_deprecated);
1101     if (real_name == NULL) {
1102       return false;
1103     }
1104     JVMFlag* flag = JVMFlag::find_flag(real_name);
1105     return set_bool_flag(flag, true, origin);
1106   }
1107 
1108   char punct;
1109   if (sscanf(arg, "%" XSTR(BUFLEN) NAME_RANGE "%c", name, &punct) == 2 && punct == '=') {
1110     const char* value = strchr(arg, '=') + 1;
1111 
1112     // this scanf pattern matches both strings (handled here) and numbers (handled later))
1113     AliasedLoggingFlag alf = catch_logging_aliases(name, true);
1114     if (alf.alias_name != NULL) {
1115       LogConfiguration::configure_stdout(alf.level, alf.exactMatch, alf.tag0, alf.tag1, alf.tag2, alf.tag3, alf.tag4, alf.tag5);
1116       return true;
1117     }
1118     real_name = handle_aliases_and_deprecation(name, warn_if_deprecated);
1119     if (real_name == NULL) {
1120       return false;
1121     }
1122     JVMFlag* flag = JVMFlag::find_flag(real_name);
1123     if (flag != NULL && flag->is_ccstr()) {
1124       if (flag->ccstr_accumulates()) {
1125         return append_to_string_flag(flag, value, origin);
1126       } else {
1127         if (value[0] == '\0') {
1128           value = NULL;
1129         }
1130         return set_string_flag(flag, value, origin);
1131       }
1132     } else {
1133       warn_if_deprecated = false; // if arg is deprecated, we've already done warning...
1134     }
1135   }
1136 
1137   if (sscanf(arg, "%" XSTR(BUFLEN) NAME_RANGE ":%c", name, &punct) == 2 && punct == '=') {
1138     const char* value = strchr(arg, '=') + 1;
1139     // -XX:Foo:=xxx will reset the string flag to the given value.
1140     if (value[0] == '\0') {
1141       value = NULL;
1142     }
1143     real_name = handle_aliases_and_deprecation(name, warn_if_deprecated);
1144     if (real_name == NULL) {
1145       return false;
1146     }
1147     JVMFlag* flag = JVMFlag::find_flag(real_name);
1148     return set_string_flag(flag, value, origin);
1149   }
1150 
1151 #define SIGNED_FP_NUMBER_RANGE "[-0123456789.eE+]"
1152 #define SIGNED_NUMBER_RANGE    "[-0123456789]"
1153 #define        NUMBER_RANGE    "[0123456789eE+-]"
1154   char value[BUFLEN + 1];
1155   char value2[BUFLEN + 1];
1156   if (sscanf(arg, "%" XSTR(BUFLEN) NAME_RANGE "=" "%" XSTR(BUFLEN) SIGNED_NUMBER_RANGE "." "%" XSTR(BUFLEN) NUMBER_RANGE "%c", name, value, value2, &dummy) == 3) {
1157     // Looks like a floating-point number -- try again with more lenient format string
1158     if (sscanf(arg, "%" XSTR(BUFLEN) NAME_RANGE "=" "%" XSTR(BUFLEN) SIGNED_FP_NUMBER_RANGE "%c", name, value, &dummy) == 2) {
1159       real_name = handle_aliases_and_deprecation(name, warn_if_deprecated);
1160       if (real_name == NULL) {
1161         return false;
1162       }
1163       JVMFlag* flag = JVMFlag::find_flag(real_name);
1164       return set_fp_numeric_flag(flag, value, origin);
1165     }
1166   }
1167 
1168 #define VALUE_RANGE "[-kmgtxKMGTX0123456789abcdefABCDEF]"
1169   if (sscanf(arg, "%" XSTR(BUFLEN) NAME_RANGE "=" "%" XSTR(BUFLEN) VALUE_RANGE "%c", name, value, &dummy) == 2) {
1170     real_name = handle_aliases_and_deprecation(name, warn_if_deprecated);
1171     if (real_name == NULL) {
1172       return false;
1173     }
1174     JVMFlag* flag = JVMFlag::find_flag(real_name);
1175     return set_numeric_flag(flag, value, origin);
1176   }
1177 
1178   return false;
1179 }

1284 bool Arguments::process_argument(const char* arg,
1285                                  jboolean ignore_unrecognized,
1286                                  JVMFlag::Flags origin) {
1287   JDK_Version since = JDK_Version();
1288 
1289   if (parse_argument(arg, origin)) {
1290     return true;
1291   }
1292 
1293   // Determine if the flag has '+', '-', or '=' characters.
1294   bool has_plus_minus = (*arg == '+' || *arg == '-');
1295   const char* const argname = has_plus_minus ? arg + 1 : arg;
1296 
1297   size_t arg_len;
1298   const char* equal_sign = strchr(argname, '=');
1299   if (equal_sign == NULL) {
1300     arg_len = strlen(argname);
1301   } else {
1302     arg_len = equal_sign - argname;
1303   }
1304 
1305   // Only make the obsolete check for valid arguments.
1306   if (arg_len <= BUFLEN) {
1307     // Construct a string which consists only of the argument name without '+', '-', or '='.
1308     char stripped_argname[BUFLEN+1]; // +1 for '\0'
1309     jio_snprintf(stripped_argname, arg_len+1, "%s", argname); // +1 for '\0'
1310     if (is_obsolete_flag(stripped_argname, &since)) {
1311       char version[256];
1312       since.to_string(version, sizeof(version));
1313       warning("Ignoring option %s; support was removed in %s", stripped_argname, version);
1314       return true;
1315     }
1316 #ifndef PRODUCT
1317     else {
1318       const char* replacement;
1319       if ((replacement = removed_develop_logging_flag_name(stripped_argname)) != NULL){
1320         log_warning(arguments)("%s has been removed. Please use %s instead.",
1321                                stripped_argname,
1322                                replacement);
1323         return false;
1324       }
1325     }
1326 #endif //PRODUCT
1327   }
1328 
1329   // For locked flags, report a custom error message if available.
1330   // Otherwise, report the standard unrecognized VM option.
1331   const JVMFlag* found_flag = JVMFlag::find_declared_flag((const char*)argname, arg_len);
1332   if (found_flag != NULL) {
1333     char locked_message_buf[BUFLEN];
1334     JVMFlag::MsgType msg_type = found_flag->get_locked_message(locked_message_buf, BUFLEN);
1335     if (strlen(locked_message_buf) == 0) {
1336       if (found_flag->is_bool() && !has_plus_minus) {
1337         jio_fprintf(defaultStream::error_stream(),
1338           "Missing +/- setting for VM option '%s'\n", argname);
1339       } else if (!found_flag->is_bool() && has_plus_minus) {
1340         jio_fprintf(defaultStream::error_stream(),
1341           "Unexpected +/- setting in VM option '%s'\n", argname);
1342       } else {
1343         jio_fprintf(defaultStream::error_stream(),
1344           "Improperly specified VM option '%s'\n", argname);
1345       }
1346     } else {
1347 #ifdef PRODUCT
1348       bool mismatched = ((msg_type == JVMFlag::NOTPRODUCT_FLAG_BUT_PRODUCT_BUILD) ||
1349                          (msg_type == JVMFlag::DEVELOPER_FLAG_BUT_PRODUCT_BUILD));
1350       if (ignore_unrecognized && mismatched) {
1351         return true;
1352       }
1353 #endif
1354       jio_fprintf(defaultStream::error_stream(), "%s", locked_message_buf);
1355     }
1356   } else {
1357     if (ignore_unrecognized) {
1358       return true;
1359     }
1360     jio_fprintf(defaultStream::error_stream(),
1361                 "Unrecognized VM option '%s'\n", argname);
1362     JVMFlag* fuzzy_matched = JVMFlag::fuzzy_match((const char*)argname, arg_len, true);
1363     if (fuzzy_matched != NULL) {
1364       jio_fprintf(defaultStream::error_stream(),
1365                   "Did you mean '%s%s%s'? ",
1366                   (fuzzy_matched->is_bool()) ? "(+/-)" : "",
1367                   fuzzy_matched->_name,
1368                   (fuzzy_matched->is_bool()) ? "" : "=<value>");
1369     }
1370   }
1371 
1372   // allow for commandline "commenting out" options like -XX:#+Verbose
1373   return arg[0] == '#';
1374 }

1376 bool Arguments::process_settings_file(const char* file_name, bool should_exist, jboolean ignore_unrecognized) {
1377   FILE* stream = fopen(file_name, "rb");
1378   if (stream == NULL) {
1379     if (should_exist) {
1380       jio_fprintf(defaultStream::error_stream(),
1381                   "Could not open settings file %s\n", file_name);
1382       return false;
1383     } else {
1384       return true;
1385     }
1386   }
1387 
1388   char token[1024];
1389   int  pos = 0;
1390 
1391   bool in_white_space = true;
1392   bool in_comment     = false;
1393   bool in_quote       = false;
1394   char quote_c        = 0;
1395   bool result         = true;
1396 
1397   int c = getc(stream);
1398   while(c != EOF && pos < (int)(sizeof(token)-1)) {
1399     if (in_white_space) {
1400       if (in_comment) {
1401         if (c == '\n') in_comment = false;
1402       } else {
1403         if (c == '#') in_comment = true;
1404         else if (!isspace(c)) {
1405           in_white_space = false;
1406           token[pos++] = c;
1407         }
1408       }
1409     } else {
1410       if (c == '\n' || (!in_quote && isspace(c))) {
1411         // token ends at newline, or at unquoted whitespace
1412         // this allows a way to include spaces in string-valued options
1413         token[pos] = '\0';
1414         logOption(token);
1415         result &= process_argument(token, ignore_unrecognized, JVMFlag::CONFIG_FILE);
1416         build_jvm_flags(token);
1417         pos = 0;
1418         in_white_space = true;
1419         in_quote = false;
1420       } else if (!in_quote && (c == '\'' || c == '"')) {
1421         in_quote = true;
1422         quote_c = c;
1423       } else if (in_quote && (c == quote_c)) {
1424         in_quote = false;
1425       } else {
1426         token[pos++] = c;
1427       }
1428     }
1429     c = getc(stream);
1430   }
1431   if (pos > 0) {
1432     token[pos] = '\0';
1433     result &= process_argument(token, ignore_unrecognized, JVMFlag::CONFIG_FILE);
1434     build_jvm_flags(token);
1435   }
1436   fclose(stream);
1437   return result;
1438 }

1447 bool Arguments::add_property(const char* prop, PropertyWriteable writeable, PropertyInternal internal) {
1448   const char* eq = strchr(prop, '=');
1449   const char* key;
1450   const char* value = "";
1451 
1452   if (eq == NULL) {
1453     // property doesn't have a value, thus use passed string
1454     key = prop;
1455   } else {
1456     // property have a value, thus extract it and save to the
1457     // allocated string
1458     size_t key_len = eq - prop;
1459     char* tmp_key = AllocateHeap(key_len + 1, mtArguments);
1460 
1461     jio_snprintf(tmp_key, key_len + 1, "%s", prop);
1462     key = tmp_key;
1463 
1464     value = &prop[key_len + 1];
1465   }
1466 
1467   if (strcmp(key, "java.compiler") == 0) {
1468     process_java_compiler_argument(value);
1469     // Record value in Arguments, but let it get passed to Java.
1470   } else if (strcmp(key, "sun.java.launcher.is_altjvm") == 0) {
1471     // sun.java.launcher.is_altjvm property is
1472     // private and is processed in process_sun_java_launcher_properties();
1473     // the sun.java.launcher property is passed on to the java application
1474   } else if (strcmp(key, "sun.boot.library.path") == 0) {
1475     // append is true, writable is true, internal is false
1476     PropertyList_unique_add(&_system_properties, key, value, AppendProperty,
1477                             WriteableProperty, ExternalProperty);
1478   } else {
1479     if (strcmp(key, "sun.java.command") == 0) {
1480       char *old_java_command = _java_command;
1481       _java_command = os::strdup_check_oom(value, mtArguments);
1482       if (old_java_command != NULL) {
1483         os::free(old_java_command);
1484       }
1485     } else if (strcmp(key, "java.vendor.url.bug") == 0) {
1486       // If this property is set on the command line then its value will be
1487       // displayed in VM error logs as the URL at which to submit such logs.
1488       // Normally the URL displayed in error logs is different from the value
1489       // of this system property, so a different property should have been
1490       // used here, but we leave this as-is in case someone depends upon it.
1491       const char* old_java_vendor_url_bug = _java_vendor_url_bug;
1492       // save it in _java_vendor_url_bug, so JVM fatal error handler can access
1493       // its value without going through the property list or making a Java call.
1494       _java_vendor_url_bug = os::strdup_check_oom(value, mtArguments);
1495       if (old_java_vendor_url_bug != NULL) {
1496         os::free((void *)old_java_vendor_url_bug);
1497       }
1498     }
1499 
1500     // Create new property and add at the end of the list
1501     PropertyList_unique_add(&_system_properties, key, value, AddProperty, writeable, internal);
1502   }
1503 
1504   if (key != prop) {
1505     // SystemProperty copy passed value, thus free previously allocated
1506     // memory
1507     FreeHeap((void *)key);
1508   }
1509 
1510   return true;
1511 }

1661 size_t Arguments::max_heap_for_compressed_oops() {
1662   // Avoid sign flip.
1663   assert(OopEncodingHeapMax > (uint64_t)os::vm_page_size(), "Unusual page size");
1664   // We need to fit both the NULL page and the heap into the memory budget, while
1665   // keeping alignment constraints of the heap. To guarantee the latter, as the
1666   // NULL page is located before the heap, we pad the NULL page to the conservative
1667   // maximum alignment that the GC may ever impose upon the heap.
1668   size_t displacement_due_to_null_page = align_up((size_t)os::vm_page_size(),
1669                                                   _conservative_max_heap_alignment);
1670 
1671   LP64_ONLY(return OopEncodingHeapMax - displacement_due_to_null_page);
1672   NOT_LP64(ShouldNotReachHere(); return 0);
1673 }

1776 // Use static initialization to get the default before parsing
1777 static const size_t DefaultHeapBaseMinAddress = HeapBaseMinAddress;
1778 
1779 void Arguments::set_heap_size() {
1780   julong phys_mem;
1781 
1782   // If the user specified one of these options, they
1783   // want specific memory sizing so do not limit memory
1784   // based on compressed oops addressability.
1785   // Also, memory limits will be calculated based on
1786   // available os physical memory, not our MaxRAM limit,
1787   // unless MaxRAM is also specified.
1788   bool override_coop_limit = (!FLAG_IS_DEFAULT(MaxRAMPercentage) ||
1789                            !FLAG_IS_DEFAULT(MaxRAMFraction) ||
1790                            !FLAG_IS_DEFAULT(MinRAMPercentage) ||
1791                            !FLAG_IS_DEFAULT(MinRAMFraction) ||
1792                            !FLAG_IS_DEFAULT(InitialRAMPercentage) ||
1793                            !FLAG_IS_DEFAULT(InitialRAMFraction) ||
1794                            !FLAG_IS_DEFAULT(MaxRAM));
1795   if (override_coop_limit) {
1796     if (FLAG_IS_DEFAULT(MaxRAM)) {
1797       phys_mem = os::physical_memory();
1798       FLAG_SET_ERGO(MaxRAM, (uint64_t)phys_mem);
1799     } else {
1800       phys_mem = (julong)MaxRAM;
1801     }
1802   } else {
1803     phys_mem = FLAG_IS_DEFAULT(MaxRAM) ? MIN2(os::physical_memory(), (julong)MaxRAM)
1804                                        : (julong)MaxRAM;
1805   }
1806 
1807 
1808   // Convert deprecated flags
1809   if (FLAG_IS_DEFAULT(MaxRAMPercentage) &&
1810       !FLAG_IS_DEFAULT(MaxRAMFraction))
1811     MaxRAMPercentage = 100.0 / MaxRAMFraction;
1812 
1813   if (FLAG_IS_DEFAULT(MinRAMPercentage) &&
1814       !FLAG_IS_DEFAULT(MinRAMFraction))
1815     MinRAMPercentage = 100.0 / MinRAMFraction;
1816 
1817   if (FLAG_IS_DEFAULT(InitialRAMPercentage) &&
1818       !FLAG_IS_DEFAULT(InitialRAMFraction))
1819     InitialRAMPercentage = 100.0 / InitialRAMFraction;
1820 
1821   // If the maximum heap size has not been set with -Xmx,
1822   // then set it as fraction of the size of physical memory,
1823   // respecting the maximum and minimum sizes of the heap.
1824   if (FLAG_IS_DEFAULT(MaxHeapSize)) {
1825     julong reasonable_max = (julong)((phys_mem * MaxRAMPercentage) / 100);
1826     const julong reasonable_min = (julong)((phys_mem * MinRAMPercentage) / 100);
1827     if (reasonable_min < MaxHeapSize) {
1828       // Small physical memory, so use a minimum fraction of it for the heap
1829       reasonable_max = reasonable_min;
1830     } else {
1831       // Not-small physical memory, so require a heap at least
1832       // as large as MaxHeapSize
1833       reasonable_max = MAX2(reasonable_max, (julong)MaxHeapSize);
1834     }
1835 
1836     if (!FLAG_IS_DEFAULT(ErgoHeapSizeLimit) && ErgoHeapSizeLimit != 0) {
1837       // Limit the heap size to ErgoHeapSizeLimit
1838       reasonable_max = MIN2(reasonable_max, (julong)ErgoHeapSizeLimit);
1839     }
1840 
1841 #ifdef _LP64
1842     if (UseCompressedOops || UseCompressedClassPointers) {
1843       // HeapBaseMinAddress can be greater than default but not less than.
1844       if (!FLAG_IS_DEFAULT(HeapBaseMinAddress)) {
1845         if (HeapBaseMinAddress < DefaultHeapBaseMinAddress) {
1846           // matches compressed oops printing flags
1847           log_debug(gc, heap, coops)("HeapBaseMinAddress must be at least " SIZE_FORMAT
1848                                      " (" SIZE_FORMAT "G) which is greater than value given " SIZE_FORMAT,
1849                                      DefaultHeapBaseMinAddress,
1850                                      DefaultHeapBaseMinAddress/G,
1851                                      HeapBaseMinAddress);
1852           FLAG_SET_ERGO(HeapBaseMinAddress, DefaultHeapBaseMinAddress);
1853         }
1854       }
1855     }
1856     if (UseCompressedOops) {
1857       // Limit the heap size to the maximum possible when using compressed oops
1858       julong max_coop_heap = (julong)max_heap_for_compressed_oops();
1859 
1860       if (HeapBaseMinAddress + MaxHeapSize < max_coop_heap) {
1861         // Heap should be above HeapBaseMinAddress to get zero based compressed oops
1862         // but it should be not less than default MaxHeapSize.
1863         max_coop_heap -= HeapBaseMinAddress;
1864       }
1865 
1866       // If user specified flags prioritizing os physical
1867       // memory limits, then disable compressed oops if
1868       // limits exceed max_coop_heap and UseCompressedOops
1869       // was not specified.
1870       if (reasonable_max > max_coop_heap) {
1871         if (FLAG_IS_ERGO(UseCompressedOops) && override_coop_limit) {
1872           log_info(cds)("UseCompressedOops and UseCompressedClassPointers have been disabled due to"
1873             " max heap " SIZE_FORMAT " > compressed oop heap " SIZE_FORMAT ". "
1874             "Please check the setting of MaxRAMPercentage %5.2f."
1875             ,(size_t)reasonable_max, (size_t)max_coop_heap, MaxRAMPercentage);
1876           FLAG_SET_ERGO(UseCompressedOops, false);
1877           if (COMPRESSED_CLASS_POINTERS_DEPENDS_ON_COMPRESSED_OOPS) {
1878             FLAG_SET_ERGO(UseCompressedClassPointers, false);
1879           }
1880         } else {
1881           reasonable_max = MIN2(reasonable_max, max_coop_heap);
1882         }
1883       }
1884     }
1885 #endif // _LP64
1886 
1887     reasonable_max = limit_by_allocatable_memory(reasonable_max);
1888 
1889     if (!FLAG_IS_DEFAULT(InitialHeapSize)) {
1890       // An initial heap size was specified on the command line,
1891       // so be sure that the maximum size is consistent.  Done
1892       // after call to limit_by_allocatable_memory because that
1893       // method might reduce the allocation size.
1894       reasonable_max = MAX2(reasonable_max, (julong)InitialHeapSize);
1895     } else if (!FLAG_IS_DEFAULT(MinHeapSize)) {
1896       reasonable_max = MAX2(reasonable_max, (julong)MinHeapSize);
1897     }
1898 
1899     log_trace(gc, heap)("  Maximum heap size " SIZE_FORMAT, (size_t) reasonable_max);
1900     FLAG_SET_ERGO(MaxHeapSize, (size_t)reasonable_max);
1901   }
1902 
1903   // If the minimum or initial heap_size have not been set or requested to be set
1904   // ergonomically, set them accordingly.
1905   if (InitialHeapSize == 0 || MinHeapSize == 0) {
1906     julong reasonable_minimum = (julong)(OldSize + NewSize);
1907 
1908     reasonable_minimum = MIN2(reasonable_minimum, (julong)MaxHeapSize);
1909 
1910     reasonable_minimum = limit_by_allocatable_memory(reasonable_minimum);
1911 
1912     if (InitialHeapSize == 0) {
1913       julong reasonable_initial = (julong)((phys_mem * InitialRAMPercentage) / 100);
1914 
1915       reasonable_initial = MAX3(reasonable_initial, reasonable_minimum, (julong)MinHeapSize);
1916       reasonable_initial = MIN2(reasonable_initial, (julong)MaxHeapSize);
1917 
1918       reasonable_initial = limit_by_allocatable_memory(reasonable_initial);
1919 
1920       FLAG_SET_ERGO(InitialHeapSize, (size_t)reasonable_initial);
1921       log_trace(gc, heap)("  Initial heap size " SIZE_FORMAT, InitialHeapSize);
1922     }
1923     // If the minimum heap size has not been set (via -Xms or -XX:MinHeapSize),
1924     // synchronize with InitialHeapSize to avoid errors with the default value.
1925     if (MinHeapSize == 0) {
1926       FLAG_SET_ERGO(MinHeapSize, MIN2((size_t)reasonable_minimum, InitialHeapSize));
1927       log_trace(gc, heap)("  Minimum heap size " SIZE_FORMAT, MinHeapSize);
1928     }
1929   }
1930 }

1932 // This option inspects the machine and attempts to set various
1933 // parameters to be optimal for long-running, memory allocation
1934 // intensive jobs.  It is intended for machines with large
1935 // amounts of cpu and memory.
1936 jint Arguments::set_aggressive_heap_flags() {
1937   // initHeapSize is needed since _initial_heap_size is 4 bytes on a 32 bit
1938   // VM, but we may not be able to represent the total physical memory
1939   // available (like having 8gb of memory on a box but using a 32bit VM).
1940   // Thus, we need to make sure we're using a julong for intermediate
1941   // calculations.
1942   julong initHeapSize;
1943   julong total_memory = os::physical_memory();
1944 
1945   if (total_memory < (julong) 256 * M) {
1946     jio_fprintf(defaultStream::error_stream(),
1947             "You need at least 256mb of memory to use -XX:+AggressiveHeap\n");
1948     vm_exit(1);
1949   }
1950 
1951   // The heap size is half of available memory, or (at most)
1952   // all of possible memory less 160mb (leaving room for the OS
1953   // when using ISM).  This is the maximum; because adaptive sizing
1954   // is turned on below, the actual space used may be smaller.
1955 
1956   initHeapSize = MIN2(total_memory / (julong) 2,
1957           total_memory - (julong) 160 * M);
1958 
1959   initHeapSize = limit_by_allocatable_memory(initHeapSize);
1960 
1961   if (FLAG_IS_DEFAULT(MaxHeapSize)) {
1962     if (FLAG_SET_CMDLINE(MaxHeapSize, initHeapSize) != JVMFlag::SUCCESS) {
1963       return JNI_EINVAL;
1964     }
1965     if (FLAG_SET_CMDLINE(InitialHeapSize, initHeapSize) != JVMFlag::SUCCESS) {
1966       return JNI_EINVAL;
1967     }
1968     if (FLAG_SET_CMDLINE(MinHeapSize, initHeapSize) != JVMFlag::SUCCESS) {
1969       return JNI_EINVAL;
1970     }
1971   }
1972   if (FLAG_IS_DEFAULT(NewSize)) {
1973     // Make the young generation 3/8ths of the total heap.
1974     if (FLAG_SET_CMDLINE(NewSize,
1975             ((julong) MaxHeapSize / (julong) 8) * (julong) 3) != JVMFlag::SUCCESS) {
1976       return JNI_EINVAL;
1977     }
1978     if (FLAG_SET_CMDLINE(MaxNewSize, NewSize) != JVMFlag::SUCCESS) {
1979       return JNI_EINVAL;
1980     }
1981   }
1982 
1983 #if !defined(_ALLBSD_SOURCE) && !defined(AIX)  // UseLargePages is not yet supported on BSD and AIX.
1984   FLAG_SET_DEFAULT(UseLargePages, true);
1985 #endif
1986 
1987   // Increase some data structure sizes for efficiency
1988   if (FLAG_SET_CMDLINE(BaseFootPrintEstimate, MaxHeapSize) != JVMFlag::SUCCESS) {
1989     return JNI_EINVAL;
1990   }
1991   if (FLAG_SET_CMDLINE(ResizeTLAB, false) != JVMFlag::SUCCESS) {
1992     return JNI_EINVAL;
1993   }
1994   if (FLAG_SET_CMDLINE(TLABSize, 256 * K) != JVMFlag::SUCCESS) {
1995     return JNI_EINVAL;
1996   }
1997 
1998   // See the OldPLABSize comment below, but replace 'after promotion'
1999   // with 'after copying'.  YoungPLABSize is the size of the survivor
2000   // space per-gc-thread buffers.  The default is 4kw.
2001   if (FLAG_SET_CMDLINE(YoungPLABSize, 256 * K) != JVMFlag::SUCCESS) { // Note: this is in words
2002     return JNI_EINVAL;
2003   }
2004 
2005   // OldPLABSize is the size of the buffers in the old gen that
2006   // UseParallelGC uses to promote live data that doesn't fit in the
2007   // survivor spaces.  At any given time, there's one for each gc thread.
2008   // The default size is 1kw. These buffers are rarely used, since the
2009   // survivor spaces are usually big enough.  For specjbb, however, there
2010   // are occasions when there's lots of live data in the young gen
2011   // and we end up promoting some of it.  We don't have a definite
2012   // explanation for why bumping OldPLABSize helps, but the theory
2013   // is that a bigger PLAB results in retaining something like the
2014   // original allocation order after promotion, which improves mutator
2015   // locality.  A minor effect may be that larger PLABs reduce the
2016   // number of PLAB allocation events during gc.  The value of 8kw
2017   // was arrived at by experimenting with specjbb.
2018   if (FLAG_SET_CMDLINE(OldPLABSize, 8 * K) != JVMFlag::SUCCESS) { // Note: this is in words
2019     return JNI_EINVAL;
2020   }
2021 
2022   // Enable parallel GC and adaptive generation sizing
2023   if (FLAG_SET_CMDLINE(UseParallelGC, true) != JVMFlag::SUCCESS) {
2024     return JNI_EINVAL;
2025   }
2026 
2027   // Encourage steady state memory management
2028   if (FLAG_SET_CMDLINE(ThresholdTolerance, 100) != JVMFlag::SUCCESS) {
2029     return JNI_EINVAL;
2030   }
2031 
2032   // This appears to improve mutator locality
2033   if (FLAG_SET_CMDLINE(ScavengeBeforeFullGC, false) != JVMFlag::SUCCESS) {
2034     return JNI_EINVAL;
2035   }
2036 
2037   return JNI_OK;
2038 }

2395 // Parse -Xss memory string parameter and convert to ThreadStackSize in K.
2396 jint Arguments::parse_xss(const JavaVMOption* option, const char* tail, intx* out_ThreadStackSize) {
2397   // The min and max sizes match the values in globals.hpp, but scaled
2398   // with K. The values have been chosen so that alignment with page
2399   // size doesn't change the max value, which makes the conversions
2400   // back and forth between Xss value and ThreadStackSize value easier.
2401   // The values have also been chosen to fit inside a 32-bit signed type.
2402   const julong min_ThreadStackSize = 0;
2403   const julong max_ThreadStackSize = 1 * M;
2404 
2405   const julong min_size = min_ThreadStackSize * K;
2406   const julong max_size = max_ThreadStackSize * K;
2407 
2408   assert(is_aligned(max_size, os::vm_page_size()), "Implementation assumption");
2409 
2410   julong size = 0;
2411   ArgsRange errcode = parse_memory_size(tail, &size, min_size, max_size);
2412   if (errcode != arg_in_range) {
2413     bool silent = (option == NULL); // Allow testing to silence error messages
2414     if (!silent) {
2415       jio_fprintf(defaultStream::error_stream(),
2416                   "Invalid thread stack size: %s\n", option->optionString);
2417       describe_range_error(errcode);
2418     }
2419     return JNI_EINVAL;
2420   }
2421 
2422   // Internally track ThreadStackSize in units of 1024 bytes.
2423   const julong size_aligned = align_up(size, K);
2424   assert(size <= size_aligned,
2425          "Overflow: " JULONG_FORMAT " " JULONG_FORMAT,
2426          size, size_aligned);
2427 
2428   const julong size_in_K = size_aligned / K;
2429   assert(size_in_K < (julong)max_intx,
2430          "size_in_K doesn't fit in the type of ThreadStackSize: " JULONG_FORMAT,
2431          size_in_K);
2432 
2433   // Check that code expanding ThreadStackSize to a page aligned number of bytes won't overflow.
2434   const julong max_expanded = align_up(size_in_K * K, os::vm_page_size());
2435   assert(max_expanded < max_uintx && max_expanded >= size_in_K,
2436          "Expansion overflowed: " JULONG_FORMAT " " JULONG_FORMAT,
2437          max_expanded, size_in_K);
2438 
2439   *out_ThreadStackSize = (intx)size_in_K;
2440 
2441   return JNI_OK;

3106 // Remove all empty paths from the app classpath (if IgnoreEmptyClassPaths is enabled)
3107 //
3108 // This is necessary because some apps like to specify classpath like -cp foo.jar:${XYZ}:bar.jar
3109 // in their start-up scripts. If XYZ is empty, the classpath will look like "-cp foo.jar::bar.jar".
3110 // Java treats such empty paths as if the user specified "-cp foo.jar:.:bar.jar". I.e., an empty
3111 // path is treated as the current directory.
3112 //
3113 // This causes problems with CDS, which requires that all directories specified in the classpath
3114 // must be empty. In most cases, applications do NOT want to load classes from the current
3115 // directory anyway. Adding -XX:+IgnoreEmptyClassPaths will make these applications' start-up
3116 // scripts compatible with CDS.
3117 void Arguments::fix_appclasspath() {
3118   if (IgnoreEmptyClassPaths) {
3119     const char separator = *os::path_separator();
3120     const char* src = _java_class_path->value();
3121 
3122     // skip over all the leading empty paths
3123     while (*src == separator) {
3124       src ++;
3125     }
3126 
3127     char* copy = os::strdup_check_oom(src, mtArguments);
3128 
3129     // trim all trailing empty paths
3130     for (char* tail = copy + strlen(copy) - 1; tail >= copy && *tail == separator; tail--) {
3131       *tail = '\0';
3132     }
3133 
3134     char from[3] = {separator, separator, '\0'};
3135     char to  [2] = {separator, '\0'};
3136     while (StringUtils::replace_no_expand(copy, from, to) > 0) {
3137       // Keep replacing "::" -> ":" until we have no more "::" (non-windows)
3138       // Keep replacing ";;" -> ";" until we have no more ";;" (windows)
3139     }
3140 
3141     _java_class_path->set_writeable_value(copy);
3142     FreeHeap(copy); // a copy was made by set_value, so don't need this anymore
3143   }

3467 jint Arguments::parse_options_buffer(const char* name, char* buffer, const size_t buf_len, ScopedVMInitArgs* vm_args) {
3468   GrowableArray<JavaVMOption> *options = new (ResourceObj::C_HEAP, mtArguments) GrowableArray<JavaVMOption>(2, true);    // Construct option array
3469 
3470   // some pointers to help with parsing
3471   char *buffer_end = buffer + buf_len;
3472   char *opt_hd = buffer;
3473   char *wrt = buffer;
3474   char *rd = buffer;
3475 
3476   // parse all options
3477   while (rd < buffer_end) {
3478     // skip leading white space from the input string
3479     while (rd < buffer_end && isspace(*rd)) {
3480       rd++;
3481     }
3482 
3483     if (rd >= buffer_end) {
3484       break;
3485     }
3486 
3487     // Remember this is where we found the head of the token.
3488     opt_hd = wrt;
3489 
3490     // Tokens are strings of non white space characters separated
3491     // by one or more white spaces.
3492     while (rd < buffer_end && !isspace(*rd)) {
3493       if (*rd == '\'' || *rd == '"') {      // handle a quoted string
3494         int quote = *rd;                    // matching quote to look for
3495         rd++;                               // don't copy open quote
3496         while (rd < buffer_end && *rd != quote) {
3497                                             // include everything (even spaces)
3498                                             // up until the close quote
3499           *wrt++ = *rd++;                   // copy to option string
3500         }
3501 
3502         if (rd < buffer_end) {
3503           rd++;                             // don't copy close quote
3504         } else {
3505                                             // did not see closing quote
3506           jio_fprintf(defaultStream::error_stream(),
3507                       "Unmatched quote in %s\n", name);
3508           delete options;
3509           return JNI_ERR;
3510         }
3511       } else {
3512         *wrt++ = *rd++;                     // copy to option string
3513       }
3514     }
3515 
3516     // steal a white space character and set it to NULL
3517     *wrt++ = '\0';
3518     // We now have a complete token
3519 
3520     JavaVMOption option;
3521     option.optionString = opt_hd;
3522     option.extraInfo = NULL;
3523 
3524     options->append(option);                // Fill in option
3525 
3526     rd++;  // Advance to next character
3527   }
3528 
3529   // Fill out JavaVMInitArgs structure.
3530   jint status = vm_args->set_args(options);
3531 
3532   delete options;
3533   return status;
3534 }

3765   for (int index = 0; index < args->nOptions; index++) {
3766     const JavaVMOption* option = args->options + index;

3768       Arguments::set_jvm_flags_file(tail);
3769       continue;
3770     }
3771     if (match_option(option, "-XX:VMOptionsFile=", &tail)) {
3772       if (vm_options_file_args.found_vm_options_file_arg()) {

3796         // current option.  So we back track to process the

3798         index--;

3813     }

3819       JVMFlag::printFlags(tty, false);
3820       vm_exit(0);
3821     }
3822     if (match_option(option, "-XX:NativeMemoryTracking", &tail)) {
3823 #if INCLUDE_NMT
3824       // The launcher did not setup nmt environment variable properly.
3825       if (!MemTracker::check_launcher_nmt_support(tail)) {
3826         warning("Native Memory Tracking did not setup properly, using wrong launcher?");
3827       }

4332 // Copies src into buf, replacing "%%" with "%" and "%p" with pid
4333 // Returns true if all of the source pointed by src has been copied over to
4334 // the destination buffer pointed by buf. Otherwise, returns false.
4335 // Notes:
4336 // 1. If the length (buflen) of the destination buffer excluding the
4337 // NULL terminator character is not long enough for holding the expanded
4338 // pid characters, it also returns false instead of returning the partially
4339 // expanded one.
4340 // 2. The passed in "buflen" should be large enough to hold the null terminator.
4341 bool Arguments::copy_expand_pid(const char* src, size_t srclen,
4342                                 char* buf, size_t buflen) {
4343   const char* p = src;
4344   char* b = buf;
4345   const char* src_end = &src[srclen];
4346   char* buf_end = &buf[buflen - 1];
4347 
4348   while (p < src_end && b < buf_end) {
4349     if (*p == '%') {
4350       switch (*(++p)) {
4351       case '%':         // "%%" ==> "%"
4352         *b++ = *p++;
4353         break;
4354       case 'p':  {       //  "%p" ==> current process id
4355         // buf_end points to the character before the last character so
4356         // that we could write '\0' to the end of the buffer.
4357         size_t buf_sz = buf_end - b + 1;
4358         int ret = jio_snprintf(b, buf_sz, "%d", os::current_process_id());
4359 
4360         // if jio_snprintf fails or the buffer is not long enough to hold
4361         // the expanded pid, returns false.
4362         if (ret < 0 || ret >= (int)buf_sz) {
4363           return false;
4364         } else {
4365           b += ret;
4366           assert(*b == '\0', "fail in copy_expand_pid");
4367           if (p == src_end && b == buf_end + 1) {
4368             // reach the end of the buffer.
4369             return true;
4370           }
4371         }
4372         p++;
4373         break;
4374       }
4375       default :
4376         *b++ = '%';
4377       }
4378     } else {
4379       *b++ = *p++;
4380     }
4381   }
4382   *b = '\0';
4383   return (p == src_end); // return false if not all of the source was copied
4384 }
