 278 void initialize_basic_type_klass(Klass* k, TRAPS) {
 279   Klass* ok = SystemDictionary::Object_klass();
 280 #if INCLUDE_CDS
 281   if (UseSharedSpaces) {
 282     ClassLoaderData* loader_data = ClassLoaderData::the_null_class_loader_data();
 283     assert(k->super() == ok, "u3");
 284     if (k->is_instance_klass()) {
 285       InstanceKlass::cast(k)->restore_unshareable_info(loader_data, Handle(), NULL, CHECK);
 286     } else {
 287       ArrayKlass::cast(k)->restore_unshareable_info(loader_data, Handle(), CHECK);
 288     }
 289   } else
 290 #endif
 291   {
 292     k->initialize_supers(ok, NULL, CHECK);
 293   }
 294   k->append_to_sibling_list();
 295 }

 297 void Universe::genesis(TRAPS) {
 298   ResourceMark rm(THREAD);
 299 
 300   { FlagSetting fs(_bootstrapping, true);
 301 
 302     { MutexLocker mc(THREAD, Compile_lock);
 303 
 304       java_lang_Class::allocate_fixup_lists();
 305 
 306       // determine base vtable size; without that we cannot create the array klasses
 307       compute_base_vtable_size();
 308 
 309       if (!UseSharedSpaces) {
 310         for (int i = T_BOOLEAN; i < T_LONG+1; i++) {
 311           _typeArrayKlassObjs[i] = TypeArrayKlass::create_klass((BasicType)i, CHECK);
 312         }
 313 
 314         ClassLoaderData* null_cld = ClassLoaderData::the_null_class_loader_data();
 315 
 316         _the_array_interfaces_array     = MetadataFactory::new_array<Klass*>(null_cld, 2, NULL, CHECK);
 317         _the_empty_int_array            = MetadataFactory::new_array<int>(null_cld, 0, CHECK);
 318         _the_empty_short_array          = MetadataFactory::new_array<u2>(null_cld, 0, CHECK);
 319         _the_empty_method_array         = MetadataFactory::new_array<Method*>(null_cld, 0, CHECK);
 320         _the_empty_klass_array          = MetadataFactory::new_array<Klass*>(null_cld, 0, CHECK);
 321         _the_empty_instance_klass_array = MetadataFactory::new_array<InstanceKlass*>(null_cld, 0, CHECK);
 322       }
 323     }
 324 
 325     vmSymbols::initialize(CHECK);
 326 
 327     SystemDictionary::initialize(CHECK);
 328 
 329     Klass* ok = SystemDictionary::Object_klass();
 330 
 331     _the_null_string            = StringTable::intern("null", CHECK);
 332     _the_min_jint_string       = StringTable::intern("-2147483648", CHECK);
 333 
 334 #if INCLUDE_CDS
 335     if (UseSharedSpaces) {
 336       // Verify shared interfaces array.
 337       assert(_the_array_interfaces_array->at(0) ==
 338              SystemDictionary::Cloneable_klass(), "u3");
 339       assert(_the_array_interfaces_array->at(1) ==
 340              SystemDictionary::Serializable_klass(), "u3");
 341     } else
 342 #endif
 343     {
 344       // Set up shared interfaces array.  (Do this before supers are set up.)
 345       _the_array_interfaces_array->at_put(0, SystemDictionary::Cloneable_klass());
 346       _the_array_interfaces_array->at_put(1, SystemDictionary::Serializable_klass());
 347     }
 348 
 349     initialize_basic_type_klass(boolArrayKlassObj(), CHECK);
 350     initialize_basic_type_klass(charArrayKlassObj(), CHECK);
 351     initialize_basic_type_klass(floatArrayKlassObj(), CHECK);
 352     initialize_basic_type_klass(doubleArrayKlassObj(), CHECK);
 353     initialize_basic_type_klass(byteArrayKlassObj(), CHECK);
 354     initialize_basic_type_klass(shortArrayKlassObj(), CHECK);
 355     initialize_basic_type_klass(intArrayKlassObj(), CHECK);
 356     initialize_basic_type_klass(longArrayKlassObj(), CHECK);
 357   } // end of core bootstrapping
 358 
 359   {
 360     Handle tns = java_lang_String::create_from_str("<null_sentinel>", CHECK);
 361     _the_null_sentinel = tns();
 362   }
 363 
 364   // Maybe this could be lifted up now that object array can be initialized
 365   // during the bootstrapping.
 366 
 367   // OLD
 368   // Initialize _objectArrayKlass after core bootstraping to make
 369   // sure the super class is set up properly for _objectArrayKlass.
 370   // ---
 371   // NEW
 372   // Since some of the old system object arrays have been converted to
 373   // ordinary object arrays, _objectArrayKlass will be loaded when
 374   // SystemDictionary::initialize(CHECK); is run. See the extra check
 375   // for Object_klass_loaded in objArrayKlassKlass::allocate_objArray_klass_impl.
 376   _objectArrayKlassObj = InstanceKlass::
 377     cast(SystemDictionary::Object_klass())->array_klass(1, CHECK);
 378   // OLD
 379   // Add the class to the class hierarchy manually to make sure that
 380   // its vtable is initialized after core bootstrapping is completed.
 381   // ---
 382   // New
 383   // Have already been initialized.
 384   _objectArrayKlassObj->append_to_sibling_list();
 385 
 386   #ifdef ASSERT
 387   if (FullGCALot) {
 388     // Allocate an array of dummy objects.
 389     // We'd like these to be at the bottom of the old generation,
 390     // so that when we free one and then collect,
 391     // (almost) the whole heap moves
 392     // and we find out if we actually update all the oops correctly.
 393     // But we can't allocate directly in the old generation,
 394     // so we allocate wherever, and hope that the first collection
 395     // moves these objects to the bottom of the old generation.
 396     // We can allocate directly in the permanent generation, so we do.
 397     int size = FullGCALotDummies * 2;
 398 
 399     objArrayOop    naked_array = oopFactory::new_objArray(SystemDictionary::Object_klass(), size, CHECK);
 400     objArrayHandle dummy_array(THREAD, naked_array);
 401     int i = 0;
 402     while (i < size) {
 403         // Allocate dummy in old generation
 404       oop dummy = SystemDictionary::Object_klass()->allocate_instance(CHECK);
 405       dummy_array->obj_at_put(i++, dummy);
 406     }
 407     {
 408       // Only modify the global variable inside the mutex.
 409       // If we had a race to here, the other dummy_array instances
 410       // and their elements just get dropped on the floor, which is fine.
 411       MutexLocker ml(THREAD, FullGCALot_lock);
 412       if (_fullgc_alot_dummy_array == NULL) {
 413         _fullgc_alot_dummy_array = dummy_array();
 414       }
 415     }
 416     assert(i == _fullgc_alot_dummy_array->length(), "just checking");
 417   }
 418   #endif
 419 }

 468 void Universe::fixup_mirrors(TRAPS) {
 469   // Bootstrap problem: all classes gets a mirror (java.lang.Class instance) assigned eagerly,
 470   // but we cannot do that for classes created before java.lang.Class is loaded. Here we simply
 471   // walk over permanent objects created so far (mostly classes) and fixup their mirrors. Note
 472   // that the number of objects allocated at this point is very small.
 473   assert(SystemDictionary::Class_klass_loaded(), "java.lang.Class should be loaded");
 474   HandleMark hm(THREAD);
 475 
 476   if (!UseSharedSpaces) {
 477     // Cache the start of the static fields
 478     InstanceMirrorKlass::init_offset_of_static_fields();
 479   }
 480 
 481   GrowableArray <Klass*>* list = java_lang_Class::fixup_mirror_list();
 482   int list_length = list->length();
 483   for (int i = 0; i < list_length; i++) {
 484     Klass* k = list->at(i);
 485     assert(k->is_klass(), "List should only hold classes");
 486     EXCEPTION_MARK;
 487     java_lang_Class::fixup_mirror(k, CATCH);
 488   }
 489   delete java_lang_Class::fixup_mirror_list();
 490   java_lang_Class::set_fixup_mirror_list(NULL);
 491 }

 525 void Universe::reinitialize_vtable_of(Klass* ko, TRAPS) {
 526   // init vtable of k and all subclasses

 561 
 562 bool Universe::should_fill_in_stack_trace(Handle throwable) {
 563   // never attempt to fill in the stack trace of preallocated errors that do not have
 564   // backtrace. These errors are kept alive forever and may be "re-used" when all
 565   // preallocated errors with backtrace have been consumed. Also need to avoid
 566   // a potential loop which could happen if an out of memory occurs when attempting
 567   // to allocate the backtrace.
 568   return ((throwable() != Universe::_out_of_memory_error_java_heap) &&
 569           (throwable() != Universe::_out_of_memory_error_metaspace)  &&
 570           (throwable() != Universe::_out_of_memory_error_class_metaspace)  &&
 571           (throwable() != Universe::_out_of_memory_error_array_size) &&
 572           (throwable() != Universe::_out_of_memory_error_gc_overhead_limit) &&
 573           (throwable() != Universe::_out_of_memory_error_realloc_objects) &&
 574           (throwable() != Universe::_out_of_memory_error_retry));
 575 }

 577 
 578 oop Universe::gen_out_of_memory_error(oop default_err) {
 579   // generate an out of memory error:
 580   // - if there is a preallocated error and stack traces are available
 581   //   (j.l.Throwable is initialized), then return the preallocated
 582   //   error with a filled in stack trace, and with the message
 583   //   provided by the default error.
 584   // - otherwise, return the default error, without a stack trace.
 585   int next;
 586   if ((_preallocated_out_of_memory_error_avail_count > 0) &&
 587       SystemDictionary::Throwable_klass()->is_initialized()) {
 588     next = (int)Atomic::add(&_preallocated_out_of_memory_error_avail_count, -1);
 589     assert(next < (int)PreallocatedOutOfMemoryErrorCount, "avail count is corrupt");
 590   } else {
 591     next = -1;
 592   }
 593   if (next < 0) {
 594     // all preallocated errors have been used.
 595     // return default
 596     return default_err;
 597   } else {
 598     Thread* THREAD = Thread::current();
 599     Handle default_err_h(THREAD, default_err);
 600     // get the error object at the slot and set set it to NULL so that the
 601     // array isn't keeping it alive anymore.
 602     Handle exc(THREAD, preallocated_out_of_memory_errors()->obj_at(next));
 603     assert(exc() != NULL, "slot has been used already");
 604     preallocated_out_of_memory_errors()->obj_at_put(next, NULL);
 605 
 606     // use the message from the default error
 607     oop msg = java_lang_Throwable::message(default_err_h());
 608     assert(msg != NULL, "no message");
 609     java_lang_Throwable::set_message(exc(), msg);
 610 
 611     // populate the stack trace and return it.
 612     java_lang_Throwable::fill_in_stack_trace_of_preallocated_backtrace(exc);
 613     return exc();
 614   }
 615 }

 617 intptr_t Universe::_non_oop_bits = 0;
 618 
 619 void* Universe::non_oop_word() {
 620   // Neither the high bits nor the low bits of this value is allowed
 621   // to look like (respectively) the high or low bits of a real oop.
 622   //
 623   // High and low are CPU-specific notions, but low always includes
 624   // the low-order bit.  Since oops are always aligned at least mod 4,
 625   // setting the low-order bit will ensure that the low half of the
 626   // word will never look like that of a real oop.
 627   //
 628   // Using the OS-supplied non-memory-address word (usually 0 or -1)
 629   // will take care of the high bits, however many there are.
 630 
 631   if (_non_oop_bits == 0) {
 632     _non_oop_bits = (intptr_t)os::non_memory_address_word() | 1;
 633   }
 634 
 635   return (void*)_non_oop_bits;
 636 }

 736 ReservedHeapSpace Universe::reserve_heap(size_t heap_size, size_t alignment) {
 737 
 738   assert(alignment <= Arguments::conservative_max_heap_alignment(),
 739          "actual alignment " SIZE_FORMAT " must be within maximum heap alignment " SIZE_FORMAT,
 740          alignment, Arguments::conservative_max_heap_alignment());
 741 
 742   size_t total_reserved = align_up(heap_size, alignment);
 743   assert(!UseCompressedOops || (total_reserved <= (OopEncodingHeapMax - os::vm_page_size())),
 744       "heap size is too big for compressed oops");
 745 
 746   bool use_large_pages = UseLargePages && is_aligned(alignment, os::large_page_size());
 747   assert(!UseLargePages
 748       || UseParallelGC
 749       || use_large_pages, "Wrong alignment to use large pages");
 750 
 751   // Now create the space.
 752   ReservedHeapSpace total_rs(total_reserved, alignment, use_large_pages, AllocateHeapAt);
 753 
 754   if (total_rs.is_reserved()) {
 755     assert((total_reserved == total_rs.size()) && ((uintptr_t)total_rs.base() % alignment == 0),
 756            "must be exactly of required size and alignment");
 757     // We are good.
 758 
 759     if (AllocateHeapAt != NULL) {
 760       log_info(gc,heap)("Successfully allocated Java heap at location %s", AllocateHeapAt);
 761     }
 762 
 763     if (UseCompressedOops) {
 764       CompressedOops::initialize(total_rs);
 765     }
 766 
 767     Universe::calculate_verify_data((HeapWord*)total_rs.base(), (HeapWord*)total_rs.end());
 768 
 769     return total_rs;
 770   }
 771 
 772   vm_exit_during_initialization(
 773     err_msg("Could not reserve enough space for " SIZE_FORMAT "KB object heap",
 774             total_reserved/K));
 775 
 776   // satisfy compiler
 777   ShouldNotReachHere();
 778   return ReservedHeapSpace(0, 0, false);
 779 }

 862   Universe::_the_empty_class_klass_array = oopFactory::new_objArray(SystemDictionary::Class_klass(), 0, CHECK_false);
 863

 865   Klass* k = SystemDictionary::OutOfMemoryError_klass();

 867   Universe::_out_of_memory_error_java_heap = ik->allocate_instance(CHECK_false);
 868   Universe::_out_of_memory_error_metaspace = ik->allocate_instance(CHECK_false);

 870   Universe::_out_of_memory_error_array_size = ik->allocate_instance(CHECK_false);
 871   Universe::_out_of_memory_error_gc_overhead_limit =
 872     ik->allocate_instance(CHECK_false);
 873   Universe::_out_of_memory_error_realloc_objects = ik->allocate_instance(CHECK_false);
 874   Universe::_out_of_memory_error_retry = ik->allocate_instance(CHECK_false);
 875 
 876   // Setup preallocated cause message for delayed StackOverflowError
 877   if (StackReservedPages > 0) {
 878     Universe::_delayed_stack_overflow_error_message =
 879       java_lang_String::create_oop_from_str("Delayed StackOverflowError due to ReservedStackAccess annotated method", CHECK_false);
 880   }

 885   Universe::_null_ptr_exception_instance = InstanceKlass::cast(k)->allocate_instance(CHECK_false);

 888   k = SystemDictionary::resolve_or_fail(vmSymbols::java_lang_ArithmeticException(), true, CHECK_false);
 889   Universe::_arithmetic_exception_instance = InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 890   // Virtual Machine Error for when we get into a situation we can't resolve

 893   if (!linked) {
 894      tty->print_cr("Unable to link/verify VirtualMachineError class");
 895      return false; // initialization failed
 896   }
 897   Universe::_virtual_machine_error_instance =
 898     InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 899 
 900   Universe::_vm_exception = InstanceKlass::cast(k)->allocate_instance(CHECK_false);
 901 
 902   Handle msg = java_lang_String::create_from_str("Java heap space", CHECK_false);
 903   java_lang_Throwable::set_message(Universe::_out_of_memory_error_java_heap, msg());
 904 
 905   msg = java_lang_String::create_from_str("Metaspace", CHECK_false);
 906   java_lang_Throwable::set_message(Universe::_out_of_memory_error_metaspace, msg());
 907   msg = java_lang_String::create_from_str("Compressed class space", CHECK_false);
 908   java_lang_Throwable::set_message(Universe::_out_of_memory_error_class_metaspace, msg());
 909 
 910   msg = java_lang_String::create_from_str("Requested array size exceeds VM limit", CHECK_false);
 911   java_lang_Throwable::set_message(Universe::_out_of_memory_error_array_size, msg());
 912 
 913   msg = java_lang_String::create_from_str("GC overhead limit exceeded", CHECK_false);
 914   java_lang_Throwable::set_message(Universe::_out_of_memory_error_gc_overhead_limit, msg());
 915 
 916   msg = java_lang_String::create_from_str("Java heap space: failed reallocation of scalar replaced objects", CHECK_false);
 917   java_lang_Throwable::set_message(Universe::_out_of_memory_error_realloc_objects, msg());
 918 
 919   msg = java_lang_String::create_from_str("Java heap space: failed retryable allocation", CHECK_false);
 920   java_lang_Throwable::set_message(Universe::_out_of_memory_error_retry, msg());
 921 
 922   msg = java_lang_String::create_from_str("/ by zero", CHECK_false);
 923   java_lang_Throwable::set_message(Universe::_arithmetic_exception_instance, msg());
 924 
 925   // Setup the array of errors that have preallocated backtrace
 926   k = Universe::_out_of_memory_error_java_heap->klass();
 927   assert(k->name() == vmSymbols::java_lang_OutOfMemoryError(), "should be out of memory error");
 928   ik = InstanceKlass::cast(k);
 929 
 930   int len = (StackTraceInThrowable) ? (int)PreallocatedOutOfMemoryErrorCount : 0;
 931   Universe::_preallocated_out_of_memory_error_array = oopFactory::new_objArray(ik, len, CHECK_false);
 932   for (int i=0; i<len; i++) {
 933     oop err = ik->allocate_instance(CHECK_false);
 934     Handle err_h = Handle(THREAD, err);
 935     java_lang_Throwable::allocate_backtrace(err_h, CHECK_false);
 936     Universe::preallocated_out_of_memory_errors()->obj_at_put(i, err_h());
 937   }
 938   Universe::_preallocated_out_of_memory_error_avail_count = (jint)len;
 939 
 940   Universe::initialize_known_methods(CHECK_false);
 941 
 942   // This needs to be done before the first scavenge/gc, since
 943   // it's an input to soft ref clearing policy.
 944   {
 945     MutexLocker x(THREAD, Heap_lock);
 946     Universe::update_heap_info_at_gc();
 947   }
 948 
 949   // ("weak") refs processing infrastructure initialization
 950   Universe::heap()->post_initialize();
 951 
 952   MemoryService::add_metaspace_memory_pools();
 953 
 954   MemoryService::set_universe_heap(Universe::heap());
 955 #if INCLUDE_CDS
 956   MetaspaceShared::post_initialize(CHECK_false);
 957 #endif
 958   return true;
 959 }

1000 void Universe::initialize_verify_flags() {
1001   verify_flags = 0;
1002   const char delimiter[] = " ,";
1003 
1004   size_t length = strlen(VerifySubSet);
1005   char* subset_list = NEW_C_HEAP_ARRAY(char, length + 1, mtInternal);
1006   strncpy(subset_list, VerifySubSet, length + 1);
1007   char* save_ptr;
1008 
1009   char* token = strtok_r(subset_list, delimiter, &save_ptr);
1010   while (token != NULL) {
1011     if (strcmp(token, "threads") == 0) {
1012       verify_flags |= Verify_Threads;
1013     } else if (strcmp(token, "heap") == 0) {
1014       verify_flags |= Verify_Heap;
1015     } else if (strcmp(token, "symbol_table") == 0) {
1016       verify_flags |= Verify_SymbolTable;
1017     } else if (strcmp(token, "string_table") == 0) {
1018       verify_flags |= Verify_StringTable;
1019     } else if (strcmp(token, "codecache") == 0) {
1020       verify_flags |= Verify_CodeCache;
1021     } else if (strcmp(token, "dictionary") == 0) {
1022       verify_flags |= Verify_SystemDictionary;
1023     } else if (strcmp(token, "classloader_data_graph") == 0) {
1024       verify_flags |= Verify_ClassLoaderDataGraph;
1025     } else if (strcmp(token, "metaspace") == 0) {
1026       verify_flags |= Verify_MetaspaceUtils;
1027     } else if (strcmp(token, "jni_handles") == 0) {
1028       verify_flags |= Verify_JNIHandles;
1029     } else if (strcmp(token, "codecache_oops") == 0) {
1030       verify_flags |= Verify_CodeCacheOops;
1031     } else if (strcmp(token, "resolved_method_table") == 0) {
1032       verify_flags |= Verify_ResolvedMethodTable;
1033     } else {
1034       vm_exit_during_initialization(err_msg("VerifySubSet: \'%s\' memory sub-system is unknown, please correct it", token));
1035     }
1036     token = strtok_r(NULL, delimiter, &save_ptr);
1037   }
1038   FREE_C_HEAP_ARRAY(char, subset_list);
1039 }

1048 void Universe::verify(VerifyOption option, const char* prefix) {
1049   // The use of _verify_in_progress is a temporary work around for
1050   // 6320749.  Don't bother with a creating a class to set and clear
1051   // it since it is only used in this method and the control flow is
1052   // straight forward.
1053   _verify_in_progress = true;
1054 
1055   COMPILER2_PRESENT(
1056     assert(!DerivedPointerTable::is_active(),
1057          "DPT should not be active during verification "
1058          "(of thread stacks below)");
1059   )
1060 
1061   ResourceMark rm;
1062   HandleMark hm;  // Handles created during verification can be zapped
1063   _verify_count++;
1064 
1065   FormatBuffer<> title("Verifying %s", prefix);
1066   GCTraceTime(Info, gc, verify) tm(title.buffer());
1067   if (should_verify_subset(Verify_Threads)) {
1068     log_debug(gc, verify)("Threads");
1069     Threads::verify();
1070   }
1071   if (should_verify_subset(Verify_Heap)) {
1072     log_debug(gc, verify)("Heap");
1073     heap()->verify(option);
1074   }
1075   if (should_verify_subset(Verify_SymbolTable)) {
1076     log_debug(gc, verify)("SymbolTable");
1077     SymbolTable::verify();
1078   }
1079   if (should_verify_subset(Verify_StringTable)) {
1080     log_debug(gc, verify)("StringTable");
1081     StringTable::verify();
1082   }
1083   if (should_verify_subset(Verify_CodeCache)) {
1084   {

1122 void Universe::calculate_verify_data(HeapWord* low_boundary, HeapWord* high_boundary) {
1123   assert(low_boundary < high_boundary, "bad interval");
1124 
1125   // decide which low-order bits we require to be clear:
1126   size_t alignSize = MinObjAlignmentInBytes;
1127   size_t min_object_size = CollectedHeap::min_fill_size();
1128 
1129   // make an inclusive limit:
1130   uintptr_t max = (uintptr_t)high_boundary - min_object_size*wordSize;
1131   uintptr_t min = (uintptr_t)low_boundary;
1132   assert(min < max, "bad interval");
1133   uintptr_t diff = max ^ min;
1134 
1135   // throw away enough low-order bits to make the diff vanish
1136   uintptr_t mask = (uintptr_t)(-1);
1137   while ((mask & diff) != 0)
1138     mask <<= 1;
1139   uintptr_t bits = (min & mask);
1140   assert(bits == (max & mask), "correct mask");
1141   // check an intermediate value between min and max, just to make sure:
1142   assert(bits == ((min + (max-min)/2) & mask), "correct mask");
1143 
1144   // require address alignment, too:
1145   mask |= (alignSize - 1);
1146 
1147   if (!(_verify_oop_mask == 0 && _verify_oop_bits == (uintptr_t)-1)) {
1148     assert(_verify_oop_mask == mask && _verify_oop_bits == bits, "mask stability");
1149   }
1150   _verify_oop_mask = mask;
1151   _verify_oop_bits = bits;
1152 }
