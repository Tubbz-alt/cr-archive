 50     private final Map<Cursor, Declaration> treeCache = new HashMap<>();

 52     public TreeMaker() {}

 54     TypeMaker typeMaker = new TypeMaker(this);

 56     public void freeze() {
 57         typeMaker.resolveTypeReferences();

 60     private <T extends Declaration> T checkCache(Cursor c, Class<T> clazz, Supplier<Declaration> factory) {
 61         // The supplier function may lead to adding some other type, which will cause CME using computeIfAbsent.
 62         // This implementation relax the constraint a bit only check for same key
 63         Declaration rv;
 64         if (treeCache.containsKey(c)) {
 65             rv = treeCache.get(c);
 66         } else {
 67             rv = factory.get();
 68             if (null != rv && treeCache.put(c, rv) != null) {
 69                 throw new ConcurrentModificationException();
 70             }
 71         }
 72         return clazz.cast(rv);
 73     }

 75     interface ScopedFactoryLayout {
 76         Declaration.Scoped make(Position pos, String name, MemoryLayout layout, Declaration... decls);
 77     }
 78 
 79     interface ScopedFactoryNoLayout {
 80         Declaration.Scoped make(Position pos, String name, Declaration... decls);
 81     }
 82 
 83     interface VarFactoryNoLayout {
 84         Declaration.Variable make(Position pos, String name, Type type);
 85     }

 87     public Declaration createTree(Cursor c) {
 88         switch (Objects.requireNonNull(c).kind()) {
 89             case EnumDecl:
 90                 return createScoped(c, Declaration.Scoped.Kind.ENUM, Declaration::enum_, Declaration::enum_);
 91             case EnumConstantDecl:
 92                 return createEnumConstant(c);
 93             case FieldDecl:
 94                 return createVar(c.isBitField() ?
 95                         Declaration.Variable.Kind.BITFIELD : Declaration.Variable.Kind.FIELD, c, Declaration::field);
 96             case ParamDecl:
 97                 return createVar(Declaration.Variable.Kind.PARAMETER, c, Declaration::parameter);
 98             case FunctionDecl:
 99                 return createFunction(c);
100             case StructDecl:
101                 return createScoped(c, Declaration.Scoped.Kind.STRUCT, Declaration::struct, Declaration::struct);
102             case UnionDecl:
103                 return createScoped(c, Declaration.Scoped.Kind.UNION, Declaration::union, Declaration::union);
104             case TypedefDecl: {
105                 return createTypedef(c);
106             }
107             case VarDecl:
108                 return createVar(Declaration.Variable.Kind.GLOBAL, c, Declaration::globalVariable);
109             default:
110                 return null;
111         }
112     }

114     Position toPos(Cursor cursor) {
115         SourceLocation loc = cursor.getSourceLocation();
116         if (loc == null) {
117             return Position.NO_POSITION;
118         }
119         SourceLocation.Location sloc = loc.getFileLocation();
120         if (sloc == null) {
121             return Position.NO_POSITION;
122         }
123         return new CursorPosition(cursor);
124     }

126     static class CursorPosition implements Position {
127         private final Cursor cursor;
128 
129         CursorPosition(Cursor cursor) {
130             this.cursor = cursor;
131         }
132 
133         @Override
134         public Path path() {
135             return cursor.getSourceLocation().getFileLocation().path();
136         }
137 
138         @Override
139         public int line() {
140             return cursor.getSourceLocation().getFileLocation().line();
141         }
142 
143         @Override
144         public int col() {
145             return cursor.getSourceLocation().getFileLocation().column();
146         }
147 
148         public Cursor cursor() {
149             return cursor;
150         }
151     }

153     public Declaration.Function createFunction(Cursor c) {
154         checkCursor(c, CursorKind.FunctionDecl);
155         List<Declaration.Variable> params = new ArrayList<>();
156         for (int i = 0 ; i < c.numberOfArgs() ; i++) {
157             params.add((Declaration.Variable)createTree(c.getArgument(i)));
158         }
159         return checkCache(c, Declaration.Function.class,
160                 ()->Declaration.function(toPos(c), c.spelling(), (Type.Function)toType(c), params.toArray(new Declaration.Variable[0])));
161     }
162 
163     public Declaration.Constant createMacro(Cursor c, Optional<MacroParserImpl.Macro> macro) {
164         checkCursorAny(c, CursorKind.MacroDefinition);
165         if (macro.isEmpty()) {
166             return null;
167         } else {
168             MacroParserImpl.Macro parsedMacro = macro.get();
169             return checkCache(c, Declaration.Constant.class,
170                     ()->Declaration.constant(toPos(c), c.spelling(), parsedMacro.value, parsedMacro.type()));
171         }
172     }
173 
174     public Declaration.Constant createEnumConstant(Cursor c) {
175         return checkCache(c, Declaration.Constant.class,
176                 ()->Declaration.constant(toPos(c), c.spelling(), c.getEnumConstantValue(), typeMaker.makeType(c.type())));
177     }
178 
179     public Declaration.Scoped createHeader(Cursor c, List<Declaration> decls) {
180         return checkCache(c, Declaration.Scoped.class,
181                 ()->Declaration.toplevel(toPos(c), filterNestedDeclarations(decls).toArray(new Declaration[0])));
182     }
183 
184     public Declaration.Scoped createScoped(Cursor c, Declaration.Scoped.Kind scopeKind, ScopedFactoryLayout factoryLayout, ScopedFactoryNoLayout factoryNoLayout) {
185         List<Declaration> decls = filterNestedDeclarations(c.children()
186                 .map(this::createTree).collect(Collectors.toList()));
187         return checkCache(c, Declaration.Scoped.class, () -> {
188             if (c.isDefinition()) {
189                 //just a declaration AND definition, we have a layout
190                 MemoryLayout layout = LayoutUtils.getLayout(c.type());
191                 List<Declaration> adaptedDecls = layout instanceof GroupLayout ?
192                         collectBitfields(layout, decls) :
193                         decls;
194                 return factoryLayout.make(toPos(c), c.spelling(), layout, adaptedDecls.toArray(new Declaration[0]));
195             } else {
196                 //just a declaration
197                 if (scopeKind == Declaration.Scoped.Kind.STRUCT ||
198                         scopeKind == Declaration.Scoped.Kind.UNION ||
199                         scopeKind == Declaration.Scoped.Kind.ENUM ||
200                         scopeKind == Declaration.Scoped.Kind.CLASS) {
201                     //if there's a real definition somewhere else, skip this redundant declaration
202                     if (!c.getDefinition().isInvalid()) {
203                         return null;
204                     }
205                 }
206                 return factoryNoLayout.make(toPos(c), c.spelling(), decls.toArray(new Declaration[0]));
207             }
208         });
209     }
210 
211     private List<Declaration> filterNestedDeclarations(List<Declaration> declarations) {
212         return declarations.stream()
213                 .filter(Objects::nonNull)
214                 .filter(d -> !d.name().isEmpty() || ((CursorPosition)d.pos()).cursor.isAnonymousStruct())
215                 .collect(Collectors.toList());
216     }
217 
218     private Declaration.Scoped createTypedef(Cursor c) {
219         Optional<Cursor> decl = c.children().findFirst();
220         if (decl.isPresent() && decl.get().isDefinition() && decl.get().spelling().isEmpty()) {
221             Declaration def = createTree(decl.get());
222             if (def instanceof Declaration.Scoped) {
223                 return checkCache(c, Declaration.Scoped.class,
224                         () -> Declaration.typedef(toPos(c), c.spelling(), def));
225             }
226         }
227         return null;
228     }
229 
230     private Declaration.Variable createVar(Declaration.Variable.Kind kind, Cursor c, VarFactoryNoLayout varFactory) {
231         checkCursorAny(c, CursorKind.VarDecl, CursorKind.FieldDecl, CursorKind.ParamDecl);
232         if (c.isBitField()) {
233             return checkCache(c, Declaration.Variable.class,
234                     () -> Declaration.bitfield(toPos(c), c.spelling(), toType(c),
235                     MemoryLayout.ofValueBits(c.getBitFieldWidth(), ByteOrder.nativeOrder())));
236         } else {
237             return checkCache(c, Declaration.Variable.class,
238                     ()->varFactory.make(toPos(c), c.spelling(), toType(c)));
239         }
240     }

242     private List<Declaration> collectBitfields(MemoryLayout layout, List<Declaration> declarations) {
243         List<Declaration> newDecls = new ArrayList<>();
244         for (MemoryLayout e : ((GroupLayout)layout).memberLayouts()) {
245             Optional<GroupLayout> contents = Utils.getContents(e);
246             if (contents.isPresent()) {
247                 List<Declaration.Variable> bfDecls = new ArrayList<>();
248                 outer: for (MemoryLayout bitfield : contents.get().memberLayouts()) {
249                     if (bitfield.name().isPresent()) {
250                         Iterator<Declaration> declIt = declarations.iterator();
251                         while (declIt.hasNext()) {
252                             Declaration d = declIt.next();
253                             if (d.name().equals(bitfield.name().get())) {
254                                 bfDecls.add((Declaration.Variable)d);
255                                 declIt.remove();
256                                 continue outer;
257                             }
258                         }
259                         throw new IllegalStateException("No matching declaration found for bitfield: " + bitfield);
260                     }
261                 }
262                 newDecls.add(Declaration.bitfields(bfDecls.get(0).pos(), "", contents.get(), bfDecls.toArray(new Declaration.Variable[0])));
263             }
264         }
265         newDecls.addAll(declarations);
266         return newDecls;
267     }
268 
269     private Type toType(Cursor c) {
270         return typeMaker.makeType(c.type());
271     }
272 
273     private void checkCursor(Cursor c, CursorKind k) {
274         if (c.kind() != k) {
275             throw new IllegalArgumentException("Invalid cursor kind");
276         }
277     }
278 
279     private void checkCursorAny(Cursor c, CursorKind... kinds) {
280         CursorKind expected = Objects.requireNonNull(c.kind());
281         for (CursorKind k : kinds) {
282             if (k == expected) {
283                 return;
284             }
285         }
286         throw new IllegalArgumentException("Invalid cursor kind");
287     }
