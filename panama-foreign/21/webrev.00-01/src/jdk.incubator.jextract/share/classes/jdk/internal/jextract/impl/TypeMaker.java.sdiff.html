 58     Type makeType(jdk.internal.clang.Type t) {
 59         Type rv = typeCache.get(t);
 60         if (rv != null) {
 61             return rv;
 62         }
 63         rv = makeTypeInternal(t);
 64         if (null != rv && typeCache.put(t, rv) != null) {
 65             throw new ConcurrentModificationException();
 66         }
 67         return rv;
 68     }

 70     Type makeTypeInternal(jdk.internal.clang.Type t) {
 71         switch(t.kind()) {
 72             case Auto:
 73                 return makeType(t.canonicalType());
 74             case Void:
 75                 return Type.void_();

135             case FunctionProto:
136             case FunctionNoProto: {
137                 List<Type> args = new ArrayList<>();
138                 for (int i = 0; i < t.numberOfArgs(); i++) {
139                     // argument could be function pointer declared locally
140                     args.add(lowerFunctionType(t.argType(i)));
141                 }
142                 return Type.function(t.isVariadic(), lowerFunctionType(t.resultType()), args.toArray(new Type[0]));
143             }
144             case Enum:
145             case Record: {
146                 if (treeMaker == null) {
147                     // Macro evaluation, type is meaningless as this can only be pointer and we only care value
148                     return Type.void_();
149                 }

151             }
152             case BlockPointer:

154                 // TODO: We can always erase type for macro evaluation, should we?
155                 TypeImpl.PointerImpl rv = new TypeImpl.PointerImpl(() -> makeType(t.getPointeeType()));

157                 return rv;
158             }
159             case Typedef: {

161                 return Type.typedef(t.spelling(), __type);

172             case Char16: //unsupported
173             case Half: //unsupported

176             default:
177                 return TypeImpl.ERROR;

 45     private final Map<jdk.internal.clang.Type, Type> typeCache = new HashMap<>();
 46     private List<ClangTypeReference> unresolved = new ArrayList<>();

 48     private class ClangTypeReference implements Supplier<Type> {
 49         jdk.internal.clang.Type origin;
 50         Type derived;
 51 
 52         private ClangTypeReference(jdk.internal.clang.Type origin) {
 53             this.origin = origin;
 54             derived = typeCache.get(origin);
 55         }
 56 
 57         public boolean isUnresolved() {
 58             return null == derived;
 59         }
 60 
 61         public void resolve() {
 62             derived = makeType(origin);
 63             Objects.requireNonNull(derived, "Clang type cannot be resolved: " + origin.spelling());
 64         }
 65 
 66         public Type get() {
 67             Objects.requireNonNull(derived, "Type is not yet resolved.");
 68             return derived;
 69         }
 70     }
 71 
 72     private ClangTypeReference reference(jdk.internal.clang.Type type) {
 73         ClangTypeReference ref = new ClangTypeReference(type);
 74         if (ref.isUnresolved()) {
 75             unresolved.add(ref);
 76         }
 77         return ref;
 78     }
 79 
 80     public TypeMaker(TreeMaker treeMaker) {
 81         this.treeMaker = treeMaker;
 82     }
 83 
 84     /**
 85      * Resolve all type references. This method should be called before discard clang cursors/types
 86      */
 87     void resolveTypeReferences() {
 88         List<ClangTypeReference> resolving = unresolved;
 89         unresolved = new ArrayList<>();
 90         while (! resolving.isEmpty()) {
 91             resolving.forEach(ClangTypeReference::resolve);
 92             resolving = unresolved;
 93             unresolved = new ArrayList<>();
 94         }
 95     }

103         if (null != rv && typeCache.put(t, rv) != null) {
104             throw new ConcurrentModificationException();
105         }
106         return rv;
107     }
