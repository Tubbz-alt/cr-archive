 26 package jdk.internal.foreign;

 28 import jdk.incubator.foreign.Foreign;
 29 import jdk.incubator.foreign.MemoryAddress;
 30 import jdk.incubator.foreign.MemoryLayout;
 31 import jdk.incubator.foreign.MemorySegment;
 32 import jdk.incubator.foreign.SystemABI;
 33 import jdk.internal.foreign.abi.aarch64.AArch64ABI;
 34 import jdk.internal.foreign.abi.x64.sysv.SysVx64ABI;
 35 import jdk.internal.foreign.abi.x64.windows.Windowsx64ABI;
 36 import sun.security.action.GetPropertyAction;
 37 
 38 import java.lang.invoke.VarHandle;
 39 import java.nio.charset.Charset;
 40 
 41 import static jdk.incubator.foreign.MemoryLayouts.C_CHAR;
 42 
 43 public class InternalForeign implements Foreign {
 44 
 45     private static final String foreignAccess
 46             = GetPropertyAction.privilegedGetProperty("jdk.incubator.foreign.Foreign", "deny");
 47     private static final InternalForeign INSTANCE = new InternalForeign();
 48 
 49     private InternalForeign() {}
 50 
 51     public static InternalForeign getInstance() {
 52         checkRestrictedAcccess();
 53         return getInstancePriviledged();
 54     }
 55 
 56     public static InternalForeign getInstancePriviledged() {
 57         return INSTANCE;
 58     }
 59 
 60     @Override
 61     public long asLong(MemoryAddress address) throws IllegalAccessError {
 62         return MemoryAddressImpl.addressof(address);
 63     }
 64 
 65     @Override
 66     public MemorySegment ofNativeUnchecked(MemoryAddress base, long byteSize) throws IllegalAccessError {
 67         return Utils.makeNativeSegmentUnchecked(base, byteSize);
 68     }
 69 
 70     @Override
 71     public SystemABI getSystemABI() {
 72         String arch = System.getProperty("os.arch");
 73         String os = System.getProperty("os.name");
 74         if (arch.equals("amd64") || arch.equals("x86_64")) {
 75             if (os.startsWith("Windows")) {
 76                 return Windowsx64ABI.getInstance();
 77             } else {
 78                 return SysVx64ABI.getInstance();
 79             }
 80         } else if (arch.equals("aarch64")) {
 81             return AArch64ABI.getInstance();
 82         }
 83         throw new UnsupportedOperationException("Unsupported os or arch: " + os + ", " + arch);
 84     }
 85 
 86     private static void checkRestrictedAcccess() {
 87         switch (foreignAccess) {
 88             case "deny" -> throwIllegalAccessError(foreignAccess);
 89             case "warn" -> System.err.println("WARNING: Accessing jdk.incubator.foreign.Foreign.");
 90             case "debug" -> {
 91                 StringBuilder sb = new StringBuilder("DEBUG: Accessing jdk.incubator.foreign.Foreign.");
 92                 StackWalker.getInstance().walk(s -> {
 93                      s
 94                      .forEach(f -> sb.append(System.lineSeparator()).append("\tat " + f));
 95                     return null;
 96                 });
 97                 System.out.println(sb.toString());
 98             }
 99             case "permit" -> {}
100             default -> throwIllegalAccessError(foreignAccess);
101         }
102     }
103 
104     private static void throwIllegalAccessError(String value) {
105         throw new IllegalAccessError("Can not access jdk.incubator.foreign.Foreign." +
106                 " System property 'jdk.incubator.foreign.Foreign' is set to '" + value + "'");
107     }
108 
109     private static VarHandle arrayHandle(MemoryLayout elemLayout, Class<?> elemCarrier) {
110         return MemoryLayout.ofSequence(1, elemLayout)
111                 .varHandle(elemCarrier, MemoryLayout.PathElement.sequenceElement());
112     }
113 
114 
115     @Override
116     public MemorySegment toCString(String str) {
117         return toCString(str.getBytes());
118     }
119 
120     @Override
121     public MemorySegment toCString(String str, Charset charset) {
122         return toCString(str.getBytes(charset));
123     }
124 
125     private MemorySegment toCString(byte[] bytes) {
126         MemoryLayout strLayout = MemoryLayout.ofSequence(bytes.length + 1, C_CHAR);
127         MemorySegment segment = MemorySegment.allocateNative(strLayout);
128         MemoryAddress addr = segment.baseAddress();
129         for (int i = 0 ; i < bytes.length; i++) {
130             Lazy.byteArrHandle.set(addr, i, bytes[i]);
131         }
132         Lazy.byteArrHandle.set(addr, (long)bytes.length, (byte)0);
133         return segment;
134     }
135 
136     @Override
137     public String toJavaString(MemoryAddress addr) {
138         StringBuilder buf = new StringBuilder();
139         try (MemorySegment seg = ofNativeUnchecked(addr, Long.MAX_VALUE)) {
140             MemoryAddress baseAddr = seg.baseAddress();
141             byte curr = (byte) Lazy.byteArrHandle.get(baseAddr, 0);
142             long offset = 0;
143             while (curr != 0) {
144                 buf.append((char) curr);
145                 curr = (byte) Lazy.byteArrHandle.get(baseAddr, ++offset);
146             }
147         }
148         return buf.toString();
149     }
150 
151     // need to lazily initialize this to prevent circular init
152     // MemoryLayouts -> Foreign -> MemoryLayouts
153     private static class Lazy {
154         final static VarHandle byteArrHandle = arrayHandle(C_CHAR, byte.class);
155     }
156 }
