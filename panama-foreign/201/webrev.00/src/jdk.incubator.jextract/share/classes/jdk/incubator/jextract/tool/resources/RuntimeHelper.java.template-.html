114             assert(args.length == nNamedArgs + 1);

119             Class<?>[] argTypes = new Class<?>[argsCount];
120             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];

122             int pos = 0;
123             for (pos = 0; pos < nNamedArgs; pos++) {
124                 argTypes[pos] = varargs.parameterType(pos);
125                 argLayouts[pos] = function.argumentLayouts().get(pos);
126             }

128             assert pos == nNamedArgs;
129             for (Object o: unnamedArgs) {
130                 argTypes[pos] = normalize(o.getClass());
131                 argLayouts[pos] = variadicLayout(argTypes[pos]);

134             assert pos == argsCount;

136             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);

139                     FunctionDescriptor.of(function.returnLayout().get(), argLayouts);
