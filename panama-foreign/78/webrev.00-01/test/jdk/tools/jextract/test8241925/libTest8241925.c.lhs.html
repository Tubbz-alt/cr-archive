26 EXPORT void square(int* v) {
27     *v = (*v) * (*v);

30 EXPORT int sum(int* arr, int num) {
31     int res = 0;
32     int i = 0;
33     for (; i < num; i++) {
34         res += arr[i];
35     }
36     return res;
37 }
