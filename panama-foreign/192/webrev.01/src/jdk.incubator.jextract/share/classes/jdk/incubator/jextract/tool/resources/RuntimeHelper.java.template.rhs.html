156         private Object invoke(Object[] args) throws Throwable {
157             // one trailing Object[]
158             int nNamedArgs = function.argumentLayouts().size();
159             assert(args.length == nNamedArgs + 1);
160             // The last argument is the array of vararg collector
161             Object[] unnamedArgs = (Object[]) args[args.length - 1];
162 
163             int argsCount = nNamedArgs + unnamedArgs.length;
164             Class<?>[] argTypes = new Class<?>[argsCount];
165             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];
166 
167             int pos = 0;
168             for (pos = 0; pos < nNamedArgs; pos++) {
169                 argTypes[pos] = varargs.parameterType(pos);
170                 argLayouts[pos] = function.argumentLayouts().get(pos);
171             }
172 
173             assert pos == nNamedArgs;
174             for (Object o: unnamedArgs) {
175                 argTypes[pos] = normalize(o.getClass());
176                 argLayouts[pos] = variadicLayout(argTypes[pos]);
177                 pos++;
178             }
179             assert pos == argsCount;
180 
181             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);
182             FunctionDescriptor f = (function.returnLayout().isEmpty()) ?
183                     FunctionDescriptor.ofVoid(argLayouts) :
184                     FunctionDescriptor.of(function.returnLayout().get(), argLayouts);
185             MethodHandle mh = ABI.downcallHandle(symbol, mt, f);
186             // flatten argument list so that it can be passed to an asSpreader MH
187             Object[] allArgs = new Object[nNamedArgs + unnamedArgs.length];
188             System.arraycopy(args, 0, allArgs, 0, nNamedArgs);
189             System.arraycopy(unnamedArgs, 0, allArgs, nNamedArgs, unnamedArgs.length);
190 
191             return mh.asSpreader(Object[].class, argsCount).invoke(allArgs);
192         }
