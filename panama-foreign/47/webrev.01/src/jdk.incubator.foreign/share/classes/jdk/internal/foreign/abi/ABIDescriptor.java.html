23 package jdk.internal.foreign.abi;
24 
25 public class ABIDescriptor {
26     final Architecture arch;
27 
28     public final VMStorage[][] inputStorage;
29     public final VMStorage[][] outputStorage;
30 
31     final VMStorage[][] volatileStorage;
32 
33     final int stackAlignment;
34     final int shadowSpace;
35 
36     public ABIDescriptor(Architecture arch, jdk.internal.foreign.abi.VMStorage[][] inputStorage, jdk.internal.foreign.abi.VMStorage[][] outputStorage,
37                          VMStorage[][] volatileStorage, int stackAlignment, int shadowSpace) {
38         this.arch = arch;
39         this.inputStorage = inputStorage;
40         this.outputStorage = outputStorage;
41         this.volatileStorage = volatileStorage;
42         this.stackAlignment = stackAlignment;
43         this.shadowSpace = shadowSpace;
44     }
45 }
