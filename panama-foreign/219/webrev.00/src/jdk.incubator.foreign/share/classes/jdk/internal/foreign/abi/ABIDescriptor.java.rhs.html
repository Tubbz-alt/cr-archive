23 package jdk.internal.foreign.abi;
24 
25 import jdk.internal.invoke.ABIDescriptorProxy;
26 
27 public class ABIDescriptor implements ABIDescriptorProxy {
28     final Architecture arch;
29 
30     public final VMStorage[][] inputStorage;
31     public final VMStorage[][] outputStorage;
32 
33     final VMStorage[][] volatileStorage;
34 
35     final int stackAlignment;
36     final int shadowSpace;
37 
38     public ABIDescriptor(Architecture arch, VMStorage[][] inputStorage, VMStorage[][] outputStorage,
39                          VMStorage[][] volatileStorage, int stackAlignment, int shadowSpace) {
40         this.arch = arch;
41         this.inputStorage = inputStorage;
42         this.outputStorage = outputStorage;
43         this.volatileStorage = volatileStorage;
44         this.stackAlignment = stackAlignment;
45         this.shadowSpace = shadowSpace;
46     }
47 
48     @Override
49     public int shadowSpaceBytes() {
50         return shadowSpace;
51     }
52 }
