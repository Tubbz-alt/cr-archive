23 package jdk.internal.foreign.abi;
24 


25 import java.util.Objects;
26 
27 public class VMStorage {
28     private final int type;
29     private final int index;
30 
31     private final String debugName;
32 
33     public VMStorage(int type, int index, String debugName) {
34         this.type = type;
35         this.index = index;
36         this.debugName = debugName;
37     }
38 
39     public int type() {
40         return type;
41     }
42 
43     public int index() {
44         return index;
45     }
46 
47     public String name() {
48         return debugName;
49     }
50 
51     @Override
52     public boolean equals(Object o) {
53         if (this == o) return true;
54         if (o == null || getClass() != o.getClass()) return false;
55         VMStorage vmStorage = (VMStorage) o;
56         return type == vmStorage.type &&
57                 index == vmStorage.index;
58     }
59 
60     @Override
61     public int hashCode() {
62         return Objects.hash(type, index);
63     }
64 
65     @Override
66     public String toString() {
67         return "VMStorage{" +
68                 "type=" + type +
69                 ", index=" + index +
70                 ", debugName='" + debugName + '\'' +
71                 '}';
72     }
73 }
