31 // ciObjArray
32 //
33 // This class represents an objArrayOop in the HotSpot virtual
34 // machine.
35 
36 ciObject* ciObjArray::obj_at(int index) {
37   VM_ENTRY_MARK;
38   objArrayOop array = get_objArrayOop();
39   if (index < 0 || index >= array->length()) return NULL;
40   oop o = array->obj_at(index);
41   if (o == NULL) {
42     return ciNullObject::make();
43   } else {
44     return CURRENT_ENV->get_object(o);
45   }
46 }
