23 package jdk.internal.foreign.abi;
24 
25 import jdk.internal.invoke.VMStorageProxy;
26 
27 import java.util.Objects;
28 
29 public class VMStorage implements VMStorageProxy {
30     private final int type;
31     private final int index;
32 
33     private final String debugName;
34 
35     public VMStorage(int type, int index, String debugName) {
36         this.type = type;
37         this.index = index;
38         this.debugName = debugName;
39     }
40 
41     public int type() {
42         return type;
43     }
44 
45     public int index() {
46         return index;
47     }
48 
49     public String name() {
50         return debugName;
51     }
52 
53     @Override
54     public boolean equals(Object o) {
55         if (this == o) return true;
56         if (o == null || getClass() != o.getClass()) return false;
57         VMStorage vmStorage = (VMStorage) o;
58         return type == vmStorage.type &&
59                 index == vmStorage.index;
60     }
61 
62     @Override
63     public int hashCode() {
64         return Objects.hash(type, index);
65     }
66 
67     @Override
68     public String toString() {
69         return "VMStorage{" +
70                 "type=" + type +
71                 ", index=" + index +
72                 ", debugName='" + debugName + '\'' +
73                 '}';
74     }
75 }
