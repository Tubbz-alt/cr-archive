143         CompletableFuture<?> r;
144         try (MemorySegment s1 = MemorySegment.allocateNative(MemoryLayout.ofSequence(2, MemoryLayouts.JAVA_INT))) {
145             r = CompletableFuture.runAsync(() -> {
146                 try {

155                     }

162                 }
163             });
