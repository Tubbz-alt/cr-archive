 51 import jdk.incubator.foreign.CSupport;
 52 import jdk.incubator.foreign.FunctionDescriptor;
 53 import jdk.incubator.foreign.LibraryLookup;
 54 import jdk.incubator.foreign.MemoryAddress;
 55 import jdk.incubator.foreign.MemoryLayout;
 56 import jdk.incubator.foreign.ForeignLinker;
 57 
 58 import java.lang.invoke.MethodHandle;
 59 import java.lang.invoke.MethodType;
 60 import java.util.ArrayList;
 61 import java.util.List;
 62 import java.util.function.Consumer;
 63 
 64 import jdk.incubator.foreign.MemorySegment;
 65 import org.testng.annotations.*;
 66 
 67 public class TestDowncall extends CallGeneratorHelper {
 68 
 69     static LibraryLookup lib = LibraryLookup.ofLibrary("TestDowncall");
 70     static ForeignLinker abi = CSupport.getSystemLinker();
 71 
 72 
 73     @Test(dataProvider="functions", dataProviderClass=CallGeneratorHelper.class)
 74     public void testDowncall(String fName, Ret ret, List<ParamType> paramTypes, List<StructFieldType> fields) throws Throwable {
 75         List<Consumer<Object>> checks = new ArrayList<>();
 76         List<MemorySegment> segments = new ArrayList<>();
 77         LibraryLookup.Symbol addr = lib.lookup(fName);
 78         MethodHandle mh = abi.downcallHandle(addr, methodType(ret, paramTypes, fields), function(ret, paramTypes, fields));
 79         Object[] args = makeArgs(paramTypes, fields, checks, segments);
 80         mh = mh.asSpreader(Object[].class, paramTypes.size());
 81         Object res = mh.invoke(args);
 82         if (ret == Ret.NON_VOID) {
 83             checks.forEach(c -> c.accept(res));
 84         }
 85         segments.forEach(MemorySegment::close);
 86     }
 87 
 88     static MethodType methodType(Ret ret, List<ParamType> params, List<StructFieldType> fields) {
 89         MethodType mt = ret == Ret.VOID ?
 90                 MethodType.methodType(void.class) : MethodType.methodType(paramCarrier(params.get(0).layout(fields)));
 91         for (ParamType p : params) {
 92             mt = mt.appendParameterTypes(paramCarrier(p.layout(fields)));
 93         }
 94         return mt;
 95     }
 96 
 97     static FunctionDescriptor function(Ret ret, List<ParamType> params, List<StructFieldType> fields) {
 98         MemoryLayout[] paramLayouts = params.stream().map(p -> p.layout(fields)).toArray(MemoryLayout[]::new);
 99         return ret == Ret.VOID ?
100                 FunctionDescriptor.ofVoid(paramLayouts) :
101                 FunctionDescriptor.of(paramLayouts[0], paramLayouts);
102     }
103 
104     static Object[] makeArgs(List<ParamType> params, List<StructFieldType> fields, List<Consumer<Object>> checks, List<MemorySegment> segments) throws ReflectiveOperationException {
105         Object[] args = new Object[params.size()];
106         for (int i = 0 ; i < params.size() ; i++) {
107             args[i] = makeArg(params.get(i).layout(fields), checks, i == 0, segments);
108         }
109         return args;
110     }
111 }
