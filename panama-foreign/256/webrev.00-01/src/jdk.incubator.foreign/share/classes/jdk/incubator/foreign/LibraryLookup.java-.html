 26 package jdk.incubator.foreign;
 27 
 28 import jdk.internal.foreign.LibrariesHelper;
 29 
 30 import java.io.File;
 31 import java.lang.invoke.MethodType;
 32 import java.util.Objects;
 33 
 34 /**
 35  * A native library lookup. Exposes lookup operation for searching symbols, see {@link LibraryLookup#lookup(String)}.
 36  * A given native library remains loaded as long as there is at least one <em>live</em> library lookup instance referring
 37  * to it.
 38  * <p>
 39  * Memory address instances generated by a library lookup will contain a strong reference to the originating lookup object,
 40  * therefore preventing library unloading; in turn method handle instances obtained from
 41  * {@link ForeignLinker#downcallHandle(Addressable, MethodType, FunctionDescriptor)}) also maintain a strong reference
 42  * to the memory address parameter used for their construction. This means that there is always a strong reachability chain
 43  * from a native method handle to a lookup object (the one that was used to lookup the native library symbol the method handle
 44  * refers to); this is useful to prevent situations where a native library is unloaded in the middle of a native call.
 45  * <p>
 46  * To allow for a library to be unloaded, a client will have to discard any strong references it
 47  * maintains, directly, or indirectly to a lookup object associated with given library.
 48  */
 49 public interface LibraryLookup {
 50 
 51     /**
 52      * A symbol retrieved during a library lookup. A lookup symbol has a <em>name</em> and can be projected
 53      * into a memory address (see {@link #name()} and {@link #address()}, respectively).
 54      */
 55     interface Symbol extends Addressable {
 56         /**
 57          * The name of this lookup symbol.
 58          * @return the name of this lookup symbol.
 59          */
 60         String name();
 61 
 62         /**
 63          * The memory address of this lookup symbol.
 64          * @return the memory address of this lookup symbol.
 65          */
 66         @Override
 67         MemoryAddress address();
 68     }
 69 
 70     /**
 71      * Lookups a symbol with given name in this library. The returned address has the same temporal bounds
 72      * as this lookup object.
 73      * @param name the symbol name.
 74      * @return the library symbol (if any)
 75      * @throws NoSuchMethodException if no symbol with given name could be found.
 76      */
 77     Symbol lookup(String name) throws NoSuchMethodException;
 78 
 79     /**
 80      * Obtain a default library lookup object.
 81      * @return the default library lookup object.
 82      */
 83     static LibraryLookup ofDefault() {
 84         SecurityManager security = System.getSecurityManager();
 85         if (security != null) {
 86             security.checkPermission(new RuntimePermission("java.foreign.getDefaultLibrary"));
 87         }
 88         return LibrariesHelper.getDefaultLibrary();
 89     }
 90 
 91     /**
 92      * Obtain a library lookup object corresponding to a library identified by given path.
 93      * @param path the library path.
 94      * @return a library lookup object for given path.
 95      */
 96     static LibraryLookup ofPath(String path) {
 97         Objects.requireNonNull(path);
 98         SecurityManager security = System.getSecurityManager();
 99         if (security != null) {
100             security.checkLink(path);
101         }
102         if (!(new File(path).isAbsolute())) {
103             throw new UnsatisfiedLinkError(
104                     "Expecting an absolute path of the library: " + path);
105         }
106         return LibrariesHelper.load(path);
107     }
108 
109     /**
110      * Obtain a library lookup object corresponding to a library identified by given library name.
111      * @param libName the library name.
112      * @return a library lookup object for given library name.
113      */
114     static LibraryLookup ofLibrary(String libName) {
115         Objects.requireNonNull(libName);
116         SecurityManager security = System.getSecurityManager();
117         if (security != null) {
118             security.checkLink(libName);
119         }
120         if (libName.indexOf(File.separatorChar) != -1) {
121             throw new UnsatisfiedLinkError(
122                     "Directory separator should not appear in library name: " + libName);
123         }
124         return LibrariesHelper.loadLibrary(libName);
125     }
126 }
