 59     private MacroParserImpl(Reparser reparser, TreeMaker treeMaker) {

 65     public static MacroParserImpl make(TreeMaker treeMaker, TranslationUnit tu, Collection<String> args) {
 66         Reparser reparser;
 67         try {
 68             reparser = new ClangReparser(tu, args);
 69         } catch (IOException | Index.ParsingFailedException ex) {
 70             throw new RuntimeException(ex);
 71         }
 72 
 73         return new MacroParserImpl(reparser, treeMaker);
 74     }

 76     /**
 77      * This method attempts to evaluate the macro. Evaluation occurs in two steps: first, an attempt is made
 78      * to see if the macro corresponds to a simple numeric constant. If so, the constant is parsed in Java directly.
 79      * If that is not possible (e.g. because the macro refers to other macro, or has a more complex grammar), fall
 80      * back to use clang evaluation support.
 81      */
 82     @Override
 83     public Optional<Declaration.Constant> parseConstant(Position pos, String name, String[] tokens) {
 84         if (!(pos instanceof TreeMaker.CursorPosition)) {
 85             return Optional.empty();
 86         } else {
 87             Cursor cursor = ((TreeMaker.CursorPosition)pos).cursor();
 88             if (cursor.isMacroFunctionLike()) {
 89                 return Optional.empty();
 90             } else if (tokens.length == 2) {
 91                 //check for fast path
 92                 Integer num = toNumber(tokens[1]);
 93                 if (num != null) {
 94                     return Optional.of(treeMaker.createMacro(cursor, name, Type.primitive(Type.Primitive.Kind.Int), (long)num));
 95                 }
 96             }
 97             macroTable.enterMacro(name, tokens, cursor);
 98             return Optional.empty();
 99         }
100     }
101 
102     private Integer toNumber(String str) {
103         try {
104             // Integer.decode supports '#' hex literals which is not valid in C.
105             return str.length() > 0 && str.charAt(0) != '#'? Integer.decode(str) : null;
106         } catch (NumberFormatException nfe) {
107             return null;
108         }
109     }

111     interface Reparser {
112         Stream<Cursor> reparse(String snippet);
113     }
114 
115     /**
116      * This class allows client to reparse a snippet of code against a given set of include files.
117      * For performance reasons, the set of includes (which comes from the jextract parser) is compiled
118      * into a precompiled header, so as to speed to incremental recompilation of the generated snippets.
119      */
120     static class ClangReparser implements Reparser {
121         final Path macro;
122         final Index macroIndex = LibClang.createIndex(true);
123         final TranslationUnit macroUnit;
124 
125         public ClangReparser(TranslationUnit tu, Collection<String> args) throws IOException, Index.ParsingFailedException {
126             Path precompiled = Files.createTempFile("jextract$", ".pch");
127             precompiled.toFile().deleteOnExit();
128             tu.save(precompiled);
129             this.macro = Files.createTempFile("jextract$", ".h");
130             this.macro.toFile().deleteOnExit();
131             String[] patchedArgs = Stream.concat(
132                 Stream.of(
133                     // Avoid system search path, use bundled instead
134                     "-nostdinc",
135                     "-ferror-limit=0",
136                     // precompiled header
137                     "-include-pch", precompiled.toAbsolutePath().toString()),
138                 args.stream()).toArray(String[]::new);
139             this.macroUnit = macroIndex.parse(macro.toAbsolutePath().toString(),
140                     this::processDiagnostics,
141                     false, //add serialization support (needed for macros)
142                     patchedArgs);
143         }
144 
145         void processDiagnostics(Diagnostic diag) {
146             if (Main.DEBUG) {
147                 System.err.println("Error while processing macro: " + diag.spelling());
148             }
149         }
150 
151         @Override
152         public Stream<Cursor> reparse(String snippet) {
153             macroUnit.reparse(this::processDiagnostics,
154                     Index.UnsavedFile.of(macro, snippet));
155             return macroUnit.getCursor().children();
156         }
157     }

159     /**
160      * This abstraction is used to collect all macros which could not be interpreted during {@link #parseConstant(Position, String, String[])}.
161      * All unparsed macros in the table can have three different states: UNPARSED (which means the macro has not been parsed yet),
162      * SUCCESS (which means the macro has been parsed and has a type and a value) and FAILURE, which means the macro has been
163      * parsed with some errors, but for which we were at least able to infer a type.
164      *
165      * The reparsing process goes as follows:
166      * 1. all unparsed macros are added to the table in the UNPARSED state.
167      * 2. a snippet for all macros in the UNPARSED state is compiled and the table state is updated
168      * 3. a recovery snippet for all macros in the FAILURE state is compiled and the table state is updated again
169      * 4. we repeat from (2) until no further progress is made.
170      * 5. we return a list of macro which are in the SUCCESS state.
171      *
172      * State transitions in the table are as follows:
173      * - an UNPARSED macro can go to either SUCCESS, to FAILURE or be removed (if not even a type can be inferred)
174      * - a FAILURE macro can go to either SUCCESS (if recovery step succeds) or be removed
175      * - a SUCCESS macro cannot go in any other state
176      */
177     class MacroTable {
178 
179         final Map<String, Entry> macrosByMangledName = new LinkedHashMap<>();
180 
181         abstract class Entry {
182             final String name;
183             final String[] tokens;
184             final Cursor cursor;
185 
186             Entry(String name, String[] tokens, Cursor cursor) {
187                 this.name = name;
188                 this.tokens = tokens;
189                 this.cursor = cursor;
190             }
191 
192             String mangledName() {
193                 return "jextract$macro$" + name;
194             }
195 
196             Entry success(Type type, Object value) {
197                 throw new IllegalStateException();
198             }
199 
200             Entry failure(Type type) {
201                 throw new IllegalStateException();
202             }
203 
204             boolean isSuccess() {
205                 return false;
206             }
207             boolean isRecoverableFailure() {
208                 return false;
209             }
210             boolean isUnparsed() {
211                 return false;
212             }
213 
214             void update() {
215                 macrosByMangledName.put(mangledName(), this);
216             }
217         }
218 
219         class Unparsed extends Entry {
220             Unparsed(String name, String[] tokens, Cursor cursor) {
221                 super(name, tokens, cursor);
222             }
223 
224             @Override
225             Entry success(Type type, Object value) {
226                 return new Success(name, tokens, cursor, type, value);
227             }
228 
229             @Override
230             Entry failure(Type type) {
231                 return type != null ?
232                         new RecoverableFailure(name, tokens, cursor, type) :
233                         new UnparseableMacro(name, tokens, cursor);
234             }
235 
236             @Override
237             boolean isUnparsed() {
238                 return true;
239             }
240 
241             @Override
242             void update() {
243                 throw new IllegalStateException();
244             }
245         }
246 
247         class RecoverableFailure extends Entry {
248 
249             final Type type;
250 
251             public RecoverableFailure(String name, String[] tokens, Cursor cursor, Type type) {
252                 super(name, tokens, cursor);
253                 this.type = type;
254             }
255 
256             @Override
257             Entry success(Type type, Object value) {
258                 return new Success(name, tokens, cursor, this.type, value);
259             }
260 
261             @Override
262             Entry failure(Type type) {
263                 return new UnparseableMacro(name, tokens, cursor);
264             }
265 
266             @Override
267             boolean isRecoverableFailure() {
268                 return true;
269             }
270         }
271 
272         class Success extends Entry {
273             final Type type;
274             final Object value;
275 
276             public Success(String name, String[] tokens, Cursor cursor, Type type, Object value) {
277                 super(name, tokens, cursor);
278                 this.type = type;
279                 this.value = value;
280             }
281 
282             @Override
283             boolean isSuccess() {
284                 return true;
285             }
286 
287             public Object value() {
288                 return value;
289             }
290         }
291 
292         class UnparseableMacro extends Entry {
293 
294             UnparseableMacro(String name, String[] tokens, Cursor cursor) {
295                 super(name, tokens, cursor);
296             }
297 
298             @Override
299             void update() {
300                 macrosByMangledName.remove(mangledName());
301             }
302         };
303 
304         void enterMacro(String name, String[] tokens, Cursor cursor) {
305             Unparsed unparsed = new Unparsed(name, tokens, cursor);
306             macrosByMangledName.put(unparsed.mangledName(), unparsed);
307         }
308 
309         public List<Declaration.Constant> reparseConstants() {
310             int last = -1;
311             while (macrosByMangledName.size() > 0 && last != macrosByMangledName.size()) {
312                 last = macrosByMangledName.size();
313                 // step 1 - try parsing macros as var declarations
314                 reparseMacros(false);
315                 // step 2 - retry failed parsed macros as pointers
316                 reparseMacros(true);
317             }
318             treeMaker.typeMaker.resolveTypeReferences();
319             return macrosByMangledName.values().stream()
320                     .filter(Entry::isSuccess)
321                     .map(e -> treeMaker.createMacro(e.cursor, e.name, ((Success)e).type, ((Success)e).value))
322                     .collect(Collectors.toList());
323         }
324 
325         void updateTable(TypeMaker typeMaker, Cursor decl) {
326             String mangledName = decl.spelling();
327             Entry entry = macrosByMangledName.get(mangledName);
328             try (EvalResult result = decl.eval()) {
329                 Entry newEntry = switch (result.getKind()) {
330                     case Integral -> {
331                         long value = result.getAsInt();
332                         yield entry.success(typeMaker.makeType(decl.type()), value);
333                     }
334                     case FloatingPoint -> {
335                         double value = result.getAsFloat();
336                         yield entry.success(typeMaker.makeType(decl.type()), value);
337                     }
338                     case StrLiteral -> {
339                         String value = result.getAsString();
340                         yield entry.success(typeMaker.makeType(decl.type()), value);
341                     }
342                     default -> {
343                         Type type = decl.type().equals(decl.type().canonicalType()) ?
344                                 null : typeMaker.makeType(decl.type());
345                         yield entry.failure(type);
346                     }
347                 };
348                 newEntry.update();
349             }
350         }
351 
352         void reparseMacros(boolean recovery) {
353             String snippet = macroDecl(recovery);
354             TreeMaker treeMaker = new TreeMaker();
355             try {
356                 reparser.reparse(snippet)
357                         .filter(c -> c.kind() == CursorKind.VarDecl &&
358                                 c.spelling().contains("jextract$"))
359                         .forEach(c -> updateTable(treeMaker.typeMaker, c));
360             } finally {
361                 treeMaker.typeMaker.resolveTypeReferences();
362             }
363         }
364 
365         String macroDecl(boolean recovery) {
366             StringBuilder buf = new StringBuilder();
367             if (recovery) {
368                 buf.append("#include <stdint.h>\n");
369             }
370             macrosByMangledName.values().stream()
371                     .filter(e -> !e.isSuccess()) // skip macros that already have passed
372                     .filter(recovery ? Entry::isRecoverableFailure : Entry::isUnparsed)
373                     .forEach(e -> {
374                         buf.append("__auto_type ")
375                                 .append(e.mangledName())
376                                 .append(" = ");
377                         if (recovery) {
378                             buf.append("(uintptr_t)");
379                         }
380                         buf.append(e.name)
381                                 .append(";\n");
382                     });
383             return buf.toString();
384         }
385     }
