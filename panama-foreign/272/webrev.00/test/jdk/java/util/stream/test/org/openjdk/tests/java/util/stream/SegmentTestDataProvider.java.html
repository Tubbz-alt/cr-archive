 41 public class SegmentTestDataProvider {
 42 
 43     static boolean compareSegmentsByte(Collection<MemorySegment> segments1, Collection<MemorySegment> segments2, boolean isOrdered) {
 44         Function<MemorySegment, Byte> mapper = MemoryAccess::getByte;
 45         List<Byte> list1 = segments1.stream()
 46                 .map(mapper)
 47                 .collect(Collectors.toList());
 48         List<Byte> list2 = segments2.stream()
 49                 .map(mapper)
 50                 .collect(Collectors.toList());
 51         return list1.equals(list2);
 52     }

109     static Consumer<MemorySegment> segmentCopier(Consumer<MemorySegment> input) {
110         return segment -> {
111             MemorySegment dest = MemorySegment.ofArray(new byte[(int)segment.byteSize()]);
112             dest.copyFrom(segment);
113             input.accept(dest);
114         };
115     }

128     static void initSegment(MemorySegment segment) {
129         for (int i = 0 ; i < segment.byteSize() ; i++) {
130             MemoryAccess.setByte(segment, (byte)i);
131         }
132     }
