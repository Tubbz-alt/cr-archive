110         private Object invoke(Object[] args) throws Throwable {
111             // one trailing Object[]
112             int nNamedArgs = function.argumentLayouts().size();
113             assert(args.length == nNamedArgs + 1);
114             // The last argument is the array of vararg collector
115             Object[] unnamedArgs = (Object[]) args[args.length - 1];
116 
117             int argsCount = nNamedArgs + unnamedArgs.length;
118             Class<?>[] argTypes = new Class<?>[argsCount];
119             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];
120 
121             int pos = 0;
122             for (pos = 0; pos < nNamedArgs; pos++) {
123                 argTypes[pos] = varargs.parameterType(pos);
124                 argLayouts[pos] = function.argumentLayouts().get(pos);
125             }
126 
127             assert pos == nNamedArgs;
128             for (Object o: unnamedArgs) {
129                 argTypes[pos] = normalize(o.getClass());
130                 argLayouts[pos] = variadicLayout(argTypes[pos]);
131                 pos++;
132             }
133             assert pos == argsCount;
134 
135             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);
136             FunctionDescriptor f = (function.returnLayout().isEmpty()) ?
137                     FunctionDescriptor.ofVoid(argLayouts) :
138                     FunctionDescriptor.of(function.returnLayout().get(), argLayouts);
139             MethodHandle mh = ABI.downcallHandle(symbol, mt, f);
140             // flatten argument list so that it can be passed to an asSpreader MH
141             Object[] allArgs = new Object[nNamedArgs + unnamedArgs.length];
142             System.arraycopy(args, 0, allArgs, 0, nNamedArgs);
143             System.arraycopy(unnamedArgs, 0, allArgs, nNamedArgs, unnamedArgs.length);
144 
145             return mh.asSpreader(Object[].class, argsCount).invoke(allArgs);
146         }
