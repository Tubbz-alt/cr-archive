 26 package jdk.javadoc.internal.doclets.formats.html.markup;
 27 
 28 import jdk.javadoc.internal.doclets.toolkit.util.Utils;
 29 
 30 /**
 31  * Enum representing HTML tags.
 32  *
 33  *  <p><b>This is NOT part of any supported API.
 34  *  If you write code that depends on this, you do so at your own risk.
 35  *  This code and its internal interfaces are subject to change or
 36  *  deletion without notice.</b>
 37  */
 38 public enum HtmlTag {
 39     A(BlockType.INLINE, EndTag.END),
 40     BUTTON(BlockType.INLINE, EndTag.END),
 41     BLOCKQUOTE,
 42     BODY(BlockType.OTHER, EndTag.END),
 43     BR(BlockType.INLINE, EndTag.NOEND),
 44     CAPTION,
 45     CODE(BlockType.INLINE, EndTag.END),
 46     DD,
 47     DIV,
 48     DL,
 49     DT,
 50     EM(BlockType.INLINE, EndTag.END),
 51     FOOTER,
 52     H1,
 53     H2,
 54     H3,
 55     H4,
 56     H5,
 57     H6,
 58     HEAD(BlockType.OTHER, EndTag.END),
 59     HEADER,
 60     HR(BlockType.BLOCK, EndTag.NOEND),
 61     HTML(BlockType.OTHER, EndTag.END),
 62     I(BlockType.INLINE, EndTag.END),

 63     IMG(BlockType.INLINE, EndTag.NOEND),
 64     INPUT(BlockType.BLOCK, EndTag.NOEND),
 65     LABEL(BlockType.INLINE, EndTag.END),
 66     LI,
 67     LISTING,
 68     LINK(BlockType.OTHER, EndTag.NOEND),
 69     MAIN,
 70     MENU,
 71     META(BlockType.OTHER, EndTag.NOEND),
 72     NAV,
 73     NOSCRIPT(BlockType.OTHER, EndTag.END),
 74     OL,
 75     P,
 76     PRE,
 77     SCRIPT(BlockType.OTHER, EndTag.END),
 78     SECTION,
 79     SMALL(BlockType.INLINE, EndTag.END),
 80     SPAN(BlockType.INLINE, EndTag.END),
 81     STRONG(BlockType.INLINE, EndTag.END),
 82     SUB(BlockType.INLINE, EndTag.END),
 83     TABLE,
 84     TBODY,
 85     THEAD,
 86     TD,
 87     TH,
 88     TITLE(BlockType.OTHER, EndTag.END),
 89     TR,
 90     UL;
 91 
 92     public final BlockType blockType;
 93     public final EndTag endTag;
 94     public final String value;
 95 
 96     /**
 97      * Enum representing the type of HTML element.
 98      */
 99     public enum BlockType {
100         BLOCK,
101         INLINE,
102         OTHER
103     }
104 
105     /**
106      * Enum representing HTML end tag requirement.
107      */
108     public enum EndTag {
109         END,
110         NOEND
111     }
112 
113     HtmlTag() {
114         this(BlockType.BLOCK, EndTag.END);
115     }
116 
117     HtmlTag(BlockType blockType, EndTag endTag) {
118         this.blockType = blockType;
119         this.endTag = endTag;
120         this.value = Utils.toLowerCase(name());
121     }
122 
123     /**
124      * Returns true if the end tag is required. This is specific to the standard
125      * doclet and does not exactly resemble the W3C specifications.
126      *
127      * @return true if end tag needs to be displayed else return false
128      */
129     public boolean endTagRequired() {
130         return (endTag == EndTag.END);
131     }
132 
133     public String toString() {
134         return value;
135     }
136 }
