 24 /*
 25  * @test
 26  * @bug 8035473 8182765
 27  * @summary make sure tool is quiet when told to, and chatty otherwise
 28  */
 29 
 30 
 31 import java.io.BufferedReader;
 32 import java.io.File;
 33 import java.io.InputStreamReader;
 34 import java.util.ArrayList;
 35 import java.util.Iterator;
 36 import java.util.List;
 37 
 38 /**
 39  * Dummy javadoc comment.
 40  */
 41 public class QuietOption {
 42 
 43     final File javadoc;
 44     final File testSrc;
 45     final String thisClassName;
 46 
 47     public QuietOption() {
 48         File javaHome = new File(System.getProperty("java.home"));
 49         if (javaHome.getName().endsWith("jre"))
 50             javaHome = javaHome.getParentFile();
 51         javadoc = new File(new File(javaHome, "bin"), "javadoc");
 52         testSrc = new File(System.getProperty("test.src"));
 53         thisClassName = QuietOption.class.getName();
 54     }
 55 
 56     public static void main(String... args) throws Exception {
 57         QuietOption test = new QuietOption();
 58         test.run1();
 59         test.run2();
 60     }
 61 
 62     // make sure javadoc is quiet
 63     void run1() throws Exception {
 64         List<String> output = doTest(javadoc.getPath(),
 65                 "-classpath", ".", // insulates us from ambient classpath
 66                 "-quiet",
 67                 new File(testSrc, thisClassName + ".java").getPath());
 68 
 69         if (!output.isEmpty()) {
 70             // Remove any lines that might have been generated by the runtime
 71             Iterator<String> iter = output.iterator();
 72             while (iter.hasNext()) {
 73                 String line = iter.next();
 74                 if (line.matches("^Picked up .*JAVA.*OPTIONS:.*")) {
 75                     System.out.println("IGNORING: " + line);
 76                     iter.remove();
 77                 }
 78             }
 79         }
 80 
 81         if (!output.isEmpty()) {
 82             System.out.println(output);
 83             throw new Exception("run1: Shhh!, very chatty javadoc!.");
 84         }
 85     }
 86 
 87     // make sure javadoc is chatty
 88     void run2() throws Exception {
 89         List<String> output = doTest(javadoc.getPath(),
 90                 "-classpath", ".", // insulates us from ambient classpath
 91                 new File(testSrc, thisClassName + ".java").getPath());
 92 
 93         if (output.isEmpty()) {
 94             System.out.println(output);
 95             throw new Exception("run2: speak up and please be heard!.");
 96         }
 97     }
 98 
 99     /**
100      * More dummy comments.
101      */
102     List<String> doTest(String... args) throws Exception {
103         List<String> output = new ArrayList<>();
104         // run javadoc in separate process to ensure doclet executed under
105         // normal user conditions w.r.t. classloader
106         Process p = new ProcessBuilder()
107                 .command(args)
108                 .redirectErrorStream(true)
109                 .start();
110         try (BufferedReader in = new BufferedReader(new InputStreamReader(p.getInputStream()))) {
111             String line = in.readLine();
112             while (line != null) {
113                 output.add(line.trim());
114                 line = in.readLine();
115             }
116         }
117         int rc = p.waitFor();
118         if (rc != 0) {
119             throw new Exception("javadoc failed, rc:" + rc);
120         }
121         return output;
122     }
123 }
