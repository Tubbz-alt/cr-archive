 40 /**
 41  * This class defines several factory methods for constructing and combining memory access var handles.
 42  * To obtain a memory access var handle, clients must start from one of the <em>leaf</em> methods
 43  * (see {@link MemoryHandles#varHandle(Class, ByteOrder)},
 44  * {@link MemoryHandles#varHandle(Class, long, ByteOrder)}). This determines the variable type
 45  * (all primitive types but {@code void} and {@code boolean} are supported), as well as the alignment constraint and the
 46  * byte order associated to a memory access var handle. The resulting memory access var handle can then be combined in various ways
 47  * to emulate different addressing modes. The var handles created by this class feature a <em>mandatory</em> coordinate type
 48  * (of type {@link MemoryAddress}), and zero or more {@code long} coordinate types, which can be used to emulate
 49  * multi-dimensional array indexing.
 50  * <p>
 51  * As an example, consider the memory layout expressed by a {@link SequenceLayout} instance constructed as follows:
 52  * <blockquote><pre>{@code
 53 SequenceLayout seq = MemoryLayout.ofSequence(5,
 54     MemoryLayout.ofStruct(
 55         MemoryLayout.ofPaddingBits(32),
 56         MemoryLayout.ofValueBits(32, ByteOrder.BIG_ENDIAN).withName("value")
 57     ));
 58  * }</pre></blockquote>

115  * which is only compatible with the alignment constraint {@code B}; in such cases, access for anything other than the
116  * {@code get} and {@code set} access modes will result in an {@code IllegalStateException}. If access is partially aligned,
117  * atomic access is only guaranteed with respect to the largest power of two that divides the GCD of {@code A} and {@code S}.
118  * <p>
119  * Finally, in all other cases, we say that a memory access operation is <em>misaligned</em>; in such cases an
120  * {@code IllegalStateException} is thrown, irrespective of the access mode being used.
121  */

301     public static VarHandle asAddressVarHandle(VarHandle target) {
302         Class<?> carrier = target.varType();
303         if (!carrier.isPrimitive() || carrier == boolean.class ||
304                 carrier == float.class || carrier == double.class) {
305             throw new IllegalArgumentException("Unsupported carrier type: " + carrier.getName());
306         }
307 
308         if (carrier != long.class) {
309             // slow-path, we need to adapt
310             return filterValue(target,
311                     MethodHandles.explicitCastArguments(ADDRESS_TO_LONG, MethodType.methodType(carrier, MemoryAddress.class)),
312                     MethodHandles.explicitCastArguments(LONG_TO_ADDRESS, MethodType.methodType(MemoryAddress.class, carrier)));
313         } else {
314             // fast-path
315             return filterValue(target, ADDRESS_TO_LONG, LONG_TO_ADDRESS);
316         }
317     }

353     /**
354      * Adapts a target var handle by narrowing incoming values and widening
355      * outgoing values, to and from the given type, respectively.
356      * <p>
357      * The returned var handle can be used to conveniently treat unsigned
358      * primitive data types as if they were a wider signed primitive type. For
359      * example, it is often convenient to model an <i>unsigned short</i> as a
360      * Java {@code int} to avoid dealing with negative values, which would be
361      * the case if modeled as a Java {@code short}. The returned var handle
362      * converts to and from wider primitive types, to a more narrow possibly
363      * unsigned primitive type.
364      * <p>
365      * When calling e.g. {@link VarHandle#set(Object...)} on the resulting var
366      * handle, the incoming value (of type {@code adaptedType}) is converted by a
367      * <i>narrowing primitive conversion</i> and then passed to the {@code
368      * target} var handle. A narrowing primitive conversion may lose information
369      * about the overall magnitude of a numeric value. Conversely, when calling
370      * e.g. {@link VarHandle#get(Object...)} on the resulting var handle, the
371      * returned value obtained from the {@code target} var handle is converted
372      * by a <i>unsigned widening conversion</i> before being returned to the
373      * caller. In an unsigned widening conversion the high-order bits greater
374      * than that of the {@code target} carrier type are zero, and the low-order
375      * bits (equal to the width of the {@code target} carrier type) are equal to
376      * the bits of the value obtained from the {@code target} var handle.
377      * <p>
378      * The returned var handle will feature the variable type {@code adaptedType},
379      * and the same access coordinates, the same access modes (see {@link
380      * java.lang.invoke.VarHandle.AccessMode}, and the same atomic access
381      * guarantees, as those featured by the {@code target} var handle.
382      *
383      * @param target the memory access var handle to be adapted
384      * @param adaptedType the adapted type
385      * @returns the adapted var handle.
386      * @throws IllegalArgumentException if the carrier type of {@code target}
387      * is not one of {@code byte}, {@code short}, or {@code int}; if {@code
388      * adaptedType} is not one of {@code int}, or {@code long}; if the bitwidth
389      * of the {@code adaptedType} is not greater than that of the {@code target}
390      * carrier type
391      * @throws NullPointerException if either of {@code target} or {@code
392      * adaptedType} is null
393      *
394      * @jls 5.1.3 Narrowing Primitive Conversion
395      */
396     public static VarHandle asUnsigned(VarHandle target, final Class<?> adaptedType) {
397         Objects.requireNonNull(target);
398         Objects.requireNonNull(adaptedType);
399         final Class<?> carrier = target.varType();
400         checkWidenable(carrier);
401         checkNarrowable(adaptedType);
402         checkTargetWiderThanCarrier(carrier, adaptedType);
403 
404         if (adaptedType == int.class && carrier == byte.class) {
405             return filterValue(target, INT_TO_BYTE, BYTE_TO_UNSIGNED_INT);
406         } else if (adaptedType == int.class && carrier == short.class) {
407             return filterValue(target, INT_TO_SHORT, SHORT_TO_UNSIGNED_INT);
408         } else if (adaptedType == long.class && carrier == byte.class) {
409             return filterValue(target, LONG_TO_BYTE, BYTE_TO_UNSIGNED_LONG);
410         } else if (adaptedType == long.class && carrier == short.class) {
411             return filterValue(target, LONG_TO_SHORT, SHORT_TO_UNSIGNED_LONG);
412         } else if (adaptedType == long.class && carrier == int.class) {
413             return filterValue(target, LONG_TO_INT, INT_TO_UNSIGNED_LONG);
414         } else {
415             throw new InternalError("should not reach here");
416         }
417     }
