 56     static VarHandle makeFieldHandle(MemberName f, Class<?> refc, Class<?> type, boolean isWriteAllowedOnFinalFields) {
 57         if (!f.isStatic()) {
 58             long foffset = MethodHandleNatives.objectFieldOffset(f);
 59             if (!type.isPrimitive()) {
 60                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 61                        ? new VarHandleReferences.FieldInstanceReadOnly(refc, foffset, type)
 62                        : new VarHandleReferences.FieldInstanceReadWrite(refc, foffset, type));
 63             }
 64             else if (type == boolean.class) {
 65                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 66                        ? new VarHandleBooleans.FieldInstanceReadOnly(refc, foffset)
 67                        : new VarHandleBooleans.FieldInstanceReadWrite(refc, foffset));
 68             }
 69             else if (type == byte.class) {
 70                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 71                        ? new VarHandleBytes.FieldInstanceReadOnly(refc, foffset)
 72                        : new VarHandleBytes.FieldInstanceReadWrite(refc, foffset));
 73             }
 74             else if (type == short.class) {
 75                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 76                        ? new VarHandleShorts.FieldInstanceReadOnly(refc, foffset)
 77                        : new VarHandleShorts.FieldInstanceReadWrite(refc, foffset));
 78             }
 79             else if (type == char.class) {
 80                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 81                        ? new VarHandleChars.FieldInstanceReadOnly(refc, foffset)
 82                        : new VarHandleChars.FieldInstanceReadWrite(refc, foffset));
 83             }
 84             else if (type == int.class) {
 85                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 86                        ? new VarHandleInts.FieldInstanceReadOnly(refc, foffset)
 87                        : new VarHandleInts.FieldInstanceReadWrite(refc, foffset));
 88             }
 89             else if (type == long.class) {
 90                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 91                        ? new VarHandleLongs.FieldInstanceReadOnly(refc, foffset)
 92                        : new VarHandleLongs.FieldInstanceReadWrite(refc, foffset));
 93             }
 94             else if (type == float.class) {
 95                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
 96                        ? new VarHandleFloats.FieldInstanceReadOnly(refc, foffset)
 97                        : new VarHandleFloats.FieldInstanceReadWrite(refc, foffset));
 98             }
 99             else if (type == double.class) {
100                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
101                        ? new VarHandleDoubles.FieldInstanceReadOnly(refc, foffset)
102                        : new VarHandleDoubles.FieldInstanceReadWrite(refc, foffset));
103             }
104             else {
105                 throw new UnsupportedOperationException();
106             }
107         }
108         else {
109             // TODO This is not lazy on first invocation
110             // and might cause some circular initialization issues
111 
112             // Replace with something similar to direct method handles
113             // where a barrier is used then elided after use
114 
115             if (UNSAFE.shouldBeInitialized(refc))
116                 UNSAFE.ensureClassInitialized(refc);
117 
118             Object base = MethodHandleNatives.staticFieldBase(f);
119             long foffset = MethodHandleNatives.staticFieldOffset(f);
120             if (!type.isPrimitive()) {
121                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
122                        ? new VarHandleReferences.FieldStaticReadOnly(base, foffset, type)
123                        : new VarHandleReferences.FieldStaticReadWrite(base, foffset, type));
124             }
125             else if (type == boolean.class) {
126                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
127                        ? new VarHandleBooleans.FieldStaticReadOnly(base, foffset)
128                        : new VarHandleBooleans.FieldStaticReadWrite(base, foffset));
129             }
130             else if (type == byte.class) {
131                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
132                        ? new VarHandleBytes.FieldStaticReadOnly(base, foffset)
133                        : new VarHandleBytes.FieldStaticReadWrite(base, foffset));
134             }
135             else if (type == short.class) {
136                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
137                        ? new VarHandleShorts.FieldStaticReadOnly(base, foffset)
138                        : new VarHandleShorts.FieldStaticReadWrite(base, foffset));
139             }
140             else if (type == char.class) {
141                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
142                        ? new VarHandleChars.FieldStaticReadOnly(base, foffset)
143                        : new VarHandleChars.FieldStaticReadWrite(base, foffset));
144             }
145             else if (type == int.class) {
146                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
147                        ? new VarHandleInts.FieldStaticReadOnly(base, foffset)
148                        : new VarHandleInts.FieldStaticReadWrite(base, foffset));
149             }
150             else if (type == long.class) {
151                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
152                        ? new VarHandleLongs.FieldStaticReadOnly(base, foffset)
153                        : new VarHandleLongs.FieldStaticReadWrite(base, foffset));
154             }
155             else if (type == float.class) {
156                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
157                        ? new VarHandleFloats.FieldStaticReadOnly(base, foffset)
158                        : new VarHandleFloats.FieldStaticReadWrite(base, foffset));
159             }
160             else if (type == double.class) {
161                 return maybeAdapt(f.isFinal() && !isWriteAllowedOnFinalFields
162                        ? new VarHandleDoubles.FieldStaticReadOnly(base, foffset)
163                        : new VarHandleDoubles.FieldStaticReadWrite(base, foffset));
164             }
165             else {
166                 throw new UnsupportedOperationException();
167             }
168         }

171     // Required by instance field handles
172     static Field getFieldFromReceiverAndOffset(Class<?> receiverType,
173                                                long offset,
174                                                Class<?> fieldType) {
175         for (Field f : receiverType.getDeclaredFields()) {
176             if (Modifier.isStatic(f.getModifiers())) continue;
177 
178             if (offset == UNSAFE.objectFieldOffset(f)) {
179                 assert f.getType() == fieldType;
180                 return f;
181             }
182         }
183         throw new InternalError("Field not found at offset");
184     }

187     static Field getStaticFieldFromBaseAndOffset(Object base,
188                                                  long offset,
189                                                  Class<?> fieldType) {
190         // @@@ This is a little fragile assuming the base is the class
191         Class<?> receiverType = (Class<?>) base;
192         for (Field f : receiverType.getDeclaredFields()) {
193             if (!Modifier.isStatic(f.getModifiers())) continue;

195             if (offset == UNSAFE.staticFieldOffset(f)) {
196                 assert f.getType() == fieldType;
197                 return f;
198             }
199         }
200         throw new InternalError("Static field not found at offset");
201     }

203     static VarHandle makeArrayElementHandle(Class<?> arrayClass) {
204         if (!arrayClass.isArray())
205             throw new IllegalArgumentException("not an array: " + arrayClass);
206 
207         Class<?> componentType = arrayClass.getComponentType();
208 
209         int aoffset = UNSAFE.arrayBaseOffset(arrayClass);
210         int ascale = UNSAFE.arrayIndexScale(arrayClass);
211         int ashift = 31 - Integer.numberOfLeadingZeros(ascale);
212 
213         if (!componentType.isPrimitive()) {
214             return maybeAdapt(new VarHandleReferences.Array(aoffset, ashift, arrayClass));
215         }
216         else if (componentType == boolean.class) {
217             return maybeAdapt(new VarHandleBooleans.Array(aoffset, ashift));
218         }
219         else if (componentType == byte.class) {
220             return maybeAdapt(new VarHandleBytes.Array(aoffset, ashift));
221         }
222         else if (componentType == short.class) {
223             return maybeAdapt(new VarHandleShorts.Array(aoffset, ashift));
224         }
225         else if (componentType == char.class) {
226             return maybeAdapt(new VarHandleChars.Array(aoffset, ashift));
227         }
228         else if (componentType == int.class) {
229             return maybeAdapt(new VarHandleInts.Array(aoffset, ashift));
230         }
231         else if (componentType == long.class) {
232             return maybeAdapt(new VarHandleLongs.Array(aoffset, ashift));
233         }
234         else if (componentType == float.class) {
235             return maybeAdapt(new VarHandleFloats.Array(aoffset, ashift));
236         }
237         else if (componentType == double.class) {
238             return maybeAdapt(new VarHandleDoubles.Array(aoffset, ashift));
239         }
240         else {
241             throw new UnsupportedOperationException();
242         }
243     }

303     /**
304      * Creates a memory access VarHandle.
305      *
306      * Resulting VarHandle will take a memory address as first argument,
307      * and a certain number of coordinate {@code long} parameters, depending on the length
308      * of the {@code strides} argument array.
309      *
310      * Coordinates are multiplied with corresponding scale factors ({@code strides}) and added
311      * to a single fixed offset to compute an effective offset from the given MemoryAddress for the access.
312      *
313      * @param carrier the Java carrier type.
314      * @param alignmentMask alignment requirement to be checked upon access. In bytes. Expressed as a mask.
315      * @param byteOrder the byte order.
316      * @param offset a constant offset for the access.
317      * @param strides the scale factors with which to multiply given access coordinates.
318      * @return the created VarHandle.
319      */
320     static VarHandle makeMemoryAddressViewHandle(Class<?> carrier, long alignmentMask,
321                                                  ByteOrder byteOrder, long offset, long[] strides) {
322         if (!carrier.isPrimitive() || carrier == void.class || carrier == boolean.class) {
323             throw new IllegalArgumentException("Invalid carrier: " + carrier.getName());
324         }
325         long size = Wrapper.forPrimitiveType(carrier).bitWidth() / 8;
326         boolean be = byteOrder == ByteOrder.BIG_ENDIAN;
327 
328         Map<Integer, MethodHandle> carrierFactory = ADDRESS_FACTORIES.get(carrier);
329         MethodHandle fac = carrierFactory.computeIfAbsent(strides.length,
330                 dims -> new MemoryAccessVarHandleGenerator(carrier, dims)
331                             .generateHandleFactory());
332 
333         try {
334             return maybeAdapt((VarHandle)fac.invoke(be, size, offset, alignmentMask, strides));
335         } catch (Throwable ex) {
336             throw new IllegalStateException(ex);
337         }
338     }

340     private static void checkWidenable(Class<?> carrier) {
341         if (!(carrier == byte.class || carrier == short.class || carrier == int.class)) {
342             throw newIllegalArgumentException("illegal carrier", carrier.getSimpleName());
343         }
344     }

346     private static void checkNarrowable(Class<?> type) {
347         if (!(type == int.class || type == long.class)) {
348             throw newIllegalArgumentException("illegal adapter type", type.getSimpleName());
349         }

352     private static void checkTargetWiderThanCarrier(Class<?> carrier, Class<?> target) {
353         if (Wrapper.forPrimitiveType(target).bitWidth() <= Wrapper.forPrimitiveType(carrier).bitWidth()) {
354             throw newIllegalArgumentException(
355                     target.getSimpleName() + " is not wider than " + carrier.getSimpleName());
356         }

359     public static VarHandle asUnsigned(VarHandle target, final Class<?> adaptedType) {
360         Objects.requireNonNull(target);
361         Objects.requireNonNull(adaptedType);
362         final Class<?> carrier = target.varType();
363         checkWidenable(carrier);
364         checkNarrowable(adaptedType);
365         checkTargetWiderThanCarrier(carrier, adaptedType);
366 
367         if (adaptedType == int.class && carrier == byte.class) {
368             return MemoryHandleIntToUnsignedByte.varHandle(target);
369         } else if (adaptedType == int.class && carrier == short.class) {
370             return MemoryHandleIntToUnsignedShort.varHandle(target);
371         } else if (adaptedType == long.class && carrier == byte.class) {
372             return MemoryHandleLongToUnsignedByte.varHandle(target);
373         } else if (adaptedType == long.class && carrier == short.class) {
374             return MemoryHandleLongToUnsignedShort.varHandle(target);
375         } else if (adaptedType == long.class && carrier == int.class) {
376             return MemoryHandleLongToUnsignedInt.varHandle(target);
377         } else {
378             throw new InternalError("should not reach here");
379         }
380     }

382     private static VarHandle maybeAdapt(VarHandle target) {
383         if (!VAR_HANDLE_IDENTITY_ADAPT) return target;
384         target = filterValue(target,
385                         MethodHandles.identity(target.varType()), MethodHandles.identity(target.varType()));
386         MethodType mtype = target.accessModeType(VarHandle.AccessMode.GET).dropParameterTypes(0, 1);
387         for (int i = 0 ; i < mtype.parameterCount() ; i++) {
388             target = filterCoordinates(target, i, MethodHandles.identity(mtype.parameterType(i)));
389         }
390         return target;
391     }

393     public static VarHandle filterValue(VarHandle target, MethodHandle filterToTarget, MethodHandle filterFromTarget) {
394         Objects.nonNull(target);
395         Objects.nonNull(filterToTarget);
396         Objects.nonNull(filterFromTarget);
397         //check that from/to filters do not throw checked exceptions
398         noCheckedExceptions(filterToTarget);
399         noCheckedExceptions(filterFromTarget);
400 
401         //check that from/to filters have right signatures
402         if (filterFromTarget.type().parameterCount() != 1) {
403             throw newIllegalArgumentException("filterFromTarget filter type has wrong arity", filterFromTarget.type());
404         } else if (filterToTarget.type().parameterCount() != 1) {
405             throw newIllegalArgumentException("filterToTarget filter type has wrong arity", filterFromTarget.type());
406         } else if (filterFromTarget.type().parameterType(0) != filterToTarget.type().returnType() ||
407                 filterToTarget.type().parameterType(0) != filterFromTarget.type().returnType()) {
408             throw newIllegalArgumentException("filterFromTarget and filterToTarget filter types do not match", filterFromTarget.type(), filterToTarget.type());
409         } else if (target.varType() != filterFromTarget.type().parameterType(0)) {
410             throw newIllegalArgumentException("filterFromTarget filter type does not match target var handle type", filterFromTarget.type(), target.varType());
411         } else if (target.varType() != filterToTarget.type().returnType()) {
412             throw newIllegalArgumentException("filterFromTarget filter type does not match target var handle type", filterToTarget.type(), target.varType());
413         }
414 
415         return new IndirectVarHandle(target, filterFromTarget.type().returnType(), target.coordinateTypes().toArray(new Class<?>[0]),
416                 (mode, modeHandle) -> {
417                     int lastParameterPos = modeHandle.type().parameterCount() - 1;
418                     return switch (mode.at) {
419                         case GET -> MethodHandles.filterReturnValue(modeHandle, filterFromTarget);
420                         case SET -> MethodHandles.filterArgument(modeHandle, lastParameterPos, filterToTarget);
421                         case GET_AND_UPDATE -> {
422                             MethodHandle adapter = MethodHandles.filterReturnValue(modeHandle, filterFromTarget);
423                             yield MethodHandles.filterArgument(adapter, lastParameterPos, filterToTarget);
424                         }
425                         case COMPARE_AND_EXCHANGE -> {
426                             MethodHandle adapter = MethodHandles.filterReturnValue(modeHandle, filterFromTarget);
427                             adapter = MethodHandles.filterArgument(adapter, lastParameterPos, filterToTarget);
428                             yield MethodHandles.filterArgument(adapter, lastParameterPos - 1, filterToTarget);
429                         }
430                         case COMPARE_AND_SET -> {
431                             MethodHandle adapter = MethodHandles.filterArgument(modeHandle, lastParameterPos, filterToTarget);
432                             yield MethodHandles.filterArgument(adapter, lastParameterPos - 1, filterToTarget);
433                         }
434                     };
435                 });
436     }

438     public static VarHandle filterCoordinates(VarHandle target, int pos, MethodHandle... filters) {
439         Objects.nonNull(target);
440         Objects.nonNull(filters);
441 
442         List<Class<?>> targetCoordinates = target.coordinateTypes();
443         if (pos < 0 || pos >= targetCoordinates.size()) {
444             throw newIllegalArgumentException("Invalid position " + pos + " for coordinate types", targetCoordinates);
445         } else if (pos + filters.length > targetCoordinates.size()) {
446             throw new IllegalArgumentException("Too many filters");
447         }
448 
449         if (filters.length == 0) return target;
450 
451         List<Class<?>> newCoordinates = new ArrayList<>(targetCoordinates);
452         for (int i = 0 ; i < filters.length ; i++) {
453             noCheckedExceptions(filters[i]);
454             MethodType filterType = filters[i].type();
455             if (filterType.parameterCount() != 1) {
456                 throw newIllegalArgumentException("Invalid filter type " + filterType);
457             } else if (newCoordinates.get(pos + i) != filterType.returnType()) {
458                 throw newIllegalArgumentException("Invalid filter type " + filterType + " for coordinate type " + newCoordinates.get(i));
459             }
460             newCoordinates.set(pos + i, filters[i].type().parameterType(0));
461         }
462 
463         return new IndirectVarHandle(target, target.varType(), newCoordinates.toArray(new Class<?>[0]),
464                 (mode, modeHandle) -> MethodHandles.filterArguments(modeHandle, 1 + pos, filters));
465     }

467     public static VarHandle insertCoordinates(VarHandle target, int pos, Object... values) {
468         Objects.nonNull(target);
469         Objects.nonNull(values);
470 
471         List<Class<?>> targetCoordinates = target.coordinateTypes();
472         if (pos < 0 || pos >= targetCoordinates.size()) {
473             throw newIllegalArgumentException("Invalid position " + pos + " for coordinate types", targetCoordinates);
474         } else if (pos + values.length > targetCoordinates.size()) {
475             throw new IllegalArgumentException("Too many values");
476         }
477 
478         if (values.length == 0) return target;
479 
480         List<Class<?>> newCoordinates = new ArrayList<>(targetCoordinates);
481         for (int i = 0 ; i < values.length ; i++) {
482             Class<?> pt = newCoordinates.get(pos);
483             if (pt.isPrimitive()) {
484                 Wrapper w = Wrapper.forPrimitiveType(pt);
485                 w.convert(values[i], pt);
486             } else {
487                 pt.cast(values[i]);
488             }
489             newCoordinates.remove(pos);
490         }
491 
492         return new IndirectVarHandle(target, target.varType(), newCoordinates.toArray(new Class<?>[0]),
493                 (mode, modeHandle) -> MethodHandles.insertArguments(modeHandle, 1 + pos, values));
494     }

496     public static VarHandle permuteCoordinates(VarHandle target, List<Class<?>> newCoordinates, int... reorder) {
497         Objects.nonNull(target);
498         Objects.nonNull(newCoordinates);
499         Objects.nonNull(reorder);
500 
501         List<Class<?>> targetCoordinates = target.coordinateTypes();
502         MethodHandles.permuteArgumentChecks(reorder,
503                 MethodType.methodType(void.class, newCoordinates),
504                 MethodType.methodType(void.class, targetCoordinates));
505 
506         return new IndirectVarHandle(target, target.varType(), newCoordinates.toArray(new Class<?>[0]),
507                 (mode, modeHandle) ->
508                         MethodHandles.permuteArguments(modeHandle,
509                                 methodTypeFor(mode.at, modeHandle.type(), targetCoordinates, newCoordinates),
510                                 reorderArrayFor(mode.at, newCoordinates, reorder)));
511     }

513     private static int numTrailingArgs(VarHandle.AccessType at) {
514         return switch (at) {
515             case GET -> 0;
516             case GET_AND_UPDATE, SET -> 1;
517             case COMPARE_AND_SET, COMPARE_AND_EXCHANGE -> 2;
518         };
519     }
520 
521     private static int[] reorderArrayFor(VarHandle.AccessType at, List<Class<?>> newCoordinates, int[] reorder) {
522         int numTrailingArgs = numTrailingArgs(at);
523         int[] adjustedReorder = new int[reorder.length + 1 + numTrailingArgs];
524         adjustedReorder[0] = 0;
525         for (int i = 0 ; i < reorder.length ; i++) {
526             adjustedReorder[i + 1] = reorder[i] + 1;
527         }
528         for (int i = 0 ; i < numTrailingArgs ; i++) {
529             adjustedReorder[i + reorder.length + 1] = i + newCoordinates.size() + 1;
530         }
531         return adjustedReorder;
532     }
533 
534     private static MethodType methodTypeFor(VarHandle.AccessType at, MethodType oldType, List<Class<?>> oldCoordinates, List<Class<?>> newCoordinates) {
535         int numTrailingArgs = numTrailingArgs(at);
536         MethodType adjustedType = MethodType.methodType(oldType.returnType(), oldType.parameterType(0));
537         adjustedType = adjustedType.appendParameterTypes(newCoordinates);
538         for (int i = 0 ; i < numTrailingArgs ; i++) {
539             adjustedType = adjustedType.appendParameterTypes(oldType.parameterType(1 + oldCoordinates.size() + i));
540         }
541         return adjustedType;
542     }

544     public static VarHandle collectCoordinates(VarHandle target, int pos, MethodHandle filter) {
545         Objects.nonNull(target);
546         Objects.nonNull(filter);
547         noCheckedExceptions(filter);
548 
549         List<Class<?>> targetCoordinates = target.coordinateTypes();
550         if (pos < 0 || pos >= targetCoordinates.size()) {
551             throw newIllegalArgumentException("Invalid position " + pos + " for coordinate types", targetCoordinates);
552         } else if (filter.type().returnType() == void.class) {
553             throw newIllegalArgumentException("Invalid filter type " + filter.type() + " ; filter cannot be void");
554         } else if (filter.type().returnType() != targetCoordinates.get(pos)) {
555             throw newIllegalArgumentException("Invalid filter type " + filter.type() + " for coordinate type " + targetCoordinates.get(pos));
556         }
557 
558         List<Class<?>> newCoordinates = new ArrayList<>(targetCoordinates);
559         newCoordinates.remove(pos);
560         newCoordinates.addAll(pos, filter.type().parameterList());
561 
562         return new IndirectVarHandle(target, target.varType(), newCoordinates.toArray(new Class<?>[0]),
563                 (mode, modeHandle) -> MethodHandles.collectArguments(modeHandle, 1 + pos, filter));
564     }

566     public static VarHandle dropCoordinates(VarHandle target, int pos, Class<?>... valueTypes) {
567         Objects.nonNull(target);
568         Objects.nonNull(valueTypes);
569 
570         List<Class<?>> targetCoordinates = target.coordinateTypes();
571         if (pos < 0 || pos > targetCoordinates.size()) {
572             throw newIllegalArgumentException("Invalid position " + pos + " for coordinate types", targetCoordinates);
573         }
574 
575         if (valueTypes.length == 0) return target;
576 
577         List<Class<?>> newCoordinates = new ArrayList<>(targetCoordinates);
578         newCoordinates.addAll(pos, List.of(valueTypes));
579 
580         return new IndirectVarHandle(target, target.varType(), newCoordinates.toArray(new Class<?>[0]),
581                 (mode, modeHandle) -> MethodHandles.dropArguments(modeHandle, 1 + pos, valueTypes));
582     }

584     private static void noCheckedExceptions(MethodHandle handle) {
585         if (handle instanceof DirectMethodHandle) {
586             DirectMethodHandle directHandle = (DirectMethodHandle)handle;
587             MethodHandleInfo info = MethodHandles.Lookup.IMPL_LOOKUP.revealDirect(directHandle);
588             Class<?>[] exceptionTypes = switch (info.getReferenceKind()) {
589                 case MethodHandleInfo.REF_invokeInterface, MethodHandleInfo.REF_invokeSpecial,
590                         MethodHandleInfo.REF_invokeStatic, MethodHandleInfo.REF_invokeVirtual ->
591                         info.reflectAs(Method.class, MethodHandles.Lookup.IMPL_LOOKUP).getExceptionTypes();
592                 case MethodHandleInfo.REF_newInvokeSpecial ->
593                         info.reflectAs(Constructor.class, MethodHandles.Lookup.IMPL_LOOKUP).getExceptionTypes();
594                 case MethodHandleInfo.REF_getField, MethodHandleInfo.REF_getStatic,
595                         MethodHandleInfo.REF_putField, MethodHandleInfo.REF_putStatic -> null;
596                 default -> throw new AssertionError("Cannot get here");
597             };
598             if (exceptionTypes != null) {
599                 if (Stream.of(exceptionTypes).anyMatch(VarHandles::isCheckedException)) {
600                     throw newIllegalArgumentException("Cannot adapt a var handle with a method handle which throws checked exceptions");
601                 }
602             }
603         } else if (handle instanceof DelegatingMethodHandle) {
604             noCheckedExceptions(((DelegatingMethodHandle)handle).getTarget());
605         } else {
606             //bound
607             BoundMethodHandle boundHandle = (BoundMethodHandle)handle;
608             for (int i = 0 ; i < boundHandle.fieldCount() ; i++) {
609                 Object arg = boundHandle.arg(i);
610                 if (arg instanceof MethodHandle){
611                     noCheckedExceptions((MethodHandle) arg);
612                 }
613             }
614         }
615     }

618         return Throwable.class.isAssignableFrom(clazz) &&
619                 !RuntimeException.class.isAssignableFrom(clazz) &&
620                 !Error.class.isAssignableFrom(clazz);
621     }
