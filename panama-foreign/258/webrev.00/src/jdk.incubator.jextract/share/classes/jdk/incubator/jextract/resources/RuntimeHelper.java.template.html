131         private Object invoke(Object[] args) throws Throwable {
132             // one trailing Object[]
133             int nNamedArgs = function.argumentLayouts().size();
134             assert(args.length == nNamedArgs + 1);
135             // The last argument is the array of vararg collector
136             Object[] unnamedArgs = (Object[]) args[args.length - 1];
137 
138             int argsCount = nNamedArgs + unnamedArgs.length;
139             Class<?>[] argTypes = new Class<?>[argsCount];
140             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];
141 
142             int pos = 0;
143             for (pos = 0; pos < nNamedArgs; pos++) {
144                 argTypes[pos] = varargs.parameterType(pos);
145                 argLayouts[pos] = function.argumentLayouts().get(pos);
146             }
147 
148             assert pos == nNamedArgs;
149             for (Object o: unnamedArgs) {
150                 argTypes[pos] = normalize(o.getClass());
151                 argLayouts[pos] = variadicLayout(argTypes[pos]);
152                 pos++;
153             }
154             assert pos == argsCount;
155 
156             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);
157             FunctionDescriptor f = (function.returnLayout().isEmpty()) ?
158                     FunctionDescriptor.ofVoid(argLayouts) :
159                     FunctionDescriptor.of(function.returnLayout().get(), argLayouts);
160             MethodHandle mh = ABI.downcallHandle(symbol, mt, f);
161             // flatten argument list so that it can be passed to an asSpreader MH
162             Object[] allArgs = new Object[nNamedArgs + unnamedArgs.length];
163             System.arraycopy(args, 0, allArgs, 0, nNamedArgs);
164             System.arraycopy(unnamedArgs, 0, allArgs, nNamedArgs, unnamedArgs.length);
165 
166             return mh.asSpreader(Object[].class, argsCount).invoke(allArgs);
167         }
