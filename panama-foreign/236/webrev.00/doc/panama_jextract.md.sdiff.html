583 public class SqliteMain {
584    private static MemoryAddress getPointer(MemoryAddress addr) {
585        return getPointer(addr, 0);
586    }
587 
588    private static MemoryAddress getPointer(MemoryAddress addr, int element) {
589        return MemoryAccess.getAddress(addr, element*C_POINTER.byteSize());
590    }
591 
592    public static void main(String[] args) throws Exception {
593         try (var scope = NativeScope.unboundedScope()) {
594             // char** errMsgPtrPtr;
595             var errMsgPtrPtr = scope.allocate(C_POINTER);
596             MemoryAccess.setAddress(errMsgPtrPtr, 0, NULL);
597 
598             // sqlite3** dbPtrPtr;
599             var dbPtrPtr = scope.allocate(C_POINTER);
600             MemoryAccess.setAddress(dbPtrPtr, 0, NULL);
601 
602             int rc = sqlite3_open(toCString("employee.db",scope), dbPtrPtr);
603             if (rc != 0) {
604                 System.err.println("sqlite3_open failed: " + rc);
605                 return;
606             } else {
607                 System.out.println("employee db opened");
608             }
609 
610             // sqlite3* dbPtr;
611             var dbPtr = getPointer(dbPtrPtr);
612 
613             // create a new table
614             var sql = toCString(
615                 "CREATE TABLE EMPLOYEE ("  +
616                 "  ID INT PRIMARY KEY NOT NULL," +
617                 "  NAME TEXT NOT NULL,"    +
618                 "  SALARY REAL NOT NULL )", scope);
619 
620             rc = sqlite3_exec(dbPtr, sql, NULL, NULL, errMsgPtrPtr);
621 
622             if (rc != 0) {
623                 System.err.println("sqlite3_exec failed: " + rc);
624                 System.err.println("SQL error: " + toJavaStringRestricted(getPointer(errMsgPtrPtr)));
625                 sqlite3_free(getPointer(errMsgPtrPtr));
626             } else {
627                 System.out.println("employee table created");
628             }
629 
630             // insert two rows
631             sql = toCString(
632                 "INSERT INTO EMPLOYEE (ID,NAME,SALARY) " +
633                     "VALUES (134, 'Xyz', 200000.0); " +
634                 "INSERT INTO EMPLOYEE (ID,NAME,SALARY) " +
635                     "VALUES (333, 'Abc', 100000.0);", scope
636             );
637             rc = sqlite3_exec(dbPtr, sql, NULL, NULL, errMsgPtrPtr);
638 
639             if (rc != 0) {
640                 System.err.println("sqlite3_exec failed: " + rc);
641                 System.err.println("SQL error: " + toJavaStringRestricted(getPointer(errMsgPtrPtr)));
642                 sqlite3_free(getPointer(errMsgPtrPtr));
643             } else {
644                 System.out.println("rows inserted");
645             }
646 
647             int[] rowNum = new int[1];
648             // callback to print rows from SELECT query
649             var callback = sqlite3_exec$callback.allocate((a, argc, argv, columnNames) -> {
650                 System.out.println("Row num: " + rowNum[0]++);
651                 System.out.println("numColumns = " + argc);
652                 argv = asArrayRestricted(argv, C_POINTER, argc);
653                 columnNames = asArrayRestricted(columnNames, C_POINTER, argc);
654                 for (int i = 0; i < argc; i++) {
655                      String name = toJavaStringRestricted(getPointer(columnNames, i));
656                      String value = toJavaStringRestricted(getPointer(argv, i));
657                      System.out.printf("%s = %s\n", name, value);
658                 }
659                 return 0;
660             }, scope);
661 
662             // select query
663             sql = toCString("SELECT * FROM EMPLOYEE", scope);
664             rc = sqlite3_exec(dbPtr, sql, callback, NULL, errMsgPtrPtr);
665 
666             if (rc != 0) {
667                 System.err.println("sqlite3_exec failed: " + rc);
668                 System.err.println("SQL error: " + toJavaStringRestricted(getPointer(errMsgPtrPtr)));
669                 sqlite3_free(getPointer(errMsgPtrPtr));
670             } else {
671                 System.out.println("done");
672             }
673

675         }
676     }
