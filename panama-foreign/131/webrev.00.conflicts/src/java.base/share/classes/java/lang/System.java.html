 314      * Sets the system-wide security manager.
 315      *
 316      * If there is a security manager already installed, this method first
 317      * calls the security manager's {@code checkPermission} method
 318      * with a {@code RuntimePermission("setSecurityManager")}
 319      * permission to ensure it's ok to replace the existing
 320      * security manager.
 321      * This may result in throwing a {@code SecurityException}.
 322      *
 323      * <p> Otherwise, the argument is established as the current
 324      * security manager. If the argument is {@code null} and no
 325      * security manager has been established, then no action is taken and
 326      * the method simply returns.
 327      *
 328      * @implNote In the JDK implementation, if the Java virtual machine is
 329      * started with the system property {@code java.security.manager} set to
 330      * the special token "{@code disallow}" then the {@code setSecurityManager}
 331      * method cannot be used to set a security manager.
 332      *
 333      * @param  sm the security manager or {@code null}
 334      * @throws SecurityException
 335      *         if the security manager has already been set and its {@code
 336      *         checkPermission} method doesn't allow it to be replaced
 337      * @throws UnsupportedOperationException
 338      *         if {@code sm} is non-null and a security manager is not allowed
 339      *         to be set dynamically
 340      * @see #getSecurityManager
 341      * @see SecurityManager#checkPermission
 342      * @see java.lang.RuntimePermission
 343      */
 344     public static void setSecurityManager(SecurityManager sm) {
 345         if (allowSecurityManager()) {
 346             if (security == null) {
 347                 // ensure image reader is initialized
 348                 Object.class.getResource("java/lang/ANY");
 349                 // ensure the default file system is initialized
 350                 DefaultFileSystemProvider.theFileSystem();
 351             }
 352             if (sm != null) {
 353                 try {
 354                     // pre-populates the SecurityManager.packageAccess cache
 355                     // to avoid recursive permission checking issues with custom
 356                     // SecurityManager implementations
 357                     sm.checkPackageAccess("java.lang");
 358                 } catch (Exception e) {
 359                     // no-op
 360                 }
 361             }
 362             setSecurityManager0(sm);
 363         } else {
 364             // security manager not allowed
 365             if (sm != null) {
 366                 throw new UnsupportedOperationException(
 367                     "Runtime configured to disallow security manager");
 368             }
 369         }
 370     }

 373     void setSecurityManager0(final SecurityManager s) {
 374         SecurityManager sm = getSecurityManager();
 375         if (sm != null) {
 376             // ask the currently installed security manager if we
 377             // can replace it.
 378             sm.checkPermission(new RuntimePermission("setSecurityManager"));
 379         }

 381         if ((s != null) && (s.getClass().getClassLoader() != null)) {
 382             // New security manager class is not on bootstrap classpath.
 383             // Force policy to get initialized before we install the new
 384             // security manager, in order to prevent infinite loops when
 385             // trying to initialize the policy (which usually involves
 386             // accessing some security and/or system properties, which in turn
 387             // calls the installed security manager's checkPermission method
 388             // which will loop infinitely if there is a non-system class

 391                 public Object run() {
 392                     s.getClass().getProtectionDomain().implies
 393                         (SecurityConstants.ALL_PERMISSION);
 394                     return null;
 395                 }
 396             });
 397         }

1416     /**
1417      * The {@code LoggerFinder} service is responsible for creating, managing,
1418      * and configuring loggers to the underlying framework it uses.
1419      *
1420      * A logger finder is a concrete implementation of this class that has a
1421      * zero-argument constructor and implements the abstract methods defined
1422      * by this class.
1423      * The loggers returned from a logger finder are capable of routing log
1424      * messages to the logging backend this provider supports.
1425      * A given invocation of the Java Runtime maintains a single
1426      * system-wide LoggerFinder instance that is loaded as follows:
1427      * <ul>
1428      *    <li>First it finds any custom {@code LoggerFinder} provider
1429      *        using the {@link java.util.ServiceLoader} facility with the
1430      *        {@linkplain ClassLoader#getSystemClassLoader() system class
1431      *        loader}.</li>
1432      *    <li>If no {@code LoggerFinder} provider is found, the system default
1433      *        {@code LoggerFinder} implementation will be used.</li>
1434      * </ul>
1435      * <p>
1436      * An application can replace the logging backend
1437      * <i>even when the java.logging module is present</i>, by simply providing
1438      * and declaring an implementation of the {@link LoggerFinder} service.
1439      * <p>
1440      * <b>Default Implementation</b>
1441      * <p>
1442      * The system default {@code LoggerFinder} implementation uses
1443      * {@code java.util.logging} as the backend framework when the
1444      * {@code java.logging} module is present.
1445      * It returns a {@linkplain System.Logger logger} instance
1446      * that will route log messages to a {@link java.util.logging.Logger
1447      * java.util.logging.Logger}. Otherwise, if {@code java.logging} is not
1448      * present, the default implementation will return a simple logger
1449      * instance that will route log messages of {@code INFO} level and above to
1450      * the console ({@code System.err}).
1451      * <p>
1452      * <b>Logging Configuration</b>
1453      * <p>
1454      * {@linkplain Logger Logger} instances obtained from the
1455      * {@code LoggerFinder} factory methods are not directly configurable by
1456      * the application. Configuration is the responsibility of the underlying
1457      * logging backend, and usually requires using APIs specific to that backend.
1458      * <p>For the default {@code LoggerFinder} implementation
1459      * using {@code java.util.logging} as its backend, refer to
1460      * {@link java.util.logging java.util.logging} for logging configuration.
1461      * For the default {@code LoggerFinder} implementation returning simple loggers
1462      * when the {@code java.logging} module is absent, the configuration
1463      * is implementation dependent.
1464      * <p>
1465      * Usually an application that uses a logging framework will log messages
1466      * through a logger facade defined (or supported) by that framework.
1467      * Applications that wish to use an external framework should log
1468      * through the facade associated with that framework.
1469      * <p>
1470      * A system class that needs to log messages will typically obtain
1471      * a {@link System.Logger} instance to route messages to the logging
1472      * framework selected by the application.
1473      * <p>
1474      * Libraries and classes that only need loggers to produce log messages
1475      * should not attempt to configure loggers by themselves, as that
1476      * would make them dependent from a specific implementation of the
1477      * {@code LoggerFinder} service.
1478      * <p>
1479      * In addition, when a security manager is present, loggers provided to
1480      * system classes should not be directly configurable through the logging
1481      * backend without requiring permissions.
1482      * <br>
1483      * It is the responsibility of the provider of
1484      * the concrete {@code LoggerFinder} implementation to ensure that
1485      * these loggers are not configured by untrusted code without proper
1486      * permission checks, as configuration performed on such loggers usually
1487      * affects all applications in the same Java Runtime.

1489      * <b>Message Levels and Mapping to backend levels</b>
1490      * <p>
1491      * A logger finder is responsible for mapping from a {@code
1492      * System.Logger.Level} to a level supported by the logging backend it uses.
1493      * <br>The default LoggerFinder using {@code java.util.logging} as the backend
1494      * maps {@code System.Logger} levels to
1495      * {@linkplain java.util.logging.Level java.util.logging} levels
1496      * of corresponding severity - as described in {@link Logger.Level
1497      * Logger.Level}.
1498      *
1499      * @see java.lang.System

1962     private static Properties createProperties(Map<String, String> initialProps) {
1963         Properties properties = new Properties(initialProps.size());
1964         for (var entry : initialProps.entrySet()) {
1965             String prop = entry.getKey();
1966             switch (prop) {

1968                 case "sun.nio.MaxDirectMemorySize":
1969                 case "sun.nio.PageAlignDirectMemory":

1987     private static void initPhase1() {

1999         // from the properties set by the VM.  If the properties are for
2000         // internal implementation use only, these properties should be

2072     /*
2073      * Invoked by VM.  Phase 3 is the final system initialization:
2074      * 1. set security manager
2075      * 2. set system class loader
2076      * 3. set TCCL
2077      *
2078      * This method must be called after the module system initialization.
2079      * The security manager and system class loader may be a custom class from
2080      * the application classpath or modulepath.
2081      */
2082     private static void initPhase3() {
2083         String smProp = System.getProperty("java.security.manager");
2084         if (smProp != null) {
2085             switch (smProp) {
2086                 case "disallow":
2087                     allowSecurityManager = NEVER;
2088                     break;
2089                 case "allow":
2090                     allowSecurityManager = MAYBE;
2091                     break;
2092                 case "":
2093                 case "default":
2094                     setSecurityManager(new SecurityManager());
2095                     allowSecurityManager = MAYBE;
2096                     break;
2097                 default:
2098                     try {
2099                         ClassLoader cl = ClassLoader.getBuiltinAppClassLoader();
2100                         Class<?> c = Class.forName(smProp, false, cl);
2101                         Constructor<?> ctor = c.getConstructor();
2102                         // Must be a public subclass of SecurityManager with
2103                         // a public no-arg constructor
2104                         if (!SecurityManager.class.isAssignableFrom(c) ||
2105                             !Modifier.isPublic(c.getModifiers()) ||
2106                             !Modifier.isPublic(ctor.getModifiers())) {
2107                             throw new Error("Could not create SecurityManager: "
2108                                              + ctor.toString());
2109                         }
2110                         // custom security manager may be in non-exported package
2111                         ctor.setAccessible(true);
2112                         SecurityManager sm = (SecurityManager) ctor.newInstance();
2113                         setSecurityManager(sm);
2114                     } catch (Exception e) {
2115                         throw new InternalError("Could not create SecurityManager", e);
2116                     }
2117                     allowSecurityManager = MAYBE;
2118             }
2119         } else {
2120             allowSecurityManager = MAYBE;
2121         }
2122 
2123         // initializing the system class loader
2124         VM.initLevel(3);
2125 
2126         // system class loader initialized
2127         ClassLoader scl = ClassLoader.initSystemClassLoader();
2128 
2129         // set TCCL
2130         Thread.currentThread().setContextClassLoader(scl);
2131 
2132         // system is fully initialized
2133         VM.initLevel(4);
2134     }
