 24 package jdk.internal.foreign.abi.aarch64;
 25 
 26 import jdk.internal.foreign.abi.ABIDescriptor;
 27 import jdk.internal.foreign.abi.Architecture;
 28 import jdk.internal.foreign.abi.VMStorage;
 29 
 30 public class AArch64Architecture implements Architecture {
 31     public static final Architecture INSTANCE = new AArch64Architecture();
 32 
 33     private static final int INTEGER_REG_SIZE = 8;
 34     private static final int VECTOR_REG_SIZE = 16;
 35     private static final int STACK_SLOT_SIZE = 8;
 36 
 37     @Override
 38     public boolean isStackType(int cls) {
 39         return cls == StorageClasses.STACK;
 40     }
 41 
 42     @Override
 43     public int typeSize(int cls) {
 44         switch (cls) {
 45             case StorageClasses.INTEGER: return INTEGER_REG_SIZE;
 46             case StorageClasses.VECTOR: return VECTOR_REG_SIZE;
 47             case StorageClasses.STACK: return STACK_SLOT_SIZE;
 48         }
 49 
 50         throw new IllegalArgumentException("Invalid Storage Class: " + cls);
 51     }
 52 
 53     @Override
 54     public int stackType() {
 55         return StorageClasses.STACK;
 56     }
 57 
 58     public interface StorageClasses {
 59         int INTEGER = 0;
 60         int VECTOR = 1;
 61         int STACK = 3;
 62     }

130     private static VMStorage integerRegister(int index) {
131         return new VMStorage(StorageClasses.INTEGER, index, "r" + index);
132     }
133 
134     private static VMStorage vectorRegister(int index) {
135         return new VMStorage(StorageClasses.VECTOR, index, "v" + index);
136     }
137 
138     public static VMStorage stackStorage(int index) {
139         return new VMStorage(StorageClasses.STACK, index, "Stack@" + index);
140     }

142     public static ABIDescriptor abiFor(VMStorage[] inputIntRegs,
143                                        VMStorage[] inputVectorRegs,
144                                        VMStorage[] outputIntRegs,
145                                        VMStorage[] outputVectorRegs,
146                                        VMStorage[] volatileIntRegs,
147                                        VMStorage[] volatileVectorRegs,
148                                        int stackAlignment,
149                                        int shadowSpace) {
150         return new ABIDescriptor(
151             INSTANCE,
152             new VMStorage[][] {
153                 inputIntRegs,
154                 inputVectorRegs,
155             },
156             new VMStorage[][] {
157                 outputIntRegs,
158                 outputVectorRegs,
159             },
160             new VMStorage[][] {
161                 volatileIntRegs,
162                 volatileVectorRegs,
163             },
164             stackAlignment,
165             shadowSpace
166         );
167     }
