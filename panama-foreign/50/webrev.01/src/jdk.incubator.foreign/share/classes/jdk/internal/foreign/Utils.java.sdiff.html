 91     public static MemorySegment makeNativeSegment(long bytesSize, long alignmentBytes) {
 92         long alignedSize = bytesSize;
 93 
 94         if (alignmentBytes > MAX_ALIGN) {
 95             alignedSize = bytesSize + (alignmentBytes - 1);
 96         }
 97 
 98         long buf = unsafe.allocateMemory(alignedSize);
 99         if (!skipZeroMemory) {
100             unsafe.setMemory(buf, alignedSize, (byte)0);
101         }
102         long alignedBuf = Utils.alignUp(buf, alignmentBytes);
103         MemoryScope scope = new MemoryScope(null, () -> unsafe.freeMemory(buf));
104         MemorySegment segment = new MemorySegmentImpl(buf, null, alignedSize, Thread.currentThread(), scope);
105         if (alignedBuf != buf) {
106             long delta = alignedBuf - buf;
107             segment = segment.asSlice(delta, bytesSize);
108         }
109         return segment;
110     }

139     public static void checkCarrier(Class<?> carrier) {
140         if (carrier == void.class || carrier == boolean.class ||
141                 (!carrier.isPrimitive() && !isAddress(carrier))) {
142             throw new IllegalArgumentException("Illegal carrier: " + carrier.getSimpleName());
143         }

146     public static long carrierSize(Class<?> carrier) {
147         if (isAddress(carrier)) {
148             return POINTER_SIZE;
149         }
150         long bitsAlignment = Math.max(8, Wrapper.forPrimitiveType(carrier).bitWidth());
151         return Utils.bitsToBytesOrThrow(bitsAlignment, IllegalStateException::new);
152     }

264         // This adaptation is required, otherwise the memory access var handle will have type MemoryAddressProxy,
