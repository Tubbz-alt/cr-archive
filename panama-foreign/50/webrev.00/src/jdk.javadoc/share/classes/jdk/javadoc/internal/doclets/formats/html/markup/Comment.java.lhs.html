26 package jdk.javadoc.internal.doclets.formats.html.markup;
27 
28 import java.io.IOException;
29 import java.io.Writer;

30 
31 import jdk.javadoc.internal.doclets.toolkit.Content;
32 import jdk.javadoc.internal.doclets.toolkit.util.DocletConstants;
33 
34 /**
35  * Class for generating a comment for HTML pages of javadoc output.
36  *
37  *  <p><b>This is NOT part of any supported API.
38  *  If you write code that depends on this, you do so at your own risk.
39  *  This code and its internal interfaces are subject to change or
40  *  deletion without notice.</b>
41  */
42 public class Comment extends Content {
43 
44     private String commentText;
45 
46     /**
47      * Constructor to construct a Comment object.
48      *
49      * @param comment comment text for the comment
50      */
51     public Comment(String comment) {
52         commentText = nullCheck(comment);
53     }
54 
55     /**
56      * This method is not supported by the class.
57      *
58      * @param content content that needs to be added
59      * @throws UnsupportedOperationException always
60      */
61     @Override
62     public void add(Content content) {
63         throw new UnsupportedOperationException();
64     }
65 
66     /**
67      * This method is not supported by the class.
68      *
69      * @param stringContent string content that needs to be added
70      * @throws UnsupportedOperationException always
71      */
72     @Override
73     public void add(CharSequence stringContent) {
74         throw new UnsupportedOperationException();
75     }
76 
77     @Override
78     public boolean isEmpty() {
79         return commentText.isEmpty();
80     }
81 
82     @Override
83     public boolean write(Writer out, boolean atNewline) throws IOException {
84         if (!atNewline)
85             out.write(DocletConstants.NL);
86         out.write("<!-- ");
87         out.write(commentText);
88         out.write(" -->" + DocletConstants.NL);
89         return true;
90     }
91 }
