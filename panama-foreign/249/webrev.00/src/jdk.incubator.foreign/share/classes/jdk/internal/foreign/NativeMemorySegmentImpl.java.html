 27 package jdk.internal.foreign;
 28 
 29 import jdk.incubator.foreign.MemoryAddress;
 30 import jdk.incubator.foreign.MemorySegment;
 31 import jdk.internal.access.JavaNioAccess;
 32 import jdk.internal.access.SharedSecrets;
 33 import jdk.internal.misc.Unsafe;
 34 import jdk.internal.vm.annotation.ForceInline;
 35 import sun.security.action.GetBooleanAction;
 36 
 37 import java.nio.ByteBuffer;
 38 
 39 /**
 40  * Implementation for native memory segments. A native memory segment is essentially a wrapper around
 41  * a native long address.
 42  */
 43 public class NativeMemorySegmentImpl extends AbstractMemorySegmentImpl {
 44 
 45     public static final MemorySegment EVERYTHING = NativeMemorySegmentImpl.makeNativeSegmentUnchecked(MemoryAddress.NULL,
 46             Long.MAX_VALUE, null, null, null).withAccessModes(READ | WRITE);
 47 
 48     private static final Unsafe unsafe = Unsafe.getUnsafe();
 49 
 50     // The maximum alignment supported by malloc - typically 16 on
 51     // 64-bit platforms and 8 on 32-bit platforms.
 52     private final static long MAX_ALIGN = Unsafe.ADDRESS_SIZE == 4 ? 8 : 16;
 53 
 54     private static final boolean skipZeroMemory = GetBooleanAction.privilegedGetProperty("jdk.internal.foreign.skipZeroMemory");
 55 
 56     final long min;
 57 
 58     @ForceInline
 59     NativeMemorySegmentImpl(long min, long length, int mask, MemoryScope scope) {
 60         super(length, mask, scope);
 61         this.min = min;
 62     }
 63 
 64     @Override
 65     NativeMemorySegmentImpl dup(long offset, long size, int mask, MemoryScope scope) {
 66         return new NativeMemorySegmentImpl(min + offset, size, mask, scope);
 67     }
 68 
 69     @Override
 70     ByteBuffer makeByteBuffer() {
 71         JavaNioAccess nioAccess = SharedSecrets.getJavaNioAccess();
 72         return nioAccess.newDirectByteBuffer(min(), (int) this.length, null, this);
 73     }
 74 
 75     @Override
 76     long min() {
 77         return min;
 78     }
 79 
 80     @Override
 81     Object base() {
 82         return null;
 83     }
 84 
 85     // factories
 86 
 87     public static MemorySegment makeNativeSegment(long bytesSize, long alignmentBytes) {
 88         long alignedSize = bytesSize;
 89 
 90         if (alignmentBytes > MAX_ALIGN) {
 91             alignedSize = bytesSize + (alignmentBytes - 1);
 92         }
 93 
 94         long buf = unsafe.allocateMemory(alignedSize);
 95         if (!skipZeroMemory) {
 96             unsafe.setMemory(buf, alignedSize, (byte)0);
 97         }
 98         long alignedBuf = Utils.alignUp(buf, alignmentBytes);
 99         MemoryScope scope = MemoryScope.create(null, () -> unsafe.freeMemory(buf));
100         MemorySegment segment = new NativeMemorySegmentImpl(buf, alignedSize,
101                 defaultAccessModes(alignedSize), scope);
102         if (alignedSize != bytesSize) {
103             long delta = alignedBuf - buf;
104             segment = segment.asSlice(delta, bytesSize);
105         }
106         return segment;
107     }
108 
109     public static MemorySegment makeNativeSegmentUnchecked(MemoryAddress min, long bytesSize, Thread owner, Runnable cleanup, Object attachment) {
110         MemoryScope scope = MemoryScope.createUnchecked(owner, attachment, cleanup);
111         return new NativeMemorySegmentImpl(min.toRawLongValue(), bytesSize, defaultAccessModes(bytesSize), scope);
112     }
113 }
