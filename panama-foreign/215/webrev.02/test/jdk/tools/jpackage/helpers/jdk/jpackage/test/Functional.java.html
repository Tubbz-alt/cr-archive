 23 package jdk.jpackage.test;
 24 
 25 import java.lang.reflect.InvocationTargetException;
 26 import java.util.function.BiConsumer;
 27 import java.util.function.Consumer;
 28 import java.util.function.Function;
 29 import java.util.function.Predicate;
 30 import java.util.function.Supplier;
 31 
 32 
 33 public class Functional {
 34     @FunctionalInterface
 35     public interface ThrowingConsumer<T> {
 36         void accept(T t) throws Throwable;
 37 
 38         public static <T> Consumer<T> toConsumer(ThrowingConsumer<T> v) {
 39             return o -> {
 40                 try {
 41                     v.accept(o);
 42                 } catch (Throwable ex) {
 43                     rethrowUnchecked(ex);
 44                 }
 45             };
 46         }
 47     }
 48 
 49     @FunctionalInterface
 50     public interface ThrowingBiConsumer<T, U> {
 51         void accept(T t, U u) throws Throwable;
 52 
 53         public static <T, U> BiConsumer<T, U> toBiConsumer(ThrowingBiConsumer<T, U> v) {
 54             return (t, u) -> {
 55                 try {
 56                     v.accept(t, u);
 57                 } catch (Throwable ex) {
 58                     rethrowUnchecked(ex);
 59                 }
 60             };
 61         }
 62     }
 63 
 64     @FunctionalInterface
 65     public interface ThrowingSupplier<T> {
 66         T get() throws Throwable;
 67 
 68         public static <T> Supplier<T> toSupplier(ThrowingSupplier<T> v) {
 69             return () -> {
 70                 try {
 71                     return v.get();
 72                 } catch (Throwable ex) {
 73                     rethrowUnchecked(ex);
 74                 }
 75                 // Unreachable
 76                 return null;
 77             };
 78         }
 79     }
 80 
 81     @FunctionalInterface
 82     public interface ThrowingFunction<T, R> {
 83         R apply(T t) throws Throwable;
 84 
 85         public static <T, R> Function<T, R> toFunction(ThrowingFunction<T, R> v) {
 86             return (t) -> {
 87                 try {
 88                     return v.apply(t);
 89                 } catch (Throwable ex) {
 90                     rethrowUnchecked(ex);
 91                 }
 92                 // Unreachable
 93                 return null;
 94             };
 95         }
 96     }
 97 
 98     @FunctionalInterface
 99     public interface ThrowingRunnable {
100         void run() throws Throwable;
101 
102         public static Runnable toRunnable(ThrowingRunnable v) {
103             return () -> {
104                 try {
105                     v.run();
106                 } catch (Throwable ex) {
107                     rethrowUnchecked(ex);
108                 }
109             };
110         }
111     }
112 
113     public static <T> Supplier<T> identity(Supplier<T> v) {
114         return v;
115     }
116 
117     public static <T> Consumer<T> identity(Consumer<T> v) {
118         return v;
119     }
120 
121     public static <T, U> BiConsumer<T, U> identity(BiConsumer<T, U> v) {
122         return v;
123     }
124 
125     public static Runnable identity(Runnable v) {
126         return v;
127     }
128 
129     public static <T, R> Function<T, R> identity(Function<T, R> v) {
130         return v;
131     }
132 
133     public static <T, R> Function<T, R> identityFunction(Function<T, R> v) {
134         return v;
135     }
136 
137     public static <T> Predicate<T> identity(Predicate<T> v) {
138         return v;
139     }
140 
141     public static <T> Predicate<T> identityPredicate(Predicate<T> v) {
142         return v;
143     }
144 
145     public static class ExceptionBox extends RuntimeException {
146         public ExceptionBox(Throwable throwable) {
147             super(throwable);
148         }
149     }
150 
151     @SuppressWarnings("unchecked")
152     public static void rethrowUnchecked(Throwable throwable) throws ExceptionBox {
153         if (throwable instanceof ExceptionBox) {
154             throw (ExceptionBox)throwable;
155         }
156 
157         if (throwable instanceof InvocationTargetException) {
158             new ExceptionBox(throwable.getCause());
159         }
160 
161         throw new ExceptionBox(throwable);
162     }
163 }
