27 /**
28  * Simple platform specific packaging test. Output of the test should be
29  * simplepackagetest*.* package bundle.
30  *
31  * Windows:
32  *
33  * The installer should not have license text. It should not have an option
34  * to change the default installation directory.
35  * Test application should be installed in %ProgramFiles%\SimplePackageTest directory.
36  * Installer should install test app for all users (machine wide).
37  * Installer should not create any shortcuts.
38  */
39 
40 /*
41  * @test
42  * @summary Simple jpackage command run
43  * @library ../helpers
44  * @key jpackagePlatformPackage
45  * @build jdk.jpackage.test.*
46  * @modules jdk.incubator.jpackage/jdk.incubator.jpackage.internal
47  * @compile SimplePackageTest.java
48  * @run main/othervm/timeout=360 -Xmx512m jdk.jpackage.test.Main
49  *  --jpt-run=SimplePackageTest
50  */
51 public class SimplePackageTest {
52 
53     @Test
54     public static void test() {
55         new PackageTest()
56         .configureHelloApp()
57         .addBundleDesktopIntegrationVerifier(false)
58         .run();
59     }
60 }
