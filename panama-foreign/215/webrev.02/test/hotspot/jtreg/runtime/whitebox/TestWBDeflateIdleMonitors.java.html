24 package runtime.whitebox;
25 
26 /*
27  * @test
28  * @bug 8246477
29  * @summary Test to verify that WB method deflateIdleMonitors works correctly.
30  * @library /test/lib
31  * @modules java.base/jdk.internal.misc
32  *          java.management
33  * @build sun.hotspot.WhiteBox
34  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
35  * @run driver runtime.whitebox.TestWBDeflateIdleMonitors
36  */
37 import jdk.test.lib.Asserts;
38 import jdk.test.lib.process.ProcessTools;
39 import jdk.test.lib.process.OutputAnalyzer;
40 import sun.hotspot.WhiteBox;
41 
42 public class TestWBDeflateIdleMonitors {
43 
44     public static void main(String args[]) throws Exception {
45         ProcessBuilder pb = ProcessTools.createTestJvm(
46                 "-Xbootclasspath/a:.",
47                 "-XX:+UnlockDiagnosticVMOptions",
48                 "-XX:+WhiteBoxAPI",
49                 "-Xlog:monitorinflation=info",
50                 InflateMonitorsTest.class.getName());
51 
52         OutputAnalyzer output = new OutputAnalyzer(pb.start());
53         System.out.println(output.getStdout());
54         output.shouldHaveExitValue(0);
55         output.shouldContain("WhiteBox initiated DeflateIdleMonitors");
56     }
57 
58     public static class InflateMonitorsTest {
59         static WhiteBox wb = WhiteBox.getWhiteBox();
60         public static Object obj;
61 
62         public static void main(String args[]) {
63             obj = new Object();
64             synchronized (obj) {
65                 // HotSpot implementation detail: asking for the hash code
66                 // when the object is locked causes monitor inflation.
67                 if (obj.hashCode() == 0xBAD) System.out.println("!");
68                 Asserts.assertEQ(wb.isMonitorInflated(obj), true,
69                                  "Monitor should be inflated.");
70             }
71             boolean did_deflation = wb.deflateIdleMonitors();
72             Asserts.assertEQ(did_deflation, true,
73                              "deflateIdleMonitors() should have worked.");
74             Asserts.assertEQ(wb.isMonitorInflated(obj), false,
75                              "Monitor should be deflated.");
76         }
77     }
78 }
