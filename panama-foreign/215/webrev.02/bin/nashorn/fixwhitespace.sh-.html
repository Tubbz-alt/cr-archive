25 fix() {
26     #convert tabs to spaces
27     find . -name $1 -exec sed -i "" 's/	/    /g' {} \;
28     #remove trailing whitespace
29     find . -name $1 -exec sed -i "" 's/[ 	]*$//' \{} \;
30 }
31 
32 if [ ! -z $1 ]; then 
33     fix $1;
34 else
35     fix "*.java"
36     fix "*.js"
37 fi
