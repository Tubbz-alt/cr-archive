26 #ifndef CPU_ZERO_FRAME_ZERO_HPP
27 #define CPU_ZERO_FRAME_ZERO_HPP
28 


29 // A frame represents a physical stack frame on the Zero stack.
30 
31  public:
32   enum {
33     pc_return_offset = 0
34   };
35 
36   // Constructor
37  public:
38   frame(ZeroFrame* zeroframe, intptr_t* sp);
39 
40  private:
41   ZeroFrame* _zeroframe;
42 
43  public:
44   const ZeroFrame *zeroframe() const {
45     return _zeroframe;
46   }
47 
48   intptr_t* fp() const {
49     return (intptr_t *) zeroframe();
50   }
51 

52   inline interpreterState get_interpreterState() const;

53 
54  public:
55   const EntryFrame *zero_entryframe() const {
56     return zeroframe()->as_entry_frame();
57   }
58   const InterpreterFrame *zero_interpreterframe() const {
59     return zeroframe()->as_interpreter_frame();
60   }
61 
62  public:
63   bool is_fake_stub_frame() const;
64 
65  public:
66   frame sender_for_nonentry_frame(RegisterMap* map) const;
67 
68  public:
69   void zero_print_on_error(int           index,
70                            outputStream* st,
71                            char*         buf,
72                            int           buflen) const;
73 
74   static jint interpreter_frame_expression_stack_direction() { return -1; }
75 
76 #endif // CPU_ZERO_FRAME_ZERO_HPP
