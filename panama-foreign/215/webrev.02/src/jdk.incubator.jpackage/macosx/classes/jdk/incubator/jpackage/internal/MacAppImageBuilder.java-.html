237     @Override
238     public void prepareApplicationFiles(Map<String, ? super Object> params)
239             throws IOException {
240         Files.createDirectories(macOSDir);
241 
242         Map<String, ? super Object> originalParams = new HashMap<>(params);
243         // Generate PkgInfo
244         File pkgInfoFile = new File(contentsDir.toFile(), "PkgInfo");
245         pkgInfoFile.createNewFile();
246         writePkgInfo(pkgInfoFile);
247 
248         Path executable = macOSDir.resolve(getLauncherName(params));
249 
250         // create the main app launcher
251         try (InputStream is_launcher =
252                 getResourceAsStream("jpackageapplauncher")) {
253             // Copy executable and library to MacOS folder
254             writeEntry(is_launcher, executable);
255         }
256         executable.toFile().setExecutable(true, false);
257         // generate main app launcher config file
258         writeCfgFile(params);
259 
260         // create additional app launcher(s) and config file(s)
261         List<Map<String, ? super Object>> entryPoints =
262                 StandardBundlerParam.ADD_LAUNCHERS.fetchFrom(params);
263         for (Map<String, ? super Object> entryPoint : entryPoints) {
264             Map<String, ? super Object> tmp =
265                     AddLauncherArguments.merge(originalParams, entryPoint);
266 
267             // add executable for add launcher
268             Path addExecutable = macOSDir.resolve(getLauncherName(tmp));
269             try (InputStream is = getResourceAsStream("jpackageapplauncher");) {
270                 writeEntry(is, addExecutable);
271             }
272             addExecutable.toFile().setExecutable(true, false);
273 
274             // add config file for add launcher
275             writeCfgFile(tmp);
276         }
277 
278         // Copy class path entries to Java folder
279         copyApplication(params);
280 
281         /*********** Take care of "config" files *******/
282 
283         createResource(TEMPLATE_BUNDLE_ICON, params)
284                 .setCategory("icon")
285                 .setExternal(ICON_ICNS.fetchFrom(params))
286                 .saveToFile(resourcesDir.resolve(APP_NAME.fetchFrom(params)
287                         + ".icns"));
288 
289         // copy file association icons
290         for (Map<String, ?
291                 super Object> fa : FILE_ASSOCIATIONS.fetchFrom(params)) {
292             File f = FA_ICON.fetchFrom(fa);
293             if (f != null && f.exists()) {
294                 try (InputStream in2 = new FileInputStream(f)) {
295                     Files.copy(in2, resourcesDir.resolve(f.getName()));
296                 }
297 
298             }
299         }
300 
301         copyRuntimeFiles(params);
302         sign(params);
303     }

305     private void copyRuntimeFiles(Map<String, ? super Object> params)
306             throws IOException {
307         // Generate Info.plist
308         writeInfoPlist(contentsDir.resolve("Info.plist").toFile(), params);
309 
310         // generate java runtime info.plist
311         writeRuntimeInfoPlist(
312                 runtimeDir.resolve("Contents/Info.plist").toFile(), params);
313 
314         // copy library
315         Path runtimeMacOSDir = Files.createDirectories(
316                 runtimeDir.resolve("Contents/MacOS"));
317 
318         final Path jliName = Path.of("libjli.dylib");
319         try (Stream<Path> walk = Files.walk(runtimeRoot.resolve("lib"))) {
320             final Path jli = walk
321                     .filter(file -> file.getFileName().equals(jliName))
322                     .findFirst()
323                     .get();
324             Files.copy(jli, runtimeMacOSDir.resolve(jliName));
325         }
326     }

328     private void sign(Map<String, ? super Object> params) throws IOException {
329         if (Optional.ofNullable(
330                 SIGN_BUNDLE.fetchFrom(params)).orElse(Boolean.TRUE)) {
331             try {
332                 addNewKeychain(params);
333             } catch (InterruptedException e) {
334                 Log.error(e.getMessage());
335             }
336             String signingIdentity =
337                     DEVELOPER_ID_APP_SIGNING_KEY.fetchFrom(params);
338             if (signingIdentity != null) {
339                 prepareEntitlements(params);
340                 signAppBundle(params, root, signingIdentity,
341                         BUNDLE_ID_SIGNING_PREFIX.fetchFrom(params),
342                         getConfig_Entitlements(params));
343             }
344             restoreKeychainList(params);
345         }
346     }

364     private String getBundleName(Map<String, ? super Object> params) {
365         if (MAC_CF_BUNDLE_NAME.fetchFrom(params) != null) {
366             String bn = MAC_CF_BUNDLE_NAME.fetchFrom(params);
367             if (bn.length() > 16) {
368                 Log.error(MessageFormat.format(I18N.getString(
369                         "message.bundle-name-too-long-warning"),
370                         MAC_CF_BUNDLE_NAME.getID(), bn));
371             }
372             return MAC_CF_BUNDLE_NAME.fetchFrom(params);
373         } else if (APP_NAME.fetchFrom(params) != null) {
374             return APP_NAME.fetchFrom(params);
375         } else {
376             String nm = MAIN_CLASS.fetchFrom(params);
377             if (nm.length() > 16) {
378                 nm = nm.substring(0, 16);
379             }
380             return nm;
381         }
382     }

384     private void writeRuntimeInfoPlist(File file,
385             Map<String, ? super Object> params) throws IOException {
386         Map<String, String> data = new HashMap<>();
387         String identifier = StandardBundlerParam.isRuntimeInstaller(params) ?
388                 MAC_CF_BUNDLE_IDENTIFIER.fetchFrom(params) :
389                 "com.oracle.java." + MAC_CF_BUNDLE_IDENTIFIER.fetchFrom(params);
390         data.put("CF_BUNDLE_IDENTIFIER", identifier);
391         String name = StandardBundlerParam.isRuntimeInstaller(params) ?
392                 getBundleName(params): "Java Runtime Image";
393         data.put("CF_BUNDLE_NAME", name);
394         data.put("CF_BUNDLE_VERSION", VERSION.fetchFrom(params));
395         data.put("CF_BUNDLE_SHORT_VERSION_STRING", VERSION.fetchFrom(params));
396 
397         createResource(TEMPLATE_RUNTIME_INFO_PLIST, params)
398                 .setPublicName("Runtime-Info.plist")
399                 .setCategory(I18N.getString("resource.runtime-info-plist"))
400                 .setSubstitutionData(data)
401                 .saveToFile(file);
402     }

404     private void writeStringArrayPlist(StringBuilder sb, String key,
405             List<String> values) {
406         if (values != null && !values.isEmpty()) {
407             sb.append("  <key>").append(key).append("</key>\n").append("   <array>\n");
408             values.forEach((value) -> {
409                 sb.append("    <string>").append(value).append("</string>\n");
410             });
411             sb.append("   </array>\n");
412         }
413     }

429     private void writeInfoPlist(File file, Map<String, ? super Object> params)
430             throws IOException {
431         Log.verbose(MessageFormat.format(I18N.getString(
432                 "message.preparing-info-plist"), file.getAbsolutePath()));
433 
434         //prepare config for exe
435         //Note: do not need CFBundleDisplayName if we don't support localization
436         Map<String, String> data = new HashMap<>();
437         data.put("DEPLOY_ICON_FILE", APP_NAME.fetchFrom(params) + ".icns");
438         data.put("DEPLOY_BUNDLE_IDENTIFIER",
439                 MAC_CF_BUNDLE_IDENTIFIER.fetchFrom(params));
440         data.put("DEPLOY_BUNDLE_NAME",
441                 getBundleName(params));
442         data.put("DEPLOY_BUNDLE_COPYRIGHT", COPYRIGHT.fetchFrom(params));
443         data.put("DEPLOY_LAUNCHER_NAME", getLauncherName(params));
444         data.put("DEPLOY_BUNDLE_SHORT_VERSION", VERSION.fetchFrom(params));
445         data.put("DEPLOY_BUNDLE_CFBUNDLE_VERSION", VERSION.fetchFrom(params));
446 
447         StringBuilder bundleDocumentTypes = new StringBuilder();
448         StringBuilder exportedTypes = new StringBuilder();
449         for (Map<String, ? super Object>
450                 fileAssociation : FILE_ASSOCIATIONS.fetchFrom(params)) {
451 
452             List<String> extensions = FA_EXTENSIONS.fetchFrom(fileAssociation);
453             if (extensions == null) {
454                 Log.verbose(I18N.getString(
455                         "message.creating-association-with-null-extension"));
456             }
457 
458             String itemContentType = MAC_CF_BUNDLE_IDENTIFIER.fetchFrom(params)
459                     + "." + ((extensions == null || extensions.isEmpty())
460                     ? "mime" : extensions.get(0));
461             String description = FA_DESCRIPTION.fetchFrom(fileAssociation);
462             File icon = FA_ICON.fetchFrom(fileAssociation);
463 
464             bundleDocumentTypes.append(" <dict>\n");
465             writeStringArrayPlist(bundleDocumentTypes, "LSItemContentTypes",
466                     Arrays.asList(itemContentType));
467             writeStringPlist(bundleDocumentTypes, "CFBundleTypeName", description);
468             writeStringPlist(bundleDocumentTypes, "LSHandlerRank",
469                     FA_MAC_LSHANDLERRANK.fetchFrom(fileAssociation));
470             writeStringPlist(bundleDocumentTypes, "CFBundleTypeRole",
471                     FA_MAC_CFBUNDLETYPEROLE.fetchFrom(fileAssociation));
472             writeStringPlist(bundleDocumentTypes, "NSPersistentStoreTypeKey",
473                     FA_MAC_NSSTORETYPEKEY.fetchFrom(fileAssociation));
474             writeStringPlist(bundleDocumentTypes, "NSDocumentClass",
475                     FA_MAC_NSDOCUMENTCLASS.fetchFrom(fileAssociation));
476             writeBoolPlist(bundleDocumentTypes, "LSIsAppleDefaultForType",
477                     "true");
478             writeBoolPlist(bundleDocumentTypes, "LSTypeIsPackage",
479                     FA_MAC_LSTYPEISPACKAGE.fetchFrom(fileAssociation));
480             writeBoolPlist(bundleDocumentTypes, "LSSupportsOpeningDocumentsInPlace",
481                     FA_MAC_LSDOCINPLACE.fetchFrom(fileAssociation));
482             writeBoolPlist(bundleDocumentTypes, "UISupportsDocumentBrowser",
483                     FA_MAC_UIDOCBROWSER.fetchFrom(fileAssociation));
484             if (icon != null && icon.exists()) {
485                 writeStringPlist(bundleDocumentTypes, "CFBundleTypeIconFile",
486                         icon.getName());
487             }
488             bundleDocumentTypes.append("  </dict>\n");
489 
490             exportedTypes.append("  <dict>\n");
491             writeStringPlist(exportedTypes, "UTTypeIdentifier",
492                     itemContentType);
493             writeStringPlist(exportedTypes, "UTTypeDescription",
494                     description);
495             writeStringArrayPlist(exportedTypes, "UTTypeConformsTo",
496                     FA_MAC_UTTYPECONFORMSTO.fetchFrom(fileAssociation));
497 
498             if (icon != null && icon.exists()) {
499                 writeStringPlist(exportedTypes, "UTTypeIconFile", icon.getName());
500             }
501             exportedTypes.append("\n")
502                     .append("  <key>UTTypeTagSpecification</key>\n")
503                     .append("  <dict>\n")
504                     .append("\n");
505             writeStringArrayPlist(exportedTypes, "public.filename-extension",
506                     extensions);
507             writeStringArrayPlist(exportedTypes, "public.mime-type",
508                     FA_CONTENT_TYPE.fetchFrom(fileAssociation));
509             writeStringArrayPlist(exportedTypes, "NSExportableTypes",
510                     FA_MAC_NSEXPORTABLETYPES.fetchFrom(fileAssociation));
511             exportedTypes.append("  </dict>\n").append(" </dict>\n");
512         }
513         String associationData;
514         if (bundleDocumentTypes.length() > 0) {
515             associationData =
516                     "\n  <key>CFBundleDocumentTypes</key>\n  <array>\n"
517                     + bundleDocumentTypes.toString()
518                     + "  </array>\n\n"
519                     + "  <key>UTExportedTypeDeclarations</key>\n  <array>\n"
520                     + exportedTypes.toString()
521                     + "  </array>\n";
522         } else {
523             associationData = "";
524         }
525         data.put("DEPLOY_FILE_ASSOCIATIONS", associationData);
526 
527         createResource(TEMPLATE_INFO_PLIST_LITE, params)
528                 .setCategory(I18N.getString("resource.app-info-plist"))
529                 .setSubstitutionData(data)
530                 .setPublicName("Info.plist")
531                 .saveToFile(file);
532     }

544     public static void addNewKeychain(Map<String, ? super Object> params)
545                                     throws IOException, InterruptedException {
546         if (Platform.getMajorVersion() < 10 ||
547                 (Platform.getMajorVersion() == 10 &&
548                 Platform.getMinorVersion() < 12)) {
549             // we need this for OS X 10.12+
550             return;
551         }
552 
553         String keyChain = SIGNING_KEYCHAIN.fetchFrom(params);
554         if (keyChain == null || keyChain.isEmpty()) {
555             return;
556         }
557 
558         // get current keychain list
559         String keyChainPath = new File (keyChain).getAbsolutePath().toString();
560         List<String> keychainList = new ArrayList<>();
561         int ret = IOUtils.getProcessOutput(
562                 keychainList, "security", "list-keychains");
563         if (ret != 0) {
564             Log.error(I18N.getString("message.keychain.error"));
565             return;
566         }
567 
568         boolean contains = keychainList.stream().anyMatch(
569                     str -> str.trim().equals("\""+keyChainPath.trim()+"\""));
570         if (contains) {
571             // keychain is already added in the search list
572             return;
573         }
574 
575         keyChains = new ArrayList<>();
576         // remove "
577         keychainList.forEach((String s) -> {
578             String path = s.trim();
579             if (path.startsWith("\"") && path.endsWith("\"")) {
580                 path = path.substring(1, path.length()-1);
581             }
582             keyChains.add(path);
583         });
584 
585         List<String> args = new ArrayList<>();
586         args.add("security");
587         args.add("list-keychains");
588         args.add("-s");
589 
590         args.addAll(keyChains);
591         args.add(keyChain);
592 
593         ProcessBuilder  pb = new ProcessBuilder(args);
594         IOUtils.exec(pb);
595     }

597     public static void restoreKeychainList(Map<String, ? super Object> params)
598             throws IOException{
599         if (Platform.getMajorVersion() < 10 ||
600                 (Platform.getMajorVersion() == 10 &&
601                 Platform.getMinorVersion() < 12)) {
602             // we need this for OS X 10.12+
603             return;
604         }
605 
606         if (keyChains == null || keyChains.isEmpty()) {
607             return;
608         }
609 
610         List<String> args = new ArrayList<>();
611         args.add("security");
612         args.add("list-keychains");
613         args.add("-s");
614 
615         args.addAll(keyChains);
616 
617         ProcessBuilder  pb = new ProcessBuilder(args);
618         IOUtils.exec(pb);
619     }

621     static void signAppBundle(
622             Map<String, ? super Object> params, Path appLocation,
623             String signingIdentity, String identifierPrefix, File entitlements)
624             throws IOException {
625         AtomicReference<IOException> toThrow = new AtomicReference<>();
626         String appExecutable = "/Contents/MacOS/" + APP_NAME.fetchFrom(params);
627         String keyChain = SIGNING_KEYCHAIN.fetchFrom(params);
628 
629         // sign all dylibs and executables
630         try (Stream<Path> stream = Files.walk(appLocation)) {
631             stream.peek(path -> { // fix permissions
632                 try {
633                     Set<PosixFilePermission> pfp =
634                             Files.getPosixFilePermissions(path);
635                     if (!pfp.contains(PosixFilePermission.OWNER_WRITE)) {
636                         pfp = EnumSet.copyOf(pfp);
637                         pfp.add(PosixFilePermission.OWNER_WRITE);
638                         Files.setPosixFilePermissions(path, pfp);
639                     }
640                 } catch (IOException e) {
641                     Log.verbose(e);
642                 }
643             }).filter(p -> Files.isRegularFile(p) &&
644                       (Files.isExecutable(p) || p.toString().endsWith(".dylib"))
645                       && !(p.toString().endsWith(appExecutable)
646                       || p.toString().contains("/Contents/runtime")
647                       || p.toString().contains("/Contents/Frameworks"))
648                      ).forEach(p -> {
649                 // noinspection ThrowableResultOfMethodCallIgnored
650                 if (toThrow.get() != null) return;
651 
652                 // If p is a symlink then skip the signing process.
653                 if (Files.isSymbolicLink(p)) {
654                     Log.verbose(MessageFormat.format(I18N.getString(
655                             "message.ignoring.symlink"), p.toString()));
656                 } else if (isFileSigned(p)) {
657                     // executable or lib already signed
658                     Log.verbose(MessageFormat.format(I18N.getString(
659                             "message.already.signed"), p.toString()));
660                 } else {
661                     List<String> args = new ArrayList<>();
662                     args.addAll(Arrays.asList("codesign",
663                             "--timestamp",
664                             "--options", "runtime",
665                             "-s", signingIdentity,
666                             "--prefix", identifierPrefix,
667                             "-vvvv"));
668                     if (keyChain != null && !keyChain.isEmpty()) {
669                         args.add("--keychain");
670                         args.add(keyChain);
671                     }
672 
673                     if (Files.isExecutable(p)) {
674                         if (entitlements != null) {
675                             args.add("--entitlements");
676                             args.add(entitlements.toString());
677                         }
678                     }
679 
680                     args.add(p.toString());
681 
682                     try {
683                         Set<PosixFilePermission> oldPermissions =
684                                 Files.getPosixFilePermissions(p);
685                         File f = p.toFile();
686                         f.setWritable(true, true);
687 
688                         ProcessBuilder pb = new ProcessBuilder(args);
689 
690                         IOUtils.exec(pb);
691 
692                         Files.setPosixFilePermissions(p, oldPermissions);
693                     } catch (IOException ioe) {
694                         toThrow.set(ioe);
695                     }
696                 }
697             });
698         }
699         IOException ioe = toThrow.get();
700         if (ioe != null) {
701             throw ioe;
702         }
703 
704         // sign all runtime and frameworks
705         Consumer<? super Path> signIdentifiedByPList = path -> {
706             //noinspection ThrowableResultOfMethodCallIgnored
707             if (toThrow.get() != null) return;
708 
709             try {
710                 List<String> args = new ArrayList<>();
711                 args.addAll(Arrays.asList("codesign",
712                         "--timestamp",
713                         "--options", "runtime",
714                         "--force",
715                         "-s", signingIdentity, // sign with this key
716                         "--prefix", identifierPrefix,
717                         // use the identifier as a prefix
718                         "-vvvv"));
719 
720                 if (keyChain != null && !keyChain.isEmpty()) {
721                     args.add("--keychain");
722                     args.add(keyChain);
723                 }
724                 args.add(path.toString());
725                 ProcessBuilder pb = new ProcessBuilder(args);
726 
727                 IOUtils.exec(pb);
728             } catch (IOException e) {
729                 toThrow.set(e);
730             }
731         };
732 
733         Path javaPath = appLocation.resolve("Contents/runtime");
734         if (Files.isDirectory(javaPath)) {
735             signIdentifiedByPList.accept(javaPath);
736 
737             ioe = toThrow.get();
738             if (ioe != null) {
739                 throw ioe;
740             }
741         }
742         Path frameworkPath = appLocation.resolve("Contents/Frameworks");
743         if (Files.isDirectory(frameworkPath)) {
744             try (var fileList = Files.list(frameworkPath)) {
745                 fileList.forEach(signIdentifiedByPList);
746             }
747 
748             ioe = toThrow.get();
749             if (ioe != null) {
750                 throw ioe;
751             }
752         }
753 
754         // sign the app itself
755         List<String> args = new ArrayList<>();
756         args.addAll(Arrays.asList("codesign",
757                 "--timestamp",
758                 "--options", "runtime",
759                 "--force",
760                 "-s", signingIdentity,
761                 "-vvvv"));
762 
763         if (keyChain != null && !keyChain.isEmpty()) {
764             args.add("--keychain");
765             args.add(keyChain);
766         }
767 
768         if (entitlements != null) {
769             args.add("--entitlements");
770             args.add(entitlements.toString());
771         }
772 
773         args.add(appLocation.toString());
774 
775         ProcessBuilder pb =
776                 new ProcessBuilder(args.toArray(new String[args.size()]));
777 
778         IOUtils.exec(pb);
779     }

781     private static boolean isFileSigned(Path file) {
782         ProcessBuilder pb =
783                 new ProcessBuilder("codesign", "--verify", file.toString());
784 
785         try {
786             IOUtils.exec(pb);
787         } catch (IOException ex) {
788             return false;
789         }
790 
791         return true;
792     }

794     private static String extractBundleIdentifier(Map<String, Object> params) {
795         if (PREDEFINED_APP_IMAGE.fetchFrom(params) == null) {
796             return null;
797         }
798 
799         try {
800             File infoPList = new File(PREDEFINED_APP_IMAGE.fetchFrom(params) +
801                                       File.separator + "Contents" +
802                                       File.separator + "Info.plist");
803 
804             DocumentBuilderFactory dbf
805                     = DocumentBuilderFactory.newDefaultInstance();
806             dbf.setFeature("http://apache.org/xml/features/" +
807                            "nonvalidating/load-external-dtd", false);
808             DocumentBuilder b = dbf.newDocumentBuilder();
809             org.w3c.dom.Document doc = b.parse(new FileInputStream(
810                     infoPList.getAbsolutePath()));
811 
812             XPath xPath = XPathFactory.newInstance().newXPath();
813             // Query for the value of <string> element preceding <key>
814             // element with value equal to CFBundleIdentifier
815             String v = (String) xPath.evaluate(
816                     "//string[preceding-sibling::key = \"CFBundleIdentifier\"][1]",
817                     doc, XPathConstants.STRING);
818 
819             if (v != null && !v.isEmpty()) {
820                 return v;
821             }
822         } catch (Exception ex) {
823             Log.verbose(ex);
824         }
825 
826         return null;
827     }
