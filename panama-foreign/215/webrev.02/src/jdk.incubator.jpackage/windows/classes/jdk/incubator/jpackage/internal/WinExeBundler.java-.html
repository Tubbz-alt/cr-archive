 25 package jdk.incubator.jpackage.internal;
 26 
 27 import java.io.*;
 28 import java.nio.file.Files;
 29 import java.nio.file.Path;
 30 import java.nio.file.Paths;
 31 import java.text.MessageFormat;
 32 import java.util.*;
 33 
 34 public class WinExeBundler extends AbstractBundler {
 35 
 36     static {
 37         System.loadLibrary("jpackage");
 38     }
 39 
 40     public static final BundlerParamInfo<File> EXE_IMAGE_DIR
 41             = new StandardBundlerParam<>(
 42                     "win.exe.imageDir",
 43                     File.class,
 44                     params -> {
 45                         File imagesRoot = IMAGES_ROOT.fetchFrom(params);
 46                         if (!imagesRoot.exists()) {
 47                             imagesRoot.mkdirs();
 48                         }
 49                         return new File(imagesRoot, "win-exe.image");
 50                     },
 51                     (s, p) -> null);
 52 
 53     private final static String EXE_WRAPPER_NAME = "msiwrapper.exe";
 54 
 55     @Override
 56     public String getName() {
 57         return I18N.getString("exe.bundler.name");
 58     }
 59 
 60     @Override
 61     public String getID() {
 62         return "exe";
 63     }
 64 
 65     @Override
 66     public String getBundleType() {
 67         return "INSTALLER";
 68     }
 69 
 70     @Override
 71     public File execute(Map<String, ? super Object> params,
 72             File outputParentDir) throws PackagerException {
 73         return bundle(params, outputParentDir);
 74     }
 75 
 76     @Override
 77     public boolean supported(boolean platformInstaller) {
 78         return msiBundler.supported(platformInstaller);
 79     }
 80 
 81     @Override
 82     public boolean isDefault() {
 83         return true;
 84     }
 85 
 86     @Override
 87     public boolean validate(Map<String, ? super Object> params)
 88             throws ConfigException {
 89         return msiBundler.validate(params);
 90     }
 91 
 92     public File bundle(Map<String, ? super Object> params, File outdir)
 93             throws PackagerException {
 94 
 95         IOUtils.writableOutputDir(outdir.toPath());
 96 
 97         File exeImageDir = EXE_IMAGE_DIR.fetchFrom(params);
 98 
 99         // Write msi to temporary directory.
100         File msi = msiBundler.execute(params, exeImageDir);
101 
102         try {
103             new ScriptRunner()
104             .setDirectory(msi.toPath().getParent())
105             .setResourceCategoryId("resource.post-msi-script")
106             .setScriptNameSuffix("post-msi")
107             .setEnvironmentVariable("JpMsiFile", msi.getAbsolutePath().toString())
108             .run(params);
109 
110             return buildEXE(params, msi, outdir);
111         } catch (IOException ex) {
112             Log.verbose(ex);
113             throw new PackagerException(ex);
114         }
115     }
116 
117     private File buildEXE(Map<String, ? super Object> params, File msi,
118             File outdir) throws IOException {
119 
120         Log.verbose(MessageFormat.format(
121                 I18N.getString("message.outputting-to-location"),
122                 outdir.getAbsolutePath()));
123 
124         // Copy template msi wrapper next to msi file
125         final Path exePath = IOUtils.replaceSuffix(msi.toPath(), ".exe");
126         try (InputStream is = OverridableResource.readDefault(EXE_WRAPPER_NAME)) {
127             Files.copy(is, exePath);
128         }
129 
130         new ExecutableRebrander().addAction((resourceLock) -> {
131             // Embed msi in msi wrapper exe.
132             embedMSI(resourceLock, msi.getAbsolutePath());
133         }).rebrandInstaller(params, exePath);
134 
135         Path dstExePath = Paths.get(outdir.getAbsolutePath(),
136                 exePath.getFileName().toString());
137         Files.deleteIfExists(dstExePath);
138 
139         Files.copy(exePath, dstExePath);
140 
141         Log.verbose(MessageFormat.format(
142                 I18N.getString("message.output-location"),
143                 outdir.getAbsolutePath()));
144 
145         return dstExePath.toFile();
146     }
147 
148     private final WinMsiBundler msiBundler = new WinMsiBundler();
149 
150     private static native int embedMSI(long resourceLock, String msiPath);
151 }
