26 package jdk.incubator.jpackage.internal;
27 
28 import java.io.File;
29 import java.io.IOException;
30 import java.util.*;
31 import java.util.jar.Attributes;
32 import java.util.jar.JarFile;
33 import java.util.jar.Manifest;
34 
35 import static jdk.incubator.jpackage.internal.StandardBundlerParam.*;


36 
37 public class BundleParams {
38 
39     final protected Map<String, ? super Object> params;
40 
41     /**
42      * create a new bundle with all default values
43      */
44     public BundleParams() {
45         params = new HashMap<>();
46     }
47 
48     /**
49      * Create a bundle params with a copy of the params
50      * @param params map of initial parameters to be copied in.
51      */
52     public BundleParams(Map<String, ?> params) {
53         this.params = new HashMap<>(params);
54     }
55 
56     public void addAllBundleParams(Map<String, ? super Object> params) {
57         this.params.putAll(params);
58     }
59 
60     // NOTE: we do not care about application parameters here
61     // as they will be embeded into jar file manifest and
62     // java launcher will take care of them!
63 
64     public Map<String, ? super Object> getBundleParamsAsMap() {
65         return new HashMap<>(params);
66     }
67 
68     public String getName() {
69         return APP_NAME.fetchFrom(params);
70     }
71 
72     private void putUnlessNull(String param, Object value) {
73         if (value != null) {
74             params.put(param, value);
75         }
76     }
77 }
