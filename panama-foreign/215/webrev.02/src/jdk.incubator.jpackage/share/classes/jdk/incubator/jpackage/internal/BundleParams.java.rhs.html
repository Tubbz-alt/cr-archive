26 package jdk.incubator.jpackage.internal;
27 






28 
29 import java.util.HashMap;
30 import java.util.Map;
31 import static jdk.incubator.jpackage.internal.StandardBundlerParam.APP_NAME;
32 
33 public class BundleParams {
34 
35     final protected Map<String, ? super Object> params;
36 
37     /**
38      * create a new bundle with all default values
39      */
40     public BundleParams() {
41         params = new HashMap<>();
42     }
43 
44     /**
45      * Create a bundle params with a copy of the params
46      * @param params map of initial parameters to be copied in.
47      */
48     public BundleParams(Map<String, ?> params) {
49         this.params = new HashMap<>(params);
50     }
51 
52     public void addAllBundleParams(Map<String, ? super Object> params) {
53         this.params.putAll(params);
54     }
55 
56     // NOTE: we do not care about application parameters here
57     // as they will be embeded into jar file manifest and
58     // java launcher will take care of them!
59 
60     public Map<String, ? super Object> getBundleParamsAsMap() {
61         return new HashMap<>(params);
62     }
63 
64     public String getName() {
65         return APP_NAME.fetchFrom(params);
66     }
67 
68     private void putUnlessNull(String param, Object value) {
69         if (value != null) {
70             params.put(param, value);
71         }
72     }
73 }
