 142 ################################################################################
 143 # Optionally create AOT libraries for specified modules before running tests.
 144 # Note, this could not be done during JDK build time.
 145 ################################################################################
 146 # Parameter 1 is the name of the rule.
 147 #
 148 # Remaining parameters are named arguments.
 149 #   MODULE      The module to generate a library for
 150 #   BIN         Output directory in which to put the library
 151 #   VM_OPTIONS  List of JVM arguments to use when creating library
 152 #   OPTIONS_VAR Name of variable to put AOT java options in
 153 #   PREREQS_VAR Name of variable to put all AOT prerequisite rule targets in
 154 #               for test rules to depend on
 155 #
 156 SetupAotModule = $(NamedParamsMacroTemplate)
 157 define SetupAotModuleBody
 158   $1_AOT_LIB := $$($1_BIN)/$$(call SHARED_LIBRARY,$$($1_MODULE))
 159   $1_AOT_CCLIST := $$(wildcard $$(TOPDIR)/test/hotspot/jtreg/compiler/aot/scripts/$$($1_MODULE)-list.txt)
 160 
 161   # Create jaotc flags.
 162   # VM flags which don't affect AOT code generation are filtered out:
 163   # -Xcomp, -XX:+-TieredCompilation
 164   $1_JAOTC_OPTS := \
 165       -J-Xmx4g --info \
 166       $$(addprefix -J, $$(filter-out -Xcomp %TieredCompilation, $$($1_VM_OPTIONS))) \
 167       $$(addprefix --compile-commands$(SPACE), $$($1_AOT_CCLIST)) \
 168       --linker-path $$(LD_JAOTC) \
 169       #
 170 
 171   ifneq ($$(filter -ea, $$($1_VM_OPTIONS)), )
 172     $1_JAOTC_OPTS += --compile-with-assertions
 173   endif
 174 
 175   $$($1_AOT_LIB): $$(JDK_UNDER_TEST)/release \
 176       $$(call DependOnVariable, $1_JAOTC_OPTS) \
 177       $$(call DependOnVariable, JDK_UNDER_TEST)
 178 	$$(call LogWarn, Generating $$(patsubst $$(OUTPUTDIR)/%, %, $$@))
 179 	$$(call MakeTargetDir)
 180 	$$(call ExecuteWithLog, $$@, \
 181 	    $((COV_ENVIRONMENT) \
 182 	    $$(FIXPATH) $$(JDK_UNDER_TEST)/bin/jaotc \
 183 	        $$($1_JAOTC_OPTS) --output $$@ --module $$($1_MODULE) \
 184 	)
 185 	$$(call ExecuteWithLog, $$@.check, ( \
 186 	    $$(FIXPATH) $$(JDK_UNDER_TEST)/bin/java \
 187 	        $$($1_VM_OPTIONS) -XX:+UnlockDiagnosticVMOptions -XX:+UnlockExperimentalVMOptions \
 188 	        -XX:+PrintAOT -XX:+UseAOTStrictLoading \
 189 	        -XX:AOTLibrary=$$@ -version \
 190 	         > $$@.verify-aot \
 191 	))
 192 
 193   $1_AOT_OPTIONS += -XX:+UnlockExperimentalVMOptions
 194   $1_AOT_OPTIONS += -XX:AOTLibrary=$$($1_AOT_LIB)
 195   $1_AOT_TARGETS += $$($1_AOT_LIB)
 196 endef

 239 ifeq ($(TEST_JOBS), 0)
 240   CORES_DIVIDER := 2
 241   # For some big multi-core machines with low ulimit -u setting we hit the max
 242   # threads/process limit. In such a setup the memory/cores-only-guided
 243   # TEST_JOBS config is insufficient. From experience a concurrency setting of
 244   # 14 works reasonably well for low ulimit values (<= 4096). Thus, use
 245   # divider 4096/14. For high ulimit -u values this shouldn't make a difference.
 246   ULIMIT_DIVIDER := (4096/14)

 255   TEST_JOBS := $(shell $(AWK) \
 256     'BEGIN { \
 257       c = $(NUM_CORES) / $(CORES_DIVIDER); \
 258       m = $(MEMORY_SIZE) / $(MEMORY_DIVIDER); \
 259       u = $(PROC_ULIMIT); \
 260       if (u > -1) { \
 261         u = u / $(ULIMIT_DIVIDER); \
 262         if (u < c) c = u; \
 263       } \
 264       if (c > m) c = m; \
 265       c = c * $(TEST_JOBS_FACTOR); \
 266       c = c * $(TEST_JOBS_FACTOR_JDL); \
 267       c = c * $(TEST_JOBS_FACTOR_MACHINE); \

 356 ################################################################################
 357 # Parse test selection
 358 #
 359 # The user has given a test selection in the TEST variable. We must parse it
 360 # and determine what that means in terms of actual calls to the test framework.
 361 #
 362 # The parse functions take as argument a test specification as given by the
 363 # user, and returns a fully qualified test descriptor if it was a match, or
 364 # nothing if not. A single test specification can result in multiple test
 365 # descriptors being returned. A valid test descriptor must always be accepted
 366 # and returned identically.
 367 ################################################################################
 368 
 369 # Helper function to determine if a test specification is a Gtest test
 370 #
 371 # It is a Gtest test if it is either "gtest", or "gtest:" followed by an optional
 372 # test filter string, and an optional "/<variant>" to select a specific JVM
 373 # variant. If no variant is specified, all found variants are tested.
 374 define ParseGtestTestSelection
 375   $(if $(filter gtest%, $1), \
 376     $(if $(filter gtest, $1), \
 377       $(addprefix gtest:all/, $(GTEST_VARIANTS)) \
 378     , \
 379       $(if $(strip $(or $(filter gtest/%, $1) $(filter gtest:/%, $1))), \
 380         $(patsubst gtest:/%, gtest:all/%, $(patsubst gtest/%, gtest:/%, $1)) \
 381       , \
 382         $(if $(filter gtest:%, $1), \
 383           $(if $(findstring /, $1), \
 384             $1 \
 385           , \
 386             $(addprefix $1/, $(GTEST_VARIANTS)) \
 387           ) \
 388         ) \
 389       ) \
 390     ) \
 391   )
 392 endef
 393 
 394 # Helper function to determine if a test specification is a microbenchmark test
 395 #
 396 # It is a microbenchmark test if it is either "micro", or "micro:" followed by
 397 # an optional test filter string.
 398 define ParseMicroTestSelection
 399   $(if $(filter micro%, $1), \
 400     $(if $(filter micro, $1), \
 401       micro:all \
 402     , \
 403       $(if $(filter micro:, $1), \
 404         micro:all \
 405       , \
 406         $1 \
 407       ) \
 408     ) \
 409   )
 410 endef

 412 # Helper function that removes the TOPDIR part
 413 CleanupJtregPath = \
 414   $(strip $(patsubst %/, %, $(subst $(JTREG_TOPDIR)/,, $1)))
 415 
 416 # Take a partial Jtreg root path and return a full, absolute path to that Jtreg
 417 # root. Also support having "hotspot" as an alias for "hotspot/jtreg".
 418 ExpandJtregRoot = \
 419   $(call CleanupJtregPath, $(wildcard \
 420     $(if $(filter /%, $1), \
 421       $(if $(wildcard $(strip $1)/TEST.ROOT), \
 422         $1 \
 423       ) \
 424     , \
 425       $(filter $(addprefix %, $1), $(JTREG_TESTROOTS) $(addsuffix /, $(JTREG_TESTROOTS))) \
 426       $(filter $(addprefix %, $(strip $1)/jtreg), $(JTREG_TESTROOTS) $(addsuffix /, $(JTREG_TESTROOTS))) \
 427     ) \
 428   ))
 429 
 430 # Take a partial Jtreg test path and return a full, absolute path to that Jtreg
 431 # test. Also support having "hotspot" as an alias for "hotspot/jtreg".
 432 ExpandJtregPath = \
 433   $(if $(call ExpandJtregRoot, $1), \
 434     $(call ExpandJtregRoot, $1) \
 435   , \
 436     $(call CleanupJtregPath, $(wildcard \
 437       $(if $(filter /%, $1), \
 438         $1 \
 439       , \
 440         $(addsuffix /$(strip $1), $(JTREG_TESTROOTS) $(TEST_BASEDIRS)) \
 441         $(addsuffix $(strip $(patsubst hotspot/%, /hotspot/jtreg/%, $1)), $(JTREG_TESTROOTS) $(TEST_BASEDIRS)) \
 442       ) \
 443     )) \
 444   )

 446 # Helper function to determine if a test specification is a Jtreg test
 447 #
 448 # It is a Jtreg test if it optionally begins with jtreg:, and then is either
 449 # an unspecified group name (possibly prefixed by :), or a group in a
 450 # specified test root, or a path to a test or test directory,
 451 # either absolute or relative to any of the TEST_BASEDIRS or test roots.
 452 define ParseJtregTestSelection
 453   $(eval TEST_NAME := $(strip $(patsubst jtreg:%, %, $1))) \
 454   $(if $(or $(findstring :, $(TEST_NAME)), $(findstring /, $(TEST_NAME))), , \
 455     $(eval TEST_NAME := :$(TEST_NAME)) \
 456   ) \
 457   $(if $(findstring :, $(TEST_NAME)), \
 458     $(if $(filter :%, $(TEST_NAME)), \
 459       $(eval TEST_GROUP := $(patsubst :%, %, $(TEST_NAME))) \
 460       $(eval TEST_ROOTS := $(foreach test_root, $(JTREG_TESTROOTS), \
 461           $(call CleanupJtregPath, $(test_root)))) \
 462     , \
 463       $(eval TEST_PATH := $(word 1, $(subst :, $(SPACE), $(TEST_NAME)))) \
 464       $(eval TEST_GROUP := $(word 2, $(subst :, $(SPACE), $(TEST_NAME)))) \
 465       $(eval TEST_ROOTS := $(call ExpandJtregRoot, $(TEST_PATH))) \
 466     ) \
 467     $(foreach test_root, $(TEST_ROOTS), \
 468       $(if $(filter /%, $(test_root)), \
 469         jtreg:$(test_root):$(TEST_GROUP) \
 470       , \
 471         $(if $(filter $(TEST_GROUP), $($(JTREG_TOPDIR)/$(test_root)_JTREG_TEST_GROUPS)), \
 472           jtreg:$(test_root):$(TEST_GROUP) \
 473         ) \
 474       ) \
 475     ) \
 476   , \
 477     $(eval TEST_PATHS := $(call ExpandJtregPath, $(TEST_NAME))) \
 478     $(foreach test_path, $(TEST_PATHS), \
 479       jtreg:$(test_path) \
 480     ) \
 481   )
 482 endef

 484 # Helper function to determine if a test specification is a special test
 485 #
 486 # It is a special test if it is "special:" followed by a test name,
 487 # if it is "make:" or "make-" followed by a make test, or any of the special
 488 # test names as a single word.
 489 define ParseSpecialTestSelection
 490   $(if $(filter special:%, $1), \
 491     $1 \
 492   ) \
 493   $(if $(filter make%, $1), \
 494     $(if $(filter make:%, $1), \
 495       special:$(strip $1) \
 496     ) \
 497     $(if $(filter make-%, $1), \
 498       special:$(patsubst make-%,make:%, $1) \
 499     ) \
 500     $(if $(filter make, $1), \
 501       special:make:all \
 502     )
 503   ) \
 504   $(if $(filter failure-handler, $1), \
 505     special:$(strip $1) \
 506   )
 507 endef

 574 ### Rules for Gtest
 575 
 576 SetupRunGtestTest = $(NamedParamsMacroTemplate)
 577 define SetupRunGtestTestBody
 578   $1_TEST_RESULTS_DIR := $$(TEST_RESULTS_DIR)/$1
 579   $1_TEST_SUPPORT_DIR := $$(TEST_SUPPORT_DIR)/$1
 580   $1_EXITCODE := $$($1_TEST_RESULTS_DIR)/exitcode.txt
 581
