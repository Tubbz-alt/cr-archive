 25 package jdk.incubator.jpackage.internal;
 26 
 27 import java.io.File;
 28 import java.io.IOException;
 29 import java.lang.module.ModuleDescriptor;
 30 import java.nio.file.Files;
 31 import java.nio.file.InvalidPathException;
 32 import java.nio.file.Path;
 33 import java.text.MessageFormat;
 34 import java.util.*;




 35 import java.util.function.Supplier;
 36 import java.util.jar.Attributes;
 37 import java.util.jar.JarFile;
 38 import java.util.jar.Manifest;
 39 import java.util.stream.Collectors;
 40 import java.util.stream.Stream;

 42 /**
 43  * Extracts data needed to run application from parameters.
 44  */
 45 final class LauncherData {
 46     boolean isModular() {
 47         return moduleDescriptor != null;
 48     }
 49 
 50     String qualifiedClassName() {
 51         return qualifiedClassName;
 52     }
 53 
 54     String packageName() {
 55         int sepIdx = qualifiedClassName.lastIndexOf('.');
 56         if (sepIdx < 0) {
 57             return "";
 58         }
 59         return qualifiedClassName.substring(sepIdx + 1);
 60     }
 61 
 62     String moduleName() {
 63         verifyIsModular(true);
 64         return moduleDescriptor.name();
 65     }
 66 
 67     List<Path> modulePath() {
 68         verifyIsModular(true);
 69         return modulePath;
 70     }
 71 
 72     Path mainJarName() {
 73         verifyIsModular(false);
 74         return mainJarName;
 75     }
 76 
 77     List<Path> classPath() {
 78         return classPath;
 79     }
 80 
 81     String getAppVersion() {
 82         if (isModular()) {
 83             ModuleDescriptor.Version ver = moduleDescriptor.version().orElse(null);
 84             if (ver != null) {
 85                 return ver.toString();
 86             }
 87             return moduleDescriptor.rawVersion().orElse(null);
 88         }
 89 
 90         return null;
 91     }
 92 
 93     private LauncherData() {
 94     }
 95 
 96     private void verifyIsModular(boolean isModular) {
 97         if ((moduleDescriptor != null) != isModular) {
 98             throw new IllegalStateException();
 99         }
100     }
101 
102     static LauncherData create(Map<String, ? super Object> params) throws
103             ConfigException, IOException {
104 
105         final String mainModule = getMainModule(params);
106         if (mainModule == null) {
107             return createNonModular(params);
108         }
109 
110         LauncherData launcherData = new LauncherData();
111 
112         final int sepIdx = mainModule.indexOf("/");
113         final String moduleName;
114         if (sepIdx > 0) {
115             launcherData.qualifiedClassName = mainModule.substring(sepIdx + 1);
116             moduleName = mainModule.substring(0, sepIdx);
117         } else {
118             moduleName = mainModule;
119         }
120         launcherData.modulePath = getModulePath(params);
121 
122         launcherData.moduleDescriptor = JLinkBundlerHelper.createModuleFinder(
123                 launcherData.modulePath).find(moduleName).orElseThrow(
124                 () -> new ConfigException(MessageFormat.format(I18N.getString(
125                         "error.no-module-in-path"), moduleName), null)).descriptor();
126 
127         if (launcherData.qualifiedClassName == null) {
128             launcherData.qualifiedClassName = launcherData.moduleDescriptor.mainClass().orElseThrow(
129                     () -> new ConfigException(I18N.getString("ERR_NoMainClass"),
130                             null));
131         }
132 
133         launcherData.initClasspath(params);
134         return launcherData;
135     }
136 
137     private static LauncherData createNonModular(
138             Map<String, ? super Object> params) throws ConfigException, IOException {
139         LauncherData launcherData = new LauncherData();
140 
141         launcherData.qualifiedClassName = getMainClass(params);
142 
143         launcherData.mainJarName = getMainJarName(params);
144         if (launcherData.mainJarName == null && launcherData.qualifiedClassName
145                 == null) {
146             throw new ConfigException(I18N.getString("error.no-main-jar-parameter"),
147                     null);
148         }
149 
150         Path mainJarDir = StandardBundlerParam.SOURCE_DIR.fetchFrom(params);
151         if (mainJarDir == null && launcherData.qualifiedClassName == null) {
152             throw new ConfigException(I18N.getString("error.no-input-parameter"),
153                     null);
154         }
155 
156         final Path mainJarPath;
157         if (launcherData.mainJarName != null && mainJarDir != null) {
158             mainJarPath = mainJarDir.resolve(launcherData.mainJarName);
159             if (!Files.exists(mainJarPath)) {
160                 throw new ConfigException(MessageFormat.format(I18N.getString(
161                         "error.main-jar-does-not-exist"),
162                         launcherData.mainJarName), I18N.getString(
163                         "error.main-jar-does-not-exist.advice"));
164             }
165         } else {
166             mainJarPath = null;
167         }
168 
169         if (launcherData.qualifiedClassName == null) {
170             if (mainJarPath == null) {
171                 throw new ConfigException(I18N.getString("error.no-main-class"),
172                         I18N.getString("error.no-main-class.advice"));
173             }
174 
175             try (JarFile jf = new JarFile(mainJarPath.toFile())) {
176                 Manifest m = jf.getManifest();
177                 Attributes attrs = (m != null) ? m.getMainAttributes() : null;
178                 if (attrs != null) {
179                     launcherData.qualifiedClassName = attrs.getValue(
180                             Attributes.Name.MAIN_CLASS);
181                 }
182             }
183         }
184 
185         if (launcherData.qualifiedClassName == null) {
186             throw new ConfigException(MessageFormat.format(I18N.getString(
187                     "error.no-main-class-with-main-jar"),
188                     launcherData.mainJarName), MessageFormat.format(
189                             I18N.getString(
190                                     "error.no-main-class-with-main-jar.advice"),
191                             launcherData.mainJarName));
192         }
193 
194         launcherData.initClasspath(params);
195         return launcherData;
196     }
197 
198     private void initClasspath(Map<String, ? super Object> params)
199             throws IOException {
200         Path inputDir = StandardBundlerParam.SOURCE_DIR.fetchFrom(params);
201         if (inputDir == null) {
202             classPath = Collections.emptyList();
203         } else {
204             try (Stream<Path> walk = Files.walk(inputDir, 1)) {
205                 Set<Path> jars = walk.filter(Files::isRegularFile)
206                         .filter(file -> file.toString().endsWith(".jar"))
207                         .map(Path::getFileName)
208                         .collect(Collectors.toSet());
209                 jars.remove(mainJarName);
210                 classPath = jars.stream().sorted().collect(Collectors.toList());
211             }
212         }
213     }
214 
215     private static String getMainClass(Map<String, ? super Object> params) {
216        return getStringParam(params, Arguments.CLIOptions.APPCLASS.getId());
217     }
218 
219     private static Path getMainJarName(Map<String, ? super Object> params)
220             throws ConfigException {
221        return getPathParam(params, Arguments.CLIOptions.MAIN_JAR.getId());
222     }
223 
224     private static String getMainModule(Map<String, ? super Object> params) {
225        return getStringParam(params, Arguments.CLIOptions.MODULE.getId());
226     }
227 
228     private static String getStringParam(Map<String, ? super Object> params,
229             String paramName) {
230         Optional<Object> value = Optional.ofNullable(params.get(paramName));
231         if (value.isPresent()) {
232             return value.get().toString();
233         }
234         return null;
235     }
236 
237     private static <T> T getPathParam(Map<String, ? super Object> params,
238             String paramName, Supplier<T> func) throws ConfigException {
239         try {
240             return func.get();
241         } catch (InvalidPathException ex) {
242             throw new ConfigException(MessageFormat.format(I18N.getString(
243                     "error.not-path-parameter"), paramName,
244                     ex.getLocalizedMessage()), null, ex);
245         }
246     }
247 
248     private static Path getPathParam(Map<String, ? super Object> params,
249             String paramName) throws ConfigException {
250         return getPathParam(params, paramName, () -> {
251             String value = getStringParam(params, paramName);
252             Path result = null;
253             if (value != null) {
254                 result = Path.of(value);
255             }
256             return result;
257         });
258     }
259 
260     private static List<Path> getModulePath(Map<String, ? super Object> params)
261             throws ConfigException {
262         return getPathListParameter(Arguments.CLIOptions.MODULE_PATH.getId(),
263                 params);
264     }
265 
266     private static List<Path> getPathListParameter(String paramName,
267             Map<String, ? super Object> params) throws ConfigException {
268         return getPathParam(params, paramName, () -> {
269             String value = params.getOrDefault(paramName, "").toString();
270         return List.of(value.split(File.pathSeparator)).stream()
271                 .map(Path::of)
272                 .collect(Collectors.toUnmodifiableList());
273         });
274     }
275 
276     private String qualifiedClassName;
277     private Path mainJarName;
278     private List<Path> classPath;
279     private List<Path> modulePath;
280     private ModuleDescriptor moduleDescriptor;
281 }
