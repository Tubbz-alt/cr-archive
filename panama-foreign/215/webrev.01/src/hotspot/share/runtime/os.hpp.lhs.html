214   // Interface for detecting multiprocessor system
215   static inline bool is_MP() {
216     // During bootstrap if _processor_count is not yet initialized
217     // we claim to be MP as that is safest. If any platform has a
218     // stub generator that might be triggered in this phase and for
219     // which being declared MP when in fact not, is a problem - then
220     // the bootstrap routine for the stub generator needs to check
221     // the processor count directly and leave the bootstrap routine
222     // in place until called after initialization has ocurred.
223     return (_processor_count != 1);
224   }

893 #ifndef _WINDOWS
894   // Suspend/resume support
895   // Protocol:
896   //
897   // a thread starts in SR_RUNNING
898   //
899   // SR_RUNNING can go to
900   //   * SR_SUSPEND_REQUEST when the WatcherThread wants to suspend it
901   // SR_SUSPEND_REQUEST can go to
902   //   * SR_RUNNING if WatcherThread decides it waited for SR_SUSPENDED too long (timeout)
903   //   * SR_SUSPENDED if the stopped thread receives the signal and switches state
904   // SR_SUSPENDED can go to
905   //   * SR_WAKEUP_REQUEST when the WatcherThread has done the work and wants to resume
906   // SR_WAKEUP_REQUEST can go to
907   //   * SR_RUNNING when the stopped thread receives the signal
908   //   * SR_WAKEUP_REQUEST on timeout (resend the signal and try again)
909   class SuspendResume {
910    public:
911     enum State {
912       SR_RUNNING,
913       SR_SUSPEND_REQUEST,
914       SR_SUSPENDED,
915       SR_WAKEUP_REQUEST
916     };
917 
918   private:
919     volatile State _state;
920 
921   private:
922     /* try to switch state from state "from" to state "to"
923      * returns the state set after the method is complete
924      */
925     State switch_state(State from, State to);
926 
927   public:
928     SuspendResume() : _state(SR_RUNNING) { }
929 
930     State state() const { return _state; }
931 
932     State request_suspend() {
933       return switch_state(SR_RUNNING, SR_SUSPEND_REQUEST);
934     }
935 
936     State cancel_suspend() {
937       return switch_state(SR_SUSPEND_REQUEST, SR_RUNNING);
938     }
939 
940     State suspended() {
941       return switch_state(SR_SUSPEND_REQUEST, SR_SUSPENDED);
942     }
943 
944     State request_wakeup() {
945       return switch_state(SR_SUSPENDED, SR_WAKEUP_REQUEST);
946     }
947 
948     State running() {
949       return switch_state(SR_WAKEUP_REQUEST, SR_RUNNING);
950     }
951 
952     bool is_running() const {
953       return _state == SR_RUNNING;
954     }
955 
956     bool is_suspended() const {
957       return _state == SR_SUSPENDED;
958     }
959   };
960 #endif // !WINDOWS
