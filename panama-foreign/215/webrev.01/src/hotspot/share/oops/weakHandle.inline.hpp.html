25 #ifndef SHARE_OOPS_WEAKHANDLE_INLINE_HPP
26 #define SHARE_OOPS_WEAKHANDLE_INLINE_HPP
27 
28 #include "oops/weakHandle.hpp"
29 #include "oops/access.inline.hpp"
30 
31 inline oop WeakHandle::resolve() const {
32   assert(!is_null(), "Must be created");
33   return NativeAccess<ON_PHANTOM_OOP_REF>::oop_load(_obj);
34 }
35 
36 inline oop WeakHandle::peek() const {
37   assert(!is_null(), "Must be created");
38   return NativeAccess<ON_PHANTOM_OOP_REF | AS_NO_KEEPALIVE>::oop_load(_obj);
39 }
40 
41 inline void WeakHandle::replace(oop with_obj) {
42   NativeAccess<ON_PHANTOM_OOP_REF>::oop_store(_obj, with_obj);
43 }
44 
45 #endif // SHARE_OOPS_WEAKHANDLE_INLINE_HPP
