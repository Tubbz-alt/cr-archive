 57     // Caching to reduce allocation pressure and heap usage
 58     private final AnnotationElement RELATIONAL = new AnnotationElement(Relational.class);
 59     private final AnnotationElement ENABLED = new AnnotationElement(Enabled.class, false);
 60     private final AnnotationElement THRESHOLD = new AnnotationElement(Threshold.class, "0 ns");
 61     private final AnnotationElement STACK_TRACE = new AnnotationElement(StackTrace.class, true);
 62     private final AnnotationElement TRANSITION_TO = new AnnotationElement(TransitionTo.class);
 63     private final AnnotationElement TRANSITION_FROM = new AnnotationElement(TransitionFrom.class);
 64     private final AnnotationElement EXPERIMENTAL = new AnnotationElement(Experimental.class);
 65     private final AnnotationElement UNSIGNED = new AnnotationElement(Unsigned.class);
 66     private final List<Object> SMALL_TEMP_LIST = new ArrayList<>();
 67     private final Type LABEL_TYPE = TypeLibrary.createAnnotationType(Label.class);
 68     private final Type DESCRIPTION_TYPE = TypeLibrary.createAnnotationType(Description.class);

 72     // <Event>, <Type> and <Relation>
 73     private final static class TypeElement {
 74         private final List<FieldElement> fields;
 75         private final String name;
 76         private final String label;
 77         private final String description;
 78         private final String category;
 79         private final String period;
 80         private final boolean thread;
 81         private final boolean startTime;
 82         private final boolean stackTrace;
 83         private final boolean cutoff;
 84         private final boolean isEvent;
 85         private final boolean isRelation;
 86         private final boolean experimental;
 87         private final long id;
 88 
 89         public TypeElement(DataInputStream dis) throws IOException {
 90             int fieldCount = dis.readInt();
 91             fields = new ArrayList<>(fieldCount);
 92             for (int i = 0; i < fieldCount; i++) {
 93                 fields.add(new FieldElement(dis));
 94             }
 95             name = dis.readUTF();
 96             label = dis.readUTF();
 97             description = dis.readUTF();
 98             category = dis.readUTF();
 99             thread = dis.readBoolean();
100             stackTrace = dis.readBoolean();
101             startTime = dis.readBoolean();
102             period = dis.readUTF();
103             cutoff = dis.readBoolean();
104             experimental = dis.readBoolean();
105             id = dis.readLong();
106             isEvent = dis.readBoolean();
107             isRelation = dis.readBoolean();
108         }
109     }

113         private final String name;
114         private final String label;
115         private final String description;
116         private final String typeName;
117         private final String annotations;
118         private final String transition;
119         private final String relation;
120         private final boolean constantPool;
121         private final boolean array;
122         private final boolean experimental;
123         private final boolean unsigned;

125         public FieldElement(DataInputStream dis) throws IOException {
126             name = dis.readUTF();
127             typeName = dis.readUTF();
128             label = dis.readUTF();
129             description = dis.readUTF();
130             constantPool = dis.readBoolean();
131             array = dis.readBoolean();
132             unsigned = dis.readBoolean();
133             annotations = dis.readUTF();
134             transition = dis.readUTF();
135             relation = dis.readUTF();
136             experimental = dis.readBoolean();
137         }

140     private final List<TypeElement> types;
141     private final Map<String, List<AnnotationElement>> anotationElements = new HashMap<>(20);
142     private final Map<String, AnnotationElement> categories = new HashMap<>();
143 
144     MetadataLoader(DataInputStream dis) throws IOException {
145         SMALL_TEMP_LIST.add(this); // add any object to expand list
146         int typeCount = dis.readInt();
147         types = new ArrayList<>(typeCount);
148         for (int i = 0; i < typeCount; i++) {
149             types.add(new TypeElement(dis));
150         }
151     }

153     private List<AnnotationElement> createAnnotationElements(String annotation) throws InternalError {
154         String[] annotations = annotation.split(",");
155         List<AnnotationElement> annotationElements = new ArrayList<>();
156         for (String a : annotations) {
157             a = a.trim();
158             int leftParenthesis = a.indexOf("(");
159             if (leftParenthesis == -1) {
160                 annotationElements.add(new AnnotationElement(createAnnotationClass(a)));
161             } else {
162                 int rightParenthesis = a.lastIndexOf(")");
163                 if (rightParenthesis == -1) {
164                     throw new InternalError("Expected closing parenthesis for 'XMLContentType'");
165                 }
166                 String value = a.substring(leftParenthesis + 1, rightParenthesis);
167                 String type = a.substring(0, leftParenthesis);
168                 annotationElements.add(new AnnotationElement(createAnnotationClass(type), value));
169             }
170         }
171         return annotationElements;
172     }
173 
174     @SuppressWarnings("unchecked")
175     private Class<? extends Annotation> createAnnotationClass(String type) {
176         try {
177             if (!type.startsWith("jdk.jfr.")) {
178                 throw new IllegalStateException("Incorrect type " + type + ". Annotation class must be located in jdk.jfr package.");
179             }
180             Class<?> c = Class.forName(type, true, null);
181             return (Class<? extends Annotation>) c;
182         } catch (ClassNotFoundException cne) {
183             throw new IllegalStateException(cne);
184         }
185     }

187     public static List<Type> createTypes() throws IOException {
188         try (DataInputStream dis = new DataInputStream(
189                 SecuritySupport.getResourceAsStream("/jdk/jfr/internal/types/metadata.bin"))) {
190             MetadataLoader ml = new MetadataLoader(dis);
191             return ml.buildTypes();
192         } catch (Exception e) {
193             throw new InternalError(e);
194         }
195     }

197     private List<Type> buildTypes() {
198         Map<String, Type> typeMap = buildTypeMap();
199         Map<String, AnnotationElement> relationMap = buildRelationMap(typeMap);
200         addFields(typeMap, relationMap);
201         return new ArrayList<>(typeMap.values());
202     }
203 
204     private Map<String, AnnotationElement> buildRelationMap(Map<String, Type> typeMap) {
205         Map<String, AnnotationElement> relationMap = new HashMap<>(20);
206         for (TypeElement t : types) {
207             if (t.isRelation) {
208                 Type relationType = typeMap.get(t.name);
209                 AnnotationElement ae = PrivateAccess.getInstance().newAnnotation(relationType, Collections.emptyList(), true);
210                 relationMap.put(t.name, ae);
211             }
212         }
213         return relationMap;
214     }
215 
216     private void addFields(Map<String, Type> lookup, Map<String, AnnotationElement> relationMap) {
217         for (TypeElement te : types) {
218             Type type = lookup.get(te.name);
219             if (te.isEvent) {
220                 boolean periodic = !te.period.isEmpty();
221                 TypeLibrary.addImplicitFields(type, periodic, te.startTime && !periodic, te.thread, te.stackTrace && !periodic, te.cutoff);
222             }
223             for (FieldElement f : te.fields) {
224                 Type fieldType = Type.getKnownType(f.typeName);
225                 if (fieldType == null) {
226                     fieldType = Objects.requireNonNull(lookup.get(f.typeName));
227                 }
228                 List<AnnotationElement> aes = new ArrayList<>();
229                 if (f.unsigned) {
230                     aes.add(UNSIGNED);
231                 }
232                 if (!f.annotations.isEmpty()) {
233                     var ae = anotationElements.get(f.annotations);
234                     if (ae == null) {
235                         ae = createAnnotationElements(f.annotations);
236                         anotationElements.put(f.annotations, ae);
237                     }
238                     aes.addAll(ae);
239                 }
240                 if (!f.relation.isEmpty()) {
241                     AnnotationElement t = relationMap.get(f.relation);
242                     aes.add(Objects.requireNonNull(t));
243                 }
244                 if (!f.label.isEmpty()) {
245                     aes.add(newAnnotation(LABEL_TYPE, f.label));
246                 }
247                 if (f.experimental) {
248                     aes.add(EXPERIMENTAL);
249                 }
250                 if (!f.description.isEmpty()) {
251                     aes.add(newAnnotation(DESCRIPTION_TYPE, f.description));
252                 }
253                 if ("from".equals(f.transition)) {
254                     aes.add(TRANSITION_FROM);
255                 }
256                 if ("to".equals(f.transition)) {
257                     aes.add(TRANSITION_TO);
258                 }
259                 type.add(PrivateAccess.getInstance().newValueDescriptor(f.name, fieldType, aes, f.array ? 1 : 0, f.constantPool, null));
260             }
261         }
262     }

264     private AnnotationElement newAnnotation(Type type, Object value) {
265         SMALL_TEMP_LIST.set(0, value);
266         return PrivateAccess.getInstance().newAnnotation(type, SMALL_TEMP_LIST, true);
267     }

269     private Map<String, Type> buildTypeMap() {
270         Map<String, Type> typeMap = new HashMap<>(2 * types.size());
271         Map<String, Type> knownTypeMap = new HashMap<>(20);
272         for (Type kt : Type.getKnownTypes()) {
273             typeMap.put(kt.getName(), kt);
274             knownTypeMap.put(kt.getName(), kt);
275         }
276         for (TypeElement t : types) {
277             List<AnnotationElement> aes = new ArrayList<>();
278             if (!t.category.isEmpty()) {
279                 AnnotationElement cat = categories.get(t.category);
280                 if (cat == null) {
281                     String[] segments = buildCategorySegments(t.category);
282                     cat = newAnnotation(CATEGORY_TYPE, segments);
283                     categories.put(t.category, cat);
284                 }
285                 aes.add(cat);
286             }
287             if (!t.label.isEmpty()) {
288                 aes.add(newAnnotation(LABEL_TYPE, t.label));
289             }
290             if (!t.description.isEmpty()) {
291                 aes.add(newAnnotation(DESCRIPTION_TYPE, t.description));
292             }
293             if (t.isEvent) {
294                 if (!t.period.isEmpty()) {
295                     aes.add(newAnnotation(PERIOD_TYPE, t.period));
296                 } else {
297                     if (t.startTime) {
298                         aes.add(THRESHOLD);
299                     }
300                     if (t.stackTrace) {
301                         aes.add(STACK_TRACE);
302                     }
303                 }
304                 if (t.cutoff) {
305                     aes.add(new AnnotationElement(Cutoff.class, Cutoff.INFINITY));
306                 }
307             }
308             if (t.experimental) {
309                 aes.add(EXPERIMENTAL);
310             }
311             Type type;
312             if (t.isEvent) {
313                 aes.add(ENABLED);
314                 type = new PlatformEventType(t.name, t.id, false, true);
315             } else {
316                 type = knownTypeMap.get(t.name);
317                 if (type == null) {
318                     if (t.isRelation) {
319                         type = new Type(t.name, Type.SUPER_TYPE_ANNOTATION, t.id);
320                         aes.add(RELATIONAL);
321                     } else {
322                         type = new Type(t.name, null, t.id);
323                     }
324                 }
325             }
326             type.setAnnotations(aes);
327             typeMap.put(t.name, type);
328         }
329         return typeMap;
330     }
331 
332     private String[] buildCategorySegments(String category) {
333         String[] segments = category.split(",");
334         for (int i = 0; i < segments.length; i++) {
335             segments[i] = segments[i].trim();
336         }
337         return segments;
338     }
