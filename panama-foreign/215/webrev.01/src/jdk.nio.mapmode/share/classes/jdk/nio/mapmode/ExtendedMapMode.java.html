26 package jdk.nio.mapmode;
27 
28 import java.nio.MappedByteBuffer;
29 import java.nio.channels.FileChannel;
30 import java.nio.channels.FileChannel.MapMode;
31 
32 /**
33  * JDK-specific map modes.
34  *
35  * @since 14
36  * @see java.nio.channels.FileChannel#map
37  */
38 public class ExtendedMapMode {
39     private ExtendedMapMode() { }
40 
41     /**
42      * File mapping mode for a read-only mapping of a file backed by
43      * non-volatile RAM.
44      *
45      * <p> The {@linkplain FileChannel#map map} method throws
46      * {@linkplain UnsupportedOperationException} when this map mode
47      * is used on an implementation that does not support it.
48      *
49      * @implNote On Linux, the {@code MAP_SYNC} and {@code
50      * MAP_SHARED_VALIDATE} flags are specified to {@code mmap} when
51      * mapping the file into memory.
52      */
53     public static final MapMode READ_ONLY_SYNC = jdk.internal.misc.ExtendedMapMode.READ_ONLY_SYNC;
54 
55     /**
56      * File mapping mode for a read-write mapping of a file backed by
57      * non-volatile RAM. {@linkplain MappedByteBuffer#force force}
58      * operations on a buffer created with this mode will be performed
59      * using cache line writeback rather than proceeding via a file
60      * device flush.
61      *
62      * <p> The {@linkplain FileChannel#map map} method throws
63      * {@linkplain UnsupportedOperationException} when this map mode
64      * is used on an implementation that does not support it.
65      *
66      * @implNote On Linux, the {@code MAP_SYNC} and {@code
67      * MAP_SHARED_VALIDATE} flags are specified to {@code mmap} when
68      * mapping the file into memory.
69      */
70     public static final MapMode READ_WRITE_SYNC = jdk.internal.misc.ExtendedMapMode.READ_WRITE_SYNC;
71 
72 }
