32 // create Java String array of 5 elements
33 var StringArray = Java.type("java.lang.String[]");
34 var a = new StringArray(5);
35 
36 // Accessing elements and length access is by usual Java syntax
37 a[0] = "scripting is great!";
38 print(a.length);
39 print(a[0]);

41 // convert a script array to Java array
42 var anArray = [1, "13", false];
43 var javaIntArray = Java.to(anArray, "int[]");
44 print(javaIntArray[0]);// prints 1
45 print(javaIntArray[1]); // prints 13, as string "13" was converted to number 13 as per ECMAScript ToNumber conversion
46 print(javaIntArray[2]);// prints 0, as boolean false was converted to number 0 as per ECMAScript ToNumber conversion

48 // convert a Java array to a JavaScript array
49 var File = Java.type("java.io.File");
50 var listCurDir = new File(".").listFiles();
