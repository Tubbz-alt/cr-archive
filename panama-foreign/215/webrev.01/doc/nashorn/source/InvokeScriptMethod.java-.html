32 import javax.script.Invocable;
33 import javax.script.ScriptEngine;
34 import javax.script.ScriptEngineManager;
35 
36 @SuppressWarnings("javadoc")
37 public class InvokeScriptMethod {
38     public static void main(final String[] args) throws Exception {
39         final ScriptEngineManager manager = new ScriptEngineManager();
40         final ScriptEngine engine = manager.getEngineByName("nashorn");
41 
42         // JavaScript code in a String. This code defines a script object 'obj'
43         // with one method called 'hello'.
44         final String script = "var obj = new Object(); obj.hello = function(name) { print('Hello, ' + name); }";
45         // evaluate script
46         engine.eval(script);
47 
48         // javax.script.Invocable is an optional interface.
49         // Check whether your script engine implements or not!
50         // Note that the JavaScript engine implements Invocable interface.
51         final Invocable inv = (Invocable) engine;
52 
53         // get script object on which we want to call the method
54         final Object obj = engine.get("obj");
55 
56         // invoke the method named "hello" on the script object "obj"
57         inv.invokeMethod(obj, "hello", "Script Method !!" );
58     }
59 }
