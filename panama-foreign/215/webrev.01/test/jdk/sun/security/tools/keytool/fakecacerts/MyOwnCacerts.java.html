24 /*

26  * @bug 8244148
27  * @library /test/lib
28  * @library /test/jdk/sun/security/util/module_patch
29  * @build java.base/sun.security.util.FilePaths
30  * @modules java.base/jdk.internal.misc
31  * @run main MyOwnCacerts
32  */
33 
34 import jdk.test.lib.SecurityTools;
35 import jdk.test.lib.process.OutputAnalyzer;
36 
37 public class MyOwnCacerts {
38 
39     // The --patch-module must be explicitly specified on the keytool
40     // command line because it's in a separate process
41     private static final String PATCH_OPTION;
42 
43     static {
44         String tmp = "";
45         for (String a : jdk.internal.misc.VM.getRuntimeArguments()) {
46             if (a.startsWith("--patch-module")) {
47                 tmp = "-J" + a + " ";
48                 break;
49             }
50         }
51         PATCH_OPTION = tmp;
52     }
53 
54     public static void main(String[] args) throws Exception {
55         kt("-keystore mycacerts -genkeypair -alias a" +
56                 " -dname CN=root -keyalg EC -storepass changeit")
57                 .shouldContain("Warning: use -cacerts option");
58         kt("-list -cacerts -storepass changeit")
59                 .shouldContain("Your keystore contains 1 entry");
60     }
61 
62     static OutputAnalyzer kt(String s) throws Exception {
63         return SecurityTools.keytool(PATCH_OPTION + s);
64     }
65 }
