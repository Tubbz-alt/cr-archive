24 import jdk.jpackage.test.TKit;
25 import jdk.jpackage.test.PackageTest;
26 import jdk.jpackage.test.PackageType;
27 import jdk.jpackage.test.Annotations.Test;
28 
29 /**
30  * Test --win-per-user-install, --win-menu, --win-menu-group parameters.
31  * Output of the test should be WinPerUserInstallTest-1.0.exe installer. The
32  * output installer should provide the same functionality as the default
33  * installer (see description of the default installer in
34  * SimplePackageTest.java) plus it should create application menu in Windows
35  * Menu and installation should be per user and not machine wide.
36  */
37 
38 /*
39  * @test
40  * @summary jpackage with --win-per-user-install, --win-menu, --win-menu-group
41  * @library ../helpers
42  * @key jpackagePlatformPackage
43  * @build jdk.jpackage.test.*
44  * @requires (os.family == "windows")
45  * @modules jdk.incubator.jpackage/jdk.incubator.jpackage.internal
46  * @compile WinPerUserInstallTest.java
47  * @run main/othervm/timeout=360 -Xmx512m jdk.jpackage.test.Main
48  *  --jpt-run=WinPerUserInstallTest
49  */
50 
51 public class WinPerUserInstallTest {
52     @Test
53     public static void test() {
54         new PackageTest()
55         .forTypes(PackageType.WINDOWS)
56         .configureHelloApp()
57         .addInitializer(cmd -> cmd.addArguments(
58                 "--win-menu",
59                 "--win-menu-group", "WinPerUserInstallTest_MenuGroup",
60                 "--win-per-user-install"))
61         .run();
62     }
63 }
