24 /*
25  * @test
26  * @bug 8246078
27  * @library ../../lib
28  * @modules jdk.javadoc/jdk.javadoc.internal.tool
29  * @build javadoc.tester.*
30  * @run main TestHelpPage
31  */
32 
33 import javadoc.tester.JavadocTester;
34 
35 import java.util.Arrays;
36 import java.util.List;
37 import java.util.regex.Matcher;
38 import java.util.regex.Pattern;
39 
40 public class TestHelpPage extends JavadocTester {
41 
42     public static void main(String... args) throws Exception {
43         TestHelpPage tester = new TestHelpPage();
44         tester.runTests();
45     }
46 
47     @Test
48     public void test() {
49 
50         setOutputDirectoryCheck(DirectoryCheck.NONE);
51         setAutomaticCheckLinks(false);
52         setAutomaticCheckAccessibility(false);
53 
54         javadoc("--version");
55         List<String> lines = getOutputLines(Output.OUT);
56         System.out.println("lines: " + Arrays.toString(lines.toArray()));
57         String firstLine = lines.get(0);
58         Matcher m = Pattern.compile("javadoc\\s+(.*)").matcher(firstLine);
59         m.matches();
60         String vstr = m.group(1);
61         System.out.printf("vstr='%s'%n", vstr);
62         Runtime.Version v = Runtime.Version.parse(vstr);
63         System.out.printf("v=%s, v.feature()=%s%n", v, v.feature());
64 
65         javadoc("-d", "out",
66                 "-sourcepath", testSrc,
67                 testSrc("TestHelpPage.java"));
68         checkExit(Exit.OK);
69 
70         checking("Reference to a particular version");
71         Pattern searchSpecLink = Pattern.compile(
72                 "\\Q<a href=\"https://docs.oracle.com/en/java/javase/\\E(\\d+)\\Q/docs/specs/javadoc/javadoc-search-spec.html\">\\E");
73 
74         String helpContents = readOutputFile("help-doc.html");
75         Matcher m2 = searchSpecLink.matcher(helpContents);
76         if (!m2.find()) {
77             failed("Reference not found: " + helpContents);
78             return;
79         }
80 
81         if (!String.valueOf(v.feature()).equals(m2.group(1))) {
82             failed("Reference to a wrong version: " + m2.group(1));
83             return;
84         }
85 
86         boolean foundMore = false;
87         while (m2.find()) {
88             // print all found before failing
89             foundMore = true;
90             System.out.println(m2.group(0));
91         }
92         if (foundMore) {
93             failed("Multiple references: " + helpContents);
94         } else {
95             passed("All good");
96         }
97     }
98 }
