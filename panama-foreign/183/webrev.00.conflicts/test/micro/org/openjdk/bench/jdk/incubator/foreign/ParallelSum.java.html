229     static class SumUnsafe extends RecursiveTask<Integer> {
230 
231         final static int SPLIT_THRESHOLD = 4 * 1024 * 8;
232 
233         private final long address;
234         private final int start, length;
235 
236         SumUnsafe(long address, int start, int length) {
237             this.address = address;
238             this.start = start;
239             this.length = length;
240         }
241 
242         @Override
243         protected Integer compute() {
244             if (length > SPLIT_THRESHOLD) {
245                 SumUnsafe s1 = new SumUnsafe(address, start, length / 2);
246                 SumUnsafe s2 = new SumUnsafe(address, length / 2, length / 2);
247                 s1.fork();
248                 s2.fork();
249                 return s1.join() + s2.join();
250             } else {
251                 int res = 0;
252                 for (int i = 0; i < length; i += CARRIER_SIZE) {
253                     res += unsafe.getInt(start + address + i);
254                 }
255                 return res;
256             }
257         }
258     }

260     static class SumSegment extends CountedCompleter<Integer> {
261 
262         final static int SPLIT_THRESHOLD = 1024 * 8;
263 
264         int localSum = 0;
265         private final ToIntFunction<MemorySegment> mapper;
266         List<SumSegment> children = new LinkedList<>();
267 
268         private Spliterator<MemorySegment> segmentSplitter;
269 
270         SumSegment(Spliterator<MemorySegment> segmentSplitter, ToIntFunction<MemorySegment> mapper) {
271             this(null, segmentSplitter, mapper);
272         }
273 
274         SumSegment(SumSegment parent, Spliterator<MemorySegment> segmentSplitter, ToIntFunction<MemorySegment> mapper) {
275             super(parent);
276             this.segmentSplitter = segmentSplitter;
277             this.mapper = mapper;
278         }
279 
280         @Override
281         public void compute() {
282             Spliterator<MemorySegment> sub;
283             while (segmentSplitter.estimateSize() > SPLIT_THRESHOLD &&
284                     (sub = segmentSplitter.trySplit()) != null) {
285                 addToPendingCount(1);
286                 SumSegment child = new SumSegment(this, sub, mapper);
287                 children.add(child);
288                 child.fork();
289             }
290             segmentSplitter.forEachRemaining(s -> {
291                 localSum += mapper.applyAsInt(s);
292             });
293             propagateCompletion();
294         }
295 
296         @Override
297         public Integer getRawResult() {
298             int sum = localSum;
299             for (SumSegment c : children) {
300                 sum += c.getRawResult();
301             }
302             children = null;
303             return sum;
304         }
