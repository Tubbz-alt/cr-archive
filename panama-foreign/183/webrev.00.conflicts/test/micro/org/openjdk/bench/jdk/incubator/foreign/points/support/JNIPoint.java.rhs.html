27 package org.openjdk.bench.jdk.incubator.foreign.points.support;
28 
29 public class JNIPoint implements AutoCloseable {
30 
31     static {
32         System.loadLibrary("JNIPoint");
33     }
34 
35     private final long peer;
36 
37     public JNIPoint(int x, int y) {
38         peer = allocate();
39         setX(peer, x);
40         setY(peer, y);
41     }
42 
43     public void free() {
44         free(peer);
45     }
46 
47     public int getX() {
48         return getX(peer);
49     }
50 
51     public void setX(int value) {
52         setX(peer, value);
53     }
54 
55     public int getY() {
56         return getY(peer);
57     }
58 
59     public void setY(int value) {
60         setY(peer, value);
61     }
62 
63     private static native long allocate();
64     private static native void free(long ptr);
65 
66     private static native int getX(long ptr);
67     private static native void setX(long ptr, int x);
68 
69     private static native int getY(long ptr);
70     private static native void setY(long ptr, int y);
71 
72     @Override
73     public void close() {
74         free();
75     }
76 }
