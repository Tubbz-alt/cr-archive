 27 package org.openjdk.bench.jdk.incubator.foreign;
 28 
 29 import jdk.incubator.foreign.MemoryAddress;
 30 import jdk.incubator.foreign.MemoryLayout;
 31 import jdk.incubator.foreign.MemorySegment;
 32 import org.openjdk.jmh.annotations.Benchmark;
 33 import org.openjdk.jmh.annotations.BenchmarkMode;
 34 import org.openjdk.jmh.annotations.Fork;
 35 import org.openjdk.jmh.annotations.Measurement;
 36 import org.openjdk.jmh.annotations.Mode;
 37 import org.openjdk.jmh.annotations.OutputTimeUnit;
 38 import org.openjdk.jmh.annotations.Setup;
 39 import org.openjdk.jmh.annotations.State;
 40 import org.openjdk.jmh.annotations.TearDown;
 41 import org.openjdk.jmh.annotations.Warmup;
 42 import sun.misc.Unsafe;
 43 
 44 import java.lang.invoke.VarHandle;
 45 import java.nio.ByteBuffer;
 46 import java.nio.ByteOrder;
 47 import java.util.concurrent.TimeUnit;
 48 
 49 import static jdk.incubator.foreign.MemoryLayout.PathElement.sequenceElement;
 50 import static jdk.incubator.foreign.MemoryLayouts.JAVA_INT;
 51 
 52 @BenchmarkMode(Mode.AverageTime)
 53 @Warmup(iterations = 5, time = 500, timeUnit = TimeUnit.MILLISECONDS)
 54 @Measurement(iterations = 10, time = 500, timeUnit = TimeUnit.MILLISECONDS)
 55 @State(org.openjdk.jmh.annotations.Scope.Thread)
 56 @OutputTimeUnit(TimeUnit.MILLISECONDS)
 57 @Fork(3)
 58 public class LoopOverNonConstantHeap {
 59 
 60     static final Unsafe unsafe = Utils.unsafe;
 61 
 62     static final int ELEM_SIZE = 1_000_000;
 63     static final int CARRIER_SIZE = (int)JAVA_INT.byteSize();
 64     static final int ALLOC_SIZE = ELEM_SIZE * CARRIER_SIZE;
 65     static final int UNSAFE_BYTE_BASE = unsafe.arrayBaseOffset(byte[].class);
 66 
 67     static final VarHandle VH_int = MemoryLayout.ofSequence(JAVA_INT).varHandle(int.class, sequenceElement());
 68     MemorySegment segment;
 69     byte[] base;
 70 
 71     ByteBuffer byteBuffer;
 72 
 73     @Setup
 74     public void setup() {
 75         base = new byte[ALLOC_SIZE];
 76         for (int i = 0; i < ELEM_SIZE; i++) {
 77             unsafe.putInt(base, UNSAFE_BYTE_BASE + (i * CARRIER_SIZE) , i);
 78         }
 79         segment = MemorySegment.ofArray(base);
 80         byteBuffer = ByteBuffer.wrap(base).order(ByteOrder.nativeOrder());
 81     }
 82 
 83     @TearDown
 84     public void tearDown() {
 85         segment.close();
 86     }
 87 
 88     @Benchmark
 89     @OutputTimeUnit(TimeUnit.NANOSECONDS)
 90     public int unsafe_get() {
 91         return unsafe.getInt(base, UNSAFE_BYTE_BASE);
 92     }
 93 
 94     @Benchmark
 95     @OutputTimeUnit(TimeUnit.NANOSECONDS)
 96     public int segment_get() {
 97         return (int) VH_int.get(segment.baseAddress(), 0L);
 98     }
 99 
100     @Benchmark
101     @OutputTimeUnit(TimeUnit.NANOSECONDS)
102     public int BB_get() {
103         return byteBuffer.getInt(0);
104     }
105 
106     @Benchmark
107     public int unsafe_loop() {
108         int res = 0;
109         for (int i = 0; i < ELEM_SIZE; i ++) {
110             res += unsafe.getInt(base, UNSAFE_BYTE_BASE + (i * CARRIER_SIZE));
111         }
112         return res;
113     }
114 
115     @Benchmark
116     public int segment_loop() {
117         int sum = 0;
118         MemoryAddress base = segment.baseAddress();
119         for (int i = 0; i < ELEM_SIZE; i++) {
120             sum += (int) VH_int.get(base, (long) i);
121         }
122         return sum;
123     }
124 
125     @Benchmark
126     public int segment_loop_slice() {
127         int sum = 0;
128         MemoryAddress base = segment.asSlice(0, segment.byteSize()).baseAddress();
129         for (int i = 0; i < ELEM_SIZE; i++) {
130             sum += (int) VH_int.get(base, (long) i);
131         }
132         return sum;
133     }
134 
135     @Benchmark
136     public int segment_loop_readonly() {
137         int sum = 0;
138         MemoryAddress base = segment.withAccessModes(MemorySegment.READ).baseAddress();
139         for (int i = 0; i < ELEM_SIZE; i++) {
140             sum += (int) VH_int.get(base, (long) i);
141         }
142         return sum;
143     }
144 
145     @Benchmark
146     public int BB_loop() {
147         int sum = 0;
148         ByteBuffer bb = byteBuffer;
149         for (int i = 0; i < ELEM_SIZE; i++) {
150             sum += bb.getInt(i * CARRIER_SIZE);
151         }
152         return sum;
153     }
154 
155 }
