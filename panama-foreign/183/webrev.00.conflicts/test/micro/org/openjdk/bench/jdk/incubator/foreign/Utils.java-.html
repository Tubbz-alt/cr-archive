23 package org.openjdk.bench.jdk.incubator.foreign;
24 
25 import sun.misc.Unsafe;
26 
27 import java.lang.reflect.Field;
28 
29 public class Utils {
30 
31     static final Unsafe unsafe;
32 
33     //setup unsafe
34     static {
35         try {
36             Field f = Unsafe.class.getDeclaredField("theUnsafe");
37             f.setAccessible(true);
38             unsafe = (Unsafe) f.get(null);
39         } catch (ReflectiveOperationException ex) {
40             throw new IllegalStateException();
41         }
42     }
43 }
