 49     static VarHandle BYTE_HANDLE = MemoryLayouts.JAVA_BYTE.varHandle(byte.class);

 52     static VarHandle INT_HANDLE = MemoryLayouts.JAVA_INT.varHandle(int.class);
 53     static VarHandle LONG_HANDLE = MemoryLayouts.JAVA_LONG.varHandle(long.class);
 54     static VarHandle FLOAT_HANDLE = MemoryLayouts.JAVA_FLOAT.varHandle(float.class);
 55     static VarHandle DOUBLE_HANDLE = MemoryLayouts.JAVA_DOUBLE.varHandle(double.class);

 57     static boolean compareSegmentsByte(Collection<MemorySegment> segments1, Collection<MemorySegment> segments2, boolean isOrdered) {
 58         Function<MemorySegment, Byte> mapper = segment -> (byte)BYTE_HANDLE.get(segment.baseAddress());
 59         List<Byte> list1 = segments1.stream()
 60                 .map(mapper)
 61                 .collect(Collectors.toList());
 62         List<Byte> list2 = segments2.stream()
 63                 .map(mapper)
 64                 .collect(Collectors.toList());
 65         return list1.equals(list2);
 66     }

123     static Consumer<MemorySegment> segmentCopier(Consumer<MemorySegment> input) {
124         return segment -> {
125             MemorySegment copy = MemorySegment.ofArray(new byte[(int)segment.byteSize()]);
126             MemoryAddress.copy(segment.baseAddress(), copy.baseAddress(), segment.byteSize());
127             input.accept(copy);
128         };
129     }

142     static void initSegment(MemorySegment segment) {
143         for (int i = 0 ; i < segment.byteSize() ; i++) {
144             BYTE_HANDLE.set(segment.baseAddress(), (byte)i);
145         }
