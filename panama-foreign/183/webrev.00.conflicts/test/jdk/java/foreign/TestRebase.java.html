 34 import jdk.incubator.foreign.MemoryAddress;
 35 import jdk.incubator.foreign.MemoryLayouts;
 36 import jdk.incubator.foreign.MemorySegment;
 37 import org.testng.annotations.DataProvider;
 38 import org.testng.annotations.Test;
 39 
 40 import java.lang.invoke.VarHandle;
 41 import java.util.ArrayList;
 42 import java.util.List;
 43 import java.util.function.IntFunction;
 44 
 45 import static org.testng.Assert.assertEquals;
 46 import static org.testng.Assert.assertTrue;
 47 import static org.testng.Assert.fail;
 48 
 49 public class TestRebase {
 50 
 51     static VarHandle BYTE_VH = MemoryLayouts.JAVA_BYTE.varHandle(byte.class);
 52 
 53     @Test(dataProvider = "slices")
 54     public void testRebase(SegmentSlice s1, SegmentSlice s2) {
 55         if (s1.contains(s2)) {
 56             //check that an address and its rebased counterpart point to same element
 57             MemoryAddress base = s2.segment.baseAddress();
 58             MemoryAddress rebased = base.rebase(s1.segment);
 59             for (int i = 0; i < s2.size(); i++) {
 60                 int expected = (int) BYTE_VH.get(base.addOffset(i));
 61                 int found = (int) BYTE_VH.get(rebased.addOffset(i));
 62                 assertEquals(found, expected);
 63             }
 64         } else if (s1.kind != s2.kind) {
 65             // check that rebase s1 to s2 fails
 66             try {
 67                 s1.segment.baseAddress().rebase(s2.segment);
 68                 fail("Rebase unexpectedly passed!");
 69             } catch (IllegalArgumentException ex) {
 70                 assertTrue(true);
 71             }
 72         } else if (!s2.contains(s1)) {
 73             //disjoint segments - check that rebased address is out of bounds
 74             MemoryAddress base = s2.segment.baseAddress();
 75             MemoryAddress rebased = base.rebase(s1.segment);
 76             for (int i = 0; i < s2.size(); i++) {
 77                 BYTE_VH.get(base.addOffset(i));
 78                 try {
 79                     BYTE_VH.get(rebased.addOffset(i));
 80                     fail("Rebased address on a disjoint segment is not out of bounds!");
 81                 } catch (IndexOutOfBoundsException ex) {
 82                     assertTrue(true);
 83                 }
 84             }
 85         }
 86     }
 87 
 88     static class SegmentSlice {
 89 
 90         enum Kind {
 91             NATIVE(MemorySegment::allocateNative),
 92             ARRAY(i -> MemorySegment.ofArray(new byte[i]));
 93 
 94             final IntFunction<MemorySegment> segmentFactory;
 95 
 96             Kind(IntFunction<MemorySegment> segmentFactory) {
 97                 this.segmentFactory = segmentFactory;
 98             }
 99 
100             MemorySegment makeSegment(int elems) {
101                 return segmentFactory.apply(elems);
102             }
103         }
104 
105         final Kind kind;
106         final int first;
107         final int last;
108         final MemorySegment segment;
109 
110         public SegmentSlice(Kind kind, int first, int last, MemorySegment segment) {
111             this.kind = kind;
112             this.first = first;
113             this.last = last;
114             this.segment = segment;
115         }
116 
117         boolean contains(SegmentSlice other) {
118             return kind == other.kind &&
119                     first <= other.first &&
120                     last >= other.last;
121         }
122 
123         int size() {
124             return last - first + 1;
125         }
126     }
127 
128     @DataProvider(name = "slices")
129     static Object[][] slices() {
130         int[] sizes = { 16, 8, 4, 2, 1 };
131         List<SegmentSlice> slices = new ArrayList<>();
132         for (SegmentSlice.Kind kind : SegmentSlice.Kind.values()) {
133             //init root segment
134             MemorySegment segment = kind.makeSegment(16);
135             for (int i = 0 ; i < 16 ; i++) {
136                 BYTE_VH.set(segment.baseAddress().addOffset(i), (byte)i);
137             }
138             //compute all slices
139             for (int size : sizes) {
140                 for (int index = 0 ; index < 16 ; index += size) {
141                     MemorySegment slice = segment.asSlice(index, size);
142                     slices.add(new SegmentSlice(kind, index, index + size - 1, slice));
143                 }
144             }
145         }
146         Object[][] sliceArray = new Object[slices.size() * slices.size()][];
147         for (int i = 0 ; i < slices.size() ; i++) {
148             for (int j = 0 ; j < slices.size() ; j++) {
149                 sliceArray[i * slices.size() + j] = new Object[] { slices.get(i), slices.get(j) };
150             }
151         }
152         return sliceArray;
153     }
154 }
