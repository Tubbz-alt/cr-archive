154     String decorateLayoutString(String s) {
155         if (name().isPresent()) {
156             s = String.format("%s(%s)", s, name().get());
157         }
158         if (!hasNaturalAlignment()) {
159             s = alignment + "%" + s;
160         }
161         if (!attributes.isEmpty()) {
162             s += attributes.entrySet().stream()
163                                       .map(e -> e.getKey() + "=" + e.getValue())
164                                       .collect(Collectors.joining(",", "[", "]"));
165         }
166         return s;
167     }

169     <T> DynamicConstantDesc<T> decorateLayoutConstant(DynamicConstantDesc<T> desc) {

171             desc = DynamicConstantDesc.ofNamed(BSM_INVOKE, "withBitAlignment", desc.constantType(), MH_WITH_BIT_ALIGNMENT,
