23 package org.openjdk.bench.jdk.incubator.foreign.points.support;
24 
25 public class JNIPoint implements AutoCloseable {
26 
27     static {
28         System.loadLibrary("JNIPoint");
29     }
30 
31     private final long peer;
32 
33     public JNIPoint(int x, int y) {
34         peer = allocate();
35         setX(peer, x);
36         setY(peer, y);
37     }
38 
39     public void free() {
40         free(peer);
41     }
42 
43     public int getX() {
44         return getX(peer);
45     }
46 
47     public void setX(int value) {
48         setX(peer, value);
49     }
50 
51     public int getY() {
52         return getY(peer);
53     }
54 
55     public void setY(int value) {
56         setY(peer, value);
57     }
58 
59     private static native long allocate();
60     private static native void free(long ptr);
61 
62     private static native int getX(long ptr);
63     private static native void setX(long ptr, int x);
64 
65     private static native int getY(long ptr);
66     private static native void setY(long ptr, int y);
67 
68     @Override
69     public void close() {
70         free();
71     }
72 }
