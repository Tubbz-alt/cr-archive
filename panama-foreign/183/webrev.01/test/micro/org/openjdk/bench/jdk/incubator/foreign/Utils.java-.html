27 package org.openjdk.bench.jdk.incubator.foreign;
28 
29 import sun.misc.Unsafe;
30 
31 import java.lang.reflect.Field;
32 
33 public class Utils {
34 
35     static final Unsafe unsafe;
36 
37     //setup unsafe
38     static {
39         try {
40             Field f = Unsafe.class.getDeclaredField("theUnsafe");
41             f.setAccessible(true);
42             unsafe = (Unsafe) f.get(null);
43         } catch (ReflectiveOperationException ex) {
44             throw new IllegalStateException();
45         }
46     }
47 }
