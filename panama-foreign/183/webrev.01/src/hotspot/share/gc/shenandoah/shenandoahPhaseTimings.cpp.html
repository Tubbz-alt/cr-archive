 51 ShenandoahPhaseTimings::ShenandoahPhaseTimings(uint max_workers) :
 52   _max_workers(max_workers) {
 53   assert(_max_workers > 0, "Must have some GC threads");
 54 
 55   // Initialize everything to sane defaults
 56   for (uint i = 0; i < _num_phases; i++) {
 57 #define SHENANDOAH_WORKER_DATA_NULL(type, title) \
 58     _worker_data[i] = NULL;
 59     SHENANDOAH_PAR_PHASE_DO(,, SHENANDOAH_WORKER_DATA_NULL)
 60 #undef SHENANDOAH_WORKER_DATA_NULL
 61     _cycle_data[i] = uninitialized();
 62   }
 63 
 64   // Then punch in the worker-related data.
 65   // Every worker phase get a bunch of internal objects, except
 66   // the very first slot, which is "<total>" and is not populated.
 67   for (uint i = 0; i < _num_phases; i++) {
 68     if (is_worker_phase(Phase(i))) {
 69       int c = 0;
 70 #define SHENANDOAH_WORKER_DATA_INIT(type, title) \
 71       if (c++ != 0) _worker_data[i + c] = new ShenandoahWorkerData(title, _max_workers);
 72       SHENANDOAH_PAR_PHASE_DO(,, SHENANDOAH_WORKER_DATA_INIT)
 73 #undef SHENANDOAH_WORKER_DATA_INIT
 74     }
 75   }
 76 
 77   _policy = ShenandoahHeap::heap()->shenandoah_policy();
 78   assert(_policy != NULL, "Can not be NULL");
 79 }

 81 ShenandoahPhaseTimings::Phase ShenandoahPhaseTimings::worker_par_phase(Phase phase, ParPhase par_phase) {
 82   assert(is_worker_phase(phase), "Phase should accept worker phase times: %s", phase_name(phase));
 83   Phase p = Phase(phase + 1 + par_phase);
 84   assert(p >= 0 && p < _num_phases, "Out of bound for: %s", phase_name(phase));
 85   return p;
 86 }
 87 
 88 ShenandoahWorkerData* ShenandoahPhaseTimings::worker_data(Phase phase, ParPhase par_phase) {
 89   Phase p = worker_par_phase(phase, par_phase);
 90   ShenandoahWorkerData* wd = _worker_data[p];
 91   assert(wd != NULL, "Counter initialized: %s", phase_name(p));
 92   return wd;
 93 }

 95 bool ShenandoahPhaseTimings::is_worker_phase(Phase phase) {
 96   assert(phase >= 0 && phase < _num_phases, "Out of bounds");
 97   switch (phase) {
 98     case init_evac:
 99     case scan_roots:
100     case update_roots:
101     case final_update_refs_roots:
102     case full_gc_scan_roots:
103     case full_gc_update_roots:
104     case full_gc_adjust_roots:
105     case degen_gc_update_roots:
106     case full_gc_purge_class_unload:
107     case full_gc_purge_weak_par:
108     case purge_class_unload:
109     case purge_weak_par:
110     case heap_iteration_roots:
111     case conc_weak_roots_work:
112     case conc_strong_roots:
113       return true;
114     default:
115       return false;
116   }
117 }
118 
119 bool ShenandoahPhaseTimings::is_root_work_phase(Phase phase) {
120   switch (phase) {
121     case scan_roots:
122     case update_roots:
123     case init_evac:
124     case final_update_refs_roots:
125     case degen_gc_update_roots:
126     case full_gc_scan_roots:
127     case full_gc_update_roots:
128     case full_gc_adjust_roots:
129       return true;
130     default:
131       return false;
132   }
133 }

135 void ShenandoahPhaseTimings::set_cycle_data(Phase phase, double time) {
136 #ifdef ASSERT
137   double d = _cycle_data[phase];
138   assert(d == uninitialized(), "Should not be set yet: %s, current value: %lf", phase_name(phase), d);
139 #endif
140   _cycle_data[phase] = time;
141 }
142 
143 void ShenandoahPhaseTimings::record_phase_time(Phase phase, double time) {
144   if (!_policy->is_at_shutdown()) {
145     set_cycle_data(phase, time);
146   }
147 }

149 void ShenandoahPhaseTimings::record_workers_start(Phase phase) {
150   assert(is_worker_phase(phase), "Phase should accept worker phase times: %s", phase_name(phase));
151 
152   // Special case: these phases can enter multiple times, need to reset
153   // their worker data every time.
154   if (phase == heap_iteration_roots) {
155     for (uint i = 1; i < _num_par_phases; i++) {
156       worker_data(phase, ParPhase(i))->reset();
157     }
158   }
159 
160 #ifdef ASSERT
161   for (uint i = 1; i < _num_par_phases; i++) {
162     ShenandoahWorkerData* wd = worker_data(phase, ParPhase(i));
163     for (uint c = 0; c < _max_workers; c++) {
164       assert(wd->get(c) == ShenandoahWorkerData::uninitialized(),
165              "Should not be set: %s", phase_name(worker_par_phase(phase, ParPhase(i))));
166     }
167   }
168 #endif
169 }
170 
171 void ShenandoahPhaseTimings::record_workers_end(Phase phase) {
172   assert(is_worker_phase(phase), "Phase should accept worker phase times: %s", phase_name(phase));
173 }

175 void ShenandoahPhaseTimings::flush_par_workers_to_cycle() {
176   for (uint pi = 0; pi < _num_phases; pi++) {
177     Phase phase = Phase(pi);
178     if (is_worker_phase(phase)) {
179       double s = uninitialized();
180       for (uint i = 1; i < _num_par_phases; i++) {
181         ShenandoahWorkerData* wd = worker_data(phase, ParPhase(i));
182         double ws = uninitialized();
183         for (uint c = 0; c < _max_workers; c++) {
184           double v = wd->get(c);
185           if (v != ShenandoahWorkerData::uninitialized()) {
186             if (ws == uninitialized()) {
187               ws = v;
188             } else {
189               ws += v;
190             }
191           }
192         }
193         if (ws != uninitialized()) {
194           // add to each line in phase
195           set_cycle_data(Phase(phase + i + 1), ws);
196           if (s == uninitialized()) {
197             s = ws;
198           } else {
199             s += ws;
200           }
201         }
202       }
203       if (s != uninitialized()) {
204         // add to total for phase
205         set_cycle_data(Phase(phase + 1), s);
206       }
207     }
208   }
209 }

211 void ShenandoahPhaseTimings::flush_cycle_to_global() {
212   for (uint i = 0; i < _num_phases; i++) {
213     if (_cycle_data[i] != uninitialized()) {
214       _global_data[i].add(_cycle_data[i]);
215       _cycle_data[i] = uninitialized();
216     }
217     if (_worker_data[i] != NULL) {
218       _worker_data[i]->reset();
219     }
220   }
221   OrderAccess::fence();
222 }

224 void ShenandoahPhaseTimings::print_cycle_on(outputStream* out) const {
225   out->cr();
226   out->print_cr("All times are wall-clock times, except per-root-class counters, that are sum over");
227   out->print_cr("all workers. Dividing the <total> over the root stage time estimates parallelism.");
228   out->cr();
229   for (uint i = 0; i < _num_phases; i++) {
230     double v = _cycle_data[i] * 1000000.0;
231     if (v > 0) {
232       out->print(SHENANDOAH_PHASE_NAME_FORMAT " " SHENANDOAH_US_TIME_FORMAT " us", _phase_names[i], v);
233 
234       if (is_worker_phase(Phase(i))) {
235         double total = _cycle_data[i + 1] * 1000000.0;
236         if (total > 0) {
237           out->print(", parallelism: " SHENANDOAH_PARALLELISM_FORMAT "x", total / v);
238         }
239       }
240 
241       if (_worker_data[i] != NULL) {
242         out->print(", workers (us): ");
243         for (uint c = 0; c < _max_workers; c++) {
244           double tv = _worker_data[i]->get(c);
245           if (tv != ShenandoahWorkerData::uninitialized()) {
246             out->print(SHENANDOAH_US_WORKER_TIME_FORMAT ", ", tv * 1000000.0);
247           } else {
248             out->print("%3s, ", "---");
249           }
250         }
251       }
252       out->cr();
253     }
254   }
255 }

257 void ShenandoahPhaseTimings::print_global_on(outputStream* out) const {
258   out->cr();
259   out->print_cr("GC STATISTICS:");
260   out->print_cr("  \"(G)\" (gross) pauses include VM time: time to notify and block threads, do the pre-");
261   out->print_cr("        and post-safepoint housekeeping. Use -Xlog:safepoint+stats to dissect.");
262   out->print_cr("  \"(N)\" (net) pauses are the times spent in the actual GC code.");
263   out->print_cr("  \"a\" is average time for each phase, look at levels to see if average makes sense.");
264   out->print_cr("  \"lvls\" are quantiles: 0%% (minimum), 25%%, 50%% (median), 75%%, 100%% (maximum).");
265   out->cr();
266   out->print_cr("  All times are wall-clock times, except per-root-class counters, that are sum over");
267   out->print_cr("  all workers. Dividing the <total> over the root stage time estimates parallelism.");
268   out->cr();
269 
270   for (uint i = 0; i < _num_phases; i++) {
271     if (_global_data[i].maximum() != 0) {
272       out->print_cr(SHENANDOAH_PHASE_NAME_FORMAT " = " SHENANDOAH_S_TIME_FORMAT " s "
273                     "(a = " SHENANDOAH_US_TIME_FORMAT " us) "
274                     "(n = " INT32_FORMAT_W(5) ") (lvls, us = "
275                     SHENANDOAH_US_TIME_FORMAT ", "
276                     SHENANDOAH_US_TIME_FORMAT ", "
277                     SHENANDOAH_US_TIME_FORMAT ", "
278                     SHENANDOAH_US_TIME_FORMAT ", "
279                     SHENANDOAH_US_TIME_FORMAT ")",
280                     _phase_names[i],
281                     _global_data[i].sum(),
282                     _global_data[i].avg() * 1000000.0,
283                     _global_data[i].num(),
284                     _global_data[i].percentile(0) * 1000000.0,
285                     _global_data[i].percentile(25) * 1000000.0,
286                     _global_data[i].percentile(50) * 1000000.0,
287                     _global_data[i].percentile(75) * 1000000.0,
288                     _global_data[i].maximum() * 1000000.0
289       );
290     }
291   }
292 }

294 ShenandoahWorkerTimingsTracker::ShenandoahWorkerTimingsTracker(ShenandoahPhaseTimings::Phase phase,
295         ShenandoahPhaseTimings::ParPhase par_phase, uint worker_id) :
296         _timings(ShenandoahHeap::heap()->phase_timings()),
297         _phase(phase), _par_phase(par_phase), _worker_id(worker_id) {
298 
299   assert(_timings->worker_data(_phase, _par_phase)->get(_worker_id) == ShenandoahWorkerData::uninitialized(),
300          "Should not be set yet: %s", ShenandoahPhaseTimings::phase_name(_timings->worker_par_phase(_phase, _par_phase)));
301   _start_time = os::elapsedTime();
302 }
303 
304 ShenandoahWorkerTimingsTracker::~ShenandoahWorkerTimingsTracker() {
305   _timings->worker_data(_phase, _par_phase)->set(_worker_id, os::elapsedTime() - _start_time);
306 
307   if (ShenandoahPhaseTimings::is_root_work_phase(_phase)) {
308     ShenandoahPhaseTimings::Phase root_phase = _phase;
309     ShenandoahPhaseTimings::Phase cur_phase = _timings->worker_par_phase(root_phase, _par_phase);
310     _event.commit(GCId::current(), _worker_id, ShenandoahPhaseTimings::phase_name(cur_phase));
311   }
312 }
