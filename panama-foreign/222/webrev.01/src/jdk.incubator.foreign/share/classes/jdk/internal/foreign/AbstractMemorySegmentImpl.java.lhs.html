 51 /**
 52  * This abstract class provides an immutable implementation for the {@code MemorySegment} interface. This class contains information
 53  * about the segment's spatial and temporal bounds; each memory segment implementation is associated with an owner thread which is set at creation time.
 54  * Access to certain sensitive operations on the memory segment will fail with {@code IllegalStateException} if the
 55  * segment is either in an invalid state (e.g. it has already been closed) or if access occurs from a thread other
 56  * than the owner thread. See {@link MemoryScope} for more details on management of temporal bounds. Subclasses
 57  * are defined for each memory segment kind, see {@link NativeMemorySegmentImpl}, {@link HeapMemorySegmentImpl} and
 58  * {@link MappedMemorySegmentImpl}.
 59  */
 60 public abstract class AbstractMemorySegmentImpl implements MemorySegment, MemorySegmentProxy {
 61 
 62     private static final Unsafe UNSAFE = Unsafe.getUnsafe();

 64     private static final boolean enableSmallSegments =
 65             Boolean.parseBoolean(GetPropertyAction.privilegedGetProperty("jdk.incubator.foreign.SmallSegments", "true"));

 67     final static int FIRST_RESERVED_FLAG = 1 << 16; // upper 16 bits are reserved
 68     final static int SMALL = FIRST_RESERVED_FLAG;
 69     final static long NONCE = new Random().nextLong();

 92     static int defaultAccessModes(long size) {
 93         return (enableSmallSegments && size < Integer.MAX_VALUE) ?
 94                 ALL_ACCESS | SMALL :

 98     @Override

100         checkBounds(offset, newSize);
101         return asSliceNoCheck(offset, newSize);

104     private AbstractMemorySegmentImpl asSliceNoCheck(long offset, long newSize) {
105         return dup(offset, newSize, mask, scope);
106     }

108     @SuppressWarnings("unchecked")
109     public static <S extends MemorySegment> Spliterator<S> spliterator(S segment, SequenceLayout sequenceLayout) {
110         ((AbstractMemorySegmentImpl)segment).checkValidState();
111         if (sequenceLayout.byteSize() != segment.byteSize()) {
112             throw new IllegalArgumentException();
113         }
114         return (Spliterator<S>)new SegmentSplitter(sequenceLayout.elementLayout().byteSize(), sequenceLayout.elementCount().getAsLong(),
115                 (AbstractMemorySegmentImpl)segment.withAccessModes(segment.accessModes() & ~CLOSE));
116     }

118     @Override
119     public final MemorySegment fill(byte value){
120         checkRange(0, length, true);
121         UNSAFE.setMemory(base(), min(), length, value);
122         return this;
123     }
124 
125     public void copyFrom(MemorySegment src) {
126         AbstractMemorySegmentImpl that = (AbstractMemorySegmentImpl)src;
127         long size = that.byteSize();
128         checkRange(0, size, true);
129         that.checkRange(0, size, false);
130         UNSAFE.copyMemory(
131                 that.base(), that.min(),
132                 base(), min(), size);
133     }

135     private final static VarHandle BYTE_HANDLE = MemoryLayout.ofSequence(MemoryLayouts.JAVA_BYTE)
136             .varHandle(byte.class, MemoryLayout.PathElement.sequenceElement());
137 
138     @Override
139     public long mismatch(MemorySegment other) {
140         AbstractMemorySegmentImpl that = (AbstractMemorySegmentImpl)other;
141         final long thisSize = this.byteSize();
142         final long thatSize = that.byteSize();
143         final long length = Math.min(thisSize, thatSize);
144         this.checkRange(0, length, false);
145         that.checkRange(0, length, false);
146         if (this == other) {
147             return -1;
148         }
149 
150         long i = 0;
151         if (length > 7) {
152             if ((byte) BYTE_HANDLE.get(this.baseAddress(), 0) != (byte) BYTE_HANDLE.get(that.baseAddress(), 0)) {
153                 return 0;
154             }
155             i = ArraysSupport.vectorizedMismatchLargeForBytes(
156                     this.base(), this.min(),
157                     that.base(), that.min(),
158                     length);
159             if (i >= 0) {
160                 return i;
161             }
162             long remaining = ~i;
163             assert remaining < 8 : "remaining greater than 7: " + remaining;
164             i = length - remaining;
165         }
166         MemoryAddress thisAddress = this.baseAddress();
167         MemoryAddress thatAddress = that.baseAddress();
168         for (; i < length; i++) {
169             if ((byte) BYTE_HANDLE.get(thisAddress, i) != (byte) BYTE_HANDLE.get(thatAddress, i)) {
170                 return i;
171             }
172         }
173         return thisSize != thatSize ? length : -1;
174     }

176     @Override

178     public final MemoryAddress baseAddress() {
179         return new MemoryAddressImpl(this, 0);

182     @Override
183     public final ByteBuffer asByteBuffer() {
184         if (!isSet(READ)) {
185             throw unsupportedAccessMode(READ);
186         }
187         checkIntSize("ByteBuffer");
188         ByteBuffer _bb = makeByteBuffer();
189         if (!isSet(WRITE)) {
190             //scope is IMMUTABLE - obtain a RO byte buffer
191             _bb = _bb.asReadOnlyBuffer();
192         }
193         return _bb;
194     }

216     @Override
217     public AbstractMemorySegmentImpl withAccessModes(int accessModes) {
218         checkAccessModes(accessModes);
219         if ((~accessModes() & accessModes) != 0) {
220             throw new IllegalArgumentException("Cannot acquire more access modes");
221         }
222         return dup(0, length, (mask & ~ALL_ACCESS) | accessModes, scope);
223     }
224 
225     @Override
226     public boolean hasAccessModes(int accessModes) {
227         checkAccessModes(accessModes);
228         return (accessModes() & accessModes) == accessModes;
229     }
230 
231     private void checkAccessModes(int accessModes) {
232         if ((accessModes & ~ALL_ACCESS) != 0) {
233             throw new IllegalArgumentException("Invalid access modes");
234         }
235     }
236 
237     @Override
238     public MemorySegment withOwnerThread(Thread newOwner) {
239         Objects.requireNonNull(newOwner);
240         if (!isSet(HANDOFF)) {
241             throw unsupportedAccessMode(HANDOFF);
242         }
243         if (scope.ownerThread() == newOwner) {
244             throw new IllegalArgumentException("Segment already owned by thread: " + newOwner);
245         } else {
246             return dupAndClose(newOwner);
247         }
248     }
249 
250     public MemorySegment dupAndClose(Thread newOwner) {
251         try {
252             return dup(0L, length, mask, scope.dup(newOwner));
253         } finally {
254             //flush read/writes to segment memory before returning the new segment
255             VarHandle.fullFence();
256         }
257     }
258 
259     @Override
260     public final void close() {
261         if (!isSet(CLOSE)) {
262             throw unsupportedAccessMode(CLOSE);
263         }
264         closeNoCheck();
265     }
266 
267     private final void closeNoCheck() {
268         scope.close();
269     }
270 
271     final AbstractMemorySegmentImpl acquire() {
272         if (Thread.currentThread() != ownerThread() && !isSet(ACQUIRE)) {
273             throw unsupportedAccessMode(ACQUIRE);
274         }
275         return dup(0, length, mask, scope.acquire());
276     }

278     @Override
279     public final byte[] toByteArray() {
280         checkIntSize("byte[]");
281         byte[] arr = new byte[(int)length];
282         MemorySegment arrSegment = MemorySegment.ofArray(arr);
283         arrSegment.copyFrom(this);
284         return arr;
285     }

287     boolean isSmall() {
288         return isSet(SMALL);
289     }

291     void checkRange(long offset, long length, boolean writeAccess) {
292         scope.checkValidState();
293         if (writeAccess && !isSet(WRITE)) {
294             throw unsupportedAccessMode(WRITE);
295         } else if (!writeAccess && !isSet(READ)) {
296             throw unsupportedAccessMode(READ);
297         }
298         checkBounds(offset, length);
299     }
300 
301     @Override
302     public final void checkValidState() {
303         scope.checkValidState();
304     }

308     private boolean isSet(int mask) {
309         return (this.mask & mask) != 0;
310     }
311 
312     private void checkIntSize(String typeName) {
313         if (length > (Integer.MAX_VALUE - 8)) { //conservative check
314             throw new UnsupportedOperationException(String.format("Segment is too large to wrap as %s. Size: %d", typeName, length));
315         }
316     }
317 
318     private void checkBounds(long offset, long length) {
319         if (isSmall()) {
320             checkBoundsSmall((int)offset, (int)length);
321         } else {
322             if (length < 0 ||
323                     offset < 0 ||
324                     offset > this.length - length) { // careful of overflow
325                 throw outOfBoundException(offset, length);
326             }
327         }
328     }
329 
330     private void checkBoundsSmall(int offset, int length) {
331         if (length < 0 ||
332                 offset < 0 ||
333                 offset > (int)this.length - length) { // careful of overflow
334             throw outOfBoundException(offset, length);
335         }
336     }
337 
338     UnsupportedOperationException unsupportedAccessMode(int expected) {
339         return new UnsupportedOperationException((String.format("Required access mode %s ; current access modes: %s",
340                 modeStrings(expected).get(0), modeStrings(mask))));
341     }
342 
343     private List<String> modeStrings(int mode) {
344         List<String> modes = new ArrayList<>();
345         if ((mode & READ) != 0) {
346             modes.add("READ");
347         }
348         if ((mode & WRITE) != 0) {
349             modes.add("WRITE");
350         }
351         if ((mode & CLOSE) != 0) {
352             modes.add("CLOSE");
353         }
354         if ((mode & ACQUIRE) != 0) {
355             modes.add("ACQUIRE");
356         }
357         if ((mode & HANDOFF) != 0) {
358             modes.add("HANDOFF");
359         }
360         return modes;
361     }
362 
363     private IndexOutOfBoundsException outOfBoundException(long offset, long length) {
364         return new IndexOutOfBoundsException(String.format("Out of bound access on segment %s; new offset = %d; new length = %d",
365                         this, offset, length));
366     }

368     protected int id() {
369         //compute a stable and random id for this memory segment
370         return Math.abs(Objects.hash(base(), min(), NONCE));
371     }

373     static class SegmentSplitter implements Spliterator<MemorySegment> {
374         AbstractMemorySegmentImpl segment;
375         long elemCount;
376         final long elementSize;
377         long currentIndex;
378 
379         SegmentSplitter(long elementSize, long elemCount, AbstractMemorySegmentImpl segment) {
380             this.segment = segment;
381             this.elementSize = elementSize;
382             this.elemCount = elemCount;
383         }
384 
385         @Override
386         public SegmentSplitter trySplit() {
387             if (currentIndex == 0 && elemCount > 1) {
388                 AbstractMemorySegmentImpl parent = segment;
389                 long rem = elemCount % 2;
390                 long split = elemCount / 2;
391                 long lobound = split * elementSize;
392                 long hibound = lobound + (rem * elementSize);
393                 elemCount  = split + rem;
394                 segment = parent.asSliceNoCheck(lobound, hibound);
395                 return new SegmentSplitter(elementSize, split, parent.asSliceNoCheck(0, lobound));
396             } else {
397                 return null;
398             }
399         }
400 
401         @Override
402         public boolean tryAdvance(Consumer<? super MemorySegment> action) {
403             Objects.requireNonNull(action);
404             if (currentIndex < elemCount) {
405                 AbstractMemorySegmentImpl acquired = segment.acquire();
406                 try {
407                     action.accept(acquired.asSliceNoCheck(currentIndex * elementSize, elementSize));
408                 } finally {
409                     acquired.closeNoCheck();
410                     currentIndex++;
411                     if (currentIndex == elemCount) {
412                         segment = null;
413                     }
414                 }
415                 return true;
416             } else {
417                 return false;
418             }
419         }
420 
421         @Override
422         public void forEachRemaining(Consumer<? super MemorySegment> action) {
423             Objects.requireNonNull(action);
424             if (currentIndex < elemCount) {
425                 AbstractMemorySegmentImpl acquired = segment.acquire();
426                 try {
427                     if (acquired.isSmall()) {
428                         int index = (int) currentIndex;
429                         int limit = (int) elemCount;
430                         int elemSize = (int) elementSize;
431                         for (; index < limit; index++) {
432                             action.accept(acquired.asSliceNoCheck(index * elemSize, elemSize));
433                         }
434                     } else {
435                         for (long i = currentIndex ; i < elemCount ; i++) {
436                             action.accept(acquired.asSliceNoCheck(i * elementSize, elementSize));
437                         }
438                     }
439                 } finally {
440                     acquired.closeNoCheck();
441                     currentIndex = elemCount;
442                     segment = null;
443                 }
444             }
445         }
446 
447         @Override
448         public long estimateSize() {
449             return elemCount;
450         }
451 
452         @Override
453         public int characteristics() {
454             return NONNULL | SUBSIZED | SIZED | IMMUTABLE | ORDERED;
455         }
456     }

465     public static AbstractMemorySegmentImpl ofBuffer(ByteBuffer bb) {
466         long bbAddress = nioAccess.getBufferAddress(bb);
467         Object base = nioAccess.getBufferBase(bb);
468         UnmapperProxy unmapper = nioAccess.unmapper(bb);
469 
470         int pos = bb.position();
471         int limit = bb.limit();
472         int size = limit - pos;
473 
474         AbstractMemorySegmentImpl bufferSegment = (AbstractMemorySegmentImpl)nioAccess.bufferSegment(bb);
475         final MemoryScope bufferScope;
476         int modes;
477         if (bufferSegment != null) {
478             bufferScope = bufferSegment.scope;
479             modes = bufferSegment.mask;
480         } else {
481             bufferScope = MemoryScope.create(bb, null);
482             modes = defaultAccessModes(size);
483         }
484         if (bb.isReadOnly()) {
485             modes &= ~WRITE;
486         }
487         if (base != null) {
488             return new HeapMemorySegmentImpl<>(bbAddress + pos, () -> (byte[])base, size, modes, bufferScope);
489         } else if (unmapper == null) {
490             return new NativeMemorySegmentImpl(bbAddress + pos, size, modes, bufferScope);
491         } else {
492             return new MappedMemorySegmentImpl(bbAddress + pos, unmapper, size, modes, bufferScope);
493         }
494     }
