130         private Object invoke(Object[] args) throws Throwable {
131             // one trailing Object[]
132             int nNamedArgs = function.argumentLayouts().size();
133             assert(args.length == nNamedArgs + 1);
134             // The last argument is the array of vararg collector
135             Object[] unnamedArgs = (Object[]) args[args.length - 1];
136 
137             int argsCount = nNamedArgs + unnamedArgs.length;
138             Class<?>[] argTypes = new Class<?>[argsCount];
139             MemoryLayout[] argLayouts = new MemoryLayout[nNamedArgs + unnamedArgs.length];
140 
141             int pos = 0;
142             for (pos = 0; pos < nNamedArgs; pos++) {
143                 argTypes[pos] = varargs.parameterType(pos);
144                 argLayouts[pos] = function.argumentLayouts().get(pos);
145             }
146 
147             assert pos == nNamedArgs;
148             for (Object o: unnamedArgs) {
149                 argTypes[pos] = normalize(o.getClass());
150                 argLayouts[pos] = variadicLayout(argTypes[pos]);
151                 pos++;
152             }
153             assert pos == argsCount;
154 
155             MethodType mt = MethodType.methodType(varargs.returnType(), argTypes);
156             FunctionDescriptor f = (function.returnLayout().isEmpty()) ?
157                     FunctionDescriptor.ofVoid(argLayouts) :
158                     FunctionDescriptor.of(function.returnLayout().get(), argLayouts);
159             MethodHandle mh = ABI.downcallHandle(symbol, mt, f);
160             // flatten argument list so that it can be passed to an asSpreader MH
161             Object[] allArgs = new Object[nNamedArgs + unnamedArgs.length];
162             System.arraycopy(args, 0, allArgs, 0, nNamedArgs);
163             System.arraycopy(unnamedArgs, 0, allArgs, nNamedArgs, unnamedArgs.length);
164 
165             return mh.asSpreader(Object[].class, argsCount).invoke(allArgs);
166         }
