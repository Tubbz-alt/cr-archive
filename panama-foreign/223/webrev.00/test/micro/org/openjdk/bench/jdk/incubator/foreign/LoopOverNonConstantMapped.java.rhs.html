 23 package org.openjdk.bench.jdk.incubator.foreign;

 25 import jdk.incubator.foreign.MappedMemorySegment;
 26 import jdk.incubator.foreign.MemoryAddress;
 27 import jdk.incubator.foreign.MemoryLayout;
 28 import jdk.incubator.foreign.MemoryLayouts;
 29 import jdk.incubator.foreign.MemorySegment;
 30 import jdk.incubator.foreign.MemorySegments;
 31 import org.openjdk.jmh.annotations.Benchmark;
 32 import org.openjdk.jmh.annotations.BenchmarkMode;
 33 import org.openjdk.jmh.annotations.Fork;
 34 import org.openjdk.jmh.annotations.Measurement;
 35 import org.openjdk.jmh.annotations.Mode;
 36 import org.openjdk.jmh.annotations.OutputTimeUnit;
 37 import org.openjdk.jmh.annotations.Setup;
 38 import org.openjdk.jmh.annotations.State;
 39 import org.openjdk.jmh.annotations.TearDown;
 40 import org.openjdk.jmh.annotations.Warmup;
 41 import sun.misc.Unsafe;
 42 
 43 import java.io.File;
 44 import java.io.IOException;
 45 import java.lang.invoke.VarHandle;
 46 import java.nio.ByteBuffer;
 47 import java.nio.ByteOrder;
 48 import java.nio.MappedByteBuffer;
 49 import java.nio.channels.FileChannel;
 50 import java.nio.file.Files;
 51 import java.nio.file.Path;
 52 import java.nio.file.StandardOpenOption;
 53 import java.util.concurrent.TimeUnit;
 54 
 55 import static jdk.incubator.foreign.MemoryLayout.PathElement.sequenceElement;
 56 import static jdk.incubator.foreign.MemoryLayouts.JAVA_INT;
 57 
 58 @BenchmarkMode(Mode.AverageTime)
 59 @Warmup(iterations = 5, time = 500, timeUnit = TimeUnit.MILLISECONDS)
 60 @Measurement(iterations = 10, time = 500, timeUnit = TimeUnit.MILLISECONDS)
 61 @State(org.openjdk.jmh.annotations.Scope.Thread)
 62 @OutputTimeUnit(TimeUnit.MILLISECONDS)
 63 @Fork(3)
 64 public class LoopOverNonConstantMapped {
 65 
 66     static final Unsafe unsafe = Utils.unsafe;
 67 
 68     static final int ELEM_SIZE = 1_000_000;
 69     static final int CARRIER_SIZE = (int)JAVA_INT.byteSize();
 70     static final int ALLOC_SIZE = ELEM_SIZE * CARRIER_SIZE;
 71 
 72     static final Path tempPath;
 73 
 74     static {
 75         try {
 76             File file = File.createTempFile("buffer", "txt");
 77             file.deleteOnExit();
 78             tempPath = file.toPath();
 79             Files.write(file.toPath(), new byte[ALLOC_SIZE], StandardOpenOption.WRITE);
 80 
 81         } catch (IOException ex) {
 82             throw new ExceptionInInitializerError(ex);
 83         }
 84     }
 85 
 86     static final VarHandle VH_int = MemoryLayout.ofSequence(JAVA_INT).varHandle(int.class, sequenceElement());
 87     MemorySegment segment;
 88     long unsafe_addr;
 89 
 90     ByteBuffer byteBuffer;
 91 
 92     @Setup
 93     public void setup() throws IOException {
 94         try (FileChannel channel = FileChannel.open(tempPath, StandardOpenOption.READ, StandardOpenOption.WRITE)) {
 95             byteBuffer = channel.map(FileChannel.MapMode.READ_WRITE, 0, ALLOC_SIZE).order(ByteOrder.nativeOrder());
 96             for (int i = 0; i < ELEM_SIZE; i++) {
 97                 byteBuffer.putInt(i * CARRIER_SIZE, i);
 98             }
 99             ((MappedByteBuffer)byteBuffer).force();
100         }
101         segment = MemorySegments.mapFromPath(tempPath, 0L, ALLOC_SIZE, FileChannel.MapMode.READ_WRITE);
102         unsafe_addr = segment.baseAddress().toRawLongValue();
103     }
104 
105     @TearDown
106     public void tearDown() {
107         segment.close();
108         unsafe.invokeCleaner(byteBuffer);
109     }
110 
111     @Benchmark
112     @OutputTimeUnit(TimeUnit.NANOSECONDS)
113     public int unsafe_get() {
114         return unsafe.getInt(unsafe_addr);
115     }
116 
117     @Benchmark
118     @OutputTimeUnit(TimeUnit.NANOSECONDS)
119     public int segment_get() {
120         return (int) VH_int.get(segment.baseAddress(), 0L);
121     }
122 
123     @Benchmark
124     @OutputTimeUnit(TimeUnit.NANOSECONDS)
125     public int BB_get() {
126         return byteBuffer.getInt(0);
127     }
128 
129     @Benchmark
130     public int unsafe_loop() {
131         int res = 0;
132         for (int i = 0; i < ELEM_SIZE; i ++) {
133             res += unsafe.getInt(unsafe_addr + (i * CARRIER_SIZE));
134         }
135         return res;
136     }
137 
138     @Benchmark
139     public int segment_loop() {
140         int sum = 0;
141         MemoryAddress base = segment.baseAddress();
142         for (int i = 0; i < ELEM_SIZE; i++) {
143             sum += (int) VH_int.get(base, (long) i);
144         }
145         return sum;
146     }
147 
148     @Benchmark
149     public int segment_loop_static() {
150         int res = 0;
151         for (int i = 0; i < ELEM_SIZE; i ++) {
152             res += MemorySegments.getInt(segment.baseAddress(), i * CARRIER_SIZE);
153         }
154         return res;
155     }
156 
157     @Benchmark
158     public int segment_loop_slice() {
159         int sum = 0;
160         MemoryAddress base = segment.asSlice(0, segment.byteSize()).baseAddress();
161         for (int i = 0; i < ELEM_SIZE; i++) {
162             sum += (int) VH_int.get(base, (long) i);
163         }
164         return sum;
165     }
166 
167     @Benchmark
168     public int segment_loop_readonly() {
169         int sum = 0;
170         MemoryAddress base = segment.withAccessModes(MemorySegment.READ).baseAddress();
171         for (int i = 0; i < ELEM_SIZE; i++) {
172             sum += (int) VH_int.get(base, (long) i);
173         }
174         return sum;
175     }
176 
177     @Benchmark
178     public int BB_loop() {
179         int sum = 0;
180         ByteBuffer bb = byteBuffer;
181         for (int i = 0; i < ELEM_SIZE; i++) {
182             sum += bb.getInt(i * CARRIER_SIZE);
183         }
184         return sum;
185     }
186 
187 }
