 51         return IntStream.range(0, 256)

 55     @Test(dataProvider = "unsignedIntToByteData")
 56     public void testUnsignedIntToByte(int intValue) {
 57         byte byteValue = (byte) (intValue & 0xFF);
 58 
 59         MemoryLayout layout = MemoryLayouts.BITS_8_BE;
 60         VarHandle byteHandle = layout.varHandle(byte.class);
 61         VarHandle intHandle = MemoryHandles.asUnsigned(byteHandle, int.class);
 62 
 63         try (MemorySegment segment = MemorySegments.allocateNative(layout)) {
 64             intHandle.set(segment.baseAddress(), intValue);
 65             int expectedIntValue = Byte.toUnsignedInt(byteValue);
 66             assertEquals((int) intHandle.get(segment.baseAddress()), expectedIntValue);
 67             assertEquals((byte) byteHandle.get(segment.baseAddress()), byteValue);
 68         }
 69     }

163     @Test
164     public void testCoordinatesSequenceLayout() {
165         MemoryLayout layout = MemoryLayout.ofSequence(2, MemoryLayouts.BITS_8_BE);
166         VarHandle byteHandle = layout.varHandle(byte.class, PathElement.sequenceElement());
167         VarHandle intHandle = MemoryHandles.asUnsigned(byteHandle, int.class);
168 
169         try (MemorySegment segment = MemorySegments.allocateNative(layout)) {
170             intHandle.set(segment.baseAddress(), 0L, (int) -1);
171             assertEquals((int) intHandle.get(segment.baseAddress(), 0L), 255);
172             intHandle.set(segment.baseAddress(), 1L, (int) 200);
173             assertEquals((int) intHandle.get(segment.baseAddress(), 1L), 200);
174         }
175     }

177     @Test
178     public void testCoordinatesStride() {
179         byte[] arr = { 0, 0, (byte) 129, 0 };
180         MemorySegment segment = MemorySegments.ofArray(arr);
181         MemoryAddress addr = segment.baseAddress();
182 
183         {
184             VarHandle byteHandle = MemoryHandles.varHandle(byte.class, ByteOrder.nativeOrder());
185             VarHandle intHandle = MemoryHandles.asUnsigned(byteHandle, int.class);
186             VarHandle strideHandle = MemoryHandles.withStride(intHandle, 1);
187             assertEquals((int) strideHandle.get(addr, 2L), 129);
188         }
189         {
190             VarHandle byteHandle = MemoryHandles.varHandle(byte.class, ByteOrder.nativeOrder());
191             VarHandle strideHandle = MemoryHandles.withStride(byteHandle, 1);
192             VarHandle intHandle = MemoryHandles.asUnsigned(strideHandle, int.class);
193             assertEquals((int) intHandle.get(addr, 2L), 129);
194         }
195     }

197     static final Class<NullPointerException> NPE = NullPointerException.class;

199     @Test
200     public void testNull() {
201         VarHandle handle = MemoryHandles.varHandle(byte.class, BIG_ENDIAN);
202         assertThrows(NPE, () -> MemoryHandles.asUnsigned(handle, null));
203         assertThrows(NPE, () -> MemoryHandles.asUnsigned(null, short.class));
204         assertThrows(NPE, () -> MemoryHandles.asUnsigned(null, null));
205     }

207     static final Class<IllegalArgumentException> IAE = IllegalArgumentException.class;
208 
209     static void assertIllegalArgumentExceptionIllegalCarrier(Class<?> carrier, Class<?> adaptedType) {
210         var vh = MemoryHandles.varHandle(carrier, BIG_ENDIAN);
211         var exception = expectThrows(IAE, () -> MemoryHandles.asUnsigned(vh, adaptedType));
212         var msg = exception.getMessage();
213         assertTrue(msg.contains("illegal carrier"), "Expected \"illegal carrier\" in:[" + msg +"]");
214     }
215 
216     static void assertIllegalArgumentExceptionIllegalAdapter(Class<?> carrier, Class<?> adaptedType) {
217         var vh = MemoryHandles.varHandle(carrier, BIG_ENDIAN);
218         var exception = expectThrows(IAE, () -> MemoryHandles.asUnsigned(vh, adaptedType));
219         var msg = exception.getMessage();
220         assertTrue(msg.contains("illegal adapter type"), "Expected \"illegal adapter type\" in:[" + msg +"]");
221     }
222 
223     static void assertIllegalArgumentExceptionIsNotWiderThan(Class<?> carrier, Class<?> adaptedType) {
224         var vh = MemoryHandles.varHandle(carrier, BIG_ENDIAN);
225         var exception = expectThrows(IAE, () -> MemoryHandles.asUnsigned(vh, adaptedType));
226         var msg = exception.getMessage();
227         assertTrue(msg.contains("is not wider than"), "Expected \"is not wider than\" in:[" + msg +"]");
228     }

230     @Test
231     public void testIllegalArgumentException() {
232         assertIllegalArgumentExceptionIllegalCarrier(char.class,   long.class);
233         assertIllegalArgumentExceptionIllegalCarrier(double.class, long.class);
234         assertIllegalArgumentExceptionIllegalCarrier(float.class,  long.class);
235         assertIllegalArgumentExceptionIllegalCarrier(long.class,   long.class);
236 
237         assertIllegalArgumentExceptionIllegalAdapter(byte.class, void.class);
238         assertIllegalArgumentExceptionIllegalAdapter(byte.class, byte.class);
239         assertIllegalArgumentExceptionIllegalAdapter(byte.class, short.class);
240         assertIllegalArgumentExceptionIllegalAdapter(byte.class, char.class);
241         assertIllegalArgumentExceptionIllegalAdapter(byte.class, double.class);
242         assertIllegalArgumentExceptionIllegalAdapter(byte.class, float.class);
243         assertIllegalArgumentExceptionIllegalAdapter(byte.class, Object.class);
244         assertIllegalArgumentExceptionIllegalAdapter(byte.class, Integer.class);
245         assertIllegalArgumentExceptionIllegalAdapter(byte.class, Long.class);
246         assertIllegalArgumentExceptionIllegalAdapter(byte.class, long[].class);
247         assertIllegalArgumentExceptionIllegalAdapter(byte.class, int[].class);
248         assertIllegalArgumentExceptionIllegalAdapter(byte.class, Integer[].class);
249         assertIllegalArgumentExceptionIllegalAdapter(byte.class, Long[].class);
250 
251         assertIllegalArgumentExceptionIsNotWiderThan(int.class, int.class);
252     }
