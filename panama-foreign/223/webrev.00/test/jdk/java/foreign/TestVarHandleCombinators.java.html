 46     @Test
 47     public void testElementAccess() {
 48         VarHandle vh = MemoryHandles.varHandle(byte.class, ByteOrder.nativeOrder());
 49         vh = MemoryHandles.withStride(vh, 1);
 50 
 51         byte[] arr = { 0, 0, -1, 0 };
 52         MemorySegment segment = MemorySegments.ofArray(arr);
 53         MemoryAddress addr = segment.baseAddress();
 54 
 55         assertEquals((byte) vh.get(addr, 2), (byte) -1);
 56     }
 57 
 58     @Test(expectedExceptions = IllegalStateException.class)
 59     public void testUnalignedElement() {
 60         VarHandle vh = MemoryHandles.varHandle(byte.class, 4, ByteOrder.nativeOrder());
 61         vh = MemoryHandles.withStride(vh, 2);
 62         MemorySegment segment = MemorySegments.ofArray(new byte[4]);
 63         vh.get(segment.baseAddress(), 1L); //should throw
 64     }

 66     public void testZeroStrideElement() {

172     @Test
173     public void testNestedSequenceAccess() {
174         int outer_size = 10;
175         int inner_size = 5;
176 
177         //[10 : [5 : [x32 i32]]]
178 
179         VarHandle vh = MemoryHandles.varHandle(int.class, ByteOrder.nativeOrder());
180         vh = MemoryHandles.withOffset(vh, 4);
181         VarHandle inner_vh = MemoryHandles.withStride(vh, 8);
182         VarHandle outer_vh = MemoryHandles.withStride(inner_vh, 5 * 8);
183         int count = 0;
184         try (MemorySegment segment = MemorySegments.allocateNative(inner_size * outer_size * 8)) {
185             for (long i = 0; i < outer_size; i++) {
186                 for (long j = 0; j < inner_size; j++) {
187                     outer_vh.set(segment.baseAddress(), i, j, count);
188                     assertEquals(
189                             (int)inner_vh.get(segment.baseAddress().addOffset(i * inner_size * 8), j),
190                             count);
191                     count++;
192                 }
193             }
194         }
195     }
