 35 NotificationThread* NotificationThread::_instance = NULL;
 36 
 37 void NotificationThread::initialize() {
 38   EXCEPTION_MARK;
 39 
 40   const char* name = "Notification Thread";
 41   Handle string = java_lang_String::create_from_str(name, CHECK);
 42 
 43   // Initialize thread_oop to put it into the system threadGroup
 44   Handle thread_group (THREAD, Universe::system_thread_group());
 45   Handle thread_oop = JavaCalls::construct_new_instance(
 46                           SystemDictionary::Thread_klass(),
 47                           vmSymbols::threadgroup_string_void_signature(),
 48                           thread_group,
 49                           string,
 50                           CHECK);
 51 
 52   Klass* group = SystemDictionary::ThreadGroup_klass();
 53   JavaValue result(T_VOID);
 54   JavaCalls::call_special(&result,
 55                           thread_group,
 56                           group,
 57                           vmSymbols::add_method_name(),
 58                           vmSymbols::thread_void_signature(),
 59                           thread_oop,
 60                           THREAD);
 61   {
 62     MutexLocker mu(THREAD, Threads_lock);
 63     NotificationThread* thread =  new NotificationThread(&notification_thread_entry);
 64 
 65     // At this point it may be possible that no osthread was created for the
 66     // JavaThread due to lack of memory. We would have to throw an exception
 67     // in that case. However, since this must work and we do not allow
 68     // exceptions anyway, check and abort if this fails.
 69     if (thread == NULL || thread->osthread() == NULL) {
 70       vm_exit_during_initialization("java.lang.OutOfMemoryError",
 71                                     os::native_thread_creation_failed_msg());
 72     }
 73 
 74     java_lang_Thread::set_thread(thread_oop(), thread);
 75     java_lang_Thread::set_priority(thread_oop(), NearMaxPriority);
 76     java_lang_Thread::set_daemon(thread_oop());
 77     thread->set_threadObj(thread_oop());
 78     _instance = thread;
 79 
 80     Threads::add(thread);
 81     Thread::start(thread);
 82   }
 83 }

 87 void NotificationThread::notification_thread_entry(JavaThread* jt, TRAPS) {
 88   while (true) {
 89     bool sensors_changed = false;
 90     bool has_dcmd_notification_event = false;
 91     bool has_gc_notification_event = false;
 92     {
 93       // Need state transition ThreadBlockInVM so that this thread
 94       // will be handled by safepoint correctly when this thread is
 95       // notified at a safepoint.
 96 
 97       ThreadBlockInVM tbivm(jt);
 98 
 99       MonitorLocker ml(Notification_lock, Mutex::_no_safepoint_check_flag);
100       // Process all available work on each (outer) iteration, rather than
101       // only the first recognized bit of work, to avoid frequently true early
102       // tests from potentially starving later work.  Hence the use of
103       // arithmetic-or to combine results; we don't want short-circuiting.
104       while (((sensors_changed = LowMemoryDetector::has_pending_requests()) |
105               (has_dcmd_notification_event = DCmdFactory::has_pending_jmx_notification()) |
106               (has_gc_notification_event = GCNotifier::has_event()))
107              == 0) {
108         // Wait as a suspend equalent until notified that there is some work to do.
109         ml.wait(0, true);
110       }
111 
112     }
113 
114     if (sensors_changed) {
115       LowMemoryDetector::process_sensor_changes(jt);
116     }
117 
118     if(has_gc_notification_event) {
119       GCNotifier::sendNotification(CHECK);
120     }
121 
122     if(has_dcmd_notification_event) {
123       DCmdFactory::send_notification(CHECK);
124     }
125 
126   }
127 }
