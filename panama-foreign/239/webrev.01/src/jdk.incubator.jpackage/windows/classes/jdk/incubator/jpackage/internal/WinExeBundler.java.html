 25 package jdk.incubator.jpackage.internal;
 26 
 27 import java.io.IOException;
 28 import java.io.InputStream;
 29 import java.nio.file.Files;
 30 import java.nio.file.Path;
 31 import java.text.MessageFormat;
 32 import java.util.Map;
 33 
 34 public class WinExeBundler extends AbstractBundler {
 35 
 36     static {
 37         System.loadLibrary("jpackage");
 38     }
 39 
 40     public static final BundlerParamInfo<Path> EXE_IMAGE_DIR
 41             = new StandardBundlerParam<>(
 42                     "win.exe.imageDir",
 43                     Path.class,
 44                     params -> {
 45                         Path imagesRoot = IMAGES_ROOT.fetchFrom(params);
 46                         if (!Files.exists(imagesRoot)) {
 47                             try {
 48                                 Files.createDirectories(imagesRoot);
 49                             } catch (IOException ioe) {
 50                                 return null;
 51                             }
 52                         }
 53                         return imagesRoot.resolve("win-exe.image");
 54                     },
 55                     (s, p) -> null);
 56 
 57     private final static String EXE_WRAPPER_NAME = "msiwrapper.exe";
 58 
 59     @Override
 60     public String getName() {
 61         return I18N.getString("exe.bundler.name");
 62     }
 63 
 64     @Override
 65     public String getID() {
 66         return "exe";
 67     }
 68 
 69     @Override
 70     public String getBundleType() {
 71         return "INSTALLER";
 72     }
 73 
 74     @Override
 75     public Path execute(Map<String, ? super Object> params,
 76             Path outputParentDir) throws PackagerException {
 77         return bundle(params, outputParentDir);
 78     }
 79 
 80     @Override
 81     public boolean supported(boolean platformInstaller) {
 82         return msiBundler.supported(platformInstaller);
 83     }
 84 
 85     @Override
 86     public boolean isDefault() {
 87         return true;
 88     }
 89 
 90     @Override
 91     public boolean validate(Map<String, ? super Object> params)
 92             throws ConfigException {
 93         return msiBundler.validate(params);
 94     }
 95 
 96     public Path bundle(Map<String, ? super Object> params, Path outdir)
 97             throws PackagerException {
 98 
 99         IOUtils.writableOutputDir(outdir);
100 
101         Path exeImageDir = EXE_IMAGE_DIR.fetchFrom(params);
102 
103         // Write msi to temporary directory.
104         Path msi = msiBundler.execute(params, exeImageDir);
105 
106         try {
107             new ScriptRunner()
108             .setDirectory(msi.getParent())
109             .setResourceCategoryId("resource.post-msi-script")
110             .setScriptNameSuffix("post-msi")
111             .setEnvironmentVariable("JpMsiFile", msi.toAbsolutePath().toString())
112             .run(params);
113 
114             return buildEXE(params, msi, outdir);
115         } catch (IOException ex) {
116             Log.verbose(ex);
117             throw new PackagerException(ex);
118         }
119     }
120 
121     private Path buildEXE(Map<String, ? super Object> params, Path msi,
122             Path outdir) throws IOException {
123 
124         Log.verbose(MessageFormat.format(
125                 I18N.getString("message.outputting-to-location"),
126                 outdir.toAbsolutePath().toString()));
127 
128         // Copy template msi wrapper next to msi file
129         final Path exePath = IOUtils.replaceSuffix(msi, ".exe");
130         try (InputStream is = OverridableResource.readDefault(EXE_WRAPPER_NAME)) {
131             Files.copy(is, exePath);
132         }
133 
134         new ExecutableRebrander().addAction((resourceLock) -> {
135             // Embed msi in msi wrapper exe.
136             embedMSI(resourceLock, msi.toAbsolutePath().toString());
137         }).rebrandInstaller(params, exePath);
138 
139         Path dstExePath = outdir.toAbsolutePath().resolve(exePath.getFileName());
140         Files.deleteIfExists(dstExePath);
141 
142         Files.copy(exePath, dstExePath);
143 
144         Log.verbose(MessageFormat.format(
145                 I18N.getString("message.output-location"),
146                 outdir.toAbsolutePath().toString()));
147 
148         return dstExePath;
149     }
150 
151     private final WinMsiBundler msiBundler = new WinMsiBundler();
152 
153     private static native int embedMSI(long resourceLock, String msiPath);
154 }
