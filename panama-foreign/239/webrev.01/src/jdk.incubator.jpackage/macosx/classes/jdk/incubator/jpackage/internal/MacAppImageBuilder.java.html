237     @Override
238     public void prepareApplicationFiles(Map<String, ? super Object> params)
239             throws IOException {
240         Files.createDirectories(macOSDir);
241 
242         Map<String, ? super Object> originalParams = new HashMap<>(params);
243         // Generate PkgInfo
244         Path pkgInfoFile = contentsDir.resolve("PkgInfo");
245         Files.createFile(pkgInfoFile);
246         writePkgInfo(pkgInfoFile);
247 
248         Path executable = macOSDir.resolve(getLauncherName(params));
249 
250         // create the main app launcher
251         try (InputStream is_launcher =
252                 getResourceAsStream("jpackageapplauncher")) {
253             // Copy executable and library to MacOS folder
254             writeEntry(is_launcher, executable);
255         }
256         executable.toFile().setExecutable(true, false);
257         // generate main app launcher config file
258         writeCfgFile(params);
259 
260         // create additional app launcher(s) and config file(s)
261         List<Map<String, ? super Object>> entryPoints =
262                 ADD_LAUNCHERS.fetchFrom(params);
263         for (Map<String, ? super Object> entryPoint : entryPoints) {
264             Map<String, ? super Object> tmp =
265                     AddLauncherArguments.merge(originalParams, entryPoint);
266 
267             // add executable for add launcher
268             Path addExecutable = macOSDir.resolve(getLauncherName(tmp));
269             try (InputStream is = getResourceAsStream("jpackageapplauncher");) {
270                 writeEntry(is, addExecutable);
271             }
272             addExecutable.toFile().setExecutable(true, false);
273 
274             // add config file for add launcher
275             writeCfgFile(tmp);
276         }
277 
278         // Copy class path entries to Java folder
279         copyApplication(params);
280 
281         /*********** Take care of "config" files *******/
282 
283         createResource(TEMPLATE_BUNDLE_ICON, params)
284                 .setCategory("icon")
285                 .setExternal(ICON_ICNS.fetchFrom(params))
286                 .saveToFile(resourcesDir.resolve(APP_NAME.fetchFrom(params)
287                         + ".icns"));
288 
289         // copy file association icons
290         for (Map<String, ?
291                 super Object> fa : FILE_ASSOCIATIONS.fetchFrom(params)) {
292             Path f = FA_ICON.fetchFrom(fa);
293             if (IOUtils.exists(f)) {
294                 IOUtils.copyFile(f, resourcesDir.resolve(f.getFileName()));
295 
296             }
297         }
298 
299         copyRuntimeFiles(params);
300         sign(params);
301     }

303     private void copyRuntimeFiles(Map<String, ? super Object> params)
304             throws IOException {
305         // Generate Info.plist
306         writeInfoPlist(contentsDir.resolve("Info.plist"), params);
307 
308         // generate java runtime info.plist
309         writeRuntimeInfoPlist(
310                 runtimeDir.resolve("Contents/Info.plist"), params);
311 
312         // copy library
313         Path runtimeMacOSDir = Files.createDirectories(
314                 runtimeDir.resolve("Contents/MacOS"));
315 
316         final Path jliName = Path.of("libjli.dylib");
317         try (Stream<Path> walk = Files.walk(runtimeRoot.resolve("lib"))) {
318             final Path jli = walk
319                     .filter(file -> file.getFileName().equals(jliName))
320                     .findFirst()
321                     .get();
322             Files.copy(jli, runtimeMacOSDir.resolve(jliName));
323         }
324     }

326     private void sign(Map<String, ? super Object> params) throws IOException {
327         if (Optional.ofNullable(
328                 SIGN_BUNDLE.fetchFrom(params)).orElse(Boolean.TRUE)) {
329             try {
330                 addNewKeychain(params);
331             } catch (InterruptedException e) {
332                 Log.error(e.getMessage());
333             }
334             String signingIdentity =
335                     DEVELOPER_ID_APP_SIGNING_KEY.fetchFrom(params);
336             if (signingIdentity != null) {
337                 prepareEntitlements(params);
338                 signAppBundle(params, root, signingIdentity,
339                         BUNDLE_ID_SIGNING_PREFIX.fetchFrom(params),
340                         getConfig_Entitlements(params));
341             }
342             restoreKeychainList(params);
343         }
344     }

362     private String getBundleName(Map<String, ? super Object> params) {
363         if (MAC_CF_BUNDLE_NAME.fetchFrom(params) != null) {
364             String bn = MAC_CF_BUNDLE_NAME.fetchFrom(params);
365             if (bn.length() > 16) {
366                 Log.error(MessageFormat.format(I18N.getString(
367                         "message.bundle-name-too-long-warning"),
368                         MAC_CF_BUNDLE_NAME.getID(), bn));
369             }
370             return MAC_CF_BUNDLE_NAME.fetchFrom(params);
371         } else if (APP_NAME.fetchFrom(params) != null) {
372             return APP_NAME.fetchFrom(params);
373         } else {
374             String nm = MAIN_CLASS.fetchFrom(params);
375             if (nm.length() > 16) {
376                 nm = nm.substring(0, 16);
377             }
378             return nm;
379         }
380     }

382     private void writeRuntimeInfoPlist(Path file,
383             Map<String, ? super Object> params) throws IOException {
384         Map<String, String> data = new HashMap<>();
385         String identifier = StandardBundlerParam.isRuntimeInstaller(params) ?
386                 MAC_CF_BUNDLE_IDENTIFIER.fetchFrom(params) :
387                 "com.oracle.java." + MAC_CF_BUNDLE_IDENTIFIER.fetchFrom(params);
388         data.put("CF_BUNDLE_IDENTIFIER", identifier);
389         String name = StandardBundlerParam.isRuntimeInstaller(params) ?
390                 getBundleName(params): "Java Runtime Image";
391         data.put("CF_BUNDLE_NAME", name);
392         data.put("CF_BUNDLE_VERSION", VERSION.fetchFrom(params));
393         data.put("CF_BUNDLE_SHORT_VERSION_STRING", VERSION.fetchFrom(params));
394 
395         createResource(TEMPLATE_RUNTIME_INFO_PLIST, params)
396                 .setPublicName("Runtime-Info.plist")
397                 .setCategory(I18N.getString("resource.runtime-info-plist"))
398                 .setSubstitutionData(data)
399                 .saveToFile(file);
400     }

402     private void writeStringArrayPlist(StringBuilder sb, String key,
403             List<String> values) {
404         if (values != null && !values.isEmpty()) {
405             sb.append("  <key>").append(key).append("</key>\n").append("   <array>\n");
406             values.forEach((value) -> {
407                 sb.append("    <string>").append(value).append("</string>\n");
408             });
409             sb.append("   </array>\n");
410         }
411     }

427     private void writeInfoPlist(Path file, Map<String, ? super Object> params)
428             throws IOException {
429         Log.verbose(MessageFormat.format(I18N.getString(
430                 "message.preparing-info-plist"), file.toAbsolutePath()));
431 
432         //prepare config for exe
433         //Note: do not need CFBundleDisplayName if we don't support localization
434         Map<String, String> data = new HashMap<>();
435         data.put("DEPLOY_ICON_FILE", APP_NAME.fetchFrom(params) + ".icns");
436         data.put("DEPLOY_BUNDLE_IDENTIFIER",
437                 MAC_CF_BUNDLE_IDENTIFIER.fetchFrom(params));
438         data.put("DEPLOY_BUNDLE_NAME",
439                 getBundleName(params));
440         data.put("DEPLOY_BUNDLE_COPYRIGHT", COPYRIGHT.fetchFrom(params));
441         data.put("DEPLOY_LAUNCHER_NAME", getLauncherName(params));
442         data.put("DEPLOY_BUNDLE_SHORT_VERSION", VERSION.fetchFrom(params));
443         data.put("DEPLOY_BUNDLE_CFBUNDLE_VERSION", VERSION.fetchFrom(params));
444 
445         StringBuilder bundleDocumentTypes = new StringBuilder();
446         StringBuilder exportedTypes = new StringBuilder();
447         for (Map<String, ? super Object>
448                 fileAssociation : FILE_ASSOCIATIONS.fetchFrom(params)) {
449 
450             List<String> extensions = FA_EXTENSIONS.fetchFrom(fileAssociation);
451             if (extensions == null) {
452                 Log.verbose(I18N.getString(
453                         "message.creating-association-with-null-extension"));
454             }
455 
456             String itemContentType = MAC_CF_BUNDLE_IDENTIFIER.fetchFrom(params)
457                     + "." + ((extensions == null || extensions.isEmpty())
458                     ? "mime" : extensions.get(0));
459             String description = FA_DESCRIPTION.fetchFrom(fileAssociation);
460             Path icon = FA_ICON.fetchFrom(fileAssociation);
461 
462             bundleDocumentTypes.append(" <dict>\n");
463             writeStringArrayPlist(bundleDocumentTypes, "LSItemContentTypes",
464                     Arrays.asList(itemContentType));
465             writeStringPlist(bundleDocumentTypes, "CFBundleTypeName", description);
466             writeStringPlist(bundleDocumentTypes, "LSHandlerRank",
467                     FA_MAC_LSHANDLERRANK.fetchFrom(fileAssociation));
468             writeStringPlist(bundleDocumentTypes, "CFBundleTypeRole",
469                     FA_MAC_CFBUNDLETYPEROLE.fetchFrom(fileAssociation));
470             writeStringPlist(bundleDocumentTypes, "NSPersistentStoreTypeKey",
471                     FA_MAC_NSSTORETYPEKEY.fetchFrom(fileAssociation));
472             writeStringPlist(bundleDocumentTypes, "NSDocumentClass",
473                     FA_MAC_NSDOCUMENTCLASS.fetchFrom(fileAssociation));
474             writeBoolPlist(bundleDocumentTypes, "LSIsAppleDefaultForType",
475                     "true");
476             writeBoolPlist(bundleDocumentTypes, "LSTypeIsPackage",
477                     FA_MAC_LSTYPEISPACKAGE.fetchFrom(fileAssociation));
478             writeBoolPlist(bundleDocumentTypes, "LSSupportsOpeningDocumentsInPlace",
479                     FA_MAC_LSDOCINPLACE.fetchFrom(fileAssociation));
480             writeBoolPlist(bundleDocumentTypes, "UISupportsDocumentBrowser",
481                     FA_MAC_UIDOCBROWSER.fetchFrom(fileAssociation));
482             if (IOUtils.exists(icon)) {
483                 writeStringPlist(bundleDocumentTypes, "CFBundleTypeIconFile",
484                         icon.getFileName().toString());
485             }
486             bundleDocumentTypes.append("  </dict>\n");
487 
488             exportedTypes.append("  <dict>\n");
489             writeStringPlist(exportedTypes, "UTTypeIdentifier",
490                     itemContentType);
491             writeStringPlist(exportedTypes, "UTTypeDescription",
492                     description);
493             writeStringArrayPlist(exportedTypes, "UTTypeConformsTo",
494                     FA_MAC_UTTYPECONFORMSTO.fetchFrom(fileAssociation));
495 
496             if (IOUtils.exists(icon)) {
497                 writeStringPlist(exportedTypes, "UTTypeIconFile",
498                         icon.getFileName().toString());
499             }
500             exportedTypes.append("\n")
501                     .append("  <key>UTTypeTagSpecification</key>\n")
502                     .append("  <dict>\n")
503                     .append("\n");
504             writeStringArrayPlist(exportedTypes, "public.filename-extension",
505                     extensions);
506             writeStringArrayPlist(exportedTypes, "public.mime-type",
507                     FA_CONTENT_TYPE.fetchFrom(fileAssociation));
508             writeStringArrayPlist(exportedTypes, "NSExportableTypes",
509                     FA_MAC_NSEXPORTABLETYPES.fetchFrom(fileAssociation));
510             exportedTypes.append("  </dict>\n").append(" </dict>\n");
511         }
512         String associationData;
513         if (bundleDocumentTypes.length() > 0) {
514             associationData =
515                     "\n  <key>CFBundleDocumentTypes</key>\n  <array>\n"
516                     + bundleDocumentTypes.toString()
517                     + "  </array>\n\n"
518                     + "  <key>UTExportedTypeDeclarations</key>\n  <array>\n"
519                     + exportedTypes.toString()
520                     + "  </array>\n";
521         } else {
522             associationData = "";
523         }
524         data.put("DEPLOY_FILE_ASSOCIATIONS", associationData);
525 
526         createResource(TEMPLATE_INFO_PLIST_LITE, params)
527                 .setCategory(I18N.getString("resource.app-info-plist"))
528                 .setSubstitutionData(data)
529                 .setPublicName("Info.plist")
530                 .saveToFile(file);
531     }

543     public static void addNewKeychain(Map<String, ? super Object> params)
544                                     throws IOException, InterruptedException {
545         if (Platform.getMajorVersion() < 10 ||
546                 (Platform.getMajorVersion() == 10 &&
547                 Platform.getMinorVersion() < 12)) {
548             // we need this for OS X 10.12+
549             return;
550         }
551 
552         String keyChain = SIGNING_KEYCHAIN.fetchFrom(params);
553         if (keyChain == null || keyChain.isEmpty()) {
554             return;
555         }
556 
557         // get current keychain list
558         String keyChainPath = Path.of(keyChain).toAbsolutePath().toString();
559         List<String> keychainList = new ArrayList<>();
560         int ret = IOUtils.getProcessOutput(
561                 keychainList, "security", "list-keychains");
562         if (ret != 0) {
563             Log.error(I18N.getString("message.keychain.error"));
564             return;
565         }
566 
567         boolean contains = keychainList.stream().anyMatch(
568                     str -> str.trim().equals("\""+keyChainPath.trim()+"\""));
569         if (contains) {
570             // keychain is already added in the search list
571             return;
572         }
573 
574         keyChains = new ArrayList<>();
575         // remove "
576         keychainList.forEach((String s) -> {
577             String path = s.trim();
578             if (path.startsWith("\"") && path.endsWith("\"")) {
579                 path = path.substring(1, path.length()-1);
580             }
581             keyChains.add(path);
582         });
583 
584         List<String> args = new ArrayList<>();
585         args.add("security");
586         args.add("list-keychains");
587         args.add("-s");
588 
589         args.addAll(keyChains);
590         args.add(keyChain);
591 
592         ProcessBuilder  pb = new ProcessBuilder(args);
593         IOUtils.exec(pb);
594     }

596     public static void restoreKeychainList(Map<String, ? super Object> params)
597             throws IOException{
598         if (Platform.getMajorVersion() < 10 ||
599                 (Platform.getMajorVersion() == 10 &&
600                 Platform.getMinorVersion() < 12)) {
601             // we need this for OS X 10.12+
602             return;
603         }
604 
605         if (keyChains == null || keyChains.isEmpty()) {
606             return;
607         }
608 
609         List<String> args = new ArrayList<>();
610         args.add("security");
611         args.add("list-keychains");
612         args.add("-s");
613 
614         args.addAll(keyChains);
615 
616         ProcessBuilder  pb = new ProcessBuilder(args);
617         IOUtils.exec(pb);
618     }

620     static void signAppBundle(
621             Map<String, ? super Object> params, Path appLocation,
622             String signingIdentity, String identifierPrefix, Path entitlements)
623             throws IOException {
624         AtomicReference<IOException> toThrow = new AtomicReference<>();
625         String appExecutable = "/Contents/MacOS/" + APP_NAME.fetchFrom(params);
626         String keyChain = SIGNING_KEYCHAIN.fetchFrom(params);
627 
628         // sign all dylibs and executables
629         try (Stream<Path> stream = Files.walk(appLocation)) {
630             stream.peek(path -> { // fix permissions
631                 try {
632                     Set<PosixFilePermission> pfp =
633                             Files.getPosixFilePermissions(path);
634                     if (!pfp.contains(PosixFilePermission.OWNER_WRITE)) {
635                         pfp = EnumSet.copyOf(pfp);
636                         pfp.add(PosixFilePermission.OWNER_WRITE);
637                         Files.setPosixFilePermissions(path, pfp);
638                     }
639                 } catch (IOException e) {
640                     Log.verbose(e);
641                 }
642             }).filter(p -> Files.isRegularFile(p) &&
643                       (Files.isExecutable(p) || p.toString().endsWith(".dylib"))
644                       && !(p.toString().endsWith(appExecutable)
645                       || p.toString().contains("/Contents/runtime")
646                       || p.toString().contains("/Contents/Frameworks"))
647                      ).forEach(p -> {
648                 // noinspection ThrowableResultOfMethodCallIgnored
649                 if (toThrow.get() != null) return;
650 
651                 // If p is a symlink then skip the signing process.
652                 if (Files.isSymbolicLink(p)) {
653                     Log.verbose(MessageFormat.format(I18N.getString(
654                             "message.ignoring.symlink"), p.toString()));
655                 } else if (isFileSigned(p)) {
656                     // executable or lib already signed
657                     Log.verbose(MessageFormat.format(I18N.getString(
658                             "message.already.signed"), p.toString()));
659                 } else {
660                     List<String> args = new ArrayList<>();
661                     args.addAll(Arrays.asList("codesign",
662                             "--timestamp",
663                             "--options", "runtime",
664                             "-s", signingIdentity,
665                             "--prefix", identifierPrefix,
666                             "-vvvv"));
667                     if (keyChain != null && !keyChain.isEmpty()) {
668                         args.add("--keychain");
669                         args.add(keyChain);
670                     }
671 
672                     if (Files.isExecutable(p)) {
673                         if (entitlements != null) {
674                             args.add("--entitlements");
675                             args.add(entitlements.toString());
676                         }
677                     }
678 
679                     args.add(p.toString());
680 
681                     try {
682                         Set<PosixFilePermission> oldPermissions =
683                                 Files.getPosixFilePermissions(p);
684                         p.toFile().setWritable(true, true);
685 
686                         ProcessBuilder pb = new ProcessBuilder(args);
687 
688                         IOUtils.exec(pb);
689 
690                         Files.setPosixFilePermissions(p, oldPermissions);
691                     } catch (IOException ioe) {
692                         toThrow.set(ioe);
693                     }
694                 }
695             });
696         }
697         IOException ioe = toThrow.get();
698         if (ioe != null) {
699             throw ioe;
700         }
701 
702         // sign all runtime and frameworks
703         Consumer<? super Path> signIdentifiedByPList = path -> {
704             //noinspection ThrowableResultOfMethodCallIgnored
705             if (toThrow.get() != null) return;
706 
707             try {
708                 List<String> args = new ArrayList<>();
709                 args.addAll(Arrays.asList("codesign",
710                         "--timestamp",
711                         "--options", "runtime",
712                         "--force",
713                         "-s", signingIdentity, // sign with this key
714                         "--prefix", identifierPrefix,
715                         // use the identifier as a prefix
716                         "-vvvv"));
717 
718                 if (keyChain != null && !keyChain.isEmpty()) {
719                     args.add("--keychain");
720                     args.add(keyChain);
721                 }
722                 args.add(path.toString());
723                 ProcessBuilder pb = new ProcessBuilder(args);
724 
725                 IOUtils.exec(pb);
726             } catch (IOException e) {
727                 toThrow.set(e);
728             }
729         };
730 
731         Path javaPath = appLocation.resolve("Contents/runtime");
732         if (Files.isDirectory(javaPath)) {
733             signIdentifiedByPList.accept(javaPath);
734 
735             ioe = toThrow.get();
736             if (ioe != null) {
737                 throw ioe;
738             }
739         }
740         Path frameworkPath = appLocation.resolve("Contents/Frameworks");
741         if (Files.isDirectory(frameworkPath)) {
742             try (var fileList = Files.list(frameworkPath)) {
743                 fileList.forEach(signIdentifiedByPList);
744             }
745 
746             ioe = toThrow.get();
747             if (ioe != null) {
748                 throw ioe;
749             }
750         }
751 
752         // sign the app itself
753         List<String> args = new ArrayList<>();
754         args.addAll(Arrays.asList("codesign",
755                 "--timestamp",
756                 "--options", "runtime",
757                 "--force",
758                 "-s", signingIdentity,
759                 "-vvvv"));
760 
761         if (keyChain != null && !keyChain.isEmpty()) {
762             args.add("--keychain");
763             args.add(keyChain);
764         }
765 
766         if (entitlements != null) {
767             args.add("--entitlements");
768             args.add(entitlements.toString());
769         }
770 
771         args.add(appLocation.toString());
772 
773         ProcessBuilder pb =
774                 new ProcessBuilder(args.toArray(new String[args.size()]));
775 
776         IOUtils.exec(pb);
777     }

779     private static boolean isFileSigned(Path file) {
780         ProcessBuilder pb =
781                 new ProcessBuilder("codesign", "--verify", file.toString());
782 
783         try {
784             IOUtils.exec(pb);
785         } catch (IOException ex) {
786             return false;
787         }
788 
789         return true;
790     }

792     private static String extractBundleIdentifier(Map<String, Object> params) {
793         if (PREDEFINED_APP_IMAGE.fetchFrom(params) == null) {
794             return null;
795         }
796 
797         try {
798             Path infoPList = PREDEFINED_APP_IMAGE.fetchFrom(params).resolve("Contents").
799                     resolve("Info.plist");
800 
801             DocumentBuilderFactory dbf
802                     = DocumentBuilderFactory.newDefaultInstance();
803             dbf.setFeature("http://apache.org/xml/features/" +
804                            "nonvalidating/load-external-dtd", false);
805             DocumentBuilder b = dbf.newDocumentBuilder();
806             org.w3c.dom.Document doc = b.parse(Files.newInputStream(infoPList));
807 
808             XPath xPath = XPathFactory.newInstance().newXPath();
809             // Query for the value of <string> element preceding <key>
810             // element with value equal to CFBundleIdentifier
811             String v = (String) xPath.evaluate(
812                     "//string[preceding-sibling::key = \"CFBundleIdentifier\"][1]",
813                     doc, XPathConstants.STRING);
814 
815             if (v != null && !v.isEmpty()) {
816                 return v;
817             }
818         } catch (Exception ex) {
819             Log.verbose(ex);
820         }
821 
822         return null;
823     }
