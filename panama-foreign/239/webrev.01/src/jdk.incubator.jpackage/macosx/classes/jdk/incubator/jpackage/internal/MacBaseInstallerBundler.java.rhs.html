 26 package jdk.incubator.jpackage.internal;
 27 
 28 import java.io.ByteArrayOutputStream;

 29 import java.io.IOException;
 30 import java.io.PrintStream;
 31 import java.nio.file.Files;
 32 import java.nio.file.Path;
 33 import java.text.MessageFormat;
 34 import java.util.ArrayList;
 35 import java.util.List;
 36 import java.util.Map;
 37 import java.util.regex.Matcher;
 38 import java.util.regex.Pattern;
 39 import static jdk.incubator.jpackage.internal.StandardBundlerParam.APP_NAME;
 40 import static jdk.incubator.jpackage.internal.StandardBundlerParam.INSTALL_DIR;
 41 import static jdk.incubator.jpackage.internal.StandardBundlerParam.PREDEFINED_APP_IMAGE;
 42 import static jdk.incubator.jpackage.internal.StandardBundlerParam.VERSION;
 43 
 44 public abstract class MacBaseInstallerBundler extends AbstractBundler {
 45 
 46     public final BundlerParamInfo<Path> APP_IMAGE_TEMP_ROOT =
 47             new StandardBundlerParam<>(
 48             "mac.app.imageRoot",
 49             Path.class,
 50             params -> {
 51                 Path imageDir = IMAGES_ROOT.fetchFrom(params);

 52                 try {
 53                     if (!IOUtils.exists(imageDir)) {
 54                         Files.createDirectories(imageDir);
 55                     }
 56                     return Files.createTempDirectory(
 57                             imageDir, "image-");
 58                 } catch (IOException e) {
 59                     return imageDir.resolve(getID()+ ".image");
 60                 }
 61             },
 62             (s, p) -> Path.of(s));
 63 
 64     public static final BundlerParamInfo<String> SIGNING_KEY_USER =
 65             new StandardBundlerParam<>(
 66             Arguments.CLIOptions.MAC_SIGNING_KEY_NAME.getId(),
 67             String.class,
 68             params -> "",
 69             null);
 70 
 71     public static final BundlerParamInfo<String> SIGNING_KEYCHAIN =
 72             new StandardBundlerParam<>(
 73             Arguments.CLIOptions.MAC_SIGNING_KEYCHAIN.getId(),
 74             String.class,
 75             params -> "",
 76             null);
 77 
 78     public static final BundlerParamInfo<String> INSTALLER_NAME =
 79             new StandardBundlerParam<> (
 80             "mac.installerName",
 81             String.class,
 82             params -> {
 83                 String nm = APP_NAME.fetchFrom(params);
 84                 if (nm == null) return null;
 85 
 86                 String version = VERSION.fetchFrom(params);
 87                 if (version == null) {
 88                     return nm;
 89                 } else {
 90                     return nm + "-" + version;
 91                 }
 92             },
 93             (s, p) -> s);
 94 
 95     protected static String getInstallDir(
 96             Map<String, ? super Object>  params) {
 97         String returnValue = INSTALL_DIR.fetchFrom(params);
 98         if (returnValue == null) {
 99             if (StandardBundlerParam.isRuntimeInstaller(params)) {
100                 returnValue = "/Library/Java/JavaVirtualMachines";
101             } else {
102                returnValue = "/Applications";
103             }
104         }
105         return returnValue;
106     }
107 
108     public MacBaseInstallerBundler() {
109         appImageBundler = new MacAppBundler().setDependentTask(true);
110     }
111 
112     protected void validateAppImageAndBundeler(
113             Map<String, ? super Object> params) throws ConfigException {
114         if (PREDEFINED_APP_IMAGE.fetchFrom(params) != null) {
115             Path applicationImage = PREDEFINED_APP_IMAGE.fetchFrom(params);
116             if (!IOUtils.exists(applicationImage)) {
117                 throw new ConfigException(
118                         MessageFormat.format(I18N.getString(
119                                 "message.app-image-dir-does-not-exist"),
120                                 PREDEFINED_APP_IMAGE.getID(),
121                                 applicationImage.toString()),
122                         MessageFormat.format(I18N.getString(
123                                 "message.app-image-dir-does-not-exist.advice"),
124                                 PREDEFINED_APP_IMAGE.getID()));
125             }
126             if (APP_NAME.fetchFrom(params) == null) {
127                 throw new ConfigException(
128                         I18N.getString("message.app-image-requires-app-name"),
129                         I18N.getString(
130                             "message.app-image-requires-app-name.advice"));
131             }
132         } else {
133             appImageBundler.validate(params);
134         }
135     }
136 
137     protected Path prepareAppBundle(Map<String, ? super Object> params)
138             throws PackagerException {
139         Path predefinedImage =
140                 StandardBundlerParam.getPredefinedAppImage(params);
141         if (predefinedImage != null) {
142             return predefinedImage;
143         }
144         Path appImageRoot = APP_IMAGE_TEMP_ROOT.fetchFrom(params);
145 
146         return appImageBundler.execute(params, appImageRoot);
147     }
148 
149     @Override
150     public String getBundleType() {
151         return "INSTALLER";
152     }
153 
154     public static String findKey(String key, String keychainName,
155             boolean verbose) {
156         if (Platform.getPlatform() != Platform.MAC) {
157             return null;
158         }
159 
160         try (ByteArrayOutputStream baos = new ByteArrayOutputStream();
161                 PrintStream ps = new PrintStream(baos)) {
162             List<String> searchOptions = new ArrayList<>();
163             searchOptions.add("security");
164             searchOptions.add("find-certificate");
165             searchOptions.add("-c");
166             searchOptions.add(key);
167             searchOptions.add("-a");
168             if (keychainName != null && !keychainName.isEmpty()) {
169                 searchOptions.add(keychainName);
170             }
171 
172             ProcessBuilder pb = new ProcessBuilder(searchOptions);
173 
174             IOUtils.exec(pb, false, ps);
175             Pattern p = Pattern.compile("\"alis\"<blob>=\"([^\"]+)\"");
176             Matcher m = p.matcher(baos.toString());
177             if (!m.find()) {
178                 Log.error(MessageFormat.format(I18N.getString(
179                         "error.cert.not.found"), key, keychainName));
180                 return null;
181             }
182             String matchedKey = m.group(1);
183             if (m.find()) {
184                 Log.error(MessageFormat.format(I18N.getString(
185                         "error.multiple.certs.found"), key, keychainName));
186                 return null;
187             }
188             Log.verbose("Using key '" + matchedKey + "'");
189             return matchedKey;
190         } catch (IOException ioe) {
191             Log.verbose(ioe);
192             return null;
193         }
194     }
195 
196     private final Bundler appImageBundler;
197 }
