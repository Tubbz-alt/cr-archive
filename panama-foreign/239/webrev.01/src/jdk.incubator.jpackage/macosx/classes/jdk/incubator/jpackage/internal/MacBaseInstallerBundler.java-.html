 39 import static jdk.incubator.jpackage.internal.StandardBundlerParam.APP_NAME;
 40 import static jdk.incubator.jpackage.internal.StandardBundlerParam.INSTALL_DIR;
 41 import static jdk.incubator.jpackage.internal.StandardBundlerParam.PREDEFINED_APP_IMAGE;
 42 import static jdk.incubator.jpackage.internal.StandardBundlerParam.VERSION;
 43 
 44 public abstract class MacBaseInstallerBundler extends AbstractBundler {
 45 
 46     public final BundlerParamInfo<File> APP_IMAGE_TEMP_ROOT =
 47             new StandardBundlerParam<>(
 48             "mac.app.imageRoot",
 49             File.class,
 50             params -> {
 51                 File imageDir = IMAGES_ROOT.fetchFrom(params);
 52                 if (!imageDir.exists()) imageDir.mkdirs();
 53                 try {
 54                     return Files.createTempDirectory(
 55                             imageDir.toPath(), "image-").toFile();
 56                 } catch (IOException e) {
 57                     return new File(imageDir, getID()+ ".image");
 58                 }
 59             },
 60             (s, p) -> new File(s));
 61 
 62     public static final BundlerParamInfo<String> SIGNING_KEY_USER =
 63             new StandardBundlerParam<>(
 64             Arguments.CLIOptions.MAC_SIGNING_KEY_NAME.getId(),
 65             String.class,
 66             params -> "",
 67             null);
 68 
 69     public static final BundlerParamInfo<String> SIGNING_KEYCHAIN =
 70             new StandardBundlerParam<>(
 71             Arguments.CLIOptions.MAC_SIGNING_KEYCHAIN.getId(),
 72             String.class,
 73             params -> "",
 74             null);
 75 
 76     public static final BundlerParamInfo<String> INSTALLER_NAME =
 77             new StandardBundlerParam<> (
 78             "mac.installerName",
 79             String.class,
 80             params -> {
 81                 String nm = APP_NAME.fetchFrom(params);
 82                 if (nm == null) return null;
 83 
 84                 String version = VERSION.fetchFrom(params);
 85                 if (version == null) {
 86                     return nm;
 87                 } else {
 88                     return nm + "-" + version;
 89                 }
 90             },
 91             (s, p) -> s);
 92 
 93     protected static String getInstallDir(
 94             Map<String, ? super Object>  params) {
 95         String returnValue = INSTALL_DIR.fetchFrom(params);
 96         if (returnValue == null) {
 97             if (StandardBundlerParam.isRuntimeInstaller(params)) {
 98                 returnValue = "/Library/Java/JavaVirtualMachines";
 99             } else {
100                returnValue = "/Applications";
101             }
102         }
103         return returnValue;
104     }
105 
106     public MacBaseInstallerBundler() {
107         appImageBundler = new MacAppBundler().setDependentTask(true);
108     }
109 
110     protected void validateAppImageAndBundeler(
111             Map<String, ? super Object> params) throws ConfigException {
112         if (PREDEFINED_APP_IMAGE.fetchFrom(params) != null) {
113             File applicationImage = PREDEFINED_APP_IMAGE.fetchFrom(params);
114             if (!applicationImage.exists()) {
115                 throw new ConfigException(
116                         MessageFormat.format(I18N.getString(
117                                 "message.app-image-dir-does-not-exist"),
118                                 PREDEFINED_APP_IMAGE.getID(),
119                                 applicationImage.toString()),
120                         MessageFormat.format(I18N.getString(
121                                 "message.app-image-dir-does-not-exist.advice"),
122                                 PREDEFINED_APP_IMAGE.getID()));
123             }
124             if (APP_NAME.fetchFrom(params) == null) {
125                 throw new ConfigException(
126                         I18N.getString("message.app-image-requires-app-name"),
127                         I18N.getString(
128                             "message.app-image-requires-app-name.advice"));
129             }
130         } else {
131             appImageBundler.validate(params);
132         }
133     }
134 
135     protected File prepareAppBundle(Map<String, ? super Object> params)
136             throws PackagerException {
137         File predefinedImage =
138                 StandardBundlerParam.getPredefinedAppImage(params);
139         if (predefinedImage != null) {
140             return predefinedImage;
141         }
142         File appImageRoot = APP_IMAGE_TEMP_ROOT.fetchFrom(params);
143 
144         return appImageBundler.execute(params, appImageRoot);
145     }
146 
147     @Override
148     public String getBundleType() {
149         return "INSTALLER";
150     }
151 
152     public static String findKey(String key, String keychainName,
153             boolean verbose) {
154         if (Platform.getPlatform() != Platform.MAC) {
155             return null;
156         }
157 
158         try (ByteArrayOutputStream baos = new ByteArrayOutputStream();
159                 PrintStream ps = new PrintStream(baos)) {
160             List<String> searchOptions = new ArrayList<>();
161             searchOptions.add("security");
162             searchOptions.add("find-certificate");
163             searchOptions.add("-c");
164             searchOptions.add(key);
165             searchOptions.add("-a");
166             if (keychainName != null && !keychainName.isEmpty()) {
167                 searchOptions.add(keychainName);
168             }
169 
170             ProcessBuilder pb = new ProcessBuilder(searchOptions);
171 
172             IOUtils.exec(pb, false, ps);
173             Pattern p = Pattern.compile("\"alis\"<blob>=\"([^\"]+)\"");
174             Matcher m = p.matcher(baos.toString());
175             if (!m.find()) {
176                 Log.error(MessageFormat.format(I18N.getString(
177                         "error.cert.not.found"), key, keychainName));
178                 return null;
179             }
180             String matchedKey = m.group(1);
181             if (m.find()) {
182                 Log.error(MessageFormat.format(I18N.getString(
183                         "error.multiple.certs.found"), key, keychainName));
184                 return null;
185             }
186             Log.verbose("Using key '" + matchedKey + "'");
187             return matchedKey;
188         } catch (IOException ioe) {
189             Log.verbose(ioe);
190             return null;
191         }
192     }
193 
194     private final Bundler appImageBundler;
