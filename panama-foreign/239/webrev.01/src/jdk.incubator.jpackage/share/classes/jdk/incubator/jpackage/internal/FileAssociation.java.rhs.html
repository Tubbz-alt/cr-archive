 26 package jdk.incubator.jpackage.internal;
 27 

 28 import java.nio.file.Path;
 29 import java.text.MessageFormat;
 30 import java.util.Collections;
 31 import java.util.List;
 32 import java.util.Map;
 33 import java.util.Objects;
 34 import java.util.Optional;
 35 import java.util.stream.Collectors;
 36 import static jdk.incubator.jpackage.internal.StandardBundlerParam.APP_NAME;
 37 import static jdk.incubator.jpackage.internal.StandardBundlerParam.FILE_ASSOCIATIONS;
 38 import static jdk.incubator.jpackage.internal.StandardBundlerParam.FA_EXTENSIONS;
 39 import static jdk.incubator.jpackage.internal.StandardBundlerParam.FA_CONTENT_TYPE;
 40 import static jdk.incubator.jpackage.internal.StandardBundlerParam.FA_ICON;
 41 import static jdk.incubator.jpackage.internal.StandardBundlerParam.FA_DESCRIPTION;
 42 
 43 final class FileAssociation {
 44     void verify() {
 45         if (extensions.isEmpty()) {
 46             Log.error(I18N.getString(
 47                     "message.creating-association-with-null-extension"));
 48         }
 49     }
 50 
 51     static void verify(List<FileAssociation> associations) throws ConfigException {
 52         // only one mime type per association, at least one file extention
 53         int assocIdx = 0;
 54         for (var assoc : associations) {
 55             ++assocIdx;
 56             if (assoc.mimeTypes.isEmpty()) {
 57                 String msgKey = "error.no-content-types-for-file-association";
 58                 throw new ConfigException(
 59                         MessageFormat.format(I18N.getString(msgKey), assocIdx),
 60                         MessageFormat.format(I18N.getString(msgKey + ".advice"), assocIdx));
 61             }
 62 
 63             if (assoc.mimeTypes.size() > 1) {
 64                 String msgKey = "error.too-many-content-types-for-file-association";
 65                 throw new ConfigException(
 66                         MessageFormat.format(I18N.getString(msgKey), assocIdx),
 67                         MessageFormat.format(I18N.getString(msgKey + ".advice"), assocIdx));
 68             }
 69 
 70             assoc.verify();
 71         }
 72     }
 73 
 74     static List<FileAssociation> fetchFrom(Map<String, ? super Object> params) {
 75         String launcherName = APP_NAME.fetchFrom(params);
 76 
 77         return FILE_ASSOCIATIONS.fetchFrom(params).stream().filter(
 78                 Objects::nonNull).map(fa -> {
 79                     FileAssociation assoc = new FileAssociation();
 80 
 81                     assoc.launcherPath = Path.of(launcherName);
 82                     assoc.description = FA_DESCRIPTION.fetchFrom(fa);
 83                     assoc.extensions = Optional.ofNullable(
 84                             FA_EXTENSIONS.fetchFrom(fa)).orElse(Collections.emptyList());
 85                     assoc.mimeTypes = Optional.ofNullable(
 86                             FA_CONTENT_TYPE.fetchFrom(fa)).orElse(Collections.emptyList());
 87 
 88                     Path icon = FA_ICON.fetchFrom(fa);
 89                     if (icon != null) {
 90                         assoc.iconPath = icon;
 91                     }
 92 
 93                     return assoc;
 94                 }).collect(Collectors.toList());
 95     }
 96 
 97     Path launcherPath;
 98     Path iconPath;
 99     List<String> mimeTypes;
100     List<String> extensions;
101     String description;
102 }
