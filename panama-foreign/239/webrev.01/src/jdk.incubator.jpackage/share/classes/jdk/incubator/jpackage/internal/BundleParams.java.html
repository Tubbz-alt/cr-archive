26 package jdk.incubator.jpackage.internal;
27 
28 import java.util.HashMap;
29 import java.util.Map;
30 import static jdk.incubator.jpackage.internal.StandardBundlerParam.APP_NAME;
31 
32 public class BundleParams {
33 
34     final protected Map<String, ? super Object> params;
35 
36     /**
37      * create a new bundle with all default values
38      */
39     public BundleParams() {
40         params = new HashMap<>();
41     }
42 
43     /**
44      * Create a bundle params with a copy of the params
45      * @param params map of initial parameters to be copied in.
46      */
47     public BundleParams(Map<String, ?> params) {
48         this.params = new HashMap<>(params);
49     }
50 
51     public void addAllBundleParams(Map<String, ? super Object> params) {
52         this.params.putAll(params);
53     }
54 
55     // NOTE: we do not care about application parameters here
56     // as they will be embeded into jar file manifest and
57     // java launcher will take care of them!
58 
59     public Map<String, ? super Object> getBundleParamsAsMap() {
60         return new HashMap<>(params);
61     }
62 
63     public String getName() {
64         return APP_NAME.fetchFrom(params);
65     }
66 
67     private void putUnlessNull(String param, Object value) {
68         if (value != null) {
69             params.put(param, value);
70         }
71     }
72 }
