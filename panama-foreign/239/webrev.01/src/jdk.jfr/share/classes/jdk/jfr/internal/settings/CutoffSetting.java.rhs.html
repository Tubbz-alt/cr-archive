26 package jdk.jfr.internal.settings;
27 
28 import java.util.Objects;
29 import java.util.Set;
30 
31 import jdk.jfr.Description;
32 import jdk.jfr.Label;
33 import jdk.jfr.MetadataDefinition;
34 import jdk.jfr.Name;
35 import jdk.jfr.Timespan;

36 import jdk.jfr.internal.PlatformEventType;
37 import jdk.jfr.internal.Type;
38 import jdk.jfr.internal.Utils;
39 
40 @MetadataDefinition
41 @Label("Cutoff")
42 @Description("Limit running time of event")
43 @Name(Type.SETTINGS_PREFIX + "Cutoff")
44 @Timespan
45 public final class CutoffSetting extends JDKSettingControl {
46     private final static long typeId = Type.getTypeId(CutoffSetting.class);
47 
48     private String value = "0 ns";
49     private final PlatformEventType eventType;
50 
51     public CutoffSetting(PlatformEventType eventType) {

52        this.eventType = Objects.requireNonNull(eventType);
53     }
54 
55     @Override
56     public String combine(Set<String> values) {
57         long max = 0;
58         String text = "0 ns";
59         for (String value : values) {
60             long l =  Utils.parseTimespanWithInfinity(value);
61             if (l > max) {
62                 text = value;
63                 max = l;
64             }
65         }
66         return text;
67     }
68 
69     @Override
70     public void setValue(String value) {
71         long l =  Utils.parseTimespanWithInfinity(value);
72         this.value = value;
73         eventType.setCutoff(l);
74     }
75 
76     @Override
77     public String getValue() {
78         return value;
79     }
80 
81     public static boolean isType(long typeId) {
82         return CutoffSetting.typeId == typeId;
83     }
84 
85     public static long parseValueSafe(String value) {
86         if (value == null) {
87             return 0L;
88         }
89         try {
90             return Utils.parseTimespanWithInfinity(value);
91         } catch (NumberFormatException nfe) {
92             return 0L;
93         }
94     }
95 }
