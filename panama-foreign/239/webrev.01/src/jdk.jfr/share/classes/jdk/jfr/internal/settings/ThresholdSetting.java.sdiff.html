26 package jdk.jfr.internal.settings;
27 
28 import java.util.Objects;
29 import java.util.Set;
30 
31 import jdk.jfr.Description;
32 import jdk.jfr.Label;
33 import jdk.jfr.MetadataDefinition;
34 import jdk.jfr.Name;
35 import jdk.jfr.Timespan;
36 import jdk.jfr.internal.PlatformEventType;
37 import jdk.jfr.internal.Control;
38 import jdk.jfr.internal.Type;
39 import jdk.jfr.internal.Utils;
40 @MetadataDefinition
41 @Label("Threshold")
42 @Name(Type.SETTINGS_PREFIX + "Threshold")
43 @Description("Record event with duration above or equal to threshold")
44 @Timespan
45 public final class ThresholdSetting extends Control {
46     private final static long typeId = Type.getTypeId(ThresholdSetting.class);
47     private String value = "0 ns";
48     private final PlatformEventType eventType;
49 
50     public ThresholdSetting(PlatformEventType eventType, String defaultValue) {
51        super(defaultValue);
52        this.eventType = Objects.requireNonNull(eventType);
53     }
54 
55     @Override
56     public String combine(Set<String> values) {
57         Long min = null;
58         String text = null;
59         for (String value : values) {
60             long l = Utils.parseTimespanWithInfinity(value);
61             // always accept first value
62             if (min == null) {
63                 min = l;
64                 text = value;
65             } else {
66                 if (l < min) {
67                     text = value;
68                     min = l;
69                 }
70             }
71         }
