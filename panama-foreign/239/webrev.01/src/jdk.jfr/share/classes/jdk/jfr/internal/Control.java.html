 26 package jdk.jfr.internal;
 27 
 28 import java.security.AccessControlContext;
 29 import java.security.AccessController;
 30 import java.security.PrivilegedAction;
 31 import java.util.Collections;
 32 import java.util.HashSet;
 33 import java.util.Objects;
 34 import java.util.Set;
 35 
 36 import jdk.jfr.SettingControl;
 37 import jdk.jfr.internal.settings.JDKSettingControl;
 38 
 39 public final class Control {
 40     private final AccessControlContext context;
 41     private final static int CACHE_SIZE = 5;
 42     private final Set<?>[] cachedUnions = new HashSet<?>[CACHE_SIZE];
 43     private final String[] cachedValues = new String[CACHE_SIZE];
 44     private final SettingControl delegate;
 45     private String defaultValue;
 46     private String lastValue;
 47 
 48     // called by exposed subclass in external API
 49     public Control(SettingControl delegate, String defaultValue) {
 50         this.context = PrivateAccess.getInstance().getContext(delegate);
 51         this.delegate = delegate;
 52         this.defaultValue = defaultValue;
 53         if (this.context == null && !(delegate instanceof JDKSettingControl)) {
 54             throw new InternalError("Security context can only be null for trusted setting controls");
 55         }
 56     }
 57 
 58     boolean isType(Class<? extends SettingControl> clazz) {
 59         return delegate.getClass() == clazz;
 60     }
 61 
 62     final void apply(Set<String> values) {
 63         setValue(findCombine(values));
 64     }
 65 
 66     final void setDefault() {
 67         if (defaultValue == null) {
 68             defaultValue = getValue();
 69         }
 70         apply(defaultValue);
 71     }
 72 
 73     public String getValue() {
 74         if (context == null) {
 75             // VM events requires no access control context
 76             return getValue();
 77         } else {
 78             return AccessController.doPrivileged(new PrivilegedAction<String>() {
 79                 @Override
 80                 public String run() {
 81                     try {
 82                         return delegate.getValue();
 83                     } catch (Throwable t) {
 84                         // Prevent malicious user to propagate exception callback in the wrong context
 85                         Logger.log(LogTag.JFR_SETTING, LogLevel.WARN, "Exception occurred when trying to get value for " + getClass());
 86                     }
 87                     return defaultValue != null ? defaultValue : ""; // Need to return something
 88                 }
 89             }, context);
 90         }
 91     }
 92 
 93     private void apply(String value) {
 94         if (lastValue != null && Objects.equals(value, lastValue)) {
 95             return;
 96         }
 97         setValue(value);
 98     }
 99 
100     public void setValue(String value) {
101         if (context == null) {
102             // VM events requires no access control context
103             try {
104                 delegate.setValue(value);
105             } catch (Throwable t) {
106                 Logger.log(LogTag.JFR_SETTING, LogLevel.WARN, "Exception occurred when setting value \"" + value + "\" for " + getClass());
107             }
108         } else {
109             AccessController.doPrivileged(new PrivilegedAction<Void>() {
110                 @Override
111                 public Void run() {
112                     try {
113                         delegate.setValue(value);
114                     } catch (Throwable t) {
115                         // Prevent malicious user to propagate exception callback in the wrong context
116                         Logger.log(LogTag.JFR_SETTING, LogLevel.WARN, "Exception occurred when setting value \"" + value + "\" for " + getClass());
117                     }
118                     return null;
119                 }
120             }, context);
121         }
122         lastValue = value;
123     }
124 
125 
126     public String combine(Set<String> values) {
127         if (context == null) {
128             // VM events requires no access control context
129             return delegate.combine(values);
130         }
131         return AccessController.doPrivileged(new PrivilegedAction<String>() {
132             @Override
133             public String run() {
134                 try {
135                     delegate.combine(Collections.unmodifiableSet(values));
136                 } catch (Throwable t) {
137                     // Prevent malicious user to propagate exception callback in the wrong context
138                     Logger.log(LogTag.JFR_SETTING, LogLevel.WARN, "Exception occurred when combining " + values + " for " + getClass());
139                 }
140                 return null;
141             }
142         }, context);
143     }
144 
145     private final String findCombine(Set<String> values) {
146         if (values.size() == 1) {
147             return values.iterator().next();
148         }
149         for (int i = 0; i < CACHE_SIZE; i++) {
150             if (Objects.equals(cachedUnions[i], values)) {
151                 return cachedValues[i];
152             }
153         }
154         String result = combine(values);
155         for (int i = 0; i < CACHE_SIZE - 1; i++) {
156             cachedUnions[i + 1] = cachedUnions[i];
157             cachedValues[i + 1] = cachedValues[i];
158         }
159         cachedValues[0] = result;
160         cachedUnions[0] = values;
161         return result;
162     }
163 
164     final String getDefaultValue() {
165         return defaultValue;
166     }
167 
168     final String getLastValue() {
169         return lastValue;
170     }
171 }
