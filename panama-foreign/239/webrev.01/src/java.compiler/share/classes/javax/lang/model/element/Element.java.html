 26 package javax.lang.model.element;
 27 
 28 
 29 import java.lang.annotation.Annotation;
 30 import java.lang.annotation.AnnotationTypeMismatchException;
 31 import java.lang.annotation.IncompleteAnnotationException;
 32 import java.util.List;
 33 import java.util.Set;
 34 
 35 import javax.lang.model.type.*;
 36 import javax.lang.model.util.*;
 37 
 38 /**
 39  * Represents a program element such as a module, package, class, or method.
 40  * Each element represents a compile-time language-level construct
 41  * (and not, for example, a runtime construct of the virtual machine).
 42  *
 43  * <p> Elements should be compared using the {@link #equals(Object)}
 44  * method.  There is no guarantee that any particular element will
 45  * always be represented by the same object.
 46  *
 47  * <p> To implement operations based on the class of an {@code
 48  * Element} object, either use a {@linkplain ElementVisitor visitor} or
 49  * use the result of the {@link #getKind} method.  Using {@code
 50  * instanceof} is <em>not</em> necessarily a reliable idiom for
 51  * determining the effective class of an object in this modeling
 52  * hierarchy since an implementation may choose to have a single object
 53  * implement multiple {@code Element} subinterfaces.
 54  *
 55  * @author Joseph D. Darcy
 56  * @author Scott Seligman
 57  * @author Peter von der Ah&eacute;
 58  * @see Elements
 59  * @see TypeMirror
 60  * @since 1.6
 61  */
 62 public interface Element extends javax.lang.model.AnnotatedConstruct {
 63     /**
 64      * Returns the type defined by this element.
 65      * @return the type defined by this element
 66      *
 67      * @see Types
 68      * @see ExecutableElement#asType
 69      * @see ModuleElement#asType
 70      * @see PackageElement#asType
 71      * @see TypeElement#asType
 72      * @see TypeParameterElement#asType
 73      * @see VariableElement#asType
 74      */
 75     TypeMirror asType();
 76 
 77     /**
 78      * Returns the {@code kind} of this element.
 79      *
 80      * <ul>
 81      *
 82      * <li> The kind of a {@linkplain PackageElement package} is
 83      * {@link ElementKind#PACKAGE PACKAGE}.
 84      *
 85      * <li> The kind of a {@linkplain ModuleElement module} is {@link
 86      * ElementKind#MODULE MODULE}.
 87      *
 88      * <li> The kind of a {@linkplain TypeElement type element} is one
 89      * of {@link ElementKind#ANNOTATION_TYPE ANNOTATION_TYPE}, {@link
 90      * ElementKind#CLASS CLASS}, {@link ElementKind#ENUM ENUM}, {@link
 91      * ElementKind#INTERFACE INTERFACE}, or {@link ElementKind#RECORD
 92      * RECORD}.
 93      *
 94      * <li> The kind of a {@linkplain VariableElement variable} is one
 95      * of {@link ElementKind#ENUM_CONSTANT ENUM_CONSTANT}, {@link
 96      * ElementKind#EXCEPTION_PARAMETER EXCEPTION_PARAMETER}, {@link
 97      * ElementKind#FIELD FIELD}, {@link ElementKind#LOCAL_VARIABLE
 98      * LOCAL_VARIABLE}, {@link ElementKind#PARAMETER PARAMETER},
 99      * {@link ElementKind#RESOURCE_VARIABLE RESOURCE_VARIABLE}, or
100      * {@link ElementKind#BINDING_VARIABLE BINDING_VARIABLE}.
101      *
102      * <li> The kind of an {@linkplain ExecutableElement executable}
103      * is one of {@link ElementKind#CONSTRUCTOR CONSTRUCTOR}, {@link
104      * ElementKind#INSTANCE_INIT INSTANCE_INIT}, {@link
105      * ElementKind#METHOD METHOD}, or {@link ElementKind#STATIC_INIT
106      * STATIC_INIT}.
107      *
108      * <li> The kind of a {@linkplain TypeParameterElement type parameter} is
109      * {@link ElementKind#TYPE_PARAMETER TYPE_PARAMETER}.
110      *
111      * <li> The kind of a {@linkplain RecordComponentElement record
112      * component} is {@link ElementKind#RECORD_COMPONENT
113      * RECORD_COMPONENT}.
114      *
115      * </ul>
116      *
117      * @return the kind of this element
118      */
119     ElementKind getKind();
120 
121     /**
122      * Returns the modifiers of this element, excluding annotations.
123      * Implicit modifiers, such as the {@code public} and {@code static}
124      * modifiers of interface members, are included.
125      *
126      * @return the modifiers of this element, or an empty set if there are none
127      */
128     Set<Modifier> getModifiers();
129 
130     /**
131      * Returns the simple (unqualified) name of this element.  The
132      * name of a generic type does not include any reference to its
133      * formal type parameters.
134      *
135      * For example, the simple name of the type element {@code
136      * java.util.Set<E>} is {@code "Set"}.
137      *
138      * If this element represents an unnamed {@linkplain
139      * PackageElement#getSimpleName package} or unnamed {@linkplain
140      * ModuleElement#getSimpleName module}, an <a
141      * href=Name.html#empty_name>empty name</a> is returned.
142      *
143      * If it represents a {@linkplain ExecutableElement#getSimpleName
144      * constructor}, the name "{@code <init>}" is returned.  If it
145      * represents a {@linkplain ExecutableElement#getSimpleName static
146      * initializer}, the name "{@code <clinit>}" is returned.
147      *
148      * If it represents an {@linkplain TypeElement#getSimpleName
149      * anonymous class} or {@linkplain ExecutableElement#getSimpleName
150      * instance initializer}, an <a href=Name.html#empty_name>empty
151      * name</a> is returned.
152      *
153      * @return the simple name of this element
154      * @see PackageElement#getSimpleName
155      * @see ExecutableElement#getSimpleName
156      * @see TypeElement#getSimpleName
157      * @see VariableElement#getSimpleName
158      * @see ModuleElement#getSimpleName
159      * @see RecordComponentElement#getSimpleName
160      * @revised 9
161      * @spec JPMS
162      */
163     Name getSimpleName();
164 
165     /**
166      * Returns the innermost element
167      * within which this element is, loosely speaking, enclosed.
168      * <ul>
169      * <li> If this element is one whose declaration is lexically enclosed
170      * immediately within the declaration of another element, that other
171      * element is returned.
172      *
173      * <li> If this is a {@linkplain TypeElement#getEnclosingElement
174      * top-level type}, its package is returned.
175      *
176      * <li> If this is a {@linkplain
177      * PackageElement#getEnclosingElement package}, its module is
178      * returned if such a module exists. Otherwise, {@code null} is returned.
179      *
180      * <li> If this is a {@linkplain
181      * TypeParameterElement#getEnclosingElement type parameter},
182      * {@linkplain TypeParameterElement#getGenericElement the
183      * generic element} of the type parameter is returned.
184      *
185      * <li> If this is a {@linkplain
186      * VariableElement#getEnclosingElement method or constructor
187      * parameter}, {@linkplain ExecutableElement the executable
188      * element} which declares the parameter is returned.
189      *
190      * <li> If this is a {@linkplain
191      * RecordComponentElement#getEnclosingElement record component},
192      * {@linkplain TypeElement the type} which declares the
193      * record component is returned.
194      *
195      * <li> If this is a {@linkplain ModuleElement#getEnclosingElement
196      * module}, {@code null} is returned.
197      *
198      * </ul>
199      *
200      * @return the enclosing element, or {@code null} if there is none
201      * @see Elements#getPackageOf
202      * @revised 9
203      * @spec JPMS
204      */
205     Element getEnclosingElement();
206 
207     /**
208      * Returns the elements that are, loosely speaking, directly
209      * enclosed by this element.
210      *
211      * A {@linkplain TypeElement#getEnclosedElements class or
212      * interface} is considered to enclose the fields, methods,
213      * constructors, record components, and member types that it directly declares.
214      *
215      * A {@linkplain PackageElement#getEnclosedElements package}
216      * encloses the top-level classes and interfaces within it, but is
217      * not considered to enclose subpackages.
218      *
219      * A {@linkplain ModuleElement#getEnclosedElements module}
220      * encloses packages within it.
221      *
222      * Enclosed elements may include implicitly declared {@linkplain
223      * Elements.Origin#MANDATED mandated} elements.
224      *
225      * Other kinds of elements are not currently considered to enclose
226      * any elements; however, that may change as this API or the
227      * programming language evolves.
228      *
229      * @apiNote Elements of certain kinds can be isolated using
230      * methods in {@link ElementFilter}.
231      *
232      * @return the enclosed elements, or an empty list if none
233      * @see TypeElement#getEnclosedElements
234      * @see PackageElement#getEnclosedElements
235      * @see ModuleElement#getEnclosedElements
236      * @see Elements#getAllMembers
237      * @jls 8.8.9 Default Constructor
238      * @jls 8.9 Enum Types
239      * @revised 9
240      * @spec JPMS
241      */
242     List<? extends Element> getEnclosedElements();
243 
244     /**
245      * Returns {@code true} if the argument represents the same
246      * element as {@code this}, or {@code false} otherwise.
247      *
248      * @apiNote The identity of an element involves implicit state
249      * not directly accessible from the element's methods, including
250      * state about the presence of unrelated types.  Element objects
251      * created by different implementations of these interfaces should
252      * <i>not</i> be expected to be equal even if &quot;the same&quot;
253      * element is being modeled; this is analogous to the inequality
254      * of {@code Class} objects for the same class file loaded through
255      * different class loaders.
256      *
257      * @param obj  the object to be compared with this element
258      * @return {@code true} if the specified object represents the same
259      *          element as this
260      */
261     @Override
262     boolean equals(Object obj);
263 
264     /**
265      * Obeys the general contract of {@link Object#hashCode Object.hashCode}.
266      *
267      * @see #equals
268      */
269     @Override
270     int hashCode();
271 
272     /**
273      * {@inheritDoc}
274      *
275      * <p>To get inherited annotations as well, use {@link
276      * Elements#getAllAnnotationMirrors(Element)
277      * getAllAnnotationMirrors}.
278      *
279      * <p>Note that any annotations returned by this method are
280      * declaration annotations.
281      *
282      * @since 1.6
283      */
284     @Override
285     List<? extends AnnotationMirror> getAnnotationMirrors();
286 
287     /**
288      * {@inheritDoc}
289      *
290      * <p>Note that any annotation returned by this method is a
291      * declaration annotation.
292      *
293      * @since 1.6
294      */
295     @Override
296     <A extends Annotation> A getAnnotation(Class<A> annotationType);
297 
298     /**
299      * {@inheritDoc}
300      *
301      * <p>Note that any annotations returned by this method are
302      * declaration annotations.
303      *
304      * @since 8
305      */
306     @Override
307     <A extends Annotation> A[] getAnnotationsByType(Class<A> annotationType);
308 
309     /**
310      * Applies a visitor to this element.
311      *
312      * @param <R> the return type of the visitor's methods
313      * @param <P> the type of the additional parameter to the visitor's methods
314      * @param v   the visitor operating on this element
315      * @param p   additional parameter to the visitor
316      * @return a visitor-specified result
317      */
318     <R, P> R accept(ElementVisitor<R, P> v, P p);
319 }
