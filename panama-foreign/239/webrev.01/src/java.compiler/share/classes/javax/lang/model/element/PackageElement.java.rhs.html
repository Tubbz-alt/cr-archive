 26 package javax.lang.model.element;
 27 
 28 import java.util.List;
 29 import javax.lang.model.type.TypeMirror;
 30 
 31 /**
 32  * Represents a package program element.  Provides access to information
 33  * about the package and its members.
 34  *
 35  * @author Joseph D. Darcy
 36  * @author Scott Seligman
 37  * @author Peter von der Ah&eacute;
 38  * @see javax.lang.model.util.Elements#getPackageOf
 39  * @since 1.6
 40  */
 41 public interface PackageElement extends Element, QualifiedNameable {
 42     /**
 43      * Returns a {@linkplain javax.lang.model.type.NoType pseudo-type}
 44      * for this package.
 45      * @return a pseudo-type for this package
 46      *
 47      * @see javax.lang.model.type.NoType
 48      * @see javax.lang.model.type.TypeKind#PACKAGE
 49      */
 50     @Override
 51     TypeMirror asType();
 52 
 53     /**
 54      * Returns the fully qualified name of this package.  This is also
 55      * known as the package's <i>canonical</i> name.  For an
 56      * {@linkplain #isUnnamed() unnamed package}, an <a
 57      * href=Name.html#empty_name>empty name</a> is returned.
 58      *
 59      * @apiNote The fully qualified name of a named package that is
 60      * not a subpackage of a named package is its simple name. The
 61      * fully qualified name of a named package that is a subpackage of
 62      * another named package consists of the fully qualified name of
 63      * the containing package, followed by "{@code .}", followed by the simple
 64      * (member) name of the subpackage.
 65      *
 66      * @return the fully qualified name of this package, or an
 67      * empty name if this is an unnamed package
 68      * @jls 6.7 Fully Qualified Names and Canonical Names
 69      */
 70     Name getQualifiedName();
 71 
 72     /**
 73      * Returns the simple name of this package.  For an {@linkplain
 74      * #isUnnamed() unnamed package}, an <a
 75      * href=Name.html#empty_name>empty name</a> is returned.
 76      *
 77      * @return the simple name of this package or an empty name if
 78      * this is an unnamed package
 79      */
 80     @Override
 81     Name getSimpleName();
 82 
 83     /**
 84      * Returns the {@linkplain NestingKind#TOP_LEVEL top-level}
 85      * classes and interfaces within this package.  Note that
 86      * subpackages are <em>not</em> considered to be enclosed by a
 87      * package.
 88      *
 89      * @return the top-level classes and interfaces within this
 90      * package
 91      */
 92     @Override
 93     List<? extends Element> getEnclosedElements();
 94 
 95     /**
 96      * Returns {@code true} if this is an unnamed package and {@code
 97      * false} otherwise.
 98      *
 99      * @return {@code true} if this is an unnamed package and {@code
100      * false} otherwise
101      * @jls 7.4.2 Unnamed Packages
102      */
103     boolean isUnnamed();
104 
105     /**
106      * Returns the enclosing module if such a module exists; otherwise
107      * returns {@code null}.
108      *
109      * One situation where a module does not exist for a package is if
110      * the environment does not include modules, such as an annotation
111      * processing environment configured for a {@linkplain
112      * javax.annotation.processing.ProcessingEnvironment#getSourceVersion
113      * source version} without modules.
114      *
115      * @return the enclosing module or {@code null} if no such module exists
116      *
117      * @revised 9
118      * @spec JPMS
119      */
120     @Override
121     Element getEnclosingElement();
122 }
