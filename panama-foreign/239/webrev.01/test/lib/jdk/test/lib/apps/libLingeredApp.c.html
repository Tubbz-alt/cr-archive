26 // Borrowed from hotspot vmError.cpp.
27 // Returns an address which is guaranteed to generate a SIGSEGV on read,
28 // which is not NULL and contains bits in every word
29 void* get_segfault_address() {
30   return (void*)
31 #ifdef _LP64
32     0xABC0000000000ABCULL;
33 #else
34     0x00000ABC;
35 #endif
36 }
37 
38 JNIEXPORT jint JNICALL
39 Java_jdk_test_lib_apps_LingeredApp_crash(JNIEnv *env, jclass clss)
40 {
41     return *(jint *)get_segfault_address();
42 }
