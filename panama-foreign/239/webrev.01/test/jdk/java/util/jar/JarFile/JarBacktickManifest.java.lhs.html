24 /**
25  * @test
26  * @bug 8186334
27  * @library /lib/testlibrary/java/util/jar
28  * @build JarBuilder

29  * @run testng JarBacktickManifest
30  * @summary Make sure scanning manifest doesn't throw AIOOBE on certain strings
31  *          containing backticks.
32  */
33 
34 import java.io.File;
35 import java.io.IOException;
36 import java.nio.file.Files;
37 import java.util.jar.JarFile;
38 
39 import org.testng.Assert;
40 import org.testng.annotations.AfterClass;
41 import org.testng.annotations.BeforeClass;
42 import org.testng.annotations.Test;
43 


44 public class JarBacktickManifest {
45 
46     public static final String VERIFY_MANIFEST_JAR = "verifyManifest.jar";
47 
48     @BeforeClass
49     public void initialize() throws Exception {
50         JarBuilder jb = new JarBuilder(VERIFY_MANIFEST_JAR);
51         jb.addAttribute("Test", " Class-`Path` ");
52         jb.addAttribute("Test2", " Multi-`Release ");
53         jb.build();
54     }
55 
56     @Test
57     public void test() throws Exception {
58         try (JarFile jf = new JarFile(VERIFY_MANIFEST_JAR)) {  // do not set runtime versioning
59             Assert.assertFalse(jf.isMultiRelease(), "Shouldn't be multi-release");
60         }
61     }
62 
63     @AfterClass
64     public void close() throws IOException {
65         Files.delete(new File(VERIFY_MANIFEST_JAR).toPath());
66     }
67 }
