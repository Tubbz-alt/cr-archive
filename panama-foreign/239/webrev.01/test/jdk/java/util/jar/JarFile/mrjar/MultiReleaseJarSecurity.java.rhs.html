 26  * @bug 8132734 8144062
 27  * @summary Test potential security related issues
 28  * @library /lib/testlibrary/java/util/jar /test/lib/
 29  * @build CreateMultiReleaseTestJars
 30  *        jdk.test.lib.compiler.Compiler
 31  *        jdk.test.lib.util.JarBuilder
 32  * @run testng MultiReleaseJarSecurity

 34 
 35 import java.io.File;
 36 import java.io.IOException;
 37 import java.io.InputStream;
 38 import java.nio.file.Files;
 39 import java.security.CodeSigner;
 40 import java.security.cert.Certificate;
 41 import java.util.Arrays;
 42 import java.util.jar.JarEntry;
 43 import java.util.jar.JarFile;
 44 import java.util.zip.ZipFile;
 45 
 46 import org.testng.Assert;
 47 import org.testng.annotations.AfterClass;
 48 import org.testng.annotations.BeforeClass;
 49 import org.testng.annotations.Test;
 50 
 51 public class MultiReleaseJarSecurity {
 52 
 53     static final int MAJOR_VERSION = Runtime.version().major();
 54 
 55     String userdir = System.getProperty("user.dir",".");
 56     File multirelease = new File(userdir, "multi-release.jar");
 57     File signedmultirelease = new File(userdir, "signed-multi-release.jar");
 58 
 59     @BeforeClass
 60     public void initialize() throws Exception {
 61         CreateMultiReleaseTestJars creator =  new CreateMultiReleaseTestJars();
 62         creator.compileEntries();
 63         creator.buildMultiReleaseJar();
 64         creator.buildSignedMultiReleaseJar();
 65     }
 66 
 67     @AfterClass
 68     public void close() throws IOException {
 69         Files.delete(multirelease.toPath());
 70         Files.delete(signedmultirelease.toPath());
 71     }
 72 
 73     @Test
 74     public void testCertsAndSigners() throws IOException {
 75         try (JarFile jf = new JarFile(signedmultirelease, true, ZipFile.OPEN_READ, Runtime.version())) {
 76             CertsAndSigners vcas = new CertsAndSigners(jf, jf.getJarEntry("version/Version.class"));
 77             CertsAndSigners rcas = new CertsAndSigners(jf, jf.getJarEntry("META-INF/versions/" + MAJOR_VERSION + "/version/Version.class"));
 78             Assert.assertTrue(Arrays.equals(rcas.getCertificates(), vcas.getCertificates()));
 79             Assert.assertTrue(Arrays.equals(rcas.getCodeSigners(), vcas.getCodeSigners()));
 80         }
 81     }
 82 
 83     private static class CertsAndSigners {
 84         final private JarFile jf;
 85         final private JarEntry je;
 86         private boolean readComplete;
 87 
 88         CertsAndSigners(JarFile jf, JarEntry je) {
 89             this.jf = jf;
 90             this.je = je;
 91         }
 92 
 93         Certificate[] getCertificates() throws IOException {
 94             readEntry();
 95             return je.getCertificates();
 96         }
 97 
 98         CodeSigner[] getCodeSigners() throws IOException {
 99             readEntry();
100             return je.getCodeSigners();
101         }
102 
103         private void readEntry() throws IOException {
104             if (!readComplete) {
105                 try (InputStream is = jf.getInputStream(je)) {
106                     is.readAllBytes();
107                 }
108                 readComplete = true;
109             }
110         }
111     }
112 }
