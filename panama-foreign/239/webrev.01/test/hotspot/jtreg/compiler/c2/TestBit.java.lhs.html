 24 package compiler.c2;
 25 
 26 import jdk.test.lib.process.OutputAnalyzer;
 27 import jdk.test.lib.process.ProcessTools;
 28 
 29 /*
 30  * @test
 31  * @bug 8247408
 32  * @summary C2 should convert ((var&16) == 16) to ((var&16) != 0) for power-of-two constants
 33  * @library /test/lib /
 34  *
 35  * @run main/othervm compiler.c2.TestBit
 36  *
 37  * @requires os.arch=="aarch64" | os.arch=="amd64" | os.arch == "ppc64le"
 38  * @requires vm.debug == true & vm.flavor == "server" & !vm.graal.enabled
 39  */
 40 public class TestBit {
 41 
 42     static void runTest(String testName) throws Exception {
 43         String className = "compiler.c2.TestBit";
 44         String[] procArgs = {


 45             "-XX:+PrintOptoAssembly",
 46             "-XX:CompileCommand=compileonly," + className + "::tst*",
 47             className, testName};
 48 
 49         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(procArgs);
 50         OutputAnalyzer output = new OutputAnalyzer(pb.start());
 51 
 52         String expectedTestBitInstruction =
 53             "ppc64le".equals(System.getProperty("os.arch")) ? "ANDI" :
 54             "aarch64".equals(System.getProperty("os.arch")) ? "tb"   :
 55             "amd64".equals(System.getProperty("os.arch"))   ? "test" : null;
 56 
 57         if (expectedTestBitInstruction != null) {
 58             output.shouldContain(expectedTestBitInstruction);
 59         } else {
 60             System.err.println("unexpected os.arch");
 61         }
 62     }
 63 
 64     static final int ITER = 100000; // ~ Tier4CompileThreshold + compilation time
 65 
 66     // dummy volatile variable
 67     public static volatile long c = 0;
 68 
 69     // C2 is expected to generate test bit instruction on the test
 70     static void tstBitLong(long value) {
 71         if (1L == (1L & value)) {
 72             c++;
 73         } else {
 74             c--;
 75         }
 76     }
 77 
 78     // C2 is expected to generate test bit instruction on the test
 79     static void tstBitInt(int value) {
 80         if (1 == (1 & value)) {
 81             c++;
 82         } else {
 83             c--;
 84         }
 85     }
 86 
 87     public static void main(String[] args) throws Exception {
 88         if (args.length == 0) {
 89             // Fork VMs to check their debug compiler output
 90             runTest("tstBitLong");
 91             runTest("tstBitInt");
 92         }
 93         if (args.length > 0) {
 94             // We are in a forked VM to execute the named test
 95             String testName = args[0];
 96             switch (testName) {
 97             case "tstBitLong":
 98                 for (int i = 0; i < ITER; i++) {
 99                     tstBitLong(i % 2);
100                 }
101                 break;
102             case "tstBitInt":
103                 for (int i = 0; i < ITER; i++) {
104                     tstBitInt(i % 2);
105                 }
106                 break;
107             default:
108                 throw new RuntimeException("unexpected test name " + testName);
109             }
110         }
111     }
112 }
