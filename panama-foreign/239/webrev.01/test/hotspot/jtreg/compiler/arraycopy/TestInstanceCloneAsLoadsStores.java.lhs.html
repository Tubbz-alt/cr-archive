 43 package compiler.arraycopy;
 44 
 45 public class TestInstanceCloneAsLoadsStores extends TestInstanceCloneUtils {
 46 
 47     // Should be compiled as loads/stores
 48     static Object m1(D src) throws CloneNotSupportedException {
 49         return src.clone();
 50     }
 51 
 52     // Should be compiled as adds of src (dest allocation eliminated)
 53     static int m2(D src) throws CloneNotSupportedException {
 54         D dest = (D)src.clone();
 55         return dest.i1 + dest.i2 + ((int)dest.i3) + dest.i4 + dest.i5;
 56     }
 57 
 58     // Should be compiled as arraycopy stub call (object too large)
 59     static int m3(E src) throws CloneNotSupportedException {
 60         E dest = (E)src.clone();
 61         return dest.i1 + dest.i2 + dest.i3 + dest.i4 + dest.i5 +
 62             dest.i6 + dest.i7 + dest.i8 + dest.i9;
 63     }
 64 
 65     // Need profiling on src's type to be able to know number of
 66     // fields. Cannot clone as loads/stores if compile doesn't use it.
 67     static Object m4(A src) throws CloneNotSupportedException {
 68         return src.clone();
 69     }
 70 
 71     // Same as above but should optimize out dest allocation
 72     static int m5(A src) throws CloneNotSupportedException {
 73         A dest = (A)src.clone();
 74         return dest.i1 + dest.i2 + dest.i3 + dest.i4 + dest.i5;
 75     }
 76 
 77     // Check that if we have no fields to clone we do fine
 78     static Object m6(F src) throws CloneNotSupportedException {
 79         return src.clone();
 80     }
 81 
 82     // With virtual call to clone: clone inlined from profling which
 83     // gives us exact type of src so we can clone it with
 84     // loads/stores.
 85     static G m7(G src) throws CloneNotSupportedException {
 86         return (G)src.myclone();
 87     }
 88 
 89     // Virtual call to clone but single target: exact type unknown,
 90     // clone intrinsic uses profiling to determine exact type and
 91     // clone with loads/stores.
 92     static J m8(J src) throws CloneNotSupportedException {
 93         return (J)src.myclone();
 94     }
 95 
 96     public static void main(String[] args) throws Exception {
 97 
 98         TestInstanceCloneAsLoadsStores test = new TestInstanceCloneAsLoadsStores();
 99 
100         test.doTest(d, "m1");
101         test.doTest(d, "m2");
102         test.doTest(e, "m3");
103         test.doTest(a, "m4");
104         test.doTest(a, "m5");
105         test.doTest(f, "m6");
106         test.doTest(g, "m7");
107         test.doTest(k, "m8");
108 
109         if (!test.success) {
110             throw new RuntimeException("some tests failed");
111         }
112 
113     }
114 }
