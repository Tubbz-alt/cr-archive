 47 package compiler.arraycopy;
 48 
 49 public class TestInstanceCloneAsLoadsStores extends TestInstanceCloneUtils {
 50 
 51     // Should be compiled as loads/stores
 52     static Object m1(D src) throws CloneNotSupportedException {
 53         return src.clone();
 54     }
 55 
 56     // Should be compiled as adds of src (dest allocation eliminated)
 57     static int m2(D src) throws CloneNotSupportedException {
 58         D dest = (D)src.clone();
 59         return dest.i1 + dest.i2 + ((int)dest.i3) + dest.i4 + dest.i5;
 60     }
 61 
 62     // Should be compiled as arraycopy stub call (object too large)
 63     static int m3(E src) throws CloneNotSupportedException {
 64         E dest = (E)src.clone();
 65         return dest.i1 + dest.i2 + dest.i3 + dest.i4 + dest.i5 +
 66             dest.i6 + dest.i7 + dest.i8 + dest.i9;
 67     }
 68 
 69     // Need profiling on src's type to be able to know number of
 70     // fields. Cannot clone as loads/stores if compile doesn't use it.
 71     static Object m4(A src) throws CloneNotSupportedException {
 72         return src.clone();
 73     }
 74 
 75     // Same as above but should optimize out dest allocation
 76     static int m5(A src) throws CloneNotSupportedException {
 77         A dest = (A)src.clone();
 78         return dest.i1 + dest.i2 + dest.i3 + dest.i4 + dest.i5;
 79     }
 80 
 81     // Check that if we have no fields to clone we do fine
 82     static Object m6(F src) throws CloneNotSupportedException {
 83         return src.clone();
 84     }
 85 
 86     // With virtual call to clone: clone inlined from profling which
 87     // gives us exact type of src so we can clone it with
 88     // loads/stores.
 89     static G m7(G src) throws CloneNotSupportedException {
 90         return (G)src.myclone();
 91     }
 92 
 93     // Virtual call to clone but single target: exact type unknown,
 94     // clone intrinsic uses profiling to determine exact type and
 95     // clone with loads/stores.
 96     static J m8(J src) throws CloneNotSupportedException {
 97         return (J)src.myclone();
 98     }
 99 
100     public static void main(String[] args) throws Exception {
101 
102         TestInstanceCloneAsLoadsStores test = new TestInstanceCloneAsLoadsStores();
103 
104         test.doTest(d, "m1");
105         test.doTest(d, "m2");
106         test.doTest(e, "m3");
107         test.doTest(a, "m4");
108         test.doTest(a, "m5");
109         test.doTest(f, "m6");
110         test.doTest(g, "m7");
111         test.doTest(k, "m8");
112 
113         if (!test.success) {
114             throw new RuntimeException("some tests failed");
115         }
116 
117     }
118 }
