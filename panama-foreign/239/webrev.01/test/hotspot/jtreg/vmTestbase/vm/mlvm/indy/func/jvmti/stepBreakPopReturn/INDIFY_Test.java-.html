 53 package vm.mlvm.indy.func.jvmti.stepBreakPopReturn;
 54 
 55 import java.lang.invoke.CallSite;
 56 import java.lang.invoke.ConstantCallSite;
 57 import java.lang.invoke.MethodHandle;
 58 import java.lang.invoke.MethodHandles;
 59 import java.lang.invoke.MethodHandles.Lookup;
 60 import java.lang.invoke.MethodType;
 61 
 62 import vm.mlvm.share.MlvmTest;
 63 
 64 public class INDIFY_Test extends MlvmTest {
 65 
 66     public static native void setDebuggeeMethodName(String name);
 67     public static native void setDebuggeeClassName(String name);
 68     public static native boolean checkStatus();
 69 
 70     static {
 71         System.loadLibrary("stepBreakPopReturn");
 72     }
 73 
 74     private static MethodType MT_bootstrap() {
 75         return MethodType.methodType(CallSite.class, Lookup.class, String.class, MethodType.class);
 76     }
 77 
 78     private static MethodHandle MH_bootstrap() throws NoSuchMethodException, IllegalAccessException {
 79         return MethodHandles.lookup().findStatic(
 80                 INDIFY_Test.class,
 81                 "bootstrap",
 82                 MT_bootstrap());
 83     }
 84 
 85     public static CallSite bootstrap(Lookup c, String name, MethodType mt) throws Throwable {
 86         int i = 0; // For single step
 87         getLog().trace(i, "Lookup " + c + "; method name = " + name + "; method type = " + mt);
 88         CallSite cs = new ConstantCallSite(MethodHandles.lookup().findStatic(
 89                 INDIFY_Test.class,
 90                 "target",
 91                 MethodType.methodType(int.class, Object.class, String.class, int.class)));
 92         return cs;
 93     }
 94 
 95     public static int target(Object o, String s, int i) {
 96         int j = 0; // For single step event
 97         getLog().trace(0, "Target called! Object = " + o + "; string = " + s + "; int = " + i);
 98         return i;
 99     }
100 
101     private static MethodHandle INDY_call1;
102     private static MethodHandle INDY_call1() throws Throwable {
103         if (INDY_call1 != null)
104             return INDY_call1;
105 
106         CallSite cs = (CallSite) MH_bootstrap().invokeWithArguments(
107                 MethodHandles.lookup(),
108                 "greet",
109                 MethodType.methodType(int.class, Object.class, String.class, int.class));
110 
111         return cs.dynamicInvoker();
112     }
113 
114     private static MethodHandle INDY_call2;
115     private static MethodHandle INDY_call2() throws Throwable {
116         if (INDY_call2 != null)
117             return INDY_call2;
118 
119         CallSite cs = (CallSite) MH_bootstrap().invokeWithArguments(
120                 MethodHandles.lookup(),
121                 "greet",
122                 MethodType.methodType(int.class, Object.class, String.class, int.class));
123 
124         return cs.dynamicInvoker();
125     }
126 
127     public boolean run() throws Throwable {
128         setDebuggeeClassName("L" + INDIFY_Test.class.getName().replace('.', '/') + ";");
129         setDebuggeeMethodName("target");
130 
131         Object o = new Object();
132         String s = "heaven";
133         int i = 789;
134 
135         // When agent is loaded and correctly working, every call to target() should return 0
136         // (without agent target() returns 3rd argument)
137         // so the sum variable should be 0 at every moment
138         getLog().trace(0, "Call site 1");
139         int sum = (int) INDY_call1().invokeExact(o, s, i);
140         for ( i = 0; i < 5; i++ ) {
141             getLog().trace(0, "Call site 2, sum=" + sum);
142             sum += (int) INDY_call2().invokeExact(o, s, i + 789);
143         }
144         getLog().trace(0, "Direct call, sum=" + sum);
145         sum += target(new Object(), "hell", 123);
146 
147         getLog().trace(0, "Done, sum=" + sum + " (should be 0)");
148 
149         return checkStatus() && sum == 0;
150     }
151 
152     public static void main(String[] args) { MlvmTest.launch(args); }
153 }
