47 package vm.mlvm.anonloader.func.isGarbageCollected;
48 
49 import java.lang.ref.PhantomReference;
50 import java.lang.ref.Reference;
51 import java.lang.ref.ReferenceQueue;
52 
53 import vm.mlvm.anonloader.share.AnonkTestee01;
54 import vm.mlvm.share.MlvmTest;
55 import vm.share.FileUtils;
56 import vm.share.UnsafeAccess;
57 
58 public class Test extends MlvmTest {
59     private static final Class<?> PARENT = AnonkTestee01.class;
60 
61     public boolean run() throws Exception {
62         ReferenceQueue<Class<?>> refQueue = new ReferenceQueue<Class<?>>();
63         PhantomReference<Class<?>> anonClassRef = createClass(refQueue);
64         System.gc();
65         Reference<? extends Class<?>> deletedObject = refQueue.remove();
66         return anonClassRef.equals(deletedObject);
67     }
68 
69     // a private method is great to keep anonClass reference local to make it GCed on the next cycle
70     private PhantomReference<Class<?>> createClass(ReferenceQueue<Class<?>> refQueue) throws Exception {
71         byte[] classBytes = FileUtils.readClass(PARENT.getName());
72         Class<?> anonClass = UnsafeAccess.unsafe.defineAnonymousClass(PARENT,
73                 classBytes, null);
74         return new PhantomReference<Class<?>>(anonClass, refQueue);
75     }
76 
77     public static void main(String[] args) { MlvmTest.launch(args); }
78 }
