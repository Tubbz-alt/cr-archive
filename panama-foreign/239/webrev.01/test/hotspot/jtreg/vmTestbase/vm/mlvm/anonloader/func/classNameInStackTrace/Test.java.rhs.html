48 package vm.mlvm.anonloader.func.classNameInStackTrace;
49 
50 import java.io.ByteArrayOutputStream;
51 import java.io.PrintStream;
52 
53 import vm.mlvm.anonloader.share.AnonkTestee01;
54 import vm.mlvm.share.MlvmTest;
55 import vm.share.FileUtils;
56 import vm.share.UnsafeAccess;
57 
58 public class Test extends MlvmTest {
59     private static final Class<?> PARENT = AnonkTestee01.class;
60 
61     public boolean run() throws Exception {
62         byte[] classBytes = FileUtils.readClass(PARENT.getName());
63         Class<?> cls = UnsafeAccess.unsafe.defineAnonymousClass(PARENT,
64                 classBytes, null);
65         try {
66             // hashCode() in AnonkTestee01 always throws an exception
67             cls.newInstance().hashCode();
68             return false;
69 
70         } catch ( RuntimeException e ) {
71             ByteArrayOutputStream byteOS = new ByteArrayOutputStream();
72             PrintStream printStream = new PrintStream(byteOS);
73             e.printStackTrace(printStream);
74             printStream.close();
75             String stackTrace = byteOS.toString("ASCII");
76             getLog().display("Caught exception stack trace: " + stackTrace);
77             return stackTrace.contains(Test.class.getName());
78         }
79     }
80 
81     public static void main(String[] args) { MlvmTest.launch(args); }
82 }
