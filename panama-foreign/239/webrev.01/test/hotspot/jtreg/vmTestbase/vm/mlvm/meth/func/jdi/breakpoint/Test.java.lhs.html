73             BreakpointInfo info = new BreakpointInfo("invokeMH");

75             result.add(info);
76         }
77         {
78             BreakpointInfo info = new BreakpointInfo("invokePlain");
79             info.stepsToTrace = 100;
80             result.add(info);

82         result.add(new BreakpointInfo("mhTarget"));
83         result.add(new BreakpointInfo("plainTarget"));
84         result.add(new BreakpointInfo("stop"));
85 
86         return result;
87     }
