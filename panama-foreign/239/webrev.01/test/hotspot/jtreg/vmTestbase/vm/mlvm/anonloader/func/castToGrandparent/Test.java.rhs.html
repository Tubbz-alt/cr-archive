46 package vm.mlvm.anonloader.func.castToGrandparent;
47 
48 import vm.mlvm.anonloader.share.AnonkTestee01;
49 import vm.mlvm.anonloader.share.AnonkTestee02;
50 import vm.mlvm.share.Env;
51 import vm.mlvm.share.MlvmTest;
52 import vm.share.FileUtils;
53 import vm.share.UnsafeAccess;
54 
55 public class Test extends MlvmTest {
56     private static final Class<?> PARENT = AnonkTestee02.class;
57 
58     @Override
59     public boolean run() throws Exception {
60         byte[] classBytes = FileUtils.readClass(PARENT.getName());
61         Class<?> cls = UnsafeAccess.unsafe.defineAnonymousClass(
62                 PARENT, classBytes, null);
63         Object anonObject = cls.newInstance();
64         // Try to cast anonymous class to its grandparent
65         AnonkTestee01 anonCastToParent = (AnonkTestee01) anonObject;
66         if ( anonCastToParent.equals(anonObject) )
67             Env.traceNormal("Anonymous object can be cast to original one");
68 
69         // Try to cast using another method
70         new AnonkTestee01().getClass().cast(anonObject);
71 
72         Env.traceNormal("Anonymous class can be cast to original one");
73 
74         return true;
75     }
76 
77     public static void main(String[] args) { MlvmTest.launch(args); }
78 }
