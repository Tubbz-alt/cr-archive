48 package vm.mlvm.anonloader.func.isGarbageCollected;
49 
50 import java.lang.ref.PhantomReference;
51 import java.lang.ref.Reference;
52 import java.lang.ref.ReferenceQueue;
53 
54 import vm.mlvm.anonloader.share.AnonkTestee01;
55 import vm.mlvm.share.MlvmTest;
56 import vm.share.FileUtils;
57 import vm.share.UnsafeAccess;
58 
59 public class Test extends MlvmTest {
60     private static final Class<?> PARENT = AnonkTestee01.class;
61 
62     public boolean run() throws Exception {
63         ReferenceQueue<Class<?>> refQueue = new ReferenceQueue<Class<?>>();
64         PhantomReference<Class<?>> anonClassRef = createClass(refQueue);
65         System.gc();
66         Reference<? extends Class<?>> deletedObject = refQueue.remove();
67         return anonClassRef.equals(deletedObject);
68     }
69 
70     // a private method is great to keep anonClass reference local to make it GCed on the next cycle
71     private PhantomReference<Class<?>> createClass(ReferenceQueue<Class<?>> refQueue) throws Exception {
72         byte[] classBytes = FileUtils.readClass(PARENT.getName());
73         Class<?> anonClass = UnsafeAccess.unsafe.defineAnonymousClass(PARENT,
74                 classBytes, null);
75         return new PhantomReference<Class<?>>(anonClass, refQueue);
76     }
77 
78     public static void main(String[] args) { MlvmTest.launch(args); }
79 }
