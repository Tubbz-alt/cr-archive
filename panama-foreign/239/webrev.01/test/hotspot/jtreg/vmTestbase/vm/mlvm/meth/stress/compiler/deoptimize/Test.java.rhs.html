 76 package vm.mlvm.meth.stress.compiler.deoptimize;
 77 
 78 import java.lang.invoke.MethodHandle;
 79 import java.lang.invoke.MethodHandles;
 80 import java.lang.invoke.MethodType;
 81 
 82 import vm.mlvm.meth.share.Argument;
 83 import vm.mlvm.meth.share.MHTransformationGen;
 84 import vm.mlvm.meth.share.RandomArgumentsGen;
 85 import vm.mlvm.share.Env;
 86 import vm.mlvm.share.MlvmTest;
 87 import vm.mlvm.share.MultiThreadedTest;
 88 
 89 // TODO: check deopt using vm.mlvm.share.comp framework
 90 public class Test extends MultiThreadedTest {
 91 
 92     static class A {
 93         Object m() {
 94             return new Integer(0);
 95         }
 96     }
 97 
 98     static class B extends A {
 99         @Override
100         Object m() {
101             return new Integer(1);
102         }
103     }
104 
105     static class TestData {
106         final A _a;
107         final int _expectedResult;
108 
109         TestData(A a) {
110             _a = a;
111             _expectedResult = (Integer) a.m();
112         }
113 
114         Object m() {
115             return _a.m();
116         }
117     }
118 
119     volatile TestData _data = new TestData(new A());
120 
121     volatile boolean _testDone = false;
122 
123     final MethodHandle _mh;
124     final Argument[] _finalArgs;
125 
126     public Test() throws Throwable {
127         _mh = MethodHandles.lookup().findVirtual(TestData.class, "m", MethodType.methodType(Object.class));
128         _finalArgs = RandomArgumentsGen.createRandomArgs(true, _mh.type());
129     }
130 
131     @Override
132     protected void initializeTest() throws Throwable {
133         super.initializeTest();
134         if (calcThreadNum() < 2) {
135             throw new IllegalArgumentException("Number of threads should be 2 or more");
136         }
137     }
138 
139     private void sleepAndDeoptimize() throws Throwable {
140         try {
141             Thread.sleep(3000);
142             // Force deoptimization in uncommon trap logic
143             Env.traceNormal("Deoptimizing");
144             _data = new TestData((A) Test.class.getClassLoader().loadClass(Test.class.getName() + "$B").newInstance());
145             Thread.sleep(3000);
146         } finally {
147             this._testDone = true;
148         }
149     }
150 
151     @Override
152     protected boolean runThread(int threadNum) throws Throwable {
153         if ( threadNum == 0 ) {
154             sleepAndDeoptimize();
155             return true;
156         }
157 
158         while ( ! Test.this._testDone) {
159             TestData dataSnapshot = _data;
160             Integer expectedResult = (Integer) dataSnapshot._a.m();
161             Argument retVal = new Argument(Object.class, expectedResult);
162             retVal.setPreserved(true);
163 
164             MHTransformationGen.createAndCallSequence(retVal, dataSnapshot, _mh, _finalArgs, true);
165         }
166 
167         return true;
168     }
169 
170     public static void main(String[] args) { MlvmTest.launch(args); }
171 }
