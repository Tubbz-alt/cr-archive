50 package vm.mlvm.anonloader.func.castToParent;
51 
52 import vm.mlvm.anonloader.share.AnonkTestee01;
53 import vm.mlvm.share.Env;
54 import vm.mlvm.share.MlvmTest;
55 import vm.share.FileUtils;
56 import vm.share.UnsafeAccess;
57 
58 public class Test extends MlvmTest {
59     private static final Class<?> PARENT = AnonkTestee01.class;
60 
61     public boolean run() throws Exception {
62         byte[] classBytes = FileUtils.readClass(PARENT.getName());
63         Class<?> cls = UnsafeAccess.unsafe.defineAnonymousClass(PARENT,
64                 classBytes, null);
65         Object o = cls.newInstance();
66         // Try to cast anonymous class to its parent
67         AnonkTestee01 anonCastToParent = (AnonkTestee01) o;
68         if ( anonCastToParent.equals(o) )
69             Env.traceNormal("Anonymous object can be cast to original one");
70 
71         // Cast the class
72         new AnonkTestee01().getClass().cast(o);
73 
74         Env.traceNormal("Anonymous can be cast to original class");
75 
76         return true;
77     }
78 
79     public static void main(String[] args) { MlvmTest.launch(args); }
80 }
