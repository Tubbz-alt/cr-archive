47 package vm.mlvm.anonloader.func.findByName;
48 
49 import vm.mlvm.anonloader.share.AnonkTestee01;
50 import vm.mlvm.share.MlvmTest;
51 import vm.share.FileUtils;
52 import vm.share.UnsafeAccess;
53 
54 public class Test extends MlvmTest {
55     private static final Class<?> PARENT = AnonkTestee01.class;
56 
57     public boolean run() throws Exception {
58         try {
59             byte[] classBytes = FileUtils.readClass(PARENT.getName());
60             Class<?> c = UnsafeAccess.unsafe.defineAnonymousClass(PARENT,
61                     classBytes, null);
62             getLog().display("Anonymous class name: " + c.getName());
63             Class.forName(c.getName()).newInstance();
64             return false;
65         } catch ( ClassNotFoundException e ) {
66             return true;
67         }
68     }
69 
70     public static void main(String[] args) { MlvmTest.launch(args); }
71 }
