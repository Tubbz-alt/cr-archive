43 package vm.mlvm.anonloader.stress.oome.metaspace;
44 
45 import java.util.List;
46 import java.io.IOException;
47 
48 import vm.mlvm.anonloader.share.AnonkTestee01;
49 import vm.mlvm.share.MlvmOOMTest;
50 import vm.mlvm.share.MlvmTestExecutor;
51 import vm.mlvm.share.Env;
52 import vm.share.FileUtils;
53 import vm.share.UnsafeAccess;
54 
55 /**
56  * This test loads classes using Unsafe.defineAnonymousClass and stores them,
57  * expecting Metaspace OOME.
58  *
59  */
60 public class Test extends MlvmOOMTest {
61     @Override
62     protected void checkOOME(OutOfMemoryError oome) {
63         String message = oome.getMessage();
64         if (!"Metaspace".equals(message) && !"Compressed class space".equals(message)) {
65             throw new RuntimeException("TEST FAIL : wrong OOME", oome);
66         }
67     }
68 
69     @Override
70     protected void eatMemory(List<Object> list) {
71         byte[] classBytes = null;
72         try {
73             classBytes = FileUtils.readClass(AnonkTestee01.class.getName());
74         } catch (IOException e) {
75             Env.throwAsUncheckedException(e);
76         }
77         while (true) {
78             list.add(UnsafeAccess.unsafe.defineAnonymousClass(AnonkTestee01.class,
79                     classBytes, null));
80         }
81     }
82 
83     public static void main(String[] args) {
84         MlvmTestExecutor.launch(args);
85     }
86 }
