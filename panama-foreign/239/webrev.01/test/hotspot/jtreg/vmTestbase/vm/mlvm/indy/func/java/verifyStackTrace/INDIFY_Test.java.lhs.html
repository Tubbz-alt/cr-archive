 45 package vm.mlvm.indy.func.java.verifyStackTrace;
 46 
 47 import java.lang.invoke.CallSite;
 48 import java.lang.invoke.ConstantCallSite;
 49 import java.lang.invoke.MethodHandle;
 50 import java.lang.invoke.MethodHandles;
 51 import java.lang.invoke.MethodHandles.Lookup;
 52 import java.lang.invoke.MethodType;
 53 
 54 import vm.mlvm.share.MlvmTest;
 55 
 56 public class INDIFY_Test extends MlvmTest {
 57 
 58     private static final String METHOD_NAME = "runFunky";
 59     private static final int MAX_FRAME = 10;
 60 
 61     public INDIFY_Test() {}
 62 
 63     private static MethodType MT_bootstrap() {
 64         return MethodType.methodType(CallSite.class, Lookup.class, String.class, MethodType.class);
 65     }
 66 
 67     private static MethodHandle MH_bootstrap() throws NoSuchMethodException, IllegalAccessException {
 68         return MethodHandles.lookup().findStatic(
 69                 INDIFY_Test.class,
 70                 "bootstrap",
 71                 MT_bootstrap());
 72     }
 73 
 74     public static CallSite bootstrap(Lookup c, String name, MethodType mt) throws Throwable {
 75         getLog().trace(0, "Lookup " + c + "; method name = " + name + "; method type = " + mt);
 76 
 77         boolean found = false;
 78         StackTraceElement trace[] = new Throwable().getStackTrace();
 79         for ( int i = 1; i < MAX_FRAME; i++ ) {
 80             StackTraceElement stackFrame = trace[i];
 81             getLog().trace(0, "Caller " + i + " stack frame: " + stackFrame);
 82             if ( stackFrame.getMethodName().equals(METHOD_NAME) ) {
 83                 getLog().trace(0, "Required stack frame found");
 84                 found = true;
 85                 break;
 86             }
 87         }
 88 
 89         if ( ! found )
 90             throw new RuntimeException("Can't find caller method name (" + METHOD_NAME + ") in a bootstrap method stack");
 91 
 92         return new ConstantCallSite(MethodHandles.lookup().findStatic(INDIFY_Test.class, "target", mt));
 93     }
 94 
 95     public static Throwable target(Object o, String s, int i) {
 96         getLog().trace(0, "Target called! Object = " + o + "; string = " + s + "; int = " + i);
 97         return new Throwable();
 98     }
 99 
100     private static MethodHandle INDY_call;
101     private static MethodHandle INDY_call() throws Throwable {
102         if (INDY_call != null)
103             return INDY_call;
104 
105         CallSite cs = (CallSite) MH_bootstrap().invokeWithArguments(
106                 MethodHandles.lookup(),
107                 "greet",
108                 MethodType.methodType(Throwable.class, Object.class, String.class, int.class));
109 
110         return cs.dynamicInvoker();
111     }
112 
113     public boolean runFunky() throws Throwable {
114         // Throwable t = (Throwable) InvokeDynamic.greet(new Object(), "world", 123);
115         Object o = new Object();
116         String s = "world";
117         int i = 123;
118         Throwable t = (Throwable) INDY_call().invokeExact(o, s, i);
119 
120         StackTraceElement stackFrame = t.getStackTrace()[1];
121         getLog().trace(0, "Stack trace element from target call: " + stackFrame);
122         if ( ! stackFrame.getMethodName().equals(METHOD_NAME) )
123             throw new Exception("Wrong method name in a bootstrap method: " + stackFrame);
124 
125         return true;
126     }
127 
128     public boolean run() throws Throwable { return runFunky(); }
129 
130     public static void main(String[] args) { MlvmTest.launch(args); }
131 }
