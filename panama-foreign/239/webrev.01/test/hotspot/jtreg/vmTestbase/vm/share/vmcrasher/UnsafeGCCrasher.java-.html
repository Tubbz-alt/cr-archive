23 package vm.share.vmcrasher;
24 
25 import vm.share.UnsafeAccess;
26 
27 public class UnsafeGCCrasher extends Crasher {
28 
29     private static class C {
30         C next;
31     }
32 
33     private static C a = null;
34 
35     @SuppressWarnings("restriction")
36     @Override
37     public void run() {
38         try {
39             a = new C();
40             a.next = new C();
41             a.next.next = new C();
42             a.next.next.next = new C();
43             UnsafeAccess.unsafe.putInt(a.next, UnsafeAccess.unsafe.objectFieldOffset(C.class.getDeclaredField("next")), 0xDEADCAFE);
44             System.gc();
45             Thread.sleep(1000);
46         } catch ( Throwable t ) {
47             t.printStackTrace();
48         }
49     }
50 
51 }
