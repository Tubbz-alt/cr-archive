 44 package vm.mlvm.anonloader.stress.byteMutation;
 45 
 46 import vm.mlvm.anonloader.share.AnonkTestee01;
 47 import vm.mlvm.anonloader.share.StressClassLoadingTest;
 48 import vm.share.FileUtils;
 49 import vm.share.options.Option;
 50 
 51 /**
 52  * The test does the following in a cycle:
 53  *
 54  * <ol>
 55  * <li>Takes bytes from a valid class file
 56  * <li>Sets 1 to 5 bytes in random positions to random values
 57  * <li>Tries to load such class using:
 58  *     <ul>
 59  *       <li>a custom class loader,
 60  *       <li>{@link sun.misc.Unsafe#defineAnonymousClass}
 61  *           when {@code -unsafeLoad true} option is passed to the test.
 62  *     </ul>
 63  * </ol>
 64  *
 65  * <p>In most cases the resulting class file is invalid and rejected by
 66  * the VM verifier. But this test tries to find pathological cases, such
 67  * as infinite loops during verification or VM crashes.
 68  *
 69  * <p>NB: There is a tool to load invalid classes saved by this test.
 70  * Please see tool documentation at {@link vm.mlvm.tools.LoadClass}
 71  *
 72  */
 73 public class Test extends StressClassLoadingTest {
 74     private final static Class<?> HOST_CLASS = AnonkTestee01.class;
 75     private final byte[] testeeBytes;
 76     @Option(name = "mutationCount", default_value = "3",
 77             description = "How many bytes to mutate in a class")
 78     private int mutationCount = 3;
 79 
 80     /**
 81      * Constructs the test.
 82      * @throws Exception if there are any errors when
 83      * reading {@link vm.mlvm.anonloader.share.AnonkTestee01} class bytecodes.
 84      */
 85     public Test() throws Exception {
 86         this.testeeBytes = FileUtils.readClass(AnonkTestee01.class.getName());
 87     }
 88 
 89     /**
 90      * Returns {@link vm.mlvm.anonloader.share.AnonkTestee01} class to the
 91      * parent.
 92      * @return {@link vm.mlvm.anonloader.share.AnonkTestee01} class.
 93      */
 94     @Override
 95     protected Class<?> getHostClass() {
 96         return HOST_CLASS;
 97     }
 98 
 99     /**
100      * Takes {@link vm.mlvm.anonloader.share.AnonkTestee01} class bytecodes
101      * and modifies mutationCount bytes setting them to random values.
102      * @return {@link vm.mlvm.anonloader.share.AnonkTestee01} class bytecodes with modified bytes.
103      */
104     @Override
105     protected byte[] generateClassBytes() {
106         // TODO: there is non-zero probability that generated bytecode will be
107         // valid, so it should be a subject of fuzzing mechanism
108         byte[] alteredBytes = testeeBytes.clone();
109         for (int j = 0; j < mutationCount; ++j) {
110             alteredBytes[getRNG().nextInt(alteredBytes.length)] = (byte) getRNG().nextInt(256);
111         }
112         return alteredBytes;
113     }
114 
115     /**
116      * Runs the test.
117      * @param args Test arguments.
118      */
119     public static void main(String[] args) {
120         StressClassLoadingTest.launch(args);
121     }
122 }
