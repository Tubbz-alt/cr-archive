 52 package vm.mlvm.indy.func.jvmti.stepBreakPopReturn;
 53 
 54 import java.lang.invoke.CallSite;
 55 import java.lang.invoke.ConstantCallSite;
 56 import java.lang.invoke.MethodHandle;
 57 import java.lang.invoke.MethodHandles;
 58 import java.lang.invoke.MethodHandles.Lookup;
 59 import java.lang.invoke.MethodType;
 60 
 61 import vm.mlvm.share.MlvmTest;
 62 
 63 public class INDIFY_Test extends MlvmTest {
 64 
 65     public static native void setDebuggeeMethodName(String name);
 66     public static native void setDebuggeeClassName(String name);
 67     public static native boolean checkStatus();
 68 
 69     static {
 70         System.loadLibrary("stepBreakPopReturn");
 71     }
 72 
 73     private static MethodType MT_bootstrap() {
 74         return MethodType.methodType(CallSite.class, Lookup.class, String.class, MethodType.class);
 75     }
 76 
 77     private static MethodHandle MH_bootstrap() throws NoSuchMethodException, IllegalAccessException {
 78         return MethodHandles.lookup().findStatic(
 79                 INDIFY_Test.class,
 80                 "bootstrap",
 81                 MT_bootstrap());
 82     }
 83 
 84     public static CallSite bootstrap(Lookup c, String name, MethodType mt) throws Throwable {
 85         int i = 0; // For single step
 86         getLog().trace(i, "Lookup " + c + "; method name = " + name + "; method type = " + mt);
 87         CallSite cs = new ConstantCallSite(MethodHandles.lookup().findStatic(
 88                 INDIFY_Test.class,
 89                 "target",
 90                 MethodType.methodType(int.class, Object.class, String.class, int.class)));
 91         return cs;
 92     }
 93 
 94     public static int target(Object o, String s, int i) {
 95         int j = 0; // For single step event
 96         getLog().trace(0, "Target called! Object = " + o + "; string = " + s + "; int = " + i);
 97         return i;
 98     }
 99 
100     private static MethodHandle INDY_call1;
101     private static MethodHandle INDY_call1() throws Throwable {
102         if (INDY_call1 != null)
103             return INDY_call1;
104 
105         CallSite cs = (CallSite) MH_bootstrap().invokeWithArguments(
106                 MethodHandles.lookup(),
107                 "greet",
108                 MethodType.methodType(int.class, Object.class, String.class, int.class));
109 
110         return cs.dynamicInvoker();
111     }
112 
113     private static MethodHandle INDY_call2;
114     private static MethodHandle INDY_call2() throws Throwable {
115         if (INDY_call2 != null)
116             return INDY_call2;
117 
118         CallSite cs = (CallSite) MH_bootstrap().invokeWithArguments(
119                 MethodHandles.lookup(),
120                 "greet",
121                 MethodType.methodType(int.class, Object.class, String.class, int.class));
122 
123         return cs.dynamicInvoker();
124     }
125 
126     public boolean run() throws Throwable {
127         setDebuggeeClassName("L" + INDIFY_Test.class.getName().replace('.', '/') + ";");
128         setDebuggeeMethodName("target");
129 
130         Object o = new Object();
131         String s = "heaven";
132         int i = 789;
133 
134         // When agent is loaded and correctly working, every call to target() should return 0
135         // (without agent target() returns 3rd argument)
136         // so the sum variable should be 0 at every moment
137         getLog().trace(0, "Call site 1");
138         int sum = (int) INDY_call1().invokeExact(o, s, i);
139         for ( i = 0; i < 5; i++ ) {
140             getLog().trace(0, "Call site 2, sum=" + sum);
141             sum += (int) INDY_call2().invokeExact(o, s, i + 789);
142         }
143         getLog().trace(0, "Direct call, sum=" + sum);
144         sum += target(new Object(), "hell", 123);
145 
146         getLog().trace(0, "Done, sum=" + sum + " (should be 0)");
147 
148         return checkStatus() && sum == 0;
149     }
150 
151     public static void main(String[] args) { MlvmTest.launch(args); }
152 }
