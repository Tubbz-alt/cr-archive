 46 class Parser {
 47     private final TreeMaker treeMaker;
 48     private final JextractTask.ConstantParser constantParser;
 49 
 50     public Parser(JextractTask.ConstantParser constantParser) {
 51         this.treeMaker = new TreeMaker();
 52         this.constantParser = constantParser;
 53     }
 54 
 55     public Declaration.Scoped parse(Path path, Collection<String> args) {
 56         final Index index = LibClang.createIndex(false);
 57 
 58         TranslationUnit tu = index.parse(path.toString(),
 59             d -> {
 60                 if (d.severity() > Diagnostic.CXDiagnostic_Warning) {
 61                     throw new RuntimeException(d.toString());
 62                 }
 63             },
 64             true, args.toArray(new String[0]));
 65 
 66         JextractTask.ConstantParser constantParser = this.constantParser != null ?
 67                 this.constantParser : new DefaultConstantParser(new MacroParserImpl(tu, args));
 68 
 69         List<Declaration> decls = new ArrayList<>();
 70         Cursor tuCursor = tu.getCursor();
 71         tuCursor.children().
 72             forEach(c -> {
 73                 SourceLocation loc = c.getSourceLocation();
 74                 if (loc == null) {
 75                     return;
 76                 }
 77 
 78                 SourceLocation.Location src = loc.getFileLocation();
 79                 if (src == null) {
 80                     return;
 81                 }
 82 
 83 
 84                 if (c.isDeclaration()) {
 85                     if (c.kind() == CursorKind.UnexposedDecl ||
 86                         c.kind() == CursorKind.Namespace) {
 87                         c.children().map(treeMaker::createTree)
 88                                 .filter(t -> t != null)
 89                                 .forEach(decls::add);
 90                     } else {
 91                         Declaration decl = treeMaker.createTree(c);
 92                         if (decl != null) {
 93                             decls.add(treeMaker.createTree(c));
 94                         }
 95                     }
 96                 } else if (isMacro(c) && src.path() != null) {
 97                     SourceRange range = c.getExtent();
 98                     String[] tokens = c.getTranslationUnit().tokens(range);
 99                     Optional<Declaration.Constant> constant = constantParser.parseConstant(treeMaker.toPos(c), c.spelling(), tokens);
100                     if (constant.isPresent()) {
101                         decls.add(constant.get());
102                     }
103                 }
104             });
105 



106         Declaration.Scoped rv = treeMaker.createHeader(tuCursor, decls);
107         treeMaker.freeze();
108         index.close();
109         return rv;
110     }
111 
112     private boolean isMacro(Cursor c) {
113         return c.isPreprocessing() && c.kind() == CursorKind.MacroDefinition;
114     }
115 
116     class DefaultConstantParser implements JextractTask.ConstantParser {
117 
118         final MacroParserImpl macroParser;
119 
120         public DefaultConstantParser(MacroParserImpl macroParser) {
121             this.macroParser = macroParser;
122         }
123 
124         @Override
125         public Optional<Declaration.Constant> parseConstant(Position pos, String name, String[] tokens) {
126             if (!(pos instanceof TreeMaker.CursorPosition)) {
127                 return Optional.empty();
128             } else {
129                 Cursor cursor = ((TreeMaker.CursorPosition)pos).cursor();
130                 if (cursor.isMacroFunctionLike()) {
131                     return Optional.empty();
132                 } else {
133                     return Optional.ofNullable(treeMaker.createMacro(cursor, macroParser.eval(name, tokens)));
134                 }
135             }
136         }
137     }

 50 class Parser {

 54     public Parser(JextractTask.ConstantParser constantParser) {
 55         this.treeMaker = new TreeMaker();
 56         this.constantParser = constantParser;
 57     }
 58 
 59     public Declaration.Scoped parse(Path path, Collection<String> args) {
 60         final Index index = LibClang.createIndex(false);
 61 
 62         TranslationUnit tu = index.parse(path.toString(),
 63             d -> {
 64                 if (d.severity() > Diagnostic.CXDiagnostic_Warning) {
 65                     throw new RuntimeException(d.toString());
 66                 }
 67             },
 68             true, args.toArray(new String[0]));
 69 
 70         JextractTask.ConstantParser constantParser = this.constantParser != null ?
 71                 this.constantParser : new MacroParserImpl(treeMaker, tu, args);
 72 
 73         List<Declaration> decls = new ArrayList<>();
 74         Cursor tuCursor = tu.getCursor();
 75         tuCursor.children().
 76             forEach(c -> {
 77                 SourceLocation loc = c.getSourceLocation();
 78                 if (loc == null) {
 79                     return;
 80                 }
 81 
 82                 SourceLocation.Location src = loc.getFileLocation();
 83                 if (src == null) {
 84                     return;
 85                 }
 86 
 87 
 88                 if (c.isDeclaration()) {
 89                     if (c.kind() == CursorKind.UnexposedDecl ||
 90                         c.kind() == CursorKind.Namespace) {
 91                         c.children().map(treeMaker::createTree)
 92                                 .filter(t -> t != null)
 93                                 .forEach(decls::add);
 94                     } else {
 95                         Declaration decl = treeMaker.createTree(c);
 96                         if (decl != null) {
 97                             decls.add(treeMaker.createTree(c));
 98                         }
 99                     }
100                 } else if (isMacro(c) && src.path() != null) {
101                     SourceRange range = c.getExtent();
102                     String[] tokens = c.getTranslationUnit().tokens(range);
103                     Optional<Declaration.Constant> constant = constantParser.parseConstant(treeMaker.toPos(c), c.spelling(), tokens);
104                     if (constant.isPresent()) {
105                         decls.add(constant.get());
106                     }
107                 }
108             });
109 
110         if (constantParser instanceof MacroParserImpl) {
111             decls.addAll(((MacroParserImpl)constantParser).macroTable.reparseConstants());
112         }
113         Declaration.Scoped rv = treeMaker.createHeader(tuCursor, decls);
114         treeMaker.freeze();
115         index.close();
116         return rv;
117     }

120         return c.isPreprocessing() && c.kind() == CursorKind.MacroDefinition;
121     }
