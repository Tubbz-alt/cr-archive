 91   -I /Library/Developer/CommandLineTools/SDKs/MacOSX.sdk/usr/include/python2.7/ \
 92   -t org.python \

362 ### Java sample code that uses LAPACK library
363 
364 ```java
365 
366 import jdk.incubator.foreign.NativeAllocationScope;
367 import lapack.*;
368 import static lapack.lapacke_h.*;
369 
370 public class TestLapack {
371     public static void main(String[] args) {
372 
373         /* Locals */
374         try (var scope = NativeAllocationScope.unboundedScope()) {
375             var A = Cdouble.allocateArray(new double[]{
376                     1, 2, 3, 4, 5, 1, 3, 5, 2, 4, 1, 4, 2, 5, 3
377             }, scope);
378             var b = Cdouble.allocateArray(new double[]{
379                     -10, 12, 14, 16, 18, -3, 14, 12, 16, 16
380             }, scope);
381             int info, m, n, lda, ldb, nrhs;
382 
383             /* Initialization */
384             m = 5;
385             n = 3;
386             nrhs = 2;
387             lda = 5;
388             ldb = 5;
389 
390             /* Print Entry Matrix */
391             print_matrix_colmajor("Entry Matrix A", m, n, A, lda );
392             /* Print Right Rand Side */
393             print_matrix_colmajor("Right Hand Side b", n, nrhs, b, ldb );
394             System.out.println();
395 
396             /* Executable statements */
397             //            printf( "LAPACKE_dgels (col-major, high-level) Example Program Results\n" );
398             /* Solve least squares problem*/
399             info = LAPACKE_dgels(LAPACK_COL_MAJOR(), (byte)'N', m, n, nrhs, A, lda, b, ldb);
400 
401             /* Print Solution */
402             print_matrix_colmajor("Solution", n, nrhs, b, ldb );
403             System.out.println();
404             System.exit(info);
405         }
406     }
407 
408     static void print_matrix_colmajor(String msg, int m, int n, MemoryAddress mat, int ldm) {
409         int i, j;
410         System.out.printf("\n %s\n", msg);
411 
412         for( i = 0; i < m; i++ ) {
413             for( j = 0; j < n; j++ ) System.out.printf(" %6.2f", Cdouble.get(mat, i+j*ldm));
414             System.out.printf( "\n" );
415         }
416     }
417 }
418 
419 ```

445 ### Java program that uses libproc to list processes
446 
447 ```java
448 
449 import jdk.incubator.foreign.NativeAllocationScope;
450 import org.unix.*;
451 import static jdk.incubator.foreign.MemoryAddress.NULL;
452 import static org.unix.libproc_h.*;
453 
454 public class LibprocMain {
455     private static final int NAME_BUF_MAX = 256;
456 
457     public static void main(String[] args) {
458         try (var scope = NativeAllocationScope.unboundedScope()) {
459             // get the number of processes
460             int numPids = proc_listallpids(NULL, 0);
461             // allocate an array
462             var pids = Cint.allocateArray(numPids, scope);
463             // list all the pids into the native array
464             proc_listallpids(pids, numPids);
465             // convert native array to java array
466             int[] jpids = Cint.toJavaArray(pids.segment());
467             // buffer for process name
468             var nameBuf = Cchar.allocateArray(NAME_BUF_MAX,scope);
469             for (int i = 0; i < jpids.length; i++) {
470                 int pid = jpids[i];
471                 // get the process name
472                 proc_name(pid, nameBuf, NAME_BUF_MAX);
473                 String procName = Cstring.toJavaString(nameBuf);
474                 // print pid and process name
475                 System.out.printf("%d %s\n", pid, procName);
476             }
477         }
478     }
479 }
480 
481 ```
