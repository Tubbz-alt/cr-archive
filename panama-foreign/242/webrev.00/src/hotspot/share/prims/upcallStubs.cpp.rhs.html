29 JVM_ENTRY(static jboolean, UH_FreeUpcallStub0(JNIEnv *env, jobject _unused, jlong addr))
30   //acquire code cache lock
31   MutexLocker mu(CodeCache_lock, Mutex::_no_safepoint_check_flag);
32   //find code blob
33   CodeBlob* cb = CodeCache::find_blob((char*)addr);
34   if (cb == NULL) {
35     return false;
36   }
37   //free global JNI handle
38   jobject* rec_ptr = (jobject*)(void*)cb -> content_begin();
39   JNIHandles::destroy_global(*rec_ptr);
40   //free code blob
41   CodeCache::free(cb);
42   return true;
43 JVM_END
