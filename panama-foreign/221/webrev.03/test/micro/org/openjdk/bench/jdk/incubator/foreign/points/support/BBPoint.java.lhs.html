23 package org.openjdk.bench.jdk.incubator.foreign.points.support;
24 
25 import java.nio.ByteBuffer;
26 import java.nio.ByteOrder;
27 
28 public class BBPoint {
29 




30     private final ByteBuffer buff;
31 
32     public BBPoint(int x, int y) {
33         this.buff = ByteBuffer.allocateDirect(4 * 2).order(ByteOrder.nativeOrder());
34         setX(x);
35         setY(y);
36     }
37 
38     public void setX(int x) {
39         buff.putInt(0, x);
40     }
41 
42     public int getX() {
43         return buff.getInt(0);
44     }
45 
46     public int getY() {
47         return buff.getInt(1);
48     }
49 
50     public void setY(int y) {
51         buff.putInt(0, y);
52     }






53 }
