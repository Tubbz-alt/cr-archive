 24 package jdk.test.lib.process;
 25 
 26 import java.io.BufferedReader;
 27 import java.io.ByteArrayOutputStream;
 28 import java.io.InputStream;
 29 import java.io.StringReader;
 30 import java.nio.charset.Charset;
 31 import java.time.Instant;
 32 import java.util.List;
 33 import java.util.concurrent.CancellationException;
 34 import java.util.concurrent.ExecutionException;
 35 import java.util.concurrent.Future;
 36 import java.util.stream.Collectors;
 37 
 38 public interface OutputBuffer {
 39   public static class OutputBufferException extends RuntimeException {
 40     private static final long serialVersionUID = 8528687792643129571L;
 41 
 42     public OutputBufferException(Throwable cause) {
 43       super(cause);
 44     }
 45   }
 46 
 47   /**
 48    * Returns the stdout result
 49    *
 50    * @return stdout result
 51    */
 52   public String getStdout();
 53 
 54   /**
 55    * Returns the stdout as a list.
 56    * Empty List if application produced no output.
 57    */
 58   default public List<String> getStdoutAsList() {
 59     return new BufferedReader(new StringReader(getStdout()))
 60             .lines()
 61             .collect(Collectors.toList());
 62   }
 63 
 64   /**
 65    * Returns the stderr result
 66    *
 67    * @return stderr result
 68    */
 69   public String getStderr();
 70   public int getExitValue();
 71 
 72   public static OutputBuffer of(Process p, Charset cs) {
 73     return new LazyOutputBuffer(p, cs);
 74   }
 75 
 76   public static OutputBuffer of(Process p) {
 77     return new LazyOutputBuffer(p, null);
 78   }
 79 
 80   public static OutputBuffer of(String stdout, String stderr, int exitValue) {
 81     return new EagerOutputBuffer(stdout, stderr, exitValue);
 82   }
 83 
 84   public static OutputBuffer of(String stdout, String stderr) {
 85     return of(stdout, stderr, -1);
 86   }
 87 
 88   class LazyOutputBuffer implements OutputBuffer {
 89     private static class StreamTask {
 90       private final ByteArrayOutputStream buffer;
 91       private final Future<Void> future;
 92       private final Charset cs;
 93 
 94       private StreamTask(InputStream stream, Charset cs) {
 95         this.buffer = new ByteArrayOutputStream();
 96         this.cs = cs;
 97         this.future = new StreamPumper(stream, buffer).process();
 98       }
 99 
100       public String get() {
101         try {
102           future.get();
103           return cs == null ? buffer.toString() : buffer.toString(cs);
104         } catch (InterruptedException e) {
105           Thread.currentThread().interrupt();
106           throw new OutputBufferException(e);
107         } catch (ExecutionException | CancellationException e) {
108           throw new OutputBufferException(e);
109         }
110       }
111     }
112 
113     private final StreamTask outTask;
114     private final StreamTask errTask;
115     private final Process p;
116 
117     private final void logProgress(String state) {
118         System.out.println("[" + Instant.now().toString() + "] " + state
119                            + " for process " + p.pid());
120         System.out.flush();
121     }
122 
123     private LazyOutputBuffer(Process p, Charset cs) {
124       this.p = p;
125       logProgress("Gathering output");
126       outTask = new StreamTask(p.getInputStream(), cs);
127       errTask = new StreamTask(p.getErrorStream(), cs);
128     }
129 
130     @Override
131     public String getStdout() {
132       return outTask.get();
133     }
134 
135     @Override
136     public String getStderr() {
137       return errTask.get();
138     }
139 
140     @Override
141     public int getExitValue() {
142       try {
143           logProgress("Waiting for completion");
144           boolean aborted = true;
145           try {
146               int result = p.waitFor();
147               logProgress("Waiting for completion finished");
148               aborted = false;
149               return result;
150           } finally {
151               if (aborted) {
152                   logProgress("Waiting for completion FAILED");
153               }
154           }
155       } catch (InterruptedException e) {
156         Thread.currentThread().interrupt();
157         throw new OutputBufferException(e);
158       }
159     }
160   }
161 
162   class EagerOutputBuffer implements OutputBuffer {
163     private final String stdout;
164     private final String stderr;
165     private final int exitValue;
166 
167     private EagerOutputBuffer(String stdout, String stderr, int exitValue) {
168       this.stdout = stdout;
169       this.stderr = stderr;
170       this.exitValue = exitValue;
171     }
172 
173     @Override
174     public String getStdout() {
175       return stdout;
176     }
177 
178     @Override
179     public String getStderr() {
180       return stderr;
181     }
182 
183     @Override
184     public int getExitValue() {
185       return exitValue;
186     }
187   }
188 }
