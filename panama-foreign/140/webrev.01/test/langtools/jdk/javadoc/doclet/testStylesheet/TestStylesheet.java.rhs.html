248     public void testStyles(Path base) throws Exception {
249         Path src = base.resolve("src");
250         tb.writeJavaFiles(src,
251                 "module mA { exports p; }",
252                 "package p; public class C {\n"
253                 + "public C() { }\n"
254                 + "public C(int i) { }\n"
255                 + "public int f1;\n"
256                 + "public int f2;\n"
257                 + "public int m1() { }\n"
258                 + "public int m2(int i) { }\n"
259                 + "}\n",
260                 "package p; public @interface Anno {\n"
261                 + "public int value();\n"
262                 + "}\n"
263         );
264 
265         javadoc("-d", base.resolve("out").toString(),
266                 "-sourcepath", src.toString(),
267                 "--module", "mA");
268         checkExit(Exit.OK);
269         checkStyles(addExtraCSSClassNamesTo(readStylesheet()));
270     }

272     Set<String> readStylesheet() {
273         // scan for class selectors, skipping '{' ... '}'
274         Set<String> styles = new TreeSet<>();
275         String stylesheet = readFile("stylesheet.css");
276         for (int i = 0; i < stylesheet.length(); i++) {
277             char ch = stylesheet.charAt(i);
278             switch (ch) {
279                 case '.':
280                     i++;
281                     int start = i;
282                     while (i < stylesheet.length()) {
283                         ch = stylesheet.charAt(i);
284                         if (!(Character.isLetterOrDigit(ch) || ch == '-')) {
285                             break;
286                         }
287                         i++;
288                     }
289                     styles.add(stylesheet.substring(start, i));
290                     break;
291 
292                 case '{':
293                     i++;
294                     while (i < stylesheet.length()) {
295                         ch = stylesheet.charAt(i);
296                         if (ch == '}') {
297                             break;
298                         }
299                         i++;
300                     }
301                     break;
302 
303                 case '@':
304                     i++;
305                     while (i < stylesheet.length()) {
306                         ch = stylesheet.charAt(i);
307                         if (ch == '{') {
308                             break;
309                         }
310                         i++;
311                     }
312                     break;
313             }
314         }
315         out.println("found styles: " + styles);
316         return styles;
317     }

319     Set<String> addExtraCSSClassNamesTo(Set<String> styles) throws Exception {
320         // The following names are used in the generated HTML,
321         // but have no corresponding definitions in the stylesheet file.
322         // They are mostly optional, in the "use if you want to" category.
323         // They are included here so that we do not get errors when these
324         // names are used in the generated HTML.
325         List<String> extra = List.of(
326                 // entries for <body> elements
327                 "all-classes-index-page",
328                 "all-packages-index-page",
329                 "constants-summary-page",
330                 "deprecated-list-page",
331                 "help-page",
332                 "index-redirect-page",
333                 "package-declaration-page",
334                 "package-tree-page",
335                 "single-index-page",
336                 "tree-page",
337                 // the following names are matched by [class$='...'] in the stylesheet
338                 "constructor-details",
339                 "constructor-summary",
340                 "field-details",
341                 "field-summary",
342                 "member-details",
343                 "method-details",
344                 "method-summary",
345                 // the following provide the ability to optionally override components of the
346                 // memberSignature structure
347                 "member-name",
348                 "modifiers",
349                 "packages",
350                 "return-type",
351                 // and others...
352                 "help-section",     // part of the help page
353                 "hierarchy",        // for the hierarchy on a tree page
354                 "index"             // on the index page
355         );
356         Set<String> all = new TreeSet<>(styles);
357         for (String e : extra) {
358             if (styles.contains(e)) {
359                 throw new Exception("extra CSS class name found in style sheet: " + e);
360             }
361             all.add(e);
362         }
363         return all;
364     }

366     /**
367      * Checks that all the CSS names found in {@code class} attributes in HTML files in the
368      * output directory are present in a given set of styles.
369      *
370      * @param styles the styles
371      */
372     void checkStyles(Set<String> styles) {
373         checking("Check CSS class names");
374         CSSClassChecker c = new CSSClassChecker(out, this::readFile, styles);
375         try {
376             c.checkDirectory(outputDir.toPath());
377             c.report();
378             int errors = c.getErrorCount();
379             if (errors == 0) {
380                 passed("No CSS class name errors found");
381             } else {
382                 failed(errors + " errors found when checking CSS class names");
383             }
384         } catch (IOException e) {
385             failed("exception thrown when reading files: " + e);
386         }
387 
388     }
389 
390     class CSSClassChecker extends HtmlChecker {
391         Set<String> styles;
392         int errors;
393 
394         protected CSSClassChecker(PrintStream out,
395                                   Function<Path, String> fileReader,
396                                   Set<String> styles) {
397             super(out, fileReader);
398             this.styles = styles;
399         }
400 
401         protected int getErrorCount() {
402             return errors;
403         }
404 
405         @Override
406         protected void report() {
407             if (getErrorCount() == 0) {
408                 out.println("All CSS class names found");
409             } else {
410                 out.println(getErrorCount() + " CSS class names not found");
411             }
412 
413         }
414 
415         @Override
416         public void startElement(String name, Map<String,String> attrs, boolean selfClosing) {
417             String style = attrs.get("class");
418             if (style != null && !styles.contains(style)) {
419                 error(currFile, getLineNumber(), "CSS class name not found: " + style);
420             }
421         }
