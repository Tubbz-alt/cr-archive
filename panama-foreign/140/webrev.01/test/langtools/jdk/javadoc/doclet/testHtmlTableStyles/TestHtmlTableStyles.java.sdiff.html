34 import javadoc.tester.JavadocTester;
35 
36 public class TestHtmlTableStyles extends JavadocTester {
37 
38     public static void main(String... args) throws Exception {
39         TestHtmlTableStyles tester = new TestHtmlTableStyles();
40         tester.runTests();
41     }
42 
43     @Test
44     public void test() {
45         javadoc("-d", "out",
46                 "-sourcepath", testSrc,
47                 "-use",
48                 "pkg1", "pkg2");
49         checkExit(Exit.ERROR);
50         checkOutput(Output.OUT, true,
51                 "attribute not supported in HTML5: summary",
52                 "attribute border for table only accepts \"\" or \"1\", use CSS instead: BORDER",
53                 "attribute not supported in HTML5: cellpadding",
54                 "attribute not supported in HTML5: cellspacing",
55                 "attribute not supported in HTML5: align");
56 
57         checkOutput("pkg1/TestTable.html", true,
58                 "<table summary=\"Summary\" border cellpadding=3 cellspacing=1>",
59                 "<div class=\"member-summary\">\n<table>",
60                 "<div class=\"member-summary\">\n<table>",
61                 "<div class=\"member-summary\">\n<table>");
62 
63         checkOutput("pkg1/package-summary.html", true,
64                 "<div class=\"type-summary\">\n<table>");
65 
66         checkOutput("pkg1/class-use/TestTable.html", true,
67                 "<div class=\"use-summary\">\n<table>");
68 
69         checkOutput("index.html", true,
70                 "<div class=\"overview-summary\">\n<table>");
71 
72         checkOutput("deprecated-list.html", true,
73             "<div class=\"deprecated-summary\" id=\"method\">\n<table>");
74 
75         checkOutput("constant-values.html", true,
76             "<div class=\"constants-summary\">\n<table>");
77     }
78 }
