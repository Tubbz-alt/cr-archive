 56     /**
 57      * This test calls Jcmd (diagnostic command tool) Compiler.queue and
 58      * then parses the output, making sure that the output look ok.
 59      *
 60      *
 61      * Output example:
 62      *
 63      * Current compiles:
 64      * C1 CompilerThread14 267       3       java.net.URLStreamHandler::parseURL (1166 bytes)
 65      * C1 CompilerThread13 760       3       javax.management.StandardMBean::getDescription (11 bytes)
 66      * C1 CompilerThread12 757  s    3       com.sun.jmx.mbeanserver.DefaultMXBeanMappingFactory::getMapping (27 bytes)
 67      * C1 CompilerThread11 756  s!   3       com.sun.jmx.mbeanserver.DefaultMXBeanMappingFactory::mappingForType (110 bytes)
 68      * C1 CompilerThread10 761       3       java.lang.StringLatin1::indexOf (121 bytes)
 69      * C2 CompilerThread7 769       4       CompilerQueueTest::testcaseMethod4 (1 bytes)
 70      *
 71      * C1 compile queue:
 72      * 762       3       java.lang.invoke.MethodType::basicType (8 bytes)
 73      * 763       3       java.util.ArrayList::rangeCheck (22 bytes)
 74      * 764       3       java.util.ArrayList::elementData (7 bytes)
 75      * 765       3       jdk.internal.org.objectweb.asm.MethodVisitor::<init> (35 bytes)
 76      * 766       1       CompilerQueueTest::testcaseMethod1 (1 bytes)
 77      * 767       2       CompilerQueueTest::testcaseMethod2 (1 bytes)
 78      * 768       3       CompilerQueueTest::testcaseMethod3 (1 bytes)
 79      * 770       3       java.util.Properties::getProperty (46 bytes)
 80      *
 81      * C2 compile queue:
 82      * Empty
 83      *
 84      **/
 85 
 86     protected static final WhiteBox WB = WhiteBox.getWhiteBox();
 87 
 88     public void run(CommandExecutor executor) {
 89 
 90         TestCase[] testcases = {
 91                 new TestCase(1, "testcaseMethod1"),
 92                 new TestCase(2, "testcaseMethod2"),
 93                 new TestCase(3, "testcaseMethod3"),
 94                 new TestCase(4, "testcaseMethod4"),
 95         };
 96 
 97         // Lock compilation makes all compiles stay in queue or compile thread before completion
 98         WB.lockCompilation();
 99 
100         // Enqueue one test method for each available level
101         int[] complevels = CompilerUtils.getAvailableCompilationLevels();
102         for (int level : complevels) {
103             TestCase testcase = testcases[level - 1];
104 
105             boolean added = WB.enqueueMethodForCompilation(testcase.method, testcase.level);
106             // Set results to false for those methods we must to find
107             // We will also assert if we find any test method we don't expect
108             Assert.assertEquals(added, WB.isMethodQueuedForCompilation(testcase.method));
109             testcase.check = false;
110         }
111 
112         // Get output from dcmd (diagnostic command)
113         OutputAnalyzer output = executor.execute("Compiler.queue");
114         Iterator<String> lines = output.asLines().iterator();
115 
116         // Loop over output set result for all found methods
117         while (lines.hasNext()) {
118             String str = lines.next();
119             // Fast check for common part of method name
120             if (str.contains("testcaseMethod")) {
121                 for (TestCase testcase : testcases) {
122                     if (str.contains(testcase.methodName)) {
123                         Assert.assertFalse(testcase.check, "Must not be found or already found.");
124                         testcase.check = true;
125                     }
126                 }
127             }
128         }
129 
130         for (TestCase testcase : testcases) {
131             if (!testcase.check) {
132                 // If this method wasn't found it must have been removed by policy,
133                 // verify that it is now removed from the queue
134                 Assert.assertFalse(WB.isMethodQueuedForCompilation(testcase.method), "Must be found or not in queue");
135             }
136             // Otherwise all good.
137         }
138 
139         // Enable compilations again
140         WB.unlockCompilation();
141     }

160     public static Method getMethod(Class klass, String name, Class<?>... parameterTypes) {
161         try {
162             return klass.getDeclaredMethod(name, parameterTypes);
163         } catch (NoSuchMethodException | SecurityException e) {
164             throw new RuntimeException("exception on getting method Helper." + name, e);
165         }
166     }
167 
168     class TestCase {
169         Method method;
170         int level;
171         String methodName;
172         Boolean check;
173 
174         public TestCase(int level, String methodName) {
175             this.method = getMethod(CompilerQueueTest.class, methodName);
176             this.level = level;
177             this.methodName = methodName;
178             this.check = true;
179         }
180     }
