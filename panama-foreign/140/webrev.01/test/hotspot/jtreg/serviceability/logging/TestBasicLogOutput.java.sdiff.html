25  * @test TestBasicLogOutput
26  * @summary Ensure logging can be enabled and successfully prints to stdout.
27  * @modules java.base/jdk.internal.misc
28  * @library /test/lib

29  */
30 
31 import jdk.test.lib.process.ProcessTools;
32 import jdk.test.lib.process.OutputAnalyzer;
33 
34 public class TestBasicLogOutput {
35 
36     public static void main(String[] args) throws Exception {
37         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-Xlog:all=trace", "-version");
38         OutputAnalyzer output = new OutputAnalyzer(pb.start());
39         output.shouldMatch("\\[logging *\\]"); // expected tag(s)
40         output.shouldContain("Log configuration fully initialized."); // expected message
41         output.shouldHaveExitValue(0);
42     }
43 }
