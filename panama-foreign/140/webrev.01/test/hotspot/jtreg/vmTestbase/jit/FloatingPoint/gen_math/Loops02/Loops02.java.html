40 // Test working with  loops and random functions.
41 import nsk.share.TestFailure;
42 
43 class Loops02
44 {
45 
46    static final int N = 300;
47 
48    public static void main (String args[])
49    {
50 
51 
52         double x;
53         double r1, r2, r3;
54 
55         double rn = N;
56         double dx = 1/rn;
57         double Min_count = rn/2;
58 
59         Loops02 ll;
60         ll = new Loops02();
61 
62         x = 0.5;
63         int count = 0;
64         do
65         {
66                 r1 = Math.random();
67                 r2 = Math.random();
68                 x = x + dx * (r1 - r2);
69                 count++;
70 
71         } while(x > 0 && x < 1);
72 
73         ll.Echeck(Min_count,count);
74 
75   }
76 
77 
78    public void Echeck(double er, double ER)
79    {
80 
81         if( er < ER)
82                 System.out.println("test PASS");
83         else
84         {
85                 System.out.println("expected error: " + ER);
86                 System.out.println("   found error: " + er);
87                 throw new TestFailure("test FAIL");
88         }
89 
90    }
