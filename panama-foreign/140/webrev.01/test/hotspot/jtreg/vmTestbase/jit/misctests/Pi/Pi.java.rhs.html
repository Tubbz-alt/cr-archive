38 package jit.misctests.Pi;
39 
40 import java.util.Random;
41 import nsk.share.TestFailure;
42 import jdk.test.lib.Utils;
43 
44 class Pi{
45     static double pi;
46     static int imKreis=0, imQuadrat=0, i=0;
47 
48     public static void main(String args[]) {
49         for(int i=0;i<=100;i++) {
50             wurf(10000);
51             pi=4.0*imKreis/imQuadrat;
52             ausgabe();
53         }
54         System.out.print("\n");
55     }
56 
57     static void wurf(int wieOft) {
58         double x,y;
59         Random zufall;
60         zufall = Utils.getRandomInstance();
61         for(int i=0;i<=wieOft;i++) {
62             x=zufall.nextDouble();
63             y=zufall.nextDouble();
64             imQuadrat++;
65             if(x*x+y*y<1)imKreis++;
66         }
67     }
68 
69     static void ausgabe() {
70         System.out.print(pi);
71         if ((++i % 4) == 0){
72             System.out.print("\n");
73         } else {
74             System.out.print(" ");
75         }
76     }
77 }
