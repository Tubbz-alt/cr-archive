38 
39 import java.util.Random;
40 import nsk.share.TestFailure;

41 
42 class Pi{
43     static double pi;
44     static int imKreis=0, imQuadrat=0, i=0;
45 
46     public static void main(String args[]) {
47         for(int i=0;i<=100;i++) {
48             wurf(10000);
49             pi=4.0*imKreis/imQuadrat;
50             ausgabe();
51         }
52         System.out.print("\n");
53     }
54 
55     static void wurf(int wieOft) {
56         double x,y;
57         Random zufall;
58         zufall=new Random();
59         for(int i=0;i<=wieOft;i++) {
60             x=zufall.nextDouble();
61             y=zufall.nextDouble();
62             imQuadrat++;
63             if(x*x+y*y<1)imKreis++;
64         }
65     }
66 
67     static void ausgabe() {
68         System.out.print(pi);
69         if ((++i % 4) == 0){
70             System.out.print("\n");
71         } else {
72             System.out.print(" ");
73         }
74     }
75 }
