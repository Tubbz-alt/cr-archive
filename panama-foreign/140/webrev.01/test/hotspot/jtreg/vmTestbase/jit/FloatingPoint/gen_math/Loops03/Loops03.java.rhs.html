 40 import nsk.share.TestFailure;
 41 
 42 class Loops03
 43 {
 44 
 45    static final int N = 100000;
 46 
 47    public static void main (String args[])
 48    {
 49 
 50 
 51         double x;
 52         double r1, r2, r3, r4, r5, r6, r7;
 53         double Error = 0.001;
 54 
 55 
 56         Loops03 ll;
 57         ll = new Loops03();
 58 
 59         int i = 1;
 60         double sum = 0;
 61         double prod = 1;
 62         while(i < N)
 63         {       r1 = ll.Random_arg(i);
 64                 r3 = Math.sin(r1);
 65                 r4 = Math.cos(r1);
 66                 r5 = r3 * r3 + r4 * r4;
 67                 r6 = i + i;
 68                 r7 = r6 * r6;
 69                 sum = sum + r5/r7;
 70 
 71                 r2 = ll.Random_arg(i);
 72                 r3 = Math.sin(r1);
 73                 r4 = Math.cos(r1);
 74                 r5 = r3 * r3 + r4 * r4;
 75                 r6 = i + i;
 76                 r7 = r6 * r6;
 77                 prod = prod * (1 + r5/r7);
 78                 i++;
 79         }
 80         double er1 = Math.abs(sum - 0.411);
 81         double er2 = Math.abs(prod - 1.465);
 82         double errrr = Math.sqrt(er1 * er1 + er2 * er2);
 83 
 84 
 85         ll.Echeck(errrr,Error);
 86 
 87   }
 88 
 89 // method below return double random number in interval
 90 //              (0, int nn)
 91 
 92    public double Random_arg(int nn)
 93    {
 94         double rr;
 95         rr = Math.random();
 96         double rn = nn;
 97         double ru = rr * rn;
 98         return ru;
 99 
100    }
101 
102    public void Echeck(double er, double ER)
103    {
104 
105         if( er < ER)
106                 System.out.println("test PASS");
107         else
108         {
109                 System.out.println("expected error: " + ER);
110                 System.out.println("   found error: " + er);
111                 throw new TestFailure("test FAIL");
112         }
113 
114    }
