 48 package nsk.jdi.HiddenClass.events;
 49 
 50 import com.sun.jdi.Field;
 51 import com.sun.jdi.Method;
 52 import com.sun.jdi.ReferenceType;
 53 
 54 import com.sun.jdi.request.EventRequest;
 55 
 56 import nsk.jdi.HiddenClass.events.DebuggerBase;
 57 import nsk.jdi.HiddenClass.events.EventHandler;
 58 
 59 import nsk.share.Log;
 60 import nsk.share.jdi.ArgumentHandler;
 61 
 62 // This class is the test debugger
 63 public class events001 extends DebuggerBase {
 64     static final String PACKAGE_NAME  = "nsk.jdi.HiddenClass.events";
 65     static final String DEBUGGEE_NAME = PACKAGE_NAME + ".events001a";
 66     static final String CHECKED_CLASS = PACKAGE_NAME + ".HiddenClass";
 67     static final String HC_FILTER = CHECKED_CLASS + "/0x*";
 68 
 69     private events001(ArgumentHandler argHandler) {
 70         super(argHandler);
 71     }
 72 
 73     public static void main (String args[]) {
 74         ArgumentHandler argHandler = new ArgumentHandler(args);
 75 
 76         events001 debugger = new events001(argHandler);
 77         System.exit(debugger.run(argHandler) + JCK_STATUS_BASE);
 78     }
 79 
 80     public int run(ArgumentHandler argHandler) {
 81         boolean testFailed = false;
 82         EventRequest breakpointRequest = null;
 83         EventRequest classPrepareRequest = null;
 84         EventRequest classUnloadRequest = null;
 85         EventRequest modWatchpointRequest = null;
 86         launchDebuggee(argHandler, DEBUGGEE_NAME);
 87 
 88         try {
 89             EventHandler eventHandler = EventHandler.createAndStart(this);
 90 
 91             // sync with debuggee
 92             readyCmdSync();
 93 
 94             // request a ClassPrepare event for the hidden class "HiddenClass/0x*"
 95             classPrepareRequest = enableClassPrepareRequest(HC_FILTER);
 96 
 97             // request a ClassUnload event for the hidden class "HiddenClass/0x*"
 98             classUnloadRequest = enableClassUnloadRequest(HC_FILTER);
 99 
100             // sync with debuggee
101             runCmdSync();
102             readyCmdSync();
103 
104             // There is a latency in getting events from the debuggee
105             // on the debugger side over the wire protocol, so we may
106             // need to wait for ClassPrepareEvent to be posted.
107             ReferenceType hcRefType = eventHandler.waitAndGetHCRefType();
108 
109             /* Hidden class has to be prepared at this point. */
110 
111             // request a Breakpoint event in the hidden class method "hcMethod"
112             Method method = findMethod(hcRefType, "hcMethod");
113             breakpointRequest = enableBreakpointRequest(method);
114 
115             // request a ModificationWatchpoint event on the hidden class field "hcField"
116             Field field = findField(hcRefType, "hcField");
117             modWatchpointRequest = enableModificationWatchpointRequest(field, HC_FILTER);
118 
119             // sync with debuggee
120             runCmdSync();
121             doneCmdSync();
122 
123             eventHandler.waitForCompleteness();
124             testFailed |= eventHandler.failedStatus();
125         } catch (Throwable t) {
126             log.complain("FAIL: " + t.getMessage());
127             t.printStackTrace(log.getOutStream());
128             testFailed = true;
129         } finally {
130             // disable event requests
131             disableRequest(breakpointRequest, "BreakpointRequest");
132             disableRequest(classPrepareRequest, "ClassPrepareRequest");
133             disableRequest(classUnloadRequest, "ClassUnloadRequest");
134             disableRequest(modWatchpointRequest, "ModificationWatchpointRequest");
135 
136             // sync with debuggee
137             quitCmdSync();
138             testFailed |= shutdownDebuggee();
139         }
140         // check test results
141         if (testFailed) {
142             log.complain("# TEST FAILED");
143             return FAILED;
144         }
145         log.display("# TEST PASSED");
146         return PASSED;
147     }
148 }
