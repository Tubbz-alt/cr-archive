 47 package vm.compiler.CodeCacheInfo;
 48 
 49 import sun.hotspot.WhiteBox;
 50 import jdk.test.lib.process.OutputAnalyzer;
 51 import jdk.test.lib.process.ProcessTools;
 52 
 53 public class Test {
 54     private static final String SEG_REGEXP;
 55     private static final String NOSEG_REGEXP;
 56 
 57     static {
 58         String p1 = " size=\\d+Kb used=\\d+Kb max_used=\\d+Kb free=\\d+Kb\\n";
 59         String p2 = " bounds \\[0x[0-9a-f]+, 0x[0-9a-f]+, 0x[0-9a-f]+\\]\\n";
 60         String p3 = " total_blobs=\\d+ nmethods=\\d+ adapters=\\d+\\n";
 61         String p4 = " compilation: enabled\\n";
 62 
 63         String segPrefix = "^(CodeHeap '[^']+':" + p1 + p2 + ")+";
 64         String nosegPrefix = "^CodeCache:" + p1 + p2;
 65 
 66         // check if AOT is enabled
 67         if (WhiteBox.getWhiteBox().aotLibrariesCount() > 0) {
 68             System.out.println("AOT is enabled");
 69             String aotSegment = "CodeHeap 'AOT':" + p1 + p2;
 70             segPrefix += aotSegment;
 71             nosegPrefix += aotSegment;
 72         } else {
 73             System.out.println("AOT is not enabled");
 74         }
 75         SEG_REGEXP = segPrefix + p3 + p4;
 76         NOSEG_REGEXP = nosegPrefix + p3 + p4;
 77     }
 78 
 79     public static void main(String[] args) throws Exception {
 80         {
 81             System.out.println("SegmentedCodeCache is enabled");
 82             var pb = ProcessTools.createJavaProcessBuilder(true,
 83                     "-XX:+SegmentedCodeCache",
 84                     "-XX:+PrintCodeCache",
 85                     "-version");
 86             var output = new OutputAnalyzer(pb.start());
 87             output.shouldHaveExitValue(0);
 88             output.stdoutShouldMatch(SEG_REGEXP);
 89         }
 90         {
 91             System.out.println("SegmentedCodeCache is disabled");
 92             var pb = ProcessTools.createJavaProcessBuilder(true,
 93                     "-XX:-SegmentedCodeCache",
 94                     "-XX:+PrintCodeCache",
 95                     "-version");
 96             var output = new OutputAnalyzer(pb.start());
 97             output.shouldHaveExitValue(0);
 98             output.stdoutShouldMatch(NOSEG_REGEXP);
 99         }
100     }
101 }
