23 package vm.compiler.coverage.parentheses.share.generation;
24 
25 import java.util.Random;
26 
27 
28 /**
29  * This class generate rrandom ight parentheses sequence in string representation: "(())", "()(())" etc
30  */
31 public class ParenthesesGenerator {
32     private static Random random = new Random();
33 
34     public static String generate(int size) {
35         if (size == 0) {
36             return "";
37         }
38         if (random.nextBoolean()) {
39             return "(" + generate(size - 1) + ")";
40         } else {
41             int splitPoint = random.nextInt(size + 1);
42             return generate(splitPoint) + generate(size - splitPoint);
43         }
44     }
45 }
