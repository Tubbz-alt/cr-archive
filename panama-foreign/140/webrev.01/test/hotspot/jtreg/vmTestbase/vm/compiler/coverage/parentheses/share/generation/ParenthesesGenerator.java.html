23 package vm.compiler.coverage.parentheses.share.generation;
24 
25 import java.util.Random;
26 import jdk.test.lib.Utils;
27 
28 
29 /**
30  * This class generate rrandom ight parentheses sequence in string representation: "(())", "()(())" etc
31  */
32 public class ParenthesesGenerator {
33     private static Random random = Utils.getRandomInstance();
34 
35     public static String generate(int size) {
36         if (size == 0) {
37             return "";
38         }
39         if (random.nextBoolean()) {
40             return "(" + generate(size - 1) + ")";
41         } else {
42             int splitPoint = random.nextInt(size + 1);
43             return generate(splitPoint) + generate(size - splitPoint);
44         }
45     }
46 }
