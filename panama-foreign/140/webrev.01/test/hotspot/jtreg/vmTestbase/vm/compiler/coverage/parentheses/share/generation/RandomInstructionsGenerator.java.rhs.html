23 package vm.compiler.coverage.parentheses.share.generation;
24 
25 import vm.compiler.coverage.parentheses.share.Instruction;
26 import vm.compiler.coverage.parentheses.share.InstructionSequence;
27 
28 import java.util.ArrayList;
29 import java.util.List;
30 import java.util.Random;
31 import jdk.test.lib.Utils;
32 
33 /**
34  * Generates random but correct list of JVM instructions
35  */
36 public class RandomInstructionsGenerator {
37     private static Random random = Utils.getRandomInstance();
38 
39     private static <E> E choseRandomElement(List<E> list) {
40         return list.get(random.nextInt(list.size()));
41     }
42 
43     private int maxStackDepth;
44 
45     public RandomInstructionsGenerator(int maxStackDepth) {
46         this.maxStackDepth = maxStackDepth;
47     }
48 
49     public InstructionSequence generate() {
50         List<Instruction> instructions = new ArrayList<Instruction>();
51 
52         //this head with constants is necessary to avoid stack underflow
53         instructions.add(Instruction.ICONST_1);
54         instructions.add(Instruction.ICONST_2);
55 
56 
57         String parenthesis = ParenthesesGenerator.generate(maxStackDepth - 2);
58 
59         for (char c : parenthesis.toCharArray()) {
60             if (c == '(') {
61                 //+1: add op that increase stack
62                 instructions.add(choseRandomElement(Instruction.stackUp));
63             } else {
64                 //-1
65                 instructions.add(choseRandomElement(Instruction.stackDown));
66             }
67 
68             //+0: add element that doesn't change stack
69             if (random.nextBoolean()) {
70                 instructions.add(choseRandomElement(Instruction.neutral));
71             }
72 
73         }
74         return new InstructionSequence(instructions, maxStackDepth);
75     }
76 }
