24 /**
25  * @test
26  * @bug 8031320
27  * @summary Verify UseRTMDeopt option processing on CPUs or OSs without rtm support
28  *          or on VMs without rtm locking support.
29  * @library /test/lib /
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  * @requires !vm.rtm.cpu & vm.rtm.compiler
33  * @build sun.hotspot.WhiteBox
34  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
35  *                                sun.hotspot.WhiteBox$WhiteBoxPermission
36  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
37  *                   -XX:+WhiteBoxAPI
38  *                   compiler.rtm.cli.TestUseRTMDeoptOptionOnUnsupportedConfig
39  */
40 
41 package compiler.rtm.cli;
42 
43 import jdk.test.lib.cli.CommandLineOptionTest;
44 
45 public class TestUseRTMDeoptOptionOnUnsupportedConfig
46         extends RTMGenericCommandLineOptionTest {
47     private static final String DEFAULT_VALUE = "false";
48 
49     private TestUseRTMDeoptOptionOnUnsupportedConfig() {
50         super("UseRTMDeopt", true, false,
51                 TestUseRTMDeoptOptionOnUnsupportedConfig.DEFAULT_VALUE,
52                 "true");
53     }
54 
55     @Override
56     protected void runX86SupportedVMTestCases() throws Throwable {
57         super.verifyJVMStartup();
58         // verify default value
59         CommandLineOptionTest.verifyOptionValueForSameVM(optionName,
60                 defaultValue, String.format("'%s' should have '%s' "
61                         + "default value on unsupported configs.",
62                         optionName, DEFAULT_VALUE));
63         // verify that until RTMLocking is not used, value
64         // will be set to default false.
65         CommandLineOptionTest.verifyOptionValueForSameVM(optionName,
66                 defaultValue, String.format("'%s' should be off on unsupported"
67                         + " configs even if '-XX:+%s' flag set", optionName,
68                         optionName),
69                 "-XX:+UseRTMDeopt");
70     }
71 
72     public static void main(String args[]) throws Throwable {
73         new TestUseRTMDeoptOptionOnUnsupportedConfig().runTestCases();
74     }
75 }
