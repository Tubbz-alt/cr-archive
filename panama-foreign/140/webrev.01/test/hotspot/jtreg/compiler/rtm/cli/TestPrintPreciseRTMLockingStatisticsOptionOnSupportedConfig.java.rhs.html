24 /**
25  * @test
26  * @bug 8031320
27  * @summary Verify PrintPreciseRTMLockingStatistics on CPUs and OSs with
28  *          rtm support and on VM with rtm locking support,
29  * @library /test/lib /
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  * @requires vm.rtm.cpu & vm.rtm.compiler
33  * @run driver compiler.rtm.cli.TestPrintPreciseRTMLockingStatisticsOptionOnSupportedConfig





34  */
35 
36 package compiler.rtm.cli;
37 
38 import jdk.test.lib.cli.CommandLineOptionTest;
39 
40 public class TestPrintPreciseRTMLockingStatisticsOptionOnSupportedConfig
41         extends TestPrintPreciseRTMLockingStatisticsBase {
42 
43     @Override
44     protected void verifyOptionValues() throws Throwable {
45         super.verifyOptionValues();
46         // verify default value
47         CommandLineOptionTest.verifyOptionValueForSameVM(optionName,
48                 TestPrintPreciseRTMLockingStatisticsBase.DEFAULT_VALUE,
49                 String.format("Option '%s' should have '%s' default value on"
50                         + " supported CPU", optionName,
51                 TestPrintPreciseRTMLockingStatisticsBase.DEFAULT_VALUE),
52                 CommandLineOptionTest.UNLOCK_DIAGNOSTIC_VM_OPTIONS,
53                 CommandLineOptionTest.UNLOCK_EXPERIMENTAL_VM_OPTIONS,
54                 "-XX:+UseRTMLocking");
55 
56         CommandLineOptionTest.verifyOptionValueForSameVM(optionName,
57                 TestPrintPreciseRTMLockingStatisticsBase.DEFAULT_VALUE,
58                 String.format("Option '%s' should have '%s' default value on"
59                         + " supported CPU when -XX:-UseRTMLocking flag set",
60                         optionName,
61                        TestPrintPreciseRTMLockingStatisticsBase.DEFAULT_VALUE),
62                 CommandLineOptionTest.UNLOCK_DIAGNOSTIC_VM_OPTIONS,
63                 CommandLineOptionTest.UNLOCK_EXPERIMENTAL_VM_OPTIONS,
64                 "-XX:-UseRTMLocking", prepareOptionValue("true"));
65 
66         // verify that option could be turned on
67         CommandLineOptionTest.verifyOptionValueForSameVM(optionName, "true",
68                 String.format("Option '%s' should have 'true' value when set "
69                         + "on supported CPU and -XX:+UseRTMLocking flag set",
70                         optionName),
71                 CommandLineOptionTest.UNLOCK_DIAGNOSTIC_VM_OPTIONS,
72                 CommandLineOptionTest.UNLOCK_EXPERIMENTAL_VM_OPTIONS,
73                 "-XX:+UseRTMLocking", prepareOptionValue("true"));
74     }
75 
76     public static void main(String args[]) throws Throwable {
77         new TestPrintPreciseRTMLockingStatisticsOptionOnSupportedConfig()
78                 .runTestCases();
79     }
80 }
