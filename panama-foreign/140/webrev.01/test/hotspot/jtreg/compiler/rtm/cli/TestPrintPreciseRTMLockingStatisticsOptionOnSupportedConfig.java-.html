24 /**
25  * @test
26  * @bug 8031320
27  * @summary Verify PrintPreciseRTMLockingStatistics on CPUs and OSs with
28  *          rtm support and on VM with rtm locking support,
29  * @library /test/lib /
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  * @requires vm.rtm.cpu & vm.rtm.compiler
33  * @build sun.hotspot.WhiteBox
34  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
35  *                                sun.hotspot.WhiteBox$WhiteBoxPermission
36  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
37  *                   -XX:+WhiteBoxAPI
38  *                   compiler.rtm.cli.TestPrintPreciseRTMLockingStatisticsOptionOnSupportedConfig
39  */
40 
41 package compiler.rtm.cli;
42 
43 import jdk.test.lib.cli.CommandLineOptionTest;
44 
45 public class TestPrintPreciseRTMLockingStatisticsOptionOnSupportedConfig
46         extends TestPrintPreciseRTMLockingStatisticsBase {
47 
48     @Override
49     protected void verifyOptionValues() throws Throwable {
50         super.verifyOptionValues();
51         // verify default value
52         CommandLineOptionTest.verifyOptionValueForSameVM(optionName,
53                 TestPrintPreciseRTMLockingStatisticsBase.DEFAULT_VALUE,
54                 String.format("Option '%s' should have '%s' default value on"
55                         + " supported CPU", optionName,
56                 TestPrintPreciseRTMLockingStatisticsBase.DEFAULT_VALUE),
57                 CommandLineOptionTest.UNLOCK_DIAGNOSTIC_VM_OPTIONS,
58                 CommandLineOptionTest.UNLOCK_EXPERIMENTAL_VM_OPTIONS,
59                 "-XX:+UseRTMLocking");
60 
61         CommandLineOptionTest.verifyOptionValueForSameVM(optionName,
62                 TestPrintPreciseRTMLockingStatisticsBase.DEFAULT_VALUE,
63                 String.format("Option '%s' should have '%s' default value on"
64                         + " supported CPU when -XX:-UseRTMLocking flag set",
65                         optionName,
66                        TestPrintPreciseRTMLockingStatisticsBase.DEFAULT_VALUE),
67                 CommandLineOptionTest.UNLOCK_DIAGNOSTIC_VM_OPTIONS,
68                 CommandLineOptionTest.UNLOCK_EXPERIMENTAL_VM_OPTIONS,
69                 "-XX:-UseRTMLocking", prepareOptionValue("true"));
70 
71         // verify that option could be turned on
72         CommandLineOptionTest.verifyOptionValueForSameVM(optionName, "true",
73                 String.format("Option '%s' should have 'true' value when set "
74                         + "on supported CPU and -XX:+UseRTMLocking flag set",
75                         optionName),
76                 CommandLineOptionTest.UNLOCK_DIAGNOSTIC_VM_OPTIONS,
77                 CommandLineOptionTest.UNLOCK_EXPERIMENTAL_VM_OPTIONS,
78                 "-XX:+UseRTMLocking", prepareOptionValue("true"));
79     }
80 
81     public static void main(String args[]) throws Throwable {
82         new TestPrintPreciseRTMLockingStatisticsOptionOnSupportedConfig()
83                 .runTestCases();
84     }
85 }
