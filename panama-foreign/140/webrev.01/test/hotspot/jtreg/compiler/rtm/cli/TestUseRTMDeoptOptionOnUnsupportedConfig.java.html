24 /**
25  * @test
26  * @bug 8031320
27  * @summary Verify UseRTMDeopt option processing on CPUs or OSs without rtm support
28  *          or on VMs without rtm locking support.
29  * @library /test/lib /
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  * @requires !vm.rtm.cpu & vm.rtm.compiler
33  * @run driver compiler.rtm.cli.TestUseRTMDeoptOptionOnUnsupportedConfig
34  */
35 
36 package compiler.rtm.cli;
37 
38 import jdk.test.lib.cli.CommandLineOptionTest;
39 
40 public class TestUseRTMDeoptOptionOnUnsupportedConfig
41         extends RTMGenericCommandLineOptionTest {
42     private static final String DEFAULT_VALUE = "false";
43 
44     private TestUseRTMDeoptOptionOnUnsupportedConfig() {
45         super("UseRTMDeopt", true, false,
46                 TestUseRTMDeoptOptionOnUnsupportedConfig.DEFAULT_VALUE,
47                 "true");
48     }
49 
50     @Override
51     protected void runX86SupportedVMTestCases() throws Throwable {
52         super.verifyJVMStartup();
53         // verify default value
54         CommandLineOptionTest.verifyOptionValueForSameVM(optionName,
55                 defaultValue, String.format("'%s' should have '%s' "
56                         + "default value on unsupported configs.",
57                         optionName, DEFAULT_VALUE));
58         // verify that until RTMLocking is not used, value
59         // will be set to default false.
60         CommandLineOptionTest.verifyOptionValueForSameVM(optionName,
61                 defaultValue, String.format("'%s' should be off on unsupported"
62                         + " configs even if '-XX:+%s' flag set", optionName,
63                         optionName),
64                 "-XX:+UseRTMDeopt");
65     }
66 
67     public static void main(String args[]) throws Throwable {
68         new TestUseRTMDeoptOptionOnUnsupportedConfig().runTestCases();
69     }
70 }
