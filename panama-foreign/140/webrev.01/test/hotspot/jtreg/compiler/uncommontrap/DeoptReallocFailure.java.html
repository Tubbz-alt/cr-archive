38 package compiler.uncommontrap;
39 
40 import sun.hotspot.WhiteBox;
41 
42 import java.lang.reflect.Method;
43 
44 public class DeoptReallocFailure {
45     static class MemoryChunk {
46         MemoryChunk other;
47         Object[][] array;
48 
49         MemoryChunk(MemoryChunk other) {
50             this.other = other;
51             array = new Object[1024 * 256][];
52         }
53     }
54 
55     static class NoEscape {
56         long f1;
57     }
58 
59     static MemoryChunk root;
60     private static final WhiteBox WB = WhiteBox.getWhiteBox();
61 
62     public static synchronized long  test() {
63         NoEscape[] noEscape = new NoEscape[45];
64         noEscape[0] = new NoEscape();
65         for (int i=0;i<1024*256;i++) {
66            root.array[i]= new Object[45];
67         }
68         return noEscape[0].f1;
69     }
70 
71     public static void main(String[] args) throws Throwable {
72 
73         //Exhaust Memory
74         root = null;
75         try {
76             while (true) {
77                 root = new MemoryChunk(root);
78             }
79         } catch (OutOfMemoryError oom) {
80         }
81 
82         if (root == null) {
83           return;
84         }
85 
86         try {
87             NoEscape dummy = new NoEscape();
88             Method m = DeoptReallocFailure.class.getMethod("test");
89             WB.enqueueMethodForCompilation(m, 4);
90             test();
91         } catch (OutOfMemoryError oom) {
92             root = null;
93             oom.printStackTrace();
94         }
95         System.out.println("TEST PASSED");
96     }
97 }
