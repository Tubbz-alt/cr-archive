24 /*
25  * @test
26  * @bug 8159620
27  * @summary testing that -XX:-UseOnStackReplacement works with both -XX:(+/-)TieredCompilation
28  * @modules java.base/jdk.internal.misc
29  * @library /test/lib /
30  *
31  * @build sun.hotspot.WhiteBox
32  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
33  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -XX:+PrintCompilation
34  *                   -XX:-BackgroundCompilation -XX:-TieredCompilation -XX:-UseOnStackReplacement
35  *                   compiler.interpreter.DisableOSRTest
36  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -XX:+PrintCompilation
37  *                   -XX:-BackgroundCompilation -XX:+TieredCompilation -XX:-UseOnStackReplacement
38  *                   compiler.interpreter.DisableOSRTest
39  */
40 
41 package compiler.interpreter;
42 
43 import sun.hotspot.WhiteBox;
44 
45 import java.lang.reflect.Method;
46 import java.util.Random;
47 
48 public class DisableOSRTest {
49     private static final WhiteBox WB = WhiteBox.getWhiteBox();
50     private static final Random RANDOM = new Random(42);
51 
52     public static int foo() {
53         return RANDOM.nextInt();
54     }
55 
56     public static void main(String[] args) throws Exception {
57         Method m = DisableOSRTest.class.getMethod("main", String[].class);
58 
59         for (int i = 0; i < 100_000; i++) {
60             foo();
61         }
62 
63         if (WB.isMethodCompiled(m, true /* isOsr */)) {
64             throw new RuntimeException("\"" + m + "\" shouldn't be OSR compiled if running with -XX:-UseOnStackReplacement!");
65         }
66     }
67 }
