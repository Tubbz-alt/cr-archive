51     /**
52      * Tests {@code WB::testSetDontInlineMethod()} by sequential calling it and
53      * checking of return value.
54      *
55      * @throws Exception if one of the checks fails.
56      */
57     @Override
58     protected void test() throws Exception {
59         if (WHITE_BOX.testSetDontInlineMethod(method, true)) {
60             throw new RuntimeException("on start " + method
61                     + " must be inlineable");
62         }
63         if (!WHITE_BOX.testSetDontInlineMethod(method, true)) {
64             throw new RuntimeException("after first change to true " + method
65                     + " must be not inlineable");
66         }
67         if (!WHITE_BOX.testSetDontInlineMethod(method, false)) {
68             throw new RuntimeException("after second change to true " + method
69                     + " must be still not inlineable");
70         }
71         if (WHITE_BOX.testSetDontInlineMethod(method, false)) {
72             throw new RuntimeException("after first change to false" + method
73                     + " must be inlineable");
74         }
75         if (WHITE_BOX.testSetDontInlineMethod(method, false)) {
76             throw new RuntimeException("after second change to false " + method
77                     + " must be inlineable");
78         }
79     }
