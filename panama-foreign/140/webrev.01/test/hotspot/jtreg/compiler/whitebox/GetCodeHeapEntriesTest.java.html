41 package compiler.whitebox;
42 
43 import jdk.test.lib.Asserts;
44 import sun.hotspot.WhiteBox;
45 import sun.hotspot.code.BlobType;
46 import sun.hotspot.code.CodeBlob;
47 
48 import java.util.Arrays;
49 import java.util.EnumSet;
50 
51 public class GetCodeHeapEntriesTest {
52     private static final WhiteBox WHITE_BOX = WhiteBox.getWhiteBox();
53     private static final int SIZE = 1024;
54     private static final String DUMMY_NAME = "WB::DummyBlob";
55     private static EnumSet<BlobType> SEGMENTED_TYPES
56             = EnumSet.complementOf(EnumSet.of(BlobType.All));
57 
58     public static void main(String[] args) {
59         EnumSet<BlobType> blobTypes = BlobType.getAvailable();
60         for (BlobType type : blobTypes) {
61             new GetCodeHeapEntriesTest(type).test();
62         }
63     }
64 
65     private final BlobType type;
66     private GetCodeHeapEntriesTest(BlobType type) {
67         this.type = type;
68     }
69 
70     private void test() {
71         System.out.printf("type %s%n", type);
72         long addr = WHITE_BOX.allocateCodeBlob(SIZE, type.id);
73         Asserts.assertNE(0, addr, "allocation failed");
74         CodeBlob[] blobs = CodeBlob.getCodeBlobs(type);
75         Asserts.assertNotNull(blobs);
76         CodeBlob blob = Arrays.stream(blobs)
77                               .filter(GetCodeHeapEntriesTest::filter)
78                               .findAny()
79                               .orElse(null);
80         Asserts.assertNotNull(blob);
81         Asserts.assertEQ(blob.code_blob_type, type);
82         Asserts.assertGTE(blob.size, SIZE);
83 
84         WHITE_BOX.freeCodeBlob(addr);
85         blobs = CodeBlob.getCodeBlobs(type);
86         long count = Arrays.stream(blobs)
87                            .filter(GetCodeHeapEntriesTest::filter)
88                            .count();
89         Asserts.assertEQ(0L, count);
90     }
91 
92     private static boolean filter(CodeBlob blob) {
93         if (blob == null) {
94             return false;
95         }
96         return DUMMY_NAME.equals(blob.name);
97     }
98 }
