52     /**
53      * Tests {@code WB::testSetDontInlineMethod()} by sequential calling it and
54      * checking of return value.
55      *
56      * @throws Exception if one of the checks fails.
57      */
58     @Override
59     protected void test() throws Exception {
60         if (WHITE_BOX.testSetDontInlineMethod(method, true)) {
61             throw new RuntimeException("on start " + method
62                     + " must be inlineable");
63         }
64         if (!WHITE_BOX.testSetDontInlineMethod(method, true)) {
65             throw new RuntimeException("after first change to true " + method
66                     + " must be not inlineable");
67         }
68         if (!WHITE_BOX.testSetDontInlineMethod(method, false)) {
69             throw new RuntimeException("after second change to true " + method
70                     + " must be still not inlineable");
71         }
72         if (WHITE_BOX.testSetDontInlineMethod(method, false)) {
73             throw new RuntimeException("after first change to false" + method
74                     + " must be inlineable");
75         }
76         if (WHITE_BOX.testSetDontInlineMethod(method, false)) {
77             throw new RuntimeException("after second change to false " + method
78                     + " must be inlineable");
79         }
80     }
