 35  *                   compiler.whitebox.EnqueueMethodForCompilationTest

 38 package compiler.whitebox;
 39 
 40 public class EnqueueMethodForCompilationTest extends CompilerWhiteBoxTest {
 41 
 42     public static void main(String[] args) throws Exception {
 43         String directive =
 44                 "[{ match:\"*SimpleTestCaseHelper.*\", BackgroundCompilation: false }, " +
 45                 " { match:\"*.*\", inline:\"-*SimpleTestCaseHelper.*\"}]";
 46         if (WHITE_BOX.addCompilerDirective(directive) != 2) {
 47             throw new RuntimeException("Could not add directive");
 48         }
 49         try {
 50             CompilerWhiteBoxTest.main(EnqueueMethodForCompilationTest::new, args);
 51         } finally {
 52             WHITE_BOX.removeCompilerDirective(2);
 53         }
 54     }
 55 
 56     private EnqueueMethodForCompilationTest(TestCase testCase) {
 57         super(testCase);
 58         // to prevent inlining of #method
 59         WHITE_BOX.testSetDontInlineMethod(method, true);
 60     }
 61 
 62     @Override
 63     protected void test() throws Exception {
 64         checkNotCompiled();
 65 
 66         // method can not be compiled on level 'none'
 67         WHITE_BOX.enqueueMethodForCompilation(method, COMP_LEVEL_NONE);
 68         if (isCompilable(COMP_LEVEL_NONE)) {
 69             throw new RuntimeException(method
 70                     + " is compilable at level COMP_LEVEL_NONE");
 71         }
 72         checkNotCompiled();
 73 
 74         // COMP_LEVEL_ANY is inapplicable as level for compilation
 75         WHITE_BOX.enqueueMethodForCompilation(method, COMP_LEVEL_ANY);
 76         checkNotCompiled();
 77 
 78         // not existing comp level
 79         WHITE_BOX.enqueueMethodForCompilation(method, 42);
 80         checkNotCompiled();
 81 
 82         compile();
 83         checkCompiled();
 84 
 85         int compLevel = getCompLevel();
 86         int bci = WHITE_BOX.getMethodEntryBci(method);
 87         deoptimize();
 88         checkNotCompiled();
 89         WHITE_BOX.clearMethodState(method);
 90 
 91         if (!WHITE_BOX.enqueueMethodForCompilation(method, compLevel, bci)) {
 92            throw new RuntimeException(method
 93                     + " could not be enqueued for compilation");
 94         }
 95         checkCompiled();
 96         deoptimize();
 97         checkNotCompiled();
 98 
 99         compile();
100         checkCompiled();
101         deoptimize();
102         checkNotCompiled();
103     }
104 }
