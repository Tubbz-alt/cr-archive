42 public class DeoptimizeMethodTest extends CompilerWhiteBoxTest {
43 
44     public static void main(String[] args) throws Exception {
45         CompilerWhiteBoxTest.main(DeoptimizeMethodTest::new, args);

47 
48     private DeoptimizeMethodTest(TestCase testCase) {
49         super(testCase);
50         // to prevent inlining of #method
51         WHITE_BOX.testSetDontInlineMethod(method, true);
52     }
53 
54     /**
55      * Tests {@code WB::deoptimizeMethod()} by calling it after
56      * compilation and checking that method isn't compiled.
57      *
58      * @throws Exception if one of the checks fails.
59      */
60     @Override
61     protected void test() throws Exception {
62         if (skipXcompOSR()) {
63             return;
64         }
65         compile();
66         checkCompiled();
67         deoptimize();
68         checkNotCompiled();
69     }
