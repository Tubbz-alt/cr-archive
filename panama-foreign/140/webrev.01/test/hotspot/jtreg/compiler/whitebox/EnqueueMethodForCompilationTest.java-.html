 39 package compiler.whitebox;
 40 
 41 public class EnqueueMethodForCompilationTest extends CompilerWhiteBoxTest {
 42 
 43     public static void main(String[] args) throws Exception {
 44         String directive =
 45                 "[{ match:\"*SimpleTestCaseHelper.*\", BackgroundCompilation: false }, " +
 46                 " { match:\"*.*\", inline:\"-*SimpleTestCaseHelper.*\"}]";
 47         if (WHITE_BOX.addCompilerDirective(directive) != 2) {
 48             throw new RuntimeException("Could not add directive");
 49         }
 50         try {
 51             CompilerWhiteBoxTest.main(EnqueueMethodForCompilationTest::new, args);
 52         } finally {
 53             WHITE_BOX.removeCompilerDirective(2);
 54         }
 55     }
 56 
 57     private EnqueueMethodForCompilationTest(TestCase testCase) {
 58         super(testCase);
 59         // to prevent inlining of #method
 60         WHITE_BOX.testSetDontInlineMethod(method, true);
 61     }
 62 
 63     @Override
 64     protected void test() throws Exception {
 65         checkNotCompiled();
 66 
 67         // method can not be compiled on level 'none'
 68         WHITE_BOX.enqueueMethodForCompilation(method, COMP_LEVEL_NONE);
 69         if (isCompilable(COMP_LEVEL_NONE)) {
 70             throw new RuntimeException(method
 71                     + " is compilable at level COMP_LEVEL_NONE");
 72         }
 73         checkNotCompiled();
 74 
 75         // COMP_LEVEL_ANY is inapplicable as level for compilation
 76         WHITE_BOX.enqueueMethodForCompilation(method, COMP_LEVEL_ANY);
 77         checkNotCompiled();
 78 
 79         // not existing comp level
 80         WHITE_BOX.enqueueMethodForCompilation(method, 42);
 81         checkNotCompiled();
 82 
 83         compile();
 84         checkCompiled();
 85 
 86         int compLevel = getCompLevel();
 87         int bci = WHITE_BOX.getMethodEntryBci(method);
 88         deoptimize();
 89         checkNotCompiled();
 90         WHITE_BOX.clearMethodState(method);
 91 
 92         if (!WHITE_BOX.enqueueMethodForCompilation(method, compLevel, bci)) {
 93            throw new RuntimeException(method
 94                     + " could not be enqueued for compilation");
 95         }
 96         checkCompiled();
 97         deoptimize();
 98         checkNotCompiled();
 99 
100         compile();
101         checkCompiled();
102         deoptimize();
103         checkNotCompiled();
104     }
105 }
