 41 package compiler.whitebox;
 42 
 43 import jdk.test.lib.Asserts;
 44 import sun.hotspot.code.BlobType;
 45 
 46 import java.util.EnumSet;
 47 
 48 public class ForceNMethodSweepTest extends CompilerWhiteBoxTest {
 49     public static void main(String[] args) throws Exception {
 50         CompilerWhiteBoxTest.main(ForceNMethodSweepTest::new, args);
 51     }
 52     private final EnumSet<BlobType> blobTypes;
 53     private ForceNMethodSweepTest(TestCase testCase) {
 54         super(testCase);
 55         // to prevent inlining of #method
 56         WHITE_BOX.testSetDontInlineMethod(method, true);
 57         blobTypes = BlobType.getAvailable();
 58     }
 59 
 60     @Override
 61     protected void test() throws Exception {
 62         // prime the asserts: get their bytecodes loaded, any lazy computation
 63         // resolved, and executed once
 64         Asserts.assertGT(1, 0, "message");
 65         Asserts.assertLTE(0, 0, "message");
 66         Asserts.assertLT(-1, 0, "message");
 67 
 68         checkNotCompiled();
 69         guaranteedSweep();
 70         int usage = getTotalUsage();
 71 
 72         compile();
 73         checkCompiled();
 74         int afterCompilation = getTotalUsage();
 75         Asserts.assertGT(afterCompilation, usage,
 76                 "compilation should increase usage");
 77 
 78         guaranteedSweep();
 79         int afterSweep = getTotalUsage();
 80         Asserts.assertLTE(afterSweep, afterCompilation,
 81                 "sweep shouldn't increase usage");
 82 
 83         deoptimize();
 84         guaranteedSweep();
 85         int afterDeoptAndSweep = getTotalUsage();
 86         Asserts.assertLT(afterDeoptAndSweep, afterSweep,
 87                 "sweep after deoptimization should decrease usage");
 88      }
 89 
 90     private int getTotalUsage() {
 91         int usage = 0;
 92         for (BlobType type : blobTypes) {
 93            usage += type.getMemoryPool().getUsage().getUsed();
 94         }
 95         return usage;
 96     }
 97     private void guaranteedSweep() {
 98         // not entrant -> ++stack_traversal_mark -> zombie -> flushed
 99         for (int i = 0; i < 5; ++i) {
100             WHITE_BOX.fullGC();
101             WHITE_BOX.forceNMethodSweep();
102         }
103     }
104 }
