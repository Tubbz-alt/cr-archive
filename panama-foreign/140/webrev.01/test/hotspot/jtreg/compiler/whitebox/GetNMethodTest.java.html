 39 package compiler.whitebox;
 40 
 41 import jdk.test.lib.Asserts;
 42 import sun.hotspot.code.BlobType;
 43 import sun.hotspot.code.NMethod;
 44 
 45 public class GetNMethodTest extends CompilerWhiteBoxTest {
 46     public static void main(String[] args) throws Exception {
 47         CompilerWhiteBoxTest.main(GetNMethodTest::new, args);
 48     }
 49 
 50     private GetNMethodTest(TestCase testCase) {
 51         super(testCase);
 52         // to prevent inlining of #method
 53         WHITE_BOX.testSetDontInlineMethod(method, true);
 54     }
 55 
 56     @Override
 57     protected void test() throws Exception {
 58         checkNotCompiled();
 59 
 60         compile();
 61         checkCompiled();
 62 
 63         NMethod nmethod = NMethod.get(method, testCase.isOsr());
 64         if (IS_VERBOSE) {
 65             System.out.println("nmethod = " + nmethod);
 66         }
 67         Asserts.assertNotNull(nmethod,
 68                 "nmethod of compiled method is null");
 69         Asserts.assertNotNull(nmethod.insts,
 70                 "nmethod.insts of compiled method is null");
 71         Asserts.assertGT(nmethod.insts.length, 0,
 72                 "compiled method's instructions is empty");
 73         Asserts.assertNotNull(nmethod.code_blob_type, "blob type is null");
 74         if (WHITE_BOX.getBooleanVMFlag("SegmentedCodeCache")) {
 75             Asserts.assertNE(nmethod.code_blob_type, BlobType.All);
 76             switch (nmethod.comp_level) {
 77             case 1:
 78             case 4:
 79                 checkBlockType(nmethod, BlobType.MethodNonProfiled);
 80                 break;
 81             case 2:
 82             case 3:
 83                 checkBlockType(nmethod, BlobType.MethodProfiled);
 84                 break;
 85             default:
 86                 throw new Error("unexpected comp level " + nmethod);
 87             }
 88         } else {
 89             Asserts.assertEQ(nmethod.code_blob_type, BlobType.All);
 90         }
 91 
 92         deoptimize();
 93         checkNotCompiled();
 94         nmethod = NMethod.get(method, testCase.isOsr());
 95         Asserts.assertNull(nmethod,
 96                 "nmethod of non-compiled method isn't null");
 97     }
 98 
 99     private void checkBlockType(NMethod nmethod, BlobType expectedType) {
100         Asserts.assertEQ(nmethod.code_blob_type, expectedType,
101                 String.format("blob_type[%s] for %d level isn't %s",
102                         nmethod.code_blob_type, nmethod.comp_level, expectedType));
103     }
104 }
