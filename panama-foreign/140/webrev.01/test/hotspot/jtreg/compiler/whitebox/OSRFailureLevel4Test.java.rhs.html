 26  * @summary check that not compilable OSR level 4 results in falling back to level 1

 33  *

 36  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
 37  *                   -XX:+WhiteBoxAPI -XX:+TieredCompilation compiler.whitebox.OSRFailureLevel4Test
 38  */

 40 package compiler.whitebox;
 41 
 42 import sun.hotspot.WhiteBox;
 43 import java.lang.reflect.Executable;
 44 import java.lang.reflect.Method;
 45 import jtreg.SkippedException;
 46 
 47 public class OSRFailureLevel4Test extends Thread {
 48     private static final WhiteBox WHITE_BOX = WhiteBox.getWhiteBox();
 49     private static final long BACKEDGE_THRESHOLD = 150000;
 50     private Method method;
 51 
 52     public static void main(String[] args) throws Exception {
 53         if (CompilerWhiteBoxTest.skipOnTieredCompilation(false)) {
 54             throw new SkippedException("Test isn't applicable for non-tiered mode");
 55         }
 56         OSRFailureLevel4Test test = new OSRFailureLevel4Test();
 57         test.test();
 58     }
 59 
 60     /**
 61      * Triggers two different OSR compilations for the same method and
 62      * checks if WhiteBox.deoptimizeMethod() deoptimizes both.
 63      *
 64      * @throws Exception
 65      */
 66     public void test() throws Exception {
 67         method = OSRFailureLevel4Test.class.getDeclaredMethod("run");
 68         WHITE_BOX.makeMethodNotCompilable(method, 4, true);
 69 
 70         Thread t = new OSRFailureLevel4Test();
 71         t.setDaemon(true);
 72         t.start();
 73 
 74         int currentLevel = 0;
 75         int loops = 0;
 76         while (true) {
 77             int level = WHITE_BOX.getMethodCompilationLevel(method, true);
 78             if (level == 1) {
 79                 System.err.println("success");
 80                 running = false;
 81                 return;
 82             }
 83             if (level == currentLevel) {
 84                 Thread.sleep(1000);
 85                 loops++;
 86                 if (loops > 100) {
 87                     running = false;
 88                     throw new AssertionError("Never reached level 1");
 89                 }
 90                 continue;
 91             }
 92             currentLevel = level;
 93             System.err.println("Current level = " + currentLevel);
 94         }
 95 
 96     }
 97 
 98     static volatile int counter = 0;
 99     static boolean running = true;
100 
101     public void run() {
102         while (running) {
103             counter++;
104         }
105     }
106 }
