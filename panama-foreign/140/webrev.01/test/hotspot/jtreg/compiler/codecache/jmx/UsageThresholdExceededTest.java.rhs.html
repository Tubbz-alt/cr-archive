24 /*

26  * @summary verifying that getUsageThresholdCount() returns correct value
27  *     after threshold has been hit
28  * @library /test/lib /
29  * @modules java.base/jdk.internal.misc
30  *          java.management
31  *
32  * @build sun.hotspot.WhiteBox

34  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
35  *     -XX:+WhiteBoxAPI -XX:-UseCodeCacheFlushing  -XX:-MethodFlushing
36  *     -XX:CompileCommand=compileonly,null::*
37  *     -XX:+SegmentedCodeCache
38  *     compiler.codecache.jmx.UsageThresholdExceededTest
39  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
40  *     -XX:+WhiteBoxAPI -XX:-UseCodeCacheFlushing  -XX:-MethodFlushing
41  *     -XX:CompileCommand=compileonly,null::*
42  *     -XX:-SegmentedCodeCache
43  *     compiler.codecache.jmx.UsageThresholdExceededTest
44  */
45 
46 package compiler.codecache.jmx;
47 
48 import sun.hotspot.code.BlobType;
49 
50 import java.lang.management.MemoryPoolMXBean;
51 
52 public class UsageThresholdExceededTest {
53 
54     protected final int iterations;
55     private final BlobType btype;
56 
57     public UsageThresholdExceededTest(BlobType btype, int iterations) {
58         this.btype = btype;
59         this.iterations = iterations;
60     }
61 
62     public static void main(String[] args) {
63         int iterationsCount = Integer.getInteger("jdk.test.lib.iterations", 1);
64         for (BlobType btype : BlobType.getAvailable()) {
65             if (CodeCacheUtils.isCodeHeapPredictable(btype)) {
66                 new UsageThresholdExceededTest(btype, iterationsCount).runTest();
67             }
68         }
69     }
70 
71     protected void runTest() {
72         MemoryPoolMXBean bean = btype.getMemoryPool();
73         long oldValue = bean.getUsageThresholdCount();
74         for (int i = 0; i < iterations; i++) {
75             CodeCacheUtils.hitUsageThreshold(bean, btype);
76         }
77         CodeCacheUtils.assertEQorGTE(btype, bean.getUsageThresholdCount(), oldValue + iterations,
78                 "Unexpected threshold usage count");
79         System.out.printf("INFO: Scenario finished successfully for %s%n", bean.getName());
80     }
81 }
