43 package compiler.codecache.jmx;
44 
45 import jdk.test.lib.Asserts;
46 import sun.hotspot.code.BlobType;
47 
48 import java.util.EnumSet;
49 
50 public class CodeHeapBeanPresenceTest {
51 
52     public static void main(String args[]) {
53         EnumSet<BlobType> shouldBeAvailable = BlobType.getAvailable();
54         EnumSet<BlobType> shouldNotBeAvailable
55                 = EnumSet.complementOf(shouldBeAvailable);
56         for (BlobType btype : shouldBeAvailable) {
57             Asserts.assertNotNull(btype.getMemoryPool(),
58                     "Can't find memory pool for " + btype.name());
59         }
60         for (BlobType btype : shouldNotBeAvailable) {
61             Asserts.assertNull(btype.getMemoryPool(),
62                     "Memory pool unexpected for " + btype.name());
63         }
64     }
65 }
