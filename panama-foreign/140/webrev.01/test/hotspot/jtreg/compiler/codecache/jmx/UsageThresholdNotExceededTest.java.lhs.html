26  * @summary verifying that usage threshold not exceeded while allocating less
27  *     than usage threshold
28  * @library /test/lib /
29  * @modules java.base/jdk.internal.misc
30  *          java.management
31  *
32  * @build sun.hotspot.WhiteBox
33  * @run driver ClassFileInstaller sun.hotspot.WhiteBox

35  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
36  *     -XX:+WhiteBoxAPI -XX:-UseCodeCacheFlushing -XX:-MethodFlushing
37  *     -XX:CompileCommand=compileonly,null::*
38  *     -XX:+SegmentedCodeCache
39  *     compiler.codecache.jmx.UsageThresholdNotExceededTest
40  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
41  *     -XX:+WhiteBoxAPI -XX:-UseCodeCacheFlushing -XX:-MethodFlushing
42  *     -XX:CompileCommand=compileonly,null::*
43  *     -XX:-SegmentedCodeCache
44  *     compiler.codecache.jmx.UsageThresholdNotExceededTest
45  */
46 
47 package compiler.codecache.jmx;
48 
49 import sun.hotspot.code.BlobType;
50 
51 import java.lang.management.MemoryPoolMXBean;
52 
53 public class UsageThresholdNotExceededTest {
54 
55     private final BlobType btype;
56 
57     public UsageThresholdNotExceededTest(BlobType btype) {
58         this.btype = btype;
59     }
60 
61     public static void main(String[] args) {
62         for (BlobType btype : BlobType.getAvailable()) {
63             new UsageThresholdNotExceededTest(btype).runTest();
64         }
65     }
66 
67     protected void runTest() {
68         MemoryPoolMXBean bean = btype.getMemoryPool();
69         long initialThresholdCount = bean.getUsageThresholdCount();
70         long initialUsage = bean.getUsage().getUsed();
71 
72         bean.setUsageThreshold(initialUsage + 1 + CodeCacheUtils.MIN_ALLOCATION);
73         long size = CodeCacheUtils.getHeaderSize(btype);
74 
75         CodeCacheUtils.WB.allocateCodeBlob(Math.max(0, CodeCacheUtils.MIN_ALLOCATION
76                 - size), btype.id);
77         // a gc cycle triggers usage threshold recalculation
78         CodeCacheUtils.WB.fullGC();
79         CodeCacheUtils.assertEQorGTE(btype, bean.getUsageThresholdCount(), initialThresholdCount,
80                 String.format("Usage threshold was hit: %d times for %s. "
81                         + "Threshold value: %d with current usage: %d",
82                         bean.getUsageThresholdCount(), bean.getName(),
83                         bean.getUsageThreshold(), bean.getUsage().getUsed()));
84         System.out.println("INFO: Case finished successfully for " + bean.getName());
85     }
86 }
