24 /**
25  * @test ManagerNamesTest
26  * @summary verify getMemoryManageNames calls in case of segmented code cache
27  * @modules java.base/jdk.internal.misc
28  *          java.management
29  * @library /test/lib
30  *
31  * @build sun.hotspot.WhiteBox
32  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
33  *                                sun.hotspot.WhiteBox$WhiteBoxPermission
34  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
35  *     -XX:+WhiteBoxAPI
36  *     -XX:+SegmentedCodeCache
37  *     compiler.codecache.jmx.ManagerNamesTest
38  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
39  *     -XX:+WhiteBoxAPI
40  *     -XX:-SegmentedCodeCache
41  *     compiler.codecache.jmx.ManagerNamesTest
42  */
43 
44 package compiler.codecache.jmx;
45 
46 import jdk.test.lib.Asserts;
47 import sun.hotspot.code.BlobType;
48 
49 import java.lang.management.MemoryPoolMXBean;
50 
51 public class ManagerNamesTest {
52 
53     private final MemoryPoolMXBean bean;
54     private final static String POOL_NAME = "CodeCacheManager";
55 
56     public static void main(String args[]) {
57         for (BlobType btype : BlobType.getAvailable()) {
58             new ManagerNamesTest(btype).runTest();
59         }
60     }
61 
62     public ManagerNamesTest(BlobType btype) {
63         bean = btype.getMemoryPool();
64     }
65 
66     protected void runTest() {
67         String[] names = bean.getMemoryManagerNames();
68         Asserts.assertEQ(names.length, 1,
69                 "Unexpected length of MemoryManagerNames");
70         Asserts.assertEQ(POOL_NAME, names[0],
71                 "Unexpected value of MemoryManagerName");
72         System.out.printf("INFO: Scenario finished successfully for %s%n",
73                 bean.getName());
74     }
75 }
