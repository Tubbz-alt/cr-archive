25 /**
26  * @test TestOnSpinWaitC1
27  * @summary (x86 only) checks that java.lang.Thread.onSpinWait is intrinsified
28  * @bug 8147844
29  * @library /test/lib
30  * @modules java.base/jdk.internal.misc
31  * @requires os.arch=="x86" | os.arch=="amd64" | os.arch=="x86_64"
32  * @requires vm.compiler1.enabled
33  * @run driver compiler.onSpinWait.TestOnSpinWait
34  */
35 
36 package compiler.onSpinWait;
37 
38 import jdk.test.lib.process.OutputAnalyzer;
39 import jdk.test.lib.process.ProcessTools;
40 
41 public class TestOnSpinWaitC1 {
42 
43     public static void main(String[] args) throws Exception {
44 
45         // Test C1 compiler
46         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
47           "-XX:+IgnoreUnrecognizedVMOptions", "-showversion",
48           "-XX:+TieredCompilation", "-XX:TieredStopAtLevel=1", "-Xbatch",
49           "-XX:+PrintCompilation", "-XX:+UnlockDiagnosticVMOptions",
50           "-XX:+PrintInlining", Launcher.class.getName());
51 
52         OutputAnalyzer analyzer = new OutputAnalyzer(pb.start());
53 
54         analyzer.shouldHaveExitValue(0);
55         analyzer.shouldContain("java.lang.Thread::onSpinWait (1 bytes)   intrinsic");
56     }
57 
58     static class Launcher {
59 
60         public static void main(final String[] args) throws Exception {
61             int end = 20_000;
62 
63             for (int i=0; i < end; i++) {
64                 test();
65             }
66         }
67         static void test() {
68             java.lang.Thread.onSpinWait();
69         }
70     }
71 }
