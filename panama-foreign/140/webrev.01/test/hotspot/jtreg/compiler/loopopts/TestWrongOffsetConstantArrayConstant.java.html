24 /*
25  * @test
26  * @bug 8239569
27  * @summary Phi of AddP transformation can cause offset in constant array to be dropped
28  *
29  * @run main/othervm -XX:-BackgroundCompilation TestWrongOffsetConstantArrayConstant
30  */
31 
32 public class TestWrongOffsetConstantArrayConstant {
33     private static volatile char volatileField;
34 
35     public static void main(String[] args) {
36         for (int i = 0; i < 20_000; i++) {
37             if (test('o') != 'o') {
38                 throw new RuntimeException("bad result");
39             }
40         }
41     }
42 
43 
44     static final byte[] str = {'f', 'o'};
45 
46     private static char test(char b1) {
47         // Trigger extra rounds of loop opts
48         for (int i = 0; i < 3; i++) {
49             for (int k = 0; k < 3; k++) {
50                 for (int l = 0; l < 3; l++) {
51 
52                 }
53             }
54         }
55         // Will fully unroll
56         int i = 0;
57         for (; i < str.length; i++) {
58             final byte b = str[i]; // LoadB
59             if (b == b1) {
60                 break;
61             }
62         }
63         // LoadUB that shares address with LoadB above
64         // Will split thru phi
65         final char c = (char) (str[i] & 0xff);
66         volatileField = c;
67         // LoadUB that shares address with LoadB and LoadUB above
68         final char c2 = (char) (str[i] & 0xff);
69         return c2;
70     }
71 }
