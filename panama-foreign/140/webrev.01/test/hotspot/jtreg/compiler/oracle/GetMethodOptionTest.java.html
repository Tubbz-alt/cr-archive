 44 package compiler.oracle;
 45 
 46 import jdk.test.lib.Asserts;
 47 import sun.hotspot.WhiteBox;
 48 
 49 import java.lang.reflect.Executable;
 50 import java.util.function.BiFunction;
 51 
 52 public class GetMethodOptionTest {
 53     private static final  WhiteBox WB = WhiteBox.getWhiteBox();
 54     public static void main(String[] args) {
 55         Executable test = getMethod("test");
 56         Executable test2 = getMethod("test2");
 57         BiFunction<Executable, String, Object> getter = WB::getMethodOption;
 58         for (TestCase testCase : TestCase.values()) {
 59             Object expected = testCase.value;
 60             String name = testCase.name();
 61             Asserts.assertEQ(expected, getter.apply(test, name),
 62                     testCase + ": universal getter returns wrong value");
 63             Asserts.assertEQ(expected, testCase.getter.apply(test, name),
 64                     testCase + ": specific getter returns wrong value");
 65             Asserts.assertEQ(null, getter.apply(test2, name),
 66                     testCase + ": universal getter returns value for unused method");
 67             Asserts.assertEQ(null, testCase.getter.apply(test2, name),
 68                     testCase + ": type specific getter returns value for unused method");
 69         }
 70     }
 71     private static void test() { }
 72     private static void test2() { }
 73 
 74     private static enum TestCase {
 75         MyListOption("_foo _bar", WB::getMethodStringOption),
 76         MyStrOption("_foo", WB::getMethodStringOption),
 77         MyBoolOption(false, WB::getMethodBooleanOption),
 78         MyIntxOption(-1L, WB::getMethodIntxOption),
 79         MyUintxOption(1L, WB::getMethodUintxOption),
 80         MyFlag(true, WB::getMethodBooleanOption),
 81         MyDoubleOption1(1.123d, WB::getMethodDoubleOption),
 82         MyDoubleOption2(1.123d, WB::getMethodDoubleOption),
 83         MyBoolOptionX(false, WB::getMethodBooleanOption),
 84         MyIntxOptionX(-1L, WB::getMethodIntxOption),
 85         MyUintxOptionX(1L, WB::getMethodUintxOption),
 86         MyFlagX(true, WB::getMethodBooleanOption),
 87         MyDoubleOptionX(1.123d, WB::getMethodDoubleOption);
 88 
 89         public final Object value;
 90         public final BiFunction<Executable, String, Object> getter;
 91         private TestCase(Object value, BiFunction<Executable, String, Object> getter) {
 92             this.value = value;
 93             this.getter = getter;
 94         }
 95     }
 96 
 97     private static Executable getMethod(String name) {
 98         Executable result;
 99         try {
100             result = GetMethodOptionTest.class.getDeclaredMethod(name);
101         } catch (NoSuchMethodException | SecurityException e) {
102             throw new Error("TESTBUG : can't get method " + name, e);
103         }
104         return result;
105     }
106 }
