48 package compiler.jvmci.compilerToVM;
49 
50 import compiler.jvmci.common.CTVMUtilities;
51 import jdk.test.lib.Asserts;
52 import jdk.vm.ci.hotspot.CompilerToVMHelper;
53 import jdk.vm.ci.hotspot.HotSpotResolvedJavaMethod;
54 import sun.hotspot.WhiteBox;
55 
56 import java.lang.reflect.Executable;
57 import java.util.ArrayList;
58 import java.util.Arrays;
59 import java.util.List;
60 
61 public class ShouldInlineMethodTest {
62 
63     private static final WhiteBox WB = WhiteBox.getWhiteBox();
64 
65     public static void main(String[] args) {
66         List<Executable> testCases = createTestCases();
67         testCases.forEach(ShouldInlineMethodTest::runSanityTest);
68     }
69 
70     private static void runSanityTest(Executable aMethod) {
71         HotSpotResolvedJavaMethod method = CTVMUtilities
72                 .getResolvedMethod(aMethod);
73         boolean shouldInline = CompilerToVMHelper.shouldInlineMethod(method);
74         boolean expectedShouldInline = WB.testSetForceInlineMethod(aMethod,
75                 true);
76         Asserts.assertEQ(shouldInline, expectedShouldInline,
77                 "Unexpected value of property 'should inline'");
78 
79         shouldInline = CompilerToVMHelper.shouldInlineMethod(method);
80         Asserts.assertTrue(shouldInline, "Unexpected value of property "
81                 + "'should inline' after setting 'force inline' to true");
82         WB.testSetForceInlineMethod(aMethod, false);
83         shouldInline = CompilerToVMHelper.shouldInlineMethod(method);
84         Asserts.assertFalse(shouldInline, "Unexpected value of property "
85                 + "'should inline' after setting 'force inline' to false");
86     }
87 
88     private static List<Executable> createTestCases() {
89         List<Executable> testCases = new ArrayList<>();
90 
91         Class<?> aClass = DummyClass.class;
92         testCases.addAll(Arrays.asList(aClass.getDeclaredMethods()));
93         testCases.addAll(Arrays.asList(aClass.getDeclaredConstructors()));
94         return testCases;
95     }
96 }
