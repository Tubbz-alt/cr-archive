24 /*
25  * @test TestBasicLogOutput
26  * @bug 8203370
27  * @summary Ensure -XX:-JVMCIPrintProperties can be enabled and successfully prints expected output to stdout.
28  * @requires vm.jvmci & !vm.graal.enabled & vm.compMode == "Xmixed"
29  * @library /test/lib
30  */
31 
32 import jdk.test.lib.process.ProcessTools;
33 import jdk.test.lib.process.OutputAnalyzer;
34 
35 public class TestJVMCIPrintProperties {
36 
37     public static void main(String[] args) throws Exception {
38         test("-XX:+EnableJVMCI");
39         test("-XX:+UseJVMCICompiler");
40     }
41 
42     static void test(String enableFlag) throws Exception {
43         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
44             "-XX:+UnlockExperimentalVMOptions",
45             enableFlag, "-Djvmci.Compiler=null",
46             "-XX:+JVMCIPrintProperties");
47         OutputAnalyzer output = new OutputAnalyzer(pb.start());
48         output.shouldContain("[JVMCI properties]"); // expected message
49         output.shouldContain("jvmci.Compiler := \"null\""); // expected message
50         output.shouldContain("jvmci.InitTimer = false"); // expected message
51         output.shouldContain("jvmci.PrintConfig = false"); // expected message
52         output.shouldContain("jvmci.TraceMethodDataFilter = null"); // expected message
53         output.shouldHaveExitValue(0);
54     }
55 }
