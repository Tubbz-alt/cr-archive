 63 /**
 64  * Test for {@code jdk.vm.ci.hotspot.CompilerToVM.lookupKlassRefIndexInPool} method
 65  */
 66 public class LookupKlassRefIndexInPoolTest {
 67 
 68     public static void main(String[] args) throws Exception {
 69         Map<ConstantTypes, Validator> typeTests = new HashMap<>();
 70         typeTests.put(CONSTANT_METHODREF, LookupKlassRefIndexInPoolTest::validate);
 71         typeTests.put(CONSTANT_INTERFACEMETHODREF, LookupKlassRefIndexInPoolTest::validate);
 72         typeTests.put(CONSTANT_FIELDREF, LookupKlassRefIndexInPoolTest::validate);
 73         ConstantPoolTestCase testCase = new ConstantPoolTestCase(typeTests);
 74         testCase.test();
 75         // The next "Class.forName" and repeating "testCase.test()"
 76         // are here for the following reason.
 77         // The first test run is without dummy class initialization,
 78         // which means no constant pool cache exists.
 79         // The second run is with initialized class (with constant pool cache available).
 80         // Some CompilerToVM methods require different input
 81         // depending on whether CP cache exists or not.
 82         for (DummyClasses dummy : DummyClasses.values()) {
 83             Class.forName(dummy.klass.getName());
 84         }
 85         testCase.test();
 86     }
 87 
 88     private static void validate(ConstantPool constantPoolCTVM,
 89                                  ConstantTypes cpType,
 90                                  DummyClasses dummyClass,
 91                                  int cpi) {
 92         TestedCPEntry entry = cpType.getTestedCPEntry(dummyClass, cpi);
 93         if (entry == null) {
 94             return;
 95         }
 96         int index = cpi;
 97         String cached = "";
 98         int cpci = dummyClass.getCPCacheIndex(cpi);
 99         if (cpci != ConstantPoolTestsHelper.NO_CP_CACHE_PRESENT) {
100             index = cpci;
101             cached = "cached ";
102         }
103         int indexToVerify = CompilerToVMHelper.lookupKlassRefIndexInPool(constantPoolCTVM, index);
104         int indexToRefer = dummyClass.constantPoolSS.getClassRefIndexAt(cpi);
105         String msg = String.format("Wrong class index returned by lookupKlassRefIndexInPool method "
106                                            + "applied to %sconstant pool index %d",
107                                    cached,
108                                    index);
109         Asserts.assertEQ(indexToRefer, indexToVerify, msg);
110     }
111 }
