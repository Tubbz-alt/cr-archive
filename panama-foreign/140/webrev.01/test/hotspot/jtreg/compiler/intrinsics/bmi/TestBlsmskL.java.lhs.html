24 /**
25  * @test
26  * @key randomness
27  * @bug 8031321
28  * @summary Verify that results of computations are the same w/
29  *          and w/o usage of BLSMSK instruction
30  * @library /test/lib /
31  * @modules java.base/jdk.internal.misc
32  *          java.management
33  *
34  * @build sun.hotspot.WhiteBox
35  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
36  *                                sun.hotspot.WhiteBox$WhiteBoxPermission
37  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
38  *                   -XX:+WhiteBoxAPI
39  *                   compiler.intrinsics.bmi.TestBlsmskL
40  */
41 
42 package compiler.intrinsics.bmi;
43 
44 import sun.hotspot.cpuinfo.CPUInfo;
45 
46 public class TestBlsmskL {
47 
48     public static void main(String args[]) throws Throwable {
49         if (!CPUInfo.hasFeature("bmi1")) {
50             System.out.println("INFO: CPU does not support bmi1 feature.");
51         }
52 
53         BMITestRunner.runTests(BlsmskLExpr.class, args,
54                                "-XX:+IgnoreUnrecognizedVMOptions",
55                                "-XX:+UseBMI1Instructions");
56         BMITestRunner.runTests(BlsmskLCommutativeExpr.class, args,
57                                "-XX:+IgnoreUnrecognizedVMOptions",
58                                "-XX:+UseBMI1Instructions");
59     }
60 
61     public static class BlsmskLExpr
62         extends Expr.BMIUnaryLongExpr {
63 
64         public long longExpr(long src) {
65             return (src - 1) ^ src;
66         }
67 
68         public long longExpr(Expr.MemL src) {
69             return (src.value - 1) ^ src.value;
70         }
71 
72     }
73 
74     public static class BlsmskLCommutativeExpr
75         extends Expr.BMIUnaryLongExpr {
76 
77         public long longExpr(long src) {
78             return src ^ (src - 1);
79         }
80 
81         public long longExpr(Expr.MemL src) {
82             return src.value ^ (src.value - 1);
83         }
84 
85     }
86 
87 }
