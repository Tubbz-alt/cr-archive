39 package compiler.intrinsics.bmi.verifycode;
40 import compiler.intrinsics.bmi.TestBlsmskI;
41 
42 import java.lang.reflect.Method;
43 
44 public class BlsmskTestI extends BmiIntrinsicBase.BmiTestCase {
45 
46     protected BlsmskTestI(Method method) {
47         super(method);
48         //from intel manual VEX.NDD.LZ.0F38.W0 F3 /2
49         instrMask = new byte[]{
50                 (byte) 0xFF,
51                 (byte) 0x1F,
52                 (byte) 0x00,
53                 (byte) 0xFF,
54                 (byte) 0b0011_1000};
55         instrPattern = new byte[]{
56                 (byte) 0xC4, // prefix for 3-byte VEX instruction
57                 (byte) 0x02, // 00010 implied 0F 38 leading opcode bytes
58                 (byte) 0x00,
59                 (byte) 0xF3,
60                 (byte) 0b0001_0000}; // bits 543 == 011 (3)
61     }
62 
63     public static void main(String[] args) throws Exception {
64         BmiIntrinsicBase.verifyTestCase(BlsmskTestI::new, TestBlsmskI.BlsmskIExpr.class.getDeclaredMethods());
65         BmiIntrinsicBase.verifyTestCase(BlsmskTestI::new, TestBlsmskI.BlsmskICommutativeExpr.class.getDeclaredMethods());
66     }
67 }
