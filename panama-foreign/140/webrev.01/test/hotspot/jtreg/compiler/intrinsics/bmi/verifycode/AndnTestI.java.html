39 package compiler.intrinsics.bmi.verifycode;
40 
41 import compiler.intrinsics.bmi.TestAndnI;
42 
43 import java.lang.reflect.Method;
44 
45 public class AndnTestI extends BmiIntrinsicBase.BmiTestCase {
46 
47     protected AndnTestI(Method method) {
48         super(method);
49         // from intel manual VEX.NDS.LZ.0F38.W0 F2 /r, example c4e260f2c2
50         instrMask = new byte[]{
51                 (byte) 0xFF,
52                 (byte) 0x1F,
53                 (byte) 0x00,
54                 (byte) 0xFF};
55         instrPattern = new byte[]{
56                 (byte) 0xC4, // prefix for 3-byte VEX instruction
57                 (byte) 0x02, // 00010 implied 0F 38 leading opcode bytes
58                 (byte) 0x00,
59                 (byte) 0xF2};
60     }
61 
62     public static void main(String[] args) throws Exception {
63         BmiIntrinsicBase.verifyTestCase(AndnTestI::new, TestAndnI.AndnIExpr.class.getDeclaredMethods());
64         BmiIntrinsicBase.verifyTestCase(AndnTestI::new, TestAndnI.AndnICommutativeExpr.class.getDeclaredMethods());
65     }
66 }
