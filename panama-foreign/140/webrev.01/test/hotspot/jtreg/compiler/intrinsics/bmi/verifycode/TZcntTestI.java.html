36  *      compiler.intrinsics.bmi.verifycode.TZcntTestI

38 package compiler.intrinsics.bmi.verifycode;
39 
40 import compiler.intrinsics.bmi.TestTzcntI;
41 
42 import java.lang.reflect.Method;
43 
44 public class TZcntTestI extends BmiIntrinsicBase.BmiTestCase_x64 {
45 
46     protected TZcntTestI(Method method) {
47         super(method);
48         instrMask = new byte[]{(byte) 0xFF, (byte) 0xFF, (byte) 0xFF};
49         instrPattern = new byte[]{(byte) 0xF3, (byte) 0x0F, (byte) 0xBC};
50 
51         instrMask_x64 = new byte[]{(byte) 0xFF, (byte) 0x00, (byte) 0xFF, (byte) 0xFF};
52         instrPattern_x64 = new byte[]{(byte) 0xF3, (byte) 0x00, (byte) 0x0F, (byte) 0xBC};
53     }
54 
55     public static void main(String[] args) throws Exception {
56         // j.l.Integer and Long should be loaded to allow a compilation of the methods that use their methods
57         System.out.println("class java.lang.Integer should be loaded. Proof: " + Integer.class);
58         // Avoid uncommon traps.
59         System.out.println("Num trailing zeroes: " + new TestTzcntI.TzcntIExpr().intExpr(12341341));
60         BmiIntrinsicBase.verifyTestCase(TZcntTestI::new, TestTzcntI.TzcntIExpr.class.getDeclaredMethods());
61     }
62 
63     @Override
64     protected String getVMFlag() {
65         return "UseCountTrailingZerosInstruction";
66     }
67 }
