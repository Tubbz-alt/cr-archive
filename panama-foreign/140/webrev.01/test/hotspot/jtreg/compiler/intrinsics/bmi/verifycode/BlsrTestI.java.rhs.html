39 package compiler.intrinsics.bmi.verifycode;
40 
41 import compiler.intrinsics.bmi.TestBlsrI;
42 
43 import java.lang.reflect.Method;
44 
45 public class BlsrTestI extends BmiIntrinsicBase.BmiTestCase {
46 
47     protected BlsrTestI(Method method) {
48         super(method);
49         //from intel manual VEX.NDD.LZ.0F38.W0 F3 /1
50         instrMask = new byte[]{
51                 (byte) 0xFF,
52                 (byte) 0x1F,
53                 (byte) 0x00,
54                 (byte) 0xFF,
55                 (byte) 0b0011_1000};
56         instrPattern = new byte[]{
57                 (byte) 0xC4, // prefix for 3-byte VEX instruction
58                 (byte) 0x02, // 00010 implied 0F 38 leading opcode bytes
59                 (byte) 0x00,
60                 (byte) 0xF3,
61                 (byte) 0b0000_1000}; // bits 543 == 011 (3)
62     }
63 
64     public static void main(String[] args) throws Exception {
65         BmiIntrinsicBase.verifyTestCase(BlsrTestI::new, TestBlsrI.BlsrIExpr.class.getDeclaredMethods());
66         BmiIntrinsicBase.verifyTestCase(BlsrTestI::new, TestBlsrI.BlsrICommutativeExpr.class.getDeclaredMethods());
67     }
68 }
