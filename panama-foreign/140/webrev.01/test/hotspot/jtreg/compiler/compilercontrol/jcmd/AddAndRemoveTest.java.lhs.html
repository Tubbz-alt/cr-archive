38 package compiler.compilercontrol.jcmd;
39 
40 import compiler.compilercontrol.share.AbstractTestBase;
41 import compiler.compilercontrol.share.method.MethodDescriptor;
42 import compiler.compilercontrol.share.scenario.Command;
43 import compiler.compilercontrol.share.scenario.CompileCommand;
44 import compiler.compilercontrol.share.scenario.JcmdCommand;
45 import compiler.compilercontrol.share.scenario.Scenario;
46 import jdk.test.lib.Utils;
47 
48 import java.lang.reflect.Executable;
49 
50 public class AddAndRemoveTest extends AbstractTestBase {
51     private static final int AMOUNT = Integer.getInteger(
52             "compiler.compilercontrol.jcmd.AddAndRemoveTest.amount", 10);
53 
54     public static void main(String[] args) {
55         new AddAndRemoveTest().test();
56     }
57 
58     @Override
59     public void test() {
60         Scenario.Builder builder = Scenario.getBuilder();
61         // Add some commands with JCMD
62         for (int i = 0; i < AMOUNT; i++) {
63             Executable exec = Utils.getRandomElement(METHODS).first;
64             MethodDescriptor md = getValidMethodDescriptor(exec);
65             CompileCommand compileCommand = new JcmdCommand(Command.COMPILEONLY,
66                     md, null, Scenario.Type.JCMD, Scenario.JcmdType.ADD);
67             compileCommand.print();
68             builder.add(compileCommand);
69         }
70         // Remove half of them
71         for (int i = 0; i < AMOUNT / 2; i++) {
72             /* remove jcmd command doesn't need method, compiler etc.
73                command will be ignored */
74             builder.add(new JcmdCommand(Command.NONEXISTENT, null, null,
75                     Scenario.Type.JCMD, Scenario.JcmdType.REMOVE));
76         }
77         Scenario scenario = builder.build();
78         scenario.execute();
79     }
80 }
