38  * @summary Test compiler control compatibility with compile command

40 
41 package compiler.compilercontrol;
42 
43 import jdk.test.lib.dcmd.CommandExecutor;
44 
45 public class TestCompilerDirectivesCompatibilityFlag extends TestCompilerDirectivesCompatibilityBase {
46 
47     public void testCompatibility(CommandExecutor executor, int comp_level) throws Exception {
48 
49         // Call all validation twice to catch error when overwriting a directive
50         // Flag is default on
51         expect(WB.getBooleanVMFlag("PrintAssembly"));
52         expect(WB.shouldPrintAssembly(method, comp_level));
53         expect(WB.shouldPrintAssembly(nomatch, comp_level));
54         expect(WB.shouldPrintAssembly(method, comp_level));
55         expect(WB.shouldPrintAssembly(nomatch, comp_level));
56 
57         // load directives that turn it off
58         executor.execute("Compiler.directives_add " + control_off);
59         expect(!WB.shouldPrintAssembly(method, comp_level));
60         expect(WB.shouldPrintAssembly(nomatch, comp_level));
61         expect(!WB.shouldPrintAssembly(method, comp_level));
62         expect(WB.shouldPrintAssembly(nomatch, comp_level));
63 
64         // remove and see that it is true again
65         executor.execute("Compiler.directives_remove");
66         expect(WB.shouldPrintAssembly(method, comp_level));
67         expect(WB.shouldPrintAssembly(nomatch, comp_level));
68         expect(WB.shouldPrintAssembly(method, comp_level));
69         expect(WB.shouldPrintAssembly(nomatch, comp_level));
70     }
71 }
