24 /**
25  * @test
26  * @requires vm.cds
27  * @bug 8067187 8200078
28  * @summary Testing CDS dumping with the -XX:MaxMetaspaceSize=<size> option
29  * @library /test/lib
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  */
33 
34 import java.util.ArrayList;
35 
36 import jdk.test.lib.cds.CDSTestUtils;
37 import jdk.test.lib.process.ProcessTools;
38 import jdk.test.lib.Platform;
39 
40 public class MaxMetaspaceSize {
41   public static void main(String[] args) throws Exception {
42     ArrayList<String> processArgs = new ArrayList<>();
43     processArgs.add("-Xshare:dump");
44 
45     if (Platform.is64bit()) {
46       processArgs.add("-XX:MaxMetaspaceSize=3m");
47       processArgs.add("-XX:CompressedClassSpaceSize=1m");
48       processArgs.add("-XX:InitialBootClassLoaderMetaspaceSize=1m");
49     } else {
50       processArgs.add("-XX:MaxMetaspaceSize=1m");
51     }
52 
53     String msg = "Failed allocating metaspace object";
54     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(processArgs.toArray(new String[0]));
55     CDSTestUtils.executeAndLog(pb, "dump").shouldContain(msg).shouldHaveExitValue(1);
56   }
57 }
