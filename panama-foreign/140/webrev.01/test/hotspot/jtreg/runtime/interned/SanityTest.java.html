24 /*
25  * @test SanityTest
26  * @summary Sanity check of String.intern() & GC
27  * @modules java.base/jdk.internal.misc
28  * @library /test/lib
29  * @build sun.hotspot.WhiteBox
30  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
31  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI SanityTest
32  */
33 
34 import java.util.*;
35 import sun.hotspot.WhiteBox;
36 
37 
38 public class SanityTest {
39     public static Object tmp;
40     public static void main(String... args) {
41 
42         WhiteBox wb = WhiteBox.getWhiteBox();
43         StringBuilder sb = new StringBuilder();
44         sb.append("1234x"); sb.append("x56789");
45         String str = sb.toString();
46 
47         if (wb.isInStringTable(str)) {
48             throw new RuntimeException("String " + str + " is already interned");
49         }
50         str.intern();
51         if (!wb.isInStringTable(str)) {
52             throw new RuntimeException("String " + str + " is not interned");
53         }
54         str = sb.toString();
55         wb.fullGC();
56         if (wb.isInStringTable(str)) {
57             throw new RuntimeException("String " + str + " is in StringTable even after GC");
58         }
59     }
60 }
