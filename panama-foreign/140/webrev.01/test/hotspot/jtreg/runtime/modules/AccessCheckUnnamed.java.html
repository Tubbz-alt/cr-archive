24 import static jdk.test.lib.Asserts.*;
25 
26 /*
27  * @test
28  * @modules java.base/jdk.internal.misc
29  * @library /test/lib ..
30  * @compile p2/c2.java
31  * @compile p1/c1.java
32  * @build sun.hotspot.WhiteBox
33  * @compile/module=java.base java/lang/ModuleHelper.java
34  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
35  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI AccessCheckUnnamed
36  */
37 
38 public class AccessCheckUnnamed {
39 
40     // Test that a class in the unnamed module can not access a package in a
41     // named module that has not been unqualifiedly exported.
42     public static void main(String args[]) throws Throwable {
43         Object m1x, m2x;
44 
45         // Get the java.lang.Module object for module java.base.
46         Class jlObject = Class.forName("java.lang.Object");
47         Object jlObject_jlM = jlObject.getModule();
48         assertNotNull(jlObject_jlM, "jlModule object of java.lang.Object should not be null");
49 
50         // Get the class loader for AccessCheckWorks and assume it's also used to
51         // load class p2.c2.
52         ClassLoader this_cldr = AccessCheckUnnamed.class.getClassLoader();
53 
54         // Define a module for p2.
55         m2x = ModuleHelper.ModuleObject("module_two", this_cldr, new String[] { "p2" });
56         assertNotNull(m2x, "Module should not be null");
57         ModuleHelper.DefineModule(m2x, false, "9.0", "m2x/there", new String[] { "p2" });
58         ModuleHelper.AddReadsModule(m2x, jlObject_jlM);
59 
60         // p1.c1's ctor tries to call a method in p2.c2.  This should fail because
61         // p1 is in the unnamed module and p2.c2 is not unqualifiedly exported.
62         Class p1_c1_class = Class.forName("p1.c1");
63         try {
64             Object c1_obj = p1_c1_class.newInstance();
65             throw new RuntimeException("Failed to get IAE (p2 in m2x is not exported to unnamed module)");
66         } catch (IllegalAccessError f) {
67             System.out.println(f.getMessage());
68             if (!f.getMessage().contains("does not export p2 to unnamed module")) {
69                 throw new RuntimeException("Wrong message: " + f.getMessage());
70             }
71         }
72     }
73 }
