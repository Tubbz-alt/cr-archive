 43 // Test that the message in a runtime ClassCastException contains module info.
 44 public class CCE_module_msg {
 45     private static final Path CLASSES_DIR = Paths.get("classes");
 46 
 47     public static void main(String[] args) throws Throwable {
 48         // Should not display version
 49         invalidObjectToDerived();
 50         invalidOriginalInnerToDerived();
 51         invalidTimeToDerived();
 52         invalidHeadersToDerived();
 53         // Should display version
 54         invalidClassToString();
 55         // Should display customer class loader
 56         invalidClassToStringCustomLoader();
 57     }
 58 
 59     public static void invalidObjectToDerived() {
 60         java.lang.Object instance = new java.lang.Object();
 61         int left = 23;
 62         int right = 42;
 63         try {
 64             for (int i = 0; i < 1; i += 1) {
 65                 left = ((Derived) instance).method(left, right);
 66             }
 67             throw new RuntimeException("ClassCastException wasn't thrown, test failed.");
 68         } catch (ClassCastException cce) {
 69             System.out.println(cce.toString());
 70             if (!cce.getMessage().contains("class java.lang.Object cannot be cast to class Derived (java.lang.Object is in module java.base of loader 'bootstrap'; Derived is in unnamed module of loader 'app')")) {
 71                 throw new RuntimeException("Wrong message: " + cce.getMessage());
 72             }
 73         }
 74     }

129     public static void invalidClassToString() throws Throwable {
130         // Get the java.lang.Module object for module java.base.
131         Class jlObject = Class.forName("java.lang.Object");
132         Object jlObject_jlM = jlObject.getModule();
133         assertNotNull(jlObject_jlM, "jlModule object of java.lang.Object should not be null");
134 
135         // Get the class loader for CCE_module_msg and assume it's also used to
136         // load classes p1.c1 and p2.c2.
137         ClassLoader this_cldr = CCE_module_msg.class.getClassLoader();
138 
139         // Define a module for p2.
140         Object m2x = ModuleHelper.ModuleObject("module_two", this_cldr, new String[] { "p2" });
141         assertNotNull(m2x, "Module should not be null");
142         ModuleHelper.DefineModule(m2x, false, "9.0", "m2x/there", new String[] { "p2" });
143         ModuleHelper.AddReadsModule(m2x, jlObject_jlM);
144 
145         try {
146             ModuleHelper.AddModuleExportsToAll(m2x, "p2");
147             Object p2Obj = new p2.c2();
148             System.out.println((String)p2Obj);
149             throw new RuntimeException("ClassCastException wasn't thrown, test failed.");
150         } catch (ClassCastException cce) {
151             String exception = cce.getMessage();
152             System.out.println(cce.toString());
153             if (!exception.contains("class p2.c2 cannot be cast to class java.lang.String (p2.c2 is in module module_two@") ||
154                 !exception.contains(" of loader 'app'; java.lang.String is in module java.base of loader 'bootstrap')")) {
155                 throw new RuntimeException("Wrong message: " + exception);
156             }
157         }
158     }

160     public static void invalidClassToStringCustomLoader() throws Throwable {
161         // Get the java.lang.Module object for module java.base.
162         Class jlObject = Class.forName("java.lang.Object");
163         Object jlObject_jlM = jlObject.getModule();
164         assertNotNull(jlObject_jlM, "jlModule object of java.lang.Object should not be null");
165 
166         // Create a customer class loader to load class p4/c4.
167         URL[] urls = new URL[] { CLASSES_DIR.toUri().toURL() };
168         ClassLoader parent = ClassLoader.getSystemClassLoader();
169         MyURLClassLoader myCldr = new MyURLClassLoader("MyClassLoader", urls, parent);
170 
171         try {
172             // Class p4.c4 should be defined to the unnamed module of myCldr
173             Class p4_c4_class = myCldr.loadClass("p4.c4");
174             Object c4Obj = p4_c4_class.newInstance();
175             System.out.println((String)c4Obj);
176             throw new RuntimeException("ClassCastException wasn't thrown, test failed.");
177         } catch (ClassCastException cce) {
178             String exception = cce.getMessage();
179             System.out.println(cce.toString());
180             if (!exception.contains("class p4.c4 cannot be cast to class java.lang.String (p4.c4 is in unnamed module of loader 'MyClassLoader' @") ||
181                 !exception.contains("; java.lang.String is in module java.base of loader 'bootstrap')")) {
182                 throw new RuntimeException("Wrong message: " + exception);
183             }
184         }
185     }

188 class OriginalInner extends java.lang.Object {
189     public int method(int left, int right) {
190         return right;
191     }
192 }
193 
194 class Derived extends java.lang.Object {
195     public int method(int left, int right) {
196         return right;
197     }
198 }
199 
200 class MyURLClassLoader extends URLClassLoader {
201     public MyURLClassLoader(String name,
202                           URL[] urls,
203                           ClassLoader parent) {
204         super(name, urls, parent);
205     }
206 
207     public Class loadClass(String name) throws ClassNotFoundException {
208         if (!name.equals("p4.c4")) {
209             return super.loadClass(name);
210         }
211         byte[] data = getClassData(name);
212         return defineClass(name, data, 0, data.length);
213     }
214 
215     byte[] getClassData(String name) {
216         try {
217            String TempName = name.replaceAll("\\.", "/");
218            String currentDir = System.getProperty("test.classes");
219            String filename = currentDir + File.separator + TempName + ".class";
220            FileInputStream fis = new FileInputStream(filename);
221            byte[] b = new byte[5000];
222            int cnt = fis.read(b, 0, 5000);
223            byte[] c = new byte[cnt];
224            for (int i=0; i<cnt; i++) c[i] = b[i];
225               return c;
226         } catch (IOException e) {
227            return null;
228         }
229     }
