24 /*
25  * @test
26  * @modules java.base/jdk.internal.misc
27  * @library /test/lib ..
28  * @compile p2/c2.java
29  * @compile p1/c1.java
30  * @build sun.hotspot.WhiteBox
31  * @compile/module=java.base java/lang/ModuleHelper.java
32  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
33  *                              sun.hotspot.WhiteBox$WhiteBoxPermission
34  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI AccessCheckOpen
35  */
36 
37 import java.lang.Module;
38 import static jdk.test.lib.Asserts.*;
39 
40 public class AccessCheckOpen {
41 
42     // Test that if module1 can read module2 and module2 is open, then
43     // p1.c1 can read p2.c2
44     public static void main(String args[]) throws Throwable {
45         Object m1, m2;
46 
47         // Get the java.lang.Module object for module java.base
48         Class jlObject = Class.forName("java.lang.Object");
49         Object jlObject_jlM = jlObject.getModule();
50         assertNotNull(jlObject_jlM, "jlModule object of java.lang.Object should not be null");
51 
52         // Get the class loader for AccessCheckOpen, which is also used to
53         // load classes p1.c1 and p2.c2
54         ClassLoader this_cldr = AccessCheckOpen.class.getClassLoader();
55 
56         // Define a module for p1
57         m1 = ModuleHelper.ModuleObject("module1", this_cldr, new String[] { "p1" });
58         assertNotNull(m1, "Module should not be null");
59         ModuleHelper.DefineModule(m1, false, "9.0", "m1/here", new String[] { "p1" });
60         ModuleHelper.AddReadsModule(m1, jlObject_jlM);
61 
62         // Define a module for p2
63         m2 = ModuleHelper.ModuleObject("module2", this_cldr, new String[] { "p2" });
64         assertNotNull(m2, "Module should not be null");
65         ModuleHelper.DefineModule(m2, true, "9.0", "m2/there", new String[] { "p2" });
66         ModuleHelper.AddReadsModule(m2, jlObject_jlM);
67 
68         // Make package p1 in m1 visible to everyone so this test can run it
69         ModuleHelper.AddModuleExportsToAll(m1, "p1");
70 
71         // Let m1 read m2
72         ModuleHelper.AddReadsModule(m1, m2);
73 
74         // p1.c1's ctor calls a method in p2.c2, and m2 is open.
75         // So should not get IllegalAccessError
76         Class p1_c1_class = Class.forName("p1.c1");
77         p1_c1_class.newInstance();
78     }
79 }
