24 /*
25  * @test
26  * @summary Module system initialization exception results if a module is specificed twice to --patch-module.
27  * @modules java.base/jdk.internal.misc
28  * @library /test/lib
29  * @run driver PatchModuleDupModule
30  */
31 
32 import jdk.test.lib.process.OutputAnalyzer;
33 import jdk.test.lib.process.ProcessTools;
34 
35 public class PatchModuleDupModule {
36 
37   // The module system initialization should generate an ExceptionInInitializerError
38   // if --patch-module is specified with the same module more than once.
39 
40   public static void main(String args[]) throws Exception {
41     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
42       "--patch-module=module_one=module_one_dir",
43       "--patch-module=module_one=module_one_dir",
44       "-version");
45     OutputAnalyzer output = new OutputAnalyzer(pb.start());
46     output.shouldContain("java.lang.ExceptionInInitializerError");
47     output.shouldHaveExitValue(1);
48   }
49 }
