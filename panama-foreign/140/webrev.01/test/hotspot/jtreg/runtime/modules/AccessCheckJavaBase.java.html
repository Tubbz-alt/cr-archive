24 /*
25  * @test
26  * @modules java.base/jdk.internal.misc
27  * @library /test/lib ..
28  * @compile p2/c2.java
29  * @build sun.hotspot.WhiteBox
30  * @compile/module=java.base java/lang/ModuleHelper.java
31  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
32  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI AccessCheckJavaBase
33  */
34 
35 import static jdk.test.lib.Asserts.*;
36 
37 public class AccessCheckJavaBase {
38 
39     // Test that a class defined to module_two always can read java.base.
40     public static void main(String args[]) throws Throwable {
41         // Get the class loader for AccessCheckJavaBase and assume it's also used to
42         // load class p2.c2.
43         ClassLoader this_cldr = AccessCheckJavaBase.class.getClassLoader();
44 
45         // Define a module for p2.
46         Object m2x = ModuleHelper.ModuleObject("module_two", this_cldr, new String[] { "p2" });
47         assertNotNull(m2x, "Module should not be null");
48         ModuleHelper.DefineModule(m2x, false, "9.0", "m2x/there", new String[] { "p2" });
49 
50         // p2.c2 can read its superclass java.lang.Object defined within java.base
51         try {
52             Class p2_c2_class = Class.forName("p2.c2");
53         } catch (IllegalAccessError e) {
54             throw new RuntimeException("Test Failed" + e.getMessage());
55         }
56     }
57 }
