29  * @compile/module=java.base java/lang/ModuleHelper.java
30  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
31  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI JVMAddReadsModule
32  */
33 
34 import static jdk.test.lib.Asserts.*;
35 
36 public class JVMAddReadsModule {
37 
38     public static void main(String args[]) throws Throwable {
39         MyClassLoader from_cl = new MyClassLoader();
40         MyClassLoader to_cl = new MyClassLoader();
41         Object from_module, to_module;
42 
43         from_module = ModuleHelper.ModuleObject("from_module", from_cl, new String[] { "mypackage" });
44         assertNotNull(from_module, "Module should not be null");
45         ModuleHelper.DefineModule(from_module, false, "9.0", "from_module/here", new String[] { "mypackage" });
46 
47         to_module = ModuleHelper.ModuleObject("to_module", to_cl, new String[] { "yourpackage" });
48         assertNotNull(to_module, "Module should not be null");
49         ModuleHelper.DefineModule(to_module, false, "9.0", "to_module/here", new String[] { "yourpackage" });
50 
51         // Null from_module argument, expect NPE
52         try {
53             ModuleHelper.AddReadsModule(null, to_module);
54             throw new RuntimeException("Failed to get the expected NPE");
55         } catch (NullPointerException e) {
56             // Expected
57         }
58 
59         // Null to_module argument, expect NPE
60         try {
61             ModuleHelper.AddReadsModule(from_module, null);
62             throw new RuntimeException("Unexpected NPE was thrown");
63         } catch (NullPointerException e) {
64             // Expected
65         }
66 
67         // Null from_module and to_module arguments, expect NPE
68         try {
69             ModuleHelper.AddReadsModule(null, null);
70             throw new RuntimeException("Failed to get the expected NPE");
71         } catch (NullPointerException e) {
72             // Expected
73         }
74 
75         // Both modules are the same, should not throw an exception
76         ModuleHelper.AddReadsModule(from_module, from_module);
77 
78         // Duplicate calls, should not throw an exception
79         ModuleHelper.AddReadsModule(from_module, to_module);
80         ModuleHelper.AddReadsModule(from_module, to_module);
81     }
82 
83     static class MyClassLoader extends ClassLoader { }
84 }
