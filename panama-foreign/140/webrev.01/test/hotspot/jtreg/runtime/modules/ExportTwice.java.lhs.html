24 /*
25  * @test
26  * @modules java.base/jdk.internal.misc
27  * @library /test/lib ..
28  * @compile p2/c2.java
29  * @compile p1/c1.java
30  * @build sun.hotspot.WhiteBox
31  * @compile/module=java.base java/lang/ModuleHelper.java
32  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
33  *                              sun.hotspot.WhiteBox$WhiteBoxPermission
34  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI ExportTwice
35  */
36 
37 import static jdk.test.lib.Asserts.*;
38 
39 public class ExportTwice {
40 
41     // Check that a package can not be exported unqualifiedly, and then exported
42     // to a specific package.
43     // Also, check that a package can be exported to a specific package and then
44     // exported unqualifiedly.
45     public static void main(String args[]) throws Throwable {
46         Object m1x, m2x, m3x;
47 
48         // Get the java.lang.Module object for module java.base.
49         Class jlObject = Class.forName("java.lang.Object");
50         Object jlObject_jlM = jlObject.getModule();
51         assertNotNull(jlObject_jlM, "jlModule object of java.lang.Object should not be null");
52 
53         // Get the class loader for ExportTwice and assume it's also used to
54         // load classes p1.c1 and p2.c2.
55         ClassLoader this_cldr = ExportTwice.class.getClassLoader();
56 
57         // Define a module for p1.
58         m1x = ModuleHelper.ModuleObject("module_one", this_cldr, new String[] { "p1" });
59         assertNotNull(m1x, "Module should not be null");
60         ModuleHelper.DefineModule(m1x, false, "9.0", "m1x/here", new String[] { "p1" });
61         ModuleHelper.AddReadsModule(m1x, jlObject_jlM);
62 
63         // Define a module for p2.
64         m2x = ModuleHelper.ModuleObject("module_two", this_cldr, new String[] { "p2" });
65         assertNotNull(m2x, "Module should not be null");
66         ModuleHelper.DefineModule(m2x, false, "9.0", "m2x/there", new String[] { "p2" });
67         ModuleHelper.AddReadsModule(m2x, jlObject_jlM);
68 
69         // Define a module for p3.
70         m3x = ModuleHelper.ModuleObject("module_three", this_cldr, new String[] { "p3" });
71         assertNotNull(m3x, "Module should not be null");
72         ModuleHelper.DefineModule(m3x, false, "9.0", "m3x/there", new String[] { "p3" });
73         ModuleHelper.AddReadsModule(m3x, jlObject_jlM);
74 
75         // Make package p1 in m1x visible to everyone.
76         ModuleHelper.AddModuleExportsToAll(m1x, "p1");
77 
78         // Try to export p1 only to m2x after it was exported unqualifiedly.  It
79         // should silently succeed.
80         ModuleHelper.AddModuleExports(m1x, "p1", m2x);
81 
82         // Export p2 to m3x then export it again unqualifiedly.
83         ModuleHelper.AddModuleExports(m2x, "p2", m3x);
84         ModuleHelper.AddModuleExportsToAll(m2x, "p2");
85 
86         // p1.c1's ctor tries to call a method in p2.c2.  This should work because
87         // p1's module can read p2's module and p2 is now exported unqualifiedly.
88         ModuleHelper.AddReadsModule(m1x, m2x);
89         Class p1_c1_class = Class.forName("p1.c1");
90         p1_c1_class.newInstance();
91     }
92 }
