24 /*
25  * @test
26  * @modules java.base/jdk.internal.misc
27  * @library /test/lib ..
28  * @compile p2/c2.java
29  * @compile p1/c1.java
30  * @build sun.hotspot.WhiteBox
31  * @compile/module=java.base java/lang/ModuleHelper.java
32  * @run driver ClassFileInstaller sun.hotspot.WhiteBox

33  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI AccessCheckRead
34  */
35 
36 import static jdk.test.lib.Asserts.*;
37 
38 public class AccessCheckRead {
39 
40     // Test that a class in a package in module_one cannot access a class in
41     // a package in module_two if module_one cannot read module_two.
42     public static void main(String args[]) throws Throwable {
43         Object m1x, m2x;
44 
45         // Get the java.lang.Module object for module java.base.
46         Class jlObject = Class.forName("java.lang.Object");
47         Object jlObject_jlM = jlObject.getModule();
48         assertNotNull(jlObject_jlM, "jlModule object of java.lang.Object should not be null");
49 
50         // Get the class loader for AccessCheckRead and assume it's also used to
51         // load classes p1.c1 and p2.c2.
52         ClassLoader this_cldr = AccessCheckRead.class.getClassLoader();
53 
54         // Define a module for p1.
55         m1x = ModuleHelper.ModuleObject("module_one", this_cldr, new String[] { "p1" });
56         assertNotNull(m1x, "Module should not be null");
57         ModuleHelper.DefineModule(m1x, false, "9.0", "m1x/here", new String[] { "p1" });
58         ModuleHelper.AddReadsModule(m1x, jlObject_jlM);
59 
60         // Define a module for p2.
61         m2x = ModuleHelper.ModuleObject("module_two", this_cldr, new String[] { "p2" });
62         assertNotNull(m2x, "Module should not be null");
63         ModuleHelper.DefineModule(m2x, false, "9.0", "m2x/there", new String[] { "p2" });
64         ModuleHelper.AddReadsModule(m2x, jlObject_jlM);
65 
66         // Make package p1 in m1x visible to everyone.
67         ModuleHelper.AddModuleExportsToAll(m1x, "p1");
68 
69         Class p1_c1_class = Class.forName("p1.c1");
70 
71         // p1.c1's ctor tries to call a method in p2.c2, but p1's module
72         // cannot read p2's module.  So should get IllegalAccessError.
73         try {
74             p1_c1_class.newInstance();
75             throw new RuntimeException("Failed to get IAE (m1x can't read m2x)");
76         } catch (IllegalAccessError e) {
77             System.out.println(e.getMessage());
78             if (!e.getMessage().contains("does not read") ||
79                 e.getMessage().contains("strict")) {
80                 throw new RuntimeException("Wrong message: " + e.getMessage());
81             }
82         }
83     }
84 }
