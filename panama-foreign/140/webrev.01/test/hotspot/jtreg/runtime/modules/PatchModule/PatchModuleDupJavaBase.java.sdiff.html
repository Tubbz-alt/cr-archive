25  * @test
26  * @summary VM exit initialization results if java.base is specificed more than once to --patch-module.
27  * @modules java.base/jdk.internal.misc
28  * @library /test/lib

29  */
30 
31 import jdk.test.lib.process.OutputAnalyzer;
32 import jdk.test.lib.process.ProcessTools;
33 
34 public class PatchModuleDupJavaBase {
35   // The VM should exit initialization if java.base is specified
36   // more than once to --patch-module.
37   public static void main(String args[]) throws Exception {
38     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
39       "--patch-module=java.base=javabase_dir",
40       "--patch-module=java.base=javabase_dir",
41       "-version");
42     OutputAnalyzer output = new OutputAnalyzer(pb.start());
43     output.shouldContain("Cannot specify java.base more than once to --patch-module");
44     output.shouldHaveExitValue(1);
45   }
46 }
