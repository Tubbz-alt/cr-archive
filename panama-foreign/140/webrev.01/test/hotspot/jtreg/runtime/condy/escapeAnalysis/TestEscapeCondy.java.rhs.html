24 /*
25  * @test
26  * @bug 8216970
27  * @summary Ensure escape analysis can handle an ldc of a dynamic
28  *          constant whose return type is an array of boolean.
29  * @modules java.base/jdk.internal.misc
30  * @library /test/lib
31  * @compile TestEscapeThroughInvokeWithCondy$A.jasm
32  * @compile TestEscapeThroughInvokeWithCondy.jasm
33  * @compile TestEscapeCondy.java
34  * @run driver TestEscapeCondy
35  */
36 
37 import jdk.test.lib.process.ProcessTools;
38 import jdk.test.lib.process.OutputAnalyzer;
39 import jdk.test.lib.compiler.InMemoryJavaCompiler;
40 
41 public class TestEscapeCondy {
42     public static void main(String args[]) throws Throwable {
43         // 1. Test escape analysis of a method that contains
44         //    a ldc instruction of a condy whose return type is an array of boolean
45         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
46              "-XX:CompileCommand=dontinline,runtime.condy.TestEscapeThroughInvokeWithCondy::create",
47              "runtime.condy.TestEscapeThroughInvokeWithCondy");
48         OutputAnalyzer oa = new OutputAnalyzer(pb.start());
49         oa.shouldContain("Test has successfully analyzed ldc bytecode within method create");
50         oa.shouldHaveExitValue(0);
51     }
52 }
