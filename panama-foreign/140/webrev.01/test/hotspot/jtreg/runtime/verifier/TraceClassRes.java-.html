24 /*
25  * @test
26  * @bug 8076318
27  * @summary split verifier needs to add class resolution tracing
28  * @modules java.base/jdk.internal.misc
29  * @library /test/lib
30  */
31 
32 import jdk.test.lib.process.ProcessTools;
33 import jdk.test.lib.process.OutputAnalyzer;
34 
35 // Test that the verifier outputs the classes it loads if -Xlog:class+resove=debug is specified"
36 public class TraceClassRes {
37   public static void main(String[] args) throws Exception {
38 
39     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
40         "-Xlog:class+resolve=debug", "-verify", "-Xshare:off", "-version");
41 
42     OutputAnalyzer output = new OutputAnalyzer(pb.start());
43     output.shouldContain("[class,resolve] java.lang.ClassLoader java.lang.Throwable ClassLoader.java (verification)");
44     output.shouldHaveExitValue(0);
45   }
46 }
