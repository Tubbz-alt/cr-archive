32 import jdk.test.lib.process.ProcessTools;
33 import jdk.test.lib.process.OutputAnalyzer;
34 
35 // Test that system property jdk.boot.class.path.append is initialized.  Otherwise,
36 // -XX:+PrintCompilation does causes a SIGSEGV.
37 public class BootClassPathAppend {
38   public static void main(String[] args) throws Exception {
39 
40     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
41         "-XX:+PrintCompilation", "-Xcomp", "-version");
42 
43     OutputAnalyzer output = new OutputAnalyzer(pb.start());
44     output.shouldContain("java.lang.Object");
45     output.shouldHaveExitValue(0);
46   }
47 }
