24 /*
25  * @test
26  * @summary Test Reserve/Commit/Uncommit/Release of virtual memory and that we track it correctly
27  * @key nmt jcmd
28  * @library /test/lib
29  * @modules java.base/jdk.internal.misc
30  *          java.management
31  * @build sun.hotspot.WhiteBox
32  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
33  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -XX:NativeMemoryTracking=detail VirtualAllocTestType
34  */
35 
36 import jdk.test.lib.process.ProcessTools;
37 import jdk.test.lib.process.OutputAnalyzer;
38 import jdk.test.lib.JDKToolFinder;
39 import sun.hotspot.WhiteBox;
40 
41 public class VirtualAllocTestType {
42 
43   public static WhiteBox wb = WhiteBox.getWhiteBox();
44   public static void main(String args[]) throws Exception {
45     OutputAnalyzer output;
46     long commitSize = 128 * 1024;
47     long reserveSize = 256 * 1024;
48     long addr;
49 
50     String pid = Long.toString(ProcessTools.getProcessId());
51     ProcessBuilder pb = new ProcessBuilder();
52 
53     addr = wb.NMTReserveMemory(reserveSize);
54     pb.command(new String[] { JDKToolFinder.getJDKTool("jcmd"), pid, "VM.native_memory", "detail"});
55 
56     output = new OutputAnalyzer(pb.start());
57     output.shouldContain("Test (reserved=256KB, committed=0KB)");
58     output.shouldMatch("\\[0x[0]*" + Long.toHexString(addr) + " - 0x[0]*" + Long.toHexString(addr + reserveSize) + "\\] reserved 256KB for Test");
59 
60     wb.NMTCommitMemory(addr, commitSize);
61 
62 
63     output = new OutputAnalyzer(pb.start());
64     output.shouldContain("Test (reserved=256KB, committed=128KB)");
65     output.shouldMatch("\\[0x[0]*" + Long.toHexString(addr) + " - 0x[0]*" + Long.toHexString(addr + commitSize) + "\\] committed 128KB");
66 
67     wb.NMTUncommitMemory(addr, commitSize);
68 
69 
70     output = new OutputAnalyzer(pb.start());
71     output.shouldContain("Test (reserved=256KB, committed=0KB)");
72     output.shouldNotMatch("\\[0x[0]*" + Long.toHexString(addr) + " - 0x[0]*" + Long.toHexString(addr + commitSize) + "\\] committed");
73 
74     wb.NMTReleaseMemory(addr, reserveSize);
75 
76     output = new OutputAnalyzer(pb.start());
77     output.shouldNotContain("Test (reserved=");
78     output.shouldNotMatch("\\[0x[0]*" + Long.toHexString(addr) + " - 0x[0]*" + Long.toHexString(addr + reserveSize) + "\\] reserved");
79   }
80 }
