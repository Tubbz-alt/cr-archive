24  /*
25  * @test
26  * @key nmt
27  * @summary Empty argument to NMT should result in an informative error message
28  * @library /test/lib
29  * @modules java.base/jdk.internal.misc
30  *          java.management

31  */
32 
33 import jdk.test.lib.process.ProcessTools;
34 import jdk.test.lib.process.OutputAnalyzer;
35 
36 public class CommandLineEmptyArgument {
37 
38   public static void main(String args[]) throws Exception {
39     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:NativeMemoryTracking=");
40     OutputAnalyzer output = new OutputAnalyzer(pb.start());
41     output.shouldContain("Syntax error, expecting -XX:NativeMemoryTracking=[off|summary|detail]");
42     output.shouldHaveExitValue(1);
43   }
44 }
