24 // Aix commits on touch, so this test won't work.
25 /*
26  * @test
27  * @key regression
28  * @bug 8012015
29  * @requires !(os.family == "aix")
30  * @summary Make sure reserved (but uncommitted) memory is not accessible
31  * @library /test/lib
32  * @modules java.base/jdk.internal.misc
33  *          java.management
34  * @build sun.hotspot.WhiteBox
35  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
36  *                              sun.hotspot.WhiteBox$WhiteBoxPermission
37  * @run main ReserveMemory
38  */
39 
40 import jdk.test.lib.process.ProcessTools;
41 import jdk.test.lib.process.OutputAnalyzer;
42 import jdk.test.lib.Platform;
43 
44 import sun.hotspot.WhiteBox;
45 
46 public class ReserveMemory {
47   public static void main(String args[]) throws Exception {
48     if (args.length > 0) {
49       WhiteBox.getWhiteBox().readReservedMemory();
50 
51       throw new Exception("Read of reserved/uncommitted memory unexpectedly succeeded, expected crash!");
52     }
53 
54     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
55           "-Xbootclasspath/a:.",
56           "-XX:+UnlockDiagnosticVMOptions",
57           "-XX:+WhiteBoxAPI",
58           "-XX:-CreateCoredumpOnCrash",
59           "-Xmx128m",
60           "ReserveMemory",
61           "test");
62 
63     OutputAnalyzer output = new OutputAnalyzer(pb.start());
64     if (Platform.isWindows()) {
65       output.shouldContain("EXCEPTION_ACCESS_VIOLATION");
66     } else if (Platform.isOSX()) {
67       output.shouldContain("SIGBUS");
68     } else {
69       output.shouldContain("SIGSEGV");
70     }
71   }
72 }
