24 /*
25  * @test
26  * @summary Test launches unit tests inside vm concurrently
27  * @library /test/lib
28  * @modules java.base/jdk.internal.misc
29  *          java.management
30  * @build sun.hotspot.WhiteBox
31  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
32  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI RunUnitTestsConcurrently 30 15000
33  */
34 
35 import jdk.test.lib.Platform;
36 import sun.hotspot.WhiteBox;
37 
38 public class RunUnitTestsConcurrently {
39 
40   private static WhiteBox wb;
41   private static long timeout;
42   private static long timeStamp;
43 
44   public static class Worker implements Runnable {
45     @Override
46     public void run() {
47       while (System.currentTimeMillis() - timeStamp < timeout) {
48         WhiteBox.getWhiteBox().runMemoryUnitTests();
49       }
50     }
51   }
52 
53   public static void main(String[] args) throws InterruptedException {
54     if (!Platform.isDebugBuild() || !Platform.is64bit()) {
55       return;
56     }
57     wb = WhiteBox.getWhiteBox();
58     System.out.println("Starting threads");
59 
60     int threads = Integer.valueOf(args[0]);
61     timeout = Long.valueOf(args[1]);
62 
63     timeStamp = System.currentTimeMillis();
64 
65     Thread[] threadsArray = new Thread[threads];
66     for (int i = 0; i < threads; i++) {
67       threadsArray[i] = new Thread(new Worker());
68       threadsArray[i].start();
69     }
70     for (int i = 0; i < threads; i++) {
71       threadsArray[i].join();
72     }
73 
74     System.out.println("Quitting test.");
75   }
76 }
