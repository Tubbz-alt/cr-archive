25 /*
26  * @test HandshakeWalkExitTest
27  * @summary This test tries to stress the handshakes with new and exiting threads
28  * @library /testlibrary /test/lib
29  * @build HandshakeWalkExitTest
30  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
31  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI HandshakeWalkExitTest
32  */
33 
34 import jdk.test.lib.Asserts;
35 import sun.hotspot.WhiteBox;
36 
37 public class HandshakeWalkExitTest  implements Runnable {
38 
39     @Override
40     public void run() {
41     }
42 
43     static volatile boolean exit_now = false;
44 
45     public static void main(String... args) throws Exception {
46         int testRuns = 20;
47         int testThreads = 128;
48 
49         HandshakeWalkExitTest test = new HandshakeWalkExitTest();
50 
51         Runnable hser = new Runnable(){
52             public void run(){
53                 WhiteBox wb = WhiteBox.getWhiteBox();
54                 while(!exit_now) {
55                     wb.handshakeWalkStack(null, true);
56                 }
57             }
58         };
59         Thread hst = new Thread(hser);
60         hst.start();
61         for (int k = 0; k<testRuns ; k++) {
62             Thread[] threads = new Thread[testThreads];
63             for (int i = 0; i<threads.length ; i++) {
64                 threads[i] = new Thread(test);
65                 threads[i].start();
66             }
67         }
68         exit_now = true;
69         hst.join();
70     }
71 }
