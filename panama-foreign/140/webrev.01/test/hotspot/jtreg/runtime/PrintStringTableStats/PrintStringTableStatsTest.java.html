25 import jdk.test.lib.process.OutputAnalyzer;

27 /*
28  * @test PrintStringTableStatsTest
29  * @bug 8211821
30  * @library /test/lib
31  * @run driver PrintStringTableStatsTest

33 
34 public class PrintStringTableStatsTest {
35     public static void main(String... args) throws Exception {
36         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
37             "-XX:+PrintStringTableStatistics",
38             "--version");
39         OutputAnalyzer output = new OutputAnalyzer(pb.start());
40         output.shouldContain("Number of buckets");
41         output.shouldHaveExitValue(0);
42     }
