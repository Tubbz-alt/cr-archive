24 /*
25  * @test
26  * @bug 8047290
27  * @summary Ensure that Monitor::lock does not assert when it correctly acquires a lock which must always have safepoint checks.
28  * @library /test/lib
29  * @modules java.base/jdk.internal.misc
30  *          java.management
31  * @build sun.hotspot.WhiteBox
32  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
33  *                              sun.hotspot.WhiteBox$WhiteBoxPermission
34  * @run main AssertSafepointCheckConsistency4
35  */
36 
37 import jdk.test.lib.process.ProcessTools;
38 import jdk.test.lib.process.OutputAnalyzer;
39 import jdk.test.lib.Platform;
40 
41 import sun.hotspot.WhiteBox;
42 
43 public class AssertSafepointCheckConsistency4 {
44     public static void main(String args[]) throws Exception {
45         if (args.length > 0) {
46             WhiteBox.getWhiteBox().assertMatchingSafepointCalls(true, false);
47         }
48         if (Platform.isDebugBuild()){
49             ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
50                "-Xbootclasspath/a:.",
51                "-XX:+UnlockDiagnosticVMOptions",
52                "-XX:+WhiteBoxAPI",
53                "-XX:-CreateCoredumpOnCrash",
54                "-Xmx32m",
55                "AssertSafepointCheckConsistency4",
56                "test");
57 
58             OutputAnalyzer output = new OutputAnalyzer(pb.start());
59             output.shouldNotContain("assert");
60             output.shouldNotContain("never");
61             output.shouldNotContain("always");
62         }
63     }
64 }
