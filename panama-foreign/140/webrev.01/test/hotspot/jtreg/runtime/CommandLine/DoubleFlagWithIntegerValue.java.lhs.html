24 /*
25  * @test DoubleFlagWithIntegerValue
26  * @bug 8178364
27  * @summary Command-line flags of type double should accept integer values
28  * @library /test/lib
29  * @modules java.base/jdk.internal.misc
30  *          java.management

31  */
32 
33 import jdk.test.lib.process.ProcessTools;
34 import jdk.test.lib.process.OutputAnalyzer;
35 
36 public class DoubleFlagWithIntegerValue {
37   public static void testDoubleFlagWithValue(String value) throws Exception {
38     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:G1ConcMarkStepDurationMillis=" + value, "-version");
39     OutputAnalyzer output = new OutputAnalyzer(pb.start());
40     output.shouldNotContain("Improperly specified VM option");
41     output.shouldHaveExitValue(0);
42   }
43 
44   public static void main(String[] args) throws Exception {
45     // Test double format
46     testDoubleFlagWithValue("10.0");
47 
48     // Test integer format
49     testDoubleFlagWithValue("10");
50   }
51 }
