24 /*
25  * @test
26  * @bug 8042885
27  * @summary Make sure there is no error using hexadecimal format in vm options
28  * @author Yumin Qi
29  * @library /test/lib
30  * @modules java.base/jdk.internal.misc
31  *          java.management

32  */
33 
34 import java.io.File;
35 import jdk.test.lib.process.ProcessTools;
36 import jdk.test.lib.process.OutputAnalyzer;
37 
38 public class TestHexArguments {
39     public static void main(String args[]) throws Exception {
40       String[] javaArgs = {"-XX:SharedBaseAddress=0x1D000000", "-version"};
41       ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(javaArgs);
42 
43       OutputAnalyzer output = new OutputAnalyzer(pb.start());
44       output.shouldNotContain("Could not create the Java Virtual Machine");
45       output.shouldHaveExitValue(0);
46 
47       String[] javaArgs1 = {"-XX:SharedBaseAddress=1D000000", "-version"};
48       pb = ProcessTools.createJavaProcessBuilder(javaArgs1);
49       output = new OutputAnalyzer(pb.start());
50       output.shouldContain("Could not create the Java Virtual Machine");
51   }
52 }
