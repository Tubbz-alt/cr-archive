24 /*
25  * @test
26  * @bug 7167142
27  * @summary Warn if unused .hotspot_compiler file is present
28  * @library /test/lib
29  * @modules java.base/jdk.internal.misc
30  *          java.management
31  * @run driver CompilerConfigFileWarning
32  */
33 
34 import java.io.PrintWriter;
35 import jdk.test.lib.process.ProcessTools;
36 import jdk.test.lib.process.OutputAnalyzer;
37 import jdk.test.lib.Platform;
38 
39 public class CompilerConfigFileWarning {
40     public static void main(String[] args) throws Exception {
41         ProcessBuilder pb;
42         OutputAnalyzer output;
43         PrintWriter pw;
44 
45         pw = new PrintWriter("hs_comp.txt");
46         pw.println("aaa, aaa");
47         pw.close();
48 
49         pb = ProcessTools.createJavaProcessBuilder("-XX:CompileCommandFile=hs_comp.txt", "-version");
50         output = new OutputAnalyzer(pb.start());
51         output.shouldContain("CompileCommand: unrecognized command");
52         output.shouldContain("aaa, aaa");
53 
54         // Skip on debug builds since we'll always read the file there
55         if (!Platform.isDebugBuild()) {
56             pw = new PrintWriter(".hotspot_compiler");
57             pw.println("aa");
58             pw.close();
59 
60             pb = ProcessTools.createJavaProcessBuilder("-version");
61             output = new OutputAnalyzer(pb.start());
62             output.shouldContain("warning: .hotspot_compiler file is present but has been ignored.  Run with -XX:CompileCommandFile=.hotspot_compiler to load the file.");
63         }
64     }
65 }
