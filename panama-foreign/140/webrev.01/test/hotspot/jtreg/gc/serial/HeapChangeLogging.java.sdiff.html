26 /*
27  * @test HeapChangeLogging.java
28  * @bug 8027440
29  * @requires vm.gc.Serial
30  * @library /test/lib
31  * @modules java.base/jdk.internal.misc
32  * @summary Allocate to get a promotion failure and verify that that heap change logging is present.
33  * @run main gc.serial.HeapChangeLogging
34  */
35 
36 import java.util.regex.Matcher;
37 import java.util.regex.Pattern;
38 
39 import jdk.test.lib.process.ProcessTools;
40 import jdk.test.lib.process.OutputAnalyzer;
41 
42 public class HeapChangeLogging {
43   public static void main(String[] args) throws Exception {
44     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-Xmx128m", "-Xmn100m", "-XX:+UseSerialGC", "-Xlog:gc", HeapFiller.class.getName());
45     OutputAnalyzer output = new OutputAnalyzer(pb.start());
46     String stdout = output.getStdout();
47     System.out.println(stdout);
48     Matcher stdoutMatcher = Pattern.compile(".*\\(Allocation Failure\\) [0-9]+[KMG]->[0-9]+[KMG]\\([0-9]+[KMG]\\)", Pattern.MULTILINE).matcher(stdout);
49     if (!stdoutMatcher.find()) {
50       throw new RuntimeException("No proper GC log line found");
51     }
52     output.shouldHaveExitValue(0);
53   }
