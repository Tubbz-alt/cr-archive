24 package gc.g1;
25 
26 /*
27  * @test TestNoEagerReclaimOfHumongousRegions
28  * @bug 8139424
29  * @summary Test to check that a live humongous object is not eagerly reclaimed. This is a regression test for
30  *          8139424 and the test will crash if an eager reclaim occur. The test is not 100% deterministic and
31  *          might pass even if there are problems in the code, but it will never crash unless there is a problem.
32  * @requires vm.gc.G1
33  * @key gc
34  * @library /test/lib
35  * @modules java.base/jdk.internal.misc
36  * @build sun.hotspot.WhiteBox
37  * @run driver ClassFileInstaller sun.hotspot.WhiteBox

38  * @run main/othervm -Xbootclasspath/a:. -Xlog:gc,gc+humongous=debug -XX:+UseG1GC -XX:MaxTenuringThreshold=0 -XX:G1RSetSparseRegionEntries=32 -XX:G1HeapRegionSize=1m -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI gc.g1.TestNoEagerReclaimOfHumongousRegions
39  */
40 
41 import java.util.LinkedList;
42 
43 import sun.hotspot.WhiteBox;
44 
45 public class TestNoEagerReclaimOfHumongousRegions {
46     // Helper class to keep reference to humongous byte[].
47     static class LargeRef {
48         private byte[] _ref;
49 
50         LargeRef(byte[] ref) {
51             _ref = ref;
52         }
53 
54         byte[] ref() { return _ref; }
55     }
56 
57     static LargeRef humongous_reference_holder;
58 
59     public static void main(String[] args) throws InterruptedException{
60         WhiteBox wb = WhiteBox.getWhiteBox();
61         LinkedList<Object> garbageAndRefList = new LinkedList<Object>();
62         // Creating a 1M large byte array. Since the test specifies the heap
63         // region size to be 1m this will be a humongous object. We then
64         // store a pointer to the array in the static object to keep it live
65         // during the whole test.
66         humongous_reference_holder = new LargeRef(new byte[1 * 1024 * 1024]);
67 
68         // Create some garbage and a reference to the humongous object each round.
69         for (int i = 0; i < 32; i++) {
70             garbageAndRefList.add(new byte[400*1000]);
71             garbageAndRefList.add(new LargeRef(humongous_reference_holder.ref()));
72 
73             // Promote to old, goal is to get rem-set entries for the humongous
74             // object from different regions. The test specifies MaxTenuringThreshold=0,
75             // this will make sure we get objects promoted to old at once.
76             wb.youngGC();
77         }
78         // Clear the garbage and reference list.
79         garbageAndRefList.clear();
80 
81         // Run a concurrent mark cycle to mark all references but the static one as dead.
82         wb.g1StartConcMarkCycle();
83         while (wb.g1InConcurrentMark()) {
84             Thread.sleep(100);
85         }
86 
87         // Run a young collection to make sure humongous object still can't be eagerly reclaimed.
88         wb.youngGC();
89         // Will crash/assert if humongous object has been reclaimed.
90         wb.fullGC();
91     }
92 }
