 24 package gc.g1.humongousObjects;
 25 
 26 import gc.testlibrary.Helpers;
 27 import jdk.test.lib.Asserts;
 28 import sun.hotspot.WhiteBox;
 29 
 30 /**
 31  * @test TestHumongousThreshold
 32  * @summary Checks that objects larger than half a region are allocated as humongous
 33  * @requires vm.gc.G1
 34  * @library /test/lib /
 35  * @modules java.base/jdk.internal.misc
 36  * @modules java.management
 37  * @build sun.hotspot.WhiteBox
 38  * @run driver ClassFileInstaller sun.hotspot.WhiteBox

 39  *
 40  * @run main/othervm -XX:+UseG1GC -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:.
 41  * -XX:G1HeapRegionSize=1M
 42  * gc.g1.humongousObjects.TestHumongousThreshold
 43  *
 44  * @run main/othervm -XX:+UseG1GC -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:.
 45  * -XX:G1HeapRegionSize=2M
 46  * gc.g1.humongousObjects.TestHumongousThreshold
 47  *
 48  * @run main/othervm -XX:+UseG1GC -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:.
 49  * -XX:G1HeapRegionSize=4M
 50  * gc.g1.humongousObjects.TestHumongousThreshold
 51  *
 52  * @run main/othervm -XX:+UseG1GC -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:.
 53  * -XX:G1HeapRegionSize=8M
 54  * gc.g1.humongousObjects.TestHumongousThreshold
 55  *
 56  * @run main/othervm -XX:+UseG1GC -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:.
 57  * -Xms128M -XX:G1HeapRegionSize=16M
 58  * gc.g1.humongousObjects.TestHumongousThreshold
 59  *
 60  * @run main/othervm -XX:+UseG1GC -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:.
 61  * -Xms200M -XX:G1HeapRegionSize=32M
 62  * gc.g1.humongousObjects.TestHumongousThreshold
 63  *
 64  */
 65 
 66 public class TestHumongousThreshold {
 67     private static final WhiteBox WHITE_BOX = WhiteBox.getWhiteBox();
 68     private static final int REGION_SIZE = WHITE_BOX.g1RegionSize();
 69     private static final int MAX_CONTINUOUS_SIZE_CHECK = 129;
 70     private static final int NON_HUMONGOUS_STEPS = 10;
 71 
 72     /**
 73      * The method allocates byte[] with specified size and checks that:
 74      * 1. byte[] is allocated as we specified in expectedHumongous.
 75      * 2. byte[] is allocated as humongous if its size is large than a half of region and non-humongous otherwise.
 76      * It uses WB to obtain the size of created byte[]. Only objects larger than half of region are expected
 77      * to be humongous.
 78      *
 79      * @param arraySize size of allocation
 80      * @param expectedHumongous expected humongous/non-humongous allocation
 81      * @return allocated byte array
 82      */
 83 
 84     private static void allocateAndCheck(int arraySize, boolean expectedHumongous) {
 85         byte[] storage = new byte[arraySize];
 86         long objectSize = WHITE_BOX.getObjectSize(storage);
 87         boolean shouldBeHumongous = objectSize > (REGION_SIZE / 2);
 88 
 89         Asserts.assertEquals(expectedHumongous, shouldBeHumongous, "Despite we expected this object to be "
 90                 + (expectedHumongous ? "humongous" : "non-humongous") + " it appeared otherwise when we checked "
 91                 + "object size - likely test bug; Allocation size = " + arraySize + "; Object size = " + objectSize
 92                 + "; region size = " + REGION_SIZE);
 93 
 94         Asserts.assertEquals(WHITE_BOX.g1IsHumongous(storage), shouldBeHumongous,
 95                 "Object should be allocated as " + (shouldBeHumongous ? "humongous"
 96                         : "non-humongous") + " but it wasn't; Allocation size = " + arraySize + "; Object size = "
 97                         + objectSize + "; region size = " + REGION_SIZE);
 98     }
 99 
100     public static void main(String[] args) {
101         int byteArrayMemoryOverhead = Helpers.detectByteArrayAllocationOverhead();
102 
103         // Largest non-humongous byte[]
104         int maxByteArrayNonHumongousSize = (REGION_SIZE / 2) - byteArrayMemoryOverhead;
105 
106         // Increment for non-humongous testing
107         int nonHumongousStep = maxByteArrayNonHumongousSize / NON_HUMONGOUS_STEPS;
108 
109         // Maximum byte[] that takes one region
110         int maxByteArrayOneRegionSize = REGION_SIZE - byteArrayMemoryOverhead;
111 
112         // Sizes in regions
113         // i,e, 1.0f means one region, 1.5f means one and half region etc
114         float[] humongousFactors = {0.8f, 1.0f, 1.2f, 1.5f, 1.7f, 2.0f, 2.5f};
115 
116         // Some diagnostic output
117         System.out.format("%s started%n", TestHumongousThreshold.class.getName());
118         System.out.format("Actual G1 region size %d%n", REGION_SIZE);
119         System.out.format("byte[] memory overhead %d%n", byteArrayMemoryOverhead);
120 
121         // Non-humongous allocations
122         System.out.format("Doing non-humongous allocations%n");
123 
124         // Testing allocations with byte[] with length from 0 to MAX_CONTINUOUS_SIZE_CHECK
125         System.out.format("Testing allocations with byte[] with length from 0 to %d%n", MAX_CONTINUOUS_SIZE_CHECK);
126         for (int i = 0; i < MAX_CONTINUOUS_SIZE_CHECK; ++i) {
127             allocateAndCheck(i, false);
128         }
129 
130         // Testing allocations with byte[] with length from 0 to nonHumongousStep * NON_HUMONGOUS_STEPS
131         System.out.format("Testing allocations with byte[] with length from 0 to %d with step %d%n",
132                 nonHumongousStep * NON_HUMONGOUS_STEPS, nonHumongousStep);
133         for (int i = 0; i < NON_HUMONGOUS_STEPS; ++i) {
134             allocateAndCheck(i * nonHumongousStep, false);
135         }
136 
137         // Testing allocations with byte[] of maximum non-humongous length
138         System.out.format("Testing allocations with byte[] of maximum non-humongous length %d%n",
139                 maxByteArrayNonHumongousSize);
140         allocateAndCheck(maxByteArrayNonHumongousSize, false);
141 
142         // Humongous allocations
143         System.out.format("Doing humongous allocations%n");
144         // Testing with minimum humongous object
145         System.out.format("Testing with byte[] of minimum humongous object %d%n", maxByteArrayNonHumongousSize + 1);
146         allocateAndCheck(maxByteArrayNonHumongousSize + 1, true);
147 
148         // Testing allocations with byte[] with length from (maxByteArrayNonHumongousSize + 1) to
149         // (maxByteArrayNonHumongousSize + 1 + MAX_CONTINUOUS_SIZE_CHECK)
150         System.out.format("Testing allocations with byte[] with length from %d to %d%n",
151                 maxByteArrayNonHumongousSize + 1, maxByteArrayNonHumongousSize + 1 + MAX_CONTINUOUS_SIZE_CHECK);
152         for (int i = 0; i < MAX_CONTINUOUS_SIZE_CHECK; ++i) {
153             allocateAndCheck(maxByteArrayNonHumongousSize + 1 + i, true);
154         }
155 
156         // Checking that large (more than a half of region size) objects are humongous
157         System.out.format("Checking that large (more than a half of region size) objects are humongous%n");
158         for (float factor : humongousFactors) {
159             allocateAndCheck((int) (maxByteArrayOneRegionSize * factor), true);
160         }
161     }
162 }
