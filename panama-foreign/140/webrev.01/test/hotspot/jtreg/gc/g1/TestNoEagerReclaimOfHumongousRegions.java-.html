24 package gc.g1;
25 
26 /*
27  * @test TestNoEagerReclaimOfHumongousRegions
28  * @bug 8139424
29  * @summary Test to check that a live humongous object is not eagerly reclaimed. This is a regression test for
30  *          8139424 and the test will crash if an eager reclaim occur. The test is not 100% deterministic and
31  *          might pass even if there are problems in the code, but it will never crash unless there is a problem.
32  * @requires vm.gc.G1
33  * @key gc
34  * @library /test/lib
35  * @modules java.base/jdk.internal.misc
36  * @build sun.hotspot.WhiteBox
37  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
38  *                              sun.hotspot.WhiteBox$WhiteBoxPermission
39  * @run main/othervm -Xbootclasspath/a:. -Xlog:gc,gc+humongous=debug -XX:+UseG1GC -XX:MaxTenuringThreshold=0 -XX:G1RSetSparseRegionEntries=32 -XX:G1HeapRegionSize=1m -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI gc.g1.TestNoEagerReclaimOfHumongousRegions
40  */
41 
42 import java.util.LinkedList;
43 
44 import sun.hotspot.WhiteBox;
45 
46 public class TestNoEagerReclaimOfHumongousRegions {
47     // Helper class to keep reference to humongous byte[].
48     static class LargeRef {
49         private byte[] _ref;
50 
51         LargeRef(byte[] ref) {
52             _ref = ref;
53         }
54 
55         byte[] ref() { return _ref; }
56     }
57 
58     static LargeRef humongous_reference_holder;
59 
60     public static void main(String[] args) throws InterruptedException{
61         WhiteBox wb = WhiteBox.getWhiteBox();
62         LinkedList<Object> garbageAndRefList = new LinkedList<Object>();
63         // Creating a 1M large byte array. Since the test specifies the heap
64         // region size to be 1m this will be a humongous object. We then
65         // store a pointer to the array in the static object to keep it live
66         // during the whole test.
67         humongous_reference_holder = new LargeRef(new byte[1 * 1024 * 1024]);
68 
69         // Create some garbage and a reference to the humongous object each round.
70         for (int i = 0; i < 32; i++) {
71             garbageAndRefList.add(new byte[400*1000]);
72             garbageAndRefList.add(new LargeRef(humongous_reference_holder.ref()));
73 
74             // Promote to old, goal is to get rem-set entries for the humongous
75             // object from different regions. The test specifies MaxTenuringThreshold=0,
76             // this will make sure we get objects promoted to old at once.
77             wb.youngGC();
78         }
79         // Clear the garbage and reference list.
80         garbageAndRefList.clear();
81 
82         // Run a concurrent mark cycle to mark all references but the static one as dead.
83         wb.g1StartConcMarkCycle();
84         while (wb.g1InConcurrentMark()) {
85             Thread.sleep(100);
86         }
87 
88         // Run a young collection to make sure humongous object still can't be eagerly reclaimed.
89         wb.youngGC();
90         // Will crash/assert if humongous object has been reclaimed.
91         wb.fullGC();
92     }
93 }
