 24 package gc.g1;
 25 
 26 /*
 27  * @test TestEagerReclaimHumongousRegionsLog
 28  * @summary Check that G1 reports humongous eager reclaim statistics correctly.
 29  * @requires vm.gc.G1
 30  * @key gc
 31  * @library /test/lib
 32  * @modules java.base/jdk.internal.misc
 33  *          java.management
 34  * @build sun.hotspot.WhiteBox
 35  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
 36  * @run driver gc.g1.TestEagerReclaimHumongousRegionsLog
 37  */
 38 
 39 import sun.hotspot.WhiteBox;
 40 
 41 import java.util.Arrays;
 42 import jdk.test.lib.Asserts;
 43 
 44 import jdk.test.lib.process.OutputAnalyzer;
 45 import jdk.test.lib.process.ProcessTools;
 46 
 47 public class TestEagerReclaimHumongousRegionsLog {
 48 
 49     private static final String LogSeparator = ": ";
 50 
 51     public static void runTest() throws Exception {
 52         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
 53             "-Xbootclasspath/a:.",
 54             "-XX:+UnlockExperimentalVMOptions",
 55             "-XX:+UnlockDiagnosticVMOptions",
 56             "-XX:+WhiteBoxAPI",
 57             "-XX:+UseG1GC",
 58             "-XX:G1HeapRegionSize=1M",
 59             "-Xms128M",
 60             "-Xmx128M",
 61             "-Xlog:gc+phases=trace,gc+heap=info",
 62             GCTest.class.getName());
 63         OutputAnalyzer output = new OutputAnalyzer(pb.start());
 64 
 65         output.shouldHaveExitValue(0);
 66 
 67         System.out.println(output.getStdout());
 68 
 69         // This gives an array of lines containing eager reclaim of humongous regions
 70         // log messages contents after the ":" in the following order for every GC:
 71         //   Region Register: a.ams
 72         //   Humongous Total: b
 73         //   Humongous Candidate: c
 74         //   Humongous Reclaim: d.dms
 75         //   Humongous Reclaimed: e
 76         //   Humongous Regions: f->g
 77 
 78         String[] lines = Arrays.stream(output.getStdout().split("\\R"))
 79                          .filter(s -> (s.contains("Humongous") || s.contains("Region Register"))).map(s -> s.substring(s.indexOf(LogSeparator) + LogSeparator.length()))
 80                          .toArray(String[]::new);
 81 
 82         Asserts.assertTrue(lines.length % 6 == 0, "There seems to be an unexpected amount of log messages (total: " + lines.length + ") per GC");
 83 
 84         for (int i = 0; i < lines.length; i += 6) {
 85             int total = Integer.parseInt(lines[i + 1]);
 86             int candidate = Integer.parseInt(lines[i + 2]);
 87             int reclaimed = Integer.parseInt(lines[i + 4]);
 88 
 89             int before = Integer.parseInt(lines[i + 5].substring(0, 1));
 90             int after = Integer.parseInt(lines[i + 5].substring(3, 4));
 91             System.out.println("total " + total + " candidate " + candidate + " reclaimed " + reclaimed + " before " + before + " after " + after);
 92 
 93             Asserts.assertEQ(total, candidate, "Not all humonguous objects are candidates");
 94             Asserts.assertLTE(reclaimed, candidate, "The number of reclaimed objects must be less or equal than the number of candidates");
 95 
 96             if (reclaimed > 0) {
 97                Asserts.assertLT(after, before, "Number of regions after must be smaller than before.");
 98                Asserts.assertEQ(reclaimed, candidate, "Must have reclaimed all candidates.");
 99                Asserts.assertGT((before - after), reclaimed, "Number of regions reclaimed (" + (before - after) +
100                                 ") must be larger than number of objects reclaimed (" + reclaimed + ")");
101             }
102         }
103     }
104 
105     public static void main(String[] args) throws Exception {
106         runTest();
107     }
108 
109     static class GCTest {
110         private static final WhiteBox WB = WhiteBox.getWhiteBox();
111 
112         public static Object holder;
113 
114         public static void main(String [] args) {
115             // Create a humongous objects spanning multiple regions so that the difference
116             // between number of humongous objects reclaimed and number of regions reclaimed
117             // is apparent.
118             holder = new byte[4 * 1024 * 1024];
119             WB.youngGC();
120             System.out.println(holder);
121             holder = null;
122             WB.youngGC();
123         }
124     }
125 }
