24 package gc;
25 
26 /* @test TestAllocateHeapAtMultiple.java
27  * @key gc
28  * @summary Test to check allocation of Java Heap with AllocateHeapAt option. Has multiple sub-tests to cover different code paths.
29  * @library /test/lib
30  * @modules java.base/jdk.internal.misc
31  * @requires vm.bits == "64" & vm.gc != "Z" & os.family != "aix"
32  * @run driver/timeout=360 gc.TestAllocateHeapAtMultiple
33  */
34 
35 import jdk.test.lib.JDKToolFinder;
36 import jdk.test.lib.process.ProcessTools;
37 import jdk.test.lib.process.OutputAnalyzer;
38 import java.util.ArrayList;
39 import java.util.Collections;
40 
41 public class TestAllocateHeapAtMultiple {
42   public static void main(String args[]) throws Exception {
43     ArrayList<String> flags = new ArrayList<>();
44 
45     String test_dir = System.getProperty("test.dir", ".");
46 
47     // Extra flags for each of the sub-tests
48     String[][] extraFlagsList = new String[][] {
49       {"-Xmx32m", "-Xms32m", "-XX:+UseCompressedOops"},     // 1. With compressedoops enabled.
50       {"-Xmx32m", "-Xms32m", "-XX:-UseCompressedOops"},     // 2. With compressedoops disabled.
51       {"-Xmx32m", "-Xms32m", "-XX:HeapBaseMinAddress=3g"},  // 3. With user specified HeapBaseMinAddress.
52       {"-Xmx32m", "-Xms32m", "-XX:+UseLargePages"},         // 4. Set UseLargePages.
53       {"-Xmx32m", "-Xms32m", "-XX:+UseNUMA"}                // 5. Set UseNUMA.
54     };
55 
56     for (String[] extraFlags : extraFlagsList) {
57       flags.clear();
58       // Add extra flags specific to the sub-test.
59       Collections.addAll(flags, extraFlags);
60       // Add common flags
61       Collections.addAll(flags, new String[] {"-XX:AllocateHeapAt=" + test_dir,
62                                               "-Xlog:gc+heap=info",
63                                               "-version"});
64 
65       ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(true, flags);
66       OutputAnalyzer output = new OutputAnalyzer(pb.start());
67 
68       System.out.println("Output:\n" + output.getOutput());
69 
70       output.shouldContain("Successfully allocated Java heap at location");
71       output.shouldHaveExitValue(0);
72     }
73   }
74 }
