26 /* @test TestAllocateHeapAtMultiple.java
27  * @key gc
28  * @summary Test to check allocation of Java Heap with AllocateHeapAt option. Has multiple sub-tests to cover different code paths.
29  * @library /test/lib
30  * @modules java.base/jdk.internal.misc
31  * @requires vm.bits == "64" & vm.gc != "Z" & os.family != "aix"
32  * @run main/timeout=360 gc.TestAllocateHeapAtMultiple
33  */
34 
35 import jdk.test.lib.JDKToolFinder;
36 import jdk.test.lib.process.ProcessTools;
37 import jdk.test.lib.process.OutputAnalyzer;
38 import java.util.ArrayList;
39 import java.util.Collections;
40 
41 public class TestAllocateHeapAtMultiple {
42   public static void main(String args[]) throws Exception {
43     ArrayList<String> vmOpts = new ArrayList<>();
44     String[] testVmOpts = null;
45 
46     String test_dir = System.getProperty("test.dir", ".");
47 
48     String testVmOptsStr = System.getProperty("test.java.opts");
49     if (!testVmOptsStr.isEmpty()) {
50       testVmOpts = testVmOptsStr.split(" ");
51     }
52 
53     // Extra options for each of the sub-tests
54     String[] extraOptsList = new String[] {
55       "-Xmx32m -Xms32m -XX:+UseCompressedOops",     // 1. With compressedoops enabled.
56       "-Xmx32m -Xms32m -XX:-UseCompressedOops",     // 2. With compressedoops disabled.
57       "-Xmx32m -Xms32m -XX:HeapBaseMinAddress=3g",  // 3. With user specified HeapBaseMinAddress.
58       "-Xmx32m -Xms32m -XX:+UseLargePages",         // 4. Set UseLargePages.
59       "-Xmx32m -Xms32m -XX:+UseNUMA"                // 5. Set UseNUMA.
60     };
61 
62     for(String extraOpts : extraOptsList) {
63       vmOpts.clear();
64       if(testVmOpts != null) {
65         Collections.addAll(vmOpts, testVmOpts);
66       }
67       // Add extra options specific to the sub-test.
68       String[] extraOptsArray = extraOpts.split(" ");
69       if(extraOptsArray != null) {
70         Collections.addAll(vmOpts, extraOptsArray);
71       }
72       // Add common options
73       Collections.addAll(vmOpts, new String[] {"-XX:AllocateHeapAt=" + test_dir,
74                                                "-Xlog:gc+heap=info",
75                                                "-version"});
76 
77       System.out.print("Testing:\n" + JDKToolFinder.getJDKTool("java"));
78       for (int i = 0; i < vmOpts.size(); i += 1) {
79         System.out.print(" " + vmOpts.get(i));
80       }
81       System.out.println();
82 
83       ProcessBuilder pb =
84         ProcessTools.createJavaProcessBuilder(vmOpts.toArray(new String[vmOpts.size()]));
85       OutputAnalyzer output = new OutputAnalyzer(pb.start());
86 
87       System.out.println("Output:\n" + output.getOutput());
88 
89       output.shouldContain("Successfully allocated Java heap at location");
90       output.shouldHaveExitValue(0);
91     }
92   }
93 }
