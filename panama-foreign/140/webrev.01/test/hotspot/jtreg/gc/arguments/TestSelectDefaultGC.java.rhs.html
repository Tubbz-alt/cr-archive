24 package gc.arguments;
25 
26 /*
27  * @test TestSelectDefaultGC
28  * @summary Test selection of GC when no GC option is specified
29  * @bug 8068582
30  * @key gc
31  * @library /test/lib
32  * @library /
33  * @requires vm.gc.Serial & vm.gc.G1
34  * @modules java.base/jdk.internal.misc
35  *          java.management
36  * @run driver gc.arguments.TestSelectDefaultGC
37  */
38 
39 import jdk.test.lib.process.OutputAnalyzer;
40 import jdk.test.lib.process.ProcessTools;
41 
42 public class TestSelectDefaultGC {
43     public static void assertVMOption(OutputAnalyzer output, String option, boolean value) {
44         output.shouldMatch(" " + option + " .*=.* " + value + " ");
45     }
46 
47     public static void testDefaultGC(boolean actAsServer) throws Exception {







48         // Start VM without specifying GC
49         ProcessBuilder pb = GCArguments.createJavaProcessBuilder(
50             "-XX:" + (actAsServer ? "+" : "-") + "AlwaysActAsServerClassMachine",
51             "-XX:" + (actAsServer ? "-" : "+") + "NeverActAsServerClassMachine",
52             "-XX:+PrintFlagsFinal",
53             "-version");
54         OutputAnalyzer output = new OutputAnalyzer(pb.start());
55         output.shouldHaveExitValue(0);
56 
57         final boolean isServer = actAsServer;
58 
59         // Verify GC selection
60         // G1 is default for server class machines
61         assertVMOption(output, "UseG1GC",            isServer);
62         // Serial is default for non-server class machines
63         assertVMOption(output, "UseSerialGC",        !isServer);
64     }
65 
66     public static void main(String[] args) throws Exception {
67         // Test server class machine
68         testDefaultGC(false);
69 
70         // Test non-server class machine
71         testDefaultGC(true);
72     }
73 }
