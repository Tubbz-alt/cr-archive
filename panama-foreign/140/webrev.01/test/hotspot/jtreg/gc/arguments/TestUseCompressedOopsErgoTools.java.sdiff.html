 63   public static boolean is64bitVM() {
 64     String val = System.getProperty("sun.arch.data.model");
 65     if (val == null) {
 66       throw new RuntimeException("Could not read sun.arch.data.model");
 67     }
 68     if (val.equals("64")) {
 69       return true;
 70     } else if (val.equals("32")) {
 71       return false;
 72     }
 73     throw new RuntimeException("Unexpected value " + val + " of sun.arch.data.model");
 74   }

 76   /**
 77    * Executes a new VM process with the given class and parameters.
 78    * @param vmargs Arguments to the VM to run
 79    * @param classname Name of the class to run
 80    * @param arguments Arguments to the class
 81    * @param useTestDotJavaDotOpts Use test.java.opts as part of the VM argument string
 82    * @return The OutputAnalyzer with the results for the invocation.
 83    */
 84   public static OutputAnalyzer runWhiteBoxTest(String[] vmargs, String classname, String[] arguments, boolean useTestDotJavaDotOpts) throws Exception {
 85     ArrayList<String> finalargs = new ArrayList<String>();
 86 
 87     String[] whiteboxOpts = new String[] {
 88       "-Xbootclasspath/a:.",
 89       "-XX:+UnlockDiagnosticVMOptions", "-XX:+WhiteBoxAPI",
 90       "-cp", System.getProperty("java.class.path"),
 91     };
 92 
 93     if (useTestDotJavaDotOpts) {
 94       // System.getProperty("test.java.opts") is '' if no options is set,
 95       // we need to skip such a result
 96       String[] externalVMOpts = new String[0];
 97       if (System.getProperty("test.java.opts") != null && System.getProperty("test.java.opts").length() != 0) {
 98         externalVMOpts = System.getProperty("test.java.opts").split(" ");
 99       }
100       finalargs.addAll(Arrays.asList(externalVMOpts));
101     }
102 
103     finalargs.addAll(Arrays.asList(vmargs));
104     finalargs.addAll(Arrays.asList(whiteboxOpts));
105     finalargs.add(classname);
106     finalargs.addAll(Arrays.asList(arguments));
107 
108     ProcessBuilder pb = GCArguments.createJavaProcessBuilder(finalargs.toArray(new String[0]));
109     OutputAnalyzer output = new OutputAnalyzer(pb.start());
110     output.shouldHaveExitValue(0);
111     return output;
112   }

114   private static String[] join(String[] part1, String part2) {
115     ArrayList<String> result = new ArrayList<String>();

117     result.add(part2);
118     return result.toArray(new String[0]);
119   }

121   public static void checkCompressedOopsErgo(String[] gcflags) throws Exception {
122     long maxHeapForCompressedOops = getMaxHeapForCompressedOops(gcflags);
123 
124     checkUseCompressedOops(gcflags, maxHeapForCompressedOops, true);
125     checkUseCompressedOops(gcflags, maxHeapForCompressedOops - 1, true);
126     checkUseCompressedOops(gcflags, maxHeapForCompressedOops + 1, false);
127 
128     // the use of HeapBaseMinAddress should not change the outcome
129     checkUseCompressedOops(join(gcflags, "-XX:HeapBaseMinAddress=32G"), maxHeapForCompressedOops, true);
130     checkUseCompressedOops(join(gcflags, "-XX:HeapBaseMinAddress=32G"), maxHeapForCompressedOops - 1, true);
131     checkUseCompressedOops(join(gcflags, "-XX:HeapBaseMinAddress=32G"), maxHeapForCompressedOops + 1, false);
132 
133     // use a different object alignment
134     maxHeapForCompressedOops = getMaxHeapForCompressedOops(join(gcflags, "-XX:ObjectAlignmentInBytes=16"));
135 
136     checkUseCompressedOops(join(gcflags, "-XX:ObjectAlignmentInBytes=16"), maxHeapForCompressedOops, true);
137     checkUseCompressedOops(join(gcflags, "-XX:ObjectAlignmentInBytes=16"), maxHeapForCompressedOops - 1, true);
138     checkUseCompressedOops(join(gcflags, "-XX:ObjectAlignmentInBytes=16"), maxHeapForCompressedOops + 1, false);
