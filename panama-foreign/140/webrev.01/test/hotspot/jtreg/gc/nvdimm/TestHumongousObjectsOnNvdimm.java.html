 26 /*
 27  * @test TestHumongousObjectsOnNvdimm
 28  * @summary Check that humongous objects reside in nv-dimm
 29  * @library /test/lib /
 30  * @requires vm.gc=="null" & os.family != "aix"
 31  * @requires test.vm.gc.nvdimm
 32  * @build sun.hotspot.WhiteBox
 33  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
 34  * @run driver gc.nvdimm.TestHumongousObjectsOnNvdimm
 35  */
 36 
 37 import jdk.test.lib.process.OutputAnalyzer;
 38 import jdk.test.lib.process.ProcessTools;
 39 import jdk.test.lib.Asserts;
 40 import sun.hotspot.WhiteBox;
 41 
 42 import java.util.ArrayList;
 43 import java.util.Collections;
 44 import gc.testlibrary.Helpers;
 45 
 46 /**
 47  * Test spawns HumongousObjectTest in a separate VM and expects that it
 48  * completes without a RuntimeException.
 49  */
 50 public class TestHumongousObjectsOnNvdimm {
 51 
 52     private static String[] commonFlags;
 53 
 54     public static void main(String args[]) throws Exception {
 55         commonFlags = new String[] {
 56             "-Xbootclasspath/a:.",
 57             "-XX:+UnlockExperimentalVMOptions",
 58             "-XX:+UnlockDiagnosticVMOptions",
 59             "-XX:+WhiteBoxAPI",
 60             "-XX:AllocateOldGenAt=" + System.getProperty("test.dir", "."),
 61             "-Xms10M", "-Xmx10M",
 62             "-XX:G1HeapRegionSize=1m"
 63         };
 64 
 65         // Test with G1 GC
 66         runTest("-XX:+UseG1GC");
 67     }
 68 
 69     private static void runTest(String... extraFlags) throws Exception {
 70         ArrayList<String> flags = new ArrayList<>();
 71         Collections.addAll(flags, commonFlags);
 72         Collections.addAll(flags, extraFlags);
 73         flags.add(HumongousObjectTest.class.getName());
 74 
 75         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(true, flags);
 76 
 77         OutputAnalyzer output = new OutputAnalyzer(pb.start());
 78         output.shouldHaveExitValue(0);
 79     }
 80 }
 81 
 82 /**
 83  * This class tests that a humongous object resides in NVDIMM.
 84  */
 85 class HumongousObjectTest {
 86     private static final WhiteBox WB = WhiteBox.getWhiteBox();
 87 
 88     private static void validateObject(Object o) {
 89         Asserts.assertTrue(WB.isObjectInOldGen(o),
 90                 "Object is supposed to be in OldGen");
 91 
 92         long obj_addr = WB.getObjectAddress(o);
 93         long nvdimm_heap_start = WB.nvdimmReservedStart();
 94         long nvdimm_heap_end = WB.nvdimmReservedEnd();
 95 
 96         Asserts.assertTrue(WB.g1BelongsToHumongousRegion(obj_addr), "Object address should be in Humongous set");
 97         Asserts.assertTrue(obj_addr >= nvdimm_heap_start && obj_addr < nvdimm_heap_end,
 98                 "Humongous object does not reside in NVDIMM");
 99     }
100 
101     public static void main(String args[]) throws Exception {
102         // allocate an humongous object
103         int byteArrayMemoryOverhead = Helpers.detectByteArrayAllocationOverhead();
104         int MinByteArrayHumongousSize = (WB.g1RegionSize() / 2) - byteArrayMemoryOverhead + 1;
105         byte[] obj = new byte[MinByteArrayHumongousSize];
106 
107         validateObject(obj);
108     }
109 }
