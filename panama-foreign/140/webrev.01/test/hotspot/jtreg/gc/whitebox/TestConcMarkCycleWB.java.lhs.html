26 /*
27  * @test TestConMarkCycleWB
28  * @bug 8065579
29  * @requires vm.gc.G1
30  * @library /test/lib
31  * @modules java.base/jdk.internal.misc
32  *          java.compiler
33  *          java.management
34  *          jdk.internal.jvmstat/sun.jvmstat.monitor
35  * @build sun.hotspot.WhiteBox
36  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
37  *                              sun.hotspot.WhiteBox$WhiteBoxPermission
38  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -XX:+UseG1GC gc.whitebox.TestConcMarkCycleWB
39  * @summary Verifies that ConcurrentMarking-related WB works properly
40  */
41 import static jdk.test.lib.Asserts.assertTrue;
42 import sun.hotspot.WhiteBox;
43 
44 public class TestConcMarkCycleWB {
45 
46     public static void main(String[] args) throws Exception {
47         WhiteBox wb = WhiteBox.getWhiteBox();
48 
49         wb.youngGC();
50         assertTrue(wb.g1StartConcMarkCycle());
51         while (wb.g1InConcurrentMark()) {
52             Thread.sleep(5);
53         }
54 
55         wb.fullGC();
56         assertTrue(wb.g1StartConcMarkCycle());
57         while (wb.g1InConcurrentMark()) {
58             Thread.sleep(5);
59         }
60         assertTrue(wb.g1StartConcMarkCycle());
61     }
62 }
