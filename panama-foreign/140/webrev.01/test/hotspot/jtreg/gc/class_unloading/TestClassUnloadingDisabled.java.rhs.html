 97 import java.io.File;
 98 import java.io.IOException;
 99 import java.nio.file.Files;
100 import java.nio.file.Path;
101 import java.nio.file.Paths;
102 
103 import sun.hotspot.WhiteBox;
104 
105 import static jdk.test.lib.Asserts.*;
106 
107 public class TestClassUnloadingDisabled {
108     public static void main(String args[]) throws Exception {
109         final WhiteBox wb = WhiteBox.getWhiteBox();
110         // Fetch the dir where the test class and the class
111         // to be loaded resides.
112         String classDir = TestClassUnloadingDisabled.class.getProtectionDomain().getCodeSource().getLocation().getPath();
113         String className = "gc.class_unloading.ClassToLoadUnload"; // can not use ClassToLoadUnload.class.getName() as that would load the class
114 
115         assertFalse(wb.isClassAlive(className), "Should not be loaded yet");
116 
117         // The NoPDClassLoader handles loading classes in the test directory
118         // and loads them without a protection domain, which in some cases
119         // keeps the class live regardless of marking state.
120         NoPDClassLoader nopd = new NoPDClassLoader(classDir);
121         nopd.loadClass(className);
122 
123         assertTrue(wb.isClassAlive(className), "Class should be loaded");
124 
125         // Clear the class-loader, class and object references to make
126         // class unloading possible.
127         nopd = null;
128 
129         System.gc();
130         assertTrue(wb.isClassAlive(className), "Class should not have ben unloaded");
131     }
132 }
133 
134 class NoPDClassLoader extends ClassLoader {
135     String path;
136 
137     NoPDClassLoader(String path) {
138         this.path = path;
139     }
140 
141     public Class<?> loadClass(String name) throws ClassNotFoundException {
142         byte[] cls = null;
143         File f = new File(path,name + ".class");
144 
145         // Delegate class loading if class not present in the given
146         // directory.
147         if (!f.exists()) {
148             return super.loadClass(name);
149         }
150 
151         try {
152             Path path = Paths.get(f.getAbsolutePath());
153             cls = Files.readAllBytes(path);
154         } catch (IOException e) {
155             throw new ClassNotFoundException(name);
156         }
157 
158         // Define class with no protection domain and resolve it.
159         return defineClass(name, cls, 0, cls.length, null);
160     }
161 }
