24 package gc;
25 
26 /*
27  * @test TestNumWorkerOutputG1
28  * @bug 8165292
29  * @summary Check that when PrintGCDetails is enabled, gc,task output is printed only once per collection.
30  * @key gc
31  * @requires vm.gc.G1
32  * @modules java.base/jdk.internal.misc
33  * @library /test/lib
34  * @build sun.hotspot.WhiteBox
35  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
36  * @run driver gc.TestNumWorkerOutput UseG1GC
37  */
38 
39 import sun.hotspot.WhiteBox;
40 
41 import java.util.regex.Matcher;
42 import java.util.regex.Pattern;
43 
44 import jdk.test.lib.process.OutputAnalyzer;
45 import jdk.test.lib.process.ProcessTools;
46 
47 public class TestNumWorkerOutput {
48 
49     public static void checkPatternOnce(String pattern, String what) throws Exception {
50         Pattern r = Pattern.compile(pattern);
51         Matcher m = r.matcher(what);
52 
53         if (!m.find()) {
54             throw new RuntimeException("Could not find pattern " + pattern + " in output");
55         }
56         if (m.find()) {
57             throw new RuntimeException("Could find pattern " + pattern + " in output more than once");
58         }
59     }
60 
61     public static void runTest(String gcArg) throws Exception {
62         ProcessBuilder pb = ProcessTools.createJavaProcessBuilder(
63             "-Xbootclasspath/a:.",
64             "-XX:+UnlockExperimentalVMOptions",
65             "-XX:+UnlockDiagnosticVMOptions",
66             "-XX:+WhiteBoxAPI",
67             "-XX:+" + gcArg,
68             "-Xmx10M",
69             "-XX:+PrintGCDetails",
70             GCTest.class.getName());



71         OutputAnalyzer output = new OutputAnalyzer(pb.start());
72 
73         output.shouldHaveExitValue(0);
74 
75         System.out.println(output.getStdout());
76 
77         String stdout = output.getStdout();
78 
79         checkPatternOnce(".*[info.*].*[gc,task.*].*GC\\(0\\) .*Using \\d+ workers of \\d+ for evacuation.*", stdout);
80     }
81 
82     public static void main(String[] args) throws Exception {
83         runTest(args[0]);
84     }
85 
86     static class GCTest {
87         private static final WhiteBox WB = WhiteBox.getWhiteBox();
88 
89         public static Object holder;
90 
91         public static void main(String [] args) {
92             holder = new byte[100];
93             WB.youngGC();
94             System.out.println(holder);
95         }
96     }
97 }
