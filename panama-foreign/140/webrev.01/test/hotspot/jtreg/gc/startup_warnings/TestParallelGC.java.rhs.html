24 package gc.startup_warnings;
25 
26 /*
27 * @test TestParallelGC
28 * @key gc
29 * @bug 8006398
30 * @requires vm.gc.Parallel
31 * @summary Test that ParallelGC does not print a warning message
32 * @library /test/lib
33 * @modules java.base/jdk.internal.misc
34 *          java.management
35 * @run driver gc.startup_warnings.TestParallelGC
36 */
37 
38 import jdk.test.lib.process.ProcessTools;
39 import jdk.test.lib.process.OutputAnalyzer;
40 
41 
42 public class TestParallelGC {
43 
44   public static void main(String args[]) throws Exception {
45     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:+UseParallelGC", "-version");
46     OutputAnalyzer output = new OutputAnalyzer(pb.start());
47     output.shouldNotContain("deprecated");
48     output.shouldNotContain("error");
49     output.shouldHaveExitValue(0);
50   }
51 
52 }
