24 package gc.startup_warnings;
25 
26 /*
27 * @test TestSerialGC
28 * @key gc
29 * @bug 8006398
30 * @summary Test that SerialGC does not print a warning message
31 * @library /test/lib
32 * @modules java.base/jdk.internal.misc
33 *          java.management
34 * @run driver gc.startup_warnings.TestSerialGC
35 */
36 
37 import jdk.test.lib.process.ProcessTools;
38 import jdk.test.lib.process.OutputAnalyzer;
39 
40 
41 public class TestSerialGC {
42 
43   public static void main(String args[]) throws Exception {
44     ProcessBuilder pb = ProcessTools.createJavaProcessBuilder("-XX:+UseSerialGC", "-version");
45     OutputAnalyzer output = new OutputAnalyzer(pb.start());
46     output.shouldNotContain("deprecated");
47     output.shouldNotContain("error");
48     output.shouldHaveExitValue(0);
49   }
50 
51 }
