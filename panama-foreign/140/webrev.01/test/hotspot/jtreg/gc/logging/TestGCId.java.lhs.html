24 package gc.logging;
25 
26 /*
27  * @test TestGCId
28  * @bug 8043607
29  * @summary Ensure that the GCId is logged
30  * @key gc
31  * @library /test/lib
32  * @modules java.base/jdk.internal.misc
33  *          java.management
34  * @build sun.hotspot.WhiteBox
35  * @run driver ClassFileInstaller sun.hotspot.WhiteBox sun.hotspot.WhiteBox$WhiteBoxPermission
36  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI gc.logging.TestGCId
37  */
38 
39 import jdk.test.lib.process.OutputAnalyzer;
40 import jdk.test.lib.process.ProcessTools;
41 import jtreg.SkippedException;
42 import sun.hotspot.gc.GC;
43 
44 public class TestGCId {
45   public static void main(String[] args) throws Exception {
46     boolean noneGCSupported = true;
47 
48     if (GC.Parallel.isSupported()) {
49       noneGCSupported = false;
50       testGCId("UseParallelGC");
51     }
52     if (GC.G1.isSupported()) {
53       noneGCSupported = false;
54       testGCId("UseG1GC");
55     }
56     if (GC.Serial.isSupported()) {
57       noneGCSupported = false;
58       testGCId("UseSerialGC");
59     }
60     if (GC.Shenandoah.isSupported()) {
61       noneGCSupported = false;
62       testGCId("UseShenandoahGC");
63     }
64 
65     if (noneGCSupported) {
66       throw new SkippedException("Skipping test because none of Parallel/G1/Serial/Shenandoah is supported.");
67     }
68   }
69 
70   private static void verifyContainsGCIDs(OutputAnalyzer output) {
71     output.shouldMatch("\\[.*\\]\\[.*\\]\\[.*\\] GC\\(0\\) ");
72     output.shouldMatch("\\[.*\\]\\[.*\\]\\[.*\\] GC\\(1\\) ");
73     output.shouldHaveExitValue(0);
74   }
75 
76   private static void testGCId(String gcFlag) throws Exception {
77     ProcessBuilder pb_default =
78       ProcessTools.createJavaProcessBuilder("-XX:+UnlockExperimentalVMOptions", "-XX:+" + gcFlag, "-Xlog:gc", "-Xmx10M", GCTest.class.getName());
79     verifyContainsGCIDs(new OutputAnalyzer(pb_default.start()));
80   }
81 
82   static class GCTest {
83     private static byte[] garbage;
84     public static void main(String [] args) {
85       System.out.println("Creating garbage");
86       // create 128MB of garbage. This should result in at least one GC
87       for (int i = 0; i < 1024; i++) {
88         garbage = new byte[128 * 1024];
89       }
90       // do a system gc to get one more gc
91       System.gc();
92       System.out.println("Done");
93     }
94   }
95 }
