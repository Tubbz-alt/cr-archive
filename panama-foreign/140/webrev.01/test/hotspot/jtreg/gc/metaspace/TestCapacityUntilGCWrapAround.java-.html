35  *                              sun.hotspot.WhiteBox$WhiteBoxPermission
36  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI gc.metaspace.TestCapacityUntilGCWrapAround
37  */
38 
39 import sun.hotspot.WhiteBox;
40 
41 import jdk.test.lib.Asserts;
42 import jdk.test.lib.Platform;
43 
44 public class TestCapacityUntilGCWrapAround {
45     private static long MB = 1024 * 1024;
46     private static long GB = 1024 * MB;
47     private static long MAX_UINT = 4 * GB - 1; // On 32-bit platforms
48 
49     public static void main(String[] args) {
50         if (Platform.is32bit()) {
51             WhiteBox wb = WhiteBox.getWhiteBox();
52 
53             long before = wb.metaspaceCapacityUntilGC();
54             // Now force possible overflow of capacity_until_GC.
55             long after = wb.incMetaspaceCapacityUntilGC(MAX_UINT);
56 
57             Asserts.assertGTE(after, before,
58                               "Increasing with MAX_UINT should not cause wrap around: " + after + " < " + before);
59             Asserts.assertLTE(after, MAX_UINT,
60                               "Increasing with MAX_UINT should not cause value larger than MAX_UINT:" + after);
61         }
62     }
63 }
