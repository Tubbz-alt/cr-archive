33  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI BlobSanityTest
34  * @summary sanity testing of allocateCodeBlob, freeCodeBlob and getCodeBlob
35  */
36 
37 
38 import sun.hotspot.WhiteBox;
39 
40 import java.util.function.Consumer;
41 import jdk.test.lib.Utils;
42 
43 public class BlobSanityTest {
44 
45     private static void runTest(Consumer<Integer> consumer, int val, String testCaseName, Class<? extends Throwable>
46             expectedException) {
47             System.out.println("Calling " + testCaseName);
48             Utils.runAndCheckException(() -> consumer.accept(val), expectedException);
49             System.out.println("Looks ok");
50     }
51 
52     public static void main(String[] args) throws Exception {
53         System.out.println("Crash means that sanity check failed");
54 
55         WhiteBox wb = WhiteBox.getWhiteBox();
56 
57         runTest(wb::freeCodeBlob, 0, "wb::freeCodeBlob(0)", null);
58         runTest(wb::getCodeBlob, 0, "wb::getCodeBlob(0)", NullPointerException.class);
59         runTest(x -> wb.allocateCodeBlob(x, 0), -1, "wb::allocateCodeBlob(-1,0)", IllegalArgumentException.class);
60     }
61 }
