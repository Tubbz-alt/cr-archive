35  * @author igor.ignatyev@oracle.com

37 import jdk.test.lib.Platform;
38 
39 public class UintxTest {
40     private static final String FLAG_NAME = "VerifyGCStartAt";
41     private static final String FLAG_DEBUG_NAME = "CodeCacheMinimumUseSpace";
42     private static final Long[] TESTS = {0L, 100L, (long) Integer.MAX_VALUE,
43         (1L << 32L) - 1L, 1L << 32L};
44     private static final Long[] EXPECTED_64 = TESTS;
45     private static final Long[] EXPECTED_32 = {0L, 100L,
46         (long) Integer.MAX_VALUE, (1L << 32L) - 1L, 0L};
47 
48     public static void main(String[] args) throws Exception {
49         VmFlagTest.runTest(FLAG_NAME, TESTS,
50             Platform.is64bit() ? EXPECTED_64 : EXPECTED_32,
51             VmFlagTest.WHITE_BOX::setUintxVMFlag,
52             VmFlagTest.WHITE_BOX::getUintxVMFlag);
53         VmFlagTest.runTest(FLAG_DEBUG_NAME, VmFlagTest.WHITE_BOX::getUintxVMFlag);
54     }
55 }
