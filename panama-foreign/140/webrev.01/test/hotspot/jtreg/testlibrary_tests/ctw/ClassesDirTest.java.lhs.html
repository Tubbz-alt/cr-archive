43 import java.nio.file.Files;
44 import java.nio.file.Paths;
45 import java.nio.file.StandardCopyOption;
46 
47 public class ClassesDirTest extends CtwTest {
48     private static final String[] SHOULD_CONTAIN
49             = {"# dir: classes", "Done (2 classes, 6 methods, "};
50 
51     private ClassesDirTest() {
52         super(SHOULD_CONTAIN);
53     }
54 
55     public static void main(String[] args) throws Exception {
56         new ClassesDirTest().run(args);
57     }
58 
59     protected void prepare() throws Exception {
60         String path = "classes";
61         Files.createDirectory(Paths.get(path));
62         Files.move(Paths.get("Foo.class"), Paths.get(path, "Foo.class"),
63                 StandardCopyOption.REPLACE_EXISTING);
64         Files.move(Paths.get("Bar.class"), Paths.get(path, "Bar.class"),
65                 StandardCopyOption.REPLACE_EXISTING);
66     }
67 }
