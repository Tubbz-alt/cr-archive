45 import jdk.test.lib.process.OutputAnalyzer;
46 
47 public class JarsTest extends CtwTest {
48     private static final String[] SHOULD_CONTAIN
49             = {"# jar: foo.jar", "# jar: bar.jar",
50                     "Done (4 classes, 12 methods, "};
51 
52     private JarsTest() {
53         super(SHOULD_CONTAIN);
54     }
55 
56     public static void main(String[] args) throws Exception {
57         new JarsTest().run(args);
58     }
59 
60     protected void prepare() throws Exception {
61         ProcessBuilder pb = createJarProcessBuilder("cf", "foo.jar",
62                 "Foo.class", "Bar.class");
63         OutputAnalyzer output = new OutputAnalyzer(pb.start());
64         dump(output, "ctw-foo.jar");
65         output.shouldHaveExitValue(0);
66 
67         pb = createJarProcessBuilder("cf", "bar.jar", "Foo.class", "Bar.class");
68         output = new OutputAnalyzer(pb.start());
69         dump(output, "ctw-bar.jar");
70         output.shouldHaveExitValue(0);
71     }
72 
73 }
