45 import java.io.File;
46 import java.nio.file.Files;
47 import java.nio.file.Paths;
48 
49 import jdk.test.lib.process.OutputAnalyzer;
50 
51 public class JarDirTest extends CtwTest {
52     private static final String[] SHOULD_CONTAIN
53             = {"# jar_in_dir: jars",
54                     "# jar: jars" + File.separator +"foo.jar",
55                     "# jar: jars" + File.separator +"bar.jar",
56                     "Done (4 classes, 12 methods, "};
57 
58     private JarDirTest() {
59         super(SHOULD_CONTAIN);
60     }
61 
62     public static void main(String[] args) throws Exception {
63         new JarDirTest().run(args);
64     }
65 
66     protected void prepare() throws Exception {
67         String path = "jars";
68         Files.createDirectory(Paths.get(path));
69 
70         ProcessBuilder pb = createJarProcessBuilder("cf", "jars/foo.jar",
71                 "Foo.class", "Bar.class");
72         OutputAnalyzer output = new OutputAnalyzer(pb.start());
73         dump(output, "ctw-foo.jar");
74         output.shouldHaveExitValue(0);
75 
76         pb = createJarProcessBuilder("cf", "jars/bar.jar", "Foo.class",
77                 "Bar.class");
78         output = new OutputAnalyzer(pb.start());
79         dump(output, "ctw-bar.jar");
80         output.shouldHaveExitValue(0);
81     }
82 
83 }
