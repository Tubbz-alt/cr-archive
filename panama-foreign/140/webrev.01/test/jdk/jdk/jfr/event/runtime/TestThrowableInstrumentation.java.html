26 package jdk.jfr.event.runtime;
27 
28 import sun.hotspot.WhiteBox;
29 import java.util.Objects;
30 import jdk.test.lib.Platform;
31 
32 /**
33  * @test
34  * @bug 8153324
35  * @summary Verify instrumented Throwable bytecode by compiling it with C1.
36  * @requires vm.hasJFR
37  * @library /test/lib
38  * @modules java.base/jdk.internal.misc
39  *          java.management
40  * @requires vm.compMode!="Xint"
41  * @build  sun.hotspot.WhiteBox
42  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
43  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI
44  *                   -Xbatch -XX:StartFlightRecording=dumponexit=true jdk.jfr.event.runtime.TestThrowableInstrumentation
45  */
46 public class TestThrowableInstrumentation {
47     private static final WhiteBox WHITE_BOX = WhiteBox.getWhiteBox();
48     private static int COMP_LEVEL_SIMPLE = 1;
49 
50     private static boolean isTieredCompilationEnabled() {
51         return Boolean.valueOf(Objects.toString(WHITE_BOX.getVMFlag("TieredCompilation")));
52     }
53 
54     public static void main(String[] args) {
55         // Compile Throwable::<clinit> with C1 (if available)
56         if (!WHITE_BOX.enqueueInitializerForCompilation(java.lang.Throwable.class, COMP_LEVEL_SIMPLE)) {
57           if (!Platform.isServer() || isTieredCompilationEnabled() || Platform.isEmulatedClient()) {
58             throw new RuntimeException("Unable to compile Throwable::<clinit> with C1");
59           }
60         }
61     }
62 }
