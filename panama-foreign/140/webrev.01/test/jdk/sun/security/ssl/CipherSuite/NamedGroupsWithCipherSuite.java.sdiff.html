160     private static boolean groupSupportdByCipher(String group,
161             CipherSuite cipherSuite) {
162         return (group.startsWith("x")
163                         && xdhGroupSupportdByCipher(cipherSuite))
164                 || (group.startsWith("secp")
165                         && ecdhGroupSupportdByCipher(cipherSuite))
166                 || (group.startsWith("ffdhe")
167                         && ffdhGroupSupportdByCipher(cipherSuite));
168     }

170     private static boolean xdhGroupSupportdByCipher(
171             CipherSuite cipherSuite) {
172         return cipherSuite.keyExAlgorithm == null
173                 || cipherSuite.keyExAlgorithm == KeyExAlgorithm.ECDHE_RSA;
