  65     private static void test(String name, String headers, List<String> body, long delay)
  66             throws Exception
  67     {
  68         try (ServerSocket server = new ServerSocket()) {
  69             server.bind(new InetSocketAddress(InetAddress.getLoopbackAddress(), 0));
  70             String path = "/testChunkExt/" + name;
  71             URI uri = new URI("http",
  72                     null,
  73                     server.getInetAddress().getHostAddress(),
  74                     server.getLocalPort(),
  75                     path,
  76                     null,
  77                     null);
  78             HttpClient client = HttpClient.newBuilder()
  79                     .proxy(HttpClient.Builder.NO_PROXY)
  80                     .build();
  81             HttpRequest request = HttpRequest.newBuilder(uri).GET().build();
  82             System.out.printf("%s: %s%n", name, uri);
  83             CompletableFuture<HttpResponse<String>> respCF =
  84                     client.sendAsync(request, HttpResponse.BodyHandlers.ofString());
  85             boolean received = false;
  86             while (!received) {
  87                 System.out.printf("%s: Recieving request...%n", name);
  88                 try (Socket serverConn = server.accept()) {
  89                     InputStream sin = serverConn.getInputStream();
  90                     BufferedReader reader = new BufferedReader(new InputStreamReader(sin));
  91                     String line;
  92                     // read headers
  93                     while (!(line = reader.readLine()).isEmpty()) {
  94                         if (line.contains(path)) {
  95                             received = true;
  96                             System.out.printf("%s: Recieved request: %s%n", name, line);
  97                         }
  98                     }
  99                     if (!received) {
 100                         System.out.printf("%s: Unexpected headers received: dropping request.%n");
 101                         continue;
 102                     }
 103                     OutputStream os = serverConn.getOutputStream();
 104                     os.write(headers.getBytes(StandardCharsets.US_ASCII));
 105                     os.flush();
 106                     System.out.printf("%s: Response headers sent.%n", name);
 107                     for (String part : body) {
 108                         os.write(part.getBytes(StandardCharsets.UTF_8));
 109                         os.flush();
 110                         if (delay > 0) {
 111                             // Give a chance to the client side to process
 112                             // the bytes sent. This is mostly  useful
 113                             // for the byte-at-a-time scenario (REQ5)
 114                             Thread.sleep(delay);
 115                         }
 116                     }
 117                     System.out.printf("%s: Response body sent.%n", name);
 118                     HttpResponse<String> resp = respCF.join();
 119                     System.out.printf("%s: Response received:%n", name);
 120                     if (PRINT_BODY) {
 121                         System.out.println(resp.body());
 122                     } else {
 123                         System.out.println(resp);
 124                     }
 125                     validate(resp.body());
 126                     System.out.printf("%n%s: body validated%n%n", name);
 127                 }
 128             }
 129         }
 130     }

 132     private static void validate(String response) {
 133         for (int i=0; i<response.length(); i++) {
 134             if (response.charAt(i) != LOREM.charAt(i % LOREM.length())) {
 135                 throw new AssertionError("Mismatch at index " + i);
 136             }
 137         }
 138     }
