 25 package jdk.incubator.jpackage.internal;
 26 
 27 import java.nio.file.Path;
 28 import java.util.Map;
 29 
 30 
 31 /**
 32  * Application directory layout.
 33  */
 34 public final class ApplicationLayout implements PathGroup.Facade<ApplicationLayout> {
 35     enum PathRole {
 36         RUNTIME, APP, LAUNCHERS, DESKTOP, APP_MODS, DLLS, RELEASE
 37     }
 38 
 39     ApplicationLayout(Map<Object, Path> paths) {
 40         data = new PathGroup(paths);
 41     }
 42 
 43     private ApplicationLayout(PathGroup data) {
 44         this.data = data;
 45     }
 46 
 47     @Override
 48     public PathGroup pathGroup() {
 49         return data;
 50     }
 51 
 52     @Override
 53     public ApplicationLayout resolveAt(Path root) {
 54         return new ApplicationLayout(pathGroup().resolveAt(root));
 55     }
 56 
 57     /**
 58      * Path to launchers directory.
 59      */
 60     public Path launchersDirectory() {
 61         return pathGroup().getPath(PathRole.LAUNCHERS);
 62     }
 63 
 64     /**
 65      * Path to directory with dynamic libraries.
 66      */
 67     public Path dllDirectory() {
 68         return pathGroup().getPath(PathRole.DLLS);
 69     }
 70 
 71     /**
 72      * Path to application data directory.
 73      */
 74     public Path appDirectory() {
 75         return pathGroup().getPath(PathRole.APP);
 76     }
 77 
 78     /**
 79      * Path to Java runtime directory.
 80      */
 81     public Path runtimeDirectory() {
 82         return pathGroup().getPath(PathRole.RUNTIME);
 83     }
 84 
 85     /**
 86      * Path to application mods directory.
 87      */
 88     public Path appModsDirectory() {
 89         return pathGroup().getPath(PathRole.APP_MODS);
 90     }
 91 
 92     /**
 93      * Path to directory with application's desktop integration files.
 94      */
 95     public Path destktopIntegrationDirectory() {
 96         return pathGroup().getPath(PathRole.DESKTOP);
 97     }
 98 
 99     /**
100      * Path to release file in the Java runtime directory.
101      */
102     public Path runtimeRelease() {
103         return pathGroup().getPath(PathRole.RELEASE);
104     }
105 
106     static ApplicationLayout linuxAppImage() {
107         return new ApplicationLayout(Map.of(
108                 PathRole.LAUNCHERS, Path.of("bin"),
109                 PathRole.APP, Path.of("lib/app"),
110                 PathRole.RUNTIME, Path.of("lib/runtime"),
111                 PathRole.DESKTOP, Path.of("lib"),
112                 PathRole.DLLS, Path.of("lib"),
113                 PathRole.APP_MODS, Path.of("lib/app/mods"),
114                 PathRole.RELEASE, Path.of("lib/runtime/release")
115         ));
116     }
117 
118     static ApplicationLayout windowsAppImage() {
119         return new ApplicationLayout(Map.of(
120                 PathRole.LAUNCHERS, Path.of(""),
121                 PathRole.APP, Path.of("app"),
122                 PathRole.RUNTIME, Path.of("runtime"),
123                 PathRole.DESKTOP, Path.of(""),
124                 PathRole.DLLS, Path.of(""),
125                 PathRole.APP_MODS, Path.of("app/mods"),
126                 PathRole.RELEASE, Path.of("runtime/release")
127         ));
128     }
129 
130     static ApplicationLayout macAppImage() {
131         return new ApplicationLayout(Map.of(
132                 PathRole.LAUNCHERS, Path.of("Contents/MacOS"),
133                 PathRole.APP, Path.of("Contents/app"),
134                 PathRole.RUNTIME, Path.of("Contents/runtime"),
135                 PathRole.DESKTOP, Path.of("Contents/Resources"),
136                 PathRole.DLLS, Path.of("Contents/MacOS"),
137                 PathRole.APP_MODS, Path.of("Contents/app/mods"),
138                 PathRole.RELEASE, Path.of("Contents/runtime/Contents/Home/release")
139         ));
140     }
141 
142     public static ApplicationLayout platformAppImage() {
143         if (Platform.isWindows()) {
144             return windowsAppImage();
145         }
146 
147         if (Platform.isLinux()) {
148             return linuxAppImage();
149         }
150 
151         if (Platform.isMac()) {
152             return macAppImage();
153         }
154 
155         throw Platform.throwUnknownPlatformError();
156     }
157 
158     public static ApplicationLayout javaRuntime() {
159         return new ApplicationLayout(Map.of(PathRole.RUNTIME, Path.of("")));
160     }
161 
162     private final PathGroup data;
163 }
