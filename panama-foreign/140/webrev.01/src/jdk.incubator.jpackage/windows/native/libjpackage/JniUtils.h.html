33 namespace jni {
34 
35 struct JniObjWithEnv {
36     JniObjWithEnv(): env(0), obj(0) {
37 
38     }
39 
40     JniObjWithEnv(JNIEnv *env, jobject obj) : env(env), obj(obj) {
41     }
42 
43     bool operator == (const JniObjWithEnv& other) const {
44         return env == other.env && obj == other.obj;
45     }
46 
47     bool operator != (const JniObjWithEnv& other) const {
48         return ! operator == (other);
49     }
50 
51     JNIEnv *env;
52     jobject obj;
53 
54     struct LocalRefDeleter {
55         typedef JniObjWithEnv pointer;
56 
57         void operator()(pointer v);
58     };
59 };
60 
61 typedef std::unique_ptr<JniObjWithEnv, JniObjWithEnv::LocalRefDeleter> LocalRef;
