28 /**
29  * Exception that is thrown if options don't follow the syntax of the command.
30  */
31 final class UserSyntaxException extends Exception {
32     private static final long serialVersionUID = 3437009454344160933L;
33 
34     /**
35      * The error message.
36      *
37      * The first letter should not be capitalized, so a context can be printed prior
38      * to the error message.
39      *
40      * @param errorMessage
41      */
42     public UserSyntaxException(String message) {
43         super(message);
44     }
45 }
