26 package jdk.jfr.consumer;
27 
28 import jdk.jfr.internal.consumer.ObjectContext;
29 
30 /**
31  * A recorded Java thread group.
32  *
33  * @since 9
34  */
35 public final class RecordedThreadGroup extends RecordedObject {
36     // package private
37     RecordedThreadGroup(ObjectContext objectContext, Object[] values) {
38         super(objectContext, values);
39     }
40 
41     /**
42      * Returns the name of the thread group, or {@code null} if doesn't exist.
43      *
44      * @return the thread group name, or {@code null} if doesn't exist
45      */
46     public String getName() {
47         return getTyped("name", String.class, null);
48     }
49 
50     /**
51      * Returns the parent thread group, or {@code null} if it doesn't exist.
52      *
53      * @return parent thread group, or {@code null} if it doesn't exist
54      */
55     public RecordedThreadGroup getParent() {
56         return getTyped("parent", RecordedThreadGroup.class, null);
57     }
58 }
