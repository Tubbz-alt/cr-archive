26 package jdk.jfr;
27 
28 import java.lang.annotation.ElementType;
29 import java.lang.annotation.Retention;
30 import java.lang.annotation.RetentionPolicy;
31 import java.lang.annotation.Target;
32 
33 /**
34  * Event field annotation, specifies that the value is a duration.
35  *
36  * @since 9
37  */
38 @MetadataDefinition
39 @ContentType
40 @Label("Timespan")
41 @Description("A duration, measured in nanoseconds by default")
42 @Retention(RetentionPolicy.RUNTIME)
43 @Target({ ElementType.FIELD, ElementType.TYPE, ElementType.METHOD})
44 public @interface Timespan {
45     /**
46      * Unit for ticks.
47      */
48     public static final String TICKS = "TICKS";
49     /**
50      * Unit for seconds.
51      */
52     public static final String SECONDS = "SECONDS";
53     /**
54      * Unit for milliseconds.
55      */
56     public static final String MILLISECONDS = "MILLISECONDS";
57     /**
58      * Unit for nanoseconds.
59      */
60     public static final String NANOSECONDS = "NANOSECONDS";
61 
62     /**
63      * Unit for microseconds.
64      */
65     public static final String MICROSECONDS = "MICROSECONDS";
66 
67     /**
68      * Returns the unit of measure for the time span.
69      * <p>
70      * By default, the unit is nanoseconds.
71      *
72      * @return the time span unit, default {@code #NANOSECONDS}, not {@code null}
73      */
74     String value() default NANOSECONDS;
75 }
