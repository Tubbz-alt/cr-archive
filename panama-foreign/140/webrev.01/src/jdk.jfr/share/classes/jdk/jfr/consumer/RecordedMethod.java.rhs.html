 26 package jdk.jfr.consumer;
 27 
 28 import java.lang.reflect.Modifier;
 29 
 30 import jdk.jfr.internal.consumer.ObjectContext;
 31 
 32 /**
 33  * A recorded method.
 34  *
 35  * @since 9
 36  */
 37 public final class RecordedMethod extends RecordedObject {
 38 
 39     // package private
 40     RecordedMethod(ObjectContext objectContext, Object[] values) {
 41         super(objectContext, values);
 42     }
 43 
 44     /**
 45      * Returns the class this method belongs to, if it belong to a Java frame.
 46      * <p>
 47      * To ensure this is a Java frame, use the {@link RecordedFrame#isJavaFrame()}
 48      * method.
 49      *
 50      * @return the class, may be {@code null} if not a Java frame
 51      *
 52      * @see RecordedFrame#isJavaFrame()
 53      */
 54     public RecordedClass getType() {
 55         return getTyped("type", RecordedClass.class, null);
 56     }
 57 
 58     /**
 59      * Returns the name of this method, for example {@code "toString"}.
 60      * <p>
 61      * If this method doesn't belong to a Java frame the result is undefined.
 62      *
 63      * @return method name, or {@code null} if doesn't exist
 64      *
 65      * @see RecordedFrame#isJavaFrame()
 66      */
 67     public String getName() {
 68         return getTyped("name", String.class, null);
 69     }
 70 
 71     /**
 72      * Returns the method descriptor for this method (for example,
 73      * {@code "(Ljava/lang/String;)V"}).
 74      * <p>
 75      * If this method doesn't belong to a Java frame then the result is undefined.


 76      *
 77      * @return method descriptor
 78      *
 79      * @see RecordedFrame#isJavaFrame()
 80      * @jvms 4.3 Descriptors
 81      */
 82     public String getDescriptor() {
 83         return getTyped("descriptor", String.class, null);
 84     }
 85 
 86     /**
 87      * Returns the modifiers for this method.
 88      * <p>
 89      * If this method doesn't belong to a Java frame, then the result is undefined.
 90      *
 91      * @return the modifiers
 92      *
 93      * @see Modifier
 94      * @see RecordedFrame#isJavaFrame
 95      */
 96     public int getModifiers() {
 97         return getTyped("modifiers", Integer.class, Integer.valueOf(0));
 98     }
 99 
100     /**
101      * Returns whether this method is hidden (for example, wrapper code in a lambda
102      * expressions).
103      *
104      * @return {@code true} if method is hidden, {@code false} otherwise
105      */
106     public boolean isHidden() {
107         return getTyped("hidden", Boolean.class, Boolean.FALSE);
108     }
109 }
