26 package jdk.jfr;
27 
28 import java.lang.annotation.ElementType;
29 import java.lang.annotation.Retention;
30 import java.lang.annotation.RetentionPolicy;
31 import java.lang.annotation.Target;
32 
33 /**
34  * Meta annotation for defining new types of event metadata.
35  * <p>
36  * In the following example, a transaction event is defined with two
37  * user-defined annotations, {@code @Severity} and {@code @TransactionId}.
38  *
39  * <pre>
40  * <code>
41  *{@literal @}MetadataDefinition
42  *{@literal @}Label("Severity")
43  *{@literal @}Description("Value between 0 and 100 that indicates severity. 100 is most severe.")
44  *{@literal @}Retention(RetentionPolicy.RUNTIME)
45  *{@literal @}Target({ ElementType.TYPE })
46  * public {@literal @}interface {@literal @}Severity {
47  *   int value() default 50;
48  * }
49  *
50  *{@literal @}MetadataDefinition
51  *{@literal @}Label("Transaction Id")
52  *{@literal @}Relational
53  *{@literal @}Retention(RetentionPolicy.RUNTIME)
54  *{@literal @}Target({ ElementType.FIELD })
55  * public {@literal @}interface {@literal @}Severity {
56  * }
57  *
58  *{@literal @}Severity(80)
59  *{@literal @}Label("Transaction Blocked");
60  * class TransactionBlocked extends Event {
61  *  {@literal @}TransactionId
62  *  {@literal @}Label("Transaction");
63  *   long transactionId;
64  *
65  *  {@literal @}TransactionId
66  *  {@literal @}Label("Transaction Blocker");
67  *   long transactionId;
68  * }
69  *
70  * </code>
71  * </pre>
72  *
73  * Adding {@code @MetadataDefinition} to the declaration of {@code @Severity} and {@code @TransactionId}
74  * ensures the information is saved by Flight Recorder.
75  *
76  * @since 9
77  */
78 @Retention(RetentionPolicy.RUNTIME)
79 @Target({ ElementType.TYPE })
80 public @interface MetadataDefinition {
81 }
