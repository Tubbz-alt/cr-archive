25 package org.graalvm.compiler.hotspot.jdk15.test;
26 
27 import org.graalvm.compiler.replacements.test.MethodSubstitutionTest;
28 import org.junit.Test;
29 
30 import java.util.function.Supplier;
31 
32 /**
33  * As of JDK 15 {@code java.lang.Class::isHidden()} was added.
34  *
35  * @see "https://openjdk.java.net/jeps/371"
36  * @see "https://bugs.openjdk.java.net/browse/JDK-8238358"
37  */
38 public class ClassReplacementsTest extends MethodSubstitutionTest {
39 
40     @SuppressWarnings("all")
41     public static boolean isHidden(Class<?> clazz) {
42         return clazz.isHidden();
43     }
44 
45     @Test
46     public void testIsHidden() {
47         testGraph("isHidden");
48 
49         Supplier<Runnable> lambda = () -> () -> System.out.println("run");
50 
51         for (Class<?> c : new Class<?>[]{getClass(), Cloneable.class, int[].class, String[][].class, lambda.getClass(), lambda.get().getClass()}) {
52             test("isHidden", c);
53         }
54     }
55 
56 }
