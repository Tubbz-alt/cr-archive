 55 /**
 56  * Substitutions for {@link java.lang.Class} methods.
 57  */
 58 @ClassSubstitution(Class.class)
 59 public class HotSpotClassSubstitutions {
 60 
 61     @MethodSubstitution(isStatic = false)
 62     public static int getModifiers(final Class<?> thisObj) {
 63         KlassPointer klass = ClassGetHubNode.readClass(thisObj);
 64         if (klass.isNull()) {
 65             // Class for primitive type
 66             return Modifier.ABSTRACT | Modifier.FINAL | Modifier.PUBLIC;
 67         } else {
 68             return klass.readInt(klassModifierFlagsOffset(INJECTED_VMCONFIG), KLASS_MODIFIER_FLAGS_LOCATION);
 69         }
 70     }
 71 
 72     @MethodSubstitution(isStatic = false)
 73     public static boolean isInterface(final Class<?> thisObj) {
 74         KlassPointer klass = ClassGetHubNode.readClass(thisObj);
 75         if (klass.isNull()) {
 76             // Class for primitive type
 77             return false;
 78         } else {
 79             int accessFlags = klass.readInt(klassAccessFlagsOffset(INJECTED_VMCONFIG), KLASS_ACCESS_FLAGS_LOCATION);
 80             return (accessFlags & Modifier.INTERFACE) != 0;
 81         }
 82     }
 83 
 84     @MethodSubstitution(isStatic = false, optional = true)
 85     public static boolean isHidden(final Class<?> thisObj) {
 86         KlassPointer klass = ClassGetHubNode.readClass(thisObj);
 87         if (klass.isNull()) {
 88             // Class for primitive type
 89             return false;
 90         } else {
 91             int accessFlags = klass.readInt(klassAccessFlagsOffset(INJECTED_VMCONFIG), KLASS_ACCESS_FLAGS_LOCATION);
 92             return (accessFlags & (jvmAccIsHiddenClass(INJECTED_VMCONFIG))) != 0;
 93         }
 94     }
 95 
 96     @MethodSubstitution(isStatic = false)
 97     public static boolean isArray(final Class<?> thisObj) {
 98         KlassPointer klass = ClassGetHubNode.readClass(thisObj);
 99         if (klass.isNull()) {
100             // Class for primitive type
101             return false;
102         } else {
103             KlassPointer klassNonNull = ClassGetHubNode.piCastNonNull(klass, SnippetAnchorNode.anchor());
104             return klassIsArray(klassNonNull);
105         }
106     }
107 
108     @MethodSubstitution(isStatic = false)
109     public static boolean isPrimitive(final Class<?> thisObj) {
110         KlassPointer klass = ClassGetHubNode.readClass(thisObj);
111         return klass.isNull();
112     }
113 
114     @Fold
115     public static ResolvedJavaType getObjectType(@Fold.InjectedParameter MetaAccessProvider metaAccesss) {
116         return metaAccesss.lookupJavaType(Object.class);
117     }
118 
119     @MethodSubstitution(isStatic = false)
120     public static Class<?> getSuperclass(final Class<?> thisObj) {
121         KlassPointer klass = ClassGetHubNode.readClass(thisObj);
122         if (!klass.isNull()) {
123             KlassPointer klassNonNull = ClassGetHubNode.piCastNonNull(klass, SnippetAnchorNode.anchor());
124             int accessFlags = klassNonNull.readInt(klassAccessFlagsOffset(INJECTED_VMCONFIG), KLASS_ACCESS_FLAGS_LOCATION);
125             if ((accessFlags & Modifier.INTERFACE) == 0) {
126                 if (klassIsArray(klassNonNull)) {
127                     return ConstantNode.forClass(getObjectType(INJECTED_METAACCESS));
128                 } else {
129                     KlassPointer superKlass = klassNonNull.readKlassPointer(klassSuperKlassOffset(INJECTED_VMCONFIG), KLASS_SUPER_KLASS_LOCATION);
130                     if (superKlass.isNull()) {
131                         return null;
132                     } else {
133                         KlassPointer superKlassNonNull = ClassGetHubNode.piCastNonNull(superKlass, SnippetAnchorNode.anchor());
134                         return HubGetClassNode.readClass(superKlassNonNull);
135                     }
136                 }
137             }
138         } else {
139             // Class for primitive type
140         }
141         return null;
142     }
143 
144     @MethodSubstitution(isStatic = false)
145     public static Class<?> getComponentType(final Class<?> thisObj) {
146         KlassPointer klass = ClassGetHubNode.readClass(thisObj);
147         if (!klass.isNull()) {
148             KlassPointer klassNonNull = ClassGetHubNode.piCastNonNull(klass, SnippetAnchorNode.anchor());
149             if (klassIsArray(klassNonNull)) {
150                 return PiNode.asNonNullClass(klassNonNull.readObject(arrayKlassComponentMirrorOffset(INJECTED_VMCONFIG), ARRAY_KLASS_COMPONENT_MIRROR));
151             }
152         } else {
153             // Class for primitive type
154         }
155         return null;
156     }
157 }
