  55 /**
  56  * Implementation of an SSL socket.
  57  * <P>
  58  * This is a normal connection type socket, implementing SSL over some lower
  59  * level socket, such as TCP.  Because it is layered over some lower level
  60  * socket, it MUST override all default socket methods.
  61  * <P>
  62  * This API offers a non-traditional option for establishing SSL
  63  * connections.  You may first establish the connection directly, then pass
  64  * that connection to the SSL socket constructor with a flag saying which
  65  * role should be taken in the handshake protocol.  (The two ends of the
  66  * connection must not choose the same role!)  This allows setup of SSL
  67  * proxying or tunneling, and also allows the kind of "role reversal"
  68  * that is required for most FTP data transfers.
  69  *
  70  * @see javax.net.ssl.SSLSocket
  71  * @see SSLServerSocket
  72  *
  73  * @author David Brownell
  74  */
  75 public final class SSLSocketImpl
  76         extends BaseSSLSocketImpl implements SSLTransport {

 221     /**
 222      * Creates a server mode {@link Socket} layered over an
 223      * existing connected socket, and is able to read data which has
 224      * already been consumed/removed from the {@link Socket}'s
 225      * underlying {@link InputStream}.
 226      */
 227     SSLSocketImpl(SSLContextImpl sslContext, Socket sock,
 228             InputStream consumed, boolean autoClose) throws IOException {
 229         super(sock, consumed);
 230         // We always layer over a connected socket
 231         if (!sock.isConnected()) {
 232             throw new SocketException("Underlying socket is not connected");
 233         }
 234 
 235         this.sslContext = sslContext;
 236         HandshakeHash handshakeHash = new HandshakeHash();
 237         this.conContext = new TransportContext(sslContext, this,
 238                 new SSLSocketInputRecord(handshakeHash),
 239                 new SSLSocketOutputRecord(handshakeHash), false);
 240         this.autoClose = autoClose;
 241         doneConnect();
 242     }

 244     /**
 245      * Layer SSL traffic over an existing connection, rather than
 246      * creating a new connection.
 247      *
 248      * The existing connection may be used only for SSL traffic (using this
 249      * SSLSocket) until the SSLSocket.close() call returns. However, if a
 250      * protocol error is detected, that existing connection is automatically
 251      * closed.
 252      * <p>
 253      * This particular constructor always uses the socket in the
 254      * role of an SSL client. It may be useful in cases which start
 255      * using SSL after some initial data transfers, for example in some
 256      * SSL tunneling applications or as part of some kinds of application
 257      * protocols which negotiate use of a SSL based security.
 258      */
 259     SSLSocketImpl(SSLContextImpl sslContext, Socket sock,
 260             String peerHost, int port, boolean autoClose) throws IOException {
 261         super(sock);
 262         // We always layer over a connected socket
 263         if (!sock.isConnected()) {
 264             throw new SocketException("Underlying socket is not connected");
 265         }
 266 
 267         this.sslContext = sslContext;
 268         HandshakeHash handshakeHash = new HandshakeHash();
 269         this.conContext = new TransportContext(sslContext, this,
 270                 new SSLSocketInputRecord(handshakeHash),
 271                 new SSLSocketOutputRecord(handshakeHash), true);
 272         this.peerHost = peerHost;
 273         this.autoClose = autoClose;
 274         doneConnect();
 275     }

 277     @Override
 278     public void connect(SocketAddress endpoint,
 279             int timeout) throws IOException {
 280 
 281         if (isLayered()) {
 282             throw new SocketException("Already connected");
 283         }
 284 
 285         if (!(endpoint instanceof InetSocketAddress)) {
 286             throw new SocketException(
 287                     "Cannot handle non-Inet socket addresses.");
 288         }
 289 
 290         super.connect(endpoint, timeout);
 291         doneConnect();
 292     }

 410     @Override
 411     public void startHandshake() throws IOException {
 412         if (!isConnected) {
 413             throw new SocketException("Socket is not connected");
 414         }
 415 
 416         if (conContext.isBroken || conContext.isInboundClosed() ||
 417                 conContext.isOutboundClosed()) {
 418             throw new SocketException("Socket has been closed or broken");
 419         }
 420 
 421         handshakeLock.lock();
 422         try {
 423             // double check the context status
 424             if (conContext.isBroken || conContext.isInboundClosed() ||
 425                     conContext.isOutboundClosed()) {
 426                 throw new SocketException("Socket has been closed or broken");
 427             }
 428 
 429             try {
 430                 conContext.kickstart();
 431 
 432                 // All initial handshaking goes through this operation until we
 433                 // have a valid SSL connection.
 434                 //
 435                 // Handle handshake messages only, need no application data.
 436                 if (!conContext.isNegotiated) {
 437                     readHandshakeRecord();
 438                 }
 439             } catch (InterruptedIOException iioe) {
 440                 handleException(iioe);
 441             } catch (IOException ioe) {
 442                 throw conContext.fatal(Alert.HANDSHAKE_FAILURE,
 443                     "Couldn't kickstart handshaking", ioe);
 444             } catch (Exception oe) {    // including RuntimeException
 445                 handleException(oe);
 446             }
 447         } finally {
 448             handshakeLock.unlock();
 449         }
 450     }

 543     // Please don't synchronized this method.  Otherwise, the read and close
 544     // locks may be deadlocked.
 545     @Override
 546     public void close() throws IOException {
 547         if (tlsIsClosed) {
 548             return;
 549         }
 550 
 551         if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
 552             SSLLogger.fine("duplex close of SSLSocket");
 553         }
 554 
 555         try {
 556             // shutdown output bound, which may have been closed previously.
 557             if (!isOutputShutdown()) {
 558                 duplexCloseOutput();
 559             }
 560 
 561             // shutdown input bound, which may have been closed previously.
 562             if (!isInputShutdown()) {
 563                 duplexCloseInput();
 564             }
 565 
 566             if (!isClosed()) {
 567                 // close the connection directly
 568                 closeSocket(false);
 569             }
 570         } catch (IOException ioe) {
 571             // ignore the exception
 572             if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
 573                 SSLLogger.warning("SSLSocket duplex close failed", ioe);
 574             }
 575         } finally {
 576             tlsIsClosed = true;
 577         }
 578     }

 580     /**
 581      * Duplex close, start from closing outbound.
 582      *
 583      * For TLS 1.2 [RFC 5246], unless some other fatal alert has been
 584      * transmitted, each party is required to send a close_notify alert
 585      * before closing the write side of the connection.  The other party
 586      * MUST respond with a close_notify alert of its own and close down
 587      * the connection immediately, discarding any pending writes.  It is
 588      * not required for the initiator of the close to wait for the responding
 589      * close_notify alert before closing the read side of the connection.
 590      *
 591      * For TLS 1.3, Each party MUST send a close_notify alert before
 592      * closing its write side of the connection, unless it has already sent
 593      * some error alert.  This does not have any effect on its read side of
 594      * the connection.  Both parties need not wait to receive a close_notify
 595      * alert before closing their read side of the connection, though doing
 596      * so would introduce the possibility of truncation.
 597      *
 598      * In order to support user initiated duplex-close for TLS 1.3 connections,
 599      * the user_canceled alert is used together with the close_notify alert.
 600      */
 601     private void duplexCloseOutput() throws IOException {
 602         boolean useUserCanceled = false;
 603         boolean hasCloseReceipt = false;
 604         if (conContext.isNegotiated) {
 605             if (!conContext.protocolVersion.useTLS13PlusSpec()) {
 606                 hasCloseReceipt = true;
 607             } else {
 608                 // Use a user_canceled alert for TLS 1.3 duplex close.
 609                 useUserCanceled = true;
 610             }
 611         } else if (conContext.handshakeContext != null) {   // initial handshake
 612             // Use user_canceled alert regardless the protocol versions.
 613             useUserCanceled = true;
 614 
 615             // The protocol version may have been negotiated.
 616             ProtocolVersion pv = conContext.handshakeContext.negotiatedProtocol;
 617             if (pv == null || (!pv.useTLS13PlusSpec())) {
 618                 hasCloseReceipt = true;
 619             }
 620         }
 621 
 622         // Need a lock here so that the user_canceled alert and the
 623         // close_notify alert can be delivered together.
 624         int linger = getSoLinger();
 625         if (linger >= 0) {
 626             // don't wait more than SO_LINGER for obtaining the
 627             // the lock.
 628             //
 629             // keep and clear the current thread interruption status.
 630             boolean interrupted = Thread.interrupted();
 631             try {
 632                 if (conContext.outputRecord.recordLock.tryLock() ||
 633                         conContext.outputRecord.recordLock.tryLock(
 634                                 linger, TimeUnit.SECONDS)) {
 635                     try {
 636                         handleClosedNotifyAlert(useUserCanceled);
 637                     } finally {
 638                         conContext.outputRecord.recordLock.unlock();
 639                     }
 640                 } else {
 641                     // For layered, non-autoclose sockets, we are not
 642                     // able to bring them into a usable state, so we
 643                     // treat it as fatal error.
 644                     if (!super.isOutputShutdown()) {
 645                         if (isLayered() && !autoClose) {
 646                             throw new SSLException(
 647                                     "SO_LINGER timeout, " +
 648                                     "close_notify message cannot be sent.");
 649                         } else {
 650                             super.shutdownOutput();
 651                             if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
 652                                 SSLLogger.warning(
 653                                     "SSLSocket output duplex close failed: " +
 654                                     "SO_LINGER timeout, " +
 655                                     "close_notify message cannot be sent.");
 656                             }
 657                         }
 658                     }
 659 
 660                     // RFC2246 requires that the session becomes
 661                     // unresumable if any connection is terminated
 662                     // without proper close_notify messages with
 663                     // level equal to warning.
 664                     //
 665                     // RFC4346 no longer requires that a session not be
 666                     // resumed if failure to properly close a connection.
 667                     //
 668                     // We choose to make the session unresumable if
 669                     // failed to send the close_notify message.
 670                     //
 671                     conContext.conSession.invalidate();
 672                     if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
 673                         SSLLogger.warning(
 674                                 "Invalidate the session: SO_LINGER timeout, " +
 675                                 "close_notify message cannot be sent.");
 676                     }
 677                 }
 678             } catch (InterruptedException ex) {
 679                 // keep interrupted status
 680                 interrupted = true;
 681             }
 682 
 683             // restore the interrupted status
 684             if (interrupted) {
 685                 Thread.currentThread().interrupt();
 686             }
 687         } else {
 688             conContext.outputRecord.recordLock.lock();
 689             try {
 690                 handleClosedNotifyAlert(useUserCanceled);
 691             } finally {
 692                 conContext.outputRecord.recordLock.unlock();
 693             }
 694         }
 695 
 696         if (!isInputShutdown()) {
 697             bruteForceCloseInput(hasCloseReceipt);
 698         }
 699     }

 701     private void handleClosedNotifyAlert(
 702             boolean useUserCanceled) throws IOException {
 703         try {
 704             // send a user_canceled alert if needed.
 705             if (useUserCanceled) {
 706                 conContext.warning(Alert.USER_CANCELED);
 707             }
 708 
 709             // send a close_notify alert
 710             conContext.warning(Alert.CLOSE_NOTIFY);
 711         } finally {
 712             if (!conContext.isOutboundClosed()) {
 713                 conContext.outputRecord.close();
 714             }
 715 
 716             if (!super.isOutputShutdown() &&
 717                     (autoClose || !isLayered())) {
 718                 super.shutdownOutput();
 719             }
 720         }
 721     }

 723     /**
 724      * Duplex close, start from closing inbound.
 725      *
 726      * This method should only be called when the outbound has been closed,
 727      * but the inbound is still open.
 728      */
 729     private void duplexCloseInput() throws IOException {
 730         boolean hasCloseReceipt = false;
 731         if (conContext.isNegotiated &&
 732                 !conContext.protocolVersion.useTLS13PlusSpec()) {
 733             hasCloseReceipt = true;
 734         }   // No close receipt if handshake has no completed.
 735 
 736         bruteForceCloseInput(hasCloseReceipt);
 737     }

 739     /**
 740      * Brute force close the input bound.
 741      *
 742      * This method should only be called when the outbound has been closed,
 743      * but the inbound is still open.
 744      */
 745     private void bruteForceCloseInput(
 746             boolean hasCloseReceipt) throws IOException {
 747         if (hasCloseReceipt) {
 748             // It is not required for the initiator of the close to wait for
 749             // the responding close_notify alert before closing the read side
 750             // of the connection.  However, if the application protocol using
 751             // TLS provides that any data may be carried over the underlying
 752             // transport after the TLS connection is closed, the TLS
 753             // implementation MUST receive a "close_notify" alert before
 754             // indicating end-of-data to the application-layer.
 755             try {
 756                 this.shutdown();
 757             } finally {
 758                 if (!isInputShutdown()) {
 759                     shutdownInput(false);
 760                 }
 761             }
 762         } else {
 763             if (!conContext.isInboundClosed()) {
 764                 try (conContext.inputRecord) {
 765                     // Try the best to use up the input records and close the
 766                     // socket gracefully, without impact the performance too
 767                     // much.
 768                     appInput.deplete();
 769                 }
 770             }
 771 
 772             if ((autoClose || !isLayered()) && !super.isInputShutdown()) {
 773                 super.shutdownInput();
 774             }
 775         }
 776     }

 886     /**
 887      * InputStream for application data as returned by
 888      * SSLSocket.getInputStream().
 889      */
 890     private class AppInputStream extends InputStream {
 891         // One element array used to implement the single byte read() method
 892         private final byte[] oneByte = new byte[1];
 893 
 894         // the temporary buffer used to read network
 895         private ByteBuffer buffer;
 896 
 897         // Is application data available in the stream?
 898         private volatile boolean appDataIsAvailable;
 899 
 900         // reading lock
 901         private final ReentrantLock readLock = new ReentrantLock();
 902 
 903         // closing status
 904         private volatile boolean isClosing;
 905         private volatile boolean hasDepleted;
 906 
 907         AppInputStream() {
 908             this.appDataIsAvailable = false;
 909             this.buffer = ByteBuffer.allocate(4096);
 910         }
 911 
 912         /**
 913          * Return the minimum number of bytes that can be read
 914          * without blocking.
 915          */
 916         @Override
 917         public int available() throws IOException {
 918             // Currently not synchronized.
 919             if ((!appDataIsAvailable) || checkEOF()) {
 920                 return 0;
 921             }
 922 
 923             return buffer.remaining();
 924         }
 925 
 926         /**
 927          * Read a single byte, returning -1 on non-fault EOF status.
 928          */
 929         @Override
 930         public int read() throws IOException {
 931             int n = read(oneByte, 0, 1);
 932             if (n <= 0) {   // EOF
 933                 return -1;
 934             }
 935 
 936             return oneByte[0] & 0xFF;
 937         }
 938 
 939         /**
 940          * Reads up to {@code len} bytes of data from the input stream
 941          * into an array of bytes.
 942          *
 943          * An attempt is made to read as many as {@code len} bytes, but a
 944          * smaller number may be read. The number of bytes actually read
 945          * is returned as an integer.
 946          *
 947          * If the layer above needs more data, it asks for more, so we
 948          * are responsible only for blocking to fill at most one buffer,
 949          * and returning "-1" on non-fault EOF status.
 950          */
 951         @Override
 952         public int read(byte[] b, int off, int len) throws IOException {
 953             if (b == null) {
 954                 throw new NullPointerException("the target buffer is null");
 955             } else if (off < 0 || len < 0 || len > b.length - off) {
 956                 throw new IndexOutOfBoundsException(
 957                         "buffer length: " + b.length + ", offset; " + off +
 958                         ", bytes to read:" + len);
 959             } else if (len == 0) {
 960                 return 0;
 961             }
 962 
 963             if (checkEOF()) {
 964                 return -1;
 965             }
 966 
 967             // start handshaking if the connection has not been negotiated.
 968             if (!conContext.isNegotiated && !conContext.isBroken &&
 969                     !conContext.isInboundClosed() &&
 970                     !conContext.isOutboundClosed()) {
 971                 ensureNegotiated();
 972             }
 973 
 974             // Check if the Socket is invalid (error or closed).
 975             if (!conContext.isNegotiated ||
 976                     conContext.isBroken || conContext.isInboundClosed()) {
 977                 throw new SocketException("Connection or inbound has closed");
 978             }
 979 
 980             // Check if the input stream has been depleted.
 981             //
 982             // Note that the "hasDepleted" rather than the isClosing
 983             // filed is checked here, in case the closing process is
 984             // still in progress.
 985             if (hasDepleted) {
 986                 if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
 987                     SSLLogger.fine("The input stream has been depleted");
 988                 }
 989 
 990                 return -1;
 991             }
 992 
 993             // Read the available bytes at first.
 994             //
 995             // Note that the receiving and processing of post-handshake message
 996             // are also synchronized with the read lock.
 997             readLock.lock();
 998             try {
 999                 // Double check if the Socket is invalid (error or closed).
1000                 if (conContext.isBroken || conContext.isInboundClosed()) {
1001                     throw new SocketException(
1002                             "Connection or inbound has closed");
1003                 }
1004 
1005                 // Double check if the input stream has been depleted.
1006                 if (hasDepleted) {
1007                     if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
1008                         SSLLogger.fine("The input stream is closing");
1009                     }
1010 
1011                     return -1;
1012                 }
1013 
1014                 int remains = available();
1015                 if (remains > 0) {
1016                     int howmany = Math.min(remains, len);
1017                     buffer.get(b, off, howmany);
1018 
1019                     return howmany;
1020                 }
1021 
1022                 appDataIsAvailable = false;
1023                 try {
1024                     ByteBuffer bb = readApplicationRecord(buffer);
1025                     if (bb == null) {   // EOF
1026                         return -1;
1027                     } else {
1028                         // The buffer may be reallocated for bigger capacity.
1029                         buffer = bb;
1030                     }
1031 
1032                     bb.flip();
1033                     int volume = Math.min(len, bb.remaining());
1034                     buffer.get(b, off, volume);
1035                     appDataIsAvailable = true;
1036 
1037                     return volume;
1038                 } catch (Exception e) {   // including RuntimeException
1039                     // shutdown and rethrow (wrapped) exception as appropriate
1040                     handleException(e);
1041 
1042                     // dummy for compiler
1043                     return -1;
1044                 }
1045             } finally {
1046                 // Check if the input stream is closing.
1047                 //
1048                 // If the deplete() did not hold the lock, clean up the
1049                 // input stream here.
1050                 try {
1051                     if (isClosing) {
1052                         readLockedDeplete();
1053                     }
1054                 } finally {
1055                     readLock.unlock();
1056                 }
1057             }
1058         }
1059 
1060         /**
1061          * Skip n bytes.
1062          *
1063          * This implementation is somewhat less efficient than possible, but
1064          * not badly so (redundant copy).  We reuse the read() code to keep
1065          * things simpler.
1066          */
1067         @Override
1068         public long skip(long n) throws IOException {
1069             // dummy array used to implement skip()
1070             byte[] skipArray = new byte[256];
1071             long skipped = 0;
1072 
1073             readLock.lock();
1074             try {
1075                 while (n > 0) {
1076                     int len = (int)Math.min(n, skipArray.length);
1077                     int r = read(skipArray, 0, len);
1078                     if (r <= 0) {
1079                         break;
1080                     }
1081                     n -= r;
1082                     skipped += r;
1083                 }
1084             } finally {
1085                 readLock.unlock();
1086             }
1087 
1088             return skipped;
1089         }

1091         @Override
1092         public void close() throws IOException {
1093             if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
1094                 SSLLogger.finest("Closing input stream");
1095             }
1096 
1097             try {
1098                 SSLSocketImpl.this.close();
1099             } catch (IOException ioe) {
1100                 // ignore the exception
1101                 if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
1102                     SSLLogger.warning("input stream close failed", ioe);
1103                 }
1104             }
1105         }
1106 
1107         /**
1108          * Return whether we have reached end-of-file.
1109          *
1110          * If the socket is not connected, has been shutdown because of an error
1111          * or has been closed, throw an Exception.
1112          */
1113         private boolean checkEOF() throws IOException {
1114             if (conContext.isBroken) {
1115                 if (conContext.closeReason == null) {
1116                     return true;
1117                 } else {
1118                     throw new SSLException(
1119                             "Connection has closed: " + conContext.closeReason,
1120                             conContext.closeReason);
1121                 }
1122             } else if (conContext.isInboundClosed()) {
1123                 return true;
1124             } else if (conContext.isInputCloseNotified) {
1125                 if (conContext.closeReason == null) {
1126                     return true;
1127                 } else {
1128                     throw new SSLException(
1129                         "Connection has closed: " + conContext.closeReason,
1130                         conContext.closeReason);
1131                 }
1132             }
1133 
1134             return false;
1135         }
1136 
1137         /**
1138          * Try the best to use up the input records so as to close the
1139          * socket gracefully, without impact the performance too much.
1140          */
1141         private void deplete() {
1142             if (conContext.isInboundClosed() || isClosing) {
1143                 return;
1144             }
1145 
1146             isClosing = true;
1147             if (readLock.tryLock()) {
1148                 try {
1149                     readLockedDeplete();
1150                 } finally {
1151                     readLock.unlock();
1152                 }
1153             }
1154         }
1155 
1156         /**
1157          * Try to use up the input records.
1158          *
1159          * Please don't call this method unless the readLock is held by
1160          * the current thread.
1161          */
1162         private void readLockedDeplete() {
1163             // double check
1164             if (hasDepleted || conContext.isInboundClosed()) {
1165                 return;
1166             }
1167 
1168             if (!(conContext.inputRecord instanceof SSLSocketInputRecord)) {
1169                 return;
1170             }
1171 
1172             SSLSocketInputRecord socketInputRecord =
1173                     (SSLSocketInputRecord)conContext.inputRecord;
1174             try {
1175                 socketInputRecord.deplete(
1176                     conContext.isNegotiated && (getSoTimeout() > 0));
1177             } catch (Exception ex) {
1178                 if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
1179                     SSLLogger.warning(
1180                         "input stream close depletion failed", ex);
1181                 }
1182             } finally {
1183                 hasDepleted = true;
1184             }
1185         }
1186     }

1225         @Override
1226         public void write(byte[] b,
1227                 int off, int len) throws IOException {
1228             if (b == null) {
1229                 throw new NullPointerException("the source buffer is null");
1230             } else if (off < 0 || len < 0 || len > b.length - off) {
1231                 throw new IndexOutOfBoundsException(
1232                         "buffer length: " + b.length + ", offset; " + off +
1233                         ", bytes to read:" + len);
1234             } else if (len == 0) {
1235                 //
1236                 // Don't bother to really write empty records.  We went this
1237                 // far to drive the handshake machinery, for correctness; not
1238                 // writing empty records improves performance by cutting CPU
1239                 // time and network resource usage.  However, some protocol
1240                 // implementations are fragile and don't like to see empty
1241                 // records, so this also increases robustness.
1242                 //
1243                 return;
1244             }
1245 
1246             // Start handshaking if the connection has not been negotiated.
1247             if (!conContext.isNegotiated && !conContext.isBroken &&
1248                     !conContext.isInboundClosed() &&
1249                     !conContext.isOutboundClosed()) {
1250                 ensureNegotiated();
1251             }
1252 
1253             // Check if the Socket is invalid (error or closed).
1254             if (!conContext.isNegotiated ||
1255                     conContext.isBroken || conContext.isOutboundClosed()) {
1256                 throw new SocketException("Connection or outbound has closed");
1257             }
1258 
1259             //
1260 
1261             // Delegate the writing to the underlying socket.
1262             try {
1263                 conContext.outputRecord.deliver(b, off, len);
1264             } catch (SSLHandshakeException she) {
1265                 // may be record sequence number overflow
1266                 throw conContext.fatal(Alert.HANDSHAKE_FAILURE, she);
1267             } catch (SSLException ssle) {
1268                 throw conContext.fatal(Alert.UNEXPECTED_MESSAGE, ssle);
1269             }   // re-throw other IOException, which should be caused by
1270                 // the underlying plain socket and could be handled by
1271                 // applications (for example, re-try the connection).
1272 
1273             // Is the sequence number is nearly overflow, or has the key usage
1274             // limit been reached?
1275             if (conContext.outputRecord.seqNumIsHuge() ||
1276                     conContext.outputRecord.writeCipher.atKeyLimit()) {
1277                 tryKeyUpdate();
1278             }
1279             // Check if NewSessionTicket PostHandshake message needs to be sent
1280             if (conContext.conSession.updateNST) {
1281                 conContext.conSession.updateNST = false;
1282                 tryNewSessionTicket();
1283             }
1284         }

1374     /**
1375      * Read the initial handshake records.
1376      */
1377     private int readHandshakeRecord() throws IOException {
1378         while (!conContext.isInboundClosed()) {
1379             try {
1380                 Plaintext plainText = decode(null);
1381                 if ((plainText.contentType == ContentType.HANDSHAKE.id) &&
1382                         conContext.isNegotiated) {
1383                     return 0;
1384                 }
1385             } catch (SSLException ssle) {
1386                 throw ssle;
1387             } catch (InterruptedIOException iioe) {
1388                 // don't change exception in case of timeouts or interrupts
1389                 throw iioe;
1390             } catch (IOException ioe) {
1391                 throw new SSLException("readHandshakeRecord", ioe);
1392             }
1393         }
1394

1398     /**
1399      * Read application data record. Used by AppInputStream only, but defined
1400      * here so as to use the socket level synchronization.
1401      *
1402      * Note that the connection guarantees that handshake, alert, and change
1403      * cipher spec data streams are handled as they arrive, so we never see
1404      * them here.
1405      *
1406      * Note: Please be careful about the synchronization, and don't use this
1407      * method other than in the AppInputStream class!
1408      */
1409     private ByteBuffer readApplicationRecord(
1410             ByteBuffer buffer) throws IOException {
1411         while (!conContext.isInboundClosed()) {
1412             /*
1413              * clean the buffer and check if it is too small, e.g. because
1414              * the AppInputStream did not have the chance to see the
1415              * current packet length but rather something like that of the
1416              * handshake before. In that case we return 0 at this point to
1417              * give the caller the chance to adjust the buffer.
1418              */
1419             buffer.clear();
1420             int inLen = conContext.inputRecord.bytesInCompletePacket();
1421             if (inLen < 0) {    // EOF
1422                 handleEOF(null);
1423 
1424                 // if no exception thrown
1425                 return null;
1426             }
1427 
1428             // Is this packet bigger than SSL/TLS normally allows?
1429             if (inLen > SSLRecord.maxLargeRecordSize) {
1430                 throw new SSLProtocolException(
1431                         "Illegal packet size: " + inLen);
1432             }
1433 
1434             if (inLen > buffer.remaining()) {
1435                 buffer = ByteBuffer.allocate(inLen);
1436             }
1437 
1438             try {
1439                 Plaintext plainText;
1440                 socketLock.lock();
1441                 try {
1442                     plainText = decode(buffer);
1443                 } finally {
1444                     socketLock.unlock();
1445                 }
1446                 if (plainText.contentType == ContentType.APPLICATION_DATA.id &&
1447                         buffer.position() > 0) {
1448                     return buffer;
1449                 }
1450             } catch (SSLException ssle) {
1451                 throw ssle;
1452             } catch (InterruptedIOException iioe) {
1453                 // don't change exception in case of timeouts or interrupts
1454                 throw iioe;
1455             } catch (IOException ioe) {
1456                 if (!(ioe instanceof SSLException)) {
1457                     throw new SSLException("readApplicationRecord", ioe);
1458                 } else {
1459                     throw ioe;
1460                 }
1461             }
1462         }
1463 
1464         //
1465         // couldn't read, due to some kind of error
1466         //
1467         return null;
1468     }

1470     private Plaintext decode(ByteBuffer destination) throws IOException {
1471         Plaintext plainText;
1472         try {
1473             if (destination == null) {
1474                 plainText = SSLTransport.decode(conContext,
1475                         null, 0, 0, null, 0, 0);
1476             } else {
1477                 plainText = SSLTransport.decode(conContext,
1478                         null, 0, 0, new ByteBuffer[]{destination}, 0, 1);
1479             }
1480         } catch (EOFException eofe) {
1481             // EOFException is special as it is related to close_notify.
1482             plainText = handleEOF(eofe);
1483         }
1484 
1485         // Is the sequence number is nearly overflow?
1486         if (plainText != Plaintext.PLAINTEXT_NULL &&
1487                 (conContext.inputRecord.seqNumIsHuge() ||
1488                 conContext.inputRecord.readCipher.atKeyLimit())) {
1489             tryKeyUpdate();
1490         }
1491 
1492         return plainText;
1493     }

1495     /**
1496      * Try key update for sequence number wrap or key usage limit.
1497      *
1498      * Note that in order to maintain the handshake status properly, we check
1499      * the sequence number and key usage limit after the last record
1500      * reading/writing process.
1501      *
1502      * As we request renegotiation or close the connection for wrapped sequence
1503      * number when there is enough sequence number space left to handle a few
1504      * more records, so the sequence number of the last record cannot be
1505      * wrapped.
1506      */
1507     private void tryKeyUpdate() throws IOException {
1508         // Don't bother to kickstart if handshaking is in progress, or if the
1509         // connection is not duplex-open.
1510         if ((conContext.handshakeContext == null) &&
1511                 !conContext.isOutboundClosed() &&
1512                 !conContext.isInboundClosed() &&
1513                 !conContext.isBroken) {
1514             if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
1515                 SSLLogger.finest("trigger key update");
1516             }
1517             startHandshake();
1518         }
1519     }

1521     // Try to generate a PostHandshake NewSessionTicket message.  This is
1522     // TLS 1.3 only.
1523     private void tryNewSessionTicket() throws IOException {
1524         // Don't bother to kickstart if handshaking is in progress, or if the
1525         // connection is not duplex-open.
1526         if (!conContext.sslConfig.isClientMode &&
1527                 conContext.protocolVersion.useTLS13PlusSpec() &&
1528                 conContext.handshakeContext == null &&
1529                 !conContext.isOutboundClosed() &&
1530                 !conContext.isInboundClosed() &&
1531                 !conContext.isBroken) {
1532             if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
1533                 SSLLogger.finest("trigger new session ticket");
1534             }
1535             NewSessionTicket.kickstartProducer.produce(
1536                     new PostHandshakeContext(conContext));
1537         }
1538     }

1540     /**
1541      * Initialize the handshaker and socket streams.
1542      *
1543      * Called by connect, the layered constructor, and SSLServerSocket.
1544      */
1545     void doneConnect() throws IOException {
1546         socketLock.lock();
1547         try {
1548             // In server mode, it is not necessary to set host and serverNames.
1549             // Otherwise, would require a reverse DNS lookup to get
1550             // the hostname.
1551             if (peerHost == null || peerHost.isEmpty()) {
1552                 boolean useNameService =
1553                         trustNameService && conContext.sslConfig.isClientMode;
1554                 useImplicitHost(useNameService);
1555             } else {
1556                 conContext.sslConfig.serverNames =
1557                         Utilities.addToSNIServerNameList(
1558                                 conContext.sslConfig.serverNames, peerHost);
1559             }
1560 
1561             InputStream sockInput = super.getInputStream();
1562             conContext.inputRecord.setReceiverStream(sockInput);
1563 
1564             OutputStream sockOutput = super.getOutputStream();
1565             conContext.inputRecord.setDeliverStream(sockOutput);
1566             conContext.outputRecord.setDeliverStream(sockOutput);
1567 
1568             this.isConnected = true;
1569         } finally {
1570             socketLock.unlock();
1571         }
1572     }

1574     private void useImplicitHost(boolean useNameService) {
1575         // Note: If the local name service is not trustworthy, reverse
1576         // host name resolution should not be performed for endpoint
1577         // identification.  Use the application original specified
1578         // hostname or IP address instead.
1579 
1580         // Get the original hostname via jdk.internal.access.SharedSecrets
1581         InetAddress inetAddress = getInetAddress();
1582         if (inetAddress == null) {      // not connected
1583             return;
1584         }
1585 
1586         JavaNetInetAddressAccess jna =
1587                 SharedSecrets.getJavaNetInetAddressAccess();
1588         String originalHostname = jna.getOriginalHostName(inetAddress);
1589         if (originalHostname != null && !originalHostname.isEmpty()) {
1590 
1591             this.peerHost = originalHostname;
1592             if (conContext.sslConfig.serverNames.isEmpty() &&
1593                     !conContext.sslConfig.noSniExtension) {
1594                 conContext.sslConfig.serverNames =
1595                         Utilities.addToSNIServerNameList(
1596                                 conContext.sslConfig.serverNames, peerHost);
1597             }
1598 
1599             return;
1600         }
1601 
1602         // No explicitly specified hostname, no server name indication.
1603         if (!useNameService) {
1604             // The local name service is not trustworthy, use IP address.
1605             this.peerHost = inetAddress.getHostAddress();
1606         } else {
1607             // Use the underlying reverse host name resolution service.
1608             this.peerHost = getInetAddress().getHostName();
1609         }
1610     }

1629     /**
1630      * Handle an exception.
1631      *
1632      * This method is called by top level exception handlers (in read(),
1633      * write()) to make sure we always shutdown the connection correctly
1634      * and do not pass runtime exception to the application.
1635      *
1636      * This method never returns normally, it always throws an IOException.
1637      */
1638     private void handleException(Exception cause) throws IOException {
1639         if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
1640             SSLLogger.warning("handling exception", cause);
1641         }
1642 
1643         // Don't close the Socket in case of timeouts or interrupts.
1644         if (cause instanceof InterruptedIOException) {
1645             throw (IOException)cause;
1646         }
1647 
1648         // need to perform error shutdown
1649         boolean isSSLException = (cause instanceof SSLException);
1650         Alert alert;
1651         if (isSSLException) {
1652             if (cause instanceof SSLHandshakeException) {
1653                 alert = Alert.HANDSHAKE_FAILURE;
1654             } else {
1655                 alert = Alert.UNEXPECTED_MESSAGE;
1656             }
1657         } else {
1658             if (cause instanceof IOException) {
1659                 alert = Alert.UNEXPECTED_MESSAGE;
1660             } else {
1661                 // RuntimeException
1662                 alert = Alert.INTERNAL_ERROR;
1663             }
1664         }
1665 
1666         throw conContext.fatal(alert, cause);
1667     }

1669     private Plaintext handleEOF(EOFException eofe) throws IOException {
1670         if (requireCloseNotify || conContext.handshakeContext != null) {
1671             SSLException ssle;
1672             if (conContext.handshakeContext != null) {
1673                 ssle = new SSLHandshakeException(
1674                         "Remote host terminated the handshake");
1675             } else {
1676                 ssle = new SSLProtocolException(
1677                         "Remote host terminated the connection");
1678             }
1679 
1680             if (eofe != null) {
1681                 ssle.initCause(eofe);
1682             }
1683             throw ssle;
1684         } else {
1685             // treat as if we had received a close_notify
1686             conContext.isInputCloseNotified = true;
1687             shutdownInput();
1688 
1689             return Plaintext.PLAINTEXT_NULL;
1690         }
1691     }

1709     @Override
1710     public void shutdown() throws IOException {
1711         if (!isClosed()) {
1712             if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
1713                 SSLLogger.fine("close the underlying socket");
1714             }
1715 
1716             try {
1717                 if (conContext.isInputCloseNotified) {
1718                     // Close the connection, no wait for more peer response.
1719                     closeSocket(false);
1720                 } else {
1721                     // Close the connection, may wait for peer close_notify.
1722                     closeSocket(true);
1723                 }
1724             } finally {
1725                 tlsIsClosed = true;
1726             }
1727         }
1728     }

1730     private void closeSocket(boolean selfInitiated) throws IOException {
1731         if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
1732             SSLLogger.fine("close the SSL connection " +
1733                 (selfInitiated ? "(initiative)" : "(passive)"));
1734         }
1735 
1736         if (autoClose || !isLayered()) {
1737             super.close();
1738         } else if (selfInitiated) {
1739             if (!conContext.isInboundClosed() && !isInputShutdown()) {
1740                 // wait for close_notify alert to clear input stream.
1741                 waitForClose();
1742             }
1743         }
1744     }

1746    /**
1747     * Wait for close_notify alert for a graceful closure.
1748     *
1749     * [RFC 5246] If the application protocol using TLS provides that any
1750     * data may be carried over the underlying transport after the TLS
1751     * connection is closed, the TLS implementation must receive the responding
1752     * close_notify alert before indicating to the application layer that
1753     * the TLS connection has ended.  If the application protocol will not
1754     * transfer any additional data, but will only close the underlying
1755     * transport connection, then the implementation MAY choose to close the
1756     * transport without waiting for the responding close_notify.
1757     */
1758     private void waitForClose() throws IOException {
1759         if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
1760             SSLLogger.fine("wait for close_notify or alert");
1761         }
1762 
1763         while (!conContext.isInboundClosed()) {
1764             try {
1765                 Plaintext plainText = decode(null);
1766                 // discard and continue
1767                 if (SSLLogger.isOn && SSLLogger.isOn("ssl")) {
1768                     SSLLogger.finest(
1769                         "discard plaintext while waiting for close", plainText);
1770                 }
1771             } catch (Exception e) {   // including RuntimeException
1772                 handleException(e);
1773             }
1774         }
1775     }
