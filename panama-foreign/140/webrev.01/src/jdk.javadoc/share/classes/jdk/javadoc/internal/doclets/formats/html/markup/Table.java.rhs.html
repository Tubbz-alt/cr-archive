 45 /**
 46  * An HTML table, such as the summary tables for various kinds of element.
 47  *
 48  * <p>The table should be used in three phases:
 49  * <ol>
 50  * <li>Configuration: the overall characteristics of the table should be specified
 51  * <li>Population: the content for the cells in each row should be added
 52  * <li>Generation: the HTML content and any associated JavaScript can be accessed
 53  * </ol>
 54  *
 55  * Many methods return the current object, to facilitate fluent builder-style usage.
 56  *
 57  * A table may support filtered views, which can be selected by clicking on
 58  * one of a list of tabs above the table. If the table does not support filtered
 59  * views, the {@code <caption>} element is typically displayed as a single (inactive)
 60  * tab.
 61  *
 62  * Tables are always enclosed in a {@code <div>} element, which will also contain
 63  * a {@code <div>} element for the list of tabs, when the table supports filtered views.
 64  *
 65  *  <p><b>This is NOT part of any supported API.
 66  *  If you write code that depends on this, you do so at your own risk.
 67  *  This code and its internal interfaces are subject to change or
 68  *  deletion without notice.</b>
 69  */

120      * Adds a tab to the table.
121      * Tabs provide a way to display subsets of rows, as determined by a
122      * predicate for the tab, and an element associated with each row.
123      * Tabs will appear left-to-right in the order they are added.
124      *
125      * @param name      the name of the tab
126      * @param predicate the predicate
127      * @return this object
128      */
129     public Table addTab(String name, Predicate<Element> predicate) {
130         if (tabMap == null) {
131             tabMap = new LinkedHashMap<>();     // preserves order that tabs are added
132             tabs = new HashSet<>();             // order not significant
133         }
134         tabMap.put(name, predicate);
135         return this;
136     }

334      * @throws NullPointerException if tabs have previously been added to the table

341     /**
342      * Adds a row of data to the table.
343      * Each item of content should be suitable for use as the content of a
344      * {@code <th>} or {@code <td>} cell.
345      *
346      * If tabs have been added to the table, the specified element will be used
347      * to determine whether the row should be displayed when any particular tab
348      * is selected, using the predicate specified when the tab was
349      * {@link #addTab(String,Predicate) added}.
350      *
351      * @param element the element
352      * @param contents the contents for the row
353      * @throws NullPointerException if tabs have previously been added to the table
354      *      and {@code element} is null
355      */
356     public void addRow(Element element, List<Content> contents) {
357         if (tabMap != null && element == null) {
358             throw new NullPointerException();
359         }
360 
361         HtmlTree row = new HtmlTree(TagName.TR);
362 
363         if (stripedStyles != null) {
364             int rowIndex = bodyRows.size();
365             row.setStyle(stripedStyles.get(rowIndex % 2));
366         }
367         int colIndex = 0;
368         for (Content c : contents) {
369             HtmlStyle cellStyle = (columnStyles == null || colIndex > columnStyles.size())
370                     ? null
371                     : columnStyles.get(colIndex);
372             HtmlTree cell = (colIndex == rowScopeColumnIndex)
373                     ? HtmlTree.TH(cellStyle, "row", c)
374                     : HtmlTree.TD(cellStyle, c);
375             row.add(cell);
376             colIndex++;
377         }
378         bodyRows.add(row);
379 
380         if (tabMap != null) {
381             int index = bodyRows.size() - 1;
382             row.put(HtmlAttr.ID, (rowIdPrefix + index));
383             int mask = 0;
384             int maskBit = 1;
385             for (Map.Entry<String, Predicate<Element>> e : tabMap.entrySet()) {
386                 String name = e.getKey();
387                 Predicate<Element> predicate = e.getValue();
388                 if (predicate.test(element)) {
389                     tabs.add(name);
390                     mask |= maskBit;
391                 }
392                 maskBit = (maskBit << 1);
393             }
394             bodyRowMasks.add(mask);
395         }
396     }

413     /**
414      * Returns the HTML for the table.
415      *
416      * @return the HTML
417      */
418     private Content toContent() {
419         HtmlTree mainDiv = new HtmlTree(TagName.DIV).setStyle(topStyle);

420         if (id != null) {
421             mainDiv.setId(id);
422         }
423         HtmlTree table = new HtmlTree(TagName.TABLE).setStyle(tableStyle);
424         if (tabMap == null || tabs.size() == 1) {
425             if (tabMap == null) {
426                 table.add(caption);
427             } else if (tabs.size() == 1) {
428                 String tabName = tabs.iterator().next();
429                 table.add(getCaption(new StringContent(tabName)));
430             }
431             table.add(getTableBody());
432             mainDiv.add(table);
433         } else {
434             HtmlTree tablist = new HtmlTree(TagName.DIV).setStyle(tabListStyle)
435                     .put(HtmlAttr.ROLE, "tablist")
436                     .put(HtmlAttr.ARIA_ORIENTATION, "horizontal");
437 
438             int tabIndex = 0;
439             tablist.add(createTab(tabId.apply(tabIndex), activeTabStyle, true, defaultTab));
440             table.put(HtmlAttr.ARIA_LABELLEDBY, tabId.apply(tabIndex));
441             for (String tabName : tabMap.keySet()) {
442                 tabIndex++;
443                 if (tabs.contains(tabName)) {
444                     String script = tabScript.apply(1 << (tabIndex - 1));
445                     HtmlTree tab = createTab(tabId.apply(tabIndex), tabStyle, false, tabName);
446                     tab.put(HtmlAttr.ONCLICK, script);
447                     tablist.add(tab);
448                 }
449             }
450             if (id == null) {
451                 throw new IllegalStateException("no id set for table");
452             }
453             HtmlTree tabpanel = new HtmlTree(TagName.DIV)
454                     .put(HtmlAttr.ID, id + ".tabpanel")
455                     .put(HtmlAttr.ROLE, "tabpanel");
456             table.add(getTableBody());
457             tabpanel.add(table);
458             mainDiv.add(tablist);
459             mainDiv.add(tabpanel);
460         }
461         return mainDiv;
462     }

464     private HtmlTree createTab(String tabId, HtmlStyle style, boolean defaultTab, String tabName) {
465         HtmlTree tab = new HtmlTree(TagName.BUTTON)
466                 .put(HtmlAttr.ROLE, "tab")
467                 .put(HtmlAttr.ARIA_SELECTED, defaultTab ? "true" : "false")
468                 .put(HtmlAttr.ARIA_CONTROLS, id + ".tabpanel")
469                 .put(HtmlAttr.TABINDEX, defaultTab ? "0" : "-1")
470                 .put(HtmlAttr.ONKEYDOWN, "switchTab(event)")
471                 .put(HtmlAttr.ID, tabId)
472                 .setStyle(style);
473         tab.add(tabName);
474         return tab;
475     }

477     private Content getTableBody() {
478         ContentBuilder tableContent = new ContentBuilder();
479         Content thead = new HtmlTree(TagName.THEAD);
480         thead.add(header);
481         tableContent.add(thead);
482         Content tbody = new HtmlTree(TagName.TBODY);
483         bodyRows.forEach(tbody::add);
484         tableContent.add(tbody);
485         return tableContent;
486     }

498     /**
499      * Returns the script to be used in conjunction with the table.
500      *
501      * @return the script
502      */
503     public String getScript() {
504         if (tabMap == null)
505             throw new IllegalStateException();
506 
507         StringBuilder sb = new StringBuilder();
508 
509         // Add the variable defining the bitmask for each row
510         sb.append("var data").append(" = {");
511         int rowIndex = 0;
512         for (int mask : bodyRowMasks) {
513             if (rowIndex > 0) {
514                 sb.append(",");
515             }
516             sb.append("\"").append(rowIdPrefix).append(rowIndex).append("\":").append(mask);
517             rowIndex++;
518         }
519         sb.append("};\n");
520 
521         // Add the variable defining the tabs
522         sb.append("var tabs = {");
523         appendTabInfo(sb, 65535, tabId.apply(0), defaultTab);
524         int tabIndex = 1;
525         int maskBit = 1;
526         for (String tabName: tabMap.keySet()) {
527             if (tabs.contains(tabName)) {
528                 sb.append(",");
529                 appendTabInfo(sb, maskBit, tabId.apply(tabIndex), tabName);
530             }
531             tabIndex++;
532             maskBit = (maskBit << 1);
533         }
534         sb.append("};\n");
535 
536         // Add the variables defining the stylenames
537         appendStyleInfo(sb,
538                 stripedStyles.get(0), stripedStyles.get(1), tabStyle, activeTabStyle);
539         return sb.toString();
540     }

542     private void appendTabInfo(StringBuilder sb, int value, String id, String name) {
543         sb.append(value)
544                 .append(":[")
545                 .append(Script.stringLiteral(id))
546                 .append(",")
547                 .append(Script.stringLiteral(name))
548                 .append("]");
549     }
550 
551     private void appendStyleInfo(StringBuilder sb, HtmlStyle... styles) {
552         for (HtmlStyle style : styles) {
553             sb.append("var ").append(style.name()).append(" = \"").append(style.cssName()).append("\";\n");
554         }
555 
556     }
