447     private final String cssName;
448 
449     HtmlStyle() {
450         cssName = Pattern.compile("\\p{Upper}")
451                 .matcher(toString())
452                 .replaceAll(mr -> "-" + mr.group().toLowerCase(Locale.US));
453     }
454 
455     HtmlStyle(String cssName) {
456         this.cssName = cssName;
457     }
458 
459     /**
460      * Returns the CSS class name associated with this style.
461      * @return the CSS class name
462      */
463     public String cssName() {
464         return cssName;
465     }
