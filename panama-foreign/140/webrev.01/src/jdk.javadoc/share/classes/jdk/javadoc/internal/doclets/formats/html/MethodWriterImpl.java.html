 79     @Override
 80     public Content getMemberSummaryHeader(TypeElement typeElement, Content memberSummaryTree) {
 81         memberSummaryTree.add(MarkerComments.START_OF_METHOD_SUMMARY);
 82         Content memberTree = new ContentBuilder();
 83         writer.addSummaryHeader(this, memberTree);
 84         return memberTree;
 85     }

 87     @Override
 88     public void addSummary(Content summariesList, Content content) {
 89         writer.addSummary(HtmlStyle.methodSummary,
 90                 SectionName.METHOD_SUMMARY, summariesList, content);
 91     }

 93     @Override
 94     public Content getMethodDetailsTreeHeader(Content memberDetailsTree) {
 95         memberDetailsTree.add(MarkerComments.START_OF_METHOD_DETAILS);
 96         Content methodDetailsTree = new ContentBuilder();
 97         Content heading = HtmlTree.HEADING(Headings.TypeDeclaration.DETAILS_HEADING,
 98                 contents.methodDetailLabel);
 99         methodDetailsTree.add(heading);
100         return methodDetailsTree;
101     }
102 
103     @Override
104     public Content getMethodDocTreeHeader(ExecutableElement method) {
105         String erasureAnchor;
106         Content methodDocTree = new ContentBuilder();
107         HtmlTree heading = HtmlTree.HEADING(Headings.TypeDeclaration.MEMBER_HEADING,
108                 new StringContent(name(method)));
109         if ((erasureAnchor = getErasureAnchor(method)) != null) {
110             heading.setId(erasureAnchor);
111         }
112         methodDocTree.add(heading);
113         return HtmlTree.SECTION(HtmlStyle.detail, methodDocTree)
114                 .setId(links.getName(writer.getAnchor(method)));
115     }

125         return new MemberSignature(method)

138     @Override
139     public void addComments(TypeMirror holderType, ExecutableElement method, Content methodDocTree) {
140         TypeElement holder = utils.asTypeElement(holderType);
141         if (!utils.getFullBody(method).isEmpty()) {
142             if (holder.equals(typeElement) ||
143                     !(utils.isPublic(holder) ||
144                     utils.isLinkable(holder))) {
145                 writer.addInlineComment(method, methodDocTree);
146             } else {
147                 Content link =
148                         writer.getDocLink(LinkInfoImpl.Kind.EXECUTABLE_ELEMENT_COPY,
149                         holder, method,
150                         utils.isIncluded(holder)
151                                 ? utils.getSimpleName(holder)
152                                 : utils.getFullyQualifiedName(holder),
153                             false);
154                 Content codeLink = HtmlTree.CODE(link);
155                 Content descfrmLabel = HtmlTree.SPAN(HtmlStyle.descfrmTypeLabel,
156                         utils.isClass(holder)
157                                 ? contents.descfrmClassLabel
158                                 : contents.descfrmInterfaceLabel);
159                 descfrmLabel.add(Entity.NO_BREAK_SPACE);
160                 descfrmLabel.add(codeLink);
161                 methodDocTree.add(HtmlTree.DIV(HtmlStyle.block, descfrmLabel));
162                 writer.addInlineComment(method, methodDocTree);
163             }
164         }
165     }

192     @Override
193     protected Table createSummaryTable() {
194         return new Table(HtmlStyle.memberSummary, HtmlStyle.summaryTable)
195                 .setHeader(getSummaryTableHeader(typeElement))
196                 .setRowScopeColumn(1)
197                 .setColumnStyles(HtmlStyle.colFirst, HtmlStyle.colSecond, HtmlStyle.colLast)
198                 .setId("method-summary-table")
199                 .setDefaultTab(resources.getText("doclet.All_Methods"))
200                 .addTab(resources.getText("doclet.Static_Methods"), utils::isStatic)
201                 .addTab(resources.getText("doclet.Instance_Methods"), e -> !utils.isStatic(e))
202                 .addTab(resources.getText("doclet.Abstract_Methods"), utils::isAbstract)
203                 .addTab(resources.getText("doclet.Concrete_Methods"),
204                         e -> !utils.isAbstract(e) && !utils.isInterface(e.getEnclosingElement()))
205                 .addTab(resources.getText("doclet.Default_Methods"), utils::isDefault)
206                 .addTab(resources.getText("doclet.Deprecated_Methods"),
207                         e -> utils.isDeprecated(e) || utils.isDeprecated(typeElement))
208                 .setTabScript(i -> "show(" + i + ");");
209     }

211     @Override
212     public void addInheritedSummaryLabel(TypeElement typeElement, Content inheritedTree) {
213         Content classLink = writer.getPreQualifiedClassLink(
214                 LinkInfoImpl.Kind.MEMBER, typeElement, false);
215         Content label;
216         if (options.summarizeOverriddenMethods()) {
217             label = new StringContent(utils.isClass(typeElement)
218                     ? resources.getText("doclet.Methods_Declared_In_Class")
219                     : resources.getText("doclet.Methods_Declared_In_Interface"));
220         } else {
221             label = new StringContent(utils.isClass(typeElement)
222                     ? resources.getText("doclet.Methods_Inherited_From_Class")
223                     : resources.getText("doclet.Methods_Inherited_From_Interface"));
224         }
225         HtmlTree labelHeading = HtmlTree.HEADING(Headings.TypeDeclaration.INHERITED_SUMMARY_HEADING,
226                 label);
227         labelHeading.setId(SectionName.METHODS_INHERITANCE.getName()
228                 + links.getName(configuration.getClassName(typeElement)));
229         labelHeading.add(Entity.NO_BREAK_SPACE);
230         labelHeading.add(classLink);
231         inheritedTree.add(labelHeading);
232     }

240     /**
241      * Adds "overrides" or "specified by" information about a method (if appropriate)
242      * into a definition list.
243      *
244      * @param writer         the writer for the element
245      * @param overriddenType the superclass
246      * @param method         the method
247      * @param dl             the list in which to add the information.
248      */
249     protected static void addOverridden(HtmlDocletWriter writer,
250                                         TypeMirror overriddenType,
251                                         ExecutableElement method,
252                                         Content dl) {
253         if (writer.options.noComment()) {
254             return;
255         }
256         Utils utils = writer.utils;
257         TypeElement holder = utils.getEnclosingTypeElement(method);
258         if (!(utils.isPublic(holder) || utils.isLinkable(holder))) {
259             //This is an implementation detail that should not be documented.
260             return;
261         }
262         if (utils.isIncluded(holder) && !utils.isIncluded(method)) {
263             //The class is included but the method is not.  That means that it
264             //is not visible so don't document this.
265             return;
266         }
267 
268         if (method != null) {
269             Contents contents = writer.contents;
270             Content label;
271             LinkInfoImpl.Kind context;
272             if (utils.isAbstract(holder) && utils.isAbstract(method)){
273                 //Abstract method is implemented from abstract class,
274                 //not overridden
275                 label = contents.specifiedByLabel;
276                 context = LinkInfoImpl.Kind.METHOD_SPECIFIED_BY;
277             } else {
278                 label = contents.overridesLabel;
279                 context = LinkInfoImpl.Kind.METHOD_OVERRIDES;
280             }
281             dl.add(HtmlTree.DT(label));
282             Content overriddenTypeLink =
283                     writer.getLink(new LinkInfoImpl(writer.configuration, context, overriddenType));
284             Content codeOverriddenTypeLink = HtmlTree.CODE(overriddenTypeLink);
285             Content methlink = writer.getLink(
286                     new LinkInfoImpl(writer.configuration, LinkInfoImpl.Kind.MEMBER, holder)
287                             .where(writer.links.getName(writer.getAnchor(method)))
288                             .label(method.getSimpleName()));
289             Content codeMethLink = HtmlTree.CODE(methlink);
290             Content dd = HtmlTree.DD(codeMethLink);
291             dd.add(Entity.NO_BREAK_SPACE);
292             dd.add(contents.inClass);
293             dd.add(Entity.NO_BREAK_SPACE);
294             dd.add(codeOverriddenTypeLink);
295             dl.add(dd);
296         }
297     }

299     /**
300      * Adds "implements" information for a method (if appropriate)
301      * into a definition list.
302      *
303      * @param writer the writer for the method
304      * @param method the method
305      * @param dl     the definition list
306      */
307     protected static void addImplementsInfo(HtmlDocletWriter writer,
308                                             ExecutableElement method,
309                                             Content dl) {
310         Utils utils = writer.utils;
311         if (utils.isStatic(method) || writer.options.noComment()) {
312             return;
313         }
314         Contents contents = writer.contents;
315         VisibleMemberTable vmt = writer.configuration
316                 .getVisibleMemberTable(utils.getEnclosingTypeElement(method));
317         SortedSet<ExecutableElement> implementedMethods =
318                 new TreeSet<>(utils.comparators.makeOverrideUseComparator());
319         implementedMethods.addAll(vmt.getImplementedMethods(method));
320         for (ExecutableElement implementedMeth : implementedMethods) {
321             TypeMirror intfac = vmt.getImplementedMethodHolder(method, implementedMeth);
322             intfac = utils.getDeclaredType(utils.getEnclosingTypeElement(method), intfac);
323             Content intfaclink = writer.getLink(new LinkInfoImpl(
324                     writer.configuration, LinkInfoImpl.Kind.METHOD_SPECIFIED_BY, intfac));
325             Content codeIntfacLink = HtmlTree.CODE(intfaclink);
326             dl.add(HtmlTree.DT(contents.specifiedByLabel));
327             Content methlink = writer.getDocLink(
328                     LinkInfoImpl.Kind.MEMBER, implementedMeth,
329                     implementedMeth.getSimpleName(), false);
330             Content codeMethLink = HtmlTree.CODE(methlink);
331             Content dd = HtmlTree.DD(codeMethLink);
332             dd.add(Entity.NO_BREAK_SPACE);
333             dd.add(contents.inInterface);
334             dd.add(Entity.NO_BREAK_SPACE);
335             dd.add(codeIntfacLink);
336             dl.add(dd);
337         }
338     }

340     /**
341      * Get the return type for the given method.
342      *
343      * @param method the method being documented.
344      * @return content containing the return type
345      */
346     protected Content getReturnType(ExecutableElement method) {
347         TypeMirror type = utils.getReturnType(typeElement, method);
348         if (type != null) {
349             return writer.getLink(new LinkInfoImpl(configuration, LinkInfoImpl.Kind.RETURN_TYPE, type));
350         }
351         return new ContentBuilder();
352     }
