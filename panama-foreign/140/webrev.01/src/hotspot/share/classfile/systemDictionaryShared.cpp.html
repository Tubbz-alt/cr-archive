  69 class DumpTimeSharedClassInfo: public CHeapObj<mtClass> {
  70   bool                         _excluded;
  71 public:
  72   struct DTConstraint {
  73     Symbol* _name;
  74     Symbol* _from_name;
  75     DTConstraint() : _name(NULL), _from_name(NULL) {}
  76     DTConstraint(Symbol* n, Symbol* fn) : _name(n), _from_name(fn) {}
  77   };
  78 
  79   InstanceKlass*               _klass;
  80   bool                         _failed_verification;
  81   int                          _id;
  82   int                          _clsfile_size;
  83   int                          _clsfile_crc32;
  84   GrowableArray<DTConstraint>* _verifier_constraints;
  85   GrowableArray<char>*         _verifier_constraint_flags;
  86 
  87   DumpTimeSharedClassInfo() {
  88     _klass = NULL;
  89     _failed_verification = false;
  90     _id = -1;
  91     _clsfile_size = -1;
  92     _clsfile_crc32 = -1;
  93     _excluded = false;
  94     _verifier_constraints = NULL;
  95     _verifier_constraint_flags = NULL;
  96   }
  97 
  98   void add_verification_constraint(InstanceKlass* k, Symbol* name,
  99          Symbol* from_name, bool from_field_is_protected, bool from_is_array, bool from_is_object);
 100 
 101   bool is_builtin() {
 102     return SystemDictionaryShared::is_builtin(_klass);
 103   }
 104 
 105   int num_constraints() {
 106     if (_verifier_constraint_flags != NULL) {
 107       return _verifier_constraint_flags->length();
 108     } else {
 109       return 0;
 110     }
 111   }
 112 
 113   void metaspace_pointers_do(MetaspaceClosure* it) {
 114     it->push(&_klass);
 115     if (_verifier_constraints != NULL) {
 116       for (int i = 0; i < _verifier_constraints->length(); i++) {
 117         DTConstraint* cons = _verifier_constraints->adr_at(i);
 118         it->push(&cons->_name);
 119         it->push(&cons->_from_name);
 120       }
 121     }
 122   }
 123 
 124   void set_excluded() {
 125     _excluded = true;
 126   }
 127 
 128   bool is_excluded() {
 129     // _klass may become NULL due to DynamicArchiveBuilder::set_to_null
 130     return _excluded || _failed_verification || _klass == NULL;
 131   }
 132 
 133   void set_failed_verification() {
 134     _failed_verification = true;
 135   }
 136 
 137   bool failed_verification() {
 138     return _failed_verification;
 139   }
 140 };

 142 class DumpTimeSharedClassTable: public ResourceHashtable<
 143   InstanceKlass*,
 144   DumpTimeSharedClassInfo,
 145   primitive_hash<InstanceKlass*>,
 146   primitive_equals<InstanceKlass*>,
 147   15889, // prime number
 148   ResourceObj::C_HEAP>
 149 {
 150   int _builtin_count;
 151   int _unregistered_count;
 152 public:
 153   DumpTimeSharedClassInfo* find_or_allocate_info_for(InstanceKlass* k) {
 154     DumpTimeSharedClassInfo* p = get(k);
 155     if (p == NULL) {
 156       assert(!SystemDictionaryShared::no_class_loading_should_happen(),
 157              "no new classes can be loaded while dumping archive");
 158       put(k, DumpTimeSharedClassInfo());
 159       p = get(k);
 160       assert(p != NULL, "sanity");
 161       p->_klass = k;
 162     }
 163     return p;
 164   }

 166   class CountClassByCategory : StackObj {
 167     DumpTimeSharedClassTable* _table;
 168   public:
 169     CountClassByCategory(DumpTimeSharedClassTable* table) : _table(table) {}
 170     bool do_entry(InstanceKlass* k, DumpTimeSharedClassInfo& info) {
 171       if (!info.is_excluded()) {
 172         if (info.is_builtin()) {
 173           ++ _table->_builtin_count;
 174         } else {
 175           ++ _table->_unregistered_count;
 176         }
 177       }
 178       return true; // keep on iterating
 179     }
 180   };
 181 
 182   void update_counts() {
 183     _builtin_count = 0;
 184     _unregistered_count = 0;
 185     CountClassByCategory counter(this);
 186     iterate(&counter);
 187   }
 188 
 189   int count_of(bool is_builtin) const {
 190     if (is_builtin) {
 191       return _builtin_count;
 192     } else {
 193       return _unregistered_count;
 194     }
 195   }

 198 class RunTimeSharedClassInfo {
 199 public:
 200   struct CrcInfo {
 201     int _clsfile_size;
 202     int _clsfile_crc32;
 203   };
 204 
 205   // This is different than  DumpTimeSharedClassInfo::DTConstraint. We use
 206   // u4 instead of Symbol* to save space on 64-bit CPU.
 207   struct RTConstraint {
 208     u4 _name;
 209     u4 _from_name;
 210   };
 211 
 212   InstanceKlass* _klass;
 213   int _num_constraints;
 214 
 215   // optional CrcInfo      _crc;  (only for UNREGISTERED classes)
 216   // optional RTConstraint _verifier_constraints[_num_constraints]
 217   // optional char         _verifier_constraint_flags[_num_constraints]
 218 
 219 private:
 220   static size_t header_size_size() {
 221     return sizeof(RunTimeSharedClassInfo);
 222   }
 223   static size_t crc_size(InstanceKlass* klass) {
 224     if (!SystemDictionaryShared::is_builtin(klass)) {
 225       return sizeof(CrcInfo);
 226     } else {
 227       return 0;
 228     }
 229   }
 230   static size_t verifier_constraints_size(int num_constraints) {
 231     return sizeof(RTConstraint) * num_constraints;
 232   }
 233   static size_t verifier_constraint_flags_size(int num_constraints) {
 234     return sizeof(char) * num_constraints;
 235   }
 236 
 237 public:
 238   static size_t byte_size(InstanceKlass* klass, int num_constraints) {
 239     return header_size_size() +
 240            crc_size(klass) +
 241            verifier_constraints_size(num_constraints) +
 242            verifier_constraint_flags_size(num_constraints);
 243   }
 244 
 245 private:
 246   size_t crc_offset() const {
 247     return header_size_size();
 248   }
 249   size_t verifier_constraints_offset() const {
 250     return crc_offset() + crc_size(_klass);
 251   }
 252   size_t verifier_constraint_flags_offset() const {
 253     return verifier_constraints_offset() + verifier_constraints_size(_num_constraints);
 254   }
 255 
 256   void check_constraint_offset(int i) const {
 257     assert(0 <= i && i < _num_constraints, "sanity");
 258   }
 259 
 260 public:
 261   CrcInfo* crc() const {
 262     assert(crc_size(_klass) > 0, "must be");
 263     return (CrcInfo*)(address(this) + crc_offset());
 264   }
 265   RTConstraint* verifier_constraints() {
 266     assert(_num_constraints > 0, "sanity");
 267     return (RTConstraint*)(address(this) + verifier_constraints_offset());
 268   }
 269   RTConstraint* verifier_constraint_at(int i) {
 270     check_constraint_offset(i);
 271     return verifier_constraints() + i;
 272   }
 273 
 274   char* verifier_constraint_flags() {
 275     assert(_num_constraints > 0, "sanity");
 276     return (char*)(address(this) + verifier_constraint_flags_offset());
 277   }
 278 
 279   static u4 object_delta_u4(Symbol* sym) {
 280     if (DynamicDumpSharedSpaces) {
 281       sym = DynamicArchive::original_to_target(sym);
 282     }
 283     return MetaspaceShared::object_delta_u4(sym);
 284   }
 285 
 286   void init(DumpTimeSharedClassInfo& info) {
 287     _klass = info._klass;
 288     if (!SystemDictionaryShared::is_builtin(_klass)) {
 289       CrcInfo* c = crc();
 290       c->_clsfile_size = info._clsfile_size;
 291       c->_clsfile_crc32 = info._clsfile_crc32;
 292     }
 293     _num_constraints = info.num_constraints();
 294     if (_num_constraints > 0) {
 295       RTConstraint* constraints = verifier_constraints();
 296       char* flags = verifier_constraint_flags();
 297       int i;
 298       for (i = 0; i < _num_constraints; i++) {
 299         constraints[i]._name      = object_delta_u4(info._verifier_constraints->at(i)._name);
 300         constraints[i]._from_name = object_delta_u4(info._verifier_constraints->at(i)._from_name);
 301       }
 302       for (i = 0; i < _num_constraints; i++) {
 303         flags[i] = info._verifier_constraint_flags->at(i);
 304       }
 305     }
 306     if (DynamicDumpSharedSpaces) {
 307       _klass = DynamicArchive::original_to_target(info._klass);
 308     }
 309     ArchivePtrMarker::mark_pointer(&_klass);
 310   }
 311 
 312   bool matches(int clsfile_size, int clsfile_crc32) const {
 313     return crc()->_clsfile_size  == clsfile_size &&
 314            crc()->_clsfile_crc32 == clsfile_crc32;
 315   }
 316 
 317   Symbol* get_constraint_name(int i) {
 318     return (Symbol*)(SharedBaseAddress + verifier_constraint_at(i)->_name);
 319   }
 320   Symbol* get_constraint_from_name(int i) {
 321     return (Symbol*)(SharedBaseAddress + verifier_constraint_at(i)->_from_name);
 322   }
 323 
 324   char get_constraint_flag(int i) {
 325     check_constraint_offset(i);
 326     return verifier_constraint_flags()[i];
 327   }
 328 
 329 private:
 330   // ArchiveCompactor::allocate() has reserved a pointer immediately before
 331   // archived InstanceKlasses. We can use this slot to do a quick
 332   // lookup of InstanceKlass* -> RunTimeSharedClassInfo* without
 333   // building a new hashtable.
 334   //
 335   //  info_pointer_addr(klass) --> 0x0100   RunTimeSharedClassInfo*
 336   //  InstanceKlass* klass     --> 0x0108   <C++ vtbl>
 337   //                               0x0110   fields from Klass ...
 338   static RunTimeSharedClassInfo** info_pointer_addr(InstanceKlass* klass) {
 339     return &((RunTimeSharedClassInfo**)klass)[-1];
 340   }
 341 
 342 public:
 343   static RunTimeSharedClassInfo* get_for(InstanceKlass* klass) {
 344     return *info_pointer_addr(klass);
 345   }
 346   static void set_for(InstanceKlass* klass, RunTimeSharedClassInfo* record) {
 347     if (DynamicDumpSharedSpaces) {
 348       klass = DynamicArchive::original_to_buffer(klass);
 349       *info_pointer_addr(klass) = DynamicArchive::buffer_to_target(record);
 350     } else {
 351       *info_pointer_addr(klass) = record;
 352     }
 353 
 354     ArchivePtrMarker::mark_pointer(info_pointer_addr(klass));
 355   }
 356 
 357   // Used by RunTimeSharedDictionary to implement OffsetCompactHashtable::EQUALS
 358   static inline bool EQUALS(
 359        const RunTimeSharedClassInfo* value, Symbol* key, int len_unused) {
 360     return (value->_klass->name() == key);
 361   }
 362 };

 390 Handle SystemDictionaryShared::get_shared_jar_manifest(int shared_path_index, TRAPS) {
 391   Handle manifest ;
 392   if (shared_jar_manifest(shared_path_index) == NULL) {
 393     SharedClassPathEntry* ent = FileMapInfo::shared_path(shared_path_index);
 394     long size = ent->manifest_size();
 395     if (size <= 0) {
 396       return Handle();
 397     }
 398 
 399     // ByteArrayInputStream bais = new ByteArrayInputStream(buf);
 400     const char* src = ent->manifest();
 401     assert(src != NULL, "No Manifest data");
 402     typeArrayOop buf = oopFactory::new_byteArray(size, CHECK_NH);
 403     typeArrayHandle bufhandle(THREAD, buf);
 404     ArrayAccess<>::arraycopy_from_native(reinterpret_cast<const jbyte*>(src),
 405                                          buf, typeArrayOopDesc::element_offset<jbyte>(0), size);
 406 
 407     Handle bais = JavaCalls::construct_new_instance(SystemDictionary::ByteArrayInputStream_klass(),
 408                       vmSymbols::byte_array_void_signature(),
 409                       bufhandle, CHECK_NH);
 410 
 411     // manifest = new Manifest(bais)
 412     manifest = JavaCalls::construct_new_instance(SystemDictionary::Jar_Manifest_klass(),
 413                       vmSymbols::input_stream_void_signature(),
 414                       bais, CHECK_NH);
 415     atomic_set_shared_jar_manifest(shared_path_index, manifest());
 416   }
 417 
 418   manifest = Handle(THREAD, shared_jar_manifest(shared_path_index));
 419   assert(manifest.not_null(), "sanity");
 420   return manifest;
 421 }

 423 Handle SystemDictionaryShared::get_shared_jar_url(int shared_path_index, TRAPS) {
 424   Handle url_h;
 425   if (shared_jar_url(shared_path_index) == NULL) {
 426     JavaValue result(T_OBJECT);
 427     const char* path = FileMapInfo::shared_path_name(shared_path_index);
 428     Handle path_string = java_lang_String::create_from_str(path, CHECK_(url_h));
 429     Klass* classLoaders_klass =
 430         SystemDictionary::jdk_internal_loader_ClassLoaders_klass();
 431     JavaCalls::call_static(&result, classLoaders_klass,
 432                            vmSymbols::toFileURL_name(),
 433                            vmSymbols::toFileURL_signature(),
 434                            path_string, CHECK_(url_h));
 435 
 436     atomic_set_shared_jar_url(shared_path_index, (oop)result.get_jobject());
 437   }
 438 
 439   url_h = Handle(THREAD, shared_jar_url(shared_path_index));
 440   assert(url_h.not_null(), "sanity");
 441   return url_h;

 456 // Define Package for shared app classes from JAR file and also checks for
 457 // package sealing (all done in Java code)
 458 // See http://docs.oracle.com/javase/tutorial/deployment/jar/sealman.html
 459 void SystemDictionaryShared::define_shared_package(Symbol*  class_name,
 460                                                    Handle class_loader,
 461                                                    Handle manifest,
 462                                                    Handle url,
 463                                                    TRAPS) {
 464   assert(SystemDictionary::is_system_class_loader(class_loader()), "unexpected class loader");
 465   // get_package_name() returns a NULL handle if the class is in unnamed package
 466   Handle pkgname_string = get_package_name(class_name, CHECK);
 467   if (pkgname_string.not_null()) {
 468     Klass* app_classLoader_klass = SystemDictionary::jdk_internal_loader_ClassLoaders_AppClassLoader_klass();
 469     JavaValue result(T_OBJECT);
 470     JavaCallArguments args(3);
 471     args.set_receiver(class_loader);
 472     args.push_oop(pkgname_string);
 473     args.push_oop(manifest);
 474     args.push_oop(url);
 475     JavaCalls::call_virtual(&result, app_classLoader_klass,
 476                             vmSymbols::defineOrCheckPackage_name(),
 477                             vmSymbols::defineOrCheckPackage_signature(),
 478                             &args,
 479                             CHECK);
 480   }
 481 }

 501 // Returns the ProtectionDomain associated with the JAR file identified by the url.
 502 Handle SystemDictionaryShared::get_shared_protection_domain(Handle class_loader,
 503                                                             int shared_path_index,
 504                                                             Handle url,
 505                                                             TRAPS) {
 506   Handle protection_domain;
 507   if (shared_protection_domain(shared_path_index) == NULL) {
 508     Handle pd = get_protection_domain_from_classloader(class_loader, url, THREAD);
 509     atomic_set_shared_protection_domain(shared_path_index, pd());
 510   }
 511 
 512   // Acquire from the cache because if another thread beats the current one to
 513   // set the shared protection_domain and the atomic_set fails, the current thread
 514   // needs to get the updated protection_domain from the cache.
 515   protection_domain = Handle(THREAD, shared_protection_domain(shared_path_index));
 516   assert(protection_domain.not_null(), "sanity");
 517   return protection_domain;
 518 }

 520 // Returns the ProtectionDomain associated with the moduleEntry.
 521 Handle SystemDictionaryShared::get_shared_protection_domain(Handle class_loader,
 522                                                             ModuleEntry* mod, TRAPS) {
 523   ClassLoaderData *loader_data = mod->loader_data();
 524   if (mod->shared_protection_domain() == NULL) {
 525     Symbol* location = mod->location();
 526     if (location != NULL) {
 527       Handle location_string = java_lang_String::create_from_symbol(
 528                                      location, CHECK_NH);
 529       Handle url;
 530       JavaValue result(T_OBJECT);
 531       if (location->starts_with("jrt:/")) {
 532         url = JavaCalls::construct_new_instance(SystemDictionary::URL_klass(),
 533                                                 vmSymbols::string_void_signature(),
 534                                                 location_string, CHECK_NH);
 535       } else {
 536         Klass* classLoaders_klass =
 537           SystemDictionary::jdk_internal_loader_ClassLoaders_klass();
 538         JavaCalls::call_static(&result, classLoaders_klass, vmSymbols::toFileURL_name(),
 539                                vmSymbols::toFileURL_signature(),
 540                                location_string, CHECK_NH);
 541         url = Handle(THREAD, (oop)result.get_jobject());
 542       }
 543 
 544       Handle pd = get_protection_domain_from_classloader(class_loader, url,
 545                                                          CHECK_NH);
 546       mod->set_shared_protection_domain(loader_data, pd);
 547     }
 548   }
 549 
 550   Handle protection_domain(THREAD, mod->shared_protection_domain());
 551   assert(protection_domain.not_null(), "sanity");
 552   return protection_domain;
 553 }

 555 // Initializes the java.lang.Package and java.security.ProtectionDomain objects associated with
 556 // the given InstanceKlass.
 557 // Returns the ProtectionDomain for the InstanceKlass.
 558 Handle SystemDictionaryShared::init_security_info(Handle class_loader, InstanceKlass* ik, PackageEntry* pkg_entry, TRAPS) {
 559   Handle pd;
 560 
 561   if (ik != NULL) {
 562     int index = ik->shared_classpath_index();
 563     assert(index >= 0, "Sanity");
 564     SharedClassPathEntry* ent = FileMapInfo::shared_path(index);
 565     Symbol* class_name = ik->name();
 566 
 567     if (ent->is_modules_image()) {
 568       // For shared app/platform classes originated from the run-time image:
 569       //   The ProtectionDomains are cached in the corresponding ModuleEntries
 570       //   for fast access by the VM.
 571       // all packages from module image are already created during VM bootstrap in
 572       // Modules::define_module().
 573       assert(pkg_entry != NULL, "archived class in module image cannot be from unnamed package");
 574       ModuleEntry* mod_entry = pkg_entry->module();
 575       pd = get_shared_protection_domain(class_loader, mod_entry, THREAD);
 576     } else {
 577       // For shared app/platform classes originated from JAR files on the class path:
 578       //   Each of the 3 SystemDictionaryShared::_shared_xxx arrays has the same length
 579       //   as the shared classpath table in the shared archive (see
 580       //   FileMap::_shared_path_table in filemap.hpp for details).
 581       //
 582       //   If a shared InstanceKlass k is loaded from the class path, let
 583       //
 584       //     index = k->shared_classpath_index():
 585       //
 586       //   FileMap::_shared_path_table[index] identifies the JAR file that contains k.
 587       //
 588       //   k's protection domain is:
 589       //
 590       //     ProtectionDomain pd = _shared_protection_domains[index];
 591       //
 592       //   and k's Package is initialized using
 593       //
 594       //     manifest = _shared_jar_manifests[index];
 595       //     url = _shared_jar_urls[index];
 596       //     define_shared_package(class_name, class_loader, manifest, url, CHECK_(pd));
 597       //
 598       //   Note that if an element of these 3 _shared_xxx arrays is NULL, it will be initialized by
 599       //   the corresponding SystemDictionaryShared::get_shared_xxx() function.
 600       Handle manifest = get_shared_jar_manifest(index, CHECK_(pd));
 601       Handle url = get_shared_jar_url(index, CHECK_(pd));
 602       define_shared_package(class_name, class_loader, manifest, url, CHECK_(pd));
 603       pd = get_shared_protection_domain(class_loader, index, url, CHECK_(pd));
 604     }
 605   }
 606   return pd;
 607 }

 609 bool SystemDictionaryShared::is_sharing_possible(ClassLoaderData* loader_data) {
 610   oop class_loader = loader_data->class_loader();
 611   return (class_loader == NULL ||
 612           SystemDictionary::is_system_class_loader(class_loader) ||

 616 // Currently AppCDS only archives classes from the run-time image, the
 617 // -Xbootclasspath/a path, the class path, and the module path.
 618 //
 619 // Check if a shared class can be loaded by the specific classloader. Following
 620 // are the "visible" archived classes for different classloaders.
 621 //
 622 // NULL classloader:
 623 //   - see SystemDictionary::is_shared_class_visible()
 624 // Platform classloader:
 625 //   - Module class from runtime image. ModuleEntry must be defined in the
 626 //     classloader.
 627 // App classloader:
 628 //   - Module Class from runtime image and module path. ModuleEntry must be defined in the
 629 //     classloader.
 630 //   - Class from -cp. The class must have no PackageEntry defined in any of the
 631 //     boot/platform/app classloader, or must be in the unnamed module defined in the
 632 //     AppClassLoader.
 633 bool SystemDictionaryShared::is_shared_class_visible_for_classloader(
 634                                                      InstanceKlass* ik,
 635                                                      Handle class_loader,
 636                                                      Symbol* pkg_name,
 637                                                      PackageEntry* pkg_entry,
 638                                                      ModuleEntry* mod_entry,
 639                                                      TRAPS) {
 640   assert(class_loader.not_null(), "Class loader should not be NULL");
 641   assert(Universe::is_module_initialized(), "Module system is not initialized");
 642   ResourceMark rm(THREAD);
 643 
 644   int path_index = ik->shared_classpath_index();
 645   SharedClassPathEntry* ent =
 646             (SharedClassPathEntry*)FileMapInfo::shared_path(path_index);
 647 
 648   if (SystemDictionary::is_platform_class_loader(class_loader())) {
 649     assert(ent != NULL, "shared class for PlatformClassLoader should have valid SharedClassPathEntry");
 650     // The PlatformClassLoader can only load archived class originated from the
 651     // run-time image. The class' PackageEntry/ModuleEntry must be
 652     // defined by the PlatformClassLoader.
 653     if (mod_entry != NULL) {
 654       // PackageEntry/ModuleEntry is found in the classloader. Check if the
 655       // ModuleEntry's location agrees with the archived class' origination.
 656       if (ent->is_modules_image() && mod_entry->location()->starts_with("jrt:")) {
 657         return true; // Module class from the runtime image
 658       }
 659     }
 660   } else if (SystemDictionary::is_system_class_loader(class_loader())) {
 661     assert(ent != NULL, "shared class for system loader should have valid SharedClassPathEntry");
 662     if (pkg_name == NULL) {
 663       // The archived class is in the unnamed package. Currently, the boot image
 664       // does not contain any class in the unnamed package.
 665       assert(!ent->is_modules_image(), "Class in the unnamed package must be from the classpath");
 666       if (path_index >= ClassLoaderExt::app_class_paths_start_index()) {
 667         assert(path_index < ClassLoaderExt::app_module_paths_start_index(), "invalid path_index");
 668         return true;
 669       }
 670     } else {
 671       // Check if this is from a PackageEntry/ModuleEntry defined in the AppClassloader.
 672       if (pkg_entry == NULL) {
 673         // It's not guaranteed that the class is from the classpath if the
 674         // PackageEntry cannot be found from the AppClassloader. Need to check
 675         // the boot and platform classloader as well.
 676         ClassLoaderData* platform_loader_data =
 677           ClassLoaderData::class_loader_data_or_null(SystemDictionary::java_platform_loader()); // can be NULL during bootstrap
 678         if ((platform_loader_data == NULL ||
 679              ClassLoader::get_package_entry(pkg_name, platform_loader_data) == NULL) &&
 680              ClassLoader::get_package_entry(pkg_name, ClassLoaderData::the_null_class_loader_data()) == NULL) {
 681           // The PackageEntry is not defined in any of the boot/platform/app classloaders.
 682           // The archived class must from -cp path and not from the runtime image.
 683           if (!ent->is_modules_image() && path_index >= ClassLoaderExt::app_class_paths_start_index() &&
 684                                           path_index < ClassLoaderExt::app_module_paths_start_index()) {
 685             return true;
 686           }
 687         }
 688       } else if (mod_entry != NULL) {
 689         // The package/module is defined in the AppClassLoader. We support
 690         // archiving application module class from the runtime image or from
 691         // a named module from a module path.
 692         // Packages from the -cp path are in the unnamed_module.
 693         if (ent->is_modules_image() && mod_entry->location()->starts_with("jrt:")) {
 694           // shared module class from runtime image
 695           return true;
 696         } else if (pkg_entry->in_unnamed_module() && path_index >= ClassLoaderExt::app_class_paths_start_index() &&
 697             path_index < ClassLoaderExt::app_module_paths_start_index()) {
 698           // shared class from -cp
 699           DEBUG_ONLY( \
 700             ClassLoaderData* loader_data = class_loader_data(class_loader); \
 701             assert(mod_entry == loader_data->unnamed_module(), "the unnamed module is not defined in the classloader");)
 702           return true;
 703         } else {
 704           if(!pkg_entry->in_unnamed_module() &&
 705               (path_index >= ClassLoaderExt::app_module_paths_start_index())&&
 706               (path_index < FileMapInfo::get_number_of_shared_paths()) &&
 707               (strcmp(ent->name(), ClassLoader::skip_uri_protocol(mod_entry->location()->as_C_string())) == 0)) {
 708             // shared module class from module path
 709             return true;
 710           } else {
 711             assert(path_index < FileMapInfo::get_number_of_shared_paths(), "invalid path_index");
 712           }
 713         }
 714       }
 715     }
 716   } else {
 717     // TEMP: if a shared class can be found by a custom loader, consider it visible now.
 718     // FIXME: is this actually correct?
 719     return true;
 720   }
 721   return false;
 722 }

 735 // The following stack shows how this code is reached:
 736 //
 737 //   [0] SystemDictionaryShared::find_or_load_shared_class()
 738 //   [1] JVM_FindLoadedClass
 739 //   [2] java.lang.ClassLoader.findLoadedClass0()
 740 //   [3] java.lang.ClassLoader.findLoadedClass()
 741 //   [4] jdk.internal.loader.BuiltinClassLoader.loadClassOrNull()
 742 //   [5] jdk.internal.loader.BuiltinClassLoader.loadClass()
 743 //   [6] jdk.internal.loader.ClassLoaders$AppClassLoader.loadClass(), or
 744 //       jdk.internal.loader.ClassLoaders$PlatformClassLoader.loadClass()
 745 //
 746 // AppCDS supports fast class loading for these 2 built-in class loaders:
 747 //    jdk.internal.loader.ClassLoaders$PlatformClassLoader
 748 //    jdk.internal.loader.ClassLoaders$AppClassLoader
 749 // with the following assumptions (based on the JDK core library source code):
 750 //
 751 // [a] these two loaders use the BuiltinClassLoader.loadClassOrNull() to
 752 //     load the named class.
 753 // [b] BuiltinClassLoader.loadClassOrNull() first calls findLoadedClass(name).
 754 // [c] At this point, if we can find the named class inside the
 755 //     shared_dictionary, we can perform further checks (see
 756 //     is_shared_class_visible_for_classloader() to ensure that this class
 757 //     was loaded by the same class loader during dump time.
 758 //
 759 // Given these assumptions, we intercept the findLoadedClass() call to invoke
 760 // SystemDictionaryShared::find_or_load_shared_class() to load the shared class from
 761 // the archive for the 2 built-in class loaders. This way,
 762 // we can improve start-up because we avoid decoding the classfile,
 763 // and avoid delegating to the parent loader.
 764 //
 765 // NOTE: there's a lot of assumption about the Java code. If any of that change, this
 766 // needs to be redesigned.
 767 
 768 InstanceKlass* SystemDictionaryShared::find_or_load_shared_class(
 769                  Symbol* name, Handle class_loader, TRAPS) {
 770   InstanceKlass* k = NULL;
 771   if (UseSharedSpaces) {
 772     if (!has_platform_or_app_classes()) {
 773       return NULL;
 774     }
 775 
 776     if (SystemDictionary::is_system_class_loader(class_loader()) ||
 777         SystemDictionary::is_platform_class_loader(class_loader())) {
 778       // Fix for 4474172; see evaluation for more details
 779       class_loader = Handle(
 780         THREAD, java_lang_ClassLoader::non_reflection_class_loader(class_loader()));
 781       ClassLoaderData *loader_data = register_loader(class_loader);
 782       Dictionary* dictionary = loader_data->dictionary();
 783 
 784       unsigned int d_hash = dictionary->compute_hash(name);
 785 
 786       bool DoObjectLock = true;
 787       if (is_parallelCapable(class_loader)) {
 788         DoObjectLock = false;
 789       }
 790 
 791       // Make sure we are synchronized on the class loader before we proceed
 792       //
 793       // Note: currently, find_or_load_shared_class is called only from
 794       // JVM_FindLoadedClass and used for PlatformClassLoader and AppClassLoader,
 795       // which are parallel-capable loaders, so this lock is NOT taken.
 796       Handle lockObject = compute_loader_lock_object(class_loader, THREAD);
 797       check_loader_lock_contention(lockObject, THREAD);
 798       ObjectLocker ol(lockObject, THREAD, DoObjectLock);
 799 
 800       {
 801         MutexLocker mu(THREAD, SystemDictionary_lock);
 802         InstanceKlass* check = find_class(d_hash, name, dictionary);
 803         if (check != NULL) {
 804           return check;
 805         }
 806       }
 807 
 808       k = load_shared_class_for_builtin_loader(name, class_loader, THREAD);
 809       if (k != NULL) {
 810         define_instance_class(k, CHECK_NULL);
 811       }
 812     }
 813   }
 814   return k;
 815 }

 878 // This function is called for loading only UNREGISTERED classes
 879 InstanceKlass* SystemDictionaryShared::lookup_from_stream(Symbol* class_name,
 880                                                           Handle class_loader,
 881                                                           Handle protection_domain,
 882                                                           const ClassFileStream* cfs,
 883                                                           TRAPS) {
 884   if (!UseSharedSpaces) {
 885     return NULL;
 886   }
 887   if (class_name == NULL) {  // don't do this for hidden and unsafe anonymous classes
 888     return NULL;
 889   }
 890   if (class_loader.is_null() ||
 891       SystemDictionary::is_system_class_loader(class_loader()) ||
 892       SystemDictionary::is_platform_class_loader(class_loader())) {
 893     // Do nothing for the BUILTIN loaders.
 894     return NULL;
 895   }
 896 
 897   const RunTimeSharedClassInfo* record = find_record(&_unregistered_dictionary, &_dynamic_unregistered_dictionary, class_name);
 898   if (record == NULL) {
 899     return NULL;
 900   }
 901 
 902   int clsfile_size  = cfs->length();
 903   int clsfile_crc32 = ClassLoader::crc32(0, (const char*)cfs->buffer(), cfs->length());
 904 
 905   if (!record->matches(clsfile_size, clsfile_crc32)) {
 906     return NULL;
 907   }
 908 
 909   return acquire_class_for_current_thread(record->_klass, class_loader,
 910                                           protection_domain, cfs,
 911                                           THREAD);
 912 }

 914 InstanceKlass* SystemDictionaryShared::acquire_class_for_current_thread(
 915                    InstanceKlass *ik,
 916                    Handle class_loader,
 917                    Handle protection_domain,
 918                    const ClassFileStream *cfs,
 919                    TRAPS) {
 920   ClassLoaderData* loader_data = ClassLoaderData::class_loader_data(class_loader());
 921 
 922   {
 923     MutexLocker mu(THREAD, SharedDictionary_lock);
 924     if (ik->class_loader_data() != NULL) {
 925       //    ik is already loaded (by this loader or by a different loader)
 926       // or ik is being loaded by a different thread (by this loader or by a different loader)
 927       return NULL;
 928     }
 929 
 930     // No other thread has acquired this yet, so give it to *this thread*
 931     ik->set_class_loader_data(loader_data);
 932   }
 933 
 934   // No longer holding SharedDictionary_lock
 935   // No need to lock, as <ik> can be held only by a single thread.
 936   loader_data->add_class(ik);
 937 
 938   // Get the package entry.
 939   PackageEntry* pkg_entry = get_package_entry_from_class_name(class_loader, ik->name());
 940 
 941   // Load and check super/interfaces, restore unsharable info
 942   InstanceKlass* shared_klass = load_shared_class(ik, class_loader, protection_domain,
 943                                                   cfs, pkg_entry, THREAD);
 944   if (shared_klass == NULL || HAS_PENDING_EXCEPTION) {
 945     // TODO: clean up <ik> so it can be used again
 946     return NULL;
 947   }
 948 
 949   return shared_klass;
 950 }

 975 // This function is called to resolve the super/interfaces of shared classes for
 976 // non-built-in loaders. E.g., ChildClass in the below example
 977 // where "super:" (and optionally "interface:") have been specified.
 978 //
 979 // java/lang/Object id: 0
 980 // Interface   id: 2 super: 0 source: cust.jar
 981 // ChildClass  id: 4 super: 0 interfaces: 2 source: cust.jar
 982 InstanceKlass* SystemDictionaryShared::dump_time_resolve_super_or_fail(
 983     Symbol* child_name, Symbol* class_name, Handle class_loader,
 984     Handle protection_domain, bool is_superclass, TRAPS) {
 985 
 986   assert(DumpSharedSpaces, "only when dumping");
 987 
 988   ClassListParser* parser = ClassListParser::instance();
 989   if (parser == NULL) {
 990     // We're still loading the well-known classes, before the ClassListParser is created.
 991     return NULL;
 992   }
 993   if (child_name->equals(parser->current_class_name())) {
 994     // When this function is called, all the numbered super and interface types
 995     // must have already been loaded. Hence this function is never recursively called.
 996     if (is_superclass) {
 997       return parser->lookup_super_for_current_class(class_name);
 998     } else {
 999       return parser->lookup_interface_for_current_class(class_name);
1000     }
1001   } else {
1002     // The VM is not trying to resolve a super type of parser->current_class_name().
1003     // Instead, it's resolving an error class (because parser->current_class_name() has
1004     // failed parsing or verification). Don't do anything here.
1005     return NULL;
1006   }
1007 }

1068 bool SystemDictionaryShared::should_be_excluded(InstanceKlass* k) {
1069   if (k->is_hidden() || k->is_unsafe_anonymous()) {
1070     warn_excluded(k, "Hidden or Unsafe anonymous class");
1071     return true; // hidden and unsafe anonymous classes are not archived, skip
1072   }
1073   if (k->is_in_error_state()) {
1074     warn_excluded(k, "In error state");
1075     return true;
1076   }
1077   if (k->has_been_redefined()) {
1078     warn_excluded(k, "Has been redefined");
1079     return true;
1080   }
1081   if (k->shared_classpath_index() < 0 && is_builtin(k)) {
1082     // These are classes loaded from unsupported locations (such as those loaded by JVMTI native
1083     // agent during dump time).
1084     warn_excluded(k, "Unsupported location");
1085     return true;
1086   }
1087   if (k->signers() != NULL) {
1088     // We cannot include signed classes in the archive because the certificates
1089     // used during dump time may be different than those used during
1090     // runtime (due to expiration, etc).
1091     warn_excluded(k, "Signed JAR");
1092     return true;
1093   }
1094   if (is_jfr_event_class(k)) {
1095     // We cannot include JFR event classes because they need runtime-specific
1096     // instrumentation in order to work with -XX:FlightRecorderOptions=retransform=false.
1097     // There are only a small number of these classes, so it's not worthwhile to
1098     // support them and make CDS more complicated.
1099     warn_excluded(k, "JFR event class");
1100     return true;
1101   }
1102   if (k->init_state() < InstanceKlass::linked) {
1103     // In CDS dumping, we will attempt to link all classes. Those that fail to link will
1104     // be recorded in DumpTimeSharedClassInfo.
1105     Arguments::assert_is_dumping_archive();
1106 
1107     // TODO -- rethink how this can be handled.
1108     // We should try to link ik, however, we can't do it here because
1109     // 1. We are at VM exit
1110     // 2. linking a class may cause other classes to be loaded, which means
1111     //    a custom ClassLoader.loadClass() may be called, at a point where the
1112     //    class loader doesn't expect it.
1113     if (has_class_failed_verification(k)) {
1114       warn_excluded(k, "Failed verification");
1115     } else {
1116       warn_excluded(k, "Not linked");
1117     }
1118     return true;
1119   }
1120   if (k->major_version() < 50 /*JAVA_6_VERSION*/) {
1121     ResourceMark rm;
1122     log_warning(cds)("Pre JDK 6 class not supported by CDS: %u.%u %s",
1123                      k->major_version(),  k->minor_version(), k->name()->as_C_string());
1124     return true;
1125   }
1126 
1127   InstanceKlass* super = k->java_super();
1128   if (super != NULL && should_be_excluded(super)) {
1129     ResourceMark rm;
1130     log_warning(cds)("Skipping %s: super class %s is excluded", k->name()->as_C_string(), super->name()->as_C_string());
1131     return true;
1132   }
1133 
1134   Array<InstanceKlass*>* interfaces = k->local_interfaces();
1135   int len = interfaces->length();
1136   for (int i = 0; i < len; i++) {
1137     InstanceKlass* intf = interfaces->at(i);
1138     if (should_be_excluded(intf)) {
1139       log_warning(cds)("Skipping %s: interface %s is excluded", k->name()->as_C_string(), intf->name()->as_C_string());
1140       return true;
1141     }
1142   }
1143 
1144   return false;
1145 }

1148 void SystemDictionaryShared::validate_before_archiving(InstanceKlass* k) {
1149   ResourceMark rm;
1150   const char* name = k->name()->as_C_string();
1151   DumpTimeSharedClassInfo* info = _dumptime_table->get(k);

1220 bool SystemDictionaryShared::add_verification_constraint(InstanceKlass* k, Symbol* name,
1221          Symbol* from_name, bool from_field_is_protected, bool from_is_array, bool from_is_object) {
1222   Arguments::assert_is_dumping_archive();
1223   DumpTimeSharedClassInfo* info = find_or_allocate_info_for(k);
1224   info->add_verification_constraint(k, name, from_name, from_field_is_protected,
1225                                     from_is_array, from_is_object);
1226 
1227   if (DynamicDumpSharedSpaces) {
1228     // For dynamic dumping, we can resolve all the constraint classes for all class loaders during
1229     // the initial run prior to creating the archive before vm exit. We will also perform verification
1230     // check when running with the archive.
1231     return false;
1232   } else {
1233     if (is_builtin(k)) {
1234       // For builtin class loaders, we can try to complete the verification check at dump time,
1235       // because we can resolve all the constraint classes. We will also perform verification check
1236       // when running with the archive.
1237       return false;
1238     } else {
1239       // For non-builtin class loaders, we cannot complete the verification check at dump time,
1240       // because at dump time we don't know how to resolve classes for such loaders.
1241       return true;
1242     }
1243   }

1246 void DumpTimeSharedClassInfo::add_verification_constraint(InstanceKlass* k, Symbol* name,
1247          Symbol* from_name, bool from_field_is_protected, bool from_is_array, bool from_is_object) {
1248   if (_verifier_constraints == NULL) {
1249     _verifier_constraints = new(ResourceObj::C_HEAP, mtClass) GrowableArray<DTConstraint>(4, true, mtClass);
1250   }
1251   if (_verifier_constraint_flags == NULL) {
1252     _verifier_constraint_flags = new(ResourceObj::C_HEAP, mtClass) GrowableArray<char>(4, true, mtClass);
1253   }
1254   GrowableArray<DTConstraint>* vc_array = _verifier_constraints;
1255   for (int i = 0; i < vc_array->length(); i++) {
1256     DTConstraint* p = vc_array->adr_at(i);
1257     if (name == p->_name && from_name == p->_from_name) {
1258       return;
1259     }
1260   }
1261   DTConstraint cons(name, from_name);
1262   vc_array->append(cons);
1263 
1264   GrowableArray<char>* vcflags_array = _verifier_constraint_flags;
1265   char c = 0;
1266   c |= from_field_is_protected ? SystemDictionaryShared::FROM_FIELD_IS_PROTECTED : 0;
1267   c |= from_is_array           ? SystemDictionaryShared::FROM_IS_ARRAY           : 0;
1268   c |= from_is_object          ? SystemDictionaryShared::FROM_IS_OBJECT          : 0;
1269   vcflags_array->append(c);
1270 
1271   if (log_is_enabled(Trace, cds, verification)) {
1272     ResourceMark rm;
1273     log_trace(cds, verification)("add_verification_constraint: %s: %s must be subclass of %s [0x%x]",
1274                                  k->external_name(), from_name->as_klass_external_name(),
1275                                  name->as_klass_external_name(), c);
1276   }
1277 }

1279 void SystemDictionaryShared::check_verification_constraints(InstanceKlass* klass,
1280                                                             TRAPS) {
1281   assert(!DumpSharedSpaces && UseSharedSpaces, "called at run time with CDS enabled only");
1282   RunTimeSharedClassInfo* record = RunTimeSharedClassInfo::get_for(klass);
1283 
1284   int length = record->_num_constraints;
1285   if (length > 0) {
1286     for (int i = 0; i < length; i++) {
1287       Symbol* name      = record->get_constraint_name(i);
1288       Symbol* from_name = record->get_constraint_from_name(i);
1289       char c            = record->get_constraint_flag(i);
1290 
1291       if (log_is_enabled(Trace, cds, verification)) {
1292         ResourceMark rm(THREAD);
1293         log_trace(cds, verification)("check_verification_constraint: %s: %s must be subclass of %s [0x%x]",
1294                                      klass->external_name(), from_name->as_klass_external_name(),
1295                                      name->as_klass_external_name(), c);
1296       }
1297 
1298       bool from_field_is_protected = (c & SystemDictionaryShared::FROM_FIELD_IS_PROTECTED) ? true : false;
1299       bool from_is_array           = (c & SystemDictionaryShared::FROM_IS_ARRAY)           ? true : false;
1300       bool from_is_object          = (c & SystemDictionaryShared::FROM_IS_OBJECT)          ? true : false;
1301 
1302       bool ok = VerificationType::resolve_and_check_assignability(klass, name,
1303          from_name, from_field_is_protected, from_is_array, from_is_object, CHECK);
1304       if (!ok) {
1305         ResourceMark rm(THREAD);
1306         stringStream ss;
1307 
1308         ss.print_cr("Bad type on operand stack");
1309         ss.print_cr("Exception Details:");
1310         ss.print_cr("  Location:\n    %s", klass->name()->as_C_string());
1311         ss.print_cr("  Reason:\n    Type '%s' is not assignable to '%s'",
1312                     from_name->as_quoted_ascii(), name->as_quoted_ascii());
1313         THROW_MSG(vmSymbols::java_lang_VerifyError(), ss.as_string());
1314       }
1315     }
1316   }

1353   CompactHashtableWriter* _writer;

1355 public:

1357                                bool is_builtin,
1358                                bool is_static_archive)
1359     : _writer(writer), _is_builtin(is_builtin) {}
1360 
1361   bool do_entry(InstanceKlass* k, DumpTimeSharedClassInfo& info) {
1362     if (!info.is_excluded() && info.is_builtin() == _is_builtin) {
1363       size_t byte_size = RunTimeSharedClassInfo::byte_size(info._klass, info.num_constraints());

1370       if (DynamicDumpSharedSpaces) {

1372       }
1373       hash = SystemDictionaryShared::hash_for_shared_dictionary(name);
1374       u4 delta;
1375       if (DynamicDumpSharedSpaces) {
1376         delta = MetaspaceShared::object_delta_u4(DynamicArchive::buffer_to_target(record));
1377       } else {
1378         delta = MetaspaceShared::object_delta_u4(record);
1379       }

1381       if (log_is_enabled(Trace, cds, hashtables)) {
1382         ResourceMark rm;
1383         log_trace(cds,hashtables)("%s dictionary: %s", (_is_builtin ? "builtin" : "unregistered"), info._klass->external_name());
1384       }
1385 
1386       // Save this for quick runtime lookup of InstanceKlass* -> RunTimeSharedClassInfo*
1387       RunTimeSharedClassInfo::set_for(info._klass, record);
1388     }
1389     return true; // keep on iterating
1390   }

1432 SystemDictionaryShared::find_record(RunTimeSharedDictionary* static_dict, RunTimeSharedDictionary* dynamic_dict, Symbol* name) {
1433   if (!UseSharedSpaces || !name->is_shared()) {
1434     // The names of all shared classes must also be a shared Symbol.

1437

1441     // The names of all shared classes in the static dict must also be in the

1444   }
1445 
1446   if (record == NULL && DynamicArchive::is_mapped()) {
1447     record = dynamic_dict->lookup(name, hash, 0);
1448   }
1449 
1450   return record;
1451 }
