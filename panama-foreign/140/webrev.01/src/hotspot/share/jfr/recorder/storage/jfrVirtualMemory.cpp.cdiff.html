    _rs = ReservedSpace(reservation_size_request_bytes,
                        os::vm_allocation_granularity(),
!                       UseLargePages && os::can_commit_large_page_memory(),
-                       false);
    if (!_rs.is_reserved()) {
      return false;
    }
    assert(_rs.base() != NULL, "invariant");
