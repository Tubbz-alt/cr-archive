25 #ifndef SHARE_GC_SHENANDOAH_SHENANDOAHPARALLELCLEANING_INLINE_HPP
26 #define SHARE_GC_SHENANDOAH_SHENANDOAHPARALLELCLEANING_INLINE_HPP
27 
28 #include "gc/shared/weakProcessor.inline.hpp"
29 #include "gc/shenandoah/shenandoahHeap.hpp"
30 #include "gc/shenandoah/shenandoahParallelCleaning.hpp"
31 #include "gc/shenandoah/shenandoahUtils.hpp"
32 #include "runtime/thread.hpp"
33 #include "runtime/safepoint.hpp"
34 
35 template<typename IsAlive, typename KeepAlive>
36 ShenandoahParallelWeakRootsCleaningTask<IsAlive, KeepAlive>::ShenandoahParallelWeakRootsCleaningTask(IsAlive* is_alive,

37                                                                                                      KeepAlive* keep_alive,
38                                                                                                      uint num_workers,
39                                                                                                      bool include_concurrent_roots) :
40   AbstractGangTask("Parallel Weak Root Cleaning Task"),
41   _weak_processing_task(num_workers),
42   _is_alive(is_alive), _keep_alive(keep_alive), _include_concurrent_roots(include_concurrent_roots) {
43   assert(SafepointSynchronize::is_at_safepoint(), "Must be at a safepoint");
44 
45   if (ShenandoahStringDedup::is_enabled()) {
46     StringDedup::gc_prologue(false);
47   }
48 }
49 
50 template<typename IsAlive, typename KeepAlive>
51 ShenandoahParallelWeakRootsCleaningTask<IsAlive, KeepAlive>::~ShenandoahParallelWeakRootsCleaningTask() {
52   if (StringDedup::is_enabled()) {
53     StringDedup::gc_epilogue();
54   }
55 }
56 
57 template<typename IsAlive, typename KeepAlive>
58 void ShenandoahParallelWeakRootsCleaningTask<IsAlive, KeepAlive>::work(uint worker_id) {
59   if (_include_concurrent_roots) {
60     _weak_processing_task.work<IsAlive, KeepAlive>(worker_id, _is_alive, _keep_alive);
61   } else {
62     _serial_weak_roots.weak_oops_do(_is_alive, _keep_alive, worker_id);
63   }
64 
65   if (ShenandoahStringDedup::is_enabled()) {
66     ShenandoahStringDedup::parallel_oops_do(_is_alive, _keep_alive, worker_id);
67   }
68 }
69 
70 #endif // SHARE_GC_SHENANDOAH_SHENANDOAHPARALLELCLEANING_INLINE_HPP
