 40 class MasterFreeRegionListChecker : public HeapRegionSetChecker {
 41 public:
 42   void check_mt_safety() {
 43     // Master Free List MT safety protocol:
 44     // (a) If we're at a safepoint, operations on the master free list
 45     // should be invoked by either the VM thread (which will serialize
 46     // them) or by the GC workers while holding the
 47     // FreeList_lock.
 48     // (b) If we're not at a safepoint, operations on the master free
 49     // list should be invoked while holding the Heap_lock.
 50 
 51     if (SafepointSynchronize::is_at_safepoint()) {
 52       guarantee(Thread::current()->is_VM_thread() ||
 53                 FreeList_lock->owned_by_self(), "master free list MT safety protocol at a safepoint");
 54     } else {
 55       guarantee(Heap_lock->owned_by_self(), "master free list MT safety protocol outside a safepoint");
 56     }
 57   }
 58   bool is_correct_type(HeapRegion* hr) { return hr->is_free(); }
 59   const char* get_description() { return "Free Regions"; }
 60 };

110 HeapRegion* HeapRegionManager::allocate_free_region(HeapRegionType type, uint requested_node_index) {
111   HeapRegion* hr = NULL;
112   bool from_head = !type.is_young();
113   G1NUMA* numa = G1NUMA::numa();
114 
115   if (requested_node_index != G1NUMA::AnyNodeIndex && numa->is_enabled()) {
116     // Try to allocate with requested node index.
117     hr = _free_list.remove_region_with_node_index(from_head, requested_node_index);
118   }
119 
120   if (hr == NULL) {
121     // If there's a single active node or we did not get a region from our requested node,
122     // try without requested node index.
123     hr = _free_list.remove_region(from_head);
124   }
125 
126   if (hr != NULL) {
127     assert(hr->next() == NULL, "Single region should not have next");
128     assert(is_available(hr->hrm_index()), "Must be committed");
129 
130     if (numa->is_enabled() && hr->node_index() < numa->num_active_nodes()) {
131       numa->update_statistics(G1NUMAStats::NewRegionAlloc, requested_node_index, hr->node_index());
132     }
133   }
134 
135   return hr;
136 }

181 void HeapRegionManager::commit_regions(uint index, size_t num_regions, WorkGang* pretouch_gang) {
182   guarantee(num_regions > 0, "Must commit more than zero regions");
183   guarantee(_num_committed + num_regions <= max_length(), "Cannot commit more than the maximum amount of regions");
184 
185   _num_committed += (uint)num_regions;
186 
187   _heap_mapper->commit_regions(index, num_regions, pretouch_gang);
188 
189   // Also commit auxiliary data
190   _prev_bitmap_mapper->commit_regions(index, num_regions, pretouch_gang);
191   _next_bitmap_mapper->commit_regions(index, num_regions, pretouch_gang);
192 
193   _bot_mapper->commit_regions(index, num_regions, pretouch_gang);
194   _cardtable_mapper->commit_regions(index, num_regions, pretouch_gang);
195 
196   _card_counts_mapper->commit_regions(index, num_regions, pretouch_gang);
197 }

199 void HeapRegionManager::uncommit_regions(uint start, size_t num_regions) {
200   guarantee(num_regions >= 1, "Need to specify at least one region to uncommit, tried to uncommit zero regions at %u", start);
201   guarantee(_num_committed >= num_regions, "pre-condition");
202 
203   // Reset node index to distinguish with committed regions.
204   for (uint i = start; i < start + num_regions; i++) {
205     at(i)->set_node_index(G1NUMA::UnknownNodeIndex);
206   }
207 
208   // Print before uncommitting.
209   if (G1CollectedHeap::heap()->hr_printer()->is_active()) {
210     for (uint i = start; i < start + num_regions; i++) {
211       HeapRegion* hr = at(i);
212       G1CollectedHeap::heap()->hr_printer()->uncommit(hr);
213     }
214   }
215 
216   _num_committed -= (uint)num_regions;
217 
218   _available_map.par_clear_range(start, start + num_regions, BitMap::unknown_range);
219   _heap_mapper->uncommit_regions(start, num_regions);
220 
221   // Also uncommit auxiliary data
222   _prev_bitmap_mapper->uncommit_regions(start, num_regions);
223   _next_bitmap_mapper->uncommit_regions(start, num_regions);
224 
225   _bot_mapper->uncommit_regions(start, num_regions);
226   _cardtable_mapper->uncommit_regions(start, num_regions);
227 
228   _card_counts_mapper->uncommit_regions(start, num_regions);
229 }

231 void HeapRegionManager::make_regions_available(uint start, uint num_regions, WorkGang* pretouch_gang) {
232   guarantee(num_regions > 0, "No point in calling this for zero regions");
233   commit_regions(start, num_regions, pretouch_gang);
234   for (uint i = start; i < start + num_regions; i++) {
235     if (_regions.get_by_index(i) == NULL) {
236       HeapRegion* new_hr = new_heap_region(i);
237       OrderAccess::storestore();
238       _regions.set_by_index(i, new_hr);
239       _allocated_heapregions_length = MAX2(_allocated_heapregions_length, i + 1);
240     }
241   }
242 
243   _available_map.par_set_range(start, start + num_regions, BitMap::unknown_range);
244 
245   for (uint i = start; i < start + num_regions; i++) {
246     assert(is_available(i), "Just made region %u available but is apparently not.", i);
247     HeapRegion* hr = at(i);
248     if (G1CollectedHeap::heap()->hr_printer()->is_active()) {
249       G1CollectedHeap::heap()->hr_printer()->commit(hr);
250     }
251 
252     hr->initialize();
253     hr->set_node_index(G1NUMA::numa()->index_for_region(hr));
254     insert_into_free_list(at(i));
255   }
256 }

280 uint HeapRegionManager::expand_at(uint start, uint num_regions, WorkGang* pretouch_workers) {
281   if (num_regions == 0) {
282     return 0;
283   }
284 
285   uint cur = start;
286   uint idx_last_found = 0;
287   uint num_last_found = 0;
288 
289   uint expanded = 0;
290 
291   while (expanded < num_regions &&
292          (num_last_found = find_unavailable_from_idx(cur, &idx_last_found)) > 0) {
293     uint to_expand = MIN2(num_regions - expanded, num_last_found);
294     make_regions_available(idx_last_found, to_expand, pretouch_workers);
295     expanded += to_expand;
296     cur = idx_last_found + num_last_found + 1;
297   }
298 
299   verify_optional();
300   return expanded;
301 }
302 
303 void HeapRegionManager::expand_exact(uint start, uint num_regions, WorkGang* pretouch_workers) {
304   assert(num_regions != 0, "Need to request at least one region");
305   uint end = start + num_regions;
306 
307   for (uint i = start; i < end; i++) {
308     if (!is_available(i)) {
309       make_regions_available(i, 1, pretouch_workers);
310     }
311   }
312 
313   verify_optional();
314 }

316 uint HeapRegionManager::expand_on_preferred_node(uint preferred_index) {
317   uint expand_candidate = UINT_MAX;
318   for (uint i = 0; i < max_length(); i++) {
319     if (is_available(i)) {
320       // Already in use continue
321       continue;
322     }
323     // Always save the candidate so we can expand later on.
324     expand_candidate = i;
325     if (is_on_preferred_index(expand_candidate, preferred_index)) {
326       // We have found a candidate on the preffered node, break.
327       break;
328     }
329   }
330 
331   if (expand_candidate == UINT_MAX) {
332      // No regions left, expand failed.
333     return 0;
334   }
335 
336   expand_exact(expand_candidate, 1, NULL);
337   return 1;
338 }

340 bool HeapRegionManager::is_on_preferred_index(uint region_index, uint preferred_node_index) {
341   uint region_node_index = G1NUMA::numa()->preferred_node_index_for_index(region_index);
342   return region_node_index == preferred_node_index;
343 }

345 #ifdef ASSERT
346 void HeapRegionManager::assert_contiguous_range(uint start, uint num_regions) {
347   // General sanity check, regions found should either be available and empty
348   // or not available so that we can make them available and use them.
349   for (uint i = start; i < (start + num_regions); i++) {
350     HeapRegion* hr = _regions.get_by_index(i);
351     assert(!is_available(i) || hr->is_free(),
352            "Found region sequence starting at " UINT32_FORMAT ", length " UINT32_FORMAT
353            " that is not free at " UINT32_FORMAT ". Hr is " PTR_FORMAT ", type is %s",
354            start, num_regions, i, p2i(hr), hr->get_type_str());
355   }
356 }
357 #endif

359 uint HeapRegionManager::find_contiguous_in_range(uint start, uint end, uint num_regions) {
360   assert(start <= end, "precondition");
361   assert(num_regions >= 1, "precondition");
362   uint candidate = start;       // First region in candidate sequence.
363   uint unchecked = candidate;   // First unchecked region in candidate.
364   // While the candidate sequence fits in the range...
365   while (num_regions <= (end - candidate)) {
366     // Walk backward over the regions for the current candidate.
367     for (uint i = candidate + num_regions - 1; true; --i) {
368       if (is_available(i) && !at(i)->is_free()) {
369         // Region i can't be used, so restart with i+1 as the start
370         // of a new candidate sequence, and with the region after the
371         // old candidate sequence being the first unchecked region.
372         unchecked = candidate + num_regions;
373         candidate = i + 1;
374         break;
375       } else if (i == unchecked) {
376         // All regions of candidate sequence have passed check.
377         assert_contiguous_range(candidate, num_regions);
378         return candidate;
379       }
380     }
381   }
382   return G1_NO_HRM_INDEX;
383 }

385 uint HeapRegionManager::find_contiguous_in_free_list(uint num_regions) {
386   BitMap::idx_t range_start = 0;
387   BitMap::idx_t range_end = range_start;
388   uint candidate = G1_NO_HRM_INDEX;
389 
390   do {
391     range_start = _available_map.get_next_one_offset(range_end);
392     range_end = _available_map.get_next_zero_offset(range_start);
393     candidate = find_contiguous_in_range((uint) range_start, (uint) range_end, num_regions);
394   } while (candidate == G1_NO_HRM_INDEX && range_end < max_length());
395 
396   return candidate;
397 }

399 uint HeapRegionManager::find_contiguous_allow_expand(uint num_regions) {
400   // Find any candidate.
401   return find_contiguous_in_range(0, max_length(), num_regions);
402 }

404 HeapRegion* HeapRegionManager::next_region_in_heap(const HeapRegion* r) const {
405   guarantee(r != NULL, "Start region must be a valid region");
406   guarantee(is_available(r->hrm_index()), "Trying to iterate starting from region %u which is not in the heap", r->hrm_index());
407   for (uint i = r->hrm_index() + 1; i < _allocated_heapregions_length; i++) {
408     HeapRegion* hr = _regions.get_by_index(i);
409     if (is_available(i)) {
410       return hr;
411     }
412   }
413   return NULL;
414 }

416 void HeapRegionManager::iterate(HeapRegionClosure* blk) const {
417   uint len = max_length();
418 
419   for (uint i = 0; i < len; i++) {
420     if (!is_available(i)) {
421       continue;
422     }
423     guarantee(at(i) != NULL, "Tried to access region %u that has a NULL HeapRegion*", i);
424     bool res = blk->do_heap_region(at(i));
425     if (res) {
426       blk->set_incomplete();
427       return;
428     }
429   }
430 }

432 uint HeapRegionManager::find_unavailable_from_idx(uint start_idx, uint* res_idx) const {
433   guarantee(res_idx != NULL, "checking");
434   guarantee(start_idx <= (max_length() + 1), "checking");
435 
436   uint num_regions = 0;
437 
438   uint cur = start_idx;
439   while (cur < max_length() && is_available(cur)) {
440     cur++;
441   }
442   if (cur == max_length()) {
443     return num_regions;
444   }
445   *res_idx = cur;
446   while (cur < max_length() && !is_available(cur)) {
447     cur++;
448   }
449   num_regions = cur - *res_idx;
450 #ifdef ASSERT
451   for (uint i = *res_idx; i < (*res_idx + num_regions); i++) {
452     assert(!is_available(i), "just checking");
453   }
454   assert(cur == max_length() || num_regions == 0 || is_available(cur),
455          "The region at the current position %u must be available or at the end of the heap.", cur);
456 #endif
457   return num_regions;
458 }

460 uint HeapRegionManager::find_highest_free(bool* expanded) {
461   // Loop downwards from the highest region index, looking for an
462   // entry which is either free or not yet committed.  If not yet
463   // committed, expand_at that index.
464   uint curr = max_length() - 1;
465   while (true) {
466     HeapRegion *hr = _regions.get_by_index(curr);
467     if (hr == NULL || !is_available(curr)) {
468       uint res = expand_at(curr, 1, NULL);
469       if (res == 1) {
470         *expanded = true;
471         return curr;
472       }
473     } else {
474       if (hr->is_free()) {
475         *expanded = false;
476         return curr;
477       }
478     }
479     if (curr == 0) {
480       return G1_NO_HRM_INDEX;
481     }
482     curr--;
483   }
484 }

486 bool HeapRegionManager::allocate_containing_regions(MemRegion range, size_t* commit_count, WorkGang* pretouch_workers) {
487   size_t commits = 0;
488   uint start_index = (uint)_regions.get_index_by_address(range.start());
489   uint last_index = (uint)_regions.get_index_by_address(range.last());
490 
491   // Ensure that each G1 region in the range is free, returning false if not.
492   // Commit those that are not yet available, and keep count.
493   for (uint curr_index = start_index; curr_index <= last_index; curr_index++) {
494     if (!is_available(curr_index)) {
495       commits++;
496       expand_at(curr_index, 1, pretouch_workers);
497     }
498     HeapRegion* curr_region  = _regions.get_by_index(curr_index);
499     if (!curr_region->is_free()) {
500       return false;
501     }
502   }
503 
504   allocate_free_regions_starting_at(start_index, (last_index - start_index) + 1);
505   *commit_count = commits;
506   return true;
507 }

509 void HeapRegionManager::par_iterate(HeapRegionClosure* blk, HeapRegionClaimer* hrclaimer, const uint start_index) const {
510   // Every worker will actually look at all regions, skipping over regions that
511   // are currently not committed.
512   // This also (potentially) iterates over regions newly allocated during GC. This
513   // is no problem except for some extra work.
514   const uint n_regions = hrclaimer->n_regions();
515   for (uint count = 0; count < n_regions; count++) {
516     const uint index = (start_index + count) % n_regions;
517     assert(index < n_regions, "sanity");
518     // Skip over unavailable regions
519     if (!is_available(index)) {
520       continue;
521     }
522     HeapRegion* r = _regions.get_by_index(index);
523     // We'll ignore regions already claimed.
524     // However, if the iteration is specified as concurrent, the values for
525     // is_starts_humongous and is_continues_humongous can not be trusted,
526     // and we should just blindly iterate over regions regardless of their
527     // humongous status.
528     if (hrclaimer->is_region_claimed(index)) {
529       continue;
530     }
531     // OK, try to claim it
532     if (!hrclaimer->claim_region(index)) {
533       continue;
534     }
535     bool res = blk->do_heap_region(r);
536     if (res) {
537       return;
538     }
539   }
540 }

542 uint HeapRegionManager::shrink_by(uint num_regions_to_remove) {
543   assert(length() > 0, "the region sequence should not be empty");
544   assert(length() <= _allocated_heapregions_length, "invariant");
545   assert(_allocated_heapregions_length > 0, "we should have at least one region committed");
546   assert(num_regions_to_remove < length(), "We should never remove all regions");
547 
548   if (num_regions_to_remove == 0) {
549     return 0;
550   }
551 
552   uint removed = 0;
553   uint cur = _allocated_heapregions_length - 1;
554   uint idx_last_found = 0;
555   uint num_last_found = 0;
556 
557   while ((removed < num_regions_to_remove) &&
558       (num_last_found = find_empty_from_idx_reverse(cur, &idx_last_found)) > 0) {
559     uint to_remove = MIN2(num_regions_to_remove - removed, num_last_found);
560 
561     shrink_at(idx_last_found + num_last_found - to_remove, to_remove);
562 
563     cur = idx_last_found;
564     removed += to_remove;
565   }
566 
567   verify_optional();
568 
569   return removed;
570 }

572 void HeapRegionManager::shrink_at(uint index, size_t num_regions) {
573 #ifdef ASSERT
574   for (uint i = index; i < (index + num_regions); i++) {
575     assert(is_available(i), "Expected available region at index %u", i);
576     assert(at(i)->is_empty(), "Expected empty region at index %u", i);
577     assert(at(i)->is_free(), "Expected free region at index %u", i);
578   }
579 #endif
580   uncommit_regions(index, num_regions);
581 }

583 uint HeapRegionManager::find_empty_from_idx_reverse(uint start_idx, uint* res_idx) const {
584   guarantee(start_idx < _allocated_heapregions_length, "checking");
585   guarantee(res_idx != NULL, "checking");
586 
587   uint num_regions_found = 0;
588 
589   jlong cur = start_idx;
590   while (cur != -1 && !(is_available(cur) && at(cur)->is_empty())) {
591     cur--;
592   }
593   if (cur == -1) {
594     return num_regions_found;
595   }
596   jlong old_cur = cur;
597   // cur indexes the first empty region
598   while (cur != -1 && is_available(cur) && at(cur)->is_empty()) {
599     cur--;
600   }
601   *res_idx = cur + 1;
602   num_regions_found = old_cur - cur;
603 
604 #ifdef ASSERT
605   for (uint i = *res_idx; i < (*res_idx + num_regions_found); i++) {
606     assert(at(i)->is_empty(), "just checking");
607   }
608 #endif
609   return num_regions_found;
610 }
611 
612 void HeapRegionManager::verify() {
613   guarantee(length() <= _allocated_heapregions_length,
614             "invariant: _length: %u _allocated_length: %u",
615             length(), _allocated_heapregions_length);
616   guarantee(_allocated_heapregions_length <= max_length(),
617             "invariant: _allocated_length: %u _max_length: %u",
618             _allocated_heapregions_length, max_length());
619 
620   bool prev_committed = true;
621   uint num_committed = 0;
622   HeapWord* prev_end = heap_bottom();
623   for (uint i = 0; i < _allocated_heapregions_length; i++) {
624     if (!is_available(i)) {
625       prev_committed = false;
626       continue;
627     }
628     num_committed++;
629     HeapRegion* hr = _regions.get_by_index(i);
630     guarantee(hr != NULL, "invariant: i: %u", i);
631     guarantee(!prev_committed || hr->bottom() == prev_end,
632               "invariant i: %u " HR_FORMAT " prev_end: " PTR_FORMAT,
633               i, HR_FORMAT_PARAMS(hr), p2i(prev_end));
634     guarantee(hr->hrm_index() == i,
635               "invariant: i: %u hrm_index(): %u", i, hr->hrm_index());
636     // Asserts will fire if i is >= _length
637     HeapWord* addr = hr->bottom();
638     guarantee(addr_to_region(addr) == hr, "sanity");
639     // We cannot check whether the region is part of a particular set: at the time
640     // this method may be called, we have only completed allocation of the regions,
641     // but not put into a region set.
642     prev_committed = true;
643     prev_end = hr->end();
644   }
645   for (uint i = _allocated_heapregions_length; i < max_length(); i++) {
646     guarantee(_regions.get_by_index(i) == NULL, "invariant i: %u", i);
647   }
648 
649   guarantee(num_committed == _num_committed, "Found %u committed regions, but should be %u", num_committed, _num_committed);
650   _free_list.verify();
651 }

653 #ifndef PRODUCT
654 void HeapRegionManager::verify_optional() {
655   verify();
656 }

687 class G1RebuildFreeListTask : public AbstractGangTask {
688   HeapRegionManager* _hrm;
689   FreeRegionList*    _worker_freelists;
690   uint               _worker_chunk_size;
691   uint               _num_workers;
692 
693 public:
694   G1RebuildFreeListTask(HeapRegionManager* hrm, uint num_workers) :
695       AbstractGangTask("G1 Rebuild Free List Task"),
696       _hrm(hrm),
697       _worker_freelists(NEW_C_HEAP_ARRAY(FreeRegionList, num_workers, mtGC)),
698       _worker_chunk_size((_hrm->max_length() + num_workers - 1) / num_workers),
699       _num_workers(num_workers) {
700     for (uint worker = 0; worker < _num_workers; worker++) {
701       ::new (&_worker_freelists[worker]) FreeRegionList("Appendable Worker Free List");
702     }
703   }
704 
705   ~G1RebuildFreeListTask() {
706     for (uint worker = 0; worker < _num_workers; worker++) {
707       _worker_freelists[worker].~FreeRegionList();
708     }
709     FREE_C_HEAP_ARRAY(FreeRegionList, _worker_freelists);
710   }
711 
712   FreeRegionList* worker_freelist(uint worker) {
713     return &_worker_freelists[worker];
714   }
715 
716   // Each worker creates a free list for a chunk of the heap. The chunks won't
717   // be overlapping so we don't need to do any claiming.
718   void work(uint worker_id) {
719     Ticks start_time = Ticks::now();
720     EventGCPhaseParallel event;
721 
722     uint start = worker_id * _worker_chunk_size;
723     uint end = MIN2(start + _worker_chunk_size, _hrm->max_length());
724 
725     // If start is outside the heap, this worker has nothing to do.
726     if (start > end) {
727       return;
728     }
729 
730     FreeRegionList *free_list = worker_freelist(worker_id);
731     for (uint i = start; i < end; i++) {
732       HeapRegion *region = _hrm->at_or_null(i);
733       if (region != NULL && region->is_free()) {
734         // Need to clear old links to allow to be added to new freelist.
735         region->unlink_from_list();
736         free_list->add_to_tail(region);
737       }
738     }
739 
740     event.commit(GCId::current(), worker_id, G1GCPhaseTimes::phase_name(G1GCPhaseTimes::RebuildFreeList));
741     G1CollectedHeap::heap()->phase_times()->record_time_secs(G1GCPhaseTimes::RebuildFreeList, worker_id, (Ticks::now() - start_time).seconds());
742   }
743 };
744 
745 void HeapRegionManager::rebuild_free_list(WorkGang* workers) {
746   // Abandon current free list to allow a rebuild.
747   _free_list.abandon();
748 
749   uint const num_workers = clamp(max_length(), 1u, workers->active_workers());
750   G1RebuildFreeListTask task(this, num_workers);
751 
752   log_debug(gc, ergo)("Running %s using %u workers for rebuilding free list of regions",
753                       task.name(), num_workers);
754   workers->run_task(&task, num_workers);
755 
756   // Link the partial free lists together.
757   Ticks serial_time = Ticks::now();
758   for (uint worker = 0; worker < num_workers; worker++) {
759     _free_list.append_ordered(task.worker_freelist(worker));
760   }
761   G1CollectedHeap::heap()->phase_times()->record_serial_rebuild_freelist_time_ms((Ticks::now() - serial_time).seconds() * 1000.0);
762 }
