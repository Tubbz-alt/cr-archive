104 $(GENSRC_DIR)/_gensrc_proc_done: $(PROC_SRCS) $(PROCESSOR_JARS)
105 	$(call MakeDir, $(@D))
106 	$(eval $(call ListPathsSafely,PROC_SRCS,$(@D)/_gensrc_proc_files))
107 	$(JAVA) $(NEW_JAVAC) \
108 	    -XDignore.symbol.file \
109 	    --upgrade-module-path $(JDK_OUTPUTDIR)/modules --system none \
110 	    $(ADD_EXPORTS) \
111 	    -sourcepath $(SOURCEPATH) \
112 	    -implicit:none \
113 	    -proc:only \
114 	    -processorpath $(PROCESSOR_PATH) \
115 	    -d $(GENSRC_DIR) \
116 	    -s $(GENSRC_DIR) \
117 	    @$(@D)/_gensrc_proc_files
118 	$(TOUCH) $@
119 
120 TARGETS += $(GENSRC_DIR)/_gensrc_proc_done

124 $(GENSRC_DIR)/module-info.java.extra: $(GENSRC_DIR)/_gensrc_proc_done
125 	$(ECHO) "" > $@;
126 	($(CD) $(GENSRC_DIR)/META-INF/providers && \
127 	    p=""; \
128 	    impl=""; \
129 	    for i in $$($(GREP) '^' * | $(SORT) -t ':' -k 2 | $(SED) 's/:.*//'); do \
130 	      c=$$($(CAT) $$i | $(TR) -d '\n\r'); \
131 	      if test x$$p != x$$c; then \
132                 if test x$$p != x; then \
133 	          $(ECHO) "    ;" >> $@; \
134 	        fi; \
135 	        $(ECHO) "provides $$c with" >> $@; \
136                 p=$$c; \
137 	        impl=""; \
138 	      fi; \
139               if test x$$impl != x; then \
140 	        $(ECHO) "  , $$i" >> $@; \
141               else \
142 	        $(ECHO) "    $$i" >> $@; \
143               fi; \
144               impl=$$i; \
145 	    done); \
146             $(ECHO) "    ;" >> $@; \
147 	$(ECHO) "uses org.graalvm.compiler.options.OptionDescriptors;" >> $@; \
148 	$(ECHO) "provides org.graalvm.compiler.options.OptionDescriptors with" >> $@; \
149         impl=""; \
150 	for i in $$($(FIND) $(GENSRC_DIR) -name '*_OptionDescriptors.java' | $(SORT)); do \
151 	    c=$$($(ECHO) $$i | $(SED) 's:.*/jdk\.internal\.vm\.compiler/\(.*\)\.java:\1:' | $(TR) '/' '.'); \
152             if test x$$impl != x; then \
153 	      $(ECHO) "  , $$c" >> $@; \
154             else \
155 	      $(ECHO) "    $$c" >> $@; \
156             fi; \
157             impl=$$c; \
158 	done; \
159 	$(ECHO) "    ;" >> $@;
