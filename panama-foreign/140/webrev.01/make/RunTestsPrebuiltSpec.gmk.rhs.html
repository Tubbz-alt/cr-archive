 33 define VerifyVariable
 34   ifeq ($$($1), )
 35     $$(info Error: Variable $1 is missing, needed by RunTestPrebuiltSpec.gmk)

 37   else
 38     ifneq ($$(findstring $$(LOG_LEVEL), debug trace), )

 40     endif
 41   endif
