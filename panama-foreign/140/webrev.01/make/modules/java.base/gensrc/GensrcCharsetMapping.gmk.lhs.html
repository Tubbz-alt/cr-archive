28 # Generate StandardCharsets.java and individul sun.nio.cs charset class using
29 # the charsetmapping tool
30 #
31 CHARSET_DATA_DIR := $(TOPDIR)/make/data/charsetmapping
32 CHARSET_EXTSRC_DIR := $(TOPDIR)/src/jdk.charsets/share/classes/sun/nio/cs/ext
33 CHARSET_GENSRC_JAVA_DIR_BASE := $(SUPPORT_OUTPUTDIR)/gensrc/java.base/sun/nio/cs
34 CHARSET_DONE_BASE := $(CHARSET_GENSRC_JAVA_DIR_BASE)/_the.charsetmapping
35 CHARSET_COPYRIGHT_HEADER := $(TOPDIR)/make/jdk/src/classes/build/tools/charsetmapping
36 CHARSET_TEMPLATES := \
37     $(CHARSET_DATA_DIR)/SingleByte-X.java.template \
38     $(CHARSET_DATA_DIR)/DoubleByte-X.java.template
39 CHARSET_STANDARD_JAVA_TEMPLATES := \
40     $(TOPDIR)/src/java.base/share/classes/sun/nio/cs/StandardCharsets.java.template
41 CHARSET_STANDARD_OS := stdcs-$(OPENJDK_TARGET_OS)
42 
43 $(CHARSET_DONE_BASE)-stdcs: $(CHARSET_DATA_DIR)/charsets \
44     $(wildcard $(CHARSET_DATA_DIR)/$(CHARSET_STANDARD_OS)) \
45     $(CHARSET_TEMPLATES) $(CHARSET_STANDARD_JAVA_TEMPLATES) \
46     $(BUILD_TOOLS_JDK)
47 	$(call LogInfo, Generating java.base charset mapping)
48 	$(call MakeDir, $(@D))
49 	$(TOOL_CHARSETMAPPING) $(CHARSET_DATA_DIR) $(CHARSET_GENSRC_JAVA_DIR_BASE) \
50 	    stdcs charsets $(CHARSET_STANDARD_OS) \
51 	    $(CHARSET_STANDARD_JAVA_TEMPLATES) $(CHARSET_EXTSRC_DIR) \
52 	    $(CHARSET_COPYRIGHT_HEADER) \
53             $(LOG_DEBUG)
54 	$(TOUCH) '$@'
55 
56 GENSRC_JAVA_BASE += $(CHARSET_DONE_BASE)-stdcs
