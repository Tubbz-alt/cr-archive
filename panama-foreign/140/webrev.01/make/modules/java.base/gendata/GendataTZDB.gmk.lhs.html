26 GENDATA_TZDB :=

28 #
29 # Time zone data file creation
30 #
31 TZDATA_DIR := $(TOPDIR)/make/data/tzdata
32 TZDATA_TZFILE := africa antarctica asia australasia europe northamerica pacificnew southamerica backward etcetera gmt jdk11_backward
33 TZDATA_TZFILES := $(addprefix $(TZDATA_DIR)/,$(TZDATA_TZFILE))
34 
35 GENDATA_TZDB_DAT := $(SUPPORT_OUTPUTDIR)/modules_libs/$(MODULE)/tzdb.dat
36 
37 $(GENDATA_TZDB_DAT): $(TZDATA_TZFILES)
38 	$(call MakeTargetDir)
39 	$(RM) $(GENDATA_TZDB_DAT)
40 	$(TOOL_TZDB) -srcdir $(TZDATA_DIR) -dstfile $(GENDATA_TZDB_DAT) $(TZDATA_TZFILE)
41 
42 TARGETS += $(GENDATA_TZDB_DAT)
