 48 /**
 49  * A helper class to generate header interface class in source form.
 50  * After aggregating various constituents of a .java source, build
 51  * method is called to get overall generated source string.
 52  */
 53 class JavaSourceBuilder {
 54     private static final String ABI = InternalForeign.getInstancePrivileged().getSystemABI().name();
 55     // buffer
 56     protected StringBuffer sb;
 57     // current line alignment (number of 4-spaces)
 58     protected int align;
 59 
 60     JavaSourceBuilder(int align) {
 61         this.align = align;
 62         this.sb = new StringBuffer();
 63     }
 64 
 65     JavaSourceBuilder() {
 66         this(0);
 67     }
 68 
 69     protected int align() {
 70         return align;
 71     }

 73     final String PUB_CLS_MODS = "public final ";
 74     final String PUB_MODS = "public static final ";
 75     final String PRI_MODS = "private static final ";
 76 
 77     protected void addPackagePrefix(String pkgName) {
 78         assert pkgName.indexOf('/') == -1 : "package name invalid: " + pkgName;
 79         sb.append("// Generated by jextract\n\n");
 80         if (!pkgName.isEmpty()) {
 81             sb.append("package ");
 82             sb.append(pkgName);
 83             sb.append(";\n\n");
 84         }
 85         addImportSection();
 86     }
 87 
 88     protected void addImportSection() {
 89         sb.append("import java.lang.invoke.MethodHandle;\n");
 90         sb.append("import java.lang.invoke.VarHandle;\n");
 91         sb.append("import jdk.incubator.foreign.*;\n");
 92         sb.append("import jdk.incubator.foreign.MemoryLayout.PathElement;\n");
 93         sb.append("import static ");
 94         sb.append(HandleSourceFactory.C_LANG_CONSTANTS_HOLDER);
 95         sb.append(".*;\n");
 96     }

 98     protected void classBegin(String name) {
 99         indent();
100         sb.append(PUB_CLS_MODS + "class ");
101         sb.append(name);
102         sb.append(" {\n\n");
103     }

105     protected void classEnd() {
106         indent();
107         sb.append("}\n\n");
108     }

110     protected void addLibraries(String[] libraryNames) {
111         incrAlign();
112         indent();
113         sb.append(PRI_MODS + "LibraryLookup[] LIBRARIES = RuntimeHelper.libraries(");
114         sb.append(stringArray(libraryNames) + ");\n");
115         decrAlign();
116     }

118     private String stringArray(String[] elements) {
119         return Stream.of(elements)
120                 .map(n -> "\"" + n + "\"")
121                 .collect(Collectors.joining(",", "new String[] {", "}"));
122     }

124     protected void addLayout(String elementName, MemoryLayout layout) {
125         incrAlign();
126         indent();
127         sb.append(PUB_MODS + "MemoryLayout " + javaSafeIdentifier(elementName) + "$LAYOUT = ");
128         addLayout(layout);
129         sb.append(";\n");
130         decrAlign();
131     }
132 
133     private void addLayout(MemoryLayout l) {
134         if (l instanceof ValueLayout) {
135             SystemABI.Type type = l.abiType().orElseThrow(()->new AssertionError("Should not get here: " + l));
136             sb.append(switch (type) {
137                 case BOOL -> "C_BOOL";
138                 case SIGNED_CHAR -> "C_SCHAR";
139                 case UNSIGNED_CHAR -> "C_UCHAR";
140                 case CHAR -> "C_CHAR";
141                 case SHORT -> "C_SHORT";
142                 case UNSIGNED_SHORT -> "C_USHORT";
143                 case INT -> "C_INT";
144                 case UNSIGNED_INT -> "C_UINT";
145                 case LONG -> "C_LONG";
146                 case UNSIGNED_LONG -> "C_ULONG";
147                 case LONG_LONG -> "C_LONGLONG";
148                 case UNSIGNED_LONG_LONG -> "C_ULONGLONG";
149                 case FLOAT -> "C_FLOAT";
150                 case DOUBLE -> "C_DOUBLE";
151                 case LONG_DOUBLE -> "C_LONGDOUBLE";
152                 case POINTER -> "C_POINTER";
153                 default -> { throw new RuntimeException("should not reach here: " + type); }
154             });
155         } else if (l instanceof SequenceLayout) {
156             sb.append("MemoryLayout.ofSequence(");
157             if (((SequenceLayout) l).elementCount().isPresent()) {
158                 sb.append(((SequenceLayout) l).elementCount().getAsLong() + ", ");
159             }
160             addLayout(((SequenceLayout) l).elementLayout());
161             sb.append(")");
162         } else if (l instanceof GroupLayout) {
163             SystemABI.Type type = l.abiType().orElse(null);
164             if (type == SystemABI.Type.COMPLEX_LONG_DOUBLE) {
165                 if (!ABI.equals(SystemABI.ABI_SYSV)) {
166                     throw new RuntimeException("complex long double is supported only for SysV ABI");
167                 } else {
168                     sb.append("C_COMPLEX_LONGDOUBLE");
169                 }
170             } else {
171                 if (((GroupLayout) l).isStruct()) {
172                     sb.append("MemoryLayout.ofStruct(\n");
173                 } else {
174                     sb.append("MemoryLayout.ofUnion(\n");
175                 }
176                 incrAlign();
177                 String delim = "";
178                 for (MemoryLayout e : ((GroupLayout) l).memberLayouts()) {
179                     sb.append(delim);
180                     indent();
181                     addLayout(e);
182                     delim = ",\n";
183                 }
184                 sb.append("\n");
185                 decrAlign();
186                 indent();
187                 sb.append(")");
188             }
189         } else {
190             //padding
191             sb.append("MemoryLayout.ofPaddingBits(" + l.bitSize() + ")");
192         }
193         if (l.name().isPresent()) {
194             sb.append(".withName(\"" +  l.name().get() + "\")");
195         }
196     }

198     protected void addVarHandle(String name, Class<?> type, String parentName) {
199         incrAlign();
200         indent();
201         parentName = parentName != null? javaSafeIdentifier(parentName) : parentName;
202         name = javaSafeIdentifier(name);
203         String vhName = parentName != null ?
204                 parentName + "$" + name : name;
205         sb.append(PUB_MODS + "VarHandle " + vhName + " = ");
206         if (parentName != null) {
207             addHandlePath(type, parentName, name);
208         } else {
209             addHandlePath(type, name);
210         }
211         sb.append(";\n");
212         decrAlign();
213     }
214 
215     protected void addHandlePath(Class<?> type, String strName, String fieldName) {
216         String ty = type.getName();
217         if (ty.contains("MemoryAddress")) {
218             ty = "long";
219         }
220         sb.append(strName + "$LAYOUT.varHandle(" + ty + ".class, ");
221         sb.append("PathElement.groupElement(\"" + fieldName +"\")");
222         sb.append(")");
223     }
224 
225     protected void addHandlePath(Class<?> type, String varName) {
226         String ty = type.getName();
227         if (ty.contains("MemoryAddress")) {
228             ty = "long";
229         }
230         sb.append(varName + "$LAYOUT.varHandle(" + ty + ".class)");
231     }

233     protected void addMethodHandle(Declaration.Function funcTree, MethodType mtype, FunctionDescriptor desc) {
234         incrAlign();
235         indent();
236         sb.append(PUB_MODS + "MethodHandle " + javaSafeIdentifier(funcTree.name()) + " = ");
237         sb.append("RuntimeHelper.downcallHandle(\n");
238         incrAlign();
239         indent();
240         sb.append("LIBRARIES, \"" + funcTree.name() + "\"");
241         sb.append(",\n");
242         indent();
243         sb.append("\"" + mtype.toMethodDescriptorString() + "\",\n");
244         indent();
245         addFunction(desc);
246         sb.append(",\n");
247         indent();
248         sb.append(funcTree.type().varargs());
249         decrAlign();
250         sb.append("\n");
251         indent();
252         sb.append(");\n");
253         decrAlign();
254     }

256     protected void addAddressLookup(String name) {
257         sb.append("RuntimeHelper.lookupGlobalVariable(LIBRARIES, \"" + name + "\")");
258     }

260     private void addFunction(FunctionDescriptor f) {
261         final boolean noArgs = f.argumentLayouts().isEmpty();
262         if (f.returnLayout().isPresent()) {
263             sb.append("FunctionDescriptor.of(");
264             addLayout(f.returnLayout().get());
265             if (!noArgs) {
266                 sb.append(", ");
267             }
268         } else {
269             sb.append("FunctionDescriptor.ofVoid(");
270         }
271         if (!noArgs) {
272             sb.append("\n");
273             incrAlign();
274             String delim = "";
275             for (MemoryLayout e : f.argumentLayouts()) {
276                 sb.append(delim);
277                 indent();
278                 addLayout(e);
279                 delim = ",\n";
280             }
281             sb.append("\n");
282             decrAlign();
283             indent();
284         }
285         sb.append(")");
286     }

288     protected void addAddress(String name) {
289         incrAlign();
290         indent();
291         sb.append(PUB_MODS + "MemoryAddress " + javaSafeIdentifier(name) + "$ADDR" + " = ");
292         addAddressLookup(name);
293         sb.append(";\n");
294         decrAlign();

297     protected void addConstant(String name, Class<?> type, Object value) {
298         incrAlign();
299         indent();
300         if (type == MemoryAddress.class || type == MemorySegment.class) {
301             //todo, skip for now (address constants and string constants)
302         } else {
303             sb.append(PUB_MODS + type.getName() + " " + javaSafeIdentifier(name));
304             sb.append(" = ");
305             if (type == float.class) {
306                 float f = ((Number)value).floatValue();
307                 if (Float.isNaN(f)) {
308                     sb.append("Float.NaN");
309                 } else if (Float.isInfinite(f)) {
310                     sb.append(f > 0? "Float.POSITIVE_INFINITY" : "Float.NEGATIVE_INFINITY");
311                 } else {
312                     sb.append(value);
313                     sb.append("f");
314                 }
315             } else if (type == long.class) {
316                 sb.append(value);
317                 sb.append("L");
318             } else if (type == double.class) {
319                 double d = ((Number)value).doubleValue();
320                 if (Double.isNaN(d)) {
321                     sb.append("Double.NaN");
322                 } else if (Double.isInfinite(d)) {
323                     sb.append(d > 0? "Double.POSITIVE_INFINITY" : "Double.NEGATIVE_INFINITY");
324                 } else {
325                    sb.append(value);
326                    sb.append("d");
327                 }
328             } else {
329                 sb.append("(" + type.getName() + ")");
330                 sb.append(value + "L");
331             }
332             sb.append(";\n");
333         }
334 
335         decrAlign();
336     }

338     static int funcIntfCounter = 0;

340     protected void addUpcallFactory(FunctionDescriptor desc) {
341         String fnName = "FI" + funcIntfCounter++;
342         incrAlign();
343         indent();
344         sb.append(PRI_MODS + "FunctionDescriptor " + fnName + "$DESC = ");
345         addFunction(desc);
346         sb.append(";\n");
347         indent();
348         sb.append(PUB_MODS + "MemoryAddress " + fnName + "$make(MethodHandle handle) {\n");
349         incrAlign();
350         indent();
351         sb.append("return RuntimeHelper.upcallStub(handle, " + fnName + "$DESC);\n");
352         decrAlign();
353         indent();
354         sb.append("}\n");
355         decrAlign();
356     }

358     protected void addStaticFunctionWrapper(Declaration.Function f, MethodType mtype) {
359         incrAlign();
360         indent();
361         sb.append(PUB_MODS + mtype.returnType().getName() + " " + javaSafeIdentifier(f.name()) + " (");
362         String delim = "";
363         List<String> pNames = new ArrayList<>();
364         final int numParams = f.parameters().size();
365         for (int i = 0 ; i < numParams; i++) {
366             String pName = f.parameters().get(i).name();
367             if (pName.isEmpty()) {
368                 pName = "x" + i;
369             }
370             pName = javaSafeIdentifier(pName);
371             pNames.add(pName);
372             sb.append(delim + mtype.parameterType(i).getName() + " " + pName);
373             delim = ", ";
374         }
375         if (f.type().varargs()) {
376             String lastArg = "x" + numParams;
377             if (numParams > 0) {
378                 sb.append(", ");
379             }
380             sb.append("Object... " + lastArg);
381             pNames.add(lastArg);
382         }
383         sb.append(") {\n");
384         incrAlign();
385         indent();
386         sb.append("try {\n");
387         incrAlign();
388         indent();
389         if (!mtype.returnType().equals(void.class)) {
390             sb.append("return (" + mtype.returnType().getName() + ")");
391         }
392         sb.append(f.name() + ".invokeExact(" + String.join(", ", pNames) + ");\n");
393         decrAlign();
394         indent();
395         sb.append("} catch (Throwable ex) {\n");
396         incrAlign();
397         indent();
398         sb.append("throw new AssertionError(ex);\n");
399         decrAlign();
400         indent();
401         sb.append("}\n");
402         decrAlign();
403         indent();
404         sb.append("}\n");
405         decrAlign();
406     }

408     void addDescriptor(String name, FunctionDescriptor desc) {
409         incrAlign();
410         indent();
411         sb.append(PRI_MODS + "FunctionDescriptor " + name + "$DESC = ");
412         addFunction(desc);

414         decrAlign();

418     void addFunctionalInterface(String name, MethodType mtype) {
419         incrAlign();
420         indent();
421         sb.append("public interface " + javaSafeIdentifier(name) + " {\n");
422         incrAlign();
423         indent();
424         sb.append(mtype.returnType().getName() + " apply(");
425         String delim = "";
426         for (int i = 0 ; i < mtype.parameterCount() ; i++) {
427             sb.append(delim + mtype.parameterType(i).getName() + " x" + i);
428             delim = ", ";
429         }
430         sb.append(");\n");
431         decrAlign();
432         indent();
433         sb.append("}\n");
434         decrAlign();
435         indent();
436     }

438     protected void addFunctionalFactory(String name, MethodType mtype) {
439         incrAlign();
440         indent();
441         sb.append(PUB_MODS + "MemoryAddress " + name + "$make(" + name + " fi) {\n");
442         incrAlign();
443         indent();
444         sb.append("return RuntimeHelper.upcallStub(" + name + ".class, fi, " + name + "$DESC, " +
445                 "\"" + mtype.toMethodDescriptorString() + "\");\n");
446         decrAlign();
447         indent();
448         sb.append("}\n");
449         decrAlign();

452     void addGetter(String name, Class<?> type, String parent) {
453         incrAlign();
454         indent();
455         name = javaSafeIdentifier(name);
456         String vhName = (parent != null ? (javaSafeIdentifier(parent) + "$") : "") + name;
457         String param = parent != null ? (MemorySegment.class.getName() + " seg") : "";
458         sb.append(PUB_MODS + type.getName() + " " + vhName + "$get(" + param + ") {\n");
459         incrAlign();
460         indent();
461         String vhParam = parent != null ?
462                 "seg.baseAddress()" : name + "$ADDR";
463         sb.append("return (" + type.getName() + ")" + vhName + ".get(" + vhParam + ");\n");
464         decrAlign();
465         indent();
466         sb.append("}\n");
467         decrAlign();
468     }

470     void addSetter(String name, Class<?> type, String parent) {
471         incrAlign();
472         indent();
473         name = javaSafeIdentifier(name);
474         String vhName = (parent != null ? (javaSafeIdentifier(parent) + "$") : "") + name;
475         String param = parent != null ? (MemorySegment.class.getName() + " seg, ") : "";
476         sb.append(PUB_MODS + "void " + vhName + "$set(" + param + type.getName() + " x) {\n");
477         incrAlign();
478         indent();
479         String vhParam = parent != null ?
480                 "seg.baseAddress()" : name + "$ADDR";
481         sb.append(vhName + ".set(" + vhParam + ", x);\n");
482         decrAlign();
483         indent();
484         sb.append("}\n");
485         decrAlign();
486     }

488     protected String build() {
489         String res = sb.toString();
490         this.sb = null;
491         return res.toString();
492     }
493 
494     protected void indent() {
495         for (int i = 0; i < align; i++) {
496             sb.append("    ");
497         }
498     }
499 
500     protected void incrAlign() {
501         align++;
502     }
503 
504     protected void decrAlign() {
505         align--;
506     }
507 
508     protected final String javaSafeIdentifier(String name) {
509         // We never get the problem of Java non-identifiers (like 123, ab-xy) as
510         // C identifiers. But we may have a java keyword used as a C identifier.
511         assert SourceVersion.isIdentifier(name);
512 
513         return SourceVersion.isKeyword(name)? (name + "_") : name;
514     }
