 25 package jdk.incubator.jextract.tool;

 27 import jdk.incubator.foreign.FunctionDescriptor;
 28 import jdk.incubator.foreign.MemoryLayout;
 29 import jdk.incubator.foreign.MemorySegment;

 31 import javax.tools.JavaFileObject;
 32 import java.lang.constant.ClassDesc;
 33 import java.lang.constant.DirectMethodHandleDesc;
 34 import java.lang.invoke.MethodType;
 35 import java.util.ArrayList;
 36 import java.util.List;

 38 /**
 39  * A helper class to generate header interface class in source form.
 40  * After aggregating various constituents of a .java source, build
 41  * method is called to get overall generated source string.
 42  */
 43 class JavaSourceBuilder {




 44     private final String pkgName;
 45     private final String[] libraryNames;
 46     // buffer
 47     protected StringBuffer sb;
 48     // current line alignment (number of 4-spaces)
 49     protected int align;
 50 
 51     private String className = null;
 52     private ConstantHelper constantHelper;
 53 
 54     JavaSourceBuilder(int align, String pkgName, String[] libraryNames) {
 55         this.align = align;
 56         this.libraryNames = libraryNames;
 57         this.sb = new StringBuffer();
 58         this.pkgName = pkgName;
 59     }
 60 
 61     JavaSourceBuilder(String pkgName, String[] libraryNames) {
 62         this(0, pkgName, libraryNames);
 63     }
 64 
 65     final String PUB_CLS_MODS = "public final ";
 66     final String PUB_MODS = "public static final ";
 67 
 68     private void addPackagePrefix() {
 69         assert pkgName.indexOf('/') == -1 : "package name invalid: " + pkgName;
 70         sb.append("// Generated by jextract\n\n");
 71         if (!pkgName.isEmpty()) {
 72             sb.append("package ");
 73             sb.append(pkgName);
 74             sb.append(";\n\n");
 75         }
 76     }
 77 
 78     private void addImportSection() {
 79         sb.append("import java.lang.invoke.MethodHandle;\n");
 80         sb.append("import java.lang.invoke.VarHandle;\n");
 81         sb.append("import jdk.incubator.foreign.*;\n");
 82         sb.append("import jdk.incubator.foreign.MemoryLayout.PathElement;\n");
 83         sb.append("import static ");
 84         sb.append(OutputFactory.C_LANG_CONSTANTS_HOLDER);
 85         sb.append(".*;\n");
 86     }
 87 
 88     public void classBegin(String name) {
 89         className = name;
 90         String qualName = pkgName.isEmpty() ? name : pkgName + "." + name;
 91         constantHelper = new ConstantHelper(qualName, ClassDesc.of(pkgName, "RuntimeHelper"), libraryNames);

 92 
 93         addPackagePrefix();
 94         addImportSection();
 95 
 96         indent();
 97         sb.append(PUB_CLS_MODS + "class ");
 98         sb.append(name);
 99         sb.append(" {\n\n");
100     }
101 
102     public void classEnd() {
103         indent();
104         sb.append("}\n\n");
105     }
106 
107     private String getterCall(DirectMethodHandleDesc desc) {
108         return desc.owner().displayName() + "." + desc.methodName() + "()";
109     }
110 
111     private void emitForwardGetter(DirectMethodHandleDesc desc) {
112         incrAlign();
113         indent();
114         sb.append(PUB_MODS + displayName(desc.invocationType().returnType()) + " " + desc.methodName() + "() {\n");
115         incrAlign();
116         indent();
117         sb.append("return " + getterCall(desc) + ";\n");
118         decrAlign();
119         indent();
120         sb.append("}\n");
121         decrAlign();
122     }
123 
124     private String displayName(ClassDesc returnType) {
125         return returnType.displayName(); // TODO shorten based on imports
126     }
127 
128     private String getFunction(String javaName, FunctionDescriptor fDesc) {
129         return getterCall(constantHelper.addFunctionDesc(javaName, fDesc));
130     }
131 
132     private String getMethodHandle(String javaName, String nativeName, MethodType mt, FunctionDescriptor fDesc, boolean varargs) {
133         return getterCall(constantHelper.addMethodHandle(javaName, nativeName, mt, fDesc, varargs));
134     }
135 
136     private String getVarHandle(String javaName, String nativeName, MemoryLayout layout, Class<?> type, MemoryLayout parentLayout) {
137         return getterCall(constantHelper.addVarHandle(javaName, nativeName, layout, type, parentLayout));
138     }
139 
140     private String getAddress(String javaName, String nativeName) {
141         return getterCall(constantHelper.addAddress(javaName, nativeName));
142     }
143 
144     public void addLayout(String javaName, MemoryLayout layout) {
145         emitForwardGetter(constantHelper.addLayout(javaName, layout));
146     }
147 
148     public void addVarHandle(String javaName, String nativeName, MemoryLayout layout, Class<?> type, MemoryLayout parentLayout) {
149         emitForwardGetter(constantHelper.addVarHandle(javaName, nativeName, layout, type, parentLayout));
150     }
151 
152     public void addMethodHandle(String javaName, String nativeName, MethodType mtype, FunctionDescriptor desc, boolean varargs) {
153         emitForwardGetter(constantHelper.addMethodHandle(javaName, nativeName, mtype, desc, varargs));
154     }
155 
156     public void addAddress(String javaName, String nativeName) {
157         emitForwardGetter(constantHelper.addAddress(javaName, nativeName));
158     }
159 
160     public void addConstant(String javaName, Class<?> type, Object value) {
161         emitForwardGetter(constantHelper.addConstant(javaName, type, value));
162     }
163 
164     public void addFunctionalFactory(String className, MethodType mtype, FunctionDescriptor fDesc) {
165         incrAlign();
166         indent();
167         sb.append(PUB_MODS + "MemoryAddress " + className + "$make(" + className + " fi) {\n");
168         incrAlign();
169         indent();
170         sb.append("return RuntimeHelper.upcallStub(" + className + ".class, fi, " + getFunction(className, fDesc) + ", " +
171                 "\"" + mtype.toMethodDescriptorString() + "\");\n");
172         decrAlign();
173         indent();
174         sb.append("}\n");
175         decrAlign();
176     }
177 
178 
179     public void addStaticFunctionWrapper(String javaName, String nativeName, MethodType mtype, FunctionDescriptor desc, boolean varargs, List<String> paramNames) {
180         incrAlign();
181         indent();
182         sb.append(PUB_MODS + mtype.returnType().getName() + " " + javaName + " (");
183         String delim = "";
184         List<String> pNames = new ArrayList<>();
185         final int numParams = paramNames.size();
186         for (int i = 0 ; i < numParams; i++) {
187             String pName = paramNames.get(i);
188             if (pName.isEmpty()) {
189                 pName = "x" + i;
190             }
191             pNames.add(pName);
192             sb.append(delim + mtype.parameterType(i).getName() + " " + pName);
193             delim = ", ";
194         }
195         if (varargs) {
196             String lastArg = "x" + numParams;
197             if (numParams > 0) {
198                 sb.append(", ");
199             }
200             sb.append("Object... " + lastArg);
201             pNames.add(lastArg);
202         }
203         sb.append(") {\n");
204         incrAlign();
205         indent();
206         sb.append("try {\n");
207         incrAlign();
208         indent();
209         if (!mtype.returnType().equals(void.class)) {
210             sb.append("return (" + mtype.returnType().getName() + ")");
211         }
212         sb.append(getMethodHandle(javaName, nativeName, mtype, desc, varargs) + ".invokeExact(" + String.join(", ", pNames) + ");\n");
213         decrAlign();
214         indent();
215         sb.append("} catch (Throwable ex) {\n");
216         incrAlign();
217         indent();
218         sb.append("throw new AssertionError(ex);\n");
219         decrAlign();
220         indent();
221         sb.append("}\n");
222         decrAlign();
223         indent();
224         sb.append("}\n");
225         decrAlign();
226     }
227 
228     public void addFunctionalInterface(String name, MethodType mtype) {
229         incrAlign();
230         indent();
231         sb.append("public interface " + name + " {\n");
232         incrAlign();
233         indent();
234         sb.append(mtype.returnType().getName() + " apply(");
235         String delim = "";
236         for (int i = 0 ; i < mtype.parameterCount() ; i++) {
237             sb.append(delim + mtype.parameterType(i).getName() + " x" + i);
238             delim = ", ";
239         }
240         sb.append(");\n");
241         decrAlign();
242         indent();
243         sb.append("}\n");
244         decrAlign();
245         indent();
246     }
247 
248     public void addGetter(String javaName, String nativeName, MemoryLayout layout, Class<?> type, MemoryLayout parentLayout) {
249         incrAlign();
250         indent();
251         String param = parentLayout != null ? (MemorySegment.class.getName() + " seg") : "";
252         sb.append(PUB_MODS + type.getName() + " " + javaName + "$get(" + param + ") {\n");
253         incrAlign();
254         indent();
255         String vhParam = parentLayout != null ?
256                 "seg.baseAddress()" : getAddress(javaName, nativeName);
257         sb.append("return (" + type.getName() + ")"
258                 + getVarHandle(javaName, nativeName, layout, type, parentLayout) + ".get(" + vhParam + ");\n");
259         decrAlign();
260         indent();
261         sb.append("}\n");
262         decrAlign();
263     }
264 
265     public void addSetter(String javaName, String nativeName, MemoryLayout layout, Class<?> type, MemoryLayout parentLayout) {
266         incrAlign();
267         indent();
268         String param = parentLayout != null ? (MemorySegment.class.getName() + " seg, ") : "";
269         sb.append(PUB_MODS + "void " + javaName + "$set(" + param + type.getName() + " x) {\n");
270         incrAlign();
271         indent();
272         String vhParam = parentLayout != null ?
273                 "seg.baseAddress()" : getAddress(javaName, nativeName);
274         sb.append(getVarHandle(javaName, nativeName, layout, type, parentLayout) + ".set(" + vhParam + ", x);\n");
275         decrAlign();
276         indent();
277         sb.append("}\n");
278         decrAlign();
279     }
280 
281     public List<JavaFileObject> build() {
282         String res = sb.toString();
283         this.sb = null;
284         List<JavaFileObject> outputs = new ArrayList<>(constantHelper.getClasses());
285         outputs.add(Utils.fileFromString(pkgName, className, res));
286         return outputs;
287     }
288 



























































289     private void indent() {
290         for (int i = 0; i < align; i++) {
291             sb.append("    ");
292         }
293     }
294 
295     private void incrAlign() {
296         align++;
297     }
298 
299     private void decrAlign() {
300         align--;
301     }
302 
303 }
