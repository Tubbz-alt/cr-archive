25 package jdk.incubator.jextract.tool;
26 
27 import jdk.incubator.foreign.FunctionDescriptor;
28 
29 import javax.lang.model.SourceVersion;
30 import javax.tools.JavaFileObject;
31 import javax.tools.SimpleJavaFileObject;
32 import java.io.ByteArrayInputStream;
33 import java.io.IOException;
34 import java.io.InputStream;
35 import java.net.URI;
36 
37 class Utils {


38 
39     private static URI fileName(String pkgName, String clsName, String extension) {
40         String pkgPrefix = pkgName.isEmpty() ? "" : pkgName.replaceAll("\\.", "/") + "/";
41         return URI.create(pkgPrefix + clsName + extension);
42     }
43 
44     static JavaFileObject fileFromString(String pkgName, String clsName, String contents) {
45         return new SimpleJavaFileObject(fileName(pkgName, clsName, ".java"), JavaFileObject.Kind.SOURCE) {
46             @Override
47             public CharSequence getCharContent(boolean ignoreEncodingErrors) throws IOException {
48                 return contents;
49             }
50         };
51     }
52 
53     static String javaSafeIdentifier(String name) {
54         // We never get the problem of Java non-identifiers (like 123, ab-xy) as
55         // C identifiers. But we may have a java keyword used as a C identifier.
56         assert SourceVersion.isIdentifier(name);
57 
58         return SourceVersion.isKeyword(name)? (name + "_") : name;
59     }
60 }
