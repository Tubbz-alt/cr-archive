 56     public static final MemoryAddress makeCString(String value) {
 57         value += '\0';
 58         MemorySegment dest = MemorySegment.allocateNative(value.length());
 59         // Utils::toJavaString casts bytes to 'char'
 60         //     -> interpreted as UTF-16
 61         //     -> get back as UTF-8 assuming no surrogate pairs
 62         byte[] bytes = value.getBytes(StandardCharsets.UTF_8);
 63         if (bytes.length != value.length()) {
 64             throw new IllegalStateException("bytes.length != value.length(). " +
 65                     "bytes: " + Arrays.toString(bytes) + ", value: " + value);
 66         }
 67         MemorySegment src = MemorySegment.ofArray(bytes);
 68         MemoryAddress.copy(src.baseAddress(), dest.baseAddress(), value.length());
 69         return dest.baseAddress();
 70     }
