117     private String stringArray(String[] elements) {
118         return Stream.of(elements)

132     private void addLayout(MemoryLayout l) {
133         if (l instanceof ValueLayout) {
134             SystemABI.Type type = l.attribute(NATIVE_TYPE)
135                                    .map(SystemABI.Type.class::cast)
136                                    .orElseThrow(()->new AssertionError("Should not get here: " + l));
137             sb.append(TypeTranslator.typeToLayoutName(type));
138         } else if (l instanceof SequenceLayout) {
139             sb.append("MemoryLayout.ofSequence(");
140             if (((SequenceLayout) l).elementCount().isPresent()) {
141                 sb.append(((SequenceLayout) l).elementCount().getAsLong() + ", ");
142             }
143             addLayout(((SequenceLayout) l).elementLayout());
144             sb.append(")");
145         } else if (l instanceof GroupLayout) {
146             SystemABI.Type type = l.attribute(NATIVE_TYPE)
147                                    .map(SystemABI.Type.class::cast)
148                                    .orElse(null);
149             if (type == SystemABI.Type.COMPLEX_LONG_DOUBLE) {
150                 if (!ABI.equals(SystemABI.ABI_SYSV)) {
151                     throw new RuntimeException("complex long double is supported only for SysV ABI");
152                 } else {
153                     sb.append("C_COMPLEX_LONGDOUBLE");
154                 }
155             } else {
156                 if (((GroupLayout) l).isStruct()) {
157                     sb.append("MemoryLayout.ofStruct(\n");
158                 } else {
159                     sb.append("MemoryLayout.ofUnion(\n");
160                 }
161                 incrAlign();
162                 String delim = "";
163                 for (MemoryLayout e : ((GroupLayout) l).memberLayouts()) {
164                     sb.append(delim);
165                     indent();
166                     addLayout(e);
167                     delim = ",\n";
168                 }
169                 sb.append("\n");
170                 decrAlign();
171                 indent();
172                 sb.append(")");
173             }
174         } else {
175             //padding
176             sb.append("MemoryLayout.ofPaddingBits(" + l.bitSize() + ")");
177         }
178         if (l.name().isPresent()) {
179             sb.append(".withName(\"" +  l.name().get() + "\")");
180         }
181     }

184         incrAlign();
185         indent();

189                 parentName + "$" + name : name;
190         sb.append(PUB_MODS + "VarHandle " + vhName + " = ");

255     private void addFunction(FunctionDescriptor f) {
256         final boolean noArgs = f.argumentLayouts().isEmpty();
257         if (f.returnLayout().isPresent()) {
258             sb.append("FunctionDescriptor.of(");
259             addLayout(f.returnLayout().get());
260             if (!noArgs) {
261                 sb.append(", ");
262             }
263         } else {
264             sb.append("FunctionDescriptor.ofVoid(");
265         }
266         if (!noArgs) {
267             sb.append("\n");
268             incrAlign();
269             String delim = "";
270             for (MemoryLayout e : f.argumentLayouts()) {
271                 sb.append(delim);
272                 indent();
273                 addLayout(e);
274                 delim = ",\n";
275             }
276             sb.append("\n");
277             decrAlign();
278             indent();
279         }
280         sb.append(")");
281     }

292     protected void addConstant(String name, Class<?> type, Object value) {
293         incrAlign();
294         indent();
295         if (type == MemoryAddress.class || type == MemorySegment.class) {
296             //todo, skip for now (address constants and string constants)
297         } else {
298             sb.append(PUB_MODS + type.getName() + " " + javaSafeIdentifier(name));
299             sb.append(" = ");
300             if (type == float.class) {
301                 float f = ((Number)value).floatValue();
302                 if (Float.isNaN(f)) {
303                     sb.append("Float.NaN");
304                 } else if (Float.isInfinite(f)) {
305                     sb.append(f > 0? "Float.POSITIVE_INFINITY" : "Float.NEGATIVE_INFINITY");
306                 } else {
307                     sb.append(value);
308                     sb.append("f");
309                 }
310             } else if (type == long.class) {
311                 sb.append(value);
312                 sb.append("L");
313             } else if (type == double.class) {
314                 double d = ((Number)value).doubleValue();
315                 if (Double.isNaN(d)) {
316                     sb.append("Double.NaN");
317                 } else if (Double.isInfinite(d)) {
318                     sb.append(d > 0? "Double.POSITIVE_INFINITY" : "Double.NEGATIVE_INFINITY");
319                 } else {
320                    sb.append(value);
321                    sb.append("d");
322                 }
323             } else {
324                 sb.append("(" + type.getName() + ")");
325                 sb.append(value + "L");
326             }
327             sb.append(";\n");
328         }
329 
330         decrAlign();
331     }

353     protected void addStaticFunctionWrapper(Declaration.Function f, MethodType mtype) {
354         incrAlign();
355         indent();
356         sb.append(PUB_MODS + mtype.returnType().getName() + " " + javaSafeIdentifier(f.name()) + " (");
357         String delim = "";
358         List<String> pNames = new ArrayList<>();
359         final int numParams = f.parameters().size();
360         for (int i = 0 ; i < numParams; i++) {
361             String pName = f.parameters().get(i).name();
362             if (pName.isEmpty()) {
363                 pName = "x" + i;
364             }
365             pName = javaSafeIdentifier(pName);
366             pNames.add(pName);
367             sb.append(delim + mtype.parameterType(i).getName() + " " + pName);
368             delim = ", ";
369         }
370         if (f.type().varargs()) {
371             String lastArg = "x" + numParams;
372             if (numParams > 0) {
373                 sb.append(", ");
374             }
375             sb.append("Object... " + lastArg);
376             pNames.add(lastArg);
377         }
378         sb.append(") {\n");
379         incrAlign();
380         indent();
381         sb.append("try {\n");
382         incrAlign();
383         indent();
384         if (!mtype.returnType().equals(void.class)) {
385             sb.append("return (" + mtype.returnType().getName() + ")");
386         }
387         sb.append(f.name() + ".invokeExact(" + String.join(", ", pNames) + ");\n");
388         decrAlign();
389         indent();
390         sb.append("} catch (Throwable ex) {\n");
391         incrAlign();
392         indent();
393         sb.append("throw new AssertionError(ex);\n");
394         decrAlign();
395         indent();
396         sb.append("}\n");
397         decrAlign();
398         indent();
399         sb.append("}\n");
400         decrAlign();
401     }

503     protected final String javaSafeIdentifier(String name) {
504         // We never get the problem of Java non-identifiers (like 123, ab-xy) as
505         // C identifiers. But we may have a java keyword used as a C identifier.
506         assert SourceVersion.isIdentifier(name);
507 
508         return SourceVersion.isKeyword(name)? (name + "_") : name;

193         incrAlign();
194         indent();
195         String param = parentLayout != null ? (MemorySegment.class.getName() + " seg") : "";
