 47 /**
 48  * A helper class to generate header interface class in source form.
 49  * After aggregating various constituents of a .java source, build
 50  * method is called to get overall generated source string.
 51  */
 52 class JavaSourceBuilder {
 53     private static final String ABI = InternalForeign.getInstancePrivileged().getSystemABI().name();
 54     // buffer
 55     protected StringBuffer sb;
 56     // current line alignment (number of 4-spaces)
 57     protected int align;
 58 
 59     JavaSourceBuilder(int align) {
 60         this.align = align;
 61         this.sb = new StringBuffer();
 62     }
 63 
 64     JavaSourceBuilder() {
 65         this(0);
 66     }
 67 
 68     protected int align() {
 69         return align;
 70     }

 72     final String PUB_CLS_MODS = "public final ";
 73     final String PUB_MODS = "public static final ";
 74     final String PRI_MODS = "private static final ";
 75 
 76     protected void addPackagePrefix(String pkgName) {
 77         assert pkgName.indexOf('/') == -1 : "package name invalid: " + pkgName;
 78         sb.append("// Generated by jextract\n\n");
 79         if (!pkgName.isEmpty()) {
 80             sb.append("package ");
 81             sb.append(pkgName);
 82             sb.append(";\n\n");
 83         }
 84         addImportSection();
 85     }
 86 
 87     protected void addImportSection() {
 88         sb.append("import java.lang.invoke.MethodHandle;\n");
 89         sb.append("import java.lang.invoke.VarHandle;\n");
 90         sb.append("import jdk.incubator.foreign.*;\n");
 91         sb.append("import jdk.incubator.foreign.MemoryLayout.PathElement;\n");
 92         sb.append("import static ");
 93         sb.append(HandleSourceFactory.C_LANG_CONSTANTS_HOLDER);
 94         sb.append(".*;\n");
 95     }

 97     protected void classBegin(String name) {
 98         indent();
 99         sb.append(PUB_CLS_MODS + "class ");
100         sb.append(name);
101         sb.append(" {\n\n");
102     }

104     protected void classEnd() {
105         indent();
106         sb.append("}\n\n");
107     }

109     protected void addLibraries(String[] libraryNames) {
110         incrAlign();
111         indent();
112         sb.append(PRI_MODS + "LibraryLookup[] LIBRARIES = RuntimeHelper.libraries(");
113         sb.append(stringArray(libraryNames) + ");\n");
114         decrAlign();
115     }

117     private String stringArray(String[] elements) {
118         return Stream.of(elements)
119                 .map(n -> "\"" + n + "\"")
120                 .collect(Collectors.joining(",", "new String[] {", "}"));
121     }

123     protected void addLayout(String elementName, MemoryLayout layout) {
124         incrAlign();
125         indent();
126         sb.append(PUB_MODS + "MemoryLayout " + javaSafeIdentifier(elementName) + "$LAYOUT = ");
127         addLayout(layout);
128         sb.append(";\n");
129         decrAlign();
130     }
131 
132     private void addLayout(MemoryLayout l) {
133         if (l instanceof ValueLayout) {
134             SystemABI.Type type = l.attribute(NATIVE_TYPE)
135                                    .map(SystemABI.Type.class::cast)
136                                    .orElseThrow(()->new AssertionError("Should not get here: " + l));
137             sb.append(TypeTranslator.typeToLayoutName(type));
138         } else if (l instanceof SequenceLayout) {
139             sb.append("MemoryLayout.ofSequence(");
140             if (((SequenceLayout) l).elementCount().isPresent()) {
141                 sb.append(((SequenceLayout) l).elementCount().getAsLong() + ", ");
142             }
143             addLayout(((SequenceLayout) l).elementLayout());
144             sb.append(")");
145         } else if (l instanceof GroupLayout) {
146             SystemABI.Type type = l.attribute(NATIVE_TYPE)
147                                    .map(SystemABI.Type.class::cast)
148                                    .orElse(null);
149             if (type == SystemABI.Type.COMPLEX_LONG_DOUBLE) {
150                 if (!ABI.equals(SystemABI.ABI_SYSV)) {
151                     throw new RuntimeException("complex long double is supported only for SysV ABI");
152                 } else {
153                     sb.append("C_COMPLEX_LONGDOUBLE");
154                 }
155             } else {
156                 if (((GroupLayout) l).isStruct()) {
157                     sb.append("MemoryLayout.ofStruct(\n");
158                 } else {
159                     sb.append("MemoryLayout.ofUnion(\n");
160                 }
161                 incrAlign();
162                 String delim = "";
163                 for (MemoryLayout e : ((GroupLayout) l).memberLayouts()) {
164                     sb.append(delim);
165                     indent();
166                     addLayout(e);
167                     delim = ",\n";
168                 }
169                 sb.append("\n");
170                 decrAlign();
171                 indent();
172                 sb.append(")");
173             }
174         } else {
175             //padding
176             sb.append("MemoryLayout.ofPaddingBits(" + l.bitSize() + ")");
177         }
178         if (l.name().isPresent()) {
179             sb.append(".withName(\"" +  l.name().get() + "\")");
180         }
181     }

183     protected void addVarHandle(String name, Class<?> type, String parentName) {
184         incrAlign();
185         indent();
186         parentName = parentName != null? javaSafeIdentifier(parentName) : parentName;
187         name = javaSafeIdentifier(name);
188         String vhName = parentName != null ?
189                 parentName + "$" + name : name;
190         sb.append(PUB_MODS + "VarHandle " + vhName + " = ");
191         if (parentName != null) {
192             addHandlePath(type, parentName, name);
193         } else {
194             addHandlePath(type, name);
195         }
196         sb.append(";\n");
197         decrAlign();
198     }
199 
200     protected void addHandlePath(Class<?> type, String strName, String fieldName) {
201         String ty = type.getName();
202         boolean isAddress = ty.contains("MemoryAddress");
203         if (isAddress) {
204             sb.append("MemoryHandles.asAddressVarHandle(");
205             ty = "long";
206         }
207         sb.append(strName + "$LAYOUT.varHandle(" + ty + ".class, ");
208         sb.append("PathElement.groupElement(\"" + fieldName +"\")");
209         sb.append(")");
210         if (isAddress) {
211             sb.append(")");
212         }
213     }
214 
215     protected void addHandlePath(Class<?> type, String varName) {
216         String ty = type.getName();
217         boolean isAddress = ty.contains("MemoryAddress");
218         if (isAddress) {
219             sb.append("MemoryHandles.asAddressVarHandle(");
220             ty = "long";
221         }
222         sb.append(varName + "$LAYOUT.varHandle(" + ty + ".class)");
223         if (isAddress) {
224             sb.append(")");
225         }
226     }

228     protected void addMethodHandle(Declaration.Function funcTree, MethodType mtype, FunctionDescriptor desc) {
229         incrAlign();
230         indent();
231         sb.append(PUB_MODS + "MethodHandle " + javaSafeIdentifier(funcTree.name()) + " = ");
232         sb.append("RuntimeHelper.downcallHandle(\n");
233         incrAlign();
234         indent();
235         sb.append("LIBRARIES, \"" + funcTree.name() + "\"");
236         sb.append(",\n");
237         indent();
238         sb.append("\"" + mtype.toMethodDescriptorString() + "\",\n");
239         indent();
240         addFunction(desc);
241         sb.append(",\n");
242         indent();
243         sb.append(funcTree.type().varargs());
244         decrAlign();
245         sb.append("\n");
246         indent();
247         sb.append(");\n");
248         decrAlign();
249     }

251     protected void addAddressLookup(String name) {
252         sb.append("RuntimeHelper.lookupGlobalVariable(LIBRARIES, \"" + name + "\")");
253     }

255     private void addFunction(FunctionDescriptor f) {
256         final boolean noArgs = f.argumentLayouts().isEmpty();
257         if (f.returnLayout().isPresent()) {
258             sb.append("FunctionDescriptor.of(");
259             addLayout(f.returnLayout().get());
260             if (!noArgs) {
261                 sb.append(", ");
262             }
263         } else {
264             sb.append("FunctionDescriptor.ofVoid(");
265         }
266         if (!noArgs) {
267             sb.append("\n");
268             incrAlign();
269             String delim = "";
270             for (MemoryLayout e : f.argumentLayouts()) {
271                 sb.append(delim);
272                 indent();
273                 addLayout(e);
274                 delim = ",\n";
275             }
276             sb.append("\n");
277             decrAlign();
278             indent();
279         }
280         sb.append(")");
281     }

283     protected void addAddress(String name) {
284         incrAlign();
285         indent();
286         sb.append(PUB_MODS + "MemoryAddress " + javaSafeIdentifier(name) + "$ADDR" + " = ");
287         addAddressLookup(name);
288         sb.append(";\n");
289         decrAlign();

292     protected void addConstant(String name, Class<?> type, Object value) {
293         incrAlign();
294         indent();
295         if (type == MemoryAddress.class || type == MemorySegment.class) {
296             //todo, skip for now (address constants and string constants)
297         } else {
298             sb.append(PUB_MODS + type.getName() + " " + javaSafeIdentifier(name));
299             sb.append(" = ");
300             if (type == float.class) {
301                 float f = ((Number)value).floatValue();
302                 if (Float.isNaN(f)) {
303                     sb.append("Float.NaN");
304                 } else if (Float.isInfinite(f)) {
305                     sb.append(f > 0? "Float.POSITIVE_INFINITY" : "Float.NEGATIVE_INFINITY");
306                 } else {
307                     sb.append(value);
308                     sb.append("f");
309                 }
310             } else if (type == long.class) {
311                 sb.append(value);
312                 sb.append("L");
313             } else if (type == double.class) {
314                 double d = ((Number)value).doubleValue();
315                 if (Double.isNaN(d)) {
316                     sb.append("Double.NaN");
317                 } else if (Double.isInfinite(d)) {
318                     sb.append(d > 0? "Double.POSITIVE_INFINITY" : "Double.NEGATIVE_INFINITY");
319                 } else {
320                    sb.append(value);
321                    sb.append("d");
322                 }
323             } else {
324                 sb.append("(" + type.getName() + ")");
325                 sb.append(value + "L");
326             }
327             sb.append(";\n");
328         }
329 
330         decrAlign();
331     }

333     static int funcIntfCounter = 0;

335     protected void addUpcallFactory(FunctionDescriptor desc) {
336         String fnName = "FI" + funcIntfCounter++;
337         incrAlign();
338         indent();
339         sb.append(PRI_MODS + "FunctionDescriptor " + fnName + "$DESC = ");
340         addFunction(desc);
341         sb.append(";\n");
342         indent();
343         sb.append(PUB_MODS + "MemoryAddress " + fnName + "$make(MethodHandle handle) {\n");
344         incrAlign();
345         indent();
346         sb.append("return RuntimeHelper.upcallStub(handle, " + fnName + "$DESC);\n");
347         decrAlign();
348         indent();
349         sb.append("}\n");
350         decrAlign();
351     }

353     protected void addStaticFunctionWrapper(Declaration.Function f, MethodType mtype) {
354         incrAlign();
355         indent();
356         sb.append(PUB_MODS + mtype.returnType().getName() + " " + javaSafeIdentifier(f.name()) + " (");
357         String delim = "";
358         List<String> pNames = new ArrayList<>();
359         final int numParams = f.parameters().size();
360         for (int i = 0 ; i < numParams; i++) {
361             String pName = f.parameters().get(i).name();
362             if (pName.isEmpty()) {
363                 pName = "x" + i;
364             }
365             pName = javaSafeIdentifier(pName);
366             pNames.add(pName);
367             sb.append(delim + mtype.parameterType(i).getName() + " " + pName);
368             delim = ", ";
369         }
370         if (f.type().varargs()) {
371             String lastArg = "x" + numParams;
372             if (numParams > 0) {
373                 sb.append(", ");
374             }
375             sb.append("Object... " + lastArg);
376             pNames.add(lastArg);
377         }
378         sb.append(") {\n");
379         incrAlign();
380         indent();
381         sb.append("try {\n");
382         incrAlign();
383         indent();
384         if (!mtype.returnType().equals(void.class)) {
385             sb.append("return (" + mtype.returnType().getName() + ")");
386         }
387         sb.append(f.name() + ".invokeExact(" + String.join(", ", pNames) + ");\n");
388         decrAlign();
389         indent();
390         sb.append("} catch (Throwable ex) {\n");
391         incrAlign();
392         indent();
393         sb.append("throw new AssertionError(ex);\n");
394         decrAlign();
395         indent();
396         sb.append("}\n");
397         decrAlign();
398         indent();
399         sb.append("}\n");
400         decrAlign();
401     }

403     void addDescriptor(String name, FunctionDescriptor desc) {
404         incrAlign();
405         indent();
406         sb.append(PRI_MODS + "FunctionDescriptor " + name + "$DESC = ");

413     void addFunctionalInterface(String name, MethodType mtype) {
414         incrAlign();
415         indent();
416         sb.append("public interface " + javaSafeIdentifier(name) + " {\n");
417         incrAlign();
418         indent();
419         sb.append(mtype.returnType().getName() + " apply(");
420         String delim = "";
421         for (int i = 0 ; i < mtype.parameterCount() ; i++) {
422             sb.append(delim + mtype.parameterType(i).getName() + " x" + i);
423             delim = ", ";
424         }
425         sb.append(");\n");
426         decrAlign();
427         indent();
428         sb.append("}\n");
429         decrAlign();
430         indent();
431     }

433     protected void addFunctionalFactory(String name, MethodType mtype) {
434         incrAlign();
435         indent();
436         sb.append(PUB_MODS + "MemoryAddress " + name + "$make(" + name + " fi) {\n");
437         incrAlign();
438         indent();
439         sb.append("return RuntimeHelper.upcallStub(" + name + ".class, fi, " + name + "$DESC, " +
440                 "\"" + mtype.toMethodDescriptorString() + "\");\n");
441         decrAlign();
442         indent();
443         sb.append("}\n");
444         decrAlign();
445     }

447     void addGetter(String name, Class<?> type, String parent) {
448         incrAlign();
449         indent();
450         name = javaSafeIdentifier(name);
451         String vhName = (parent != null ? (javaSafeIdentifier(parent) + "$") : "") + name;
452         String param = parent != null ? (MemorySegment.class.getName() + " seg") : "";
453         sb.append(PUB_MODS + type.getName() + " " + vhName + "$get(" + param + ") {\n");
454         incrAlign();
455         indent();
456         String vhParam = parent != null ?
457                 "seg.baseAddress()" : name + "$ADDR";
458         sb.append("return (" + type.getName() + ")" + vhName + ".get(" + vhParam + ");\n");
459         decrAlign();
460         indent();
461         sb.append("}\n");
462         decrAlign();
463     }

465     void addSetter(String name, Class<?> type, String parent) {
466         incrAlign();
467         indent();
468         name = javaSafeIdentifier(name);
469         String vhName = (parent != null ? (javaSafeIdentifier(parent) + "$") : "") + name;
470         String param = parent != null ? (MemorySegment.class.getName() + " seg, ") : "";
471         sb.append(PUB_MODS + "void " + vhName + "$set(" + param + type.getName() + " x) {\n");
472         incrAlign();
473         indent();
474         String vhParam = parent != null ?
475                 "seg.baseAddress()" : name + "$ADDR";
476         sb.append(vhName + ".set(" + vhParam + ", x);\n");
477         decrAlign();
478         indent();
479         sb.append("}\n");
480         decrAlign();
481     }

483     protected String build() {
484         String res = sb.toString();
485         this.sb = null;
486         return res.toString();
487     }
488 
489     protected void indent() {
490         for (int i = 0; i < align; i++) {
491             sb.append("    ");
492         }
493     }
494 
495     protected void incrAlign() {
496         align++;
497     }
498 
499     protected void decrAlign() {
500         align--;
501     }
502 
503     protected final String javaSafeIdentifier(String name) {
504         // We never get the problem of Java non-identifiers (like 123, ab-xy) as
505         // C identifiers. But we may have a java keyword used as a C identifier.
506         assert SourceVersion.isIdentifier(name);
507 
508         return SourceVersion.isKeyword(name)? (name + "_") : name;
509     }
