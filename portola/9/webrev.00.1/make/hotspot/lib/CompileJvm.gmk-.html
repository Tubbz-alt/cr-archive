208 ################################################################################
209 # Hotspot disallows the use of global operators 'new' and 'delete'. This build
210 # time check helps enforce this requirement. If you trigger this check and the
211 # reference is not obvious from the source, GNU objdump can be used to help find
212 # the reference if compiled with GCC:
213 #
214 # objdump -lrdSC <path/to/file.o>
215 #
216 # -C demangle
217 # -d disassemble
218 # -r print relocation entries, interspersed with the disassembly
219 # -S print source code, intermixed with disassembly
220 # -l include filenames and line numbers
221 #
222 # Search the output for the operator(s) of interest, to see where they are
223 # referenced.
224 #
225 # When a reference to the global 'operator delete' is reported, it might be
226 # due to a "deleting destructor".  In this case, objdump may show the
227 # reference to be associated with but not actually in a destructor.  A
228 # deleting destructor is automatically generated for a class whose destructor
229 # is virtual.  A deleting destructor requires an accessible 'operator delete'
230 # for the associated class.  If the class doesn't provide a more specific
231 # declaration (either directly or by inheriting from a class that provides
232 # one) then the global definition will be used, triggering this check.
233 
234 ifneq ($(GENERATE_COMPILE_COMMANDS_ONLY), true)
235   ifneq ($(filter $(TOOLCHAIN_TYPE), gcc clang), )
236 
237     DEMANGLED_REGEXP := [^:]operator (new|delete)
238 
239     # Running c++filt to find offending symbols in all files is too expensive,
240     # so use mangled names when looking for symbols.
241     # Save the demangling for when something is actually found.
242     MANGLED_SYMS := \
243         _ZdaPv \
244         _ZdlPv \
245         _Znam \
246         _Znwm \
247         #
248     UNDEF_PATTERN := ' U '
249 
250     define SetupOperatorNewDeleteCheck
251         $1.op_check: $1
252 	  if [ -n "`$(NM) $$< | $(GREP) $(addprefix -e , $(MANGLED_SYMS)) \
253 	      | $(GREP) $(UNDEF_PATTERN)`" ]; then \
254 	    $(ECHO) "$$<: Error: Use of global operators new and delete is not allowed in Hotspot:"; \
255 	    $(NM) $$< | $(CXXFILT) | $(EGREP) '$(DEMANGLED_REGEXP)' | $(GREP) $(UNDEF_PATTERN); \
256 	    $(ECHO) "See: $(TOPDIR)/make/hotspot/lib/CompileJvm.gmk"; \
257 	    exit 1; \
258 	  fi
259 	  $(TOUCH) $$@
260 
261       TARGETS += $1.op_check
262     endef
263 
264     $(foreach o, $(BUILD_LIBJVM_ALL_OBJS), $(eval $(call SetupOperatorNewDeleteCheck,$o)))
265   endif
266 endif
