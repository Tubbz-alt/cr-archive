852 /*
853  * Provide a CounterGet() implementation based on gettimeofday() which
854  * is universally available, even though it may not be 'high resolution'
855  * compared to platforms that provide gethrtime() (like Solaris). It is
856  * also subject to time-of-day changes, but alternatives may not be
857  * known to be available at either build time or run time.
858  */
859 uint64_t CounterGet() {
860     uint64_t result = 0;
861     struct timeval tv;
862     if (gettimeofday(&tv, NULL) != -1) {
863         result = 1000000LL * (uint64_t)tv.tv_sec;
864         result += (uint64_t)tv.tv_usec;
865     }
866     return result;
867 }
