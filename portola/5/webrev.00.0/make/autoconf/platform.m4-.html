171 # Support macro for PLATFORM_EXTRACT_TARGET_AND_BUILD.
172 # Converts autoconf style OS name to OpenJDK style, into
173 # VAR_OS, VAR_OS_TYPE and VAR_OS_ENV.
174 AC_DEFUN([PLATFORM_EXTRACT_VARS_FROM_OS],
175 [
176   case "$1" in
177     *linux*)
178       VAR_OS=linux
179       VAR_OS_TYPE=unix
180       ;;
181     *solaris*)
182       VAR_OS=solaris
183       VAR_OS_TYPE=unix
184       ;;
185     *darwin*)
186       VAR_OS=macosx
187       VAR_OS_TYPE=unix
188       ;;
189     *bsd*)
190       VAR_OS=bsd
191       VAR_OS_TYPE=unix
192       ;;
193     *cygwin*)
194       VAR_OS=windows
195       VAR_OS_ENV=windows.cygwin
196       ;;
197     *wsl*)
198       VAR_OS=windows
199       VAR_OS_ENV=windows.wsl
200       ;;
201     *mingw*)
202       VAR_OS=windows
203       VAR_OS_ENV=windows.msys
204       ;;
205     *aix*)
206       VAR_OS=aix
207       VAR_OS_TYPE=unix
208       ;;
209     *)
210       AC_MSG_ERROR([unsupported operating system $1])
211       ;;
212   esac
213 
214   case "$1" in
215     *linux*-musl)
216       VAR_LIBC=musl
217       ;;
218     *linux*-gnu)
219       VAR_LIBC=gnu
220       ;;
221     *)
222       VAR_LIBC=default
223       ;;
224   esac
225 ])

227 # Expects $host_os $host_cpu $build_os and $build_cpu
228 # and $with_target_bits to have been setup!
229 #
230 # Translate the standard triplet(quadruplet) definition
231 # of the target/build system into OPENJDK_TARGET_OS, OPENJDK_TARGET_CPU,
232 # OPENJDK_BUILD_OS, etc.
233 AC_DEFUN([PLATFORM_EXTRACT_TARGET_AND_BUILD],
234 [
235   # Copy the autoconf trip/quadruplet verbatim to OPENJDK_TARGET_AUTOCONF_NAME
236   # (from the autoconf "host") and OPENJDK_BUILD_AUTOCONF_NAME
237   # Note that we might later on rewrite e.g. OPENJDK_TARGET_CPU due to reduced build,
238   # but this will not change the value of OPENJDK_TARGET_AUTOCONF_NAME.
239   OPENJDK_TARGET_AUTOCONF_NAME="$host"
240   OPENJDK_BUILD_AUTOCONF_NAME="$build"
241   AC_SUBST(OPENJDK_TARGET_AUTOCONF_NAME)
242   AC_SUBST(OPENJDK_BUILD_AUTOCONF_NAME)
243 
244   # Convert the autoconf OS/CPU value to our own data, into the VAR_OS/CPU variables.
245   PLATFORM_EXTRACT_VARS_FROM_OS($build_os)
246   PLATFORM_EXTRACT_VARS_FROM_CPU($build_cpu)
247   # ..and setup our own variables. (Do this explicitly to facilitate searching)
248   OPENJDK_BUILD_OS="$VAR_OS"
249   if test "x$VAR_OS_TYPE" != x; then
250     OPENJDK_BUILD_OS_TYPE="$VAR_OS_TYPE"
251   else
252     OPENJDK_BUILD_OS_TYPE="$VAR_OS"
253   fi
254   if test "x$VAR_OS_ENV" != x; then
255     OPENJDK_BUILD_OS_ENV="$VAR_OS_ENV"
256   else
257     OPENJDK_BUILD_OS_ENV="$VAR_OS"
258   fi
259   OPENJDK_BUILD_CPU="$VAR_CPU"
260   OPENJDK_BUILD_CPU_ARCH="$VAR_CPU_ARCH"
261   OPENJDK_BUILD_CPU_BITS="$VAR_CPU_BITS"
262   OPENJDK_BUILD_CPU_ENDIAN="$VAR_CPU_ENDIAN"
263   OPENJDK_BUILD_LIBC="$VAR_LIBC"
264   AC_SUBST(OPENJDK_BUILD_OS)
265   AC_SUBST(OPENJDK_BUILD_OS_TYPE)
266   AC_SUBST(OPENJDK_BUILD_OS_ENV)
267   AC_SUBST(OPENJDK_BUILD_CPU)
268   AC_SUBST(OPENJDK_BUILD_CPU_ARCH)
269   AC_SUBST(OPENJDK_BUILD_CPU_BITS)
270   AC_SUBST(OPENJDK_BUILD_CPU_ENDIAN)
271   AC_SUBST(OPENJDK_BUILD_LIBC)
272 
273   AC_MSG_CHECKING([openjdk-build os-cpu])
274   AC_MSG_RESULT([$OPENJDK_BUILD_OS-$OPENJDK_BUILD_CPU])
275 
276   if test "x$OPENJDK_BUILD_OS" = "xlinux"; then
277     AC_MSG_CHECKING([openjdk-build C library])
278     AC_MSG_RESULT([$OPENJDK_BUILD_LIBC])
279   fi
280 
281   # Convert the autoconf OS/CPU value to our own data, into the VAR_OS/CPU variables.
282   PLATFORM_EXTRACT_VARS_FROM_OS($host_os)
283   PLATFORM_EXTRACT_VARS_FROM_CPU($host_cpu)
284   # ... and setup our own variables. (Do this explicitly to facilitate searching)
285   OPENJDK_TARGET_OS="$VAR_OS"
286   if test "x$VAR_OS_TYPE" != x; then
287     OPENJDK_TARGET_OS_TYPE="$VAR_OS_TYPE"
288   else
289     OPENJDK_TARGET_OS_TYPE="$VAR_OS"
290   fi
291   if test "x$VAR_OS_ENV" != x; then
292     OPENJDK_TARGET_OS_ENV="$VAR_OS_ENV"
293   else
294     OPENJDK_TARGET_OS_ENV="$VAR_OS"
295   fi
296   OPENJDK_TARGET_CPU="$VAR_CPU"
297   OPENJDK_TARGET_CPU_ARCH="$VAR_CPU_ARCH"
298   OPENJDK_TARGET_CPU_BITS="$VAR_CPU_BITS"
299   OPENJDK_TARGET_CPU_ENDIAN="$VAR_CPU_ENDIAN"
300   OPENJDK_TARGET_OS_UPPERCASE=`$ECHO $OPENJDK_TARGET_OS | $TR 'abcdefghijklmnopqrstuvwxyz' 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'`
301   OPENJDK_TARGET_LIBC="$VAR_LIBC"
302 
303   AC_SUBST(OPENJDK_TARGET_OS)
304   AC_SUBST(OPENJDK_TARGET_OS_TYPE)
305   AC_SUBST(OPENJDK_TARGET_OS_ENV)
306   AC_SUBST(OPENJDK_TARGET_OS_UPPERCASE)
307   AC_SUBST(OPENJDK_TARGET_CPU)
308   AC_SUBST(OPENJDK_TARGET_CPU_ARCH)
309   AC_SUBST(OPENJDK_TARGET_CPU_BITS)
310   AC_SUBST(OPENJDK_TARGET_CPU_ENDIAN)
311   AC_SUBST(OPENJDK_TARGET_LIBC)
312 
313   AC_MSG_CHECKING([openjdk-target os-cpu])
314   AC_MSG_RESULT([$OPENJDK_TARGET_OS-$OPENJDK_TARGET_CPU])
315 
316   if test "x$OPENJDK_TARGET_OS" = "xlinux"; then
317     AC_MSG_CHECKING([openjdk-target C library])
318     AC_MSG_RESULT([$OPENJDK_TARGET_LIBC])
319   fi
320 ])

322 # Check if a reduced build (32-bit on 64-bit platforms) is requested, and modify behaviour
323 # accordingly. Must be done after setting up build and target system, but before
324 # doing anything else with these values.
325 AC_DEFUN([PLATFORM_SETUP_TARGET_CPU_BITS],
326 [
327   AC_ARG_WITH(target-bits, [AS_HELP_STRING([--with-target-bits],
328        [build 32-bit or 64-bit binaries (for platforms that support it), e.g. --with-target-bits=32 @<:@guessed@:>@])])
329 
330   # We have three types of compiles:
331   # native  == normal compilation, target system == build system
332   # cross   == traditional cross compilation, target system != build system; special toolchain needed
333   # reduced == using native compilers, but with special flags (e.g. -m32) to produce 32-bit builds on 64-bit machines
334   #
335   if test "x$OPENJDK_BUILD_AUTOCONF_NAME" != "x$OPENJDK_TARGET_AUTOCONF_NAME"; then
336     # We're doing a proper cross-compilation
337     COMPILE_TYPE="cross"
338   else
339     COMPILE_TYPE="native"
340   fi
341 
342   if test "x$with_target_bits" != x; then
343     if test "x$COMPILE_TYPE" = "xcross"; then
344       AC_MSG_ERROR([It is not possible to combine --with-target-bits=X and proper cross-compilation. Choose either.])
345     fi
346 
347     if test "x$with_target_bits" = x32 && test "x$OPENJDK_TARGET_CPU_BITS" = x64; then
348       # A reduced build is requested
349       COMPILE_TYPE="reduced"
350       OPENJDK_TARGET_CPU_BITS=32
351       if test "x$OPENJDK_TARGET_CPU_ARCH" = "xx86"; then
352         OPENJDK_TARGET_CPU=x86
353       elif test "x$OPENJDK_TARGET_CPU_ARCH" = "xsparc"; then
354         OPENJDK_TARGET_CPU=sparc
355       else
356         AC_MSG_ERROR([Reduced build (--with-target-bits=32) is only supported on x86_64 and sparcv9])
357       fi
358     elif test "x$with_target_bits" = x64 && test "x$OPENJDK_TARGET_CPU_BITS" = x32; then
359       AC_MSG_ERROR([It is not possible to use --with-target-bits=64 on a 32 bit system. Use proper cross-compilation instead.])
360     elif test "x$with_target_bits" = "x$OPENJDK_TARGET_CPU_BITS"; then
361       AC_MSG_NOTICE([--with-target-bits are set to build platform address size; argument has no meaning])
362     else
363       AC_MSG_ERROR([--with-target-bits can only be 32 or 64, you specified $with_target_bits!])
364     fi
365   fi
366   AC_SUBST(COMPILE_TYPE)
367 
368   AC_MSG_CHECKING([compilation type])
369   AC_MSG_RESULT([$COMPILE_TYPE])
370 ])

380 # $1 - Either TARGET or BUILD to setup the variables for.
381 AC_DEFUN([PLATFORM_SETUP_LEGACY_VARS_HELPER],
382 [
383   # Also store the legacy naming of the cpu.
384   # Ie i586 and amd64 instead of x86 and x86_64
385   OPENJDK_$1_CPU_LEGACY="$OPENJDK_$1_CPU"
386   if test "x$OPENJDK_$1_CPU" = xx86; then
387     OPENJDK_$1_CPU_LEGACY="i586"
388   elif test "x$OPENJDK_$1_OS" != xmacosx && test "x$OPENJDK_$1_CPU" = xx86_64; then
389     # On all platforms except MacOSX replace x86_64 with amd64.
390     OPENJDK_$1_CPU_LEGACY="amd64"
391   elif test "x$OPENJDK_$1_CPU" = xalpha; then
392     # Avoid name collisions with variables named alpha
393     OPENJDK_$1_CPU_LEGACY="_alpha_"
394   elif test "x$OPENJDK_$1_CPU" = xsh; then
395     # Avoid name collisions with variables named sh
396     OPENJDK_$1_CPU_LEGACY="_sh_"
397   fi
398   AC_SUBST(OPENJDK_$1_CPU_LEGACY)
399 
400   # And the second legacy naming of the cpu.
401   # Ie i386 and amd64 instead of x86 and x86_64.
402   OPENJDK_$1_CPU_LEGACY_LIB="$OPENJDK_$1_CPU"
403   if test "x$OPENJDK_$1_CPU" = xx86; then
404     OPENJDK_$1_CPU_LEGACY_LIB="i386"
405   elif test "x$OPENJDK_$1_CPU" = xx86_64; then
406     OPENJDK_$1_CPU_LEGACY_LIB="amd64"
407   fi
408   AC_SUBST(OPENJDK_$1_CPU_LEGACY_LIB)
409 
410   # OPENJDK_$1_CPU_ISADIR is normally empty. On 64-bit Solaris systems, it is set to
411   # /amd64 or /sparcv9. This string is appended to some library paths, like this:
412   # /usr/lib${OPENJDK_$1_CPU_ISADIR}/libexample.so
413   OPENJDK_$1_CPU_ISADIR=""
414   if test "x$OPENJDK_$1_OS" = xsolaris; then
415     if test "x$OPENJDK_$1_CPU" = xx86_64; then
416       OPENJDK_$1_CPU_ISADIR="/amd64"
417     elif test "x$OPENJDK_$1_CPU" = xsparcv9; then
418       OPENJDK_$1_CPU_ISADIR="/sparcv9"
419     fi
420   fi
421   AC_SUBST(OPENJDK_$1_CPU_ISADIR)
422 
423   # Setup OPENJDK_$1_CPU_OSARCH, which is used to set the os.arch Java system property
424   OPENJDK_$1_CPU_OSARCH="$OPENJDK_$1_CPU"
425   if test "x$OPENJDK_$1_OS" = xlinux && test "x$OPENJDK_$1_CPU" = xx86; then
426     # On linux only, we replace x86 with i386.
427     OPENJDK_$1_CPU_OSARCH="i386"
428   elif test "x$OPENJDK_$1_OS" != xmacosx && test "x$OPENJDK_$1_CPU" = xx86_64; then
429     # On all platforms except macosx, we replace x86_64 with amd64.
430     OPENJDK_$1_CPU_OSARCH="amd64"
431   fi
432   AC_SUBST(OPENJDK_$1_CPU_OSARCH)
433 
434   OPENJDK_$1_CPU_JLI="$OPENJDK_$1_CPU"
435   if test "x$OPENJDK_$1_CPU" = xx86; then
436     OPENJDK_$1_CPU_JLI="i386"
437   elif test "x$OPENJDK_$1_OS" != xmacosx && test "x$OPENJDK_$1_CPU" = xx86_64; then
438     # On all platforms except macosx, we replace x86_64 with amd64.
439     OPENJDK_$1_CPU_JLI="amd64"
440   fi
441 
442   # The new version string in JDK 9 also defined new naming of OS and ARCH for bundles
443   # Macosx is osx and x86_64 is x64
444   if test "x$OPENJDK_$1_OS" = xmacosx; then
445     OPENJDK_$1_OS_BUNDLE="osx"
446   else
447     OPENJDK_$1_OS_BUNDLE="$OPENJDK_TARGET_OS"
448   fi
449   if test "x$OPENJDK_$1_CPU" = xx86_64; then
450     OPENJDK_$1_CPU_BUNDLE="x64"
451   else
452     OPENJDK_$1_CPU_BUNDLE="$OPENJDK_$1_CPU"
453   fi
454 
455   OPENJDK_$1_LIBC_BUNDLE=""
456   if test "x$OPENJDK_$1_LIBC" = "xmusl"; then  
457     OPENJDK_$1_LIBC_BUNDLE="-$OPENJDK_$1_LIBC"
458   fi
459 
460   OPENJDK_$1_BUNDLE_PLATFORM="${OPENJDK_$1_OS_BUNDLE}-${OPENJDK_$1_CPU_BUNDLE}${OPENJDK_$1_LIBC_BUNDLE}"
461   AC_SUBST(OPENJDK_$1_BUNDLE_PLATFORM)
462 
463   if test "x$COMPILE_TYPE" = "xcross"; then
464     # FIXME: ... or should this include reduced builds..?
465     DEFINE_CROSS_COMPILE_ARCH="CROSS_COMPILE_ARCH:=$OPENJDK_$1_CPU_LEGACY"
466   else
467     DEFINE_CROSS_COMPILE_ARCH=""
468   fi
469   AC_SUBST(DEFINE_CROSS_COMPILE_ARCH)
470 
471   # Convert openjdk platform names to hotspot names
472 
473   HOTSPOT_$1_OS=${OPENJDK_$1_OS}
474   if test "x$OPENJDK_$1_OS" = xmacosx; then
475     HOTSPOT_$1_OS=bsd
476   fi
477   AC_SUBST(HOTSPOT_$1_OS)
478 
479   HOTSPOT_$1_OS_TYPE=${OPENJDK_$1_OS_TYPE}
480   if test "x$OPENJDK_$1_OS_TYPE" = xunix; then
481     HOTSPOT_$1_OS_TYPE=posix
482   fi
483   AC_SUBST(HOTSPOT_$1_OS_TYPE)
484 
485   HOTSPOT_$1_CPU=${OPENJDK_$1_CPU}
486   if test "x$OPENJDK_$1_CPU" = xx86; then
487     HOTSPOT_$1_CPU=x86_32
488   elif test "x$OPENJDK_$1_CPU" = xsparcv9; then
489     HOTSPOT_$1_CPU=sparc
490   elif test "x$OPENJDK_$1_CPU" = xppc64; then
491     HOTSPOT_$1_CPU=ppc_64
492   elif test "x$OPENJDK_$1_CPU" = xppc64le; then
493     HOTSPOT_$1_CPU=ppc_64
494   fi
495   AC_SUBST(HOTSPOT_$1_CPU)
496 
497   # This is identical with OPENJDK_*, but define anyway for consistency.
498   HOTSPOT_$1_CPU_ARCH=${OPENJDK_$1_CPU_ARCH}
499   AC_SUBST(HOTSPOT_$1_CPU_ARCH)
500 
501   # Setup HOTSPOT_$1_CPU_DEFINE
502   if test "x$OPENJDK_$1_CPU" = xx86; then
503     HOTSPOT_$1_CPU_DEFINE=IA32
504   elif test "x$OPENJDK_$1_CPU" = xx86_64; then
505     HOTSPOT_$1_CPU_DEFINE=AMD64
506   elif test "x$OPENJDK_$1_CPU" = xx32; then
507     HOTSPOT_$1_CPU_DEFINE=X32
508   elif test "x$OPENJDK_$1_CPU" = xsparcv9; then
509     HOTSPOT_$1_CPU_DEFINE=SPARC
510   elif test "x$OPENJDK_$1_CPU" = xaarch64; then
511     HOTSPOT_$1_CPU_DEFINE=AARCH64
512   elif test "x$OPENJDK_$1_CPU" = xppc64; then
513     HOTSPOT_$1_CPU_DEFINE=PPC64
514   elif test "x$OPENJDK_$1_CPU" = xppc64le; then
515     HOTSPOT_$1_CPU_DEFINE=PPC64
516 
517   # The cpu defines below are for zero, we don't support them directly.
518   elif test "x$OPENJDK_$1_CPU" = xsparc; then
519     HOTSPOT_$1_CPU_DEFINE=SPARC
520   elif test "x$OPENJDK_$1_CPU" = xppc; then
521     HOTSPOT_$1_CPU_DEFINE=PPC32
522   elif test "x$OPENJDK_$1_CPU" = xs390; then
523     HOTSPOT_$1_CPU_DEFINE=S390
524   elif test "x$OPENJDK_$1_CPU" = xs390x; then
525     HOTSPOT_$1_CPU_DEFINE=S390
526   elif test "x$OPENJDK_$1_CPU" = xriscv64; then
527     HOTSPOT_$1_CPU_DEFINE=RISCV
528   elif test "x$OPENJDK_$1_CPU" != x; then
529     HOTSPOT_$1_CPU_DEFINE=$(echo $OPENJDK_$1_CPU | tr a-z A-Z)
530   fi
531   AC_SUBST(HOTSPOT_$1_CPU_DEFINE)
532 
533   if test "x$OPENJDK_$1_LIBC" = "xmusl"; then
534     HOTSPOT_$1_LIBC=$OPENJDK_$1_LIBC
535   else
536     HOTSPOT_$1_LIBC=""
537   fi
538   AC_SUBST(HOTSPOT_$1_LIBC)
539 
540   # For historical reasons, the OS include directories have odd names.
541   OPENJDK_$1_OS_INCLUDE_SUBDIR="$OPENJDK_TARGET_OS"
542   if test "x$OPENJDK_TARGET_OS" = "xwindows"; then
543     OPENJDK_$1_OS_INCLUDE_SUBDIR="win32"
544   elif test "x$OPENJDK_TARGET_OS" = "xmacosx"; then
545     OPENJDK_$1_OS_INCLUDE_SUBDIR="darwin"
546   fi
547   AC_SUBST(OPENJDK_$1_OS_INCLUDE_SUBDIR)
548 ])

644 AC_DEFUN_ONCE([PLATFORM_SETUP_OPENJDK_TARGET_BITS],
645 [
646   ###############################################################################
647   #
648   # Now we check if libjvm.so will use 32 or 64 bit pointers for the C/C++ code.
649   # (The JVM can use 32 or 64 bit Java pointers but that decision
650   # is made at runtime.)
651   #
652 
653   # Make compilation sanity check
654   AC_CHECK_HEADERS([stdio.h], , [
655     AC_MSG_NOTICE([Failed to compile stdio.h. This likely implies missing compile dependencies.])
656     if test "x$COMPILE_TYPE" = xreduced; then
657       HELP_MSG_MISSING_DEPENDENCY([reduced])
658       AC_MSG_NOTICE([You are doing a reduced build. Check that you have 32-bit libraries installed. $HELP_MSG])
659     elif test "x$COMPILE_TYPE" = xcross; then
660       AC_MSG_NOTICE([You are doing a cross-compilation. Check that you have all target platform libraries installed.])
661     fi
662     AC_MSG_ERROR([Cannot continue.])
663   ])
664 
665   AC_CHECK_SIZEOF([int *], [1111])
666 
667   # AC_CHECK_SIZEOF defines 'ac_cv_sizeof_int_p' to hold the number of bytes used by an 'int*'
668   if test "x$ac_cv_sizeof_int_p" = x; then
669     # The test failed, lets stick to the assumed value.
670     AC_MSG_WARN([The number of bits in the target could not be determined, using $OPENJDK_TARGET_CPU_BITS.])
671   else
672     TESTED_TARGET_CPU_BITS=`expr 8 \* $ac_cv_sizeof_int_p`
673 
674     if test "x$TESTED_TARGET_CPU_BITS" != "x$OPENJDK_TARGET_CPU_BITS"; then
675       AC_MSG_NOTICE([The tested number of bits in the target ($TESTED_TARGET_CPU_BITS) differs from the number of bits expected to be found in the target ($OPENJDK_TARGET_CPU_BITS)])
676       if test "x$COMPILE_TYPE" = xreduced; then
677         HELP_MSG_MISSING_DEPENDENCY([reduced])
678         AC_MSG_NOTICE([You are doing a reduced build. Check that you have 32-bit libraries installed. $HELP_MSG])
679       elif test "x$COMPILE_TYPE" = xcross; then
680         AC_MSG_NOTICE([You are doing a cross-compilation. Check that you have all target platform libraries installed.])
681       fi
682       AC_MSG_ERROR([Cannot continue.])
683     fi
684   fi
685 
686   AC_MSG_CHECKING([for target address size])
687   AC_MSG_RESULT([$OPENJDK_TARGET_CPU_BITS bits])
688 ])
