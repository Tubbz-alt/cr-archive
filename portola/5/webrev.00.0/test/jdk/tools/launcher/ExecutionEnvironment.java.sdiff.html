178     /*
179      * This test ensures that LD_LIBRARY_PATH* values are interpreted by the VM
180      * and the expected java.library.path behaviour.
181      * For Generic platforms (All *nixes):
182      *    * All LD_LIBRARY_PATH variable should be on java.library.path
183      * For Solaris 32-bit
184      *    * The LD_LIBRARY_PATH_32 should override LD_LIBRARY_PATH if specified
185      * For Solaris 64-bit
186      *    * The LD_LIBRARY_PATH_64 should override LD_LIBRARY_PATH if specified
187      */
188     @Test
189     void testJavaLibraryPath() {
190         TestResult tr;
191 
192         Map<String, String> env = new HashMap<>();
193 
194         if (TestHelper.isSolaris) {
195             // no override
196             env.clear();
197             env.put(LD_LIBRARY_PATH, LD_LIBRARY_PATH_VALUE);
198             tr = doExec(env, javaCmd, "-jar", testJarFile.getAbsolutePath());
199             verifyJavaLibraryPathGeneric(tr);
200 
201             env.clear();
202             for (String x : LD_PATH_STRINGS) {
203                 String pairs[] = x.split("=");
204                 env.put(pairs[0], pairs[1]);
205             }
206 
207             // verify the override occurs for 64-bit system
208             tr = doExec(env, javaCmd, "-jar", testJarFile.getAbsolutePath());
209             verifyJavaLibraryPathOverride(tr, false);
210         } else {
211             for (String x : LD_PATH_STRINGS) {
212                 String pairs[] = x.split("=");
213                 env.put(pairs[0], pairs[1]);
214             }
215 
216             tr = doExec(env, javaCmd, "-jar", testJarFile.getAbsolutePath());
217             verifyJavaLibraryPathGeneric(tr);
218         }



219     }

221     private void verifyJavaLibraryPathGeneric(TestResult tr) {
222         if (!tr.matches("java.library.path=.*" + LD_LIBRARY_PATH_VALUE + ".*")) {
223             flagError(tr, "testJavaLibraryPath: java.library.path does not contain " +
224                     LD_LIBRARY_PATH_VALUE);
225         }
226     }

228     private void verifyJavaLibraryPathOverride(TestResult tr,
229             boolean is32Bit) {
230         // make sure the 32/64 bit value exists
231         if (!tr.matches("java.library.path=.*" +
232                 (is32Bit ? LD_LIBRARY_PATH_32_VALUE : LD_LIBRARY_PATH_64_VALUE) + ".*")) {
233             flagError(tr, "verifyJavaLibraryPathOverride: " +
234                 " java.library.path does not contain " +
235                     (is32Bit ? LD_LIBRARY_PATH_32_VALUE : LD_LIBRARY_PATH_64_VALUE));
236 
237         }
