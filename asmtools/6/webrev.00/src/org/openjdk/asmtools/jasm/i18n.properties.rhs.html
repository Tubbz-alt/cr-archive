 23 Usage: java -jar asmtools.jar jasm [options] file.jasm...\n\
 24 where possible options include:
 25 
 26 jasm.opt.d=\
 27 \     -d destdir  directory to place resulting .class files
 28 jasm.opt.v=\
 29 \     -v          add trace information
 30 jasm.opt.g=\
 31 \     -g          add debug information
 32 jasm.opt.version=\
 33 \     -version    prints the program version
 34 jasm.opt.nowrite=\
 35 \     -nowrite    do not write resulting .class files
 36 jasm.opt.strict=\
 37 \     -strict     consider warnings as errors
 38 jasm.opt.nowarn=\
 39 \     -nowarn     do not print warnings
 40 jasm.opt.cv=\
 41 \     -cv major.minor  set operating class file version (by default {0}.{1})
