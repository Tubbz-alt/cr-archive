25 package com.sun.tdk.jcov.runtime;

27 /**
28  *
29  * @author Sergey Borodin
30  */
31 class JCovServerSocketSaver implements JCovSaver {
32 
33     protected static final int DEFAULT_PORT = 3335;
34     protected static final String PORT = "server.port";
35     protected int port;
36 
37     static int detectPort() {
38         String p = null;
39         try {
40             p = PropertyFinder.findValue(PORT, "" + DEFAULT_PORT);
41             return Integer.parseInt(p);
42         } catch (NumberFormatException e) {
43             System.err.println("JCovRT: Port parse error (not a number) " + p);
44         } catch (Throwable ignore) {
45         }
46         return DEFAULT_PORT;
47     }
48 
49     public JCovServerSocketSaver() {
50         port = detectPort();
51     }
52 
53     public JCovServerSocketSaver(int port) {
54         this.port = port;
55     }
56 
57     protected void startServer(Thread server) {
58         server.start();
59     }
60 
61     public void saveResults() {
62         //throw new UnsupportedOperationException("Not supported yet.");
63     }
64 }
