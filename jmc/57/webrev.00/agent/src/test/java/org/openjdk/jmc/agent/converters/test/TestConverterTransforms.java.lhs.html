33 package org.openjdk.jmc.agent.converters.test;
34 
35 import static org.junit.Assert.assertTrue;
36 
37 import java.io.IOException;
38 import java.lang.instrument.IllegalClassFormatException;
39 import java.util.concurrent.atomic.AtomicInteger;
40 
41 import javax.xml.stream.XMLStreamException;
42 
43 import org.junit.Test;
44 import org.objectweb.asm.Type;
45 import org.openjdk.jmc.agent.TransformRegistry;
46 import org.openjdk.jmc.agent.impl.DefaultTransformRegistry;
47 import org.openjdk.jmc.agent.test.util.TestToolkit;
48 
49 public class TestConverterTransforms {
50 	private static AtomicInteger runCount = new AtomicInteger(0);
51 
52 	public static String getTemplate() throws IOException {
53 		return TestToolkit.readTemplate(TestConverterTransforms.class, TestToolkit.DEFAULT_TEMPLATE_NAME);
54 	}
55 
56 	@Test
57 	public void testRunConverterTransforms() throws XMLStreamException, IllegalClassFormatException, IOException {
58 		TransformRegistry registry = DefaultTransformRegistry.from(TestToolkit.getProbesXMLFromTemplate(getTemplate(),
59 				"testRunConverterTransforms" + runCount.getAndIncrement())); //$NON-NLS-1$
60 
61 		assertTrue(registry.hasPendingTransforms(Type.getInternalName(InstrumentMeConverter.class)));
62 	}
63 }
