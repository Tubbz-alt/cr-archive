33 package org.openjdk.jmc.agent;
34 
35 /**
36  * Definition of a method to be logged by the agent.
37  */
38 public class Method {
39 	private String name;
40 	private String signature;
41 
42 	public Method(String name, String signature) {
43 		this.name = name;
44 		this.signature = signature;
45 	}
46 
47 	public String getName() {
48 		return name;
49 	}
50 
51 	public String getSignature() {
52 		return signature;
53 	}
54 
55 	@Override
56 	public int hashCode() {
57 		final int prime = 31;
58 		int result = 1;
59 		result = prime * result + ((name == null) ? 0 : name.hashCode());
60 		result = prime * result + ((signature == null) ? 0 : signature.hashCode());
61 		return result;
62 	}
63 
64 	@Override
65 	public boolean equals(Object obj) {
66 		if (this == obj) {
67 			return true;
68 		}
69 		if (obj == null) {
70 			return false;
71 		}
72 		if (getClass() != obj.getClass()) {
73 			return false;
74 		}
75 		Method other = (Method) obj;
76 		if (name == null) {
77 			if (other.name != null) {
78 				return false;
79 			}
80 		} else if (!name.equals(other.name)) {
81 			return false;
82 		}
83 		if (signature == null) {
84 			if (other.signature != null) {
85 				return false;
86 			}
87 		} else if (!signature.equals(other.signature)) {
88 			return false;
89 		}
90 		return true;
91 	}
92 
93 	@Override
94 	public String toString() {
95 		return name + signature;
96 	}
97 }
