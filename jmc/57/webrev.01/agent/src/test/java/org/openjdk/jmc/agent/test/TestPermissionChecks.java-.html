34 package org.openjdk.jmc.agent.test;
35 
36 import static org.junit.Assert.assertTrue;
37 import java.lang.management.ManagementFactory;
38 
39 import javax.management.JMX;
40 import javax.management.ObjectName;
41 
42 import org.junit.Test;
43 import org.openjdk.jmc.agent.jmx.AgentControllerMBean;
44 
45 public class TestPermissionChecks {
46 
47 	private static final String AGENT_OBJECT_NAME = "org.openjdk.jmc.jfr.agent:type=AgentController"; //$NON-NLS-1$
48 
49 	@Test
50 	public void testPermissionChecks() throws Exception {
51 		boolean exceptionThrown = false;
52 		try {
53 			doDefineEventProbes("");
54 		} catch(SecurityException e) {
55 			exceptionThrown = true;
56 		}
57 		assertTrue(exceptionThrown);
58 	}
59 
60 	private void doDefineEventProbes(String xmlDescription) throws Exception  {
61 		AgentControllerMBean mbean = JMX.newMXBeanProxy(ManagementFactory.getPlatformMBeanServer(),
62 				new ObjectName(AGENT_OBJECT_NAME), AgentControllerMBean.class, false);
63 		mbean.defineEventProbes(xmlDescription);
64 	}
65 
66 	public void test() {
67 		//Dummy method for instrumentation
68 	}
69 }
