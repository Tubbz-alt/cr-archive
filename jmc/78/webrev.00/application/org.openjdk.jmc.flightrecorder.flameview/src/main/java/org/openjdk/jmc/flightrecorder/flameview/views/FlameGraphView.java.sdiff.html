345 	private void rebuildModel(IItemCollection items) {
346 		// Release old model before building the new
347 		if (currentModelCalculator != null) {
348 			currentModelCalculator.cancel(true);
349 		}
350 		currentModelCalculator = getModelPreparer(items, frameSeparator, true);

351 		currentModelCalculator.thenAcceptAsync(this::setModel, DisplayToolkit.inDisplayThread())
352 				.exceptionally(FlameGraphView::handleModelBuildException);
353 	}

355 	private CompletableFuture<TraceNode> getModelPreparer(
356 		final IItemCollection items, final FrameSeparator separator, final boolean materializeSelectedBranches) {




357 		return CompletableFuture.supplyAsync(() -> {
358 			StacktraceModel model = new StacktraceModel(threadRootAtTop, frameSeparator, items);
359 			TraceNode root = TraceTreeUtils.createRootWithDescription(items, model.getRootFork().getBranchCount());
360 			return TraceTreeUtils.createTree(root, model);









361 		}, MODEL_EXECUTOR);
362 	}

391 	private void saveFlameGraph() {
392 		CompletableFuture<String> future = new CompletableFuture<>();
393 		String[] destination = new String[2];
394 
395 		DisplayToolkit.inDisplayThread().execute(() -> {
396 			FileDialog fd = new FileDialog(browser.getShell(), SWT.SAVE);
397 			fd.setText(getFlameviewMessage(FLAMEVIEW_SAVE_FLAME_GRAPH_AS));
398 			fd.setFilterNames(
399 					new String[] {getFlameviewMessage(FLAMEVIEW_JPEG_IMAGE), getFlameviewMessage(FLAMEVIEW_PNG_IMAGE)});
400 			fd.setFilterExtensions(new String[] {"*.jpg", "*.png"}); //$NON-NLS-1$ //$NON-NLS-2$
401 			fd.setFileName("flame_graph"); //$NON-NLS-1$
402 			fd.setOverwrite(true);
403 			if (fd.open() == null) {
404 				future.cancel(true);
405 				return;
406 			}
407 
408 			String type;
409 			String fileName = fd.getFileName().toLowerCase();
410 			// FIXME: FileDialog filterIndex returns -1 (https://bugs.eclipse.org/bugs/show_bug.cgi?id=546256)

411 			if (fileName.endsWith(".jpg") || fileName.endsWith(".jpeg")) { //$NON-NLS-1$ //$NON-NLS-2$
412 				type = "image/jpeg"; //$NON-NLS-1$
413 			} else if (fileName.endsWith(".png")) { //$NON-NLS-1$
414 				type = "image/png"; //$NON-NLS-1$
415 			} else {
416 				future.completeExceptionally(new UnsupportedOperationException("Unsupported image format")); //$NON-NLS-1$
417 				return;
418 			}
419 			destination[0] = fd.getFilterPath();
420 			destination[1] = fd.getFileName();
421 
422 			String callback = "_saveFlameGraphCallback"; //$NON-NLS-1$
423 			new BrowserFunction(browser, callback) {
424 				@Override
425 				public Object function(Object[] arguments) {
426 					if (arguments.length > 1) {
427 						future.completeExceptionally(new RuntimeException((String) arguments[1]));
428 						return null;
429 					}
430 					future.complete((String) arguments[0]);

383 	private void rebuildModel() {
384 		// try to cancel downstream calculation
385 		if (currentModelCalculator != null) {
386 			currentModelCalculator.cancel(true);
387 		}
388 
389 		currentModelCalculator = getModelPreparer(frameSeparator, true);
390 		currentModelCalculator.thenAcceptAsync(this::setModel, DisplayToolkit.inDisplayThread())
391 				.exceptionally(FlameGraphView::handleModelBuildException);
392 	}

398 	private CompletableFuture<ModelsContainer> getModelPreparer(final FrameSeparator separator,
399 			final boolean materializeSelectedBranches) {
400 		return CompletableFuture.supplyAsync(() -> {
401 			StacktraceModel model = createStacktraceModel();
402 			Fork rootFork = model.getRootFork();
403 			if (materializeSelectedBranches) {
404 				Branch selectedBranch = getLastSelectedBranch(rootFork);
405 				if (selectedBranch != null) {
406 					selectedBranch.getEndFork();
407 				}
408 			}
409 			
410 			TraceNode root = TraceTreeUtils.createRootWithDescription(currentItems, rootFork.getBranchCount());
411 			return new ModelsContainer(TraceTreeUtils.createTree(root, model), model);
412 
413 		}, MODEL_EXECUTOR);
414 	}

444 	private void saveFlameGraph() {
445 		CompletableFuture<String> future = new CompletableFuture<>();
446 		String[] destination = new String[2];
447 
448 		DisplayToolkit.inDisplayThread().execute(() -> {
449 			FileDialog fd = new FileDialog(browser.getShell(), SWT.SAVE);
450 			fd.setText(getFlameviewMessage(FLAMEVIEW_SAVE_FLAME_GRAPH_AS));
451 			fd.setFilterNames(new String[] { getFlameviewMessage(FLAMEVIEW_JPEG_IMAGE),
452 					getFlameviewMessage(FLAMEVIEW_PNG_IMAGE) });
453 			fd.setFilterExtensions(new String[] { "*.jpg", "*.png" }); //$NON-NLS-1$ //$NON-NLS-2$
454 			fd.setFileName("flame_graph"); //$NON-NLS-1$
455 			fd.setOverwrite(true);
456 			if (fd.open() == null) {
457 				future.cancel(true);
458 				return;
459 			}
460 
461 			String type;
462 			String fileName = fd.getFileName().toLowerCase();
463 			// FIXME: FileDialog filterIndex returns -1
464 			// (https://bugs.eclipse.org/bugs/show_bug.cgi?id=546256)
465 			if (fileName.endsWith(".jpg") || fileName.endsWith(".jpeg")) { //$NON-NLS-1$ //$NON-NLS-2$
466 				type = "image/jpeg"; //$NON-NLS-1$
467 			} else if (fileName.endsWith(".png")) { //$NON-NLS-1$
468 				type = "image/png"; //$NON-NLS-1$
469 			} else {
470 				future.completeExceptionally(new UnsupportedOperationException("Unsupported image format")); //$NON-NLS-1$
471 				return;
472 			}
473 			destination[0] = fd.getFilterPath();
474 			destination[1] = fd.getFileName();
475 
476 			String callback = "_saveFlameGraphCallback"; //$NON-NLS-1$
477 			new BrowserFunction(browser, callback) {
478 				@Override
479 				public Object function(Object[] arguments) {
480 					if (arguments.length > 1) {
481 						future.completeExceptionally(new RuntimeException((String) arguments[1]));
482 						return null;
483 					}
484 					future.complete((String) arguments[0]);
