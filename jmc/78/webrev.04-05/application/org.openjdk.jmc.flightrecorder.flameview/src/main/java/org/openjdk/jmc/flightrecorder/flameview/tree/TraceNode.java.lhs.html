 34 package org.openjdk.jmc.flightrecorder.flameview.tree;
 35 
 36 import java.util.ArrayList;
 37 import java.util.List;
 38 
 39 public class TraceNode {
 40 	public static final TraceNode EMPTY = new TraceNode(null, 0, null);
 41 	private final int value;
 42 	private final String name;
 43 	private final String packageName;
 44 	private final List<TraceNode> children = new ArrayList<>();

 45 
 46 	public TraceNode(String name, int value, String packageName) {
 47 		this.name = name;
 48 		this.value = value;
 49 		this.packageName = packageName;
 50 	}
 51 
 52 	public int getValue() {
 53 		return value;
 54 	}
 55 
 56 	public String getName() {
 57 		return name;
 58 	}
 59 
 60 	public String getPackageName() {
 61 		return packageName;
 62 	}
 63 
 64 	public List<TraceNode> getChildren() {
 65 		return children;
 66 	}
 67 
 68 	public void addChild(TraceNode child) {
 69 		children.add(child);
 70 	}
 71 








 72 	@Override
 73 	public int hashCode() {
 74 		final int prime = 31;
 75 		int result = 1;
 76 		result = prime * result + ((children == null) ? 0 : children.hashCode());
 77 		result = prime * result + ((name == null) ? 0 : name.hashCode());
 78 		result = prime * result + ((packageName == null) ? 0 : packageName.hashCode());

 79 		result = prime * result + value;
 80 		return result;
 81 	}
 82 
 83 	@Override
 84 	public boolean equals(Object obj) {
 85 		if (this == obj)
 86 			return true;
 87 		if (obj == null)
 88 			return false;
 89 		if (getClass() != obj.getClass())
 90 			return false;
 91 		TraceNode other = (TraceNode) obj;
 92 		if (children == null) {
 93 			if (other.children != null)
 94 				return false;
 95 		} else if (!children.equals(other.children))
 96 			return false;
 97 		if (name == null) {
 98 			if (other.name != null)
 99 				return false;
100 		} else if (!name.equals(other.name))
101 			return false;
102 		if (packageName == null) {
103 			if (other.packageName != null) {
104 				return false;
105 			}
106 		} else if (!packageName.equals(other.packageName))
107 			return false;


108 		if (value != other.value)
109 			return false;
110 		return true;
111 	}
112 
113 	public String toString() {
114 		return "TraceNode [name: " + name + ", value: " + value + ", packageName: " + packageName + ", children: "
115 				+ children.size() + "]";
116 	}
117 }
