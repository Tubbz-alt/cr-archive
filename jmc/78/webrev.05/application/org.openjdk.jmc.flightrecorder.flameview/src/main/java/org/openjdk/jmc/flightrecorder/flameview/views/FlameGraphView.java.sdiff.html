345 	private void rebuildModel(IItemCollection items) {
346 		// Release old model before building the new
347 		if (currentModelCalculator != null) {
348 			currentModelCalculator.cancel(true);
349 		}
350 		currentModelCalculator = getModelPreparer(items, frameSeparator, true);

351 		currentModelCalculator.thenAcceptAsync(this::setModel, DisplayToolkit.inDisplayThread())
352 				.exceptionally(FlameGraphView::handleModelBuildException);
353 	}

355 	private CompletableFuture<TraceNode> getModelPreparer(
356 		final IItemCollection items, final FrameSeparator separator, final boolean materializeSelectedBranches) {





357 		return CompletableFuture.supplyAsync(() -> {
358 			StacktraceModel model = new StacktraceModel(threadRootAtTop, frameSeparator, items);
359 			TraceNode root = TraceTreeUtils.createRootWithDescription(items, model.getRootFork().getBranchCount());
360 			return TraceTreeUtils.createTree(root, model);












361 		}, MODEL_EXECUTOR);
362 	}

391 	private void saveFlameGraph() {
392 		CompletableFuture<String> future = new CompletableFuture<>();
393 		String[] destination = new String[2];
394 
395 		DisplayToolkit.inDisplayThread().execute(() -> {
396 			FileDialog fd = new FileDialog(browser.getShell(), SWT.SAVE);
397 			fd.setText(getFlameviewMessage(FLAMEVIEW_SAVE_FLAME_GRAPH_AS));
398 			fd.setFilterNames(
399 					new String[] {getFlameviewMessage(FLAMEVIEW_JPEG_IMAGE), getFlameviewMessage(FLAMEVIEW_PNG_IMAGE)});
400 			fd.setFilterExtensions(new String[] {"*.jpg", "*.png"}); //$NON-NLS-1$ //$NON-NLS-2$
401 			fd.setFileName("flame_graph"); //$NON-NLS-1$
402 			fd.setOverwrite(true);
403 			if (fd.open() == null) {
404 				future.cancel(true);
405 				return;
406 			}
407 
408 			String type;
409 			String fileName = fd.getFileName().toLowerCase();
410 			// FIXME: FileDialog filterIndex returns -1 (https://bugs.eclipse.org/bugs/show_bug.cgi?id=546256)

411 			if (fileName.endsWith(".jpg") || fileName.endsWith(".jpeg")) { //$NON-NLS-1$ //$NON-NLS-2$
412 				type = "image/jpeg"; //$NON-NLS-1$
413 			} else if (fileName.endsWith(".png")) { //$NON-NLS-1$
414 				type = "image/png"; //$NON-NLS-1$
415 			} else {
416 				future.completeExceptionally(new UnsupportedOperationException("Unsupported image format")); //$NON-NLS-1$
417 				return;
418 			}
419 			destination[0] = fd.getFilterPath();
420 			destination[1] = fd.getFileName();
421 
422 			String callback = "_saveFlameGraphCallback"; //$NON-NLS-1$
423 			new BrowserFunction(browser, callback) {
424 				@Override
425 				public Object function(Object[] arguments) {
426 					if (arguments.length > 1) {
427 						future.completeExceptionally(new RuntimeException((String) arguments[1]));
428 						return null;
429 					}
430 					future.complete((String) arguments[0]);

440 		try {
441 			String b64 = future.get();
442 			byte[] bytes = Base64.getDecoder().decode(b64);
443 			FileOutputStream fos = new FileOutputStream(new File(destination[0], destination[1]));
444 			fos.write(bytes);
445 			fos.close();
446 		} catch (CancellationException e) {
447 			// noop
448 		} catch (InterruptedException | ExecutionException | IOException e) {
449 			FlightRecorderUI.getDefault().getLogger().log(Level.SEVERE, "Failed to save flame graph", e); //$NON-NLS-1$
450 		}
451 	}

467 	private static String render(TraceNode root) {
468 		StringBuilder builder = new StringBuilder();
469 		String rootNodeStart = createJsonRootTraceNode(root);
470 		builder.append(rootNodeStart);
471 		renderChildren(builder, root);

472 		builder.append("]}");
473 		return builder.toString();
474 	}

476 	private static void render(StringBuilder builder, TraceNode node) {
477 		String start = UNCLASSIFIABLE_FRAME.equals(node.getName()) ? createJsonDescTraceNode(node)
478 				: createJsonTraceNode(node);
479 		builder.append(start);
480 		renderChildren(builder, node);
481 		builder.append("]}");
482 	}

484 	private static void renderChildren(StringBuilder builder, TraceNode node) {
485 		for (int i = 0; i < node.getChildren().size(); i++) {
486 			render(builder, node.getChildren().get(i));







487 			if (i < node.getChildren().size() - 1) {
488 				builder.append(",");
489 			}

490 		}
491 	}

413 	private void rebuildModel() {
414 		if (modelCalculationActive.get()) {
415 			modelCalculationActive.set(false);

416 		}
417 		final CompletableFuture<ModelsContainer> currentModelCalculator = getModelPreparer(createStacktraceModel(),
418 				true);
419 		currentModelCalculator.thenAcceptAsync(this::setModel, DisplayToolkit.inDisplayThread())
420 				.exceptionally(FlameGraphView::handleModelBuildException);
421 	}

427 	private CompletableFuture<ModelsContainer> getModelPreparer(
428 		final StacktraceModel model, final boolean materializeSelectedBranches) {
429 		modelCalculationActive.set(true);
430 		return CompletableFuture.supplyAsync(() -> {
431 			Fork rootFork = model.getRootFork();
432 			if (materializeSelectedBranches) {
433 				Branch selectedBranch = getLastSelectedBranch(rootFork);
434 				if (selectedBranch != null) {
435 					selectedBranch.getEndFork();
436 				}
437 			}
438 
439 			if (modelCalculationActive.get()) {
440 				TraceNode root = TraceTreeUtils.createRootWithDescription(currentItems, rootFork.getBranchCount());
441 				TraceNode traceNode = TraceTreeUtils.createTree(modelCalculationActive, root, model);
442 				return traceNode.isCanceled() ? ModelsContainer.EMPTY : new ModelsContainer(traceNode, model);
443 			} else {
444 				return ModelsContainer.EMPTY;
445 			}
446 		}, MODEL_EXECUTOR);
447 	}

481 	private void saveFlameGraph() {
482 		CompletableFuture<String> future = new CompletableFuture<>();
483 		String[] destination = new String[2];
484 
485 		DisplayToolkit.inDisplayThread().execute(() -> {
486 			FileDialog fd = new FileDialog(browser.getShell(), SWT.SAVE);
487 			fd.setText(getFlameviewMessage(FLAMEVIEW_SAVE_FLAME_GRAPH_AS));
488 			fd.setFilterNames(
489 					new String[] {getFlameviewMessage(FLAMEVIEW_JPEG_IMAGE), getFlameviewMessage(FLAMEVIEW_PNG_IMAGE)});
490 			fd.setFilterExtensions(new String[] {"*.jpg", "*.png"}); //$NON-NLS-1$ //$NON-NLS-2$
491 			fd.setFileName("flame_graph"); //$NON-NLS-1$
492 			fd.setOverwrite(true);
493 			if (fd.open() == null) {
494 				future.cancel(true);
495 				return;
496 			}
497 
498 			String type;
499 			String fileName = fd.getFileName().toLowerCase();
500 			// FIXME: FileDialog filterIndex returns -1
501 			// (https://bugs.eclipse.org/bugs/show_bug.cgi?id=546256)
502 			if (fileName.endsWith(".jpg") || fileName.endsWith(".jpeg")) { //$NON-NLS-1$ //$NON-NLS-2$
503 				type = "image/jpeg"; //$NON-NLS-1$
504 			} else if (fileName.endsWith(".png")) { //$NON-NLS-1$
505 				type = "image/png"; //$NON-NLS-1$
506 			} else {
507 				future.completeExceptionally(new UnsupportedOperationException("Unsupported image format")); //$NON-NLS-1$
508 				return;
509 			}
510 			destination[0] = fd.getFilterPath();
511 			destination[1] = fd.getFileName();
512 
513 			String callback = "_saveFlameGraphCallback"; //$NON-NLS-1$
514 			new BrowserFunction(browser, callback) {
515 				@Override
516 				public Object function(Object[] arguments) {
517 					if (arguments.length > 1) {
518 						future.completeExceptionally(new RuntimeException((String) arguments[1]));
519 						return null;
520 					}
521 					future.complete((String) arguments[0]);

531 		try {
532 			String b64 = future.get();
533 			byte[] bytes = Base64.getDecoder().decode(b64);
534 			FileOutputStream fos = new FileOutputStream(new File(destination[0], destination[1]));
535 			fos.write(bytes);
536 			fos.close();
537 		} catch (CancellationException e) {
538 			// noop
539 		} catch (InterruptedException | ExecutionException | IOException e) {
540 			FlightRecorderUI.getDefault().getLogger().log(Level.SEVERE, "Failed to save flame graph", e); //$NON-NLS-1$
541 		}
542 	}

558 	private JSonModelBuilder render(TraceNode root) {
559 		JSonModelBuilder builder = new JSonModelBuilder();
560 		String rootNodeStart = createJsonRootTraceNode(root);
561 		builder.append(rootNodeStart);
562 		AtomicBoolean renderActive = new AtomicBoolean(true);
563 		renderChildren(renderActive, builder, root);
564 		builder.append("]}");
565 		return builder;
566 	}
567 
568 	private void render(AtomicBoolean renderActive, JSonModelBuilder builder, TraceNode node) {
569 		String start = UNCLASSIFIABLE_FRAME.equals(node.getName()) ? createJsonDescTraceNode(node)
570 				: createJsonTraceNode(node);
571 		builder.append(start);
572 		renderChildren(renderActive, builder, node);
573 		builder.append("]}");
574 	}
575 
576 	private void renderChildren(AtomicBoolean renderActive, JSonModelBuilder builder, TraceNode node) {
577 
578 		int i = 0;
579 		while (renderActive.get() && i < node.getChildren().size()) {
580 			if (!modelCalculationActive.get()) {
581 				renderActive.set(false);
582 				builder.setCanceled();
583 			}
584 
585 			render(renderActive, builder, node.getChildren().get(i));
586 			if (i < node.getChildren().size() - 1) {
587 				builder.append(",");
588 			}
589 			i++;
590 		}
591 	}
