345 	private void rebuildModel(IItemCollection items) {
346 		// Release old model before building the new
347 		if (currentModelCalculator != null) {
348 			currentModelCalculator.cancel(true);
349 		}
350 		currentModelCalculator = getModelPreparer(items, frameSeparator, true);

351 		currentModelCalculator.thenAcceptAsync(this::setModel, DisplayToolkit.inDisplayThread())
352 				.exceptionally(FlameGraphView::handleModelBuildException);
353 	}
354 
355 	private CompletableFuture<TraceNode> getModelPreparer(
356 		final IItemCollection items, final FrameSeparator separator, final boolean materializeSelectedBranches) {




357 		return CompletableFuture.supplyAsync(() -> {
358 			StacktraceModel model = new StacktraceModel(threadRootAtTop, frameSeparator, items);
359 			TraceNode root = TraceTreeUtils.createRootWithDescription(items, model.getRootFork().getBranchCount());
360 			return TraceTreeUtils.createTree(root, model);









361 		}, MODEL_EXECUTOR);
362 	}

391 	private void saveFlameGraph() {
392 		CompletableFuture<String> future = new CompletableFuture<>();
393 		String[] destination = new String[2];
394 
395 		DisplayToolkit.inDisplayThread().execute(() -> {
396 			FileDialog fd = new FileDialog(browser.getShell(), SWT.SAVE);
397 			fd.setText(getFlameviewMessage(FLAMEVIEW_SAVE_FLAME_GRAPH_AS));
398 			fd.setFilterNames(
399 					new String[] {getFlameviewMessage(FLAMEVIEW_JPEG_IMAGE), getFlameviewMessage(FLAMEVIEW_PNG_IMAGE)});
400 			fd.setFilterExtensions(new String[] {"*.jpg", "*.png"}); //$NON-NLS-1$ //$NON-NLS-2$
401 			fd.setFileName("flame_graph"); //$NON-NLS-1$
402 			fd.setOverwrite(true);
403 			if (fd.open() == null) {
404 				future.cancel(true);
405 				return;
406 			}
407 
408 			String type;
409 			String fileName = fd.getFileName().toLowerCase();
410 			// FIXME: FileDialog filterIndex returns -1 (https://bugs.eclipse.org/bugs/show_bug.cgi?id=546256)

411 			if (fileName.endsWith(".jpg") || fileName.endsWith(".jpeg")) { //$NON-NLS-1$ //$NON-NLS-2$
412 				type = "image/jpeg"; //$NON-NLS-1$
413 			} else if (fileName.endsWith(".png")) { //$NON-NLS-1$
414 				type = "image/png"; //$NON-NLS-1$
415 			} else {
416 				future.completeExceptionally(new UnsupportedOperationException("Unsupported image format")); //$NON-NLS-1$
417 				return;
418 			}
419 			destination[0] = fd.getFilterPath();
420 			destination[1] = fd.getFileName();
421 
422 			String callback = "_saveFlameGraphCallback"; //$NON-NLS-1$
423 			new BrowserFunction(browser, callback) {
424 				@Override
425 				public Object function(Object[] arguments) {
426 					if (arguments.length > 1) {
427 						future.completeExceptionally(new RuntimeException((String) arguments[1]));
428 						return null;
429 					}
430 					future.complete((String) arguments[0]);

378 	private void rebuildModel() {
379 		// try to cancel downstream calculation
380 		if (currentModelCalculator != null) {
381 			currentModelCalculator.cancel(true);
382 		}
383 
384 		currentModelCalculator = getModelPreparer(frameSeparator, true);
385 		currentModelCalculator.thenAcceptAsync(this::setModel, DisplayToolkit.inDisplayThread())
386 				.exceptionally(FlameGraphView::handleModelBuildException);
387 	}

393 	private CompletableFuture<ModelsContainer> getModelPreparer(
394 		final FrameSeparator separator, final boolean materializeSelectedBranches) {
395 		return CompletableFuture.supplyAsync(() -> {
396 			StacktraceModel model = createStacktraceModel();
397 			Fork rootFork = model.getRootFork();
398 			if (materializeSelectedBranches) {
399 				Branch selectedBranch = getLastSelectedBranch(rootFork);
400 				if (selectedBranch != null) {
401 					selectedBranch.getEndFork();
402 				}
403 			}
404 
405 			TraceNode root = TraceTreeUtils.createRootWithDescription(currentItems, rootFork.getBranchCount());
406 			return new ModelsContainer(TraceTreeUtils.createTree(root, model), model);
407 
408 		}, MODEL_EXECUTOR);
409 	}

439 	private void saveFlameGraph() {
440 		CompletableFuture<String> future = new CompletableFuture<>();
441 		String[] destination = new String[2];
442 
443 		DisplayToolkit.inDisplayThread().execute(() -> {
444 			FileDialog fd = new FileDialog(browser.getShell(), SWT.SAVE);
445 			fd.setText(getFlameviewMessage(FLAMEVIEW_SAVE_FLAME_GRAPH_AS));
446 			fd.setFilterNames(
447 					new String[] {getFlameviewMessage(FLAMEVIEW_JPEG_IMAGE), getFlameviewMessage(FLAMEVIEW_PNG_IMAGE)});
448 			fd.setFilterExtensions(new String[] {"*.jpg", "*.png"}); //$NON-NLS-1$ //$NON-NLS-2$
449 			fd.setFileName("flame_graph"); //$NON-NLS-1$
450 			fd.setOverwrite(true);
451 			if (fd.open() == null) {
452 				future.cancel(true);
453 				return;
454 			}
455 
456 			String type;
457 			String fileName = fd.getFileName().toLowerCase();
458 			// FIXME: FileDialog filterIndex returns -1
459 			// (https://bugs.eclipse.org/bugs/show_bug.cgi?id=546256)
460 			if (fileName.endsWith(".jpg") || fileName.endsWith(".jpeg")) { //$NON-NLS-1$ //$NON-NLS-2$
461 				type = "image/jpeg"; //$NON-NLS-1$
462 			} else if (fileName.endsWith(".png")) { //$NON-NLS-1$
463 				type = "image/png"; //$NON-NLS-1$
464 			} else {
465 				future.completeExceptionally(new UnsupportedOperationException("Unsupported image format")); //$NON-NLS-1$
466 				return;
467 			}
468 			destination[0] = fd.getFilterPath();
469 			destination[1] = fd.getFileName();
470 
471 			String callback = "_saveFlameGraphCallback"; //$NON-NLS-1$
472 			new BrowserFunction(browser, callback) {
473 				@Override
474 				public Object function(Object[] arguments) {
475 					if (arguments.length > 1) {
476 						future.completeExceptionally(new RuntimeException((String) arguments[1]));
477 						return null;
478 					}
479 					future.complete((String) arguments[0]);
