 34 package org.openjdk.jmc.flightrecorder.flameview.tree;
 35 
 36 import java.util.ArrayList;
 37 import java.util.List;
 38 
 39 public class TraceNode {

 40 	private final int value;
 41 	private final String name;
 42 	private final String packageName;
 43 	private final List<TraceNode> children = new ArrayList<>();
 44 
 45 	public TraceNode(String name, int value, String packageName) {
 46 		this.name = name;
 47 		this.value = value;
 48 		this.packageName = packageName;
 49 	}
 50 
 51 	public int getValue() {
 52 		return value;
 53 	}
 54 
 55 	public String getName() {
 56 		return name;
 57 	}
 58 
 59 	public String getPackageName() {
 60 		return packageName;
 61 	}
 62 
 63 	public List<TraceNode> getChildren() {
 64 		return children;
 65 	}
 66 
 67 	public void addChild(TraceNode child) {
 68 		children.add(child);
 69 	}
 70 
 71 	@Override
 72 	public int hashCode() {
 73 		final int prime = 31;
 74 		int result = 1;
 75 		result = prime * result + ((children == null) ? 0 : children.hashCode());
 76 		result = prime * result + ((name == null) ? 0 : name.hashCode());
 77 		result = prime * result + ((packageName == null) ? 0 : packageName.hashCode());
 78 		result = prime * result + value;
 79 		return result;
 80 	}
 81 
 82 	@Override
 83 	public boolean equals(Object obj) {
 84 		if (this == obj)
 85 			return true;
 86 		if (obj == null)
 87 			return false;
 88 		if (getClass() != obj.getClass())
 89 			return false;
 90 		TraceNode other = (TraceNode) obj;
 91 		if (children == null) {
 92 			if (other.children != null)
 93 				return false;
 94 		} else if (!children.equals(other.children))
 95 			return false;
 96 		if (name == null) {
 97 			if (other.name != null)
 98 				return false;
 99 		} else if (!name.equals(other.name))
100 			return false;
101 		if (packageName == null) {
102 			if (other.packageName != null) {
103 				return false;
104 			}
105 		} else if (!packageName.equals(other.packageName))
106 			return false;
107 		if (value != other.value)
108 			return false;
109 		return true;
110 	}
111 
112 	public String toString() {
113 		return "TraceNode [name: " + name + ", value: " + value + ", packageName: " + packageName + ", children: "
114 				+ children.size() + "]";
115 	}
116 }
