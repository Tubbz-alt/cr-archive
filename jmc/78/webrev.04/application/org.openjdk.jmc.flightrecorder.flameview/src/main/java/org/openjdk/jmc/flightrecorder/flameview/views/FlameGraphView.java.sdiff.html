345 	private void rebuildModel(IItemCollection items) {
346 		// Release old model before building the new
347 		if (currentModelCalculator != null) {
348 			currentModelCalculator.cancel(true);
349 		}
350 		currentModelCalculator = getModelPreparer(items, frameSeparator, true);

351 		currentModelCalculator.thenAcceptAsync(this::setModel, DisplayToolkit.inDisplayThread())
352 				.exceptionally(FlameGraphView::handleModelBuildException);
353 	}

355 	private CompletableFuture<TraceNode> getModelPreparer(
356 		final IItemCollection items, final FrameSeparator separator, final boolean materializeSelectedBranches) {




357 		return CompletableFuture.supplyAsync(() -> {
358 			StacktraceModel model = new StacktraceModel(threadRootAtTop, frameSeparator, items);
359 			TraceNode root = TraceTreeUtils.createRootWithDescription(items, model.getRootFork().getBranchCount());
360 			return TraceTreeUtils.createTree(root, model);













361 		}, MODEL_EXECUTOR);
362 	}

391 	private void saveFlameGraph() {
392 		CompletableFuture<String> future = new CompletableFuture<>();
393 		String[] destination = new String[2];
394 
395 		DisplayToolkit.inDisplayThread().execute(() -> {
396 			FileDialog fd = new FileDialog(browser.getShell(), SWT.SAVE);
397 			fd.setText(getFlameviewMessage(FLAMEVIEW_SAVE_FLAME_GRAPH_AS));
398 			fd.setFilterNames(
399 					new String[] {getFlameviewMessage(FLAMEVIEW_JPEG_IMAGE), getFlameviewMessage(FLAMEVIEW_PNG_IMAGE)});
400 			fd.setFilterExtensions(new String[] {"*.jpg", "*.png"}); //$NON-NLS-1$ //$NON-NLS-2$
401 			fd.setFileName("flame_graph"); //$NON-NLS-1$
402 			fd.setOverwrite(true);
403 			if (fd.open() == null) {
404 				future.cancel(true);
405 				return;
406 			}
407 
408 			String type;
409 			String fileName = fd.getFileName().toLowerCase();
410 			// FIXME: FileDialog filterIndex returns -1 (https://bugs.eclipse.org/bugs/show_bug.cgi?id=546256)

411 			if (fileName.endsWith(".jpg") || fileName.endsWith(".jpeg")) { //$NON-NLS-1$ //$NON-NLS-2$
412 				type = "image/jpeg"; //$NON-NLS-1$
413 			} else if (fileName.endsWith(".png")) { //$NON-NLS-1$
414 				type = "image/png"; //$NON-NLS-1$
415 			} else {
416 				future.completeExceptionally(new UnsupportedOperationException("Unsupported image format")); //$NON-NLS-1$
417 				return;
418 			}
419 			destination[0] = fd.getFilterPath();
420 			destination[1] = fd.getFileName();
421 
422 			String callback = "_saveFlameGraphCallback"; //$NON-NLS-1$
423 			new BrowserFunction(browser, callback) {
424 				@Override
425 				public Object function(Object[] arguments) {
426 					if (arguments.length > 1) {
427 						future.completeExceptionally(new RuntimeException((String) arguments[1]));
428 						return null;
429 					}
430 					future.complete((String) arguments[0]);

440 		try {
441 			String b64 = future.get();
442 			byte[] bytes = Base64.getDecoder().decode(b64);
443 			FileOutputStream fos = new FileOutputStream(new File(destination[0], destination[1]));
444 			fos.write(bytes);
445 			fos.close();
446 		} catch (CancellationException e) {
447 			// noop
448 		} catch (InterruptedException | ExecutionException | IOException e) {
449 			FlightRecorderUI.getDefault().getLogger().log(Level.SEVERE, "Failed to save flame graph", e); //$NON-NLS-1$
450 		}
451 	}

467 	private static String render(TraceNode root) {
468 		StringBuilder builder = new StringBuilder();
469 		String rootNodeStart = createJsonRootTraceNode(root);
470 		builder.append(rootNodeStart);
471 		renderChildren(builder, root);

472 		builder.append("]}");
473 		return builder.toString();

476 	private static void render(StringBuilder builder, TraceNode node) {
477 		String start = UNCLASSIFIABLE_FRAME.equals(node.getName()) ? createJsonDescTraceNode(node)
478 				: createJsonTraceNode(node);
479 		builder.append(start);
480 		renderChildren(builder, node);
481 		builder.append("]}");
482 	}

484 	private static void renderChildren(StringBuilder builder, TraceNode node) {
485 		for (int i = 0; i < node.getChildren().size(); i++) {
486 			render(builder, node.getChildren().get(i));
487 			if (i < node.getChildren().size() - 1) {
488 				builder.append(",");






489 			}

490 		}







491 	}

417 		if (modelCalculationActive.get()) {
418 			modelCalculationActive.set(false);

419 		}
420 		final CompletableFuture<ModelsContainer> currentModelCalculator = getModelPreparer(createStacktraceModel(),
421 				true);

423 				.exceptionally(FlameGraphView::handleModelBuildException);

430 	private CompletableFuture<ModelsContainer> getModelPreparer(
431 		final StacktraceModel model, final boolean materializeSelectedBranches) {
432 		return CompletableFuture.supplyAsync(() -> {
433 			modelCalculationActive.set(true);
434 			Fork rootFork = model.getRootFork();
435 			if (materializeSelectedBranches) {
436 				Branch selectedBranch = getLastSelectedBranch(rootFork);
437 				if (selectedBranch != null) {
438 					selectedBranch.getEndFork();
439 				}
440 			}
441 
442 			if (modelCalculationActive.get()) {
443 				TraceNode root = TraceTreeUtils.createRootWithDescription(currentItems, rootFork.getBranchCount());
444 				return new ModelsContainer(TraceTreeUtils.createTree(root, model), model);
445 			} else {
446 				System.out.println("FLAME, getModelPreparer CANCEL");
447 				return ModelsContainer.EMPTY;
448 			}
449 		}, MODEL_EXECUTOR);
450 	}

486 	private void saveFlameGraph() {
487 		CompletableFuture<String> future = new CompletableFuture<>();
488 		String[] destination = new String[2];
489 
490 		DisplayToolkit.inDisplayThread().execute(() -> {
491 			FileDialog fd = new FileDialog(browser.getShell(), SWT.SAVE);
492 			fd.setText(getFlameviewMessage(FLAMEVIEW_SAVE_FLAME_GRAPH_AS));
493 			fd.setFilterNames(
494 					new String[] {getFlameviewMessage(FLAMEVIEW_JPEG_IMAGE), getFlameviewMessage(FLAMEVIEW_PNG_IMAGE)});
495 			fd.setFilterExtensions(new String[] {"*.jpg", "*.png"}); //$NON-NLS-1$ //$NON-NLS-2$
496 			fd.setFileName("flame_graph"); //$NON-NLS-1$
497 			fd.setOverwrite(true);
498 			if (fd.open() == null) {
499 				future.cancel(true);
500 				return;
501 			}
502 
503 			String type;
504 			String fileName = fd.getFileName().toLowerCase();
505 			// FIXME: FileDialog filterIndex returns -1
506 			// (https://bugs.eclipse.org/bugs/show_bug.cgi?id=546256)
507 			if (fileName.endsWith(".jpg") || fileName.endsWith(".jpeg")) { //$NON-NLS-1$ //$NON-NLS-2$
508 				type = "image/jpeg"; //$NON-NLS-1$
509 			} else if (fileName.endsWith(".png")) { //$NON-NLS-1$
510 				type = "image/png"; //$NON-NLS-1$
511 			} else {
512 				future.completeExceptionally(new UnsupportedOperationException("Unsupported image format")); //$NON-NLS-1$
513 				return;
514 			}
515 			destination[0] = fd.getFilterPath();
516 			destination[1] = fd.getFileName();
517 
518 			String callback = "_saveFlameGraphCallback"; //$NON-NLS-1$
519 			new BrowserFunction(browser, callback) {
520 				@Override
521 				public Object function(Object[] arguments) {
522 					if (arguments.length > 1) {
523 						future.completeExceptionally(new RuntimeException((String) arguments[1]));
524 						return null;
525 					}
526 					future.complete((String) arguments[0]);

536 		try {
537 			String b64 = future.get();
538 			byte[] bytes = Base64.getDecoder().decode(b64);
539 			FileOutputStream fos = new FileOutputStream(new File(destination[0], destination[1]));
540 			fos.write(bytes);
541 			fos.close();
542 		} catch (CancellationException e) {
543 			// noop
544 		} catch (InterruptedException | ExecutionException | IOException e) {
545 			FlightRecorderUI.getDefault().getLogger().log(Level.SEVERE, "Failed to save flame graph", e); //$NON-NLS-1$
546 		}
547 	}

563 	private JSonModelBuilder render(TraceNode root) {
564 		JSonModelBuilder builder = new JSonModelBuilder();
565 		String rootNodeStart = createJsonRootTraceNode(root);
566 		builder.append(rootNodeStart);
567 		AtomicBoolean renderActive = new AtomicBoolean(true);
568 		renderChildren(renderActive, builder, root);
569 		builder.append("]}");
570 		if (renderActive.get()) {
571 			builder.setReady();
572 		}
573 		return builder;
574 	}

577 		String start = UNCLASSIFIABLE_FRAME.equals(node.getName()) ? createJsonDescTraceNode(node)
578 				: createJsonTraceNode(node);
579 		builder.append(start);
580 		renderChildren(renderActive, builder, node);
581 		builder.append("]}");
582 	}

652 			return "";
653 		} else {
654 			ByteArrayOutputStream baos = new ByteArrayOutputStream();
655 			ImageLoader loader = new ImageLoader();
656 			loader.data = new ImageData[] {image.getImageData()};
