155 		if (prefix != null && TypeUtils.isValidJavaIdentifier(prefix)) {
156 			return prefix;
157 		}
158 		return DEFAULT_CLASS_PREFIX;

166 					"Could not find an event name, generated one: " + eventName); //$NON-NLS-1$
