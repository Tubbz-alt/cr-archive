 33 package org.openjdk.jmc.agent;
 34 
 35 import java.util.List;
 36 import java.util.Map;
 37 
 38 import org.openjdk.jmc.agent.jfr.JFRTransformDescriptor;
 39 
 40 /**
 41  * General metadata describing a transform to take place for a method.
 42  */
 43 public abstract class TransformDescriptor {
 44 	public static final String ATTRIBUTE_CLASS_PREFIX = "classprefix"; //$NON-NLS-1$
 45 	public static final String ATTRIBUTE_ALLOW_TO_STRING = "allowtostring"; //$NON-NLS-1$
 46 	public static final String ATTRIBUTE_ALLOW_CONVERTER = "allowconverter"; //$NON-NLS-1$
 47 	public static final String ATTRIBUTE_EMIT_ON_EXCEPTION = "emitonexception"; //$NON-NLS-1$
 48 
 49 	public static final String DEFAULT_CLASS_PREFIX = "__JFREvent"; //$NON-NLS-1$
 50 
 51 	private final String id;
 52 	private final String className;
 53 	private final Method method;
 54 	private final Map<String, String> transformationAttributes;
 55 	private volatile boolean pendingTransforms = true;
 56 
 57 	public TransformDescriptor(String id, String className, Method method,
 58 			Map<String, String> transformationAttributes) {
 59 		this.id = id;
 60 		this.className = className;
 61 		this.method = method;
 62 		this.transformationAttributes = transformationAttributes;
 63 	}
 64 
 65 	public String getId() {
 66 		return id;
 67 	}
 68 
 69 	public String getClassName() {
 70 		return className;
 71 	}
 72 
 73 	public Method getMethod() {
 74 		return method;
 75 	}
 76 
 77 	public Map<String, String> getTransformationAttributes() {
 78 		return transformationAttributes;
 79 	}
 80 
 81 	public boolean isPendingTransforms() {
 82 		return pendingTransforms;
 83 	}
 84 
 85 	public void setPendingTransforms(boolean hasPendingTransforms) {
 86 		this.pendingTransforms = hasPendingTransforms;
 87 	}
 88 
 89 	protected String getTransformationAttribute(String attribute) {
 90 		return transformationAttributes.get(attribute);
 91 	}
 92 
 93 	/**
 94 	 * Factory method for creating {@link TransformDescriptor} instances.
 95 	 *
 96 	 * @param id
 97 	 *            transform id
 98 	 * @param internalName
 99 	 *            the class name in VM internal form.
100 	 * @param method
101 	 *            the method (see {@link Method})
102 	 * @param values
103 	 *            the values describing the transform.
104 	 * @param parameters
105 	 *            the parameters to include (see {@link Parameter}).
106 	 * @return the instantiated {@link TransformDescriptor}.
107 	 */
108 	public static TransformDescriptor create(
109 		String id, String internalName, Method method, Map<String, String> values, List<Parameter> parameters, ReturnValue returnValue, List<Field> fields) {
110 		return new JFRTransformDescriptor(id, internalName, method, values, parameters, returnValue, fields);
111 	}
112 
113 	@Override
114 	public String toString() {
115 		return String.format("TransformDescriptor [method:%s]", method.toString()); //$NON-NLS-1$
116 	}
117 }
