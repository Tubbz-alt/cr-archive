121 	 * Loads the configuration from the file specified in the agentArguments, and initializes the
122 	 * agent.
123 	 *
124 	 * @param agentArguments
125 	 *            the file to load from.
126 	 * @param instrumentation
127 	 *            the {@link Instrumentation} instance.
128 	 */
129 	private static void initializeAgent(String agentArguments, Instrumentation instrumentation) {
130 		if (agentArguments == null || agentArguments.trim().length() == 0) {
131 			try {
132 				initializeAgent((InputStream) null, instrumentation);
133 			} catch (XMLStreamException e) {
134 				// noop
135 			}
136 			return;
137 		}
138 
139 		File file = new File(agentArguments);
140 		try (InputStream stream = new FileInputStream(file)) {
141 			initializeAgent(stream, instrumentation);
142 		} catch (XMLStreamException | IOException e) {
143 			getLogger().log(Level.SEVERE, "Failed to read jfr probe definitions from " + file.getPath(), e); //$NON-NLS-1$
144 		}
145 	}
