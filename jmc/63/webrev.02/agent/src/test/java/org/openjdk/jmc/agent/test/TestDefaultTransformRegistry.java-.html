 33 package org.openjdk.jmc.agent.test;
 34 
 35 import static org.junit.Assert.assertEquals;
 36 import static org.junit.Assert.assertNotNull;
 37 import static org.junit.Assert.assertNull;
 38 import static org.junit.Assert.assertTrue;
 39 
 40 import java.io.IOException;
 41 import java.util.List;
 42 
 43 import javax.xml.stream.XMLStreamException;
 44 
 45 import org.junit.Test;
 46 import org.objectweb.asm.Type;
 47 import org.openjdk.jmc.agent.TransformDescriptor;
 48 import org.openjdk.jmc.agent.TransformRegistry;
 49 import org.openjdk.jmc.agent.impl.DefaultTransformRegistry;
 50 import org.openjdk.jmc.agent.test.util.TestToolkit;
 51 
 52 public class TestDefaultTransformRegistry {
 53 
 54 	private static final String XML_EVENT_DESCRIPTION = "<event id=\"demo.jfr.test1\">"
 55 			+ "<name>JFR Hello World Event 1 Modify </name>"
 56 			+ "<description>Defined in the xml file and added by the agent.</description>"
 57 			+ "<path>demo/jfrhelloworldevent1</path>"
 58 			+ "<stacktrace>true</stacktrace>"
 59 			+ "<class>org.openjdk.jmc.agent.test.InstrumentMe</class>"
 60 			+ "<method>"
 61 			+ "<name>printHelloWorldJFR1</name>"
 62 			+ "<descriptor>()V</descriptor>"
 63 			+ "</method>"
 64 			+ "<location>WRAP</location>"
 65 			+ "</event>";
 66 	
 67 	public static String getTemplate() throws IOException {
 68 		return TestToolkit.readTemplate(TestDefaultTransformRegistry.class, TestToolkit.DEFAULT_TEMPLATE_NAME);
 69 	}
 70 	
 71 	@Test
 72 	public void testHasPendingTransforms() throws XMLStreamException, IOException {
 73 		TransformRegistry registry = DefaultTransformRegistry
 74 				.from(TestToolkit.getProbesXMLFromTemplate(getTemplate(), "HasPendingTransforms")); //$NON-NLS-1$
 75 		assertNotNull(registry);
 76 		assertTrue(registry.hasPendingTransforms(Type.getInternalName(InstrumentMe.class)));
 77 	}
 78 
 79 	@Test
 80 	public void testFrom() throws XMLStreamException, IOException {
 81 		TransformRegistry registry = DefaultTransformRegistry
 82 				.from(TestToolkit.getProbesXMLFromTemplate(getTemplate(), "From")); //$NON-NLS-1$
 83 		assertNotNull(registry);
 84 	}
 85 
 86 	@Test
 87 	public void testGetTransformData() throws XMLStreamException, IOException {
 88 		TransformRegistry registry = DefaultTransformRegistry
 89 				.from(TestToolkit.getProbesXMLFromTemplate(getTemplate(), "GetTransformData")); //$NON-NLS-1$
 90 		assertNotNull(registry);
 91 		List<TransformDescriptor> transformData = registry.getTransformData(Type.getInternalName(InstrumentMe.class));
 92 		assertNotNull(transformData);
 93 		assertTrue(transformData.size() > 0);
 94 	}
 95 
 96 	@Test
 97 	public void testModify() throws XMLStreamException, IOException {
 98 		TransformRegistry registry = DefaultTransformRegistry
 99 				.from(TestToolkit.getProbesXMLFromTemplate(getTemplate(), "Modify")); //$NON-NLS-1$
100 		assertNotNull(registry);
101 		List<TransformDescriptor> descriptors = registry.modify(getXMLDescription(XML_EVENT_DESCRIPTION));
102 		assertNotNull(descriptors);
103 		assertTrue(descriptors.size() == 1);
104 		assertEquals(descriptors.get(0).getClassName(), "org/openjdk/jmc/agent/test/InstrumentMe");
105 		assertEquals(descriptors.get(0).getMethod().toString(), "printHelloWorldJFR1()V");
106 		assertTrue(registry.hasPendingTransforms("org/openjdk/jmc/agent/test/InstrumentMe"));
107 	}
108 
109 	@Test
110 	public void testModifyNameCollision() throws XMLStreamException, IOException {
111 		TransformRegistry registry = DefaultTransformRegistry
112 				.from(TestToolkit.getProbesXMLFromTemplate(getTemplate(), "Modify")); //$NON-NLS-1$
113 		assertNotNull(registry);
114 		final String collisionDescirption = getXMLDescription(XML_EVENT_DESCRIPTION.concat(XML_EVENT_DESCRIPTION));
115 		List<TransformDescriptor> descriptors = registry.modify(collisionDescirption);
116 		assertNotNull(descriptors);
117 		assertTrue(descriptors.size() == 1);
118 	}
119 
120 	@Test
121 	public void testClearAllTransformData() throws XMLStreamException, IOException {
122 		TransformRegistry registry = DefaultTransformRegistry
123 				.from(TestToolkit.getProbesXMLFromTemplate(getTemplate(), "clearAllTransformData")); //$NON-NLS-1$
124 		assertNotNull(registry);
125 		List<String> classesCleared = registry.clearAllTransformData();
126 		assertEquals(classesCleared.get(0),Type.getInternalName(InstrumentMe.class));
127 		assertNull(registry.getTransformData(Type.getInternalName(InstrumentMe.class)));
128 	}
129 
130 	private String getXMLDescription(String eventsDescription) {
131 		return "<jfragent><events>".concat(eventsDescription).concat("</events></jfragent>");
132 	}
133 
134 }
