 33 package org.openjdk.jmc.flightrecorder.internal;
 34 
 35 import java.io.IOException;
 36 import java.io.InputStream;
 37 import java.util.Collections;
 38 import java.util.HashMap;
 39 import java.util.Locale;
 40 import java.util.Map;
 41 import java.util.Map.Entry;
 42 import java.util.Properties;
 43 import java.util.regex.Pattern;
 44 
 45 /**
 46  * Contain algorithmic conversion and overrides for the display name of event path segments. Also,
 47  * algorithmic generation and overrides for the color of event types. For these purposes, a case
 48  * insensitive matching of the path is made. (That is, it is converted to the canonical lower case
 49  * representation.)
 50  */
 51 /*
 52  * FIXME: This class provides knowledge about JDK events. It would make sense to make this
 53  * extendable.
 54  *
 55  * FIXME: This class (or to be precise, the flightrecorder.internal package) is exposed using a
 56  * friends-only export to flightrecorder.configuration. We could instead move it to for example the
 57  * flightrecorder.jdk package and perhaps rename it to something more related to path segments.
 58  */
 59 public class EventAppearance {
 60 	private static final Pattern PATH_SPLIT_REGEX = Pattern.compile("\\/"); //$NON-NLS-1$
 61 	private static final Map<String, String> HUMAN_NAMES;
 62 	static {
 63 		{
 64 			Properties props = loadProperties("segments.properties"); //$NON-NLS-1$
 65 			HashMap<String, String> segments = new HashMap<>(props.size());
 66 			for (Entry<Object, Object> entry : props.entrySet()) {
 67 				String key = (String) entry.getKey();
 68 				segments.put(key.toLowerCase(Locale.ENGLISH), (String) entry.getValue());
 69 			}
 70 			HUMAN_NAMES = Collections.unmodifiableMap(segments);
 71 		}
 72 	}
 73 
 74 	private static Properties loadProperties(String fileName) {
 75 		// Reading through Properties now, for simplicity.
 76 		// Might change to ResourceBundle, or do as FieldToolkit (or NLS),
 77 		// if localization is needed. (Which I doubt, since it would be confusing.)
 78 		Properties properties = new Properties();
 79 		try (InputStream in = EventAppearance.class.getResourceAsStream(fileName)) {
 80 			if (in != null) {
 81 				properties.load(in);
 82 			} else {
 83 				System.err.println("Couldn't find file '" + fileName + "'"); //$NON-NLS-1$ //$NON-NLS-2$
 84 			}
 85 		} catch (IOException e) {
 86 			System.err.println("Problem loading file '" + fileName + "'"); //$NON-NLS-1$ //$NON-NLS-2$
 87 			e.printStackTrace();
 88 		}
 89 		return properties;
 90 	}
 91 
 92 	public static String[] getHumanSegmentArray(String path) {
 93 		String[] pathArray = PATH_SPLIT_REGEX.split(path);
 94 		for (int i = 0; i < pathArray.length; i++) {
 95 			pathArray[i] = getHumanSegmentName(pathArray[i].trim());
 96 		}
 97 		return pathArray;
 98 	}
 99 
100 	private static String getHumanSegmentName(String path) {
101 		path = path.toLowerCase(Locale.ENGLISH);
102 		String humanName = HUMAN_NAMES.get(path);
103 		if (humanName != null) {
104 			return humanName;
105 		}
106 		// NOTE: In order to be thread safe, do not save the human readable name.
107 		return humanifyName(path);
108 	}
109 
110 	// Migrated from PathDescriptorRepository
111 	public static String humanifyName(String identifier) {
112 		if (identifier == null) {
113 			return null;
114 		} else if (identifier.length() == 0) {
115 			// This is so that malformed event paths, should they get this far, are noticed and not hidden.
116 			return "<Empty>"; //$NON-NLS-1$
117 		}
118 
119 		StringBuilder humanReadable = new StringBuilder(identifier.length());
120 		boolean firstLetter = true;
121 		for (int n = 0; n < identifier.length(); n++) {
122 			char c = identifier.charAt(n);
123 			if (c == '_') {
124 				c = ' ';
125 			}
126 			if (firstLetter && Character.isLetter(c)) {
127 				humanReadable.append(Character.toUpperCase(c));
128 			} else {
129 				humanReadable.append(c);
130 			}
131 			firstLetter = (c == ' ');
132 		}
133 		return humanReadable.toString();
134 	}
