38 import org.eclipse.ui.activities.IWorkbenchActivitySupport;
39 
40 /**
41  * Class responsible for activating activities that can be specified on the command line. Currently
42  * only the -designer role is supported.
43  */
44 public final class ActivitiesToolkit {
45 	private static final Set<String> ACTIVITIES = new HashSet<>();
46 
47 	static void updateActivities() {
48 		getActivitySupport().setEnabledActivityIds(ACTIVITIES);
49 	}
50 
51 	public static IWorkbenchActivitySupport getActivitySupport() {
52 		return UIPlugin.getDefault().getWorkbench().getActivitySupport();
53 	}
54 
55 	public static boolean enableActivity(String id) {
56 		boolean success = ACTIVITIES.add(id);
57 		updateActivities();
58 		return success;
59 	}
60 
61 	public static boolean disableActivity(String id) {
62 		boolean success = ACTIVITIES.remove(id);
63 		updateActivities();
64 		return success;
65 	}
66 }
