175 	private static ImageDescriptor icon(String filename) {
176 		return createDescriptor("$nl$/icons/" + filename); //$NON-NLS-1$
177 	}
178 
179 	private static ImageDescriptor createDescriptor(String relPath) {
180 		ImageDescriptor desc = AbstractUIPlugin.imageDescriptorFromPlugin(UIPlugin.PLUGIN_ID, relPath);
181 		if (desc == null) {
182 			// FIXME: Throwing an exception has the development time advantage of being very intrusive. For release time, logging might be better.
183 			throw new MissingResourceException("Missing image '" + relPath + '\'', ImageDescriptor.class.getName(), //$NON-NLS-1$
184 					UIPlugin.PLUGIN_ID + '/' + relPath);
185 		}
186 		return desc;
187 	}
