 33 package org.openjdk.jmc.console.jconsole;
 34 
 35 import java.io.File;
 36 import java.util.logging.Logger;
 37 
 38 import org.eclipse.jface.preference.IPreferenceStore;

 39 import org.eclipse.ui.plugin.AbstractUIPlugin;


 40 import org.openjdk.jmc.console.jconsole.preferences.PreferenceConstants;
 41 import org.osgi.framework.BundleContext;
 42 
 43 /**
 44  * The activator class controls the plug-in life cycle
 45  */
 46 public class Activator extends AbstractUIPlugin {
 47 	// The plug-in ID
 48 	public static final String PLUGIN_ID = "org.openjdk.jmc.console.jconsole"; //$NON-NLS-1$
 49 	private static Logger LOGGER = Logger.getLogger(PLUGIN_ID);
 50 
 51 	// The shared instance
 52 	private static Activator plugin;
 53 
 54 	/**
 55 	 * The constructor
 56 	 */
 57 	public Activator() {
 58 	}
 59 
 60 	@Override
 61 	public void start(BundleContext context) throws Exception {
 62 		super.start(context);
 63 		plugin = this;
 64 	}
 65 
 66 	@Override
 67 	public void stop(BundleContext context) throws Exception {
 68 		plugin = null;
 69 		super.stop(context);
 70 	}
 71 
 72 	/**
 73 	 * Returns the shared instance
 74 	 *
 75 	 * @return the shared instance
 76 	 */
 77 	public static Activator getDefault() {
 78 		return plugin;
 79 	}
 80 











 81 	/**
 82 	 * @return the logger used by this plug-in.
 83 	 */
 84 	public static Logger getLogger() {
 85 		return LOGGER;
 86 	}
 87 
 88 	private static IPreferenceStore getPrefs() {
 89 		return Activator.getDefault().getPreferenceStore();
 90 	}
 91 
 92 	/**
 93 	 * @return the currently configured JConsole plug-in dir.
 94 	 */
 95 	public static File getJConsolePluginDir() {
 96 		return new File(getPrefs().getString(PreferenceConstants.PROPERTY_JCONSOLE_PLUGIN_DIR));
 97 	}
 98 
 99 	public static long getUpdateInterval() {
100 		return getPrefs().getLong(PreferenceConstants.PROPERTY_JCONSOLE_UPDATE_INTERVAL);
101 
102 	}
103 }
