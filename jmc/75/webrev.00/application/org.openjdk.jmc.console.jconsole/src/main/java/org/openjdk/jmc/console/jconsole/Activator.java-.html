 33 package org.openjdk.jmc.console.jconsole;
 34 
 35 import java.io.File;
 36 import java.util.logging.Logger;
 37 
 38 import org.eclipse.jface.preference.IPreferenceStore;
 39 import org.eclipse.jface.resource.ImageDescriptor;
 40 import org.eclipse.ui.plugin.AbstractUIPlugin;
 41 import org.osgi.framework.BundleContext;
 42 
 43 import org.openjdk.jmc.console.jconsole.preferences.PreferenceConstants;
 44 
 45 /**
 46  * The activator class controls the plug-in life cycle
 47  */
 48 public class Activator extends AbstractUIPlugin {
 49 	// The plug-in ID
 50 	public static final String PLUGIN_ID = "org.openjdk.jmc.console.jconsole"; //$NON-NLS-1$
 51 	private static Logger LOGGER = Logger.getLogger(PLUGIN_ID);
 52 
 53 	// The shared instance
 54 	private static Activator plugin;
 55 
 56 	/**
 57 	 * The constructor
 58 	 */
 59 	public Activator() {
 60 	}
 61 
 62 	@Override
 63 	public void start(BundleContext context) throws Exception {
 64 		super.start(context);
 65 		plugin = this;
 66 	}
 67 
 68 	@Override
 69 	public void stop(BundleContext context) throws Exception {
 70 		plugin = null;
 71 		super.stop(context);
 72 	}
 73 
 74 	/**
 75 	 * Returns the shared instance
 76 	 *
 77 	 * @return the shared instance
 78 	 */
 79 	public static Activator getDefault() {
 80 		return plugin;
 81 	}
 82 
 83 	/**
 84 	 * Returns an image descriptor for the image file at the given plug-in relative path
 85 	 *
 86 	 * @param path
 87 	 *            the path
 88 	 * @return the image descriptor
 89 	 */
 90 	public static ImageDescriptor getImageDescriptor(String path) {
 91 		return imageDescriptorFromPlugin(PLUGIN_ID, path);
 92 	}
 93 
 94 	/**
 95 	 * @return the logger used by this plug-in.
 96 	 */
 97 	public static Logger getLogger() {
 98 		return LOGGER;
 99 	}
100 
101 	private static IPreferenceStore getPrefs() {
102 		return Activator.getDefault().getPreferenceStore();
103 	}
104 
105 	/**
106 	 * @return the currently configured JConsole plug-in dir.
107 	 */
108 	public static File getJConsolePluginDir() {
109 		return new File(getPrefs().getString(PreferenceConstants.PROPERTY_JCONSOLE_PLUGIN_DIR));
110 	}
111 
112 	public static long getUpdateInterval() {
113 		return getPrefs().getLong(PreferenceConstants.PROPERTY_JCONSOLE_UPDATE_INTERVAL);
114 
115 	}
116 }
