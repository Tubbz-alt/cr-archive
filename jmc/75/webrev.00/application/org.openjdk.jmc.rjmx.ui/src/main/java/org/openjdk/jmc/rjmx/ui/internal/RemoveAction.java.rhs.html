33 package org.openjdk.jmc.rjmx.ui.internal;
34 
35 import org.eclipse.jface.action.Action;
36 import org.eclipse.jface.resource.ImageDescriptor;
37 import org.eclipse.jface.resource.ResourceLocator;
38 import org.eclipse.ui.forms.SectionPart;


39 import org.openjdk.jmc.ui.misc.MCSectionPart;
40 
41 /**
42  * Action for removing a {@link SectionPart} in a {@link SectionPartManager}
43  */
44 public class RemoveAction extends Action {
45 	private final static ImageDescriptor ICON = ResourceLocator.imageDescriptorFromBundle("org.eclipse.ui", //$NON-NLS-1$
46 			"icons/full/elcl16/close_view.gif").orElse(null); //$NON-NLS-1$
47 	final private SectionPartManager m_sectionPartmanager;
48 	final private MCSectionPart m_part;
49 
50 	public RemoveAction(SectionPartManager manager, MCSectionPart part) {
51 		setImageDescriptor(ICON);
52 		setText(Messages.RemoveAction_REMOVE_CONTROL_TEXT);
53 		setToolTipText(Messages.RemoveAction_REMOVE_CONTROL_TEXT);
54 		setId("remove"); //$NON-NLS-1$
55 		m_sectionPartmanager = manager;
56 		m_part = part;
57 	}
58 
59 	@Override
60 	public void run() {
61 		m_sectionPartmanager.destroyPart(m_part);
62 	}
63 }
