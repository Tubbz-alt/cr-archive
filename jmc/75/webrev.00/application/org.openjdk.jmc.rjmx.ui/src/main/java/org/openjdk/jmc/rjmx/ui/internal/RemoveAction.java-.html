33 package org.openjdk.jmc.rjmx.ui.internal;
34 
35 import org.eclipse.jface.action.Action;
36 import org.eclipse.jface.resource.ImageDescriptor;
37 import org.eclipse.ui.forms.SectionPart;
38 import org.eclipse.ui.plugin.AbstractUIPlugin;
39 
40 import org.openjdk.jmc.ui.misc.MCSectionPart;
41 
42 /**
43  * Action for removing a {@link SectionPart} in a {@link SectionPartManager}
44  */
45 public class RemoveAction extends Action {
46 	private final static ImageDescriptor ICON = AbstractUIPlugin.imageDescriptorFromPlugin("org.eclipse.ui", //$NON-NLS-1$
47 			"icons/full/elcl16/close_view.gif"); //$NON-NLS-1$
48 	final private SectionPartManager m_sectionPartmanager;
49 	final private MCSectionPart m_part;
50 
51 	public RemoveAction(SectionPartManager manager, MCSectionPart part) {
52 		setImageDescriptor(ICON);
53 		setText(Messages.RemoveAction_REMOVE_CONTROL_TEXT);
54 		setToolTipText(Messages.RemoveAction_REMOVE_CONTROL_TEXT);
55 		setId("remove"); //$NON-NLS-1$
56 		m_sectionPartmanager = manager;
57 		m_part = part;
58 	}
59 
60 	@Override
61 	public void run() {
62 		m_sectionPartmanager.destroyPart(m_part);
63 	}
64 }
