33 package org.openjdk.jmc.rcp.application.commands;
34 
35 import java.io.PrintStream;
36 
37 import org.eclipse.ui.IEditorPart;
38 import org.eclipse.ui.IWorkbenchPage;
39 import org.eclipse.ui.IWorkbenchWindow;
40 import org.eclipse.ui.PlatformUI;
41 import org.openjdk.jmc.commands.IExecute;
42 import org.openjdk.jmc.commands.Statement;
43 
44 /**
45  * This class should move to core
46  */
47 public final class CloseEditor implements IExecute {
48 
49 	@Override
50 	public boolean execute(Statement statement, PrintStream out) {
51 		IWorkbenchWindow ww = PlatformUI.getWorkbench().getActiveWorkbenchWindow();
52 		if (ww != null) {
53 			IWorkbenchPage wp = ww.getActivePage();
54 			if (wp != null) {
55 				IEditorPart ep = wp.getActiveEditor();
56 				wp.closeEditor(ep, false);
57 			}
58 		}
59 		return false;
60 	}
61 
62 }
