33 package org.openjdk.jmc.flightrecorder.controlpanel.ui.actions;
34 
35 import org.eclipse.jface.wizard.IWizard;
36 import org.openjdk.jmc.flightrecorder.controlpanel.ui.ControlPanel;
37 import org.openjdk.jmc.flightrecorder.controlpanel.ui.FlightRecorderProvider;
38 import org.openjdk.jmc.flightrecorder.controlpanel.ui.ImageConstants;
39 import org.openjdk.jmc.flightrecorder.controlpanel.ui.messages.internal.Messages;
40 import org.openjdk.jmc.flightrecorder.controlpanel.ui.wizards.RecordingWizardModel;
41 import org.openjdk.jmc.flightrecorder.controlpanel.ui.wizards.StartRecordingWizard;
42 import org.openjdk.jmc.rjmx.IConnectionHandle;
43 import org.openjdk.jmc.rjmx.JVMSupportToolkit;
44 import org.openjdk.jmc.rjmx.services.jfr.FlightRecorderException;
45 import org.openjdk.jmc.rjmx.services.jfr.IFlightRecorderService;
46 import org.openjdk.jmc.ui.common.resource.MCFile;
47 import org.openjdk.jmc.ui.wizards.AbstractWizardUserAction;
48 
49 public class StartRecordingAction extends AbstractWizardUserAction {
50 
51 	private final FlightRecorderProvider recorder;
52 
53 	public StartRecordingAction(FlightRecorderProvider recorder) {
54 		super(Messages.ACTION_START_RECORDING_LABEL, Messages.ACTION_START_RECORDING_TOOLTIP,
55 				ControlPanel.getDefault().getMCImageDescriptor(ImageConstants.ICON_FLIGHT_RECORDING_ON));
56 		this.recorder = recorder;
57 	}
58 
59 	@Override
60 	public IWizard doCreateWizard() throws Exception {
61 		try (IConnectionHandle handle = recorder.getServerHandle().connect(Messages.ACTION_START_RECORDING_LABEL)) {
62 			IFlightRecorderService flrService = handle.getServiceOrNull(IFlightRecorderService.class);
63 			if (flrService == null || !JVMSupportToolkit.hasFlightRecorder(handle)) {
64 				throw new FlightRecorderException(JVMSupportToolkit.getNoFlightRecorderErrorMessage(handle, false));
65 			} else if (flrService.isEnabled()
66 					|| ControlPanel.askUserForEnable(flrService, Messages.COMMERCIAL_FEATURES_QUESTION)) {
67 				MCFile recFile = ControlPanel.getDefaultRecordingFile(recorder.getServerHandle());
68 				RecordingWizardModel model = new RecordingWizardModel(flrService, recFile);
69 				recorder.resetWarning();
70 				return new StartRecordingWizard(model, recorder);
71 			} else {
72 				return null;
73 			}
74 		} catch (Exception e) {
75 			recorder.setWarning(e.getLocalizedMessage());
76 			throw e;
77 		}
78 	}
79 
80 }
