33 package org.openjdk.jmc.console.ui.misc;
34 
35 import java.io.PrintStream;
36 
37 import org.eclipse.ui.IEditorInput;
38 import org.eclipse.ui.IWorkbenchWindow;
39 import org.eclipse.ui.PartInitException;
40 
41 import org.openjdk.jmc.commands.Statement;
42 import org.openjdk.jmc.console.ui.editor.internal.ConsoleEditorInput;
43 import org.openjdk.jmc.rjmx.util.internal.RJMXStartCommand;
44 import org.openjdk.jmc.ui.UIPlugin;
45 
46 /**
47  * Command for starting up the JMX Console on a RJMX-descriptor.
48  */
49 public class StartConsole extends RJMXStartCommand {
50 	@Override
51 	public boolean execute(Statement statement, PrintStream out) {
52 		IWorkbenchWindow window = UIPlugin.getDefault().getWorkbench().getActiveWorkbenchWindow();
53 		if (window != null) {
54 			IEditorInput input = new ConsoleEditorInput(createConnectionDescriptor(statement, out));
55 			try {
56 				window.getActivePage().openEditor(input, ConsoleEditorInput.EDITOR_ID);
57 			} catch (PartInitException e) {
58 				out.println(e.getMessage());
59 				return false;
60 			}
61 		}
62 		return true;
63 	}
64 }
