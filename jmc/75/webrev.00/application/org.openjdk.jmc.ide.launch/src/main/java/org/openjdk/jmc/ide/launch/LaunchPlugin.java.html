33 package org.openjdk.jmc.ide.launch;
34 
35 import org.eclipse.jface.resource.ImageRegistry;
36 import org.eclipse.swt.widgets.Shell;
37 import org.eclipse.ui.IWorkbenchWindow;
38 import org.eclipse.ui.PlatformUI;
39 import org.osgi.framework.BundleContext;
40 
41 import org.openjdk.jmc.ui.MCAbstractUIPlugin;
42 
43 /**
44  * The activator class controls the plug-in life cycle.
45  */
46 public class LaunchPlugin extends MCAbstractUIPlugin {
47 
48 	// The plug-in ID
49 	public static final String PLUGIN_ID = "org.openjdk.jmc.ide.launch"; //$NON-NLS-1$
50 
51 	public static final String JFR_ICON = "jfr.png"; //$NON-NLS-1$
52 
53 	// The shared instance
54 	private static LaunchPlugin plugin;
55 
56 	/**
57 	 * The constructor
58 	 */
59 	public LaunchPlugin() {
60 		super(PLUGIN_ID);
61 	}
62 
63 	@Override
64 	public void start(BundleContext context) throws Exception {
65 		super.start(context);
66 		plugin = this;
67 	}
68 
69 	@Override
70 	public void stop(BundleContext context) throws Exception {
71 		plugin = null;
72 		super.stop(context);
73 	}
74 
75 	/**
76 	 * Returns the shared instance
77 	 *
78 	 * @return the shared instance
79 	 */
80 	public static LaunchPlugin getDefault() {
81 		return plugin;
82 	}
83 
84 	@Override
85 	protected void initializeImageRegistry(ImageRegistry registry) {
86 		registerImage(registry, JFR_ICON, JFR_ICON);
87 	}
88 
89 	static Shell getActiveWorkbenchShell() {
90 		IWorkbenchWindow window = PlatformUI.getWorkbench().getActiveWorkbenchWindow();
91 
92 		if (window != null) {
93 			return window.getShell();
94 		}
95 		return null;
96 	}
97 
98 }
