33 package org.openjdk.jmc.ide.launch;
34 
35 import org.eclipse.jface.resource.ImageRegistry;
36 import org.eclipse.swt.widgets.Shell;
37 import org.eclipse.ui.IWorkbenchWindow;

38 import org.osgi.framework.BundleContext;
39 
40 import org.openjdk.jmc.ui.MCAbstractUIPlugin;
41 
42 /**
43  * The activator class controls the plug-in life cycle.
44  */
45 public class LaunchPlugin extends MCAbstractUIPlugin {
46 
47 	// The plug-in ID
48 	public static final String PLUGIN_ID = "org.openjdk.jmc.ide.launch"; //$NON-NLS-1$
49 
50 	public static final String JFR_ICON = "jfr.png"; //$NON-NLS-1$
51 
52 	// The shared instance
53 	private static LaunchPlugin plugin;
54 
55 	/**
56 	 * The constructor
57 	 */
58 	public LaunchPlugin() {
59 		super(PLUGIN_ID);
60 	}
61 
62 	@Override
63 	public void start(BundleContext context) throws Exception {
64 		super.start(context);
65 		plugin = this;
66 	}
67 
68 	@Override
69 	public void stop(BundleContext context) throws Exception {
70 		plugin = null;
71 		super.stop(context);
72 	}
73 
74 	/**
75 	 * Returns the shared instance
76 	 *
77 	 * @return the shared instance
78 	 */
79 	public static LaunchPlugin getDefault() {
80 		return plugin;
81 	}
82 
83 	@Override
84 	protected void initializeImageRegistry(ImageRegistry registry) {
85 		registerImage(registry, JFR_ICON, JFR_ICON);
86 	}
87 
88 	static Shell getActiveWorkbenchShell() {
89 		IWorkbenchWindow window = LaunchPlugin.getDefault().getWorkbench().getActiveWorkbenchWindow();
90 
91 		if (window != null) {
92 			return window.getShell();
93 		}
94 		return null;
95 	}
96 
97 }
