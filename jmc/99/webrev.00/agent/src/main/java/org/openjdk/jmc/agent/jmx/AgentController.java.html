 33 package org.openjdk.jmc.agent.jmx;
 34 
 35 import java.lang.instrument.Instrumentation;
 36 import java.lang.management.ManagementPermission;
 37 import java.util.ArrayList;
 38 import java.util.HashSet;
 39 import java.util.List;
 40 import java.util.Set;
 41 import java.util.logging.Level;
 42 import java.util.logging.Logger;
 43 
 44 import org.openjdk.jmc.agent.TransformDescriptor;
 45 import org.openjdk.jmc.agent.TransformRegistry;
 46 import org.openjdk.jmc.agent.jfr.JFRTransformDescriptor;
 47 
 48 public class AgentController implements AgentControllerMXBean {
 49 	
 50 	private static final Logger logger = Logger.getLogger(AgentController.class.getName());
 51 	
 52 	private final Instrumentation instrumentation;
 53 	private final TransformRegistry registry;
 54 
 55 	public AgentController(Instrumentation instrumentation, TransformRegistry registry) {
 56 		this.instrumentation = instrumentation;
 57 		this.registry = registry;
 58 	}
 59 
 60 	public void defineEventProbes(String xmlDescription) throws Exception{
 61 		checkSecurity();
 62 		Class<?>[] classesToRetransformArray;
 63 		boolean revertAll = xmlDescription == null ? true : xmlDescription.isEmpty();
 64 		if (revertAll) {
 65 			classesToRetransformArray = revertAllTransforms();
 66 			registry.setCurrentConfiguration("");
 67 		} else {
 68 			List<TransformDescriptor> descriptors = registry.modify(xmlDescription);
 69 			if (descriptors == null) {
 70 				logger.log(Level.SEVERE, "Failed to identify transformations: " + xmlDescription);
 71 				return;
 72 			} else if (descriptors.isEmpty()) {
 73 				classesToRetransformArray = revertAllTransforms();
 74 			} else {
 75 				classesToRetransformArray = defineSpecificTransforms(descriptors);
 76 			}
 77 		}
 78 		registry.setRevertInstrumentation(true);
 79 		instrumentation.retransformClasses(classesToRetransformArray);
 80 		registry.setRevertInstrumentation(false);
 81 	}
 82 
 83 	private Class<?>[] revertAllTransforms() {
 84 		Set<Class<?>> classesToRetransform = new HashSet<>();
 85 		List<String> classNames = registry.clearAllTransformData();
 86 		for (String className : classNames ) {
 87 			try {
 88 				Class<?> classToRetransform = Class.forName(className.replace('/', '.'));
 89 				classesToRetransform.add(classToRetransform);
 90 			} catch (ClassNotFoundException cnfe) {
 91 				logger.log(Level.SEVERE, "Unable to find class: " + className, cnfe);
 92 			}
 93 		}
 94 		return classesToRetransform.toArray(new Class<?>[0]);
 95 	}
 96 
 97 	private Class<?>[] defineSpecificTransforms(List<TransformDescriptor> descriptors) {
 98 		Set<Class<?>> classesToRetransform = new HashSet<>();
 99 		for (TransformDescriptor descriptor : descriptors) {
100 			try {
101 				Class<?> classToRetransform = Class.forName(descriptor.getClassName().replace('/', '.'));
102 				classesToRetransform.add(classToRetransform);
103 			} catch (ClassNotFoundException cnfe) {
104 				logger.log(Level.SEVERE, "Unable to find class: " + descriptor.getClassName(), cnfe);
105 			}
106 		}
107 		return classesToRetransform.toArray(new Class<?>[0]);
108 	}
109 
110 	public JFRTransformDescriptor[] retrieveCurrentTransforms() {
111 		checkSecurity();
112 		Set<String> classNames = registry.getClassNames();
113 		List<TransformDescriptor> tds  = new ArrayList<>();
114 		for (String className : classNames) {
115 			tds.addAll(registry.getTransformData(className));
116 		}
117 
118 		List<JFRTransformDescriptor> jfrTds = new ArrayList<>();
119 		for (TransformDescriptor td :tds) {
120 			jfrTds.add((JFRTransformDescriptor) td);
121 		}
122 		return (jfrTds.toArray(new JFRTransformDescriptor[0]));
123 	}
124 
125 	public String retrieveEventProbes() {
126 		checkSecurity();
127 		return registry.getCurrentConfiguration();
128 	}
129 
130 	private void checkSecurity() {
131 		  SecurityManager secMan = System.getSecurityManager();
132 		  if (secMan != null) {
133 		    secMan.checkPermission(new ManagementPermission("control"));
134 		  }
135 	}
136 
137 }
