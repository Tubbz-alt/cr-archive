34 package org.openjdk.jmc.joverflow.ext.treemap;
35 
36 import org.eclipse.jface.action.Action;
37 import org.eclipse.jface.action.IAction;
38 import org.eclipse.jface.resource.ImageDescriptor;
39 import org.openjdk.jmc.ui.CoreImages;
40 
41 /* package-private */ class TreemapAction extends Action {
42 	private final TreemapActionType actionType;
43 	private Runnable runnable = () -> {
44 	};
45 
46 	TreemapAction(TreemapActionType actionType) {
47 		super(actionType.message, actionType.action);
48 		this.actionType = actionType;
49 		setToolTipText(actionType.message);
50 		setImageDescriptor(actionType.imageDescriptor);
51 	}
52 
53 	@Override
54 	public void run() {
55 		runnable.run();
56 	}
57 
58 	public void setRunnable(Runnable callback) {
59 		runnable = callback;
60 	}
61 
62 	public TreemapActionType getType() {
63 		return actionType;
64 	}
65 
66 	enum TreemapActionType {
67 		ZOOM_IN(Messages.TreemapAction_ZOOM_IN_DESCRIPTION, IAction.AS_PUSH_BUTTON, CoreImages.ZOOM_IN), // 
68 		ZOOM_OUT(Messages.TreemapAction_ZOOM_OUT_DESCRIPTION, IAction.AS_PUSH_BUTTON, CoreImages.ZOOM_OUT), // 
69 		ZOOM_OFF(Messages.TreemapAction_ZOOM_OFF_DESCRIPTION, IAction.AS_PUSH_BUTTON, CoreImages.ZOOM_OFF);
70 
71 		private final String message;
72 		private final int action;
73 		private final ImageDescriptor imageDescriptor;
74 
75 		TreemapActionType(String message, int action, ImageDescriptor imageDescriptor) {
76 			this.message = message;
77 			this.action = action;
78 			this.imageDescriptor = imageDescriptor;
79 		}
80 	}
81 }
