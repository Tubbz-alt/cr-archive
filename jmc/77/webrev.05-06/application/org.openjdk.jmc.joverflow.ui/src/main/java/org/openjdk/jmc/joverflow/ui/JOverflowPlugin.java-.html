33 package org.openjdk.jmc.joverflow.ui;
34 
35 import org.openjdk.jmc.ui.MCAbstractUIPlugin;
36 import org.osgi.framework.BundleContext;
37 
38 /**
39  * The activator class controls the plug-in life cycle.
40  */
41 public class JOverflowPlugin extends MCAbstractUIPlugin {
42 
43 	// The plug-in ID
44 	private static final String PLUGIN_ID = "org.openjdk.jmc.joverflow.ui"; //$NON-NLS-1$
45 
46 	static final String ICON_UNDO_EDIT = "undo_edit.gif"; //$NON-NLS-1$
47 
48 	// The shared instance
49 	private static JOverflowPlugin m_plugin;
50 
51 	/**
52 	 * The constructor
53 	 */
54 	public JOverflowPlugin() {
55 		super(PLUGIN_ID);
56 	}
57 
58 	@Override
59 	public void start(BundleContext context) throws Exception {
60 		super.start(context);
61 		m_plugin = this;
62 	}
63 
64 	@Override
65 	public void stop(BundleContext context) throws Exception {
66 		m_plugin = null;
67 		super.stop(context);
68 	}
69 
70 	/**
71 	 * Returns the shared instance
72 	 *
73 	 * @return the shared instance
74 	 */
75 	public static JOverflowPlugin getDefault() {
76 		return m_plugin;
77 	}
78 }
