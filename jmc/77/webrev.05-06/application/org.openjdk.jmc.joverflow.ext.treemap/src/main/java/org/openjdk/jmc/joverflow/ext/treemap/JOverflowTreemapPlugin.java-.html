34 package org.openjdk.jmc.joverflow.ext.treemap;
35 
36 import org.openjdk.jmc.ui.MCAbstractUIPlugin;
37 import org.osgi.framework.BundleContext;
38 
39 /**
40  * The activator class controls the plug-in life cycle.
41  */
42 public class JOverflowTreemapPlugin extends MCAbstractUIPlugin {
43 
44 	// The plug-in ID
45 	private static final String PLUGIN_ID = "org.openjdk.jmc.joverflow.ext.treemap"; //$NON-NLS-1$
46 
47 	static final String ICON_RESET = "reset.gif"; //$NON-NLS-1$
48 
49 	// The shared instance
50 	private static JOverflowTreemapPlugin m_plugin;
51 
52 	/**
53 	 * The constructor
54 	 */
55 	public JOverflowTreemapPlugin() {
56 		super(PLUGIN_ID);
57 	}
58 
59 	@Override
60 	public void start(BundleContext context) throws Exception {
61 		super.start(context);
62 		m_plugin = this;
63 	}
64 
65 	@Override
66 	public void stop(BundleContext context) throws Exception {
67 		m_plugin = null;
68 		super.stop(context);
69 	}
70 
71 	/**
72 	 * Returns the shared instance
73 	 *
74 	 * @return the shared instance
75 	 */
76 	public static JOverflowTreemapPlugin getDefault() {
77 		return m_plugin;
78 	}
79 }
