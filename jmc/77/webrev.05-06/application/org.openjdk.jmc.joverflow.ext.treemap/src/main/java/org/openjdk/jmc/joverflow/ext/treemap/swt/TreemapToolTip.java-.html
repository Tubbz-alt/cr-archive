34 package org.openjdk.jmc.joverflow.ext.treemap.swt;
35 
36 import org.eclipse.swt.SWT;
37 import org.eclipse.swt.layout.RowLayout;
38 import org.eclipse.swt.widgets.Composite;
39 import org.eclipse.swt.widgets.Control;
40 import org.eclipse.swt.widgets.Event;
41 import org.eclipse.swt.widgets.Label;
42 import org.eclipse.jface.window.ToolTip;
43 
44 /*package-private*/ class TreemapToolTip extends ToolTip {
45 	private static final int PADDING = 5;
46 
47 	private TreemapItem item = null;
48 
49 	public TreemapToolTip(Control parent) {
50 		super(parent);
51 	}
52 
53 	@Override
54 	protected Composite createToolTipContentArea(Event event, Composite parent) {
55 		Composite container = new Composite(parent, SWT.NONE);
56 
57 		RowLayout rowLayout = new RowLayout();
58 		rowLayout.marginLeft = PADDING;
59 		rowLayout.marginTop = PADDING;
60 		rowLayout.marginRight = PADDING;
61 		rowLayout.marginBottom = PADDING;
62 
63 		container.setLayout(rowLayout);
64 		container.setBackground(parent.getBackground());
65 
66 		Label label = new Label(container, SWT.NONE);
67 		label.setText(item != null ? item.getToolTipText() : ""); //$NON-NLS-1$
68 		label.setForeground(parent.getForeground());
69 
70 		return container;
71 	}
72 
73 	public void setItem(TreemapItem item) {
74 		this.item = item;
75 
76 		if (item.getToolTipText().isEmpty()) {
77 			deactivate();
78 		} else {
79 			activate();
80 		}
81 	}
82 }
