383 	private static String getIconBase64(String iconName) {

385 		if (image == null) {
386 			return "";
387 		} else {

389 			ImageLoader loader = new ImageLoader();

392 			return Base64.getEncoder().encodeToString(baos.toByteArray());
393 		}
