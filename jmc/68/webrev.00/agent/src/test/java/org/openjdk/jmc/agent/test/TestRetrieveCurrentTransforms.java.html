34 package org.openjdk.jmc.agent.test;
35 
36 import java.lang.management.ManagementFactory;
37 
38 import javax.management.JMX;
39 import javax.management.ObjectName;
40 
41 import org.junit.Assert;
42 import org.junit.Test;
43 import org.openjdk.jmc.agent.jfr.JFRTransformDescriptor;
44 import org.openjdk.jmc.agent.jmx.AgentControllerMXBean;
45 
46 public class TestRetrieveCurrentTransforms {
47 
48 	private static final String AGENT_OBJECT_NAME = "org.openjdk.jmc.jfr.agent:type=AgentController"; //$NON-NLS-1$
49 	private static final String EVENT_ID = "demo.jfr.test1"; //$NON-NLS-1$
50 	private static final String METHOD_NAME = "printHelloWorldJFR1"; //$NON-NLS-1$
51 	private static final String FIELD_NAME = "'InstrumentMe.STATIC_STRING_FIELD'"; //$NON-NLS-1$
52 
53 	@Test
54 	public void testRetreiveCurrentTransforms() throws Exception {
55 		JFRTransformDescriptor[] jfrTds = doRetrieveCurrentTransforms();
56 		Assert.assertTrue(jfrTds.length == 1);
57 		for (JFRTransformDescriptor jfrTd : jfrTds) {
58 			Assert.assertEquals(EVENT_ID, jfrTd.getId());
59 			Assert.assertEquals(METHOD_NAME, jfrTd.getMethod().getName());
60 			Assert.assertEquals(FIELD_NAME, jfrTd.getFields().get(0).getName());
61 		}
62 	}
63 
64 	private JFRTransformDescriptor[] doRetrieveCurrentTransforms() throws Exception {
65 		AgentControllerMXBean mbean = JMX.newMXBeanProxy(ManagementFactory.getPlatformMBeanServer(),
66 				new ObjectName(AGENT_OBJECT_NAME), AgentControllerMXBean.class, false);
67 		return mbean.retrieveCurrentTransforms();
68 	}
69 
70 	public void test() {
71 		//Dummy method for instrumentation
72 	}
73 }
