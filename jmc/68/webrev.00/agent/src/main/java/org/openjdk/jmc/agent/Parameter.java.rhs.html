 33 package org.openjdk.jmc.agent;
 34 
 35 import javax.management.openmbean.CompositeData;
 36 
 37 import org.openjdk.jmc.agent.util.TypeUtils;
 38 
 39 /**
 40  * Metadata for a parameter to be logged by the agent.
 41  */
 42 public final class Parameter implements Attribute {
 43 	public static final int INDEX_INVALID = -1;
 44 
 45 	private final int index;
 46 	private final String name;
 47 	private final String fieldName;
 48 	private final String description;
 49 	private final String contentType;
 50 	private final String relationKey;
 51 	private final String converterClassName;
 52 
 53 	public Parameter(int index, String name, String description, String contentType, String relationKey,
 54 			String converterClassName) {
 55 		this.index = index;
 56 		this.name = name;
 57 		this.description = description;
 58 		this.contentType = contentType;
 59 		this.relationKey = relationKey;
 60 		this.converterClassName = converterClassName;
 61 		this.fieldName = "field" + TypeUtils.deriveIdentifierPart(name); //$NON-NLS-1$
 62 	}
 63 
 64 	public static Parameter from(CompositeData cd) {
 65 		return new Parameter((int) cd.get("index"), (String) cd.get("name"), (String) cd.get("description"),
 66 				(String) cd.get("contentType"), (String) cd.get("relationKey"), (String) cd.get("converterClassName"));
 67 	}
 68 
 69 	public int getIndex() {
 70 		return index;
 71 	}
 72 
 73 	public String getName() {
 74 		return name;
 75 	}
 76 
 77 	public String getDescription() {
 78 		return description;
 79 	}
 80 
 81 	public String getContentType() {
 82 		return contentType;
 83 	}
 84 
 85 	public String getRelationKey() {
 86 		return relationKey;
 87 	}
 88 
 89 	public String getFieldName() {
 90 		return fieldName;
 91 	}
 92 
 93 	public boolean isInvalid() {
 94 		return index == INDEX_INVALID;
 95 	}
 96 
 97 	public String getConverterClassName() {
 98 		return converterClassName;
 99 	}
100 }
