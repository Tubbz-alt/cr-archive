 33 package org.openjdk.jmc.agent;
 34 
 35 import javax.management.openmbean.CompositeData;
 36 
 37 /**
 38  * Definition of a method to be logged by the agent.
 39  */
 40 public class Method {
 41 	private String name;
 42 	private String signature;
 43 
 44 	public Method(String name, String signature) {
 45 		this.name = name;
 46 		this.signature = signature;
 47 	}
 48 
 49 	public static Method from(CompositeData cd) {
 50 		return new Method((String) cd.get("name"), (String) cd.get("signature"));
 51 	}
 52 
 53 	public String getName() {
 54 		return name;
 55 	}
 56 
 57 	public String getSignature() {
 58 		return signature;
 59 	}
 60 
 61 	@Override
 62 	public int hashCode() {
 63 		final int prime = 31;
 64 		int result = 1;
 65 		result = prime * result + ((name == null) ? 0 : name.hashCode());
 66 		result = prime * result + ((signature == null) ? 0 : signature.hashCode());
 67 		return result;
 68 	}
 69 
 70 	@Override
 71 	public boolean equals(Object obj) {
 72 		if (this == obj) {
 73 			return true;
 74 		}
 75 		if (obj == null) {
 76 			return false;
 77 		}
 78 		if (getClass() != obj.getClass()) {
 79 			return false;
 80 		}
 81 		Method other = (Method) obj;
 82 		if (name == null) {
 83 			if (other.name != null) {
 84 				return false;
 85 			}
 86 		} else if (!name.equals(other.name)) {
 87 			return false;
 88 		}
 89 		if (signature == null) {
 90 			if (other.signature != null) {
 91 				return false;
 92 			}
 93 		} else if (!signature.equals(other.signature)) {
 94 			return false;
 95 		}
 96 		return true;
 97 	}
 98 
 99 	@Override
100 	public String toString() {
101 		return name + signature;
102 	}
103 }
