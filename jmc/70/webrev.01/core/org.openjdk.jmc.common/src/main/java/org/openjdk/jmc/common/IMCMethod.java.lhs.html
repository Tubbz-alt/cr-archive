33 package org.openjdk.jmc.common;
34 
35 import java.lang.reflect.Modifier;
36 
37 /**
38  * A method representation.
39  * <p>
40  * There are many different ways methods are represented in Mission Control and in Java:
41  * MethodIdentfier, java.lang.String, stack trace locations, profiled methods, JFR methods etc.
42  * <p>
43  * Some IMCMethod implementations may support a wide range of method information while others might
44  * only be able provide the basic method name. It's up to the user of this interface to examine what
45  * is available. If information is missing {@code null} is returned.
46  */
47 // FIXME: Move IMC* classes to a subpackage
48 public interface IMCMethod {
49 	/**
50 	 * Returns the class this method is declared in.
51 	 *
52 	 * @return the class declaring this method
53 	 */
54 	IMCType getType();
55 
56 	/**
57 	 * Returns the method name not including parameters.
58 	 * <p>
59 	 * An example is "mymethod". If the method is native the format is undefined.
60 	 *
61 	 * @return the name of this method, or {@code null} if unavailable
62 	 */
63 	String getMethodName();
64 
65 	/**
66 	 * Returns the formal descriptor.
67 	 * <p>
68 	 * For example, the method descriptor for the method
69 	 * {@code Object mymethod(int i, double d, Thread t)} is
70 	 * {@code (IDLjava/lang/Thread;)Ljava/lang/Object;}
71 	 *
72 	 * @return the formal method descriptor, or {@code null} if unavailable
73 	 */
74 	String getFormalDescriptor();
75 
76 	/**
77 	 * Returns the modifier used in the Java class file.
78 	 * <p>
79 	 * Examples of modifiers are "protected", "public", etc.
80 	 * <p>
81 	 * See {@link Modifier} for more information about the bit pattern and for methods that can be
82 	 * used to decode it.
83 	 *
84 	 * @return the modifier used in the class file, or {@code null} if not available
85 	 */
86 	Integer getModifier();
87 
88 	/**
89 	 * Whether this method is native.
90 	 *
91 	 * @return {@code Boolean.TRUE} if the method is native, {@code Boolean.FALSE} if not, or
92 	 *         {@code null} if the information is not available
93 	 */
94 	Boolean isNative();








95 }
