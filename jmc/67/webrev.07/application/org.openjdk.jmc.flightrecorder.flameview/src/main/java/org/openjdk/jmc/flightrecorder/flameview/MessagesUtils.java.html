34 package org.openjdk.jmc.flightrecorder.flameview;
35 
36 import java.text.MessageFormat;
37 
38 public final class MessagesUtils {
39 
40 	/**
41 	 * Localized Stack Trace message
42 	 * 
43 	 * @param key
44 	 *            stack trace message
45 	 * @return message
46 	 */
47 	public static String getStacktraceMessage(String key) {
48 		return org.openjdk.jmc.flightrecorder.stacktrace.Messages.getString(key);
49 	}
50 
51 	/**
52 	 * Localized Flameview Message
53 	 * 
54 	 * @param key
55 	 *            flameview message
56 	 * @param values
57 	 *            message values
58 	 * @return message
59 	 */
60 	public static String getFlameviewMessage(String key, Object ... values) {
61 		if (values == null || values.length == 0) {
62 			return getFlameviewMessage(key);
63 		} else {
64 			return MessageFormat.format(getFlameviewMessage(key), values);
65 		}
66 	}
67 
68 	private static String getFlameviewMessage(String key) {
69 		return org.openjdk.jmc.flightrecorder.flameview.Messages.getString(key);
70 	}
71 
72 }
