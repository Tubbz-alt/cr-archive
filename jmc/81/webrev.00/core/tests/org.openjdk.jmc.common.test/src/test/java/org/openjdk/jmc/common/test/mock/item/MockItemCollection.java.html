 33 package org.openjdk.jmc.common.test.mock.item;
 34 
 35 import java.util.ArrayList;
 36 import java.util.Collections;
 37 import java.util.Iterator;
 38 import java.util.List;
 39 import java.util.Set;
 40 
 41 import org.openjdk.jmc.common.collection.IteratorToolkit;
 42 import org.openjdk.jmc.common.item.IAggregator;
 43 import org.openjdk.jmc.common.item.IItem;
 44 import org.openjdk.jmc.common.item.IItemCollection;
 45 import org.openjdk.jmc.common.item.IItemConsumer;
 46 import org.openjdk.jmc.common.item.IItemFilter;
 47 import org.openjdk.jmc.common.item.IItemIterable;
 48 import org.openjdk.jmc.common.item.IType;
 49 import org.openjdk.jmc.common.unit.IQuantity;
 50 import org.openjdk.jmc.common.unit.IRange;
 51 
 52 public class MockItemCollection<T, CT extends IType<?>> implements IItemCollection {
 53 	private List<IItem> items = new ArrayList<>();
 54 	private IType<?> type;
 55 
 56 	public MockItemCollection(T[] values, IType<?> type) {
 57 		build(values, type);
 58 		this.type = type;
 59 	}
 60 
 61 	private void build(T[] values, IType<?> type) {
 62 		long index = 0;
 63 		for (T value : values) {
 64 			items.add(new MockItem<T, CT>(value, type, index++));
 65 		}
 66 	}
 67 
 68 	@Override
 69 	public IItemCollection apply(IItemFilter filter) {
 70 		// Filtering not supported in this implementation
 71 		return null;
 72 	}
 73 
 74 	@Override
 75 	public Iterator<IItemIterable> iterator() {
 76 		return IteratorToolkit.of(new IItemIterable[] {new MockItemIterable<CT>(items, type)});
 77 	}
 78 
 79 	@Override
 80 	public <V, C extends IItemConsumer<C>> V getAggregate(IAggregator<V, C> aggregator) {
 81 		return aggregate(aggregator, iterator());
 82 	}
 83 
 84 	@Override
 85 	public boolean hasItems() {
 86 		return !items.isEmpty();
 87 	}
 88 
 89 	// Copied from EventCollection
 90 	private static <V, C extends IItemConsumer<C>> V aggregate(
 91 		final IAggregator<V, C> aggregator, final Iterator<? extends IItemIterable> items) {
 92 		return aggregator.getValue(new Iterator<C>() {
 93 
 94 			IItemIterable next = findNext();
 95 
 96 			@Override
 97 			public boolean hasNext() {
 98 				return next != null;
 99 			}
100 
101 			@Override
102 			public C next() {
103 				C calc = aggregator.newItemConsumer(next.getType());
104 				Iterator<? extends IItem> iterator = next.iterator();
105 				while (iterator.hasNext()) {
106 					calc.consume(iterator.next());
107 				}
108 				next = findNext();
109 				return calc;
110 			}
111 
112 			IItemIterable findNext() {
113 				while (items.hasNext()) {
114 					IItemIterable ii = items.next();
115 					if (aggregator.acceptType(ii.getType())) {
116 						return ii;
117 					}
118 				}
119 				return null;
120 			}
121 
122 			@Override
123 			public void remove() {
124 				throw new UnsupportedOperationException();
125 			}
126 		});
127 	}
128 
129 	@Override
130 	public Set<IRange<IQuantity>> getChunkRanges() {
131 		return Collections.emptySet();
132 	}
133 }
