24 /*
25  * @test
26  * @author Gary Ellison
27  * @bug 4170635
28  * @summary Verify equals()/hashCode() contract honored
29  * @modules java.base/sun.security.util
30  *          java.base/sun.security.x509
31  */
32 
33 import java.io.*;
34 import sun.security.util.*;
35 import sun.security.x509.*;
36 
37 
38 public class DerValueEqualsHashCode {
39 
40     public static void main(String[] args) throws Exception {
41 
42         String name = "CN=user";
43         X500Name dn = new X500Name(name);
44 
45         DerOutputStream deros;
46         byte[] ba;
47         //
48         // get busy
49         deros = new DerOutputStream();
50         dn.derEncode(deros);
51         ba = deros.toByteArray();
52 
53         DerValue dv1 = new DerValue(ba);
54         DerValue dv2 = new DerValue(ba);
55 
56         if ( (dv1.equals(dv2)) == (dv1.hashCode()==dv2.hashCode()) )
57             System.out.println("PASSED");
58         else
59             throw new Exception("FAILED equals()/hashCode() contract");
60 
61     }
62 
63 }
