24 /*
25  * @test
26  * @bug 8011867 8242151
27  * @summary Accept unknown PKCS #9 attributes
28  * @library /test/lib
29  * @modules java.base/sun.security.pkcs
30  *          java.base/sun.security.util
31  */
32 
33 import java.io.*;
34 import java.util.Arrays;
35 
36 import sun.security.pkcs.PKCS9Attribute;
37 import sun.security.util.DerOutputStream;
38 import sun.security.util.DerValue;
39 import sun.security.util.ObjectIdentifier;
40 import jdk.test.lib.hexdump.HexPrinter;
41 
42 public class UnknownAttribute {
43 
44     public static void main(String[] args) throws Exception {
45         // Unknown attr
46         PKCS9Attribute p1 = new PKCS9Attribute(
47                 PKCS9Attribute.CHALLENGE_PASSWORD_OID, "t0p5ecr3t");
48         if (!p1.isKnown()) {
49             throw new Exception();
50         }
51         // Unknown attr from DER
52         byte[] data = {
53                 0x30, 0x08,                 // SEQUENCE OF
54                 0x06, 0x02, 0x2A, 0x03,     // OID 1.2.3 and
55                 0x31, 0x02, 0x05, 0x00      // an empty SET
56         };
57         PKCS9Attribute p2 = new PKCS9Attribute(new DerValue(data));
58         if (p2.isKnown()) {
59             throw new Exception();
60         }
61         DerOutputStream bout = new DerOutputStream();
62         p2.derEncode(bout);
63         HexPrinter.simple().dest(System.err).format(bout.toByteArray());
64         if (!Arrays.equals(data, bout.toByteArray())) {
65             throw new Exception();
66         }
67         // Unknown attr from value
68         try {
69             new PKCS9Attribute(ObjectIdentifier.of("1.2.3"), "hello");
70             throw new Exception();
71         } catch (IllegalArgumentException iae) {
72             // Good. Unknown attr must have byte[] value type
73         }
74         PKCS9Attribute p3 = new PKCS9Attribute(
75                 ObjectIdentifier.of("1.2.3"), new byte[]{0x31,0x02,0x05,0x00});
76         if (p3.isKnown()) {
77             throw new Exception();
78         }
79         bout = new DerOutputStream();
80         p3.derEncode(bout);
81         if (!Arrays.equals(data, bout.toByteArray())) {
82             throw new Exception();
83         }
84     }
85 }
