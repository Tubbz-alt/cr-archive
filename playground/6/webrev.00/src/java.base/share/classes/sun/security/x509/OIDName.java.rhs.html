 26 package sun.security.x509;
 27 
 28 import java.io.IOException;
 29 
 30 import sun.security.util.*;
 31 
 32 /**
 33  * This class implements the OIDName as required by the GeneralNames
 34  * ASN.1 object.
 35  *
 36  * @author Amit Kapoor
 37  * @author Hemma Prafullchandra
 38  * @see GeneralName
 39  * @see GeneralNames
 40  * @see GeneralNameInterface
 41  */
 42 public class OIDName implements GeneralNameInterface {
 43      private ObjectIdentifier oid;
 44 
 45     /**
 46      * Create the OIDName object from the passed encoded Der value.
 47      *
 48      * @param derValue the encoded DER OIDName.
 49      * @exception IOException on error.
 50      */
 51     public OIDName(DerValue derValue) throws IOException {
 52         oid = derValue.getOID();
 53     }
 54 
 55     /**
 56      * Create the OIDName object with the specified name.
 57      *
 58      * @param oid the OIDName.
 59      */
 60     public OIDName(ObjectIdentifier oid) {
 61         this.oid = oid;
 62     }
 63 
 64     /**
 65      * Create the OIDName from the String form of the OID
 66      *
 67      * @param name the OIDName in form "x.y.z..."
 68      * @throws IOException on error
 69      */
 70     public OIDName(String name) throws IOException {
 71         try {
 72             oid = ObjectIdentifier.of(name);
 73         } catch (Exception e) {
 74             throw new IOException("Unable to create OIDName: " + e);
 75         }
 76     }
 77 
 78     /**
 79      * Return the type of the GeneralName.
 80      */
 81     public int getType() {
 82         return (GeneralNameInterface.NAME_OID);
 83     }
 84 
 85     /**
 86      * Encode the OID name into the DerOutputStream.
 87      *
 88      * @param out the DER stream to encode the OIDName to.
 89      * @exception IOException on encoding errors.
 90      */
 91     @Override
 92     public void derEncode(DerOutputStream out) {
 93         out.putOID(oid);
 94     }
 95 
 96     /**
 97      * Convert the name into user readable string.
 98      */
 99     public String toString() {
100         return ("OIDName: " + oid.toString());
101     }
102 
103     /**
104      * Returns this OID name.
105      */
106     public ObjectIdentifier getOID() {
107         return oid;
108     }
109 
110     /**
111      * Compares this name with another, for equality.
112      *
113      * @return true iff the names are identical
114      */
115     public boolean equals(Object obj) {
116         if (this == obj)
117             return true;
118 
119         if (!(obj instanceof OIDName))
120             return false;
121 
122         OIDName other = (OIDName)obj;
123 
124         return oid.equals(other.oid);
125     }
126 
127     /**
128      * Returns the hash code value for this object.
129      *
130      * @return a hash code value for this object.
131      */
132     public int hashCode() {
133         return oid.hashCode();
134     }
135 
136     /**
137      * Return type of constraint inputName places on this name:<ul>
138      *   <li>NAME_DIFF_TYPE = -1: input name is different type from name (i.e. does not constrain).
139      *   <li>NAME_MATCH = 0: input name matches name.
140      *   <li>NAME_NARROWS = 1: input name narrows name (is lower in the naming subtree)
141      *   <li>NAME_WIDENS = 2: input name widens name (is higher in the naming subtree)
142      *   <li>NAME_SAME_TYPE = 3: input name does not match or narrow name, but is same type.
143      * </ul>.  These results are used in checking NameConstraints during
144      * certification path verification.
145      *
146      * @param inputName to be checked for being constrained
147      * @return constraint type above
148      * @throws UnsupportedOperationException if name is not exact match, but narrowing and widening are
149      *          not supported for this name type.
150      */
151     public int constrains(GeneralNameInterface inputName) throws UnsupportedOperationException {
152         int constraintType;
153         if (inputName == null)
154             constraintType = NAME_DIFF_TYPE;
155         else if (inputName.getType() != NAME_OID)
156             constraintType = NAME_DIFF_TYPE;
157         else if (this.equals((OIDName)inputName))
158             constraintType = NAME_MATCH;
159         else
160             //widens and narrows not defined in RFC 5280 for OIDName (aka registeredID)
161             throw new UnsupportedOperationException("Narrowing and widening are not supported for OIDNames");
162         return constraintType;
163     }
164 
165     /**
166      * Return subtree depth of this name for purposes of determining
167      * NameConstraints minimum and maximum bounds and for calculating
168      * path lengths in name subtrees.
169      *
170      * @return distance of name from root
171      * @throws UnsupportedOperationException if not supported for this name type
172      */
173     public int subtreeDepth() throws UnsupportedOperationException {
174         throw new UnsupportedOperationException("subtreeDepth() not supported for OIDName.");
175    }
176 }
