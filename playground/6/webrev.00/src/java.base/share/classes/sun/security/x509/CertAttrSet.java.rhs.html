 26 package sun.security.x509;
 27 
 28 import sun.security.util.DerOutputStream;
 29 
 30 import java.io.IOException;
 31 import java.io.OutputStream;
 32 import java.security.cert.CertificateException;
 33 import java.util.Enumeration;
 34 
 35 /**
 36  * This interface defines the methods required of a certificate attribute.
 37  * Examples of X.509 certificate attributes are Validity, Issuer_Name, and
 38  * Subject Name. A CertAttrSet may comprise one attribute or many
 39  * attributes.
 40  * <p>
 41  * A CertAttrSet itself can also be comprised of other sub-sets.
 42  * In the case of X.509 V3 certificates, for example, the "extensions"
 43  * attribute has subattributes, such as those for KeyUsage and
 44  * AuthorityKeyIdentifier.
 45  *
 46  * @author Amit Kapoor
 47  * @author Hemma Prafullchandra
 48  * @see CertificateException
 49  */
 50 public interface CertAttrSet<T> {
 51     /**
 52      * Returns a short string describing this certificate attribute.
 53      *
 54      * @return value of this certificate attribute in
 55      *         printable form.
 56      */
 57     String toString();
 58 
 59     /**
 60      * Encodes the attribute to the output stream in a format
 61      * that can be parsed by the <code>decode</code> method.
 62      *
 63      * @param out the OutputStream to encode the attribute to.
 64      *
 65      * @exception CertificateException on encoding or validity errors.
 66      * @exception IOException on other errors.
 67      */
 68     void encode(DerOutputStream out) throws CertificateException;

 69 
 70     /**
 71      * Sets an attribute value within this CertAttrSet.
 72      *
 73      * @param name the name of the attribute (e.g. "x509.info.key")
 74      * @param obj the attribute object.
 75      *
 76      * @exception CertificateException on attribute handling errors.
 77      * @exception IOException on other errors.
 78      */
 79     void set(String name, Object obj)
 80         throws CertificateException, IOException;
 81 
 82     /**
 83      * Gets an attribute value for this CertAttrSet.
 84      *
 85      * @param name the name of the attribute to return.
 86      *
 87      * @exception CertificateException on attribute handling errors.
 88      * @exception IOException on other errors.
 89      */
 90     Object get(String name)
 91         throws CertificateException, IOException;
 92 
 93     /**
 94      * Deletes an attribute value from this CertAttrSet.
 95      *
 96      * @param name the name of the attribute to delete.
 97      *
 98      * @exception CertificateException on attribute handling errors.
 99      * @exception IOException on other errors.
100      */
101     void delete(String name)
102         throws CertificateException, IOException;
103 
104     /**
105      * Returns an enumeration of the names of the attributes existing within
106      * this attribute.
107      *
108      * @return an enumeration of the attribute names.
109      */
110     Enumeration<T> getElements();
111 
112     /**
113      * Returns the name (identifier) of this CertAttrSet.
114      *
115      * @return the name of this CertAttrSet.
116      */
117     String getName();
118 }
