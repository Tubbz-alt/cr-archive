 418     private void run(String[] args, PrintStream out) throws Exception {
 419         try {
 420             args = parseArgs(args);
 421             if (command != null) {
 422                 doCommands(out);
 423             }
 424         } catch (Exception e) {
 425             System.out.println(rb.getString("keytool.error.") + e);
 426             if (verbose) {
 427                 e.printStackTrace(System.out);
 428             }
 429             if (!debug) {
 430                 System.exit(1);
 431             } else {
 432                 throw e;
 433             }
 434         } finally {
 435             printWeakWarnings(false);
 436             for (char[] pass : passwords) {
 437                 if (pass != null) {
 438                     Arrays.fill(pass, ' ');
 439                     pass = null;
 440                 }
 441             }
 442 
 443             if (ksStream != null) {
 444                 ksStream.close();
 445             }
 446         }
 447     }

 449     /**
 450      * Parse command line arguments.
 451      */
 452     String[] parseArgs(String[] args) throws Exception {
 453 
 454         int i=0;
 455         boolean help = args.length == 0;
 456 
 457         String confFile = null;
 458 
 459         // Records all commands and options set. Used to check dups.
 460         Set<String> optionsSet = new HashSet<>();
 461 
 462         for (i=0; i < args.length; i++) {
 463             String flags = args[i];
 464             if (flags.startsWith("-")) {
 465                 String lowerFlags = flags.toLowerCase(Locale.ROOT);
 466                 if (optionsSet.contains(lowerFlags)) {
 467                     switch (lowerFlags) {
 468                         case "-ext":
 469                         case "-id":
 470                         case "-provider":
 471                         case "-addprovider":
 472                         case "-providerclass":
 473                         case "-providerarg":
 474                             // These options are allowed multiple times
 475                             break;
 476                         default:
 477                             weakWarnings.add(String.format(
 478                                     rb.getString("option.1.set.twice"),
 479                                     lowerFlags));
 480                     }
 481                 } else {
 482                     optionsSet.add(lowerFlags);
 483                 }
 484                 if (collator.compare(flags, "-conf") == 0) {
 485                     if (i == args.length - 1) {
 486                         errorNeedArgument(flags);
 487                     }
 488                     confFile = args[++i];
 489                 } else {
 490                     Command c = Command.getCommand(flags);
 491                     if (c != null) {
 492                         if (command == null) {
 493                             command = c;
 494                         } else {
 495                             throw new Exception(String.format(
 496                                     rb.getString("multiple.commands.1.2"),
 497                                     command.name, c.name));
 498                         }
 499                     }
 500                 }
 501             }
 502         }
 503 
 504         if (confFile != null && command != null) {
 505             args = KeyStoreUtil.expandArgs("keytool", confFile,
 506                     command.toString(),
 507                     command.getAltName(), args);
 508         }
 509 
 510         debug = Arrays.stream(args).anyMatch(
 511                 x -> collator.compare(x, "-debug") == 0);
 512 
 513         if (debug) {
 514             // No need to localize debug output
 515             System.out.println("Command line args: " +
 516                     Arrays.toString(args));
 517         }
 518 
 519         for (i=0; (i < args.length) && args[i].startsWith("-"); i++) {
 520 
 521             String flags = args[i];
 522 
 523             // Check if the last option needs an arg
 524             if (i == args.length - 1) {
 525                 for (Option option: Option.values()) {
 526                     // Only options with an arg need to be checked
 527                     if (collator.compare(flags, option.toString()) == 0) {
 528                         if (option.arg != null) errorNeedArgument(flags);
 529                         break;
 530                     }
 531                 }
 532             }
 533 
 534             /*
 535              * Check modifiers
 536              */
 537             String modifier = null;
 538             int pos = flags.indexOf(':');
 539             if (pos > 0) {
 540                 modifier = flags.substring(pos+1);
 541                 flags = flags.substring(0, pos);
 542             }
 543 
 544             /*
 545              * command modes
 546              */
 547             Command c = Command.getCommand(flags);
 548 
 549             if (c != null) {
 550                 command = c;
 551             } else if (collator.compare(flags, "--help") == 0 ||
 552                        collator.compare(flags, "-h") == 0 ||
 553                        collator.compare(flags, "-?") == 0 ||
 554                        // -help: legacy.
 555                        collator.compare(flags, "-help") == 0) {
 556                 help = true;
 557             } else if (collator.compare(flags, "-conf") == 0) {
 558                 i++;
 559             } else if (collator.compare(flags, "-nowarn") == 0) {
 560                 nowarn = true;
 561             } else if (collator.compare(flags, "-keystore") == 0) {
 562                 ksfname = args[++i];
 563                 if (new File(ksfname).getCanonicalPath().equals(
 564                         new File(KeyStoreUtil.getCacerts()).getCanonicalPath())) {
 565                     System.err.println(rb.getString("warning.cacerts.option"));
 566                 }
 567             } else if (collator.compare(flags, "-destkeystore") == 0) {
 568                 ksfname = args[++i];
 569             } else if (collator.compare(flags, "-cacerts") == 0) {
 570                 cacerts = true;
 571             } else if (collator.compare(flags, "-storepass") == 0 ||
 572                     collator.compare(flags, "-deststorepass") == 0) {
 573                 storePass = getPass(modifier, args[++i]);
 574                 passwords.add(storePass);
 575             } else if (collator.compare(flags, "-storetype") == 0 ||
 576                     collator.compare(flags, "-deststoretype") == 0) {
 577                 storetype = KeyStoreUtil.niceStoreTypeName(args[++i]);
 578             } else if (collator.compare(flags, "-srcstorepass") == 0) {
 579                 srcstorePass = getPass(modifier, args[++i]);
 580                 passwords.add(srcstorePass);
 581             } else if (collator.compare(flags, "-srcstoretype") == 0) {
 582                 srcstoretype = KeyStoreUtil.niceStoreTypeName(args[++i]);
 583             } else if (collator.compare(flags, "-srckeypass") == 0) {
 584                 srckeyPass = getPass(modifier, args[++i]);
 585                 passwords.add(srckeyPass);
 586             } else if (collator.compare(flags, "-srcprovidername") == 0) {
 587                 srcProviderName = args[++i];
 588             } else if (collator.compare(flags, "-providername") == 0 ||
 589                     collator.compare(flags, "-destprovidername") == 0) {
 590                 providerName = args[++i];

 757             (command == KEYPASSWD || command == STOREPASSWD)) {

 759                         (".storepasswd.and.keypasswd.commands.not.supported.if.storetype.is.{0}"), storetype));
 760         }
 761 
 762         if (token && (keyPass != null || newPass != null || destKeyPass != null)) {
 763             throw new IllegalArgumentException(MessageFormat.format(rb.getString
 764                 (".keypass.and.new.can.not.be.specified.if.storetype.is.{0}"), storetype));

 767         if (protectedPath) {

 777                 throw new IllegalArgumentException(rb.getString

 779             }

 783             if (storePass != null || keyPass != null ||
 784                     newPass != null || destKeyPass != null) {
 785                 throw new IllegalArgumentException(rb.getString
 786                         ("if.keystore.is.not.password.protected.then.storepass.keypass.and.new.must.not.be.specified"));
 787             }
 788         }
 789 
 790         if (KeyStoreUtil.isWindowsKeyStore(srcstoretype)) {
 791             if (srcstorePass != null || srckeyPass != null) {
 792                 throw new IllegalArgumentException(rb.getString
 793                         ("if.source.keystore.is.not.password.protected.then.srcstorepass.and.srckeypass.must.not.be.specified"));
 794             }
 795         }
 796 
 797         if (validity <= (long)0) {
 798             throw new Exception
 799                 (rb.getString("Validity.must.be.greater.than.zero"));
 800         }
 801 
 802         // Try to load and install specified provider
 803         if (providers != null) {
 804             for (Pair<String, String> provider : providers) {
 805                 try {
 806                     KeyStoreUtil.loadProviderByName(
 807                             provider.fst, provider.snd);
 808                     if (debug) {
 809                         System.out.println("loadProviderByName: " + provider.fst);
 810                     }
 811                 } catch (IllegalArgumentException e) {
 812                     throw new Exception(String.format(rb.getString(
 813                             "provider.name.not.found"), provider.fst));
 814                 }
 815             }
 816         }
 817         if (providerClasses != null) {
 818             ClassLoader cl = null;
 819             if (pathlist != null) {
 820                 String path = null;
 821                 path = PathList.appendPath(
 822                         path, System.getProperty("java.class.path"));
 823                 path = PathList.appendPath(
 824                         path, System.getProperty("env.class.path"));
 825                 path = PathList.appendPath(path, pathlist);
 826 
 827                 URL[] urls = PathList.pathToURLs(path);
 828                 cl = new URLClassLoader(urls);
 829             } else {
 830                 cl = ClassLoader.getSystemClassLoader();
 831             }
 832             for (Pair<String, String> provider : providerClasses) {
 833                 try {
 834                     KeyStoreUtil.loadProviderByClass(
 835                             provider.fst, provider.snd, cl);
 836                     if (debug) {
 837                         System.out.println("loadProviderByClass: " + provider.fst);
 838                     }
 839                 } catch (ClassCastException cce) {
 840                     throw new Exception(String.format(rb.getString(
 841                             "provclass.not.a.provider"), provider.fst));
 842                 } catch (IllegalArgumentException e) {
 843                     throw new Exception(String.format(rb.getString(
 844                             "provider.class.not.found"), provider.fst), e.getCause());
 845                 }
 846             }
 847         }
 848 
 849         if (command == LIST && verbose && rfc) {
 850             System.err.println(rb.getString
 851                 ("Must.not.specify.both.v.and.rfc.with.list.command"));
 852             tinyHelp();
 853         }
 854 
 855         // Make sure provided passwords are at least 6 characters long
 856         if (command == GENKEYPAIR && keyPass!=null && keyPass.length < 6) {
 857             throw new Exception(rb.getString
 858                 ("Key.password.must.be.at.least.6.characters"));
 859         }
 860         if (newPass != null && newPass.length < 6) {
 861             throw new Exception(rb.getString
 862                 ("New.password.must.be.at.least.6.characters"));
 863         }
 864         if (destKeyPass != null && destKeyPass.length < 6) {
 865             throw new Exception(rb.getString
 866                 ("New.password.must.be.at.least.6.characters"));
 867         }
 868 
 869         // Set this before inplaceImport check so we can compare name.
 870         if (ksfname == null) {
 871             ksfname = System.getProperty("user.home") + File.separator
 872                     + ".keystore";
 873         }
 874 
 875         KeyStore srcKeyStore = null;
 876         if (command == IMPORTKEYSTORE) {
 877             inplaceImport = inplaceImportCheck();
 878             if (inplaceImport) {
 879                 // We load srckeystore first so we have srcstorePass that
 880                 // can be assigned to storePass
 881                 srcKeyStore = loadSourceKeyStore();
 882                 if (storePass == null) {
 883                     storePass = srcstorePass;
 884                 }
 885             }
 886         }
 887 
 888         // Check if keystore exists.
 889         // If no keystore has been specified at the command line, try to use
 890         // the default, which is located in $HOME/.keystore.
 891         // No need to check if isKeyStoreRelated(command) is false.
 892 
 893         // DO NOT open the existing keystore if this is an in-place import.
 894         // The keystore should be created as brand new.
 895         if (isKeyStoreRelated(command) && !nullStream && !inplaceImport) {
 896             try {
 897                 ksfile = new File(ksfname);
 898                 // Check if keystore file is empty
 899                 if (ksfile.exists() && ksfile.length() == 0) {
 900                     throw new Exception(rb.getString
 901                             ("Keystore.file.exists.but.is.empty.") + ksfname);
 902                 }
 903                 ksStream = new FileInputStream(ksfile);
 904             } catch (FileNotFoundException e) {
 905                 // These commands do not need the keystore to be existing.
 906                 // Either it will create a new one or the keystore is
 907                 // optional (i.e. PRINTCRL).
 908                 if (command != GENKEYPAIR &&
 909                         command != GENSECKEY &&
 910                         command != IDENTITYDB &&
 911                         command != IMPORTCERT &&
 912                         command != IMPORTPASS &&
 913                         command != IMPORTKEYSTORE &&
 914                         command != PRINTCRL) {
 915                     throw new Exception(rb.getString
 916                             ("Keystore.file.does.not.exist.") + ksfname);
 917                 }
 918             }
 919         }
 920 
 921         if ((command == KEYCLONE || command == CHANGEALIAS)
 922                 && dest == null) {
 923             dest = getAlias("destination");
 924             if ("".equals(dest)) {
 925                 throw new Exception(rb.getString
 926                         ("Must.specify.destination.alias"));
 927             }
 928         }
 929 
 930         if (command == DELETE && alias == null) {
 931             alias = getAlias(null);
 932             if ("".equals(alias)) {
 933                 throw new Exception(rb.getString("Must.specify.alias"));
 934             }
 935         }
 936 
 937         // Create new keystore
 938         // Probe for keystore type when filename is available
 939         if (ksfile != null && ksStream != null && providerName == null &&
 940                 storetype == null && !inplaceImport) {
 941             keyStore = KeyStore.getInstance(ksfile, storePass);
 942             storetype = keyStore.getType();
 943             if (storetype.equalsIgnoreCase("pkcs12")) {
 944                 isPasswordlessKeyStore = PKCS12KeyStore.isPasswordless(ksfile);
 945             }
 946         } else {
 947             if (storetype == null) {
 948                 storetype = KeyStore.getDefaultType();
 949             }
 950             if (providerName == null) {
 951                 keyStore = KeyStore.getInstance(storetype);
 952             } else {
 953                 keyStore = KeyStore.getInstance(storetype, providerName);
 954             }
 955             // When creating a new pkcs12 file, Do not prompt for storepass
 956             // if certProtectionAlgorithm and macAlgorithm are both NONE.
 957             if (storetype.equalsIgnoreCase("pkcs12")) {
 958                 isPasswordlessKeyStore =
 959                         "NONE".equals(SecurityProperties.privilegedGetOverridable(
 960                                 "keystore.pkcs12.certProtectionAlgorithm"))
 961                         && "NONE".equals(SecurityProperties.privilegedGetOverridable(
 962                                 "keystore.pkcs12.macAlgorithm"));
 963             }
 964 
 965             /*
 966              * Load the keystore data.
 967              *
 968              * At this point, it's OK if no keystore password has been provided.
 969              * We want to make sure that we can load the keystore data, i.e.,
 970              * the keystore data has the right format. If we cannot load the
 971              * keystore, why bother asking the user for his or her password?
 972              * Only if we were able to load the keystore, and no keystore
 973              * password has been provided, will we prompt the user for the
 974              * keystore password to verify the keystore integrity.
 975              * This means that the keystore is loaded twice: first load operation
 976              * checks the keystore format, second load operation verifies the
 977              * keystore integrity.
 978              *
 979              * If the keystore password has already been provided (at the
 980              * command line), however, the keystore is loaded only once, and the
 981              * keystore format and integrity are checked "at the same time".
 982              *
 983              * Null stream keystores are loaded later.
 984              */
 985             if (!nullStream) {
 986                 if (inplaceImport) {
 987                     keyStore.load(null, storePass);
 988                 } else {
 989                     keyStore.load(ksStream, storePass);
 990                 }
 991                 if (ksStream != null) {
 992                     ksStream.close();
 993                 }
 994             }
 995         }
 996 
 997         if (P12KEYSTORE.equalsIgnoreCase(storetype) && command == KEYPASSWD) {
 998             throw new UnsupportedOperationException(rb.getString
 999                     (".keypasswd.commands.not.supported.if.storetype.is.PKCS12"));
1000         }
1001 
1002         // All commands that create or modify the keystore require a keystore
1003         // password.
1004 
1005         if (nullStream && storePass != null) {
1006             keyStore.load(null, storePass);
1007         } else if (!nullStream && storePass != null) {
1008             // If we are creating a new non nullStream-based keystore,
1009             // insist that the password be at least 6 characters
1010             if (ksStream == null && storePass.length < 6) {
1011                 throw new Exception(rb.getString
1012                         ("Keystore.password.must.be.at.least.6.characters"));
1013             }
1014         } else if (storePass == null) {
1015             if (!protectedPath && !KeyStoreUtil.isWindowsKeyStore(storetype)
1016                     && isKeyStoreRelated(command)
1017                     && !isPasswordlessKeyStore) {
1018                 if (command == CERTREQ ||
1019                         command == DELETE ||
1020                         command == GENKEYPAIR ||
1021                         command == GENSECKEY ||
1022                         command == IMPORTCERT ||
1023                         command == IMPORTPASS ||
1024                         command == IMPORTKEYSTORE ||
1025                         command == KEYCLONE ||
1026                         command == CHANGEALIAS ||
1027                         command == SELFCERT ||
1028                         command == STOREPASSWD ||
1029                         command == KEYPASSWD ||
1030                         command == IDENTITYDB) {
1031                     int count = 0;
1032                     do {
1033                         if (command == IMPORTKEYSTORE) {
1034                             System.err.print
1035                                     (rb.getString("Enter.destination.keystore.password."));
1036                         } else {
1037                             System.err.print
1038                                     (rb.getString("Enter.keystore.password."));
1039                         }
1040                         System.err.flush();
1041                         storePass = Password.readPassword(System.in);
1042                         passwords.add(storePass);
1043 
1044                         // If we are creating a new non nullStream-based keystore,
1045                         // insist that the password be at least 6 characters
1046                         if (!nullStream && (storePass == null || storePass.length < 6)) {
1047                             System.err.println(rb.getString
1048                                     ("Keystore.password.is.too.short.must.be.at.least.6.characters"));
1049                             storePass = null;
1050                         }
1051 
1052                         // If the keystore file does not exist and needs to be
1053                         // created, the storepass should be prompted twice.
1054                         if (storePass != null && !nullStream && ksStream == null) {
1055                             System.err.print(rb.getString("Re.enter.new.password."));
1056                             char[] storePassAgain = Password.readPassword(System.in);
1057                             passwords.add(storePassAgain);
1058                             if (!Arrays.equals(storePass, storePassAgain)) {
1059                                 System.err.println
1060                                         (rb.getString("They.don.t.match.Try.again"));
1061                                 storePass = null;
1062                             }
1063                         }
1064 
1065                         count++;
1066                     } while ((storePass == null) && count < 3);
1067 
1068 
1069                     if (storePass == null) {
1070                         System.err.println
1071                                 (rb.getString("Too.many.failures.try.later"));
1072                         return;
1073                     }
1074                 } else {
1075                     // here we have EXPORTCERT and LIST (info valid until STOREPASSWD)
1076                     if (command != PRINTCRL) {
1077                         System.err.print(rb.getString("Enter.keystore.password."));
1078                         System.err.flush();
1079                         storePass = Password.readPassword(System.in);
1080                         passwords.add(storePass);
1081                     }
1082                 }
1083             }
1084 
1085             // Now load a nullStream-based keystore,
1086             // or verify the integrity of an input stream-based keystore
1087             if (nullStream) {
1088                 keyStore.load(null, storePass);
1089             } else if (ksStream != null) {
1090                 ksStream = new FileInputStream(ksfile);
1091                 keyStore.load(ksStream, storePass);
1092                 ksStream.close();
1093             }
1094         }
1095 
1096         if (storePass != null && P12KEYSTORE.equalsIgnoreCase(storetype)) {
1097             MessageFormat form = new MessageFormat(rb.getString(
1098                 "Warning.Different.store.and.key.passwords.not.supported.for.PKCS12.KeyStores.Ignoring.user.specified.command.value."));
1099             if (keyPass != null && !Arrays.equals(storePass, keyPass)) {
1100                 Object[] source = {"-keypass"};
1101                 System.err.println(form.format(source));
1102                 keyPass = storePass;
1103             }
1104             if (destKeyPass != null && !Arrays.equals(storePass, destKeyPass)) {
1105                 Object[] source = {"-destkeypass"};
1106                 System.err.println(form.format(source));
1107                 destKeyPass = storePass;
1108             }
1109         }
1110 
1111         // -trustcacerts can only be specified on -importcert.
1112         // Reset it so that warnings on CA cert will remain for
1113         // -printcert, etc.
1114         if (command != IMPORTCERT) {
1115             trustcacerts = false;
1116         }
1117 
1118         if (trustcacerts) {
1119             caks = KeyStoreUtil.getCacertsKeyStore();
1120         }
1121 
1122         // Perform the specified command
1123         if (command == CERTREQ) {
1124             if (filename != null) {
1125                 try (PrintStream ps = new PrintStream(new FileOutputStream
1126                                                       (filename))) {
1127                     doCertReq(alias, sigAlgName, ps);
1128                 }
1129             } else {
1130                 doCertReq(alias, sigAlgName, out);
1131             }
1132             if (verbose && filename != null) {
1133                 MessageFormat form = new MessageFormat(rb.getString
1134                         ("Certification.request.stored.in.file.filename."));
1135                 Object[] source = {filename};
1136                 System.err.println(form.format(source));
1137                 System.err.println(rb.getString("Submit.this.to.your.CA"));
1138             }
1139         } else if (command == DELETE) {
1140             doDeleteEntry(alias);
1141             kssave = true;
1142         } else if (command == EXPORTCERT) {
1143             if (filename != null) {
1144                 try (PrintStream ps = new PrintStream(new FileOutputStream
1145                                                    (filename))) {
1146                     doExportCert(alias, ps);
1147                 }
1148             } else {
1149                 doExportCert(alias, out);
1150             }
1151             if (filename != null) {
1152                 MessageFormat form = new MessageFormat(rb.getString
1153                         ("Certificate.stored.in.file.filename."));
1154                 Object[] source = {filename};
1155                 System.err.println(form.format(source));
1156             }
1157         } else if (command == GENKEYPAIR) {
1158             if (keyAlgName == null) {
1159                 throw new Exception(rb.getString(
1160                         "keyalg.option.missing.error"));
1161             }
1162             doGenKeyPair(alias, dname, keyAlgName, keysize, groupName, sigAlgName);
1163             kssave = true;
1164         } else if (command == GENSECKEY) {
1165             if (keyAlgName == null) {
1166                 throw new Exception(rb.getString(
1167                         "keyalg.option.missing.error"));
1168             }
1169             doGenSecretKey(alias, keyAlgName, keysize);
1170             kssave = true;
1171         } else if (command == IMPORTPASS) {
1172             if (keyAlgName == null) {
1173                 keyAlgName = "PBE";
1174             }
1175             // password is stored as a secret key
1176             doGenSecretKey(alias, keyAlgName, keysize);
1177             kssave = true;
1178         } else if (command == IDENTITYDB) {
1179             if (filename != null) {
1180                 try (InputStream inStream = new FileInputStream(filename)) {
1181                     doImportIdentityDatabase(inStream);
1182                 }
1183             } else {
1184                 doImportIdentityDatabase(System.in);
1185             }
1186         } else if (command == IMPORTCERT) {
1187             InputStream inStream = System.in;
1188             if (filename != null) {

2467      * Loads CRLs from a source. This method is also called in JarSigner.
2468      * @param src the source, which means System.in if null, or a URI,
2469      *        or a bare file path name
2470      */
2471     public static Collection<? extends CRL> loadCRLs(String src) throws Exception {
2472         InputStream in = null;
2473         URI uri = null;
2474         if (src == null) {
2475             in = System.in;
2476         } else {
2477             try {
2478                 uri = new URI(src);
2479                 if (uri.getScheme().equals("ldap")) {
2480                     // No input stream for LDAP
2481                 } else {
2482                     in = uri.toURL().openStream();
2483                 }
2484             } catch (Exception e) {
2485                 try {
2486                     in = new FileInputStream(src);
2487                 } catch (Exception e2) {
2488                     if (uri == null || uri.getScheme() == null) {
2489                         throw e2;   // More likely a bare file path
2490                     } else {
2491                         throw e;    // More likely a protocol or network problem
2492                     }
2493                 }
2494             }
2495         }
2496         if (in != null) {
2497             try {
2498                 // Read the full stream before feeding to X509Factory,
2499                 // otherwise, keytool -gencrl | keytool -printcrl
2500                 // might not work properly, since -gencrl is slow
2501                 // and there's no data in the pipe at the beginning.
2502                 ByteArrayOutputStream bout = new ByteArrayOutputStream();
2503                 byte[] b = new byte[4096];
2504                 while (true) {
2505                     int len = in.read(b);
2506                     if (len < 0) break;
2507                     bout.write(b, 0, len);
2508                 }
2509                 return CertificateFactory.getInstance("X509").generateCRLs(
2510                         new ByteArrayInputStream(bout.toByteArray()));
2511             } finally {
2512                 if (in != System.in) {
2513                     in.close();
2514                 }
2515             }
2516         } else {    // must be LDAP, and uri is not null
2517             URICertStoreParameters params =
2518                 new URICertStoreParameters(uri);
2519             CertStore s = CertStore.getInstance("LDAP", params);
2520             return s.getCRLs(new X509CRLSelector());
2521         }

3917     /**
3918      * Recursively tries to establish chain from pool of certs starting from
3919      * certToVerify until a self-signed cert is found, and fill the certs found
3920      * into chain. Each cert in the chain signs the next one.
3921      *
3922      * This method is able to recover from an error, say, if certToVerify
3923      * is signed by certA but certA has no issuer in certs and itself is not
3924      * self-signed, the method can try another certB that also signs
3925      * certToVerify and look for signer of certB, etc, etc.
3926      *
3927      * Each cert in chain comes with a label showing its origin. The label is
3928      * used in the warning message when the cert is considered a risk.
3929      *
3930      * @param certToVerify the cert that needs to be verified.
3931      * @param chain the chain that's being built.
3932      * @param certs the pool of trusted certs
3933      *
3934      * @return true if successful, false otherwise.
3935      */
3936     private boolean buildChain(Pair<String,X509Certificate> certToVerify,
3937             Vector<Pair<String,X509Certificate>> chain,
3938             Hashtable<Principal, Vector<Pair<String,X509Certificate>>> certs) {
3939         if (KeyStoreUtil.isSelfSigned(certToVerify.snd)) {
3940             // reached self-signed root cert;
3941             // no verification needed because it's trusted.
3942             chain.addElement(certToVerify);
3943             return true;
3944         }
3945 
3946         Principal issuer = certToVerify.snd.getIssuerDN();
3947 
3948         // Get the issuer's certificate(s)
3949         Vector<Pair<String,X509Certificate>> vec = certs.get(issuer);
3950         if (vec == null) {
3951             return false;
3952         }
3953 
3954         // Try out each certificate in the vector, until we find one
3955         // whose public key verifies the signature of the certificate
3956         // in question.
3957         for (Enumeration<Pair<String,X509Certificate>> issuerCerts = vec.elements();
3958                 issuerCerts.hasMoreElements(); ) {
3959             Pair<String,X509Certificate> issuerCert = issuerCerts.nextElement();
3960             PublicKey issuerPubKey = issuerCert.snd.getPublicKey();
3961             try {
3962                 certToVerify.snd.verify(issuerPubKey);
3963             } catch (Exception e) {
3964                 continue;
3965             }
3966             if (buildChain(issuerCert, chain, certs)) {
3967                 chain.addElement(certToVerify);
3968                 return true;
3969             }
3970         }
3971         return false;
3972     }
