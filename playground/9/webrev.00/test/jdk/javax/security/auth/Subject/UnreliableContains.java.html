24 /*
25  * @test
26  * @bug 8243592
27  * @summary Subject$SecureSet::addAll should not call contains(null)
28  */
29 
30 import javax.security.auth.Subject;
31 import java.security.Principal;
32 import java.util.Collections;
33 import java.util.HashSet;
34 import java.util.Objects;
35 
36 public class UnreliableContains {
37 
38     public static void main(String[] args) {
39         MySet<Principal> set = new MySet<>();
40         set.add(null);
41         Subject s = null;
42         try {
43             s = new Subject(false, set, Collections.emptySet(),
44                     Collections.emptySet());
45         } catch (NullPointerException e) {
46             // The correct exit
47             return;
48         }
49         // Suppose NPE was not caught. At least null was not added?
50         for (Principal p : s.getPrincipals()) {
51             Objects.requireNonNull(p);
52         }
53         // Still must fail. We don't want this Subject created
54         throw new RuntimeException("Fail");
55     }
56 
57     // This is a Map that implements contains(null) differently
58     static class MySet<E> extends HashSet<E> {
59         @Override
60         public boolean contains(Object o) {
61             if (o == null) {
62                 return false;
63             } else {
64                 return super.contains(o);
65             }
66         }
67     }
68 }
