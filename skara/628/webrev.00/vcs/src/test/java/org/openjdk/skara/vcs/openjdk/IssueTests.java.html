23 package org.openjdk.skara.vcs.openjdk;
24 
25 import org.junit.jupiter.api.Test;
26 
27 import static org.junit.jupiter.api.Assertions.assertEquals;
28 
29 public class IssueTests {
30     @Test
31     void parseRelaxed() {
32         assertEquals("Description", Issue.fromStringRelaxed("1234: Description").orElseThrow().description());
33         assertEquals("Description", Issue.fromStringRelaxed("1234 Description").orElseThrow().description());
34         assertEquals("Description", Issue.fromStringRelaxed("1234   Description").orElseThrow().description());
35         assertEquals("Description", Issue.fromStringRelaxed("1234 - Description").orElseThrow().description());
36         assertEquals("Description", Issue.fromStringRelaxed("1234   -  Description").orElseThrow().description());
37         assertEquals("-Description", Issue.fromStringRelaxed("1234   -Description").orElseThrow().description());
38         assertEquals("-Description", Issue.fromStringRelaxed("1234 - -Description").orElseThrow().description());
39         assertEquals("[Description]", Issue.fromStringRelaxed("1234: [Description]").orElseThrow().description());
40         assertEquals("[Description]", Issue.fromStringRelaxed("1234 [Description]").orElseThrow().description());
41         assertEquals("[Description]", Issue.fromStringRelaxed("1234 - [Description]").orElseThrow().description());
42     }
43 }
