23 package org.openjdk.skara.bots.notify;
24 
25 import org.openjdk.skara.forge.HostedRepository;
26 import org.openjdk.skara.storage.StorageBuilder;
27 
28 import java.nio.file.Path;
29 import java.util.*;
30 import java.util.regex.Pattern;
31 
32 public class NotifyBotBuilder {
33     private HostedRepository repository;
34     private Path storagePath;
35     private Pattern branches;
36     private StorageBuilder<UpdatedTag> tagStorageBuilder;
37     private StorageBuilder<UpdatedBranch> branchStorageBuilder;
38     private StorageBuilder<PullRequestState> prStateStorageBuilder;
39     private Set<String> readyLabels = Set.of();
40     private Map<String, Pattern> readyComments = Map.of();
41     private String integratorId;
42 
43     public NotifyBotBuilder repository(HostedRepository repository) {
44         this.repository = repository;
45         return this;
46     }
47 
48     public NotifyBotBuilder storagePath(Path storagePath) {
49         this.storagePath = storagePath;
50         return this;
51     }
52 
53     public NotifyBotBuilder branches(Pattern branches) {
54         this.branches = branches;
55         return this;
56     }
57 
58     public NotifyBotBuilder tagStorageBuilder(StorageBuilder<UpdatedTag> tagStorageBuilder) {
59         this.tagStorageBuilder = tagStorageBuilder;
60         return this;
61     }
62 
63     public NotifyBotBuilder branchStorageBuilder(StorageBuilder<UpdatedBranch> branchStorageBuilder) {
64         this.branchStorageBuilder = branchStorageBuilder;
65         return this;
66     }
67 
68     public NotifyBotBuilder prStateStorageBuilder(StorageBuilder<PullRequestState> prStateStorageBuilder) {
69         this.prStateStorageBuilder = prStateStorageBuilder;
70         return this;
71     }
72 
73     public NotifyBotBuilder readyLabels(Set<String> readyLabels) {
74         this.readyLabels = readyLabels;
75         return this;
76     }
77 
78     public NotifyBotBuilder readyComments(Map<String, Pattern> readyComments) {
79         this.readyComments = readyComments;
80         return this;
81     }
82 
83     public NotifyBotBuilder integratorId(String integratorId) {
84         this.integratorId = integratorId;
85         return this;
86     }
87 
88     public NotifyBot build() {
89         return new NotifyBot(repository, storagePath, branches, tagStorageBuilder, branchStorageBuilder, prStateStorageBuilder, readyLabels, readyComments, integratorId);
90     }
91 }
