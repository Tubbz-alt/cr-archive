23 package org.openjdk.skara.bots.notify;
24 
25 import org.openjdk.skara.forge.HostedRepository;
26 import org.openjdk.skara.storage.StorageBuilder;
27 
28 import java.nio.file.Path;
29 import java.util.*;
30 import java.util.regex.Pattern;
31 
32 public class NotifyBotBuilder {
33     private HostedRepository repository;
34     private Path storagePath;
35     private Pattern branches;
36     private StorageBuilder<UpdatedTag> tagStorageBuilder;
37     private StorageBuilder<UpdatedBranch> branchStorageBuilder;
38     private StorageBuilder<PullRequestState> prStateStorageBuilder;
39     private Map<String, Pattern> readyComments = Map.of();
40     private String integratorId;
41 
42     public NotifyBotBuilder repository(HostedRepository repository) {
43         this.repository = repository;
44         return this;
45     }
46 
47     public NotifyBotBuilder storagePath(Path storagePath) {
48         this.storagePath = storagePath;
49         return this;
50     }
51 
52     public NotifyBotBuilder branches(Pattern branches) {
53         this.branches = branches;
54         return this;
55     }
56 
57     public NotifyBotBuilder tagStorageBuilder(StorageBuilder<UpdatedTag> tagStorageBuilder) {
58         this.tagStorageBuilder = tagStorageBuilder;
59         return this;
60     }
61 
62     public NotifyBotBuilder branchStorageBuilder(StorageBuilder<UpdatedBranch> branchStorageBuilder) {
63         this.branchStorageBuilder = branchStorageBuilder;
64         return this;
65     }
66 
67     public NotifyBotBuilder prStateStorageBuilder(StorageBuilder<PullRequestState> prStateStorageBuilder) {
68         this.prStateStorageBuilder = prStateStorageBuilder;
69         return this;
70     }
71 
72     public NotifyBotBuilder readyComments(Map<String, Pattern> readyComments) {
73         this.readyComments = readyComments;
74         return this;
75     }
76 
77     public NotifyBotBuilder integratorId(String integratorId) {
78         this.integratorId = integratorId;
79         return this;
80     }
81 
82     public NotifyBot build() {
83         return new NotifyBot(repository, storagePath, branches, tagStorageBuilder, branchStorageBuilder, prStateStorageBuilder, readyComments, integratorId);
84     }
85 }
