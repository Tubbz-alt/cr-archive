23 package org.openjdk.skara.bots.mlbridge;
24 
25 import org.junit.jupiter.api.Test;
26 
27 import static org.junit.jupiter.api.Assertions.assertEquals;
28 
29 class TextToMarkdownTests {
30     @Test
31     void punctuation() {
32         assertEquals("1\\. Not a list", TextToMarkdown.escapeFormatting("1. Not a list"));
33         assertEquals("\\*not emphasized\\*", TextToMarkdown.escapeFormatting("*not emphasized*"));
34         assertEquals("\\\\n", TextToMarkdown.escapeFormatting("\\n"));
35     }
36 
37     @Test
38     void indented() {
39         assertEquals("&#32;      hello", TextToMarkdown.escapeFormatting("       hello"));
40     }
41 
42     @Test
43     void preserveQuoting() {
44         assertEquals("> quoted", TextToMarkdown.escapeFormatting("> quoted"));
45     }
46 
47     @Test
48     void escapedPattern() {
49         assertEquals("1\\$2", TextToMarkdown.escapeFormatting("1$2"));
50     }
51 }
