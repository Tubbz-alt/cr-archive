 82                         List<Review> allReviews, List<Review> activeReviews, Set<String> labels, CensusInstance censusInstance,

 97     private List<Issue> issues() {
 98         var issue = Issue.fromStringRelaxed(pr.title());
 99         if (issue.isPresent()) {
100             var issues = new ArrayList<Issue>();
101             issues.add(issue.get());
102             issues.addAll(SolvesTracker.currentSolved(pr.repository().forge().currentUser(), comments));

112     private List<org.openjdk.skara.issuetracker.Issue> issuesOfDisallowedType() {
113         var issueProject = issueProject();
114         var allowed = allowedIssueTypes();
115         if (issueProject != null && allowed != null) {
116             return issues().stream()
117                            .filter(i -> i.project().equals(Optional.of(issueProject.name())))
118                            .map(i -> issueProject.issue(i.shortId()))
119                            .filter(Optional::isPresent)
120                            .map(Optional::get)
121                            .filter(i -> i.properties().containsKey("issuetype"))
122                            .filter(i -> !allowed.contains(i.properties().get("issuetype").asString()))
123                            .collect(Collectors.toList());
124         }
125         return List.of();
126     }

128     private List<String> allowedTargetBranches() {
129         return pr.repository()
130                  .branches()
131                  .stream()
132                  .map(HostedBranch::name)
133                  .map(name -> workItem.bot.allowedTargetBranches().matcher(name))
134                  .filter(Matcher::matches)
135                  .map(Matcher::group)
136                  .collect(Collectors.toList());

139     // For unknown contributors, check that all commits have the same name and email
140     private boolean checkCommitAuthor(List<CommitMetadata> commits) throws IOException {
141         var author = censusInstance.namespace().get(pr.author().id());
142         if (author != null) {
143             return true;
144         }
145 
146         var names = new HashSet<String>();
147         var emails = new HashSet<String>();
148 
149         for (var commit : commits) {
150             names.add(commit.author().name());
151             emails.add(commit.author().email());
152         }
153 
154         return ((names.size() == 1) && emails.size() == 1);
155     }

157     // Additional bot-specific checks that are not handled by JCheck
158     private List<String> botSpecificChecks(Hash finalHash) throws IOException {
159         var ret = new ArrayList<String>();
160 
161         if (bodyWithoutStatus().isBlank()) {
162             var error = "The pull request body must not be empty.";
163             ret.add(error);
164         }
165 
166         if (!isTargetBranchAllowed()) {
167             var error = "The branch `" + pr.targetRef() + "` is not allowed as target branch. The allowed target branches are:\n" +
168                     allowedTargetBranches().stream()
169                     .map(name -> "   - " + name)
170                     .collect(Collectors.joining("\n"));
171             ret.add(error);
172         }
173 
174         var disallowedIssues = issuesOfDisallowedType();
175         if (!disallowedIssues.isEmpty()) {
176             var s = disallowedIssues.size() > 1 ? "s " : " ";
177             var are = disallowedIssues.size() > 1 ? "are" : "is";
178             var links = disallowedIssues.stream()
179                                         .map(i -> "[" + i.id() + "](" + i.webUrl() + ")")
180                                         .collect(Collectors.toList());
181             var error = "The issue" + s + String.join(",", links) + " " + are + " not of the expected type. The allowed issue types are:\n" +
182                 allowedIssueTypes().stream()
183                 .map(name -> "   - " + name)
184                 .collect(Collectors.joining("\n"));
185             ret.add(error);
186         }
187 
188         var headHash = pr.headHash();
189         var originalCommits = localRepo.commitMetadata(baseHash, headHash);
190 
191         if (!checkCommitAuthor(originalCommits)) {
192             var error = "For contributors who are not existing OpenJDK Authors, commit attribution will be taken from " +
193                     "the commits in the PR. However, the commits in this PR have inconsistent user names and/or " +
194                     "email addresses. Please amend the commits.";
195             ret.add(error);
196         }
197 
198         for (var blocker : workItem.bot.blockingCheckLabels().entrySet()) {
199             if (labels.contains(blocker.getKey())) {
200                 ret.add(blocker.getValue());
201             }
202         }
203 
204         return ret;
205     }

207     private void updateCheckBuilder(CheckBuilder checkBuilder, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
208         if (visitor.isReadyForReview() && additionalErrors.isEmpty()) {
209             checkBuilder.complete(true);
210         } else {
211             checkBuilder.title("Required");
212             var summary = Stream.concat(visitor.messages().stream(), additionalErrors.stream())
213                                 .sorted()
214                                 .map(m -> "- " + m)
215                                 .collect(Collectors.joining("\n"));
216             checkBuilder.summary(summary);
217             for (var annotation : visitor.getAnnotations()) {
218                 checkBuilder.annotation(annotation);
219             }
220             checkBuilder.complete(false);
221         }
222     }

224     private void updateReadyForReview(PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
225         // Additional errors are not allowed
226         if (!additionalErrors.isEmpty()) {
227             newLabels.remove("rfr");
228             return;
229         }
230 
231         // Draft requests are not for review
232         if (pr.isDraft()) {
233             newLabels.remove("rfr");
234             return;
235         }
236 
237         // Check if the visitor found any issues that should be resolved before reviewing
238         if (visitor.isReadyForReview()) {
239             newLabels.add("rfr");
240         } else {
241             newLabels.remove("rfr");
242         }
243     }

245     private String getRole(String username) {
246         var project = censusInstance.project();
247         var version = censusInstance.census().version().format();
248         if (project.isReviewer(username, version)) {
249             return "**Reviewer**";
250         } else if (project.isCommitter(username, version)) {
251             return "Committer";
252         } else if (project.isAuthor(username, version)) {
253             return "Author";
254         } else {
255             return "no project role";
256         }
257     }

259     private String formatReviewer(HostUser reviewer) {
260         var namespace = censusInstance.namespace();
261         var contributor = namespace.get(reviewer.id());
262         if (contributor == null) {
263             return reviewer.userName() + " (no known " + namespace.name() + " user name / role)";
264         } else {
265             var userNameLink = "[" + contributor.username() + "](@" + reviewer.userName() + ")";
266             return contributor.fullName().orElse(contributor.username()) + " (" + userNameLink + " - " +
267                     getRole(contributor.username()) + ")";
268         }
269     }

284     private Optional<String> getReviewersList(List<Review> reviews) {
285         var reviewers = reviews.stream()
286                                .filter(review -> review.verdict() == Review.Verdict.APPROVED)
287                                .map(review -> {
288                                    var entry = " * " + formatReviewer(review.reviewer());
289                                    if (!review.hash().equals(pr.headHash())) {
290                                        if (ignoreStaleReviews) {
291                                            entry += " 🔄 Re-review required (review applies to " + review.hash() + ")";
292                                        } else {
293                                            entry += " ⚠️ Review applies to " + review.hash();
294                                        }
295                                    }
296                                    return entry;
297                                })
298                                .collect(Collectors.joining("\n"));
299 
300         // Check for manually added reviewers
301         if (!ignoreStaleReviews) {
302             var namespace = censusInstance.namespace();
303             var allReviewers = PullRequestUtils.reviewerNames(reviews, namespace);
304             var additionalEntries = new ArrayList<String>();
305             for (var additional : Reviewers.reviewers(pr.repository().forge().currentUser(), comments)) {
306                 if (!allReviewers.contains(additional)) {
307                     additionalEntries.add(" * " + additional + " - " + getRole(additional) + " ⚠️ Added manually");
308                 }
309             }
310             if (!reviewers.isBlank()) {
311                 reviewers += "\n";
312             }
313             reviewers += String.join("\n", additionalEntries);
314         }
315 
316         if (reviewers.length() > 0) {
317             return Optional.of(reviewers);
318         } else {
319             return Optional.empty();
320         }
321     }

340     private boolean relaxedEquals(String s1, String s2) {
341         return s1.trim()
342                  .replaceAll("\\s+", " ")
343                  .equalsIgnoreCase(s2.trim()
344                                      .replaceAll("\\s+", " "));
345     }

348     private String getStatusMessage(List<Comment> comments, List<Review> reviews, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
349         var progressBody = new StringBuilder();
350         progressBody.append("---------\n");
351         progressBody.append("### Progress\n");
352         progressBody.append(getChecksList(visitor));
353 
354         var allAdditionalErrors = Stream.concat(visitor.hiddenMessages().stream(), additionalErrors.stream())
355                                         .sorted()
356                                         .collect(Collectors.toList());
357         if (!allAdditionalErrors.isEmpty()) {
358             progressBody.append("\n\n### Error");
359             if (allAdditionalErrors.size() > 1) {
360                 progressBody.append("s");
361             }
362             progressBody.append("\n");
363             progressBody.append(getAdditionalErrorsList(allAdditionalErrors));
364         }
365 
366         var issues = issues();
367         var issueProject = issueProject();
368         if (issueProject != null && !issues.isEmpty()) {



369             progressBody.append("\n\n### Issue");
370             if (issues.size() > 1) {
371                 progressBody.append("s");
372             }
373             progressBody.append("\n");
374             for (var currentIssue : issues) {
375                 progressBody.append(" * ");
376                 if (currentIssue.project().isPresent() && !currentIssue.project().get().equals(issueProject.name())) {
377                     progressBody.append("⚠️ Issue `");
378                     progressBody.append(currentIssue.id());
379                     progressBody.append("` does not belong to the `");
380                     progressBody.append(issueProject.name());
381                     progressBody.append("` project.");
382                 } else {
383                     try {
384                         var iss = issueProject.issue(currentIssue.shortId());
385                         if (iss.isPresent()) {
386                             progressBody.append("[");
387                             progressBody.append(iss.get().id());
388                             progressBody.append("](");
389                             progressBody.append(iss.get().webUrl());
390                             progressBody.append("): ");
391                             progressBody.append(iss.get().title());
392                             if (!relaxedEquals(iss.get().title(), currentIssue.description())) {
393                                 progressBody.append(" ⚠️ Title mismatch between PR and JBS.");
394                             }
395                             progressBody.append("\n");
396                         } else {
397                             progressBody.append("⚠️ Failed to retrieve information on issue `");
398                             progressBody.append(currentIssue.id());
399                             progressBody.append("`.\n");
400                         }
401                     } catch (RuntimeException e) {
402                         progressBody.append("⚠️ Temporary failure when trying to retrieve information on issue `");
403                         progressBody.append(currentIssue.id());
404                         progressBody.append("`.\n");
405                     }
406                 }
407             }
408         }
409 
410         getReviewersList(reviews).ifPresent(reviewers -> {
411             progressBody.append("\n\n### Reviewers\n");
412             progressBody.append(reviewers);
413         });
414 
415         getContributorsList(comments).ifPresent(contributors -> {
416             progressBody.append("\n\n### Contributors\n");
417             progressBody.append(contributors);
418         });
419 
420         progressBody.append("\n\n### Download\n");
421         progressBody.append(checkoutCommands());
422 
423         return progressBody.toString();
424     }

426     private String checkoutCommands() {
427         var repoUrl = pr.repository().webUrl();
428         return
429            "`$ git fetch " + repoUrl + " " + pr.fetchRef() + ":pull/" + pr.id() + "`\n" +
430            "`$ git checkout pull/" + pr.id() + "`\n";
431     }

433     private String bodyWithoutStatus() {
434         var description = pr.body();
435         var markerIndex = description.lastIndexOf(progressMarker);
436         return (markerIndex < 0 ?
437                 description :
438                 description.substring(0, markerIndex)).trim();
439     }

441     private String updateStatusMessage(String message) {
442         var description = pr.body();
443         var markerIndex = description.lastIndexOf(progressMarker);
444 
445         if (markerIndex >= 0 && description.substring(markerIndex).equals(message)) {
446             log.info("Progress already up to date");
447             return description;
448         }
449         var originalBody = bodyWithoutStatus();
450         if (originalBody.isBlank()) {
451             originalBody = emptyPrBodyMarker;
452         }
453         var newBody = originalBody + "\n" + progressMarker + "\n" + message;
454 
455         // TODO? Retrieve the body again here to lower the chance of concurrent updates
456         pr.setBody(newBody);
457         return newBody;
458     }

460     private String verdictToString(Review.Verdict verdict) {
461         switch (verdict) {
462             case APPROVED:
463                 return "changes are approved";
464             case DISAPPROVED:
465                 return "more changes needed";
466             case NONE:
467                 return "comment added";
468             default:
469                 throw new RuntimeException("Unknown verdict: " + verdict);
470         }
471     }

474         var reviewTracker = new ReviewTracker(comments, reviews);

493     private String getMergeReadyComment(String commitMessage, List<Review> reviews) {
494         var message = new StringBuilder();
495         message.append("@");
496         message.append(pr.author().userName());
497         message.append(" This change now passes all automated pre-integration checks");
498 
499         try {
500             var hasContributingFile =
501                 !localRepo.files(pr.targetHash(), Path.of("CONTRIBUTING.md")).isEmpty();
502             if (hasContributingFile) {
503                 message.append(". When the change also fulfills all ");
504                 message.append("[project specific requirements](https://github.com/");
505                 message.append(pr.repository().name());
506                 message.append("/blob/");
507                 message.append(pr.targetRef());
508                 message.append("/CONTRIBUTING.md)");
509             }
510         } catch (IOException e) {
511             throw new UncheckedIOException(e);
512         }
513 
514         message.append(", type `/integrate` in a new comment to proceed. After integration, ");
515         message.append("the commit message will be:\n");
516         message.append("```\n");
517         message.append(commitMessage);
518         message.append("\n```\n");
519 
520         message.append("- If you would like to add a summary, use the `/summary` command.\n");
521         message.append("- To credit additional contributors, use the `/contributor` command.\n");
522         message.append("- To add additional solved issues, use the `/issue` command.\n");
523 
524         var divergingCommits = checkablePullRequest.divergingCommits();
525         if (divergingCommits.size() > 0) {
526             message.append("\n");
527             message.append("Since the source branch of this PR was last updated there ");
528             if (divergingCommits.size() == 1) {
529                 message.append("has been 1 commit ");
530             } else {
531                 message.append("have been ");
532                 message.append(divergingCommits.size());
533                 message.append(" commits ");
534             }
535             message.append("pushed to the `");
536             message.append(pr.targetRef());
537             message.append("` branch:\n\n");
538             divergingCommits.stream()
539                             .limit(10)
540                             .forEach(c -> message.append(" * ").append(c.hash().hex()).append(": ").append(c.message().get(0)).append("\n"));
541             if (divergingCommits.size() > 10) {
542                 message.append(" * ... and ").append(divergingCommits.size() - 10).append(" more: ")
543                        .append(pr.repository().webUrl(baseHash.hex(), pr.targetRef())).append("\n");
544             }
545 
546             message.append("\n");
547             message.append("As there are no conflicts, your changes will automatically be rebased on top of ");
548             message.append("these commits when integrating. If you prefer to avoid automatic rebasing, please merge `");
549             message.append(pr.targetRef());
550             message.append("` into your branch, and then specify the current head hash when integrating, like this: ");
551             message.append("`/integrate ");
552             message.append(pr.targetHash());
553             message.append("`.\n");
554         } else {
555             message.append("\n");
556             message.append("There are currently no new commits on the `");
557             message.append(pr.targetRef());
558             message.append("` branch since the last update of the source branch of this PR. If another commit should be pushed before ");
559             message.append("you perform the `/integrate` command, your PR will be automatically rebased. If you would like to avoid ");
560             message.append("potential automatic rebasing, specify the current head hash when integrating, like this: ");
561             message.append("`/integrate ");
562             message.append(pr.targetHash());
563             message.append("`.\n");
564         }
565 
566         if (!ProjectPermissions.mayCommit(censusInstance, pr.author())) {
567             message.append("\n");
568             var contributor = censusInstance.namespace().get(pr.author().id());
569             if (contributor == null) {
570                 message.append("As you are not a known OpenJDK [Author](https://openjdk.java.net/bylaws#author), ");
571             } else {
572                 message.append("As you do not have Committer status in this project, ");
573             }
574 
575             message.append("an existing [Committer](https://openjdk.java.net/bylaws#committer) must agree to ");
576             message.append("[sponsor](https://openjdk.java.net/sponsor/) your change. ");
577             var candidates = reviews.stream()
578                                     .filter(review -> ProjectPermissions.mayCommit(censusInstance, review.reviewer()))
579                                     .map(review -> "@" + review.reviewer().userName())
580                                     .collect(Collectors.joining(", "));
581             if (candidates.length() > 0) {
582                 message.append("Possible candidates are the reviewers of this PR (");
583                 message.append(candidates);
584                 message.append(") but any other Committer may sponsor as well. ");
585             }
586             message.append("\n\n");
587             message.append("➡️ To flag this PR as ready for integration with the above commit message, type ");
588             message.append("`/integrate` in a new comment. (Afterwards, your sponsor types ");
589             message.append("`/sponsor` in a new comment to perform the integration).\n");
590         } else {
591             message.append("\n");
592             message.append("➡️ To integrate this PR with the above commit message to the `" + pr.targetRef() + "` branch, type ");
593             message.append("`/integrate` in a new comment.\n");
594         }
595         message.append(mergeReadyMarker);
596         return message.toString();
597     }

608     private void updateMergeReadyComment(boolean isReady, String commitMessage, List<Comment> comments, List<Review> reviews, boolean rebasePossible) {
609         var existing = findComment(comments, mergeReadyMarker);
610         if (isReady && rebasePossible) {
611             var message = getMergeReadyComment(commitMessage, reviews);
612             if (existing.isEmpty()) {
613                 pr.addComment(message);
614             } else {
615                 pr.updateComment(existing.get().id(), message);
616             }
617         } else {
618             existing.ifPresent(comment -> pr.updateComment(comment.id(), getMergeNoLongerReadyComment()));
619         }
620     }

622     private void addSourceBranchWarningComment(List<Comment> comments) throws IOException {
623         var existing = findComment(comments, sourceBranchWarningMarker);
624         if (existing.isPresent()) {
625             // Only add the comment once per PR
626             return;
627         }
628         var branch = pr.sourceRef();
629         var message = ":warning: @" + pr.author().userName() + " " +
630             "a branch with the same name as the source branch for this pull request (`" + branch + "`) " +
631             "is present in the [target repository](" + pr.repository().nonTransformedWebUrl() + "). " +
632             "If you eventually integrate this pull request then the branch `" + branch + "` " +
633             "in your [personal fork](" + pr.sourceRepository().nonTransformedWebUrl() + ") will diverge once you sync " +
634             "your personal fork with the upstream repository.\n" +
635             "\n" +
636             "To avoid this situation, create a new branch for your changes and reset the `" + branch + "` branch. " +
637             "You can do this by running the following commands in a local repository for your personal fork. " +
638             "_Note_: you do *not* have to name the new branch `NEW-BRANCH-NAME`." +
639             "\n" +
640             "```" +
641             "$ git checkout " + branch + "\n" +
642             "$ git checkout -b NEW-BRANCH-NAME\n" +
643             "$ git branch -f " + branch + " " + baseHash.hex() + "\n" +
644             "$ git push -f origin " + branch + "\n" +
645             "```\n" +
646             "\n" +
647             "Then proceed to create a new pull request with `NEW-BRANCH-NAME` as the source branch and " +
648             "close this one.\n" +
649             sourceBranchWarningMarker;
650         pr.addComment(message);
651     }

653     private void addOutdatedComment(List<Comment> comments) {
654         var existing = findComment(comments, outdatedHelpMarker);
655         if (existing.isPresent()) {
656             // Only add the comment once per PR
657             return;
658         }
659         var message = "@" + pr.author().userName() + " this pull request can not be integrated into " +
660                 "`" + pr.targetRef() + "` due to one or more merge conflicts. To resolve these merge conflicts " +
661                 "and update this pull request you can run the following commands in the local repository for your personal fork:\n" +
662                 "```bash\n" +
663                 "git checkout " + pr.sourceRef() + "\n" +
664                 "git fetch " + pr.repository().webUrl() + " " + pr.targetRef() + "\n" +
665                 "git merge FETCH_HEAD\n" +
666                 "# resolve conflicts and follow the instructions given by git merge\n" +
667                 "git commit -m \"Merge " + pr.targetRef() + "\"\n" +
668                 "git push\n" +
669                 "```\n" +
670                 outdatedHelpMarker;
671         pr.addComment(message);
672     }

674     private void checkStatus() {
675         var checkBuilder = CheckBuilder.create("jcheck", pr.headHash());
676         var censusDomain = censusInstance.configuration().census().domain();
677         Exception checkException = null;
678 
679         try {
680             // Post check in-progress
681             log.info("Starting to run jcheck on PR head");
682             pr.createCheck(checkBuilder.build());
683 
684             var ignored = new PrintWriter(new StringWriter());
685             var rebasePossible = true;
686             var commitHash = pr.headHash();
687             var mergedHash = checkablePullRequest.mergeTarget(ignored);
688             if (mergedHash.isPresent()) {
689                 commitHash = mergedHash.get();
690             } else {
691                 rebasePossible = false;
692             }
693 
694             List<String> additionalErrors = List.of();
695             Hash localHash;
696             try {
697                 localHash = checkablePullRequest.commit(commitHash, censusInstance.namespace(), censusDomain, null);
698             } catch (CommitFailure e) {
699                 additionalErrors = List.of(e.getMessage());
700                 localHash = baseHash;
701             }
702             PullRequestCheckIssueVisitor visitor = checkablePullRequest.createVisitor(localHash, censusInstance);
703             if (!localHash.equals(baseHash)) {
704                 // Determine current status
705                 var additionalConfiguration = AdditionalConfiguration.get(localRepo, localHash, pr.repository().forge().currentUser(), comments);
706                 checkablePullRequest.executeChecks(localHash, censusInstance, visitor, additionalConfiguration);
707                 additionalErrors = botSpecificChecks(localHash);
708             } else {
709                 if (additionalErrors.isEmpty()) {
710                     additionalErrors = List.of("This PR contains no changes");
711                 }
712             }
713             updateCheckBuilder(checkBuilder, visitor, additionalErrors);
714             updateReadyForReview(visitor, additionalErrors);
715 
716             // Calculate and update the status message if needed
717             var statusMessage = getStatusMessage(comments, activeReviews, visitor, additionalErrors);
718             var updatedBody = updateStatusMessage(statusMessage);
719 
720             // Post / update approval messages (only needed if the review itself can't contain a body)
721             if (!pr.repository().forge().supportsReviewBody()) {
722                 updateReviewedMessages(comments, allReviews);
723             }
724 
725             var commit = localRepo.lookup(localHash).orElseThrow();
726             var commitMessage = String.join("\n", commit.message());
727             var readyForIntegration = visitor.messages().isEmpty() && additionalErrors.isEmpty();
728             updateMergeReadyComment(readyForIntegration, commitMessage, comments, activeReviews, rebasePossible);
729             if (readyForIntegration && rebasePossible) {
730                 newLabels.add("ready");
731             } else {
732                 newLabels.remove("ready");
733             }
734             if (!rebasePossible) {
735                 if (!labels.contains("failed-auto-merge")) {
736                     addOutdatedComment(comments);
737                 }
738                 newLabels.add("merge-conflict");
739             } else {
740                 newLabels.remove("merge-conflict");
741             }
742 
743             var branchNames = pr.repository().branches().stream().map(HostedBranch::name).collect(Collectors.toSet());
744             if (!pr.repository().url().equals(pr.sourceRepository().url()) && branchNames.contains(pr.sourceRef())) {
745                 addSourceBranchWarningComment(comments);
746             }
747 
748             // Ensure that the ready for sponsor label is up to date
749             newLabels.remove("sponsor");
750             var readyHash = ReadyForSponsorTracker.latestReadyForSponsor(pr.repository().forge().currentUser(), comments);
751             if (readyHash.isPresent() && readyForIntegration) {
752                 var acceptedHash = readyHash.get();
753                 if (pr.headHash().equals(acceptedHash)) {
754                     newLabels.add("sponsor");
755                 }
756             }
757 
758             // Calculate current metadata to avoid unnecessary future checks
759             var metadata = workItem.getMetadata(pr.title(), updatedBody, pr.comments(), activeReviews, newLabels,
760                                                 censusInstance, pr.targetHash(), pr.isDraft());
761             checkBuilder.metadata(metadata);
762         } catch (Exception e) {
763             log.throwing("CommitChecker", "checkStatus", e);
764             newLabels.remove("ready");
765             checkBuilder.metadata("invalid");
766             checkBuilder.title("Exception occurred during jcheck - the operation will be retried");
767             checkBuilder.summary(e.getMessage());
768             checkBuilder.complete(false);
769             checkException = e;
770         }
771         var check = checkBuilder.build();
772         pr.updateCheck(check);
773 
774         // Synchronize the wanted set of labels
775         for (var newLabel : newLabels) {
776             if (!labels.contains(newLabel)) {
777                 pr.addLabel(newLabel);
778             }
779         }
780         for (var oldLabel : labels) {
781             if (!newLabels.contains(oldLabel)) {
782                 pr.removeLabel(oldLabel);
783             }
784         }
785 
786         // After updating the PR, rethrow any exception to automatically retry on transient errors
787         if (checkException != null) {
788             throw new RuntimeException("Exception during jcheck", checkException);
789         }
790     }
