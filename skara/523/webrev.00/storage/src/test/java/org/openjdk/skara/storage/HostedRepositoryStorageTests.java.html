23 package org.openjdk.skara.storage;
24 
25 import org.junit.jupiter.api.*;
26 import org.openjdk.skara.test.*;
27 import org.openjdk.skara.vcs.*;
28 
29 import java.io.IOException;
30 import java.util.*;
31 import java.util.stream.*;
32 
33 import static org.junit.jupiter.api.Assertions.*;
34 
35 public class HostedRepositoryStorageTests {
36     String serializer(Collection<String> added, Set<String> existing) {
37         return Stream.concat(added.stream(), existing.stream())
38                 .distinct()
39                 .sorted()
40                 .collect(Collectors.joining("\n"));
41     }
42 
43     Set<String> deserializer(String serialized) {
44         return serialized.lines().collect(Collectors.toSet());
45     }
46 
47     @Test
48     void failedMaterialization(TestInfo testInfo) throws IOException {
49         try (var credentials = new HostCredentials(testInfo);
50                 var tempFolder = new TemporaryDirectory()) {
51             var repo = credentials.getHostedRepository();
52             var storage = new HostedRepositoryStorage<>(repo, tempFolder.path(), "master", "test.txt",
53                                                         "duke", "duke@openjdk.java.org",
54                                                         "Updated storage", this::serializer, this::deserializer);
55             storage.put(List.of("a", "b"));
56 
57             // Corrupt the destination path and materialize again
58             var localRepo = Repository.init(tempFolder.path(), VCS.GIT);
59             localRepo.checkout(new Branch("storage"));
60             assertThrows(RuntimeException.class, () -> new HostedRepositoryStorage<>(repo, tempFolder.path(), "master", "test.txt",
61                                                                                      "duke", "duke@openjdk.java.org",
62                                                                                      "Updated storage", this::serializer, this::deserializer));
63         }
64     }
65 }
