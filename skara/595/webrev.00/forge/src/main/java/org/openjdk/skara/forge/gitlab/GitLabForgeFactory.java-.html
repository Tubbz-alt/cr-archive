1 package org.openjdk.skara.forge.gitlab;
 2 
 3 import org.openjdk.skara.forge.*;
 4 import org.openjdk.skara.host.Credential;
 5 import org.openjdk.skara.json.JSONObject;
 6 
 7 import java.net.URI;
 8 import java.util.Set;
 9 
10 public class GitLabForgeFactory implements ForgeFactory {
11     @Override
12     public String name() {
13         return "gitlab";
14     }
15 
16     @Override
17     public Set<String> knownHosts() {
18         return Set.of("gitlab.com");
19     }
20 
21     @Override
22     public Forge create(URI uri, Credential credential, JSONObject configuration) {
23         var name = "GitLab";
24         if (configuration.contains("name")) {
25             name = configuration.get("name").asString();
26         }
27         if (credential != null) {
28             return new GitLabHost(name, uri, credential);
29         } else {
30             return new GitLabHost(name, uri);
31         }
32     }
33 }
