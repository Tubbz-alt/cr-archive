  49     private java.lang.Process start(List<String> cmd) throws IOException {
  50         log.fine("Executing " + String.join(" ", cmd));
  51         var pb = new ProcessBuilder(cmd);
  52         pb.directory(dir.toFile());
  53         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
  54         return pb.start();
  55     }

  57     private static void stop(java.lang.Process p) throws IOException {
  58         if (p != null && p.isAlive()) {
  59             var stream = p.getInputStream();
  60             var read = 0;
  61             var buf = new byte[128];
  62             while (read != -1) {
  63                 read = stream.read(buf);
  64             }
  65             try {
  66                 p.waitFor();
  67             } catch (InterruptedException e) {
  68                 throw new IOException(e);
  69             }
  70         }
  71     }

  91     private static Execution.Result await(Execution e) throws IOException {
  92         var result = e.await();
  93         if (result.status() != 0) {
  94             throw new IOException("Unexpected exit code\n" + result);
  95         }
  96         return result;
  97     }

  99     private static void await(java.lang.Process p) throws IOException {
 100         try {
 101             var res = p.waitFor();
 102             if (res != 0) {
 103                 throw new IOException("Unexpected exit code: " + res);
 104             }
 105         } catch (InterruptedException e) {
 106             throw new IOException(e);
 107         }
 108     }

 202     @Override
 203     public List<CommitMetadata> commitMetadata(String range, List<Path> paths, boolean reverse) throws IOException {
 204         var args = new ArrayList<String>();
 205         args.addAll(List.of("git", "rev-list",
 206                                    "--format=" + GitCommitMetadata.FORMAT,
 207                                    "--no-abbrev",
 208                                    "--no-color",
 209                                    range));
 210         if (reverse) {
 211             args.add("--reverse");
 212         }
 213         if (paths != null && !paths.isEmpty()) {
 214             args.add("--");
 215             for (var path : paths) {
 216                 args.add(path.toString());
 217             }
 218         }
 219         var p = start(args);
 220         var reader = new UnixStreamReader(p.getInputStream());
 221         var result = new ArrayList<CommitMetadata>();
 222 
 223         var line = reader.readLine();
 224         while (line != null) {
 225             if (!line.startsWith("commit")) {
 226                 throw new IOException("Unexpected line: " + line);
 227             }
 228 
 229             result.add(GitCommitMetadata.read(reader));
 230             line = reader.readLine();
 231         }
 232 
 233         await(p);
 234         return result;
 235     }

 292     private List<Hash> refs() throws IOException {
 293         try (var p = capture("git", "show-ref", "--hash", "--abbrev")) {
 294             var res = p.await();
 295             if (res.status() == -1) {
 296                 if (res.stdout().size() != 0) {
 297                     throw new IOException("Unexpected output\n" + res);
 298                 }
 299                 return new ArrayList<>();
 300             } else {
 301                 return res.stdout().stream()
 302                           .map(Hash::new)
 303                           .collect(Collectors.toList());
 304             }
 305         }
 306     }

 308     @Override
 309     public boolean isEmpty() throws IOException {
 310         int numLooseObjects = -1;
 311         int numPackedObjects = -1;
 312 
 313         try (var p = capture("git", "count-objects", "-v")) {
 314             var res = await(p);
 315             var stdout = res.stdout();
 316 
 317             for (var line : stdout) {
 318                 if (line.startsWith("count: ")) {
 319                     try {
 320                         numLooseObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 321                     } catch (NumberFormatException e) {
 322                         throw new IOException("Unexpected 'count' value\n" + res, e);
 323                     }
 324 
 325                 } else if (line.startsWith("in-pack: ")) {
 326                     try {
 327                         numPackedObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 328                     } catch (NumberFormatException e) {
 329                         throw new IOException("Unexpected 'in-pack' value\n" + res, e);
 330                     }
 331                 }
 332             }
 333         }
 334 
 335         return numLooseObjects == 0 && numPackedObjects == 0 && refs().size() == 0;
 336     }

 511     @Override
 512     public Path root() throws IOException {
 513         if (cachedRoot != null) {
 514             return cachedRoot;
 515         }
 516 
 517         try (var p = capture("git", "rev-parse", "--show-toplevel")) {
 518             var res = p.await();
 519             if (res.status() != 0 || res.stdout().size() != 1) {
 520                 // Perhaps this is a bare repository
 521                 try (var p2 = capture("git", "rev-parse", "--git-dir")) {
 522                     var res2 = await(p2);
 523                     if (res2.stdout().size() != 1) {
 524                         throw new IOException("Unexpected output\n" + res2);
 525                     }
 526                     cachedRoot = dir.resolve(Path.of(res2.stdout().get(0)));
 527                     return cachedRoot;
 528                 }
 529             }
 530 
 531             cachedRoot = Path.of(res.stdout().get(0));
 532             return cachedRoot;
 533         }
 534     }

 543     @FunctionalInterface
 544     private static interface Operation {
 545         void execute(List<Path> args) throws IOException;
 546     }
 547 
 548     private void batch(Operation op, List<Path> args) throws IOException {
 549         var batchSize = 64;
 550         var start = 0;
 551         while (start < args.size()) {
 552             var end = start + batchSize;
 553             if (end > args.size()) {
 554                 end = args.size();
 555             }
 556             op.execute(args.subList(start, end));
 557             start = end;
 558         }
 559     }

 624     @Override
 625     public Hash commit(String message,
 626                        String authorName,
 627                        String authorEmail,
 628                        ZonedDateTime authorDate,
 629                        String committerName,
 630                        String committerEmail,
 631                        ZonedDateTime committerDate) throws IOException {
 632         var cmd = Process.capture("git", "commit", "--message=" + message)
 633                          .workdir(dir)
 634                          .environ("GIT_AUTHOR_NAME", authorName)
 635                          .environ("GIT_AUTHOR_EMAIL", authorEmail)
 636                          .environ("GIT_COMMITTER_NAME", committerName)
 637                          .environ("GIT_COMMITTER_EMAIL", committerEmail);
 638         if (authorDate != null) {
 639             cmd = cmd.environ("GIT_AUTHOR_DATE",
 640                               authorDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 641         }
 642         if (committerDate != null) {
 643             cmd = cmd.environ("GIT_COMMITTER_DATE",
 644                               committerDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 645         }
 646         try (var p = cmd.execute()) {
 647             await(p);
 648             return head();
 649         }
 650     }

 806         var cmd = new ArrayList<String>();

 815             for (var line : res.stdout()) {
 816                 var parts = line.split("\t");

 821                                           FileType.fromOctal(metadata[0]),

 823                                           Path.of(filename));
 824                 entries.add(entry);
 825             }

 830     @Override
 831     public List<FileEntry> files(Hash hash, List<Path> paths) throws IOException {
 832         if (paths.isEmpty()) {
 833             return allFiles(hash, paths);
 834         }
 835 
 836         var entries = new ArrayList<FileEntry>();
 837         var batchSize = 64;
 838         var start = 0;
 839         while (start < paths.size()) {
 840             var end = start + batchSize;
 841             if (end > paths.size()) {
 842                 end = paths.size();
 843             }
 844             entries.addAll(allFiles(hash, paths.subList(start, end)));
 845             start = end;
 846         }
 847         return entries;
 848     }

 862     public Optional<byte[]> show(Path path, Hash hash) throws IOException {
 863         var entries = files(hash, path);
 864         if (entries.size() == 0) {
 865             return Optional.empty();
 866         } else if (entries.size() > 1) {
 867             throw new IOException("Multiple files match path " + path.toString() + " in commit " + hash.hex());
 868         }
 869 
 870         var entry = entries.get(0);
 871         var type = entry.type();
 872         if (type.isVCSLink()) {
 873             var content = "Subproject commit " + entry.hash().hex() + " " + entry.path().toString();
 874             return Optional.of(content.getBytes(StandardCharsets.UTF_8));
 875         } else if (type.isRegular()) {
 876             var tmp = unpackFile(entry.hash().hex());
 877             var content = Files.readAllBytes(tmp);
 878             Files.delete(tmp);
 879             return Optional.of(content);
 880         }
 881 
 882         return Optional.empty();

1238     @Override
1239     public List<Submodule> submodules() throws IOException {
1240         var gitModules = root().resolve(".gitmodules");
1241         if (!Files.exists(gitModules)) {
1242             return List.of();
1243         }
1244 
1245         var urls = new HashMap<String, String>();
1246         var paths = new HashMap<String, String>();
1247         try (var p = capture("git", "config", "--file", gitModules.toAbsolutePath().toString(),
1248                                               "--list")) {
1249             for (var line : await(p).stdout()) {
1250                 if (line.startsWith("submodule.")) {
1251                     line = line.substring("submodule.".length());
1252                     var parts = line.split("=");
1253                     var nameAndProperty = parts[0].split("\\.");
1254                     var name = nameAndProperty[0];
1255                     var prop = nameAndProperty[1];
1256                     var value = parts[1];
1257                     if (prop.equals("path")) {
1258                         paths.put(name, value);
1259                     } else if (prop.equals("url")) {
1260                         urls.put(name, value);
1261                     } else {
1262                         throw new IOException("Unexpected submodule property: " + prop);
1263                     }
1264                 }
1265             }
1266         }
1267 
1268         var hashes = new HashMap<String, String>();
1269         try (var p = capture("git", "submodule", "status")) {
1270             for (var line : await(p).stdout()) {
1271                 var parts = line.substring(1).split(" ");
1272                 var hash = parts[0];
1273                 var path = parts[1];
1274                 hashes.put(path, hash);
1275             }
1276         }
1277 
1278         var modules = new ArrayList<Submodule>();
1279         for (var name : paths.keySet()) {
1280             var url = urls.get(name);
1281             var path = paths.get(name);
1282             var hash = hashes.get(path);
1283 
1284             modules.add(new Submodule(new Hash(hash), Path.of(path), url));
1285         }
1286
