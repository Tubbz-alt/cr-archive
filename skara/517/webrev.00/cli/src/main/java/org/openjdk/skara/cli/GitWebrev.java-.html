 42 public class GitWebrev {
 43     private static final List<String> KNOWN_JBS_PROJECTS =
 44         List.of("JDK", "CODETOOLS", "SKARA", "JMC");
 45     private static void clearDirectory(Path directory) {
 46         try {
 47             Files.walk(directory)
 48                     .map(Path::toFile)
 49                     .sorted(Comparator.reverseOrder())
 50                     .forEach(File::delete);
 51         } catch (IOException io) {
 52             throw new RuntimeException(io);
 53         }
 54     }
 55 
 56     private static String arg(String name, Arguments args, ReadOnlyRepository repo) throws IOException {
 57         if (args.contains(name)) {
 58             return args.get(name).asString();
 59         }
 60 
 61         var config = repo.config("webrev." + name);
 62         if (config.size() == 1) {
 63             return config.get(0);
 64         }
 65 
 66         return null;
 67     }
 68 
 69     private static void die(String message) {
 70         System.err.println(message);
 71         System.exit(1);
 72     }
 73 
 74     private static Hash resolve(ReadOnlyRepository repo, String ref) {
 75         var message = "error: could not resolve reference '" + ref + "'";
 76         try {
 77             var hash = repo.resolve(ref);
 78             if (!hash.isPresent()) {
 79                 die(message);
 80             }
 81             return hash.get();
 82         } catch (IOException e) {
 83             die(message);
 84             return null; // impossible
 85         }
 86     }
 87 
 88     private static boolean isDigit(char c) {
 89         return Character.isDigit(c);
 90     }

 92     private static void generate(String[] args) throws IOException {
 93         var flags = List.of(
 94             Option.shortcut("r")
 95                   .fullname("rev")
 96                   .describe("REV")
 97                   .helptext("Compare against a specified revision")
 98                   .optional(),
 99             Option.shortcut("o")
100                   .fullname("output")
101                   .describe("DIR")
102                   .helptext("Output directory")
103                   .optional(),
104             Option.shortcut("u")
105                   .fullname("username")
106                   .describe("NAME")
107                   .helptext("Use that username instead of 'guessing' one")
108                   .optional(),
109             Option.shortcut("")
110                   .fullname("upstream")
111                   .describe("URL")
112                   .helptext("The URL to the upstream repository")
113                   .optional(),
114             Option.shortcut("t")
115                   .fullname("title")
116                   .describe("TITLE")
117                   .helptext("The title of the webrev")
118                   .optional(),
119             Option.shortcut("c")
120                   .fullname("cr")
121                   .describe("CR#")
122                   .helptext("Include link to CR (aka bugid) in the main page")
123                   .optional(),
124             Switch.shortcut("b")
125                   .fullname("")
126                   .helptext("Do not ignore changes in whitespace (always true)")
127                   .optional(),
128             Switch.shortcut("m")
129                   .fullname("mercurial")
130                   .helptext("Deprecated: force use of mercurial")
131                   .optional(),
132             Switch.shortcut("C")
133                   .fullname("no-comments")
134                   .helptext("Don't show comments")
135                   .optional(),
136             Switch.shortcut("N")
137                   .fullname("no-outgoing")
138                   .helptext("Do not compare against remote, use only 'status'")
139                   .optional(),
140             Switch.shortcut("v")
141                   .fullname("version")
142                   .helptext("Print the version of this tool")
143                   .optional());
144 
145         var inputs = List.of(
146             Input.position(0)
147                  .describe("FILE")
148                  .singular()
149                  .optional());
150 
151         var parser = new ArgumentParser("git webrev", flags, inputs);
152         var arguments = parser.parse(args);
153 
154         var version = Version.fromManifest().orElse("unknown");
155         if (arguments.contains("version")) {
156             System.out.println("git-webrev version: " + version);
157             System.exit(0);
158         }
159 
160         var cwd = Paths.get("").toAbsolutePath();
161         var repository = ReadOnlyRepository.get(cwd);
162         if (!repository.isPresent()) {
163             System.err.println(String.format("error: %s is not a repository", cwd.toString()));
164             System.exit(1);
165         }
166         var repo = repository.get();
167         var isMercurial = arguments.contains("mercurial");
168 
169         var upstream = arg("upstream", arguments, repo);
170         if (upstream == null) {
171             var remotes = repo.remotes();
172             if (remotes.contains("upstream")) {
173                 var pullPath = Remote.toWebURI(repo.pullPath("upstream"));
174                 var host = pullPath.getHost();
175                 if (host != null && host.endsWith("openjdk.java.net")) {
176                     upstream = pullPath.toString();
177                 } else if (host != null && host.equals("github.com")) {
178                     var path = pullPath.getPath();
179                     if (path != null && path.startsWith("/openjdk/")) {
180                         upstream = pullPath.toString();
181                     }
182                 }
183             } else if (remotes.contains("origin") || remotes.contains("default")) {
184                 var remote = isMercurial ? "default" : "origin";
185                 var pullPath = Remote.toWebURI(repo.pullPath(remote));
186                 var host = pullPath.getHost();
187                 if (host != null && host.endsWith("openjdk.java.net")) {
188                     upstream = pullPath.toString();
189                 } else if (host != null && host.equals("github.com")) {
190                     var path = pullPath.getPath();
191                     if (path != null && path.startsWith("/openjdk/")) {
192                         upstream = pullPath.toString();
193                     }
194                 }
195             }
196         }
197         var upstreamURL = upstream;
198 
199         var noOutgoing = arguments.contains("no-outgoing");
200         if (!noOutgoing) {
201             var config = repo.config("webrev.no-outgoing");
202             if (config.size() == 1) {
203                 var enabled = Set.of("TRUE", "ON", "1", "ENABLED");
204                 noOutgoing = enabled.contains(config.get(0).toUpperCase());
205             }
206         }
207 
208         var rev = arguments.contains("rev") ? resolve(repo, arguments.get("rev").asString()) : null;
209         if (rev == null) {
210             if (isMercurial) {
211                 resolve(repo, noOutgoing ? "tip" : "min(outgoing())^");
212             } else {
213                 if (noOutgoing) {
214                     rev = resolve(repo, "HEAD");
215                 } else {
216                     var commits = repo.commitMetadata("origin..HEAD", true);
217                     if (commits.isEmpty()) {
218                         rev = resolve(repo, "HEAD");
219                     } else {
220                         rev = resolve(repo, commits.get(0).hash().hex() + "^");
221                     }
222                 }
223             }
224         }
225 
226         var issue = arguments.contains("cr") ? arguments.get("cr").asString() : null;
227         if (issue != null) {
228             if (issue.startsWith("http")) {
229                 var uri = URI.create(issue);
230                 issue = Path.of(uri.getPath()).getFileName().toString();
231             } else if (isDigit(issue.charAt(0))) {
232                 issue = "JDK-" + issue;
233             }
234         }
235         if (issue == null) {
236             var pattern = Pattern.compile("(?:(" + String.join("|", KNOWN_JBS_PROJECTS) + ")-)?([0-9]+).*");
237             var currentBranch = repo.currentBranch();
238             if (currentBranch.isPresent()) {
239                 var branchName = currentBranch.get().name().toUpperCase();
240                 var m = pattern.matcher(branchName);
241                 if (m.matches()) {
242                     var project = m.group(1);
243                     if (project == null) {
244                         project = "JDK";
245                     }
246                     var id = m.group(2);
247                     issue = project + "-" + id;
248                 }
249             }
250         }
251 
252         var out = arg("output", arguments, repo);
253         if (out == null) {
254             out = "webrev";
255         }
256         var output = Path.of(out);
257 
258         var title = arguments.contains("title") ?
259             arguments.get("title").asString() : null;
260         if (title == null && issue != null) {
261             try {
262                 var uri = new URI(issue);
263                 title = Path.of(uri.getPath()).getFileName().toString();
264             } catch (URISyntaxException e) {
265                 title = null;
266             }
267         }
268         if (title == null && upstream != null) {
269             var index = upstream.lastIndexOf("/");
270             if (index != -1 && index + 1 < upstream.length()) {
271                 title = upstream.substring(index + 1);
272             }
273         }
274         if (title == null) {
275             title = Path.of("").toAbsolutePath().getFileName().toString();
276         }
277 
278         var username = arg("username", arguments, repo);
279         if (username == null) {
280             username = repo.username().orElse(System.getProperty("user.name"));
281         }
282         var author = Author.fromString(username);
283 
284         if (Files.exists(output)) {
285             clearDirectory(output);
286         }
287 
288         List<Path> files = List.of();
289         if (arguments.at(0).isPresent()) {
290             var path = arguments.at(0).via(Path::of);
291             if (path.equals(Path.of("-"))) {
292                 var reader = new BufferedReader(new InputStreamReader(System.in));
293                 files = reader.lines().map(Path::of).collect(Collectors.toList());
294             } else {
295                 files = Files.readAllLines(path).stream().map(Path::of).collect(Collectors.toList());
296             }
297         }
298 
299         var jbs = "https://bugs.openjdk.java.net/browse/";
300         var issueParts = issue != null ? issue.split("-") : new String[0];
301         var jbsProject = issueParts.length == 2 && KNOWN_JBS_PROJECTS.contains(issueParts[0])?
302             issueParts[0] : "JDK";
303         Webrev.repository(repo)
304               .output(output)
305               .title(title)
306               .upstream(upstream)
307               .username(author.name())
308               .commitLinker(hash -> upstreamURL == null ? null : upstreamURL + "/commit/" + hash)
309               .issueLinker(id -> jbs + (isDigit(id.charAt(0)) ? jbsProject + "-" : "") + id)
310               .issue(issue)
311               .version(version)
312               .files(files)
313               .generate(rev);
314     }

316     private static void apply(String[] args) throws Exception {
317         var inputs = List.of(
318             Input.position(0)
319                  .describe("webrev url")
320                  .singular()
321                  .required());
322 
323         var parser = new ArgumentParser("git webrev apply", List.of(), inputs);
324         var arguments = parser.parse(args);
325 
326         var cwd = Paths.get("").toAbsolutePath();
327         var repository = Repository.get(cwd).orElseGet(() -> {
328             System.err.println(String.format("error: %s is not a repository", cwd.toString()));
329             System.exit(1);
330             return null;
331         });
332 
333         var inputString = arguments.at(0).asString();
334         var webrevMetaData = WebrevMetaData.from(URI.create(inputString));
335         var patchFileURI = webrevMetaData.patchURI()
336                 .orElseThrow(() -> new IllegalStateException("Could not find patch file in webrev"));
337         var patchFile = downloadPatchFile(patchFileURI);
338 
339         repository.apply(patchFile, false);
340     }
341 
342     private static Path downloadPatchFile(URI uri) throws IOException, InterruptedException {
343         var client = HttpClient.newHttpClient();
344         var patchFile = Files.createTempFile("patch", ".patch");
345         var patchFileRequest = HttpRequest.newBuilder()
346                 .uri(uri)
347                 .build();
348         client.send(patchFileRequest, HttpResponse.BodyHandlers.ofFile(patchFile));
349         return patchFile;
350     }
351 
352     public static void main(String[] args) throws Exception {
353         var commands = List.of(
354                     Default.name("generate")
355                            .helptext("generate a webrev")
356                            .main(GitWebrev::generate),
357                     Command.name("apply")
358                            .helptext("apply a webrev from a webrev url")
359                            .main(GitWebrev::apply)
360                 );
361         HttpProxy.setup();
362 
363         var parser = new MultiCommandParser("git webrev", commands);
364         var command = parser.parse(args);
365         command.execute();
366     }
