 77     final static class Spec {
 78         final static class Frequency {
 79             static enum Interval {
 80                 HOURLY,
 81                 DAILY,
 82                 WEEKLY,
 83                 MONTHLY,
 84                 YEARLY;
 85 
 86                 boolean isHourly() {
 87                     return this.equals(HOURLY);
 88                 }
 89 
 90                 boolean isDaily() {
 91                     return this.equals(DAILY);
 92                 }
 93 
 94                 boolean isWeekly() {
 95                     return this.equals(WEEKLY);
 96                 }
 97 
 98                 boolean isMonthly() {
 99                     return this.equals(MONTHLY);
100                 }
101 
102                 boolean isYearly() {
103                     return this.equals(YEARLY);
104                 }
105             }

107             private final Interval interval;
108             private final DayOfWeek weekday;
109             private final Month month;
110             private final int day;
111             private final int hour;
112             private final int minute;
113 
114             private Frequency(Interval interval, DayOfWeek weekday, Month month, int day, int hour, int minute) {
115                 this.interval = interval;
116                 this.weekday = weekday;
117                 this.month = month;
118                 this.day = day;
119                 this.hour = hour;
120                 this.minute = minute;
121             }

123             static Frequency hourly(int minute) {
124                 return new Frequency(Interval.HOURLY, null, null, -1, -1, minute);
125             }
126 
127             static Frequency daily(int hour) {
128                 return new Frequency(Interval.DAILY, null, null, -1, hour, -1);
129             }
130 
131             static Frequency weekly(DayOfWeek weekday, int hour) {
132                 return new Frequency(Interval.WEEKLY, weekday, null, -1, hour, -1);
133             }
134 
135             static Frequency monthly(int day, int hour) {
136                 return new Frequency(Interval.MONTHLY, null, null, day, hour, -1);
137             }
138 
139             static Frequency yearly(Month month, int day, int hour) {
140                 return new Frequency(Interval.YEARLY, null, month, day, hour, -1);
141             }

217     @Override
218     public boolean concurrentWith(WorkItem other) {
219         if (!(other instanceof MergeBot)) {
220             return true;
221         }
222         var otherBot = (MergeBot) other;
223         return !target.name().equals(otherBot.target.name());
224     }

226     @Override
227     public void run(Path scratchPath) {
228         try {
229             var sanitizedUrl =
230                 URLEncoder.encode(fork.webUrl().toString(), StandardCharsets.UTF_8);
231             var dir = storage.resolve(sanitizedUrl);
232 
233             Repository repo = null;
234             if (!Files.exists(dir)) {
235                 log.info("Cloning " + fork.name());
236                 Files.createDirectories(dir);
237                 repo = Repository.clone(fork.url(), dir);
238             } else {
239                 log.info("Found existing scratch directory for " + fork.name());
240                 repo = Repository.get(dir).orElseThrow(() -> {
241                         return new RuntimeException("Repository in " + dir + " has vanished");
242                 });
243             }
244 
245             // Sync personal fork
246             var remoteBranches = repo.remoteBranches(target.url().toString());
247             for (var branch : remoteBranches) {
248                 var fetchHead = repo.fetch(target.url(), branch.hash().hex());
249                 repo.push(fetchHead, fork.url(), branch.name());
250             }
251 
252             // Must fetch once to update refs/heads
253             repo.fetchAll();
254 
255             var prTarget = fork.forge().repository(target.name()).orElseThrow(() ->
256                     new IllegalStateException("Can't get well-known repository " + target.name())
257             );
258             var prs = prTarget.pullRequests();
259             var currentUser = prTarget.forge().currentUser();
260 
261             for (var spec : specs) {
262                 var toBranch = spec.toBranch();
263                 var fromRepo = spec.fromRepo();
264                 var fromBranch = spec.fromBranch();
265 
266                 log.info("Deciding whether to merge " + fromRepo.name() + ":" + fromBranch.name() + " to " + toBranch.name());
267 
268                 // Checkout the branch to merge into
269                 repo.checkout(toBranch, false);
270                 var remoteBranch = new Branch(repo.upstreamFor(toBranch).orElseThrow(() ->
271                     new IllegalStateException("Could not get remote branch name for " + toBranch.name())
272                 ));
273                 repo.merge(remoteBranch); // should always be a fast-forward merge
274 
275                 var targetName = Path.of(target.name()).getFileName();
276                 var fromName = Path.of(fromRepo.name()).getFileName();
277                 var fromDesc = targetName.equals(fromName) ? fromBranch.name() : fromName + ":" + fromBranch.name();
278 
279                 // Check if merge conflict pull request is present
280                 var shouldMerge = true;
281                 var title = "Cannot automatically merge " + fromDesc + " to " + toBranch.name();
282                 var marker = "<!-- MERGE CONFLICTS -->";
283                 for (var pr : prs) {
284                     if (pr.title().equals(title) &&

285                         pr.body().startsWith(marker) &&
286                         currentUser.equals(pr.author())) {
287                         var lines = pr.body().split("\n");
288                         var head = new Hash(lines[1].substring(5, 45));
289                         if (repo.contains(toBranch, head)) {
290                             log.info("Closing resolved merge conflict PR " + pr.id() + ", will try merge");
291                             pr.addComment("Merge conflicts have been resolved, closing this PR");
292                             pr.setState(PullRequest.State.CLOSED);
293                         } else {
294                             log.info("Outstanding unresolved merge already present, will not merge");
295                             shouldMerge = false;





























296                         }
297                         break;
298                     }
299                 }
300 
301                 if (spec.frequency().isPresent()) {
302                     var now = clock.now();
303                     var desc = toBranch.name() + "->" + fromRepo.name() + ":" + fromBranch.name();
304                     var freq = spec.frequency().get();
305                     if (freq.isHourly()) {
306                         if (!hourly.containsKey(desc)) {
307                             hourly.put(desc, new HashSet<Integer>());
308                         }
309                         var minute = now.getMinute();
310                         var hour = now.getHour();
311                         if (freq.minute() == minute && !hourly.get(desc).contains(hour)) {
312                             hourly.get(desc).add(hour);
313                         } else {
314                             shouldMerge = false;
315                         }
316                     } else if (freq.isDaily()) {
317                         if (!daily.containsKey(desc)) {
318                             daily.put(desc, new HashSet<Integer>());
319                         }
320                         var hour = now.getHour();
321                         var day = now.getDayOfYear();
322                         if (freq.hour() == hour && !daily.get(desc).contains(day)) {
323                             daily.get(desc).add(day);
324                         } else {
325                             shouldMerge = false;
326                         }
327                     } else if (freq.isWeekly()) {
328                         if (!weekly.containsKey(desc)) {
329                             weekly.put(desc, new HashSet<Integer>());
330                         }
331                         var weekOfYear = now.get(WeekFields.ISO.weekOfYear());
332                         var weekday = now.getDayOfWeek();
333                         var hour = now.getHour();
334                         if (freq.weekday().equals(weekday) &&
335                             freq.hour() == hour &&
336                             !weekly.get(desc).contains(weekOfYear)) {
337                             weekly.get(desc).add(weekOfYear);
338                         } else {
339                             shouldMerge = false;
340                         }
341                     } else if (freq.isMonthly()) {
342                         if (!monthly.containsKey(desc)) {
343                             monthly.put(desc, new HashSet<Month>());
344                         }
345                         var day = now.getDayOfMonth();
346                         var hour = now.getHour();
347                         var month = now.getMonth();
348                         if (freq.day() == day && freq.hour() == hour &&
349                             !monthly.get(desc).contains(month)) {
350                             monthly.get(desc).add(month);
351                         } else {
352                             shouldMerge = false;
353                         }
354                     } else if (freq.isYearly()) {
355                         if (!yearly.containsKey(desc)) {
356                             yearly.put(desc, new HashSet<Integer>());
357                         }
358                         var month = now.getMonth();
359                         var day = now.getDayOfMonth();
360                         var hour = now.getHour();
361                         var year = now.getYear();
362                         if (freq.month().equals(month) &&
363                             freq.day() == day &&
364                             freq.hour() == hour &&
365                             !yearly.get(desc).contains(year)) {
366                             yearly.get(desc).add(year);
367                         } else {
368                             shouldMerge = false;
369                         }
370                     }
371                 }
372 
373                 if (!shouldMerge) {
374                     log.info("Will not merge " + fromRepo.name() + ":" + fromBranch.name() + " to " + toBranch.name());
375                     continue;
376                 }
377 
378                 log.info("Merging " + fromRepo.name() + ":" + fromBranch.name() + " to " + toBranch.name());







379                 log.info("Fetching " + fromRepo.name() + ":" + fromBranch.name());
380                 var fetchHead = repo.fetch(fromRepo.url(), fromBranch.name());
381                 var head = repo.resolve(toBranch.name()).orElseThrow(() ->
382                         new IOException("Could not resolve branch " + toBranch.name())
383                 );
384                 if (repo.contains(toBranch, fetchHead)) {
385                     log.info("Nothing to merge");
386                     continue;
387                 }
388 
389                 var isAncestor = repo.isAncestor(head, fetchHead);
390 
391                 log.info("Trying to merge into " + toBranch.name());
392                 IOException error = null;
393                 try {
394                     repo.merge(fetchHead);
395                 } catch (IOException e) {
396                     error = e;
397                 }
398 
399                 if (error == null) {
400                     log.info("Pushing successful merge");
401                     if (!isAncestor) {
402                         repo.commit("Automatic merge of " + fromDesc + " into " + toBranch,
403                                 "duke", "duke@openjdk.org");
404                     }
405                     repo.push(toBranch, target.url().toString(), false);
406                 } else {
407                     log.info("Got error: " + error.getMessage());
408                     log.info("Aborting unsuccesful merge");
409                     var status = repo.status();
410                     repo.abortMerge();
411 
412                     var fromRepoName = Path.of(fromRepo.webUrl().getPath()).getFileName();
413                     var branchDesc = fromRepoName + "/" + fromBranch.name() + "->" + toBranch.name();
414                     repo.push(fetchHead, fork.url(), branchDesc, true);
415 
416                     log.info("Creating pull request to alert");
417                     var mergeBase = repo.mergeBase(fetchHead, head);
418 
419                     var message = new ArrayList<String>();
420                     message.add(marker);
421                     message.add("<!-- " + fetchHead.hex() + " -->");
422 
423                     var commits = repo.commits(mergeBase.hex() + ".." + fetchHead.hex(), true).asList();
424                     var numCommits = commits.size();
425                     var are = numCommits > 1 ? "are" : "is";
426                     var s = numCommits > 1 ? "s" : "";
427 
428                     message.add("Hi all,");
429                     message.add("");
430                     message.add("this is an _automatically_ generated merge request to notify you that there " +
431                                 are + " " + numCommits + " commit" + s + " from the branch `" + fromDesc + "`" +
432                                 "that can **not** be merged into the branch `" + toBranch.name() + "`:");
433 
434                     message.add("");
435                     var unmerged = status.stream().filter(entry -> entry.status().isUnmerged()).collect(Collectors.toList());
436                     if (unmerged.size() <= 10) {
437                         var files = unmerged.size() > 1 ? "files" : "file";
438                         message.add("The following " + files + " contains merge conflicts:");
439                         message.add("");
440                         for (var fileStatus : unmerged) {
441                             message.add("- " + fileStatus.source().path().orElseThrow());
442                         }
443                     } else {
444                         message.add("Over " + unmerged.size() + " files contains merge conflicts.");
445                     }
446                     message.add("");
447 











448                     message.add("The following paragraphs will give an example on how to solve these " +
449                                 "merge conflicts and create a pull request to integrate them. If you are " +
450                                 "using a workflow different from the one below, feel free to use that instead. " +
451                                 "It is important that the title of the pull request you create is " +
452                                 "`Merge " + fromDesc + "`, otherwise the bots will _not_ understand that the " +
453                                 "pull request you create is a \"merge style\" pull request.");
454                     message.add("");
455                     var localBranchName = "merge-" + fromDesc.replace(":", "-") + "-into-" + toBranch.name() + "-" + commits.get(0).hash().abbreviate();
456                     message.add("The below commands should be run in a local clone of your " +
457                                 "[personal fork](https://wiki.openjdk.java.net/display/skara#Skara-Personalforks) " +
458                                 "of the [" + target.name() + "](" + target.nonTransformedWebUrl() + ") repository. " +
459                                 "These commands will allow you to view and resolve the merge conflicts. Note that " +
460                                 "the name of the local branch in the commands, " +
461                                 "`" + localBranchName + "`" +
462                                 ", is just an example, feel free to give the local branch any name you prefer.");
463                     message.add("");

464                     message.add("```bash");

465                     message.add("$ git checkout " + toBranch.name());
466                     message.add("$ git pull " + target.nonTransformedWebUrl() + " " + toBranch.name());
467                     message.add("$ git checkout -b " + localBranchName);
468                     message.add("$ git pull " + fromRepo.nonTransformedWebUrl() + " " + fromBranch.name());





469                     message.add("```");
470                     message.add("");
471                     message.add("When you have resolved the conflicts resulting from the `git pull` command " +
472                                 "above, run the following commands to create a merge commit:");
473                     message.add("");
474                     message.add("```bash");
475                     message.add("$ git add paths/to/files/with/conflicts");
476                     message.add("$ git commit -m 'Merge " + fromDesc + "'");
477                     message.add("```");
478                     message.add("");
479                     message.add("The commit message does not have to be `Merge " + fromDesc + "`, " +
480                                 "but it is convenient for when you will create a pull request. Many tools " +
481                                 "will by default use the commit message of the most recent commit on a branch " +
482                                 "as the title for a pull request from that branch. This means that if you use " +
483                                 "the commit message `Merge " + fromDesc + "` as the commit message then the " +
484                                 "title of the pull request will (depending to tools used to create the " +
485                                 "pull request) be of a format that the bots expect.");
486                     message.add("");
487                     message.add("Proceed to [publish the local branch](https://wiki.openjdk.java.net/display/SKARA/FAQ#FAQ-HowdoIpushalocalbranchtoaremoterepository?) " +
488                                 "and [create a pull request](https://wiki.openjdk.java.net/display/SKARA/FAQ#FAQ-HowdoIcreateapullrequest?) " +
489                                 "towards the `" + toBranch.name() + "` branch in the " +
490                                 "[" + target.name() + "](" + target.webUrl() + ") repository. The resulting pull " +
491                                 "request can then integrated as usual once it has passed all required " +
492                                 "pre-integration checks.");


493                     message.add("");
494                     message.add("Thanks,");
495                     message.add("J. Duke");
496 
497                     var prFromFork = fork.createPullRequest(prTarget,
498                                                             toBranch.name(),
499                                                             branchDesc,
500                                                             title,
501                                                             message);
502                     var prFromTarget = target.pullRequest(prFromFork.id());
503                     prFromTarget.addLabel("failed-auto-merge");
504                 }
505             }
506         } catch (IOException e) {
507             throw new UncheckedIOException(e);
508         }
509     }
