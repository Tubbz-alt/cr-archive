23 package org.openjdk.skara.issuetracker;
24 
25 import org.openjdk.skara.host.HostUser;
26 
27 import java.time.ZonedDateTime;
28 import java.util.Objects;
29 
30 public class Comment {
31     private final String id;
32     private final String body;
33     private final HostUser author;
34     private final ZonedDateTime createdAt;
35     private final ZonedDateTime updatedAt;
36 
37     public Comment(String id, String body, HostUser author, ZonedDateTime createdAt, ZonedDateTime updatedAt) {
38         this.id = id;
39         this.body = body;
40         this.author = author;
41         this.createdAt = createdAt;
42         this.updatedAt = updatedAt;
43     }
44 
45     public String id() {
46         return id;
47     }
48 
49     public String body() {
50         return body;
51     }
52 
53     public HostUser author() {
54         return author;
55     }
56 
57     public ZonedDateTime createdAt() {
58         return createdAt;
59     }
60 
61     public ZonedDateTime updatedAt() {
62         return updatedAt;
63     }
64 
65     @Override
66     public boolean equals(Object o) {
67         if (this == o) {
68             return true;
69         }
70         if (o == null || getClass() != o.getClass()) {
71             return false;
72         }
73         Comment comment = (Comment) o;
74         return id.equals(comment.id) &&
75                 body.equals(comment.body) &&
76                 author.equals(comment.author) &&
77                 createdAt.equals(comment.createdAt) &&
78                 updatedAt.equals(comment.updatedAt);
79     }
80 
81     @Override
82     public int hashCode() {
83         return Objects.hash(id, body, author, createdAt, updatedAt);
84     }





85 }
