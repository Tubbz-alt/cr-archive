23 package org.openjdk.skara.bots.notify;
24 
25 import org.openjdk.skara.vcs.*;
26 
27 import java.util.Objects;
28 
29 class ResolvedBranch {
30     private final Branch branch;
31     private final Hash hash;
32 
33     ResolvedBranch(Branch branch, Hash hash) {
34         this.branch = branch;
35         this.hash = hash;
36     }
37 
38     public Branch branch() {
39         return branch;
40     }
41 
42     public Hash hash() {
43         return hash;
44     }
45 
46     @Override
47     public String toString() {
48         return branch.name() + ":" + hash().hex();
49     }
50 
51     @Override
52     public boolean equals(Object o) {
53         if (this == o) {
54             return true;
55         }
56         if (o == null || getClass() != o.getClass()) {
57             return false;
58         }
59         ResolvedBranch that = (ResolvedBranch) o;
60         return Objects.equals(branch, that.branch) &&
61                 Objects.equals(hash, that.hash);
62     }
63 
64     @Override
65     public int hashCode() {
66         return Objects.hash(branch, hash);
67     }
68 }
