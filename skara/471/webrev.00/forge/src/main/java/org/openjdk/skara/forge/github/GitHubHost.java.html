 48     public GitHubHost(URI uri, GitHubApplication application, Pattern webUriPattern, String webUriReplacement) {
 49         this.uri = uri;
 50         this.webUriPattern = webUriPattern;
 51         this.webUriReplacement = webUriReplacement;
 52         this.application = application;
 53         this.pat = null;
 54 
 55         var baseApi = URIBuilder.base(uri)
 56                 .appendSubDomain("api")
 57                 .setPath("/")
 58                 .build();
 59 
 60         request = new RestRequest(baseApi, application.authId(), () -> Arrays.asList(
 61                 "Authorization", "token " + getInstallationToken().orElseThrow(),
 62                 "Accept", "application/vnd.github.machine-man-preview+json",
 63                 "Accept", "application/vnd.github.antiope-preview+json"));
 64 
 65         var graphQLAPI = URIBuilder.base(uri)
 66                 .appendSubDomain("api")
 67                 .setPath("/graphql")
 68                 .build();
 69         graphQL = new RestRequest(graphQLAPI, application.authId(), () -> Arrays.asList(
 70                 "Authorization", "bearer " + getInstallationToken().orElseThrow(),
 71                 "Accept", "application/vnd.github.machine-man-preview+json",
 72                 "Accept", "application/vnd.github.antiope-preview+json",
 73                 "Accept", "application/vnd.github.shadow-cat-preview+json",
 74                 "Accept", "application/vnd.github.comfort-fade-preview+json"
 75         ));
 76     }
 77 
 78     RestRequest graphQL() {
 79         if (graphQL == null) {
 80             throw new IllegalStateException("Cannot use GraphQL API without authorization");
 81         }
 82         return graphQL;
 83     }

102                 .setPath("/graphql")

117         this.pat = null;

129     public URI getURI() {

133     URI getWebURI(String endpoint) {
134         var baseWebUri = URIBuilder.base(uri)
135                                    .setPath(endpoint)
136                                    .build();
137 
138         if (webUriPattern == null) {
139             return baseWebUri;
140         }
141 
142         var matcher = webUriPattern.matcher(baseWebUri.toString());
143         if (!matcher.matches()) {
144             return baseWebUri;
145 
146         }
147         return URIBuilder.base(matcher.replaceAll(webUriReplacement)).build();
148     }

150     Optional<String> getInstallationToken() {
151         if (application != null) {
152             return Optional.of(application.getInstallationToken());
153         }
154 
155         if (pat != null) {
156             return Optional.of(pat.password());
157         }
158 
159         return Optional.empty();
160     }

162     Optional<String> authId() {
163         if (application != null) {
164             return Optional.of(application.authId());
165         }
166 
167         if (pat != null) {
168             return Optional.of(pat.username());
169         }
170 
171         return Optional.empty();
172     }

179     // Most GitHub API's return user information in this format

181         return parseUserObject(json.get("user"));

184     HostUser parseUserObject(JSONValue json) {
185         return new HostUser(json.get("id").asInt(), json.get("login").asString(),
186                             () -> getFullName(json.get("login").asString()));
187     }

189     @Override
190     public boolean isValid() {
191         try {
192             var endpoints = request.get("")
193                                    .executeUnparsed();
194             var parsed = JSON.parse(endpoints);
195             if (parsed != null && parsed.contains("current_user_url")) {
196                 return true;
197             } else {
198                 log.fine("Error during GitHub host validation: unexpected endpoint list: " + endpoints);
199                 return false;
200             }
201         } catch (IOException e) {
202             log.fine("Error during GitHub host validation: " + e);
203             return false;
204         }
205     }

229     @Override
230     public Optional<HostUser> user(String username) {
231         var details = request.get("users/" + URLEncoder.encode(username, StandardCharsets.UTF_8))
232                              .onError(r -> Optional.of(JSON.of()))
233                              .execute();
234         if (details.isNull()) {
235             return Optional.empty();
236         }
237 
238         return Optional.of(asHostUser(details.asObject()));
239     }
240 
241     private static HostUser asHostUser(JSONObject details) {
242         // Always present
243         var login = details.get("login").asString();
244         var id = details.get("id").asInt();
245 
246         var name = details.get("name").asString();
247         if (name == null) {
248             name = login;
249         }
250         var email = details.get("email").asString();
251         return new HostUser(id, login, name, email);
252     }
253 
254     @Override
255     public HostUser currentUser() {
256         if (currentUser == null) {
257             if (application != null) {
258                 var appDetails = application.getAppDetails();
259                 var appName = appDetails.get("name").asString() + "[bot]";
260                 currentUser = user(appName).get();
261             } else if (pat != null) {
262                 // Cannot always trust username in PAT, e.g. Git Credential Manager
263                 // on Windows always return "PersonalAccessToken" as username.
264                 // Query GitHub for the username instead.
265                 var details = request.get("user").execute().asObject();
266                 currentUser = asHostUser(details);
267             } else {
268                 throw new IllegalStateException("No credentials present");
269             }
270         }
271         return currentUser;
272     }

274     @Override
275     public boolean supportsReviewBody() {
276         return true;

279     @Override
280     public boolean isMemberOf(String groupId, HostUser user) {
281         long gid = 0L;
282         try {
283             gid = Long.parseLong(groupId);
284         } catch (NumberFormatException e) {
285             throw new IllegalArgumentException("Group id is not a number: " + groupId);
286         }
287         var username = URLEncoder.encode(user.userName(), StandardCharsets.UTF_8);
288         var orgs = request.get("users/" + username + "/orgs").execute().asArray();
289         for (var org : orgs) {
290             if (org.get("id").asLong() == gid) {
291                 return true;
292             }
293         }
294 
295         return false;
296     }
