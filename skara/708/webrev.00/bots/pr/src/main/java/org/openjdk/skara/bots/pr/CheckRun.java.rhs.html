100     private List<Issue> issues() {

115     private List<org.openjdk.skara.issuetracker.Issue> issuesOfDisallowedType() {
116         var issueProject = issueProject();
117         var allowed = allowedIssueTypes();
118         if (issueProject != null && allowed != null) {
119             return issues().stream()
120                            .filter(i -> i.project().equals(Optional.of(issueProject.name())))
121                            .map(i -> issueProject.issue(i.shortId()))
122                            .filter(Optional::isPresent)
123                            .map(Optional::get)
124                            .filter(i -> i.properties().containsKey("issuetype"))
125                            .filter(i -> !allowed.contains(i.properties().get("issuetype").asString()))
126                            .collect(Collectors.toList());
127         }
128         return List.of();
129     }

131     private List<String> allowedTargetBranches() {
132         return pr.repository()
133                  .branches()
134                  .stream()
135                  .map(HostedBranch::name)
136                  .map(name -> workItem.bot.allowedTargetBranches().matcher(name))
137                  .filter(Matcher::matches)
138                  .map(Matcher::group)
139                  .collect(Collectors.toList());

142     // For unknown contributors, check that all commits have the same name and email
143     private boolean checkCommitAuthor(List<CommitMetadata> commits) throws IOException {
144         var author = censusInstance.namespace().get(pr.author().id());
145         if (author != null) {
146             return true;
147         }
148 
149         var names = new HashSet<String>();
150         var emails = new HashSet<String>();
151 
152         for (var commit : commits) {
153             names.add(commit.author().name());
154             emails.add(commit.author().email());
155         }
156 
157         return ((names.size() == 1) && emails.size() == 1);
158     }

160     // Additional bot-specific checks that are not handled by JCheck
161     private List<String> botSpecificChecks(Hash finalHash) throws IOException {
162         var ret = new ArrayList<String>();
163 
164         if (bodyWithoutStatus().isBlank()) {
165             var error = "The pull request body must not be empty.";
166             ret.add(error);
167         }
168 
169         if (!isTargetBranchAllowed()) {
170             var error = "The branch `" + pr.targetRef() + "` is not allowed as target branch. The allowed target branches are:\n" +
171                     allowedTargetBranches().stream()
172                     .map(name -> "   - " + name)
173                     .collect(Collectors.joining("\n"));
174             ret.add(error);
175         }
176 
177         var disallowedIssues = issuesOfDisallowedType();
178         if (!disallowedIssues.isEmpty()) {
179             var s = disallowedIssues.size() > 1 ? "s " : " ";
180             var are = disallowedIssues.size() > 1 ? "are" : "is";
181             var links = disallowedIssues.stream()
182                                         .map(i -> "[" + i.id() + "](" + i.webUrl() + ")")
183                                         .collect(Collectors.toList());
184             var error = "The issue" + s + String.join(",", links) + " " + are + " not of the expected type. The allowed issue types are:\n" +
185                 allowedIssueTypes().stream()
186                 .map(name -> "   - " + name)
187                 .collect(Collectors.joining("\n"));
188             ret.add(error);
189         }
190 
191         var headHash = pr.headHash();
192         var originalCommits = localRepo.commitMetadata(baseHash, headHash);
193 
194         if (!checkCommitAuthor(originalCommits)) {
195             var error = "For contributors who are not existing OpenJDK Authors, commit attribution will be taken from " +
196                     "the commits in the PR. However, the commits in this PR have inconsistent user names and/or " +
197                     "email addresses. Please amend the commits.";
198             ret.add(error);
199         }
200 
201         for (var blocker : workItem.bot.blockingCheckLabels().entrySet()) {
202             if (labels.contains(blocker.getKey())) {
203                 ret.add(blocker.getValue());
204             }
205         }
206 
207         return ret;
208     }

210     private void updateCheckBuilder(CheckBuilder checkBuilder, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
211         if (visitor.isReadyForReview() && additionalErrors.isEmpty()) {
212             checkBuilder.complete(true);
213         } else {
214             checkBuilder.title("Required");
215             var summary = Stream.concat(visitor.messages().stream(), additionalErrors.stream())
216                                 .sorted()
217                                 .map(m -> "- " + m)
218                                 .collect(Collectors.joining("\n"));
219             checkBuilder.summary(summary);
220             for (var annotation : visitor.getAnnotations()) {
221                 checkBuilder.annotation(annotation);
222             }
223             checkBuilder.complete(false);
224         }
225     }

227     private void updateReadyForReview(PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
228         // Additional errors are not allowed
229         if (!additionalErrors.isEmpty()) {
230             newLabels.remove("rfr");
231             return;
232         }
233 
234         // Draft requests are not for review
235         if (pr.isDraft()) {
236             newLabels.remove("rfr");
237             return;
238         }
239 
240         // Check if the visitor found any issues that should be resolved before reviewing
241         if (visitor.isReadyForReview()) {
242             newLabels.add("rfr");
243         } else {
244             newLabels.remove("rfr");
245         }
246     }

248     private String getRole(String username) {
249         var project = censusInstance.project();
250         var version = censusInstance.census().version().format();
251         if (project.isReviewer(username, version)) {
252             return "**Reviewer**";
253         } else if (project.isCommitter(username, version)) {
254             return "Committer";
255         } else if (project.isAuthor(username, version)) {
256             return "Author";
257         } else {
258             return "no project role";
259         }
260     }

262     private String formatReviewer(HostUser reviewer) {
263         var namespace = censusInstance.namespace();
264         var contributor = namespace.get(reviewer.id());
265         if (contributor == null) {
266             return reviewer.userName() + " (no known " + namespace.name() + " user name / role)";
267         } else {
268             var userNameLink = "[" + contributor.username() + "](@" + reviewer.userName() + ")";
269             return contributor.fullName().orElse(contributor.username()) + " (" + userNameLink + " - " +
270                     getRole(contributor.username()) + ")";
271         }
272     }

287     private Optional<String> getReviewersList(List<Review> reviews) {
288         var reviewers = reviews.stream()
289                                .filter(review -> review.verdict() == Review.Verdict.APPROVED)
290                                .map(review -> {
291                                    var entry = " * " + formatReviewer(review.reviewer());
292                                    if (!review.hash().equals(pr.headHash())) {
293                                        if (ignoreStaleReviews) {
294                                            entry += " 🔄 Re-review required (review applies to " + review.hash() + ")";
295                                        } else {
296                                            entry += " ⚠️ Review applies to " + review.hash();
297                                        }
298                                    }
299                                    return entry;
300                                })
301                                .collect(Collectors.joining("\n"));
302 
303         // Check for manually added reviewers
304         if (!ignoreStaleReviews) {
305             var namespace = censusInstance.namespace();
306             var allReviewers = PullRequestUtils.reviewerNames(reviews, namespace);
307             var additionalEntries = new ArrayList<String>();
308             for (var additional : Reviewers.reviewers(pr.repository().forge().currentUser(), comments)) {
309                 if (!allReviewers.contains(additional)) {
310                     additionalEntries.add(" * " + additional + " - " + getRole(additional) + " ⚠️ Added manually");
311                 }
312             }
313             if (!reviewers.isBlank()) {
314                 reviewers += "\n";
315             }
316             reviewers += String.join("\n", additionalEntries);
317         }
318 
319         if (reviewers.length() > 0) {
320             return Optional.of(reviewers);
321         } else {
322             return Optional.empty();
323         }
324     }

343     private boolean relaxedEquals(String s1, String s2) {
344         return s1.trim()
345                  .replaceAll("\\s+", " ")
346                  .equalsIgnoreCase(s2.trim()
347                                      .replaceAll("\\s+", " "));

351     private String getStatusMessage(List<Comment> comments, List<Review> reviews, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
352         var progressBody = new StringBuilder();
353         progressBody.append("---------\n");
354         progressBody.append("### Progress\n");
355         progressBody.append(getChecksList(visitor));
356 
357         var allAdditionalErrors = Stream.concat(visitor.hiddenMessages().stream(), additionalErrors.stream())
358                                         .sorted()
359                                         .collect(Collectors.toList());
360         if (!allAdditionalErrors.isEmpty()) {
361             progressBody.append("\n\n### Error");
362             if (allAdditionalErrors.size() > 1) {
363                 progressBody.append("s");
364             }
365             progressBody.append("\n");
366             progressBody.append(getAdditionalErrorsList(allAdditionalErrors));
367         }
368 
369         var issues = issues();
370         var issueProject = issueProject();
371         if (issueProject != null && !issues.isEmpty()) {
372             progressBody.append("\n\n### Issue");
373             if (issues.size() > 1) {
374                 progressBody.append("s");
375             }
376             progressBody.append("\n");
377             for (var currentIssue : issues) {
378                 progressBody.append(" * ");
379                 if (currentIssue.project().isPresent() && !currentIssue.project().get().equals(issueProject.name())) {
380                     progressBody.append("⚠️ Issue `");
381                     progressBody.append(currentIssue.id());
382                     progressBody.append("` does not belong to the `");
383                     progressBody.append(issueProject.name());
384                     progressBody.append("` project.");
385                 } else {
386                     try {
387                         var iss = issueProject.issue(currentIssue.shortId());
388                         if (iss.isPresent()) {
389                             progressBody.append("[");
390                             progressBody.append(iss.get().id());
391                             progressBody.append("](");
392                             progressBody.append(iss.get().webUrl());
393                             progressBody.append("): ");
394                             progressBody.append(iss.get().title());
395                             if (!relaxedEquals(iss.get().title(), currentIssue.description())) {
396                                 progressBody.append(" ⚠️ Title mismatch between PR and JBS.");
397                             }
398                             progressBody.append("\n");
399                         } else {
400                             progressBody.append("⚠️ Failed to retrieve information on issue `");
401                             progressBody.append(currentIssue.id());
402                             progressBody.append("`.\n");
403                         }
404                     } catch (RuntimeException e) {
405                         progressBody.append("⚠️ Temporary failure when trying to retrieve information on issue `");
406                         progressBody.append(currentIssue.id());
407                         progressBody.append("`.\n");
408                     }
409                 }
410             }
411         }
412 
413         getReviewersList(reviews).ifPresent(reviewers -> {
414             progressBody.append("\n\n### Reviewers\n");
415             progressBody.append(reviewers);
416         });
417 
418         getContributorsList(comments).ifPresent(contributors -> {
419             progressBody.append("\n\n### Contributors\n");
420             progressBody.append(contributors);
421         });
422 
423         progressBody.append("\n\n### Download\n");
424         progressBody.append(checkoutCommands());
425 
426         return progressBody.toString();
427     }

429     private String checkoutCommands() {
430         var repoUrl = pr.repository().webUrl();
431         return
432            "`$ git fetch " + repoUrl + " " + pr.fetchRef() + ":pull/" + pr.id() + "`\n" +
433            "`$ git checkout pull/" + pr.id() + "`\n";

436     private String bodyWithoutStatus() {
437         var description = pr.body();
438         var markerIndex = description.lastIndexOf(progressMarker);
439         return (markerIndex < 0 ?
440                 description :
441                 description.substring(0, markerIndex)).trim();
442     }

444     private String updateStatusMessage(String message) {
445         var description = pr.body();
446         var markerIndex = description.lastIndexOf(progressMarker);
447 
448         if (markerIndex >= 0 && description.substring(markerIndex).equals(message)) {
449             log.info("Progress already up to date");
450             return description;
451         }
452         var originalBody = bodyWithoutStatus();
453         if (originalBody.isBlank()) {
454             originalBody = emptyPrBodyMarker;
455         }
456         var newBody = originalBody + "\n" + progressMarker + "\n" + message;
457 
458         // TODO? Retrieve the body again here to lower the chance of concurrent updates
459         pr.setBody(newBody);
460         return newBody;
461     }

463     private String updateTitle() {
464         var title = pr.title();
465         var m = ISSUE_ID_PATTERN.matcher(title);
466         var project = issueProject();
467 
468         var newTitle = title;
469         if (m.matches() && project != null) {
470             var id = m.group(1);
471             var issue = project.issue(id);
472             if (issue.isPresent()) {
473                 newTitle = id + ": " + issue.get().title();
474             }
475         }
476 
477         if (!title.equals(newTitle)) {
478             pr.setTitle(newTitle);
479         }
480 
481         return newTitle;
482     }

484     private String verdictToString(Review.Verdict verdict) {
485         switch (verdict) {
486             case APPROVED:
487                 return "changes are approved";
488             case DISAPPROVED:
489                 return "more changes needed";
490             case NONE:
491                 return "comment added";
492             default:
493                 throw new RuntimeException("Unknown verdict: " + verdict);
494         }
495     }

517     private String getMergeReadyComment(String commitMessage, List<Review> reviews) {
518         var message = new StringBuilder();
519         message.append("@");
520         message.append(pr.author().userName());
521         message.append(" This change now passes all automated pre-integration checks");
522 
523         try {
524             var hasContributingFile =
525                 !localRepo.files(pr.targetHash(), Path.of("CONTRIBUTING.md")).isEmpty();
526             if (hasContributingFile) {
527                 message.append(". When the change also fulfills all ");
528                 message.append("[project specific requirements](https://github.com/");
529                 message.append(pr.repository().name());
530                 message.append("/blob/");
531                 message.append(pr.targetRef());
532                 message.append("/CONTRIBUTING.md)");
533             }
534         } catch (IOException e) {
535             throw new UncheckedIOException(e);
536         }
537 
538         message.append(", type `/integrate` in a new comment to proceed. After integration, ");
539         message.append("the commit message will be:\n");
540         message.append("```\n");
541         message.append(commitMessage);
542         message.append("\n```\n");
543 
544         message.append("- If you would like to add a summary, use the `/summary` command.\n");
545         message.append("- To credit additional contributors, use the `/contributor` command.\n");
546         message.append("- To add additional solved issues, use the `/issue` command.\n");
547 
548         var divergingCommits = checkablePullRequest.divergingCommits();
549         if (divergingCommits.size() > 0) {
550             message.append("\n");
551             message.append("Since the source branch of this PR was last updated there ");
552             if (divergingCommits.size() == 1) {
553                 message.append("has been 1 commit ");
554             } else {
555                 message.append("have been ");
556                 message.append(divergingCommits.size());
557                 message.append(" commits ");
558             }
559             message.append("pushed to the `");
560             message.append(pr.targetRef());
561             message.append("` branch:\n\n");
562             divergingCommits.stream()
563                             .limit(10)
564                             .forEach(c -> message.append(" * ").append(c.hash().hex()).append(": ").append(c.message().get(0)).append("\n"));
565             if (divergingCommits.size() > 10) {
566                 message.append(" * ... and ").append(divergingCommits.size() - 10).append(" more: ")
567                        .append(pr.repository().webUrl(baseHash.hex(), pr.targetRef())).append("\n");
568             }
569 
570             message.append("\n");
571             message.append("As there are no conflicts, your changes will automatically be rebased on top of ");
572             message.append("these commits when integrating. If you prefer to avoid automatic rebasing, please merge `");
573             message.append(pr.targetRef());
574             message.append("` into your branch, and then specify the current head hash when integrating, like this: ");
575             message.append("`/integrate ");
576             message.append(pr.targetHash());
577             message.append("`.\n");
578         } else {
579             message.append("\n");
580             message.append("There are currently no new commits on the `");
581             message.append(pr.targetRef());
582             message.append("` branch since the last update of the source branch of this PR. If another commit should be pushed before ");
583             message.append("you perform the `/integrate` command, your PR will be automatically rebased. If you would like to avoid ");
584             message.append("potential automatic rebasing, specify the current head hash when integrating, like this: ");
585             message.append("`/integrate ");
586             message.append(pr.targetHash());
587             message.append("`.\n");
588         }
589 
590         if (!censusInstance.isCommitter(pr.author())) {
591             message.append("\n");
592             var contributor = censusInstance.namespace().get(pr.author().id());
593             if (contributor == null) {
594                 message.append("As you are not a known OpenJDK [Author](https://openjdk.java.net/bylaws#author), ");
595             } else {
596                 message.append("As you do not have Committer status in this project, ");
597             }
598 
599             message.append("an existing [Committer](https://openjdk.java.net/bylaws#committer) must agree to ");
600             message.append("[sponsor](https://openjdk.java.net/sponsor/) your change. ");
601             var candidates = reviews.stream()
602                                     .filter(review -> censusInstance.isCommitter(review.reviewer()))
603                                     .map(review -> "@" + review.reviewer().userName())
604                                     .collect(Collectors.joining(", "));
605             if (candidates.length() > 0) {
606                 message.append("Possible candidates are the reviewers of this PR (");
607                 message.append(candidates);
608                 message.append(") but any other Committer may sponsor as well. ");
609             }
610             message.append("\n\n");
611             message.append("➡️ To flag this PR as ready for integration with the above commit message, type ");
612             message.append("`/integrate` in a new comment. (Afterwards, your sponsor types ");
613             message.append("`/sponsor` in a new comment to perform the integration).\n");
614         } else {
615             message.append("\n");
616             message.append("➡️ To integrate this PR with the above commit message to the `" + pr.targetRef() + "` branch, type ");
617             message.append("`/integrate` in a new comment.\n");
618         }
619         message.append(mergeReadyMarker);
620         return message.toString();
621     }

623     private String getMergeNoLongerReadyComment() {
624         var message = new StringBuilder();
625         message.append("@");
626         message.append(pr.author().userName());
627         message.append(" This change is no longer ready for integration - check the PR body for details.\n");
628         message.append(mergeReadyMarker);
629         return message.toString();
630     }

632     private void updateMergeReadyComment(boolean isReady, String commitMessage, List<Comment> comments, List<Review> reviews, boolean rebasePossible) {
633         var existing = findComment(comments, mergeReadyMarker);
634         if (isReady && rebasePossible) {
635             var message = getMergeReadyComment(commitMessage, reviews);
636             if (existing.isEmpty()) {
637                 pr.addComment(message);
638             } else {
639                 pr.updateComment(existing.get().id(), message);
640             }
641         } else {
642             existing.ifPresent(comment -> pr.updateComment(comment.id(), getMergeNoLongerReadyComment()));
643         }

647         var existing = findComment(comments, sourceBranchWarningMarker);
648         if (existing.isPresent()) {

650             return;

652         var branch = pr.sourceRef();

658             "your personal fork with the upstream repository.\n" +
659             "\n" +
660             "To avoid this situation, create a new branch for your changes and reset the `" + branch + "` branch. " +
661             "You can do this by running the following commands in a local repository for your personal fork. " +
662             "_Note_: you do *not* have to name the new branch `NEW-BRANCH-NAME`." +

664             "```" +

666             "$ git checkout -b NEW-BRANCH-NAME\n" +

668             "$ git push -f origin " + branch + "\n" +

671             "Then proceed to create a new pull request with `NEW-BRANCH-NAME` as the source branch and " +
672             "close this one.\n" +
673             sourceBranchWarningMarker;

715     private void checkStatus() {
716         var checkBuilder = CheckBuilder.create("jcheck", pr.headHash());
717         var censusDomain = censusInstance.configuration().census().domain();
718         Exception checkException = null;
719 
720         try {
721             // Post check in-progress
722             log.info("Starting to run jcheck on PR head");
723             pr.createCheck(checkBuilder.build());
724 
725             var ignored = new PrintWriter(new StringWriter());
726             var rebasePossible = true;
727             var commitHash = pr.headHash();
728             var mergedHash = checkablePullRequest.mergeTarget(ignored);
729             if (mergedHash.isPresent()) {
730                 commitHash = mergedHash.get();
731             } else {
732                 rebasePossible = false;
733             }
734 
735             List<String> additionalErrors = List.of();
736             Hash localHash;
737             try {
738                 localHash = checkablePullRequest.commit(commitHash, censusInstance.namespace(), censusDomain, null);
739             } catch (CommitFailure e) {
740                 additionalErrors = List.of(e.getMessage());
741                 localHash = baseHash;
742             }
743             PullRequestCheckIssueVisitor visitor = checkablePullRequest.createVisitor(localHash, censusInstance);
744             if (!localHash.equals(baseHash)) {
745                 // Determine current status
746                 var additionalConfiguration = AdditionalConfiguration.get(localRepo, localHash, pr.repository().forge().currentUser(), comments);
747                 checkablePullRequest.executeChecks(localHash, censusInstance, visitor, additionalConfiguration);
748                 additionalErrors = botSpecificChecks(localHash);
749             } else {
750                 if (additionalErrors.isEmpty()) {
751                     additionalErrors = List.of("This PR contains no changes");
752                 }
753             }
754             updateCheckBuilder(checkBuilder, visitor, additionalErrors);
755             updateReadyForReview(visitor, additionalErrors);
756 
757             // Calculate and update the status message if needed
758             var statusMessage = getStatusMessage(comments, activeReviews, visitor, additionalErrors);
759             var updatedBody = updateStatusMessage(statusMessage);
760             var title = updateTitle();
761 
762             // Post / update approval messages (only needed if the review itself can't contain a body)
763             if (!pr.repository().forge().supportsReviewBody()) {
764                 updateReviewedMessages(comments, allReviews);
765             }
766 
767             var commit = localRepo.lookup(localHash).orElseThrow();
768             var commitMessage = String.join("\n", commit.message());
769             var readyForIntegration = visitor.messages().isEmpty() && additionalErrors.isEmpty();
770             updateMergeReadyComment(readyForIntegration, commitMessage, comments, activeReviews, rebasePossible);
771             if (readyForIntegration && rebasePossible) {
772                 newLabels.add("ready");
773             } else {
774                 newLabels.remove("ready");
775             }
776             if (!rebasePossible) {
777                 if (!labels.contains("failed-auto-merge")) {
778                     addOutdatedComment(comments);
779                 }
780                 newLabels.add("merge-conflict");
781             } else {
782                 newLabels.remove("merge-conflict");
783             }
784 
785             if (pr.sourceRepository().isPresent()) {
786                 var branchNames = pr.repository().branches().stream().map(HostedBranch::name).collect(Collectors.toSet());
787                 if (!pr.repository().url().equals(pr.sourceRepository().get().url()) && branchNames.contains(pr.sourceRef())) {
788                     addSourceBranchWarningComment(comments);
789                 }
790             }
791 
792             if (!PullRequestUtils.isMerge(pr) && PullRequestUtils.containsForeignMerge(pr, localRepo)) {
793                 addMergeCommitWarningComment(comments);
794             }
795 
796             // Ensure that the ready for sponsor label is up to date
797             newLabels.remove("sponsor");
798             var readyHash = ReadyForSponsorTracker.latestReadyForSponsor(pr.repository().forge().currentUser(), comments);
799             if (readyHash.isPresent() && readyForIntegration) {
800                 var acceptedHash = readyHash.get();
801                 if (pr.headHash().equals(acceptedHash)) {
802                     newLabels.add("sponsor");
803                 }
804             }
805 
806             // Calculate current metadata to avoid unnecessary future checks
807             var metadata = workItem.getMetadata(title, updatedBody, pr.comments(), activeReviews, newLabels,
808                                                 censusInstance, pr.targetHash(), pr.isDraft());
809             checkBuilder.metadata(metadata);
810         } catch (Exception e) {
811             log.throwing("CommitChecker", "checkStatus", e);
812             newLabels.remove("ready");
813             checkBuilder.metadata("invalid");
814             checkBuilder.title("Exception occurred during jcheck - the operation will be retried");
815             checkBuilder.summary(e.getMessage());
816             checkBuilder.complete(false);
817             checkException = e;
818         }
819         var check = checkBuilder.build();
820         pr.updateCheck(check);
821 
822         // Synchronize the wanted set of labels
823         for (var newLabel : newLabels) {
824             if (!labels.contains(newLabel)) {
825                 pr.addLabel(newLabel);
826             }
827         }
828         for (var oldLabel : labels) {
829             if (!newLabels.contains(oldLabel)) {
830                 pr.removeLabel(oldLabel);
831             }
832         }
833 
834         // After updating the PR, rethrow any exception to automatically retry on transient errors
835         if (checkException != null) {
836             throw new RuntimeException("Exception during jcheck", checkException);
837         }
838     }
