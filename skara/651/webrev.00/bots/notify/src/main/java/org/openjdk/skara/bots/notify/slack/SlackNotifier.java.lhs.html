 23 package org.openjdk.skara.bots.notify.slack;
 24 
 25 import org.openjdk.skara.bots.notify.*;
 26 import org.openjdk.skara.forge.HostedRepository;
 27 import org.openjdk.skara.forge.PullRequest;
 28 import org.openjdk.skara.json.*;
 29 import org.openjdk.skara.vcs.*;
 30 import org.openjdk.skara.network.*;
 31 
 32 import java.io.IOException;
 33 import java.io.UncheckedIOException;
 34 import java.net.URI;
 35 import java.util.List;
 36 import java.time.format.DateTimeFormatter;
 37 
 38 class SlackNotifier implements RepositoryUpdateConsumer, PullRequestUpdateConsumer {
 39     private final RestRequest prWebhook;
 40     private final RestRequest commitWebhook;
 41     private final String username;
 42 
 43     SlackNotifier(URI prWebhook, URI commitWebhook, String username) {
 44         this.prWebhook = prWebhook != null ? new RestRequest(prWebhook) : null;
 45         this.commitWebhook = commitWebhook != null ? new RestRequest(commitWebhook) : null;
 46         this.username = username;
 47     }
 48 






 49     @Override
 50     public void handleNewPullRequest(PullRequest pr) {
 51         if (prWebhook == null) {
 52             return;
 53         }
 54 
 55         try {
 56             var query = JSON.object();
 57             query.put("text", pr.nonTransformedWebUrl().toString());
 58             if (username != null && !username.isEmpty()) {
 59                 query.put("username", username);
 60             }
 61             prWebhook.post("").body(query).executeUnparsed();
 62         } catch (IOException e) {
 63             throw new UncheckedIOException(e);
 64         }
 65     }
 66 
 67     @Override
 68     public void handleCommits(HostedRepository repository,
 69                               Repository localRepository,
 70                               List<Commit> commits,
 71                               Branch branch) throws NonRetriableException {
 72         if (commitWebhook == null) {
 73             return;
 74         }
 75 
 76         try {
 77             for (var commit : commits) {
 78                 var query = JSON.object();
 79                 if (username != null && !username.isEmpty()) {
 80                     query.put("username", username);
 81                 }
 82                 var title = commit.message().get(0);
 83                 query.put("text", branch.name() + ": " + commit.hash().abbreviate() + ": " + title + "\n" +
 84                                   "Author: " + commit.author().name() + "\n" +
 85                                   "Committer: " + commit.author().name() + "\n" +
 86                                   "Date: " + commit.date().format(DateTimeFormatter.RFC_1123_DATE_TIME) + "\n");
 87 
 88                 var attachment = JSON.object();
 89                 attachment.put("fallback", "Link to commit");
 90                 attachment.put("color", "#cc0e31");
 91                 attachment.put("title", "View on " + repository.forge().name());
 92                 attachment.put("title_link", repository.webUrl(commit.hash()).toString());
 93                 var attachments = JSON.array();
 94                 attachments.add(attachment);
 95                 query.put("attachments", attachments);
 96                 commitWebhook.post("").body(query).executeUnparsed();
 97             }
 98         } catch (IOException e) {
 99             throw new UncheckedIOException(e);
100         }
101     }
102 
103     @Override
104     public String name() {
105         return "slack";
106     }
107 }
