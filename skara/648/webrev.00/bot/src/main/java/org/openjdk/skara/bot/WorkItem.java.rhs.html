23 package org.openjdk.skara.bot;
24 
25 import java.nio.file.Path;
26 import java.util.*;
27 
28 public interface WorkItem {

29     /**
30      * Return true if this item can run concurrently with <code>other</code>, otherwise false.
31      * @param other
32      * @return
33      */
34     boolean concurrentWith(WorkItem other);
35 
36     /**
37      * Execute the appropriate tasks with the provided scratch folder.
38      * @param scratchPath
39      */
40     Collection<WorkItem> run(Path scratchPath);
41 
42     /**
43      * The BotRunner will catch <code>RuntimeException</code>s, implementing this method allows a WorkItem to
44      * perform additional cleanup if necessary (avoiding the need for catching and rethrowing the exception).
45      * @param e
46      */
47     default void handleRuntimeException(RuntimeException e) {}
48 }
