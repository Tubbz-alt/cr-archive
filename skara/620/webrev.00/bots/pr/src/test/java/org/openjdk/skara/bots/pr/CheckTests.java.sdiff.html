 726     @Test
 727     void executableFile(TestInfo testInfo) throws IOException {
 728         try (var credentials = new HostCredentials(testInfo);
 729              var tempFolder = new TemporaryDirectory()) {
 730             var author = credentials.getHostedRepository();
 731             var reviewer = credentials.getHostedRepository();
 732 
 733             var censusBuilder = credentials.getCensusBuilder()
 734                                            .addAuthor(author.forge().currentUser().id())
 735                                            .addReviewer(reviewer.forge().currentUser().id());
 736             var checkBot = PullRequestBot.newBuilder()
 737                                          .repo(author)
 738                                          .censusRepo(censusBuilder.build())
 739                                          .build();
 740 
 741             // Populate the projects repository
 742             var localRepo = CheckableRepository.init(tempFolder.path(), author.repositoryType(),
 743                     Path.of("executable.exe"), Set.of("reviewers", "executable"), "0.1");
 744             var masterHash = localRepo.resolve("master").orElseThrow();
 745             localRepo.push(masterHash, author.url(), "master", true);
 746 
 747             // Make a change with a corresponding PR
 748             Files.writeString(tempFolder.path().resolve("executable.exe"), "Executable file contents", StandardCharsets.UTF_8);
 749             Files.setPosixFilePermissions(tempFolder.path().resolve("executable.exe"), Set.of(PosixFilePermission.OWNER_EXECUTE, PosixFilePermission.OWNER_READ));
 750             localRepo.add(Path.of("executable.exe"));
 751             var editHash = localRepo.commit("Make it executable", "duke", "duke@openjdk.org");
 752             localRepo.push(editHash, author.url(), "edit", true);
 753             var pr = credentials.createPullRequest(author, "master", "edit", "Another PR");
 754             pr.setBody("This should now be ready");
 755 
 756             // Check the status
 757             TestBotRunner.runPeriodicItems(checkBot);
 758 
 759             // Verify that the check failed
 760             var checks = pr.checks(editHash);
 761             assertEquals(1, checks.size());
 762             var check = checks.get("jcheck");
 763             assertEquals(CheckStatus.FAILURE, check.status());
 764             assertTrue(check.summary().orElseThrow().contains("Executable files are not allowed (file: executable.exe)"));
 765 
 766             // Additional errors should be displayed in the body
 767             var updatedPr = author.pullRequest(pr.id());
 768             assertTrue(updatedPr.body().contains("## Error"));
 769             assertTrue(updatedPr.body().contains("Executable files are not allowed (file: executable.exe)"));
 770 
 771             // The PR should not yet be ready for review
 772             assertFalse(pr.labels().contains("rfr"));
 773             assertFalse(pr.labels().contains("ready"));
 774 
 775             // Drop that error
 776             Files.setPosixFilePermissions(tempFolder.path().resolve("executable.exe"), Set.of(PosixFilePermission.OWNER_READ));
 777             localRepo.add(Path.of("executable.exe"));
 778             var updatedHash = localRepo.commit("Make it unexecutable", "duke", "duke@openjdk.org");
 779             localRepo.push(updatedHash, author.url(), "edit");
 780             TestBotRunner.runPeriodicItems(checkBot);
 781 
 782             // The PR should now be ready for review
 783             assertTrue(pr.labels().contains("rfr"));
 784             assertFalse(pr.labels().contains("ready"));
 785 
 786             // The additional errors should be gone
 787             updatedPr = author.pullRequest(pr.id());
 788             assertFalse(updatedPr.body().contains("## Error"));
 789             assertFalse(updatedPr.body().contains("Executable files are not allowed"));
 790         }
 791     }
