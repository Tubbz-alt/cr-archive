 78     static void execute(CheckWorkItem workItem, PullRequest pr, Repository localRepo, List<Comment> comments,
 79                         List<Review> allReviews, List<Review> activeReviews, Set<String> labels, CensusInstance censusInstance,
 80                         boolean ignoreStaleReviews) throws IOException {

 85     private boolean isTargetBranchAllowed() {
 86         var matcher = workItem.bot.allowedTargetBranches().matcher(pr.targetRef());
 87         return matcher.matches();
 88     }

 90     private List<String> allowedTargetBranches() {
 91         return pr.repository()
 92                  .branches()
 93                  .stream()
 94                  .map(HostedBranch::name)
 95                  .map(name -> workItem.bot.allowedTargetBranches().matcher(name))
 96                  .filter(Matcher::matches)
 97                  .map(Matcher::group)
 98                  .collect(Collectors.toList());
 99     }

101     // For unknown contributors, check that all commits have the same name and email
102     private boolean checkCommitAuthor(List<CommitMetadata> commits) throws IOException {
103         var author = censusInstance.namespace().get(pr.author().id());
104         if (author != null) {
105             return true;
106         }
107 
108         var names = new HashSet<String>();
109         var emails = new HashSet<String>();
110 
111         for (var commit : commits) {
112             names.add(commit.author().name());
113             emails.add(commit.author().email());
114         }
115 
116         return ((names.size() == 1) && emails.size() == 1);
117     }

119     // Additional bot-specific checks that are not handled by JCheck
120     private List<String> botSpecificChecks(Hash finalHash) throws IOException {
121         var ret = new ArrayList<String>();
122 
123         if (bodyWithoutStatus().isBlank()) {
124             var error = "The pull request body must not be empty.";
125             ret.add(error);
126         }
127 
128         if (!isTargetBranchAllowed()) {
129             var error = "The branch `" + pr.targetRef() + "` is not allowed as target branch. The allowed target branches are:\n" +
130                     allowedTargetBranches().stream()
131                     .map(name -> "   - " + name)
132                     .collect(Collectors.joining("\n"));
133             ret.add(error);
134         }
135 
136         var headHash = pr.headHash();
137         var originalCommits = localRepo.commitMetadata(baseHash, headHash);
138 
139         if (!checkCommitAuthor(originalCommits)) {
140             var error = "For contributors who are not existing OpenJDK Authors, commit attribution will be taken from " +
141                     "the commits in the PR. However, the commits in this PR have inconsistent user names and/or " +
142                     "email addresses. Please amend the commits.";
143             ret.add(error);
144         }
145 
146         for (var blocker : workItem.bot.blockingCheckLabels().entrySet()) {
147             if (labels.contains(blocker.getKey())) {
148                 ret.add(blocker.getValue());
149             }
150         }
151 
152         return ret;
153     }

155     private void updateCheckBuilder(CheckBuilder checkBuilder, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
156         if (visitor.isReadyForReview() && additionalErrors.isEmpty()) {
157             checkBuilder.complete(true);
158         } else {
159             checkBuilder.title("Required");
160             var summary = Stream.concat(visitor.messages().stream(), additionalErrors.stream())
161                                 .sorted()
162                                 .map(m -> "- " + m)
163                                 .collect(Collectors.joining("\n"));
164             checkBuilder.summary(summary);
165             for (var annotation : visitor.getAnnotations()) {
166                 checkBuilder.annotation(annotation);
167             }
168             checkBuilder.complete(false);
169         }
170     }

172     private void updateReadyForReview(PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
173         // Additional errors are not allowed
174         if (!additionalErrors.isEmpty()) {
175             newLabels.remove("rfr");
176             return;
177         }
178 
179         // Draft requests are not for review
180         if (pr.isDraft()) {
181             newLabels.remove("rfr");
182             return;
183         }
184 
185         // Check if the visitor found any issues that should be resolved before reviewing
186         if (visitor.isReadyForReview()) {
187             newLabels.add("rfr");
188         } else {
189             newLabels.remove("rfr");
190         }
191     }

193     private String getRole(String username) {
194         var project = censusInstance.project();
195         var version = censusInstance.census().version().format();
196         if (project.isReviewer(username, version)) {
197             return "**Reviewer**";
198         } else if (project.isCommitter(username, version)) {
199             return "Committer";
200         } else if (project.isAuthor(username, version)) {
201             return "Author";
202         } else {
203             return "no project role";
204         }
205     }

207     private String formatReviewer(HostUser reviewer) {
208         var namespace = censusInstance.namespace();
209         var contributor = namespace.get(reviewer.id());
210         if (contributor == null) {
211             return reviewer.userName() + " (no known " + namespace.name() + " user name / role)";
212         } else {
213             var userNameLink = "[" + contributor.username() + "](@" + reviewer.userName() + ")";
214             return contributor.fullName().orElse(contributor.username()) + " (" + userNameLink + " - " +
215                     getRole(contributor.username()) + ")";
216         }
217     }

232     private Optional<String> getReviewersList(List<Review> reviews) {
233         var reviewers = reviews.stream()
234                                .filter(review -> review.verdict() == Review.Verdict.APPROVED)
235                                .map(review -> {
236                                    var entry = " * " + formatReviewer(review.reviewer());
237                                    if (!review.hash().equals(pr.headHash())) {
238                                        if (ignoreStaleReviews) {
239                                            entry += " 🔄 Re-review required (review applies to " + review.hash() + ")";
240                                        } else {
241                                            entry += " ⚠️ Review applies to " + review.hash();
242                                        }
243                                    }
244                                    return entry;
245                                })
246                                .collect(Collectors.joining("\n"));
247         if (reviewers.length() > 0) {
248             return Optional.of(reviewers);
249         } else {
250             return Optional.empty();
251         }
252     }

271     private String getStatusMessage(List<Comment> comments, List<Review> reviews, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
272         var progressBody = new StringBuilder();
273         progressBody.append("---------\n");
274         progressBody.append("### Progress\n");
275         progressBody.append(getChecksList(visitor));
276 
277         var allAdditionalErrors = Stream.concat(visitor.hiddenMessages().stream(), additionalErrors.stream())
278                                         .sorted()
279                                         .collect(Collectors.toList());
280         if (!allAdditionalErrors.isEmpty()) {
281             progressBody.append("\n\n### Error");
282             if (allAdditionalErrors.size() > 1) {
283                 progressBody.append("s");
284             }
285             progressBody.append("\n");
286             progressBody.append(getAdditionalErrorsList(allAdditionalErrors));
287         }
288 
289         var issue = Issue.fromString(pr.title());
290         var issueProject = workItem.bot.issueProject();
291         if (issueProject != null && issue.isPresent()) {
292             var allIssues = new ArrayList<Issue>();
293             allIssues.add(issue.get());
294             allIssues.addAll(SolvesTracker.currentSolved(pr.repository().forge().currentUser(), comments));
295             progressBody.append("\n\n### Issue");
296             if (allIssues.size() > 1) {
297                 progressBody.append("s");
298             }
299             progressBody.append("\n");
300             for (var currentIssue : allIssues) {
301                 progressBody.append(" * ");
302                 try {
303                     var iss = issueProject.issue(currentIssue.id());
304                     if (iss.isPresent()) {
305                         progressBody.append("[");
306                         progressBody.append(iss.get().id());
307                         progressBody.append("](");
308                         progressBody.append(iss.get().webUrl());
309                         progressBody.append("): ");
310                         progressBody.append(iss.get().title());
311                         progressBody.append("\n");
312                     } else {
313                         progressBody.append("⚠️ Failed to retrieve information on issue `");
314                         progressBody.append(currentIssue.id());
315                         progressBody.append("`.\n");
316                     }
317                 } catch (RuntimeException e) {
318                     progressBody.append("⚠️ Temporary failure when trying to retrieve information on issue `");
319                     progressBody.append(currentIssue.id());
320                     progressBody.append("`.\n");
321                 }
322             }
323         }
324 
325         getReviewersList(reviews).ifPresent(reviewers -> {
326             progressBody.append("\n\n### Reviewers\n");
327             progressBody.append(reviewers);
328         });
329 
330         getContributorsList(comments).ifPresent(contributors -> {
331             progressBody.append("\n\n### Contributors\n");
332             progressBody.append(contributors);
333         });
334 
335         progressBody.append("\n\n### Download\n");
336         progressBody.append(checkoutCommands());
337 
338         return progressBody.toString();
339     }

341     private String checkoutCommands() {
342         var repoUrl = pr.repository().webUrl();
343         return
344            "`$ git fetch " + repoUrl + " " + pr.fetchRef() + ":pull/" + pr.id() + "`\n" +
345            "`$ git checkout pull/" + pr.id() + "`\n";
346     }

348     private String bodyWithoutStatus() {
349         var description = pr.body();
350         var markerIndex = description.lastIndexOf(progressMarker);
351         return (markerIndex < 0 ?
352                 description :
353                 description.substring(0, markerIndex)).trim();
354     }

356     private String updateStatusMessage(String message) {
357         var description = pr.body();
358         var markerIndex = description.lastIndexOf(progressMarker);
359 
360         if (markerIndex >= 0 && description.substring(markerIndex).equals(message)) {
361             log.info("Progress already up to date");
362             return description;
363         }
364         var newBody = bodyWithoutStatus() + "\n" + progressMarker + "\n" + message;
365 
366         // TODO? Retrieve the body again here to lower the chance of concurrent updates
367         pr.setBody(newBody);
368         return newBody;
369     }

371     private String verdictToString(Review.Verdict verdict) {
372         switch (verdict) {
373             case APPROVED:
374                 return "changes are approved";
375             case DISAPPROVED:
376                 return "more changes needed";
377             case NONE:
378                 return "comment added";
379             default:
380                 throw new RuntimeException("Unknown verdict: " + verdict);
381         }
382     }

384     private void updateReviewedMessages(List<Comment> comments, List<Review> reviews) {
385         var reviewTracker = new ReviewTracker(comments, reviews);

396     private Optional<Comment> findComment(List<Comment> comments, String marker) {
397         var self = pr.repository().forge().currentUser();
398         return comments.stream()
399                        .filter(comment -> comment.author().equals(self))
400                        .filter(comment -> comment.body().contains(marker))
401                        .findAny();

404     private String getMergeReadyComment(String commitMessage, List<Review> reviews) {
405         var message = new StringBuilder();
406         message.append("@");
407         message.append(pr.author().userName());
408         message.append(" This change now passes all automated pre-integration checks");
409 
410         try {
411             var hasContributingFile =
412                 !localRepo.files(pr.targetHash(), Path.of("CONTRIBUTING.md")).isEmpty();
413             if (hasContributingFile) {
414                 message.append(". When the change also fulfills all ");
415                 message.append("[project specific requirements](https://github.com/");
416                 message.append(pr.repository().name());
417                 message.append("/blob/");
418                 message.append(pr.targetRef());
419                 message.append("/CONTRIBUTING.md)");
420             }
421         } catch (IOException e) {
422             throw new UncheckedIOException(e);
423         }
424 
425         message.append(", type `/integrate` in a new comment to proceed. After integration, ");
426         message.append("the commit message will be:\n");
427         message.append("```\n");
428         message.append(commitMessage);
429         message.append("\n```\n");
430 
431         message.append("- If you would like to add a summary, use the `/summary` command.\n");
432         message.append("- To credit additional contributors, use the `/contributor` command.\n");
433         message.append("- To add additional solved issues, use the `/issue` command.\n");
434 
435         var divergingCommits = checkablePullRequest.divergingCommits();
436         if (divergingCommits.size() > 0) {
437             message.append("\n");
438             message.append("Since the source branch of this PR was last updated there ");
439             if (divergingCommits.size() == 1) {
440                 message.append("has been 1 commit ");
441             } else {
442                 message.append("have been ");
443                 message.append(divergingCommits.size());
444                 message.append(" commits ");
445             }
446             message.append("pushed to the `");
447             message.append(pr.targetRef());
448             message.append("` branch:\n\n");
449             divergingCommits.stream()
450                             .limit(10)
451                             .forEach(c -> message.append(" * ").append(c.hash().hex()).append(": ").append(c.message().get(0)).append("\n"));
452             if (divergingCommits.size() > 10) {
453                 message.append(" * ... and ").append(divergingCommits.size() - 10).append(" more: ")
454                        .append(pr.repository().webUrl(baseHash.hex(), pr.targetRef())).append("\n");
455             }
456 
457             message.append("\n");
458             message.append("As there are no conflicts, your changes will automatically be rebased on top of ");
459             message.append("these commits when integrating. If you prefer to avoid automatic rebasing, please merge `");
460             message.append(pr.targetRef());
461             message.append("` into your branch, and then specify the current head hash when integrating, like this: ");
462             message.append("`/integrate ");
463             message.append(pr.targetHash());
464             message.append("`.\n");
465         } else {
466             message.append("\n");
467             message.append("There are currently no new commits on the `");
468             message.append(pr.targetRef());
469             message.append("` branch since the last update of the source branch of this PR. If another commit should be pushed before ");
470             message.append("you perform the `/integrate` command, your PR will be automatically rebased. If you would like to avoid ");
471             message.append("potential automatic rebasing, specify the current head hash when integrating, like this: ");
472             message.append("`/integrate ");
473             message.append(pr.targetHash());
474             message.append("`.\n");
475         }
476 
477         if (!ProjectPermissions.mayCommit(censusInstance, pr.author())) {
478             message.append("\n");
479             var contributor = censusInstance.namespace().get(pr.author().id());
480             if (contributor == null) {
481                 message.append("As you are not a known OpenJDK [Author](https://openjdk.java.net/bylaws#author), ");
482             } else {
483                 message.append("As you do not have Committer status in this project, ");
484             }
485 
486             message.append("an existing [Committer](https://openjdk.java.net/bylaws#committer) must agree to ");
487             message.append("[sponsor](https://openjdk.java.net/sponsor/) your change. ");
488             var candidates = reviews.stream()
489                                     .filter(review -> ProjectPermissions.mayCommit(censusInstance, review.reviewer()))
490                                     .map(review -> "@" + review.reviewer().userName())
491                                     .collect(Collectors.joining(", "));
492             if (candidates.length() > 0) {
493                 message.append("Possible candidates are the reviewers of this PR (");
494                 message.append(candidates);
495                 message.append(") but any other Committer may sponsor as well. ");
496             }
497             message.append("\n\n");
498             message.append("➡️ To flag this PR as ready for integration with the above commit message, type ");
499             message.append("`/integrate` in a new comment. (Afterwards, your sponsor types ");
500             message.append("`/sponsor` in a new comment to perform the integration).\n");
501         } else {
502             message.append("\n");
503             message.append("➡️ To integrate this PR with the above commit message to the `" + pr.targetRef() + "` branch, type ");
504             message.append("`/integrate` in a new comment.\n");
505         }
506         message.append(mergeReadyMarker);
507         return message.toString();
508     }

519     private void updateMergeReadyComment(boolean isReady, String commitMessage, List<Comment> comments, List<Review> reviews, boolean rebasePossible) {
520         var existing = findComment(comments, mergeReadyMarker);
521         if (isReady && rebasePossible) {
522             var message = getMergeReadyComment(commitMessage, reviews);
523             if (existing.isEmpty()) {
524                 pr.addComment(message);
525             } else {
526                 pr.updateComment(existing.get().id(), message);
527             }
528         } else {
529             existing.ifPresent(comment -> pr.updateComment(comment.id(), getMergeNoLongerReadyComment()));
530         }
531     }

533     private void addSourceBranchWarningComment(List<Comment> comments) throws IOException {
534         var existing = findComment(comments, sourceBranchWarningMarker);
535         if (existing.isPresent()) {
536             // Only add the comment once per PR
537             return;
538         }
539         var branch = pr.sourceRef();
540         var message = ":warning: @" + pr.author().userName() + " " +
541             "a branch with the same name as the source branch for this pull request (`" + branch + "`) " +
542             "is present in the [target repository](" + pr.repository().nonTransformedWebUrl() + "). " +
543             "If you eventually integrate this pull request then the branch `" + branch + "` " +
544             "in your [personal fork](" + pr.sourceRepository().nonTransformedWebUrl() + ") will diverge once you sync " +
545             "your personal fork with the upstream repository.\n" +
546             "\n" +
547             "To avoid this situation, create a new branch for your changes and reset the `" + branch + "` branch. " +
548             "You can do this by running the following commands in a local repository for your personal fork. " +
549             "_Note_: you do *not* have to name the new branch `NEW-BRANCH-NAME`." +
550             "\n" +
551             "```" +
552             "$ git checkout " + branch + "\n" +
553             "$ git checkout -b NEW-BRANCH-NAME\n" +
554             "$ git branch -f " + branch + " " + baseHash.hex() + "\n" +
555             "$ git push -f origin " + branch + "\n" +
556             "```\n" +
557             "\n" +
558             "Then proceed to create a new pull request with `NEW-BRANCH-NAME` as the source branch and " +
559             "close this one.\n" +
560             sourceBranchWarningMarker;
561         pr.addComment(message);
562     }

564     private void addOutdatedComment(List<Comment> comments) {
565         var existing = findComment(comments, outdatedHelpMarker);
566         if (existing.isPresent()) {
567             // Only add the comment once per PR
568             return;
569         }
570         var message = "@" + pr.author().userName() + " this pull request can not be integrated into " +
571                 "`" + pr.targetRef() + "` due to one or more merge conflicts. To resolve these merge conflicts " +
572                 "and update this pull request you can run the following commands in the local repository for your personal fork:\n" +
573                 "```bash\n" +
574                 "git checkout " + pr.sourceRef() + "\n" +
575                 "git fetch " + pr.repository().webUrl() + " " + pr.targetRef() + "\n" +
576                 "git merge FETCH_HEAD\n" +
577                 "# resolve conflicts and follow the instructions given by git merge\n" +
578                 "git commit -m \"Merge " + pr.targetRef() + "\"\n" +
579                 "git push\n" +
580                 "```\n" +
581                 outdatedHelpMarker;
582         pr.addComment(message);
583     }

585     private void checkStatus() {
586         var checkBuilder = CheckBuilder.create("jcheck", pr.headHash());
587         var censusDomain = censusInstance.configuration().census().domain();
588         Exception checkException = null;
589 
590         try {
591             // Post check in-progress
592             log.info("Starting to run jcheck on PR head");
593             pr.createCheck(checkBuilder.build());
594 
595             var ignored = new PrintWriter(new StringWriter());
596             var rebasePossible = true;
597             var commitHash = pr.headHash();
598             var mergedHash = checkablePullRequest.mergeTarget(ignored);
599             if (mergedHash.isPresent()) {
600                 commitHash = mergedHash.get();
601             } else {
602                 rebasePossible = false;
603             }
604 
605             List<String> additionalErrors = List.of();
606             Hash localHash;
607             try {
608                 localHash = checkablePullRequest.commit(commitHash, censusInstance.namespace(), censusDomain, null);
609             } catch (CommitFailure e) {
610                 additionalErrors = List.of(e.getMessage());
611                 localHash = baseHash;
612             }
613             PullRequestCheckIssueVisitor visitor = checkablePullRequest.createVisitor(localHash, censusInstance);
614             if (!localHash.equals(baseHash)) {
615                 // Determine current status
616                 var additionalConfiguration = AdditionalConfiguration.get(localRepo, localHash, pr.repository().forge().currentUser(), comments);
617                 checkablePullRequest.executeChecks(localHash, censusInstance, visitor, additionalConfiguration);
618                 additionalErrors = botSpecificChecks(localHash);
619             } else {
620                 if (additionalErrors.isEmpty()) {
621                     additionalErrors = List.of("This PR contains no changes");
622                 }
623             }
624             updateCheckBuilder(checkBuilder, visitor, additionalErrors);
625             updateReadyForReview(visitor, additionalErrors);
626 
627             // Calculate and update the status message if needed
628             var statusMessage = getStatusMessage(comments, activeReviews, visitor, additionalErrors);
629             var updatedBody = updateStatusMessage(statusMessage);
630 
631             // Post / update approval messages (only needed if the review itself can't contain a body)
632             if (!pr.repository().forge().supportsReviewBody()) {
633                 updateReviewedMessages(comments, allReviews);
634             }
635 
636             var commit = localRepo.lookup(localHash).orElseThrow();
637             var commitMessage = String.join("\n", commit.message());
638             var readyForIntegration = visitor.messages().isEmpty() && additionalErrors.isEmpty();
639             updateMergeReadyComment(readyForIntegration, commitMessage, comments, activeReviews, rebasePossible);
640             if (readyForIntegration && rebasePossible) {
641                 newLabels.add("ready");
642             } else {
643                 newLabels.remove("ready");
644             }
645             if (!rebasePossible) {
646                 if (!labels.contains("failed-auto-merge")) {
647                     addOutdatedComment(comments);
648                 }
649                 newLabels.add("merge-conflict");
650             } else {
651                 newLabels.remove("merge-conflict");
652             }
653 
654             var branchNames = pr.repository().branches().stream().map(HostedBranch::name).collect(Collectors.toSet());
655             if (!pr.repository().url().equals(pr.sourceRepository().url()) && branchNames.contains(pr.sourceRef())) {
656                 addSourceBranchWarningComment(comments);
657             }
658 
659             // Ensure that the ready for sponsor label is up to date
660             newLabels.remove("sponsor");
661             var readyHash = ReadyForSponsorTracker.latestReadyForSponsor(pr.repository().forge().currentUser(), comments);
662             if (readyHash.isPresent() && readyForIntegration) {
663                 var acceptedHash = readyHash.get();
664                 if (pr.headHash().equals(acceptedHash)) {
665                     newLabels.add("sponsor");
666                 }
667             }
668 
669             // Calculate current metadata to avoid unnecessary future checks
670             var metadata = workItem.getMetadata(pr.title(), updatedBody, pr.comments(), activeReviews, newLabels,
671                                                 censusInstance, pr.targetHash(), pr.isDraft());
672             checkBuilder.metadata(metadata);
673         } catch (Exception e) {
674             log.throwing("CommitChecker", "checkStatus", e);
675             newLabels.remove("ready");
676             checkBuilder.metadata("invalid");
677             checkBuilder.title("Exception occurred during jcheck - the operation will be retried");
678             checkBuilder.summary(e.getMessage());
679             checkBuilder.complete(false);
680             checkException = e;
681         }
682         var check = checkBuilder.build();
683         pr.updateCheck(check);
684 
685         // Synchronize the wanted set of labels
686         for (var newLabel : newLabels) {
687             if (!labels.contains(newLabel)) {
688                 pr.addLabel(newLabel);
689             }
690         }
691         for (var oldLabel : labels) {
692             if (!newLabels.contains(oldLabel)) {
693                 pr.removeLabel(oldLabel);
694             }
695         }
696 
697         // After updating the PR, rethrow any exception to automatically retry on transient errors
698         if (checkException != null) {
699             throw new RuntimeException("Exception during jcheck", checkException);
700         }
701     }
702 }
