23 package org.openjdk.skara.forge;
24 
25 import org.openjdk.skara.host.HostUser;
26 import org.openjdk.skara.issuetracker.Comment;
27 import org.openjdk.skara.vcs.Hash;
28 
29 import java.nio.file.Path;
30 import java.time.ZonedDateTime;
31 import java.util.*;
32 
33 public class CommitComment extends Comment {
34     private final Hash commit;
35     private final Path path;
36     private final int line;
37 
38     public CommitComment(Hash commit, Path path, int line, String id, String body, HostUser author, ZonedDateTime createdAt, ZonedDateTime updatedAt) {
39         super(id, body, author, createdAt, updatedAt);
40 
41         this.commit = commit;
42         this.path = path;
43         this.line = line;
44     }
45 
46     /**
47      * Returns the hash of the commit.
48      */
49     public Hash commit() {
50         return commit;
51     }
52 
53     /**
54      * Returns the relative path of the file.
55      */
56     public Optional<Path> path() {
57         return Optional.ofNullable(path);
58     }
59 
60     /**
61      * Returns the line number in the file.
62      */
63     public Optional<Integer> line() {
64         return line == -1 ? Optional.empty() : Optional.of(line);
65     }
66 
67     @Override
68     public String toString() {
69         return commit.hex() + ": " + body();
70     }
71 
72     @Override
73     public boolean equals(Object o) {
74         if (this == o) {
75             return true;
76         }
77         if (o == null || getClass() != o.getClass()) {
78             return false;
79         }
80         if (!super.equals(o)) {
81             return false;
82         }
83         var other = (CommitComment) o;
84         return Objects.equals(commit, other.commit) &&
85                Objects.equals(path, other.path) &&
86                line == other.line;
87     }
88 
89     @Override
90     public int hashCode() {
91         return Objects.hash(super.hashCode(), commit, path, line);
92     }
93 }
