  49     private java.lang.Process start(List<String> cmd) throws IOException {
  50         log.fine("Executing " + String.join(" ", cmd));
  51         var pb = new ProcessBuilder(cmd);
  52         pb.directory(dir.toFile());
  53         pb.redirectError(ProcessBuilder.Redirect.DISCARD);
  54         return pb.start();
  55     }

  57     private static void stop(java.lang.Process p) throws IOException {
  58         if (p != null && p.isAlive()) {
  59             var stream = p.getInputStream();
  60             var read = 0;
  61             var buf = new byte[128];
  62             while (read != -1) {
  63                 read = stream.read(buf);
  64             }
  65             try {
  66                 p.waitFor();
  67             } catch (InterruptedException e) {
  68                 throw new IOException(e);
  69             }
  70         }
  71     }

  91     private static Execution.Result await(Execution e) throws IOException {
  92         var result = e.await();
  93         if (result.status() != 0) {
  94             throw new IOException("Unexpected exit code\n" + result);
  95         }
  96         return result;
  97     }

  99     private static void await(java.lang.Process p) throws IOException {
 100         try {
 101             var res = p.waitFor();
 102             if (res != 0) {
 103                 throw new IOException("Unexpected exit code: " + res);
 104             }
 105         } catch (InterruptedException e) {
 106             throw new IOException(e);
 107         }
 108     }

 202     @Override
 203     public List<CommitMetadata> commitMetadata(String range, List<Path> paths, boolean reverse) throws IOException {
 204         var args = new ArrayList<String>();
 205         args.addAll(List.of("git", "rev-list",
 206                                    "--format=" + GitCommitMetadata.FORMAT,
 207                                    "--no-abbrev",
 208                                    "--no-color",
 209                                    range));
 210         if (reverse) {
 211             args.add("--reverse");
 212         }
 213         if (paths != null && !paths.isEmpty()) {
 214             args.add("--");
 215             for (var path : paths) {
 216                 args.add(path.toString());
 217             }
 218         }
 219         var p = start(args);
 220         var reader = new UnixStreamReader(p.getInputStream());
 221         var result = new ArrayList<CommitMetadata>();
 222 
 223         var line = reader.readLine();
 224         while (line != null) {
 225             if (!line.startsWith("commit")) {
 226                 throw new IOException("Unexpected line: " + line);
 227             }
 228 
 229             result.add(GitCommitMetadata.read(reader));
 230             line = reader.readLine();
 231         }
 232 
 233         await(p);
 234         return result;
 235     }

 292     private List<Hash> refs() throws IOException {
 293         try (var p = capture("git", "show-ref", "--hash", "--abbrev")) {
 294             var res = p.await();
 295             if (res.status() == -1) {
 296                 if (res.stdout().size() != 0) {
 297                     throw new IOException("Unexpected output\n" + res);
 298                 }
 299                 return new ArrayList<>();
 300             } else {
 301                 return res.stdout().stream()
 302                           .map(Hash::new)
 303                           .collect(Collectors.toList());
 304             }
 305         }
 306     }

 308     @Override
 309     public boolean isEmpty() throws IOException {
 310         int numLooseObjects = -1;
 311         int numPackedObjects = -1;
 312 
 313         try (var p = capture("git", "count-objects", "-v")) {
 314             var res = await(p);
 315             var stdout = res.stdout();
 316 
 317             for (var line : stdout) {
 318                 if (line.startsWith("count: ")) {
 319                     try {
 320                         numLooseObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 321                     } catch (NumberFormatException e) {
 322                         throw new IOException("Unexpected 'count' value\n" + res, e);
 323                     }
 324 
 325                 } else if (line.startsWith("in-pack: ")) {
 326                     try {
 327                         numPackedObjects = Integer.parseUnsignedInt(line.split(" ")[1]);
 328                     } catch (NumberFormatException e) {
 329                         throw new IOException("Unexpected 'in-pack' value\n" + res, e);
 330                     }
 331                 }
 332             }
 333         }
 334 
 335         return numLooseObjects == 0 && numPackedObjects == 0 && refs().size() == 0;
 336     }

 399     @Override
 400     public Hash fetch(URI uri, String refspec, boolean includeTags) throws IOException {
 401         var cmd = new ArrayList<String>();
 402         cmd.addAll(List.of("git", "fetch", "--recurse-submodules=on-demand"));
 403         if (includeTags) {
 404             cmd.add("--tags");
 405         } else {
 406             cmd.add("--no-tags");
 407         }
 408         cmd.add(uri.toString());
 409         cmd.add(refspec);
 410         try (var p = capture(cmd)) {
 411             await(p);
 412             return resolve("FETCH_HEAD").get();
 413         }
 414     }

 424         } else {

 427         cmd.add("--all");

 429             await(p);
 430         }

 530     @Override
 531     public Path root() throws IOException {
 532         if (cachedRoot != null) {
 533             return cachedRoot;
 534         }
 535 
 536         try (var p = capture("git", "rev-parse", "--show-toplevel")) {
 537             var res = p.await();
 538             if (res.status() != 0 || res.stdout().size() != 1) {
 539                 // Perhaps this is a bare repository
 540                 try (var p2 = capture("git", "rev-parse", "--git-dir")) {
 541                     var res2 = await(p2);
 542                     if (res2.stdout().size() != 1) {
 543                         throw new IOException("Unexpected output\n" + res2);
 544                     }
 545                     cachedRoot = dir.resolve(Path.of(res2.stdout().get(0)));
 546                     return cachedRoot;
 547                 }
 548             }
 549 
 550             cachedRoot = Path.of(res.stdout().get(0));
 551             return cachedRoot;
 552         }
 553     }

 562     @FunctionalInterface
 563     private static interface Operation {
 564         void execute(List<Path> args) throws IOException;
 565     }

 567     private void batch(Operation op, List<Path> args) throws IOException {
 568         var batchSize = 64;
 569         var start = 0;
 570         while (start < args.size()) {
 571             var end = start + batchSize;
 572             if (end > args.size()) {
 573                 end = args.size();
 574             }
 575             op.execute(args.subList(start, end));
 576             start = end;
 577         }
 578     }

 649                        String committerEmail,
 650                        ZonedDateTime committerDate) throws IOException {

 653                          .environ("GIT_AUTHOR_NAME", authorName)
 654                          .environ("GIT_AUTHOR_EMAIL", authorEmail)
 655                          .environ("GIT_COMMITTER_NAME", committerName)
 656                          .environ("GIT_COMMITTER_EMAIL", committerEmail);
 657         if (authorDate != null) {
 658             cmd = cmd.environ("GIT_AUTHOR_DATE",
 659                               authorDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 660         }
 661         if (committerDate != null) {
 662             cmd = cmd.environ("GIT_COMMITTER_DATE",
 663                               committerDate.format(DateTimeFormatter.ISO_OFFSET_DATE_TIME));
 664         }
 665         try (var p = cmd.execute()) {
 666             await(p);
 667             return head();
 668         }
 669     }

 728     @Override
 729     public Hash mergeBase(Hash first, Hash second) throws IOException {
 730         try (var p = capture("git", "merge-base", first.hex(), second.hex())) {

 824     private List<FileEntry> allFiles(Hash hash, List<Path> paths) throws IOException {
 825         var cmd = new ArrayList<String>();
 826         cmd.addAll(List.of("git", "ls-tree", "-r"));
 827         cmd.add(hash.hex());
 828         cmd.addAll(paths.stream().map(Path::toString).collect(Collectors.toList()));
 829         try (var p = Process.capture(cmd.toArray(new String[0]))
 830                             .workdir(root())
 831                             .execute()) {
 832             var res = await(p);
 833             var entries = new ArrayList<FileEntry>();
 834             for (var line : res.stdout()) {
 835                 var parts = line.split("\t");
 836                 var metadata = parts[0].split(" ");
 837                 var filename = parts[1];
 838 
 839                 var entry = new FileEntry(hash,
 840                                           FileType.fromOctal(metadata[0]),
 841                                           new Hash(metadata[2]),
 842                                           Path.of(filename));
 843                 entries.add(entry);
 844             }
 845             return entries;
 846         }
 847     }

 849     @Override
 850     public List<FileEntry> files(Hash hash, List<Path> paths) throws IOException {
 851         if (paths.isEmpty()) {
 852             return allFiles(hash, paths);
 853         }
 854 
 855         var entries = new ArrayList<FileEntry>();
 856         var batchSize = 64;
 857         var start = 0;
 858         while (start < paths.size()) {
 859             var end = start + batchSize;
 860             if (end > paths.size()) {
 861                 end = paths.size();
 862             }
 863             entries.addAll(allFiles(hash, paths.subList(start, end)));
 864             start = end;
 865         }
 866         return entries;
 867     }

 880     @Override
 881     public Optional<byte[]> show(Path path, Hash hash) throws IOException {
 882         var entries = files(hash, path);
 883         if (entries.size() == 0) {
 884             return Optional.empty();
 885         } else if (entries.size() > 1) {
 886             throw new IOException("Multiple files match path " + path.toString() + " in commit " + hash.hex());
 887         }
 888 
 889         var entry = entries.get(0);
 890         var type = entry.type();
 891         if (type.isVCSLink()) {
 892             var content = "Subproject commit " + entry.hash().hex() + " " + entry.path().toString();
 893             return Optional.of(content.getBytes(StandardCharsets.UTF_8));
 894         } else if (type.isRegular()) {
 895             var tmp = unpackFile(entry.hash().hex());
 896             var content = Files.readAllBytes(tmp);
 897             Files.delete(tmp);
 898             return Optional.of(content);
 899         }
 900 
 901         return Optional.empty();
 902     }

1264     @Override
1265     public List<Submodule> submodules() throws IOException {
1266         var gitModules = root().resolve(".gitmodules");
1267         if (!Files.exists(gitModules)) {
1268             return List.of();
1269         }
1270 
1271         var urls = new HashMap<String, String>();
1272         var paths = new HashMap<String, String>();
1273         try (var p = capture("git", "config", "--file", gitModules.toAbsolutePath().toString(),
1274                                               "--list")) {
1275             for (var line : await(p).stdout()) {
1276                 if (line.startsWith("submodule.")) {
1277                     line = line.substring("submodule.".length());
1278                     var parts = line.split("=");
1279                     var nameAndProperty = parts[0].split("\\.");
1280                     var name = nameAndProperty[0];
1281                     var prop = nameAndProperty[1];
1282                     var value = parts[1];
1283                     if (prop.equals("path")) {
1284                         paths.put(name, value);
1285                     } else if (prop.equals("url")) {
1286                         urls.put(name, value);
1287                     } else {
1288                         throw new IOException("Unexpected submodule property: " + prop);
1289                     }
1290                 }
1291             }
1292         }
1293 
1294         var hashes = new HashMap<String, String>();
1295         try (var p = capture("git", "submodule", "status")) {
1296             for (var line : await(p).stdout()) {
1297                 var parts = line.substring(1).split(" ");
1298                 var hash = parts[0];
1299                 var path = parts[1];
1300                 hashes.put(path, hash);
1301             }
1302         }
1303 
1304         var modules = new ArrayList<Submodule>();
1305         for (var name : paths.keySet()) {
1306             var url = urls.get(name);
1307             var path = paths.get(name);
1308             var hash = hashes.get(path);
1309 
1310             modules.add(new Submodule(new Hash(hash), Path.of(path), url));
1311         }
1312 
1313         return modules;
1314     }
