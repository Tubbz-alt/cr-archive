23 package org.openjdk.skara.vcs;
24 
25 import java.util.Objects;
26 
27 public class Tree {
28     private final Hash hash;
29 
30     public Tree(Hash hash) {
31         this.hash = hash;
32     }
33 
34     public Hash hash() {
35         return hash;
36     }
37 
38     @Override
39     public boolean equals(Object o) {
40         if (this == o) return true;
41         if (o == null || getClass() != o.getClass()) return false;
42         Tree tree = (Tree) o;
43         return hash.equals(tree.hash);
44     }
45 
46     @Override
47     public int hashCode() {
48         return Objects.hash(hash);
49     }
50 
51     @Override
52     public String toString() {
53         return "Tree{" +
54                 "hash=" + hash +
55                 '}';
56     }
57 }
