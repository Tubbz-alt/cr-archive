23 package org.openjdk.skara.bots.notify;
24 
25 import org.openjdk.skara.vcs.Tag;
26 
27 import java.util.Objects;
28 
29 public class UpdatedTag {
30     private final Tag tag;
31     private final String updater;
32     private final boolean shouldRetry;
33 
34     public UpdatedTag(Tag tag, String updater, boolean shouldRetry) {
35         this.tag = tag;
36         this.updater = updater;
37         this.shouldRetry = shouldRetry;
38     }
39 
40     public Tag tag() {
41         return tag;
42     }
43 
44     public String updater() {
45         return updater;
46     }
47 
48     public boolean shouldRetry() {
49         return shouldRetry;
50     }
51 
52     @Override
53     public boolean equals(Object o) {
54         if (this == o) return true;
55         if (o == null || getClass() != o.getClass()) return false;
56         UpdatedTag that = (UpdatedTag) o;
57         return shouldRetry == that.shouldRetry &&
58                 tag.equals(that.tag) &&
59                 updater.equals(that.updater);
60     }
61 
62     @Override
63     public int hashCode() {
64         return Objects.hash(tag, updater, shouldRetry);
65     }
66 }
