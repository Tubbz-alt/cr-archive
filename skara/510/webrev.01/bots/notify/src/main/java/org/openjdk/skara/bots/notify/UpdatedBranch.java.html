23 package org.openjdk.skara.bots.notify;
24 
25 import org.openjdk.skara.vcs.*;
26 
27 import java.util.Objects;
28 
29 class UpdatedBranch {
30     private final Branch branch;
31     private final String updater;
32     private final Hash hash;
33 
34     UpdatedBranch(Branch branch, String updater, Hash hash) {
35         this.branch = branch;
36         this.updater = updater;
37         this.hash = hash;
38     }
39 
40     public Branch branch() {
41         return branch;
42     }
43 
44     public String updater() {
45         return updater;
46     }
47 
48     public Hash hash() {
49         return hash;
50     }
51 
52     @Override
53     public boolean equals(Object o) {
54         if (this == o) {
55             return true;
56         }
57         if (o == null || getClass() != o.getClass()) {
58             return false;
59         }
60         UpdatedBranch that = (UpdatedBranch) o;
61         return branch.equals(that.branch) && updater.equals(that.updater) && hash.equals(that.hash);
62     }
63 
64     @Override
65     public int hashCode() {
66         return Objects.hash(branch, updater, hash);
67     }
68 }
