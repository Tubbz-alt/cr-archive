 23 package org.openjdk.skara.cli;
 24 
 25 import org.openjdk.skara.args.Main;
 26 import org.openjdk.skara.vcs.Repository;
 27 import org.openjdk.skara.vcs.openjdk.CommitMessageParsers;
 28 import org.openjdk.skara.version.Version;
 29 
 30 import java.io.IOException;
 31 import java.nio.file.Path;
 32 import java.nio.file.Files;
 33 import java.nio.charset.StandardCharsets;
 34 import java.util.*;
 35 
 36 public class GitSkara {
 37     private static final Map<String, Main> commands = new TreeMap<>();
 38     private static final Set<String> mercurialCommands = Set.of("webrev", "defpath", "jcheck");
 39 
 40     private static void usage(String[] args) {
 41         var isMercurial = args.length > 0 && args[0].equals("--mercurial");
 42         var skaraCommands = Set.of("help", "version", "update");
 43 
 44         var names = new ArrayList<String>();
 45         if (isMercurial) {
 46             names.addAll(mercurialCommands);
 47             names.addAll(skaraCommands);
 48         } else {
 49             names.addAll(commands.keySet());
 50         }
 51 
 52         var vcs = isMercurial ? "hg" : "git";
 53         System.out.println("usage: " + vcs + " skara <" + String.join("|", names) + ">");
 54         System.out.println("");
 55         System.out.println("Additional available " + vcs + " commands:");
 56         for (var name : names) {
 57             if (!skaraCommands.contains(name)) {
 58                 if (isMercurial) {
 59                     if (mercurialCommands.contains(name)) {
 60                         System.out.println("- hg " + name);
 61                     }
 62                 } else {
 63                     System.out.println("- git " + name);
 64                 }
 65             }
 66         }
 67         System.out.println("");
 68         System.out.println("To learn more about a particular command, run:");
 69         System.out.println("");
 70         System.out.println("    " + vcs + " <command> -h");
 71         System.out.println("");
 72         System.out.println("For more information, please see the Skara wiki:");
 73         System.out.println("");
 74         if (isMercurial) {
 75             System.out.println("    https://wiki.openjdk.java.net/display/SKARA/Mercurial");
 76         } else {
 77             System.out.println("    https://wiki.openjdk.java.net/display/skara");
 78         }
 79         System.out.println("");
 80         System.exit(0);
 81     }
 82 
 83     private static void version(String[] args) {
 84         var isMercurial = args.length > 0 && args[0].equals("--mercurial");
 85         var vcs = isMercurial ? "hg" : "git";
 86         System.out.println(vcs + " skara version: " + Version.fromManifest().orElse("unknown"));
 87         System.exit(0);
 88     }
 89 
 90     private static List<String> config(String key, boolean isMercurial) throws IOException, InterruptedException {
 91         var vcs = isMercurial ? "hg" : "git";
 92         var pb = new ProcessBuilder(vcs, "config", key);
 93         pb.redirectOutput(ProcessBuilder.Redirect.PIPE);
 94         pb.redirectError(ProcessBuilder.Redirect.INHERIT);
 95         var p = pb.start();
 96         var value = new String(p.getInputStream().readAllBytes(), StandardCharsets.UTF_8);
 97         p.waitFor();
 98         return Arrays.asList(value.split("\n"));
 99     }
100 
101     private static void update(String[] args) throws IOException, InterruptedException {
102         var isMercurial = args.length > 0 && args[0].equals("--mercurial");
103 
104         String line = null;
105         if (isMercurial) {
106             var lines = config("extensions.skara", true);
107             if (lines.size() == 1) {
108                 line = lines.get(0);
109             } else {
110                 System.err.println("error: could not find skara repository");
111                 System.exit(1);
112             }
113         } else {
114             var lines = config("include.path", false);
115             var entry = lines.stream().filter(l -> l.endsWith("skara.gitconfig")).findFirst();
116             if (entry.isEmpty()) {
117                 System.err.println("error: could not find skara repository");
118                 System.exit(1);
119             }
120             line = entry.get();
121         }
122 
123         var expanded = line.startsWith("~") ?
124             System.getProperty("user.home") + line.substring(1) : line;
125         var path = Path.of(expanded);
126         if (!Files.exists(path)) {
127             System.err.println("error: " + path + " does not exist");
128             System.exit(1);
129         }
130         var parent = path.getParent();
131         var repo = Repository.get(parent);
132         if (repo.isEmpty()) {
133             System.err.println("error: could not find skara repository");
134             System.exit(1);
135         }
136 
137         var head = repo.get().head();
138         System.out.println("Checking for updates ...");
139         repo.get().pull();
140         for (var s : repo.get().submodules()) {
141             repo.get().updateSubmodule(s);
142         }
143         var newHead = repo.get().head();
144 
145         if (!head.equals(newHead)) {
146             System.out.println("Found the following updates:");
147             var commits = repo.get().commitMetadata(head, newHead);
148             for (var commit : commits) {
149                 var message = CommitMessageParsers.v1.parse(commit);
150                 System.out.println("- " + message.title());
151             }
152             System.out.println("Rebuilding ...");
153             var cmd = new ArrayList<String>();
154             if (System.getProperty("os.name").toLowerCase().startsWith("win")) {
155                 cmd.add("gradlew.bat");
156             } else {
157                 cmd.addAll(List.of("sh", "gradlew"));
158             }
159 
160             var pb = new ProcessBuilder(cmd);
161             pb.inheritIO();
162             pb.directory(parent.toFile());
163             var p = pb.start();
164             var res = p.waitFor();
165             if (res != 0) {
166                 System.err.println("error: could not build Skara tooling");
167                 System.exit(1);
168             }
169         } else {
170             System.out.println("No updates found");
171         }
172     }
173 
174     public static void main(String[] args) throws Exception {
175         commands.put("jcheck", GitJCheck::main);
176         commands.put("webrev", GitWebrev::main);
177         commands.put("defpath", GitDefpath::main);
178         commands.put("verify-import", GitVerifyImport::main);
179         commands.put("openjdk-import", GitOpenJDKImport::main);
180         commands.put("fork", GitFork::main);
181         commands.put("pr", GitPr::main);
182         commands.put("token", GitToken::main);
183         commands.put("info", GitInfo::main);
184         commands.put("translate", GitTranslate::main);
185         commands.put("sync", GitSync::main);
186         commands.put("publish", GitPublish::main);
187         commands.put("proxy", GitProxy::main);
188         commands.put("trees", GitTrees::main);
189 
190         commands.put("update", GitSkara::update);
191         commands.put("help", GitSkara::usage);
192         commands.put("version", GitSkara::version);
193 
194         var isEmpty = args.length == 0;
195         var command = isEmpty ? "help" : args[0];
196         var commandArgs = isEmpty ? new String[0] : Arrays.copyOfRange(args, 1, args.length);
197         if (commands.containsKey(command)) {
198             commands.get(command).main(commandArgs);
199         } else {
200             System.err.println("error: unknown command: " + command);
201             usage(args);
202         }
203     }
204 }
