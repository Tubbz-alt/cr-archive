 81     private boolean isTargetBranchAllowed() {
 82         var matcher = workItem.bot.allowedTargetBranches().matcher(pr.targetRef());
 83         return matcher.matches();
 84     }

 86     private List<String> allowedTargetBranches() {
 87         return pr.repository()
 88                  .branches()
 89                  .stream()
 90                  .map(HostedBranch::name)
 91                  .map(name -> workItem.bot.allowedTargetBranches().matcher(name))
 92                  .filter(Matcher::matches)
 93                  .map(Matcher::group)
 94                  .collect(Collectors.toList());
 95     }

 97     // For unknown contributors, check that all commits have the same name and email
 98     private boolean checkCommitAuthor(List<CommitMetadata> commits) throws IOException {
 99         var author = censusInstance.namespace().get(pr.author().id());
100         if (author != null) {
101             return true;
102         }
103 
104         var names = new HashSet<String>();
105         var emails = new HashSet<String>();
106 
107         for (var commit : commits) {
108             names.add(commit.author().name());
109             emails.add(commit.author().email());
110         }
111 
112         return ((names.size() == 1) && emails.size() == 1);
113     }

125     private Optional<MergeSource> mergeSource() {
126         var repoMatcher = mergeSourceFullPattern.matcher(pr.title());
127         if (!repoMatcher.matches()) {
128             var branchMatcher = mergeSourceBranchOnlyPattern.matcher(pr.title());
129             if (!branchMatcher.matches()) {
130                 return Optional.empty();
131             }
132 
133             // Verify that the branch exists
134             var isValidBranch = prInstance.remoteBranches().stream()
135                                           .map(Reference::name)
136                                           .anyMatch(branch -> branch.equals(branchMatcher.group(1)));
137             if (!isValidBranch) {
138                 // Assume the name refers to a sibling repository
139                 var repoName = Path.of(pr.repository().name()).resolveSibling(branchMatcher.group(1)).toString();
140                 return Optional.of(new MergeSource(repoName, "master"));
141             }
142 
143             return Optional.of(new MergeSource(pr.repository().name(), branchMatcher.group(1)));
144         }
145 
146         return Optional.of(new MergeSource(repoMatcher.group(1), repoMatcher.group(2)));
147     }

149     // Additional bot-specific checks that are not handled by JCheck
150     private List<String> botSpecificChecks() throws IOException {
151         var ret = new ArrayList<String>();
152 
153         if (bodyWithoutStatus().isBlank()) {
154             var error = "The pull request body must not be empty.";
155             ret.add(error);
156         }
157 
158         if (!isTargetBranchAllowed()) {
159             var error = "The branch `" + pr.targetRef() + "` is not allowed as target branch. The allowed target branches are:\n" +
160                     allowedTargetBranches().stream()
161                     .map(name -> "   - " + name)
162                     .collect(Collectors.joining("\n"));
163             ret.add(error);
164         }
165 
166         var baseHash = prInstance.baseHash();
167         var headHash = pr.headHash();
168         var commits = prInstance.localRepo().commitMetadata(baseHash, headHash);
169 
170         if (!checkCommitAuthor(commits)) {
171             var error = "For contributors who are not existing OpenJDK Authors, commit attribution will be taken from " +
172                     "the commits in the PR. However, the commits in this PR have inconsistent user names and/or " +
173                     "email addresses. Please amend the commits.";
174             ret.add(error);
175         }
176 
177         if (pr.title().startsWith("Merge")) {
178             if (commits.size() < 2) {
179                 ret.add("A Merge PR must contain at least two commits that are not already present in the target.");
180             } else {
181                 // Find the last merge commit - the very last commit is not eligible (as the merge needs a parent)
182                 int mergeCommitIndex = commits.size();
183                 for (int i = 0; i < commits.size() - 1; ++i) {
184                     if (commits.get(i).isMerge()) {
185                         mergeCommitIndex = i;
186                     }
187                 }
188                 if (mergeCommitIndex >= commits.size() - 1) {
189                     ret.add("A Merge PR must contain a merge commit as well as at least one other commit from the merge source.");
190                 }
191 
192                 var source = mergeSource();
193                 if (source.isPresent()) {
194                     try {
195                         var mergeSourceRepo = pr.repository().forge().repository(source.get().repositoryName).orElseThrow(() ->
196                                 new RuntimeException("Could not find repository " + source.get().repositoryName)
197                         );
198                         try {
199                             var sourceHash = prInstance.localRepo().fetch(mergeSourceRepo.url(), source.get().branchName);
200                             var mergeCommit = commits.get(mergeCommitIndex);
201                             for (int i = 1; i < mergeCommit.parents().size(); ++i) {
202                                 if (!prInstance.localRepo().isAncestor(mergeCommit.parents().get(i), sourceHash)) {
203                                     ret.add("The merge contains commits that are not ancestors of the source.");
204                                     break;
205                                 }
206                             }
207                         } catch (IOException e) {
208                             ret.add("Could not fetch branch `" + source.get().branchName + "` from project `" +
209                                             source.get().repositoryName + "` - check that they are correct.");
210                         }
211                     } catch (RuntimeException e) {
212                         ret.add("Could not find project `" +
213                                         source.get().repositoryName + "` - check that it is correct.");
214                     }
215                 } else {
216                     ret.add("Could not determine the source for this merge. A Merge PR title must be specified on the format: " +
217                             "Merge `project`:`branch` to allow verification of the merge contents.");
218                 }
219             }
220         }
221 
222         for (var blocker : workItem.bot.blockingCheckLabels().entrySet()) {
223             if (labels.contains(blocker.getKey())) {
224                 ret.add(blocker.getValue());
225             }
226         }
227 
228         return ret;
229     }

231     private void updateCheckBuilder(CheckBuilder checkBuilder, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
232         if (visitor.isReadyForReview() && additionalErrors.isEmpty()) {
233             checkBuilder.complete(true);
234         } else {
235             checkBuilder.title("Required");
236             var summary = Stream.concat(visitor.getMessages().stream(), additionalErrors.stream())
237                                 .sorted()
238                                 .map(m -> "- " + m)
239                                 .collect(Collectors.joining("\n"));
240             checkBuilder.summary(summary);
241             for (var annotation : visitor.getAnnotations()) {
242                 checkBuilder.annotation(annotation);
243             }
244             checkBuilder.complete(false);
245         }
246     }

248     private void updateReadyForReview(PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
249         // Additional errors are not allowed
250         if (!additionalErrors.isEmpty()) {
251             newLabels.remove("rfr");
252             return;
253         }
254 
255         // Draft requests are not for review
256         if (pr.isDraft()) {
257             newLabels.remove("rfr");
258             return;
259         }
260 
261         // Check if the visitor found any issues that should be resolved before reviewing
262         if (visitor.isReadyForReview()) {
263             newLabels.add("rfr");
264         } else {
265             newLabels.remove("rfr");
266         }
267     }

269     private String getRole(String username) {
270         var project = censusInstance.project();
271         var version = censusInstance.census().version().format();
272         if (project.isReviewer(username, version)) {
273             return "**Reviewer**";
274         } else if (project.isCommitter(username, version)) {
275             return "Committer";
276         } else if (project.isAuthor(username, version)) {
277             return "Author";
278         } else {
279             return "no project role";
280         }
281     }

283     private String formatReviewer(HostUser reviewer) {
284         var namespace = censusInstance.namespace();
285         var contributor = namespace.get(reviewer.id());
286         if (contributor == null) {
287             return reviewer.userName() + " (no known " + namespace.name() + " user name / role)";
288         } else {
289             var userNameLink = "[" + contributor.username() + "](@" + reviewer.userName() + ")";
290             return contributor.fullName().orElse(contributor.username()) + " (" + userNameLink + " - " +
291                     getRole(contributor.username()) + ")";
292         }
293     }

308     private Optional<String> getReviewersList(List<Review> reviews) {
309         var reviewers = reviews.stream()
310                                .filter(review -> review.verdict() == Review.Verdict.APPROVED)
311                                .map(review -> {
312                                    var entry = " * " + formatReviewer(review.reviewer());
313                                    if (!review.hash().equals(pr.headHash())) {
314                                        if (ignoreStaleReviews) {
315                                            entry += " 🔄 Re-review required (review applies to " + review.hash() + ")";
316                                        } else {
317                                            entry += " ⚠️ Review applies to " + review.hash();
318                                        }
319                                    }
320                                    return entry;
321                                })
322                                .collect(Collectors.joining("\n"));
323         if (reviewers.length() > 0) {
324             return Optional.of(reviewers);
325         } else {
326             return Optional.empty();
327         }
328     }

347     private String getStatusMessage(List<Comment> comments, List<Review> reviews, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
348         var progressBody = new StringBuilder();
349         progressBody.append("---------\n");
350         progressBody.append("### Progress\n");
351         progressBody.append(getChecksList(visitor));
352 
353         if (!additionalErrors.isEmpty()) {
354             progressBody.append("\n\n### Error");
355             if (additionalErrors.size() > 1) {
356                 progressBody.append("s");
357             }
358             progressBody.append("\n");
359             progressBody.append(getAdditionalErrorsList(additionalErrors));
360         }
361 
362         var issue = Issue.fromString(pr.title());
363         var issueProject = workItem.bot.issueProject();
364         if (issueProject != null && issue.isPresent()) {
365             var allIssues = new ArrayList<Issue>();
366             allIssues.add(issue.get());
367             allIssues.addAll(SolvesTracker.currentSolved(pr.repository().forge().currentUser(), comments));
368             progressBody.append("\n\n### Issue");
369             if (allIssues.size() > 1) {
370                 progressBody.append("s");
371             }
372             progressBody.append("\n");
373             for (var currentIssue : allIssues) {
374                 progressBody.append(" * ");
375                 try {
376                     var iss = issueProject.issue(currentIssue.id());
377                     if (iss.isPresent()) {
378                         progressBody.append("[");
379                         progressBody.append(iss.get().id());
380                         progressBody.append("](");
381                         progressBody.append(iss.get().webUrl());
382                         progressBody.append("): ");
383                         progressBody.append(iss.get().title());
384                         progressBody.append("\n");
385                     } else {
386                         progressBody.append("⚠️ Failed to retrieve information on issue `");
387                         progressBody.append(currentIssue.id());
388                         progressBody.append("`.\n");
389                     }
390                 } catch (RuntimeException e) {
391                     progressBody.append("⚠️ Temporary failure when trying to retrieve information on issue `");
392                     progressBody.append(currentIssue.id());
393                     progressBody.append("`.\n");
394                 }
395             }
396         }
397 
398         getReviewersList(reviews).ifPresent(reviewers -> {
399             progressBody.append("\n\n### Reviewers\n");
400             progressBody.append(reviewers);
401         });
402 
403         getContributorsList(comments).ifPresent(contributors -> {
404             progressBody.append("\n\n### Contributors\n");
405             progressBody.append(contributors);
406         });
407 
408         progressBody.append("\n\n### Download\n");
409         progressBody.append(checkoutCommands());
410 
411         return progressBody.toString();
412     }

414     private String checkoutCommands() {
415         var repoUrl = pr.repository().webUrl();
416         return
417            "`$ git fetch " + repoUrl + " " + pr.fetchRef() + ":pull/" + pr.id() + "`\n" +
418            "`$ git checkout pull/" + pr.id() + "`\n";
419     }

421     private String bodyWithoutStatus() {
422         var description = pr.body();
423         var markerIndex = description.lastIndexOf(progressMarker);
424         return (markerIndex < 0 ?
425                 description :
426                 description.substring(0, markerIndex)).trim();
427     }

429     private String updateStatusMessage(String message) {
430         var description = pr.body();
431         var markerIndex = description.lastIndexOf(progressMarker);
432 
433         if (markerIndex >= 0 && description.substring(markerIndex).equals(message)) {
434             log.info("Progress already up to date");
435             return description;
436         }
437         var newBody = bodyWithoutStatus() + "\n" + progressMarker + "\n" + message;
438 
439         // TODO? Retrieve the body again here to lower the chance of concurrent updates
440         pr.setBody(newBody);
441         return newBody;
442     }

444     private String verdictToString(Review.Verdict verdict) {
445         switch (verdict) {
446             case APPROVED:
447                 return "changes are approved";
448             case DISAPPROVED:
449                 return "more changes needed";
450             case NONE:
451                 return "comment added";
452             default:
453                 throw new RuntimeException("Unknown verdict: " + verdict);
454         }
455     }

477     private String getMergeReadyComment(String commitMessage, List<Review> reviews, boolean rebasePossible) {
478         var message = new StringBuilder();
479         message.append("@");
480         message.append(pr.author().userName());
481         message.append(" This change now passes all automated pre-integration checks");
482 
483         try {
484             var hasContributingFile =
485                 !prInstance.localRepo().files(prInstance.targetHash(), Path.of("CONTRIBUTING.md")).isEmpty();
486             if (hasContributingFile) {
487                 message.append(". When the change also fulfills all ");
488                 message.append("[project specific requirements](https://github.com/");
489                 message.append(pr.repository().name());
490                 message.append("/blob/");
491                 message.append(pr.targetRef());
492                 message.append("/CONTRIBUTING.md)");
493             }
494         } catch (IOException e) {
495             throw new UncheckedIOException(e);
496         }
497 
498         message.append(", type `/integrate` in a new comment to proceed. After integration, ");
499         message.append("the commit message will be:\n");
500         message.append("```\n");
501         message.append(commitMessage);
502         message.append("\n```\n");
503 
504         message.append("- If you would like to add a summary, use the `/summary` command.\n");
505         message.append("- To credit additional contributors, use the `/contributor` command.\n");
506         message.append("- To add additional solved issues, use the `/solves` command.\n");
507 
508         var divergingCommits = prInstance.divergingCommits();
509         if (divergingCommits.size() > 0) {
510             message.append("\n");
511             message.append("Since the source branch of this PR was last updated there ");
512             if (divergingCommits.size() == 1) {
513                 message.append("has been 1 commit ");
514             } else {
515                 message.append("have been ");
516                 message.append(divergingCommits.size());
517                 message.append(" commits ");
518             }
519             message.append("pushed to the `");
520             message.append(pr.targetRef());
521             message.append("` branch. ");
522             if (rebasePossible) {
523                 message.append("Since there are no conflicts, your changes will automatically be rebased on top of ");
524                 message.append("these commits when integrating. If you prefer to avoid automatic rebasing, please merge `");
525                 message.append(pr.targetRef());
526                 message.append("` into your branch, and then specify the current head hash when integrating, like this: ");
527                 message.append("`/integrate ");
528                 message.append(prInstance.targetHash());
529                 message.append("`.\n");
530             } else {
531                 message.append("Your changes cannot be rebased automatically without conflicts, so you will need to ");
532                 message.append("merge `");
533                 message.append(pr.targetRef());
534                 message.append("` into your branch before integrating.\n");
535             }
536         }
537 
538         if (!ProjectPermissions.mayCommit(censusInstance, pr.author())) {
539             message.append("\n");
540             var contributor = censusInstance.namespace().get(pr.author().id());
541             if (contributor == null) {
542                 message.append("As you are not a known OpenJDK [Author](https://openjdk.java.net/bylaws#author), ");
543             } else {
544                 message.append("As you do not have Committer status in this project, ");
545             }
546 
547             message.append("an existing [Committer](https://openjdk.java.net/bylaws#committer) must agree to ");
548             message.append("[sponsor](https://openjdk.java.net/sponsor/) your change. ");
549             var candidates = reviews.stream()
550                                     .filter(review -> ProjectPermissions.mayCommit(censusInstance, review.reviewer()))
551                                     .map(review -> "@" + review.reviewer().userName())
552                                     .collect(Collectors.joining(", "));
553             if (candidates.length() > 0) {
554                 message.append("Possible candidates are the reviewers of this PR (");
555                 message.append(candidates);
556                 message.append(") but any other Committer may sponsor as well. ");
557             }
558             if (rebasePossible) {
559                 message.append("\n\n");
560                 message.append("➡️ To flag this PR as ready for integration with the above commit message, type ");
561                 message.append("`/integrate` in a new comment. (Afterwards, your sponsor types ");
562                 message.append("`/sponsor` in a new comment to perform the integration).\n");
563             }
564         } else if (rebasePossible) {
565             message.append("\n");
566             message.append("➡️ To integrate this PR with the above commit message to the `" + pr.targetRef() + "` branch, type ");
567             message.append("`/integrate` in a new comment.\n");
568         }
569         message.append(mergeReadyMarker);
570         return message.toString();
571     }

582     private void updateMergeReadyComment(boolean isReady, String commitMessage, List<Comment> comments, List<Review> reviews, boolean rebasePossible) {
583         var existing = findComment(comments, mergeReadyMarker);
584         if (isReady) {
585             var message = getMergeReadyComment(commitMessage, reviews, rebasePossible);
586             if (existing.isEmpty()) {
587                 pr.addComment(message);
588             } else {
589                 pr.updateComment(existing.get().id(), message);
590             }
591         } else {
592             existing.ifPresent(comment -> pr.updateComment(comment.id(), getMergeNoLongerReadyComment()));
593         }
594     }

596     private void addSourceBranchWarningComment(List<Comment> comments) {
597         var existing = findComment(comments, sourceBranchWarningMarker);
598         if (existing.isPresent()) {
599             // Only add the comment once per PR
600             return;
601         }
602         var branch = pr.sourceRef();
603         var message = ":warning: @" + pr.author().userName() + " " +
604             "a branch with the same name as the source branch for this pull request (`" + branch + "`) " +
605             "is present in the [target repository](" + pr.repository().nonTransformedWebUrl() + "). " +
606             "If you eventually integrate this pull request then the branch `" + branch + "` " +
607             "in your [personal fork](" + pr.sourceRepository().nonTransformedWebUrl() + ") will diverge once you sync " +
608             "your personal fork with the upstream repository.\n" +
609             "\n" +
610             "To avoid this situation, create a new branch for your changes and reset the `" + branch + "` branch. " +
611             "You can do this by running the following commands in a local repository for your personal fork. " +
612             "_Note_: you do *not* have to name the new branch `NEW-BRANCH-NAME`." +
613             "\n" +
614             "```" +
615             "$ git checkout " + branch + "\n" +
616             "$ git checkout -b NEW-BRANCH-NAME\n" +
617             "$ git branch -f " + branch + " " + prInstance.baseHash().hex() + "\n" +
618             "$ git push -f origin " + branch + "\n" +
619             "```\n" +
620             "\n" +
621             "Then proceed to create a new pull request with `NEW-BRANCH-NAME` as the source branch and " +
622             "close this one.\n" +
623             sourceBranchWarningMarker;
624         pr.addComment(message);
625     }

627     private void addOutdatedComment(List<Comment> comments) {
628         var existing = findComment(comments, outdatedHelpMarker);
629         if (existing.isPresent()) {
630             // Only add the comment once per PR
631             return;
632         }
633         var message = "@" + pr.author().userName() + " this pull request can no longer be integrated into " +
634                 "`" + pr.targetRef() + "` due to one or more merge conflicts. To resolve these merge conflicts " +
635                 "and update this pull request you can run the following commands in the local repository for your personal fork:\n" +
636                 "```bash\n" +
637                 "git checkout " + pr.sourceRef() + "\n" +
638                 "git fetch " + pr.repository().webUrl() + " " + pr.targetRef() + "\n" +
639                 "git merge FETCH_HEAD\n" +
640                 "# resolve conflicts and follow the instructions given by git merge\n" +
641                 "git commit -m \"Merge " + pr.targetRef() + "\"\n" +
642                 "git push\n" +
643                 "```\n" +
644                 outdatedHelpMarker;
645         pr.addComment(message);
646     }

648     private void checkStatus() {
649         var checkBuilder = CheckBuilder.create("jcheck", pr.headHash());
650         var censusDomain = censusInstance.configuration().census().domain();
651         Exception checkException = null;
652 
653         try {
654             // Post check in-progress
655             log.info("Starting to run jcheck on PR head");
656             pr.createCheck(checkBuilder.build());
657             List<String> additionalErrors = List.of();
658             Hash localHash;
659             try {
660                 localHash = prInstance.commit(censusInstance.namespace(), censusDomain, null);
661             } catch (CommitFailure e) {
662                 additionalErrors = List.of("It was not possible to create a commit for the changes in this PR: " + e.getMessage());
663                 localHash = prInstance.baseHash();
664             }
665             boolean rebasePossible = true;
666             PullRequestCheckIssueVisitor visitor = prInstance.createVisitor(localHash, censusInstance);
667             if (!localHash.equals(prInstance.baseHash())) {
668                 // Try to rebase
669                 var ignored = new PrintWriter(new StringWriter());
670                 var rebasedHash = prInstance.rebase(localHash, ignored);
671                 if (rebasedHash.isEmpty()) {
672                     rebasePossible = false;
673                 } else {
674                     localHash = rebasedHash.get();
675                 }
676 
677                 // Determine current status
678                 var additionalConfiguration = AdditionalConfiguration.get(prInstance.localRepo(), localHash, pr.repository().forge().currentUser(), comments);
679                 prInstance.executeChecks(localHash, censusInstance, visitor, additionalConfiguration);
680                 additionalErrors = botSpecificChecks();
681             } else {
682                 if (additionalErrors.isEmpty()) {
683                     additionalErrors = List.of("This PR contains no changes");
684                 }
685             }
686             updateCheckBuilder(checkBuilder, visitor, additionalErrors);
687             updateReadyForReview(visitor, additionalErrors);
688 
689             // Calculate and update the status message if needed
690             var statusMessage = getStatusMessage(comments, activeReviews, visitor, additionalErrors);
691             var updatedBody = updateStatusMessage(statusMessage);
692 
693             // Post / update approval messages (only needed if the review itself can't contain a body)
694             if (!pr.repository().forge().supportsReviewBody()) {
695                 updateReviewedMessages(comments, allReviews);
696             }
697 
698             var commit = prInstance.localRepo().lookup(localHash).orElseThrow();
699             var commitMessage = String.join("\n", commit.message());
700             var readyForIntegration = visitor.getMessages().isEmpty() && additionalErrors.isEmpty();
701             updateMergeReadyComment(readyForIntegration, commitMessage, comments, activeReviews, rebasePossible);
702             if (readyForIntegration && rebasePossible) {
703                 newLabels.add("ready");
704             } else {
705                 newLabels.remove("ready");
706             }
707             if (!rebasePossible) {
708                 if (!labels.contains("failed-auto-merge")) {
709                     addOutdatedComment(comments);
710                 }
711                 newLabels.add("merge-conflict");
712             } else {
713                 newLabels.remove("merge-conflict");
714             }
715 
716             var branchNames = pr.repository().branches().stream().map(HostedBranch::name).collect(Collectors.toSet());
717             if (!pr.repository().url().equals(pr.sourceRepository().url()) && branchNames.contains(pr.sourceRef())) {
718                 addSourceBranchWarningComment(comments);
719             }
720 
721             // Ensure that the ready for sponsor label is up to date
722             newLabels.remove("sponsor");
723             var readyHash = ReadyForSponsorTracker.latestReadyForSponsor(pr.repository().forge().currentUser(), comments);
724             if (readyHash.isPresent() && readyForIntegration) {
725                 var acceptedHash = readyHash.get();
726                 if (pr.headHash().equals(acceptedHash)) {
727                     newLabels.add("sponsor");
728                 }
729             }
730 
731             // Calculate current metadata to avoid unnecessary future checks
732             var metadata = workItem.getMetadata(pr.title(), updatedBody, pr.comments(), activeReviews, newLabels,
733                                                 censusInstance, pr.targetHash(), pr.isDraft());
734             checkBuilder.metadata(metadata);
735         } catch (Exception e) {
736             log.throwing("CommitChecker", "checkStatus", e);
737             newLabels.remove("ready");
738             checkBuilder.metadata("invalid");
739             checkBuilder.title("Exception occurred during jcheck - the operation will be retried");
740             checkBuilder.summary(e.getMessage());
741             checkBuilder.complete(false);
742             checkException = e;
743         }
744         var check = checkBuilder.build();
745         pr.updateCheck(check);
746 
747         // Synchronize the wanted set of labels
748         for (var newLabel : newLabels) {
749             if (!labels.contains(newLabel)) {
750                 pr.addLabel(newLabel);
751             }
752         }
753         for (var oldLabel : labels) {
754             if (!newLabels.contains(oldLabel)) {
755                 pr.removeLabel(oldLabel);
756             }
757         }
758 
759         // After updating the PR, rethrow any exception to automatically retry on transient errors
760         if (checkException != null) {
761             throw new RuntimeException("Exception during jcheck", checkException);
762         }
763     }
