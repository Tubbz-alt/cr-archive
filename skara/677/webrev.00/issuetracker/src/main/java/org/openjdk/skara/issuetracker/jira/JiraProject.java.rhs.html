 51     private JSONObject project() {
 52         if (projectMetadataCache == null) {
 53             projectMetadataCache = request.get("project/" + projectName).execute().asObject();
 54         }
 55         return projectMetadataCache;
 56     }

 58     private JSONObject createMeta() {
 59         if (createMetaCache == null) {
 60             createMetaCache = request.get("issue/createmeta")
 61                                      .param("projectKeys", projectName)
 62                                      .param("expand", "projects.issuetypes.fields")
 63                                      .execute()
 64                                      .asObject();
 65         }
 66         return createMetaCache;
 67     }

 69     private Map<String, String> issueTypes() {
 70         var ret = new TreeMap<String, String>(String.CASE_INSENSITIVE_ORDER);
 71         for (var type : project().get("issueTypes").asArray()) {
 72             ret.put(type.get("name").asString(), type.get("id").asString());
 73         }
 74         return ret;
 75     }
 76 
 77     private String issueTypeId(String name) {
 78         var ret = issueTypes().get(name);
 79         if (ret == null) {
 80             var allowedList = issueTypes().keySet().stream()
 81                                           .map(s -> "`" + s + "`")
 82                                           .collect(Collectors.joining(", "));
 83             throw new RuntimeException("Unknown issue type `" + name + "`` Known issue types are " + allowedList + ".");
 84         }
 85         return ret;
 86     }

 88     private Map<String, String> components() {
 89         var ret = new HashMap<String, String>();
 90         for (var type : project().get("components").asArray()) {
 91             ret.put(type.get("name").asString(), type.get("id").asString());
 92         }
 93         return ret;
 94     }

 96     private String componentId(String name) {
 97         var ret = components().get(name);
 98         if (ret == null) {
 99             var allowedList = components().keySet().stream()

101                                           .collect(Collectors.joining(", "));
102             throw new RuntimeException("Unknown component `" + name + "`. Known components are " + allowedList + ".");
103         }
104         return ret;
105     }

156     private static final Set<String> knownProperties = Set.of("issuetype", "fixVersions", "versions", "priority", "components");
157     private static final Set<String> readOnlyProperties = Set.of("resolution");
158 
159     boolean isAllowedProperty(String name, boolean forWrite) {
160         if (knownProperties.contains(name)) {
161             return true;
162         }
163         if (!forWrite && readOnlyProperties.contains(name)) {
164             return true;
165         }
166         return name.startsWith("customfield_");
167     }
168 
169     Optional<JSONValue> decodeProperty(String name, JSONValue value) {
170         if (!isAllowedProperty(name, false)) {
171             return Optional.empty();
172         }
173         if (value.isNull()) {
174             return Optional.empty();
175         }
176 
177         // Transform known fields to a better representation
178         switch (name) {
179             case "fixVersions": // fall-through
180             case "versions": // fall-through
181             case "components":
182                 return Optional.of(new JSONArray(value.stream()
183                                                       .map(obj -> obj.get("name"))
184                                                       .collect(Collectors.toList())));
185             case "customfield_10008": // fall-through
186             case "issuetype":
187                 return Optional.of(JSON.of(value.get("name").asString()));
188             case "priority":
189                 return Optional.of(JSON.of(value.get("id").asString()));
190             default:
191                 return Optional.of(value);
192         }
193     }
194 
195     Optional<JSONValue> encodeProperty(String name, JSONValue value) {
196         if (!isAllowedProperty(name, true)) {
197             return Optional.empty();
198         }
199 
200         switch (name) {
201             case "fixVersions": // fall-through
202             case "versions":
203                 return Optional.of(new JSONArray(value.stream()
204                                                       .map(JSONValue::asString)
205                                                       .map(s -> JSON.object().put("id", versions().get(s)))
206                                                       .collect(Collectors.toList())));
207             case "components":
208                 return Optional.of(new JSONArray(value.stream()
209                                                       .map(JSONValue::asString)
210                                                       .map(s -> JSON.object().put("id", componentId(s)))
211                                                       .collect(Collectors.toList())));
212             case "issuetype":
213                 return Optional.of(JSON.object().put("id", issueTypeId(value.asString())));
214             case "priority":
215                 return Optional.of(JSON.object().put("id", value.asString()));
216             default:
217                 return Optional.of(value);
218         }
219     }
220 
221     JSONValue encodeCustomFields(String name, JSONValue value, Map<String, JSONValue> allProperties) {
222         if (!name.startsWith("customfield_")) {
223             return value;
224         }
225 
226         if (!name.equals("customfield_10008")) {
227             if (value.isObject()) {
228                 if (value.asObject().contains("id")) {
229                     return value.get("id");
230                 } else {
231                     return value;
232                 }
233             } else {
234                 return value;
235             }
236         }
237 
238         var createMeta = createMeta();
239         var fields = createMeta.get("projects").stream()
240                                .filter(p -> p.contains("name"))
241                                .filter(p -> p.get("name").asString().equalsIgnoreCase(projectName))
242                                .findAny().orElseThrow()
243                                .get("issuetypes").stream()
244                                .filter(i -> i.get("id").asString().equals(allProperties.get("issuetype").get("id").asString()))
245                                .findAny().orElseThrow()
246                                .get("fields")
247                                .asObject();
248         System.out.println("Allowed fields " + fields);
249 
250         var field = fields.get(name);
251         var componentIds = allProperties.get("components").stream()
252                                         .map(c -> c.get("id").asString())
253                                         .map(Integer::parseInt)
254                                         .collect(Collectors.toSet());
255         var allowed = field.get("allowedValues").stream()
256                            .filter(c -> componentIds.contains(c.get("id").asInt()))
257                            .flatMap(c -> c.get("subComponents").stream())
258                            .collect(Collectors.toMap(s -> s.get("name").asString(),
259                                                      s -> s.get("id").asInt()));
260         if (!allowed.containsKey(value.asString())) {
261             var allowedList = allowed.keySet().stream()
262                                      .map(s -> "`" + s + "`")
263                                      .collect(Collectors.joining(", "));
264             throw new RuntimeException("Unknown subcomponent `" + value.asString() + "`. Known subcomponents are " +
265                                                allowedList + ".");
266         }
267 
268         return JSON.of(allowed.get(value.asString()));
269     }

281     private boolean isInitialField(String issueType, String name, JSONValue value) {
282         var createMeta = createMeta();
283         var fields = createMeta.get("projects").stream()
284                                .filter(p -> p.contains("name"))
285                                .filter(p -> p.get("name").asString().equalsIgnoreCase(projectName))
286                                .findAny().orElseThrow()
287                                .get("issuetypes").stream()
288                                .filter(i -> i.get("id").asString().equals(issueType))
289                                .findAny().orElseThrow()
290                                .get("fields").fields().stream()
291                                .map(JSONObject.Field::name)
292                                .collect(Collectors.toSet());
293 
294         return fields.contains(name);







295     }

297     @Override
298     public Issue createIssue(String title, List<String> body, Map<String, JSONValue> properties) {
299         var query = JSON.object();
300 
301         // Encode optional properties as fields
302         var finalProperties = new HashMap<String, JSONValue>();
303         for (var property : properties.entrySet()) {
304             var encoded = encodeProperty(property.getKey(), property.getValue());
305             if (encoded.isEmpty()) {
306                 continue;
307             }
308             finalProperties.put(property.getKey(), encoded.get());
309         }
310 
311         // Always override certain fields
312         finalProperties.put("project", JSON.object().put("id", projectId()));
313         finalProperties.put("summary", JSON.of(title));
314         finalProperties.put("description", JSON.of(String.join("\n", body)));
315 
316         // Provide default values for required fields if not present
317         finalProperties.putIfAbsent("components", JSON.array().add(JSON.object().put("id", defaultComponent())));
318         finalProperties.putIfAbsent("issuetype", JSON.object().put("id", defaultIssueType()));
319 
320         // Filter out the fields that can be set at creation time
321         var issueType = finalProperties.get("issuetype").get("id").asString();
322         var fields = JSON.object();
323         finalProperties.entrySet().stream()
324                        .filter(entry -> isInitialField(issueType, entry.getKey(), entry.getValue()))
325                        .forEach(entry -> fields.put(entry.getKey(), encodeCustomFields(entry.getKey(),
326                                                                                        entry.getValue(),
327                                                                                        finalProperties)));


328         query.put("fields", fields);
329         jiraHost.securityLevel().ifPresent(securityLevel -> fields.put("security", JSON.object()
330                                                                                        .put("id", securityLevel)));
331         var data = request.post("issue")
332                           .body(query)
333                           .execute();
334 
335         // Apply fields that have to be set later (if any)
336         var editFields = JSON.object();
337         finalProperties.entrySet().stream()
338                        .filter(entry -> !isInitialField(issueType, entry.getKey(), entry.getValue()))
339                        .forEach(entry -> editFields.put(entry.getKey(), encodeCustomFields(entry.getKey(),
340                                                                                            entry.getValue(),
341                                                                                            finalProperties)));
342 
343         if (editFields.fields().size() > 0) {
344             var id = data.get("key").asString();
345             if (id.indexOf('-') < 0) {
346                 id = projectName.toUpperCase() + "-" + id;
347             }
348             var updateQuery = JSON.object().put("fields", editFields);
349             request.put("issue/" + id)
350                    .body(updateQuery)
351                    .execute();
352 
353         }
354 
355         return issue(data.get("key").asString()).orElseThrow();
356     }

358     @Override
359     public Optional<Issue> issue(String id) {
360         if (id.indexOf('-') < 0) {
361             id = projectName.toUpperCase() + "-" + id;
362         }
363         var issueRequest = request.restrict("issue/" + id);
364         var issue = issueRequest.get("")
365                                 .onError(r -> r.statusCode() < 500 ? Optional.of(JSON.object().put("NOT_FOUND", true)) : Optional.empty())
366                                 .execute();
367         if (!issue.contains("NOT_FOUND")) {
368             return Optional.of(new JiraIssue(this, issueRequest, issue));
369         } else {
370             return Optional.empty();
371         }
372     }

374     @Override
375     public List<Issue> issues() {
376         var ret = new ArrayList<Issue>();
377         var issues = request.post("search")
378                             .body("jql", "project = " + projectName + " AND status in (Open, New)")
379                             .execute();
380         for (var issue : issues.get("issues").asArray()) {
381             ret.add(new JiraIssue(this, request, issue));
382         }
383         return ret;
384     }
