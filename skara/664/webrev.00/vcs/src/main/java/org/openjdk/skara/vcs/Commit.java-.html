23 package org.openjdk.skara.vcs;
24 
25 import java.time.*;
26 import java.time.format.*;
27 import java.util.*;
28 
29 public class Commit {
30     private final CommitMetadata metadata;
31     private final List<Diff> parentDiffs;
32 
33     public Commit(CommitMetadata metadata, List<Diff> parentDiffs) {
34         this.metadata = metadata;
35         this.parentDiffs = parentDiffs;
36     }
37 
38     public Hash hash() {
39         return metadata.hash();
40     }
41 
42     public Author author() {
43         return metadata.author();
44     }
45 
46     public Author committer() {
47         return metadata.committer();
48     }
49 
50     public List<String> message() {
51         return metadata.message();
52     }
53 
54     public List<Hash> parents() {
55         return metadata.parents();
56     }
57 
58     public List<Diff> parentDiffs() {
59         return parentDiffs;
60     }
61 
62     public boolean isInitialCommit() {
63         return metadata.isInitialCommit();
64     }
65 
66     public ZonedDateTime date() {
67         return metadata.date();
68     }
69 
70     public boolean isMerge() {
71         return metadata.isMerge();
72     }
73 
74     public int numParents() {
75         return metadata.numParents();
76     }
77 
78     @Override
79     public String toString() {
80         return metadata.toString();
81     }
82 
83     @Override
84     public int hashCode() {
85         return Objects.hash(metadata, parentDiffs);
86     }
87 
88     @Override
89     public boolean equals(Object o) {
90         if (!(o instanceof Commit)) {
91             return false;
92         }
93 
94         var other = (Commit) o;
95         return Objects.equals(metadata, other.metadata) && Objects.equals(parentDiffs, other.parentDiffs);
96     }
97 }
