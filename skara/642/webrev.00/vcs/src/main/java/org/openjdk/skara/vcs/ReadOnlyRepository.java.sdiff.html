 53         var l = commitMetadata(range(hash));
 54         if (l.size() > 1) {
 55             throw new IllegalStateException("More than one commit for hash: " + hash.hex());
 56         }
 57         if (l.size() == 0) {
 58             return Optional.empty();
 59         }
 60         return Optional.of(l.get(0));
 61     }
