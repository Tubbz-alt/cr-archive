23 package org.openjdk.skara.args;
24 
25 public class Input {
26     private final int position;
27     private final String description;
28     private final int occurrences;
29     private final boolean required;
30 
31     Input(int position, String description, int occurrences, boolean required) {
32         this.position = position;
33         this.description = description;
34         this.occurrences = occurrences;
35         this.required = required;
36     }
37 
38     public static InputDescriber position(int p) {
39         return new InputDescriber(p);
40     }
41 
42     public int getPosition() {
43         return position;
44     }
45 
46     public String getDescription() {
47         return description;
48     }
49 
50     public int getOccurrences() {
51         return occurrences;
52     }
53 
54     public boolean isTrailing() {
55         return occurrences == -1;
56     }
57 
58     public boolean isRequired() {
59         return required;
60     }
61 
62     @Override
63     public String toString() {
64         var builder = new StringBuilder();
65         for (var i = 0; i < occurrences; i++) {

66             if (!isRequired()) {
67                 builder.append("[");
68             }
69             builder.append("<");
70             builder.append(description);
71             builder.append(">");
72             if (!isRequired()) {
73                 builder.append("]");
74             }
75             if (i != (occurrences - 1)) {
76                 builder.append(" ");
77             }
78 
79             if (isTrailing()) {
80                 builder.append("...");
81             }
82         }
83 
84         return builder.toString();
85     }
86 }
