23 package org.openjdk.skara.bots.pr;
24 
25 import org.junit.jupiter.api.Test;
26 
27 import java.nio.file.Path;
28 import java.util.*;
29 import java.util.regex.Pattern;
30 
31 import static org.junit.jupiter.api.Assertions.assertEquals;
32 
33 public class LabelConfigurationTests {
34     @Test
35     void simple() {
36         var config = LabelConfiguration.newBuilder()
37                                        .addMatchers("1", List.of(Pattern.compile("cpp$")))
38                                        .addMatchers("2", List.of(Pattern.compile("hpp$")))
39                                        .build();
40 
41         assertEquals(Set.of("1", "2"), config.allowed());
42 
43         assertEquals(Set.of("1"), config.fromChanges(Set.of(Path.of("a.cpp"))));
44         assertEquals(Set.of("2"), config.fromChanges(Set.of(Path.of("a.hpp"))));
45         assertEquals(Set.of("1", "2"), config.fromChanges(Set.of(Path.of("a.cpp"), Path.of("a.hpp"))));
46     }
47 
48     @Test
49     void group() {
50         var config = LabelConfiguration.newBuilder()
51                                        .addMatchers("1", List.of(Pattern.compile("cpp$")))
52                                        .addMatchers("2", List.of(Pattern.compile("hpp$")))
53                                        .addGroup("both", List.of("1", "2"))
54                                        .build();
55 
56         assertEquals(Set.of("1", "2", "both"), config.allowed());
57 
58         assertEquals(Set.of("1"), config.fromChanges(Set.of(Path.of("a.cpp"))));
59         assertEquals(Set.of("2"), config.fromChanges(Set.of(Path.of("a.hpp"))));
60         assertEquals(Set.of("1", "2", "both"), config.fromChanges(Set.of(Path.of("a.cpp"), Path.of("a.hpp"))));
61     }
62 }
