23 package org.openjdk.skara.vcs;
24 
25 public class Hash {

26     private final String hex;
27 
28     public Hash(String hex) {
29         this.hex = hex;
30     }
31 




32     @Override
33     public int hashCode() {
34         return hex.hashCode();
35     }
36 
37     @Override
38     public boolean equals(Object o) {
39         if (!(o instanceof Hash)) {
40             return false;
41         }
42 
43         if (o == this) {
44             return true;
45         }
46 
47         return hex.equals(((Hash) o).hex());
48     }
49 
50     @Override
51     public String toString() {
52         return hex();
53     }
54 
55     public String hex() {
56         return hex;
57     }
58 
59     public String abbreviate() {
60         return hex().substring(0, 8);
61     }
62 }
