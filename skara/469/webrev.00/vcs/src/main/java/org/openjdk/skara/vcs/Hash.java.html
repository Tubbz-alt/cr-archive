23 package org.openjdk.skara.vcs;
24 
25 public class Hash {
26     private static final Hash ZERO = new Hash("0".repeat(40));
27     private final String hex;
28 
29     public Hash(String hex) {
30         this.hex = hex;
31     }
32 
33     public static Hash zero() {
34         return ZERO;
35     }
36 
37     @Override
38     public int hashCode() {
39         return hex.hashCode();
40     }
41 
42     @Override
43     public boolean equals(Object o) {
44         if (!(o instanceof Hash)) {
45             return false;
46         }
47 
48         if (o == this) {
49             return true;
50         }
51 
52         return hex.equals(((Hash) o).hex());
53     }
54 
55     @Override
56     public String toString() {
57         return hex();
58     }
59 
60     public String hex() {
61         return hex;
62     }
63 
64     public String abbreviate() {
65         return hex().substring(0, 8);
66     }
67 }
