 23 package org.openjdk.skara.forge;
 24 
 25 import org.openjdk.skara.vcs.Hash;
 26 
 27 import org.junit.jupiter.api.Test;
 28 
 29 import java.time.ZonedDateTime;
 30 import java.util.List;
 31 
 32 import static org.junit.jupiter.api.Assertions.assertEquals;
 33 
 34 class CheckBuilderTests {
 35     @Test
 36     void testFrom() {
 37         var hash = new Hash("0".repeat(40));
 38         var name = "test";
 39         var title = "title";
 40         var summary = "summary";
 41         var metadata = "metadata";
 42         var annotation = CheckAnnotationBuilder.create("README", 0, 1, CheckAnnotationLevel.NOTICE, "Message")
 43                                                .build();
 44         var startedAt = ZonedDateTime.now();
 45         var completedAt = ZonedDateTime.now();
 46         var success = true;
 47 
 48         var existing = CheckBuilder.create(name, hash)
 49                                    .title(title)
 50                                    .summary(summary)
 51                                    .metadata(metadata)
 52                                    .annotation(annotation)
 53                                    .startedAt(startedAt)
 54                                    .complete(success, completedAt)
 55                                    .build();
 56         var dup = CheckBuilder.from(existing)
 57                               .build();
 58 
 59         assertEquals(existing.name(), dup.name());
 60         assertEquals(existing.hash(), dup.hash());
 61         assertEquals(existing.status(), dup.status());
 62         assertEquals(existing.startedAt(), dup.startedAt());
 63         assertEquals(existing.completedAt(), dup.completedAt());
 64         assertEquals(existing.title(), dup.title());
 65         assertEquals(existing.summary(), dup.summary());
 66         assertEquals(existing.metadata(), dup.metadata());
 67         assertEquals(existing.annotations(), dup.annotations());
 68 
 69         var newTitle = "new title";
 70         var newSummary = "new summary";
 71         var newMetadata = "new metadata";
 72         var newAnnotation = CheckAnnotationBuilder.create("FILE", 0, 1, CheckAnnotationLevel.NOTICE, "Message")
 73                                                   .build();
 74         var newStartedAt = ZonedDateTime.now();
 75         var newCompletedAt = ZonedDateTime.now();
 76         var newSuccess = false;
 77 
 78         var modified = CheckBuilder.from(existing)
 79                                    .title(newTitle)
 80                                    .summary(newSummary)
 81                                    .metadata(newMetadata)
 82                                    .annotation(newAnnotation)
 83                                    .startedAt(newStartedAt)
 84                                    .complete(newSuccess, newCompletedAt)
 85                                    .build();
 86 
 87         // existing check should not have changed
 88         assertEquals(dup.name(), existing.name());
 89         assertEquals(dup.hash(), existing.hash());
 90         assertEquals(dup.status(), existing.status());
 91         assertEquals(dup.startedAt(), existing.startedAt());
 92         assertEquals(dup.completedAt(), existing.completedAt());
 93         assertEquals(dup.title(), existing.title());
 94         assertEquals(dup.summary(), existing.summary());
 95         assertEquals(dup.metadata(), existing.metadata());
 96         assertEquals(dup.annotations(), existing.annotations());
 97 
 98         // modified should have new values except name and hash and inherit annotations
 99         assertEquals(existing.name(), modified.name());
100         assertEquals(existing.hash(), modified.hash());
101         assertEquals(newStartedAt, modified.startedAt());
102         assertEquals(newCompletedAt, modified.completedAt().get());
103         assertEquals(newTitle, modified.title().get());
104         assertEquals(newSummary, modified.summary().get());
105         assertEquals(newMetadata, modified.metadata().get());
106         assertEquals(List.of(annotation, newAnnotation), modified.annotations());
107     }
108 }
