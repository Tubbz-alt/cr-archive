23 package org.openjdk.skara.jcheck;
24 
25 import org.openjdk.skara.ini.Section;
26 
27 import java.util.List;
28 import java.util.Set;
29 import java.util.stream.Collectors;
30 
31 public class IssuesConfiguration {
32     static final IssuesConfiguration DEFAULT =
33         new IssuesConfiguration("^(([A-Z][A-Z0-9]+-)?[0-9]+): (\\S.*)$", true);
34 
35     private final String pattern;
36     private final boolean required;
37 
38     IssuesConfiguration(String pattern, boolean required) {
39         this.pattern = pattern;
40         this.required = required;
41     }
42 
43     public String pattern() {
44         return pattern;
45     }
46 
47     public boolean required() {
48         return required;
49     }
50 
51     static String name() {
52         return "issues";
53     }
54 
55     static IssuesConfiguration parse(Section s) {
56         if (s == null) {
57             return DEFAULT;
58         }
59 
60         var pattern = s.get("pattern", DEFAULT.pattern());
61         var required = s.get("required", DEFAULT.required());
62         return new IssuesConfiguration(pattern, required);
63     }
64 }
