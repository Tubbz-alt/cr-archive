23 package org.openjdk.skara.jcheck;
24 
25 import org.openjdk.skara.ini.Section;
26 
27 import java.util.List;
28 import java.util.Set;
29 import java.util.stream.Collectors;
30 
31 public class IssuesConfiguration {
32     static final IssuesConfiguration DEFAULT =
33         new IssuesConfiguration("^(([A-Z][A-Z0-9]+-)?[0-9]+): (\\S.*)$");
34 
35     private final String pattern;

36 
37     IssuesConfiguration(String pattern) {
38         this.pattern = pattern;

39     }
40 
41     public String pattern() {
42         return pattern;
43     }
44 




45     static String name() {
46         return "issues";
47     }
48 
49     static IssuesConfiguration parse(Section s) {
50         if (s == null) {
51             return DEFAULT;
52         }
53 
54         var pattern = s.get("pattern", DEFAULT.pattern());
55         return new IssuesConfiguration(pattern);

56     }
57 }
