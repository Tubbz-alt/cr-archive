23 package org.openjdk.skara.bots.notify.issue;
24 
25 import org.openjdk.skara.issuetracker.IssueProject;
26 
27 import java.net.URI;
28 import java.util.Map;
29 
30 class IssueNotifierBuilder {
31     private IssueProject issueProject;
32     private boolean reviewLink = true;
33     private URI reviewIcon = null;
34     private boolean commitLink = true;
35     private URI commitIcon = null;
36     private boolean setFixVersion = false;
37     private Map<String, String> fixVersions = null;
38     private JbsVault vault = null;
39     private String securityLevel = null;
40     private boolean prOnly = true;
41 
42     IssueNotifierBuilder issueProject(IssueProject issueProject) {
43         this.issueProject = issueProject;
44         return this;
45     }
46 
47     IssueNotifierBuilder reviewLink(boolean reviewLink) {
48         this.reviewLink = reviewLink;
49         return this;
50     }
51 
52     IssueNotifierBuilder reviewIcon(URI reviewIcon) {
53         this.reviewIcon = reviewIcon;
54         return this;
55     }
56 
57     IssueNotifierBuilder commitLink(boolean commitLink) {
58         this.commitLink = commitLink;
59         return this;
60     }
61 
62     IssueNotifierBuilder commitIcon(URI commitIcon) {
63         this.commitIcon = commitIcon;
64         return this;
65     }
66 
67     public IssueNotifierBuilder setFixVersion(boolean setFixVersion) {
68         prOnly = false;
69         this.setFixVersion = setFixVersion;
70         return this;
71     }
72 
73     public IssueNotifierBuilder fixVersions(Map<String, String> fixVersions) {
74         this.fixVersions = fixVersions;
75         return this;
76     }
77 
78     public IssueNotifierBuilder vault(JbsVault vault) {
79         this.vault = vault;
80         return this;
81     }
82 
83     public IssueNotifierBuilder securityLevel(String securityLevel) {
84         this.securityLevel = securityLevel;
85         return this;
86     }
87 
88     public IssueNotifierBuilder prOnly(boolean prOnly) {
89         this.prOnly = prOnly;
90         return this;
91     }
92 
93     IssueNotifier build() {
94         var jbsBackport = new JbsBackport(issueProject.webUrl(), vault, securityLevel);
95         return new IssueNotifier(issueProject, reviewLink, reviewIcon, commitLink, commitIcon,
96                                  setFixVersion, fixVersions, jbsBackport, prOnly);
97     }
98 }
