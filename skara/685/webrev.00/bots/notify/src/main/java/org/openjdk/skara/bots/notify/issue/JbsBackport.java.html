23 package org.openjdk.skara.bots.notify.issue;
24 
25 import org.openjdk.skara.issuetracker.*;
26 import org.openjdk.skara.json.JSON;
27 import org.openjdk.skara.network.*;
28 
29 import java.net.URI;
30 import java.util.*;
31 import java.util.stream.Collectors;
32 
33 public class JbsBackport {
34     private final String securityLevel;
35     private final RestRequest backportRequest;
36 
37     private static URI backportRequest(URI uri) {
38         return URIBuilder.base(uri)
39                          .setPath("/rest/jbs/1.0/backport/")
40                          .build();
41     }
42 
43     JbsBackport(URI uri, JbsVault vault, String securityLevel) {
44         this.securityLevel = securityLevel;
45         if (vault != null) {
46             backportRequest = new RestRequest(backportRequest(uri), vault.authId(), () -> Arrays.asList("Cookie", vault.getCookie()));
47         } else {
48             backportRequest = null;
49         }
50     }
51 
52     private Issue createBackportIssue(Issue primary) {
53         var finalProperties = new HashMap<>(primary.properties());
54         finalProperties.put("issuetype", JSON.of("Backport"));
55 
56         var backport = primary.project().createIssue(primary.title(), primary.body().lines().collect(Collectors.toList()), finalProperties);
57 
58         var backportLink = Link.create(backport, "backported by").build();
59         primary.addLink(backportLink);
60         return backport;
61     }
62 
63     public Issue createBackport(Issue primary, String fixVersion, String assignee) {
64         if (backportRequest == null) {
65             if (primary.project().webUrl().toString().contains("openjdk.java.net")) {
66                 throw new RuntimeException("Backports on JBS require vault authentication");
67             } else {
68                 return createBackportIssue(primary);
69             }
70         }
71 
72         var request = backportRequest.post()
73                                      .body("parentIssueKey", primary.id())
74                                      .body("fixVersion", fixVersion);
75         if (assignee != null) {
76             request.body("assignee", assignee);
77         }
78         if (securityLevel != null) {
79             request.body("level", securityLevel);
80         }
81         var response = request.execute();
82         return primary.project().issue(response.get("key").asString()).orElseThrow();
83     }
84 }
