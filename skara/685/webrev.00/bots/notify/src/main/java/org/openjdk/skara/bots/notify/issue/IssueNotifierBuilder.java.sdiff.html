23 package org.openjdk.skara.bots.notify.issue;
24 
25 import org.openjdk.skara.issuetracker.IssueProject;
26 
27 import java.net.URI;
28 import java.util.Map;
29 
30 class IssueNotifierBuilder {
31     private IssueProject issueProject;
32     private boolean reviewLink = true;
33     private URI reviewIcon = null;
34     private boolean commitLink = true;
35     private URI commitIcon = null;
36     private boolean setFixVersion = false;
37     private Map<String, String> fixVersions = null;
38     private JbsVault vault = null;
39     private String securityLevel = null;
40 
41     IssueNotifierBuilder issueProject(IssueProject issueProject) {
42         this.issueProject = issueProject;
43         return this;
44     }
45 
46     IssueNotifierBuilder reviewLink(boolean reviewLink) {
47         this.reviewLink = reviewLink;
48         return this;
49     }
50 
51     IssueNotifierBuilder reviewIcon(URI reviewIcon) {
52         this.reviewIcon = reviewIcon;
53         return this;
54     }
55 
56     IssueNotifierBuilder commitLink(boolean commitLink) {
57         this.commitLink = commitLink;
58         return this;
59     }
60 
61     IssueNotifierBuilder commitIcon(URI commitIcon) {
62         this.commitIcon = commitIcon;
63         return this;
64     }
65 
66     public IssueNotifierBuilder setFixVersion(boolean setFixVersion) {
67         this.setFixVersion = setFixVersion;
68         return this;
69     }
70 
71     public IssueNotifierBuilder fixVersions(Map<String, String> fixVersions) {
72         this.fixVersions = fixVersions;
73         return this;
74     }
75 
76     public IssueNotifierBuilder vault(JbsVault vault) {
77         this.vault = vault;
78         return this;
79     }
80 
81     public IssueNotifierBuilder securityLevel(String securityLevel) {
82         this.securityLevel = securityLevel;
83         return this;
84     }
85 
86     IssueNotifier build() {
87         var jbsBackport = new JbsBackport(issueProject.webUrl(), vault, securityLevel);
88         return new IssueNotifier(issueProject, reviewLink, reviewIcon, commitLink, commitIcon,
89                                  setFixVersion, fixVersions, jbsBackport);
90     }
