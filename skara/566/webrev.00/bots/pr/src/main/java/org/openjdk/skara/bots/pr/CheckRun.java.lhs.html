 81     private boolean isTargetBranchAllowed() {
 82         var matcher = workItem.bot.allowedTargetBranches().matcher(pr.targetRef());
 83         return matcher.matches();

 86     private List<String> allowedTargetBranches() {
 87         return pr.repository()
 88                  .branches()
 89                  .stream()
 90                  .map(HostedBranch::name)
 91                  .map(name -> workItem.bot.allowedTargetBranches().matcher(name))
 92                  .filter(Matcher::matches)
 93                  .map(Matcher::group)
 94                  .collect(Collectors.toList());
 95     }

 97     // For unknown contributors, check that all commits have the same name and email
 98     private boolean checkCommitAuthor(List<CommitMetadata> commits) throws IOException {
 99         var author = censusInstance.namespace().get(pr.author().id());
100         if (author != null) {
101             return true;
102         }
103 
104         var names = new HashSet<String>();
105         var emails = new HashSet<String>();
106 
107         for (var commit : commits) {
108             names.add(commit.author().name());
109             emails.add(commit.author().email());
110         }
111 
112         return ((names.size() == 1) && emails.size() == 1);
113     }

121             this.branchName = branchName;

125     private Optional<MergeSource> mergeSource() {
126         var repoMatcher = mergeSourceFullPattern.matcher(pr.title());
127         if (!repoMatcher.matches()) {
128             var branchMatcher = mergeSourceBranchOnlyPattern.matcher(pr.title());
129             if (!branchMatcher.matches()) {
130                 return Optional.empty();
131             }
132 
133             // Verify that the branch exists
134             var isValidBranch = prInstance.remoteBranches().stream()
135                                           .map(Reference::name)
136                                           .anyMatch(branch -> branch.equals(branchMatcher.group(1)));
137             if (!isValidBranch) {
138                 // Assume the name refers to a sibling repository
139                 var repoName = Path.of(pr.repository().name()).resolveSibling(branchMatcher.group(1)).toString();
140                 return Optional.of(new MergeSource(repoName, "master"));
141             }
142 
143             return Optional.of(new MergeSource(pr.repository().name(), branchMatcher.group(1)));
144         }
145 
146         return Optional.of(new MergeSource(repoMatcher.group(1), repoMatcher.group(2)));
147     }

149     // Additional bot-specific checks that are not handled by JCheck
150     private List<String> botSpecificChecks() throws IOException {
151         var ret = new ArrayList<String>();
152 
153         if (bodyWithoutStatus().isBlank()) {
154             var error = "The pull request body must not be empty.";
155             ret.add(error);
156         }
157 
158         if (!isTargetBranchAllowed()) {
159             var error = "The branch `" + pr.targetRef() + "` is not allowed as target branch. The allowed target branches are:\n" +
160                     allowedTargetBranches().stream()
161                     .map(name -> "   - " + name)
162                     .collect(Collectors.joining("\n"));
163             ret.add(error);
164         }
165 
166         var baseHash = prInstance.baseHash();
167         var headHash = pr.headHash();
168         var commits = prInstance.localRepo().commitMetadata(baseHash, headHash);
169 
170         if (!checkCommitAuthor(commits)) {
171             var error = "For contributors who are not existing OpenJDK Authors, commit attribution will be taken from " +
172                     "the commits in the PR. However, the commits in this PR have inconsistent user names and/or " +
173                     "email addresses. Please amend the commits.";
174             ret.add(error);
175         }
176 
177         if (pr.title().startsWith("Merge")) {
178             if (commits.size() < 2) {
179                 ret.add("A Merge PR must contain at least two commits that are not already present in the target.");
180             } else {
181                 // Find the last merge commit - the very last commit is not eligible (as the merge needs a parent)
182                 int mergeCommitIndex = commits.size();
183                 for (int i = 0; i < commits.size() - 1; ++i) {
184                     if (commits.get(i).isMerge()) {
185                         mergeCommitIndex = i;
186                     }
187                 }
188                 if (mergeCommitIndex >= commits.size() - 1) {
189                     ret.add("A Merge PR must contain a merge commit as well as at least one other commit from the merge source.");
190                 }
191 
192                 var source = mergeSource();
193                 if (source.isPresent()) {
194                     try {
195                         var mergeSourceRepo = pr.repository().forge().repository(source.get().repositoryName).orElseThrow(() ->
196                                 new RuntimeException("Could not find repository " + source.get().repositoryName)
197                         );
198                         try {
199                             var sourceHash = prInstance.localRepo().fetch(mergeSourceRepo.url(), source.get().branchName);
200                             var mergeCommit = commits.get(mergeCommitIndex);
201                             for (int i = 1; i < mergeCommit.parents().size(); ++i) {
202                                 if (!prInstance.localRepo().isAncestor(mergeCommit.parents().get(i), sourceHash)) {
203                                     ret.add("The merge contains commits that are not ancestors of the source.");
204                                     break;
205                                 }
206                             }
207                         } catch (IOException e) {
208                             ret.add("Could not fetch branch `" + source.get().branchName + "` from project `" +
209                                             source.get().repositoryName + "` - check that they are correct.");
210                         }
211                     } catch (RuntimeException e) {
212                         ret.add("Could not find project `" +
213                                         source.get().repositoryName + "` - check that it is correct.");
214                     }
215                 } else {
216                     ret.add("Could not determine the source for this merge. A Merge PR title must be specified on the format: " +
217                             "Merge `project`:`branch` to allow verification of the merge contents.");
218                 }
219             }
220         }
221 
222         for (var blocker : workItem.bot.blockingCheckLabels().entrySet()) {
223             if (labels.contains(blocker.getKey())) {
224                 ret.add(blocker.getValue());
225             }
226         }
227 
228         return ret;
229     }

231     private void updateCheckBuilder(CheckBuilder checkBuilder, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
232         if (visitor.isReadyForReview() && additionalErrors.isEmpty()) {
233             checkBuilder.complete(true);
234         } else {
235             checkBuilder.title("Required");
236             var summary = Stream.concat(visitor.getMessages().stream(), additionalErrors.stream())
237                                 .sorted()
238                                 .map(m -> "- " + m)
239                                 .collect(Collectors.joining("\n"));
240             checkBuilder.summary(summary);
241             for (var annotation : visitor.getAnnotations()) {
242                 checkBuilder.annotation(annotation);
243             }
244             checkBuilder.complete(false);
245         }
246     }

248     private void updateReadyForReview(PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
249         // Additional errors are not allowed
250         if (!additionalErrors.isEmpty()) {
251             newLabels.remove("rfr");
252             return;
253         }
254 
255         // Draft requests are not for review
256         if (pr.isDraft()) {
257             newLabels.remove("rfr");
258             return;
259         }
260 
261         // Check if the visitor found any issues that should be resolved before reviewing
262         if (visitor.isReadyForReview()) {
263             newLabels.add("rfr");
264         } else {
265             newLabels.remove("rfr");
266         }
267     }

269     private String getRole(String username) {
270         var project = censusInstance.project();
271         var version = censusInstance.census().version().format();
272         if (project.isReviewer(username, version)) {
273             return "**Reviewer**";
274         } else if (project.isCommitter(username, version)) {
275             return "Committer";
276         } else if (project.isAuthor(username, version)) {
277             return "Author";
278         } else {
279             return "no project role";
280         }
281     }

283     private String formatReviewer(HostUser reviewer) {
284         var namespace = censusInstance.namespace();
285         var contributor = namespace.get(reviewer.id());
286         if (contributor == null) {
287             return reviewer.userName() + " (no known " + namespace.name() + " user name / role)";
288         } else {
289             var userNameLink = "[" + contributor.username() + "](@" + reviewer.userName() + ")";
290             return contributor.fullName().orElse(contributor.username()) + " (" + userNameLink + " - " +
291                     getRole(contributor.username()) + ")";
292         }
293     }

301     private Optional<String> getReviewersList(List<Review> reviews) {
302         var reviewers = reviews.stream()
303                                .filter(review -> review.verdict() == Review.Verdict.APPROVED)
304                                .map(review -> {
305                                    var entry = " * " + formatReviewer(review.reviewer());
306                                    if (!review.hash().equals(pr.headHash())) {
307                                        if (ignoreStaleReviews) {
308                                            entry += " 🔄 Re-review required (review applies to " + review.hash() + ")";
309                                        } else {
310                                            entry += " ⚠️ Review applies to " + review.hash();
311                                        }
312                                    }
313                                    return entry;
314                                })
315                                .collect(Collectors.joining("\n"));
316         if (reviewers.length() > 0) {
317             return Optional.of(reviewers);
318         } else {
319             return Optional.empty();
320         }
321     }

340     private String getStatusMessage(List<Comment> comments, List<Review> reviews, PullRequestCheckIssueVisitor visitor) {
341         var progressBody = new StringBuilder();
342         progressBody.append("---------\n");
343         progressBody.append("### Progress\n");
344         progressBody.append(getChecksList(visitor));
345 





346         var issue = Issue.fromString(pr.title());
347         var issueProject = workItem.bot.issueProject();
348         if (issueProject != null && issue.isPresent()) {
349             var allIssues = new ArrayList<Issue>();
350             allIssues.add(issue.get());
351             allIssues.addAll(SolvesTracker.currentSolved(pr.repository().forge().currentUser(), comments));
352             progressBody.append("\n\n### Issue");
353             if (allIssues.size() > 1) {
354                 progressBody.append("s");
355             }
356             progressBody.append("\n");
357             for (var currentIssue : allIssues) {
358                 progressBody.append(" * ");
359                 try {
360                     var iss = issueProject.issue(currentIssue.id());
361                     if (iss.isPresent()) {
362                         progressBody.append("[");
363                         progressBody.append(iss.get().id());
364                         progressBody.append("](");
365                         progressBody.append(iss.get().webUrl());
366                         progressBody.append("): ");
367                         progressBody.append(iss.get().title());
368                         progressBody.append("\n");
369                     } else {
370                         progressBody.append("⚠️ Failed to retrieve information on issue `");
371                         progressBody.append(currentIssue.id());
372                         progressBody.append("`.\n");
373                     }
374                 } catch (RuntimeException e) {
375                     progressBody.append("⚠️ Temporary failure when trying to retrieve information on issue `");
376                     progressBody.append(currentIssue.id());
377                     progressBody.append("`.\n");
378                 }
379             }
380         }
381 
382         getReviewersList(reviews).ifPresent(reviewers -> {
383             progressBody.append("\n\n### Reviewers\n");
384             progressBody.append(reviewers);
385         });
386 
387         getContributorsList(comments).ifPresent(contributors -> {
388             progressBody.append("\n\n### Contributors\n");
389             progressBody.append(contributors);
390         });
391 
392         progressBody.append("\n\n### Download\n");
393         progressBody.append(checkoutCommands());
394 
395         return progressBody.toString();
396     }

402            "`$ git checkout pull/" + pr.id() + "`\n";

405     private String bodyWithoutStatus() {
406         var description = pr.body();
407         var markerIndex = description.lastIndexOf(progressMarker);
408         return (markerIndex < 0 ?
409                 description :
410                 description.substring(0, markerIndex)).trim();
411     }

413     private String updateStatusMessage(String message) {
414         var description = pr.body();
415         var markerIndex = description.lastIndexOf(progressMarker);
416 
417         if (markerIndex >= 0 && description.substring(markerIndex).equals(message)) {
418             log.info("Progress already up to date");
419             return description;
420         }
421         var newBody = bodyWithoutStatus() + "\n" + progressMarker + "\n" + message;
422 
423         // TODO? Retrieve the body again here to lower the chance of concurrent updates
424         pr.setBody(newBody);
425         return newBody;
426     }

428     private String verdictToString(Review.Verdict verdict) {
429         switch (verdict) {
430             case APPROVED:
431                 return "changes are approved";
432             case DISAPPROVED:
433                 return "more changes needed";
434             case NONE:
435                 return "comment added";
436             default:
437                 throw new RuntimeException("Unknown verdict: " + verdict);
438         }
439     }

461     private String getMergeReadyComment(String commitMessage, List<Review> reviews, boolean rebasePossible) {
462         var message = new StringBuilder();
463         message.append("@");
464         message.append(pr.author().userName());
465         message.append(" This change now passes all automated pre-integration checks");
466 
467         try {
468             var hasContributingFile =
469                 !prInstance.localRepo().files(prInstance.targetHash(), Path.of("CONTRIBUTING.md")).isEmpty();
470             if (hasContributingFile) {
471                 message.append(". When the change also fulfills all ");
472                 message.append("[project specific requirements](https://github.com/");
473                 message.append(pr.repository().name());
474                 message.append("/blob/");
475                 message.append(pr.targetRef());
476                 message.append("/CONTRIBUTING.md)");
477             }
478         } catch (IOException e) {
479             throw new UncheckedIOException(e);
480         }
481 
482         message.append(", type `/integrate` in a new comment to proceed. After integration, ");
483         message.append("the commit message will be:\n");
484         message.append("```\n");
485         message.append(commitMessage);
486         message.append("\n```\n");
487 
488         message.append("- If you would like to add a summary, use the `/summary` command.\n");
489         message.append("- To credit additional contributors, use the `/contributor` command.\n");
490         message.append("- To add additional solved issues, use the `/solves` command.\n");
491 
492         var divergingCommits = prInstance.divergingCommits();
493         if (divergingCommits.size() > 0) {
494             message.append("\n");
495             message.append("Since the source branch of this PR was last updated there ");
496             if (divergingCommits.size() == 1) {
497                 message.append("has been 1 commit ");
498             } else {
499                 message.append("have been ");
500                 message.append(divergingCommits.size());
501                 message.append(" commits ");
502             }
503             message.append("pushed to the `");
504             message.append(pr.targetRef());
505             message.append("` branch. ");
506             if (rebasePossible) {
507                 message.append("Since there are no conflicts, your changes will automatically be rebased on top of ");
508                 message.append("these commits when integrating. If you prefer to avoid automatic rebasing, please merge `");
509                 message.append(pr.targetRef());
510                 message.append("` into your branch, and then specify the current head hash when integrating, like this: ");
511                 message.append("`/integrate ");
512                 message.append(prInstance.targetHash());
513                 message.append("`.\n");
514             } else {
515                 message.append("Your changes cannot be rebased automatically without conflicts, so you will need to ");
516                 message.append("merge `");
517                 message.append(pr.targetRef());
518                 message.append("` into your branch before integrating.\n");
519             }
520         }
521 
522         if (!ProjectPermissions.mayCommit(censusInstance, pr.author())) {
523             message.append("\n");
524             var contributor = censusInstance.namespace().get(pr.author().id());
525             if (contributor == null) {
526                 message.append("As you are not a known OpenJDK [Author](https://openjdk.java.net/bylaws#author), ");
527             } else {
528                 message.append("As you do not have Committer status in this project, ");
529             }
530 
531             message.append("an existing [Committer](https://openjdk.java.net/bylaws#committer) must agree to ");
532             message.append("[sponsor](https://openjdk.java.net/sponsor/) your change. ");
533             var candidates = reviews.stream()
534                                     .filter(review -> ProjectPermissions.mayCommit(censusInstance, review.reviewer()))
535                                     .map(review -> "@" + review.reviewer().userName())
536                                     .collect(Collectors.joining(", "));
537             if (candidates.length() > 0) {
538                 message.append("Possible candidates are the reviewers of this PR (");
539                 message.append(candidates);
540                 message.append(") but any other Committer may sponsor as well. ");
541             }
542             if (rebasePossible) {
543                 message.append("\n\n");
544                 message.append("➡️ To flag this PR as ready for integration with the above commit message, type ");
545                 message.append("`/integrate` in a new comment. (Afterwards, your sponsor types ");
546                 message.append("`/sponsor` in a new comment to perform the integration).\n");
547             }
548         } else if (rebasePossible) {
549             message.append("\n");
550             message.append("➡️ To integrate this PR with the above commit message to the `" + pr.targetRef() + "` branch, type ");
551             message.append("`/integrate` in a new comment.\n");
552         }
553         message.append(mergeReadyMarker);
554         return message.toString();
555     }

566     private void updateMergeReadyComment(boolean isReady, String commitMessage, List<Comment> comments, List<Review> reviews, boolean rebasePossible) {
567         var existing = findComment(comments, mergeReadyMarker);
568         if (isReady) {
569             var message = getMergeReadyComment(commitMessage, reviews, rebasePossible);
570             if (existing.isEmpty()) {
571                 pr.addComment(message);
572             } else {
573                 pr.updateComment(existing.get().id(), message);
574             }
575         } else {
576             existing.ifPresent(comment -> pr.updateComment(comment.id(), getMergeNoLongerReadyComment()));
577         }
578     }

580     private void addSourceBranchWarningComment(List<Comment> comments) {
581         var existing = findComment(comments, sourceBranchWarningMarker);
582         if (existing.isPresent()) {
583             // Only add the comment once per PR
584             return;
585         }
586         var branch = pr.sourceRef();
587         var message = ":warning: @" + pr.author().userName() + " " +
588             "a branch with the same name as the source branch for this pull request (`" + branch + "`) " +
589             "is present in the [target repository](" + pr.repository().nonTransformedWebUrl() + "). " +
590             "If you eventually integrate this pull request then the branch `" + branch + "` " +
591             "in your [personal fork](" + pr.sourceRepository().nonTransformedWebUrl() + ") will diverge once you sync " +
592             "your personal fork with the upstream repository.\n" +
593             "\n" +
594             "To avoid this situation, create a new branch for your changes and reset the `" + branch + "` branch. " +
595             "You can do this by running the following commands in a local repository for your personal fork. " +
596             "_Note_: you do *not* have to name the new branch `NEW-BRANCH-NAME`." +
597             "\n" +
598             "```" +
599             "$ git checkout " + branch + "\n" +
600             "$ git checkout -b NEW-BRANCH-NAME\n" +
601             "$ git branch -f " + branch + " " + prInstance.baseHash().hex() + "\n" +
602             "$ git push -f origin " + branch + "\n" +
603             "```\n" +
604             "\n" +
605             "Then proceed to create a new pull request with `NEW-BRANCH-NAME` as the source branch and " +
606             "close this one.\n" +
607             sourceBranchWarningMarker;
608         pr.addComment(message);
609     }

611     private void addOutdatedComment(List<Comment> comments) {
612         var existing = findComment(comments, outdatedHelpMarker);
613         if (existing.isPresent()) {
614             // Only add the comment once per PR
615             return;
616         }
617         var message = "@" + pr.author().userName() + " this pull request can no longer be integrated into " +
618                 "`" + pr.targetRef() + "` due to one or more merge conflicts. To resolve these merge conflicts " +
619                 "and update this pull request you can run the following commands in the local repository for your personal fork:\n" +
620                 "```bash\n" +
621                 "git checkout " + pr.sourceRef() + "\n" +
622                 "git fetch " + pr.repository().webUrl() + " " + pr.targetRef() + "\n" +
623                 "git merge FETCH_HEAD\n" +
624                 "# resolve conflicts and follow the instructions given by git merge\n" +
625                 "git commit -m \"Merge " + pr.targetRef() + "\"\n" +
626                 "git push\n" +
627                 "```\n" +
628                 outdatedHelpMarker;
629         pr.addComment(message);
630     }

632     private void checkStatus() {
633         var checkBuilder = CheckBuilder.create("jcheck", pr.headHash());
634         var censusDomain = censusInstance.configuration().census().domain();
635         Exception checkException = null;
636 
637         try {
638             // Post check in-progress
639             log.info("Starting to run jcheck on PR head");
640             pr.createCheck(checkBuilder.build());
641             List<String> additionalErrors = List.of();
642             Hash localHash;
643             try {
644                 localHash = prInstance.commit(censusInstance.namespace(), censusDomain, null);
645             } catch (CommitFailure e) {
646                 additionalErrors = List.of("It was not possible to create a commit for the changes in this PR: " + e.getMessage());
647                 localHash = prInstance.baseHash();
648             }
649             boolean rebasePossible = true;
650             PullRequestCheckIssueVisitor visitor = prInstance.createVisitor(localHash, censusInstance);
651             if (!localHash.equals(prInstance.baseHash())) {
652                 // Try to rebase
653                 var ignored = new PrintWriter(new StringWriter());
654                 var rebasedHash = prInstance.rebase(localHash, ignored);
655                 if (rebasedHash.isEmpty()) {
656                     rebasePossible = false;
657                 } else {
658                     localHash = rebasedHash.get();
659                 }
660 
661                 // Determine current status
662                 var additionalConfiguration = AdditionalConfiguration.get(prInstance.localRepo(), localHash, pr.repository().forge().currentUser(), comments);
663                 prInstance.executeChecks(localHash, censusInstance, visitor, additionalConfiguration);
664                 additionalErrors = botSpecificChecks();
665             } else {
666                 if (additionalErrors.isEmpty()) {
667                     additionalErrors = List.of("This PR contains no changes");
668                 }
669             }
670             updateCheckBuilder(checkBuilder, visitor, additionalErrors);
671             updateReadyForReview(visitor, additionalErrors);
672 
673             // Calculate and update the status message if needed
674             var statusMessage = getStatusMessage(comments, activeReviews, visitor);
675             var updatedBody = updateStatusMessage(statusMessage);
676 
677             // Post / update approval messages (only needed if the review itself can't contain a body)
678             if (!pr.repository().forge().supportsReviewBody()) {
679                 updateReviewedMessages(comments, allReviews);
680             }
681 
682             var commit = prInstance.localRepo().lookup(localHash).orElseThrow();
683             var commitMessage = String.join("\n", commit.message());
684             var readyForIntegration = visitor.getMessages().isEmpty() && additionalErrors.isEmpty();
685             updateMergeReadyComment(readyForIntegration, commitMessage, comments, activeReviews, rebasePossible);
686             if (readyForIntegration && rebasePossible) {
687                 newLabels.add("ready");
688             } else {
689                 newLabels.remove("ready");
690             }
691             if (!rebasePossible) {
692                 if (!labels.contains("failed-auto-merge")) {
693                     addOutdatedComment(comments);
694                 }
695                 newLabels.add("merge-conflict");
696             } else {
697                 newLabels.remove("merge-conflict");
698             }
699 
700             var branchNames = pr.repository().branches().stream().map(HostedBranch::name).collect(Collectors.toSet());
701             if (!pr.repository().url().equals(pr.sourceRepository().url()) && branchNames.contains(pr.sourceRef())) {
702                 addSourceBranchWarningComment(comments);
703             }
704 
705             // Ensure that the ready for sponsor label is up to date
706             newLabels.remove("sponsor");
707             var readyHash = ReadyForSponsorTracker.latestReadyForSponsor(pr.repository().forge().currentUser(), comments);
708             if (readyHash.isPresent() && readyForIntegration) {
709                 var acceptedHash = readyHash.get();
710                 if (pr.headHash().equals(acceptedHash)) {
711                     newLabels.add("sponsor");
712                 }
713             }
714 
715             // Calculate current metadata to avoid unnecessary future checks
716             var metadata = workItem.getMetadata(pr.title(), updatedBody, pr.comments(), activeReviews, newLabels,
717                                                 censusInstance, pr.targetHash(), pr.isDraft());
718             checkBuilder.metadata(metadata);
719         } catch (Exception e) {
720             log.throwing("CommitChecker", "checkStatus", e);
721             newLabels.remove("ready");
722             checkBuilder.metadata("invalid");
723             checkBuilder.title("Exception occurred during jcheck - the operation will be retried");
724             checkBuilder.summary(e.getMessage());
725             checkBuilder.complete(false);
726             checkException = e;
727         }
728         var check = checkBuilder.build();
729         pr.updateCheck(check);
730 
731         // Synchronize the wanted set of labels
732         for (var newLabel : newLabels) {
733             if (!labels.contains(newLabel)) {
734                 pr.addLabel(newLabel);
735             }
736         }
737         for (var oldLabel : labels) {
738             if (!newLabels.contains(oldLabel)) {
739                 pr.removeLabel(oldLabel);
740             }
741         }
742 
743         // After updating the PR, rethrow any exception to automatically retry on transient errors
744         if (checkException != null) {
745             throw new RuntimeException("Exception during jcheck", checkException);
746         }
747     }
