 81     private boolean isTargetBranchAllowed() {
 82         var matcher = workItem.bot.allowedTargetBranches().matcher(pr.targetRef());
 83         return matcher.matches();
 84     }

 86     private List<String> allowedTargetBranches() {
 87         return pr.repository()
 88                  .branches()
 89                  .stream()
 90                  .map(HostedBranch::name)
 91                  .map(name -> workItem.bot.allowedTargetBranches().matcher(name))
 92                  .filter(Matcher::matches)
 93                  .map(Matcher::group)
 94                  .collect(Collectors.toList());
 95     }

 97     // For unknown contributors, check that all commits have the same name and email
 98     private boolean checkCommitAuthor(List<CommitMetadata> commits) throws IOException {
 99         var author = censusInstance.namespace().get(pr.author().id());
100         if (author != null) {
101             return true;
102         }
103 
104         var names = new HashSet<String>();
105         var emails = new HashSet<String>();
106 
107         for (var commit : commits) {
108             names.add(commit.author().name());
109             emails.add(commit.author().email());
110         }
111 
112         return ((names.size() == 1) && emails.size() == 1);
113     }

121             this.branchName = branchName;

125     private Optional<MergeSource> mergeSource() {
126         var repoMatcher = mergeSourceFullPattern.matcher(pr.title());
127         if (!repoMatcher.matches()) {
128             var branchMatcher = mergeSourceBranchOnlyPattern.matcher(pr.title());
129             if (!branchMatcher.matches()) {
130                 return Optional.empty();
131             }
132 
133             // Verify that the branch exists
134             var isValidBranch = prInstance.remoteBranches().stream()
135                                           .map(Reference::name)
136                                           .anyMatch(branch -> branch.equals(branchMatcher.group(1)));
137             if (!isValidBranch) {
138                 // Assume the name refers to a sibling repository
139                 var repoName = Path.of(pr.repository().name()).resolveSibling(branchMatcher.group(1)).toString();
140                 return Optional.of(new MergeSource(repoName, "master"));
141             }
142 
143             return Optional.of(new MergeSource(pr.repository().name(), branchMatcher.group(1)));
144         }
145 
146         return Optional.of(new MergeSource(repoMatcher.group(1), repoMatcher.group(2)));
147     }

149     // Additional bot-specific checks that are not handled by JCheck
150     private List<String> botSpecificChecks() throws IOException {
151         var ret = new ArrayList<String>();
152 
153         if (bodyWithoutStatus().isBlank()) {
154             var error = "The pull request body must not be empty.";
155             ret.add(error);
156         }
157 
158         if (!isTargetBranchAllowed()) {
159             var error = "The branch `" + pr.targetRef() + "` is not allowed as target branch. The allowed target branches are:\n" +
160                     allowedTargetBranches().stream()
161                     .map(name -> "   - " + name)
162                     .collect(Collectors.joining("\n"));
163             ret.add(error);
164         }
165 
166         var baseHash = prInstance.baseHash();
167         var headHash = pr.headHash();
168         var commits = prInstance.localRepo().commitMetadata(baseHash, headHash);
169 
170         if (!checkCommitAuthor(commits)) {
171             var error = "For contributors who are not existing OpenJDK Authors, commit attribution will be taken from " +
172                     "the commits in the PR. However, the commits in this PR have inconsistent user names and/or " +
173                     "email addresses. Please amend the commits.";
174             ret.add(error);
175         }
176 
177         if (pr.title().startsWith("Merge")) {
178             if (commits.size() < 2) {
179                 ret.add("A Merge PR must contain at least two commits that are not already present in the target.");
180             } else {
181                 // Find the last merge commit - the very last commit is not eligible (as the merge needs a parent)
182                 int mergeCommitIndex = commits.size();
183                 for (int i = 0; i < commits.size() - 1; ++i) {
184                     if (commits.get(i).isMerge()) {
185                         mergeCommitIndex = i;
186                     }
187                 }
188                 if (mergeCommitIndex >= commits.size() - 1) {
189                     ret.add("A Merge PR must contain a merge commit as well as at least one other commit from the merge source.");
190                 }
191 
192                 var source = mergeSource();
193                 if (source.isPresent()) {
194                     try {
195                         var mergeSourceRepo = pr.repository().forge().repository(source.get().repositoryName).orElseThrow(() ->
196                                 new RuntimeException("Could not find repository " + source.get().repositoryName)
197                         );
198                         try {
199                             var sourceHash = prInstance.localRepo().fetch(mergeSourceRepo.url(), source.get().branchName);
200                             var mergeCommit = commits.get(mergeCommitIndex);
201                             for (int i = 1; i < mergeCommit.parents().size(); ++i) {
202                                 if (!prInstance.localRepo().isAncestor(mergeCommit.parents().get(i), sourceHash)) {
203                                     ret.add("The merge contains commits that are not ancestors of the source.");
204                                     break;
205                                 }
206                             }
207                         } catch (IOException e) {
208                             ret.add("Could not fetch branch `" + source.get().branchName + "` from project `" +
209                                             source.get().repositoryName + "` - check that they are correct.");
210                         }
211                     } catch (RuntimeException e) {
212                         ret.add("Could not find project `" +
213                                         source.get().repositoryName + "` - check that it is correct.");
214                     }
215                 } else {
216                     ret.add("Could not determine the source for this merge. A Merge PR title must be specified on the format: " +
217                             "Merge `project`:`branch` to allow verification of the merge contents.");
218                 }
219             }
220         }
221 
222         for (var blocker : workItem.bot.blockingCheckLabels().entrySet()) {
223             if (labels.contains(blocker.getKey())) {
224                 ret.add(blocker.getValue());
225             }
226         }
227 
228         return ret;
229     }

231     private void updateCheckBuilder(CheckBuilder checkBuilder, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
232         if (visitor.isReadyForReview() && additionalErrors.isEmpty()) {
233             checkBuilder.complete(true);
234         } else {
235             checkBuilder.title("Required");
236             var summary = Stream.concat(visitor.getMessages().stream(), additionalErrors.stream())
237                                 .sorted()
238                                 .map(m -> "- " + m)
239                                 .collect(Collectors.joining("\n"));
240             checkBuilder.summary(summary);
241             for (var annotation : visitor.getAnnotations()) {
242                 checkBuilder.annotation(annotation);
243             }
244             checkBuilder.complete(false);
245         }
246     }

248     private void updateReadyForReview(PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
249         // Additional errors are not allowed
250         if (!additionalErrors.isEmpty()) {
251             newLabels.remove("rfr");
252             return;
253         }
254 
255         // Draft requests are not for review
256         if (pr.isDraft()) {
257             newLabels.remove("rfr");
258             return;
259         }
260 
261         // Check if the visitor found any issues that should be resolved before reviewing
262         if (visitor.isReadyForReview()) {
263             newLabels.add("rfr");
264         } else {
265             newLabels.remove("rfr");
266         }
267     }

269     private String getRole(String username) {
270         var project = censusInstance.project();
271         var version = censusInstance.census().version().format();
272         if (project.isReviewer(username, version)) {
273             return "**Reviewer**";
274         } else if (project.isCommitter(username, version)) {
275             return "Committer";
276         } else if (project.isAuthor(username, version)) {
277             return "Author";
278         } else {
279             return "no project role";
280         }
281     }

283     private String formatReviewer(HostUser reviewer) {
284         var namespace = censusInstance.namespace();
285         var contributor = namespace.get(reviewer.id());
286         if (contributor == null) {
287             return reviewer.userName() + " (no known " + namespace.name() + " user name / role)";
288         } else {
289             var userNameLink = "[" + contributor.username() + "](@" + reviewer.userName() + ")";
290             return contributor.fullName().orElse(contributor.username()) + " (" + userNameLink + " - " +
291                     getRole(contributor.username()) + ")";
292         }
293     }

302         return additionalErrors.stream()

305                                .collect(Collectors.joining("\n"));

308     private Optional<String> getReviewersList(List<Review> reviews) {
309         var reviewers = reviews.stream()
310                                .filter(review -> review.verdict() == Review.Verdict.APPROVED)
311                                .map(review -> {
312                                    var entry = " * " + formatReviewer(review.reviewer());
313                                    if (!review.hash().equals(pr.headHash())) {
314                                        if (ignoreStaleReviews) {
315                                            entry += " 🔄 Re-review required (review applies to " + review.hash() + ")";
316                                        } else {
317                                            entry += " ⚠️ Review applies to " + review.hash();
318                                        }
319                                    }
320                                    return entry;
321                                })
322                                .collect(Collectors.joining("\n"));
323         if (reviewers.length() > 0) {
324             return Optional.of(reviewers);
325         } else {
326             return Optional.empty();
327         }
328     }

347     private String getStatusMessage(List<Comment> comments, List<Review> reviews, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
348         var progressBody = new StringBuilder();
349         progressBody.append("---------\n");
350         progressBody.append("### Progress\n");
351         progressBody.append(getChecksList(visitor));
352 
353         if (!additionalErrors.isEmpty()) {
354             progressBody.append("\n\n### Problems\n");
355             progressBody.append(getAdditionalErrorsList(additionalErrors));
356         }
357 
358         var issue = Issue.fromString(pr.title());
359         var issueProject = workItem.bot.issueProject();
360         if (issueProject != null && issue.isPresent()) {
361             var allIssues = new ArrayList<Issue>();
362             allIssues.add(issue.get());
363             allIssues.addAll(SolvesTracker.currentSolved(pr.repository().forge().currentUser(), comments));
364             progressBody.append("\n\n### Issue");
365             if (allIssues.size() > 1) {
366                 progressBody.append("s");
367             }
368             progressBody.append("\n");
369             for (var currentIssue : allIssues) {
370                 progressBody.append(" * ");
371                 try {
372                     var iss = issueProject.issue(currentIssue.id());
373                     if (iss.isPresent()) {
374                         progressBody.append("[");
375                         progressBody.append(iss.get().id());
376                         progressBody.append("](");
377                         progressBody.append(iss.get().webUrl());
378                         progressBody.append("): ");
379                         progressBody.append(iss.get().title());
380                         progressBody.append("\n");
381                     } else {
382                         progressBody.append("⚠️ Failed to retrieve information on issue `");
383                         progressBody.append(currentIssue.id());
384                         progressBody.append("`.\n");
385                     }
386                 } catch (RuntimeException e) {
387                     progressBody.append("⚠️ Temporary failure when trying to retrieve information on issue `");
388                     progressBody.append(currentIssue.id());
389                     progressBody.append("`.\n");
390                 }
391             }
392         }
393 
394         getReviewersList(reviews).ifPresent(reviewers -> {
395             progressBody.append("\n\n### Reviewers\n");
396             progressBody.append(reviewers);
397         });
398 
399         getContributorsList(comments).ifPresent(contributors -> {
400             progressBody.append("\n\n### Contributors\n");
401             progressBody.append(contributors);
402         });
403 
404         progressBody.append("\n\n### Download\n");
405         progressBody.append(checkoutCommands());
406 
407         return progressBody.toString();
408     }

410     private String checkoutCommands() {
411         var repoUrl = pr.repository().webUrl();
412         return
413            "`$ git fetch " + repoUrl + " " + pr.fetchRef() + ":pull/" + pr.id() + "`\n" +
414            "`$ git checkout pull/" + pr.id() + "`\n";
415     }

417     private String bodyWithoutStatus() {
418         var description = pr.body();
419         var markerIndex = description.lastIndexOf(progressMarker);
420         return (markerIndex < 0 ?
421                 description :
422                 description.substring(0, markerIndex)).trim();
423     }

425     private String updateStatusMessage(String message) {
426         var description = pr.body();
427         var markerIndex = description.lastIndexOf(progressMarker);
428 
429         if (markerIndex >= 0 && description.substring(markerIndex).equals(message)) {
430             log.info("Progress already up to date");
431             return description;
432         }
433         var newBody = bodyWithoutStatus() + "\n" + progressMarker + "\n" + message;
434 
435         // TODO? Retrieve the body again here to lower the chance of concurrent updates
436         pr.setBody(newBody);
437         return newBody;
438     }

440     private String verdictToString(Review.Verdict verdict) {
441         switch (verdict) {
442             case APPROVED:
443                 return "changes are approved";
444             case DISAPPROVED:
445                 return "more changes needed";
446             case NONE:
447                 return "comment added";
448             default:
449                 throw new RuntimeException("Unknown verdict: " + verdict);
450         }
451     }

473     private String getMergeReadyComment(String commitMessage, List<Review> reviews, boolean rebasePossible) {
474         var message = new StringBuilder();
475         message.append("@");
476         message.append(pr.author().userName());
477         message.append(" This change now passes all automated pre-integration checks");
478 
479         try {
480             var hasContributingFile =
481                 !prInstance.localRepo().files(prInstance.targetHash(), Path.of("CONTRIBUTING.md")).isEmpty();
482             if (hasContributingFile) {
483                 message.append(". When the change also fulfills all ");
484                 message.append("[project specific requirements](https://github.com/");
485                 message.append(pr.repository().name());
486                 message.append("/blob/");
487                 message.append(pr.targetRef());
488                 message.append("/CONTRIBUTING.md)");
489             }
490         } catch (IOException e) {
491             throw new UncheckedIOException(e);
492         }
493 
494         message.append(", type `/integrate` in a new comment to proceed. After integration, ");
495         message.append("the commit message will be:\n");
496         message.append("```\n");
497         message.append(commitMessage);
498         message.append("\n```\n");
499 
500         message.append("- If you would like to add a summary, use the `/summary` command.\n");
501         message.append("- To credit additional contributors, use the `/contributor` command.\n");
502         message.append("- To add additional solved issues, use the `/solves` command.\n");
503 
504         var divergingCommits = prInstance.divergingCommits();
505         if (divergingCommits.size() > 0) {
506             message.append("\n");
507             message.append("Since the source branch of this PR was last updated there ");
508             if (divergingCommits.size() == 1) {
509                 message.append("has been 1 commit ");
510             } else {
511                 message.append("have been ");
512                 message.append(divergingCommits.size());
513                 message.append(" commits ");
514             }
515             message.append("pushed to the `");
516             message.append(pr.targetRef());
517             message.append("` branch. ");
518             if (rebasePossible) {
519                 message.append("Since there are no conflicts, your changes will automatically be rebased on top of ");
520                 message.append("these commits when integrating. If you prefer to avoid automatic rebasing, please merge `");
521                 message.append(pr.targetRef());
522                 message.append("` into your branch, and then specify the current head hash when integrating, like this: ");
523                 message.append("`/integrate ");
524                 message.append(prInstance.targetHash());
525                 message.append("`.\n");
526             } else {
527                 message.append("Your changes cannot be rebased automatically without conflicts, so you will need to ");
528                 message.append("merge `");
529                 message.append(pr.targetRef());
530                 message.append("` into your branch before integrating.\n");
531             }
532         }
533 
534         if (!ProjectPermissions.mayCommit(censusInstance, pr.author())) {
535             message.append("\n");
536             var contributor = censusInstance.namespace().get(pr.author().id());
537             if (contributor == null) {
538                 message.append("As you are not a known OpenJDK [Author](https://openjdk.java.net/bylaws#author), ");
539             } else {
540                 message.append("As you do not have Committer status in this project, ");
541             }
542 
543             message.append("an existing [Committer](https://openjdk.java.net/bylaws#committer) must agree to ");
544             message.append("[sponsor](https://openjdk.java.net/sponsor/) your change. ");
545             var candidates = reviews.stream()
546                                     .filter(review -> ProjectPermissions.mayCommit(censusInstance, review.reviewer()))
547                                     .map(review -> "@" + review.reviewer().userName())
548                                     .collect(Collectors.joining(", "));
549             if (candidates.length() > 0) {
550                 message.append("Possible candidates are the reviewers of this PR (");
551                 message.append(candidates);
552                 message.append(") but any other Committer may sponsor as well. ");
553             }
554             if (rebasePossible) {
555                 message.append("\n\n");
556                 message.append("➡️ To flag this PR as ready for integration with the above commit message, type ");
557                 message.append("`/integrate` in a new comment. (Afterwards, your sponsor types ");
558                 message.append("`/sponsor` in a new comment to perform the integration).\n");
559             }
560         } else if (rebasePossible) {
561             message.append("\n");
562             message.append("➡️ To integrate this PR with the above commit message to the `" + pr.targetRef() + "` branch, type ");
563             message.append("`/integrate` in a new comment.\n");
564         }
565         message.append(mergeReadyMarker);
566         return message.toString();
567     }

578     private void updateMergeReadyComment(boolean isReady, String commitMessage, List<Comment> comments, List<Review> reviews, boolean rebasePossible) {
579         var existing = findComment(comments, mergeReadyMarker);
580         if (isReady) {
581             var message = getMergeReadyComment(commitMessage, reviews, rebasePossible);
582             if (existing.isEmpty()) {
583                 pr.addComment(message);
584             } else {
585                 pr.updateComment(existing.get().id(), message);
586             }
587         } else {
588             existing.ifPresent(comment -> pr.updateComment(comment.id(), getMergeNoLongerReadyComment()));
589         }
590     }

592     private void addSourceBranchWarningComment(List<Comment> comments) {
593         var existing = findComment(comments, sourceBranchWarningMarker);
594         if (existing.isPresent()) {
595             // Only add the comment once per PR
596             return;
597         }
598         var branch = pr.sourceRef();
599         var message = ":warning: @" + pr.author().userName() + " " +
600             "a branch with the same name as the source branch for this pull request (`" + branch + "`) " +
601             "is present in the [target repository](" + pr.repository().nonTransformedWebUrl() + "). " +
602             "If you eventually integrate this pull request then the branch `" + branch + "` " +
603             "in your [personal fork](" + pr.sourceRepository().nonTransformedWebUrl() + ") will diverge once you sync " +
604             "your personal fork with the upstream repository.\n" +
605             "\n" +
606             "To avoid this situation, create a new branch for your changes and reset the `" + branch + "` branch. " +
607             "You can do this by running the following commands in a local repository for your personal fork. " +
608             "_Note_: you do *not* have to name the new branch `NEW-BRANCH-NAME`." +
609             "\n" +
610             "```" +
611             "$ git checkout " + branch + "\n" +
612             "$ git checkout -b NEW-BRANCH-NAME\n" +
613             "$ git branch -f " + branch + " " + prInstance.baseHash().hex() + "\n" +
614             "$ git push -f origin " + branch + "\n" +
615             "```\n" +
616             "\n" +
617             "Then proceed to create a new pull request with `NEW-BRANCH-NAME` as the source branch and " +
618             "close this one.\n" +
619             sourceBranchWarningMarker;
620         pr.addComment(message);
621     }

623     private void addOutdatedComment(List<Comment> comments) {
624         var existing = findComment(comments, outdatedHelpMarker);
625         if (existing.isPresent()) {
626             // Only add the comment once per PR
627             return;
628         }
629         var message = "@" + pr.author().userName() + " this pull request can no longer be integrated into " +
630                 "`" + pr.targetRef() + "` due to one or more merge conflicts. To resolve these merge conflicts " +
631                 "and update this pull request you can run the following commands in the local repository for your personal fork:\n" +
632                 "```bash\n" +
633                 "git checkout " + pr.sourceRef() + "\n" +
634                 "git fetch " + pr.repository().webUrl() + " " + pr.targetRef() + "\n" +
635                 "git merge FETCH_HEAD\n" +
636                 "# resolve conflicts and follow the instructions given by git merge\n" +
637                 "git commit -m \"Merge " + pr.targetRef() + "\"\n" +
638                 "git push\n" +
639                 "```\n" +
640                 outdatedHelpMarker;
641         pr.addComment(message);
642     }

644     private void checkStatus() {
645         var checkBuilder = CheckBuilder.create("jcheck", pr.headHash());
646         var censusDomain = censusInstance.configuration().census().domain();
647         Exception checkException = null;
648 
649         try {
650             // Post check in-progress
651             log.info("Starting to run jcheck on PR head");
652             pr.createCheck(checkBuilder.build());
653             List<String> additionalErrors = List.of();
654             Hash localHash;
655             try {
656                 localHash = prInstance.commit(censusInstance.namespace(), censusDomain, null);
657             } catch (CommitFailure e) {
658                 additionalErrors = List.of("It was not possible to create a commit for the changes in this PR: " + e.getMessage());
659                 localHash = prInstance.baseHash();
660             }
661             boolean rebasePossible = true;
662             PullRequestCheckIssueVisitor visitor = prInstance.createVisitor(localHash, censusInstance);
663             if (!localHash.equals(prInstance.baseHash())) {
664                 // Try to rebase
665                 var ignored = new PrintWriter(new StringWriter());
666                 var rebasedHash = prInstance.rebase(localHash, ignored);
667                 if (rebasedHash.isEmpty()) {
668                     rebasePossible = false;
669                 } else {
670                     localHash = rebasedHash.get();
671                 }
672 
673                 // Determine current status
674                 var additionalConfiguration = AdditionalConfiguration.get(prInstance.localRepo(), localHash, pr.repository().forge().currentUser(), comments);
675                 prInstance.executeChecks(localHash, censusInstance, visitor, additionalConfiguration);
676                 additionalErrors = botSpecificChecks();
677             } else {
678                 if (additionalErrors.isEmpty()) {
679                     additionalErrors = List.of("This PR contains no changes");
680                 }
681             }
682             updateCheckBuilder(checkBuilder, visitor, additionalErrors);
683             updateReadyForReview(visitor, additionalErrors);
684 
685             // Calculate and update the status message if needed
686             var statusMessage = getStatusMessage(comments, activeReviews, visitor, additionalErrors);
687             var updatedBody = updateStatusMessage(statusMessage);
688 
689             // Post / update approval messages (only needed if the review itself can't contain a body)
690             if (!pr.repository().forge().supportsReviewBody()) {
691                 updateReviewedMessages(comments, allReviews);
692             }
693 
694             var commit = prInstance.localRepo().lookup(localHash).orElseThrow();
695             var commitMessage = String.join("\n", commit.message());
696             var readyForIntegration = visitor.getMessages().isEmpty() && additionalErrors.isEmpty();
697             updateMergeReadyComment(readyForIntegration, commitMessage, comments, activeReviews, rebasePossible);
698             if (readyForIntegration && rebasePossible) {
699                 newLabels.add("ready");
700             } else {
701                 newLabels.remove("ready");
702             }
703             if (!rebasePossible) {
704                 if (!labels.contains("failed-auto-merge")) {
705                     addOutdatedComment(comments);
706                 }
707                 newLabels.add("merge-conflict");
708             } else {
709                 newLabels.remove("merge-conflict");
710             }
711 
712             var branchNames = pr.repository().branches().stream().map(HostedBranch::name).collect(Collectors.toSet());
713             if (!pr.repository().url().equals(pr.sourceRepository().url()) && branchNames.contains(pr.sourceRef())) {
714                 addSourceBranchWarningComment(comments);
715             }
716 
717             // Ensure that the ready for sponsor label is up to date
718             newLabels.remove("sponsor");
719             var readyHash = ReadyForSponsorTracker.latestReadyForSponsor(pr.repository().forge().currentUser(), comments);
720             if (readyHash.isPresent() && readyForIntegration) {
721                 var acceptedHash = readyHash.get();
722                 if (pr.headHash().equals(acceptedHash)) {
723                     newLabels.add("sponsor");
724                 }
725             }
726 
727             // Calculate current metadata to avoid unnecessary future checks
728             var metadata = workItem.getMetadata(pr.title(), updatedBody, pr.comments(), activeReviews, newLabels,
729                                                 censusInstance, pr.targetHash(), pr.isDraft());
730             checkBuilder.metadata(metadata);
731         } catch (Exception e) {
732             log.throwing("CommitChecker", "checkStatus", e);
733             newLabels.remove("ready");
734             checkBuilder.metadata("invalid");
735             checkBuilder.title("Exception occurred during jcheck - the operation will be retried");
736             checkBuilder.summary(e.getMessage());
737             checkBuilder.complete(false);
738             checkException = e;
739         }
740         var check = checkBuilder.build();
741         pr.updateCheck(check);
742 
743         // Synchronize the wanted set of labels
744         for (var newLabel : newLabels) {
745             if (!labels.contains(newLabel)) {
746                 pr.addLabel(newLabel);
747             }
748         }
749         for (var oldLabel : labels) {
750             if (!newLabels.contains(oldLabel)) {
751                 pr.removeLabel(oldLabel);
752             }
753         }
754 
755         // After updating the PR, rethrow any exception to automatically retry on transient errors
756         if (checkException != null) {
757             throw new RuntimeException("Exception during jcheck", checkException);
758         }
759     }
