22 BUILD=build
23 prefix=$(HOME)/.local
24 bindir=$(prefix)/bin
25 sharedir=$(prefix)/share
26 mandir=$(prefix)/man
27 
28 LAUNCHERS=$(addprefix $(bindir)/,$(notdir $(wildcard $(BUILD)/bin/git-*)))
29 MANPAGES=$(addprefix $(mandir)/man1/,$(notdir $(wildcard $(BUILD)/bin/man/man1/*)))
30 
31 all:
32 	@sh gradlew
33 
34 check:
35 	@sh gradlew test
36 
37 test:
38 	@sh gradlew test
39 
40 clean:
41 	@sh gradlew clean
42 
43 images:
44 	@sh gradlew images
45 
46 bots:
47 	@sh gradlew :bots:cli:images
48 
49 offline:
50 	@sh gradlew :offline
51 
52 reproduce:
53 	@sh gradlew :reproduce
54 
55 install: all $(LAUNCHERS) $(MANPAGES) $(sharedir)/skara
56 	@echo "Successfully installed to $(prefix)"
57 
58 uninstall:
59 	@rm -rf $(sharedir)/skara
60 	@rm $(LAUNCHERS)
61 	@rm $(MANPAGES)
62 
63 $(mandir)/man1/%: $(BUILD)/bin/man/man1/%
64 	@mkdir -p $(mandir)/man1
65 	@cp $< $@
66 
67 $(sharedir)/skara: $(BUILD)/image
68 	@mkdir -p $(sharedir)
69 	@rm -rf $@
70 	@cp -r $< $@
71 
72 $(bindir)/%: $(BUILD)/bin/%
73 	@mkdir -p $(bindir)
74 	@sed 's~export JAVA_HOME=.*$$~export JAVA_HOME\=$(sharedir)\/skara~' < $< > $@
75 	@chmod 755 $@
76 
77 .PHONY: all bots check clean images install test uninstall
