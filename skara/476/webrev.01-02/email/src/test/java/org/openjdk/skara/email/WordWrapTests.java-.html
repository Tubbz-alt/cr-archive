1 package org.openjdk.skara.email;
 2 
 3 import org.junit.jupiter.api.Test;
 4 
 5 import static org.junit.jupiter.api.Assertions.assertEquals;
 6 
 7 public class WordWrapTests {
 8     @Test
 9     void simple() {
10         assertEquals("hello\nthere\nyou", WordWrap.wrapBody("hello there you", 2));
11         assertEquals("hello\nthere", WordWrap.wrapBody("hello there", 7));
12         assertEquals("hello there", WordWrap.wrapBody("hello there", 20));
13         assertEquals("hello\nthere", WordWrap.wrapBody("hello   there", 7));
14     }
15 
16     @Test
17     void indented() {
18         assertEquals("  hello\n  there", WordWrap.wrapBody("  hello there", 10));
19         assertEquals("  hello\n  there\n you", WordWrap.wrapBody("  hello there\n you", 10));
20     }
21 
22     @Test
23     void quoted() {
24         assertEquals("> hello\n> there", WordWrap.wrapBody("> hello there", 10));
25         assertEquals("> hello\n> there\n> you", WordWrap.wrapBody("> hello there\n> you", 2));
26         assertEquals(">> hello\n>> there\n> you", WordWrap.wrapBody(">> hello there\n> you", 2));
27     }
28 
29     @Test
30     void list() {
31         assertEquals(" - hello\n   there\n - you", WordWrap.wrapBody(" - hello there\n - you", 10));
32         assertEquals(" - hello\n   there", WordWrap.wrapBody(" - hello there", 10));
33     }
34 }
