24 DIR=$(dirname "${0}")
25 OS=$(uname)
26 
27 if [ "${OS}" = "Linux" -o "${OS}" = "Darwin" ]; then
28     if [ ! -x "${DIR}/bin/bin/git-skara" ]; then
29         echo "Compiling ..."
30         (cd "${DIR}" && sh gradlew)
31     fi
32 else
33     if [ ! -f "${DIR}/bin/bin/git-skara.bat" ]; then
34         echo "Compiling ..."
35         (cd "${DIR}" && ./gradlew.bat)
36     fi
37 
38 fi
39 
40 if [ -d "${DIR}/build" ]; then
41     rm -rf "${DIR}/bin"
42     mv "${DIR}/build" "${DIR}/bin"
43 fi
44 
45 if [ "${OS}" = "Linux" -o "${OS}" = "Darwin" ]; then
46     exec "${DIR}/bin/bin/git-skara" "${@}"
47 else
48     exec "${DIR}/bin/bin/git-skara.bat" "${@}"
49 fi
