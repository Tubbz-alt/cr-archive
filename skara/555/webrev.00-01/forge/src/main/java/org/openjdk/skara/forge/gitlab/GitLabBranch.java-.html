23 package org.openjdk.skara.forge.gitlab;
24 
25 import org.openjdk.skara.forge.*;
26 import org.openjdk.skara.vcs.Hash;
27 
28 import java.util.Objects;
29 
30 class GitLabBranch implements HostedBranch {
31     private final String name;
32     private final Hash hash;
33 
34     GitLabBranch(String name, Hash hash) {
35         this.name = name;
36         this.hash = hash;
37     }
38 
39     @Override
40     public String name() {
41         return name;
42     }
43 
44     @Override
45     public Hash hash() {
46         return hash;
47     }
48 
49     @Override
50     public String toString() {
51         return name + "@" + hash.hex();
52     }
53 
54     @Override
55     public int hashCode() {
56         return Objects.hash(name, hash);
57     }
58 
59     @Override
60     public boolean equals(Object other) {
61         if (!(other instanceof GitLabBranch)) {
62             return false;
63         }
64 
65         var o = (GitLabBranch) other;
66         return Objects.equals(name, o.name) && Objects.equals(hash, o.hash);
67     }
68 }
