23 package org.openjdk.skara.forge;
24 
25 import org.openjdk.skara.vcs.Hash;
26 
27 import java.util.Objects;
28 
29 public class HostedBranch {
30     private final String name;
31     private final Hash hash;
32 
33     public HostedBranch(String name, Hash hash) {
34         this.name = name;
35         this.hash = hash;
36     }
37 
38     public String name() {
39         return name;
40     }
41 
42     public Hash hash() {
43         return hash;
44     }
45 
46     @Override
47     public String toString() {
48         return name + "@" + hash.hex();
49     }
50 
51     @Override
52     public int hashCode() {
53         return Objects.hash(name, hash);
54     }
55 
56     @Override
57     public boolean equals(Object other) {
58         if (!(other instanceof HostedBranch)) {
59             return false;
60         }
61 
62         var o = (HostedBranch) other;
63         return Objects.equals(name, o.name) && Objects.equals(hash, o.hash);
64     }
65 }
