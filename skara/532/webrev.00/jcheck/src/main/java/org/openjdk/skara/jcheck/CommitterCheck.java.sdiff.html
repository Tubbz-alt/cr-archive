 30 import java.util.Iterator;
 31 import java.util.logging.Logger;
 32 
 33 public class CommitterCheck extends CommitCheck {
 34     private final Logger log = Logger.getLogger("org.openjdk.skara.jcheck.committer");
 35     private final Census census;
 36 
 37     CommitterCheck(Census census) {
 38         this.census = census;
 39     }
 40 
 41     private boolean hasRole(Project project, String role, String username, int version) {
 42         switch (role) {
 43             case "lead":
 44                 return project.isLead(username, version);
 45             case "reviewer":
 46                 return project.isReviewer(username, version);
 47             case "committer":
 48                 return project.isCommitter(username, version);
 49             case "author":
 50                 return project.isAuthor(username, version);
 51             case "contributor":
 52                 return census.isContributor(username);
 53             default:
 54                 throw new IllegalStateException("Unsupported role: " + role);
 55         }
 56     }
 57 
 58 
 59     @Override
 60     Iterator<Issue> check(Commit commit, CommitMessage message, JCheckConfiguration conf) {

 61         var project = census.project(conf.general().project());
 62         var version = conf.census().version();
 63         var domain = conf.census().domain();
 64         var role = conf.checks().committer().role();
 65         var metadata = CommitIssue.metadata(commit, message, conf, this);
 66 
 67         var committer = commit.committer();
 68         if (committer.name() == null || committer.name().isEmpty()) {
 69             log.finer("issue: committer.name is null or empty");
 70             return iterator(new CommitterNameIssue(metadata));
 71         }
 72         if (committer.email() == null || !committer.email().endsWith("@" + domain)) {
 73             log.finer("issue: committer.email is null or does not end with @" + domain);
 74             return iterator(new CommitterEmailIssue(domain, metadata));
 75         }
 76 
 77         var username = committer.email().split("@")[0];
 78         var allowedToMerge = conf.checks().committer().allowedToMerge();
 79         if (!commit.isMerge() || !allowedToMerge.contains(username)) {
 80             if (!hasRole(project, role, username, version)) {
 81                 log.finer("issue: committer does not have role " + role);
 82                 return iterator(new CommitterIssue(project, metadata));



 83             }
 84         }
 85 
 86         return iterator();
 87     }
 88 
 89     @Override
 90     public String name() {
 91         return "committer";
 92     }
 93 
 94     @Override
 95     public String description() {
 96         return "Change must contain a proper committer";
 97     }
 98 }

 23 package org.openjdk.skara.jcheck;
 24 
 25 import org.openjdk.skara.census.Census;
 26 import org.openjdk.skara.census.Project;
 27 import org.openjdk.skara.vcs.Commit;
 28 import org.openjdk.skara.vcs.openjdk.CommitMessage;
 29 
 30 import java.util.ArrayList;
 31 import java.util.Iterator;
 32 import java.util.logging.Logger;
 33 
 34 public class CommitterCheck extends CommitCheck {
 35     private final Logger log = Logger.getLogger("org.openjdk.skara.jcheck.committer");
 36     private final Census census;
 37 
 38     CommitterCheck(Census census) {
 39         this.census = census;
 40     }
 41 
 42     private boolean hasRole(Project project, String role, String username, int version) {
 43         switch (role) {
 44             case "lead":
 45                 return project.isLead(username, version);
 46             case "reviewer":
 47                 return project.isReviewer(username, version);
 48             case "committer":
 49                 return project.isCommitter(username, version);
 50             case "author":
 51                 return project.isAuthor(username, version);
 52             case "contributor":
 53                 return census.isContributor(username);
 54             default:
 55                 throw new IllegalStateException("Unsupported role: " + role);
 56         }
 57     }
 58 
 59 
 60     @Override
 61     Iterator<Issue> check(Commit commit, CommitMessage message, JCheckConfiguration conf) {
 62         var issues = new ArrayList<Issue>();
 63         var project = census.project(conf.general().project());
 64         var version = conf.census().version();
 65         var domain = conf.census().domain();
 66         var role = conf.checks().committer().role();
 67         var metadata = CommitIssue.metadata(commit, message, conf, this);
 68 
 69         var committer = commit.committer();
 70         if (committer.name() == null || committer.name().isEmpty()) {
 71             log.finer("issue: committer.name is null or empty");
 72             issues.add(new CommitterNameIssue(metadata));
 73         }
 74         if (committer.email() == null || !committer.email().endsWith("@" + domain)) {
 75             log.finer("issue: committer.email is null or does not end with @" + domain);
 76             issues.add(new CommitterEmailIssue(domain, metadata));
 77         }
 78 
 79         if (committer.name() != null || committer.email() != null) {
 80             var username = committer.email() == null ?
 81                 committer.name() : committer.email().split("@")[0];
 82             var allowedToMerge = conf.checks().committer().allowedToMerge();
 83             if (!commit.isMerge() || !allowedToMerge.contains(username)) {
 84                 if (!hasRole(project, role, username, version)) {
 85                     log.finer("issue: committer does not have role " + role);
 86                     issues.add(new CommitterIssue(project, metadata));
 87                 }
 88             }
 89         }
 90 
 91         return issues.iterator();
 92     }
 93 
 94     @Override
 95     public String name() {
 96         return "committer";
 97     }
 98 
 99     @Override
100     public String description() {
101         return "Change must contain a proper committer";
102     }
103 }
