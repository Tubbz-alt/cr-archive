 23 package org.openjdk.skara.json;
 24 
 25 import java.util.stream.Stream;
 26 import java.util.List;
 27 
 28 public interface JSONValue {
 29     default int asInt() {
 30         throw new IllegalStateException("Unsupported conversion to int");
 31     }
 32 
 33     default long asLong() {
 34         throw new IllegalStateException("Unsupported conversion to long");
 35     }
 36 
 37     default double asDouble() {
 38         throw new IllegalStateException("Unsupported conversion to double");
 39     }
 40 
 41     default String asString() {
 42         throw new IllegalStateException("Unsupported conversion to String");
 43     }
 44 
 45     default boolean asBoolean() {
 46         throw new IllegalStateException("Unsupported conversion to boolean");
 47     }
 48 
 49     default JSONArray asArray() {
 50         throw new IllegalStateException("Unsupported conversion to array");
 51     }
 52 
 53     default JSONObject asObject() {
 54         throw new IllegalStateException("Unsupported conversion to object");
 55     }
 56 
 57     default boolean isInt() {
 58         return false;
 59     }
 60 
 61     default boolean isLong() {
 62         return false;
 63     }
 64 
 65     default boolean isDouble() {
 66         return false;
 67     }
 68 
 69     default boolean isString() {
 70         return false;
 71     }
 72 
 73     default boolean isBoolean() {
 74         return false;
 75     }
 76 
 77     default boolean isArray() {
 78         return false;
 79     }
 80 
 81     default boolean isObject() {
 82         return false;
 83     }
 84 
 85     default boolean isNull() {
 86         return false;
 87     }
 88 
 89     default List<JSONObject.Field> fields() {
 90         return asObject().fields();
 91     }
 92 
 93     default boolean contains(String field) {
 94         return asObject().contains(field);
 95     }
 96 
 97     default JSONValue get(String field) {
 98         return asObject().get(field);
 99     }
100 
101     default JSONValue getOrDefault(String field, JSONValue fallback) {
102         return asObject().getOrDefault(field, fallback);
103     }
104 
105     default JSONValue get(int i) {
106         return asArray().get(i);
107     }
108 
109     default Stream<JSONValue> stream() {
110         return Stream.of(this);
111     }
112 
113     static JSONValue from(int i) {
114         return new JSONNumber(i);
115     }
116 
117     static JSONValue from(long l) {
118         return new JSONNumber(l);
119     }
120 
121     static JSONValue from(double d) {
122         return new JSONDecimal(d);
123     }
124 
125     static JSONValue from(boolean b) {
126         return new JSONBoolean(b);
127     }
128 
129     static JSONValue from(String s) {
130         return new JSONString(s);
131     }
132 
133     static JSONValue fromNull() {
134         return new JSONNull();
135     }
136 }
