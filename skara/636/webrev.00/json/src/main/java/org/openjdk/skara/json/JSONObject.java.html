 23 package org.openjdk.skara.json;
 24 
 25 import java.util.*;
 26 import java.util.stream.Collectors;
 27 
 28 public class JSONObject implements JSONValue {
 29     public static class Field {
 30         private final String name;
 31         private final JSONValue value;
 32 
 33         private Field(String name, JSONValue value) {
 34             this.name = name;
 35             this.value = value;
 36         }
 37 
 38         public String name() {
 39             return name;
 40         }
 41 
 42         public JSONValue value() {
 43             return value;
 44         }
 45     }
 46 
 47     private final Map<String, JSONValue> value;
 48 
 49     public JSONObject() {
 50         this.value = new HashMap<String, JSONValue>();
 51     }
 52 
 53     public JSONObject(Map<String, JSONValue> map) {
 54         this.value = new HashMap<String, JSONValue>(map);
 55     }
 56 
 57     @Override
 58     public boolean isObject() {
 59         return true;
 60     }
 61 
 62     @Override
 63     public JSONObject asObject() {
 64         return this;
 65     }
 66 
 67     public JSONObject put(String k, boolean v) {
 68         value.put(k, JSON.of(v));
 69         return this;
 70     }
 71 
 72     public JSONObject put(String k, int v) {
 73         value.put(k, JSON.of(v));
 74         return this;
 75     }
 76 
 77     public JSONObject put(String k, long v) {
 78         value.put(k, JSON.of(v));
 79         return this;
 80     }
 81 
 82     public JSONObject put(String k, String v) {
 83         value.put(k, JSON.of(v));
 84         return this;
 85     }
 86 
 87     public JSONObject put(String k, double v) {
 88         value.put(k, JSON.of(v));
 89         return this;
 90     }
 91 
 92     public JSONObject put(String k, JSONArray v) {
 93         value.put(k, v);
 94         return this;
 95     }
 96 
 97     public JSONObject put(String k, JSONObject v) {
 98         value.put(k, v);
 99         return this;
100     }
101 
102     public JSONObject put(String k, JSONValue v) {
103         value.put(k, v);
104         return this;
105     }
106 
107     public JSONObject putNull(String k) {
108         value.put(k, JSON.of());
109         return this;
110     }
111 
112     public JSONValue remove(String k) {
113         return value.remove(k);
114     }
115 
116     public JSONValue get(String k) {
117         return value.get(k);
118     }
119 
120     public JSONValue getOrDefault(String k, JSONValue fallback) {
121         return value.getOrDefault(k, fallback);
122     }
123 
124     public List<Field> fields() {
125         return value.entrySet()
126                     .stream()
127                     .map(e -> new Field(e.getKey(), e.getValue()))
128                     .collect(Collectors.toList());
129     }
130 
131     public boolean contains(String field) {
132         return value.containsKey(field);
133     }
134 
135     public Set<String> keys() {
136         return value.keySet();
137     }
138 
139     @Override
140     public String toString() {
141         var builder = new StringBuilder();
142         builder.append("{");
143         for (var key : value.keySet()) {
144             builder.append("\"");
145             builder.append(key);
146             builder.append("\":");
147             builder.append(value.get(key).toString());
148             builder.append(",");
149         }
150 
151         var end = builder.length() - 1;
152         if (builder.charAt(end) == ',') {
153             builder.deleteCharAt(end);
154         }
155 
156         builder.append("}");
157         return builder.toString();
158     }
159 }
