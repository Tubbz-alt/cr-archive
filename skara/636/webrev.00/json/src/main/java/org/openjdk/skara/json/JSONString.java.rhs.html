23 package org.openjdk.skara.json;
24 
25 public class JSONString implements JSONValue {
26     private String value;
27 
28     public JSONString(String value) {
29         this.value = value;
30     }
31 
32     @Override
33     public boolean isString() {
34         return true;
35     }
36 
37     @Override
38     public String asString() {
39         return value;
40     }
41 
42     @Override
43     public String toString() {
44         var builder = new StringBuilder();
45         builder.append("\"");
46 
47         for (var i = 0; i < value.length(); i++) {
48             var c = value.charAt(i);
49 
50             switch (c) {
51                 case '"':
52                     builder.append("\\\"");
53                     break;
54                 case '\\':
55                     builder.append("\\\\");
56                     break;
57                 case '/':
58                     builder.append("\\/");
59                     break;
60                 case '\b':
61                     builder.append("\\b");
62                     break;
63                 case '\f':
64                     builder.append("\\f");
65                     break;
66                 case '\n':
67                     builder.append("\\n");
68                     break;
69                 case '\r':
70                     builder.append("\\r");
71                     break;
72                 case '\t':
73                     builder.append("\\t");
74                     break;
75                 default:
76                     builder.append(c);
77                     break;
78             }
79         }
80 
81         builder.append("\"");
82         return builder.toString();
83     }
84 }
