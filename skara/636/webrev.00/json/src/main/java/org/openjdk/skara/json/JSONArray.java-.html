 23 package org.openjdk.skara.json;
 24 
 25 import java.util.*;
 26 import java.util.stream.Stream;
 27 
 28 public class JSONArray implements JSONValue, Iterable<JSONValue> {
 29     private final List<JSONValue> values;
 30 
 31     public JSONArray() {
 32         this.values = new ArrayList<JSONValue>();
 33     }
 34 
 35     public JSONArray(JSONValue[] array) {
 36         this.values = new ArrayList<JSONValue>(array.length);
 37         for (var v : array) {
 38             values.add(v);
 39         }
 40     }
 41 
 42     public JSONArray(List<JSONValue> values) {
 43         this.values = new ArrayList<JSONValue>(values);
 44     }
 45 
 46     @Override
 47     public boolean isArray() {
 48         return true;
 49     }
 50 
 51     @Override
 52     public JSONArray asArray() {
 53         return this;
 54     }
 55 
 56     public JSONArray set(int i, boolean value) {
 57         values.set(i, JSON.of(value));
 58         return this;
 59     }
 60 
 61     public JSONArray set(int i, int value) {
 62         values.set(i, JSON.of(value));
 63         return this;
 64     }
 65 
 66     public JSONArray set(int i, long value) {
 67         values.set(i, JSON.of(value));
 68         return this;
 69     }
 70 
 71     public JSONArray set(int i, String value) {
 72         values.set(i, JSON.of(value));
 73         return this;
 74     }
 75 
 76     public JSONArray set(int i, double value) {
 77         values.set(i, JSON.of(value));
 78         return this;
 79     }
 80 
 81     public JSONArray set(int i, JSONValue value) {
 82         values.set(i, value);
 83         return this;
 84     }
 85 
 86     public JSONArray setNull(int i) {
 87         values.set(i, JSON.of());
 88         return this;
 89     }
 90 
 91     public JSONArray add(boolean value) {
 92         values.add(JSON.of(value));
 93         return this;
 94     }
 95 
 96     public JSONArray add(int value) {
 97         values.add(JSON.of(value));
 98         return this;
 99     }
100 
101     public JSONArray add(long value) {
102         values.add(JSON.of(value));
103         return this;
104     }
105 
106     public JSONArray add(String value) {
107         values.add(JSON.of(value));
108         return this;
109     }
110 
111     public JSONArray add(double value) {
112         values.add(JSON.of(value));
113         return this;
114     }
115 
116     public JSONArray add(JSONValue value) {
117         values.add(value);
118         return this;
119     }
120 
121     public JSONArray addNull() {
122         values.add(JSON.of());
123         return this;
124     }
125 
126     public JSONValue get(int i) {
127         return values.get(i);
128     }
129 
130     public int size() {
131         return values.size();
132     }
133 
134     @Override
135     public String toString() {
136         var builder = new StringBuilder();
137 
138         builder.append("[");
139         for (var i = 0; i < size(); i++) {
140             builder.append(get(i).toString());
141             if (i != (size() - 1)) {
142                 builder.append(",");
143             }
144         }
145         builder.append("]");
146         return builder.toString();
147     }
148 
149     @Override
150     public Stream<JSONValue> stream() {
151         return values.stream();
152     }
153 
154     @Override
155     public Iterator<JSONValue> iterator() {
156         return values.iterator();
157     }
158 }
