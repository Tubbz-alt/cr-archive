 23 package org.openjdk.skara.vcs.git;
 24 
 25 import org.openjdk.skara.vcs.*;
 26 import org.openjdk.skara.vcs.tools.*;
 27 
 28 import java.nio.charset.StandardCharsets;
 29 import java.util.concurrent.TimeUnit;
 30 import java.io.*;
 31 import java.util.*;
 32 import java.nio.file.Path;
 33 
 34 class GitCommits implements Commits, AutoCloseable {
 35     private static final String COMMIT_DELIMITER = "#@!_-=&";
 36 
 37     private final Path dir;
 38     private final String range;
 39     private final boolean reverse;
 40     private final int num;
 41     private final String format;
 42 
 43     private final List<Process> processes = new ArrayList<Process>();
 44     private final List<List<String>> commands = new ArrayList<List<String>>();
 45     private boolean closed = false;
 46 
 47     public GitCommits(Path dir, String range, boolean reverse, int num) throws IOException {
 48         this.dir = dir;
 49         this.range = range;
 50         this.reverse = reverse;
 51         this.num = num;
 52         this.format = String.join("%n",
 53                                   COMMIT_DELIMITER,
 54                                   GitCommitMetadata.FORMAT);
 55 
 56     }
 57 
 58     @Override
 59     public Iterator<Commit> iterator() {
 60         var cmd = new ArrayList<String>();
 61         cmd.addAll(List.of("git", "log", "--format=" + format,
 62                                          "--patch",
 63                                          "--find-renames=99%",
 64                                          "--find-copies=99%",
 65                                          "--find-copies-harder",
 66                                          "--topo-order",
 67                                          "--binary",
 68                                          "-c",

 69                                          "--raw",
 70                                          "--no-abbrev",
 71                                          "--unified=0",
 72                                          "--no-color"));
 73         if (reverse) {
 74             cmd.add("--reverse");
 75         }
 76         if (num > 0) {
 77             cmd.add("-n");
 78             cmd.add(Integer.toString(num));
 79         }
 80         cmd.add(range);
 81         var pb = new ProcessBuilder(cmd);
 82         pb.directory(dir.toFile());
 83         var command = pb.command();
 84         try {
 85             var p = pb.start();
 86             processes.add(p);
 87             commands.add(command);
 88             var reader = new UnixStreamReader(p.getInputStream());
 89 
 90             return new GitCommitIterator(reader, COMMIT_DELIMITER);
 91         } catch (IOException e) {
 92             throw new UncheckedIOException(e);
 93         }
 94     }
 95 
 96     @Override
 97     public void close() throws IOException {
 98         synchronized (this) {
 99             if (!closed) {
100                 closed = true;
101             } else {
102                 return;
103             }
104         }
105 
106         for (var i = 0; i < processes.size(); i++) {
107             var p = processes.get(i);
108             var command = commands.get(i);
109             try {
110                 var exited = p.waitFor(30L, TimeUnit.SECONDS);
111                 if (!exited) {
112                     throw new IOException("'" + String.join(" ", command) + "' timed out");
113                 }
114                 var exitCode = p.exitValue();
115                 if (exitCode == 128) {
116                     var stderr = new BufferedReader(new InputStreamReader(p.getErrorStream(), StandardCharsets.UTF_8));
117                     var message = stderr.readLine();
118                     if (message.equals("fatal: bad default revision 'HEAD'")) {
119                         // this is an empty repository, this is not an error case
120                     } else {
121                         throw new IOException("'" + String.join(" ", command) + "' exited with code: " + exitCode);
122                     }
123                 }
124             } catch (InterruptedException e) {
125                 throw new IOException("'" + String.join(" ", command) + "' was interrupted", e);
126             }
127         }
128     }
129 }
