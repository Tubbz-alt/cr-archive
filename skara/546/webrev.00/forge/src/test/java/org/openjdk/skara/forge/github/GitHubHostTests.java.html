23 package org.openjdk.skara.forge.github;
24 
25 import org.junit.jupiter.api.Test;
26 import org.openjdk.skara.network.URIBuilder;
27 import org.openjdk.skara.test.TemporaryDirectory;
28 
29 import java.io.IOException;
30 import java.net.*;
31 import java.util.regex.Pattern;
32 
33 import static org.junit.jupiter.api.Assertions.assertEquals;
34 
35 class GitHubHostTests {
36     @Test
37     void webUriPatternReplacement() throws IOException, URISyntaxException {
38         try (var tempFolder = new TemporaryDirectory()) {
39             var host = new GitHubHost(URIBuilder.base("http://www.example.com").build(),
40                                       Pattern.compile("^(http://www.example.com)/test/(.*)$"), "$1/another/$2");
41             assertEquals(new URI("http://www.example.com/another/hello"), host.getWebURI("/test/hello"));
42         }
43     }
44 
45     @Test
46     void nonTransformedWebUrl() throws IOException, URISyntaxException {
47         try (var tempFolder = new TemporaryDirectory()) {
48             var host = new GitHubHost(URIBuilder.base("http://www.example.com").build(),
49                                       Pattern.compile("^(http://www.example.com)/test/(.*)$"), "$1/another/$2");
50             assertEquals(new URI("http://www.example.com/another/hello"), host.getWebURI("/test/hello"));
51             assertEquals(new URI("http://www.example.com/test/hello"), host.getWebURI("/test/hello", false));
52         }
53     }
54 }
