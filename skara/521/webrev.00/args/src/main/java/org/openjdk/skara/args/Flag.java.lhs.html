 23 package org.openjdk.skara.args;
 24 
 25 import java.util.Objects;
 26 
 27 public class Flag {
 28     private boolean isSwitch;
 29     private final String shortcut;
 30     private final String fullname;
 31     private final String description;
 32     private final String helptext;
 33     private final boolean isRequired;
 34 
 35     Flag(boolean isSwitch, String shortcut, String fullname, String description, String helptext, boolean isRequired) {
 36         this.isSwitch = isSwitch;
 37         this.shortcut = shortcut;
 38         this.fullname = fullname;
 39         this.description = description;
 40         this.helptext = helptext;
 41         this.isRequired = isRequired;
 42     }
 43 
 44     boolean isSwitch() {
 45         return isSwitch;
 46     }
 47 
 48     String fullname() {
 49         return fullname;
 50     }
 51 
 52     String shortcut() {
 53         return shortcut;
 54     }
 55 
 56     String description() {
 57         return description;
 58     }
 59 
 60     String helptext() {
 61         return helptext;
 62     }
 63 
 64     boolean isRequired() {
 65         return isRequired;
 66     }
 67 
 68     @Override
 69     public boolean equals(Object o) {
 70         if (!(o instanceof Flag)) {
 71             return false;
 72         }
 73 
 74         Flag other = (Flag) o;
 75         return Objects.equals(isSwitch, other.isSwitch) &&
 76                Objects.equals(shortcut, other.shortcut) &&
 77                Objects.equals(fullname, other.fullname) &&
 78                Objects.equals(helptext, other.helptext) &&
 79                Objects.equals(isRequired, other.isRequired);
 80     }
 81 
 82     @Override
 83     public int hashCode() {
 84         return Objects.hash(isSwitch,
 85                             shortcut,
 86                             fullname,
 87                             helptext,
 88                             isRequired);
 89     }
 90 
 91     @Override
 92     public String toString() {
 93         if (shortcut.equals("")) {
 94             return "--" + fullname;
 95         }
 96 
 97         if (fullname.equals("")) {
 98             return "-" + shortcut;
 99         }
100 
101         return "-" + shortcut + ", --" + fullname;
102     }
103 }
