 23 package org.openjdk.skara.issuetracker;
 24 
 25 import org.openjdk.skara.host.HostUser;
 26 import org.openjdk.skara.json.JSONValue;
 27 
 28 import java.net.URI;
 29 import java.time.ZonedDateTime;
 30 import java.util.*;
 31 
 32 public interface Issue {
 33     /**
 34      * Project containing the issue.
 35      * @return
 36      */
 37     IssueProject project();
 38 
 39     /**
 40      * The repository-specific identifier.
 41      * @return
 42      */
 43     String id();
 44 
 45     /**
 46      * The host-specific author name.
 47      * @return
 48      */
 49     HostUser author();
 50 
 51     /**
 52      * Title of the request.
 53      * @return
 54      */
 55     String title();
 56 
 57     /**
 58      * Update the title of the request.
 59      * @param title
 60      */
 61     void setTitle(String title);
 62 
 63     /**
 64      * The main body of the request.
 65      * @return
 66      */
 67     String body();
 68 
 69     /**
 70      * Update the main body of the request.
 71      * @param body
 72      */
 73     void setBody(String body);
 74 
 75     /**
 76      * All comments on the issue, in ascending creation time order.
 77      * @return
 78      */
 79     List<Comment> comments();
 80 
 81     /**
 82      * Posts a new comment.
 83      * @param body
 84      */
 85     Comment addComment(String body);
 86 
 87     /**
 88      * Updates an existing comment.
 89      * @param id
 90      * @param body
 91      */
 92     Comment updateComment(String id, String body);
 93 
 94     /**
 95      * When the request was created.
 96      * @return
 97      */
 98     ZonedDateTime createdAt();
 99 
100     /**
101      * When the request was last updated.
102      * @return
103      */
104     ZonedDateTime updatedAt();
105 
106     enum State {
107         OPEN,
108         RESOLVED,
109         CLOSED
110     }
111 
112     /**
113      * Returns the current state.
114      * @return
115      */
116     State state();
117 
118     /**
119      * Set the state.
120      * @param state Desired state
121      */
122     void setState(State state);
123 
124     /**
125      * Adds the given label.
126      * @param label
127      */
128     void addLabel(String label);
129 
130     /**
131      * Removes the given label.
132      * @param label
133      */
134     void removeLabel(String label);
135 
136     /**
137      * Retrieves all the currently set labels.
138      * @return
139      */
140     List<String> labels();
141 
142     /**
143      * Returns a link that will lead to the issue.
144      */
145     URI webUrl();
146 







147     /**
148      * Returns all usernames assigned to the issue.
149      */
150     List<HostUser> assignees();
151 
152     /**
153      * Update the list of assignees.
154      * @param assignees
155      */
156     void setAssignees(List<HostUser> assignees);
157 
158     List<Link> links();
159 
160     void addLink(Link link);
161 
162     void removeLink(Link link);
163 
164     Map<String, JSONValue> properties();
165 
166     void setProperty(String name, JSONValue value);
167 
168     void removeProperty(String name);
169 }
