 23 package org.openjdk.skara.bots.notify;
 24 
 25 import org.openjdk.skara.bot.*;
 26 import org.openjdk.skara.email.EmailAddress;
 27 import org.openjdk.skara.json.*;
 28 import org.openjdk.skara.mailinglist.MailingListServerFactory;
 29 import org.openjdk.skara.network.URIBuilder;
 30 import org.openjdk.skara.storage.StorageBuilder;
 31 import org.openjdk.skara.vcs.Tag;
 32 
 33 import java.net.URI;
 34 import java.nio.file.Path;
 35 import java.time.Duration;
 36 import java.util.*;
 37 import java.util.logging.Logger;
 38 import java.util.regex.Pattern;
 39 import java.util.stream.Collectors;
 40 
 41 public class NotifyBotFactory implements BotFactory {
 42     private final Logger log = Logger.getLogger("org.openjdk.skara.bots");;
 43 
 44     @Override
 45     public String name() {
 46         return "notify";
 47     }
 48 
 49     @Override
 50     public List<Bot> create(BotConfiguration configuration) {
 51         var ret = new ArrayList<Bot>();
 52         var specific = configuration.specific();
 53 
 54         var database = specific.get("database").asObject();
 55         var databaseRepo = configuration.repository(database.get("repository").asString());
 56         var databaseRef = configuration.repositoryRef(database.get("repository").asString());
 57         var databaseName = database.get("name").asString();
 58         var databaseEmail = database.get("email").asString();
 59 
 60         var readyLabels = specific.get("ready").get("labels").stream()
 61                                   .map(JSONValue::asString)
 62                                   .collect(Collectors.toSet());
 63         var readyComments = specific.get("ready").get("comments").stream()
 64                                     .map(JSONValue::asObject)
 65                                     .collect(Collectors.toMap(obj -> obj.get("user").asString(),
 66                                                               obj -> Pattern.compile(obj.get("pattern").asString())));
 67 
 68         URI reviewIcon = null;
 69         if (specific.contains("reviews")) {
 70             if (specific.get("reviews").contains("icon")) {
 71                 reviewIcon = URI.create(specific.get("reviews").get("icon").asString());
 72             }
 73         }
 74         URI commitIcon = null;
 75         if (specific.contains("commits")) {
 76             if (specific.get("commits").contains("icon")) {
 77                 commitIcon = URI.create(specific.get("commits").get("icon").asString());
 78             }
 79         }
 80         for (var repo : specific.get("repositories").fields()) {
 81             var repoName = repo.name();
 82             var branchPattern = Pattern.compile("^master$");
 83             if (repo.value().contains("branches")) {
 84                 branchPattern = Pattern.compile(repo.value().get("branches").asString());
 85             }
 86 
 87             var updaters = new ArrayList<RepositoryUpdateConsumer>();
 88             var prUpdaters = new ArrayList<PullRequestUpdateConsumer>();
 89             if (repo.value().contains("json")) {
 90                 var folder = repo.value().get("folder").asString();
 91                 var build = repo.value().get("build").asString();
 92                 var version = repo.value().get("version").asString();
 93                 updaters.add(new JsonUpdater(Path.of(folder), version, build));
 94             }
 95 
 96             if (repo.value().contains("slack")) {
 97                 var slackConf = repo.value().get("slack");
 98                 URI prWebhook = null;
 99                 if (slackConf.contains("pr")) {
100                     prWebhook = URIBuilder.base(slackConf.get("pr").asString()).build();
101                 }
102                 URI commitWebhook = null;
103                 if (slackConf.contains("commit")) {
104                     commitWebhook = URIBuilder.base(slackConf.get("commit").asString()).build();
105                 }
106                 var username = slackConf.get("username").asString();
107                 var updater = new SlackUpdater(prWebhook, commitWebhook, username);
108                 updaters.add(updater);
109                 prUpdaters.add(updater);
110             }
111 
112             if (repo.value().contains("mailinglists")) {
113                 var email = specific.get("email").asObject();
114                 var smtp = email.get("smtp").asString();
115                 var sender = EmailAddress.parse(email.get("sender").asString());
116                 var archive = URIBuilder.base(email.get("archive").asString()).build();
117                 var interval = email.contains("interval") ? Duration.parse(email.get("interval").asString()) : Duration.ofSeconds(1);
118                 var listServer = MailingListServerFactory.createMailmanServer(archive, smtp, interval);
119 
120                 for (var mailinglist : repo.value().get("mailinglists").asArray()) {
121                     var recipient = mailinglist.get("recipient").asString();
122                     var recipientAddress = EmailAddress.parse(recipient);
123 
124                     var author = mailinglist.contains("author") ? EmailAddress.parse(mailinglist.get("author").asString()) : null;
125                     var allowedDomains = author == null ? Pattern.compile(mailinglist.get("domains").asString()) : null;
126 
127                     var mailingListUpdaterBuilder = MailingListUpdater.newBuilder()
128                                                                       .list(listServer.getList(recipient))
129                                                                       .recipient(recipientAddress)
130                                                                       .sender(sender)
131                                                                       .author(author)
132                                                                       .allowedAuthorDomains(allowedDomains);
133 
134                     if (mailinglist.contains("mode")) {
135                         var mode = MailingListUpdater.Mode.ALL;
136                         switch (mailinglist.get("mode").asString()) {
137                             case "pr":
138                                 mode = MailingListUpdater.Mode.PR;
139                                 break;
140                             case "pr-only":
141                                 mode = MailingListUpdater.Mode.PR_ONLY;
142                                 break;
143                             default:
144                                 throw new RuntimeException("Unknown mode");
145                         }
146                         mailingListUpdaterBuilder.mode(mode);
147                     }
148                     if (mailinglist.contains("headers")) {
149                         mailingListUpdaterBuilder.headers(mailinglist.get("headers").fields().stream()
150                                                                      .collect(Collectors.toMap(JSONObject.Field::name,
151                                                                                                field -> field.value().asString())));
152                     }
153                     if (mailinglist.contains("branchnames")) {
154                         mailingListUpdaterBuilder.includeBranch(mailinglist.get("branchnames").asBoolean());
155                     }
156                     if (mailinglist.contains("tags")) {
157                         mailingListUpdaterBuilder.reportNewTags(mailinglist.get("tags").asBoolean());
158                     }
159                     if (mailinglist.contains("branches")) {
160                         mailingListUpdaterBuilder.reportNewBranches(mailinglist.get("branches").asBoolean());
161                     }
162                     if (mailinglist.contains("builds")) {
163                         mailingListUpdaterBuilder.reportNewBuilds(mailinglist.get("builds").asBoolean());
164                     }
165                     if (mailinglist.contains("reponame")) {
166                         mailingListUpdaterBuilder.repoInSubject(mailinglist.get("reponame").asBoolean());
167                     }
168                     if (mailinglist.contains("branchname")) {
169                         mailingListUpdaterBuilder.branchInSubject(Pattern.compile(mailinglist.get("branchname").asString()));
170                     }
171 
172                     updaters.add(mailingListUpdaterBuilder.build());
173                 }
174             }
175             if (repo.value().contains("issues")) {
176                 var issuesConf = repo.value().get("issues");
177                 var issueProject = configuration.issueProject(issuesConf.get("project").asString());
178                 var issueUpdaterBuilder = IssueUpdater.newBuilder()
179                                                       .issueProject(issueProject);
180 
181                 if (issuesConf.contains("reviewlink")) {
182                     issueUpdaterBuilder.reviewLink(issuesConf.get("reviewlink").asBoolean());
183                 }
184                 if (issuesConf.contains("commitlink")) {
185                     issueUpdaterBuilder.commitLink(issuesConf.get("commitlink").asBoolean());
186                 }
187                 if (issuesConf.contains("fixversions")) {
188                     issueUpdaterBuilder.setFixVersion(true);
189                     issueUpdaterBuilder.fixVersions(issuesConf.get("fixversions").fields().stream()
190                                                               .collect(Collectors.toMap(JSONObject.Field::name,
191                                                                                         f -> f.value().asString())));
192                 }
193                 if (issuesConf.contains("pronly")) {
194                     issueUpdaterBuilder.prOnly(issuesConf.get("pronly").asBoolean());
195                 }
196                 updaters.add(issueUpdaterBuilder.build());
197                 prUpdaters.add(issueUpdaterBuilder.build());
198             }
199 
200             if (updaters.isEmpty()) {
201                 log.warning("No consumers configured for notify bot repository: " + repoName);
202                 continue;
203             }
204 
205             var baseName = repo.value().contains("basename") ? repo.value().get("basename").asString() : configuration.repositoryName(repoName);
206 
207             var tagStorageBuilder = new StorageBuilder<UpdatedTag>(baseName + ".tags.txt")
208                     .remoteRepository(databaseRepo, databaseRef, databaseName, databaseEmail, "Added tag for " + repoName);
209             var branchStorageBuilder = new StorageBuilder<UpdatedBranch>(baseName + ".branches.txt")
210                     .remoteRepository(databaseRepo, databaseRef, databaseName, databaseEmail, "Added branch hash for " + repoName);
211             var issueStorageBuilder = new StorageBuilder<PullRequestIssues>(baseName + ".prissues.txt")
212                     .remoteRepository(databaseRepo, databaseRef, databaseName, databaseEmail, "Added pull request issue info for " + repoName);
213             var bot = NotifyBot.newBuilder()
214                                .repository(configuration.repository(repoName))
215                                .storagePath(configuration.storageFolder())
216                                .branches(branchPattern)
217                                .tagStorageBuilder(tagStorageBuilder)
218                                .branchStorageBuilder(branchStorageBuilder)
219                                .prIssuesStorageBuilder(issueStorageBuilder)
220                                .updaters(updaters)
221                                .prUpdaters(prUpdaters)
222                                .readyLabels(readyLabels)
223                                .readyComments(readyComments)
224                                .build();
225             ret.add(bot);
226         }
227 
228         return ret;
229     }
230 }
