23 package org.openjdk.skara.webrev;
24 
25 import java.io.IOException;
26 import java.io.Writer;
27 import java.nio.file.Files;
28 import java.nio.file.Path;
29 import java.util.List;
30 
31 class RawView implements View {
32     private final Path out;
33     private final Path file;
34     private final List<String> text;
35     private final byte[] binary;
36 
37     public RawView(Path out, Path file, List<String> text) {
38         this.out = out;
39         this.file = file;
40         this.text = text;
41         this.binary = null;
42     }
43 
44     public RawView(Path out, Path file, byte[] binary) {
45         this.out = out;
46         this.file = file;
47         this.binary = binary;
48         this.text = null;
49     }
50 
51     public void render(Writer w) throws IOException {
52         var rawFile = out.resolve(file.toString());
53         if (Files.exists(rawFile)) {
54             // If the raw file collides with a file generated by the webrev (such as index.html), rename it
55             rawFile = out.resolve("_" + file.toString());
56         } else {
57             Files.createDirectories(rawFile.getParent());
58         }
59 
60         if (binary != null) {
61             Files.write(rawFile, binary);
62         } else {
63             Files.write(rawFile, text);
64         }
65 
66         w.write("<a href=\"");
67         w.write(Webrev.relativeToIndex(out, rawFile));
68         w.write("\">Raw</a>\n");
69     }
70 }
