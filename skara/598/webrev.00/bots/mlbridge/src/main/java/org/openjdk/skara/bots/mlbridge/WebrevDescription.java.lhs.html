23 package org.openjdk.skara.bots.mlbridge;
24 
25 import java.net.URI;
26 
27 public class WebrevDescription {
28     public enum Type {
29         FULL,
30         INCREMENTAL,
31         MERGE_TARGET,
32         MERGE_SOURCE

33     }
34 
35     private final URI uri;
36     private final Type type;
37     private final String description;
38 
39     public WebrevDescription(URI uri, Type type, String description) {
40         this.uri = uri;
41         this.type = type;
42         this.description = description;
43     }
44 
45     public WebrevDescription(URI uri, Type type) {
46         this.uri = uri;
47         this.type = type;
48         this.description = null;
49     }
50 




51     public URI uri() {
52         return uri;
53     }
54 
55     public String label() {
56         switch (type) {
57             case FULL:
58                 return "Full";
59             case INCREMENTAL:
60                 return "Incremental";
61             case MERGE_TARGET:
62                 return "Merge target" + (description != null ? " (" + description + ")" : "");
63             case MERGE_SOURCE:
64                 return "Merge source" + (description != null ? " (" + description + ")" : "");


65 
66         }
67         throw new RuntimeException("Unknown type");
68     }
69 
70     public String shortLabel() {
71         switch (type) {
72             case FULL:
73                 return "full";
74             case INCREMENTAL:
75                 return "incr";
76             case MERGE_TARGET:
77                 return description != null ? description : "merge target";
78             case MERGE_SOURCE:
79                 return description != null ? description : "merge source";


80 
81         }
82         throw new RuntimeException("Unknown type");
83     }
84 }
