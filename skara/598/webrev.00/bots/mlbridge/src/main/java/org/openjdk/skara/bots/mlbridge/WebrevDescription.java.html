23 package org.openjdk.skara.bots.mlbridge;
24 
25 import java.net.URI;
26 
27 public class WebrevDescription {
28     public enum Type {
29         FULL,
30         INCREMENTAL,
31         MERGE_TARGET,
32         MERGE_SOURCE,
33         MERGE_CONFLICT
34     }
35 
36     private final URI uri;
37     private final Type type;
38     private final String description;
39 
40     public WebrevDescription(URI uri, Type type, String description) {
41         this.uri = uri;
42         this.type = type;
43         this.description = description;
44     }
45 
46     public WebrevDescription(URI uri, Type type) {
47         this.uri = uri;
48         this.type = type;
49         this.description = null;
50     }
51 
52     public Type type() {
53         return type;
54     }
55 
56     public URI uri() {
57         return uri;
58     }
59 
60     public String label() {
61         switch (type) {
62             case FULL:
63                 return "Full";
64             case INCREMENTAL:
65                 return "Incremental";
66             case MERGE_TARGET:
67                 return "Merge target" + (description != null ? " (" + description + ")" : "");
68             case MERGE_SOURCE:
69                 return "Merge source" + (description != null ? " (" + description + ")" : "");
70             case MERGE_CONFLICT:
71                 return "Merge conflicts" + (description != null ? " (" + description + ")" : "");
72 
73         }
74         throw new RuntimeException("Unknown type");
75     }
76 
77     public String shortLabel() {
78         switch (type) {
79             case FULL:
80                 return "full";
81             case INCREMENTAL:
82                 return "incr";
83             case MERGE_TARGET:
84                 return description != null ? description : "merge target";
85             case MERGE_SOURCE:
86                 return description != null ? description : "merge source";
87             case MERGE_CONFLICT:
88                 return "merge conflicts";
89 
90         }
91         throw new RuntimeException("Unknown type");
92     }
93 }
