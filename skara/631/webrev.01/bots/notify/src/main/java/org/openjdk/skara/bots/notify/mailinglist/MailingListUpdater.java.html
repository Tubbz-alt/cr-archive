 77     private String tagAnnotationToText(HostedRepository repository, Tag.Annotated annotation) {
 78         var writer = new StringWriter();
 79         var printer = new PrintWriter(writer);
 80 
 81         printer.println("Tagged by: " + annotation.author().name() + " <" + annotation.author().email() + ">");
 82         printer.println("Date:      " + annotation.date().format(DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss +0000")));
 83         printer.println();
 84         printer.print(String.join("\n", annotation.message()));
 85 
 86         return writer.toString();
 87     }
 88 
 89     private EmailAddress filteredAuthor(EmailAddress commitAddress) {
 90         if (author != null) {
 91             return author;
 92         }
 93         var allowedAuthorMatcher = allowedAuthorDomains.matcher(commitAddress.domain());
 94         if (!allowedAuthorMatcher.matches()) {
 95             return sender;
 96         } else {
 97             return commitAddress;
 98         }
 99     }
100 
101     private EmailAddress commitToAuthor(Commit commit) {
102         return filteredAuthor(EmailAddress.from(commit.committer().name(), commit.committer().email()));
103     }
104 
105     private EmailAddress annotationToAuthor(Tag.Annotated annotation) {
106         return filteredAuthor(EmailAddress.from(annotation.author().name(), annotation.author().email()));
107     }

109     private String commitsToSubject(HostedRepository repository, List<Commit> commits, Branch branch) {
110         var subject = new StringBuilder();
111         subject.append(repository.repositoryType().shortName());
112         subject.append(": ");
113         subject.append(repository.name());
114         subject.append(": ");
115         if (includeBranch) {
116             subject.append(branch.name());
117             subject.append(": ");
118         }
119         if (commits.size() > 1) {
120             subject.append(commits.size());
121             subject.append(" new changesets");
122         } else {
123             subject.append(commits.get(0).message().get(0));
124         }
125         return subject.toString();
126     }
127 
128     private String tagToSubject(HostedRepository repository, Hash hash, Tag tag) {
129         return repository.repositoryType().shortName() +
130                 ": " +
131                 repository.name() +
132                 ": Added tag " +
133                 tag +
134                 " for changeset " +
135                 hash.abbreviate();
136     }

138     private List<Commit> filterPrCommits(HostedRepository repository, Repository localRepository, List<Commit> commits, Branch branch) throws NonRetriableException {
139         var ret = new ArrayList<Commit>();
140         var mergedHashes = new HashSet<Hash>();
141 
142         for (var commit : commits) {
143             if (mergedHashes.contains(commit.hash())) {
144                 log.info("Commit " + commit.hash() + " belongs to a merge PR - skipping");
145                 continue;
146             }
147 
148             var candidates = repository.findPullRequestsWithComment(null, "Pushed as commit " + commit.hash() + ".");
149             if (candidates.size() != 1) {
150                 log.warning("Commit " + commit.hash() + " matches " + candidates.size() + " pull requests - expected 1");
151                 ret.add(commit);
152                 continue;
153             }
154 
155             var candidate = candidates.get(0);
156             var prLink = candidate.webUrl();
157             if (!candidate.targetRef().equals(branch.name())) {
158                 log.info("Pull request " + prLink + " targets " + candidate.targetRef() + " - commit is on " + branch.toString() + " - skipping");
159                 ret.add(commit);
160                 continue;
161             }
162 
163             // For a merge PR, many other of these commits could belong here as well
164             try {
165                 localRepository.fetch(repository.url(), candidate.fetchRef());
166                 var baseHash = PullRequestUtils.baseHash(candidate, localRepository);
167                 var prCommits = localRepository.commitMetadata(baseHash, candidate.headHash());
168                 prCommits.forEach(prCommit -> mergedHashes.add(prCommit.hash()));
169             } catch (IOException e) {
170                 log.warning("Could not fetch commits from " + prLink + " - cannot see if the belong to the PR");
171             }
172         }
173 
174         return ret;
175     }

177     private void sendCombinedCommits(HostedRepository repository, List<Commit> commits, Branch branch) throws NonRetriableException {
178         if (commits.size() == 0) {
179             return;
180         }
181 
182         var writer = new StringWriter();
183         var printer = new PrintWriter(writer);
184 
185         for (var commit : commits) {
186             printer.println(CommitFormatters.toText(repository, commit));
187         }
188 
189         var subject = commitsToSubject(repository, commits, branch);
190         var lastCommit = commits.get(commits.size() - 1);
191         var commitAddress = filteredAuthor(EmailAddress.from(lastCommit.committer().name(), lastCommit.committer().email()));
192         var email = Email.create(subject, writer.toString())
193                          .sender(sender)
194                          .author(commitAddress)
195                          .recipient(recipient)
196                          .headers(headers)
197                          .headers(commitHeaders(repository, commits))
198                          .build();
199 
200         try {
201             list.post(email);
202         } catch (RuntimeException e) {
203             throw new NonRetriableException(e);
204         }
205     }

207     private Map<String, String> commitHeaders(HostedRepository repository, List<Commit> commits) {
208         var ret = new HashMap<String, String>();
209         ret.put("X-Git-URL", repository.webUrl().toString());
210         if (!commits.isEmpty()) {
211             ret.put("X-Git-Changeset", commits.get(0).hash().hex());
212         }
213         return ret;
214     }

216     @Override
217     public void handleCommits(HostedRepository repository, Repository localRepository, List<Commit> commits, Branch branch) throws NonRetriableException {
218         if (mode == Mode.PR) {
219             commits = filterPrCommits(repository, localRepository, commits, branch);
220         }
221         sendCombinedCommits(repository, commits, branch);
222     }

224     @Override
225     public void handleOpenJDKTagCommits(HostedRepository repository, Repository localRepository, List<Commit> commits, OpenJDKTag tag, Tag.Annotated annotation) throws NonRetriableException {
226         if (!reportNewTags) {
227             return;
228         }
229         if (!reportNewBuilds) {
230             handleTagCommit(repository, localRepository, commits.get(commits.size() - 1), tag.tag(), annotation);
231             return;
232         }
233         var writer = new StringWriter();
234         var printer = new PrintWriter(writer);
235 
236         var taggedCommit = commits.get(commits.size() - 1);
237         if (annotation != null) {
238             printer.println(tagAnnotationToText(repository, annotation));
239         }
240         printer.println(CommitFormatters.toTextBrief(repository, taggedCommit));
241 
242         printer.println("The following commits are included in " + tag.tag());
243         printer.println("========================================================");
244         for (var commit : commits) {
245             printer.print(commit.hash().abbreviate());
246             if (commit.message().size() > 0) {
247                 printer.print(": " + commit.message().get(0));
248             }
249             printer.println();
250         }
251 
252         var subject = tagToSubject(repository, taggedCommit.hash(), tag.tag());
253         var email = Email.create(subject, writer.toString())
254                          .sender(sender)
255                          .recipient(recipient)
256                          .headers(headers)
257                          .headers(commitHeaders(repository, commits));
258 
259         if (annotation != null) {
260             email.author(annotationToAuthor(annotation));
261         } else {
262             email.author(commitToAuthor(taggedCommit));
263         }
264 
265         try {
266             list.post(email.build());
267         } catch (RuntimeException e) {
268             throw new NonRetriableException(e);
269         }
270     }
271 
272     @Override
273     public void handleTagCommit(HostedRepository repository, Repository localRepository, Commit commit, Tag tag, Tag.Annotated annotation) throws NonRetriableException {
274         if (!reportNewTags) {
275             return;
276         }
277         var writer = new StringWriter();
278         var printer = new PrintWriter(writer);
279 
280         if (annotation != null) {
281             printer.println(tagAnnotationToText(repository, annotation));
282         }
283         printer.println(CommitFormatters.toTextBrief(repository, commit));
284 
285         var subject = tagToSubject(repository, commit.hash(), tag);
286         var email = Email.create(subject, writer.toString())
287                          .sender(sender)
288                          .recipient(recipient)
289                          .headers(headers)
290                          .headers(commitHeaders(repository, List.of(commit)));
291 
292         if (annotation != null) {
293             email.author(annotationToAuthor(annotation));
294         } else {
295             email.author(commitToAuthor(commit));
296         }
297 
298         try {
299             list.post(email.build());
300         } catch (RuntimeException e) {
301             throw new NonRetriableException(e);
302         }
303     }

305     private String newBranchSubject(HostedRepository repository, Repository localRepository, List<Commit> commits, Branch parent, Branch branch) {
306         var subject = new StringBuilder();
307         subject.append(repository.repositoryType().shortName());
308         subject.append(": ");
309         subject.append(repository.name());
310         subject.append(": created branch ");
311         subject.append(branch);
312         subject.append(" based on the branch ");
313         subject.append(parent);
314         subject.append(" containing ");
315         subject.append(commits.size());
316         subject.append(" unique commit");
317         if (commits.size() != 1) {
318             subject.append("s");
319         }
320 
321         return subject.toString();
322     }

324     @Override
325     public void handleNewBranch(HostedRepository repository, Repository localRepository, List<Commit> commits, Branch parent, Branch branch) throws NonRetriableException {
326         if (!reportNewBranches) {
327             return;
328         }
329         var writer = new StringWriter();
330         var printer = new PrintWriter(writer);
331 
332         if (commits.size() > 0) {
333             printer.println("The following commits are unique to the " + branch.name() + " branch:");
334             printer.println("========================================================");
335             for (var commit : commits) {
336                 printer.print(commit.hash().abbreviate());
337                 if (commit.message().size() > 0) {
338                     printer.print(": " + commit.message().get(0));
339                 }
340                 printer.println();
341             }
342         } else {
343             printer.println("The new branch " + branch.name() + " is currently identical to the " + parent.name() + " branch.");
344         }
345 
346         var subject = newBranchSubject(repository, localRepository, commits, parent, branch);
347         var finalAuthor = commits.size() > 0 ? commitToAuthor(commits.get(commits.size() - 1)) : sender;
348 
349         var email = Email.create(subject, writer.toString())
350                          .sender(sender)
351                          .author(finalAuthor)
352                          .recipient(recipient)
353                          .headers(headers)
354                          .headers(commitHeaders(repository, commits))
355                          .build();
356         try {
357             list.post(email);
358         } catch (RuntimeException e) {
359             throw new NonRetriableException(e);
360         }
361     }
