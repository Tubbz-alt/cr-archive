 43 def _update_if_needed(ui):
 44     skara = os.path.dirname(os.path.realpath(__file__))
 45     git_skara = os.path.join(skara, 'bin', 'bin', 'git-skara')
 46     if not os.path.isfile(git_skara):
 47         ui.status(b'Compiling ...\n')
 48         cmd = ['gradlew.bat'] if os.name == 'nt' else ['/bin/sh', 'gradlew']
 49         p = subprocess.Popen(cmd, cwd=skara)
 50         ret = p.wait()
 51         if ret != 0:
 52             ui.error(b"Error: could not compile Skara\n")
 53             sys.exit(1)
 54 
 55     skara_bin = os.path.join(skara, 'bin')
 56     skara_build = os.path.join(skara, 'build')
 57     if os.path.isdir(skara_build):
 58         if os.path.isdir(skara_bin):
 59             shutil.rmtree(skara_bin)
 60         shutil.move(skara_build, skara_bin)
 61     return git_skara

 63 def _opts_to_flags(cmd = None, **opts):
 64     args = []
 65     for k in opts:
 66         name = k.replace('_', '-')
 67         v = opts[k]
 68         if v == True:
 69             args.append('--' + name)
 70         elif k == b'terse' and cmd == 'status':
 71             if v != b'nothing':
 72                 args.append('--' + name)
 73                 args.append(str(v))
 74         elif v != b'' and v != [] and v != None and v != False:
 75             args.append('--' + name)
 76             args.append(str(v))
 77     return args

 79 def _skara(ui, args, **opts):
 80     git_skara = _update_if_needed(ui)
 81     flags = _opts_to_flags(**opts)
 82     ret = subprocess.call([git_skara] + args + flags)
 83     sys.exit(ret)
 84 
 85 def _trees(ui, command, *args, **opts):
 86     git_skara = _update_if_needed(ui)
 87     flags = _opts_to_flags(command, **opts)
 88     ret = subprocess.call([git_skara] + ['trees', '--mercurial', command] + flags + list(args))
 89     sys.exit(ret)

151 @command(b'tclone', norepo=True)
152 def tclone(ui, source, dest=None, **opts):
153     repo = mercurial.hg.peer(ui, opts, source)
154     trees = sorted(list(repo.listkeys(b'trees').values()))
155     ui.status(b'cloning %s\n' % source)
156     if mercurial.commands.clone(ui, source, dest, **opts):
157         return True
158 
159     dest = os.path.basename(source) if dest == None else dest
160     with open(os.path.join(dest, '.hg', 'files'), 'w') as f:
161         f.write('\n'.join(trees))
162 
163     for t in sorted(trees, key=len):
164         tsource = source + b'/' + t
165         tdest = os.path.join(dest, t)
166 
167         ui.status(b'\n')
168         ui.status(b'cloning %s\n' % tsource)
169         ui.status(b'destination directory: %s\n' % tdest)
170         if mercurial.commands.clone(ui, tsource, tdest, **opts):
171             return True
