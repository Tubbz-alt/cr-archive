 81     private boolean isTargetBranchAllowed() {
 82         var matcher = workItem.bot.allowedTargetBranches().matcher(pr.targetRef());
 83         return matcher.matches();
 84     }

 86     private List<String> allowedTargetBranches() {
 87         return pr.repository()
 88                  .branches()
 89                  .stream()
 90                  .map(HostedBranch::name)
 91                  .map(name -> workItem.bot.allowedTargetBranches().matcher(name))
 92                  .filter(Matcher::matches)
 93                  .map(Matcher::group)
 94                  .collect(Collectors.toList());
 95     }

 97     // For unknown contributors, check that all commits have the same name and email
 98     private boolean checkCommitAuthor(List<CommitMetadata> commits) throws IOException {
 99         var author = censusInstance.namespace().get(pr.author().id());
100         if (author != null) {
101             return true;
102         }
103 
104         var names = new HashSet<String>();
105         var emails = new HashSet<String>();
106 
107         for (var commit : commits) {
108             names.add(commit.author().name());
109             emails.add(commit.author().email());
110         }
111 
112         return ((names.size() == 1) && emails.size() == 1);
113     }

117         private final String branchName;

120             this.repositoryName = repositoryName;
121             this.branchName = branchName;

125     private Optional<MergeSource> mergeSource() {
126         var repoMatcher = mergeSourceFullPattern.matcher(pr.title());
127         if (!repoMatcher.matches()) {
128             var branchMatcher = mergeSourceBranchOnlyPattern.matcher(pr.title());
129             if (!branchMatcher.matches()) {
130                 return Optional.empty();
131             }
132 
133             // Verify that the branch exists
134             var isValidBranch = prInstance.remoteBranches().stream()
135                                           .map(Reference::name)
136                                           .anyMatch(branch -> branch.equals(branchMatcher.group(1)));
137             if (!isValidBranch) {
138                 // Assume the name refers to a sibling repository
139                 var repoName = Path.of(pr.repository().name()).resolveSibling(branchMatcher.group(1)).toString();
140                 return Optional.of(new MergeSource(repoName, "master"));
141             }
142 
143             return Optional.of(new MergeSource(pr.repository().name(), branchMatcher.group(1)));
144         }
145 
146         return Optional.of(new MergeSource(repoMatcher.group(1), repoMatcher.group(2)));
147     }

149     // Additional bot-specific checks that are not handled by JCheck
150     private List<String> botSpecificChecks() throws IOException {
151         var ret = new ArrayList<String>();
152 
153         if (bodyWithoutStatus().isBlank()) {
154             var error = "The pull request body must not be empty.";
155             ret.add(error);
156         }
157 
158         if (!isTargetBranchAllowed()) {
159             var error = "The branch `" + pr.targetRef() + "` is not allowed as target branch. The allowed target branches are:\n" +
160                     allowedTargetBranches().stream()
161                     .map(name -> "   - " + name)
162                     .collect(Collectors.joining("\n"));
163             ret.add(error);
164         }
165 
166         var baseHash = prInstance.baseHash();
167         var headHash = pr.headHash();
168         var commits = prInstance.localRepo().commitMetadata(baseHash, headHash);
169 
170         if (!checkCommitAuthor(commits)) {
171             var error = "For contributors who are not existing OpenJDK Authors, commit attribution will be taken from " +
172                     "the commits in the PR. However, the commits in this PR have inconsistent user names and/or " +
173                     "email addresses. Please amend the commits.";
174             ret.add(error);
175         }
176 
177         if (pr.title().startsWith("Merge")) {
178             if (commits.size() < 2) {
179                 ret.add("A Merge PR must contain at least two commits that are not already present in the target.");
180             } else {
181                 // Find the first merge commit - the very last commit is not eligible (as the merge needs a parent)
182                 int mergeCommitIndex = commits.size();
183                 for (int i = 0; i < commits.size() - 1; ++i) {
184                     if (commits.get(i).isMerge()) {
185                         mergeCommitIndex = i;
186                         break;
187                     }
188                 }
189                 if (mergeCommitIndex >= commits.size() - 1) {
190                     ret.add("A Merge PR must contain a merge commit as well as at least one other commit from the merge source.");
191                 }
192 
193                 var source = mergeSource();
194                 if (source.isPresent()) {
195                     try {
196                         var mergeSourceRepo = pr.repository().forge().repository(source.get().repositoryName).orElseThrow(() ->
197                                 new RuntimeException("Could not find repository " + source.get().repositoryName)
198                         );
199                         try {
200                             var sourceHash = prInstance.localRepo().fetch(mergeSourceRepo.url(), source.get().branchName);
201                             var mergeCommit = commits.get(mergeCommitIndex);
202                             for (int i = 0; i < mergeCommit.parents().size(); ++i) {
203                                 if (!prInstance.localRepo().isAncestor(mergeCommit.parents().get(i), sourceHash)) {
204                                     if (!mergeCommit.parents().get(i).equals(prInstance.targetHash())) {
205                                         ret.add("The merge contains commits that are neither ancestors of the source nor the target.");
206                                         break;
207                                     }
208                                 }
209                             }
210                         } catch (IOException e) {
211                             ret.add("Could not fetch branch `" + source.get().branchName + "` from project `" +
212                                             source.get().repositoryName + "` - check that they are correct.");
213                         }
214                     } catch (RuntimeException e) {
215                         ret.add("Could not find project `" +
216                                         source.get().repositoryName + "` - check that it is correct.");
217                     }
218                 } else {
219                     ret.add("Could not determine the source for this merge. A Merge PR title must be specified on the format: " +
220                             "Merge `project`:`branch` to allow verification of the merge contents.");
221                 }
222             }
223         }
224 
225         for (var blocker : workItem.bot.blockingCheckLabels().entrySet()) {
226             if (labels.contains(blocker.getKey())) {
227                 ret.add(blocker.getValue());
228             }
229         }
230 
231         return ret;
232     }

234     private void updateCheckBuilder(CheckBuilder checkBuilder, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
235         if (visitor.isReadyForReview() && additionalErrors.isEmpty()) {
236             checkBuilder.complete(true);
237         } else {
238             checkBuilder.title("Required");
239             var summary = Stream.concat(visitor.getMessages().stream(), additionalErrors.stream())
240                                 .sorted()
241                                 .map(m -> "- " + m)
242                                 .collect(Collectors.joining("\n"));
243             checkBuilder.summary(summary);
244             for (var annotation : visitor.getAnnotations()) {
245                 checkBuilder.annotation(annotation);
246             }
247             checkBuilder.complete(false);
248         }
249     }

251     private void updateReadyForReview(PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
252         // Additional errors are not allowed
253         if (!additionalErrors.isEmpty()) {
254             newLabels.remove("rfr");
255             return;
256         }
257 
258         // Draft requests are not for review
259         if (pr.isDraft()) {
260             newLabels.remove("rfr");
261             return;
262         }
263 
264         // Check if the visitor found any issues that should be resolved before reviewing
265         if (visitor.isReadyForReview()) {
266             newLabels.add("rfr");
267         } else {
268             newLabels.remove("rfr");
269         }
270     }

272     private String getRole(String username) {
273         var project = censusInstance.project();
274         var version = censusInstance.census().version().format();
275         if (project.isReviewer(username, version)) {
276             return "**Reviewer**";
277         } else if (project.isCommitter(username, version)) {
278             return "Committer";
279         } else if (project.isAuthor(username, version)) {
280             return "Author";
281         } else {
282             return "no project role";
283         }
284     }

286     private String formatReviewer(HostUser reviewer) {
287         var namespace = censusInstance.namespace();
288         var contributor = namespace.get(reviewer.id());
289         if (contributor == null) {
290             return reviewer.userName() + " (no known " + namespace.name() + " user name / role)";
291         } else {
292             var userNameLink = "[" + contributor.username() + "](@" + reviewer.userName() + ")";
293             return contributor.fullName().orElse(contributor.username()) + " (" + userNameLink + " - " +
294                     getRole(contributor.username()) + ")";
295         }
296     }

311     private Optional<String> getReviewersList(List<Review> reviews) {
312         var reviewers = reviews.stream()
313                                .filter(review -> review.verdict() == Review.Verdict.APPROVED)
314                                .map(review -> {
315                                    var entry = " * " + formatReviewer(review.reviewer());
316                                    if (!review.hash().equals(pr.headHash())) {
317                                        if (ignoreStaleReviews) {
318                                            entry += " 🔄 Re-review required (review applies to " + review.hash() + ")";
319                                        } else {
320                                            entry += " ⚠️ Review applies to " + review.hash();
321                                        }
322                                    }
323                                    return entry;
324                                })
325                                .collect(Collectors.joining("\n"));
326         if (reviewers.length() > 0) {
327             return Optional.of(reviewers);
328         } else {
329             return Optional.empty();
330         }
331     }

350     private String getStatusMessage(List<Comment> comments, List<Review> reviews, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
351         var progressBody = new StringBuilder();
352         progressBody.append("---------\n");
353         progressBody.append("### Progress\n");
354         progressBody.append(getChecksList(visitor));
355 
356         if (!additionalErrors.isEmpty()) {
357             progressBody.append("\n\n### Error");
358             if (additionalErrors.size() > 1) {
359                 progressBody.append("s");
360             }
361             progressBody.append("\n");
362             progressBody.append(getAdditionalErrorsList(additionalErrors));
363         }
364 
365         var issue = Issue.fromString(pr.title());
366         var issueProject = workItem.bot.issueProject();
367         if (issueProject != null && issue.isPresent()) {
368             var allIssues = new ArrayList<Issue>();
369             allIssues.add(issue.get());
370             allIssues.addAll(SolvesTracker.currentSolved(pr.repository().forge().currentUser(), comments));
371             progressBody.append("\n\n### Issue");
372             if (allIssues.size() > 1) {
373                 progressBody.append("s");
374             }
375             progressBody.append("\n");
376             for (var currentIssue : allIssues) {
377                 progressBody.append(" * ");
378                 try {
379                     var iss = issueProject.issue(currentIssue.id());
380                     if (iss.isPresent()) {
381                         progressBody.append("[");
382                         progressBody.append(iss.get().id());
383                         progressBody.append("](");
384                         progressBody.append(iss.get().webUrl());
385                         progressBody.append("): ");
386                         progressBody.append(iss.get().title());
387                         progressBody.append("\n");
388                     } else {
389                         progressBody.append("⚠️ Failed to retrieve information on issue `");
390                         progressBody.append(currentIssue.id());
391                         progressBody.append("`.\n");
392                     }
393                 } catch (RuntimeException e) {
394                     progressBody.append("⚠️ Temporary failure when trying to retrieve information on issue `");
395                     progressBody.append(currentIssue.id());
396                     progressBody.append("`.\n");
397                 }
398             }
399         }
400 
401         getReviewersList(reviews).ifPresent(reviewers -> {
402             progressBody.append("\n\n### Reviewers\n");
403             progressBody.append(reviewers);
404         });
405 
406         getContributorsList(comments).ifPresent(contributors -> {
407             progressBody.append("\n\n### Contributors\n");
408             progressBody.append(contributors);
409         });
410 
411         progressBody.append("\n\n### Download\n");
412         progressBody.append(checkoutCommands());
413 
414         return progressBody.toString();
415     }

417     private String checkoutCommands() {
418         var repoUrl = pr.repository().webUrl();
419         return
420            "`$ git fetch " + repoUrl + " " + pr.fetchRef() + ":pull/" + pr.id() + "`\n" +
421            "`$ git checkout pull/" + pr.id() + "`\n";
422     }

424     private String bodyWithoutStatus() {
425         var description = pr.body();
426         var markerIndex = description.lastIndexOf(progressMarker);
427         return (markerIndex < 0 ?
428                 description :
429                 description.substring(0, markerIndex)).trim();
430     }

432     private String updateStatusMessage(String message) {
433         var description = pr.body();
434         var markerIndex = description.lastIndexOf(progressMarker);
435 
436         if (markerIndex >= 0 && description.substring(markerIndex).equals(message)) {
437             log.info("Progress already up to date");
438             return description;
439         }
440         var newBody = bodyWithoutStatus() + "\n" + progressMarker + "\n" + message;
441 
442         // TODO? Retrieve the body again here to lower the chance of concurrent updates
443         pr.setBody(newBody);
444         return newBody;
445     }

447     private String verdictToString(Review.Verdict verdict) {
448         switch (verdict) {
449             case APPROVED:
450                 return "changes are approved";
451             case DISAPPROVED:
452                 return "more changes needed";
453             case NONE:
454                 return "comment added";
455             default:
456                 throw new RuntimeException("Unknown verdict: " + verdict);
457         }
458     }

461         var reviewTracker = new ReviewTracker(comments, reviews);

465                     "This PR has been reviewed by " +

472     private Optional<Comment> findComment(List<Comment> comments, String marker) {
473         var self = pr.repository().forge().currentUser();
474         return comments.stream()
475                        .filter(comment -> comment.author().equals(self))
476                        .filter(comment -> comment.body().contains(marker))
477                        .findAny();

480     private String getMergeReadyComment(String commitMessage, List<Review> reviews, boolean rebasePossible) {
481         var message = new StringBuilder();
482         message.append("@");
483         message.append(pr.author().userName());
484         message.append(" This change now passes all automated pre-integration checks");
485 
486         try {
487             var hasContributingFile =
488                 !prInstance.localRepo().files(prInstance.targetHash(), Path.of("CONTRIBUTING.md")).isEmpty();
489             if (hasContributingFile) {
490                 message.append(". When the change also fulfills all ");
491                 message.append("[project specific requirements](https://github.com/");
492                 message.append(pr.repository().name());
493                 message.append("/blob/");
494                 message.append(pr.targetRef());
495                 message.append("/CONTRIBUTING.md)");
496             }
497         } catch (IOException e) {
498             throw new UncheckedIOException(e);
499         }
500 
501         message.append(", type `/integrate` in a new comment to proceed. After integration, ");
502         message.append("the commit message will be:\n");
503         message.append("```\n");
504         message.append(commitMessage);
505         message.append("\n```\n");
506 
507         message.append("- If you would like to add a summary, use the `/summary` command.\n");
508         message.append("- To credit additional contributors, use the `/contributor` command.\n");
509         message.append("- To add additional solved issues, use the `/solves` command.\n");
510 
511         var divergingCommits = prInstance.divergingCommits();
512         if (divergingCommits.size() > 0) {
513             message.append("\n");
514             message.append("Since the source branch of this PR was last updated there ");
515             if (divergingCommits.size() == 1) {
516                 message.append("has been 1 commit ");
517             } else {
518                 message.append("have been ");
519                 message.append(divergingCommits.size());
520                 message.append(" commits ");
521             }
522             message.append("pushed to the `");
523             message.append(pr.targetRef());
524             message.append("` branch. ");
525             if (rebasePossible) {
526                 message.append("As there are no conflicts, your changes will automatically be rebased on top of ");
527                 message.append("these commits when integrating. If you prefer to avoid automatic rebasing, please merge `");
528                 message.append(pr.targetRef());
529                 message.append("` into your branch, and then specify the current head hash when integrating, like this: ");
530                 message.append("`/integrate ");
531                 message.append(prInstance.targetHash());
532                 message.append("`.\n");
533             } else {
534                 message.append("Your changes cannot be rebased automatically without conflicts, so you will need to ");
535                 message.append("merge `");
536                 message.append(pr.targetRef());
537                 message.append("` into your branch before integrating.\n");
538             }
539         } else {
540             message.append("\n");
541             message.append("There are currently no new commits on the `");
542             message.append(pr.targetRef());
543             message.append("` branch since the last update of the source branch of this PR. If another commit should be pushed before ");
544             message.append("you perform the `/integrate` command, your PR will be automatically rebased. If you would like to avoid ");
545             message.append("potential automatic rebasing, specify the current head hash when integrating, like this: ");
546             message.append("`/integrate ");
547             message.append(prInstance.targetHash());
548             message.append("`.\n");
549         }
550 
551         if (!ProjectPermissions.mayCommit(censusInstance, pr.author())) {
552             message.append("\n");
553             var contributor = censusInstance.namespace().get(pr.author().id());
554             if (contributor == null) {
555                 message.append("As you are not a known OpenJDK [Author](https://openjdk.java.net/bylaws#author), ");
556             } else {
557                 message.append("As you do not have Committer status in this project, ");
558             }
559 
560             message.append("an existing [Committer](https://openjdk.java.net/bylaws#committer) must agree to ");
561             message.append("[sponsor](https://openjdk.java.net/sponsor/) your change. ");
562             var candidates = reviews.stream()
563                                     .filter(review -> ProjectPermissions.mayCommit(censusInstance, review.reviewer()))
564                                     .map(review -> "@" + review.reviewer().userName())
565                                     .collect(Collectors.joining(", "));
566             if (candidates.length() > 0) {
567                 message.append("Possible candidates are the reviewers of this PR (");
568                 message.append(candidates);
569                 message.append(") but any other Committer may sponsor as well. ");
570             }
571             if (rebasePossible) {
572                 message.append("\n\n");
573                 message.append("➡️ To flag this PR as ready for integration with the above commit message, type ");
574                 message.append("`/integrate` in a new comment. (Afterwards, your sponsor types ");
575                 message.append("`/sponsor` in a new comment to perform the integration).\n");
576             }
577         } else if (rebasePossible) {
578             message.append("\n");
579             message.append("➡️ To integrate this PR with the above commit message to the `" + pr.targetRef() + "` branch, type ");
580             message.append("`/integrate` in a new comment.\n");
581         }
582         message.append(mergeReadyMarker);
583         return message.toString();
584     }

595     private void updateMergeReadyComment(boolean isReady, String commitMessage, List<Comment> comments, List<Review> reviews, boolean rebasePossible) {
596         var existing = findComment(comments, mergeReadyMarker);
597         if (isReady) {
598             var message = getMergeReadyComment(commitMessage, reviews, rebasePossible);
599             if (existing.isEmpty()) {
600                 pr.addComment(message);
601             } else {
602                 pr.updateComment(existing.get().id(), message);
603             }
604         } else {
605             existing.ifPresent(comment -> pr.updateComment(comment.id(), getMergeNoLongerReadyComment()));
606         }
607     }

609     private void addSourceBranchWarningComment(List<Comment> comments) {
610         var existing = findComment(comments, sourceBranchWarningMarker);
611         if (existing.isPresent()) {
612             // Only add the comment once per PR
613             return;
614         }
615         var branch = pr.sourceRef();
616         var message = ":warning: @" + pr.author().userName() + " " +
617             "a branch with the same name as the source branch for this pull request (`" + branch + "`) " +
618             "is present in the [target repository](" + pr.repository().nonTransformedWebUrl() + "). " +
619             "If you eventually integrate this pull request then the branch `" + branch + "` " +
620             "in your [personal fork](" + pr.sourceRepository().nonTransformedWebUrl() + ") will diverge once you sync " +
621             "your personal fork with the upstream repository.\n" +
622             "\n" +
623             "To avoid this situation, create a new branch for your changes and reset the `" + branch + "` branch. " +
624             "You can do this by running the following commands in a local repository for your personal fork. " +
625             "_Note_: you do *not* have to name the new branch `NEW-BRANCH-NAME`." +
626             "\n" +
627             "```" +
628             "$ git checkout " + branch + "\n" +
629             "$ git checkout -b NEW-BRANCH-NAME\n" +
630             "$ git branch -f " + branch + " " + prInstance.baseHash().hex() + "\n" +
631             "$ git push -f origin " + branch + "\n" +
632             "```\n" +
633             "\n" +
634             "Then proceed to create a new pull request with `NEW-BRANCH-NAME` as the source branch and " +
635             "close this one.\n" +
636             sourceBranchWarningMarker;
637         pr.addComment(message);
638     }

640     private void addOutdatedComment(List<Comment> comments) {
641         var existing = findComment(comments, outdatedHelpMarker);
642         if (existing.isPresent()) {
643             // Only add the comment once per PR
644             return;
645         }
646         var message = "@" + pr.author().userName() + " this pull request can no longer be integrated into " +
647                 "`" + pr.targetRef() + "` due to one or more merge conflicts. To resolve these merge conflicts " +
648                 "and update this pull request you can run the following commands in the local repository for your personal fork:\n" +
649                 "```bash\n" +
650                 "git checkout " + pr.sourceRef() + "\n" +
651                 "git fetch " + pr.repository().webUrl() + " " + pr.targetRef() + "\n" +
652                 "git merge FETCH_HEAD\n" +
653                 "# resolve conflicts and follow the instructions given by git merge\n" +
654                 "git commit -m \"Merge " + pr.targetRef() + "\"\n" +
655                 "git push\n" +
656                 "```\n" +
657                 outdatedHelpMarker;
658         pr.addComment(message);
659     }

661     private void checkStatus() {
662         var checkBuilder = CheckBuilder.create("jcheck", pr.headHash());
663         var censusDomain = censusInstance.configuration().census().domain();
664         Exception checkException = null;
665 
666         try {
667             // Post check in-progress
668             log.info("Starting to run jcheck on PR head");
669             pr.createCheck(checkBuilder.build());
670             List<String> additionalErrors = List.of();
671             Hash localHash;
672             try {
673                 localHash = prInstance.commit(censusInstance.namespace(), censusDomain, null);
674             } catch (CommitFailure e) {
675                 additionalErrors = List.of("It was not possible to create a commit for the changes in this PR: " + e.getMessage());
676                 localHash = prInstance.baseHash();
677             }
678             boolean rebasePossible = true;
679             PullRequestCheckIssueVisitor visitor = prInstance.createVisitor(localHash, censusInstance);
680             if (!localHash.equals(prInstance.baseHash())) {
681                 // Try to rebase
682                 var ignored = new PrintWriter(new StringWriter());
683                 var rebasedHash = prInstance.rebase(localHash, ignored);
684                 if (rebasedHash.isEmpty()) {
685                     rebasePossible = false;
686                 } else {
687                     localHash = rebasedHash.get();
688                 }
689 
690                 // Determine current status
691                 var additionalConfiguration = AdditionalConfiguration.get(prInstance.localRepo(), localHash, pr.repository().forge().currentUser(), comments);
692                 prInstance.executeChecks(localHash, censusInstance, visitor, additionalConfiguration);
693                 additionalErrors = botSpecificChecks();
694             } else {
695                 if (additionalErrors.isEmpty()) {
696                     additionalErrors = List.of("This PR contains no changes");
697                 }
698             }
699             updateCheckBuilder(checkBuilder, visitor, additionalErrors);
700             updateReadyForReview(visitor, additionalErrors);
701 
702             // Calculate and update the status message if needed
703             var statusMessage = getStatusMessage(comments, activeReviews, visitor, additionalErrors);
704             var updatedBody = updateStatusMessage(statusMessage);
705 
706             // Post / update approval messages (only needed if the review itself can't contain a body)
707             if (!pr.repository().forge().supportsReviewBody()) {
708                 updateReviewedMessages(comments, allReviews);
709             }
710 
711             var commit = prInstance.localRepo().lookup(localHash).orElseThrow();
712             var commitMessage = String.join("\n", commit.message());
713             var readyForIntegration = visitor.getMessages().isEmpty() && additionalErrors.isEmpty();
714             updateMergeReadyComment(readyForIntegration, commitMessage, comments, activeReviews, rebasePossible);
715             if (readyForIntegration && rebasePossible) {
716                 newLabels.add("ready");
717             } else {
718                 newLabels.remove("ready");
719             }
720             if (!rebasePossible) {
721                 if (!labels.contains("failed-auto-merge")) {
722                     addOutdatedComment(comments);
723                 }
724                 newLabels.add("merge-conflict");
725             } else {
726                 newLabels.remove("merge-conflict");
727             }
728 
729             var branchNames = pr.repository().branches().stream().map(HostedBranch::name).collect(Collectors.toSet());
730             if (!pr.repository().url().equals(pr.sourceRepository().url()) && branchNames.contains(pr.sourceRef())) {
731                 addSourceBranchWarningComment(comments);
732             }
733 
734             // Ensure that the ready for sponsor label is up to date
735             newLabels.remove("sponsor");
736             var readyHash = ReadyForSponsorTracker.latestReadyForSponsor(pr.repository().forge().currentUser(), comments);
737             if (readyHash.isPresent() && readyForIntegration) {
738                 var acceptedHash = readyHash.get();
739                 if (pr.headHash().equals(acceptedHash)) {
740                     newLabels.add("sponsor");
741                 }
742             }
743 
744             // Calculate current metadata to avoid unnecessary future checks
745             var metadata = workItem.getMetadata(pr.title(), updatedBody, pr.comments(), activeReviews, newLabels,
746                                                 censusInstance, pr.targetHash(), pr.isDraft());
747             checkBuilder.metadata(metadata);
748         } catch (Exception e) {
749             log.throwing("CommitChecker", "checkStatus", e);
750             newLabels.remove("ready");
751             checkBuilder.metadata("invalid");
752             checkBuilder.title("Exception occurred during jcheck - the operation will be retried");
753             checkBuilder.summary(e.getMessage());
754             checkBuilder.complete(false);
755             checkException = e;
756         }
757         var check = checkBuilder.build();
758         pr.updateCheck(check);
759 
760         // Synchronize the wanted set of labels
761         for (var newLabel : newLabels) {
762             if (!labels.contains(newLabel)) {
763                 pr.addLabel(newLabel);
764             }
765         }
766         for (var oldLabel : labels) {
767             if (!newLabels.contains(oldLabel)) {
768                 pr.removeLabel(oldLabel);
769             }
770         }
771 
772         // After updating the PR, rethrow any exception to automatically retry on transient errors
773         if (checkException != null) {
774             throw new RuntimeException("Exception during jcheck", checkException);
775         }
776     }
