23 package org.openjdk.skara.bots.mlbridge;
24 
25 import java.net.URI;
26 import java.util.*;
27 import java.util.regex.Pattern;
28 
29 class QuoteFilter {
30     private static final Pattern leadingQuotesPattern = Pattern.compile("^([>\\s]+).*");
31 
32     private static Optional<String> leadingQuotes(String line) {
33         var leadingQuotesMatcher = leadingQuotesPattern.matcher(line);
34         if (leadingQuotesMatcher.matches()) {
35             if (leadingQuotesMatcher.group(1).contains(">")) {
36                 return Optional.of(leadingQuotesMatcher.group(1).trim());
37             }
38         }
39         return Optional.empty();
40     }
41 
42     // Strip all quote blocks containing a certain link
43     public static String stripLinkBlock(String body, URI link) {
44         var ret = new ArrayList<String>();
45         var buffer = new LinkedList<String>();
46         String dropPrefix = null;
47 
48         for (var line : body.split("\\R")) {
49             if (dropPrefix != null && line.startsWith(dropPrefix)) {
50                 continue;
51             }
52             dropPrefix = null;
53 
54             if (line.contains(link.toString())) {
55                 var prefix = leadingQuotes(line);
56                 if (prefix.isEmpty()) {
57                     buffer.add(line);
58                     continue;
59                 }
60                 dropPrefix = prefix.get();
61 
62                 // Drop any previous lines with the same prefix
63                 while (!buffer.isEmpty()) {
64                     if (buffer.peekLast().startsWith(dropPrefix)) {
65                         buffer.removeLast();
66                     } else {
67                         break;
68                     }
69                 }
70                 // Any remaining lines in buffer should be kept in the final result
71                 ret.addAll(buffer);
72                 buffer.clear();
73             } else {
74                 buffer.add(line);
75             }
76         }
77 
78         ret.addAll(buffer);
79         return String.join("\n", ret);
80     }
81 }
