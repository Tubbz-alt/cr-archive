23 package org.openjdk.skara.cli;
24 
25 import java.util.logging.*;
26 
27 class Logging {
28     private static Logger log;
29 
30     static void setup(Level level) {
31         setup(level, "");
32     }
33 
34     static void setup(Level level, String component) {
35         LogManager.getLogManager().reset();
36         log = level == Level.FINE ?
37             Logger.getLogger("org.openjdk.skara" + "." + component) :
38             Logger.getLogger("org.openjdk.skara");
39         log.setLevel(level);
40 
41         ConsoleHandler handler = new ConsoleHandler();
42         handler.setFormatter(new MinimalFormatter());
43         handler.setLevel(level);
44         log.addHandler(handler);
45     }
46 }
