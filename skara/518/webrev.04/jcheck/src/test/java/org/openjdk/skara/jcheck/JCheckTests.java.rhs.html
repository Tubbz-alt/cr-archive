 41     static class CheckableRepository {
 42         public static Repository create(Path path, VCS vcs) throws IOException {
 43             var repo = Repository.init(path, vcs);
 44 
 45             Files.createDirectories(path.resolve(".jcheck"));
 46             var checkConf = path.resolve(".jcheck/conf");
 47             try (var output = new FileWriter(checkConf.toFile())) {
 48                 output.append("[general]\n");
 49                 output.append("project=test\n");
 50                 output.append("\n");
 51                 output.append("[checks]\n");
 52                 output.append("error=reviewers,whitespace\n");
 53                 output.append("\n");
 54                 output.append("[census]\n");
 55                 output.append("version=0\n");
 56                 output.append("domain=openjdk.java.net\n");
 57                 output.append("\n");
 58                 output.append("[checks \"whitespace\"]\n");
 59                 output.append("suffixes=.txt\n");
 60                 output.append("\n");
 61                 output.append("[checks \"reviewers\"]\n");
 62                 output.append("minimum=1\n");
 63             }
 64             repo.add(checkConf);
 65 
 66             repo.commit("Initial commit\n\nReviewed-by: user2", "user3", "user3@openjdk.java.net");
 67 
 68             return repo;
 69         }
 70     }

256     @ParameterizedTest
257     @EnumSource(VCS.class)
258     void checksForCommit(VCS vcs) throws Exception {
259         try (var dir = new TemporaryDirectory()) {
260             var repoPath = dir.path().resolve("repo");
261             var repo = CheckableRepository.create(repoPath, vcs);
262 
263             var readme = repoPath.resolve("README");
264             Files.write(readme, List.of("Hello, readme!"));
265             repo.add(readme);
266             var first = repo.commit("Add README", "duke", "duke@openjdk.java.net");
267 
268             var censusPath = dir.path().resolve("census");
269             Files.createDirectories(censusPath);
270             CensusCreator.populateCensusDirectory(censusPath);
271             var census = Census.parse(censusPath);
272 
273             var checks = JCheck.checksFor(repo, census, first);
274             var checkNames = checks.stream()
275                                    .map(Check::name)
276                                    .collect(Collectors.toSet());
277             assertEquals(Set.of("whitespace", "reviewers"), checkNames);
278         }
279     }

281     @ParameterizedTest
282     @EnumSource(VCS.class)
283     void checkRemoval(VCS vcs) throws Exception {
284         try (var dir = new TemporaryDirectory()) {
285             var repoPath = dir.path().resolve("repo");
286             var repo = CheckableRepository.create(repoPath, vcs);
287 
288             var file = repoPath.resolve("file.txt");
289             Files.write(file, List.of("Hello, file!"));
290             repo.add(file);
291             var first = repo.commit("Add file", "duke", "duke@openjdk.java.net");
292 
293             Files.delete(file);
294             repo.remove(file);
295             var second = repo.commit("Remove file", "duke", "duke@openjdk.java.net");
296 
297             var censusPath = dir.path().resolve("census");
298             Files.createDirectories(censusPath);
299             CensusCreator.populateCensusDirectory(censusPath);
300             var census = Census.parse(censusPath);
301 
302             var visitor = new TestVisitor();
303             try (var issues = JCheck.check(repo, census, CommitMessageParsers.v1, first.hex() + ".." + second.hex(), Map.of(), Set.of())) {
304                 for (var issue : issues) {
305                     issue.accept(visitor);
306                 }
307             }
308             assertEquals(Set.of("org.openjdk.skara.jcheck.TooFewReviewersIssue"), visitor.issueNames());
309         }
310     }

312     @ParameterizedTest
313     @EnumSource(VCS.class)
314     void checkOverridingConfiguration(VCS vcs) throws Exception {
315         try (var dir = new TemporaryDirectory()) {
316             var repoPath = dir.path().resolve("repo");
317             var repo = CheckableRepository.create(repoPath, vcs);
318 
319             var initialCommit = repo.commits().asList().get(0);
320 
321             var jcheckConf = repoPath.resolve(".jcheck").resolve("conf");
322             assertTrue(Files.exists(jcheckConf));
323             Files.writeString(jcheckConf, "[checks \"reviewers\"]\nminimum = 0\n",
324                               StandardOpenOption.WRITE, StandardOpenOption.APPEND);
325             repo.add(jcheckConf);
326             var secondCommit = repo.commit("Do not require reviews", "user3", "user3@openjdk.java.net");
327 
328             var censusPath = dir.path().resolve("census");
329             Files.createDirectories(censusPath);
330             CensusCreator.populateCensusDirectory(censusPath);
331             var census = Census.parse(censusPath);
332 
333             // Check the last commit without reviewers, should pass since .jcheck/conf was updated
334             var range = initialCommit.hash().hex() + ".." + secondCommit.hex();
335             var visitor = new TestVisitor();
336             try (var issues = JCheck.check(repo, census, CommitMessageParsers.v1, range, Map.of(), Set.of())) {
337                 for (var issue : issues) {
338                     issue.accept(visitor);
339                 }
340             }
341             assertEquals(Set.of(), visitor.issues());
342 
343             // Check the last commit without reviewers with the initial .jcheck/conf. Should fail
344             // due to missing reviewers.
345             try (var issues = JCheck.check(repo, census, CommitMessageParsers.v1, secondCommit, initialCommit.hash(), List.of())) {
346                 for (var issue : issues) {
347                     issue.accept(visitor);
348                 }
349             }
350             assertEquals(Set.of("org.openjdk.skara.jcheck.TooFewReviewersIssue"), visitor.issueNames());
351         }
