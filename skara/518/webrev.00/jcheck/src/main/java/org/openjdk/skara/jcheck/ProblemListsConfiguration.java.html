23 package org.openjdk.skara.jcheck;
24 
25 import org.openjdk.skara.ini.Section;
26 
27 public class ProblemListsConfiguration {
28     static final ProblemListsConfiguration DEFAULT =
29             new ProblemListsConfiguration("test", "^ProblemList.*.txt$");
30 
31     private final String dirs;
32     private final String pattern;
33 
34     ProblemListsConfiguration(String dirs, String patterns) {
35         this.dirs = dirs;
36         this.pattern = patterns;
37     }
38 
39     public String dirs() {
40         return dirs;
41     }
42 
43     public String pattern() {
44         return pattern;
45     }
46 
47     static String name() {
48         return "problemlists";
49     }
50 
51     static ProblemListsConfiguration parse(Section s) {
52         if (s == null) {
53             return DEFAULT;
54         }
55 
56         var dirs = s.get("dirs", DEFAULT.dirs());
57         var pattern = s.get("pattern", DEFAULT.pattern());
58         return new ProblemListsConfiguration(dirs, pattern);
59     }
60 }
