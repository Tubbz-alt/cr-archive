23 package org.openjdk.skara.jcheck;
24 
25 import org.openjdk.skara.vcs.Commit;
26 import org.openjdk.skara.vcs.openjdk.CommitMessage;
27 
28 import java.util.Iterator;
29 import java.util.logging.Logger;
30 import java.util.regex.Pattern;
31 
32 public class MergeMessageCheck extends CommitCheck {
33     private final Logger log = Logger.getLogger("org.openjdk.skara.jcheck.merge");
34     @Override
35     Iterator<Issue> check(Commit commit, CommitMessage message, JCheckConfiguration conf) {
36         if (!commit.isMerge()) {
37             return iterator();
38         }
39 
40         var pattern = Pattern.compile(conf.checks().merge().message());
41         if (commit.message().size() != 1 || !pattern.matcher(commit.message().get(0)).matches()) {
42             var metadata = CommitIssue.metadata(commit, message, conf, this);
43             log.finer("issue: wrong merge message");
44             return iterator(new MergeMessageIssue(pattern.toString(), metadata));
45         }
46 
47         return iterator();
48     }
49 
50     @Override
51     public String name() {
52         return "merge";
53     }
54 
55     @Override
56     public String description() {
57         return "Merge commit must contain a proper message";
58     }
59 }
