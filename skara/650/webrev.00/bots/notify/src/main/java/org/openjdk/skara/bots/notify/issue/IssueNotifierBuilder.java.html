23 package org.openjdk.skara.bots.notify.issue;
24 
25 import org.openjdk.skara.issuetracker.IssueProject;
26 
27 import java.net.URI;
28 import java.util.Map;
29 
30 class IssueNotifierBuilder {
31     private IssueProject issueProject;
32     private boolean reviewLink = true;
33     private URI reviewIcon = null;
34     private boolean commitLink = true;
35     private URI commitIcon = null;
36 
37     IssueNotifierBuilder issueProject(IssueProject issueProject) {
38         this.issueProject = issueProject;
39         return this;
40     }
41 
42     IssueNotifierBuilder reviewLink(boolean reviewLink) {
43         this.reviewLink = reviewLink;
44         return this;
45     }
46 
47     IssueNotifierBuilder reviewIcon(URI reviewIcon) {
48         this.reviewIcon = reviewIcon;
49         return this;
50     }
51 
52     IssueNotifierBuilder commitLink(boolean commitLink) {
53         this.commitLink = commitLink;
54         return this;
55     }
56 
57     IssueNotifierBuilder commitIcon(URI commitIcon) {
58         this.commitIcon = commitIcon;
59         return this;
60     }
61 
62     IssueNotifier build() {
63         return new IssueNotifier(issueProject, reviewLink, reviewIcon, commitLink, commitIcon);
64     }
65 }
