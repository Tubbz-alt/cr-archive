23 package org.openjdk.skara.jcheck;
24 
25 import org.openjdk.skara.ini.Section;
26 
27 import java.util.List;
28 import java.util.stream.Collectors;
29 
30 public class CommitterConfiguration {
31     static final CommitterConfiguration DEFAULT = new CommitterConfiguration("committer");
32 
33     private final String role;
34 
35     CommitterConfiguration(String role) {
36         this.role = role;
37     }
38 
39     public String role() {
40         return role;
41     }
42 
43     static String name() {
44         return "committer";
45     }
46 
47     static CommitterConfiguration parse(Section s) {
48         if (s == null) {
49             return DEFAULT;
50         }
51 
52         var role = s.get("role", DEFAULT.role());
53         return new CommitterConfiguration(role);
54     }
55 }
