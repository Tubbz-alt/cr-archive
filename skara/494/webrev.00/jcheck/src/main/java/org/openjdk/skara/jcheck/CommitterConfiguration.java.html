23 package org.openjdk.skara.jcheck;
24 
25 import org.openjdk.skara.ini.Section;
26 
27 import java.util.HashSet;
28 import java.util.List;
29 import java.util.Set;
30 import java.util.stream.Collectors;
31 
32 public class CommitterConfiguration {
33     static final CommitterConfiguration DEFAULT = new CommitterConfiguration("committer", Set.of());
34 
35     private final String role;
36     private final Set<String> allowedToMerge;
37 
38     CommitterConfiguration(String role, Set<String> allowedToMerge) {
39         this.role = role;
40         this.allowedToMerge = allowedToMerge;
41     }
42 
43     public String role() {
44         return role;
45     }
46 
47     public Set<String> allowedToMerge() {
48         return allowedToMerge;
49     }
50 
51     static String name() {
52         return "committer";
53     }
54 
55     static CommitterConfiguration parse(Section s) {
56         if (s == null) {
57             return DEFAULT;
58         }
59 
60         var role = s.get("role", DEFAULT.role());
61         var usernames = s.get("allowed-to-merge", "");
62         var allowedToMerge = new HashSet<String>();
63         for (var username : usernames.split(",")) {
64             allowedToMerge.add(username.trim());
65         }
66         return new CommitterConfiguration(role, allowedToMerge);
67     }
68 }
