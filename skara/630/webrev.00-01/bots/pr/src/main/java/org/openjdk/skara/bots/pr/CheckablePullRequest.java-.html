 36 public class CheckablePullRequest {
 37     private final PullRequest pr;
 38     private final Repository localRepo;
 39     private final boolean ignoreStaleReviews;
 40 
 41     CheckablePullRequest(PullRequest pr, Repository localRepo, boolean ignoreStaleReviews) {
 42         this.pr = pr;
 43         this.localRepo = localRepo;
 44         this.ignoreStaleReviews = ignoreStaleReviews;
 45     }
 46 
 47     private String commitMessage(List<Review> activeReviews, Namespace namespace) throws IOException {
 48         var reviewers = activeReviews.stream()
 49                                      .filter(review -> !ignoreStaleReviews || review.hash().equals(pr.headHash()))
 50                                      .filter(review -> review.verdict() == Review.Verdict.APPROVED)
 51                                      .map(review -> review.reviewer().id())
 52                                      .map(namespace::get)
 53                                      .filter(Objects::nonNull)
 54                                      .map(Contributor::username)
 55                                      .collect(Collectors.toList());
 56 
 57         var comments = pr.comments();
 58         var currentUser = pr.repository().forge().currentUser();
 59 
 60         if (!ignoreStaleReviews) {
 61             var allReviewers = activeReviews.stream()
 62                                             .map(review -> namespace.get(review.reviewer().id()))
 63                                             .filter(Objects::nonNull)
 64                                             .map(Contributor::username)
 65                                             .collect(Collectors.toSet());
 66             var additionalReviewers = Reviewers.reviewers(currentUser, comments);
 67             for (var additionalReviewer : additionalReviewers) {
 68                 if (!allReviewers.contains(additionalReviewer)) {
 69                     reviewers.add(additionalReviewer);
 70                 }
 71             }
 72         }
 73 
 74         var additionalContributors = Contributors.contributors(currentUser,
 75                                                                comments).stream()
 76                                                  .map(email -> Author.fromString(email.toString()))
 77                                                  .collect(Collectors.toList());
 78 
 79         var additionalIssues = SolvesTracker.currentSolved(currentUser, comments);
 80         var summary = Summary.summary(currentUser, comments);
 81         var issue = Issue.fromStringRelaxed(pr.title());
 82         var commitMessageBuilder = issue.map(CommitMessage::title).orElseGet(() -> CommitMessage.title(pr.title()));
 83         if (issue.isPresent()) {
 84             commitMessageBuilder.issues(additionalIssues);
 85         }
 86         commitMessageBuilder.contributors(additionalContributors)
 87                             .reviewers(reviewers);
 88         summary.ifPresent(commitMessageBuilder::summary);
 89 
 90         return String.join("\n", commitMessageBuilder.format(CommitMessageFormatters.v1));
 91     }
 92 
 93     /**
 94      * The Review list is in chronological order, the latest one from a particular reviewer is the
 95      * one that is "active".
 96      * @param allReviews
 97      * @return
 98      */
 99     static List<Review> filterActiveReviews(List<Review> allReviews) {
100         var reviewPerUser = new LinkedHashMap<HostUser, Review>();
101         for (var review : allReviews) {
102             reviewPerUser.put(review.reviewer(), review);
103         }
104         return new ArrayList<>(reviewPerUser.values());
105     }
106 
107     Hash commit(Hash finalHead, Namespace namespace, String censusDomain, String sponsorId) throws IOException, CommitFailure {
108         Author committer;
109         Author author;
110         var contributor = namespace.get(pr.author().id());
111 
112         if (contributor == null) {
113             if (PullRequestUtils.isMerge(pr)) {
114                 throw new CommitFailure("Merge PRs can only be created by known OpenJDK authors.");
115             }
116 
117             // Use the information contained in the head commit - jcheck has verified that it contains sane values
118             var headCommit = localRepo.commitMetadata(pr.headHash().hex() + "^.." + pr.headHash().hex()).get(0);
119             author = headCommit.author();
120         } else {
121             author = new Author(contributor.fullName().orElseThrow(), contributor.username() + "@" + censusDomain);
122         }
123 
124         if (sponsorId != null) {
125             var sponsorContributor = namespace.get(sponsorId);
126             committer = new Author(sponsorContributor.fullName().orElseThrow(), sponsorContributor.username() + "@" + censusDomain);
127         } else {
128             committer = author;
129         }
130 
131         var activeReviews = filterActiveReviews(pr.reviews());
132         var commitMessage = commitMessage(activeReviews, namespace);
133         return PullRequestUtils.createCommit(pr, localRepo, finalHead, author, committer, commitMessage);
134     }
135 
136     PullRequestCheckIssueVisitor createVisitor(Hash localHash, CensusInstance censusInstance) throws IOException {
137         var checks = JCheck.checksFor(localRepo, censusInstance.census(), pr.targetHash());
138         return new PullRequestCheckIssueVisitor(checks);
139     }
140 
141     void executeChecks(Hash localHash, CensusInstance censusInstance, PullRequestCheckIssueVisitor visitor, List<String> additionalConfiguration) throws Exception {
142         try (var issues = JCheck.check(localRepo, censusInstance.census(), CommitMessageParsers.v1, localHash,
143                                        pr.targetHash(), additionalConfiguration)) {
144             for (var issue : issues) {
145                 issue.accept(visitor);
146             }
147         }
148     }
149 
150     List<CommitMetadata> divergingCommits() {
151         return divergingCommits(pr.headHash());
152     }
153 
154     private List<CommitMetadata> divergingCommits(Hash commitHash) {
155         try {
156             var updatedBase = localRepo.mergeBase(pr.targetHash(), commitHash);
157             return localRepo.commitMetadata(updatedBase, pr.targetHash());
158         } catch (IOException e) {
159             throw new RuntimeException(e);
160         }
161     }
162 
163     Optional<Hash> mergeTarget(PrintWriter reply) {
164         var divergingCommits = divergingCommits(pr.headHash());
165         if (divergingCommits.size() > 0) {
166             reply.print("The following commits have been pushed to ");
167             reply.print(pr.targetRef());
168             reply.println(" since your change was applied:");
169             divergingCommits.forEach(c -> reply.println(" * " + c.hash().hex() + ": " + c.message().get(0)));
170 
171             try {
172                 localRepo.checkout(pr.headHash(), true);
173                 localRepo.merge(pr.targetHash());
174                 var hash = localRepo.commit("Automatic merge with latest target", "duke", "duke@openjdk.org");
175                 reply.println();
176                 reply.println("Your commit was automatically rebased without conflicts.");
177                 return Optional.of(hash);
178             } catch (IOException e) {
179                 reply.println();
180                 reply.print("It was not possible to rebase your changes automatically. Please merge `");
181                 reply.print(pr.targetRef());
182                 reply.println("` into your branch and try again.");
183                 return Optional.empty();
184             }
185         } else {
186             // No merge needed
187             return Optional.of(pr.headHash());
188         }
189     }
