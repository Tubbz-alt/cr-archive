 88     private boolean isTargetBranchAllowed() {
 89         var matcher = workItem.bot.allowedTargetBranches().matcher(pr.targetRef());
 90         return matcher.matches();
 91     }

 93     private List<String> allowedTargetBranches() {
 94         return pr.repository()
 95                  .branches()
 96                  .stream()
 97                  .map(HostedBranch::name)
 98                  .map(name -> workItem.bot.allowedTargetBranches().matcher(name))
 99                  .filter(Matcher::matches)
100                  .map(Matcher::group)
101                  .collect(Collectors.toList());
102     }

104     // For unknown contributors, check that all commits have the same name and email
105     private boolean checkCommitAuthor(List<CommitMetadata> commits) throws IOException {
106         var author = censusInstance.namespace().get(pr.author().id());
107         if (author != null) {
108             return true;
109         }
110 
111         var names = new HashSet<String>();
112         var emails = new HashSet<String>();
113 
114         for (var commit : commits) {
115             names.add(commit.author().name());
116             emails.add(commit.author().email());
117         }
118 
119         return ((names.size() == 1) && emails.size() == 1);
120     }

122     // Additional bot-specific checks that are not handled by JCheck
123     private List<String> botSpecificChecks(Hash finalHash) throws IOException {
124         var ret = new ArrayList<String>();
125 
126         if (bodyWithoutStatus().isBlank()) {
127             var error = "The pull request body must not be empty.";
128             ret.add(error);
129         }
130 
131         if (!isTargetBranchAllowed()) {
132             var error = "The branch `" + pr.targetRef() + "` is not allowed as target branch. The allowed target branches are:\n" +
133                     allowedTargetBranches().stream()
134                     .map(name -> "   - " + name)
135                     .collect(Collectors.joining("\n"));
136             ret.add(error);
137         }
138 
139         var headHash = pr.headHash();
140         var originalCommits = localRepo.commitMetadata(baseHash, headHash);
141 
142         if (!checkCommitAuthor(originalCommits)) {
143             var error = "For contributors who are not existing OpenJDK Authors, commit attribution will be taken from " +
144                     "the commits in the PR. However, the commits in this PR have inconsistent user names and/or " +
145                     "email addresses. Please amend the commits.";
146             ret.add(error);
147         }
148 
149         for (var blocker : workItem.bot.blockingCheckLabels().entrySet()) {
150             if (labels.contains(blocker.getKey())) {
151                 ret.add(blocker.getValue());
152             }
153         }
154 
155         return ret;
156     }

158     private void updateCheckBuilder(CheckBuilder checkBuilder, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
159         if (visitor.isReadyForReview() && additionalErrors.isEmpty()) {
160             checkBuilder.complete(true);
161         } else {
162             checkBuilder.title("Required");
163             var summary = Stream.concat(visitor.messages().stream(), additionalErrors.stream())
164                                 .sorted()
165                                 .map(m -> "- " + m)
166                                 .collect(Collectors.joining("\n"));
167             checkBuilder.summary(summary);
168             for (var annotation : visitor.getAnnotations()) {
169                 checkBuilder.annotation(annotation);
170             }
171             checkBuilder.complete(false);
172         }
173     }

175     private void updateReadyForReview(PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
176         // Additional errors are not allowed
177         if (!additionalErrors.isEmpty()) {
178             newLabels.remove("rfr");
179             return;
180         }
181 
182         // Draft requests are not for review
183         if (pr.isDraft()) {
184             newLabels.remove("rfr");
185             return;
186         }
187 
188         // Check if the visitor found any issues that should be resolved before reviewing
189         if (visitor.isReadyForReview()) {
190             newLabels.add("rfr");
191         } else {
192             newLabels.remove("rfr");
193         }
194     }

196     private String getRole(String username) {
197         var project = censusInstance.project();
198         var version = censusInstance.census().version().format();
199         if (project.isReviewer(username, version)) {
200             return "**Reviewer**";
201         } else if (project.isCommitter(username, version)) {
202             return "Committer";
203         } else if (project.isAuthor(username, version)) {
204             return "Author";
205         } else {
206             return "no project role";
207         }
208     }

210     private String formatReviewer(HostUser reviewer) {
211         var namespace = censusInstance.namespace();
212         var contributor = namespace.get(reviewer.id());
213         if (contributor == null) {
214             return reviewer.userName() + " (no known " + namespace.name() + " user name / role)";
215         } else {
216             var userNameLink = "[" + contributor.username() + "](@" + reviewer.userName() + ")";
217             return contributor.fullName().orElse(contributor.username()) + " (" + userNameLink + " - " +
218                     getRole(contributor.username()) + ")";
219         }
220     }

235     private Optional<String> getReviewersList(List<Review> reviews) {
236         var reviewers = reviews.stream()
237                                .filter(review -> review.verdict() == Review.Verdict.APPROVED)
238                                .map(review -> {
239                                    var entry = " * " + formatReviewer(review.reviewer());
240                                    if (!review.hash().equals(pr.headHash())) {
241                                        if (ignoreStaleReviews) {
242                                            entry += " 🔄 Re-review required (review applies to " + review.hash() + ")";
243                                        } else {
244                                            entry += " ⚠️ Review applies to " + review.hash();
245                                        }
246                                    }
247                                    return entry;
248                                })
249                                .collect(Collectors.joining("\n"));
250 
251         // Check for manually added reviewers
252         if (!ignoreStaleReviews) {
253             var namespace = censusInstance.namespace();
254             var allReviewers = reviews.stream()
255                                       .map(review -> namespace.get(review.reviewer().id()))
256                                       .filter(Objects::nonNull)
257                                       .map(Contributor::username)
258                                       .collect(Collectors.toSet());
259             var additionalEntries = new ArrayList<String>();
260             for (var additional : Reviewers.reviewers(pr.repository().forge().currentUser(), comments)) {
261                 if (!allReviewers.contains(additional)) {
262                     additionalEntries.add(" * " + additional + " - " + getRole(additional) + " ⚠️ Added manually");
263                 }
264             }
265             if (!reviewers.isBlank()) {
266                 reviewers += "\n";
267             }
268             reviewers += String.join("\n", additionalEntries);
269         }
270 
271         if (reviewers.length() > 0) {
272             return Optional.of(reviewers);
273         } else {
274             return Optional.empty();
275         }
276     }

284         var contributors = Contributors.contributors(pr.repository().forge().currentUser(), comments)

295     private boolean relaxedEquals(String s1, String s2) {
296         return s1.trim()
297                  .replaceAll("\\s+", " ")
298                  .equalsIgnoreCase(s2.trim()
299                                      .replaceAll("\\s+", " "));
300     }

303     private String getStatusMessage(List<Comment> comments, List<Review> reviews, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
304         var progressBody = new StringBuilder();
305         progressBody.append("---------\n");
306         progressBody.append("### Progress\n");
307         progressBody.append(getChecksList(visitor));
308 
309         var allAdditionalErrors = Stream.concat(visitor.hiddenMessages().stream(), additionalErrors.stream())
310                                         .sorted()
311                                         .collect(Collectors.toList());
312         if (!allAdditionalErrors.isEmpty()) {
313             progressBody.append("\n\n### Error");
314             if (allAdditionalErrors.size() > 1) {
315                 progressBody.append("s");
316             }
317             progressBody.append("\n");
318             progressBody.append(getAdditionalErrorsList(allAdditionalErrors));
319         }
320 
321         var issue = Issue.fromStringRelaxed(pr.title());
322         var issueProject = workItem.bot.issueProject();
323         if (issueProject != null && issue.isPresent()) {
324             var allIssues = new ArrayList<Issue>();
325             allIssues.add(issue.get());
326             allIssues.addAll(SolvesTracker.currentSolved(pr.repository().forge().currentUser(), comments));
327             progressBody.append("\n\n### Issue");
328             if (allIssues.size() > 1) {
329                 progressBody.append("s");
330             }
331             progressBody.append("\n");
332             for (var currentIssue : allIssues) {
333                 progressBody.append(" * ");
334                 if (currentIssue.project().isPresent() && !currentIssue.project().get().equals(issueProject.name())) {
335                     progressBody.append("⚠️ Issue `");
336                     progressBody.append(currentIssue.id());
337                     progressBody.append("` does not belong to the `");
338                     progressBody.append(issueProject.name());
339                     progressBody.append("` project.");
340                 } else {
341                     try {
342                         var iss = issueProject.issue(currentIssue.shortId());
343                         if (iss.isPresent()) {
344                             progressBody.append("[");
345                             progressBody.append(iss.get().id());
346                             progressBody.append("](");
347                             progressBody.append(iss.get().webUrl());
348                             progressBody.append("): ");
349                             progressBody.append(iss.get().title());
350                             if (!relaxedEquals(iss.get().title(), currentIssue.description())) {
351                                 progressBody.append(" ⚠️ Title mismatch between PR and JBS.");
352                             }
353                             progressBody.append("\n");
354                         } else {
355                             progressBody.append("⚠️ Failed to retrieve information on issue `");
356                             progressBody.append(currentIssue.id());
357                             progressBody.append("`.\n");
358                         }
359                     } catch (RuntimeException e) {
360                         progressBody.append("⚠️ Temporary failure when trying to retrieve information on issue `");
361                         progressBody.append(currentIssue.id());
362                         progressBody.append("`.\n");
363                     }
364                 }
365             }
366         }
367 
368         getReviewersList(reviews).ifPresent(reviewers -> {
369             progressBody.append("\n\n### Reviewers\n");
370             progressBody.append(reviewers);
371         });
372 
373         getContributorsList(comments).ifPresent(contributors -> {
374             progressBody.append("\n\n### Contributors\n");
375             progressBody.append(contributors);
376         });
377 
378         progressBody.append("\n\n### Download\n");
379         progressBody.append(checkoutCommands());
380 
381         return progressBody.toString();
382     }

384     private String checkoutCommands() {
385         var repoUrl = pr.repository().webUrl();
386         return
387            "`$ git fetch " + repoUrl + " " + pr.fetchRef() + ":pull/" + pr.id() + "`\n" +
388            "`$ git checkout pull/" + pr.id() + "`\n";
389     }

391     private String bodyWithoutStatus() {
392         var description = pr.body();
393         var markerIndex = description.lastIndexOf(progressMarker);
394         return (markerIndex < 0 ?
395                 description :
396                 description.substring(0, markerIndex)).trim();
397     }

399     private String updateStatusMessage(String message) {
400         var description = pr.body();
401         var markerIndex = description.lastIndexOf(progressMarker);
402 
403         if (markerIndex >= 0 && description.substring(markerIndex).equals(message)) {
404             log.info("Progress already up to date");
405             return description;
406         }
407         var originalBody = bodyWithoutStatus();
408         if (originalBody.isBlank()) {
409             originalBody = emptyPrBodyMarker;
410         }
411         var newBody = originalBody + "\n" + progressMarker + "\n" + message;
412 
413         // TODO? Retrieve the body again here to lower the chance of concurrent updates
414         pr.setBody(newBody);
415         return newBody;
416     }

418     private String verdictToString(Review.Verdict verdict) {
419         switch (verdict) {
420             case APPROVED:
421                 return "changes are approved";
422             case DISAPPROVED:
423                 return "more changes needed";
424             case NONE:
425                 return "comment added";
426             default:
427                 throw new RuntimeException("Unknown verdict: " + verdict);
428         }
429     }

431     private void updateReviewedMessages(List<Comment> comments, List<Review> reviews) {
432         var reviewTracker = new ReviewTracker(comments, reviews);
433 
434         for (var added : reviewTracker.newReviews().entrySet()) {
435             var body = added.getValue() + "\n" +
436                     "This PR has been reviewed by " +
437                     formatReviewer(added.getKey().reviewer()) + " - " +
438                     verdictToString(added.getKey().verdict()) + ".";
439             pr.addComment(body);
440         }
441     }

443     private Optional<Comment> findComment(List<Comment> comments, String marker) {
444         var self = pr.repository().forge().currentUser();
445         return comments.stream()
446                        .filter(comment -> comment.author().equals(self))
447                        .filter(comment -> comment.body().contains(marker))
448                        .findAny();

451     private String getMergeReadyComment(String commitMessage, List<Review> reviews) {
452         var message = new StringBuilder();
453         message.append("@");
454         message.append(pr.author().userName());
455         message.append(" This change now passes all automated pre-integration checks");
456 
457         try {
458             var hasContributingFile =
459                 !localRepo.files(pr.targetHash(), Path.of("CONTRIBUTING.md")).isEmpty();
460             if (hasContributingFile) {
461                 message.append(". When the change also fulfills all ");
462                 message.append("[project specific requirements](https://github.com/");
463                 message.append(pr.repository().name());
464                 message.append("/blob/");
465                 message.append(pr.targetRef());
466                 message.append("/CONTRIBUTING.md)");
467             }
468         } catch (IOException e) {
469             throw new UncheckedIOException(e);
470         }
471 
472         message.append(", type `/integrate` in a new comment to proceed. After integration, ");
473         message.append("the commit message will be:\n");
474         message.append("```\n");
475         message.append(commitMessage);
476         message.append("\n```\n");
477 
478         message.append("- If you would like to add a summary, use the `/summary` command.\n");
479         message.append("- To credit additional contributors, use the `/contributor` command.\n");
480         message.append("- To add additional solved issues, use the `/issue` command.\n");
481 
482         var divergingCommits = checkablePullRequest.divergingCommits();
483         if (divergingCommits.size() > 0) {
484             message.append("\n");
485             message.append("Since the source branch of this PR was last updated there ");
486             if (divergingCommits.size() == 1) {
487                 message.append("has been 1 commit ");
488             } else {
489                 message.append("have been ");
490                 message.append(divergingCommits.size());
491                 message.append(" commits ");
492             }
493             message.append("pushed to the `");
494             message.append(pr.targetRef());
495             message.append("` branch:\n\n");
496             divergingCommits.stream()
497                             .limit(10)
498                             .forEach(c -> message.append(" * ").append(c.hash().hex()).append(": ").append(c.message().get(0)).append("\n"));
499             if (divergingCommits.size() > 10) {
500                 message.append(" * ... and ").append(divergingCommits.size() - 10).append(" more: ")
501                        .append(pr.repository().webUrl(baseHash.hex(), pr.targetRef())).append("\n");
502             }
503 
504             message.append("\n");
505             message.append("As there are no conflicts, your changes will automatically be rebased on top of ");
506             message.append("these commits when integrating. If you prefer to avoid automatic rebasing, please merge `");
507             message.append(pr.targetRef());
508             message.append("` into your branch, and then specify the current head hash when integrating, like this: ");
509             message.append("`/integrate ");
510             message.append(pr.targetHash());
511             message.append("`.\n");
512         } else {
513             message.append("\n");
514             message.append("There are currently no new commits on the `");
515             message.append(pr.targetRef());
516             message.append("` branch since the last update of the source branch of this PR. If another commit should be pushed before ");
517             message.append("you perform the `/integrate` command, your PR will be automatically rebased. If you would like to avoid ");
518             message.append("potential automatic rebasing, specify the current head hash when integrating, like this: ");
519             message.append("`/integrate ");
520             message.append(pr.targetHash());
521             message.append("`.\n");
522         }
523 
524         if (!ProjectPermissions.mayCommit(censusInstance, pr.author())) {
525             message.append("\n");
526             var contributor = censusInstance.namespace().get(pr.author().id());
527             if (contributor == null) {
528                 message.append("As you are not a known OpenJDK [Author](https://openjdk.java.net/bylaws#author), ");
529             } else {
530                 message.append("As you do not have Committer status in this project, ");
531             }
532 
533             message.append("an existing [Committer](https://openjdk.java.net/bylaws#committer) must agree to ");
534             message.append("[sponsor](https://openjdk.java.net/sponsor/) your change. ");
535             var candidates = reviews.stream()
536                                     .filter(review -> ProjectPermissions.mayCommit(censusInstance, review.reviewer()))
537                                     .map(review -> "@" + review.reviewer().userName())
538                                     .collect(Collectors.joining(", "));
539             if (candidates.length() > 0) {
540                 message.append("Possible candidates are the reviewers of this PR (");
541                 message.append(candidates);
542                 message.append(") but any other Committer may sponsor as well. ");
543             }
544             message.append("\n\n");
545             message.append("➡️ To flag this PR as ready for integration with the above commit message, type ");
546             message.append("`/integrate` in a new comment. (Afterwards, your sponsor types ");
547             message.append("`/sponsor` in a new comment to perform the integration).\n");
548         } else {
549             message.append("\n");
550             message.append("➡️ To integrate this PR with the above commit message to the `" + pr.targetRef() + "` branch, type ");
551             message.append("`/integrate` in a new comment.\n");
552         }
553         message.append(mergeReadyMarker);
554         return message.toString();
555     }

566     private void updateMergeReadyComment(boolean isReady, String commitMessage, List<Comment> comments, List<Review> reviews, boolean rebasePossible) {
567         var existing = findComment(comments, mergeReadyMarker);
568         if (isReady && rebasePossible) {
569             var message = getMergeReadyComment(commitMessage, reviews);
570             if (existing.isEmpty()) {
571                 pr.addComment(message);
572             } else {
573                 pr.updateComment(existing.get().id(), message);
574             }
575         } else {
576             existing.ifPresent(comment -> pr.updateComment(comment.id(), getMergeNoLongerReadyComment()));
577         }
578     }

580     private void addSourceBranchWarningComment(List<Comment> comments) throws IOException {
581         var existing = findComment(comments, sourceBranchWarningMarker);
582         if (existing.isPresent()) {
583             // Only add the comment once per PR
584             return;
585         }
586         var branch = pr.sourceRef();
587         var message = ":warning: @" + pr.author().userName() + " " +
588             "a branch with the same name as the source branch for this pull request (`" + branch + "`) " +
589             "is present in the [target repository](" + pr.repository().nonTransformedWebUrl() + "). " +
590             "If you eventually integrate this pull request then the branch `" + branch + "` " +
591             "in your [personal fork](" + pr.sourceRepository().nonTransformedWebUrl() + ") will diverge once you sync " +
592             "your personal fork with the upstream repository.\n" +
593             "\n" +
594             "To avoid this situation, create a new branch for your changes and reset the `" + branch + "` branch. " +
595             "You can do this by running the following commands in a local repository for your personal fork. " +
596             "_Note_: you do *not* have to name the new branch `NEW-BRANCH-NAME`." +
597             "\n" +
598             "```" +
599             "$ git checkout " + branch + "\n" +
600             "$ git checkout -b NEW-BRANCH-NAME\n" +
601             "$ git branch -f " + branch + " " + baseHash.hex() + "\n" +
602             "$ git push -f origin " + branch + "\n" +
603             "```\n" +
604             "\n" +
605             "Then proceed to create a new pull request with `NEW-BRANCH-NAME` as the source branch and " +
606             "close this one.\n" +
607             sourceBranchWarningMarker;
608         pr.addComment(message);
609     }

611     private void addOutdatedComment(List<Comment> comments) {
612         var existing = findComment(comments, outdatedHelpMarker);
613         if (existing.isPresent()) {
614             // Only add the comment once per PR
615             return;
616         }
617         var message = "@" + pr.author().userName() + " this pull request can not be integrated into " +
618                 "`" + pr.targetRef() + "` due to one or more merge conflicts. To resolve these merge conflicts " +
619                 "and update this pull request you can run the following commands in the local repository for your personal fork:\n" +
620                 "```bash\n" +
621                 "git checkout " + pr.sourceRef() + "\n" +
622                 "git fetch " + pr.repository().webUrl() + " " + pr.targetRef() + "\n" +
623                 "git merge FETCH_HEAD\n" +
624                 "# resolve conflicts and follow the instructions given by git merge\n" +
625                 "git commit -m \"Merge " + pr.targetRef() + "\"\n" +
626                 "git push\n" +
627                 "```\n" +
628                 outdatedHelpMarker;
629         pr.addComment(message);
630     }

632     private void checkStatus() {
633         var checkBuilder = CheckBuilder.create("jcheck", pr.headHash());
634         var censusDomain = censusInstance.configuration().census().domain();
635         Exception checkException = null;
636 
637         try {
638             // Post check in-progress
639             log.info("Starting to run jcheck on PR head");
640             pr.createCheck(checkBuilder.build());
641 
642             var ignored = new PrintWriter(new StringWriter());
643             var rebasePossible = true;
644             var commitHash = pr.headHash();
645             var mergedHash = checkablePullRequest.mergeTarget(ignored);
646             if (mergedHash.isPresent()) {
647                 commitHash = mergedHash.get();
648             } else {
649                 rebasePossible = false;
650             }
651 
652             List<String> additionalErrors = List.of();
653             Hash localHash;
654             try {
655                 localHash = checkablePullRequest.commit(commitHash, censusInstance.namespace(), censusDomain, null);
656             } catch (CommitFailure e) {
657                 additionalErrors = List.of(e.getMessage());
658                 localHash = baseHash;
659             }
660             PullRequestCheckIssueVisitor visitor = checkablePullRequest.createVisitor(localHash, censusInstance);
661             if (!localHash.equals(baseHash)) {
662                 // Determine current status
663                 var additionalConfiguration = AdditionalConfiguration.get(localRepo, localHash, pr.repository().forge().currentUser(), comments);
664                 checkablePullRequest.executeChecks(localHash, censusInstance, visitor, additionalConfiguration);
665                 additionalErrors = botSpecificChecks(localHash);
666             } else {
667                 if (additionalErrors.isEmpty()) {
668                     additionalErrors = List.of("This PR contains no changes");
669                 }
670             }
671             updateCheckBuilder(checkBuilder, visitor, additionalErrors);
672             updateReadyForReview(visitor, additionalErrors);
673 
674             // Calculate and update the status message if needed
675             var statusMessage = getStatusMessage(comments, activeReviews, visitor, additionalErrors);
676             var updatedBody = updateStatusMessage(statusMessage);
677 
678             // Post / update approval messages (only needed if the review itself can't contain a body)
679             if (!pr.repository().forge().supportsReviewBody()) {
680                 updateReviewedMessages(comments, allReviews);
681             }
682 
683             var commit = localRepo.lookup(localHash).orElseThrow();
684             var commitMessage = String.join("\n", commit.message());
685             var readyForIntegration = visitor.messages().isEmpty() && additionalErrors.isEmpty();
686             updateMergeReadyComment(readyForIntegration, commitMessage, comments, activeReviews, rebasePossible);
687             if (readyForIntegration && rebasePossible) {
688                 newLabels.add("ready");
689             } else {
690                 newLabels.remove("ready");
691             }
692             if (!rebasePossible) {
693                 if (!labels.contains("failed-auto-merge")) {
694                     addOutdatedComment(comments);
695                 }
696                 newLabels.add("merge-conflict");
697             } else {
698                 newLabels.remove("merge-conflict");
699             }
700 
701             var branchNames = pr.repository().branches().stream().map(HostedBranch::name).collect(Collectors.toSet());
702             if (!pr.repository().url().equals(pr.sourceRepository().url()) && branchNames.contains(pr.sourceRef())) {
703                 addSourceBranchWarningComment(comments);
704             }
705 
706             // Ensure that the ready for sponsor label is up to date
707             newLabels.remove("sponsor");
708             var readyHash = ReadyForSponsorTracker.latestReadyForSponsor(pr.repository().forge().currentUser(), comments);
709             if (readyHash.isPresent() && readyForIntegration) {
710                 var acceptedHash = readyHash.get();
711                 if (pr.headHash().equals(acceptedHash)) {
712                     newLabels.add("sponsor");
713                 }
714             }
715 
716             // Calculate current metadata to avoid unnecessary future checks
717             var metadata = workItem.getMetadata(pr.title(), updatedBody, pr.comments(), activeReviews, newLabels,
718                                                 censusInstance, pr.targetHash(), pr.isDraft());
719             checkBuilder.metadata(metadata);
720         } catch (Exception e) {
721             log.throwing("CommitChecker", "checkStatus", e);
722             newLabels.remove("ready");
723             checkBuilder.metadata("invalid");
724             checkBuilder.title("Exception occurred during jcheck - the operation will be retried");
725             checkBuilder.summary(e.getMessage());
726             checkBuilder.complete(false);
727             checkException = e;
728         }
729         var check = checkBuilder.build();
730         pr.updateCheck(check);
731 
732         // Synchronize the wanted set of labels
733         for (var newLabel : newLabels) {
734             if (!labels.contains(newLabel)) {
735                 pr.addLabel(newLabel);
736             }
737         }
738         for (var oldLabel : labels) {
739             if (!newLabels.contains(oldLabel)) {
740                 pr.removeLabel(oldLabel);
741             }
742         }
743 
744         // After updating the PR, rethrow any exception to automatically retry on transient errors
745         if (checkException != null) {
746             throw new RuntimeException("Exception during jcheck", checkException);
747         }
748     }
