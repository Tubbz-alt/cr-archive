 23 package org.openjdk.skara.forge;
 24 
 25 import java.net.URI;
 26 import java.util.List;
 27 import org.junit.jupiter.api.Test;
 28 import static org.junit.jupiter.api.Assertions.*;
 29 
 30 class PullRequestBodyTests {
 31     @Test
 32     void parseEmpty() {
 33         var body = PullRequestBody.parse(List.of());
 34         assertTrue(body.issues().isEmpty());
 35         assertTrue(body.contributors().isEmpty());
 36     }
 37 
 38     @Test
 39     void parseText() {
 40         var text = List.of(
 41             "Hi all,",
 42             "",
 43             "please review this patch!",
 44             ""
 45         );
 46         var body = PullRequestBody.parse(text);
 47         assertTrue(body.issues().isEmpty());
 48         assertTrue(body.contributors().isEmpty());
 49     }
 50 
 51     @Test
 52     void parseEmptySections() {
 53         var text = List.of(
 54             "### Issues",
 55             "",
 56             "### Contributors",
 57             ""
 58         );
 59         var body = PullRequestBody.parse(text);
 60         assertTrue(body.issues().isEmpty());
 61         assertTrue(body.contributors().isEmpty());
 62     }
 63 
 64     @Test
 65     void parseSingleIssue() {
 66         var text = List.of(
 67             "### Issues",
 68             " * [JDK-1234567](https://bugs/JDK-1234567): A bug"
 69         );
 70         var body = PullRequestBody.parse(text);
 71         assertTrue(body.contributors().isEmpty());
 72         assertEquals(1, body.issues().size());
 73         assertEquals(URI.create("https://bugs/JDK-1234567"),
 74                      body.issues().get(0));
 75     }
 76 
 77     @Test
 78     void parseMultipleIssues() {
 79         var text = List.of(
 80             "### Issues",
 81             " * [JDK-1234567](https://bugs/JDK-1234567): A bug",
 82             " * [JDK-4567890](https://bugs/JDK-4567890): Another bug"
 83         );
 84         var body = PullRequestBody.parse(text);
 85         assertTrue(body.contributors().isEmpty());
 86         assertEquals(2, body.issues().size());
 87         assertEquals(URI.create("https://bugs/JDK-1234567"),
 88                      body.issues().get(0));
 89         assertEquals(URI.create("https://bugs/JDK-4567890"),
 90                      body.issues().get(1));
 91     }
 92 
 93     @Test
 94     void parseSingleContributor() {
 95         var text = List.of(
 96             "### Contributors",
 97             " * Foo Bar `<foo@bar.com>`"
 98         );
 99         var body = PullRequestBody.parse(text);
100         assertTrue(body.issues().isEmpty());
101         assertEquals(1, body.contributors().size());
102         assertEquals("Foo Bar <foo@bar.com>", body.contributors().get(0));
103     }
104 
105     @Test
106     void parseMultipleContributors() {
107         var text = List.of(
108             "### Contributors",
109             " * Foo Bar `<foo@bar.com>`",
110             " * J Duke `<j@duke.com>`"
111         );
112         var body = PullRequestBody.parse(text);
113         assertTrue(body.issues().isEmpty());
114         assertEquals(2, body.contributors().size());
115         assertEquals("Foo Bar <foo@bar.com>", body.contributors().get(0));
116         assertEquals("J Duke <j@duke.com>", body.contributors().get(1));
117     }
118 
119     @Test
120     void parseMultipleContributorsAndMultipleIssues() {
121         var text = List.of(
122             "### Issues",
123             " * [JDK-1234567](https://bugs/JDK-1234567): A bug",
124             " * [JDK-4567890](https://bugs/JDK-4567890): Another bug",
125             "### Contributors",
126             " * Foo Bar `<foo@bar.com>`",
127             " * J Duke `<j@duke.com>`"
128         );
129         var body = PullRequestBody.parse(text);
130         assertEquals(URI.create("https://bugs/JDK-1234567"),
131                      body.issues().get(0));
132         assertEquals(URI.create("https://bugs/JDK-4567890"),
133                      body.issues().get(1));
134         assertEquals(2, body.contributors().size());
135         assertEquals("Foo Bar <foo@bar.com>", body.contributors().get(0));
136         assertEquals("J Duke <j@duke.com>", body.contributors().get(1));
137     }
138 
139     @Test
140     void parseMultipleContributorsAndMultipleIssuesWithAdditionalText() {
141         var text = List.of(
142             "Hi all,",
143             "",
144             "please review this patch!",
145             "",
146             "<!-- A COMMENT ->",
147             "### Progress",
148             "",
149             "### Issues",
150             " * [JDK-1234567](https://bugs/JDK-1234567): A bug",
151             " * [JDK-4567890](https://bugs/JDK-4567890): Another bug",
152             "",
153             "### Contributors",
154             " * Foo Bar `<foo@bar.com>`",
155             " * J Duke `<j@duke.com>`",
156             "",
157             "### Download"
158         );
159         var body = PullRequestBody.parse(text);
160         assertEquals(URI.create("https://bugs/JDK-1234567"),
161                      body.issues().get(0));
162         assertEquals(URI.create("https://bugs/JDK-4567890"),
163                      body.issues().get(1));
164         assertEquals(2, body.contributors().size());
165         assertEquals("Foo Bar <foo@bar.com>", body.contributors().get(0));
166         assertEquals("J Duke <j@duke.com>", body.contributors().get(1));
167     }
168 }
