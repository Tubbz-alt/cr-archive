 99     private List<Issue> issues() {
100         var issue = Issue.fromStringRelaxed(pr.title());
101         if (issue.isPresent()) {
102             var issues = new ArrayList<Issue>();
103             issues.add(issue.get());

114     private List<org.openjdk.skara.issuetracker.Issue> issuesOfDisallowedType() {
115         var issueProject = issueProject();
116         var allowed = allowedIssueTypes();
117         if (issueProject != null && allowed != null) {
118             return issues().stream()
119                            .filter(i -> i.project().equals(Optional.of(issueProject.name())))
120                            .map(i -> issueProject.issue(i.shortId()))
121                            .filter(Optional::isPresent)
122                            .map(Optional::get)
123                            .filter(i -> i.properties().containsKey("issuetype"))
124                            .filter(i -> !allowed.contains(i.properties().get("issuetype").asString()))
125                            .collect(Collectors.toList());
126         }
127         return List.of();
128     }

130     private List<String> allowedTargetBranches() {
131         return pr.repository()
132                  .branches()
133                  .stream()
134                  .map(HostedBranch::name)
135                  .map(name -> workItem.bot.allowedTargetBranches().matcher(name))
136                  .filter(Matcher::matches)
137                  .map(Matcher::group)
138                  .collect(Collectors.toList());

141     // For unknown contributors, check that all commits have the same name and email
142     private boolean checkCommitAuthor(List<CommitMetadata> commits) throws IOException {
143         var author = censusInstance.namespace().get(pr.author().id());
144         if (author != null) {
145             return true;
146         }
147 
148         var names = new HashSet<String>();
149         var emails = new HashSet<String>();
150 
151         for (var commit : commits) {
152             names.add(commit.author().name());
153             emails.add(commit.author().email());
154         }
155 
156         return ((names.size() == 1) && emails.size() == 1);
157     }

159     // Additional bot-specific checks that are not handled by JCheck
160     private List<String> botSpecificChecks(Hash finalHash) throws IOException {
161         var ret = new ArrayList<String>();
162 
163         if (bodyWithoutStatus().isBlank()) {
164             var error = "The pull request body must not be empty.";
165             ret.add(error);
166         }
167 
168         if (!isTargetBranchAllowed()) {
169             var error = "The branch `" + pr.targetRef() + "` is not allowed as target branch. The allowed target branches are:\n" +
170                     allowedTargetBranches().stream()
171                     .map(name -> "   - " + name)
172                     .collect(Collectors.joining("\n"));
173             ret.add(error);
174         }
175 
176         var disallowedIssues = issuesOfDisallowedType();
177         if (!disallowedIssues.isEmpty()) {
178             var s = disallowedIssues.size() > 1 ? "s " : " ";
179             var are = disallowedIssues.size() > 1 ? "are" : "is";
180             var links = disallowedIssues.stream()
181                                         .map(i -> "[" + i.id() + "](" + i.webUrl() + ")")
182                                         .collect(Collectors.toList());
183             var error = "The issue" + s + String.join(",", links) + " " + are + " not of the expected type. The allowed issue types are:\n" +
184                 allowedIssueTypes().stream()
185                 .map(name -> "   - " + name)
186                 .collect(Collectors.joining("\n"));
187             ret.add(error);
188         }
189 
190         var headHash = pr.headHash();
191         var originalCommits = localRepo.commitMetadata(baseHash, headHash);
192 
193         if (!checkCommitAuthor(originalCommits)) {
194             var error = "For contributors who are not existing OpenJDK Authors, commit attribution will be taken from " +
195                     "the commits in the PR. However, the commits in this PR have inconsistent user names and/or " +
196                     "email addresses. Please amend the commits.";
197             ret.add(error);
198         }
199 
200         for (var blocker : workItem.bot.blockingCheckLabels().entrySet()) {
201             if (labels.contains(blocker.getKey())) {
202                 ret.add(blocker.getValue());
203             }
204         }
205 
206         return ret;
207     }

209     private void updateCheckBuilder(CheckBuilder checkBuilder, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
210         if (visitor.isReadyForReview() && additionalErrors.isEmpty()) {
211             checkBuilder.complete(true);
212         } else {
213             checkBuilder.title("Required");
214             var summary = Stream.concat(visitor.messages().stream(), additionalErrors.stream())
215                                 .sorted()
216                                 .map(m -> "- " + m)
217                                 .collect(Collectors.joining("\n"));
218             checkBuilder.summary(summary);
219             for (var annotation : visitor.getAnnotations()) {
220                 checkBuilder.annotation(annotation);
221             }
222             checkBuilder.complete(false);
223         }
224     }

226     private void updateReadyForReview(PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
227         // Additional errors are not allowed
228         if (!additionalErrors.isEmpty()) {
229             newLabels.remove("rfr");
230             return;
231         }
232 
233         // Draft requests are not for review
234         if (pr.isDraft()) {
235             newLabels.remove("rfr");
236             return;
237         }
238 
239         // Check if the visitor found any issues that should be resolved before reviewing
240         if (visitor.isReadyForReview()) {
241             newLabels.add("rfr");
242         } else {
243             newLabels.remove("rfr");
244         }
245     }

247     private String getRole(String username) {
248         var project = censusInstance.project();
249         var version = censusInstance.census().version().format();
250         if (project.isReviewer(username, version)) {
251             return "**Reviewer**";
252         } else if (project.isCommitter(username, version)) {
253             return "Committer";
254         } else if (project.isAuthor(username, version)) {
255             return "Author";
256         } else {
257             return "no project role";
258         }
259     }

261     private String formatReviewer(HostUser reviewer) {
262         var namespace = censusInstance.namespace();
263         var contributor = namespace.get(reviewer.id());
264         if (contributor == null) {
265             return reviewer.userName() + " (no known " + namespace.name() + " user name / role)";
266         } else {
267             var userNameLink = "[" + contributor.username() + "](@" + reviewer.userName() + ")";
268             return contributor.fullName().orElse(contributor.username()) + " (" + userNameLink + " - " +
269                     getRole(contributor.username()) + ")";
270         }
271     }

286     private Optional<String> getReviewersList(List<Review> reviews) {
287         var reviewers = reviews.stream()
288                                .filter(review -> review.verdict() == Review.Verdict.APPROVED)
289                                .map(review -> {
290                                    var entry = " * " + formatReviewer(review.reviewer());
291                                    if (!review.hash().equals(pr.headHash())) {
292                                        if (ignoreStaleReviews) {
293                                            entry += " 🔄 Re-review required (review applies to " + review.hash() + ")";
294                                        } else {
295                                            entry += " ⚠️ Review applies to " + review.hash();
296                                        }
297                                    }
298                                    return entry;
299                                })
300                                .collect(Collectors.joining("\n"));
301 
302         // Check for manually added reviewers
303         if (!ignoreStaleReviews) {
304             var namespace = censusInstance.namespace();
305             var allReviewers = PullRequestUtils.reviewerNames(reviews, namespace);
306             var additionalEntries = new ArrayList<String>();
307             for (var additional : Reviewers.reviewers(pr.repository().forge().currentUser(), comments)) {
308                 if (!allReviewers.contains(additional)) {
309                     additionalEntries.add(" * " + additional + " - " + getRole(additional) + " ⚠️ Added manually");
310                 }
311             }
312             if (!reviewers.isBlank()) {
313                 reviewers += "\n";
314             }
315             reviewers += String.join("\n", additionalEntries);
316         }
317 
318         if (reviewers.length() > 0) {
319             return Optional.of(reviewers);
320         } else {
321             return Optional.empty();
322         }
323     }

342     private boolean relaxedEquals(String s1, String s2) {
343         return s1.trim()
344                  .replaceAll("\\s+", " ")
345                  .equalsIgnoreCase(s2.trim()
346                                      .replaceAll("\\s+", " "));

350     private String getStatusMessage(List<Comment> comments, List<Review> reviews, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
351         var progressBody = new StringBuilder();
352         progressBody.append("---------\n");
353         progressBody.append("### Progress\n");
354         progressBody.append(getChecksList(visitor));
355 
356         var allAdditionalErrors = Stream.concat(visitor.hiddenMessages().stream(), additionalErrors.stream())
357                                         .sorted()
358                                         .collect(Collectors.toList());
359         if (!allAdditionalErrors.isEmpty()) {
360             progressBody.append("\n\n### Error");
361             if (allAdditionalErrors.size() > 1) {
362                 progressBody.append("s");
363             }
364             progressBody.append("\n");
365             progressBody.append(getAdditionalErrorsList(allAdditionalErrors));
366         }
367 
368         var issues = issues();
369         var issueProject = issueProject();
370         if (issueProject != null && !issues.isEmpty()) {
371             progressBody.append("\n\n### Issue");
372             if (issues.size() > 1) {
373                 progressBody.append("s");
374             }
375             progressBody.append("\n");
376             for (var currentIssue : issues) {
377                 progressBody.append(" * ");
378                 if (currentIssue.project().isPresent() && !currentIssue.project().get().equals(issueProject.name())) {
379                     progressBody.append("⚠️ Issue `");
380                     progressBody.append(currentIssue.id());
381                     progressBody.append("` does not belong to the `");
382                     progressBody.append(issueProject.name());
383                     progressBody.append("` project.");
384                 } else {
385                     try {
386                         var iss = issueProject.issue(currentIssue.shortId());
387                         if (iss.isPresent()) {
388                             progressBody.append("[");
389                             progressBody.append(iss.get().id());
390                             progressBody.append("](");
391                             progressBody.append(iss.get().webUrl());
392                             progressBody.append("): ");
393                             progressBody.append(iss.get().title());
394                             if (!relaxedEquals(iss.get().title(), currentIssue.description())) {
395                                 progressBody.append(" ⚠️ Title mismatch between PR and JBS.");
396                             }
397                             progressBody.append("\n");
398                         } else {
399                             progressBody.append("⚠️ Failed to retrieve information on issue `");
400                             progressBody.append(currentIssue.id());
401                             progressBody.append("`.\n");
402                         }
403                     } catch (RuntimeException e) {
404                         progressBody.append("⚠️ Temporary failure when trying to retrieve information on issue `");
405                         progressBody.append(currentIssue.id());
406                         progressBody.append("`.\n");
407                     }
408                 }
409             }
410         }
411 
412         getReviewersList(reviews).ifPresent(reviewers -> {
413             progressBody.append("\n\n### Reviewers\n");
414             progressBody.append(reviewers);
415         });
416 
417         getContributorsList(comments).ifPresent(contributors -> {
418             progressBody.append("\n\n### Contributors\n");
419             progressBody.append(contributors);
420         });
421 
422         progressBody.append("\n\n### Download\n");
423         progressBody.append(checkoutCommands());
424 
425         return progressBody.toString();
426     }

428     private String checkoutCommands() {
429         var repoUrl = pr.repository().webUrl();
430         return
431            "`$ git fetch " + repoUrl + " " + pr.fetchRef() + ":pull/" + pr.id() + "`\n" +
432            "`$ git checkout pull/" + pr.id() + "`\n";

435     private String bodyWithoutStatus() {
436         var description = pr.body();
437         var markerIndex = description.lastIndexOf(progressMarker);
438         return (markerIndex < 0 ?
439                 description :
440                 description.substring(0, markerIndex)).trim();
441     }

443     private String updateStatusMessage(String message) {
444         var description = pr.body();
445         var markerIndex = description.lastIndexOf(progressMarker);
446 
447         if (markerIndex >= 0 && description.substring(markerIndex).equals(message)) {
448             log.info("Progress already up to date");
449             return description;
450         }
451         var originalBody = bodyWithoutStatus();
452         if (originalBody.isBlank()) {
453             originalBody = emptyPrBodyMarker;
454         }
455         var newBody = originalBody + "\n" + progressMarker + "\n" + message;
456 
457         // TODO? Retrieve the body again here to lower the chance of concurrent updates
458         pr.setBody(newBody);
459         return newBody;
460     }

462     private String updateTitle() {
463         var title = pr.title();
464         var m = ISSUE_ID_PATTERN.matcher(title);
465         var project = issueProject();
466 
467         var newTitle = title;
468         if (m.matches() && project != null) {
469             var id = m.group(1);
470             var issue = project.issue(id);
471             if (issue.isPresent()) {
472                 newTitle = id + ": " + issue.get().title();
473             }
474         }
475 
476         if (!title.equals(newTitle)) {
477             pr.setTitle(newTitle);
478         }
479 
480         return newTitle;
481     }

483     private String verdictToString(Review.Verdict verdict) {
484         switch (verdict) {
485             case APPROVED:
486                 return "changes are approved";
487             case DISAPPROVED:
488                 return "more changes needed";
489             case NONE:
490                 return "comment added";
491             default:
492                 throw new RuntimeException("Unknown verdict: " + verdict);
493         }
494     }

516     private String getMergeReadyComment(String commitMessage, List<Review> reviews) {
517         var message = new StringBuilder();
518         message.append("@");
519         message.append(pr.author().userName());
520         message.append(" This change now passes all automated pre-integration checks");
521 
522         try {
523             var hasContributingFile =
524                 !localRepo.files(pr.targetHash(), Path.of("CONTRIBUTING.md")).isEmpty();
525             if (hasContributingFile) {
526                 message.append(". When the change also fulfills all ");
527                 message.append("[project specific requirements](https://github.com/");
528                 message.append(pr.repository().name());
529                 message.append("/blob/");
530                 message.append(pr.targetRef());
531                 message.append("/CONTRIBUTING.md)");
532             }
533         } catch (IOException e) {
534             throw new UncheckedIOException(e);
535         }
536 
537         message.append(", type `/integrate` in a new comment to proceed. After integration, ");
538         message.append("the commit message will be:\n");
539         message.append("```\n");
540         message.append(commitMessage);
541         message.append("\n```\n");
542 
543         message.append("- If you would like to add a summary, use the `/summary` command.\n");
544         message.append("- To credit additional contributors, use the `/contributor` command.\n");
545         message.append("- To add additional solved issues, use the `/issue` command.\n");
546 
547         var divergingCommits = checkablePullRequest.divergingCommits();
548         if (divergingCommits.size() > 0) {
549             message.append("\n");
550             message.append("Since the source branch of this PR was last updated there ");
551             if (divergingCommits.size() == 1) {
552                 message.append("has been 1 commit ");
553             } else {
554                 message.append("have been ");
555                 message.append(divergingCommits.size());
556                 message.append(" commits ");
557             }
558             message.append("pushed to the `");
559             message.append(pr.targetRef());
560             message.append("` branch:\n\n");
561             divergingCommits.stream()
562                             .limit(10)
563                             .forEach(c -> message.append(" * ").append(c.hash().hex()).append(": ").append(c.message().get(0)).append("\n"));
564             if (divergingCommits.size() > 10) {
565                 message.append(" * ... and ").append(divergingCommits.size() - 10).append(" more: ")
566                        .append(pr.repository().webUrl(baseHash.hex(), pr.targetRef())).append("\n");
567             }
568 
569             message.append("\n");
570             message.append("As there are no conflicts, your changes will automatically be rebased on top of ");
571             message.append("these commits when integrating. If you prefer to avoid automatic rebasing, please merge `");
572             message.append(pr.targetRef());
573             message.append("` into your branch, and then specify the current head hash when integrating, like this: ");
574             message.append("`/integrate ");
575             message.append(pr.targetHash());
576             message.append("`.\n");
577         } else {
578             message.append("\n");
579             message.append("There are currently no new commits on the `");
580             message.append(pr.targetRef());
581             message.append("` branch since the last update of the source branch of this PR. If another commit should be pushed before ");
582             message.append("you perform the `/integrate` command, your PR will be automatically rebased. If you would like to avoid ");
583             message.append("potential automatic rebasing, specify the current head hash when integrating, like this: ");
584             message.append("`/integrate ");
585             message.append(pr.targetHash());
586             message.append("`.\n");
587         }
588 
589         if (!ProjectPermissions.mayCommit(censusInstance, pr.author())) {
590             message.append("\n");
591             var contributor = censusInstance.namespace().get(pr.author().id());
592             if (contributor == null) {
593                 message.append("As you are not a known OpenJDK [Author](https://openjdk.java.net/bylaws#author), ");
594             } else {
595                 message.append("As you do not have Committer status in this project, ");
596             }
597 
598             message.append("an existing [Committer](https://openjdk.java.net/bylaws#committer) must agree to ");
599             message.append("[sponsor](https://openjdk.java.net/sponsor/) your change. ");
600             var candidates = reviews.stream()
601                                     .filter(review -> ProjectPermissions.mayCommit(censusInstance, review.reviewer()))
602                                     .map(review -> "@" + review.reviewer().userName())
603                                     .collect(Collectors.joining(", "));
604             if (candidates.length() > 0) {
605                 message.append("Possible candidates are the reviewers of this PR (");
606                 message.append(candidates);
607                 message.append(") but any other Committer may sponsor as well. ");
608             }
609             message.append("\n\n");
610             message.append("➡️ To flag this PR as ready for integration with the above commit message, type ");
611             message.append("`/integrate` in a new comment. (Afterwards, your sponsor types ");
612             message.append("`/sponsor` in a new comment to perform the integration).\n");
613         } else {
614             message.append("\n");
615             message.append("➡️ To integrate this PR with the above commit message to the `" + pr.targetRef() + "` branch, type ");
616             message.append("`/integrate` in a new comment.\n");
617         }
618         message.append(mergeReadyMarker);
619         return message.toString();
620     }

622     private String getMergeNoLongerReadyComment() {
623         var message = new StringBuilder();
624         message.append("@");
625         message.append(pr.author().userName());
626         message.append(" This change is no longer ready for integration - check the PR body for details.\n");
627         message.append(mergeReadyMarker);
628         return message.toString();
629     }

631     private void updateMergeReadyComment(boolean isReady, String commitMessage, List<Comment> comments, List<Review> reviews, boolean rebasePossible) {
632         var existing = findComment(comments, mergeReadyMarker);
633         if (isReady && rebasePossible) {
634             var message = getMergeReadyComment(commitMessage, reviews);
635             if (existing.isEmpty()) {
636                 pr.addComment(message);
637             } else {
638                 pr.updateComment(existing.get().id(), message);
639             }
640         } else {
641             existing.ifPresent(comment -> pr.updateComment(comment.id(), getMergeNoLongerReadyComment()));
642         }

645     private void addSourceBranchWarningComment(List<Comment> comments) throws IOException {
646         var existing = findComment(comments, sourceBranchWarningMarker);
647         if (existing.isPresent()) {
648             // Only add the comment once per PR
649             return;
650         }
651         var branch = pr.sourceRef();
652         var message = ":warning: @" + pr.author().userName() + " " +
653             "a branch with the same name as the source branch for this pull request (`" + branch + "`) " +

655             "If you eventually integrate this pull request then the branch `" + branch + "` " +

657             "your personal fork with the upstream repository.\n" +
658             "\n" +
659             "To avoid this situation, create a new branch for your changes and reset the `" + branch + "` branch. " +
660             "You can do this by running the following commands in a local repository for your personal fork. " +
661             "_Note_: you do *not* have to name the new branch `NEW-BRANCH-NAME`." +
662             "\n" +
663             "```" +
664             "$ git checkout " + branch + "\n" +
665             "$ git checkout -b NEW-BRANCH-NAME\n" +
666             "$ git branch -f " + branch + " " + baseHash.hex() + "\n" +
667             "$ git push -f origin " + branch + "\n" +
668             "```\n" +
669             "\n" +
670             "Then proceed to create a new pull request with `NEW-BRANCH-NAME` as the source branch and " +
671             "close this one.\n" +
672             sourceBranchWarningMarker;
673         pr.addComment(message);
674     }

693                 outdatedHelpMarker;

697     private void checkStatus() {
698         var checkBuilder = CheckBuilder.create("jcheck", pr.headHash());
699         var censusDomain = censusInstance.configuration().census().domain();
700         Exception checkException = null;
701 
702         try {
703             // Post check in-progress
704             log.info("Starting to run jcheck on PR head");
705             pr.createCheck(checkBuilder.build());
706 
707             var ignored = new PrintWriter(new StringWriter());
708             var rebasePossible = true;
709             var commitHash = pr.headHash();
710             var mergedHash = checkablePullRequest.mergeTarget(ignored);
711             if (mergedHash.isPresent()) {
712                 commitHash = mergedHash.get();
713             } else {
714                 rebasePossible = false;
715             }
716 
717             List<String> additionalErrors = List.of();
718             Hash localHash;
719             try {
720                 localHash = checkablePullRequest.commit(commitHash, censusInstance.namespace(), censusDomain, null);
721             } catch (CommitFailure e) {
722                 additionalErrors = List.of(e.getMessage());
723                 localHash = baseHash;
724             }
725             PullRequestCheckIssueVisitor visitor = checkablePullRequest.createVisitor(localHash, censusInstance);
726             if (!localHash.equals(baseHash)) {
727                 // Determine current status
728                 var additionalConfiguration = AdditionalConfiguration.get(localRepo, localHash, pr.repository().forge().currentUser(), comments);
729                 checkablePullRequest.executeChecks(localHash, censusInstance, visitor, additionalConfiguration);
730                 additionalErrors = botSpecificChecks(localHash);
731             } else {
732                 if (additionalErrors.isEmpty()) {
733                     additionalErrors = List.of("This PR contains no changes");
734                 }
735             }
736             updateCheckBuilder(checkBuilder, visitor, additionalErrors);
737             updateReadyForReview(visitor, additionalErrors);
738 
739             // Calculate and update the status message if needed
740             var statusMessage = getStatusMessage(comments, activeReviews, visitor, additionalErrors);
741             var updatedBody = updateStatusMessage(statusMessage);
742             var title = updateTitle();
743 
744             // Post / update approval messages (only needed if the review itself can't contain a body)
745             if (!pr.repository().forge().supportsReviewBody()) {
746                 updateReviewedMessages(comments, allReviews);
747             }
748 
749             var commit = localRepo.lookup(localHash).orElseThrow();
750             var commitMessage = String.join("\n", commit.message());
751             var readyForIntegration = visitor.messages().isEmpty() && additionalErrors.isEmpty();
752             updateMergeReadyComment(readyForIntegration, commitMessage, comments, activeReviews, rebasePossible);
753             if (readyForIntegration && rebasePossible) {
754                 newLabels.add("ready");
755             } else {
756                 newLabels.remove("ready");
757             }
758             if (!rebasePossible) {
759                 if (!labels.contains("failed-auto-merge")) {
760                     addOutdatedComment(comments);
761                 }
762                 newLabels.add("merge-conflict");
763             } else {
764                 newLabels.remove("merge-conflict");
765             }
766 
767             var branchNames = pr.repository().branches().stream().map(HostedBranch::name).collect(Collectors.toSet());
768             if (!pr.repository().url().equals(pr.sourceRepository().url()) && branchNames.contains(pr.sourceRef())) {
769                 addSourceBranchWarningComment(comments);
770             }
771 




772             // Ensure that the ready for sponsor label is up to date
773             newLabels.remove("sponsor");
774             var readyHash = ReadyForSponsorTracker.latestReadyForSponsor(pr.repository().forge().currentUser(), comments);
775             if (readyHash.isPresent() && readyForIntegration) {
776                 var acceptedHash = readyHash.get();
777                 if (pr.headHash().equals(acceptedHash)) {
778                     newLabels.add("sponsor");
779                 }
780             }
781 
782             // Calculate current metadata to avoid unnecessary future checks
783             var metadata = workItem.getMetadata(title, updatedBody, pr.comments(), activeReviews, newLabels,
784                                                 censusInstance, pr.targetHash(), pr.isDraft());
785             checkBuilder.metadata(metadata);
786         } catch (Exception e) {
787             log.throwing("CommitChecker", "checkStatus", e);
788             newLabels.remove("ready");
789             checkBuilder.metadata("invalid");
790             checkBuilder.title("Exception occurred during jcheck - the operation will be retried");
791             checkBuilder.summary(e.getMessage());
792             checkBuilder.complete(false);
793             checkException = e;
794         }
795         var check = checkBuilder.build();
796         pr.updateCheck(check);
797 
798         // Synchronize the wanted set of labels
799         for (var newLabel : newLabels) {
800             if (!labels.contains(newLabel)) {
801                 pr.addLabel(newLabel);
802             }
803         }
804         for (var oldLabel : labels) {
805             if (!newLabels.contains(oldLabel)) {
806                 pr.removeLabel(oldLabel);
807             }
808         }
809 
810         // After updating the PR, rethrow any exception to automatically retry on transient errors
811         if (checkException != null) {
812             throw new RuntimeException("Exception during jcheck", checkException);
813         }
814     }
