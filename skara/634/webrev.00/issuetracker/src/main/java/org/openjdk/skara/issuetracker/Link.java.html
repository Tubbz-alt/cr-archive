 23 package org.openjdk.skara.issuetracker;
 24 
 25 import java.net.URI;
 26 import java.util.*;
 27 
 28 public class Link {
 29     private final URI uri;
 30     private final String title;
 31     private final String relationship;
 32     private final String summary;
 33     private final URI iconUrl;
 34     private final String iconTitle;
 35     private final URI statusIconUrl;
 36     private final String statusIconTitle;
 37     private final boolean resolved;
 38     private final Issue linked;
 39 
 40     Link(URI uri, String title, String relationship, String summary, URI iconUrl, String iconTitle, URI statusIconUrl, String statusIconTitle, boolean resolved, Issue linked) {
 41         this.uri = uri;
 42         this.title = title;
 43         this.relationship = relationship;
 44         this.summary = summary;
 45         this.iconUrl = iconUrl;
 46         this.iconTitle = iconTitle;
 47         this.statusIconUrl = statusIconUrl;
 48         this.statusIconTitle = statusIconTitle;
 49         this.resolved = resolved;
 50         this.linked = linked;
 51     }
 52 
 53     public static WebLinkBuilder create(URI uri, String title) {
 54         return new WebLinkBuilder(uri, title);
 55     }
 56 
 57     public static IssueLinkBuilder create(Issue issue, String relationship) {
 58         return new IssueLinkBuilder(issue, relationship);
 59     }
 60 
 61     public Optional<URI> uri() {
 62         return Optional.ofNullable(uri);
 63     }
 64 
 65     public Optional<String> title() {
 66         return Optional.ofNullable(title);
 67     }
 68 
 69     public Optional<Issue> issue() {
 70         return Optional.ofNullable(linked);
 71     }
 72 
 73     public Optional<String> relationship() {
 74         return Optional.ofNullable(relationship);
 75     }
 76 
 77     public Optional<String> summary() {
 78         return Optional.ofNullable(summary);
 79     }
 80 
 81     public Optional<URI> iconUrl() {
 82         return Optional.ofNullable(iconUrl);
 83     }
 84 
 85     public Optional<String> iconTitle() {
 86         return Optional.ofNullable(iconTitle);
 87     }
 88 
 89     public Optional<URI> statusIconUrl() {
 90         return Optional.ofNullable(statusIconUrl);
 91     }
 92 
 93     public Optional<String> statusIconTitle() {
 94         return Optional.ofNullable(statusIconTitle);
 95     }
 96 
 97     public boolean resolved() {
 98         return resolved;
 99     }
100 
101     @Override
102     public boolean equals(Object o) {
103         if (this == o) {
104             return true;
105         }
106         if (o == null || getClass() != o.getClass()) {
107             return false;
108         }
109         Link link = (Link) o;
110         return resolved == link.resolved &&
111                 Objects.equals(uri, link.uri) &&
112                 Objects.equals(title, link.title) &&
113                 Objects.equals(relationship, link.relationship) &&
114                 Objects.equals(linked, link.linked) &&
115                 Objects.equals(summary, link.summary) &&
116                 Objects.equals(iconUrl, link.iconUrl) &&
117                 Objects.equals(iconTitle, link.iconTitle) &&
118                 Objects.equals(statusIconUrl, link.statusIconUrl) &&
119                 Objects.equals(statusIconTitle, link.statusIconTitle);
120     }
121 
122     @Override
123     public int hashCode() {
124         return Objects.hash(uri, title, relationship, summary, iconUrl, iconTitle, statusIconUrl, statusIconTitle, resolved);
125     }
126 
127     @Override
128     public String toString() {
129         return "Link{" +
130                 "uri=" + uri +
131                 ", title='" + title + '\'' +
132                 ", relationship='" + relationship + '\'' +
133                 ", linked='" + linked + '\'' +
134                 ", summary='" + summary + '\'' +
135                 ", iconUrl=" + iconUrl +
136                 ", iconTitle='" + iconTitle + '\'' +
137                 ", statusIconUrl=" + statusIconUrl +
138                 ", statusIconTitle='" + statusIconTitle + '\'' +
139                 ", resolved=" + resolved +
140                 '}';
141     }
142 }
