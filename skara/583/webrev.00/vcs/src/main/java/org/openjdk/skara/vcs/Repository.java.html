136     static Repository init(Path p, VCS vcs) throws IOException {
137         switch (vcs) {
138             case GIT:
139                 return new GitRepository(p).init();
140             case HG:
141                 return new HgRepository(p).init();
142             default:
143                 throw new IllegalArgumentException("Invalid enum value: " + vcs);
144         }
145     }
146 
147     static Optional<Repository> get(Path p) throws IOException {
148         var gitRepo = GitRepository.get(p);
149         var hgRepo = HgRepository.get(p);
150         if (gitRepo.isPresent() && hgRepo.isEmpty()) {
151             return gitRepo;
152         } else if (gitRepo.isEmpty() && hgRepo.isPresent()) {
153             return hgRepo;
154         } else if (gitRepo.isPresent() && hgRepo.isPresent()) {
155             // Nested repositories
156             var gitRoot = gitRepo.get().root();
157             var hgRoot = hgRepo.get().root();
158             if (gitRoot.equals(hgRoot)) {
159                 throw new IOException(p.toString() + " contains both a hg and git repository");
160             }
161             if (hgRoot.startsWith(gitRoot)) {
162                 return hgRepo;
163             } else {
164                 return gitRepo;
165             }
166         }
167         return Optional.empty();
168     }
169 
170     static boolean exists(Path p) throws IOException {
171         return get(p).isPresent();
172     }
173 
174     static Repository materialize(Path p, URI remote, String ref) throws IOException {
175         return materialize(p, remote, ref, true);
176     }
177 
178     static Repository materialize(Path p, URI remote, String ref, boolean checkout) throws IOException {
179         var localRepo = remote.getPath().endsWith(".git") ?
180             Repository.init(p, VCS.GIT) : Repository.init(p, VCS.HG);
181         if (!localRepo.exists()) {
182             localRepo.init();
183         } else if (!localRepo.isHealthy()) {
184             localRepo.reinitialize();
185         } else {
186             try {
187                 localRepo.clean();
188             } catch (IOException e) {
189                 localRepo.reinitialize();
190             }
191         }
192 
193         var baseHash = localRepo.fetch(remote, ref);
194 
195         if (checkout) {
196             try {
197                 localRepo.checkout(baseHash, true);
198             } catch (IOException e) {
199                 localRepo.reinitialize();
200                 baseHash = localRepo.fetch(remote, ref);
201                 localRepo.checkout(baseHash, true);
202             }
203         }
204 
205         return localRepo;
206     }

208     static Repository clone(URI from) throws IOException {
209         var to = Path.of(from).getFileName();
210         if (to.toString().endsWith(".git")) {
211             to = Path.of(to.toString().replace(".git", ""));
212         }
213         return clone(from, to);
214     }
215 
216     static Repository clone(URI from, Path to) throws IOException {
217         return clone(from, to, false);
218     }
219 
220     static Repository clone(URI from, Path to, boolean isBare) throws IOException {
221         return clone(from, to, isBare, null);
222     }
223 
224     static Repository clone(URI from, Path to, boolean isBare, Path seed) throws IOException {
225         return from.getPath().endsWith(".git") ?
226             GitRepository.clone(from, to, isBare, seed) : HgRepository.clone(from, to, isBare, seed);
227     }
228 
229     static Repository mirror(URI from, Path to) throws IOException {
230         return from.getPath().toString().endsWith(".git") ?
231             GitRepository.mirror(from, to) :
232             HgRepository.clone(from, to, true, null); // hg does not have concept of "mirror"
233     }
