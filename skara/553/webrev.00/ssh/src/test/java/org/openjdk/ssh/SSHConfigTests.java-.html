24 package org.openjdk.skara.ssh;
25 
26 import java.util.*;
27 import org.junit.jupiter.api.Test;
28 
29 import static org.junit.jupiter.api.Assertions.*;
30 
31 public class SSHConfigTests {
32     @Test
33     void testSimpleConfig() {
34         var lines = List.of(
35             "Host test",
36             "  User git",
37             "  HostName git.openjdk.java.net",
38             "  Port 22"
39         );
40         var config = SSHConfig.parse(lines);
41         var hosts = config.hosts();
42         assertEquals(1, hosts.size());
43         var host = hosts.get(0);
44         assertEquals("test", host.name());
45         assertEquals("git", host.user());
46         assertEquals(22, host.port());
47     }
48 }
