 23 package org.openjdk.skara.forge.gitlab;
 24 
 25 import org.openjdk.skara.forge.*;
 26 import org.openjdk.skara.host.*;
 27 import org.openjdk.skara.json.*;
 28 import org.openjdk.skara.network.*;
 29 
 30 import java.io.IOException;
 31 import java.net.*;
 32 import java.nio.charset.StandardCharsets;
 33 import java.util.*;
 34 import java.util.logging.Logger;
 35 
 36 public class GitLabHost implements Forge {
 37     private final String name;
 38     private final URI uri;
 39     private final Credential pat;
 40     private final RestRequest request;
 41     private final Logger log = Logger.getLogger("org.openjdk.skara.forge.gitlab");
 42 
 43     GitLabHost(String name, URI uri, Credential pat) {
 44         this.name = name;
 45         this.uri = uri;
 46         this.pat = pat;
 47 
 48         var baseApi = URIBuilder.base(uri)
 49                                 .setPath("/api/v4/")
 50                                 .build();
 51         request = new RestRequest(baseApi, pat.username(), () -> Arrays.asList("Private-Token", pat.password()));
 52     }
 53 
 54     GitLabHost(String name, URI uri) {
 55         this.name = name;
 56         this.uri = uri;
 57         this.pat = null;
 58 
 59         var baseApi = URIBuilder.base(uri)
 60                                 .setPath("/api/v4/")
 61                                 .build();
 62         request = new RestRequest(baseApi);
 63     }
 64 
 65     public URI getUri() {
 66         return uri;
 67     }
 68 
 69     Optional<Credential> getPat() {
 70         return Optional.ofNullable(pat);
 71     }
 72 
 73     @Override
 74     public String name() {
 75         return name;
 76     }
 77 
 78     @Override
 79     public boolean isValid() {
 80         try {
 81             var version = request.get("version")
 82                                   .executeUnparsed();
 83             var parsed = JSON.parse(version);
 84             if (parsed != null && parsed.contains("version")) {
 85                 return true;
 86             } else {
 87                 log.fine("Error during GitLab host validation: unexpected version: " + version);
 88                 return false;
 89             }
 90         } catch (IOException e) {
 91             log.fine("Error during GitLab host validation: " + e);
 92             return false;
 93         }
 94     }
 95 
 96     JSONObject getProjectInfo(String name) {
 97         var encodedName = URLEncoder.encode(name, StandardCharsets.US_ASCII);
 98         var project = request.get("projects/" + encodedName)
 99                              .onError(r -> r.statusCode() == 404 ? Optional.of(JSON.object().put("retry", true)) : Optional.empty())
100                              .execute();
101         if (project.contains("retry")) {
102             // Depending on web server configuration, GitLab may need double escaping of project names
103             encodedName = URLEncoder.encode(encodedName, StandardCharsets.US_ASCII);
104             project = request.get("projects/" + encodedName).execute();
105         }
106         return project.asObject();
107     }
108 
109     JSONObject getProjectInfo(int id) {
110         return request.get("projects/" + Integer.toString(id))
111                       .execute()
112                       .asObject();
113     }
114 
115     @Override
116     public Optional<HostedRepository> repository(String name) {
117         try {
118             return Optional.of(new GitLabRepository(this, name));
119         } catch (Throwable t) {
120             return Optional.empty();
121         }
122     }
123 
124     private HostUser parseUserDetails(JSONObject details) {
125         var id = details.get("id").asInt();
126         var username = details.get("username").asString();
127         var name = details.get("name").asString();
128         var email = details.get("email").asString();
129         return new HostUser(id, username, name, email);
130     }
131 
132     @Override
133     public Optional<HostUser> user(String username) {
134         var details = request.get("users")
135                              .param("username", username)
136                              .onError(r -> Optional.of(JSON.of()))
137                              .execute();
138 
139         if (details.isNull()) {
140             return Optional.empty();
141         }
142 
143         var users = details.asArray();
144         if (users.size() != 1) {
145             return Optional.empty();
146         }
147 
148         return Optional.of(parseUserDetails(users.get(0).asObject()));
149     }
150 
151     @Override
152     public HostUser currentUser() {
153         var details = request.get("user").execute().asObject();
154         return parseUserDetails(details);
155     }
156 
157     @Override
158     public boolean supportsReviewBody() {
159         // GitLab CE does not support this
160         return false;
161     }
162 
163     boolean isProjectForkComplete(String name) {
164         var project = getProjectInfo(name);
165         if (project.contains("import_status")) {
166             var status = project.get("import_status").asString();
167             switch (status) {
168                 case "finished":
169                     return true;
170                 case "started":
171                     return false;
172                 default:
173                     throw new RuntimeException("Unknown fork status: " + status);
174             }
175         } else {
176             throw new RuntimeException("Project does not seem to be a fork");
177         }
178     }
179 
180     @Override
181     public boolean isMemberOf(String groupId, HostUser user) {
182         long gid = 0L;
183         try {
184             gid = Long.parseLong(groupId);
185         } catch (NumberFormatException e) {
186             throw new IllegalArgumentException("Group id is not a number: " + groupId);
187         }
188         var details = request.get("groups/" + gid + "/members/" + user.id())
189                              .onError(r -> Optional.of(JSON.of()))
190                              .execute();
191         return !details.isNull();
192     }
193 }
