23 package org.openjdk.skara.cli;
24 
25 import org.openjdk.skara.args.*;
26 import org.openjdk.skara.cli.pr.*;
27 import org.openjdk.skara.proxy.HttpProxy;
28 
29 import java.util.List;
30 
31 public class GitPr {
32     public static void main(String[] args) throws Exception {
33         var commands = List.of(
34                     Default.name("help")
35                            .helptext("show help text")
36                            .main(GitPrHelp::main),
37                     Command.name("list")
38                            .helptext("list open pull requests")
39                            .main(GitPrList::main),
40                     Command.name("fetch")
41                            .helptext("fetch a pull request")
42                            .main(GitPrFetch::main),
43                     Command.name("show")
44                            .helptext("show a pull request")
45                            .main(GitPrShow::main),
46                     Command.name("checkout")
47                            .helptext("checkout a pull request")
48                            .main(GitPrCheckout::main),
49                     Command.name("apply")
50                            .helptext("apply a pull request")
51                            .main(GitPrApply::main),
52                     Command.name("integrate")
53                            .helptext("integrate a pull request")
54                            .main(GitPrIntegrate::main),
55                     Command.name("approve")
56                            .helptext("approve a pull request")
57                            .main(GitPrApprove::main),
58                     Command.name("create")
59                            .helptext("create a pull request")
60                            .main(GitPrCreate::main),
61                     Command.name("close")
62                            .helptext("close a pull request")
63                            .main(GitPrClose::main),
64                     Command.name("set")
65                            .helptext("set properties of a pull request")
66                            .main(GitPrSet::main),
67                     Command.name("sponsor")
68                            .helptext("sponsor a pull request")
69                            .main(GitPrSet::main),
70                     Command.name("test")
71                            .helptext("test a pull request")
72                            .main(GitPrTest::main),
73                     Command.name("info")
74                            .helptext("show status of a pull request")
75                            .main(GitPrInfo::main),
76                     Command.name("issue")
77                            .helptext("add, remove or create issues")
78                            .main(GitPrIssue::main),
79                     Command.name("reviewer")
80                            .helptext("add or remove reviewers")
81                            .main(GitPrReviewer::main),
82                     Command.name("cc")
83                            .helptext("add one or more labels")
84                            .main(GitPrCC::main)
85         );
86 
87         HttpProxy.setup();
88 
89         var parser = new MultiCommandParser("git pr", commands);
90         var command = parser.parse(args);
91         command.execute();
92     }
93 }
