 44     private Optional<Contributor> parseUser(String user, PullRequest pr, CensusInstance censusInstance) {
 45         user = user.strip();
 46         if (user.isEmpty()) {
 47             return Optional.empty();
 48         }
 49         Contributor contributor;

 52             if (platformUser.isEmpty()) {

 55             contributor = censusInstance.namespace().get(platformUser.get().id());
 56             if (contributor == null) {
 57                 return Optional.empty();
 58             }
 59         } else {
 60             contributor = censusInstance.census().contributor(user);
 61             if (contributor == null) {

 68     @Override
 69     public void handle(PullRequestBot bot, PullRequest pr, CensusInstance censusInstance, Path scratchPath, CommandInvocation command, List<Comment> allComments, PrintWriter reply) {
 70         if (!command.user().equals(pr.author())) {
 71             reply.println("Only the author (@" + pr.author().userName() + ") is allowed to issue the `reviewer` command.");
 72             return;
 73         }
 74         if (bot.ignoreStaleReviews()) {
 75             reply.println("This project requires authenticated reviews - please ask your reviewer to flag this PR as reviewed.");
 76             return;
 77         }
 78 
 79         var matcher = commandPattern.matcher(command.args());
 80         if (!matcher.matches()) {
 81             showHelp(pr, reply);
 82             return;
 83         }
 84 
 85         var reviewer = parseUser(matcher.group(2), pr, censusInstance);
 86         if (reviewer.isEmpty()) {
 87             reply.println("Could not parse `" + matcher.group(2) + "` as a valid reviewer.");
 88             showHelp(pr, reply);;
 89             return;





 90         }
 91 
 92         var namespace = censusInstance.namespace();
 93         var authenticatedReviewers = PullRequestUtils.reviewerNames(pr.reviews(), namespace);
 94         switch (matcher.group(1)) {
 95             case "add": {
 96                 if (!authenticatedReviewers.contains(reviewer.get().username())) {
 97                     reply.println(Reviewers.addReviewerMarker(reviewer.get()));
 98                     reply.println("Reviewer `" + reviewer.get().username() + "` successfully added.");

 99                 } else {
100                     reply.println("Reviewer `" + reviewer.get().username() + "` has already made an authenticated review of this PR, and does not need to be added manually.");
101                 }
102                 break;
103             }
104             case "remove": {
105                 var existing = new HashSet<>(Reviewers.reviewers(pr.repository().forge().currentUser(), allComments));
106                 if (existing.contains(reviewer.get().username())) {
107                     reply.println(Reviewers.removeReviewerMarker(reviewer.get()));
108                     reply.println("Reviewer `" + reviewer.get().username() + "` successfully removed.");


109                 } else {
110                     if (existing.isEmpty()) {
111                         reply.println("There are no additional reviewers associated with this pull request.");

112                     } else {
113                         reply.println("Reviewer `" + reviewer.get().username() + "` was not found.");
114                         reply.println("Current additional reviewers are:");
115                         for (var e : existing) {
116                             reply.println("- `" + e + "`");
117                         }
118                     }
119                     break;
120                 }
121                 break;






122             }
123         }
124     }

 69     @Override
 70     public void handle(PullRequestBot bot, PullRequest pr, CensusInstance censusInstance, Path scratchPath, CommandInvocation command, List<Comment> allComments, PrintWriter reply) {
 71         if (!command.user().equals(pr.author())) {
 72             reply.println("Only the author (@" + pr.author().userName() + ") is allowed to issue the `reviewer` command.");
 73             return;
 74         }
 75         if (bot.ignoreStaleReviews()) {
 76             reply.println("This project requires authenticated reviews - please ask your reviewer to flag this PR as reviewed.");
 77             return;
 78         }
 79 
 80         var matcher = commandPattern.matcher(command.args());
 81         if (!matcher.matches()) {
 82             showHelp(pr, reply);
 83             return;
 84         }
 85 
 86         var reviewers = new ArrayList<Contributor>();
 87         for (var entry : matcher.group(2).split("[\\s,]+")) {
 88             var reviewer = parseUser(entry, pr, censusInstance);
 89             if (reviewer.isEmpty()) {
 90                 reply.println("Could not parse `" + entry + "` as a valid reviewer.");
 91                 showHelp(pr, reply);
 92                 return;
 93             }
 94 
 95             reviewers.add(reviewer.get());
 96         }
 97 
 98         var namespace = censusInstance.namespace();
 99         var authenticatedReviewers = PullRequestUtils.reviewerNames(pr.reviews(), namespace);
100         var action = matcher.group(1);
101         if (action.equals("add")) {
102             for (var reviewer : reviewers) {
103                 if (!authenticatedReviewers.contains(reviewer.username())) {
104                     reply.println(Reviewers.addReviewerMarker(reviewer));
105                     reply.println("Reviewer `" + reviewer.username() + "` successfully added.");
106                 } else {
107                     reply.println("Reviewer `" + reviewer.username() + "` has already made an authenticated review of this PR, and does not need to be added manually.");
108                 }

109             }
110         } else if (action.equals("remove")) {
111             var failed = false;
112             var existing = new HashSet<>(Reviewers.reviewers(pr.repository().forge().currentUser(), allComments));
113             for (var reviewer : reviewers) {
114                 if (existing.contains(reviewer.username())) {
115                     reply.println(Reviewers.removeReviewerMarker(reviewer));
116                     reply.println("Reviewer `" + reviewer.username() + "` successfully removed.");
117                 } else {
118                     if (existing.isEmpty()) {
119                         reply.println("There are no additional reviewers associated with this pull request.");
120                         failed = true;
121                     } else {
122                         reply.println("Reviewer `" + reviewer.username() + "` was not found.");
123                         failed = true;



124                     }

125                 }
126             }
127 
128             if (failed) {
129                 reply.println("Current additional reviewers are:");
130                 for (var e : existing) {
131                     reply.println("- `" + e + "`");
132                 }
133             }
134         }
135     }
