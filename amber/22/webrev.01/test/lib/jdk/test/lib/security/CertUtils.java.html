376      *         implementation for the specified algorithm

417      * @throws IOException on input stream errors.

419     public static X509Certificate getCertFromStream(InputStream input)
420             throws CertificateException, IOException {
421         try {

611     /**
612      * Get the content of a file with given filter condition.
613      *
614      * @param relativeFilePath path to file that relative to test.src directory.
615      * @param predicate The condition for filtering file content
616      * @return the file content
617      */
618     private static String readFile(String relativeFilePath,
619             Predicate<String> predicate) {
620         Path filePath = Paths.get(TEST_SRC, relativeFilePath);
621         try (Stream<String> lines = Files.lines(filePath)) {
622             Stream<String> interStream = null;
623             if (predicate != null) {
624                 interStream = lines.filter(predicate);
625             }
626             return interStream != null
627                    ? interStream.collect(Collectors.joining("\n"))
628                    : lines.collect(Collectors.joining("\n"));
629         } catch (IOException e) {
630             throw new RuntimeException("Cannot read file", e);
631         }
632     }
