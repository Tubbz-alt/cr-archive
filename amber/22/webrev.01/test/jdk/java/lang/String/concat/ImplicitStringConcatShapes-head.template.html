41 public class ImplicitStringConcatShapes {
42     public static void test(String expected, String actual) {
43         // Fingers crossed: String concat should work.
44         if (!expected.equals(actual)) {
45             StringBuilder sb = new StringBuilder();
46             sb.append("Expected = ");
47             sb.append(expected);
48             sb.append(", actual = ");
49             sb.append(actual);
50             throw new IllegalStateException(sb.toString());
51         }
52     }
53 
54     static class MyClass {
55         private final int i;
56 
57         public MyClass(int i) {
58             this.i = i;
59         }
60 
61         public String toString() {
62             return new StringBuilder("C(").append(i).append(")").toString();
63         }
64     }
65 
66     static class MyClassNullToString {
67         public String toString() {
68             return null;
69         }
70     }
71 
72     public static void main(String[] args) throws Exception {
73         new ImplicitStringConcatShapes().run();
74     }
