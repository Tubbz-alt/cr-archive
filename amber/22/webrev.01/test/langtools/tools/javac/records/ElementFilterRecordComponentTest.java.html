24 /*
25  * @test
26  * @bug 8241312
27  * @summary test for javax.lang.model.util.ElementFilter::recordComponentsIn
28  * @modules jdk.compiler/com.sun.tools.javac.util
29  */
30 
31 import java.io.IOException;
32 import java.net.URI;
33 import java.util.List;
34 import javax.lang.model.element.RecordComponentElement;
35 import javax.lang.model.element.TypeElement;
36 import javax.tools.JavaCompiler;
37 import javax.tools.SimpleJavaFileObject;
38 import javax.tools.ToolProvider;
39 import com.sun.source.util.JavacTask;
40 import com.sun.tools.javac.util.Assert;
41 import java.util.HashSet;
42 import java.util.Set;
43 import javax.lang.model.util.ElementFilter;
44 import javax.tools.JavaFileObject;
45 
46 
47 public class ElementFilterRecordComponentTest {
48     public static void main(String... args) throws IOException {
49         JavaCompiler c = ToolProvider.getSystemJavaCompiler();
50         JavacTask t = (JavacTask) c.getTask(null, null, null,
51                 List.of("--enable-preview", "-source", Integer.toString(Runtime.version().feature())), null,
52                 List.of(new SimpleJavaFileObject(URI.create("TestClass.java"), JavaFileObject.Kind.SOURCE) {
53             @Override
54             public CharSequence getCharContent(boolean ignoreEncodingErrors) throws IOException {
55                 return "record R(int val1, int val2) {}";
56             }
57         }));
58         TypeElement record = (TypeElement) t.analyze().iterator().next();
59         Set<RecordComponentElement> recordSet = ElementFilter.recordComponentsIn(new HashSet<>(record.getEnclosedElements()));
60         Assert.check(recordSet.size() == 2);
61         List<RecordComponentElement> recordList = ElementFilter.recordComponentsIn(record.getEnclosedElements());
62         Assert.check(recordList.size() == 2);
63         Assert.check(recordSet.containsAll(recordList));
64     }
65 }
