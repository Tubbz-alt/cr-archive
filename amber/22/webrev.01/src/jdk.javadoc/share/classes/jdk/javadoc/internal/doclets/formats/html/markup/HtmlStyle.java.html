 38  *
 39  * @apiNote
 40  * Despite the name, the members of this enum provide values for the HTML {@code class} attribute,
 41  * and <strong>not</strong> the HTML {@code style} attribute.
 42  * This is to avoid confusion with the widespread use of the word "class" in the Java ecosystem,

754     private final String cssName;
755 
756     HtmlStyle() {
757         cssName = Pattern.compile("\\p{Upper}")
758                 .matcher(toString())
759                 .replaceAll(mr -> "-" + mr.group().toLowerCase(Locale.US));
760     }
761 
762     HtmlStyle(String cssName) {
763         this.cssName = cssName;
764     }
765 
766     /**
767      * Returns the CSS class name associated with this style.
768      * @return the CSS class name
769      */
770     public String cssName() {
771         return cssName;
772     }
