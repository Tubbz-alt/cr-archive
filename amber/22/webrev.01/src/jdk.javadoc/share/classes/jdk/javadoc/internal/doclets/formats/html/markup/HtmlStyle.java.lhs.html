440     private final String cssName;
441 
442     HtmlStyle() {
443         cssName = Pattern.compile("\\p{Upper}")
444                 .matcher(toString())
445                 .replaceAll(mr -> "-" + mr.group().toLowerCase(Locale.US));
446     }
447 
448     HtmlStyle(String cssName) {
449         this.cssName = cssName;
450     }
451 
452     /**
453      * Returns the CSS class name associated with this style.
454      * @return the CSS class name
455      */
456     public String cssName() {
457         return cssName;
458     }
