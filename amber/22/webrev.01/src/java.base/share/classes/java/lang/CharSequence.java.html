 26 package java.lang;
 27 
 28 import java.util.NoSuchElementException;
 29 import java.util.Objects;
 30 import java.util.PrimitiveIterator;
 31 import java.util.Spliterator;
 32 import java.util.Spliterators;
 33 import java.util.function.IntConsumer;
 34 import java.util.stream.IntStream;
 35 import java.util.stream.StreamSupport;
 36 
 37 /**
 38  * A {@code CharSequence} is a readable sequence of {@code char} values. This
 39  * interface provides uniform, read-only access to many different kinds of
 40  * {@code char} sequences.
 41  * A {@code char} value represents a character in the <i>Basic
 42  * Multilingual Plane (BMP)</i> or a surrogate. Refer to <a
 43  * href="Character.html#unicode">Unicode Character Representation</a> for details.
 44  *
 45  * <p> This interface does not refine the general contracts of the {@link
 46  * java.lang.Object#equals(java.lang.Object) equals} and {@link
 47  * java.lang.Object#hashCode() hashCode} methods. The result of testing two objects
 48  * that implement {@code CharSequence} for equality is therefore, in general, undefined.
 49  * Each object may be implemented by a different class, and there
 50  * is no guarantee that each class will be capable of testing its instances
 51  * for equality with those of the other.  It is therefore inappropriate to use
 52  * arbitrary {@code CharSequence} instances as elements in a set or as keys in
 53  * a map. </p>
 54  *
 55  * @author Mike McCloskey
 56  * @since 1.4
 57  * @spec JSR-51
 58  */
 59 
 60 public interface CharSequence {
 61 
 62     /**
 63      * Returns the length of this character sequence.  The length is the number
 64      * of 16-bit {@code char}s in the sequence.
 65      *
 66      * @return  the number of {@code char}s in this sequence
 67      */
 68     int length();
 69 
 70     /**
 71      * Returns the {@code char} value at the specified index.  An index ranges from zero
 72      * to {@code length() - 1}.  The first {@code char} value of the sequence is at
 73      * index zero, the next at index one, and so on, as for array
 74      * indexing.
 75      *
 76      * <p>If the {@code char} value specified by the index is a
 77      * <a href="{@docRoot}/java.base/java/lang/Character.html#unicode">surrogate</a>, the surrogate
 78      * value is returned.
 79      *
 80      * @param   index   the index of the {@code char} value to be returned
 81      *
 82      * @return  the specified {@code char} value
 83      *
 84      * @throws  IndexOutOfBoundsException
 85      *          if the {@code index} argument is negative or not less than
 86      *          {@code length()}
 87      */
 88     char charAt(int index);
 89 
 90     /**
 91      * Returns {@code true} if this character sequence is empty.
 92      *
 93      * @implSpec
 94      * The default implementation returns the result of calling {@code length() == 0}.
 95      *
 96      * @return {@code true} if {@link #length()} is {@code 0}, otherwise
 97      * {@code false}
 98      *
 99      * @since 15
100      */
101     default boolean isEmpty() {
102         return this.length() == 0;
103     }
104 
105     /**
106      * Returns a {@code CharSequence} that is a subsequence of this sequence.
107      * The subsequence starts with the {@code char} value at the specified index and
108      * ends with the {@code char} value at index {@code end - 1}.  The length
109      * (in {@code char}s) of the
110      * returned sequence is {@code end - start}, so if {@code start == end}
111      * then an empty sequence is returned.
112      *
113      * @param   start   the start index, inclusive
114      * @param   end     the end index, exclusive
115      *
116      * @return  the specified subsequence
117      *
118      * @throws  IndexOutOfBoundsException
119      *          if {@code start} or {@code end} are negative,
120      *          if {@code end} is greater than {@code length()},
121      *          or if {@code start} is greater than {@code end}
122      */
123     CharSequence subSequence(int start, int end);
124 
125     /**
126      * Returns a string containing the characters in this sequence in the same
127      * order as this sequence.  The length of the string will be the length of
128      * this sequence.
129      *
130      * @return  a string consisting of exactly this sequence of characters
131      */
132     public String toString();
133 
134     /**
135      * Returns a stream of {@code int} zero-extending the {@code char} values
136      * from this sequence.  Any char which maps to a <a
137      * href="{@docRoot}/java.base/java/lang/Character.html#unicode">surrogate code
138      * point</a> is passed through uninterpreted.
139      *
140      * <p>The stream binds to this sequence when the terminal stream operation
141      * commences (specifically, for mutable sequences the spliterator for the
142      * stream is <a href="../util/Spliterator.html#binding"><em>late-binding</em></a>).
143      * If the sequence is modified during that operation then the result is
144      * undefined.
145      *
146      * @return an IntStream of char values from this sequence
147      * @since 1.8
148      */
149     public default IntStream chars() {
150         class CharIterator implements PrimitiveIterator.OfInt {
151             int cur = 0;
152 
153             public boolean hasNext() {
154                 return cur < length();
155             }
156 
157             public int nextInt() {
158                 if (hasNext()) {
159                     return charAt(cur++);
160                 } else {
161                     throw new NoSuchElementException();
162                 }
163             }
164 
165             @Override
166             public void forEachRemaining(IntConsumer block) {
167                 for (; cur < length(); cur++) {
168                     block.accept(charAt(cur));
169                 }
170             }
171         }
172 
173         return StreamSupport.intStream(() ->
174                 Spliterators.spliterator(
175                         new CharIterator(),
176                         length(),
177                         Spliterator.ORDERED),
178                 Spliterator.SUBSIZED | Spliterator.SIZED | Spliterator.ORDERED,
179                 false);
180     }
181 
182     /**
183      * Returns a stream of code point values from this sequence.  Any surrogate
184      * pairs encountered in the sequence are combined as if by {@linkplain
185      * Character#toCodePoint Character.toCodePoint} and the result is passed
186      * to the stream. Any other code units, including ordinary BMP characters,
187      * unpaired surrogates, and undefined code units, are zero-extended to
188      * {@code int} values which are then passed to the stream.
189      *
190      * <p>The stream binds to this sequence when the terminal stream operation
191      * commences (specifically, for mutable sequences the spliterator for the
192      * stream is <a href="../util/Spliterator.html#binding"><em>late-binding</em></a>).
193      * If the sequence is modified during that operation then the result is
194      * undefined.
195      *
196      * @return an IntStream of Unicode code points from this sequence
197      * @since 1.8
198      */
199     public default IntStream codePoints() {
200         class CodePointIterator implements PrimitiveIterator.OfInt {
201             int cur = 0;
202 
203             @Override
204             public void forEachRemaining(IntConsumer block) {
205                 final int length = length();
206                 int i = cur;
207                 try {
208                     while (i < length) {
209                         char c1 = charAt(i++);
210                         if (!Character.isHighSurrogate(c1) || i >= length) {
211                             block.accept(c1);
212                         } else {
213                             char c2 = charAt(i);
214                             if (Character.isLowSurrogate(c2)) {
215                                 i++;
216                                 block.accept(Character.toCodePoint(c1, c2));
217                             } else {
218                                 block.accept(c1);
219                             }
220                         }
221                     }
222                 } finally {
223                     cur = i;
224                 }
225             }
226 
227             public boolean hasNext() {
228                 return cur < length();
229             }
230 
231             public int nextInt() {
232                 final int length = length();
233 
234                 if (cur >= length) {
235                     throw new NoSuchElementException();
236                 }
237                 char c1 = charAt(cur++);
238                 if (Character.isHighSurrogate(c1) && cur < length) {
239                     char c2 = charAt(cur);
240                     if (Character.isLowSurrogate(c2)) {
241                         cur++;
242                         return Character.toCodePoint(c1, c2);
243                     }
244                 }
245                 return c1;
246             }
247         }
248 
249         return StreamSupport.intStream(() ->
250                 Spliterators.spliteratorUnknownSize(
251                         new CodePointIterator(),
252                         Spliterator.ORDERED),
253                 Spliterator.ORDERED,
254                 false);
255     }
256 
257     /**
258      * Compares two {@code CharSequence} instances lexicographically. Returns a
259      * negative value, zero, or a positive value if the first sequence is lexicographically
260      * less than, equal to, or greater than the second, respectively.
261      *
262      * <p>
263      * The lexicographical ordering of {@code CharSequence} is defined as follows.
264      * Consider a {@code CharSequence} <i>cs</i> of length <i>len</i> to be a
265      * sequence of char values, <i>cs[0]</i> to <i>cs[len-1]</i>. Suppose <i>k</i>
266      * is the lowest index at which the corresponding char values from each sequence
267      * differ. The lexicographic ordering of the sequences is determined by a numeric
268      * comparison of the char values <i>cs1[k]</i> with <i>cs2[k]</i>. If there is
269      * no such index <i>k</i>, the shorter sequence is considered lexicographically
270      * less than the other. If the sequences have the same length, the sequences are
271      * considered lexicographically equal.
272      *
273      *
274      * @param cs1 the first {@code CharSequence}
275      * @param cs2 the second {@code CharSequence}
276      *
277      * @return  the value {@code 0} if the two {@code CharSequence} are equal;
278      *          a negative integer if the first {@code CharSequence}
279      *          is lexicographically less than the second; or a
280      *          positive integer if the first {@code CharSequence} is
281      *          lexicographically greater than the second.
282      *
283      * @since 11
284      */
285     @SuppressWarnings("unchecked")
286     public static int compare(CharSequence cs1, CharSequence cs2) {
287         if (Objects.requireNonNull(cs1) == Objects.requireNonNull(cs2)) {
288             return 0;
289         }
290 
291         if (cs1.getClass() == cs2.getClass() && cs1 instanceof Comparable) {
292             return ((Comparable<Object>) cs1).compareTo(cs2);
293         }
294 
295         for (int i = 0, len = Math.min(cs1.length(), cs2.length()); i < len; i++) {
296             char a = cs1.charAt(i);
297             char b = cs2.charAt(i);
298             if (a != b) {
299                 return a - b;
300             }
301         }
302 
303         return cs1.length() - cs2.length();
304     }
