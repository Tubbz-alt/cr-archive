26 /*
27  * Usage: jjs jrtls.js
28  *
29  * Recursively list the content of / directory of the jrt fs.
30  *
31  * @implNote This script needs to maintain JDK 8 source compatibility.
32  *
33  * It is used internally in the JDK to implement jimage/jrtfs access,
34  * but also compiled and delivered as part of the jrtfs.jar to support access
35  * to the jimage file provided by the shipped JDK by tools running on JDK 8.
36  */
37 
38 // classes used
39 var Files = Java.type("java.nio.file.Files");
40 var FileSystems = Java.type("java.nio.file.FileSystems");
41 var URI = Java.type("java.net.URI");
42 
43 var fs = FileSystems.getFileSystem(URI.create("jrt:/"));
44 var root = fs.getPath('/');
45 Files.walk(root).forEach(print);
