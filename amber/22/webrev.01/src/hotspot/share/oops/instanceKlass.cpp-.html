 139 static inline bool is_class_loader(const Symbol* class_name,
 140                                    const ClassFileParser& parser) {
 141   assert(class_name != NULL, "invariant");
 142 
 143   if (class_name == vmSymbols::java_lang_ClassLoader()) {
 144     return true;
 145   }
 146 
 147   if (SystemDictionary::ClassLoader_klass_loaded()) {
 148     const Klass* const super_klass = parser.super_klass();
 149     if (super_klass != NULL) {
 150       if (super_klass->is_subtype_of(SystemDictionary::ClassLoader_klass())) {
 151         return true;
 152       }
 153     }
 154   }
 155   return false;
 156 }

 158 // private: called to verify that k is a static member of this nest.
 159 // We know that k is an instance class in the same package and hence the
 160 // same classloader.
 161 bool InstanceKlass::has_nest_member(InstanceKlass* k, TRAPS) const {
 162   assert(!is_hidden(), "unexpected hidden class");
 163   if (_nest_members == NULL || _nest_members == Universe::the_empty_short_array()) {
 164     if (log_is_enabled(Trace, class, nestmates)) {
 165       ResourceMark rm(THREAD);
 166       log_trace(class, nestmates)("Checked nest membership of %s in non-nest-host class %s",
 167                                   k->external_name(), this->external_name());
 168     }
 169     return false;
 170   }
 171 
 172   if (log_is_enabled(Trace, class, nestmates)) {
 173     ResourceMark rm(THREAD);
 174     log_trace(class, nestmates)("Checking nest membership of %s in %s",
 175                                 k->external_name(), this->external_name());
 176   }
 177 
 178   // Check for a resolved cp entry , else fall back to a name check.
 179   // We don't want to resolve any class other than the one being checked.
 180   for (int i = 0; i < _nest_members->length(); i++) {
 181     int cp_index = _nest_members->at(i);
 182     if (_constants->tag_at(cp_index).is_klass()) {
 183       Klass* k2 = _constants->klass_at(cp_index, THREAD);
 184       assert(!HAS_PENDING_EXCEPTION || PENDING_EXCEPTION->is_a(SystemDictionary::VirtualMachineError_klass()),
 185              "Exceptions should not be possible here");
 186       if (k2 == k) {
 187         log_trace(class, nestmates)("- class is listed at nest_members[%d] => cp[%d]", i, cp_index);
 188         return true;
 189       }
 190     }
 191     else {
 192       Symbol* name = _constants->klass_name_at(cp_index);
 193       if (name == k->name()) {
 194         log_trace(class, nestmates)("- Found it at nest_members[%d] => cp[%d]", i, cp_index);
 195 
 196         // Names match so check actual klass. This may trigger class loading if
 197         // it doesn't match though that should be impossible as it means one classloader
 198         // has defined two different classes with the same name! A compiler thread won't be
 199         // able to perform that loading but we can't exclude the compiler threads from
 200         // executing this logic. But it should actually be impossible to trigger loading here.
 201         Klass* k2 = _constants->klass_at(cp_index, THREAD);
 202         assert(!HAS_PENDING_EXCEPTION || PENDING_EXCEPTION->is_a(SystemDictionary::VirtualMachineError_klass()),
 203                "Exceptions should not be possible here");
 204         if (k2 == k) {
 205           log_trace(class, nestmates)("- class is listed as a nest member");
 206           return true;
 207         }
 208         else {
 209           // same name but different klass!
 210           log_trace(class, nestmates)(" - klass comparison failed!");
 211           // can't have two names the same, so we're done
 212           return false;
 213         }
 214       }
 215     }
 216   }
 217   log_trace(class, nestmates)("- class is NOT a nest member!");
 218   return false;
 219 }

 221 // Called to verify that k is a permitted subclass of this class
 222 bool InstanceKlass::has_as_permitted_subclass(const InstanceKlass* k) const {
 223   Thread* THREAD = Thread::current();
 224   assert(k != NULL, "sanity check");
 225   assert(_permitted_subclasses != NULL && _permitted_subclasses != Universe::the_empty_short_array(),
 226          "unexpected empty _permitted_subclasses array");
 227 
 228   if (log_is_enabled(Trace, class, sealed)) {
 229     ResourceMark rm(THREAD);
 230     log_trace(class, sealed)("Checking for permitted subclass of %s in %s",
 231                              k->external_name(), this->external_name());
 232   }
 233 
 234   // Check that the class and its super are in the same module.
 235   if (k->module() != this->module()) {
 236     log_trace(class, sealed)("Check failed for same module of permitted subclass of %s and sealed class %s",
 237                              k->external_name(), this->external_name());
 238     return false;
 239   }
 240 
 241   // Check for a resolved cp entry, else fall back to a name check.
 242   // We don't want to resolve any class other than the one being checked.
 243   for (int i = 0; i < _permitted_subclasses->length(); i++) {
 244     int cp_index = _permitted_subclasses->at(i);
 245     if (_constants->tag_at(cp_index).is_klass()) {
 246       Klass* k2 = _constants->klass_at(cp_index, THREAD);
 247       assert(!HAS_PENDING_EXCEPTION, "Unexpected exception");
 248       if (k2 == k) {
 249         log_trace(class, sealed)("- class is listed at permitted_subclasses[%d] => cp[%d]", i, cp_index);
 250         return true;
 251       }
 252     } else {
 253       Symbol* name = _constants->klass_name_at(cp_index);
 254       if (name == k->name()) {
 255         log_trace(class, sealed)("- Found it at permitted_subclasses[%d] => cp[%d]", i, cp_index);
 256         return true;
 257       }
 258     }
 259   }
 260   log_trace(class, sealed)("- class is NOT a permitted subclass!");
 261   return false;
 262 }

 264 // Return nest-host class, resolving, validating and saving it if needed.
 265 // In cases where this is called from a thread that cannot do classloading
 266 // (such as a native JIT thread) then we simply return NULL, which in turn
 267 // causes the access check to return false. Such code will retry the access
 268 // from a more suitable environment later. Otherwise the _nest_host is always
 269 // set once this method returns.
 270 // Any errors from nest-host resolution must be preserved so they can be queried
 271 // from higher-level access checking code, and reported as part of access checking
 272 // exceptions.
 273 // VirtualMachineErrors are propagated with a NULL return.
 274 // Under any conditions where the _nest_host can be set to non-NULL the resulting
 275 // value of it and, if applicable, the nest host resolution/validation error,
 276 // are idempotent.
 277 InstanceKlass* InstanceKlass::nest_host(TRAPS) {
 278   InstanceKlass* nest_host_k = _nest_host;
 279   if (nest_host_k != NULL) {
 280     return nest_host_k;
 281   }
 282 
 283   ResourceMark rm(THREAD);
 284 
 285   // need to resolve and save our nest-host class.
 286   if (_nest_host_index != 0) { // we have a real nest_host
 287     // Before trying to resolve check if we're in a suitable context
 288     if (!THREAD->can_call_java() && !_constants->tag_at(_nest_host_index).is_klass()) {
 289       log_trace(class, nestmates)("Rejected resolution of nest-host of %s in unsuitable thread",
 290                                   this->external_name());
 291       return NULL; // sentinel to say "try again from a different context"
 292     }
 293 
 294     log_trace(class, nestmates)("Resolving nest-host of %s using cp entry for %s",
 295                                 this->external_name(),
 296                                 _constants->klass_name_at(_nest_host_index)->as_C_string());
 297 
 298     Klass* k = _constants->klass_at(_nest_host_index, THREAD);
 299     if (HAS_PENDING_EXCEPTION) {
 300       if (PENDING_EXCEPTION->is_a(SystemDictionary::VirtualMachineError_klass())) {
 301         return NULL; // propagate VMEs
 302       }
 303       stringStream ss;
 304       char* target_host_class = _constants->klass_name_at(_nest_host_index)->as_C_string();
 305       ss.print("Nest host resolution of %s with host %s failed: ",
 306                this->external_name(), target_host_class);
 307       java_lang_Throwable::print(PENDING_EXCEPTION, &ss);
 308       const char* msg = ss.as_string(true /* on C-heap */);
 309       constantPoolHandle cph(THREAD, constants());
 310       SystemDictionary::add_nest_host_error(cph, _nest_host_index, msg);
 311       CLEAR_PENDING_EXCEPTION;
 312 
 313       log_trace(class, nestmates)("%s", msg);
 314     } else {
 315       // A valid nest-host is an instance class in the current package that lists this
 316       // class as a nest member. If any of these conditions are not met the class is
 317       // its own nest-host.
 318       const char* error = NULL;
 319 
 320       // JVMS 5.4.4 indicates package check comes first
 321       if (is_same_class_package(k)) {
 322         // Now check actual membership. We can't be a member if our "host" is
 323         // not an instance class.
 324         if (k->is_instance_klass()) {
 325           nest_host_k = InstanceKlass::cast(k);
 326           bool is_member = nest_host_k->has_nest_member(this, THREAD);
 327           // exception is rare, perhaps impossible
 328           if (!HAS_PENDING_EXCEPTION) {
 329             if (is_member) {
 330               _nest_host = nest_host_k; // save resolved nest-host value
 331 
 332               log_trace(class, nestmates)("Resolved nest-host of %s to %s",
 333                                           this->external_name(), k->external_name());
 334               return nest_host_k;
 335             } else {
 336               error = "current type is not listed as a nest member";
 337             }
 338           } else {
 339             if (PENDING_EXCEPTION->is_a(SystemDictionary::VirtualMachineError_klass())) {
 340               return NULL; // propagate VMEs
 341             }
 342             stringStream ss;
 343             ss.print("exception on member check: ");
 344             java_lang_Throwable::print(PENDING_EXCEPTION, &ss);
 345             error = ss.as_string();
 346           }
 347         } else {
 348           error = "host is not an instance class";
 349         }
 350       } else {
 351         error = "types are in different packages";
 352       }
 353 
 354       // something went wrong, so record what and log it
 355       {
 356         stringStream ss;
 357         ss.print("Type %s (loader: %s) is not a nest member of type %s (loader: %s): %s",
 358                  this->external_name(),
 359                  this->class_loader_data()->loader_name_and_id(),
 360                  k->external_name(),
 361                  k->class_loader_data()->loader_name_and_id(),
 362                  error);
 363         const char* msg = ss.as_string(true /* on C-heap */);
 364         constantPoolHandle cph(THREAD, constants());
 365         SystemDictionary::add_nest_host_error(cph, _nest_host_index, msg);
 366         log_trace(class, nestmates)("%s", msg);
 367       }
 368     }
 369   } else {
 370     log_trace(class, nestmates)("Type %s is not part of a nest: setting nest-host to self",
 371                                 this->external_name());
 372   }
 373 
 374   // Either not in an explicit nest, or else an error occurred, so
 375   // the nest-host is set to `this`. Any thread that sees this assignment
 376   // will also see any setting of nest_host_error(), if applicable.
 377   return (_nest_host = this);
 378 }

 380 // Dynamic nest member support: set this class's nest host to the given class.
 381 // This occurs as part of the class definition, as soon as the instanceKlass
 382 // has been created and doesn't require further resolution. The code:
 383 //    lookup().defineHiddenClass(bytes_for_X, NESTMATE);
 384 // results in:
 385 //    class_of_X.set_nest_host(lookup().lookupClass().getNestHost())
 386 // If it has an explicit _nest_host_index or _nest_members, these will be ignored.
 387 // We also know the "host" is a valid nest-host in the same package so we can
 388 // assert some of those facts.
 389 void InstanceKlass::set_nest_host(InstanceKlass* host, TRAPS) {
 390   assert(is_hidden(), "must be a hidden class");
 391   assert(host != NULL, "NULL nest host specified");
 392   assert(_nest_host == NULL, "current class has resolved nest-host");
 393   assert(nest_host_error(THREAD) == NULL, "unexpected nest host resolution error exists: %s",
 394          nest_host_error(THREAD));
 395   assert((host->_nest_host == NULL && host->_nest_host_index == 0) ||
 396          (host->_nest_host == host), "proposed host is not a valid nest-host");
 397   // Can't assert this as package is not set yet:
 398   // assert(is_same_class_package(host), "proposed host is in wrong package");
 399 
 400   if (log_is_enabled(Trace, class, nestmates)) {
 401     ResourceMark rm(THREAD);
 402     const char* msg = "";
 403     // a hidden class does not expect a statically defined nest-host
 404     if (_nest_host_index > 0) {
 405       msg = "(the NestHost attribute in the current class is ignored)";
 406     } else if (_nest_members != NULL && _nest_members != Universe::the_empty_short_array()) {
 407       msg = "(the NestMembers attribute in the current class is ignored)";
 408     }
 409     log_trace(class, nestmates)("Injected type %s into the nest of %s %s",
 410                                 this->external_name(),
 411                                 host->external_name(),
 412                                 msg);
 413   }
 414   // set dynamic nest host
 415   _nest_host = host;
 416   // Record dependency to keep nest host from being unloaded before this class.
 417   ClassLoaderData* this_key = class_loader_data();
 418   this_key->record_dependency(host);
 419 }

 421 // check if 'this' and k are nestmates (same nest_host), or k is our nest_host,
 422 // or we are k's nest_host - all of which is covered by comparing the two
 423 // resolved_nest_hosts.
 424 // Any exceptions (i.e. VMEs) are propagated.
 425 bool InstanceKlass::has_nestmate_access_to(InstanceKlass* k, TRAPS) {
 426 
 427   assert(this != k, "this should be handled by higher-level code");
 428 
 429   // Per JVMS 5.4.4 we first resolve and validate the current class, then
 430   // the target class k.
 431 
 432   InstanceKlass* cur_host = nest_host(CHECK_false);
 433   if (cur_host == NULL) {
 434     return false;
 435   }
 436 
 437   Klass* k_nest_host = k->nest_host(CHECK_false);
 438   if (k_nest_host == NULL) {
 439     return false;
 440   }
 441 
 442   bool access = (cur_host == k_nest_host);
 443 
 444   ResourceMark rm(THREAD);
 445   log_trace(class, nestmates)("Class %s does %shave nestmate access to %s",
 446                               this->external_name(),
 447                               access ? "" : "NOT ",
 448                               k->external_name());
 449   return access;
 450 }

 452 const char* InstanceKlass::nest_host_error(TRAPS) {
 453   if (_nest_host_index == 0) {
 454     return NULL;
 455   } else {
 456     constantPoolHandle cph(THREAD, constants());
 457     return SystemDictionary::find_nest_host_error(cph, (int)_nest_host_index);
 458   }

 461 InstanceKlass* InstanceKlass::allocate_instance_klass(const ClassFileParser& parser, TRAPS) {
 462   bool is_hidden_or_anonymous = parser.is_hidden() || parser.is_unsafe_anonymous();
 463   const int size = InstanceKlass::size(parser.vtable_size(),
 464                                        parser.itable_size(),
 465                                        nonstatic_oop_map_size(parser.total_oop_map_count()),
 466                                        parser.is_interface(),
 467                                        parser.is_unsafe_anonymous(),
 468                                        should_store_fingerprint(is_hidden_or_anonymous));
 469 
 470   const Symbol* const class_name = parser.class_name();
 471   assert(class_name != NULL, "invariant");
 472   ClassLoaderData* loader_data = parser.loader_data();
 473   assert(loader_data != NULL, "invariant");
 474 
 475   InstanceKlass* ik;
 476 
 477   // Allocation
 478   if (REF_NONE == parser.reference_type()) {
 479     if (class_name == vmSymbols::java_lang_Class()) {
 480       // mirror
 481       ik = new (loader_data, size, THREAD) InstanceMirrorKlass(parser);
 482     }
 483     else if (is_class_loader(class_name, parser)) {
 484       // class loader
 485       ik = new (loader_data, size, THREAD) InstanceClassLoaderKlass(parser);
 486     } else {
 487       // normal
 488       ik = new (loader_data, size, THREAD) InstanceKlass(parser, InstanceKlass::_kind_other);
 489     }
 490   } else {
 491     // reference
 492     ik = new (loader_data, size, THREAD) InstanceRefKlass(parser);
 493   }
 494 
 495   // Check for pending exception before adding to the loader data and incrementing
 496   // class count.  Can get OOM here.
 497   if (HAS_PENDING_EXCEPTION) {
 498     return NULL;
 499   }
 500 
 501   return ik;
 502 }

 526 InstanceKlass::InstanceKlass(const ClassFileParser& parser, unsigned kind, KlassID id) :
 527   Klass(id),
 528   _nest_members(NULL),
 529   _nest_host(NULL),
 530   _permitted_subclasses(NULL),
 531   _record_components(NULL),
 532   _static_field_size(parser.static_field_size()),
 533   _nonstatic_oop_map_size(nonstatic_oop_map_size(parser.total_oop_map_count())),
 534   _itable_len(parser.itable_size()),
 535   _nest_host_index(0),
 536   _init_state(allocated),
 537   _reference_type(parser.reference_type()),
 538   _init_thread(NULL)
 539 {
 540   set_vtable_length(parser.vtable_size());
 541   set_kind(kind);
 542   set_access_flags(parser.access_flags());
 543   if (parser.is_hidden()) set_is_hidden();
 544   set_is_unsafe_anonymous(parser.is_unsafe_anonymous());
 545   set_layout_helper(Klass::instance_layout_helper(parser.layout_size(),
 546                                                     false));
 547 
 548   assert(NULL == _methods, "underlying memory not zeroed?");
 549   assert(is_instance_klass(), "is layout incorrect?");
 550   assert(size_helper() == parser.layout_size(), "incorrect size_helper?");
 551 
 552   // Set biased locking bit for all instances of this class; it will be
 553   // cleared if revocation occurs too often for this type
 554   if (UseBiasedLocking && BiasedLocking::enabled()) {
 555     set_prototype_header(markWord::biased_locking_prototype());
 556   }
 557 }

 561   if (methods != NULL && methods != Universe::the_empty_method_array() &&
 562       !methods->is_shared()) {
 563     for (int i = 0; i < methods->length(); i++) {
 564       Method* method = methods->at(i);
 565       if (method == NULL) continue;  // maybe null if error processing
 566       // Only want to delete methods that are not executing for RedefineClasses.
 567       // The previous version will point to them so they're not totally dangling
 568       assert (!method->on_stack(), "shouldn't be called with methods on stack");
 569       MetadataFactory::free_metadata(loader_data, method);
 570     }
 571     MetadataFactory::free_array<Method*>(loader_data, methods);
 572   }

 609 // This function deallocates the metadata and C heap pointers that the
 610 // InstanceKlass points to.
 611 void InstanceKlass::deallocate_contents(ClassLoaderData* loader_data) {
 612 
 613   // Orphan the mirror first, CMS thinks it's still live.
 614   if (java_mirror() != NULL) {
 615     java_lang_Class::set_klass(java_mirror(), NULL);
 616   }
 617 
 618   // Also remove mirror from handles
 619   loader_data->remove_handle(_java_mirror);
 620 
 621   // Need to take this class off the class loader data list.
 622   loader_data->remove_class(this);
 623 
 624   // The array_klass for this class is created later, after error handling.
 625   // For class redefinition, we keep the original class so this scratch class
 626   // doesn't have an array class.  Either way, assert that there is nothing
 627   // to deallocate.
 628   assert(array_klasses() == NULL, "array classes shouldn't be created for this class yet");
 629 
 630   // Release C heap allocated data that this points to, which includes
 631   // reference counting symbol names.
 632   release_C_heap_structures_internal();
 633 
 634   deallocate_methods(loader_data, methods());
 635   set_methods(NULL);
 636 
 637   deallocate_record_components(loader_data, record_components());
 638   set_record_components(NULL);
 639 
 640   if (method_ordering() != NULL &&
 641       method_ordering() != Universe::the_empty_int_array() &&
 642       !method_ordering()->is_shared()) {
 643     MetadataFactory::free_array<int>(loader_data, method_ordering());
 644   }
 645   set_method_ordering(NULL);
 646 
 647   // default methods can be empty
 648   if (default_methods() != NULL &&
 649       default_methods() != Universe::the_empty_method_array() &&
 650       !default_methods()->is_shared()) {
 651     MetadataFactory::free_array<Method*>(loader_data, default_methods());
 652   }
 653   // Do NOT deallocate the default methods, they are owned by superinterfaces.
 654   set_default_methods(NULL);
 655 
 656   // default methods vtable indices can be empty
 657   if (default_vtable_indices() != NULL &&
 658       !default_vtable_indices()->is_shared()) {
 659     MetadataFactory::free_array<int>(loader_data, default_vtable_indices());
 660   }
 661   set_default_vtable_indices(NULL);
 662 
 663 
 664   // This array is in Klass, but remove it with the InstanceKlass since
 665   // this place would be the only caller and it can share memory with transitive
 666   // interfaces.
 667   if (secondary_supers() != NULL &&
 668       secondary_supers() != Universe::the_empty_klass_array() &&
 669       // see comments in compute_secondary_supers about the following cast
 670       (address)(secondary_supers()) != (address)(transitive_interfaces()) &&
 671       !secondary_supers()->is_shared()) {
 672     MetadataFactory::free_array<Klass*>(loader_data, secondary_supers());
 673   }
 674   set_secondary_supers(NULL);
 675 
 676   deallocate_interfaces(loader_data, super(), local_interfaces(), transitive_interfaces());
 677   set_transitive_interfaces(NULL);
 678   set_local_interfaces(NULL);
 679 
 680   if (fields() != NULL && !fields()->is_shared()) {
 681     MetadataFactory::free_array<jushort>(loader_data, fields());
 682   }
 683   set_fields(NULL, 0);
 684 
 685   // If a method from a redefined class is using this constant pool, don't
 686   // delete it, yet.  The new class's previous version will point to this.
 687   if (constants() != NULL) {
 688     assert (!constants()->on_stack(), "shouldn't be called if anything is onstack");
 689     if (!constants()->is_shared()) {
 690       MetadataFactory::free_metadata(loader_data, constants());
 691     }
 692     // Delete any cached resolution errors for the constant pool
 693     SystemDictionary::delete_resolution_error(constants());
 694 
 695     set_constants(NULL);
 696   }
 697 
 698   if (inner_classes() != NULL &&
 699       inner_classes() != Universe::the_empty_short_array() &&
 700       !inner_classes()->is_shared()) {
 701     MetadataFactory::free_array<jushort>(loader_data, inner_classes());
 702   }
 703   set_inner_classes(NULL);
 704 
 705   if (nest_members() != NULL &&
 706       nest_members() != Universe::the_empty_short_array() &&
 707       !nest_members()->is_shared()) {
 708     MetadataFactory::free_array<jushort>(loader_data, nest_members());
 709   }
 710   set_nest_members(NULL);
 711 
 712   if (permitted_subclasses() != NULL &&
 713       permitted_subclasses() != Universe::the_empty_short_array() &&
 714       !permitted_subclasses()->is_shared()) {
 715     MetadataFactory::free_array<jushort>(loader_data, permitted_subclasses());
 716   }
 717   set_permitted_subclasses(NULL);
 718 
 719   // We should deallocate the Annotations instance if it's not in shared spaces.
 720   if (annotations() != NULL && !annotations()->is_shared()) {
 721     MetadataFactory::free_metadata(loader_data, annotations());
 722   }
 723   set_annotations(NULL);
 724 
 725   if (Arguments::is_dumping_archive()) {
 726     SystemDictionaryShared::remove_dumptime_info(this);
 727   }
 728 }

 744 void InstanceKlass::eager_initialize(Thread *thread) {
 745   if (!EagerInitialization) return;
 746 
 747   if (this->is_not_initialized()) {
 748     // abort if the the class has a class initializer
 749     if (this->class_initializer() != NULL) return;
 750 
 751     // abort if it is java.lang.Object (initialization is handled in genesis)
 752     Klass* super_klass = super();
 753     if (super_klass == NULL) return;
 754 
 755     // abort if the super class should be initialized
 756     if (!InstanceKlass::cast(super_klass)->is_initialized()) return;
 757 
 758     // call body to expose the this pointer
 759     eager_initialize_impl();
 760   }
 761 }

 777 oop InstanceKlass::init_lock() const {
 778   // return the init lock from the mirror
 779   oop lock = java_lang_Class::init_lock(java_mirror());
 780   // Prevent reordering with any access of initialization state
 781   OrderAccess::loadload();
 782   assert((oop)lock != NULL || !is_not_initialized(), // initialized or in_error state
 783          "only fully initialized state can have a null lock");
 784   return lock;

 787 // Set the initialization lock to null so the object can be GC'ed.  Any racing
 788 // threads to get this lock will see a null lock and will not lock.
 789 // That's okay because they all check for initialized state after getting
 790 // the lock and return.
 791 void InstanceKlass::fence_and_clear_init_lock() {
 792   // make sure previous stores are all done, notably the init_state.
 793   OrderAccess::storestore();
 794   java_lang_Class::set_init_lock(java_mirror(), NULL);
 795   assert(!is_not_initialized(), "class must be initialized now");
 796 }

 798 void InstanceKlass::eager_initialize_impl() {
 799   EXCEPTION_MARK;
 800   HandleMark hm(THREAD);
 801   Handle h_init_lock(THREAD, init_lock());
 802   ObjectLocker ol(h_init_lock, THREAD, h_init_lock() != NULL);
 803 
 804   // abort if someone beat us to the initialization
 805   if (!is_not_initialized()) return;  // note: not equivalent to is_initialized()
 806 
 807   ClassState old_state = init_state();
 808   link_class_impl(THREAD);
 809   if (HAS_PENDING_EXCEPTION) {
 810     CLEAR_PENDING_EXCEPTION;
 811     // Abort if linking the class throws an exception.
 812 
 813     // Use a test to avoid redundantly resetting the state if there's
 814     // no change.  Set_init_state() asserts that state changes make
 815     // progress, whereas here we might just be spinning in place.
 816     if (old_state != _init_state)
 817       set_init_state(old_state);
 818   } else {
 819     // linking successfull, mark class as initialized
 820     set_init_state(fully_initialized);
 821     fence_and_clear_init_lock();
 822     // trace
 823     if (log_is_enabled(Info, class, init)) {
 824       ResourceMark rm(THREAD);
 825       log_info(class, init)("[Initialized %s without side effects]", external_name());
 826     }
 827   }
 828 }

 868 bool InstanceKlass::link_class_impl(TRAPS) {
 869   if (DumpSharedSpaces && SystemDictionaryShared::has_class_failed_verification(this)) {
 870     // This is for CDS dumping phase only -- we use the in_error_state to indicate that
 871     // the class has failed verification. Throwing the NoClassDefFoundError here is just
 872     // a convenient way to stop repeat attempts to verify the same (bad) class.
 873     //
 874     // Note that the NoClassDefFoundError is not part of the JLS, and should not be thrown
 875     // if we are executing Java code. This is not a problem for CDS dumping phase since
 876     // it doesn't execute any Java code.
 877     ResourceMark rm(THREAD);
 878     Exceptions::fthrow(THREAD_AND_LOCATION,
 879                        vmSymbols::java_lang_NoClassDefFoundError(),
 880                        "Class %s, or one of its supertypes, failed class initialization",
 881                        external_name());
 882     return false;
 883   }
 884   // return if already verified
 885   if (is_linked()) {
 886     return true;
 887   }
 888 
 889   // Timing
 890   // timer handles recursion
 891   assert(THREAD->is_Java_thread(), "non-JavaThread in link_class_impl");
 892   JavaThread* jt = (JavaThread*)THREAD;
 893 
 894   // link super class before linking this class
 895   Klass* super_klass = super();
 896   if (super_klass != NULL) {
 897     if (super_klass->is_interface()) {  // check if super class is an interface
 898       ResourceMark rm(THREAD);
 899       Exceptions::fthrow(
 900         THREAD_AND_LOCATION,
 901         vmSymbols::java_lang_IncompatibleClassChangeError(),
 902         "class %s has interface %s as super class",
 903         external_name(),
 904         super_klass->external_name()
 905       );
 906       return false;
 907     }
 908 
 909     InstanceKlass* ik_super = InstanceKlass::cast(super_klass);
 910     ik_super->link_class_impl(CHECK_false);
 911   }
 912 
 913   // link all interfaces implemented by this class before linking this class
 914   Array<InstanceKlass*>* interfaces = local_interfaces();
 915   int num_interfaces = interfaces->length();
 916   for (int index = 0; index < num_interfaces; index++) {
 917     InstanceKlass* interk = interfaces->at(index);
 918     interk->link_class_impl(CHECK_false);
 919   }
 920 
 921   // in case the class is linked in the process of linking its superclasses
 922   if (is_linked()) {
 923     return true;
 924   }
 925 
 926   // trace only the link time for this klass that includes
 927   // the verification time
 928   PerfClassTraceTime vmtimer(ClassLoader::perf_class_link_time(),
 929                              ClassLoader::perf_class_link_selftime(),
 930                              ClassLoader::perf_classes_linked(),
 931                              jt->get_thread_stat()->perf_recursion_counts_addr(),
 932                              jt->get_thread_stat()->perf_timers_addr(),
 933                              PerfClassTraceTime::CLASS_LINK);
 934 
 935   // verification & rewriting
 936   {
 937     HandleMark hm(THREAD);
 938     Handle h_init_lock(THREAD, init_lock());
 939     ObjectLocker ol(h_init_lock, THREAD, h_init_lock() != NULL);
 940     // rewritten will have been set if loader constraint error found
 941     // on an earlier link attempt
 942     // don't verify or rewrite if already rewritten
 943     //
 944 
 945     if (!is_linked()) {
 946       if (!is_rewritten()) {
 947         {
 948           bool verify_ok = verify_code(THREAD);
 949           if (!verify_ok) {
 950             return false;
 951           }
 952         }
 953 
 954         // Just in case a side-effect of verify linked this class already
 955         // (which can sometimes happen since the verifier loads classes
 956         // using custom class loaders, which are free to initialize things)
 957         if (is_linked()) {
 958           return true;
 959         }
 960 
 961         // also sets rewritten
 962         rewrite_class(CHECK_false);
 963       } else if (is_shared()) {
 964         SystemDictionaryShared::check_verification_constraints(this, CHECK_false);
 965       }
 966 
 967       // relocate jsrs and link methods after they are all rewritten
 968       link_methods(CHECK_false);
 969 
 970       // Initialize the vtable and interface table after
 971       // methods have been rewritten since rewrite may
 972       // fabricate new Method*s.
 973       // also does loader constraint checking
 974       //
 975       // initialize_vtable and initialize_itable need to be rerun
 976       // for a shared class if
 977       // 1) the class is loaded by custom class loader or
 978       // 2) the class is loaded by built-in class loader but failed to add archived loader constraints
 979       bool need_init_table = true;
 980       if (is_shared() && SystemDictionaryShared::check_linking_constraints(this, THREAD)) {
 981         need_init_table = false;
 982       }
 983       if (need_init_table) {
 984         vtable().initialize_vtable(true, CHECK_false);
 985         itable().initialize_itable(true, CHECK_false);
 986       }
 987 #ifdef ASSERT
 988       vtable().verify(tty, true);
 989       // In case itable verification is ever added.
 990       // itable().verify(tty, true);
 991 #endif
 992       set_init_state(linked);
 993       if (JvmtiExport::should_post_class_prepare()) {
 994         Thread *thread = THREAD;
 995         assert(thread->is_Java_thread(), "thread->is_Java_thread()");
 996         JvmtiExport::post_class_prepare((JavaThread *) thread, this);
 997       }
 998     }
 999   }
1000   return true;
1001 }

1029 // Eagerly initialize superinterfaces that declare default methods (concrete instance: any access)
1030 void InstanceKlass::initialize_super_interfaces(TRAPS) {
1031   assert (has_nonstatic_concrete_methods(), "caller should have checked this");
1032   for (int i = 0; i < local_interfaces()->length(); ++i) {
1033     InstanceKlass* ik = local_interfaces()->at(i);
1034 
1035     // Initialization is depth first search ie. we start with top of the inheritance tree
1036     // has_nonstatic_concrete_methods drives searching superinterfaces since it
1037     // means has_nonstatic_concrete_methods in its superinterface hierarchy
1038     if (ik->has_nonstatic_concrete_methods()) {
1039       ik->initialize_super_interfaces(CHECK);
1040     }
1041 
1042     // Only initialize() interfaces that "declare" concrete methods.
1043     if (ik->should_be_initialized() && ik->declares_nonstatic_concrete_methods()) {
1044       ik->initialize(CHECK);
1045     }
1046   }
1047 }

1049 void InstanceKlass::initialize_impl(TRAPS) {
1050   HandleMark hm(THREAD);
1051 
1052   // Make sure klass is linked (verified) before initialization
1053   // A class could already be verified, since it has been reflected upon.
1054   link_class(CHECK);
1055 
1056   DTRACE_CLASSINIT_PROBE(required, -1);
1057 
1058   bool wait = false;
1059 
1060   assert(THREAD->is_Java_thread(), "non-JavaThread in initialize_impl");
1061   JavaThread* jt = (JavaThread*)THREAD;
1062 
1063   // refer to the JVM book page 47 for description of steps
1064   // Step 1
1065   {
1066     Handle h_init_lock(THREAD, init_lock());
1067     ObjectLocker ol(h_init_lock, THREAD, h_init_lock() != NULL);
1068 
1069     // Step 2
1070     // If we were to use wait() instead of waitInterruptibly() then
1071     // we might end up throwing IE from link/symbol resolution sites
1072     // that aren't expected to throw.  This would wreak havoc.  See 6320309.
1073     while (is_being_initialized() && !is_reentrant_initialization(jt)) {
1074       wait = true;
1075       jt->set_class_to_be_initialized(this);
1076       ol.wait_uninterruptibly(jt);
1077       jt->set_class_to_be_initialized(NULL);
1078     }
1079 
1080     // Step 3
1081     if (is_being_initialized() && is_reentrant_initialization(jt)) {
1082       DTRACE_CLASSINIT_PROBE_WAIT(recursive, -1, wait);
1083       return;
1084     }
1085 
1086     // Step 4
1087     if (is_initialized()) {
1088       DTRACE_CLASSINIT_PROBE_WAIT(concurrent, -1, wait);
1089       return;
1090     }
1091 
1092     // Step 5
1093     if (is_in_error_state()) {
1094       DTRACE_CLASSINIT_PROBE_WAIT(erroneous, -1, wait);
1095       ResourceMark rm(THREAD);
1096       const char* desc = "Could not initialize class ";
1097       const char* className = external_name();
1098       size_t msglen = strlen(desc) + strlen(className) + 1;
1099       char* message = NEW_RESOURCE_ARRAY(char, msglen);
1100       if (NULL == message) {
1101         // Out of memory: can't create detailed error message
1102           THROW_MSG(vmSymbols::java_lang_NoClassDefFoundError(), className);
1103       } else {
1104         jio_snprintf(message, msglen, "%s%s", desc, className);
1105           THROW_MSG(vmSymbols::java_lang_NoClassDefFoundError(), message);
1106       }
1107     }
1108 
1109     // Step 6
1110     set_init_state(being_initialized);
1111     set_init_thread(jt);
1112   }
1113 
1114   // Step 7
1115   // Next, if C is a class rather than an interface, initialize it's super class and super
1116   // interfaces.
1117   if (!is_interface()) {
1118     Klass* super_klass = super();
1119     if (super_klass != NULL && super_klass->should_be_initialized()) {
1120       super_klass->initialize(THREAD);
1121     }
1122     // If C implements any interface that declares a non-static, concrete method,
1123     // the initialization of C triggers initialization of its super interfaces.
1124     // Only need to recurse if has_nonstatic_concrete_methods which includes declaring and
1125     // having a superinterface that declares, non-static, concrete methods
1126     if (!HAS_PENDING_EXCEPTION && has_nonstatic_concrete_methods()) {
1127       initialize_super_interfaces(THREAD);
1128     }
1129 
1130     // If any exceptions, complete abruptly, throwing the same exception as above.
1131     if (HAS_PENDING_EXCEPTION) {
1132       Handle e(THREAD, PENDING_EXCEPTION);
1133       CLEAR_PENDING_EXCEPTION;
1134       {
1135         EXCEPTION_MARK;
1136         // Locks object, set state, and notify all waiting threads
1137         set_initialization_state_and_notify(initialization_error, THREAD);
1138         CLEAR_PENDING_EXCEPTION;
1139       }
1140       DTRACE_CLASSINIT_PROBE_WAIT(super__failed, -1, wait);
1141       THROW_OOP(e());
1142     }
1143   }
1144 
1145 
1146   // Look for aot compiled methods for this klass, including class initializer.
1147   AOTLoader::load_for_klass(this, THREAD);
1148 
1149   // Step 8
1150   {
1151     DTRACE_CLASSINIT_PROBE_WAIT(clinit, -1, wait);
1152     // Timer includes any side effects of class initialization (resolution,
1153     // etc), but not recursive entry into call_class_initializer().
1154     PerfClassTraceTime timer(ClassLoader::perf_class_init_time(),
1155                              ClassLoader::perf_class_init_selftime(),
1156                              ClassLoader::perf_classes_inited(),
1157                              jt->get_thread_stat()->perf_recursion_counts_addr(),

1251 // The embedded _implementor field can only record one implementor.
1252 // When there are more than one implementors, the _implementor field
1253 // is set to the interface Klass* itself. Following are the possible
1254 // values for the _implementor field:
1255 //   NULL                  - no implementor
1256 //   implementor Klass*    - one implementor
1257 //   self                  - more than one implementor
1258 //
1259 // The _implementor field only exists for interfaces.
1260 void InstanceKlass::add_implementor(Klass* k) {
1261   if (Universe::is_fully_initialized()) {
1262     assert_lock_strong(Compile_lock);
1263   }
1264   assert(is_interface(), "not interface");
1265   // Filter out my subinterfaces.
1266   // (Note: Interfaces are never on the subklass list.)
1267   if (InstanceKlass::cast(k)->is_interface()) return;
1268 
1269   // Filter out subclasses whose supers already implement me.
1270   // (Note: CHA must walk subclasses of direct implementors
1271   // in order to locate indirect implementors.)
1272   Klass* sk = k->super();
1273   if (sk != NULL && InstanceKlass::cast(sk)->implements_interface(this))
1274     // We only need to check one immediate superclass, since the
1275     // implements_interface query looks at transitive_interfaces.
1276     // Any supers of the super have the same (or fewer) transitive_interfaces.
1277     return;
1278 
1279   Klass* ik = implementor();
1280   if (ik == NULL) {
1281     set_implementor(k);
1282   } else if (ik != this && ik != k) {
1283     // There is already an implementor. Use itself as an indicator of
1284     // more than one implementors.
1285     set_implementor(this);
1286   }
1287 
1288   // The implementor also implements the transitive_interfaces
1289   for (int index = 0; index < local_interfaces()->length(); index++) {
1290     InstanceKlass::cast(local_interfaces()->at(index))->add_implementor(k);
1291   }
1292 }

1547   // search order according to newest JVM spec (5.4.3.2, p.167).
1548   // 1) search for field in current klass

1558     if (supr != NULL) return InstanceKlass::cast(supr)->find_field(name, sig, fd);

1722 inline int InstanceKlass::quick_search(const Array<Method*>* methods, const Symbol* name) {
1723   if (_disable_method_binary_search) {
1724     assert(DynamicDumpSharedSpaces, "must be");
1725     // At the final stage of dynamic dumping, the methods array may not be sorted
1726     // by ascending addresses of their names, so we can't use binary search anymore.
1727     // However, methods with the same name are still laid out consecutively inside the
1728     // methods array, so let's look for the first one that matches.
1729     return linear_search(methods, name);
1730   }
1731 
1732   int len = methods->length();
1733   int l = 0;
1734   int h = len - 1;
1735 
1736   // methods are sorted by ascending addresses of their names, so do binary search
1737   while (l <= h) {

1745       l = mid + 1;
1746     } else {
1747       h = mid - 1;
1748     }
1749   }
1750   return -1;

1855 // true if method matches signature and conforms to skipping_X conditions.
1856 static bool method_matches(const Method* m,
1857                            const Symbol* signature,
1858                            bool skipping_overpass,
1859                            bool skipping_static,
1860                            bool skipping_private) {
1861   return ((m->signature() == signature) &&

1867 // Used directly for default_methods to find the index into the
1868 // default_vtable_indices, and indirectly by find_method
1869 // find_method_index looks in the local methods array to return the index
1870 // of the matching name/signature. If, overpass methods are being ignored,
1871 // the search continues to find a potential non-overpass match.  This capability
1872 // is important during method resolution to prefer a static method, for example,
1873 // over an overpass method.
1874 // There is the possibility in any _method's array to have the same name/signature
1875 // for a static method, an overpass method and a local instance method
1876 // To correctly catch a given method, the search criteria may need
1877 // to explicitly skip the other two. For local instance methods, it
1878 // is often necessary to skip private methods
1879 int InstanceKlass::find_method_index(const Array<Method*>* methods,
1880                                      const Symbol* name,
1881                                      const Symbol* signature,
1882                                      OverpassLookupMode overpass_mode,
1883                                      StaticLookupMode static_mode,
1884                                      PrivateLookupMode private_mode) {
1885   const bool skipping_overpass = (overpass_mode == skip_overpass);
1886   const bool skipping_static = (static_mode == skip_static);
1887   const bool skipping_private = (private_mode == skip_private);
1888   const int hit = quick_search(methods, name);
1889   if (hit != -1) {
1890     const Method* const m = methods->at(hit);
1891 
1892     // Do linear search to find matching signature.  First, quick check
1893     // for common case, ignoring overpasses if requested.
1894     if (method_matches(m, signature, skipping_overpass, skipping_static, skipping_private)) {
1895       return hit;
1896     }
1897 
1898     // search downwards through overloaded methods
1899     int i;
1900     for (i = hit - 1; i >= 0; --i) {
1901         const Method* const m = methods->at(i);
1902         assert(m->is_method(), "must be method");
1903         if (m->name() != name) {
1904           break;
1905         }
1906         if (method_matches(m, signature, skipping_overpass, skipping_static, skipping_private)) {
1907           return i;
1908         }
1909     }
1910     // search upwards
1911     for (i = hit + 1; i < methods->length(); ++i) {
1912         const Method* const m = methods->at(i);
1913         assert(m->is_method(), "must be method");
1914         if (m->name() != name) {
1915           break;
1916         }
1917         if (method_matches(m, signature, skipping_overpass, skipping_static, skipping_private)) {
1918           return i;
1919         }
1920     }
1921     // not found
1922 #ifdef ASSERT
1923     const int index = (skipping_overpass || skipping_static || skipping_private) ? -1 :
1924       linear_search(methods, name, signature);
1925     assert(-1 == index, "binary search should have found entry %d", index);
1926 #endif
1927   }
1928   return -1;
1929 }

2075 // Lookup or create a jmethodID.
2076 // This code is called by the VMThread and JavaThreads so the
2077 // locking has to be done very carefully to avoid deadlocks
2078 // and/or other cache consistency problems.
2079 //
2080 jmethodID InstanceKlass::get_jmethod_id(const methodHandle& method_h) {
2081   size_t idnum = (size_t)method_h->method_idnum();
2082   jmethodID* jmeths = methods_jmethod_ids_acquire();
2083   size_t length = 0;
2084   jmethodID id = NULL;
2085 
2086   // We use a double-check locking idiom here because this cache is
2087   // performance sensitive. In the normal system, this cache only
2088   // transitions from NULL to non-NULL which is safe because we use
2089   // release_set_methods_jmethod_ids() to advertise the new cache.
2090   // A partially constructed cache should never be seen by a racing
2091   // thread. We also use release_store() to save a new jmethodID
2092   // in the cache so a partially constructed jmethodID should never be
2093   // seen either. Cache reads of existing jmethodIDs proceed without a
2094   // lock, but cache writes of a new jmethodID requires uniqueness and
2095   // creation of the cache itself requires no leaks so a lock is
2096   // generally acquired in those two cases.
2097   //
2098   // If the RedefineClasses() API has been used, then this cache can
2099   // grow and we'll have transitions from non-NULL to bigger non-NULL.
2100   // Cache creation requires no leaks and we require safety between all
2101   // cache accesses and freeing of the old cache so a lock is generally
2102   // acquired when the RedefineClasses() API has been used.
2103 
2104   if (jmeths != NULL) {
2105     // the cache already exists
2106     if (!idnum_can_increment()) {
2107       // the cache can't grow so we can just get the current values
2108       get_jmethod_id_length_value(jmeths, idnum, &length, &id);
2109     } else {
2110       // cache can grow so we have to be more careful
2111       if (Threads::number_of_threads() == 0 ||
2112           SafepointSynchronize::is_at_safepoint()) {
2113         // we're single threaded or at a safepoint - no locking needed
2114         get_jmethod_id_length_value(jmeths, idnum, &length, &id);
2115       } else {
2116         MutexLocker ml(JmethodIdCreation_lock, Mutex::_no_safepoint_check_flag);
2117         get_jmethod_id_length_value(jmeths, idnum, &length, &id);
2118       }
2119     }
2120   }
2121   // implied else:
2122   // we need to allocate a cache so default length and id values are good
2123 
2124   if (jmeths == NULL ||   // no cache yet
2125       length <= idnum ||  // cache is too short
2126       id == NULL) {       // cache doesn't contain entry
2127 
2128     // This function can be called by the VMThread so we have to do all
2129     // things that might block on a safepoint before grabbing the lock.
2130     // Otherwise, we can deadlock with the VMThread or have a cache
2131     // consistency issue. These vars keep track of what we might have
2132     // to free after the lock is dropped.
2133     jmethodID  to_dealloc_id     = NULL;
2134     jmethodID* to_dealloc_jmeths = NULL;
2135 
2136     // may not allocate new_jmeths or use it if we allocate it
2137     jmethodID* new_jmeths = NULL;
2138     if (length <= idnum) {
2139       // allocate a new cache that might be used
2140       size_t size = MAX2(idnum+1, (size_t)idnum_allocated_count());
2141       new_jmeths = NEW_C_HEAP_ARRAY(jmethodID, size+1, mtClass);
2142       memset(new_jmeths, 0, (size+1)*sizeof(jmethodID));
2143       // cache size is stored in element[0], other elements offset by one
2144       new_jmeths[0] = (jmethodID)size;
2145     }
2146 
2147     // allocate a new jmethodID that might be used
2148     jmethodID new_id = NULL;
2149     if (method_h->is_old() && !method_h->is_obsolete()) {
2150       // The method passed in is old (but not obsolete), we need to use the current version
2151       Method* current_method = method_with_idnum((int)idnum);
2152       assert(current_method != NULL, "old and but not obsolete, so should exist");
2153       new_id = Method::make_jmethod_id(class_loader_data(), current_method);
2154     } else {
2155       // It is the current version of the method or an obsolete method,
2156       // use the version passed in
2157       new_id = Method::make_jmethod_id(class_loader_data(), method_h());
2158     }
2159 
2160     if (Threads::number_of_threads() == 0 ||
2161         SafepointSynchronize::is_at_safepoint()) {
2162       // we're single threaded or at a safepoint - no locking needed
2163       id = get_jmethod_id_fetch_or_update(idnum, new_id, new_jmeths,
2164                                           &to_dealloc_id, &to_dealloc_jmeths);
2165     } else {
2166       MutexLocker ml(JmethodIdCreation_lock, Mutex::_no_safepoint_check_flag);
2167       id = get_jmethod_id_fetch_or_update(idnum, new_id, new_jmeths,
2168                                           &to_dealloc_id, &to_dealloc_jmeths);
2169     }
2170 
2171     // The lock has been dropped so we can free resources.
2172     // Free up either the old cache or the new cache if we allocated one.
2173     if (to_dealloc_jmeths != NULL) {
2174       FreeHeap(to_dealloc_jmeths);
2175     }
2176     // free up the new ID since it wasn't needed
2177     if (to_dealloc_id != NULL) {
2178       Method::destroy_jmethod_id(class_loader_data(), to_dealloc_id);
2179     }
2180   }
2181   return id;
2182 }

2204 // Common code to fetch the jmethodID from the cache or update the
2205 // cache with the new jmethodID. This function should never do anything
2206 // that causes the caller to go to a safepoint or we can deadlock with
2207 // the VMThread or have cache consistency issues.
2208 //
2209 jmethodID InstanceKlass::get_jmethod_id_fetch_or_update(
2210             size_t idnum, jmethodID new_id,
2211             jmethodID* new_jmeths, jmethodID* to_dealloc_id_p,
2212             jmethodID** to_dealloc_jmeths_p) {
2213   assert(new_id != NULL, "sanity check");
2214   assert(to_dealloc_id_p != NULL, "sanity check");
2215   assert(to_dealloc_jmeths_p != NULL, "sanity check");
2216   assert(Threads::number_of_threads() == 0 ||
2217          SafepointSynchronize::is_at_safepoint() ||
2218          JmethodIdCreation_lock->owned_by_self(), "sanity check");
2219 
2220   // reacquire the cache - we are locked, single threaded or at a safepoint
2221   jmethodID* jmeths = methods_jmethod_ids_acquire();
2222   jmethodID  id     = NULL;
2223   size_t     length = 0;
2224 
2225   if (jmeths == NULL ||                         // no cache yet
2226       (length = (size_t)jmeths[0]) <= idnum) {  // cache is too short
2227     if (jmeths != NULL) {
2228       // copy any existing entries from the old cache
2229       for (size_t index = 0; index < length; index++) {
2230         new_jmeths[index+1] = jmeths[index+1];
2231       }
2232       *to_dealloc_jmeths_p = jmeths;  // save old cache for later delete
2233     }
2234     release_set_methods_jmethod_ids(jmeths = new_jmeths);
2235   } else {
2236     // fetch jmethodID (if any) from the existing cache
2237     id = jmeths[idnum+1];
2238     *to_dealloc_jmeths_p = new_jmeths;  // save new cache for later delete
2239   }
2240   if (id == NULL) {
2241     // No matching jmethodID in the existing cache or we have a new
2242     // cache or we just grew the cache. This cache write is done here
2243     // by the first thread to win the foot race because a jmethodID
2244     // needs to be unique once it is generally available.
2245     id = new_id;
2246 
2247     // The jmethodID cache can be read while unlocked so we have to
2248     // make sure the new jmethodID is complete before installing it
2249     // in the cache.
2250     Atomic::release_store(&jmeths[idnum+1], id);
2251   } else {
2252     *to_dealloc_id_p = new_id; // save new id for later delete
2253   }
2254   return id;
2255 }

3238 nmethod* InstanceKlass::lookup_osr_nmethod(const Method* m, int bci, int comp_level, bool match_level) const {
3239   MutexLocker ml(CompiledMethod_lock->owned_by_self() ? NULL : CompiledMethod_lock,
3240                  Mutex::_no_safepoint_check_flag);
3241   nmethod* osr = osr_nmethods_head();

3246     // for a c2 version. When c2 completes its osr nmethod we will trash
3247     // the c1 version and only be able to find the c2 version. However
3248     // while we overflow in the c1 code at back branches we don't want to
3249     // try and switch to the same code as we are already running
3250 
3251     if (osr->method() == m &&
3252         (bci == InvocationEntryBci || osr->osr_entry_bci() == bci)) {
3253       if (match_level) {
3254         if (osr->comp_level() == comp_level) {
3255           // Found a match - return it.
3256           return osr;
3257         }
3258       } else {
3259         if (best == NULL || (osr->comp_level() > best->comp_level())) {
3260           if (osr->comp_level() == CompLevel_highest_tier) {
3261             // Found the best possible - return it.
3262             return osr;
3263           }
3264           best = osr;
3265         }
3266       }
3267     }
3268     osr = osr->osr_link();
3269   }
3270

3272   if (best != NULL && best->comp_level() >= comp_level) {

3870 // Purge previous versions before adding new previous versions of the class and
3871 // during class unloading.
3872 void InstanceKlass::purge_previous_version_list() {
3873   assert(SafepointSynchronize::is_at_safepoint(), "only called at safepoint");
3874   assert(has_been_redefined(), "Should only be called for main class");
3875 
3876   // Quick exit.
3877   if (previous_versions() == NULL) {
3878     return;
3879   }
3880 
3881   // This klass has previous versions so see what we can cleanup
3882   // while it is safe to do so.
3883 
3884   int deleted_count = 0;    // leave debugging breadcrumbs
3885   int live_count = 0;
3886   ClassLoaderData* loader_data = class_loader_data();
3887   assert(loader_data != NULL, "should never be null");
3888 
3889   ResourceMark rm;
3890   log_trace(redefine, class, iklass, purge)("%s: previous versions", external_name());
3891 
3892   // previous versions are linked together through the InstanceKlass
3893   InstanceKlass* pv_node = previous_versions();
3894   InstanceKlass* last = this;
3895   int version = 0;
3896 
3897   // check the previous versions list
3898   for (; pv_node != NULL; ) {
3899 
3900     ConstantPool* pvcp = pv_node->constants();
3901     assert(pvcp != NULL, "cp ref was unexpectedly cleared");
3902 
3903     if (!pvcp->on_stack()) {
3904       // If the constant pool isn't on stack, none of the methods
3905       // are executing.  Unlink this previous_version.
3906       // The previous version InstanceKlass is on the ClassLoaderData deallocate list
3907       // so will be deallocated during the next phase of class unloading.
3908       log_trace(redefine, class, iklass, purge)
3909         ("previous version " INTPTR_FORMAT " is dead.", p2i(pv_node));
3910       // For debugging purposes.
3911       pv_node->set_is_scratch_class();
3912       // Unlink from previous version list.
3913       assert(pv_node->class_loader_data() == loader_data, "wrong loader_data");
3914       InstanceKlass* next = pv_node->previous_versions();
3915       pv_node->link_previous_versions(NULL);   // point next to NULL
3916       last->link_previous_versions(next);
3917       // Add to the deallocate list after unlinking
3918       loader_data->add_to_deallocate_list(pv_node);
3919       pv_node = next;
3920       deleted_count++;
3921       version++;
3922       continue;
3923     } else {
3924       log_trace(redefine, class, iklass, purge)("previous version " INTPTR_FORMAT " is alive", p2i(pv_node));
3925       assert(pvcp->pool_holder() != NULL, "Constant pool with no holder");
3926       guarantee (!loader_data->is_unloading(), "unloaded classes can't be on the stack");
3927       live_count++;
3928       // found a previous version for next time we do class unloading
3929       _has_previous_versions = true;
3930     }
3931 
3932     // At least one method is live in this previous version.
3933     // Reset dead EMCP methods not to get breakpoints.
3934     // All methods are deallocated when all of the methods for this class are no
3935     // longer running.
3936     Array<Method*>* method_refs = pv_node->methods();
3937     if (method_refs != NULL) {
3938       log_trace(redefine, class, iklass, purge)("previous methods length=%d", method_refs->length());
3939       for (int j = 0; j < method_refs->length(); j++) {
3940         Method* method = method_refs->at(j);
3941 
3942         if (!method->on_stack()) {
3943           // no breakpoints for non-running methods
3944           if (method->is_running_emcp()) {
3945             method->set_running_emcp(false);
3946           }
3947         } else {
3948           assert (method->is_obsolete() || method->is_running_emcp(),
3949                   "emcp method cannot run after emcp bit is cleared");
3950           log_trace(redefine, class, iklass, purge)
3951             ("purge: %s(%s): prev method @%d in version @%d is alive",
3952              method->name()->as_C_string(), method->signature()->as_C_string(), j, version);
3953         }
3954       }
3955     }
3956     // next previous version
3957     last = pv_node;
3958     pv_node = pv_node->previous_versions();
3959     version++;
3960   }
3961   log_trace(redefine, class, iklass, purge)
3962     ("previous version stats: live=%d, deleted=%d", live_count, deleted_count);

4022 // Save the scratch_class as the previous version if any of the methods are running.
4023 // The previous_versions are used to set breakpoints in EMCP methods and they are
4024 // also used to clean MethodData links to redefined methods that are no longer running.
4025 void InstanceKlass::add_previous_version(InstanceKlass* scratch_class,
4026                                          int emcp_method_count) {
4027   assert(Thread::current()->is_VM_thread(),
4028          "only VMThread can add previous versions");
4029 
4030   ResourceMark rm;
4031   log_trace(redefine, class, iklass, add)
4032     ("adding previous version ref for %s, EMCP_cnt=%d", scratch_class->external_name(), emcp_method_count);
4033 
4034   // Clean out old previous versions for this class
4035   purge_previous_version_list();
4036 
4037   // Mark newly obsolete methods in remaining previous versions.  An EMCP method from
4038   // a previous redefinition may be made obsolete by this redefinition.
4039   Array<Method*>* old_methods = scratch_class->methods();
4040   mark_newly_obsolete_methods(old_methods, emcp_method_count);
4041 
4042   // If the constant pool for this previous version of the class
4043   // is not marked as being on the stack, then none of the methods
4044   // in this previous version of the class are on the stack so
4045   // we don't need to add this as a previous version.
4046   ConstantPool* cp_ref = scratch_class->constants();
4047   if (!cp_ref->on_stack()) {
4048     log_trace(redefine, class, iklass, add)("scratch class not added; no methods are running");
4049     // For debugging purposes.
4050     scratch_class->set_is_scratch_class();
4051     scratch_class->class_loader_data()->add_to_deallocate_list(scratch_class);
4052     return;
4053   }
4054 
4055   if (emcp_method_count != 0) {
4056     // At least one method is still running, check for EMCP methods
4057     for (int i = 0; i < old_methods->length(); i++) {
4058       Method* old_method = old_methods->at(i);
4059       if (!old_method->is_obsolete() && old_method->on_stack()) {
4060         // if EMCP method (not obsolete) is on the stack, mark as EMCP so that
4061         // we can add breakpoints for it.
4062 
4063         // We set the method->on_stack bit during safepoints for class redefinition
4064         // and use this bit to set the is_running_emcp bit.
4065         // After the safepoint, the on_stack bit is cleared and the running emcp
4066         // method may exit.   If so, we would set a breakpoint in a method that
4067         // is never reached, but this won't be noticeable to the programmer.
4068         old_method->set_running_emcp(true);
4069         log_trace(redefine, class, iklass, add)
4070           ("EMCP method %s is on_stack " INTPTR_FORMAT, old_method->name_and_sig_as_C_string(), p2i(old_method));
4071       } else if (!old_method->is_obsolete()) {
4072         log_trace(redefine, class, iklass, add)
4073           ("EMCP method %s is NOT on_stack " INTPTR_FORMAT, old_method->name_and_sig_as_C_string(), p2i(old_method));
4074       }
4075     }
4076   }
4077 
4078   // Add previous version if any methods are still running.
4079   // Set has_previous_version flag for processing during class unloading.
4080   _has_previous_versions = true;
4081   log_trace(redefine, class, iklass, add) ("scratch class added; one of its methods is on_stack.");
4082   assert(scratch_class->previous_versions() == NULL, "shouldn't have a previous version");
4083   scratch_class->link_previous_versions(previous_versions());
4084   link_previous_versions(scratch_class);
