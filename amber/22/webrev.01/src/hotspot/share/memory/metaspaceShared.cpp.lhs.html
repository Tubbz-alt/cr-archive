  91 // The CDS archive is divided into the following regions:
  92 //     mc  - misc code (the method entry trampolines, c++ vtables)
  93 //     rw  - read-write metadata
  94 //     ro  - read-only metadata and read-only tables
  95 //
  96 //     ca0 - closed archive heap space #0
  97 //     ca1 - closed archive heap space #1 (may be empty)
  98 //     oa0 - open archive heap space #0
  99 //     oa1 - open archive heap space #1 (may be empty)
 100 //
 101 // The mc, rw, and ro regions are linearly allocated, starting from
 102 // SharedBaseAddress, in the order of mc->rw->ro. The size of these 3 regions
 103 // are page-aligned, and there's no gap between any consecutive regions.
 104 //
 105 // These 3 regions are populated in the following steps:
 106 // [1] All classes are loaded in MetaspaceShared::preload_classes(). All metadata are
 107 //     temporarily allocated outside of the shared regions. Only the method entry
 108 //     trampolines are written into the mc region.
 109 // [2] C++ vtables are copied into the mc region.
 110 // [3] ArchiveCompactor copies RW metadata into the rw region.
 111 // [4] ArchiveCompactor copies RO metadata into the ro region.
 112 // [5] SymbolTable, StringTable, SystemDictionary, and a few other read-only data
 113 //     are copied into the ro region as read-only tables.
 114 //
 115 // The s0/s1 and oa0/oa1 regions are populated inside HeapShared::archive_java_heap_objects.
 116 // Their layout is independent of the other 4 regions.
 117 
 118 char* DumpRegion::expand_top_to(char* newtop) {
 119   assert(is_allocatable(), "must be initialized and not packed");
 120   assert(newtop >= _top, "must not grow backwards");
 121   if (newtop > _end) {
 122     MetaspaceShared::report_out_of_space(_name, newtop - _top);
 123     ShouldNotReachHere();
 124   }
 125 
 126   if (_rs == MetaspaceShared::shared_rs()) {
 127     uintx delta;
 128     if (DynamicDumpSharedSpaces) {
 129       delta = DynamicArchive::object_delta_uintx(newtop);
 130     } else {
 131       delta = MetaspaceShared::object_delta_uintx(newtop);
 132     }
 133     if (delta > MAX_SHARED_DELTA) {
 134       // This is just a sanity check and should not appear in any real world usage. This
 135       // happens only if you allocate more than 2GB of shared objects and would require
 136       // millions of shared classes.
 137       vm_exit_during_initialization("Out of memory in the CDS archive",
 138                                     "Please reduce the number of shared classes.");
 139     }
 140   }
 141 
 142   MetaspaceShared::commit_to(_rs, _vs, newtop);
 143   _top = newtop;
 144   return _top;
 145 }

 147 char* DumpRegion::allocate(size_t num_bytes, size_t alignment) {
 148   char* p = (char*)align_up(_top, alignment);
 149   char* newtop = p + align_up(num_bytes, alignment);
 150   expand_top_to(newtop);
 151   memset(p, 0, newtop - p);
 152   return p;
 153 }

 155 void DumpRegion::append_intptr_t(intptr_t n, bool need_to_mark) {
 156   assert(is_aligned(_top, sizeof(intptr_t)), "bad alignment");
 157   intptr_t *p = (intptr_t*)_top;
 158   char* newtop = _top + sizeof(intptr_t);
 159   expand_top_to(newtop);
 160   *p = n;
 161   if (need_to_mark) {
 162     ArchivePtrMarker::mark_pointer(p);
 163   }
 164 }

 180 void DumpRegion::init(ReservedSpace* rs, VirtualSpace* vs) {
 181   _rs = rs;
 182   _vs = vs;
 183   // Start with 0 committed bytes. The memory will be committed as needed by
 184   // MetaspaceShared::commit_to().
 185   if (!_vs->initialize(*_rs, 0)) {
 186     fatal("Unable to allocate memory for shared space");
 187   }
 188   _base = _top = _rs->base();
 189   _end = _rs->end();
 190 }

 192 void DumpRegion::pack(DumpRegion* next) {
 193   assert(!is_packed(), "sanity");
 194   _end = (char*)align_up(_top, Metaspace::reserve_alignment());
 195   _is_packed = true;
 196   if (next != NULL) {
 197     next->_rs = _rs;
 198     next->_vs = _vs;
 199     next->_base = next->_top = this->_end;
 200     next->_end = _rs->end();
 201   }
 202 }

 240 // When reserving an address range using ReservedSpace, we need an alignment that satisfies both:
 241 // os::vm_allocation_granularity() -- so that we can sub-divide this range into multiple mmap regions,
 242 //                                    while keeping the first range at offset 0 of this range.
 243 // Metaspace::reserve_alignment()  -- so we can pass the region to
 244 //                                    Metaspace::allocate_metaspace_compressed_klass_ptrs.
 245 size_t MetaspaceShared::reserved_space_alignment() {
 246   size_t os_align = os::vm_allocation_granularity();
 247   size_t ms_align = Metaspace::reserve_alignment();
 248   if (os_align >= ms_align) {
 249     assert(os_align % ms_align == 0, "must be a multiple");
 250     return os_align;
 251   } else {
 252     assert(ms_align % os_align == 0, "must be a multiple");
 253     return ms_align;
 254   }
 255 }

 262 void MetaspaceShared::initialize_dumptime_shared_and_meta_spaces() {
 263   assert(DumpSharedSpaces, "should be called for dump time only");
 264   const size_t reserve_alignment = reserved_space_alignment();





 265   char* shared_base = (char*)align_up((char*)SharedBaseAddress, reserve_alignment);
 266 
 267 #ifdef _LP64
 268   // On 64-bit VM, the heap and class space layout will be the same as if
 269   // you're running in -Xshare:on mode:
 270   //
 271   //                              +-- SharedBaseAddress (default = 0x800000000)
 272   //                              v
 273   // +-..---------+---------+ ... +----+----+----+--------------------+
 274   // |    Heap    | Archive |     | MC | RW | RO |    class space     |
 275   // +-..---------+---------+ ... +----+----+----+--------------------+
 276   // |<--   MaxHeapSize  -->|     |<-- UnscaledClassSpaceMax = 4GB -->|
 277   //
 278   const uint64_t UnscaledClassSpaceMax = (uint64_t(max_juint) + 1);
 279   const size_t cds_total = align_down(UnscaledClassSpaceMax, reserve_alignment);
 280 #else
 281   // We don't support archives larger than 256MB on 32-bit due to limited virtual address space.

 282   size_t cds_total = align_down(256*M, reserve_alignment);
 283 #endif
 284 

 285   bool use_requested_base = true;





 286   if (ArchiveRelocationMode == 1) {
 287     log_info(cds)("ArchiveRelocationMode == 1: always allocate class space at an alternative address");
 288     use_requested_base = false;
 289   }
 290 
 291   // First try to reserve the space at the specified SharedBaseAddress.
 292   assert(!_shared_rs.is_reserved(), "must be");
 293   if (use_requested_base) {
 294     _shared_rs = reserve_shared_space(cds_total, shared_base);







 295   }
 296   if (_shared_rs.is_reserved()) {
 297     assert(shared_base == 0 || _shared_rs.base() == shared_base, "should match");
 298   } else {
 299     // Get a mmap region anywhere if the SharedBaseAddress fails.
 300     _shared_rs = reserve_shared_space(cds_total);









 301   }

 302   if (!_shared_rs.is_reserved()) {
 303     vm_exit_during_initialization("Unable to reserve memory for shared space",
 304                                   err_msg(SIZE_FORMAT " bytes.", cds_total));
 305   }
 306 
 307 #ifdef _LP64
 308   // During dump time, we allocate 4GB (UnscaledClassSpaceMax) of space and split it up:
 309   // + The upper 1 GB is used as the "temporary compressed class space" -- preload_classes()
 310   //   will store Klasses into this space.
 311   // + The lower 3 GB is used for the archive -- when preload_classes() is done,
 312   //   ArchiveCompactor will copy the class metadata into this space, first the RW parts,
 313   //   then the RO parts.
 314 
 315   size_t max_archive_size = align_down(cds_total * 3 / 4, reserve_alignment);
 316   ReservedSpace tmp_class_space = _shared_rs.last_part(max_archive_size);
 317   CompressedClassSpaceSize = align_down(tmp_class_space.size(), reserve_alignment);
 318   _shared_rs = _shared_rs.first_part(max_archive_size);
 319 
 320   if (UseCompressedClassPointers) {
 321     // Set up compress class pointers.
 322     CompressedKlassPointers::set_base((address)_shared_rs.base());
 323     // Set narrow_klass_shift to be LogKlassAlignmentInBytes. This is consistent
 324     // with AOT.
 325     CompressedKlassPointers::set_shift(LogKlassAlignmentInBytes);
 326     // Set the range of klass addresses to 4GB.
 327     CompressedKlassPointers::set_range(cds_total);









































 328     Metaspace::initialize_class_space(tmp_class_space);














 329   }
 330   log_info(cds)("narrow_klass_base = " PTR_FORMAT ", narrow_klass_shift = %d",
 331                 p2i(CompressedKlassPointers::base()), CompressedKlassPointers::shift());
 332 
 333   log_info(cds)("Allocated temporary class space: " SIZE_FORMAT " bytes at " PTR_FORMAT,
 334                 CompressedClassSpaceSize, p2i(tmp_class_space.base()));
 335 #endif
 336 
 337   init_shared_dump_space(&_mc_region);
 338   SharedBaseAddress = (size_t)_shared_rs.base();
 339   log_info(cds)("Allocated shared space: " SIZE_FORMAT " bytes at " PTR_FORMAT,
 340                 _shared_rs.size(), p2i(_shared_rs.base()));
 341 
 342   size_t symbol_rs_size = LP64_ONLY(3 * G) NOT_LP64(128 * M);
 343   _symbol_rs = ReservedSpace(symbol_rs_size);
 344   if (!_symbol_rs.is_reserved()) {
 345     vm_exit_during_initialization("Unable to reserve memory for symbols",
 346                                   err_msg(SIZE_FORMAT " bytes.", symbol_rs_size));
 347   }
 348   _symbol_region.init(&_symbol_rs, &_symbol_vs);
 349 }

 371 static GrowableArray<Handle>* _extra_interned_strings = NULL;
 372 
 373 void MetaspaceShared::read_extra_data(const char* filename, TRAPS) {
 374   _extra_interned_strings = new (ResourceObj::C_HEAP, mtInternal)GrowableArray<Handle>(10000, true);
 375 
 376   HashtableTextDump reader(filename);
 377   reader.check_version("VERSION: 1.0");
 378 
 379   while (reader.remain() > 0) {
 380     int utf8_length;
 381     int prefix_type = reader.scan_prefix(&utf8_length);
 382     ResourceMark rm(THREAD);
 383     if (utf8_length == 0x7fffffff) {
 384       // buf_len will overflown 32-bit value.
 385       vm_exit_during_initialization(err_msg("string length too large: %d", utf8_length));
 386     }
 387     int buf_len = utf8_length+1;
 388     char* utf8_buffer = NEW_RESOURCE_ARRAY(char, buf_len);
 389     reader.get_utf8(utf8_buffer, utf8_length);
 390     utf8_buffer[utf8_length] = '\0';
 391 
 392     if (prefix_type == HashtableTextDump::SymbolPrefix) {
 393       SymbolTable::new_permanent_symbol(utf8_buffer);
 394     } else{
 395       assert(prefix_type == HashtableTextDump::StringPrefix, "Sanity");
 396       oop s = StringTable::intern(utf8_buffer, THREAD);
 397 
 398       if (HAS_PENDING_EXCEPTION) {
 399         log_warning(cds, heap)("[line %d] extra interned string allocation failed; size too large: %d",
 400                                reader.last_line_no(), utf8_length);
 401         CLEAR_PENDING_EXCEPTION;
 402       } else {
 403 #if INCLUDE_G1GC
 404         if (UseG1GC) {
 405           typeArrayOop body = java_lang_String::value(s);
 406           const HeapRegion* hr = G1CollectedHeap::heap()->heap_region_containing(body);
 407           if (hr->is_humongous()) {
 408             // Don't keep it alive, so it will be GC'ed before we dump the strings, in order
 409             // to maximize free heap space and minimize fragmentation.
 410             log_warning(cds, heap)("[line %d] extra interned string ignored; size too large: %d",
 411                                 reader.last_line_no(), utf8_length);
 412             continue;
 413           }
 414         }
 415 #endif
 416         // Interned strings are GC'ed if there are no references to it, so let's
 417         // add a reference to keep this string alive.
 418         assert(s != NULL, "must succeed");
 419         Handle h(THREAD, s);
 420         _extra_interned_strings->append(h);
 421       }
 422     }
 423   }
 424 }

 426 void MetaspaceShared::commit_to(ReservedSpace* rs, VirtualSpace* vs, char* newtop) {
 427   Arguments::assert_is_dumping_archive();
 428   char* base = rs->base();
 429   size_t need_committed_size = newtop - base;
 430   size_t has_committed_size = vs->committed_size();
 431   if (need_committed_size < has_committed_size) {
 432     return;
 433   }
 434 
 435   size_t min_bytes = need_committed_size - has_committed_size;
 436   size_t preferred_bytes = 1 * M;
 437   size_t uncommitted = vs->reserved_size() - has_committed_size;
 438 
 439   size_t commit =MAX2(min_bytes, preferred_bytes);
 440   commit = MIN2(commit, uncommitted);
 441   assert(commit <= uncommitted, "sanity");
 442 
 443   bool result = vs->expand_by(commit, false);
 444   if (rs == &_shared_rs) {
 445     ArchivePtrMarker::expand_ptr_end((address*)vs->high());
 446   }
 447 
 448   if (!result) {
 449     vm_exit_during_initialization(err_msg("Failed to expand shared space to " SIZE_FORMAT " bytes",
 450                                           need_committed_size));
 451   }
 452 
 453   assert(rs == &_shared_rs || rs == &_symbol_rs, "must be");
 454   const char* which = (rs == &_shared_rs) ? "shared" : "symbol";
 455   log_debug(cds)("Expanding %s spaces by " SIZE_FORMAT_W(7) " bytes [total " SIZE_FORMAT_W(9)  " bytes ending at %p]",
 456                  which, commit, vs->actual_committed_size(), vs->high());
 457 }

 478   soc->do_tag(sizeof(Symbol));
 479 
 480   // Dump/restore miscellaneous metadata.
 481   JavaClasses::serialize_offsets(soc);
 482   Universe::serialize(soc);
 483   soc->do_tag(--tag);
 484 
 485   // Dump/restore references to commonly used names and signatures.
 486   vmSymbols::serialize(soc);
 487   soc->do_tag(--tag);
 488 
 489   // Dump/restore the symbol/string/subgraph_info tables
 490   SymbolTable::serialize_shared_table_header(soc);

 492   HeapShared::serialize_subgraph_info_table_header(soc);
 493   SystemDictionaryShared::serialize_dictionary_headers(soc);
 494 
 495   InstanceMirrorKlass::serialize_offsets(soc);
 496 
 497   // Dump/restore well known classes (pointers)
 498   SystemDictionaryShared::serialize_well_known_klasses(soc);
 499   soc->do_tag(--tag);
 500 
 501   serialize_cloned_cpp_vtptrs(soc);
 502   soc->do_tag(--tag);
 503 
 504   soc->do_tag(666);
 505 }

 523 uintx MetaspaceShared::object_delta_uintx(void* obj) {
 524   Arguments::assert_is_dumping_archive();
 525   if (DumpSharedSpaces) {
 526     assert(shared_rs()->contains(obj), "must be");
 527   } else {
 528     assert(is_in_shared_metaspace(obj) || DynamicArchive::is_in_target_space(obj), "must be");
 529   }
 530   address base_address = address(SharedBaseAddress);
 531   uintx deltax = address(obj) - base_address;
 532   return deltax;
 533 }

 615 static void rewrite_nofast_bytecode(const methodHandle& method) {
 616   BytecodeStream bcs(method);
 617   while (!bcs.is_last_bytecode()) {
 618     Bytecodes::Code opcode = bcs.next();
 619     switch (opcode) {
 620     case Bytecodes::_getfield:      *bcs.bcp() = Bytecodes::_nofast_getfield;      break;
 621     case Bytecodes::_putfield:      *bcs.bcp() = Bytecodes::_nofast_putfield;      break;
 622     case Bytecodes::_aload_0:       *bcs.bcp() = Bytecodes::_nofast_aload_0;       break;
 623     case Bytecodes::_iload: {
 624       if (!bcs.is_wide()) {
 625         *bcs.bcp() = Bytecodes::_nofast_iload;
 626       }
 627       break;
 628     }
 629     default: break;
 630     }
 631   }
 632 }

 659 // Objects of the Metadata types (such as Klass and ConstantPool) have C++ vtables.
 660 // (In GCC this is the field <Type>::_vptr, i.e., first word in the object.)
 661 //
 662 // Addresses of the vtables and the methods may be different across JVM runs,
 663 // if libjvm.so is dynamically loaded at a different base address.
 664 //
 665 // To ensure that the Metadata objects in the CDS archive always have the correct vtable:
 666 //
 667 // + at dump time:  we redirect the _vptr to point to our own vtables inside
 668 //                  the CDS image
 669 // + at run time:   we clone the actual contents of the vtables from libjvm.so
 670 //                  into our own tables.
 671 
 672 // Currently, the archive contain ONLY the following types of objects that have C++ vtables.
 673 #define CPP_VTABLE_PATCH_TYPES_DO(f) \
 674   f(ConstantPool) \
 675   f(InstanceKlass) \
 676   f(InstanceClassLoaderKlass) \
 677   f(InstanceMirrorKlass) \
 678   f(InstanceRefKlass) \
 679   f(Method) \
 680   f(ObjArrayKlass) \
 681   f(TypeArrayKlass)

 683 class CppVtableInfo {
 684   intptr_t _vtable_size;
 685   intptr_t _cloned_vtable[1];
 686 public:
 687   static int num_slots(int vtable_size) {
 688     return 1 + vtable_size; // Need to add the space occupied by _vtable_size;
 689   }
 690   int vtable_size()           { return int(uintx(_vtable_size)); }
 691   void set_vtable_size(int n) { _vtable_size = intptr_t(n); }
 692   intptr_t* cloned_vtable()   { return &_cloned_vtable[0]; }
 693   void zero()                 { memset(_cloned_vtable, 0, sizeof(intptr_t) * vtable_size()); }
 694   // Returns the address of the next CppVtableInfo that can be placed immediately after this CppVtableInfo
 695   static size_t byte_size(int vtable_size) {
 696     CppVtableInfo i;
 697     return pointer_delta(&i._cloned_vtable[vtable_size], &i, sizeof(u1));
 698   }
 699 };

 742 template <class T>
 743 intptr_t* CppVtableCloner<T>::clone_vtable(const char* name, CppVtableInfo* info) {
 744   if (!DumpSharedSpaces) {
 745     assert(_info == 0, "_info is initialized only at dump time");
 746     _info = info; // Remember it -- it will be used by MetaspaceShared::is_valid_shared_method()
 747   }
 748   T tmp; // Allocate temporary dummy metadata object to get to the original vtable.
 749   int n = info->vtable_size();
 750   intptr_t* srcvtable = vtable_of(tmp);
 751   intptr_t* dstvtable = info->cloned_vtable();
 752 
 753   // We already checked (and, if necessary, adjusted n) when the vtables were allocated, so we are
 754   // safe to do memcpy.
 755   log_debug(cds, vtables)("Copying %3d vtable entries for %s", n, name);
 756   memcpy(dstvtable, srcvtable, sizeof(intptr_t) * n);
 757   return dstvtable + n;
 758 }

 760 // To determine the size of the vtable for each type, we use the following
 761 // trick by declaring 2 subclasses:
 762 //
 763 //   class CppVtableTesterA: public InstanceKlass {virtual int   last_virtual_method() {return 1;}    };
 764 //   class CppVtableTesterB: public InstanceKlass {virtual void* last_virtual_method() {return NULL}; };
 765 //
 766 // CppVtableTesterA and CppVtableTesterB's vtables have the following properties:
 767 // - Their size (N+1) is exactly one more than the size of InstanceKlass's vtable (N)
 768 // - The first N entries have are exactly the same as in InstanceKlass's vtable.
 769 // - Their last entry is different.
 770 //
 771 // So to determine the value of N, we just walk CppVtableTesterA and CppVtableTesterB's tables
 772 // and find the first entry that's different.
 773 //
 774 // This works on all C++ compilers supported by Oracle, but you may need to tweak it for more
 775 // esoteric compilers.
 776 
 777 template <class T> class CppVtableTesterB: public T {
 778 public:
 779   virtual int last_virtual_method() {return 1;}
 780 };
 781 
 782 template <class T> class CppVtableTesterA : public T {
 783 public:
 784   virtual void* last_virtual_method() {
 785     // Make this different than CppVtableTesterB::last_virtual_method so the C++
 786     // compiler/linker won't alias the two functions.
 787     return NULL;
 788   }
 789 };
 790 
 791 template <class T>
 792 int CppVtableCloner<T>::get_vtable_length(const char* name) {
 793   CppVtableTesterA<T> a;
 794   CppVtableTesterB<T> b;
 795 
 796   intptr_t* avtable = vtable_of(a);
 797   intptr_t* bvtable = vtable_of(b);
 798 
 799   // Start at slot 1, because slot 0 may be RTTI (on Solaris/Sparc)
 800   int vtable_len = 1;
 801   for (; ; vtable_len++) {
 802     if (avtable[vtable_len] != bvtable[vtable_len]) {
 803       break;
 804     }
 805   }
 806   log_debug(cds, vtables)("Found   %3d vtable entries for %s", vtable_len, name);
 807 
 808   return vtable_len;
 809 }

1407   static void iterate_roots(MetaspaceClosure* it) {
1408     // To ensure deterministic contents in the archive, we just need to ensure that
1409     // we iterate the MetsapceObjs in a deterministic order. It doesn't matter where
1410     // the MetsapceObjs are located originally, as they are copied sequentially into
1411     // the archive during the iteration.
1412     //
1413     // The only issue here is that the symbol table and the system directories may be
1414     // randomly ordered, so we copy the symbols and klasses into two arrays and sort
1415     // them deterministically.
1416     //
1417     // During -Xshare:dump, the order of Symbol creation is strictly determined by
1418     // the SharedClassListFile (class loading is done in a single thread and the JIT
1419     // is disabled). Also, Symbols are allocated in monotonically increasing addresses
1420     // (see Symbol::operator new(size_t, int)). So if we iterate the Symbols by
1421     // ascending address order, we ensure that all Symbols are copied into deterministic
1422     // locations in the archive.
1423     GrowableArray<Symbol*>* symbols = _ssc->get_sorted_symbols();
1424     for (int i=0; i<symbols->length(); i++) {
1425       it->push(symbols->adr_at(i));
1426     }
1427     if (_global_klass_objects != NULL) {
1428       // Need to fix up the pointers
1429       for (int i = 0; i < _global_klass_objects->length(); i++) {
1430         // NOTE -- this requires that the vtable is NOT yet patched, or else we are hosed.
1431         it->push(_global_klass_objects->adr_at(i));
1432       }
1433     }
1434     FileMapInfo::metaspace_pointers_do(it, false);
1435     SystemDictionaryShared::dumptime_classes_do(it);
1436     Universe::metaspace_pointers_do(it);
1437     SymbolTable::metaspace_pointers_do(it);
1438     vmSymbols::metaspace_pointers_do(it);
1439 
1440     it->finish();

1510 void VM_PopulateDumpSharedSpace::relocate_to_default_base_address(CHeapBitMap* ptrmap) {
1511   intx addr_delta = MetaspaceShared::final_delta();
1512   if (addr_delta == 0) {
1513     ArchivePtrMarker::compact((address)SharedBaseAddress, (address)_ro_region.top());
1514   } else {
1515     // We are not able to reserve space at Arguments::default_SharedBaseAddress() (due to ASLR).
1516     // This means that the current content of the archive is based on a random
1517     // address. Let's relocate all the pointers, so that it can be mapped to
1518     // Arguments::default_SharedBaseAddress() without runtime relocation.
1519     //
1520     // Note: both the base and dynamic archive are written with
1521     // FileMapHeader::_shared_base_address == Arguments::default_SharedBaseAddress()
1522 
1523     // Patch all pointers that are marked by ptrmap within this region,
1524     // where we have just dumped all the metaspace data.
1525     address patch_base = (address)SharedBaseAddress;
1526     address patch_end  = (address)_ro_region.top();
1527     size_t size = patch_end - patch_base;
1528 
1529     // the current value of the pointers to be patched must be within this
1530     // range (i.e., must point to valid metaspace objects)
1531     address valid_old_base = patch_base;
1532     address valid_old_end  = patch_end;
1533 
1534     // after patching, the pointers must point inside this range
1535     // (the requested location of the archive, as mapped at runtime).
1536     address valid_new_base = (address)Arguments::default_SharedBaseAddress();
1537     address valid_new_end  = valid_new_base + size;
1538 
1539     log_debug(cds)("Relocating archive from [" INTPTR_FORMAT " - " INTPTR_FORMAT " ] to "
1540                    "[" INTPTR_FORMAT " - " INTPTR_FORMAT " ]", p2i(patch_base), p2i(patch_end),
1541                    p2i(valid_new_base), p2i(valid_new_end));
1542 
1543     SharedDataRelocator<true> patcher((address*)patch_base, (address*)patch_end, valid_old_base, valid_old_end,
1544                                       valid_new_base, valid_new_end, addr_delta, ptrmap);
1545     ptrmap->iterate(&patcher);
1546     ArchivePtrMarker::compact(patcher.max_non_null_offset());
1547   }
1548 }

1550 void VM_PopulateDumpSharedSpace::doit() {
1551   CHeapBitMap ptrmap;
1552   MetaspaceShared::initialize_ptr_marker(&ptrmap);
1553 
1554   // We should no longer allocate anything from the metaspace, so that:
1555   //
1556   // (1) Metaspace::allocate might trigger GC if we have run out of
1557   //     committed metaspace, but we can't GC because we're running
1558   //     in the VM thread.
1559   // (2) ArchiveCompactor needs to work with a stable set of MetaspaceObjs.
1560   Metaspace::freeze();
1561   DEBUG_ONLY(SystemDictionaryShared::NoClassLoadingMark nclm);
1562 
1563   Thread* THREAD = VMThread::vm_thread();
1564 
1565   FileMapInfo::check_nonempty_dir_in_shared_path_table();
1566 
1567   NOT_PRODUCT(SystemDictionary::verify();)
1568   // The following guarantee is meant to ensure that no loader constraints
1569   // exist yet, since the constraints table is not shared.  This becomes
1570   // more important now that we don't re-initialize vtables/itables for
1571   // shared classes at runtime, where constraints were previously created.
1572   guarantee(SystemDictionary::constraints()->number_of_entries() == 0,
1573             "loader constraints are not saved");
1574   guarantee(SystemDictionary::placeholders()->number_of_entries() == 0,
1575           "placeholders are not saved");
1576 
1577   // At this point, many classes have been loaded.
1578   // Gather systemDictionary classes in a global array and do everything to
1579   // that so we don't have to walk the SystemDictionary again.
1580   SystemDictionaryShared::check_excluded_classes();
1581   _global_klass_objects = new GrowableArray<Klass*>(1000);
1582   CollectClassesClosure collect_classes;
1583   ClassLoaderDataGraph::loaded_classes_do(&collect_classes);
1584   _global_klass_objects->sort(global_klass_compare);
1585 
1586   print_class_stats();
1587 
1588   // Ensure the ConstMethods won't be modified at run-time
1589   log_info(cds)("Updating ConstMethods ... ");
1590   rewrite_nofast_bytecodes_and_calculate_fingerprints(THREAD);
1591   log_info(cds)("done. ");
1592 
1593   // Remove all references outside the metadata
1594   log_info(cds)("Removing unshareable information ... ");
1595   remove_unshareable_in_classes();
1596   log_info(cds)("done. ");
1597 
1598   MetaspaceShared::allocate_cloned_cpp_vtptrs();
1599   char* cloned_vtables = _mc_region.top();
1600   MetaspaceShared::allocate_cpp_vtable_clones();
1601 
1602   ArchiveCompactor::initialize();
1603   ArchiveCompactor::copy_and_compact();
1604 
1605   dump_symbols();
1606 
1607   // Dump supported java heap objects
1608   _closed_archive_heap_regions = NULL;
1609   _open_archive_heap_regions = NULL;
1610   dump_java_heap_objects();
1611 
1612   ArchiveCompactor::relocate_well_known_klasses();
1613 
1614   char* serialized_data = dump_read_only_tables();
1615   _ro_region.pack();
1616 
1617   // The vtable clones contain addresses of the current process.
1618   // We don't want to write these addresses into the archive. Same for i2i buffer.
1619   MetaspaceShared::zero_cpp_vtable_clones_for_writing();
1620   memset(MetaspaceShared::i2i_entry_code_buffers(), 0,
1621          MetaspaceShared::i2i_entry_code_buffers_size());
1622 
1623   // relocate the data so that it can be mapped to Arguments::default_SharedBaseAddress()
1624   // without runtime relocation.
1625   relocate_to_default_base_address(&ptrmap);
1626 
1627   // Create and write the archive file that maps the shared spaces.
1628 
1629   FileMapInfo* mapinfo = new FileMapInfo(true);
1630   mapinfo->populate_header(os::vm_allocation_granularity());
1631   mapinfo->set_serialized_data(serialized_data);
1632   mapinfo->set_cloned_vtables(cloned_vtables);
1633   mapinfo->set_i2i_entry_code_buffers(MetaspaceShared::i2i_entry_code_buffers(),
1634                                       MetaspaceShared::i2i_entry_code_buffers_size());
1635   mapinfo->open_for_write();
1636   MetaspaceShared::write_core_archive_regions(mapinfo, _closed_archive_heap_oopmaps, _open_archive_heap_oopmaps);
1637   _total_closed_archive_region_size = mapinfo->write_archive_heap_regions(
1638                                         _closed_archive_heap_regions,
1639                                         _closed_archive_heap_oopmaps,
1640                                         MetaspaceShared::first_closed_archive_heap_region,
1641                                         MetaspaceShared::max_closed_archive_heap_region);
1642   _total_open_archive_region_size = mapinfo->write_archive_heap_regions(
1643                                         _open_archive_heap_regions,
1644                                         _open_archive_heap_oopmaps,
1645                                         MetaspaceShared::first_open_archive_heap_region,
1646                                         MetaspaceShared::max_open_archive_heap_region);
1647 
1648   mapinfo->set_final_requested_base((char*)Arguments::default_SharedBaseAddress());

1817         char *end = strrchr(class_list_path_str, *os::file_separator());

1823           if (class_list_path_len < JVM_MAXPATHLEN - 4) {

1828           }
1829         }
1830       }

1832         jio_snprintf(class_list_path_str + class_list_path_len,
1833                      sizeof(class_list_path_str) - class_list_path_len,

1882 int MetaspaceShared::preload_classes(const char* class_list_path, TRAPS) {
1883   ClassListParser parser(class_list_path);
1884   int class_count = 0;
1885 
1886   while (parser.parse_one_line()) {
1887     Klass* klass = parser.load_current_class(THREAD);
1888     if (HAS_PENDING_EXCEPTION) {
1889       if (klass == NULL &&
1890           (PENDING_EXCEPTION->klass()->name() == vmSymbols::java_lang_ClassNotFoundException())) {
1891         // print a warning only when the pending exception is class not found
1892         log_warning(cds)("Preload Warning: Cannot find %s", parser.current_class_name());
1893       }
1894       CLEAR_PENDING_EXCEPTION;
1895     }
1896     if (klass != NULL) {
1897       if (log_is_enabled(Trace, cds)) {
1898         ResourceMark rm(THREAD);
1899         log_trace(cds)("Shared spaces preloaded: %s", klass->external_name());
1900       }
1901 
1902       if (klass->is_instance_klass()) {
1903         InstanceKlass* ik = InstanceKlass::cast(klass);
1904 
1905         // Link the class to cause the bytecodes to be rewritten and the
1906         // cpcache to be created. The linking is done as soon as classes
1907         // are loaded in order that the related data structures (klass and
1908         // cpCache) are located together.
1909         try_link_class(ik, THREAD);
1910         guarantee(!HAS_PENDING_EXCEPTION, "exception in link_class");
1911       }
1912 
1913       class_count++;
1914     }
1915   }
1916 
1917   return class_count;

1920 // Returns true if the class's status has changed
1921 bool MetaspaceShared::try_link_class(InstanceKlass* ik, TRAPS) {
1922   Arguments::assert_is_dumping_archive();
1923   if (ik->init_state() < InstanceKlass::linked &&
1924       !SystemDictionaryShared::has_class_failed_verification(ik)) {
1925     bool saved = BytecodeVerificationLocal;
1926     if (ik->is_shared_unregistered_class() && ik->class_loader() == NULL) {
1927       // The verification decision is based on BytecodeVerificationRemote
1928       // for non-system classes. Since we are using the NULL classloader
1929       // to load non-system classes for customized class loaders during dumping,
1930       // we need to temporarily change BytecodeVerificationLocal to be the same as
1931       // BytecodeVerificationRemote. Note this can cause the parent system
1932       // classes also being verified. The extra overhead is acceptable during
1933       // dumping.
1934       BytecodeVerificationLocal = BytecodeVerificationRemote;
1935     }
1936     ik->link_class(THREAD);
1937     if (HAS_PENDING_EXCEPTION) {
1938       ResourceMark rm(THREAD);
1939       log_warning(cds)("Preload Warning: Verification failed for %s",
1940                     ik->external_name());
1941       CLEAR_PENDING_EXCEPTION;
1942       SystemDictionaryShared::set_class_has_failed_verification(ik);
1943       _has_error_classes = true;
1944     }
1945     BytecodeVerificationLocal = saved;
1946     return true;
1947   } else {
1948     return false;
1949   }
1950 }

2145 // use_requested_addr:
2146 //  true  = map at FileMapHeader::_requested_base_address
2147 //  false = map at an alternative address picked by OS.
2148 MapArchiveResult MetaspaceShared::map_archives(FileMapInfo* static_mapinfo, FileMapInfo* dynamic_mapinfo,
2149                                                bool use_requested_addr) {
2150   PRODUCT_ONLY(if (ArchiveRelocationMode == 1 && use_requested_addr) {
2151       // For product build only -- this is for benchmarking the cost of doing relocation.
2152       // For debug builds, the check is done in FileMapInfo::map_regions for better test coverage.

2153       log_info(cds)("ArchiveRelocationMode == 1: always map archive(s) at an alternative address");
2154       return MAP_ARCHIVE_MMAP_FAILURE;
2155     });
2156 
2157   if (ArchiveRelocationMode == 2 && !use_requested_addr) {
2158     log_info(cds)("ArchiveRelocationMode == 2: never map archive(s) at an alternative address");
2159     return MAP_ARCHIVE_MMAP_FAILURE;
2160   };
2161 
2162   if (dynamic_mapinfo != NULL) {
2163     // Ensure that the OS won't be able to allocate new memory spaces between the two
2164     // archives, or else it would mess up the simple comparision in MetaspaceObj::is_shared().
2165     assert(static_mapinfo->mapping_end_offset() == dynamic_mapinfo->mapping_base_offset(), "no gap");
2166   }
2167

2244 char* MetaspaceShared::reserve_address_space_for_archives(FileMapInfo* static_mapinfo,
2245                                                           FileMapInfo* dynamic_mapinfo,
2246                                                           bool use_requested_addr,
2247                                                           ReservedSpace& main_rs,
2248                                                           ReservedSpace& archive_space_rs,
2249                                                           ReservedSpace& class_space_rs) {
2250   const bool use_klass_space = NOT_LP64(false) LP64_ONLY(Metaspace::using_class_space());
2251   const size_t class_space_size = NOT_LP64(0) LP64_ONLY(Metaspace::compressed_class_space_size());
2252 
2253   if (use_klass_space) {
2254     assert(class_space_size > 0, "CompressedClassSpaceSize must have been validated");
2255   }
2256   if (use_requested_addr && !is_aligned(static_mapinfo->requested_base_address(), reserved_space_alignment())) {
2257     return NULL;
2258   }
2259 
2260   // Size and requested location of the archive_space_rs (for both static and dynamic archives)
2261   size_t base_offset = static_mapinfo->mapping_base_offset();
2262   size_t end_offset  = (dynamic_mapinfo == NULL) ? static_mapinfo->mapping_end_offset() : dynamic_mapinfo->mapping_end_offset();
2263   assert(base_offset == 0, "must be");
2264   assert(is_aligned(end_offset,  os::vm_allocation_granularity()), "must be");
2265   assert(is_aligned(base_offset, os::vm_allocation_granularity()), "must be");
2266 
2267   // In case reserved_space_alignment() != os::vm_allocation_granularity()
2268   assert((size_t)os::vm_allocation_granularity() <= reserved_space_alignment(), "must be");
2269   end_offset = align_up(end_offset, reserved_space_alignment());
2270 
2271   size_t archive_space_size = end_offset - base_offset;
2272 
2273   // Special handling for Windows because it cannot mmap into a reserved space:
2274   //    use_requested_addr: We just map each region individually, and give up if any one of them fails.
2275   //   !use_requested_addr: We reserve the space first, and then os::read in all the regions (instead of mmap).
2276   //                        We're going to patch all the pointers anyway so there's no benefit for mmap.
2277 
2278   if (use_requested_addr) {
2279     char* archive_space_base = static_mapinfo->requested_base_address() + base_offset;
2280     char* archive_space_end  = archive_space_base + archive_space_size;
2281     if (!MetaspaceShared::use_windows_memory_mapping()) {
2282       archive_space_rs = reserve_shared_space(archive_space_size, archive_space_base);
2283       if (!archive_space_rs.is_reserved()) {
2284         return NULL;
2285       }
2286     }
2287     if (use_klass_space) {
2288       // Make sure we can map the klass space immediately following the archive_space space
2289       // Don't call reserve_shared_space here as that may try to enforce platform-specific
2290       // alignment rules which only apply to the archive base address
2291       char* class_space_base = archive_space_end;
2292       class_space_rs = ReservedSpace(class_space_size, reserved_space_alignment(),
2293                                      false /* large_pages */, class_space_base);
2294       if (!class_space_rs.is_reserved()) {
2295         return NULL;
2296       }
2297     }
2298     return static_mapinfo->requested_base_address();
2299   } else {
2300     if (use_klass_space) {
2301       main_rs = reserve_shared_space(archive_space_size + class_space_size);
2302       if (main_rs.is_reserved()) {
2303         archive_space_rs = main_rs.first_part(archive_space_size, reserved_space_alignment(), /*split=*/true);
2304         class_space_rs = main_rs.last_part(archive_space_size);
2305       }
2306     } else {
2307       main_rs = reserve_shared_space(archive_space_size);
2308       archive_space_rs = main_rs;
2309     }







2310     if (archive_space_rs.is_reserved()) {


2311       return archive_space_rs.base();
2312     } else {
2313       return NULL;
2314     }








































2315   }
