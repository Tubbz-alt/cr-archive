34 size_t EpsilonArguments::conservative_max_heap_alignment() {
35   return UseLargePages ? os::large_page_size() : os::vm_page_size();
36 }
37 
38 void EpsilonArguments::initialize() {
39   GCArguments::initialize();
40 
41   assert(UseEpsilonGC, "Sanity");
42 
43   // Forcefully exit when OOME is detected. Nothing we can do at that point.
44   if (FLAG_IS_DEFAULT(ExitOnOutOfMemoryError)) {
45     FLAG_SET_DEFAULT(ExitOnOutOfMemoryError, true);
46   }
47 












48   if (EpsilonMaxTLABSize < MinTLABSize) {
49     log_warning(gc)("EpsilonMaxTLABSize < MinTLABSize, adjusting it to " SIZE_FORMAT, MinTLABSize);
50     EpsilonMaxTLABSize = MinTLABSize;
51   }
52 
53   if (!EpsilonElasticTLAB && EpsilonElasticTLABDecay) {
54     log_warning(gc)("Disabling EpsilonElasticTLABDecay because EpsilonElasticTLAB is disabled");
55     FLAG_SET_DEFAULT(EpsilonElasticTLABDecay, false);
56   }
57 
58 #ifdef COMPILER2
59   // Enable loop strip mining: there are still non-GC safepoints, no need to make it worse
60   if (FLAG_IS_DEFAULT(UseCountedLoopSafepoints)) {
61     FLAG_SET_DEFAULT(UseCountedLoopSafepoints, true);
62     if (FLAG_IS_DEFAULT(LoopStripMiningIter)) {
63       FLAG_SET_DEFAULT(LoopStripMiningIter, 1000);
64     }
65   }
66 #endif
67 }
68 
69 void EpsilonArguments::initialize_alignments() {
70   size_t page_size = UseLargePages ? os::large_page_size() : os::vm_page_size();
71   size_t align = MAX2((size_t)os::vm_allocation_granularity(), page_size);
72   SpaceAlignment = align;
73   HeapAlignment  = align;
74 }
75 
76 CollectedHeap* EpsilonArguments::create_heap() {
77   return new EpsilonHeap();
78 }
