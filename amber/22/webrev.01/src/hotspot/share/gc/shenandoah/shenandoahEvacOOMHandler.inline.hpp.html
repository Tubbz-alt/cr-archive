25 #ifndef SHARE_GC_SHENANDOAH_SHENANDOAHEVACOOMHANDLER_INLINE_HPP
26 #define SHARE_GC_SHENANDOAH_SHENANDOAHEVACOOMHANDLER_INLINE_HPP
27 
28 #include "gc/shenandoah/shenandoahEvacOOMHandler.hpp"
29 #include "gc/shenandoah/shenandoahHeap.inline.hpp"
30 #include "gc/shenandoah/shenandoahThreadLocalData.hpp"
31 #include "runtime/atomic.hpp"
32 
33 void ShenandoahEvacOOMHandler::enter_evacuation(Thread* thr) {
34   jint threads_in_evac = Atomic::load_acquire(&_threads_in_evac);
35 
36   uint8_t level = ShenandoahThreadLocalData::push_evac_oom_scope(thr);
37   if ((threads_in_evac & OOM_MARKER_MASK) != 0) {
38     wait_for_no_evac_threads();
39     return;
40   }
41 
42   // Nesting case, this thread already registered
43   if (level != 0) {
44      return;
45   }
46   // Entering top level scope, register this thread.
47   register_thread(thr);
48 }
49 
50 void ShenandoahEvacOOMHandler::leave_evacuation(Thread* thr) {
51   uint8_t level = ShenandoahThreadLocalData::pop_evac_oom_scope(thr);
52   // Not top level, just return
53   if (level > 1) {
54     return;
55   }
56 
57   // Leaving top level scope, unregister this thread.
58   unregister_thread(thr);
59 }
60 
61 ShenandoahEvacOOMScope::ShenandoahEvacOOMScope() :
62   _thread(Thread::current()) {
63   ShenandoahHeap::heap()->enter_evacuation(_thread);
64 }
65 
66 ShenandoahEvacOOMScope::ShenandoahEvacOOMScope(Thread* t) :
67   _thread(t) {
68   ShenandoahHeap::heap()->enter_evacuation(_thread);
69 }
70 
71 ShenandoahEvacOOMScope::~ShenandoahEvacOOMScope() {
72   ShenandoahHeap::heap()->leave_evacuation(_thread);
73 }
74 
75 #endif // SHARE_GC_SHENANDOAH_SHENANDOAHEVACOOMHANDLER_INLINE_HPP
