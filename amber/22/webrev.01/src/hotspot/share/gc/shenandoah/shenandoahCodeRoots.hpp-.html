 69 class ShenandoahCodeRootsIterator {

 75 
 76 protected:
 77   ShenandoahCodeRootsIterator();

 83   template<bool CSET_FILTER>

 85 };

 99 class ShenandoahCodeRoots : public AllStatic {
100   friend class ShenandoahHeap;
101   friend class ShenandoahCodeRootsIterator;
102 
103 public:
104   static void initialize();
105   static void register_nmethod(nmethod* nm);
106   static void unregister_nmethod(nmethod* nm);
107   static void flush_nmethod(nmethod* nm);
108 
109   static ShenandoahNMethodTable* table() {
110     return _nmethod_table;
111   }
112 
113   // Concurrent nmethod unloading support
114   static void unlink(WorkGang* workers, bool unloading_occurred);
115   static void purge(WorkGang* workers);
116   static void arm_nmethods();
117   static void disarm_nmethods();
118   static int  disarmed_value()         { return _disarmed_value; }
119   static int* disarmed_value_address() { return &_disarmed_value; }
120 
121 private:
122   static ShenandoahNMethodTable* _nmethod_table;
123   static int                     _disarmed_value;
124 };
