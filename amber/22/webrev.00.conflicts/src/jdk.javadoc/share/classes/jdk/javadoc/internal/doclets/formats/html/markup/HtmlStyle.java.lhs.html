718     private final String cssName;
719 
720     HtmlStyle() {
721         cssName = Pattern.compile("\\p{Upper}")
722                 .matcher(toString())
723                 .replaceAll(mr -> "-" + mr.group().toLowerCase(Locale.US));
724     }
725 
726     HtmlStyle(String cssName) {
727         this.cssName = cssName;
728     }
729 
730     /**
731      * Returns the CSS class name associated with this style.
732      * @return the CSS class name
733      */
734     public String cssName() {
735         return cssName;
736     }
