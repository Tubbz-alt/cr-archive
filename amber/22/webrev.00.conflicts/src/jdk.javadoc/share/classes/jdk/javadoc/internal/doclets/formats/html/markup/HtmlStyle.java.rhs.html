726     private final String cssName;
727 
728     HtmlStyle() {
729         cssName = Pattern.compile("\\p{Upper}")
730                 .matcher(toString())
731                 .replaceAll(mr -> "-" + mr.group().toLowerCase(Locale.US));
732     }
733 
734     HtmlStyle(String cssName) {
735         this.cssName = cssName;
736     }
737 
738     /**
739      * Returns the CSS class name associated with this style.
740      * @return the CSS class name
741      */
742     public String cssName() {
743         return cssName;
744     }
