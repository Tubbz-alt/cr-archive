107     private static final Context.Key<Source> sourceKey = new Context.Key<>();

109     public static Source instance(Context context) {
110         Source instance = context.get(sourceKey);
111         if (instance == null) {
112             Options options = Options.instance(context);
113             String sourceString = options.get(SOURCE);
114             if (sourceString != null) instance = lookup(sourceString);
115             if (instance == null) instance = DEFAULT;
116             context.put(sourceKey, instance);
117         }
118         return instance;
119     }

220         enum DiagKind {
221             NORMAL,
222             PLURAL;
223         }
224 
225         private final Source minLevel;
226         private final Source maxLevel;
227         private final Fragment optFragment;
228         private final DiagKind optKind;
229 
230         Feature(Source minLevel) {
231             this(minLevel, null, null);
232         }
233 
234         Feature(Source minLevel, Fragment optFragment, DiagKind optKind) {
235             this(minLevel, MAX, optFragment, optKind);
236         }
237 
238         Feature(Source minLevel, Source maxLevel) {
239             this(minLevel, maxLevel, null, null);
240         }
241 
242         Feature(Source minLevel, Source maxLevel, Fragment optFragment, DiagKind optKind) {
243             this.minLevel = minLevel;
244             this.maxLevel = maxLevel;
245             this.optFragment = optFragment;
246             this.optKind = optKind;
247         }
248 
249         public boolean allowedInSource(Source source) {
250             return source.compareTo(minLevel) >= 0 &&
251                     source.compareTo(maxLevel) <= 0;
252         }
253 
254         public boolean isPlural() {
255             Assert.checkNonNull(optKind);
256             return optKind == DiagKind.PLURAL;
257         }
258 
259         public Fragment nameFragment() {
260             Assert.checkNonNull(optFragment);
261             return optFragment;
262         }
263 
264         public Fragment fragment(String sourceName) {
265             Assert.checkNonNull(optFragment);
266             return optKind == DiagKind.NORMAL ?
267                     Fragments.FeatureNotSupportedInSource(optFragment, sourceName, minLevel.name) :
268                     Fragments.FeatureNotSupportedInSourcePlural(optFragment, sourceName, minLevel.name);
269         }
270 
271         public Error error(String sourceName) {
272             Assert.checkNonNull(optFragment);
273             return optKind == DiagKind.NORMAL ?
274                     Errors.FeatureNotSupportedInSource(optFragment, sourceName, minLevel.name) :
275                     Errors.FeatureNotSupportedInSourcePlural(optFragment, sourceName, minLevel.name);
276         }
