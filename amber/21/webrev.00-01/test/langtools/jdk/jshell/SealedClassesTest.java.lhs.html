35 import jdk.jshell.TypeDeclSnippet;

38 import org.testng.annotations.BeforeMethod;
39 import org.testng.annotations.Test;
40 
41 import static jdk.jshell.Snippet.Status.VALID;
42 
43 @Test
44 public class SealedClassesTest extends KullaTesting {
45 
46     public void testSealed() {
47         TypeDeclSnippet base = classKey(
48                 assertEval("sealed class B permits I {}",
49                            ste(MAIN_SNIPPET, Status.NONEXISTENT, Status.RECOVERABLE_NOT_DEFINED, false, null)));
50         assertEval("final class I extends B {}",
51                    added(VALID),
52                    ste(base, Status.RECOVERABLE_NOT_DEFINED, Status.VALID, true, null));












53     }
54 
55     @BeforeMethod
56     public void setUp() {
57         setUp(b -> b.compilerOptions("--enable-preview", "-source", String.valueOf(SourceVersion.latest().ordinal()))
58                     .remoteVMOptions("--enable-preview"));
59     }
