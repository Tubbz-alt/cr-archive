40     ACCEPTED_BIN_DIFF="
41         ./lib/server/libjvm.so

45         ./hotspot/gtest/server/libjvm.so
46         "

61   "

64   SKIP_BIN_DIFF="true"

75       ./lib/libosxui.dylib
