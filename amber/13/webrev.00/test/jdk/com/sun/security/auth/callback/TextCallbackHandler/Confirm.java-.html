25  * @test
26  * @bug 6599079
27  * @summary Non-standard ConfirmationCallback throws NPE
28  */
29 
30 import javax.security.auth.callback.Callback;
31 import javax.security.auth.callback.ConfirmationCallback;
32 import com.sun.security.auth.callback.TextCallbackHandler;
33 import java.io.ByteArrayInputStream;
34 
35 public class Confirm {
36 
37     public static void main(String[] args) throws Exception {
38         // Provide answer in an individual stream so that the program
39         // does not block.
40         System.setIn(new ByteArrayInputStream("1\n".getBytes()));
41 
42         new TextCallbackHandler().handle(new Callback[]{
43             new ConfirmationCallback("Prince", ConfirmationCallback.INFORMATION,
44                     new String[]{"To be", "Not to be"}, 0)});
45     }
46 }
