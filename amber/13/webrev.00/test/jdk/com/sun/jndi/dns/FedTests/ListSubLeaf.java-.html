24 import javax.naming.Context;
25 import javax.naming.NamingEnumeration;
26 import javax.naming.directory.InitialDirContext;
27 
28 /*
29  * @test
30  * @bug 8210339
31  * @summary list of a leaf using a composite name with DNS name as
32  *          first component
33  * @library ../lib/
34  * @modules java.naming/com.sun.jndi.toolkit.dir
35  *          java.base/sun.security.util
36  * @build FedSubordinateNs FedObjectFactory
37  * @run main/othervm ListSubLeaf
38  */
39 
40 public class ListSubLeaf extends ListFedBase {
41 
42     private static final int COUNT_LIMIT = 0; // no entry
43 
44     public static void main(String[] args) throws Exception {
45         new ListSubLeaf().run(args);
46     }
47 
48     /*
49      * list of a leaf using a composite name with DNS name as
50      * first component
51      */
52     @Override
53     public void runTest() throws Exception {
54         env().put(Context.OBJECT_FACTORIES, "FedObjectFactory");
55         setContext(new InitialDirContext(env()));
56 
57         NamingEnumeration enumObj = context().listBindings(getKey() + "/a/b/c");
58         verifyNamingEnumeration(enumObj, COUNT_LIMIT);
59     }
60 }
