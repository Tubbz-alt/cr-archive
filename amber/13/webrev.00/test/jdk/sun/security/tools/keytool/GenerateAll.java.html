24 /*
25  * @test
26  * @bug 8242184
27  * @summary CRL generation error with RSASSA-PSS
28  * @library /test/lib
29  */
30 
31 import jdk.test.lib.SecurityTools;
32 import jdk.test.lib.process.OutputAnalyzer;
33 
34 public class GenerateAll {
35 
36     public static void main(String[] args) throws Throwable {
37 
38         kt("-genkeypair -alias ca -dname CN=CA -keyalg ec");
39 
40         String[] aliases = {
41                 "rsa", "dsa", "rrr", "rsassa-pss", "ec"};
42 
43         for (String alias : aliases) {
44             // "rrr": keyalg is rsa, sigalg is rsassa-pss
45             // otherwise: keyalg is alias, sigalg auto derived
46             String keyAlg = alias.equals("rrr") ? "rsa" : alias;
47             String extra = alias.equals("rrr") ? " -sigalg rsassa-pss" : "";
48 
49             // gen
50             kt("-genkeypair -alias " + alias + " -dname CN=" + alias
51                     + " -keyalg " + keyAlg + extra);
52 
53             // req
54             kt("-certreq -alias " + alias + " -file " + alias + ".req");
55             kt("-printcertreq -file " + alias + ".req");
56 
57             // gencert
58             kt("-gencert -alias ca -infile " + alias
59                     + ".req -outfile " + alias + ".crt");
60             kt("-printcert -file " + alias + ".crt");
61 
62             // crl
63             kt("-gencrl -alias " + alias + " -id 0 -file " + alias + ".crl");
64             kt("-printcrl -file " + alias + ".crl")
65                     .shouldContain("Verified by " + alias);
66         }
67     }
68 
69     static OutputAnalyzer kt(String arg) throws Exception {
70         return SecurityTools.keytool("-keystore ks -storepass changeit " + arg)
71                 .shouldHaveExitValue(0);
72     }
73 }
