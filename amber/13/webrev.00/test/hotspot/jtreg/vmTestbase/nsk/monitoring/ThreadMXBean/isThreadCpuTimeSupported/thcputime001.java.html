24 package nsk.monitoring.ThreadMXBean.isThreadCpuTimeSupported;
25 
26 import java.io.*;
27 import nsk.share.*;
28 import nsk.monitoring.share.*;
29 
30 public class thcputime001 {
31     public static void main(String[] argv) {
32         System.exit(Consts.JCK_STATUS_BASE + run(argv, System.out));
33     }
34 
35     public static int run(String[] argv, PrintStream out) {
36         ArgumentHandler argHandler = new ArgumentHandler(argv);
37         Log log = new Log(out, argHandler);
38         ThreadMonitor monitor = Monitor.getThreadMonitor(log, argHandler);
39 
40         // Check the method is... for the specified way of access to MBeans
41         boolean isSupported = monitor.isThreadCpuTimeSupported();
42         if (isSupported) {
43             log.display("Thread cpu time is supported.");
44         } else {
45             log.display("Thread cpu time is not supported.");
46         }
47         return 0;
48     }
49 }
