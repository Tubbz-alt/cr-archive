24 /**
25  * @test

26  * @bug 8031321
27  * @summary Verify that results of computations are the same w/
28  *          and w/o usage of BLSMSK instruction
29  * @library /test/lib /
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  *
33  * @build sun.hotspot.WhiteBox
34  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
35  *                                sun.hotspot.WhiteBox$WhiteBoxPermission
36  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
37  *                   -XX:+WhiteBoxAPI
38  *                   compiler.intrinsics.bmi.TestBlsmskL
39  */
40 
41 package compiler.intrinsics.bmi;
42 
43 import sun.hotspot.cpuinfo.CPUInfo;
44 
45 public class TestBlsmskL {
46 
47     public static void main(String args[]) throws Throwable {
48         if (!CPUInfo.hasFeature("bmi1")) {
49             System.out.println("INFO: CPU does not support bmi1 feature.");
50         }
51 
52         BMITestRunner.runTests(BlsmskLExpr.class, args,
53                                "-XX:+IgnoreUnrecognizedVMOptions",
54                                "-XX:+UseBMI1Instructions");
55         BMITestRunner.runTests(BlsmskLCommutativeExpr.class, args,
56                                "-XX:+IgnoreUnrecognizedVMOptions",
57                                "-XX:+UseBMI1Instructions");
58     }
59 
60     public static class BlsmskLExpr
61         extends Expr.BMIUnaryLongExpr {
62 
63         public long longExpr(long src) {
64             return (src - 1) ^ src;
65         }
66 
67         public long longExpr(Expr.MemL src) {
68             return (src.value - 1) ^ src.value;
69         }
70 
71     }
72 
73     public static class BlsmskLCommutativeExpr
74         extends Expr.BMIUnaryLongExpr {
75 
76         public long longExpr(long src) {
77             return src ^ (src - 1);
78         }
79 
80         public long longExpr(Expr.MemL src) {
81             return src.value ^ (src.value - 1);
82         }
83 
84     }
85 
86 }
