41 package compiler.intrinsics.bmi;
42 
43 import sun.hotspot.cpuinfo.CPUInfo;
44 
45 public class TestTzcntI {
46 
47     public static void main(String args[]) throws Throwable {
48         if (!CPUInfo.hasFeature("bmi1")) {
49             System.out.println("INFO: CPU does not support bmi1 feature.");
50         }
51 
52         BMITestRunner.runTests(TzcntIExpr.class, args,
53                                "-XX:+IgnoreUnrecognizedVMOptions",
54                                "-XX:+UseCountTrailingZerosInstruction");
55     }
56 
57     public static class TzcntIExpr extends Expr.BitCountingIntExpr {
58 
59         public int intExpr(int src) {
60             return Integer.numberOfTrailingZeros(src);
61         }
62 
63     }
64 
65 }
