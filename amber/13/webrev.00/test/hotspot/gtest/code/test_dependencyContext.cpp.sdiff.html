61   void wipe() {
62     DependencyContext ctx(&_dependency_context, &_last_cleanup);
63     nmethodBucket* b = ctx.dependencies();
64     ctx.set_dependencies(NULL);
65     while (b != NULL) {
66       nmethodBucket* next = b->next();
67       delete b;
68       b = next;
69     }
70   }
71 };
