24 /*
25  * @test
26  * @bug 8211901
27  * @summary  javadoc generates broken links on deprecated items page
28  * @library  ../../lib
29  * @modules  jdk.javadoc/jdk.javadoc.internal.tool
30  * @build    javadoc.tester.*
31  * @run main TestOverriddenDeprecatedMethods
32  */
33 
34 import javadoc.tester.JavadocTester;
35 
36 public class TestOverriddenDeprecatedMethods extends JavadocTester {
37 
38     public static void main(String args[]) throws Exception {
39         TestOverriddenDeprecatedMethods tester = new TestOverriddenDeprecatedMethods();
40         tester.runTests();
41     }
42 
43     @Test
44     public void test() {
45         javadoc("-d", "out-deprecated",
46                 "-sourcepath", testSrc,
47                 "--override-methods","summary",
48                 "pkg1");
49 
50         checkExit(Exit.OK);
51 
52         checkOrder("pkg1/SubClass.html",
53                 "Method Summary",
54                 "Methods declared in class&nbsp;pkg1.<a href=\"BaseClass.html\" title=\"class in pkg1\">"
55                 + "BaseClass</a>",
56                 "<a href=\"BaseClass.html#func3()\">func3</a>");
57 
58         checkOrder("pkg1/SubClass.html",
59                 "Method Detail",
60                 "<span class=\"annotations\">@Deprecated\n</span><span class=\"modifiers\">public</span>&nbsp;"
61                 + "<span class=\"return-type\">void</span>&nbsp;<span class=\"member-name\">func1</span>()",
62                 "<div class=\"deprecation-block\"><span class=\"deprecated-label\">Deprecated.</span></div>",
63                 "<span class=\"annotations\">@Deprecated\n</span><span class=\"modifiers\">public</span>&nbsp;"
64                 + "<span class=\"return-type\">void</span>&nbsp;<span class=\"member-name\">func2</span>()",
65                 "<div class=\"deprecation-block\"><span class=\"deprecated-label\">Deprecated.</span></div>",
66                 "<div class=\"block\">deprecated with comments</div>");
67     }
68 }
