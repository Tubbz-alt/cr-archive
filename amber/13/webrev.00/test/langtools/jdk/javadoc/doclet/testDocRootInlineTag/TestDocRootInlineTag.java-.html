24 /*
25  * @test
26  * @bug 4369014 4851991 8164407 8205593
27  * @summary Determine if the docRoot inline tag works properly.
28  * If docRoot performs as documented, the test passes.
29  * Make sure that the docRoot tag works with the -bottom option.
30  * @library ../../lib
31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
32  * @build javadoc.tester.*
33  * @run main TestDocRootInlineTag
34  */
35 
36 import javadoc.tester.JavadocTester;
37 
38 public class TestDocRootInlineTag extends JavadocTester {
39 
40     public static void main(String... args) throws Exception {
41         TestDocRootInlineTag tester = new TestDocRootInlineTag();
42         tester.runTests();
43     }
44 
45     @Test
46     public void test() {
47         String uri = "http://www.java.sun.com/j2se/1.4/docs/api";
48 
49         javadoc("-bottom", "The value of @docRoot is \"{@docRoot}\"",
50                 "-d", "out",
51                 "-source", "8",
52                 "-sourcepath", testSrc,
53                 "-linkoffline", uri, testSrc,
54                 testSrc("TestDocRootTag.java"), "pkg");
55         checkExit(Exit.OK);
56 
57         checkOutput("TestDocRootTag.html", true,
58                 "<a href=\"" + uri + "/java/io/File.html\" "
59                 + "title=\"class or interface in java.io\" class=\"externalLink\"><code>File</code></a>",
60                 "<a href=\"./index-all.html\">index</a>",
61                 "<a href=\"" + uri + "/java/io/File.html\" "
62                 + "title=\"class or interface in java.io\" class=\"externalLink\"><code>Second File Link</code></a>",
63                 "The value of @docRoot is \"./\"");
64 
65         checkOutput("index-all.html", true,
66                 "My package page is <a href=\"./pkg/package-summary.html\">here</a>");
67     }
68 }
