24 /*
25  * @test
26  * @summary Ensure the modifiers are correct
27  * @library ../../lib
28  * @modules jdk.javadoc/jdk.javadoc.internal.tool
29  * @build javadoc.tester.*
30  * @run main TestModifierEx
31  */
32 
33 import javadoc.tester.JavadocTester;
34 
35 public class TestModifierEx extends JavadocTester {
36 
37     public static void main(String... args) throws Exception {
38         TestModifierEx tester = new TestModifierEx();
39         tester.runTests();
40     }
41 
42     @Test
43     public void test1(){
44         javadoc("-d", "out-1",
45                 "-sourcepath", testSrc,
46                 "-package", "pkg1");
47         checkExit(Exit.OK);
48         checkOutput("pkg1/Abstract.html", true,
49                 "<pre>public abstract class <span class=\"typeNameLabel\">Abstract</span>");
50         checkOutput("pkg1/Interface.html", true,
51                 "<pre>interface <span class=\"typeNameLabel\">Interface</span></pre>");
52         checkOutput("pkg1/Interface.Kind.html", true,
53                 "<pre>public static interface <span class=\"typeNameLabel\">Interface.Kind</span></pre>");
54         checkOutput("pkg1/Enum.html", true,
55                 "<pre>public enum <span class=\"typeNameLabel\">Enum</span>");
56         checkOutput("pkg1/Klass.StaticEnum.html", true,
57                 "<pre>public static enum <span class=\"typeNameLabel\">Klass.StaticEnum</span>");
58     }
59 
60 }
