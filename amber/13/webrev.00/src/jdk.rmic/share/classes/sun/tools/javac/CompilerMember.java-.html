26 package sun.tools.javac;
27 
28 import sun.tools.java.*;
29 import sun.tools.asm.Assembler;
30 
31 /**
32  * This class is used to represents fields while they are
33  * being compiled
34  *
35  * WARNING: The contents of this source file are not part of any
36  * supported API.  Code that depends on them does so at its own risk:
37  * they are subject to change or removal without notice.
38  */
39 @Deprecated
40 final
41 class CompilerMember implements Comparable<Object> {
42     MemberDefinition field;
43     Assembler asm;
44     Object value;
45     String name;
46     String sig;
47     String key;
48 
49     CompilerMember(MemberDefinition field, Assembler asm) {
50         this.field = field;
51         this.asm = asm;
52         name = field.getName().toString();
53         sig = field.getType().getTypeSignature();
54     }
55 
56     public int compareTo(Object o) {
57         CompilerMember cm = (CompilerMember) o;
58         return getKey().compareTo(cm.getKey());
59     }
60 
61     String getKey() {
62         if (key==null)
63             key = name+sig;
64         return key;
65     }
66 
67 }
