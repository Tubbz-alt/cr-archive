26 package sun.tools.asm;
27 
28 import sun.tools.java.*;
29 import java.util.Vector;
30 
31 /**
32  * WARNING: The contents of this source file are not part of any
33  * supported API.  Code that depends on them does so at its own risk:
34  * they are subject to change or removal without notice.
35  */
36 public final
37 class TryData {
38     Vector<CatchData> catches = new Vector<>();
39     Label endLabel = new Label();
40 
41     /**
42      * Add a label
43      */
44     public CatchData add(Object type) {
45         CatchData cd = new CatchData(type);
46         catches.addElement(cd);
47         return cd;
48     }
49 
50     /**
51      * Get a label
52      */
53     public CatchData getCatch(int n) {
54         return catches.elementAt(n);
55     }
56 
57     /**
58      * Get the default label
59      */
60     public Label getEndLabel() {
61         return endLabel;
62     }
63 }
