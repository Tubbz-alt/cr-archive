35 public
36 class BinaryCompareExpression extends BinaryExpression {
37     /**
38      * constructor
39      */
40     public BinaryCompareExpression(int op, long where, Expression left, Expression right) {
41         super(op, where, Type.tBoolean, left, right);
42     }
43 
44     /**
45      * Select the type
46      */
47     void selectType(Environment env, Context ctx, int tm) {
48         Type t = Type.tInt;
49         if ((tm & TM_DOUBLE) != 0) {
50             t = Type.tDouble;
51         } else if ((tm & TM_FLOAT) != 0) {
52             t = Type.tFloat;
53         } else if ((tm & TM_LONG) != 0) {
54             t = Type.tLong;
55         }
56         left = convert(env, ctx, t, left);
57         right = convert(env, ctx, t, right);
58     }
59 }
