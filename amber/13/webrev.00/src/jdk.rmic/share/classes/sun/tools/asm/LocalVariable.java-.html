26 package sun.tools.asm;
27 
28 import sun.tools.java.*;
29 
30 /**
31  * This class is used to assemble the local variables in the local
32  * variable table.
33  *
34  * WARNING: The contents of this source file are not part of any
35  * supported API.  Code that depends on them does so at its own risk:
36  * they are subject to change or removal without notice.
37  *
38  * @author Arthur van Hoff
39  */
40 public final
41 class LocalVariable {
42     MemberDefinition field;
43     int slot;
44     int from;
45     int to;
46 
47     public LocalVariable(MemberDefinition field, int slot) {
48         if (field == null) {
49             new Exception().printStackTrace();
50         }
51         this.field = field;
52         this.slot = slot;
53         to = -1;
54     }
55 
56     LocalVariable(MemberDefinition field, int slot, int from, int to) {
57         this.field = field;
58         this.slot = slot;
59         this.from = from;
60         this.to = to;
61     }
62 
63     public String toString() {
64         return field + "/" + slot;
65     }
66 }
