26 package sun.tools.asm;
27 
28 import sun.tools.java.*;
29 import sun.tools.tree.StringExpression;
30 import java.io.IOException;
31 import java.io.DataOutputStream;
32 
33 /**
34  * This is a string expression constant. This constant
35  * represents an Java string constant.
36  *
37  * WARNING: The contents of this source file are not part of any
38  * supported API.  Code that depends on them does so at its own risk:
39  * they are subject to change or removal without notice.
40  */
41 final
42 class StringExpressionConstantData extends ConstantPoolData {
43     StringExpression str;
44 
45     /**
46      * Constructor
47      */
48     StringExpressionConstantData(ConstantPool tab, StringExpression str) {
49         this.str = str;
50         tab.put(str.getValue());
51     }
52 
53     /**
54      * Write the constant to the output stream
55      */
56     void write(Environment env, DataOutputStream out, ConstantPool tab) throws IOException {
57         out.writeByte(CONSTANT_STRING);
58         out.writeShort(tab.index(str.getValue()));
59     }
60 
61     /**
62      * Return the order of the constant
63      */
64     int order() {
65         return 0;
66     }
67 
68     /**
69      * toString
70      */
71     public String toString() {
72         return "StringExpressionConstantData[" + str.getValue() + "]=" + str.getValue().hashCode();
73     }
74 }
