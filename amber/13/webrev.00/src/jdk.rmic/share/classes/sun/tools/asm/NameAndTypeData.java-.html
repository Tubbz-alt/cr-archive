26 package sun.tools.asm;
27 
28 import sun.tools.java.*;
29 
30 /**
31  * An object to represent a name and type constant pool data item.
32  *
33  * WARNING: The contents of this source file are not part of any
34  * supported API.  Code that depends on them does so at its own risk:
35  * they are subject to change or removal without notice.
36  */
37 final
38 class NameAndTypeData {
39     MemberDefinition field;
40 
41     /**
42      * Constructor
43      */
44     NameAndTypeData(MemberDefinition field) {
45         this.field = field;
46     }
47 
48     /**
49      * Hashcode
50      */
51     public int hashCode() {
52         return field.getName().hashCode() * field.getType().hashCode();
53     }
54 
55     /**
56      * Equality
57      */
58     public boolean equals(Object obj) {
59         if ((obj != null) && (obj instanceof NameAndTypeData)) {
60             NameAndTypeData nt = (NameAndTypeData)obj;
61             return field.getName().equals(nt.field.getName()) &&
62                 field.getType().equals(nt.field.getType());
63         }
64         return false;
65     }
66 
67     /**
68      * Convert to string
69      */
70     public String toString() {
71         return "%%" + field.toString() + "%%";
72     }
73 }
