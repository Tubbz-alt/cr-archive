26 package sun.tools.asm;
27 
28 import sun.tools.java.*;
29 import java.io.IOException;
30 import java.io.DataOutputStream;
31 
32 /**
33  * This is a name and type constant pool data item
34  *
35  * WARNING: The contents of this source file are not part of any
36  * supported API.  Code that depends on them does so at its own risk:
37  * they are subject to change or removal without notice.
38  */
39 final
40 class NameAndTypeConstantData extends ConstantPoolData {
41     String name;
42     String type;
43 
44     /**
45      * Constructor
46      */
47     NameAndTypeConstantData(ConstantPool tab, NameAndTypeData nt) {
48         name = nt.field.getName().toString();
49         type = nt.field.getType().getTypeSignature();
50         tab.put(name);
51         tab.put(type);
52     }
53 
54     /**
55      * Write the constant to the output stream
56      */
57     void write(Environment env, DataOutputStream out, ConstantPool tab) throws IOException {
58         out.writeByte(CONSTANT_NAMEANDTYPE);
59         out.writeShort(tab.index(name));
60         out.writeShort(tab.index(type));
61     }
62 
63     /**
64      * Return the order of the constant
65      */
66     int order() {
67         return 3;
68     }
69 }
