37 abstract public
38 class DivRemExpression extends BinaryArithmeticExpression {
39     /**
40      * constructor
41      */
42     public DivRemExpression(int op, long where, Expression left, Expression right) {
43         super(op, where, left, right);
44     }
45 
46     /**
47      * Inline
48      */
49     public Expression inline(Environment env, Context ctx) {
50         // Do not toss out integer divisions or remainders since they
51         // can cause a division by zero.
52         if (type.inMask(TM_INTEGER)) {
53             right = right.inlineValue(env, ctx);
54             if (right.isConstant() && !right.equals(0)) {
55                 // We know the division can be elided
56                 left = left.inline(env, ctx);
57                 return left;
58             } else {
59                 left = left.inlineValue(env, ctx);
60                 try {
61                     return eval().simplify();
62                 } catch (ArithmeticException e) {
63                     env.error(where, "arithmetic.exception");
64                     return this;
65                 }
66             }
67         } else {
68             // float & double divisions don't cause arithmetic errors
69             return super.inline(env, ctx);
70         }
71     }
72 }
