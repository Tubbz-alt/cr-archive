 26 package sun.tools.java;
 27 
 28 import java.io.File;
 29 import java.io.InputStream;
 30 import java.io.FileInputStream;
 31 import java.io.IOException;
 32 import java.nio.file.Files;
 33 import java.nio.file.Path;
 34 import java.util.zip.*;
 35 
 36 /**
 37  * Abstract class to represent a class file.
 38  *
 39  * WARNING: The contents of this source file are not part of any
 40  * supported API.  Code that depends on them does so at its own risk:
 41  * they are subject to change or removal without notice.
 42  */
 43 public
 44 abstract class ClassFile {
 45     /**
 46      * Factory method to create a ClassFile backed by a File.
 47      *
 48      * @param file a File object
 49      * @return a new ClassFile
 50      */
 51     public static ClassFile newClassFile(File file) {
 52         return new FileClassFile(file);
 53     }
 54 
 55     /**
 56      * Factory method to create a ClassFile backed by a ZipEntry.
 57      *
 58      * @param zf a ZipFile
 59      * @param ze a ZipEntry within the zip file
 60      * @return a new ClassFile
 61      */
 62     public static ClassFile newClassFile(ZipFile zf, ZipEntry ze) {
 63         return new ZipClassFile(zf, ze);
 64     }
 65 
 66     /**
 67      * Factory method to create a ClassFile backed by a nio Path.
 68      *
 69      * @param path nio Path object
 70      * @return a new ClassFile
 71      */
 72     public static ClassFile newClassFile(Path path) {
 73         return Files.exists(path)? new PathClassFile(path) : null;
 74     }
 75 
 76     /**
 77      * Returns true if this is zip file entry
 78      */
 79     public abstract boolean isZipped();
 80 
 81     /**
 82      * Returns input stream to either regular file or zip file entry
 83      */
 84     public abstract InputStream getInputStream() throws IOException;
 85 
 86     /**
 87      * Returns true if file exists.
 88      */
 89     public abstract boolean exists();
 90 
 91     /**
 92      * Returns true if this is a directory.
 93      */
 94     public abstract boolean isDirectory();
 95 
 96     /**
 97      * Return last modification time
 98      */
 99     public abstract long lastModified();
100 
101     /**
102      * Get file path. The path for a zip file entry will also include
103      * the zip file name.
104      */
105     public abstract String getPath();
106 
107     /**
108      * Get name of file entry excluding directory name
109      */
110     public abstract String getName();
111 
112     /**
113      * Get absolute name of file entry
114      */
115     public abstract String getAbsoluteName();
116 
117     /**
118      * Get length of file
119      */
120     public abstract long length();
121 }
