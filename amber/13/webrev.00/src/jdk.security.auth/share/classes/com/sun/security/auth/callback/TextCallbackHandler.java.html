26 package com.sun.security.auth.callback;
27 
28 /* JAAS imports */
29 import javax.security.auth.callback.Callback;
30 import javax.security.auth.callback.CallbackHandler;
31 import javax.security.auth.callback.NameCallback;        // javadoc
32 import javax.security.auth.callback.PasswordCallback;    // javadoc
33 import javax.security.auth.callback.UnsupportedCallbackException;
34 
35 /* Java imports */
36 import java.io.IOException;
37 
38 import sun.security.util.ConsoleCallbackHandler;
39 
40 /**
41  * Prompts and reads from the command line for answers to authentication
42  * questions.
43  * This can be used by a JAAS application to instantiate a
44  * CallbackHandler
45  * @see javax.security.auth.callback
46  */
47 
48 public class TextCallbackHandler implements CallbackHandler {
49     private final CallbackHandler consoleHandler;
50 
51     /**
52      * Creates a callback handler that prompts and reads from the
53      * command line for answers to authentication questions.
54      * This can be used by JAAS applications to instantiate a
55      * CallbackHandler.
56      */
57     public TextCallbackHandler() {
58         this.consoleHandler = new ConsoleCallbackHandler();
59     }
60 
61     /**
62      * Handles the specified set of callbacks.
63      *
64      * @param callbacks the callbacks to handle
65      * @throws IOException if an input or output error occurs.
66      * @throws UnsupportedCallbackException if the callback is not an
67      * instance of NameCallback or PasswordCallback
68      */
69     public void handle(Callback[] callbacks)
70         throws IOException, UnsupportedCallbackException
71     {
72         // delegate to console handler
73         consoleHandler.handle(callbacks);
74     }
75 }
