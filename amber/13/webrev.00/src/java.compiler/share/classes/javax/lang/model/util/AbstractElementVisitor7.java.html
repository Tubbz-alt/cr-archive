26 package javax.lang.model.util;

28 import javax.annotation.processing.SupportedSourceVersion;
29 import javax.lang.model.SourceVersion;
30 import static javax.lang.model.SourceVersion.*;
31 
32 
33 /**
34  * A skeletal visitor of program elements with default behavior
35  * appropriate for the {@link SourceVersion#RELEASE_7 RELEASE_7}
36  * source version.
37  *
38  * @param <R> the return type of this visitor's methods.  Use {@link
39  *            Void} for visitors that do not need to return results.
40  * @param <P> the type of the additional parameter to this visitor's
41  *            methods.  Use {@code Void} for visitors that do not need an
42  *            additional parameter.
43  *
44  * @see <a href="AbstractElementVisitor6.html#note_for_subclasses">
45  * <strong>Compatibility note for subclasses</strong></a>
46  * @see AbstractElementVisitor6
47  * @see AbstractElementVisitor8
48  * @see AbstractElementVisitor9
49  * @see AbstractElementVisitor14
50  * @since 1.7
51  */
52 @SupportedSourceVersion(RELEASE_7)
53 public abstract class AbstractElementVisitor7<R, P> extends AbstractElementVisitor6<R, P> {
54     /**
55      * Constructor for concrete subclasses to call.
56      *
57      * @deprecated Release 7 is obsolete; update to a visitor for a newer
58      * release level.
59      */
60     @Deprecated(since="12")
61     protected AbstractElementVisitor7(){
62         super(); // Superclass constructor deprecated too
63     }
64 }
