26 package javax.lang.model.util;
27 
28 import javax.annotation.processing.SupportedSourceVersion;
29 import javax.lang.model.SourceVersion;
30 import static javax.lang.model.SourceVersion.*;
31 
32 /**
33  * A simple visitor for annotation values with default behavior
34  * appropriate for the {@link SourceVersion#RELEASE_8 RELEASE_8}
35  * source version.  Visit methods call {@link #defaultAction
36  * defaultAction} passing their arguments to {@code defaultAction}'s
37  * corresponding parameters.
38  *
39  * @apiNote
40  * Methods in this class may be overridden subject to their general
41  * contract.


















42  *
43  * @param <R> the return type of this visitor's methods
44  * @param <P> the type of the additional parameter to this visitor's methods.
45  *
46  * @see <a href="SimpleAnnotationValueVisitor6.html#note_for_subclasses">
47  * <strong>Compatibility note for subclasses</strong></a>
48  * @see SimpleAnnotationValueVisitor6
49  * @see SimpleAnnotationValueVisitor7
50  * @see SimpleAnnotationValueVisitor8
51  * @see SimpleAnnotationValueVisitor14
52  * @since 1.8
53  */
54 @SupportedSourceVersion(RELEASE_8)
55 public class SimpleAnnotationValueVisitor8<R, P> extends SimpleAnnotationValueVisitor7<R, P> {
56     /**
57      * Constructor for concrete subclasses; uses {@code null} for the
58      * default value.
59      */
60     @SuppressWarnings("deprecation") // Superclass constructor deprecated
61     protected SimpleAnnotationValueVisitor8() {
62         super(null);
63     }
64 
65     /**
66      * Constructor for concrete subclasses; uses the argument for the
67      * default value.
68      *
69      * @param defaultValue the value to assign to {@link #DEFAULT_VALUE}
70      */
71     @SuppressWarnings("deprecation") // Superclass constructor deprecated
72     protected SimpleAnnotationValueVisitor8(R defaultValue) {
73         super(defaultValue);
74     }
75 }
