135     protected void addContents(Character uc, List<? extends Element> memberlist,
136             List<SearchIndexItem> searchList, Content contentTree) {
137         addHeading(uc, contentTree);
138         int memberListSize = memberlist.size();
139         int searchListSize = searchList.size();
140         int i = 0;
141         int j = 0;
142         HtmlTree dl = HtmlTree.DL(HtmlStyle.index);
143         while (i < memberListSize && j < searchListSize) {
144             Element elem = memberlist.get(i);
145             String name = (utils.isModule(elem))
146                     ? utils.getFullyQualifiedName(elem) : utils.getSimpleName(elem);
147             if (name.compareTo(searchList.get(j).getLabel()) < 0) {
148                 addDescription(dl, memberlist.get(i));
149                 i++;
150             } else if (name.compareTo(searchList.get(j).getLabel()) > 0) {
151                 addDescription(searchList.get(j), dl);
152                 j++;
153             } else {
154                 addDescription(dl, memberlist.get(i));
155                 addDescription(searchList.get(j), dl);
156                 j++;
157                 i++;
158             }
159         }
160         if (i >= memberListSize) {
161             while (j < searchListSize) {
162                 addDescription(searchList.get(j), dl);
163                 j++;
164             }
165         }
166         if (j >= searchListSize) {
167             while (i < memberListSize) {
168                 addDescription(dl, memberlist.get(i));
169                 i++;
170             }
171         }
172         contentTree.add(dl);
173     }

313 
314         si.setContainingPackage(utils.getPackageName(utils.containingPackage(member)));
315         si.setContainingClass(utils.getSimpleName(utils.getEnclosingTypeElement(member)));
316         String name = utils.getSimpleName(member);

318             ExecutableElement ee = (ExecutableElement)member;
319             name = name + utils.flatSignature(ee);
320             si.setLabel(name);
321             String url = HtmlTree.encodeURL(links.getName(getAnchor(ee)));
322             if (!name.equals(url)) {
323                 si.setUrl(url);
324             }
325         }  else {
326             si.setLabel(name);
327         }
328         si.setCategory(SearchIndexItem.Category.MEMBERS);
329         Content span = HtmlTree.SPAN(HtmlStyle.memberNameLink,
330                 getDocLink(LinkInfoImpl.Kind.INDEX, member, name));
331         Content dt = HtmlTree.DT(span);
332         dt.add(" - ");
333         addMemberDesc(member, dt);
334         dlTree.add(dt);
335         Content dd = new HtmlTree(HtmlTag.DD);
336         addComment(member, dd);
337         dlTree.add(dd);

340     protected void addDescription(SearchIndexItem sii, Content dlTree) {
341         String siiPath = pathToRoot.isEmpty() ? "" : pathToRoot.getPath() + "/";
342         siiPath += sii.getUrl();
343         HtmlTree labelLink = HtmlTree.A(siiPath, new StringContent(sii.getLabel()));
344         Content dt = HtmlTree.DT(HtmlTree.SPAN(HtmlStyle.searchTagLink, labelLink));
345         dt.add(" - ");
346         dt.add(contents.getContent("doclet.Search_tag_in", sii.getHolder()));
347         dlTree.add(dt);
348         Content dd = new HtmlTree(HtmlTag.DD);
349         if (sii.getDescription().isEmpty()) {
350             dd.add(Entity.NO_BREAK_SPACE);
351         } else {
352             dd.add(sii.getDescription());
353         }
354         dlTree.add(dd);
355     }

357     /**
358      * Add comment for each element in the index. If the element is deprecated
359      * and it has a @deprecated tag, use that comment. Else if the containing
360      * class for this element is deprecated, then add the word "Deprecated." at
361      * the start and then print the normal comment.
362      *
363      * @param element Index element
364      * @param contentTree the content tree to which the comment will be added
365      */
366     protected void addComment(Element element, Content contentTree) {
367         List<? extends DocTree> tags;
368         Content span = HtmlTree.SPAN(HtmlStyle.deprecatedLabel, getDeprecatedPhrase(element));
369         HtmlTree div = new HtmlTree(HtmlTag.DIV);
370         div.setStyle(HtmlStyle.deprecationBlock);
371         if (utils.isDeprecated(element)) {
372             div.add(span);
373             tags = utils.getBlockTags(element, DocTree.Kind.DEPRECATED);
374             if (!tags.isEmpty())
375                 addInlineDeprecatedComment(element, tags.get(0), div);
376             contentTree.add(div);
377         } else {
378             TypeElement encl = utils.getEnclosingTypeElement(element);
379             while (encl != null) {
380                 if (utils.isDeprecated(encl)) {
381                     div.add(span);
382                     contentTree.add(div);
383                     break;
384                 }
385                 encl = utils.getEnclosingTypeElement(encl);
386             }
387             addSummaryComment(element, contentTree);
388         }
389     }

391     /**
392      * Add description about the Static Variable/Method/Constructor for a
393      * member.
394      *
395      * @param member MemberDoc for the member within the Class Kind

396      * @param contentTree the content tree to which the member description will be added
397      */
398     protected void addMemberDesc(Element member, Content contentTree) {
399         TypeElement containing = utils.getEnclosingTypeElement(member);
400         String classdesc = utils.getTypeElementName(containing, true) + " ";
401         if (utils.isField(member)) {
402             Content resource = contents.getContent(utils.isStatic(member)
403                     ? "doclet.Static_variable_in"
404                     : "doclet.Variable_in", classdesc);
405             contentTree.add(resource);
406         } else if (utils.isConstructor(member)) {
407             contentTree.add(
408                     contents.getContent("doclet.Constructor_for", classdesc));
409         } else if (utils.isMethod(member)) {
410             Content resource = contents.getContent(utils.isStatic(member)
411                     ? "doclet.Static_method_in"
412                     : "doclet.Method_in", classdesc);
413             contentTree.add(resource);
414         }
415         addPreQualifiedClassLink(LinkInfoImpl.Kind.INDEX, containing,
416                 false, contentTree);
417     }

464     /**
465      * Creates a search index file.
466      *
467      * @param searchIndexJS     the file for the JavaScript to be generated
468      * @param searchIndex       the search index items
469      * @param varName           the variable name to write in the JavaScript file
470      * @throws DocFileIOException if there is a problem creating the search index file
471      */
472     protected void createSearchIndexFile(DocPath searchIndexJS,
473                                          Collection<SearchIndexItem> searchIndex,
474                                          String varName)
475             throws DocFileIOException
476     {
477         // The file needs to be created even if there are no searchIndex items
478         // File could be written straight-through, without an intermediate StringBuilder
479         if (!searchIndex.isEmpty()) {

480             StringBuilder searchVar = new StringBuilder("[");
481             boolean first = true;
482             for (SearchIndexItem item : searchIndex) {

483                 if (first) {
484                     searchVar.append(item.toString());
485                     first = false;
486                 } else {
487                     searchVar.append(",").append(item.toString());
488                 }
489             }
490             searchVar.append("]");
491             DocFile jsFile = DocFile.createFileForOutput(configuration, searchIndexJS);
492             try (Writer wr = jsFile.openWriter()) {
493                 wr.write(varName);
494                 wr.write(" = ");
495                 wr.write(searchVar.toString());
496             } catch (IOException ie) {
497                 throw new DocFileIOException(jsFile, DocFileIOException.Mode.WRITE, ie);
498             }
499         }
500     }

502     protected static Map<Character, List<SearchIndexItem>> buildSearchTagIndex(
503             Collection<? extends SearchIndexItem> searchItems)
504     {
505         Map<Character, List<SearchIndexItem>> map = new HashMap<>();
506         for (SearchIndexItem sii : searchItems) {
507             String tagLabel = sii.getLabel();
508             Character unicode = (tagLabel.length() == 0)
509                     ? '*'
510                     : Character.toUpperCase(tagLabel.charAt(0));
511             map.computeIfAbsent(unicode, k -> new ArrayList<>()).add(sii);
512         }
513         return map;
514     }

141      */
142     protected void addElementDescription(IndexItem indexItem, Content dlTree, SearchIndexItem si) {
143         Content dt;
144         Element element = indexItem.getElement();
145         String label = indexItem.getLabel();
146         switch (element.getKind()) {
147             case MODULE:
148                 dt = HtmlTree.DT(getModuleLink((ModuleElement)element, new StringContent(label)));
149                 si.setCategory(Category.MODULES);
150                 dt.add(" - ").add(contents.module_).add(" " + label);
151                 break;
152             case PACKAGE:
153                 dt = HtmlTree.DT(getPackageLink((PackageElement)element, new StringContent(label)));
154                 if (configuration.showModules) {
155                     si.setContainingModule(utils.getFullyQualifiedName(utils.containingModule(element)));
156                 }
157                 si.setCategory(Category.PACKAGES);
158                 dt.add(" - ").add(contents.package_).add(" " + label);
159                 break;
160             case CLASS:
161             case ENUM:
162             case ANNOTATION_TYPE:
163             case INTERFACE:
164                 dt = HtmlTree.DT(getLink(new LinkInfoImpl(configuration,
165                         LinkInfoImpl.Kind.INDEX, (TypeElement)element).strong(true)));
166                 si.setContainingPackage(utils.getPackageName(utils.containingPackage(element)));
167                 si.setCategory(Category.TYPES);
168                 dt.add(" - ");
169                 addClassInfo((TypeElement)element, dt);
170                 break;
171             default:
172                 TypeElement containingType = indexItem.getTypeElement();
173                 dt = HtmlTree.DT(HtmlTree.SPAN(HtmlStyle.memberNameLink,
174                         getDocLink(LinkInfoImpl.Kind.INDEX, containingType, element, new StringContent(label))));
175                 si.setContainingPackage(utils.getPackageName(utils.containingPackage(element)));
176                 si.setContainingClass(utils.getSimpleName(containingType));
177                 if (utils.isExecutableElement(element)) {
178                     String url = HtmlTree.encodeURL(links.getName(getAnchor((ExecutableElement)element)));
179                     if (!label.equals(url)) {
180                         si.setUrl(url);
181                     }
182                 }
183                 si.setCategory(Category.MEMBERS);
184                 dt.add(" - ");
185                 addMemberDesc(element, containingType, dt);
186                 break;
187         }




























188         dlTree.add(dt);
189         Content dd = new HtmlTree(TagName.DD);
190         if (element.getKind() == ElementKind.MODULE || element.getKind() == ElementKind.PACKAGE) {
191             addSummaryComment(element, dd);
192         } else {
193             addComment(element, dd);
194         }
195         dlTree.add(dd);
196     }

230     /**
231      * Add comment for each element in the index. If the element is deprecated
232      * and it has a @deprecated tag, use that comment. Else if the containing
233      * class for this element is deprecated, then add the word "Deprecated." at
234      * the start and then print the normal comment.
235      *
236      * @param element Index element
237      * @param contentTree the content tree to which the comment will be added
238      */
239     protected void addComment(Element element, Content contentTree) {
240         List<? extends DocTree> tags;
241         Content span = HtmlTree.SPAN(HtmlStyle.deprecatedLabel, getDeprecatedPhrase(element));
242         HtmlTree div = new HtmlTree(TagName.DIV);
243         div.setStyle(HtmlStyle.deprecationBlock);
244         if (utils.isDeprecated(element)) {
245             div.add(span);
246             tags = utils.getBlockTags(element, DocTree.Kind.DEPRECATED);
247             if (!tags.isEmpty())
248                 addInlineDeprecatedComment(element, tags.get(0), div);
249             contentTree.add(div);
250         } else {
251             TypeElement encl = utils.getEnclosingTypeElement(element);
252             while (encl != null) {
253                 if (utils.isDeprecated(encl)) {
254                     div.add(span);
255                     contentTree.add(div);
256                     break;
257                 }
258                 encl = utils.getEnclosingTypeElement(encl);
259             }
260             addSummaryComment(element, contentTree);
261         }
262     }

265      * Add description about the Static Variable/Method/Constructor for a
266      * member.
267      *
268      * @param member element for the member
269      * @param enclosing the enclosing type element
270      * @param contentTree the content tree to which the member description will be added
271      */
272     protected void addMemberDesc(Element member, TypeElement enclosing, Content contentTree) {
273         String classdesc = utils.getTypeElementName(enclosing, true) + " ";

274         if (utils.isField(member)) {
275             Content resource = contents.getContent(utils.isStatic(member)
276                     ? "doclet.Static_variable_in"
277                     : "doclet.Variable_in", classdesc);
278             contentTree.add(resource);
279         } else if (utils.isConstructor(member)) {

281                     contents.getContent("doclet.Constructor_for", classdesc));

284                     ? "doclet.Static_method_in"
285                     : "doclet.Method_in", classdesc);
286             contentTree.add(resource);
287         }
288         addPreQualifiedClassLink(LinkInfoImpl.Kind.INDEX, enclosing,
289                 false, contentTree);
290     }

337     /**
338      * Creates a search index file.
339      *
340      * @param searchIndexJS     the file for the JavaScript to be generated
341      * @param searchIndex       the search index items
342      * @param varName           the variable name to write in the JavaScript file
343      * @throws DocFileIOException if there is a problem creating the search index file
344      */
345     protected void createSearchIndexFile(DocPath searchIndexJS,
346                                          Stream<SearchIndexItem> searchIndex,
347                                          String varName)
348             throws DocFileIOException
349     {
350         // The file needs to be created even if there are no searchIndex items
351         // File could be written straight-through, without an intermediate StringBuilder
352         Iterator<SearchIndexItem> index = searchIndex.iterator();
353         if (index.hasNext()) {
354             StringBuilder searchVar = new StringBuilder("[");
355             boolean first = true;
356             while (index.hasNext()) {
357                 SearchIndexItem item = index.next();
358                 if (first) {
359                     searchVar.append(item.toString());
360                     first = false;
361                 } else {
362                     searchVar.append(",").append(item.toString());
363                 }
364             }
365             searchVar.append("]");
366             DocFile jsFile = DocFile.createFileForOutput(configuration, searchIndexJS);
367             try (Writer wr = jsFile.openWriter()) {
368                 wr.write(varName);
369                 wr.write(" = ");
370                 wr.write(searchVar.toString());
371             } catch (IOException ie) {
372                 throw new DocFileIOException(jsFile, DocFileIOException.Mode.WRITE, ie);
373             }
374         }
375     }
