 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import java.io.IOException;
 29 import java.io.StringWriter;
 30 import java.io.Writer;
 31 import java.util.Objects;
 32 
 33 /**
 34  * A class to create content for javadoc output pages.
 35  *
 36  *  <p><b>This is NOT part of any supported API.
 37  *  If you write code that depends on this, you do so at your own risk.
 38  *  This code and its internal interfaces are subject to change or
 39  *  deletion without notice.</b>
 40  */
 41 public abstract class Content {
 42 
 43     /**
 44      * Returns a string representation of the content.
 45      *
 46      * @return string representation of the content
 47      */
 48     @Override
 49     public String toString() {
 50         StringWriter out = new StringWriter();
 51         try {
 52             write(out, true);
 53         } catch (IOException e) {
 54             // cannot happen from StringWriter
 55             throw new AssertionError(e);
 56         }
 57         return out.toString();
 58     }
 59 
 60     /**
 61      * Adds content to the existing content.
 62      * This is an optional operation.
 63      *
 64      * @implSpec This implementation throws {@linkplain UnsupportedOperationException}.
 65      *
 66      * @param content content to be added
 67      * @return this object
 68      * @throws UnsupportedOperationException if this operation is not supported by
 69      *                                       a particular implementation
 70      * @throws IllegalArgumentException      if the content is not suitable to be added
 71      */
 72     public Content add(Content content) {
 73         throw new UnsupportedOperationException();
 74     }
 75 
 76     /**
 77      * Adds a string content to the existing content.
 78      * This is an optional operation.
 79      *
 80      * @implSpec
 81      * This implementation throws {@linkplain UnsupportedOperationException}.
 82      *
 83      * @param stringContent the string content to be added
 84      * @return this object
 85      * @throws UnsupportedOperationException if this operation is not supported by
 86      *                                       a particular implementation
 87      * @throws IllegalArgumentException      if the content is not suitable to be added
 88      */
 89     public Content add(CharSequence stringContent) {
 90         throw new UnsupportedOperationException();
 91     }
 92 
 93     /**
 94      * Writes content to a writer.
 95      *
 96      * @param writer the writer
 97      * @param atNewline whether the writer has just written a newline
 98      * @return  whether the writer has just written a newline
 99      * @throws IOException if an error occurs while writing the output
100      */
101     public abstract boolean write(Writer writer, boolean atNewline) throws IOException ;
102 
103     /**
104      * Returns true if the content is empty.
105      *
106      * @return true if no content to be displayed else return false
107      */
108     public abstract boolean isEmpty();
109 
110     /**
111      * Returns true if the content is valid.
112      *
113      * @return true if the content is valid else return false
114      */
115     public boolean isValid() {
116         return !isEmpty();
117     }
118 
119     /**
120      * Return the number of characters of plain text content in this object
121      * (optional operation.)
122      * @return the number of characters of plain text content in this
123      */
124     public int charCount() {
125         return 0;
126     }











127 }
