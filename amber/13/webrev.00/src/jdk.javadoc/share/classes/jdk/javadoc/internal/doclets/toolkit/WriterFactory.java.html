 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import javax.lang.model.element.Element;
 29 import javax.lang.model.element.ModuleElement;
 30 import javax.lang.model.element.PackageElement;
 31 import javax.lang.model.element.TypeElement;
 32 
 33 import jdk.javadoc.internal.doclets.toolkit.util.ClassTree;
 34 import jdk.javadoc.internal.doclets.toolkit.util.VisibleMemberTable;
 35 
 36 /**
 37  * The interface for a factory creates writers.
 38  *
 39  *  <p><b>This is NOT part of any supported API.
 40  *  If you write code that depends on this, you do so at your own risk.
 41  *  This code and its internal interfaces are subject to change or
 42  *  deletion without notice.</b>
 43  */
 44 
 45 public interface WriterFactory {
 46 
 47     /**
 48      * Return the writer for the constant summary.
 49      *
 50      * @return the writer for the constant summary.  Return null if this
 51      * writer is not supported by the doclet.
 52      */
 53     ConstantsSummaryWriter getConstantsSummaryWriter();
 54 
 55     /**
 56      * Return the writer for the package summary.
 57      *
 58      * @param packageElement the package being documented.
 59      * @return the writer for the package summary.  Return null if this
 60      * writer is not supported by the doclet.
 61      */
 62     PackageSummaryWriter getPackageSummaryWriter(PackageElement packageElement);
 63 
 64     /**
 65      * Return the writer for the module summary.
 66      *
 67      * @param mdle the module being documented.
 68      * @return the writer for the module summary.  Return null if this
 69      * writer is not supported by the doclet.
 70      */
 71     ModuleSummaryWriter getModuleSummaryWriter(ModuleElement mdle);
 72 
 73     /**
 74      * Returns the writer for a given type element,
 75      * or null if this writer is not supported by the doclet.
 76      *
 77      * @param typeElement the class being documented
 78      * @param classTree   the class tree
 79      * @return the writer
 80      */
 81     ClassWriter getClassWriter(TypeElement typeElement, ClassTree classTree);
 82 
 83     /**
 84      * Return the method writer for a given type element,
 85      * or null if this writer is not supported by the doclet.
 86      *
 87      * @param classWriter the writer for the class being documented
 88      * @return the method writer
 89      */
 90     MethodWriter getMethodWriter(ClassWriter classWriter);
 91 
 92     /**
 93      * Return the annotation type optional member writer for a given annotation
 94      * type, or null if this writer is not supported by the doclet.
 95      *
 96      * @param classWriter the writer for the annotation type being documented
 97      * @return the member writer
 98      */
 99     AnnotationTypeOptionalMemberWriter getAnnotationTypeOptionalMemberWriter(
100             ClassWriter classWriter);
101 
102     /**
103      * Return the annotation type required member writer for a given annotation
104      * type, or null if this writer is not supported by the doclet.
105      *
106      * @param classWriter the writer for the annotation type being documented
107      * @return the member writer
108      */
109     AnnotationTypeRequiredMemberWriter getAnnotationTypeRequiredMemberWriter(
110             ClassWriter classWriter);
111 
112     /**
113      * Return the enum constant writer for a given type element,
114      * or null if this writer is not supported by the doclet.
115      *
116      * @param classWriter the writer for the type element being documented
117      * @return the enum constant writer
118      */
119     EnumConstantWriter getEnumConstantWriter(ClassWriter classWriter);
120 
121     /**
122      * Return the field writer for a given type element,
123      * or null if this writer is not supported by the doclet.
124      *
125      * @param classWriter the writer for the class being documented.
126      * @return the field writer for the given class.  Return null if this
127      * writer is not supported by the doclet.
128      */
129     FieldWriter getFieldWriter(ClassWriter classWriter);
130 
131     /**
132      * Return the property writer for a given class,
133      * or null if this writer is not supported by the doclet.
134      *
135      * @param classWriter the writer for the type elemen t being documented.
136      * @return the property writer
137      */
138     PropertyWriter getPropertyWriter(ClassWriter classWriter);
139 
140     /**
141      * Return the constructor writer for a given type element,
142      * or null if this writer is not supported by the doclet.
143      *
144      * @param classWriter the writer for the type element being documented
145      * @return the constructor writer
146      */
147     ConstructorWriter getConstructorWriter(ClassWriter classWriter);
148 
149     /**
150      * Return the specified member summary writer for a given type element,
151      * or null if this writer is not supported by the doclet.
152      *
153      * @param classWriter the writer for the class being documented.
154      * @param memberType  the {@link VisibleMemberTable} member type indicating
155      *                    the type of member summary that should be returned.
156      * @return the summary writer
157      *
158      * @see VisibleMemberTable
159      */
160     MemberSummaryWriter getMemberSummaryWriter(ClassWriter classWriter,
161                                                VisibleMemberTable.Kind memberType);
162 
163     /**
164      * Return the writer for the serialized form.
165      *
166      * @return the writer for the serialized form
167      */
168     SerializedFormWriter getSerializedFormWriter();
169 
170     /**
171      * Return the handler for doc files.
172      *
173      * @return the handler for the doc files
174      */
175     DocFilesHandler getDocFilesHandler(Element pkg);
176 }
