40 /**
41  * Intrinsic for closing a {@linkplain BeginLockScopeNode scope} binding a stack-based lock with an
42  * object.
43  */
44 @NodeInfo(allowedUsageTypes = Memory, cycles = CYCLES_0, size = SIZE_0)
45 public final class EndLockScopeNode extends AbstractMemoryCheckpoint implements LIRLowerable, MonitorExit {
46     public static final NodeClass<EndLockScopeNode> TYPE = NodeClass.create(EndLockScopeNode.class);
47 
48     public EndLockScopeNode() {
49         super(TYPE, StampFactory.forVoid());
50     }
51 
52     @Override
53     public boolean hasSideEffect() {
54         return false;
55     }
56 
57     @Override
58     public LocationIdentity getKilledLocationIdentity() {
59         return LocationIdentity.any();
60     }
61 
62     @Override
63     public void generate(NodeLIRBuilderTool gen) {
64     }
65 
66     @NodeIntrinsic
67     public static native void endLockScope();
68 }
