31 /**
32  * Strange optimisation in for loop with cyclic integer condition.
33  *
34  * @test
35  * @bug 6753639
36  * @run main/othervm -Xbatch Test6753639
37  */
38 // @formatter:off
39 public class Test6753639 extends JTTTest {
40 
41     public static int test() {
42         int end = Integer.MAX_VALUE;
43         int count = 0;
44         for (int i = Integer.MAX_VALUE - 5; i <= end; i++) {
45             count++;
46             if (count > 100000) {
47                 return 95;
48             }
49         }
50         return 97;
51     }
52 
53     @Test
54     public void run0() throws Throwable {
55         runTest("test");
56     }
57 
58 }
