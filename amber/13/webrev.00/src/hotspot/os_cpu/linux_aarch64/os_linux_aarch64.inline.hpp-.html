29 #include "runtime/os.hpp"

31 // See http://www.technovelty.org/code/c/reading-rdtsc.htl for details
32 inline jlong os::rdtsc() {
33   uint64_t res;
34   uint32_t ts1, ts2;
35   __asm__ __volatile__ ("rdtsc" : "=a" (ts1), "=d" (ts2));
36   res = ((uint64_t)ts1 | (uint64_t)ts2 << 32);
37   return (jlong)res;
38 }
