32 bool VM_ShenandoahReferenceOperation::doit_prologue() {
33   Heap_lock->lock();
34   return true;
35 }
36 
37 void VM_ShenandoahReferenceOperation::doit_epilogue() {
38   if (Universe::has_reference_pending_list()) {
39     Heap_lock->notify_all();
40   }
41   Heap_lock->unlock();
42 }
