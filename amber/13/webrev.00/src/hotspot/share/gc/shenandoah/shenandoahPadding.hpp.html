25 #ifndef SHARE_GC_SHENANDOAH_SHENANDOAHPADDING_INLINE_HPP
26 #define SHARE_GC_SHENANDOAH_SHENANDOAHPADDING_INLINE_HPP
27 
28 // 64 bytes is enough to cover all existing architectures. If we have some
29 // other platforms, we would need to provide the architecture-specific
30 // versions here. Shared code provides DEFAULT_CACHE_LINE_SIZE, which is
31 // inconveniently large by default.
32 
33 #define SHENANDOAH_CACHE_LINE_SIZE 64
34 
35 #define shenandoah_padding(id) \
36   DEFINE_PAD_MINUS_SIZE(id, SHENANDOAH_CACHE_LINE_SIZE, 0)
37 
38 #endif // SHARE_GC_SHENANDOAH_SHENANDOAHPADDING_INLINE_HPP
