25 #ifndef SHARE_GC_G1_HEAPREGIONBOUNDS_HPP
26 #define SHARE_GC_G1_HEAPREGIONBOUNDS_HPP
27 
28 #include "memory/allocation.hpp"
29 
30 class HeapRegionBounds : public AllStatic {
31 private:
32   // Minimum region size; we won't go lower than that.
33   // We might want to decrease this in the future, to deal with small
34   // heaps a bit more efficiently.
35   static const size_t MIN_REGION_SIZE = 1024 * 1024;
36 
37   // Maximum region size; we don't go higher than that. There's a good
38   // reason for having an upper bound. We don't want regions to get too
39   // large, otherwise cleanup's effectiveness would decrease as there
40   // will be fewer opportunities to find totally empty regions after
41   // marking.
42   static const size_t MAX_REGION_SIZE = 32 * 1024 * 1024;
43 
44   // The automatic region size calculation will try to have around this
45   // many regions in the heap.
46   static const size_t TARGET_REGION_NUMBER = 2048;
47 
48 public:
49   static inline size_t min_size();
50   static inline size_t max_size();
51   static inline size_t target_number();
52 };
53 
54 #endif // SHARE_GC_G1_HEAPREGIONBOUNDS_HPP
