  73 bool G1CMBitMapClosure::do_addr(HeapWord* const addr) {
  74   assert(addr < _cm->finger(), "invariant");
  75   assert(addr >= _task->finger(), "invariant");
  76 
  77   // We move that task's local finger along.
  78   _task->move_finger_to(addr);
  79 
  80   _task->scan_task_entry(G1TaskQueueEntry::from_oop(oop(addr)));
  81   // we only partially drain the local queue and global stack
  82   _task->drain_local_queue(true);
  83   _task->drain_global_stack(true);
  84 
  85   // if the has_aborted flag has been raised, we need to bail out of
  86   // the iteration
  87   return !_task->has_aborted();
  88 }

  97 bool G1CMMarkStack::resize(size_t new_capacity) {
  98   assert(is_empty(), "Only resize when stack is empty.");
  99   assert(new_capacity <= _max_chunk_capacity,
 100          "Trying to resize stack to " SIZE_FORMAT " chunks when the maximum is " SIZE_FORMAT, new_capacity, _max_chunk_capacity);
 101 
 102   TaskQueueEntryChunk* new_base = MmapArrayAllocator<TaskQueueEntryChunk>::allocate_or_null(new_capacity, mtGC);
 103 
 104   if (new_base == NULL) {
 105     log_warning(gc)("Failed to reserve memory for new overflow mark stack with " SIZE_FORMAT " chunks and size " SIZE_FORMAT "B.", new_capacity, new_capacity * sizeof(TaskQueueEntryChunk));
 106     return false;
 107   }
 108   // Release old mapping.
 109   if (_base != NULL) {
 110     MmapArrayAllocator<TaskQueueEntryChunk>::free(_base, _chunk_capacity);
 111   }
 112 
 113   _base = new_base;
 114   _chunk_capacity = new_capacity;
 115   set_empty();
 116 
 117   return true;
 118 }

 120 size_t G1CMMarkStack::capacity_alignment() {
 121   return (size_t)lcm(os::vm_allocation_granularity(), sizeof(TaskQueueEntryChunk)) / sizeof(G1TaskQueueEntry);

 124 bool G1CMMarkStack::initialize(size_t initial_capacity, size_t max_capacity) {
 125   guarantee(_max_chunk_capacity == 0, "G1CMMarkStack already initialized.");
 126 
 127   size_t const TaskEntryChunkSizeInVoidStar = sizeof(TaskQueueEntryChunk) / sizeof(G1TaskQueueEntry);
 128 
 129   _max_chunk_capacity = align_up(max_capacity, capacity_alignment()) / TaskEntryChunkSizeInVoidStar;
 130   size_t initial_chunk_capacity = align_up(initial_capacity, capacity_alignment()) / TaskEntryChunkSizeInVoidStar;
 131 
 132   guarantee(initial_chunk_capacity <= _max_chunk_capacity,
 133             "Maximum chunk capacity " SIZE_FORMAT " smaller than initial capacity " SIZE_FORMAT,
 134             _max_chunk_capacity,
 135             initial_chunk_capacity);
 136 
 137   log_debug(gc)("Initialize mark stack with " SIZE_FORMAT " chunks, maximum " SIZE_FORMAT,
 138                 initial_chunk_capacity, _max_chunk_capacity);
 139 
 140   return resize(initial_chunk_capacity);
 141 }

 143 void G1CMMarkStack::expand() {
 144   if (_chunk_capacity == _max_chunk_capacity) {
 145     log_debug(gc)("Can not expand overflow mark stack further, already at maximum capacity of " SIZE_FORMAT " chunks.", _chunk_capacity);
 146     return;
 147   }
 148   size_t old_capacity = _chunk_capacity;
 149   // Double capacity if possible
 150   size_t new_capacity = MIN2(old_capacity * 2, _max_chunk_capacity);
 151 
 152   if (resize(new_capacity)) {
 153     log_debug(gc)("Expanded mark stack capacity from " SIZE_FORMAT " to " SIZE_FORMAT " chunks",
 154                   old_capacity, new_capacity);
 155   } else {
 156     log_warning(gc)("Failed to expand mark stack capacity from " SIZE_FORMAT " to " SIZE_FORMAT " chunks",
 157                     old_capacity, new_capacity);
 158   }
 159 }

 193   TaskQueueEntryChunk* result = remove_chunk_from_list(&_chunk_list);
 194   if (result != NULL) {
 195     _chunks_in_chunk_list--;
 196   }
 197   return result;

 205 G1CMMarkStack::TaskQueueEntryChunk* G1CMMarkStack::allocate_new_chunk() {
 206   // This dirty read of _hwm is okay because we only ever increase the _hwm in parallel code.
 207   // Further this limits _hwm to a value of _chunk_capacity + #threads, avoiding
 208   // wraparound of _hwm.
 209   if (_hwm >= _chunk_capacity) {
 210     return NULL;
 211   }
 212 
 213   size_t cur_idx = Atomic::fetch_and_add(&_hwm, 1u);
 214   if (cur_idx >= _chunk_capacity) {
 215     return NULL;
 216   }
 217 
 218   TaskQueueEntryChunk* result = ::new (&_base[cur_idx]) TaskQueueEntryChunk;
 219   result->next = NULL;
 220   return result;
 221 }

 223 bool G1CMMarkStack::par_push_chunk(G1TaskQueueEntry* ptr_arr) {
 224   // Get a new chunk.
 225   TaskQueueEntryChunk* new_chunk = remove_chunk_from_free_list();
 226 
 227   if (new_chunk == NULL) {
 228     // Did not get a chunk from the free list. Allocate from backing memory.
 229     new_chunk = allocate_new_chunk();
 230 
 231     if (new_chunk == NULL) {
 232       return false;
 233     }
 234   }
 235 
 236   Copy::conjoint_memory_atomic(ptr_arr, new_chunk->data, EntriesPerChunk * sizeof(G1TaskQueueEntry));
 237 
 238   add_chunk_to_chunk_list(new_chunk);
 239 
 240   return true;
 241 }
 242 
 243 bool G1CMMarkStack::par_pop_chunk(G1TaskQueueEntry* ptr_arr) {
 244   TaskQueueEntryChunk* cur = remove_chunk_from_chunk_list();
 245 
 246   if (cur == NULL) {
 247     return false;
 248   }
 249 
 250   Copy::conjoint_memory_atomic(cur->data, ptr_arr, EntriesPerChunk * sizeof(G1TaskQueueEntry));
 251 
 252   add_chunk_to_free_list(cur);
 253   return true;
 254 }

 298 const MemRegion* G1CMRootMemRegions::claim_next() {
 299   if (_should_abort) {
 300     // If someone has set the should_abort flag, we return NULL to
 301     // force the caller to bail out of their loop.
 302     return NULL;
 303   }
 304 
 305   if (_claimed_root_regions >= _num_root_regions) {
 306     return NULL;
 307   }
 308 
 309   size_t claimed_index = Atomic::fetch_and_add(&_claimed_root_regions, 1u);
 310   if (claimed_index < _num_root_regions) {
 311     return &_root_regions[claimed_index];
 312   }
 313   return NULL;
 314 }

 320 void G1CMRootMemRegions::notify_scan_done() {
 321   MutexLocker x(RootRegionScan_lock, Mutex::_no_safepoint_check_flag);
 322   _scan_in_progress = false;
 323   RootRegionScan_lock->notify_all();
 324 }
 325 
 326 void G1CMRootMemRegions::cancel_scan() {
 327   notify_scan_done();
 328 }
 329 
 330 void G1CMRootMemRegions::scan_finished() {
 331   assert(scan_in_progress(), "pre-condition");
 332 
 333   if (!_should_abort) {
 334     assert(_claimed_root_regions >= num_root_regions(),
 335            "we should have claimed all root regions, claimed " SIZE_FORMAT ", length = %u",
 336            _claimed_root_regions, num_root_regions());
 337   }
 338 
 339   notify_scan_done();
 340 }

 342 bool G1CMRootMemRegions::wait_until_scan_finished() {
 343   if (!scan_in_progress()) {
 344     return false;
 345   }
 346 
 347   {
 348     MonitorLocker ml(RootRegionScan_lock, Mutex::_no_safepoint_check_flag);
 349     while (scan_in_progress()) {
 350       ml.wait();
 351     }
 352   }
 353   return true;
 354 }

 454 void G1ConcurrentMark::reset() {
 455   _has_aborted = false;
 456 
 457   reset_marking_for_restart();
 458 
 459   // Reset all tasks, since different phases will use different number of active
 460   // threads. So, it's easiest to have all of them ready.
 461   for (uint i = 0; i < _max_num_tasks; ++i) {
 462     _tasks[i]->reset(_next_mark_bitmap);
 463   }
 464 
 465   uint max_regions = _g1h->max_regions();
 466   for (uint i = 0; i < max_regions; i++) {
 467     _top_at_rebuild_starts[i] = NULL;
 468     _region_mark_stats[i].clear();
 469   }
 470 }

 514 void G1ConcurrentMark::reset_marking_for_restart() {
 515   _global_mark_stack.set_empty();
 516 
 517   // Expand the marking stack, if we have to and if we can.
 518   if (has_overflown()) {
 519     _global_mark_stack.expand();
 520 
 521     uint max_regions = _g1h->max_regions();
 522     for (uint i = 0; i < max_regions; i++) {
 523       _region_mark_stats[i].clear_during_overflow();
 524     }
 525   }
 526 
 527   clear_has_overflown();
 528   _finger = _heap.start();
 529 
 530   for (uint i = 0; i < _max_num_tasks; ++i) {
 531     G1CMTaskQueue* queue = _task_queues->queue(i);
 532     queue->set_empty();
 533   }
 534 }

 536 void G1ConcurrentMark::set_concurrency(uint active_tasks) {
 537   assert(active_tasks <= _max_num_tasks, "we should not have more");
 538 
 539   _num_active_tasks = active_tasks;
 540   // Need to update the three data structures below according to the
 541   // number of active threads for this phase.
 542   _terminator.reset_for_reuse(active_tasks);
 543   _first_overflow_barrier_sync.set_n_workers((int) active_tasks);
 544   _second_overflow_barrier_sync.set_n_workers((int) active_tasks);
 545 }

 547 void G1ConcurrentMark::set_concurrency_and_phase(uint active_tasks, bool concurrent) {

 556            "only way to get here: _finger: " PTR_FORMAT ", _heap_end: " PTR_FORMAT,

 558   }

 575 class G1ClearBitMapTask : public AbstractGangTask {
 576 public:
 577   static size_t chunk_size() { return M; }
 578 
 579 private:
 580   // Heap region closure used for clearing the given mark bitmap.
 581   class G1ClearBitmapHRClosure : public HeapRegionClosure {
 582   private:
 583     G1CMBitMap* _bitmap;
 584     G1ConcurrentMark* _cm;
 585   public:
 586     G1ClearBitmapHRClosure(G1CMBitMap* bitmap, G1ConcurrentMark* cm) : HeapRegionClosure(), _bitmap(bitmap), _cm(cm) {
 587     }
 588 
 589     virtual bool do_heap_region(HeapRegion* r) {
 590       size_t const chunk_size_in_words = G1ClearBitMapTask::chunk_size() / HeapWordSize;
 591 
 592       HeapWord* cur = r->bottom();
 593       HeapWord* const end = r->end();
 594 
 595       while (cur < end) {
 596         MemRegion mr(cur, MIN2(cur + chunk_size_in_words, end));
 597         _bitmap->clear_range(mr);
 598 
 599         cur += chunk_size_in_words;
 600 
 601         // Abort iteration if after yielding the marking has been aborted.

 610         assert(_cm == NULL || !G1CollectedHeap::heap()->collector_state()->mark_or_rebuild_in_progress(), "invariant");
 611       }
 612       assert(cur == end, "Must have completed iteration over the bitmap for region %u.", r->hrm_index());
 613 
 614       return false;
 615     }

 657   // the current cycle.

 700 void G1ConcurrentMark::post_initial_mark() {
 701   // Start Concurrent Marking weak-reference discovery.
 702   ReferenceProcessor* rp = _g1h->ref_processor_cm();
 703   // enable ("weak") refs discovery
 704   rp->enable_discovery();
 705   rp->setup_policy(false); // snapshot the soft ref policy to be used in this cycle
 706 
 707   SATBMarkQueueSet& satb_mq_set = G1BarrierSet::satb_mark_queue_set();
 708   // This is the start of  the marking cycle, we're expected all
 709   // threads to have SATB queues with active set to false.
 710   satb_mq_set.set_active_all_threads(true, /* new active value */
 711                                      false /* expected_active */);
 712 
 713   _root_regions.prepare_for_scan();
 714 
 715   // update_g1_committed() will be called at the end of an evac pause
 716   // when marking is on. So, it's also called at the end of the
 717   // initial-mark pause to update the heap end, if the heap expands
 718   // during it. No need to call it here.
 719 }

 721 /*
 722  * Notice that in the next two methods, we actually leave the STS
 723  * during the barrier sync and join it immediately afterwards. If we
 724  * do not do this, the following deadlock can occur: one thread could
 725  * be in the barrier sync code, waiting for the other thread to also
 726  * sync up, whereas another one could be trying to yield, while also
 727  * waiting for the other threads to sync up too.
 728  *
 729  * Note, however, that this code is also used during remark and in
 730  * this case we should not attempt to leave / enter the STS, otherwise
 731  * we'll either hit an assert (debug / fastdebug) or deadlock
 732  * (product). So we should only leave / enter the STS if we are
 733  * operating concurrently.
 734  *
 735  * Because the thread that does the sync barrier has left the STS, it
 736  * is possible to be suspended for a Full GC or an evacuation pause
 737  * could occur. This is actually safe, since the entering the sync
 738  * barrier is one of the last things do_marking_step() does, and it
 739  * doesn't manipulate any data structures afterwards.
 740  */
 741 
 742 void G1ConcurrentMark::enter_first_sync_barrier(uint worker_id) {
 743   bool barrier_aborted;
 744   {
 745     SuspendibleThreadSetLeaver sts_leave(concurrent());
 746     barrier_aborted = !_first_overflow_barrier_sync.enter();
 747   }
 748 
 749   // at this point everyone should have synced up and not be doing any
 750   // more work
 751 
 752   if (barrier_aborted) {
 753     // If the barrier aborted we ignore the overflow condition and
 754     // just abort the whole marking phase as quickly as possible.
 755     return;
 756   }
 757 }

 760   SuspendibleThreadSetLeaver sts_leave(concurrent());
 761   _second_overflow_barrier_sync.enter();
 762 
 763   // at this point everything should be re-initialized and ready to go
 764 }

 825 void G1ConcurrentMark::scan_root_region(const MemRegion* region, uint worker_id) {
 826 #ifdef ASSERT
 827   HeapWord* last = region->last();

 832          "MemRegion start should be equal to nTAMS");

 836 
 837   const uintx interval = PrefetchScanIntervalInBytes;

 965     G1CollectedHeap* _g1h;
 966     G1ConcurrentMark* _cm;
 967 
 968     G1PrintRegionLivenessInfoClosure* _cl;
 969 
 970     uint _num_regions_selected_for_rebuild;  // The number of regions actually selected for rebuild.
 971 
 972     void update_remset_before_rebuild(HeapRegion* hr) {
 973       G1RemSetTrackingPolicy* tracking_policy = _g1h->policy()->remset_tracker();
 974 
 975       bool selected_for_rebuild;
 976       if (hr->is_humongous()) {
 977         bool const is_live = _cm->liveness(hr->humongous_start_region()->hrm_index()) > 0;
 978         selected_for_rebuild = tracking_policy->update_humongous_before_rebuild(hr, is_live);
 979       } else {
 980         size_t const live_bytes = _cm->liveness(hr->hrm_index());
 981         selected_for_rebuild = tracking_policy->update_before_rebuild(hr, live_bytes);
 982       }
 983       if (selected_for_rebuild) {
 984         _num_regions_selected_for_rebuild++;
 985       }
 986       _cm->update_top_at_rebuild_start(hr);
 987     }
 988 
 989     // Distribute the given words across the humongous object starting with hr and
 990     // note end of marking.
 991     void distribute_marked_bytes(HeapRegion* hr, size_t marked_words) {
 992       uint const region_idx = hr->hrm_index();
 993       size_t const obj_size_in_words = (size_t)oop(hr->bottom())->size();
 994       uint const num_regions_in_humongous = (uint)G1CollectedHeap::humongous_obj_size_in_regions(obj_size_in_words);
 995 
 996       // "Distributing" zero words means that we only note end of marking for these
 997       // regions.
 998       assert(marked_words == 0 || obj_size_in_words == marked_words,
 999              "Marked words should either be 0 or the same as humongous object (" SIZE_FORMAT ") but is " SIZE_FORMAT,
1000              obj_size_in_words, marked_words);
1001 
1002       for (uint i = region_idx; i < (region_idx + num_regions_in_humongous); i++) {
1003         HeapRegion* const r = _g1h->region_at(i);
1004         size_t const words_to_add = MIN2(HeapRegion::GrainWords, marked_words);
1005 
1006         log_trace(gc, marking)("Adding " SIZE_FORMAT " words to humongous region %u (%s)",
1007                                words_to_add, i, r->get_type_str());
1008         add_marked_bytes_and_note_end(r, words_to_add * HeapWordSize);
1009         marked_words -= words_to_add;
1010       }
1011       assert(marked_words == 0,
1012              SIZE_FORMAT " words left after distributing space across %u regions",
1013              marked_words, num_regions_in_humongous);
1014     }
1015 
1016     void update_marked_bytes(HeapRegion* hr) {
1017       uint const region_idx = hr->hrm_index();
1018       size_t const marked_words = _cm->liveness(region_idx);
1019       // The marking attributes the object's size completely to the humongous starts
1020       // region. We need to distribute this value across the entire set of regions a
1021       // humongous object spans.
1022       if (hr->is_humongous()) {
1023         assert(hr->is_starts_humongous() || marked_words == 0,
1024                "Should not have marked words " SIZE_FORMAT " in non-starts humongous region %u (%s)",
1025                marked_words, region_idx, hr->get_type_str());
1026         if (hr->is_starts_humongous()) {
1027           distribute_marked_bytes(hr, marked_words);
1028         }
1029       } else {
1030         log_trace(gc, marking)("Adding " SIZE_FORMAT " words to region %u (%s)", marked_words, region_idx, hr->get_type_str());
1031         add_marked_bytes_and_note_end(hr, marked_words * HeapWordSize);
1032       }
1033     }
1034 
1035     void add_marked_bytes_and_note_end(HeapRegion* hr, size_t marked_bytes) {
1036       hr->add_to_marked_bytes(marked_bytes);
1037       _cl->do_heap_region(hr);
1038       hr->note_end_of_marking();
1039     }

1042     G1UpdateRemSetTrackingBeforeRebuild(G1CollectedHeap* g1h, G1ConcurrentMark* cm, G1PrintRegionLivenessInfoClosure* cl) :

1046       update_remset_before_rebuild(r);
1047       update_marked_bytes(r);
1048 
1049       return false;
1050     }

1053   };

1083 void G1ConcurrentMark::remark() {
1084   assert_at_safepoint_on_vm_thread();
1085 
1086   // If a full collection has happened, we should not continue. However we might
1087   // have ended up here as the Remark VM operation has been scheduled already.
1088   if (has_aborted()) {
1089     return;
1090   }
1091 
1092   G1Policy* policy = _g1h->policy();
1093   policy->record_concurrent_mark_remark_start();
1094 
1095   double start = os::elapsedTime();
1096 
1097   verify_during_pause(G1HeapVerifier::G1VerifyRemark, VerifyOption_G1UsePrevMarking, "Remark before");
1098 
1099   {
1100     GCTraceTime(Debug, gc, phases) debug("Finalize Marking", _gc_timer_cm);
1101     finalize_marking();
1102   }
1103 
1104   double mark_work_end = os::elapsedTime();
1105 
1106   bool const mark_finished = !has_overflown();
1107   if (mark_finished) {
1108     weak_refs_work(false /* clear_all_soft_refs */);
1109 
1110     SATBMarkQueueSet& satb_mq_set = G1BarrierSet::satb_mark_queue_set();
1111     // We're done with marking.
1112     // This is the end of the marking cycle, we're expected all
1113     // threads to have SATB queues with active set to true.
1114     satb_mq_set.set_active_all_threads(false, /* new active value */
1115                                        true /* expected_active */);
1116 
1117     {
1118       GCTraceTime(Debug, gc, phases) debug("Flush Task Caches", _gc_timer_cm);
1119       flush_all_task_caches();
1120     }
1121 
1122     // Install newly created mark bitmap as "prev".
1123     swap_mark_bitmaps();
1124     {
1125       GCTraceTime(Debug, gc, phases) debug("Update Remembered Set Tracking Before Rebuild", _gc_timer_cm);
1126 
1127       uint const workers_by_capacity = (_g1h->num_regions() + G1UpdateRemSetTrackingBeforeRebuildTask::RegionsPerThread - 1) /
1128                                        G1UpdateRemSetTrackingBeforeRebuildTask::RegionsPerThread;
1129       uint const num_workers = MIN2(_g1h->workers()->active_workers(), workers_by_capacity);
1130 
1131       G1UpdateRemSetTrackingBeforeRebuildTask cl(_g1h, this, num_workers);
1132       log_debug(gc,ergo)("Running %s using %u workers for %u regions in heap", cl.name(), num_workers, _g1h->num_regions());
1133       _g1h->workers()->run_task(&cl, num_workers);
1134 
1135       log_debug(gc, remset, tracking)("Remembered Set Tracking update regions total %u, selected %u",
1136                                       _g1h->num_regions(), cl.total_selected_for_rebuild());
1137     }
1138     {
1139       GCTraceTime(Debug, gc, phases) debug("Reclaim Empty Regions", _gc_timer_cm);
1140       reclaim_empty_regions();
1141     }

1333 // 'Keep Alive' oop closure used by both serial parallel reference processing.
1334 // Uses the G1CMTask associated with a worker thread (for serial reference
1335 // processing the G1CMTask for worker 0 is used) to preserve (mark) and
1336 // trace referent objects.
1337 //
1338 // Using the G1CMTask and embedded local queues avoids having the worker
1339 // threads operating on the global mark stack. This reduces the risk
1340 // of overflowing the stack - which we would rather avoid at this late
1341 // state. Also using the tasks' local queues removes the potential
1342 // of the workers interfering with each other that could occur if
1343 // operating on the global stack.
1344 
1345 class G1CMKeepAliveAndDrainClosure : public OopClosure {
1346   G1ConcurrentMark* _cm;
1347   G1CMTask*         _task;
1348   uint              _ref_counter_limit;
1349   uint              _ref_counter;
1350   bool              _is_serial;
1351 public:
1352   G1CMKeepAliveAndDrainClosure(G1ConcurrentMark* cm, G1CMTask* task, bool is_serial) :
1353     _cm(cm), _task(task), _ref_counter_limit(G1RefProcDrainInterval),
1354     _ref_counter(_ref_counter_limit), _is_serial(is_serial) {
1355     assert(!_is_serial || _task->worker_id() == 0, "only task 0 for serial code");
1356   }
1357 
1358   virtual void do_oop(narrowOop* p) { do_oop_work(p); }
1359   virtual void do_oop(      oop* p) { do_oop_work(p); }
1360 
1361   template <class T> void do_oop_work(T* p) {
1362     if (_cm->has_overflown()) {
1363       return;
1364     }
1365     if (!_task->deal_with_reference(p)) {
1366       // We did not add anything to the mark bitmap (or mark stack), so there is
1367       // no point trying to drain it.
1368       return;
1369     }
1370     _ref_counter--;
1371 
1372     if (_ref_counter == 0) {
1373       // We have dealt with _ref_counter_limit references, pushing them
1374       // and objects reachable from them on to the local stack (and
1375       // possibly the global stack). Call G1CMTask::do_marking_step() to
1376       // process these entries.
1377       //
1378       // We call G1CMTask::do_marking_step() in a loop, which we'll exit if
1379       // there's nothing more to do (i.e. we're done with the entries that
1380       // were pushed as a result of the G1CMTask::deal_with_reference() calls
1381       // above) or we overflow.
1382       //
1383       // Note: G1CMTask::do_marking_step() can set the G1CMTask::has_aborted()
1384       // flag while there may still be some work to do. (See the comment at
1385       // the beginning of G1CMTask::do_marking_step() for those conditions -
1386       // one of which is reaching the specified time target.) It is only
1387       // when G1CMTask::do_marking_step() returns without setting the
1388       // has_aborted() flag that the marking step has completed.
1389       do {
1390         double mark_step_duration_ms = G1ConcMarkStepDurationMillis;
1391         _task->do_marking_step(mark_step_duration_ms,
1392                                false      /* do_termination */,
1393                                _is_serial);
1394       } while (_task->has_aborted() && !_cm->has_overflown());
1395       _ref_counter = _ref_counter_limit;
1396     }
1397   }
1398 };

1400 // 'Drain' oop closure used by both serial and parallel reference processing.
1401 // Uses the G1CMTask associated with a given worker thread (for serial
1402 // reference processing the G1CMtask for worker 0 is used). Calls the
1403 // do_marking_step routine, with an unbelievably large timeout value,
1404 // to drain the marking data structures of the remaining entries
1405 // added by the 'keep alive' oop closure above.
1406 
1407 class G1CMDrainMarkingStackClosure : public VoidClosure {
1408   G1ConcurrentMark* _cm;
1409   G1CMTask*         _task;
1410   bool              _is_serial;
1411  public:
1412   G1CMDrainMarkingStackClosure(G1ConcurrentMark* cm, G1CMTask* task, bool is_serial) :
1413     _cm(cm), _task(task), _is_serial(is_serial) {
1414     assert(!_is_serial || _task->worker_id() == 0, "only task 0 for serial code");
1415   }
1416 
1417   void do_void() {
1418     do {
1419       // We call G1CMTask::do_marking_step() to completely drain the local
1420       // and global marking stacks of entries pushed by the 'keep alive'
1421       // oop closure (an instance of G1CMKeepAliveAndDrainClosure above).
1422       //
1423       // G1CMTask::do_marking_step() is called in a loop, which we'll exit
1424       // if there's nothing more to do (i.e. we've completely drained the
1425       // entries that were pushed as a a result of applying the 'keep alive'
1426       // closure to the entries on the discovered ref lists) or we overflow
1427       // the global marking stack.
1428       //
1429       // Note: G1CMTask::do_marking_step() can set the G1CMTask::has_aborted()
1430       // flag while there may still be some work to do. (See the comment at
1431       // the beginning of G1CMTask::do_marking_step() for those conditions -
1432       // one of which is reaching the specified time target.) It is only
1433       // when G1CMTask::do_marking_step() returns without setting the
1434       // has_aborted() flag that the marking step has completed.
1435 
1436       _task->do_marking_step(1000000000.0 /* something very large */,
1437                              true         /* do_termination */,
1438                              _is_serial);
1439     } while (_task->has_aborted() && !_cm->has_overflown());
1440   }
1441 };

1831 HeapRegion*
1832 G1ConcurrentMark::claim_region(uint worker_id) {
1833   // "checkpoint" the finger
1834   HeapWord* finger = _finger;
1835 
1836   while (finger < _heap.end()) {
1837     assert(_g1h->is_in_g1_reserved(finger), "invariant");
1838 
1839     HeapRegion* curr_region = _g1h->heap_region_containing(finger);
1840     // Make sure that the reads below do not float before loading curr_region.
1841     OrderAccess::loadload();
1842     // Above heap_region_containing may return NULL as we always scan claim
1843     // until the end of the heap. In this case, just jump to the next region.
1844     HeapWord* end = curr_region != NULL ? curr_region->end() : finger + HeapRegion::GrainWords;
1845 
1846     // Is the gap between reading the finger and doing the CAS too long?
1847     HeapWord* res = Atomic::cmpxchg(&_finger, finger, end);
1848     if (res == finger && curr_region != NULL) {
1849       // we succeeded
1850       HeapWord*   bottom        = curr_region->bottom();
1851       HeapWord*   limit         = curr_region->next_top_at_mark_start();
1852 
1853       // notice that _finger == end cannot be guaranteed here since,
1854       // someone else might have moved the finger even further
1855       assert(_finger >= end, "the finger should have moved forward");
1856 
1857       if (limit > bottom) {
1858         return curr_region;
1859       } else {
1860         assert(limit == bottom,
1861                "the region limit should be at bottom");
1862         // we return NULL and the caller should try calling
1863         // claim_region() again.
1864         return NULL;
1865       }
1866     } else {
1867       assert(_finger > finger, "the finger should have moved forward");
1868       // read it again
1869       finger = _finger;
1870     }
1871   }
1872 
1873   return NULL;
1874 }

1962 void G1ConcurrentMark::concurrent_cycle_abort() {
1963   if (!cm_thread()->during_cycle() || _has_aborted) {
1964     // We haven't started a concurrent cycle or we have already aborted it. No need to do anything.
1965     return;
1966   }
1967 
1968   // Clear all marks in the next bitmap for the next marking cycle. This will allow us to skip the next
1969   // concurrent bitmap clearing.
1970   {
1971     GCTraceTime(Debug, gc) debug("Clear Next Bitmap");
1972     clear_bitmap(_next_mark_bitmap, _g1h->workers(), false);
1973   }
1974   // Note we cannot clear the previous marking bitmap here
1975   // since VerifyDuringGC verifies the objects marked during
1976   // a full GC against the previous bitmap.
1977 
1978   // Empty mark stack
1979   reset_marking_for_restart();
1980   for (uint i = 0; i < _max_num_tasks; ++i) {
1981     _tasks[i]->clear_region_fields();
1982   }
1983   _first_overflow_barrier_sync.abort();
1984   _second_overflow_barrier_sync.abort();
1985   _has_aborted = true;
1986 
1987   SATBMarkQueueSet& satb_mq_set = G1BarrierSet::satb_mark_queue_set();
1988   satb_mq_set.abandon_partial_marking();
1989   // This can be called either during or outside marking, we'll read
1990   // the expected_active value from the SATB queue set.
1991   satb_mq_set.set_active_all_threads(
1992                                  false, /* new active value */
1993                                  satb_mq_set.is_active() /* expected_active */);
1994 }

2064 void G1CMTask::update_region_limit() {
2065   HeapRegion* hr            = _curr_region;
2066   HeapWord* bottom          = hr->bottom();
2067   HeapWord* limit           = hr->next_top_at_mark_start();
2068 
2069   if (limit == bottom) {
2070     // The region was collected underneath our feet.
2071     // We set the finger to bottom to ensure that the bitmap
2072     // iteration that will follow this will not do anything.
2073     // (this is not a condition that holds when we set the region up,
2074     // as the region is not supposed to be empty in the first place)
2075     _finger = bottom;
2076   } else if (limit >= _region_limit) {
2077     assert(limit >= _finger, "peace of mind");
2078   } else {
2079     assert(limit < _region_limit, "only way to get here");
2080     // This can happen under some pretty unusual circumstances.  An
2081     // evacuation pause empties the region underneath our feet (NTAMS
2082     // at bottom). We then do some allocation in the region (NTAMS
2083     // stays at bottom), followed by the region being used as a GC
2084     // alloc region (NTAMS will move to top() and the objects
2085     // originally below it will be grayed). All objects now marked in
2086     // the region are explicitly grayed, if below the global finger,
2087     // and we do not need in fact to scan anything else. So, we simply
2088     // set _finger to be limit to ensure that the bitmap iteration
2089     // doesn't do anything.
2090     _finger = limit;
2091   }
2092 
2093   _region_limit = limit;
2094 }

2149 bool G1CMTask::regular_clock_call() {
2150   if (has_aborted()) {
2151     return false;
2152   }
2153 
2154   // First, we need to recalculate the words scanned and refs reached
2155   // limits for the next clock call.
2156   recalculate_limits();
2157 
2158   // During the regular clock call we do the following
2159 
2160   // (1) If an overflow has been flagged, then we abort.
2161   if (_cm->has_overflown()) {
2162     return false;
2163   }
2164 
2165   // If we are not concurrent (i.e. we're doing remark) we don't need
2166   // to check anything else. The other steps are only needed during
2167   // the concurrent marking phase.
2168   if (!_cm->concurrent()) {
2169     return true;
2170   }
2171 
2172   // (2) If marking has been aborted for Full GC, then we also abort.
2173   if (_cm->has_aborted()) {
2174     return false;
2175   }
2176 
2177   double curr_time_ms = os::elapsedVTime() * 1000.0;
2178 
2179   // (4) We check whether we should yield. If we have to, then we abort.
2180   if (SuspendibleThreadSet::should_yield()) {
2181     // We should yield. To do this we abort the task. The caller is
2182     // responsible for yielding.
2183     return false;
2184   }
2185 
2186   // (5) We check whether we've reached our time quota. If we have,
2187   // then we abort.
2188   double elapsed_time_ms = curr_time_ms - _start_time_ms;
2189   if (elapsed_time_ms > _time_target_ms) {
2190     _has_timed_out = true;
2191     return false;
2192   }
2193 
2194   // (6) Finally, we check whether there are enough completed STAB
2195   // buffers available for processing. If there are, we abort.
2196   SATBMarkQueueSet& satb_mq_set = G1BarrierSet::satb_mark_queue_set();
2197   if (!_draining_satb_buffers && satb_mq_set.process_completed_buffers()) {
2198     // we do need to process SATB buffers, we'll abort and restart
2199     // the marking task to do so
2200     return false;
2201   }
2202   return true;
2203 }

2223 void G1CMTask::move_entries_to_global_stack() {
2224   // Local array where we'll store the entries that will be popped
2225   // from the local queue.
2226   G1TaskQueueEntry buffer[G1CMMarkStack::EntriesPerChunk];
2227 
2228   size_t n = 0;
2229   G1TaskQueueEntry task_entry;
2230   while (n < G1CMMarkStack::EntriesPerChunk && _task_queue->pop_local(task_entry)) {
2231     buffer[n] = task_entry;
2232     ++n;
2233   }
2234   if (n < G1CMMarkStack::EntriesPerChunk) {
2235     buffer[n] = G1TaskQueueEntry();
2236   }
2237 
2238   if (n > 0) {
2239     if (!_cm->mark_stack_push(buffer)) {

2400 /*****************************************************************************
2401 
2402     The do_marking_step(time_target_ms, ...) method is the building
2403     block of the parallel marking framework. It can be called in parallel
2404     with other invocations of do_marking_step() on different tasks
2405     (but only one per task, obviously) and concurrently with the
2406     mutator threads, or during remark, hence it eliminates the need
2407     for two versions of the code. When called during remark, it will
2408     pick up from where the task left off during the concurrent marking
2409     phase. Interestingly, tasks are also claimable during evacuation
2410     pauses too, since do_marking_step() ensures that it aborts before
2411     it needs to yield.
2412 
2413     The data structures that it uses to do marking work are the
2414     following:
2415 
2416       (1) Marking Bitmap. If there are gray objects that appear only
2417       on the bitmap (this happens either when dealing with an overflow
2418       or when the initial marking phase has simply marked the roots
2419       and didn't push them on the stack), then tasks claim heap
2420       regions whose bitmap they then scan to find gray objects. A
2421       global finger indicates where the end of the last claimed region
2422       is. A local finger indicates how far into the region a task has
2423       scanned. The two fingers are used to determine how to gray an
2424       object (i.e. whether simply marking it is OK, as it will be
2425       visited by a task in the future, or whether it needs to be also
2426       pushed on a stack).
2427 
2428       (2) Local Queue. The local queue of the task which is accessed
2429       reasonably efficiently by the task. Other tasks can steal from
2430       it when they run out of work. Throughout the marking phase, a
2431       task attempts to keep its local queue short but not totally
2432       empty, so that entries are available for stealing by other
2433       tasks. Only when there is no more work, a task will totally
2434       drain its local queue.
2435 
2436       (3) Global Mark Stack. This handles local queue overflow. During
2437       marking only sets of entries are moved between it and the local
2438       queues, as access to it requires a mutex and more fine-grain
2439       interaction with it which might cause contention. If it
2440       overflows, then the marking phase should restart and iterate
2441       over the bitmap to identify gray objects. Throughout the marking
2442       phase, tasks attempt to keep the global mark stack at a small
2443       length but not totally empty, so that entries are available for
2444       popping by other tasks. Only when there is no more work, tasks
2445       will totally drain the global mark stack.
2446 
2447       (4) SATB Buffer Queue. This is where completed SATB buffers are
2448       made available. Buffers are regularly removed from this queue
2449       and scanned for roots, so that the queue doesn't get too
2450       long. During remark, all completed buffers are processed, as
2451       well as the filled in parts of any uncompleted buffers.
2452 
2453     The do_marking_step() method tries to abort when the time target
2454     has been reached. There are a few other cases when the
2455     do_marking_step() method also aborts:
2456 
2457       (1) When the marking phase has been aborted (after a Full GC).
2458 
2459       (2) When a global overflow (on the global stack) has been
2460       triggered. Before the task aborts, it will actually sync up with
2461       the other tasks to ensure that all the marking data structures
2462       (local queues, stacks, fingers etc.)  are re-initialized so that
2463       when do_marking_step() completes, the marking phase can
2464       immediately restart.
2465 
2466       (3) When enough completed SATB buffers are available. The
2467       do_marking_step() method only tries to drain SATB buffers right
2468       at the beginning. So, if enough buffers are available, the
2469       marking step aborts and the SATB buffers are processed at
2470       the beginning of the next invocation.
2471 
2472       (4) To yield. when we have to yield then we abort and yield
2473       right at the end of do_marking_step(). This saves us from a lot
2474       of hassle as, by yielding we might allow a Full GC. If this
2475       happens then objects will be compacted underneath our feet, the
2476       heap might shrink, etc. We save checking for this by just
2477       aborting and doing the yield right at the end.
2478 
2479     From the above it follows that the do_marking_step() method should
2480     be called in a loop (or, otherwise, regularly) until it completes.
2481 
2482     If a marking step completes without its has_aborted() flag being
2483     true, it means it has completed the current marking phase (and
2484     also all other marking tasks have done so and have all synced up).
2485 
2486     A method called regular_clock_call() is invoked "regularly" (in
2487     sub ms intervals) throughout marking. It is this clock method that
2488     checks all the abort conditions which were mentioned above and
2489     decides when the task should abort. A work-based scheme is used to
2490     trigger this clock method: when the number of object words the
2491     marking phase has scanned or the number of references the marking
2492     phase has visited reach a given limit. Additional invocations to
2493     the method clock have been planted in a few other strategic places
2494     too. The initial reason for the clock method was to avoid calling
2495     vtime too regularly, as it is quite expensive. So, once it was in
2496     place, it was natural to piggy-back all the other conditions on it
2497     too and not constantly check them throughout the code.
2498 
2499     If do_termination is true then do_marking_step will enter its
2500     termination protocol.
2501 
2502     The value of is_serial must be true when do_marking_step is being
2503     called serially (i.e. by the VMThread) and do_marking_step should
2504     skip any synchronization in the termination and overflow code.
2505     Examples include the serial remark code and the serial reference
2506     processing closures.
2507 
2508     The value of is_serial must be false when do_marking_step is
2509     being called by any of the worker threads in a work gang.
2510     Examples include the concurrent marking code (CMMarkingTask),
2511     the MT remark code, and the MT reference processing closures.
2512 
2513  *****************************************************************************/

2515 void G1CMTask::do_marking_step(double time_target_ms,
2516                                bool do_termination,
2517                                bool is_serial) {
2518   assert(time_target_ms >= 1.0, "minimum granularity is 1ms");
2519 
2520   _start_time_ms = os::elapsedVTime() * 1000.0;
2521 
2522   // If do_stealing is true then do_marking_step will attempt to
2523   // steal work from the other G1CMTasks. It only makes sense to
2524   // enable stealing when the termination protocol is enabled
2525   // and do_marking_step() is not being called serially.
2526   bool do_stealing = do_termination && !is_serial;
2527 
2528   G1Predictions const& predictor = _g1h->policy()->predictor();
2529   double diff_prediction_ms = predictor.predict_zero_bounded(&_marking_step_diff_ms);
2530   _time_target_ms = time_target_ms - diff_prediction_ms;
2531 
2532   // set up the variables that are used in the work-based scheme to
2533   // call the regular clock method
2534   _words_scanned = 0;
2535   _refs_reached  = 0;
2536   recalculate_limits();
2537 
2538   // clear all flags
2539   clear_has_aborted();
2540   _has_timed_out = false;
2541   _draining_satb_buffers = false;
2542 
2543   ++_calls;
2544 
2545   // Set up the bitmap and oop closures. Anything that uses them is
2546   // eventually called from this method, so it is OK to allocate these
2547   // statically.
2548   G1CMBitMapClosure bitmap_closure(this, _cm);
2549   G1CMOopClosure cm_oop_closure(_g1h, this);
2550   set_cm_oop_closure(&cm_oop_closure);
2551 
2552   if (_cm->has_overflown()) {
2553     // This can happen if the mark stack overflows during a GC pause
2554     // and this task, after a yield point, restarts. We have to abort
2555     // as we need to get into the overflow protocol which happens
2556     // right at the end of this task.
2557     set_has_aborted();
2558   }
2559 
2560   // First drain any available SATB buffers. After this, we will not
2561   // look at SATB buffers before the next invocation of this method.
2562   // If enough completed SATB buffers are queued up, the regular clock
2563   // will abort this task so that it restarts.
2564   drain_satb_buffers();
2565   // ...then partially drain the local queue and the global stack
2566   drain_local_queue(true);
2567   drain_global_stack(true);
2568 
2569   do {
2570     if (!has_aborted() && _curr_region != NULL) {
2571       // This means that we're already holding on to a region.
2572       assert(_finger != NULL, "if region is not NULL, then the finger "
2573              "should not be NULL either");
2574 
2575       // We might have restarted this task after an evacuation pause
2576       // which might have evacuated the region we're holding on to
2577       // underneath our feet. Let's read its limit again to make sure
2578       // that we do not iterate over a region of the heap that
2579       // contains garbage (update_region_limit() will also move
2580       // _finger to the start of the region if it is found empty).
2581       update_region_limit();
2582       // We will start from _finger not from the start of the region,
2583       // as we might be restarting this task after aborting half-way
2584       // through scanning this region. In this case, _finger points to
2585       // the address where we last found a marked object. If this is a
2586       // fresh region, _finger points to start().
2587       MemRegion mr = MemRegion(_finger, _region_limit);
2588 
2589       assert(!_curr_region->is_humongous() || mr.start() == _curr_region->bottom(),
2590              "humongous regions should go around loop once only");
2591 
2592       // Some special cases:
2593       // If the memory region is empty, we can just give up the region.
2594       // If the current region is humongous then we only need to check
2595       // the bitmap for the bit associated with the start of the object,
2596       // scan the object if it's live, and give up the region.
2597       // Otherwise, let's iterate over the bitmap of the part of the region
2598       // that is left.
2599       // If the iteration is successful, give up the region.
2600       if (mr.is_empty()) {
2601         giveup_current_region();
2602         abort_marking_if_regular_check_fail();
2603       } else if (_curr_region->is_humongous() && mr.start() == _curr_region->bottom()) {
2604         if (_next_mark_bitmap->is_marked(mr.start())) {
2605           // The object is marked - apply the closure
2606           bitmap_closure.do_addr(mr.start());
2607         }
2608         // Even if this task aborted while scanning the humongous object
2609         // we can (and should) give up the current region.
2610         giveup_current_region();
2611         abort_marking_if_regular_check_fail();
2612       } else if (_next_mark_bitmap->iterate(&bitmap_closure, mr)) {
2613         giveup_current_region();
2614         abort_marking_if_regular_check_fail();
2615       } else {
2616         assert(has_aborted(), "currently the only way to do so");
2617         // The only way to abort the bitmap iteration is to return
2618         // false from the do_bit() method. However, inside the
2619         // do_bit() method we move the _finger to point to the
2620         // object currently being looked at. So, if we bail out, we
2621         // have definitely set _finger to something non-null.
2622         assert(_finger != NULL, "invariant");
2623 
2624         // Region iteration was actually aborted. So now _finger
2625         // points to the address of the object we last scanned. If we
2626         // leave it there, when we restart this task, we will rescan
2627         // the object. It is easy to avoid this. We move the finger by
2628         // enough to point to the next possible object header.
2629         assert(_finger < _region_limit, "invariant");
2630         HeapWord* const new_finger = _finger + ((oop)_finger)->size();
2631         // Check if bitmap iteration was aborted while scanning the last object
2632         if (new_finger >= _region_limit) {
2633           giveup_current_region();
2634         } else {
2635           move_finger_to(new_finger);
2636         }
2637       }
2638     }
2639     // At this point we have either completed iterating over the
2640     // region we were holding on to, or we have aborted.
2641 
2642     // We then partially drain the local queue and the global stack.
2643     // (Do we really need this?)
2644     drain_local_queue(true);
2645     drain_global_stack(true);
2646 
2647     // Read the note on the claim_region() method on why it might
2648     // return NULL with potentially more regions available for
2649     // claiming and why we have to check out_of_regions() to determine
2650     // whether we're done or not.
2651     while (!has_aborted() && _curr_region == NULL && !_cm->out_of_regions()) {
2652       // We are going to try to claim a new region. We should have
2653       // given up on the previous one.
2654       // Separated the asserts so that we know which one fires.
2655       assert(_curr_region  == NULL, "invariant");
2656       assert(_finger       == NULL, "invariant");
2657       assert(_region_limit == NULL, "invariant");
2658       HeapRegion* claimed_region = _cm->claim_region(_worker_id);
2659       if (claimed_region != NULL) {
2660         // Yes, we managed to claim one
2661         setup_for_region(claimed_region);
2662         assert(_curr_region == claimed_region, "invariant");
2663       }
2664       // It is important to call the regular clock here. It might take
2665       // a while to claim a region if, for example, we hit a large
2666       // block of empty regions. So we need to call the regular clock
2667       // method once round the loop to make sure it's called
2668       // frequently enough.
2669       abort_marking_if_regular_check_fail();
2670     }
2671 
2672     if (!has_aborted() && _curr_region == NULL) {
2673       assert(_cm->out_of_regions(),
2674              "at this point we should be out of regions");
2675     }
2676   } while ( _curr_region != NULL && !has_aborted());
2677 
2678   if (!has_aborted()) {
2679     // We cannot check whether the global stack is empty, since other
2680     // tasks might be pushing objects to it concurrently.
2681     assert(_cm->out_of_regions(),
2682            "at this point we should be out of regions");
2683     // Try to reduce the number of available SATB buffers so that
2684     // remark has less work to do.
2685     drain_satb_buffers();
2686   }
2687 
2688   // Since we've done everything else, we can now totally drain the
2689   // local queue and global stack.
2690   drain_local_queue(false);
2691   drain_global_stack(false);
2692 
2693   // Attempt at work stealing from other task's queues.
2694   if (do_stealing && !has_aborted()) {
2695     // We have not aborted. This means that we have finished all that
2696     // we could. Let's try to do some stealing...
2697 
2698     // We cannot check whether the global stack is empty, since other
2699     // tasks might be pushing objects to it concurrently.
2700     assert(_cm->out_of_regions() && _task_queue->size() == 0,
2701            "only way to reach here");
2702     while (!has_aborted()) {
2703       G1TaskQueueEntry entry;
2704       if (_cm->try_stealing(_worker_id, entry)) {
2705         scan_task_entry(entry);
2706 
2707         // And since we're towards the end, let's totally drain the
2708         // local queue and global stack.
2709         drain_local_queue(false);
2710         drain_global_stack(false);
2711       } else {
2712         break;
2713       }
2714     }
2715   }
2716 
2717   // We still haven't aborted. Now, let's try to get into the
2718   // termination protocol.
2719   if (do_termination && !has_aborted()) {
2720     // We cannot check whether the global stack is empty, since other
2721     // tasks might be concurrently pushing objects on it.
2722     // Separated the asserts so that we know which one fires.
2723     assert(_cm->out_of_regions(), "only way to reach here");
2724     assert(_task_queue->size() == 0, "only way to reach here");
2725     _termination_start_time_ms = os::elapsedVTime() * 1000.0;
2726 
2727     // The G1CMTask class also extends the TerminatorTerminator class,
2728     // hence its should_exit_termination() method will also decide
2729     // whether to exit the termination protocol or not.
2730     bool finished = (is_serial ||
2731                      _cm->terminator()->offer_termination(this));
2732     double termination_end_time_ms = os::elapsedVTime() * 1000.0;
2733     _termination_time_ms +=
2734       termination_end_time_ms - _termination_start_time_ms;
2735 
2736     if (finished) {
2737       // We're all done.
2738 
2739       // We can now guarantee that the global stack is empty, since
2740       // all other tasks have finished. We separated the guarantees so
2741       // that, if a condition is false, we can immediately find out
2742       // which one.
2743       guarantee(_cm->out_of_regions(), "only way to reach here");
2744       guarantee(_cm->mark_stack_empty(), "only way to reach here");
2745       guarantee(_task_queue->size() == 0, "only way to reach here");
2746       guarantee(!_cm->has_overflown(), "only way to reach here");
2747       guarantee(!has_aborted(), "should never happen if termination has completed");
2748     } else {
2749       // Apparently there's more work to do. Let's abort this task. It
2750       // will restart it and we can hopefully find more things to do.
2751       set_has_aborted();
2752     }
2753   }
2754 
2755   // Mainly for debugging purposes to make sure that a pointer to the
2756   // closure which was statically allocated in this frame doesn't
2757   // escape it by accident.
2758   set_cm_oop_closure(NULL);
2759   double end_time_ms = os::elapsedVTime() * 1000.0;
2760   double elapsed_time_ms = end_time_ms - _start_time_ms;
2761   // Update the step history.
2762   _step_times_ms.add(elapsed_time_ms);
2763 
2764   if (has_aborted()) {
2765     // The task was aborted for some reason.
2766     if (_has_timed_out) {
2767       double diff_ms = elapsed_time_ms - _time_target_ms;
2768       // Keep statistics of how well we did with respect to hitting
2769       // our target only if we actually timed out (if we aborted for
2770       // other reasons, then the results might get skewed).
2771       _marking_step_diff_ms.add(diff_ms);
2772     }
2773 
2774     if (_cm->has_overflown()) {
2775       // This is the interesting one. We aborted because a global
2776       // overflow was raised. This means we have to restart the
2777       // marking phase and start iterating over regions. However, in
2778       // order to do this we have to make sure that all tasks stop
2779       // what they are doing and re-initialize in a safe manner. We
2780       // will achieve this with the use of two barrier sync points.
2781 
2782       if (!is_serial) {
2783         // We only need to enter the sync barrier if being called
2784         // from a parallel context
2785         _cm->enter_first_sync_barrier(_worker_id);
2786 
2787         // When we exit this sync barrier we know that all tasks have
2788         // stopped doing marking work. So, it's now safe to
2789         // re-initialize our data structures.
2790       }
2791 
2792       clear_region_fields();
2793       flush_mark_stats_cache();
2794 
2795       if (!is_serial) {
2796         // If we're executing the concurrent phase of marking, reset the marking
2797         // state; otherwise the marking state is reset after reference processing,
2798         // during the remark pause.
2799         // If we reset here as a result of an overflow during the remark we will
2800         // see assertion failures from any subsequent set_concurrency_and_phase()
2801         // calls.
2802         if (_cm->concurrent() && _worker_id == 0) {
2803           // Worker 0 is responsible for clearing the global data structures because
2804           // of an overflow. During STW we should not clear the overflow flag (in
2805           // G1ConcurrentMark::reset_marking_state()) since we rely on it being true when we exit
2806           // method to abort the pause and restart concurrent marking.
2807           _cm->reset_marking_for_restart();
2808 
2809           log_info(gc, marking)("Concurrent Mark reset for overflow");
2810         }
2811 
2812         // ...and enter the second barrier.
2813         _cm->enter_second_sync_barrier(_worker_id);
2814       }
2815       // At this point, if we're during the concurrent phase of
2816       // marking, everything has been re-initialized and we're
2817       // ready to restart.
2818     }
2819   }
2820 }
