 26 package jdk.tools.jaotc;
 27 
 28 import jdk.tools.jaotc.binformat.BinaryContainer;
 29 import jdk.tools.jaotc.binformat.Relocation;
 30 import jdk.tools.jaotc.binformat.Relocation.RelocType;
 31 import jdk.tools.jaotc.binformat.Symbol;
 32 
 33 import jdk.vm.ci.code.site.Mark;
 34 
 35 final class MarkProcessor {
 36 
 37     private final BinaryContainer binaryContainer;
 38 
 39     MarkProcessor(DataBuilder dataBuilder) {
 40         binaryContainer = dataBuilder.getBinaryContainer();
 41     }
 42 
 43     /**
 44      * Parse a {@link Mark} generated by the compiler and create all needed binary section
 45      * constructs.
 46      *
 47      * @param methodInfo compiled method info
 48      * @param mark mark being processed
 49      */
 50     @SuppressWarnings("fallthrough")
 51     void process(CompiledMethodInfo methodInfo, Mark mark) {
 52         MarkId markId = MarkId.getEnum((int) mark.id);
 53         switch (markId) {
 54             case EXCEPTION_HANDLER_ENTRY:
 55             case DEOPT_HANDLER_ENTRY:
 56                 break;
 57             case POLL_FAR:
 58             case POLL_RETURN_FAR:
 59                 if (binaryContainer.getThreadLocalHandshakes()) {
 60                     // skip relocation
 61                     break;
 62                 }
 63                 // fallthrough
 64             case CARD_TABLE_ADDRESS:
 65             case HEAP_TOP_ADDRESS:
 66             case HEAP_END_ADDRESS:
 67             case NARROW_KLASS_BASE_ADDRESS:
 68             case NARROW_OOP_BASE_ADDRESS:
 69             case CRC_TABLE_ADDRESS:
 70             case LOG_OF_HEAP_REGION_GRAIN_BYTES:
 71             case INLINE_CONTIGUOUS_ALLOCATION_SUPPORTED:
 72                 String vmSymbolName;
 73                 switch (markId) {
 74                     case CARD_TABLE_ADDRESS:
 75                         vmSymbolName = BinaryContainer.getCardTableAddressSymbolName();
 76                         break;
 77                     case HEAP_TOP_ADDRESS:
 78                         vmSymbolName = BinaryContainer.getHeapTopAddressSymbolName();
 79                         break;
 80                     case HEAP_END_ADDRESS:
 81                         vmSymbolName = BinaryContainer.getHeapEndAddressSymbolName();
 82                         break;
 83                     case NARROW_KLASS_BASE_ADDRESS:
 84                         vmSymbolName = BinaryContainer.getNarrowKlassBaseAddressSymbolName();
 85                         break;
 86                     case NARROW_OOP_BASE_ADDRESS:
 87                         vmSymbolName = BinaryContainer.getNarrowOopBaseAddressSymbolName();
 88                         break;
 89                     case CRC_TABLE_ADDRESS:
 90                         vmSymbolName = BinaryContainer.getCrcTableAddressSymbolName();
 91                         break;
 92                     case LOG_OF_HEAP_REGION_GRAIN_BYTES:
 93                         vmSymbolName = BinaryContainer.getLogOfHeapRegionGrainBytesSymbolName();
 94                         break;
 95                     case INLINE_CONTIGUOUS_ALLOCATION_SUPPORTED:
 96                         vmSymbolName = BinaryContainer.getInlineContiguousAllocationSupportedSymbolName();
 97                         break;
 98                     default:
 99                         throw new InternalError("Unhandled mark: " + mark);
100                 }
101                 String s = "got." + vmSymbolName;
102                 Symbol gotSymbol = binaryContainer.getGotSymbol(s);
103                 assert gotSymbol != null : " Processing Mark: Encountered undefined got symbol for  " + mark;
104                 final int textBaseOffset = methodInfo.getTextSectionOffset();
105                 final int textOffset = textBaseOffset + mark.pcOffset;
106                 Relocation reloc = new Relocation(textOffset, RelocType.EXTERNAL_PLT_TO_GOT, 8, binaryContainer.getCodeContainer(), gotSymbol);
107                 binaryContainer.addRelocation(reloc);
108                 break;
109             case VERIFIED_ENTRY:
110             case UNVERIFIED_ENTRY:
111             case OSR_ENTRY:
112             case INVOKEINTERFACE:
113             case INVOKEVIRTUAL:
114             case INVOKESTATIC:
115             case INVOKESPECIAL:
116             case INLINE_INVOKE:
117             case POLL_NEAR:
118             case POLL_RETURN_NEAR:
119                 // Nothing to do.
120                 break;
121             default:
122                 throw new InternalError("Unexpected mark found: " + mark);
123         }
124     }
125 }
