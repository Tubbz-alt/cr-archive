19 package jdk.internal.org.jline.reader;
20 
21 public class EOFError extends SyntaxError {
22 
23     private static final long serialVersionUID = 1L;
24 
25     private final String missing;
26 
27     public EOFError(int line, int column, String message) {
28         this(line, column, message, null);
29     }
30 
31     public EOFError(int line, int column, String message, String missing) {
32         super(line, column, message);
33         this.missing = missing;
34     }
35 
36     public String getMissing() {
37         return missing;
38     }
39 }
