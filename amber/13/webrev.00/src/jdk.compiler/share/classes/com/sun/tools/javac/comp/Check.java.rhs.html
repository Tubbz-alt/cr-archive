 329     /** Report duplicate declaration error.
 330      */
 331     void duplicateError(DiagnosticPosition pos, Symbol sym) {
 332         if (!sym.type.isErroneous()) {
 333             Symbol location = sym.location();
 334             if (location.kind == MTH &&
 335                     ((MethodSymbol)location).isStaticOrInstanceInit()) {
 336                 log.error(pos,
 337                           Errors.AlreadyDefinedInClinit(kindName(sym),
 338                                                         sym,
 339                                                         kindName(sym.location()),
 340                                                         kindName(sym.location().enclClass()),
 341                                                         sym.location().enclClass()));
 342             } else {
 343                 /* dont error if this is a duplicated parameter of a generated canonical constructor
 344                  * as we should have issued an error for the duplicated fields
 345                  */
 346                 if (location.kind != MTH ||
 347                         ((sym.owner.flags_field & GENERATEDCONSTR) == 0) ||
 348                         ((sym.owner.flags_field & RECORD) == 0)) {
 349                     log.error(pos,
 350                             Errors.AlreadyDefined(kindName(sym),
 351                                     sym,
 352                                     kindName(sym.location()),
 353                                     sym.location()));
 354                 }
 355             }
 356         }
 357     }

 371     /** Check that variable does not hide variable with same name in
 372      *  immediately enclosing local scope.
 373      *  @param pos           Position for error reporting.
 374      *  @param v             The symbol.
 375      *  @param s             The scope.
 376      */
 377     void checkTransparentVar(DiagnosticPosition pos, VarSymbol v, Scope s) {
 378         for (Symbol sym : s.getSymbolsByName(v.name)) {
 379             if (sym.owner != v.owner) break;
 380             if (sym.kind == VAR &&
 381                 sym.owner.kind.matches(KindSelector.VAL_MTH) &&
 382                 v.name != names.error) {
 383                 duplicateError(pos, sym);
 384                 return;
 385             }
 386         }
 387     }
 388 
 389     /** Check that a class or interface does not hide a class or
 390      *  interface with same name in immediately enclosing local scope.
 391      *  @param pos           Position for error reporting.
 392      *  @param c             The symbol.
 393      *  @param s             The scope.
 394      */
 395     void checkTransparentClass(DiagnosticPosition pos, ClassSymbol c, Scope s) {
 396         for (Symbol sym : s.getSymbolsByName(c.name)) {
 397             if (sym.owner != c.owner) break;
 398             if (sym.kind == TYP && !sym.type.hasTag(TYPEVAR) &&
 399                 sym.owner.kind.matches(KindSelector.VAL_MTH) &&
 400                 c.name != names.error) {
 401                 duplicateError(pos, sym);
 402                 return;
 403             }
 404         }
 405     }

 407     /** Check that class does not have the same name as one of
 408      *  its enclosing classes, or as a class defined in its enclosing scope.
 409      *  return true if class is unique in its enclosing scope.
 410      *  @param pos           Position for error reporting.
 411      *  @param name          The class name.
 412      *  @param s             The enclosing scope.
 413      */
 414     boolean checkUniqueClassName(DiagnosticPosition pos, Name name, Scope s) {
 415         for (Symbol sym : s.getSymbolsByName(name, NON_RECURSIVE)) {
 416             if (sym.kind == TYP && sym.name != names.error) {
 417                 duplicateError(pos, sym);
 418                 return false;
 419             }
 420         }
 421         for (Symbol sym = s.owner; sym != null; sym = sym.owner) {
 422             if (sym.kind == TYP && sym.name == name && sym.name != names.error) {
 423                 duplicateError(pos, sym);
 424                 return true;
 425             }
 426         }
 427         return true;
 428     }

 437     /** Return name of local class.
 438      *  This is of the form   {@code <enclClass> $ n <classname> }
 439      *  where
 440      *    enclClass is the flat name of the enclosing class,
 441      *    classname is the simple name of the local class
 442      */
 443     public Name localClassName(ClassSymbol c) {
 444         Name enclFlatname = c.owner.enclClass().flatname;
 445         String enclFlatnameStr = enclFlatname.toString();
 446         Pair<Name, Name> key = new Pair<>(enclFlatname, c.name);
 447         Integer index = localClassNameIndexes.get(key);
 448         for (int i = (index == null) ? 1 : index; ; i++) {
 449             Name flatname = names.fromString(enclFlatnameStr
 450                     + syntheticNameChar + i + c.name);
 451             if (getCompiled(c.packge().modle, flatname) == null) {
 452                 localClassNameIndexes.put(key, i + 1);
 453                 return flatname;
 454             }
 455         }
 456     }
 457 
 458     public void clearLocalClassNameIndexes(ClassSymbol c) {
 459         if (c.owner != null && c.owner.kind != NIL) {
 460             localClassNameIndexes.remove(new Pair<>(
 461                     c.owner.enclClass().flatname, c.name));
 462         }
 463     }

 594     Type checkType(final DiagnosticPosition pos, final Type found, final Type req, final CheckContext checkContext) {
 595         final InferenceContext inferenceContext = checkContext.inferenceContext();
 596         if (inferenceContext.free(req) || inferenceContext.free(found)) {
 597             inferenceContext.addFreeTypeListener(List.of(req, found),
 598                     solvedContext -> checkType(pos, solvedContext.asInstType(found), solvedContext.asInstType(req), checkContext));
 599         }
 600         if (req.hasTag(ERROR))
 601             return req;
 602         if (req.hasTag(NONE))
 603             return found;
 604         if (checkContext.compatible(found, req, checkContext.checkWarner(pos, found, req))) {
 605             return found;
 606         } else {
 607             if (found.isNumeric() && req.isNumeric()) {
 608                 checkContext.report(pos, diags.fragment(Fragments.PossibleLossOfPrecision(found, req)));
 609                 return types.createErrorType(found);
 610             }
 611             checkContext.report(pos, diags.fragment(Fragments.InconvertibleTypes(found, req)));
 612             return types.createErrorType(found);
 613         }
 614     }

 643     /** Check for redundant casts (i.e. where source type is a subtype of target type)
 644      * The problem should only be reported for non-292 cast

 990     Type checkMethod(final Type mtype,
 991             final Symbol sym,
 992             final Env<AttrContext> env,
 993             final List<JCExpression> argtrees,
 994             final List<Type> argtypes,
 995             final boolean useVarargs,
 996             InferenceContext inferenceContext) {
 997         // System.out.println("call   : " + env.tree);
 998         // System.out.println("method : " + owntype);
 999         // System.out.println("actuals: " + argtypes);
1000         if (inferenceContext.free(mtype)) {
1001             inferenceContext.addFreeTypeListener(List.of(mtype),
1002                     solvedContext -> checkMethod(solvedContext.asInstType(mtype), sym, env, argtrees, argtypes, useVarargs, solvedContext));
1003             return mtype;
1004         }
1005         Type owntype = mtype;
1006         List<Type> formals = owntype.getParameterTypes();
1007         List<Type> nonInferred = sym.type.getParameterTypes();
1008         if (nonInferred.length() != formals.length()) nonInferred = formals;
1009         Type last = useVarargs ? formals.last() : null;
1010         if (sym.name == names.init && sym.owner == syms.enumSym) {
1011             formals = formals.tail.tail;
1012             nonInferred = nonInferred.tail.tail;
1013         }
1014         if ((sym.flags() & ANONCONSTR_BASED) != 0) {
1015             formals = formals.tail;
1016             nonInferred = nonInferred.tail;
1017         }
1018         List<JCExpression> args = argtrees;
1019         if (args != null) {
1020             //this is null when type-checking a method reference
1021             while (formals.head != last) {
1022                 JCTree arg = args.head;
1023                 Warner warn = convertWarner(arg.pos(), arg.type, nonInferred.head);
1024                 assertConvertible(arg, arg.type, formals.head, warn);
1025                 args = args.tail;
1026                 formals = formals.tail;
1027                 nonInferred = nonInferred.tail;
1028             }
1029             if (useVarargs) {
1030                 Type varArg = types.elemtype(last);
1031                 while (args.tail != null) {
1032                     JCTree arg = args.head;
1033                     Warner warn = convertWarner(arg.pos(), arg.type, varArg);
1034                     assertConvertible(arg, arg.type, varArg, warn);
1035                     args = args.tail;
1036                 }
1037             } else if ((sym.flags() & (VARARGS | SIGNATURE_POLYMORPHIC)) == VARARGS) {
1038                 // non-varargs call to varargs method
1039                 Type varParam = owntype.getParameterTypes().last();
1040                 Type lastArg = argtypes.last();
1041                 if (types.isSubtypeUnchecked(lastArg, types.elemtype(varParam)) &&
1042                     !types.isSameType(types.erasure(varParam), types.erasure(lastArg)))
1043                     log.warning(argtrees.last().pos(),
1044                                 Warnings.InexactNonVarargsCall(types.elemtype(varParam),varParam));
1045             }
1046         }

1049             if (!types.isReifiable(argtype) &&
1050                 (sym.baseSymbol().attribute(syms.trustMeType.tsym) == null ||
1051                  !isTrustMeAllowedOnMethod(sym))) {
1052                 warnUnchecked(env.tree.pos(), Warnings.UncheckedGenericArrayCreation(argtype));
1053             }
1054             TreeInfo.setVarargsElement(env.tree, types.elemtype(argtype));
1055          }
1056          return owntype;
1057     }

1084             List<Type> forms = type.tsym.type.getTypeArguments();
1085             ListBuffer<Type> bounds_buf = new ListBuffer<>();

1087             // For matching pairs of actual argument types `a' and
1088             // formal type parameters with declared bound `b' ...
1089             while (args.nonEmpty() && forms.nonEmpty()) {
1090                 // exact type arguments needs to know their
1091                 // bounds (for upper and lower bound
1092                 // calculations).  So we create new bounds where
1093                 // type-parameters are replaced with actuals argument types.
1094                 bounds_buf.append(types.subst(forms.head.getUpperBound(), formals, actuals));
1095                 args = args.tail;
1096                 forms = forms.tail;
1097             }
1098 
1099             args = type.getTypeArguments();
1100             List<Type> tvars_cap = types.substBounds(formals,
1101                                       formals,
1102                                       types.capture(type).allparams());
1103             while (args.nonEmpty() && tvars_cap.nonEmpty()) {
1104                 // Let the actual arguments know their bound
1105                 args.head.withTypeVar((TypeVar)tvars_cap.head);
1106                 args = args.tail;
1107                 tvars_cap = tvars_cap.tail;
1108             }
1109 
1110             args = type.getTypeArguments();
1111             List<Type> bounds = bounds_buf.toList();
1112 
1113             while (args.nonEmpty() && bounds.nonEmpty()) {
1114                 Type actual = args.head;
1115                 if (!isTypeArgErroneous(actual) &&
1116                         !bounds.head.isErroneous() &&
1117                         !checkExtends(actual, bounds.head)) {
1118                     return args.head;
1119                 }
1120                 args = args.tail;
1121                 bounds = bounds.tail;
1122             }
1123 
1124             args = type.getTypeArguments();
1125             bounds = bounds_buf.toList();
1126 
1127             for (Type arg : types.capture(type).getTypeArguments()) {
1128                 if (arg.hasTag(TYPEVAR) &&
1129                         arg.getUpperBound().isErroneous() &&
1130                         !bounds.head.isErroneous() &&
1131                         !isTypeArgErroneous(args.head)) {

1133                 }
1134                 bounds = bounds.tail;

1164     /** Check that given modifiers are legal for given symbol and
1165      *  return modifiers together with any implicit modifiers for that symbol.
1166      *  Warning: we can't use flags() here since this method
1167      *  is called during class enter, when flags() would cause a premature
1168      *  completion.
1169      *  @param pos           Position to be used for error reporting.
1170      *  @param flags         The set of modifiers given in a definition.
1171      *  @param sym           The defined symbol.
1172      */
1173     long checkFlags(DiagnosticPosition pos, long flags, Symbol sym, JCTree tree) {
1174         long mask;
1175         long implicit = 0;
1176 
1177         switch (sym.kind) {
1178         case VAR:
1179             if (TreeInfo.isReceiverParam(tree))
1180                 mask = ReceiverParamFlags;
1181             else if (sym.owner.kind != TYP)
1182                 mask = LocalVarFlags;
1183             else if ((sym.owner.flags_field & INTERFACE) != 0)
1184                 mask = implicit = InterfaceVarFlags;
1185             else
1186                 mask = VarFlags;
1187             break;
1188         case MTH:
1189             if (sym.name == names.init) {
1190                 if ((sym.owner.flags_field & ENUM) != 0) {
1191                     // enum constructors cannot be declared public or
1192                     // protected and must be implicitly or explicitly
1193                     // private
1194                     implicit = PRIVATE;
1195                     mask = PRIVATE;
1196                 } else
1197                     mask = ConstructorFlags;
1198             }  else if ((sym.owner.flags_field & INTERFACE) != 0) {
1199                 if ((sym.owner.flags_field & ANNOTATION) != 0) {
1200                     mask = AnnotationTypeElementMask;
1201                     implicit = PUBLIC | ABSTRACT;
1202                 } else if ((flags & (DEFAULT | STATIC | PRIVATE)) != 0) {
1203                     mask = InterfaceMethodMask;
1204                     implicit = (flags & PRIVATE) != 0 ? 0 : PUBLIC;
1205                     if ((flags & DEFAULT) != 0) {
1206                         implicit |= ABSTRACT;
1207                     }
1208                 } else {
1209                     mask = implicit = InterfaceMethodFlags;
1210                 }
1211             } else if ((sym.owner.flags_field & RECORD) != 0) {
1212                 mask = RecordMethodFlags;
1213             } else {
1214                 mask = MethodFlags;
1215             }
1216             // Imply STRICTFP if owner has STRICTFP set.
1217             if (((flags|implicit) & Flags.ABSTRACT) == 0 ||
1218                 ((flags) & Flags.DEFAULT) != 0)
1219                 implicit |= sym.owner.flags_field & STRICTFP;
1220             break;
1221         case TYP:
1222             if (sym.isLocal()) {
1223                 mask = (flags & RECORD) != 0 ? LocalRecordFlags : LocalClassFlags;
1224                 if ((flags & RECORD) != 0) {
1225                     implicit = STATIC;
1226                     if (sym.owner.kind == TYP) {
1227                         log.error(pos, Errors.RecordDeclarationNotAllowedInInnerClasses);
1228                     }
1229                 }
1230                 if ((sym.owner.flags_field & STATIC) == 0 &&
1231                     (flags & ENUM) != 0) {
1232                     log.error(pos, Errors.EnumsMustBeStatic);
1233                 }
1234             } else if (sym.owner.kind == TYP) {
1235                 mask = (flags & RECORD) != 0 ? MemberRecordFlags : MemberClassFlags;
1236                 if (sym.owner.owner.kind == PCK ||
1237                     (sym.owner.flags_field & STATIC) != 0)
1238                     mask |= STATIC;
1239                 else if ((flags & ENUM) != 0) {
1240                     log.error(pos, Errors.EnumsMustBeStatic);
1241                 } else if ((flags & RECORD) != 0) {
1242                     log.error(pos, Errors.RecordDeclarationNotAllowedInInnerClasses);
1243                 }
1244                 // Nested interfaces and enums are always STATIC (Spec ???)
1245                 if ((flags & (INTERFACE | ENUM | RECORD)) != 0 ) implicit = STATIC;
1246             } else {
1247                 mask = ClassFlags;
1248             }
1249             // Interfaces are always ABSTRACT
1250             if ((flags & INTERFACE) != 0) implicit |= ABSTRACT;
1251 
1252             if ((flags & ENUM) != 0) {
1253                 // enums can't be declared abstract or final
1254                 mask &= ~(ABSTRACT | FINAL);
1255                 implicit |= implicitEnumFinalFlag(tree);
1256             }
1257             if ((flags & RECORD) != 0) {
1258                 // records can't be declared abstract
1259                 mask &= ~ABSTRACT;
1260                 implicit |= FINAL;
1261             }
1262             // Imply STRICTFP if owner has STRICTFP set.
1263             implicit |= sym.owner.flags_field & STRICTFP;
1264             break;
1265         default:
1266             throw new AssertionError();
1267         }
1268         long illegal = flags & ExtendedStandardFlags & ~mask;
1269         if (illegal != 0) {
1270             if ((illegal & INTERFACE) != 0) {
1271                 log.error(pos, ((flags & ANNOTATION) != 0) ? Errors.AnnotationDeclNotAllowedHere : Errors.IntfNotAllowedHere);
1272                 mask |= INTERFACE;
1273             }
1274             else {
1275                 log.error(pos,
1276                           Errors.ModNotAllowedHere(asFlagSet(illegal)));
1277             }
1278         }
1279         else if ((sym.kind == TYP ||
1280                   // ISSUE: Disallowing abstract&private is no longer appropriate
1281                   // in the presence of inner classes. Should it be deleted here?
1282                   checkDisjoint(pos, flags,
1283                                 ABSTRACT,
1284                                 PRIVATE | STATIC | DEFAULT))
1285                  &&
1286                  checkDisjoint(pos, flags,
1287                                 STATIC | PRIVATE,
1288                                 DEFAULT)
1289                  &&
1290                  checkDisjoint(pos, flags,
1291                                ABSTRACT | INTERFACE,
1292                                FINAL | NATIVE | SYNCHRONIZED)
1293                  &&
1294                  checkDisjoint(pos, flags,
1295                                PUBLIC,
1296                                PRIVATE | PROTECTED)
1297                  &&
1298                  checkDisjoint(pos, flags,
1299                                PRIVATE,
1300                                PUBLIC | PROTECTED)
1301                  &&
1302                  checkDisjoint(pos, flags,
1303                                FINAL,
1304                                VOLATILE)
1305                  &&
1306                  (sym.kind == TYP ||
1307                   checkDisjoint(pos, flags,
1308                                 ABSTRACT | NATIVE,
1309                                 STRICTFP))) {
1310             // skip
1311         }
1312         return flags & (mask | ~ExtendedStandardFlags) | implicit;
1313     }

1732     /** Check that this method conforms with overridden method 'other'.
1733      *  where `origin' is the class where checking started.
1734      *  Complications:
1735      *  (1) Do not check overriding of synthetic methods
1736      *      (reason: they might be final).
1737      *      todo: check whether this is still necessary.
1738      *  (2) Admit the case where an interface proxy throws fewer exceptions
1739      *      than the method it implements. Augment the proxy methods with the
1740      *      undeclared exceptions in this case.
1741      *  (3) When generics are enabled, admit the case where an interface proxy
1742      *      has a result type
1743      *      extended by the result type of the method it implements.
1744      *      Change the proxies result type to the smaller type in this case.
1745      *
1746      *  @param tree         The tree from which positions
1747      *                      are extracted for errors.
1748      *  @param m            The overriding method.
1749      *  @param other        The overridden method.
1750      *  @param origin       The class of which the overriding method
1751      *                      is a member.
1752      */
1753     void checkOverride(JCTree tree,
1754                        MethodSymbol m,
1755                        MethodSymbol other,
1756                        ClassSymbol origin) {
1757         // Don't check overriding of synthetic methods or by bridge methods.
1758         if ((m.flags() & (SYNTHETIC|BRIDGE)) != 0 || (other.flags() & SYNTHETIC) != 0) {
1759             return;
1760         }
1761 
1762         // Error if static method overrides instance method (JLS 8.4.6.2).
1763         if ((m.flags() & STATIC) != 0 &&
1764                    (other.flags() & STATIC) == 0) {
1765             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1766                       Errors.OverrideStatic(cannotOverride(m, other)));
1767             m.flags_field |= BAD_OVERRIDE;
1768             return;
1769         }
1770 
1771         // Error if instance method overrides static or final
1772         // method (JLS 8.4.6.1).
1773         if ((other.flags() & FINAL) != 0 ||
1774                  (m.flags() & STATIC) == 0 &&
1775                  (other.flags() & STATIC) != 0) {
1776             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1777                       Errors.OverrideMeth(cannotOverride(m, other),
1778                                           asFlagSet(other.flags() & (FINAL | STATIC))));
1779             m.flags_field |= BAD_OVERRIDE;
1780             return;
1781         }
1782 
1783         if ((m.owner.flags() & ANNOTATION) != 0) {
1784             // handled in validateAnnotationMethod
1785             return;
1786         }
1787 
1788         // Error if overriding method has weaker access (JLS 8.4.6.3).
1789         if (protection(m.flags()) > protection(other.flags())) {
1790             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1791                       (other.flags() & AccessFlags) == 0 ?
1792                               Errors.OverrideWeakerAccess(cannotOverride(m, other),
1793                                                           "package") :
1794                               Errors.OverrideWeakerAccess(cannotOverride(m, other),
1795                                                           asFlagSet(other.flags() & AccessFlags)));
1796             m.flags_field |= BAD_OVERRIDE;
1797             return;
1798         }
1799 
1800         Type mt = types.memberType(origin.type, m);
1801         Type ot = types.memberType(origin.type, other);
1802         // Error if overriding result type is different
1803         // (or, in the case of generics mode, not a subtype) of
1804         // overridden result type. We have to rename any type parameters
1805         // before comparing types.
1806         List<Type> mtvars = mt.getTypeArguments();
1807         List<Type> otvars = ot.getTypeArguments();
1808         Type mtres = mt.getReturnType();
1809         Type otres = types.subst(ot.getReturnType(), otvars, mtvars);
1810 
1811         overrideWarner.clear();
1812         boolean resultTypesOK =
1813             types.returnTypeSubstitutable(mt, ot, otres, overrideWarner);
1814         if (!resultTypesOK) {
1815             if ((m.flags() & STATIC) != 0 && (other.flags() & STATIC) != 0) {
1816                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1817                           Errors.OverrideIncompatibleRet(Fragments.CantHide(m, m.location(), other,
1818                                         other.location()), mtres, otres));
1819                 m.flags_field |= BAD_OVERRIDE;
1820             } else {
1821                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1822                           Errors.OverrideIncompatibleRet(cannotOverride(m, other), mtres, otres));
1823                 m.flags_field |= BAD_OVERRIDE;
1824             }
1825             return;
1826         } else if (overrideWarner.hasNonSilentLint(LintCategory.UNCHECKED)) {
1827             warnUnchecked(TreeInfo.diagnosticPositionFor(m, tree),
1828                     Warnings.OverrideUncheckedRet(uncheckedOverrides(m, other), mtres, otres));
1829         }
1830 
1831         // Error if overriding method throws an exception not reported
1832         // by overridden method.
1833         List<Type> otthrown = types.subst(ot.getThrownTypes(), otvars, mtvars);
1834         List<Type> unhandledErased = unhandled(mt.getThrownTypes(), types.erasure(otthrown));
1835         List<Type> unhandledUnerased = unhandled(mt.getThrownTypes(), otthrown);
1836         if (unhandledErased.nonEmpty()) {
1837             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1838                       Errors.OverrideMethDoesntThrow(cannotOverride(m, other), unhandledUnerased.head));
1839             m.flags_field |= BAD_OVERRIDE;
1840             return;
1841         }
1842         else if (unhandledUnerased.nonEmpty()) {

2652     /**
2653       * Report warnings for potentially ambiguous method declarations. Two declarations
2654       * are potentially ambiguous if they feature two unrelated functional interface
2655       * in same argument position (in which case, a call site passing an implicit
2656       * lambda would be ambiguous).
2657       */
2658     void checkPotentiallyAmbiguousOverloads(DiagnosticPosition pos, Type site,
2659             MethodSymbol msym1, MethodSymbol msym2) {
2660         if (msym1 != msym2 &&
2661                 Feature.DEFAULT_METHODS.allowedInSource(source) &&
2662                 lint.isEnabled(LintCategory.OVERLOADS) &&
2663                 (msym1.flags() & POTENTIALLY_AMBIGUOUS) == 0 &&
2664                 (msym2.flags() & POTENTIALLY_AMBIGUOUS) == 0) {
2665             Type mt1 = types.memberType(site, msym1);
2666             Type mt2 = types.memberType(site, msym2);
2667             //if both generic methods, adjust type variables
2668             if (mt1.hasTag(FORALL) && mt2.hasTag(FORALL) &&
2669                     types.hasSameBounds((ForAll)mt1, (ForAll)mt2)) {
2670                 mt2 = types.subst(mt2, ((ForAll)mt2).tvars, ((ForAll)mt1).tvars);
2671             }
2672             //expand varargs methods if needed
2673             int maxLength = Math.max(mt1.getParameterTypes().length(), mt2.getParameterTypes().length());
2674             List<Type> args1 = rs.adjustArgs(mt1.getParameterTypes(), msym1, maxLength, true);
2675             List<Type> args2 = rs.adjustArgs(mt2.getParameterTypes(), msym2, maxLength, true);
2676             //if arities don't match, exit
2677             if (args1.length() != args2.length()) return;
2678             boolean potentiallyAmbiguous = false;
2679             while (args1.nonEmpty() && args2.nonEmpty()) {
2680                 Type s = args1.head;
2681                 Type t = args2.head;
2682                 if (!types.isSubtype(t, s) && !types.isSubtype(s, t)) {
2683                     if (types.isFunctionalInterface(s) && types.isFunctionalInterface(t) &&
2684                             types.findDescriptorType(s).getParameterTypes().length() > 0 &&
2685                             types.findDescriptorType(s).getParameterTypes().length() ==
2686                             types.findDescriptorType(t).getParameterTypes().length()) {
2687                         potentiallyAmbiguous = true;
2688                     } else {
2689                         break;
2690                     }
2691                 }
2692                 args1 = args1.tail;
2693                 args2 = args2.tail;
2694             }
2695             if (potentiallyAmbiguous) {
2696                 //we found two incompatible functional interfaces with same arity
2697                 //this means a call site passing an implicit lambda would be ambiguous
2698                 msym1.flags_field |= POTENTIALLY_AMBIGUOUS;
2699                 msym2.flags_field |= POTENTIALLY_AMBIGUOUS;
2700                 log.warning(LintCategory.OVERLOADS, pos,
2701                             Warnings.PotentiallyAmbiguousOverload(msym1, msym1.location(),
2702                                                                   msym2, msym2.location()));
2703                 return;
2704             }
2705         }

2894     /** Check an annotation of a symbol.
2895      */
2896     private void validateAnnotation(JCAnnotation a, JCTree declarationTree, Symbol s) {
2897         validateAnnotationTree(a);
2898         boolean isRecordMember = (s.flags_field & RECORD) != 0 || s.enclClass() != null && s.enclClass().isRecord();
2899 
2900         boolean isRecordField = isRecordMember &&
2901                 (s.flags_field & (Flags.PRIVATE | Flags.FINAL | Flags.GENERATED_MEMBER | Flags.RECORD)) != 0 &&
2902                 declarationTree.hasTag(VARDEF) &&
2903                 s.owner.kind == TYP;
2904 
2905         if (isRecordField) {
2906             // first we need to check if the annotation is applicable to records
2907             Name[] targets = getTargetNames(a);
2908             boolean appliesToRecords = false;
2909             for (Name target : targets) {
2910                 appliesToRecords =
2911                                 target == names.FIELD ||
2912                                 target == names.PARAMETER ||
2913                                 target == names.METHOD ||
2914                                 target == names.TYPE_USE ||
2915                                 target == names.RECORD_COMPONENT;
2916                 if (appliesToRecords) {
2917                     break;
2918                 }
2919             }
2920             if (!appliesToRecords) {
2921                 log.error(a.pos(), Errors.AnnotationTypeNotApplicable);
2922             } else {
2923                 /* lets now find the annotations in the field that are targeted to record components and append them to
2924                  * the corresponding record component
2925                  */
2926                 ClassSymbol recordClass = (ClassSymbol) s.owner;
2927                 RecordComponent rc = recordClass.getRecordComponent((VarSymbol)s);
2928                 SymbolMetadata metadata = rc.getMetadata();
2929                 if (metadata == null || metadata.isEmpty()) {
2930                     /* if not is empty then we have already been here, which is the case if multiple annotations are applied
2931                      * to the record component declaration
2932                      */
2933                     rc.appendAttributes(s.getRawAttributes().stream().filter(anno ->
2934                             Arrays.stream(getTargetNames(anno.type.tsym)).anyMatch(name -> name == names.RECORD_COMPONENT)
2935                     ).collect(List.collector()));
2936                     rc.setTypeAttributes(s.getRawTypeAttributes());
2937                     // to get all the type annotations applied to the type
2938                     rc.type = s.type;
2939                 }
2940             }
2941         }
2942 
2943         /* the section below is tricky. Annotations applied to record components are propagated to the corresponding
2944          * record member so if an annotation has target: FIELD, it is propagated to the corresponding FIELD, if it has
2945          * target METHOD, it is propagated to the accessor and so on. But at the moment when method members are generated
2946          * there is no enough information to propagate only the right annotations. So all the annotations are propagated
2947          * to all the possible locations.
2948          *
2949          * At this point we need to remove all the annotations that are not in place before going on with the annotation
2950          * party. On top of the above there is the issue that there is no AST representing record components, just symbols
2951          * so the corresponding field has been holding all the annotations and it's metadata has been modified as if it
2952          * was both a field and a record component.
2953          *
2954          * So there are two places where we need to trim annotations from: the metadata of the symbol and / or the modifiers
2955          * in the AST. Whatever is in the metadata will be written to the class file, whatever is in the modifiers could
2956          * be see by annotation processors.
2957          *
2958          * The metadata contains both type annotations and declaration annotations. At this point of the game we don't
2959          * need to care about type annotations, they are all in the right place. But we could need to remove declaration
2960          * annotations. So for declaration annotations if they are not applicable to the record member, excluding type
