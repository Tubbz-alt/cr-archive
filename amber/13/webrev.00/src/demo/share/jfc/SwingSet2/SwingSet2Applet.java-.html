56 public class SwingSet2Applet extends JApplet {
57     public void init() {
58         getContentPane().setLayout(new BorderLayout());
59         getContentPane().add(new SwingSet2(this), BorderLayout.CENTER);
60     }
61 
62     public URL getURL(String filename) {
63         URL codeBase = this.getCodeBase();
64         URL url = null;
65 
66         try {
67             url = new URL(codeBase, filename);
68             System.out.println(url);
69         } catch (java.net.MalformedURLException e) {
70             System.out.println("Error: badly specified URL");
71             return null;
72         }
73 
74         return url;
75     }
