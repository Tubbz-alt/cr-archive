 566     private boolean hiddenIn(ClassSymbol clazz, Types types) {
 567         Symbol sym = hiddenInInternal(clazz, types);
 568         Assert.check(sym != null, "the result of hiddenInInternal() can't be null");
 569         /* If we find the current symbol then there is no symbol hiding it
 570          */
 571         return sym != this;
 572     }
 573 
 574     /** This method looks in the supertypes graph that has the current class as the
 575      * initial node, till it finds the current symbol or another symbol that hides it.
 576      * If the current class has more than one supertype (extends one class and
 577      * implements one or more interfaces) then null can be returned, meaning that
 578      * a wrong path in the supertypes graph was selected. Null can only be returned
 579      * as a temporary value, as a result of the recursive call.
 580      */
 581     private Symbol hiddenInInternal(ClassSymbol currentClass, Types types) {
 582         if (currentClass == owner) {
 583             return this;
 584         }
 585         for (Symbol sym : currentClass.members().getSymbolsByName(name)) {
 586             if (sym.kind == kind &&
 587                     (kind != MTH ||
 588                     (sym.flags() & STATIC) != 0 &&
 589                     types.isSubSignature(sym.type, type))) {
 590                 return sym;
 591             }
 592         }
 593         Symbol hiddenSym = null;
 594         for (Type st : types.interfaces(currentClass.type)
 595                 .prepend(types.supertype(currentClass.type))) {
 596             if (st != null && (st.hasTag(CLASS))) {
 597                 Symbol sym = hiddenInInternal((ClassSymbol)st.tsym, types);
 598                 if (sym == this) {
 599                     return this;
 600                 } else if (sym != null) {
 601                     hiddenSym = sym;
 602                 }
 603             }
 604         }
 605         return hiddenSym;
 606     }

 608     /** Is this symbol accessible in a given class?
 609      *  PRE: If symbol's owner is a interface,
 610      *       it is already assumed that the interface is a superinterface
 611      *       the given class.
 612      *  @param clazz  The class for which we want to establish membership.
 613      *                This must be a subclass of the member's owner.
 614      */
 615     public final boolean isAccessibleIn(Symbol clazz, Types types) {
 616         switch ((int)(flags_field & Flags.AccessFlags)) {
 617         default: // error recovery
 618         case PUBLIC:
 619             return true;
 620         case PRIVATE:
 621             return this.owner == clazz;
 622         case PROTECTED:
 623             // we model interfaces as extending Object
 624             return (clazz.flags() & INTERFACE) == 0;
 625         case 0:
 626             PackageSymbol thisPackage = this.packge();
 627             for (Symbol sup = clazz;
 628                  sup != null && sup != this.owner;
 629                  sup = types.supertype(sup.type).tsym) {
 630                 while (sup.type.hasTag(TYPEVAR))
 631                     sup = sup.type.getUpperBound().tsym;
 632                 if (sup.type.isErroneous())
 633                     return true; // error recovery
 634                 if ((sup.flags() & COMPOUND) != 0)
 635                     continue;
 636                 if (sup.packge() != thisPackage)
 637                     return false;
 638             }
 639             return (clazz.flags() & INTERFACE) == 0;
 640         }
 641     }

 835             if (this == that)

2037         /** Does this symbol override `other' symbol, when both are seen as
2038          *  members of class `origin'?  It is assumed that _other is a member
2039          *  of origin.
2040          *
2041          *  Caveat: If `this' is an abstract inherited member of origin, it is
2042          *  deemed to override `other' only when `requireConcreteIfInherited'
2043          *  is false.
2044          *
2045          *  It is assumed that both symbols have the same name.  The static
2046          *  modifier is ignored for this test.
2047          *
2048          *  See JLS 8.4.6.1 (without transitivity) and 8.4.6.4
2049          */
2050         public boolean overrides(Symbol _other, TypeSymbol origin, Types types, boolean checkResult,
2051                                             boolean requireConcreteIfInherited) {
2052             if (isConstructor() || _other.kind != MTH) return false;
2053 
2054             if (this == _other) return true;
2055             MethodSymbol other = (MethodSymbol)_other;
2056 
2057             // check for a direct implementation
2058             if (other.isOverridableIn((TypeSymbol)owner) &&
2059                 types.asSuper(owner.type, other.owner) != null) {
2060                 Type mt = types.memberType(owner.type, this);
2061                 Type ot = types.memberType(owner.type, other);
2062                 if (types.isSubSignature(mt, ot)) {
2063                     if (!checkResult)
2064                         return true;
2065                     if (types.returnTypeSubstitutable(mt, ot))
2066                         return true;
2067                 }
2068             }
2069 
2070             // check for an inherited implementation
2071             if (((flags() & ABSTRACT) != 0 && requireConcreteIfInherited) ||
2072                     ((other.flags() & ABSTRACT) == 0 && (other.flags() & DEFAULT) == 0) ||
2073                     !other.isOverridableIn(origin) ||
2074                     !this.isMemberOf(origin, types))
2075                 return false;
2076 
2077             // assert types.asSuper(origin.type, other.owner) != null;
2078             Type mt = types.memberType(origin.type, this);
2079             Type ot = types.memberType(origin.type, other);

2415         /** Access codes for dereferencing, assignment,
2416          *  and pre/post increment/decrement.
2417 
2418          *  All access codes for accesses to the current class are even.
2419          *  If a member of the superclass should be accessed instead (because
2420          *  access was via a qualified super), add one to the corresponding code
2421          *  for the current class, making the number odd.
2422          *  This numbering scheme is used by the backend to decide whether
2423          *  to issue an invokevirtual or invokespecial call.
2424          *
2425          *  @see Gen#visitSelect(JCFieldAccess tree)
2426          */
2427         public enum AccessCode {
2428             UNKNOWN(-1, Tag.NO_TAG),
2429             DEREF(0, Tag.NO_TAG),
2430             ASSIGN(2, Tag.ASSIGN),
2431             PREINC(4, Tag.PREINC),
2432             PREDEC(6, Tag.PREDEC),
2433             POSTINC(8, Tag.POSTINC),
2434             POSTDEC(10, Tag.POSTDEC),
2435             FIRSTASGOP(12, Tag.NO_TAG);
2436 
2437             public final int code;
2438             public final Tag tag;
2439             public static final int numberOfAccessCodes = (lushrl - ishll + lxor + 2 - iadd) * 2 + FIRSTASGOP.code + 2;

2442                 this.code = code;

2444             }

2446             static public AccessCode getFromCode(int code) {
2447                 for (AccessCode aCodes : AccessCode.values()) {
2448                     if (aCodes.code == code) {
2449                         return aCodes;
2450                     }
2451                 }
2452                 return UNKNOWN;
2453             }

2455             static int from(Tag tag, int opcode) {
2456                 /** Map bytecode of binary operation to access code of corresponding
2457                 *  assignment operation. This is always an even number.
2458                 */
2459                 switch (tag) {
2460                     case PREINC:
2461                         return AccessCode.PREINC.code;
2462                     case PREDEC:
2463                         return AccessCode.PREDEC.code;
2464                     case POSTINC:
2465                         return AccessCode.POSTINC.code;
2466                     case POSTDEC:
2467                         return AccessCode.POSTDEC.code;
2468                 }
2469                 if (iadd <= opcode && opcode <= lxor) {
2470                     return (opcode - iadd) * 2 + FIRSTASGOP.code;
2471                 } else if (opcode == string_add) {
2472                     return (lxor + 1 - iadd) * 2 + FIRSTASGOP.code;
2473                 } else if (ishll <= opcode && opcode <= lushrl) {
2474                     return (opcode - ishll + lxor + 2 - iadd) * 2 + FIRSTASGOP.code;
2475                 }
2476                 return -1;
2477             }
