 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import javax.lang.model.element.ExecutableElement;
 29 import javax.lang.model.element.TypeElement;
 30 
 31 /**
 32  * The interface for writing property output.
 33  *
 34  *  <p><b>This is NOT part of any supported API.
 35  *  If you write code that depends on this, you do so at your own risk.
 36  *  This code and its internal interfaces are subject to change or
 37  *  deletion without notice.</b>
 38  */
 39 
 40 public interface PropertyWriter {
 41 
 42     /**
 43      * Get the property details tree header.
 44      *
 45      * @param memberDetailsTree the content tree representing member details
 46      * @return content tree for the property details header
 47      */
 48     Content getPropertyDetailsTreeHeader(Content memberDetailsTree);
 49 
 50     /**
 51      * Get the property documentation tree header.
 52      *
 53      * @param property the property being documented
 54      * @return content tree for the property documentation header
 55      */
 56     Content getPropertyDocTreeHeader(ExecutableElement property);
 57 
 58     /**
 59      * Get the signature for the given property.
 60      *
 61      * @param property the property being documented
 62      * @return content tree for the property signature
 63      */
 64     Content getSignature(ExecutableElement property);
 65 
 66     /**
 67      * Add the deprecated output for the given property.
 68      *
 69      * @param property the property being documented
 70      * @param propertyDocTree content tree to which the deprecated information will be added
 71      */
 72     void addDeprecated(ExecutableElement property, Content propertyDocTree);
 73 
 74     /**
 75      * Add the comments for the given property.
 76      *
 77      * @param property the property being documented
 78      * @param propertyDocTree the content tree to which the comments will be added
 79      */
 80     void addComments(ExecutableElement property, Content propertyDocTree);
 81 
 82     /**
 83      * Add the tags for the given property.
 84      *
 85      * @param property the property being documented
 86      * @param propertyDocTree the content tree to which the tags will be added
 87      */
 88     void addTags(ExecutableElement property, Content propertyDocTree);
 89 
 90     /**
 91      * Get the property details tree.
 92      *
 93      * @param memberDetailsTreeHeader the content tree representing member details header
 94      * @param memberDetailsTree the content tree representing member details
 95      * @return content tree for the property details
 96      */
 97     Content getPropertyDetails(Content memberDetailsTreeHeader, Content memberDetailsTree);
 98 
 99     /**
100      * Get the property documentation.
101      *
102      * @param propertyDocTree the content tree representing property documentation
103      * @return content tree for the property documentation
104      */
105     Content getPropertyDoc(Content propertyDocTree);
106 
107     /**
108      * Gets the member header tree.
109      *
110      * @return a content tree for the member header
111      */
112     Content getMemberTreeHeader();
113 }
