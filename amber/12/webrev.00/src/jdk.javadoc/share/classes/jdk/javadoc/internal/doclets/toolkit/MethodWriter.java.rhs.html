 26 package jdk.javadoc.internal.doclets.toolkit;
 27 
 28 import javax.lang.model.element.ExecutableElement;

 29 import javax.lang.model.type.TypeMirror;
 30 
 31 /**
 32  * The interface for writing method output.
 33  *
 34  *  <p><b>This is NOT part of any supported API.
 35  *  If you write code that depends on this, you do so at your own risk.
 36  *  This code and its internal interfaces are subject to change or
 37  *  deletion without notice.</b>
 38  */
 39 
 40 public interface MethodWriter extends MemberWriter {
 41 
 42     /**
 43      * Get the method details tree header.
 44      *
 45      * @param memberDetailsTree the content tree representing member details
 46      * @return content tree for the method details header
 47      */
 48     Content getMethodDetailsTreeHeader(Content memberDetailsTree);
 49 
 50     /**
 51      * Get the method documentation tree header.
 52      *
 53      * @param method the method being documented
 54      * @return content tree for the method documentation header
 55      */
 56     Content getMethodDocTreeHeader(ExecutableElement method);
 57 
 58     /**
 59      * Get the signature for the given method.
 60      *
 61      * @param method the method being documented
 62      * @return content tree for the method signature
 63      */
 64     Content getSignature(ExecutableElement method);
 65 
 66     /**
 67      * Add the deprecated output for the given method.
 68      *
 69      * @param method the method being documented
 70      * @param methodDocTree content tree to which the deprecated information will be added
 71      */
 72     void addDeprecated(ExecutableElement method, Content methodDocTree);
 73 
 74     /**
 75      * Add the comments for the given method.
 76      *
 77      * @param holder the holder type (not erasure) of the method
 78      * @param method the method being documented
 79      * @param methodDocTree the content tree to which the comments will be added
 80      */
 81     void addComments(TypeMirror holder, ExecutableElement method, Content methodDocTree);
 82 
 83     /**
 84      * Add the tags for the given method.
 85      *
 86      * @param method the method being documented
 87      * @param methodDocTree the content tree to which the tags will be added
 88      */
 89     void addTags(ExecutableElement method, Content methodDocTree);
 90 
 91     /**
 92      * Get the method details tree.
 93      *
 94      * @param methodDetailsTreeHeader the content tree representing method details header
 95      * @param methodDetailsTree the content tree representing method details
 96      * @return content tree for the method details
 97      */
 98     Content getMethodDetails(Content methodDetailsTreeHeader, Content methodDetailsTree);
 99 








100     /**
101      * Gets the member header tree.
102      *
103      * @return a content tree for the member header
104      */
105     Content getMemberTreeHeader();
106 }
