369     private final String cssName;
370 
371     HtmlStyle() {
372         cssName = Pattern.compile("\\p{Upper}")
373                 .matcher(toString())
374                 .replaceAll(mr -> "-" + mr.group().toLowerCase(Locale.US));
375     }
376 
377     HtmlStyle(String cssName) {
378         this.cssName = cssName;
379     }
380 
381     /**
382      * Returns the CSS class name associated with this style.
383      * @return the CSS class name
384      */
385     public String cssName() {
386         return cssName;
387     }
