 96     /**
 97      * Details about a package in a module.
 98      * A package may be not exported, or exported to some modules, or exported to all modules.
 99      * A package may be not opened, or opened to some modules, or opened to all modules.

113          * Summary of package opens:
114          * If null, the package is not opened to any modules;
115          * if empty, the package is opened to all modules;
116          * otherwise, the package is opened to these modules.
117          */
118         Set<ModuleElement> openedTo;
119     }

244     /**
245      * Compute the modules data that will be displayed in various tables on the module summary page.
246      */
247     public void computeModulesData() {
248         CommentHelper ch = utils.getCommentHelper(mdle);
249         // Get module dependencies using the module's transitive closure.

272 
273         // Get all packages if module is open or if displaying concealed modules
274         for (PackageElement pkg : utils.getModulePackageMap().getOrDefault(mdle, Collections.emptySet())) {
275             if (shouldDocument(pkg) && (mdle.isOpen() || moduleMode == ModuleMode.ALL)) {
276                 PackageEntry e = new PackageEntry();
277                 if (mdle.isOpen()) {
278                     e.openedTo = Collections.emptySet();
279                 }
280                 packages.put(pkg, e);
281             }
282         }
283 
284         // Get all exported packages for the module, using the exports directive for the module.
285         for (ModuleElement.ExportsDirective directive : ElementFilter.exportsIn(mdle.getDirectives())) {
286             PackageElement p = directive.getPackage();
287             if (shouldDocument(p)) {
288                 List<? extends ModuleElement> targetMdles = directive.getTargetModules();
289                 // Include package if in details mode, or exported to all (i.e. targetModules == null)
290                 if (moduleMode == ModuleMode.ALL || targetMdles == null) {
291                     PackageEntry packageEntry = packages.computeIfAbsent(p, pkg -> new PackageEntry());
292                     SortedSet<ModuleElement> mdleList = new TreeSet<>(utils.makeModuleComparator());
293                     if (targetMdles != null) {
294                         mdleList.addAll(targetMdles);
295                     }
296                     packageEntry.exportedTo = mdleList;
297                 }
298             }
299         }
300 
301         // Get all opened packages for the module, using the opens directive for the module.
302         // If it is an open module, there will be no separate opens directives.
303         for (ModuleElement.OpensDirective directive : ElementFilter.opensIn(mdle.getDirectives())) {
304             PackageElement p = directive.getPackage();
305             if (shouldDocument(p)) {
306                 List<? extends ModuleElement> targetMdles = directive.getTargetModules();
307                 // Include package if in details mode, or opened to all (i.e. targetModules == null)
308                 if (moduleMode == ModuleMode.ALL || targetMdles == null) {
309                     PackageEntry packageEntry = packages.computeIfAbsent(p, pkg -> new PackageEntry());
310                     SortedSet<ModuleElement> mdleList = new TreeSet<>(utils.makeModuleComparator());
311                     if (targetMdles != null) {
312                         mdleList.addAll(targetMdles);
313                     }
314                     packageEntry.openedTo = mdleList;
315                 }
316             }
317         }
318 
319         // Get all the exported and opened packages, for the transitive closure of the module, to be displayed in
320         // the indirect packages tables.
321         dependentModules.forEach((module, mod) -> {
322             SortedSet<PackageElement> exportedPackages = new TreeSet<>(utils.makePackageComparator());
323             ElementFilter.exportsIn(module.getDirectives()).forEach(directive -> {
324                 PackageElement pkg = directive.getPackage();
325                 if (shouldDocument(pkg)) {
326                     // Qualified exports are not displayed in API mode
327                     if (moduleMode == ModuleMode.ALL || directive.getTargetModules() == null) {
328                         exportedPackages.add(pkg);
329                     }
330                 }
331             });
332             // If none of the indirect modules have exported packages to be displayed, we should not be
333             // displaying the table and so it should not be added to the map.
334             if (!exportedPackages.isEmpty()) {
335                 indirectPackages.put(module, exportedPackages);
336             }
337             SortedSet<PackageElement> openPackages = new TreeSet<>(utils.makePackageComparator());
338             if (module.isOpen()) {
339                 openPackages.addAll(utils.getModulePackageMap().getOrDefault(module, Collections.emptySet()));
340             } else {
341                 ElementFilter.opensIn(module.getDirectives()).forEach(directive -> {
342                     PackageElement pkg = directive.getPackage();
343                     if (shouldDocument(pkg)) {
344                         // Qualified opens are not displayed in API mode
345                         if (moduleMode == ModuleMode.ALL || directive.getTargetModules() == null) {
346                             openPackages.add(pkg);
347                         }
348                     }
349                 });
350             }
351             // If none of the indirect modules have opened packages to be displayed, we should not be
352             // displaying the table and so it should not be added to the map.
353             if (!openPackages.isEmpty()) {
354                 indirectOpenPackages.put(module, openPackages);
355             }
356         });
357         // Get all the services listed as uses directive.
358         ElementFilter.usesIn(mdle.getDirectives()).forEach(directive -> {
359             TypeElement u = directive.getService();
360             if (shouldDocument(u)) {
361                 uses.add(u);
362             }
363         });
364         // Get all the services and implementations listed as provides directive.
365         ElementFilter.providesIn(mdle.getDirectives()).forEach(directive -> {
366             TypeElement u = directive.getService();
367             if (shouldDocument(u)) {
368                 List<? extends TypeElement> implList = directive.getImplementations();
369                 SortedSet<TypeElement> implSet = new TreeSet<>(utils.makeAllClassesComparator());
370                 implSet.addAll(implList);
371                 provides.put(u, implSet);
372             }
373         });
374         // Generate the map of all services listed using @provides, and the description.
375         utils.getBlockTags(mdle, DocTree.Kind.PROVIDES).forEach(tree -> {
376             TypeElement t = ch.getServiceType(tree);
377             if (t != null) {
378                 providesTrees.put(t, commentTagsToContent(tree, mdle, ch.getDescription(tree), false, true));
379             }
380         });
381         // Generate the map of all services listed using @uses, and the description.
382         utils.getBlockTags(mdle, DocTree.Kind.USES).forEach(tree -> {
383             TypeElement t = ch.getServiceType(tree);
384             if (t != null) {
385                 usesTrees.put(t, commentTagsToContent(tree, mdle, ch.getDescription(tree), false, true));
386             }
387         });
388     }

243     /**
244      * Compute the modules data that will be displayed in various tables on the module summary page.
245      */
246     public void computeModulesData() {
247         CommentHelper ch = utils.getCommentHelper(mdle);
248         // Get module dependencies using the module's transitive closure.

271 
272         // Get all packages if module is open or if displaying concealed modules
273         for (PackageElement pkg : utils.getModulePackageMap().getOrDefault(mdle, Collections.emptySet())) {
274             if (shouldDocument(pkg) && (mdle.isOpen() || moduleMode == ModuleMode.ALL)) {
275                 PackageEntry e = new PackageEntry();
276                 if (mdle.isOpen()) {
277                     e.openedTo = Collections.emptySet();
278                 }
279                 packages.put(pkg, e);
280             }
281         }
282 
283         // Get all exported packages for the module, using the exports directive for the module.
284         for (ModuleElement.ExportsDirective directive : ElementFilter.exportsIn(mdle.getDirectives())) {
285             PackageElement p = directive.getPackage();
286             if (shouldDocument(p)) {
287                 List<? extends ModuleElement> targetMdles = directive.getTargetModules();
288                 // Include package if in details mode, or exported to all (i.e. targetModules == null)
289                 if (moduleMode == ModuleMode.ALL || targetMdles == null) {
290                     PackageEntry packageEntry = packages.computeIfAbsent(p, pkg -> new PackageEntry());
291                     SortedSet<ModuleElement> mdleList = new TreeSet<>(utils.comparators.makeModuleComparator());
292                     if (targetMdles != null) {
293                         mdleList.addAll(targetMdles);
294                     }
295                     packageEntry.exportedTo = mdleList;
296                 }
297             }
298         }
299 
300         // Get all opened packages for the module, using the opens directive for the module.
301         // If it is an open module, there will be no separate opens directives.
302         for (ModuleElement.OpensDirective directive : ElementFilter.opensIn(mdle.getDirectives())) {
303             PackageElement p = directive.getPackage();
304             if (shouldDocument(p)) {
305                 List<? extends ModuleElement> targetMdles = directive.getTargetModules();
306                 // Include package if in details mode, or opened to all (i.e. targetModules == null)
307                 if (moduleMode == ModuleMode.ALL || targetMdles == null) {
308                     PackageEntry packageEntry = packages.computeIfAbsent(p, pkg -> new PackageEntry());
309                     SortedSet<ModuleElement> mdleList = new TreeSet<>(utils.comparators.makeModuleComparator());
310                     if (targetMdles != null) {
311                         mdleList.addAll(targetMdles);
312                     }
313                     packageEntry.openedTo = mdleList;
314                 }
315             }
316         }
317 
318         // Get all the exported and opened packages, for the transitive closure of the module, to be displayed in
319         // the indirect packages tables.
320         dependentModules.forEach((module, mod) -> {
321             SortedSet<PackageElement> exportedPackages = new TreeSet<>(utils.comparators.makePackageComparator());
322             ElementFilter.exportsIn(module.getDirectives()).forEach(directive -> {
323                 PackageElement pkg = directive.getPackage();
324                 if (shouldDocument(pkg)) {
325                     // Qualified exports are not displayed in API mode
326                     if (moduleMode == ModuleMode.ALL || directive.getTargetModules() == null) {
327                         exportedPackages.add(pkg);
328                     }
329                 }
330             });
331             // If none of the indirect modules have exported packages to be displayed, we should not be
332             // displaying the table and so it should not be added to the map.
333             if (!exportedPackages.isEmpty()) {
334                 indirectPackages.put(module, exportedPackages);
335             }
336             SortedSet<PackageElement> openPackages = new TreeSet<>(utils.comparators.makePackageComparator());
337             if (module.isOpen()) {
338                 openPackages.addAll(utils.getModulePackageMap().getOrDefault(module, Collections.emptySet()));
339             } else {
340                 ElementFilter.opensIn(module.getDirectives()).forEach(directive -> {
341                     PackageElement pkg = directive.getPackage();
342                     if (shouldDocument(pkg)) {
343                         // Qualified opens are not displayed in API mode
344                         if (moduleMode == ModuleMode.ALL || directive.getTargetModules() == null) {
345                             openPackages.add(pkg);
346                         }
347                     }
348                 });
349             }
350             // If none of the indirect modules have opened packages to be displayed, we should not be
351             // displaying the table and so it should not be added to the map.
352             if (!openPackages.isEmpty()) {
353                 indirectOpenPackages.put(module, openPackages);
354             }
355         });
356         // Get all the services listed as uses directive.
357         ElementFilter.usesIn(mdle.getDirectives()).forEach(directive -> {
358             TypeElement u = directive.getService();
359             if (shouldDocument(u)) {
360                 uses.add(u);
361             }
362         });
363         // Get all the services and implementations listed as provides directive.
364         ElementFilter.providesIn(mdle.getDirectives()).forEach(directive -> {
365             TypeElement u = directive.getService();
366             if (shouldDocument(u)) {
367                 List<? extends TypeElement> implList = directive.getImplementations();
368                 SortedSet<TypeElement> implSet = new TreeSet<>(utils.comparators.makeAllClassesComparator());
369                 implSet.addAll(implList);
370                 provides.put(u, implSet);
371             }
372         });
373         // Generate the map of all services listed using @provides, and the description.
374         utils.getBlockTags(mdle, DocTree.Kind.PROVIDES).forEach(tree -> {
375             TypeElement t = ch.getServiceType(tree);
376             if (t != null) {
377                 providesTrees.put(t, commentTagsToContent(tree, mdle, ch.getDescription(tree), false, true));
378             }
379         });
380         // Generate the map of all services listed using @uses, and the description.
381         utils.getBlockTags(mdle, DocTree.Kind.USES).forEach(tree -> {
382             TypeElement t = ch.getServiceType(tree);
383             if (t != null) {
384                 usesTrees.put(t, commentTagsToContent(tree, mdle, ch.getDescription(tree), false, true));
385             }
386         });
387     }

748         Content typeLinkContent;
749         Content description;

751             if (!displayServiceDirective(t, usesTrees)) {
752                 continue;
753             }
754             typeLinkContent = getLink(new LinkInfoImpl(configuration, LinkInfoImpl.Kind.PACKAGE, t));
755             Content summary = new ContentBuilder();
756             if (display(usesTrees)) {
757                 description = usesTrees.get(t);
758                 if (description != null && !description.isEmpty()) {

776         SortedSet<TypeElement> implSet;
777         Content description;
778         for (Map.Entry<TypeElement, SortedSet<TypeElement>> entry : provides.entrySet()) {
779             TypeElement srv = entry.getKey();
780             if (!displayServiceDirective(srv, providesTrees)) {
781                 continue;
782             }
783             implSet = entry.getValue();
784             Content srvLinkContent = getLink(new LinkInfoImpl(configuration, LinkInfoImpl.Kind.PACKAGE, srv));
785             Content desc = new ContentBuilder();
786             if (display(providesTrees)) {
787                 description = providesTrees.get(srv);

789                         ? HtmlTree.DIV(HtmlStyle.block, description)
790                         : Entity.NO_BREAK_SPACE);
791             } else {
792                 desc.add(Entity.NO_BREAK_SPACE);
793                 }
794             // Only display the implementation details in the "all" mode.
795             if (moduleMode == ModuleMode.ALL && !implSet.isEmpty()) {
796                 desc.add(new HtmlTree(TagName.BR));
797                 desc.add("(");
798                 HtmlTree implSpan = HtmlTree.SPAN(HtmlStyle.implementationLabel, contents.implementation);
799                 desc.add(implSpan);
800                 desc.add(Entity.NO_BREAK_SPACE);
801                 String sep = "";
802                 for (TypeElement impl : implSet) {
803                     desc.add(sep);
804                     desc.add(getLink(new LinkInfoImpl(configuration, LinkInfoImpl.Kind.PACKAGE, impl)));
805                     sep = ", ";
806                 }
807                 desc.add(")");
808             }
809             table.addRow(srvLinkContent, desc);
810         }
811     }
