28 /**
29  * Common behavior for writing members of a type.
30  *
31  *  <p><b>This is NOT part of any supported API.
32  *  If you write code that depends on this, you do so at your own risk.
33  *  This code and its internal interfaces are subject to change or
34  *  deletion without notice.</b>
35  */
36 public interface MemberWriter {
37 
38     /**
39      * Returns a list to be used for the members of a given kind.
40      *
41      * @return a list to be used for the members of a given kind
42      * @see #getMemberListItem(Content)
43      */
44     Content getMemberList();
45 
46     /**
47      * Returns an item for the list of elements of a given kind.
48      *
49      * @param content the content tree of the member to be documented
50      * @return an item for the list of elements of a given kind
51      * @see #getMemberList()
52      */
53     Content getMemberListItem(Content content);
54 }
