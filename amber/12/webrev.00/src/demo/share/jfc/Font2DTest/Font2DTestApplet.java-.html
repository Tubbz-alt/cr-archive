58 /// Applet version of Font2DTest that wraps the actual demo
59 
60 public final class Font2DTestApplet extends JApplet {
61     public void init() {
62         /// Check if necessary permission is given...
63         SecurityManager security = System.getSecurityManager();
64         if ( security != null ) {
65             try {
66                 security.checkPermission( new AWTPermission( "showWindowWithoutWarningBanner" ));
67             }
68             catch ( SecurityException e ) {
69                 System.out.println( "NOTE: showWindowWithoutWarningBanner AWTPermission not given.\n" +
70                                     "Zoom window will contain warning banner at bottom when shown\n" );
71             }
72             try {
73                 security.checkPrintJobAccess();
74             }
75             catch ( SecurityException e ) {
76                 System.out.println( "NOTE: queuePrintJob RuntimePermission not given.\n" +
77                                     "Printing feature will not be available\n" );
78             }
79         }
80 
81         final JFrame f = new JFrame( "Font2DTest" );
82         final Font2DTest f2dt = new Font2DTest( f, true );
83         f.addWindowListener( new WindowAdapter() {
84             public void windowClosing( WindowEvent e ) { f.dispose(); }
85         });
86 
87         f.getContentPane().add( f2dt );
88         f.pack();
89         f.show();
90     }
91 }
