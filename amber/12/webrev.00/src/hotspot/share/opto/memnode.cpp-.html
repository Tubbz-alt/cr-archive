 106 void MemNode::dump_adr_type(const Node* mem, const TypePtr* adr_type, outputStream *st) {
 107   st->print(" @");
 108   if (adr_type == NULL) {
 109     st->print("NULL");
 110   } else {
 111     adr_type->dump_on(st);
 112     Compile* C = Compile::current();
 113     Compile::AliasType* atp = NULL;
 114     if (C->have_alias_type(adr_type))  atp = C->alias_type(adr_type);
 115     if (atp == NULL)
 116       st->print(", idx=?\?;");
 117     else if (atp->index() == Compile::AliasIdxBot)
 118       st->print(", idx=Bot;");
 119     else if (atp->index() == Compile::AliasIdxTop)
 120       st->print(", idx=Top;");
 121     else if (atp->index() == Compile::AliasIdxRaw)
 122       st->print(", idx=Raw;");
 123     else {
 124       ciField* field = atp->field();
 125       if (field) {
 126         st->print(", name=");
 127         field->print_name_on(st);
 128       }
 129       st->print(", idx=%d;", atp->index());
 130     }
 131   }
 132 }

 138 Node *MemNode::optimize_simple_memory_chain(Node *mchain, const TypeOopPtr *t_oop, Node *load, PhaseGVN *phase) {
 139   assert((t_oop != NULL), "sanity");
 140   bool is_instance = t_oop->is_known_instance_field();
 141   bool is_boxed_value_load = t_oop->is_ptr_to_boxed_value() &&
 142                              (load != NULL) && load->is_Load() &&
 143                              (phase->is_IterGVN() != NULL);
 144   if (!(is_instance || is_boxed_value_load))
 145     return mchain;  // don't try to optimize non-instance types
 146   uint instance_id = t_oop->instance_id();
 147   Node *start_mem = phase->C->start()->proj_out_or_null(TypeFunc::Memory);
 148   Node *prev = NULL;
 149   Node *result = mchain;
 150   while (prev != result) {
 151     prev = result;
 152     if (result == start_mem)
 153       break;  // hit one of our sentinels
 154     // skip over a call which does not affect this memory slice
 155     if (result->is_Proj() && result->as_Proj()->_con == TypeFunc::Memory) {
 156       Node *proj_in = result->in(0);
 157       if (proj_in->is_Allocate() && proj_in->_idx == instance_id) {
 158         break;  // hit one of our sentinels
 159       } else if (proj_in->is_Call()) {
 160         // ArrayCopyNodes processed here as well
 161         CallNode *call = proj_in->as_Call();
 162         if (!call->may_modify(t_oop, phase)) { // returns false for instances
 163           result = call->in(TypeFunc::Memory);
 164         }
 165       } else if (proj_in->is_Initialize()) {
 166         AllocateNode* alloc = proj_in->as_Initialize()->allocation();
 167         // Stop if this is the initialization for the object instance which
 168         // contains this memory slice, otherwise skip over it.
 169         if ((alloc == NULL) || (alloc->_idx == instance_id)) {
 170           break;
 171         }
 172         if (is_instance) {
 173           result = proj_in->in(TypeFunc::Memory);
 174         } else if (is_boxed_value_load) {
 175           Node* klass = alloc->in(AllocateNode::KlassNode);
 176           const TypeKlassPtr* tklass = phase->type(klass)->is_klassptr();
 177           if (tklass->klass_is_exact() && !tklass->klass()->equals(t_oop->klass())) {
 178             result = proj_in->in(TypeFunc::Memory); // not related allocation
 179           }
 180         }
 181       } else if (proj_in->is_MemBar()) {
 182         ArrayCopyNode* ac = NULL;
 183         if (ArrayCopyNode::may_modify(t_oop, proj_in->as_MemBar(), phase, ac)) {
 184           break;
 185         }
 186         result = proj_in->in(TypeFunc::Memory);
 187       } else {
 188         assert(false, "unexpected projection");
 189       }
 190     } else if (result->is_ClearArray()) {
 191       if (!is_instance || !ClearArrayNode::step_through(&result, instance_id, phase)) {
 192         // Can not bypass initialization of the instance
 193         // we are looking for.
 194         break;
 195       }
 196       // Otherwise skip it (the call updated 'result' value).
 197     } else if (result->is_MergeMem()) {
 198       result = step_through_mergemem(phase, result->as_MergeMem(), t_oop, NULL, tty);
 199     }
 200   }
 201   return result;
 202 }

 204 Node *MemNode::optimize_memory_chain(Node *mchain, const TypePtr *t_adr, Node *load, PhaseGVN *phase) {
 205   const TypeOopPtr* t_oop = t_adr->isa_oopptr();
 206   if (t_oop == NULL)
 207     return mchain;  // don't try to optimize non-oop types
 208   Node* result = optimize_simple_memory_chain(mchain, t_oop, load, phase);
 209   bool is_instance = t_oop->is_known_instance_field();
 210   PhaseIterGVN *igvn = phase->is_IterGVN();
 211   if (is_instance && igvn != NULL && result->is_Phi()) {
 212     PhiNode *mphi = result->as_Phi();
 213     assert(mphi->bottom_type() == Type::MEMORY, "memory phi required");
 214     const TypePtr *t = mphi->adr_type();
 215     if (t == TypePtr::BOTTOM || t == TypeRawPtr::BOTTOM ||
 216         (t->isa_oopptr() && !t->is_oopptr()->is_known_instance() &&
 217          t->is_oopptr()->cast_to_exactness(true)
 218            ->is_oopptr()->cast_to_ptr_type(t_oop->ptr())
 219             ->is_oopptr()->cast_to_instance_id(t_oop->instance_id()) == t_oop)) {
 220       // clone the Phi with our address type
 221       result = mphi->split_out_instance(t_adr, igvn);
 222     } else {
 223       assert(phase->C->get_alias_index(t) == phase->C->get_alias_index(t_adr), "correct memory chain");
 224     }
 225   }
 226   return result;
 227 }

 229 static Node *step_through_mergemem(PhaseGVN *phase, MergeMemNode *mmem,  const TypePtr *tp, const TypePtr *adr_check, outputStream *st) {
 230   uint alias_idx = phase->C->get_alias_index(tp);
 231   Node *mem = mmem;
 232 #ifdef ASSERT
 233   {
 234     // Check that current type is consistent with the alias index used during graph construction
 235     assert(alias_idx >= Compile::AliasIdxRaw, "must not be a bad alias_idx");
 236     bool consistent =  adr_check == NULL || adr_check->empty() ||
 237                        phase->C->must_alias(adr_check, alias_idx );
 238     // Sometimes dead array references collapse to a[-1], a[-2], or a[-3]
 239     if( !consistent && adr_check != NULL && !adr_check->empty() &&
 240                tp->isa_aryptr() &&        tp->offset() == Type::OffsetBot &&
 241         adr_check->isa_aryptr() && adr_check->offset() != Type::OffsetBot &&
 242         ( adr_check->offset() == arrayOopDesc::length_offset_in_bytes() ||
 243           adr_check->offset() == oopDesc::klass_offset_in_bytes() ||
 244           adr_check->offset() == oopDesc::mark_offset_in_bytes() ) ) {
 245       // don't assert if it is dead code.
 246       consistent = true;
 247     }
 248     if( !consistent ) {
 249       st->print("alias_idx==%d, adr_check==", alias_idx);
 250       if( adr_check == NULL ) {
 251         st->print("NULL");
 252       } else {
 253         adr_check->dump();
 254       }
 255       st->cr();
 256       print_alias_types();
 257       assert(consistent, "adr_check must match alias idx");
 258     }
 259   }
 260 #endif
 261   // TypeOopPtr::NOTNULL+any is an OOP with unknown offset - generally
 262   // means an array I have not precisely typed yet.  Do not do any
 263   // alias stuff with it any time soon.
 264   const TypeOopPtr *toop = tp->isa_oopptr();
 265   if( tp->base() != Type::AnyPtr &&
 266       !(toop &&
 267         toop->klass() != NULL &&
 268         toop->klass()->is_java_lang_Object() &&
 269         toop->offset() == Type::OffsetBot) ) {
 270     // compress paths and change unreachable cycles to TOP
 271     // If not, we can update the input infinitely along a MergeMem cycle
 272     // Equivalent code in PhiNode::Ideal
 273     Node* m  = phase->transform(mmem);
 274     // If transformed to a MergeMem, get the desired slice
 275     // Otherwise the returned node represents memory for every slice
 276     mem = (m->is_MergeMem())? m->as_MergeMem()->memory_at(alias_idx) : m;
 277     // Update input if it is progress over what we have now
 278   }
 279   return mem;
 280 }

 282 //--------------------------Ideal_common---------------------------------------
 283 // Look for degenerate control and memory inputs.  Bypass MergeMem inputs.
 284 // Unhook non-raw memories from complete (macro-expanded) initializations.
 285 Node *MemNode::Ideal_common(PhaseGVN *phase, bool can_reshape) {
 286   // If our control input is a dead region, kill all below the region
 287   Node *ctl = in(MemNode::Control);
 288   if (ctl && remove_dead_region(phase, can_reshape))
 289     return this;
 290   ctl = in(MemNode::Control);
 291   // Don't bother trying to transform a dead node
 292   if (ctl && ctl->is_top())  return NodeSentinel;
 293 
 294   PhaseIterGVN *igvn = phase->is_IterGVN();
 295   // Wait if control on the worklist.
 296   if (ctl && can_reshape && igvn != NULL) {
 297     Node* bol = NULL;
 298     Node* cmp = NULL;
 299     if (ctl->in(0)->is_If()) {
 300       assert(ctl->is_IfTrue() || ctl->is_IfFalse(), "sanity");
 301       bol = ctl->in(0)->in(1);
 302       if (bol->is_Bool())
 303         cmp = ctl->in(0)->in(1)->in(1);
 304     }
 305     if (igvn->_worklist.member(ctl) ||
 306         (bol != NULL && igvn->_worklist.member(bol)) ||
 307         (cmp != NULL && igvn->_worklist.member(cmp)) ) {
 308       // This control path may be dead.
 309       // Delay this memory node transformation until the control is processed.
 310       phase->is_IterGVN()->_worklist.push(this);
 311       return NodeSentinel; // caller will return NULL
 312     }
 313   }
 314   // Ignore if memory is dead, or self-loop
 315   Node *mem = in(MemNode::Memory);
 316   if (phase->type( mem ) == Type::TOP) return NodeSentinel; // caller will return NULL
 317   assert(mem != this, "dead loop in MemNode::Ideal");
 318 
 319   if (can_reshape && igvn != NULL && igvn->_worklist.member(mem)) {
 320     // This memory slice may be dead.
 321     // Delay this mem node transformation until the memory is processed.
 322     phase->is_IterGVN()->_worklist.push(this);
 323     return NodeSentinel; // caller will return NULL
 324   }
 325 
 326   Node *address = in(MemNode::Address);
 327   const Type *t_adr = phase->type(address);
 328   if (t_adr == Type::TOP)              return NodeSentinel; // caller will return NULL
 329 
 330   if (can_reshape && is_unsafe_access() && (t_adr == TypePtr::NULL_PTR)) {
 331     // Unsafe off-heap access with zero address. Remove access and other control users
 332     // to not confuse optimizations and add a HaltNode to fail if this is ever executed.
 333     assert(ctl != NULL, "unsafe accesses should be control dependent");
 334     for (DUIterator_Fast imax, i = ctl->fast_outs(imax); i < imax; i++) {
 335       Node* u = ctl->fast_out(i);
 336       if (u != ctl) {
 337         igvn->rehash_node_delayed(u);
 338         int nb = u->replace_edge(ctl, phase->C->top());
 339         --i, imax -= nb;
 340       }
 341     }
 342     Node* frame = igvn->transform(new ParmNode(phase->C->start(), TypeFunc::FramePtr));
 343     Node* halt = igvn->transform(new HaltNode(ctl, frame, "unsafe off-heap access with zero address"));
 344     phase->C->root()->add_req(halt);
 345     return this;
 346   }
 347 
 348   if (can_reshape && igvn != NULL &&
 349       (igvn->_worklist.member(address) ||
 350        (igvn->_worklist.size() > 0 && t_adr != adr_type())) ) {
 351     // The address's base and type may change when the address is processed.
 352     // Delay this mem node transformation until the address is processed.
 353     phase->is_IterGVN()->_worklist.push(this);
 354     return NodeSentinel; // caller will return NULL
 355   }
 356 
 357   // Do NOT remove or optimize the next lines: ensure a new alias index
 358   // is allocated for an oop pointer type before Escape Analysis.
 359   // Note: C++ will not remove it since the call has side effect.
 360   if (t_adr->isa_oopptr()) {
 361     int alias_idx = phase->C->get_alias_index(t_adr->is_ptr());
 362   }
 363 
 364   Node* base = NULL;
 365   if (address->is_AddP()) {
 366     base = address->in(AddPNode::Base);
 367   }
 368   if (base != NULL && phase->type(base)->higher_equal(TypePtr::NULL_PTR) &&
 369       !t_adr->isa_rawptr()) {
 370     // Note: raw address has TOP base and top->higher_equal(TypePtr::NULL_PTR) is true.
 371     // Skip this node optimization if its address has TOP base.
 372     return NodeSentinel; // caller will return NULL
 373   }
 374 
 375   // Avoid independent memory operations
 376   Node* old_mem = mem;
 377 
 378   // The code which unhooks non-raw memories from complete (macro-expanded)
 379   // initializations was removed. After macro-expansion all stores catched
 380   // by Initialize node became raw stores and there is no information
 381   // which memory slices they modify. So it is unsafe to move any memory
 382   // operation above these stores. Also in most cases hooked non-raw memories
 383   // were already unhooked by using information from detect_ptr_independence()
 384   // and find_previous_store().
 385 
 386   if (mem->is_MergeMem()) {
 387     MergeMemNode* mmem = mem->as_MergeMem();
 388     const TypePtr *tp = t_adr->is_ptr();
 389 
 390     mem = step_through_mergemem(phase, mmem, tp, adr_type(), tty);
 391   }
 392 
 393   if (mem != old_mem) {
 394     set_req(MemNode::Memory, mem);
 395     if (can_reshape && old_mem->outcnt() == 0 && igvn != NULL) {
 396       igvn->_worklist.push(old_mem);
 397     }
 398     if (phase->type(mem) == Type::TOP) return NodeSentinel;
 399     return this;
 400   }
 401 
 402   // let the subclass continue analyzing...
 403   return NULL;
 404 }

 406 // Helper function for proving some simple control dominations.
 407 // Attempt to prove that all control inputs of 'dom' dominate 'sub'.
 408 // Already assumes that 'dom' is available at 'sub', and that 'sub'
 409 // is not a constant (dominated by the method's StartNode).
 410 // Used by MemNode::find_previous_store to prove that the
 411 // control input of a memory operation predates (dominates)
 412 // an allocation it wants to look past.
 413 bool MemNode::all_controls_dominate(Node* dom, Node* sub) {
 414   if (dom == NULL || dom->is_top() || sub == NULL || sub->is_top())
 415     return false; // Conservative answer for dead code
 416 
 417   // Check 'dom'. Skip Proj and CatchProj nodes.
 418   dom = dom->find_exact_control(dom);
 419   if (dom == NULL || dom->is_top())
 420     return false; // Conservative answer for dead code
 421 
 422   if (dom == sub) {
 423     // For the case when, for example, 'sub' is Initialize and the original
 424     // 'dom' is Proj node of the 'sub'.
 425     return false;
 426   }
 427 
 428   if (dom->is_Con() || dom->is_Start() || dom->is_Root() || dom == sub)
 429     return true;
 430 
 431   // 'dom' dominates 'sub' if its control edge and control edges
 432   // of all its inputs dominate or equal to sub's control edge.
 433 
 434   // Currently 'sub' is either Allocate, Initialize or Start nodes.
 435   // Or Region for the check in LoadNode::Ideal();
 436   // 'sub' should have sub->in(0) != NULL.
 437   assert(sub->is_Allocate() || sub->is_Initialize() || sub->is_Start() ||
 438          sub->is_Region() || sub->is_Call(), "expecting only these nodes");
 439 
 440   // Get control edge of 'sub'.
 441   Node* orig_sub = sub;
 442   sub = sub->find_exact_control(sub->in(0));
 443   if (sub == NULL || sub->is_top())
 444     return false; // Conservative answer for dead code
 445 
 446   assert(sub->is_CFG(), "expecting control");
 447 
 448   if (sub == dom)
 449     return true;
 450 
 451   if (sub->is_Start() || sub->is_Root())
 452     return false;
 453 
 454   {
 455     // Check all control edges of 'dom'.
 456 
 457     ResourceMark rm;
 458     Arena* arena = Thread::current()->resource_area();
 459     Node_List nlist(arena);
 460     Unique_Node_List dom_list(arena);
 461 
 462     dom_list.push(dom);
 463     bool only_dominating_controls = false;
 464 
 465     for (uint next = 0; next < dom_list.size(); next++) {
 466       Node* n = dom_list.at(next);
 467       if (n == orig_sub)
 468         return false; // One of dom's inputs dominated by sub.
 469       if (!n->is_CFG() && n->pinned()) {
 470         // Check only own control edge for pinned non-control nodes.
 471         n = n->find_exact_control(n->in(0));
 472         if (n == NULL || n->is_top())
 473           return false; // Conservative answer for dead code
 474         assert(n->is_CFG(), "expecting control");
 475         dom_list.push(n);
 476       } else if (n->is_Con() || n->is_Start() || n->is_Root()) {
 477         only_dominating_controls = true;
 478       } else if (n->is_CFG()) {
 479         if (n->dominates(sub, nlist))
 480           only_dominating_controls = true;
 481         else
 482           return false;
 483       } else {
 484         // First, own control edge.
 485         Node* m = n->find_exact_control(n->in(0));
 486         if (m != NULL) {
 487           if (m->is_top())
 488             return false; // Conservative answer for dead code
 489           dom_list.push(m);
 490         }
 491         // Now, the rest of edges.
 492         uint cnt = n->req();
 493         for (uint i = 1; i < cnt; i++) {
 494           m = n->find_exact_control(n->in(i));
 495           if (m == NULL || m->is_top())
 496             continue;
 497           dom_list.push(m);
 498         }
 499       }
 500     }
 501     return only_dominating_controls;
 502   }
 503 }

 505 //---------------------detect_ptr_independence---------------------------------
 506 // Used by MemNode::find_previous_store to prove that two base
 507 // pointers are never equal.
 508 // The pointers are accompanied by their associated allocations,
 509 // if any, which have been previously discovered by the caller.
 510 bool MemNode::detect_ptr_independence(Node* p1, AllocateNode* a1,
 511                                       Node* p2, AllocateNode* a2,
 512                                       PhaseTransform* phase) {
 513   // Attempt to prove that these two pointers cannot be aliased.
 514   // They may both manifestly be allocations, and they should differ.
 515   // Or, if they are not both allocations, they can be distinct constants.
 516   // Otherwise, one is an allocation and the other a pre-existing value.
 517   if (a1 == NULL && a2 == NULL) {           // neither an allocation
 518     return (p1 != p2) && p1->is_Con() && p2->is_Con();
 519   } else if (a1 != NULL && a2 != NULL) {    // both allocations
 520     return (a1 != a2);
 521   } else if (a1 != NULL) {                  // one allocation a1
 522     // (Note:  p2->is_Con implies p2->in(0)->is_Root, which dominates.)
 523     return all_controls_dominate(p2, a1);
 524   } else { //(a2 != NULL)                   // one allocation a2
 525     return all_controls_dominate(p1, a2);
 526   }
 527   return false;
 528 }

 531 // Find an arraycopy that must have set (can_see_stored_value=true) or
 532 // could have set (can_see_stored_value=false) the value for this load
 533 Node* LoadNode::find_previous_arraycopy(PhaseTransform* phase, Node* ld_alloc, Node*& mem, bool can_see_stored_value) const {
 534   if (mem->is_Proj() && mem->in(0) != NULL && (mem->in(0)->Opcode() == Op_MemBarStoreStore ||
 535                                                mem->in(0)->Opcode() == Op_MemBarCPUOrder)) {
 536     Node* mb = mem->in(0);
 537     if (mb->in(0) != NULL && mb->in(0)->is_Proj() &&
 538         mb->in(0)->in(0) != NULL && mb->in(0)->in(0)->is_ArrayCopy()) {
 539       ArrayCopyNode* ac = mb->in(0)->in(0)->as_ArrayCopy();
 540       if (ac->is_clonebasic()) {
 541         intptr_t offset;
 542         AllocateNode* alloc = AllocateNode::Ideal_allocation(ac->in(ArrayCopyNode::Dest), phase, offset);
 543         if (alloc != NULL && alloc == ld_alloc) {
 544           return ac;
 545         }
 546       }
 547     }
 548   } else if (mem->is_Proj() && mem->in(0) != NULL && mem->in(0)->is_ArrayCopy()) {
 549     ArrayCopyNode* ac = mem->in(0)->as_ArrayCopy();
 550 
 551     if (ac->is_arraycopy_validated() ||
 552         ac->is_copyof_validated() ||
 553         ac->is_copyofrange_validated()) {
 554       Node* ld_addp = in(MemNode::Address);
 555       if (ld_addp->is_AddP()) {
 556         Node* ld_base = ld_addp->in(AddPNode::Address);
 557         Node* ld_offs = ld_addp->in(AddPNode::Offset);
 558 
 559         Node* dest = ac->in(ArrayCopyNode::Dest);
 560 
 561         if (dest == ld_base) {
 562           const TypeX *ld_offs_t = phase->type(ld_offs)->isa_intptr_t();
 563           if (ac->modifies(ld_offs_t->_lo, ld_offs_t->_hi, phase, can_see_stored_value)) {
 564             return ac;
 565           }
 566           if (!can_see_stored_value) {
 567             mem = ac->in(TypeFunc::Memory);
 568           }
 569         }
 570       }
 571     }
 572   }
 573   return NULL;

 576 // The logic for reordering loads and stores uses four steps:
 577 // (a) Walk carefully past stores and initializations which we
 578 //     can prove are independent of this load.
 579 // (b) Observe that the next memory state makes an exact match
 580 //     with self (load or store), and locate the relevant store.
 581 // (c) Ensure that, if we were to wire self directly to the store,
 582 //     the optimizer would fold it up somehow.
 583 // (d) Do the rewiring, and return, depending on some other part of
 584 //     the optimizer to fold up the load.
 585 // This routine handles steps (a) and (b).  Steps (c) and (d) are
 586 // specific to loads and stores, so they are handled by the callers.
 587 // (Currently, only LoadNode::Ideal has steps (c), (d).  More later.)
 588 //
 589 Node* MemNode::find_previous_store(PhaseTransform* phase) {
 590   Node*         ctrl   = in(MemNode::Control);
 591   Node*         adr    = in(MemNode::Address);
 592   intptr_t      offset = 0;
 593   Node*         base   = AddPNode::Ideal_base_and_offset(adr, phase, offset);
 594   AllocateNode* alloc  = AllocateNode::Ideal_allocation(base, phase);
 595 
 596   if (offset == Type::OffsetBot)
 597     return NULL;            // cannot unalias unless there are precise offsets
 598 
 599   const bool adr_maybe_raw = check_if_adr_maybe_raw(adr);
 600   const TypeOopPtr *addr_t = adr->bottom_type()->isa_oopptr();
 601 
 602   intptr_t size_in_bytes = memory_size();
 603 
 604   Node* mem = in(MemNode::Memory);   // start searching here...
 605 
 606   int cnt = 50;             // Cycle limiter
 607   for (;;) {                // While we can dance past unrelated stores...
 608     if (--cnt < 0)  break;  // Caught in cycle or a complicated dance?
 609 
 610     Node* prev = mem;
 611     if (mem->is_Store()) {
 612       Node* st_adr = mem->in(MemNode::Address);
 613       intptr_t st_offset = 0;
 614       Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_offset);
 615       if (st_base == NULL)
 616         break;              // inscrutable pointer
 617 
 618       // For raw accesses it's not enough to prove that constant offsets don't intersect.
 619       // We need the bases to be the equal in order for the offset check to make sense.
 620       if ((adr_maybe_raw || check_if_adr_maybe_raw(st_adr)) && st_base != base) {
 621         break;
 622       }
 623 
 624       if (st_offset != offset && st_offset != Type::OffsetBot) {
 625         const int MAX_STORE = BytesPerLong;
 626         if (st_offset >= offset + size_in_bytes ||
 627             st_offset <= offset - MAX_STORE ||
 628             st_offset <= offset - mem->as_Store()->memory_size()) {
 629           // Success:  The offsets are provably independent.
 630           // (You may ask, why not just test st_offset != offset and be done?
 631           // The answer is that stores of different sizes can co-exist
 632           // in the same sequence of RawMem effects.  We sometimes initialize
 633           // a whole 'tile' of array elements with a single jint or jlong.)
 634           mem = mem->in(MemNode::Memory);
 635           continue;           // (a) advance through independent store memory
 636         }
 637       }
 638       if (st_base != base &&
 639           detect_ptr_independence(base, alloc,
 640                                   st_base,
 641                                   AllocateNode::Ideal_allocation(st_base, phase),
 642                                   phase)) {
 643         // Success:  The bases are provably independent.
 644         mem = mem->in(MemNode::Memory);
 645         continue;           // (a) advance through independent store memory
 646       }
 647 
 648       // (b) At this point, if the bases or offsets do not agree, we lose,
 649       // since we have not managed to prove 'this' and 'mem' independent.
 650       if (st_base == base && st_offset == offset) {
 651         return mem;         // let caller handle steps (c), (d)
 652       }
 653 
 654     } else if (mem->is_Proj() && mem->in(0)->is_Initialize()) {
 655       InitializeNode* st_init = mem->in(0)->as_Initialize();
 656       AllocateNode*  st_alloc = st_init->allocation();
 657       if (st_alloc == NULL)
 658         break;              // something degenerated
 659       bool known_identical = false;
 660       bool known_independent = false;
 661       if (alloc == st_alloc)
 662         known_identical = true;
 663       else if (alloc != NULL)
 664         known_independent = true;
 665       else if (all_controls_dominate(this, st_alloc))
 666         known_independent = true;
 667 
 668       if (known_independent) {
 669         // The bases are provably independent: Either they are
 670         // manifestly distinct allocations, or else the control
 671         // of this load dominates the store's allocation.
 672         int alias_idx = phase->C->get_alias_index(adr_type());
 673         if (alias_idx == Compile::AliasIdxRaw) {
 674           mem = st_alloc->in(TypeFunc::Memory);
 675         } else {
 676           mem = st_init->memory(alias_idx);
 677         }
 678         continue;           // (a) advance through independent store memory
 679       }
 680 
 681       // (b) at this point, if we are not looking at a store initializing
 682       // the same allocation we are loading from, we lose.
 683       if (known_identical) {
 684         // From caller, can_see_stored_value will consult find_captured_store.
 685         return mem;         // let caller handle steps (c), (d)
 686       }
 687 
 688     } else if (find_previous_arraycopy(phase, alloc, mem, false) != NULL) {
 689       if (prev != mem) {
 690         // Found an arraycopy but it doesn't affect that load
 691         continue;
 692       }
 693       // Found an arraycopy that may affect that load
 694       return mem;
 695     } else if (addr_t != NULL && addr_t->is_known_instance_field()) {
 696       // Can't use optimize_simple_memory_chain() since it needs PhaseGVN.
 697       if (mem->is_Proj() && mem->in(0)->is_Call()) {
 698         // ArrayCopyNodes processed here as well.
 699         CallNode *call = mem->in(0)->as_Call();
 700         if (!call->may_modify(addr_t, phase)) {
 701           mem = call->in(TypeFunc::Memory);
 702           continue;         // (a) advance through independent call memory
 703         }
 704       } else if (mem->is_Proj() && mem->in(0)->is_MemBar()) {
 705         ArrayCopyNode* ac = NULL;
 706         if (ArrayCopyNode::may_modify(addr_t, mem->in(0)->as_MemBar(), phase, ac)) {
 707           break;
 708         }
 709         mem = mem->in(0)->in(TypeFunc::Memory);
 710         continue;           // (a) advance through independent MemBar memory
 711       } else if (mem->is_ClearArray()) {
 712         if (ClearArrayNode::step_through(&mem, (uint)addr_t->instance_id(), phase)) {
 713           // (the call updated 'mem' value)
 714           continue;         // (a) advance through independent allocation memory
 715         } else {
 716           // Can not bypass initialization of the instance
 717           // we are looking for.
 718           return mem;
 719         }
 720       } else if (mem->is_MergeMem()) {
 721         int alias_idx = phase->C->get_alias_index(adr_type());
 722         mem = mem->as_MergeMem()->memory_at(alias_idx);
 723         continue;           // (a) advance through independent MergeMem memory
 724       }
 725     }
 726 
 727     // Unless there is an explicit 'continue', we must bail out here,
 728     // because 'mem' is an inscrutable memory state (e.g., a call).
 729     break;
 730   }
 731 
 732   return NULL;              // bail out
 733 }

 736 // Helper function.  Notices when the given type of address hits top or bottom.
 737 // Also, asserts a cross-check of the type against the expected address type.
 738 const TypePtr* MemNode::calculate_adr_type(const Type* t, const TypePtr* cross_check) {
 739   if (t == Type::TOP)  return NULL; // does not touch memory any more?
 740   #ifdef ASSERT

 742   #endif

 744   if (tp == NULL) {
 745     assert(cross_check == NULL || cross_check == TypePtr::BOTTOM, "expected memory type must be wide");
 746     return TypePtr::BOTTOM;           // touches lots of memory
 747   } else {
 748     #ifdef ASSERT
 749     // %%%% [phh] We don't check the alias index if cross_check is
 750     //            TypeRawPtr::BOTTOM.  Needs to be investigated.
 751     if (cross_check != NULL &&
 752         cross_check != TypePtr::BOTTOM &&
 753         cross_check != TypeRawPtr::BOTTOM) {
 754       // Recheck the alias index, to see if it has changed (due to a bug).

 763              "real address must not escape from expected memory type");
 764     }
 765     #endif
 766     return tp;
 767   }

 925 // Is the value loaded previously stored by an arraycopy? If so return
 926 // a load node that reads from the source array so we may be able to
 927 // optimize out the ArrayCopy node later.
 928 Node* LoadNode::can_see_arraycopy_value(Node* st, PhaseGVN* phase) const {
 929   Node* ld_adr = in(MemNode::Address);
 930   intptr_t ld_off = 0;
 931   AllocateNode* ld_alloc = AllocateNode::Ideal_allocation(ld_adr, phase, ld_off);
 932   Node* ac = find_previous_arraycopy(phase, ld_alloc, st, true);
 933   if (ac != NULL) {
 934     assert(ac->is_ArrayCopy(), "what kind of node can this be?");
 935 
 936     Node* mem = ac->in(TypeFunc::Memory);
 937     Node* ctl = ac->in(0);
 938     Node* src = ac->in(ArrayCopyNode::Src);
 939 
 940     if (!ac->as_ArrayCopy()->is_clonebasic() && !phase->type(src)->isa_aryptr()) {
 941       return NULL;
 942     }
 943 
 944     LoadNode* ld = clone()->as_Load();
 945     Node* addp = in(MemNode::Address)->clone();
 946     if (ac->as_ArrayCopy()->is_clonebasic()) {
 947       assert(ld_alloc != NULL, "need an alloc");
 948       assert(addp->is_AddP(), "address must be addp");
 949       assert(ac->in(ArrayCopyNode::Dest)->is_AddP(), "dest must be an address");

 951       assert(bs->step_over_gc_barrier(addp->in(AddPNode::Base)) == bs->step_over_gc_barrier(ac->in(ArrayCopyNode::Dest)->in(AddPNode::Base)), "strange pattern");
 952       assert(bs->step_over_gc_barrier(addp->in(AddPNode::Address)) == bs->step_over_gc_barrier(ac->in(ArrayCopyNode::Dest)->in(AddPNode::Address)), "strange pattern");
 953       addp->set_req(AddPNode::Base, src->in(AddPNode::Base));
 954       addp->set_req(AddPNode::Address, src->in(AddPNode::Address));
 955     } else {
 956       assert(ac->as_ArrayCopy()->is_arraycopy_validated() ||
 957              ac->as_ArrayCopy()->is_copyof_validated() ||
 958              ac->as_ArrayCopy()->is_copyofrange_validated(), "only supported cases");
 959       assert(addp->in(AddPNode::Base) == addp->in(AddPNode::Address), "should be");
 960       addp->set_req(AddPNode::Base, src);
 961       addp->set_req(AddPNode::Address, src);
 962 
 963       const TypeAryPtr* ary_t = phase->type(in(MemNode::Address))->isa_aryptr();
 964       BasicType ary_elem  = ary_t->klass()->as_array_klass()->element_type()->basic_type();
 965       uint header = arrayOopDesc::base_offset_in_bytes(ary_elem);
 966       uint shift  = exact_log2(type2aelembytes(ary_elem));
 967 
 968       Node* diff = phase->transform(new SubINode(ac->in(ArrayCopyNode::SrcPos), ac->in(ArrayCopyNode::DestPos)));
 969 #ifdef _LP64
 970       diff = phase->transform(new ConvI2LNode(diff));
 971 #endif
 972       diff = phase->transform(new LShiftXNode(diff, phase->intcon(shift)));
 973 
 974       Node* offset = phase->transform(new AddXNode(addp->in(AddPNode::Offset), diff));
 975       addp->set_req(AddPNode::Offset, offset);
 976     }
 977     addp = phase->transform(addp);
 978 #ifdef ASSERT
 979     const TypePtr* adr_type = phase->type(addp)->is_ptr();
 980     ld->_adr_type = adr_type;
 981 #endif
 982     ld->set_req(MemNode::Address, addp);
 983     ld->set_req(0, ctl);
 984     ld->set_req(MemNode::Memory, mem);
 985     // load depends on the tests that validate the arraycopy
 986     ld->_control_dependency = UnknownControl;
 987     return ld;
 988   }
 989   return NULL;
 990 }

 993 //---------------------------can_see_stored_value------------------------------
 994 // This routine exists to make sure this set of tests is done the same
 995 // everywhere.  We need to make a coordinated change: first LoadNode::Ideal
 996 // will change the graph shape in a way which makes memory alive twice at the
 997 // same time (uses the Oracle model of aliasing), then some
 998 // LoadXNode::Identity will fold things back to the equivalence-class model
 999 // of aliasing.
1000 Node* MemNode::can_see_stored_value(Node* st, PhaseTransform* phase) const {
1001   Node* ld_adr = in(MemNode::Address);
1002   intptr_t ld_off = 0;
1003   Node* ld_base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ld_off);
1004   Node* ld_alloc = AllocateNode::Ideal_allocation(ld_base, phase);
1005   const TypeInstPtr* tp = phase->type(ld_adr)->isa_instptr();
1006   Compile::AliasType* atp = (tp != NULL) ? phase->C->alias_type(tp) : NULL;
1007   // This is more general than load from boxing objects.
1008   if (skip_through_membars(atp, tp, phase->C->eliminate_boxing())) {
1009     uint alias_idx = atp->index();
1010     bool final = !atp->is_rewritable();
1011     Node* result = NULL;
1012     Node* current = st;
1013     // Skip through chains of MemBarNodes checking the MergeMems for
1014     // new states for the slice of this load.  Stop once any other
1015     // kind of node is encountered.  Loads from final memory can skip
1016     // through any kind of MemBar but normal loads shouldn't skip
1017     // through MemBarAcquire since the could allow them to move out of
1018     // a synchronized region.
1019     while (current->is_Proj()) {
1020       int opc = current->in(0)->Opcode();
1021       if ((final && (opc == Op_MemBarAcquire ||
1022                      opc == Op_MemBarAcquireLock ||
1023                      opc == Op_LoadFence)) ||
1024           opc == Op_MemBarRelease ||
1025           opc == Op_StoreFence ||
1026           opc == Op_MemBarReleaseLock ||
1027           opc == Op_MemBarStoreStore ||
1028           opc == Op_MemBarCPUOrder) {
1029         Node* mem = current->in(0)->in(TypeFunc::Memory);
1030         if (mem->is_MergeMem()) {
1031           MergeMemNode* merge = mem->as_MergeMem();
1032           Node* new_st = merge->memory_at(alias_idx);
1033           if (new_st == merge->base_memory()) {
1034             // Keep searching
1035             current = new_st;
1036             continue;
1037           }
1038           // Save the new memory state for the slice and fall through
1039           // to exit.
1040           result = new_st;
1041         }
1042       }
1043       break;
1044     }
1045     if (result != NULL) {
1046       st = result;
1047     }
1048   }
1049 
1050   // Loop around twice in the case Load -> Initialize -> Store.
1051   // (See PhaseIterGVN::add_users_to_worklist, which knows about this case.)
1052   for (int trip = 0; trip <= 1; trip++) {
1053 
1054     if (st->is_Store()) {
1055       Node* st_adr = st->in(MemNode::Address);
1056       if (!phase->eqv(st_adr, ld_adr)) {
1057         // Try harder before giving up. Unify base pointers with casts (e.g., raw/non-raw pointers).
1058         intptr_t st_off = 0;
1059         Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_off);
1060         if (ld_base == NULL)                                   return NULL;
1061         if (st_base == NULL)                                   return NULL;
1062         if (!ld_base->eqv_uncast(st_base, /*keep_deps=*/true)) return NULL;
1063         if (ld_off != st_off)                                  return NULL;
1064         if (ld_off == Type::OffsetBot)                         return NULL;
1065         // Same base, same offset.
1066         // Possible improvement for arrays: check index value instead of absolute offset.
1067 
1068         // At this point we have proven something like this setup:
1069         //   B = << base >>
1070         //   L =  LoadQ(AddP(Check/CastPP(B), #Off))
1071         //   S = StoreQ(AddP(             B , #Off), V)
1072         // (Actually, we haven't yet proven the Q's are the same.)
1073         // In other words, we are loading from a casted version of
1074         // the same pointer-and-offset that we stored to.
1075         // Casted version may carry a dependency and it is respected.
1076         // Thus, we are able to replace L by V.
1077       }
1078       // Now prove that we have a LoadQ matched to a StoreQ, for some Q.
1079       if (store_Opcode() != st->Opcode())
1080         return NULL;
1081       return st->in(MemNode::ValueIn);
1082     }
1083 
1084     // A load from a freshly-created object always returns zero.
1085     // (This can happen after LoadNode::Ideal resets the load's memory input
1086     // to find_captured_store, which returned InitializeNode::zero_memory.)
1087     if (st->is_Proj() && st->in(0)->is_Allocate() &&
1088         (st->in(0) == ld_alloc) &&
1089         (ld_off >= st->in(0)->as_Allocate()->minimum_header_size())) {
1090       // return a zero value for the load's basic type
1091       // (This is one of the few places where a generic PhaseTransform
1092       // can create new nodes.  Think of it as lazily manifesting
1093       // virtually pre-existing constants.)
1094       return phase->zerocon(memory_type());
1095     }
1096 
1097     // A load from an initialization barrier can match a captured store.
1098     if (st->is_Proj() && st->in(0)->is_Initialize()) {
1099       InitializeNode* init = st->in(0)->as_Initialize();
1100       AllocateNode* alloc = init->allocation();
1101       if ((alloc != NULL) && (alloc == ld_alloc)) {
1102         // examine a captured store value
1103         st = init->find_captured_store(ld_off, memory_size(), phase);
1104         if (st != NULL) {
1105           continue;             // take one more trip around
1106         }
1107       }
1108     }
1109 
1110     // Load boxed value from result of valueOf() call is input parameter.
1111     if (this->is_Load() && ld_adr->is_AddP() &&
1112         (tp != NULL) && tp->is_ptr_to_boxed_value()) {
1113       intptr_t ignore = 0;
1114       Node* base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ignore);
1115       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1116       base = bs->step_over_gc_barrier(base);
1117       if (base != NULL && base->is_Proj() &&
1118           base->as_Proj()->_con == TypeFunc::Parms &&
1119           base->in(0)->is_CallStaticJava() &&
1120           base->in(0)->as_CallStaticJava()->is_boxing_method()) {
1121         return base->in(0)->in(TypeFunc::Parms);
1122       }
1123     }
1124 
1125     break;
1126   }
1127 
1128   return NULL;
1129 }

1148 //------------------------------Identity---------------------------------------
1149 // Loads are identity if previous store is to same address
1150 Node* LoadNode::Identity(PhaseGVN* phase) {
1151   // If the previous store-maker is the right kind of Store, and the store is
1152   // to the same address, then we are equal to the value stored.
1153   Node* mem = in(Memory);
1154   Node* value = can_see_stored_value(mem, phase);
1155   if( value ) {
1156     // byte, short & char stores truncate naturally.
1157     // A load has to load the truncated value which requires
1158     // some sort of masking operation and that requires an
1159     // Ideal call instead of an Identity call.
1160     if (memory_size() < BytesPerInt) {
1161       // If the input to the store does not fit with the load's result type,
1162       // it must be truncated via an Ideal call.
1163       if (!phase->type(value)->higher_equal(phase->type(this)))
1164         return this;
1165     }
1166     // (This works even when value is a Con, but LoadNode::Value
1167     // usually runs first, producing the singleton type of the Con.)
1168     return value;
1169   }
1170 
1171   // Search for an existing data phi which was generated before for the same
1172   // instance's field to avoid infinite generation of phis in a loop.
1173   Node *region = mem->in(0);
1174   if (is_instance_field_load_with_local_phi(region)) {
1175     const TypeOopPtr *addr_t = in(Address)->bottom_type()->isa_oopptr();
1176     int this_index  = phase->C->get_alias_index(addr_t);
1177     int this_offset = addr_t->offset();
1178     int this_iid    = addr_t->instance_id();
1179     if (!addr_t->is_known_instance() &&
1180          addr_t->is_ptr_to_boxed_value()) {
1181       // Use _idx of address base (could be Phi node) for boxed values.
1182       intptr_t   ignore = 0;
1183       Node*      base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1184       if (base == NULL) {
1185         return this;
1186       }
1187       this_iid = base->_idx;
1188     }
1189     const Type* this_type = bottom_type();
1190     for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
1191       Node* phi = region->fast_out(i);
1192       if (phi->is_Phi() && phi != mem &&
1193           phi->as_Phi()->is_same_inst_field(this_type, (int)mem->_idx, this_iid, this_index, this_offset)) {
1194         return phi;
1195       }
1196     }
1197   }
1198 
1199   return this;
1200 }

1240 // We're loading from an object which has autobox behaviour.
1241 // If this object is result of a valueOf call we'll have a phi
1242 // merging a newly allocated object and a load from the cache.
1243 // We want to replace this load with the original incoming
1244 // argument to the valueOf call.
1245 Node* LoadNode::eliminate_autobox(PhaseGVN* phase) {
1246   assert(phase->C->eliminate_boxing(), "sanity");
1247   intptr_t ignore = 0;
1248   Node* base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1249   if ((base == NULL) || base->is_Phi()) {
1250     // Push the loads from the phi that comes from valueOf up
1251     // through it to allow elimination of the loads and the recovery
1252     // of the original value. It is done in split_through_phi().
1253     return NULL;
1254   } else if (base->is_Load() ||
1255              (base->is_DecodeN() && base->in(1)->is_Load())) {
1256     // Eliminate the load of boxed value for integer types from the cache
1257     // array by deriving the value from the index into the array.
1258     // Capture the offset of the load and then reverse the computation.
1259 
1260     // Get LoadN node which loads a boxing object from 'cache' array.
1261     if (base->is_DecodeN()) {
1262       base = base->in(1);
1263     }
1264     if (!base->in(Address)->is_AddP()) {
1265       return NULL; // Complex address
1266     }
1267     AddPNode* address = base->in(Address)->as_AddP();
1268     Node* cache_base = address->in(AddPNode::Base);
1269     if ((cache_base != NULL) && cache_base->is_DecodeN()) {
1270       // Get ConP node which is static 'cache' field.
1271       cache_base = cache_base->in(1);
1272     }
1273     if ((cache_base != NULL) && cache_base->is_Con()) {
1274       const TypeAryPtr* base_type = cache_base->bottom_type()->isa_aryptr();
1275       if ((base_type != NULL) && base_type->is_autobox_cache()) {
1276         Node* elements[4];
1277         int shift = exact_log2(type2aelembytes(T_OBJECT));
1278         int count = address->unpack_offsets(elements, ARRAY_SIZE(elements));
1279         if (count > 0 && elements[0]->is_Con() &&
1280             (count == 1 ||
1281              (count == 2 && elements[1]->Opcode() == Op_LShiftX &&
1282                             elements[1]->in(2) == phase->intcon(shift)))) {
1283           ciObjArray* array = base_type->const_oop()->as_obj_array();
1284           // Fetch the box object cache[0] at the base of the array and get its value
1285           ciInstance* box = array->obj_at(0)->as_instance();
1286           ciInstanceKlass* ik = box->klass()->as_instance_klass();
1287           assert(ik->is_box_klass(), "sanity");
1288           assert(ik->nof_nonstatic_fields() == 1, "change following code");
1289           if (ik->nof_nonstatic_fields() == 1) {
1290             // This should be true nonstatic_field_at requires calling
1291             // nof_nonstatic_fields so check it anyway
1292             ciConstant c = box->field_value(ik->nonstatic_field_at(0));
1293             BasicType bt = c.basic_type();
1294             // Only integer types have boxing cache.
1295             assert(bt == T_BOOLEAN || bt == T_CHAR  ||
1296                    bt == T_BYTE    || bt == T_SHORT ||
1297                    bt == T_INT     || bt == T_LONG, "wrong type = %s", type2name(bt));
1298             jlong cache_low = (bt == T_LONG) ? c.as_long() : c.as_int();
1299             if (cache_low != (int)cache_low) {
1300               return NULL; // should not happen since cache is array indexed by value
1301             }
1302             jlong offset = arrayOopDesc::base_offset_in_bytes(T_OBJECT) - (cache_low << shift);
1303             if (offset != (int)offset) {
1304               return NULL; // should not happen since cache is array indexed by value
1305             }
1306            // Add up all the offsets making of the address of the load
1307             Node* result = elements[0];
1308             for (int i = 1; i < count; i++) {
1309               result = phase->transform(new AddXNode(result, elements[i]));
1310             }
1311             // Remove the constant offset from the address and then
1312             result = phase->transform(new AddXNode(result, phase->MakeConX(-(int)offset)));
1313             // remove the scaling of the offset to recover the original index.
1314             if (result->Opcode() == Op_LShiftX && result->in(2) == phase->intcon(shift)) {
1315               // Peel the shift off directly but wrap it in a dummy node
1316               // since Ideal can't return existing nodes
1317               result = new RShiftXNode(result->in(1), phase->intcon(0));
1318             } else if (result->is_Add() && result->in(2)->is_Con() &&
1319                        result->in(1)->Opcode() == Op_LShiftX &&
1320                        result->in(1)->in(2) == phase->intcon(shift)) {
1321               // We can't do general optimization: ((X<<Z) + Y) >> Z ==> X + (Y>>Z)
1322               // but for boxing cache access we know that X<<Z will not overflow
1323               // (there is range check) so we do this optimizatrion by hand here.
1324               Node* add_con = new RShiftXNode(result->in(2), phase->intcon(shift));
1325               result = new AddXNode(result->in(1)->in(1), phase->transform(add_con));
1326             } else {
1327               result = new RShiftXNode(result, phase->intcon(shift));
1328             }
1329 #ifdef _LP64
1330             if (bt != T_LONG) {
1331               result = new ConvL2INode(phase->transform(result));
1332             }
1333 #else
1334             if (bt == T_LONG) {
1335               result = new ConvI2LNode(phase->transform(result));
1336             }
1337 #endif
1338             // Boxing/unboxing can be done from signed & unsigned loads (e.g. LoadUB -> ... -> LoadB pair).
1339             // Need to preserve unboxing load type if it is unsigned.
1340             switch(this->Opcode()) {
1341               case Op_LoadUB:
1342                 result = new AndINode(phase->transform(result), phase->intcon(0xFF));
1343                 break;
1344               case Op_LoadUS:
1345                 result = new AndINode(phase->transform(result), phase->intcon(0xFFFF));
1346                 break;
1347             }
1348             return result;
1349           }
1350         }
1351       }
1352     }
1353   }
1354   return NULL;
1355 }

1357 static bool stable_phi(PhiNode* phi, PhaseGVN *phase) {
1358   Node* region = phi->in(0);
1359   if (region == NULL) {
1360     return false; // Wait stable graph
1361   }
1362   uint cnt = phi->req();
1363   for (uint i = 1; i < cnt; i++) {
1364     Node* rc = region->in(i);
1365     if (rc == NULL || phase->type(rc) == Type::TOP)
1366       return false; // Wait stable graph
1367     Node* in = phi->in(i);
1368     if (in == NULL || phase->type(in) == Type::TOP)
1369       return false; // Wait stable graph
1370   }
1371   return true;

1373 //------------------------------split_through_phi------------------------------
1374 // Split instance or boxed field load through Phi.
1375 Node *LoadNode::split_through_phi(PhaseGVN *phase) {
1376   Node* mem     = in(Memory);
1377   Node* address = in(Address);
1378   const TypeOopPtr *t_oop = phase->type(address)->isa_oopptr();
1379 
1380   assert((t_oop != NULL) &&
1381          (t_oop->is_known_instance_field() ||
1382           t_oop->is_ptr_to_boxed_value()), "invalide conditions");
1383 
1384   Compile* C = phase->C;
1385   intptr_t ignore = 0;
1386   Node*    base = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1387   bool base_is_phi = (base != NULL) && base->is_Phi();
1388   bool load_boxed_values = t_oop->is_ptr_to_boxed_value() && C->aggressive_unboxing() &&
1389                            (base != NULL) && (base == address->in(AddPNode::Base)) &&
1390                            phase->type(base)->higher_equal(TypePtr::NOTNULL);
1391 
1392   if (!((mem->is_Phi() || base_is_phi) &&
1393         (load_boxed_values || t_oop->is_known_instance_field()))) {
1394     return NULL; // memory is not Phi
1395   }
1396 
1397   if (mem->is_Phi()) {
1398     if (!stable_phi(mem->as_Phi(), phase)) {
1399       return NULL; // Wait stable graph
1400     }
1401     uint cnt = mem->req();
1402     // Check for loop invariant memory.
1403     if (cnt == 3) {
1404       for (uint i = 1; i < cnt; i++) {
1405         Node* in = mem->in(i);
1406         Node*  m = optimize_memory_chain(in, t_oop, this, phase);
1407         if (m == mem) {
1408           if (i == 1) {
1409             // if the first edge was a loop, check second edge too.
1410             // If both are replaceable - we are in an infinite loop
1411             Node *n = optimize_memory_chain(mem->in(2), t_oop, this, phase);
1412             if (n == mem) {
1413               break;
1414             }
1415           }
1416           set_req(Memory, mem->in(cnt - i));
1417           return this; // made change
1418         }
1419       }
1420     }
1421   }
1422   if (base_is_phi) {
1423     if (!stable_phi(base->as_Phi(), phase)) {
1424       return NULL; // Wait stable graph
1425     }
1426     uint cnt = base->req();
1427     // Check for loop invariant memory.
1428     if (cnt == 3) {
1429       for (uint i = 1; i < cnt; i++) {
1430         if (base->in(i) == base) {
1431           return NULL; // Wait stable graph
1432         }
1433       }
1434     }
1435   }
1436 
1437   // Split through Phi (see original code in loopopts.cpp).
1438   assert(C->have_alias_type(t_oop), "instance should have alias type");
1439 
1440   // Do nothing here if Identity will find a value
1441   // (to avoid infinite chain of value phis generation).
1442   if (!phase->eqv(this, this->Identity(phase))) {
1443     return NULL;
1444   }
1445 
1446   // Select Region to split through.
1447   Node* region;
1448   if (!base_is_phi) {
1449     assert(mem->is_Phi(), "sanity");
1450     region = mem->in(0);
1451     // Skip if the region dominates some control edge of the address.
1452     if (!MemNode::all_controls_dominate(address, region))
1453       return NULL;
1454   } else if (!mem->is_Phi()) {
1455     assert(base_is_phi, "sanity");
1456     region = base->in(0);
1457     // Skip if the region dominates some control edge of the memory.
1458     if (!MemNode::all_controls_dominate(mem, region))
1459       return NULL;
1460   } else if (base->in(0) != mem->in(0)) {
1461     assert(base_is_phi && mem->is_Phi(), "sanity");
1462     if (MemNode::all_controls_dominate(mem, base->in(0))) {
1463       region = base->in(0);
1464     } else if (MemNode::all_controls_dominate(address, mem->in(0))) {
1465       region = mem->in(0);
1466     } else {
1467       return NULL; // complex graph
1468     }
1469   } else {
1470     assert(base->in(0) == mem->in(0), "sanity");
1471     region = mem->in(0);
1472   }
1473 
1474   const Type* this_type = this->bottom_type();
1475   int this_index  = C->get_alias_index(t_oop);
1476   int this_offset = t_oop->offset();
1477   int this_iid    = t_oop->instance_id();
1478   if (!t_oop->is_known_instance() && load_boxed_values) {
1479     // Use _idx of address base for boxed values.
1480     this_iid = base->_idx;
1481   }
1482   PhaseIterGVN* igvn = phase->is_IterGVN();
1483   Node* phi = new PhiNode(region, this_type, NULL, mem->_idx, this_iid, this_index, this_offset);
1484   for (uint i = 1; i < region->req(); i++) {
1485     Node* x;
1486     Node* the_clone = NULL;
1487     Node* in = region->in(i);
1488     if (region->is_CountedLoop() && region->as_Loop()->is_strip_mined() && i == LoopNode::EntryControl &&
1489         in != NULL && in->is_OuterStripMinedLoop()) {
1490       // No node should go in the outer strip mined loop
1491       in = in->in(LoopNode::EntryControl);
1492     }
1493     if (in == NULL || in == C->top()) {
1494       x = C->top();      // Dead path?  Use a dead data op
1495     } else {
1496       x = this->clone();        // Else clone up the data op
1497       the_clone = x;            // Remember for possible deletion.
1498       // Alter data node to use pre-phi inputs
1499       if (this->in(0) == region) {
1500         x->set_req(0, in);
1501       } else {
1502         x->set_req(0, NULL);
1503       }
1504       if (mem->is_Phi() && (mem->in(0) == region)) {
1505         x->set_req(Memory, mem->in(i)); // Use pre-Phi input for the clone.
1506       }
1507       if (address->is_Phi() && address->in(0) == region) {
1508         x->set_req(Address, address->in(i)); // Use pre-Phi input for the clone
1509       }
1510       if (base_is_phi && (base->in(0) == region)) {
1511         Node* base_x = base->in(i); // Clone address for loads from boxed objects.
1512         Node* adr_x = phase->transform(new AddPNode(base_x,base_x,address->in(AddPNode::Offset)));
1513         x->set_req(Address, adr_x);
1514       }
1515     }
1516     // Check for a 'win' on some paths
1517     const Type *t = x->Value(igvn);
1518 
1519     bool singleton = t->singleton();
1520 
1521     // See comments in PhaseIdealLoop::split_thru_phi().
1522     if (singleton && t == Type::TOP) {
1523       singleton &= region->is_Loop() && (i != LoopNode::EntryControl);
1524     }
1525 
1526     if (singleton) {
1527       x = igvn->makecon(t);
1528     } else {
1529       // We now call Identity to try to simplify the cloned node.
1530       // Note that some Identity methods call phase->type(this).
1531       // Make sure that the type array is big enough for
1532       // our new node, even though we may throw the node away.
1533       // (This tweaking with igvn only works because x is a new node.)
1534       igvn->set_type(x, t);
1535       // If x is a TypeNode, capture any more-precise type permanently into Node
1536       // otherwise it will be not updated during igvn->transform since
1537       // igvn->type(x) is set to x->Value() already.
1538       x->raise_bottom_type(t);
1539       Node* y = x->Identity(igvn);
1540       if (y != x) {
1541         x = y;
1542       } else {
1543         y = igvn->hash_find_insert(x);
1544         if (y) {
1545           x = y;
1546         } else {
1547           // Else x is a new node we are keeping
1548           // We do not need register_new_node_with_optimizer
1549           // because set_type has already been called.
1550           igvn->_worklist.push(x);
1551         }
1552       }
1553     }
1554     if (x != the_clone && the_clone != NULL) {
1555       igvn->remove_dead_node(the_clone);
1556     }
1557     phi->set_req(i, x);
1558   }
1559   // Record Phi
1560   igvn->register_new_node_with_optimizer(phi);
1561   return phi;
1562 }

1580 //------------------------------Ideal------------------------------------------
1581 // If the load is from Field memory and the pointer is non-null, it might be possible to
1582 // zero out the control input.
1583 // If the offset is constant and the base is an object allocation,
1584 // try to hook me up to the exact initializing store.
1585 Node *LoadNode::Ideal(PhaseGVN *phase, bool can_reshape) {
1586   Node* p = MemNode::Ideal_common(phase, can_reshape);
1587   if (p)  return (p == NodeSentinel) ? NULL : p;
1588 
1589   Node* ctrl    = in(MemNode::Control);
1590   Node* address = in(MemNode::Address);
1591   bool progress = false;
1592 
1593   bool addr_mark = ((phase->type(address)->isa_oopptr() || phase->type(address)->isa_narrowoop()) &&
1594          phase->type(address)->is_ptr()->offset() == oopDesc::mark_offset_in_bytes());
1595 
1596   // Skip up past a SafePoint control.  Cannot do this for Stores because
1597   // pointer stores & cardmarks must stay on the same side of a SafePoint.
1598   if( ctrl != NULL && ctrl->Opcode() == Op_SafePoint &&
1599       phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw  &&
1600       !addr_mark &&
1601       (depends_only_on_test() || has_unknown_control_dependency())) {
1602     ctrl = ctrl->in(0);
1603     set_req(MemNode::Control,ctrl);
1604     progress = true;
1605   }
1606 
1607   intptr_t ignore = 0;
1608   Node*    base   = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1609   if (base != NULL
1610       && phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw) {
1611     // Check for useless control edge in some common special cases
1612     if (in(MemNode::Control) != NULL
1613         && can_remove_control()
1614         && phase->type(base)->higher_equal(TypePtr::NOTNULL)
1615         && all_controls_dominate(base, phase->C->start())) {
1616       // A method-invariant, non-null address (constant or 'this' argument).
1617       set_req(MemNode::Control, NULL);
1618       progress = true;
1619     }
1620   }
1621 
1622   Node* mem = in(MemNode::Memory);
1623   const TypePtr *addr_t = phase->type(address)->isa_ptr();
1624 
1625   if (can_reshape && (addr_t != NULL)) {
1626     // try to optimize our memory input
1627     Node* opt_mem = MemNode::optimize_memory_chain(mem, addr_t, this, phase);
1628     if (opt_mem != mem) {
1629       set_req(MemNode::Memory, opt_mem);
1630       if (phase->type( opt_mem ) == Type::TOP) return NULL;
1631       return this;
1632     }
1633     const TypeOopPtr *t_oop = addr_t->isa_oopptr();
1634     if ((t_oop != NULL) &&
1635         (t_oop->is_known_instance_field() ||
1636          t_oop->is_ptr_to_boxed_value())) {
1637       PhaseIterGVN *igvn = phase->is_IterGVN();
1638       if (igvn != NULL && igvn->_worklist.member(opt_mem)) {
1639         // Delay this transformation until memory Phi is processed.
1640         phase->is_IterGVN()->_worklist.push(this);
1641         return NULL;
1642       }
1643       // Split instance field load through Phi.
1644       Node* result = split_through_phi(phase);
1645       if (result != NULL) return result;
1646 
1647       if (t_oop->is_ptr_to_boxed_value()) {
1648         Node* result = eliminate_autobox(phase);
1649         if (result != NULL) return result;
1650       }
1651     }
1652   }
1653 
1654   // Is there a dominating load that loads the same value?  Leave
1655   // anything that is not a load of a field/array element (like
1656   // barriers etc.) alone
1657   if (in(0) != NULL && !adr_type()->isa_rawptr() && can_reshape) {
1658     for (DUIterator_Fast imax, i = mem->fast_outs(imax); i < imax; i++) {
1659       Node *use = mem->fast_out(i);
1660       if (use != this &&
1661           use->Opcode() == Opcode() &&
1662           use->in(0) != NULL &&
1663           use->in(0) != in(0) &&
1664           use->in(Address) == in(Address)) {
1665         Node* ctl = in(0);
1666         for (int i = 0; i < 10 && ctl != NULL; i++) {
1667           ctl = IfNode::up_one_dom(ctl);
1668           if (ctl == use->in(0)) {
1669             set_req(0, use->in(0));
1670             return this;
1671           }
1672         }
1673       }
1674     }
1675   }
1676 
1677   // Check for prior store with a different base or offset; make Load
1678   // independent.  Skip through any number of them.  Bail out if the stores
1679   // are in an endless dead cycle and report no progress.  This is a key
1680   // transform for Reflection.  However, if after skipping through the Stores
1681   // we can't then fold up against a prior store do NOT do the transform as
1682   // this amounts to using the 'Oracle' model of aliasing.  It leaves the same
1683   // array memory alive twice: once for the hoisted Load and again after the
1684   // bypassed Store.  This situation only works if EVERYBODY who does
1685   // anti-dependence work knows how to bypass.  I.e. we need all
1686   // anti-dependence checks to ask the same Oracle.  Right now, that Oracle is
1687   // the alias index stuff.  So instead, peek through Stores and IFF we can
1688   // fold up, do so.
1689   Node* prev_mem = find_previous_store(phase);
1690   if (prev_mem != NULL) {
1691     Node* value = can_see_arraycopy_value(prev_mem, phase);
1692     if (value != NULL) {
1693       return value;
1694     }
1695   }
1696   // Steps (a), (b):  Walk past independent stores to find an exact match.
1697   if (prev_mem != NULL && prev_mem != in(MemNode::Memory)) {
1698     // (c) See if we can fold up on the spot, but don't fold up here.
1699     // Fold-up might require truncation (for LoadB/LoadS/LoadUS) or
1700     // just return a prior value, which is done by Identity calls.
1701     if (can_see_stored_value(prev_mem, phase)) {
1702       // Make ready for step (d):
1703       set_req(MemNode::Memory, prev_mem);
1704       return this;
1705     }
1706   }
1707 
1708   AllocateNode* alloc = is_new_object_mark_load(phase);
1709   if (alloc != NULL && alloc->Opcode() == Op_Allocate && UseBiasedLocking) {
1710     InitializeNode* init = alloc->initialization();
1711     Node* control = init->proj_out(0);
1712     return alloc->make_ideal_mark(phase, address, control, mem);
1713   }
1714 
1715   return progress ? this : NULL;
1716 }

1718 // Helper to recognize certain Klass fields which are invariant across
1719 // some group of array types (e.g., int[] or all T[] where T < Object).
1720 const Type*
1721 LoadNode::load_array_final_field(const TypeKlassPtr *tkls,
1722                                  ciKlass* klass) const {
1723   if (tkls->offset() == in_bytes(Klass::modifier_flags_offset())) {
1724     // The field is Klass::_modifier_flags.  Return its (constant) value.
1725     // (Folds up the 2nd indirection in aClassConstant.getModifiers().)
1726     assert(this->Opcode() == Op_LoadI, "must load an int from _modifier_flags");
1727     return TypeInt::make(klass->modifier_flags());
1728   }
1729   if (tkls->offset() == in_bytes(Klass::access_flags_offset())) {
1730     // The field is Klass::_access_flags.  Return its (constant) value.
1731     // (Folds up the 2nd indirection in Reflection.getClassAccessFlags(aClassConstant).)
1732     assert(this->Opcode() == Op_LoadI, "must load an int from _access_flags");
1733     return TypeInt::make(klass->access_flags());
1734   }
1735   if (tkls->offset() == in_bytes(Klass::layout_helper_offset())) {
1736     // The field is Klass::_layout_helper.  Return its constant value if known.
1737     assert(this->Opcode() == Op_LoadI, "must load an int from _layout_helper");
1738     return TypeInt::make(klass->layout_helper());
1739   }
1740 
1741   // No match.
1742   return NULL;
1743 }

1745 //------------------------------Value-----------------------------------------
1746 const Type* LoadNode::Value(PhaseGVN* phase) const {
1747   // Either input is TOP ==> the result is TOP
1748   Node* mem = in(MemNode::Memory);
1749   const Type *t1 = phase->type(mem);
1750   if (t1 == Type::TOP)  return Type::TOP;
1751   Node* adr = in(MemNode::Address);
1752   const TypePtr* tp = phase->type(adr)->isa_ptr();
1753   if (tp == NULL || tp->empty())  return Type::TOP;
1754   int off = tp->offset();
1755   assert(off != Type::OffsetTop, "case covered by TypePtr::empty");
1756   Compile* C = phase->C;
1757 
1758   // Try to guess loaded type from pointer type
1759   if (tp->isa_aryptr()) {
1760     const TypeAryPtr* ary = tp->is_aryptr();
1761     const Type* t = ary->elem();
1762 
1763     // Determine whether the reference is beyond the header or not, by comparing
1764     // the offset against the offset of the start of the array's data.
1765     // Different array types begin at slightly different offsets (12 vs. 16).
1766     // We choose T_BYTE as an example base type that is least restrictive
1767     // as to alignment, which will therefore produce the smallest
1768     // possible base offset.
1769     const int min_base_off = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1770     const bool off_beyond_header = (off >= min_base_off);
1771 
1772     // Try to constant-fold a stable array element.
1773     if (FoldStableValues && !is_mismatched_access() && ary->is_stable()) {
1774       // Make sure the reference is not into the header and the offset is constant
1775       ciObject* aobj = ary->const_oop();
1776       if (aobj != NULL && off_beyond_header && adr->is_AddP() && off != Type::OffsetBot) {
1777         int stable_dimension = (ary->stable_dimension() > 0 ? ary->stable_dimension() - 1 : 0);
1778         const Type* con_type = Type::make_constant_from_array_element(aobj->as_array(), off,
1779                                                                       stable_dimension,
1780                                                                       memory_type(), is_unsigned());
1781         if (con_type != NULL) {
1782           return con_type;
1783         }
1784       }
1785     }
1786 
1787     // Don't do this for integer types. There is only potential profit if
1788     // the element type t is lower than _type; that is, for int types, if _type is
1789     // more restrictive than t.  This only happens here if one is short and the other
1790     // char (both 16 bits), and in those cases we've made an intentional decision
1791     // to use one kind of load over the other. See AndINode::Ideal and 4965907.
1792     // Also, do not try to narrow the type for a LoadKlass, regardless of offset.
1793     //
1794     // Yes, it is possible to encounter an expression like (LoadKlass p1:(AddP x x 8))
1795     // where the _gvn.type of the AddP is wider than 8.  This occurs when an earlier
1796     // copy p0 of (AddP x x 8) has been proven equal to p1, and the p0 has been
1797     // subsumed by p1.  If p1 is on the worklist but has not yet been re-transformed,
1798     // it is possible that p1 will have a type like Foo*[int+]:NotNull*+any.
1799     // In fact, that could have been the original type of p1, and p1 could have
1800     // had an original form like p1:(AddP x x (LShiftL quux 3)), where the
1801     // expression (LShiftL quux 3) independently optimized to the constant 8.
1802     if ((t->isa_int() == NULL) && (t->isa_long() == NULL)
1803         && (_type->isa_vect() == NULL)
1804         && Opcode() != Op_LoadKlass && Opcode() != Op_LoadNKlass) {
1805       // t might actually be lower than _type, if _type is a unique
1806       // concrete subclass of abstract class t.
1807       if (off_beyond_header || off == Type::OffsetBot) {  // is the offset beyond the header?
1808         const Type* jt = t->join_speculative(_type);
1809         // In any case, do not allow the join, per se, to empty out the type.
1810         if (jt->empty() && !t->empty()) {
1811           // This can happen if a interface-typed array narrows to a class type.
1812           jt = _type;
1813         }
1814 #ifdef ASSERT
1815         if (phase->C->eliminate_boxing() && adr->is_AddP()) {
1816           // The pointers in the autobox arrays are always non-null
1817           Node* base = adr->in(AddPNode::Base);
1818           if ((base != NULL) && base->is_DecodeN()) {
1819             // Get LoadN node which loads IntegerCache.cache field
1820             base = base->in(1);
1821           }
1822           if ((base != NULL) && base->is_Con()) {
1823             const TypeAryPtr* base_type = base->bottom_type()->isa_aryptr();
1824             if ((base_type != NULL) && base_type->is_autobox_cache()) {
1825               // It could be narrow oop
1826               assert(jt->make_ptr()->ptr() == TypePtr::NotNull,"sanity");
1827             }
1828           }
1829         }
1830 #endif
1831         return jt;
1832       }
1833     }
1834   } else if (tp->base() == Type::InstPtr) {
1835     assert( off != Type::OffsetBot ||
1836             // arrays can be cast to Objects
1837             tp->is_oopptr()->klass()->is_java_lang_Object() ||
1838             // unsafe field access may not have a constant offset
1839             C->has_unsafe_access(),
1840             "Field accesses must be precise" );
1841     // For oop loads, we expect the _type to be precise.
1842 
1843     // Optimize loads from constant fields.
1844     const TypeInstPtr* tinst = tp->is_instptr();
1845     ciObject* const_oop = tinst->const_oop();
1846     if (!is_mismatched_access() && off != Type::OffsetBot && const_oop != NULL && const_oop->is_instance()) {
1847       const Type* con_type = Type::make_constant_from_field(const_oop->as_instance(), off, is_unsigned(), memory_type());
1848       if (con_type != NULL) {
1849         return con_type;
1850       }
1851     }
1852   } else if (tp->base() == Type::KlassPtr) {
1853     assert( off != Type::OffsetBot ||
1854             // arrays can be cast to Objects
1855             tp->is_klassptr()->klass()->is_java_lang_Object() ||
1856             // also allow array-loading from the primary supertype
1857             // array during subtype checks
1858             Opcode() == Op_LoadKlass,
1859             "Field accesses must be precise" );
1860     // For klass/static loads, we expect the _type to be precise
1861   } else if (tp->base() == Type::RawPtr && adr->is_Load() && off == 0) {
1862     /* With mirrors being an indirect in the Klass*
1863      * the VM is now using two loads. LoadKlass(LoadP(LoadP(Klass, mirror_offset), zero_offset))
1864      * The LoadP from the Klass has a RawPtr type (see LibraryCallKit::load_mirror_from_klass).
1865      *
1866      * So check the type and klass of the node before the LoadP.
1867      */
1868     Node* adr2 = adr->in(MemNode::Address);
1869     const TypeKlassPtr* tkls = phase->type(adr2)->isa_klassptr();
1870     if (tkls != NULL && !StressReflectiveCode) {
1871       ciKlass* klass = tkls->klass();
1872       if (klass->is_loaded() && tkls->klass_is_exact() && tkls->offset() == in_bytes(Klass::java_mirror_offset())) {
1873         assert(adr->Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1874         assert(Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1875         return TypeInstPtr::make(klass->java_mirror());
1876       }
1877     }
1878   }
1879 
1880   const TypeKlassPtr *tkls = tp->isa_klassptr();
1881   if (tkls != NULL && !StressReflectiveCode) {
1882     ciKlass* klass = tkls->klass();
1883     if (klass->is_loaded() && tkls->klass_is_exact()) {
1884       // We are loading a field from a Klass metaobject whose identity
1885       // is known at compile time (the type is "exact" or "precise").
1886       // Check for fields we know are maintained as constants by the VM.
1887       if (tkls->offset() == in_bytes(Klass::super_check_offset_offset())) {
1888         // The field is Klass::_super_check_offset.  Return its (constant) value.
1889         // (Folds up type checking code.)
1890         assert(Opcode() == Op_LoadI, "must load an int from _super_check_offset");
1891         return TypeInt::make(klass->super_check_offset());
1892       }
1893       // Compute index into primary_supers array
1894       juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1895       // Check for overflowing; use unsigned compare to handle the negative case.
1896       if( depth < ciKlass::primary_super_limit() ) {
1897         // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1898         // (Folds up type checking code.)
1899         assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
1900         ciKlass *ss = klass->super_of_depth(depth);
1901         return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
1902       }
1903       const Type* aift = load_array_final_field(tkls, klass);
1904       if (aift != NULL)  return aift;
1905     }
1906 
1907     // We can still check if we are loading from the primary_supers array at a
1908     // shallow enough depth.  Even though the klass is not exact, entries less
1909     // than or equal to its super depth are correct.
1910     if (klass->is_loaded() ) {
1911       ciType *inner = klass;
1912       while( inner->is_obj_array_klass() )
1913         inner = inner->as_obj_array_klass()->base_element_type();
1914       if( inner->is_instance_klass() &&
1915           !inner->as_instance_klass()->flags().is_interface() ) {
1916         // Compute index into primary_supers array
1917         juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1918         // Check for overflowing; use unsigned compare to handle the negative case.
1919         if( depth < ciKlass::primary_super_limit() &&
1920             depth <= klass->super_depth() ) { // allow self-depth checks to handle self-check case
1921           // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1922           // (Folds up type checking code.)
1923           assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
1924           ciKlass *ss = klass->super_of_depth(depth);
1925           return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
1926         }
1927       }
1928     }
1929 
1930     // If the type is enough to determine that the thing is not an array,
1931     // we can give the layout_helper a positive interval type.
1932     // This will help short-circuit some reflective code.
1933     if (tkls->offset() == in_bytes(Klass::layout_helper_offset())
1934         && !klass->is_array_klass() // not directly typed as an array
1935         && !klass->is_interface()  // specifically not Serializable & Cloneable
1936         && !klass->is_java_lang_Object()   // not the supertype of all T[]
1937         ) {
1938       // Note:  When interfaces are reliable, we can narrow the interface
1939       // test to (klass != Serializable && klass != Cloneable).
1940       assert(Opcode() == Op_LoadI, "must load an int from _layout_helper");
1941       jint min_size = Klass::instance_layout_helper(oopDesc::header_size(), false);
1942       // The key property of this type is that it folds up tests
1943       // for array-ness, since it proves that the layout_helper is positive.
1944       // Thus, a generic value like the basic object layout helper works fine.
1945       return TypeInt::make(min_size, max_jint, Type::WidenMin);
1946     }
1947   }
1948 
1949   // If we are loading from a freshly-allocated object, produce a zero,
1950   // if the load is provably beyond the header of the object.
1951   // (Also allow a variable load from a fresh array to produce zero.)
1952   const TypeOopPtr *tinst = tp->isa_oopptr();
1953   bool is_instance = (tinst != NULL) && tinst->is_known_instance_field();
1954   bool is_boxed_value = (tinst != NULL) && tinst->is_ptr_to_boxed_value();
1955   if (ReduceFieldZeroing || is_instance || is_boxed_value) {
1956     Node* value = can_see_stored_value(mem,phase);
1957     if (value != NULL && value->is_Con()) {
1958       assert(value->bottom_type()->higher_equal(_type),"sanity");
1959       return value->bottom_type();
1960     }
1961   }
1962 
1963   if (is_instance) {
1964     // If we have an instance type and our memory input is the
1965     // programs's initial memory state, there is no matching store,
1966     // so just return a zero of the appropriate type
1967     Node *mem = in(MemNode::Memory);
1968     if (mem->is_Parm() && mem->in(0)->is_Start()) {
1969       assert(mem->as_Parm()->_con == TypeFunc::Memory, "must be memory Parm");
1970       return Type::get_zero_type(_type->basic_type());
1971     }
1972   }
1973 
1974   Node* alloc = is_new_object_mark_load(phase);
1975   if (alloc != NULL && !(alloc->Opcode() == Op_Allocate && UseBiasedLocking)) {
1976     return TypeX::make(markWord::prototype().value());
1977   }
1978 
1979   return _type;
1980 }

2513 //------------------------------Ideal------------------------------------------
2514 // Change back-to-back Store(, p, x) -> Store(m, p, y) to Store(m, p, x).
2515 // When a store immediately follows a relevant allocation/initialization,
2516 // try to capture it into the initialization, or hoist it above.
2517 Node *StoreNode::Ideal(PhaseGVN *phase, bool can_reshape) {
2518   Node* p = MemNode::Ideal_common(phase, can_reshape);
2519   if (p)  return (p == NodeSentinel) ? NULL : p;
2520 
2521   Node* mem     = in(MemNode::Memory);
2522   Node* address = in(MemNode::Address);
2523   // Back-to-back stores to same address?  Fold em up.  Generally
2524   // unsafe if I have intervening uses...  Also disallowed for StoreCM
2525   // since they must follow each StoreP operation.  Redundant StoreCMs
2526   // are eliminated just before matching in final_graph_reshape.
2527   {
2528     Node* st = mem;
2529     // If Store 'st' has more than one use, we cannot fold 'st' away.
2530     // For example, 'st' might be the final state at a conditional
2531     // return.  Or, 'st' might be used by some node which is live at
2532     // the same time 'st' is live, which might be unschedulable.  So,
2533     // require exactly ONE user until such time as we clone 'mem' for
2534     // each of 'mem's uses (thus making the exactly-1-user-rule hold
2535     // true).
2536     while (st->is_Store() && st->outcnt() == 1 && st->Opcode() != Op_StoreCM) {
2537       // Looking at a dead closed cycle of memory?
2538       assert(st != st->in(MemNode::Memory), "dead loop in StoreNode::Ideal");
2539       assert(Opcode() == st->Opcode() ||
2540              st->Opcode() == Op_StoreVector ||
2541              Opcode() == Op_StoreVector ||
2542              phase->C->get_alias_index(adr_type()) == Compile::AliasIdxRaw ||
2543              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreI) || // expanded ClearArrayNode
2544              (Opcode() == Op_StoreI && st->Opcode() == Op_StoreL) || // initialization by arraycopy
2545              (is_mismatched_access() || st->as_Store()->is_mismatched_access()),
2546              "no mismatched stores, except on raw memory: %s %s", NodeClassNames[Opcode()], NodeClassNames[st->Opcode()]);
2547 
2548       if (st->in(MemNode::Address)->eqv_uncast(address) &&
2549           st->as_Store()->memory_size() <= this->memory_size()) {
2550         Node* use = st->raw_out(0);
2551         phase->igvn_rehash_node_delayed(use);
2552         if (can_reshape) {
2553           use->set_req_X(MemNode::Memory, st->in(MemNode::Memory), phase->is_IterGVN());
2554         } else {
2555           // It's OK to do this in the parser, since DU info is always accurate,
2556           // and the parser always refers to nodes via SafePointNode maps.
2557           use->set_req(MemNode::Memory, st->in(MemNode::Memory));
2558         }
2559         return this;
2560       }
2561       st = st->in(MemNode::Memory);
2562     }
2563   }
2564 
2565 
2566   // Capture an unaliased, unconditional, simple store into an initializer.
2567   // Or, if it is independent of the allocation, hoist it above the allocation.
2568   if (ReduceFieldZeroing && /*can_reshape &&*/
2569       mem->is_Proj() && mem->in(0)->is_Initialize()) {
2570     InitializeNode* init = mem->in(0)->as_Initialize();

2572     if (offset > 0) {
2573       Node* moved = init->capture_store(this, offset, phase, can_reshape);
2574       // If the InitializeNode captured me, it made a raw copy of me,
2575       // and I need to disappear.
2576       if (moved != NULL) {
2577         // %%% hack to ensure that Ideal returns a new node:
2578         mem = MergeMemNode::make(mem);
2579         return mem;             // fold me away
2580       }
2581     }
2582   }
2583 
2584   return NULL;                  // No further progress
2585 }

2599 //------------------------------Identity---------------------------------------
2600 // Remove redundant stores:
2601 //   Store(m, p, Load(m, p)) changes to m.
2602 //   Store(, p, x) -> Store(m, p, x) changes to Store(m, p, x).
2603 Node* StoreNode::Identity(PhaseGVN* phase) {
2604   Node* mem = in(MemNode::Memory);
2605   Node* adr = in(MemNode::Address);
2606   Node* val = in(MemNode::ValueIn);
2607 
2608   Node* result = this;
2609 
2610   // Load then Store?  Then the Store is useless
2611   if (val->is_Load() &&
2612       val->in(MemNode::Address)->eqv_uncast(adr) &&
2613       val->in(MemNode::Memory )->eqv_uncast(mem) &&
2614       val->as_Load()->store_Opcode() == Opcode()) {
2615     result = mem;
2616   }
2617 
2618   // Two stores in a row of the same value?
2619   if (result == this &&
2620       mem->is_Store() &&
2621       mem->in(MemNode::Address)->eqv_uncast(adr) &&
2622       mem->in(MemNode::ValueIn)->eqv_uncast(val) &&
2623       mem->Opcode() == Opcode()) {
2624     result = mem;
2625   }
2626 
2627   // Store of zero anywhere into a freshly-allocated object?
2628   // Then the store is useless.
2629   // (It must already have been captured by the InitializeNode.)
2630   if (result == this &&
2631       ReduceFieldZeroing && phase->type(val)->is_zero_type()) {
2632     // a newly allocated object is already all-zeroes everywhere
2633     if (mem->is_Proj() && mem->in(0)->is_Allocate()) {
2634       result = mem;
2635     }
2636 
2637     if (result == this) {
2638       // the store may also apply to zero-bits in an earlier object
2639       Node* prev_mem = find_previous_store(phase);
2640       // Steps (a), (b):  Walk past independent stores to find an exact match.
2641       if (prev_mem != NULL) {
2642         Node* prev_val = can_see_stored_value(prev_mem, phase);
2643         if (prev_val != NULL && phase->eqv(prev_val, val)) {
2644           // prev_val and val might differ by a cast; it would be good
2645           // to keep the more informative of the two.
2646           result = mem;
2647         }
2648       }
2649     }
2650   }
2651 
2652   if (result != this && phase->is_IterGVN() != NULL) {
2653     MemBarNode* trailing = trailing_membar();
2654     if (trailing != NULL) {
2655 #ifdef ASSERT
2656       const TypeOopPtr* t_oop = phase->type(in(Address))->isa_oopptr();
2657       assert(t_oop == NULL || t_oop->is_known_instance_field(), "only for non escaping objects");
2658 #endif
2659       PhaseIterGVN* igvn = phase->is_IterGVN();
2660       trailing->remove(igvn);
2661     }
2662   }
2663 
2664   return result;
2665 }

2934 //------------------------------Idealize---------------------------------------
2935 // Clearing a short array is faster with stores
2936 Node *ClearArrayNode::Ideal(PhaseGVN *phase, bool can_reshape) {
2937   // Already know this is a large node, do not try to ideal it
2938   if (!IdealizeClearArrayNode || _is_large) return NULL;
2939 
2940   const int unit = BytesPerLong;
2941   const TypeX* t = phase->type(in(2))->isa_intptr_t();
2942   if (!t)  return NULL;
2943   if (!t->is_con())  return NULL;
2944   intptr_t raw_count = t->get_con();
2945   intptr_t size = raw_count;
2946   if (!Matcher::init_array_count_is_in_bytes) size *= unit;
2947   // Clearing nothing uses the Identity call.
2948   // Negative clears are possible on dead ClearArrays
2949   // (see jck test stmt114.stmt11402.val).
2950   if (size <= 0 || size % unit != 0)  return NULL;
2951   intptr_t count = size / unit;
2952   // Length too long; communicate this to matchers and assemblers.
2953   // Assemblers are responsible to produce fast hardware clears for it.
2954   if (size > InitArrayShortSize) {
2955     return new ClearArrayNode(in(0), in(1), in(2), in(3), true);
2956   }
2957   Node *mem = in(1);
2958   if( phase->type(mem)==Type::TOP ) return NULL;
2959   Node *adr = in(3);
2960   const Type* at = phase->type(adr);
2961   if( at==Type::TOP ) return NULL;
2962   const TypePtr* atp = at->isa_ptr();
2963   // adjust atp to be the correct array element address type
2964   if (atp == NULL)  atp = TypePtr::BOTTOM;
2965   else              atp = atp->add_offset(Type::OffsetBot);
2966   // Get base for derived pointer purposes
2967   if( adr->Opcode() != Op_AddP ) Unimplemented();
2968   Node *base = adr->in(1);
2969 
2970   Node *zero = phase->makecon(TypeLong::ZERO);
2971   Node *off  = phase->MakeConX(BytesPerLong);
2972   mem = new StoreLNode(in(0),mem,adr,atp,zero,MemNode::unordered,false);
2973   count--;
2974   while( count-- ) {
2975     mem = phase->transform(mem);
2976     adr = phase->transform(new AddPNode(base,adr,off));
2977     mem = new StoreLNode(in(0),mem,adr,atp,zero,MemNode::unordered,false);
2978   }
2979   return mem;
2980 }

3031 Node* ClearArrayNode::clear_memory(Node* ctl, Node* mem, Node* dest,
3032                                    Node* start_offset,
3033                                    Node* end_offset,
3034                                    PhaseGVN* phase) {
3035   if (start_offset == end_offset) {
3036     // nothing to do
3037     return mem;
3038   }
3039 
3040   int unit = BytesPerLong;
3041   Node* zbase = start_offset;
3042   Node* zend  = end_offset;
3043 
3044   // Scale to the unit required by the CPU:
3045   if (!Matcher::init_array_count_is_in_bytes) {
3046     Node* shift = phase->intcon(exact_log2(unit));
3047     zbase = phase->transform(new URShiftXNode(zbase, shift) );
3048     zend  = phase->transform(new URShiftXNode(zend,  shift) );
3049   }
3050 
3051   // Bulk clear double-words
3052   Node* zsize = phase->transform(new SubXNode(zend, zbase) );
3053   Node* adr = phase->transform(new AddPNode(dest, dest, start_offset) );
3054   mem = new ClearArrayNode(ctl, mem, zsize, adr, false);
3055   return phase->transform(mem);
3056 }

3059                                    intptr_t start_offset,

3061                                    PhaseGVN* phase) {
3062   if (start_offset == end_offset) {

3065   }
3066 
3067   assert((end_offset % BytesPerInt) == 0, "odd end offset");

3359 //===========================InitializeNode====================================
3360 // SUMMARY:
3361 // This node acts as a memory barrier on raw memory, after some raw stores.
3362 // The 'cooked' oop value feeds from the Initialize, not the Allocation.
3363 // The Initialize can 'capture' suitably constrained stores as raw inits.
3364 // It can coalesce related raw stores into larger units (called 'tiles').
3365 // It can avoid zeroing new storage for memory units which have raw inits.
3366 // At macro-expansion, it is marked 'complete', and does not optimize further.
3367 //
3368 // EXAMPLE:
3369 // The object 'new short[2]' occupies 16 bytes in a 32-bit machine.
3370 //   ctl = incoming control; mem* = incoming memory
3371 // (Note:  A star * on a memory edge denotes I/O and other standard edges.)
3372 // First allocate uninitialized memory and fill in the header:
3373 //   alloc = (Allocate ctl mem* 16 #short[].klass ...)
3374 //   ctl := alloc.Control; mem* := alloc.Memory*
3375 //   rawmem = alloc.Memory; rawoop = alloc.RawAddress
3376 // Then initialize to zero the non-header parts of the raw memory block:
3377 //   init = (Initialize alloc.Control alloc.Memory* alloc.RawAddress)
3378 //   ctl := init.Control; mem.SLICE(#short[*]) := init.Memory
3379 // After the initialize node executes, the object is ready for service:
3380 //   oop := (CheckCastPP init.Control alloc.RawAddress #short[])
3381 // Suppose its body is immediately initialized as {1,2}:
3382 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3383 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3384 //   mem.SLICE(#short[*]) := store2
3385 //
3386 // DETAILS:
3387 // An InitializeNode collects and isolates object initialization after
3388 // an AllocateNode and before the next possible safepoint.  As a
3389 // memory barrier (MemBarNode), it keeps critical stores from drifting
3390 // down past any safepoint or any publication of the allocation.
3391 // Before this barrier, a newly-allocated object may have uninitialized bits.
3392 // After this barrier, it may be treated as a real oop, and GC is allowed.
3393 //
3394 // The semantics of the InitializeNode include an implicit zeroing of
3395 // the new object from object header to the end of the object.
3396 // (The object header and end are determined by the AllocateNode.)
3397 //
3398 // Certain stores may be added as direct inputs to the InitializeNode.
3399 // These stores must update raw memory, and they must be to addresses
3400 // derived from the raw address produced by AllocateNode, and with
3401 // a constant offset.  They must be ordered by increasing offset.
3402 // The first one is at in(RawStores), the last at in(req()-1).
3403 // Unlike most memory operations, they are not linked in a chain,
3404 // but are displayed in parallel as users of the rawmem output of
3405 // the allocation.
3406 //
3407 // (See comments in InitializeNode::capture_store, which continue
3408 // the example given above.)
3409 //
3410 // When the associated Allocate is macro-expanded, the InitializeNode
3411 // may be rewritten to optimize collected stores.  A ClearArrayNode
3412 // may also be created at that point to represent any required zeroing.
3413 // The InitializeNode is then marked 'complete', prohibiting further
3414 // capturing of nearby memory operations.
3415 //
3416 // During macro-expansion, all captured initializations which store
3417 // constant values of 32 bits or smaller are coalesced (if advantageous)
3418 // into larger 'tiles' 32 or 64 bits.  This allows an object to be
3419 // initialized in fewer memory operations.  Memory words which are
3420 // covered by neither tiles nor non-constant stores are pre-zeroed
3421 // by explicit stores of zero.  (The code shape happens to do all
3422 // zeroing first, then all other stores, with both sequences occurring
3423 // in order of ascending offsets.)
3424 //
3425 // Alternatively, code may be inserted between an AllocateNode and its
3426 // InitializeNode, to perform arbitrary initialization of the new object.
3427 // E.g., the object copying intrinsics insert complex data transfers here.
3428 // The initialization must then be marked as 'complete' disable the
3429 // built-in zeroing semantics and the collection of initializing stores.
3430 //
3431 // While an InitializeNode is incomplete, reads from the memory state
3432 // produced by it are optimizable if they match the control edge and
3433 // new oop address associated with the allocation/initialization.
3434 // They return a stored value (if the offset matches) or else zero.
3435 // A write to the memory state, if it matches control and address,
3436 // and if it is to a constant offset, may be 'captured' by the
3437 // InitializeNode.  It is cloned as a raw memory operation and rewired
3438 // inside the initialization, to the raw oop produced by the allocation.
3439 // Operations on addresses which are provably distinct (e.g., to
3440 // other AllocateNodes) are allowed to bypass the initialization.
3441 //
3442 // The effect of all this is to consolidate object initialization
3443 // (both arrays and non-arrays, both piecewise and bulk) into a
3444 // single location, where it can be optimized as a unit.
3445 //
3446 // Only stores with an offset less than TrackedInitializationLimit words
3447 // will be considered for capture by an InitializeNode.  This puts a
3448 // reasonable limit on the complexity of optimized initializations.
3449

3678                 // after the InitializeNode. We check the control of the
3679                 // object/array that is loaded from. If it's the same as
3680                 // the store control then we cannot capture the store.
3681                 assert(!n->is_Store(), "2 stores to same slice on same control?");
3682                 Node* base = other_adr;
3683                 assert(base->is_AddP(), "should be addp but is %s", base->Name());

3688                     failed = true;

3715 // Find the captured store in(i) which corresponds to the range
3716 // [start..start+size) in the initialized object.
3717 // If there is one, return its index i.  If there isn't, return the
3718 // negative of the index where it should be inserted.
3719 // Return 0 if the queried range overlaps an initialization boundary
3720 // or if dead code is encountered.
3721 // If size_in_bytes is zero, do not bother with overlap checks.
3722 int InitializeNode::captured_store_insertion_point(intptr_t start,
3723                                                    int size_in_bytes,
3724                                                    PhaseTransform* phase) {
3725   const int FAIL = 0, MAX_STORE = BytesPerLong;
3726 
3727   if (is_complete())
3728     return FAIL;                // arraycopy got here first; punt
3729 
3730   assert(allocation() != NULL, "must be present");
3731 
3732   // no negatives, no header fields:
3733   if (start < (intptr_t) allocation()->minimum_header_size())  return FAIL;
3734 
3735   // after a certain size, we bail out on tracking all the stores:
3736   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
3737   if (start >= ti_limit)  return FAIL;
3738 
3739   for (uint i = InitializeNode::RawStores, limit = req(); ; ) {
3740     if (i >= limit)  return -(int)i; // not found; here is where to put it
3741 
3742     Node*    st     = in(i);
3743     intptr_t st_off = get_store_offset(st, phase);
3744     if (st_off < 0) {
3745       if (st != zero_memory()) {
3746         return FAIL;            // bail out if there is dead garbage
3747       }
3748     } else if (st_off > start) {
3749       // ...we are done, since stores are ordered
3750       if (st_off < start + size_in_bytes) {
3751         return FAIL;            // the next store overlaps
3752       }
3753       return -(int)i;           // not found; here is where to put it
3754     } else if (st_off < start) {
3755       if (size_in_bytes != 0 &&
3756           start < st_off + MAX_STORE &&
3757           start < st_off + st->as_Store()->memory_size()) {
3758         return FAIL;            // the previous store overlaps
3759       }
3760     } else {
3761       if (size_in_bytes != 0 &&
3762           st->as_Store()->memory_size() != size_in_bytes) {
3763         return FAIL;            // mismatched store size
3764       }
3765       return i;
3766     }
3767 
3768     ++i;
3769   }
3770 }

3803 // Clone the given store, converting it into a raw store
3804 // initializing a field or element of my new object.
3805 // Caller is responsible for retiring the original store,
3806 // with subsume_node or the like.
3807 //
3808 // From the example above InitializeNode::InitializeNode,
3809 // here are the old stores to be captured:
3810 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3811 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3812 //
3813 // Here is the changed code; note the extra edges on init:
3814 //   alloc = (Allocate ...)
3815 //   rawoop = alloc.RawAddress
3816 //   rawstore1 = (StoreC alloc.Control alloc.Memory (+ rawoop 12) 1)
3817 //   rawstore2 = (StoreC alloc.Control alloc.Memory (+ rawoop 14) 2)
3818 //   init = (Initialize alloc.Control alloc.Memory rawoop
3819 //                      rawstore1 rawstore2)
3820 //
3821 Node* InitializeNode::capture_store(StoreNode* st, intptr_t start,
3822                                     PhaseGVN* phase, bool can_reshape) {
3823   assert(stores_are_sane(phase), "");
3824 
3825   if (start < 0)  return NULL;
3826   assert(can_capture_store(st, phase, can_reshape) == start, "sanity");
3827 
3828   Compile* C = phase->C;
3829   int size_in_bytes = st->memory_size();
3830   int i = captured_store_insertion_point(start, size_in_bytes, phase);
3831   if (i == 0)  return NULL;     // bail out
3832   Node* prev_mem = NULL;        // raw memory for the captured store
3833   if (i > 0) {
3834     prev_mem = in(i);           // there is a pre-existing store under this one
3835     set_req(i, C->top());       // temporarily disconnect it
3836     // See StoreNode::Ideal 'st->outcnt() == 1' for the reason to disconnect.
3837   } else {
3838     i = -i;                     // no pre-existing store
3839     prev_mem = zero_memory();   // a slice of the newly allocated object
3840     if (i > InitializeNode::RawStores && in(i-1) == prev_mem)
3841       set_req(--i, C->top());   // reuse this edge; it has been folded away
3842     else
3843       ins_req(i, C->top());     // build a new edge
3844   }
3845   Node* new_st = st->clone();
3846   new_st->set_req(MemNode::Control, in(Control));
3847   new_st->set_req(MemNode::Memory,  prev_mem);
3848   new_st->set_req(MemNode::Address, make_raw_address(start, phase));
3849   new_st = phase->transform(new_st);
3850 
3851   // At this point, new_st might have swallowed a pre-existing store
3852   // at the same offset, or perhaps new_st might have disappeared,
3853   // if it redundantly stored the same value (or zero to fresh memory).
3854 
3855   // In any case, wire it in:
3856   phase->igvn_rehash_node_delayed(this);
3857   set_req(i, new_st);
3858 
3859   // The caller may now kill the old guy.
3860   DEBUG_ONLY(Node* check_st = find_captured_store(start, size_in_bytes, phase));
3861   assert(check_st == new_st || check_st == NULL, "must be findable");
3862   assert(!is_complete(), "");
3863   return new_st;
3864 }

3866 static bool store_constant(jlong* tiles, int num_tiles,
3867                            intptr_t st_off, int st_size,
3868                            jlong con) {

3870     return false;               // strange store offset (assume size==2**N)
3871   address addr = (address)tiles + st_off;
3872   assert(st_off >= 0 && addr+st_size <= (address)&tiles[num_tiles], "oob");
3873   switch (st_size) {
3874   case sizeof(jbyte):  *(jbyte*) addr = (jbyte) con; break;
3875   case sizeof(jchar):  *(jchar*) addr = (jchar) con; break;
3876   case sizeof(jint):   *(jint*)  addr = (jint)  con; break;
3877   case sizeof(jlong):  *(jlong*) addr = (jlong) con; break;
3878   default: return false;        // strange store size (detect size!=2**N here)
3879   }
3880   return true;                  // return success to caller
3881 }

3883 // Coalesce subword constants into int constants and possibly
3884 // into long constants.  The goal, if the CPU permits,
3885 // is to initialize the object with a small number of 64-bit tiles.
3886 // Also, convert floating-point constants to bit patterns.
3887 // Non-constants are not relevant to this pass.
3888 //
3889 // In terms of the running example on InitializeNode::InitializeNode
3890 // and InitializeNode::capture_store, here is the transformation
3891 // of rawstore1 and rawstore2 into rawstore12:
3892 //   alloc = (Allocate ...)
3893 //   rawoop = alloc.RawAddress
3894 //   tile12 = 0x00010002
3895 //   rawstore12 = (StoreI alloc.Control alloc.Memory (+ rawoop 12) tile12)
3896 //   init = (Initialize alloc.Control alloc.Memory rawoop rawstore12)
3897 //
3898 void
3899 InitializeNode::coalesce_subword_stores(intptr_t header_size,
3900                                         Node* size_in_bytes,
3901                                         PhaseGVN* phase) {
3902   Compile* C = phase->C;
3903 
3904   assert(stores_are_sane(phase), "");
3905   // Note:  After this pass, they are not completely sane,
3906   // since there may be some overlaps.
3907 
3908   int old_subword = 0, old_long = 0, new_int = 0, new_long = 0;
3909 
3910   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
3911   intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, ti_limit);
3912   size_limit = MIN2(size_limit, ti_limit);
3913   size_limit = align_up(size_limit, BytesPerLong);
3914   int num_tiles = size_limit / BytesPerLong;
3915 
3916   // allocate space for the tile map:
3917   const int small_len = DEBUG_ONLY(true ? 3 :) 30; // keep stack frames small
3918   jlong  tiles_buf[small_len];
3919   Node*  nodes_buf[small_len];
3920   jlong  inits_buf[small_len];

3922                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
3923   Node** nodes = ((num_tiles <= small_len) ? &nodes_buf[0]
3924                   : NEW_RESOURCE_ARRAY(Node*, num_tiles));
3925   jlong* inits = ((num_tiles <= small_len) ? &inits_buf[0]
3926                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
3927   // tiles: exact bitwise model of all primitive constants
3928   // nodes: last constant-storing node subsumed into the tiles model
3929   // inits: which bytes (in each tile) are touched by any initializations
3930 
3931   //// Pass A: Fill in the tile model with any relevant stores.
3932 
3933   Copy::zero_to_bytes(tiles, sizeof(tiles[0]) * num_tiles);
3934   Copy::zero_to_bytes(nodes, sizeof(nodes[0]) * num_tiles);
3935   Copy::zero_to_bytes(inits, sizeof(inits[0]) * num_tiles);
3936   Node* zmem = zero_memory(); // initially zero memory state
3937   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
3938     Node* st = in(i);
3939     intptr_t st_off = get_store_offset(st, phase);
3940 
3941     // Figure out the store's offset and constant value:
3942     if (st_off < header_size)             continue; //skip (ignore header)
3943     if (st->in(MemNode::Memory) != zmem)  continue; //skip (odd store chain)
3944     int st_size = st->as_Store()->memory_size();
3945     if (st_off + st_size > size_limit)    break;
3946 
3947     // Record which bytes are touched, whether by constant or not.
3948     if (!store_constant(inits, num_tiles, st_off, st_size, (jlong) -1))
3949       continue;                 // skip (strange store size)
3950 
3951     const Type* val = phase->type(st->in(MemNode::ValueIn));
3952     if (!val->singleton())                continue; //skip (non-con store)
3953     BasicType type = val->basic_type();
3954 
3955     jlong con = 0;
3956     switch (type) {
3957     case T_INT:    con = val->is_int()->get_con();  break;
3958     case T_LONG:   con = val->is_long()->get_con(); break;
3959     case T_FLOAT:  con = jint_cast(val->getf());    break;
3960     case T_DOUBLE: con = jlong_cast(val->getd());   break;
3961     default:                              continue; //skip (odd store type)
3962     }
3963 
3964     if (type == T_LONG && Matcher::isSimpleConstant64(con) &&
3965         st->Opcode() == Op_StoreL) {
3966       continue;                 // This StoreL is already optimal.
3967     }
3968 
3969     // Store down the constant.
3970     store_constant(tiles, num_tiles, st_off, st_size, con);
3971 
3972     intptr_t j = st_off >> LogBytesPerLong;
3973 
3974     if (type == T_INT && st_size == BytesPerInt
3975         && (st_off & BytesPerInt) == BytesPerInt) {
3976       jlong lcon = tiles[j];
3977       if (!Matcher::isSimpleConstant64(lcon) &&
3978           st->Opcode() == Op_StoreI) {
3979         // This StoreI is already optimal by itself.
3980         jint* intcon = (jint*) &tiles[j];
3981         intcon[1] = 0;  // undo the store_constant()
3982 
3983         // If the previous store is also optimal by itself, back up and
3984         // undo the action of the previous loop iteration... if we can.
3985         // But if we can't, just let the previous half take care of itself.
3986         st = nodes[j];
3987         st_off -= BytesPerInt;
3988         con = intcon[0];

3990           assert(st_off >= header_size, "still ignoring header");
3991           assert(get_store_offset(st, phase) == st_off, "must be");
3992           assert(in(i-1) == zmem, "must be");
3993           DEBUG_ONLY(const Type* tcon = phase->type(st->in(MemNode::ValueIn)));
3994           assert(con == tcon->is_int()->get_con(), "must be");
3995           // Undo the effects of the previous loop trip, which swallowed st:
3996           intcon[0] = 0;        // undo store_constant()
3997           set_req(i-1, st);     // undo set_req(i, zmem)
3998           nodes[j] = NULL;      // undo nodes[j] = st
3999           --old_subword;        // undo ++old_subword

4001         continue;               // This StoreI is already optimal.
4002       }

4030       init0 = u.intcon[0];
4031       init1 = u.intcon[1];

4120 // Explore forward from in(start) to find the first fully initialized
4121 // word, and return its offset.  Skip groups of subword stores which
4122 // together initialize full words.  If in(start) is itself part of a
4123 // fully initialized word, return the offset of in(start).  If there
4124 // are no following full-word stores, or if something is fishy, return
4125 // a negative value.
4126 intptr_t InitializeNode::find_next_fullword_store(uint start, PhaseGVN* phase) {
4127   int       int_map = 0;
4128   intptr_t  int_map_off = 0;
4129   const int FULL_MAP = right_n_bits(BytesPerInt);  // the int_map we hope for
4130 
4131   for (uint i = start, limit = req(); i < limit; i++) {
4132     Node* st = in(i);
4133 
4134     intptr_t st_off = get_store_offset(st, phase);
4135     if (st_off < 0)  break;  // return conservative answer
4136 
4137     int st_size = st->as_Store()->memory_size();
4138     if (st_size >= BytesPerInt && (st_off % BytesPerInt) == 0) {
4139       return st_off;            // we found a complete word init
4140     }
4141 
4142     // update the map:
4143 
4144     intptr_t this_int_off = align_down(st_off, BytesPerInt);
4145     if (this_int_off != int_map_off) {
4146       // reset the map:
4147       int_map = 0;
4148       int_map_off = this_int_off;
4149     }
4150 
4151     int subword_off = st_off - this_int_off;
4152     int_map |= right_n_bits(st_size) << subword_off;
4153     if ((int_map & FULL_MAP) == FULL_MAP) {
4154       return this_int_off;      // we found a complete word init
4155     }
4156 
4157     // Did this store hit or cross the word boundary?
4158     intptr_t next_int_off = align_down(st_off + st_size, BytesPerInt);
4159     if (next_int_off == this_int_off + BytesPerInt) {
4160       // We passed the current int, without fully initializing it.
4161       int_map_off = next_int_off;
4162       int_map >>= BytesPerInt;
4163     } else if (next_int_off > this_int_off + BytesPerInt) {
4164       // We passed the current and next int.
4165       return this_int_off + BytesPerInt;
4166     }
4167   }
4168 
4169   return -1;

4189     coalesce_subword_stores(header_size, size_in_bytes, phase);
4190 
4191   Node* zmem = zero_memory();   // initially zero memory state
4192   Node* inits = zmem;           // accumulating a linearized chain of inits
4193   #ifdef ASSERT
4194   intptr_t first_offset = allocation()->minimum_header_size();
4195   intptr_t last_init_off = first_offset;  // previous init offset
4196   intptr_t last_init_end = first_offset;  // previous init offset+size
4197   intptr_t last_tile_end = first_offset;  // previous tile offset+size
4198   #endif
4199   intptr_t zeroes_done = header_size;
4200 
4201   bool do_zeroing = true;       // we might give up if inits are very sparse
4202   int  big_init_gaps = 0;       // how many large gaps have we seen?
4203 
4204   if (UseTLAB && ZeroTLAB)  do_zeroing = false;
4205   if (!ReduceFieldZeroing && !ReduceBulkZeroing)  do_zeroing = false;
4206 
4207   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
4208     Node* st = in(i);
4209     intptr_t st_off = get_store_offset(st, phase);
4210     if (st_off < 0)
4211       break;                    // unknown junk in the inits
4212     if (st->in(MemNode::Memory) != zmem)
4213       break;                    // complicated store chains somehow in list
4214 
4215     int st_size = st->as_Store()->memory_size();
4216     intptr_t next_init_off = st_off + st_size;
4217 
4218     if (do_zeroing && zeroes_done < next_init_off) {
4219       // See if this store needs a zero before it or under it.
4220       intptr_t zeroes_needed = st_off;
4221 
4222       if (st_size < BytesPerInt) {
4223         // Look for subword stores which only partially initialize words.
4224         // If we find some, we must lay down some word-level zeroes first,
4225         // underneath the subword stores.
4226         //
4227         // Examples:
4228         //   byte[] a = { p,q,r,s }  =>  a[0]=p,a[1]=q,a[2]=r,a[3]=s
4229         //   byte[] a = { x,y,0,0 }  =>  a[0..3] = 0, a[0]=x,a[1]=y
4230         //   byte[] a = { 0,0,z,0 }  =>  a[0..3] = 0, a[2]=z
4231         //
4232         // Note:  coalesce_subword_stores may have already done this,
4233         // if it was prompted by constant non-zero subword initializers.
4234         // But this case can still arise with non-constant stores.
4235 
4236         intptr_t next_full_store = find_next_fullword_store(i, phase);
4237 
4238         // In the examples above:
4239         //   in(i)          p   q   r   s     x   y     z
4240         //   st_off        12  13  14  15    12  13    14
4241         //   st_size        1   1   1   1     1   1     1
4242         //   next_full_s.  12  16  16  16    16  16    16
4243         //   z's_done      12  16  16  16    12  16    12
4244         //   z's_needed    12  16  16  16    16  16    16
4245         //   zsize          0   0   0   0     4   0     4
4246         if (next_full_store < 0) {
4247           // Conservative tack:  Zero to end of current word.
4248           zeroes_needed = align_up(zeroes_needed, BytesPerInt);
4249         } else {
4250           // Zero to beginning of next fully initialized word.
4251           // Or, don't zero at all, if we are already in that word.
4252           assert(next_full_store >= zeroes_needed, "must go forward");
4253           assert((next_full_store & (BytesPerInt-1)) == 0, "even boundary");
4254           zeroes_needed = next_full_store;
4255         }
4256       }
4257 
4258       if (zeroes_needed > zeroes_done) {
4259         intptr_t zsize = zeroes_needed - zeroes_done;
4260         // Do some incremental zeroing on rawmem, in parallel with inits.
4261         zeroes_done = align_down(zeroes_done, BytesPerInt);
4262         rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4263                                               zeroes_done, zeroes_needed,
4264                                               phase);
4265         zeroes_done = zeroes_needed;
4266         if (zsize > InitArrayShortSize && ++big_init_gaps > 2)
4267           do_zeroing = false;   // leave the hole, next time
4268       }
4269     }
4270 
4271     // Collect the store and move on:
4272     phase->replace_input_of(st, MemNode::Memory, inits);
4273     inits = st;                 // put it on the linearized chain
4274     set_req(i, zmem);           // unhook from previous position
4275 
4276     if (zeroes_done == st_off)
4277       zeroes_done = next_init_off;
4278 
4279     assert(!do_zeroing || zeroes_done >= next_init_off, "don't miss any");
4280 
4281     #ifdef ASSERT
4282     // Various order invariants.  Weaker than stores_are_sane because
4283     // a large constant tile can be filled in by smaller non-constant stores.
4284     assert(st_off >= last_init_off, "inits do not reverse");
4285     last_init_off = st_off;
4286     const Type* val = NULL;
4287     if (st_size >= BytesPerInt &&
4288         (val = phase->type(st->in(MemNode::ValueIn)))->singleton() &&
4289         (int)val->basic_type() < (int)T_OBJECT) {
4290       assert(st_off >= last_tile_end, "tiles do not overlap");
4291       assert(st_off >= last_init_end, "tiles do not overwrite inits");
4292       last_tile_end = MAX2(last_tile_end, next_init_off);
4293     } else {
4294       intptr_t st_tile_end = align_up(next_init_off, BytesPerLong);
4295       assert(st_tile_end >= last_tile_end, "inits stay with tiles");
4296       assert(st_off      >= last_init_end, "inits do not overlap");
4297       last_init_end = next_init_off;  // it's a non-tile
4298     }
4299     #endif //ASSERT
4300   }
4301 
4302   remove_extra_zeroes();        // clear out all the zmems left over
4303   add_req(inits);
4304 
4305   if (!(UseTLAB && ZeroTLAB)) {
4306     // If anything remains to be zeroed, zero it all now.
4307     zeroes_done = align_down(zeroes_done, BytesPerInt);
4308     // if it is the last unused 4 bytes of an instance, forget about it
4309     intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, max_jint);
4310     if (zeroes_done + BytesPerLong >= size_limit) {
4311       AllocateNode* alloc = allocation();
4312       assert(alloc != NULL, "must be present");
4313       if (alloc != NULL && alloc->Opcode() == Op_Allocate) {
4314         Node* klass_node = alloc->in(AllocateNode::KlassNode);
4315         ciKlass* k = phase->type(klass_node)->is_klassptr()->klass();
4316         if (zeroes_done == k->layout_helper())
4317           zeroes_done = size_limit;
4318       }
4319     }
4320     if (zeroes_done < size_limit) {
4321       rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4322                                             zeroes_done, size_in_bytes, phase);
4323     }
4324   }
4325 
4326   set_complete(phase);
4327   return rawmem;
4328 }

4356 //============================MergeMemNode=====================================
4357 //
4358 // SEMANTICS OF MEMORY MERGES:  A MergeMem is a memory state assembled from several
4359 // contributing store or call operations.  Each contributor provides the memory
4360 // state for a particular "alias type" (see Compile::alias_type).  For example,
4361 // if a MergeMem has an input X for alias category #6, then any memory reference
4362 // to alias category #6 may use X as its memory state input, as an exact equivalent
4363 // to using the MergeMem as a whole.
4364 //   Load<6>( MergeMem(<6>: X, ...), p ) <==> Load<6>(X,p)
4365 //
4366 // (Here, the <N> notation gives the index of the relevant adr_type.)
4367 //
4368 // In one special case (and more cases in the future), alias categories overlap.
4369 // The special alias category "Bot" (Compile::AliasIdxBot) includes all memory
4370 // states.  Therefore, if a MergeMem has only one contributing input W for Bot,
4371 // it is exactly equivalent to that state W:
4372 //   MergeMem(<Bot>: W) <==> W
4373 //
4374 // Usually, the merge has more than one input.  In that case, where inputs
4375 // overlap (i.e., one is Bot), the narrower alias type determines the memory
4376 // state for that type, and the wider alias type (Bot) fills in everywhere else:
4377 //   Load<5>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<5>(W,p)
4378 //   Load<6>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<6>(X,p)
4379 //
4380 // A merge can take a "wide" memory state as one of its narrow inputs.
4381 // This simply means that the merge observes out only the relevant parts of
4382 // the wide input.  That is, wide memory states arriving at narrow merge inputs
4383 // are implicitly "filtered" or "sliced" as necessary.  (This is rare.)
4384 //
4385 // These rules imply that MergeMem nodes may cascade (via their <Bot> links),
4386 // and that memory slices "leak through":
4387 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y)) <==> MergeMem(<Bot>: W, <7>: Y)
4388 //
4389 // But, in such a cascade, repeated memory slices can "block the leak":
4390 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y), <7>: Y') <==> MergeMem(<Bot>: W, <7>: Y')
4391 //
4392 // In the last example, Y is not part of the combined memory state of the
4393 // outermost MergeMem.  The system must, of course, prevent unschedulable
4394 // memory states from arising, so you can be sure that the state Y is somehow
4395 // a precursor to state Y'.
4396 //
4397 //
4398 // REPRESENTATION OF MEMORY MERGES: The indexes used to address the Node::in array
4399 // of each MergeMemNode array are exactly the numerical alias indexes, including
4400 // but not limited to AliasIdxTop, AliasIdxBot, and AliasIdxRaw.  The functions
4401 // Compile::alias_type (and kin) produce and manage these indexes.
4402 //
4403 // By convention, the value of in(AliasIdxTop) (i.e., in(1)) is always the top node.
4404 // (Note that this provides quick access to the top node inside MergeMem methods,
4405 // without the need to reach out via TLS to Compile::current.)
4406 //
4407 // As a consequence of what was just described, a MergeMem that represents a full
4408 // memory state has an edge in(AliasIdxBot) which is a "wide" memory state,
4409 // containing all alias categories.
4410 //
4411 // MergeMem nodes never (?) have control inputs, so in(0) is NULL.
4412 //
4413 // All other edges in(N) (including in(AliasIdxRaw), which is in(3)) are either
4414 // a memory state for the alias type <N>, or else the top node, meaning that
4415 // there is no particular input for that alias type.  Note that the length of
4416 // a MergeMem is variable, and may be extended at any time to accommodate new
4417 // memory states at larger alias indexes.  When merges grow, they are of course
4418 // filled with "top" in the unused in() positions.
4419 //
4420 // This use of top is named "empty_memory()", or "empty_mem" (no-memory) as a variable.
4421 // (Top was chosen because it works smoothly with passes like GCM.)
4422 //
4423 // For convenience, we hardwire the alias index for TypeRawPtr::BOTTOM.  (It is
4424 // the type of random VM bits like TLS references.)  Since it is always the
4425 // first non-Bot memory slice, some low-level loops use it to initialize an
4426 // index variable:  for (i = AliasIdxRaw; i < req(); i++).
4427 //
4428 //
4429 // ACCESSORS:  There is a special accessor MergeMemNode::base_memory which returns
4430 // the distinguished "wide" state.  The accessor MergeMemNode::memory_at(N) returns
4431 // the memory state for alias type <N>, or (if there is no particular slice at <N>,
4432 // it returns the base memory.  To prevent bugs, memory_at does not accept <Top>
4433 // or <Bot> indexes.  The iterator MergeMemStream provides robust iteration over
4434 // MergeMem nodes or pairs of such nodes, ensuring that the non-top edges are visited.
4435 //
4436 // %%%% We may get rid of base_memory as a separate accessor at some point; it isn't
4437 // really that different from the other memory inputs.  An abbreviation called
4438 // "bot_memory()" for "memory_at(AliasIdxBot)" would keep code tidy.
4439 //
4440 //
4441 // PARTIAL MEMORY STATES:  During optimization, MergeMem nodes may arise that represent
4442 // partial memory states.  When a Phi splits through a MergeMem, the copy of the Phi
4443 // that "emerges though" the base memory will be marked as excluding the alias types
4444 // of the other (narrow-memory) copies which "emerged through" the narrow edges:
4445 //
4446 //   Phi<Bot>(U, MergeMem(<Bot>: W, <8>: Y))
4447 //     ==Ideal=>  MergeMem(<Bot>: Phi<Bot-8>(U, W), Phi<8>(U, Y))
4448 //
4449 // This strange "subtraction" effect is necessary to ensure IGVN convergence.
4450 // (It is currently unimplemented.)  As you can see, the resulting merge is
4451 // actually a disjoint union of memory states, rather than an overlay.
4452 //

4514 //------------------------------Ideal------------------------------------------
4515 // This method is invoked recursively on chains of MergeMem nodes
4516 Node *MergeMemNode::Ideal(PhaseGVN *phase, bool can_reshape) {
4517   // Remove chain'd MergeMems
4518   //
4519   // This is delicate, because the each "in(i)" (i >= Raw) is interpreted
4520   // relative to the "in(Bot)".  Since we are patching both at the same time,
4521   // we have to be careful to read each "in(i)" relative to the old "in(Bot)",
4522   // but rewrite each "in(i)" relative to the new "in(Bot)".
4523   Node *progress = NULL;
4524 
4525 
4526   Node* old_base = base_memory();
4527   Node* empty_mem = empty_memory();
4528   if (old_base == empty_mem)
4529     return NULL; // Dead memory path.
4530 
4531   MergeMemNode* old_mbase;
4532   if (old_base != NULL && old_base->is_MergeMem())
4533     old_mbase = old_base->as_MergeMem();
4534   else
4535     old_mbase = NULL;
4536   Node* new_base = old_base;
4537 
4538   // simplify stacked MergeMems in base memory
4539   if (old_mbase)  new_base = old_mbase->base_memory();
4540 
4541   // the base memory might contribute new slices beyond my req()
4542   if (old_mbase)  grow_to_match(old_mbase);
4543 
4544   // Look carefully at the base node if it is a phi.
4545   PhiNode* phi_base;
4546   if (new_base != NULL && new_base->is_Phi())
4547     phi_base = new_base->as_Phi();
4548   else
4549     phi_base = NULL;
4550 
4551   Node*    phi_reg = NULL;
4552   uint     phi_len = (uint)-1;
4553   if (phi_base != NULL && !phi_base->is_copy()) {
4554     // do not examine phi if degraded to a copy
4555     phi_reg = phi_base->region();
4556     phi_len = phi_base->req();
4557     // see if the phi is unfinished
4558     for (uint i = 1; i < phi_len; i++) {
4559       if (phi_base->in(i) == NULL) {
4560         // incomplete phi; do not look at it yet!
4561         phi_reg = NULL;
4562         phi_len = (uint)-1;
4563         break;
4564       }
4565     }
4566   }
4567 
4568   // Note:  We do not call verify_sparse on entry, because inputs
4569   // can normalize to the base_memory via subsume_node or similar
4570   // mechanisms.  This method repairs that damage.
4571 
4572   assert(!old_mbase || old_mbase->is_empty_memory(empty_mem), "consistent sentinels");
4573 
4574   // Look at each slice.
4575   for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4576     Node* old_in = in(i);
4577     // calculate the old memory value
4578     Node* old_mem = old_in;
4579     if (old_mem == empty_mem)  old_mem = old_base;
4580     assert(old_mem == memory_at(i), "");
4581 
4582     // maybe update (reslice) the old memory value
4583 
4584     // simplify stacked MergeMems
4585     Node* new_mem = old_mem;
4586     MergeMemNode* old_mmem;
4587     if (old_mem != NULL && old_mem->is_MergeMem())
4588       old_mmem = old_mem->as_MergeMem();
4589     else
4590       old_mmem = NULL;
4591     if (old_mmem == this) {
4592       // This can happen if loops break up and safepoints disappear.
4593       // A merge of BotPtr (default) with a RawPtr memory derived from a
4594       // safepoint can be rewritten to a merge of the same BotPtr with
4595       // the BotPtr phi coming into the loop.  If that phi disappears
4596       // also, we can end up with a self-loop of the mergemem.
4597       // In general, if loops degenerate and memory effects disappear,
4598       // a mergemem can be left looking at itself.  This simply means
4599       // that the mergemem's default should be used, since there is
4600       // no longer any apparent effect on this slice.
4601       // Note: If a memory slice is a MergeMem cycle, it is unreachable
4602       //       from start.  Update the input to TOP.
4603       new_mem = (new_base == this || new_base == empty_mem)? empty_mem : new_base;
4604     }
4605     else if (old_mmem != NULL) {
4606       new_mem = old_mmem->memory_at(i);
4607     }
4608     // else preceding memory was not a MergeMem
4609 
4610     // replace equivalent phis (unfortunately, they do not GVN together)
4611     if (new_mem != NULL && new_mem != new_base &&
4612         new_mem->req() == phi_len && new_mem->in(0) == phi_reg) {
4613       if (new_mem->is_Phi()) {
4614         PhiNode* phi_mem = new_mem->as_Phi();
4615         for (uint i = 1; i < phi_len; i++) {
4616           if (phi_base->in(i) != phi_mem->in(i)) {
4617             phi_mem = NULL;
4618             break;
4619           }
4620         }
4621         if (phi_mem != NULL) {
4622           // equivalent phi nodes; revert to the def
4623           new_mem = new_base;
4624         }
4625       }
4626     }
4627 
4628     // maybe store down a new value
4629     Node* new_in = new_mem;
4630     if (new_in == new_base)  new_in = empty_mem;
4631 
4632     if (new_in != old_in) {
4633       // Warning:  Do not combine this "if" with the previous "if"
4634       // A memory slice might have be be rewritten even if it is semantically
4635       // unchanged, if the base_memory value has changed.
4636       set_req(i, new_in);
4637       progress = this;          // Report progress
4638     }
4639   }
4640 
4641   if (new_base != old_base) {
4642     set_req(Compile::AliasIdxBot, new_base);
4643     // Don't use set_base_memory(new_base), because we need to update du.
4644     assert(base_memory() == new_base, "");
4645     progress = this;
4646   }
4647 
4648   if( base_memory() == this ) {
4649     // a self cycle indicates this memory path is dead
4650     set_req(Compile::AliasIdxBot, empty_mem);
4651   }
4652 
4653   // Resolve external cycles by calling Ideal on a MergeMem base_memory
4654   // Recursion must occur after the self cycle check above
4655   if( base_memory()->is_MergeMem() ) {
4656     MergeMemNode *new_mbase = base_memory()->as_MergeMem();
4657     Node *m = phase->transform(new_mbase);  // Rollup any cycles
4658     if( m != NULL &&
4659         (m->is_top() ||
4660          (m->is_MergeMem() && m->as_MergeMem()->base_memory() == empty_mem)) ) {
4661       // propagate rollup of dead cycle to self
4662       set_req(Compile::AliasIdxBot, empty_mem);
4663     }
4664   }
4665 
4666   if( base_memory() == empty_mem ) {
4667     progress = this;
4668     // Cut inputs during Parse phase only.
4669     // During Optimize phase a dead MergeMem node will be subsumed by Top.
4670     if( !can_reshape ) {
4671       for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4672         if( in(i) != empty_mem ) { set_req(i, empty_mem); }
4673       }
4674     }
4675   }
4676 
4677   if( !progress && base_memory()->is_Phi() && can_reshape ) {
4678     // Check if PhiNode::Ideal's "Split phis through memory merges"
4679     // transform should be attempted. Look for this->phi->this cycle.
4680     uint merge_width = req();
4681     if (merge_width > Compile::AliasIdxRaw) {
4682       PhiNode* phi = base_memory()->as_Phi();
4683       for( uint i = 1; i < phi->req(); ++i ) {// For all paths in
4684         if (phi->in(i) == this) {
4685           phase->is_IterGVN()->_worklist.push(phi);
4686           break;
4687         }
4688       }
4689     }
4690   }
4691 
4692   assert(progress || verify_sparse(), "please, no dups of base");
4693   return progress;
4694 }

4696 //-------------------------set_base_memory-------------------------------------

4738 // verify a narrow slice (either incoming or outgoing)
4739 static void verify_memory_slice(const MergeMemNode* m, int alias_idx, Node* n) {
4740   if (!VerifyAliases)                return;  // don't bother to verify unless requested
4741   if (VMError::is_error_reported())  return;  // muzzle asserts when debugging an error
4742   if (Node::in_dump())               return;  // muzzle asserts when printing
4743   assert(alias_idx >= Compile::AliasIdxRaw, "must not disturb base_memory or sentinel");
4744   assert(n != NULL, "");
4745   // Elide intervening MergeMem's
4746   while (n->is_MergeMem()) {
4747     n = n->as_MergeMem()->memory_at(alias_idx);
4748   }
4749   Compile* C = Compile::current();
4750   const TypePtr* n_adr_type = n->adr_type();
4751   if (n == m->empty_memory()) {
4752     // Implicit copy of base_memory()
4753   } else if (n_adr_type != TypePtr::BOTTOM) {
4754     assert(n_adr_type != NULL, "new memory must have a well-defined adr_type");
4755     assert(C->must_alias(n_adr_type, alias_idx), "new memory must match selected slice");
4756   } else {
4757     // A few places like make_runtime_call "know" that VM calls are narrow,
4758     // and can be used to update only the VM bits stored as TypeRawPtr::BOTTOM.
4759     bool expected_wide_mem = false;
4760     if (n == m->base_memory()) {
4761       expected_wide_mem = true;
4762     } else if (alias_idx == Compile::AliasIdxRaw ||
4763                n == m->memory_at(Compile::AliasIdxRaw)) {
4764       expected_wide_mem = true;
4765     } else if (!C->alias_type(alias_idx)->is_rewritable()) {
4766       // memory can "leak through" calls on channels that
4767       // are write-once.  Allow this also.
4768       expected_wide_mem = true;
4769     }
4770     assert(expected_wide_mem, "expected narrow slice replacement");
4771   }
4772 }

4774 #define verify_memory_slice(m,i,n) (void)(0)  // PRODUCT version is no-op

4778 //-----------------------------memory_at---------------------------------------
4779 Node* MergeMemNode::memory_at(uint alias_idx) const {
4780   assert(alias_idx >= Compile::AliasIdxRaw ||
4781          alias_idx == Compile::AliasIdxBot && Compile::current()->AliasLevel() == 0,
4782          "must avoid base_memory and AliasIdxTop");
4783 
4784   // Otherwise, it is a narrow slice.
4785   Node* n = alias_idx < req() ? in(alias_idx) : empty_memory();
4786   Compile *C = Compile::current();
4787   if (is_empty_memory(n)) {
4788     // the array is sparse; empty slots are the "top" node
4789     n = base_memory();
4790     assert(Node::in_dump()
4791            || n == NULL || n->bottom_type() == Type::TOP
4792            || n->adr_type() == NULL // address is TOP
4793            || n->adr_type() == TypePtr::BOTTOM
4794            || n->adr_type() == TypeRawPtr::BOTTOM
4795            || Compile::current()->AliasLevel() == 0,
4796            "must be a wide memory");
4797     // AliasLevel == 0 if we are organizing the memory states manually.
4798     // See verify_memory_slice for comments on TypeRawPtr::BOTTOM.
4799   } else {
4800     // make sure the stored slice is sane
4801     #ifdef ASSERT
4802     if (VMError::is_error_reported() || Node::in_dump()) {
4803     } else if (might_be_same(n, base_memory())) {
4804       // Give it a pass:  It is a mostly harmless repetition of the base.
4805       // This can arise normally from node subsumption during optimization.
4806     } else {
4807       verify_memory_slice(this, alias_idx, n);
4808     }
4809     #endif
4810   }
4811   return n;
4812 }

4814 //---------------------------set_memory_at-------------------------------------
4815 void MergeMemNode::set_memory_at(uint alias_idx, Node *n) {
4816   verify_memory_slice(this, alias_idx, n);
4817   Node* empty_mem = empty_memory();
4818   if (n == base_memory())  n = empty_mem;  // collapse default
4819   uint need_req = alias_idx+1;
4820   if (req() < need_req) {
4821     if (n == empty_mem)  return;  // already the default, so do not grow me
4822     // grow the sparse array
4823     do {
4824       add_req(empty_mem);
4825     } while (req() < need_req);
4826   }
4827   set_req( alias_idx, n );
4828 }

4832 //--------------------------iteration_setup------------------------------------
4833 void MergeMemNode::iteration_setup(const MergeMemNode* other) {
4834   if (other != NULL) {
4835     grow_to_match(other);
4836     // invariant:  the finite support of mm2 is within mm->req()
4837     #ifdef ASSERT
4838     for (uint i = req(); i < other->req(); i++) {
4839       assert(other->is_empty_memory(other->in(i)), "slice left uncovered");
4840     }
4841     #endif
4842   }
4843   // Replace spurious copies of base_memory by top.
4844   Node* base_mem = base_memory();
4845   if (base_mem != NULL && !base_mem->is_top()) {
4846     for (uint i = Compile::AliasIdxBot+1, imax = req(); i < imax; i++) {
4847       if (in(i) == base_mem)
4848         set_req(i, empty_memory());
4849     }
4850   }
4851 }

4853 //---------------------------grow_to_match-------------------------------------
4854 void MergeMemNode::grow_to_match(const MergeMemNode* other) {
4855   Node* empty_mem = empty_memory();
4856   assert(other->is_empty_memory(empty_mem), "consistent sentinels");
4857   // look for the finite support of the other memory
4858   for (uint i = other->req(); --i >= req(); ) {
4859     if (other->in(i) != empty_mem) {
4860       uint new_len = i+1;
4861       while (req() < new_len)  add_req(empty_mem);
4862       break;
4863     }
4864   }
4865 }

4882   Node* n;
