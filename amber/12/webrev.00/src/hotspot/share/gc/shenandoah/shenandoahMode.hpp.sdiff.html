32 #define SHENANDOAH_CHECK_FLAG_SET(name)                                     \
33   do {                                                                      \
34     if (!(name)) {                                                          \
35       err_msg message("GC mode needs -XX:+" #name " to work correctly");    \
36       vm_exit_during_initialization("Error", message);                      \
37     }                                                                       \
38   } while (0)
39 
40 #define SHENANDOAH_CHECK_FLAG_UNSET(name)                                   \
41   do {                                                                      \
42     if ((name)) {                                                           \
43       err_msg message("GC mode needs -XX:-" #name " to work correctly");    \
44       vm_exit_during_initialization("Error", message);                      \
45     }                                                                       \
46   } while (0)
47 
48 class ShenandoahMode : public CHeapObj<mtGC> {
49 public:
50   virtual void initialize_flags() const = 0;
51   virtual ShenandoahHeuristics* initialize_heuristics() const = 0;
52   virtual const char* name() = 0;
53   virtual bool is_diagnostic() = 0;
54   virtual bool is_experimental() = 0;
55 };
