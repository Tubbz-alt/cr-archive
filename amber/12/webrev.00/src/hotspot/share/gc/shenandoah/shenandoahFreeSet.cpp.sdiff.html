150 HeapWord* ShenandoahFreeSet::try_allocate_in(ShenandoahHeapRegion* r, ShenandoahAllocRequest& req, bool& in_new_region) {
151   assert (!has_no_alloc_capacity(r), "Performance: should avoid full regions on this path: " SIZE_FORMAT, r->region_number());
152 
153   if (_heap->is_concurrent_root_in_progress() &&
154       r->is_trash()) {
155     return NULL;
156   }
157 
158   try_recycle_trashed(r);
159 
160   in_new_region = r->is_empty();
161 
162   HeapWord* result = NULL;
163   size_t size = req.size();
164 
165   if (ShenandoahElasticTLAB && req.is_lab_alloc()) {
166     size_t free = align_down(r->free() >> LogHeapWordSize, MinObjAlignment);
167     if (size > free) {
168       size = free;
169     }
170     if (size >= req.min_size()) {
171       result = r->allocate(size, req.type());
172       assert (result != NULL, "Allocation must succeed: free " SIZE_FORMAT ", actual " SIZE_FORMAT, free, size);
173     }
174   } else {
175     result = r->allocate(size, req.type());
176   }
177 
178   if (result != NULL) {
179     // Allocation successful, bump stats:
180     if (req.is_mutator_alloc()) {
181       increase_used(size * HeapWordSize);
182     }
183 
184     // Record actual allocation size
185     req.set_actual_size(size);
186 
187     if (req.is_gc_alloc() && _heap->is_concurrent_traversal_in_progress()) {
188       // Traversal needs to traverse through GC allocs. Adjust TAMS to the new top
189       // so that these allocations appear below TAMS, and thus get traversed.
190       // See top of shenandoahTraversal.cpp for an explanation.
191       _heap->marking_context()->capture_top_at_mark_start(r);
192       _heap->traversal_gc()->traversal_set()->add_region_check_for_duplicates(r);
193       OrderAccess::fence();
194     }
195   }
196 
197   if (result == NULL || has_no_alloc_capacity(r)) {
198     // Region cannot afford this or future allocations. Retire it.
199     //
200     // While this seems a bit harsh, especially in the case when this large allocation does not
201     // fit, but the next small one would, we are risking to inflate scan times when lots of
202     // almost-full regions precede the fully-empty region where we want allocate the entire TLAB.
203     // TODO: Record first fully-empty region, and use that for large allocations
204 
205     // Record the remainder as allocation waste
206     if (req.is_mutator_alloc()) {
207       size_t waste = r->free();
208       if (waste > 0) {
209         increase_used(waste);
210         _heap->notify_mutator_alloc_words(waste >> LogHeapWordSize, true);
211       }
212     }
213 
214     size_t num = r->region_number();
215     _collector_free_bitmap.clear_bit(num);
216     _mutator_free_bitmap.clear_bit(num);
217     // Touched the bounds? Need to update:
218     if (touches_bounds(num)) {
219       adjust_bounds();
220     }
221     assert_bounds();
222   }
223   return result;
224 }

228 }

241 void ShenandoahFreeSet::adjust_bounds() {
242   // Rewind both mutator bounds until the next bit.
243   while (_mutator_leftmost < _max && !is_mutator_free(_mutator_leftmost)) {
244     _mutator_leftmost++;
245   }
246   while (_mutator_rightmost > 0 && !is_mutator_free(_mutator_rightmost)) {
247     _mutator_rightmost--;
248   }
249   // Rewind both collector bounds until the next bit.
250   while (_collector_leftmost < _max && !is_collector_free(_collector_leftmost)) {
251     _collector_leftmost++;
252   }
253   while (_collector_rightmost > 0 && !is_collector_free(_collector_rightmost)) {
254     _collector_rightmost--;
255   }
256 }

258 HeapWord* ShenandoahFreeSet::allocate_contiguous(ShenandoahAllocRequest& req) {
259   assert_heaplock_owned_by_current_thread();
260 
261   size_t words_size = req.size();
262   size_t num = ShenandoahHeapRegion::required_regions(words_size * HeapWordSize);
263 
264   // No regions left to satisfy allocation, bye.
265   if (num > mutator_count()) {
266     return NULL;
267   }
268 
269   // Find the continuous interval of $num regions, starting from $beg and ending in $end,
270   // inclusive. Contiguous allocations are biased to the beginning.
271 
272   size_t beg = _mutator_leftmost;
273   size_t end = beg;
274 
275   while (true) {
276     if (end >= _max) {
277       // Hit the end, goodbye
278       return NULL;
279     }

284       end++;
285       beg = end;
286       continue;
287     }
288 
289     if ((end - beg + 1) == num) {
290       // found the match
291       break;
292     }
293 
294     end++;
295   };
296 
297   size_t remainder = words_size & ShenandoahHeapRegion::region_size_words_mask();
298 
299   // Initialize regions:
300   for (size_t i = beg; i <= end; i++) {
301     ShenandoahHeapRegion* r = _heap->get_region(i);
302     try_recycle_trashed(r);
303 
304     assert(i == beg || _heap->get_region(i-1)->region_number() + 1 == r->region_number(), "Should be contiguous");
305     assert(r->is_empty(), "Should be empty");
306 
307     if (i == beg) {
308       r->make_humongous_start();
309     } else {
310       r->make_humongous_cont();
311     }
312 
313     // Trailing region may be non-full, record the remainder there
314     size_t used_words;
315     if ((i == end) && (remainder != 0)) {
316       used_words = remainder;
317     } else {
318       used_words = ShenandoahHeapRegion::region_size_words();
319     }
320 
321     r->set_top(r->bottom() + used_words);
322     r->reset_alloc_metadata_to_shared();
323 
324     _mutator_free_bitmap.clear_bit(r->region_number());
325   }
326 
327   // While individual regions report their true use, all humongous regions are
328   // marked used in the free set.
329   increase_used(ShenandoahHeapRegion::region_size_bytes() * num);
330 
331   if (remainder != 0) {
332     // Record this remainder as allocation waste
333     _heap->notify_mutator_alloc_words(ShenandoahHeapRegion::region_size_words() - remainder, true);
334   }
335 
336   // Allocated at left/rightmost? Move the bounds appropriately.
337   if (beg == _mutator_leftmost || end == _mutator_rightmost) {
338     adjust_bounds();
339   }
340   assert_bounds();
341 
342   req.set_actual_size(words_size);
343   return _heap->get_region(beg)->bottom();
344 }

352     // This would be recycled on allocation path
353     return ShenandoahHeapRegion::region_size_bytes();
354   } else {
355     return r->free();
356   }
357 }

363 void ShenandoahFreeSet::try_recycle_trashed(ShenandoahHeapRegion *r) {
364   if (r->is_trash()) {

366     r->recycle();
367   }

370 void ShenandoahFreeSet::recycle_trash() {
371   // lock is not reentrable, check we don't have it
372   assert_heaplock_not_owned_by_current_thread();
373 
374   for (size_t i = 0; i < _heap->num_regions(); i++) {
375     ShenandoahHeapRegion* r = _heap->get_region(i);
376     if (r->is_trash()) {
377       ShenandoahHeapLocker locker(_heap->lock());
378       try_recycle_trashed(r);
379     }
380     SpinPause(); // allow allocators to take the lock
381   }

384 void ShenandoahFreeSet::flip_to_gc(ShenandoahHeapRegion* r) {
385   size_t idx = r->region_number();
386 
387   assert(_mutator_free_bitmap.at(idx), "Should be in mutator view");
388   assert(can_allocate_from(r), "Should not be allocated");
389 
390   _mutator_free_bitmap.clear_bit(idx);
391   _collector_free_bitmap.set_bit(idx);
392   _collector_leftmost = MIN2(idx, _collector_leftmost);
393   _collector_rightmost = MAX2(idx, _collector_rightmost);
394 
395   _capacity -= alloc_capacity(r);
396 
397   if (touches_bounds(idx)) {
398     adjust_bounds();
399   }
400   assert_bounds();
401 }

419 void ShenandoahFreeSet::rebuild() {
420   assert_heaplock_owned_by_current_thread();
421   clear();
422 
423   for (size_t idx = 0; idx < _heap->num_regions(); idx++) {
424     ShenandoahHeapRegion* region = _heap->get_region(idx);
425     if (region->is_alloc_allowed() || region->is_trash()) {
426       assert(!region->is_cset(), "Shouldn't be adding those to the free set");
427 
428       // Do not add regions that would surely fail allocation
429       if (has_no_alloc_capacity(region)) continue;
430 
431       _capacity += alloc_capacity(region);
432       assert(_used <= _capacity, "must not use more than we have");
433 
434       assert(!is_mutator_free(idx), "We are about to add it, it shouldn't be there already");
435       _mutator_free_bitmap.set_bit(idx);
436     }
437   }
438 
439   // Evac reserve: reserve trailing space for evacuations
440   size_t to_reserve = _heap->max_capacity() / 100 * ShenandoahEvacReserve;
441   size_t reserved = 0;
442 
443   for (size_t idx = _heap->num_regions() - 1; idx > 0; idx--) {
444     if (reserved >= to_reserve) break;
445 
446     ShenandoahHeapRegion* region = _heap->get_region(idx);
447     if (_mutator_free_bitmap.at(idx) && can_allocate_from(region)) {
448       _mutator_free_bitmap.clear_bit(idx);
449       _collector_free_bitmap.set_bit(idx);
450       size_t ac = alloc_capacity(region);
451       _capacity -= ac;
452       reserved += ac;
453     }
454   }
455 
456   recompute_bounds();
457   assert_bounds();
458 }

150   assert (!has_no_alloc_capacity(r), "Performance: should avoid full regions on this path: " SIZE_FORMAT, r->index());
151 
152   if (_heap->is_concurrent_root_in_progress() &&
153       r->is_trash()) {
154     return NULL;
155   }
156 
157   try_recycle_trashed(r);
158 
159   in_new_region = r->is_empty();
160 
161   HeapWord* result = NULL;
162   size_t size = req.size();
163 
164   if (ShenandoahElasticTLAB && req.is_lab_alloc()) {
165     size_t free = align_down(r->free() >> LogHeapWordSize, MinObjAlignment);
166     if (size > free) {
167       size = free;
168     }
169     if (size >= req.min_size()) {
170       result = r->allocate(size, req.type());
171       assert (result != NULL, "Allocation must succeed: free " SIZE_FORMAT ", actual " SIZE_FORMAT, free, size);
172     }
173   } else {
174     result = r->allocate(size, req.type());
175   }
176 
177   if (result != NULL) {
178     // Allocation successful, bump stats:
179     if (req.is_mutator_alloc()) {
180       increase_used(size * HeapWordSize);
181     }
182 
183     // Record actual allocation size
184     req.set_actual_size(size);
185 
186     if (req.is_gc_alloc()) {
187       r->set_update_watermark(r->top());





188     }
189   }
190 
191   if (result == NULL || has_no_alloc_capacity(r)) {
192     // Region cannot afford this or future allocations. Retire it.
193     //
194     // While this seems a bit harsh, especially in the case when this large allocation does not
195     // fit, but the next small one would, we are risking to inflate scan times when lots of
196     // almost-full regions precede the fully-empty region where we want allocate the entire TLAB.
197     // TODO: Record first fully-empty region, and use that for large allocations
198 
199     // Record the remainder as allocation waste
200     if (req.is_mutator_alloc()) {
201       size_t waste = r->free();
202       if (waste > 0) {
203         increase_used(waste);
204         _heap->notify_mutator_alloc_words(waste >> LogHeapWordSize, true);
205       }
206     }
207 
208     size_t num = r->index();
209     _collector_free_bitmap.clear_bit(num);
210     _mutator_free_bitmap.clear_bit(num);
211     // Touched the bounds? Need to update:
212     if (touches_bounds(num)) {
213       adjust_bounds();
214     }
215     assert_bounds();
216   }
217   return result;
218 }

235 void ShenandoahFreeSet::adjust_bounds() {
236   // Rewind both mutator bounds until the next bit.
237   while (_mutator_leftmost < _max && !is_mutator_free(_mutator_leftmost)) {
238     _mutator_leftmost++;
239   }
240   while (_mutator_rightmost > 0 && !is_mutator_free(_mutator_rightmost)) {
241     _mutator_rightmost--;
242   }
243   // Rewind both collector bounds until the next bit.
244   while (_collector_leftmost < _max && !is_collector_free(_collector_leftmost)) {
245     _collector_leftmost++;
246   }
247   while (_collector_rightmost > 0 && !is_collector_free(_collector_rightmost)) {
248     _collector_rightmost--;
249   }
250 }

252 HeapWord* ShenandoahFreeSet::allocate_contiguous(ShenandoahAllocRequest& req) {
253   shenandoah_assert_heaplocked();
254 
255   size_t words_size = req.size();
256   size_t num = ShenandoahHeapRegion::required_regions(words_size * HeapWordSize);
257 
258   // No regions left to satisfy allocation, bye.
259   if (num > mutator_count()) {
260     return NULL;
261   }
262 
263   // Find the continuous interval of $num regions, starting from $beg and ending in $end,
264   // inclusive. Contiguous allocations are biased to the beginning.
265 
266   size_t beg = _mutator_leftmost;
267   size_t end = beg;
268 
269   while (true) {
270     if (end >= _max) {
271       // Hit the end, goodbye
272       return NULL;
273     }

278       end++;
279       beg = end;
280       continue;
281     }
282 
283     if ((end - beg + 1) == num) {
284       // found the match
285       break;
286     }
287 
288     end++;
289   };
290 
291   size_t remainder = words_size & ShenandoahHeapRegion::region_size_words_mask();
292 
293   // Initialize regions:
294   for (size_t i = beg; i <= end; i++) {
295     ShenandoahHeapRegion* r = _heap->get_region(i);
296     try_recycle_trashed(r);
297 
298     assert(i == beg || _heap->get_region(i - 1)->index() + 1 == r->index(), "Should be contiguous");
299     assert(r->is_empty(), "Should be empty");
300 
301     if (i == beg) {
302       r->make_humongous_start();
303     } else {
304       r->make_humongous_cont();
305     }
306 
307     // Trailing region may be non-full, record the remainder there
308     size_t used_words;
309     if ((i == end) && (remainder != 0)) {
310       used_words = remainder;
311     } else {
312       used_words = ShenandoahHeapRegion::region_size_words();
313     }
314 
315     r->set_top(r->bottom() + used_words);

316 
317     _mutator_free_bitmap.clear_bit(r->index());
318   }
319 
320   // While individual regions report their true use, all humongous regions are
321   // marked used in the free set.
322   increase_used(ShenandoahHeapRegion::region_size_bytes() * num);
323 
324   if (remainder != 0) {
325     // Record this remainder as allocation waste
326     _heap->notify_mutator_alloc_words(ShenandoahHeapRegion::region_size_words() - remainder, true);
327   }
328 
329   // Allocated at left/rightmost? Move the bounds appropriately.
330   if (beg == _mutator_leftmost || end == _mutator_rightmost) {
331     adjust_bounds();
332   }
333   assert_bounds();
334 
335   req.set_actual_size(words_size);
336   return _heap->get_region(beg)->bottom();
337 }

345     // This would be recycled on allocation path
346     return ShenandoahHeapRegion::region_size_bytes();
347   } else {
348     return r->free();
349   }
350 }

363 void ShenandoahFreeSet::recycle_trash() {
364   // lock is not reentrable, check we don't have it
365   shenandoah_assert_not_heaplocked();
366 
367   for (size_t i = 0; i < _heap->num_regions(); i++) {
368     ShenandoahHeapRegion* r = _heap->get_region(i);
369     if (r->is_trash()) {
370       ShenandoahHeapLocker locker(_heap->lock());
371       try_recycle_trashed(r);
372     }
373     SpinPause(); // allow allocators to take the lock
374   }

377 void ShenandoahFreeSet::flip_to_gc(ShenandoahHeapRegion* r) {
378   size_t idx = r->index();
379 
380   assert(_mutator_free_bitmap.at(idx), "Should be in mutator view");
381   assert(can_allocate_from(r), "Should not be allocated");
382 
383   _mutator_free_bitmap.clear_bit(idx);
384   _collector_free_bitmap.set_bit(idx);
385   _collector_leftmost = MIN2(idx, _collector_leftmost);
386   _collector_rightmost = MAX2(idx, _collector_rightmost);
387 
388   _capacity -= alloc_capacity(r);
389 
390   if (touches_bounds(idx)) {
391     adjust_bounds();
392   }
393   assert_bounds();
394 }

412 void ShenandoahFreeSet::rebuild() {
413   shenandoah_assert_heaplocked();
414   clear();
415 
416   for (size_t idx = 0; idx < _heap->num_regions(); idx++) {
417     ShenandoahHeapRegion* region = _heap->get_region(idx);
418     if (region->is_alloc_allowed() || region->is_trash()) {
419       assert(!region->is_cset(), "Shouldn't be adding those to the free set");
420 
421       // Do not add regions that would surely fail allocation
422       if (has_no_alloc_capacity(region)) continue;
423 
424       _capacity += alloc_capacity(region);
425       assert(_used <= _capacity, "must not use more than we have");
426 
427       assert(!is_mutator_free(idx), "We are about to add it, it shouldn't be there already");
428       _mutator_free_bitmap.set_bit(idx);
429     }
430   }
431 
432   // Evac reserve: reserve trailing space for evacuations
433   size_t to_reserve = _heap->max_capacity() / 100 * ShenandoahEvacReserve;
434   size_t reserved = 0;
435 
436   for (size_t idx = _heap->num_regions() - 1; idx > 0; idx--) {
437     if (reserved >= to_reserve) break;
438 
439     ShenandoahHeapRegion* region = _heap->get_region(idx);
440     if (_mutator_free_bitmap.at(idx) && can_allocate_from(region)) {
441       _mutator_free_bitmap.clear_bit(idx);
442       _collector_free_bitmap.set_bit(idx);
443       size_t ac = alloc_capacity(region);
444       _capacity -= ac;
445       reserved += ac;
446     }
447   }
448 
449   recompute_bounds();
450   assert_bounds();
451 }

600 /*
601  * Internal fragmentation metric: describes how fragmented the heap regions are.
602  *
603  * It is derived as:
604  *
605  *               sum(used[i]^2, i=0..k)
606  *   IF = 1 - ------------------------------
607  *              C * sum(used[i], i=0..k)
608  *
609  * ...where k is the number of regions in computation, C is the region capacity, and
610  * used[i] is the used space in the region.
611  *
612  * The non-linearity causes IF to be lower for the cases where the same total heap
613  * used is densely packed. For example:
614  *   a) Heap is completely full  => IF = 0
615  *   b) Heap is half full, first 50% regions are completely full => IF = 0
616  *   c) Heap is half full, each region is 50% full => IF = 1/2
617  *   d) Heap is quarter full, first 50% regions are completely full => IF = 0
618  *   e) Heap is quarter full, each region is 25% full => IF = 3/4
619  *   f) Heap has one small object per each region => IF =~ 1
620  */
621 double ShenandoahFreeSet::internal_fragmentation() {
622   double squared = 0;
623   double linear = 0;
624   int count = 0;
625 
626   for (size_t index = _mutator_leftmost; index <= _mutator_rightmost; index++) {
627     if (is_mutator_free(index)) {
628       ShenandoahHeapRegion* r = _heap->get_region(index);
629       size_t used = r->used();
630       squared += used * used;
631       linear += used;
632       count++;
633     }
634   }
635 
636   if (count > 0) {
637     double s = squared / (ShenandoahHeapRegion::region_size_bytes() * linear);
638     return 1 - s;
639   } else {
640     return 0;
641   }
642 }

644 /*
645  * External fragmentation metric: describes how fragmented the heap is.
646  *
647  * It is derived as:
648  *
649  *   EF = 1 - largest_contiguous_free / total_free
650  *
651  * For example:
652  *   a) Heap is completely empty => EF = 0
653  *   b) Heap is completely full => EF = 0
654  *   c) Heap is first-half full => EF = 1/2
655  *   d) Heap is half full, full and empty regions interleave => EF =~ 1
656  */
657 double ShenandoahFreeSet::external_fragmentation() {
658   size_t last_idx = 0;
659   size_t max_contig = 0;
660   size_t empty_contig = 0;
661 
662   size_t free = 0;
663 
664   for (size_t index = _mutator_leftmost; index <= _mutator_rightmost; index++) {
665     if (is_mutator_free(index)) {
666       ShenandoahHeapRegion* r = _heap->get_region(index);
667       if (r->is_empty()) {
668         free += ShenandoahHeapRegion::region_size_bytes();
669         if (last_idx + 1 == index) {
670           empty_contig++;
671         } else {
672           empty_contig = 1;
673         }
674       } else {
675         empty_contig = 0;
676       }
677 
678       max_contig = MAX2(max_contig, empty_contig);
679       last_idx = index;
680     }
681   }
682 
683   if (free > 0) {
684     return 1 - (1.0 * max_contig * ShenandoahHeapRegion::region_size_bytes() / free);
685   } else {
686     return 0;
687   }
688 }
