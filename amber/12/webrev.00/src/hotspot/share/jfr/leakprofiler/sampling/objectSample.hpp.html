 25 #ifndef SHARE_JFR_LEAKPROFILER_SAMPLING_OBJECTSAMPLE_HPP
 26 #define SHARE_JFR_LEAKPROFILER_SAMPLING_OBJECTSAMPLE_HPP
 27 
 28 #include "jfr/utilities/jfrAllocation.hpp"
 29 #include "jfr/utilities/jfrBlob.hpp"
 30 #include "jfr/utilities/jfrTime.hpp"
 31 #include "jfr/utilities/jfrTypes.hpp"
 32 #include "memory/allocation.hpp"
 33 #include "oops/oop.hpp"
 34 #include "utilities/ticks.hpp"
 35 
 36 /*
 37  * Handle for diagnosing Java memory leaks.
 38  *
 39  * The class tracks the time the object was
 40  * allocated, the thread and the stack trace.
 41  */
 42 class ObjectSample : public JfrCHeapObj {
 43   friend class ObjectSampler;
 44   friend class SampleList;
 45  private:
 46   ObjectSample* _next;
 47   ObjectSample* _previous;
 48   JfrBlobHandle _stacktrace;
 49   JfrBlobHandle _thread;
 50   JfrBlobHandle _type_set;
 51   oop _object;
 52   Ticks _allocation_time;
 53   traceid _stack_trace_id;
 54   traceid _thread_id;
 55   int _index;
 56   size_t _span;
 57   size_t _allocated;
 58   size_t _heap_used_at_last_gc;
 59   unsigned int _stack_trace_hash;
 60 
 61   void release_references() {
 62     _stacktrace.~JfrBlobHandle();
 63     _thread.~JfrBlobHandle();
 64     _type_set.~JfrBlobHandle();
 65   }
 66 
 67   void reset() {
 68     _object = NULL;
 69     set_stack_trace_id(0);
 70     set_stack_trace_hash(0);
 71     release_references();
 72   }
 73 
 74  public:
 75   ObjectSample() : _next(NULL),
 76                    _previous(NULL),
 77                    _stacktrace(),
 78                    _thread(),
 79                    _type_set(),
 80                    _object(NULL),
 81                    _allocation_time(),
 82                    _stack_trace_id(0),
 83                    _thread_id(0),
 84                    _index(0),
 85                    _span(0),
 86                    _allocated(0),
 87                    _heap_used_at_last_gc(0),
 88                    _stack_trace_hash(0) {}
 89 
 90   ObjectSample* next() const {
 91     return _next;
 92   }
 93 
 94   void set_next(ObjectSample* next) {
 95     _next = next;
 96   }
 97 
 98   ObjectSample* prev() const {
 99     return _previous;
100   }
101 
102   void set_prev(ObjectSample* prev) {
103     _previous = prev;
104   }
105 
106   bool is_dead() const {
107     return object() == NULL;
108   }
109 
110   const oop object() const;
111   const oop object_raw() const;
112   void set_object(oop object);
113 
114   const oop* object_addr() const {
115     return &_object;
116   }
117 
118   int index() const {
119     return _index;
120   }
121 
122   void set_index(int index) {
123     _index = index;
124   }
125 
126   size_t span() const {
127     return _span;
128   }
129 
130   void set_span(size_t span) {
131     _span = span;
132   }
133 
134   void add_span(size_t span) {
135     _span += span;
136   }
137 
138   size_t allocated() const {
139     return _allocated;
140   }
141 
142   void set_allocated(size_t size) {
143     _allocated = size;
144   }
145 
146   const Ticks& allocation_time() const {
147     return _allocation_time;
148   }
149 
150   const void set_allocation_time(const JfrTicks& time) {
151     _allocation_time = Ticks(time.value());
152   }
153 
154   void set_heap_used_at_last_gc(size_t heap_used) {
155     _heap_used_at_last_gc = heap_used;
156   }
157 
158   size_t heap_used_at_last_gc() const {
159     return _heap_used_at_last_gc;
160   }
161 
162   bool has_stack_trace_id() const {
163     return stack_trace_id() != 0;
164   }
165 
166   traceid stack_trace_id() const {
167     return _stack_trace_id;
168   }
169 
170   void set_stack_trace_id(traceid id) {
171     _stack_trace_id = id;
172   }
173 
174   unsigned int stack_trace_hash() const {
175     return _stack_trace_hash;
176   }
177 
178   void set_stack_trace_hash(unsigned int hash) {
179     _stack_trace_hash = hash;
180   }
181 
182   traceid thread_id() const {
183     return _thread_id;
184   }
185 
186   void set_thread_id(traceid id) {
187     _thread_id = id;
188   }
189 
190   bool is_alive_and_older_than(jlong time_stamp) const {
191     return !is_dead() && (JfrTime::is_ft_enabled() ?
192       _allocation_time.ft_value() : _allocation_time.value()) < time_stamp;
193   }
194 
195   const JfrBlobHandle& stacktrace() const {
196     return _stacktrace;
197   }
198 
199   bool has_stacktrace() const {
200     return _stacktrace.valid();
201   }
202 
203   // JfrBlobHandle assignment operator
204   // maintains proper reference counting
205   void set_stacktrace(const JfrBlobHandle& ref) {
206     if (_stacktrace != ref) {
207       _stacktrace = ref;
208     }
209   }
210 
211   const JfrBlobHandle& thread() const {
212     return _thread;
213   }
214 
215   bool has_thread() const {
216     return _thread.valid();
217   }
218 
219   void set_thread(const JfrBlobHandle& ref) {
220     if (_thread != ref) {
221       _thread = ref;
222     }
223   }
224 
225   const JfrBlobHandle& type_set() const {
226     return _type_set;
227   }
228 
229   bool has_type_set() const {
230     return _type_set.valid();
231   }
232 
233   void set_type_set(const JfrBlobHandle& ref) {
234     if (_type_set != ref) {
235       if (_type_set.valid()) {
236         _type_set->set_next(ref);
237         return;
238       }
239       _type_set = ref;
240     }
241   }
242 };
243 
244 #endif // SHARE_JFR_LEAKPROFILER_SAMPLING_OBJECTSAMPLE_HPP
