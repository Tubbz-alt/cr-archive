26 /**
27  * Defines the {@linkplain javax.management.remote.rmi RMI connector}
28  * for the Java Management Extensions (JMX) Remote API.
29  *
30  * <dl>
31  * <dt class="simpleTagLabel" style="font-family:'DejaVu Sans', Arial, Helvetica, sans serif">Providers:</dt>
32  * <dd>This module provides the
33  * {@link javax.management.remote.JMXConnectorProvider} service,
34  * which creates JMX connector clients using the RMI protocol.
35  * Instances of {@code JMXConnector} can be obtained via the
36  * {@link javax.management.remote.JMXConnectorFactory#newJMXConnector
37  * JMXConnectorFactory.newJMXConnector} factory method.
38  * It also provides the {@link javax.management.remote.JMXConnectorServerProvider} service,
39  * which creates JMX connector servers using the RMI protocol.
40  * Instances of {@code JMXConnectorServer} can be obtained via the
41  * {@link javax.management.remote.JMXConnectorServerFactory#newJMXConnectorServer
42  * JMXConnectorServerFactory.newJMXConnectorServer} factory method.
43  * </dd>
44  * </dl>
45  *
46  * @provides javax.management.remote.JMXConnectorProvider
47  * @provides javax.management.remote.JMXConnectorServerProvider
48  *
49  * @moduleGraph
50  * @since 9
51  */
52 module java.management.rmi {
53 
54     requires java.naming;
55 
56     requires transitive java.management;
57     requires transitive java.rmi;
58 
59     exports javax.management.remote.rmi;
60 
61     // The qualified export below is required to preserve backward
62     // compatibility for the legacy case where an ordered list
63     // of package prefixes can be specified to the factory.
64     exports com.sun.jmx.remote.protocol.rmi to java.management;
65 
66     // jdk.management.agent needs to create an RMIExporter instance.
67     exports com.sun.jmx.remote.internal.rmi to jdk.management.agent;
68 
69     // The java.management.rmi module provides implementations
70     // of the JMXConnectorProvider and JMXConnectorServerProvider
71     // services supporting the RMI protocol.
72     provides javax.management.remote.JMXConnectorProvider with
73         com.sun.jmx.remote.protocol.rmi.ClientProvider;
74     provides javax.management.remote.JMXConnectorServerProvider with
75         com.sun.jmx.remote.protocol.rmi.ServerProvider;
76 
77 }
