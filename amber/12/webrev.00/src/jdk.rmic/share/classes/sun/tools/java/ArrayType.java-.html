26 package sun.tools.java;
27 
28 /**
29  * This class represents an Java array type.
30  * It overrides the relevant methods in class Type.
31  *
32  * WARNING: The contents of this source file are not part of any
33  * supported API.  Code that depends on them does so at its own risk:
34  * they are subject to change or removal without notice.
35  *
36  * @author      Arthur van Hoff
37  */
38 public final
39 class ArrayType extends Type {
40     /**
41      * The type of the element.
42      */
43     Type elemType;
44 
45     /**
46      * Construct an array type. Use Type.tArray to create
47      * a new array type.
48      */
49     ArrayType(String typeSig, Type elemType) {
50         super(TC_ARRAY, typeSig);
51         this.elemType = elemType;
52     }
53 
54     public Type getElementType() {
55         return elemType;
56     }
57 
58     public int getArrayDimension() {
59         return elemType.getArrayDimension() + 1;
60     }
61 
62     public String typeString(String id, boolean abbrev, boolean ret) {
63         return getElementType().typeString(id, abbrev, ret) + "[]";
64     }
65 }
