26 package sun.tools.tree;
27 
28 import sun.tools.java.*;
29 import sun.tools.asm.Assembler;
30 
31 /**
32  * WARNING: The contents of this source file are not part of any
33  * supported API.  Code that depends on them does so at its own risk:
34  * they are subject to change or removal without notice.
35  */
36 public
37 class BitNotExpression extends UnaryExpression {
38     /**
39      * Constructor
40      */
41     public BitNotExpression(long where, Expression right) {
42         super(BITNOT, where, right.type, right);
43     }
44 
45     /**
46      * Select the type of the expression
47      */
48     void selectType(Environment env, Context ctx, int tm) {
49         if ((tm & TM_LONG) != 0) {
50             type = Type.tLong;
51         } else {
52             type = Type.tInt;
53         }
54         right = convert(env, ctx, type, right);
55     }
56 
57     /**
58      * Evaluate
59      */
60     Expression eval(int a) {
61         return new IntExpression(where, ~a);
62     }
63     Expression eval(long a) {
64         return new LongExpression(where, ~a);
65     }
66 
67     /**
68      * Simplify
69      */
70     Expression simplify() {
71         if (right.op == BITNOT) {
72             return ((BitNotExpression)right).right;
73         }
74         return this;
75     }
76 
77     /**
78      * Code
79      */
80     public void codeValue(Environment env, Context ctx, Assembler asm) {
81         right.codeValue(env, ctx, asm);
82         if (type.isType(TC_INT)) {
83             asm.add(where, opc_ldc, -1);
84             asm.add(where, opc_ixor);
85         } else {
86             asm.add(where, opc_ldc2_w, -1L);
87             asm.add(where, opc_lxor);
88         }
89     }
90 }
