26 package sun.tools.tree;
27 
28 import sun.tools.java.*;
29 import java.io.PrintStream;
30 import java.util.Hashtable;
31 
32 /**
33  * WARNING: The contents of this source file are not part of any
34  * supported API.  Code that depends on them does so at its own risk:
35  * they are subject to change or removal without notice.
36  */
37 public
38 class TypeExpression extends Expression {
39     /**
40      * Constructor
41      */
42     public TypeExpression(long where, Type type) {
43         super(TYPE, where, type);
44     }
45 
46     /**
47      * Convert to a type
48      */
49     Type toType(Environment env, Context ctx) {
50         return type;
51     }
52 
53     /**
54      * Check an expression
55      */
56     public Vset checkValue(Environment env, Context ctx, Vset vset, Hashtable<Object, Object> exp) {
57         env.error(where, "invalid.term");
58         type = Type.tError;
59         return vset;
60     }
61 
62     public Vset checkAmbigName(Environment env, Context ctx, Vset vset, Hashtable<Object, Object> exp,
63                                UnaryExpression loc) {
64         return vset;
65     }
66 
67     public Expression inline(Environment env, Context ctx) {
68         return null;
69     }
70 
71     /**
72      * Print
73      */
74     public void print(PrintStream out) {
75         out.print(type.toString());
76     }
77 }
