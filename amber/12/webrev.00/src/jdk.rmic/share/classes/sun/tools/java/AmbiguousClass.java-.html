37 @SuppressWarnings("serial") // JDK implementation class
38 public
39 class AmbiguousClass extends ClassNotFound {
40     /**
41      * The class that was not found
42      */
43     public Identifier name1;
44     public Identifier name2;
45 
46     /**
47      * Constructor
48      */
49     public AmbiguousClass(Identifier name1, Identifier name2) {
50         super(name1.getName());
51         this.name1 = name1;
52         this.name2 = name2;
53     }
54 }
