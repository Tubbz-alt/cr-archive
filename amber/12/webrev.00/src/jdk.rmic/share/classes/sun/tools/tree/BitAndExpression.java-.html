28 import sun.tools.java.*;

30 
31 /**
32  * WARNING: The contents of this source file are not part of any
33  * supported API.  Code that depends on them does so at its own risk:
34  * they are subject to change or removal without notice.
35  */
36 public
37 class BitAndExpression extends BinaryBitExpression {
38     /**
39      * constructor
40      */
41     public BitAndExpression(long where, Expression left, Expression right) {
42         super(BITAND, where, left, right);
43     }
44 
45     /**
46      * Evaluate
47      */
48     Expression eval(boolean a, boolean b) {
49         return new BooleanExpression(where, a & b);
50     }
51     Expression eval(int a, int b) {
52         return new IntExpression(where, a & b);
53     }
54     Expression eval(long a, long b) {
55         return new LongExpression(where, a & b);
56     }
57 
58     /**
59      * Simplify
60      */
61     Expression simplify() {
62         if (left.equals(true))
63             return right;
64         if (right.equals(true))
65             return left;
66         if (left.equals(false) || left.equals(0))
67             return new CommaExpression(where, right, left).simplify();
68         if (right.equals(false) || right.equals(0))
69             return new CommaExpression(where, left, right).simplify();
70         return this;
71     }
72 
73     /**
74      * Code
75      */
76     void codeOperation(Environment env, Context ctx, Assembler asm) {
77         asm.add(where, opc_iand + type.getTypeCodeOffset());
78     }
79 }
