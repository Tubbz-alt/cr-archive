26 package sun.tools.tree;
27 
28 import sun.tools.java.*;
29 import sun.tools.asm.Assembler;
30 import java.io.PrintStream;
31 
32 /**
33  * WARNING: The contents of this source file are not part of any
34  * supported API.  Code that depends on them does so at its own risk:
35  * they are subject to change or removal without notice.
36  */
37 public
38 class FloatExpression extends ConstantExpression {
39     float value;
40 
41     /**
42      * Constructor
43      */
44     public FloatExpression(long where, float value) {
45         super(FLOATVAL, where, Type.tFloat);
46         this.value = value;
47     }
48 
49     /**
50      * Get the value
51      */
52     public Object getValue() {
53         return new Float(value);
54     }
55 
56     /**
57      * Check if the expression is equal to a value
58      */
59     public boolean equals(int i) {
60         return value == i;
61     }
62 
63     /**
64      * Check if the expression is equal to its default static value
65      */
66     public boolean equalsDefault() {
67         // don't allow -0.0
68         return (Float.floatToIntBits(value) == 0);
69     }
70 
71     /**
72      * Code
73      */
74     public void codeValue(Environment env, Context ctx, Assembler asm) {
75         asm.add(where, opc_ldc, new Float(value));
76     }
77 
78     /**
79      * Print
80      */
81     public void print(PrintStream out) {
82         out.print(value +"F");
83     }
84 }
