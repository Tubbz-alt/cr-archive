25 package org.graalvm.compiler.options;
26 
27 import java.util.Iterator;
28 
29 import jdk.internal.vm.compiler.collections.EconomicMap;
30 
31 /**
32  * A {@link OptionDescriptor} implementation that wraps an existing map from {@link String}s to
33  * {@link OptionDescriptor}s.
34  */
35 public final class OptionDescriptorsMap implements OptionDescriptors {
36     private final EconomicMap<String, OptionDescriptor> map;
37 
38     public OptionDescriptorsMap(EconomicMap<String, OptionDescriptor> map) {
39         this.map = map;
40     }
41 
42     @Override
43     public OptionDescriptor get(String key) {
44         return map.get(key);
45     }
46 
47     @Override
48     public Iterator<OptionDescriptor> iterator() {
49         return map.getValues().iterator();
50     }
51 }
