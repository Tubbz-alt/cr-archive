 26 package java.security.interfaces;
 27 
 28 import java.math.BigInteger;
 29 import java.security.spec.RSAOtherPrimeInfo;
 30 
 31 /**
 32  * The interface to an RSA multi-prime private key, as defined in the
 33  * <a href="https://tools.ietf.org/rfc/rfc8017.txt">PKCS#1 v2.2</a> standard,
 34  * using the <i>Chinese Remainder Theorem</i> (CRT) information values.
 35  *
 36  * @author Valerie Peng
 37  *
 38  *
 39  * @see java.security.spec.RSAPrivateKeySpec
 40  * @see java.security.spec.RSAMultiPrimePrivateCrtKeySpec
 41  * @see RSAPrivateKey
 42  * @see RSAPrivateCrtKey
 43  *
 44  * @since 1.4
 45  */
 46 
 47 public interface RSAMultiPrimePrivateCrtKey extends RSAPrivateKey {
 48 
 49     /**
 50      * The type fingerprint that is set to indicate
 51      * serialization compatibility with a previous
 52      * version of the type.
 53     *
 54     * @deprecated A {@code serialVersionUID} field in an interface is
 55     * ineffectual. Do not use; no replacement.
 56      */
 57     @Deprecated
 58     @SuppressWarnings("serial")
 59     @java.io.Serial
 60     static final long serialVersionUID = 618058533534628008L;
 61 
 62     /**
 63      * Returns the public exponent.
 64      *
 65      * @return the public exponent.
 66      */
 67     public BigInteger getPublicExponent();
 68 
 69     /**
 70      * Returns the primeP.
 71      *
 72      * @return the primeP.
 73      */
 74     public BigInteger getPrimeP();
 75 
 76     /**
 77      * Returns the primeQ.
 78      *
 79      * @return the primeQ.
 80      */
 81     public BigInteger getPrimeQ();
 82 
 83     /**
 84      * Returns the primeExponentP.
 85      *
 86      * @return the primeExponentP.
 87      */
 88     public BigInteger getPrimeExponentP();
 89 
 90     /**
 91      * Returns the primeExponentQ.
 92      *
 93      * @return the primeExponentQ.
 94      */
 95     public BigInteger getPrimeExponentQ();
 96 
 97     /**
 98      * Returns the crtCoefficient.
 99      *
100      * @return the crtCoefficient.
101      */
102     public BigInteger getCrtCoefficient();
103 
104     /**
105      * Returns the otherPrimeInfo or null if there are only
106      * two prime factors (p and q).
107      *
108      * @return the otherPrimeInfo.
109      */
110     public RSAOtherPrimeInfo[] getOtherPrimeInfo();
111 }
