26 package javax.lang.model.util;
27 
28 import javax.annotation.processing.SupportedSourceVersion;
29 import javax.lang.model.SourceVersion;
30 import static javax.lang.model.SourceVersion.*;
31 
32 /**
33  * A simple visitor for annotation values with default behavior
34  * appropriate for source version {@link SourceVersion#RELEASE_14 RELEASE_14}.
35  *
36  * Visit methods call {@link #defaultAction
37  * defaultAction} passing their arguments to {@code defaultAction}'s
38  * corresponding parameters.
39  *
40  * @apiNote
41  * Methods in this class may be overridden subject to their general
42  * contract.
43  *
44  * @param <R> the return type of this visitor's methods
45  * @param <P> the type of the additional parameter to this visitor's methods.
46  *
47  * @see <a href="SimpleAnnotationValueVisitor6.html#note_for_subclasses">
48  * <strong>Compatibility note for subclasses</strong></a>
49  * @see SimpleAnnotationValueVisitor6
50  * @see SimpleAnnotationValueVisitor7
51  * @see SimpleAnnotationValueVisitor8
52  * @see SimpleAnnotationValueVisitor9
53  * @since 14
54  */
55 @SupportedSourceVersion(RELEASE_15)
56 public class SimpleAnnotationValueVisitor14<R, P> extends SimpleAnnotationValueVisitor9<R, P> {
57     /**
58      * Constructor for concrete subclasses; uses {@code null} for the
59      * default value.
60      */
61     protected SimpleAnnotationValueVisitor14() {
62         super(null);
63     }
64 
65     /**
66      * Constructor for concrete subclasses; uses the argument for the
67      * default value.
68      *
69      * @param defaultValue the value to assign to {@link #DEFAULT_VALUE}
70      */
71     protected SimpleAnnotationValueVisitor14(R defaultValue) {
72         super(defaultValue);
73     }
74 }
