26 package com.sun.security.sasl.digest;
27 
28 import javax.security.sasl.SaslException;
29 
30 /**
31   * Interface used for classes implementing integrity checking and privacy
32   * for DIGEST-MD5 SASL mechanism implementation.
33   *
34   * @see <a href="http://www.ietf.org/rfc/rfc2831.txt">RFC 2831</a>
35   * - Using Digest Authentication as a SASL Mechanism
36   *
37   * @author Jonathan Bruce
38   */
39 
40 interface SecurityCtx {
41 
42     /**
43      * Wrap out-going message and return wrapped message
44      *
45      * @throws SaslException
46      */
47     byte[] wrap(byte[] dest, int start, int len)
48         throws SaslException;
49 
50     /**
51      * Unwrap incoming message and return original message
52      *
53      * @throws SaslException
54      */
55     byte[] unwrap(byte[] outgoing, int start, int len)
56         throws SaslException;
57 }
