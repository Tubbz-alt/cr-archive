26 package javax.security.sasl;
27 
28 import javax.security.auth.callback.TextInputCallback;
29 
30 /**
31   * This callback is used by {@code SaslClient} and {@code SaslServer}
32   * to retrieve realm information.
33   *
34   * @since 1.5
35   *
36   * @author Rosanna Lee
37   * @author Rob Weltman
38   */
39 public class RealmCallback extends TextInputCallback {
40 
41     /**
42      * Constructs a {@code RealmCallback} with a prompt.
43      *
44      * @param prompt The non-null prompt to use to request the realm information.
45      * @throws IllegalArgumentException If {@code prompt} is null or
46      * the empty string.
47      */
48     public RealmCallback(String prompt) {
49         super(prompt);
50     }
51 
52     /**
53      * Constructs a {@code RealmCallback} with a prompt and default
54      * realm information.
55      *
56      * @param prompt The non-null prompt to use to request the realm information.
57      * @param defaultRealmInfo The non-null default realm information to use.
58      * @throws IllegalArgumentException If {@code prompt} is null or
59      * the empty string,
60      * or if {@code defaultRealm} is empty or null.
61      */
62     public RealmCallback(String prompt, String defaultRealmInfo) {
63         super(prompt, defaultRealmInfo);
64     }
65 
66     private static final long serialVersionUID = -4342673378785456908L;
67 }
