26 package java.rmi.server;
27 
28 import java.rmi.RemoteException;
29 
30 /**
31  * This exception is thrown when a call is received that does not
32  * match the available skeleton.  It indicates either that the
33  * remote method names or signatures in this interface have changed or
34  * that the stub class used to make the call and the skeleton
35  * receiving the call were not generated by the same version of
36  * the stub protocol.
37  *
38  * @author  Roger Riggs
39  * @since   1.1
40  * @deprecated no replacement.  Skeletons are no longer required for remote
41  * method calls in the Java 2 platform v1.2 and greater.
42  */
43 @Deprecated
44 public class SkeletonMismatchException extends RemoteException {
45 
46     /* indicate compatibility with JDK 1.1.x version of class */
47     private static final long serialVersionUID = -7780460454818859281L;
48 
49     /**
50      * Constructs a new <code>SkeletonMismatchException</code> with
51      * a specified detail message.
52      *
53      * @param s the detail message
54      * @since 1.1
55      * @deprecated no replacement
56      */
57     @Deprecated
58     public SkeletonMismatchException(String s) {
59         super(s);
60     }
61 
62 }
