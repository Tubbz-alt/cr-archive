26 package java.rmi.server;
27 
28 /**
29  * An <code>Operation</code> contains a description of a Java method.
30  * <code>Operation</code> objects were used in JDK1.1 version stubs and
31  * skeletons. The <code>Operation</code> class is not needed for 1.2 style
32  * stubs; hence, this class is deprecated.
33  *
34  * @since 1.1
35  * @deprecated no replacement
36  */
37 @Deprecated
38 public class Operation {
39     private String operation;
40 
41     /**
42      * Creates a new Operation object.
43      * @param op method name
44      * @deprecated no replacement
45      * @since 1.1
46      */
47     @Deprecated
48     public Operation(String op) {
49         operation = op;
50     }
51 
52     /**
53      * Returns the name of the method.
54      * @return method name
55      * @deprecated no replacement
56      * @since 1.1
57      */
58     @Deprecated
59     public String getOperation() {
60         return operation;
61     }
62 
63     /**
64      * Returns the string representation of the operation.
65      * @deprecated no replacement
66      * @since 1.1
67      */
68     @Deprecated
69     public String toString() {
70         return operation;
71     }
72 }
