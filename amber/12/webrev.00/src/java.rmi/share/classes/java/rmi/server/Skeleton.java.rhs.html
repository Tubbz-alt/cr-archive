25 package java.rmi.server;
26 
27 import java.rmi.Remote;
28 
29 /**
30  * The <code>Skeleton</code> interface is used solely by the RMI
31  * implementation.
32  *
33  * <p> Every version 1.1 compatible skeleton implements this interface.
34  * A skeleton for a remote object is a server-side entity that dispatches calls
35  * to the actual remote object implementation.


36  *
37  * @author  Ann Wollrath
38  * @since   1.1
39  * @deprecated no replacement.  Skeletons are no longer required for remote
40  * method calls in the Java 2 platform v1.2 and greater.
41  */
42 @Deprecated
43 public interface Skeleton {
44     /**
45      * Unmarshals arguments, calls the actual remote object implementation,
46      * and marshals the return value or any exception.
47      *
48      * @param obj remote implementation to dispatch call to
49      * @param theCall object representing remote call
50      * @param opnum operation number
51      * @param hash stub/skeleton interface hash
52      * @exception java.lang.Exception if a general exception occurs.
53      * @since 1.1
54      * @deprecated no replacement
55      */
56     @Deprecated
57     void dispatch(Remote obj, RemoteCall theCall, int opnum, long hash)
58         throws Exception;
59 
60     /**
61      * Returns the operations supported by the skeleton.
62      * @return operations supported by skeleton
63      * @since 1.1
64      * @deprecated no replacement
65      */
66     @Deprecated
67     Operation[] getOperations();
68 }
