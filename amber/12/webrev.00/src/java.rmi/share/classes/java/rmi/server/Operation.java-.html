26 package java.rmi.server;
27 
28 /**
29  * An <code>Operation</code> contains a description of a Java method.
30  * <code>Operation</code> objects were used in JDK1.1 version stubs and
31  * skeletons. The <code>Operation</code> class is not needed for 1.2 style
32  * stubs (stubs generated with <code>rmic -v1.2</code>); hence, this class
33  * is deprecated.
34  *
35  * @since 1.1
36  * @deprecated no replacement
37  */
38 @Deprecated
39 public class Operation {
40     private String operation;
41 
42     /**
43      * Creates a new Operation object.
44      * @param op method name
45      * @deprecated no replacement
46      * @since 1.1
47      */
48     @Deprecated
49     public Operation(String op) {
50         operation = op;
51     }
52 
53     /**
54      * Returns the name of the method.
55      * @return method name
56      * @deprecated no replacement
57      * @since 1.1
58      */
59     @Deprecated
60     public String getOperation() {
61         return operation;
62     }
63 
64     /**
65      * Returns the string representation of the operation.
66      * @deprecated no replacement
67      * @since 1.1
68      */
69     @Deprecated
70     public String toString() {
71         return operation;
72     }
73 }
