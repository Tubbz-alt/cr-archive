  9 package jdk.internal.org.jline.reader;
 10 
 11 import java.util.Objects;
 12 
 13 /**
 14  * A completion candidate.
 15  *
 16  * @author <a href="mailto:gnodet@gmail.com">Guillaume Nodet</a>
 17  */
 18 public class Candidate implements Comparable<Candidate> {
 19 
 20     private final String value;
 21     private final String displ;
 22     private final String group;
 23     private final String descr;
 24     private final String suffix;
 25     private final String key;
 26     private final boolean complete;
 27 
 28     /**
 29      * Simple constructor with only a single String as an argument.
 30      *
 31      * @param value the candidate
 32      */
 33     public Candidate(String value) {
 34         this(value, value, null, null, null, null, true);
 35     }
 36 
 37     /**
 38      * Constructs a new Candidate.
 39      *
 40      * @param value the value
 41      * @param displ the display string
 42      * @param group the group
 43      * @param descr the description
 44      * @param suffix the suffix
 45      * @param key the key
 46      * @param complete the complete flag
 47      */
 48     public Candidate(String value, String displ, String group, String descr, String suffix, String key, boolean complete) {
 49         this.value = Objects.requireNonNull(value);
 50         this.displ = Objects.requireNonNull(displ);
 51         this.group = group;
 52         this.descr = descr;
 53         this.suffix = suffix;
 54         this.key = key;
 55         this.complete = complete;
 56     }
 57 
 58     /**
 59      * The value that will be used for the actual completion.
 60      * This string should not contain ANSI sequences.
 61      * @return the value
 62      */
 63     public String value() {
 64         return value;
 65     }
 66 
 67     /**
 68      * The string that will be displayed to the user.
 69      * This string may contain ANSI sequences.
 70      * @return the display string
 71      */
 72     public String displ() {
 73         return displ;
 74     }
 75 
 76     /**
 77      * The group name for this candidate.
 78      * Candidates can be grouped together and this string is used
 79      * as a key for the group and displayed to the user.
 80      * @return the group
 81      *
 82      * @see LineReader.Option#GROUP
 83      * @see LineReader.Option#AUTO_GROUP
 84      */
 85     public String group() {
 86         return group;
 87     }
 88 
 89     /**
 90      * Description of this candidate, usually a small help message
 91      * to understand the meaning of this candidate.
 92      * This string may contain ANSI sequences.
 93      * @return the description
 94      */
 95     public String descr() {
 96         return descr;
 97     }
 98 
 99     /**
100      * The suffix is added when this candidate is displayed.
101      * However, if the next character entered does not match,
102      * the suffix will be automatically removed.
103      * This string should not contain ANSI sequences.
104      * @return the suffix
105      *
106      * @see LineReader.Option#AUTO_REMOVE_SLASH
107      * @see LineReader#REMOVE_SUFFIX_CHARS
108      */
109     public String suffix() {
110         return suffix;
111     }
112 
113     /**
114      * Candidates which have the same key will be merged together.
115      * For example, if a command has multiple aliases, they can be merged
116      * if they are using the same key.
117      * @return the key
118      */
119     public String key() {
120         return key;
121     }
122 
123     /**
124      * Boolean indicating whether this candidate is complete or
125      * if the completer may further expand the candidate value
126      * after this candidate has been selected.
127      * This can be the case when completing folders for example.
128      * If the candidate is complete and is selected, a space
129      * separator will be added.
130      * @return the completion flag
131      */
132     public boolean complete() {
133         return complete;
134     }
135 
136     @Override
137     public int compareTo(Candidate o) {
138         return value.compareTo(o.value);
139     }
140 }
