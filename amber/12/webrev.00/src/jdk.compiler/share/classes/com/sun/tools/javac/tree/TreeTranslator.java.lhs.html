 32 /** A subclass of Tree.Visitor, this class defines
 33  *  a general tree translator pattern. Translation proceeds recursively in
 34  *  left-to-right order down a tree, constructing translated nodes by
 35  *  overwriting existing ones. There is one visitor method in this class
 36  *  for every possible kind of tree node.  To obtain a specific
 37  *  translator, it suffices to override those visitor methods which
 38  *  do some interesting work. The translator class itself takes care of all
 39  *  navigational aspects.
 40  *
 41  *  <p><b>This is NOT part of any supported API.
 42  *  If you write code that depends on this, you do so at your own risk.
 43  *  This code and its internal interfaces are subject to change or
 44  *  deletion without notice.</b>
 45  */
 46 public class TreeTranslator extends JCTree.Visitor {
 47 
 48     /** Visitor result field: a tree
 49      */
 50     protected JCTree result;
 51 
 52     /** Visitor method: Translate a single node.
 53      */
 54     @SuppressWarnings("unchecked")
 55     public <T extends JCTree> T translate(T tree) {
 56         if (tree == null) {
 57             return null;
 58         } else {
 59             tree.accept(this);
 60             JCTree tmpResult = this.result;
 61             this.result = null;
 62             return (T)tmpResult; // XXX cast
 63         }
 64     }
 65 
 66     /** Visitor method: translate a list of nodes.
 67      */
 68     public <T extends JCTree> List<T> translate(List<T> trees) {
 69         if (trees == null) return null;
 70         for (List<T> l = trees; l.nonEmpty(); l = l.tail)
 71             l.head = translate(l.head);
 72         return trees;
 73     }

310     public void visitNewArray(JCNewArray tree) {
311         tree.annotations = translate(tree.annotations);
312         List<List<JCAnnotation>> dimAnnos = List.nil();
313         for (List<JCAnnotation> origDimAnnos : tree.dimAnnotations)
314             dimAnnos = dimAnnos.append(translate(origDimAnnos));
315         tree.dimAnnotations = dimAnnos;
316         tree.elemtype = translate(tree.elemtype);
317         tree.dims = translate(tree.dims);
318         tree.elems = translate(tree.elems);
319         result = tree;
320     }
