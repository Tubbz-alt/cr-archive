600     public JCTree visitOther(Tree node, P p) {
601         JCTree tree = (JCTree) node;
602         switch (tree.getTag()) {
603             case LETEXPR: {
604                 LetExpr t = (LetExpr) node;
605                 List<? extends JCStatement> defs = copy(t.defs, p);
606                 JCExpression expr = copy(t.expr, p);
607                 return M.at(t.pos).LetExpr(defs, expr);
608             }
609             default:
610                 throw new AssertionError("unknown tree tag: " + tree.getTag());
611         }
612     }
