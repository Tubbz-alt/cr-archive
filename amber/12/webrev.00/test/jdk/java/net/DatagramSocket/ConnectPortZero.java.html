 49 /*
 50  * @test
 51  * @bug 8240533
 52  * @summary Check that DatagramSocket, MulticastSocket and DatagramSocketAdaptor
 53  *          throw expected Exception when connecting to port 0
 54  * @run testng/othervm ConnectPortZero
 55  */
 56 
 57 public class ConnectPortZero{
 58     private InetAddress loopbackAddr, wildcardAddr;
 59     private DatagramSocket datagramSocket, datagramSocketAdaptor;
 60     private MulticastSocket multicastSocket;
 61 
 62     private static final Class<SocketException> SE = SocketException.class;
 63     private static final Class<UncheckedIOException> UCIOE =
 64             UncheckedIOException.class;
 65     private static final Class<AccessControlException> ACE =
 66             AccessControlException.class;
 67 
 68     @BeforeTest
 69     public void setUp() throws IOException {
 70         loopbackAddr = InetAddress.getLoopbackAddress();
 71         wildcardAddr = new InetSocketAddress(0).getAddress();
 72 
 73         datagramSocket = new DatagramSocket();
 74         multicastSocket = new MulticastSocket();
 75         datagramSocketAdaptor = DatagramChannel.open().socket();
 76     }
 77 
 78     @DataProvider(name = "data")
 79     public Object[][] variants() {
 80         return new Object[][]{
 81                 { datagramSocket,        loopbackAddr },
 82                 { datagramSocketAdaptor, loopbackAddr },
 83                 { multicastSocket,       loopbackAddr },
 84                 { datagramSocket,        wildcardAddr },
 85                 { datagramSocketAdaptor, wildcardAddr },
 86                 { multicastSocket,       wildcardAddr }
 87         };
 88     }
 89 
 90     @Test(dataProvider = "data")
 91     public void testConnect(DatagramSocket ds, InetAddress addr) {
 92         Throwable t = expectThrows(UCIOE, () -> ds.connect(addr, 0));
 93         assertEquals(t.getCause().getClass(), SE);
 94 
 95         assertThrows(SE, () -> ds
 96                 .connect(new InetSocketAddress(addr, 0)));
 97     }
 98 
 99 
100     // Check that 0 port check doesn't override security manager check
101     @Test(dataProvider = "data")
102     public void testConnectWithSecurityManager(DatagramSocket ds,
103                                                InetAddress addr) {
104         Policy defaultPolicy = Policy.getPolicy();
105         try {
106             Policy.setPolicy(new NoSendPolicy());
107             System.setSecurityManager(new SecurityManager());
108 
109             assertThrows(ACE, () -> ds.connect(addr, 0));
110             assertThrows(ACE, () ->
111                     ds.connect(new InetSocketAddress(addr, 0)));
112         } finally {
113             System.setSecurityManager(null);
114             Policy.setPolicy(defaultPolicy);
115         }
116     }
117 
118     static class NoSendPolicy extends Policy {
119         final PermissionCollection perms = new Permissions();
120         { perms.add(new SocketPermission("*:0", "connect")); }
121 
122         public boolean implies(ProtectionDomain domain, Permission perm) {
123             return !perms.implies(perm);
124         }
125     }
126 
127     @AfterTest
128     public void tearDown() {
129         datagramSocket.close();
130         multicastSocket.close();
131         datagramSocketAdaptor.close();
132     }
133 }
