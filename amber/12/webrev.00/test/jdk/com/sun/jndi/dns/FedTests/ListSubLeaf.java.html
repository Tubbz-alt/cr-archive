24 import javax.naming.Binding;
25 import javax.naming.Context;
26 import javax.naming.NamingEnumeration;
27 import javax.naming.directory.InitialDirContext;
28 
29 /*
30  * @test
31  * @bug 8210339
32  * @summary list of a leaf using a composite name with DNS name as
33  *          first component
34  * @library ../lib/
35  * @modules java.naming/com.sun.jndi.toolkit.dir
36  *          java.base/sun.security.util
37  * @build FedSubordinateNs FedObjectFactory
38  * @run main/othervm ListSubLeaf
39  */
40 
41 public class ListSubLeaf extends ListFedBase {
42 
43     private static final int COUNT_LIMIT = 0; // no entry
44 
45     public static void main(String[] args) throws Exception {
46         new ListSubLeaf().run(args);
47     }
48 
49     /*
50      * list of a leaf using a composite name with DNS name as
51      * first component
52      */
53     @Override
54     public void runTest() throws Exception {
55         env().put(Context.OBJECT_FACTORIES, "FedObjectFactory");
56         setContext(new InitialDirContext(env()));
57 
58         NamingEnumeration<Binding> enumObj = context().listBindings(getKey() + "/a/b/c");
59         verifyNamingEnumeration(enumObj, COUNT_LIMIT);
60     }
61 }
