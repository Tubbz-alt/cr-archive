24 import javax.naming.Binding;
25 import javax.naming.NamingEnumeration;
26 import javax.naming.NamingException;
27 
28 /**
29  * Abstract test base for Fed list related test, this class extends DNSTestBase.
30  *
31  * @see DNSTestBase
32  * @see TestBase
33  */
34 abstract class ListFedBase extends DNSTestBase {
35     private String key;
36 
37     public ListFedBase() {
38         // set default test data
39         setKey("host1");
40     }
41 
42     /**
43      * Verify given NamingEnumeration match expected count.
44      *
45      * @param enumObj       given NamingEnumeration instance
46      * @param expectedCount given expected count
47      * @throws NamingException
48      */
49     public void verifyNamingEnumeration(NamingEnumeration enumObj,
50             int expectedCount) throws NamingException {
51         DNSTestUtils.debug("Enum is: " + enumObj);
52 
53         int count = 0;
54         Binding res;
55 
56         while (enumObj.hasMore()) {
57             res = (Binding) enumObj.next();
58             DNSTestUtils.debug(res);
59             ++count;
60         }
61 
62         if (count != expectedCount) {
63             throw new RuntimeException(
64                     "Expecting " + expectedCount + " entries but got " + count);
65         }
66     }
67 
68     public String getKey() {
69         return key;
70     }
71 
72     public void setKey(String key) {
73         this.key = key;
74     }
75 }
