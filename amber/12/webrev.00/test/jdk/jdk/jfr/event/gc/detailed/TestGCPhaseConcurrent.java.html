43 /**
44  * @test TestGCPhaseConcurrent
45  * @key jfr
46  * @library /test/lib /test/jdk /test/hotspot/jtreg
47  * @requires vm.hasJFR & vm.gc.Shenandoah
48  * @run main/othervm -XX:+UnlockExperimentalVMOptions -XX:+UseShenandoahGC -Xmx32M jdk.jfr.event.gc.detailed.TestGCPhaseConcurrent
49  */
50 public class TestGCPhaseConcurrent {
51     public static void main(String[] args) throws Exception {
52         try (Recording recording = new Recording()) {
53             // Activate the event we are interested in and start recording
54             recording.enable(EventNames.GCPhaseConcurrent);
55             recording.enable(EventNames.GCPhaseConcurrentLevel1);
56             recording.start();
57 
58             // Run GC to get concurrent phases
59             System.gc();
60 
61             recording.stop();
62 
63             // Verify recording
64             List<RecordedEvent> events = Events.fromRecording(recording);
65             Events.hasEvents(events);
66             Events.hasEvent(events, EventNames.GCPhaseConcurrent);
67             Events.hasEvent(events, EventNames.GCPhaseConcurrentLevel1);
68 
69             for (RecordedEvent event : events) {
70                 System.out.println("Event:" + event);
71                 Events.assertField(event, "name").notEmpty();
72             }
73         }
74     }
75 }
