26 package jdk.jfr.event.runtime;
27 
28 import java.lang.management.ManagementFactory;
29 import java.lang.management.RuntimeMXBean;
30 import java.nio.file.Files;
31 import java.nio.file.Paths;
32 import java.util.List;
33 import java.util.stream.Collectors;
34 
35 import jdk.jfr.Recording;
36 import jdk.jfr.consumer.RecordedEvent;
37 import jdk.test.lib.Asserts;
38 import jdk.test.lib.jfr.EventNames;
39 import jdk.test.lib.jfr.Events;
40 
41 /**
42  * @test
43  * @key jfr
44  * @requires vm.gc == "Serial" | vm.gc == null
45  * @requires vm.hasJFR
46  * @library /test/lib
47  * @run driver jdk.jfr.event.runtime.TestVMInfoEvent generateFlagsFile
48  * @run main/othervm -XX:Flags=TestVMInfoEvent.flags -Xmx500m jdk.jfr.event.runtime.TestVMInfoEvent arg1 arg2
49  */
50 public class TestVMInfoEvent {
51     private final static String EVENT_NAME = EventNames.JVMInformation;
52 
53     public static void main(String[] args) throws Exception {
54         if( (args.length > 0) && ("generateFlagsFile".equals(args[0])) ) {
55             generateFlagsFile();
56             return;
57         }
58         RuntimeMXBean mbean = ManagementFactory.getRuntimeMXBean();
59         Recording recording = new Recording();
60         recording.enable(EVENT_NAME);
61         recording.start();
62         recording.stop();
63 
64         List<RecordedEvent> events = Events.fromRecording(recording);
65         Events.hasEvents(events);
66         for (RecordedEvent event : events) {
67             System.out.println("Event:" + event);
68             Events.assertField(event, "jvmName").equal(mbean.getVmName());
69             String jvmVersion = Events.assertField(event, "jvmVersion").notEmpty().getValue();
70             if (!jvmVersion.contains(mbean.getVmVersion())) {
71                 Asserts.fail(String.format("%s does not contain %s", jvmVersion, mbean.getVmVersion()));
72             }
73 
74             String jvmArgs = Events.assertField(event, "jvmArguments").notNull().getValue();
75             String jvmFlags = Events.assertField(event, "jvmFlags").notNull().getValue();
76             Long pid = Events.assertField(event, "pid").atLeast(0L).getValue();
77             Asserts.assertEquals(pid, ProcessHandle.current().pid());
78             String eventArgs = (jvmFlags.trim() + " " + jvmArgs).trim();
79             String beanArgs = mbean.getInputArguments().stream().collect(Collectors.joining(" "));
80             Asserts.assertEquals(eventArgs, beanArgs, "Wrong inputArgs");
81 
82             final String javaCommand = mbean.getSystemProperties().get("sun.java.command");
83             Events.assertField(event, "javaArguments").equal(javaCommand);
84             Events.assertField(event, "jvmStartTime").equal(mbean.getStartTime());
85         }
86     }
87 
88     public static void generateFlagsFile() throws Exception {
89         Files.writeString(Paths.get("", "TestVMInfoEvent.flags"), "+UseSerialGC");
90     }
91 }
