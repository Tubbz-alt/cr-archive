 24 /*
 25  * @test
 26  * @bug 4652655 4857717 8025633 8026567 8071982 8164407 8182765 8205593 8240169
 27  * @summary This test verifies that class cross references work properly.
 28  * @library ../../lib
 29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
 30  * @build javadoc.tester.*
 31  * @build TestClassCrossReferences
 32  * @run main TestClassCrossReferences
 33  */
 34 
 35 import javadoc.tester.JavadocTester;
 36 
 37 public class TestClassCrossReferences extends JavadocTester {
 38 
 39     static final String uri = "http://docs.oracle.com/javase/8/docs/api/";
 40 
 41     public static void main(String... args) throws Exception {
 42         TestClassCrossReferences tester = new TestClassCrossReferences();
 43         tester.runTests();
 44     }
 45 
 46     @Test
 47     public void test() {
 48         javadoc("-d", "out",
 49                 "-source", "8",
 50                 "-Xdoclint:none",
 51                 "-sourcepath", testSrc,
 52                 "-linkoffline", uri, testSrc,
 53                 testSrc("C.java"));
 54         checkExit(Exit.OK);
 55 
 56         checkOutput("C.html", true,
 57                 "<a href=\"" + uri + "java/math/package-summary.html\" class=\"external-link\">"
 58                 + "<code>Link to math package</code></a>",
 59                 "<a href=\"" + uri + "javax/swing/text/AbstractDocument.AttributeContext.html\" "
 60                 + "title=\"class or interface in javax.swing.text\" class=\"external-link\"><code>Link to AttributeContext innerclass</code></a>",
 61                 "<a href=\"" + uri + "java/math/BigDecimal.html\" "
 62                 + "title=\"class or interface in java.math\" class=\"external-link\"><code>Link to external class BigDecimal</code></a>",
 63                 "<a href=\"" + uri + "java/math/BigInteger.html#gcd(java.math.BigInteger)\" "
 64                 + "title=\"class or interface in java.math\" class=\"external-link\"><code>Link to external member gcd</code></a>",
 65                 "<a href=\"" + uri + "javax/tools/SimpleJavaFileObject.html#uri\" "
 66                 + "title=\"class or interface in javax.tools\" class=\"external-link\"><code>Link to external member URI</code></a>",
 67                 "<dl class=\"notes\">\n"
 68                 + "<dt>Overrides:</dt>\n"
 69                 + "<dd><code>toString</code>&nbsp;in class&nbsp;<code>java.lang.Object</code></dd>\n"
 70                 + "</dl>");
 71     }
 72 
 73     @Test
 74     public void test_warning() {
 75         javadoc("-d", "out-warning",
 76                 "-Xdoclint:none",
 77                 "-sourcepath", testSrc,
 78                 "-linkoffline", uri, testSrc,
 79                 testSrc("C.java"));
 80         checkExit(Exit.OK);
 81 
 82         checkOutput(Output.OUT, true,
 83                 "The code being documented uses modules but the packages defined"
 84                 + " in http://docs.oracle.com/javase/8/docs/api/ are in the unnamed module");
 85 
 86         checkOutput("C.html", true,
 87                 "<a href=\"" + uri + "java/math/package-summary.html\" class=\"external-link\">"
 88                 + "<code>Link to math package</code></a>",
 89                 "<a href=\"" + uri + "javax/swing/text/AbstractDocument.AttributeContext.html\" "
 90                 + "title=\"class or interface in javax.swing.text\" class=\"external-link\"><code>Link to AttributeContext innerclass</code></a>",
 91                 "<a href=\"" + uri + "java/math/BigDecimal.html\" "
 92                 + "title=\"class or interface in java.math\" class=\"external-link\"><code>Link to external class BigDecimal</code></a>",
 93                 "<a href=\"" + uri + "java/math/BigInteger.html#gcd(java.math.BigInteger)\" "
 94                 + "title=\"class or interface in java.math\" class=\"external-link\"><code>Link to external member gcd</code></a>",
 95                 "<a href=\"" + uri + "javax/tools/SimpleJavaFileObject.html#uri\" "
 96                 + "title=\"class or interface in javax.tools\" class=\"external-link\"><code>Link to external member URI</code></a>",
 97                 "<dl class=\"notes\">\n"
 98                 + "<dt>Overrides:</dt>\n"
 99                 + "<dd><code>toString</code>&nbsp;in class&nbsp;<code>java.lang.Object</code></dd>\n"
100                 + "</dl>");
101     }
