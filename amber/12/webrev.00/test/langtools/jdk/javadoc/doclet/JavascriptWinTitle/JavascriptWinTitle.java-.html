24 /*
25  * @test
26  * @bug 4645058 4747738 4855054 8024756 8141492 8196202 8205593 8215599
27  * @summary  Javascript IE load error when linked by -linkoffline
28  *           Window title shouldn't change when loading left frames (javascript)
29  * @library ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build javadoc.tester.*
32  * @run main JavascriptWinTitle
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class JavascriptWinTitle extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         JavascriptWinTitle tester = new JavascriptWinTitle();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-d", "out",
47                 "-source", "8",
48                 "-doctitle", "Document Title",
49                 "-windowtitle", "Window Title",
50                 "-overview", testSrc("overview.html"),
51                 "-linkoffline", "http://java.sun.com/j2se/1.4/docs/api", testSrc,
52                 "-sourcepath", testSrc,
53                 "p1", "p2");
54         checkExit(Exit.OK);
55         checkOutput("index.html", true,
56                 "<script type=\"text/javascript\">",
57                 "<body class=\"package-index\">");
58 
59         // Test that "onload" is not present in BODY tag:
60         checkOutput("p1/package-summary.html", true, "<body class=\"package-declaration\">");
61 
62         checkOutput("p1/C.html", true, "<title>C (Window Title)</title>");
63     }
64 }
