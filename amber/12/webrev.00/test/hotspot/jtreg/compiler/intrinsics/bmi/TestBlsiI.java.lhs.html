24 /**
25  * @test

26  * @bug 8031321
27  * @summary Verify that results of computations are the same w/
28  *          and w/o usage of BLSI instruction
29  * @library /test/lib /
30  * @modules java.base/jdk.internal.misc
31  *          java.management
32  *
33  * @build sun.hotspot.WhiteBox
34  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
35  *                                sun.hotspot.WhiteBox$WhiteBoxPermission
36  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
37  *                   -XX:+WhiteBoxAPI
38  *                   compiler.intrinsics.bmi.TestBlsiI
39  */
40 
41 package compiler.intrinsics.bmi;
42 
43 import sun.hotspot.cpuinfo.CPUInfo;
44 
45 public class TestBlsiI {
46 
47     public static void main(String args[]) throws Throwable {
48         if (!CPUInfo.hasFeature("bmi1")) {
49             System.out.println("INFO: CPU does not support bmi1 feature.");
50         }
51 
52         BMITestRunner.runTests(BlsiIExpr.class, args,
53                                "-XX:+IgnoreUnrecognizedVMOptions",
54                                "-XX:+UseBMI1Instructions");
55         BMITestRunner.runTests(BlsiICommutativeExpr.class, args,
56                                "-XX:+IgnoreUnrecognizedVMOptions",
57                                "-XX:+UseBMI1Instructions");
58     }
59 
60     public static class BlsiIExpr extends Expr.BMIUnaryIntExpr {
61 
62         public int intExpr(int src) {
63             return -src & src;
64         }
65 
66         public int intExpr(Expr.MemI src) {
67             return -src.value & src.value;
68         }
69 
70     }
71 
72     public static class BlsiICommutativeExpr extends Expr.BMIUnaryIntExpr {
73 
74         public int intExpr(int src) {
75             return src & -src;
76         }
77 
78         public int intExpr(Expr.MemI src) {
79             return src.value & -src.value;
80         }
81 
82     }
83 
84 }
