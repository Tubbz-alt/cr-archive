32  * @summary Verify that direct byte buffers are correctly accessed.
33  */
34 
35 public class DirectByteBufferTest extends ByteBufferTest {
36 
37     public DirectByteBufferTest(long iterations, boolean direct) {
38         super(iterations, direct);
39     }
40 
41     public static void main(String[] args) {
42         // The number of iterations is high to ensure that tiered
43         // compilation kicks in all the way up to C2.
44         long iterations = 100000;
45         if (args.length > 0)
46             iterations = Long.parseLong(args[0]);
47 
48         new DirectByteBufferTest(iterations, true).run();
49     }
50 }
