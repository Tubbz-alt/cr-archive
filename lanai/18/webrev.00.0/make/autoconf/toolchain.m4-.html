  63 # Prepare the system so that TOOLCHAIN_CHECK_COMPILER_VERSION can be called.
  64 # Must have CC_VERSION_NUMBER and CXX_VERSION_NUMBER.
  65 # $1 - optional variable prefix for compiler and version variables (BUILD_)
  66 # $2 - optional variable prefix for comparable variable (OPENJDK_BUILD_)
  67 # $3 - optional human readable description for the type of compilers ("build " or "")
  68 AC_DEFUN([TOOLCHAIN_PREPARE_FOR_VERSION_COMPARISONS],
  69 [
  70   if test "x[$]$1CC_VERSION_NUMBER" != "x[$]$1CXX_VERSION_NUMBER"; then
  71     AC_MSG_WARN([The $3C and C++ compilers have different version numbers, [$]$1CC_VERSION_NUMBER vs [$]$1CXX_VERSION_NUMBER.])
  72     AC_MSG_WARN([This typically indicates a broken setup, and is not supported])
  73   fi
  74 
  75   # We only check CC_VERSION_NUMBER since we assume CXX_VERSION_NUMBER is equal.
  76   if [ [[ "[$]$1CC_VERSION_NUMBER" =~ (.*\.){4} ]] ]; then
  77     AC_MSG_WARN([C compiler version number has more than four parts (W.X.Y.Z): [$]$1CC_VERSION_NUMBER. Comparisons might be wrong.])
  78   fi
  79 
  80   if [ [[  "[$]$1CC_VERSION_NUMBER" =~ [0-9]{6} ]] ]; then
  81     AC_MSG_WARN([C compiler version number has a part larger than 99999: [$]$1CC_VERSION_NUMBER. Comparisons might be wrong.])
  82   fi
  83 
  84   $2COMPARABLE_ACTUAL_VERSION=`$AWK -F. '{ printf("%05d%05d%05d%05d\n", [$]1, [$]2, [$]3, [$]4) }' <<< "[$]$1CC_VERSION_NUMBER"`
  85 ])

  87 # Check if the configured compiler (C and C++) is of a specific version or
  88 # newer. TOOLCHAIN_PREPARE_FOR_VERSION_COMPARISONS must have been called before.
  89 #
  90 # Arguments:
  91 #   VERSION:   The version string to check against the found version
  92 #   IF_AT_LEAST:   block to run if the compiler is at least this version (>=)
  93 #   IF_OLDER_THAN:   block to run if the compiler is older than this version (<)
  94 #   PREFIX:   Optional variable prefix for compiler to compare version for (OPENJDK_BUILD_)
  95 BASIC_DEFUN_NAMED([TOOLCHAIN_CHECK_COMPILER_VERSION],
  96     [*VERSION PREFIX IF_AT_LEAST IF_OLDER_THAN], [$@],
  97 [
  98   # Need to assign to a variable since m4 is blocked from modifying parts in [].
  99   REFERENCE_VERSION=ARG_VERSION
 100 
 101   if [ [[ "$REFERENCE_VERSION" =~ (.*\.){4} ]] ]; then
 102     AC_MSG_ERROR([Internal error: Cannot compare to ARG_VERSION, only four parts (W.X.Y.Z) is supported])
 103   fi
 104 
 105   if [ [[ "$REFERENCE_VERSION" =~ [0-9]{6} ]] ]; then
 106     AC_MSG_ERROR([Internal error: Cannot compare to ARG_VERSION, only parts < 99999 is supported])
 107   fi
 108 
 109   # Version comparison method inspired by http://stackoverflow.com/a/24067243
 110   COMPARABLE_REFERENCE_VERSION=`$AWK -F. '{ printf("%05d%05d%05d%05d\n", [$]1, [$]2, [$]3, [$]4) }' <<< "$REFERENCE_VERSION"`
 111 
 112   if test [$]ARG_PREFIX[COMPARABLE_ACTUAL_VERSION] -ge $COMPARABLE_REFERENCE_VERSION ; then
 113     :
 114     ARG_IF_AT_LEAST
 115   else
 116     :
 117     ARG_IF_OLDER_THAN
 118   fi
 119 ])

 172 # Setup a number of variables describing how native output files are
 173 # named on this platform/toolchain.
 174 AC_DEFUN([TOOLCHAIN_SETUP_FILENAME_PATTERNS],
 175 [
 176   # Define filename patterns
 177   if test "x$OPENJDK_TARGET_OS" = xwindows; then
 178     LIBRARY_PREFIX=
 179     SHARED_LIBRARY_SUFFIX='.dll'
 180     STATIC_LIBRARY_SUFFIX='.lib'
 181     SHARED_LIBRARY='[$]1.dll'
 182     STATIC_LIBRARY='[$]1.lib'
 183     OBJ_SUFFIX='.obj'
 184   else
 185     LIBRARY_PREFIX=lib
 186     SHARED_LIBRARY_SUFFIX='.so'
 187     STATIC_LIBRARY_SUFFIX='.a'
 188     SHARED_LIBRARY='lib[$]1.so'
 189     STATIC_LIBRARY='lib[$]1.a'
 190     OBJ_SUFFIX='.o'
 191     if test "x$OPENJDK_TARGET_OS" = xmacosx; then
 192       # For full static builds, we're overloading the SHARED_LIBRARY
 193       # variables in order to limit the amount of changes required.
 194       # It would be better to remove SHARED and just use LIBRARY and
 195       # LIBRARY_SUFFIX for libraries that can be built either
 196       # shared or static and use STATIC_* for libraries that are
 197       # always built statically.
 198       if test "x$STATIC_BUILD" = xtrue; then
 199         SHARED_LIBRARY='lib[$]1.a'
 200         SHARED_LIBRARY_SUFFIX='.a'
 201       else
 202         SHARED_LIBRARY='lib[$]1.dylib'
 203         SHARED_LIBRARY_SUFFIX='.dylib'
 204       fi
 205     fi
 206   fi
 207 
 208   AC_SUBST(LIBRARY_PREFIX)
 209   AC_SUBST(SHARED_LIBRARY_SUFFIX)
 210   AC_SUBST(STATIC_LIBRARY_SUFFIX)
 211   AC_SUBST(SHARED_LIBRARY)
 212   AC_SUBST(STATIC_LIBRARY)
 213   AC_SUBST(OBJ_SUFFIX)
 214 ])

 216 # Determine which toolchain type to use, and make sure it is valid for this
 217 # platform. Setup various information about the selected toolchain.
 218 AC_DEFUN_ONCE([TOOLCHAIN_DETERMINE_TOOLCHAIN_TYPE],
 219 [
 220   AC_ARG_WITH(toolchain-type, [AS_HELP_STRING([--with-toolchain-type],
 221       [the toolchain type (or family) to use, use '--help' to show possible values @<:@platform dependent@:>@])])
 222 
 223   # Use indirect variable referencing
 224   toolchain_var_name=VALID_TOOLCHAINS_$OPENJDK_BUILD_OS
 225   VALID_TOOLCHAINS=${!toolchain_var_name}
 226 
 227   if test "x$OPENJDK_TARGET_OS" = xmacosx; then
 228     if test -n "$XCODEBUILD"; then
 229       # On Mac OS X, default toolchain to clang after Xcode 5
 230       XCODE_VERSION_OUTPUT=`"$XCODEBUILD" -version 2>&1 | $HEAD -n 1`
 231       $ECHO "$XCODE_VERSION_OUTPUT" | $GREP "Xcode " > /dev/null
 232       if test $? -ne 0; then
 233         AC_MSG_NOTICE([xcodebuild output: $XCODE_VERSION_OUTPUT])
 234         AC_MSG_ERROR([Failed to determine Xcode version.])
 235       fi
 236       XCODE_MAJOR_VERSION=`$ECHO $XCODE_VERSION_OUTPUT | \
 237           $SED -e 's/^Xcode \(@<:@1-9@:>@@<:@0-9.@:>@*\)/\1/' | \
 238           $CUT -f 1 -d .`
 239       AC_MSG_NOTICE([Xcode major version: $XCODE_MAJOR_VERSION])
 240       if test $XCODE_MAJOR_VERSION -ge 5; then
 241           DEFAULT_TOOLCHAIN="clang"
 242       else
 243           DEFAULT_TOOLCHAIN="gcc"
 244       fi
 245     else
 246       # If Xcode is not installed, but the command line tools are
 247       # then we can't run xcodebuild. On these systems we should
 248       # default to clang
 249       DEFAULT_TOOLCHAIN="clang"
 250     fi
 251   else
 252     # First toolchain type in the list is the default
 253     DEFAULT_TOOLCHAIN=${VALID_TOOLCHAINS%% *}
 254   fi
 255 
 256   if test "x$with_toolchain_type" = xlist; then
 257     # List all toolchains
 258     AC_MSG_NOTICE([The following toolchains are valid on this platform:])
 259     for toolchain in $VALID_TOOLCHAINS; do
 260       toolchain_var_name=TOOLCHAIN_DESCRIPTION_$toolchain
 261       TOOLCHAIN_DESCRIPTION=${!toolchain_var_name}
 262       $PRINTF "  %-10s  %s\n" $toolchain "$TOOLCHAIN_DESCRIPTION"
 263     done
 264 
 265     exit 0
 266   elif test "x$with_toolchain_type" != x; then
 267     # User override; check that it is valid
 268     if test "x${VALID_TOOLCHAINS/$with_toolchain_type/}" = "x${VALID_TOOLCHAINS}"; then
 269       AC_MSG_NOTICE([Toolchain type $with_toolchain_type is not valid on this platform.])
 270       AC_MSG_NOTICE([Valid toolchains: $VALID_TOOLCHAINS.])
 271       AC_MSG_ERROR([Cannot continue.])
 272     fi
 273     TOOLCHAIN_TYPE=$with_toolchain_type
 274   else
 275     # No flag given, use default
 276     TOOLCHAIN_TYPE=$DEFAULT_TOOLCHAIN
 277   fi
 278   AC_SUBST(TOOLCHAIN_TYPE)
 279 
 280   # on AIX, check for xlclang++ on the PATH and TOOLCHAIN_PATH and use it if it is available
 281   if test "x$OPENJDK_TARGET_OS" = xaix; then
 282     if test "x$TOOLCHAIN_PATH" != x; then
 283       XLC_TEST_PATH=${TOOLCHAIN_PATH}/
 284     fi
 285 
 286     XLCLANG_VERSION_OUTPUT=`${XLC_TEST_PATH}xlclang++ -qversion 2>&1 | $HEAD -n 1`
 287     $ECHO "$XLCLANG_VERSION_OUTPUT" | $GREP "IBM XL C/C++ for AIX" > /dev/null
 288     if test $? -eq 0; then
 289       AC_MSG_NOTICE([xlclang++ output: $XLCLANG_VERSION_OUTPUT])
 290     else
 291       AC_MSG_ERROR([xlclang++ version output check failed, output: $XLCLANG_VERSION_OUTPUT])
 292     fi
 293   fi
 294 
 295   TOOLCHAIN_CC_BINARY_clang="clang"
 296   TOOLCHAIN_CC_BINARY_gcc="gcc"
 297   TOOLCHAIN_CC_BINARY_microsoft="cl$EXE_SUFFIX"
 298   TOOLCHAIN_CC_BINARY_solstudio="cc"
 299   TOOLCHAIN_CC_BINARY_xlc="xlclang"
 300 
 301   TOOLCHAIN_CXX_BINARY_clang="clang++"
 302   TOOLCHAIN_CXX_BINARY_gcc="g++"
 303   TOOLCHAIN_CXX_BINARY_microsoft="cl$EXE_SUFFIX"
 304   TOOLCHAIN_CXX_BINARY_solstudio="CC"
 305   TOOLCHAIN_CXX_BINARY_xlc="xlclang++"
 306 
 307   # Use indirect variable referencing
 308   toolchain_var_name=TOOLCHAIN_DESCRIPTION_$TOOLCHAIN_TYPE
 309   TOOLCHAIN_DESCRIPTION=${!toolchain_var_name}
 310   toolchain_var_name=TOOLCHAIN_MINIMUM_VERSION_$TOOLCHAIN_TYPE
 311   TOOLCHAIN_MINIMUM_VERSION=${!toolchain_var_name}
 312   toolchain_var_name=TOOLCHAIN_MINIMUM_LD_VERSION_$TOOLCHAIN_TYPE
 313   TOOLCHAIN_MINIMUM_LD_VERSION=${!toolchain_var_name}
 314   toolchain_var_name=TOOLCHAIN_CC_BINARY_$TOOLCHAIN_TYPE
 315   TOOLCHAIN_CC_BINARY=${!toolchain_var_name}
 316   toolchain_var_name=TOOLCHAIN_CXX_BINARY_$TOOLCHAIN_TYPE
 317   TOOLCHAIN_CXX_BINARY=${!toolchain_var_name}
 318 
 319   TOOLCHAIN_SETUP_FILENAME_PATTERNS
 320 
 321   if test "x$TOOLCHAIN_TYPE" = "x$DEFAULT_TOOLCHAIN"; then
 322     AC_MSG_NOTICE([Using default toolchain $TOOLCHAIN_TYPE ($TOOLCHAIN_DESCRIPTION)])
 323   else
 324     AC_MSG_NOTICE([Using user selected toolchain $TOOLCHAIN_TYPE ($TOOLCHAIN_DESCRIPTION). Default toolchain is $DEFAULT_TOOLCHAIN.])
 325   fi
 326 ])

 328 # Before we start detecting the toolchain executables, we might need some
 329 # special setup, e.g. additional paths etc.
 330 AC_DEFUN_ONCE([TOOLCHAIN_PRE_DETECTION],
 331 [
 332   # FIXME: Is this needed?
 333   AC_LANG(C++)
 334 
 335   # Store the CFLAGS etc passed to the configure script.
 336   ORG_CFLAGS="$CFLAGS"
 337   ORG_CXXFLAGS="$CXXFLAGS"
 338 
 339   # autoconf magic only relies on PATH, so update it if tools dir is specified
 340   OLD_PATH="$PATH"
 341 
 342   # On Windows, we need to detect the visual studio installation first.
 343   # This will change the PATH, but we need to keep that new PATH even
 344   # after toolchain detection is done, since the compiler (on x86) uses
 345   # it for DLL resolution in runtime.
 346   if test "x$OPENJDK_BUILD_OS" = "xwindows" \
 347       && test "x$TOOLCHAIN_TYPE" = "xmicrosoft"; then
 348     TOOLCHAIN_SETUP_VISUAL_STUDIO_ENV
 349     if test "x$OPENJDK_BUILD_OS_ENV" = "xwindows.wsl"; then
 350       # Append VS_PATH. In WSL, VS_PATH will not contain the WSL env path needed
 351       # for using basic Unix tools, so need to keep the original PATH.
 352       BASIC_APPEND_TO_PATH(PATH, $VS_PATH)
 353       BASIC_APPEND_TO_PATH(WSLENV, "PATH/l:LIB:INCLUDE")
 354       export WSLENV
 355     else
 356       # Reset path to VS_PATH. It will include everything that was on PATH at the time we
 357       # ran TOOLCHAIN_SETUP_VISUAL_STUDIO_ENV.
 358       PATH="$VS_PATH"
 359     fi
 360     # The microsoft toolchain also requires INCLUDE and LIB to be set.
 361     export INCLUDE="$VS_INCLUDE"
 362     export LIB="$VS_LIB"
 363   else
 364     if test "x$XCODE_VERSION_OUTPUT" != x; then
 365       # For Xcode, we set the Xcode version as TOOLCHAIN_VERSION
 366       TOOLCHAIN_VERSION=`$ECHO $XCODE_VERSION_OUTPUT | $CUT -f 2 -d ' '`
 367       TOOLCHAIN_DESCRIPTION="$TOOLCHAIN_DESCRIPTION from Xcode $TOOLCHAIN_VERSION"
 368     else
 369       # Currently we do not define this for other toolchains. This might change as the need arise.
 370       TOOLCHAIN_VERSION=
 371     fi
 372   fi
 373   AC_SUBST(TOOLCHAIN_VERSION)
 374 
 375   # Finally add TOOLCHAIN_PATH at the beginning, to allow --with-tools-dir to
 376   # override all other locations.
 377   if test "x$TOOLCHAIN_PATH" != x; then
 378     PATH=$TOOLCHAIN_PATH:$PATH
 379   fi
 380 ])

 398 # Check if a compiler is of the toolchain type we expect, and save the version
 399 # information from it. If the compiler does not match the expected type,
 400 # this function will abort using AC_MSG_ERROR. If it matches, the version will
 401 # be stored in CC_VERSION_NUMBER/CXX_VERSION_NUMBER (as a dotted number), and
 402 # the full version string in CC_VERSION_STRING/CXX_VERSION_STRING.
 403 #
 404 # $1 = compiler to test (CC or CXX)
 405 # $2 = human readable name of compiler (C or C++)
 406 AC_DEFUN([TOOLCHAIN_EXTRACT_COMPILER_VERSION],
 407 [
 408   COMPILER=[$]$1
 409   COMPILER_NAME=$2
 410 
 411   if test "x$TOOLCHAIN_TYPE" = xsolstudio; then
 412     # cc -V output typically looks like
 413     #     cc: Sun C 5.12 Linux_i386 2011/11/16
 414     # or
 415     #     cc: Studio 12.5 Sun C 5.14 SunOS_sparc 2016/05/31
 416     COMPILER_VERSION_OUTPUT=`$COMPILER -V 2>&1`
 417     # Check that this is likely to be the Solaris Studio cc.
 418     $ECHO "$COMPILER_VERSION_OUTPUT" | $GREP "^.* Sun $COMPILER_NAME" > /dev/null
 419     if test $? -ne 0; then
 420       ALT_VERSION_OUTPUT=`$COMPILER --version 2>&1`
 421       AC_MSG_NOTICE([The $COMPILER_NAME compiler (located as $COMPILER) does not seem to be the required $TOOLCHAIN_TYPE compiler.])
 422       AC_MSG_NOTICE([The result from running with -V was: "$COMPILER_VERSION_OUTPUT"])
 423       AC_MSG_NOTICE([The result from running with --version was: "$ALT_VERSION_OUTPUT"])
 424       AC_MSG_ERROR([A $TOOLCHAIN_TYPE compiler is required. Try setting --with-tools-dir.])
 425     fi
 426     # Remove usage instructions (if present), and
 427     # collapse compiler output into a single line
 428     COMPILER_VERSION_STRING=`$ECHO $COMPILER_VERSION_OUTPUT | \
 429         $SED -e 's/ *@<:@Uu@:>@sage:.*//'`
 430     COMPILER_VERSION_NUMBER=`$ECHO $COMPILER_VERSION_OUTPUT | \
 431         $SED -e "s/^.*@<:@ ,\t@:>@$COMPILER_NAME@<:@ ,\t@:>@\(@<:@1-9@:>@\.@<:@0-9@:>@@<:@0-9@:>@*\).*/\1/"`
 432   elif test  "x$TOOLCHAIN_TYPE" = xxlc; then
 433     # xlc -qversion output typically looks like
 434     #     IBM XL C/C++ for AIX, V11.1 (5724-X13)
 435     #     Version: 11.01.0000.0015
 436     COMPILER_VERSION_OUTPUT=`$COMPILER -qversion 2>&1`
 437     # Check that this is likely to be the IBM XL C compiler.
 438     $ECHO "$COMPILER_VERSION_OUTPUT" | $GREP "IBM XL C" > /dev/null
 439     if test $? -ne 0; then
 440       ALT_VERSION_OUTPUT=`$COMPILER --version 2>&1`
 441       AC_MSG_NOTICE([The $COMPILER_NAME compiler (located as $COMPILER) does not seem to be the required $TOOLCHAIN_TYPE compiler.])
 442       AC_MSG_NOTICE([The result from running with -qversion was: "$COMPILER_VERSION_OUTPUT"])
 443       AC_MSG_NOTICE([The result from running with --version was: "$ALT_VERSION_OUTPUT"])
 444       AC_MSG_ERROR([A $TOOLCHAIN_TYPE compiler is required. Try setting --with-tools-dir.])
 445     fi
 446     # Collapse compiler output into a single line
 447     COMPILER_VERSION_STRING=`$ECHO $COMPILER_VERSION_OUTPUT`
 448     COMPILER_VERSION_NUMBER=`$ECHO $COMPILER_VERSION_OUTPUT | \
 449         $SED -e 's/^.*, V\(@<:@1-9@:>@@<:@0-9.@:>@*\).*$/\1/'`
 450   elif test  "x$TOOLCHAIN_TYPE" = xmicrosoft; then
 451     # There is no specific version flag, but all output starts with a version string.
 452     # First line typically looks something like:
 453     # Microsoft (R) 32-bit C/C++ Optimizing Compiler Version 16.00.40219.01 for 80x86
 454     # but the compiler name may vary depending on locale.
 455     COMPILER_VERSION_OUTPUT=`"$COMPILER" 2>&1 | $GREP -v 'ERROR.*UtilTranslatePathList' | $HEAD -n 1 | $TR -d '\r'`
 456     # Check that this is likely to be Microsoft CL.EXE.
 457     $ECHO "$COMPILER_VERSION_OUTPUT" | $GREP "Microsoft" > /dev/null
 458     if test $? -ne 0; then
 459       AC_MSG_NOTICE([The $COMPILER_NAME compiler (located as $COMPILER) does not seem to be the required $TOOLCHAIN_TYPE compiler.])
 460       AC_MSG_NOTICE([The result from running it was: "$COMPILER_VERSION_OUTPUT"])
 461       AC_MSG_ERROR([A $TOOLCHAIN_TYPE compiler is required. Try setting --with-tools-dir.])
 462     fi
 463     # Collapse compiler output into a single line
 464     COMPILER_VERSION_STRING=`$ECHO $COMPILER_VERSION_OUTPUT`
 465     COMPILER_VERSION_NUMBER=`$ECHO $COMPILER_VERSION_OUTPUT | \
 466         $SED -e 's/^.*ersion.\(@<:@1-9@:>@@<:@0-9.@:>@*\) .*$/\1/'`
 467   elif test  "x$TOOLCHAIN_TYPE" = xgcc; then
 468     # gcc --version output typically looks like
 469     #     gcc (Ubuntu/Linaro 4.8.1-10ubuntu9) 4.8.1
 470     #     Copyright (C) 2013 Free Software Foundation, Inc.
 471     #     This is free software; see the source for copying conditions.  There is NO
 472     #     warranty; not even for MERCHANTABILITY or FITNESS FOR A PARTICULAR PURPOSE.
 473     COMPILER_VERSION_OUTPUT=`$COMPILER --version 2>&1`
 474     # Check that this is likely to be GCC.
 475     $ECHO "$COMPILER_VERSION_OUTPUT" | $GREP "Free Software Foundation" > /dev/null
 476     if test $? -ne 0; then
 477       AC_MSG_NOTICE([The $COMPILER_NAME compiler (located as $COMPILER) does not seem to be the required $TOOLCHAIN_TYPE compiler.])

 518 # Try to locate the given C or C++ compiler in the path, or otherwise.
 519 #
 520 # $1 = compiler to test (CC or CXX)
 521 # $2 = human readable name of compiler (C or C++)
 522 # $3 = compiler name to search for
 523 AC_DEFUN([TOOLCHAIN_FIND_COMPILER],
 524 [
 525   COMPILER_NAME=$2
 526   SEARCH_LIST="$3"
 527 
 528   if test "x[$]$1" != x; then
 529     # User has supplied compiler name already, always let that override.
 530     AC_MSG_NOTICE([Will use user supplied compiler $1=[$]$1])
 531     if test "x`basename [$]$1`" = "x[$]$1"; then
 532       # A command without a complete path is provided, search $PATH.
 533 
 534       AC_PATH_PROGS(POTENTIAL_$1, [$]$1)
 535       if test "x$POTENTIAL_$1" != x; then
 536         $1=$POTENTIAL_$1
 537       else
 538         AC_MSG_ERROR([User supplied compiler $1=[$]$1 could not be found])
 539       fi
 540     else
 541       # Otherwise it might already be a complete path
 542       if test ! -x "[$]$1"; then
 543         AC_MSG_ERROR([User supplied compiler $1=[$]$1 does not exist])
 544       fi
 545     fi
 546   else
 547     # No user supplied value. Locate compiler ourselves.
 548 
 549     # If we are cross compiling, assume cross compilation tools follows the
 550     # cross compilation standard where they are prefixed with the autoconf
 551     # standard name for the target. For example the binary
 552     # i686-sun-solaris2.10-gcc will cross compile for i686-sun-solaris2.10.
 553     # If we are not cross compiling, then the default compiler name will be
 554     # used.
 555 
 556     $1=
 557     # If TOOLCHAIN_PATH is set, check for all compiler names in there first
 558     # before checking the rest of the PATH.
 559     # FIXME: Now that we prefix the TOOLS_DIR to the PATH in the PRE_DETECTION
 560     # step, this should not be necessary.
 561     if test -n "$TOOLCHAIN_PATH"; then
 562       PATH_save="$PATH"
 563       PATH="$TOOLCHAIN_PATH"
 564       AC_PATH_TOOL(TOOLCHAIN_PATH_$1, $SEARCH_LIST)
 565       $1=$TOOLCHAIN_PATH_$1
 566       PATH="$PATH_save"
 567     fi
 568 
 569     # AC_PATH_TOOL can't be run multiple times with the same variable,
 570     # so create a new name for this run.
 571     if test "x[$]$1" = x; then
 572       AC_PATH_TOOL(POTENTIAL_$1, $SEARCH_LIST)
 573       $1=$POTENTIAL_$1
 574     fi
 575 
 576     if test "x[$]$1" = x; then
 577       HELP_MSG_MISSING_DEPENDENCY([devkit])
 578       AC_MSG_ERROR([Could not find a $COMPILER_NAME compiler. $HELP_MSG])
 579     fi
 580   fi
 581 
 582   # Now we have a compiler binary in $1. Make sure it's okay.
 583   BASIC_FIXUP_EXECUTABLE($1)
 584   TEST_COMPILER="[$]$1"
 585 
 586   AC_MSG_CHECKING([resolved symbolic links for $1])
 587   SYMLINK_ORIGINAL="$TEST_COMPILER"
 588   BASIC_REMOVE_SYMBOLIC_LINKS(SYMLINK_ORIGINAL)
 589   if test "x$TEST_COMPILER" = "x$SYMLINK_ORIGINAL"; then
 590     AC_MSG_RESULT([no symlink])
 591   else
 592     AC_MSG_RESULT([$SYMLINK_ORIGINAL])
 593 
 594     # We can't handle ccache by gcc wrappers, since we need to know if we're
 595     # using ccache. Instead ccache usage must be controlled by a configure option.
 596     COMPILER_BASENAME=`$BASENAME "$SYMLINK_ORIGINAL"`
 597     if test "x$COMPILER_BASENAME" = "xccache"; then
 598       AC_MSG_NOTICE([Please use --enable-ccache instead of providing a wrapped compiler.])
 599       AC_MSG_ERROR([$TEST_COMPILER is a symbolic link to ccache. This is not supported.])
 600     fi
 601   fi
 602 
 603   TOOLCHAIN_EXTRACT_COMPILER_VERSION([$1], [$COMPILER_NAME])
 604 ])

 606 # Retrieve the linker version number and store it in LD_VERSION_NUMBER
 607 # (as a dotted number), and
 608 # the full version string in LD_VERSION_STRING.
 609 #
 610 # $1 = linker to test (LD or BUILD_LD)
 611 # $2 = human readable name of linker (Linker or BuildLinker)
 612 AC_DEFUN([TOOLCHAIN_EXTRACT_LD_VERSION],
 613 [
 614   LINKER=[$]$1
 615   LINKER_NAME="$2"
 616 
 617   if test "x$TOOLCHAIN_TYPE" = xsolstudio; then
 618     # cc -Wl,-V output typically looks like
 619     #   ld: Software Generation Utilities - Solaris Link Editors: 5.11-1.2329
 620 
 621     # solstudio cc requires us to have an existing file to pass as argument,
 622     # but it need not be a syntactically correct C file, so just use
 623     # ourself. :) The intermediate 'cat' is needed to stop ld from leaving
 624     # a lingering a.out (!).
 625     LINKER_VERSION_STRING=`$LD -Wl,-V $TOPDIR/configure 2>&1 | $CAT | $HEAD -n 1 | $SED -e 's/ld: //'`
 626     # Extract version number
 627     [ LINKER_VERSION_NUMBER=`$ECHO $LINKER_VERSION_STRING | \
 628         $SED -e 's/.* \([0-9][0-9]*\.[0-9][0-9]*\)-\([0-9][0-9]*\.[0-9][0-9]*\)/\1.\2/'` ]
 629   elif test  "x$TOOLCHAIN_TYPE" = xxlc; then
 630     LINKER_VERSION_STRING="Unknown"
 631     LINKER_VERSION_NUMBER="0.0"
 632   elif test  "x$TOOLCHAIN_TYPE" = xmicrosoft; then
 633     # There is no specific version flag, but all output starts with a version string.
 634     # First line typically looks something like:
 635     #   Microsoft (R) Incremental Linker Version 12.00.31101.0
 636     LINKER_VERSION_STRING=`$LD 2>&1 | $HEAD -n 1 | $TR -d '\r'`
 637     # Extract version number
 638     [ LINKER_VERSION_NUMBER=`$ECHO $LINKER_VERSION_STRING | \
 639         $SED -e 's/.* \([0-9][0-9]*\(\.[0-9][0-9]*\)*\).*/\1/'` ]
 640   elif test  "x$TOOLCHAIN_TYPE" = xgcc; then
 641     # gcc -Wl,-version output typically looks like:
 642     #   GNU ld (GNU Binutils for Ubuntu) 2.26.1
 643     #   Copyright (C) 2015 Free Software Foundation, Inc.
 644     #   This program is free software; [...]
 645     # If using gold it will look like:
 646     #   GNU gold (GNU Binutils 2.30) 1.15
 647     LINKER_VERSION_STRING=`$LD -Wl,--version 2> /dev/null | $HEAD -n 1`
 648     # Extract version number
 649     if [ [[ "$LINKER_VERSION_STRING" == *gold* ]] ]; then
 650       [ LINKER_VERSION_NUMBER=`$ECHO $LINKER_VERSION_STRING | \
 651           $SED -e 's/.* \([0-9][0-9]*\(\.[0-9][0-9]*\)*\).*) .*/\1/'` ]
 652     else
 653       [ LINKER_VERSION_NUMBER=`$ECHO $LINKER_VERSION_STRING | \
 654           $SED -e 's/.* \([0-9][0-9]*\(\.[0-9][0-9]*\)*\).*/\1/'` ]
 655     fi
 656   elif test  "x$TOOLCHAIN_TYPE" = xclang; then
 657     # clang -Wl,-v output typically looks like
 658     #   @(#)PROGRAM:ld  PROJECT:ld64-305
 659     #   configured to support archs: armv6 armv7 armv7s arm64 i386 x86_64 x86_64h armv6m armv7k armv7m armv7em (tvOS)
 660     #   Library search paths: [...]
 661     # or
 662     #   GNU ld (GNU Binutils for Ubuntu) 2.26.1
 663 
 664     LINKER_VERSION_STRING=`$LD -Wl,-v 2>&1 | $HEAD -n 1`
 665     # Check if we're using the GNU ld
 666     $ECHO "$LINKER_VERSION_STRING" | $GREP "GNU" > /dev/null
 667     if test $? -eq 0; then
 668       # Extract version number
 669       [ LINKER_VERSION_NUMBER=`$ECHO $LINKER_VERSION_STRING | \
 670           $SED -e 's/.* \([0-9][0-9]*\(\.[0-9][0-9]*\)*\).*/\1/'` ]
 671     else
 672       # Extract version number
 673       [ LINKER_VERSION_NUMBER=`$ECHO $LINKER_VERSION_STRING | \
 674           $SED -e 's/.*-\([0-9][0-9]*\)/\1/'` ]
 675     fi
 676   fi
 677 
 678   $1_VERSION_NUMBER="$LINKER_VERSION_NUMBER"
 679   $1_VERSION_STRING="$LINKER_VERSION_STRING"
 680 
 681   AC_MSG_NOTICE([Using $TOOLCHAIN_TYPE $LINKER_NAME version $LINKER_VERSION_NUMBER @<:@$LINKER_VERSION_STRING@:>@])
 682 ])
