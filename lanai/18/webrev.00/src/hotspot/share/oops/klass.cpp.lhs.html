 83 bool Klass::is_subclass_of(const Klass* k) const {
 84   // Run up the super chain and check
 85   if (this == k) return true;
 86 
 87   Klass* t = const_cast<Klass*>(this)->super();
 88 
 89   while (t != NULL) {
 90     if (t == k) return true;
 91     t = t->super();
 92   }
 93   return false;
 94 }

 96 bool Klass::search_secondary_supers(Klass* k) const {
 97   // Put some extra logic here out-of-line, before the search proper.
 98   // This cuts down the size of the inline method.
 99 
100   // This is necessary, since I am never in my own secondary_super list.
101   if (this == k)
102     return true;
103   // Scan the array-of-objects for a match
104   int cnt = secondary_supers()->length();
105   for (int i = 0; i < cnt; i++) {
106     if (secondary_supers()->at(i) == k) {
107       ((Klass*)this)->set_secondary_super_cache(k);
108       return true;
109     }
110   }
111   return false;
112 }

114 // Return self, except for abstract classes with exactly 1
115 // implementor.  Then return the 1 concrete implementation.
116 Klass *Klass::up_cast_abstract() {
117   Klass *r = this;
118   while( r->is_abstract() ) {   // Receiver is abstract?
119     Klass *s = r->subklass();   // Check for exactly 1 subklass
120     if (s == NULL || s->next_sibling() != NULL) // Oops; wrong count; give up
121       return this;              // Return 'this' as a no-progress flag
122     r = s;                    // Loop till find concrete class
123   }
124   return r;                   // Return the 1 concrete class
125 }

127 // Find LCA in class hierarchy
128 Klass *Klass::LCA( Klass *k2 ) {
129   Klass *k1 = this;
130   while( 1 ) {
131     if( k1->is_subtype_of(k2) ) return k2;
132     if( k2->is_subtype_of(k1) ) return k1;
133     k1 = k1->super();
134     k2 = k2->super();
135   }
136 }

200 jint Klass::array_layout_helper(BasicType etype) {
201   assert(etype >= T_BOOLEAN && etype <= T_OBJECT, "valid etype");
202   // Note that T_ARRAY is not allowed here.
203   int  hsize = arrayOopDesc::base_offset_in_bytes(etype);
204   int  esize = type2aelembytes(etype);
205   bool isobj = (etype == T_OBJECT);
206   int  tag   =  isobj ? _lh_array_tag_obj_value : _lh_array_tag_type_value;
207   int lh = array_layout_helper(tag, hsize, etype, exact_log2(esize));
208 
209   assert(lh < (int)_lh_neutral_value, "must look like an array layout");
210   assert(layout_helper_is_array(lh), "correct kind");
211   assert(layout_helper_is_objArray(lh) == isobj, "correct kind");
212   assert(layout_helper_is_typeArray(lh) == !isobj, "correct kind");
213   assert(layout_helper_header_size(lh) == hsize, "correct decode");
214   assert(layout_helper_element_type(lh) == etype, "correct decode");
215   assert(1 << layout_helper_log2_element_size(lh) == esize, "correct decode");
216 
217   return lh;
218 }

220 bool Klass::can_be_primary_super_slow() const {
221   if (super() == NULL)
222     return true;
223   else if (super()->super_depth() >= primary_super_limit()-1)
224     return false;
225   else
226     return true;

229 void Klass::initialize_supers(Klass* k, Array<InstanceKlass*>* transitive_interfaces, TRAPS) {
230   if (k == NULL) {
231     set_super(NULL);
232     _primary_supers[0] = this;
233     assert(super_depth() == 0, "Object must already be initialized properly");
234   } else if (k != super() || k == SystemDictionary::Object_klass()) {
235     assert(super() == NULL || super() == SystemDictionary::Object_klass(),
236            "initialize this only once to a non-trivial value");
237     set_super(k);
238     Klass* sup = k;
239     int sup_depth = sup->super_depth();
240     juint my_depth  = MIN2(sup_depth + 1, (int)primary_super_limit());
241     if (!can_be_primary_super_slow())
242       my_depth = primary_super_limit();
243     for (juint i = 0; i < my_depth; i++) {
244       _primary_supers[i] = sup->_primary_supers[i];
245     }
246     Klass* *super_check_cell;
247     if (my_depth < primary_super_limit()) {
248       _primary_supers[my_depth] = this;
249       super_check_cell = &_primary_supers[my_depth];
250     } else {
251       // Overflow of the primary_supers array forces me to be secondary.
252       super_check_cell = &_secondary_super_cache;
253     }
254     set_super_check_offset((address)super_check_cell - (address) this);
255 
256 #ifdef ASSERT
257     {
258       juint j = super_depth();
259       assert(j == my_depth, "computed accessor gets right answer");
260       Klass* t = this;
261       while (!t->can_be_primary_super()) {
262         t = t->super();
263         j = t->super_depth();
264       }
265       for (juint j1 = j+1; j1 < primary_super_limit(); j1++) {
266         assert(primary_super_of_depth(j1) == NULL, "super list padding");
267       }
268       while (t != NULL) {
269         assert(primary_super_of_depth(j) == t, "super list initialization");
270         t = t->super();
271         --j;
272       }
273       assert(j == (juint)-1, "correct depth count");
274     }
275 #endif
276   }
277 
278   if (secondary_supers() == NULL) {
279 
280     // Now compute the list of secondary supertypes.
281     // Secondaries can occasionally be on the super chain,
282     // if the inline "_primary_supers" array overflows.
283     int extras = 0;
284     Klass* p;
285     for (p = super(); !(p == NULL || p->can_be_primary_super()); p = p->super()) {
286       ++extras;
287     }
288 
289     ResourceMark rm(THREAD);  // need to reclaim GrowableArrays allocated below
290 
291     // Compute the "real" non-extra secondaries.
292     GrowableArray<Klass*>* secondaries = compute_secondary_supers(extras, transitive_interfaces);
293     if (secondaries == NULL) {
294       // secondary_supers set by compute_secondary_supers
295       return;
296     }
297 
298     GrowableArray<Klass*>* primaries = new GrowableArray<Klass*>(extras);
299 
300     for (p = super(); !(p == NULL || p->can_be_primary_super()); p = p->super()) {
301       int i;                    // Scan for overflow primaries being duplicates of 2nd'arys
302 
303       // This happens frequently for very deeply nested arrays: the
304       // primary superclass chain overflows into the secondary.  The
305       // secondary list contains the element_klass's secondaries with
306       // an extra array dimension added.  If the element_klass's
307       // secondary list already contains some primary overflows, they
308       // (with the extra level of array-ness) will collide with the
309       // normal primary superclass overflows.
310       for( i = 0; i < secondaries->length(); i++ ) {
311         if( secondaries->at(i) == p )
312           break;
313       }
314       if( i < secondaries->length() )
315         continue;               // It's a dup, don't put it in
316       primaries->push(p);
317     }
318     // Combine the two arrays into a metadata object to pack the array.
319     // The primaries are added in the reverse order, then the secondaries.
320     int new_length = primaries->length() + secondaries->length();
321     Array<Klass*>* s2 = MetadataFactory::new_array<Klass*>(
322                                        class_loader_data(), new_length, CHECK);
323     int fill_p = primaries->length();
324     for (int j = 0; j < fill_p; j++) {
325       s2->at_put(j, primaries->pop());  // add primaries in reverse order.
326     }
327     for( int j = 0; j < secondaries->length(); j++ ) {
328       s2->at_put(j+fill_p, secondaries->at(j));  // add secondaries on the end.
329     }
330 
331   #ifdef ASSERT
332       // We must not copy any NULL placeholders left over from bootstrap.
333     for (int j = 0; j < s2->length(); j++) {
334       assert(s2->at(j) != NULL, "correct bootstrapping order");
335     }
336   #endif
337 
338     set_secondary_supers(s2);
339   }
340 }

357 // subklass links.  Used by the compiler (and vtable initialization)
358 // May be cleaned concurrently, so must use the Compile_lock.
359 // The log parameter is for clean_weak_klass_links to report unlinked classes.
360 Klass* Klass::subklass(bool log) const {
361   // Need load_acquire on the _subklass, because it races with inserts that
362   // publishes freshly initialized data.
363   for (Klass* chain = Atomic::load_acquire(&_subklass);
364        chain != NULL;
365        // Do not need load_acquire on _next_sibling, because inserts never
366        // create _next_sibling edges to dead data.
367        chain = Atomic::load(&chain->_next_sibling))
368   {
369     if (chain->is_loader_alive()) {
370       return chain;
371     } else if (log) {
372       if (log_is_enabled(Trace, class, unload)) {
373         ResourceMark rm;
374         log_trace(class, unload)("unlinking class (subclass): %s", chain->external_name());
375       }
376     }
377   }
378   return NULL;
379 }
380 
381 Klass* Klass::next_sibling(bool log) const {
382   // Do not need load_acquire on _next_sibling, because inserts never
383   // create _next_sibling edges to dead data.
384   for (Klass* chain = Atomic::load(&_next_sibling);
385        chain != NULL;
386        chain = Atomic::load(&chain->_next_sibling)) {
387     // Only return alive klass, there may be stale klass
388     // in this chain if cleaned concurrently.
389     if (chain->is_loader_alive()) {
390       return chain;
391     } else if (log) {
392       if (log_is_enabled(Trace, class, unload)) {
393         ResourceMark rm;
394         log_trace(class, unload)("unlinking class (sibling): %s", chain->external_name());
395       }
396     }
397   }
398   return NULL;
399 }

406 void Klass::set_next_sibling(Klass* s) {
407   assert(s != this, "sanity check");
408   // Does not need release semantics. If used by cleanup, it will link to
409   // already safely published data, and if used by inserts, will be published
410   // safely using cmpxchg.
411   Atomic::store(&_next_sibling, s);
412 }

414 void Klass::append_to_sibling_list() {
415   assert_locked_or_safepoint(Compile_lock);


416   debug_only(verify();)
417   // add ourselves to superklass' subklass list
418   InstanceKlass* super = superklass();
419   if (super == NULL) return;        // special case: class Object
420   assert((!super->is_interface()    // interfaces cannot be supers
421           && (super->superklass() == NULL || !is_interface())),
422          "an interface can only be a subklass of Object");
423 
424   // Make sure there is no stale subklass head
425   super->clean_subklass();
426 
427   for (;;) {
428     Klass* prev_first_subklass = Atomic::load_acquire(&_super->_subklass);
429     if (prev_first_subklass != NULL) {
430       // set our sibling to be the superklass' previous first subklass
431       assert(prev_first_subklass->is_loader_alive(), "May not attach not alive klasses");
432       set_next_sibling(prev_first_subklass);
433     }
434     // Note that the prev_first_subklass is always alive, meaning no sibling_next links
435     // are ever created to not alive klasses. This is an important invariant of the lock-free
436     // cleaning protocol, that allows us to safely unlink dead klasses from the sibling list.
437     if (Atomic::cmpxchg(&super->_subklass, prev_first_subklass, this) == prev_first_subklass) {
438       return;
439     }
440   }
441   debug_only(verify();)
442 }
443 
444 void Klass::clean_subklass() {
445   for (;;) {
446     // Need load_acquire, due to contending with concurrent inserts
447     Klass* subklass = Atomic::load_acquire(&_subklass);
448     if (subklass == NULL || subklass->is_loader_alive()) {
449       return;
450     }
451     // Try to fix _subklass until it points at something not dead.
452     Atomic::cmpxchg(&_subklass, subklass, subklass->next_sibling());
453   }
454 }

456 void Klass::clean_weak_klass_links(bool unloading_occurred, bool clean_alive_klasses) {
457   if (!ClassUnloading || !unloading_occurred) {
458     return;
459   }
460 
461   Klass* root = SystemDictionary::Object_klass();
462   Stack<Klass*, mtGC> stack;
463 
464   stack.push(root);
465   while (!stack.is_empty()) {
466     Klass* current = stack.pop();
467 
468     assert(current->is_loader_alive(), "just checking, this should be live");
469 
470     // Find and set the first alive subklass
471     Klass* sub = current->subklass(true);
472     current->clean_subklass();
473     if (sub != NULL) {
474       stack.push(sub);
475     }
476 
477     // Find and set the first alive sibling
478     Klass* sibling = current->next_sibling(true);
479     current->set_next_sibling(sibling);
480     if (sibling != NULL) {
481       stack.push(sibling);
482     }
483 
484     // Clean the implementors list and method data.
485     if (clean_alive_klasses && current->is_instance_klass()) {
486       InstanceKlass* ik = InstanceKlass::cast(current);
487       ik->clean_weak_instanceklass_links();
488 
489       // JVMTI RedefineClasses creates previous versions that are not in
490       // the class hierarchy, so process them here.
491       while ((ik = ik->previous_versions()) != NULL) {
492         ik->clean_weak_instanceklass_links();
493       }
494     }
495   }
496 }

498 void Klass::metaspace_pointers_do(MetaspaceClosure* it) {
499   if (log_is_enabled(Trace, cds)) {
500     ResourceMark rm;
501     log_trace(cds)("Iter(Klass): %p (%s)", this, external_name());
502   }
503 
504   it->push(&_name);
505   it->push(&_secondary_super_cache);
506   it->push(&_secondary_supers);
507   for (int i = 0; i < _primary_super_limit; i++) {
508     it->push(&_primary_supers[i]);
509   }
510   it->push(&_super);
511   it->push((Klass**)&_subklass);
512   it->push((Klass**)&_next_sibling);
513   it->push(&_next_link);
514 
515   vtableEntry* vt = start_of_vtable();
516   for (int i=0; i<vtable_length(); i++) {
517     it->push(vt[i].method_addr());
518   }
519 }

549 void Klass::restore_unshareable_info(ClassLoaderData* loader_data, Handle protection_domain, TRAPS) {
550   assert(is_klass(), "ensure C++ vtable is restored");
551   assert(is_shared(), "must be set");
552   JFR_ONLY(RESTORE_ID(this);)
553   if (log_is_enabled(Trace, cds, unshareable)) {
554     ResourceMark rm(THREAD);
555     log_trace(cds, unshareable)("restore: %s", external_name());
556   }
557 
558   // If an exception happened during CDS restore, some of these fields may already be
559   // set.  We leave the class on the CLD list, even if incomplete so that we don't
560   // modify the CLD list outside a safepoint.
561   if (class_loader_data() == NULL) {
562     // Restore class_loader_data to the null class loader data
563     set_class_loader_data(loader_data);
564 
565     // Add to null class loader list first before creating the mirror
566     // (same order as class file parsing)
567     loader_data->add_class(this);
568   }
569 
570   Handle loader(THREAD, loader_data->class_loader());
571   ModuleEntry* module_entry = NULL;
572   Klass* k = this;
573   if (k->is_objArray_klass()) {
574     k = ObjArrayKlass::cast(k)->bottom_klass();
575   }
576   // Obtain klass' module.
577   if (k->is_instance_klass()) {
578     InstanceKlass* ik = (InstanceKlass*) k;
579     module_entry = ik->module();
580   } else {
581     module_entry = ModuleEntryTable::javabase_moduleEntry();
582   }
583   // Obtain java.lang.Module, if available
584   Handle module_handle(THREAD, ((module_entry != NULL) ? module_entry->module() : (oop)NULL));
585 
586   if (this->has_raw_archived_mirror()) {
587     ResourceMark rm(THREAD);
588     log_debug(cds, mirror)("%s has raw archived mirror", external_name());
589     if (HeapShared::open_archive_heap_region_mapped()) {
590       bool present = java_lang_Class::restore_archived_mirror(this, loader, module_handle,
591                                                               protection_domain,
592                                                               CHECK);
593       if (present) {
594         return;
595       }
596     }
597 
598     // No archived mirror data
599     log_debug(cds, mirror)("No archived mirror data for %s", external_name());
600     _java_mirror = NULL;
601     this->clear_has_raw_archived_mirror();
602   }
603 
604   // Only recreate it if not present.  A previous attempt to restore may have
605   // gotten an OOM later but keep the mirror if it was created.
606   if (java_mirror() == NULL) {
607     log_trace(cds, mirror)("Recreate mirror for %s", external_name());
608     java_lang_Class::create_mirror(this, loader, module_handle, protection_domain, CHECK);
609   }
610 }

658 void Klass::check_array_allocation_length(int length, int max_length, TRAPS) {

668     THROW_MSG(vmSymbols::java_lang_NegativeArraySizeException(), err_msg("%d", length));
669   }

672 // In product mode, this function doesn't have virtual function calls so
673 // there might be some performance advantage to handling InstanceKlass here.
674 const char* Klass::external_name() const {
675   if (is_instance_klass()) {
676     const InstanceKlass* ik = static_cast<const InstanceKlass*>(this);
677     if (ik->is_unsafe_anonymous()) {
678       char addr_buf[20];
679       jio_snprintf(addr_buf, 20, "/" INTPTR_FORMAT, p2i(ik));
680       size_t addr_len = strlen(addr_buf);
681       size_t name_len = name()->utf8_length();
682       char*  result   = NEW_RESOURCE_ARRAY(char, name_len + addr_len + 1);
683       name()->as_klass_external_name(result, (int) name_len + 1);
684       assert(strlen(result) == name_len, "");
685       strcpy(result + name_len, addr_buf);
686       assert(strlen(result) == name_len + addr_len, "");
687       return result;
688     }
689   }
690   if (name() == NULL)  return "<unknown>";
691   return name()->as_klass_external_name();
692 }

758 // Verification
759 
760 void Klass::verify_on(outputStream* st) {
761 
762   // This can be expensive, but it is worth checking that this klass is actually
763   // in the CLD graph but not in production.
764   assert(Metaspace::contains((address)this), "Should be");
765 
766   guarantee(this->is_klass(),"should be klass");
767 
768   if (super() != NULL) {
769     guarantee(super()->is_klass(), "should be klass");
770   }
771   if (secondary_super_cache() != NULL) {
772     Klass* ko = secondary_super_cache();
773     guarantee(ko->is_klass(), "should be klass");
774   }
775   for ( uint i = 0; i < primary_super_limit(); i++ ) {
776     Klass* ko = _primary_supers[i];
777     if (ko != NULL) {
778       guarantee(ko->is_klass(), "should be klass");
779     }
780   }
781 
782   if (java_mirror_no_keepalive() != NULL) {
783     guarantee(oopDesc::is_oop(java_mirror_no_keepalive()), "should be instance");
784   }
785 }

789   guarantee(obj->klass()->is_klass(), "klass field is not a klass");

792 bool Klass::is_valid(Klass* k) {
793   if (!is_aligned(k, sizeof(MetaWord))) return false;
794   if ((size_t)k < os::min_page_size()) return false;
795 
796   if (!os::is_readable_range(k, k + 1)) return false;
797   if (!Metaspace::contains(k)) return false;
798 
799   if (!Symbol::is_valid(k->name())) return false;
800   return ClassLoaderDataGraph::is_valid(k->class_loader_data());
801 }

824 // Caller needs ResourceMark
825 // joint_in_module_of_loader provides an optimization if 2 classes are in
826 // the same module to succinctly print out relevant information about their
827 // module name and class loader's name_and_id for error messages.
828 // Format:
829 //   <fully-qualified-external-class-name1> and <fully-qualified-external-class-name2>
830 //                      are in module <module-name>[@<version>]
831 //                      of loader <loader-name_and_id>[, parent loader <parent-loader-name_and_id>]
832 const char* Klass::joint_in_module_of_loader(const Klass* class2, bool include_parent_loader) const {
833   assert(module() == class2->module(), "classes do not have the same module");
834   const char* class1_name = external_name();
835   size_t len = strlen(class1_name) + 1;
836 
837   const char* class2_description = class2->class_in_module_of_loader(true, include_parent_loader);
838   len += strlen(class2_description);
839 
840   len += strlen(" and ");
841 
842   char* joint_description = NEW_RESOURCE_ARRAY_RETURN_NULL(char, len);
843 
844   // Just return the FQN if error when allocating string
845   if (joint_description == NULL) {
846     return class1_name;
847   }
848 
849   jio_snprintf(joint_description, len, "%s and %s",
850                class1_name,
851                class2_description);
852 
853   return joint_description;
854 }

856 // Caller needs ResourceMark
857 // class_in_module_of_loader provides a standard way to include
858 // relevant information about a class, such as its module name as
859 // well as its class loader's name_and_id, in error messages and logging.
860 // Format:
861 //   <fully-qualified-external-class-name> is in module <module-name>[@<version>]
862 //                                         of loader <loader-name_and_id>[, parent loader <parent-loader-name_and_id>]
863 const char* Klass::class_in_module_of_loader(bool use_are, bool include_parent_loader) const {
864   // 1. fully qualified external name of class
865   const char* klass_name = external_name();
866   size_t len = strlen(klass_name) + 1;
867 
868   // 2. module name + @version
869   const char* module_name = "";
870   const char* version = "";
871   bool has_version = false;
872   bool module_is_named = false;
873   const char* module_name_phrase = "";
874   const Klass* bottom_klass = is_objArray_klass() ?
875                                 ObjArrayKlass::cast(this)->bottom_klass() : this;
876   if (bottom_klass->is_instance_klass()) {
877     ModuleEntry* module = InstanceKlass::cast(bottom_klass)->module();
878     if (module->is_named()) {
879       module_is_named = true;
880       module_name_phrase = "module ";
881       module_name = module->name()->as_C_string();
882       len += strlen(module_name);
883       // Use version if exists and is not a jdk module
884       if (module->should_show_version()) {
885         has_version = true;
886         version = module->version()->as_C_string();
887         // Include stlen(version) + 1 for the "@"
888         len += strlen(version) + 1;
889       }
890     } else {
891       module_name = UNNAMED_MODULE;
892       len += UNNAMED_MODULE_LEN;
893     }
894   } else {
895     // klass is an array of primitives, module is java.base
896     module_is_named = true;
897     module_name_phrase = "module ";
898     module_name = JAVA_BASE_NAME;
899     len += JAVA_BASE_NAME_LEN;
900   }
901 
902   // 3. class loader's name_and_id
903   ClassLoaderData* cld = class_loader_data();
904   assert(cld != NULL, "class_loader_data should not be null");
905   const char* loader_name_and_id = cld->loader_name_and_id();
906   len += strlen(loader_name_and_id);
907 
908   // 4. include parent loader information
909   const char* parent_loader_phrase = "";
910   const char* parent_loader_name_and_id = "";
911   if (include_parent_loader &&
912       !cld->is_builtin_class_loader_data()) {
913     oop parent_loader = java_lang_ClassLoader::parent(class_loader());
914     ClassLoaderData *parent_cld = ClassLoaderData::class_loader_data_or_null(parent_loader);
915     // The parent loader's ClassLoaderData could be null if it is
916     // a delegating class loader that has never defined a class.
917     // In this case the loader's name must be obtained via the parent loader's oop.
918     if (parent_cld == NULL) {
919       oop cl_name_and_id = java_lang_ClassLoader::nameAndId(parent_loader);
920       if (cl_name_and_id != NULL) {
921         parent_loader_name_and_id = java_lang_String::as_utf8_string(cl_name_and_id);
922       }
923     } else {
924       parent_loader_name_and_id = parent_cld->loader_name_and_id();
925     }
926     parent_loader_phrase = ", parent loader ";
927     len += strlen(parent_loader_phrase) + strlen(parent_loader_name_and_id);
928   }
929 
930   // Start to construct final full class description string
931   len += ((use_are) ? strlen(" are in ") : strlen(" is in "));
932   len += strlen(module_name_phrase) + strlen(" of loader ");
933 
934   char* class_description = NEW_RESOURCE_ARRAY_RETURN_NULL(char, len);
935 
936   // Just return the FQN if error when allocating string
937   if (class_description == NULL) {
938     return klass_name;
939   }
940 
941   jio_snprintf(class_description, len, "%s %s in %s%s%s%s of loader %s%s%s",
942                klass_name,
943                (use_are) ? "are" : "is",
944                module_name_phrase,
945                module_name,
946                (has_version) ? "@" : "",
947                (has_version) ? version : "",
948                loader_name_and_id,
949                parent_loader_phrase,
950                parent_loader_name_and_id);
951 
952   return class_description;
953 }
