  65 // Set a breakpoint here to identify where a particular node index is built.
  66 void Node::verify_construction() {
  67   _debug_orig = NULL;
  68   int old_debug_idx = Compile::debug_idx();
  69   int new_debug_idx = old_debug_idx+1;
  70   if (new_debug_idx > 0) {
  71     // Arrange that the lowest five decimal digits of _debug_idx
  72     // will repeat those of _idx. In case this is somehow pathological,
  73     // we continue to assign negative numbers (!) consecutively.
  74     const int mod = 100000;
  75     int bump = (int)(_idx - new_debug_idx) % mod;
  76     if (bump < 0)  bump += mod;
  77     assert(bump >= 0 && bump < mod, "");
  78     new_debug_idx += bump;
  79   }
  80   Compile::set_debug_idx(new_debug_idx);
  81   set_debug_idx( new_debug_idx );
  82   assert(Compile::current()->unique() < (INT_MAX - 1), "Node limit exceeded INT_MAX");
  83   assert(Compile::current()->live_nodes() < Compile::current()->max_node_limit(), "Live Node limit exceeded limit");
  84   if (BreakAtNode != 0 && (_debug_idx == BreakAtNode || (int)_idx == BreakAtNode)) {
  85     tty->print_cr("BreakAtNode: _idx=%d _debug_idx=%d", _idx, _debug_idx);
  86     BREAKPOINT;
  87   }
  88 #if OPTO_DU_ITERATOR_ASSERT
  89   _last_del = NULL;
  90   _del_tick = 0;
  91 #endif
  92   _hash_lock = 0;
  93 }

  99 void DUIterator_Common::sample(const Node* node) {
 100   _vdui     = VerifyDUIterators;
 101   _node     = node;
 102   _outcnt   = node->_outcnt;
 103   _del_tick = node->_del_tick;
 104   _last     = NULL;
 105 }
 106 
 107 void DUIterator_Common::verify(const Node* node, bool at_end_ok) {
 108   assert(_node     == node, "consistent iterator source");
 109   assert(_del_tick == node->_del_tick, "no unexpected deletions allowed");
 110 }

 112 void DUIterator_Common::verify_resync() {
 113   // Ensure that the loop body has just deleted the last guy produced.
 114   const Node* node = _node;
 115   // Ensure that at least one copy of the last-seen edge was deleted.
 116   // Note:  It is OK to delete multiple copies of the last-seen edge.
 117   // Unfortunately, we have no way to verify that all the deletions delete
 118   // that same edge.  On this point we must use the Honor System.
 119   assert(node->_del_tick >= _del_tick+1, "must have deleted an edge");
 120   assert(node->_last_del == _last, "must have deleted the edge just produced");
 121   // We liked this deletion, so accept the resulting outcnt and tick.
 122   _outcnt   = node->_outcnt;
 123   _del_tick = node->_del_tick;
 124 }

 126 void DUIterator_Common::reset(const DUIterator_Common& that) {
 127   if (this == &that)  return;  // ignore assignment to self
 128   if (!_vdui) {
 129     // We need to initialize everything, overwriting garbage values.
 130     _last = that._last;
 131     _vdui = that._vdui;
 132   }
 133   // Note:  It is legal (though odd) for an iterator over some node x
 134   // to be reassigned to iterate over another node y.  Some doubly-nested
 135   // progress loops depend on being able to do this.
 136   const Node* node = that._node;
 137   // Re-initialize everything, except _last.
 138   _node     = node;
 139   _outcnt   = node->_outcnt;
 140   _del_tick = node->_del_tick;
 141 }
 142 
 143 void DUIterator::sample(const Node* node) {
 144   DUIterator_Common::sample(node);      // Initialize the assertion data.
 145   _refresh_tick = 0;                    // No refreshes have happened, as yet.
 146 }
 147 
 148 void DUIterator::verify(const Node* node, bool at_end_ok) {
 149   DUIterator_Common::verify(node, at_end_ok);
 150   assert(_idx      <  node->_outcnt + (uint)at_end_ok, "idx in range");
 151 }
 152 
 153 void DUIterator::verify_increment() {
 154   if (_refresh_tick & 1) {
 155     // We have refreshed the index during this loop.
 156     // Fix up _idx to meet asserts.
 157     if (_idx > _outcnt)  _idx = _outcnt;
 158   }
 159   verify(_node, true);
 160 }
 161 
 162 void DUIterator::verify_resync() {
 163   // Note:  We do not assert on _outcnt, because insertions are OK here.
 164   DUIterator_Common::verify_resync();
 165   // Make sure we are still in sync, possibly with no more out-edges:
 166   verify(_node, true);
 167 }

 169 void DUIterator::reset(const DUIterator& that) {
 170   if (this == &that)  return;  // self assignment is always a no-op
 171   assert(that._refresh_tick == 0, "assign only the result of Node::outs()");
 172   assert(that._idx          == 0, "assign only the result of Node::outs()");
 173   assert(_idx               == that._idx, "already assigned _idx");
 174   if (!_vdui) {
 175     // We need to initialize everything, overwriting garbage values.
 176     sample(that._node);
 177   } else {
 178     DUIterator_Common::reset(that);
 179     if (_refresh_tick & 1) {
 180       _refresh_tick++;                  // Clear the "was refreshed" flag.
 181     }
 182     assert(_refresh_tick < 2*100000, "DU iteration must converge quickly");
 183   }
 184 }
 185 
 186 void DUIterator::refresh() {
 187   DUIterator_Common::sample(_node);     // Re-fetch assertion data.
 188   _refresh_tick |= 1;                   // Set the "was refreshed" flag.
 189 }
 190 
 191 void DUIterator::verify_finish() {
 192   // If the loop has killed the node, do not require it to re-run.
 193   if (_node->_outcnt == 0)  _refresh_tick &= ~1;
 194   // If this assert triggers, it means that a loop used refresh_out_pos
 195   // to re-synch an iteration index, but the loop did not correctly
 196   // re-run itself, using a "while (progress)" construct.
 197   // This iterator enforces the rule that you must keep trying the loop
 198   // until it "runs clean" without any need for refreshing.
 199   assert(!(_refresh_tick & 1), "the loop must run once with no refreshing");
 200 }

 203 void DUIterator_Fast::verify(const Node* node, bool at_end_ok) {
 204   DUIterator_Common::verify(node, at_end_ok);
 205   Node** out    = node->_out;
 206   uint   cnt    = node->_outcnt;
 207   assert(cnt == _outcnt, "no insertions allowed");
 208   assert(_outp >= out && _outp <= out + cnt - !at_end_ok, "outp in range");
 209   // This last check is carefully designed to work for NO_OUT_ARRAY.
 210 }
 211 
 212 void DUIterator_Fast::verify_limit() {
 213   const Node* node = _node;
 214   verify(node, true);
 215   assert(_outp == node->_out + node->_outcnt, "limit still correct");
 216 }
 217 
 218 void DUIterator_Fast::verify_resync() {
 219   const Node* node = _node;
 220   if (_outp == node->_out + _outcnt) {
 221     // Note that the limit imax, not the pointer i, gets updated with the
 222     // exact count of deletions.  (For the pointer it's always "--i".)
 223     assert(node->_outcnt+node->_del_tick == _outcnt+_del_tick, "no insertions allowed with deletion(s)");
 224     // This is a limit pointer, with a name like "imax".
 225     // Fudge the _last field so that the common assert will be happy.
 226     _last = (Node*) node->_last_del;
 227     DUIterator_Common::verify_resync();
 228   } else {
 229     assert(node->_outcnt < _outcnt, "no insertions allowed with deletion(s)");
 230     // A normal internal pointer.
 231     DUIterator_Common::verify_resync();
 232     // Make sure we are still in sync, possibly with no more out-edges:
 233     verify(node, true);
 234   }
 235 }

 237 void DUIterator_Fast::verify_relimit(uint n) {
 238   const Node* node = _node;
 239   assert((int)n > 0, "use imax -= n only with a positive count");
 240   // This must be a limit pointer, with a name like "imax".
 241   assert(_outp == node->_out + node->_outcnt, "apply -= only to a limit (imax)");
 242   // The reported number of deletions must match what the node saw.
 243   assert(node->_del_tick == _del_tick + n, "must have deleted n edges");
 244   // Fudge the _last field so that the common assert will be happy.
 245   _last = (Node*) node->_last_del;
 246   DUIterator_Common::verify_resync();
 247 }

 249 void DUIterator_Fast::reset(const DUIterator_Fast& that) {
 250   assert(_outp              == that._outp, "already assigned _outp");
 251   DUIterator_Common::reset(that);

 254 void DUIterator_Last::verify(const Node* node, bool at_end_ok) {
 255   // at_end_ok means the _outp is allowed to underflow by 1
 256   _outp += at_end_ok;
 257   DUIterator_Fast::verify(node, at_end_ok);  // check _del_tick, etc.
 258   _outp -= at_end_ok;
 259   assert(_outp == (node->_out + node->_outcnt) - 1, "pointer must point to end of nodes");
 260 }

 268 void DUIterator_Last::verify_step(uint num_edges) {
 269   assert((int)num_edges > 0, "need non-zero edge count for loop progress");
 270   _outcnt   -= num_edges;
 271   _del_tick += num_edges;
 272   // Make sure we are still in sync, possibly with no more out-edges:
 273   const Node* node = _node;
 274   verify(node, true);
 275   assert(node->_last_del == _last, "must have deleted the edge just produced");
 276 }

 294 // Shared initialization code.
 295 inline int Node::Init(int req) {
 296   Compile* C = Compile::current();
 297   int idx = C->next_unique();
 298 
 299   // Allocate memory for the necessary number of edges.
 300   if (req > 0) {
 301     // Allocate space for _in array to have double alignment.
 302     _in = (Node **) ((char *) (C->node_arena()->Amalloc_D(req * sizeof(void*))));
 303   }
 304   // If there are default notes floating around, capture them:
 305   Node_Notes* nn = C->default_node_notes();
 306   if (nn != NULL)  init_node_notes(C, idx, nn);
 307 
 308   // Note:  At this point, C is dead,
 309   // and we begin to initialize the new Node.
 310 
 311   _cnt = _max = req;
 312   _outcnt = _outmax = 0;
 313   _class_id = Class_Node;
 314   _flags = 0;
 315   _out = NO_OUT_ARRAY;
 316   return idx;
 317 }

 480 //------------------------------clone------------------------------------------
 481 // Clone a Node.
 482 Node *Node::clone() const {
 483   Compile* C = Compile::current();
 484   uint s = size_of();           // Size of inherited Node
 485   Node *n = (Node*)C->node_arena()->Amalloc_D(size_of() + _max*sizeof(Node*));
 486   Copy::conjoint_words_to_lower((HeapWord*)this, (HeapWord*)n, s);
 487   // Set the new input pointer array
 488   n->_in = (Node**)(((char*)n)+s);
 489   // Cannot share the old output pointer array, so kill it
 490   n->_out = NO_OUT_ARRAY;
 491   // And reset the counters to 0
 492   n->_outcnt = 0;
 493   n->_outmax = 0;
 494   // Unlock this guy, since he is not in any hash table.
 495   debug_only(n->_hash_lock = 0);
 496   // Walk the old node's input list to duplicate its edges
 497   uint i;
 498   for( i = 0; i < len(); i++ ) {
 499     Node *x = in(i);
 500     n->_in[i] = x;
 501     if (x != NULL) x->add_out(n);
 502   }
 503   if (is_macro())
 504     C->add_macro_node(n);
 505   if (is_expensive())
 506     C->add_expensive_node(n);
 507   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 508   bs->register_potential_barrier_node(n);
 509   // If the cloned node is a range check dependent CastII, add it to the list.
 510   CastIINode* cast = n->isa_CastII();
 511   if (cast != NULL && cast->has_range_check()) {
 512     C->add_range_check_cast(cast);
 513   }
 514   if (n->Opcode() == Op_Opaque4) {
 515     C->add_opaque4_node(n);
 516   }
 517 
 518   n->set_idx(C->next_unique()); // Get new unique index as well
 519   debug_only( n->verify_construction() );
 520   NOT_PRODUCT(nodes_created++);
 521   // Do not patch over the debug_idx of a clone, because it makes it
 522   // impossible to break on the clone's moment of creation.
 523   //debug_only( n->set_debug_idx( debug_idx() ) );
 524 
 525   C->copy_node_notes_to(n, (Node*) this);
 526 
 527   // MachNode clone
 528   uint nopnds;
 529   if (this->is_Mach() && (nopnds = this->as_Mach()->num_opnds()) > 0) {
 530     MachNode *mach  = n->as_Mach();
 531     MachNode *mthis = this->as_Mach();
 532     // Get address of _opnd_array.
 533     // It should be the same offset since it is the clone of this node.
 534     MachOper **from = mthis->_opnds;
 535     MachOper **to = (MachOper **)((size_t)(&mach->_opnds) +
 536                     pointer_delta((const void*)from,
 537                                   (const void*)(&mthis->_opnds), 1));
 538     mach->_opnds = to;
 539     for ( uint i = 0; i < nopnds; ++i ) {
 540       to[i] = from[i]->clone();
 541     }
 542   }
 543   // cloning CallNode may need to clone JVMState
 544   if (n->is_Call()) {
 545     n->as_Call()->clone_jvms(C);
 546   }
 547   if (n->is_SafePoint()) {
 548     n->as_SafePoint()->clone_replaced_nodes();
 549   }
 550   return n;                     // Return the clone
 551 }

 553 //---------------------------setup_is_top--------------------------------------
 554 // Call this when changing the top node, to reassert the invariants
 555 // required by Node::is_top.  See Compile::set_cached_top_node.
 556 void Node::setup_is_top() {
 557   if (this == (Node*)Compile::current()->top()) {
 558     // This node has just become top.  Kill its out array.
 559     _outcnt = _outmax = 0;
 560     _out = NULL;                           // marker value for top
 561     assert(is_top(), "must be top");
 562   } else {
 563     if (_out == NULL)  _out = NO_OUT_ARRAY;
 564     assert(!is_top(), "must not be top");

 568 //------------------------------~Node------------------------------------------
 569 // Fancy destructor; eagerly attempt to reclaim Node numberings and storage
 570 void Node::destruct() {
 571   // Eagerly reclaim unique Node numberings
 572   Compile* compile = Compile::current();
 573   if ((uint)_idx+1 == compile->unique()) {
 574     compile->set_unique(compile->unique()-1);
 575   }
 576   // Clear debug info:
 577   Node_Notes* nn = compile->node_notes_at(_idx);
 578   if (nn != NULL)  nn->clear();
 579   // Walk the input array, freeing the corresponding output edges
 580   _cnt = _max;  // forget req/prec distinction
 581   uint i;
 582   for( i = 0; i < _max; i++ ) {
 583     set_req(i, NULL);
 584     //assert(def->out(def->outcnt()-1) == (Node *)this,"bad def-use hacking in reclaim");
 585   }
 586   assert(outcnt() == 0, "deleting a node must not leave a dangling use");
 587   // See if the input array was allocated just prior to the object
 588   int edge_size = _max*sizeof(void*);
 589   int out_edge_size = _outmax*sizeof(void*);
 590   char *edge_end = ((char*)_in) + edge_size;
 591   char *out_array = (char*)(_out == NO_OUT_ARRAY? NULL: _out);
 592   int node_size = size_of();
 593 
 594   // Free the output edge array
 595   if (out_edge_size > 0) {
 596     compile->node_arena()->Afree(out_array, out_edge_size);
 597   }
 598 
 599   // Free the input edge array and the node itself
 600   if( edge_end == (char*)this ) {
 601     // It was; free the input array and object all in one hit
 602 #ifndef ASSERT
 603     compile->node_arena()->Afree(_in,edge_size+node_size);
 604 #endif
 605   } else {
 606     // Free just the input array
 607     compile->node_arena()->Afree(_in,edge_size);
 608 
 609     // Free just the object
 610 #ifndef ASSERT
 611     compile->node_arena()->Afree(this,node_size);
 612 #endif
 613   }
 614   if (is_macro()) {
 615     compile->remove_macro_node(this);
 616   }
 617   if (is_expensive()) {
 618     compile->remove_expensive_node(this);
 619   }
 620   CastIINode* cast = isa_CastII();
 621   if (cast != NULL && cast->has_range_check()) {
 622     compile->remove_range_check_cast(cast);
 623   }
 624   if (Opcode() == Op_Opaque4) {
 625     compile->remove_opaque4_node(this);
 626   }
 627 
 628   if (is_SafePoint()) {
 629     as_SafePoint()->delete_replaced_nodes();
 630   }
 631   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 632   bs->unregister_potential_barrier_node(this);
 633 #ifdef ASSERT
 634   // We will not actually delete the storage, but we'll make the node unusable.
 635   *(address*)this = badAddress;  // smash the C++ vtbl, probably
 636   _in = _out = (Node**) badAddress;
 637   _max = _cnt = _outmax = _outcnt = 0;
 638   compile->remove_modified_node(this);
 639 #endif
 640 }

 642 //------------------------------grow-------------------------------------------
 643 // Grow the input array, making space for more edges
 644 void Node::grow( uint len ) {
 645   Arena* arena = Compile::current()->node_arena();
 646   uint new_max = _max;
 647   if( new_max == 0 ) {
 648     _max = 4;
 649     _in = (Node**)arena->Amalloc(4*sizeof(Node*));
 650     Node** to = _in;
 651     to[0] = NULL;
 652     to[1] = NULL;
 653     to[2] = NULL;
 654     to[3] = NULL;
 655     return;
 656   }
 657   new_max = next_power_of_2(len);
 658   // Trimming to limit allows a uint8 to handle up to 255 edges.
 659   // Previously I was using only powers-of-2 which peaked at 128 edges.
 660   //if( new_max >= limit ) new_max = limit-1;
 661   _in = (Node**)arena->Arealloc(_in, _max*sizeof(Node*), new_max*sizeof(Node*));
 662   Copy::zero_to_bytes(&_in[_max], (new_max-_max)*sizeof(Node*)); // NULL all new space
 663   _max = new_max;               // Record new max length
 664   // This assertion makes sure that Node::_max is wide enough to
 665   // represent the numerical value of new_max.
 666   assert(_max == new_max && _max > len, "int width of _max is too small");
 667 }

 731 //------------------------------add_req----------------------------------------
 732 // Add a new required input at the end
 733 void Node::add_req( Node *n ) {
 734   assert( is_not_dead(n), "can not use dead node");
 735 
 736   // Look to see if I can move precedence down one without reallocating
 737   if( (_cnt >= _max) || (in(_max-1) != NULL) )
 738     grow( _max+1 );
 739 
 740   // Find a precedence edge to move
 741   if( in(_cnt) != NULL ) {       // Next precedence edge is busy?
 742     uint i;
 743     for( i=_cnt; i<_max; i++ )
 744       if( in(i) == NULL )       // Find the NULL at end of prec edge list
 745         break;                  // There must be one, since we grew the array
 746     _in[i] = in(_cnt);          // Move prec over, making space for req edge
 747   }
 748   _in[_cnt++] = n;            // Stuff over old prec edge
 749   if (n != NULL) n->add_out((Node *)this);
 750 }

 752 //---------------------------add_req_batch-------------------------------------
 753 // Add a new required input at the end
 754 void Node::add_req_batch( Node *n, uint m ) {
 755   assert( is_not_dead(n), "can not use dead node");
 756   // check various edge cases
 757   if ((int)m <= 1) {
 758     assert((int)m >= 0, "oob");
 759     if (m != 0)  add_req(n);
 760     return;
 761   }
 762 
 763   // Look to see if I can move precedence down one without reallocating
 764   if( (_cnt+m) > _max || _in[_max-m] )
 765     grow( _max+m );
 766 
 767   // Find a precedence edge to move
 768   if( _in[_cnt] != NULL ) {     // Next precedence edge is busy?
 769     uint i;
 770     for( i=_cnt; i<_max; i++ )
 771       if( _in[i] == NULL )      // Find the NULL at end of prec edge list
 772         break;                  // There must be one, since we grew the array
 773     // Slide all the precs over by m positions (assume #prec << m).
 774     Copy::conjoint_words_to_higher((HeapWord*)&_in[_cnt], (HeapWord*)&_in[_cnt+m], ((i-_cnt)*sizeof(Node*)));
 775   }
 776 
 777   // Stuff over the old prec edges
 778   for(uint i=0; i<m; i++ ) {
 779     _in[_cnt++] = n;
 780   }
 781 
 782   // Insert multiple out edges on the node.
 783   if (n != NULL && !n->is_top()) {
 784     for(uint i=0; i<m; i++ ) {
 785       n->add_out((Node *)this);
 786     }
 787   }
 788 }

 791 // Delete the required edge and compact the edge array

 797   Node *n = in(idx);

 845 int Node::replace_edge(Node* old, Node* neww) {
 846   if (old == neww)  return 0;  // nothing to do
 847   uint nrep = 0;
 848   for (uint i = 0; i < len(); i++) {
 849     if (in(i) == old) {

 853         assert(find_prec_edge(neww) == -1, "spec violation: duplicated prec edge (node %d -> %d)", _idx, neww->_idx);
 854         set_prec(i, neww);
 855       }
 856       nrep++;
 857     }

 877 //-------------------------disconnect_inputs-----------------------------------
 878 // NULL out all inputs to eliminate incoming Def-Use edges.
 879 // Return the number of edges between 'n' and 'this'
 880 int Node::disconnect_inputs(Node *n, Compile* C) {
 881   int edges_to_n = 0;
 882 
 883   uint cnt = req();
 884   for( uint i = 0; i < cnt; ++i ) {
 885     if( in(i) == 0 ) continue;
 886     if( in(i) == n ) ++edges_to_n;
 887     set_req(i, NULL);
 888   }
 889   // Remove precedence edges if any exist
 890   // Note: Safepoints may have precedence edges, even during parsing
 891   if( (req() != len()) && (in(req()) != NULL) ) {
 892     uint max = len();
 893     for( uint i = 0; i < max; ++i ) {
 894       if( in(i) == 0 ) continue;
 895       if( in(i) == n ) ++edges_to_n;
 896       set_prec(i, NULL);
 897     }
 898   }
 899 
 900   // Node::destruct requires all out edges be deleted first
 901   // debug_only(destruct();)   // no reuse benefit expected
 902   if (edges_to_n == 0) {
 903     C->record_dead_node(_idx);
 904   }
 905   return edges_to_n;
 906 }

 980 //------------------------------add_prec---------------------------------------
 981 // Add a new precedence input.  Precedence inputs are unordered, with
 982 // duplicates removed and NULLs packed down at the end.
 983 void Node::add_prec( Node *n ) {
 984   assert( is_not_dead(n), "can not use dead node");
 985 
 986   // Check for NULL at end
 987   if( _cnt >= _max || in(_max-1) )
 988     grow( _max+1 );

 990   // Find a precedence edge to move
 991   uint i = _cnt;
 992   while( in(i) != NULL ) {
 993     if (in(i) == n) return; // Avoid spec violation: duplicated prec edge.
 994     i++;
 995   }
 996   _in[i] = n;                                // Stuff prec edge over NULL
 997   if ( n != NULL) n->add_out((Node *)this);  // Add mirror edge
 998 
 999 #ifdef ASSERT
1000   while ((++i)<_max) { assert(_in[i] == NULL, "spec violation: Gap in prec edges (node %d)", _idx); }
1001 #endif
1002 }

1089 //------------------------------Ideal------------------------------------------
1090 //
1091 // 'Idealize' the graph rooted at this Node.
1092 //
1093 // In order to be efficient and flexible there are some subtle invariants
1094 // these Ideal calls need to hold.  Running with '+VerifyIterativeGVN' checks
1095 // these invariants, although its too slow to have on by default.  If you are
1096 // hacking an Ideal call, be sure to test with +VerifyIterativeGVN!
1097 //
1098 // The Ideal call almost arbitrarily reshape the graph rooted at the 'this'
1099 // pointer.  If ANY change is made, it must return the root of the reshaped
1100 // graph - even if the root is the same Node.  Example: swapping the inputs
1101 // to an AddINode gives the same answer and same root, but you still have to
1102 // return the 'this' pointer instead of NULL.
1103 //
1104 // You cannot return an OLD Node, except for the 'this' pointer.  Use the
1105 // Identity call to return an old Node; basically if Identity can find
1106 // another Node have the Ideal call make no change and return NULL.
1107 // Example: AddINode::Ideal must check for add of zero; in this case it
1108 // returns NULL instead of doing any graph reshaping.
1109 //
1110 // You cannot modify any old Nodes except for the 'this' pointer.  Due to
1111 // sharing there may be other users of the old Nodes relying on their current
1112 // semantics.  Modifying them will break the other users.
1113 // Example: when reshape "(X+3)+4" into "X+7" you must leave the Node for
1114 // "X+3" unchanged in case it is shared.
1115 //
1116 // If you modify the 'this' pointer's inputs, you should use
1117 // 'set_req'.  If you are making a new Node (either as the new root or
1118 // some new internal piece) you may use 'init_req' to set the initial
1119 // value.  You can make a new Node with either 'new' or 'clone'.  In
1120 // either case, def-use info is correctly maintained.
1121 //
1122 // Example: reshape "(X+3)+4" into "X+7":
1123 //    set_req(1, in(1)->in(1));
1124 //    set_req(2, phase->intcon(7));
1125 //    return this;
1126 // Example: reshape "X*4" into "X<<2"
1127 //    return new LShiftINode(in(1), phase->intcon(2));
1128 //
1129 // You must call 'phase->transform(X)' on any new Nodes X you make, except
1130 // for the returned root node.  Example: reshape "X*31" with "(X<<5)-X".
1131 //    Node *shift=phase->transform(new LShiftINode(in(1),phase->intcon(5)));
1132 //    return new AddINode(shift, in(1));
1133 //
1134 // When making a Node for a constant use 'phase->makecon' or 'phase->intcon'.
1135 // These forms are faster than 'phase->transform(new ConNode())' and Do
1136 // The Right Thing with def-use info.
1137 //
1138 // You cannot bury the 'this' Node inside of a graph reshape.  If the reshaped
1139 // graph uses the 'this' Node it must be the root.  If you want a Node with
1140 // the same Opcode as the 'this' pointer use 'clone'.
1141 //
1142 Node *Node::Ideal(PhaseGVN *phase, bool can_reshape) {
1143   return NULL;                  // Default to being Ideal already
1144 }

1146 // Some nodes have specific Ideal subgraph transformations only if they are
1147 // unique users of specific nodes. Such nodes should be put on IGVN worklist
1148 // for the transformations to happen.
1149 bool Node::has_special_unique_user() const {
1150   assert(outcnt() == 1, "match only for unique out");
1151   Node* n = unique_out();
1152   int op  = Opcode();
1153   if (this->is_Store()) {
1154     // Condition for back-to-back stores folding.
1155     return n->Opcode() == op && n->in(MemNode::Memory) == this;
1156   } else if (this->is_Load() || this->is_DecodeN() || this->is_Phi()) {
1157     // Condition for removing an unused LoadNode or DecodeNNode from the MemBarAcquire precedence input
1158     return n->Opcode() == Op_MemBarAcquire;
1159   } else if (op == Op_AddL) {
1160     // Condition for convL2I(addL(x,y)) ==> addI(convL2I(x),convL2I(y))
1161     return n->Opcode() == Op_ConvL2I && n->in(1) == this;
1162   } else if (op == Op_SubI || op == Op_SubL) {
1163     // Condition for subI(x,subI(y,z)) ==> subI(addI(x,z),y)
1164     return n->Opcode() == op && n->in(2) == this;
1165   } else if (is_If() && (n->is_IfFalse() || n->is_IfTrue())) {
1166     // See IfProjNode::Identity()
1167     return true;
1168   } else {
1169     return false;
1170   }
1171 };

1192 //--------------------------dominates------------------------------------------
1193 // Helper function for MemNode::all_controls_dominate().
1194 // Check if 'this' control node dominates or equal to 'sub' control node.
1195 // We already know that if any path back to Root or Start reaches 'this',
1196 // then all paths so, so this is a simple search for one example,
1197 // not an exhaustive search for a counterexample.
1198 bool Node::dominates(Node* sub, Node_List &nlist) {
1199   assert(this->is_CFG(), "expecting control");
1200   assert(sub != NULL && sub->is_CFG(), "expecting control");
1201 
1202   // detect dead cycle without regions
1203   int iterations_without_region_limit = DominatorSearchLimit;
1204 
1205   Node* orig_sub = sub;
1206   Node* dom      = this;
1207   bool  met_dom  = false;
1208   nlist.clear();
1209 
1210   // Walk 'sub' backward up the chain to 'dom', watching for regions.
1211   // After seeing 'dom', continue up to Root or Start.
1212   // If we hit a region (backward split point), it may be a loop head.
1213   // Keep going through one of the region's inputs.  If we reach the
1214   // same region again, go through a different input.  Eventually we
1215   // will either exit through the loop head, or give up.
1216   // (If we get confused, break out and return a conservative 'false'.)
1217   while (sub != NULL) {
1218     if (sub->is_top())  break; // Conservative answer for dead code.
1219     if (sub == dom) {
1220       if (nlist.size() == 0) {
1221         // No Region nodes except loops were visited before and the EntryControl
1222         // path was taken for loops: it did not walk in a cycle.
1223         return true;
1224       } else if (met_dom) {
1225         break;          // already met before: walk in a cycle
1226       } else {
1227         // Region nodes were visited. Continue walk up to Start or Root
1228         // to make sure that it did not walk in a cycle.
1229         met_dom = true; // first time meet
1230         iterations_without_region_limit = DominatorSearchLimit; // Reset
1231      }
1232     }
1233     if (sub->is_Start() || sub->is_Root()) {
1234       // Success if we met 'dom' along a path to Start or Root.
1235       // We assume there are no alternative paths that avoid 'dom'.
1236       // (This assumption is up to the caller to ensure!)
1237       return met_dom;
1238     }
1239     Node* up = sub->in(0);
1240     // Normalize simple pass-through regions and projections:
1241     up = sub->find_exact_control(up);
1242     // If sub == up, we found a self-loop.  Try to push past it.
1243     if (sub == up && sub->is_Loop()) {
1244       // Take loop entry path on the way up to 'dom'.
1245       up = sub->in(1); // in(LoopNode::EntryControl);
1246     } else if (sub == up && sub->is_Region() && sub->req() == 2) {
1247       // Take in(1) path on the way up to 'dom' for regions with only one input
1248       up = sub->in(1);
1249     } else if (sub == up && sub->is_Region() && sub->req() == 3) {
1250       // Try both paths for Regions with 2 input paths (it may be a loop head).
1251       // It could give conservative 'false' answer without information
1252       // which region's input is the entry path.
1253       iterations_without_region_limit = DominatorSearchLimit; // Reset
1254 
1255       bool region_was_visited_before = false;
1256       // Was this Region node visited before?
1257       // If so, we have reached it because we accidentally took a
1258       // loop-back edge from 'sub' back into the body of the loop,
1259       // and worked our way up again to the loop header 'sub'.
1260       // So, take the first unexplored path on the way up to 'dom'.
1261       for (int j = nlist.size() - 1; j >= 0; j--) {
1262         intptr_t ni = (intptr_t)nlist.at(j);
1263         Node* visited = (Node*)(ni & ~1);
1264         bool  visited_twice_already = ((ni & 1) != 0);
1265         if (visited == sub) {
1266           if (visited_twice_already) {
1267             // Visited 2 paths, but still stuck in loop body.  Give up.
1268             return false;
1269           }
1270           // The Region node was visited before only once.
1271           // (We will repush with the low bit set, below.)
1272           nlist.remove(j);
1273           // We will find a new edge and re-insert.
1274           region_was_visited_before = true;
1275           break;
1276         }
1277       }
1278 
1279       // Find an incoming edge which has not been seen yet; walk through it.
1280       assert(up == sub, "");
1281       uint skip = region_was_visited_before ? 1 : 0;
1282       for (uint i = 1; i < sub->req(); i++) {
1283         Node* in = sub->in(i);
1284         if (in != NULL && !in->is_top() && in != sub) {
1285           if (skip == 0) {
1286             up = in;
1287             break;
1288           }
1289           --skip;               // skip this nontrivial input
1290         }
1291       }
1292 
1293       // Set 0 bit to indicate that both paths were taken.
1294       nlist.push((Node*)((intptr_t)sub + (region_was_visited_before ? 1 : 0)));
1295     }
1296 
1297     if (up == sub) {
1298       break;    // some kind of tight cycle
1299     }
1300     if (up == orig_sub && met_dom) {
1301       // returned back after visiting 'dom'
1302       break;    // some kind of cycle
1303     }
1304     if (--iterations_without_region_limit < 0) {
1305       break;    // dead cycle
1306     }
1307     sub = up;
1308   }
1309 
1310   // Did not meet Root or Start node in pred. chain.
1311   // Conservative answer for dead code.
1312   return false;
1313 }

1315 //------------------------------remove_dead_region-----------------------------
1316 // This control node is dead.  Follow the subgraph below it making everything
1317 // using it dead as well.  This will happen normally via the usual IterGVN
1318 // worklist but this call is more efficient.  Do not update use-def info
1319 // inside the dead region, just at the borders.
1320 static void kill_dead_code( Node *dead, PhaseIterGVN *igvn ) {
1321   // Con's are a popular node to re-hit in the hash table again.
1322   if( dead->is_Con() ) return;
1323 
1324   ResourceMark rm;
1325   Node_List  nstack(Thread::current()->resource_area());
1326 
1327   Node *top = igvn->C->top();
1328   nstack.push(dead);
1329   bool has_irreducible_loop = igvn->C->has_irreducible_loop();
1330 
1331   while (nstack.size() > 0) {
1332     dead = nstack.pop();
1333     if (dead->Opcode() == Op_SafePoint) {
1334       dead->as_SafePoint()->disconnect_from_root(igvn);
1335     }
1336     if (dead->outcnt() > 0) {
1337       // Keep dead node on stack until all uses are processed.
1338       nstack.push(dead);
1339       // For all Users of the Dead...    ;-)
1340       for (DUIterator_Last kmin, k = dead->last_outs(kmin); k >= kmin; ) {
1341         Node* use = dead->last_out(k);
1342         igvn->hash_delete(use);       // Yank from hash table prior to mod
1343         if (use->in(0) == dead) {     // Found another dead node
1344           assert (!use->is_Con(), "Control for Con node should be Root node.");
1345           use->set_req(0, top);       // Cut dead edge to prevent processing
1346           nstack.push(use);           // the dead node again.
1347         } else if (!has_irreducible_loop && // Backedge could be alive in irreducible loop
1348                    use->is_Loop() && !use->is_Root() &&       // Don't kill Root (RootNode extends LoopNode)
1349                    use->in(LoopNode::EntryControl) == dead) { // Dead loop if its entry is dead
1350           use->set_req(LoopNode::EntryControl, top);          // Cut dead edge to prevent processing
1351           use->set_req(0, top);       // Cut self edge
1352           nstack.push(use);
1353         } else {                      // Else found a not-dead user
1354           // Dead if all inputs are top or null
1355           bool dead_use = !use->is_Root(); // Keep empty graph alive
1356           for (uint j = 1; j < use->req(); j++) {
1357             Node* in = use->in(j);
1358             if (in == dead) {         // Turn all dead inputs into TOP
1359               use->set_req(j, top);
1360             } else if (in != NULL && !in->is_top()) {
1361               dead_use = false;
1362             }
1363           }
1364           if (dead_use) {
1365             if (use->is_Region()) {
1366               use->set_req(0, top);   // Cut self edge
1367             }
1368             nstack.push(use);
1369           } else {
1370             igvn->_worklist.push(use);
1371           }
1372         }
1373         // Refresh the iterator, since any number of kills might have happened.
1374         k = dead->last_outs(kmin);
1375       }
1376     } else { // (dead->outcnt() == 0)
1377       // Done with outputs.
1378       igvn->hash_delete(dead);
1379       igvn->_worklist.remove(dead);
1380       igvn->C->remove_modified_node(dead);
1381       igvn->set_type(dead, Type::TOP);
1382       if (dead->is_macro()) {
1383         igvn->C->remove_macro_node(dead);
1384       }
1385       if (dead->is_expensive()) {
1386         igvn->C->remove_expensive_node(dead);
1387       }
1388       CastIINode* cast = dead->isa_CastII();
1389       if (cast != NULL && cast->has_range_check()) {
1390         igvn->C->remove_range_check_cast(cast);
1391       }
1392       if (dead->Opcode() == Op_Opaque4) {
1393         igvn->C->remove_opaque4_node(dead);
1394       }
1395       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1396       bs->unregister_potential_barrier_node(dead);
1397       igvn->C->record_dead_node(dead->_idx);
1398       // Kill all inputs to the dead guy
1399       for (uint i=0; i < dead->req(); i++) {
1400         Node *n = dead->in(i);      // Get input to dead guy
1401         if (n != NULL && !n->is_top()) { // Input is valid?
1402           dead->set_req(i, top);    // Smash input away
1403           if (n->outcnt() == 0) {   // Input also goes dead?
1404             if (!n->is_Con())
1405               nstack.push(n);       // Clear it out as well
1406           } else if (n->outcnt() == 1 &&
1407                      n->has_special_unique_user()) {
1408             igvn->add_users_to_worklist( n );
1409           } else if (n->outcnt() <= 2 && n->is_Store()) {
1410             // Push store's uses on worklist to enable folding optimization for
1411             // store/store and store/load to the same address.
1412             // The restriction (outcnt() <= 2) is the same as in set_req_X()
1413             // and remove_globally_dead_node().
1414             igvn->add_users_to_worklist( n );
1415           } else {
1416             BarrierSet::barrier_set()->barrier_set_c2()->enqueue_useful_gc_barrier(igvn, n);
1417           }
1418         }
1419       }
1420     } // (dead->outcnt() == 0)
1421   }   // while (nstack.size() > 0) for outputs
1422   return;
1423 }

1545 //------------------------------find------------------------------------------
1546 // Find a neighbor of this Node with the given _idx
1547 // If idx is negative, find its absolute value, following both _in and _out.
1548 static void find_recur(Compile* C,  Node* &result, Node *n, int idx, bool only_ctrl,
1549                         VectorSet* old_space, VectorSet* new_space ) {
1550   int node_idx = (idx >= 0) ? idx : -idx;
1551   if (NotANode(n))  return;  // Gracefully handle NULL, -1, 0xabababab, etc.
1552   // Contained in new_space or old_space?   Check old_arena first since it's mostly empty.
1553   VectorSet *v = C->old_arena()->contains(n) ? old_space : new_space;
1554   if( v->test(n->_idx) ) return;
1555   if( (int)n->_idx == node_idx
1556       debug_only(|| n->debug_idx() == node_idx) ) {
1557     if (result != NULL)
1558       tty->print("find: " INTPTR_FORMAT " and " INTPTR_FORMAT " both have idx==%d\n",
1559                  (uintptr_t)result, (uintptr_t)n, node_idx);
1560     result = n;
1561   }
1562   v->set(n->_idx);
1563   for( uint i=0; i<n->len(); i++ ) {
1564     if( only_ctrl && !(n->is_Region()) && (n->Opcode() != Op_Root) && (i != TypeFunc::Control) ) continue;
1565     find_recur(C, result, n->in(i), idx, only_ctrl, old_space, new_space );
1566   }
1567   // Search along forward edges also:
1568   if (idx < 0 && !only_ctrl) {
1569     for( uint j=0; j<n->outcnt(); j++ ) {
1570       find_recur(C, result, n->raw_out(j), idx, only_ctrl, old_space, new_space );

1579       find_recur(C, result, orig, idx, only_ctrl, old_space, new_space );

1806 //----------------------------collect_nodes_i----------------------------------
1807 // Collects nodes from an Ideal graph, starting from a given start node and
1808 // moving in a given direction until a certain depth (distance from the start
1809 // node) is reached. Duplicates are ignored.
1810 // Arguments:
1811 //   nstack:        the nodes are collected into this array.
1812 //   start:         the node at which to start collecting.
1813 //   direction:     if this is a positive number, collect input nodes; if it is
1814 //                  a negative number, collect output nodes.
1815 //   depth:         collect nodes up to this distance from the start node.
1816 //   include_start: whether to include the start node in the result collection.
1817 //   only_ctrl:     whether to regard control edges only during traversal.
1818 //   only_data:     whether to regard data edges only during traversal.
1819 static void collect_nodes_i(GrowableArray<Node*> *nstack, const Node* start, int direction, uint depth, bool include_start, bool only_ctrl, bool only_data) {
1820   Node* s = (Node*) start; // remove const
1821   nstack->append(s);
1822   int begin = 0;
1823   int end = 0;
1824   for(uint i = 0; i < depth; i++) {
1825     end = nstack->length();
1826     for(int j = begin; j < end; j++) {
1827       Node* tp  = nstack->at(j);
1828       uint limit = direction > 0 ? tp->len() : tp->outcnt();
1829       for(uint k = 0; k < limit; k++) {
1830         Node* n = direction > 0 ? tp->in(k) : tp->raw_out(k);
1831 
1832         if (NotANode(n))  continue;
1833         // do not recurse through top or the root (would reach unrelated stuff)
1834         if (n->is_Root() || n->is_top()) continue;
1835         if (only_ctrl && !n->is_CFG()) continue;
1836         if (only_data && n->is_CFG()) continue;
1837 
1838         bool on_stack = nstack->contains(n);
1839         if (!on_stack) {
1840           nstack->append(n);
1841         }
1842       }
1843     }
1844     begin = end;
1845   }
1846   if (!include_start) {
1847     nstack->remove(s);
1848   }
1849 }

2002 //--------------------------collect_nodes_in-----------------------------------
2003 static void collect_nodes_in(Node* start, GrowableArray<Node*> *ns, bool primary_is_data, bool collect_secondary) {
2004   // The maximum depth is determined using a BFS that visits all primary (data
2005   // or control) inputs and increments the depth at each level.
2006   uint d_in = 0;
2007   GrowableArray<Node*> nodes(Compile::current()->unique());
2008   nodes.push(start);
2009   int nodes_at_current_level = 1;
2010   int n_idx = 0;
2011   while (nodes_at_current_level > 0) {
2012     // Add all primary inputs reachable from the current level to the list, and
2013     // increase the depth if there were any.
2014     int nodes_at_next_level = 0;
2015     bool nodes_added = false;
2016     while (nodes_at_current_level > 0) {
2017       nodes_at_current_level--;
2018       Node* current = nodes.at(n_idx++);
2019       for (uint i = 0; i < current->len(); i++) {
2020         Node* n = current->in(i);
2021         if (NotANode(n)) {
2022           continue;
2023         }
2024         if ((primary_is_data && n->is_CFG()) || (!primary_is_data && !n->is_CFG())) {
2025           continue;
2026         }
2027         if (!nodes.contains(n)) {
2028           nodes.push(n);
2029           nodes_added = true;
2030           nodes_at_next_level++;
2031         }
2032       }
2033     }
2034     if (nodes_added) {
2035       d_in++;
2036     }
2037     nodes_at_current_level = nodes_at_next_level;
2038   }
2039   start->collect_nodes(ns, d_in, !primary_is_data, primary_is_data);
2040   if (collect_secondary) {
2041     // Now, iterate over the secondary nodes in ns and add the respective
2042     // boundary reachable from them.
2043     GrowableArray<Node*> sns(Compile::current()->unique());
2044     for (GrowableArrayIterator<Node*> it = ns->begin(); it != ns->end(); ++it) {
2045       Node* n = *it;
2046       n->collect_nodes(&sns, 1, primary_is_data, !primary_is_data);
2047       for (GrowableArrayIterator<Node*> d = sns.begin(); d != sns.end(); ++d) {
2048         ns->append_if_missing(*d);
2049       }
2050       sns.clear();
2051     }
2052   }
2053 }

2096 // VERIFICATION CODE
2097 // For each input edge to a node (ie - for each Use-Def edge), verify that
2098 // there is a corresponding Def-Use edge.
2099 //------------------------------verify_edges-----------------------------------
2100 void Node::verify_edges(Unique_Node_List &visited) {
2101   uint i, j, idx;
2102   int  cnt;
2103   Node *n;
2104 
2105   // Recursive termination test
2106   if (visited.member(this))  return;
2107   visited.push(this);
2108 
2109   // Walk over all input edges, checking for correspondence
2110   for( i = 0; i < len(); i++ ) {
2111     n = in(i);
2112     if (n != NULL && !n->is_top()) {
2113       // Count instances of (Node *)this
2114       cnt = 0;
2115       for (idx = 0; idx < n->_outcnt; idx++ ) {
2116         if (n->_out[idx] == (Node *)this)  cnt++;
2117       }
2118       assert( cnt > 0,"Failed to find Def-Use edge." );
2119       // Check for duplicate edges
2120       // walk the input array downcounting the input edges to n
2121       for( j = 0; j < len(); j++ ) {
2122         if( in(j) == n ) cnt--;
2123       }
2124       assert( cnt == 0,"Mismatched edge count.");
2125     } else if (n == NULL) {
2126       assert(i >= req() || i == 0 || is_Region() || is_Phi(), "only regions or phis have null data edges");
2127     } else {
2128       assert(n->is_top(), "sanity");
2129       // Nothing to check.
2130     }
2131   }
2132   // Recursive walk over all input edges
2133   for( i = 0; i < len(); i++ ) {
2134     n = in(i);
2135     if( n != NULL )
2136       in(i)->verify_edges(visited);
2137   }
2138 }

2297 // It is possible for this operation to return false for a loop-varying
2298 // value, if it appears (by local graph inspection) to be computed by a simple conditional.
2299 bool Node::is_iteratively_computed() {
2300   if (ideal_reg()) { // does operation have a result register?
2301     for (uint i = 1; i < req(); i++) {
2302       Node* n = in(i);
2303       if (n != NULL && n->is_Phi()) {
2304         for (uint j = 1; j < n->req(); j++) {
2305           if (n->in(j) == this) {
2306             return true;
2307           }
2308         }
2309       }
2310     }
2311   }
2312   return false;
2313 }
