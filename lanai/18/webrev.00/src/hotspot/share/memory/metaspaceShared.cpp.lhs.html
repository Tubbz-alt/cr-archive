  89 // The CDS archive is divided into the following regions:
  90 //     mc  - misc code (the method entry trampolines, c++ vtables)
  91 //     rw  - read-write metadata
  92 //     ro  - read-only metadata and read-only tables
  93 //
  94 //     ca0 - closed archive heap space #0
  95 //     ca1 - closed archive heap space #1 (may be empty)
  96 //     oa0 - open archive heap space #0
  97 //     oa1 - open archive heap space #1 (may be empty)
  98 //
  99 // The mc, rw, and ro regions are linearly allocated, starting from
 100 // SharedBaseAddress, in the order of mc->rw->ro. The size of these 3 regions
 101 // are page-aligned, and there's no gap between any consecutive regions.
 102 //
 103 // These 3 regions are populated in the following steps:
 104 // [1] All classes are loaded in MetaspaceShared::preload_classes(). All metadata are
 105 //     temporarily allocated outside of the shared regions. Only the method entry
 106 //     trampolines are written into the mc region.
 107 // [2] C++ vtables are copied into the mc region.
 108 // [3] ArchiveCompactor copies RW metadata into the rw region.
 109 // [4] ArchiveCompactor copies RO metadata into the ro region.
 110 // [5] SymbolTable, StringTable, SystemDictionary, and a few other read-only data
 111 //     are copied into the ro region as read-only tables.
 112 //
 113 // The s0/s1 and oa0/oa1 regions are populated inside HeapShared::archive_java_heap_objects.
 114 // Their layout is independent of the other 4 regions.
 115 
 116 char* DumpRegion::expand_top_to(char* newtop) {
 117   assert(is_allocatable(), "must be initialized and not packed");
 118   assert(newtop >= _top, "must not grow backwards");
 119   if (newtop > _end) {
 120     MetaspaceShared::report_out_of_space(_name, newtop - _top);
 121     ShouldNotReachHere();
 122   }
 123   uintx delta;
 124   if (DynamicDumpSharedSpaces) {
 125     delta = DynamicArchive::object_delta_uintx(newtop);
 126   } else {
 127     delta = MetaspaceShared::object_delta_uintx(newtop);
 128   }
 129   if (delta > MAX_SHARED_DELTA) {
 130     // This is just a sanity check and should not appear in any real world usage. This
 131     // happens only if you allocate more than 2GB of shared objects and would require
 132     // millions of shared classes.
 133     vm_exit_during_initialization("Out of memory in the CDS archive",
 134                                   "Please reduce the number of shared classes.");
 135   }
 136 
 137   MetaspaceShared::commit_shared_space_to(newtop);
 138   _top = newtop;
 139   return _top;
 140 }

 142 char* DumpRegion::allocate(size_t num_bytes, size_t alignment) {
 143   char* p = (char*)align_up(_top, alignment);
 144   char* newtop = p + align_up(num_bytes, alignment);
 145   expand_top_to(newtop);
 146   memset(p, 0, newtop - p);
 147   return p;
 148 }

 150 void DumpRegion::append_intptr_t(intptr_t n, bool need_to_mark) {
 151   assert(is_aligned(_top, sizeof(intptr_t)), "bad alignment");
 152   intptr_t *p = (intptr_t*)_top;
 153   char* newtop = _top + sizeof(intptr_t);
 154   expand_top_to(newtop);
 155   *p = n;
 156   if (need_to_mark) {
 157     ArchivePtrMarker::mark_pointer(p);
 158   }
 159 }

 175 void DumpRegion::pack(DumpRegion* next) {
 176   assert(!is_packed(), "sanity");
 177   _end = (char*)align_up(_top, Metaspace::reserve_alignment());
 178   _is_packed = true;
 179   if (next != NULL) {
 180     next->_base = next->_top = this->_end;
 181     next->_end = MetaspaceShared::shared_rs()->end();
 182   }
 183 }

 222 // When reserving an address range using ReservedSpace, we need an alignment that satisfies both:
 223 // os::vm_allocation_granularity() -- so that we can sub-divide this range into multiple mmap regions,
 224 //                                    while keeping the first range at offset 0 of this range.
 225 // Metaspace::reserve_alignment()  -- so we can pass the region to
 226 //                                    Metaspace::allocate_metaspace_compressed_klass_ptrs.
 227 size_t MetaspaceShared::reserved_space_alignment() {
 228   size_t os_align = os::vm_allocation_granularity();
 229   size_t ms_align = Metaspace::reserve_alignment();
 230   if (os_align >= ms_align) {
 231     assert(os_align % ms_align == 0, "must be a multiple");
 232     return os_align;
 233   } else {
 234     assert(ms_align % os_align == 0, "must be a multiple");
 235     return ms_align;
 236   }
 237 }

 244 void MetaspaceShared::initialize_dumptime_shared_and_meta_spaces() {
 245   assert(DumpSharedSpaces, "should be called for dump time only");
 246   const size_t reserve_alignment = reserved_space_alignment();
 247   char* shared_base = (char*)align_up((char*)SharedBaseAddress, reserve_alignment);
 248 
 249 #ifdef _LP64
 250   // On 64-bit VM, the heap and class space layout will be the same as if
 251   // you're running in -Xshare:on mode:
 252   //
 253   //                              +-- SharedBaseAddress (default = 0x800000000)
 254   //                              v
 255   // +-..---------+---------+ ... +----+----+----+--------------------+
 256   // |    Heap    | Archive |     | MC | RW | RO |    class space     |
 257   // +-..---------+---------+ ... +----+----+----+--------------------+
 258   // |<--   MaxHeapSize  -->|     |<-- UnscaledClassSpaceMax = 4GB -->|
 259   //
 260   const uint64_t UnscaledClassSpaceMax = (uint64_t(max_juint) + 1);
 261   const size_t cds_total = align_down(UnscaledClassSpaceMax, reserve_alignment);
 262 #else
 263   // We don't support archives larger than 256MB on 32-bit due to limited virtual address space.
 264   size_t cds_total = align_down(256*M, reserve_alignment);
 265 #endif
 266 
 267   bool use_requested_base = true;
 268   if (ArchiveRelocationMode == 1) {
 269     log_info(cds)("ArchiveRelocationMode == 1: always allocate class space at an alternative address");
 270     use_requested_base = false;
 271   }
 272 
 273   // First try to reserve the space at the specified SharedBaseAddress.
 274   assert(!_shared_rs.is_reserved(), "must be");
 275   if (use_requested_base) {
 276     _shared_rs = reserve_shared_space(cds_total, shared_base);
 277   }
 278   if (_shared_rs.is_reserved()) {
 279     assert(shared_base == 0 || _shared_rs.base() == shared_base, "should match");
 280   } else {
 281     // Get a mmap region anywhere if the SharedBaseAddress fails.
 282     _shared_rs = reserve_shared_space(cds_total);
 283   }
 284   if (!_shared_rs.is_reserved()) {
 285     vm_exit_during_initialization("Unable to reserve memory for shared space",
 286                                   err_msg(SIZE_FORMAT " bytes.", cds_total));
 287   }
 288 
 289 #ifdef _LP64
 290   // During dump time, we allocate 4GB (UnscaledClassSpaceMax) of space and split it up:
 291   // + The upper 1 GB is used as the "temporary compressed class space" -- preload_classes()
 292   //   will store Klasses into this space.
 293   // + The lower 3 GB is used for the archive -- when preload_classes() is done,
 294   //   ArchiveCompactor will copy the class metadata into this space, first the RW parts,
 295   //   then the RO parts.
 296 
 297   assert(UseCompressedOops && UseCompressedClassPointers,
 298       "UseCompressedOops and UseCompressedClassPointers must be set");
 299 
 300   size_t max_archive_size = align_down(cds_total * 3 / 4, reserve_alignment);
 301   ReservedSpace tmp_class_space = _shared_rs.last_part(max_archive_size);
 302   CompressedClassSpaceSize = align_down(tmp_class_space.size(), reserve_alignment);
 303   _shared_rs = _shared_rs.first_part(max_archive_size);
 304 
 305   // Set up compress class pointers.
 306   CompressedKlassPointers::set_base((address)_shared_rs.base());
 307   // Set narrow_klass_shift to be LogKlassAlignmentInBytes. This is consistent
 308   // with AOT.
 309   CompressedKlassPointers::set_shift(LogKlassAlignmentInBytes);
 310   // Set the range of klass addresses to 4GB.
 311   CompressedKlassPointers::set_range(cds_total);
 312 
 313   Metaspace::initialize_class_space(tmp_class_space);

 314   log_info(cds)("narrow_klass_base = " PTR_FORMAT ", narrow_klass_shift = %d",
 315                 p2i(CompressedKlassPointers::base()), CompressedKlassPointers::shift());
 316 
 317   log_info(cds)("Allocated temporary class space: " SIZE_FORMAT " bytes at " PTR_FORMAT,
 318                 CompressedClassSpaceSize, p2i(tmp_class_space.base()));
 319 #endif
 320 
 321   init_shared_dump_space(&_mc_region);
 322   SharedBaseAddress = (size_t)_shared_rs.base();
 323   log_info(cds)("Allocated shared space: " SIZE_FORMAT " bytes at " PTR_FORMAT,
 324                 _shared_rs.size(), p2i(_shared_rs.base()));
 325 }

 349 void MetaspaceShared::read_extra_data(const char* filename, TRAPS) {
 350   _extra_interned_strings = new (ResourceObj::C_HEAP, mtInternal)GrowableArray<Handle>(10000, true);
 351 
 352   HashtableTextDump reader(filename);
 353   reader.check_version("VERSION: 1.0");
 354 
 355   while (reader.remain() > 0) {
 356     int utf8_length;
 357     int prefix_type = reader.scan_prefix(&utf8_length);
 358     ResourceMark rm(THREAD);
 359     if (utf8_length == 0x7fffffff) {
 360       // buf_len will overflown 32-bit value.
 361       vm_exit_during_initialization(err_msg("string length too large: %d", utf8_length));
 362     }
 363     int buf_len = utf8_length+1;
 364     char* utf8_buffer = NEW_RESOURCE_ARRAY(char, buf_len);
 365     reader.get_utf8(utf8_buffer, utf8_length);
 366     utf8_buffer[utf8_length] = '\0';
 367 
 368     if (prefix_type == HashtableTextDump::SymbolPrefix) {
 369       SymbolTable::new_permanent_symbol(utf8_buffer);
 370     } else{
 371       assert(prefix_type == HashtableTextDump::StringPrefix, "Sanity");
 372       oop s = StringTable::intern(utf8_buffer, THREAD);
 373 
 374       if (HAS_PENDING_EXCEPTION) {
 375         log_warning(cds, heap)("[line %d] extra interned string allocation failed; size too large: %d",
 376                                reader.last_line_no(), utf8_length);
 377         CLEAR_PENDING_EXCEPTION;
 378       } else {
 379 #if INCLUDE_G1GC
 380         if (UseG1GC) {
 381           typeArrayOop body = java_lang_String::value(s);
 382           const HeapRegion* hr = G1CollectedHeap::heap()->heap_region_containing(body);
 383           if (hr->is_humongous()) {
 384             // Don't keep it alive, so it will be GC'ed before we dump the strings, in order
 385             // to maximize free heap space and minimize fragmentation.
 386             log_warning(cds, heap)("[line %d] extra interned string ignored; size too large: %d",
 387                                 reader.last_line_no(), utf8_length);
 388             continue;
 389           }
 390         }
 391 #endif
 392         // Interned strings are GC'ed if there are no references to it, so let's
 393         // add a reference to keep this string alive.
 394         assert(s != NULL, "must succeed");
 395         Handle h(THREAD, s);
 396         _extra_interned_strings->append(h);
 397       }
 398     }
 399   }
 400 }

 402 void MetaspaceShared::commit_shared_space_to(char* newtop) {
 403   Arguments::assert_is_dumping_archive();
 404   char* base = _shared_rs.base();
 405   size_t need_committed_size = newtop - base;
 406   size_t has_committed_size = _shared_vs.committed_size();
 407   if (need_committed_size < has_committed_size) {
 408     return;
 409   }
 410 
 411   size_t min_bytes = need_committed_size - has_committed_size;
 412   size_t preferred_bytes = 1 * M;
 413   size_t uncommitted = _shared_vs.reserved_size() - has_committed_size;
 414 
 415   size_t commit =MAX2(min_bytes, preferred_bytes);
 416   commit = MIN2(commit, uncommitted);
 417   assert(commit <= uncommitted, "sanity");
 418 
 419   bool result = _shared_vs.expand_by(commit, false);
 420   ArchivePtrMarker::expand_ptr_end((address*)_shared_vs.high());
 421 
 422   if (!result) {
 423     vm_exit_during_initialization(err_msg("Failed to expand shared space to " SIZE_FORMAT " bytes",
 424                                           need_committed_size));
 425   }
 426 
 427   log_debug(cds)("Expanding shared spaces by " SIZE_FORMAT_W(7) " bytes [total " SIZE_FORMAT_W(9)  " bytes ending at %p]",
 428                  commit, _shared_vs.actual_committed_size(), _shared_vs.high());
 429 }

 438 void MetaspaceShared::serialize(SerializeClosure* soc) {
 439   int tag = 0;

 445   soc->do_tag(arrayOopDesc::base_offset_in_bytes(T_BYTE));

 449   soc->do_tag(typeArrayOopDesc::base_offset_in_bytes(T_BYTE));
 450   soc->do_tag(sizeof(Symbol));
 451 
 452   // Dump/restore miscellaneous metadata.
 453   JavaClasses::serialize_offsets(soc);
 454   Universe::serialize(soc);
 455   soc->do_tag(--tag);
 456 
 457   // Dump/restore references to commonly used names and signatures.
 458   vmSymbols::serialize(soc);
 459   soc->do_tag(--tag);
 460 
 461   // Dump/restore the symbol/string/subgraph_info tables
 462   SymbolTable::serialize_shared_table_header(soc);
 463   StringTable::serialize_shared_table_header(soc);
 464   HeapShared::serialize_subgraph_info_table_header(soc);
 465   SystemDictionaryShared::serialize_dictionary_headers(soc);
 466 
 467   InstanceMirrorKlass::serialize_offsets(soc);



 468   soc->do_tag(--tag);
 469 
 470   serialize_cloned_cpp_vtptrs(soc);
 471   soc->do_tag(--tag);
 472 
 473   soc->do_tag(666);
 474 }

 492 uintx MetaspaceShared::object_delta_uintx(void* obj) {
 493   Arguments::assert_is_dumping_archive();
 494   if (DumpSharedSpaces) {
 495     assert(shared_rs()->contains(obj), "must be");
 496   } else {
 497     assert(is_in_shared_metaspace(obj) || DynamicArchive::is_in_target_space(obj), "must be");
 498   }
 499   address base_address = address(SharedBaseAddress);
 500   uintx deltax = address(obj) - base_address;
 501   return deltax;
 502 }

 536       if (h != NULL) {

 580 static void rewrite_nofast_bytecode(const methodHandle& method) {
 581   BytecodeStream bcs(method);
 582   while (!bcs.is_last_bytecode()) {
 583     Bytecodes::Code opcode = bcs.next();
 584     switch (opcode) {
 585     case Bytecodes::_getfield:      *bcs.bcp() = Bytecodes::_nofast_getfield;      break;
 586     case Bytecodes::_putfield:      *bcs.bcp() = Bytecodes::_nofast_putfield;      break;
 587     case Bytecodes::_aload_0:       *bcs.bcp() = Bytecodes::_nofast_aload_0;       break;
 588     case Bytecodes::_iload: {
 589       if (!bcs.is_wide()) {
 590         *bcs.bcp() = Bytecodes::_nofast_iload;
 591       }
 592       break;
 593     }
 594     default: break;
 595     }
 596   }
 597 }

 624 // Objects of the Metadata types (such as Klass and ConstantPool) have C++ vtables.
 625 // (In GCC this is the field <Type>::_vptr, i.e., first word in the object.)
 626 //
 627 // Addresses of the vtables and the methods may be different across JVM runs,
 628 // if libjvm.so is dynamically loaded at a different base address.
 629 //
 630 // To ensure that the Metadata objects in the CDS archive always have the correct vtable:
 631 //
 632 // + at dump time:  we redirect the _vptr to point to our own vtables inside
 633 //                  the CDS image
 634 // + at run time:   we clone the actual contents of the vtables from libjvm.so
 635 //                  into our own tables.

 638 #define CPP_VTABLE_PATCH_TYPES_DO(f) \
 639   f(ConstantPool) \
 640   f(InstanceKlass) \
 641   f(InstanceClassLoaderKlass) \
 642   f(InstanceMirrorKlass) \
 643   f(InstanceRefKlass) \
 644   f(Method) \
 645   f(ObjArrayKlass) \
 646   f(TypeArrayKlass)

 648 class CppVtableInfo {
 649   intptr_t _vtable_size;
 650   intptr_t _cloned_vtable[1];
 651 public:
 652   static int num_slots(int vtable_size) {
 653     return 1 + vtable_size; // Need to add the space occupied by _vtable_size;
 654   }
 655   int vtable_size()           { return int(uintx(_vtable_size)); }
 656   void set_vtable_size(int n) { _vtable_size = intptr_t(n); }
 657   intptr_t* cloned_vtable()   { return &_cloned_vtable[0]; }
 658   void zero()                 { memset(_cloned_vtable, 0, sizeof(intptr_t) * vtable_size()); }
 659   // Returns the address of the next CppVtableInfo that can be placed immediately after this CppVtableInfo
 660   static size_t byte_size(int vtable_size) {
 661     CppVtableInfo i;
 662     return pointer_delta(&i._cloned_vtable[vtable_size], &i, sizeof(u1));
 663   }
 664 };

 707 template <class T>
 708 intptr_t* CppVtableCloner<T>::clone_vtable(const char* name, CppVtableInfo* info) {
 709   if (!DumpSharedSpaces) {
 710     assert(_info == 0, "_info is initialized only at dump time");
 711     _info = info; // Remember it -- it will be used by MetaspaceShared::is_valid_shared_method()
 712   }
 713   T tmp; // Allocate temporary dummy metadata object to get to the original vtable.
 714   int n = info->vtable_size();
 715   intptr_t* srcvtable = vtable_of(tmp);
 716   intptr_t* dstvtable = info->cloned_vtable();
 717 
 718   // We already checked (and, if necessary, adjusted n) when the vtables were allocated, so we are
 719   // safe to do memcpy.
 720   log_debug(cds, vtables)("Copying %3d vtable entries for %s", n, name);
 721   memcpy(dstvtable, srcvtable, sizeof(intptr_t) * n);
 722   return dstvtable + n;
 723 }

 725 // To determine the size of the vtable for each type, we use the following
 726 // trick by declaring 2 subclasses:
 727 //
 728 //   class CppVtableTesterA: public InstanceKlass {virtual int   last_virtual_method() {return 1;}    };
 729 //   class CppVtableTesterB: public InstanceKlass {virtual void* last_virtual_method() {return NULL}; };
 730 //
 731 // CppVtableTesterA and CppVtableTesterB's vtables have the following properties:
 732 // - Their size (N+1) is exactly one more than the size of InstanceKlass's vtable (N)
 733 // - The first N entries have are exactly the same as in InstanceKlass's vtable.
 734 // - Their last entry is different.
 735 //
 736 // So to determine the value of N, we just walk CppVtableTesterA and CppVtableTesterB's tables
 737 // and find the first entry that's different.
 738 //
 739 // This works on all C++ compilers supported by Oracle, but you may need to tweak it for more
 740 // esoteric compilers.
 741 
 742 template <class T> class CppVtableTesterB: public T {
 743 public:
 744   virtual int last_virtual_method() {return 1;}
 745 };
 746 
 747 template <class T> class CppVtableTesterA : public T {
 748 public:
 749   virtual void* last_virtual_method() {
 750     // Make this different than CppVtableTesterB::last_virtual_method so the C++
 751     // compiler/linker won't alias the two functions.
 752     return NULL;
 753   }
 754 };
 755 
 756 template <class T>
 757 int CppVtableCloner<T>::get_vtable_length(const char* name) {
 758   CppVtableTesterA<T> a;
 759   CppVtableTesterB<T> b;
 760 
 761   intptr_t* avtable = vtable_of(a);
 762   intptr_t* bvtable = vtable_of(b);
 763 
 764   // Start at slot 1, because slot 0 may be RTTI (on Solaris/Sparc)
 765   int vtable_len = 1;
 766   for (; ; vtable_len++) {
 767     if (avtable[vtable_len] != bvtable[vtable_len]) {
 768       break;
 769     }
 770   }
 771   log_debug(cds, vtables)("Found   %3d vtable entries for %s", vtable_len, name);
 772 
 773   return vtable_len;
 774 }

 814   Arguments::assert_is_dumping_archive();
 815   int kind = -1;
 816   switch (msotype) {
 817   case MetaspaceObj::SymbolType:
 818   case MetaspaceObj::TypeArrayU1Type:
 819   case MetaspaceObj::TypeArrayU2Type:
 820   case MetaspaceObj::TypeArrayU4Type:
 821   case MetaspaceObj::TypeArrayU8Type:
 822   case MetaspaceObj::TypeArrayOtherType:
 823   case MetaspaceObj::ConstMethodType:
 824   case MetaspaceObj::ConstantPoolCacheType:
 825   case MetaspaceObj::AnnotationsType:
 826   case MetaspaceObj::MethodCountersType:
 827   case MetaspaceObj::RecordComponentType:
 828     // These have no vtables.
 829     break;

 831     {

 856       Method* m = (Method*)obj;

 859     }

1184   static void allocate(MetaspaceClosure::Ref* ref, bool read_only) {
1185     address obj = ref->obj();
1186     int bytes = ref->size() * BytesPerWord;
1187     char* p;
1188     size_t alignment = BytesPerWord;
1189     char* oldtop;
1190     char* newtop;

1192     if (read_only) {
1193       oldtop = _ro_region.top();
1194       p = _ro_region.allocate(bytes, alignment);
1195       newtop = _ro_region.top();
1196     } else {
1197       oldtop = _rw_region.top();
1198       if (ref->msotype() == MetaspaceObj::ClassType) {
1199         // Save a pointer immediate in front of an InstanceKlass, so
1200         // we can do a quick lookup from InstanceKlass* -> RunTimeSharedClassInfo*
1201         // without building another hashtable. See RunTimeSharedClassInfo::get_for()
1202         // in systemDictionaryShared.cpp.
1203         Klass* klass = (Klass*)obj;
1204         if (klass->is_instance_klass()) {
1205           SystemDictionaryShared::validate_before_archiving(InstanceKlass::cast(klass));
1206           _rw_region.allocate(sizeof(address), BytesPerWord);
1207         }
1208       }
1209       p = _rw_region.allocate(bytes, alignment);
1210       newtop = _rw_region.top();
1211     }
1212     memcpy(p, obj, bytes);
1213 
1214     intptr_t* cloned_vtable = MetaspaceShared::fix_cpp_vtable_for_dynamic_archive(ref->msotype(), (address)p);
1215     if (cloned_vtable != NULL) {
1216       *(address*)p = (address)cloned_vtable;
1217       ArchivePtrMarker::mark_pointer((address*)p);
1218     }
1219 
1220     assert(_new_loc_table->lookup(obj) == NULL, "each object can be relocated at most once");
1221     _new_loc_table->add(obj, (address)p);
1222     log_trace(cds)("Copy: " PTR_FORMAT " ==> " PTR_FORMAT " %d", p2i(obj), p2i(p), bytes);
1223     if (_new_loc_table->maybe_grow(MAX_TABLE_SIZE)) {
1224       log_info(cds, hashtables)("Expanded _new_loc_table to %d", _new_loc_table->table_size());
1225     }
1226     _alloc_stats->record(ref->msotype(), int(newtop - oldtop), read_only);
1227   }

1451 void VM_PopulateDumpSharedSpace::relocate_to_default_base_address(CHeapBitMap* ptrmap) {
1452   intx addr_delta = MetaspaceShared::final_delta();
1453   if (addr_delta == 0) {
1454     ArchivePtrMarker::compact((address)SharedBaseAddress, (address)_ro_region.top());
1455   } else {
1456     // We are not able to reserve space at Arguments::default_SharedBaseAddress() (due to ASLR).
1457     // This means that the current content of the archive is based on a random
1458     // address. Let's relocate all the pointers, so that it can be mapped to
1459     // Arguments::default_SharedBaseAddress() without runtime relocation.
1460     //
1461     // Note: both the base and dynamic archive are written with
1462     // FileMapHeader::_shared_base_address == Arguments::default_SharedBaseAddress()
1463 
1464     // Patch all pointers that are marked by ptrmap within this region,
1465     // where we have just dumped all the metaspace data.
1466     address patch_base = (address)SharedBaseAddress;
1467     address patch_end  = (address)_ro_region.top();
1468     size_t size = patch_end - patch_base;
1469 
1470     // the current value of the pointers to be patched must be within this
1471     // range (i.e., must point to valid metaspace objects)
1472     address valid_old_base = patch_base;
1473     address valid_old_end  = patch_end;
1474 
1475     // after patching, the pointers must point inside this range
1476     // (the requested location of the archive, as mapped at runtime).
1477     address valid_new_base = (address)Arguments::default_SharedBaseAddress();
1478     address valid_new_end  = valid_new_base + size;
1479 
1480     log_debug(cds)("Relocating archive from [" INTPTR_FORMAT " - " INTPTR_FORMAT " ] to "
1481                    "[" INTPTR_FORMAT " - " INTPTR_FORMAT " ]", p2i(patch_base), p2i(patch_end),
1482                    p2i(valid_new_base), p2i(valid_new_end));
1483 
1484     SharedDataRelocator<true> patcher((address*)patch_base, (address*)patch_end, valid_old_base, valid_old_end,
1485                                       valid_new_base, valid_new_end, addr_delta, ptrmap);
1486     ptrmap->iterate(&patcher);
1487     ArchivePtrMarker::compact(patcher.max_non_null_offset());
1488   }
1489 }

1491 void VM_PopulateDumpSharedSpace::doit() {
1492   CHeapBitMap ptrmap;
1493   MetaspaceShared::initialize_ptr_marker(&ptrmap);
1494 
1495   // We should no longer allocate anything from the metaspace, so that:
1496   //
1497   // (1) Metaspace::allocate might trigger GC if we have run out of
1498   //     committed metaspace, but we can't GC because we're running
1499   //     in the VM thread.
1500   // (2) ArchiveCompactor needs to work with a stable set of MetaspaceObjs.
1501   Metaspace::freeze();
1502   DEBUG_ONLY(SystemDictionaryShared::NoClassLoadingMark nclm);
1503 
1504   Thread* THREAD = VMThread::vm_thread();
1505 
1506   FileMapInfo::check_nonempty_dir_in_shared_path_table();
1507 
1508   NOT_PRODUCT(SystemDictionary::verify();)
1509   // The following guarantee is meant to ensure that no loader constraints
1510   // exist yet, since the constraints table is not shared.  This becomes
1511   // more important now that we don't re-initialize vtables/itables for
1512   // shared classes at runtime, where constraints were previously created.
1513   guarantee(SystemDictionary::constraints()->number_of_entries() == 0,
1514             "loader constraints are not saved");
1515   guarantee(SystemDictionary::placeholders()->number_of_entries() == 0,
1516           "placeholders are not saved");
1517 
1518   // At this point, many classes have been loaded.
1519   // Gather systemDictionary classes in a global array and do everything to
1520   // that so we don't have to walk the SystemDictionary again.
1521   SystemDictionaryShared::check_excluded_classes();
1522   _global_klass_objects = new GrowableArray<Klass*>(1000);
1523   CollectClassesClosure collect_classes;
1524   ClassLoaderDataGraph::loaded_classes_do(&collect_classes);
1525 
1526   print_class_stats();
1527 
1528   // Ensure the ConstMethods won't be modified at run-time
1529   log_info(cds)("Updating ConstMethods ... ");
1530   rewrite_nofast_bytecodes_and_calculate_fingerprints(THREAD);
1531   log_info(cds)("done. ");
1532 
1533   // Remove all references outside the metadata
1534   log_info(cds)("Removing unshareable information ... ");
1535   remove_unshareable_in_classes();
1536   log_info(cds)("done. ");
1537 
1538   MetaspaceShared::allocate_cloned_cpp_vtptrs();
1539   char* cloned_vtables = _mc_region.top();
1540   MetaspaceShared::allocate_cpp_vtable_clones();
1541 
1542   ArchiveCompactor::initialize();
1543   ArchiveCompactor::copy_and_compact();
1544 
1545   dump_symbols();
1546 
1547   // Dump supported java heap objects
1548   _closed_archive_heap_regions = NULL;
1549   _open_archive_heap_regions = NULL;
1550   dump_java_heap_objects();
1551 
1552   ArchiveCompactor::relocate_well_known_klasses();
1553 
1554   char* serialized_data = dump_read_only_tables();
1555   _ro_region.pack();
1556 
1557   // The vtable clones contain addresses of the current process.
1558   // We don't want to write these addresses into the archive.
1559   MetaspaceShared::zero_cpp_vtable_clones_for_writing();
1560 
1561   // relocate the data so that it can be mapped to Arguments::default_SharedBaseAddress()
1562   // without runtime relocation.
1563   relocate_to_default_base_address(&ptrmap);
1564 
1565   // Create and write the archive file that maps the shared spaces.
1566 
1567   FileMapInfo* mapinfo = new FileMapInfo(true);
1568   mapinfo->populate_header(os::vm_allocation_granularity());
1569   mapinfo->set_serialized_data(serialized_data);
1570   mapinfo->set_cloned_vtables(cloned_vtables);
1571   mapinfo->set_i2i_entry_code_buffers(MetaspaceShared::i2i_entry_code_buffers(),
1572                                       MetaspaceShared::i2i_entry_code_buffers_size());
1573   mapinfo->open_for_write();
1574   MetaspaceShared::write_core_archive_regions(mapinfo);
1575   _total_closed_archive_region_size = mapinfo->write_archive_heap_regions(
1576                                         _closed_archive_heap_regions,
1577                                         _closed_archive_heap_oopmaps,
1578                                         MetaspaceShared::first_closed_archive_heap_region,
1579                                         MetaspaceShared::max_closed_archive_heap_region);
1580   _total_open_archive_region_size = mapinfo->write_archive_heap_regions(
1581                                         _open_archive_heap_regions,
1582                                         _open_archive_heap_oopmaps,
1583                                         MetaspaceShared::first_open_archive_heap_region,
1584                                         MetaspaceShared::max_open_archive_heap_region);
1585 
1586   mapinfo->set_final_requested_base((char*)Arguments::default_SharedBaseAddress());
1587   mapinfo->set_header_crc(mapinfo->compute_header_crc());
1588   mapinfo->write_header();
1589   mapinfo->close();
1590 
1591   print_region_stats();
1592 
1593   if (log_is_enabled(Info, cds)) {
1594     ArchiveCompactor::alloc_stats()->print_stats(int(_ro_region.used()), int(_rw_region.used()),
1595                                                  int(_mc_region.used()));
1596   }
1597 
1598   if (PrintSystemDictionaryAtExit) {
1599     SystemDictionary::print();
1600   }
1601 
1602   if (AllowArchivingWithJavaAgent) {
1603     warning("This archive was created with AllowArchivingWithJavaAgent. It should be used "
1604             "for testing purposes only and should not be used in a production environment");
1605   }
1606 
1607   // There may be other pending VM operations that operate on the InstanceKlasses,
1608   // which will fail because InstanceKlasses::remove_unshareable_info()
1609   // has been called. Forget these operations and exit the VM directly.
1610   vm_direct_exit(0);
1611 }

1782           }
1783         }
1784       }

1787                      sizeof(class_list_path_str) - class_list_path_len,
1788                      "%sclasslist", os::file_separator());

1836 int MetaspaceShared::preload_classes(const char* class_list_path, TRAPS) {
1837   ClassListParser parser(class_list_path);
1838   int class_count = 0;
1839 
1840   while (parser.parse_one_line()) {
1841     Klass* klass = parser.load_current_class(THREAD);
1842     if (HAS_PENDING_EXCEPTION) {
1843       if (klass == NULL &&
1844           (PENDING_EXCEPTION->klass()->name() == vmSymbols::java_lang_ClassNotFoundException())) {
1845         // print a warning only when the pending exception is class not found
1846         log_warning(cds)("Preload Warning: Cannot find %s", parser.current_class_name());
1847       }
1848       CLEAR_PENDING_EXCEPTION;
1849     }
1850     if (klass != NULL) {
1851       if (log_is_enabled(Trace, cds)) {
1852         ResourceMark rm(THREAD);
1853         log_trace(cds)("Shared spaces preloaded: %s", klass->external_name());
1854       }
1855 
1856       if (klass->is_instance_klass()) {
1857         InstanceKlass* ik = InstanceKlass::cast(klass);
1858 
1859         // Link the class to cause the bytecodes to be rewritten and the
1860         // cpcache to be created. The linking is done as soon as classes
1861         // are loaded in order that the related data structures (klass and
1862         // cpCache) are located together.
1863         try_link_class(ik, THREAD);
1864         guarantee(!HAS_PENDING_EXCEPTION, "exception in link_class");
1865       }
1866 
1867       class_count++;
1868     }
1869   }
1870 
1871   return class_count;
1872 }

1874 // Returns true if the class's status has changed
1875 bool MetaspaceShared::try_link_class(InstanceKlass* ik, TRAPS) {
1876   assert(DumpSharedSpaces, "should only be called during dumping");
1877   if (ik->init_state() < InstanceKlass::linked) {

1878     bool saved = BytecodeVerificationLocal;
1879     if (ik->loader_type() == 0 && ik->class_loader() == NULL) {
1880       // The verification decision is based on BytecodeVerificationRemote
1881       // for non-system classes. Since we are using the NULL classloader
1882       // to load non-system classes for customized class loaders during dumping,
1883       // we need to temporarily change BytecodeVerificationLocal to be the same as
1884       // BytecodeVerificationRemote. Note this can cause the parent system
1885       // classes also being verified. The extra overhead is acceptable during
1886       // dumping.
1887       BytecodeVerificationLocal = BytecodeVerificationRemote;
1888     }
1889     ik->link_class(THREAD);
1890     if (HAS_PENDING_EXCEPTION) {
1891       ResourceMark rm(THREAD);
1892       log_warning(cds)("Preload Warning: Verification failed for %s",
1893                     ik->external_name());
1894       CLEAR_PENDING_EXCEPTION;
1895       ik->set_in_error_state();
1896       _has_error_classes = true;
1897     }
1898     BytecodeVerificationLocal = saved;
1899     return true;
1900   } else {
1901     return false;
1902   }
1903 }

2097 // use_requested_addr:
2098 //  true  = map at FileMapHeader::_requested_base_address
2099 //  false = map at an alternative address picked by OS.
2100 MapArchiveResult MetaspaceShared::map_archives(FileMapInfo* static_mapinfo, FileMapInfo* dynamic_mapinfo,
2101                                                bool use_requested_addr) {
2102   PRODUCT_ONLY(if (ArchiveRelocationMode == 1 && use_requested_addr) {
2103       // For product build only -- this is for benchmarking the cost of doing relocation.
2104       // For debug builds, the check is done in FileMapInfo::map_regions for better test coverage.
2105       log_info(cds)("ArchiveRelocationMode == 1: always map archive(s) at an alternative address");
2106       return MAP_ARCHIVE_MMAP_FAILURE;
2107     });
2108 
2109   if (ArchiveRelocationMode == 2 && !use_requested_addr) {
2110     log_info(cds)("ArchiveRelocationMode == 2: never map archive(s) at an alternative address");
2111     return MAP_ARCHIVE_MMAP_FAILURE;
2112   };
2113 
2114   if (dynamic_mapinfo != NULL) {
2115     // Ensure that the OS won't be able to allocate new memory spaces between the two
2116     // archives, or else it would mess up the simple comparision in MetaspaceObj::is_shared().
2117     assert(static_mapinfo->mapping_end_offset() == dynamic_mapinfo->mapping_base_offset(), "no gap");
2118   }
2119 
2120   ReservedSpace main_rs, archive_space_rs, class_space_rs;
2121   MapArchiveResult result = MAP_ARCHIVE_OTHER_FAILURE;
2122   char* mapped_base_address = reserve_address_space_for_archives(static_mapinfo, dynamic_mapinfo,
2123                                                                  use_requested_addr, main_rs, archive_space_rs,
2124                                                                  class_space_rs);
2125   if (mapped_base_address == NULL) {
2126     result = MAP_ARCHIVE_MMAP_FAILURE;
2127   } else {
2128     log_debug(cds)("Reserved archive_space_rs     [" INTPTR_FORMAT " - " INTPTR_FORMAT "] (" SIZE_FORMAT ") bytes",
2129                    p2i(archive_space_rs.base()), p2i(archive_space_rs.end()), archive_space_rs.size());
2130     log_debug(cds)("Reserved class_space_rs [" INTPTR_FORMAT " - " INTPTR_FORMAT "] (" SIZE_FORMAT ") bytes",
2131                    p2i(class_space_rs.base()), p2i(class_space_rs.end()), class_space_rs.size());
2132     MapArchiveResult static_result = map_archive(static_mapinfo, mapped_base_address, archive_space_rs);
2133     MapArchiveResult dynamic_result = (static_result == MAP_ARCHIVE_SUCCESS) ?
2134                                      map_archive(dynamic_mapinfo, mapped_base_address, archive_space_rs) : MAP_ARCHIVE_OTHER_FAILURE;
2135 
2136     DEBUG_ONLY(if (ArchiveRelocationMode == 1 && use_requested_addr) {
2137       // This is for simulating mmap failures at the requested address. In debug builds, we do it
2138       // here (after all archives have possibly been mapped), so we can thoroughly test the code for
2139       // failure handling (releasing all allocated resource, etc).
2140       log_info(cds)("ArchiveRelocationMode == 1: always map archive(s) at an alternative address");
2141       if (static_result == MAP_ARCHIVE_SUCCESS) {
2142         static_result = MAP_ARCHIVE_MMAP_FAILURE;
2143       }
2144       if (dynamic_result == MAP_ARCHIVE_SUCCESS) {
2145         dynamic_result = MAP_ARCHIVE_MMAP_FAILURE;
2146       }
2147     });
2148 
2149     if (static_result == MAP_ARCHIVE_SUCCESS) {
2150       if (dynamic_result == MAP_ARCHIVE_SUCCESS) {
2151         result = MAP_ARCHIVE_SUCCESS;
2152       } else if (dynamic_result == MAP_ARCHIVE_OTHER_FAILURE) {
2153         assert(dynamic_mapinfo != NULL && !dynamic_mapinfo->is_mapped(), "must have failed");
2154         // No need to retry mapping the dynamic archive again, as it will never succeed
2155         // (bad file, etc) -- just keep the base archive.
2156         log_warning(cds, dynamic)("Unable to use shared archive. The top archive failed to load: %s",
2157                                   dynamic_mapinfo->full_path());
2158         result = MAP_ARCHIVE_SUCCESS;
2159         // TODO, we can give the unused space for the dynamic archive to class_space_rs, but there's no
2160         // easy API to do that right now.
2161       } else {
2162         result = MAP_ARCHIVE_MMAP_FAILURE;
2163       }
2164     } else if (static_result == MAP_ARCHIVE_OTHER_FAILURE) {
2165       result = MAP_ARCHIVE_OTHER_FAILURE;
2166     } else {
2167       result = MAP_ARCHIVE_MMAP_FAILURE;
2168     }
2169   }
2170 
2171   if (result == MAP_ARCHIVE_SUCCESS) {
2172     if (!main_rs.is_reserved() && class_space_rs.is_reserved()) {
2173       MemTracker::record_virtual_memory_type((address)class_space_rs.base(), mtClass);
2174     }
2175     SharedBaseAddress = (size_t)mapped_base_address;
2176     LP64_ONLY({
2177         if (Metaspace::using_class_space()) {
2178           assert(class_space_rs.is_reserved(), "must be");
2179           char* cds_base = static_mapinfo->mapped_base();
2180           Metaspace::allocate_metaspace_compressed_klass_ptrs(class_space_rs, NULL, (address)cds_base);
2181           // map_heap_regions() compares the current narrow oop and klass encodings
2182           // with the archived ones, so it must be done after all encodings are determined.
2183           static_mapinfo->map_heap_regions();

2184         }
2185         CompressedKlassPointers::set_range(CompressedClassSpaceSize);
2186       });
2187   } else {
2188     unmap_archive(static_mapinfo);
2189     unmap_archive(dynamic_mapinfo);
2190     release_reserved_spaces(main_rs, archive_space_rs, class_space_rs);
2191   }
2192 
2193   return result;

2196 char* MetaspaceShared::reserve_address_space_for_archives(FileMapInfo* static_mapinfo,
2197                                                           FileMapInfo* dynamic_mapinfo,
2198                                                           bool use_requested_addr,
2199                                                           ReservedSpace& main_rs,
2200                                                           ReservedSpace& archive_space_rs,
2201                                                           ReservedSpace& class_space_rs) {
2202   const bool use_klass_space = NOT_LP64(false) LP64_ONLY(Metaspace::using_class_space());
2203   const size_t class_space_size = NOT_LP64(0) LP64_ONLY(Metaspace::compressed_class_space_size());
2204 
2205   if (use_klass_space) {
2206     assert(class_space_size > 0, "CompressedClassSpaceSize must have been validated");
2207   }
2208   if (use_requested_addr && !is_aligned(static_mapinfo->requested_base_address(), reserved_space_alignment())) {
2209     return NULL;
2210   }
2211 
2212   // Size and requested location of the archive_space_rs (for both static and dynamic archives)
2213   size_t base_offset = static_mapinfo->mapping_base_offset();
2214   size_t end_offset  = (dynamic_mapinfo == NULL) ? static_mapinfo->mapping_end_offset() : dynamic_mapinfo->mapping_end_offset();
2215   assert(base_offset == 0, "must be");
2216   assert(is_aligned(end_offset,  os::vm_allocation_granularity()), "must be");
2217   assert(is_aligned(base_offset, os::vm_allocation_granularity()), "must be");
2218 
2219   // In case reserved_space_alignment() != os::vm_allocation_granularity()
2220   assert((size_t)os::vm_allocation_granularity() <= reserved_space_alignment(), "must be");
2221   end_offset = align_up(end_offset, reserved_space_alignment());
2222 
2223   size_t archive_space_size = end_offset - base_offset;
2224 
2225   // Special handling for Windows because it cannot mmap into a reserved space:
2226   //    use_requested_addr: We just map each region individually, and give up if any one of them fails.
2227   //   !use_requested_addr: We reserve the space first, and then os::read in all the regions (instead of mmap).
2228   //                        We're going to patch all the pointers anyway so there's no benefit for mmap.
2229 
2230   if (use_requested_addr) {
2231     char* archive_space_base = static_mapinfo->requested_base_address() + base_offset;
2232     char* archive_space_end  = archive_space_base + archive_space_size;
2233     if (!MetaspaceShared::use_windows_memory_mapping()) {
2234       archive_space_rs = reserve_shared_space(archive_space_size, archive_space_base);
2235       if (!archive_space_rs.is_reserved()) {
2236         return NULL;
2237       }
2238     }
2239     if (use_klass_space) {
2240       // Make sure we can map the klass space immediately following the archive_space space
2241       // Don't call reserve_shared_space here as that may try to enforce platform-specific
2242       // alignment rules which only apply to the archive base address
2243       char* class_space_base = archive_space_end;
2244       class_space_rs = ReservedSpace(class_space_size, reserved_space_alignment(),
2245                                      false /* large_pages */, class_space_base);
2246       if (!class_space_rs.is_reserved()) {
2247         return NULL;
2248       }
2249     }
2250     return static_mapinfo->requested_base_address();
2251   } else {
2252     if (use_klass_space) {
2253       main_rs = reserve_shared_space(archive_space_size + class_space_size);
2254       if (main_rs.is_reserved()) {
2255         archive_space_rs = main_rs.first_part(archive_space_size, reserved_space_alignment(), /*split=*/true);
2256         class_space_rs = main_rs.last_part(archive_space_size);
2257       }
2258     } else {
2259       main_rs = reserve_shared_space(archive_space_size);
2260       archive_space_rs = main_rs;
2261     }
2262     if (archive_space_rs.is_reserved()) {
2263       return archive_space_rs.base();
2264     } else {
2265       return NULL;
2266     }
2267   }
2268 }

2296   assert(UseSharedSpaces, "must be runtime");
2297   if (mapinfo == NULL) {

2299   }
2300 
2301   mapinfo->set_is_mapped(false);
2302 
2303   if (mapinfo->alignment() != (size_t)os::vm_allocation_granularity()) {
2304     log_error(cds)("Unable to map CDS archive -- os::vm_allocation_granularity() expected: " SIZE_FORMAT

2310     mapinfo->map_regions(archive_regions, archive_regions_count, mapped_base_address, rs);
2311

2314     return result;

2316

2318     if (!mapinfo->validate_shared_path_table()) {
2319       unmap_archive(mapinfo);

2321     }
2322   } else {
2323     if (!DynamicArchive::validate(mapinfo)) {

2326     }
2327   }
2328
