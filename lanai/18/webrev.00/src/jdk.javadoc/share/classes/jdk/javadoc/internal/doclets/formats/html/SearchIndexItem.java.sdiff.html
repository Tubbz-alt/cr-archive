 48     private String url = "";
 49     private String containingModule = "";

112     public String toString() {
113         StringBuilder item = new StringBuilder("");

114         switch (category) {
115         case MODULES:
116             item.append("{")
117                     .append("\"l\":\"").append(label).append("\"")
118                     .append("}");
119             break;
120         case PACKAGES:
121             item.append("{");
122             if (!containingModule.isEmpty()) {
123                 item.append("\"m\":\"").append(containingModule).append("\",");
124             }
125             item.append("\"l\":\"").append(label).append("\"");
126             if (!url.isEmpty()) {
127                 item.append(",\"url\":\"").append(url).append("\"");
128             }
129             item.append("}");
130             break;
131         case TYPES:
132             item.append("{");
133             if (!containingPackage.isEmpty()) {
134                 item.append("\"p\":\"").append(containingPackage).append("\",");
135             }
136             item.append("\"l\":\"").append(label).append("\"");
137             if (!url.isEmpty()) {
138                 item.append(",\"url\":\"").append(url).append("\"");
139             }
140             item.append("}");
141             break;
142         case MEMBERS:
143             item.append("{")
144                     .append("\"p\":\"").append(containingPackage).append("\",")
145                     .append("\"c\":\"").append(containingClass).append("\",")
146                     .append("\"l\":\"").append(label).append("\"");
147             if (!url.isEmpty()) {
148                 item.append(",\"url\":\"").append(url).append("\"");
149             }
150             item.append("}");
151             break;
152         case SEARCH_TAGS:
153             item.append("{")
154                     .append("\"l\":\"").append(label).append("\",")
155                     .append("\"h\":\"").append(holder).append("\",");
156             if (!description.isEmpty()) {
157                 item.append("\"d\":\"").append(description).append("\",");
158             }
159             item.append("\"u\":\"").append(url).append("\"")
160                     .append("}");
161             break;
162         default:
163             throw new IllegalStateException("category not set");

164         }
165         return item.toString();
166     }
167 
168     /**

170      *
171      * @return the simple name
172      */
173     public String getSimpleName() {
174         return label.substring(label.lastIndexOf('.') + 1);
175     }
