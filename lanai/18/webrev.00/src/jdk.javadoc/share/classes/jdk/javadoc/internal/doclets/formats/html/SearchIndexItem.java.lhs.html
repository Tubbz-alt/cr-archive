 26 package jdk.javadoc.internal.doclets.formats.html;
 27 


 28 /**
 29  * Index item for search.
 30  *
 31  *  <p><b>This is NOT part of any supported API.
 32  *  If you write code that depends on this, you do so at your own risk.
 33  *  This code and its internal interfaces are subject to change or
 34  *  deletion without notice.</b>
 35  */
 36 public class SearchIndexItem {
 37 
 38     enum Category {
 39         MODULES,
 40         PACKAGES,
 41         TYPES,
 42         MEMBERS,
 43         SEARCH_TAGS







 44     }
 45 
 46     private Category category;
 47     private String label = "";
 48     private String url = "";
 49     private String containingModule = "";
 50     private String containingPackage = "";
 51     private String containingClass = "";
 52     private String holder = "";
 53     private String description = "";
 54     private boolean systemProperty;
 55 
 56     public void setLabel(String l) {
 57         label = l;
 58     }
 59 
 60     public String getLabel() {
 61         return label;
 62     }
 63 
 64     public void setUrl(String u) {
 65         url = u;
 66     }
 67 
 68     public String getUrl() {
 69         return url;
 70     }
 71 
 72     public void setContainingModule(String m) {
 73         containingModule = m;
 74     }
 75 
 76     public void setContainingPackage(String p) {
 77         containingPackage = p;
 78     }
 79 
 80     public void setContainingClass(String c) {
 81         containingClass = c;
 82     }
 83 
 84     public void setCategory(Category c) {
 85         category = c;
 86     }
 87 
 88     public void setHolder(String h) {
 89         holder = h;
 90     }
 91 
 92     public String getHolder() {
 93         return holder;
 94     }
 95 
 96     public void setDescription(String d) {
 97         description = d;
 98     }
 99 
100     public String getDescription() {
101         return description;
102     }
103 
104     public void setSystemProperty(boolean value) {
105         systemProperty = value;




106     }
107 
108     public boolean isSystemProperty() {
109         return systemProperty;
110     }
111 

112     public String toString() {
113         StringBuilder item = new StringBuilder("");

114         switch (category) {
115         case MODULES:
116             item.append("{")
117                     .append("\"l\":\"").append(label).append("\"")
118                     .append("}");
119             break;
120         case PACKAGES:
121             item.append("{");
122             if (!containingModule.isEmpty()) {
123                 item.append("\"m\":\"").append(containingModule).append("\",");
124             }
125             item.append("\"l\":\"").append(label).append("\"");
126             if (!url.isEmpty()) {
127                 item.append(",\"url\":\"").append(url).append("\"");
128             }
129             item.append("}");
130             break;
131         case TYPES:
132             item.append("{");
133             if (!containingPackage.isEmpty()) {
134                 item.append("\"p\":\"").append(containingPackage).append("\",");
135             }
136             item.append("\"l\":\"").append(label).append("\"");
137             if (!url.isEmpty()) {
138                 item.append(",\"url\":\"").append(url).append("\"");
139             }
140             item.append("}");
141             break;
142         case MEMBERS:
143             item.append("{")
144                     .append("\"p\":\"").append(containingPackage).append("\",")
145                     .append("\"c\":\"").append(containingClass).append("\",")
146                     .append("\"l\":\"").append(label).append("\"");
147             if (!url.isEmpty()) {
148                 item.append(",\"url\":\"").append(url).append("\"");
149             }
150             item.append("}");
151             break;
152         case SEARCH_TAGS:
153             item.append("{")
154                     .append("\"l\":\"").append(label).append("\",")
155                     .append("\"h\":\"").append(holder).append("\",");
156             if (!description.isEmpty()) {
157                 item.append("\"d\":\"").append(description).append("\",");
158             }
159             item.append("\"u\":\"").append(url).append("\"")
160                     .append("}");
161             break;
162         default:
163             throw new IllegalStateException("category not set");

164         }
165         return item.toString();
166     }
167 
168     /**
169      * Get the part of the label after the last dot, or whole label if no dots.
170      *
171      * @return the simple name
172      */
173     public String getSimpleName() {
174         return label.substring(label.lastIndexOf('.') + 1);
175     }
176 }
