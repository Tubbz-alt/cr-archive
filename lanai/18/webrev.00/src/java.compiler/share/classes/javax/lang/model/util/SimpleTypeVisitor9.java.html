26 package javax.lang.model.util;
27 
28 import javax.annotation.processing.SupportedSourceVersion;
29 import javax.lang.model.SourceVersion;
30 import javax.lang.model.type.IntersectionType;
31 import static javax.lang.model.SourceVersion.*;
32 
33 /**
34  * A simple visitor of types with default behavior appropriate for
35  * source versions {@link SourceVersion#RELEASE_9 RELEASE_9} through
36  * {@link SourceVersion#RELEASE_14 RELEASE_14}.
37  *
38  * Visit methods corresponding to {@code RELEASE_9} and earlier
39  * language constructs call {@link #defaultAction defaultAction},
40  * passing their arguments to {@code defaultAction}'s corresponding
41  * parameters.
42  *
43  * @apiNote
44  * Methods in this class may be overridden subject to their general
45  * contract.
46  *
47  * @param <R> the return type of this visitor's methods.  Use {@link
48  *            Void} for visitors that do not need to return results.
49  * @param <P> the type of the additional parameter to this visitor's
50  *            methods.  Use {@code Void} for visitors that do not need an
51  *            additional parameter.
52  *
53  * @see <a href="SimpleTypeVisitor6.html#note_for_subclasses">
54  * <strong>Compatibility note for subclasses</strong></a>
55  * @see SimpleTypeVisitor6
56  * @see SimpleTypeVisitor7
57  * @see SimpleTypeVisitor8
58  * @see SimpleTypeVisitor14
59  * @since 9
60  */
61 @SupportedSourceVersion(RELEASE_14)
62 public class SimpleTypeVisitor9<R, P> extends SimpleTypeVisitor8<R, P> {
63     /**
64      * Constructor for concrete subclasses; uses {@code null} for the
65      * default value.
66      */
67     protected SimpleTypeVisitor9(){
68         super(null);
69     }
70 
71     /**
72      * Constructor for concrete subclasses; uses the argument for the
73      * default value.
74      *
75      * @param defaultValue the value to assign to {@link #DEFAULT_VALUE}
76      */
77     protected SimpleTypeVisitor9(R defaultValue){
78         super(defaultValue);
79     }
80 }
