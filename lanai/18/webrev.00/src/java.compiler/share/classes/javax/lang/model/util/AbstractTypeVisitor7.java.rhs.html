26 package javax.lang.model.util;
27 
28 import javax.annotation.processing.SupportedSourceVersion;
29 import javax.lang.model.type.*;
30 
31 import static javax.lang.model.SourceVersion.*;
32 
33 /**
34  * A skeletal visitor of types with default behavior appropriate for
35  * the {@link javax.lang.model.SourceVersion#RELEASE_7 RELEASE_7}
36  * source version.
37  *

















38  * @param <R> the return type of this visitor's methods.  Use {@link
39  *            Void} for visitors that do not need to return results.
40  * @param <P> the type of the additional parameter to this visitor's
41  *            methods.  Use {@code Void} for visitors that do not need an
42  *            additional parameter.
43  *
44  * @see <a href="AbstractTypeVisitor6.html#note_for_subclasses">
45  * <strong>Compatibility note for subclasses</strong></a>
46  * @see AbstractTypeVisitor6
47  * @see AbstractTypeVisitor8
48  * @see AbstractTypeVisitor9
49  * @see AbstractTypeVisitor14
50  * @since 1.7
51  */
52 @SupportedSourceVersion(RELEASE_7)
53 public abstract class AbstractTypeVisitor7<R, P> extends AbstractTypeVisitor6<R, P> {
54     /**
55      * Constructor for concrete subclasses to call.
56      *
57      * @deprecated Release 7 is obsolete; update to a visitor for a newer
58      * release level.
59      */
60     @Deprecated(since="12")
61     protected AbstractTypeVisitor7() {
62         super();  // Superclass constructor deprecated too
63     }
64 
65     /**
66      * Visits a {@code UnionType} in a manner defined by a subclass.
67      *
68      * @param t  {@inheritDoc}
69      * @param p  {@inheritDoc}
70      * @return the result of the visit as defined by a subclass
71      */
72     @Override
73     public abstract R visitUnion(UnionType t, P p);
74 }
