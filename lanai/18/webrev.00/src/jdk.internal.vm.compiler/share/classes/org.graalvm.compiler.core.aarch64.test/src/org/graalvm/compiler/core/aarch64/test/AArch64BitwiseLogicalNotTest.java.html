 27 package org.graalvm.compiler.core.aarch64.test;
 28 
 29 import org.graalvm.compiler.lir.LIRInstruction;
 30 import org.graalvm.compiler.lir.aarch64.AArch64ArithmeticOp;
 31 import org.graalvm.compiler.lir.aarch64.AArch64ArithmeticOp.BinaryOp;
 32 import org.junit.Test;
 33 
 34 import java.util.function.Predicate;
 35 
 36 public class AArch64BitwiseLogicalNotTest extends AArch64MatchRuleTest {
 37     private static final String BIC = AArch64ArithmeticOp.BIC.name();
 38     private static final String ORN = AArch64ArithmeticOp.ORN.name();
 39     private static final String EON = AArch64ArithmeticOp.EON.name();
 40 
 41     private void test(String methodName, String opName, Object... args) {
 42         test(methodName, args);
 43         Predicate<LIRInstruction> predicate = op -> {
 44             if (op instanceof BinaryOp && op.name().equalsIgnoreCase(opName)) {
 45                 return true;
 46             }
 47             return false;
 48         };
 49         checkLIR(methodName, predicate, 1);
 50     }
 51 
 52     // Tests for and not.
 53     public int andNotInt1(int m, int n) {
 54         return n & (~m);
 55     }
 56 
 57     @Test
 58     public void testAndNotInt1() {
 59         test("andNotInt1", BIC, 5, 6);
 60     }
 61 
 62     public int andNotInt2(int m, int n) {
 63         return n & (m ^ 0XFFFFFFFF);
 64     }
 65 
 66     @Test
 67     public void testAndNotInt2() {
 68         test("andNotInt2", BIC, 325, -1);
 69     }
 70 
 71     public long andNotLong(long m, long n) {
 72         return m & (n ^ -1L);
 73     }
 74 
 75     @Test
 76     public void testAndNotLong() {
 77         test("andNotLong", BIC, 3L, 425L);
 78     }
 79 
 80     // Tests for or not.
 81     public int orNotInt(int m, int n) {
 82         return (n ^ 0XFFFFFFFF) | m;
 83     }
 84 
 85     @Test
 86     public void testOrNotInt() {
 87         test("orNotInt", ORN, -1, Integer.MAX_VALUE);
 88     }
 89 
 90     public long orNotLong(long m, long n) {
 91         return m | (~n);
 92     }
 93 
 94     @Test
 95     public void testOrNotLong() {
 96         test("orNotLong", ORN, 23L, -1L);
 97     }
 98 
 99     // Tests for xor not.
100     public int xorNotInt(int m, int n) {
101         return (~n) ^ m;
102     }
103 
104     @Test
105     public void testXorNotInt() {
106         test("xorNotInt", EON, 4132, 24);
107     }
108 
109     public long xorNotLong(long m, long n) {
110         return m ^ (~n);
111     }
112 
113     @Test
114     public void testXorNotLong() {
115         test("xorNotLong", EON, Long.MIN_VALUE, Long.MAX_VALUE);
116     }
117 
118     // Tests for not xor.
119     public int notXorInt1(int m, int n) {
120         return ~(m ^ n);
121     }
122 
123     @Test
124     public void testNotXorInt1() {
125         test("notXorInt1", EON, 235, 98);
126     }
127 
128     public int notXorInt2(int m, int n) {
129         return (m ^ n) ^ 0XFFFFFFFF;
130     }
131 
132     @Test
133     public void testNotXorInt2() {
134         test("notXorInt2", EON, 245, 34654);
135     }
136 
137     public long notXorLong(long m, long n) {
138         return ~(m ^ n);
139     }
140 
141     @Test
142     public void testNotXorLong() {
143         test("notXorLong", EON, 324L, 235L);
144     }
145 }
