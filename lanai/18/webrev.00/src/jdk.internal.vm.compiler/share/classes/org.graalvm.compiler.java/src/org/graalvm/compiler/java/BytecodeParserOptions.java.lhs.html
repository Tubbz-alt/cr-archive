25 package org.graalvm.compiler.java;
26 
27 import org.graalvm.compiler.options.Option;
28 import org.graalvm.compiler.options.OptionType;
29 import org.graalvm.compiler.options.OptionKey;
30 
31 /**
32  * Options related to {@link BytecodeParser}.
33  *
34  * Note: This must be a top level class to work around for
35  * <a href="https://bugs.eclipse.org/bugs/show_bug.cgi?id=477597">Eclipse bug 477597</a>.
36  */
37 public class BytecodeParserOptions {
38     // @formatter:off
39     @Option(help = "The trace level for the bytecode parser. A value of 1 enables instruction tracing " +
40                    "and any greater value emits a frame state trace just prior to each instruction trace." +
41                    "Instruction tracing output from multiple compiler threads will be interleaved so " +
42                    "use of this option make most sense for single threaded compilation. " +
43                    "The MethodFilter option can be used to refine tracing to selected methods.", type = OptionType.Debug)
44     public static final OptionKey<Integer> TraceBytecodeParserLevel = new OptionKey<>(0);
45 
46     @Option(help = "Inlines trivial methods during bytecode parsing.", type = OptionType.Expert)
47     public static final OptionKey<Boolean> InlineDuringParsing = new OptionKey<>(true);
48 
49     @Option(help = "Inlines partial intrinsic exits during bytecode parsing when possible. " +
50                    "A partial intrinsic exit is a call within an intrinsic to the method " +
51                    "being intrinsified and denotes semantics of the original method that " +
52                    "the intrinsic does not support.", type = OptionType.Expert)
53     public static final OptionKey<Boolean> InlinePartialIntrinsicExitDuringParsing = new OptionKey<>(true);
54 
55     @Option(help = "Inlines intrinsic methods during bytecode parsing.", type = OptionType.Expert)
56     public static final OptionKey<Boolean> InlineIntrinsicsDuringParsing = new OptionKey<>(true);
57 
58     @Option(help = "Traces inlining performed during bytecode parsing.", type = OptionType.Debug)
59     public static final OptionKey<Boolean> TraceInlineDuringParsing = new OptionKey<>(false);
60 
61     @Option(help = "Traces use of plugins during bytecode parsing.", type = OptionType.Debug)
62     public static final OptionKey<Boolean> TraceParserPlugins = new OptionKey<>(false);
63 
64     @Option(help = "Maximum depth when inlining during bytecode parsing.", type = OptionType.Debug)
65     public static final OptionKey<Integer> InlineDuringParsingMaxDepth = new OptionKey<>(10);
66 
67     @Option(help = "When creating info points hide the methods of the substitutions.", type = OptionType.Debug)
68     public static final OptionKey<Boolean> HideSubstitutionStates = new OptionKey<>(false);
69     // @formatter:on
70 }
