 219     /** Check kind and type of given tree against protokind and prototype.
 220      *  If check succeeds, store type in tree and return it.
 221      *  If check fails, store errType in tree and return it.
 222      *  No checks are performed if the prototype is a method type.
 223      *  It is not necessary in this case since we know that kind and type
 224      *  are correct.
 225      *
 226      *  @param tree     The tree whose kind and type is checked
 227      *  @param found    The computed type of the tree
 228      *  @param ownkind  The computed kind of the tree
 229      *  @param resultInfo  The expected result of the tree
 230      */
 231     Type check(final JCTree tree,
 232                final Type found,
 233                final KindSelector ownkind,
 234                final ResultInfo resultInfo) {
 235         InferenceContext inferenceContext = resultInfo.checkContext.inferenceContext();
 236         Type owntype;
 237         boolean shouldCheck = !found.hasTag(ERROR) &&
 238                 !resultInfo.pt.hasTag(METHOD) &&
 239                 !resultInfo.pt.hasTag(FORALL);
 240         if (shouldCheck && !ownkind.subset(resultInfo.pkind)) {
 241             log.error(tree.pos(),
 242                       Errors.UnexpectedType(resultInfo.pkind.kindNames(),
 243                                             ownkind.kindNames()));
 244             owntype = types.createErrorType(found);
 245         } else if (allowPoly && inferenceContext.free(found)) {
 246             //delay the check if there are inference variables in the found type
 247             //this means we are dealing with a partially inferred poly expression
 248             owntype = shouldCheck ? resultInfo.pt : found;
 249             if (resultInfo.checkMode.installPostInferenceHook()) {
 250                 inferenceContext.addFreeTypeListener(List.of(found),
 251                         instantiatedContext -> {
 252                             ResultInfo pendingResult =
 253                                     resultInfo.dup(inferenceContext.asInstType(resultInfo.pt));
 254                             check(tree, inferenceContext.asInstType(found), ownkind, pendingResult);
 255                         });
 256             }
 257         } else {
 258             owntype = shouldCheck ?
 259             resultInfo.check(tree, found) :
 260             found;
 261         }
 262         if (resultInfo.checkMode.updateTreeType()) {
 263             tree.type = owntype;
 264         }
 265         return owntype;
 266     }

 268     /** Is given blank final variable assignable, i.e. in a scope where it
 269      *  may be assigned to even though it is final?
 270      *  @param v      The blank final variable.
 271      *  @param env    The current environment.
 272      */
 273     boolean isAssignableAsBlankFinal(VarSymbol v, Env<AttrContext> env) {
 274         Symbol owner = env.info.scope.owner;
 275            // owner refers to the innermost variable, method or
 276            // initializer block declaration at this point.
 277         return
 278             v.owner == owner
 279             ||
 280             ((owner.name == names.init ||    // i.e. we are in a constructor
 281               owner.kind == VAR ||           // i.e. we are in a variable initializer
 282               (owner.flags() & BLOCK) != 0)  // i.e. we are in an initializer block
 283              &&
 284              v.owner == owner.owner
 285              &&
 286              ((v.flags() & STATIC) != 0) == Resolve.isStatic(env));
 287     }

 289     /** Check that variable can be assigned to.
 290      *  @param pos    The current source code position.
 291      *  @param v      The assigned variable
 292      *  @param base   If the variable is referred to in a Select, the part
 293      *                to the left of the `.', null otherwise.
 294      *  @param env    The current environment.
 295      */
 296     void checkAssignable(DiagnosticPosition pos, VarSymbol v, JCTree base, Env<AttrContext> env) {
 297         if (v.name == names._this) {
 298             log.error(pos, Errors.CantAssignValToThis);
 299         } else if ((v.flags() & FINAL) != 0 &&
 300             ((v.flags() & HASINIT) != 0
 301              ||
 302              !((base == null ||
 303                TreeInfo.isThisQualifier(base)) &&
 304                isAssignableAsBlankFinal(v, env)))) {
 305             if (v.isResourceVariable()) { //TWR resource
 306                 log.error(pos, Errors.TryResourceMayNotBeAssigned(v));
 307             } else if ((v.flags() & MATCH_BINDING) != 0) {
 308                 log.error(pos, Errors.PatternBindingMayNotBeAssigned(v));
 309             } else {
 310                 log.error(pos, Errors.CantAssignValToFinalVar(v));
 311             }
 312         }
 313     }

 364         private TreeVisitor<Symbol,Env<AttrContext>> identAttributer = new IdentAttributer();
 365         private class IdentAttributer extends SimpleTreeVisitor<Symbol,Env<AttrContext>> {
 366             @Override @DefinedBy(Api.COMPILER_TREE)
 367             public Symbol visitMemberSelect(MemberSelectTree node, Env<AttrContext> env) {
 368                 Symbol site = visit(node.getExpression(), env);
 369                 if (site.kind == ERR || site.kind == ABSENT_TYP || site.kind == HIDDEN)
 370                     return site;
 371                 Name name = (Name)node.getIdentifier();
 372                 if (site.kind == PCK) {
 373                     env.toplevel.packge = (PackageSymbol)site;
 374                     return rs.findIdentInPackage(null, env, (TypeSymbol)site, name,
 375                             KindSelector.TYP_PCK);
 376                 } else {
 377                     env.enclClass.sym = (ClassSymbol)site;
 378                     return rs.findMemberType(env, site.asType(), name, (TypeSymbol)site);
 379                 }
 380             }
 381 
 382             @Override @DefinedBy(Api.COMPILER_TREE)
 383             public Symbol visitIdentifier(IdentifierTree node, Env<AttrContext> env) {
 384                 return rs.findIdent(null, env, (Name)node.getName(), KindSelector.TYP_PCK);

 415     private Env<AttrContext> attribToTree(JCTree root, Env<AttrContext> env, JCTree tree, ResultInfo resultInfo) {
 416         breakTree = tree;
 417         JavaFileObject prev = log.useSource(env.toplevel.sourcefile);
 418         try {
 419             deferredAttr.attribSpeculative(root, env, resultInfo,
 420                     null, DeferredAttr.AttributionMode.ANALYZER,
 421                     argumentAttr.withLocalCacheContext());
 422         } catch (BreakAttr b) {
 423             return b.env;
 424         } catch (AssertionError ae) {
 425             if (ae.getCause() instanceof BreakAttr) {
 426                 return ((BreakAttr)(ae.getCause())).env;
 427             } else {
 428                 throw ae;
 429             }
 430         } finally {
 431             breakTree = null;
 432             log.useSource(prev);
 433         }
 434         return env;
 435     }

 647     /** Visitor method: attribute a tree, catching any completion failure
 648      *  exceptions. Return the tree's type.
 649      *
 650      *  @param tree    The tree to be visited.
 651      *  @param env     The environment visitor argument.
 652      *  @param resultInfo   The result info visitor argument.
 653      */
 654     Type attribTree(JCTree tree, Env<AttrContext> env, ResultInfo resultInfo) {
 655         Env<AttrContext> prevEnv = this.env;
 656         ResultInfo prevResult = this.resultInfo;
 657         try {
 658             this.env = env;
 659             this.resultInfo = resultInfo;
 660             if (resultInfo.needsArgumentAttr(tree)) {
 661                 result = argumentAttr.attribArg(tree, env);
 662             } else {
 663                 tree.accept(this);
 664             }
 665             matchBindings = matchBindingsComputer.finishBindings(tree,
 666                                                                  matchBindings);
 667             if (tree == breakTree &&
 668                     resultInfo.checkContext.deferredAttrContext().mode == AttrMode.CHECK) {
 669                 breakTreeFound(copyEnv(env));
 670             }
 671             return result;
 672         } catch (CompletionFailure ex) {
 673             tree.type = syms.errType;
 674             return chk.completionError(tree.pos(), ex);
 675         } finally {
 676             this.env = prevEnv;
 677             this.resultInfo = prevResult;
 678         }
 679     }

 867      *  @param env               The environment current at the reference.

 869      *  @param interfaceExpected true if only an interface is expected here.

 871     Type attribBase(JCTree tree,
 872                     Env<AttrContext> env,
 873                     boolean classExpected,
 874                     boolean interfaceExpected,
 875                     boolean checkExtensible) {
 876         Type t = tree.type != null ?
 877             tree.type :
 878             attribType(tree, env);
 879         return checkBase(t, tree, env, classExpected, interfaceExpected, checkExtensible);

 881     Type checkBase(Type t,
 882                    JCTree tree,
 883                    Env<AttrContext> env,
 884                    boolean classExpected,
 885                    boolean interfaceExpected,
 886                    boolean checkExtensible) {
 887         final DiagnosticPosition pos = tree.hasTag(TYPEAPPLY) ?
 888                 (((JCTypeApply) tree).clazz).pos() : tree.pos();
 889         if (t.tsym.isAnonymous()) {
 890             log.error(pos, Errors.CantInheritFromAnon);
 891             return types.createErrorType(t);
 892         }
 893         if (t.isErroneous())
 894             return t;
 895         if (t.hasTag(TYPEVAR) && !classExpected && !interfaceExpected) {
 896             // check that type variable is already visible
 897             if (t.getUpperBound() == null) {
 898                 log.error(pos, Errors.IllegalForwardRef);
 899                 return types.createErrorType(t);
 900             }
 901         } else {
 902             t = chk.checkClassType(pos, t, checkExtensible);
 903         }
 904         if (interfaceExpected && (t.tsym.flags() & INTERFACE) == 0) {
 905             log.error(pos, Errors.IntfExpectedHere);
 906             // return errType is necessary since otherwise there might
 907             // be undetected cycles which cause attribution to loop
 908             return types.createErrorType(t);
 909         } else if (checkExtensible &&
 910                    classExpected &&
 911                    (t.tsym.flags() & INTERFACE) != 0) {
 912             log.error(pos, Errors.NoIntfExpectedHere);
 913             return types.createErrorType(t);
 914         }
 915         if (checkExtensible &&
 916             ((t.tsym.flags() & FINAL) != 0)) {
 917             log.error(pos,
 918                       Errors.CantInheritFromFinal(t.tsym));
 919         }
 920         chk.checkNonCyclic(pos, t);
 921         return t;
 922     }

 931     public void visitClassDef(JCClassDecl tree) {
 932         Optional<ArgumentAttr.LocalCacheContext> localCacheContext =
 933                 Optional.ofNullable(env.info.attributionMode.isSpeculative ?
 934                         argumentAttr.withLocalCacheContext() : null);
 935         try {
 936             // Local and anonymous classes have not been entered yet, so we need to
 937             // do it now.
 938             if (env.info.scope.owner.kind.matches(KindSelector.VAL_MTH)) {
 939                 enter.classEnter(tree, env);
 940             } else {
 941                 // If this class declaration is part of a class level annotation,
 942                 // as in @MyAnno(new Object() {}) class MyClass {}, enter it in
 943                 // order to simplify later steps and allow for sensible error
 944                 // messages.
 945                 if (env.tree.hasTag(NEWCLASS) && TreeInfo.isInAnnotation(env, tree))
 946                     enter.classEnter(tree, env);
 947             }
 948 
 949             ClassSymbol c = tree.sym;
 950             if (c == null) {
 951                 // exit in case something drastic went wrong during enter.
 952                 result = null;
 953             } else {
 954                 // make sure class has been completed:
 955                 c.complete();
 956 
 957                 // If this class appears as an anonymous class
 958                 // in a superclass constructor call
 959                 // disable implicit outer instance from being passed.
 960                 // (This would be an illegal access to "this before super").
 961                 if (env.info.isSelfCall &&
 962                         env.tree.hasTag(NEWCLASS)) {
 963                     c.flags_field |= NOOUTERTHIS;
 964                 }
 965                 attribClass(tree.pos(), c);
 966                 result = tree.type = c.type;
 967             }
 968         } finally {
 969             localCacheContext.ifPresent(LocalCacheContext::leave);
 970         }
 971     }

 973     public void visitMethodDef(JCMethodDecl tree) {
 974         MethodSymbol m = tree.sym;
 975         boolean isDefaultMethod = (m.flags() & DEFAULT) != 0;
 976 
 977         Lint lint = env.info.lint.augment(m);
 978         Lint prevLint = chk.setLint(lint);
 979         MethodSymbol prevMethod = chk.setMethod(m);
 980         try {
 981             deferredLintHandler.flush(tree.pos());
 982             chk.checkDeprecatedAnnotation(tree.pos(), m);
 983 
 984 
 985             // Create a new environment with local scope
 986             // for attributing the method.
 987             Env<AttrContext> localEnv = memberEnter.methodEnv(tree, env);
 988             localEnv.info.lint = lint;
 989 
 990             attribStats(tree.typarams, localEnv);
 991 
 992             // If we override any other methods, check that we do so properly.
 993             // JLS ???
 994             if (m.isStatic()) {
 995                 chk.checkHideClashes(tree.pos(), env.enclClass.type, m);
 996             } else {
 997                 chk.checkOverrideClashes(tree.pos(), env.enclClass.type, m);
 998             }
 999             chk.checkOverride(env, tree, m);
1000 
1001             if (isDefaultMethod && types.overridesObjectMethod(m.enclClass(), m)) {
1002                 log.error(tree, Errors.DefaultOverridesObjectMember(m.name, Kinds.kindName(m.location()), m.location()));
1003             }
1004 
1005             // Enter all type parameters into the local method scope.
1006             for (List<JCTypeParameter> l = tree.typarams; l.nonEmpty(); l = l.tail)
1007                 localEnv.info.scope.enterIfAbsent(l.head.type.tsym);
1008 
1009             ClassSymbol owner = env.enclClass.sym;
1010             if ((owner.flags() & ANNOTATION) != 0 &&
1011                     (tree.params.nonEmpty() ||
1012                     tree.recvparam != null))
1013                 log.error(tree.params.nonEmpty() ?
1014                         tree.params.head.pos() :
1015                         tree.recvparam.pos(),
1016                         Errors.IntfAnnotationMembersCantHaveParams);
1017 
1018             // Attribute all value parameters.
1019             for (List<JCVariableDecl> l = tree.params; l.nonEmpty(); l = l.tail) {
1020                 attribStat(l.head, localEnv);
1021             }
1022 
1023             chk.checkVarargsMethodDecl(localEnv, tree);
1024 
1025             // Check that type parameters are well-formed.
1026             chk.validate(tree.typarams, localEnv);
1027 
1028             // Check that result type is well-formed.
1029             if (tree.restype != null && !tree.restype.type.hasTag(VOID))
1030                 chk.validate(tree.restype, localEnv);
1031 
1032             // Check that receiver type is well-formed.
1033             if (tree.recvparam != null) {
1034                 // Use a new environment to check the receiver parameter.
1035                 // Otherwise I get "might not have been initialized" errors.
1036                 // Is there a better way?
1037                 Env<AttrContext> newEnv = memberEnter.methodEnv(tree, env);
1038                 attribType(tree.recvparam, newEnv);
1039                 chk.validate(tree.recvparam, newEnv);
1040             }
1041 
1042             if (env.enclClass.sym.isRecord() && tree.sym.owner.kind == TYP) {
1043                 // lets find if this method is an accessor
1044                 Optional<? extends RecordComponent> recordComponent = env.enclClass.sym.getRecordComponents().stream()
1045                         .filter(rc -> rc.accessor == tree.sym && (rc.accessor.flags_field & GENERATED_MEMBER) == 0).findFirst();
1046                 if (recordComponent.isPresent()) {
1047                     // the method is a user defined accessor lets check that everything is fine
1048                     if (!tree.sym.isPublic()) {
1049                         log.error(tree, Errors.InvalidAccessorMethodInRecord(env.enclClass.sym, Fragments.MethodMustBePublic));
1050                     }
1051                     if (!types.isSameType(tree.sym.type.getReturnType(), recordComponent.get().type)) {
1052                         log.error(tree, Errors.InvalidAccessorMethodInRecord(env.enclClass.sym,
1053                                 Fragments.AccessorReturnTypeDoesntMatch(tree.sym, recordComponent.get())));
1054                     }
1055                     if (tree.sym.type.asMethodType().thrown != null && !tree.sym.type.asMethodType().thrown.isEmpty()) {
1056                         log.error(tree,
1057                                 Errors.InvalidAccessorMethodInRecord(env.enclClass.sym, Fragments.AccessorMethodCantThrowException));
1058                     }
1059                     if (!tree.typarams.isEmpty()) {
1060                         log.error(tree,
1061                                 Errors.InvalidAccessorMethodInRecord(env.enclClass.sym, Fragments.AccessorMethodMustNotBeGeneric));
1062                     }
1063                     if (tree.sym.isStatic()) {
1064                         log.error(tree,
1065                                 Errors.InvalidAccessorMethodInRecord(env.enclClass.sym, Fragments.AccessorMethodMustNotBeStatic));
1066                     }
1067                 }
1068 
1069                 if (tree.name == names.init) {
1070                     // if this a constructor other than the canonical one
1071                     if ((tree.sym.flags_field & RECORD) == 0) {
1072                         JCMethodInvocation app = TreeInfo.firstConstructorCall(tree);
1073                         if (app == null ||
1074                                 TreeInfo.name(app.meth) != names._this ||
1075                                 !checkFirstConstructorStat(app, tree, false)) {
1076                             log.error(tree, Errors.FirstStatementMustBeCallToAnotherConstructor);
1077                         }
1078                     } else {
1079                         // but if it is the canonical:
1080 
1081                         // if user generated, then it shouldn't explicitly invoke any other constructor
1082                         if ((tree.sym.flags_field & GENERATEDCONSTR) == 0) {
1083                             JCMethodInvocation app = TreeInfo.firstConstructorCall(tree);
1084                             if (app != null &&
1085                                     (TreeInfo.name(app.meth) == names._this ||
1086                                             TreeInfo.name(app.meth) == names._super) &&
1087                                     checkFirstConstructorStat(app, tree, false)) {
1088                                 log.error(tree, Errors.InvalidCanonicalConstructorInRecord(
1089                                         Fragments.Canonical, tree.sym,
1090                                         Fragments.CanonicalMustNotContainExplicitConstructorInvocation));
1091                             }
1092                         }
1093 
1094                         // also we want to check that no type variables have been defined
1095                         if (!tree.typarams.isEmpty()) {
1096                             log.error(tree, Errors.InvalidCanonicalConstructorInRecord(
1097                                     Fragments.Canonical, tree.sym, Fragments.CanonicalMustNotDeclareTypeVariables));
1098                         }
1099 
1100                         /* and now we need to check that the constructor's arguments are exactly the same as those of the
1101                          * record components
1102                          */
1103                         List<Type> recordComponentTypes = TreeInfo.recordFields(env.enclClass).map(vd -> vd.sym.type);
1104                         for (JCVariableDecl param: tree.params) {
1105                             if (!types.isSameType(param.type, recordComponentTypes.head)) {
1106                                 log.error(param, Errors.InvalidCanonicalConstructorInRecord(
1107                                         Fragments.Canonical, tree.sym, Fragments.TypeMustBeIdenticalToCorrespondingRecordComponentType));
1108                             }
1109                             recordComponentTypes = recordComponentTypes.tail;
1110                         }
1111                     }
1112                 }
1113             }
1114 
1115             // annotation method checks
1116             if ((owner.flags() & ANNOTATION) != 0) {
1117                 // annotation method cannot have throws clause
1118                 if (tree.thrown.nonEmpty()) {
1119                     log.error(tree.thrown.head.pos(),
1120                               Errors.ThrowsNotAllowedInIntfAnnotation);
1121                 }
1122                 // annotation method cannot declare type-parameters
1123                 if (tree.typarams.nonEmpty()) {
1124                     log.error(tree.typarams.head.pos(),
1125                               Errors.IntfAnnotationMembersCantHaveTypeParams);
1126                 }
1127                 // validate annotation method's return type (could be an annotation type)
1128                 chk.validateAnnotationType(tree.restype);
1129                 // ensure that annotation method does not clash with members of Object/Annotation
1130                 chk.validateAnnotationMethod(tree.pos(), m);
1131             }
1132 
1133             for (List<JCExpression> l = tree.thrown; l.nonEmpty(); l = l.tail)
1134                 chk.checkType(l.head.pos(), l.head.type, syms.throwableType);
1135 
1136             if (tree.body == null) {
1137                 // Empty bodies are only allowed for
1138                 // abstract, native, or interface methods, or for methods
1139                 // in a retrofit signature class.
1140                 if (tree.defaultValue != null) {
1141                     if ((owner.flags() & ANNOTATION) == 0)
1142                         log.error(tree.pos(),
1143                                   Errors.DefaultAllowedInIntfAnnotationMember);
1144                 }
1145                 if (isDefaultMethod || (tree.sym.flags() & (ABSTRACT | NATIVE)) == 0)
1146                     log.error(tree.pos(), Errors.MissingMethBodyOrDeclAbstract);
1147             } else if ((tree.sym.flags() & (ABSTRACT|DEFAULT|PRIVATE)) == ABSTRACT) {
1148                 if ((owner.flags() & INTERFACE) != 0) {
1149                     log.error(tree.body.pos(), Errors.IntfMethCantHaveBody);
1150                 } else {
1151                     log.error(tree.pos(), Errors.AbstractMethCantHaveBody);
1152                 }
1153             } else if ((tree.mods.flags & NATIVE) != 0) {
1154                 log.error(tree.pos(), Errors.NativeMethCantHaveBody);
1155             } else {
1156                 // Add an implicit super() call unless an explicit call to
1157                 // super(...) or this(...) is given
1158                 // or we are compiling class java.lang.Object.
1159                 if (tree.name == names.init && owner.type != syms.objectType) {
1160                     JCBlock body = tree.body;
1161                     if (body.stats.isEmpty() ||
1162                             TreeInfo.getConstructorInvocationName(body.stats, names) == names.empty) {
1163                         JCStatement supCall = make.at(body.pos).Exec(make.Apply(List.nil(),
1164                                 make.Ident(names._super), make.Idents(List.nil())));
1165                         body.stats = body.stats.prepend(supCall);
1166                     } else if ((env.enclClass.sym.flags() & ENUM) != 0 &&
1167                             (tree.mods.flags & GENERATEDCONSTR) == 0 &&
1168                             TreeInfo.isSuperCall(body.stats.head)) {
1169                         // enum constructors are not allowed to call super
1170                         // directly, so make sure there aren't any super calls
1171                         // in enum constructors, except in the compiler
1172                         // generated one.
1173                         log.error(tree.body.stats.head.pos(),
1174                                   Errors.CallToSuperNotAllowedInEnumCtor(env.enclClass.sym));
1175                     }
1176                     if (env.enclClass.sym.isRecord() && (tree.sym.flags_field & RECORD) != 0) { // we are seeing the canonical constructor
1177                         List<Name> recordComponentNames = TreeInfo.recordFields(env.enclClass).map(vd -> vd.sym.name);
1178                         List<Name> initParamNames = tree.sym.params.map(p -> p.name);
1179                         if (!initParamNames.equals(recordComponentNames)) {
1180                             log.error(tree, Errors.InvalidCanonicalConstructorInRecord(
1181                                     Fragments.Canonical, env.enclClass.sym, Fragments.CanonicalWithNameMismatch));
1182                         }
1183                         if (!tree.sym.isPublic()) {
1184                             log.error(tree, Errors.InvalidCanonicalConstructorInRecord(
1185                                     TreeInfo.isCompactConstructor(tree) ? Fragments.Compact : Fragments.Canonical,
1186                                     env.enclClass.sym, Fragments.CanonicalConstructorMustBePublic));
1187                         }
1188                         if (tree.sym.type.asMethodType().thrown != null && !tree.sym.type.asMethodType().thrown.isEmpty()) {
1189                             log.error(tree,
1190                                     Errors.InvalidCanonicalConstructorInRecord(
1191                                             TreeInfo.isCompactConstructor(tree) ? Fragments.Compact : Fragments.Canonical,
1192                                             env.enclClass.sym, Fragments.ThrowsClauseNotAllowedForCanonicalConstructor));


1193                         }
1194                     }
1195                 }
1196 
1197                 // Attribute all type annotations in the body
1198                 annotate.queueScanTreeAndTypeAnnotate(tree.body, localEnv, m, null);
1199                 annotate.flush();
1200 
1201                 // Attribute method body.
1202                 attribStat(tree.body, localEnv);
1203             }
1204 
1205             localEnv.info.scope.leave();
1206             result = tree.type = m.type;
1207         } finally {
1208             chk.setLint(prevLint);
1209             chk.setMethod(prevMethod);
1210         }
1211     }

1213     public void visitVarDef(JCVariableDecl tree) {
1214         // Local variables have not been entered yet, so we need to do it now:
1215         if (env.info.scope.owner.kind == MTH || env.info.scope.owner.kind == VAR) {
1216             if (tree.sym != null) {
1217                 // parameters have already been entered
1218                 env.info.scope.enter(tree.sym);
1219             } else {
1220                 if (tree.isImplicitlyTyped() && (tree.getModifiers().flags & PARAMETER) == 0) {
1221                     if (tree.init == null) {
1222                         //cannot use 'var' without initializer
1223                         log.error(tree, Errors.CantInferLocalVarType(tree.name, Fragments.LocalMissingInit));
1224                         tree.vartype = make.Erroneous();
1225                     } else {
1226                         Fragment msg = canInferLocalVarType(tree);
1227                         if (msg != null) {
1228                             //cannot use 'var' with initializer which require an explicit target
1229                             //(e.g. lambda, method reference, array initializer).
1230                             log.error(tree, Errors.CantInferLocalVarType(tree.name, msg));
1231                             tree.vartype = make.Erroneous();
1232                         }
1233                     }
1234                 }
1235                 try {
1236                     annotate.blockAnnotations();
1237                     memberEnter.memberEnter(tree, env);
1238                 } finally {
1239                     annotate.unblockAnnotations();
1240                 }
1241             }
1242         } else {
1243             if (tree.init != null) {
1244                 // Field initializer expression need to be entered.
1245                 annotate.queueScanTreeAndTypeAnnotate(tree.init, env, tree.sym, tree.pos());
1246                 annotate.flush();
1247             }
1248         }
1249 
1250         VarSymbol v = tree.sym;
1251         Lint lint = env.info.lint.augment(v);
1252         Lint prevLint = chk.setLint(lint);
1253 
1254         // Check that the variable's declared type is well-formed.
1255         boolean isImplicitLambdaParameter = env.tree.hasTag(LAMBDA) &&
1256                 ((JCLambda)env.tree).paramKind == JCLambda.ParameterKind.IMPLICIT &&
1257                 (tree.sym.flags() & PARAMETER) != 0;
1258         chk.validate(tree.vartype, env, !isImplicitLambdaParameter && !tree.isImplicitlyTyped());
1259 
1260         try {
1261             v.getConstValue(); // ensure compile-time constant initializer is evaluated
1262             deferredLintHandler.flush(tree.pos());
1263             chk.checkDeprecatedAnnotation(tree.pos(), v);
1264 
1265             if (tree.init != null) {
1266                 if ((v.flags_field & FINAL) == 0 ||
1267                     !memberEnter.needsLazyConstValue(tree.init)) {
1268                     // Not a compile-time constant
1269                     // Attribute initializer in a new environment
1270                     // with the declared variable as owner.
1271                     // Check that initializer conforms to variable's declared type.
1272                     Env<AttrContext> initEnv = memberEnter.initEnv(tree, env);
1273                     initEnv.info.lint = lint;
1274                     // In order to catch self-references, we set the variable's
1275                     // declaration position to maximal possible value, effectively
1276                     // marking the variable as undefined.
1277                     initEnv.info.enclVar = v;
1278                     attribExpr(tree.init, initEnv, v.type);
1279                     if (tree.isImplicitlyTyped()) {
1280                         //fixup local variable type
1281                         v.type = chk.checkLocalVarType(tree, tree.init.type.baseType(), tree.name);
1282                     }
1283                 }
1284                 if (tree.isImplicitlyTyped()) {
1285                     setSyntheticVariableType(tree, v.type);
1286                 }
1287             }
1288             result = tree.type = v.type;
1289             if (env.enclClass.sym.isRecord() && tree.sym.owner.kind == TYP && !v.isStatic()) {
1290                 if (isNonArgsMethodInObject(v.name)) {
1291                     log.error(tree, Errors.IllegalRecordComponentName(v));
1292                 }
1293             }
1294         }
1295         finally {
1296             chk.setLint(prevLint);
1297         }
1298     }

1611     private void handleSwitch(JCTree switchTree,
1612                               JCExpression selector,
1613                               List<JCCase> cases,
1614                               BiConsumer<JCCase, Env<AttrContext>> attribCase) {
1615         Type seltype = attribExpr(selector, env);
1616 
1617         Env<AttrContext> switchEnv =
1618             env.dup(switchTree, env.info.dup(env.info.scope.dup()));
1619 
1620         try {
1621             boolean enumSwitch = (seltype.tsym.flags() & Flags.ENUM) != 0;
1622             boolean stringSwitch = types.isSameType(seltype, syms.stringType);
1623             if (!enumSwitch && !stringSwitch)
1624                 seltype = chk.checkType(selector.pos(), seltype, syms.intType);
1625 
1626             // Attribute all cases and
1627             // check that there are no duplicate case labels or default clauses.
1628             Set<Object> labels = new HashSet<>(); // The set of case labels.
1629             boolean hasDefault = false;      // Is there a default label?
1630             CaseTree.CaseKind caseKind = null;
1631             boolean wasError = false;
1632             for (List<JCCase> l = cases; l.nonEmpty(); l = l.tail) {
1633                 JCCase c = l.head;
1634                 if (caseKind == null) {
1635                     caseKind = c.caseKind;
1636                 } else if (caseKind != c.caseKind && !wasError) {
1637                     log.error(c.pos(),
1638                               Errors.SwitchMixingCaseTypes);
1639                     wasError = true;
1640                 }
1641                 if (c.getExpressions().nonEmpty()) {
1642                     for (JCExpression pat : c.getExpressions()) {
1643                         if (TreeInfo.isNull(pat)) {
1644                             log.error(pat.pos(),
1645                                       Errors.SwitchNullNotAllowed);
1646                         } else if (enumSwitch) {
1647                             Symbol sym = enumConstant(pat, seltype);
1648                             if (sym == null) {
1649                                 log.error(pat.pos(), Errors.EnumLabelMustBeUnqualifiedEnum);
1650                             } else if (!labels.add(sym)) {
1651                                 log.error(c.pos(), Errors.DuplicateCaseLabel);
1652                             }
1653                         } else {
1654                             Type pattype = attribExpr(pat, switchEnv, seltype);
1655                             if (!pattype.hasTag(ERROR)) {
1656                                 if (pattype.constValue() == null) {
1657                                     log.error(pat.pos(),
1658                                               (stringSwitch ? Errors.StringConstReq : Errors.ConstExprReq));
1659                                 } else if (!labels.add(pattype.constValue())) {
1660                                     log.error(c.pos(), Errors.DuplicateCaseLabel);
1661                                 }
1662                             }
1663                         }
1664                     }
1665                 } else if (hasDefault) {
1666                     log.error(c.pos(), Errors.DuplicateDefaultLabel);
1667                 } else {
1668                     hasDefault = true;
1669                 }
1670                 Env<AttrContext> caseEnv =
1671                     switchEnv.dup(c, env.info.dup(switchEnv.info.scope.dup()));
1672                 try {
1673                     attribCase.accept(c, caseEnv);
1674                 } finally {
1675                     caseEnv.info.scope.leave();
1676                 }
1677                 addVars(c.stats, switchEnv.info.scope);
1678             }
1679         } finally {
1680             switchEnv.info.scope.leave();
1681         }
1682     }

1718         // Create a new local environment with a local
1719         Env<AttrContext> localEnv = env.dup(tree, env.info.dup(env.info.scope.dup()));
1720         try {
1721             boolean isTryWithResource = tree.resources.nonEmpty();
1722             // Create a nested environment for attributing the try block if needed
1723             Env<AttrContext> tryEnv = isTryWithResource ?
1724                 env.dup(tree, localEnv.info.dup(localEnv.info.scope.dup())) :
1725                 localEnv;
1726             try {
1727                 // Attribute resource declarations
1728                 for (JCTree resource : tree.resources) {
1729                     CheckContext twrContext = new Check.NestedCheckContext(resultInfo.checkContext) {
1730                         @Override
1731                         public void report(DiagnosticPosition pos, JCDiagnostic details) {
1732                             chk.basicHandler.report(pos, diags.fragment(Fragments.TryNotApplicableToType(details)));
1733                         }
1734                     };
1735                     ResultInfo twrResult =
1736                         new ResultInfo(KindSelector.VAR,
1737                                        syms.autoCloseableType,
1738                                        twrContext);
1739                     if (resource.hasTag(VARDEF)) {
1740                         attribStat(resource, tryEnv);
1741                         twrResult.check(resource, resource.type);
1742 
1743                         //check that resource type cannot throw InterruptedException
1744                         checkAutoCloseable(resource.pos(), localEnv, resource.type);
1745 
1746                         VarSymbol var = ((JCVariableDecl) resource).sym;
1747                         var.setData(ElementKind.RESOURCE_VARIABLE);
1748                     } else {
1749                         attribTree(resource, tryEnv, twrResult);
1750                     }
1751                 }
1752                 // Attribute body
1753                 attribStat(tree.body, tryEnv);
1754             } finally {
1755                 if (isTryWithResource)
1756                     tryEnv.info.scope.leave();
1757             }
1758 
1759             // Attribute catch clauses
1760             for (List<JCCatch> l = tree.catchers; l.nonEmpty(); l = l.tail) {
1761                 JCCatch c = l.head;
1762                 Env<AttrContext> catchEnv =
1763                     localEnv.dup(c, localEnv.info.dup(localEnv.info.scope.dup()));
1764                 try {
1765                     Type ctype = attribStat(c.param, catchEnv);
1766                     if (TreeInfo.isMultiCatch(c)) {
1767                         //multi-catch parameter is implicitly marked as final
1768                         c.param.sym.flags_field |= FINAL | UNION;
1769                     }
1770                     if (c.param.sym.kind == VAR) {
1771                         c.param.sym.setData(ElementKind.EXCEPTION_PARAMETER);
1772                     }
1773                     chk.checkType(c.param.vartype.pos(),
1774                                   chk.checkClassType(c.param.vartype.pos(), ctype),
1775                                   syms.throwableType);
1776                     attribStat(c.body, catchEnv);
1777                 } finally {
1778                     catchEnv.info.scope.leave();
1779                 }
1780             }
1781 
1782             // Attribute finalizer
1783             if (tree.finalizer != null) attribStat(tree.finalizer, localEnv);
1784             result = null;
1785         }
1786         finally {
1787             localEnv.info.scope.leave();
1788         }

1936         /** Compute the type of a conditional expression, after
1937          *  checking that it exists.  See JLS 15.25. Does not take into
1938          *  account the special case where condition and both arms
1939          *  are constants.
1940          *
1941          *  @param pos      The source position to be used for error
1942          *                  diagnostics.
1943          *  @param thentype The type of the expression's then-part.
1944          *  @param elsetype The type of the expression's else-part.
1945          */
1946         Type condType(List<DiagnosticPosition> positions, List<Type> condTypes) {
1947             if (condTypes.isEmpty()) {
1948                 return syms.objectType; //TODO: how to handle?
1949             }
1950             Type first = condTypes.head;
1951             // If same type, that is the result
1952             if (condTypes.tail.stream().allMatch(t -> types.isSameType(first, t)))
1953                 return first.baseType();
1954 
1955             List<Type> unboxedTypes = condTypes.stream()
1956                                                .map(t -> t.isPrimitive() ? t : types.unboxedType(t))
1957                                                .collect(List.collector());
1958 
1959             // Otherwise, if both arms can be converted to a numeric
1960             // type, return the least numeric type that fits both arms
1961             // (i.e. return larger of the two, or return int if one
1962             // arm is short, the other is char).
1963             if (unboxedTypes.stream().allMatch(t -> t.isPrimitive())) {
1964                 // If one arm has an integer subrange type (i.e., byte,
1965                 // short, or char), and the other is an integer constant
1966                 // that fits into the subrange, return the subrange type.
1967                 for (Type type : unboxedTypes) {
1968                     if (!type.getTag().isStrictSubRangeOf(INT)) {
1969                         continue;
1970                     }
1971                     if (unboxedTypes.stream().filter(t -> t != type).allMatch(t -> t.hasTag(INT) && types.isAssignable(t, type)))
1972                         return type.baseType();
1973                 }
1974 
1975                 for (TypeTag tag : primitiveTags) {
1976                     Type candidate = syms.typeOfTag[tag.ordinal()];
1977                     if (unboxedTypes.stream().allMatch(t -> types.isSubtype(t, candidate))) {
1978                         return candidate;
1979                     }
1980                 }
1981             }
1982 
1983             // Those were all the cases that could result in a primitive
1984             condTypes = condTypes.stream()
1985                                  .map(t -> t.isPrimitive() ? types.boxedClass(t).type : t)
1986                                  .collect(List.collector());
1987 
1988             for (Type type : condTypes) {
1989                 if (condTypes.stream().filter(t -> t != type).allMatch(t -> types.isAssignable(t, type)))
1990                     return type.baseType();
1991             }
1992 
1993             Iterator<DiagnosticPosition> posIt = positions.iterator();
1994 
1995             condTypes = condTypes.stream()
1996                                  .map(t -> chk.checkNonVoid(posIt.next(), t))
1997                                  .collect(List.collector());
1998 
1999             // both are known to be reference types.  The result is
2000             // lub(thentype,elsetype). This cannot fail, as it will
2001             // always be possible to infer "Object" if nothing better.
2002             return types.lub(condTypes.stream().map(t -> t.baseType()).collect(List.collector()));
2003         }

2146          *  not the tree representing the labelled statement itself.
2147          *
2148          *  @param tag     The tag of the jump statement. This is either
2149          *                 Tree.BREAK or Tree.CONTINUE.
2150          *  @param label   The label of the jump statement, or null if no
2151          *                 label is given.
2152          *  @param env     The environment current at the jump statement.
2153          */
2154         private Pair<JCTree, JCDiagnostic.Error> findJumpTargetNoError(JCTree.Tag tag,
2155                                                                        Name label,
2156                                                                        Env<AttrContext> env) {
2157             // Search environments outwards from the point of jump.
2158             Env<AttrContext> env1 = env;
2159             JCDiagnostic.Error pendingError = null;
2160             LOOP:
2161             while (env1 != null) {
2162                 switch (env1.tree.getTag()) {
2163                     case LABELLED:
2164                         JCLabeledStatement labelled = (JCLabeledStatement)env1.tree;
2165                         if (label == labelled.label) {
2166                             // If jump is a continue, check that target is a loop.
2167                             if (tag == CONTINUE) {
2168                                 if (!labelled.body.hasTag(DOLOOP) &&
2169                                         !labelled.body.hasTag(WHILELOOP) &&
2170                                         !labelled.body.hasTag(FORLOOP) &&
2171                                         !labelled.body.hasTag(FOREACHLOOP)) {
2172                                     pendingError = Errors.NotLoopLabel(label);
2173                                 }
2174                                 // Found labelled statement target, now go inwards
2175                                 // to next non-labelled tree.
2176                                 return Pair.of(TreeInfo.referencedStatement(labelled), pendingError);
2177                             } else {
2178                                 return Pair.of(labelled, pendingError);
2179                             }
2180                         }
2181                         break;
2182                     case DOLOOP:
2183                     case WHILELOOP:
2184                     case FORLOOP:
2185                     case FOREACHLOOP:
2186                         if (label == null) return Pair.of(env1.tree, pendingError);
2187                         break;
2188                     case SWITCH:
2189                         if (label == null && tag == BREAK) return Pair.of(env1.tree, null);
2190                         break;
2191                     case SWITCH_EXPRESSION:
2192                         if (tag == YIELD) {
2193                             return Pair.of(env1.tree, null);
2194                         } else if (tag == BREAK) {
2195                             pendingError = Errors.BreakOutsideSwitchExpression;
2196                         } else {
2197                             pendingError = Errors.ContinueOutsideSwitchExpression;
2198                         }
2199                         break;
2200                     case LAMBDA:
2201                     case METHODDEF:
2202                     case CLASSDEF:
2203                         break LOOP;
2204                     default:
2205                 }
2206                 env1 = env1.next;
2207             }
2208             if (label != null)
2209                 return Pair.of(null, Errors.UndefLabel(label));
2210             else if (pendingError != null)
2211                 return Pair.of(null, pendingError);
2212             else if (tag == CONTINUE)
2213                 return Pair.of(null, Errors.ContOutsideLoop);
2214             else
2215                 return Pair.of(null, Errors.BreakOutsideSwitchLoop);
2216         }

2686                                                    KindSelector pkind) {

2688             // In this case, the class instance creation
2689             // expression
2690             //
2691             //    E.new <typeargs1>C<typargs2>(args) { ... }
2692             //
2693             // is represented internally as
2694             //
2695             //    E . new <typeargs1>C<typargs2>(args) ( class <empty-name> { ... } )  .
2696             //
2697             // This expression is then *transformed* as follows:
2698             //
2699             // (1) add an extends or implements clause
2700             // (2) add a constructor.
2701             //
2702             // For instance, if C is a class, and ET is the type of E,
2703             // the expression
2704             //
2705             //    E.new <typeargs1>C<typargs2>(args) { ... }
2706             //
2707             // is translated to (where X is a fresh name and typarams is the
2708             // parameter list of the super constructor):
2709             //
2710             //   new <typeargs1>X(<*nullchk*>E, args) where
2711             //     X extends C<typargs2> {
2712             //       <typarams> X(ET e, args) {
2713             //         e.<typeargs1>super(args)
2714             //       }
2715             //       ...
2716             //     }
2717             InferenceContext inferenceContext = resultInfo.checkContext.inferenceContext();
2718             final boolean isDiamond = TreeInfo.isDiamond(tree);
2719             if (isDiamond
2720                     && ((tree.constructorType != null && inferenceContext.free(tree.constructorType))
2721                     || (tree.clazz.type != null && inferenceContext.free(tree.clazz.type)))) {
2722                 final ResultInfo resultInfoForClassDefinition = this.resultInfo;
2723                 inferenceContext.addFreeTypeListener(List.of(tree.constructorType, tree.clazz.type),
2724                         instantiatedContext -> {
2725                             tree.constructorType = instantiatedContext.asInstType(tree.constructorType);
2726                             tree.clazz.type = clazz.type = instantiatedContext.asInstType(clazz.type);
2727                             ResultInfo prevResult = this.resultInfo;

2729                                 this.resultInfo = resultInfoForClassDefinition;
2730                                 visitAnonymousClassDefinition(tree, clazz, clazz.type, cdef,

2732                             } finally {
2733                                 this.resultInfo = prevResult;

2735                         });

2738                     List<Type> invalidDiamondArgs = chk.checkDiamondDenotable((ClassType)clazztype);
2739                     if (!clazztype.isErroneous() && invalidDiamondArgs.nonEmpty()) {

2745                                         invalidDiamondArgs.size() > 1 ?

2758                         clazztype.isErroneous() && !clazztype.getOriginalType().hasTag(NONE) &&
2759                         clazztype.getOriginalType().tsym.isInterface();

2763                 } else {

2765                 }

2768                     isSerializable(clazztype)) {
2769                     localEnv.info.isSerializable = true;
2770                 }
2771

2773

2775                 // If an outer instance is given,
2776                 // prefix it to the constructor arguments

3610     boolean checkExConstraints(
3611             List<Type> thrownByFuncExpr,
3612             List<Type> thrownAtFuncType,
3613             InferenceContext inferenceContext) {
3614         /** 18.2.5: Otherwise, let E1, ..., En be the types in the function type's throws clause that
3615          *  are not proper types
3616          */
3617         List<Type> nonProperList = thrownAtFuncType.stream()
3618                 .filter(e -> inferenceContext.free(e)).collect(List.collector());
3619         List<Type> properList = thrownAtFuncType.diff(nonProperList);
3620 
3621         /** Let X1,...,Xm be the checked exception types that the lambda body can throw or
3622          *  in the throws clause of the invocation type of the method reference's compile-time
3623          *  declaration
3624          */
3625         List<Type> checkedList = thrownByFuncExpr.stream()
3626                 .filter(e -> chk.isChecked(e)).collect(List.collector());
3627 
3628         /** If n = 0 (the function type's throws clause consists only of proper types), then
3629          *  if there exists some i (1 <= i <= m) such that Xi is not a subtype of any proper type
3630          *  in the throws clause, the constraint reduces to false; otherwise, the constraint
3631          *  reduces to true
3632          */
3633         ListBuffer<Type> uncaughtByProperTypes = new ListBuffer<>();
3634         for (Type checked : checkedList) {
3635             boolean isSubtype = false;
3636             for (Type proper : properList) {
3637                 if (types.isSubtype(checked, proper)) {
3638                     isSubtype = true;
3639                     break;
3640                 }
3641             }
3642             if (!isSubtype) {
3643                 uncaughtByProperTypes.add(checked);
3644             }
3645         }
3646 
3647         if (nonProperList.isEmpty() && !uncaughtByProperTypes.isEmpty()) {
3648             return false;
3649         }
3650 
3651         /** If n > 0, the constraint reduces to a set of subtyping constraints:
3652          *  for all i (1 <= i <= m), if Xi is not a subtype of any proper type in the
3653          *  throws clause, then the constraints include, for all j (1 <= j <= n), <Xi <: Ej>
3654          */
3655         List<Type> nonProperAsUndet = inferenceContext.asUndetVars(nonProperList);
3656         uncaughtByProperTypes.forEach(checkedEx -> {
3657             nonProperAsUndet.forEach(nonProper -> {
3658                 types.isSubtype(checkedEx, nonProper);
3659             });
3660         });
3661 
3662         /** In addition, for all j (1 <= j <= n), the constraint reduces to the bound throws Ej
3663          */
3664         nonProperAsUndet.stream()
3665                 .filter(t -> t.hasTag(UNDETVAR))
3666                 .forEach(t -> ((UndetVar)t).setThrow());
3667         return true;
3668     }

4494             // JLS:
4495             //

4512                 // subclass thereof) declaring v.  If not, this

4514                 if (v.owner != enclClass && !types.isSubtype(enclClass.type, v.owner.type))
4515                     return;

4518                 // the reference is OK.
4519                 if (!Resolve.isInitializer(env))
4520                     return;

4795     Type checkIntersection(JCTree tree, List<JCExpression> bounds) {
4796         Set<Type> boundSet = new HashSet<>();
4797         if (bounds.nonEmpty()) {
4798             // accept class or interface or typevar as first bound.
4799             bounds.head.type = checkBase(bounds.head.type, bounds.head, env, false, false, false);
4800             boundSet.add(types.erasure(bounds.head.type));
4801             if (bounds.head.type.isErroneous()) {
4802                 return bounds.head.type;
4803             }
4804             else if (bounds.head.type.hasTag(TYPEVAR)) {
4805                 // if first bound was a typevar, do not accept further bounds.
4806                 if (bounds.tail.nonEmpty()) {
4807                     log.error(bounds.tail.head.pos(),
4808                               Errors.TypeVarMayNotBeFollowedByOtherBounds);
4809                     return bounds.head.type;
4810                 }
4811             } else {
4812                 // if first bound was a class or interface, accept only interfaces
4813                 // as further bounds.
4814                 for (JCExpression bound : bounds.tail) {
4815                     bound.type = checkBase(bound.type, bound, env, false, true, false);
4816                     if (bound.type.isErroneous()) {
4817                         bounds = List.of(bound);
4818                     }
4819                     else if (bound.type.hasTag(CLASS)) {
4820                         chk.checkNotRepeated(bound.pos(), types.erasure(bound.type), boundSet);
4821                     }
4822                 }
4823             }
4824         }
4825 
4826         if (bounds.length() == 0) {
4827             return syms.objectType;
4828         } else if (bounds.length() == 1) {
4829             return bounds.head.type;
4830         } else {
4831             Type owntype = types.makeIntersectionType(TreeInfo.types(bounds));
4832             // ... the variable's bound is a class type flagged COMPOUND
4833             // (see comment for TypeVar.bound).
4834             // In this case, generate a class tree that represents the
4835             // bound class, ...
4836             JCExpression extending;
4837             List<JCExpression> implementing;
4838             if (!bounds.head.type.isInterface()) {
4839                 extending = bounds.head;
4840                 implementing = bounds.tail;
4841             } else {
4842                 extending = null;
4843                 implementing = bounds;
4844             }
4845             JCClassDecl cd = make.at(tree).ClassDef(
4846                 make.Modifiers(PUBLIC | ABSTRACT),
4847                 names.empty, List.nil(),
4848                 extending, implementing, List.nil());
4849 
4850             ClassSymbol c = (ClassSymbol)owntype.tsym;
4851             Assert.check((c.flags() & COMPOUND) != 0);
4852             cd.sym = c;
4853             c.sourcefile = env.toplevel.sourcefile;
4854 
4855             // ... and attribute the bound class
4856             c.flags_field |= UNATTRIBUTED;
4857             Env<AttrContext> cenv = enter.classEnv(cd, env);
4858             typeEnvs.put(c, cenv);
4859             attribClass(c);
4860             return owntype;
4861         }

4978     /** Attribute class definition associated with given class symbol.
4979      *  @param c   The class symbol whose definition will be attributed.
4980      */
4981     void attribClass(ClassSymbol c) throws CompletionFailure {
4982         if (c.type.hasTag(ERROR)) return;
4983 
4984         // Check for cycles in the inheritance graph, which can arise from
4985         // ill-formed class files.
4986         chk.checkNonCyclic(null, c.type);
4987 
4988         Type st = types.supertype(c.type);
4989         if ((c.flags_field & Flags.COMPOUND) == 0) {
4990             // First, attribute superclass.
4991             if (st.hasTag(CLASS))
4992                 attribClass((ClassSymbol)st.tsym);
4993 
4994             // Next attribute owner, if it is a class.
4995             if (c.owner.kind == TYP && c.owner.type.hasTag(CLASS))
4996                 attribClass((ClassSymbol)c.owner);
4997         }
4998 
4999         // The previous operations might have attributed the current class
5000         // if there was a cycle. So we test first whether the class is still
5001         // UNATTRIBUTED.
5002         if ((c.flags_field & UNATTRIBUTED) != 0) {
5003             c.flags_field &= ~UNATTRIBUTED;
5004 
5005             // Get environment current at the point of class definition.
5006             Env<AttrContext> env = typeEnvs.get(c);
5007 
5008             // The info.lint field in the envs stored in typeEnvs is deliberately uninitialized,
5009             // because the annotations were not available at the time the env was created. Therefore,
5010             // we look up the environment chain for the first enclosing environment for which the
5011             // lint value is set. Typically, this is the parent env, but might be further if there
5012             // are any envs created as a result of TypeParameter nodes.
5013             Env<AttrContext> lintEnv = env;
5014             while (lintEnv.info.lint == null)
5015                 lintEnv = lintEnv.next;
5016 
5017             // Having found the enclosing lint value, we can initialize the lint value for this class
5018             env.info.lint = lintEnv.info.lint.augment(c);
5019 
5020             Lint prevLint = chk.setLint(env.info.lint);
5021             JavaFileObject prev = log.useSource(c.sourcefile);
5022             ResultInfo prevReturnRes = env.info.returnResult;
5023 
5024             try {
5025                 deferredLintHandler.flush(env.tree);
5026                 env.info.returnResult = null;
5027                 // java.lang.Enum may not be subclassed by a non-enum
5028                 if (st.tsym == syms.enumSym &&
5029                     ((c.flags_field & (Flags.ENUM|Flags.COMPOUND)) == 0))
5030                     log.error(env.tree.pos(), Errors.EnumNoSubclassing);
5031 
5032                 // Enums may not be extended by source-level classes
5033                 if (st.tsym != null &&
5034                     ((st.tsym.flags_field & Flags.ENUM) != 0) &&
5035                     ((c.flags_field & (Flags.ENUM | Flags.COMPOUND)) == 0)) {
5036                     log.error(env.tree.pos(), Errors.EnumTypesNotExtensible);
5037                 }

5039                 if (isSerializable(c.type)) {
5040                     env.info.isSerializable = true;
5041                 }

5044 
5045                 chk.checkDeprecatedAnnotation(env.tree.pos(), c);

5048                 chk.checkLeaksNotAccessible(env, (JCClassDecl) env.tree);
5049             } finally {
5050                 env.info.returnResult = prevReturnRes;
5051                 log.useSource(prev);

5077     /** Finish the attribution of a class. */
5078     private void attribClassBody(Env<AttrContext> env, ClassSymbol c) {
5079         JCClassDecl tree = (JCClassDecl)env.tree;
5080         Assert.check(c == tree.sym);
5081 
5082         // Validate type parameters, supertype and interfaces.
5083         attribStats(tree.typarams, env);
5084         if (!c.isAnonymous()) {
5085             //already checked if anonymous
5086             chk.validate(tree.typarams, env);
5087             chk.validate(tree.extending, env);
5088             chk.validate(tree.implementing, env);
5089         }
5090 
5091         c.markAbstractIfNeeded(types);
5092 
5093         // If this is a non-abstract class, check that it has no abstract
5094         // methods or unimplemented methods of an implemented interface.
5095         if ((c.flags() & (ABSTRACT | INTERFACE)) == 0) {
5096             chk.checkAllDefined(tree.pos(), c);
5097         }
5098 
5099         if ((c.flags() & ANNOTATION) != 0) {
5100             if (tree.implementing.nonEmpty())
5101                 log.error(tree.implementing.head.pos(),
5102                           Errors.CantExtendIntfAnnotation);
5103             if (tree.typarams.nonEmpty()) {
5104                 log.error(tree.typarams.head.pos(),
5105                           Errors.IntfAnnotationCantHaveTypeParams(c));
5106             }
5107 
5108             // If this annotation type has a @Repeatable, validate
5109             Attribute.Compound repeatable = c.getAnnotationTypeMetadata().getRepeatable();
5110             // If this annotation type has a @Repeatable, validate
5111             if (repeatable != null) {
5112                 // get diagnostic position for error reporting
5113                 DiagnosticPosition cbPos = getDiagnosticPosition(tree, repeatable.type);
5114                 Assert.checkNonNull(cbPos);
5115 
5116                 chk.validateRepeatable(c, repeatable, cbPos);
5117             }
5118         } else {
5119             // Check that all extended classes and interfaces
5120             // are compatible (i.e. no two define methods with same arguments
5121             // yet different return types).  (JLS 8.4.6.3)
5122             chk.checkCompatibleSupertypes(tree.pos(), c.type);
5123             if (allowDefaultMethods) {
5124                 chk.checkDefaultMethodClashes(tree.pos(), c.type);
5125             }
5126         }
5127 
5128         // Check that class does not import the same parameterized interface
5129         // with two different argument lists.
5130         chk.checkClassBounds(tree.pos(), c.type);
5131 
5132         tree.type = c.type;
5133 
5134         for (List<JCTypeParameter> l = tree.typarams;
5135              l.nonEmpty(); l = l.tail) {
5136              Assert.checkNonNull(env.info.scope.findFirst(l.head.name));
5137         }
5138 
5139         // Check that a generic class doesn't extend Throwable
5140         if (!c.type.allparams().isEmpty() && types.isSubtype(c.type, syms.throwableType))
5141             log.error(tree.extending.pos(), Errors.GenericThrowable);
5142 
5143         // Check that all methods which implement some
5144         // method conform to the method they implement.
5145         chk.checkImplementations(tree);
5146 
5147         //check that a resource implementing AutoCloseable cannot throw InterruptedException
5148         checkAutoCloseable(tree.pos(), env, c.type);
5149 
5150         for (List<JCTree> l = tree.defs; l.nonEmpty(); l = l.tail) {
5151             // Attribute declaration
5152             attribStat(l.head, env);
5153             // Check that declarations in inner classes are not static (JLS 8.1.2)
5154             // Make an exception for static constants.
5155             if (c.owner.kind != PCK &&
5156                 ((c.flags() & STATIC) == 0 || c.name == names.empty) &&
5157                 (TreeInfo.flags(l.head) & (STATIC | INTERFACE)) != 0) {
5158                 Symbol sym = null;
5159                 if (l.head.hasTag(VARDEF)) sym = ((JCVariableDecl) l.head).sym;
5160                 if (sym == null ||
5161                     sym.kind != VAR ||
5162                     ((VarSymbol) sym).getConstValue() == null)
5163                     log.error(l.head.pos(), Errors.IclsCantHaveStaticDecl(c));
5164             }
5165         }
5166 
5167         // Check for cycles among non-initial constructors.
5168         chk.checkCyclicConstructors(tree);
5169 
5170         // Check for cycles among annotation elements.
5171         chk.checkNonCyclicElements(tree);
5172 
5173         // Check for proper use of serialVersionUID
5174         if (env.info.lint.isEnabled(LintCategory.SERIAL)
5175                 && isSerializable(c.type)
5176                 && (c.flags() & (Flags.ENUM | Flags.INTERFACE)) == 0
5177                 && !c.isAnonymous()) {
5178             checkSerialVersionUID(tree, c);
5179         }
5180         if (allowTypeAnnos) {
5181             // Correctly organize the positions of the type annotations
5182             typeAnnotations.organizeTypeAnnotationsBodies(tree);
5183 
5184             // Check type annotations applicability rules
5185             validateTypeAnnotations(tree, false);

5371         /* I would want to model this after
5372          * com.sun.tools.javac.comp.Check.Validator.visitSelectInternal(JCFieldAccess)
5373          * and override visitSelect and visitTypeApply.
5374          * However, we only set the annotated type in the top-level type
5375          * of the symbol.
5376          * Therefore, we need to override each individual location where a type
5377          * can occur.
5378          */
5379         private void validateAnnotatedType(final JCTree errtree, final Type type) {
5380             //System.err.println("Attr.validateAnnotatedType: " + errtree + " type: " + type);
5381 
5382             if (type.isPrimitiveOrVoid()) {
5383                 return;
5384             }
5385 
5386             JCTree enclTr = errtree;
5387             Type enclTy = type;
5388 
5389             boolean repeat = true;
5390             while (repeat) {
5391                 if (enclTr.hasTag(TYPEAPPLY)) {
5392                     List<Type> tyargs = enclTy.getTypeArguments();
5393                     List<JCExpression> trargs = ((JCTypeApply)enclTr).getTypeArguments();
5394                     if (trargs.length() > 0) {
5395                         // Nothing to do for diamonds
5396                         if (tyargs.length() == trargs.length()) {
5397                             for (int i = 0; i < tyargs.length(); ++i) {
5398                                 validateAnnotatedType(trargs.get(i), tyargs.get(i));
5399                             }
5400                         }
5401                         // If the lengths don't match, it's either a diamond
5402                         // or some nested type that redundantly provides
5403                         // type arguments in the tree.
5404                     }
5405 
5406                     // Look at the clazz part of a generic type
5407                     enclTr = ((JCTree.JCTypeApply)enclTr).clazz;
5408                 }
5409 
5410                 if (enclTr.hasTag(SELECT)) {
5411                     enclTr = ((JCTree.JCFieldAccess)enclTr).getExpression();
5412                     if (enclTy != null &&
5413                             !enclTy.hasTag(NONE)) {
5414                         enclTy = enclTy.getEnclosingType();
5415                     }
5416                 } else if (enclTr.hasTag(ANNOTATED_TYPE)) {
5417                     JCAnnotatedType at = (JCTree.JCAnnotatedType) enclTr;
5418                     if (enclTy == null || enclTy.hasTag(NONE)) {
5419                         if (at.getAnnotations().size() == 1) {
5420                             log.error(at.underlyingType.pos(), Errors.CantTypeAnnotateScoping1(at.getAnnotations().head.attribute));
5421                         } else {
5422                             ListBuffer<Attribute.Compound> comps = new ListBuffer<>();
5423                             for (JCAnnotation an : at.getAnnotations()) {
5424                                 comps.add(an.attribute);
5425                             }
5426                             log.error(at.underlyingType.pos(), Errors.CantTypeAnnotateScoping(comps.toList()));
5427                         }
5428                         repeat = false;
5429                     }
5430                     enclTr = at.underlyingType;
5431                     // enclTy doesn't need to be changed
5432                 } else if (enclTr.hasTag(IDENT)) {
5433                     repeat = false;
5434                 } else if (enclTr.hasTag(JCTree.Tag.WILDCARD)) {
5435                     JCWildcard wc = (JCWildcard) enclTr;
5436                     if (wc.getKind() == JCTree.Kind.EXTENDS_WILDCARD ||
5437                             wc.getKind() == JCTree.Kind.SUPER_WILDCARD) {
5438                         validateAnnotatedType(wc.getBound(), wc.getBound().type);
5439                     } else {
5440                         // Nothing to do for UNBOUND
5441                     }
5442                     repeat = false;
5443                 } else if (enclTr.hasTag(TYPEARRAY)) {
5444                     JCArrayTypeTree art = (JCArrayTypeTree) enclTr;
5445                     validateAnnotatedType(art.getType(), art.elemtype.type);
5446                     repeat = false;
5447                 } else if (enclTr.hasTag(TYPEUNION)) {
5448                     JCTypeUnion ut = (JCTypeUnion) enclTr;
5449                     for (JCTree t : ut.getTypeAlternatives()) {
5450                         validateAnnotatedType(t, t.type);
5451                     }
5452                     repeat = false;
5453                 } else if (enclTr.hasTag(TYPEINTERSECTION)) {
5454                     JCTypeIntersection it = (JCTypeIntersection) enclTr;
5455                     for (JCTree t : it.getBounds()) {
5456                         validateAnnotatedType(t, t.type);
5457                     }
5458                     repeat = false;
5459                 } else if (enclTr.getKind() == JCTree.Kind.PRIMITIVE_TYPE ||
5460                            enclTr.getKind() == JCTree.Kind.ERRONEOUS) {
5461                     repeat = false;
5462                 } else {
5463                     Assert.error("Unexpected tree: " + enclTr + " with kind: " + enclTr.getKind() +
5464                             " within: "+ errtree + " with kind: " + errtree.getKind());
5465                 }
5466             }
5467         }
