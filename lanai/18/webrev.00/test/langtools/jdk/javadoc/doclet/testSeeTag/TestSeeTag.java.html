24 /*
25  * @test
26  * @bug      8017191 8182765 8200432 8239804
27  * @summary  Javadoc is confused by at-link to imported classes outside of the set of generated packages
28  * @library  ../../lib
29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
30  * @build    javadoc.tester.*
31  * @run main TestSeeTag
32  */
33 
34 import javadoc.tester.JavadocTester;
35 
36 public class TestSeeTag extends JavadocTester {
37 
38     public static void main(String... args) throws Exception {
39         TestSeeTag tester = new TestSeeTag();
40         tester.runTests();
41     }
42 
43     @Test
44     public void test() {
45         javadoc("-d", "out",
46                 "-sourcepath", testSrc,
47                 "pkg");
48         checkExit(Exit.OK);
49 
50         checkOutput("pkg/Test.html", true,
51             "<code>List</code>",
52             "<dl class=\"notes\">\n"
53             + "<dt>See Also:</dt>\n"
54             + "<dd><a href=\"Test.InnerOne.html#foo()\"><code>Test.InnerOne.foo()</code></a>, \n"
55             + "<a href=\"Test.InnerOne.html#bar(java.lang.Object)\"><code>Test.InnerOne.bar(Object)</code></a>, \n"
56             + "<a href=\"http://docs.oracle.com/javase/7/docs/technotes/tools/windows/javadoc.html#see\">Javadoc</a>, \n"
57             + "<a href=\"Test.InnerOne.html#baz(float)\"><code>something</code></a></dd>\n"
58             + "</dl>");
59 
60         checkOutput("pkg/Test.html", false,
61           "&lt;code&gt;List&lt;/code&gt;");
62 
63         checkOutput("pkg/Test2.html", true,
64            "<code>Serializable</code>");
65 
66         checkOutput("pkg/Test2.html", false,
67            ">Serialized Form<");
68     }
69 
70     @Test
71     public void testBadReference() {
72         javadoc("-d", "out-badref",
73                 "-sourcepath", testSrc,
74                 "badref");
75         checkExit(Exit.ERROR);
76 
77         checkOutput("badref/Test.html", true,
78                 "<dl class=\"notes\">\n"
79                 + "<dt>See Also:</dt>\n"
80                 + "<dd><code>Object</code>, \n"
81                 + "<code>Foo<String></code></dd>\n"
82                 + "</dl>");
83     }
84 }
