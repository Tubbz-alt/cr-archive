24 /*
25  * @test
26  * @bug      4904037 8026567 8239804
27  * @summary  The constructor comments should be surrounded by
28  *           <dl></dl>.  Check for this in the output.
29  * @library  ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build    javadoc.tester.*
32  * @run main TestConstructorIndent
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestConstructorIndent extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestConstructorIndent tester = new TestConstructorIndent();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-d", "out",
47                 "-sourcepath", testSrc,
48                 testSrc("C.java"));
49         checkExit(Exit.OK);
50 
51         checkOutput("C.html", true,
52                 "<div class=\"block\">"
53                 + "This is just a simple constructor.</div>\n"
54                 + "<dl class=\"notes\">\n"
55                 + "<dt>Parameters:</dt>\n"
56                 + "<dd><code>i</code> - a param.</dd>\n"
57                 + "</dl>");
58     }
59 }
