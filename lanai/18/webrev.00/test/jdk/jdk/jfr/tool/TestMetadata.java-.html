26 package jdk.jfr.tool;
27 
28 import java.nio.file.Path;
29 
30 import jdk.jfr.EventType;
31 import jdk.jfr.consumer.RecordingFile;
32 import jdk.test.lib.process.OutputAnalyzer;
33 
34 /**
35  * @test
36  * @summary Test jfr info
37  * @key jfr
38  * @requires vm.hasJFR
39  * @library /test/lib /test/jdk
40  * @run main/othervm jdk.jfr.tool.TestMetadata
41  */
42 public class TestMetadata {
43 
44     public static void main(String[] args) throws Throwable {
45         Path f = ExecuteHelper.createProfilingRecording().toAbsolutePath();
46         String file = f.toAbsolutePath().toString();
47 
48         OutputAnalyzer output = ExecuteHelper.jfr("metadata");
49         output.shouldContain("missing file");
50 
51         output = ExecuteHelper.jfr("metadata", "--wrongOption", file);
52         output.shouldContain("unknown option --wrongOption");
53 
54         output = ExecuteHelper.jfr("metadata", file);
55         try (RecordingFile rf = new RecordingFile(f)) {
56             for (EventType t : rf.readEventTypes()) {
57                 String name = t.getName();
58                 name = name.substring(name.lastIndexOf(".") + 1);
59                 output.shouldContain(name);
60             }
61         }
62     }
63 }
