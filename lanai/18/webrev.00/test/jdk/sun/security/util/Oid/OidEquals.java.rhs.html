24 /*
25  * @test
26  * @bug 8022444
27  * @summary Test ObjectIdentifier.equals(Object obj)
28  * @modules java.base/sun.security.util
29  */
30 
31 import sun.security.util.ObjectIdentifier;
32 
33 public class OidEquals {
34     public static void main(String[] args) throws Exception {
35         ObjectIdentifier oid1 = new ObjectIdentifier("1.3.6.1.4.1.42.2.17");
36         ObjectIdentifier oid2 = new ObjectIdentifier("1.2.3.4");


37 
38         assertEquals(oid1, oid1);
39         assertNotEquals(oid1, oid2);

40         assertNotEquals(oid1, "1.3.6.1.4.1.42.2.17");
41 
42         System.out.println("Tests passed.");
43     }
44 
45     static void assertEquals(ObjectIdentifier oid, Object obj)
46             throws Exception {
47         if (!oid.equals(obj)) {
48             throw new Exception("The ObjectIdentifier " + oid.toString() +
49                     " should be equal to the Object " + obj.toString());
50         }
51     }
52 
53     static void assertNotEquals(ObjectIdentifier oid, Object obj)
54             throws Exception {
55         if (oid.equals(obj)) {
56             throw new Exception("The ObjectIdentifier " + oid.toString() +
57                     " should not be equal to the Object " + obj.toString());
58         }
59     }
60 }
