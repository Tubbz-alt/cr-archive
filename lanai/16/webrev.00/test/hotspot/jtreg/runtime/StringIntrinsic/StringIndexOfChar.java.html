24 /* @test
25  * @bug 8239787
26  * @summary String.indexOf(char) for empty string must give -1
27  * @run main/othervm -XX:-CompactStrings StringIndexOfChar
28  */
29 public class StringIndexOfChar {
30     public static void main(String[] args) throws Exception {
31         String emptyString = "";
32         for (int i = 0; i < 100; i++) {
33             for(int c = 0; c < 0xFFFF; c++) {
34                 int result = emptyString.indexOf((char)c, -1);
35                 if (result != -1) {
36                     throw new Exception("new String(\"\").indexOf(char, -1) must be -1, but got " + result);
37                 }
38             }
39         }
40     }
41 }
