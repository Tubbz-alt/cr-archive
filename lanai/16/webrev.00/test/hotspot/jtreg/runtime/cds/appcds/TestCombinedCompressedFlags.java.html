 24 /**
 25  * @test
 26  * @bug 8232069
 27  * @summary Testing different combination of CompressedOops and CompressedClassPointers
 28  * @requires vm.cds
 29  * @requires (vm.gc=="null")
 30  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
 31  * @compile test-classes/Hello.java
 32  * @modules java.base/jdk.internal.misc
 33  * @run main/othervm  TestCombinedCompressedFlags
 34  */
 35 
 36 import jdk.test.lib.Platform;
 37 import jdk.test.lib.process.OutputAnalyzer;
 38 import java.util.List;
 39 import java.util.ArrayList;
 40 import jtreg.SkippedException;
 41 
 42 public class TestCombinedCompressedFlags {
 43     public static String HELLO_STRING = "Hello World";
 44     public static String EXEC_ABNORMAL_MSG = "Unable to use shared archive.";
 45     public static final int PASS = 0;
 46     public static final int FAIL = 1;
 47 
 48     static class ConfArg {
 49         public boolean useCompressedOops;            // UseCompressedOops
 50         public boolean useCompressedClassPointers;   // UseCompressedClassPointers
 51         public String  msg;
 52         public int code;
 53         public ConfArg(boolean useCompressedOops, boolean useCompressedClassPointers, String msg, int code) {
 54             this.useCompressedOops = useCompressedOops;
 55             this.useCompressedClassPointers = useCompressedClassPointers;
 56             this.msg  = msg;
 57             this.code = code;
 58         }
 59     }
 60 
 61     static class RunArg {
 62         public ConfArg dumpArg;
 63         public List<ConfArg> execArgs;
 64         public RunArg(ConfArg arg) {
 65             dumpArg = arg;
 66             initExecArgs();
 67         }
 68         private void initExecArgs() {
 69            /* The combinations have four cases. Note COOP off, CCPTR must be off
 70             *          UseCompressedOops   UseCompressedClassPointers  Result
 71             *    1.
 72             *    dump: on                  on
 73             *    test: on                  on                          Pass
 74             *          on                  off                         Fail
 75             *          off                 on                          Fail
 76             *          off                 off                         Fail
 77             *    2.
 78             *    dump: on                  off
 79             *    test: on                  off                         Pass
 80             *          on                  on                          Fail
 81             *          off                 on                          Pass
 82             *          off                 off                         Fail
 83             *    3.
 84             *    dump: off                 on
 85             *    test: off                 on                          Pass
 86             *          off                 off                         Pass
 87             *          on                  on                          Fail
 88             *          on                  off                         Fail
 89             *    4.
 90             *    dump: off                 off
 91             *    test: off                 off                         Pass
 92             *          off                 on                          Pass
 93             *          on                  on                          Fail
 94             *          on                  off                         Fail
 95             **/
 96             execArgs = new ArrayList<ConfArg>();
 97             if (dumpArg.useCompressedOops && dumpArg.useCompressedClassPointers) {
 98                 execArgs
 99                     .add(new ConfArg(true, true, HELLO_STRING, PASS));
100                 execArgs
101                     .add(new ConfArg(true, false, EXEC_ABNORMAL_MSG, FAIL));
102                 execArgs
103                     .add(new ConfArg(false, true, EXEC_ABNORMAL_MSG, FAIL));
104                 execArgs
105                     .add(new ConfArg(false, false, EXEC_ABNORMAL_MSG, FAIL));
106 
107             }  else if(dumpArg.useCompressedOops && !dumpArg.useCompressedClassPointers) {
108                 execArgs
109                     .add(new ConfArg(true, false, HELLO_STRING, PASS));
110                 execArgs
111                     .add(new ConfArg(true, true, EXEC_ABNORMAL_MSG, FAIL));
112                 execArgs
113                     .add(new ConfArg(false, true, EXEC_ABNORMAL_MSG, FAIL));
114                 execArgs
115                     .add(new ConfArg(false, false, EXEC_ABNORMAL_MSG, FAIL));
116 
117             } else if (!dumpArg.useCompressedOops && dumpArg.useCompressedClassPointers) {
118                 execArgs
119                     .add(new ConfArg(false, false, HELLO_STRING, PASS));
120                 execArgs
121                     .add(new ConfArg(false, true, HELLO_STRING, PASS));
122                 execArgs
123                     .add(new ConfArg(true, true, EXEC_ABNORMAL_MSG, FAIL));
124                 execArgs
125                     .add(new ConfArg(true, false, EXEC_ABNORMAL_MSG, FAIL));
126             } else if (!dumpArg.useCompressedOops && !dumpArg.useCompressedClassPointers) {
127                 execArgs
128                     .add(new ConfArg(false, false, HELLO_STRING, PASS));
129                 execArgs
130                     .add(new ConfArg(false, true, HELLO_STRING, PASS));
131                 execArgs
132                     .add(new ConfArg(true, true, EXEC_ABNORMAL_MSG, FAIL));
133                 execArgs
134                     .add(new ConfArg(true, false, EXEC_ABNORMAL_MSG, FAIL));
135             }
136         }
137     }
138 
139     public static String getCompressedOopsArg(boolean on) {
140         if (on) return "-XX:+UseCompressedOops";
141         else    return "-XX:-UseCompressedOops";
142     }
143 
144     public static String getCompressedClassPointersArg(boolean on) {
145         if (on) return "-XX:+UseCompressedClassPointers";
146         else    return "-XX:-UseCompressedClassPointers";
147     }
148 
149     public static List<RunArg> runList;
150 
151     public static void configureRunArgs() {
152         runList = new ArrayList<RunArg>();
153         runList
154             .add(new RunArg(new ConfArg(true, true, null, PASS)));
155         runList
156             .add(new RunArg(new ConfArg(true, false, null, PASS)));
157         runList
158             .add(new RunArg(new ConfArg(false, true, null, PASS)));
159         runList
160             .add(new RunArg(new ConfArg(false, false, null, PASS)));
161     }
162 
163     public static void main(String[] args) throws Exception {
164         if (!Platform.is64bit()) {
165             throw new SkippedException("Platform is not 64 bit, skipped");
166         }
167 
168         String helloJar = JarBuilder.build("hello", "Hello");
169         configureRunArgs();
170         OutputAnalyzer out;
171         for (RunArg t: runList) {
172             out = TestCommon
173                 .dump(helloJar,
174                       new String[] {"Hello"},
175                       getCompressedOopsArg(t.dumpArg.useCompressedOops),
176                       getCompressedClassPointersArg(t.dumpArg.useCompressedClassPointers),
177                       "-Xlog:cds");
178             out.shouldContain("Dumping shared data to file:");
179             out.shouldHaveExitValue(0);
180 
181             for (ConfArg c : t.execArgs) {
182                 out = TestCommon.exec(helloJar,
183                                       "-cp",
184                                       helloJar,
185                                       "-Xlog:cds",
186                                       getCompressedOopsArg(c.useCompressedOops),
187                                       getCompressedClassPointersArg(c.useCompressedClassPointers),
188                                       "Hello");
189                 out.shouldContain(c.msg);
190                 out.shouldHaveExitValue(c.code);
191             }
192         }
193     }
