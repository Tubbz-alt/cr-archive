24 package org.openjdk.bench.java.lang;
25 
26 import org.openjdk.jmh.annotations.*;
27 import java.util.concurrent.TimeUnit;
28 
29 @BenchmarkMode(Mode.AverageTime)
30 @OutputTimeUnit(TimeUnit.NANOSECONDS)
31 @State(Scope.Benchmark)
32 public class StringSubstring {
33 
34     public String s = new String("An arbitrary string that happened to be of length 52");
35 
36     @Benchmark
37     public String from26toEnd0() {
38         return s.substring(26);
39     }
40 
41     @Benchmark
42     public String from26toEnd1() {
43         return s.substring(26, s.length());
44     }
45 }
