24 package org.openjdk.bench.java.lang;
25 
26 import org.openjdk.jmh.annotations.*;
27 import java.util.concurrent.TimeUnit;
28 
29 @BenchmarkMode(Mode.AverageTime)
30 @OutputTimeUnit(TimeUnit.NANOSECONDS)
31 @Fork(value = 3)
32 @Warmup(iterations = 5, time = 1, timeUnit = TimeUnit.SECONDS)
33 @Measurement(iterations = 5, time = 1, timeUnit = TimeUnit.SECONDS)
34 @State(Scope.Benchmark)
35 public class StringSubstring {
36 
37     public String s = new String("An arbitrary string that happened to be of length 52");
38 
39     @Benchmark
40     public String from26toEnd0() {
41         return s.substring(26);
42     }
43 
44     @Benchmark
45     public String from26toEnd1() {
46         return s.substring(26, s.length());
47     }
48 
49     /**
50      * An empty substring should not allocate a new String
51      */
52     @Benchmark
53     public String empty() {
54         return s.substring(17, 17);
55     }
56 }
