24 /*
25  * @test
26  * @bug 4530727 8026567 8239804
27  * @summary When an exception is declared in the method signature but
28  * not documented with a throws tag, we generate a link to it in the
29  * throws section.  Make sure that the link is below a Throws heading.
30  * @library ../../lib
31  * @modules jdk.javadoc/jdk.javadoc.internal.tool
32  * @build javadoc.tester.*
33  * @run main TestThrowsHead
34  */
35 
36 import javadoc.tester.JavadocTester;
37 
38 public class TestThrowsHead extends JavadocTester {
39 
40     public static void main(String... args) throws Exception {
41         TestThrowsHead tester = new TestThrowsHead();
42         tester.runTests();
43     }
44 
45     @Test
46     public void test() {
47         javadoc("-d", "out",
48                 testSrc("C.java"));
49         checkExit(Exit.OK);
50 
51         checkOutput("C.html", true,
52                 "<dt>Throws:</dt>");
53     }
54 }
