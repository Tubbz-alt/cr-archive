24 /*
25  * @test
26  * @bug 4695326 4750173 4920381 8078320 8071982 8239804
27  * @summary Test the declaration of simple tags using -tag. Verify that
28  * "-tag name" is a shortcut for "-tag name:a:Name:".  Also verity that
29  * you can escape the ":" character with a back slash so that it is not
30  * considered a separator when parsing the simple tag argument.
31  * @library ../../lib
32  * @modules jdk.javadoc/jdk.javadoc.internal.tool
33  * @build javadoc.tester.*
34  * @run main TestSimpleTag
35  */
36 
37 import javadoc.tester.JavadocTester;
38 
39 public class TestSimpleTag extends JavadocTester {
40 
41     public static void main(String... args) throws Exception {
42         TestSimpleTag tester = new TestSimpleTag();
43         tester.runTests();
44     }
45 
46     @Test
47     public void test() {
48         javadoc("-d", "out",
49                 "-sourcepath", testSrc,
50                 "-tag", "param",
51                 "-tag", "todo",
52                 "-tag", "ejb\\:bean:a:EJB Beans:",
53                 "-tag", "regular:a:Regular Tag:",
54                 "-tag", "tag-with-hyphens:a:Tag-With-Hyphens:",
55                 testSrc("C.java"));
56         checkExit(Exit.OK);
57 
58         checkOutput("C.html", true,
59                 "<dl class=\"notes\">",
60                 "<dt>Todo:</dt>",
61                 "<dt>EJB Beans:</dt>",
62                 "<dt>Regular Tag:</dt>",
63                 "<dt>Tag-With-Hyphens:</dt>",
64                 "<dt>Parameters:</dt>\n"
65                 + "<dd><code>arg</code> - this is an int argument.</dd>");
66     }
67 }
