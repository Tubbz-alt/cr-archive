24 /*
25  * @test
26  * @bug      7180906 8026567 8239804
27  * @summary  Test to make sure that the since tag works correctly
28  * @library  ../../lib
29  * @modules jdk.javadoc/jdk.javadoc.internal.tool
30  * @build    javadoc.tester.*
31  * @run main TestSinceTag
32  */
33 
34 import javadoc.tester.JavadocTester;
35 
36 public class TestSinceTag extends JavadocTester {
37 
38     public static void main(String... args) throws Exception {
39         TestSinceTag tester = new TestSinceTag();
40         tester.runTests();
41         tester.printSummary();
42     }
43 
44     @Test
45     public void testSince() {
46         javadoc("-d", "out-since",
47                 "-sourcepath", testSrc,
48                 "pkg1");
49         checkExit(Exit.OK);
50 
51         checkSince(true);
52     }
53 
54     @Test
55     public void testNoSince() {
56         javadoc("-d", "out-nosince",
57                 "-sourcepath", testSrc,
58                 "-nosince",
59                 "pkg1");
60         checkExit(Exit.OK);
61 
62         checkSince(false);
63     }
64 
65     void checkSince(boolean on) {
66         checkOutput("pkg1/C1.html", on,
67                 "<dl class=\"notes\">\n"
68                 + "<dt>Since:</dt>\n"
69                 + "<dd>JDK1.0</dd>");
70 
71         checkOutput("serialized-form.html", on,
72                 "<dl class=\"notes\">\n"
73                 + "<dt>Since:</dt>\n"
74                 + "<dd>1.4</dd>");
75     }
76 }
