24 /*
25  * @test
26  * @bug      4732864 6280605 7064544 8014636 8016328 8025633 8071982 8182765
27  * @summary  Make sure that you can link from one member to another using
28  *           non-qualified name, furthermore, ensure the right one is linked.
29  * @library  ../../lib
30  * @modules jdk.javadoc/jdk.javadoc.internal.tool
31  * @build    javadoc.tester.*
32  * @run main TestLinkTaglet
33  */
34 
35 import javadoc.tester.JavadocTester;
36 
37 public class TestLinkTaglet extends JavadocTester {
38 
39     public static void main(String... args) throws Exception {
40         TestLinkTaglet tester = new TestLinkTaglet();
41         tester.runTests();
42     }
43 
44     @Test
45     public void test() {
46         javadoc("-Xdoclint:none",
47                 "-d", "out",
48                 "-sourcepath", testSrc,
49                 "pkg", testSrc("checkPkg/B.java"));
50         checkExit(Exit.OK);
51 
52         checkOutput("pkg/C.html", true,
53                 "Qualified Link: <a href=\"C.InnerC.html\" title=\"class in pkg\"><code>C.InnerC</code></a>.<br/>\n"
54                 + " Unqualified Link1: <a href=\"C.InnerC.html\" title=\"class in pkg\"><code>C.InnerC</code></a>.<br/>\n"
55                 + " Unqualified Link2: <a href=\"C.InnerC.html\" title=\"class in pkg\"><code>C.InnerC</code></a>.<br/>\n"
56                 + " Qualified Link: <a href=\"#method(pkg.C.InnerC,pkg.C.InnerC2)\"><code>method(pkg.C.InnerC, pkg.C.InnerC2)</code></a>.<br/>\n"
57                 + " Unqualified Link: <a href=\"#method(pkg.C.InnerC,pkg.C.InnerC2)\"><code>method(C.InnerC, C.InnerC2)</code></a>.<br/>\n"
58                 + " Unqualified Link: <a href=\"#method(pkg.C.InnerC,pkg.C.InnerC2)\"><code>method(InnerC, InnerC2)</code></a>.<br/>\n"
59                 + " Package Link: <a href=\"package-summary.html\"><code>pkg</code></a>.<br/>");
60 
61         checkOutput("pkg/C.InnerC.html", true,
62                 "Link to member in outer class: <a href=\"C.html#MEMBER\"><code>C.MEMBER</code></a> <br/>\n"
63                 + " Link to member in inner class: <a href=\"C.InnerC2.html#MEMBER2\"><code>C.InnerC2.MEMBER2</code></a> <br/>\n"
64                 + " Link to another inner class: <a href=\"C.InnerC2.html\" title=\"class in pkg\"><code>C.InnerC2</code></a>");
65 
66         checkOutput("pkg/C.InnerC2.html", true,
67                 "<dl class=\"notes\">\n"
68                 + "<dt>Enclosing class:</dt>\n"
69                 + "<dd><a href=\"C.html\" title=\"class in pkg\">C</a></dd>\n"
70                 + "</dl>");
71 
72         checkOutput(Output.OUT, false,
73                 "Tag @see: reference not found: A");
74 
75         checkFiles(false, "checkPkg/A.html");
76     }
77 }
