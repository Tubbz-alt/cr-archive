        STLPORT_LIB="$SOLARIS_STUDIO_DIR/lib/stlport4/v9/libstlport.so.1"
      fi
      if test -f "$STLPORT_LIB"; then
        AC_MSG_RESULT([yes, $STLPORT_LIB])
!       BASIC_FIXUP_PATH([STLPORT_LIB])

!       AC_MSG_RESULT([no, not found at $STLPORT_LIB])

    fi
