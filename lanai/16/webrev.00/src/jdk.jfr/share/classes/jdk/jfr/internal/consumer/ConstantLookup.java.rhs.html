26 package jdk.jfr.internal.consumer;
27 
28 import jdk.jfr.internal.Type;
29 
30 final class ConstantLookup {
31     private final Type type;
32     private ConstantMap current;
33     private ConstantMap previous = ConstantMap.EMPTY;
34 
35     ConstantLookup(ConstantMap current, Type type) {
36         this.current = current;
37         this.type = type;
38     }
39 
40     public Type getType() {
41         return type;
42     }
43 
44     public ConstantMap getLatestPool() {
45         return current;
46     }
47 
48     public void newPool() {
49         previous = current;
50         current = new ConstantMap(current.factory, current.name);
51     }
52 
53     public Object getPreviousResolved(long key) {
54         return previous.getResolved(key);
55     }
56 
57     public Object getCurrentResolved(long key) {
58         return current.getResolved(key);
59     }
60 
61     public Object getCurrent(long key) {
62         return current.get(key);
63     }
64 }
