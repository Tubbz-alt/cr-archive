 26 package jdk.internal.platform;
 27 
 28 /**
 29  * Cgroup v1 Metrics extensions
 30  *
 31  */
 32 public class CgroupV1MetricsImpl extends CgroupMetrics implements CgroupV1Metrics {
 33 
 34     private final CgroupV1Metrics metrics;
 35 
 36     CgroupV1MetricsImpl(CgroupV1Metrics metrics) {
 37         super((CgroupSubsystem)metrics);
 38         this.metrics = metrics;
 39     }
 40 
 41     @Override
 42     public long getMemoryMaxUsage() {
 43         return metrics.getMemoryMaxUsage();
 44     }
 45 
 46     @Override
 47     public long getKernelMemoryFailCount() {
 48         return metrics.getKernelMemoryFailCount();
 49     }
 50 
 51     @Override
 52     public long getKernelMemoryLimit() {
 53         return metrics.getKernelMemoryLimit();
 54     }
 55 
 56     @Override
 57     public long getKernelMemoryMaxUsage() {
 58         return metrics.getKernelMemoryMaxUsage();
 59     }
 60 
 61     @Override
 62     public long getKernelMemoryUsage() {
 63         return metrics.getKernelMemoryUsage();
 64     }
 65 
 66     @Override
 67     public long getTcpMemoryFailCount() {
 68         return metrics.getTcpMemoryFailCount();
 69     }
 70 
 71     @Override
 72     public long getTcpMemoryLimit() {
 73         return metrics.getTcpMemoryLimit();
 74     }
 75 
 76     @Override
 77     public long getTcpMemoryMaxUsage() {
 78         return metrics.getTcpMemoryMaxUsage();
 79     }
 80 
 81     @Override
 82     public long getMemoryAndSwapFailCount() {
 83         return metrics.getMemoryAndSwapFailCount();
 84     }
 85 
 86     @Override
 87     public long getMemoryAndSwapMaxUsage() {
 88         return metrics.getMemoryAndSwapMaxUsage();
 89     }
 90 
 91     @Override
 92     public Boolean isMemoryOOMKillEnabled() {
 93         return metrics.isMemoryOOMKillEnabled();
 94     }
 95 
 96     @Override
 97     public double getCpuSetMemoryPressure() {
 98         return metrics.getCpuSetMemoryPressure();
 99     }
100 
101     @Override
102     public Boolean isCpuSetMemoryPressureEnabled() {
103         return metrics.isCpuSetMemoryPressureEnabled();
104     }
105 
106 }
