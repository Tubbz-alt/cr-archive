26 #include "FileSystemSupport_md.h"
27 
28 /**
29  * Post-process the given URI path string if necessary.  This is used on
30  * win32, e.g., to transform "/c:/foo" into "c:/foo".  The path string
31  * still has slash separators; code in the File class will translate them
32  * after this method returns.
33  */
34 char* fromURIPath(const char* path);
35 
36 /**
37  * Return the basen path of the given pathname. If the string is already
38  * the base path then it is simply returned.
39  */
40 char* basePath(const char* path);
41 
42 /**
43  * Convert the given pathname string to normal form.  If the string is
44  * already in normal form then it is simply returned.
45  */
46 char* normalize(const char* path);
47 
48 /**
49  * Tell whether or not the given abstract pathname is absolute.
50  */
51 int isAbsolute(const char * path);
52 
53 /**
54  * Resolve the child pathname string against the parent.
55  */
56 char* resolve(const char* parent, const char* child);
