178     /**
179      * Returns the path for a type element.
180      * For example, if the type element is {@code java.lang.Object},
181      * the path is {@code java/lang/Object.html}.
182      *
183      * @param typeElement the type element
184      * @return the path
185      */
186     public DocPath forClass(TypeElement typeElement) {
187         return (typeElement == null)
188                 ? DocPath.empty
189                 : forPackage(utils.containingPackage(typeElement)).resolve(forName(typeElement));
190     }
191 
192     /**
193      * Returns the path for the simple name of a type element.
194      * For example, if the type element is {@code java.lang.Object},
195      * the path is {@code Object.html}.
196      *
197      * @param typeElement the type element
198      * @return the path
199      */
200     public DocPath forName(TypeElement typeElement) {
201         return (typeElement == null)
202                 ? DocPath.empty
203                 : new DocPath(utils.getSimpleName(typeElement) + ".html");
204     }
205 
206     public static DocPath forModule(ModuleElement mdle) {
207         return mdle == null || mdle.isUnnamed()
208                 ? DocPath.empty
209                 : DocPath.create(mdle.getQualifiedName().toString());
210     }
211 
212     /**
213      * Returns the path for the package of a type element.
214      * For example, if the type element is {@code java.lang.Object},
215      * the path is {@code java/lang}.
216      *
217      * @param typeElement the type element
218      * @return the path
219      */
220     public DocPath forPackage(TypeElement typeElement) {
221         return (typeElement == null)
222                 ? DocPath.empty
223                 : forPackage(utils.containingPackage(typeElement));
224     }
225 
226     /**
227      * Returns the path for a package.
228      * For example, if the package is {@code java.lang},
229      * the path is {@code java/lang}.
230      *
231      * @param pkgElement the package element
232      * @return the path
233      */
234     public DocPath forPackage(PackageElement pkgElement) {
235         if (pkgElement == null || pkgElement.isUnnamed()) {
236             return DocPath.empty;
237         }
238 
239         DocPath pkgPath = DocPath.create(pkgElement.getQualifiedName().toString().replace('.', '/'));
240         ModuleElement mdle = (ModuleElement) pkgElement.getEnclosingElement();
241         return forModule(mdle).resolve(pkgPath);
242     }
243 
244     /**
245      * Returns the inverse path for a package.
246      * For example, if the package is {@code java.lang},
247      * the inverse path is {@code ../..}.
248      *
249      * @param pkgElement the package element
250      * @return the path
251      */
252     public static DocPath forRoot(PackageElement pkgElement) {
253         String name = (pkgElement == null || pkgElement.isUnnamed())
254                 ? ""
255                 : pkgElement.getQualifiedName().toString();
256         return new DocPath(name.replace('.', '/').replaceAll("[^/]+", ".."));
257     }
258 
259     /**
260      * Returns a relative path from one package to another.
261      *
262      * @param from the origin of the relative path
263      * @param to the destination of the relative path
264      * @return the path
265      */
266     public DocPath relativePath(PackageElement from, PackageElement to) {
267         return forRoot(from).resolve(forPackage(to));
268     }

297     private DocPath createModulePath(ModuleElement mdle, String path) {
298         return DocPath.create(mdle.getQualifiedName() + moduleSeparator + path);
299     }
300 
301     private DocPath createModulePath(String moduleName, String path) {
302         return DocPath.create(moduleName + moduleSeparator + path);
303     }
