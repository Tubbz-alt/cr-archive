 26 package jdk.javadoc.internal.doclets.formats.html;
 27 
 28 import jdk.javadoc.internal.doclets.formats.html.markup.HtmlTag;
 29 
 30 /**
 31  * Aliases for HTML heading tags (H1..H6) for different kinds of pages.
 32  *
 33  *  <p><b>This is NOT part of any supported API.
 34  *  If you write code that depends on this, you do so at your own risk.
 35  *  This code and its internal interfaces are subject to change or
 36  *  deletion without notice.</b>
 37  */
 38 class Headings {
 39     /**
 40      * Standard top-level heading for the page title for all pages.
 41      */
 42     static final HtmlTag PAGE_TITLE_HEADING = HtmlTag.H1;
 43 
 44     /**
 45      * Standard second-level heading for sundry pages that do
 46      * not have their own page group.
 47      */
 48     static final HtmlTag CONTENT_HEADING = HtmlTag.H2;
 49 
 50     /**
 51      * Headings for the page for a module declaration.
 52      */
 53     static class ModuleDeclaration {
 54         static final HtmlTag SUMMARY_HEADING = HtmlTag.H2;
 55     }
 56 
 57     /**
 58      * Headings for the page for a type declaration.
 59      * This includes classes, interfaces, enums and annotation types.
 60      */
 61     static class TypeDeclaration {
 62         /**
 63          * Heading for the different summary lists:
 64          * Field Summary, Constructor Summary, Method Summary, etc.
 65          */
 66         static final HtmlTag SUMMARY_HEADING = HtmlTag.H2;
 67 
 68         /**
 69          * Subheading within a summary for the inherited elements:
 70          * inherited methods, etc
 71          */
 72         static final HtmlTag INHERITED_SUMMARY_HEADING = HtmlTag.H3;
 73 
 74         /**
 75          * Heading for the different detail lists:
 76          * Field Details, Constructor Details, Method Details, etc.
 77          */
 78         static final HtmlTag DETAILS_HEADING = HtmlTag.H2;
 79 
 80         /**
 81          * Subheading with a Details list for an individual element.
 82          */
 83         static final HtmlTag MEMBER_HEADING = HtmlTag.H3;
 84     }
 85 
 86     /**
 87      * Headings for the Constants Summary page.
 88      */
 89     static class ConstantsSummary {
 90         static final HtmlTag PACKAGE_HEADING = HtmlTag.H2;
 91     }
 92 
 93     /**
 94      * Headings for the Serialized Form page.
 95      */
 96     static class SerializedForm {
 97         /**
 98          * Heading for the package name, preceding a list of types.
 99          */
100         static final HtmlTag PACKAGE_HEADING = HtmlTag.H2;
101 
102         /**
103          * Heading for a type name within a package.
104          */
105         static final HtmlTag CLASS_HEADING = HtmlTag.H3;
106 
107         /**
108          * Subheading for info within a type.
109          */
110         static final HtmlTag CLASS_SUBHEADING = HtmlTag.H4;
111 
112         /**
113          * Heading for an individual member element within a type.
114          */
115         static final HtmlTag MEMBER_HEADING = HtmlTag.H5;
116     }
117 
118     /**
119      * Headings for a type Use page.
120      */
121     static class TypeUse {
122         static final HtmlTag SUMMARY_HEADING = HtmlTag.H2;
123     }
124 
125     /**
126      * Headings for index frames pages.
127      */
128     static class IndexFrames {
129         /** Heading for a list of module names in an index frame. */
130         static final HtmlTag MODULE_HEADING = HtmlTag.H2;
131         /** Heading for a list of package names in an index frame. */
132         static final HtmlTag PACKAGE_HEADING = HtmlTag.H2;
133     }
134 }
