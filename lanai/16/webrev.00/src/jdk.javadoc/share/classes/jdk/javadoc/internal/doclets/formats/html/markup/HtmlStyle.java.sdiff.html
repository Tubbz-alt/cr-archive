 31 /**

 36  *  This code and its internal interfaces are subject to change or
 37  *  deletion without notice.</b>
 38  *
 39  * @apiNote
 40  * Despite the name, the members of this enum provide values for the HTML {@code class} attribute,

153     private final String cssName;
154 
155     HtmlStyle() {
156         cssName = Pattern.compile("\\p{Upper}")
157                 .matcher(toString())
158                 .replaceAll(mr -> "-" + mr.group().toLowerCase(Locale.US));
159     }
160 
161     HtmlStyle(String cssName) {
162         this.cssName = cssName;
163     }
164 
165     /**

167      * @return the CSS class name
168      */
169     public String cssName() {

171     }
