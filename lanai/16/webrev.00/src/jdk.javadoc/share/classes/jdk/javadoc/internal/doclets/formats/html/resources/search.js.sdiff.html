333                         url += ui.item.p.replace(/\./g, '/') + "/" + ui.item.l + ".html";
334                     }
335                 } else if (ui.item.category === catMembers) {
336                     if (ui.item.p === "<Unnamed>") {
337                         url += ui.item.c + ".html" + "#";

341                     if (ui.item.url) {

343                     } else {
344                         url += ui.item.l;
345                     }
346                 } else if (ui.item.category === catSearchTags) {
347                     url += ui.item.u;
348                 }
