25 package org.graalvm.compiler.hotspot.replacements;
26 
27 import org.graalvm.compiler.hotspot.replacements.HotSpotAllocationSnippets.ProfileContext;
28 import org.graalvm.compiler.options.EnumOptionKey;
29 import org.graalvm.compiler.options.Option;
30 import org.graalvm.compiler.options.OptionType;
31 import org.graalvm.compiler.options.OptionKey;
32 
33 /**
34  * Options related to HotSpot snippets in this package.
35  *
36  * Note: This must be a top level class to work around for
37  * <a href="https://bugs.eclipse.org/bugs/show_bug.cgi?id=477597">Eclipse bug 477597</a>.
38  */
39 public class HotspotSnippetsOptions {
40 
41     // @formatter:off
42     @Option(help = "If the probability that a type check will hit one the profiled types (up to " +
43                    "TypeCheckMaxHints) is below this value, the type check will be compiled without profiling info", type = OptionType.Expert)
44     public static final OptionKey<Double> TypeCheckMinProfileHitProbability = new OptionKey<>(0.5);
45 
46     @Option(help = "The maximum number of profiled types that will be used when compiling a profiled type check. " +
47                     "Note that TypeCheckMinProfileHitProbability also influences whether profiling info is used in compiled type checks.", type = OptionType.Expert)
48     public static final OptionKey<Integer> TypeCheckMaxHints = new OptionKey<>(2);
49 
50     @Option(help = "Use a VM runtime call to load and clear the exception object from the thread at the start of a compiled exception handler.", type = OptionType.Debug)
51     public static final OptionKey<Boolean> LoadExceptionObjectInVM = new OptionKey<>(false);
52 
53     @Option(help = "Enable profiling of allocation sites.", type = OptionType.Debug)
54     public static final OptionKey<Boolean> ProfileAllocations = new OptionKey<>(false);
55 
56     @Option(help = "file:doc-files/ProfileAllocationsContextHelp.txt", type = OptionType.Debug)
57     public static final EnumOptionKey<ProfileContext> ProfileAllocationsContext = new EnumOptionKey<>(ProfileContext.AllocatingMethod);
58 
59     @Option(help = "Enable profiling of monitor operations.", type = OptionType.Debug)
60     public static final OptionKey<Boolean> ProfileMonitors = new OptionKey<>(false);
61 
62     @Option(help = "Handle simple cases for inflated monitors in the fast-path.", type = OptionType.Expert)
63     public static final OptionKey<Boolean> SimpleFastInflatedLocking = new OptionKey<>(true);
64 
65     @Option(help = "Trace monitor operations on objects whose type contains this substring.", type = OptionType.Debug)
66     public static final OptionKey<String> TraceMonitorsTypeFilter = new OptionKey<>(null);
67 
68     @Option(help = "Trace monitor operations in methods whose fully qualified name contains this substring.", type = OptionType.Debug)
69     public static final OptionKey<String> TraceMonitorsMethodFilter = new OptionKey<>(null);
70 
71     @Option(help = "Emit extra code to dynamically check monitor operations are balanced.", type = OptionType.Debug)
72     public static final OptionKey<Boolean> VerifyBalancedMonitors = new OptionKey<>(false);
73     //@formatter:on
74 }
