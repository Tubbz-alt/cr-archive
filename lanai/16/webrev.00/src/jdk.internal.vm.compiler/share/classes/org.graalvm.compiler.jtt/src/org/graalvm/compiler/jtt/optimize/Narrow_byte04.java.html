32 public class Narrow_byte04 extends JTTTest {
33     static final byte NEG_FLAG = (byte) -80;
34 
35     public static byte test(byte b) {
36         byte hBits = (byte) (b & 0xF0);
37         byte loBits = (byte) (b & 0x0F);
38         return hBits == NEG_FLAG ? loBits : b;
39     }
40 
41     @Test
42     public void run0() throws Throwable {
43         runTest("test", ((byte) 0xb1));
44     }
45 
46     @Test
47     public void run1() throws Throwable {
48         runTest("test", ((byte) 0xFF));
49     }
50 
51     @Test
52     public void run3() throws Throwable {
53         runTest("test", ((byte) 0));
54     }
55 }
