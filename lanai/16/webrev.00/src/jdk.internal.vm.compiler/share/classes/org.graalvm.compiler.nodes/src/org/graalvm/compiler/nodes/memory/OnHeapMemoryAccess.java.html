25 package org.graalvm.compiler.nodes.memory;
26 
27 /**
28  *
29  * A special form of {@linkplain MemoryAccess} requiring barrier information for garbage collection.
30  */
31 public interface OnHeapMemoryAccess extends MemoryAccess {
32 
33     /**
34      * The types of (write/read) barriers attached to stores.
35      */
36     enum BarrierType {
37         /**
38          * Primitive access which do not necessitate barriers.
39          */
40         NONE,
41         /**
42          * Array object access.
43          */
44         ARRAY,
45         /**
46          * Field object access.
47          */
48         FIELD,
49         /**
50          * Unknown (aka field or array) object access.
51          */
52         UNKNOWN,
53         /**
54          * Weak field access (e.g. Hotspot's Reference.referent field).
55          */
56         WEAK_FIELD,
57         /**
58          * An access which requires a dynamic check for Weak field access (e.g. Hotspot's
59          * Reference.referent field).
60          */
61         MAYBE_WEAK_FIELD
62     }
63 
64     /**
65      * Gets the write barrier type for that particular access.
66      */
67     BarrierType getBarrierType();
68 
69 }
