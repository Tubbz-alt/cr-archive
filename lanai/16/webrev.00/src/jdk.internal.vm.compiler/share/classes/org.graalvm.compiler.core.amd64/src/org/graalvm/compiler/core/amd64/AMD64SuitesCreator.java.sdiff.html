33 
34 public class AMD64SuitesCreator extends DefaultSuitesCreator {
35 
36     public AMD64SuitesCreator(CompilerConfiguration compilerConfiguration, Plugins plugins) {
37         super(compilerConfiguration, plugins);
38     }
39 




40     @Override
41     public LIRSuites createLIRSuites(OptionValues options) {
42         LIRSuites lirSuites = super.createLIRSuites(options);
43         if (StackMoveOptimizationPhase.Options.LIROptStackMoveOptimizer.getValue(options)) {
44             /* Note: this phase must be inserted <b>after</b> RedundantMoveElimination */
45             lirSuites.getPostAllocationOptimizationStage().appendPhase(new StackMoveOptimizationPhase());
46         }
47         return lirSuites;
48     }
49 }
