 37 // This class represents an arrayOop in the HotSpot virtual
 38 // machine.
 39 static BasicType fixup_element_type(BasicType bt) {
 40   if (is_reference_type(bt))  return T_OBJECT;
 41   if (bt == T_BOOLEAN)  return T_BYTE;
 42   return bt;
 43 }
