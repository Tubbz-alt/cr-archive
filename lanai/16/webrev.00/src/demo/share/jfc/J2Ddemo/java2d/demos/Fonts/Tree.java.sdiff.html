 63     }

 81         at.scale(mindim, mindim);
 82         at.translate(0.5, 0.5);
 83         at.scale(0.3, 0.3);
 84         at.translate(-(Twidth + Rwidth), FontHeight / 4.0);
 85         g2.transform(at);
 86         tree(g2, mindim * 0.3, 0);
