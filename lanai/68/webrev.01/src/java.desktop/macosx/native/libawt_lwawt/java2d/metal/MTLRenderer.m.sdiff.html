 79     // DrawLine implementation same as in OGLRenderer.c
 80     struct Vertex verts[2];
 81     if (y1 == y2) {
 82         // horizontal
 83         float fx1 = (float)x1;

 85         float fy  = ((float)y1) + 0.2f;

 87         if (x1 > x2) {

 95     } else if (x1 == x2) {
 96         // vertical

107         verts[1].position[0] = fx;

109     } else {
