32 SOURCEPATH=src
33 CLASSES=build
34 DIST=dist
35 RESOURCES=resources
36 
37 RENDERPERF_CLASSES = $(CLASSES)/renderperf/RenderPerfTest.class
38 RENDERPERF_SOURCES = $(SOURCEPATH)/renderperf/RenderPerfTest.java
39 
40 RENDERPERF_RESOURCES = $(CLASSES)/renderperf/images/duke.png 
41 
42 all: mkdirs $(DIST)/RenderPerfTest.jar
43 
44 run: mkdirs $(DIST)/RenderPerfTest.jar
45 	java -jar $(DIST)/RenderPerfTest.jar
46 
47 $(DIST)/RenderPerfTest.jar: \
48 	$(RENDERPERF_CLASSES) $(RENDERPERF_RESOURCES) \
49 	$(CLASSES)/renderperf.manifest
50 	jar cvmf $(CLASSES)/renderperf.manifest $(DIST)/RenderPerfTest.jar -C $(CLASSES) .
51 
52 
53 $(CLASSES)/renderperf/images/%: $(RESOURCES)/images/%
54 	cp -r $< $@
55 
56 
57 $(CLASSES)/renderperf.manifest:
58 	echo "Main-Class: renderperf.RenderPerfTest" > $@
59 
60 $(DIST):
61 	mkdir  $(DIST)
62 
63 $(CLASSES):
64 	mkdir $(CLASSES)
65 	mkdir -p $(CLASSES)/renderperf/images
66 
67 mkdirs: $(DIST) $(CLASSES)
68 
69 $(RENDERPERF_CLASSES): $(RENDERPERF_SOURCES)
70 	javac -g:none -d $(CLASSES) -sourcepath $(SOURCEPATH) $<
71 
72 clean:
73 	rm -rf $(CLASSES)
74 	rm -rf $(DIST)
