 47 static jboolean MTLSurfaceData_initTexture(BMTLSDOps *bmtlsdo, jboolean isOpaque, jboolean rtt, jint width, jint height) {
 48     @autoreleasepool {
 49         if (bmtlsdo == NULL) {
 50             J2dRlsTraceLn(J2D_TRACE_ERROR, "MTLSurfaceData_initTexture: ops are null");
 51             return JNI_FALSE;
 52         }
 53         if (width <= 0 || height <= 0) {
 54             J2dRlsTraceLn2(J2D_TRACE_ERROR, "MTLSurfaceData_initTexture: texture dimensions is incorrect, w=%d, h=%d", width, height);
 55             return JNI_FALSE;
 56         }
 57 
 58         MTLSDOps *mtlsdo = (MTLSDOps *)bmtlsdo->privOps;
 59 
 60         if (mtlsdo == NULL) {
 61             J2dRlsTraceLn(J2D_TRACE_ERROR, "MTLSurfaceData_initTexture: MTLSDOps are null");
 62             return JNI_FALSE;
 63         }
 64         if (mtlsdo->configInfo == NULL || mtlsdo->configInfo->context == NULL) {
 65             J2dRlsTraceLn(J2D_TRACE_ERROR, "MTLSurfaceData_initTexture: MTLSDOps wasn't initialized (context is null)");
 66             return JNI_FALSE;
 67         }
 68 
 69         MTLContext* ctx = mtlsdo->configInfo->context;
 70 
 71         MTLTextureDescriptor *textureDescriptor = [MTLTextureDescriptor texture2DDescriptorWithPixelFormat: MTLPixelFormatBGRA8Unorm width: width height: height mipmapped: NO];
 72         textureDescriptor.usage = MTLTextureUsageUnknown;
 73         bmtlsdo->pTexture = [ctx.device newTextureWithDescriptor: textureDescriptor];
 74 
 75         MTLTextureDescriptor *stencilDataDescriptor =
 76             [MTLTextureDescriptor texture2DDescriptorWithPixelFormat:MTLPixelFormatR8Uint width:width height:height mipmapped:NO];
 77         stencilDataDescriptor.usage = MTLTextureUsageRenderTarget | MTLTextureUsageShaderRead;
 78         stencilDataDescriptor.storageMode = MTLStorageModePrivate;
 79         bmtlsdo->pStencilData = [ctx.device newTextureWithDescriptor:stencilDataDescriptor];
 80         bmtlsdo->pAAStencilData = [ctx.device newTextureWithDescriptor:textureDescriptor];
 81         bmtlsdo->pStencilDataBuf = [ctx.device newBufferWithLength:width*height options:MTLResourceStorageModePrivate];
 82         bmtlsdo->pAAStencilDataBuf = [ctx.device newBufferWithLength:width*height*4 options:MTLResourceStorageModePrivate];
 83 
 84 
 85         MTLTextureDescriptor *stencilTextureDescriptor =
 86             [MTLTextureDescriptor texture2DDescriptorWithPixelFormat:MTLPixelFormatStencil8 width:width height:height mipmapped:NO];
 87         stencilTextureDescriptor.usage = MTLTextureUsageRenderTarget | MTLTextureUsageShaderRead | MTLTextureUsageShaderWrite;
 88         stencilTextureDescriptor.storageMode = MTLStorageModePrivate;
 89         bmtlsdo->pStencilTexture = [ctx.device newTextureWithDescriptor:stencilTextureDescriptor];
 90 
 91         MTLTextureDescriptor *lcdTextureDescriptor = 
 92             [MTLTextureDescriptor texture2DDescriptorWithPixelFormat:MTLPixelFormatBGRA8Unorm width:width height:height mipmapped:NO];
 93 
 94         bmtlsdo->textureLCD = [ctx.device newTextureWithDescriptor:lcdTextureDescriptor];
 95 
 96         bmtlsdo->isOpaque = isOpaque;
 97         bmtlsdo->xOffset = 0;
 98         bmtlsdo->yOffset = 0;
 99         bmtlsdo->width = width;
100         bmtlsdo->height = height;
101         bmtlsdo->textureWidth = width;
102         bmtlsdo->textureHeight = height;
103         bmtlsdo->textureTarget = -1;
104         bmtlsdo->drawableType = rtt ? MTLSD_RT_TEXTURE : MTLSD_TEXTURE;
105 
106         J2dTraceLn6(J2D_TRACE_VERBOSE, "MTLSurfaceData_initTexture: w=%d h=%d bp=%p [tex=%p] opaque=%d rtt=%d", width, height, bmtlsdo, bmtlsdo->pTexture, isOpaque, rtt);
107         return JNI_TRUE;
108     }
109 }

169 /**
170  * Initializes nativeWidth/Height fields of the surfaceData object with
171  * passed arguments.
172  */
173 void
174 MTLSD_SetNativeDimensions(JNIEnv *env, BMTLSDOps *mtlsdo,
175                           jint width, jint height)
176 {
177     jobject sdObject;
178 
179     sdObject = (*env)->NewLocalRef(env, mtlsdo->sdOps.sdObject);
180     if (sdObject == NULL) {
181         return;
182     }
183 
184     JNU_SetFieldByName(env, NULL, sdObject, "nativeWidth", "I", width);
185     if (!((*env)->ExceptionOccurred(env))) {
186         JNU_SetFieldByName(env, NULL, sdObject, "nativeHeight", "I", height);
187     }
188 
189     (*env)->DeleteLocalRef(env, sdObject);
190 }

192 /**
193  * Deletes native OpenGL resources associated with this surface.
194  */
195 void
196 MTLSD_Delete(JNIEnv *env, BMTLSDOps *bmtlsdo)
197 {
198     J2dTraceLn3(J2D_TRACE_VERBOSE, "MTLSD_Delete: type=%d %p [tex=%p]", bmtlsdo->drawableType, bmtlsdo, bmtlsdo->pTexture);
199     if (bmtlsdo->drawableType == MTLSD_WINDOW) {
200         MTLSD_DestroyMTLSurface(env, bmtlsdo);
201     } else if (
202             bmtlsdo->drawableType == MTLSD_RT_TEXTURE
203             || bmtlsdo->drawableType == MTLSD_TEXTURE
204             || bmtlsdo->drawableType == MTLSD_FLIP_BACKBUFFER
205     ) {
206         [(NSObject *)bmtlsdo->pTexture release];
207         [(NSObject *)bmtlsdo->pStencilTexture release];
208         [(NSObject *)bmtlsdo->pStencilData release];
209         [(NSObject *)bmtlsdo->pStencilDataBuf release];
210         [(NSObject *)bmtlsdo->pAAStencilData release];
211         [(NSObject *)bmtlsdo->pAAStencilDataBuf release];
212         bmtlsdo->pTexture = NULL;
213         bmtlsdo->drawableType = MTLSD_UNDEFINED;
214     }
215 }

217 /**
218  * This is the implementation of the general DisposeFunc defined in
219  * SurfaceData.h and used by the Disposer mechanism.  It first flushes all
220  * native OpenGL resources and then frees any memory allocated within the
221  * native MTLSDOps structure.
222  */
223 void
224 MTLSD_Dispose(JNIEnv *env, SurfaceDataOps *ops)
225 {
226     BMTLSDOps *bmtlsdo = (BMTLSDOps *)ops;
227     jobject graphicsConfig = bmtlsdo->graphicsConfig;
228 
229     JNU_CallStaticMethodByName(env, NULL, "sun/java2d/metal/MTLSurfaceData",
230                                "dispose",
231                                "(JLsun/java2d/metal/MTLGraphicsConfig;)V",
232                                ptr_to_jlong(ops), graphicsConfig);
233     (*env)->DeleteGlobalRef(env, graphicsConfig);
234     bmtlsdo->graphicsConfig = NULL;
235 }

292 /**
293  * This function initializes a native window surface and caches the window
294  * bounds in the given BMTLSDOps.  Returns JNI_TRUE if the operation was
295  * successful; JNI_FALSE otherwise.
296  */
297 jboolean
298 MTLSD_InitMTLWindow(JNIEnv *env, BMTLSDOps *bmtlsdo)
299 {
300     if (bmtlsdo == NULL) {
301         J2dRlsTraceLn(J2D_TRACE_ERROR, "MTLSD_InitMTLWindow: ops are null");
302         return JNI_FALSE;
303     }
304 
305     MTLSDOps *mtlsdo = (MTLSDOps *)bmtlsdo->privOps;
306     if (mtlsdo == NULL) {
307         J2dRlsTraceLn(J2D_TRACE_ERROR, "MTLSD_InitMTLWindow: priv ops are null");
308         return JNI_FALSE;
309     }
310 
311     AWTView *v = mtlsdo->peerData;
312     if (v == NULL) {
313         J2dRlsTraceLn(J2D_TRACE_ERROR, "MTLSD_InitMTLWindow: view is invalid");
314         return JNI_FALSE;
315     }
316 
317     JNF_COCOA_ENTER(env);
318             NSRect surfaceBounds = [v bounds];
319             bmtlsdo->drawableType = MTLSD_WINDOW;
320             bmtlsdo->isOpaque = JNI_TRUE;
321             bmtlsdo->width = surfaceBounds.size.width;
322             bmtlsdo->height = surfaceBounds.size.height;
323     JNF_COCOA_EXIT(env);
324 
325     J2dTraceLn2(J2D_TRACE_VERBOSE, "  created window: w=%d h=%d", bmtlsdo->width, bmtlsdo->height);
326     return JNI_TRUE;
327 }

343 JNIEXPORT void JNICALL
344 Java_sun_java2d_metal_MTLSurfaceData_initOps
345     (JNIEnv *env, jobject mtlsd, jobject gc,
346      jlong pConfigInfo, jlong pPeerData, jlong layerPtr,
347      jint xoff, jint yoff, jboolean isOpaque)
348 {
349     BMTLSDOps *bmtlsdo = (BMTLSDOps *)SurfaceData_InitOps(env, mtlsd, sizeof(BMTLSDOps));
350     MTLSDOps *mtlsdo = (MTLSDOps *)malloc(sizeof(MTLSDOps));
351 
352     J2dTraceLn1(J2D_TRACE_INFO, "MTLSurfaceData_initOps p=%p", bmtlsdo);
353     J2dTraceLn1(J2D_TRACE_INFO, "  pPeerData=%p", jlong_to_ptr(pPeerData));
354     J2dTraceLn1(J2D_TRACE_INFO, "  layerPtr=%p", jlong_to_ptr(layerPtr));
355     J2dTraceLn2(J2D_TRACE_INFO, "  xoff=%d, yoff=%d", (int)xoff, (int)yoff);
356 
357     gc = (*env)->NewGlobalRef(env, gc);
358     if (gc == NULL) {
359         JNU_ThrowOutOfMemoryError(env, "Initialization of SurfaceData failed.");
360         return;
361     }
362 
363     if (mtlsdo == NULL) {
364         (*env)->DeleteGlobalRef(env, gc);
365         JNU_ThrowOutOfMemoryError(env, "Initialization of SurfaceData failed.");
366         return;
367     }
368 
369     // later the graphicsConfig will be used for deallocation of mtlsdo
370     bmtlsdo->privOps = mtlsdo;
371     bmtlsdo->graphicsConfig = gc;
372 
373     bmtlsdo->sdOps.Lock               = MTLSD_Lock;
374     bmtlsdo->sdOps.GetRasInfo         = MTLSD_GetRasInfo;
375     bmtlsdo->sdOps.Unlock             = MTLSD_Unlock;
376     bmtlsdo->sdOps.Dispose            = MTLSD_Dispose;
377 
378     bmtlsdo->drawableType = MTLSD_UNDEFINED;
379     bmtlsdo->needsInit = JNI_TRUE;
380     bmtlsdo->xOffset = xoff;
381     bmtlsdo->yOffset = yoff;
382     bmtlsdo->isOpaque = isOpaque;
383 
384     mtlsdo->peerData = (AWTView *)jlong_to_ptr(pPeerData);
385     mtlsdo->layer = (MTLLayer *)jlong_to_ptr(layerPtr);
386     mtlsdo->configInfo = (MTLGraphicsConfigInfo *)jlong_to_ptr(pConfigInfo);
387 
388     if (mtlsdo->configInfo == NULL) {
389         free(mtlsdo);
390         JNU_ThrowNullPointerException(env, "Config info is null in initOps");
391     }
392 }
