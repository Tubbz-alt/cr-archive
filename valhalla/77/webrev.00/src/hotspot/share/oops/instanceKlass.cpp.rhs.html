 140 static inline bool is_class_loader(const Symbol* class_name,
 141                                    const ClassFileParser& parser) {
 142   assert(class_name != NULL, "invariant");
 143 
 144   if (class_name == vmSymbols::java_lang_ClassLoader()) {
 145     return true;
 146   }
 147 
 148   if (SystemDictionary::ClassLoader_klass_loaded()) {
 149     const Klass* const super_klass = parser.super_klass();
 150     if (super_klass != NULL) {
 151       if (super_klass->is_subtype_of(SystemDictionary::ClassLoader_klass())) {
 152         return true;
 153       }
 154     }
 155   }
 156   return false;
 157 }

 161 // private: called to verify that k is a static member of this nest.
 162 // We know that k is an instance class in the same package and hence the
 163 // same classloader.
 164 bool InstanceKlass::has_nest_member(InstanceKlass* k, TRAPS) const {
 165   assert(!is_hidden(), "unexpected hidden class");
 166   if (_nest_members == NULL || _nest_members == Universe::the_empty_short_array()) {
 167     if (log_is_enabled(Trace, class, nestmates)) {
 168       ResourceMark rm(THREAD);
 169       log_trace(class, nestmates)("Checked nest membership of %s in non-nest-host class %s",
 170                                   k->external_name(), this->external_name());
 171     }
 172     return false;
 173   }
 174 
 175   if (log_is_enabled(Trace, class, nestmates)) {
 176     ResourceMark rm(THREAD);
 177     log_trace(class, nestmates)("Checking nest membership of %s in %s",
 178                                 k->external_name(), this->external_name());
 179   }
 180 
 181   // Check for a resolved cp entry , else fall back to a name check.
 182   // We don't want to resolve any class other than the one being checked.
 183   for (int i = 0; i < _nest_members->length(); i++) {
 184     int cp_index = _nest_members->at(i);
 185     if (_constants->tag_at(cp_index).is_klass()) {
 186       Klass* k2 = _constants->klass_at(cp_index, THREAD);
 187       assert(!HAS_PENDING_EXCEPTION || PENDING_EXCEPTION->is_a(SystemDictionary::VirtualMachineError_klass()),
 188              "Exceptions should not be possible here");
 189       if (k2 == k) {
 190         log_trace(class, nestmates)("- class is listed at nest_members[%d] => cp[%d]", i, cp_index);
 191         return true;
 192       }
 193     }
 194     else {
 195       Symbol* name = _constants->klass_name_at(cp_index);
 196       if (name == k->name()) {
 197         log_trace(class, nestmates)("- Found it at nest_members[%d] => cp[%d]", i, cp_index);
 198 
 199         // Names match so check actual klass. This may trigger class loading if
 200         // it doesn't match though that should be impossible as it means one classloader
 201         // has defined two different classes with the same name! A compiler thread won't be
 202         // able to perform that loading but we can't exclude the compiler threads from
 203         // executing this logic. But it should actually be impossible to trigger loading here.
 204         Klass* k2 = _constants->klass_at(cp_index, THREAD);
 205         assert(!HAS_PENDING_EXCEPTION || PENDING_EXCEPTION->is_a(SystemDictionary::VirtualMachineError_klass()),
 206                "Exceptions should not be possible here");
 207         if (k2 == k) {
 208           log_trace(class, nestmates)("- class is listed as a nest member");
 209           return true;
 210         }
 211         else {
 212           // same name but different klass!
 213           log_trace(class, nestmates)(" - klass comparison failed!");
 214           // can't have two names the same, so we're done
 215           return false;
 216         }
 217       }
 218     }
 219   }
 220   log_trace(class, nestmates)("- class is NOT a nest member!");
 221   return false;
 222 }

 224 // Return nest-host class, resolving, validating and saving it if needed.
 225 // In cases where this is called from a thread that cannot do classloading
 226 // (such as a native JIT thread) then we simply return NULL, which in turn
 227 // causes the access check to return false. Such code will retry the access
 228 // from a more suitable environment later. Otherwise the _nest_host is always
 229 // set once this method returns.
 230 // Any errors from nest-host resolution must be preserved so they can be queried
 231 // from higher-level access checking code, and reported as part of access checking
 232 // exceptions.
 233 // VirtualMachineErrors are propagated with a NULL return.
 234 // Under any conditions where the _nest_host can be set to non-NULL the resulting
 235 // value of it and, if applicable, the nest host resolution/validation error,
 236 // are idempotent.
 237 InstanceKlass* InstanceKlass::nest_host(TRAPS) {
 238   InstanceKlass* nest_host_k = _nest_host;
 239   if (nest_host_k != NULL) {
 240     return nest_host_k;
 241   }
 242 
 243   ResourceMark rm(THREAD);
 244 
 245   // need to resolve and save our nest-host class.
 246   if (_nest_host_index != 0) { // we have a real nest_host
 247     // Before trying to resolve check if we're in a suitable context
 248     if (!THREAD->can_call_java() && !_constants->tag_at(_nest_host_index).is_klass()) {
 249       log_trace(class, nestmates)("Rejected resolution of nest-host of %s in unsuitable thread",
 250                                   this->external_name());
 251       return NULL; // sentinel to say "try again from a different context"
 252     }
 253 
 254     log_trace(class, nestmates)("Resolving nest-host of %s using cp entry for %s",
 255                                 this->external_name(),
 256                                 _constants->klass_name_at(_nest_host_index)->as_C_string());
 257 
 258     Klass* k = _constants->klass_at(_nest_host_index, THREAD);
 259     if (HAS_PENDING_EXCEPTION) {
 260       if (PENDING_EXCEPTION->is_a(SystemDictionary::VirtualMachineError_klass())) {
 261         return NULL; // propagate VMEs
 262       }
 263       stringStream ss;
 264       char* target_host_class = _constants->klass_name_at(_nest_host_index)->as_C_string();
 265       ss.print("Nest host resolution of %s with host %s failed: ",
 266                this->external_name(), target_host_class);
 267       java_lang_Throwable::print(PENDING_EXCEPTION, &ss);
 268       const char* msg = ss.as_string(true /* on C-heap */);
 269       constantPoolHandle cph(THREAD, constants());
 270       SystemDictionary::add_nest_host_error(cph, _nest_host_index, msg);
 271       CLEAR_PENDING_EXCEPTION;
 272 
 273       log_trace(class, nestmates)("%s", msg);
 274     } else {
 275       // A valid nest-host is an instance class in the current package that lists this
 276       // class as a nest member. If any of these conditions are not met the class is
 277       // its own nest-host.
 278       const char* error = NULL;
 279 
 280       // JVMS 5.4.4 indicates package check comes first
 281       if (is_same_class_package(k)) {
 282         // Now check actual membership. We can't be a member if our "host" is
 283         // not an instance class.
 284         if (k->is_instance_klass()) {
 285           nest_host_k = InstanceKlass::cast(k);
 286           bool is_member = nest_host_k->has_nest_member(this, THREAD);
 287           // exception is rare, perhaps impossible
 288           if (!HAS_PENDING_EXCEPTION) {
 289             if (is_member) {
 290               _nest_host = nest_host_k; // save resolved nest-host value
 291 
 292               log_trace(class, nestmates)("Resolved nest-host of %s to %s",
 293                                           this->external_name(), k->external_name());
 294               return nest_host_k;
 295             } else {
 296               error = "current type is not listed as a nest member";
 297             }
 298           } else {
 299             if (PENDING_EXCEPTION->is_a(SystemDictionary::VirtualMachineError_klass())) {
 300               return NULL; // propagate VMEs
 301             }
 302             stringStream ss;
 303             ss.print("exception on member check: ");
 304             java_lang_Throwable::print(PENDING_EXCEPTION, &ss);
 305             error = ss.as_string();
 306           }
 307         } else {
 308           error = "host is not an instance class";
 309         }
 310       } else {
 311         error = "types are in different packages";
 312       }
 313 
 314       // something went wrong, so record what and log it
 315       {
 316         stringStream ss;
 317         ss.print("Type %s (loader: %s) is not a nest member of type %s (loader: %s): %s",
 318                  this->external_name(),
 319                  this->class_loader_data()->loader_name_and_id(),
 320                  k->external_name(),
 321                  k->class_loader_data()->loader_name_and_id(),
 322                  error);
 323         const char* msg = ss.as_string(true /* on C-heap */);
 324         constantPoolHandle cph(THREAD, constants());
 325         SystemDictionary::add_nest_host_error(cph, _nest_host_index, msg);
 326         log_trace(class, nestmates)("%s", msg);
 327       }
 328     }
 329   } else {
 330     log_trace(class, nestmates)("Type %s is not part of a nest: setting nest-host to self",
 331                                 this->external_name());
 332   }
 333 
 334   // Either not in an explicit nest, or else an error occurred, so
 335   // the nest-host is set to `this`. Any thread that sees this assignment
 336   // will also see any setting of nest_host_error(), if applicable.
 337   return (_nest_host = this);
 338 }

 340 // Dynamic nest member support: set this class's nest host to the given class.
 341 // This occurs as part of the class definition, as soon as the instanceKlass
 342 // has been created and doesn't require further resolution. The code:
 343 //    lookup().defineHiddenClass(bytes_for_X, NESTMATE);
 344 // results in:
 345 //    class_of_X.set_nest_host(lookup().lookupClass().getNestHost())
 346 // If it has an explicit _nest_host_index or _nest_members, these will be ignored.
 347 // We also know the "host" is a valid nest-host in the same package so we can
 348 // assert some of those facts.
 349 void InstanceKlass::set_nest_host(InstanceKlass* host, TRAPS) {
 350   assert(is_hidden(), "must be a hidden class");
 351   assert(host != NULL, "NULL nest host specified");
 352   assert(_nest_host == NULL, "current class has resolved nest-host");
 353   assert(nest_host_error(THREAD) == NULL, "unexpected nest host resolution error exists: %s",
 354          nest_host_error(THREAD));
 355   assert((host->_nest_host == NULL && host->_nest_host_index == 0) ||
 356          (host->_nest_host == host), "proposed host is not a valid nest-host");
 357   // Can't assert this as package is not set yet:
 358   // assert(is_same_class_package(host), "proposed host is in wrong package");
 359 
 360   if (log_is_enabled(Trace, class, nestmates)) {
 361     ResourceMark rm(THREAD);
 362     const char* msg = "";
 363     // a hidden class does not expect a statically defined nest-host
 364     if (_nest_host_index > 0) {
 365       msg = "(the NestHost attribute in the current class is ignored)";
 366     } else if (_nest_members != NULL && _nest_members != Universe::the_empty_short_array()) {
 367       msg = "(the NestMembers attribute in the current class is ignored)";
 368     }
 369     log_trace(class, nestmates)("Injected type %s into the nest of %s %s",
 370                                 this->external_name(),
 371                                 host->external_name(),
 372                                 msg);
 373   }
 374   // set dynamic nest host
 375   _nest_host = host;
 376   // Record dependency to keep nest host from being unloaded before this class.
 377   ClassLoaderData* this_key = class_loader_data();
 378   this_key->record_dependency(host);
 379 }

 381 // check if 'this' and k are nestmates (same nest_host), or k is our nest_host,
 382 // or we are k's nest_host - all of which is covered by comparing the two
 383 // resolved_nest_hosts.
 384 // Any exceptions (i.e. VMEs) are propagated.
 385 bool InstanceKlass::has_nestmate_access_to(InstanceKlass* k, TRAPS) {
 386 
 387   assert(this != k, "this should be handled by higher-level code");
 388 
 389   // Per JVMS 5.4.4 we first resolve and validate the current class, then
 390   // the target class k.
 391 
 392   InstanceKlass* cur_host = nest_host(CHECK_false);
 393   if (cur_host == NULL) {
 394     return false;
 395   }
 396 
 397   Klass* k_nest_host = k->nest_host(CHECK_false);
 398   if (k_nest_host == NULL) {
 399     return false;
 400   }
 401 
 402   bool access = (cur_host == k_nest_host);
 403 
 404   ResourceMark rm(THREAD);
 405   log_trace(class, nestmates)("Class %s does %shave nestmate access to %s",
 406                               this->external_name(),
 407                               access ? "" : "NOT ",
 408                               k->external_name());
 409   return access;
 410 }

 414     return NULL;
 415   } else {

 418   }

 421 InstanceKlass* InstanceKlass::allocate_instance_klass(const ClassFileParser& parser, TRAPS) {
 422   bool is_hidden_or_anonymous = parser.is_hidden() || parser.is_unsafe_anonymous();
 423   const int size = InstanceKlass::size(parser.vtable_size(),
 424                                        parser.itable_size(),
 425                                        nonstatic_oop_map_size(parser.total_oop_map_count()),
 426                                        parser.is_interface(),
 427                                        parser.is_unsafe_anonymous(),
 428                                        should_store_fingerprint(is_hidden_or_anonymous),
 429                                        parser.has_inline_fields() ? parser.java_fields_count() : 0,
 430                                        parser.is_inline_type());
 431 
 432   const Symbol* const class_name = parser.class_name();
 433   assert(class_name != NULL, "invariant");
 434   ClassLoaderData* loader_data = parser.loader_data();
 435   assert(loader_data != NULL, "invariant");
 436 
 437   InstanceKlass* ik;
 438 
 439   // Allocation
 440   if (REF_NONE == parser.reference_type()) {
 441     if (class_name == vmSymbols::java_lang_Class()) {
 442       // mirror
 443       ik = new (loader_data, size, THREAD) InstanceMirrorKlass(parser);
 444     } else if (is_class_loader(class_name, parser)) {
 445       // class loader
 446       ik = new (loader_data, size, THREAD) InstanceClassLoaderKlass(parser);
 447     } else if (parser.is_inline_type()) {
 448       // inline type
 449       ik = new (loader_data, size, THREAD) ValueKlass(parser);
 450     } else {
 451       // normal
 452       ik = new (loader_data, size, THREAD) InstanceKlass(parser, InstanceKlass::_kind_other);
 453     }
 454   } else {
 455     // reference
 456     ik = new (loader_data, size, THREAD) InstanceRefKlass(parser);
 457   }
 458 
 459   // Check for pending exception before adding to the loader data and incrementing
 460   // class count.  Can get OOM here.
 461   if (HAS_PENDING_EXCEPTION) {
 462     return NULL;
 463   }
 464 
 465 #ifdef ASSERT
 466   assert(ik->size() == size, "");
 467   ik->bounds_check((address) ik->start_of_vtable(), false, size);
 468   ik->bounds_check((address) ik->start_of_itable(), false, size);
 469   ik->bounds_check((address) ik->end_of_itable(), true, size);
 470   ik->bounds_check((address) ik->end_of_nonstatic_oop_maps(), true, size);
 471 #endif //ASSERT
 472   return ik;
 473 }

 494   Verbose = WizardMode = true; this->print(); //@@

 520 InstanceKlass::InstanceKlass(const ClassFileParser& parser, unsigned kind, KlassID id) :
 521   Klass(id),
 522   _nest_members(NULL),
 523   _nest_host(NULL),
 524   _record_components(NULL),
 525   _static_field_size(parser.static_field_size()),
 526   _nonstatic_oop_map_size(nonstatic_oop_map_size(parser.total_oop_map_count())),
 527   _itable_len(parser.itable_size()),
 528   _nest_host_index(0),
 529   _init_state(allocated),
 530   _reference_type(parser.reference_type()),
 531   _init_thread(NULL),
 532   _value_field_klasses(NULL),
 533   _adr_valueklass_fixed_block(NULL)
 534 {
 535   set_vtable_length(parser.vtable_size());
 536   set_kind(kind);
 537   set_access_flags(parser.access_flags());
 538   if (parser.is_hidden()) set_is_hidden();
 539   set_is_unsafe_anonymous(parser.is_unsafe_anonymous());
 540   set_layout_helper(Klass::instance_layout_helper(parser.layout_size(),
 541                                                     false));
 542     if (parser.has_inline_fields()) {
 543       set_has_inline_fields();
 544     }
 545     _java_fields_count = parser.java_fields_count();
 546 
 547     assert(NULL == _methods, "underlying memory not zeroed?");
 548     assert(is_instance_klass(), "is layout incorrect?");
 549     assert(size_helper() == parser.layout_size(), "incorrect size_helper?");
 550 
 551   // Set biased locking bit for all instances of this class; it will be
 552   // cleared if revocation occurs too often for this type
 553   if (UseBiasedLocking && BiasedLocking::enabled()) {
 554     set_prototype_header(markWord::biased_locking_prototype());

 613 // This function deallocates the metadata and C heap pointers that the
 614 // InstanceKlass points to.
 615 void InstanceKlass::deallocate_contents(ClassLoaderData* loader_data) {
 616 
 617   // Orphan the mirror first, CMS thinks it's still live.
 618   if (java_mirror() != NULL) {
 619     java_lang_Class::set_klass(java_mirror(), NULL);
 620   }
 621 
 622   // Also remove mirror from handles
 623   loader_data->remove_handle(_java_mirror);
 624 
 625   // Need to take this class off the class loader data list.
 626   loader_data->remove_class(this);
 627 
 628   // The array_klass for this class is created later, after error handling.
 629   // For class redefinition, we keep the original class so this scratch class
 630   // doesn't have an array class.  Either way, assert that there is nothing
 631   // to deallocate.
 632   assert(array_klasses() == NULL, "array classes shouldn't be created for this class yet");
 633 
 634   // Release C heap allocated data that this points to, which includes
 635   // reference counting symbol names.
 636   release_C_heap_structures_internal();
 637 
 638   deallocate_methods(loader_data, methods());
 639   set_methods(NULL);
 640 
 641   deallocate_record_components(loader_data, record_components());
 642   set_record_components(NULL);
 643 
 644   if (method_ordering() != NULL &&
 645       method_ordering() != Universe::the_empty_int_array() &&
 646       !method_ordering()->is_shared()) {
 647     MetadataFactory::free_array<int>(loader_data, method_ordering());
 648   }
 649   set_method_ordering(NULL);
 650 
 651   // default methods can be empty
 652   if (default_methods() != NULL &&
 653       default_methods() != Universe::the_empty_method_array() &&
 654       !default_methods()->is_shared()) {
 655     MetadataFactory::free_array<Method*>(loader_data, default_methods());
 656   }
 657   // Do NOT deallocate the default methods, they are owned by superinterfaces.

 664   }
 665   set_default_vtable_indices(NULL);
 666 
 667 
 668   // This array is in Klass, but remove it with the InstanceKlass since
 669   // this place would be the only caller and it can share memory with transitive
 670   // interfaces.
 671   if (secondary_supers() != NULL &&
 672       secondary_supers() != Universe::the_empty_klass_array() &&
 673       // see comments in compute_secondary_supers about the following cast
 674       (address)(secondary_supers()) != (address)(transitive_interfaces()) &&

 676     MetadataFactory::free_array<Klass*>(loader_data, secondary_supers());
 677   }
 678   set_secondary_supers(NULL);
 679 
 680   deallocate_interfaces(loader_data, super(), local_interfaces(), transitive_interfaces());
 681   set_transitive_interfaces(NULL);
 682   set_local_interfaces(NULL);
 683 
 684   if (fields() != NULL && !fields()->is_shared()) {
 685     MetadataFactory::free_array<jushort>(loader_data, fields());
 686   }
 687   set_fields(NULL, 0);
 688 
 689   // If a method from a redefined class is using this constant pool, don't
 690   // delete it, yet.  The new class's previous version will point to this.
 691   if (constants() != NULL) {
 692     assert (!constants()->on_stack(), "shouldn't be called if anything is onstack");
 693     if (!constants()->is_shared()) {
 694       MetadataFactory::free_metadata(loader_data, constants());
 695     }
 696     // Delete any cached resolution errors for the constant pool
 697     SystemDictionary::delete_resolution_error(constants());
 698 
 699     set_constants(NULL);
 700   }
 701 
 702   if (inner_classes() != NULL &&
 703       inner_classes() != Universe::the_empty_short_array() &&
 704       !inner_classes()->is_shared()) {
 705     MetadataFactory::free_array<jushort>(loader_data, inner_classes());
 706   }
 707   set_inner_classes(NULL);
 708 
 709   if (nest_members() != NULL &&
 710       nest_members() != Universe::the_empty_short_array() &&
 711       !nest_members()->is_shared()) {
 712     MetadataFactory::free_array<jushort>(loader_data, nest_members());
 713   }
 714   set_nest_members(NULL);
 715 
 716   // We should deallocate the Annotations instance if it's not in shared spaces.
 717   if (annotations() != NULL && !annotations()->is_shared()) {
 718     MetadataFactory::free_metadata(loader_data, annotations());
 719   }
 720   set_annotations(NULL);
 721 
 722   if (Arguments::is_dumping_archive()) {
 723     SystemDictionaryShared::remove_dumptime_info(this);
 724   }

 735 void InstanceKlass::eager_initialize(Thread *thread) {
 736   if (!EagerInitialization) return;
 737 
 738   if (this->is_not_initialized()) {
 739     // abort if the the class has a class initializer

 751   }

 789 void InstanceKlass::eager_initialize_impl() {
 790   EXCEPTION_MARK;
 791   HandleMark hm(THREAD);
 792   Handle h_init_lock(THREAD, init_lock());
 793   ObjectLocker ol(h_init_lock, THREAD, h_init_lock() != NULL);
 794 
 795   // abort if someone beat us to the initialization
 796   if (!is_not_initialized()) return;  // note: not equivalent to is_initialized()
 797 
 798   ClassState old_state = init_state();
 799   link_class_impl(THREAD);
 800   if (HAS_PENDING_EXCEPTION) {
 801     CLEAR_PENDING_EXCEPTION;
 802     // Abort if linking the class throws an exception.
 803 
 804     // Use a test to avoid redundantly resetting the state if there's
 805     // no change.  Set_init_state() asserts that state changes make
 806     // progress, whereas here we might just be spinning in place.
 807     if (old_state != _init_state)
 808       set_init_state(old_state);
 809   } else {
 810     // linking successfull, mark class as initialized
 811     set_init_state(fully_initialized);
 812     fence_and_clear_init_lock();
 813     // trace
 814     if (log_is_enabled(Info, class, init)) {
 815       ResourceMark rm(THREAD);
 816       log_info(class, init)("[Initialized %s without side effects]", external_name());
 817     }
 818   }
 819 }

 859 bool InstanceKlass::link_class_impl(TRAPS) {
 860   if (DumpSharedSpaces && SystemDictionaryShared::has_class_failed_verification(this)) {
 861     // This is for CDS dumping phase only -- we use the in_error_state to indicate that
 862     // the class has failed verification. Throwing the NoClassDefFoundError here is just
 863     // a convenient way to stop repeat attempts to verify the same (bad) class.
 864     //
 865     // Note that the NoClassDefFoundError is not part of the JLS, and should not be thrown
 866     // if we are executing Java code. This is not a problem for CDS dumping phase since
 867     // it doesn't execute any Java code.
 868     ResourceMark rm(THREAD);
 869     Exceptions::fthrow(THREAD_AND_LOCATION,
 870                        vmSymbols::java_lang_NoClassDefFoundError(),
 871                        "Class %s, or one of its supertypes, failed class initialization",
 872                        external_name());
 873     return false;
 874   }
 875   // return if already verified
 876   if (is_linked()) {
 877     return true;
 878   }
 879 
 880   // Timing
 881   // timer handles recursion
 882   assert(THREAD->is_Java_thread(), "non-JavaThread in link_class_impl");
 883   JavaThread* jt = (JavaThread*)THREAD;
 884 
 885   // link super class before linking this class
 886   Klass* super_klass = super();
 887   if (super_klass != NULL) {
 888     if (super_klass->is_interface()) {  // check if super class is an interface
 889       ResourceMark rm(THREAD);
 890       Exceptions::fthrow(
 891         THREAD_AND_LOCATION,
 892         vmSymbols::java_lang_IncompatibleClassChangeError(),
 893         "class %s has interface %s as super class",
 894         external_name(),
 895         super_klass->external_name()
 896       );
 897       return false;
 898     }
 899 
 900     InstanceKlass* ik_super = InstanceKlass::cast(super_klass);
 901     ik_super->link_class_impl(CHECK_false);
 902   }
 903 
 904   // link all interfaces implemented by this class before linking this class
 905   Array<InstanceKlass*>* interfaces = local_interfaces();
 906   int num_interfaces = interfaces->length();
 907   for (int index = 0; index < num_interfaces; index++) {
 908     InstanceKlass* interk = interfaces->at(index);
 909     interk->link_class_impl(CHECK_false);
 910   }
 911 
 912 
 913   // If a class declares a method that uses an inline class as an argument
 914   // type or return inline type, this inline class must be loaded during the
 915   // linking of this class because size and properties of the inline class
 916   // must be known in order to be able to perform inline type optimizations.
 917   // The implementation below is an approximation of this rule, the code
 918   // iterates over all methods of the current class (including overridden
 919   // methods), not only the methods declared by this class. This
 920   // approximation makes the code simpler, and doesn't change the semantic
 921   // because classes declaring methods overridden by the current class are
 922   // linked (and have performed their own pre-loading) before the linking
 923   // of the current class.
 924 
 925 
 926   // Note:
 927   // Inline class types are loaded during
 928   // the loading phase (see ClassFileParser::post_process_parsed_stream()).
 929   // Inline class types used as element types for array creation
 930   // are not pre-loaded. Their loading is triggered by either anewarray
 931   // or multianewarray bytecodes.
 932 
 933   // Could it be possible to do the following processing only if the
 934   // class uses inline types?
 935   {
 936     ResourceMark rm(THREAD);
 937     for (int i = 0; i < methods()->length(); i++) {
 938       Method* m = methods()->at(i);
 939       for (SignatureStream ss(m->signature()); !ss.is_done(); ss.next()) {
 940         if (ss.is_reference()) {
 941           if (ss.is_array()) {
 942             ss.skip_array_prefix();
 943           }
 944           if (ss.type() == T_VALUETYPE) {
 945             Symbol* symb = ss.as_symbol();
 946 
 947             oop loader = class_loader();
 948             oop protection_domain = this->protection_domain();
 949             Klass* klass = SystemDictionary::resolve_or_fail(symb,
 950                                                              Handle(THREAD, loader), Handle(THREAD, protection_domain), true,
 951                                                              CHECK_false);
 952             if (klass == NULL) {
 953               THROW_(vmSymbols::java_lang_LinkageError(), false);
 954             }
 955             if (!klass->is_value()) {
 956               Exceptions::fthrow(
 957                 THREAD_AND_LOCATION,
 958                 vmSymbols::java_lang_IncompatibleClassChangeError(),
 959                 "class %s is not an inline type",
 960                 klass->external_name());
 961             }
 962           }
 963         }
 964       }
 965     }
 966   }
 967 
 968   // in case the class is linked in the process of linking its superclasses
 969   if (is_linked()) {
 970     return true;
 971   }
 972 
 973   // trace only the link time for this klass that includes
 974   // the verification time
 975   PerfClassTraceTime vmtimer(ClassLoader::perf_class_link_time(),
 976                              ClassLoader::perf_class_link_selftime(),
 977                              ClassLoader::perf_classes_linked(),
 978                              jt->get_thread_stat()->perf_recursion_counts_addr(),
 979                              jt->get_thread_stat()->perf_timers_addr(),
 980                              PerfClassTraceTime::CLASS_LINK);
 981 
 982   // verification & rewriting
 983   {
 984     HandleMark hm(THREAD);
 985     Handle h_init_lock(THREAD, init_lock());
 986     ObjectLocker ol(h_init_lock, THREAD, h_init_lock() != NULL);
 987     // rewritten will have been set if loader constraint error found
 988     // on an earlier link attempt
 989     // don't verify or rewrite if already rewritten
 990     //
 991 
 992     if (!is_linked()) {
 993       if (!is_rewritten()) {
 994         {
 995           bool verify_ok = verify_code(THREAD);
 996           if (!verify_ok) {
 997             return false;
 998           }
 999         }
1000 
1001         // Just in case a side-effect of verify linked this class already
1002         // (which can sometimes happen since the verifier loads classes
1003         // using custom class loaders, which are free to initialize things)
1004         if (is_linked()) {
1005           return true;
1006         }
1007 
1008         // also sets rewritten
1009         rewrite_class(CHECK_false);
1010       } else if (is_shared()) {
1011         SystemDictionaryShared::check_verification_constraints(this, CHECK_false);
1012       }
1013 
1014       // relocate jsrs and link methods after they are all rewritten
1015       link_methods(CHECK_false);
1016 
1017       // Initialize the vtable and interface table after
1018       // methods have been rewritten since rewrite may
1019       // fabricate new Method*s.
1020       // also does loader constraint checking
1021       //
1022       // initialize_vtable and initialize_itable need to be rerun
1023       // for a shared class if
1024       // 1) the class is loaded by custom class loader or
1025       // 2) the class is loaded by built-in class loader but failed to add archived loader constraints
1026       bool need_init_table = true;
1027       if (is_shared() && SystemDictionaryShared::check_linking_constraints(this, THREAD)) {
1028         need_init_table = false;
1029       }
1030       if (need_init_table) {
1031         vtable().initialize_vtable(true, CHECK_false);
1032         itable().initialize_itable(true, CHECK_false);
1033       }
1034 #ifdef ASSERT
1035       vtable().verify(tty, true);
1036       // In case itable verification is ever added.
1037       // itable().verify(tty, true);
1038 #endif
1039 
1040       set_init_state(linked);
1041       if (JvmtiExport::should_post_class_prepare()) {
1042         Thread *thread = THREAD;
1043         assert(thread->is_Java_thread(), "thread->is_Java_thread()");
1044         JvmtiExport::post_class_prepare((JavaThread *) thread, this);
1045       }
1046     }
1047   }
1048   return true;
1049 }

1077 // Eagerly initialize superinterfaces that declare default methods (concrete instance: any access)
1078 void InstanceKlass::initialize_super_interfaces(TRAPS) {
1079   assert (has_nonstatic_concrete_methods(), "caller should have checked this");
1080   for (int i = 0; i < local_interfaces()->length(); ++i) {
1081     InstanceKlass* ik = local_interfaces()->at(i);
1082 
1083     // Initialization is depth first search ie. we start with top of the inheritance tree
1084     // has_nonstatic_concrete_methods drives searching superinterfaces since it

1092       ik->initialize(CHECK);
1093     }
1094   }

1097 void InstanceKlass::initialize_impl(TRAPS) {
1098   HandleMark hm(THREAD);
1099 
1100   // Make sure klass is linked (verified) before initialization
1101   // A class could already be verified, since it has been reflected upon.
1102   link_class(CHECK);
1103 
1104   DTRACE_CLASSINIT_PROBE(required, -1);
1105 
1106   bool wait = false;
1107 
1108   assert(THREAD->is_Java_thread(), "non-JavaThread in initialize_impl");
1109   JavaThread* jt = (JavaThread*)THREAD;
1110 
1111   // refer to the JVM book page 47 for description of steps
1112   // Step 1
1113   {
1114     Handle h_init_lock(THREAD, init_lock());
1115     ObjectLocker ol(h_init_lock, THREAD, h_init_lock() != NULL);
1116 
1117     // Step 2
1118     // If we were to use wait() instead of waitInterruptibly() then
1119     // we might end up throwing IE from link/symbol resolution sites
1120     // that aren't expected to throw.  This would wreak havoc.  See 6320309.
1121     while (is_being_initialized() && !is_reentrant_initialization(jt)) {
1122       wait = true;
1123       jt->set_class_to_be_initialized(this);
1124       ol.wait_uninterruptibly(jt);
1125       jt->set_class_to_be_initialized(NULL);
1126     }
1127 
1128     // Step 3
1129     if (is_being_initialized() && is_reentrant_initialization(jt)) {
1130       DTRACE_CLASSINIT_PROBE_WAIT(recursive, -1, wait);
1131       return;
1132     }
1133 
1134     // Step 4
1135     if (is_initialized()) {
1136       DTRACE_CLASSINIT_PROBE_WAIT(concurrent, -1, wait);
1137       return;
1138     }
1139 
1140     // Step 5
1141     if (is_in_error_state()) {
1142       DTRACE_CLASSINIT_PROBE_WAIT(erroneous, -1, wait);
1143       ResourceMark rm(THREAD);
1144       const char* desc = "Could not initialize class ";
1145       const char* className = external_name();
1146       size_t msglen = strlen(desc) + strlen(className) + 1;
1147       char* message = NEW_RESOURCE_ARRAY(char, msglen);
1148       if (NULL == message) {
1149         // Out of memory: can't create detailed error message
1150           THROW_MSG(vmSymbols::java_lang_NoClassDefFoundError(), className);
1151       } else {
1152         jio_snprintf(message, msglen, "%s%s", desc, className);
1153           THROW_MSG(vmSymbols::java_lang_NoClassDefFoundError(), message);
1154       }
1155     }
1156 
1157     // Step 6
1158     set_init_state(being_initialized);
1159     set_init_thread(jt);
1160   }
1161 
1162   // Step 7
1163   // Next, if C is a class rather than an interface, initialize it's super class and super
1164   // interfaces.
1165   if (!is_interface()) {
1166     Klass* super_klass = super();
1167     if (super_klass != NULL && super_klass->should_be_initialized()) {
1168       super_klass->initialize(THREAD);
1169     }
1170     // If C implements any interface that declares a non-static, concrete method,
1171     // the initialization of C triggers initialization of its super interfaces.
1172     // Only need to recurse if has_nonstatic_concrete_methods which includes declaring and
1173     // having a superinterface that declares, non-static, concrete methods
1174     if (!HAS_PENDING_EXCEPTION && has_nonstatic_concrete_methods()) {
1175       initialize_super_interfaces(THREAD);
1176     }
1177 
1178     // If any exceptions, complete abruptly, throwing the same exception as above.
1179     if (HAS_PENDING_EXCEPTION) {
1180       Handle e(THREAD, PENDING_EXCEPTION);
1181       CLEAR_PENDING_EXCEPTION;
1182       {
1183         EXCEPTION_MARK;
1184         // Locks object, set state, and notify all waiting threads
1185         set_initialization_state_and_notify(initialization_error, THREAD);
1186         CLEAR_PENDING_EXCEPTION;
1187       }
1188       DTRACE_CLASSINIT_PROBE_WAIT(super__failed, -1, wait);
1189       THROW_OOP(e());
1190     }
1191   }
1192 
1193   // Step 8
1194   // Initialize classes of inline fields
1195   {
1196     for (AllFieldStream fs(this); !fs.done(); fs.next()) {
1197       if (Signature::basic_type(fs.signature()) == T_VALUETYPE) {
1198         Klass* klass = this->get_value_field_klass_or_null(fs.index());
1199         if (klass == NULL) {
1200           assert(fs.access_flags().is_static(), "Otherwise should have been pre-loaded");

1201           klass = SystemDictionary::resolve_or_fail(field_signature(fs.index())->fundamental_name(THREAD),
1202               Handle(THREAD, class_loader()),
1203               Handle(THREAD, protection_domain()),
1204               true, CHECK);
1205           if (klass == NULL) {
1206             THROW(vmSymbols::java_lang_NoClassDefFoundError());
1207           }
1208           if (!klass->is_value()) {
1209             THROW(vmSymbols::java_lang_IncompatibleClassChangeError());
1210           }
1211           this->set_value_field_klass(fs.index(), klass);
1212         }
1213         InstanceKlass::cast(klass)->initialize(CHECK);
1214         if (fs.access_flags().is_static()) {
1215           if (java_mirror()->obj_field(fs.offset()) == NULL) {
1216             java_mirror()->obj_field_put(fs.offset(), ValueKlass::cast(klass)->default_value());
1217           }
1218         }
1219       }
1220     }
1221   }
1222 
1223 
1224   // Look for aot compiled methods for this klass, including class initializer.
1225   AOTLoader::load_for_klass(this, THREAD);
1226 
1227   // Step 9
1228   {
1229     DTRACE_CLASSINIT_PROBE_WAIT(clinit, -1, wait);
1230     // Timer includes any side effects of class initialization (resolution,
1231     // etc), but not recursive entry into call_class_initializer().
1232     PerfClassTraceTime timer(ClassLoader::perf_class_init_time(),
1233                              ClassLoader::perf_class_init_selftime(),
1234                              ClassLoader::perf_classes_inited(),
1235                              jt->get_thread_stat()->perf_recursion_counts_addr(),
1236                              jt->get_thread_stat()->perf_timers_addr(),
1237                              PerfClassTraceTime::CLASS_CLINIT);
1238     call_class_initializer(THREAD);
1239   }
1240 
1241   // Step 10
1242   if (!HAS_PENDING_EXCEPTION) {
1243     set_initialization_state_and_notify(fully_initialized, CHECK);
1244     {
1245       debug_only(vtable().verify(tty, true);)
1246     }
1247   }
1248   else {
1249     // Step 11 and 12
1250     Handle e(THREAD, PENDING_EXCEPTION);
1251     CLEAR_PENDING_EXCEPTION;
1252     // JVMTI has already reported the pending exception
1253     // JVMTI internal flag reset is needed in order to report ExceptionInInitializerError
1254     JvmtiExport::clear_detected_exception(jt);
1255     {
1256       EXCEPTION_MARK;
1257       set_initialization_state_and_notify(initialization_error, THREAD);
1258       CLEAR_PENDING_EXCEPTION;   // ignore any exception thrown, class initialization error is thrown below
1259       // JVMTI has already reported the pending exception
1260       // JVMTI internal flag reset is needed in order to report ExceptionInInitializerError
1261       JvmtiExport::clear_detected_exception(jt);
1262     }
1263     DTRACE_CLASSINIT_PROBE_WAIT(error, -1, wait);
1264     if (e->is_a(SystemDictionary::Error_klass())) {
1265       THROW_OOP(e());
1266     } else {
1267       JavaCallArguments args(e);
1268       THROW_ARG(vmSymbols::java_lang_ExceptionInInitializerError(),
1269                 vmSymbols::throwable_void_signature(),
1270                 &args);
1271     }
1272   }
1273   DTRACE_CLASSINIT_PROBE_WAIT(end, -1, wait);
1274 }

1329 // The embedded _implementor field can only record one implementor.
1330 // When there are more than one implementors, the _implementor field
1331 // is set to the interface Klass* itself. Following are the possible
1332 // values for the _implementor field:
1333 //   NULL                  - no implementor
1334 //   implementor Klass*    - one implementor
1335 //   self                  - more than one implementor
1336 //
1337 // The _implementor field only exists for interfaces.
1338 void InstanceKlass::add_implementor(Klass* k) {
1339   if (Universe::is_fully_initialized()) {
1340     assert_lock_strong(Compile_lock);
1341   }
1342   assert(is_interface(), "not interface");
1343   // Filter out my subinterfaces.
1344   // (Note: Interfaces are never on the subklass list.)
1345   if (InstanceKlass::cast(k)->is_interface()) return;
1346 
1347   // Filter out subclasses whose supers already implement me.
1348   // (Note: CHA must walk subclasses of direct implementors
1349   // in order to locate indirect implementors.)
1350   Klass* sk = k->super();
1351   if (sk != NULL && InstanceKlass::cast(sk)->implements_interface(this))
1352     // We only need to check one immediate superclass, since the
1353     // implements_interface query looks at transitive_interfaces.
1354     // Any supers of the super have the same (or fewer) transitive_interfaces.
1355     return;
1356 
1357   Klass* ik = implementor();
1358   if (ik == NULL) {
1359     set_implementor(k);
1360   } else if (ik != this && ik != k) {
1361     // There is already an implementor. Use itself as an indicator of
1362     // more than one implementors.
1363     set_implementor(this);
1364   }
1365 
1366   // The implementor also implements the transitive_interfaces
1367   for (int index = 0; index < local_interfaces()->length(); index++) {
1368     InstanceKlass::cast(local_interfaces()->at(index))->add_implementor(k);
1369   }
1370 }

1397                                                                Array<InstanceKlass*>* transitive_interfaces) {

1400   int num_secondaries = num_extra_slots + interfaces->length();

1417     for (int i = 0; i < interfaces->length(); i++) {

1419     }
1420     return secondaries;
1421   }

1619 Klass* InstanceKlass::find_field(Symbol* name, Symbol* sig, fieldDescriptor* fd) const {
1620   // search order according to newest JVM spec (5.4.3.2, p.167).
1621   // 1) search for field in current klass
1622   if (find_local_field(name, sig, fd)) {
1623     return const_cast<InstanceKlass*>(this);
1624   }
1625   // 2) search for field recursively in direct superinterfaces
1626   { Klass* intf = find_interface_field(name, sig, fd);
1627     if (intf != NULL) return intf;
1628   }
1629   // 3) apply field lookup recursively if superclass exists
1630   { Klass* supr = super();
1631     if (supr != NULL) return InstanceKlass::cast(supr)->find_field(name, sig, fd);
1632   }
1633   // 4) otherwise field lookup fails
1634   return NULL;
1635 }

1804 inline int InstanceKlass::quick_search(const Array<Method*>* methods, const Symbol* name) {
1805   if (_disable_method_binary_search) {
1806     assert(DynamicDumpSharedSpaces, "must be");
1807     // At the final stage of dynamic dumping, the methods array may not be sorted
1808     // by ascending addresses of their names, so we can't use binary search anymore.
1809     // However, methods with the same name are still laid out consecutively inside the
1810     // methods array, so let's look for the first one that matches.
1811     return linear_search(methods, name);
1812   }
1813 
1814   int len = methods->length();
1815   int l = 0;
1816   int h = len - 1;
1817 
1818   // methods are sorted by ascending addresses of their names, so do binary search
1819   while (l <= h) {
1820     int mid = (l + h) >> 1;
1821     Method* m = methods->at(mid);
1822     assert(m->is_method(), "must be method");
1823     int res = m->name()->fast_compare(name);
1824     if (res == 0) {
1825       return mid;
1826     } else if (res < 0) {
1827       l = mid + 1;
1828     } else {
1829       h = mid - 1;
1830     }
1831   }
1832   return -1;
1833 }

1937 // true if method matches signature and conforms to skipping_X conditions.
1938 static bool method_matches(const Method* m,
1939                            const Symbol* signature,
1940                            bool skipping_overpass,
1941                            bool skipping_static,
1942                            bool skipping_private) {
1943   return ((m->signature() == signature) &&

1949 // Used directly for default_methods to find the index into the
1950 // default_vtable_indices, and indirectly by find_method
1951 // find_method_index looks in the local methods array to return the index
1952 // of the matching name/signature. If, overpass methods are being ignored,
1953 // the search continues to find a potential non-overpass match.  This capability
1954 // is important during method resolution to prefer a static method, for example,
1955 // over an overpass method.
1956 // There is the possibility in any _method's array to have the same name/signature
1957 // for a static method, an overpass method and a local instance method
1958 // To correctly catch a given method, the search criteria may need
1959 // to explicitly skip the other two. For local instance methods, it
1960 // is often necessary to skip private methods
1961 int InstanceKlass::find_method_index(const Array<Method*>* methods,
1962                                      const Symbol* name,
1963                                      const Symbol* signature,
1964                                      OverpassLookupMode overpass_mode,
1965                                      StaticLookupMode static_mode,
1966                                      PrivateLookupMode private_mode) {
1967   const bool skipping_overpass = (overpass_mode == skip_overpass);
1968   const bool skipping_static = (static_mode == skip_static);
1969   const bool skipping_private = (private_mode == skip_private);
1970   const int hit = quick_search(methods, name);
1971   if (hit != -1) {
1972     const Method* const m = methods->at(hit);
1973 
1974     // Do linear search to find matching signature.  First, quick check
1975     // for common case, ignoring overpasses if requested.
1976     if (method_matches(m, signature, skipping_overpass, skipping_static, skipping_private)) {
1977       return hit;
1978     }
1979 
1980     // search downwards through overloaded methods
1981     int i;
1982     for (i = hit - 1; i >= 0; --i) {
1983         const Method* const m = methods->at(i);
1984         assert(m->is_method(), "must be method");
1985         if (m->name() != name) {
1986           break;
1987         }
1988         if (method_matches(m, signature, skipping_overpass, skipping_static, skipping_private)) {
1989           return i;
1990         }
1991     }
1992     // search upwards
1993     for (i = hit + 1; i < methods->length(); ++i) {
1994         const Method* const m = methods->at(i);
1995         assert(m->is_method(), "must be method");
1996         if (m->name() != name) {
1997           break;
1998         }
1999         if (method_matches(m, signature, skipping_overpass, skipping_static, skipping_private)) {
2000           return i;
2001         }
2002     }
2003     // not found
2004 #ifdef ASSERT
2005     const int index = (skipping_overpass || skipping_static || skipping_private) ? -1 :
2006       linear_search(methods, name, signature);
2007     assert(-1 == index, "binary search should have found entry %d", index);
2008 #endif
2009   }
2010   return -1;
2011 }

2160 // Lookup or create a jmethodID.
2161 // This code is called by the VMThread and JavaThreads so the
2162 // locking has to be done very carefully to avoid deadlocks
2163 // and/or other cache consistency problems.
2164 //
2165 jmethodID InstanceKlass::get_jmethod_id(const methodHandle& method_h) {
2166   size_t idnum = (size_t)method_h->method_idnum();
2167   jmethodID* jmeths = methods_jmethod_ids_acquire();
2168   size_t length = 0;
2169   jmethodID id = NULL;
2170 
2171   // We use a double-check locking idiom here because this cache is
2172   // performance sensitive. In the normal system, this cache only
2173   // transitions from NULL to non-NULL which is safe because we use
2174   // release_set_methods_jmethod_ids() to advertise the new cache.
2175   // A partially constructed cache should never be seen by a racing
2176   // thread. We also use release_store() to save a new jmethodID
2177   // in the cache so a partially constructed jmethodID should never be
2178   // seen either. Cache reads of existing jmethodIDs proceed without a
2179   // lock, but cache writes of a new jmethodID requires uniqueness and
2180   // creation of the cache itself requires no leaks so a lock is
2181   // generally acquired in those two cases.
2182   //
2183   // If the RedefineClasses() API has been used, then this cache can
2184   // grow and we'll have transitions from non-NULL to bigger non-NULL.
2185   // Cache creation requires no leaks and we require safety between all
2186   // cache accesses and freeing of the old cache so a lock is generally
2187   // acquired when the RedefineClasses() API has been used.
2188 
2189   if (jmeths != NULL) {
2190     // the cache already exists
2191     if (!idnum_can_increment()) {
2192       // the cache can't grow so we can just get the current values
2193       get_jmethod_id_length_value(jmeths, idnum, &length, &id);
2194     } else {
2195       // cache can grow so we have to be more careful
2196       if (Threads::number_of_threads() == 0 ||
2197           SafepointSynchronize::is_at_safepoint()) {
2198         // we're single threaded or at a safepoint - no locking needed
2199         get_jmethod_id_length_value(jmeths, idnum, &length, &id);
2200       } else {
2201         MutexLocker ml(JmethodIdCreation_lock, Mutex::_no_safepoint_check_flag);
2202         get_jmethod_id_length_value(jmeths, idnum, &length, &id);
2203       }
2204     }
2205   }
2206   // implied else:
2207   // we need to allocate a cache so default length and id values are good
2208 
2209   if (jmeths == NULL ||   // no cache yet
2210       length <= idnum ||  // cache is too short
2211       id == NULL) {       // cache doesn't contain entry
2212 
2213     // This function can be called by the VMThread so we have to do all
2214     // things that might block on a safepoint before grabbing the lock.
2215     // Otherwise, we can deadlock with the VMThread or have a cache
2216     // consistency issue. These vars keep track of what we might have
2217     // to free after the lock is dropped.
2218     jmethodID  to_dealloc_id     = NULL;
2219     jmethodID* to_dealloc_jmeths = NULL;
2220 
2221     // may not allocate new_jmeths or use it if we allocate it
2222     jmethodID* new_jmeths = NULL;
2223     if (length <= idnum) {
2224       // allocate a new cache that might be used
2225       size_t size = MAX2(idnum+1, (size_t)idnum_allocated_count());
2226       new_jmeths = NEW_C_HEAP_ARRAY(jmethodID, size+1, mtClass);
2227       memset(new_jmeths, 0, (size+1)*sizeof(jmethodID));
2228       // cache size is stored in element[0], other elements offset by one
2229       new_jmeths[0] = (jmethodID)size;
2230     }
2231 
2232     // allocate a new jmethodID that might be used
2233     jmethodID new_id = NULL;
2234     if (method_h->is_old() && !method_h->is_obsolete()) {
2235       // The method passed in is old (but not obsolete), we need to use the current version
2236       Method* current_method = method_with_idnum((int)idnum);
2237       assert(current_method != NULL, "old and but not obsolete, so should exist");
2238       new_id = Method::make_jmethod_id(class_loader_data(), current_method);
2239     } else {
2240       // It is the current version of the method or an obsolete method,
2241       // use the version passed in
2242       new_id = Method::make_jmethod_id(class_loader_data(), method_h());
2243     }
2244 
2245     if (Threads::number_of_threads() == 0 ||
2246         SafepointSynchronize::is_at_safepoint()) {
2247       // we're single threaded or at a safepoint - no locking needed
2248       id = get_jmethod_id_fetch_or_update(idnum, new_id, new_jmeths,
2249                                           &to_dealloc_id, &to_dealloc_jmeths);
2250     } else {
2251       MutexLocker ml(JmethodIdCreation_lock, Mutex::_no_safepoint_check_flag);
2252       id = get_jmethod_id_fetch_or_update(idnum, new_id, new_jmeths,
2253                                           &to_dealloc_id, &to_dealloc_jmeths);
2254     }
2255 
2256     // The lock has been dropped so we can free resources.
2257     // Free up either the old cache or the new cache if we allocated one.
2258     if (to_dealloc_jmeths != NULL) {
2259       FreeHeap(to_dealloc_jmeths);
2260     }
2261     // free up the new ID since it wasn't needed
2262     if (to_dealloc_id != NULL) {
2263       Method::destroy_jmethod_id(class_loader_data(), to_dealloc_id);
2264     }
2265   }
2266   return id;
2267 }

2304 
2305   // reacquire the cache - we are locked, single threaded or at a safepoint
2306   jmethodID* jmeths = methods_jmethod_ids_acquire();
2307   jmethodID  id     = NULL;
2308   size_t     length = 0;
2309 
2310   if (jmeths == NULL ||                         // no cache yet
2311       (length = (size_t)jmeths[0]) <= idnum) {  // cache is too short
2312     if (jmeths != NULL) {
2313       // copy any existing entries from the old cache
2314       for (size_t index = 0; index < length; index++) {
2315         new_jmeths[index+1] = jmeths[index+1];
2316       }
2317       *to_dealloc_jmeths_p = jmeths;  // save old cache for later delete
2318     }
2319     release_set_methods_jmethod_ids(jmeths = new_jmeths);
2320   } else {
2321     // fetch jmethodID (if any) from the existing cache
2322     id = jmeths[idnum+1];
2323     *to_dealloc_jmeths_p = new_jmeths;  // save new cache for later delete
2324   }
2325   if (id == NULL) {
2326     // No matching jmethodID in the existing cache or we have a new
2327     // cache or we just grew the cache. This cache write is done here
2328     // by the first thread to win the foot race because a jmethodID
2329     // needs to be unique once it is generally available.
2330     id = new_id;
2331 
2332     // The jmethodID cache can be read while unlocked so we have to
2333     // make sure the new jmethodID is complete before installing it
2334     // in the cache.
2335     Atomic::release_store(&jmeths[idnum+1], id);
2336   } else {
2337     *to_dealloc_id_p = new_id; // save new id for later delete
2338   }
2339   return id;
2340 }

3778   // Avoid redundant verifies, this really should be in product.
3779   if (_verify_count == Universe::verify_count()) return;
3780   _verify_count = Universe::verify_count();
3781 #endif
3782 
3783   // Verify Klass

3785

3789

3791   if (is_linked()) {
3792     // $$$ This used to be done only for m/s collections.  Doing it
3793     // always seemed a valid generalization.  (DLD -- 6/00)
3794     vtable().verify(st);
3795   }
3796 
3797   // Verify first subklass
3798   if (subklass() != NULL) {
3799     guarantee(subklass()->is_klass(), "should be klass");
3800   }
3801 
3802   // Verify siblings
3803   Klass* super = this->super();
3804   Klass* sib = next_sibling();
3805   if (sib != NULL) {
3806     if (sib == this) {
3807       fatal("subclass points to itself " PTR_FORMAT, p2i(sib));
3808     }
3809

3811     guarantee(sib->super() == super, "siblings should have same superklass");
3812   }

3815   if (local_interfaces()) {
3816     Array<InstanceKlass*>* local_interfaces = this->local_interfaces();
3817     for (int j = 0; j < local_interfaces->length(); j++) {
3818       InstanceKlass* e = local_interfaces->at(j);
3819       guarantee(e->is_klass() && e->is_interface(), "invalid local interface");
3820     }
3821   }
3822

3824   if (transitive_interfaces() != NULL) {
3825     Array<InstanceKlass*>* transitive_interfaces = this->transitive_interfaces();
3826     for (int j = 0; j < transitive_interfaces->length(); j++) {
3827       InstanceKlass* e = transitive_interfaces->at(j);

3829     }
3830   }
3831

3836       guarantee(methods->at(j)->is_method(), "non-method in methods array");
3837     }
3838     for (int j = 0; j < methods->length() - 1; j++) {
3839       Method* m1 = methods->at(j);
3840       Method* m2 = methods->at(j + 1);
3841       guarantee(m1->name()->fast_compare(m2->name()) <= 0, "methods not sorted correctly");
3842     }

3856         guarantee(original_index < length, "invalid method ordering index");
3857         sum += original_index;
3858       }
3859       // Verify sum of indices 0,1,...,length-1
3860       guarantee(sum == ((jlong)length*(length-1))/2, "invalid method ordering sum");

3863     }
3864   }
3865 
3866   // Verify default methods
3867   if (default_methods() != NULL) {
3868     Array<Method*>* methods = this->default_methods();
3869     for (int j = 0; j < methods->length(); j++) {
3870       guarantee(methods->at(j)->is_method(), "non-method in methods array");
3871     }
3872     for (int j = 0; j < methods->length() - 1; j++) {
3873       Method* m1 = methods->at(j);
3874       Method* m2 = methods->at(j + 1);
3875       guarantee(m1->name()->fast_compare(m2->name()) <= 0, "methods not sorted correctly");
3876     }
3877   }
3878

3880   if (jni_ids() != NULL) {
3881     jni_ids()->verify(this);
3882   }
3883

3885   if (constants() != NULL) {
3886     guarantee(constants()->is_constantPool(), "should be constant pool");
3887   }
3888   const Klass* anonymous_host = unsafe_anonymous_host();
3889   if (anonymous_host != NULL) {
3890     guarantee(anonymous_host->is_klass(), "should be klass");
3891   }

3981 // Purge previous versions before adding new previous versions of the class and
3982 // during class unloading.
3983 void InstanceKlass::purge_previous_version_list() {
3984   assert(SafepointSynchronize::is_at_safepoint(), "only called at safepoint");
3985   assert(has_been_redefined(), "Should only be called for main class");
3986 
3987   // Quick exit.
3988   if (previous_versions() == NULL) {
3989     return;
3990   }
3991 
3992   // This klass has previous versions so see what we can cleanup
3993   // while it is safe to do so.
3994 
3995   int deleted_count = 0;    // leave debugging breadcrumbs
3996   int live_count = 0;
3997   ClassLoaderData* loader_data = class_loader_data();
3998   assert(loader_data != NULL, "should never be null");
3999 
4000   ResourceMark rm;
4001   log_trace(redefine, class, iklass, purge)("%s: previous versions", external_name());
4002 
4003   // previous versions are linked together through the InstanceKlass
4004   InstanceKlass* pv_node = previous_versions();
4005   InstanceKlass* last = this;
4006   int version = 0;
4007 
4008   // check the previous versions list
4009   for (; pv_node != NULL; ) {
4010 
4011     ConstantPool* pvcp = pv_node->constants();
4012     assert(pvcp != NULL, "cp ref was unexpectedly cleared");
4013 
4014     if (!pvcp->on_stack()) {
4015       // If the constant pool isn't on stack, none of the methods
4016       // are executing.  Unlink this previous_version.
4017       // The previous version InstanceKlass is on the ClassLoaderData deallocate list
4018       // so will be deallocated during the next phase of class unloading.
4019       log_trace(redefine, class, iklass, purge)
4020         ("previous version " INTPTR_FORMAT " is dead.", p2i(pv_node));
4021       // For debugging purposes.
4022       pv_node->set_is_scratch_class();
4023       // Unlink from previous version list.
4024       assert(pv_node->class_loader_data() == loader_data, "wrong loader_data");
4025       InstanceKlass* next = pv_node->previous_versions();
4026       pv_node->link_previous_versions(NULL);   // point next to NULL
4027       last->link_previous_versions(next);
4028       // Add to the deallocate list after unlinking
4029       loader_data->add_to_deallocate_list(pv_node);
4030       pv_node = next;
4031       deleted_count++;
4032       version++;
4033       continue;
4034     } else {
4035       log_trace(redefine, class, iklass, purge)("previous version " INTPTR_FORMAT " is alive", p2i(pv_node));
4036       assert(pvcp->pool_holder() != NULL, "Constant pool with no holder");
4037       guarantee (!loader_data->is_unloading(), "unloaded classes can't be on the stack");
4038       live_count++;
4039       // found a previous version for next time we do class unloading
4040       _has_previous_versions = true;
4041     }
4042 
4043     // At least one method is live in this previous version.
4044     // Reset dead EMCP methods not to get breakpoints.
4045     // All methods are deallocated when all of the methods for this class are no
4046     // longer running.
4047     Array<Method*>* method_refs = pv_node->methods();
4048     if (method_refs != NULL) {
4049       log_trace(redefine, class, iklass, purge)("previous methods length=%d", method_refs->length());
4050       for (int j = 0; j < method_refs->length(); j++) {
4051         Method* method = method_refs->at(j);
4052 
4053         if (!method->on_stack()) {
4054           // no breakpoints for non-running methods
4055           if (method->is_running_emcp()) {
4056             method->set_running_emcp(false);
4057           }
4058         } else {
4059           assert (method->is_obsolete() || method->is_running_emcp(),
4060                   "emcp method cannot run after emcp bit is cleared");
4061           log_trace(redefine, class, iklass, purge)
4062             ("purge: %s(%s): prev method @%d in version @%d is alive",
4063              method->name()->as_C_string(), method->signature()->as_C_string(), j, version);
4064         }
4065       }
4066     }
4067     // next previous version
4068     last = pv_node;
4069     pv_node = pv_node->previous_versions();
4070     version++;
4071   }
4072   log_trace(redefine, class, iklass, purge)
4073     ("previous version stats: live=%d, deleted=%d", live_count, deleted_count);
4074 }

4076 void InstanceKlass::mark_newly_obsolete_methods(Array<Method*>* old_methods,
4077                                                 int emcp_method_count) {
4078   int obsolete_method_count = old_methods->length() - emcp_method_count;
4079 
4080   if (emcp_method_count != 0 && obsolete_method_count != 0 &&
4081       _previous_versions != NULL) {
4082     // We have a mix of obsolete and EMCP methods so we have to
4083     // clear out any matching EMCP method entries the hard way.
4084     int local_count = 0;
4085     for (int i = 0; i < old_methods->length(); i++) {
4086       Method* old_method = old_methods->at(i);
4087       if (old_method->is_obsolete()) {
4088         // only obsolete methods are interesting
4089         Symbol* m_name = old_method->name();
4090         Symbol* m_signature = old_method->signature();
4091 
4092         // previous versions are linked together through the InstanceKlass
4093         int j = 0;
4094         for (InstanceKlass* prev_version = _previous_versions;
4095              prev_version != NULL;
4096              prev_version = prev_version->previous_versions(), j++) {
4097 
4098           Array<Method*>* method_refs = prev_version->methods();
4099           for (int k = 0; k < method_refs->length(); k++) {
4100             Method* method = method_refs->at(k);
4101 
4102             if (!method->is_obsolete() &&
4103                 method->name() == m_name &&
4104                 method->signature() == m_signature) {
4105               // The current RedefineClasses() call has made all EMCP
4106               // versions of this method obsolete so mark it as obsolete
4107               log_trace(redefine, class, iklass, add)
4108                 ("%s(%s): flush obsolete method @%d in version @%d",
4109                  m_name->as_C_string(), m_signature->as_C_string(), k, j);
4110 
4111               method->set_is_obsolete();
4112               break;
4113             }
4114           }
4115 
4116           // The previous loop may not find a matching EMCP method, but
4117           // that doesn't mean that we can optimize and not go any
4118           // further back in the PreviousVersion generations. The EMCP
4119           // method for this generation could have already been made obsolete,
4120           // but there still may be an older EMCP method that has not
4121           // been made obsolete.
4122         }
4123 
4124         if (++local_count >= obsolete_method_count) {
4125           // no more obsolete methods so bail out now
4126           break;
4127         }
4128       }
4129     }
4130   }
4131 }

4133 // Save the scratch_class as the previous version if any of the methods are running.
4134 // The previous_versions are used to set breakpoints in EMCP methods and they are
4135 // also used to clean MethodData links to redefined methods that are no longer running.
4136 void InstanceKlass::add_previous_version(InstanceKlass* scratch_class,
4137                                          int emcp_method_count) {
4138   assert(Thread::current()->is_VM_thread(),
4139          "only VMThread can add previous versions");
4140 
4141   ResourceMark rm;
4142   log_trace(redefine, class, iklass, add)
4143     ("adding previous version ref for %s, EMCP_cnt=%d", scratch_class->external_name(), emcp_method_count);
4144 
4145   // Clean out old previous versions for this class
4146   purge_previous_version_list();
4147 
4148   // Mark newly obsolete methods in remaining previous versions.  An EMCP method from
4149   // a previous redefinition may be made obsolete by this redefinition.
4150   Array<Method*>* old_methods = scratch_class->methods();
4151   mark_newly_obsolete_methods(old_methods, emcp_method_count);
4152 
4153   // If the constant pool for this previous version of the class
4154   // is not marked as being on the stack, then none of the methods
4155   // in this previous version of the class are on the stack so
4156   // we don't need to add this as a previous version.
4157   ConstantPool* cp_ref = scratch_class->constants();
4158   if (!cp_ref->on_stack()) {
4159     log_trace(redefine, class, iklass, add)("scratch class not added; no methods are running");
4160     // For debugging purposes.
4161     scratch_class->set_is_scratch_class();
4162     scratch_class->class_loader_data()->add_to_deallocate_list(scratch_class);
4163     return;
4164   }
4165 
4166   if (emcp_method_count != 0) {
4167     // At least one method is still running, check for EMCP methods
4168     for (int i = 0; i < old_methods->length(); i++) {
4169       Method* old_method = old_methods->at(i);
4170       if (!old_method->is_obsolete() && old_method->on_stack()) {
4171         // if EMCP method (not obsolete) is on the stack, mark as EMCP so that
4172         // we can add breakpoints for it.
4173 
4174         // We set the method->on_stack bit during safepoints for class redefinition
4175         // and use this bit to set the is_running_emcp bit.
4176         // After the safepoint, the on_stack bit is cleared and the running emcp
4177         // method may exit.   If so, we would set a breakpoint in a method that
4178         // is never reached, but this won't be noticeable to the programmer.
4179         old_method->set_running_emcp(true);
4180         log_trace(redefine, class, iklass, add)
4181           ("EMCP method %s is on_stack " INTPTR_FORMAT, old_method->name_and_sig_as_C_string(), p2i(old_method));
4182       } else if (!old_method->is_obsolete()) {
4183         log_trace(redefine, class, iklass, add)
4184           ("EMCP method %s is NOT on_stack " INTPTR_FORMAT, old_method->name_and_sig_as_C_string(), p2i(old_method));
4185       }
4186     }
4187   }
4188 
4189   // Add previous version if any methods are still running.
4190   // Set has_previous_version flag for processing during class unloading.
4191   _has_previous_versions = true;
4192   log_trace(redefine, class, iklass, add) ("scratch class added; one of its methods is on_stack.");
4193   assert(scratch_class->previous_versions() == NULL, "shouldn't have a previous version");
4194   scratch_class->link_previous_versions(previous_versions());
4195   link_previous_versions(scratch_class);
4196 } // end add_previous_version()
