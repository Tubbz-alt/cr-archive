40 import java.util.EnumSet;
41 import java.util.Set;
42 import javax.lang.model.SourceVersion;
43 import javax.tools.Tool;
44 import javax.tools.ToolProvider;
45 import static javax.lang.model.SourceVersion.*;
46 
47 public class TestGetSourceVersions {
48     public static void main(String... args) {
49         Tool compiler = ToolProvider.getSystemJavaCompiler();
50         Set<SourceVersion> expected = EnumSet.noneOf(SourceVersion.class);
51         for (String arg : args)
52             expected.add(SourceVersion.valueOf(arg));
53         Set<SourceVersion> found = compiler.getSourceVersions();
54         Set<SourceVersion> notExpected = EnumSet.copyOf(found);
55         for (SourceVersion version : expected) {
56             if (!found.contains(version))
57                 throw new AssertionError("Expected source version not found: " + version);
58             else
59                 notExpected.remove(version);
60         }
61         if (!notExpected.isEmpty())
62             throw new AssertionError("Unexpected source versions: " + notExpected);
63     }
64 }
