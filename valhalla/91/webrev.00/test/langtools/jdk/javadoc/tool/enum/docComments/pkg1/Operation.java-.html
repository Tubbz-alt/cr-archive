26 /*
27  * Example from JSR 201, which borrowed it from "Effective Java".
28  */
29 
30 /**
31  * Arithmetic operations.
32  */
33 public abstract enum Operation {
34     /** Addition */
35     plus {
36         /** Add 'em up. */
37         double eval(double x, double y) { return x + y; }
38     },
39     minus {
40         double eval(double x, double y) { return x - y; }
41     },
42     times {
43         double eval(double x, double y) { return x * y; }
44     },
45     divided_by {
46         double eval(double x, double y) { return x / y; }
47     };
48 
49     /**
50      * Perform arithmetic operation represented by this constant.
51      */
52     abstract double eval(double x, double y);
53 }
