24 package javadoc.tester;
25 
26 import java.util.Locale;
27 import java.util.Set;
28 import javax.lang.model.SourceVersion;
29 
30 import jdk.javadoc.doclet.Doclet;
31 import jdk.javadoc.doclet.DocletEnvironment;
32 import jdk.javadoc.doclet.Reporter;
33 
34 /**
35  * A minimal base class for test doclets.
36  *
37  * The {@link Doclet#run(DocletEnvironment) run} method must still be provided by subtypes.
38  */
39 public abstract class TestDoclet implements Doclet {
40     protected Locale locale;
41     protected Reporter reporter;
42 
43     @Override
44     public void init(Locale locale, Reporter reporter) {
45         this.locale = locale;
46         this.reporter = reporter;
47     }
48 
49     @Override
50     public String getName() {
51         return getClass().getSimpleName();
52     }
53 
54     @Override
55     public Set<? extends Option> getSupportedOptions() {
56         return Set.of();
57     }
58 
59     @Override
60     public SourceVersion getSupportedSourceVersion() {
61         return SourceVersion.latestSupported();
62     }
63 }
