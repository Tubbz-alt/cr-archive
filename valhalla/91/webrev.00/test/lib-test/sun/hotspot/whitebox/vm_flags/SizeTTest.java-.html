24 /*
25  * @test SizeTTest
26  * @bug 8054823
27  * @library /test/lib
28  * @modules java.base/jdk.internal.misc
29  *          java.management/sun.management
30  * @build sun.hotspot.WhiteBox
31  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
32  * @run main/othervm/timeout=600 -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -XX:+UnlockExperimentalVMOptions SizeTTest
33  * @summary testing of WB::set/getSizeTVMFlag()
34  */
35 import jdk.test.lib.Platform;
36 
37 public class SizeTTest {
38     private static final String FLAG_NAME = "ArrayAllocatorMallocLimit";
39     private static final Long[] TESTS = {0L, 100L, (long) Integer.MAX_VALUE,
40         (1L << 32L) - 1L, 1L << 32L};
41     private static final Long[] EXPECTED_64 = TESTS;
42     private static final Long[] EXPECTED_32 = {0L, 100L,
43         (long) Integer.MAX_VALUE, (1L << 32L) - 1L, 0L};
44 
45     public static void main(String[] args) throws Exception {
46         VmFlagTest.runTest(FLAG_NAME, TESTS,
47             Platform.is64bit() ? EXPECTED_64 : EXPECTED_32,
48             VmFlagTest.WHITE_BOX::setSizeTVMFlag,
49             VmFlagTest.WHITE_BOX::getSizeTVMFlag);
50     }
51 }
