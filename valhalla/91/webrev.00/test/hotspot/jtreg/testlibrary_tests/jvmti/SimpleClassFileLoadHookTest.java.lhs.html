24 /*
25  * @test
26  * @library /test/lib
27  * @requires vm.flavor != "minimal"
28  * @run main/othervm/native -agentlib:SimpleClassFileLoadHook=Foo,XXX,YYY
29  *      SimpleClassFileLoadHookTest
30  */
31 import jdk.test.lib.Asserts;
32 
33 class Foo {
34     static String getValue() {
35         return "XXX";
36     }
37     static String getOtherValue() {
38         return "xXXXxx";
39     }
40 }
41 public class SimpleClassFileLoadHookTest {
42     public static void main(String args[]) {
43         System.out.println(Foo.getValue());
44         System.out.println(Foo.getOtherValue());
45         Asserts.assertTrue("YYY".equals(Foo.getValue()) &&
46                            "xYYYxx".equals(Foo.getOtherValue()),
47                            "SimpleClassFileLoadHook should replace XXX with YYY");
48     }
49 }
