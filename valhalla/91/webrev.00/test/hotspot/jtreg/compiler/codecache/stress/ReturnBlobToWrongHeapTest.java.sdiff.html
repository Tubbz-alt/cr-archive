49 public class ReturnBlobToWrongHeapTest {
50     private static final long largeBlobSize = Helper.WHITE_BOX.getUintxVMFlag("ReservedCodeCacheSize") >> 6;
51     private static final long codeCacheMinBlockLength = Helper.WHITE_BOX.getUintxVMFlag("CodeCacheMinBlockLength");
52     private static final BlobType[] BLOB_TYPES = BlobType.getAvailable().toArray(new BlobType[0]);
53 
54     // Allocate blob in first code heap (the code heap with index 0).
55     private static long allocate(int size) {
56         return Helper.WHITE_BOX.allocateCodeBlob(size, BLOB_TYPES[0].id);
57     }
58 
59     // Free blob.
60     private static void free(long address) {
61         Helper.WHITE_BOX.freeCodeBlob(address);
62     }
63 
64     public static void main(String[] args) {
65         if (codeCacheMinBlockLength == 1) {
66             // start with allocating a small block
67             long firstSegmentSizedAddress = 0;
68             firstSegmentSizedAddress = allocate(0);
69             if (firstSegmentSizedAddress == 0) {
70                 throw new RuntimeException("Test failed: Failed allocating first segment-sized blob");
71             }
72 
73             // Fill first code heap with large blobs until allocation fails.
74             long address;
75             while ((address = allocate((int)largeBlobSize)) != 0) {
76             }
77 
78             // Allocate segment-sized blocks in first code heap until it runs out
79             // Remember the last one
80             // Use the pre-allocated one as backup if the code cache is already completely full.
81             long lastSegmentSizedAddress = firstSegmentSizedAddress;
82             while ((address = allocate(0)) != 0) {
83                 lastSegmentSizedAddress = address;
84             }
85 
86             if (lastSegmentSizedAddress == 0) {
87                 throw new RuntimeException("Test failed: Not possible to allocate segment-sized blob");
88             }
89 
90             // Remove last segment-sized block from the first code heap.
91             free(lastSegmentSizedAddress);
92         } else {
93             throw new RuntimeException("Test requires CodeCacheMinBlockLength==1; CodeCacheMinBlockLength is " +
94                                        codeCacheMinBlockLength);
95         }
96     }
