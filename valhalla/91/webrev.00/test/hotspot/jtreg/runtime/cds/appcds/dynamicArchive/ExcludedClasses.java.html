26  * @test
27  * @summary Unreferenced app classes during dump time should not be included in the archive.
28  * @requires vm.cds
29  * @library /test/lib /test/hotspot/jtreg/runtime/cds/appcds
30  *          /test/hotspot/jtreg/runtime/cds/appcds/dynamicArchive/test-classes
31  * @build ExcludedClassesApp
32  * @build sun.hotspot.WhiteBox
33  * @run driver ClassFileInstaller -jar ExcludedClasses.jar
34  *             ExcludedClassesApp
35  *             ExcludedClassesApp$NotLinkedSuper
36  *             ExcludedClassesApp$NotLinkedChild
37  *             ExcludedClassesApp$NotLinkedInterface
38  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
39  * @run main/othervm -XX:+UnlockDiagnosticVMOptions -XX:+WhiteBoxAPI -Xbootclasspath/a:. ExcludedClasses
40  */
41 
42 public class ExcludedClasses extends DynamicArchiveTestBase {
43     public static void main(String[] args) throws Exception {
44         runTest(ExcludedClasses::test);
45     }
46 
47     static void test() throws Exception {
48         String topArchiveName = getNewArchiveName();
49         String appJar = ClassFileInstaller.getJarPath("ExcludedClasses.jar");
50         String mainClass = "ExcludedClassesApp";
51 
52         dumpAndRun(topArchiveName, "-Xlog:cds+dynamic=debug", "-cp", appJar, mainClass);
53     }
54 }
