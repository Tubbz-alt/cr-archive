25 import java.lang.instrument.ClassFileTransformer;
26 import java.lang.instrument.Instrumentation;
27 import java.lang.instrument.IllegalClassFormatException;
28 import java.security.ProtectionDomain;
29 
30 public class GCDuringDumpTransformer implements ClassFileTransformer {
31     public byte[] transform(ClassLoader loader, String name, Class<?> classBeingRedefined,
32                             ProtectionDomain pd, byte[] buffer) throws IllegalClassFormatException {
33         try {
34             makeGarbage();
35         } catch (Throwable t) {
36             t.printStackTrace();
37             try {
38                 Thread.sleep(200); // let GC to have a chance to run
39             } catch (Throwable t2) {}
40         }
41 
42         return null;
43     }
44 
45     private static Instrumentation savedInstrumentation;
46 
47     public static void premain(String agentArguments, Instrumentation instrumentation) {
48         System.out.println("ClassFileTransformer.premain() is called");
49         instrumentation.addTransformer(new GCDuringDumpTransformer(), /*canRetransform=*/true);
50         savedInstrumentation = instrumentation;
51     }
52 
53     public static Instrumentation getInstrumentation() {
54         return savedInstrumentation;
55     }
56 
57     public static void agentmain(String args, Instrumentation inst) throws Exception {
58         premain(args, inst);
59     }
60 
61     public static void makeGarbage() {
62         for (int x=0; x<10; x++) {
63             Object[] a = new Object[10000];
64         }
65     }
66 }
