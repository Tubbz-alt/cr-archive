1 #####################################################################
  2 #	Default Configuration File for Java Platform Management
  3 #####################################################################
  4 #
  5 # The Management Configuration file (in java.util.Properties format)
  6 # will be read if one of the following system properties is set:
  7 #    -Dcom.sun.management.jmxremote.port=<port-number>
  8 # or -Dcom.sun.management.config.file=<this-file>
  9 #
 10 # The default Management Configuration file is:
 11 #
 12 #       $JRE/conf/management/management.properties
 13 #
 14 # Another location for the Management Configuration File can be specified
 15 # by the following property on the Java command line:
 16 #
 17 #    -Dcom.sun.management.config.file=<this-file>
 18 #
 19 # If -Dcom.sun.management.config.file=<this-file> is set, the port
 20 # number for the management agent can be specified in the config file
 21 # using the following lines:
 22 #
 23 # ################ Management Agent Port #########################
 24 #
 25 # For setting the JMX RMI agent port use the following line
 26 com.sun.management.jmxremote.port=${getFreePort}
 27 #

 29 #####################################################################
 30 #                   Optional Instrumentation
 31 #####################################################################
 32 #
 33 # By default only the basic instrumentation with low overhead is on.
 34 # The following properties allow to selectively turn on optional
 35 # instrumentation which are off by default and may have some
 36 # additional overhead.
 37 #
 38 # com.sun.management.enableThreadContentionMonitoring
 39 #
 40 #      This option enables thread contention monitoring if the
 41 #      Java virtual machine supports such instrumentation.
 42 #      Refer to the specification for the java.lang.management.ThreadMBean
 43 #      interface - see isThreadContentionMonitoringSupported() method.
 44 #
 45 
 46 # To enable thread contention monitoring, uncomment the following line
 47 # com.sun.management.enableThreadContentionMonitoring

 49 #####################################################################
 50 #			RMI Management Properties
 51 #####################################################################
 52 #
 53 # If system property -Dcom.sun.management.jmxremote.port=<port-number>
 54 # is set then
 55 #     - A MBean server is started
 56 #     - JRE Platform MBeans are registered in the MBean server
 57 #     - RMI connector is published  in a private readonly registry at
 58 #       specified port using a well known name, "jmxrmi"
 59 #     - the following properties are read for JMX remote management.
 60 #
 61 # The configuration can be specified only at startup time.
 62 # Later changes to above system property (e.g. via setProperty method),
 63 # this config file, the password file, or the access file have no effect to the
 64 # running MBean server, the connector, or the registry.
 65 #
 66 
 67 #
 68 # ###################### RMI SSL #############################
 69 #
 70 # com.sun.management.jmxremote.ssl=true|false
 71 #      Default for this property is true. (Case for true/false ignored)
 72 #      If this property is specified as false then SSL is not used.
 73 #
 74 
 75 # For RMI monitoring without SSL use the following line
 76 # com.sun.management.jmxremote.ssl=false
 77 
 78 # com.sun.management.jmxremote.ssl.enabled.cipher.suites=<cipher-suites>
 79 #      The value of this property is a string that is a comma-separated list
 80 #      of SSL/TLS cipher suites to enable. This property can be specified in
 81 #      conjunction with the previous property "com.sun.management.jmxremote.ssl"
 82 #      in order to control which particular SSL/TLS cipher suites are enabled
 83 #      for use by accepted connections. If this property is not specified then
 84 #      the SSL/TLS RMI Server Socket Factory uses the SSL/TLS cipher suites that
 85 #      are enabled by default.
 86 #
 87 
 88 # com.sun.management.jmxremote.ssl.enabled.protocols=<protocol-versions>
 89 #      The value of this property is a string that is a comma-separated list
 90 #      of SSL/TLS protocol versions to enable. This property can be specified in
 91 #      conjunction with the previous property "com.sun.management.jmxremote.ssl"
 92 #      in order to control which particular SSL/TLS protocol versions are
 93 #      enabled for use by accepted connections. If this property is not
 94 #      specified then the SSL/TLS RMI Server Socket Factory uses the SSL/TLS
 95 #      protocol versions that are enabled by default.
 96 #
 97 
 98 # com.sun.management.jmxremote.ssl.need.client.auth=true|false
 99 #      Default for this property is false. (Case for true/false ignored)
100 #      If this property is specified as true in conjunction with the previous
101 #      property "com.sun.management.jmxremote.ssl" then the SSL/TLS RMI Server
102 #      Socket Factory will require client authentication.
103 #
104 
105 # For RMI monitoring with SSL client authentication use the following line
106 # com.sun.management.jmxremote.ssl.need.client.auth=true
107 
108 # com.sun.management.jmxremote.registry.ssl=true|false
109 #      Default for this property is false. (Case for true/false ignored)
110 #      If this property is specified as true then the RMI registry used
111 #      to bind the RMIServer remote object is protected with SSL/TLS
112 #      RMI Socket Factories that can be configured with the properties:
113 #          com.sun.management.jmxremote.ssl.enabled.cipher.suites
114 #          com.sun.management.jmxremote.ssl.enabled.protocols
115 #          com.sun.management.jmxremote.ssl.need.client.auth
116 #      If the two properties below are true at the same time, i.e.
117 #          com.sun.management.jmxremote.ssl=true
118 #          com.sun.management.jmxremote.registry.ssl=true
119 #      then the RMIServer remote object and the RMI registry are
120 #      both exported with the same SSL/TLS RMI Socket Factories.
121 #
122 
123 # For using an SSL/TLS protected RMI registry use the following line
124 com.sun.management.jmxremote.registry.ssl=true
125 
126 #
127 # ################ RMI User authentication ################
128 #
129 # com.sun.management.jmxremote.authenticate=true|false
130 #      Default for this property is true. (Case for true/false ignored)
131 #      If this property is specified as false then no authentication is
132 #      performed and all users are allowed all access.
133 #
134 
135 # For RMI monitoring without any checking use the following line
136 com.sun.management.jmxremote.authenticate=false
137 
138 #
139 # ################ RMI Login configuration ###################
140 #
141 # com.sun.management.jmxremote.login.config=<config-name>
142 #      Specifies the name of a JAAS login configuration entry to use when
143 #      authenticating users of RMI monitoring.
144 #
145 #      Setting this property is optional - the default login configuration
146 #      specifies a file-based authentication that uses the password file.
147 #
148 #      When using this property to override the default login configuration
149 #      then the named configuration entry must be in a file that gets loaded
150 #      by JAAS. In addition, the login module(s) specified in the configuration
151 #      should use the name and/or password callbacks to acquire the user's
152 #      credentials. See the NameCallback and PasswordCallback classes in the
153 #      javax.security.auth.callback package for more details.
154 #
155 #      If the property "com.sun.management.jmxremote.authenticate" is set to
156 #      false, then this property and the password & access files are ignored.
157 #
158 
159 # For a non-default login configuration use the following line
160 # com.sun.management.jmxremote.login.config=<config-name>
161 
162 #
163 # ################ RMI Password file location ##################
164 #
165 # com.sun.management.jmxremote.password.file=filepath
166 #      Specifies location for password file
167 #      This is optional - default location is
168 #      $JRE/conf/management/jmxremote.password
169 #
170 #      If the property "com.sun.management.jmxremote.authenticate" is set to
171 #      false, then this property and the password & access files are ignored.
172 #      Otherwise the password file must exist and be in the valid format.
173 #      If the password file is empty or non-existent then no access is allowed.
174 #
175 
176 # For a non-default password file location use the following line
177 # com.sun.management.jmxremote.password.file=filepath
178 
179 #
180 # ################ RMI Access file location #####################
181 #
182 # com.sun.management.jmxremote.access.file=filepath
183 #      Specifies location for access  file
184 #      This is optional - default location is
185 #      $JRE/conf/management/jmxremote.access
186 #
187 #      If the property "com.sun.management.jmxremote.authenticate" is set to
188 #      false, then this property and the password & access files are ignored.
189 #      Otherwise, the access file must exist and be in the valid format.
190 #      If the access file is empty or non-existent then no access is allowed.
191 #
192 
193 # For a non-default password file location use the following line
194 # com.sun.management.jmxremote.access.file=filepath
