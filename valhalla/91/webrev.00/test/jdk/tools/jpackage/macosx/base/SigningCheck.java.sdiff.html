 86     private static void validateCertificateTrust(String name) {
 87         // Certificates using the default user name must be trusted by user.
 88         // User supplied certs whose trust is set to "Use System Defaults"
 89         // will not be listed as trusted by dump-trust-settings
 90         if (SigningBase.DEV_NAME.equals("jpackage.openjdk.java.net")) {
 91             List<String> result = new Executor()
 92                     .setExecutable("security")
 93                     .addArguments("dump-trust-settings")
 94                     .executeWithoutExitCodeCheckAndGetOutput();
 95             result.stream().forEachOrdered(TKit::trace);
 96             TKit.assertTextStream(name)
 97                     .predicate((line, what) -> line.trim().endsWith(what))
 98                     .orElseThrow(() -> TKit.throwSkippedException(
 99                             "Certifcate not trusted by current user: " + name))
100                     .apply(result.stream());
101         }
102     }
