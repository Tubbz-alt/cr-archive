25 #ifndef SHARE_GC_SHENANDOAH_SHENANDOAHPACER_INLINE_HPP
26 #define SHARE_GC_SHENANDOAH_SHENANDOAHPACER_INLINE_HPP
27 
28 #include "gc/shenandoah/shenandoahPacer.hpp"
29 #include "runtime/atomic.hpp"
30 
31 inline void ShenandoahPacer::report_mark(size_t words) {
32   report_internal(words);
33   report_progress_internal(words);
34 }
35 
36 inline void ShenandoahPacer::report_evac(size_t words) {
37   report_internal(words);
38 }
39 
40 inline void ShenandoahPacer::report_updaterefs(size_t words) {
41   report_internal(words);
42 }
43 
44 inline void ShenandoahPacer::report_alloc(size_t words) {
45   report_internal(words);
46 }
47 
48 inline void ShenandoahPacer::report_internal(size_t words) {
49   assert(ShenandoahPacing, "Only be here when pacing is enabled");
50   add_budget(words);
51 }
52 
53 inline void ShenandoahPacer::report_progress_internal(size_t words) {
54   assert(ShenandoahPacing, "Only be here when pacing is enabled");
55   STATIC_ASSERT(sizeof(size_t) <= sizeof(intptr_t));
56   Atomic::add(&_progress, (intptr_t)words);
57 }
58 
59 inline void ShenandoahPacer::add_budget(size_t words) {
60   STATIC_ASSERT(sizeof(size_t) <= sizeof(intptr_t));
61   intptr_t inc = (intptr_t) words;
62   intptr_t new_budget = Atomic::add(&_budget, inc);
63 
64   // Was the budget replenished beyond zero? Then all pacing claims
65   // are satisfied, notify the waiters. Avoid taking any locks here,
66   // as it can be called from hot paths and/or while holding other locks.
67   if (new_budget >= 0 && (new_budget - inc) < 0) {
68     _need_notify_waiters.try_set();
69   }
70 }
71 
72 #endif // SHARE_GC_SHENANDOAH_SHENANDOAHPACER_INLINE_HPP
