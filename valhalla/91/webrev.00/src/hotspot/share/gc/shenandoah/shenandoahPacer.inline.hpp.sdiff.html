48 inline void ShenandoahPacer::report_internal(size_t words) {
49   assert(ShenandoahPacing, "Only be here when pacing is enabled");










50   STATIC_ASSERT(sizeof(size_t) <= sizeof(intptr_t));
51   intptr_t inc = (intptr_t) words;
52   intptr_t new_budget = Atomic::add(&_budget, inc);
53 
54   // Was the budget replenished beyond zero? Then all pacing claims
55   // are satisfied, notify the waiters.

56   if (new_budget >= 0 && (new_budget - inc) < 0) {
57     notify_waiters();
58   }
59 }

59 inline void ShenandoahPacer::add_budget(size_t words) {
60   STATIC_ASSERT(sizeof(size_t) <= sizeof(intptr_t));
61   intptr_t inc = (intptr_t) words;
62   intptr_t new_budget = Atomic::add(&_budget, inc);
63 
64   // Was the budget replenished beyond zero? Then all pacing claims
65   // are satisfied, notify the waiters. Avoid taking any locks here,
66   // as it can be called from hot paths and/or while holding other locks.
67   if (new_budget >= 0 && (new_budget - inc) < 0) {
68     _need_notify_waiters.try_set();
69   }
70 }
