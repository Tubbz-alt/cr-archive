  95 // If any of the classes are being redefined, wait
  96 // Parallel constant pool merging leads to indeterminate constant pools.
  97 void VM_RedefineClasses::lock_classes() {
  98   MonitorLocker ml(RedefineClasses_lock);
  99   bool has_redefined;
 100   do {
 101     has_redefined = false;
 102     // Go through classes each time until none are being redefined.
 103     for (int i = 0; i < _class_count; i++) {
 104       if (get_ik(_class_defs[i].klass)->is_being_redefined()) {
 105         ml.wait();
 106         has_redefined = true;
 107         break;  // for loop
 108       }
 109     }
 110   } while (has_redefined);
 111   for (int i = 0; i < _class_count; i++) {
 112     get_ik(_class_defs[i].klass)->set_is_being_redefined(true);
 113   }
 114   ml.notify_all();
 115 }
 116 
 117 void VM_RedefineClasses::unlock_classes() {
 118   MonitorLocker ml(RedefineClasses_lock);
 119   for (int i = 0; i < _class_count; i++) {
 120     assert(get_ik(_class_defs[i].klass)->is_being_redefined(),
 121            "should be being redefined to get here");
 122     get_ik(_class_defs[i].klass)->set_is_being_redefined(false);
 123   }
 124   ml.notify_all();
 125 }

 127 bool VM_RedefineClasses::doit_prologue() {
 128   if (_class_count == 0) {
 129     _res = JVMTI_ERROR_NONE;
 130     return false;
 131   }
 132   if (_class_defs == NULL) {
 133     _res = JVMTI_ERROR_NULL_POINTER;
 134     return false;
 135   }
 136 
 137   for (int i = 0; i < _class_count; i++) {
 138     if (_class_defs[i].klass == NULL) {
 139       _res = JVMTI_ERROR_INVALID_CLASS;
 140       return false;
 141     }
 142     if (_class_defs[i].class_byte_count == 0) {
 143       _res = JVMTI_ERROR_INVALID_CLASS_FORMAT;
 144       return false;
 145     }
 146     if (_class_defs[i].class_bytes == NULL) {
 147       _res = JVMTI_ERROR_NULL_POINTER;
 148       return false;
 149     }
 150 
 151     oop mirror = JNIHandles::resolve_non_null(_class_defs[i].klass);
 152     // classes for primitives, arrays, hidden and vm unsafe anonymous classes
 153     // cannot be redefined.
 154     if (!is_modifiable_class(mirror)) {
 155       _res = JVMTI_ERROR_UNMODIFIABLE_CLASS;
 156       return false;
 157     }
 158   }
 159 
 160   // Start timer after all the sanity checks; not quite accurate, but
 161   // better than adding a bunch of stop() calls.
 162   if (log_is_enabled(Info, redefine, class, timer)) {
 163     _timer_vm_op_prologue.start();
 164   }
 165 
 166   lock_classes();
 167   // We first load new class versions in the prologue, because somewhere down the
 168   // call chain it is required that the current thread is a Java thread.
 169   _res = load_new_class_versions(Thread::current());
 170   if (_res != JVMTI_ERROR_NONE) {
 171     // free any successfully created classes, since none are redefined
 172     for (int i = 0; i < _class_count; i++) {
 173       if (_scratch_classes[i] != NULL) {
 174         ClassLoaderData* cld = _scratch_classes[i]->class_loader_data();
 175         // Free the memory for this class at class unloading time.  Not before
 176         // because CMS might think this is still live.
 177         InstanceKlass* ik = get_ik(_class_defs[i].klass);
 178         if (ik->get_cached_class_file() == _scratch_classes[i]->get_cached_class_file()) {
 179           // Don't double-free cached_class_file copied from the original class if error.
 180           _scratch_classes[i]->set_cached_class_file(NULL);
 181         }
 182         cld->add_to_deallocate_list(InstanceKlass::cast(_scratch_classes[i]));
 183       }
 184     }
 185     // Free os::malloc allocated memory in load_new_class_version.
 186     os::free(_scratch_classes);
 187     _timer_vm_op_prologue.stop();
 188     unlock_classes();
 189     return false;
 190   }
 191 
 192   _timer_vm_op_prologue.stop();
 193   return true;
 194 }

 196 void VM_RedefineClasses::doit() {
 197   Thread *thread = Thread::current();
 198 
 199 #if INCLUDE_CDS
 200   if (UseSharedSpaces) {
 201     // Sharing is enabled so we remap the shared readonly space to
 202     // shared readwrite, private just in case we need to redefine
 203     // a shared class. We do the remap during the doit() phase of
 204     // the safepoint to be safer.
 205     if (!MetaspaceShared::remap_shared_readonly_as_readwrite()) {
 206       log_info(redefine, class, load)("failed to remap shared readonly space to readwrite, private");
 207       _res = JVMTI_ERROR_INTERNAL;
 208       return;
 209     }
 210   }
 211 #endif
 212 
 213   // Mark methods seen on stack and everywhere else so old methods are not
 214   // cleaned up if they're on the stack.
 215   MetadataOnStackMark md_on_stack(/*walk_all_metadata*/true, /*redefinition_walk*/true);
 216   HandleMark hm(thread);   // make sure any handles created are deleted
 217                            // before the stack walk again.
 218 
 219   for (int i = 0; i < _class_count; i++) {
 220     redefine_single_class(_class_defs[i].klass, _scratch_classes[i], thread);
 221   }
 222 
 223   // Flush all compiled code that depends on the classes redefined.
 224   flush_dependent_code();
 225 
 226   // Adjust constantpool caches and vtables for all classes
 227   // that reference methods of the evolved classes.
 228   // Have to do this after all classes are redefined and all methods that
 229   // are redefined are marked as old.
 230   AdjustAndCleanMetadata adjust_and_clean_metadata(thread);
 231   ClassLoaderDataGraph::classes_do(&adjust_and_clean_metadata);
 232 
 233   // JSR-292 support
 234   if (_any_class_has_resolved_methods) {
 235     bool trace_name_printed = false;
 236     ResolvedMethodTable::adjust_method_entries(&trace_name_printed);
 237   }
 238 
 239   // Increment flag indicating that some invariants are no longer true.
 240   // See jvmtiExport.hpp for detailed explanation.
 241   JvmtiExport::increment_redefinition_count();
 242 
 243   // check_class() is optionally called for product bits, but is
 244   // always called for non-product bits.
 245 #ifdef PRODUCT
 246   if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
 247 #endif
 248     log_trace(redefine, class, obsolete, metadata)("calling check_class");
 249     CheckClass check_class(thread);
 250     ClassLoaderDataGraph::classes_do(&check_class);
 251 #ifdef PRODUCT
 252   }
 253 #endif
 254 
 255   // Clean up any metadata now unreferenced while MetadataOnStackMark is set.
 256   ClassLoaderDataGraph::clean_deallocate_lists(false);
 257 }

 259 void VM_RedefineClasses::doit_epilogue() {
 260   unlock_classes();
 261 
 262   // Free os::malloc allocated memory.
 263   os::free(_scratch_classes);
 264 
 265   // Reset the_class to null for error printing.
 266   _the_class = NULL;
 267 
 268   if (log_is_enabled(Info, redefine, class, timer)) {
 269     // Used to have separate timers for "doit" and "all", but the timer
 270     // overhead skewed the measurements.
 271     julong doit_time = _timer_rsc_phase1.milliseconds() +
 272                        _timer_rsc_phase2.milliseconds();
 273     julong all_time = _timer_vm_op_prologue.milliseconds() + doit_time;
 274 
 275     log_info(redefine, class, timer)
 276       ("vm_op: all=" JULONG_FORMAT "  prologue=" JULONG_FORMAT "  doit=" JULONG_FORMAT,
 277        all_time, (julong)_timer_vm_op_prologue.milliseconds(), doit_time);
 278     log_info(redefine, class, timer)
 279       ("redefine_single_class: phase1=" JULONG_FORMAT "  phase2=" JULONG_FORMAT,
 280        (julong)_timer_rsc_phase1.milliseconds(), (julong)_timer_rsc_phase2.milliseconds());
 281   }
 282 }

 284 bool VM_RedefineClasses::is_modifiable_class(oop klass_mirror) {
 285   // classes for primitives cannot be redefined
 286   if (java_lang_Class::is_primitive(klass_mirror)) {
 287     return false;
 288   }
 289   Klass* k = java_lang_Class::as_Klass(klass_mirror);
 290   // classes for arrays cannot be redefined
 291   if (k == NULL || !k->is_instance_klass()) {
 292     return false;
 293   }
 294 
 295   // Cannot redefine or retransform a hidden or an unsafe anonymous class.
 296   if (InstanceKlass::cast(k)->is_hidden() ||
 297       InstanceKlass::cast(k)->is_unsafe_anonymous()) {
 298     return false;
 299   }
 300   return true;
 301 }

 303 // Append the current entry at scratch_i in scratch_cp to *merge_cp_p
 304 // where the end of *merge_cp_p is specified by *merge_cp_length_p. For
 305 // direct CP entries, there is just the current entry to append. For
 306 // indirect and double-indirect CP entries, there are zero or more
 307 // referenced CP entries along with the current entry to append.
 308 // Indirect and double-indirect CP entries are handled by recursive
 309 // calls to append_entry() as needed. The referenced CP entries are
 310 // always appended to *merge_cp_p before the referee CP entry. These
 311 // referenced CP entries may already exist in *merge_cp_p in which case
 312 // there is nothing extra to append and only the current entry is
 313 // appended.
 314 void VM_RedefineClasses::append_entry(const constantPoolHandle& scratch_cp,
 315        int scratch_i, constantPoolHandle *merge_cp_p, int *merge_cp_length_p,
 316        TRAPS) {
 317 
 318   // append is different depending on entry tag type
 319   switch (scratch_cp->tag_at(scratch_i).value()) {
 320 
 321     // The old verifier is implemented outside the VM. It loads classes,
 322     // but does not resolve constant pool entries directly so we never
 323     // see Class entries here with the old verifier. Similarly the old
 324     // verifier does not like Class entries in the input constant pool.
 325     // The split-verifier is implemented in the VM so it can optionally
 326     // and directly resolve constant pool entries to load classes. The
 327     // split-verifier can accept either Class entries or UnresolvedClass
 328     // entries in the input constant pool. We revert the appended copy
 329     // back to UnresolvedClass so that either verifier will be happy
 330     // with the constant pool entry.
 331     //
 332     // this is an indirect CP entry so it needs special handling
 333     case JVM_CONSTANT_Class:
 334     case JVM_CONSTANT_UnresolvedClass:
 335     {
 336       int name_i = scratch_cp->klass_name_index_at(scratch_i);
 337       int new_name_i = find_or_append_indirect_entry(scratch_cp, name_i, merge_cp_p,
 338                                                      merge_cp_length_p, THREAD);
 339 
 340       if (new_name_i != name_i) {
 341         log_trace(redefine, class, constantpool)
 342           ("Class entry@%d name_index change: %d to %d",
 343            *merge_cp_length_p, name_i, new_name_i);
 344       }
 345 
 346       (*merge_cp_p)->temp_unresolved_klass_at_put(*merge_cp_length_p, new_name_i);
 347       if (scratch_i != *merge_cp_length_p) {
 348         // The new entry in *merge_cp_p is at a different index than
 349         // the new entry in scratch_cp so we need to map the index values.
 350         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 351       }
 352       (*merge_cp_length_p)++;
 353     } break;
 354 
 355     // these are direct CP entries so they can be directly appended,
 356     // but double and long take two constant pool entries
 357     case JVM_CONSTANT_Double:  // fall through
 358     case JVM_CONSTANT_Long:
 359     {
 360       ConstantPool::copy_entry_to(scratch_cp, scratch_i, *merge_cp_p, *merge_cp_length_p,
 361         THREAD);
 362 
 363       if (scratch_i != *merge_cp_length_p) {
 364         // The new entry in *merge_cp_p is at a different index than
 365         // the new entry in scratch_cp so we need to map the index values.
 366         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 367       }
 368       (*merge_cp_length_p) += 2;
 369     } break;
 370 
 371     // these are direct CP entries so they can be directly appended
 372     case JVM_CONSTANT_Float:   // fall through
 373     case JVM_CONSTANT_Integer: // fall through
 374     case JVM_CONSTANT_Utf8:    // fall through
 375 
 376     // This was an indirect CP entry, but it has been changed into
 377     // Symbol*s so this entry can be directly appended.
 378     case JVM_CONSTANT_String:      // fall through
 379     {
 380       ConstantPool::copy_entry_to(scratch_cp, scratch_i, *merge_cp_p, *merge_cp_length_p,
 381         THREAD);
 382 
 383       if (scratch_i != *merge_cp_length_p) {
 384         // The new entry in *merge_cp_p is at a different index than
 385         // the new entry in scratch_cp so we need to map the index values.
 386         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 387       }
 388       (*merge_cp_length_p)++;
 389     } break;
 390 
 391     // this is an indirect CP entry so it needs special handling
 392     case JVM_CONSTANT_NameAndType:
 393     {
 394       int name_ref_i = scratch_cp->name_ref_index_at(scratch_i);
 395       int new_name_ref_i = find_or_append_indirect_entry(scratch_cp, name_ref_i, merge_cp_p,
 396                                                          merge_cp_length_p, THREAD);
 397 
 398       int signature_ref_i = scratch_cp->signature_ref_index_at(scratch_i);
 399       int new_signature_ref_i = find_or_append_indirect_entry(scratch_cp, signature_ref_i,
 400                                                               merge_cp_p, merge_cp_length_p,
 401                                                               THREAD);
 402 
 403       // If the referenced entries already exist in *merge_cp_p, then
 404       // both new_name_ref_i and new_signature_ref_i will both be 0.
 405       // In that case, all we are appending is the current entry.
 406       if (new_name_ref_i != name_ref_i) {
 407         log_trace(redefine, class, constantpool)
 408           ("NameAndType entry@%d name_ref_index change: %d to %d",
 409            *merge_cp_length_p, name_ref_i, new_name_ref_i);
 410       }
 411       if (new_signature_ref_i != signature_ref_i) {
 412         log_trace(redefine, class, constantpool)
 413           ("NameAndType entry@%d signature_ref_index change: %d to %d",
 414            *merge_cp_length_p, signature_ref_i, new_signature_ref_i);
 415       }
 416 
 417       (*merge_cp_p)->name_and_type_at_put(*merge_cp_length_p,
 418         new_name_ref_i, new_signature_ref_i);
 419       if (scratch_i != *merge_cp_length_p) {
 420         // The new entry in *merge_cp_p is at a different index than
 421         // the new entry in scratch_cp so we need to map the index values.
 422         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 423       }
 424       (*merge_cp_length_p)++;
 425     } break;
 426 
 427     // this is a double-indirect CP entry so it needs special handling
 428     case JVM_CONSTANT_Fieldref:           // fall through
 429     case JVM_CONSTANT_InterfaceMethodref: // fall through
 430     case JVM_CONSTANT_Methodref:
 431     {
 432       int klass_ref_i = scratch_cp->uncached_klass_ref_index_at(scratch_i);
 433       int new_klass_ref_i = find_or_append_indirect_entry(scratch_cp, klass_ref_i,
 434                                                           merge_cp_p, merge_cp_length_p, THREAD);
 435 
 436       int name_and_type_ref_i = scratch_cp->uncached_name_and_type_ref_index_at(scratch_i);
 437       int new_name_and_type_ref_i = find_or_append_indirect_entry(scratch_cp, name_and_type_ref_i,
 438                                                           merge_cp_p, merge_cp_length_p, THREAD);
 439 
 440       const char *entry_name = NULL;
 441       switch (scratch_cp->tag_at(scratch_i).value()) {
 442       case JVM_CONSTANT_Fieldref:
 443         entry_name = "Fieldref";
 444         (*merge_cp_p)->field_at_put(*merge_cp_length_p, new_klass_ref_i,
 445           new_name_and_type_ref_i);
 446         break;
 447       case JVM_CONSTANT_InterfaceMethodref:
 448         entry_name = "IFMethodref";
 449         (*merge_cp_p)->interface_method_at_put(*merge_cp_length_p,
 450           new_klass_ref_i, new_name_and_type_ref_i);
 451         break;
 452       case JVM_CONSTANT_Methodref:
 453         entry_name = "Methodref";
 454         (*merge_cp_p)->method_at_put(*merge_cp_length_p, new_klass_ref_i,
 455           new_name_and_type_ref_i);
 456         break;
 457       default:
 458         guarantee(false, "bad switch");
 459         break;
 460       }
 461 
 462       if (klass_ref_i != new_klass_ref_i) {
 463         log_trace(redefine, class, constantpool)
 464           ("%s entry@%d class_index changed: %d to %d", entry_name, *merge_cp_length_p, klass_ref_i, new_klass_ref_i);
 465       }
 466       if (name_and_type_ref_i != new_name_and_type_ref_i) {
 467         log_trace(redefine, class, constantpool)
 468           ("%s entry@%d name_and_type_index changed: %d to %d",
 469            entry_name, *merge_cp_length_p, name_and_type_ref_i, new_name_and_type_ref_i);
 470       }
 471 
 472       if (scratch_i != *merge_cp_length_p) {
 473         // The new entry in *merge_cp_p is at a different index than
 474         // the new entry in scratch_cp so we need to map the index values.
 475         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 476       }
 477       (*merge_cp_length_p)++;
 478     } break;
 479 
 480     // this is an indirect CP entry so it needs special handling
 481     case JVM_CONSTANT_MethodType:
 482     {
 483       int ref_i = scratch_cp->method_type_index_at(scratch_i);
 484       int new_ref_i = find_or_append_indirect_entry(scratch_cp, ref_i, merge_cp_p,
 485                                                     merge_cp_length_p, THREAD);
 486       if (new_ref_i != ref_i) {
 487         log_trace(redefine, class, constantpool)
 488           ("MethodType entry@%d ref_index change: %d to %d", *merge_cp_length_p, ref_i, new_ref_i);
 489       }
 490       (*merge_cp_p)->method_type_index_at_put(*merge_cp_length_p, new_ref_i);
 491       if (scratch_i != *merge_cp_length_p) {
 492         // The new entry in *merge_cp_p is at a different index than
 493         // the new entry in scratch_cp so we need to map the index values.
 494         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 495       }
 496       (*merge_cp_length_p)++;
 497     } break;
 498 
 499     // this is an indirect CP entry so it needs special handling
 500     case JVM_CONSTANT_MethodHandle:
 501     {
 502       int ref_kind = scratch_cp->method_handle_ref_kind_at(scratch_i);
 503       int ref_i = scratch_cp->method_handle_index_at(scratch_i);
 504       int new_ref_i = find_or_append_indirect_entry(scratch_cp, ref_i, merge_cp_p,
 505                                                     merge_cp_length_p, THREAD);
 506       if (new_ref_i != ref_i) {
 507         log_trace(redefine, class, constantpool)
 508           ("MethodHandle entry@%d ref_index change: %d to %d", *merge_cp_length_p, ref_i, new_ref_i);
 509       }
 510       (*merge_cp_p)->method_handle_index_at_put(*merge_cp_length_p, ref_kind, new_ref_i);
 511       if (scratch_i != *merge_cp_length_p) {
 512         // The new entry in *merge_cp_p is at a different index than
 513         // the new entry in scratch_cp so we need to map the index values.
 514         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 515       }
 516       (*merge_cp_length_p)++;
 517     } break;
 518 
 519     // this is an indirect CP entry so it needs special handling
 520     case JVM_CONSTANT_Dynamic:  // fall through
 521     case JVM_CONSTANT_InvokeDynamic:
 522     {
 523       // Index of the bootstrap specifier in the operands array
 524       int old_bs_i = scratch_cp->bootstrap_methods_attribute_index(scratch_i);
 525       int new_bs_i = find_or_append_operand(scratch_cp, old_bs_i, merge_cp_p,
 526                                             merge_cp_length_p, THREAD);
 527       // The bootstrap method NameAndType_info index
 528       int old_ref_i = scratch_cp->bootstrap_name_and_type_ref_index_at(scratch_i);
 529       int new_ref_i = find_or_append_indirect_entry(scratch_cp, old_ref_i, merge_cp_p,
 530                                                     merge_cp_length_p, THREAD);
 531       if (new_bs_i != old_bs_i) {
 532         log_trace(redefine, class, constantpool)
 533           ("Dynamic entry@%d bootstrap_method_attr_index change: %d to %d",
 534            *merge_cp_length_p, old_bs_i, new_bs_i);
 535       }
 536       if (new_ref_i != old_ref_i) {
 537         log_trace(redefine, class, constantpool)
 538           ("Dynamic entry@%d name_and_type_index change: %d to %d", *merge_cp_length_p, old_ref_i, new_ref_i);
 539       }
 540 
 541       if (scratch_cp->tag_at(scratch_i).is_dynamic_constant())
 542         (*merge_cp_p)->dynamic_constant_at_put(*merge_cp_length_p, new_bs_i, new_ref_i);
 543       else
 544         (*merge_cp_p)->invoke_dynamic_at_put(*merge_cp_length_p, new_bs_i, new_ref_i);
 545       if (scratch_i != *merge_cp_length_p) {
 546         // The new entry in *merge_cp_p is at a different index than
 547         // the new entry in scratch_cp so we need to map the index values.
 548         map_index(scratch_cp, scratch_i, *merge_cp_length_p);
 549       }
 550       (*merge_cp_length_p)++;
 551     } break;
 552 
 553     // At this stage, Class or UnresolvedClass could be in scratch_cp, but not
 554     // ClassIndex
 555     case JVM_CONSTANT_ClassIndex: // fall through
 556 
 557     // Invalid is used as the tag for the second constant pool entry
 558     // occupied by JVM_CONSTANT_Double or JVM_CONSTANT_Long. It should
 559     // not be seen by itself.
 560     case JVM_CONSTANT_Invalid: // fall through
 561 
 562     // At this stage, String could be here, but not StringIndex
 563     case JVM_CONSTANT_StringIndex: // fall through
 564 
 565     // At this stage JVM_CONSTANT_UnresolvedClassInError should not be here
 566     case JVM_CONSTANT_UnresolvedClassInError: // fall through
 567 
 568     default:
 569     {
 570       // leave a breadcrumb
 571       jbyte bad_value = scratch_cp->tag_at(scratch_i).value();
 572       ShouldNotReachHere();
 573     } break;
 574   } // end switch tag value
 575 } // end append_entry()

 578 int VM_RedefineClasses::find_or_append_indirect_entry(const constantPoolHandle& scratch_cp,
 579       int ref_i, constantPoolHandle *merge_cp_p, int *merge_cp_length_p, TRAPS) {
 580 
 581   int new_ref_i = ref_i;
 582   bool match = (ref_i < *merge_cp_length_p) &&
 583                scratch_cp->compare_entry_to(ref_i, *merge_cp_p, ref_i, THREAD);
 584 
 585   if (!match) {
 586     // forward reference in *merge_cp_p or not a direct match
 587     int found_i = scratch_cp->find_matching_entry(ref_i, *merge_cp_p, THREAD);
 588     if (found_i != 0) {
 589       guarantee(found_i != ref_i, "compare_entry_to() and find_matching_entry() do not agree");
 590       // Found a matching entry somewhere else in *merge_cp_p so just need a mapping entry.
 591       new_ref_i = found_i;
 592       map_index(scratch_cp, ref_i, found_i);
 593     } else {
 594       // no match found so we have to append this entry to *merge_cp_p
 595       append_entry(scratch_cp, ref_i, merge_cp_p, merge_cp_length_p, THREAD);
 596       // The above call to append_entry() can only append one entry
 597       // so the post call query of *merge_cp_length_p is only for
 598       // the sake of consistency.
 599       new_ref_i = *merge_cp_length_p - 1;
 600     }
 601   }
 602 
 603   return new_ref_i;
 604 } // end find_or_append_indirect_entry()

 607 // Append a bootstrap specifier into the merge_cp operands that is semantically equal
 608 // to the scratch_cp operands bootstrap specifier passed by the old_bs_i index.
 609 // Recursively append new merge_cp entries referenced by the new bootstrap specifier.
 610 void VM_RedefineClasses::append_operand(const constantPoolHandle& scratch_cp, int old_bs_i,
 611        constantPoolHandle *merge_cp_p, int *merge_cp_length_p, TRAPS) {
 612 
 613   int old_ref_i = scratch_cp->operand_bootstrap_method_ref_index_at(old_bs_i);
 614   int new_ref_i = find_or_append_indirect_entry(scratch_cp, old_ref_i, merge_cp_p,
 615                                                 merge_cp_length_p, THREAD);
 616   if (new_ref_i != old_ref_i) {
 617     log_trace(redefine, class, constantpool)
 618       ("operands entry@%d bootstrap method ref_index change: %d to %d", _operands_cur_length, old_ref_i, new_ref_i);
 619   }
 620 
 621   Array<u2>* merge_ops = (*merge_cp_p)->operands();
 622   int new_bs_i = _operands_cur_length;
 623   // We have _operands_cur_length == 0 when the merge_cp operands is empty yet.
 624   // However, the operand_offset_at(0) was set in the extend_operands() call.
 625   int new_base = (new_bs_i == 0) ? (*merge_cp_p)->operand_offset_at(0)
 626                                  : (*merge_cp_p)->operand_next_offset_at(new_bs_i - 1);
 627   int argc     = scratch_cp->operand_argument_count_at(old_bs_i);
 628 
 629   ConstantPool::operand_offset_at_put(merge_ops, _operands_cur_length, new_base);
 630   merge_ops->at_put(new_base++, new_ref_i);
 631   merge_ops->at_put(new_base++, argc);
 632 
 633   for (int i = 0; i < argc; i++) {
 634     int old_arg_ref_i = scratch_cp->operand_argument_index_at(old_bs_i, i);
 635     int new_arg_ref_i = find_or_append_indirect_entry(scratch_cp, old_arg_ref_i, merge_cp_p,
 636                                                       merge_cp_length_p, THREAD);
 637     merge_ops->at_put(new_base++, new_arg_ref_i);
 638     if (new_arg_ref_i != old_arg_ref_i) {
 639       log_trace(redefine, class, constantpool)
 640         ("operands entry@%d bootstrap method argument ref_index change: %d to %d",
 641          _operands_cur_length, old_arg_ref_i, new_arg_ref_i);
 642     }
 643   }
 644   if (old_bs_i != _operands_cur_length) {
 645     // The bootstrap specifier in *merge_cp_p is at a different index than
 646     // that in scratch_cp so we need to map the index values.
 647     map_operand_index(old_bs_i, new_bs_i);
 648   }
 649   _operands_cur_length++;
 650 } // end append_operand()

 653 int VM_RedefineClasses::find_or_append_operand(const constantPoolHandle& scratch_cp,
 654       int old_bs_i, constantPoolHandle *merge_cp_p, int *merge_cp_length_p, TRAPS) {
 655 
 656   int new_bs_i = old_bs_i; // bootstrap specifier index
 657   bool match = (old_bs_i < _operands_cur_length) &&
 658                scratch_cp->compare_operand_to(old_bs_i, *merge_cp_p, old_bs_i, THREAD);
 659 
 660   if (!match) {
 661     // forward reference in *merge_cp_p or not a direct match
 662     int found_i = scratch_cp->find_matching_operand(old_bs_i, *merge_cp_p,
 663                                                     _operands_cur_length, THREAD);
 664     if (found_i != -1) {
 665       guarantee(found_i != old_bs_i, "compare_operand_to() and find_matching_operand() disagree");
 666       // found a matching operand somewhere else in *merge_cp_p so just need a mapping
 667       new_bs_i = found_i;
 668       map_operand_index(old_bs_i, found_i);
 669     } else {
 670       // no match found so we have to append this bootstrap specifier to *merge_cp_p
 671       append_operand(scratch_cp, old_bs_i, merge_cp_p, merge_cp_length_p, THREAD);
 672       new_bs_i = _operands_cur_length - 1;
 673     }
 674   }
 675   return new_bs_i;
 676 } // end find_or_append_operand()

 711 // The caller must have an active ResourceMark.
 712 static jvmtiError check_attribute_arrays(const char* attr_name,
 713            InstanceKlass* the_class, InstanceKlass* scratch_class,
 714            Array<u2>* the_array, Array<u2>* scr_array) {
 715   bool the_array_exists = the_array != Universe::the_empty_short_array();
 716   bool scr_array_exists = scr_array != Universe::the_empty_short_array();
 717 
 718   int array_len = the_array->length();
 719   if (the_array_exists && scr_array_exists) {
 720     if (array_len != scr_array->length()) {
 721       log_trace(redefine, class)
 722         ("redefined class %s attribute change error: %s len=%d changed to len=%d",
 723          the_class->external_name(), attr_name, array_len, scr_array->length());
 724       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 725     }
 726 
 727     // The order of entries in the attribute array is not specified so we
 728     // have to explicitly check for the same contents. We do this by copying
 729     // the referenced symbols into their own arrays, sorting them and then
 730     // comparing each element pair.
 731 
 732     Symbol** the_syms = NEW_RESOURCE_ARRAY_RETURN_NULL(Symbol*, array_len);
 733     Symbol** scr_syms = NEW_RESOURCE_ARRAY_RETURN_NULL(Symbol*, array_len);
 734 
 735     if (the_syms == NULL || scr_syms == NULL) {
 736       return JVMTI_ERROR_OUT_OF_MEMORY;
 737     }
 738 
 739     for (int i = 0; i < array_len; i++) {
 740       int the_cp_index = the_array->at(i);
 741       int scr_cp_index = scr_array->at(i);
 742       the_syms[i] = the_class->constants()->klass_name_at(the_cp_index);
 743       scr_syms[i] = scratch_class->constants()->klass_name_at(scr_cp_index);
 744     }
 745 
 746     qsort(the_syms, array_len, sizeof(Symbol*), symcmp);
 747     qsort(scr_syms, array_len, sizeof(Symbol*), symcmp);
 748 
 749     for (int i = 0; i < array_len; i++) {
 750       if (the_syms[i] != scr_syms[i]) {
 751         log_trace(redefine, class)
 752           ("redefined class %s attribute change error: %s[%d]: %s changed to %s",
 753            the_class->external_name(), attr_name, i,
 754            the_syms[i]->as_C_string(), scr_syms[i]->as_C_string());
 755         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 756       }
 757     }
 758   } else if (the_array_exists ^ scr_array_exists) {
 759     const char* action_str = (the_array_exists) ? "removed" : "added";
 760     log_trace(redefine, class)
 761       ("redefined class %s attribute change error: %s attribute %s",
 762        the_class->external_name(), attr_name, action_str);
 763     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 764   }
 765   return JVMTI_ERROR_NONE;
 766 }

 802   // Get lists of record components.
 803   Array<RecordComponent*>* the_record = the_class->record_components();
 804   Array<RecordComponent*>* scr_record = scratch_class->record_components();
 805   bool the_record_exists = the_record != NULL;
 806   bool scr_record_exists = scr_record != NULL;
 807 
 808   if (the_record_exists && scr_record_exists) {
 809     int the_num_components = the_record->length();
 810     int scr_num_components = scr_record->length();

 813         ("redefined class %s attribute change error: Record num_components=%d changed to num_components=%d",
 814          the_class->external_name(), the_num_components, scr_num_components);
 815       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 816     }
 817 
 818     // Compare each field in each record component.
 819     ConstantPool* the_cp =  the_class->constants();
 820     ConstantPool* scr_cp =  scratch_class->constants();
 821     for (int x = 0; x < the_num_components; x++) {
 822       RecordComponent* the_component = the_record->at(x);
 823       RecordComponent* scr_component = scr_record->at(x);
 824       const Symbol* const the_name = the_cp->symbol_at(the_component->name_index());
 825       const Symbol* const scr_name = scr_cp->symbol_at(scr_component->name_index());
 826       const Symbol* const the_descr = the_cp->symbol_at(the_component->descriptor_index());
 827       const Symbol* const scr_descr = scr_cp->symbol_at(scr_component->descriptor_index());
 828       if (the_name != scr_name || the_descr != scr_descr) {
 829         log_trace(redefine, class, record)
 830           ("redefined class %s attribute change error: Record name_index, descriptor_index, and/or attributes_count changed",
 831            the_class->external_name());
 832         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 833       }
 834 
 835       int the_gen_sig = the_component->generic_signature_index();
 836       int scr_gen_sig = scr_component->generic_signature_index();
 837       const Symbol* const the_gen_sig_sym = (the_gen_sig == 0 ? NULL :
 838         the_cp->symbol_at(the_component->generic_signature_index()));
 839       const Symbol* const scr_gen_sig_sym = (scr_gen_sig == 0 ? NULL :
 840         scr_cp->symbol_at(scr_component->generic_signature_index()));
 841       if (the_gen_sig_sym != scr_gen_sig_sym) {
 842         log_trace(redefine, class, record)
 843           ("redefined class %s attribute change error: Record generic_signature attribute changed",
 844            the_class->external_name());
 845         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 846       }
 847 
 848       // It's okay if a record component's annotations were changed.
 849     }
 850 
 851   } else if (the_record_exists ^ scr_record_exists) {
 852     const char* action_str = (the_record_exists) ? "removed" : "added";
 853     log_trace(redefine, class, record)
 854       ("redefined class %s attribute change error: Record attribute %s",
 855        the_class->external_name(), action_str);
 856     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_ATTRIBUTE_CHANGED;
 857   }
 858 
 859   return JVMTI_ERROR_NONE;
 860 }

 881 jvmtiError VM_RedefineClasses::compare_and_normalize_class_versions(
 882              InstanceKlass* the_class,
 883              InstanceKlass* scratch_class) {
 884   int i;
 885 
 886   // Check superclasses, or rather their names, since superclasses themselves can be
 887   // requested to replace.
 888   // Check for NULL superclass first since this might be java.lang.Object
 889   if (the_class->super() != scratch_class->super() &&
 890       (the_class->super() == NULL || scratch_class->super() == NULL ||
 891        the_class->super()->name() !=
 892        scratch_class->super()->name())) {
 893     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_HIERARCHY_CHANGED;
 894   }
 895 
 896   // Check if the number, names and order of directly implemented interfaces are the same.
 897   // I think in principle we should just check if the sets of names of directly implemented
 898   // interfaces are the same, i.e. the order of declaration (which, however, if changed in the
 899   // .java file, also changes in .class file) should not matter. However, comparing sets is
 900   // technically a bit more difficult, and, more importantly, I am not sure at present that the
 901   // order of interfaces does not matter on the implementation level, i.e. that the VM does not
 902   // rely on it somewhere.
 903   Array<InstanceKlass*>* k_interfaces = the_class->local_interfaces();
 904   Array<InstanceKlass*>* k_new_interfaces = scratch_class->local_interfaces();
 905   int n_intfs = k_interfaces->length();
 906   if (n_intfs != k_new_interfaces->length()) {
 907     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_HIERARCHY_CHANGED;
 908   }
 909   for (i = 0; i < n_intfs; i++) {
 910     if (k_interfaces->at(i)->name() !=
 911         k_new_interfaces->at(i)->name()) {
 912       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_HIERARCHY_CHANGED;
 913     }
 914   }
 915 
 916   // Check whether class is in the error init state.
 917   if (the_class->is_in_error_state()) {
 918     // TBD #5057930: special error code is needed in 1.6
 919     return JVMTI_ERROR_INVALID_CLASS;
 920   }
 921 
 922   // Check whether the nest-related attributes have been changed.
 923   jvmtiError err = check_nest_attributes(the_class, scratch_class);
 924   if (err != JVMTI_ERROR_NONE) {
 925     return err;
 926   }
 927 
 928   // Check whether the Record attribute has been changed.
 929   err = check_record_attribute(the_class, scratch_class);
 930   if (err != JVMTI_ERROR_NONE) {
 931     return err;
 932   }
 933 
 934   // Check whether the PermittedSubclasses attribute has been changed.
 935   err = check_permitted_subclasses_attribute(the_class, scratch_class);
 936   if (err != JVMTI_ERROR_NONE) {
 937     return err;
 938   }
 939 
 940   // Check whether class modifiers are the same.
 941   jushort old_flags = (jushort) the_class->access_flags().get_flags();
 942   jushort new_flags = (jushort) scratch_class->access_flags().get_flags();
 943   if (old_flags != new_flags) {
 944     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_CLASS_MODIFIERS_CHANGED;
 945   }
 946 
 947   // Check if the number, names, types and order of fields declared in these classes
 948   // are the same.
 949   JavaFieldStream old_fs(the_class);
 950   JavaFieldStream new_fs(scratch_class);
 951   for (; !old_fs.done() && !new_fs.done(); old_fs.next(), new_fs.next()) {
 952     // access
 953     old_flags = old_fs.access_flags().as_short();
 954     new_flags = new_fs.access_flags().as_short();
 955     if ((old_flags ^ new_flags) & JVM_RECOGNIZED_FIELD_MODIFIERS) {
 956       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_SCHEMA_CHANGED;
 957     }
 958     // offset
 959     if (old_fs.offset() != new_fs.offset()) {
 960       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_SCHEMA_CHANGED;
 961     }
 962     // name and signature
 963     Symbol* name_sym1 = the_class->constants()->symbol_at(old_fs.name_index());
 964     Symbol* sig_sym1 = the_class->constants()->symbol_at(old_fs.signature_index());
 965     Symbol* name_sym2 = scratch_class->constants()->symbol_at(new_fs.name_index());
 966     Symbol* sig_sym2 = scratch_class->constants()->symbol_at(new_fs.signature_index());
 967     if (name_sym1 != name_sym2 || sig_sym1 != sig_sym2) {
 968       return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_SCHEMA_CHANGED;
 969     }
 970   }
 971 
 972   // If both streams aren't done then we have a differing number of
 973   // fields.
 974   if (!old_fs.done() || !new_fs.done()) {
 975     return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_SCHEMA_CHANGED;
 976   }
 977 
 978   // Do a parallel walk through the old and new methods. Detect
 979   // cases where they match (exist in both), have been added in
 980   // the new methods, or have been deleted (exist only in the
 981   // old methods).  The class file parser places methods in order
 982   // by method name, but does not order overloaded methods by
 983   // signature.  In order to determine what fate befell the methods,
 984   // this code places the overloaded new methods that have matching
 985   // old methods in the same order as the old methods and places
 986   // new overloaded methods at the end of overloaded methods of
 987   // that name. The code for this order normalization is adapted
 988   // from the algorithm used in InstanceKlass::find_method().
 989   // Since we are swapping out of order entries as we find them,
 990   // we only have to search forward through the overloaded methods.
 991   // Methods which are added and have the same name as an existing
 992   // method (but different signature) will be put at the end of
 993   // the methods with that name, and the name mismatch code will
 994   // handle them.
 995   Array<Method*>* k_old_methods(the_class->methods());
 996   Array<Method*>* k_new_methods(scratch_class->methods());
 997   int n_old_methods = k_old_methods->length();
 998   int n_new_methods = k_new_methods->length();
 999   Thread* thread = Thread::current();
1000 
1001   int ni = 0;
1002   int oi = 0;
1003   while (true) {
1004     Method* k_old_method;
1005     Method* k_new_method;
1006     enum { matched, added, deleted, undetermined } method_was = undetermined;
1007 
1008     if (oi >= n_old_methods) {
1009       if (ni >= n_new_methods) {
1010         break; // we've looked at everything, done
1011       }
1012       // New method at the end
1013       k_new_method = k_new_methods->at(ni);
1014       method_was = added;
1015     } else if (ni >= n_new_methods) {
1016       // Old method, at the end, is deleted
1017       k_old_method = k_old_methods->at(oi);
1018       method_was = deleted;
1019     } else {
1020       // There are more methods in both the old and new lists
1021       k_old_method = k_old_methods->at(oi);
1022       k_new_method = k_new_methods->at(ni);
1023       if (k_old_method->name() != k_new_method->name()) {
1024         // Methods are sorted by method name, so a mismatch means added
1025         // or deleted
1026         if (k_old_method->name()->fast_compare(k_new_method->name()) > 0) {
1027           method_was = added;
1028         } else {
1029           method_was = deleted;
1030         }
1031       } else if (k_old_method->signature() == k_new_method->signature()) {
1032         // Both the name and signature match
1033         method_was = matched;
1034       } else {
1035         // The name matches, but the signature doesn't, which means we have to
1036         // search forward through the new overloaded methods.
1037         int nj;  // outside the loop for post-loop check
1038         for (nj = ni + 1; nj < n_new_methods; nj++) {
1039           Method* m = k_new_methods->at(nj);
1040           if (k_old_method->name() != m->name()) {
1041             // reached another method name so no more overloaded methods
1042             method_was = deleted;
1043             break;
1044           }
1045           if (k_old_method->signature() == m->signature()) {
1046             // found a match so swap the methods
1047             k_new_methods->at_put(ni, m);
1048             k_new_methods->at_put(nj, k_new_method);
1049             k_new_method = m;
1050             method_was = matched;
1051             break;
1052           }
1053         }
1054 
1055         if (nj >= n_new_methods) {
1056           // reached the end without a match; so method was deleted
1057           method_was = deleted;
1058         }
1059       }
1060     }
1061 
1062     switch (method_was) {
1063     case matched:
1064       // methods match, be sure modifiers do too
1065       old_flags = (jushort) k_old_method->access_flags().get_flags();
1066       new_flags = (jushort) k_new_method->access_flags().get_flags();
1067       if ((old_flags ^ new_flags) & ~(JVM_ACC_NATIVE)) {
1068         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_METHOD_MODIFIERS_CHANGED;
1069       }
1070       {
1071         u2 new_num = k_new_method->method_idnum();
1072         u2 old_num = k_old_method->method_idnum();
1073         if (new_num != old_num) {
1074           Method* idnum_owner = scratch_class->method_with_idnum(old_num);
1075           if (idnum_owner != NULL) {
1076             // There is already a method assigned this idnum -- switch them
1077             // Take current and original idnum from the new_method
1078             idnum_owner->set_method_idnum(new_num);
1079             idnum_owner->set_orig_method_idnum(k_new_method->orig_method_idnum());
1080           }
1081           // Take current and original idnum from the old_method
1082           k_new_method->set_method_idnum(old_num);
1083           k_new_method->set_orig_method_idnum(k_old_method->orig_method_idnum());
1084           if (thread->has_pending_exception()) {
1085             return JVMTI_ERROR_OUT_OF_MEMORY;
1086           }
1087         }
1088       }
1089       log_trace(redefine, class, normalize)
1090         ("Method matched: new: %s [%d] == old: %s [%d]",
1091          k_new_method->name_and_sig_as_C_string(), ni, k_old_method->name_and_sig_as_C_string(), oi);
1092       // advance to next pair of methods
1093       ++oi;
1094       ++ni;
1095       break;
1096     case added:
1097       // method added, see if it is OK
1098       if (!can_add_or_delete(k_new_method)) {
1099         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_METHOD_ADDED;
1100       }
1101       {
1102         u2 num = the_class->next_method_idnum();
1103         if (num == ConstMethod::UNSET_IDNUM) {
1104           // cannot add any more methods
1105           return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_METHOD_ADDED;
1106         }
1107         u2 new_num = k_new_method->method_idnum();
1108         Method* idnum_owner = scratch_class->method_with_idnum(num);
1109         if (idnum_owner != NULL) {
1110           // There is already a method assigned this idnum -- switch them
1111           // Take current and original idnum from the new_method
1112           idnum_owner->set_method_idnum(new_num);
1113           idnum_owner->set_orig_method_idnum(k_new_method->orig_method_idnum());
1114         }
1115         k_new_method->set_method_idnum(num);
1116         k_new_method->set_orig_method_idnum(num);
1117         if (thread->has_pending_exception()) {
1118           return JVMTI_ERROR_OUT_OF_MEMORY;
1119         }
1120       }
1121       log_trace(redefine, class, normalize)
1122         ("Method added: new: %s [%d]", k_new_method->name_and_sig_as_C_string(), ni);
1123       ++ni; // advance to next new method
1124       break;
1125     case deleted:
1126       // method deleted, see if it is OK
1127       if (!can_add_or_delete(k_old_method)) {
1128         return JVMTI_ERROR_UNSUPPORTED_REDEFINITION_METHOD_DELETED;
1129       }
1130       log_trace(redefine, class, normalize)
1131         ("Method deleted: old: %s [%d]", k_old_method->name_and_sig_as_C_string(), oi);
1132       ++oi; // advance to next old method
1133       break;
1134     default:
1135       ShouldNotReachHere();
1136     }
1137   }
1138 
1139   return JVMTI_ERROR_NONE;
1140 }

1453 // Merge old_cp and scratch_cp and return the results of the merge via
1454 // merge_cp_p. The number of entries in *merge_cp_p is returned via
1455 // merge_cp_length_p. The entries in old_cp occupy the same locations
1456 // in *merge_cp_p. Also creates a map of indices from entries in
1457 // scratch_cp to the corresponding entry in *merge_cp_p. Index map
1458 // entries are only created for entries in scratch_cp that occupy a
1459 // different location in *merged_cp_p.
1460 bool VM_RedefineClasses::merge_constant_pools(const constantPoolHandle& old_cp,
1461        const constantPoolHandle& scratch_cp, constantPoolHandle *merge_cp_p,
1462        int *merge_cp_length_p, TRAPS) {
1463 
1464   if (merge_cp_p == NULL) {
1465     assert(false, "caller must provide scratch constantPool");
1466     return false; // robustness
1467   }
1468   if (merge_cp_length_p == NULL) {
1469     assert(false, "caller must provide scratch CP length");
1470     return false; // robustness
1471   }
1472   // Worst case we need old_cp->length() + scratch_cp()->length(),
1473   // but the caller might be smart so make sure we have at least
1474   // the minimum.
1475   if ((*merge_cp_p)->length() < old_cp->length()) {
1476     assert(false, "merge area too small");
1477     return false; // robustness
1478   }
1479 
1480   log_info(redefine, class, constantpool)("old_cp_len=%d, scratch_cp_len=%d", old_cp->length(), scratch_cp->length());
1481 
1482   {
1483     // Pass 0:
1484     // The old_cp is copied to *merge_cp_p; this means that any code
1485     // using old_cp does not have to change. This work looks like a
1486     // perfect fit for ConstantPool*::copy_cp_to(), but we need to
1487     // handle one special case:
1488     // - revert JVM_CONSTANT_Class to JVM_CONSTANT_UnresolvedClass
1489     // This will make verification happy.
1490 
1491     int old_i;  // index into old_cp
1492 
1493     // index zero (0) is not used in constantPools
1494     for (old_i = 1; old_i < old_cp->length(); old_i++) {
1495       // leave debugging crumb
1496       jbyte old_tag = old_cp->tag_at(old_i).value();
1497       switch (old_tag) {
1498       case JVM_CONSTANT_Class:
1499       case JVM_CONSTANT_UnresolvedClass:
1500         // revert the copy to JVM_CONSTANT_UnresolvedClass
1501         // May be resolving while calling this so do the same for
1502         // JVM_CONSTANT_UnresolvedClass (klass_name_at() deals with transition)
1503         (*merge_cp_p)->temp_unresolved_klass_at_put(old_i,
1504           old_cp->klass_name_index_at(old_i));
1505         break;
1506 
1507       case JVM_CONSTANT_Double:
1508       case JVM_CONSTANT_Long:
1509         // just copy the entry to *merge_cp_p, but double and long take
1510         // two constant pool entries
1511         ConstantPool::copy_entry_to(old_cp, old_i, *merge_cp_p, old_i, CHECK_false);
1512         old_i++;
1513         break;
1514 
1515       default:
1516         // just copy the entry to *merge_cp_p
1517         ConstantPool::copy_entry_to(old_cp, old_i, *merge_cp_p, old_i, CHECK_false);
1518         break;
1519       }
1520     } // end for each old_cp entry
1521 
1522     ConstantPool::copy_operands(old_cp, *merge_cp_p, CHECK_false);
1523     (*merge_cp_p)->extend_operands(scratch_cp, CHECK_false);
1524 
1525     // We don't need to sanity check that *merge_cp_length_p is within
1526     // *merge_cp_p bounds since we have the minimum on-entry check above.
1527     (*merge_cp_length_p) = old_i;
1528   }
1529 
1530   // merge_cp_len should be the same as old_cp->length() at this point
1531   // so this trace message is really a "warm-and-breathing" message.
1532   log_debug(redefine, class, constantpool)("after pass 0: merge_cp_len=%d", *merge_cp_length_p);
1533 
1534   int scratch_i;  // index into scratch_cp
1535   {
1536     // Pass 1a:
1537     // Compare scratch_cp entries to the old_cp entries that we have
1538     // already copied to *merge_cp_p. In this pass, we are eliminating
1539     // exact duplicates (matching entry at same index) so we only
1540     // compare entries in the common indice range.
1541     int increment = 1;
1542     int pass1a_length = MIN2(old_cp->length(), scratch_cp->length());
1543     for (scratch_i = 1; scratch_i < pass1a_length; scratch_i += increment) {
1544       switch (scratch_cp->tag_at(scratch_i).value()) {
1545       case JVM_CONSTANT_Double:
1546       case JVM_CONSTANT_Long:
1547         // double and long take two constant pool entries
1548         increment = 2;
1549         break;
1550 
1551       default:
1552         increment = 1;
1553         break;
1554       }
1555 
1556       bool match = scratch_cp->compare_entry_to(scratch_i, *merge_cp_p,
1557         scratch_i, CHECK_false);
1558       if (match) {
1559         // found a match at the same index so nothing more to do
1560         continue;
1561       } else if (is_unresolved_class_mismatch(scratch_cp, scratch_i,
1562                                               *merge_cp_p, scratch_i)) {
1563         // The mismatch in compare_entry_to() above is because of a
1564         // resolved versus unresolved class entry at the same index
1565         // with the same string value. Since Pass 0 reverted any
1566         // class entries to unresolved class entries in *merge_cp_p,
1567         // we go with the unresolved class entry.
1568         continue;
1569       }
1570 
1571       int found_i = scratch_cp->find_matching_entry(scratch_i, *merge_cp_p,
1572         CHECK_false);
1573       if (found_i != 0) {
1574         guarantee(found_i != scratch_i,
1575           "compare_entry_to() and find_matching_entry() do not agree");
1576 
1577         // Found a matching entry somewhere else in *merge_cp_p so
1578         // just need a mapping entry.
1579         map_index(scratch_cp, scratch_i, found_i);
1580         continue;
1581       }
1582 
1583       // The find_matching_entry() call above could fail to find a match
1584       // due to a resolved versus unresolved class or string entry situation
1585       // like we solved above with the is_unresolved_*_mismatch() calls.
1586       // However, we would have to call is_unresolved_*_mismatch() over
1587       // all of *merge_cp_p (potentially) and that doesn't seem to be
1588       // worth the time.
1589 
1590       // No match found so we have to append this entry and any unique
1591       // referenced entries to *merge_cp_p.
1592       append_entry(scratch_cp, scratch_i, merge_cp_p, merge_cp_length_p,
1593         CHECK_false);
1594     }
1595   }
1596 
1597   log_debug(redefine, class, constantpool)
1598     ("after pass 1a: merge_cp_len=%d, scratch_i=%d, index_map_len=%d",
1599      *merge_cp_length_p, scratch_i, _index_map_count);
1600 
1601   if (scratch_i < scratch_cp->length()) {
1602     // Pass 1b:
1603     // old_cp is smaller than scratch_cp so there are entries in
1604     // scratch_cp that we have not yet processed. We take care of
1605     // those now.
1606     int increment = 1;
1607     for (; scratch_i < scratch_cp->length(); scratch_i += increment) {
1608       switch (scratch_cp->tag_at(scratch_i).value()) {
1609       case JVM_CONSTANT_Double:
1610       case JVM_CONSTANT_Long:
1611         // double and long take two constant pool entries
1612         increment = 2;
1613         break;
1614 
1615       default:
1616         increment = 1;
1617         break;
1618       }
1619 
1620       int found_i =
1621         scratch_cp->find_matching_entry(scratch_i, *merge_cp_p, CHECK_false);
1622       if (found_i != 0) {
1623         // Found a matching entry somewhere else in *merge_cp_p so
1624         // just need a mapping entry.
1625         map_index(scratch_cp, scratch_i, found_i);
1626         continue;
1627       }
1628 
1629       // No match found so we have to append this entry and any unique
1630       // referenced entries to *merge_cp_p.
1631       append_entry(scratch_cp, scratch_i, merge_cp_p, merge_cp_length_p,
1632         CHECK_false);
1633     }
1634 
1635     log_debug(redefine, class, constantpool)
1636       ("after pass 1b: merge_cp_len=%d, scratch_i=%d, index_map_len=%d",
1637        *merge_cp_length_p, scratch_i, _index_map_count);
1638   }
1639   finalize_operands_merge(*merge_cp_p, THREAD);
1640 
1641   return true;
1642 } // end merge_constant_pools()

1662 // Merge constant pools between the_class and scratch_class and
1663 // potentially rewrite bytecodes in scratch_class to use the merged
1664 // constant pool.
1665 jvmtiError VM_RedefineClasses::merge_cp_and_rewrite(
1666              InstanceKlass* the_class, InstanceKlass* scratch_class,
1667              TRAPS) {
1668   // worst case merged constant pool length is old and new combined
1669   int merge_cp_length = the_class->constants()->length()
1670         + scratch_class->constants()->length();
1671 
1672   // Constant pools are not easily reused so we allocate a new one
1673   // each time.
1674   // merge_cp is created unsafe for concurrent GC processing.  It
1675   // should be marked safe before discarding it. Even though
1676   // garbage,  if it crosses a card boundary, it may be scanned
1677   // in order to find the start of the first complete object on the card.
1678   ClassLoaderData* loader_data = the_class->class_loader_data();
1679   ConstantPool* merge_cp_oop =
1680     ConstantPool::allocate(loader_data,
1681                            merge_cp_length,
1682                            CHECK_(JVMTI_ERROR_OUT_OF_MEMORY));
1683   MergeCPCleaner cp_cleaner(loader_data, merge_cp_oop);
1684 
1685   HandleMark hm(THREAD);  // make sure handles are cleared before
1686                           // MergeCPCleaner clears out merge_cp_oop
1687   constantPoolHandle merge_cp(THREAD, merge_cp_oop);
1688 
1689   // Get constants() from the old class because it could have been rewritten
1690   // while we were at a safepoint allocating a new constant pool.
1691   constantPoolHandle old_cp(THREAD, the_class->constants());
1692   constantPoolHandle scratch_cp(THREAD, scratch_class->constants());
1693 
1694   // If the length changed, the class was redefined out from under us. Return
1695   // an error.
1696   if (merge_cp_length != the_class->constants()->length()
1697          + scratch_class->constants()->length()) {
1698     return JVMTI_ERROR_INTERNAL;
1699   }
1700 
1701   // Update the version number of the constant pools (may keep scratch_cp)
1702   merge_cp->increment_and_save_version(old_cp->version());
1703   scratch_cp->increment_and_save_version(old_cp->version());
1704 
1705   ResourceMark rm(THREAD);
1706   _index_map_count = 0;
1707   _index_map_p = new intArray(scratch_cp->length(), scratch_cp->length(), -1);
1708 
1709   _operands_cur_length = ConstantPool::operand_array_length(old_cp->operands());
1710   _operands_index_map_count = 0;
1711   int operands_index_map_len = ConstantPool::operand_array_length(scratch_cp->operands());
1712   _operands_index_map_p = new intArray(operands_index_map_len, operands_index_map_len, -1);
1713 
1714   // reference to the cp holder is needed for copy_operands()
1715   merge_cp->set_pool_holder(scratch_class);
1716   bool result = merge_constant_pools(old_cp, scratch_cp, &merge_cp,
1717                   &merge_cp_length, THREAD);
1718   merge_cp->set_pool_holder(NULL);
1719 
1720   if (!result) {
1721     // The merge can fail due to memory allocation failure or due
1722     // to robustness checks.
1723     return JVMTI_ERROR_INTERNAL;
1724   }
1725 
1726   // Save fields from the old_cp.
1727   merge_cp->copy_fields(old_cp());
1728   scratch_cp->copy_fields(old_cp());
1729 
1730   log_info(redefine, class, constantpool)("merge_cp_len=%d, index_map_len=%d", merge_cp_length, _index_map_count);
1731 
1732   if (_index_map_count == 0) {
1733     // there is nothing to map between the new and merged constant pools
1734 
1735     if (old_cp->length() == scratch_cp->length()) {
1736       // The old and new constant pools are the same length and the
1737       // index map is empty. This means that the three constant pools
1738       // are equivalent (but not the same). Unfortunately, the new
1739       // constant pool has not gone through link resolution nor have
1740       // the new class bytecodes gone through constant pool cache
1741       // rewriting so we can't use the old constant pool with the new
1742       // class.
1743 
1744       // toss the merged constant pool at return
1745     } else if (old_cp->length() < scratch_cp->length()) {
1746       // The old constant pool has fewer entries than the new constant
1747       // pool and the index map is empty. This means the new constant
1748       // pool is a superset of the old constant pool. However, the old
1749       // class bytecodes have already gone through constant pool cache
1750       // rewriting so we can't use the new constant pool with the old
1751       // class.
1752 
1753       // toss the merged constant pool at return
1754     } else {
1755       // The old constant pool has more entries than the new constant
1756       // pool and the index map is empty. This means that both the old
1757       // and merged constant pools are supersets of the new constant
1758       // pool.
1759 
1760       // Replace the new constant pool with a shrunken copy of the
1761       // merged constant pool
1762       set_new_constant_pool(loader_data, scratch_class, merge_cp, merge_cp_length,
1763                             CHECK_(JVMTI_ERROR_OUT_OF_MEMORY));
1764       // The new constant pool replaces scratch_cp so have cleaner clean it up.
1765       // It can't be cleaned up while there are handles to it.
1766       cp_cleaner.add_scratch_cp(scratch_cp());
1767     }
1768   } else {
1769     if (log_is_enabled(Trace, redefine, class, constantpool)) {
1770       // don't want to loop unless we are tracing
1771       int count = 0;
1772       for (int i = 1; i < _index_map_p->length(); i++) {
1773         int value = _index_map_p->at(i);
1774 
1775         if (value != -1) {
1776           log_trace(redefine, class, constantpool)("index_map[%d]: old=%d new=%d", count, i, value);
1777           count++;
1778         }
1779       }
1780     }
1781 
1782     // We have entries mapped between the new and merged constant pools
1783     // so we have to rewrite some constant pool references.
1784     if (!rewrite_cp_refs(scratch_class, THREAD)) {
1785       return JVMTI_ERROR_INTERNAL;
1786     }
1787 
1788     // Replace the new constant pool with a shrunken copy of the
1789     // merged constant pool so now the rewritten bytecodes have
1790     // valid references; the previous new constant pool will get
1791     // GCed.
1792     set_new_constant_pool(loader_data, scratch_class, merge_cp, merge_cp_length,
1793                           CHECK_(JVMTI_ERROR_OUT_OF_MEMORY));
1794     // The new constant pool replaces scratch_cp so have cleaner clean it up.
1795     // It can't be cleaned up while there are handles to it.
1796     cp_cleaner.add_scratch_cp(scratch_cp());
1797   }
1798 
1799   return JVMTI_ERROR_NONE;
1800 } // end merge_cp_and_rewrite()

1978        InstanceKlass* scratch_class, TRAPS) {
1979 
1980   Array<Method*>* methods = scratch_class->methods();
1981

1983     // no methods so nothing to do
1984     return true;
1985   }

1988   for (int i = methods->length() - 1; i >= 0; i--) {

2001       // Need to clear pending exception here as the super caller sets
2002       // the JVMTI_ERROR_INTERNAL if the returned value is false.
2003       CLEAR_PENDING_EXCEPTION;
2004       return false;
2005     }
2006   }
2007 
2008   return true;

2012 // Rewrite constant pool references in the specific method. This code
2013 // was adapted from Rewriter::rewrite_method().
2014 void VM_RedefineClasses::rewrite_cp_refs_in_method(methodHandle method,
2015        methodHandle *new_method_p, TRAPS) {
2016 
2017   *new_method_p = methodHandle();  // default is no new method
2018 
2019   // We cache a pointer to the bytecodes here in code_base. If GC
2020   // moves the Method*, then the bytecodes will also move which
2021   // will likely cause a crash. We create a NoSafepointVerifier
2022   // object to detect whether we pass a possible safepoint in this
2023   // code block.
2024   NoSafepointVerifier nsv;
2025 
2026   // Bytecodes and their length
2027   address code_base = method->code_base();
2028   int code_length = method->code_size();
2029 
2030   int bc_length;
2031   for (int bci = 0; bci < code_length; bci += bc_length) {
2032     address bcp = code_base + bci;
2033     Bytecodes::Code c = (Bytecodes::Code)(*bcp);
2034 
2035     bc_length = Bytecodes::length_for(c);
2036     if (bc_length == 0) {
2037       // More complicated bytecodes report a length of zero so
2038       // we have to try again a slightly different way.
2039       bc_length = Bytecodes::length_at(method(), bcp);
2040     }
2041 
2042     assert(bc_length != 0, "impossible bytecode length");
2043 
2044     switch (c) {
2045       case Bytecodes::_ldc:
2046       {
2047         int cp_index = *(bcp + 1);
2048         int new_index = find_new_index(cp_index);
2049 
2050         if (StressLdcRewrite && new_index == 0) {
2051           // If we are stressing ldc -> ldc_w rewriting, then we
2052           // always need a new_index value.
2053           new_index = cp_index;
2054         }
2055         if (new_index != 0) {
2056           // the original index is mapped so we have more work to do
2057           if (!StressLdcRewrite && new_index <= max_jubyte) {
2058             // The new value can still use ldc instead of ldc_w
2059             // unless we are trying to stress ldc -> ldc_w rewriting
2060             log_trace(redefine, class, constantpool)
2061               ("%s@" INTPTR_FORMAT " old=%d, new=%d", Bytecodes::name(c), p2i(bcp), cp_index, new_index);
2062             *(bcp + 1) = new_index;
2063           } else {
2064             log_trace(redefine, class, constantpool)
2065               ("%s->ldc_w@" INTPTR_FORMAT " old=%d, new=%d", Bytecodes::name(c), p2i(bcp), cp_index, new_index);
2066             // the new value needs ldc_w instead of ldc
2067             u_char inst_buffer[4]; // max instruction size is 4 bytes
2068             bcp = (address)inst_buffer;
2069             // construct new instruction sequence
2070             *bcp = Bytecodes::_ldc_w;
2071             bcp++;
2072             // Rewriter::rewrite_method() does not rewrite ldc -> ldc_w.
2073             // See comment below for difference between put_Java_u2()
2074             // and put_native_u2().
2075             Bytes::put_Java_u2(bcp, new_index);
2076 
2077             Relocator rc(method, NULL /* no RelocatorListener needed */);
2078             methodHandle m;
2079             {
2080               PauseNoSafepointVerifier pnsv(&nsv);
2081 
2082               // ldc is 2 bytes and ldc_w is 3 bytes
2083               m = rc.insert_space_at(bci, 3, inst_buffer, CHECK);
2084             }
2085 
2086             // return the new method so that the caller can update
2087             // the containing class
2088             *new_method_p = method = m;
2089             // switch our bytecode processing loop from the old method
2090             // to the new method
2091             code_base = method->code_base();

3140 // Rewrite constant pool references in the method's stackmap table.
3141 // These "structures" are adapted from the StackMapTable_attribute that
3142 // is described in section 4.8.4 of the 6.0 version of the VM spec
3143 // (dated 2005.10.26):
3144 // file:///net/quincunx.sfbay/export/gbracha/ClassFile-Java6.pdf
3145 //
3146 // stack_map {
3147 //   u2 number_of_entries;
3148 //   stack_map_frame entries[number_of_entries];
3149 // }
3150 //
3151 void VM_RedefineClasses::rewrite_cp_refs_in_stack_map_table(
3152        const methodHandle& method, TRAPS) {
3153 
3154   if (!method->has_stackmap_table()) {
3155     return;
3156   }
3157 
3158   AnnotationArray* stackmap_data = method->stackmap_data();
3159   address stackmap_p = (address)stackmap_data->adr_at(0);
3160   address stackmap_end = stackmap_p + stackmap_data->length();
3161 
3162   assert(stackmap_p + 2 <= stackmap_end, "no room for number_of_entries");
3163   u2 number_of_entries = Bytes::get_Java_u2(stackmap_p);
3164   stackmap_p += 2;
3165 
3166   log_debug(redefine, class, stackmap)("number_of_entries=%u", number_of_entries);
3167 
3168   // walk through each stack_map_frame
3169   u2 calc_number_of_entries = 0;
3170   for (; calc_number_of_entries < number_of_entries; calc_number_of_entries++) {
3171     // The stack_map_frame structure is a u1 frame_type followed by
3172     // 0 or more bytes of data:
3173     //
3174     // union stack_map_frame {
3175     //   same_frame;
3176     //   same_locals_1_stack_item_frame;
3177     //   same_locals_1_stack_item_frame_extended;
3178     //   chop_frame;
3179     //   same_frame_extended;
3180     //   append_frame;
3181     //   full_frame;
3182     // }
3183 
3184     assert(stackmap_p + 1 <= stackmap_end, "no room for frame_type");
3185     u1 frame_type = *stackmap_p;
3186     stackmap_p++;
3187 
3188     // same_frame {
3189     //   u1 frame_type = SAME; /* 0-63 */
3190     // }
3191     if (frame_type <= 63) {
3192       // nothing more to do for same_frame
3193     }
3194 
3195     // same_locals_1_stack_item_frame {
3196     //   u1 frame_type = SAME_LOCALS_1_STACK_ITEM; /* 64-127 */
3197     //   verification_type_info stack[1];
3198     // }
3199     else if (frame_type >= 64 && frame_type <= 127) {
3200       rewrite_cp_refs_in_verification_type_info(stackmap_p, stackmap_end,
3201         calc_number_of_entries, frame_type, THREAD);
3202     }
3203 
3204     // reserved for future use
3205     else if (frame_type >= 128 && frame_type <= 246) {
3206       // nothing more to do for reserved frame_types
3207     }
3208 
3209     // same_locals_1_stack_item_frame_extended {
3210     //   u1 frame_type = SAME_LOCALS_1_STACK_ITEM_EXTENDED; /* 247 */
3211     //   u2 offset_delta;
3212     //   verification_type_info stack[1];
3213     // }
3214     else if (frame_type == 247) {
3215       stackmap_p += 2;
3216       rewrite_cp_refs_in_verification_type_info(stackmap_p, stackmap_end,
3217         calc_number_of_entries, frame_type, THREAD);
3218     }
3219 
3220     // chop_frame {
3221     //   u1 frame_type = CHOP; /* 248-250 */
3222     //   u2 offset_delta;
3223     // }
3224     else if (frame_type >= 248 && frame_type <= 250) {
3225       stackmap_p += 2;
3226     }
3227 
3228     // same_frame_extended {
3229     //   u1 frame_type = SAME_FRAME_EXTENDED; /* 251*/
3230     //   u2 offset_delta;
3231     // }
3232     else if (frame_type == 251) {
3233       stackmap_p += 2;
3234     }
3235 
3236     // append_frame {
3237     //   u1 frame_type = APPEND; /* 252-254 */
3238     //   u2 offset_delta;
3239     //   verification_type_info locals[frame_type - 251];
3240     // }
3241     else if (frame_type >= 252 && frame_type <= 254) {
3242       assert(stackmap_p + 2 <= stackmap_end,
3243         "no room for offset_delta");
3244       stackmap_p += 2;
3245       u1 len = frame_type - 251;
3246       for (u1 i = 0; i < len; i++) {
3247         rewrite_cp_refs_in_verification_type_info(stackmap_p, stackmap_end,
3248           calc_number_of_entries, frame_type, THREAD);
3249       }
3250     }
3251 
3252     // full_frame {
3253     //   u1 frame_type = FULL_FRAME; /* 255 */
3254     //   u2 offset_delta;
3255     //   u2 number_of_locals;
3256     //   verification_type_info locals[number_of_locals];
3257     //   u2 number_of_stack_items;
3258     //   verification_type_info stack[number_of_stack_items];
3259     // }
3260     else if (frame_type == 255) {
3261       assert(stackmap_p + 2 + 2 <= stackmap_end,
3262         "no room for smallest full_frame");
3263       stackmap_p += 2;
3264 
3265       u2 number_of_locals = Bytes::get_Java_u2(stackmap_p);
3266       stackmap_p += 2;
3267 
3268       for (u2 locals_i = 0; locals_i < number_of_locals; locals_i++) {
3269         rewrite_cp_refs_in_verification_type_info(stackmap_p, stackmap_end,
3270           calc_number_of_entries, frame_type, THREAD);
3271       }
3272 
3273       // Use the largest size for the number_of_stack_items, but only get
3274       // the right number of bytes.
3275       u2 number_of_stack_items = Bytes::get_Java_u2(stackmap_p);

3288 // Rewrite constant pool references in the verification type info
3289 // portion of the method's stackmap table. These "structures" are
3290 // adapted from the StackMapTable_attribute that is described in
3291 // section 4.8.4 of the 6.0 version of the VM spec (dated 2005.10.26):
3292 // file:///net/quincunx.sfbay/export/gbracha/ClassFile-Java6.pdf
3293 //
3294 // The verification_type_info structure is a u1 tag followed by 0 or
3295 // more bytes of data:
3296 //
3297 // union verification_type_info {
3298 //   Top_variable_info;
3299 //   Integer_variable_info;
3300 //   Float_variable_info;
3301 //   Long_variable_info;
3302 //   Double_variable_info;
3303 //   Null_variable_info;
3304 //   UninitializedThis_variable_info;
3305 //   Object_variable_info;
3306 //   Uninitialized_variable_info;
3307 // }
3308 //
3309 void VM_RedefineClasses::rewrite_cp_refs_in_verification_type_info(
3310        address& stackmap_p_ref, address stackmap_end, u2 frame_i,
3311        u1 frame_type, TRAPS) {
3312 
3313   assert(stackmap_p_ref + 1 <= stackmap_end, "no room for tag");
3314   u1 tag = *stackmap_p_ref;
3315   stackmap_p_ref++;
3316 
3317   switch (tag) {
3318   // Top_variable_info {
3319   //   u1 tag = ITEM_Top; /* 0 */
3320   // }
3321   // verificationType.hpp has zero as ITEM_Bogus instead of ITEM_Top
3322   case 0:  // fall through
3323 
3324   // Integer_variable_info {
3325   //   u1 tag = ITEM_Integer; /* 1 */
3326   // }
3327   case ITEM_Integer:  // fall through
3328 
3329   // Float_variable_info {
3330   //   u1 tag = ITEM_Float; /* 2 */
3331   // }
3332   case ITEM_Float:  // fall through
3333 
3334   // Double_variable_info {
3335   //   u1 tag = ITEM_Double; /* 3 */
3336   // }
3337   case ITEM_Double:  // fall through
3338 
3339   // Long_variable_info {
3340   //   u1 tag = ITEM_Long; /* 4 */
3341   // }
3342   case ITEM_Long:  // fall through
3343 
3344   // Null_variable_info {
3345   //   u1 tag = ITEM_Null; /* 5 */
3346   // }
3347   case ITEM_Null:  // fall through
3348 
3349   // UninitializedThis_variable_info {
3350   //   u1 tag = ITEM_UninitializedThis; /* 6 */
3351   // }
3352   case ITEM_UninitializedThis:
3353     // nothing more to do for the above tag types
3354     break;
3355 
3356   // Object_variable_info {
3357   //   u1 tag = ITEM_Object; /* 7 */
3358   //   u2 cpool_index;
3359   // }
3360   case ITEM_Object:
3361   {
3362     assert(stackmap_p_ref + 2 <= stackmap_end, "no room for cpool_index");
3363     u2 cpool_index = Bytes::get_Java_u2(stackmap_p_ref);
3364     u2 new_cp_index = find_new_index(cpool_index);
3365     if (new_cp_index != 0) {
3366       log_debug(redefine, class, stackmap)("mapped old cpool_index=%d", cpool_index);
3367       Bytes::put_Java_u2(stackmap_p_ref, new_cp_index);
3368       cpool_index = new_cp_index;
3369     }
3370     stackmap_p_ref += 2;
3371 
3372     log_debug(redefine, class, stackmap)
3373       ("frame_i=%u, frame_type=%u, cpool_index=%d", frame_i, frame_type, cpool_index);
3374   } break;
3375 
3376   // Uninitialized_variable_info {
3377   //   u1 tag = ITEM_Uninitialized; /* 8 */
3378   //   u2 offset;
3379   // }
3380   case ITEM_Uninitialized:
3381     assert(stackmap_p_ref + 2 <= stackmap_end, "no room for offset");
3382     stackmap_p_ref += 2;
3383     break;
3384 
3385   default:
3386     log_debug(redefine, class, stackmap)("frame_i=%u, frame_type=%u, bad tag=0x%x", frame_i, frame_type, tag);

3388     break;
3389   } // end switch (tag)

3684 int VM_RedefineClasses::check_methods_and_mark_as_obsolete() {
3685   int emcp_method_count = 0;
3686   int obsolete_count = 0;
3687   int old_index = 0;
3688   for (int j = 0; j < _matching_methods_length; ++j, ++old_index) {
3689     Method* old_method = _matching_old_methods[j];
3690     Method* new_method = _matching_new_methods[j];
3691     Method* old_array_method;
3692 
3693     // Maintain an old_index into the _old_methods array by skipping
3694     // deleted methods
3695     while ((old_array_method = _old_methods->at(old_index)) != old_method) {
3696       ++old_index;
3697     }
3698 
3699     if (MethodComparator::methods_EMCP(old_method, new_method)) {
3700       // The EMCP definition from JSR-163 requires the bytecodes to be
3701       // the same with the exception of constant pool indices which may
3702       // differ. However, the constants referred to by those indices
3703       // must be the same.
3704       //
3705       // We use methods_EMCP() for comparison since constant pool
3706       // merging can remove duplicate constant pool entries that were
3707       // present in the old method and removed from the rewritten new
3708       // method. A faster binary comparison function would consider the
3709       // old and new methods to be different when they are actually
3710       // EMCP.
3711       //
3712       // The old and new methods are EMCP and you would think that we
3713       // could get rid of one of them here and now and save some space.
3714       // However, the concept of EMCP only considers the bytecodes and
3715       // the constant pool entries in the comparison. Other things,
3716       // e.g., the line number table (LNT) or the local variable table
3717       // (LVT) don't count in the comparison. So the new (and EMCP)
3718       // method can have a new LNT that we need so we can't just
3719       // overwrite the new method with the old method.
3720       //
3721       // When this routine is called, we have already attached the new
3722       // methods to the_class so the old methods are effectively
3723       // overwritten. However, if an old method is still executing,
3724       // then the old method cannot be collected until sometime after
3725       // the old method call has returned. So the overwriting of old
3726       // methods by new methods will save us space except for those
3727       // (hopefully few) old methods that are still executing.
3728       //
3729       // A method refers to a ConstMethod* and this presents another
3730       // possible avenue to space savings. The ConstMethod* in the
3731       // new method contains possibly new attributes (LNT, LVT, etc).
3732       // At first glance, it seems possible to save space by replacing
3733       // the ConstMethod* in the old method with the ConstMethod*
3734       // from the new method. The old and new methods would share the
3735       // same ConstMethod* and we would save the space occupied by
3736       // the old ConstMethod*. However, the ConstMethod* contains
3737       // a back reference to the containing method. Sharing the
3738       // ConstMethod* between two methods could lead to confusion in
3739       // the code that uses the back reference. This would lead to
3740       // brittle code that could be broken in non-obvious ways now or
3741       // in the future.
3742       //
3743       // Another possibility is to copy the ConstMethod* from the new
3744       // method to the old method and then overwrite the new method with
3745       // the old method. Since the ConstMethod* contains the bytecodes
3746       // for the method embedded in the oop, this option would change
3747       // the bytecodes out from under any threads executing the old
3748       // method and make the thread's bcp invalid. Since EMCP requires
3749       // that the bytecodes be the same modulo constant pool indices, it
3750       // is straight forward to compute the correct new bcp in the new
3751       // ConstMethod* from the old bcp in the old ConstMethod*. The
3752       // time consuming part would be searching all the frames in all
3753       // of the threads to find all of the calls to the old method.
3754       //
3755       // It looks like we will have to live with the limited savings
3756       // that we get from effectively overwriting the old methods
3757       // when the new methods are attached to the_class.
3758 
3759       // Count number of methods that are EMCP.  The method will be marked
3760       // old but not obsolete if it is EMCP.
3761       emcp_method_count++;
3762 
3763       // An EMCP method is _not_ obsolete. An obsolete method has a
3764       // different jmethodID than the current method. An EMCP method
3765       // has the same jmethodID as the current method. Having the
3766       // same jmethodID for all EMCP versions of a method allows for
3767       // a consistent view of the EMCP methods regardless of which
3768       // EMCP method you happen to have in hand. For example, a
3769       // breakpoint set in one EMCP method will work for all EMCP
3770       // versions of the method including the current one.
3771     } else {
3772       // mark obsolete methods as such
3773       old_method->set_is_obsolete();
3774       obsolete_count++;
3775 
3776       // obsolete methods need a unique idnum so they become new entries in
3777       // the jmethodID cache in InstanceKlass
3778       assert(old_method->method_idnum() == new_method->method_idnum(), "must match");
3779       u2 num = InstanceKlass::cast(_the_class)->next_method_idnum();
3780       if (num != ConstMethod::UNSET_IDNUM) {
3781         old_method->set_method_idnum(num);
3782       }
3783 
3784       // With tracing we try not to "yack" too much. The position of
3785       // this trace assumes there are fewer obsolete methods than
3786       // EMCP methods.

3821 // This internal class transfers the native function registration from old methods
3822 // to new methods.  It is designed to handle both the simple case of unchanged
3823 // native methods and the complex cases of native method prefixes being added and/or
3824 // removed.
3825 // It expects only to be used during the VM_RedefineClasses op (a safepoint).
3826 //
3827 // This class is used after the new methods have been installed in "the_class".
3828 //
3829 // So, for example, the following must be handled.  Where 'm' is a method and
3830 // a number followed by an underscore is a prefix.
3831 //
3832 //                                      Old Name    New Name
3833 // Simple transfer to new method        m       ->  m
3834 // Add prefix                           m       ->  1_m
3835 // Remove prefix                        1_m     ->  m
3836 // Simultaneous add of prefixes         m       ->  3_2_1_m
3837 // Simultaneous removal of prefixes     3_2_1_m ->  m
3838 // Simultaneous add and remove          1_m     ->  2_m
3839 // Same, caused by prefix removal only  3_2_1_m ->  3_2_m
3840 //
3841 class TransferNativeFunctionRegistration {
3842  private:
3843   InstanceKlass* the_class;
3844   int prefix_count;
3845   char** prefixes;
3846 
3847   // Recursively search the binary tree of possibly prefixed method names.
3848   // Iteration could be used if all agents were well behaved. Full tree walk is
3849   // more resilent to agents not cleaning up intermediate methods.
3850   // Branch at each depth in the binary tree is:
3851   //    (1) without the prefix.
3852   //    (2) with the prefix.
3853   // where 'prefix' is the prefix at that 'depth' (first prefix, second prefix,...)
3854   Method* search_prefix_name_space(int depth, char* name_str, size_t name_len,
3855                                      Symbol* signature) {
3856     TempNewSymbol name_symbol = SymbolTable::probe(name_str, (int)name_len);
3857     if (name_symbol != NULL) {
3858       Method* method = the_class->lookup_method(name_symbol, signature);
3859       if (method != NULL) {
3860         // Even if prefixed, intermediate methods must exist.
3861         if (method->is_native()) {
3862           // Wahoo, we found a (possibly prefixed) version of the method, return it.
3863           return method;
3864         }
3865         if (depth < prefix_count) {
3866           // Try applying further prefixes (other than this one).
3867           method = search_prefix_name_space(depth+1, name_str, name_len, signature);
3868           if (method != NULL) {
3869             return method; // found
3870           }
3871 
3872           // Try adding this prefix to the method name and see if it matches
3873           // another method name.
3874           char* prefix = prefixes[depth];
3875           size_t prefix_len = strlen(prefix);
3876           size_t trial_len = name_len + prefix_len;
3877           char* trial_name_str = NEW_RESOURCE_ARRAY(char, trial_len + 1);
3878           strcpy(trial_name_str, prefix);
3879           strcat(trial_name_str, name_str);
3880           method = search_prefix_name_space(depth+1, trial_name_str, trial_len,
3881                                             signature);
3882           if (method != NULL) {
3883             // If found along this branch, it was prefixed, mark as such
3884             method->set_is_prefixed_native();
3885             return method; // found
3886           }
3887         }
3888       }
3889     }
3890     return NULL;  // This whole branch bore nothing
3891   }
3892 
3893   // Return the method name with old prefixes stripped away.
3894   char* method_name_without_prefixes(Method* method) {
3895     Symbol* name = method->name();
3896     char* name_str = name->as_utf8();
3897 
3898     // Old prefixing may be defunct, strip prefixes, if any.
3899     for (int i = prefix_count-1; i >= 0; i--) {
3900       char* prefix = prefixes[i];
3901       size_t prefix_len = strlen(prefix);
3902       if (strncmp(prefix, name_str, prefix_len) == 0) {
3903         name_str += prefix_len;
3904       }
3905     }
3906     return name_str;
3907   }
3908 
3909   // Strip any prefixes off the old native method, then try to find a
3910   // (possibly prefixed) new native that matches it.
3911   Method* strip_and_search_for_new_native(Method* method) {
3912     ResourceMark rm;
3913     char* name_str = method_name_without_prefixes(method);
3914     return search_prefix_name_space(0, name_str, strlen(name_str),
3915                                     method->signature());
3916   }

3928   // Attempt to transfer any of the old or deleted methods that are native
3929   void transfer_registrations(Method** old_methods, int methods_length) {

3931       Method* old_method = old_methods[j];

4006 void VM_RedefineClasses::compute_added_deleted_matching_methods() {
4007   Method* old_method;
4008   Method* new_method;
4009 
4010   _matching_old_methods = NEW_RESOURCE_ARRAY(Method*, _old_methods->length());
4011   _matching_new_methods = NEW_RESOURCE_ARRAY(Method*, _old_methods->length());
4012   _added_methods        = NEW_RESOURCE_ARRAY(Method*, _new_methods->length());
4013   _deleted_methods      = NEW_RESOURCE_ARRAY(Method*, _old_methods->length());
4014 
4015   _matching_methods_length = 0;
4016   _deleted_methods_length  = 0;
4017   _added_methods_length    = 0;
4018 
4019   int nj = 0;
4020   int oj = 0;
4021   while (true) {
4022     if (oj >= _old_methods->length()) {
4023       if (nj >= _new_methods->length()) {
4024         break; // we've looked at everything, done
4025       }
4026       // New method at the end
4027       new_method = _new_methods->at(nj);
4028       _added_methods[_added_methods_length++] = new_method;
4029       ++nj;
4030     } else if (nj >= _new_methods->length()) {
4031       // Old method, at the end, is deleted
4032       old_method = _old_methods->at(oj);
4033       _deleted_methods[_deleted_methods_length++] = old_method;
4034       ++oj;
4035     } else {
4036       old_method = _old_methods->at(oj);
4037       new_method = _new_methods->at(nj);
4038       if (old_method->name() == new_method->name()) {
4039         if (old_method->signature() == new_method->signature()) {
4040           _matching_old_methods[_matching_methods_length  ] = old_method;
4041           _matching_new_methods[_matching_methods_length++] = new_method;
4042           ++nj;
4043           ++oj;
4044         } else {
4045           // added overloaded have already been moved to the end,
4046           // so this is a deleted overloaded method
4047           _deleted_methods[_deleted_methods_length++] = old_method;
4048           ++oj;
4049         }
4050       } else { // names don't match
4051         if (old_method->name()->fast_compare(new_method->name()) > 0) {
4052           // new method
4053           _added_methods[_added_methods_length++] = new_method;
4054           ++nj;
4055         } else {
4056           // deleted method
4057           _deleted_methods[_deleted_methods_length++] = old_method;
4058           ++oj;
4059         }
4060       }
4061     }
4062   }
4063   assert(_matching_methods_length + _deleted_methods_length == _old_methods->length(), "sanity");
4064   assert(_matching_methods_length + _added_methods_length == _new_methods->length(), "sanity");
4065 }

4349 void VM_RedefineClasses::increment_class_counter(InstanceKlass *ik, TRAPS) {

4354

4357     log_debug(redefine, class, subclass)("updated count in subclass=%s to %d", ik->external_name(), new_count);
4358   }
4359 
4360   for (Klass *subk = ik->subklass(); subk != NULL;
4361        subk = subk->next_sibling()) {

4364       InstanceKlass *subik = InstanceKlass::cast(subk);

4371 void VM_RedefineClasses::CheckClass::do_klass(Klass* k) {
4372   bool no_old_methods = true;  // be optimistic
4373 
4374   // Both array and instance classes have vtables.
4375   // a vtable should never contain old or obsolete methods
4376   ResourceMark rm(_thread);
4377   if (k->vtable_length() > 0 &&
4378       !k->vtable().check_no_old_or_obsolete_entries()) {
4379     if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
4380       log_trace(redefine, class, obsolete, metadata)
4381         ("klassVtable::check_no_old_or_obsolete_entries failure -- OLD or OBSOLETE method found -- class: %s",
4382          k->signature_name());
4383       k->vtable().dump_vtable();
4384     }
4385     no_old_methods = false;
4386   }
4387 
4388   if (k->is_instance_klass()) {
4389     HandleMark hm(_thread);
4390     InstanceKlass *ik = InstanceKlass::cast(k);
4391 
4392     // an itable should never contain old or obsolete methods
4393     if (ik->itable_length() > 0 &&
4394         !ik->itable().check_no_old_or_obsolete_entries()) {
4395       if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
4396         log_trace(redefine, class, obsolete, metadata)
4397           ("klassItable::check_no_old_or_obsolete_entries failure -- OLD or OBSOLETE method found -- class: %s",
4398            ik->signature_name());
4399         ik->itable().dump_itable();
4400       }
4401       no_old_methods = false;
4402     }
4403 
4404     // the constant pool cache should never contain non-deleted old or obsolete methods
4405     if (ik->constants() != NULL &&
4406         ik->constants()->cache() != NULL &&
4407         !ik->constants()->cache()->check_no_old_or_obsolete_entries()) {
4408       if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
4409         log_trace(redefine, class, obsolete, metadata)
4410           ("cp-cache::check_no_old_or_obsolete_entries failure -- OLD or OBSOLETE method found -- class: %s",
4411            ik->signature_name());
4412         ik->constants()->cache()->dump_cache();
4413       }
4414       no_old_methods = false;
4415     }
4416   }
4417 
4418   // print and fail guarantee if old methods are found.
4419   if (!no_old_methods) {
4420     if (log_is_enabled(Trace, redefine, class, obsolete, metadata)) {
4421       dump_methods();
4422     } else {
4423       log_trace(redefine, class)("Use the '-Xlog:redefine+class*:' option "
4424         "to see more info about the following guarantee() failure.");
4425     }
4426     guarantee(false, "OLD and/or OBSOLETE method(s) found");
4427   }
4428 }
