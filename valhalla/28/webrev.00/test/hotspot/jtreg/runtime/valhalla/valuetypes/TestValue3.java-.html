25 final class ContainerValue3 {
26     static TestValue3? staticValueField;
27     TestValue3 nonStaticValueField;
28     TestValue3[] valueArray;
29 }
30 
31 public inline final class TestValue3 {
32 
33     static TestValue3? staticValue = getInstance();
34 
35     final byte b;
36 
37     private TestValue3() {
38         b = 123;
39     }
40 
41     public static TestValue3 create(byte b) {
42         TestValue3 v = TestValue3.default;
43         v = __WithField(v.b, b);
44         return v;
45     }
46 
47     public static TestValue3 create() {
48         TestValue3 v = TestValue3.default;
49         v = __WithField(v.b, 123);
50         return v;
51     }
52 
53     public static TestValue3 getInstance() {
54         return create();
55     }
56 
57     public static TestValue3 getNonBufferedInstance() {
58         return (TestValue3) staticValue;
59     }
60 
61     public boolean verify() {
62         return b == 0 || b == 123;
63     }
64 }
