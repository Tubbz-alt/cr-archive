27  * @test
28  * @bug 8247567
29  * @summary Javac chokes on static member selection via the reference projection.
30  * @run main StaticSelectedThroughProjection
31  */
32 
33 public class StaticSelectedThroughProjection {
34     static inline class MyValue {
35         int x = 42;
36         static String test() {
37             return "OK";
38         };
39     }
40     public static void main(String[] args) {
41         if (!MyValue.ref.test().equals("OK"))
42             throw new AssertionError("Broken");
43     }
44 }
