 75 /** A class that defines all predefined constants and operators
 76  *  as well as special classes such as java.lang.Object, which need
 77  *  to be known to the compiler. All symbols are held in instance
 78  *  fields. This makes it possible to work in multiple concurrent
 79  *  projects, which might use different class files for library classes.
 80  *
 81  *  <p><b>This is NOT part of any supported API.
 82  *  If you write code that depends on this, you do so at your own risk.
 83  *  This code and its internal interfaces are subject to change or
 84  *  deletion without notice.</b>
 85  */

 90     /** Get the symbol table instance. */
 91     public static Symtab instance(Context context) {
 92         Symtab instance = context.get(symtabKey);
 93         if (instance == null)
 94             instance = new Symtab(context);
 95         return instance;
 96     }

265     public VarSymbol getClassField(Type type, Types types) {
266         return classFields.computeIfAbsent(
267             new UniqueType(type, types), k -> {
268                 Type arg = null;
269                 if (type.getTag() == ARRAY || type.getTag() == CLASS) {
270                     /* Temporary treatment for inline class: Given an inline class V that implements
271                        I1, I2, ... In, V.class is typed to be Class<? extends Object & I1 & I2 .. & In>
272                     */
273                     if (type.isValue()) {
274                         ClassType ct = (ClassType) type;
275                         Type it;
276                         if (ct.interfaces_field == null || ct.interfaces_field.isEmpty()) {
277                             it = objectType;
278                         } else {
279                             it = types.makeIntersectionType(ct.interfaces_field, true);
280                         }
281                         arg = new WildcardType(it, BoundKind.EXTENDS, boundClass);
282                     } else {
283                         arg = types.erasure(type);
284                     }
285                 }
286                 else if (type.isPrimitiveOrVoid())
287                     arg = types.boxedClass(type).type;
288                 else
289                     throw new AssertionError(type);
290 
291                 Type t = new ClassType(
292                     classType.getEnclosingType(), List.of(arg), classType.tsym);
293                 return new VarSymbol(
294                     STATIC | PUBLIC | FINAL, names._class, t, type.tsym);
295             });
296     }

326     public void synthesizeEmptyInterfaceIfMissing(final Type type) {
327         final Completer completer = type.tsym.completer;
328         type.tsym.completer = new Completer() {
329             @Override
330             public void complete(Symbol sym) throws CompletionFailure {
331                 try {
332                     completer.complete(sym);
333                 } catch (CompletionFailure e) {
334                     sym.flags_field |= (PUBLIC | INTERFACE);
335                     ((ClassType) sym.type).supertype_field = objectType;
336                 }
337             }
338 
339             @Override
340             public boolean isTerminal() {
341                 return completer.isTerminal();
342             }
343         };
344     }
345 
346     public void synthesizeBoxTypeIfMissing(final Type type) {
347         ClassSymbol sym = enterClass(java_base, boxedName[type.getTag().ordinal()]);
348         final Completer completer = sym.completer;
349         sym.completer = new Completer() {
350             @Override
351             public void complete(Symbol sym) throws CompletionFailure {
352                 try {
353                     completer.complete(sym);
354                 } catch (CompletionFailure e) {
355                     sym.flags_field |= PUBLIC;
356                     ((ClassType) sym.type).supertype_field = objectType;
357                     MethodSymbol boxMethod =
358                         new MethodSymbol(PUBLIC | STATIC, names.valueOf,
359                                          new MethodType(List.of(type), sym.type,
360                                 List.nil(), methodClass),
361                             sym);
362                     sym.members().enter(boxMethod);
363                     MethodSymbol unboxMethod =
364                         new MethodSymbol(PUBLIC,
365                             type.tsym.name.append(names.Value), // x.intValue()
366                             new MethodType(List.nil(), type,
367                                 List.nil(), methodClass),
368                             sym);
369                     sym.members().enter(unboxMethod);
370                 }
371             }
372 
373             @Override
374             public boolean isTerminal() {
375                 return completer.isTerminal();
376             }
377         };
378     }

380     // Enter a synthetic class that is used to mark classes in ct.sym.
381     // This class does not have a class file.
382     private Type enterSyntheticAnnotation(String name) {
383         // for now, leave the module null, to prevent problems from synthesizing the
384         // existence of a class in any specific module, including noModule
385         ClassType type = (ClassType)enterClass(java_base, names.fromString(name)).type;
386         ClassSymbol sym = (ClassSymbol)type.tsym;
387         sym.completer = Completer.NULL_COMPLETER;
388         sym.flags_field = PUBLIC|ACYCLIC|ANNOTATION|INTERFACE;
389         sym.erasure_field = type;
390         sym.members_field = WriteableScope.create(sym);
391         type.typarams_field = List.nil();
392         type.allparams_field = List.nil();
393         type.supertype_field = annotationType;
394         type.interfaces_field = List.nil();
395         return type;
396     }

402         context.put(symtabKey, this);
403 
404         names = Names.instance(context);
405 
406         // Create the unknown type
407         unknownType = new UnknownType();

411         MissingInfoHandler missingInfoHandler = MissingInfoHandler.instance(context);
412 
413         rootPackage = new RootPackageSymbol(names.empty, null, missingInfoHandler);
414 
415         // create the basic builtin symbols
416         unnamedModule = new ModuleSymbol(names.empty, null) {
417                 {
418                     directives = List.nil();
419                     exports = List.nil();
420                     provides = List.nil();
421                     uses = List.nil();

423                     com.sun.tools.javac.code.Directive.RequiresDirective d =
424                             new com.sun.tools.javac.code.Directive.RequiresDirective(java_base,
425                                     EnumSet.of(com.sun.tools.javac.code.Directive.RequiresFlag.MANDATED));
426                     requires = List.of(d);
427                 }
428                 @Override
429                 public String toString() {
430                     return messages.getLocalizedString("compiler.misc.unnamed.module");
431                 }
432             };
433         addRootPackageFor(unnamedModule);
434         unnamedModule.enclosedPackages = unnamedModule.enclosedPackages.prepend(unnamedModule.unnamedPackage);
435 
436         errModule = new ModuleSymbol(names.empty, null) {
437                 {
438                     directives = List.nil();

465         // create the error symbols
466         errSymbol = new ClassSymbol(PUBLIC|STATIC|ACYCLIC, names.any, null, rootPackage);

475         initType(shortType, "short", "Short");
476         initType(charType, "char", "Character");
477         initType(intType, "int", "Integer");
478         initType(longType, "long", "Long");
479         initType(floatType, "float", "Float");
480         initType(doubleType, "double", "Double");
481         initType(booleanType, "boolean", "Boolean");
482         initType(voidType, "void", "Void");
483         initType(botType, "<nulltype>");
484         initType(errType, errSymbol);
485         initType(unknownType, unknownSymbol);
486 
487         // the builtin class of all arrays
488         arrayClass = new ClassSymbol(PUBLIC|ACYCLIC, names.Array, noSymbol);
489 
490         // VGJ
491         boundClass = new ClassSymbol(PUBLIC|ACYCLIC, names.Bound, noSymbol);
492         boundClass.members_field = new Scope.ErrorScope(boundClass);
493 
494         // the builtin class of all methods
495         methodClass = new ClassSymbol(PUBLIC|ACYCLIC, names.Method, noSymbol);

664     /** Create a new toplevel or member class symbol with given name
665      *  and owner and enter in `classes' unless already there.
666      */
667     public ClassSymbol enterClass(ModuleSymbol msym, Name name, TypeSymbol owner) {
668         Assert.checkNonNull(msym);
669         Name flatname = TypeSymbol.formFlatName(name, owner);
670         ClassSymbol c = getClass(msym, flatname);
671         if (c == null) {
672             c = defineClass(name, owner);
673             doEnterClass(msym, c);
674         } else if ((c.name != name || c.owner != owner) && owner.kind == TYP && c.owner.kind == PCK) {
675             // reassign fields of classes that might have been loaded with
676             // their flat names.
677             c.owner.members().remove(c);
678             c.name = name;
679             c.owner = owner;
680             c.fullname = ClassSymbol.formFullName(name, owner);
681         }
682         return c;
683     }

690     public PackageSymbol lookupPackage(ModuleSymbol msym, Name flatName) {
691         Assert.checkNonNull(msym);
692 
693         if (flatName.isEmpty()) {
694             //unnamed packages only from the current module - visiblePackages contains *root* package, not unnamed package!
695             return msym.unnamedPackage;
696         }
697 
698         if (msym == noModule) {
699             return enterPackage(msym, flatName);
700         }
701 
702         msym.complete();
703 
704         PackageSymbol pack;
705 
706         pack = msym.visiblePackages.get(flatName);
707 
708         if (pack != null)
709             return pack;
710 
711         pack = getPackage(msym, flatName);
712 
713         if (pack != null && pack.exists())
714             return pack;
715 
716         boolean dependsOnUnnamed = msym.requires != null &&
717                                    msym.requires.stream()
718                                                 .map(rd -> rd.module)
719                                                 .anyMatch(mod -> mod == unnamedModule);
720 
721         if (dependsOnUnnamed) {
722             //msyms depends on the unnamed module, for which we generally don't know
723             //the list of packages it "exports" ahead of time. So try to lookup the package in the
724             //current module, and in the unnamed module and see if it exists in one of them
725             PackageSymbol unnamedPack = getPackage(unnamedModule, flatName);
726 
727             if (unnamedPack != null && unnamedPack.exists()) {
728                 msym.visiblePackages.put(unnamedPack.fullname, unnamedPack);
729                 return unnamedPack;
730             }
731 
732             pack = enterPackage(msym, flatName);
733             pack.complete();
734             if (pack.exists())
735                 return pack;
736 
737             unnamedPack = enterPackage(unnamedModule, flatName);
738             unnamedPack.complete();
739             if (unnamedPack.exists()) {
740                 msym.visiblePackages.put(unnamedPack.fullname, unnamedPack);
741                 return unnamedPack;
742             }
743 
744             return pack;
745         }
746 
747         return enterPackage(msym, flatName);
748     }

767     public ClassSymbol enterClass(ModuleSymbol msym, Name flatname) {

788     /** Make a package, given its fully qualified name.
789      */
790     public PackageSymbol enterPackage(ModuleSymbol currModule, Name fullname) {
791         Assert.checkNonNull(currModule);
792         PackageSymbol p = getPackage(currModule, fullname);
793         if (p == null) {
794             Assert.check(!fullname.isEmpty(), () -> "rootPackage missing!; currModule: " + currModule);
795             p = new PackageSymbol(
796                     Convert.shortName(fullname),
797                     enterPackage(currModule, Convert.packagePart(fullname)));
798             p.completer = initialCompleter;
799             p.modle = currModule;
800             doEnterPackage(currModule, p);
801         }
802         return p;
803     }

806         packages.computeIfAbsent(pack.fullname, n -> new HashMap<>()).put(msym, pack);
807         msym.enclosedPackages = msym.enclosedPackages.prepend(pack);
808     }

810     private void addRootPackageFor(ModuleSymbol module) {
811         doEnterPackage(module, rootPackage);
812         PackageSymbol unnamedPackage = new PackageSymbol(names.empty, rootPackage) {
813                 @Override

815                     return messages.getLocalizedString("compiler.misc.unnamed.package");
816                 }
817             };
818         unnamedPackage.modle = module;
819         //we cannot use a method reference below, as initialCompleter might be null now
820         unnamedPackage.completer = s -> initialCompleter.complete(s);
821         unnamedPackage.flags_field |= EXISTS;
822         module.unnamedPackage = unnamedPackage;

865     public List<ModuleSymbol> listPackageModules(Name packageName) {

867             return List.nil();

869         List<ModuleSymbol> result = List.nil();
