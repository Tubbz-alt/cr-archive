26 /*
27  * @test
28  * @bug 8222784
29  * @summary Check that we are able to get a class literal for a reference projection type
30  * @run main RefDotClass
31  */
32 
33 public inline class RefDotClass {
34     final int value;
35 
36     public RefDotClass(int value) {
37         this.value = value;
38     }
39 
40     private static <T> T foo(Class<T> type) {
41         return null;
42     }
43 
44     public static void main(String[] args) {
45         foo(RefDotClass.ref.class);
46         String tName = RefDotClass.ref.class.getTypeName();
47         if (!tName.equals("RefDotClass$ref"))
48             throw new AssertionError("Unexpected type name " + tName);
49     }
50 }
