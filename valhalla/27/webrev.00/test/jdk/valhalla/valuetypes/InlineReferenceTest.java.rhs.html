32 /*
33  * @test

35  * @run testng/othervm InlineReferenceTest

37 @Test
38 public class InlineReferenceTest {
39 
40     @Test(expectedExceptions = IllegalArgumentException.class)
41     static void test1() {
42         Point.ref p = new Point(10,20);

44     }
45 
46     @Test(expectedExceptions = IllegalArgumentException.class)
47     static void test2() {
48         ReferenceQueue<Object> q = new ReferenceQueue<>();

50         WeakReference<Point.ref> r = new WeakReference<>(p, q);
51     }
52 }
