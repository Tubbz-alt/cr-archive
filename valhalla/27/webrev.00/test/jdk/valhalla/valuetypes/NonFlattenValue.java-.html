24 public inline class NonFlattenValue {
25     Point? nfp;
26 
27     NonFlattenValue() {
28         this.nfp = Point.makePoint(0,0);
29     }
30     NonFlattenValue(Point p) {
31         this.nfp = p;
32     }
33     public Point? point() {
34         return nfp;
35     }
36     public Point pointValue() {
37         return (Point) nfp;
38     }
39     public boolean has(Point p1, Point? p2) {
40         return nfp.equals(p1) || nfp.equals(p2);
41     }
42 
43     public static NonFlattenValue make(int x, int y) {
44         return new NonFlattenValue(Point.makePoint(x, y));
45     }
46 
47     @Override
48     public String toString() {
49         // nfp may be null when NonFlattenValue[] is created and filled with default value
50         return nfp != null ? nfp.toString() : "default NonFlattenValue";
51     }
52 }
