28 import java.lang.annotation.*;
29 import static java.lang.annotation.ElementType.*;

31 /**
32  * A class annotated {@code @__inline__} is an inline class.
33  * This is a temporary workaround to enable use of inline types
34  * in editors and IDEs that do not yet understand the 'inline' modifier.
35  * @since 1.12
36  */
37 @Retention(RetentionPolicy.SOURCE)
38 @Target(value={ElementType.TYPE, ElementType.TYPE_USE})
39 public @interface __inline__ {
40 }
