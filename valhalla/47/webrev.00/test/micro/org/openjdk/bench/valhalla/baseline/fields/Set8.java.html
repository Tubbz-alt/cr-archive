23 package org.openjdk.bench.valhalla.baseline.fields;
24 
25 import org.openjdk.jmh.annotations.Benchmark;
26 import org.openjdk.jmh.annotations.Setup;
27 import org.openjdk.bench.valhalla.SizedBase;
28 import org.openjdk.bench.valhalla.baseline.types.Ref8;
29 
30 public class Set8 extends SizedBase {
31 
32     NodePrim8[] primitive;
33     NodeRef8[] reference;
34 
35     @Setup
36     public void setup() {
37         primitive = NodePrim8.set(new NodePrim8[size]);
38         reference = NodeRef8.set(new NodeRef8[size]);
39     }
40 
41     @Benchmark
42     public Object primitive() {
43         NodePrim8[] values = primitive;
44         for (int i = 0, k = 0; i < size; i++, k += 8) {
45             values[i].f0 = k;
46             values[i].f1 = k + 1;
47             values[i].f2 = k + 2;
48             values[i].f3 = k + 3;
49             values[i].f4 = k + 4;
50             values[i].f5 = k + 5;
51             values[i].f6 = k + 6;
52             values[i].f7 = k + 7;
53         }
54         return values;
55     }
56 
57     @Benchmark
58     public Object reference() {
59         NodeRef8[] values = reference;
60         for (int i = 0, k = 0; i < size; i++, k += 8) {
61             values[i].f = new Ref8(k, k + 1, k + 2, k + 3, k + 4, k + 5, k + 6, k + 7);
62         }
63         return values;
64     }
65 
66 }
