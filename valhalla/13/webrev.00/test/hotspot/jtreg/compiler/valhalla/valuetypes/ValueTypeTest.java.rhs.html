228     public static String[] concat(String prefix[], String... extra) {
229         ArrayList<String> list = new ArrayList<String>();
230         if (prefix != null) {
231             for (String s : prefix) {
232                 list.add(s);
233             }
234         }
235         if (extra != null) {
236             for (String s : extra) {
237                 list.add(s);
238             }
239         }
240 
241         return list.toArray(new String[list.size()]);
242     }

328     public static void main(String[] args) throws Throwable {
329         if (args.length != 1) {
330             throw new RuntimeException("Usage: @run main/othervm/timeout=120 -Xbootclasspath/a:." +
331                                        " -XX:+IgnoreUnrecognizedVMOptions -XX:+UnlockDiagnosticVMOptions" +
332                                        " -XX:+UnlockExperimentalVMOptions -XX:+WhiteBoxAPI" +
333                                        " compiler.valhalla.valuetypes.ValueTypeTest <YourTestMainClass>");
334         }
335         String testMainClassName = args[0];
336         Class testMainClass = Class.forName(testMainClassName);
337         ValueTypeTest test = (ValueTypeTest)testMainClass.newInstance();
338         List<String> scenarios = null;
339         if (!SCENARIOS.isEmpty()) {
340            scenarios = Arrays.asList(SCENARIOS.split(","));
341         }
342         for (int i=0; i<test.getNumScenarios(); i++) {
343             String reason;
344             if ((reason = test.isScenarioIgnored(i)) != null) {
345                 System.out.println("Scenario #" + i + " is ignored: " + reason);
346             } else if (scenarios != null && !scenarios.contains(Integer.toString(i))) {
347                 System.out.println("Scenario #" + i + " is skipped due to -Dscenarios=" + SCENARIOS);
348             } else {
349                 System.out.println("Scenario #" + i + " -------- ");
350                 String[] cmds = InputArguments.getVmInputArgs();
351                 if (RUN_SCENARIOS_WITH_XCOMP) {
352                     cmds = concat(cmds, "-Xcomp");
353                 }
354                 cmds = concat(cmds, test.getVMParameters(i));
355                 cmds = concat(cmds, test.getExtraVMParameters(i));
356                 cmds = concat(cmds, testMainClassName);
357 
358                 OutputAnalyzer oa = ProcessTools.executeTestJvm(cmds);
359                 String output = oa.getOutput();
360                 oa.shouldHaveExitValue(0);
361                 System.out.println(output);
362             }
363         }
364     }

366     // To exclude test cases, use -DExclude=<case1>,<case2>,...
367     // Each case can be just the method name, or can be <class>.<method>. The latter form is useful
368     // when you are running several tests at the same time.
369     //
370     // jtreg -DExclude=test12 TestArrays.java
371     // jtreg -DExclude=test34 TestLWorld.java
372     // -- or --
373     // jtreg -DExclude=TestArrays.test12,TestLWorld.test34 TestArrays.java TestLWorld.java
374     //
375     private List<String> buildExcludeList() {
376         List<String> exclude = null;
377         String classPrefix = getClass().getSimpleName() + ".";
378         if (!EXCLUDELIST.isEmpty()) {
379             exclude = new ArrayList(Arrays.asList(EXCLUDELIST.split(",")));
380             for (int i = exclude.size() - 1; i >= 0; i--) {
381                 String ex = exclude.get(i);
382                 if (ex.indexOf(".") > 0) {
383                     if (ex.startsWith(classPrefix)) {
384                         ex = ex.substring(classPrefix.length());
385                         exclude.set(i, ex);
386                     } else {
387                         exclude.remove(i);
388                     }
389                 }
390             }
391         }
392         return exclude;
393     }

395     protected ValueTypeTest() {
396         List<String> list = null;
397         if (!TESTLIST.isEmpty()) {
398            list = Arrays.asList(TESTLIST.split(","));
399         }
400         List<String> exclude = buildExcludeList();
401 
402         // Gather all test methods and put them in Hashtable
403         for (Method m : getClass().getDeclaredMethods()) {
404             Test[] annos = m.getAnnotationsByType(Test.class);
405             if (annos.length != 0 &&
406                 ((list == null || list.contains(m.getName())) && (exclude == null || !exclude.contains(m.getName())))) {
407                 tests.put(getClass().getSimpleName() + "::" + m.getName(), m);
408             }
409         }
410     }

412     protected void run(String[] args, Class<?>... classes) throws Throwable {
413         if (args.length == 0) {
414             // Spawn a new VM instance
415             execute_vm();
416         } else {
417             // Execute tests in the VM spawned by the above code.
418             Asserts.assertTrue(args.length == 1 && args[0].equals("run"), "must be");
419             run(classes);
420         }
421     }

423     private void execute_vm() throws Throwable {
424         Asserts.assertFalse(tests.isEmpty(), "no tests to execute");
425         String[] vmInputArgs = InputArguments.getVmInputArgs();
426         for (String arg : vmInputArgs) {
427             if (arg.startsWith("-XX:CompileThreshold")) {
428                 // Disable IR verification if non-default CompileThreshold is set
429                 VERIFY_IR = false;
430             }
431         }
432         // Each VM is launched with flags in this order, so the later ones can override the earlier one:
433         //     VERIFY_IR/VERIFY_VM flags specified below
434         //     vmInputArgs, which consists of:
435         //        @run options
436         //        getVMParameters()
437         //        getExtraVMParameters()
438         //     defaultFlags
439         //     compileCommandFlags
440         String cmds[] = null;
441         if (VERIFY_IR) {
442             // Add print flags for IR verification
443             cmds = concat(cmds, printFlags);
444             // Always trap for exception throwing to not confuse IR verification
445             cmds = concat(cmds, "-XX:-OmitStackTraceInFastThrow");
446         }
447         if (VERIFY_VM) {
448             cmds = concat(cmds, verifyFlags);
449         }
450         cmds = concat(cmds, vmInputArgs);
451         cmds = concat(cmds, defaultFlags);
452         if (COMPILE_COMMANDS) {
453           cmds = concat(cmds, compileCommandFlags);
454         }
455 
456         // Run tests in own process and verify output
457         cmds = concat(cmds, getClass().getName(), "run");
458         OutputAnalyzer oa = ProcessTools.executeTestJvm(cmds);
459         // If ideal graph printing is enabled/supported, verify output
460         String output = oa.getOutput();
461         oa.shouldHaveExitValue(0);
462         if (VERIFY_IR) {
463             if (output.contains("PrintIdeal enabled")) {
464                 parseOutput(output);
465             } else {
466                 System.out.println(output);
467                 System.out.println("WARNING: IR verification failed! Running with -Xint, -Xcomp or release build?");
468             }
469         }
470     }

472     static final class TestAnnotation {
473         private final int flag;
474         private final BooleanSupplier predicate;
475 
476         private static final TestAnnotation testAnnotations[] = {
477             new TestAnnotation(ValueTypePassFieldsAsArgsOn, () -> ValueTypePassFieldsAsArgs),
478             new TestAnnotation(ValueTypePassFieldsAsArgsOff, () -> !ValueTypePassFieldsAsArgs),
479             new TestAnnotation(ValueTypeArrayFlattenOn, () -> ValueTypeArrayFlatten),
480             new TestAnnotation(ValueTypeArrayFlattenOff, () -> !ValueTypeArrayFlatten),
481             new TestAnnotation(ValueTypeReturnedAsFieldsOn, () -> ValueTypeReturnedAsFields),
482             new TestAnnotation(ValueTypeReturnedAsFieldsOff, () -> !ValueTypeReturnedAsFields),
483             new TestAnnotation(AlwaysIncrementalInlineOn, () -> AlwaysIncrementalInline),
484             new TestAnnotation(AlwaysIncrementalInlineOff, () -> !AlwaysIncrementalInline),
485             new TestAnnotation(G1GCOn, () -> G1GC),
486             new TestAnnotation(G1GCOff, () -> !G1GC),
487             new TestAnnotation(ZGCOn, () -> ZGC),
488             new TestAnnotation(ZGCOff, () -> !ZGC),
489             new TestAnnotation(ArrayLoadStoreProfileOn, () -> UseArrayLoadStoreProfile),
490             new TestAnnotation(ArrayLoadStoreProfileOff, () -> !UseArrayLoadStoreProfile),
491             new TestAnnotation(TypeProfileOn, () -> TypeProfileLevel == 222),
492             new TestAnnotation(TypeProfileOff, () -> TypeProfileLevel == 0),
493         };
494 
495         private TestAnnotation(int flag, BooleanSupplier predicate) {
496             this.flag = flag;
497             this.predicate = predicate;
498         }
499 
500         private boolean match(Test a) {
501             return (a.valid() & flag) != 0 && predicate.getAsBoolean();
502         }
503 
504         static boolean find(Test a) {
505             Stream<TestAnnotation> s = Arrays.stream(testAnnotations).filter(t -> t.match(a));
506             long c = s.count();
507             if (c > 1) {
508                 throw new RuntimeException("At most one Test annotation should match");
509             }
510             return c > 0;
511         }
512     }

514     private void parseOutput(String output) throws Exception {
515         Pattern comp_re = Pattern.compile("\\n\\s+\\d+\\s+\\d+\\s+(%| )(s| )(!| )b(n| )\\s+\\d?\\s+\\S+\\.(?<name>[^.]+::\\S+)\\s+(?<osr>@ \\d+\\s+)?[(]\\d+ bytes[)]");
516         Matcher m = comp_re.matcher(output);
517         Map<String,String> compilations = new LinkedHashMap<>();
518         int prev = 0;
519         String methodName = null;
520         while (m.find()) {
521             if (prev == 0) {
522                 // Print header
523                 System.out.print(output.substring(0, m.start()+1));
524             } else if (methodName != null) {
525                 compilations.put(methodName, output.substring(prev, m.start()+1));
526             }
527             if (m.group("osr") != null) {
528                 methodName = null;
529             } else {
530                 methodName = m.group("name");
531             }
532             prev = m.end();
533         }
534         if (prev == 0) {
535             // Print header
536             System.out.print(output);
537         } else if (methodName != null) {
538             compilations.put(methodName, output.substring(prev));
539         }
540         // Iterate over compilation output
541         for (String testName : compilations.keySet()) {
542             Method test = tests.get(testName);
543             if (test == null) {
544                 // Skip helper methods
545                 continue;
546             }
547             String graph = compilations.get(testName);
548             if (PRINT_GRAPH) {
549                 System.out.println("\nGraph for " + testName + "\n" + graph);
550             }
551             // Parse graph using regular expressions to determine if it contains forbidden nodes
552             Test[] annos = test.getAnnotationsByType(Test.class);
553             Test anno = Arrays.stream(annos).filter(TestAnnotation::find).findFirst().orElse(null);
554             if (anno == null) {
555                 Object[] res = Arrays.stream(annos).filter(a -> a.valid() == 0).toArray();
556                 if (res.length != 1) {
557                     throw new RuntimeException("Only one Test annotation should match");
558                 }
559                 anno = (Test)res[0];
560             }
561             String regexFail = anno.failOn();
562             if (!regexFail.isEmpty()) {
563                 Pattern pattern = Pattern.compile(regexFail.substring(0, regexFail.length()-1));
564                 Matcher matcher = pattern.matcher(graph);
565                 boolean found = matcher.find();
566                 Asserts.assertFalse(found, "Graph for '" + testName + "' contains forbidden node:\n" + (found ? matcher.group() : ""));
567             }
568             String[] regexMatch = anno.match();
569             int[] matchCount = anno.matchCount();
570             for (int i = 0; i < regexMatch.length; ++i) {
571                 Pattern pattern = Pattern.compile(regexMatch[i].substring(0, regexMatch[i].length()-1));
572                 Matcher matcher = pattern.matcher(graph);
573                 int count = 0;
574                 String nodes = "";
575                 while (matcher.find()) {
576                     count++;
577                     nodes += matcher.group() + "\n";
578                 }
579                 if (matchCount[i] < 0) {
580                     Asserts.assertLTE(Math.abs(matchCount[i]), count, "Graph for '" + testName + "' contains different number of match nodes (expected <= " + matchCount[i] + " but got " + count + "):\n" + nodes);
581                 } else {
582                     Asserts.assertEQ(matchCount[i], count, "Graph for '" + testName + "' contains different number of match nodes (expected " + matchCount[i] + " but got " + count + "):\n" + nodes);
583                 }
584             }
585             tests.remove(testName);
586             System.out.println(testName + " passed");
587         }
588         // Check if all tests were compiled
589         if (tests.size() != 0) {
590             for (String name : tests.keySet()) {
591                 System.out.println("Test '" + name + "' not compiled!");
592             }
593             throw new RuntimeException("Not all tests were compiled");
594         }
595     }

597     private void setup(Class<?> clazz) {
598         if (XCOMP) {
599             // Don't control compilation if -Xcomp is enabled
600             return;
601         }
602         if (DUMP_REPLAY) {
603             // Generate replay compilation files
604             String directive = "[{ match: \"*.*\", DumpReplay: true }]";
605             if (WHITE_BOX.addCompilerDirective(directive) != 1) {
606                 throw new RuntimeException("Failed to add compiler directive");
607             }
608         }
609 
610         Method[] methods = clazz.getDeclaredMethods();
611         for (Method m : methods) {
612             if (m.isAnnotationPresent(Test.class)) {
613                 // Don't inline tests
614                 WHITE_BOX.testSetDontInlineMethod(m, true);
615             }
616             if (m.isAnnotationPresent(DontCompile.class)) {
617                 WHITE_BOX.makeMethodNotCompilable(m, COMP_LEVEL_ANY, true);
618                 WHITE_BOX.makeMethodNotCompilable(m, COMP_LEVEL_ANY, false);
619                 WHITE_BOX.testSetDontInlineMethod(m, true);
620             }
621             if (m.isAnnotationPresent(ForceInline.class)) {
622                 Asserts.assertFalse(m.isAnnotationPresent(DontInline.class), "Method " + m.getName() + " has contradicting DontInline annotation");
623                 WHITE_BOX.testSetForceInlineMethod(m, true);
624             }
625             if (m.isAnnotationPresent(DontInline.class)) {
626                 Asserts.assertFalse(m.isAnnotationPresent(ForceInline.class), "Method " + m.getName() + " has contradicting ForceInline annotation");
627                 WHITE_BOX.testSetDontInlineMethod(m, true);
628             }
629             if (STRESS_CC) {
630                 // Exclude some methods from compilation with C2 to stress test the calling convention
631                 if (Utils.getRandomInstance().nextBoolean()) {
632                     System.out.println("Excluding from C2 compilation: " + m);
633                     WHITE_BOX.makeMethodNotCompilable(m, COMP_LEVEL_FULL_OPTIMIZATION, false);
634                 }
635             }
636         }
637         // Only force compilation now because above annotations affect inlining
638         for (Method m : methods) {
639             if (m.isAnnotationPresent(ForceCompile.class)) {
640                 Asserts.assertFalse(m.isAnnotationPresent(DontCompile.class), "Method " + m.getName() + " has contradicting DontCompile annotation");
641                 int compLevel = getCompLevel(m.getAnnotation(ForceCompile.class));
642                 enqueueMethodForCompilation(m, compLevel);
643             }
644         }
645         // Compile class initializers
646         int compLevel = getCompLevel(null);
647         WHITE_BOX.enqueueInitializerForCompilation(clazz, compLevel);
648     }

650     private void run(Class<?>... classes) throws Exception {
651         if (USE_COMPILER && PRINT_IDEAL && !XCOMP && !STRESS_CC) {
652             System.out.println("PrintIdeal enabled");
653         }
654         System.out.format("rI = %d, rL = %d\n", rI, rL);
655 
656         setup(getClass());
657         for (Class<?> clazz : classes) {
658             setup(clazz);
659         }
660 
661         // Execute tests
662         TreeMap<Long, String> durations = (PRINT_TIMES || VERBOSE) ? new TreeMap<Long, String>() : null;
663         for (Method test : tests.values()) {
664             if (VERBOSE) {
665                 System.out.println("Starting " + test.getName());
666             }
667             TempSkipForC1 c1skip = test.getAnnotation(TempSkipForC1.class);
668             if (TEST_C1 && c1skip != null) {
669                 System.out.println("Skipped " + test.getName() + " for C1 testing: " + c1skip.reason());
670                 continue;
671             }
672             long startTime = System.nanoTime();
673             Method verifier = getClass().getMethod(test.getName() + "_verifier", boolean.class);
674             // Warmup using verifier method
675             Warmup anno = test.getAnnotation(Warmup.class);
676             int warmup = anno == null ? WARMUP : anno.value();
677             for (int i = 0; i < warmup; ++i) {
678                 verifier.invoke(this, true);
679             }
680             boolean osrOnly = (test.getAnnotation(OSRCompileOnly.class) != null);
681 
682             // C1 generates a lot of code when VerifyOops is enabled and may run out of space (for a small
683             // number of test cases).
684             boolean maybeCodeBufferOverflow = (TEST_C1 && VerifyOops);
685 
686             if (osrOnly) {
687                 long started = System.currentTimeMillis();
688                 boolean stateCleared = false;
689                 for (;;) {
690                     long elapsed = System.currentTimeMillis() - started;
691                     if (maybeCodeBufferOverflow && elapsed > 5000 && !WHITE_BOX.isMethodCompiled(test, false)) {
692                         System.out.println("Temporarily disabling VerifyOops");
693                         try {
694                             WHITE_BOX.setBooleanVMFlag("VerifyOops", false);
695                             if (!stateCleared) {
696                                 WHITE_BOX.clearMethodState(test);
697                                 stateCleared = true;
698                             }
699                             verifier.invoke(this, false);
700                         } finally {
701                             WHITE_BOX.setBooleanVMFlag("VerifyOops", true);
702                             System.out.println("Re-enabled VerifyOops");
703                         }
704                     } else {
705                         verifier.invoke(this, false);
706                     }
707 
708                     boolean b = WHITE_BOX.isMethodCompiled(test, false);
709                     if (VERBOSE) {
710                         System.out.println("Is " + test.getName() + " compiled? " + b);
711                     }
712                     if (b || XCOMP || STRESS_CC || !USE_COMPILER) {
713                         // Don't control compilation if -Xcomp is enabled, or if compiler is disabled
714                         break;
715                     }
716                     Asserts.assertTrue(OSR_TEST_TIMEOUT < 0 || elapsed < OSR_TEST_TIMEOUT, test + " not compiled after " + OSR_TEST_TIMEOUT + " ms");
717                 }
718                 Asserts.assertTrue(XCOMP || STRESS_CC || !USE_COMPILER || WHITE_BOX.isMethodCompiled(test, false), test + " not compiled");


719             } else {
720                 int compLevel = getCompLevel(test.getAnnotation(Test.class));
721                 // Trigger compilation
722                 enqueueMethodForCompilation(test, compLevel);
723                 if (maybeCodeBufferOverflow && !WHITE_BOX.isMethodCompiled(test, false)) {
724                     // Let's disable VerifyOops temporarily and retry.
725                     WHITE_BOX.setBooleanVMFlag("VerifyOops", false);
726                     WHITE_BOX.clearMethodState(test);
727                     enqueueMethodForCompilation(test, compLevel);
728                     WHITE_BOX.setBooleanVMFlag("VerifyOops", true);
729                 }
730                 Asserts.assertTrue(STRESS_CC || !USE_COMPILER || WHITE_BOX.isMethodCompiled(test, false), test + " not compiled");
731                 // Check result
732                 verifier.invoke(this, false);
733             }
734             if (PRINT_TIMES || VERBOSE) {
735                 long endTime = System.nanoTime();
736                 long duration = (endTime - startTime);
737                 durations.put(duration, test.getName());
738                 if (VERBOSE) {
739                     System.out.println("Done " + test.getName() + ": " + duration + " ns = " + (duration / 1000000) + " ms");
740                 }
741             }
742             if (GC_AFTER) {
743                 System.out.println("doing GC");
744                 System.gc();
745             }
746         }
747 
748         // Print execution times
749         if (PRINT_TIMES) {
750           System.out.println("\n\nTest execution times:");
751           for (Map.Entry<Long, String> entry : durations.entrySet()) {
752               System.out.format("%-10s%15d ns\n", entry.getValue() + ":", entry.getKey());
753           }
754         }
755     }

757     // Get the appropriate compilation level for a method, according to the
758     // given annotation, as well as the current test scenario and VM options.
759     //
760     private int getCompLevel(Object annotation) {
761         int compLevel;
762         if (annotation == null) {
763             compLevel = COMP_LEVEL_ANY;
764         } else if (annotation instanceof Test) {
765             compLevel = ((Test)annotation).compLevel();
766         } else {
767             compLevel = ((ForceCompile)annotation).compLevel();
768         }
769 
770         return restrictCompLevel(compLevel);
771     }
772 
773     // Get the appropriate level as permitted by the test scenario and VM options.
774     private static int restrictCompLevel(int compLevel) {
775         if (compLevel == COMP_LEVEL_ANY) {
776             compLevel = COMP_LEVEL_FULL_OPTIMIZATION;
777         }
778         if (FLIP_C1_C2) {
779             // Effectively treat all (compLevel = C1) as (compLevel = C2), and
780             //                       (compLevel = C2) as (compLevel = C1).
781             if (compLevel == COMP_LEVEL_SIMPLE) {
782                 compLevel = COMP_LEVEL_FULL_OPTIMIZATION;
783             } else if (compLevel == COMP_LEVEL_FULL_OPTIMIZATION) {
784                 compLevel = COMP_LEVEL_SIMPLE;
785             }
786         }
787         if (!TEST_C1 && compLevel < COMP_LEVEL_FULL_OPTIMIZATION) {
788             compLevel = COMP_LEVEL_FULL_OPTIMIZATION;
789         }
790         if (compLevel > (int)TieredStopAtLevel) {
791             compLevel = (int)TieredStopAtLevel;
792         }
793         return compLevel;
794     }

796     public static void enqueueMethodForCompilation(Method m, int level) {
797         level = restrictCompLevel(level);
798         if (VERBOSE) {
799             System.out.println("enqueueMethodForCompilation " + m + ", level = " + level);
800         }
801         WHITE_BOX.enqueueMethodForCompilation(m, level);
802     }

804     // Unlike C2, C1 intrinsics never deoptimize System.arraycopy. Instead, we fall back to
805     // a normal method invocation when encountering flattened arrays.
806     static boolean isCompiledByC2(Method m) {
807         return USE_COMPILER && !XCOMP && WHITE_BOX.isMethodCompiled(m, false) &&
808             WHITE_BOX.getMethodCompilationLevel(m, false) >= COMP_LEVEL_FULL_OPTIMIZATION;
