  96 void GrowableCache::recache() {
  97   int len = _elements->length();
  98 
  99   FREE_C_HEAP_ARRAY(address, _cache);
 100   _cache = NEW_C_HEAP_ARRAY(address,len+1, mtInternal);
 101 
 102   for (int i=0; i<len; i++) {
 103     _cache[i] = _elements->at(i)->getCacheValue();
 104     //
 105     // The cache entry has gone bad. Without a valid frame pointer
 106     // value, the entry is useless so we simply delete it in product
 107     // mode. The call to remove() will rebuild the cache again
 108     // without the bad entry.
 109     //
 110     if (_cache[i] == NULL) {
 111       assert(false, "cannot recache NULL elements");
 112       remove(i);
 113       return;
 114     }
 115   }
 116   _cache[len] = NULL;
 117 
 118   _listener_fun(_this_obj,_cache);
 119 }

 179   assert(e != NULL, "e != NULL");
 180   _elements->remove(e);
 181   delete e;
 182   recache();
 183 }

 185 // clear out all elements, release all heap space and
 186 // let our listener know that things have changed.

 191   }
 192   _elements->clear();
 193   recache();
 194 }

 200 JvmtiBreakpoint::JvmtiBreakpoint(Method* m_method, jlocation location)
 201     : _method(m_method), _bci((int)location), _class_holder(NULL) {
 202   assert(_method != NULL, "No method for breakpoint.");
 203   assert(_bci >= 0, "Negative bci for breakpoint.");
 204   oop class_holder_oop  = _method->method_holder()->klass_holder();
 205   _class_holder = OopStorageSet::vm_global()->allocate();
 206   if (_class_holder == NULL) {
 207     vm_exit_out_of_memory(sizeof(oop), OOM_MALLOC_ERROR,
 208                           "Cannot create breakpoint oop handle");
 209   }
 210   NativeAccess<>::oop_store(_class_holder, class_holder_oop);
 211 }
 212 
 213 JvmtiBreakpoint::~JvmtiBreakpoint() {
 214   if (_class_holder != NULL) {
 215     NativeAccess<>::oop_store(_class_holder, (oop)NULL);
 216     OopStorageSet::vm_global()->release(_class_holder);
 217   }
 218 }
 219 
 220 void JvmtiBreakpoint::copy(JvmtiBreakpoint& bp) {
 221   _method   = bp._method;
 222   _bci      = bp._bci;
 223   _class_holder = OopStorageSet::vm_global()->allocate();
 224   if (_class_holder == NULL) {
 225     vm_exit_out_of_memory(sizeof(oop), OOM_MALLOC_ERROR,
 226                           "Cannot create breakpoint oop handle");
 227   }
 228   oop resolved_ch = NativeAccess<>::oop_load(bp._class_holder);
 229   NativeAccess<>::oop_store(_class_holder, resolved_ch);
 230 }

 241 void JvmtiBreakpoint::each_method_version_do(method_action meth_act) {
 242   ((Method*)_method->*meth_act)(_bci);
 243 
 244   // add/remove breakpoint to/from versions of the method that are EMCP.
 245   Thread *thread = Thread::current();
 246   InstanceKlass* ik = _method->method_holder();
 247   Symbol* m_name = _method->name();
 248   Symbol* m_signature = _method->signature();
 249 
 250   // search previous versions if they exist
 251   for (InstanceKlass* pv_node = ik->previous_versions();
 252        pv_node != NULL;
 253        pv_node = pv_node->previous_versions()) {
 254     Array<Method*>* methods = pv_node->methods();
 255 
 256     for (int i = methods->length() - 1; i >= 0; i--) {
 257       Method* method = methods->at(i);
 258       // Only set breakpoints in running EMCP methods.
 259       if (method->is_running_emcp() &&
 260           method->name() == m_name &&
 261           method->signature() == m_signature) {
 262         ResourceMark rm;
 263         log_debug(redefine, class, breakpoint)
 264           ("%sing breakpoint in %s(%s)", meth_act == &Method::set_breakpoint ? "sett" : "clear",
 265            method->name()->as_C_string(), method->signature()->as_C_string());
 266         (method->*meth_act)(_bci);
 267         break;
 268       }
 269     }
 270   }
 271 }
 272 
 273 void JvmtiBreakpoint::set() {
 274   each_method_version_do(&Method::set_breakpoint);
 275 }
 276 
 277 void JvmtiBreakpoint::clear() {
 278   each_method_version_do(&Method::clear_breakpoint);
 279 }

 296 
 297 void VM_ChangeBreakpoints::doit() {
 298   switch (_operation) {
 299   case SET_BREAKPOINT:
 300     _breakpoints->set_at_safepoint(*_bp);
 301     break;
 302   case CLEAR_BREAKPOINT:
 303     _breakpoints->clear_at_safepoint(*_bp);
 304     break;
 305   default:
 306     assert(false, "Unknown operation");
 307   }
 308 }

 338 void JvmtiBreakpoints::set_at_safepoint(JvmtiBreakpoint& bp) {
 339   assert(SafepointSynchronize::is_at_safepoint(), "must be at safepoint");
 340 
 341   int i = _bps.find(bp);
 342   if (i == -1) {
 343     _bps.append(bp);
 344     bp.set();
 345   }
 346 }
 347 
 348 void JvmtiBreakpoints::clear_at_safepoint(JvmtiBreakpoint& bp) {
 349   assert(SafepointSynchronize::is_at_safepoint(), "must be at safepoint");
 350 
 351   int i = _bps.find(bp);
 352   if (i != -1) {
 353     _bps.remove(i);
 354     bp.clear();
 355   }
 356 }

 360 int JvmtiBreakpoints::set(JvmtiBreakpoint& bp) {
 361   if ( _bps.find(bp) != -1) {
 362      return JVMTI_ERROR_DUPLICATE;
 363   }
 364   VM_ChangeBreakpoints set_breakpoint(VM_ChangeBreakpoints::SET_BREAKPOINT, &bp);
 365   VMThread::execute(&set_breakpoint);

 371      return JVMTI_ERROR_NOT_FOUND;

 379 void JvmtiBreakpoints::clearall_in_class_at_safepoint(Klass* klass) {
 380   bool changed = true;
 381   // We are going to run thru the list of bkpts
 382   // and delete some.  This deletion probably alters
 383   // the list in some implementation defined way such
 384   // that when we delete entry i, the next entry might
 385   // no longer be at i+1.  To be safe, each time we delete
 386   // an entry, we'll just start again from the beginning.
 387   // We'll stop when we make a pass thru the whole list without
 388   // deleting anything.
 389   while (changed) {
 390     int len = _bps.length();
 391     changed = false;
 392     for (int i = 0; i < len; i++) {
 393       JvmtiBreakpoint& bp = _bps.at(i);
 394       if (bp.method()->method_holder() == klass) {
 395         bp.clear();
 396         _bps.remove(i);
 397         // This changed 'i' so we have to start over.
 398         changed = true;
 399         break;
 400       }
 401     }
 402   }
 403 }

 504 // Check that the klass is assignable to a type with the given signature.
 505 // Another solution could be to use the function Klass::is_subtype_of(type).
 506 // But the type class can be forced to load/initialize eagerly in such a case.
 507 // This may cause unexpected consequences like CFLH or class-init JVMTI events.
 508 // It is better to avoid such a behavior.
 509 bool VM_GetOrSetLocal::is_assignable(const char* ty_sign, Klass* klass, Thread* thread) {
 510   assert(ty_sign != NULL, "type signature must not be NULL");
 511   assert(thread != NULL, "thread must not be NULL");
 512   assert(klass != NULL, "klass must not be NULL");
 513 
 514   int len = (int) strlen(ty_sign);
 515   if ((ty_sign[0] == JVM_SIGNATURE_CLASS ||
 516        ty_sign[0] == JVM_SIGNATURE_INLINE_TYPE) &&
 517       ty_sign[len-1] == JVM_SIGNATURE_ENDCLASS) { // Need pure class/interface name
 518     ty_sign++;
 519     len -= 2;
 520   }
 521   TempNewSymbol ty_sym = SymbolTable::new_symbol(ty_sign, len);
 522   if (klass->name() == ty_sym) {
 523     return true;
 524   }
 525   // Compare primary supers
 526   int super_depth = klass->super_depth();
 527   int idx;
 528   for (idx = 0; idx < super_depth; idx++) {
 529     if (klass->primary_super_of_depth(idx)->name() == ty_sym) {
 530       return true;
 531     }
 532   }
 533   // Compare secondary supers
 534   const Array<Klass*>* sec_supers = klass->secondary_supers();
 535   for (idx = 0; idx < sec_supers->length(); idx++) {
 536     if (((Klass*) sec_supers->at(idx))->name() == ty_sym) {
 537       return true;
 538     }
 539   }
 540   return false;
 541 }

 543 // Checks error conditions:
 544 //   JVMTI_ERROR_INVALID_SLOT
 545 //   JVMTI_ERROR_TYPE_MISMATCH
 546 // Returns: 'true' - everything is Ok, 'false' - error code
 547 
 548 bool VM_GetOrSetLocal::check_slot_type_lvt(javaVFrame* jvf) {
 549   Method* method_oop = jvf->method();
 550   jint num_entries = method_oop->localvariable_table_length();
 551   if (num_entries == 0) {
 552     _result = JVMTI_ERROR_INVALID_SLOT;
 553     return false;       // There are no slots
 554   }
 555   int signature_idx = -1;
 556   int vf_bci = jvf->bci();
 557   LocalVariableTableElement* table = method_oop->localvariable_table_start();
 558   for (int i = 0; i < num_entries; i++) {
 559     int start_bci = table[i].start_bci;
 560     int end_bci = start_bci + table[i].length;
 561 
 562     // Here we assume that locations of LVT entries
 563     // with the same slot number cannot be overlapped
 564     if (_index == (jint) table[i].slot && start_bci <= vf_bci && vf_bci <= end_bci) {
 565       signature_idx = (int) table[i].descriptor_cp_index;
 566       break;
 567     }
 568   }
 569   if (signature_idx == -1) {
 570     _result = JVMTI_ERROR_INVALID_SLOT;
 571     return false;       // Incorrect slot index
 572   }
 573   Symbol*   sign_sym  = method_oop->constants()->symbol_at(signature_idx);
 574   BasicType slot_type = Signature::basic_type(sign_sym);
 575 
 576   switch (slot_type) {
 577   case T_BYTE:
 578   case T_SHORT:
 579   case T_CHAR:
 580   case T_BOOLEAN:
 581     slot_type = T_INT;
 582     break;
 583   case T_ARRAY:
 584   case T_VALUETYPE:
 585     slot_type = T_OBJECT;
 586     break;
 587   default:
 588     break;
 589   };
 590   if (_type != slot_type) {
 591     _result = JVMTI_ERROR_TYPE_MISMATCH;
 592     return false;
 593   }
 594 
 595   jobject jobj = _value.l;
 596   if (_set && slot_type == T_OBJECT && jobj != NULL) { // NULL reference is allowed
 597     // Check that the jobject class matches the return type signature.
 598     JavaThread* cur_thread = JavaThread::current();
 599     HandleMark hm(cur_thread);
 600 
 601     Handle obj(cur_thread, JNIHandles::resolve_external_guard(jobj));
 602     NULL_CHECK(obj, (_result = JVMTI_ERROR_INVALID_OBJECT, false));
 603     Klass* ob_k = obj->klass();
 604     NULL_CHECK(ob_k, (_result = JVMTI_ERROR_INVALID_OBJECT, false));
 605 
 606     const char* signature = (const char *) sign_sym->as_utf8();
 607     if (!is_assignable(signature, ob_k, cur_thread)) {
 608       _result = JVMTI_ERROR_TYPE_MISMATCH;
 609       return false;
 610     }
 611   }
 612   return true;
 613 }

 615 bool VM_GetOrSetLocal::check_slot_type_no_lvt(javaVFrame* jvf) {
 616   Method* method_oop = jvf->method();
 617   jint extra_slot = (_type == T_LONG || _type == T_DOUBLE) ? 1 : 0;
 618 
 619   if (_index < 0 || _index + extra_slot >= method_oop->max_locals()) {
 620     _result = JVMTI_ERROR_INVALID_SLOT;
 621     return false;
 622   }
 623   StackValueCollection *locals = _jvf->locals();
 624   BasicType slot_type = locals->at(_index)->type();
 625 
 626   if (slot_type == T_CONFLICT) {
 627     _result = JVMTI_ERROR_INVALID_SLOT;
 628     return false;
 629   }
 630   if (extra_slot) {
 631     BasicType extra_slot_type = locals->at(_index + 1)->type();
 632     if (extra_slot_type != T_INT) {
 633       _result = JVMTI_ERROR_INVALID_SLOT;
 634       return false;
 635     }
 636   }
 637   if (_type != slot_type && (_type == T_OBJECT || slot_type != T_INT)) {
 638     _result = JVMTI_ERROR_TYPE_MISMATCH;
 639     return false;
 640   }
 641   return true;
 642 }

 648 bool VM_GetOrSetLocal::doit_prologue() {
 649   _jvf = get_java_vframe();
 650   NULL_CHECK(_jvf, false);
 651 
 652   Method* method_oop = _jvf->method();
 653   if (getting_receiver()) {
 654     if (method_oop->is_static()) {
 655       _result = JVMTI_ERROR_INVALID_SLOT;
 656       return false;
 657     }
 658     return true;
 659   }
 660 
 661   if (method_oop->is_native()) {
 662     _result = JVMTI_ERROR_OPAQUE_FRAME;
 663     return false;
 664   }
 665 
 666   if (!check_slot_type_no_lvt(_jvf)) {
 667     return false;
 668   }
 669   if (method_oop->has_localvariable_table()) {
 670     return check_slot_type_lvt(_jvf);
 671   }
 672   return true;
 673 }
 674 
 675 void VM_GetOrSetLocal::doit() {
 676   InterpreterOopMap oop_mask;
 677   _jvf->method()->mask_for(_jvf->bci(), &oop_mask);
 678   if (oop_mask.is_dead(_index)) {
 679     // The local can be invalid and uninitialized in the scope of current bci
 680     _result = JVMTI_ERROR_INVALID_SLOT;
 681     return;
 682   }
 683   if (_set) {
 684     // Force deoptimization of frame if compiled because it's
 685     // possible the compiler emitted some locals as constant values,
 686     // meaning they are not mutable.
 687     if (can_be_deoptimized(_jvf)) {
 688 
 689       // Schedule deoptimization so that eventually the local
 690       // update will be written to an interpreter frame.
 691       Deoptimization::deoptimize_frame(_jvf->thread(), _jvf->fr().id());
 692 
 693       // Now store a new value for the local which will be applied
 694       // once deoptimization occurs. Note however that while this
 695       // write is deferred until deoptimization actually happens
 696       // can vframe created after this point will have its locals
 697       // reflecting this update so as far as anyone can see the
 698       // write has already taken place.
 699 
 700       // If we are updating an oop then get the oop from the handle
 701       // since the handle will be long gone by the time the deopt
 702       // happens. The oop stored in the deferred local will be
 703       // gc'd on its own.
 704       if (_type == T_OBJECT || _type == T_VALUETYPE) {
 705         _value.l = cast_from_oop<jobject>(JNIHandles::resolve_external_guard(_value.l));
 706       }
 707       // Re-read the vframe so we can see that it is deoptimized
 708       // [ Only need because of assert in update_local() ]
 709       _jvf = get_java_vframe();
 710       ((compiledVFrame*)_jvf)->update_local(_type, _index, _value);
 711       return;
 712     }
 713     StackValueCollection *locals = _jvf->locals();
 714     HandleMark hm;
 715 
 716     switch (_type) {
 717       case T_INT:    locals->set_int_at   (_index, _value.i); break;
 718       case T_LONG:   locals->set_long_at  (_index, _value.j); break;
 719       case T_FLOAT:  locals->set_float_at (_index, _value.f); break;
 720       case T_DOUBLE: locals->set_double_at(_index, _value.d); break;
 721       case T_OBJECT:
 722       case T_VALUETYPE: {
 723         Handle ob_h(Thread::current(), JNIHandles::resolve_external_guard(_value.l));
 724         locals->set_obj_at (_index, ob_h);
 725         break;
 726       }
 727       default: ShouldNotReachHere();
 728     }
 729     _jvf->set_locals(locals);
 730   } else {
 731     if (_jvf->method()->is_native() && _jvf->is_compiled_frame()) {
 732       assert(getting_receiver(), "Can only get here when getting receiver");
 733       oop receiver = _jvf->fr().get_native_receiver();
 734       _value.l = JNIHandles::make_local(_calling_thread, receiver);
 735     } else {
 736       StackValueCollection *locals = _jvf->locals();
 737 
 738       switch (_type) {
 739         case T_INT:    _value.i = locals->int_at   (_index);   break;
 740         case T_LONG:   _value.j = locals->long_at  (_index);   break;
 741         case T_FLOAT:  _value.f = locals->float_at (_index);   break;
 742         case T_DOUBLE: _value.d = locals->double_at(_index);   break;
 743         case T_OBJECT:
 744         case T_VALUETYPE: {
 745           // Wrap the oop to be returned in a local JNI handle since
 746           // oops_do() no longer applies after doit() is finished.
 747           oop obj = locals->obj_at(_index)();
 748           _value.l = JNIHandles::make_local(_calling_thread, obj);
 749           break;
 750         }
 751         default: ShouldNotReachHere();
 752       }
 753     }
 754   }
 755 }

 773 bool JvmtiSuspendControl::suspend(JavaThread *java_thread) {
 774   // external suspend should have caught suspending a thread twice
 775 
 776   // Immediate suspension required for JPDA back-end so JVMTI agent threads do
 777   // not deadlock due to later suspension on transitions while holding
 778   // raw monitors.  Passing true causes the immediate suspension.
 779   // java_suspend() will catch threads in the process of exiting
 780   // and will ignore them.
 781   java_thread->java_suspend();
 782 
 783   // It would be nice to have the following assertion in all the time,
 784   // but it is possible for a racing resume request to have resumed
 785   // this thread right after we suspended it. Temporarily enable this
 786   // assertion if you are chasing a different kind of bug.
 787   //
 788   // assert(java_lang_Thread::thread(java_thread->threadObj()) == NULL ||
 789   //   java_thread->is_being_ext_suspended(), "thread is not suspended");
 790 
 791   if (java_lang_Thread::thread(java_thread->threadObj()) == NULL) {
 792     // check again because we can get delayed in java_suspend():
 793     // the thread is in process of exiting.
 794     return false;
 795   }
 796 
 797   return true;
 798 }
 799 
 800 bool JvmtiSuspendControl::resume(JavaThread *java_thread) {
 801   // external suspend should have caught resuming a thread twice
 802   assert(java_thread->is_being_ext_suspended(), "thread should be suspended");
 803 
 804   // resume thread
 805   {
 806     // must always grab Threads_lock, see JVM_SuspendThread
 807     MutexLocker ml(Threads_lock);
 808     java_thread->java_resume();
 809   }
 810 
 811   return true;
 812 }

 874 void JvmtiDeferredEvent::post() {
 875   assert(Thread::current()->is_service_thread(),
 876          "Service thread must post enqueued events");
 877   switch(_type) {
 878     case TYPE_COMPILED_METHOD_LOAD: {
 879       nmethod* nm = _event_data.compiled_method_load;
 880       JvmtiExport::post_compiled_method_load(nm);
 881       break;
 882     }
 883     case TYPE_COMPILED_METHOD_UNLOAD: {
 884       JvmtiExport::post_compiled_method_unload(
 885         _event_data.compiled_method_unload.method_id,
 886         _event_data.compiled_method_unload.code_begin);
 887       break;
 888     }
 889     case TYPE_DYNAMIC_CODE_GENERATED: {
 890       JvmtiExport::post_dynamic_code_generated_internal(
 891         // if strdup failed give the event a default name
 892         (_event_data.dynamic_code_generated.name == NULL)
 893           ? "unknown_code" : _event_data.dynamic_code_generated.name,
 894         _event_data.dynamic_code_generated.code_begin,
 895         _event_data.dynamic_code_generated.code_end);
 896       if (_event_data.dynamic_code_generated.name != NULL) {
 897         // release our copy
 898         os::free((void *)_event_data.dynamic_code_generated.name);
 899       }
 900       break;
 901     }
 902     case TYPE_CLASS_UNLOAD: {
 903       JvmtiExport::post_class_unload_internal(
 904         // if strdup failed give the event a default name
 905         (_event_data.class_unload.name == NULL)
 906           ? "unknown_class" : _event_data.class_unload.name);
 907       if (_event_data.class_unload.name != NULL) {
 908         // release our copy
 909         os::free((void *)_event_data.class_unload.name);
 910       }
 911       break;
 912     }
 913     default:
 914       ShouldNotReachHere();
 915   }
 916 }

 947 bool JvmtiDeferredEventQueue::has_events() {
 948   // We save the queued events before the live phase and post them when it starts.
 949   // This code could skip saving the events on the queue before the live
 950   // phase and ignore them, but this would change how we do things now.
 951   // Starting the service thread earlier causes this to be called before the live phase begins.
 952   // The events on the queue should all be posted after the live phase so this is an
 953   // ok check.  Before the live phase, DynamicCodeGenerated events are posted directly.
 954   // If we add other types of events to the deferred queue, this could get ugly.
 955   return JvmtiEnvBase::get_phase() == JVMTI_PHASE_LIVE  && _queue_head != NULL;
 956 }

 958 void JvmtiDeferredEventQueue::enqueue(JvmtiDeferredEvent event) {
 959   // Events get added to the end of the queue (and are pulled off the front).
 960   QueueNode* node = new QueueNode(event);
 961   if (_queue_tail == NULL) {
 962     _queue_tail = _queue_head = node;
 963   } else {
 964     assert(_queue_tail->next() == NULL, "Must be the last element in the list");
 965     _queue_tail->set_next(node);
 966     _queue_tail = node;
 967   }
 968 
 969   assert((_queue_head == NULL) == (_queue_tail == NULL),
 970          "Inconsistent queue markers");
 971 }
 972 
 973 JvmtiDeferredEvent JvmtiDeferredEventQueue::dequeue() {
 974   assert(_queue_head != NULL, "Nothing to dequeue");
 975 
 976   if (_queue_head == NULL) {
 977     // Just in case this happens in product; it shouldn't but let's not crash
 978     return JvmtiDeferredEvent();
 979   }
 980 
 981   QueueNode* node = _queue_head;
 982   _queue_head = _queue_head->next();
 983   if (_queue_head == NULL) {
 984     _queue_tail = NULL;
 985   }
 986 
 987   assert((_queue_head == NULL) == (_queue_tail == NULL),
 988          "Inconsistent queue markers");
 989 
 990   JvmtiDeferredEvent event = node->event();
 991   delete node;
 992   return event;
 993 }
 994 
 995 void JvmtiDeferredEventQueue::post(JvmtiEnv* env) {
 996   // Post and destroy queue nodes
 997   while (_queue_head != NULL) {
 998      JvmtiDeferredEvent event = dequeue();
 999      event.post_compiled_method_load_event(env);
1000   }
1001 }

1006   }
