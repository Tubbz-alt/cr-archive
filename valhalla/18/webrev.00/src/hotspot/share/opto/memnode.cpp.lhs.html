 107 void MemNode::dump_adr_type(const Node* mem, const TypePtr* adr_type, outputStream *st) {
 108   st->print(" @");
 109   if (adr_type == NULL) {
 110     st->print("NULL");
 111   } else {
 112     adr_type->dump_on(st);
 113     Compile* C = Compile::current();
 114     Compile::AliasType* atp = NULL;
 115     if (C->have_alias_type(adr_type))  atp = C->alias_type(adr_type);
 116     if (atp == NULL)
 117       st->print(", idx=?\?;");
 118     else if (atp->index() == Compile::AliasIdxBot)
 119       st->print(", idx=Bot;");
 120     else if (atp->index() == Compile::AliasIdxTop)
 121       st->print(", idx=Top;");
 122     else if (atp->index() == Compile::AliasIdxRaw)
 123       st->print(", idx=Raw;");
 124     else {
 125       ciField* field = atp->field();
 126       if (field) {
 127         st->print(", name=");
 128         field->print_name_on(st);
 129       }
 130       st->print(", idx=%d;", atp->index());
 131     }
 132   }
 133 }

 139 Node *MemNode::optimize_simple_memory_chain(Node *mchain, const TypeOopPtr *t_oop, Node *load, PhaseGVN *phase) {
 140   assert((t_oop != NULL), "sanity");
 141   bool is_instance = t_oop->is_known_instance_field();
 142   bool is_boxed_value_load = t_oop->is_ptr_to_boxed_value() &&
 143                              (load != NULL) && load->is_Load() &&
 144                              (phase->is_IterGVN() != NULL);
 145   if (!(is_instance || is_boxed_value_load))
 146     return mchain;  // don't try to optimize non-instance types
 147   uint instance_id = t_oop->instance_id();
 148   Node *start_mem = phase->C->start()->proj_out_or_null(TypeFunc::Memory);
 149   Node *prev = NULL;
 150   Node *result = mchain;
 151   while (prev != result) {
 152     prev = result;
 153     if (result == start_mem)
 154       break;  // hit one of our sentinels
 155     // skip over a call which does not affect this memory slice
 156     if (result->is_Proj() && result->as_Proj()->_con == TypeFunc::Memory) {
 157       Node *proj_in = result->in(0);
 158       if (proj_in->is_Allocate() && proj_in->_idx == instance_id) {
 159         break;  // hit one of our sentinels
 160       } else if (proj_in->is_Call()) {
 161         // ArrayCopyNodes processed here as well
 162         CallNode *call = proj_in->as_Call();
 163         if (!call->may_modify(t_oop, phase)) { // returns false for instances
 164           result = call->in(TypeFunc::Memory);
 165         }
 166       } else if (proj_in->is_Initialize()) {
 167         AllocateNode* alloc = proj_in->as_Initialize()->allocation();
 168         // Stop if this is the initialization for the object instance which
 169         // contains this memory slice, otherwise skip over it.
 170         if ((alloc == NULL) || (alloc->_idx == instance_id)) {
 171           break;
 172         }
 173         if (is_instance) {
 174           result = proj_in->in(TypeFunc::Memory);
 175         } else if (is_boxed_value_load) {
 176           Node* klass = alloc->in(AllocateNode::KlassNode);
 177           const TypeKlassPtr* tklass = phase->type(klass)->is_klassptr();
 178           if (tklass->klass_is_exact() && !tklass->klass()->equals(t_oop->klass())) {
 179             result = proj_in->in(TypeFunc::Memory); // not related allocation
 180           }
 181         }
 182       } else if (proj_in->is_MemBar()) {
 183         ArrayCopyNode* ac = NULL;
 184         if (ArrayCopyNode::may_modify(t_oop, proj_in->as_MemBar(), phase, ac)) {
 185           break;
 186         }
 187         result = proj_in->in(TypeFunc::Memory);
 188       } else {
 189         assert(false, "unexpected projection");
 190       }
 191     } else if (result->is_ClearArray()) {
 192       if (!is_instance || !ClearArrayNode::step_through(&result, instance_id, phase)) {
 193         // Can not bypass initialization of the instance
 194         // we are looking for.
 195         break;
 196       }
 197       // Otherwise skip it (the call updated 'result' value).
 198     } else if (result->is_MergeMem()) {
 199       result = step_through_mergemem(phase, result->as_MergeMem(), t_oop, NULL, tty);
 200     }
 201   }
 202   return result;
 203 }

 205 Node *MemNode::optimize_memory_chain(Node *mchain, const TypePtr *t_adr, Node *load, PhaseGVN *phase) {
 206   const TypeOopPtr* t_oop = t_adr->isa_oopptr();
 207   if (t_oop == NULL)
 208     return mchain;  // don't try to optimize non-oop types
 209   Node* result = optimize_simple_memory_chain(mchain, t_oop, load, phase);
 210   bool is_instance = t_oop->is_known_instance_field();
 211   PhaseIterGVN *igvn = phase->is_IterGVN();
 212   if (is_instance && igvn != NULL && result->is_Phi()) {
 213     PhiNode *mphi = result->as_Phi();
 214     assert(mphi->bottom_type() == Type::MEMORY, "memory phi required");
 215     const TypePtr *t = mphi->adr_type();
 216     if (t == TypePtr::BOTTOM || t == TypeRawPtr::BOTTOM ||
 217         (t->isa_oopptr() && !t->is_oopptr()->is_known_instance() &&
 218          t->is_oopptr()->cast_to_exactness(true)
 219            ->is_oopptr()->cast_to_ptr_type(t_oop->ptr())
 220             ->is_oopptr()->cast_to_instance_id(t_oop->instance_id()) == t_oop)) {
 221       // clone the Phi with our address type
 222       result = mphi->split_out_instance(t_adr, igvn);
 223     } else {
 224       assert(phase->C->get_alias_index(t) == phase->C->get_alias_index(t_adr), "correct memory chain");
 225     }
 226   }
 227   return result;
 228 }

 230 static Node *step_through_mergemem(PhaseGVN *phase, MergeMemNode *mmem,  const TypePtr *tp, const TypePtr *adr_check, outputStream *st) {
 231   uint alias_idx = phase->C->get_alias_index(tp);
 232   Node *mem = mmem;
 233 #ifdef ASSERT
 234   {
 235     // Check that current type is consistent with the alias index used during graph construction
 236     assert(alias_idx >= Compile::AliasIdxRaw, "must not be a bad alias_idx");
 237     bool consistent =  adr_check == NULL || adr_check->empty() ||
 238                        phase->C->must_alias(adr_check, alias_idx );
 239     // Sometimes dead array references collapse to a[-1], a[-2], or a[-3]
 240     if( !consistent && adr_check != NULL && !adr_check->empty() &&
 241         tp->isa_aryptr() &&        tp->offset() == Type::OffsetBot &&
 242         adr_check->isa_aryptr() && adr_check->offset() != Type::OffsetBot &&
 243         ( adr_check->offset() == arrayOopDesc::length_offset_in_bytes() ||
 244           adr_check->offset() == oopDesc::klass_offset_in_bytes() ||
 245           adr_check->offset() == oopDesc::mark_offset_in_bytes() ) ) {
 246       // don't assert if it is dead code.
 247       consistent = true;
 248     }
 249     if( !consistent ) {
 250       st->print("alias_idx==%d, adr_check==", alias_idx);
 251       if( adr_check == NULL ) {
 252         st->print("NULL");
 253       } else {
 254         adr_check->dump();
 255       }
 256       st->cr();
 257       print_alias_types();
 258       assert(consistent, "adr_check must match alias idx");
 259     }
 260   }
 261 #endif
 262   // TypeOopPtr::NOTNULL+any is an OOP with unknown offset - generally
 263   // means an array I have not precisely typed yet.  Do not do any
 264   // alias stuff with it any time soon.
 265   const TypeOopPtr *toop = tp->isa_oopptr();
 266   if( tp->base() != Type::AnyPtr &&
 267       !(toop &&
 268         toop->klass() != NULL &&
 269         toop->klass()->is_java_lang_Object() &&
 270         toop->offset() == Type::OffsetBot) ) {
 271     // compress paths and change unreachable cycles to TOP
 272     // If not, we can update the input infinitely along a MergeMem cycle
 273     // Equivalent code in PhiNode::Ideal
 274     Node* m  = phase->transform(mmem);
 275     // If transformed to a MergeMem, get the desired slice
 276     // Otherwise the returned node represents memory for every slice
 277     mem = (m->is_MergeMem())? m->as_MergeMem()->memory_at(alias_idx) : m;
 278     // Update input if it is progress over what we have now
 279   }
 280   return mem;
 281 }

 283 //--------------------------Ideal_common---------------------------------------
 284 // Look for degenerate control and memory inputs.  Bypass MergeMem inputs.
 285 // Unhook non-raw memories from complete (macro-expanded) initializations.
 286 Node *MemNode::Ideal_common(PhaseGVN *phase, bool can_reshape) {
 287   // If our control input is a dead region, kill all below the region
 288   Node *ctl = in(MemNode::Control);
 289   if (ctl && remove_dead_region(phase, can_reshape))
 290     return this;
 291   ctl = in(MemNode::Control);
 292   // Don't bother trying to transform a dead node
 293   if (ctl && ctl->is_top())  return NodeSentinel;
 294 
 295   PhaseIterGVN *igvn = phase->is_IterGVN();
 296   // Wait if control on the worklist.
 297   if (ctl && can_reshape && igvn != NULL) {
 298     Node* bol = NULL;
 299     Node* cmp = NULL;
 300     if (ctl->in(0)->is_If()) {
 301       assert(ctl->is_IfTrue() || ctl->is_IfFalse(), "sanity");
 302       bol = ctl->in(0)->in(1);
 303       if (bol->is_Bool())
 304         cmp = ctl->in(0)->in(1)->in(1);
 305     }
 306     if (igvn->_worklist.member(ctl) ||
 307         (bol != NULL && igvn->_worklist.member(bol)) ||
 308         (cmp != NULL && igvn->_worklist.member(cmp)) ) {
 309       // This control path may be dead.
 310       // Delay this memory node transformation until the control is processed.
 311       phase->is_IterGVN()->_worklist.push(this);
 312       return NodeSentinel; // caller will return NULL
 313     }
 314   }
 315   // Ignore if memory is dead, or self-loop
 316   Node *mem = in(MemNode::Memory);
 317   if (phase->type( mem ) == Type::TOP) return NodeSentinel; // caller will return NULL
 318   assert(mem != this, "dead loop in MemNode::Ideal");
 319 
 320   if (can_reshape && igvn != NULL && igvn->_worklist.member(mem)) {
 321     // This memory slice may be dead.
 322     // Delay this mem node transformation until the memory is processed.
 323     phase->is_IterGVN()->_worklist.push(this);
 324     return NodeSentinel; // caller will return NULL
 325   }
 326 
 327   Node *address = in(MemNode::Address);
 328   const Type *t_adr = phase->type(address);
 329   if (t_adr == Type::TOP)              return NodeSentinel; // caller will return NULL
 330 
 331   if (can_reshape && is_unsafe_access() && (t_adr == TypePtr::NULL_PTR)) {
 332     // Unsafe off-heap access with zero address. Remove access and other control users
 333     // to not confuse optimizations and add a HaltNode to fail if this is ever executed.
 334     assert(ctl != NULL, "unsafe accesses should be control dependent");
 335     for (DUIterator_Fast imax, i = ctl->fast_outs(imax); i < imax; i++) {
 336       Node* u = ctl->fast_out(i);
 337       if (u != ctl) {
 338         igvn->rehash_node_delayed(u);
 339         int nb = u->replace_edge(ctl, phase->C->top());
 340         --i, imax -= nb;
 341       }
 342     }
 343     Node* frame = igvn->transform(new ParmNode(phase->C->start(), TypeFunc::FramePtr));
 344     Node* halt = igvn->transform(new HaltNode(ctl, frame, "unsafe off-heap access with zero address"));
 345     phase->C->root()->add_req(halt);
 346     return this;
 347   }
 348 
 349   if (can_reshape && igvn != NULL &&
 350       (igvn->_worklist.member(address) ||
 351        (igvn->_worklist.size() > 0 && t_adr != adr_type())) ) {
 352     // The address's base and type may change when the address is processed.
 353     // Delay this mem node transformation until the address is processed.
 354     phase->is_IterGVN()->_worklist.push(this);
 355     return NodeSentinel; // caller will return NULL
 356   }
 357 
 358   // Do NOT remove or optimize the next lines: ensure a new alias index
 359   // is allocated for an oop pointer type before Escape Analysis.
 360   // Note: C++ will not remove it since the call has side effect.
 361   if (t_adr->isa_oopptr()) {
 362     int alias_idx = phase->C->get_alias_index(t_adr->is_ptr());
 363   }
 364 
 365   Node* base = NULL;
 366   if (address->is_AddP()) {
 367     base = address->in(AddPNode::Base);
 368   }
 369   if (base != NULL && phase->type(base)->higher_equal(TypePtr::NULL_PTR) &&
 370       !t_adr->isa_rawptr()) {
 371     // Note: raw address has TOP base and top->higher_equal(TypePtr::NULL_PTR) is true.
 372     // Skip this node optimization if its address has TOP base.
 373     return NodeSentinel; // caller will return NULL
 374   }
 375 
 376   // Avoid independent memory operations
 377   Node* old_mem = mem;
 378 
 379   // The code which unhooks non-raw memories from complete (macro-expanded)
 380   // initializations was removed. After macro-expansion all stores catched
 381   // by Initialize node became raw stores and there is no information
 382   // which memory slices they modify. So it is unsafe to move any memory
 383   // operation above these stores. Also in most cases hooked non-raw memories
 384   // were already unhooked by using information from detect_ptr_independence()
 385   // and find_previous_store().
 386 
 387   if (mem->is_MergeMem()) {
 388     MergeMemNode* mmem = mem->as_MergeMem();
 389     const TypePtr *tp = t_adr->is_ptr();
 390 
 391     mem = step_through_mergemem(phase, mmem, tp, adr_type(), tty);
 392   }
 393 
 394   if (mem != old_mem) {
 395     set_req(MemNode::Memory, mem);
 396     if (can_reshape && old_mem->outcnt() == 0 && igvn != NULL) {
 397       igvn->_worklist.push(old_mem);
 398     }
 399     if (phase->type(mem) == Type::TOP) return NodeSentinel;
 400     return this;
 401   }
 402 
 403   // let the subclass continue analyzing...
 404   return NULL;
 405 }

 407 // Helper function for proving some simple control dominations.
 408 // Attempt to prove that all control inputs of 'dom' dominate 'sub'.
 409 // Already assumes that 'dom' is available at 'sub', and that 'sub'
 410 // is not a constant (dominated by the method's StartNode).
 411 // Used by MemNode::find_previous_store to prove that the
 412 // control input of a memory operation predates (dominates)
 413 // an allocation it wants to look past.
 414 bool MemNode::all_controls_dominate(Node* dom, Node* sub) {
 415   if (dom == NULL || dom->is_top() || sub == NULL || sub->is_top())
 416     return false; // Conservative answer for dead code
 417 
 418   // Check 'dom'. Skip Proj and CatchProj nodes.
 419   dom = dom->find_exact_control(dom);
 420   if (dom == NULL || dom->is_top())
 421     return false; // Conservative answer for dead code
 422 
 423   if (dom == sub) {
 424     // For the case when, for example, 'sub' is Initialize and the original
 425     // 'dom' is Proj node of the 'sub'.
 426     return false;
 427   }
 428 
 429   if (dom->is_Con() || dom->is_Start() || dom->is_Root() || dom == sub)
 430     return true;
 431 
 432   // 'dom' dominates 'sub' if its control edge and control edges
 433   // of all its inputs dominate or equal to sub's control edge.
 434 
 435   // Currently 'sub' is either Allocate, Initialize or Start nodes.
 436   // Or Region for the check in LoadNode::Ideal();
 437   // 'sub' should have sub->in(0) != NULL.
 438   assert(sub->is_Allocate() || sub->is_Initialize() || sub->is_Start() ||
 439          sub->is_Region() || sub->is_Call(), "expecting only these nodes");
 440 
 441   // Get control edge of 'sub'.
 442   Node* orig_sub = sub;
 443   sub = sub->find_exact_control(sub->in(0));
 444   if (sub == NULL || sub->is_top())
 445     return false; // Conservative answer for dead code
 446 
 447   assert(sub->is_CFG(), "expecting control");
 448 
 449   if (sub == dom)
 450     return true;
 451 
 452   if (sub->is_Start() || sub->is_Root())
 453     return false;
 454 
 455   {
 456     // Check all control edges of 'dom'.
 457 
 458     ResourceMark rm;
 459     Arena* arena = Thread::current()->resource_area();
 460     Node_List nlist(arena);
 461     Unique_Node_List dom_list(arena);
 462 
 463     dom_list.push(dom);
 464     bool only_dominating_controls = false;
 465 
 466     for (uint next = 0; next < dom_list.size(); next++) {
 467       Node* n = dom_list.at(next);
 468       if (n == orig_sub)
 469         return false; // One of dom's inputs dominated by sub.
 470       if (!n->is_CFG() && n->pinned()) {
 471         // Check only own control edge for pinned non-control nodes.
 472         n = n->find_exact_control(n->in(0));
 473         if (n == NULL || n->is_top())
 474           return false; // Conservative answer for dead code
 475         assert(n->is_CFG(), "expecting control");
 476         dom_list.push(n);
 477       } else if (n->is_Con() || n->is_Start() || n->is_Root()) {
 478         only_dominating_controls = true;
 479       } else if (n->is_CFG()) {
 480         if (n->dominates(sub, nlist))
 481           only_dominating_controls = true;
 482         else
 483           return false;
 484       } else {
 485         // First, own control edge.
 486         Node* m = n->find_exact_control(n->in(0));
 487         if (m != NULL) {
 488           if (m->is_top())
 489             return false; // Conservative answer for dead code
 490           dom_list.push(m);
 491         }
 492         // Now, the rest of edges.
 493         uint cnt = n->req();
 494         for (uint i = 1; i < cnt; i++) {
 495           m = n->find_exact_control(n->in(i));
 496           if (m == NULL || m->is_top())
 497             continue;
 498           dom_list.push(m);
 499         }
 500       }
 501     }
 502     return only_dominating_controls;
 503   }
 504 }

 506 //---------------------detect_ptr_independence---------------------------------
 507 // Used by MemNode::find_previous_store to prove that two base
 508 // pointers are never equal.
 509 // The pointers are accompanied by their associated allocations,
 510 // if any, which have been previously discovered by the caller.
 511 bool MemNode::detect_ptr_independence(Node* p1, AllocateNode* a1,
 512                                       Node* p2, AllocateNode* a2,
 513                                       PhaseTransform* phase) {
 514   // Attempt to prove that these two pointers cannot be aliased.
 515   // They may both manifestly be allocations, and they should differ.
 516   // Or, if they are not both allocations, they can be distinct constants.
 517   // Otherwise, one is an allocation and the other a pre-existing value.
 518   if (a1 == NULL && a2 == NULL) {           // neither an allocation
 519     return (p1 != p2) && p1->is_Con() && p2->is_Con();
 520   } else if (a1 != NULL && a2 != NULL) {    // both allocations
 521     return (a1 != a2);
 522   } else if (a1 != NULL) {                  // one allocation a1
 523     // (Note:  p2->is_Con implies p2->in(0)->is_Root, which dominates.)
 524     return all_controls_dominate(p2, a1);
 525   } else { //(a2 != NULL)                   // one allocation a2
 526     return all_controls_dominate(p1, a2);
 527   }
 528   return false;
 529 }

 532 // Find an arraycopy that must have set (can_see_stored_value=true) or
 533 // could have set (can_see_stored_value=false) the value for this load
 534 Node* LoadNode::find_previous_arraycopy(PhaseTransform* phase, Node* ld_alloc, Node*& mem, bool can_see_stored_value) const {
 535   if (mem->is_Proj() && mem->in(0) != NULL && (mem->in(0)->Opcode() == Op_MemBarStoreStore ||
 536                                                mem->in(0)->Opcode() == Op_MemBarCPUOrder)) {
 537     Node* mb = mem->in(0);
 538     if (mb->in(0) != NULL && mb->in(0)->is_Proj() &&
 539         mb->in(0)->in(0) != NULL && mb->in(0)->in(0)->is_ArrayCopy()) {
 540       ArrayCopyNode* ac = mb->in(0)->in(0)->as_ArrayCopy();
 541       if (ac->is_clonebasic()) {
 542         intptr_t offset;
 543         AllocateNode* alloc = AllocateNode::Ideal_allocation(ac->in(ArrayCopyNode::Dest), phase, offset);
 544         if (alloc != NULL && alloc == ld_alloc) {
 545           return ac;
 546         }
 547       }
 548     }
 549   } else if (mem->is_Proj() && mem->in(0) != NULL && mem->in(0)->is_ArrayCopy()) {
 550     ArrayCopyNode* ac = mem->in(0)->as_ArrayCopy();
 551 
 552     if (ac->is_arraycopy_validated() ||
 553         ac->is_copyof_validated() ||
 554         ac->is_copyofrange_validated()) {
 555       Node* ld_addp = in(MemNode::Address);
 556       if (ld_addp->is_AddP()) {
 557         Node* ld_base = ld_addp->in(AddPNode::Address);
 558         Node* ld_offs = ld_addp->in(AddPNode::Offset);
 559 
 560         Node* dest = ac->in(ArrayCopyNode::Dest);
 561 
 562         if (dest == ld_base) {
 563           const TypeX *ld_offs_t = phase->type(ld_offs)->isa_intptr_t();
 564           if (ac->modifies(ld_offs_t->_lo, ld_offs_t->_hi, phase, can_see_stored_value)) {
 565             return ac;
 566           }
 567           if (!can_see_stored_value) {
 568             mem = ac->in(TypeFunc::Memory);
 569           }
 570         }
 571       }
 572     }
 573   }
 574   return NULL;

 577 // The logic for reordering loads and stores uses four steps:
 578 // (a) Walk carefully past stores and initializations which we
 579 //     can prove are independent of this load.
 580 // (b) Observe that the next memory state makes an exact match
 581 //     with self (load or store), and locate the relevant store.
 582 // (c) Ensure that, if we were to wire self directly to the store,
 583 //     the optimizer would fold it up somehow.
 584 // (d) Do the rewiring, and return, depending on some other part of
 585 //     the optimizer to fold up the load.
 586 // This routine handles steps (a) and (b).  Steps (c) and (d) are
 587 // specific to loads and stores, so they are handled by the callers.
 588 // (Currently, only LoadNode::Ideal has steps (c), (d).  More later.)
 589 //
 590 Node* MemNode::find_previous_store(PhaseTransform* phase) {
 591   Node*         ctrl   = in(MemNode::Control);
 592   Node*         adr    = in(MemNode::Address);
 593   intptr_t      offset = 0;
 594   Node*         base   = AddPNode::Ideal_base_and_offset(adr, phase, offset);
 595   AllocateNode* alloc  = AllocateNode::Ideal_allocation(base, phase);
 596 
 597   if (offset == Type::OffsetBot)
 598     return NULL;            // cannot unalias unless there are precise offsets
 599 
 600   const bool adr_maybe_raw = check_if_adr_maybe_raw(adr);
 601   const TypeOopPtr *addr_t = adr->bottom_type()->isa_oopptr();
 602 
 603   intptr_t size_in_bytes = memory_size();
 604 
 605   Node* mem = in(MemNode::Memory);   // start searching here...
 606 
 607   int cnt = 50;             // Cycle limiter
 608   for (;;) {                // While we can dance past unrelated stores...
 609     if (--cnt < 0)  break;  // Caught in cycle or a complicated dance?
 610 
 611     Node* prev = mem;
 612     if (mem->is_Store()) {
 613       Node* st_adr = mem->in(MemNode::Address);
 614       intptr_t st_offset = 0;
 615       Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_offset);
 616       if (st_base == NULL)
 617         break;              // inscrutable pointer
 618 
 619       // For raw accesses it's not enough to prove that constant offsets don't intersect.
 620       // We need the bases to be the equal in order for the offset check to make sense.
 621       if ((adr_maybe_raw || check_if_adr_maybe_raw(st_adr)) && st_base != base) {
 622         break;
 623       }
 624 
 625       if (st_offset != offset && st_offset != Type::OffsetBot) {
 626         const int MAX_STORE = BytesPerLong;
 627         if (st_offset >= offset + size_in_bytes ||
 628             st_offset <= offset - MAX_STORE ||
 629             st_offset <= offset - mem->as_Store()->memory_size()) {
 630           // Success:  The offsets are provably independent.
 631           // (You may ask, why not just test st_offset != offset and be done?
 632           // The answer is that stores of different sizes can co-exist
 633           // in the same sequence of RawMem effects.  We sometimes initialize
 634           // a whole 'tile' of array elements with a single jint or jlong.)
 635           mem = mem->in(MemNode::Memory);
 636           continue;           // (a) advance through independent store memory
 637         }
 638       }
 639       if (st_base != base &&
 640           detect_ptr_independence(base, alloc,
 641                                   st_base,
 642                                   AllocateNode::Ideal_allocation(st_base, phase),
 643                                   phase)) {
 644         // Success:  The bases are provably independent.
 645         mem = mem->in(MemNode::Memory);
 646         continue;           // (a) advance through independent store memory
 647       }
 648 
 649       // (b) At this point, if the bases or offsets do not agree, we lose,
 650       // since we have not managed to prove 'this' and 'mem' independent.
 651       if (st_base == base && st_offset == offset) {
 652         return mem;         // let caller handle steps (c), (d)
 653       }
 654 
 655     } else if (mem->is_Proj() && mem->in(0)->is_Initialize()) {
 656       InitializeNode* st_init = mem->in(0)->as_Initialize();
 657       AllocateNode*  st_alloc = st_init->allocation();
 658       if (st_alloc == NULL)
 659         break;              // something degenerated
 660       bool known_identical = false;
 661       bool known_independent = false;
 662       if (alloc == st_alloc)
 663         known_identical = true;
 664       else if (alloc != NULL)
 665         known_independent = true;
 666       else if (all_controls_dominate(this, st_alloc))
 667         known_independent = true;
 668 
 669       if (known_independent) {
 670         // The bases are provably independent: Either they are
 671         // manifestly distinct allocations, or else the control
 672         // of this load dominates the store's allocation.
 673         int alias_idx = phase->C->get_alias_index(adr_type());
 674         if (alias_idx == Compile::AliasIdxRaw) {
 675           mem = st_alloc->in(TypeFunc::Memory);
 676         } else {
 677           mem = st_init->memory(alias_idx);
 678         }
 679         continue;           // (a) advance through independent store memory
 680       }
 681 
 682       // (b) at this point, if we are not looking at a store initializing
 683       // the same allocation we are loading from, we lose.
 684       if (known_identical) {
 685         // From caller, can_see_stored_value will consult find_captured_store.
 686         return mem;         // let caller handle steps (c), (d)
 687       }
 688 
 689     } else if (find_previous_arraycopy(phase, alloc, mem, false) != NULL) {
 690       if (prev != mem) {
 691         // Found an arraycopy but it doesn't affect that load
 692         continue;
 693       }
 694       // Found an arraycopy that may affect that load
 695       return mem;
 696     } else if (addr_t != NULL && addr_t->is_known_instance_field()) {
 697       // Can't use optimize_simple_memory_chain() since it needs PhaseGVN.
 698       if (mem->is_Proj() && mem->in(0)->is_Call()) {
 699         // ArrayCopyNodes processed here as well.
 700         CallNode *call = mem->in(0)->as_Call();
 701         if (!call->may_modify(addr_t, phase)) {
 702           mem = call->in(TypeFunc::Memory);
 703           continue;         // (a) advance through independent call memory
 704         }
 705       } else if (mem->is_Proj() && mem->in(0)->is_MemBar()) {
 706         ArrayCopyNode* ac = NULL;
 707         if (ArrayCopyNode::may_modify(addr_t, mem->in(0)->as_MemBar(), phase, ac)) {
 708           break;
 709         }
 710         mem = mem->in(0)->in(TypeFunc::Memory);
 711         continue;           // (a) advance through independent MemBar memory
 712       } else if (mem->is_ClearArray()) {
 713         if (ClearArrayNode::step_through(&mem, (uint)addr_t->instance_id(), phase)) {
 714           // (the call updated 'mem' value)
 715           continue;         // (a) advance through independent allocation memory
 716         } else {
 717           // Can not bypass initialization of the instance
 718           // we are looking for.
 719           return mem;
 720         }
 721       } else if (mem->is_MergeMem()) {
 722         int alias_idx = phase->C->get_alias_index(adr_type());
 723         mem = mem->as_MergeMem()->memory_at(alias_idx);
 724         continue;           // (a) advance through independent MergeMem memory
 725       }
 726     }
 727 
 728     // Unless there is an explicit 'continue', we must bail out here,
 729     // because 'mem' is an inscrutable memory state (e.g., a call).
 730     break;
 731   }
 732 
 733   return NULL;              // bail out
 734 }

 736 //----------------------calculate_adr_type-------------------------------------
 737 // Helper function.  Notices when the given type of address hits top or bottom.
 738 // Also, asserts a cross-check of the type against the expected address type.
 739 const TypePtr* MemNode::calculate_adr_type(const Type* t, const TypePtr* cross_check) {
 740   if (t == Type::TOP)  return NULL; // does not touch memory any more?
 741   #ifdef ASSERT
 742   if (!VerifyAliases || VMError::is_error_reported() || Node::in_dump())  cross_check = NULL;
 743   #endif
 744   const TypePtr* tp = t->isa_ptr();
 745   if (tp == NULL) {
 746     assert(cross_check == NULL || cross_check == TypePtr::BOTTOM, "expected memory type must be wide");
 747     return TypePtr::BOTTOM;           // touches lots of memory
 748   } else {
 749     #ifdef ASSERT
 750     // %%%% [phh] We don't check the alias index if cross_check is
 751     //            TypeRawPtr::BOTTOM.  Needs to be investigated.
 752     if (cross_check != NULL &&
 753         cross_check != TypePtr::BOTTOM &&
 754         cross_check != TypeRawPtr::BOTTOM) {
 755       // Recheck the alias index, to see if it has changed (due to a bug).

 764              "real address must not escape from expected memory type");
 765     }
 766     #endif
 767     return tp;
 768   }

 927 // Is the value loaded previously stored by an arraycopy? If so return
 928 // a load node that reads from the source array so we may be able to
 929 // optimize out the ArrayCopy node later.
 930 Node* LoadNode::can_see_arraycopy_value(Node* st, PhaseGVN* phase) const {
 931   Node* ld_adr = in(MemNode::Address);
 932   intptr_t ld_off = 0;
 933   AllocateNode* ld_alloc = AllocateNode::Ideal_allocation(ld_adr, phase, ld_off);
 934   Node* ac = find_previous_arraycopy(phase, ld_alloc, st, true);
 935   if (ac != NULL) {
 936     assert(ac->is_ArrayCopy(), "what kind of node can this be?");
 937 
 938     Node* mem = ac->in(TypeFunc::Memory);
 939     Node* ctl = ac->in(0);
 940     Node* src = ac->in(ArrayCopyNode::Src);
 941 
 942     if (!ac->as_ArrayCopy()->is_clonebasic() && !phase->type(src)->isa_aryptr()) {
 943       return NULL;
 944     }
 945 
 946     LoadNode* ld = clone()->as_Load();
 947     Node* addp = in(MemNode::Address)->clone();
 948     if (ac->as_ArrayCopy()->is_clonebasic()) {
 949       assert(ld_alloc != NULL, "need an alloc");
 950       assert(addp->is_AddP(), "address must be addp");

 952       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 953       assert(bs->step_over_gc_barrier(addp->in(AddPNode::Base)) == bs->step_over_gc_barrier(ac->in(ArrayCopyNode::Dest)->in(AddPNode::Base)), "strange pattern");
 954       assert(bs->step_over_gc_barrier(addp->in(AddPNode::Address)) == bs->step_over_gc_barrier(ac->in(ArrayCopyNode::Dest)->in(AddPNode::Address)), "strange pattern");
 955       addp->set_req(AddPNode::Base, src->in(AddPNode::Base));
 956       addp->set_req(AddPNode::Address, src->in(AddPNode::Address));
 957     } else {
 958       assert(ac->as_ArrayCopy()->is_arraycopy_validated() ||
 959              ac->as_ArrayCopy()->is_copyof_validated() ||
 960              ac->as_ArrayCopy()->is_copyofrange_validated(), "only supported cases");
 961       assert(addp->in(AddPNode::Base) == addp->in(AddPNode::Address), "should be");
 962       addp->set_req(AddPNode::Base, src);
 963       addp->set_req(AddPNode::Address, src);
 964 
 965       const TypeAryPtr* ary_t = phase->type(in(MemNode::Address))->isa_aryptr();
 966       BasicType ary_elem  = ary_t->klass()->as_array_klass()->element_type()->basic_type();
 967       uint header = arrayOopDesc::base_offset_in_bytes(ary_elem);
 968       uint shift  = exact_log2(type2aelembytes(ary_elem));
 969       if (ary_t->klass()->is_value_array_klass()) {
 970         ciValueArrayKlass* vak = ary_t->klass()->as_value_array_klass();
 971         shift = vak->log2_element_size();
 972       }
 973 
 974       Node* diff = phase->transform(new SubINode(ac->in(ArrayCopyNode::SrcPos), ac->in(ArrayCopyNode::DestPos)));
 975 #ifdef _LP64
 976       diff = phase->transform(new ConvI2LNode(diff));
 977 #endif
 978       diff = phase->transform(new LShiftXNode(diff, phase->intcon(shift)));
 979 
 980       Node* offset = phase->transform(new AddXNode(addp->in(AddPNode::Offset), diff));
 981       addp->set_req(AddPNode::Offset, offset);
 982     }
 983     addp = phase->transform(addp);
 984 #ifdef ASSERT
 985     const TypePtr* adr_type = phase->type(addp)->is_ptr();
 986     ld->_adr_type = adr_type;
 987 #endif
 988     ld->set_req(MemNode::Address, addp);
 989     ld->set_req(0, ctl);
 990     ld->set_req(MemNode::Memory, mem);
 991     // load depends on the tests that validate the arraycopy
 992     ld->_control_dependency = UnknownControl;
 993     return ld;
 994   }
 995   return NULL;
 996 }

 999 //---------------------------can_see_stored_value------------------------------
1000 // This routine exists to make sure this set of tests is done the same
1001 // everywhere.  We need to make a coordinated change: first LoadNode::Ideal
1002 // will change the graph shape in a way which makes memory alive twice at the
1003 // same time (uses the Oracle model of aliasing), then some
1004 // LoadXNode::Identity will fold things back to the equivalence-class model
1005 // of aliasing.
1006 Node* MemNode::can_see_stored_value(Node* st, PhaseTransform* phase) const {
1007   Node* ld_adr = in(MemNode::Address);
1008   intptr_t ld_off = 0;
1009   Node* ld_base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ld_off);
1010   Node* ld_alloc = AllocateNode::Ideal_allocation(ld_base, phase);
1011   const TypeInstPtr* tp = phase->type(ld_adr)->isa_instptr();
1012   Compile::AliasType* atp = (tp != NULL) ? phase->C->alias_type(tp) : NULL;
1013   // This is more general than load from boxing objects.
1014   if (skip_through_membars(atp, tp, phase->C->eliminate_boxing())) {
1015     uint alias_idx = atp->index();
1016     bool final = !atp->is_rewritable();
1017     Node* result = NULL;
1018     Node* current = st;
1019     // Skip through chains of MemBarNodes checking the MergeMems for
1020     // new states for the slice of this load.  Stop once any other
1021     // kind of node is encountered.  Loads from final memory can skip
1022     // through any kind of MemBar but normal loads shouldn't skip
1023     // through MemBarAcquire since the could allow them to move out of
1024     // a synchronized region.
1025     while (current->is_Proj()) {
1026       int opc = current->in(0)->Opcode();
1027       if ((final && (opc == Op_MemBarAcquire ||
1028                      opc == Op_MemBarAcquireLock ||
1029                      opc == Op_LoadFence)) ||
1030           opc == Op_MemBarRelease ||
1031           opc == Op_StoreFence ||
1032           opc == Op_MemBarReleaseLock ||
1033           opc == Op_MemBarStoreStore ||
1034           opc == Op_MemBarCPUOrder) {
1035         Node* mem = current->in(0)->in(TypeFunc::Memory);
1036         if (mem->is_MergeMem()) {
1037           MergeMemNode* merge = mem->as_MergeMem();
1038           Node* new_st = merge->memory_at(alias_idx);
1039           if (new_st == merge->base_memory()) {
1040             // Keep searching
1041             current = new_st;
1042             continue;
1043           }
1044           // Save the new memory state for the slice and fall through
1045           // to exit.
1046           result = new_st;
1047         }
1048       }
1049       break;
1050     }
1051     if (result != NULL) {
1052       st = result;
1053     }
1054   }
1055 
1056   // Loop around twice in the case Load -> Initialize -> Store.
1057   // (See PhaseIterGVN::add_users_to_worklist, which knows about this case.)
1058   for (int trip = 0; trip <= 1; trip++) {
1059 
1060     if (st->is_Store()) {
1061       Node* st_adr = st->in(MemNode::Address);
1062       if (!phase->eqv(st_adr, ld_adr)) {
1063         // Try harder before giving up. Unify base pointers with casts (e.g., raw/non-raw pointers).
1064         intptr_t st_off = 0;
1065         Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_off);
1066         if (ld_base == NULL)                                   return NULL;
1067         if (st_base == NULL)                                   return NULL;
1068         if (!ld_base->eqv_uncast(st_base, /*keep_deps=*/true)) return NULL;
1069         if (ld_off != st_off)                                  return NULL;
1070         if (ld_off == Type::OffsetBot)                         return NULL;
1071         // Same base, same offset.
1072         // Possible improvement for arrays: check index value instead of absolute offset.
1073 
1074         // At this point we have proven something like this setup:
1075         //   B = << base >>
1076         //   L =  LoadQ(AddP(Check/CastPP(B), #Off))
1077         //   S = StoreQ(AddP(             B , #Off), V)
1078         // (Actually, we haven't yet proven the Q's are the same.)
1079         // In other words, we are loading from a casted version of
1080         // the same pointer-and-offset that we stored to.
1081         // Casted version may carry a dependency and it is respected.
1082         // Thus, we are able to replace L by V.
1083       }
1084       // Now prove that we have a LoadQ matched to a StoreQ, for some Q.
1085       if (store_Opcode() != st->Opcode())
1086         return NULL;
1087       return st->in(MemNode::ValueIn);
1088     }
1089 
1090     // A load from a freshly-created object always returns zero.
1091     // (This can happen after LoadNode::Ideal resets the load's memory input
1092     // to find_captured_store, which returned InitializeNode::zero_memory.)
1093     if (st->is_Proj() && st->in(0)->is_Allocate() &&
1094         (st->in(0) == ld_alloc) &&
1095         (ld_off >= st->in(0)->as_Allocate()->minimum_header_size())) {
1096       // return a zero value for the load's basic type
1097       // (This is one of the few places where a generic PhaseTransform
1098       // can create new nodes.  Think of it as lazily manifesting
1099       // virtually pre-existing constants.)
1100       assert(memory_type() != T_VALUETYPE, "should not be used for value types");
1101       Node* default_value = ld_alloc->in(AllocateNode::DefaultValue);
1102       if (default_value != NULL) {
1103         return default_value;
1104       }
1105       assert(ld_alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
1106       return phase->zerocon(memory_type());
1107     }
1108 
1109     // A load from an initialization barrier can match a captured store.
1110     if (st->is_Proj() && st->in(0)->is_Initialize()) {
1111       InitializeNode* init = st->in(0)->as_Initialize();
1112       AllocateNode* alloc = init->allocation();
1113       if ((alloc != NULL) && (alloc == ld_alloc)) {
1114         // examine a captured store value
1115         st = init->find_captured_store(ld_off, memory_size(), phase);
1116         if (st != NULL) {
1117           continue;             // take one more trip around
1118         }
1119       }
1120     }
1121 
1122     // Load boxed value from result of valueOf() call is input parameter.
1123     if (this->is_Load() && ld_adr->is_AddP() &&
1124         (tp != NULL) && tp->is_ptr_to_boxed_value()) {
1125       intptr_t ignore = 0;
1126       Node* base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ignore);
1127       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1128       base = bs->step_over_gc_barrier(base);
1129       if (base != NULL && base->is_Proj() &&
1130           base->as_Proj()->_con == TypeFunc::Parms &&
1131           base->in(0)->is_CallStaticJava() &&
1132           base->in(0)->as_CallStaticJava()->is_boxing_method()) {
1133         return base->in(0)->in(TypeFunc::Parms);
1134       }
1135     }
1136 
1137     break;
1138   }
1139 
1140   return NULL;
1141 }

1160 //------------------------------Identity---------------------------------------
1161 // Loads are identity if previous store is to same address
1162 Node* LoadNode::Identity(PhaseGVN* phase) {
1163   // Loading from a ValueTypePtr? The ValueTypePtr has the values of
1164   // all fields as input. Look for the field with matching offset.
1165   Node* addr = in(Address);
1166   intptr_t offset;
1167   Node* base = AddPNode::Ideal_base_and_offset(addr, phase, offset);
1168   if (base != NULL && base->is_ValueTypePtr() && offset > oopDesc::klass_offset_in_bytes()) {
1169     Node* value = base->as_ValueTypePtr()->field_value_by_offset((int)offset, true);
1170     if (value->is_ValueType()) {
1171       // Non-flattened value type field
1172       ValueTypeNode* vt = value->as_ValueType();
1173       if (vt->is_allocated(phase)) {
1174         value = vt->get_oop();
1175       } else {
1176         // Not yet allocated, bail out
1177         value = NULL;
1178       }
1179     }
1180     if (value != NULL) {
1181       if (Opcode() == Op_LoadN) {
1182         // Encode oop value if we are loading a narrow oop
1183         assert(!phase->type(value)->isa_narrowoop(), "should already be decoded");
1184         value = phase->transform(new EncodePNode(value, bottom_type()));
1185       }
1186       return value;
1187     }
1188   }
1189 
1190   // If the previous store-maker is the right kind of Store, and the store is
1191   // to the same address, then we are equal to the value stored.
1192   Node* mem = in(Memory);
1193   Node* value = can_see_stored_value(mem, phase);
1194   if( value ) {
1195     // byte, short & char stores truncate naturally.
1196     // A load has to load the truncated value which requires
1197     // some sort of masking operation and that requires an
1198     // Ideal call instead of an Identity call.
1199     if (memory_size() < BytesPerInt) {
1200       // If the input to the store does not fit with the load's result type,
1201       // it must be truncated via an Ideal call.
1202       if (!phase->type(value)->higher_equal(phase->type(this)))
1203         return this;
1204     }
1205     // (This works even when value is a Con, but LoadNode::Value
1206     // usually runs first, producing the singleton type of the Con.)
1207     return value;
1208   }
1209 
1210   // Search for an existing data phi which was generated before for the same
1211   // instance's field to avoid infinite generation of phis in a loop.
1212   Node *region = mem->in(0);
1213   if (is_instance_field_load_with_local_phi(region)) {
1214     const TypeOopPtr *addr_t = in(Address)->bottom_type()->isa_oopptr();
1215     int this_index  = phase->C->get_alias_index(addr_t);
1216     int this_offset = addr_t->offset();
1217     int this_iid    = addr_t->instance_id();
1218     if (!addr_t->is_known_instance() &&
1219          addr_t->is_ptr_to_boxed_value()) {
1220       // Use _idx of address base (could be Phi node) for boxed values.
1221       intptr_t   ignore = 0;
1222       Node*      base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1223       if (base == NULL) {
1224         return this;
1225       }
1226       this_iid = base->_idx;
1227     }
1228     const Type* this_type = bottom_type();
1229     for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
1230       Node* phi = region->fast_out(i);
1231       if (phi->is_Phi() && phi != mem &&
1232           phi->as_Phi()->is_same_inst_field(this_type, (int)mem->_idx, this_iid, this_index, this_offset)) {
1233         return phi;
1234       }
1235     }
1236   }
1237 
1238   return this;
1239 }

1279 // We're loading from an object which has autobox behaviour.
1280 // If this object is result of a valueOf call we'll have a phi
1281 // merging a newly allocated object and a load from the cache.
1282 // We want to replace this load with the original incoming
1283 // argument to the valueOf call.
1284 Node* LoadNode::eliminate_autobox(PhaseGVN* phase) {
1285   assert(phase->C->eliminate_boxing(), "sanity");
1286   intptr_t ignore = 0;
1287   Node* base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1288   if ((base == NULL) || base->is_Phi()) {
1289     // Push the loads from the phi that comes from valueOf up
1290     // through it to allow elimination of the loads and the recovery
1291     // of the original value. It is done in split_through_phi().
1292     return NULL;
1293   } else if (base->is_Load() ||
1294              (base->is_DecodeN() && base->in(1)->is_Load())) {
1295     // Eliminate the load of boxed value for integer types from the cache
1296     // array by deriving the value from the index into the array.
1297     // Capture the offset of the load and then reverse the computation.
1298 
1299     // Get LoadN node which loads a boxing object from 'cache' array.
1300     if (base->is_DecodeN()) {
1301       base = base->in(1);
1302     }
1303     if (!base->in(Address)->is_AddP()) {
1304       return NULL; // Complex address
1305     }
1306     AddPNode* address = base->in(Address)->as_AddP();
1307     Node* cache_base = address->in(AddPNode::Base);
1308     if ((cache_base != NULL) && cache_base->is_DecodeN()) {
1309       // Get ConP node which is static 'cache' field.
1310       cache_base = cache_base->in(1);
1311     }
1312     if ((cache_base != NULL) && cache_base->is_Con()) {
1313       const TypeAryPtr* base_type = cache_base->bottom_type()->isa_aryptr();
1314       if ((base_type != NULL) && base_type->is_autobox_cache()) {
1315         Node* elements[4];
1316         int shift = exact_log2(type2aelembytes(T_OBJECT));
1317         int count = address->unpack_offsets(elements, ARRAY_SIZE(elements));
1318         if (count > 0 && elements[0]->is_Con() &&
1319             (count == 1 ||
1320              (count == 2 && elements[1]->Opcode() == Op_LShiftX &&
1321                             elements[1]->in(2) == phase->intcon(shift)))) {
1322           ciObjArray* array = base_type->const_oop()->as_obj_array();
1323           // Fetch the box object cache[0] at the base of the array and get its value
1324           ciInstance* box = array->obj_at(0)->as_instance();
1325           ciInstanceKlass* ik = box->klass()->as_instance_klass();
1326           assert(ik->is_box_klass(), "sanity");
1327           assert(ik->nof_nonstatic_fields() == 1, "change following code");
1328           if (ik->nof_nonstatic_fields() == 1) {
1329             // This should be true nonstatic_field_at requires calling
1330             // nof_nonstatic_fields so check it anyway
1331             ciConstant c = box->field_value(ik->nonstatic_field_at(0));
1332             BasicType bt = c.basic_type();
1333             // Only integer types have boxing cache.
1334             assert(bt == T_BOOLEAN || bt == T_CHAR  ||
1335                    bt == T_BYTE    || bt == T_SHORT ||
1336                    bt == T_INT     || bt == T_LONG, "wrong type = %s", type2name(bt));
1337             jlong cache_low = (bt == T_LONG) ? c.as_long() : c.as_int();
1338             if (cache_low != (int)cache_low) {
1339               return NULL; // should not happen since cache is array indexed by value
1340             }
1341             jlong offset = arrayOopDesc::base_offset_in_bytes(T_OBJECT) - (cache_low << shift);
1342             if (offset != (int)offset) {
1343               return NULL; // should not happen since cache is array indexed by value
1344             }
1345            // Add up all the offsets making of the address of the load
1346             Node* result = elements[0];
1347             for (int i = 1; i < count; i++) {
1348               result = phase->transform(new AddXNode(result, elements[i]));
1349             }
1350             // Remove the constant offset from the address and then
1351             result = phase->transform(new AddXNode(result, phase->MakeConX(-(int)offset)));
1352             // remove the scaling of the offset to recover the original index.
1353             if (result->Opcode() == Op_LShiftX && result->in(2) == phase->intcon(shift)) {
1354               // Peel the shift off directly but wrap it in a dummy node
1355               // since Ideal can't return existing nodes
1356               result = new RShiftXNode(result->in(1), phase->intcon(0));
1357             } else if (result->is_Add() && result->in(2)->is_Con() &&
1358                        result->in(1)->Opcode() == Op_LShiftX &&
1359                        result->in(1)->in(2) == phase->intcon(shift)) {
1360               // We can't do general optimization: ((X<<Z) + Y) >> Z ==> X + (Y>>Z)
1361               // but for boxing cache access we know that X<<Z will not overflow
1362               // (there is range check) so we do this optimizatrion by hand here.
1363               Node* add_con = new RShiftXNode(result->in(2), phase->intcon(shift));
1364               result = new AddXNode(result->in(1)->in(1), phase->transform(add_con));
1365             } else {
1366               result = new RShiftXNode(result, phase->intcon(shift));
1367             }
1368 #ifdef _LP64
1369             if (bt != T_LONG) {
1370               result = new ConvL2INode(phase->transform(result));
1371             }
1372 #else
1373             if (bt == T_LONG) {
1374               result = new ConvI2LNode(phase->transform(result));
1375             }
1376 #endif
1377             // Boxing/unboxing can be done from signed & unsigned loads (e.g. LoadUB -> ... -> LoadB pair).
1378             // Need to preserve unboxing load type if it is unsigned.
1379             switch(this->Opcode()) {
1380               case Op_LoadUB:
1381                 result = new AndINode(phase->transform(result), phase->intcon(0xFF));
1382                 break;
1383               case Op_LoadUS:
1384                 result = new AndINode(phase->transform(result), phase->intcon(0xFFFF));
1385                 break;
1386             }
1387             return result;
1388           }
1389         }
1390       }
1391     }
1392   }
1393   return NULL;
1394 }

1396 static bool stable_phi(PhiNode* phi, PhaseGVN *phase) {
1397   Node* region = phi->in(0);
1398   if (region == NULL) {
1399     return false; // Wait stable graph
1400   }
1401   uint cnt = phi->req();
1402   for (uint i = 1; i < cnt; i++) {
1403     Node* rc = region->in(i);
1404     if (rc == NULL || phase->type(rc) == Type::TOP)
1405       return false; // Wait stable graph
1406     Node* in = phi->in(i);
1407     if (in == NULL || phase->type(in) == Type::TOP)
1408       return false; // Wait stable graph
1409   }
1410   return true;

1412 //------------------------------split_through_phi------------------------------
1413 // Split instance or boxed field load through Phi.
1414 Node *LoadNode::split_through_phi(PhaseGVN *phase) {
1415   Node* mem     = in(Memory);
1416   Node* address = in(Address);
1417   const TypeOopPtr *t_oop = phase->type(address)->isa_oopptr();
1418 
1419   assert((t_oop != NULL) &&
1420          (t_oop->is_known_instance_field() ||
1421           t_oop->is_ptr_to_boxed_value()), "invalide conditions");
1422 
1423   Compile* C = phase->C;
1424   intptr_t ignore = 0;
1425   Node*    base = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1426   bool base_is_phi = (base != NULL) && base->is_Phi();
1427   bool load_boxed_values = t_oop->is_ptr_to_boxed_value() && C->aggressive_unboxing() &&
1428                            (base != NULL) && (base == address->in(AddPNode::Base)) &&
1429                            phase->type(base)->higher_equal(TypePtr::NOTNULL);
1430 
1431   if (!((mem->is_Phi() || base_is_phi) &&
1432         (load_boxed_values || t_oop->is_known_instance_field()))) {
1433     return NULL; // memory is not Phi
1434   }
1435 
1436   if (mem->is_Phi()) {
1437     if (!stable_phi(mem->as_Phi(), phase)) {
1438       return NULL; // Wait stable graph
1439     }
1440     uint cnt = mem->req();
1441     // Check for loop invariant memory.
1442     if (cnt == 3) {
1443       for (uint i = 1; i < cnt; i++) {
1444         Node* in = mem->in(i);
1445         Node*  m = optimize_memory_chain(in, t_oop, this, phase);
1446         if (m == mem) {
1447           if (i == 1) {
1448             // if the first edge was a loop, check second edge too.
1449             // If both are replaceable - we are in an infinite loop
1450             Node *n = optimize_memory_chain(mem->in(2), t_oop, this, phase);
1451             if (n == mem) {
1452               break;
1453             }
1454           }
1455           set_req(Memory, mem->in(cnt - i));
1456           return this; // made change
1457         }
1458       }
1459     }
1460   }
1461   if (base_is_phi) {
1462     if (!stable_phi(base->as_Phi(), phase)) {
1463       return NULL; // Wait stable graph
1464     }
1465     uint cnt = base->req();
1466     // Check for loop invariant memory.
1467     if (cnt == 3) {
1468       for (uint i = 1; i < cnt; i++) {
1469         if (base->in(i) == base) {
1470           return NULL; // Wait stable graph
1471         }
1472       }
1473     }
1474   }
1475 
1476   // Split through Phi (see original code in loopopts.cpp).
1477   assert(C->have_alias_type(t_oop), "instance should have alias type");
1478 
1479   // Do nothing here if Identity will find a value
1480   // (to avoid infinite chain of value phis generation).
1481   if (!phase->eqv(this, this->Identity(phase))) {
1482     return NULL;
1483   }
1484 
1485   // Select Region to split through.
1486   Node* region;
1487   if (!base_is_phi) {
1488     assert(mem->is_Phi(), "sanity");
1489     region = mem->in(0);
1490     // Skip if the region dominates some control edge of the address.
1491     if (!MemNode::all_controls_dominate(address, region))
1492       return NULL;
1493   } else if (!mem->is_Phi()) {
1494     assert(base_is_phi, "sanity");
1495     region = base->in(0);
1496     // Skip if the region dominates some control edge of the memory.
1497     if (!MemNode::all_controls_dominate(mem, region))
1498       return NULL;
1499   } else if (base->in(0) != mem->in(0)) {
1500     assert(base_is_phi && mem->is_Phi(), "sanity");
1501     if (MemNode::all_controls_dominate(mem, base->in(0))) {
1502       region = base->in(0);
1503     } else if (MemNode::all_controls_dominate(address, mem->in(0))) {
1504       region = mem->in(0);
1505     } else {
1506       return NULL; // complex graph
1507     }
1508   } else {
1509     assert(base->in(0) == mem->in(0), "sanity");
1510     region = mem->in(0);
1511   }
1512 
1513   const Type* this_type = this->bottom_type();
1514   int this_index  = C->get_alias_index(t_oop);
1515   int this_offset = t_oop->offset();
1516   int this_iid    = t_oop->instance_id();
1517   if (!t_oop->is_known_instance() && load_boxed_values) {
1518     // Use _idx of address base for boxed values.
1519     this_iid = base->_idx;
1520   }
1521   PhaseIterGVN* igvn = phase->is_IterGVN();
1522   Node* phi = new PhiNode(region, this_type, NULL, mem->_idx, this_iid, this_index, this_offset);
1523   for (uint i = 1; i < region->req(); i++) {
1524     Node* x;
1525     Node* the_clone = NULL;
1526     Node* in = region->in(i);
1527     if (region->is_CountedLoop() && region->as_Loop()->is_strip_mined() && i == LoopNode::EntryControl &&
1528         in != NULL && in->is_OuterStripMinedLoop()) {
1529       // No node should go in the outer strip mined loop
1530       in = in->in(LoopNode::EntryControl);
1531     }
1532     if (in == NULL || in == C->top()) {
1533       x = C->top();      // Dead path?  Use a dead data op
1534     } else {
1535       x = this->clone();        // Else clone up the data op
1536       the_clone = x;            // Remember for possible deletion.
1537       // Alter data node to use pre-phi inputs
1538       if (this->in(0) == region) {
1539         x->set_req(0, in);
1540       } else {
1541         x->set_req(0, NULL);
1542       }
1543       if (mem->is_Phi() && (mem->in(0) == region)) {
1544         x->set_req(Memory, mem->in(i)); // Use pre-Phi input for the clone.
1545       }
1546       if (address->is_Phi() && address->in(0) == region) {
1547         x->set_req(Address, address->in(i)); // Use pre-Phi input for the clone
1548       }
1549       if (base_is_phi && (base->in(0) == region)) {
1550         Node* base_x = base->in(i); // Clone address for loads from boxed objects.
1551         Node* adr_x = phase->transform(new AddPNode(base_x,base_x,address->in(AddPNode::Offset)));
1552         x->set_req(Address, adr_x);
1553       }
1554     }
1555     // Check for a 'win' on some paths
1556     const Type *t = x->Value(igvn);
1557 
1558     bool singleton = t->singleton();
1559 
1560     // See comments in PhaseIdealLoop::split_thru_phi().
1561     if (singleton && t == Type::TOP) {
1562       singleton &= region->is_Loop() && (i != LoopNode::EntryControl);
1563     }
1564 
1565     if (singleton) {
1566       x = igvn->makecon(t);
1567     } else {
1568       // We now call Identity to try to simplify the cloned node.
1569       // Note that some Identity methods call phase->type(this).
1570       // Make sure that the type array is big enough for
1571       // our new node, even though we may throw the node away.
1572       // (This tweaking with igvn only works because x is a new node.)
1573       igvn->set_type(x, t);
1574       // If x is a TypeNode, capture any more-precise type permanently into Node
1575       // otherwise it will be not updated during igvn->transform since
1576       // igvn->type(x) is set to x->Value() already.
1577       x->raise_bottom_type(t);
1578       Node* y = x->Identity(igvn);
1579       if (y != x) {
1580         x = y;
1581       } else {
1582         y = igvn->hash_find_insert(x);
1583         if (y) {
1584           x = y;
1585         } else {
1586           // Else x is a new node we are keeping
1587           // We do not need register_new_node_with_optimizer
1588           // because set_type has already been called.
1589           igvn->_worklist.push(x);
1590         }
1591       }
1592     }
1593     if (x != the_clone && the_clone != NULL) {
1594       igvn->remove_dead_node(the_clone);
1595     }
1596     phi->set_req(i, x);
1597   }
1598   // Record Phi
1599   igvn->register_new_node_with_optimizer(phi);
1600   return phi;
1601 }

1619 //------------------------------Ideal------------------------------------------
1620 // If the load is from Field memory and the pointer is non-null, it might be possible to
1621 // zero out the control input.
1622 // If the offset is constant and the base is an object allocation,
1623 // try to hook me up to the exact initializing store.
1624 Node *LoadNode::Ideal(PhaseGVN *phase, bool can_reshape) {
1625   Node* p = MemNode::Ideal_common(phase, can_reshape);
1626   if (p)  return (p == NodeSentinel) ? NULL : p;
1627 
1628   Node* ctrl    = in(MemNode::Control);
1629   Node* address = in(MemNode::Address);
1630   bool progress = false;
1631 
1632   bool addr_mark = ((phase->type(address)->isa_oopptr() || phase->type(address)->isa_narrowoop()) &&
1633          phase->type(address)->is_ptr()->offset() == oopDesc::mark_offset_in_bytes());
1634 
1635   // Skip up past a SafePoint control.  Cannot do this for Stores because
1636   // pointer stores & cardmarks must stay on the same side of a SafePoint.
1637   if( ctrl != NULL && ctrl->Opcode() == Op_SafePoint &&
1638       phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw  &&
1639       !addr_mark &&
1640       (depends_only_on_test() || has_unknown_control_dependency())) {
1641     ctrl = ctrl->in(0);
1642     set_req(MemNode::Control,ctrl);
1643     progress = true;
1644   }
1645 
1646   intptr_t ignore = 0;
1647   Node*    base   = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1648   if (base != NULL
1649       && phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw) {
1650     // Check for useless control edge in some common special cases
1651     if (in(MemNode::Control) != NULL
1652         && can_remove_control()
1653         && phase->type(base)->higher_equal(TypePtr::NOTNULL)
1654         && all_controls_dominate(base, phase->C->start())) {
1655       // A method-invariant, non-null address (constant or 'this' argument).
1656       set_req(MemNode::Control, NULL);
1657       progress = true;
1658     }
1659   }
1660 
1661   Node* mem = in(MemNode::Memory);
1662   const TypePtr *addr_t = phase->type(address)->isa_ptr();
1663 
1664   if (can_reshape && (addr_t != NULL)) {
1665     // try to optimize our memory input
1666     Node* opt_mem = MemNode::optimize_memory_chain(mem, addr_t, this, phase);
1667     if (opt_mem != mem) {
1668       set_req(MemNode::Memory, opt_mem);
1669       if (phase->type( opt_mem ) == Type::TOP) return NULL;
1670       return this;
1671     }
1672     const TypeOopPtr *t_oop = addr_t->isa_oopptr();
1673     if ((t_oop != NULL) &&
1674         (t_oop->is_known_instance_field() ||
1675          t_oop->is_ptr_to_boxed_value())) {
1676       PhaseIterGVN *igvn = phase->is_IterGVN();
1677       if (igvn != NULL && igvn->_worklist.member(opt_mem)) {
1678         // Delay this transformation until memory Phi is processed.
1679         phase->is_IterGVN()->_worklist.push(this);
1680         return NULL;
1681       }
1682       // Split instance field load through Phi.
1683       Node* result = split_through_phi(phase);
1684       if (result != NULL) return result;
1685 
1686       if (t_oop->is_ptr_to_boxed_value()) {
1687         Node* result = eliminate_autobox(phase);
1688         if (result != NULL) return result;
1689       }
1690     }
1691   }
1692 
1693   // Is there a dominating load that loads the same value?  Leave
1694   // anything that is not a load of a field/array element (like
1695   // barriers etc.) alone
1696   if (in(0) != NULL && !adr_type()->isa_rawptr() && can_reshape) {
1697     for (DUIterator_Fast imax, i = mem->fast_outs(imax); i < imax; i++) {
1698       Node *use = mem->fast_out(i);
1699       if (use != this &&
1700           use->Opcode() == Opcode() &&
1701           use->in(0) != NULL &&
1702           use->in(0) != in(0) &&
1703           use->in(Address) == in(Address)) {
1704         Node* ctl = in(0);
1705         for (int i = 0; i < 10 && ctl != NULL; i++) {
1706           ctl = IfNode::up_one_dom(ctl);
1707           if (ctl == use->in(0)) {
1708             set_req(0, use->in(0));
1709             return this;
1710           }
1711         }
1712       }
1713     }
1714   }
1715 
1716   // Check for prior store with a different base or offset; make Load
1717   // independent.  Skip through any number of them.  Bail out if the stores
1718   // are in an endless dead cycle and report no progress.  This is a key
1719   // transform for Reflection.  However, if after skipping through the Stores
1720   // we can't then fold up against a prior store do NOT do the transform as
1721   // this amounts to using the 'Oracle' model of aliasing.  It leaves the same
1722   // array memory alive twice: once for the hoisted Load and again after the
1723   // bypassed Store.  This situation only works if EVERYBODY who does
1724   // anti-dependence work knows how to bypass.  I.e. we need all
1725   // anti-dependence checks to ask the same Oracle.  Right now, that Oracle is
1726   // the alias index stuff.  So instead, peek through Stores and IFF we can
1727   // fold up, do so.
1728   Node* prev_mem = find_previous_store(phase);
1729   if (prev_mem != NULL) {
1730     Node* value = can_see_arraycopy_value(prev_mem, phase);
1731     if (value != NULL) {
1732       return value;
1733     }
1734   }
1735   // Steps (a), (b):  Walk past independent stores to find an exact match.
1736   if (prev_mem != NULL && prev_mem != in(MemNode::Memory)) {
1737     // (c) See if we can fold up on the spot, but don't fold up here.
1738     // Fold-up might require truncation (for LoadB/LoadS/LoadUS) or
1739     // just return a prior value, which is done by Identity calls.
1740     if (can_see_stored_value(prev_mem, phase)) {
1741       // Make ready for step (d):
1742       set_req(MemNode::Memory, prev_mem);
1743       return this;
1744     }
1745   }
1746 
1747   AllocateNode* alloc = AllocateNode::Ideal_allocation(address, phase);
1748   if (alloc != NULL && mem->is_Proj() &&
1749       mem->in(0) != NULL &&
1750       mem->in(0) == alloc->initialization() &&
1751       Opcode() == Op_LoadX &&
1752       alloc->initialization()->proj_out_or_null(0) != NULL) {
1753     InitializeNode* init = alloc->initialization();
1754     Node* control = init->proj_out(0);
1755     return alloc->make_ideal_mark(phase, control, mem);
1756   }
1757 
1758   return progress ? this : NULL;
1759 }

1761 // Helper to recognize certain Klass fields which are invariant across
1762 // some group of array types (e.g., int[] or all T[] where T < Object).
1763 const Type*
1764 LoadNode::load_array_final_field(const TypeKlassPtr *tkls,
1765                                  ciKlass* klass) const {
1766   if (tkls->offset() == in_bytes(Klass::modifier_flags_offset())) {
1767     // The field is Klass::_modifier_flags.  Return its (constant) value.
1768     // (Folds up the 2nd indirection in aClassConstant.getModifiers().)
1769     assert(this->Opcode() == Op_LoadI, "must load an int from _modifier_flags");
1770     return TypeInt::make(klass->modifier_flags());
1771   }
1772   if (tkls->offset() == in_bytes(Klass::access_flags_offset())) {
1773     // The field is Klass::_access_flags.  Return its (constant) value.
1774     // (Folds up the 2nd indirection in Reflection.getClassAccessFlags(aClassConstant).)
1775     assert(this->Opcode() == Op_LoadI, "must load an int from _access_flags");
1776     return TypeInt::make(klass->access_flags());
1777   }
1778   if (tkls->offset() == in_bytes(Klass::layout_helper_offset())) {
1779     // The field is Klass::_layout_helper.  Return its constant value if known.
1780     assert(this->Opcode() == Op_LoadI, "must load an int from _layout_helper");
1781     return TypeInt::make(klass->layout_helper());
1782   }
1783 
1784   // No match.
1785   return NULL;
1786 }

1788 //------------------------------Value-----------------------------------------
1789 const Type* LoadNode::Value(PhaseGVN* phase) const {
1790   // Either input is TOP ==> the result is TOP
1791   Node* mem = in(MemNode::Memory);
1792   const Type *t1 = phase->type(mem);
1793   if (t1 == Type::TOP)  return Type::TOP;
1794   Node* adr = in(MemNode::Address);
1795   const TypePtr* tp = phase->type(adr)->isa_ptr();
1796   if (tp == NULL || tp->empty())  return Type::TOP;
1797   int off = tp->offset();
1798   assert(off != Type::OffsetTop, "case covered by TypePtr::empty");
1799   Compile* C = phase->C;
1800 
1801   // Try to guess loaded type from pointer type
1802   if (tp->isa_aryptr()) {
1803     const TypeAryPtr* ary = tp->is_aryptr();
1804     const Type* t = ary->elem();
1805 
1806     // Determine whether the reference is beyond the header or not, by comparing
1807     // the offset against the offset of the start of the array's data.
1808     // Different array types begin at slightly different offsets (12 vs. 16).
1809     // We choose T_BYTE as an example base type that is least restrictive
1810     // as to alignment, which will therefore produce the smallest
1811     // possible base offset.
1812     const int min_base_off = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1813     const bool off_beyond_header = (off >= min_base_off);
1814 
1815     // Try to constant-fold a stable array element.
1816     if (FoldStableValues && !is_mismatched_access() && ary->is_stable()) {
1817       // Make sure the reference is not into the header and the offset is constant
1818       ciObject* aobj = ary->const_oop();
1819       if (aobj != NULL && off_beyond_header && adr->is_AddP() && off != Type::OffsetBot) {
1820         int stable_dimension = (ary->stable_dimension() > 0 ? ary->stable_dimension() - 1 : 0);
1821         const Type* con_type = Type::make_constant_from_array_element(aobj->as_array(), off,
1822                                                                       stable_dimension,
1823                                                                       memory_type(), is_unsigned());
1824         if (con_type != NULL) {
1825           return con_type;
1826         }
1827       }
1828     }
1829 
1830     // Don't do this for integer types. There is only potential profit if
1831     // the element type t is lower than _type; that is, for int types, if _type is
1832     // more restrictive than t.  This only happens here if one is short and the other
1833     // char (both 16 bits), and in those cases we've made an intentional decision
1834     // to use one kind of load over the other. See AndINode::Ideal and 4965907.
1835     // Also, do not try to narrow the type for a LoadKlass, regardless of offset.
1836     //
1837     // Yes, it is possible to encounter an expression like (LoadKlass p1:(AddP x x 8))
1838     // where the _gvn.type of the AddP is wider than 8.  This occurs when an earlier
1839     // copy p0 of (AddP x x 8) has been proven equal to p1, and the p0 has been
1840     // subsumed by p1.  If p1 is on the worklist but has not yet been re-transformed,
1841     // it is possible that p1 will have a type like Foo*[int+]:NotNull*+any.
1842     // In fact, that could have been the original type of p1, and p1 could have
1843     // had an original form like p1:(AddP x x (LShiftL quux 3)), where the
1844     // expression (LShiftL quux 3) independently optimized to the constant 8.
1845     if ((t->isa_int() == NULL) && (t->isa_long() == NULL)
1846         && (_type->isa_vect() == NULL)
1847         && t->isa_valuetype() == NULL
1848         && Opcode() != Op_LoadKlass && Opcode() != Op_LoadNKlass) {
1849       // t might actually be lower than _type, if _type is a unique
1850       // concrete subclass of abstract class t.
1851       if (off_beyond_header || off == Type::OffsetBot) {  // is the offset beyond the header?
1852         const Type* jt = t->join_speculative(_type);
1853         // In any case, do not allow the join, per se, to empty out the type.
1854         if (jt->empty() && !t->empty()) {
1855           // This can happen if a interface-typed array narrows to a class type.
1856           jt = _type;
1857         }
1858 #ifdef ASSERT
1859         if (phase->C->eliminate_boxing() && adr->is_AddP()) {
1860           // The pointers in the autobox arrays are always non-null
1861           Node* base = adr->in(AddPNode::Base);
1862           if ((base != NULL) && base->is_DecodeN()) {
1863             // Get LoadN node which loads IntegerCache.cache field
1864             base = base->in(1);
1865           }
1866           if ((base != NULL) && base->is_Con()) {
1867             const TypeAryPtr* base_type = base->bottom_type()->isa_aryptr();
1868             if ((base_type != NULL) && base_type->is_autobox_cache()) {
1869               // It could be narrow oop
1870               assert(jt->make_ptr()->ptr() == TypePtr::NotNull,"sanity");
1871             }
1872           }
1873         }
1874 #endif
1875         return jt;
1876       }
1877     }
1878   } else if (tp->base() == Type::InstPtr) {
1879     assert( off != Type::OffsetBot ||
1880             // arrays can be cast to Objects
1881             tp->is_oopptr()->klass()->is_java_lang_Object() ||
1882             tp->is_oopptr()->klass() == ciEnv::current()->Class_klass() ||
1883             // unsafe field access may not have a constant offset
1884             C->has_unsafe_access(),
1885             "Field accesses must be precise" );
1886     // For oop loads, we expect the _type to be precise.
1887 
1888     const TypeInstPtr* tinst = tp->is_instptr();
1889     BasicType bt = memory_type();
1890 
1891     // Fold component and value mirror loads
1892     ciInstanceKlass* ik = tinst->klass()->as_instance_klass();
1893     if (ik == phase->C->env()->Class_klass() && (off == java_lang_Class::component_mirror_offset_in_bytes() ||
1894                                                  off == java_lang_Class::inline_mirror_offset_in_bytes())) {
1895       ciType* mirror_type = tinst->java_mirror_type();
1896       if (mirror_type != NULL) {
1897         const Type* const_oop = TypePtr::NULL_PTR;
1898         if (mirror_type->is_array_klass()) {
1899           const_oop = TypeInstPtr::make(mirror_type->as_array_klass()->component_mirror_instance());
1900         } else if (mirror_type->is_valuetype()) {
1901           const_oop = TypeInstPtr::make(mirror_type->as_value_klass()->inline_mirror_instance());
1902         }
1903         return (bt == T_NARROWOOP) ? const_oop->make_narrowoop() : const_oop;
1904       }
1905     }
1906 
1907     // Optimize loads from constant fields.
1908     ciObject* const_oop = tinst->const_oop();
1909     if (!is_mismatched_access() && off != Type::OffsetBot && const_oop != NULL && const_oop->is_instance()) {
1910       ciType* mirror_type = const_oop->as_instance()->java_mirror_type();
1911       if (mirror_type != NULL && mirror_type->is_valuetype()) {
1912         ciValueKlass* vk = mirror_type->as_value_klass();
1913         if (off == vk->default_value_offset()) {
1914           // Loading a special hidden field that contains the oop of the default value type
1915           const Type* const_oop = TypeInstPtr::make(vk->default_value_instance());
1916           return (bt == T_NARROWOOP) ? const_oop->make_narrowoop() : const_oop;
1917         }
1918       }
1919       const Type* con_type = Type::make_constant_from_field(const_oop->as_instance(), off, is_unsigned(), bt);
1920       if (con_type != NULL) {
1921         return con_type;
1922       }
1923     }
1924   } else if (tp->base() == Type::KlassPtr) {
1925     assert( off != Type::OffsetBot ||
1926             // arrays can be cast to Objects
1927             tp->is_klassptr()->klass() == NULL ||
1928             tp->is_klassptr()->klass()->is_java_lang_Object() ||
1929             // also allow array-loading from the primary supertype
1930             // array during subtype checks
1931             Opcode() == Op_LoadKlass,
1932             "Field accesses must be precise" );
1933     // For klass/static loads, we expect the _type to be precise
1934   } else if (tp->base() == Type::RawPtr && !StressReflectiveCode) {
1935     if (adr->is_Load() && off == 0) {
1936       /* With mirrors being an indirect in the Klass*
1937        * the VM is now using two loads. LoadKlass(LoadP(LoadP(Klass, mirror_offset), zero_offset))
1938        * The LoadP from the Klass has a RawPtr type (see LibraryCallKit::load_mirror_from_klass).
1939        *
1940        * So check the type and klass of the node before the LoadP.
1941        */
1942       Node* adr2 = adr->in(MemNode::Address);
1943       const TypeKlassPtr* tkls = phase->type(adr2)->isa_klassptr();
1944       if (tkls != NULL) {
1945         ciKlass* klass = tkls->klass();
1946         if (klass != NULL && klass->is_loaded() && tkls->klass_is_exact() && tkls->offset() == in_bytes(Klass::java_mirror_offset())) {
1947           assert(adr->Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1948           assert(Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1949           return TypeInstPtr::make(klass->java_mirror());
1950         }
1951       }
1952     } else {
1953       // Check for a load of the default value offset from the ValueKlassFixedBlock:
1954       // LoadI(LoadP(value_klass, adr_valueklass_fixed_block_offset), default_value_offset_offset)
1955       intptr_t offset = 0;
1956       Node* base = AddPNode::Ideal_base_and_offset(adr, phase, offset);
1957       if (base != NULL && base->is_Load() && offset == in_bytes(ValueKlass::default_value_offset_offset())) {
1958         const TypeKlassPtr* tkls = phase->type(base->in(MemNode::Address))->isa_klassptr();
1959         if (tkls != NULL && tkls->is_loaded() && tkls->klass_is_exact() && tkls->isa_valuetype() &&
1960             tkls->offset() == in_bytes(InstanceKlass::adr_valueklass_fixed_block_offset())) {
1961           assert(base->Opcode() == Op_LoadP, "must load an oop from klass");
1962           assert(Opcode() == Op_LoadI, "must load an int from fixed block");
1963           return TypeInt::make(tkls->klass()->as_value_klass()->default_value_offset());
1964         }
1965       }
1966     }
1967   }
1968 
1969   const TypeKlassPtr *tkls = tp->isa_klassptr();
1970   if (tkls != NULL && !StressReflectiveCode) {
1971     ciKlass* klass = tkls->klass();
1972     if (tkls->is_loaded() && tkls->klass_is_exact()) {
1973       // We are loading a field from a Klass metaobject whose identity
1974       // is known at compile time (the type is "exact" or "precise").
1975       // Check for fields we know are maintained as constants by the VM.
1976       if (tkls->offset() == in_bytes(Klass::super_check_offset_offset())) {
1977         // The field is Klass::_super_check_offset.  Return its (constant) value.
1978         // (Folds up type checking code.)
1979         assert(Opcode() == Op_LoadI, "must load an int from _super_check_offset");
1980         return TypeInt::make(klass->super_check_offset());
1981       }
1982       // Compute index into primary_supers array
1983       juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1984       // Check for overflowing; use unsigned compare to handle the negative case.
1985       if( depth < ciKlass::primary_super_limit() ) {
1986         // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1987         // (Folds up type checking code.)
1988         assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
1989         ciKlass *ss = klass->super_of_depth(depth);
1990         return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
1991       }
1992       const Type* aift = load_array_final_field(tkls, klass);
1993       if (aift != NULL)  return aift;
1994     }
1995 
1996     // We can still check if we are loading from the primary_supers array at a
1997     // shallow enough depth.  Even though the klass is not exact, entries less
1998     // than or equal to its super depth are correct.
1999     if (tkls->is_loaded()) {
2000       ciType *inner = klass;
2001       while( inner->is_obj_array_klass() )
2002         inner = inner->as_obj_array_klass()->base_element_type();
2003       if( inner->is_instance_klass() &&
2004           !inner->as_instance_klass()->flags().is_interface() ) {
2005         // Compute index into primary_supers array
2006         juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
2007         // Check for overflowing; use unsigned compare to handle the negative case.
2008         if( depth < ciKlass::primary_super_limit() &&
2009             depth <= klass->super_depth() ) { // allow self-depth checks to handle self-check case
2010           // The field is an element of Klass::_primary_supers.  Return its (constant) value.
2011           // (Folds up type checking code.)
2012           assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
2013           ciKlass *ss = klass->super_of_depth(depth);
2014           return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
2015         }
2016       }
2017     }
2018 
2019     // If the type is enough to determine that the thing is not an array,
2020     // we can give the layout_helper a positive interval type.
2021     // This will help short-circuit some reflective code.
2022     if (tkls->offset() == in_bytes(Klass::layout_helper_offset())
2023         && !klass->is_array_klass() // not directly typed as an array
2024         && !klass->is_interface()  // specifically not Serializable & Cloneable
2025         && !klass->is_java_lang_Object()   // not the supertype of all T[]
2026         ) {
2027       // Note:  When interfaces are reliable, we can narrow the interface
2028       // test to (klass != Serializable && klass != Cloneable).
2029       assert(Opcode() == Op_LoadI, "must load an int from _layout_helper");
2030       jint min_size = Klass::instance_layout_helper(oopDesc::header_size(), false);
2031       // The key property of this type is that it folds up tests
2032       // for array-ness, since it proves that the layout_helper is positive.
2033       // Thus, a generic value like the basic object layout helper works fine.
2034       return TypeInt::make(min_size, max_jint, Type::WidenMin);
2035     }
2036   }
2037 
2038   // If we are loading from a freshly-allocated object, produce a zero,
2039   // if the load is provably beyond the header of the object.
2040   // (Also allow a variable load from a fresh array to produce zero.)
2041   const TypeOopPtr *tinst = tp->isa_oopptr();
2042   bool is_instance = (tinst != NULL) && tinst->is_known_instance_field();
2043   bool is_boxed_value = (tinst != NULL) && tinst->is_ptr_to_boxed_value();

2232   // Either input is TOP ==> the result is TOP
2233   const Type *t1 = phase->type( in(MemNode::Memory) );

2235   Node *adr = in(MemNode::Address);
2236   const Type *t2 = phase->type( adr );
2237   if (t2 == Type::TOP)  return Type::TOP;
2238   const TypePtr *tp = t2->is_ptr();
2239   if (TypePtr::above_centerline(tp->ptr()) ||
2240       tp->ptr() == TypePtr::Null)  return Type::TOP;
2241 
2242   // Return a more precise klass, if possible
2243   const TypeInstPtr *tinst = tp->isa_instptr();
2244   if (tinst != NULL) {
2245     ciInstanceKlass* ik = tinst->klass()->as_instance_klass();
2246     int offset = tinst->offset();
2247     if (ik == phase->C->env()->Class_klass()
2248         && (offset == java_lang_Class::klass_offset_in_bytes() ||
2249             offset == java_lang_Class::array_klass_offset_in_bytes())) {
2250       // We are loading a special hidden field from a Class mirror object,
2251       // the field which points to the VM's Klass metaobject.
2252       bool is_indirect_type = true;
2253       ciType* t = tinst->java_mirror_type(&is_indirect_type);
2254       // java_mirror_type returns non-null for compile-time Class constants.
2255       if (t != NULL) {
2256         // constant oop => constant klass

2258           if (t->is_void()) {
2259             // We cannot create a void array.  Since void is a primitive type return null
2260             // klass.  Users of this result need to do a null check on the returned klass.
2261             return TypePtr::NULL_PTR;
2262           }
2263           return TypeKlassPtr::make(ciArrayKlass::make(t, /* never_null */ !is_indirect_type));
2264         }
2265         if (!t->is_klass()) {
2266           // a primitive Class (e.g., int.class) has NULL for a klass field
2267           return TypePtr::NULL_PTR;
2268         }
2269         // (Folds up the 1st indirection in aClassConstant.getModifiers().)
2270         return TypeKlassPtr::make(t->as_klass());
2271       }
2272       // non-constant mirror, so we can't tell what's going on
2273     }
2274     if( !ik->is_loaded() )
2275       return _type;             // Bail out if not loaded
2276     if (offset == oopDesc::klass_offset_in_bytes()) {
2277       if (tinst->klass_is_exact()) {
2278         return TypeKlassPtr::make(ik);
2279       }
2280       // See if we can become precise: no subklasses and no interface
2281       // (Note:  We need to support verified interfaces.)
2282       if (!ik->is_interface() && !ik->has_subklass()) {
2283         //assert(!UseExactTypes, "this code should be useless with exact types");
2284         // Add a dependence; if any subclass added we need to recompile
2285         if (!ik->is_final()) {
2286           // %%% should use stronger assert_unique_concrete_subtype instead
2287           phase->C->dependencies()->assert_leaf_type(ik);
2288         }
2289         // Return precise klass
2290         return TypeKlassPtr::make(ik);
2291       }
2292 
2293       // Return root of possible klass
2294       return TypeKlassPtr::make(TypePtr::NotNull, ik, Type::Offset(0), tinst->flat_array());
2295     }
2296   }
2297 
2298   // Check for loading klass from an array
2299   const TypeAryPtr *tary = tp->isa_aryptr();
2300   if (tary != NULL) {
2301     ciKlass *tary_klass = tary->klass();
2302     if (tary_klass != NULL   // can be NULL when at BOTTOM or TOP
2303         && tary->offset() == oopDesc::klass_offset_in_bytes()) {
2304       ciArrayKlass* ak = tary_klass->as_array_klass();
2305       // Do not fold klass loads from [V?. The runtime type might be [V due to [V <: [V?
2306       // and the klass for [V is not equal to the klass for [V?.
2307       if (tary->klass_is_exact()) {
2308         return TypeKlassPtr::make(tary_klass);
2309       }
2310 
2311       // If the klass is an object array, we defer the question to the
2312       // array component klass.
2313       if (ak->is_obj_array_klass()) {
2314         assert(ak->is_loaded(), "");
2315         ciKlass *base_k = ak->as_obj_array_klass()->base_element_klass();
2316         if (base_k->is_loaded() && base_k->is_instance_klass()) {
2317           ciInstanceKlass *ik = base_k->as_instance_klass();
2318           // See if we can become precise: no subklasses and no interface
2319           if (!ik->is_interface() && !ik->has_subklass() && (!ik->is_valuetype() || ak->storage_properties().is_null_free())) {
2320             //assert(!UseExactTypes, "this code should be useless with exact types");
2321             // Add a dependence; if any subclass added we need to recompile
2322             if (!ik->is_final()) {
2323               phase->C->dependencies()->assert_leaf_type(ik);
2324             }
2325             // Return precise array klass
2326             return TypeKlassPtr::make(ak);
2327           }
2328         }
2329         return TypeKlassPtr::make(TypePtr::NotNull, ak, Type::Offset(0), false);
2330       } else if (ak->is_type_array_klass()) {
2331         //assert(!UseExactTypes, "this code should be useless with exact types");
2332         return TypeKlassPtr::make(ak); // These are always precise
2333       }
2334     }
2335   }
2336 
2337   // Check for loading klass from an array klass
2338   const TypeKlassPtr *tkls = tp->isa_klassptr();
2339   if (tkls != NULL && !StressReflectiveCode) {
2340     if (!tkls->is_loaded()) {
2341       return _type;             // Bail out if not loaded
2342     }
2343     ciKlass* klass = tkls->klass();
2344     if( klass->is_obj_array_klass() &&
2345         tkls->offset() == in_bytes(ObjArrayKlass::element_klass_offset())) {
2346       ciKlass* elem = klass->as_obj_array_klass()->element_klass();
2347       // // Always returning precise element type is incorrect,
2348       // // e.g., element type could be object and array may contain strings
2349       // return TypeKlassPtr::make(TypePtr::Constant, elem, 0);
2350 
2351       // The array's TypeKlassPtr was declared 'precise' or 'not precise'
2352       // according to the element type's subclassing.
2353       return TypeKlassPtr::make(tkls->ptr(), elem, Type::Offset(0), elem->flatten_array());
2354     } else if (klass->is_value_array_klass() &&
2355                tkls->offset() == in_bytes(ObjArrayKlass::element_klass_offset())) {
2356       ciKlass* elem = klass->as_value_array_klass()->element_klass();
2357       return TypeKlassPtr::make(tkls->ptr(), elem, Type::Offset(0), true);

2360         tkls->offset() == in_bytes(Klass::super_offset())) {

2363       // (Folds up the 2nd indirection in aClassConstant.getSuperClass().)

2366   }

2425 Node* LoadNode::klass_identity_common(PhaseGVN* phase) {
2426   Node* x = LoadNode::Identity(phase);
2427   if (x != this)  return x;
2428 
2429   // Take apart the address into an oop and and offset.
2430   // Return 'this' if we cannot.
2431   Node*    adr    = in(MemNode::Address);
2432   intptr_t offset = 0;
2433   Node*    base   = AddPNode::Ideal_base_and_offset(adr, phase, offset);
2434   if (base == NULL)     return this;
2435   const TypeOopPtr* toop = phase->type(adr)->isa_oopptr();
2436   if (toop == NULL)     return this;
2437 
2438   // Step over potential GC barrier for OopHandle resolve
2439   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
2440   if (bs->is_gc_barrier_node(base)) {
2441     base = bs->step_over_gc_barrier(base);
2442   }
2443 
2444   // We can fetch the klass directly through an AllocateNode.
2445   // This works even if the klass is not constant (clone or newArray).
2446   if (offset == oopDesc::klass_offset_in_bytes()) {
2447     Node* allocated_klass = AllocateNode::Ideal_klass(base, phase);
2448     if (allocated_klass != NULL) {
2449       return allocated_klass;
2450     }
2451   }
2452 
2453   // Simplify k.java_mirror.as_klass to plain k, where k is a Klass*.
2454   // See inline_native_Class_query for occurrences of these patterns.
2455   // Java Example:  x.getClass().isAssignableFrom(y)
2456   //
2457   // This improves reflective code, often making the Class
2458   // mirror go completely dead.  (Current exception:  Class
2459   // mirrors may appear in debug info, but we could clean them out by
2460   // introducing a new debug info operator for Klass.java_mirror).
2461 
2462   if (toop->isa_instptr() && toop->klass() == phase->C->env()->Class_klass()
2463       && offset == java_lang_Class::klass_offset_in_bytes()) {
2464     if (base->is_Load()) {
2465       Node* base2 = base->in(MemNode::Address);
2466       if (base2->is_Load()) { /* direct load of a load which is the OopHandle */
2467         Node* adr2 = base2->in(MemNode::Address);
2468         const TypeKlassPtr* tkls = phase->type(adr2)->isa_klassptr();
2469         if (tkls != NULL && !tkls->empty()
2470             && (tkls->klass()->is_instance_klass() ||
2471               tkls->klass()->is_array_klass())
2472             && adr2->is_AddP()
2473            ) {
2474           int mirror_field = in_bytes(Klass::java_mirror_offset());
2475           if (tkls->offset() == mirror_field) {
2476             return adr2->in(AddPNode::Base);
2477           }
2478         }
2479       }
2480     }
2481   }
2482 
2483   return this;
2484 }

2658 //------------------------------Ideal------------------------------------------
2659 // Change back-to-back Store(, p, x) -> Store(m, p, y) to Store(m, p, x).
2660 // When a store immediately follows a relevant allocation/initialization,
2661 // try to capture it into the initialization, or hoist it above.
2662 Node *StoreNode::Ideal(PhaseGVN *phase, bool can_reshape) {
2663   Node* p = MemNode::Ideal_common(phase, can_reshape);
2664   if (p)  return (p == NodeSentinel) ? NULL : p;
2665 
2666   Node* mem     = in(MemNode::Memory);
2667   Node* address = in(MemNode::Address);
2668   // Back-to-back stores to same address?  Fold em up.  Generally
2669   // unsafe if I have intervening uses...  Also disallowed for StoreCM
2670   // since they must follow each StoreP operation.  Redundant StoreCMs
2671   // are eliminated just before matching in final_graph_reshape.
2672   if (phase->C->get_adr_type(phase->C->get_alias_index(adr_type())) != TypeAryPtr::VALUES) {
2673     Node* st = mem;
2674     // If Store 'st' has more than one use, we cannot fold 'st' away.
2675     // For example, 'st' might be the final state at a conditional
2676     // return.  Or, 'st' might be used by some node which is live at
2677     // the same time 'st' is live, which might be unschedulable.  So,
2678     // require exactly ONE user until such time as we clone 'mem' for
2679     // each of 'mem's uses (thus making the exactly-1-user-rule hold
2680     // true).
2681     while (st->is_Store() && st->outcnt() == 1 && st->Opcode() != Op_StoreCM) {
2682       // Looking at a dead closed cycle of memory?
2683       assert(st != st->in(MemNode::Memory), "dead loop in StoreNode::Ideal");
2684       assert(Opcode() == st->Opcode() ||
2685              st->Opcode() == Op_StoreVector ||
2686              Opcode() == Op_StoreVector ||
2687              phase->C->get_alias_index(adr_type()) == Compile::AliasIdxRaw ||
2688              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreI) || // expanded ClearArrayNode
2689              (Opcode() == Op_StoreI && st->Opcode() == Op_StoreL) || // initialization by arraycopy
2690              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreN) ||
2691              (is_mismatched_access() || st->as_Store()->is_mismatched_access()),
2692              "no mismatched stores, except on raw memory: %s %s", NodeClassNames[Opcode()], NodeClassNames[st->Opcode()]);
2693 
2694       if (st->in(MemNode::Address)->eqv_uncast(address) &&
2695           st->as_Store()->memory_size() <= this->memory_size()) {
2696         Node* use = st->raw_out(0);
2697         phase->igvn_rehash_node_delayed(use);
2698         if (can_reshape) {
2699           use->set_req_X(MemNode::Memory, st->in(MemNode::Memory), phase->is_IterGVN());
2700         } else {
2701           // It's OK to do this in the parser, since DU info is always accurate,
2702           // and the parser always refers to nodes via SafePointNode maps.
2703           use->set_req(MemNode::Memory, st->in(MemNode::Memory));
2704         }
2705         return this;
2706       }
2707       st = st->in(MemNode::Memory);
2708     }
2709   }
2710 
2711 
2712   // Capture an unaliased, unconditional, simple store into an initializer.
2713   // Or, if it is independent of the allocation, hoist it above the allocation.
2714   if (ReduceFieldZeroing && /*can_reshape &&*/
2715       mem->is_Proj() && mem->in(0)->is_Initialize()) {
2716     InitializeNode* init = mem->in(0)->as_Initialize();
2717     intptr_t offset = init->can_capture_store(this, phase, can_reshape);
2718     if (offset > 0) {
2719       Node* moved = init->capture_store(this, offset, phase, can_reshape);
2720       // If the InitializeNode captured me, it made a raw copy of me,
2721       // and I need to disappear.
2722       if (moved != NULL) {
2723         // %%% hack to ensure that Ideal returns a new node:
2724         mem = MergeMemNode::make(mem);
2725         return mem;             // fold me away
2726       }
2727     }
2728   }
2729 
2730   return NULL;                  // No further progress
2731 }

2745 //------------------------------Identity---------------------------------------
2746 // Remove redundant stores:
2747 //   Store(m, p, Load(m, p)) changes to m.
2748 //   Store(, p, x) -> Store(m, p, x) changes to Store(m, p, x).
2749 Node* StoreNode::Identity(PhaseGVN* phase) {
2750   Node* mem = in(MemNode::Memory);
2751   Node* adr = in(MemNode::Address);
2752   Node* val = in(MemNode::ValueIn);
2753 
2754   Node* result = this;
2755 
2756   // Load then Store?  Then the Store is useless
2757   if (val->is_Load() &&
2758       val->in(MemNode::Address)->eqv_uncast(adr) &&
2759       val->in(MemNode::Memory )->eqv_uncast(mem) &&
2760       val->as_Load()->store_Opcode() == Opcode()) {
2761     result = mem;
2762   }
2763 
2764   // Two stores in a row of the same value?
2765   if (result == this &&
2766       mem->is_Store() &&
2767       mem->in(MemNode::Address)->eqv_uncast(adr) &&
2768       mem->in(MemNode::ValueIn)->eqv_uncast(val) &&
2769       mem->Opcode() == Opcode()) {
2770     result = mem;
2771   }
2772 
2773   // Store of zero anywhere into a freshly-allocated object?
2774   // Then the store is useless.
2775   // (It must already have been captured by the InitializeNode.)
2776   if (result == this && ReduceFieldZeroing) {
2777     // a newly allocated object is already all-zeroes everywhere
2778     if (mem->is_Proj() && mem->in(0)->is_Allocate() &&
2779         (phase->type(val)->is_zero_type() || mem->in(0)->in(AllocateNode::DefaultValue) == val)) {
2780       assert(!phase->type(val)->is_zero_type() || mem->in(0)->in(AllocateNode::DefaultValue) == NULL, "storing null to value array is forbidden");
2781       result = mem;
2782     }
2783 
2784     if (result == this) {
2785       // the store may also apply to zero-bits in an earlier object
2786       Node* prev_mem = find_previous_store(phase);
2787       // Steps (a), (b):  Walk past independent stores to find an exact match.
2788       if (prev_mem != NULL) {
2789         Node* prev_val = can_see_stored_value(prev_mem, phase);
2790         if (prev_val != NULL && phase->eqv(prev_val, val)) {
2791           // prev_val and val might differ by a cast; it would be good
2792           // to keep the more informative of the two.
2793           if (phase->type(val)->is_zero_type()) {
2794             result = mem;
2795           } else if (prev_mem->is_Proj() && prev_mem->in(0)->is_Initialize()) {
2796             InitializeNode* init = prev_mem->in(0)->as_Initialize();
2797             AllocateNode* alloc = init->allocation();
2798             if (alloc != NULL && alloc->in(AllocateNode::DefaultValue) == val) {
2799               result = mem;
2800             }
2801           }
2802         }
2803       }
2804     }
2805   }
2806 
2807   if (result != this && phase->is_IterGVN() != NULL) {
2808     MemBarNode* trailing = trailing_membar();
2809     if (trailing != NULL) {
2810 #ifdef ASSERT
2811       const TypeOopPtr* t_oop = phase->type(in(Address))->isa_oopptr();
2812       assert(t_oop == NULL || t_oop->is_known_instance_field(), "only for non escaping objects");
2813 #endif
2814       PhaseIterGVN* igvn = phase->is_IterGVN();
2815       trailing->remove(igvn);
2816     }
2817   }
2818 
2819   return result;
2820 }

3534 //===========================InitializeNode====================================
3535 // SUMMARY:
3536 // This node acts as a memory barrier on raw memory, after some raw stores.
3537 // The 'cooked' oop value feeds from the Initialize, not the Allocation.
3538 // The Initialize can 'capture' suitably constrained stores as raw inits.
3539 // It can coalesce related raw stores into larger units (called 'tiles').
3540 // It can avoid zeroing new storage for memory units which have raw inits.
3541 // At macro-expansion, it is marked 'complete', and does not optimize further.
3542 //
3543 // EXAMPLE:
3544 // The object 'new short[2]' occupies 16 bytes in a 32-bit machine.
3545 //   ctl = incoming control; mem* = incoming memory
3546 // (Note:  A star * on a memory edge denotes I/O and other standard edges.)
3547 // First allocate uninitialized memory and fill in the header:
3548 //   alloc = (Allocate ctl mem* 16 #short[].klass ...)
3549 //   ctl := alloc.Control; mem* := alloc.Memory*
3550 //   rawmem = alloc.Memory; rawoop = alloc.RawAddress
3551 // Then initialize to zero the non-header parts of the raw memory block:
3552 //   init = (Initialize alloc.Control alloc.Memory* alloc.RawAddress)
3553 //   ctl := init.Control; mem.SLICE(#short[*]) := init.Memory
3554 // After the initialize node executes, the object is ready for service:
3555 //   oop := (CheckCastPP init.Control alloc.RawAddress #short[])
3556 // Suppose its body is immediately initialized as {1,2}:
3557 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3558 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3559 //   mem.SLICE(#short[*]) := store2
3560 //
3561 // DETAILS:
3562 // An InitializeNode collects and isolates object initialization after
3563 // an AllocateNode and before the next possible safepoint.  As a
3564 // memory barrier (MemBarNode), it keeps critical stores from drifting
3565 // down past any safepoint or any publication of the allocation.
3566 // Before this barrier, a newly-allocated object may have uninitialized bits.
3567 // After this barrier, it may be treated as a real oop, and GC is allowed.
3568 //
3569 // The semantics of the InitializeNode include an implicit zeroing of
3570 // the new object from object header to the end of the object.
3571 // (The object header and end are determined by the AllocateNode.)
3572 //
3573 // Certain stores may be added as direct inputs to the InitializeNode.
3574 // These stores must update raw memory, and they must be to addresses
3575 // derived from the raw address produced by AllocateNode, and with
3576 // a constant offset.  They must be ordered by increasing offset.
3577 // The first one is at in(RawStores), the last at in(req()-1).
3578 // Unlike most memory operations, they are not linked in a chain,
3579 // but are displayed in parallel as users of the rawmem output of
3580 // the allocation.
3581 //
3582 // (See comments in InitializeNode::capture_store, which continue
3583 // the example given above.)
3584 //
3585 // When the associated Allocate is macro-expanded, the InitializeNode
3586 // may be rewritten to optimize collected stores.  A ClearArrayNode
3587 // may also be created at that point to represent any required zeroing.
3588 // The InitializeNode is then marked 'complete', prohibiting further
3589 // capturing of nearby memory operations.
3590 //
3591 // During macro-expansion, all captured initializations which store
3592 // constant values of 32 bits or smaller are coalesced (if advantageous)
3593 // into larger 'tiles' 32 or 64 bits.  This allows an object to be
3594 // initialized in fewer memory operations.  Memory words which are
3595 // covered by neither tiles nor non-constant stores are pre-zeroed
3596 // by explicit stores of zero.  (The code shape happens to do all
3597 // zeroing first, then all other stores, with both sequences occurring
3598 // in order of ascending offsets.)
3599 //
3600 // Alternatively, code may be inserted between an AllocateNode and its
3601 // InitializeNode, to perform arbitrary initialization of the new object.
3602 // E.g., the object copying intrinsics insert complex data transfers here.
3603 // The initialization must then be marked as 'complete' disable the
3604 // built-in zeroing semantics and the collection of initializing stores.
3605 //
3606 // While an InitializeNode is incomplete, reads from the memory state
3607 // produced by it are optimizable if they match the control edge and
3608 // new oop address associated with the allocation/initialization.
3609 // They return a stored value (if the offset matches) or else zero.
3610 // A write to the memory state, if it matches control and address,
3611 // and if it is to a constant offset, may be 'captured' by the
3612 // InitializeNode.  It is cloned as a raw memory operation and rewired
3613 // inside the initialization, to the raw oop produced by the allocation.
3614 // Operations on addresses which are provably distinct (e.g., to
3615 // other AllocateNodes) are allowed to bypass the initialization.
3616 //
3617 // The effect of all this is to consolidate object initialization
3618 // (both arrays and non-arrays, both piecewise and bulk) into a
3619 // single location, where it can be optimized as a unit.
3620 //
3621 // Only stores with an offset less than TrackedInitializationLimit words
3622 // will be considered for capture by an InitializeNode.  This puts a
3623 // reasonable limit on the complexity of optimized initializations.
3624 
3625 //---------------------------InitializeNode------------------------------------
3626 InitializeNode::InitializeNode(Compile* C, int adr_type, Node* rawoop)
3627   : MemBarNode(C, adr_type, rawoop),

3717 // Helper for proving that an initialization expression is
3718 // "simple enough" to be folded into an object initialization.
3719 // Attempts to prove that a store's initial value 'n' can be captured
3720 // within the initialization without creating a vicious cycle, such as:
3721 //     { Foo p = new Foo(); p.next = p; }
3722 // True for constants and parameters and small combinations thereof.
3723 bool InitializeNode::detect_init_independence(Node* value, PhaseGVN* phase) {
3724   ResourceMark rm;
3725   Unique_Node_List worklist;
3726   worklist.push(value);
3727 
3728   uint complexity_limit = 20;
3729   for (uint j = 0; j < worklist.size(); j++) {
3730     if (j >= complexity_limit) {
3731       return false;  // Bail out if processed too many nodes
3732     }
3733 
3734     Node* n = worklist.at(j);
3735     if (n == NULL)      continue;   // (can this really happen?)
3736     if (n->is_Proj())   n = n->in(0);
3737     if (n == this)      return false;  // found a cycle
3738     if (n->is_Con())    continue;
3739     if (n->is_Start())  continue;   // params, etc., are OK
3740     if (n->is_Root())   continue;   // even better
3741 
3742     // There cannot be any dependency if 'n' is a CFG node that dominates the current allocation
3743     if (n->is_CFG() && phase->is_dominator(n, allocation())) {
3744       continue;
3745     }
3746 
3747     Node* ctl = n->in(0);
3748     if (ctl != NULL && !ctl->is_top()) {
3749       if (ctl->is_Proj())  ctl = ctl->in(0);
3750       if (ctl == this)  return false;
3751 
3752       // If we already know that the enclosing memory op is pinned right after
3753       // the init, then any control flow that the store has picked up
3754       // must have preceded the init, or else be equal to the init.
3755       // Even after loop optimizations (which might change control edges)
3756       // a store is never pinned *before* the availability of its inputs.
3757       if (!MemNode::all_controls_dominate(n, this))
3758         return false;                  // failed to prove a good control
3759     }
3760 
3761     // Check data edges for possible dependencies on 'this'.
3762     for (uint i = 1; i < n->req(); i++) {
3763       Node* m = n->in(i);
3764       if (m == NULL || m == n || m->is_top())  continue;
3765 
3766       // Only process data inputs once
3767       worklist.push(m);
3768     }
3769   }
3770 
3771   return true;
3772 }

3774 // Here are all the checks a Store must pass before it can be moved into
3775 // an initialization.  Returns zero if a check fails.
3776 // On success, returns the (constant) offset to which the store applies,
3777 // within the initialized memory.
3778 intptr_t InitializeNode::can_capture_store(StoreNode* st, PhaseGVN* phase, bool can_reshape) {
3779   const int FAIL = 0;
3780   if (st->req() != MemNode::ValueIn + 1)
3781     return FAIL;                // an inscrutable StoreNode (card mark?)
3782   Node* ctl = st->in(MemNode::Control);
3783   if (!(ctl != NULL && ctl->is_Proj() && ctl->in(0) == this))
3784     return FAIL;                // must be unconditional after the initialization
3785   Node* mem = st->in(MemNode::Memory);
3786   if (!(mem->is_Proj() && mem->in(0) == this))
3787     return FAIL;                // must not be preceded by other stores
3788   Node* adr = st->in(MemNode::Address);
3789   intptr_t offset;
3790   AllocateNode* alloc = AllocateNode::Ideal_allocation(adr, phase, offset);
3791   if (alloc == NULL)
3792     return FAIL;                // inscrutable address
3793   if (alloc != allocation())
3794     return FAIL;                // wrong allocation!  (store needs to float up)
3795   int size_in_bytes = st->memory_size();
3796   if ((size_in_bytes != 0) && (offset % size_in_bytes) != 0) {
3797     return FAIL;                // mismatched access
3798   }
3799   Node* val = st->in(MemNode::ValueIn);
3800 
3801   if (!detect_init_independence(val, phase))
3802     return FAIL;                // stored value must be 'simple enough'
3803 
3804   // The Store can be captured only if nothing after the allocation
3805   // and before the Store is using the memory location that the store
3806   // overwrites.
3807   bool failed = false;
3808   // If is_complete_with_arraycopy() is true the shape of the graph is
3809   // well defined and is safe so no need for extra checks.
3810   if (!is_complete_with_arraycopy()) {
3811     // We are going to look at each use of the memory state following
3812     // the allocation to make sure nothing reads the memory that the
3813     // Store writes.
3814     const TypePtr* t_adr = phase->type(adr)->isa_ptr();
3815     int alias_idx = phase->C->get_alias_index(t_adr);
3816     ResourceMark rm;
3817     Unique_Node_List mems;
3818     mems.push(mem);
3819     Node* unique_merge = NULL;
3820     for (uint next = 0; next < mems.size(); ++next) {
3821       Node *m  = mems.at(next);
3822       for (DUIterator_Fast jmax, j = m->fast_outs(jmax); j < jmax; j++) {
3823         Node *n = m->fast_out(j);
3824         if (n->outcnt() == 0) {
3825           continue;
3826         }
3827         if (n == st) {
3828           continue;
3829         } else if (n->in(0) != NULL && n->in(0) != ctl) {
3830           // If the control of this use is different from the control
3831           // of the Store which is right after the InitializeNode then
3832           // this node cannot be between the InitializeNode and the
3833           // Store.
3834           continue;
3835         } else if (n->is_MergeMem()) {
3836           if (n->as_MergeMem()->memory_at(alias_idx) == m) {
3837             // We can hit a MergeMemNode (that will likely go away
3838             // later) that is a direct use of the memory state
3839             // following the InitializeNode on the same slice as the
3840             // store node that we'd like to capture. We need to check
3841             // the uses of the MergeMemNode.
3842             mems.push(n);
3843           }
3844         } else if (n->is_Mem()) {
3845           Node* other_adr = n->in(MemNode::Address);
3846           if (other_adr == adr) {
3847             failed = true;
3848             break;
3849           } else {
3850             const TypePtr* other_t_adr = phase->type(other_adr)->isa_ptr();
3851             if (other_t_adr != NULL) {
3852               int other_alias_idx = phase->C->get_alias_index(other_t_adr);
3853               if (other_alias_idx == alias_idx) {
3854                 // A load from the same memory slice as the store right
3855                 // after the InitializeNode. We check the control of the
3856                 // object/array that is loaded from. If it's the same as
3857                 // the store control then we cannot capture the store.
3858                 assert(!n->is_Store(), "2 stores to same slice on same control?");
3859                 Node* base = other_adr;
3860                 assert(base->is_AddP(), "should be addp but is %s", base->Name());
3861                 base = base->in(AddPNode::Base);
3862                 if (base != NULL) {
3863                   base = base->uncast();
3864                   if (base->is_Proj() && base->in(0) == alloc) {
3865                     failed = true;
3866                     break;
3867                   }
3868                 }
3869               }
3870             }
3871           }
3872         } else {
3873           failed = true;
3874           break;
3875         }
3876       }
3877     }
3878   }
3879   if (failed) {
3880     if (!can_reshape) {
3881       // We decided we couldn't capture the store during parsing. We
3882       // should try again during the next IGVN once the graph is
3883       // cleaner.
3884       phase->C->record_for_igvn(st);
3885     }
3886     return FAIL;
3887   }
3888 
3889   return offset;                // success
3890 }

3892 // Find the captured store in(i) which corresponds to the range
3893 // [start..start+size) in the initialized object.
3894 // If there is one, return its index i.  If there isn't, return the
3895 // negative of the index where it should be inserted.
3896 // Return 0 if the queried range overlaps an initialization boundary
3897 // or if dead code is encountered.
3898 // If size_in_bytes is zero, do not bother with overlap checks.
3899 int InitializeNode::captured_store_insertion_point(intptr_t start,
3900                                                    int size_in_bytes,
3901                                                    PhaseTransform* phase) {
3902   const int FAIL = 0, MAX_STORE = BytesPerLong;
3903 
3904   if (is_complete())
3905     return FAIL;                // arraycopy got here first; punt
3906 
3907   assert(allocation() != NULL, "must be present");
3908 
3909   // no negatives, no header fields:
3910   if (start < (intptr_t) allocation()->minimum_header_size())  return FAIL;
3911 
3912   // after a certain size, we bail out on tracking all the stores:
3913   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
3914   if (start >= ti_limit)  return FAIL;
3915 
3916   for (uint i = InitializeNode::RawStores, limit = req(); ; ) {
3917     if (i >= limit)  return -(int)i; // not found; here is where to put it
3918 
3919     Node*    st     = in(i);
3920     intptr_t st_off = get_store_offset(st, phase);
3921     if (st_off < 0) {
3922       if (st != zero_memory()) {
3923         return FAIL;            // bail out if there is dead garbage
3924       }
3925     } else if (st_off > start) {
3926       // ...we are done, since stores are ordered
3927       if (st_off < start + size_in_bytes) {
3928         return FAIL;            // the next store overlaps
3929       }
3930       return -(int)i;           // not found; here is where to put it
3931     } else if (st_off < start) {
3932       if (size_in_bytes != 0 &&
3933           start < st_off + MAX_STORE &&
3934           start < st_off + st->as_Store()->memory_size()) {
3935         return FAIL;            // the previous store overlaps
3936       }
3937     } else {
3938       if (size_in_bytes != 0 &&
3939           st->as_Store()->memory_size() != size_in_bytes) {
3940         return FAIL;            // mismatched store size
3941       }
3942       return i;
3943     }
3944 
3945     ++i;
3946   }
3947 }

3980 // Clone the given store, converting it into a raw store
3981 // initializing a field or element of my new object.
3982 // Caller is responsible for retiring the original store,
3983 // with subsume_node or the like.
3984 //
3985 // From the example above InitializeNode::InitializeNode,
3986 // here are the old stores to be captured:
3987 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3988 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3989 //
3990 // Here is the changed code; note the extra edges on init:
3991 //   alloc = (Allocate ...)
3992 //   rawoop = alloc.RawAddress
3993 //   rawstore1 = (StoreC alloc.Control alloc.Memory (+ rawoop 12) 1)
3994 //   rawstore2 = (StoreC alloc.Control alloc.Memory (+ rawoop 14) 2)
3995 //   init = (Initialize alloc.Control alloc.Memory rawoop
3996 //                      rawstore1 rawstore2)
3997 //
3998 Node* InitializeNode::capture_store(StoreNode* st, intptr_t start,
3999                                     PhaseGVN* phase, bool can_reshape) {
4000   assert(stores_are_sane(phase), "");
4001 
4002   if (start < 0)  return NULL;
4003   assert(can_capture_store(st, phase, can_reshape) == start, "sanity");
4004 
4005   Compile* C = phase->C;
4006   int size_in_bytes = st->memory_size();
4007   int i = captured_store_insertion_point(start, size_in_bytes, phase);
4008   if (i == 0)  return NULL;     // bail out
4009   Node* prev_mem = NULL;        // raw memory for the captured store
4010   if (i > 0) {
4011     prev_mem = in(i);           // there is a pre-existing store under this one
4012     set_req(i, C->top());       // temporarily disconnect it
4013     // See StoreNode::Ideal 'st->outcnt() == 1' for the reason to disconnect.
4014   } else {
4015     i = -i;                     // no pre-existing store
4016     prev_mem = zero_memory();   // a slice of the newly allocated object
4017     if (i > InitializeNode::RawStores && in(i-1) == prev_mem)
4018       set_req(--i, C->top());   // reuse this edge; it has been folded away
4019     else
4020       ins_req(i, C->top());     // build a new edge
4021   }
4022   Node* new_st = st->clone();
4023   new_st->set_req(MemNode::Control, in(Control));
4024   new_st->set_req(MemNode::Memory,  prev_mem);
4025   new_st->set_req(MemNode::Address, make_raw_address(start, phase));
4026   new_st = phase->transform(new_st);
4027 
4028   // At this point, new_st might have swallowed a pre-existing store
4029   // at the same offset, or perhaps new_st might have disappeared,
4030   // if it redundantly stored the same value (or zero to fresh memory).
4031 
4032   // In any case, wire it in:
4033   phase->igvn_rehash_node_delayed(this);
4034   set_req(i, new_st);
4035 
4036   // The caller may now kill the old guy.
4037   DEBUG_ONLY(Node* check_st = find_captured_store(start, size_in_bytes, phase));
4038   assert(check_st == new_st || check_st == NULL, "must be findable");
4039   assert(!is_complete(), "");
4040   return new_st;
4041 }

4043 static bool store_constant(jlong* tiles, int num_tiles,
4044                            intptr_t st_off, int st_size,
4045                            jlong con) {
4046   if ((st_off & (st_size-1)) != 0)
4047     return false;               // strange store offset (assume size==2**N)
4048   address addr = (address)tiles + st_off;
4049   assert(st_off >= 0 && addr+st_size <= (address)&tiles[num_tiles], "oob");
4050   switch (st_size) {
4051   case sizeof(jbyte):  *(jbyte*) addr = (jbyte) con; break;
4052   case sizeof(jchar):  *(jchar*) addr = (jchar) con; break;
4053   case sizeof(jint):   *(jint*)  addr = (jint)  con; break;
4054   case sizeof(jlong):  *(jlong*) addr = (jlong) con; break;
4055   default: return false;        // strange store size (detect size!=2**N here)
4056   }
4057   return true;                  // return success to caller
4058 }

4060 // Coalesce subword constants into int constants and possibly
4061 // into long constants.  The goal, if the CPU permits,
4062 // is to initialize the object with a small number of 64-bit tiles.
4063 // Also, convert floating-point constants to bit patterns.
4064 // Non-constants are not relevant to this pass.
4065 //
4066 // In terms of the running example on InitializeNode::InitializeNode
4067 // and InitializeNode::capture_store, here is the transformation
4068 // of rawstore1 and rawstore2 into rawstore12:
4069 //   alloc = (Allocate ...)
4070 //   rawoop = alloc.RawAddress
4071 //   tile12 = 0x00010002
4072 //   rawstore12 = (StoreI alloc.Control alloc.Memory (+ rawoop 12) tile12)
4073 //   init = (Initialize alloc.Control alloc.Memory rawoop rawstore12)
4074 //
4075 void
4076 InitializeNode::coalesce_subword_stores(intptr_t header_size,
4077                                         Node* size_in_bytes,
4078                                         PhaseGVN* phase) {
4079   Compile* C = phase->C;
4080 
4081   assert(stores_are_sane(phase), "");
4082   // Note:  After this pass, they are not completely sane,
4083   // since there may be some overlaps.
4084 
4085   int old_subword = 0, old_long = 0, new_int = 0, new_long = 0;
4086 
4087   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
4088   intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, ti_limit);
4089   size_limit = MIN2(size_limit, ti_limit);
4090   size_limit = align_up(size_limit, BytesPerLong);
4091   int num_tiles = size_limit / BytesPerLong;
4092 
4093   // allocate space for the tile map:
4094   const int small_len = DEBUG_ONLY(true ? 3 :) 30; // keep stack frames small
4095   jlong  tiles_buf[small_len];
4096   Node*  nodes_buf[small_len];
4097   jlong  inits_buf[small_len];
4098   jlong* tiles = ((num_tiles <= small_len) ? &tiles_buf[0]
4099                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
4100   Node** nodes = ((num_tiles <= small_len) ? &nodes_buf[0]
4101                   : NEW_RESOURCE_ARRAY(Node*, num_tiles));
4102   jlong* inits = ((num_tiles <= small_len) ? &inits_buf[0]
4103                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
4104   // tiles: exact bitwise model of all primitive constants
4105   // nodes: last constant-storing node subsumed into the tiles model
4106   // inits: which bytes (in each tile) are touched by any initializations
4107 
4108   //// Pass A: Fill in the tile model with any relevant stores.
4109 
4110   Copy::zero_to_bytes(tiles, sizeof(tiles[0]) * num_tiles);
4111   Copy::zero_to_bytes(nodes, sizeof(nodes[0]) * num_tiles);
4112   Copy::zero_to_bytes(inits, sizeof(inits[0]) * num_tiles);
4113   Node* zmem = zero_memory(); // initially zero memory state
4114   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
4115     Node* st = in(i);
4116     intptr_t st_off = get_store_offset(st, phase);
4117 
4118     // Figure out the store's offset and constant value:
4119     if (st_off < header_size)             continue; //skip (ignore header)
4120     if (st->in(MemNode::Memory) != zmem)  continue; //skip (odd store chain)
4121     int st_size = st->as_Store()->memory_size();
4122     if (st_off + st_size > size_limit)    break;
4123 
4124     // Record which bytes are touched, whether by constant or not.
4125     if (!store_constant(inits, num_tiles, st_off, st_size, (jlong) -1))
4126       continue;                 // skip (strange store size)
4127 
4128     const Type* val = phase->type(st->in(MemNode::ValueIn));
4129     if (!val->singleton())                continue; //skip (non-con store)
4130     BasicType type = val->basic_type();
4131 
4132     jlong con = 0;
4133     switch (type) {
4134     case T_INT:    con = val->is_int()->get_con();  break;
4135     case T_LONG:   con = val->is_long()->get_con(); break;
4136     case T_FLOAT:  con = jint_cast(val->getf());    break;
4137     case T_DOUBLE: con = jlong_cast(val->getd());   break;
4138     default:                              continue; //skip (odd store type)
4139     }
4140 
4141     if (type == T_LONG && Matcher::isSimpleConstant64(con) &&
4142         st->Opcode() == Op_StoreL) {
4143       continue;                 // This StoreL is already optimal.
4144     }
4145 
4146     // Store down the constant.
4147     store_constant(tiles, num_tiles, st_off, st_size, con);
4148 
4149     intptr_t j = st_off >> LogBytesPerLong;
4150 
4151     if (type == T_INT && st_size == BytesPerInt
4152         && (st_off & BytesPerInt) == BytesPerInt) {
4153       jlong lcon = tiles[j];
4154       if (!Matcher::isSimpleConstant64(lcon) &&
4155           st->Opcode() == Op_StoreI) {
4156         // This StoreI is already optimal by itself.
4157         jint* intcon = (jint*) &tiles[j];
4158         intcon[1] = 0;  // undo the store_constant()
4159 
4160         // If the previous store is also optimal by itself, back up and
4161         // undo the action of the previous loop iteration... if we can.
4162         // But if we can't, just let the previous half take care of itself.
4163         st = nodes[j];
4164         st_off -= BytesPerInt;
4165         con = intcon[0];
4166         if (con != 0 && st != NULL && st->Opcode() == Op_StoreI) {
4167           assert(st_off >= header_size, "still ignoring header");

4171           assert(con == tcon->is_int()->get_con(), "must be");

4173           intcon[0] = 0;        // undo store_constant()

4176           --old_subword;        // undo ++old_subword

4187     else  ++old_long;

4194   // Be sure to insert them before overlapping non-constant stores.
4195   // (E.g., byte[] x = { 1,2,y,4 }  =>  x[int 0] = 0x01020004, x[2]=y.)
4196   for (int j = 0; j < num_tiles; j++) {
4197     jlong con  = tiles[j];

4200     jint con0,  con1;           // split the constant, address-wise
4201     jint init0, init1;          // split the init map, address-wise
4202     { union { jlong con; jint intcon[2]; } u;
4203       u.con = con;
4204       con0  = u.intcon[0];
4205       con1  = u.intcon[1];
4206       u.con = init;
4207       init0 = u.intcon[0];
4208       init1 = u.intcon[1];
4209     }

4212     assert(old != NULL, "need the prior store");
4213     intptr_t offset = (j * BytesPerLong);
4214 
4215     bool split = !Matcher::isSimpleConstant64(con);
4216 
4217     if (offset < header_size) {
4218       assert(offset + BytesPerInt >= header_size, "second int counts");
4219       assert(*(jint*)&tiles[j] == 0, "junk in header");

4221       // Example:  int a[] = { 42 ... }
4222     } else if (con0 == 0 && init0 == -1) {
4223       split = true;             // first word is covered by full inits
4224       // Example:  int a[] = { ... foo(), 42 ... }
4225     } else if (con1 == 0 && init1 == -1) {
4226       split = true;             // second word is covered by full inits
4227       // Example:  int a[] = { ... 42, foo() ... }
4228     }
4229 
4230     // Here's a case where init0 is neither 0 nor -1:
4231     //   byte a[] = { ... 0,0,foo(),0,  0,0,0,42 ... }
4232     // Assuming big-endian memory, init0, init1 are 0x0000FF00, 0x000000FF.
4233     // In this case the tile is not split; it is (jlong)42.
4234     // The big tile is stored down, and then the foo() value is inserted.
4235     // (If there were foo(),foo() instead of foo(),0, init0 would be -1.)

4237     Node* ctl = old->in(MemNode::Control);
4238     Node* adr = make_raw_address(offset, phase);
4239     const TypePtr* atp = TypeRawPtr::BOTTOM;
4240 
4241     // One or two coalesced stores to plop down.
4242     Node*    st[2];
4243     intptr_t off[2];
4244     int  nst = 0;
4245     if (!split) {
4246       ++new_long;
4247       off[nst] = offset;
4248       st[nst++] = StoreNode::make(*phase, ctl, zmem, adr, atp,
4249                                   phase->longcon(con), T_LONG, MemNode::unordered);
4250     } else {
4251       // Omit either if it is a zero.
4252       if (con0 != 0) {
4253         ++new_int;
4254         off[nst]  = offset;
4255         st[nst++] = StoreNode::make(*phase, ctl, zmem, adr, atp,
4256                                     phase->intcon(con0), T_INT, MemNode::unordered);
4257       }
4258       if (con1 != 0) {
4259         ++new_int;
4260         offset += BytesPerInt;
4261         adr = make_raw_address(offset, phase);
4262         off[nst]  = offset;
4263         st[nst++] = StoreNode::make(*phase, ctl, zmem, adr, atp,
4264                                     phase->intcon(con1), T_INT, MemNode::unordered);
4265       }
4266     }
4267 
4268     // Insert second store first, then the first before the second.
4269     // Insert each one just before any overlapping non-constant stores.
4270     while (nst > 0) {
4271       Node* st1 = st[--nst];
4272       C->copy_node_notes_to(st1, old);
4273       st1 = phase->transform(st1);
4274       offset = off[nst];
4275       assert(offset >= header_size, "do not smash header");
4276       int ins_idx = captured_store_insertion_point(offset, /*size:*/0, phase);
4277       guarantee(ins_idx != 0, "must re-insert constant store");
4278       if (ins_idx < 0)  ins_idx = -ins_idx;  // never overlap
4279       if (ins_idx > InitializeNode::RawStores && in(ins_idx-1) == zmem)
4280         set_req(--ins_idx, st1);
4281       else
4282         ins_req(ins_idx, st1);
4283     }
4284   }
4285 
4286   if (PrintCompilation && WizardMode)
4287     tty->print_cr("Changed %d/%d subword/long constants into %d/%d int/long",
4288                   old_subword, old_long, new_int, new_long);

4290     C->log()->elem("comment that='%d/%d subword/long to %d/%d int/long'",
4291                    old_subword, old_long, new_int, new_long);
4292 
4293   // Clean up any remaining occurrences of zmem:
4294   remove_extra_zeroes();
4295 }

4297 // Explore forward from in(start) to find the first fully initialized
4298 // word, and return its offset.  Skip groups of subword stores which
4299 // together initialize full words.  If in(start) is itself part of a
4300 // fully initialized word, return the offset of in(start).  If there
4301 // are no following full-word stores, or if something is fishy, return
4302 // a negative value.
4303 intptr_t InitializeNode::find_next_fullword_store(uint start, PhaseGVN* phase) {
4304   int       int_map = 0;
4305   intptr_t  int_map_off = 0;
4306   const int FULL_MAP = right_n_bits(BytesPerInt);  // the int_map we hope for
4307 
4308   for (uint i = start, limit = req(); i < limit; i++) {
4309     Node* st = in(i);
4310 
4311     intptr_t st_off = get_store_offset(st, phase);
4312     if (st_off < 0)  break;  // return conservative answer
4313 
4314     int st_size = st->as_Store()->memory_size();
4315     if (st_size >= BytesPerInt && (st_off % BytesPerInt) == 0) {
4316       return st_off;            // we found a complete word init
4317     }
4318 
4319     // update the map:
4320 
4321     intptr_t this_int_off = align_down(st_off, BytesPerInt);
4322     if (this_int_off != int_map_off) {
4323       // reset the map:
4324       int_map = 0;
4325       int_map_off = this_int_off;
4326     }
4327 
4328     int subword_off = st_off - this_int_off;
4329     int_map |= right_n_bits(st_size) << subword_off;
4330     if ((int_map & FULL_MAP) == FULL_MAP) {
4331       return this_int_off;      // we found a complete word init
4332     }
4333 
4334     // Did this store hit or cross the word boundary?
4335     intptr_t next_int_off = align_down(st_off + st_size, BytesPerInt);
4336     if (next_int_off == this_int_off + BytesPerInt) {
4337       // We passed the current int, without fully initializing it.
4338       int_map_off = next_int_off;
4339       int_map >>= BytesPerInt;
4340     } else if (next_int_off > this_int_off + BytesPerInt) {
4341       // We passed the current and next int.
4342       return this_int_off + BytesPerInt;
4343     }
4344   }
4345 
4346   return -1;
4347 }

4350 // Called when the associated AllocateNode is expanded into CFG.
4351 // At this point, we may perform additional optimizations.
4352 // Linearize the stores by ascending offset, to make memory
4353 // activity as coherent as possible.
4354 Node* InitializeNode::complete_stores(Node* rawctl, Node* rawmem, Node* rawptr,
4355                                       intptr_t header_size,
4356                                       Node* size_in_bytes,
4357                                       PhaseIterGVN* phase) {
4358   assert(!is_complete(), "not already complete");
4359   assert(stores_are_sane(phase), "");
4360   assert(allocation() != NULL, "must be present");
4361 
4362   remove_extra_zeroes();
4363 
4364   if (ReduceFieldZeroing || ReduceBulkZeroing)
4365     // reduce instruction count for common initialization patterns
4366     coalesce_subword_stores(header_size, size_in_bytes, phase);
4367 
4368   Node* zmem = zero_memory();   // initially zero memory state
4369   Node* inits = zmem;           // accumulating a linearized chain of inits
4370   #ifdef ASSERT
4371   intptr_t first_offset = allocation()->minimum_header_size();
4372   intptr_t last_init_off = first_offset;  // previous init offset
4373   intptr_t last_init_end = first_offset;  // previous init offset+size
4374   intptr_t last_tile_end = first_offset;  // previous tile offset+size
4375   #endif
4376   intptr_t zeroes_done = header_size;
4377 
4378   bool do_zeroing = true;       // we might give up if inits are very sparse
4379   int  big_init_gaps = 0;       // how many large gaps have we seen?
4380 
4381   if (UseTLAB && ZeroTLAB)  do_zeroing = false;
4382   if (!ReduceFieldZeroing && !ReduceBulkZeroing)  do_zeroing = false;
4383 
4384   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
4385     Node* st = in(i);
4386     intptr_t st_off = get_store_offset(st, phase);
4387     if (st_off < 0)
4388       break;                    // unknown junk in the inits
4389     if (st->in(MemNode::Memory) != zmem)
4390       break;                    // complicated store chains somehow in list
4391 
4392     int st_size = st->as_Store()->memory_size();
4393     intptr_t next_init_off = st_off + st_size;
4394 
4395     if (do_zeroing && zeroes_done < next_init_off) {
4396       // See if this store needs a zero before it or under it.
4397       intptr_t zeroes_needed = st_off;
4398 
4399       if (st_size < BytesPerInt) {
4400         // Look for subword stores which only partially initialize words.
4401         // If we find some, we must lay down some word-level zeroes first,
4402         // underneath the subword stores.
4403         //
4404         // Examples:
4405         //   byte[] a = { p,q,r,s }  =>  a[0]=p,a[1]=q,a[2]=r,a[3]=s
4406         //   byte[] a = { x,y,0,0 }  =>  a[0..3] = 0, a[0]=x,a[1]=y
4407         //   byte[] a = { 0,0,z,0 }  =>  a[0..3] = 0, a[2]=z
4408         //
4409         // Note:  coalesce_subword_stores may have already done this,
4410         // if it was prompted by constant non-zero subword initializers.
4411         // But this case can still arise with non-constant stores.
4412 
4413         intptr_t next_full_store = find_next_fullword_store(i, phase);
4414 
4415         // In the examples above:
4416         //   in(i)          p   q   r   s     x   y     z
4417         //   st_off        12  13  14  15    12  13    14
4418         //   st_size        1   1   1   1     1   1     1
4419         //   next_full_s.  12  16  16  16    16  16    16
4420         //   z's_done      12  16  16  16    12  16    12
4421         //   z's_needed    12  16  16  16    16  16    16
4422         //   zsize          0   0   0   0     4   0     4
4423         if (next_full_store < 0) {
4424           // Conservative tack:  Zero to end of current word.
4425           zeroes_needed = align_up(zeroes_needed, BytesPerInt);
4426         } else {
4427           // Zero to beginning of next fully initialized word.
4428           // Or, don't zero at all, if we are already in that word.
4429           assert(next_full_store >= zeroes_needed, "must go forward");
4430           assert((next_full_store & (BytesPerInt-1)) == 0, "even boundary");
4431           zeroes_needed = next_full_store;
4432         }
4433       }
4434 
4435       if (zeroes_needed > zeroes_done) {
4436         intptr_t zsize = zeroes_needed - zeroes_done;
4437         // Do some incremental zeroing on rawmem, in parallel with inits.
4438         zeroes_done = align_down(zeroes_done, BytesPerInt);
4439         rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4440                                               allocation()->in(AllocateNode::DefaultValue),
4441                                               allocation()->in(AllocateNode::RawDefaultValue),
4442                                               zeroes_done, zeroes_needed,
4443                                               phase);
4444         zeroes_done = zeroes_needed;
4445         if (zsize > InitArrayShortSize && ++big_init_gaps > 2)
4446           do_zeroing = false;   // leave the hole, next time
4447       }
4448     }
4449 
4450     // Collect the store and move on:
4451     phase->replace_input_of(st, MemNode::Memory, inits);
4452     inits = st;                 // put it on the linearized chain
4453     set_req(i, zmem);           // unhook from previous position
4454 
4455     if (zeroes_done == st_off)
4456       zeroes_done = next_init_off;
4457 
4458     assert(!do_zeroing || zeroes_done >= next_init_off, "don't miss any");
4459 
4460     #ifdef ASSERT
4461     // Various order invariants.  Weaker than stores_are_sane because
4462     // a large constant tile can be filled in by smaller non-constant stores.
4463     assert(st_off >= last_init_off, "inits do not reverse");
4464     last_init_off = st_off;
4465     const Type* val = NULL;
4466     if (st_size >= BytesPerInt &&
4467         (val = phase->type(st->in(MemNode::ValueIn)))->singleton() &&
4468         (int)val->basic_type() < (int)T_OBJECT) {
4469       assert(st_off >= last_tile_end, "tiles do not overlap");
4470       assert(st_off >= last_init_end, "tiles do not overwrite inits");
4471       last_tile_end = MAX2(last_tile_end, next_init_off);
4472     } else {
4473       intptr_t st_tile_end = align_up(next_init_off, BytesPerLong);
4474       assert(st_tile_end >= last_tile_end, "inits stay with tiles");
4475       assert(st_off      >= last_init_end, "inits do not overlap");
4476       last_init_end = next_init_off;  // it's a non-tile
4477     }
4478     #endif //ASSERT
4479   }
4480 
4481   remove_extra_zeroes();        // clear out all the zmems left over
4482   add_req(inits);
4483 
4484   if (!(UseTLAB && ZeroTLAB)) {
4485     // If anything remains to be zeroed, zero it all now.
4486     zeroes_done = align_down(zeroes_done, BytesPerInt);
4487     // if it is the last unused 4 bytes of an instance, forget about it
4488     intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, max_jint);
4489     if (zeroes_done + BytesPerLong >= size_limit) {
4490       AllocateNode* alloc = allocation();
4491       assert(alloc != NULL, "must be present");
4492       if (alloc != NULL && alloc->Opcode() == Op_Allocate) {
4493         Node* klass_node = alloc->in(AllocateNode::KlassNode);
4494         ciKlass* k = phase->type(klass_node)->is_klassptr()->klass();
4495         if (zeroes_done == k->layout_helper())
4496           zeroes_done = size_limit;
4497       }
4498     }
4499     if (zeroes_done < size_limit) {
4500       rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4501                                             allocation()->in(AllocateNode::DefaultValue),
4502                                             allocation()->in(AllocateNode::RawDefaultValue),
4503                                             zeroes_done, size_in_bytes, phase);
4504     }
4505   }
4506 
4507   set_complete(phase);
4508   return rawmem;
4509 }

4537 //============================MergeMemNode=====================================
4538 //
4539 // SEMANTICS OF MEMORY MERGES:  A MergeMem is a memory state assembled from several
4540 // contributing store or call operations.  Each contributor provides the memory
4541 // state for a particular "alias type" (see Compile::alias_type).  For example,
4542 // if a MergeMem has an input X for alias category #6, then any memory reference
4543 // to alias category #6 may use X as its memory state input, as an exact equivalent
4544 // to using the MergeMem as a whole.
4545 //   Load<6>( MergeMem(<6>: X, ...), p ) <==> Load<6>(X,p)
4546 //
4547 // (Here, the <N> notation gives the index of the relevant adr_type.)
4548 //
4549 // In one special case (and more cases in the future), alias categories overlap.
4550 // The special alias category "Bot" (Compile::AliasIdxBot) includes all memory
4551 // states.  Therefore, if a MergeMem has only one contributing input W for Bot,
4552 // it is exactly equivalent to that state W:
4553 //   MergeMem(<Bot>: W) <==> W
4554 //
4555 // Usually, the merge has more than one input.  In that case, where inputs
4556 // overlap (i.e., one is Bot), the narrower alias type determines the memory
4557 // state for that type, and the wider alias type (Bot) fills in everywhere else:
4558 //   Load<5>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<5>(W,p)
4559 //   Load<6>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<6>(X,p)
4560 //
4561 // A merge can take a "wide" memory state as one of its narrow inputs.
4562 // This simply means that the merge observes out only the relevant parts of
4563 // the wide input.  That is, wide memory states arriving at narrow merge inputs
4564 // are implicitly "filtered" or "sliced" as necessary.  (This is rare.)
4565 //
4566 // These rules imply that MergeMem nodes may cascade (via their <Bot> links),
4567 // and that memory slices "leak through":
4568 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y)) <==> MergeMem(<Bot>: W, <7>: Y)
4569 //
4570 // But, in such a cascade, repeated memory slices can "block the leak":
4571 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y), <7>: Y') <==> MergeMem(<Bot>: W, <7>: Y')
4572 //
4573 // In the last example, Y is not part of the combined memory state of the
4574 // outermost MergeMem.  The system must, of course, prevent unschedulable
4575 // memory states from arising, so you can be sure that the state Y is somehow
4576 // a precursor to state Y'.
4577 //
4578 //
4579 // REPRESENTATION OF MEMORY MERGES: The indexes used to address the Node::in array
4580 // of each MergeMemNode array are exactly the numerical alias indexes, including
4581 // but not limited to AliasIdxTop, AliasIdxBot, and AliasIdxRaw.  The functions
4582 // Compile::alias_type (and kin) produce and manage these indexes.
4583 //
4584 // By convention, the value of in(AliasIdxTop) (i.e., in(1)) is always the top node.
4585 // (Note that this provides quick access to the top node inside MergeMem methods,
4586 // without the need to reach out via TLS to Compile::current.)
4587 //
4588 // As a consequence of what was just described, a MergeMem that represents a full
4589 // memory state has an edge in(AliasIdxBot) which is a "wide" memory state,
4590 // containing all alias categories.
4591 //
4592 // MergeMem nodes never (?) have control inputs, so in(0) is NULL.
4593 //
4594 // All other edges in(N) (including in(AliasIdxRaw), which is in(3)) are either
4595 // a memory state for the alias type <N>, or else the top node, meaning that
4596 // there is no particular input for that alias type.  Note that the length of
4597 // a MergeMem is variable, and may be extended at any time to accommodate new
4598 // memory states at larger alias indexes.  When merges grow, they are of course
4599 // filled with "top" in the unused in() positions.
4600 //
4601 // This use of top is named "empty_memory()", or "empty_mem" (no-memory) as a variable.
4602 // (Top was chosen because it works smoothly with passes like GCM.)
4603 //
4604 // For convenience, we hardwire the alias index for TypeRawPtr::BOTTOM.  (It is
4605 // the type of random VM bits like TLS references.)  Since it is always the
4606 // first non-Bot memory slice, some low-level loops use it to initialize an
4607 // index variable:  for (i = AliasIdxRaw; i < req(); i++).
4608 //
4609 //
4610 // ACCESSORS:  There is a special accessor MergeMemNode::base_memory which returns
4611 // the distinguished "wide" state.  The accessor MergeMemNode::memory_at(N) returns
4612 // the memory state for alias type <N>, or (if there is no particular slice at <N>,
4613 // it returns the base memory.  To prevent bugs, memory_at does not accept <Top>
4614 // or <Bot> indexes.  The iterator MergeMemStream provides robust iteration over
4615 // MergeMem nodes or pairs of such nodes, ensuring that the non-top edges are visited.
4616 //
4617 // %%%% We may get rid of base_memory as a separate accessor at some point; it isn't
4618 // really that different from the other memory inputs.  An abbreviation called
4619 // "bot_memory()" for "memory_at(AliasIdxBot)" would keep code tidy.
4620 //
4621 //
4622 // PARTIAL MEMORY STATES:  During optimization, MergeMem nodes may arise that represent
4623 // partial memory states.  When a Phi splits through a MergeMem, the copy of the Phi
4624 // that "emerges though" the base memory will be marked as excluding the alias types
4625 // of the other (narrow-memory) copies which "emerged through" the narrow edges:
4626 //
4627 //   Phi<Bot>(U, MergeMem(<Bot>: W, <8>: Y))
4628 //     ==Ideal=>  MergeMem(<Bot>: Phi<Bot-8>(U, W), Phi<8>(U, Y))
4629 //
4630 // This strange "subtraction" effect is necessary to ensure IGVN convergence.
4631 // (It is currently unimplemented.)  As you can see, the resulting merge is
4632 // actually a disjoint union of memory states, rather than an overlay.
4633 //

4695 //------------------------------Ideal------------------------------------------
4696 // This method is invoked recursively on chains of MergeMem nodes
4697 Node *MergeMemNode::Ideal(PhaseGVN *phase, bool can_reshape) {
4698   // Remove chain'd MergeMems
4699   //
4700   // This is delicate, because the each "in(i)" (i >= Raw) is interpreted
4701   // relative to the "in(Bot)".  Since we are patching both at the same time,
4702   // we have to be careful to read each "in(i)" relative to the old "in(Bot)",
4703   // but rewrite each "in(i)" relative to the new "in(Bot)".
4704   Node *progress = NULL;
4705 
4706 
4707   Node* old_base = base_memory();
4708   Node* empty_mem = empty_memory();
4709   if (old_base == empty_mem)
4710     return NULL; // Dead memory path.
4711 
4712   MergeMemNode* old_mbase;
4713   if (old_base != NULL && old_base->is_MergeMem())
4714     old_mbase = old_base->as_MergeMem();
4715   else
4716     old_mbase = NULL;
4717   Node* new_base = old_base;
4718 
4719   // simplify stacked MergeMems in base memory
4720   if (old_mbase)  new_base = old_mbase->base_memory();
4721 
4722   // the base memory might contribute new slices beyond my req()
4723   if (old_mbase)  grow_to_match(old_mbase);
4724 
4725   // Look carefully at the base node if it is a phi.
4726   PhiNode* phi_base;
4727   if (new_base != NULL && new_base->is_Phi())
4728     phi_base = new_base->as_Phi();
4729   else
4730     phi_base = NULL;
4731 
4732   Node*    phi_reg = NULL;
4733   uint     phi_len = (uint)-1;
4734   if (phi_base != NULL && !phi_base->is_copy()) {
4735     // do not examine phi if degraded to a copy
4736     phi_reg = phi_base->region();
4737     phi_len = phi_base->req();
4738     // see if the phi is unfinished
4739     for (uint i = 1; i < phi_len; i++) {
4740       if (phi_base->in(i) == NULL) {
4741         // incomplete phi; do not look at it yet!
4742         phi_reg = NULL;
4743         phi_len = (uint)-1;
4744         break;
4745       }
4746     }
4747   }
4748 
4749   // Note:  We do not call verify_sparse on entry, because inputs
4750   // can normalize to the base_memory via subsume_node or similar
4751   // mechanisms.  This method repairs that damage.
4752 
4753   assert(!old_mbase || old_mbase->is_empty_memory(empty_mem), "consistent sentinels");
4754 
4755   // Look at each slice.
4756   for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4757     Node* old_in = in(i);
4758     // calculate the old memory value
4759     Node* old_mem = old_in;
4760     if (old_mem == empty_mem)  old_mem = old_base;
4761     assert(old_mem == memory_at(i), "");
4762 
4763     // maybe update (reslice) the old memory value
4764 
4765     // simplify stacked MergeMems
4766     Node* new_mem = old_mem;
4767     MergeMemNode* old_mmem;
4768     if (old_mem != NULL && old_mem->is_MergeMem())
4769       old_mmem = old_mem->as_MergeMem();
4770     else
4771       old_mmem = NULL;
4772     if (old_mmem == this) {
4773       // This can happen if loops break up and safepoints disappear.
4774       // A merge of BotPtr (default) with a RawPtr memory derived from a
4775       // safepoint can be rewritten to a merge of the same BotPtr with
4776       // the BotPtr phi coming into the loop.  If that phi disappears
4777       // also, we can end up with a self-loop of the mergemem.
4778       // In general, if loops degenerate and memory effects disappear,
4779       // a mergemem can be left looking at itself.  This simply means
4780       // that the mergemem's default should be used, since there is
4781       // no longer any apparent effect on this slice.
4782       // Note: If a memory slice is a MergeMem cycle, it is unreachable
4783       //       from start.  Update the input to TOP.
4784       new_mem = (new_base == this || new_base == empty_mem)? empty_mem : new_base;
4785     }
4786     else if (old_mmem != NULL) {
4787       new_mem = old_mmem->memory_at(i);
4788     }
4789     // else preceding memory was not a MergeMem
4790 
4791     // replace equivalent phis (unfortunately, they do not GVN together)
4792     if (new_mem != NULL && new_mem != new_base &&
4793         new_mem->req() == phi_len && new_mem->in(0) == phi_reg) {
4794       if (new_mem->is_Phi()) {
4795         PhiNode* phi_mem = new_mem->as_Phi();
4796         for (uint i = 1; i < phi_len; i++) {
4797           if (phi_base->in(i) != phi_mem->in(i)) {
4798             phi_mem = NULL;
4799             break;
4800           }
4801         }
4802         if (phi_mem != NULL) {
4803           // equivalent phi nodes; revert to the def
4804           new_mem = new_base;
4805         }
4806       }
4807     }
4808 
4809     // maybe store down a new value
4810     Node* new_in = new_mem;
4811     if (new_in == new_base)  new_in = empty_mem;
4812 
4813     if (new_in != old_in) {
4814       // Warning:  Do not combine this "if" with the previous "if"
4815       // A memory slice might have be be rewritten even if it is semantically
4816       // unchanged, if the base_memory value has changed.
4817       set_req(i, new_in);
4818       progress = this;          // Report progress
4819     }
4820   }
4821 
4822   if (new_base != old_base) {
4823     set_req(Compile::AliasIdxBot, new_base);
4824     // Don't use set_base_memory(new_base), because we need to update du.
4825     assert(base_memory() == new_base, "");
4826     progress = this;
4827   }
4828 
4829   if( base_memory() == this ) {
4830     // a self cycle indicates this memory path is dead
4831     set_req(Compile::AliasIdxBot, empty_mem);
4832   }
4833 
4834   // Resolve external cycles by calling Ideal on a MergeMem base_memory
4835   // Recursion must occur after the self cycle check above
4836   if( base_memory()->is_MergeMem() ) {
4837     MergeMemNode *new_mbase = base_memory()->as_MergeMem();
4838     Node *m = phase->transform(new_mbase);  // Rollup any cycles
4839     if( m != NULL &&
4840         (m->is_top() ||
4841          (m->is_MergeMem() && m->as_MergeMem()->base_memory() == empty_mem)) ) {
4842       // propagate rollup of dead cycle to self
4843       set_req(Compile::AliasIdxBot, empty_mem);
4844     }
4845   }
4846 
4847   if( base_memory() == empty_mem ) {
4848     progress = this;
4849     // Cut inputs during Parse phase only.
4850     // During Optimize phase a dead MergeMem node will be subsumed by Top.
4851     if( !can_reshape ) {
4852       for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4853         if( in(i) != empty_mem ) { set_req(i, empty_mem); }
4854       }
4855     }
4856   }
4857 
4858   if( !progress && base_memory()->is_Phi() && can_reshape ) {
4859     // Check if PhiNode::Ideal's "Split phis through memory merges"
4860     // transform should be attempted. Look for this->phi->this cycle.
4861     uint merge_width = req();
4862     if (merge_width > Compile::AliasIdxRaw) {
4863       PhiNode* phi = base_memory()->as_Phi();
4864       for( uint i = 1; i < phi->req(); ++i ) {// For all paths in
4865         if (phi->in(i) == this) {
4866           phase->is_IterGVN()->_worklist.push(phi);
4867           break;
4868         }
4869       }
4870     }
4871   }
4872 
4873   assert(progress || verify_sparse(), "please, no dups of base");
4874   return progress;
4875 }

4920 static void verify_memory_slice(const MergeMemNode* m, int alias_idx, Node* n) {
4921   if (!VerifyAliases)                return;  // don't bother to verify unless requested
4922   if (VMError::is_error_reported())  return;  // muzzle asserts when debugging an error
4923   if (Node::in_dump())               return;  // muzzle asserts when printing
4924   assert(alias_idx >= Compile::AliasIdxRaw, "must not disturb base_memory or sentinel");
4925   assert(n != NULL, "");
4926   // Elide intervening MergeMem's
4927   while (n->is_MergeMem()) {
4928     n = n->as_MergeMem()->memory_at(alias_idx);
4929   }
4930   Compile* C = Compile::current();
4931   const TypePtr* n_adr_type = n->adr_type();
4932   if (n == m->empty_memory()) {
4933     // Implicit copy of base_memory()
4934   } else if (n_adr_type != TypePtr::BOTTOM) {
4935     assert(n_adr_type != NULL, "new memory must have a well-defined adr_type");
4936     assert(C->must_alias(n_adr_type, alias_idx), "new memory must match selected slice");
4937   } else {
4938     // A few places like make_runtime_call "know" that VM calls are narrow,
4939     // and can be used to update only the VM bits stored as TypeRawPtr::BOTTOM.
4940     bool expected_wide_mem = false;
4941     if (n == m->base_memory()) {
4942       expected_wide_mem = true;
4943     } else if (alias_idx == Compile::AliasIdxRaw ||
4944                n == m->memory_at(Compile::AliasIdxRaw)) {
4945       expected_wide_mem = true;
4946     } else if (!C->alias_type(alias_idx)->is_rewritable()) {
4947       // memory can "leak through" calls on channels that
4948       // are write-once.  Allow this also.
4949       expected_wide_mem = true;
4950     }
4951     assert(expected_wide_mem, "expected narrow slice replacement");
4952   }
4953 }

4972            || n == NULL || n->bottom_type() == Type::TOP
4973            || n->adr_type() == NULL // address is TOP
