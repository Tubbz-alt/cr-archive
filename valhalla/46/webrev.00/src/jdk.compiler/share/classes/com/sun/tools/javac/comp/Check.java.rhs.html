 233             }
 234         }

 330     /** Report duplicate declaration error.
 331      */
 332     void duplicateError(DiagnosticPosition pos, Symbol sym) {
 333         if (!sym.type.isErroneous()) {
 334             Symbol location = sym.location();
 335             if (location.kind == MTH &&
 336                     ((MethodSymbol)location).isStaticOrInstanceInit()) {
 337                 log.error(pos,
 338                           Errors.AlreadyDefinedInClinit(kindName(sym),
 339                                                         sym,
 340                                                         kindName(sym.location()),
 341                                                         kindName(sym.location().enclClass()),
 342                                                         sym.location().enclClass()));
 343             } else {
 344                 /* dont error if this is a duplicated parameter of a generated canonical constructor
 345                  * as we should have issued an error for the duplicated fields
 346                  */
 347                 if (location.kind != MTH ||
 348                         ((sym.owner.flags_field & GENERATEDCONSTR) == 0) ||
 349                         ((sym.owner.flags_field & RECORD) == 0)) {
 350                     log.error(pos,
 351                             Errors.AlreadyDefined(kindName(sym),
 352                                     sym,
 353                                     kindName(sym.location()),
 354                                     sym.location()));
 355                 }
 356             }
 357         }
 358     }

 372     /** Check that variable does not hide variable with same name in
 373      *  immediately enclosing local scope.
 374      *  @param pos           Position for error reporting.
 375      *  @param v             The symbol.
 376      *  @param s             The scope.
 377      */
 378     void checkTransparentVar(DiagnosticPosition pos, VarSymbol v, Scope s) {
 379         for (Symbol sym : s.getSymbolsByName(v.name)) {
 380             if (sym.owner != v.owner) break;
 381             if (sym.kind == VAR &&
 382                 sym.owner.kind.matches(KindSelector.VAL_MTH) &&
 383                 v.name != names.error) {
 384                 duplicateError(pos, sym);
 385                 return;
 386             }
 387         }
 388     }
 389 
 390     /** Check that a class or interface does not hide a class or
 391      *  interface with same name in immediately enclosing local scope.
 392      *  @param pos           Position for error reporting.
 393      *  @param c             The symbol.
 394      *  @param s             The scope.
 395      */
 396     void checkTransparentClass(DiagnosticPosition pos, ClassSymbol c, Scope s) {
 397         for (Symbol sym : s.getSymbolsByName(c.name)) {
 398             if (sym.owner != c.owner) break;
 399             if (sym.kind == TYP && !sym.type.hasTag(TYPEVAR) &&
 400                 sym.owner.kind.matches(KindSelector.VAL_MTH) &&
 401                 c.name != names.error) {
 402                 duplicateError(pos, sym);
 403                 return;
 404             }
 405         }
 406     }

 408     /** Check that class does not have the same name as one of
 409      *  its enclosing classes, or as a class defined in its enclosing scope.
 410      *  return true if class is unique in its enclosing scope.
 411      *  @param pos           Position for error reporting.
 412      *  @param name          The class name.
 413      *  @param s             The enclosing scope.
 414      */
 415     boolean checkUniqueClassName(DiagnosticPosition pos, Name name, Scope s) {
 416         for (Symbol sym : s.getSymbolsByName(name, NON_RECURSIVE)) {
 417             if (sym.kind == TYP && sym.name != names.error) {
 418                 duplicateError(pos, sym);
 419                 return false;
 420             }
 421         }
 422         for (Symbol sym = s.owner; sym != null; sym = sym.owner) {
 423             if (sym.kind == TYP && sym.name == name && sym.name != names.error) {
 424                 duplicateError(pos, sym);
 425                 return true;
 426             }
 427         }
 428         return true;
 429     }

 438     /** Return name of local class.
 439      *  This is of the form   {@code <enclClass> $ n <classname> }
 440      *  where
 441      *    enclClass is the flat name of the enclosing class,
 442      *    classname is the simple name of the local class
 443      */
 444     public Name localClassName(ClassSymbol c) {
 445         Name enclFlatname = c.owner.enclClass().flatname;
 446         String enclFlatnameStr = enclFlatname.toString();
 447         Pair<Name, Name> key = new Pair<>(enclFlatname, c.name);
 448         Integer index = localClassNameIndexes.get(key);
 449         for (int i = (index == null) ? 1 : index; ; i++) {
 450             Name flatname = names.fromString(enclFlatnameStr
 451                     + syntheticNameChar + i + c.name);
 452             if (getCompiled(c.packge().modle, flatname) == null) {
 453                 localClassNameIndexes.put(key, i + 1);
 454                 return flatname;
 455             }
 456         }
 457     }
 458 
 459     public void clearLocalClassNameIndexes(ClassSymbol c) {
 460         if (c.owner != null && c.owner.kind != NIL) {
 461             localClassNameIndexes.remove(new Pair<>(
 462                     c.owner.enclClass().flatname, c.name));
 463         }
 464     }

 595     Type checkType(final DiagnosticPosition pos, final Type found, final Type req, final CheckContext checkContext) {
 596         final InferenceContext inferenceContext = checkContext.inferenceContext();
 597         if (inferenceContext.free(req) || inferenceContext.free(found)) {
 598             inferenceContext.addFreeTypeListener(List.of(req, found),
 599                     solvedContext -> checkType(pos, solvedContext.asInstType(found), solvedContext.asInstType(req), checkContext));
 600         } else {
 601             if (found.hasTag(CLASS)) {
 602                 if (inferenceContext != infer.emptyContext)
 603                     checkParameterizationWithValues(pos, found);
 604             }
 605         }
 606         if (req.hasTag(ERROR))
 607             return req;
 608         if (req.hasTag(NONE))
 609             return found;
 610         if (checkContext.compatible(found, req, checkContext.checkWarner(pos, found, req))) {
 611             if (found.hasTag(BOT) && types.isValueBased(req)) {
 612                 log.warning(pos, Warnings.SuspiciousMixOfNullWithValueBasedClass(req));
 613             }
 614             return found;
 615         } else {
 616             if (found.isNumeric() && req.isNumeric()) {
 617                 checkContext.report(pos, diags.fragment(Fragments.PossibleLossOfPrecision(found, req)));
 618                 return types.createErrorType(found);
 619             }
 620             checkContext.report(pos, diags.fragment(Fragments.InconvertibleTypes(found, req)));
 621             return types.createErrorType(found);
 622         }
 623     }

 625     /** Check that a given type can be cast to a given target type.
 626      *  Return the result of the cast.

 628      *  @param found      The type that is being cast.
 629      *  @param req        The target type of the cast.

 634     Type checkCastable(DiagnosticPosition pos, Type found, Type req, CheckContext checkContext) {
 635         if (types.isCastable(found, req, castWarner(pos, found, req))) {
 636             if (types.isValueBased(req)) {
 637                 if (found.hasTag(BOT)) {
 638                     log.warning(pos, Warnings.SuspiciousMixOfNullWithValueBasedClass(req));
 639                 } else if (!types.isValueBased(found)) {
 640                     log.warning(pos, Warnings.PotentialNullPollution(found));
 641                 }

1137             final Symbol sym,

1173                 nonInferred = nonInferred.tail;
1174             }

1176                 Type varArg = types.elemtype(last);
1177                 while (args.tail != null) {
1178                     JCTree arg = args.head;
1179                     Warner warn = convertWarner(arg.pos(), arg.type, varArg);
1180                     assertConvertible(arg, arg.type, varArg, warn);
1181                     args = args.tail;
1182                 }
1183             } else if ((sym.flags() & (VARARGS | SIGNATURE_POLYMORPHIC)) == VARARGS) {
1184                 // non-varargs call to varargs method

1197                  !isTrustMeAllowedOnMethod(sym))) {
1198                 warnUnchecked(env.tree.pos(), Warnings.UncheckedGenericArrayCreation(argtype));
1199             }
1200             TreeInfo.setVarargsElement(env.tree, types.elemtype(argtype));
1201          }
1202          return owntype;

1226         private Type firstIncompatibleTypeArg(Type type) {
1227             List<Type> formals = type.tsym.type.allparams();
1228             List<Type> actuals = type.allparams();
1229             List<Type> args = type.getTypeArguments();
1230             List<Type> forms = type.tsym.type.getTypeArguments();
1231             ListBuffer<Type> bounds_buf = new ListBuffer<>();
1232 
1233             // For matching pairs of actual argument types `a' and
1234             // formal type parameters with declared bound `b' ...
1235             while (args.nonEmpty() && forms.nonEmpty()) {
1236                 // exact type arguments needs to know their
1237                 // bounds (for upper and lower bound
1238                 // calculations).  So we create new bounds where
1239                 // type-parameters are replaced with actuals argument types.
1240                 bounds_buf.append(types.subst(forms.head.getUpperBound(), formals, actuals));
1241                 args = args.tail;
1242                 forms = forms.tail;
1243             }
1244 
1245             args = type.getTypeArguments();
1246             List<Type> tvars_cap = types.substBounds(formals,
1247                                       formals,
1248                                       types.capture(type).allparams());
1249             while (args.nonEmpty() && tvars_cap.nonEmpty()) {
1250                 // Let the actual arguments know their bound
1251                 args.head.withTypeVar((TypeVar)tvars_cap.head);
1252                 args = args.tail;
1253                 tvars_cap = tvars_cap.tail;
1254             }
1255 
1256             args = type.getTypeArguments();
1257             List<Type> bounds = bounds_buf.toList();
1258 
1259             while (args.nonEmpty() && bounds.nonEmpty()) {
1260                 Type actual = args.head;
1261                 if (!isTypeArgErroneous(actual) &&
1262                         !bounds.head.isErroneous() &&
1263                         !checkExtends(actual, bounds.head)) {
1264                     return args.head;
1265                 }
1266                 args = args.tail;
1267                 bounds = bounds.tail;
1268             }
1269 
1270             args = type.getTypeArguments();
1271             bounds = bounds_buf.toList();
1272 
1273             for (Type arg : types.capture(type).getTypeArguments()) {
1274                 if (arg.hasTag(TYPEVAR) &&
1275                         arg.getUpperBound().isErroneous() &&
1276                         !bounds.head.isErroneous() &&
1277                         !isTypeArgErroneous(args.head)) {
1278                     return args.head;
1279                 }
1280                 bounds = bounds.tail;
1281                 args = args.tail;
1282             }
1283 
1284             return null;
1285         }

1288             return isTypeArgErroneous.visit(t);
1289         }

1310     /** Check that given modifiers are legal for given symbol and
1311      *  return modifiers together with any implicit modifiers for that symbol.
1312      *  Warning: we can't use flags() here since this method
1313      *  is called during class enter, when flags() would cause a premature
1314      *  completion.
1315      *  @param pos           Position to be used for error reporting.
1316      *  @param flags         The set of modifiers given in a definition.
1317      *  @param sym           The defined symbol.
1318      */
1319     long checkFlags(DiagnosticPosition pos, long flags, Symbol sym, JCTree tree) {
1320         long mask;
1321         long implicit = 0;
1322 
1323         switch (sym.kind) {
1324         case VAR:
1325             if (TreeInfo.isReceiverParam(tree))
1326                 mask = ReceiverParamFlags;
1327             else if (sym.owner.kind != TYP)
1328                 mask = LocalVarFlags;
1329             else if ((sym.owner.flags_field & INTERFACE) != 0)
1330                 mask = implicit = InterfaceVarFlags;
1331             else {
1332                 mask = VarFlags;
1333                 if (types.isValue(sym.owner.type) && (flags & STATIC) == 0) {
1334                     implicit |= FINAL;
1335                 }
1336             }
1337             break;
1338         case MTH:
1339             if (sym.name == names.init) {
1340                 if ((sym.owner.flags_field & ENUM) != 0) {
1341                     // enum constructors cannot be declared public or
1342                     // protected and must be implicitly or explicitly
1343                     // private
1344                     implicit = PRIVATE;
1345                     mask = PRIVATE;
1346                 } else
1347                     mask = ConstructorFlags;
1348             }  else if ((sym.owner.flags_field & INTERFACE) != 0) {
1349                 if ((sym.owner.flags_field & ANNOTATION) != 0) {
1350                     mask = AnnotationTypeElementMask;
1351                     implicit = PUBLIC | ABSTRACT;
1352                 } else if ((flags & (DEFAULT | STATIC | PRIVATE)) != 0) {
1353                     mask = InterfaceMethodMask;
1354                     implicit = (flags & PRIVATE) != 0 ? 0 : PUBLIC;
1355                     if ((flags & DEFAULT) != 0) {
1356                         implicit |= ABSTRACT;
1357                     }
1358                 } else {
1359                     mask = implicit = InterfaceMethodFlags;
1360                 }
1361             } else if ((sym.owner.flags_field & RECORD) != 0) {
1362                 mask = RecordMethodFlags;
1363             } else {
1364                 // instance methods of value types do not have a monitor associated with their `this'
1365                 mask = ((sym.owner.flags_field & VALUE) != 0 && (flags & Flags.STATIC) == 0) ?
1366                         MethodFlags & ~SYNCHRONIZED : MethodFlags;
1367             }
1368             // Imply STRICTFP if owner has STRICTFP set.
1369             if (((flags|implicit) & Flags.ABSTRACT) == 0 ||
1370                 ((flags) & Flags.DEFAULT) != 0)
1371                 implicit |= sym.owner.flags_field & STRICTFP;
1372             break;
1373         case TYP:
1374             if (sym.isLocal()) {
1375                 mask = (flags & RECORD) != 0 ? LocalRecordFlags : LocalClassFlags;
1376                 if ((flags & RECORD) != 0) {
1377                     implicit = STATIC;
1378                     if (sym.owner.kind == TYP) {
1379                         log.error(pos, Errors.RecordDeclarationNotAllowedInInnerClasses);
1380                     }
1381                 }
1382                 if ((sym.owner.flags_field & STATIC) == 0 &&
1383                     (flags & ENUM) != 0) {
1384                     log.error(pos, Errors.EnumsMustBeStatic);
1385                 }
1386             } else if (sym.owner.kind == TYP) {
1387                 mask = (flags & RECORD) != 0 ? MemberRecordFlags : MemberClassFlags;
1388                 if (sym.owner.owner.kind == PCK ||
1389                     (sym.owner.flags_field & STATIC) != 0)
1390                     mask |= STATIC;
1391                 else if ((flags & ENUM) != 0) {
1392                     log.error(pos, Errors.EnumsMustBeStatic);
1393                 } else if ((flags & RECORD) != 0) {
1394                     log.error(pos, Errors.RecordDeclarationNotAllowedInInnerClasses);
1395                 }
1396                 // Nested interfaces and enums are always STATIC (Spec ???)
1397                 if ((flags & (INTERFACE | ENUM | RECORD)) != 0 ) implicit = STATIC;
1398             } else {
1399                 mask = ClassFlags;
1400             }
1401             // Interfaces are always ABSTRACT
1402             if ((flags & INTERFACE) != 0) implicit |= ABSTRACT;
1403 
1404             if ((flags & ENUM) != 0) {
1405                 // enums can't be declared abstract or final or value type
1406                 mask &= ~(ABSTRACT | FINAL | VALUE);
1407                 implicit |= implicitEnumFinalFlag(tree);
1408             }
1409             if ((flags & RECORD) != 0) {
1410                 // records can't be declared abstract
1411                 mask &= ~ABSTRACT;
1412                 implicit |= FINAL;
1413             }
1414             // Imply STRICTFP if owner has STRICTFP set.
1415             implicit |= sym.owner.flags_field & STRICTFP;
1416             break;
1417         default:
1418             throw new AssertionError();
1419         }
1420         long illegal = flags & ExtendedStandardFlags & ~mask;
1421         if (illegal != 0) {
1422             if ((illegal & INTERFACE) != 0) {
1423                 log.error(pos, ((flags & ANNOTATION) != 0) ? Errors.AnnotationDeclNotAllowedHere : Errors.IntfNotAllowedHere);
1424                 mask |= INTERFACE;
1425             }
1426             else {
1427                 log.error(pos,
1428                           Errors.ModNotAllowedHere(asFlagSet(illegal)));
1429             }
1430         }
1431         else if ((sym.kind == TYP ||
1432                   // ISSUE: Disallowing abstract&private is no longer appropriate
1433                   // in the presence of inner classes. Should it be deleted here?
1434                   checkDisjoint(pos, flags,
1435                                 ABSTRACT,
1436                                 PRIVATE | STATIC | DEFAULT))
1437                  &&
1438                  checkDisjoint(pos, flags,
1439                                 STATIC | PRIVATE,
1440                                 DEFAULT)
1441                  &&
1442                  checkDisjoint(pos, flags,
1443                                ABSTRACT | INTERFACE,
1444                                FINAL | NATIVE | SYNCHRONIZED | VALUE)
1445                  &&
1446                  checkDisjoint(pos, flags,
1447                                PUBLIC,
1448                                PRIVATE | PROTECTED)
1449                  &&
1450                  checkDisjoint(pos, flags,
1451                                PRIVATE,
1452                                PUBLIC | PROTECTED)
1453                  &&
1454                  checkDisjoint(pos, (flags | implicit), // complain against volatile & implcitly final entities too.
1455                                FINAL,
1456                                VOLATILE)
1457                  &&
1458                  (sym.kind == TYP ||
1459                   checkDisjoint(pos, flags,
1460                                 ABSTRACT | NATIVE,
1461                                 STRICTFP))) {
1462             // skip
1463         }
1464         return flags & (mask | ~ExtendedStandardFlags) | implicit;
1465     }

1469      *
1470      *  If the enum has no specialized enum constants, it is final.
1471      *
1472      *  If the enum does have specialized enum constants, it is
1473      *  <i>not</i> final.
1474      */

1476         if (!tree.hasTag(CLASSDEF)) return 0;

1478             boolean specialized;
1479             SpecialTreeVisitor() {
1480                 this.specialized = false;

1501             if (sts.specialized) return 0;
1502         }

1721      *  in the list.
1722      */

1724         return subset(t, ts) ? ts : excl(t, ts).prepend(t);

1727     /** Remove type set from type set list.
1728      */
1729     List<Type> excl(Type t, List<Type> ts) {
1730         if (ts.isEmpty()) {
1731             return ts;
1732         } else {
1733             List<Type> ts1 = excl(t, ts.tail);
1734             if (types.isSubtype(ts.head, t)) return ts1;
1735             else if (ts1 == ts.tail) return ts;
1736             else return ts1.prepend(ts.head);
1737         }

1887     /** Check that this method conforms with overridden method 'other'.
1888      *  where `origin' is the class where checking started.
1889      *  Complications:
1890      *  (1) Do not check overriding of synthetic methods
1891      *      (reason: they might be final).
1892      *      todo: check whether this is still necessary.
1893      *  (2) Admit the case where an interface proxy throws fewer exceptions
1894      *      than the method it implements. Augment the proxy methods with the
1895      *      undeclared exceptions in this case.
1896      *  (3) When generics are enabled, admit the case where an interface proxy
1897      *      has a result type
1898      *      extended by the result type of the method it implements.
1899      *      Change the proxies result type to the smaller type in this case.
1900      *
1901      *  @param tree         The tree from which positions
1902      *                      are extracted for errors.
1903      *  @param m            The overriding method.
1904      *  @param other        The overridden method.
1905      *  @param origin       The class of which the overriding method
1906      *                      is a member.
1907      */
1908     void checkOverride(JCTree tree,
1909                        MethodSymbol m,
1910                        MethodSymbol other,
1911                        ClassSymbol origin) {
1912         // Don't check overriding of synthetic methods or by bridge methods.
1913         if ((m.flags() & (SYNTHETIC|BRIDGE)) != 0 || (other.flags() & SYNTHETIC) != 0) {
1914             return;
1915         }
1916 
1917         // Error if static method overrides instance method (JLS 8.4.6.2).
1918         if ((m.flags() & STATIC) != 0 &&
1919                    (other.flags() & STATIC) == 0) {
1920             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1921                       Errors.OverrideStatic(cannotOverride(m, other)));
1922             m.flags_field |= BAD_OVERRIDE;
1923             return;
1924         }
1925 
1926         // Error if instance method overrides static or final
1927         // method (JLS 8.4.6.1).
1928         if ((other.flags() & FINAL) != 0 ||
1929                  (m.flags() & STATIC) == 0 &&
1930                  (other.flags() & STATIC) != 0) {
1931             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1932                       Errors.OverrideMeth(cannotOverride(m, other),
1933                                           asFlagSet(other.flags() & (FINAL | STATIC))));
1934             m.flags_field |= BAD_OVERRIDE;
1935             return;
1936         }
1937 
1938         if ((m.owner.flags() & ANNOTATION) != 0) {
1939             // handled in validateAnnotationMethod
1940             return;
1941         }
1942 
1943         // Error if overriding method has weaker access (JLS 8.4.6.3).
1944         if (protection(m.flags()) > protection(other.flags())) {
1945             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1946                       (other.flags() & AccessFlags) == 0 ?
1947                               Errors.OverrideWeakerAccess(cannotOverride(m, other),
1948                                                           "package") :
1949                               Errors.OverrideWeakerAccess(cannotOverride(m, other),
1950                                                           asFlagSet(other.flags() & AccessFlags)));
1951             m.flags_field |= BAD_OVERRIDE;
1952             return;
1953         }
1954 
1955         if (origin.isValue() && other.owner == syms.objectType.tsym && m.type.getParameterTypes().size() == 0) {
1956             if (m.name == names.clone || m.name == names.finalize) {
1957                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1958                         Errors.InlineClassMayNotOverride(m.name));
1959                 m.flags_field |= BAD_OVERRIDE;
1960                 return;
1961             }
1962         }
1963 
1964         Type mt = types.memberType(origin.type, m);
1965         Type ot = types.memberType(origin.type, other);
1966         // Error if overriding result type is different
1967         // (or, in the case of generics mode, not a subtype) of
1968         // overridden result type. We have to rename any type parameters
1969         // before comparing types.
1970         List<Type> mtvars = mt.getTypeArguments();
1971         List<Type> otvars = ot.getTypeArguments();
1972         Type mtres = mt.getReturnType();
1973         Type otres = types.subst(ot.getReturnType(), otvars, mtvars);
1974 
1975         overrideWarner.clear();
1976         boolean resultTypesOK =
1977             types.returnTypeSubstitutable(mt, ot, otres, overrideWarner);
1978         if (!resultTypesOK) {
1979             if ((m.flags() & STATIC) != 0 && (other.flags() & STATIC) != 0) {
1980                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1981                           Errors.OverrideIncompatibleRet(Fragments.CantHide(m, m.location(), other,
1982                                         other.location()), mtres, otres));
1983                 m.flags_field |= BAD_OVERRIDE;
1984             } else {
1985                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1986                           Errors.OverrideIncompatibleRet(cannotOverride(m, other), mtres, otres));
1987                 m.flags_field |= BAD_OVERRIDE;
1988             }
1989             return;
1990         } else if (overrideWarner.hasNonSilentLint(LintCategory.UNCHECKED)) {
1991             warnUnchecked(TreeInfo.diagnosticPositionFor(m, tree),
1992                     Warnings.OverrideUncheckedRet(uncheckedOverrides(m, other), mtres, otres));
1993         }
1994 
1995         // Error if overriding method throws an exception not reported
1996         // by overridden method.
1997         List<Type> otthrown = types.subst(ot.getThrownTypes(), otvars, mtvars);
1998         List<Type> unhandledErased = unhandled(mt.getThrownTypes(), types.erasure(otthrown));
1999         List<Type> unhandledUnerased = unhandled(mt.getThrownTypes(), otthrown);
2000         if (unhandledErased.nonEmpty()) {
2001             log.error(TreeInfo.diagnosticPositionFor(m, tree),
2002                       Errors.OverrideMethDoesntThrow(cannotOverride(m, other), unhandledUnerased.head));
2003             m.flags_field |= BAD_OVERRIDE;
2004             return;
2005         }
2006         else if (unhandledUnerased.nonEmpty()) {
2007             warnUnchecked(TreeInfo.diagnosticPositionFor(m, tree),
2008                           Warnings.OverrideUncheckedThrown(cannotOverride(m, other), unhandledUnerased.head));
2009             return;
2010         }
2011 
2012         // Optional warning if varargs don't agree
2013         if ((((m.flags() ^ other.flags()) & Flags.VARARGS) != 0)
2014             && lint.isEnabled(LintCategory.OVERRIDES)) {
2015             log.warning(TreeInfo.diagnosticPositionFor(m, tree),
2016                         ((m.flags() & Flags.VARARGS) != 0)
2017                         ? Warnings.OverrideVarargsMissing(varargsOverrides(m, other))
2018                         : Warnings.OverrideVarargsExtra(varargsOverrides(m, other)));
2019         }
2020 
2021         // Warn if instance method overrides bridge method (compiler spec ??)
2022         if ((other.flags() & BRIDGE) != 0) {
2023             log.warning(TreeInfo.diagnosticPositionFor(m, tree),
2024                         Warnings.OverrideBridge(uncheckedOverrides(m, other)));
2025         }
2026 
2027         // Warn if a deprecated method overridden by a non-deprecated one.
2028         if (!isDeprecatedOverrideIgnorable(other, origin)) {
2029             Lint prevLint = setLint(lint.augment(m));
2030             try {
2031                 checkDeprecated(() -> TreeInfo.diagnosticPositionFor(m, tree), m, other);
2032             } finally {
2033                 setLint(prevLint);
2034             }
2035         }

2866     /**
2867       * Report warnings for potentially ambiguous method declarations. Two declarations
2868       * are potentially ambiguous if they feature two unrelated functional interface
2869       * in same argument position (in which case, a call site passing an implicit
2870       * lambda would be ambiguous).
2871       */
2872     void checkPotentiallyAmbiguousOverloads(DiagnosticPosition pos, Type site,
2873             MethodSymbol msym1, MethodSymbol msym2) {
2874         if (msym1 != msym2 &&
2875                 Feature.DEFAULT_METHODS.allowedInSource(source) &&
2876                 lint.isEnabled(LintCategory.OVERLOADS) &&
2877                 (msym1.flags() & POTENTIALLY_AMBIGUOUS) == 0 &&
2878                 (msym2.flags() & POTENTIALLY_AMBIGUOUS) == 0) {
2879             Type mt1 = types.memberType(site, msym1);
2880             Type mt2 = types.memberType(site, msym2);
2881             //if both generic methods, adjust type variables
2882             if (mt1.hasTag(FORALL) && mt2.hasTag(FORALL) &&
2883                     types.hasSameBounds((ForAll)mt1, (ForAll)mt2)) {
2884                 mt2 = types.subst(mt2, ((ForAll)mt2).tvars, ((ForAll)mt1).tvars);
2885             }
2886             //expand varargs methods if needed
2887             int maxLength = Math.max(mt1.getParameterTypes().length(), mt2.getParameterTypes().length());
2888             List<Type> args1 = rs.adjustArgs(mt1.getParameterTypes(), msym1, maxLength, true);
2889             List<Type> args2 = rs.adjustArgs(mt2.getParameterTypes(), msym2, maxLength, true);
2890             //if arities don't match, exit
2891             if (args1.length() != args2.length()) return;
2892             boolean potentiallyAmbiguous = false;
2893             while (args1.nonEmpty() && args2.nonEmpty()) {
2894                 Type s = args1.head;
2895                 Type t = args2.head;
2896                 if (!types.isSubtype(t, s) && !types.isSubtype(s, t)) {
2897                     if (types.isFunctionalInterface(s) && types.isFunctionalInterface(t) &&
2898                             types.findDescriptorType(s).getParameterTypes().length() > 0 &&
2899                             types.findDescriptorType(s).getParameterTypes().length() ==
2900                             types.findDescriptorType(t).getParameterTypes().length()) {
2901                         potentiallyAmbiguous = true;
2902                     } else {
2903                         break;
2904                     }
2905                 }
2906                 args1 = args1.tail;
2907                 args2 = args2.tail;
2908             }
2909             if (potentiallyAmbiguous) {
2910                 //we found two incompatible functional interfaces with same arity
2911                 //this means a call site passing an implicit lambda would be ambiguous
2912                 msym1.flags_field |= POTENTIALLY_AMBIGUOUS;
2913                 msym2.flags_field |= POTENTIALLY_AMBIGUOUS;
2914                 log.warning(LintCategory.OVERLOADS, pos,
2915                             Warnings.PotentiallyAmbiguousOverload(msym1, msym1.location(),
2916                                                                   msym2, msym2.location()));
2917                 return;
2918             }
2919         }

4321             PackageSymbol whatPackage = what.packge();
4322             ExportsDirective whatExport = findExport(whatPackage);
4323             ExportsDirective inExport = findExport(inPackage);
4324 
4325             if (whatExport == null) { //package not exported:

4331                 if (inExport.modules == null || !whatExport.modules.containsAll(inExport.modules)) {
4332                     log.warning(LintCategory.EXPORTS, pos, Warnings.LeaksNotAccessibleUnexportedQualified(kindName(what), what, what.packge().modle));
