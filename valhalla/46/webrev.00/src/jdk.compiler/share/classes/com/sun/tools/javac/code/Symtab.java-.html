 74 /** A class that defines all predefined constants and operators
 75  *  as well as special classes such as java.lang.Object, which need
 76  *  to be known to the compiler. All symbols are held in instance
 77  *  fields. This makes it possible to work in multiple concurrent
 78  *  projects, which might use different class files for library classes.
 79  *
 80  *  <p><b>This is NOT part of any supported API.
 81  *  If you write code that depends on this, you do so at your own risk.
 82  *  This code and its internal interfaces are subject to change or
 83  *  deletion without notice.</b>
 84  */
 85 public class Symtab {
 86     /** The context key for the symbol table. */
 87     protected static final Context.Key<Symtab> symtabKey = new Context.Key<>();

 89     /** Get the symbol table instance. */
 90     public static Symtab instance(Context context) {
 91         Symtab instance = context.get(symtabKey);
 92         if (instance == null)
 93             instance = new Symtab(context);
 94         return instance;
 95     }

267     public VarSymbol getClassField(Type type, Types types) {
268         return classFields.computeIfAbsent(
269             new UniqueType(type, types), k -> {
270                 Type arg = null;
271                 if (type.getTag() == ARRAY || type.getTag() == CLASS)
272                     arg = types.erasure(type);
273                 else if (type.isPrimitiveOrVoid())
274                     arg = types.boxedClass(type).type;
275                 else
276                     throw new AssertionError(type);
277 
278                 Type t = new ClassType(
279                     classType.getEnclosingType(), List.of(arg), classType.tsym);
280                 return new VarSymbol(
281                     STATIC | PUBLIC | FINAL, names._class, t, type.tsym);
282             });
283     }

313     public void synthesizeEmptyInterfaceIfMissing(final Type type) {
314         final Completer completer = type.tsym.completer;
315         type.tsym.completer = new Completer() {
316             @Override
317             public void complete(Symbol sym) throws CompletionFailure {
318                 try {
319                     completer.complete(sym);
320                 } catch (CompletionFailure e) {
321                     sym.flags_field |= (PUBLIC | INTERFACE);
322                     ((ClassType) sym.type).supertype_field = objectType;
323                 }
324             }
325 
326             @Override
327             public boolean isTerminal() {
328                 return completer.isTerminal();
329             }
330         };
331     }
332 
333     public void synthesizeBoxTypeIfMissing(final Type type) {
334         ClassSymbol sym = enterClass(java_base, boxedName[type.getTag().ordinal()]);
335         final Completer completer = sym.completer;
336         sym.completer = new Completer() {
337             @Override
338             public void complete(Symbol sym) throws CompletionFailure {
339                 try {
340                     completer.complete(sym);
341                 } catch (CompletionFailure e) {
342                     sym.flags_field |= PUBLIC;
343                     ((ClassType) sym.type).supertype_field = objectType;
344                     MethodSymbol boxMethod =
345                         new MethodSymbol(PUBLIC | STATIC, names.valueOf,
346                                          new MethodType(List.of(type), sym.type,
347                                 List.nil(), methodClass),
348                             sym);
349                     sym.members().enter(boxMethod);
350                     MethodSymbol unboxMethod =
351                         new MethodSymbol(PUBLIC,
352                             type.tsym.name.append(names.Value), // x.intValue()
353                             new MethodType(List.nil(), type,
354                                 List.nil(), methodClass),
355                             sym);
356                     sym.members().enter(unboxMethod);
357                 }
358             }
359 
360             @Override
361             public boolean isTerminal() {
362                 return completer.isTerminal();
363             }
364         };
365     }

367     // Enter a synthetic class that is used to mark classes in ct.sym.
368     // This class does not have a class file.
369     private Type enterSyntheticAnnotation(String name) {
370         // for now, leave the module null, to prevent problems from synthesizing the
371         // existence of a class in any specific module, including noModule
372         ClassType type = (ClassType)enterClass(java_base, names.fromString(name)).type;
373         ClassSymbol sym = (ClassSymbol)type.tsym;
374         sym.completer = Completer.NULL_COMPLETER;
375         sym.flags_field = PUBLIC|ACYCLIC|ANNOTATION|INTERFACE;
376         sym.erasure_field = type;
377         sym.members_field = WriteableScope.create(sym);
378         type.typarams_field = List.nil();
379         type.allparams_field = List.nil();
380         type.supertype_field = annotationType;
381         type.interfaces_field = List.nil();
382         return type;
383     }

412                                     EnumSet.of(com.sun.tools.javac.code.Directive.RequiresFlag.MANDATED));
413                     requires = List.of(d);
414                 }
415                 @Override
416                 public String toString() {
417                     return messages.getLocalizedString("compiler.misc.unnamed.module");
418                 }
419             };
420         addRootPackageFor(unnamedModule);

423         errModule = new ModuleSymbol(names.empty, null) {

472         initType(unknownType, unknownSymbol);
473 
474         // the builtin class of all arrays
475         arrayClass = new ClassSymbol(PUBLIC|ACYCLIC, names.Array, noSymbol);
476 
477         // VGJ
478         boundClass = new ClassSymbol(PUBLIC|ACYCLIC, names.Bound, noSymbol);
479         boundClass.members_field = new Scope.ErrorScope(boundClass);

481         // the builtin class of all methods

615         profileType.tsym.members().enter(m);
616 
617         injectTopInterfaceTypes = Options.instance(context).isUnset("noTopInterfaceInjection") &&
618                 Feature.INLINE_TYPES.allowedInSource(source) &&
619                 Target.instance(context).hasTopInterfaces();
620 
621         // Enter a class for arrays.
622         // The class implements java.lang.Cloneable and java.io.Serializable.
623         // It has a final length field and a clone method.
624         ClassType arrayClassType = (ClassType)arrayClass.type;
625         arrayClassType.supertype_field = objectType;
626         arrayClassType.interfaces_field = injectTopInterfaceTypes ?
627                 List.of(cloneableType, serializableType, identityObjectType):
628                 List.of(cloneableType, serializableType);
629 
630         arrayClass.members_field = WriteableScope.create(arrayClass);
631         lengthVar = new VarSymbol(
632             PUBLIC | FINAL,
633             names.length,
634             intType,
635             arrayClass);
636         arrayClass.members().enter(lengthVar);
637         arrayCloneMethod = new MethodSymbol(
638             PUBLIC,
639             names.clone,
640             new MethodType(List.nil(), objectType,
641                            List.nil(), methodClass),
642             arrayClass);
643         arrayClass.members().enter(arrayCloneMethod);
644 
645         if (java_base != noModule)
646             java_base.completer = moduleCompleter::complete; //bootstrap issues
647 
648     }

658     /** Create a new toplevel or member class symbol with given name
659      *  and owner and enter in `classes' unless already there.
660      */
661     public ClassSymbol enterClass(ModuleSymbol msym, Name name, TypeSymbol owner) {
662         Assert.checkNonNull(msym);
663         Name flatname = TypeSymbol.formFlatName(name, owner);
664         ClassSymbol c = getClass(msym, flatname);
665         if (c == null) {
666             c = defineClass(name, owner);
667             doEnterClass(msym, c);
668         } else if ((c.name != name || c.owner != owner) && owner.kind == TYP && c.owner.kind == PCK) {
669             // reassign fields of classes that might have been loaded with
670             // their flat names.
671             c.owner.members().remove(c);
672             c.name = name;
673             c.owner = owner;
674             c.fullname = ClassSymbol.formFullName(name, owner);
675         }
676         return c;
677     }

684     public PackageSymbol lookupPackage(ModuleSymbol msym, Name flatName) {
685         Assert.checkNonNull(msym);
686 
687         if (flatName.isEmpty()) {
688             //unnamed packages only from the current module - visiblePackages contains *root* package, not unnamed package!
689             return msym.unnamedPackage;
690         }
691 
692         if (msym == noModule) {
693             return enterPackage(msym, flatName);
694         }
695 
696         msym.complete();
697 
698         PackageSymbol pack;
699 
700         pack = msym.visiblePackages.get(flatName);
701 
702         if (pack != null)
703             return pack;
704 
705         pack = getPackage(msym, flatName);
706 
707         if (pack != null && pack.exists())
708             return pack;
709 
710         boolean dependsOnUnnamed = msym.requires != null &&
711                                    msym.requires.stream()
712                                                 .map(rd -> rd.module)
713                                                 .anyMatch(mod -> mod == unnamedModule);
714 
715         if (dependsOnUnnamed) {
716             //msyms depends on the unnamed module, for which we generally don't know
717             //the list of packages it "exports" ahead of time. So try to lookup the package in the
718             //current module, and in the unnamed module and see if it exists in one of them
719             PackageSymbol unnamedPack = getPackage(unnamedModule, flatName);
720 
721             if (unnamedPack != null && unnamedPack.exists()) {
722                 msym.visiblePackages.put(unnamedPack.fullname, unnamedPack);
723                 return unnamedPack;
724             }
725 
726             pack = enterPackage(msym, flatName);
727             pack.complete();
728             if (pack.exists())
729                 return pack;
730 
731             unnamedPack = enterPackage(unnamedModule, flatName);
732             unnamedPack.complete();
733             if (unnamedPack.exists()) {
734                 msym.visiblePackages.put(unnamedPack.fullname, unnamedPack);
735                 return unnamedPack;
736             }
737 
738             return pack;
739         }
740 
741         return enterPackage(msym, flatName);
742     }

772             return c;

782     /** Make a package, given its fully qualified name.
783      */
784     public PackageSymbol enterPackage(ModuleSymbol currModule, Name fullname) {
785         Assert.checkNonNull(currModule);
786         PackageSymbol p = getPackage(currModule, fullname);
787         if (p == null) {
788             Assert.check(!fullname.isEmpty(), () -> "rootPackage missing!; currModule: " + currModule);
789             p = new PackageSymbol(
790                     Convert.shortName(fullname),
791                     enterPackage(currModule, Convert.packagePart(fullname)));
792             p.completer = initialCompleter;
793             p.modle = currModule;
794             doEnterPackage(currModule, p);
795         }
796         return p;
797     }

800         packages.computeIfAbsent(pack.fullname, n -> new HashMap<>()).put(msym, pack);
801         msym.enclosedPackages = msym.enclosedPackages.prepend(pack);
802     }

805         doEnterPackage(module, rootPackage);
806         PackageSymbol unnamedPackage = new PackageSymbol(names.empty, rootPackage) {
807                 @Override
808                 public String toString() {
809                     return messages.getLocalizedString("compiler.misc.unnamed.package");
810                 }
811             };
812         unnamedPackage.modle = module;
813         //we cannot use a method reference below, as initialCompleter might be null now
814         unnamedPackage.completer = s -> initialCompleter.complete(s);
815         unnamedPackage.flags_field |= EXISTS;
816         module.unnamedPackage = unnamedPackage;

859     public List<ModuleSymbol> listPackageModules(Name packageName) {
860         if (packageName.isEmpty())
861             return List.nil();
862 
863         List<ModuleSymbol> result = List.nil();
864         Map<ModuleSymbol,PackageSymbol> map = packages.get(packageName);
865         if (map != null) {
866             for (Map.Entry<ModuleSymbol, PackageSymbol> e: map.entrySet()) {
867                 if (!e.getValue().members().isEmpty()) {
868                     result = result.prepend(e.getKey());

870             }

872         return result;
