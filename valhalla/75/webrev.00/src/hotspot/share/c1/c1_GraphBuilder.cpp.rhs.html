 132 void BlockListBuilder::set_entries(int osr_bci) {
 133   // generate start blocks
 134   BlockBegin* std_entry = make_block_at(0, NULL);
 135   if (scope()->caller() == NULL) {
 136     std_entry->set(BlockBegin::std_entry_flag);
 137   }
 138   if (osr_bci != -1) {
 139     BlockBegin* osr_entry = make_block_at(osr_bci, NULL);
 140     osr_entry->set(BlockBegin::osr_entry_flag);
 141   }
 142 
 143   // generate exception entry blocks
 144   XHandlers* list = xhandlers();
 145   const int n = list->length();
 146   for (int i = 0; i < n; i++) {
 147     XHandler* h = list->handler_at(i);
 148     BlockBegin* entry = make_block_at(h->handler_bci(), NULL);
 149     entry->set(BlockBegin::exception_entry_flag);
 150     h->set_entry_block(entry);
 151   }
 152 }

 155 BlockBegin* BlockListBuilder::make_block_at(int cur_bci, BlockBegin* predecessor) {
 156   assert(method()->bci_block_start().at(cur_bci), "wrong block starts of MethodLivenessAnalyzer");
 157 
 158   BlockBegin* block = _bci2block->at(cur_bci);
 159   if (block == NULL) {
 160     block = new BlockBegin(cur_bci);
 161     block->init_stores_to_locals(method()->max_locals());
 162     _bci2block->at_put(cur_bci, block);
 163     _blocks.append(block);
 164 
 165     assert(predecessor == NULL || predecessor->bci() < cur_bci, "targets for backward branches must already exist");
 166   }
 167 
 168   if (predecessor != NULL) {
 169     if (block->is_set(BlockBegin::exception_entry_flag)) {
 170       BAILOUT_("Exception handler can be reached by both normal and exceptional control flow", block);
 171     }
 172 
 173     predecessor->add_successor(block);
 174     block->increment_total_preds();
 175   }
 176 
 177   return block;
 178 }

 181 inline void BlockListBuilder::store_one(BlockBegin* current, int local) {
 182   current->stores_to_locals().set_bit(local);

 184 inline void BlockListBuilder::store_two(BlockBegin* current, int local) {
 185   store_one(current, local);
 186   store_one(current, local + 1);

 190 void BlockListBuilder::handle_exceptions(BlockBegin* current, int cur_bci) {
 191   // Draws edges from a block to its exception handlers
 192   XHandlers* list = xhandlers();
 193   const int n = list->length();
 194 
 195   for (int i = 0; i < n; i++) {
 196     XHandler* h = list->handler_at(i);
 197 
 198     if (h->covers(cur_bci)) {
 199       BlockBegin* entry = h->entry_block();
 200       assert(entry != NULL && entry == _bci2block->at(h->handler_bci()), "entry must be set");
 201       assert(entry->is_set(BlockBegin::exception_entry_flag), "flag must be set");
 202 
 203       // add each exception handler only once
 204       if (!current->is_successor(entry)) {
 205         current->add_successor(entry);
 206         entry->increment_total_preds();
 207       }
 208 
 209       // stop when reaching catchall
 210       if (h->catch_type() == 0) break;
 211     }
 212   }
 213 }

 215 void BlockListBuilder::handle_jsr(BlockBegin* current, int sr_bci, int next_bci) {
 216   // start a new block after jsr-bytecode and link this block into cfg
 217   make_block_at(next_bci, current);
 218 
 219   // start a new block at the subroutine entry at mark it with special flag
 220   BlockBegin* sr_block = make_block_at(sr_bci, current);
 221   if (!sr_block->is_set(BlockBegin::subroutine_entry_flag)) {
 222     sr_block->set(BlockBegin::subroutine_entry_flag);
 223   }
 224 }

 227 void BlockListBuilder::set_leaders() {
 228   bool has_xhandlers = xhandlers()->has_handlers();
 229   BlockBegin* current = NULL;
 230 
 231   // The information which bci starts a new block simplifies the analysis
 232   // Without it, backward branches could jump to a bci where no block was created
 233   // during bytecode iteration. This would require the creation of a new block at the
 234   // branch target and a modification of the successor lists.
 235   const BitMap& bci_block_start = method()->bci_block_start();
 236 
 237   ciBytecodeStream s(method());
 238   while (s.next() != ciBytecodeStream::EOBC()) {
 239     int cur_bci = s.cur_bci();
 240 
 241     if (bci_block_start.at(cur_bci)) {
 242       current = make_block_at(cur_bci, current);
 243     }
 244     assert(current != NULL, "must have current block");
 245 
 246     if (has_xhandlers && GraphBuilder::can_trap(method(), s.cur_bc())) {
 247       handle_exceptions(current, cur_bci);
 248     }
 249 
 250     switch (s.cur_bc()) {
 251       // track stores to local variables for selective creation of phi functions
 252       case Bytecodes::_iinc:     store_one(current, s.get_index()); break;
 253       case Bytecodes::_istore:   store_one(current, s.get_index()); break;
 254       case Bytecodes::_lstore:   store_two(current, s.get_index()); break;
 255       case Bytecodes::_fstore:   store_one(current, s.get_index()); break;
 256       case Bytecodes::_dstore:   store_two(current, s.get_index()); break;
 257       case Bytecodes::_astore:   store_one(current, s.get_index()); break;
 258       case Bytecodes::_istore_0: store_one(current, 0); break;
 259       case Bytecodes::_istore_1: store_one(current, 1); break;
 260       case Bytecodes::_istore_2: store_one(current, 2); break;
 261       case Bytecodes::_istore_3: store_one(current, 3); break;
 262       case Bytecodes::_lstore_0: store_two(current, 0); break;
 263       case Bytecodes::_lstore_1: store_two(current, 1); break;
 264       case Bytecodes::_lstore_2: store_two(current, 2); break;
 265       case Bytecodes::_lstore_3: store_two(current, 3); break;
 266       case Bytecodes::_fstore_0: store_one(current, 0); break;
 267       case Bytecodes::_fstore_1: store_one(current, 1); break;
 268       case Bytecodes::_fstore_2: store_one(current, 2); break;
 269       case Bytecodes::_fstore_3: store_one(current, 3); break;
 270       case Bytecodes::_dstore_0: store_two(current, 0); break;
 271       case Bytecodes::_dstore_1: store_two(current, 1); break;
 272       case Bytecodes::_dstore_2: store_two(current, 2); break;
 273       case Bytecodes::_dstore_3: store_two(current, 3); break;
 274       case Bytecodes::_astore_0: store_one(current, 0); break;
 275       case Bytecodes::_astore_1: store_one(current, 1); break;
 276       case Bytecodes::_astore_2: store_one(current, 2); break;
 277       case Bytecodes::_astore_3: store_one(current, 3); break;
 278 
 279       // track bytecodes that affect the control flow
 280       case Bytecodes::_athrow:  // fall through
 281       case Bytecodes::_ret:     // fall through
 282       case Bytecodes::_ireturn: // fall through
 283       case Bytecodes::_lreturn: // fall through
 284       case Bytecodes::_freturn: // fall through
 285       case Bytecodes::_dreturn: // fall through
 286       case Bytecodes::_areturn: // fall through
 287       case Bytecodes::_return:
 288         current = NULL;
 289         break;
 290 
 291       case Bytecodes::_ifeq:      // fall through
 292       case Bytecodes::_ifne:      // fall through
 293       case Bytecodes::_iflt:      // fall through
 294       case Bytecodes::_ifge:      // fall through
 295       case Bytecodes::_ifgt:      // fall through
 296       case Bytecodes::_ifle:      // fall through
 297       case Bytecodes::_if_icmpeq: // fall through
 298       case Bytecodes::_if_icmpne: // fall through
 299       case Bytecodes::_if_icmplt: // fall through
 300       case Bytecodes::_if_icmpge: // fall through
 301       case Bytecodes::_if_icmpgt: // fall through
 302       case Bytecodes::_if_icmple: // fall through
 303       case Bytecodes::_if_acmpeq: // fall through
 304       case Bytecodes::_if_acmpne: // fall through
 305       case Bytecodes::_ifnull:    // fall through
 306       case Bytecodes::_ifnonnull:
 307         make_block_at(s.next_bci(), current);
 308         make_block_at(s.get_dest(), current);
 309         current = NULL;
 310         break;
 311 
 312       case Bytecodes::_goto:
 313         make_block_at(s.get_dest(), current);
 314         current = NULL;
 315         break;
 316 
 317       case Bytecodes::_goto_w:
 318         make_block_at(s.get_far_dest(), current);
 319         current = NULL;
 320         break;
 321 
 322       case Bytecodes::_jsr:
 323         handle_jsr(current, s.get_dest(), s.next_bci());
 324         current = NULL;
 325         break;
 326 
 327       case Bytecodes::_jsr_w:
 328         handle_jsr(current, s.get_far_dest(), s.next_bci());
 329         current = NULL;
 330         break;
 331 
 332       case Bytecodes::_tableswitch: {
 333         // set block for each case
 334         Bytecode_tableswitch sw(&s);
 335         int l = sw.length();
 336         for (int i = 0; i < l; i++) {
 337           make_block_at(cur_bci + sw.dest_offset_at(i), current);
 338         }
 339         make_block_at(cur_bci + sw.default_offset(), current);
 340         current = NULL;
 341         break;
 342       }
 343 
 344       case Bytecodes::_lookupswitch: {
 345         // set block for each case
 346         Bytecode_lookupswitch sw(&s);
 347         int l = sw.number_of_pairs();
 348         for (int i = 0; i < l; i++) {
 349           make_block_at(cur_bci + sw.pair_at(i).offset(), current);
 350         }
 351         make_block_at(cur_bci + sw.default_offset(), current);
 352         current = NULL;
 353         break;
 354       }
 355 
 356       default:
 357         break;
 358     }
 359   }
 360 }

 363 void BlockListBuilder::mark_loops() {
 364   ResourceMark rm;
 365 
 366   _active.initialize(BlockBegin::number_of_blocks());
 367   _visited.initialize(BlockBegin::number_of_blocks());
 368   _loop_map = intArray(BlockBegin::number_of_blocks(), BlockBegin::number_of_blocks(), 0);
 369   _next_loop_index = 0;
 370   _next_block_number = _blocks.length();
 371 
 372   // recursively iterate the control flow graph
 373   mark_loops(_bci2block->at(0), false);
 374   assert(_next_block_number >= 0, "invalid block numbers");
 375 
 376   // Remove dangling Resource pointers before the ResourceMark goes out-of-scope.
 377   _active.resize(0);
 378   _visited.resize(0);
 379 }

 381 void BlockListBuilder::make_loop_header(BlockBegin* block) {
 382   if (block->is_set(BlockBegin::exception_entry_flag)) {
 383     // exception edges may look like loops but don't mark them as such
 384     // since it screws up block ordering.
 385     return;
 386   }
 387   if (!block->is_set(BlockBegin::parser_loop_header_flag)) {
 388     block->set(BlockBegin::parser_loop_header_flag);
 389 
 390     assert(_loop_map.at(block->block_id()) == 0, "must not be set yet");
 391     assert(0 <= _next_loop_index && _next_loop_index < BitsPerInt, "_next_loop_index is used as a bit-index in integer");
 392     _loop_map.at_put(block->block_id(), 1 << _next_loop_index);
 393     if (_next_loop_index < 31) _next_loop_index++;
 394   } else {
 395     // block already marked as loop header
 396     assert(is_power_of_2((unsigned int)_loop_map.at(block->block_id())), "exactly one bit must be set");
 397   }
 398 }

 400 int BlockListBuilder::mark_loops(BlockBegin* block, bool in_subroutine) {
 401   int block_id = block->block_id();
 402 
 403   if (_visited.at(block_id)) {
 404     if (_active.at(block_id)) {
 405       // reached block via backward branch
 406       make_loop_header(block);
 407     }
 408     // return cached loop information for this block
 409     return _loop_map.at(block_id);
 410   }
 411 
 412   if (block->is_set(BlockBegin::subroutine_entry_flag)) {
 413     in_subroutine = true;
 414   }
 415 
 416   // set active and visited bits before successors are processed
 417   _visited.set_bit(block_id);
 418   _active.set_bit(block_id);
 419 
 420   intptr_t loop_state = 0;
 421   for (int i = block->number_of_sux() - 1; i >= 0; i--) {
 422     // recursively process all successors
 423     loop_state |= mark_loops(block->sux_at(i), in_subroutine);
 424   }
 425 
 426   // clear active-bit after all successors are processed
 427   _active.clear_bit(block_id);
 428 
 429   // reverse-post-order numbering of all blocks
 430   block->set_depth_first_number(_next_block_number);
 431   _next_block_number--;
 432 
 433   if (loop_state != 0 || in_subroutine ) {
 434     // block is contained at least in one loop, so phi functions are necessary
 435     // phi functions are also necessary for all locals stored in a subroutine
 436     scope()->requires_phi_function().set_union(block->stores_to_locals());
 437   }
 438 
 439   if (block->is_set(BlockBegin::parser_loop_header_flag)) {
 440     int header_loop_state = _loop_map.at(block_id);
 441     assert(is_power_of_2((unsigned)header_loop_state), "exactly one bit must be set");
 442 
 443     // If the highest bit is set (i.e. when integer value is negative), the method
 444     // has 32 or more loops. This bit is never cleared because it is used for multiple loops
 445     if (header_loop_state >= 0) {
 446       clear_bits(loop_state, header_loop_state);
 447     }
 448   }
 449 
 450   // cache and return loop information for this block
 451   _loop_map.at_put(block_id, loop_state);
 452   return loop_state;
 453 }

 462 void BlockListBuilder::print() {
 463   tty->print("----- initial block list of BlockListBuilder for method ");
 464   method()->print_short_name();
 465   tty->cr();
 466 
 467   // better readability if blocks are sorted in processing order
 468   _blocks.sort(compare_depth_first);
 469 
 470   for (int i = 0; i < _blocks.length(); i++) {
 471     BlockBegin* cur = _blocks.at(i);
 472     tty->print("%4d: B%-4d bci: %-4d  preds: %-4d ", cur->depth_first_number(), cur->block_id(), cur->bci(), cur->total_preds());
 473 
 474     tty->print(cur->is_set(BlockBegin::std_entry_flag)               ? " std" : "    ");
 475     tty->print(cur->is_set(BlockBegin::osr_entry_flag)               ? " osr" : "    ");
 476     tty->print(cur->is_set(BlockBegin::exception_entry_flag)         ? " ex" : "   ");
 477     tty->print(cur->is_set(BlockBegin::subroutine_entry_flag)        ? " sr" : "   ");
 478     tty->print(cur->is_set(BlockBegin::parser_loop_header_flag)      ? " lh" : "   ");
 479 
 480     if (cur->number_of_sux() > 0) {
 481       tty->print("    sux: ");
 482       for (int j = 0; j < cur->number_of_sux(); j++) {
 483         BlockBegin* sux = cur->sux_at(j);
 484         tty->print("B%d ", sux->block_id());
 485       }
 486     }
 487     tty->cr();
 488   }
 489 }

 494 // A simple growable array of Values indexed by ciFields
 495 class FieldBuffer: public CompilationResourceObj {
 496  private:
 497   GrowableArray<Value> _values;
 498 
 499  public:
 500   FieldBuffer() {}
 501 
 502   void kill() {
 503     _values.trunc_to(0);
 504   }
 505 
 506   Value at(ciField* field) {
 507     assert(field->holder()->is_loaded(), "must be a loaded field");
 508     int offset = field->offset();
 509     if (offset < _values.length()) {
 510       return _values.at(offset);
 511     } else {
 512       return NULL;
 513     }
 514   }
 515 
 516   void at_put(ciField* field, Value value) {
 517     assert(field->holder()->is_loaded(), "must be a loaded field");
 518     int offset = field->offset();
 519     _values.at_put_grow(offset, value, NULL);
 520   }
 521 
 522 };

 525 // MemoryBuffer is fairly simple model of the current state of memory.
 526 // It partitions memory into several pieces.  The first piece is
 527 // generic memory where little is known about the owner of the memory.
 528 // This is conceptually represented by the tuple <O, F, V> which says
 529 // that the field F of object O has value V.  This is flattened so
 530 // that F is represented by the offset of the field and the parallel
 531 // arrays _objects and _values are used for O and V.  Loads of O.F can
 532 // simply use V.  Newly allocated objects are kept in a separate list
 533 // along with a parallel array for each object which represents the
 534 // current value of its fields.  Stores of the default value to fields
 535 // which have never been stored to before are eliminated since they
 536 // are redundant.  Once newly allocated objects are stored into
 537 // another object or they are passed out of the current compile they
 538 // are treated like generic memory.
 539 
 540 class MemoryBuffer: public CompilationResourceObj {
 541  private:
 542   FieldBuffer                 _values;
 543   GrowableArray<Value>        _objects;
 544   GrowableArray<Value>        _newobjects;
 545   GrowableArray<FieldBuffer*> _fields;
 546 
 547  public:
 548   MemoryBuffer() {}
 549 
 550   StoreField* store(StoreField* st) {
 551     if (!EliminateFieldAccess) {
 552       return st;
 553     }
 554 
 555     Value object = st->obj();
 556     Value value = st->value();
 557     ciField* field = st->field();
 558     if (field->holder()->is_loaded()) {
 559       int offset = field->offset();
 560       int index = _newobjects.find(object);
 561       if (index != -1) {
 562         // newly allocated object with no other stores performed on this field
 563         FieldBuffer* buf = _fields.at(index);
 564         if (buf->at(field) == NULL && is_default_value(value)) {
 565 #ifndef PRODUCT
 566           if (PrintIRDuringConstruction && Verbose) {
 567             tty->print_cr("Eliminated store for object %d:", index);
 568             st->print_line();
 569           }
 570 #endif
 571           return NULL;
 572         } else {
 573           buf->at_put(field, value);
 574         }
 575       } else {
 576         _objects.at_put_grow(offset, object, NULL);
 577         _values.at_put(field, value);
 578       }
 579 
 580       store_value(value);
 581     } else {
 582       // if we held onto field names we could alias based on names but
 583       // we don't know what's being stored to so kill it all.
 584       kill();
 585     }
 586     return st;
 587   }
 588 
 589 
 590   // return true if this value correspond to the default value of a field.
 591   bool is_default_value(Value value) {
 592     Constant* con = value->as_Constant();
 593     if (con) {
 594       switch (con->type()->tag()) {
 595         case intTag:    return con->type()->as_IntConstant()->value() == 0;
 596         case longTag:   return con->type()->as_LongConstant()->value() == 0;
 597         case floatTag:  return jint_cast(con->type()->as_FloatConstant()->value()) == 0;
 598         case doubleTag: return jlong_cast(con->type()->as_DoubleConstant()->value()) == jlong_cast(0);
 599         case objectTag: return con->type() == objectNull;
 600         default:  ShouldNotReachHere();
 601       }
 602     }
 603     return false;
 604   }
 605 
 606 
 607   // return either the actual value of a load or the load itself
 608   Value load(LoadField* load) {
 609     if (!EliminateFieldAccess) {
 610       return load;
 611     }
 612 
 613     if (strict_fp_requires_explicit_rounding && load->type()->is_float_kind()) {
 614 #ifdef IA32
 615       if (UseSSE < 2) {
 616         // can't skip load since value might get rounded as a side effect
 617         return load;
 618       }
 619 #else
 620       Unimplemented();
 621 #endif // IA32
 622     }
 623 
 624     ciField* field = load->field();
 625     Value object   = load->obj();
 626     if (field->holder()->is_loaded() && !field->is_volatile()) {
 627       int offset = field->offset();
 628       Value result = NULL;
 629       int index = _newobjects.find(object);
 630       if (index != -1) {
 631         result = _fields.at(index)->at(field);
 632       } else if (_objects.at_grow(offset, NULL) == object) {
 633         result = _values.at(field);
 634       }
 635       if (result != NULL) {
 636 #ifndef PRODUCT
 637         if (PrintIRDuringConstruction && Verbose) {
 638           tty->print_cr("Eliminated load: ");
 639           load->print_line();
 640         }
 641 #endif
 642         assert(result->type()->tag() == load->type()->tag(), "wrong types");
 643         return result;
 644       }
 645     }
 646     return load;
 647   }
 648 
 649   // Record this newly allocated object
 650   void new_instance(NewInstance* object) {
 651     int index = _newobjects.length();
 652     _newobjects.append(object);
 653     if (_fields.at_grow(index, NULL) == NULL) {
 654       _fields.at_put(index, new FieldBuffer());
 655     } else {
 656       _fields.at(index)->kill();
 657     }
 658   }
 659 
 660   // Record this newly allocated object
 661   void new_instance(NewValueTypeInstance* object) {
 662     int index = _newobjects.length();
 663     _newobjects.append(object);
 664     if (_fields.at_grow(index, NULL) == NULL) {
 665       _fields.at_put(index, new FieldBuffer());
 666     } else {
 667       _fields.at(index)->kill();
 668     }
 669   }
 670 
 671   void store_value(Value value) {
 672     int index = _newobjects.find(value);
 673     if (index != -1) {
 674       // stored a newly allocated object into another object.
 675       // Assume we've lost track of it as separate slice of memory.
 676       // We could do better by keeping track of whether individual
 677       // fields could alias each other.
 678       _newobjects.remove_at(index);
 679       // pull out the field info and store it at the end up the list
 680       // of field info list to be reused later.
 681       _fields.append(_fields.at(index));
 682       _fields.remove_at(index);
 683     }
 684   }
 685 
 686   void kill() {
 687     _newobjects.trunc_to(0);
 688     _objects.trunc_to(0);
 689     _values.kill();
 690   }
 691 };

 732 BlockBegin* GraphBuilder::ScopeData::block_at(int bci) {
 733   if (parsing_jsr()) {
 734     // It is necessary to clone all blocks associated with a
 735     // subroutine, including those for exception handlers in the scope
 736     // of the method containing the jsr (because those exception
 737     // handlers may contain ret instructions in some cases).
 738     BlockBegin* block = bci2block()->at(bci);
 739     if (block != NULL && block == parent()->bci2block()->at(bci)) {
 740       BlockBegin* new_block = new BlockBegin(block->bci());
 741       if (PrintInitialBlockList) {
 742         tty->print_cr("CFG: cloned block %d (bci %d) as block %d for jsr",
 743                       block->block_id(), block->bci(), new_block->block_id());
 744       }
 745       // copy data from cloned blocked
 746       new_block->set_depth_first_number(block->depth_first_number());
 747       if (block->is_set(BlockBegin::parser_loop_header_flag)) new_block->set(BlockBegin::parser_loop_header_flag);
 748       // Preserve certain flags for assertion checking
 749       if (block->is_set(BlockBegin::subroutine_entry_flag)) new_block->set(BlockBegin::subroutine_entry_flag);
 750       if (block->is_set(BlockBegin::exception_entry_flag))  new_block->set(BlockBegin::exception_entry_flag);
 751 
 752       // copy was_visited_flag to allow early detection of bailouts
 753       // if a block that is used in a jsr has already been visited before,
 754       // it is shared between the normal control flow and a subroutine
 755       // BlockBegin::try_merge returns false when the flag is set, this leads
 756       // to a compilation bailout
 757       if (block->is_set(BlockBegin::was_visited_flag))  new_block->set(BlockBegin::was_visited_flag);
 758 
 759       bci2block()->at_put(bci, new_block);
 760       block = new_block;
 761     }
 762     return block;
 763   } else {
 764     return bci2block()->at(bci);
 765   }
 766 }

 823 void GraphBuilder::sort_top_into_worklist(BlockList* worklist, BlockBegin* top) {
 824   assert(worklist->top() == top, "");
 825   // sort block descending into work list
 826   const int dfn = top->depth_first_number();
 827   assert(dfn != -1, "unknown depth first number");
 828   int i = worklist->length()-2;
 829   while (i >= 0) {
 830     BlockBegin* b = worklist->at(i);
 831     if (b->depth_first_number() < dfn) {
 832       worklist->at_put(i+1, b);
 833     } else {
 834       break;
 835     }
 836     i --;
 837   }
 838   if (i >= -1) worklist->at_put(i + 1, top);
 839 }

 896 void GraphBuilder::load_constant() {
 897   ciConstant con = stream()->get_constant();
 898   if (con.basic_type() == T_ILLEGAL) {
 899     // FIXME: an unresolved Dynamic constant can get here,
 900     // and that should not terminate the whole compilation.
 901     BAILOUT("could not resolve a constant");
 902   } else {
 903     ValueType* t = illegalType;
 904     ValueStack* patch_state = NULL;
 905     switch (con.basic_type()) {
 906       case T_BOOLEAN: t = new IntConstant     (con.as_boolean()); break;
 907       case T_BYTE   : t = new IntConstant     (con.as_byte   ()); break;
 908       case T_CHAR   : t = new IntConstant     (con.as_char   ()); break;
 909       case T_SHORT  : t = new IntConstant     (con.as_short  ()); break;
 910       case T_INT    : t = new IntConstant     (con.as_int    ()); break;
 911       case T_LONG   : t = new LongConstant    (con.as_long   ()); break;
 912       case T_FLOAT  : t = new FloatConstant   (con.as_float  ()); break;
 913       case T_DOUBLE : t = new DoubleConstant  (con.as_double ()); break;
 914       case T_ARRAY  : t = new ArrayConstant   (con.as_object ()->as_array   ()); break;
 915       case T_OBJECT :
 916        {
 917         ciObject* obj = con.as_object();
 918         if (!obj->is_loaded()
 919             || (PatchALot && obj->klass() != ciEnv::current()->String_klass())) {
 920           // A Class, MethodType, MethodHandle, or String.
 921           // Unloaded condy nodes show up as T_ILLEGAL, above.
 922           patch_state = copy_state_before();
 923           t = new ObjectConstant(obj);
 924         } else {
 925           // Might be a Class, MethodType, MethodHandle, or Dynamic constant
 926           // result, which might turn out to be an array.
 927           if (obj->is_null_object())
 928             t = objectNull;
 929           else if (obj->is_array())
 930             t = new ArrayConstant(obj->as_array());
 931           else
 932             t = new InstanceConstant(obj->as_instance());
 933         }
 934         break;
 935        }
 936       default       : ShouldNotReachHere();
 937     }
 938     Value x;
 939     if (patch_state != NULL) {
 940       x = new Constant(t, patch_state);
 941     } else {
 942       x = new Constant(t);
 943     }
 944     push(t, append(x));
 945   }
 946 }

 962 void GraphBuilder::store_local(ValueStack* state, Value x, int index) {
 963   if (parsing_jsr()) {
 964     // We need to do additional tracking of the location of the return
 965     // address for jsrs since we don't handle arbitrary jsr/ret
 966     // constructs. Here we are figuring out in which circumstances we
 967     // need to bail out.
 968     if (x->type()->is_address()) {
 969       scope_data()->set_jsr_return_address_local(index);
 970 
 971       // Also check parent jsrs (if any) at this time to see whether
 972       // they are using this local. We don't handle skipping over a
 973       // ret.
 974       for (ScopeData* cur_scope_data = scope_data()->parent();
 975            cur_scope_data != NULL && cur_scope_data->parsing_jsr() && cur_scope_data->scope() == scope();
 976            cur_scope_data = cur_scope_data->parent()) {
 977         if (cur_scope_data->jsr_return_address_local() == index) {
 978           BAILOUT("subroutine overwrites return address from previous subroutine");
 979         }
 980       }
 981     } else if (index == scope_data()->jsr_return_address_local()) {
 982       scope_data()->set_jsr_return_address_local(-1);
 983     }
 984   }
 985 
 986   x->set_local_index(index);
 987   state->store_local(index, round_fp(x));
 988 }

 991 void GraphBuilder::load_indexed(BasicType type) {
 992   // In case of in block code motion in range check elimination
 993   ValueStack* state_before = NULL;
 994   int array_idx = state()->stack_size() - 2;
 995   if (type == T_OBJECT && state()->stack_at(array_idx)->maybe_flattened_array()) {
 996     // Save the entire state and re-execute on deopt when accessing flattened arrays
 997     state_before = copy_state_before();
 998     state_before->set_should_reexecute(true);
 999   } else {
1000     state_before = copy_state_indexed_access();
1001   }
1002   compilation()->set_has_access_indexed(true);
1003   Value index = ipop();
1004   Value array = apop();
1005   Value length = NULL;
1006   if (CSEArrayLength ||
1007       (array->as_AccessField() && array->as_AccessField()->field()->is_constant()) ||
1008       (array->as_NewArray() && array->as_NewArray()->length() && array->as_NewArray()->length()->type()->is_constant())) {
1009     length = append(new ArrayLength(array, state_before));
1010   }
1011 
1012   LoadIndexed* load_indexed = NULL;
1013   Instruction* result = NULL;
1014   if (array->is_loaded_flattened_array()) {
1015     ciType* array_type = array->declared_type();
1016     ciValueKlass* elem_klass = array_type->as_value_array_klass()->element_klass()->as_value_klass();
1017     NewValueTypeInstance* new_instance = new NewValueTypeInstance(elem_klass, state_before, false);
1018     _memory->new_instance(new_instance);
1019     apush(append_split(new_instance));
1020     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1021     load_indexed->set_vt(new_instance);
1022   } else {
1023     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1024   }
1025   if (profile_array_accesses() && is_reference_type(type)) {
1026     compilation()->set_would_profile(true);
1027     load_indexed->set_should_profile(true);
1028     load_indexed->set_profiled_method(method());
1029     load_indexed->set_profiled_bci(bci());
1030   }
1031   result = append(load_indexed);
1032   assert(!(profile_array_accesses() && is_reference_type(type)) || load_indexed == result, "should not be optimized out");
1033   if (!array->is_loaded_flattened_array()) {
1034     push(as_ValueType(type), result);
1035   }
1036 }

1089   switch (code) {
1090     case Bytecodes::_pop:
1091       { state()->raw_pop();
1092       }
1093       break;
1094     case Bytecodes::_pop2:
1095       { state()->raw_pop();
1096         state()->raw_pop();
1097       }
1098       break;
1099     case Bytecodes::_dup:
1100       { Value w = state()->raw_pop();
1101         state()->raw_push(w);
1102         state()->raw_push(w);

1104       break;

1118         state()->raw_push(w3);

1120         state()->raw_push(w1);
1121       }
1122       break;

1126         state()->raw_push(w2);

1190 void GraphBuilder::shift_op(ValueType* type, Bytecodes::Code code) {
1191   Value s = ipop();
1192   Value x = pop(type);
1193   // try to simplify
1194   // Note: This code should go into the canonicalizer as soon as it can
1195   //       can handle canonicalized forms that contain more than one node.
1196   if (CanonicalizeNodes && code == Bytecodes::_iushr) {
1197     // pattern: x >>> s
1198     IntConstant* s1 = s->type()->as_IntConstant();
1199     if (s1 != NULL) {
1200       // pattern: x >>> s1, with s1 constant
1201       ShiftOp* l = x->as_ShiftOp();
1202       if (l != NULL && l->op() == Bytecodes::_ishl) {
1203         // pattern: (a << b) >>> s1
1204         IntConstant* s0 = l->y()->type()->as_IntConstant();
1205         if (s0 != NULL) {
1206           // pattern: (a << s0) >>> s1
1207           const int s0c = s0->value() & 0x1F; // only the low 5 bits are significant for shifts
1208           const int s1c = s1->value() & 0x1F; // only the low 5 bits are significant for shifts
1209           if (s0c == s1c) {
1210             if (s0c == 0) {
1211               // pattern: (a << 0) >>> 0 => simplify to: a
1212               ipush(l->x());
1213             } else {
1214               // pattern: (a << s0c) >>> s0c => simplify to: a & m, with m constant
1215               assert(0 < s0c && s0c < BitsPerInt, "adjust code below to handle corner cases");
1216               const int m = (1 << (BitsPerInt - s0c)) - 1;
1217               Value s = append(new Constant(new IntConstant(m)));
1218               ipush(append(new LogicOp(Bytecodes::_iand, l->x(), s)));
1219             }
1220             return;
1221           }
1222         }
1223       }
1224     }
1225   }
1226   // could not simplify
1227   push(type, append(new ShiftOp(code, x, s)));
1228 }

1275 void GraphBuilder::if_node(Value x, If::Condition cond, Value y, ValueStack* state_before) {
1276   BlockBegin* tsux = block_at(stream()->get_dest());
1277   BlockBegin* fsux = block_at(stream()->next_bci());
1278   bool is_bb = tsux->bci() < stream()->cur_bci() || fsux->bci() < stream()->cur_bci();
1279 
1280   bool subst_check = false;
1281   if (EnableValhalla && (stream()->cur_bc() == Bytecodes::_if_acmpeq || stream()->cur_bc() == Bytecodes::_if_acmpne) &&
1282       method() != ciEnv::current()->ValueBootstrapMethods_klass()->find_method(ciSymbol::isSubstitutable_name(), ciSymbol::object_object_boolean_signature())) {
1283     // If current method is ValueBootstrapMethods::isSubstitutable(),
1284     // compile the acmp as a regular pointer comparison otherwise we
1285     // could call ValueBootstrapMethods::isSubstitutable() back
1286     ValueType* left_vt = x->type();
1287     ValueType* right_vt = y->type();
1288     if (left_vt->is_object()) {
1289       assert(right_vt->is_object(), "must be");
1290       ciKlass* left_klass = x->as_loaded_klass_or_null();
1291       ciKlass* right_klass = y->as_loaded_klass_or_null();
1292 
1293       if (left_klass == NULL || right_klass == NULL) {
1294         // The klass is still unloaded, or came from a Phi node. Go slow case;
1295         subst_check = true;
1296       } else if (left_klass->can_be_value_klass() || right_klass->can_be_value_klass()) {
1297         // Either operand may be a value object, but we're not sure. Go slow case;
1298         subst_check = true;
1299       } else {
1300         // No need to do substitutability check
1301       }
1302     }
1303   }
1304 
1305   // In case of loop invariant code motion or predicate insertion
1306   // before the body of a loop the state is needed
1307   Instruction *i = append(new If(x, cond, false, y, tsux, fsux, (is_bb || compilation()->is_optimistic() || subst_check) ? state_before : NULL, is_bb, subst_check));
1308 
1309   assert(i->as_Goto() == NULL ||
1310          (i->as_Goto()->sux_at(0) == tsux  && i->as_Goto()->is_safepoint() == tsux->bci() < stream()->cur_bci()) ||
1311          (i->as_Goto()->sux_at(0) == fsux  && i->as_Goto()->is_safepoint() == fsux->bci() < stream()->cur_bci()),
1312          "safepoint state of Goto returned by canonicalizer incorrect");
1313 
1314   if (is_profiling()) {
1315     If* if_node = i->as_If();
1316     if (if_node != NULL) {
1317       // Note that we'd collect profile data in this method if we wanted it.
1318       compilation()->set_would_profile(true);
1319       // At level 2 we need the proper bci to count backedges
1320       if_node->set_profiled_bci(bci());
1321       if (profile_branches()) {
1322         // Successors can be rotated by the canonicalizer, check for this case.
1323         if_node->set_profiled_method(method());
1324         if_node->set_should_profile(true);
1325         if (if_node->tsux() == fsux) {
1326           if_node->set_swapped(true);
1327         }
1328       }
1329       return;
1330     }
1331 
1332     // Check if this If was reduced to Goto.
1333     Goto *goto_node = i->as_Goto();
1334     if (goto_node != NULL) {
1335       compilation()->set_would_profile(true);
1336       goto_node->set_profiled_bci(bci());
1337       if (profile_branches()) {
1338         goto_node->set_profiled_method(method());
1339         goto_node->set_should_profile(true);
1340         // Find out which successor is used.
1341         if (goto_node->default_sux() == tsux) {
1342           goto_node->set_direction(Goto::taken);
1343         } else if (goto_node->default_sux() == fsux) {
1344           goto_node->set_direction(Goto::not_taken);
1345         } else {
1346           ShouldNotReachHere();
1347         }
1348       }
1349       return;
1350     }
1351   }
1352 }

1411 void GraphBuilder::table_switch() {
1412   Bytecode_tableswitch sw(stream());
1413   const int l = sw.length();
1414   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1415     // total of 2 successors => use If instead of switch
1416     // Note: This code should go into the canonicalizer as soon as it can
1417     //       can handle canonicalized forms that contain more than one node.
1418     Value key = append(new Constant(new IntConstant(sw.low_key())));
1419     BlockBegin* tsux = block_at(bci() + sw.dest_offset_at(0));
1420     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1421     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1422     // In case of loop invariant code motion or predicate insertion
1423     // before the body of a loop the state is needed
1424     ValueStack* state_before = copy_state_if_bb(is_bb);
1425     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1426   } else {
1427     // collect successors
1428     BlockList* sux = new BlockList(l + 1, NULL);
1429     int i;
1430     bool has_bb = false;
1431     for (i = 0; i < l; i++) {
1432       sux->at_put(i, block_at(bci() + sw.dest_offset_at(i)));
1433       if (sw.dest_offset_at(i) < 0) has_bb = true;
1434     }
1435     // add default successor
1436     if (sw.default_offset() < 0) has_bb = true;
1437     sux->at_put(i, block_at(bci() + sw.default_offset()));
1438     // In case of loop invariant code motion or predicate insertion
1439     // before the body of a loop the state is needed
1440     ValueStack* state_before = copy_state_if_bb(has_bb);
1441     Instruction* res = append(new TableSwitch(ipop(), sux, sw.low_key(), state_before, has_bb));
1442 #ifdef ASSERT
1443     if (res->as_Goto()) {
1444       for (i = 0; i < l; i++) {
1445         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1446           assert(res->as_Goto()->is_safepoint() == sw.dest_offset_at(i) < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1447         }
1448       }
1449     }
1450 #endif
1451   }
1452 }

1455 void GraphBuilder::lookup_switch() {
1456   Bytecode_lookupswitch sw(stream());
1457   const int l = sw.number_of_pairs();
1458   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1459     // total of 2 successors => use If instead of switch
1460     // Note: This code should go into the canonicalizer as soon as it can
1461     //       can handle canonicalized forms that contain more than one node.
1462     // simplify to If
1463     LookupswitchPair pair = sw.pair_at(0);
1464     Value key = append(new Constant(new IntConstant(pair.match())));
1465     BlockBegin* tsux = block_at(bci() + pair.offset());
1466     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1467     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1468     // In case of loop invariant code motion or predicate insertion
1469     // before the body of a loop the state is needed
1470     ValueStack* state_before = copy_state_if_bb(is_bb);;
1471     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1472   } else {
1473     // collect successors & keys
1474     BlockList* sux = new BlockList(l + 1, NULL);
1475     intArray* keys = new intArray(l, l, 0);
1476     int i;
1477     bool has_bb = false;
1478     for (i = 0; i < l; i++) {
1479       LookupswitchPair pair = sw.pair_at(i);
1480       if (pair.offset() < 0) has_bb = true;
1481       sux->at_put(i, block_at(bci() + pair.offset()));
1482       keys->at_put(i, pair.match());
1483     }
1484     // add default successor
1485     if (sw.default_offset() < 0) has_bb = true;
1486     sux->at_put(i, block_at(bci() + sw.default_offset()));
1487     // In case of loop invariant code motion or predicate insertion
1488     // before the body of a loop the state is needed
1489     ValueStack* state_before = copy_state_if_bb(has_bb);
1490     Instruction* res = append(new LookupSwitch(ipop(), sux, keys, state_before, has_bb));
1491 #ifdef ASSERT
1492     if (res->as_Goto()) {
1493       for (i = 0; i < l; i++) {
1494         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1495           assert(res->as_Goto()->is_safepoint() == sw.pair_at(i).offset() < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1496         }
1497       }
1498     }
1499 #endif
1500   }
1501 }

1503 void GraphBuilder::call_register_finalizer() {
1504   // If the receiver requires finalization then emit code to perform
1505   // the registration on return.
1506 
1507   // Gather some type information about the receiver
1508   Value receiver = state()->local_at(0);
1509   assert(receiver != NULL, "must have a receiver");
1510   ciType* declared_type = receiver->declared_type();
1511   ciType* exact_type = receiver->exact_type();
1512   if (exact_type == NULL &&
1513       receiver->as_Local() &&
1514       receiver->as_Local()->java_index() == 0) {
1515     ciInstanceKlass* ik = compilation()->method()->holder();
1516     if (ik->is_final()) {
1517       exact_type = ik;
1518     } else if (UseCHA && !(ik->has_subklass() || ik->is_interface())) {
1519       // test class is leaf class
1520       compilation()->dependency_recorder()->assert_leaf_type(ik);
1521       exact_type = ik;
1522     } else {
1523       declared_type = ik;
1524     }
1525   }
1526 
1527   // see if we know statically that registration isn't required
1528   bool needs_check = true;
1529   if (exact_type != NULL) {
1530     needs_check = exact_type->as_instance_klass()->has_finalizer();
1531   } else if (declared_type != NULL) {
1532     ciInstanceKlass* ik = declared_type->as_instance_klass();
1533     if (!Dependencies::has_finalizable_subclass(ik)) {
1534       compilation()->dependency_recorder()->assert_has_no_finalizable_subclasses(ik);
1535       needs_check = false;
1536     }
1537   }
1538 
1539   if (needs_check) {
1540     // Perform the registration of finalizable objects.
1541     ValueStack* state_before = copy_state_for_exception();
1542     load_local(objectType, 0);
1543     append_split(new Intrinsic(voidType, vmIntrinsics::_Object_init,
1544                                state()->pop_arguments(1),
1545                                true, state_before, true));
1546   }
1547 }

1550 void GraphBuilder::method_return(Value x, bool ignore_return) {
1551   if (RegisterFinalizersAtInit &&
1552       method()->intrinsic_id() == vmIntrinsics::_Object_init) {
1553     call_register_finalizer();
1554   }
1555 
1556   // The conditions for a memory barrier are described in Parse::do_exits().
1557   bool need_mem_bar = false;
1558   if (method()->is_object_constructor() &&
1559        (scope()->wrote_final() ||
1560          (AlwaysSafeConstructors && scope()->wrote_fields()) ||
1561          (support_IRIW_for_not_multiple_copy_atomic_cpu && scope()->wrote_volatile()))) {
1562     need_mem_bar = true;
1563   }
1564 
1565   BasicType bt = method()->return_type()->basic_type();
1566   switch (bt) {
1567     case T_BYTE:
1568     {
1569       Value shift = append(new Constant(new IntConstant(24)));
1570       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1571       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1572       break;
1573     }
1574     case T_SHORT:
1575     {
1576       Value shift = append(new Constant(new IntConstant(16)));
1577       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1578       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1579       break;
1580     }
1581     case T_CHAR:
1582     {
1583       Value mask = append(new Constant(new IntConstant(0xFFFF)));
1584       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1585       break;
1586     }
1587     case T_BOOLEAN:
1588     {
1589       Value mask = append(new Constant(new IntConstant(1)));
1590       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1591       break;
1592     }
1593     default:
1594       break;
1595   }
1596 
1597   // Check to see whether we are inlining. If so, Return
1598   // instructions become Gotos to the continuation point.
1599   if (continuation() != NULL) {
1600 
1601     int invoke_bci = state()->caller_state()->bci();
1602 
1603     if (x != NULL  && !ignore_return) {
1604       ciMethod* caller = state()->scope()->caller()->method();
1605       Bytecodes::Code invoke_raw_bc = caller->raw_code_at_bci(invoke_bci);
1606       if (invoke_raw_bc == Bytecodes::_invokehandle || invoke_raw_bc == Bytecodes::_invokedynamic) {
1607         ciType* declared_ret_type = caller->get_declared_signature_at_bci(invoke_bci)->return_type();
1608         if (declared_ret_type->is_klass() && x->exact_type() == NULL &&
1609             x->declared_type() != declared_ret_type && declared_ret_type != compilation()->env()->Object_klass()) {
1610           x = append(new TypeCast(declared_ret_type->as_klass(), x, copy_state_before()));
1611         }
1612       }
1613     }
1614 
1615     assert(!method()->is_synchronized() || InlineSynchronizedMethods, "can not inline synchronized methods yet");
1616 
1617     if (compilation()->env()->dtrace_method_probes()) {
1618       // Report exit from inline methods
1619       Values* args = new Values(1);
1620       args->push(append(new Constant(new MethodConstant(method()))));
1621       append(new RuntimeCall(voidType, "dtrace_method_exit", CAST_FROM_FN_PTR(address, SharedRuntime::dtrace_method_exit), args));
1622     }
1623 
1624     // If the inlined method is synchronized, the monitor must be
1625     // released before we jump to the continuation block.
1626     if (method()->is_synchronized()) {
1627       assert(state()->locks_size() == 1, "receiver must be locked here");
1628       monitorexit(state()->lock_at(0), SynchronizationEntryBCI);
1629     }
1630 
1631     if (need_mem_bar) {
1632       append(new MemBar(lir_membar_storestore));
1633     }
1634 
1635     // State at end of inlined method is the state of the caller
1636     // without the method parameters on stack, including the
1637     // return value, if any, of the inlined method on operand stack.
1638     set_state(state()->caller_state()->copy_for_parsing());
1639     if (x != NULL) {
1640       if (!ignore_return) {
1641         state()->push(x->type(), x);
1642       }
1643       if (profile_return() && x->type()->is_object_kind()) {
1644         ciMethod* caller = state()->scope()->method();
1645         profile_return_type(x, method(), caller, invoke_bci);
1646       }
1647     }
1648     Goto* goto_callee = new Goto(continuation(), false);
1649 
1650     // See whether this is the first return; if so, store off some
1651     // of the state for later examination
1652     if (num_returns() == 0) {
1653       set_inline_cleanup_info();
1654     }
1655 
1656     // The current bci() is in the wrong scope, so use the bci() of
1657     // the continuation point.
1658     append_with_bci(goto_callee, scope_data()->continuation()->bci());
1659     incr_num_returns();
1660     return;
1661   }
1662 
1663   state()->truncate_stack(0);
1664   if (method()->is_synchronized()) {
1665     // perform the unlocking before exiting the method
1666     Value receiver;
1667     if (!method()->is_static()) {
1668       receiver = _initial_state->local_at(0);
1669     } else {
1670       receiver = append(new Constant(new ClassConstant(method()->holder())));
1671     }
1672     append_split(new MonitorExit(receiver, state()->unlock()));
1673   }
1674 
1675   if (need_mem_bar) {
1676       append(new MemBar(lir_membar_storestore));
1677   }
1678 
1679   assert(!ignore_return, "Ignoring return value works only for inlining");
1680   append(new Return(x));
1681 }

1723 void GraphBuilder::access_field(Bytecodes::Code code) {
1724   bool will_link;
1725   ciField* field = stream()->get_field(will_link);
1726   ciInstanceKlass* holder = field->holder();
1727   BasicType field_type = field->type()->basic_type();
1728   ValueType* type = as_ValueType(field_type);
1729 
1730   // call will_link again to determine if the field is valid.
1731   const bool needs_patching = !holder->is_loaded() ||
1732                               !field->will_link(method(), code) ||
1733                               PatchALot;
1734 
1735   ValueStack* state_before = NULL;
1736   if (!holder->is_initialized() || needs_patching) {
1737     // save state before instruction for debug info when
1738     // deoptimization happens during patching
1739     state_before = copy_state_before();
1740   }
1741 
1742   Value obj = NULL;
1743   if (code == Bytecodes::_getstatic || code == Bytecodes::_putstatic) {
1744     if (state_before != NULL) {
1745       // build a patching constant
1746       obj = new Constant(new InstanceConstant(holder->java_mirror()), state_before);
1747     } else {
1748       obj = new Constant(new InstanceConstant(holder->java_mirror()));
1749     }
1750   }
1751 
1752   if (field->is_final() && (code == Bytecodes::_putfield || code == Bytecodes::_withfield)) {
1753     scope()->set_wrote_final();
1754   }
1755 
1756   if (code == Bytecodes::_putfield || code == Bytecodes::_withfield) {
1757     scope()->set_wrote_fields();
1758     if (field->is_volatile()) {
1759       scope()->set_wrote_volatile();
1760     }
1761   }
1762 
1763   const int offset = !needs_patching ? field->offset() : -1;
1764   switch (code) {
1765     case Bytecodes::_getstatic: {
1766       // check for compile-time constants, i.e., initialized static final fields
1767       Value constant = NULL;
1768       if (field->is_static_constant() && !PatchALot) {
1769         ciConstant field_value = field->constant_value();
1770         assert(!field->is_stable() || !field_value.is_null_or_zero(),
1771                "stable static w/ default value shouldn't be a constant");
1772         constant = make_constant(field_value, field);
1773       }
1774       if (constant != NULL) {
1775         push(type, append(constant));
1776       } else {
1777         if (state_before == NULL) {
1778           state_before = copy_state_for_exception();
1779         }
1780         LoadField* load_field = new LoadField(append(obj), offset, field, true,
1781                                         state_before, needs_patching);
1782         if (field->is_flattenable()) {
1783           load_field->set_never_null(true);
1784         }
1785         push(type, append(load_field));
1786       }
1787       break;
1788     }
1789     case Bytecodes::_putstatic: {
1790       Value val = pop(type);
1791       val->set_escaped();
1792       if (state_before == NULL) {
1793         state_before = copy_state_for_exception();
1794       }
1795       if (field->type()->basic_type() == T_BOOLEAN) {
1796         Value mask = append(new Constant(new IntConstant(1)));
1797         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1798       }
1799       append(new StoreField(append(obj), offset, field, val, true, state_before, needs_patching));
1800       break;
1801     }
1802     case Bytecodes::_getfield: {
1803       // Check for compile-time constants, i.e., trusted final non-static fields.
1804       Value constant = NULL;
1805       if (state_before == NULL && field->is_flattened()) {
1806         // Save the entire state and re-execute on deopt when accessing flattened fields
1807         assert(Interpreter::bytecode_should_reexecute(code), "should reexecute");
1808         state_before = copy_state_before();
1809       }
1810       obj = apop();
1811       ObjectType* obj_type = obj->type()->as_ObjectType();
1812       if (field->is_constant() && !field->is_flattened() && obj_type->is_constant() && !PatchALot) {
1813         ciObject* const_oop = obj_type->constant_value();
1814         if (!const_oop->is_null_object() && const_oop->is_loaded()) {
1815           ciConstant field_value = field->constant_value_of(const_oop);
1816           if (field_value.is_valid()) {
1817             if (field->is_flattenable() && field_value.is_null_or_zero()) {
1818               // Non-flattened but flattenable inline type field. Replace null by the default value.
1819               constant = new Constant(new InstanceConstant(field->type()->as_value_klass()->default_value_instance()));
1820             } else {
1821               constant = make_constant(field_value, field);
1822             }
1823             // For CallSite objects add a dependency for invalidation of the optimization.
1824             if (field->is_call_site_target()) {
1825               ciCallSite* call_site = const_oop->as_call_site();
1826               if (!call_site->is_fully_initialized_constant_call_site()) {
1827                 ciMethodHandle* target = field_value.as_object()->as_method_handle();
1828                 dependency_recorder()->assert_call_site_target_value(call_site, target);
1829               }
1830             }
1831           }
1832         }
1833       }
1834       if (constant != NULL) {
1835         push(type, append(constant));
1836       } else {
1837         if (state_before == NULL) {
1838           state_before = copy_state_for_exception();
1839         }
1840         if (!field->is_flattened()) {
1841           LoadField* load = new LoadField(obj, offset, field, false, state_before, needs_patching);
1842           Value replacement = !needs_patching ? _memory->load(load) : load;
1843           if (replacement != load) {
1844             assert(replacement->is_linked() || !replacement->can_be_linked(), "should already by linked");
1845             // Writing an (integer) value to a boolean, byte, char or short field includes an implicit narrowing
1846             // conversion. Emit an explicit conversion here to get the correct field value after the write.
1847             BasicType bt = field->type()->basic_type();
1848             switch (bt) {
1849             case T_BOOLEAN:
1850             case T_BYTE:
1851               replacement = append(new Convert(Bytecodes::_i2b, replacement, as_ValueType(bt)));
1852               break;
1853             case T_CHAR:
1854               replacement = append(new Convert(Bytecodes::_i2c, replacement, as_ValueType(bt)));
1855               break;
1856             case T_SHORT:
1857               replacement = append(new Convert(Bytecodes::_i2s, replacement, as_ValueType(bt)));
1858               break;
1859             default:
1860               break;
1861             }
1862             push(type, replacement);
1863           } else {
1864             push(type, append(load));
1865           }
1866         } else { // flattened field, not optimized solution: re-instantiate the flattened value
1867           assert(field->type()->is_valuetype(), "Sanity check");
1868           ciValueKlass* value_klass = field->type()->as_value_klass();
1869           int flattening_offset = field->offset() - value_klass->first_field_offset();
1870           assert(field->type()->is_valuetype(), "Sanity check");
1871           scope()->set_wrote_final();
1872           scope()->set_wrote_fields();
1873           NewValueTypeInstance* new_instance = new NewValueTypeInstance(value_klass, state_before, false);
1874           _memory->new_instance(new_instance);
1875           apush(append_split(new_instance));
1876           copy_value_content(value_klass, obj, field->offset(), new_instance, value_klass->first_field_offset(),
1877                        state_before, needs_patching);
1878         }
1879       }
1880       break;
1881     }
1882     case Bytecodes::_withfield:
1883     case Bytecodes::_putfield: {
1884       Value val = pop(type);
1885       val->set_escaped();
1886       obj = apop();
1887       if (state_before == NULL) {
1888         state_before = copy_state_for_exception();
1889       }
1890       if (field->type()->basic_type() == T_BOOLEAN) {
1891         Value mask = append(new Constant(new IntConstant(1)));
1892         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1893       }
1894 
1895       if (!field->is_flattened()) {
1896         StoreField* store = new StoreField(obj, offset, field, val, false, state_before, needs_patching);
1897         if (!needs_patching) store = _memory->store(store);
1898         if (store != NULL) {
1899           append(store);
1900         }
1901       } else {
1902         assert(field->type()->is_valuetype(), "Sanity check");
1903         ciValueKlass* value_klass = field->type()->as_value_klass();
1904         int flattening_offset = field->offset() - value_klass->first_field_offset();
1905         copy_value_content(value_klass, val, value_klass->first_field_offset(), obj, field->offset(),
1906                    state_before, needs_patching);
1907       }
1908       break;
1909     }
1910     default:
1911       ShouldNotReachHere();
1912       break;
1913   }
1914 }

1916 // Baseline version of withfield, allocate every time
1917 void GraphBuilder::withfield(int field_index)
1918 {
1919   bool will_link;
1920   ciField* field_modify = stream()->get_field(will_link);
1921   ciInstanceKlass* holder = field_modify->holder();
1922   BasicType field_type = field_modify->type()->basic_type();
1923   ValueType* type = as_ValueType(field_type);
1924 
1925   // call will_link again to determine if the field is valid.
1926   const bool needs_patching = !holder->is_loaded() ||
1927                               !field_modify->will_link(method(), Bytecodes::_withfield) ||
1928                               PatchALot;
1929 
1930 
1931   scope()->set_wrote_final();
1932   scope()->set_wrote_fields();
1933 
1934   const int offset = !needs_patching ? field_modify->offset() : -1;
1935 
1936   if (!holder->is_loaded()
1937       || needs_patching /* FIXME: 8228634 - field_modify->will_link() may incorrectly return false */
1938       ) {
1939     ValueStack* state_before = copy_state_before();
1940     Value val = pop(type);
1941     Value obj = apop();
1942     apush(append_split(new WithField(state_before)));
1943     return;
1944   }
1945   ValueStack* state_before = copy_state_before();
1946 
1947   Value val = pop(type);
1948   Value obj = apop();
1949 
1950   if (!needs_patching && obj->is_optimizable_for_withfield()) {
1951     int astore_index;
1952     ciBytecodeStream s(method());
1953     s.force_bci(bci());
1954     s.next();
1955     switch (s.cur_bc()) {
1956     case Bytecodes::_astore:    astore_index = s.get_index(); break;
1957     case Bytecodes::_astore_0:  astore_index = 0; break;
1958     case Bytecodes::_astore_1:  astore_index = 1; break;
1959     case Bytecodes::_astore_2:  astore_index = 2; break;
1960     case Bytecodes::_astore_3:  astore_index = 3; break;
1961     default: astore_index = -1;
1962     }
1963 
1964     if (astore_index >= 0 && obj == state()->local_at(astore_index)) {
1965       // We have a sequence like this, where we load a value object from a local slot,
1966       // and overwrite the same local slot with a modified copy of the value object.
1967       //      defaultvalue #1 // class compiler/valhalla/valuetypes/MyValue1
1968       //      astore 9
1969       //      ...
1970       //      iload_0
1971       //      aload 9
1972       //      swap
1973       //      withfield #7 // Field x:I
1974       //      astore 9
1975       // If this object was created by defaultvalue, and has not escaped, and is not stored
1976       // in any other local slots, we can effectively treat the withfield/astore
1977       // sequence as a single putfield bytecode.
1978       push(objectType, obj);
1979       push(type, val);
1980       access_field(Bytecodes::_withfield);
1981       stream()->next(); // skip the next astore/astore_n bytecode.
1982       return;
1983     }
1984   }
1985 
1986   assert(holder->is_valuetype(), "must be a value klass");
1987   // Save the entire state and re-execute on deopt when executing withfield
1988   state_before->set_should_reexecute(true);
1989   NewValueTypeInstance* new_instance = new NewValueTypeInstance(holder->as_value_klass(), state_before, false);
1990   _memory->new_instance(new_instance);
1991   apush(append_split(new_instance));
1992 
1993   for (int i = 0; i < holder->nof_nonstatic_fields(); i++) {
1994     ciField* field = holder->nonstatic_field_at(i);
1995     int off = field->offset();
1996 
1997     if (field->offset() != offset) {
1998       if (field->is_flattened()) {
1999         assert(field->type()->is_valuetype(), "Sanity check");
2000         assert(field->type()->is_valuetype(), "Only value types can be flattened");
2001         ciValueKlass* vk = field->type()->as_value_klass();
2002         copy_value_content(vk, obj, off, new_instance, vk->first_field_offset(), state_before, needs_patching);
2003       } else {
2004         // Only load those fields who are not modified
2005         LoadField* load = new LoadField(obj, off, field, false, state_before, needs_patching);
2006         Value replacement = append(load);
2007         StoreField* store = new StoreField(new_instance, off, field, replacement, false, state_before, needs_patching);
2008         append(store);
2009       }
2010     }
2011   }
2012 
2013   // Field to modify
2014   if (field_modify->type()->basic_type() == T_BOOLEAN) {
2015     Value mask = append(new Constant(new IntConstant(1)));
2016     val = append(new LogicOp(Bytecodes::_iand, val, mask));
2017   }
2018   if (field_modify->is_flattened()) {
2019     assert(field_modify->type()->is_valuetype(), "Only value types can be flattened");
2020     ciValueKlass* vk = field_modify->type()->as_value_klass();
2021     copy_value_content(vk, val, vk->first_field_offset(), new_instance, field_modify->offset(), state_before, needs_patching);
2022   } else {
2023     StoreField* store = new StoreField(new_instance, offset, field_modify, val, false, state_before, needs_patching);
2024     append(store);
2025   }
2026 }

2033 // How many arguments do we want to profile?
2034 Values* GraphBuilder::args_list_for_profiling(ciMethod* target, int& start, bool may_have_receiver) {
2035   int n = 0;
2036   bool has_receiver = may_have_receiver && Bytecodes::has_receiver(method()->java_code_at_bci(bci()));
2037   start = has_receiver ? 1 : 0;
2038   if (profile_arguments()) {
2039     ciProfileData* data = method()->method_data()->bci_to_data(bci());
2040     if (data != NULL && (data->is_CallTypeData() || data->is_VirtualCallTypeData())) {
2041       n = data->is_CallTypeData() ? data->as_CallTypeData()->number_of_arguments() : data->as_VirtualCallTypeData()->number_of_arguments();
2042     }
2043   }
2044   // If we are inlining then we need to collect arguments to profile parameters for the target
2045   if (profile_parameters() && target != NULL) {
2046     if (target->method_data() != NULL && target->method_data()->parameters_type_data() != NULL) {
2047       // The receiver is profiled on method entry so it's included in
2048       // the number of parameters but here we're only interested in
2049       // actual arguments.
2050       n = MAX2(n, target->method_data()->parameters_type_data()->number_of_parameters() - start);
2051     }
2052   }
2053   if (n > 0) {
2054     return new Values(n);
2055   }
2056   return NULL;
2057 }

2088 void GraphBuilder::invoke(Bytecodes::Code code) {
2089   bool will_link;
2090   ciSignature* declared_signature = NULL;
2091   ciMethod*             target = stream()->get_method(will_link, &declared_signature);
2092   ciKlass*              holder = stream()->get_declared_method_holder();
2093   const Bytecodes::Code bc_raw = stream()->cur_bc_raw();
2094   assert(declared_signature != NULL, "cannot be null");
2095   assert(will_link == target->is_loaded(), "");
2096 
2097   ciInstanceKlass* klass = target->holder();
2098   assert(!target->is_loaded() || klass->is_loaded(), "loaded target must imply loaded klass");
2099 
2100   // check if CHA possible: if so, change the code to invoke_special
2101   ciInstanceKlass* calling_klass = method()->holder();
2102   ciInstanceKlass* callee_holder = ciEnv::get_instance_klass_for_declared_method_holder(holder);
2103   ciInstanceKlass* actual_recv = callee_holder;
2104 
2105   CompileLog* log = compilation()->log();
2106   if (log != NULL)
2107       log->elem("call method='%d' instr='%s'",
2108                 log->identify(target),
2109                 Bytecodes::name(code));
2110 
2111   // invoke-special-super
2112   if (bc_raw == Bytecodes::_invokespecial && !target->is_object_constructor()) {
2113     ciInstanceKlass* sender_klass =
2114           calling_klass->is_unsafe_anonymous() ? calling_klass->unsafe_anonymous_host() :
2115                                                  calling_klass;
2116     if (sender_klass->is_interface()) {
2117       int index = state()->stack_size() - (target->arg_size_no_receiver() + 1);
2118       Value receiver = state()->stack_at(index);
2119       CheckCast* c = new CheckCast(sender_klass, receiver, copy_state_before());
2120       c->set_invokespecial_receiver_check();
2121       state()->stack_at_put(index, append_split(c));
2122     }
2123   }
2124 
2125   // Some methods are obviously bindable without any type checks so
2126   // convert them directly to an invokespecial or invokestatic.
2127   if (target->is_loaded() && !target->is_abstract() && target->can_be_statically_bound()) {
2128     switch (bc_raw) {
2129     case Bytecodes::_invokevirtual:
2130       code = Bytecodes::_invokespecial;
2131       break;
2132     case Bytecodes::_invokehandle:
2133       code = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokespecial;
2134       break;
2135     default:
2136       break;
2137     }
2138   } else {
2139     if (bc_raw == Bytecodes::_invokehandle) {
2140       assert(!will_link, "should come here only for unlinked call");
2141       code = Bytecodes::_invokespecial;
2142     }
2143   }
2144 
2145   // Push appendix argument (MethodType, CallSite, etc.), if one.
2146   bool patch_for_appendix = false;
2147   int patching_appendix_arg = 0;
2148   if (Bytecodes::has_optional_appendix(bc_raw) && (!will_link || PatchALot)) {
2149     Value arg = append(new Constant(new ObjectConstant(compilation()->env()->unloaded_ciinstance()), copy_state_before()));
2150     apush(arg);
2151     patch_for_appendix = true;
2152     patching_appendix_arg = (will_link && stream()->has_appendix()) ? 0 : 1;
2153   } else if (stream()->has_appendix()) {
2154     ciObject* appendix = stream()->get_appendix();
2155     Value arg = append(new Constant(new ObjectConstant(appendix)));
2156     apush(arg);
2157   }
2158 
2159   ciMethod* cha_monomorphic_target = NULL;
2160   ciMethod* exact_target = NULL;
2161   Value better_receiver = NULL;
2162   if (UseCHA && DeoptC1 && target->is_loaded() &&
2163       !(// %%% FIXME: Are both of these relevant?
2164         target->is_method_handle_intrinsic() ||
2165         target->is_compiled_lambda_form()) &&
2166       !patch_for_appendix) {
2167     Value receiver = NULL;
2168     ciInstanceKlass* receiver_klass = NULL;
2169     bool type_is_exact = false;
2170     // try to find a precise receiver type
2171     if (will_link && !target->is_static()) {
2172       int index = state()->stack_size() - (target->arg_size_no_receiver() + 1);
2173       receiver = state()->stack_at(index);
2174       ciType* type = receiver->exact_type();
2175       if (type != NULL && type->is_loaded() &&
2176           type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {
2177         receiver_klass = (ciInstanceKlass*) type;
2178         type_is_exact = true;
2179       }
2180       if (type == NULL) {
2181         type = receiver->declared_type();
2182         if (type != NULL && type->is_loaded() &&
2183             type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {
2184           receiver_klass = (ciInstanceKlass*) type;
2185           if (receiver_klass->is_leaf_type() && !receiver_klass->is_final()) {
2186             // Insert a dependency on this type since
2187             // find_monomorphic_target may assume it's already done.
2188             dependency_recorder()->assert_leaf_type(receiver_klass);
2189             type_is_exact = true;
2190           }
2191         }
2192       }
2193     }
2194     if (receiver_klass != NULL && type_is_exact &&
2195         receiver_klass->is_loaded() && code != Bytecodes::_invokespecial) {
2196       // If we have the exact receiver type we can bind directly to
2197       // the method to call.
2198       exact_target = target->resolve_invoke(calling_klass, receiver_klass);
2199       if (exact_target != NULL) {
2200         target = exact_target;
2201         code = Bytecodes::_invokespecial;
2202       }
2203     }
2204     if (receiver_klass != NULL &&
2205         receiver_klass->is_subtype_of(actual_recv) &&
2206         actual_recv->is_initialized()) {
2207       actual_recv = receiver_klass;
2208     }
2209 
2210     if ((code == Bytecodes::_invokevirtual && callee_holder->is_initialized()) ||
2211         (code == Bytecodes::_invokeinterface && callee_holder->is_initialized() && !actual_recv->is_interface())) {
2212       // Use CHA on the receiver to select a more precise method.
2213       cha_monomorphic_target = target->find_monomorphic_target(calling_klass, callee_holder, actual_recv);
2214     } else if (code == Bytecodes::_invokeinterface && callee_holder->is_loaded() && receiver != NULL) {
2215       assert(callee_holder->is_interface(), "invokeinterface to non interface?");
2216       // If there is only one implementor of this interface then we
2217       // may be able bind this invoke directly to the implementing
2218       // klass but we need both a dependence on the single interface
2219       // and on the method we bind to.  Additionally since all we know
2220       // about the receiver type is the it's supposed to implement the
2221       // interface we have to insert a check that it's the class we
2222       // expect.  Interface types are not checked by the verifier so
2223       // they are roughly equivalent to Object.
2224       // The number of implementors for declared_interface is less or
2225       // equal to the number of implementors for target->holder() so
2226       // if number of implementors of target->holder() == 1 then
2227       // number of implementors for decl_interface is 0 or 1. If
2228       // it's 0 then no class implements decl_interface and there's
2229       // no point in inlining.
2230       ciInstanceKlass* declared_interface = callee_holder;
2231       ciInstanceKlass* singleton = declared_interface->unique_implementor();
2232       if (singleton != NULL &&
2233           (!target->is_default_method() || target->is_overpass()) /* CHA doesn't support default methods yet. */ ) {
2234         assert(singleton != declared_interface, "not a unique implementor");
2235         cha_monomorphic_target = target->find_monomorphic_target(calling_klass, declared_interface, singleton);
2236         if (cha_monomorphic_target != NULL) {
2237           if (cha_monomorphic_target->holder() != compilation()->env()->Object_klass()) {
2238             // If CHA is able to bind this invoke then update the class
2239             // to match that class, otherwise klass will refer to the
2240             // interface.
2241             klass = cha_monomorphic_target->holder();
2242             actual_recv = declared_interface;
2243 
2244             // insert a check it's really the expected class.
2245             CheckCast* c = new CheckCast(klass, receiver, copy_state_for_exception());
2246             c->set_incompatible_class_change_check();
2247             c->set_direct_compare(klass->is_final());
2248             // pass the result of the checkcast so that the compiler has
2249             // more accurate type info in the inlinee
2250             better_receiver = append_split(c);
2251           } else {
2252             cha_monomorphic_target = NULL; // subtype check against Object is useless
2253           }
2254         }
2255       }
2256     }
2257   }
2258 
2259   if (cha_monomorphic_target != NULL) {
2260     assert(!target->can_be_statically_bound() || target == cha_monomorphic_target, "");
2261     assert(!cha_monomorphic_target->is_abstract(), "");
2262     if (!cha_monomorphic_target->can_be_statically_bound(actual_recv)) {
2263       // If we inlined because CHA revealed only a single target method,
2264       // then we are dependent on that target method not getting overridden
2265       // by dynamic class loading.  Be sure to test the "static" receiver
2266       // dest_method here, as opposed to the actual receiver, which may
2267       // falsely lead us to believe that the receiver is final or private.
2268       dependency_recorder()->assert_unique_concrete_method(actual_recv, cha_monomorphic_target);
2269     }
2270     code = Bytecodes::_invokespecial;
2271   }
2272 
2273   // check if we could do inlining
2274   if (!PatchALot && Inline && target->is_loaded() &&
2275       (klass->is_initialized() || (klass->is_interface() && target->holder()->is_initialized()))
2276       && !patch_for_appendix) {
2277     // callee is known => check if we have static binding
2278     if (code == Bytecodes::_invokestatic  ||
2279         code == Bytecodes::_invokespecial ||
2280         (code == Bytecodes::_invokevirtual && target->is_final_method()) ||
2281         code == Bytecodes::_invokedynamic) {
2282       ciMethod* inline_target = (cha_monomorphic_target != NULL) ? cha_monomorphic_target : target;
2283       // static binding => check if callee is ok
2284       bool success = try_inline(inline_target, (cha_monomorphic_target != NULL) || (exact_target != NULL), false, code, better_receiver);
2285 
2286       CHECK_BAILOUT();
2287       clear_inline_bailout();
2288 
2289       if (success) {
2290         // Register dependence if JVMTI has either breakpoint
2291         // setting or hotswapping of methods capabilities since they may
2292         // cause deoptimization.
2293         if (compilation()->env()->jvmti_can_hotswap_or_post_breakpoint()) {
2294           dependency_recorder()->assert_evol_method(inline_target);
2295         }
2296         return;
2297       }
2298     } else {
2299       print_inlining(target, "no static binding", /*success*/ false);
2300     }
2301   } else {
2302     print_inlining(target, "not inlineable", /*success*/ false);
2303   }
2304 
2305   // If we attempted an inline which did not succeed because of a
2306   // bailout during construction of the callee graph, the entire
2307   // compilation has to be aborted. This is fairly rare and currently
2308   // seems to only occur for jasm-generated classes which contain
2309   // jsr/ret pairs which are not associated with finally clauses and
2310   // do not have exception handlers in the containing method, and are
2311   // therefore not caught early enough to abort the inlining without
2312   // corrupting the graph. (We currently bail out with a non-empty
2313   // stack at a ret in these situations.)
2314   CHECK_BAILOUT();
2315 
2316   // inlining not successful => standard invoke
2317   ValueType* result_type = as_ValueType(declared_signature->return_type());
2318   ValueStack* state_before = copy_state_exhandling();
2319 
2320   // The bytecode (code) might change in this method so we are checking this very late.
2321   const bool has_receiver =
2322     code == Bytecodes::_invokespecial   ||
2323     code == Bytecodes::_invokevirtual   ||
2324     code == Bytecodes::_invokeinterface;
2325   Values* args = state()->pop_arguments(target->arg_size_no_receiver() + patching_appendix_arg);
2326   Value recv = has_receiver ? apop() : NULL;
2327   int vtable_index = Method::invalid_vtable_index;
2328 
2329 #ifdef SPARC
2330   // Currently only supported on Sparc.
2331   // The UseInlineCaches only controls dispatch to invokevirtuals for
2332   // loaded classes which we weren't able to statically bind.
2333   if (!UseInlineCaches && target->is_loaded() && code == Bytecodes::_invokevirtual
2334       && !target->can_be_statically_bound()) {
2335     // Find a vtable index if one is available
2336     // For arrays, callee_holder is Object. Resolving the call with
2337     // Object would allow an illegal call to finalize() on an
2338     // array. We use holder instead: illegal calls to finalize() won't
2339     // be compiled as vtable calls (IC call resolution will catch the
2340     // illegal call) and the few legal calls on array types won't be
2341     // either.
2342     vtable_index = target->resolve_vtable_index(calling_klass, holder);
2343   }
2344 #endif
2345 
2346   // A null check is required here (when there is a receiver) for any of the following cases
2347   // - invokespecial, always need a null check.
2348   // - invokevirtual, when the target is final and loaded. Calls to final targets will become optimized
2349   //   and require null checking. If the target is loaded a null check is emitted here.
2350   //   If the target isn't loaded the null check must happen after the call resolution. We achieve that
2351   //   by using the target methods unverified entry point (see CompiledIC::compute_monomorphic_entry).
2352   //   (The JVM specification requires that LinkageError must be thrown before a NPE. An unloaded target may
2353   //   potentially fail, and can't have the null check before the resolution.)
2354   // - A call that will be profiled. (But we can't add a null check when the target is unloaded, by the same
2355   //   reason as above, so calls with a receiver to unloaded targets can't be profiled.)
2356   //
2357   // Normal invokevirtual will perform the null check during lookup
2358 
2359   bool need_null_check = (code == Bytecodes::_invokespecial) ||
2360       (target->is_loaded() && (target->is_final_method() || (is_profiling() && profile_calls())));
2361 
2362   if (need_null_check) {
2363     if (recv != NULL) {
2364       null_check(recv);
2365     }
2366 
2367     if (is_profiling()) {
2368       // Note that we'd collect profile data in this method if we wanted it.
2369       compilation()->set_would_profile(true);
2370 
2371       if (profile_calls()) {
2372         assert(cha_monomorphic_target == NULL || exact_target == NULL, "both can not be set");
2373         ciKlass* target_klass = NULL;
2374         if (cha_monomorphic_target != NULL) {
2375           target_klass = cha_monomorphic_target->holder();
2376         } else if (exact_target != NULL) {
2377           target_klass = exact_target->holder();
2378         }
2379         profile_call(target, recv, target_klass, collect_args_for_profiling(args, NULL, false), false);
2380       }
2381     }
2382   }
2383 
2384   if (recv != NULL) {
2385     recv->set_escaped();
2386   }
2387   for (int i=0; i<args->length(); i++) {
2388     args->at(0)->set_escaped();
2389   }
2390 
2391   Invoke* result = new Invoke(code, result_type, recv, args, vtable_index, target, state_before,
2392                               declared_signature->returns_never_null());
2393   // push result
2394   append_split(result);
2395 
2396   if (result_type != voidType) {
2397     if (method()->is_strict()) {
2398       push(result_type, round_fp(result));
2399     } else {
2400       push(result_type, result);
2401     }
2402   }
2403   if (profile_return() && result_type->is_object_kind()) {
2404     profile_return_type(result, target);
2405   }
2406 }

2589 Instruction* GraphBuilder::append_with_bci(Instruction* instr, int bci) {
2590   Canonicalizer canon(compilation(), instr, bci);
2591   Instruction* i1 = canon.canonical();
2592   if (i1->is_linked() || !i1->can_be_linked()) {
2593     // Canonicalizer returned an instruction which was already
2594     // appended so simply return it.
2595     return i1;
2596   }
2597 
2598   if (UseLocalValueNumbering) {
2599     // Lookup the instruction in the ValueMap and add it to the map if
2600     // it's not found.
2601     Instruction* i2 = vmap()->find_insert(i1);
2602     if (i2 != i1) {
2603       // found an entry in the value map, so just return it.
2604       assert(i2->is_linked(), "should already be linked");
2605       return i2;
2606     }
2607     ValueNumberingEffects vne(vmap());
2608     i1->visit(&vne);
2609   }
2610 
2611   // i1 was not eliminated => append it
2612   assert(i1->next() == NULL, "shouldn't already be linked");
2613   _last = _last->set_next(i1, canon.bci());
2614 
2615   if (++_instruction_count >= InstructionCountCutoff && !bailed_out()) {
2616     // set the bailout state but complete normal processing.  We
2617     // might do a little more work before noticing the bailout so we
2618     // want processing to continue normally until it's noticed.
2619     bailout("Method and/or inlining is too large");
2620   }
2621 
2622 #ifndef PRODUCT
2623   if (PrintIRDuringConstruction) {
2624     InstructionPrinter ip;
2625     ip.print_line(i1);
2626     if (Verbose) {
2627       state()->print();
2628     }
2629   }
2630 #endif
2631 
2632   // save state after modification of operand stack for StateSplit instructions
2633   StateSplit* s = i1->as_StateSplit();
2634   if (s != NULL) {
2635     if (EliminateFieldAccess) {
2636       Intrinsic* intrinsic = s->as_Intrinsic();
2637       if (s->as_Invoke() != NULL || (intrinsic && !intrinsic->preserves_state())) {
2638         _memory->kill();
2639       }
2640     }
2641     s->set_state(state()->copy(ValueStack::StateAfter, canon.bci()));
2642   }
2643 
2644   // set up exception handlers for this instruction if necessary
2645   if (i1->can_trap()) {
2646     i1->set_exception_handlers(handle_exception(i1));
2647     assert(i1->exception_state() != NULL || !i1->needs_exception_state() || bailed_out(), "handle_exception must set exception state");
2648   }
2649   return i1;
2650 }

2684 XHandlers* GraphBuilder::handle_exception(Instruction* instruction) {
2685   if (!has_handler() && (!instruction->needs_exception_state() || instruction->exception_state() != NULL)) {
2686     assert(instruction->exception_state() == NULL
2687            || instruction->exception_state()->kind() == ValueStack::EmptyExceptionState
2688            || (instruction->exception_state()->kind() == ValueStack::ExceptionState && _compilation->env()->should_retain_local_variables()),
2689            "exception_state should be of exception kind");
2690     return new XHandlers();
2691   }
2692 
2693   XHandlers*  exception_handlers = new XHandlers();
2694   ScopeData*  cur_scope_data = scope_data();
2695   ValueStack* cur_state = instruction->state_before();
2696   ValueStack* prev_state = NULL;
2697   int scope_count = 0;
2698 
2699   assert(cur_state != NULL, "state_before must be set");
2700   do {
2701     int cur_bci = cur_state->bci();
2702     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2703     assert(cur_bci == SynchronizationEntryBCI || cur_bci == cur_scope_data->stream()->cur_bci(), "invalid bci");
2704 
2705     // join with all potential exception handlers
2706     XHandlers* list = cur_scope_data->xhandlers();
2707     const int n = list->length();
2708     for (int i = 0; i < n; i++) {
2709       XHandler* h = list->handler_at(i);
2710       if (h->covers(cur_bci)) {
2711         // h is a potential exception handler => join it
2712         compilation()->set_has_exception_handlers(true);
2713 
2714         BlockBegin* entry = h->entry_block();
2715         if (entry == block()) {
2716           // It's acceptable for an exception handler to cover itself
2717           // but we don't handle that in the parser currently.  It's
2718           // very rare so we bailout instead of trying to handle it.
2719           BAILOUT_("exception handler covers itself", exception_handlers);
2720         }
2721         assert(entry->bci() == h->handler_bci(), "must match");
2722         assert(entry->bci() == -1 || entry == cur_scope_data->block_at(entry->bci()), "blocks must correspond");
2723 
2724         // previously this was a BAILOUT, but this is not necessary
2725         // now because asynchronous exceptions are not handled this way.
2726         assert(entry->state() == NULL || cur_state->total_locks_size() == entry->state()->total_locks_size(), "locks do not match");
2727 
2728         // xhandler start with an empty expression stack
2729         if (cur_state->stack_size() != 0) {
2730           cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2731         }
2732         if (instruction->exception_state() == NULL) {
2733           instruction->set_exception_state(cur_state);
2734         }
2735 
2736         // Note: Usually this join must work. However, very
2737         // complicated jsr-ret structures where we don't ret from
2738         // the subroutine can cause the objects on the monitor
2739         // stacks to not match because blocks can be parsed twice.
2740         // The only test case we've seen so far which exhibits this
2741         // problem is caught by the infinite recursion test in
2742         // GraphBuilder::jsr() if the join doesn't work.
2743         if (!entry->try_merge(cur_state)) {
2744           BAILOUT_("error while joining with exception handler, prob. due to complicated jsr/rets", exception_handlers);
2745         }
2746 
2747         // add current state for correct handling of phi functions at begin of xhandler
2748         int phi_operand = entry->add_exception_state(cur_state);
2749 
2750         // add entry to the list of xhandlers of this block
2751         _block->add_exception_handler(entry);
2752 
2753         // add back-edge from xhandler entry to this block
2754         if (!entry->is_predecessor(_block)) {
2755           entry->add_predecessor(_block);
2756         }
2757 
2758         // clone XHandler because phi_operand and scope_count can not be shared
2759         XHandler* new_xhandler = new XHandler(h);
2760         new_xhandler->set_phi_operand(phi_operand);
2761         new_xhandler->set_scope_count(scope_count);
2762         exception_handlers->append(new_xhandler);
2763 
2764         // fill in exception handler subgraph lazily
2765         assert(!entry->is_set(BlockBegin::was_visited_flag), "entry must not be visited yet");
2766         cur_scope_data->add_to_work_list(entry);
2767 
2768         // stop when reaching catchall
2769         if (h->catch_type() == 0) {
2770           return exception_handlers;
2771         }
2772       }
2773     }
2774 
2775     if (exception_handlers->length() == 0) {
2776       // This scope and all callees do not handle exceptions, so the local
2777       // variables of this scope are not needed. However, the scope itself is
2778       // required for a correct exception stack trace -> clear out the locals.
2779       if (_compilation->env()->should_retain_local_variables()) {
2780         cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2781       } else {
2782         cur_state = cur_state->copy(ValueStack::EmptyExceptionState, cur_state->bci());
2783       }
2784       if (prev_state != NULL) {
2785         prev_state->set_caller_state(cur_state);
2786       }
2787       if (instruction->exception_state() == NULL) {
2788         instruction->set_exception_state(cur_state);
2789       }
2790     }
2791 
2792     // Set up iteration for next time.
2793     // If parsing a jsr, do not grab exception handlers from the
2794     // parent scopes for this method (already got them, and they
2795     // needed to be cloned)
2796 
2797     while (cur_scope_data->parsing_jsr()) {
2798       cur_scope_data = cur_scope_data->parent();
2799     }
2800 
2801     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2802     assert(cur_state->locks_size() == 0 || cur_state->locks_size() == 1, "unlocking must be done in a catchall exception handler");
2803 
2804     prev_state = cur_state;
2805     cur_state = cur_state->caller_state();
2806     cur_scope_data = cur_scope_data->parent();
2807     scope_count++;
2808   } while (cur_scope_data != NULL);
2809 
2810   return exception_handlers;
2811 }

2837     return v;

2839     // already substituted; subst can be phi itself -> simplify
2840     return simplify(v->subst());
2841   } else if (phi->is_set(Phi::cannot_simplify)) {
2842     // already tried to simplify phi before
2843     return phi;
2844   } else if (phi->is_set(Phi::visited)) {
2845     // break cycles in phi functions
2846     return phi;
2847   } else if (phi->type()->is_illegal()) {
2848     // illegal phi functions are ignored anyway
2849     return phi;
2850 
2851   } else {
2852     // mark phi function as processed to break cycles in phi functions
2853     phi->set(Phi::visited);
2854 
2855     // simplify x = [y, x] and x = [y, y] to y
2856     Value subst = NULL;
2857     int opd_count = phi->operand_count();
2858     for (int i = 0; i < opd_count; i++) {
2859       Value opd = phi->operand_at(i);
2860       assert(opd != NULL, "Operand must exist!");
2861 
2862       if (opd->type()->is_illegal()) {
2863         // if one operand is illegal, the entire phi function is illegal
2864         phi->make_illegal();
2865         phi->clear(Phi::visited);
2866         return phi;
2867       }
2868 
2869       Value new_opd = simplify(opd);
2870       assert(new_opd != NULL, "Simplified operand must exist!");
2871 
2872       if (new_opd != phi && new_opd != subst) {
2873         if (subst == NULL) {
2874           subst = new_opd;
2875         } else {
2876           // no simplification possible
2877           phi->set(Phi::cannot_simplify);
2878           phi->clear(Phi::visited);
2879           return phi;
2880         }
2881       }
2882     }
2883 
2884     // sucessfully simplified phi function
2885     assert(subst != NULL, "illegal phi function");
2886     _has_substitutions = true;
2887     phi->clear(Phi::visited);
2888     phi->set_subst(subst);
2889 
2890 #ifndef PRODUCT
2891     if (PrintPhiFunctions) {
2892       tty->print_cr("simplified phi function %c%d to %c%d (Block B%d)", phi->type()->tchar(), phi->id(), subst->type()->tchar(), subst->id(), phi->block()->block_id());
2893     }
2894 #endif
2895 
2896     return subst;
2897   }
2898 }

3233   }
3234 
3235   scope_data()->set_stream(NULL);
3236 
3237   // done
3238   return end;
3239 }

3242 void GraphBuilder::iterate_all_blocks(bool start_in_current_block_for_inlining) {
3243   do {
3244     if (start_in_current_block_for_inlining && !bailed_out()) {
3245       iterate_bytecodes_for_block(0);
3246       start_in_current_block_for_inlining = false;
3247     } else {
3248       BlockBegin* b;
3249       while ((b = scope_data()->remove_from_work_list()) != NULL) {
3250         if (!b->is_set(BlockBegin::was_visited_flag)) {
3251           if (b->is_set(BlockBegin::osr_entry_flag)) {
3252             // we're about to parse the osr entry block, so make sure
3253             // we setup the OSR edge leading into this block so that
3254             // Phis get setup correctly.
3255             setup_osr_entry_block();
3256             // this is no longer the osr entry block, so clear it.
3257             b->clear(BlockBegin::osr_entry_flag);
3258           }
3259           b->set(BlockBegin::was_visited_flag);
3260           connect_to_end(b);
3261         }
3262       }
3263     }
3264   } while (!bailed_out() && !scope_data()->is_work_list_empty());
3265 }

3391 void GraphBuilder::setup_osr_entry_block() {
3392   assert(compilation()->is_osr_compile(), "only for osrs");
3393 
3394   int osr_bci = compilation()->osr_bci();
3395   ciBytecodeStream s(method());
3396   s.reset_to_bci(osr_bci);
3397   s.next();
3398   scope_data()->set_stream(&s);
3399 
3400   // create a new block to be the osr setup code
3401   _osr_entry = new BlockBegin(osr_bci);
3402   _osr_entry->set(BlockBegin::osr_entry_flag);
3403   _osr_entry->set_depth_first_number(0);
3404   BlockBegin* target = bci2block()->at(osr_bci);
3405   assert(target != NULL && target->is_set(BlockBegin::osr_entry_flag), "must be there");
3406   // the osr entry has no values for locals
3407   ValueStack* state = target->state()->copy();
3408   _osr_entry->set_state(state);
3409 
3410   kill_all();
3411   _block = _osr_entry;
3412   _state = _osr_entry->state()->copy();
3413   assert(_state->bci() == osr_bci, "mismatch");
3414   _last  = _osr_entry;
3415   Value e = append(new OsrEntry());
3416   e->set_needs_null_check(false);
3417 
3418   // OSR buffer is
3419   //
3420   // locals[nlocals-1..0]
3421   // monitors[number_of_locks-1..0]
3422   //
3423   // locals is a direct copy of the interpreter frame so in the osr buffer
3424   // so first slot in the local array is the last local from the interpreter
3425   // and last slot is local[0] (receiver) from the interpreter
3426   //
3427   // Similarly with locks. The first lock slot in the osr buffer is the nth lock
3428   // from the interpreter frame, the nth lock slot in the osr buffer is 0th lock
3429   // in the interpreter frame (the method lock if a sync method)
3430 
3431   // Initialize monitors in the compiled activation.
3432 
3433   int index;
3434   Value local;
3435 
3436   // find all the locals that the interpreter thinks contain live oops
3437   const ResourceBitMap live_oops = method()->live_local_oops_at_bci(osr_bci);
3438 
3439   // compute the offset into the locals so that we can treat the buffer
3440   // as if the locals were still in the interpreter frame
3441   int locals_offset = BytesPerWord * (method()->max_locals() - 1);
3442   for_each_local_value(state, index, local) {
3443     int offset = locals_offset - (index + local->type()->size() - 1) * BytesPerWord;
3444     Value get;
3445     if (local->type()->is_object_kind() && !live_oops.at(index)) {
3446       // The interpreter thinks this local is dead but the compiler
3447       // doesn't so pretend that the interpreter passed in null.
3448       get = append(new Constant(objectNull));
3449     } else {
3450       get = append(new UnsafeGetRaw(as_BasicType(local->type()), e,
3451                                     append(new Constant(new IntConstant(offset))),
3452                                     0,
3453                                     true /*unaligned*/, true /*wide*/));
3454     }
3455     _state->store_local(index, get);
3456   }
3457 
3458   // the storage for the OSR buffer is freed manually in the LIRGenerator.
3459 
3460   assert(state->caller_state() == NULL, "should be top scope");
3461   state->clear_locals();
3462   Goto* g = new Goto(target, false);
3463   append(g);
3464   _osr_entry->set_end(g);
3465   target->merge(_osr_entry->end()->state());
3466 
3467   scope_data()->set_stream(NULL);
3468 }

3471 ValueStack* GraphBuilder::state_at_entry() {
3472   ValueStack* state = new ValueStack(scope(), NULL);
3473 
3474   // Set up locals for receiver
3475   int idx = 0;
3476   if (!method()->is_static()) {
3477     // we should always see the receiver
3478     state->store_local(idx, new Local(method()->holder(), objectType, idx,
3479              /*receiver*/ true, /*never_null*/ method()->holder()->is_value_array_klass()));
3480     idx = 1;
3481   }
3482 
3483   // Set up locals for incoming arguments
3484   ciSignature* sig = method()->signature();
3485   for (int i = 0; i < sig->count(); i++) {
3486     ciType* type = sig->type_at(i);
3487     BasicType basic_type = type->basic_type();
3488     // don't allow T_ARRAY to propagate into locals types
3489     if (is_reference_type(basic_type)) basic_type = T_OBJECT;
3490     ValueType* vt = as_ValueType(basic_type);
3491     state->store_local(idx, new Local(type, vt, idx, false, sig->is_never_null_at(i)));
3492     idx += type->size();
3493   }
3494 
3495   // lock synchronized method
3496   if (method()->is_synchronized()) {
3497     state->lock(NULL);
3498   }
3499 
3500   return state;
3501 }

3504 GraphBuilder::GraphBuilder(Compilation* compilation, IRScope* scope)
3505   : _scope_data(NULL)
3506   , _compilation(compilation)
3507   , _memory(new MemoryBuffer())
3508   , _inline_bailout_msg(NULL)
3509   , _instruction_count(0)
3510   , _osr_entry(NULL)
3511 {
3512   int osr_bci = compilation->osr_bci();
3513 
3514   // determine entry points and bci2block mapping
3515   BlockListBuilder blm(compilation, scope, osr_bci);
3516   CHECK_BAILOUT();
3517 
3518   BlockList* bci2block = blm.bci2block();
3519   BlockBegin* start_block = bci2block->at(0);
3520 
3521   push_root_scope(scope, bci2block, start_block);
3522 
3523   // setup state for std entry
3524   _initial_state = state_at_entry();
3525   start_block->merge(_initial_state);
3526 
3527   // complete graph
3528   _vmap        = new ValueMap();
3529   switch (scope->method()->intrinsic_id()) {
3530   case vmIntrinsics::_dabs          : // fall through
3531   case vmIntrinsics::_dsqrt         : // fall through
3532   case vmIntrinsics::_dsin          : // fall through
3533   case vmIntrinsics::_dcos          : // fall through
3534   case vmIntrinsics::_dtan          : // fall through
3535   case vmIntrinsics::_dlog          : // fall through
3536   case vmIntrinsics::_dlog10        : // fall through
3537   case vmIntrinsics::_dexp          : // fall through
3538   case vmIntrinsics::_dpow          : // fall through
3539     {
3540       // Compiles where the root method is an intrinsic need a special
3541       // compilation environment because the bytecodes for the method
3542       // shouldn't be parsed during the compilation, only the special
3543       // Intrinsic node should be emitted.  If this isn't done the the
3544       // code for the inlined version will be different than the root
3545       // compiled version which could lead to monotonicity problems on
3546       // intel.
3547       if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3548         BAILOUT("failed to inline intrinsic, method not annotated");
3549       }
3550 
3551       // Set up a stream so that appending instructions works properly.
3552       ciBytecodeStream s(scope->method());
3553       s.reset_to_bci(0);
3554       scope_data()->set_stream(&s);
3555       s.next();
3556 
3557       // setup the initial block state
3558       _block = start_block;
3559       _state = start_block->state()->copy_for_parsing();
3560       _last  = start_block;
3561       load_local(doubleType, 0);
3562       if (scope->method()->intrinsic_id() == vmIntrinsics::_dpow) {
3563         load_local(doubleType, 2);
3564       }
3565 
3566       // Emit the intrinsic node.
3567       bool result = try_inline_intrinsics(scope->method());
3568       if (!result) BAILOUT("failed to inline intrinsic");
3569       method_return(dpop());
3570 
3571       // connect the begin and end blocks and we're all done.
3572       BlockEnd* end = last()->as_BlockEnd();
3573       block()->set_end(end);
3574       break;
3575     }
3576 
3577   case vmIntrinsics::_Reference_get:
3578     {
3579       {
3580         // With java.lang.ref.reference.get() we must go through the
3581         // intrinsic - when G1 is enabled - even when get() is the root
3582         // method of the compile so that, if necessary, the value in
3583         // the referent field of the reference object gets recorded by
3584         // the pre-barrier code.
3585         // Specifically, if G1 is enabled, the value in the referent
3586         // field is recorded by the G1 SATB pre barrier. This will
3587         // result in the referent being marked live and the reference
3588         // object removed from the list of discovered references during
3589         // reference processing.
3590         if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3591           BAILOUT("failed to inline intrinsic, method not annotated");
3592         }
3593 
3594         // Also we need intrinsic to prevent commoning reads from this field
3595         // across safepoint since GC can change its value.
3596 
3597         // Set up a stream so that appending instructions works properly.
3598         ciBytecodeStream s(scope->method());
3599         s.reset_to_bci(0);
3600         scope_data()->set_stream(&s);
3601         s.next();
3602 
3603         // setup the initial block state
3604         _block = start_block;
3605         _state = start_block->state()->copy_for_parsing();
3606         _last  = start_block;
3607         load_local(objectType, 0);
3608 
3609         // Emit the intrinsic node.
3610         bool result = try_inline_intrinsics(scope->method());
3611         if (!result) BAILOUT("failed to inline intrinsic");
3612         method_return(apop());
3613 
3614         // connect the begin and end blocks and we're all done.
3615         BlockEnd* end = last()->as_BlockEnd();
3616         block()->set_end(end);
3617         break;
3618       }
3619       // Otherwise, fall thru
3620     }
3621 
3622   default:
3623     scope_data()->add_to_work_list(start_block);
3624     iterate_all_blocks();
3625     break;
3626   }
3627   CHECK_BAILOUT();
3628 
3629   _start = setup_start_block(osr_bci, start_block, _osr_entry, _initial_state);
3630 
3631   eliminate_redundant_phis(_start);
3632 
3633   NOT_PRODUCT(if (PrintValueNumbering && Verbose) print_stats());
3634   // for osr compile, bailout if some requirements are not fulfilled
3635   if (osr_bci != -1) {
3636     BlockBegin* osr_block = blm.bci2block()->at(osr_bci);
3637     if (!osr_block->is_set(BlockBegin::was_visited_flag)) {
3638       BAILOUT("osr entry must have been visited for osr compile");
3639     }
3640 
3641     // check if osr entry point has empty stack - we cannot handle non-empty stacks at osr entry points
3642     if (!osr_block->state()->stack_is_empty()) {
3643       BAILOUT("stack not empty at OSR entry point");
3644     }
3645   }
3646 #ifndef PRODUCT
3647   if (PrintCompilation && Verbose) tty->print_cr("Created %d Instructions", _instruction_count);
3648 #endif
3649 }

3696 bool GraphBuilder::try_inline(ciMethod* callee, bool holder_known, bool ignore_return, Bytecodes::Code bc, Value receiver) {
3697   const char* msg = NULL;
3698 
3699   // clear out any existing inline bailout condition
3700   clear_inline_bailout();
3701 
3702   // exclude methods we don't want to inline
3703   msg = should_not_inline(callee);
3704   if (msg != NULL) {
3705     print_inlining(callee, msg, /*success*/ false);
3706     return false;
3707   }
3708 
3709   // method handle invokes
3710   if (callee->is_method_handle_intrinsic()) {
3711     if (try_method_handle_inline(callee, ignore_return)) {
3712       if (callee->has_reserved_stack_access()) {
3713         compilation()->set_has_reserved_stack_access(true);
3714       }
3715       return true;
3716     }
3717     return false;
3718   }
3719 
3720   // handle intrinsics
3721   if (callee->intrinsic_id() != vmIntrinsics::_none &&
3722       (CheckIntrinsics ? callee->intrinsic_candidate() : true)) {
3723     if (try_inline_intrinsics(callee, ignore_return)) {
3724       print_inlining(callee, "intrinsic");
3725       if (callee->has_reserved_stack_access()) {
3726         compilation()->set_has_reserved_stack_access(true);
3727       }
3728       return true;
3729     }
3730     // try normal inlining
3731   }
3732 
3733   // certain methods cannot be parsed at all
3734   msg = check_can_parse(callee);
3735   if (msg != NULL) {
3736     print_inlining(callee, msg, /*success*/ false);
3737     return false;
3738   }
3739 
3740   // If bytecode not set use the current one.
3741   if (bc == Bytecodes::_illegal) {
3742     bc = code();
3743   }
3744   if (try_inline_full(callee, holder_known, ignore_return, bc, receiver)) {
3745     if (callee->has_reserved_stack_access()) {
3746       compilation()->set_has_reserved_stack_access(true);
3747     }
3748     return true;
3749   }
3750 
3751   // Entire compilation could fail during try_inline_full call.
3752   // In that case printing inlining decision info is useless.
3753   if (!bailed_out())
3754     print_inlining(callee, _inline_bailout_msg, /*success*/ false);
3755 
3756   return false;
3757 }

3904 bool GraphBuilder::try_inline_jsr(int jsr_dest_bci) {
3905   // Introduce a new callee continuation point - all Ret instructions
3906   // will be replaced with Gotos to this point.
3907   BlockBegin* cont = block_at(next_bci());
3908   assert(cont != NULL, "continuation must exist (BlockListBuilder starts a new block after a jsr");
3909 
3910   // Note: can not assign state to continuation yet, as we have to
3911   // pick up the state from the Ret instructions.
3912 
3913   // Push callee scope
3914   push_scope_for_jsr(cont, jsr_dest_bci);
3915 
3916   // Temporarily set up bytecode stream so we can append instructions
3917   // (only using the bci of this stream)
3918   scope_data()->set_stream(scope_data()->parent()->stream());
3919 
3920   BlockBegin* jsr_start_block = block_at(jsr_dest_bci);
3921   assert(jsr_start_block != NULL, "jsr start block must exist");
3922   assert(!jsr_start_block->is_set(BlockBegin::was_visited_flag), "should not have visited jsr yet");
3923   Goto* goto_sub = new Goto(jsr_start_block, false);
3924   // Must copy state to avoid wrong sharing when parsing bytecodes
3925   assert(jsr_start_block->state() == NULL, "should have fresh jsr starting block");
3926   jsr_start_block->set_state(copy_state_before_with_bci(jsr_dest_bci));
3927   append(goto_sub);
3928   _block->set_end(goto_sub);
3929   _last = _block = jsr_start_block;
3930 
3931   // Clear out bytecode stream
3932   scope_data()->set_stream(NULL);
3933 
3934   scope_data()->add_to_work_list(jsr_start_block);
3935 
3936   // Ready to resume parsing in subroutine
3937   iterate_all_blocks();
3938 
3939   // If we bailed out during parsing, return immediately (this is bad news)
3940   CHECK_BAILOUT_(false);
3941 
3942   // Detect whether the continuation can actually be reached. If not,
3943   // it has not had state set by the join() operations in
3944   // iterate_bytecodes_for_block()/ret() and we should not touch the
3945   // iteration state. The calling activation of
3946   // iterate_bytecodes_for_block will then complete normally.
3947   if (cont->state() != NULL) {
3948     if (!cont->is_set(BlockBegin::was_visited_flag)) {
3949       // add continuation to work list instead of parsing it immediately
3950       scope_data()->parent()->add_to_work_list(cont);
3951     }
3952   }
3953 
3954   assert(jsr_continuation() == cont, "continuation must not have changed");
3955   assert(!jsr_continuation()->is_set(BlockBegin::was_visited_flag) ||
3956          jsr_continuation()->is_set(BlockBegin::parser_loop_header_flag),
3957          "continuation can only be visited in case of backward branches");
3958   assert(_last && _last->as_BlockEnd(), "block must have end");
3959 
3960   // continuation is in work list, so end iteration of current block
3961   _skip_block = true;
3962   pop_scope_for_jsr();
3963 
3964   return true;
3965 }

4056   if (CompilationPolicy::policy()->should_not_inline(compilation()->env(), callee)) {
4057     INLINE_BAILOUT("inlining prohibited by policy");
4058   }
4059   // first perform tests of things it's not possible to inline
4060   if (callee->has_exception_handlers() &&
4061       !InlineMethodsWithExceptionHandlers) INLINE_BAILOUT("callee has exception handlers");
4062   if (callee->is_synchronized() &&
4063       !InlineSynchronizedMethods         ) INLINE_BAILOUT("callee is synchronized");
4064   if (!callee->holder()->is_initialized()) INLINE_BAILOUT("callee's klass not initialized yet");
4065   if (!callee->has_balanced_monitors())    INLINE_BAILOUT("callee's monitors do not match");
4066 
4067   // Proper inlining of methods with jsrs requires a little more work.
4068   if (callee->has_jsrs()                 ) INLINE_BAILOUT("jsrs not handled properly by inliner yet");
4069 
4070   if (strict_fp_requires_explicit_rounding &&
4071       method()->is_strict() != callee->is_strict()) {
4072 #ifdef IA32
4073     // If explicit rounding is required, do not inline strict code into non-strict code (or the reverse).
4074     // When SSE2 is present, no special handling is needed.
4075     if (UseSSE < 2) {
4076       INLINE_BAILOUT("caller and callee have different strict fp requirements");
4077     }
4078 #else
4079     Unimplemented();
4080 #endif // IA32
4081   }
4082 
4083   if (is_profiling() && !callee->ensure_method_data()) {
4084     INLINE_BAILOUT("mdo allocation failed");
4085   }
4086 
4087   const bool is_invokedynamic = (bc == Bytecodes::_invokedynamic);
4088   const bool has_receiver = (bc != Bytecodes::_invokestatic && !is_invokedynamic);
4089 
4090   const int args_base = state()->stack_size() - callee->arg_size();
4091   assert(args_base >= 0, "stack underflow during inlining");
4092 
4093   Value recv = NULL;
4094   if (has_receiver) {
4095     assert(!callee->is_static(), "callee must not be static");
4096     assert(callee->arg_size() > 0, "must have at least a receiver");
4097 
4098     recv = state()->stack_at(args_base);
4099     if (recv->is_null_obj()) {
4100       INLINE_BAILOUT("receiver is always null");
4101     }
4102   }
4103 
4104   // now perform tests that are based on flag settings
4105   bool inlinee_by_directive = compilation()->directive()->should_inline(callee);
4106   if (callee->force_inline() || inlinee_by_directive) {
4107     if (inline_level() > MaxForceInlineLevel                      ) INLINE_BAILOUT("MaxForceInlineLevel");
4108     if (recursive_inline_level(callee) > C1MaxRecursiveInlineLevel) INLINE_BAILOUT("recursive inlining too deep");
4109

4112     if (inlinee_by_directive)    msg = "force inline by CompileCommand";

4114   } else {
4115     // use heuristic controls on inlining
4116     if (inline_level() > C1MaxInlineLevel                       ) INLINE_BAILOUT("inlining too deep");
4117     int callee_recursive_level = recursive_inline_level(callee);
4118     if (callee_recursive_level > C1MaxRecursiveInlineLevel      ) INLINE_BAILOUT("recursive inlining too deep");
4119     if (callee->code_size_for_inlining() > max_inline_size()    ) INLINE_BAILOUT("callee is too large");
4120     // Additional condition to limit stack usage for non-recursive calls.
4121     if ((callee_recursive_level == 0) &&
4122         (callee->max_stack() + callee->max_locals() - callee->size_of_parameters() > C1InlineStackLimit)) {
4123       INLINE_BAILOUT("callee uses too much stack");
4124     }

4289   assert(continuation_existed ||
4290          !continuation()->is_set(BlockBegin::was_visited_flag),
4291          "continuation should not have been parsed yet if we created it");
4292 
4293   // At this point we are almost ready to return and resume parsing of
4294   // the caller back in the GraphBuilder. The only thing we want to do
4295   // first is an optimization: during parsing of the callee we
4296   // generated at least one Goto to the continuation block. If we
4297   // generated exactly one, and if the inlined method spanned exactly
4298   // one block (and we didn't have to Goto its entry), then we snip
4299   // off the Goto to the continuation, allowing control to fall
4300   // through back into the caller block and effectively performing
4301   // block merging. This allows load elimination and CSE to take place
4302   // across multiple callee scopes if they are relatively simple, and
4303   // is currently essential to making inlining profitable.
4304   if (num_returns() == 1
4305       && block() == orig_block
4306       && block() == inline_cleanup_block()) {
4307     _last  = inline_cleanup_return_prev();
4308     _state = inline_cleanup_state();
4309   } else if (continuation_preds == cont->number_of_preds()) {
4310     // Inlining caused that the instructions after the invoke in the
4311     // caller are not reachable any more. So skip filling this block
4312     // with instructions!
4313     assert(cont == continuation(), "");
4314     assert(_last && _last->as_BlockEnd(), "");
4315     _skip_block = true;
4316   } else {
4317     // Resume parsing in continuation block unless it was already parsed.
4318     // Note that if we don't change _last here, iteration in
4319     // iterate_bytecodes_for_block will stop when we return.
4320     if (!continuation()->is_set(BlockBegin::was_visited_flag)) {
4321       // add continuation to work list instead of parsing it immediately
4322       assert(_last && _last->as_BlockEnd(), "");
4323       scope_data()->parent()->add_to_work_list(continuation());
4324       _skip_block = true;
4325     }
4326   }
4327 
4328   // Fill the exception handler for synchronized methods with instructions
4329   if (callee->is_synchronized() && sync_handler->state() != NULL) {
4330     fill_sync_handler(lock, sync_handler);
4331   } else {
4332     pop_scope();
4333   }
4334 
4335   compilation()->notice_inlined_method(callee);
4336 
4337   return true;
4338 }

4341 bool GraphBuilder::try_method_handle_inline(ciMethod* callee, bool ignore_return) {
4342   ValueStack* state_before = copy_state_before();
4343   vmIntrinsics::ID iid = callee->intrinsic_id();
4344   switch (iid) {
4345   case vmIntrinsics::_invokeBasic:
4346     {
4347       // get MethodHandle receiver
4348       const int args_base = state()->stack_size() - callee->arg_size();
4349       ValueType* type = state()->stack_at(args_base)->type();
4350       if (type->is_constant()) {
4351         ciMethod* target = type->as_ObjectType()->constant_value()->as_method_handle()->get_vmtarget();
4352         // We don't do CHA here so only inline static and statically bindable methods.
4353         if (target->is_static() || target->can_be_statically_bound()) {
4354           if (ciMethod::is_consistent_info(callee, target)) {
4355             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4356             ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4357             if (try_inline(target, /*holder_known*/ true, ignore_return, bc)) {
4358               return true;
4359             }
4360           } else {
4361             print_inlining(target, "signatures mismatch", /*success*/ false);
4362           }
4363         } else {
4364           print_inlining(target, "not static or statically bindable", /*success*/ false);
4365         }
4366       } else {
4367         print_inlining(callee, "receiver not constant", /*success*/ false);
4368       }
4369     }
4370     break;
4371 
4372   case vmIntrinsics::_linkToVirtual:
4373   case vmIntrinsics::_linkToStatic:
4374   case vmIntrinsics::_linkToSpecial:
4375   case vmIntrinsics::_linkToInterface:
4376     {
4377       // pop MemberName argument
4378       const int args_base = state()->stack_size() - callee->arg_size();
4379       ValueType* type = apop()->type();
4380       if (type->is_constant()) {
4381         ciMethod* target = type->as_ObjectType()->constant_value()->as_member_name()->get_vmtarget();
4382         ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4383         // If the target is another method handle invoke, try to recursively get
4384         // a better target.
4385         if (target->is_method_handle_intrinsic()) {
4386           if (try_method_handle_inline(target, ignore_return)) {
4387             return true;
4388           }
4389         } else if (!ciMethod::is_consistent_info(callee, target)) {
4390           print_inlining(target, "signatures mismatch", /*success*/ false);
4391         } else {
4392           ciSignature* signature = target->signature();
4393           const int receiver_skip = target->is_static() ? 0 : 1;
4394           // Cast receiver to its type.
4395           if (!target->is_static()) {
4396             ciKlass* tk = signature->accessing_klass();
4397             Value obj = state()->stack_at(args_base);
4398             if (obj->exact_type() == NULL &&
4399                 obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4400               TypeCast* c = new TypeCast(tk, obj, state_before);
4401               append(c);
4402               state()->stack_at_put(args_base, c);
4403             }
4404           }
4405           // Cast reference arguments to its type.
4406           for (int i = 0, j = 0; i < signature->count(); i++) {
4407             ciType* t = signature->type_at(i);
4408             if (t->is_klass()) {
4409               ciKlass* tk = t->as_klass();
4410               Value obj = state()->stack_at(args_base + receiver_skip + j);
4411               if (obj->exact_type() == NULL &&
4412                   obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4413                 TypeCast* c = new TypeCast(t, obj, state_before);
4414                 append(c);
4415                 state()->stack_at_put(args_base + receiver_skip + j, c);
4416               }
4417             }
4418             j += t->size();  // long and double take two slots
4419           }
4420           // We don't do CHA here so only inline static and statically bindable methods.
4421           if (target->is_static() || target->can_be_statically_bound()) {
4422             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4423             if (try_inline(target, /*holder_known*/ true, ignore_return, bc)) {
4424               return true;
4425             }
4426           } else {
4427             print_inlining(target, "not static or statically bindable", /*success*/ false);
4428           }
4429         }
4430       } else {
4431         print_inlining(callee, "MemberName not constant", /*success*/ false);
4432       }
4433     }
4434     break;
4435 
4436   default:
4437     fatal("unexpected intrinsic %d: %s", iid, vmIntrinsics::name_at(iid));
4438     break;
4439   }
4440   set_state(state_before->copy_for_parsing());
4441   return false;
4442 }
