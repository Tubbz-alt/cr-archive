 132 void BlockListBuilder::set_entries(int osr_bci) {
 133   // generate start blocks
 134   BlockBegin* std_entry = make_block_at(0, NULL);
 135   if (scope()->caller() == NULL) {
 136     std_entry->set(BlockBegin::std_entry_flag);
 137   }
 138   if (osr_bci != -1) {
 139     BlockBegin* osr_entry = make_block_at(osr_bci, NULL);
 140     osr_entry->set(BlockBegin::osr_entry_flag);
 141   }
 142 
 143   // generate exception entry blocks
 144   XHandlers* list = xhandlers();
 145   const int n = list->length();
 146   for (int i = 0; i < n; i++) {
 147     XHandler* h = list->handler_at(i);
 148     BlockBegin* entry = make_block_at(h->handler_bci(), NULL);
 149     entry->set(BlockBegin::exception_entry_flag);
 150     h->set_entry_block(entry);
 151   }
 152 }

 155 BlockBegin* BlockListBuilder::make_block_at(int cur_bci, BlockBegin* predecessor) {
 156   assert(method()->bci_block_start().at(cur_bci), "wrong block starts of MethodLivenessAnalyzer");
 157 
 158   BlockBegin* block = _bci2block->at(cur_bci);
 159   if (block == NULL) {
 160     block = new BlockBegin(cur_bci);
 161     block->init_stores_to_locals(method()->max_locals());
 162     _bci2block->at_put(cur_bci, block);
 163     _blocks.append(block);
 164 
 165     assert(predecessor == NULL || predecessor->bci() < cur_bci, "targets for backward branches must already exist");
 166   }
 167 
 168   if (predecessor != NULL) {
 169     if (block->is_set(BlockBegin::exception_entry_flag)) {
 170       BAILOUT_("Exception handler can be reached by both normal and exceptional control flow", block);
 171     }
 172 
 173     predecessor->add_successor(block);
 174     block->increment_total_preds();
 175   }
 176 
 177   return block;
 178 }

 181 inline void BlockListBuilder::store_one(BlockBegin* current, int local) {
 182   current->stores_to_locals().set_bit(local);

 184 inline void BlockListBuilder::store_two(BlockBegin* current, int local) {
 185   store_one(current, local);
 186   store_one(current, local + 1);

 190 void BlockListBuilder::handle_exceptions(BlockBegin* current, int cur_bci) {
 191   // Draws edges from a block to its exception handlers
 192   XHandlers* list = xhandlers();
 193   const int n = list->length();
 194 
 195   for (int i = 0; i < n; i++) {
 196     XHandler* h = list->handler_at(i);
 197 
 198     if (h->covers(cur_bci)) {
 199       BlockBegin* entry = h->entry_block();
 200       assert(entry != NULL && entry == _bci2block->at(h->handler_bci()), "entry must be set");
 201       assert(entry->is_set(BlockBegin::exception_entry_flag), "flag must be set");
 202 
 203       // add each exception handler only once
 204       if (!current->is_successor(entry)) {
 205         current->add_successor(entry);
 206         entry->increment_total_preds();
 207       }
 208 
 209       // stop when reaching catchall
 210       if (h->catch_type() == 0) break;
 211     }
 212   }
 213 }

 215 void BlockListBuilder::handle_jsr(BlockBegin* current, int sr_bci, int next_bci) {
 216   // start a new block after jsr-bytecode and link this block into cfg
 217   make_block_at(next_bci, current);
 218 
 219   // start a new block at the subroutine entry at mark it with special flag
 220   BlockBegin* sr_block = make_block_at(sr_bci, current);
 221   if (!sr_block->is_set(BlockBegin::subroutine_entry_flag)) {
 222     sr_block->set(BlockBegin::subroutine_entry_flag);
 223   }
 224 }

 227 void BlockListBuilder::set_leaders() {
 228   bool has_xhandlers = xhandlers()->has_handlers();
 229   BlockBegin* current = NULL;
 230 
 231   // The information which bci starts a new block simplifies the analysis
 232   // Without it, backward branches could jump to a bci where no block was created
 233   // during bytecode iteration. This would require the creation of a new block at the
 234   // branch target and a modification of the successor lists.
 235   const BitMap& bci_block_start = method()->bci_block_start();
 236 
 237   ciBytecodeStream s(method());
 238   while (s.next() != ciBytecodeStream::EOBC()) {
 239     int cur_bci = s.cur_bci();
 240 
 241     if (bci_block_start.at(cur_bci)) {
 242       current = make_block_at(cur_bci, current);
 243     }
 244     assert(current != NULL, "must have current block");
 245 
 246     if (has_xhandlers && GraphBuilder::can_trap(method(), s.cur_bc())) {
 247       handle_exceptions(current, cur_bci);
 248     }
 249 
 250     switch (s.cur_bc()) {
 251       // track stores to local variables for selective creation of phi functions
 252       case Bytecodes::_iinc:     store_one(current, s.get_index()); break;
 253       case Bytecodes::_istore:   store_one(current, s.get_index()); break;
 254       case Bytecodes::_lstore:   store_two(current, s.get_index()); break;
 255       case Bytecodes::_fstore:   store_one(current, s.get_index()); break;
 256       case Bytecodes::_dstore:   store_two(current, s.get_index()); break;
 257       case Bytecodes::_astore:   store_one(current, s.get_index()); break;
 258       case Bytecodes::_istore_0: store_one(current, 0); break;
 259       case Bytecodes::_istore_1: store_one(current, 1); break;
 260       case Bytecodes::_istore_2: store_one(current, 2); break;
 261       case Bytecodes::_istore_3: store_one(current, 3); break;
 262       case Bytecodes::_lstore_0: store_two(current, 0); break;
 263       case Bytecodes::_lstore_1: store_two(current, 1); break;
 264       case Bytecodes::_lstore_2: store_two(current, 2); break;
 265       case Bytecodes::_lstore_3: store_two(current, 3); break;
 266       case Bytecodes::_fstore_0: store_one(current, 0); break;
 267       case Bytecodes::_fstore_1: store_one(current, 1); break;
 268       case Bytecodes::_fstore_2: store_one(current, 2); break;
 269       case Bytecodes::_fstore_3: store_one(current, 3); break;
 270       case Bytecodes::_dstore_0: store_two(current, 0); break;
 271       case Bytecodes::_dstore_1: store_two(current, 1); break;
 272       case Bytecodes::_dstore_2: store_two(current, 2); break;
 273       case Bytecodes::_dstore_3: store_two(current, 3); break;
 274       case Bytecodes::_astore_0: store_one(current, 0); break;
 275       case Bytecodes::_astore_1: store_one(current, 1); break;
 276       case Bytecodes::_astore_2: store_one(current, 2); break;
 277       case Bytecodes::_astore_3: store_one(current, 3); break;
 278 
 279       // track bytecodes that affect the control flow
 280       case Bytecodes::_athrow:  // fall through
 281       case Bytecodes::_ret:     // fall through
 282       case Bytecodes::_ireturn: // fall through
 283       case Bytecodes::_lreturn: // fall through
 284       case Bytecodes::_freturn: // fall through
 285       case Bytecodes::_dreturn: // fall through
 286       case Bytecodes::_areturn: // fall through
 287       case Bytecodes::_return:
 288         current = NULL;
 289         break;
 290 
 291       case Bytecodes::_ifeq:      // fall through
 292       case Bytecodes::_ifne:      // fall through
 293       case Bytecodes::_iflt:      // fall through
 294       case Bytecodes::_ifge:      // fall through
 295       case Bytecodes::_ifgt:      // fall through
 296       case Bytecodes::_ifle:      // fall through
 297       case Bytecodes::_if_icmpeq: // fall through
 298       case Bytecodes::_if_icmpne: // fall through
 299       case Bytecodes::_if_icmplt: // fall through
 300       case Bytecodes::_if_icmpge: // fall through
 301       case Bytecodes::_if_icmpgt: // fall through
 302       case Bytecodes::_if_icmple: // fall through
 303       case Bytecodes::_if_acmpeq: // fall through
 304       case Bytecodes::_if_acmpne: // fall through
 305       case Bytecodes::_ifnull:    // fall through
 306       case Bytecodes::_ifnonnull:
 307         make_block_at(s.next_bci(), current);
 308         make_block_at(s.get_dest(), current);
 309         current = NULL;
 310         break;
 311 
 312       case Bytecodes::_goto:
 313         make_block_at(s.get_dest(), current);
 314         current = NULL;
 315         break;
 316 
 317       case Bytecodes::_goto_w:
 318         make_block_at(s.get_far_dest(), current);
 319         current = NULL;
 320         break;
 321 
 322       case Bytecodes::_jsr:
 323         handle_jsr(current, s.get_dest(), s.next_bci());
 324         current = NULL;
 325         break;
 326 
 327       case Bytecodes::_jsr_w:
 328         handle_jsr(current, s.get_far_dest(), s.next_bci());
 329         current = NULL;
 330         break;
 331 
 332       case Bytecodes::_tableswitch: {
 333         // set block for each case
 334         Bytecode_tableswitch sw(&s);
 335         int l = sw.length();
 336         for (int i = 0; i < l; i++) {
 337           make_block_at(cur_bci + sw.dest_offset_at(i), current);
 338         }
 339         make_block_at(cur_bci + sw.default_offset(), current);
 340         current = NULL;
 341         break;
 342       }
 343 
 344       case Bytecodes::_lookupswitch: {
 345         // set block for each case
 346         Bytecode_lookupswitch sw(&s);
 347         int l = sw.number_of_pairs();
 348         for (int i = 0; i < l; i++) {
 349           make_block_at(cur_bci + sw.pair_at(i).offset(), current);
 350         }
 351         make_block_at(cur_bci + sw.default_offset(), current);
 352         current = NULL;
 353         break;
 354       }
 355 
 356       default:
 357         break;
 358     }
 359   }
 360 }

 363 void BlockListBuilder::mark_loops() {
 364   ResourceMark rm;
 365 
 366   _active.initialize(BlockBegin::number_of_blocks());
 367   _visited.initialize(BlockBegin::number_of_blocks());
 368   _loop_map = intArray(BlockBegin::number_of_blocks(), BlockBegin::number_of_blocks(), 0);
 369   _next_loop_index = 0;
 370   _next_block_number = _blocks.length();
 371 
 372   // recursively iterate the control flow graph
 373   mark_loops(_bci2block->at(0), false);
 374   assert(_next_block_number >= 0, "invalid block numbers");
 375 
 376   // Remove dangling Resource pointers before the ResourceMark goes out-of-scope.
 377   _active.resize(0);
 378   _visited.resize(0);
 379 }

 381 void BlockListBuilder::make_loop_header(BlockBegin* block) {
 382   if (block->is_set(BlockBegin::exception_entry_flag)) {
 383     // exception edges may look like loops but don't mark them as such
 384     // since it screws up block ordering.
 385     return;
 386   }
 387   if (!block->is_set(BlockBegin::parser_loop_header_flag)) {
 388     block->set(BlockBegin::parser_loop_header_flag);
 389 
 390     assert(_loop_map.at(block->block_id()) == 0, "must not be set yet");
 391     assert(0 <= _next_loop_index && _next_loop_index < BitsPerInt, "_next_loop_index is used as a bit-index in integer");
 392     _loop_map.at_put(block->block_id(), 1 << _next_loop_index);
 393     if (_next_loop_index < 31) _next_loop_index++;
 394   } else {
 395     // block already marked as loop header
 396     assert(is_power_of_2((unsigned int)_loop_map.at(block->block_id())), "exactly one bit must be set");
 397   }
 398 }

 400 int BlockListBuilder::mark_loops(BlockBegin* block, bool in_subroutine) {
 401   int block_id = block->block_id();
 402 
 403   if (_visited.at(block_id)) {
 404     if (_active.at(block_id)) {
 405       // reached block via backward branch
 406       make_loop_header(block);
 407     }
 408     // return cached loop information for this block
 409     return _loop_map.at(block_id);
 410   }
 411 
 412   if (block->is_set(BlockBegin::subroutine_entry_flag)) {
 413     in_subroutine = true;
 414   }
 415 
 416   // set active and visited bits before successors are processed
 417   _visited.set_bit(block_id);
 418   _active.set_bit(block_id);
 419 
 420   intptr_t loop_state = 0;
 421   for (int i = block->number_of_sux() - 1; i >= 0; i--) {
 422     // recursively process all successors
 423     loop_state |= mark_loops(block->sux_at(i), in_subroutine);
 424   }
 425 
 426   // clear active-bit after all successors are processed
 427   _active.clear_bit(block_id);
 428 
 429   // reverse-post-order numbering of all blocks
 430   block->set_depth_first_number(_next_block_number);
 431   _next_block_number--;
 432 
 433   if (loop_state != 0 || in_subroutine ) {
 434     // block is contained at least in one loop, so phi functions are necessary
 435     // phi functions are also necessary for all locals stored in a subroutine
 436     scope()->requires_phi_function().set_union(block->stores_to_locals());
 437   }
 438 
 439   if (block->is_set(BlockBegin::parser_loop_header_flag)) {
 440     int header_loop_state = _loop_map.at(block_id);
 441     assert(is_power_of_2((unsigned)header_loop_state), "exactly one bit must be set");
 442 
 443     // If the highest bit is set (i.e. when integer value is negative), the method
 444     // has 32 or more loops. This bit is never cleared because it is used for multiple loops
 445     if (header_loop_state >= 0) {
 446       clear_bits(loop_state, header_loop_state);
 447     }
 448   }
 449 
 450   // cache and return loop information for this block
 451   _loop_map.at_put(block_id, loop_state);
 452   return loop_state;
 453 }

 462 void BlockListBuilder::print() {
 463   tty->print("----- initial block list of BlockListBuilder for method ");
 464   method()->print_short_name();
 465   tty->cr();
 466 
 467   // better readability if blocks are sorted in processing order
 468   _blocks.sort(compare_depth_first);
 469 
 470   for (int i = 0; i < _blocks.length(); i++) {
 471     BlockBegin* cur = _blocks.at(i);
 472     tty->print("%4d: B%-4d bci: %-4d  preds: %-4d ", cur->depth_first_number(), cur->block_id(), cur->bci(), cur->total_preds());
 473 
 474     tty->print(cur->is_set(BlockBegin::std_entry_flag)               ? " std" : "    ");
 475     tty->print(cur->is_set(BlockBegin::osr_entry_flag)               ? " osr" : "    ");
 476     tty->print(cur->is_set(BlockBegin::exception_entry_flag)         ? " ex" : "   ");
 477     tty->print(cur->is_set(BlockBegin::subroutine_entry_flag)        ? " sr" : "   ");
 478     tty->print(cur->is_set(BlockBegin::parser_loop_header_flag)      ? " lh" : "   ");
 479 
 480     if (cur->number_of_sux() > 0) {
 481       tty->print("    sux: ");
 482       for (int j = 0; j < cur->number_of_sux(); j++) {
 483         BlockBegin* sux = cur->sux_at(j);
 484         tty->print("B%d ", sux->block_id());
 485       }
 486     }
 487     tty->cr();
 488   }
 489 }

 494 // A simple growable array of Values indexed by ciFields
 495 class FieldBuffer: public CompilationResourceObj {
 496  private:
 497   GrowableArray<Value> _values;
 498 
 499  public:
 500   FieldBuffer() {}
 501 
 502   void kill() {
 503     _values.trunc_to(0);
 504   }
 505 
 506   Value at(ciField* field) {
 507     assert(field->holder()->is_loaded(), "must be a loaded field");
 508     int offset = field->offset();
 509     if (offset < _values.length()) {
 510       return _values.at(offset);
 511     } else {
 512       return NULL;
 513     }
 514   }
 515 
 516   void at_put(ciField* field, Value value) {
 517     assert(field->holder()->is_loaded(), "must be a loaded field");
 518     int offset = field->offset();
 519     _values.at_put_grow(offset, value, NULL);
 520   }
 521 
 522 };

 525 // MemoryBuffer is fairly simple model of the current state of memory.
 526 // It partitions memory into several pieces.  The first piece is
 527 // generic memory where little is known about the owner of the memory.
 528 // This is conceptually represented by the tuple <O, F, V> which says
 529 // that the field F of object O has value V.  This is flattened so
 530 // that F is represented by the offset of the field and the parallel
 531 // arrays _objects and _values are used for O and V.  Loads of O.F can
 532 // simply use V.  Newly allocated objects are kept in a separate list
 533 // along with a parallel array for each object which represents the
 534 // current value of its fields.  Stores of the default value to fields
 535 // which have never been stored to before are eliminated since they
 536 // are redundant.  Once newly allocated objects are stored into
 537 // another object or they are passed out of the current compile they
 538 // are treated like generic memory.
 539 
 540 class MemoryBuffer: public CompilationResourceObj {
 541  private:
 542   FieldBuffer                 _values;
 543   GrowableArray<Value>        _objects;
 544   GrowableArray<Value>        _newobjects;
 545   GrowableArray<FieldBuffer*> _fields;
 546 
 547  public:
 548   MemoryBuffer() {}
 549 
 550   StoreField* store(StoreField* st) {
 551     if (!EliminateFieldAccess) {
 552       return st;
 553     }
 554 
 555     Value object = st->obj();
 556     Value value = st->value();
 557     ciField* field = st->field();
 558     if (field->holder()->is_loaded()) {
 559       int offset = field->offset();
 560       int index = _newobjects.find(object);
 561       if (index != -1) {
 562         // newly allocated object with no other stores performed on this field
 563         FieldBuffer* buf = _fields.at(index);
 564         if (buf->at(field) == NULL && is_default_value(value)) {
 565 #ifndef PRODUCT
 566           if (PrintIRDuringConstruction && Verbose) {
 567             tty->print_cr("Eliminated store for object %d:", index);
 568             st->print_line();
 569           }
 570 #endif
 571           return NULL;
 572         } else {
 573           buf->at_put(field, value);
 574         }
 575       } else {
 576         _objects.at_put_grow(offset, object, NULL);
 577         _values.at_put(field, value);
 578       }
 579 
 580       store_value(value);
 581     } else {
 582       // if we held onto field names we could alias based on names but
 583       // we don't know what's being stored to so kill it all.
 584       kill();
 585     }
 586     return st;
 587   }
 588 
 589 
 590   // return true if this value correspond to the default value of a field.
 591   bool is_default_value(Value value) {
 592     Constant* con = value->as_Constant();
 593     if (con) {
 594       switch (con->type()->tag()) {
 595         case intTag:    return con->type()->as_IntConstant()->value() == 0;
 596         case longTag:   return con->type()->as_LongConstant()->value() == 0;
 597         case floatTag:  return jint_cast(con->type()->as_FloatConstant()->value()) == 0;
 598         case doubleTag: return jlong_cast(con->type()->as_DoubleConstant()->value()) == jlong_cast(0);
 599         case objectTag: return con->type() == objectNull;
 600         default:  ShouldNotReachHere();
 601       }
 602     }
 603     return false;
 604   }
 605 
 606 
 607   // return either the actual value of a load or the load itself
 608   Value load(LoadField* load) {
 609     if (!EliminateFieldAccess) {
 610       return load;
 611     }
 612 
 613     if (strict_fp_requires_explicit_rounding && load->type()->is_float_kind()) {
 614 #ifdef IA32
 615       if (UseSSE < 2) {
 616         // can't skip load since value might get rounded as a side effect
 617         return load;
 618       }
 619 #else
 620       Unimplemented();
 621 #endif // IA32
 622     }
 623 
 624     ciField* field = load->field();
 625     Value object   = load->obj();
 626     if (field->holder()->is_loaded() && !field->is_volatile()) {
 627       int offset = field->offset();
 628       Value result = NULL;
 629       int index = _newobjects.find(object);
 630       if (index != -1) {
 631         result = _fields.at(index)->at(field);
 632       } else if (_objects.at_grow(offset, NULL) == object) {
 633         result = _values.at(field);
 634       }
 635       if (result != NULL) {
 636 #ifndef PRODUCT
 637         if (PrintIRDuringConstruction && Verbose) {
 638           tty->print_cr("Eliminated load: ");
 639           load->print_line();
 640         }
 641 #endif
 642         assert(result->type()->tag() == load->type()->tag(), "wrong types");
 643         return result;
 644       }
 645     }
 646     return load;
 647   }
 648 
 649   // Record this newly allocated object
 650   void new_instance(NewInstance* object) {
 651     int index = _newobjects.length();
 652     _newobjects.append(object);
 653     if (_fields.at_grow(index, NULL) == NULL) {
 654       _fields.at_put(index, new FieldBuffer());
 655     } else {
 656       _fields.at(index)->kill();
 657     }
 658   }
 659 
 660   // Record this newly allocated object
 661   void new_instance(NewValueTypeInstance* object) {
 662     int index = _newobjects.length();
 663     _newobjects.append(object);
 664     if (_fields.at_grow(index, NULL) == NULL) {
 665       _fields.at_put(index, new FieldBuffer());
 666     } else {
 667       _fields.at(index)->kill();
 668     }
 669   }
 670 
 671   void store_value(Value value) {
 672     int index = _newobjects.find(value);
 673     if (index != -1) {
 674       // stored a newly allocated object into another object.
 675       // Assume we've lost track of it as separate slice of memory.
 676       // We could do better by keeping track of whether individual
 677       // fields could alias each other.
 678       _newobjects.remove_at(index);
 679       // pull out the field info and store it at the end up the list
 680       // of field info list to be reused later.
 681       _fields.append(_fields.at(index));
 682       _fields.remove_at(index);
 683     }
 684   }
 685 
 686   void kill() {
 687     _newobjects.trunc_to(0);
 688     _objects.trunc_to(0);
 689     _values.kill();
 690   }
 691 };

 732 BlockBegin* GraphBuilder::ScopeData::block_at(int bci) {
 733   if (parsing_jsr()) {
 734     // It is necessary to clone all blocks associated with a
 735     // subroutine, including those for exception handlers in the scope
 736     // of the method containing the jsr (because those exception
 737     // handlers may contain ret instructions in some cases).
 738     BlockBegin* block = bci2block()->at(bci);
 739     if (block != NULL && block == parent()->bci2block()->at(bci)) {
 740       BlockBegin* new_block = new BlockBegin(block->bci());
 741       if (PrintInitialBlockList) {
 742         tty->print_cr("CFG: cloned block %d (bci %d) as block %d for jsr",
 743                       block->block_id(), block->bci(), new_block->block_id());
 744       }
 745       // copy data from cloned blocked
 746       new_block->set_depth_first_number(block->depth_first_number());
 747       if (block->is_set(BlockBegin::parser_loop_header_flag)) new_block->set(BlockBegin::parser_loop_header_flag);
 748       // Preserve certain flags for assertion checking
 749       if (block->is_set(BlockBegin::subroutine_entry_flag)) new_block->set(BlockBegin::subroutine_entry_flag);
 750       if (block->is_set(BlockBegin::exception_entry_flag))  new_block->set(BlockBegin::exception_entry_flag);
 751 
 752       // copy was_visited_flag to allow early detection of bailouts
 753       // if a block that is used in a jsr has already been visited before,
 754       // it is shared between the normal control flow and a subroutine
 755       // BlockBegin::try_merge returns false when the flag is set, this leads
 756       // to a compilation bailout
 757       if (block->is_set(BlockBegin::was_visited_flag))  new_block->set(BlockBegin::was_visited_flag);
 758 
 759       bci2block()->at_put(bci, new_block);
 760       block = new_block;
 761     }
 762     return block;
 763   } else {
 764     return bci2block()->at(bci);
 765   }
 766 }

 823 void GraphBuilder::sort_top_into_worklist(BlockList* worklist, BlockBegin* top) {
 824   assert(worklist->top() == top, "");
 825   // sort block descending into work list
 826   const int dfn = top->depth_first_number();
 827   assert(dfn != -1, "unknown depth first number");
 828   int i = worklist->length()-2;
 829   while (i >= 0) {
 830     BlockBegin* b = worklist->at(i);
 831     if (b->depth_first_number() < dfn) {
 832       worklist->at_put(i+1, b);
 833     } else {
 834       break;
 835     }
 836     i --;
 837   }
 838   if (i >= -1) worklist->at_put(i + 1, top);
 839 }

 896 void GraphBuilder::load_constant() {
 897   ciConstant con = stream()->get_constant();
 898   if (con.basic_type() == T_ILLEGAL) {
 899     // FIXME: an unresolved Dynamic constant can get here,
 900     // and that should not terminate the whole compilation.
 901     BAILOUT("could not resolve a constant");
 902   } else {
 903     ValueType* t = illegalType;
 904     ValueStack* patch_state = NULL;
 905     switch (con.basic_type()) {
 906       case T_BOOLEAN: t = new IntConstant     (con.as_boolean()); break;
 907       case T_BYTE   : t = new IntConstant     (con.as_byte   ()); break;
 908       case T_CHAR   : t = new IntConstant     (con.as_char   ()); break;
 909       case T_SHORT  : t = new IntConstant     (con.as_short  ()); break;
 910       case T_INT    : t = new IntConstant     (con.as_int    ()); break;
 911       case T_LONG   : t = new LongConstant    (con.as_long   ()); break;
 912       case T_FLOAT  : t = new FloatConstant   (con.as_float  ()); break;
 913       case T_DOUBLE : t = new DoubleConstant  (con.as_double ()); break;
 914       case T_ARRAY  : t = new ArrayConstant   (con.as_object ()->as_array   ()); break;
 915       case T_OBJECT :
 916        {
 917         ciObject* obj = con.as_object();
 918         if (!obj->is_loaded()
 919             || (PatchALot && obj->klass() != ciEnv::current()->String_klass())) {
 920           // A Class, MethodType, MethodHandle, or String.
 921           // Unloaded condy nodes show up as T_ILLEGAL, above.
 922           patch_state = copy_state_before();
 923           t = new ObjectConstant(obj);
 924         } else {
 925           // Might be a Class, MethodType, MethodHandle, or Dynamic constant
 926           // result, which might turn out to be an array.
 927           if (obj->is_null_object())
 928             t = objectNull;
 929           else if (obj->is_array())
 930             t = new ArrayConstant(obj->as_array());
 931           else
 932             t = new InstanceConstant(obj->as_instance());
 933         }
 934         break;
 935        }
 936       default       : ShouldNotReachHere();
 937     }
 938     Value x;
 939     if (patch_state != NULL) {
 940       x = new Constant(t, patch_state);
 941     } else {
 942       x = new Constant(t);
 943     }
 944     push(t, append(x));
 945   }
 946 }

 962 void GraphBuilder::store_local(ValueStack* state, Value x, int index) {
 963   if (parsing_jsr()) {
 964     // We need to do additional tracking of the location of the return
 965     // address for jsrs since we don't handle arbitrary jsr/ret
 966     // constructs. Here we are figuring out in which circumstances we
 967     // need to bail out.
 968     if (x->type()->is_address()) {
 969       scope_data()->set_jsr_return_address_local(index);
 970 
 971       // Also check parent jsrs (if any) at this time to see whether
 972       // they are using this local. We don't handle skipping over a
 973       // ret.
 974       for (ScopeData* cur_scope_data = scope_data()->parent();
 975            cur_scope_data != NULL && cur_scope_data->parsing_jsr() && cur_scope_data->scope() == scope();
 976            cur_scope_data = cur_scope_data->parent()) {
 977         if (cur_scope_data->jsr_return_address_local() == index) {
 978           BAILOUT("subroutine overwrites return address from previous subroutine");
 979         }
 980       }
 981     } else if (index == scope_data()->jsr_return_address_local()) {
 982       scope_data()->set_jsr_return_address_local(-1);
 983     }
 984   }
 985 
 986   x->set_local_index(index);
 987   state->store_local(index, round_fp(x));
 988 }

 991 void GraphBuilder::load_indexed(BasicType type) {
 992   // In case of in block code motion in range check elimination
 993   ValueStack* state_before = NULL;
 994   int array_idx = state()->stack_size() - 2;
 995   if (type == T_OBJECT && state()->stack_at(array_idx)->maybe_flattened_array()) {
 996     // Save the entire state and re-execute on deopt when accessing flattened arrays
 997     state_before = copy_state_before();
 998     state_before->set_should_reexecute(true);
 999   } else {
1000     state_before = copy_state_indexed_access();
1001   }
1002   compilation()->set_has_access_indexed(true);
1003   Value index = ipop();
1004   Value array = apop();
1005   Value length = NULL;
1006   if (CSEArrayLength ||
1007       (array->as_AccessField() && array->as_AccessField()->field()->is_constant()) ||
1008       (array->as_NewArray() && array->as_NewArray()->length() && array->as_NewArray()->length()->type()->is_constant())) {
1009     length = append(new ArrayLength(array, state_before));
1010   }
1011 
1012   LoadIndexed* load_indexed = NULL;
1013   Instruction* result = NULL;
1014   if (array->is_loaded_flattened_array()) {
1015     ciType* array_type = array->declared_type();
1016     ciValueKlass* elem_klass = array_type->as_value_array_klass()->element_klass()->as_value_klass();
1017     NewValueTypeInstance* new_instance = new NewValueTypeInstance(elem_klass, state_before, false);
1018     _memory->new_instance(new_instance);
1019     apush(append_split(new_instance));
1020     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1021     load_indexed->set_vt(new_instance);
1022   } else {
1023     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1024   }
1025   if (profile_array_accesses() && is_reference_type(type)) {
1026     compilation()->set_would_profile(true);
1027     load_indexed->set_should_profile(true);
1028     load_indexed->set_profiled_method(method());
1029     load_indexed->set_profiled_bci(bci());
1030   }
1031   result = append(load_indexed);
1032   assert(!(profile_array_accesses() && is_reference_type(type)) || load_indexed == result, "should not be optimized out");
1033   if (!array->is_loaded_flattened_array()) {
1034     push(as_ValueType(type), result);
1035   }
1036 }

1089   switch (code) {
1090     case Bytecodes::_pop:
1091       { state()->raw_pop();
1092       }
1093       break;
1094     case Bytecodes::_pop2:
1095       { state()->raw_pop();
1096         state()->raw_pop();
1097       }
1098       break;
1099     case Bytecodes::_dup:
1100       { Value w = state()->raw_pop();
1101         state()->raw_push(w);
1102         state()->raw_push(w);

1104       break;

1118         state()->raw_push(w3);

1120         state()->raw_push(w1);
1121       }
1122       break;

1126         state()->raw_push(w2);

1190 void GraphBuilder::shift_op(ValueType* type, Bytecodes::Code code) {
1191   Value s = ipop();
1192   Value x = pop(type);
1193   // try to simplify
1194   // Note: This code should go into the canonicalizer as soon as it can
1195   //       can handle canonicalized forms that contain more than one node.
1196   if (CanonicalizeNodes && code == Bytecodes::_iushr) {
1197     // pattern: x >>> s
1198     IntConstant* s1 = s->type()->as_IntConstant();
1199     if (s1 != NULL) {
1200       // pattern: x >>> s1, with s1 constant
1201       ShiftOp* l = x->as_ShiftOp();
1202       if (l != NULL && l->op() == Bytecodes::_ishl) {
1203         // pattern: (a << b) >>> s1
1204         IntConstant* s0 = l->y()->type()->as_IntConstant();
1205         if (s0 != NULL) {
1206           // pattern: (a << s0) >>> s1
1207           const int s0c = s0->value() & 0x1F; // only the low 5 bits are significant for shifts
1208           const int s1c = s1->value() & 0x1F; // only the low 5 bits are significant for shifts
1209           if (s0c == s1c) {
1210             if (s0c == 0) {
1211               // pattern: (a << 0) >>> 0 => simplify to: a
1212               ipush(l->x());
1213             } else {
1214               // pattern: (a << s0c) >>> s0c => simplify to: a & m, with m constant
1215               assert(0 < s0c && s0c < BitsPerInt, "adjust code below to handle corner cases");
1216               const int m = (1 << (BitsPerInt - s0c)) - 1;
1217               Value s = append(new Constant(new IntConstant(m)));
1218               ipush(append(new LogicOp(Bytecodes::_iand, l->x(), s)));
1219             }
1220             return;
1221           }
1222         }
1223       }
1224     }
1225   }
1226   // could not simplify
1227   push(type, append(new ShiftOp(code, x, s)));
1228 }

1275 void GraphBuilder::if_node(Value x, If::Condition cond, Value y, ValueStack* state_before) {
1276   BlockBegin* tsux = block_at(stream()->get_dest());
1277   BlockBegin* fsux = block_at(stream()->next_bci());
1278   bool is_bb = tsux->bci() < stream()->cur_bci() || fsux->bci() < stream()->cur_bci();
1279 
1280   bool subst_check = false;
1281   if (EnableValhalla && (stream()->cur_bc() == Bytecodes::_if_acmpeq || stream()->cur_bc() == Bytecodes::_if_acmpne) &&
1282       method() != ciEnv::current()->ValueBootstrapMethods_klass()->find_method(ciSymbol::isSubstitutable_name(), ciSymbol::object_object_boolean_signature())) {
1283     // If current method is ValueBootstrapMethods::isSubstitutable(),
1284     // compile the acmp as a regular pointer comparison otherwise we
1285     // could call ValueBootstrapMethods::isSubstitutable() back
1286     ValueType* left_vt = x->type();
1287     ValueType* right_vt = y->type();
1288     if (left_vt->is_object()) {
1289       assert(right_vt->is_object(), "must be");
1290       ciKlass* left_klass = x->as_loaded_klass_or_null();
1291       ciKlass* right_klass = y->as_loaded_klass_or_null();
1292 
1293       if (left_klass == NULL || right_klass == NULL) {
1294         // The klass is still unloaded, or came from a Phi node. Go slow case;
1295         subst_check = true;
1296       } else if (left_klass->can_be_value_klass() || right_klass->can_be_value_klass()) {
1297         // Either operand may be a value object, but we're not sure. Go slow case;
1298         subst_check = true;
1299       } else {
1300         // No need to do substitutability check
1301       }
1302     }
1303   }
1304 
1305   // In case of loop invariant code motion or predicate insertion
1306   // before the body of a loop the state is needed
1307   Instruction *i = append(new If(x, cond, false, y, tsux, fsux, (is_bb || compilation()->is_optimistic() || subst_check) ? state_before : NULL, is_bb, subst_check));
1308 
1309   assert(i->as_Goto() == NULL ||
1310          (i->as_Goto()->sux_at(0) == tsux  && i->as_Goto()->is_safepoint() == tsux->bci() < stream()->cur_bci()) ||
1311          (i->as_Goto()->sux_at(0) == fsux  && i->as_Goto()->is_safepoint() == fsux->bci() < stream()->cur_bci()),
1312          "safepoint state of Goto returned by canonicalizer incorrect");
1313 
1314   if (is_profiling()) {
1315     If* if_node = i->as_If();
1316     if (if_node != NULL) {
1317       // Note that we'd collect profile data in this method if we wanted it.
1318       compilation()->set_would_profile(true);
1319       // At level 2 we need the proper bci to count backedges
1320       if_node->set_profiled_bci(bci());
1321       if (profile_branches()) {
1322         // Successors can be rotated by the canonicalizer, check for this case.
1323         if_node->set_profiled_method(method());
1324         if_node->set_should_profile(true);
1325         if (if_node->tsux() == fsux) {
1326           if_node->set_swapped(true);
1327         }
1328       }
1329       return;
1330     }
1331 
1332     // Check if this If was reduced to Goto.
1333     Goto *goto_node = i->as_Goto();
1334     if (goto_node != NULL) {
1335       compilation()->set_would_profile(true);
1336       goto_node->set_profiled_bci(bci());
1337       if (profile_branches()) {
1338         goto_node->set_profiled_method(method());
1339         goto_node->set_should_profile(true);
1340         // Find out which successor is used.
1341         if (goto_node->default_sux() == tsux) {
1342           goto_node->set_direction(Goto::taken);
1343         } else if (goto_node->default_sux() == fsux) {
1344           goto_node->set_direction(Goto::not_taken);
1345         } else {
1346           ShouldNotReachHere();
1347         }
1348       }
1349       return;
1350     }
1351   }
1352 }

1411 void GraphBuilder::table_switch() {
1412   Bytecode_tableswitch sw(stream());
1413   const int l = sw.length();
1414   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1415     // total of 2 successors => use If instead of switch
1416     // Note: This code should go into the canonicalizer as soon as it can
1417     //       can handle canonicalized forms that contain more than one node.
1418     Value key = append(new Constant(new IntConstant(sw.low_key())));
1419     BlockBegin* tsux = block_at(bci() + sw.dest_offset_at(0));
1420     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1421     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1422     // In case of loop invariant code motion or predicate insertion
1423     // before the body of a loop the state is needed
1424     ValueStack* state_before = copy_state_if_bb(is_bb);
1425     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1426   } else {
1427     // collect successors
1428     BlockList* sux = new BlockList(l + 1, NULL);
1429     int i;
1430     bool has_bb = false;
1431     for (i = 0; i < l; i++) {
1432       sux->at_put(i, block_at(bci() + sw.dest_offset_at(i)));
1433       if (sw.dest_offset_at(i) < 0) has_bb = true;
1434     }
1435     // add default successor
1436     if (sw.default_offset() < 0) has_bb = true;
1437     sux->at_put(i, block_at(bci() + sw.default_offset()));
1438     // In case of loop invariant code motion or predicate insertion
1439     // before the body of a loop the state is needed
1440     ValueStack* state_before = copy_state_if_bb(has_bb);
1441     Instruction* res = append(new TableSwitch(ipop(), sux, sw.low_key(), state_before, has_bb));
1442 #ifdef ASSERT
1443     if (res->as_Goto()) {
1444       for (i = 0; i < l; i++) {
1445         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1446           assert(res->as_Goto()->is_safepoint() == sw.dest_offset_at(i) < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1447         }
1448       }
1449     }
1450 #endif
1451   }
1452 }

1455 void GraphBuilder::lookup_switch() {
1456   Bytecode_lookupswitch sw(stream());
1457   const int l = sw.number_of_pairs();
1458   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1459     // total of 2 successors => use If instead of switch
1460     // Note: This code should go into the canonicalizer as soon as it can
1461     //       can handle canonicalized forms that contain more than one node.
1462     // simplify to If
1463     LookupswitchPair pair = sw.pair_at(0);
1464     Value key = append(new Constant(new IntConstant(pair.match())));
1465     BlockBegin* tsux = block_at(bci() + pair.offset());
1466     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1467     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1468     // In case of loop invariant code motion or predicate insertion
1469     // before the body of a loop the state is needed
1470     ValueStack* state_before = copy_state_if_bb(is_bb);;
1471     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1472   } else {
1473     // collect successors & keys
1474     BlockList* sux = new BlockList(l + 1, NULL);
1475     intArray* keys = new intArray(l, l, 0);
1476     int i;
1477     bool has_bb = false;
1478     for (i = 0; i < l; i++) {
1479       LookupswitchPair pair = sw.pair_at(i);
1480       if (pair.offset() < 0) has_bb = true;
1481       sux->at_put(i, block_at(bci() + pair.offset()));
1482       keys->at_put(i, pair.match());
1483     }
1484     // add default successor
1485     if (sw.default_offset() < 0) has_bb = true;
1486     sux->at_put(i, block_at(bci() + sw.default_offset()));
1487     // In case of loop invariant code motion or predicate insertion
1488     // before the body of a loop the state is needed
1489     ValueStack* state_before = copy_state_if_bb(has_bb);
1490     Instruction* res = append(new LookupSwitch(ipop(), sux, keys, state_before, has_bb));
1491 #ifdef ASSERT
1492     if (res->as_Goto()) {
1493       for (i = 0; i < l; i++) {
1494         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1495           assert(res->as_Goto()->is_safepoint() == sw.pair_at(i).offset() < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1496         }
1497       }
1498     }
1499 #endif
1500   }
1501 }

1503 void GraphBuilder::call_register_finalizer() {
1504   // If the receiver requires finalization then emit code to perform
1505   // the registration on return.
1506 
1507   // Gather some type information about the receiver
1508   Value receiver = state()->local_at(0);
1509   assert(receiver != NULL, "must have a receiver");
1510   ciType* declared_type = receiver->declared_type();
1511   ciType* exact_type = receiver->exact_type();
1512   if (exact_type == NULL &&
1513       receiver->as_Local() &&
1514       receiver->as_Local()->java_index() == 0) {
1515     ciInstanceKlass* ik = compilation()->method()->holder();
1516     if (ik->is_final()) {
1517       exact_type = ik;
1518     } else if (UseCHA && !(ik->has_subklass() || ik->is_interface())) {
1519       // test class is leaf class
1520       compilation()->dependency_recorder()->assert_leaf_type(ik);
1521       exact_type = ik;
1522     } else {
1523       declared_type = ik;
1524     }
1525   }
1526 
1527   // see if we know statically that registration isn't required
1528   bool needs_check = true;
1529   if (exact_type != NULL) {
1530     needs_check = exact_type->as_instance_klass()->has_finalizer();
1531   } else if (declared_type != NULL) {
1532     ciInstanceKlass* ik = declared_type->as_instance_klass();
1533     if (!Dependencies::has_finalizable_subclass(ik)) {
1534       compilation()->dependency_recorder()->assert_has_no_finalizable_subclasses(ik);
1535       needs_check = false;
1536     }
1537   }
1538 
1539   if (needs_check) {
1540     // Perform the registration of finalizable objects.
1541     ValueStack* state_before = copy_state_for_exception();
1542     load_local(objectType, 0);
1543     append_split(new Intrinsic(voidType, vmIntrinsics::_Object_init,
1544                                state()->pop_arguments(1),
1545                                true, state_before, true));
1546   }
1547 }

1550 void GraphBuilder::method_return(Value x, bool ignore_return) {
1551   if (RegisterFinalizersAtInit &&
1552       method()->intrinsic_id() == vmIntrinsics::_Object_init) {
1553     call_register_finalizer();
1554   }
1555 
1556   // The conditions for a memory barrier are described in Parse::do_exits().
1557   bool need_mem_bar = false;
1558   if (method()->is_object_constructor() &&
1559        (scope()->wrote_final() ||
1560          (AlwaysSafeConstructors && scope()->wrote_fields()) ||
1561          (support_IRIW_for_not_multiple_copy_atomic_cpu && scope()->wrote_volatile()))) {
1562     need_mem_bar = true;
1563   }
1564 
1565   BasicType bt = method()->return_type()->basic_type();
1566   switch (bt) {
1567     case T_BYTE:
1568     {
1569       Value shift = append(new Constant(new IntConstant(24)));
1570       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1571       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1572       break;
1573     }
1574     case T_SHORT:
1575     {
1576       Value shift = append(new Constant(new IntConstant(16)));
1577       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1578       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1579       break;
1580     }
1581     case T_CHAR:
1582     {
1583       Value mask = append(new Constant(new IntConstant(0xFFFF)));
1584       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1585       break;
1586     }
1587     case T_BOOLEAN:
1588     {
1589       Value mask = append(new Constant(new IntConstant(1)));
1590       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1591       break;
1592     }
1593     default:
1594       break;
1595   }
1596 
1597   // Check to see whether we are inlining. If so, Return
1598   // instructions become Gotos to the continuation point.
1599   if (continuation() != NULL) {
1600 
1601     int invoke_bci = state()->caller_state()->bci();
1602 
1603     if (x != NULL  && !ignore_return) {
1604       ciMethod* caller = state()->scope()->caller()->method();
1605       Bytecodes::Code invoke_raw_bc = caller->raw_code_at_bci(invoke_bci);
1606       if (invoke_raw_bc == Bytecodes::_invokehandle || invoke_raw_bc == Bytecodes::_invokedynamic) {
1607         ciType* declared_ret_type = caller->get_declared_signature_at_bci(invoke_bci)->return_type();
1608         if (declared_ret_type->is_klass() && x->exact_type() == NULL &&
1609             x->declared_type() != declared_ret_type && declared_ret_type != compilation()->env()->Object_klass()) {
1610           x = append(new TypeCast(declared_ret_type->as_klass(), x, copy_state_before()));
1611         }
1612       }
1613     }
1614 
1615     assert(!method()->is_synchronized() || InlineSynchronizedMethods, "can not inline synchronized methods yet");
1616 
1617     if (compilation()->env()->dtrace_method_probes()) {
1618       // Report exit from inline methods
1619       Values* args = new Values(1);
1620       args->push(append(new Constant(new MethodConstant(method()))));
1621       append(new RuntimeCall(voidType, "dtrace_method_exit", CAST_FROM_FN_PTR(address, SharedRuntime::dtrace_method_exit), args));
1622     }
1623 
1624     // If the inlined method is synchronized, the monitor must be
1625     // released before we jump to the continuation block.
1626     if (method()->is_synchronized()) {
1627       assert(state()->locks_size() == 1, "receiver must be locked here");
1628       monitorexit(state()->lock_at(0), SynchronizationEntryBCI);
1629     }
1630 
1631     if (need_mem_bar) {
1632       append(new MemBar(lir_membar_storestore));
1633     }
1634 
1635     // State at end of inlined method is the state of the caller
1636     // without the method parameters on stack, including the
1637     // return value, if any, of the inlined method on operand stack.
1638     set_state(state()->caller_state()->copy_for_parsing());
1639     if (x != NULL) {
1640       if (!ignore_return) {
1641         state()->push(x->type(), x);
1642       }
1643       if (profile_return() && x->type()->is_object_kind()) {
1644         ciMethod* caller = state()->scope()->method();
1645         profile_return_type(x, method(), caller, invoke_bci);
1646       }
1647     }
1648     Goto* goto_callee = new Goto(continuation(), false);
1649 
1650     // See whether this is the first return; if so, store off some
1651     // of the state for later examination
1652     if (num_returns() == 0) {
1653       set_inline_cleanup_info();
1654     }
1655 
1656     // The current bci() is in the wrong scope, so use the bci() of
1657     // the continuation point.
1658     append_with_bci(goto_callee, scope_data()->continuation()->bci());
1659     incr_num_returns();
1660     return;
1661   }
1662 
1663   state()->truncate_stack(0);
1664   if (method()->is_synchronized()) {
1665     // perform the unlocking before exiting the method
1666     Value receiver;
1667     if (!method()->is_static()) {
1668       receiver = _initial_state->local_at(0);
1669     } else {
1670       receiver = append(new Constant(new ClassConstant(method()->holder())));
1671     }
1672     append_split(new MonitorExit(receiver, state()->unlock()));
1673   }
1674 
1675   if (need_mem_bar) {
1676       append(new MemBar(lir_membar_storestore));
1677   }
1678 
1679   assert(!ignore_return, "Ignoring return value works only for inlining");
1680   append(new Return(x));
1681 }

1723 void GraphBuilder::access_field(Bytecodes::Code code) {
1724   bool will_link;
1725   ciField* field = stream()->get_field(will_link);
1726   ciInstanceKlass* holder = field->holder();
1727   BasicType field_type = field->type()->basic_type();
1728   ValueType* type = as_ValueType(field_type);
1729 
1730   // call will_link again to determine if the field is valid.
1731   const bool needs_patching = !holder->is_loaded() ||
1732                               !field->will_link(method(), code) ||
1733                               PatchALot;
1734 
1735   ValueStack* state_before = NULL;
1736   if (!holder->is_initialized() || needs_patching) {
1737     // save state before instruction for debug info when
1738     // deoptimization happens during patching
1739     state_before = copy_state_before();
1740   }
1741 
1742   Value obj = NULL;
1743   if (code == Bytecodes::_getstatic || code == Bytecodes::_putstatic) {
1744     if (state_before != NULL) {
1745       // build a patching constant
1746       obj = new Constant(new InstanceConstant(holder->java_mirror()), state_before);
1747     } else {
1748       obj = new Constant(new InstanceConstant(holder->java_mirror()));
1749     }
1750   }
1751 
1752   if (field->is_final() && (code == Bytecodes::_putfield || code == Bytecodes::_withfield)) {
1753     scope()->set_wrote_final();
1754   }
1755 
1756   if (code == Bytecodes::_putfield || code == Bytecodes::_withfield) {
1757     scope()->set_wrote_fields();
1758     if (field->is_volatile()) {
1759       scope()->set_wrote_volatile();
1760     }
1761   }
1762 
1763   const int offset = !needs_patching ? field->offset() : -1;
1764   switch (code) {
1765     case Bytecodes::_getstatic: {
1766       // check for compile-time constants, i.e., initialized static final fields
1767       Value constant = NULL;
1768       if (field->is_static_constant() && !PatchALot) {
1769         ciConstant field_value = field->constant_value();
1770         assert(!field->is_stable() || !field_value.is_null_or_zero(),
1771                "stable static w/ default value shouldn't be a constant");
1772         constant = make_constant(field_value, field);
1773       }
1774       if (constant != NULL) {
1775         push(type, append(constant));
1776       } else {
1777         if (state_before == NULL) {
1778           state_before = copy_state_for_exception();
1779         }
1780         LoadField* load_field = new LoadField(append(obj), offset, field, true,
1781                                         state_before, needs_patching);
1782         if (field->is_flattenable()) {
1783           load_field->set_never_null(true);
1784         }
1785         push(type, append(load_field));
1786       }
1787       break;
1788     }
1789     case Bytecodes::_putstatic: {
1790       Value val = pop(type);
1791       val->set_escaped();
1792       if (state_before == NULL) {
1793         state_before = copy_state_for_exception();
1794       }
1795       if (field->type()->basic_type() == T_BOOLEAN) {
1796         Value mask = append(new Constant(new IntConstant(1)));
1797         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1798       }
1799       append(new StoreField(append(obj), offset, field, val, true, state_before, needs_patching));
1800       break;
1801     }
1802     case Bytecodes::_getfield: {
1803       // Check for compile-time constants, i.e., trusted final non-static fields.
1804       Value constant = NULL;
1805       if (state_before == NULL && field->is_flattened()) {
1806         // Save the entire state and re-execute on deopt when accessing flattened fields
1807         assert(Interpreter::bytecode_should_reexecute(code), "should reexecute");
1808         state_before = copy_state_before();
1809       }
1810       obj = apop();
1811       ObjectType* obj_type = obj->type()->as_ObjectType();
1812       if (field->is_constant() && !field->is_flattened() && obj_type->is_constant() && !PatchALot) {
1813         ciObject* const_oop = obj_type->constant_value();
1814         if (!const_oop->is_null_object() && const_oop->is_loaded()) {
1815           ciConstant field_value = field->constant_value_of(const_oop);
1816           if (field_value.is_valid()) {
1817             constant = make_constant(field_value, field);
1818             // For CallSite objects add a dependency for invalidation of the optimization.
1819             if (field->is_call_site_target()) {
1820               ciCallSite* call_site = const_oop->as_call_site();
1821               if (!call_site->is_fully_initialized_constant_call_site()) {
1822                 ciMethodHandle* target = field_value.as_object()->as_method_handle();
1823                 dependency_recorder()->assert_call_site_target_value(call_site, target);
1824               }
1825             }
1826           }
1827         }
1828       }
1829       if (constant != NULL) {
1830         push(type, append(constant));
1831       } else {
1832         if (state_before == NULL) {
1833           state_before = copy_state_for_exception();
1834         }
1835         if (!field->is_flattened()) {
1836           LoadField* load = new LoadField(obj, offset, field, false, state_before, needs_patching);
1837           Value replacement = !needs_patching ? _memory->load(load) : load;
1838           if (replacement != load) {
1839             assert(replacement->is_linked() || !replacement->can_be_linked(), "should already by linked");
1840             // Writing an (integer) value to a boolean, byte, char or short field includes an implicit narrowing
1841             // conversion. Emit an explicit conversion here to get the correct field value after the write.
1842             BasicType bt = field->type()->basic_type();
1843             switch (bt) {
1844             case T_BOOLEAN:
1845             case T_BYTE:
1846               replacement = append(new Convert(Bytecodes::_i2b, replacement, as_ValueType(bt)));
1847               break;
1848             case T_CHAR:
1849               replacement = append(new Convert(Bytecodes::_i2c, replacement, as_ValueType(bt)));
1850               break;
1851             case T_SHORT:
1852               replacement = append(new Convert(Bytecodes::_i2s, replacement, as_ValueType(bt)));
1853               break;
1854             default:
1855               break;
1856             }
1857             push(type, replacement);
1858           } else {
1859             push(type, append(load));
1860           }
1861         } else { // flattened field, not optimized solution: re-instantiate the flattened value
1862           assert(field->type()->is_valuetype(), "Sanity check");
1863           ciValueKlass* value_klass = field->type()->as_value_klass();
1864           int flattening_offset = field->offset() - value_klass->first_field_offset();
1865           assert(field->type()->is_valuetype(), "Sanity check");
1866           scope()->set_wrote_final();
1867           scope()->set_wrote_fields();
1868           NewValueTypeInstance* new_instance = new NewValueTypeInstance(value_klass, state_before, false);
1869           _memory->new_instance(new_instance);
1870           apush(append_split(new_instance));
1871           copy_value_content(value_klass, obj, field->offset(), new_instance, value_klass->first_field_offset(),
1872                        state_before, needs_patching);
1873         }
1874       }
1875       break;
1876     }
1877     case Bytecodes::_withfield:
1878     case Bytecodes::_putfield: {
1879       Value val = pop(type);
1880       val->set_escaped();
1881       obj = apop();
1882       if (state_before == NULL) {
1883         state_before = copy_state_for_exception();
1884       }
1885       if (field->type()->basic_type() == T_BOOLEAN) {
1886         Value mask = append(new Constant(new IntConstant(1)));
1887         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1888       }
1889 
1890       if (!field->is_flattened()) {
1891         StoreField* store = new StoreField(obj, offset, field, val, false, state_before, needs_patching);
1892         if (!needs_patching) store = _memory->store(store);
1893         if (store != NULL) {
1894           append(store);
1895         }
1896       } else {
1897         assert(field->type()->is_valuetype(), "Sanity check");
1898         ciValueKlass* value_klass = field->type()->as_value_klass();
1899         int flattening_offset = field->offset() - value_klass->first_field_offset();
1900         copy_value_content(value_klass, val, value_klass->first_field_offset(), obj, field->offset(),
1901                    state_before, needs_patching);
1902       }
1903       break;
1904     }
1905     default:
1906       ShouldNotReachHere();
1907       break;
1908   }
1909 }

1911 // Baseline version of withfield, allocate every time
1912 void GraphBuilder::withfield(int field_index)
1913 {
1914   bool will_link;
1915   ciField* field_modify = stream()->get_field(will_link);
1916   ciInstanceKlass* holder = field_modify->holder();
1917   BasicType field_type = field_modify->type()->basic_type();
1918   ValueType* type = as_ValueType(field_type);
1919 
1920   // call will_link again to determine if the field is valid.
1921   const bool needs_patching = !holder->is_loaded() ||
1922                               !field_modify->will_link(method(), Bytecodes::_withfield) ||
1923                               PatchALot;
1924 
1925 
1926   scope()->set_wrote_final();
1927   scope()->set_wrote_fields();
1928 
1929   const int offset = !needs_patching ? field_modify->offset() : -1;
1930 
1931   if (!holder->is_loaded()
1932       || needs_patching /* FIXME: 8228634 - field_modify->will_link() may incorrectly return false */
1933       ) {
1934     ValueStack* state_before = copy_state_before();
1935     Value val = pop(type);
1936     Value obj = apop();
1937     apush(append_split(new WithField(state_before)));
1938     return;
1939   }
1940   ValueStack* state_before = copy_state_before();
1941 
1942   Value val = pop(type);
1943   Value obj = apop();
1944 
1945   if (!needs_patching && obj->is_optimizable_for_withfield()) {
1946     int astore_index;
1947     ciBytecodeStream s(method());
1948     s.force_bci(bci());
1949     s.next();
1950     switch (s.cur_bc()) {
1951     case Bytecodes::_astore:    astore_index = s.get_index(); break;
1952     case Bytecodes::_astore_0:  astore_index = 0; break;
1953     case Bytecodes::_astore_1:  astore_index = 1; break;
1954     case Bytecodes::_astore_2:  astore_index = 2; break;
1955     case Bytecodes::_astore_3:  astore_index = 3; break;
1956     default: astore_index = -1;
1957     }
1958 
1959     if (astore_index >= 0 && obj == state()->local_at(astore_index)) {
1960       // We have a sequence like this, where we load a value object from a local slot,
1961       // and overwrite the same local slot with a modified copy of the value object.
1962       //      defaultvalue #1 // class compiler/valhalla/valuetypes/MyValue1
1963       //      astore 9
1964       //      ...
1965       //      iload_0
1966       //      aload 9
1967       //      swap
1968       //      withfield #7 // Field x:I
1969       //      astore 9
1970       // If this object was created by defaultvalue, and has not escaped, and is not stored
1971       // in any other local slots, we can effectively treat the withfield/astore
1972       // sequence as a single putfield bytecode.
1973       push(objectType, obj);
1974       push(type, val);
1975       access_field(Bytecodes::_withfield);
1976       stream()->next(); // skip the next astore/astore_n bytecode.
1977       return;
1978     }
1979   }
1980 
1981   assert(holder->is_valuetype(), "must be a value klass");
1982   // Save the entire state and re-execute on deopt when executing withfield
1983   state_before->set_should_reexecute(true);
1984   NewValueTypeInstance* new_instance = new NewValueTypeInstance(holder->as_value_klass(), state_before, false);
1985   _memory->new_instance(new_instance);
1986   apush(append_split(new_instance));
1987 
1988   for (int i = 0; i < holder->nof_nonstatic_fields(); i++) {
1989     ciField* field = holder->nonstatic_field_at(i);
1990     int off = field->offset();
1991 
1992     if (field->offset() != offset) {
1993       if (field->is_flattened()) {
1994         assert(field->type()->is_valuetype(), "Sanity check");
1995         assert(field->type()->is_valuetype(), "Only value types can be flattened");
1996         ciValueKlass* vk = field->type()->as_value_klass();
1997         copy_value_content(vk, obj, off, new_instance, vk->first_field_offset(), state_before, needs_patching);
1998       } else {
1999         // Only load those fields who are not modified
2000         LoadField* load = new LoadField(obj, off, field, false, state_before, needs_patching);
2001         Value replacement = append(load);
2002         StoreField* store = new StoreField(new_instance, off, field, replacement, false, state_before, needs_patching);
2003         append(store);
2004       }
2005     }
2006   }
2007 
2008   // Field to modify
2009   if (field_modify->type()->basic_type() == T_BOOLEAN) {
2010     Value mask = append(new Constant(new IntConstant(1)));
2011     val = append(new LogicOp(Bytecodes::_iand, val, mask));
2012   }
2013   if (field_modify->is_flattened()) {
2014     assert(field_modify->type()->is_valuetype(), "Only value types can be flattened");
2015     ciValueKlass* vk = field_modify->type()->as_value_klass();
2016     copy_value_content(vk, val, vk->first_field_offset(), new_instance, field_modify->offset(), state_before, needs_patching);
2017   } else {
2018     StoreField* store = new StoreField(new_instance, offset, field_modify, val, false, state_before, needs_patching);
2019     append(store);
2020   }
2021 }

2028 // How many arguments do we want to profile?
2029 Values* GraphBuilder::args_list_for_profiling(ciMethod* target, int& start, bool may_have_receiver) {
2030   int n = 0;
2031   bool has_receiver = may_have_receiver && Bytecodes::has_receiver(method()->java_code_at_bci(bci()));
2032   start = has_receiver ? 1 : 0;
2033   if (profile_arguments()) {
2034     ciProfileData* data = method()->method_data()->bci_to_data(bci());
2035     if (data != NULL && (data->is_CallTypeData() || data->is_VirtualCallTypeData())) {
2036       n = data->is_CallTypeData() ? data->as_CallTypeData()->number_of_arguments() : data->as_VirtualCallTypeData()->number_of_arguments();
2037     }
2038   }
2039   // If we are inlining then we need to collect arguments to profile parameters for the target
2040   if (profile_parameters() && target != NULL) {
2041     if (target->method_data() != NULL && target->method_data()->parameters_type_data() != NULL) {
2042       // The receiver is profiled on method entry so it's included in
2043       // the number of parameters but here we're only interested in
2044       // actual arguments.
2045       n = MAX2(n, target->method_data()->parameters_type_data()->number_of_parameters() - start);
2046     }
2047   }
2048   if (n > 0) {
2049     return new Values(n);
2050   }
2051   return NULL;
2052 }

2083 void GraphBuilder::invoke(Bytecodes::Code code) {
2084   bool will_link;
2085   ciSignature* declared_signature = NULL;
2086   ciMethod*             target = stream()->get_method(will_link, &declared_signature);
2087   ciKlass*              holder = stream()->get_declared_method_holder();
2088   const Bytecodes::Code bc_raw = stream()->cur_bc_raw();
2089   assert(declared_signature != NULL, "cannot be null");
2090   assert(will_link == target->is_loaded(), "");
2091 
2092   ciInstanceKlass* klass = target->holder();
2093   assert(!target->is_loaded() || klass->is_loaded(), "loaded target must imply loaded klass");
2094 
2095   // check if CHA possible: if so, change the code to invoke_special
2096   ciInstanceKlass* calling_klass = method()->holder();
2097   ciInstanceKlass* callee_holder = ciEnv::get_instance_klass_for_declared_method_holder(holder);
2098   ciInstanceKlass* actual_recv = callee_holder;
2099 
2100   CompileLog* log = compilation()->log();
2101   if (log != NULL)
2102       log->elem("call method='%d' instr='%s'",
2103                 log->identify(target),
2104                 Bytecodes::name(code));
2105 
2106   // invoke-special-super
2107   if (bc_raw == Bytecodes::_invokespecial && !target->is_object_constructor()) {
2108     ciInstanceKlass* sender_klass =
2109           calling_klass->is_unsafe_anonymous() ? calling_klass->unsafe_anonymous_host() :
2110                                                  calling_klass;
2111     if (sender_klass->is_interface()) {
2112       int index = state()->stack_size() - (target->arg_size_no_receiver() + 1);
2113       Value receiver = state()->stack_at(index);
2114       CheckCast* c = new CheckCast(sender_klass, receiver, copy_state_before());
2115       c->set_invokespecial_receiver_check();
2116       state()->stack_at_put(index, append_split(c));
2117     }
2118   }
2119 
2120   // Some methods are obviously bindable without any type checks so
2121   // convert them directly to an invokespecial or invokestatic.
2122   if (target->is_loaded() && !target->is_abstract() && target->can_be_statically_bound()) {
2123     switch (bc_raw) {
2124     case Bytecodes::_invokevirtual:
2125       code = Bytecodes::_invokespecial;
2126       break;
2127     case Bytecodes::_invokehandle:
2128       code = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokespecial;
2129       break;
2130     default:
2131       break;
2132     }
2133   } else {
2134     if (bc_raw == Bytecodes::_invokehandle) {
2135       assert(!will_link, "should come here only for unlinked call");
2136       code = Bytecodes::_invokespecial;
2137     }
2138   }
2139 
2140   // Push appendix argument (MethodType, CallSite, etc.), if one.
2141   bool patch_for_appendix = false;
2142   int patching_appendix_arg = 0;
2143   if (Bytecodes::has_optional_appendix(bc_raw) && (!will_link || PatchALot)) {
2144     Value arg = append(new Constant(new ObjectConstant(compilation()->env()->unloaded_ciinstance()), copy_state_before()));
2145     apush(arg);
2146     patch_for_appendix = true;
2147     patching_appendix_arg = (will_link && stream()->has_appendix()) ? 0 : 1;
2148   } else if (stream()->has_appendix()) {
2149     ciObject* appendix = stream()->get_appendix();
2150     Value arg = append(new Constant(new ObjectConstant(appendix)));
2151     apush(arg);
2152   }
2153 
2154   ciMethod* cha_monomorphic_target = NULL;
2155   ciMethod* exact_target = NULL;
2156   Value better_receiver = NULL;
2157   if (UseCHA && DeoptC1 && target->is_loaded() &&
2158       !(// %%% FIXME: Are both of these relevant?
2159         target->is_method_handle_intrinsic() ||
2160         target->is_compiled_lambda_form()) &&
2161       !patch_for_appendix) {
2162     Value receiver = NULL;
2163     ciInstanceKlass* receiver_klass = NULL;
2164     bool type_is_exact = false;
2165     // try to find a precise receiver type
2166     if (will_link && !target->is_static()) {
2167       int index = state()->stack_size() - (target->arg_size_no_receiver() + 1);
2168       receiver = state()->stack_at(index);
2169       ciType* type = receiver->exact_type();
2170       if (type != NULL && type->is_loaded() &&
2171           type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {
2172         receiver_klass = (ciInstanceKlass*) type;
2173         type_is_exact = true;
2174       }
2175       if (type == NULL) {
2176         type = receiver->declared_type();
2177         if (type != NULL && type->is_loaded() &&
2178             type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {
2179           receiver_klass = (ciInstanceKlass*) type;
2180           if (receiver_klass->is_leaf_type() && !receiver_klass->is_final()) {
2181             // Insert a dependency on this type since
2182             // find_monomorphic_target may assume it's already done.
2183             dependency_recorder()->assert_leaf_type(receiver_klass);
2184             type_is_exact = true;
2185           }
2186         }
2187       }
2188     }
2189     if (receiver_klass != NULL && type_is_exact &&
2190         receiver_klass->is_loaded() && code != Bytecodes::_invokespecial) {
2191       // If we have the exact receiver type we can bind directly to
2192       // the method to call.
2193       exact_target = target->resolve_invoke(calling_klass, receiver_klass);
2194       if (exact_target != NULL) {
2195         target = exact_target;
2196         code = Bytecodes::_invokespecial;
2197       }
2198     }
2199     if (receiver_klass != NULL &&
2200         receiver_klass->is_subtype_of(actual_recv) &&
2201         actual_recv->is_initialized()) {
2202       actual_recv = receiver_klass;
2203     }
2204 
2205     if ((code == Bytecodes::_invokevirtual && callee_holder->is_initialized()) ||
2206         (code == Bytecodes::_invokeinterface && callee_holder->is_initialized() && !actual_recv->is_interface())) {
2207       // Use CHA on the receiver to select a more precise method.
2208       cha_monomorphic_target = target->find_monomorphic_target(calling_klass, callee_holder, actual_recv);
2209     } else if (code == Bytecodes::_invokeinterface && callee_holder->is_loaded() && receiver != NULL) {
2210       assert(callee_holder->is_interface(), "invokeinterface to non interface?");
2211       // If there is only one implementor of this interface then we
2212       // may be able bind this invoke directly to the implementing
2213       // klass but we need both a dependence on the single interface
2214       // and on the method we bind to.  Additionally since all we know
2215       // about the receiver type is the it's supposed to implement the
2216       // interface we have to insert a check that it's the class we
2217       // expect.  Interface types are not checked by the verifier so
2218       // they are roughly equivalent to Object.
2219       // The number of implementors for declared_interface is less or
2220       // equal to the number of implementors for target->holder() so
2221       // if number of implementors of target->holder() == 1 then
2222       // number of implementors for decl_interface is 0 or 1. If
2223       // it's 0 then no class implements decl_interface and there's
2224       // no point in inlining.
2225       ciInstanceKlass* declared_interface = callee_holder;
2226       ciInstanceKlass* singleton = declared_interface->unique_implementor();
2227       if (singleton != NULL &&
2228           (!target->is_default_method() || target->is_overpass()) /* CHA doesn't support default methods yet. */ ) {
2229         assert(singleton != declared_interface, "not a unique implementor");
2230         cha_monomorphic_target = target->find_monomorphic_target(calling_klass, declared_interface, singleton);
2231         if (cha_monomorphic_target != NULL) {
2232           if (cha_monomorphic_target->holder() != compilation()->env()->Object_klass()) {
2233             // If CHA is able to bind this invoke then update the class
2234             // to match that class, otherwise klass will refer to the
2235             // interface.
2236             klass = cha_monomorphic_target->holder();
2237             actual_recv = declared_interface;
2238 
2239             // insert a check it's really the expected class.
2240             CheckCast* c = new CheckCast(klass, receiver, copy_state_for_exception());
2241             c->set_incompatible_class_change_check();
2242             c->set_direct_compare(klass->is_final());
2243             // pass the result of the checkcast so that the compiler has
2244             // more accurate type info in the inlinee
2245             better_receiver = append_split(c);
2246           } else {
2247             cha_monomorphic_target = NULL; // subtype check against Object is useless
2248           }
2249         }
2250       }
2251     }
2252   }
2253 
2254   if (cha_monomorphic_target != NULL) {
2255     assert(!target->can_be_statically_bound() || target == cha_monomorphic_target, "");
2256     assert(!cha_monomorphic_target->is_abstract(), "");
2257     if (!cha_monomorphic_target->can_be_statically_bound(actual_recv)) {
2258       // If we inlined because CHA revealed only a single target method,
2259       // then we are dependent on that target method not getting overridden
2260       // by dynamic class loading.  Be sure to test the "static" receiver
2261       // dest_method here, as opposed to the actual receiver, which may
2262       // falsely lead us to believe that the receiver is final or private.
2263       dependency_recorder()->assert_unique_concrete_method(actual_recv, cha_monomorphic_target);
2264     }
2265     code = Bytecodes::_invokespecial;
2266   }
2267 
2268   // check if we could do inlining
2269   if (!PatchALot && Inline && target->is_loaded() &&
2270       (klass->is_initialized() || (klass->is_interface() && target->holder()->is_initialized()))
2271       && !patch_for_appendix) {
2272     // callee is known => check if we have static binding
2273     if (code == Bytecodes::_invokestatic  ||
2274         code == Bytecodes::_invokespecial ||
2275         (code == Bytecodes::_invokevirtual && target->is_final_method()) ||
2276         code == Bytecodes::_invokedynamic) {
2277       ciMethod* inline_target = (cha_monomorphic_target != NULL) ? cha_monomorphic_target : target;
2278       // static binding => check if callee is ok
2279       bool success = try_inline(inline_target, (cha_monomorphic_target != NULL) || (exact_target != NULL), false, code, better_receiver);
2280 
2281       CHECK_BAILOUT();
2282       clear_inline_bailout();
2283 
2284       if (success) {
2285         // Register dependence if JVMTI has either breakpoint
2286         // setting or hotswapping of methods capabilities since they may
2287         // cause deoptimization.
2288         if (compilation()->env()->jvmti_can_hotswap_or_post_breakpoint()) {
2289           dependency_recorder()->assert_evol_method(inline_target);
2290         }
2291         return;
2292       }
2293     } else {
2294       print_inlining(target, "no static binding", /*success*/ false);
2295     }
2296   } else {
2297     print_inlining(target, "not inlineable", /*success*/ false);
2298   }
2299 
2300   // If we attempted an inline which did not succeed because of a
2301   // bailout during construction of the callee graph, the entire
2302   // compilation has to be aborted. This is fairly rare and currently
2303   // seems to only occur for jasm-generated classes which contain
2304   // jsr/ret pairs which are not associated with finally clauses and
2305   // do not have exception handlers in the containing method, and are
2306   // therefore not caught early enough to abort the inlining without
2307   // corrupting the graph. (We currently bail out with a non-empty
2308   // stack at a ret in these situations.)
2309   CHECK_BAILOUT();
2310 
2311   // inlining not successful => standard invoke
2312   ValueType* result_type = as_ValueType(declared_signature->return_type());
2313   ValueStack* state_before = copy_state_exhandling();
2314 
2315   // The bytecode (code) might change in this method so we are checking this very late.
2316   const bool has_receiver =
2317     code == Bytecodes::_invokespecial   ||
2318     code == Bytecodes::_invokevirtual   ||
2319     code == Bytecodes::_invokeinterface;
2320   Values* args = state()->pop_arguments(target->arg_size_no_receiver() + patching_appendix_arg);
2321   Value recv = has_receiver ? apop() : NULL;
2322   int vtable_index = Method::invalid_vtable_index;
2323 
2324 #ifdef SPARC
2325   // Currently only supported on Sparc.
2326   // The UseInlineCaches only controls dispatch to invokevirtuals for
2327   // loaded classes which we weren't able to statically bind.
2328   if (!UseInlineCaches && target->is_loaded() && code == Bytecodes::_invokevirtual
2329       && !target->can_be_statically_bound()) {
2330     // Find a vtable index if one is available
2331     // For arrays, callee_holder is Object. Resolving the call with
2332     // Object would allow an illegal call to finalize() on an
2333     // array. We use holder instead: illegal calls to finalize() won't
2334     // be compiled as vtable calls (IC call resolution will catch the
2335     // illegal call) and the few legal calls on array types won't be
2336     // either.
2337     vtable_index = target->resolve_vtable_index(calling_klass, holder);
2338   }
2339 #endif
2340 
2341   // A null check is required here (when there is a receiver) for any of the following cases
2342   // - invokespecial, always need a null check.
2343   // - invokevirtual, when the target is final and loaded. Calls to final targets will become optimized
2344   //   and require null checking. If the target is loaded a null check is emitted here.
2345   //   If the target isn't loaded the null check must happen after the call resolution. We achieve that
2346   //   by using the target methods unverified entry point (see CompiledIC::compute_monomorphic_entry).
2347   //   (The JVM specification requires that LinkageError must be thrown before a NPE. An unloaded target may
2348   //   potentially fail, and can't have the null check before the resolution.)
2349   // - A call that will be profiled. (But we can't add a null check when the target is unloaded, by the same
2350   //   reason as above, so calls with a receiver to unloaded targets can't be profiled.)
2351   //
2352   // Normal invokevirtual will perform the null check during lookup
2353 
2354   bool need_null_check = (code == Bytecodes::_invokespecial) ||
2355       (target->is_loaded() && (target->is_final_method() || (is_profiling() && profile_calls())));
2356 
2357   if (need_null_check) {
2358     if (recv != NULL) {
2359       null_check(recv);
2360     }
2361 
2362     if (is_profiling()) {
2363       // Note that we'd collect profile data in this method if we wanted it.
2364       compilation()->set_would_profile(true);
2365 
2366       if (profile_calls()) {
2367         assert(cha_monomorphic_target == NULL || exact_target == NULL, "both can not be set");
2368         ciKlass* target_klass = NULL;
2369         if (cha_monomorphic_target != NULL) {
2370           target_klass = cha_monomorphic_target->holder();
2371         } else if (exact_target != NULL) {
2372           target_klass = exact_target->holder();
2373         }
2374         profile_call(target, recv, target_klass, collect_args_for_profiling(args, NULL, false), false);
2375       }
2376     }
2377   }
2378 
2379   if (recv != NULL) {
2380     recv->set_escaped();
2381   }
2382   for (int i=0; i<args->length(); i++) {
2383     args->at(0)->set_escaped();
2384   }
2385 
2386   Invoke* result = new Invoke(code, result_type, recv, args, vtable_index, target, state_before,
2387                               declared_signature->returns_never_null());
2388   // push result
2389   append_split(result);
2390 
2391   if (result_type != voidType) {
2392     if (method()->is_strict()) {
2393       push(result_type, round_fp(result));
2394     } else {
2395       push(result_type, result);
2396     }
2397   }
2398   if (profile_return() && result_type->is_object_kind()) {
2399     profile_return_type(result, target);
2400   }
2401 }

2584 Instruction* GraphBuilder::append_with_bci(Instruction* instr, int bci) {
2585   Canonicalizer canon(compilation(), instr, bci);
2586   Instruction* i1 = canon.canonical();
2587   if (i1->is_linked() || !i1->can_be_linked()) {
2588     // Canonicalizer returned an instruction which was already
2589     // appended so simply return it.
2590     return i1;
2591   }
2592 
2593   if (UseLocalValueNumbering) {
2594     // Lookup the instruction in the ValueMap and add it to the map if
2595     // it's not found.
2596     Instruction* i2 = vmap()->find_insert(i1);
2597     if (i2 != i1) {
2598       // found an entry in the value map, so just return it.
2599       assert(i2->is_linked(), "should already be linked");
2600       return i2;
2601     }
2602     ValueNumberingEffects vne(vmap());
2603     i1->visit(&vne);
2604   }
2605 
2606   // i1 was not eliminated => append it
2607   assert(i1->next() == NULL, "shouldn't already be linked");
2608   _last = _last->set_next(i1, canon.bci());
2609 
2610   if (++_instruction_count >= InstructionCountCutoff && !bailed_out()) {
2611     // set the bailout state but complete normal processing.  We
2612     // might do a little more work before noticing the bailout so we
2613     // want processing to continue normally until it's noticed.
2614     bailout("Method and/or inlining is too large");
2615   }
2616 
2617 #ifndef PRODUCT
2618   if (PrintIRDuringConstruction) {
2619     InstructionPrinter ip;
2620     ip.print_line(i1);
2621     if (Verbose) {
2622       state()->print();
2623     }
2624   }
2625 #endif
2626 
2627   // save state after modification of operand stack for StateSplit instructions
2628   StateSplit* s = i1->as_StateSplit();
2629   if (s != NULL) {
2630     if (EliminateFieldAccess) {
2631       Intrinsic* intrinsic = s->as_Intrinsic();
2632       if (s->as_Invoke() != NULL || (intrinsic && !intrinsic->preserves_state())) {
2633         _memory->kill();
2634       }
2635     }
2636     s->set_state(state()->copy(ValueStack::StateAfter, canon.bci()));
2637   }
2638 
2639   // set up exception handlers for this instruction if necessary
2640   if (i1->can_trap()) {
2641     i1->set_exception_handlers(handle_exception(i1));
2642     assert(i1->exception_state() != NULL || !i1->needs_exception_state() || bailed_out(), "handle_exception must set exception state");
2643   }
2644   return i1;
2645 }

2679 XHandlers* GraphBuilder::handle_exception(Instruction* instruction) {
2680   if (!has_handler() && (!instruction->needs_exception_state() || instruction->exception_state() != NULL)) {
2681     assert(instruction->exception_state() == NULL
2682            || instruction->exception_state()->kind() == ValueStack::EmptyExceptionState
2683            || (instruction->exception_state()->kind() == ValueStack::ExceptionState && _compilation->env()->should_retain_local_variables()),
2684            "exception_state should be of exception kind");
2685     return new XHandlers();
2686   }
2687 
2688   XHandlers*  exception_handlers = new XHandlers();
2689   ScopeData*  cur_scope_data = scope_data();
2690   ValueStack* cur_state = instruction->state_before();
2691   ValueStack* prev_state = NULL;
2692   int scope_count = 0;
2693 
2694   assert(cur_state != NULL, "state_before must be set");
2695   do {
2696     int cur_bci = cur_state->bci();
2697     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2698     assert(cur_bci == SynchronizationEntryBCI || cur_bci == cur_scope_data->stream()->cur_bci(), "invalid bci");
2699 
2700     // join with all potential exception handlers
2701     XHandlers* list = cur_scope_data->xhandlers();
2702     const int n = list->length();
2703     for (int i = 0; i < n; i++) {
2704       XHandler* h = list->handler_at(i);
2705       if (h->covers(cur_bci)) {
2706         // h is a potential exception handler => join it
2707         compilation()->set_has_exception_handlers(true);
2708 
2709         BlockBegin* entry = h->entry_block();
2710         if (entry == block()) {
2711           // It's acceptable for an exception handler to cover itself
2712           // but we don't handle that in the parser currently.  It's
2713           // very rare so we bailout instead of trying to handle it.
2714           BAILOUT_("exception handler covers itself", exception_handlers);
2715         }
2716         assert(entry->bci() == h->handler_bci(), "must match");
2717         assert(entry->bci() == -1 || entry == cur_scope_data->block_at(entry->bci()), "blocks must correspond");
2718 
2719         // previously this was a BAILOUT, but this is not necessary
2720         // now because asynchronous exceptions are not handled this way.
2721         assert(entry->state() == NULL || cur_state->total_locks_size() == entry->state()->total_locks_size(), "locks do not match");
2722 
2723         // xhandler start with an empty expression stack
2724         if (cur_state->stack_size() != 0) {
2725           cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2726         }
2727         if (instruction->exception_state() == NULL) {
2728           instruction->set_exception_state(cur_state);
2729         }
2730 
2731         // Note: Usually this join must work. However, very
2732         // complicated jsr-ret structures where we don't ret from
2733         // the subroutine can cause the objects on the monitor
2734         // stacks to not match because blocks can be parsed twice.
2735         // The only test case we've seen so far which exhibits this
2736         // problem is caught by the infinite recursion test in
2737         // GraphBuilder::jsr() if the join doesn't work.
2738         if (!entry->try_merge(cur_state)) {
2739           BAILOUT_("error while joining with exception handler, prob. due to complicated jsr/rets", exception_handlers);
2740         }
2741 
2742         // add current state for correct handling of phi functions at begin of xhandler
2743         int phi_operand = entry->add_exception_state(cur_state);
2744 
2745         // add entry to the list of xhandlers of this block
2746         _block->add_exception_handler(entry);
2747 
2748         // add back-edge from xhandler entry to this block
2749         if (!entry->is_predecessor(_block)) {
2750           entry->add_predecessor(_block);
2751         }
2752 
2753         // clone XHandler because phi_operand and scope_count can not be shared
2754         XHandler* new_xhandler = new XHandler(h);
2755         new_xhandler->set_phi_operand(phi_operand);
2756         new_xhandler->set_scope_count(scope_count);
2757         exception_handlers->append(new_xhandler);
2758 
2759         // fill in exception handler subgraph lazily
2760         assert(!entry->is_set(BlockBegin::was_visited_flag), "entry must not be visited yet");
2761         cur_scope_data->add_to_work_list(entry);
2762 
2763         // stop when reaching catchall
2764         if (h->catch_type() == 0) {
2765           return exception_handlers;
2766         }
2767       }
2768     }
2769 
2770     if (exception_handlers->length() == 0) {
2771       // This scope and all callees do not handle exceptions, so the local
2772       // variables of this scope are not needed. However, the scope itself is
2773       // required for a correct exception stack trace -> clear out the locals.
2774       if (_compilation->env()->should_retain_local_variables()) {
2775         cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2776       } else {
2777         cur_state = cur_state->copy(ValueStack::EmptyExceptionState, cur_state->bci());
2778       }
2779       if (prev_state != NULL) {
2780         prev_state->set_caller_state(cur_state);
2781       }
2782       if (instruction->exception_state() == NULL) {
2783         instruction->set_exception_state(cur_state);
2784       }
2785     }
2786 
2787     // Set up iteration for next time.
2788     // If parsing a jsr, do not grab exception handlers from the
2789     // parent scopes for this method (already got them, and they
2790     // needed to be cloned)
2791 
2792     while (cur_scope_data->parsing_jsr()) {
2793       cur_scope_data = cur_scope_data->parent();
2794     }
2795 
2796     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2797     assert(cur_state->locks_size() == 0 || cur_state->locks_size() == 1, "unlocking must be done in a catchall exception handler");
2798 
2799     prev_state = cur_state;
2800     cur_state = cur_state->caller_state();
2801     cur_scope_data = cur_scope_data->parent();
2802     scope_count++;
2803   } while (cur_scope_data != NULL);
2804 
2805   return exception_handlers;
2806 }

2837     // already tried to simplify phi before
2838     return phi;
2839   } else if (phi->is_set(Phi::visited)) {
2840     // break cycles in phi functions
2841     return phi;
2842   } else if (phi->type()->is_illegal()) {
2843     // illegal phi functions are ignored anyway
2844     return phi;
2845 
2846   } else {
2847     // mark phi function as processed to break cycles in phi functions
2848     phi->set(Phi::visited);
2849 
2850     // simplify x = [y, x] and x = [y, y] to y
2851     Value subst = NULL;
2852     int opd_count = phi->operand_count();
2853     for (int i = 0; i < opd_count; i++) {
2854       Value opd = phi->operand_at(i);
2855       assert(opd != NULL, "Operand must exist!");
2856 
2857       if (opd->type()->is_illegal()) {
2858         // if one operand is illegal, the entire phi function is illegal
2859         phi->make_illegal();
2860         phi->clear(Phi::visited);
2861         return phi;
2862       }
2863 
2864       Value new_opd = simplify(opd);
2865       assert(new_opd != NULL, "Simplified operand must exist!");
2866 
2867       if (new_opd != phi && new_opd != subst) {
2868         if (subst == NULL) {
2869           subst = new_opd;
2870         } else {
2871           // no simplification possible
2872           phi->set(Phi::cannot_simplify);
2873           phi->clear(Phi::visited);
2874           return phi;
2875         }
2876       }
2877     }
2878 
2879     // sucessfully simplified phi function
2880     assert(subst != NULL, "illegal phi function");
2881     _has_substitutions = true;
2882     phi->clear(Phi::visited);
2883     phi->set_subst(subst);
2884 
2885 #ifndef PRODUCT
2886     if (PrintPhiFunctions) {
2887       tty->print_cr("simplified phi function %c%d to %c%d (Block B%d)", phi->type()->tchar(), phi->id(), subst->type()->tchar(), subst->id(), phi->block()->block_id());
2888     }
2889 #endif
2890 
2891     return subst;
2892   }
2893 }

3237 void GraphBuilder::iterate_all_blocks(bool start_in_current_block_for_inlining) {
3238   do {
3239     if (start_in_current_block_for_inlining && !bailed_out()) {
3240       iterate_bytecodes_for_block(0);
3241       start_in_current_block_for_inlining = false;
3242     } else {
3243       BlockBegin* b;
3244       while ((b = scope_data()->remove_from_work_list()) != NULL) {
3245         if (!b->is_set(BlockBegin::was_visited_flag)) {
3246           if (b->is_set(BlockBegin::osr_entry_flag)) {
3247             // we're about to parse the osr entry block, so make sure
3248             // we setup the OSR edge leading into this block so that
3249             // Phis get setup correctly.
3250             setup_osr_entry_block();
3251             // this is no longer the osr entry block, so clear it.
3252             b->clear(BlockBegin::osr_entry_flag);
3253           }
3254           b->set(BlockBegin::was_visited_flag);
3255           connect_to_end(b);
3256         }
3257       }
3258     }
3259   } while (!bailed_out() && !scope_data()->is_work_list_empty());
3260 }

3346 BlockBegin* GraphBuilder::setup_start_block(int osr_bci, BlockBegin* std_entry, BlockBegin* osr_entry, ValueStack* state) {
3347   BlockBegin* start = new BlockBegin(0);
3348 
3349   // This code eliminates the empty start block at the beginning of
3350   // each method.  Previously, each method started with the
3351   // start-block created below, and this block was followed by the
3352   // header block that was always empty.  This header block is only
3353   // necesary if std_entry is also a backward branch target because
3354   // then phi functions may be necessary in the header block.  It's
3355   // also necessary when profiling so that there's a single block that
3356   // can increment the interpreter_invocation_count.
3357   BlockBegin* new_header_block;
3358   if (std_entry->number_of_preds() > 0 || count_invocations() || count_backedges()) {
3359     new_header_block = header_block(std_entry, BlockBegin::std_entry_flag, state);
3360   } else {
3361     new_header_block = std_entry;
3362   }
3363 
3364   // setup start block (root for the IR graph)
3365   Base* base =
3366     new Base(
3367       new_header_block,
3368       osr_entry
3369     );
3370   start->set_next(base, 0);
3371   start->set_end(base);
3372   // create & setup state for start block
3373   start->set_state(state->copy(ValueStack::StateAfter, std_entry->bci()));

3375 
3376   if (base->std_entry()->state() == NULL) {
3377     // setup states for header blocks
3378     base->std_entry()->merge(state);
3379   }
3380 
3381   assert(base->std_entry()->state() != NULL, "");
3382   return start;

3386 void GraphBuilder::setup_osr_entry_block() {
3387   assert(compilation()->is_osr_compile(), "only for osrs");
3388 
3389   int osr_bci = compilation()->osr_bci();
3390   ciBytecodeStream s(method());
3391   s.reset_to_bci(osr_bci);
3392   s.next();
3393   scope_data()->set_stream(&s);
3394 
3395   // create a new block to be the osr setup code
3396   _osr_entry = new BlockBegin(osr_bci);
3397   _osr_entry->set(BlockBegin::osr_entry_flag);
3398   _osr_entry->set_depth_first_number(0);
3399   BlockBegin* target = bci2block()->at(osr_bci);
3400   assert(target != NULL && target->is_set(BlockBegin::osr_entry_flag), "must be there");
3401   // the osr entry has no values for locals
3402   ValueStack* state = target->state()->copy();
3403   _osr_entry->set_state(state);
3404 
3405   kill_all();
3406   _block = _osr_entry;
3407   _state = _osr_entry->state()->copy();
3408   assert(_state->bci() == osr_bci, "mismatch");
3409   _last  = _osr_entry;
3410   Value e = append(new OsrEntry());
3411   e->set_needs_null_check(false);
3412 
3413   // OSR buffer is
3414   //
3415   // locals[nlocals-1..0]
3416   // monitors[number_of_locks-1..0]
3417   //
3418   // locals is a direct copy of the interpreter frame so in the osr buffer
3419   // so first slot in the local array is the last local from the interpreter
3420   // and last slot is local[0] (receiver) from the interpreter
3421   //
3422   // Similarly with locks. The first lock slot in the osr buffer is the nth lock
3423   // from the interpreter frame, the nth lock slot in the osr buffer is 0th lock
3424   // in the interpreter frame (the method lock if a sync method)
3425 
3426   // Initialize monitors in the compiled activation.
3427 
3428   int index;
3429   Value local;
3430 
3431   // find all the locals that the interpreter thinks contain live oops
3432   const ResourceBitMap live_oops = method()->live_local_oops_at_bci(osr_bci);
3433 
3434   // compute the offset into the locals so that we can treat the buffer
3435   // as if the locals were still in the interpreter frame
3436   int locals_offset = BytesPerWord * (method()->max_locals() - 1);
3437   for_each_local_value(state, index, local) {
3438     int offset = locals_offset - (index + local->type()->size() - 1) * BytesPerWord;
3439     Value get;
3440     if (local->type()->is_object_kind() && !live_oops.at(index)) {
3441       // The interpreter thinks this local is dead but the compiler
3442       // doesn't so pretend that the interpreter passed in null.
3443       get = append(new Constant(objectNull));
3444     } else {
3445       get = append(new UnsafeGetRaw(as_BasicType(local->type()), e,
3446                                     append(new Constant(new IntConstant(offset))),
3447                                     0,
3448                                     true /*unaligned*/, true /*wide*/));
3449     }
3450     _state->store_local(index, get);
3451   }
3452 
3453   // the storage for the OSR buffer is freed manually in the LIRGenerator.
3454 
3455   assert(state->caller_state() == NULL, "should be top scope");
3456   state->clear_locals();
3457   Goto* g = new Goto(target, false);
3458   append(g);
3459   _osr_entry->set_end(g);
3460   target->merge(_osr_entry->end()->state());
3461 
3462   scope_data()->set_stream(NULL);
3463 }

3466 ValueStack* GraphBuilder::state_at_entry() {
3467   ValueStack* state = new ValueStack(scope(), NULL);
3468 
3469   // Set up locals for receiver
3470   int idx = 0;
3471   if (!method()->is_static()) {
3472     // we should always see the receiver
3473     state->store_local(idx, new Local(method()->holder(), objectType, idx,
3474              /*receiver*/ true, /*never_null*/ method()->holder()->is_value_array_klass()));
3475     idx = 1;
3476   }
3477 
3478   // Set up locals for incoming arguments
3479   ciSignature* sig = method()->signature();
3480   for (int i = 0; i < sig->count(); i++) {
3481     ciType* type = sig->type_at(i);
3482     BasicType basic_type = type->basic_type();
3483     // don't allow T_ARRAY to propagate into locals types
3484     if (is_reference_type(basic_type)) basic_type = T_OBJECT;
3485     ValueType* vt = as_ValueType(basic_type);
3486     state->store_local(idx, new Local(type, vt, idx, false, sig->is_never_null_at(i)));
3487     idx += type->size();
3488   }
3489 
3490   // lock synchronized method
3491   if (method()->is_synchronized()) {
3492     state->lock(NULL);
3493   }
3494 
3495   return state;
3496 }

3499 GraphBuilder::GraphBuilder(Compilation* compilation, IRScope* scope)
3500   : _scope_data(NULL)
3501   , _compilation(compilation)
3502   , _memory(new MemoryBuffer())
3503   , _inline_bailout_msg(NULL)
3504   , _instruction_count(0)
3505   , _osr_entry(NULL)
3506 {
3507   int osr_bci = compilation->osr_bci();
3508 
3509   // determine entry points and bci2block mapping
3510   BlockListBuilder blm(compilation, scope, osr_bci);
3511   CHECK_BAILOUT();
3512 
3513   BlockList* bci2block = blm.bci2block();
3514   BlockBegin* start_block = bci2block->at(0);
3515 
3516   push_root_scope(scope, bci2block, start_block);
3517 
3518   // setup state for std entry
3519   _initial_state = state_at_entry();
3520   start_block->merge(_initial_state);
3521 
3522   // complete graph
3523   _vmap        = new ValueMap();
3524   switch (scope->method()->intrinsic_id()) {
3525   case vmIntrinsics::_dabs          : // fall through
3526   case vmIntrinsics::_dsqrt         : // fall through
3527   case vmIntrinsics::_dsin          : // fall through
3528   case vmIntrinsics::_dcos          : // fall through
3529   case vmIntrinsics::_dtan          : // fall through
3530   case vmIntrinsics::_dlog          : // fall through
3531   case vmIntrinsics::_dlog10        : // fall through
3532   case vmIntrinsics::_dexp          : // fall through
3533   case vmIntrinsics::_dpow          : // fall through
3534     {
3535       // Compiles where the root method is an intrinsic need a special
3536       // compilation environment because the bytecodes for the method
3537       // shouldn't be parsed during the compilation, only the special
3538       // Intrinsic node should be emitted.  If this isn't done the the
3539       // code for the inlined version will be different than the root
3540       // compiled version which could lead to monotonicity problems on
3541       // intel.
3542       if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3543         BAILOUT("failed to inline intrinsic, method not annotated");
3544       }
3545 
3546       // Set up a stream so that appending instructions works properly.
3547       ciBytecodeStream s(scope->method());
3548       s.reset_to_bci(0);
3549       scope_data()->set_stream(&s);
3550       s.next();
3551 
3552       // setup the initial block state
3553       _block = start_block;
3554       _state = start_block->state()->copy_for_parsing();
3555       _last  = start_block;
3556       load_local(doubleType, 0);
3557       if (scope->method()->intrinsic_id() == vmIntrinsics::_dpow) {
3558         load_local(doubleType, 2);
3559       }
3560 
3561       // Emit the intrinsic node.
3562       bool result = try_inline_intrinsics(scope->method());
3563       if (!result) BAILOUT("failed to inline intrinsic");
3564       method_return(dpop());
3565 
3566       // connect the begin and end blocks and we're all done.
3567       BlockEnd* end = last()->as_BlockEnd();
3568       block()->set_end(end);
3569       break;
3570     }
3571 
3572   case vmIntrinsics::_Reference_get:
3573     {
3574       {
3575         // With java.lang.ref.reference.get() we must go through the
3576         // intrinsic - when G1 is enabled - even when get() is the root
3577         // method of the compile so that, if necessary, the value in
3578         // the referent field of the reference object gets recorded by
3579         // the pre-barrier code.
3580         // Specifically, if G1 is enabled, the value in the referent
3581         // field is recorded by the G1 SATB pre barrier. This will
3582         // result in the referent being marked live and the reference
3583         // object removed from the list of discovered references during
3584         // reference processing.
3585         if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3586           BAILOUT("failed to inline intrinsic, method not annotated");
3587         }
3588 
3589         // Also we need intrinsic to prevent commoning reads from this field
3590         // across safepoint since GC can change its value.
3591 
3592         // Set up a stream so that appending instructions works properly.
3593         ciBytecodeStream s(scope->method());
3594         s.reset_to_bci(0);
3595         scope_data()->set_stream(&s);
3596         s.next();
3597 
3598         // setup the initial block state
3599         _block = start_block;
3600         _state = start_block->state()->copy_for_parsing();
3601         _last  = start_block;
3602         load_local(objectType, 0);
3603 
3604         // Emit the intrinsic node.
3605         bool result = try_inline_intrinsics(scope->method());
3606         if (!result) BAILOUT("failed to inline intrinsic");
3607         method_return(apop());
3608 
3609         // connect the begin and end blocks and we're all done.
3610         BlockEnd* end = last()->as_BlockEnd();
3611         block()->set_end(end);
3612         break;
3613       }
3614       // Otherwise, fall thru
3615     }
3616 
3617   default:
3618     scope_data()->add_to_work_list(start_block);
3619     iterate_all_blocks();
3620     break;
3621   }
3622   CHECK_BAILOUT();
3623 
3624   _start = setup_start_block(osr_bci, start_block, _osr_entry, _initial_state);
3625 
3626   eliminate_redundant_phis(_start);
3627 
3628   NOT_PRODUCT(if (PrintValueNumbering && Verbose) print_stats());
3629   // for osr compile, bailout if some requirements are not fulfilled
3630   if (osr_bci != -1) {
3631     BlockBegin* osr_block = blm.bci2block()->at(osr_bci);
3632     if (!osr_block->is_set(BlockBegin::was_visited_flag)) {
3633       BAILOUT("osr entry must have been visited for osr compile");
3634     }
3635 
3636     // check if osr entry point has empty stack - we cannot handle non-empty stacks at osr entry points
3637     if (!osr_block->state()->stack_is_empty()) {
3638       BAILOUT("stack not empty at OSR entry point");
3639     }
3640   }
3641 #ifndef PRODUCT
3642   if (PrintCompilation && Verbose) tty->print_cr("Created %d Instructions", _instruction_count);
3643 #endif
3644 }

3691 bool GraphBuilder::try_inline(ciMethod* callee, bool holder_known, bool ignore_return, Bytecodes::Code bc, Value receiver) {
3692   const char* msg = NULL;
3693 
3694   // clear out any existing inline bailout condition
3695   clear_inline_bailout();
3696 
3697   // exclude methods we don't want to inline
3698   msg = should_not_inline(callee);
3699   if (msg != NULL) {
3700     print_inlining(callee, msg, /*success*/ false);
3701     return false;
3702   }
3703 
3704   // method handle invokes
3705   if (callee->is_method_handle_intrinsic()) {
3706     if (try_method_handle_inline(callee, ignore_return)) {
3707       if (callee->has_reserved_stack_access()) {
3708         compilation()->set_has_reserved_stack_access(true);
3709       }
3710       return true;
3711     }
3712     return false;
3713   }
3714 
3715   // handle intrinsics
3716   if (callee->intrinsic_id() != vmIntrinsics::_none &&
3717       (CheckIntrinsics ? callee->intrinsic_candidate() : true)) {
3718     if (try_inline_intrinsics(callee, ignore_return)) {
3719       print_inlining(callee, "intrinsic");
3720       if (callee->has_reserved_stack_access()) {
3721         compilation()->set_has_reserved_stack_access(true);
3722       }
3723       return true;
3724     }
3725     // try normal inlining
3726   }
3727 
3728   // certain methods cannot be parsed at all
3729   msg = check_can_parse(callee);
3730   if (msg != NULL) {
3731     print_inlining(callee, msg, /*success*/ false);
3732     return false;
3733   }
3734 
3735   // If bytecode not set use the current one.
3736   if (bc == Bytecodes::_illegal) {
3737     bc = code();
3738   }
3739   if (try_inline_full(callee, holder_known, ignore_return, bc, receiver)) {
3740     if (callee->has_reserved_stack_access()) {
3741       compilation()->set_has_reserved_stack_access(true);
3742     }
3743     return true;
3744   }
3745 
3746   // Entire compilation could fail during try_inline_full call.
3747   // In that case printing inlining decision info is useless.
3748   if (!bailed_out())
3749     print_inlining(callee, _inline_bailout_msg, /*success*/ false);
3750 
3751   return false;
3752 }

3899 bool GraphBuilder::try_inline_jsr(int jsr_dest_bci) {
3900   // Introduce a new callee continuation point - all Ret instructions
3901   // will be replaced with Gotos to this point.
3902   BlockBegin* cont = block_at(next_bci());
3903   assert(cont != NULL, "continuation must exist (BlockListBuilder starts a new block after a jsr");
3904 
3905   // Note: can not assign state to continuation yet, as we have to
3906   // pick up the state from the Ret instructions.
3907 
3908   // Push callee scope
3909   push_scope_for_jsr(cont, jsr_dest_bci);
3910 
3911   // Temporarily set up bytecode stream so we can append instructions
3912   // (only using the bci of this stream)
3913   scope_data()->set_stream(scope_data()->parent()->stream());
3914 
3915   BlockBegin* jsr_start_block = block_at(jsr_dest_bci);
3916   assert(jsr_start_block != NULL, "jsr start block must exist");
3917   assert(!jsr_start_block->is_set(BlockBegin::was_visited_flag), "should not have visited jsr yet");
3918   Goto* goto_sub = new Goto(jsr_start_block, false);
3919   // Must copy state to avoid wrong sharing when parsing bytecodes
3920   assert(jsr_start_block->state() == NULL, "should have fresh jsr starting block");
3921   jsr_start_block->set_state(copy_state_before_with_bci(jsr_dest_bci));
3922   append(goto_sub);
3923   _block->set_end(goto_sub);
3924   _last = _block = jsr_start_block;
3925 
3926   // Clear out bytecode stream
3927   scope_data()->set_stream(NULL);
3928 
3929   scope_data()->add_to_work_list(jsr_start_block);
3930 
3931   // Ready to resume parsing in subroutine
3932   iterate_all_blocks();
3933 
3934   // If we bailed out during parsing, return immediately (this is bad news)
3935   CHECK_BAILOUT_(false);
3936 
3937   // Detect whether the continuation can actually be reached. If not,
3938   // it has not had state set by the join() operations in
3939   // iterate_bytecodes_for_block()/ret() and we should not touch the
3940   // iteration state. The calling activation of
3941   // iterate_bytecodes_for_block will then complete normally.
3942   if (cont->state() != NULL) {
3943     if (!cont->is_set(BlockBegin::was_visited_flag)) {
3944       // add continuation to work list instead of parsing it immediately
3945       scope_data()->parent()->add_to_work_list(cont);
3946     }
3947   }
3948 
3949   assert(jsr_continuation() == cont, "continuation must not have changed");
3950   assert(!jsr_continuation()->is_set(BlockBegin::was_visited_flag) ||
3951          jsr_continuation()->is_set(BlockBegin::parser_loop_header_flag),
3952          "continuation can only be visited in case of backward branches");
3953   assert(_last && _last->as_BlockEnd(), "block must have end");
3954 
3955   // continuation is in work list, so end iteration of current block
3956   _skip_block = true;
3957   pop_scope_for_jsr();
3958 
3959   return true;
3960 }

4049 bool GraphBuilder::try_inline_full(ciMethod* callee, bool holder_known, bool ignore_return, Bytecodes::Code bc, Value receiver) {
4050   assert(!callee->is_native(), "callee must not be native");
4051   if (CompilationPolicy::policy()->should_not_inline(compilation()->env(), callee)) {
4052     INLINE_BAILOUT("inlining prohibited by policy");
4053   }
4054   // first perform tests of things it's not possible to inline
4055   if (callee->has_exception_handlers() &&
4056       !InlineMethodsWithExceptionHandlers) INLINE_BAILOUT("callee has exception handlers");
4057   if (callee->is_synchronized() &&
4058       !InlineSynchronizedMethods         ) INLINE_BAILOUT("callee is synchronized");
4059   if (!callee->holder()->is_initialized()) INLINE_BAILOUT("callee's klass not initialized yet");
4060   if (!callee->has_balanced_monitors())    INLINE_BAILOUT("callee's monitors do not match");
4061 
4062   // Proper inlining of methods with jsrs requires a little more work.
4063   if (callee->has_jsrs()                 ) INLINE_BAILOUT("jsrs not handled properly by inliner yet");
4064 
4065   if (strict_fp_requires_explicit_rounding &&
4066       method()->is_strict() != callee->is_strict()) {
4067 #ifdef IA32
4068     // If explicit rounding is required, do not inline strict code into non-strict code (or the reverse).
4069     // When SSE2 is present, no special handling is needed.
4070     if (UseSSE < 2) {
4071       INLINE_BAILOUT("caller and callee have different strict fp requirements");
4072     }
4073 #else
4074     Unimplemented();
4075 #endif // IA32
4076   }
4077 
4078   if (is_profiling() && !callee->ensure_method_data()) {
4079     INLINE_BAILOUT("mdo allocation failed");
4080   }
4081 
4082   const bool is_invokedynamic = (bc == Bytecodes::_invokedynamic);
4083   const bool has_receiver = (bc != Bytecodes::_invokestatic && !is_invokedynamic);
4084 
4085   const int args_base = state()->stack_size() - callee->arg_size();
4086   assert(args_base >= 0, "stack underflow during inlining");
4087 
4088   Value recv = NULL;
4089   if (has_receiver) {
4090     assert(!callee->is_static(), "callee must not be static");
4091     assert(callee->arg_size() > 0, "must have at least a receiver");
4092 
4093     recv = state()->stack_at(args_base);
4094     if (recv->is_null_obj()) {
4095       INLINE_BAILOUT("receiver is always null");
4096     }
4097   }
4098 
4099   // now perform tests that are based on flag settings
4100   bool inlinee_by_directive = compilation()->directive()->should_inline(callee);
4101   if (callee->force_inline() || inlinee_by_directive) {
4102     if (inline_level() > MaxForceInlineLevel                      ) INLINE_BAILOUT("MaxForceInlineLevel");
4103     if (recursive_inline_level(callee) > C1MaxRecursiveInlineLevel) INLINE_BAILOUT("recursive inlining too deep");
4104 
4105     const char* msg = "";
4106     if (callee->force_inline())  msg = "force inline by annotation";
4107     if (inlinee_by_directive)    msg = "force inline by CompileCommand";
4108     print_inlining(callee, msg);
4109   } else {
4110     // use heuristic controls on inlining
4111     if (inline_level() > C1MaxInlineLevel                       ) INLINE_BAILOUT("inlining too deep");
4112     int callee_recursive_level = recursive_inline_level(callee);
4113     if (callee_recursive_level > C1MaxRecursiveInlineLevel      ) INLINE_BAILOUT("recursive inlining too deep");
4114     if (callee->code_size_for_inlining() > max_inline_size()    ) INLINE_BAILOUT("callee is too large");
4115     // Additional condition to limit stack usage for non-recursive calls.
4116     if ((callee_recursive_level == 0) &&
4117         (callee->max_stack() + callee->max_locals() - callee->size_of_parameters() > C1InlineStackLimit)) {
4118       INLINE_BAILOUT("callee uses too much stack");
4119     }
4120 
4121     // don't inline throwable methods unless the inlining tree is rooted in a throwable class
4122     if (callee->name() == ciSymbol::object_initializer_name() &&
4123         callee->holder()->is_subclass_of(ciEnv::current()->Throwable_klass())) {
4124       // Throwable constructor call
4125       IRScope* top = scope();
4126       while (top->caller() != NULL) {
4127         top = top->caller();
4128       }
4129       if (!top->method()->holder()->is_subclass_of(ciEnv::current()->Throwable_klass())) {
4130         INLINE_BAILOUT("don't inline Throwable constructors");
4131       }
4132     }
4133

4135       INLINE_BAILOUT("total inlining greater than DesiredMethodLimit");
4136     }

4138     print_inlining(callee, "inline", /*success*/ true);
4139   }

4142   // GraphBuilder level, do not cause bailout just of the inlining but
4143   // in fact of the entire compilation.

4146 
4147   // Insert null check if necessary
4148   if (has_receiver) {
4149     // note: null check must happen even if first instruction of callee does
4150     //       an implicit null check since the callee is in a different scope
4151     //       and we must make sure exception handling does the right thing
4152     null_check(recv);
4153   }
4154 
4155   if (is_profiling()) {
4156     // Note that we'd collect profile data in this method if we wanted it.

4159

4197   // inlining, to detect if inlined method has edges to its
4198   // continuation after inlining.
4199   int continuation_preds = cont->number_of_preds();

4202   push_scope(callee, cont);

4209   // (only using the bci of this stream)
4210   scope_data()->set_stream(scope_data()->parent()->stream());

4213   // note: this will also ensure that all arguments are computed before being passed
4214   ValueStack* callee_state = state();

4217     const int arg_no = i - args_base;
4218     Value arg = caller_state->stack_at_inc(i);
4219     store_local(callee_state, arg, arg_no);

4221

4223   // Note that we preserve locals state in case we can use it later
4224   // (see use of pop_scope() below)
4225   caller_state->truncate_stack(args_base);
4226   assert(callee_state->stack_size() == 0, "callee stack must be empty");
4227

4229   BlockBegin* sync_handler = NULL;
4230

4243   }

4247   }

4250   if (callee_start_block != NULL) {

4253     // The state for this goto is in the scope of the callee, so use
4254     // the entry bci for the callee instead of the call site bci.
4255     append_with_bci(goto_callee, 0);
4256     _block->set_end(goto_callee);
4257     callee_start_block->merge(callee_state);

4272   // were in before or in the callee's start block)
4273   iterate_all_blocks(callee_start_block == NULL);
4274

4276

4281   // iterate_all_blocks theoretically traverses in random order; in
4282   // practice, we have only traversed the continuation if we are
4283   // inlining into a subroutine
4284   assert(continuation_existed ||

4289   // the caller back in the GraphBuilder. The only thing we want to do
4290   // first is an optimization: during parsing of the callee we
4291   // generated at least one Goto to the continuation block. If we
4292   // generated exactly one, and if the inlined method spanned exactly
4293   // one block (and we didn't have to Goto its entry), then we snip
4294   // off the Goto to the continuation, allowing control to fall
4295   // through back into the caller block and effectively performing
4296   // block merging. This allows load elimination and CSE to take place
4297   // across multiple callee scopes if they are relatively simple, and
4298   // is currently essential to making inlining profitable.
4299   if (num_returns() == 1
4300       && block() == orig_block
4301       && block() == inline_cleanup_block()) {
4302     _last  = inline_cleanup_return_prev();
4303     _state = inline_cleanup_state();
4304   } else if (continuation_preds == cont->number_of_preds()) {
4305     // Inlining caused that the instructions after the invoke in the
4306     // caller are not reachable any more. So skip filling this block
4307     // with instructions!
4308     assert(cont == continuation(), "");
4309     assert(_last && _last->as_BlockEnd(), "");
4310     _skip_block = true;
4311   } else {
4312     // Resume parsing in continuation block unless it was already parsed.
4313     // Note that if we don't change _last here, iteration in
4314     // iterate_bytecodes_for_block will stop when we return.
4315     if (!continuation()->is_set(BlockBegin::was_visited_flag)) {
4316       // add continuation to work list instead of parsing it immediately
4317       assert(_last && _last->as_BlockEnd(), "");
4318       scope_data()->parent()->add_to_work_list(continuation());
4319       _skip_block = true;
4320     }
4321   }
4322 
4323   // Fill the exception handler for synchronized methods with instructions
4324   if (callee->is_synchronized() && sync_handler->state() != NULL) {
4325     fill_sync_handler(lock, sync_handler);
4326   } else {
4327     pop_scope();
4328   }
4329 
4330   compilation()->notice_inlined_method(callee);
4331 
4332   return true;
4333 }

4336 bool GraphBuilder::try_method_handle_inline(ciMethod* callee, bool ignore_return) {
4337   ValueStack* state_before = copy_state_before();
4338   vmIntrinsics::ID iid = callee->intrinsic_id();
4339   switch (iid) {
4340   case vmIntrinsics::_invokeBasic:
4341     {
4342       // get MethodHandle receiver
4343       const int args_base = state()->stack_size() - callee->arg_size();
4344       ValueType* type = state()->stack_at(args_base)->type();
4345       if (type->is_constant()) {
4346         ciMethod* target = type->as_ObjectType()->constant_value()->as_method_handle()->get_vmtarget();
4347         // We don't do CHA here so only inline static and statically bindable methods.
4348         if (target->is_static() || target->can_be_statically_bound()) {
4349           if (ciMethod::is_consistent_info(callee, target)) {
4350             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4351             ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4352             if (try_inline(target, /*holder_known*/ true, ignore_return, bc)) {
4353               return true;
4354             }
4355           } else {
4356             print_inlining(target, "signatures mismatch", /*success*/ false);
4357           }
4358         } else {
4359           print_inlining(target, "not static or statically bindable", /*success*/ false);
4360         }
4361       } else {
4362         print_inlining(callee, "receiver not constant", /*success*/ false);
4363       }
4364     }
4365     break;
4366 
4367   case vmIntrinsics::_linkToVirtual:
4368   case vmIntrinsics::_linkToStatic:
4369   case vmIntrinsics::_linkToSpecial:
4370   case vmIntrinsics::_linkToInterface:
4371     {
4372       // pop MemberName argument
4373       const int args_base = state()->stack_size() - callee->arg_size();
4374       ValueType* type = apop()->type();
4375       if (type->is_constant()) {
4376         ciMethod* target = type->as_ObjectType()->constant_value()->as_member_name()->get_vmtarget();
4377         ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4378         // If the target is another method handle invoke, try to recursively get
4379         // a better target.
4380         if (target->is_method_handle_intrinsic()) {
4381           if (try_method_handle_inline(target, ignore_return)) {
4382             return true;
4383           }
4384         } else if (!ciMethod::is_consistent_info(callee, target)) {
4385           print_inlining(target, "signatures mismatch", /*success*/ false);
4386         } else {
4387           ciSignature* signature = target->signature();
4388           const int receiver_skip = target->is_static() ? 0 : 1;
4389           // Cast receiver to its type.
4390           if (!target->is_static()) {
4391             ciKlass* tk = signature->accessing_klass();
4392             Value obj = state()->stack_at(args_base);
4393             if (obj->exact_type() == NULL &&
4394                 obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4395               TypeCast* c = new TypeCast(tk, obj, state_before);
4396               append(c);
4397               state()->stack_at_put(args_base, c);
4398             }
4399           }
4400           // Cast reference arguments to its type.
4401           for (int i = 0, j = 0; i < signature->count(); i++) {
4402             ciType* t = signature->type_at(i);
4403             if (t->is_klass()) {
4404               ciKlass* tk = t->as_klass();
4405               Value obj = state()->stack_at(args_base + receiver_skip + j);
4406               if (obj->exact_type() == NULL &&
4407                   obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4408                 TypeCast* c = new TypeCast(t, obj, state_before);
4409                 append(c);
4410                 state()->stack_at_put(args_base + receiver_skip + j, c);
4411               }
4412             }
4413             j += t->size();  // long and double take two slots
4414           }
4415           // We don't do CHA here so only inline static and statically bindable methods.
4416           if (target->is_static() || target->can_be_statically_bound()) {
4417             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4418             if (try_inline(target, /*holder_known*/ true, ignore_return, bc)) {
4419               return true;
4420             }
4421           } else {
4422             print_inlining(target, "not static or statically bindable", /*success*/ false);
4423           }
4424         }
4425       } else {
4426         print_inlining(callee, "MemberName not constant", /*success*/ false);
4427       }
4428     }
4429     break;
4430 
4431   default:
4432     fatal("unexpected intrinsic %d: %s", iid, vmIntrinsics::name_at(iid));
4433     break;
4434   }
4435   set_state(state_before->copy_for_parsing());
4436   return false;
4437 }
