 112 void ConstantPool::deallocate_contents(ClassLoaderData* loader_data) {
 113   if (cache() != NULL) {
 114     MetadataFactory::free_metadata(loader_data, cache());
 115     set_cache(NULL);
 116   }
 117 
 118   MetadataFactory::free_array<Klass*>(loader_data, resolved_klasses());
 119   set_resolved_klasses(NULL);
 120 
 121   MetadataFactory::free_array<jushort>(loader_data, operands());
 122   set_operands(NULL);
 123 
 124   release_C_heap_structures();
 125 
 126   // free tag array
 127   MetadataFactory::free_array<u1>(loader_data, tags());
 128   set_tags(NULL);
 129 }

 136 void ConstantPool::metaspace_pointers_do(MetaspaceClosure* it) {
 137   log_trace(cds)("Iter(ConstantPool): %p", this);
 138 
 139   it->push(&_tags, MetaspaceClosure::_writable);
 140   it->push(&_cache);
 141   it->push(&_pool_holder);
 142   it->push(&_operands);
 143   it->push(&_resolved_klasses, MetaspaceClosure::_writable);
 144 
 145   for (int i = 0; i < length(); i++) {
 146     // The only MSO's embedded in the CP entries are Symbols:
 147     //   JVM_CONSTANT_String (normal and pseudo)
 148     //   JVM_CONSTANT_Utf8
 149     constantTag ctag = tag_at(i);
 150     if (ctag.is_string() || ctag.is_utf8()) {
 151       it->push(symbol_at_addr(i));
 152     }
 153   }
 154 }

 170 // Create resolved_references array and mapping array for original cp indexes
 171 // The ldc bytecode was rewritten to have the resolved reference array index so need a way
 172 // to map it back for resolving and some unlikely miscellaneous uses.
 173 // The objects created by invokedynamic are appended to this list.
 174 void ConstantPool::initialize_resolved_references(ClassLoaderData* loader_data,
 175                                                   const intStack& reference_map,
 176                                                   int constant_pool_map_length,
 177                                                   TRAPS) {
 178   // Initialized the resolved object cache.
 179   int map_length = reference_map.length();
 180   if (map_length > 0) {
 181     // Only need mapping back to constant pool entries.  The map isn't used for
 182     // invokedynamic resolved_reference entries.  For invokedynamic entries,
 183     // the constant pool cache index has the mapping back to both the constant
 184     // pool and to the resolved reference index.
 185     if (constant_pool_map_length > 0) {
 186       Array<u2>* om = MetadataFactory::new_array<u2>(loader_data, constant_pool_map_length, CHECK);
 187 
 188       for (int i = 0; i < constant_pool_map_length; i++) {
 189         int x = reference_map.at(i);
 190         assert(x == (int)(jushort) x, "klass index is too big");
 191         om->at_put(i, (jushort)x);
 192       }
 193       set_reference_map(om);
 194     }
 195 
 196     // Create Java array for holding resolved strings, methodHandles,
 197     // methodTypes, invokedynamic and invokehandle appendix objects, etc.
 198     objArrayOop stom = oopFactory::new_objArray(SystemDictionary::Object_klass(), map_length, CHECK);
 199     Handle refs_handle (THREAD, (oop)stom);  // must handleize.
 200     set_resolved_references(loader_data->add_handle(refs_handle));
 201   }
 202 }

 204 void ConstantPool::allocate_resolved_klasses(ClassLoaderData* loader_data, int num_klasses, TRAPS) {
 205   // A ConstantPool can't possibly have 0xffff valid class entries,
 206   // because entry #0 must be CONSTANT_Invalid, and each class entry must refer to a UTF8
 207   // entry for the class's name. So at most we will have 0xfffe class entries.
 208   // This allows us to use 0xffff (ConstantPool::_temp_resolved_klass_index) to indicate
 209   // UnresolvedKlass entries that are temporarily created during class redefinition.
 210   assert(num_klasses < CPKlassSlot::_temp_resolved_klass_index, "sanity");
 211   assert(resolved_klasses() == NULL, "sanity");
 212   Array<Klass*>* rk = MetadataFactory::new_array<Klass*>(loader_data, num_klasses, CHECK);
 213   set_resolved_klasses(rk);
 214 }

 216 void ConstantPool::initialize_unresolved_klasses(ClassLoaderData* loader_data, TRAPS) {
 217   int len = length();
 218   int num_klasses = 0;
 219   for (int i = 1; i <len; i++) {
 220     switch (tag_at(i).value()) {
 221     case JVM_CONSTANT_ClassIndex:
 222       {
 223         const int class_index = klass_index_at(i);
 224         unresolved_klass_at_put(i, class_index, num_klasses++);
 225       }
 226       break;
 227 #ifndef PRODUCT
 228     case JVM_CONSTANT_Class:
 229     case JVM_CONSTANT_UnresolvedClass:
 230     case JVM_CONSTANT_UnresolvedClassInError:
 231       // All of these should have been reverted back to Unresolved before calling
 232       // this function.
 233       ShouldNotReachHere();
 234 #endif
 235     }
 236   }
 237   allocate_resolved_klasses(loader_data, num_klasses, THREAD);
 238 }

 240 // Unsafe anonymous class support:
 241 void ConstantPool::klass_at_put(int class_index, int name_index, int resolved_klass_index, Klass* k, Symbol* name) {
 242   assert(is_within_bounds(class_index), "index out of bounds");
 243   assert(is_within_bounds(name_index), "index out of bounds");
 244   assert((resolved_klass_index & 0xffff0000) == 0, "must be");
 245   *int_at_addr(class_index) =
 246     build_int_from_shorts((jushort)resolved_klass_index, (jushort)name_index);
 247 
 248   symbol_at_put(name_index, name);
 249   name->increment_refcount();
 250   Klass** adr = resolved_klasses()->adr_at(resolved_klass_index);
 251   Atomic::release_store(adr, k);
 252 
 253   // The interpreter assumes when the tag is stored, the klass is resolved
 254   // and the Klass* non-NULL, so we need hardware store ordering here.
 255   jbyte qdesc_bit = (name->is_Q_signature()) ? (jbyte) JVM_CONSTANT_QDescBit : 0;
 256   if (k != NULL) {
 257     release_tag_at_put(class_index, JVM_CONSTANT_Class | qdesc_bit);
 258   } else {
 259     release_tag_at_put(class_index, JVM_CONSTANT_UnresolvedClass | qdesc_bit);
 260   }
 261 }

 263 // Unsafe anonymous class support:
 264 void ConstantPool::klass_at_put(int class_index, Klass* k) {
 265   assert(k != NULL, "must be valid klass");
 266   CPKlassSlot kslot = klass_slot_at(class_index);
 267   int resolved_klass_index = kslot.resolved_klass_index();
 268   Klass** adr = resolved_klasses()->adr_at(resolved_klass_index);
 269   Atomic::release_store(adr, k);
 270 
 271   // The interpreter assumes when the tag is stored, the klass is resolved
 272   // and the Klass* non-NULL, so we need hardware store ordering here.
 273   assert(!k->name()->is_Q_signature(), "Q-type without JVM_CONSTANT_QDescBit");
 274   release_tag_at_put(class_index, JVM_CONSTANT_Class);
 275 }

 277 #if INCLUDE_CDS_JAVA_HEAP
 278 // Archive the resolved references
 279 void ConstantPool::archive_resolved_references(Thread* THREAD) {
 280   if (_cache == NULL) {
 281     return; // nothing to do
 282   }
 283 
 284   InstanceKlass *ik = pool_holder();
 285   if (!(ik->is_shared_boot_class() || ik->is_shared_platform_class() ||
 286         ik->is_shared_app_class())) {
 287     // Archiving resolved references for classes from non-builtin loaders
 288     // is not yet supported.
 289     set_resolved_references(NULL);
 290     return;
 291   }
 292 
 293   objArrayOop rr = resolved_references();
 294   Array<u2>* ref_map = reference_map();
 295   if (rr != NULL) {
 296     int ref_map_len = ref_map == NULL ? 0 : ref_map->length();
 297     int rr_len = rr->length();
 298     for (int i = 0; i < rr_len; i++) {
 299       oop p = rr->obj_at(i);
 300       rr->obj_at_put(i, NULL);
 301       if (p != NULL && i < ref_map_len) {
 302         int index = object_to_cp_index(i);
 303         if (tag_at(index).is_string()) {
 304           oop op = StringTable::create_archived_string(p, THREAD);
 305           // If the String object is not archived (possibly too large),
 306           // NULL is returned. Also set it in the array, so we won't
 307           // have a 'bad' reference in the archived resolved_reference
 308           // array.
 309           rr->obj_at_put(i, op);
 310         }
 311       }
 312     }
 313 
 314     oop archived = HeapShared::archive_heap_object(rr, THREAD);
 315     // If the resolved references array is not archived (too large),
 316     // the 'archived' object is NULL. No need to explicitly check
 317     // the return value of archive_heap_object here. At runtime, the
 318     // resolved references will be created using the normal process
 319     // when there is no archived value.
 320     _cache->set_archived_references(archived);
 321     set_resolved_references(NULL);
 322   }
 323 }

 325 void ConstantPool::resolve_class_constants(TRAPS) {
 326   assert(DumpSharedSpaces, "used during dump time only");
 327   // The _cache may be NULL if the _pool_holder klass fails verification
 328   // at dump time due to missing dependencies.
 329   if (cache() == NULL || reference_map() == NULL) {
 330     return; // nothing to do
 331   }
 332 
 333   constantPoolHandle cp(THREAD, this);
 334   for (int index = 1; index < length(); index++) { // Index 0 is unused
 335     if (tag_at(index).is_string() && !cp->is_pseudo_string_at(index)) {
 336       int cache_index = cp->cp_to_object_index(index);
 337       string_at_impl(cp, index, cache_index, CHECK);
 338     }
 339   }
 340 }

 343 // CDS support. Create a new resolved_references array.
 344 void ConstantPool::restore_unshareable_info(TRAPS) {
 345   assert(is_constantPool(), "ensure C++ vtable is restored");
 346   assert(on_stack(), "should always be set for shared constant pools");
 347   assert(is_shared(), "should always be set for shared constant pools");
 348   assert(_cache != NULL, "constant pool _cache should not be NULL");
 349 
 350   // Only create the new resolved references array if it hasn't been attempted before
 351   if (resolved_references() != NULL) return;
 352 
 353   // restore the C++ vtable from the shared archive
 354   restore_vtable();
 355 
 356   if (SystemDictionary::Object_klass_loaded()) {
 357     ClassLoaderData* loader_data = pool_holder()->class_loader_data();
 358 #if INCLUDE_CDS_JAVA_HEAP
 359     if (HeapShared::open_archive_heap_region_mapped() &&
 360         _cache->archived_references() != NULL) {
 361       oop archived = _cache->archived_references();
 362       // Create handle for the archived resolved reference array object
 363       Handle refs_handle(THREAD, archived);
 364       set_resolved_references(loader_data->add_handle(refs_handle));
 365     } else
 366 #endif
 367     {
 368       // No mapped archived resolved reference array
 369       // Recreate the object array and add to ClassLoaderData.
 370       int map_length = resolved_reference_length();
 371       if (map_length > 0) {
 372         objArrayOop stom = oopFactory::new_objArray(SystemDictionary::Object_klass(), map_length, CHECK);
 373         Handle refs_handle(THREAD, (oop)stom);  // must handleize.
 374         set_resolved_references(loader_data->add_handle(refs_handle));
 375       }
 376     }
 377   }
 378 }

 380 void ConstantPool::remove_unshareable_info() {
 381   // Resolved references are not in the shared archive.
 382   // Save the length for restoration.  It is not necessarily the same length
 383   // as reference_map.length() if invokedynamic is saved. It is needed when
 384   // re-creating the resolved reference array if archived heap data cannot be map
 385   // at runtime.
 386   set_resolved_reference_length(
 387     resolved_references() != NULL ? resolved_references()->length() : 0);
 388 
 389   // If archiving heap objects is not allowed, clear the resolved references.
 390   // Otherwise, it is cleared after the resolved references array is cached
 391   // (see archive_resolved_references()).
 392   // If DynamicDumpSharedSpaces is enabled, clear the resolved references also
 393   // as java objects are not archived in the top layer.
 394   if (!HeapShared::is_heap_object_archiving_allowed() || DynamicDumpSharedSpaces) {
 395     set_resolved_references(NULL);
 396   }
 397 
 398   // Shared ConstantPools are in the RO region, so the _flags cannot be modified.
 399   // The _on_stack flag is used to prevent ConstantPools from deallocation during
 400   // class redefinition. Since shared ConstantPools cannot be deallocated anyway,
 401   // we always set _on_stack to true to avoid having to change _flags during runtime.
 402   _flags |= (_on_stack | _is_shared);
 403   int num_klasses = 0;
 404   for (int index = 1; index < length(); index++) { // Index 0 is unused
 405     if (!DynamicDumpSharedSpaces) {
 406       assert(!tag_at(index).is_unresolved_klass_in_error(), "This must not happen during static dump time");
 407     } else {
 408       if (tag_at(index).is_unresolved_klass_in_error() ||
 409           tag_at(index).is_method_handle_in_error()    ||
 410           tag_at(index).is_method_type_in_error()      ||
 411           tag_at(index).is_dynamic_constant_in_error()) {
 412         tag_at_put(index, JVM_CONSTANT_UnresolvedClass);
 413       }
 414     }
 415     if (tag_at(index).is_klass()) {
 416       // This class was resolved as a side effect of executing Java code
 417       // during dump time. We need to restore it back to an UnresolvedClass,
 418       // so that the proper class loading and initialization can happen
 419       // at runtime.
 420       CPKlassSlot kslot = klass_slot_at(index);
 421       int resolved_klass_index = kslot.resolved_klass_index();
 422       int name_index = kslot.name_index();
 423       assert(tag_at(name_index).is_symbol(), "sanity");
 424       resolved_klasses()->at_put(resolved_klass_index, NULL);
 425       tag_at_put(index, JVM_CONSTANT_UnresolvedClass);
 426       assert(klass_name_at(index) == symbol_at(name_index), "sanity");
 427     }
 428   }
 429   if (cache() != NULL) {
 430     cache()->remove_unshareable_info();
 431   }
 432 }

 445 void ConstantPool::trace_class_resolution(const constantPoolHandle& this_cp, Klass* k) {
 446   ResourceMark rm;
 447   int line_number = -1;
 448   const char * source_file = NULL;
 449   if (JavaThread::current()->has_last_Java_frame()) {
 450     // try to identify the method which called this function.
 451     vframeStream vfst(JavaThread::current());
 452     if (!vfst.at_end()) {
 453       line_number = vfst.method()->line_number_from_bci(vfst.bci());
 454       Symbol* s = vfst.method()->method_holder()->source_file_name();
 455       if (s != NULL) {
 456         source_file = s->as_C_string();
 457       }
 458     }
 459   }
 460   if (k != this_cp->pool_holder()) {
 461     // only print something if the classes are different
 462     if (source_file != NULL) {
 463       log_debug(class, resolve)("%s %s %s:%d",
 464                  this_cp->pool_holder()->external_name(),
 465                  k->external_name(), source_file, line_number);
 466     } else {
 467       log_debug(class, resolve)("%s %s",
 468                  this_cp->pool_holder()->external_name(),
 469                  k->external_name());
 470     }
 471   }
 472 }

 480 Klass* ConstantPool::klass_at_impl(const constantPoolHandle& this_cp, int which,
 481                                    bool save_resolution_error, TRAPS) {
 482   assert(THREAD->is_Java_thread(), "must be a Java thread");
 483   JavaThread* javaThread = (JavaThread*)THREAD;
 484 
 485   // A resolved constantPool entry will contain a Klass*, otherwise a Symbol*.
 486   // It is not safe to rely on the tag bit's here, since we don't have a lock, and
 487   // the entry and tag is not updated atomicly.
 488   CPKlassSlot kslot = this_cp->klass_slot_at(which);
 489   int resolved_klass_index = kslot.resolved_klass_index();
 490   int name_index = kslot.name_index();
 491   assert(this_cp->tag_at(name_index).is_symbol(), "sanity");
 492 
 493   Klass* klass = this_cp->resolved_klasses()->at(resolved_klass_index);
 494   if (klass != NULL) {
 495     return klass;
 496   }
 497 
 498   // This tag doesn't change back to unresolved class unless at a safepoint.
 499   if (this_cp->tag_at(which).is_unresolved_klass_in_error()) {
 500     // The original attempt to resolve this constant pool entry failed so find the
 501     // class of the original error and throw another error of the same class
 502     // (JVMS 5.4.3).
 503     // If there is a detail message, pass that detail message to the error.
 504     // The JVMS does not strictly require us to duplicate the same detail message,
 505     // or any internal exception fields such as cause or stacktrace.  But since the
 506     // detail message is often a class name or other literal string, we will repeat it
 507     // if we can find it in the symbol table.
 508     throw_resolution_error(this_cp, which, CHECK_NULL);
 509     ShouldNotReachHere();
 510   }
 511 
 512   Handle mirror_handle;
 513   Symbol* name = this_cp->symbol_at(name_index);
 514   bool inline_type_signature = false;
 515   if (name->is_Q_signature()) {
 516     name = name->fundamental_name(THREAD);
 517     inline_type_signature = true;
 518   }
 519   Handle loader (THREAD, this_cp->pool_holder()->class_loader());
 520   Handle protection_domain (THREAD, this_cp->pool_holder()->protection_domain());
 521 
 522   Klass* k;
 523   {
 524     // Turn off the single stepping while doing class resolution
 525     JvmtiHideSingleStepping jhss(javaThread);
 526     k = SystemDictionary::resolve_or_fail(name, loader, protection_domain, true, THREAD);
 527   } //  JvmtiHideSingleStepping jhss(javaThread);
 528   if (inline_type_signature) {
 529     name->decrement_refcount();
 530   }
 531 
 532   if (!HAS_PENDING_EXCEPTION) {
 533     // preserve the resolved klass from unloading
 534     mirror_handle = Handle(THREAD, k->java_mirror());
 535     // Do access check for klasses
 536     verify_constant_pool_resolve(this_cp, k, THREAD);
 537   }
 538 
 539   if (!HAS_PENDING_EXCEPTION && inline_type_signature) {
 540     check_is_inline_type(k, THREAD);
 541   }
 542 
 543   if (!HAS_PENDING_EXCEPTION) {
 544     Klass* bottom_klass = NULL;
 545     if (k->is_objArray_klass()) {
 546       bottom_klass = ObjArrayKlass::cast(k)->bottom_klass();
 547       assert(bottom_klass != NULL, "Should be set");
 548       assert(bottom_klass->is_instance_klass() || bottom_klass->is_typeArray_klass(), "Sanity check");
 549     } else if (k->is_valueArray_klass()) {
 550       bottom_klass = ValueArrayKlass::cast(k)->element_klass();
 551       assert(bottom_klass != NULL, "Should be set");
 552     }
 553   }
 554 
 555   // Failed to resolve class. We must record the errors so that subsequent attempts
 556   // to resolve this constant pool entry fail with the same error (JVMS 5.4.3).
 557   if (HAS_PENDING_EXCEPTION) {
 558     if (save_resolution_error) {
 559       save_and_throw_exception(this_cp, which, constantTag(JVM_CONSTANT_UnresolvedClass), CHECK_NULL);
 560       // If CHECK_NULL above doesn't return the exception, that means that
 561       // some other thread has beaten us and has resolved the class.
 562       // To preserve old behavior, we return the resolved class.
 563       klass = this_cp->resolved_klasses()->at(resolved_klass_index);
 564       assert(klass != NULL, "must be resolved if exception was cleared");
 565       return klass;
 566     } else {
 567       return NULL;  // return the pending exception
 568     }
 569   }
 570 
 571   // logging for class+resolve.
 572   if (log_is_enabled(Debug, class, resolve)){
 573     trace_class_resolution(this_cp, k);
 574   }
 575   Klass** adr = this_cp->resolved_klasses()->adr_at(resolved_klass_index);
 576   Atomic::release_store(adr, k);
 577   // The interpreter assumes when the tag is stored, the klass is resolved
 578   // and the Klass* stored in _resolved_klasses is non-NULL, so we need
 579   // hardware store ordering here.
 580   jbyte tag = JVM_CONSTANT_Class;
 581   if (this_cp->tag_at(which).is_Qdescriptor_klass()) {
 582     tag |= JVM_CONSTANT_QDescBit;
 583   }
 584   this_cp->release_tag_at_put(which, tag);
 585   return k;
 586 }

 589 // Does not update ConstantPool* - to avoid any exception throwing. Used
 590 // by compiler and exception handling.  Also used to avoid classloads for
 591 // instanceof operations. Returns NULL if the class has not been loaded or
 592 // if the verification of constant pool failed
 593 Klass* ConstantPool::klass_at_if_loaded(const constantPoolHandle& this_cp, int which) {
 594   CPKlassSlot kslot = this_cp->klass_slot_at(which);
 595   int resolved_klass_index = kslot.resolved_klass_index();
 596   int name_index = kslot.name_index();
 597   assert(this_cp->tag_at(name_index).is_symbol(), "sanity");
 598 
 599   Klass* k = this_cp->resolved_klasses()->at(resolved_klass_index);
 600   if (k != NULL) {
 601     return k;
 602   } else {
 603     Thread *thread = Thread::current();
 604     Symbol* name = this_cp->symbol_at(name_index);
 605     oop loader = this_cp->pool_holder()->class_loader();
 606     oop protection_domain = this_cp->pool_holder()->protection_domain();
 607     Handle h_prot (thread, protection_domain);
 608     Handle h_loader (thread, loader);
 609     Klass* k = SystemDictionary::find(name, h_loader, h_prot, thread);
 610 
 611     // Avoid constant pool verification at a safepoint, which takes the Module_lock.
 612     if (k != NULL && !SafepointSynchronize::is_at_safepoint()) {
 613       // Make sure that resolving is legal
 614       EXCEPTION_MARK;
 615       // return NULL if verification fails
 616       verify_constant_pool_resolve(this_cp, k, THREAD);
 617       if (HAS_PENDING_EXCEPTION) {
 618         CLEAR_PENDING_EXCEPTION;
 619         return NULL;
 620       }
 621       return k;
 622     } else {
 623       return k;
 624     }
 625   }
 626 }

 676   int i = which;
 677   if (!uncached && cache() != NULL) {
 678     if (ConstantPool::is_invokedynamic_index(which)) {
 679       // Invokedynamic index is index into the constant pool cache

 682       assert(tag_at(pool_index).is_name_and_type(), "");

 684     }

 690       assert(tag_at(pool_index).is_name_and_type(), "");
 691       return pool_index;
 692     }
 693   }
 694   assert(tag_at(i).is_field_or_method(), "Corrupted constant pool");
 695   assert(!tag_at(i).has_bootstrap(), "Must be handled above");
 696   jint ref_index = *int_at_addr(i);
 697   return extract_high_short_from_int(ref_index);

 794 Symbol* ConstantPool::exception_message(const constantPoolHandle& this_cp, int which, constantTag tag, oop pending_exception) {
 795   // Dig out the detailed message to reuse if possible
 796   Symbol* message = java_lang_Throwable::detail_message(pending_exception);
 797   if (message != NULL) {
 798     return message;
 799   }
 800 
 801   // Return specific message for the tag
 802   switch (tag.value()) {
 803   case JVM_CONSTANT_UnresolvedClass:
 804     // return the class name in the error message
 805     message = this_cp->klass_name_at(which);
 806     break;
 807   case JVM_CONSTANT_MethodHandle:
 808     // return the method handle name in the error message
 809     message = this_cp->method_handle_name_ref_at(which);
 810     break;
 811   case JVM_CONSTANT_MethodType:
 812     // return the method type signature in the error message
 813     message = this_cp->method_type_signature_at(which);
 814     break;
 815   case JVM_CONSTANT_Dynamic:
 816     // return the name of the condy in the error message
 817     message = this_cp->uncached_name_ref_at(which);
 818     break;
 819   default:
 820     ShouldNotReachHere();
 821   }
 822 
 823   return message;

 839 // If resolution for Class, Dynamic constant, MethodHandle or MethodType fails, save the
 840 // exception in the resolution error table, so that the same exception is thrown again.
 841 void ConstantPool::save_and_throw_exception(const constantPoolHandle& this_cp, int which,
 842                                             constantTag tag, TRAPS) {
 843   Symbol* error = PENDING_EXCEPTION->klass()->name();
 844 
 845   int error_tag = tag.error_value();
 846 
 847   if (!PENDING_EXCEPTION->
 848     is_a(SystemDictionary::LinkageError_klass())) {
 849     // Just throw the exception and don't prevent these classes from
 850     // being loaded due to virtual machine errors like StackOverflow
 851     // and OutOfMemoryError, etc, or if the thread was hit by stop()
 852     // Needs clarification to section 5.4.3 of the VM spec (see 6308271)
 853   } else if (this_cp->tag_at(which).value() != error_tag) {
 854     Symbol* message = exception_message(this_cp, which, tag, PENDING_EXCEPTION);
 855     SystemDictionary::add_resolution_error(this_cp, which, error, message);
 856     // CAS in the tag.  If a thread beat us to registering this error that's fine.
 857     // If another thread resolved the reference, this is a race condition. This
 858     // thread may have had a security manager or something temporary.
 859     // This doesn't deterministically get an error.   So why do we save this?
 860     // We save this because jvmti can add classes to the bootclass path after
 861     // this error, so it needs to get the same error if the error is first.
 862     jbyte old_tag = Atomic::cmpxchg((jbyte*)this_cp->tag_addr_at(which),
 863                                     (jbyte)tag.value(),
 864                                     (jbyte)error_tag);
 865     if (old_tag != error_tag && old_tag != tag.value()) {
 866       // MethodHandles and MethodType doesn't change to resolved version.
 867       assert(this_cp->tag_at(which).is_klass(), "Wrong tag value");
 868       // Forget the exception and use the resolved class.
 869       CLEAR_PENDING_EXCEPTION;
 870     }
 871   } else {
 872     // some other thread put this in error state
 873     throw_resolution_error(this_cp, which, CHECK);
 874   }
 875 }

 899 // Called to resolve constants in the constant pool and return an oop.
 900 // Some constant pool entries cache their resolved oop. This is also
 901 // called to create oops from constants to use in arguments for invokedynamic
 902 oop ConstantPool::resolve_constant_at_impl(const constantPoolHandle& this_cp,
 903                                            int index, int cache_index,
 904                                            bool* status_return, TRAPS) {
 905   oop result_oop = NULL;
 906   Handle throw_exception;
 907 
 908   if (cache_index == _possible_index_sentinel) {
 909     // It is possible that this constant is one which is cached in the objects.
 910     // We'll do a linear search.  This should be OK because this usage is rare.
 911     // FIXME: If bootstrap specifiers stress this code, consider putting in
 912     // a reverse index.  Binary search over a short array should do it.
 913     assert(index > 0, "valid index");
 914     cache_index = this_cp->cp_to_object_index(index);
 915   }
 916   assert(cache_index == _no_index_sentinel || cache_index >= 0, "");
 917   assert(index == _no_index_sentinel || index >= 0, "");
 918 
 919   if (cache_index >= 0) {
 920     result_oop = this_cp->resolved_references()->obj_at(cache_index);
 921     if (result_oop != NULL) {
 922       if (result_oop == Universe::the_null_sentinel()) {
 923         DEBUG_ONLY(int temp_index = (index >= 0 ? index : this_cp->object_to_cp_index(cache_index)));
 924         assert(this_cp->tag_at(temp_index).is_dynamic_constant(), "only condy uses the null sentinel");
 925         result_oop = NULL;
 926       }
 927       if (status_return != NULL)  (*status_return) = true;
 928       return result_oop;
 929       // That was easy...
 930     }
 931     index = this_cp->object_to_cp_index(cache_index);
 932   }
 933 
 934   jvalue prim_value;  // temp used only in a few cases below
 935 
 936   constantTag tag = this_cp->tag_at(index);
 937 
 938   if (status_return != NULL) {
 939     // don't trigger resolution if the constant might need it
 940     switch (tag.value()) {
 941     case JVM_CONSTANT_Class:
 942     {
 943       CPKlassSlot kslot = this_cp->klass_slot_at(index);
 944       int resolved_klass_index = kslot.resolved_klass_index();
 945       if (this_cp->resolved_klasses()->at(resolved_klass_index) == NULL) {
 946         (*status_return) = false;
 947         return NULL;
 948       }
 949       // the klass is waiting in the CP; go get it
 950       break;
 951     }
 952     case JVM_CONSTANT_String:
 953     case JVM_CONSTANT_Integer:
 954     case JVM_CONSTANT_Float:
 955     case JVM_CONSTANT_Long:
 956     case JVM_CONSTANT_Double:
 957       // these guys trigger OOM at worst
 958       break;
 959     default:
 960       (*status_return) = false;
 961       return NULL;
 962     }
 963     // from now on there is either success or an OOME
 964     (*status_return) = true;
 965   }
 966 
 967   switch (tag.value()) {
 968 
 969   case JVM_CONSTANT_UnresolvedClass:
 970   case JVM_CONSTANT_UnresolvedClassInError:
 971   case JVM_CONSTANT_Class:
 972     {
 973       assert(cache_index == _no_index_sentinel, "should not have been set");
 974       Klass* resolved = klass_at_impl(this_cp, index, true, CHECK_NULL);
 975       // ldc wants the java mirror.
 976       result_oop = resolved->java_mirror();
 977       break;
 978     }
 979 
 980   case JVM_CONSTANT_Dynamic:
 981     {
 982       // Resolve the Dynamically-Computed constant to invoke the BSM in order to obtain the resulting oop.
 983       BootstrapInfo bootstrap_specifier(this_cp, index);
 984 
 985       // The initial step in resolving an unresolved symbolic reference to a
 986       // dynamically-computed constant is to resolve the symbolic reference to a
 987       // method handle which will be the bootstrap method for the dynamically-computed
 988       // constant. If resolution of the java.lang.invoke.MethodHandle for the bootstrap
 989       // method fails, then a MethodHandleInError is stored at the corresponding
 990       // bootstrap method's CP index for the CONSTANT_MethodHandle_info. No need to
 991       // set a DynamicConstantInError here since any subsequent use of this
 992       // bootstrap method will encounter the resolution of MethodHandleInError.
 993       // Both the first, (resolution of the BSM and its static arguments), and the second tasks,
 994       // (invocation of the BSM), of JVMS Section 5.4.3.6 occur within invoke_bootstrap_method()
 995       // for the bootstrap_specifier created above.
 996       SystemDictionary::invoke_bootstrap_method(bootstrap_specifier, THREAD);
 997       Exceptions::wrap_dynamic_exception(/* is_indy */ false, THREAD);
 998       if (HAS_PENDING_EXCEPTION) {
 999         // Resolution failure of the dynamically-computed constant, save_and_throw_exception
1000         // will check for a LinkageError and store a DynamicConstantInError.
1001         save_and_throw_exception(this_cp, index, tag, CHECK_NULL);
1002       }
1003       result_oop = bootstrap_specifier.resolved_value()();
1004       BasicType type = Signature::basic_type(bootstrap_specifier.signature());
1005       if (!is_reference_type(type)) {
1006         // Make sure the primitive value is properly boxed.
1007         // This is a JDK responsibility.
1008         const char* fail = NULL;
1009         if (result_oop == NULL) {
1010           fail = "null result instead of box";
1011         } else if (!is_java_primitive(type)) {
1012           // FIXME: support value types via unboxing
1013           fail = "can only handle references and primitives";
1014         } else if (!java_lang_boxing_object::is_instance(result_oop, type)) {
1015           fail = "primitive is not properly boxed";
1016         }
1017         if (fail != NULL) {
1018           // Since this exception is not a LinkageError, throw exception
1019           // but do not save a DynamicInError resolution result.
1020           // See section 5.4.3 of the VM spec.
1021           THROW_MSG_NULL(vmSymbols::java_lang_InternalError(), fail);
1022         }
1023       }
1024 
1025       LogTarget(Debug, methodhandles, condy) lt_condy;
1026       if (lt_condy.is_enabled()) {
1027         LogStream ls(lt_condy);
1028         bootstrap_specifier.print_msg_on(&ls, "resolve_constant_at_impl");
1029       }
1030       break;
1031     }
1032 
1033   case JVM_CONSTANT_String:
1034     assert(cache_index != _no_index_sentinel, "should have been set");
1035     if (this_cp->is_pseudo_string_at(index)) {
1036       result_oop = this_cp->pseudo_string_at(index, cache_index);
1037       break;
1038     }
1039     result_oop = string_at_impl(this_cp, index, cache_index, CHECK_NULL);
1040     break;
1041 
1042   case JVM_CONSTANT_DynamicInError:
1043   case JVM_CONSTANT_MethodHandleInError:
1044   case JVM_CONSTANT_MethodTypeInError:
1045     {
1046       throw_resolution_error(this_cp, index, CHECK_NULL);
1047       break;
1048     }
1049 
1050   case JVM_CONSTANT_MethodHandle:
1051     {
1052       int ref_kind                 = this_cp->method_handle_ref_kind_at(index);
1053       int callee_index             = this_cp->method_handle_klass_index_at(index);
1054       Symbol*  name =      this_cp->method_handle_name_ref_at(index);
1055       Symbol*  signature = this_cp->method_handle_signature_ref_at(index);
1056       constantTag m_tag  = this_cp->tag_at(this_cp->method_handle_index_at(index));
1057       { ResourceMark rm(THREAD);
1058         log_debug(class, resolve)("resolve JVM_CONSTANT_MethodHandle:%d [%d/%d/%d] %s.%s",
1059                               ref_kind, index, this_cp->method_handle_index_at(index),
1060                               callee_index, name->as_C_string(), signature->as_C_string());
1061       }
1062 
1063       Klass* callee = klass_at_impl(this_cp, callee_index, true, CHECK_NULL);
1064 
1065       // Check constant pool method consistency
1066       if ((callee->is_interface() && m_tag.is_method()) ||
1067           ((!callee->is_interface() && m_tag.is_interface_method()))) {
1068         ResourceMark rm(THREAD);
1069         stringStream ss;
1070         ss.print("Inconsistent constant pool data in classfile for class %s. "
1071                  "Method '", callee->name()->as_C_string());
1072         signature->print_as_signature_external_return_type(&ss);
1073         ss.print(" %s(", name->as_C_string());
1074         signature->print_as_signature_external_parameters(&ss);
1075         ss.print(")' at index %d is %s and should be %s",
1076                  index,
1077                  callee->is_interface() ? "CONSTANT_MethodRef" : "CONSTANT_InterfaceMethodRef",
1078                  callee->is_interface() ? "CONSTANT_InterfaceMethodRef" : "CONSTANT_MethodRef");
1079         THROW_MSG_NULL(vmSymbols::java_lang_IncompatibleClassChangeError(), ss.as_string());
1080       }
1081 
1082       Klass* klass = this_cp->pool_holder();
1083       Handle value = SystemDictionary::link_method_handle_constant(klass, ref_kind,
1084                                                                    callee, name, signature,
1085                                                                    THREAD);
1086       result_oop = value();
1087       if (HAS_PENDING_EXCEPTION) {
1088         save_and_throw_exception(this_cp, index, tag, CHECK_NULL);
1089       }
1090       break;
1091     }
1092 
1093   case JVM_CONSTANT_MethodType:
1094     {
1095       Symbol*  signature = this_cp->method_type_signature_at(index);
1096       { ResourceMark rm(THREAD);
1097         log_debug(class, resolve)("resolve JVM_CONSTANT_MethodType [%d/%d] %s",
1098                               index, this_cp->method_type_index_at(index),
1099                               signature->as_C_string());
1100       }
1101       Klass* klass = this_cp->pool_holder();
1102       Handle value = SystemDictionary::find_method_handle_type(signature, klass, THREAD);
1103       result_oop = value();
1104       if (HAS_PENDING_EXCEPTION) {
1105         save_and_throw_exception(this_cp, index, tag, CHECK_NULL);
1106       }
1107       break;
1108     }
1109 
1110   case JVM_CONSTANT_Integer:
1111     assert(cache_index == _no_index_sentinel, "should not have been set");
1112     prim_value.i = this_cp->int_at(index);
1113     result_oop = java_lang_boxing_object::create(T_INT, &prim_value, CHECK_NULL);
1114     break;
1115 
1116   case JVM_CONSTANT_Float:
1117     assert(cache_index == _no_index_sentinel, "should not have been set");
1118     prim_value.f = this_cp->float_at(index);
1119     result_oop = java_lang_boxing_object::create(T_FLOAT, &prim_value, CHECK_NULL);
1120     break;
1121 
1122   case JVM_CONSTANT_Long:
1123     assert(cache_index == _no_index_sentinel, "should not have been set");
1124     prim_value.j = this_cp->long_at(index);
1125     result_oop = java_lang_boxing_object::create(T_LONG, &prim_value, CHECK_NULL);
1126     break;
1127 
1128   case JVM_CONSTANT_Double:
1129     assert(cache_index == _no_index_sentinel, "should not have been set");
1130     prim_value.d = this_cp->double_at(index);
1131     result_oop = java_lang_boxing_object::create(T_DOUBLE, &prim_value, CHECK_NULL);
1132     break;
1133 
1134   default:
1135     DEBUG_ONLY( tty->print_cr("*** %p: tag at CP[%d/%d] = %d",
1136                               this_cp(), index, cache_index, tag.value()));
1137     assert(false, "unexpected constant tag");
1138     break;
1139   }
1140 
1141   if (cache_index >= 0) {
1142     // Benign race condition:  resolved_references may already be filled in.
1143     // The important thing here is that all threads pick up the same result.
1144     // It doesn't matter which racing thread wins, as long as only one
1145     // result is used by all threads, and all future queries.
1146     oop new_result = (result_oop == NULL ? Universe::the_null_sentinel() : result_oop);
1147     oop old_result = this_cp->resolved_references()
1148       ->atomic_compare_exchange_oop(cache_index, new_result, NULL);
1149     if (old_result == NULL) {
1150       return result_oop;  // was installed
1151     } else {
1152       // Return the winning thread's result.  This can be different than
1153       // the result here for MethodHandles.
1154       if (old_result == Universe::the_null_sentinel())
1155         old_result = NULL;
1156       return old_result;
1157     }
1158   } else {
1159     assert(result_oop != Universe::the_null_sentinel(), "");
1160     return result_oop;
1161   }
1162 }

1171 void ConstantPool::copy_bootstrap_arguments_at_impl(const constantPoolHandle& this_cp, int index,
1172                                                     int start_arg, int end_arg,
1173                                                     objArrayHandle info, int pos,
1174                                                     bool must_resolve, Handle if_not_available,
1175                                                     TRAPS) {
1176   int argc;
1177   int limit = pos + end_arg - start_arg;
1178   // checks: index in range [0..this_cp->length),
1179   // tag at index, start..end in range [0..argc],
1180   // info array non-null, pos..limit in [0..info.length]
1181   if ((0 >= index    || index >= this_cp->length())  ||
1182       !(this_cp->tag_at(index).is_invoke_dynamic()    ||
1183         this_cp->tag_at(index).is_dynamic_constant()) ||
1184       (0 > start_arg || start_arg > end_arg) ||
1185       (end_arg > (argc = this_cp->bootstrap_argument_count_at(index))) ||
1186       (0 > pos       || pos > limit)         ||
1187       (info.is_null() || limit > info->length())) {
1188     // An index or something else went wrong; throw an error.
1189     // Since this is an internal API, we don't expect this,
1190     // so we don't bother to craft a nice message.
1191     THROW_MSG(vmSymbols::java_lang_LinkageError(), "bad BSM argument access");
1192   }
1193   // now we can loop safely
1194   int info_i = pos;
1195   for (int i = start_arg; i < end_arg; i++) {
1196     int arg_index = this_cp->bootstrap_argument_index_at(index, i);
1197     oop arg_oop;
1198     if (must_resolve) {
1199       arg_oop = this_cp->resolve_possibly_cached_constant_at(arg_index, CHECK);
1200     } else {
1201       bool found_it = false;
1202       arg_oop = this_cp->find_cached_constant_at(arg_index, found_it, CHECK);
1203       if (!found_it)  arg_oop = if_not_available();
1204     }
1205     info->obj_at_put(info_i++, arg_oop);
1206   }
1207 }

1243 // Compare this constant pool's entry at index1 to the constant pool
1244 // cp2's entry at index2.
1245 bool ConstantPool::compare_entry_to(int index1, const constantPoolHandle& cp2,
1246        int index2, TRAPS) {
1247 
1248   // The error tags are equivalent to non-error tags when comparing
1249   jbyte t1 = tag_at(index1).non_error_value();
1250   jbyte t2 = cp2->tag_at(index2).non_error_value();
1251 
1252   if (t1 != t2) {
1253     // Not the same entry type so there is nothing else to check. Note
1254     // that this style of checking will consider resolved/unresolved
1255     // class pairs as different.
1256     // From the ConstantPool* API point of view, this is correct
1257     // behavior. See VM_RedefineClasses::merge_constant_pools() to see how this
1258     // plays out in the context of ConstantPool* merging.
1259     return false;
1260   }
1261 
1262   switch (t1) {
1263   case JVM_CONSTANT_Class:
1264   {
1265     Klass* k1 = klass_at(index1, CHECK_false);
1266     Klass* k2 = cp2->klass_at(index2, CHECK_false);
1267     if (k1 == k2) {
1268       return true;
1269     }
1270   } break;
1271 
1272   case JVM_CONSTANT_ClassIndex:
1273   {
1274     int recur1 = klass_index_at(index1);
1275     int recur2 = cp2->klass_index_at(index2);
1276     bool match = compare_entry_to(recur1, cp2, recur2, CHECK_false);
1277     if (match) {
1278       return true;
1279     }
1280   } break;
1281 
1282   case JVM_CONSTANT_Double:
1283   {
1284     jdouble d1 = double_at(index1);
1285     jdouble d2 = cp2->double_at(index2);
1286     if (d1 == d2) {
1287       return true;
1288     }
1289   } break;
1290 
1291   case JVM_CONSTANT_Fieldref:
1292   case JVM_CONSTANT_InterfaceMethodref:
1293   case JVM_CONSTANT_Methodref:
1294   {
1295     int recur1 = uncached_klass_ref_index_at(index1);
1296     int recur2 = cp2->uncached_klass_ref_index_at(index2);
1297     bool match = compare_entry_to(recur1, cp2, recur2, CHECK_false);
1298     if (match) {
1299       recur1 = uncached_name_and_type_ref_index_at(index1);
1300       recur2 = cp2->uncached_name_and_type_ref_index_at(index2);
1301       match = compare_entry_to(recur1, cp2, recur2, CHECK_false);
1302       if (match) {
1303         return true;
1304       }
1305     }
1306   } break;
1307 
1308   case JVM_CONSTANT_Float:
1309   {
1310     jfloat f1 = float_at(index1);
1311     jfloat f2 = cp2->float_at(index2);
1312     if (f1 == f2) {
1313       return true;
1314     }
1315   } break;
1316 
1317   case JVM_CONSTANT_Integer:
1318   {
1319     jint i1 = int_at(index1);
1320     jint i2 = cp2->int_at(index2);
1321     if (i1 == i2) {
1322       return true;
1323     }
1324   } break;
1325 
1326   case JVM_CONSTANT_Long:
1327   {
1328     jlong l1 = long_at(index1);
1329     jlong l2 = cp2->long_at(index2);
1330     if (l1 == l2) {
1331       return true;
1332     }
1333   } break;
1334 
1335   case JVM_CONSTANT_NameAndType:
1336   {
1337     int recur1 = name_ref_index_at(index1);
1338     int recur2 = cp2->name_ref_index_at(index2);
1339     bool match = compare_entry_to(recur1, cp2, recur2, CHECK_false);
1340     if (match) {
1341       recur1 = signature_ref_index_at(index1);
1342       recur2 = cp2->signature_ref_index_at(index2);
1343       match = compare_entry_to(recur1, cp2, recur2, CHECK_false);
1344       if (match) {
1345         return true;
1346       }
1347     }
1348   } break;
1349 
1350   case JVM_CONSTANT_StringIndex:
1351   {
1352     int recur1 = string_index_at(index1);
1353     int recur2 = cp2->string_index_at(index2);
1354     bool match = compare_entry_to(recur1, cp2, recur2, CHECK_false);
1355     if (match) {
1356       return true;
1357     }
1358   } break;
1359 
1360   case JVM_CONSTANT_UnresolvedClass:
1361   {
1362     Symbol* k1 = klass_name_at(index1);
1363     Symbol* k2 = cp2->klass_name_at(index2);
1364     if (k1 == k2) {
1365       return true;
1366     }
1367   } break;
1368 
1369   case JVM_CONSTANT_MethodType:
1370   {
1371     int k1 = method_type_index_at(index1);
1372     int k2 = cp2->method_type_index_at(index2);
1373     bool match = compare_entry_to(k1, cp2, k2, CHECK_false);
1374     if (match) {
1375       return true;
1376     }
1377   } break;
1378 
1379   case JVM_CONSTANT_MethodHandle:
1380   {
1381     int k1 = method_handle_ref_kind_at(index1);
1382     int k2 = cp2->method_handle_ref_kind_at(index2);
1383     if (k1 == k2) {
1384       int i1 = method_handle_index_at(index1);
1385       int i2 = cp2->method_handle_index_at(index2);
1386       bool match = compare_entry_to(i1, cp2, i2, CHECK_false);
1387       if (match) {
1388         return true;
1389       }
1390     }
1391   } break;
1392 
1393   case JVM_CONSTANT_Dynamic:
1394   {
1395     int k1 = bootstrap_name_and_type_ref_index_at(index1);
1396     int k2 = cp2->bootstrap_name_and_type_ref_index_at(index2);
1397     int i1 = bootstrap_methods_attribute_index(index1);
1398     int i2 = cp2->bootstrap_methods_attribute_index(index2);
1399     // separate statements and variables because CHECK_false is used
1400     bool match_entry = compare_entry_to(k1, cp2, k2, CHECK_false);
1401     bool match_operand = compare_operand_to(i1, cp2, i2, CHECK_false);
1402     return (match_entry && match_operand);
1403   } break;
1404 
1405   case JVM_CONSTANT_InvokeDynamic:
1406   {
1407     int k1 = bootstrap_name_and_type_ref_index_at(index1);
1408     int k2 = cp2->bootstrap_name_and_type_ref_index_at(index2);
1409     int i1 = bootstrap_methods_attribute_index(index1);
1410     int i2 = cp2->bootstrap_methods_attribute_index(index2);
1411     // separate statements and variables because CHECK_false is used
1412     bool match_entry = compare_entry_to(k1, cp2, k2, CHECK_false);
1413     bool match_operand = compare_operand_to(i1, cp2, i2, CHECK_false);
1414     return (match_entry && match_operand);
1415   } break;
1416 
1417   case JVM_CONSTANT_String:
1418   {
1419     Symbol* s1 = unresolved_string_at(index1);
1420     Symbol* s2 = cp2->unresolved_string_at(index2);
1421     if (s1 == s2) {
1422       return true;
1423     }
1424   } break;
1425 
1426   case JVM_CONSTANT_Utf8:
1427   {
1428     Symbol* s1 = symbol_at(index1);
1429     Symbol* s2 = cp2->symbol_at(index2);
1430     if (s1 == s2) {
1431       return true;
1432     }
1433   } break;
1434 
1435   // Invalid is used as the tag for the second constant pool entry
1436   // occupied by JVM_CONSTANT_Double or JVM_CONSTANT_Long. It should
1437   // not be seen by itself.
1438   case JVM_CONSTANT_Invalid: // fall through
1439 
1440   default:
1441     ShouldNotReachHere();
1442     break;
1443   }
1444 
1445   return false;
1446 } // end compare_entry_to()

1449 // Resize the operands array with delta_len and delta_size.
1450 // Used in RedefineClasses for CP merge.
1451 void ConstantPool::resize_operands(int delta_len, int delta_size, TRAPS) {
1452   int old_len  = operand_array_length(operands());
1453   int new_len  = old_len + delta_len;
1454   int min_len  = (delta_len > 0) ? old_len : new_len;
1455 
1456   int old_size = operands()->length();
1457   int new_size = old_size + delta_size;
1458   int min_size = (delta_size > 0) ? old_size : new_size;
1459 
1460   ClassLoaderData* loader_data = pool_holder()->class_loader_data();
1461   Array<u2>* new_ops = MetadataFactory::new_array<u2>(loader_data, new_size, CHECK);
1462 
1463   // Set index in the resized array for existing elements only
1464   for (int idx = 0; idx < min_len; idx++) {
1465     int offset = operand_offset_at(idx);                       // offset in original array
1466     operand_offset_at_put(new_ops, idx, offset + 2*delta_len); // offset in resized array
1467   }
1468   // Copy the bootstrap specifiers only
1469   Copy::conjoint_memory_atomic(operands()->adr_at(2*old_len),
1470                                new_ops->adr_at(2*new_len),
1471                                (min_size - 2*min_len) * sizeof(u2));
1472   // Explicitly deallocate old operands array.
1473   // Note, it is not needed for 7u backport.
1474   if ( operands() != NULL) { // the safety check
1475     MetadataFactory::free_array<u2>(loader_data, operands());
1476   }
1477   set_operands(new_ops);
1478 } // end resize_operands()

1482 // Used in RedefineClasses for CP merge.
1483 void ConstantPool::extend_operands(const constantPoolHandle& ext_cp, TRAPS) {

1486     return; // nothing to do

1499     resize_operands(delta_len, delta_size, CHECK);
1500   }

1507 void ConstantPool::shrink_operands(int new_len, TRAPS) {

1523 void ConstantPool::copy_operands(const constantPoolHandle& from_cp,
1524                                  const constantPoolHandle& to_cp,
1525                                  TRAPS) {
1526 
1527   int from_oplen = operand_array_length(from_cp->operands());
1528   int old_oplen  = operand_array_length(to_cp->operands());
1529   if (from_oplen != 0) {
1530     ClassLoaderData* loader_data = to_cp->pool_holder()->class_loader_data();
1531     // append my operands to the target's operands array
1532     if (old_oplen == 0) {
1533       // Can't just reuse from_cp's operand list because of deallocation issues
1534       int len = from_cp->operands()->length();
1535       Array<u2>* new_ops = MetadataFactory::new_array<u2>(loader_data, len, CHECK);
1536       Copy::conjoint_memory_atomic(
1537           from_cp->operands()->adr_at(0), new_ops->adr_at(0), len * sizeof(u2));
1538       to_cp->set_operands(new_ops);
1539     } else {
1540       int old_len  = to_cp->operands()->length();
1541       int from_len = from_cp->operands()->length();
1542       int old_off  = old_oplen * sizeof(u2);
1543       int from_off = from_oplen * sizeof(u2);
1544       // Use the metaspace for the destination constant pool
1545       Array<u2>* new_operands = MetadataFactory::new_array<u2>(loader_data, old_len + from_len, CHECK);
1546       int fillp = 0, len = 0;
1547       // first part of dest
1548       Copy::conjoint_memory_atomic(to_cp->operands()->adr_at(0),
1549                                    new_operands->adr_at(fillp),
1550                                    (len = old_off) * sizeof(u2));
1551       fillp += len;
1552       // first part of src
1553       Copy::conjoint_memory_atomic(from_cp->operands()->adr_at(0),
1554                                    new_operands->adr_at(fillp),
1555                                    (len = from_off) * sizeof(u2));
1556       fillp += len;
1557       // second part of dest
1558       Copy::conjoint_memory_atomic(to_cp->operands()->adr_at(old_off),
1559                                    new_operands->adr_at(fillp),
1560                                    (len = old_len - old_off) * sizeof(u2));
1561       fillp += len;
1562       // second part of src
1563       Copy::conjoint_memory_atomic(from_cp->operands()->adr_at(from_off),
1564                                    new_operands->adr_at(fillp),
1565                                    (len = from_len - from_off) * sizeof(u2));
1566       fillp += len;
1567       assert(fillp == new_operands->length(), "");
1568 
1569       // Adjust indexes in the first part of the copied operands array.
1570       for (int j = 0; j < from_oplen; j++) {
1571         int offset = operand_offset_at(new_operands, old_oplen + j);
1572         assert(offset == operand_offset_at(from_cp->operands(), j), "correct copy");
1573         offset += old_len;  // every new tuple is preceded by old_len extra u2's
1574         operand_offset_at_put(new_operands, old_oplen + j, offset);
1575       }
1576 
1577       // replace target operands array with combined array
1578       to_cp->set_operands(new_operands);
1579     }
1580   }
1581 } // end copy_operands()

1588        const constantPoolHandle& to_cp, int to_i, TRAPS) {
1589

1616 // Copy this constant pool's entry at from_i to the constant pool
1617 // to_cp's entry at to_i.
1618 void ConstantPool::copy_entry_to(const constantPoolHandle& from_cp, int from_i,
1619                                         const constantPoolHandle& to_cp, int to_i,
1620                                         TRAPS) {
1621 
1622   int tag = from_cp->tag_at(from_i).value();
1623   switch (tag) {
1624   case JVM_CONSTANT_ClassIndex:
1625   {
1626     jint ki = from_cp->klass_index_at(from_i);
1627     to_cp->klass_index_at_put(to_i, ki);
1628   } break;
1629 
1630   case JVM_CONSTANT_Double:
1631   {
1632     jdouble d = from_cp->double_at(from_i);
1633     to_cp->double_at_put(to_i, d);
1634     // double takes two constant pool entries so init second entry's tag
1635     to_cp->tag_at_put(to_i + 1, JVM_CONSTANT_Invalid);
1636   } break;
1637 
1638   case JVM_CONSTANT_Fieldref:
1639   {
1640     int class_index = from_cp->uncached_klass_ref_index_at(from_i);
1641     int name_and_type_index = from_cp->uncached_name_and_type_ref_index_at(from_i);
1642     to_cp->field_at_put(to_i, class_index, name_and_type_index);
1643   } break;
1644 
1645   case JVM_CONSTANT_Float:
1646   {
1647     jfloat f = from_cp->float_at(from_i);
1648     to_cp->float_at_put(to_i, f);
1649   } break;
1650 
1651   case JVM_CONSTANT_Integer:
1652   {
1653     jint i = from_cp->int_at(from_i);
1654     to_cp->int_at_put(to_i, i);
1655   } break;
1656 
1657   case JVM_CONSTANT_InterfaceMethodref:
1658   {
1659     int class_index = from_cp->uncached_klass_ref_index_at(from_i);
1660     int name_and_type_index = from_cp->uncached_name_and_type_ref_index_at(from_i);
1661     to_cp->interface_method_at_put(to_i, class_index, name_and_type_index);
1662   } break;
1663 
1664   case JVM_CONSTANT_Long:
1665   {
1666     jlong l = from_cp->long_at(from_i);
1667     to_cp->long_at_put(to_i, l);
1668     // long takes two constant pool entries so init second entry's tag
1669     to_cp->tag_at_put(to_i + 1, JVM_CONSTANT_Invalid);
1670   } break;
1671 
1672   case JVM_CONSTANT_Methodref:
1673   {
1674     int class_index = from_cp->uncached_klass_ref_index_at(from_i);
1675     int name_and_type_index = from_cp->uncached_name_and_type_ref_index_at(from_i);
1676     to_cp->method_at_put(to_i, class_index, name_and_type_index);
1677   } break;
1678 
1679   case JVM_CONSTANT_NameAndType:
1680   {
1681     int name_ref_index = from_cp->name_ref_index_at(from_i);
1682     int signature_ref_index = from_cp->signature_ref_index_at(from_i);
1683     to_cp->name_and_type_at_put(to_i, name_ref_index, signature_ref_index);
1684   } break;
1685 
1686   case JVM_CONSTANT_StringIndex:
1687   {
1688     jint si = from_cp->string_index_at(from_i);
1689     to_cp->string_index_at_put(to_i, si);
1690   } break;
1691 
1692   case JVM_CONSTANT_Class:
1693   case JVM_CONSTANT_UnresolvedClass:
1694   case JVM_CONSTANT_UnresolvedClassInError:
1695   {
1696     // Revert to JVM_CONSTANT_ClassIndex
1697     int name_index = from_cp->klass_slot_at(from_i).name_index();
1698     assert(from_cp->tag_at(name_index).is_symbol(), "sanity");
1699     to_cp->klass_index_at_put(to_i, name_index);
1700   } break;
1701 
1702   case JVM_CONSTANT_String:
1703   {
1704     Symbol* s = from_cp->unresolved_string_at(from_i);
1705     to_cp->unresolved_string_at_put(to_i, s);
1706   } break;
1707 
1708   case JVM_CONSTANT_Utf8:
1709   {
1710     Symbol* s = from_cp->symbol_at(from_i);
1711     // Need to increase refcount, the old one will be thrown away and deferenced
1712     s->increment_refcount();
1713     to_cp->symbol_at_put(to_i, s);
1714   } break;
1715 
1716   case JVM_CONSTANT_MethodType:
1717   case JVM_CONSTANT_MethodTypeInError:
1718   {
1719     jint k = from_cp->method_type_index_at(from_i);
1720     to_cp->method_type_index_at_put(to_i, k);
1721   } break;
1722 
1723   case JVM_CONSTANT_MethodHandle:
1724   case JVM_CONSTANT_MethodHandleInError:
1725   {
1726     int k1 = from_cp->method_handle_ref_kind_at(from_i);
1727     int k2 = from_cp->method_handle_index_at(from_i);
1728     to_cp->method_handle_index_at_put(to_i, k1, k2);
1729   } break;
1730 
1731   case JVM_CONSTANT_Dynamic:
1732   case JVM_CONSTANT_DynamicInError:
1733   {
1734     int k1 = from_cp->bootstrap_methods_attribute_index(from_i);
1735     int k2 = from_cp->bootstrap_name_and_type_ref_index_at(from_i);
1736     k1 += operand_array_length(to_cp->operands());  // to_cp might already have operands
1737     to_cp->dynamic_constant_at_put(to_i, k1, k2);
1738   } break;
1739 
1740   case JVM_CONSTANT_InvokeDynamic:
1741   {
1742     int k1 = from_cp->bootstrap_methods_attribute_index(from_i);
1743     int k2 = from_cp->bootstrap_name_and_type_ref_index_at(from_i);
1744     k1 += operand_array_length(to_cp->operands());  // to_cp might already have operands
1745     to_cp->invoke_dynamic_at_put(to_i, k1, k2);
1746   } break;
1747 
1748   // Invalid is used as the tag for the second constant pool entry
1749   // occupied by JVM_CONSTANT_Double or JVM_CONSTANT_Long. It should
1750   // not be seen by itself.
1751   case JVM_CONSTANT_Invalid: // fall through
1752 
1753   default:
1754   {
1755     ShouldNotReachHere();
1756   } break;
1757   }
1758 } // end copy_entry_to()

1760 // Search constant pool search_cp for an entry that matches this
1761 // constant pool's entry at pattern_i. Returns the index of a

1778 // Compare this constant pool's bootstrap specifier at idx1 to the constant pool
1779 // cp2's bootstrap specifier at idx2.
1780 bool ConstantPool::compare_operand_to(int idx1, const constantPoolHandle& cp2, int idx2, TRAPS) {
1781   int k1 = operand_bootstrap_method_ref_index_at(idx1);
1782   int k2 = cp2->operand_bootstrap_method_ref_index_at(idx2);
1783   bool match = compare_entry_to(k1, cp2, k2, CHECK_false);
1784 
1785   if (!match) {
1786     return false;
1787   }
1788   int argc = operand_argument_count_at(idx1);
1789   if (argc == cp2->operand_argument_count_at(idx2)) {
1790     for (int j = 0; j < argc; j++) {
1791       k1 = operand_argument_index_at(idx1, j);
1792       k2 = cp2->operand_argument_index_at(idx2, j);
1793       match = compare_entry_to(k1, cp2, k2, CHECK_false);
1794       if (!match) {
1795         return false;
1796       }
1797     }
1798     return true;           // got through loop; all elements equal
1799   }
1800   return false;
1801 } // end compare_operand_to()

2065 //    0, in case of OutOfMemoryError
2066 //   -1, in case of internal error

2070                                           unsigned char *bytes) {

2480 void SymbolHashMap::add_entry(Symbol* sym, u2 value) {
2481   char *str = sym->as_utf8();
2482   unsigned int hash = compute_hash(str, sym->utf8_length());
2483   unsigned int index = hash % table_size();
2484 
2485   // check if already in map
2486   // we prefer the first entry since it is more likely to be what was used in
2487   // the class file
2488   for (SymbolHashMapEntry *en = bucket(index); en != NULL; en = en->next()) {
2489     assert(en->symbol() != NULL, "SymbolHashMapEntry symbol is NULL");
2490     if (en->hash() == hash && en->symbol() == sym) {
2491         return;  // already there
2492     }
2493   }
2494 
2495   SymbolHashMapEntry* entry = new SymbolHashMapEntry(hash, sym, value);
2496   entry->set_next(bucket(index));
2497   _buckets[index].set_entry(entry);
2498   assert(entry->symbol() != NULL, "SymbolHashMapEntry symbol is NULL");
