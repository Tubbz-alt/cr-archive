24 import jdk.test.lib.Asserts;
25 import jdk.test.lib.Platform;
26 import sun.hotspot.WhiteBox;
27 
28 /**
29  * @test
30  * @summary Verifies that Platform::isTieredSupported returns correct value.
31  * @requires vm.opt.final.TieredCompilation
32  * @library /test/lib
33  * @modules java.base/jdk.internal.misc
34  *          java.management
35  * @build sun.hotspot.WhiteBox
36  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
37  *                              sun.hotspot.WhiteBox$WhiteBoxPermission
38  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
39  *                   -XX:+WhiteBoxAPI -XX:+TieredCompilation
40  *                   TestPlatformIsTieredSupported
41  */
42 public class TestPlatformIsTieredSupported {
43     public static void main(String args[]) {
44         WhiteBox whiteBox = WhiteBox.getWhiteBox();
45         boolean tieredCompilation = whiteBox.getBooleanVMFlag(
46                 "TieredCompilation");
47         Asserts.assertEQ(Platform.isTieredSupported(), tieredCompilation,
48                 "Platform::isTieredSupported should report the same value as "
49                         + "TieredCompilation flag's value when "
50                         + "+TieredCompilation was explicitly passed to JVM.");
51     }
52 }
