24 import jdk.test.lib.Asserts;
25 import jdk.test.lib.Platform;
26 import sun.hotspot.WhiteBox;
27 
28 /**
29  * @test
30  * @summary Verifies that Platform::isTieredSupported returns correct value.
31  * @requires vm.opt.final.TieredCompilation
32  * @library /test/lib
33  * @modules java.base/jdk.internal.misc
34  *          java.management
35  * @build sun.hotspot.WhiteBox
36  * @run driver ClassFileInstaller sun.hotspot.WhiteBox
37  * @run main/othervm -Xbootclasspath/a:. -XX:+UnlockDiagnosticVMOptions
38  *                   -XX:+WhiteBoxAPI -XX:+TieredCompilation
39  *                   TestPlatformIsTieredSupported
40  */
41 public class TestPlatformIsTieredSupported {
42     public static void main(String args[]) {
43         WhiteBox whiteBox = WhiteBox.getWhiteBox();
44         boolean tieredCompilation = whiteBox.getBooleanVMFlag(
45                 "TieredCompilation");
46         Asserts.assertEQ(Platform.isTieredSupported(), tieredCompilation,
47                 "Platform::isTieredSupported should report the same value as "
48                         + "TieredCompilation flag's value when "
49                         + "+TieredCompilation was explicitly passed to JVM.");
50     }
51 }
