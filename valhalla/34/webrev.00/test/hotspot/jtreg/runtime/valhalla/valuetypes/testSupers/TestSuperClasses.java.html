24 /*
25  * @test
26  * @bug 8243204
27  * @summary test that the JVM detects illegal super classes for inline types.
28  * @compile NotAbstract.java HasNonStaticFields.java CtorHasArgs.java CtorIsNotEmpty.java
29  * @compile HasSynchMethod.java ValidSuper.java InlineClassWithBadSupers.jcod
30  * @run main/othervm -verify TestSuperClasses
31  */
32 
33 public class TestSuperClasses {
34 
35     public static void runTestIncompatibleClassChangeError(String test_name, String message) throws Exception {
36         System.out.println("Testing: " + test_name);
37         try {
38             Class newClass = Class.forName(test_name);
39             throw new RuntimeException("Expected IncompatibleClassChangeError exception not thrown");
40         } catch (java.lang.IncompatibleClassChangeError e) {
41             if (!e.getMessage().contains(message)) {
42                 throw new RuntimeException("Wrong IncompatibleClassChangeError: " + e.getMessage());
43             }
44         }
45     }
46 
47     public static void main(String[] args) throws Exception {
48 
49         // Test that the super class of an inline type must be java.lang.Object or an abstract class.
50         runTestIncompatibleClassChangeError("SuperNotAbstract",
51             "has a super class NotAbstract that is not abstract");
52 
53         // Test that the super class of an inline type cannot have instance fields.
54         runTestIncompatibleClassChangeError("SuperHasNonStaticFields",
55             "has a super class HasNonStaticFields containing instance fields");
56 
57         // Test that the super class of an inline type cannot contain a synchronized instance method.
58         runTestIncompatibleClassChangeError("SuperHasSynchMethod",
59             "has a super class HasSynchMethod containing a synchronized instance method");
60 
61         // Test that the constructor in a super class of an inline type must have a signature of "()V".
62         runTestIncompatibleClassChangeError("SuperCtorHasArgs",
63             "has a super class CtorHasArgs containing a constructor with a non-void signature");
64 
65         // Test that the constructor in a super class of an inline type must be empty.
66         runTestIncompatibleClassChangeError("SuperCtorIsNotEmpty",
67             "has a super class CtorIsNotEmpty containing a non-empty constructor");
68     }
69 }
