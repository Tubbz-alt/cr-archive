  77 Method* Method::allocate(ClassLoaderData* loader_data,
  78                          int byte_code_size,
  79                          AccessFlags access_flags,
  80                          InlineTableSizes* sizes,
  81                          ConstMethod::MethodType method_type,
  82                          TRAPS) {
  83   assert(!access_flags.is_native() || byte_code_size == 0,
  84          "native methods should not contain byte codes");
  85   ConstMethod* cm = ConstMethod::allocate(loader_data,
  86                                           byte_code_size,
  87                                           sizes,
  88                                           method_type,
  89                                           CHECK_NULL);
  90   int size = Method::size(access_flags.is_native());
  91   return new (loader_data, size, MetaspaceObj::MethodType, THREAD) Method(cm, access_flags);
  92 }

 115   }
 116   NOT_PRODUCT(set_compiled_invocation_count(0);)

 181 char* Method::name_and_sig_as_C_string(Klass* klass, Symbol* method_name, Symbol* signature) {
 182   const char* klass_name = klass->external_name();
 183   int klass_name_len  = (int)strlen(klass_name);
 184   int method_name_len = method_name->utf8_length();
 185   int len             = klass_name_len + 1 + method_name_len + signature->utf8_length();
 186   char* dest          = NEW_RESOURCE_ARRAY(char, len + 1);
 187   strcpy(dest, klass_name);
 188   dest[klass_name_len] = '.';
 189   strcpy(&dest[klass_name_len + 1], method_name->as_C_string());
 190   strcpy(&dest[klass_name_len + 1 + method_name_len], signature->as_C_string());
 191   dest[len] = 0;
 192   return dest;
 193 }

 195 char* Method::name_and_sig_as_C_string(Klass* klass, Symbol* method_name, Symbol* signature, char* buf, int size) {

 204     len = (int)strlen(buf);
 205 
 206     signature->as_C_string(&(buf[len]), size - len);
 207   }
 208 
 209   return buf;

 233 int Method::fast_exception_handler_bci_for(const methodHandle& mh, Klass* ex_klass, int throw_bci, TRAPS) {
 234   // exception table holds quadruple entries of the form (beg_bci, end_bci, handler_bci, klass_index)
 235   // access exception table
 236   ExceptionTable table(mh());
 237   int length = table.length();
 238   // iterate through all entries sequentially
 239   constantPoolHandle pool(THREAD, mh->constants());
 240   for (int i = 0; i < length; i ++) {
 241     //reacquire the table in case a GC happened
 242     ExceptionTable table(mh());
 243     int beg_bci = table.start_pc(i);
 244     int end_bci = table.end_pc(i);
 245     assert(beg_bci <= end_bci, "inconsistent exception table");
 246     if (beg_bci <= throw_bci && throw_bci < end_bci) {
 247       // exception handler bci range covers throw_bci => investigate further
 248       int handler_bci = table.handler_pc(i);
 249       int klass_index = table.catch_type_index(i);
 250       if (klass_index == 0) {
 251         return handler_bci;
 252       } else if (ex_klass == NULL) {
 253         return handler_bci;
 254       } else {
 255         // we know the exception class => get the constraint class
 256         // this may require loading of the constraint class; if verification
 257         // fails or some other exception occurs, return handler_bci
 258         Klass* k = pool->klass_at(klass_index, CHECK_(handler_bci));
 259         assert(k != NULL, "klass not loaded");
 260         if (ex_klass->is_subtype_of(k)) {
 261           return handler_bci;
 262         }
 263       }
 264     }
 265   }
 266 
 267   return -1;
 268 }

 270 void Method::mask_for(int bci, InterpreterOopMap* mask) {
 271   methodHandle h_this(Thread::current(), this);
 272   // Only GC uses the OopMapCache during thread stack root scanning
 273   // any other uses generate an oopmap but do not save it in the cache.
 274   if (Universe::heap()->is_gc_active()) {
 275     method_holder()->mask_for(h_this, bci, mask);
 276   } else {
 277     OopMapCache::compute_one_oop_map(h_this, bci, mask);
 278   }
 279   return;
 280 }

 303 // Return bci if it appears to be a valid bcp
 304 // Return -1 otherwise.
 305 // Used by profiling code, when invalid data is a possibility.
 306 // The caller is responsible for validating the Method* itself.
 307 int Method::validate_bci_from_bcp(address bcp) const {
 308   // keep bci as -1 if not a valid bci
 309   int bci = -1;
 310   if (bcp == 0 || bcp == code_base()) {
 311     // code_size() may return 0 and we allow 0 here
 312     // the method may be native
 313     bci = 0;
 314   } else if (contains(bcp)) {
 315     bci = bcp - code_base();
 316   }
 317   // Assert that if we have dodged any asserts, bci is negative.
 318   assert(bci == -1 || bci == bci_from(bcp_from(bci)), "sane bci if >=0");
 319   return bci;
 320 }

 322 address Method::bcp_from(int bci) const {
 323   assert((is_native() && bci == 0) || (!is_native() && 0 <= bci && bci < code_size()),
 324          "illegal bci: %d for %s method", bci, is_native() ? "native" : "non-native");
 325   address bcp = code_base() + bci;
 326   assert(is_native() && bcp == code_base() || contains(bcp), "bcp doesn't belong to this method");
 327   return bcp;
 328 }

 339   // If native, then include pointers for native_function and signature_handler

 342   return align_metadata_size(header_size() + extra_words);

 399 // The RegisterNatives call being attempted tried to register with a method that
 400 // is not native.  Ask JVM TI what prefixes have been specified.  Then check
 401 // to see if the native method is now wrapped with the prefixes.  See the
 402 // SetNativeMethodPrefix(es) functions in the JVM TI Spec for details.
 403 static Method* find_prefixed_native(Klass* k, Symbol* name, Symbol* signature, TRAPS) {
 404 #if INCLUDE_JVMTI
 405   ResourceMark rm(THREAD);
 406   Method* method;
 407   int name_len = name->utf8_length();
 408   char* name_str = name->as_utf8();
 409   int prefix_count;
 410   char** prefixes = JvmtiExport::get_all_native_method_prefixes(&prefix_count);
 411   for (int i = 0; i < prefix_count; i++) {
 412     char* prefix = prefixes[i];
 413     int prefix_len = (int)strlen(prefix);
 414 
 415     // try adding this prefix to the method name and see if it matches another method name
 416     int trial_len = name_len + prefix_len;
 417     char* trial_name_str = NEW_RESOURCE_ARRAY(char, trial_len + 1);
 418     strcpy(trial_name_str, prefix);
 419     strcat(trial_name_str, name_str);
 420     TempNewSymbol trial_name = SymbolTable::probe(trial_name_str, trial_len);
 421     if (trial_name == NULL) {
 422       continue; // no such symbol, so this prefix wasn't used, try the next prefix
 423     }
 424     method = k->lookup_method(trial_name, signature);
 425     if (method == NULL) {
 426       continue; // signature doesn't match, try the next prefix
 427     }
 428     if (method->is_native()) {
 429       method->set_is_prefixed_native();
 430       return method; // wahoo, we found a prefixed version of the method, return it
 431     }
 432     // found as non-native, so prefix is good, add it, probably just need more prefixes
 433     name_len = trial_len;
 434     name_str = trial_name_str;
 435   }
 436 #endif // INCLUDE_JVMTI
 437   return NULL; // not found
 438 }

 440 bool Method::register_native(Klass* k, Symbol* name, Symbol* signature, address entry, TRAPS) {
 441   Method* method = k->lookup_method(name, signature);
 442   if (method == NULL) {
 443     ResourceMark rm(THREAD);
 444     stringStream st;
 445     st.print("Method '");
 446     print_external_name(&st, k, name, signature);
 447     st.print("' name or signature does not match");
 448     THROW_MSG_(vmSymbols::java_lang_NoSuchMethodError(), st.as_string(), false);
 449   }
 450   if (!method->is_native()) {
 451     // trying to register to a non-native method, see if a JVM TI agent has added prefix(es)
 452     method = find_prefixed_native(k, name, signature, THREAD);
 453     if (method == NULL) {
 454       ResourceMark rm(THREAD);
 455       stringStream st;
 456       st.print("Method '");
 457       print_external_name(&st, k, name, signature);
 458       st.print("' is not declared as native");
 459       THROW_MSG_(vmSymbols::java_lang_NoSuchMethodError(), st.as_string(), false);
 460     }
 461   }
 462 
 463   if (entry != NULL) {
 464     method->set_native_function(entry, native_bind_event_is_interesting);
 465   } else {
 466     method->clear_native_function();
 467   }
 468   if (log_is_enabled(Debug, jni, resolve)) {
 469     ResourceMark rm(THREAD);
 470     log_debug(jni, resolve)("[Registering JNI native method %s.%s]",
 471                             method->method_holder()->external_name(),
 472                             method->name()->as_C_string());
 473   }
 474   return true;
 475 }

 477 bool Method::was_executed_more_than(int n) {
 478   // Invocation counter is reset when the Method* is compiled.
 479   // If the method has compiled code we therefore assume it has
 480   // be excuted more than n times.
 481   if (is_accessor() || is_empty_method() || (code() != NULL)) {
 482     // interpreter doesn't bump invocation counter of trivial methods
 483     // compiler does not bump invocation counter of compiled methods
 484     return true;
 485   }
 486   else if ((method_counters() != NULL &&
 487             method_counters()->invocation_counter()->carry()) ||
 488            (method_data() != NULL &&
 489             method_data()->invocation_counter()->carry())) {
 490     // The carry bit is set when the counter overflows and causes
 491     // a compilation to occur.  We don't know how many times
 492     // the counter has been reset, so we simply assume it has
 493     // been executed more than n times.
 494     return true;
 495   } else {
 496     return invocation_count() > n;
 497   }
 498 }

 500 void Method::print_invocation_count() {
 501   if (is_static()) tty->print("static ");
 502   if (is_final()) tty->print("final ");
 503   if (is_synchronized()) tty->print("synchronized ");
 504   if (is_native()) tty->print("native ");
 505   tty->print("%s::", method_holder()->external_name());
 506   name()->print_symbol_on(tty);
 507   signature()->print_symbol_on(tty);
 508 
 509   if (WizardMode) {
 510     // dump the size of the byte codes
 511     tty->print(" {%d}", code_size());
 512   }
 513   tty->cr();
 514 
 515   tty->print_cr ("  interpreter_invocation_count: %8d ", interpreter_invocation_count());
 516   tty->print_cr ("  invocation_counter:           %8d ", invocation_count());
 517   tty->print_cr ("  backedge_counter:             %8d ", backedge_count());
 518 #ifndef PRODUCT
 519   if (CountCompiledCalls) {
 520     tty->print_cr ("  compiled_invocation_count: %8d ", compiled_invocation_count());
 521   }
 522 #endif
 523 }

 525 // Build a MethodData* object to hold information about this method
 526 // collected in the interpreter.
 527 void Method::build_interpreter_method_data(const methodHandle& method, TRAPS) {
 528   // Do not profile the method if metaspace has hit an OOM previously
 529   // allocating profiling data. Callers clear pending exception so don't
 530   // add one here.
 531   if (ClassLoaderDataGraph::has_metaspace_oom()) {
 532     return;
 533   }
 534 
 535   // Grab a lock here to prevent multiple
 536   // MethodData*s from being created.
 537   MutexLocker ml(THREAD, MethodData_lock);
 538   if (method->method_data() == NULL) {
 539     ClassLoaderData* loader_data = method->method_holder()->class_loader_data();
 540     MethodData* method_data = MethodData::allocate(loader_data, method, THREAD);
 541     if (HAS_PENDING_EXCEPTION) {
 542       CompileBroker::log_metaspace_failure();
 543       ClassLoaderDataGraph::set_metaspace_oom(true);
 544       return;   // return the exception (which is cleared)
 545     }
 546 
 547     method->set_method_data(method_data);
 548     if (PrintMethodData && (Verbose || WizardMode)) {
 549       ResourceMark rm(THREAD);
 550       tty->print("build_interpreter_method_data for ");
 551       method->print_name(tty);
 552       tty->cr();
 553       // At the end of the run, the MDO, full of data, will be dumped.
 554     }
 555   }
 556 }

 558 MethodCounters* Method::build_method_counters(Method* m, TRAPS) {
 559   // Do not profile the method if metaspace has hit an OOM previously
 560   if (ClassLoaderDataGraph::has_metaspace_oom()) {
 561     return NULL;
 562   }
 563 
 564   methodHandle mh(THREAD, m);
 565   MethodCounters* counters = MethodCounters::allocate(mh, THREAD);
 566   if (HAS_PENDING_EXCEPTION) {
 567     CompileBroker::log_metaspace_failure();
 568     ClassLoaderDataGraph::set_metaspace_oom(true);
 569     return NULL;   // return the exception (which is cleared)
 570   }
 571   if (!mh->init_method_counters(counters)) {
 572     MetadataFactory::free_metadata(mh->method_holder()->class_loader_data(), counters);
 573   }
 574 
 575   if (LogTouchedMethods) {
 576     mh->log_touched(CHECK_NULL);
 577   }
 578 
 579   return mh->method_counters();
 580 }

 627 bool Method::is_vanilla_constructor() const {
 628   // Returns true if this method is a vanilla constructor, i.e. an "<init>" "()V" method
 629   // which only calls the superclass vanilla constructor and possibly does stores of
 630   // zero constants to local fields:
 631   //
 632   //   aload_0, _fast_aload_0, or _nofast_aload_0
 633   //   invokespecial
 634   //   indexbyte1
 635   //   indexbyte2
 636   //
 637   // followed by an (optional) sequence of:
 638   //
 639   //   aload_0
 640   //   aconst_null / iconst_0 / fconst_0 / dconst_0
 641   //   putfield
 642   //   indexbyte1
 643   //   indexbyte2
 644   //
 645   // followed by:
 646   //
 647   //   return
 648 
 649   assert(name() == vmSymbols::object_initializer_name(),    "Should only be called for default constructors");
 650   assert(signature() == vmSymbols::void_method_signature(), "Should only be called for default constructors");
 651   int size = code_size();
 652   // Check if size match
 653   if (size == 0 || size % 5 != 0) return false;
 654   address cb = code_base();
 655   int last = size - 1;
 656   if ((cb[0] != Bytecodes::_aload_0 && cb[0] != Bytecodes::_fast_aload_0 && cb[0] != Bytecodes::_nofast_aload_0) ||
 657        cb[1] != Bytecodes::_invokespecial || cb[last] != Bytecodes::_return) {
 658     // Does not call superclass default constructor
 659     return false;
 660   }
 661   // Check optional sequence
 662   for (int i = 4; i < last; i += 5) {
 663     if (cb[i] != Bytecodes::_aload_0) return false;
 664     if (!Bytecodes::is_zero_const(Bytecodes::cast(cb[i+1]))) return false;
 665     if (cb[i+2] != Bytecodes::_putfield) return false;
 666   }
 667   return true;
 668 }

 672   BytecodeStream bcs(methodHandle(Thread::current(), this));
 673   Bytecodes::Code bc;
 674 
 675   while ((bc = bcs.next()) >= 0) {
 676     switch( bc ) {
 677       case Bytecodes::_ifeq:
 678       case Bytecodes::_ifnull:
 679       case Bytecodes::_iflt:
 680       case Bytecodes::_ifle:
 681       case Bytecodes::_ifne:
 682       case Bytecodes::_ifnonnull:
 683       case Bytecodes::_ifgt:
 684       case Bytecodes::_ifge:
 685       case Bytecodes::_if_icmpeq:
 686       case Bytecodes::_if_icmpne:
 687       case Bytecodes::_if_icmplt:
 688       case Bytecodes::_if_icmpgt:
 689       case Bytecodes::_if_icmple:
 690       case Bytecodes::_if_icmpge:
 691       case Bytecodes::_if_acmpeq:
 692       case Bytecodes::_if_acmpne:
 693       case Bytecodes::_goto:

 705     }

 861 int Method::line_number_from_bci(int bci) const {
 862   int best_bci  =  0;
 863   int best_line = -1;
 864   if (bci == SynchronizationEntryBCI) bci = 0;
 865   if (0 <= bci && bci < code_size() && has_linenumber_table()) {
 866     // The line numbers are a short array of 2-tuples [start_pc, line_number].
 867     // Not necessarily sorted and not necessarily one-to-one.
 868     CompressedLineNumberReadStream stream(compressed_linenumber_table());
 869     while (stream.read_pair()) {
 870       if (stream.bci() == bci) {
 871         // perfect match
 872         return stream.line();
 873       } else {
 874         // update best_bci/line
 875         if (stream.bci() < bci && stream.bci() >= best_bci) {
 876           best_bci  = stream.bci();
 877           best_line = stream.line();
 878         }
 879       }
 880     }
 881   }
 882   return best_line;
 883 }

 911 void Method::set_native_function(address function, bool post_event_flag) {
 912   assert(function != NULL, "use clear_native_function to unregister natives");
 913   assert(!is_method_handle_intrinsic() || function == SharedRuntime::native_method_throw_unsatisfied_link_error_entry(), "");
 914   address* native_function = native_function_addr();
 915 
 916   // We can see racers trying to place the same native function into place. Once
 917   // is plenty.
 918   address current = *native_function;
 919   if (current == function) return;
 920   if (post_event_flag && JvmtiExport::should_post_native_method_bind() &&
 921       function != NULL) {
 922     // native_method_throw_unsatisfied_link_error_entry() should only
 923     // be passed when post_event_flag is false.
 924     assert(function !=
 925       SharedRuntime::native_method_throw_unsatisfied_link_error_entry(),
 926       "post_event_flag mis-match");
 927 
 928     // post the bind event, and possible change the bind function
 929     JvmtiExport::post_native_method_bind(this, &function);
 930   }
 931   *native_function = function;
 932   // This function can be called more than once. We must make sure that we always
 933   // use the latest registered method -> check if a stub already has been generated.
 934   // If so, we have to make it not_entrant.
 935   CompiledMethod* nm = code(); // Put it into local variable to guard against concurrent updates
 936   if (nm != NULL) {
 937     nm->make_not_entrant();
 938   }
 939 }

1155 //
1156 // In this example, both Methods A and B have the _i2i_entry of "zero_locals".
1157 // They also have similar signatures so that they will share the same
1158 // AdapterHandlerEntry.
1159 //
1160 // _adapter_trampoline points to a fixed location in the RW section of
1161 // the CDS archive. This location initially contains a NULL pointer. When the
1162 // first of method A or B is linked, an AdapterHandlerEntry is allocated
1163 // dynamically, and its c2i/i2c entries are generated.
1164 //
1165 // _i2i_entry and _from_interpreted_entry initially points to the same
1166 // (fixed) location in the CODE section of the CDS archive. This contains
1167 // an unconditional branch to the actual entry for "zero_locals", which is
1168 // generated at run time and may be on an arbitrary address. Thus, the
1169 // unconditional branch is also generated at run time to jump to the correct
1170 // address.
1171 //
1172 // Similarly, _from_compiled_entry points to a fixed address in the CODE
1173 // section. This address has enough space for an unconditional branch
1174 // instruction, and is initially zero-filled. After the AdapterHandlerEntry is
1175 // initialized, and the address for the actual c2i_entry is known, we emit a
1176 // branch instruction here to branch to the actual c2i_entry.
1177 //
1178 // The effect of the extra branch on the i2i and c2i entries is negligible.
1179 //
1180 // The reason for putting _adapter_trampoline in RO is many shared Methods
1181 // share the same AdapterHandlerEntry, so we can save space in the RW section
1182 // by having the extra indirection.
1183 
1184 
1185 [Method A: RW]
1186   _constMethod ----> [ConstMethod: RO]
1187                        _adapter_trampoline -----------+
1188                                                       |
1189   _i2i_entry              (same value as method B)    |
1190   _from_interpreted_entry (same value as method B)    |
1191   _from_compiled_entry    (same value as method B)    |
1192                                                       |
1193                                                       |
1194 [Method B: RW]                               +--------+
1195   _constMethod ----> [ConstMethod: RO]       |
1196                        _adapter_trampoline --+--->(AdapterHandlerEntry* ptr: RW)-+
1197                                                                                  |
1198                                                  +-------------------------------+
1199                                                  |
1200                                                  +----> [AdapterHandlerEntry] (allocated at run time)
1201                                                               _fingerprint
1202                                                               _c2i_entry ---------------------------------+->[c2i entry..]
1203  _i2i_entry  -------------+                                   _i2c_entry ---------------+-> [i2c entry..] |
1204  _from_interpreted_entry  |                                   _c2i_unverified_entry     |                 |
1205          |                |                                   _c2i_no_clinit_check_entry|                 |
1206          |                |  (_cds_entry_table: CODE)                                   |                 |
1207          |                +->[0]: jmp _entry_table[0] --> (i2i_entry_for "zero_locals") |                 |
1208          |                |                               (allocated at run time)       |                 |
1209          |                |  ...                           [asm code ...]               |                 |
1210          +-[not compiled]-+  [n]: jmp _entry_table[n]                                   |                 |
1211          |                                                                              |                 |
1212          |                                                                              |                 |
1213          +-[compiled]-------------------------------------------------------------------+                 |
1214                                                                                                           |
1215  _from_compiled_entry------------>  (_c2i_entry_trampoline: CODE)                                         |
1216                                     [jmp c2i_entry] ------------------------------------------------------+
1217 
1218 ***/

1220 // Called when the method_holder is getting linked. Setup entrypoints so the method
1221 // is ready to be called from interpreter, compiler, and vtables.
1222 void Method::link_method(const methodHandle& h_method, TRAPS) {
1223   // If the code cache is full, we may reenter this function for the
1224   // leftover methods that weren't linked.
1225   if (is_shared()) {
1226 #ifdef ASSERT
1227     address entry = Interpreter::entry_for_cds_method(h_method);
1228     assert(entry != NULL && entry == _i2i_entry,
1229            "should be correctly set during dump time");
1230 #endif
1231     if (adapter() != NULL) {
1232       return;
1233     }
1234     assert(entry == _from_interpreted_entry,
1235            "should be correctly set during dump time");
1236   } else if (_i2i_entry != NULL) {
1237     return;
1238   }
1239   assert( _code == NULL, "nothing compiled yet" );
1240 
1241   // Setup interpreter entrypoint
1242   assert(this == h_method(), "wrong h_method()" );
1243 
1244   if (!is_shared()) {
1245     assert(adapter() == NULL, "init'd to NULL");
1246     address entry = Interpreter::entry_for_method(h_method);
1247     assert(entry != NULL, "interpreter entry must be non-null");
1248     // Sets both _i2i_entry and _from_interpreted_entry
1249     set_interpreter_entry(entry);
1250   }
1251 
1252   // Don't overwrite already registered native entries.
1253   if (is_native() && !has_native_function()) {
1254     set_native_function(
1255       SharedRuntime::native_method_throw_unsatisfied_link_error_entry(),
1256       !native_bind_event_is_interesting);
1257   }
1258 
1259   // Setup compiler entrypoint.  This is made eagerly, so we do not need
1260   // special handling of vtables.  An alternative is to make adapters more
1261   // lazily by calling make_adapter() from from_compiled_entry() for the
1262   // normal calls.  For vtable calls life gets more complicated.  When a
1263   // call-site goes mega-morphic we need adapters in all methods which can be
1264   // called from the vtable.  We need adapters on such methods that get loaded
1265   // later.  Ditto for mega-morphic itable calls.  If this proves to be a
1266   // problem we'll make these lazily later.
1267   (void) make_adapters(h_method, CHECK);
1268 
1269   // ONLY USE the h_method now as make_adapter may have blocked
1270 
1271 }

1273 address Method::make_adapters(const methodHandle& mh, TRAPS) {
1274   // Adapters for compiled code are made eagerly here.  They are fairly
1275   // small (generally < 100 bytes) and quick to make (and cached and shared)
1276   // so making them eagerly shouldn't be too expensive.
1277   AdapterHandlerEntry* adapter = AdapterHandlerLibrary::get_adapter(mh);
1278   if (adapter == NULL ) {
1279     if (!is_init_completed()) {
1280       // Don't throw exceptions during VM initialization because java.lang.* classes
1281       // might not have been initialized, causing problems when constructing the
1282       // Java exception object.
1283       vm_exit_during_initialization("Out of space in CodeCache for adapters");
1284     } else {
1285       THROW_MSG_NULL(vmSymbols::java_lang_VirtualMachineError(), "Out of space in CodeCache for adapters");
1286     }
1287   }
1288 
1289   if (mh->is_shared()) {
1290     assert(mh->adapter() == adapter, "must be");

1299   }
1300   return adapter->get_c2i_entry();

1380 // Install compiled code.  Instantly it can execute.
1381 void Method::set_code(const methodHandle& mh, CompiledMethod *code) {
1382   assert_lock_strong(CompiledMethod_lock);
1383   assert( code, "use clear_code to remove code" );
1384   assert( mh->check_code(), "" );
1385 
1386   guarantee(mh->adapter() != NULL, "Adapter blob must already exist!");
1387 
1388   // These writes must happen in this order, because the interpreter will
1389   // directly jump to from_interpreted_entry which jumps to an i2c adapter
1390   // which jumps to _from_compiled_entry.
1391   mh->_code = code;             // Assign before allowing compiled code to exec
1392 
1393   int comp_level = code->comp_level();
1394   // In theory there could be a race here. In practice it is unlikely
1395   // and not worth worrying about.
1396   if (comp_level > mh->highest_comp_level()) {
1397     mh->set_highest_comp_level(comp_level);
1398   }

1400   OrderAccess::storestore();
1401   mh->_from_compiled_entry = code->verified_entry_point();
1402   mh->_from_compiled_value_entry = code->verified_value_entry_point();
1403   mh->_from_compiled_value_ro_entry = code->verified_value_ro_entry_point();
1404   OrderAccess::storestore();
1405   // Instantly compiled code can execute.
1406   if (!mh->is_method_handle_intrinsic())
1407     mh->_from_interpreted_entry = mh->get_i2c_entry();

1499 // Make an instance of a signature-polymorphic internal MH primitive.
1500 methodHandle Method::make_method_handle_intrinsic(vmIntrinsics::ID iid,
1501                                                          Symbol* signature,
1502                                                          TRAPS) {
1503   ResourceMark rm(THREAD);
1504   methodHandle empty;
1505 
1506   InstanceKlass* holder = SystemDictionary::MethodHandle_klass();
1507   Symbol* name = MethodHandles::signature_polymorphic_intrinsic_name(iid);
1508   assert(iid == MethodHandles::signature_polymorphic_name_id(name), "");
1509   if (TraceMethodHandles) {
1510     tty->print_cr("make_method_handle_intrinsic MH.%s%s", name->as_C_string(), signature->as_C_string());
1511   }
1512 
1513   // invariant:   cp->symbol_at_put is preceded by a refcount increment (more usually a lookup)
1514   name->increment_refcount();
1515   signature->increment_refcount();
1516 
1517   int cp_length = _imcp_limit;
1518   ClassLoaderData* loader_data = holder->class_loader_data();
1519   constantPoolHandle cp;
1520   {
1521     ConstantPool* cp_oop = ConstantPool::allocate(loader_data, cp_length, CHECK_(empty));
1522     cp = constantPoolHandle(THREAD, cp_oop);
1523   }
1524   cp->set_pool_holder(holder);
1525   cp->symbol_at_put(_imcp_invoke_name,       name);
1526   cp->symbol_at_put(_imcp_invoke_signature,  signature);
1527   cp->set_has_preresolution();
1528 
1529   // decide on access bits:  public or not?
1530   int flags_bits = (JVM_ACC_NATIVE | JVM_ACC_SYNTHETIC | JVM_ACC_FINAL);
1531   bool must_be_static = MethodHandles::is_signature_polymorphic_static(iid);
1532   if (must_be_static)  flags_bits |= JVM_ACC_STATIC;
1533   assert((flags_bits & JVM_ACC_PUBLIC) == 0, "do not expose these methods");
1534 
1535   methodHandle m;
1536   {
1537     InlineTableSizes sizes;
1538     Method* m_oop = Method::allocate(loader_data, 0,
1539                                      accessFlags_from(flags_bits), &sizes,
1540                                      ConstMethod::NORMAL, CHECK_(empty));
1541     m = methodHandle(THREAD, m_oop);
1542   }
1543   m->set_constants(cp());
1544   m->set_name_index(_imcp_invoke_name);
1545   m->set_signature_index(_imcp_invoke_signature);
1546   assert(MethodHandles::is_signature_polymorphic_name(m->name()), "");
1547   assert(m->signature() == signature, "");
1548   m->compute_from_signature(signature);
1549   m->init_intrinsic_id();
1550   assert(m->is_method_handle_intrinsic(), "");
1551 #ifdef ASSERT
1552   if (!MethodHandles::is_signature_polymorphic(m->intrinsic_id()))  m->print();
1553   assert(MethodHandles::is_signature_polymorphic(m->intrinsic_id()), "must be an invoker");
1554   assert(m->intrinsic_id() == iid, "correctly predicted iid");
1555 #endif //ASSERT
1556 
1557   // Finally, set up its entry points.
1558   assert(m->can_be_statically_bound(), "");
1559   m->set_vtable_index(Method::nonvirtual_vtable_index);
1560   m->link_method(m, CHECK_(empty));
1561 
1562   if (TraceMethodHandles && (Verbose || WizardMode)) {
1563     ttyLocker ttyl;
1564     m->print_on(tty);
1565   }
1566 
1567   return m;
1568 }

1598       localvariable_len,

1600       exception_table_len,

1602       method_parameters_len,
1603       cm->generic_signature_index(),
1604       method_annotations_len,
1605       parameter_annotations_len,
1606       type_annotations_len,
1607       default_annotations_len,
1608       0);
1609 
1610   ClassLoaderData* loader_data = m->method_holder()->class_loader_data();
1611   Method* newm_oop = Method::allocate(loader_data,
1612                                       new_code_length,
1613                                       flags,
1614                                       &sizes,
1615                                       m->method_type(),
1616                                       CHECK_(methodHandle()));
1617   methodHandle newm (THREAD, newm_oop);

1647   }

1652            method_parameters_len * sizeof(MethodParametersElement));
1653   }
1654   // Copy checked_exceptions
1655   if (checked_exceptions_len > 0) {
1656     memcpy(newm->checked_exceptions_start(),
1657            m->checked_exceptions_start(),
1658            checked_exceptions_len * sizeof(CheckedExceptionElement));
1659   }
1660   // Copy exception table
1661   if (exception_table_len > 0) {
1662     memcpy(newm->exception_table_start(),
1663            m->exception_table_start(),
1664            exception_table_len * sizeof(ExceptionTableElement));
1665   }
1666   // Copy local variable number table
1667   if (localvariable_len > 0) {
1668     memcpy(newm->localvariable_table_start(),
1669            m->localvariable_table_start(),
1670            localvariable_len * sizeof(LocalVariableTableElement));
1671   }
1672   // Copy stackmap table
1673   if (m->has_stackmap_table()) {
1674     int code_attribute_length = m->stackmap_data()->length();

1676       MetadataFactory::new_array<u1>(loader_data, code_attribute_length, 0, CHECK_(methodHandle()));

1678            (void*)m->stackmap_data()->adr_at(0), code_attribute_length);

1680   }
1681 
1682   // copy annotations over to new method
1683   newcm->copy_annotations_from(loader_data, cm, CHECK_(methodHandle()));
1684   return newm;
1685 }

1702 void Method::init_intrinsic_id() {
1703   assert(_intrinsic_id == vmIntrinsics::_none, "do this just once");
1704   const uintptr_t max_id_uint = right_n_bits((int)(sizeof(_intrinsic_id) * BitsPerByte));
1705   assert((uintptr_t)vmIntrinsics::ID_LIMIT <= max_id_uint, "else fix size");
1706   assert(intrinsic_id_size_in_bytes() == sizeof(_intrinsic_id), "");
1707 
1708   // the klass name is well-known:
1709   vmSymbols::SID klass_id = klass_id_for_intrinsics(method_holder());
1710   assert(klass_id != vmSymbols::NO_SID, "caller responsibility");
1711 
1712   // ditto for method and signature:
1713   vmSymbols::SID  name_id = vmSymbols::find_sid(name());
1714   if (klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_MethodHandle)
1715       && klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_VarHandle)
1716       && name_id == vmSymbols::NO_SID) {
1717     return;
1718   }
1719   vmSymbols::SID   sig_id = vmSymbols::find_sid(signature());
1720   if (klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_MethodHandle)
1721       && klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_VarHandle)
1722       && sig_id == vmSymbols::NO_SID) {
1723     return;
1724   }
1725   jshort flags = access_flags().as_short();
1726 
1727   vmIntrinsics::ID id = vmIntrinsics::find_id(klass_id, name_id, sig_id, flags);
1728   if (id != vmIntrinsics::_none) {
1729     set_intrinsic_id(id);
1730     if (id == vmIntrinsics::_Class_cast) {
1731       // Even if the intrinsic is rejected, we want to inline this simple method.
1732       set_force_inline(true);
1733     }
1734     return;
1735   }
1736 
1737   // A few slightly irregular cases:
1738   switch (klass_id) {
1739   case vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_StrictMath):
1740     // Second chance: check in regular Math.
1741     switch (name_id) {
1742     case vmSymbols::VM_SYMBOL_ENUM_NAME(min_name):
1743     case vmSymbols::VM_SYMBOL_ENUM_NAME(max_name):
1744     case vmSymbols::VM_SYMBOL_ENUM_NAME(sqrt_name):
1745       // pretend it is the corresponding method in the non-strict class:
1746       klass_id = vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_Math);
1747       id = vmIntrinsics::find_id(klass_id, name_id, sig_id, flags);
1748       break;
1749     default:
1750       break;
1751     }
1752     break;
1753 
1754   // Signature-polymorphic methods: MethodHandle.invoke*, InvokeDynamic.*., VarHandle
1755   case vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_MethodHandle):
1756   case vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_VarHandle):
1757     if (!is_native())  break;
1758     id = MethodHandles::signature_polymorphic_name_id(method_holder(), name());
1759     if (is_static() != MethodHandles::is_signature_polymorphic_static(id))
1760       id = vmIntrinsics::_none;
1761     break;
1762 
1763   default:
1764     break;
1765   }
1766 
1767   if (id != vmIntrinsics::_none) {
1768     // Set up its iid.  It is an alias method.
1769     set_intrinsic_id(id);
1770     return;
1771   }
1772 }

1978 static void clear_matches(Method* m, int bci) {
1979   InstanceKlass* ik = m->method_holder();
1980   BreakpointInfo* prev_bp = NULL;
1981   BreakpointInfo* next_bp;
1982   for (BreakpointInfo* bp = ik->breakpoints(); bp != NULL; bp = next_bp) {
1983     next_bp = bp->next();
1984     // bci value of -1 is used to delete all breakpoints in method m (ex: clear_all_breakpoint).
1985     if (bci >= 0 ? bp->match(m, bci) : bp->match(m)) {
1986       // do this first:
1987       bp->clear(m);
1988       // unhook it
1989       if (prev_bp != NULL)
1990         prev_bp->set_next(next_bp);
1991       else
1992         ik->set_breakpoints(next_bp);
1993       delete bp;
1994       // When class is redefined JVMTI sets breakpoint in all versions of EMCP methods
1995       // at same location. So we have multiple matching (method_index and bci)
1996       // BreakpointInfo nodes in BreakpointInfo list. We should just delete one
1997       // breakpoint for clear_breakpoint request and keep all other method versions
1998       // BreakpointInfo for future clear_breakpoint request.
1999       // bcivalue of -1 is used to clear all breakpoints (see clear_all_breakpoints)
2000       // which is being called when class is unloaded. We delete all the Breakpoint
2001       // information for all versions of method. We may not correctly restore the original
2002       // bytecode in all method versions, but that is ok. Because the class is being unloaded
2003       // so these methods won't be used anymore.
2004       if (bci >= 0) {
2005         break;
2006       }
2007     } else {
2008       // This one is a keeper.
2009       prev_bp = bp;
2010     }
2011   }
2012 }

2138 class JNIMethodBlockNode : public CHeapObj<mtClass> {

2166 class JNIMethodBlock : public CHeapObj<mtClass> {
2167   JNIMethodBlockNode _head;
2168   JNIMethodBlockNode *_last_free;
2169  public:
2170   static Method* const _free_method;
2171 
2172   JNIMethodBlock(int initial_capacity = min_block_size)
2173       : _head(initial_capacity), _last_free(&_head) {}
2174 
2175   void ensure_methods(int num_addl_methods) {
2176     _last_free->ensure_methods(num_addl_methods);
2177   }
2178 
2179   Method** add_method(Method* m) {
2180     for (JNIMethodBlockNode* b = _last_free; b != NULL; b = b->_next) {
2181       if (b->_top < b->_number_of_methods) {
2182         // top points to the next free entry.
2183         int i = b->_top;
2184         b->_methods[i] = m;
2185         b->_top++;
2186         _last_free = b;
2187         return &(b->_methods[i]);
2188       } else if (b->_top == b->_number_of_methods) {
2189         // if the next free entry ran off the block see if there's a free entry
2190         for (int i = 0; i < b->_number_of_methods; i++) {
2191           if (b->_methods[i] == _free_method) {
2192             b->_methods[i] = m;
2193             _last_free = b;
2194             return &(b->_methods[i]);
2195           }
2196         }
2197         // Only check each block once for frees.  They're very unlikely.
2198         // Increment top past the end of the block.
2199         b->_top++;
2200       }
2201       // need to allocate a next block.
2202       if (b->_next == NULL) {
2203         b->_next = _last_free = new JNIMethodBlockNode();
2204       }
2205     }
2206     guarantee(false, "Should always allocate a free block");
2207     return NULL;
2208   }
2209 
2210   bool contains(Method** m) {
2211     if (m == NULL) return false;
2212     for (JNIMethodBlockNode* b = &_head; b != NULL; b = b->_next) {
2213       if (b->_methods <= m && m < b->_methods + b->_number_of_methods) {
2214         // This is a bit of extra checking, for two reasons.  One is
2215         // that contains() deals with pointers that are passed in by
2216         // JNI code, so making sure that the pointer is aligned
2217         // correctly is valuable.  The other is that <= and > are
2218         // technically not defined on pointers, so the if guard can
2219         // pass spuriously; no modern compiler is likely to make that
2220         // a problem, though (and if one did, the guard could also
2221         // fail spuriously, which would be bad).
2222         ptrdiff_t idx = m - b->_methods;
2223         if (b->_methods + idx == m) {
2224           return true;
2225         }
2226       }
2227     }
2228     return false;  // not found
2229   }

2246     }
2247   }

2526 // TouchedMethodRecord -- we can't use a HashtableEntry<Method*> because
2527 // the Method may be garbage collected. Let's roll our own hash table.
2528 class TouchedMethodRecord : CHeapObj<mtTracing> {
2529 public:
2530   // It's OK to store Symbols here because they will NOT be GC'ed if
2531   // LogTouchedMethods is enabled.
2532   TouchedMethodRecord* _next;
2533   Symbol* _class_name;
2534   Symbol* _method_name;
2535   Symbol* _method_signature;
2536 };
2537 
2538 static const int TOUCHED_METHOD_TABLE_SIZE = 20011;
2539 static TouchedMethodRecord** _touched_method_table = NULL;
2540 
2541 void Method::log_touched(TRAPS) {
2542 
2543   const int table_size = TOUCHED_METHOD_TABLE_SIZE;
2544   Symbol* my_class = klass_name();
2545   Symbol* my_name  = name();
2546   Symbol* my_sig   = signature();
2547 
2548   unsigned int hash = my_class->identity_hash() +
2549                       my_name->identity_hash() +
2550                       my_sig->identity_hash();
2551   juint index = juint(hash) % table_size;
2552 
2553   MutexLocker ml(THREAD, TouchedMethodLog_lock);
2554   if (_touched_method_table == NULL) {
2555     _touched_method_table = NEW_C_HEAP_ARRAY2(TouchedMethodRecord*, table_size,
2556                                               mtTracing, CURRENT_PC);
2557     memset(_touched_method_table, 0, sizeof(TouchedMethodRecord*)*table_size);
2558   }
2559 
2560   TouchedMethodRecord* ptr = _touched_method_table[index];
2561   while (ptr) {
2562     if (ptr->_class_name       == my_class &&
2563         ptr->_method_name      == my_name &&
2564         ptr->_method_signature == my_sig) {
2565       return;
2566     }
2567     if (ptr->_next == NULL) break;
2568     ptr = ptr->_next;
2569   }
2570   TouchedMethodRecord* nptr = NEW_C_HEAP_OBJ(TouchedMethodRecord, mtTracing);
2571   my_class->increment_refcount();
2572   my_name->increment_refcount();
2573   my_sig->increment_refcount();
2574   nptr->_class_name         = my_class;
2575   nptr->_method_name        = my_name;
2576   nptr->_method_signature   = my_sig;
2577   nptr->_next               = NULL;
2578 
2579   if (ptr == NULL) {
2580     // first
2581     _touched_method_table[index] = nptr;
2582   } else {
2583     ptr->_next = nptr;
2584   }
2585 }
