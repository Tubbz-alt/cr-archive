 132 void BlockListBuilder::set_entries(int osr_bci) {
 133   // generate start blocks
 134   BlockBegin* std_entry = make_block_at(0, NULL);
 135   if (scope()->caller() == NULL) {
 136     std_entry->set(BlockBegin::std_entry_flag);
 137   }
 138   if (osr_bci != -1) {
 139     BlockBegin* osr_entry = make_block_at(osr_bci, NULL);
 140     osr_entry->set(BlockBegin::osr_entry_flag);
 141   }
 142 
 143   // generate exception entry blocks
 144   XHandlers* list = xhandlers();
 145   const int n = list->length();
 146   for (int i = 0; i < n; i++) {
 147     XHandler* h = list->handler_at(i);
 148     BlockBegin* entry = make_block_at(h->handler_bci(), NULL);
 149     entry->set(BlockBegin::exception_entry_flag);
 150     h->set_entry_block(entry);
 151   }
 152 }

 155 BlockBegin* BlockListBuilder::make_block_at(int cur_bci, BlockBegin* predecessor) {
 156   assert(method()->bci_block_start().at(cur_bci), "wrong block starts of MethodLivenessAnalyzer");
 157 
 158   BlockBegin* block = _bci2block->at(cur_bci);
 159   if (block == NULL) {
 160     block = new BlockBegin(cur_bci);
 161     block->init_stores_to_locals(method()->max_locals());
 162     _bci2block->at_put(cur_bci, block);
 163     _blocks.append(block);
 164 
 165     assert(predecessor == NULL || predecessor->bci() < cur_bci, "targets for backward branches must already exist");
 166   }
 167 
 168   if (predecessor != NULL) {
 169     if (block->is_set(BlockBegin::exception_entry_flag)) {
 170       BAILOUT_("Exception handler can be reached by both normal and exceptional control flow", block);
 171     }
 172 
 173     predecessor->add_successor(block);
 174     block->increment_total_preds();
 175   }
 176 
 177   return block;
 178 }

 181 inline void BlockListBuilder::store_one(BlockBegin* current, int local) {
 182   current->stores_to_locals().set_bit(local);

 184 inline void BlockListBuilder::store_two(BlockBegin* current, int local) {
 185   store_one(current, local);
 186   store_one(current, local + 1);

 190 void BlockListBuilder::handle_exceptions(BlockBegin* current, int cur_bci) {
 191   // Draws edges from a block to its exception handlers
 192   XHandlers* list = xhandlers();
 193   const int n = list->length();
 194 
 195   for (int i = 0; i < n; i++) {
 196     XHandler* h = list->handler_at(i);
 197 
 198     if (h->covers(cur_bci)) {
 199       BlockBegin* entry = h->entry_block();
 200       assert(entry != NULL && entry == _bci2block->at(h->handler_bci()), "entry must be set");
 201       assert(entry->is_set(BlockBegin::exception_entry_flag), "flag must be set");
 202 
 203       // add each exception handler only once
 204       if (!current->is_successor(entry)) {
 205         current->add_successor(entry);
 206         entry->increment_total_preds();
 207       }
 208 
 209       // stop when reaching catchall
 210       if (h->catch_type() == 0) break;
 211     }
 212   }
 213 }

 215 void BlockListBuilder::handle_jsr(BlockBegin* current, int sr_bci, int next_bci) {
 216   // start a new block after jsr-bytecode and link this block into cfg
 217   make_block_at(next_bci, current);
 218 
 219   // start a new block at the subroutine entry at mark it with special flag
 220   BlockBegin* sr_block = make_block_at(sr_bci, current);
 221   if (!sr_block->is_set(BlockBegin::subroutine_entry_flag)) {
 222     sr_block->set(BlockBegin::subroutine_entry_flag);
 223   }
 224 }

 227 void BlockListBuilder::set_leaders() {
 228   bool has_xhandlers = xhandlers()->has_handlers();
 229   BlockBegin* current = NULL;
 230 
 231   // The information which bci starts a new block simplifies the analysis
 232   // Without it, backward branches could jump to a bci where no block was created
 233   // during bytecode iteration. This would require the creation of a new block at the
 234   // branch target and a modification of the successor lists.
 235   const BitMap& bci_block_start = method()->bci_block_start();
 236 
 237   ciBytecodeStream s(method());
 238   while (s.next() != ciBytecodeStream::EOBC()) {
 239     int cur_bci = s.cur_bci();
 240 
 241     if (bci_block_start.at(cur_bci)) {
 242       current = make_block_at(cur_bci, current);
 243     }
 244     assert(current != NULL, "must have current block");
 245 
 246     if (has_xhandlers && GraphBuilder::can_trap(method(), s.cur_bc())) {
 247       handle_exceptions(current, cur_bci);
 248     }
 249 
 250     switch (s.cur_bc()) {
 251       // track stores to local variables for selective creation of phi functions
 252       case Bytecodes::_iinc:     store_one(current, s.get_index()); break;
 253       case Bytecodes::_istore:   store_one(current, s.get_index()); break;
 254       case Bytecodes::_lstore:   store_two(current, s.get_index()); break;
 255       case Bytecodes::_fstore:   store_one(current, s.get_index()); break;
 256       case Bytecodes::_dstore:   store_two(current, s.get_index()); break;
 257       case Bytecodes::_astore:   store_one(current, s.get_index()); break;
 258       case Bytecodes::_istore_0: store_one(current, 0); break;
 259       case Bytecodes::_istore_1: store_one(current, 1); break;
 260       case Bytecodes::_istore_2: store_one(current, 2); break;
 261       case Bytecodes::_istore_3: store_one(current, 3); break;
 262       case Bytecodes::_lstore_0: store_two(current, 0); break;
 263       case Bytecodes::_lstore_1: store_two(current, 1); break;
 264       case Bytecodes::_lstore_2: store_two(current, 2); break;
 265       case Bytecodes::_lstore_3: store_two(current, 3); break;
 266       case Bytecodes::_fstore_0: store_one(current, 0); break;
 267       case Bytecodes::_fstore_1: store_one(current, 1); break;
 268       case Bytecodes::_fstore_2: store_one(current, 2); break;
 269       case Bytecodes::_fstore_3: store_one(current, 3); break;
 270       case Bytecodes::_dstore_0: store_two(current, 0); break;
 271       case Bytecodes::_dstore_1: store_two(current, 1); break;
 272       case Bytecodes::_dstore_2: store_two(current, 2); break;
 273       case Bytecodes::_dstore_3: store_two(current, 3); break;
 274       case Bytecodes::_astore_0: store_one(current, 0); break;
 275       case Bytecodes::_astore_1: store_one(current, 1); break;
 276       case Bytecodes::_astore_2: store_one(current, 2); break;
 277       case Bytecodes::_astore_3: store_one(current, 3); break;
 278 
 279       // track bytecodes that affect the control flow
 280       case Bytecodes::_athrow:  // fall through
 281       case Bytecodes::_ret:     // fall through
 282       case Bytecodes::_ireturn: // fall through
 283       case Bytecodes::_lreturn: // fall through
 284       case Bytecodes::_freturn: // fall through
 285       case Bytecodes::_dreturn: // fall through
 286       case Bytecodes::_areturn: // fall through
 287       case Bytecodes::_return:
 288         current = NULL;
 289         break;
 290 
 291       case Bytecodes::_ifeq:      // fall through
 292       case Bytecodes::_ifne:      // fall through
 293       case Bytecodes::_iflt:      // fall through
 294       case Bytecodes::_ifge:      // fall through
 295       case Bytecodes::_ifgt:      // fall through
 296       case Bytecodes::_ifle:      // fall through
 297       case Bytecodes::_if_icmpeq: // fall through
 298       case Bytecodes::_if_icmpne: // fall through
 299       case Bytecodes::_if_icmplt: // fall through
 300       case Bytecodes::_if_icmpge: // fall through
 301       case Bytecodes::_if_icmpgt: // fall through
 302       case Bytecodes::_if_icmple: // fall through
 303       case Bytecodes::_if_acmpeq: // fall through
 304       case Bytecodes::_if_acmpne: // fall through
 305       case Bytecodes::_ifnull:    // fall through
 306       case Bytecodes::_ifnonnull:
 307         make_block_at(s.next_bci(), current);
 308         make_block_at(s.get_dest(), current);
 309         current = NULL;
 310         break;
 311 
 312       case Bytecodes::_goto:
 313         make_block_at(s.get_dest(), current);
 314         current = NULL;
 315         break;
 316 
 317       case Bytecodes::_goto_w:
 318         make_block_at(s.get_far_dest(), current);
 319         current = NULL;
 320         break;
 321 
 322       case Bytecodes::_jsr:
 323         handle_jsr(current, s.get_dest(), s.next_bci());
 324         current = NULL;
 325         break;
 326 
 327       case Bytecodes::_jsr_w:
 328         handle_jsr(current, s.get_far_dest(), s.next_bci());
 329         current = NULL;
 330         break;
 331 
 332       case Bytecodes::_tableswitch: {
 333         // set block for each case
 334         Bytecode_tableswitch sw(&s);
 335         int l = sw.length();
 336         for (int i = 0; i < l; i++) {
 337           make_block_at(cur_bci + sw.dest_offset_at(i), current);
 338         }
 339         make_block_at(cur_bci + sw.default_offset(), current);
 340         current = NULL;
 341         break;
 342       }
 343 
 344       case Bytecodes::_lookupswitch: {
 345         // set block for each case
 346         Bytecode_lookupswitch sw(&s);
 347         int l = sw.number_of_pairs();
 348         for (int i = 0; i < l; i++) {
 349           make_block_at(cur_bci + sw.pair_at(i).offset(), current);
 350         }
 351         make_block_at(cur_bci + sw.default_offset(), current);
 352         current = NULL;
 353         break;
 354       }
 355 
 356       default:
 357         break;
 358     }
 359   }
 360 }

 363 void BlockListBuilder::mark_loops() {
 364   ResourceMark rm;
 365 
 366   _active.initialize(BlockBegin::number_of_blocks());
 367   _visited.initialize(BlockBegin::number_of_blocks());
 368   _loop_map = intArray(BlockBegin::number_of_blocks(), BlockBegin::number_of_blocks(), 0);
 369   _next_loop_index = 0;
 370   _next_block_number = _blocks.length();
 371 
 372   // recursively iterate the control flow graph
 373   mark_loops(_bci2block->at(0), false);
 374   assert(_next_block_number >= 0, "invalid block numbers");
 375 
 376   // Remove dangling Resource pointers before the ResourceMark goes out-of-scope.
 377   _active.resize(0);
 378   _visited.resize(0);
 379 }

 381 void BlockListBuilder::make_loop_header(BlockBegin* block) {
 382   if (block->is_set(BlockBegin::exception_entry_flag)) {
 383     // exception edges may look like loops but don't mark them as such
 384     // since it screws up block ordering.
 385     return;
 386   }
 387   if (!block->is_set(BlockBegin::parser_loop_header_flag)) {
 388     block->set(BlockBegin::parser_loop_header_flag);
 389 
 390     assert(_loop_map.at(block->block_id()) == 0, "must not be set yet");
 391     assert(0 <= _next_loop_index && _next_loop_index < BitsPerInt, "_next_loop_index is used as a bit-index in integer");
 392     _loop_map.at_put(block->block_id(), 1 << _next_loop_index);
 393     if (_next_loop_index < 31) _next_loop_index++;
 394   } else {
 395     // block already marked as loop header
 396     assert(is_power_of_2((unsigned int)_loop_map.at(block->block_id())), "exactly one bit must be set");
 397   }
 398 }

 400 int BlockListBuilder::mark_loops(BlockBegin* block, bool in_subroutine) {
 401   int block_id = block->block_id();
 402 
 403   if (_visited.at(block_id)) {
 404     if (_active.at(block_id)) {
 405       // reached block via backward branch
 406       make_loop_header(block);
 407     }
 408     // return cached loop information for this block
 409     return _loop_map.at(block_id);
 410   }
 411 
 412   if (block->is_set(BlockBegin::subroutine_entry_flag)) {
 413     in_subroutine = true;
 414   }
 415 
 416   // set active and visited bits before successors are processed
 417   _visited.set_bit(block_id);
 418   _active.set_bit(block_id);
 419 
 420   intptr_t loop_state = 0;
 421   for (int i = block->number_of_sux() - 1; i >= 0; i--) {
 422     // recursively process all successors
 423     loop_state |= mark_loops(block->sux_at(i), in_subroutine);
 424   }
 425 
 426   // clear active-bit after all successors are processed
 427   _active.clear_bit(block_id);
 428 
 429   // reverse-post-order numbering of all blocks
 430   block->set_depth_first_number(_next_block_number);
 431   _next_block_number--;
 432 
 433   if (loop_state != 0 || in_subroutine ) {
 434     // block is contained at least in one loop, so phi functions are necessary
 435     // phi functions are also necessary for all locals stored in a subroutine
 436     scope()->requires_phi_function().set_union(block->stores_to_locals());
 437   }
 438 
 439   if (block->is_set(BlockBegin::parser_loop_header_flag)) {
 440     int header_loop_state = _loop_map.at(block_id);
 441     assert(is_power_of_2((unsigned)header_loop_state), "exactly one bit must be set");
 442 
 443     // If the highest bit is set (i.e. when integer value is negative), the method
 444     // has 32 or more loops. This bit is never cleared because it is used for multiple loops
 445     if (header_loop_state >= 0) {
 446       clear_bits(loop_state, header_loop_state);
 447     }
 448   }
 449 
 450   // cache and return loop information for this block
 451   _loop_map.at_put(block_id, loop_state);
 452   return loop_state;
 453 }

 462 void BlockListBuilder::print() {
 463   tty->print("----- initial block list of BlockListBuilder for method ");
 464   method()->print_short_name();
 465   tty->cr();
 466 
 467   // better readability if blocks are sorted in processing order
 468   _blocks.sort(compare_depth_first);
 469 
 470   for (int i = 0; i < _blocks.length(); i++) {
 471     BlockBegin* cur = _blocks.at(i);
 472     tty->print("%4d: B%-4d bci: %-4d  preds: %-4d ", cur->depth_first_number(), cur->block_id(), cur->bci(), cur->total_preds());
 473 
 474     tty->print(cur->is_set(BlockBegin::std_entry_flag)               ? " std" : "    ");
 475     tty->print(cur->is_set(BlockBegin::osr_entry_flag)               ? " osr" : "    ");
 476     tty->print(cur->is_set(BlockBegin::exception_entry_flag)         ? " ex" : "   ");
 477     tty->print(cur->is_set(BlockBegin::subroutine_entry_flag)        ? " sr" : "   ");
 478     tty->print(cur->is_set(BlockBegin::parser_loop_header_flag)      ? " lh" : "   ");
 479 
 480     if (cur->number_of_sux() > 0) {
 481       tty->print("    sux: ");
 482       for (int j = 0; j < cur->number_of_sux(); j++) {
 483         BlockBegin* sux = cur->sux_at(j);
 484         tty->print("B%d ", sux->block_id());
 485       }
 486     }
 487     tty->cr();
 488   }
 489 }

 494 // A simple growable array of Values indexed by ciFields
 495 class FieldBuffer: public CompilationResourceObj {
 496  private:
 497   GrowableArray<Value> _values;
 498 
 499  public:
 500   FieldBuffer() {}
 501 
 502   void kill() {
 503     _values.trunc_to(0);
 504   }
 505 
 506   Value at(ciField* field) {
 507     assert(field->holder()->is_loaded(), "must be a loaded field");
 508     int offset = field->offset();
 509     if (offset < _values.length()) {
 510       return _values.at(offset);
 511     } else {
 512       return NULL;
 513     }
 514   }
 515 
 516   void at_put(ciField* field, Value value) {
 517     assert(field->holder()->is_loaded(), "must be a loaded field");
 518     int offset = field->offset();
 519     _values.at_put_grow(offset, value, NULL);
 520   }
 521 
 522 };

 525 // MemoryBuffer is fairly simple model of the current state of memory.
 526 // It partitions memory into several pieces.  The first piece is
 527 // generic memory where little is known about the owner of the memory.
 528 // This is conceptually represented by the tuple <O, F, V> which says
 529 // that the field F of object O has value V.  This is flattened so
 530 // that F is represented by the offset of the field and the parallel
 531 // arrays _objects and _values are used for O and V.  Loads of O.F can
 532 // simply use V.  Newly allocated objects are kept in a separate list
 533 // along with a parallel array for each object which represents the
 534 // current value of its fields.  Stores of the default value to fields
 535 // which have never been stored to before are eliminated since they
 536 // are redundant.  Once newly allocated objects are stored into
 537 // another object or they are passed out of the current compile they
 538 // are treated like generic memory.
 539 
 540 class MemoryBuffer: public CompilationResourceObj {
 541  private:
 542   FieldBuffer                 _values;
 543   GrowableArray<Value>        _objects;
 544   GrowableArray<Value>        _newobjects;
 545   GrowableArray<FieldBuffer*> _fields;
 546 
 547  public:
 548   MemoryBuffer() {}
 549 
 550   StoreField* store(StoreField* st) {
 551     if (!EliminateFieldAccess) {
 552       return st;
 553     }
 554 
 555     Value object = st->obj();
 556     Value value = st->value();
 557     ciField* field = st->field();
 558     if (field->holder()->is_loaded()) {
 559       int offset = field->offset();
 560       int index = _newobjects.find(object);
 561       if (index != -1) {
 562         // newly allocated object with no other stores performed on this field
 563         FieldBuffer* buf = _fields.at(index);
 564         if (buf->at(field) == NULL && is_default_value(value)) {
 565 #ifndef PRODUCT
 566           if (PrintIRDuringConstruction && Verbose) {
 567             tty->print_cr("Eliminated store for object %d:", index);
 568             st->print_line();
 569           }
 570 #endif
 571           return NULL;
 572         } else {
 573           buf->at_put(field, value);
 574         }
 575       } else {
 576         _objects.at_put_grow(offset, object, NULL);
 577         _values.at_put(field, value);
 578       }
 579 
 580       store_value(value);
 581     } else {
 582       // if we held onto field names we could alias based on names but
 583       // we don't know what's being stored to so kill it all.
 584       kill();
 585     }
 586     return st;
 587   }
 588 
 589 
 590   // return true if this value correspond to the default value of a field.
 591   bool is_default_value(Value value) {
 592     Constant* con = value->as_Constant();
 593     if (con) {
 594       switch (con->type()->tag()) {
 595         case intTag:    return con->type()->as_IntConstant()->value() == 0;
 596         case longTag:   return con->type()->as_LongConstant()->value() == 0;
 597         case floatTag:  return jint_cast(con->type()->as_FloatConstant()->value()) == 0;
 598         case doubleTag: return jlong_cast(con->type()->as_DoubleConstant()->value()) == jlong_cast(0);
 599         case objectTag: return con->type() == objectNull;
 600         default:  ShouldNotReachHere();
 601       }
 602     }
 603     return false;
 604   }
 605 
 606 
 607   // return either the actual value of a load or the load itself
 608   Value load(LoadField* load) {
 609     if (!EliminateFieldAccess) {
 610       return load;
 611     }
 612 
 613     if (strict_fp_requires_explicit_rounding && load->type()->is_float_kind()) {
 614 #ifdef IA32
 615       if (UseSSE < 2) {
 616         // can't skip load since value might get rounded as a side effect
 617         return load;
 618       }
 619 #else
 620       Unimplemented();
 621 #endif // IA32
 622     }
 623 
 624     ciField* field = load->field();
 625     Value object   = load->obj();
 626     if (field->holder()->is_loaded() && !field->is_volatile()) {
 627       int offset = field->offset();
 628       Value result = NULL;
 629       int index = _newobjects.find(object);
 630       if (index != -1) {
 631         result = _fields.at(index)->at(field);
 632       } else if (_objects.at_grow(offset, NULL) == object) {
 633         result = _values.at(field);
 634       }
 635       if (result != NULL) {
 636 #ifndef PRODUCT
 637         if (PrintIRDuringConstruction && Verbose) {
 638           tty->print_cr("Eliminated load: ");
 639           load->print_line();
 640         }
 641 #endif
 642         assert(result->type()->tag() == load->type()->tag(), "wrong types");
 643         return result;
 644       }
 645     }
 646     return load;
 647   }
 648 
 649   // Record this newly allocated object
 650   void new_instance(NewInstance* object) {
 651     int index = _newobjects.length();
 652     _newobjects.append(object);
 653     if (_fields.at_grow(index, NULL) == NULL) {
 654       _fields.at_put(index, new FieldBuffer());
 655     } else {
 656       _fields.at(index)->kill();
 657     }
 658   }
 659 
 660   // Record this newly allocated object
 661   void new_instance(NewValueTypeInstance* object) {
 662     int index = _newobjects.length();
 663     _newobjects.append(object);
 664     if (_fields.at_grow(index, NULL) == NULL) {
 665       _fields.at_put(index, new FieldBuffer());
 666     } else {
 667       _fields.at(index)->kill();
 668     }
 669   }
 670 
 671   void store_value(Value value) {
 672     int index = _newobjects.find(value);
 673     if (index != -1) {
 674       // stored a newly allocated object into another object.
 675       // Assume we've lost track of it as separate slice of memory.
 676       // We could do better by keeping track of whether individual
 677       // fields could alias each other.
 678       _newobjects.remove_at(index);
 679       // pull out the field info and store it at the end up the list
 680       // of field info list to be reused later.
 681       _fields.append(_fields.at(index));
 682       _fields.remove_at(index);
 683     }
 684   }
 685 
 686   void kill() {
 687     _newobjects.trunc_to(0);
 688     _objects.trunc_to(0);
 689     _values.kill();
 690   }
 691 };

 732 BlockBegin* GraphBuilder::ScopeData::block_at(int bci) {
 733   if (parsing_jsr()) {
 734     // It is necessary to clone all blocks associated with a
 735     // subroutine, including those for exception handlers in the scope
 736     // of the method containing the jsr (because those exception
 737     // handlers may contain ret instructions in some cases).
 738     BlockBegin* block = bci2block()->at(bci);
 739     if (block != NULL && block == parent()->bci2block()->at(bci)) {
 740       BlockBegin* new_block = new BlockBegin(block->bci());
 741       if (PrintInitialBlockList) {
 742         tty->print_cr("CFG: cloned block %d (bci %d) as block %d for jsr",
 743                       block->block_id(), block->bci(), new_block->block_id());
 744       }
 745       // copy data from cloned blocked
 746       new_block->set_depth_first_number(block->depth_first_number());
 747       if (block->is_set(BlockBegin::parser_loop_header_flag)) new_block->set(BlockBegin::parser_loop_header_flag);
 748       // Preserve certain flags for assertion checking
 749       if (block->is_set(BlockBegin::subroutine_entry_flag)) new_block->set(BlockBegin::subroutine_entry_flag);
 750       if (block->is_set(BlockBegin::exception_entry_flag))  new_block->set(BlockBegin::exception_entry_flag);
 751 
 752       // copy was_visited_flag to allow early detection of bailouts
 753       // if a block that is used in a jsr has already been visited before,
 754       // it is shared between the normal control flow and a subroutine
 755       // BlockBegin::try_merge returns false when the flag is set, this leads
 756       // to a compilation bailout
 757       if (block->is_set(BlockBegin::was_visited_flag))  new_block->set(BlockBegin::was_visited_flag);
 758 
 759       bci2block()->at_put(bci, new_block);
 760       block = new_block;
 761     }
 762     return block;
 763   } else {
 764     return bci2block()->at(bci);
 765   }
 766 }

 823 void GraphBuilder::sort_top_into_worklist(BlockList* worklist, BlockBegin* top) {
 824   assert(worklist->top() == top, "");
 825   // sort block descending into work list
 826   const int dfn = top->depth_first_number();
 827   assert(dfn != -1, "unknown depth first number");
 828   int i = worklist->length()-2;
 829   while (i >= 0) {
 830     BlockBegin* b = worklist->at(i);
 831     if (b->depth_first_number() < dfn) {
 832       worklist->at_put(i+1, b);
 833     } else {
 834       break;
 835     }
 836     i --;
 837   }
 838   if (i >= -1) worklist->at_put(i + 1, top);
 839 }

 896 void GraphBuilder::load_constant() {
 897   ciConstant con = stream()->get_constant();
 898   if (con.basic_type() == T_ILLEGAL) {
 899     // FIXME: an unresolved Dynamic constant can get here,
 900     // and that should not terminate the whole compilation.
 901     BAILOUT("could not resolve a constant");
 902   } else {
 903     ValueType* t = illegalType;
 904     ValueStack* patch_state = NULL;
 905     switch (con.basic_type()) {
 906       case T_BOOLEAN: t = new IntConstant     (con.as_boolean()); break;
 907       case T_BYTE   : t = new IntConstant     (con.as_byte   ()); break;
 908       case T_CHAR   : t = new IntConstant     (con.as_char   ()); break;
 909       case T_SHORT  : t = new IntConstant     (con.as_short  ()); break;
 910       case T_INT    : t = new IntConstant     (con.as_int    ()); break;
 911       case T_LONG   : t = new LongConstant    (con.as_long   ()); break;
 912       case T_FLOAT  : t = new FloatConstant   (con.as_float  ()); break;
 913       case T_DOUBLE : t = new DoubleConstant  (con.as_double ()); break;
 914       case T_ARRAY  : t = new ArrayConstant   (con.as_object ()->as_array   ()); break;
 915       case T_OBJECT :
 916        {
 917         ciObject* obj = con.as_object();
 918         if (!obj->is_loaded()
 919             || (PatchALot && obj->klass() != ciEnv::current()->String_klass())) {
 920           // A Class, MethodType, MethodHandle, or String.
 921           // Unloaded condy nodes show up as T_ILLEGAL, above.
 922           patch_state = copy_state_before();
 923           t = new ObjectConstant(obj);
 924         } else {
 925           // Might be a Class, MethodType, MethodHandle, or Dynamic constant
 926           // result, which might turn out to be an array.
 927           if (obj->is_null_object())
 928             t = objectNull;
 929           else if (obj->is_array())
 930             t = new ArrayConstant(obj->as_array());
 931           else
 932             t = new InstanceConstant(obj->as_instance());
 933         }
 934         break;
 935        }
 936       default       : ShouldNotReachHere();
 937     }
 938     Value x;
 939     if (patch_state != NULL) {
 940       x = new Constant(t, patch_state);
 941     } else {
 942       x = new Constant(t);
 943     }
 944     push(t, append(x));
 945   }
 946 }

 962 void GraphBuilder::store_local(ValueStack* state, Value x, int index) {
 963   if (parsing_jsr()) {
 964     // We need to do additional tracking of the location of the return
 965     // address for jsrs since we don't handle arbitrary jsr/ret
 966     // constructs. Here we are figuring out in which circumstances we
 967     // need to bail out.
 968     if (x->type()->is_address()) {
 969       scope_data()->set_jsr_return_address_local(index);
 970 
 971       // Also check parent jsrs (if any) at this time to see whether
 972       // they are using this local. We don't handle skipping over a
 973       // ret.
 974       for (ScopeData* cur_scope_data = scope_data()->parent();
 975            cur_scope_data != NULL && cur_scope_data->parsing_jsr() && cur_scope_data->scope() == scope();
 976            cur_scope_data = cur_scope_data->parent()) {
 977         if (cur_scope_data->jsr_return_address_local() == index) {
 978           BAILOUT("subroutine overwrites return address from previous subroutine");
 979         }
 980       }
 981     } else if (index == scope_data()->jsr_return_address_local()) {
 982       scope_data()->set_jsr_return_address_local(-1);
 983     }
 984   }
 985 
 986   x->set_local_index(index);
 987   state->store_local(index, round_fp(x));
 988 }

 991 void GraphBuilder::load_indexed(BasicType type) {
 992   // In case of in block code motion in range check elimination
 993   ValueStack* state_before = NULL;
 994   int array_idx = state()->stack_size() - 2;
 995   if (type == T_OBJECT && state()->stack_at(array_idx)->maybe_flattened_array()) {
 996     // Save the entire state and re-execute on deopt when accessing flattened arrays
 997     state_before = copy_state_before();
 998     state_before->set_should_reexecute(true);
 999   } else {
1000     state_before = copy_state_indexed_access();
1001   }
1002   compilation()->set_has_access_indexed(true);
1003   Value index = ipop();
1004   Value array = apop();
1005   Value length = NULL;
1006   if (CSEArrayLength ||
1007       (array->as_AccessField() && array->as_AccessField()->field()->is_constant()) ||
1008       (array->as_NewArray() && array->as_NewArray()->length() && array->as_NewArray()->length()->type()->is_constant())) {
1009     length = append(new ArrayLength(array, state_before));
1010   }
1011 
1012   LoadIndexed* load_indexed = NULL;
1013   Instruction* result = NULL;
1014   if (array->is_loaded_flattened_array()) {
1015     ciType* array_type = array->declared_type();
1016     ciValueKlass* elem_klass = array_type->as_value_array_klass()->element_klass()->as_value_klass();
1017     NewValueTypeInstance* new_instance = new NewValueTypeInstance(elem_klass, state_before, false);
1018     _memory->new_instance(new_instance);
1019     apush(append_split(new_instance));
1020     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1021     load_indexed->set_vt(new_instance);
1022   } else {
1023     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1024   }
1025   if (profile_array_accesses() && is_reference_type(type)) {
1026     compilation()->set_would_profile(true);
1027     load_indexed->set_should_profile(true);
1028     load_indexed->set_profiled_method(method());
1029     load_indexed->set_profiled_bci(bci());
1030   }
1031   result = append(load_indexed);
1032   assert(!(profile_array_accesses() && is_reference_type(type)) || load_indexed == result, "should not be optimized out");
1033   if (!array->is_loaded_flattened_array()) {
1034     push(as_ValueType(type), result);
1035   }
1036 }

1089   switch (code) {
1090     case Bytecodes::_pop:
1091       { state()->raw_pop();
1092       }
1093       break;
1094     case Bytecodes::_pop2:
1095       { state()->raw_pop();
1096         state()->raw_pop();
1097       }
1098       break;
1099     case Bytecodes::_dup:
1100       { Value w = state()->raw_pop();
1101         state()->raw_push(w);
1102         state()->raw_push(w);

1104       break;

1118         state()->raw_push(w3);

1120         state()->raw_push(w1);
1121       }
1122       break;

1126         state()->raw_push(w2);

1190 void GraphBuilder::shift_op(ValueType* type, Bytecodes::Code code) {
1191   Value s = ipop();
1192   Value x = pop(type);
1193   // try to simplify
1194   // Note: This code should go into the canonicalizer as soon as it can
1195   //       can handle canonicalized forms that contain more than one node.
1196   if (CanonicalizeNodes && code == Bytecodes::_iushr) {
1197     // pattern: x >>> s
1198     IntConstant* s1 = s->type()->as_IntConstant();
1199     if (s1 != NULL) {
1200       // pattern: x >>> s1, with s1 constant
1201       ShiftOp* l = x->as_ShiftOp();
1202       if (l != NULL && l->op() == Bytecodes::_ishl) {
1203         // pattern: (a << b) >>> s1
1204         IntConstant* s0 = l->y()->type()->as_IntConstant();
1205         if (s0 != NULL) {
1206           // pattern: (a << s0) >>> s1
1207           const int s0c = s0->value() & 0x1F; // only the low 5 bits are significant for shifts
1208           const int s1c = s1->value() & 0x1F; // only the low 5 bits are significant for shifts
1209           if (s0c == s1c) {
1210             if (s0c == 0) {
1211               // pattern: (a << 0) >>> 0 => simplify to: a
1212               ipush(l->x());
1213             } else {
1214               // pattern: (a << s0c) >>> s0c => simplify to: a & m, with m constant
1215               assert(0 < s0c && s0c < BitsPerInt, "adjust code below to handle corner cases");
1216               const int m = (1 << (BitsPerInt - s0c)) - 1;
1217               Value s = append(new Constant(new IntConstant(m)));
1218               ipush(append(new LogicOp(Bytecodes::_iand, l->x(), s)));
1219             }
1220             return;
1221           }
1222         }
1223       }
1224     }
1225   }
1226   // could not simplify
1227   push(type, append(new ShiftOp(code, x, s)));
1228 }

1275 void GraphBuilder::if_node(Value x, If::Condition cond, Value y, ValueStack* state_before) {
1276   BlockBegin* tsux = block_at(stream()->get_dest());
1277   BlockBegin* fsux = block_at(stream()->next_bci());
1278   bool is_bb = tsux->bci() < stream()->cur_bci() || fsux->bci() < stream()->cur_bci();
1279 
1280   bool subst_check = false;
1281   if (EnableValhalla && (stream()->cur_bc() == Bytecodes::_if_acmpeq || stream()->cur_bc() == Bytecodes::_if_acmpne) &&
1282       method() != ciEnv::current()->ValueBootstrapMethods_klass()->find_method(ciSymbol::isSubstitutable_name(), ciSymbol::object_object_boolean_signature())) {
1283     // If current method is ValueBootstrapMethods::isSubstitutable(),
1284     // compile the acmp as a regular pointer comparison otherwise we
1285     // could call ValueBootstrapMethods::isSubstitutable() back
1286     ValueType* left_vt = x->type();
1287     ValueType* right_vt = y->type();
1288     if (left_vt->is_object()) {
1289       assert(right_vt->is_object(), "must be");
1290       ciKlass* left_klass = x->as_loaded_klass_or_null();
1291       ciKlass* right_klass = y->as_loaded_klass_or_null();
1292 
1293       if (left_klass == NULL || right_klass == NULL) {
1294         // The klass is still unloaded, or came from a Phi node. Go slow case;
1295         subst_check = true;
1296       } else if (left_klass->can_be_value_klass() || right_klass->can_be_value_klass()) {
1297         // Either operand may be a value object, but we're not sure. Go slow case;
1298         subst_check = true;
1299       } else {
1300         // No need to do substitutability check
1301       }
1302     }
1303   }
1304 
1305   // In case of loop invariant code motion or predicate insertion
1306   // before the body of a loop the state is needed
1307   Instruction *i = append(new If(x, cond, false, y, tsux, fsux, (is_bb || compilation()->is_optimistic() || subst_check) ? state_before : NULL, is_bb, subst_check));
1308 
1309   assert(i->as_Goto() == NULL ||
1310          (i->as_Goto()->sux_at(0) == tsux  && i->as_Goto()->is_safepoint() == tsux->bci() < stream()->cur_bci()) ||
1311          (i->as_Goto()->sux_at(0) == fsux  && i->as_Goto()->is_safepoint() == fsux->bci() < stream()->cur_bci()),
1312          "safepoint state of Goto returned by canonicalizer incorrect");
1313 
1314   if (is_profiling()) {
1315     If* if_node = i->as_If();
1316     if (if_node != NULL) {
1317       // Note that we'd collect profile data in this method if we wanted it.
1318       compilation()->set_would_profile(true);
1319       // At level 2 we need the proper bci to count backedges
1320       if_node->set_profiled_bci(bci());
1321       if (profile_branches()) {
1322         // Successors can be rotated by the canonicalizer, check for this case.
1323         if_node->set_profiled_method(method());
1324         if_node->set_should_profile(true);
1325         if (if_node->tsux() == fsux) {
1326           if_node->set_swapped(true);
1327         }
1328       }
1329       return;
1330     }
1331 
1332     // Check if this If was reduced to Goto.
1333     Goto *goto_node = i->as_Goto();
1334     if (goto_node != NULL) {
1335       compilation()->set_would_profile(true);
1336       goto_node->set_profiled_bci(bci());
1337       if (profile_branches()) {
1338         goto_node->set_profiled_method(method());
1339         goto_node->set_should_profile(true);
1340         // Find out which successor is used.
1341         if (goto_node->default_sux() == tsux) {
1342           goto_node->set_direction(Goto::taken);
1343         } else if (goto_node->default_sux() == fsux) {
1344           goto_node->set_direction(Goto::not_taken);
1345         } else {
1346           ShouldNotReachHere();
1347         }
1348       }
1349       return;
1350     }
1351   }
1352 }

1411 void GraphBuilder::table_switch() {
1412   Bytecode_tableswitch sw(stream());
1413   const int l = sw.length();
1414   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1415     // total of 2 successors => use If instead of switch
1416     // Note: This code should go into the canonicalizer as soon as it can
1417     //       can handle canonicalized forms that contain more than one node.
1418     Value key = append(new Constant(new IntConstant(sw.low_key())));
1419     BlockBegin* tsux = block_at(bci() + sw.dest_offset_at(0));
1420     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1421     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1422     // In case of loop invariant code motion or predicate insertion
1423     // before the body of a loop the state is needed
1424     ValueStack* state_before = copy_state_if_bb(is_bb);
1425     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1426   } else {
1427     // collect successors
1428     BlockList* sux = new BlockList(l + 1, NULL);
1429     int i;
1430     bool has_bb = false;
1431     for (i = 0; i < l; i++) {
1432       sux->at_put(i, block_at(bci() + sw.dest_offset_at(i)));
1433       if (sw.dest_offset_at(i) < 0) has_bb = true;
1434     }
1435     // add default successor
1436     if (sw.default_offset() < 0) has_bb = true;
1437     sux->at_put(i, block_at(bci() + sw.default_offset()));
1438     // In case of loop invariant code motion or predicate insertion
1439     // before the body of a loop the state is needed
1440     ValueStack* state_before = copy_state_if_bb(has_bb);
1441     Instruction* res = append(new TableSwitch(ipop(), sux, sw.low_key(), state_before, has_bb));
1442 #ifdef ASSERT
1443     if (res->as_Goto()) {
1444       for (i = 0; i < l; i++) {
1445         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1446           assert(res->as_Goto()->is_safepoint() == sw.dest_offset_at(i) < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1447         }
1448       }
1449     }
1450 #endif
1451   }
1452 }

1455 void GraphBuilder::lookup_switch() {
1456   Bytecode_lookupswitch sw(stream());
1457   const int l = sw.number_of_pairs();
1458   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1459     // total of 2 successors => use If instead of switch
1460     // Note: This code should go into the canonicalizer as soon as it can
1461     //       can handle canonicalized forms that contain more than one node.
1462     // simplify to If
1463     LookupswitchPair pair = sw.pair_at(0);
1464     Value key = append(new Constant(new IntConstant(pair.match())));
1465     BlockBegin* tsux = block_at(bci() + pair.offset());
1466     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1467     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1468     // In case of loop invariant code motion or predicate insertion
1469     // before the body of a loop the state is needed
1470     ValueStack* state_before = copy_state_if_bb(is_bb);;
1471     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1472   } else {
1473     // collect successors & keys
1474     BlockList* sux = new BlockList(l + 1, NULL);
1475     intArray* keys = new intArray(l, l, 0);
1476     int i;
1477     bool has_bb = false;
1478     for (i = 0; i < l; i++) {
1479       LookupswitchPair pair = sw.pair_at(i);
1480       if (pair.offset() < 0) has_bb = true;
1481       sux->at_put(i, block_at(bci() + pair.offset()));
1482       keys->at_put(i, pair.match());
1483     }
1484     // add default successor
1485     if (sw.default_offset() < 0) has_bb = true;
1486     sux->at_put(i, block_at(bci() + sw.default_offset()));
1487     // In case of loop invariant code motion or predicate insertion
1488     // before the body of a loop the state is needed
1489     ValueStack* state_before = copy_state_if_bb(has_bb);
1490     Instruction* res = append(new LookupSwitch(ipop(), sux, keys, state_before, has_bb));
1491 #ifdef ASSERT
1492     if (res->as_Goto()) {
1493       for (i = 0; i < l; i++) {
1494         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1495           assert(res->as_Goto()->is_safepoint() == sw.pair_at(i).offset() < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1496         }
1497       }
1498     }
1499 #endif
1500   }
1501 }

1503 void GraphBuilder::call_register_finalizer() {
1504   // If the receiver requires finalization then emit code to perform
1505   // the registration on return.
1506 
1507   // Gather some type information about the receiver
1508   Value receiver = state()->local_at(0);
1509   assert(receiver != NULL, "must have a receiver");
1510   ciType* declared_type = receiver->declared_type();
1511   ciType* exact_type = receiver->exact_type();
1512   if (exact_type == NULL &&
1513       receiver->as_Local() &&
1514       receiver->as_Local()->java_index() == 0) {
1515     ciInstanceKlass* ik = compilation()->method()->holder();
1516     if (ik->is_final()) {
1517       exact_type = ik;
1518     } else if (UseCHA && !(ik->has_subklass() || ik->is_interface())) {
1519       // test class is leaf class
1520       compilation()->dependency_recorder()->assert_leaf_type(ik);
1521       exact_type = ik;
1522     } else {
1523       declared_type = ik;
1524     }
1525   }
1526 
1527   // see if we know statically that registration isn't required
1528   bool needs_check = true;
1529   if (exact_type != NULL) {
1530     needs_check = exact_type->as_instance_klass()->has_finalizer();
1531   } else if (declared_type != NULL) {
1532     ciInstanceKlass* ik = declared_type->as_instance_klass();
1533     if (!Dependencies::has_finalizable_subclass(ik)) {
1534       compilation()->dependency_recorder()->assert_has_no_finalizable_subclasses(ik);
1535       needs_check = false;
1536     }
1537   }
1538 
1539   if (needs_check) {
1540     // Perform the registration of finalizable objects.
1541     ValueStack* state_before = copy_state_for_exception();
1542     load_local(objectType, 0);
1543     append_split(new Intrinsic(voidType, vmIntrinsics::_Object_init,
1544                                state()->pop_arguments(1),
1545                                true, state_before, true));
1546   }
1547 }

1550 void GraphBuilder::method_return(Value x, bool ignore_return) {
1551   if (RegisterFinalizersAtInit &&
1552       method()->intrinsic_id() == vmIntrinsics::_Object_init) {
1553     call_register_finalizer();
1554   }
1555 
1556   // The conditions for a memory barrier are described in Parse::do_exits().
1557   bool need_mem_bar = false;
1558   if (method()->is_object_constructor() &&
1559        (scope()->wrote_final() ||
1560          (AlwaysSafeConstructors && scope()->wrote_fields()) ||
1561          (support_IRIW_for_not_multiple_copy_atomic_cpu && scope()->wrote_volatile()))) {
1562     need_mem_bar = true;
1563   }
1564 
1565   BasicType bt = method()->return_type()->basic_type();
1566   switch (bt) {
1567     case T_BYTE:
1568     {
1569       Value shift = append(new Constant(new IntConstant(24)));
1570       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1571       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1572       break;
1573     }
1574     case T_SHORT:
1575     {
1576       Value shift = append(new Constant(new IntConstant(16)));
1577       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1578       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1579       break;
1580     }
1581     case T_CHAR:
1582     {
1583       Value mask = append(new Constant(new IntConstant(0xFFFF)));
1584       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1585       break;
1586     }
1587     case T_BOOLEAN:
1588     {
1589       Value mask = append(new Constant(new IntConstant(1)));
1590       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1591       break;
1592     }
1593     default:
1594       break;
1595   }
1596 
1597   // Check to see whether we are inlining. If so, Return
1598   // instructions become Gotos to the continuation point.
1599   if (continuation() != NULL) {
1600 
1601     int invoke_bci = state()->caller_state()->bci();
1602 
1603     if (x != NULL  && !ignore_return) {
1604       ciMethod* caller = state()->scope()->caller()->method();
1605       Bytecodes::Code invoke_raw_bc = caller->raw_code_at_bci(invoke_bci);
1606       if (invoke_raw_bc == Bytecodes::_invokehandle || invoke_raw_bc == Bytecodes::_invokedynamic) {
1607         ciType* declared_ret_type = caller->get_declared_signature_at_bci(invoke_bci)->return_type();
1608         if (declared_ret_type->is_klass() && x->exact_type() == NULL &&
1609             x->declared_type() != declared_ret_type && declared_ret_type != compilation()->env()->Object_klass()) {
1610           x = append(new TypeCast(declared_ret_type->as_klass(), x, copy_state_before()));
1611         }
1612       }
1613     }
1614 
1615     assert(!method()->is_synchronized() || InlineSynchronizedMethods, "can not inline synchronized methods yet");
1616 
1617     if (compilation()->env()->dtrace_method_probes()) {
1618       // Report exit from inline methods
1619       Values* args = new Values(1);
1620       args->push(append(new Constant(new MethodConstant(method()))));
1621       append(new RuntimeCall(voidType, "dtrace_method_exit", CAST_FROM_FN_PTR(address, SharedRuntime::dtrace_method_exit), args));
1622     }
1623 
1624     // If the inlined method is synchronized, the monitor must be
1625     // released before we jump to the continuation block.
1626     if (method()->is_synchronized()) {
1627       assert(state()->locks_size() == 1, "receiver must be locked here");
1628       monitorexit(state()->lock_at(0), SynchronizationEntryBCI);
1629     }
1630 
1631     if (need_mem_bar) {
1632       append(new MemBar(lir_membar_storestore));
1633     }
1634 
1635     // State at end of inlined method is the state of the caller
1636     // without the method parameters on stack, including the
1637     // return value, if any, of the inlined method on operand stack.
1638     set_state(state()->caller_state()->copy_for_parsing());
1639     if (x != NULL) {
1640       if (!ignore_return) {
1641         state()->push(x->type(), x);
1642       }
1643       if (profile_return() && x->type()->is_object_kind()) {
1644         ciMethod* caller = state()->scope()->method();
1645         profile_return_type(x, method(), caller, invoke_bci);
1646       }
1647     }
1648     Goto* goto_callee = new Goto(continuation(), false);
1649 
1650     // See whether this is the first return; if so, store off some
1651     // of the state for later examination
1652     if (num_returns() == 0) {
1653       set_inline_cleanup_info();
1654     }
1655 
1656     // The current bci() is in the wrong scope, so use the bci() of
1657     // the continuation point.
1658     append_with_bci(goto_callee, scope_data()->continuation()->bci());
1659     incr_num_returns();
1660     return;
1661   }
1662 
1663   state()->truncate_stack(0);
1664   if (method()->is_synchronized()) {
1665     // perform the unlocking before exiting the method
1666     Value receiver;
1667     if (!method()->is_static()) {
1668       receiver = _initial_state->local_at(0);
1669     } else {
1670       receiver = append(new Constant(new ClassConstant(method()->holder())));
1671     }
1672     append_split(new MonitorExit(receiver, state()->unlock()));
1673   }
1674 
1675   if (need_mem_bar) {
1676       append(new MemBar(lir_membar_storestore));
1677   }
1678 
1679   assert(!ignore_return, "Ignoring return value works only for inlining");
1680   append(new Return(x));
1681 }

1723 void GraphBuilder::access_field(Bytecodes::Code code) {
1724   bool will_link;
1725   ciField* field = stream()->get_field(will_link);
1726   ciInstanceKlass* holder = field->holder();
1727   BasicType field_type = field->type()->basic_type();
1728   ValueType* type = as_ValueType(field_type);
1729 
1730   // call will_link again to determine if the field is valid.
1731   const bool needs_patching = !holder->is_loaded() ||
1732                               !field->will_link(method(), code) ||
1733                               PatchALot;
1734 
1735   ValueStack* state_before = NULL;
1736   if (!holder->is_initialized() || needs_patching) {
1737     // save state before instruction for debug info when
1738     // deoptimization happens during patching
1739     state_before = copy_state_before();
1740   }
1741 
1742   Value obj = NULL;
1743   if (code == Bytecodes::_getstatic || code == Bytecodes::_putstatic) {
1744     if (state_before != NULL) {
1745       // build a patching constant
1746       obj = new Constant(new InstanceConstant(holder->java_mirror()), state_before);
1747     } else {
1748       obj = new Constant(new InstanceConstant(holder->java_mirror()));
1749     }
1750   }
1751 
1752   if (field->is_final() && (code == Bytecodes::_putfield || code == Bytecodes::_withfield)) {
1753     scope()->set_wrote_final();
1754   }
1755 
1756   if (code == Bytecodes::_putfield || code == Bytecodes::_withfield) {
1757     scope()->set_wrote_fields();
1758     if (field->is_volatile()) {
1759       scope()->set_wrote_volatile();
1760     }
1761   }
1762 
1763   const int offset = !needs_patching ? field->offset() : -1;
1764   switch (code) {
1765     case Bytecodes::_getstatic: {
1766       // check for compile-time constants, i.e., initialized static final fields
1767       Value constant = NULL;
1768       if (field->is_static_constant() && !PatchALot) {
1769         ciConstant field_value = field->constant_value();
1770         assert(!field->is_stable() || !field_value.is_null_or_zero(),
1771                "stable static w/ default value shouldn't be a constant");
1772         constant = make_constant(field_value, field);
1773       }
1774       if (constant != NULL) {
1775         push(type, append(constant));
1776       } else {
1777         if (state_before == NULL) {
1778           state_before = copy_state_for_exception();
1779         }
1780         LoadField* load_field = new LoadField(append(obj), offset, field, true,
1781                                         state_before, needs_patching);
1782         push(type, append(load_field));
1783       }
1784       break;
1785     }
1786     case Bytecodes::_putstatic: {
1787       Value val = pop(type);
1788       val->set_escaped();
1789       if (state_before == NULL) {
1790         state_before = copy_state_for_exception();
1791       }
1792       if (field_type == T_BOOLEAN) {
1793         Value mask = append(new Constant(new IntConstant(1)));
1794         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1795       }
1796       append(new StoreField(append(obj), offset, field, val, true, state_before, needs_patching));
1797       break;
1798     }
1799     case Bytecodes::_getfield: {
1800       // Check for compile-time constants, i.e., trusted final non-static fields.
1801       Value constant = NULL;
1802       if (state_before == NULL && field->is_flattened()) {
1803         // Save the entire state and re-execute on deopt when accessing flattened fields
1804         assert(Interpreter::bytecode_should_reexecute(code), "should reexecute");
1805         state_before = copy_state_before();
1806       }
1807       obj = apop();
1808       ObjectType* obj_type = obj->type()->as_ObjectType();
1809       if (field->is_constant() && !field->is_flattened() && obj_type->is_constant() && !PatchALot) {
1810         ciObject* const_oop = obj_type->constant_value();
1811         if (!const_oop->is_null_object() && const_oop->is_loaded()) {
1812           ciConstant field_value = field->constant_value_of(const_oop);
1813           if (field_value.is_valid()) {
1814             if (field_type == T_INLINE_TYPE && field_value.is_null_or_zero()) {
1815               // Non-flattened inline type field. Replace null by the default value.
1816               constant = new Constant(new InstanceConstant(field->type()->as_value_klass()->default_value_instance()));
1817             } else {
1818               constant = make_constant(field_value, field);
1819             }
1820             // For CallSite objects add a dependency for invalidation of the optimization.
1821             if (field->is_call_site_target()) {
1822               ciCallSite* call_site = const_oop->as_call_site();
1823               if (!call_site->is_fully_initialized_constant_call_site()) {
1824                 ciMethodHandle* target = field_value.as_object()->as_method_handle();
1825                 dependency_recorder()->assert_call_site_target_value(call_site, target);
1826               }
1827             }
1828           }
1829         }
1830       }
1831       if (constant != NULL) {
1832         push(type, append(constant));
1833       } else {
1834         if (state_before == NULL) {
1835           state_before = copy_state_for_exception();
1836         }
1837         if (!field->is_flattened()) {
1838           LoadField* load = new LoadField(obj, offset, field, false, state_before, needs_patching);
1839           Value replacement = !needs_patching ? _memory->load(load) : load;
1840           if (replacement != load) {
1841             assert(replacement->is_linked() || !replacement->can_be_linked(), "should already by linked");
1842             // Writing an (integer) value to a boolean, byte, char or short field includes an implicit narrowing
1843             // conversion. Emit an explicit conversion here to get the correct field value after the write.
1844             BasicType bt = field->type()->basic_type();
1845             switch (bt) {
1846             case T_BOOLEAN:
1847             case T_BYTE:
1848               replacement = append(new Convert(Bytecodes::_i2b, replacement, as_ValueType(bt)));
1849               break;
1850             case T_CHAR:
1851               replacement = append(new Convert(Bytecodes::_i2c, replacement, as_ValueType(bt)));
1852               break;
1853             case T_SHORT:
1854               replacement = append(new Convert(Bytecodes::_i2s, replacement, as_ValueType(bt)));
1855               break;
1856             default:
1857               break;
1858             }
1859             push(type, replacement);
1860           } else {
1861             push(type, append(load));
1862           }
1863         } else { // flattened field, not optimized solution: re-instantiate the flattened value
1864           assert(field->type()->is_valuetype(), "Sanity check");
1865           ciValueKlass* value_klass = field->type()->as_value_klass();
1866           int flattening_offset = field->offset() - value_klass->first_field_offset();
1867           assert(field->type()->is_valuetype(), "Sanity check");
1868           scope()->set_wrote_final();
1869           scope()->set_wrote_fields();
1870           NewValueTypeInstance* new_instance = new NewValueTypeInstance(value_klass, state_before, false);
1871           _memory->new_instance(new_instance);
1872           apush(append_split(new_instance));
1873           copy_value_content(value_klass, obj, field->offset(), new_instance, value_klass->first_field_offset(),
1874                        state_before, needs_patching);
1875         }
1876       }
1877       break;
1878     }
1879     case Bytecodes::_withfield:
1880     case Bytecodes::_putfield: {
1881       Value val = pop(type);
1882       val->set_escaped();
1883       obj = apop();
1884       if (state_before == NULL) {
1885         state_before = copy_state_for_exception();
1886       }
1887       if (field->type()->basic_type() == T_BOOLEAN) {
1888         Value mask = append(new Constant(new IntConstant(1)));
1889         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1890       }
1891 
1892       if (!field->is_flattened()) {
1893         StoreField* store = new StoreField(obj, offset, field, val, false, state_before, needs_patching);
1894         if (!needs_patching) store = _memory->store(store);
1895         if (store != NULL) {
1896           append(store);
1897         }
1898       } else {
1899         assert(field->type()->is_valuetype(), "Sanity check");
1900         ciValueKlass* value_klass = field->type()->as_value_klass();
1901         int flattening_offset = field->offset() - value_klass->first_field_offset();
1902         copy_value_content(value_klass, val, value_klass->first_field_offset(), obj, field->offset(),
1903                    state_before, needs_patching);
1904       }
1905       break;
1906     }
1907     default:
1908       ShouldNotReachHere();
1909       break;
1910   }
1911 }

1913 // Baseline version of withfield, allocate every time
1914 void GraphBuilder::withfield(int field_index)
1915 {
1916   bool will_link;
1917   ciField* field_modify = stream()->get_field(will_link);
1918   ciInstanceKlass* holder = field_modify->holder();
1919   BasicType field_type = field_modify->type()->basic_type();
1920   ValueType* type = as_ValueType(field_type);
1921 
1922   // call will_link again to determine if the field is valid.
1923   const bool needs_patching = !holder->is_loaded() ||
1924                               !field_modify->will_link(method(), Bytecodes::_withfield) ||
1925                               PatchALot;
1926 
1927 
1928   scope()->set_wrote_final();
1929   scope()->set_wrote_fields();
1930 
1931   const int offset = !needs_patching ? field_modify->offset() : -1;
1932 
1933   if (!holder->is_loaded()
1934       || needs_patching /* FIXME: 8228634 - field_modify->will_link() may incorrectly return false */
1935       ) {
1936     ValueStack* state_before = copy_state_before();
1937     Value val = pop(type);
1938     Value obj = apop();
1939     apush(append_split(new WithField(state_before)));
1940     return;
1941   }
1942   ValueStack* state_before = copy_state_before();
1943 
1944   Value val = pop(type);
1945   Value obj = apop();
1946 
1947   if (!needs_patching && obj->is_optimizable_for_withfield()) {
1948     int astore_index;
1949     ciBytecodeStream s(method());
1950     s.force_bci(bci());
1951     s.next();
1952     switch (s.cur_bc()) {
1953     case Bytecodes::_astore:    astore_index = s.get_index(); break;
1954     case Bytecodes::_astore_0:  astore_index = 0; break;
1955     case Bytecodes::_astore_1:  astore_index = 1; break;
1956     case Bytecodes::_astore_2:  astore_index = 2; break;
1957     case Bytecodes::_astore_3:  astore_index = 3; break;
1958     default: astore_index = -1;
1959     }
1960 
1961     if (astore_index >= 0 && obj == state()->local_at(astore_index)) {
1962       // We have a sequence like this, where we load a value object from a local slot,
1963       // and overwrite the same local slot with a modified copy of the value object.
1964       //      defaultvalue #1 // class compiler/valhalla/valuetypes/MyValue1
1965       //      astore 9
1966       //      ...
1967       //      iload_0
1968       //      aload 9
1969       //      swap
1970       //      withfield #7 // Field x:I
1971       //      astore 9
1972       // If this object was created by defaultvalue, and has not escaped, and is not stored
1973       // in any other local slots, we can effectively treat the withfield/astore
1974       // sequence as a single putfield bytecode.
1975       push(objectType, obj);
1976       push(type, val);
1977       access_field(Bytecodes::_withfield);
1978       stream()->next(); // skip the next astore/astore_n bytecode.
1979       return;
1980     }
1981   }
1982 
1983   assert(holder->is_valuetype(), "must be a value klass");
1984   // Save the entire state and re-execute on deopt when executing withfield
1985   state_before->set_should_reexecute(true);
1986   NewValueTypeInstance* new_instance = new NewValueTypeInstance(holder->as_value_klass(), state_before, false);
1987   _memory->new_instance(new_instance);
1988   apush(append_split(new_instance));
1989 
1990   for (int i = 0; i < holder->nof_nonstatic_fields(); i++) {
1991     ciField* field = holder->nonstatic_field_at(i);
1992     int off = field->offset();
1993 
1994     if (field->offset() != offset) {
1995       if (field->is_flattened()) {
1996         assert(field->type()->is_valuetype(), "Sanity check");
1997         assert(field->type()->is_valuetype(), "Only value types can be flattened");
1998         ciValueKlass* vk = field->type()->as_value_klass();
1999         copy_value_content(vk, obj, off, new_instance, vk->first_field_offset(), state_before, needs_patching);
2000       } else {
2001         // Only load those fields who are not modified
2002         LoadField* load = new LoadField(obj, off, field, false, state_before, needs_patching);
2003         Value replacement = append(load);
2004         StoreField* store = new StoreField(new_instance, off, field, replacement, false, state_before, needs_patching);
2005         append(store);
2006       }
2007     }
2008   }
2009 
2010   // Field to modify
2011   if (field_modify->type()->basic_type() == T_BOOLEAN) {
2012     Value mask = append(new Constant(new IntConstant(1)));
2013     val = append(new LogicOp(Bytecodes::_iand, val, mask));
2014   }
2015   if (field_modify->is_flattened()) {
2016     assert(field_modify->type()->is_valuetype(), "Only value types can be flattened");
2017     ciValueKlass* vk = field_modify->type()->as_value_klass();
2018     copy_value_content(vk, val, vk->first_field_offset(), new_instance, field_modify->offset(), state_before, needs_patching);
2019   } else {
2020     StoreField* store = new StoreField(new_instance, offset, field_modify, val, false, state_before, needs_patching);
2021     append(store);
2022   }
2023 }

2032   int n = 0;
2033   bool has_receiver = may_have_receiver && Bytecodes::has_receiver(method()->java_code_at_bci(bci()));
2034   start = has_receiver ? 1 : 0;
2035   if (profile_arguments()) {
2036     ciProfileData* data = method()->method_data()->bci_to_data(bci());
2037     if (data != NULL && (data->is_CallTypeData() || data->is_VirtualCallTypeData())) {
2038       n = data->is_CallTypeData() ? data->as_CallTypeData()->number_of_arguments() : data->as_VirtualCallTypeData()->number_of_arguments();
2039     }
2040   }
2041   // If we are inlining then we need to collect arguments to profile parameters for the target
2042   if (profile_parameters() && target != NULL) {
2043     if (target->method_data() != NULL && target->method_data()->parameters_type_data() != NULL) {
2044       // The receiver is profiled on method entry so it's included in
2045       // the number of parameters but here we're only interested in
2046       // actual arguments.
2047       n = MAX2(n, target->method_data()->parameters_type_data()->number_of_parameters() - start);
2048     }
2049   }
2050   if (n > 0) {
2051     return new Values(n);
2052   }
2053   return NULL;
2054 }

2270   // check if we could do inlining
2271   if (!PatchALot && Inline && target->is_loaded() &&
2272       (klass->is_initialized() || (klass->is_interface() && target->holder()->is_initialized()))
2273       && !patch_for_appendix) {
2274     // callee is known => check if we have static binding
2275     if (code == Bytecodes::_invokestatic  ||
2276         code == Bytecodes::_invokespecial ||
2277         (code == Bytecodes::_invokevirtual && target->is_final_method()) ||
2278         code == Bytecodes::_invokedynamic) {
2279       ciMethod* inline_target = (cha_monomorphic_target != NULL) ? cha_monomorphic_target : target;
2280       // static binding => check if callee is ok
2281       bool success = try_inline(inline_target, (cha_monomorphic_target != NULL) || (exact_target != NULL), false, code, better_receiver);
2282 
2283       CHECK_BAILOUT();
2284       clear_inline_bailout();
2285 
2286       if (success) {
2287         // Register dependence if JVMTI has either breakpoint
2288         // setting or hotswapping of methods capabilities since they may
2289         // cause deoptimization.
2290         if (compilation()->env()->jvmti_can_hotswap_or_post_breakpoint()) {
2291           dependency_recorder()->assert_evol_method(inline_target);
2292         }
2293         return;
2294       }
2295     } else {
2296       print_inlining(target, "no static binding", /*success*/ false);
2297     }
2298   } else {
2299     print_inlining(target, "not inlineable", /*success*/ false);
2300   }
2301 
2302   // If we attempted an inline which did not succeed because of a
2303   // bailout during construction of the callee graph, the entire
2304   // compilation has to be aborted. This is fairly rare and currently
2305   // seems to only occur for jasm-generated classes which contain
2306   // jsr/ret pairs which are not associated with finally clauses and
2307   // do not have exception handlers in the containing method, and are
2308   // therefore not caught early enough to abort the inlining without
2309   // corrupting the graph. (We currently bail out with a non-empty
2310   // stack at a ret in these situations.)
2311   CHECK_BAILOUT();
2312 
2313   // inlining not successful => standard invoke
2314   ValueType* result_type = as_ValueType(declared_signature->return_type());
2315   ValueStack* state_before = copy_state_exhandling();
2316 
2317   // The bytecode (code) might change in this method so we are checking this very late.
2318   const bool has_receiver =
2319     code == Bytecodes::_invokespecial   ||
2320     code == Bytecodes::_invokevirtual   ||
2321     code == Bytecodes::_invokeinterface;
2322   Values* args = state()->pop_arguments(target->arg_size_no_receiver() + patching_appendix_arg);
2323   Value recv = has_receiver ? apop() : NULL;
2324   int vtable_index = Method::invalid_vtable_index;
2325 
2326   // A null check is required here (when there is a receiver) for any of the following cases
2327   // - invokespecial, always need a null check.
2328   // - invokevirtual, when the target is final and loaded. Calls to final targets will become optimized
2329   //   and require null checking. If the target is loaded a null check is emitted here.
2330   //   If the target isn't loaded the null check must happen after the call resolution. We achieve that
2331   //   by using the target methods unverified entry point (see CompiledIC::compute_monomorphic_entry).
2332   //   (The JVM specification requires that LinkageError must be thrown before a NPE. An unloaded target may
2333   //   potentially fail, and can't have the null check before the resolution.)
2334   // - A call that will be profiled. (But we can't add a null check when the target is unloaded, by the same
2335   //   reason as above, so calls with a receiver to unloaded targets can't be profiled.)
2336   //
2337   // Normal invokevirtual will perform the null check during lookup
2338 
2339   bool need_null_check = (code == Bytecodes::_invokespecial) ||
2340       (target->is_loaded() && (target->is_final_method() || (is_profiling() && profile_calls())));
2341 
2342   if (need_null_check) {
2343     if (recv != NULL) {
2344       null_check(recv);
2345     }
2346 
2347     if (is_profiling()) {
2348       // Note that we'd collect profile data in this method if we wanted it.
2349       compilation()->set_would_profile(true);
2350 
2351       if (profile_calls()) {
2352         assert(cha_monomorphic_target == NULL || exact_target == NULL, "both can not be set");
2353         ciKlass* target_klass = NULL;
2354         if (cha_monomorphic_target != NULL) {
2355           target_klass = cha_monomorphic_target->holder();
2356         } else if (exact_target != NULL) {
2357           target_klass = exact_target->holder();
2358         }
2359         profile_call(target, recv, target_klass, collect_args_for_profiling(args, NULL, false), false);
2360       }
2361     }
2362   }
2363 
2364   if (recv != NULL) {
2365     recv->set_escaped();
2366   }
2367   for (int i=0; i<args->length(); i++) {
2368     args->at(0)->set_escaped();
2369   }
2370 
2371   Invoke* result = new Invoke(code, result_type, recv, args, vtable_index, target, state_before,
2372                               declared_signature->return_type()->is_valuetype());
2373   // push result
2374   append_split(result);
2375 
2376   if (result_type != voidType) {
2377     if (method()->is_strict()) {
2378       push(result_type, round_fp(result));
2379     } else {
2380       push(result_type, result);
2381     }
2382   }
2383   if (profile_return() && result_type->is_object_kind()) {
2384     profile_return_type(result, target);
2385   }
2386 }

2569 Instruction* GraphBuilder::append_with_bci(Instruction* instr, int bci) {
2570   Canonicalizer canon(compilation(), instr, bci);
2571   Instruction* i1 = canon.canonical();
2572   if (i1->is_linked() || !i1->can_be_linked()) {
2573     // Canonicalizer returned an instruction which was already
2574     // appended so simply return it.
2575     return i1;
2576   }
2577 
2578   if (UseLocalValueNumbering) {
2579     // Lookup the instruction in the ValueMap and add it to the map if
2580     // it's not found.
2581     Instruction* i2 = vmap()->find_insert(i1);
2582     if (i2 != i1) {
2583       // found an entry in the value map, so just return it.
2584       assert(i2->is_linked(), "should already be linked");
2585       return i2;
2586     }
2587     ValueNumberingEffects vne(vmap());
2588     i1->visit(&vne);
2589   }
2590 
2591   // i1 was not eliminated => append it
2592   assert(i1->next() == NULL, "shouldn't already be linked");
2593   _last = _last->set_next(i1, canon.bci());
2594 
2595   if (++_instruction_count >= InstructionCountCutoff && !bailed_out()) {
2596     // set the bailout state but complete normal processing.  We
2597     // might do a little more work before noticing the bailout so we
2598     // want processing to continue normally until it's noticed.
2599     bailout("Method and/or inlining is too large");
2600   }
2601 
2602 #ifndef PRODUCT
2603   if (PrintIRDuringConstruction) {
2604     InstructionPrinter ip;
2605     ip.print_line(i1);
2606     if (Verbose) {
2607       state()->print();
2608     }
2609   }
2610 #endif
2611 
2612   // save state after modification of operand stack for StateSplit instructions
2613   StateSplit* s = i1->as_StateSplit();
2614   if (s != NULL) {
2615     if (EliminateFieldAccess) {
2616       Intrinsic* intrinsic = s->as_Intrinsic();
2617       if (s->as_Invoke() != NULL || (intrinsic && !intrinsic->preserves_state())) {
2618         _memory->kill();
2619       }
2620     }
2621     s->set_state(state()->copy(ValueStack::StateAfter, canon.bci()));
2622   }
2623 
2624   // set up exception handlers for this instruction if necessary
2625   if (i1->can_trap()) {
2626     i1->set_exception_handlers(handle_exception(i1));
2627     assert(i1->exception_state() != NULL || !i1->needs_exception_state() || bailed_out(), "handle_exception must set exception state");
2628   }
2629   return i1;
2630 }

2664 XHandlers* GraphBuilder::handle_exception(Instruction* instruction) {
2665   if (!has_handler() && (!instruction->needs_exception_state() || instruction->exception_state() != NULL)) {
2666     assert(instruction->exception_state() == NULL
2667            || instruction->exception_state()->kind() == ValueStack::EmptyExceptionState
2668            || (instruction->exception_state()->kind() == ValueStack::ExceptionState && _compilation->env()->should_retain_local_variables()),
2669            "exception_state should be of exception kind");
2670     return new XHandlers();
2671   }
2672 
2673   XHandlers*  exception_handlers = new XHandlers();
2674   ScopeData*  cur_scope_data = scope_data();
2675   ValueStack* cur_state = instruction->state_before();
2676   ValueStack* prev_state = NULL;
2677   int scope_count = 0;
2678 
2679   assert(cur_state != NULL, "state_before must be set");
2680   do {
2681     int cur_bci = cur_state->bci();
2682     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2683     assert(cur_bci == SynchronizationEntryBCI || cur_bci == cur_scope_data->stream()->cur_bci(), "invalid bci");
2684 
2685     // join with all potential exception handlers
2686     XHandlers* list = cur_scope_data->xhandlers();
2687     const int n = list->length();
2688     for (int i = 0; i < n; i++) {
2689       XHandler* h = list->handler_at(i);
2690       if (h->covers(cur_bci)) {
2691         // h is a potential exception handler => join it
2692         compilation()->set_has_exception_handlers(true);
2693 
2694         BlockBegin* entry = h->entry_block();
2695         if (entry == block()) {
2696           // It's acceptable for an exception handler to cover itself
2697           // but we don't handle that in the parser currently.  It's
2698           // very rare so we bailout instead of trying to handle it.
2699           BAILOUT_("exception handler covers itself", exception_handlers);
2700         }
2701         assert(entry->bci() == h->handler_bci(), "must match");
2702         assert(entry->bci() == -1 || entry == cur_scope_data->block_at(entry->bci()), "blocks must correspond");
2703 
2704         // previously this was a BAILOUT, but this is not necessary
2705         // now because asynchronous exceptions are not handled this way.
2706         assert(entry->state() == NULL || cur_state->total_locks_size() == entry->state()->total_locks_size(), "locks do not match");
2707 
2708         // xhandler start with an empty expression stack
2709         if (cur_state->stack_size() != 0) {
2710           cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2711         }
2712         if (instruction->exception_state() == NULL) {
2713           instruction->set_exception_state(cur_state);
2714         }
2715 
2716         // Note: Usually this join must work. However, very
2717         // complicated jsr-ret structures where we don't ret from
2718         // the subroutine can cause the objects on the monitor
2719         // stacks to not match because blocks can be parsed twice.
2720         // The only test case we've seen so far which exhibits this
2721         // problem is caught by the infinite recursion test in
2722         // GraphBuilder::jsr() if the join doesn't work.
2723         if (!entry->try_merge(cur_state)) {
2724           BAILOUT_("error while joining with exception handler, prob. due to complicated jsr/rets", exception_handlers);
2725         }
2726 
2727         // add current state for correct handling of phi functions at begin of xhandler
2728         int phi_operand = entry->add_exception_state(cur_state);
2729 
2730         // add entry to the list of xhandlers of this block
2731         _block->add_exception_handler(entry);
2732 
2733         // add back-edge from xhandler entry to this block
2734         if (!entry->is_predecessor(_block)) {
2735           entry->add_predecessor(_block);
2736         }
2737 
2738         // clone XHandler because phi_operand and scope_count can not be shared
2739         XHandler* new_xhandler = new XHandler(h);
2740         new_xhandler->set_phi_operand(phi_operand);
2741         new_xhandler->set_scope_count(scope_count);
2742         exception_handlers->append(new_xhandler);
2743 
2744         // fill in exception handler subgraph lazily
2745         assert(!entry->is_set(BlockBegin::was_visited_flag), "entry must not be visited yet");
2746         cur_scope_data->add_to_work_list(entry);
2747 
2748         // stop when reaching catchall
2749         if (h->catch_type() == 0) {
2750           return exception_handlers;
2751         }
2752       }
2753     }
2754 
2755     if (exception_handlers->length() == 0) {
2756       // This scope and all callees do not handle exceptions, so the local
2757       // variables of this scope are not needed. However, the scope itself is
2758       // required for a correct exception stack trace -> clear out the locals.
2759       if (_compilation->env()->should_retain_local_variables()) {
2760         cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2761       } else {
2762         cur_state = cur_state->copy(ValueStack::EmptyExceptionState, cur_state->bci());
2763       }
2764       if (prev_state != NULL) {
2765         prev_state->set_caller_state(cur_state);
2766       }
2767       if (instruction->exception_state() == NULL) {
2768         instruction->set_exception_state(cur_state);
2769       }
2770     }
2771 
2772     // Set up iteration for next time.
2773     // If parsing a jsr, do not grab exception handlers from the
2774     // parent scopes for this method (already got them, and they
2775     // needed to be cloned)
2776 
2777     while (cur_scope_data->parsing_jsr()) {
2778       cur_scope_data = cur_scope_data->parent();
2779     }
2780 
2781     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2782     assert(cur_state->locks_size() == 0 || cur_state->locks_size() == 1, "unlocking must be done in a catchall exception handler");
2783 
2784     prev_state = cur_state;
2785     cur_state = cur_state->caller_state();
2786     cur_scope_data = cur_scope_data->parent();
2787     scope_count++;
2788   } while (cur_scope_data != NULL);
2789 
2790   return exception_handlers;
2791 }

2811 
2812 Value PhiSimplifier::simplify(Value v) {
2813   Phi* phi = v->as_Phi();
2814 
2815   if (phi == NULL) {
2816     // no phi function
2817     return v;
2818   } else if (v->has_subst()) {
2819     // already substituted; subst can be phi itself -> simplify
2820     return simplify(v->subst());
2821   } else if (phi->is_set(Phi::cannot_simplify)) {
2822     // already tried to simplify phi before
2823     return phi;
2824   } else if (phi->is_set(Phi::visited)) {
2825     // break cycles in phi functions
2826     return phi;
2827   } else if (phi->type()->is_illegal()) {
2828     // illegal phi functions are ignored anyway
2829     return phi;
2830 
2831   } else {
2832     // mark phi function as processed to break cycles in phi functions
2833     phi->set(Phi::visited);
2834 
2835     // simplify x = [y, x] and x = [y, y] to y
2836     Value subst = NULL;
2837     int opd_count = phi->operand_count();
2838     for (int i = 0; i < opd_count; i++) {

2840       assert(opd != NULL, "Operand must exist!");

2844         phi->make_illegal();

2846         return phi;
2847       }
2848 
2849       Value new_opd = simplify(opd);
2850       assert(new_opd != NULL, "Simplified operand must exist!");
2851 
2852       if (new_opd != phi && new_opd != subst) {
2853         if (subst == NULL) {
2854           subst = new_opd;
2855         } else {
2856           // no simplification possible
2857           phi->set(Phi::cannot_simplify);
2858           phi->clear(Phi::visited);
2859           return phi;
2860         }
2861       }
2862     }
2863 
2864     // sucessfully simplified phi function
2865     assert(subst != NULL, "illegal phi function");

2867     phi->clear(Phi::visited);
2868     phi->set_subst(subst);
2869 
2870 #ifndef PRODUCT
2871     if (PrintPhiFunctions) {
2872       tty->print_cr("simplified phi function %c%d to %c%d (Block B%d)", phi->type()->tchar(), phi->id(), subst->type()->tchar(), subst->id(), phi->block()->block_id());
2873     }
2874 #endif
2875 
2876     return subst;
2877   }

3240           connect_to_end(b);
3241         }
3242       }
3243     }
3244   } while (!bailed_out() && !scope_data()->is_work_list_empty());

3371 void GraphBuilder::setup_osr_entry_block() {
3372   assert(compilation()->is_osr_compile(), "only for osrs");
3373 
3374   int osr_bci = compilation()->osr_bci();
3375   ciBytecodeStream s(method());
3376   s.reset_to_bci(osr_bci);
3377   s.next();
3378   scope_data()->set_stream(&s);
3379 
3380   // create a new block to be the osr setup code
3381   _osr_entry = new BlockBegin(osr_bci);
3382   _osr_entry->set(BlockBegin::osr_entry_flag);
3383   _osr_entry->set_depth_first_number(0);
3384   BlockBegin* target = bci2block()->at(osr_bci);
3385   assert(target != NULL && target->is_set(BlockBegin::osr_entry_flag), "must be there");
3386   // the osr entry has no values for locals
3387   ValueStack* state = target->state()->copy();
3388   _osr_entry->set_state(state);
3389 
3390   kill_all();
3391   _block = _osr_entry;
3392   _state = _osr_entry->state()->copy();
3393   assert(_state->bci() == osr_bci, "mismatch");
3394   _last  = _osr_entry;
3395   Value e = append(new OsrEntry());
3396   e->set_needs_null_check(false);
3397 
3398   // OSR buffer is
3399   //
3400   // locals[nlocals-1..0]
3401   // monitors[number_of_locks-1..0]
3402   //
3403   // locals is a direct copy of the interpreter frame so in the osr buffer
3404   // so first slot in the local array is the last local from the interpreter
3405   // and last slot is local[0] (receiver) from the interpreter
3406   //
3407   // Similarly with locks. The first lock slot in the osr buffer is the nth lock
3408   // from the interpreter frame, the nth lock slot in the osr buffer is 0th lock
3409   // in the interpreter frame (the method lock if a sync method)
3410 
3411   // Initialize monitors in the compiled activation.
3412 
3413   int index;
3414   Value local;
3415 
3416   // find all the locals that the interpreter thinks contain live oops
3417   const ResourceBitMap live_oops = method()->live_local_oops_at_bci(osr_bci);
3418 
3419   // compute the offset into the locals so that we can treat the buffer
3420   // as if the locals were still in the interpreter frame
3421   int locals_offset = BytesPerWord * (method()->max_locals() - 1);
3422   for_each_local_value(state, index, local) {
3423     int offset = locals_offset - (index + local->type()->size() - 1) * BytesPerWord;
3424     Value get;
3425     if (local->type()->is_object_kind() && !live_oops.at(index)) {
3426       // The interpreter thinks this local is dead but the compiler
3427       // doesn't so pretend that the interpreter passed in null.
3428       get = append(new Constant(objectNull));
3429     } else {
3430       get = append(new UnsafeGetRaw(as_BasicType(local->type()), e,
3431                                     append(new Constant(new IntConstant(offset))),
3432                                     0,
3433                                     true /*unaligned*/, true /*wide*/));
3434     }
3435     _state->store_local(index, get);
3436   }
3437 
3438   // the storage for the OSR buffer is freed manually in the LIRGenerator.
3439 
3440   assert(state->caller_state() == NULL, "should be top scope");
3441   state->clear_locals();
3442   Goto* g = new Goto(target, false);
3443   append(g);
3444   _osr_entry->set_end(g);
3445   target->merge(_osr_entry->end()->state());
3446 
3447   scope_data()->set_stream(NULL);
3448 }

3452   ValueStack* state = new ValueStack(scope(), NULL);
3453 
3454   // Set up locals for receiver
3455   int idx = 0;
3456   if (!method()->is_static()) {
3457     // we should always see the receiver
3458     state->store_local(idx, new Local(method()->holder(), objectType, idx,
3459              /*receiver*/ true, /*never_null*/ method()->holder()->is_value_array_klass()));

3461   }
3462 
3463   // Set up locals for incoming arguments
3464   ciSignature* sig = method()->signature();

3466     ciType* type = sig->type_at(i);
3467     BasicType basic_type = type->basic_type();

3469     if (is_reference_type(basic_type)) basic_type = T_OBJECT;
3470     ValueType* vt = as_ValueType(basic_type);
3471     state->store_local(idx, new Local(type, vt, idx, false, type->is_valuetype()));
3472     idx += type->size();
3473   }
3474 
3475   // lock synchronized method
3476   if (method()->is_synchronized()) {
3477     state->lock(NULL);
3478   }
3479

3495   BlockListBuilder blm(compilation, scope, osr_bci);

3500

3502

3511   case vmIntrinsics::_dsqrt         : // fall through
3512   case vmIntrinsics::_dsin          : // fall through
3513   case vmIntrinsics::_dcos          : // fall through
3514   case vmIntrinsics::_dtan          : // fall through
3515   case vmIntrinsics::_dlog          : // fall through
3516   case vmIntrinsics::_dlog10        : // fall through
3517   case vmIntrinsics::_dexp          : // fall through
3518   case vmIntrinsics::_dpow          : // fall through

3521       // compilation environment because the bytecodes for the method
3522       // shouldn't be parsed during the compilation, only the special
3523       // Intrinsic node should be emitted.  If this isn't done the the
3524       // code for the inlined version will be different than the root
3525       // compiled version which could lead to monotonicity problems on
3526       // intel.

3528         BAILOUT("failed to inline intrinsic, method not annotated");

3563         // the referent field of the reference object gets recorded by
3564         // the pre-barrier code.

3576

3578         ciBytecodeStream s(scope->method());

3582

3584         _block = start_block;
3585         _state = start_block->state()->copy_for_parsing();

3601

3612 
3613   NOT_PRODUCT(if (PrintValueNumbering && Verbose) print_stats());

3615   if (osr_bci != -1) {
3616     BlockBegin* osr_block = blm.bci2block()->at(osr_bci);
3617     if (!osr_block->is_set(BlockBegin::was_visited_flag)) {

3624     }
3625   }

3676 bool GraphBuilder::try_inline(ciMethod* callee, bool holder_known, bool ignore_return, Bytecodes::Code bc, Value receiver) {
3677   const char* msg = NULL;
3678 
3679   // clear out any existing inline bailout condition
3680   clear_inline_bailout();
3681 
3682   // exclude methods we don't want to inline
3683   msg = should_not_inline(callee);
3684   if (msg != NULL) {
3685     print_inlining(callee, msg, /*success*/ false);
3686     return false;
3687   }
3688 
3689   // method handle invokes
3690   if (callee->is_method_handle_intrinsic()) {
3691     if (try_method_handle_inline(callee, ignore_return)) {
3692       if (callee->has_reserved_stack_access()) {
3693         compilation()->set_has_reserved_stack_access(true);
3694       }
3695       return true;
3696     }
3697     return false;
3698   }
3699 
3700   // handle intrinsics
3701   if (callee->intrinsic_id() != vmIntrinsics::_none &&
3702       (CheckIntrinsics ? callee->intrinsic_candidate() : true)) {
3703     if (try_inline_intrinsics(callee, ignore_return)) {
3704       print_inlining(callee, "intrinsic");
3705       if (callee->has_reserved_stack_access()) {
3706         compilation()->set_has_reserved_stack_access(true);
3707       }
3708       return true;
3709     }
3710     // try normal inlining
3711   }
3712 
3713   // certain methods cannot be parsed at all
3714   msg = check_can_parse(callee);
3715   if (msg != NULL) {
3716     print_inlining(callee, msg, /*success*/ false);
3717     return false;
3718   }
3719 
3720   // If bytecode not set use the current one.
3721   if (bc == Bytecodes::_illegal) {
3722     bc = code();
3723   }
3724   if (try_inline_full(callee, holder_known, ignore_return, bc, receiver)) {
3725     if (callee->has_reserved_stack_access()) {
3726       compilation()->set_has_reserved_stack_access(true);
3727     }
3728     return true;
3729   }
3730 
3731   // Entire compilation could fail during try_inline_full call.
3732   // In that case printing inlining decision info is useless.
3733   if (!bailed_out())
3734     print_inlining(callee, _inline_bailout_msg, /*success*/ false);
3735 
3736   return false;
3737 }

3884 bool GraphBuilder::try_inline_jsr(int jsr_dest_bci) {
3885   // Introduce a new callee continuation point - all Ret instructions
3886   // will be replaced with Gotos to this point.
3887   BlockBegin* cont = block_at(next_bci());
3888   assert(cont != NULL, "continuation must exist (BlockListBuilder starts a new block after a jsr");
3889 
3890   // Note: can not assign state to continuation yet, as we have to
3891   // pick up the state from the Ret instructions.
3892 
3893   // Push callee scope
3894   push_scope_for_jsr(cont, jsr_dest_bci);
3895 
3896   // Temporarily set up bytecode stream so we can append instructions
3897   // (only using the bci of this stream)
3898   scope_data()->set_stream(scope_data()->parent()->stream());
3899 
3900   BlockBegin* jsr_start_block = block_at(jsr_dest_bci);
3901   assert(jsr_start_block != NULL, "jsr start block must exist");
3902   assert(!jsr_start_block->is_set(BlockBegin::was_visited_flag), "should not have visited jsr yet");
3903   Goto* goto_sub = new Goto(jsr_start_block, false);
3904   // Must copy state to avoid wrong sharing when parsing bytecodes
3905   assert(jsr_start_block->state() == NULL, "should have fresh jsr starting block");
3906   jsr_start_block->set_state(copy_state_before_with_bci(jsr_dest_bci));
3907   append(goto_sub);
3908   _block->set_end(goto_sub);
3909   _last = _block = jsr_start_block;
3910 
3911   // Clear out bytecode stream
3912   scope_data()->set_stream(NULL);
3913 
3914   scope_data()->add_to_work_list(jsr_start_block);
3915 
3916   // Ready to resume parsing in subroutine
3917   iterate_all_blocks();
3918 
3919   // If we bailed out during parsing, return immediately (this is bad news)
3920   CHECK_BAILOUT_(false);
3921 
3922   // Detect whether the continuation can actually be reached. If not,
3923   // it has not had state set by the join() operations in
3924   // iterate_bytecodes_for_block()/ret() and we should not touch the
3925   // iteration state. The calling activation of
3926   // iterate_bytecodes_for_block will then complete normally.
3927   if (cont->state() != NULL) {
3928     if (!cont->is_set(BlockBegin::was_visited_flag)) {
3929       // add continuation to work list instead of parsing it immediately
3930       scope_data()->parent()->add_to_work_list(cont);
3931     }
3932   }
3933 
3934   assert(jsr_continuation() == cont, "continuation must not have changed");

3937          "continuation can only be visited in case of backward branches");

3939 
3940   // continuation is in work list, so end iteration of current block
3941   _skip_block = true;
3942   pop_scope_for_jsr();
3943 
3944   return true;
3945 }

4036   if (CompilationPolicy::policy()->should_not_inline(compilation()->env(), callee)) {
4037     INLINE_BAILOUT("inlining prohibited by policy");
4038   }
4039   // first perform tests of things it's not possible to inline
4040   if (callee->has_exception_handlers() &&
4041       !InlineMethodsWithExceptionHandlers) INLINE_BAILOUT("callee has exception handlers");
4042   if (callee->is_synchronized() &&
4043       !InlineSynchronizedMethods         ) INLINE_BAILOUT("callee is synchronized");
4044   if (!callee->holder()->is_initialized()) INLINE_BAILOUT("callee's klass not initialized yet");
4045   if (!callee->has_balanced_monitors())    INLINE_BAILOUT("callee's monitors do not match");
4046 
4047   // Proper inlining of methods with jsrs requires a little more work.
4048   if (callee->has_jsrs()                 ) INLINE_BAILOUT("jsrs not handled properly by inliner yet");
4049 
4050   if (strict_fp_requires_explicit_rounding &&
4051       method()->is_strict() != callee->is_strict()) {
4052 #ifdef IA32
4053     // If explicit rounding is required, do not inline strict code into non-strict code (or the reverse).
4054     // When SSE2 is present, no special handling is needed.
4055     if (UseSSE < 2) {
4056       INLINE_BAILOUT("caller and callee have different strict fp requirements");
4057     }
4058 #else
4059     Unimplemented();
4060 #endif // IA32
4061   }
4062 
4063   if (is_profiling() && !callee->ensure_method_data()) {
4064     INLINE_BAILOUT("mdo allocation failed");
4065   }
4066 
4067   const bool is_invokedynamic = (bc == Bytecodes::_invokedynamic);
4068   const bool has_receiver = (bc != Bytecodes::_invokestatic && !is_invokedynamic);
4069 
4070   const int args_base = state()->stack_size() - callee->arg_size();
4071   assert(args_base >= 0, "stack underflow during inlining");
4072

4074   if (has_receiver) {
4075     assert(!callee->is_static(), "callee must not be static");
4076     assert(callee->arg_size() > 0, "must have at least a receiver");
4077 
4078     recv = state()->stack_at(args_base);
4079     if (recv->is_null_obj()) {
4080       INLINE_BAILOUT("receiver is always null");
4081     }
4082   }
4083 
4084   // now perform tests that are based on flag settings
4085   bool inlinee_by_directive = compilation()->directive()->should_inline(callee);
4086   if (callee->force_inline() || inlinee_by_directive) {
4087     if (inline_level() > MaxForceInlineLevel                      ) INLINE_BAILOUT("MaxForceInlineLevel");
4088     if (recursive_inline_level(callee) > C1MaxRecursiveInlineLevel) INLINE_BAILOUT("recursive inlining too deep");
4089

4091     if (callee->force_inline())  msg = "force inline by annotation";
4092     if (inlinee_by_directive)    msg = "force inline by CompileCommand";
4093     print_inlining(callee, msg);
4094   } else {
4095     // use heuristic controls on inlining
4096     if (inline_level() > C1MaxInlineLevel                       ) INLINE_BAILOUT("inlining too deep");
4097     int callee_recursive_level = recursive_inline_level(callee);
4098     if (callee_recursive_level > C1MaxRecursiveInlineLevel      ) INLINE_BAILOUT("recursive inlining too deep");
4099     if (callee->code_size_for_inlining() > max_inline_size()    ) INLINE_BAILOUT("callee is too large");
4100     // Additional condition to limit stack usage for non-recursive calls.
4101     if ((callee_recursive_level == 0) &&
4102         (callee->max_stack() + callee->max_locals() - callee->size_of_parameters() > C1InlineStackLimit)) {
4103       INLINE_BAILOUT("callee uses too much stack");
4104     }
4105 
4106     // don't inline throwable methods unless the inlining tree is rooted in a throwable class
4107     if (callee->name() == ciSymbol::object_initializer_name() &&
4108         callee->holder()->is_subclass_of(ciEnv::current()->Throwable_klass())) {
4109       // Throwable constructor call
4110       IRScope* top = scope();
4111       while (top->caller() != NULL) {
4112         top = top->caller();

4115         INLINE_BAILOUT("don't inline Throwable constructors");
4116       }
4117     }
4118

4120       INLINE_BAILOUT("total inlining greater than DesiredMethodLimit");
4121     }

4124   }

4127   // GraphBuilder level, do not cause bailout just of the inlining but

4132   // Insert null check if necessary
4133   if (has_receiver) {
4134     // note: null check must happen even if first instruction of callee does
4135     //       an implicit null check since the callee is in a different scope
4136     //       and we must make sure exception handling does the right thing
4137     null_check(recv);
4138   }
4139

4141     // Note that we'd collect profile data in this method if we wanted it.
4142     // this may be redundant here...

4144

4194   // (only using the bci of this stream)

4209   // (see use of pop_scope() below)
4210   caller_state->truncate_stack(args_base);
4211   assert(callee_state->stack_size() == 0, "callee stack must be empty");

4214   BlockBegin* sync_handler = NULL;
4215

4219                                : state()->local_at(0);

4235   if (callee_start_block != NULL) {

4239     // the entry bci for the callee instead of the call site bci.
4240     append_with_bci(goto_callee, 0);
4241     _block->set_end(goto_callee);

4259

4267   // practice, we have only traversed the continuation if we are
4268   // inlining into a subroutine

4274   // the caller back in the GraphBuilder. The only thing we want to do
4275   // first is an optimization: during parsing of the callee we
4276   // generated at least one Goto to the continuation block. If we
4277   // generated exactly one, and if the inlined method spanned exactly
4278   // one block (and we didn't have to Goto its entry), then we snip
4279   // off the Goto to the continuation, allowing control to fall
4280   // through back into the caller block and effectively performing
4281   // block merging. This allows load elimination and CSE to take place
4282   // across multiple callee scopes if they are relatively simple, and
4283   // is currently essential to making inlining profitable.

4287     _last  = inline_cleanup_return_prev();
4288     _state = inline_cleanup_state();
4289   } else if (continuation_preds == cont->number_of_preds()) {
4290     // Inlining caused that the instructions after the invoke in the
4291     // caller are not reachable any more. So skip filling this block
4292     // with instructions!
4293     assert(cont == continuation(), "");
4294     assert(_last && _last->as_BlockEnd(), "");
4295     _skip_block = true;
4296   } else {
4297     // Resume parsing in continuation block unless it was already parsed.
4298     // Note that if we don't change _last here, iteration in
4299     // iterate_bytecodes_for_block will stop when we return.
4300     if (!continuation()->is_set(BlockBegin::was_visited_flag)) {
4301       // add continuation to work list instead of parsing it immediately
4302       assert(_last && _last->as_BlockEnd(), "");
4303       scope_data()->parent()->add_to_work_list(continuation());
4304       _skip_block = true;
4305     }
4306   }
4307 
4308   // Fill the exception handler for synchronized methods with instructions
4309   if (callee->is_synchronized() && sync_handler->state() != NULL) {
4310     fill_sync_handler(lock, sync_handler);
4311   } else {
4312     pop_scope();
4313   }
4314 
4315   compilation()->notice_inlined_method(callee);
4316 
4317   return true;
4318 }

4321 bool GraphBuilder::try_method_handle_inline(ciMethod* callee, bool ignore_return) {
4322   ValueStack* state_before = copy_state_before();
4323   vmIntrinsics::ID iid = callee->intrinsic_id();
4324   switch (iid) {
4325   case vmIntrinsics::_invokeBasic:
4326     {
4327       // get MethodHandle receiver
4328       const int args_base = state()->stack_size() - callee->arg_size();
4329       ValueType* type = state()->stack_at(args_base)->type();
4330       if (type->is_constant()) {
4331         ciMethod* target = type->as_ObjectType()->constant_value()->as_method_handle()->get_vmtarget();
4332         // We don't do CHA here so only inline static and statically bindable methods.
4333         if (target->is_static() || target->can_be_statically_bound()) {
4334           if (ciMethod::is_consistent_info(callee, target)) {
4335             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4336             ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4337             if (try_inline(target, /*holder_known*/ !callee->is_static(), ignore_return, bc)) {
4338               return true;
4339             }
4340           } else {
4341             print_inlining(target, "signatures mismatch", /*success*/ false);
4342           }
4343         } else {
4344           print_inlining(target, "not static or statically bindable", /*success*/ false);
4345         }
4346       } else {
4347         print_inlining(callee, "receiver not constant", /*success*/ false);
4348       }
4349     }
4350     break;
4351 
4352   case vmIntrinsics::_linkToVirtual:
4353   case vmIntrinsics::_linkToStatic:
4354   case vmIntrinsics::_linkToSpecial:
4355   case vmIntrinsics::_linkToInterface:
4356     {
4357       // pop MemberName argument
4358       const int args_base = state()->stack_size() - callee->arg_size();
4359       ValueType* type = apop()->type();
4360       if (type->is_constant()) {
4361         ciMethod* target = type->as_ObjectType()->constant_value()->as_member_name()->get_vmtarget();
4362         ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4363         // If the target is another method handle invoke, try to recursively get
4364         // a better target.
4365         if (target->is_method_handle_intrinsic()) {
4366           if (try_method_handle_inline(target, ignore_return)) {
4367             return true;
4368           }
4369         } else if (!ciMethod::is_consistent_info(callee, target)) {
4370           print_inlining(target, "signatures mismatch", /*success*/ false);
4371         } else {
4372           ciSignature* signature = target->signature();
4373           const int receiver_skip = target->is_static() ? 0 : 1;
4374           // Cast receiver to its type.
4375           if (!target->is_static()) {
4376             ciKlass* tk = signature->accessing_klass();
4377             Value obj = state()->stack_at(args_base);
4378             if (obj->exact_type() == NULL &&
4379                 obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4380               TypeCast* c = new TypeCast(tk, obj, state_before);
4381               append(c);
4382               state()->stack_at_put(args_base, c);
4383             }
4384           }
4385           // Cast reference arguments to its type.
4386           for (int i = 0, j = 0; i < signature->count(); i++) {
4387             ciType* t = signature->type_at(i);
4388             if (t->is_klass()) {
4389               ciKlass* tk = t->as_klass();
4390               Value obj = state()->stack_at(args_base + receiver_skip + j);
4391               if (obj->exact_type() == NULL &&
4392                   obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4393                 TypeCast* c = new TypeCast(t, obj, state_before);
4394                 append(c);
4395                 state()->stack_at_put(args_base + receiver_skip + j, c);
4396               }
4397             }
4398             j += t->size();  // long and double take two slots
4399           }
4400           // We don't do CHA here so only inline static and statically bindable methods.
4401           if (target->is_static() || target->can_be_statically_bound()) {
4402             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4403             if (try_inline(target, /*holder_known*/ !callee->is_static(), ignore_return, bc)) {
4404               return true;
4405             }
4406           } else {
4407             print_inlining(target, "not static or statically bindable", /*success*/ false);
4408           }
4409         }
4410       } else {
4411         print_inlining(callee, "MemberName not constant", /*success*/ false);
4412       }
4413     }
4414     break;
4415 
4416   default:
4417     fatal("unexpected intrinsic %d: %s", iid, vmIntrinsics::name_at(iid));
4418     break;
4419   }
4420   set_state(state_before->copy_for_parsing());
4421   return false;
4422 }
