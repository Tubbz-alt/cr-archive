  55 Node* Parse::record_profile_for_speculation_at_array_load(Node* ld) {
  56   // Feed unused profile data to type speculation
  57   if (UseTypeSpeculation && UseArrayLoadStoreProfile) {
  58     ciKlass* array_type = NULL;
  59     ciKlass* element_type = NULL;
  60     ProfilePtrKind element_ptr = ProfileMaybeNull;
  61     bool flat_array = true;
  62     bool null_free_array = true;
  63     method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
  64     if (element_type != NULL || element_ptr != ProfileMaybeNull) {
  65       ld = record_profile_for_speculation(ld, element_type, element_ptr);
  66     }
  67   }
  68   return ld;
  69 }

  72 //---------------------------------array_load----------------------------------
  73 void Parse::array_load(BasicType bt) {
  74   const Type* elemtype = Type::TOP;
  75   Node* adr = array_addressing(bt, 0, elemtype);
  76   if (stopped())  return;     // guaranteed null or range check
  77 
  78   Node* idx = pop();
  79   Node* ary = pop();
  80 
  81   // Handle value type arrays
  82   const TypeOopPtr* elemptr = elemtype->make_oopptr();
  83   const TypeAryPtr* ary_t = _gvn.type(ary)->is_aryptr();
  84   if (elemtype->isa_valuetype() != NULL) {
  85     C->set_flattened_accesses();
  86     // Load from flattened value type array
  87     Node* vt = ValueTypeNode::make_from_flattened(this, elemtype->value_klass(), ary, adr);
  88     push(vt);
  89     return;
  90   } else if (elemptr != NULL && elemptr->is_valuetypeptr() && !elemptr->maybe_null()) {
  91     // Load from non-flattened inline type array (elements can never be null)
  92     bt = T_INLINE_TYPE;
  93   } else if (!ary_t->is_not_flat()) {
  94     // Cannot statically determine if array is flattened, emit runtime check
  95     assert(UseFlatArray && is_reference_type(bt) && elemptr->can_be_value_type() && !ary_t->klass_is_exact() && !ary_t->is_not_null_free() &&
  96            (!elemptr->is_valuetypeptr() || elemptr->value_klass()->flatten_array()), "array can't be flattened");
  97     IdealKit ideal(this);
  98     IdealVariable res(ideal);
  99     ideal.declarations_done();
 100     ideal.if_then(is_non_flattened_array(ary)); {
 101       // non-flattened
 102       assert(ideal.ctrl()->in(0)->as_If()->is_non_flattened_array_check(&_gvn), "Should be found");
 103       sync_kit(ideal);
 104       const TypeAryPtr* adr_type = TypeAryPtr::get_array_body_type(bt);
 105       Node* ld = access_load_at(ary, adr, adr_type, elemptr, bt,
 106                                 IN_HEAP | IS_ARRAY | C2_CONTROL_DEPENDENT_LOAD);
 107       ideal.sync_kit(this);
 108       ideal.set(res, ld);
 109     } ideal.else_(); {
 110       // flattened
 111       sync_kit(ideal);
 112       if (elemptr->is_valuetypeptr()) {
 113         // Element type is known, cast and load from flattened representation
 114         ciValueKlass* vk = elemptr->value_klass();
 115         assert(vk->flatten_array() && elemptr->maybe_null(), "never/always flat - should be optimized");
 116         ciArrayKlass* array_klass = ciArrayKlass::make(vk);
 117         const TypeAryPtr* arytype = TypeOopPtr::make_from_klass(array_klass)->isa_aryptr();
 118         Node* cast = _gvn.transform(new CheckCastPPNode(control(), ary, arytype));
 119         Node* casted_adr = array_element_address(cast, idx, T_INLINE_TYPE, ary_t->size(), control());
 120         // Re-execute flattened array load if buffering triggers deoptimization
 121         PreserveReexecuteState preexecs(this);
 122         jvms()->set_should_reexecute(true);
 123         inc_sp(2);
 124         Node* vt = ValueTypeNode::make_from_flattened(this, vk, cast, casted_adr)->buffer(this, false);
 125         ideal.set(res, vt);
 126         ideal.sync_kit(this);
 127       } else {
 128         // Element type is unknown, emit runtime call
 129         Node* kls = load_object_klass(ary);
 130         Node* k_adr = basic_plus_adr(kls, in_bytes(ArrayKlass::element_klass_offset()));
 131         Node* elem_klass = _gvn.transform(LoadKlassNode::make(_gvn, NULL, immutable_memory(), k_adr, TypeInstPtr::KLASS));
 132         Node* obj_size  = NULL;
 133         kill_dead_locals();
 134         // Re-execute flattened array load if buffering triggers deoptimization
 135         PreserveReexecuteState preexecs(this);
 136         jvms()->set_bci(_bci);
 137         jvms()->set_should_reexecute(true);
 138         inc_sp(2);
 139         Node* alloc_obj = new_instance(elem_klass, NULL, &obj_size, /*deoptimize_on_exception=*/true);
 140 
 141         AllocateNode* alloc = AllocateNode::Ideal_allocation(alloc_obj, &_gvn);
 142         assert(alloc->maybe_set_complete(&_gvn), "");
 143         alloc->initialization()->set_complete_with_arraycopy();
 144 
 145         // This membar keeps this access to an unknown flattened array
 146         // correctly ordered with other unknown and known flattened
 147         // array accesses.
 148         insert_mem_bar_volatile(Op_MemBarCPUOrder, C->get_alias_index(TypeAryPtr::VALUES));
 149 
 150         BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 151         // Unknown value type might contain reference fields
 152         if (false && !bs->array_copy_requires_gc_barriers(false, T_OBJECT, false, BarrierSetC2::Parsing)) {
 153           // FIXME 8230656 also merge changes from 8238759 in
 154           int base_off = sizeof(instanceOopDesc);
 155           Node* dst_base = basic_plus_adr(alloc_obj, base_off);
 156           Node* countx = obj_size;
 157           countx = _gvn.transform(new SubXNode(countx, MakeConX(base_off)));
 158           countx = _gvn.transform(new URShiftXNode(countx, intcon(LogBytesPerLong)));
 159 
 160           assert(Klass::_lh_log2_element_size_shift == 0, "use shift in place");
 161           Node* lhp = basic_plus_adr(kls, in_bytes(Klass::layout_helper_offset()));
 162           Node* elem_shift = make_load(NULL, lhp, TypeInt::INT, T_INT, MemNode::unordered);
 163           uint header = arrayOopDesc::base_offset_in_bytes(T_INLINE_TYPE);
 164           Node* base  = basic_plus_adr(ary, header);
 165           idx = Compile::conv_I2X_index(&_gvn, idx, TypeInt::POS, control());
 166           Node* scale = _gvn.transform(new LShiftXNode(idx, elem_shift));
 167           Node* adr = basic_plus_adr(ary, base, scale);
 168 
 169           access_clone(adr, dst_base, countx, false);
 170         } else {
 171           ideal.sync_kit(this);
 172           ideal.make_leaf_call(OptoRuntime::load_unknown_value_Type(),
 173                                CAST_FROM_FN_PTR(address, OptoRuntime::load_unknown_value),
 174                                "load_unknown_value",
 175                                ary, idx, alloc_obj);
 176           sync_kit(ideal);
 177         }
 178 
 179         // This makes sure no other thread sees a partially initialized buffered value
 180         insert_mem_bar_volatile(Op_MemBarStoreStore, Compile::AliasIdxRaw, alloc->proj_out_or_null(AllocateNode::RawAddress));
 181 
 182         // Same as MemBarCPUOrder above: keep this unknown flattened
 183         // array access correctly ordered with other flattened array
 184         // access
 185         insert_mem_bar_volatile(Op_MemBarCPUOrder, C->get_alias_index(TypeAryPtr::VALUES));
 186 
 187         // Prevent any use of the newly allocated value before it is
 188         // fully initialized
 189         alloc_obj = new CastPPNode(alloc_obj, _gvn.type(alloc_obj), true);
 190         alloc_obj->set_req(0, control());
 191         alloc_obj = _gvn.transform(alloc_obj);
 192 
 193         const Type* unknown_value = elemptr->is_instptr()->cast_to_flat_array();
 194         alloc_obj = _gvn.transform(new CheckCastPPNode(control(), alloc_obj, unknown_value));
 195 
 196         ideal.sync_kit(this);
 197         ideal.set(res, alloc_obj);
 198       }
 199     } ideal.end_if();
 200     sync_kit(ideal);
 201     Node* ld = _gvn.transform(ideal.value(res));
 202     ld = record_profile_for_speculation_at_array_load(ld);
 203     push_node(bt, ld);
 204     return;
 205   }
 206 
 207   if (elemtype == TypeInt::BOOL) {
 208     bt = T_BOOLEAN;
 209   }
 210   const TypeAryPtr* adr_type = TypeAryPtr::get_array_body_type(bt);
 211   Node* ld = access_load_at(ary, adr, adr_type, elemtype, bt,
 212                             IN_HEAP | IS_ARRAY | C2_CONTROL_DEPENDENT_LOAD);
 213   if (bt == T_INLINE_TYPE) {
 214     // Loading a non-flattened value type from an array
 215     assert(!gvn().type(ld)->maybe_null(), "value type array elements should never be null");
 216     if (elemptr->value_klass()->is_scalarizable()) {
 217       ld = ValueTypeNode::make_from_oop(this, ld, elemptr->value_klass());
 218     }
 219   }
 220   if (!ld->is_ValueType()) {
 221     ld = record_profile_for_speculation_at_array_load(ld);
 222   }
 223 
 224   push_node(bt, ld);
 225 }

 228 //--------------------------------array_store----------------------------------
 229 void Parse::array_store(BasicType bt) {
 230   const Type* elemtype = Type::TOP;
 231   Node* adr = array_addressing(bt, type2size[bt], elemtype);
 232   if (stopped())  return;     // guaranteed null or range check
 233   Node* cast_val = NULL;
 234   if (bt == T_OBJECT) {
 235     cast_val = array_store_check();
 236     if (stopped()) return;
 237   }
 238   Node* val = pop_node(bt); // Value to store
 239   Node* idx = pop();        // Index in the array
 240   Node* ary = pop();        // The array itself
 241 
 242   const TypeAryPtr* ary_t = _gvn.type(ary)->is_aryptr();
 243   const TypeAryPtr* adr_type = TypeAryPtr::get_array_body_type(bt);
 244 
 245   if (elemtype == TypeInt::BOOL) {
 246     bt = T_BOOLEAN;
 247   } else if (bt == T_OBJECT) {
 248     elemtype = elemtype->make_oopptr();
 249     const Type* tval = _gvn.type(cast_val);
 250     // We may have lost type information for 'val' here due to the casts
 251     // emitted by the array_store_check code (see JDK-6312651)
 252     // TODO Remove this code once JDK-6312651 is in.
 253     const Type* tval_init = _gvn.type(val);
 254     bool not_inline = !tval->isa_valuetype() && (tval == TypePtr::NULL_PTR || !tval_init->is_oopptr()->can_be_value_type() || !tval->is_oopptr()->can_be_value_type());
 255     bool not_flattened = !UseFlatArray || not_inline || ((tval_init->is_valuetypeptr() || tval_init->isa_valuetype()) && !tval_init->value_klass()->flatten_array());
 256 
 257     if (!ary_t->is_not_null_free() && not_inline && (!tval->maybe_null() || !tval_init->maybe_null())) {
 258       // Storing a non-inline type, mark array as not null-free (-> not flat).
 259       // This is only legal for non-null stores because the array_store_check always passes for null.
 260       // Null stores are handled in GraphKit::gen_value_array_null_guard().
 261       ary_t = ary_t->cast_to_not_null_free();
 262       Node* cast = _gvn.transform(new CheckCastPPNode(control(), ary, ary_t));
 263       replace_in_map(ary, cast);
 264       ary = cast;
 265     } else if (!ary_t->is_not_flat() && not_flattened) {
 266       // Storing a non-flattened value, mark array as not flat.
 267       ary_t = ary_t->cast_to_not_flat();
 268       Node* cast = _gvn.transform(new CheckCastPPNode(control(), ary, ary_t));
 269       replace_in_map(ary, cast);
 270       ary = cast;
 271     }
 272 
 273     if (ary_t->elem()->isa_valuetype() != NULL) {
 274       // Store to flattened value type array
 275       C->set_flattened_accesses();
 276       if (!cast_val->is_ValueType()) {
 277         inc_sp(3);
 278         cast_val = null_check(cast_val);
 279         if (stopped()) return;
 280         dec_sp(3);
 281         cast_val = ValueTypeNode::make_from_oop(this, cast_val, ary_t->elem()->value_klass());
 282       }
 283       // Re-execute flattened array store if buffering triggers deoptimization
 284       PreserveReexecuteState preexecs(this);
 285       inc_sp(3);
 286       jvms()->set_should_reexecute(true);
 287       cast_val->as_ValueType()->store_flattened(this, ary, adr, NULL, 0, MO_UNORDERED | IN_HEAP | IS_ARRAY);
 288       return;
 289     } else if (elemtype->is_valuetypeptr() && !elemtype->maybe_null()) {
 290       // Store to non-flattened inline type array (elements can never be null)
 291       if (!cast_val->is_ValueType() && tval->maybe_null()) {
 292         inc_sp(3);
 293         cast_val = null_check(cast_val);
 294         if (stopped()) return;
 295         dec_sp(3);
 296       }
 297     } else if (!ary_t->is_not_flat()) {
 298       // Array might be flattened, emit runtime checks
 299       assert(UseFlatArray && !not_flattened && elemtype->is_oopptr()->can_be_value_type() &&
 300              !ary_t->klass_is_exact() && !ary_t->is_not_null_free(), "array can't be flattened");
 301       IdealKit ideal(this);
 302       ideal.if_then(is_non_flattened_array(ary)); {
 303         // non-flattened
 304         assert(ideal.ctrl()->in(0)->as_If()->is_non_flattened_array_check(&_gvn), "Should be found");
 305         sync_kit(ideal);
 306         gen_value_array_null_guard(ary, cast_val, 3);
 307         inc_sp(3);
 308         access_store_at(ary, adr, adr_type, cast_val, elemtype, bt, MO_UNORDERED | IN_HEAP | IS_ARRAY, false);
 309         dec_sp(3);
 310         ideal.sync_kit(this);
 311       } ideal.else_(); {
 312         Node* val = cast_val;
 313         // flattened
 314         if (!val->is_ValueType() && tval->maybe_null()) {
 315           // Add null check
 316           sync_kit(ideal);
 317           Node* null_ctl = top();
 318           val = null_check_oop(val, &null_ctl);
 319           if (null_ctl != top()) {
 320             PreserveJVMState pjvms(this);
 321             inc_sp(3);
 322             set_control(null_ctl);
 323             uncommon_trap(Deoptimization::Reason_null_check, Deoptimization::Action_none);
 324             dec_sp(3);
 325           }
 326           ideal.sync_kit(this);
 327         }
 328         // Try to determine the value klass
 329         ciValueKlass* vk = NULL;
 330         if (tval->isa_valuetype() || tval->is_valuetypeptr()) {
 331           vk = tval->value_klass();
 332         } else if (tval_init->isa_valuetype() || tval_init->is_valuetypeptr()) {
 333           vk = tval_init->value_klass();
 334         } else if (elemtype->is_valuetypeptr()) {
 335           vk = elemtype->value_klass();
 336         }
 337         Node* casted_ary = ary;
 338         if (vk != NULL && !stopped()) {
 339           // Element type is known, cast and store to flattened representation
 340           sync_kit(ideal);
 341           assert(vk->flatten_array() && elemtype->maybe_null(), "never/always flat - should be optimized");
 342           ciArrayKlass* array_klass = ciArrayKlass::make(vk);
 343           const TypeAryPtr* arytype = TypeOopPtr::make_from_klass(array_klass)->isa_aryptr();
 344           casted_ary = _gvn.transform(new CheckCastPPNode(control(), casted_ary, arytype));
 345           Node* casted_adr = array_element_address(casted_ary, idx, T_OBJECT, arytype->size(), control());
 346           if (!val->is_ValueType()) {
 347             assert(!gvn().type(val)->maybe_null(), "value type array elements should never be null");
 348             val = ValueTypeNode::make_from_oop(this, val, vk);
 349           }
 350           // Re-execute flattened array store if buffering triggers deoptimization
 351           PreserveReexecuteState preexecs(this);
 352           inc_sp(3);
 353           jvms()->set_should_reexecute(true);
 354           val->as_ValueType()->store_flattened(this, casted_ary, casted_adr, NULL, 0, MO_UNORDERED | IN_HEAP | IS_ARRAY);
 355           ideal.sync_kit(this);
 356         } else if (!ideal.ctrl()->is_top()) {
 357           // Element type is unknown, emit runtime call
 358           sync_kit(ideal);
 359 
 360           // This membar keeps this access to an unknown flattened
 361           // array correctly ordered with other unknown and known
 362           // flattened array accesses.
 363           insert_mem_bar_volatile(Op_MemBarCPUOrder, C->get_alias_index(TypeAryPtr::VALUES));
 364           ideal.sync_kit(this);
 365 
 366           ideal.make_leaf_call(OptoRuntime::store_unknown_value_Type(),
 367                                CAST_FROM_FN_PTR(address, OptoRuntime::store_unknown_value),
 368                                "store_unknown_value",
 369                                val, casted_ary, idx);
 370 
 371           sync_kit(ideal);
 372           // Same as MemBarCPUOrder above: keep this unknown
 373           // flattened array access correctly ordered with other
 374           // flattened array accesses.
 375           insert_mem_bar_volatile(Op_MemBarCPUOrder, C->get_alias_index(TypeAryPtr::VALUES));
 376           ideal.sync_kit(this);
 377         }
 378       }
 379       ideal.end_if();
 380       sync_kit(ideal);
 381       return;
 382     } else if (!ary_t->is_not_null_free()) {
 383       // Array is not flattened but may be null free
 384       assert(elemtype->is_oopptr()->can_be_value_type() && !ary_t->klass_is_exact(), "array can't be null free");
 385       ary = gen_value_array_null_guard(ary, cast_val, 3, true);
 386     }
 387   }
 388   inc_sp(3);
 389   access_store_at(ary, adr, adr_type, val, elemtype, bt, MO_UNORDERED | IN_HEAP | IS_ARRAY);
 390   dec_sp(3);
 391 }

 394 //------------------------------array_addressing-------------------------------
 395 // Pull array and index from the stack.  Compute pointer-to-element.
 396 Node* Parse::array_addressing(BasicType type, int vals, const Type*& elemtype) {
 397   Node *idx   = peek(0+vals);   // Get from stack without popping
 398   Node *ary   = peek(1+vals);   // in case of exception
 399 
 400   // Null check the array base, with correct stack contents
 401   ary = null_check(ary, T_ARRAY);
 402   // Compile-time detect of null-exception?
 403   if (stopped())  return top();
 404 
 405   const TypeAryPtr* arytype  = _gvn.type(ary)->is_aryptr();
 406   const TypeInt*    sizetype = arytype->size();
 407   elemtype = arytype->elem();
 408 
 409   if (UseUniqueSubclasses) {
 410     const Type* el = elemtype->make_ptr();
 411     if (el && el->isa_instptr()) {
 412       const TypeInstPtr* toop = el->is_instptr();
 413       if (toop->klass()->as_instance_klass()->unique_concrete_subklass()) {
 414         // If we load from "AbstractClass[]" we must see "ConcreteSubClass".
 415         const Type* subklass = Type::get_const_type(toop->klass());
 416         elemtype = subklass->join_speculative(el);
 417       }
 418     }
 419   }
 420 
 421   // Check for big class initializers with all constant offsets
 422   // feeding into a known-size array.
 423   const TypeInt* idxtype = _gvn.type(idx)->is_int();
 424   // See if the highest idx value is less than the lowest array bound,
 425   // and if the idx value cannot be negative:
 426   bool need_range_check = true;
 427   if (idxtype->_hi < sizetype->_lo && idxtype->_lo >= 0) {
 428     need_range_check = false;
 429     if (C->log() != NULL)   C->log()->elem("observe that='!need_range_check'");
 430   }
 431 
 432   ciKlass * arytype_klass = arytype->klass();
 433   if ((arytype_klass != NULL) && (!arytype_klass->is_loaded())) {
 434     // Only fails for some -Xcomp runs
 435     // The class is unloaded.  We have to run this bytecode in the interpreter.
 436     uncommon_trap(Deoptimization::Reason_unloaded,
 437                   Deoptimization::Action_reinterpret,
 438                   arytype->klass(), "!loaded array");
 439     return top();
 440   }
 441 
 442   // Do the range check
 443   if (GenerateRangeChecks && need_range_check) {
 444     Node* tst;
 445     if (sizetype->_hi <= 0) {
 446       // The greatest array bound is negative, so we can conclude that we're
 447       // compiling unreachable code, but the unsigned compare trick used below
 448       // only works with non-negative lengths.  Instead, hack "tst" to be zero so
 449       // the uncommon_trap path will always be taken.
 450       tst = _gvn.intcon(0);
 451     } else {
 452       // Range is constant in array-oop, so we can use the original state of mem
 453       Node* len = load_array_length(ary);
 454 
 455       // Test length vs index (standard trick using unsigned compare)
 456       Node* chk = _gvn.transform( new CmpUNode(idx, len) );
 457       BoolTest::mask btest = BoolTest::lt;
 458       tst = _gvn.transform( new BoolNode(chk, btest) );
 459     }
 460     RangeCheckNode* rc = new RangeCheckNode(control(), tst, PROB_MAX, COUNT_UNKNOWN);
 461     _gvn.set_type(rc, rc->Value(&_gvn));
 462     if (!tst->is_Con()) {
 463       record_for_igvn(rc);
 464     }
 465     set_control(_gvn.transform(new IfTrueNode(rc)));
 466     // Branch to failure if out of bounds
 467     {
 468       PreserveJVMState pjvms(this);
 469       set_control(_gvn.transform(new IfFalseNode(rc)));
 470       if (C->allow_range_check_smearing()) {
 471         // Do not use builtin_throw, since range checks are sometimes
 472         // made more stringent by an optimistic transformation.
 473         // This creates "tentative" range checks at this point,
 474         // which are not guaranteed to throw exceptions.
 475         // See IfNode::Ideal, is_range_check, adjust_check.
 476         uncommon_trap(Deoptimization::Reason_range_check,
 477                       Deoptimization::Action_make_not_entrant,
 478                       NULL, "range_check");
 479       } else {
 480         // If we have already recompiled with the range-check-widening
 481         // heroic optimization turned off, then we must really be throwing
 482         // range check exceptions.
 483         builtin_throw(Deoptimization::Reason_range_check, idx);
 484       }
 485     }
 486   }
 487   // Check for always knowing you are throwing a range-check exception
 488   if (stopped())  return top();
 489 
 490   // This could be an access to a value array. We can't tell if it's
 491   // flat or not. Speculating it's not leads to a much simpler graph
 492   // shape. Check profiling.
 493   // For aastore, by the time we're here, the array store check should
 494   // have already taken advantage of profiling to cast the array to an
 495   // exact type reported by profiling
 496   const TypeOopPtr* elemptr = elemtype->make_oopptr();
 497   if (elemtype->isa_valuetype() == NULL &&
 498       (elemptr == NULL || !elemptr->is_valuetypeptr() || elemptr->maybe_null()) &&
 499       !arytype->is_not_flat()) {
 500     assert(is_reference_type(type), "Only references");
 501     // First check the speculative type
 502     Deoptimization::DeoptReason reason = Deoptimization::Reason_speculate_class_check;
 503     ciKlass* array_type = arytype->speculative_type();
 504     if (too_many_traps_or_recompiles(reason) || array_type == NULL) {
 505       // No speculative type, check profile data at this bci
 506       array_type = NULL;
 507       reason = Deoptimization::Reason_class_check;
 508       if (UseArrayLoadStoreProfile && !too_many_traps_or_recompiles(reason)) {
 509         ciKlass* element_type = NULL;
 510         ProfilePtrKind element_ptr = ProfileMaybeNull;
 511         bool flat_array = true;
 512         bool null_free_array = true;
 513         method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
 514       }
 515     }
 516     if (array_type != NULL) {
 517       // Speculate that this array has the exact type reported by profile data
 518       Node* better_ary = NULL;
 519       Node* slow_ctl = type_check_receiver(ary, array_type, 1.0, &better_ary);
 520       { PreserveJVMState pjvms(this);
 521         set_control(slow_ctl);
 522         uncommon_trap_exact(reason, Deoptimization::Action_maybe_recompile);
 523       }
 524       replace_in_map(ary, better_ary);
 525       ary = better_ary;
 526       arytype  = _gvn.type(ary)->is_aryptr();
 527       elemtype = arytype->elem();
 528     }
 529   } else if (UseTypeSpeculation && UseArrayLoadStoreProfile) {
 530     // No need to speculate: feed profile data at this bci for the
 531     // array to type speculation
 532     ciKlass* array_type = NULL;
 533     ciKlass* element_type = NULL;
 534     ProfilePtrKind element_ptr = ProfileMaybeNull;
 535     bool flat_array = true;
 536     bool null_free_array = true;
 537     method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
 538     if (array_type != NULL) {
 539       record_profile_for_speculation(ary, array_type, ProfileMaybeNull);
 540     }
 541   }
 542 
 543   // We have no exact array type from profile data. Check profile data
 544   // for a non null free or non flat array. Non null free implies non
 545   // flat so check this one first. Speculating on a non null free
 546   // array doesn't help aaload but could be profitable for a
 547   // subsequent aastore.
 548   elemptr = elemtype->make_oopptr();
 549   if (!arytype->is_not_null_free() &&
 550       elemtype->isa_valuetype() == NULL &&
 551       (elemptr == NULL || !elemptr->is_valuetypeptr()) &&
 552       UseArrayLoadStoreProfile) {
 553     assert(is_reference_type(type), "");
 554     bool null_free_array = true;
 555     Deoptimization::DeoptReason reason = Deoptimization::Reason_none;
 556     if (arytype->speculative() != NULL &&
 557         arytype->speculative()->is_aryptr()->is_not_null_free() &&
 558         !too_many_traps_or_recompiles(Deoptimization::Reason_speculate_class_check)) {
 559       null_free_array = false;
 560       reason = Deoptimization::Reason_speculate_class_check;
 561     } else if (!too_many_traps_or_recompiles(Deoptimization::Reason_class_check)) {
 562       ciKlass* array_type = NULL;
 563       ciKlass* element_type = NULL;
 564       ProfilePtrKind element_ptr = ProfileMaybeNull;
 565       bool flat_array = true;
 566       method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
 567       reason = Deoptimization::Reason_class_check;
 568     }
 569     if (!null_free_array) {
 570       { // Deoptimize if null-free array
 571         BuildCutout unless(this, is_nullable_array(ary), PROB_MAX);
 572         uncommon_trap_exact(reason, Deoptimization::Action_maybe_recompile);
 573       }
 574       Node* better_ary = _gvn.transform(new CheckCastPPNode(control(), ary, arytype->cast_to_not_null_free()));
 575       replace_in_map(ary, better_ary);
 576       ary = better_ary;
 577       arytype = _gvn.type(ary)->is_aryptr();
 578     }
 579   }
 580 
 581   if (!arytype->is_not_flat() && elemtype->isa_valuetype() == NULL) {
 582     assert(is_reference_type(type), "");
 583     bool flat_array = true;
 584     Deoptimization::DeoptReason reason = Deoptimization::Reason_none;
 585     if (arytype->speculative() != NULL &&
 586         arytype->speculative()->is_aryptr()->is_not_flat() &&
 587         !too_many_traps_or_recompiles(Deoptimization::Reason_speculate_class_check)) {
 588       flat_array = false;
 589       reason = Deoptimization::Reason_speculate_class_check;
 590     } else if (UseArrayLoadStoreProfile && !too_many_traps_or_recompiles(reason)) {
 591       ciKlass* array_type = NULL;
 592       ciKlass* element_type = NULL;
 593       ProfilePtrKind element_ptr = ProfileMaybeNull;
 594       bool null_free_array = true;
 595       method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
 596       reason = Deoptimization::Reason_class_check;
 597     }
 598     if (!flat_array) {
 599       { // Deoptimize if flat array
 600         BuildCutout unless(this, is_non_flattened_array(ary), PROB_MAX);
 601         uncommon_trap_exact(reason, Deoptimization::Action_maybe_recompile);
 602       }
 603       Node* better_ary = _gvn.transform(new CheckCastPPNode(control(), ary, arytype->cast_to_not_flat()));
 604       replace_in_map(ary, better_ary);
 605       ary = better_ary;
 606       arytype = _gvn.type(ary)->is_aryptr();
 607     }
 608   }
 609 
 610   // Make array address computation control dependent to prevent it
 611   // from floating above the range check during loop optimizations.
 612   Node* ptr = array_element_address(ary, idx, type, sizetype, control());
 613   assert(ptr != top(), "top should go hand-in-hand with stopped");
 614 
 615   return ptr;
 616 }

 719 class SwitchRange : public StackObj {
 720   // a range of integers coupled with a bci destination
 721   jint _lo;                     // inclusive lower limit
 722   jint _hi;                     // inclusive upper limit
 723   int _dest;
 724   int _table_index;             // index into method data table
 725   float _cnt;                   // how many times this range was hit according to profiling
 726 
 727 public:
 728   jint lo() const              { return _lo;   }
 729   jint hi() const              { return _hi;   }
 730   int  dest() const            { return _dest; }
 731   int  table_index() const     { return _table_index; }
 732   bool is_singleton() const    { return _lo == _hi; }
 733   float cnt() const            { return _cnt; }
 734 
 735   void setRange(jint lo, jint hi, int dest, int table_index, float cnt) {
 736     assert(lo <= hi, "must be a non-empty range");
 737     _lo = lo, _hi = hi; _dest = dest; _table_index = table_index; _cnt = cnt;
 738     assert(_cnt >= 0, "");
 739   }
 740   bool adjoinRange(jint lo, jint hi, int dest, int table_index, float cnt, bool trim_ranges) {
 741     assert(lo <= hi, "must be a non-empty range");
 742     if (lo == _hi+1 && table_index == _table_index) {
 743       // see merge_ranges() comment below
 744       if (trim_ranges) {
 745         if (cnt == 0) {
 746           if (_cnt != 0) {
 747             return false;
 748           }
 749           if (dest != _dest) {
 750             _dest = never_reached;
 751           }
 752         } else {
 753           if (_cnt == 0) {
 754             return false;
 755           }
 756           if (dest != _dest) {
 757             return false;
 758           }
 759         }
 760       } else {
 761         if (dest != _dest) {
 762           return false;
 763         }
 764       }
 765       _hi = hi;
 766       _cnt += cnt;
 767       return true;
 768     }
 769     return false;
 770   }
 771 
 772   void set (jint value, int dest, int table_index, float cnt) {
 773     setRange(value, value, dest, table_index, cnt);
 774   }
 775   bool adjoin(jint value, int dest, int table_index, float cnt, bool trim_ranges) {
 776     return adjoinRange(value, value, dest, table_index, cnt, trim_ranges);
 777   }
 778   bool adjoin(SwitchRange& other) {
 779     return adjoinRange(other._lo, other._hi, other._dest, other._table_index, other._cnt, false);
 780   }

 782   void print() {
 783     if (is_singleton())

 785     else if (lo() == min_jint)
 786       tty->print(" {..%d}=>%d (cnt=%f)", hi(), dest(), cnt());
 787     else if (hi() == max_jint)
 788       tty->print(" {%d..}=>%d (cnt=%f)", lo(), dest(), cnt());
 789     else
 790       tty->print(" {%d..%d}=>%d (cnt=%f)", lo(), hi(), dest(), cnt());
 791   }

 794 // We try to minimize the number of ranges and the size of the taken
 795 // ones using profiling data. When ranges are created,
 796 // SwitchRange::adjoinRange() only allows 2 adjoining ranges to merge
 797 // if both were never hit or both were hit to build longer unreached
 798 // ranges. Here, we now merge adjoining ranges with the same
 799 // destination and finally set destination of unreached ranges to the
 800 // special value never_reached because it can help minimize the number
 801 // of tests that are necessary.
 802 //
 803 // For instance:
 804 // [0, 1] to target1 sometimes taken
 805 // [1, 2] to target1 never taken
 806 // [2, 3] to target2 never taken
 807 // would lead to:
 808 // [0, 1] to target1 sometimes taken
 809 // [1, 3] never taken
 810 //
 811 // (first 2 ranges to target1 are not merged)
 812 static void merge_ranges(SwitchRange* ranges, int& rp) {
 813   if (rp == 0) {
 814     return;
 815   }
 816   int shift = 0;
 817   for (int j = 0; j < rp; j++) {
 818     SwitchRange& r1 = ranges[j-shift];
 819     SwitchRange& r2 = ranges[j+1];
 820     if (r1.adjoin(r2)) {
 821       shift++;
 822     } else if (shift > 0) {
 823       ranges[j+1-shift] = r2;
 824     }
 825   }
 826   rp -= shift;
 827   for (int j = 0; j <= rp; j++) {
 828     SwitchRange& r = ranges[j];
 829     if (r.cnt() == 0 && r.dest() != never_reached) {
 830       r.setRange(r.lo(), r.hi(), never_reached, r.table_index(), r.cnt());
 831     }
 832   }
 833 }

 835 //-------------------------------do_tableswitch--------------------------------
 836 void Parse::do_tableswitch() {
 837   Node* lookup = pop();
 838   // Get information about tableswitch
 839   int default_dest = iter().get_dest_table(0);
 840   int lo_index     = iter().get_int_table(1);
 841   int hi_index     = iter().get_int_table(2);
 842   int len          = hi_index - lo_index + 1;
 843 
 844   if (len < 1) {
 845     // If this is a backward branch, add safepoint
 846     maybe_add_safepoint(default_dest);
 847     merge(default_dest);
 848     return;
 849   }
 850 
 851   ciMethodData* methodData = method()->method_data();
 852   ciMultiBranchData* profile = NULL;
 853   if (methodData->is_mature() && UseSwitchProfiling) {
 854     ciProfileData* data = methodData->bci_to_data(bci());
 855     if (data != NULL && data->is_MultiBranchData()) {
 856       profile = (ciMultiBranchData*)data;
 857     }
 858   }
 859   bool trim_ranges = !method_data_update() && !C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if);
 860 
 861   // generate decision tree, using trichotomy when possible
 862   int rnum = len+2;
 863   bool makes_backward_branch = false;
 864   SwitchRange* ranges = NEW_RESOURCE_ARRAY(SwitchRange, rnum);
 865   int rp = -1;
 866   if (lo_index != min_jint) {
 867     uint cnt = 1;
 868     if (profile != NULL) {
 869       cnt = profile->default_count() / (hi_index != max_jint ? 2 : 1);
 870     }
 871     ranges[++rp].setRange(min_jint, lo_index-1, default_dest, NullTableIndex, cnt);
 872   }
 873   for (int j = 0; j < len; j++) {
 874     jint match_int = lo_index+j;
 875     int  dest      = iter().get_dest_table(j+3);
 876     makes_backward_branch |= (dest <= bci());
 877     int  table_index = method_data_update() ? j : NullTableIndex;
 878     uint cnt = 1;
 879     if (profile != NULL) {
 880       cnt = profile->count_at(j);
 881     }
 882     if (rp < 0 || !ranges[rp].adjoin(match_int, dest, table_index, cnt, trim_ranges)) {
 883       ranges[++rp].set(match_int, dest, table_index, cnt);
 884     }
 885   }
 886   jint highest = lo_index+(len-1);
 887   assert(ranges[rp].hi() == highest, "");
 888   if (highest != max_jint) {
 889     uint cnt = 1;
 890     if (profile != NULL) {
 891       cnt = profile->default_count() / (lo_index != min_jint ? 2 : 1);
 892     }
 893     if (!ranges[rp].adjoinRange(highest+1, max_jint, default_dest, NullTableIndex, cnt, trim_ranges)) {
 894       ranges[++rp].setRange(highest+1, max_jint, default_dest, NullTableIndex, cnt);
 895     }
 896   }
 897   assert(rp < len+2, "not too many ranges");
 898 
 899   if (trim_ranges) {
 900     merge_ranges(ranges, rp);
 901   }
 902 
 903   // Safepoint in case if backward branch observed
 904   if( makes_backward_branch && UseLoopSafepoints )
 905     add_safepoint();
 906 
 907   jump_switch_ranges(lookup, &ranges[0], &ranges[rp]);
 908 }

 911 //------------------------------do_lookupswitch--------------------------------
 912 void Parse::do_lookupswitch() {
 913   Node *lookup = pop();         // lookup value
 914   // Get information about lookupswitch
 915   int default_dest = iter().get_dest_table(0);
 916   int len          = iter().get_int_table(1);
 917 
 918   if (len < 1) {    // If this is a backward branch, add safepoint
 919     maybe_add_safepoint(default_dest);
 920     merge(default_dest);
 921     return;
 922   }
 923 
 924   ciMethodData* methodData = method()->method_data();
 925   ciMultiBranchData* profile = NULL;
 926   if (methodData->is_mature() && UseSwitchProfiling) {
 927     ciProfileData* data = methodData->bci_to_data(bci());
 928     if (data != NULL && data->is_MultiBranchData()) {
 929       profile = (ciMultiBranchData*)data;
 930     }
 931   }
 932   bool trim_ranges = !method_data_update() && !C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if);
 933 
 934   // generate decision tree, using trichotomy when possible
 935   jint* table = NEW_RESOURCE_ARRAY(jint, len*3);
 936   {
 937     for (int j = 0; j < len; j++) {
 938       table[3*j+0] = iter().get_int_table(2+2*j);
 939       table[3*j+1] = iter().get_dest_table(2+2*j+1);
 940       table[3*j+2] = profile == NULL ? 1 : profile->count_at(j);
 941     }
 942     qsort(table, len, 3*sizeof(table[0]), jint_cmp);
 943   }
 944 
 945   float defaults = 0;
 946   jint prev = min_jint;
 947   for (int j = 0; j < len; j++) {
 948     jint match_int = table[3*j+0];
 949     if (match_int != prev) {
 950       defaults += (float)match_int - prev;
 951     }
 952     prev = match_int+1;
 953   }
 954   if (prev-1 != max_jint) {
 955     defaults += (float)max_jint - prev + 1;
 956   }
 957   float default_cnt = 1;
 958   if (profile != NULL) {
 959     default_cnt = profile->default_count()/defaults;
 960   }
 961 
 962   int rnum = len*2+1;
 963   bool makes_backward_branch = false;
 964   SwitchRange* ranges = NEW_RESOURCE_ARRAY(SwitchRange, rnum);
 965   int rp = -1;
 966   for (int j = 0; j < len; j++) {
 967     jint match_int   = table[3*j+0];
 968     int  dest        = table[3*j+1];
 969     int  cnt         = table[3*j+2];
 970     int  next_lo     = rp < 0 ? min_jint : ranges[rp].hi()+1;
 971     int  table_index = method_data_update() ? j : NullTableIndex;
 972     makes_backward_branch |= (dest <= bci());
 973     float c = default_cnt * ((float)match_int - next_lo);
 974     if (match_int != next_lo && (rp < 0 || !ranges[rp].adjoinRange(next_lo, match_int-1, default_dest, NullTableIndex, c, trim_ranges))) {
 975       assert(default_dest != never_reached, "sentinel value for dead destinations");
 976       ranges[++rp].setRange(next_lo, match_int-1, default_dest, NullTableIndex, c);
 977     }
 978     if (rp < 0 || !ranges[rp].adjoin(match_int, dest, table_index, cnt, trim_ranges)) {
 979       assert(dest != never_reached, "sentinel value for dead destinations");
 980       ranges[++rp].set(match_int, dest, table_index, cnt);
 981     }
 982   }
 983   jint highest = table[3*(len-1)];
 984   assert(ranges[rp].hi() == highest, "");
 985   if (highest != max_jint &&
 986       !ranges[rp].adjoinRange(highest+1, max_jint, default_dest, NullTableIndex, default_cnt * ((float)max_jint - highest), trim_ranges)) {
 987     ranges[++rp].setRange(highest+1, max_jint, default_dest, NullTableIndex, default_cnt * ((float)max_jint - highest));
 988   }
 989   assert(rp < rnum, "not too many ranges");
 990 
 991   if (trim_ranges) {
 992     merge_ranges(ranges, rp);
 993   }
 994 
 995   // Safepoint in case backward branch observed
 996   if (makes_backward_branch && UseLoopSafepoints)
 997     add_safepoint();
 998 
 999   jump_switch_ranges(lookup, &ranges[0], &ranges[rp]);
1000 }

1051 static float compute_tree_cost(SwitchRange *lo, SwitchRange *hi, float total_cnt) {
1052   GrowableArray<SwitchRanges> tree;
1053   SwitchRanges root(lo, hi);
1054   tree.push(root);
1055 
1056   float cost = 0;
1057   do {
1058     SwitchRanges& r = *tree.adr_at(tree.length()-1);
1059     if (r._hi != r._lo) {
1060       if (r._mid == NULL) {
1061         float r_cnt = sum_of_cnts(r._lo, r._hi);
1062 
1063         if (r_cnt == 0) {
1064           tree.pop();
1065           cost = 0;
1066           continue;
1067         }
1068 
1069         SwitchRange* mid = NULL;
1070         mid = r._lo;
1071         for (float cnt = 0; ; ) {
1072           assert(mid <= r._hi, "out of bounds");
1073           cnt += mid->cnt();
1074           if (cnt > r_cnt / 2) {
1075             break;
1076           }
1077           mid++;
1078         }
1079         assert(mid <= r._hi, "out of bounds");
1080         r._mid = mid;
1081         r._cost = r_cnt / total_cnt;
1082       }
1083       r._cost += cost;
1084       if (r._state < SwitchRanges::LeftDone && r._mid > r._lo) {
1085         cost = 0;
1086         r._state = SwitchRanges::LeftDone;
1087         tree.push(SwitchRanges(r._lo, r._mid-1));
1088       } else if (r._state < SwitchRanges::RightDone) {
1089         cost = 0;
1090         r._state = SwitchRanges::RightDone;
1091         tree.push(SwitchRanges(r._mid == r._lo ? r._mid+1 : r._mid, r._hi));
1092       } else {
1093         tree.pop();
1094         cost = r._cost;
1095       }
1096     } else {
1097       tree.pop();
1098       cost = r._cost;
1099     }
1100   } while (tree.length() > 0);
1101 
1102 
1103   return cost;
1104 }

1106 // It sometimes pays off to test most common ranges before the binary search
1107 void Parse::linear_search_switch_ranges(Node* key_val, SwitchRange*& lo, SwitchRange*& hi) {
1108   uint nr = hi - lo + 1;
1109   float total_cnt = sum_of_cnts(lo, hi);
1110 
1111   float min = compute_tree_cost(lo, hi, total_cnt);
1112   float extra = 1;
1113   float sub = 0;
1114 
1115   SwitchRange* array1 = lo;
1116   SwitchRange* array2 = NEW_RESOURCE_ARRAY(SwitchRange, nr);
1117 
1118   SwitchRange* ranges = NULL;
1119 
1120   while (nr >= 2) {
1121     assert(lo == array1 || lo == array2, "one the 2 already allocated arrays");
1122     ranges = (lo == array1) ? array2 : array1;
1123 
1124     // Find highest frequency range
1125     SwitchRange* candidate = lo;
1126     for (SwitchRange* sr = lo+1; sr <= hi; sr++) {
1127       if (sr->cnt() > candidate->cnt()) {
1128         candidate = sr;
1129       }
1130     }
1131     SwitchRange most_freq = *candidate;
1132     if (most_freq.cnt() == 0) {
1133       break;
1134     }
1135 
1136     // Copy remaining ranges into another array
1137     int shift = 0;
1138     for (uint i = 0; i < nr; i++) {
1139       SwitchRange* sr = &lo[i];
1140       if (sr != candidate) {
1141         ranges[i-shift] = *sr;
1142       } else {
1143         shift++;
1144         if (i > 0 && i < nr-1) {
1145           SwitchRange prev = lo[i-1];
1146           prev.setRange(prev.lo(), sr->hi(), prev.dest(), prev.table_index(), prev.cnt());
1147           if (prev.adjoin(lo[i+1])) {
1148             shift++;
1149             i++;
1150           }
1151           ranges[i-shift] = prev;
1152         }
1153       }
1154     }
1155     nr -= shift;
1156 
1157     // Evaluate cost of testing the most common range and performing a
1158     // binary search on the other ranges
1159     float cost = extra + compute_tree_cost(&ranges[0], &ranges[nr-1], total_cnt);
1160     if (cost >= min) {
1161       break;
1162     }
1163     // swap arrays
1164     lo = &ranges[0];
1165     hi = &ranges[nr-1];
1166 
1167     // It pays off: emit the test for the most common range
1168     assert(most_freq.cnt() > 0, "must be taken");
1169     Node* val = _gvn.transform(new SubINode(key_val, _gvn.intcon(most_freq.lo())));
1170     Node* cmp = _gvn.transform(new CmpUNode(val, _gvn.intcon(most_freq.hi() - most_freq.lo())));
1171     Node* tst = _gvn.transform(new BoolNode(cmp, BoolTest::le));
1172     IfNode* iff = create_and_map_if(control(), tst, if_prob(most_freq.cnt(), total_cnt), if_cnt(most_freq.cnt()));
1173     jump_if_true_fork(iff, most_freq.dest(), most_freq.table_index(), false);
1174 
1175     sub += most_freq.cnt() / total_cnt;
1176     extra += 1 - sub;
1177     min = cost;
1178   }
1179 }

1181 //----------------------------create_jump_tables-------------------------------
1182 bool Parse::create_jump_tables(Node* key_val, SwitchRange* lo, SwitchRange* hi) {
1183   // Are jumptables enabled
1184   if (!UseJumpTables)  return false;
1185 
1186   // Are jumptables supported
1187   if (!Matcher::has_match_rule(Op_Jump))  return false;
1188 
1189   // Don't make jump table if profiling
1190   if (method_data_update())  return false;
1191 
1192   bool trim_ranges = !C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if);
1193 
1194   // Decide if a guard is needed to lop off big ranges at either (or
1195   // both) end(s) of the input set. We'll call this the default target
1196   // even though we can't be sure that it is the true "default".
1197 
1198   bool needs_guard = false;
1199   int default_dest;
1200   int64_t total_outlier_size = 0;
1201   int64_t hi_size = ((int64_t)hi->hi()) - ((int64_t)hi->lo()) + 1;
1202   int64_t lo_size = ((int64_t)lo->hi()) - ((int64_t)lo->lo()) + 1;
1203 
1204   if (lo->dest() == hi->dest()) {
1205     total_outlier_size = hi_size + lo_size;
1206     default_dest = lo->dest();
1207   } else if (lo_size > hi_size) {
1208     total_outlier_size = lo_size;
1209     default_dest = lo->dest();
1210   } else {
1211     total_outlier_size = hi_size;
1212     default_dest = hi->dest();
1213   }
1214 
1215   float total = sum_of_cnts(lo, hi);
1216   float cost = compute_tree_cost(lo, hi, total);
1217 
1218   // If a guard test will eliminate very sparse end ranges, then
1219   // it is worth the cost of an extra jump.
1220   float trimmed_cnt = 0;
1221   if (total_outlier_size > (MaxJumpTableSparseness * 4)) {
1222     needs_guard = true;
1223     if (default_dest == lo->dest()) {
1224       trimmed_cnt += lo->cnt();
1225       lo++;
1226     }
1227     if (default_dest == hi->dest()) {
1228       trimmed_cnt += hi->cnt();
1229       hi--;
1230     }
1231   }
1232 
1233   // Find the total number of cases and ranges
1234   int64_t num_cases = ((int64_t)hi->hi()) - ((int64_t)lo->lo()) + 1;
1235   int num_range = hi - lo + 1;
1236 
1237   // Don't create table if: too large, too small, or too sparse.
1238   if (num_cases > MaxJumpTableSize)
1239     return false;
1240   if (UseSwitchProfiling) {
1241     // MinJumpTableSize is set so with a well balanced binary tree,
1242     // when the number of ranges is MinJumpTableSize, it's cheaper to
1243     // go through a JumpNode that a tree of IfNodes. Average cost of a
1244     // tree of IfNodes with MinJumpTableSize is
1245     // log2f(MinJumpTableSize) comparisons. So if the cost computed
1246     // from profile data is less than log2f(MinJumpTableSize) then
1247     // going with the binary search is cheaper.
1248     if (cost < log2f(MinJumpTableSize)) {
1249       return false;
1250     }
1251   } else {
1252     if (num_cases < MinJumpTableSize)
1253       return false;
1254   }
1255   if (num_cases > (MaxJumpTableSparseness * num_range))
1256     return false;
1257 
1258   // Normalize table lookups to zero
1259   int lowval = lo->lo();
1260   key_val = _gvn.transform( new SubINode(key_val, _gvn.intcon(lowval)) );
1261 
1262   // Generate a guard to protect against input keyvals that aren't
1263   // in the switch domain.
1264   if (needs_guard) {
1265     Node*   size = _gvn.intcon(num_cases);
1266     Node*   cmp = _gvn.transform(new CmpUNode(key_val, size));
1267     Node*   tst = _gvn.transform(new BoolNode(cmp, BoolTest::ge));
1268     IfNode* iff = create_and_map_if(control(), tst, if_prob(trimmed_cnt, total), if_cnt(trimmed_cnt));
1269     jump_if_true_fork(iff, default_dest, NullTableIndex, trim_ranges && trimmed_cnt == 0);
1270 
1271     total -= trimmed_cnt;
1272   }
1273 
1274   // Create an ideal node JumpTable that has projections
1275   // of all possible ranges for a switch statement
1276   // The key_val input must be converted to a pointer offset and scaled.
1277   // Compare Parse::array_addressing above.
1278 
1279   // Clean the 32-bit int into a real 64-bit offset.
1280   // Otherwise, the jint value 0 might turn into an offset of 0x0800000000.
1281   const TypeInt* ikeytype = TypeInt::make(0, num_cases, Type::WidenMin);
1282   // Make I2L conversion control dependent to prevent it from
1283   // floating above the range check during loop optimizations.
1284   key_val = C->conv_I2X_index(&_gvn, key_val, ikeytype, control());
1285 
1286   // Shift the value by wordsize so we have an index into the table, rather
1287   // than a switch value
1288   Node *shiftWord = _gvn.MakeConX(wordSize);
1289   key_val = _gvn.transform( new MulXNode( key_val, shiftWord));
1290 
1291   // Create the JumpNode
1292   Arena* arena = C->comp_arena();
1293   float* probs = (float*)arena->Amalloc(sizeof(float)*num_cases);
1294   int i = 0;
1295   if (total == 0) {
1296     for (SwitchRange* r = lo; r <= hi; r++) {
1297       for (int64_t j = r->lo(); j <= r->hi(); j++, i++) {
1298         probs[i] = 1.0F / num_cases;
1299       }
1300     }
1301   } else {
1302     for (SwitchRange* r = lo; r <= hi; r++) {
1303       float prob = r->cnt()/total;
1304       for (int64_t j = r->lo(); j <= r->hi(); j++, i++) {
1305         probs[i] = prob / (r->hi() - r->lo() + 1);
1306       }
1307     }
1308   }
1309 
1310   ciMethodData* methodData = method()->method_data();
1311   ciMultiBranchData* profile = NULL;
1312   if (methodData->is_mature()) {
1313     ciProfileData* data = methodData->bci_to_data(bci());
1314     if (data != NULL && data->is_MultiBranchData()) {
1315       profile = (ciMultiBranchData*)data;
1316     }
1317   }
1318 
1319   Node* jtn = _gvn.transform(new JumpNode(control(), key_val, num_cases, probs, profile == NULL ? COUNT_UNKNOWN : total));
1320 
1321   // These are the switch destinations hanging off the jumpnode
1322   i = 0;
1323   for (SwitchRange* r = lo; r <= hi; r++) {
1324     for (int64_t j = r->lo(); j <= r->hi(); j++, i++) {
1325       Node* input = _gvn.transform(new JumpProjNode(jtn, i, r->dest(), (int)(j - lowval)));
1326       {
1327         PreserveJVMState pjvms(this);
1328         set_control(input);
1329         jump_if_always_fork(r->dest(), r->table_index(), trim_ranges && r->cnt() == 0);
1330       }
1331     }
1332   }
1333   assert(i == num_cases, "miscount of cases");
1334   stop_and_kill_map();  // no more uses for this JVMS
1335   return true;
1336 }

1338 //----------------------------jump_switch_ranges-------------------------------
1339 void Parse::jump_switch_ranges(Node* key_val, SwitchRange *lo, SwitchRange *hi, int switch_depth) {
1340   Block* switch_block = block();
1341   bool trim_ranges = !method_data_update() && !C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if);
1342 
1343   if (switch_depth == 0) {
1344     // Do special processing for the top-level call.
1345     assert(lo->lo() == min_jint, "initial range must exhaust Type::INT");
1346     assert(hi->hi() == max_jint, "initial range must exhaust Type::INT");
1347 
1348     // Decrement pred-numbers for the unique set of nodes.
1349 #ifdef ASSERT
1350     if (!trim_ranges) {
1351       // Ensure that the block's successors are a (duplicate-free) set.
1352       int successors_counted = 0;  // block occurrences in [hi..lo]
1353       int unique_successors = switch_block->num_successors();
1354       for (int i = 0; i < unique_successors; i++) {
1355         Block* target = switch_block->successor_at(i);
1356 
1357         // Check that the set of successors is the same in both places.
1358         int successors_found = 0;
1359         for (SwitchRange* p = lo; p <= hi; p++) {
1360           if (p->dest() == target->start())  successors_found++;
1361         }
1362         assert(successors_found > 0, "successor must be known");
1363         successors_counted += successors_found;
1364       }
1365       assert(successors_counted == (hi-lo)+1, "no unexpected successors");
1366     }
1367 #endif
1368 
1369     // Maybe prune the inputs, based on the type of key_val.
1370     jint min_val = min_jint;
1371     jint max_val = max_jint;
1372     const TypeInt* ti = key_val->bottom_type()->isa_int();
1373     if (ti != NULL) {
1374       min_val = ti->_lo;
1375       max_val = ti->_hi;
1376       assert(min_val <= max_val, "invalid int type");
1377     }
1378     while (lo->hi() < min_val) {
1379       lo++;
1380     }
1381     if (lo->lo() < min_val)  {
1382       lo->setRange(min_val, lo->hi(), lo->dest(), lo->table_index(), lo->cnt());
1383     }
1384     while (hi->lo() > max_val) {
1385       hi--;
1386     }
1387     if (hi->hi() > max_val) {
1388       hi->setRange(hi->lo(), max_val, hi->dest(), hi->table_index(), hi->cnt());
1389     }
1390 
1391     linear_search_switch_ranges(key_val, lo, hi);
1392   }
1393 
1394 #ifndef PRODUCT
1395   if (switch_depth == 0) {
1396     _max_switch_depth = 0;
1397     _est_switch_depth = log2_intptr((hi-lo+1)-1)+1;
1398   }
1399 #endif
1400 
1401   assert(lo <= hi, "must be a non-empty set of ranges");
1402   if (lo == hi) {
1403     jump_if_always_fork(lo->dest(), lo->table_index(), trim_ranges && lo->cnt() == 0);
1404   } else {
1405     assert(lo->hi() == (lo+1)->lo()-1, "contiguous ranges");
1406     assert(hi->lo() == (hi-1)->hi()+1, "contiguous ranges");
1407 
1408     if (create_jump_tables(key_val, lo, hi)) return;
1409 
1410     SwitchRange* mid = NULL;
1411     float total_cnt = sum_of_cnts(lo, hi);
1412 
1413     int nr = hi - lo + 1;
1414     if (UseSwitchProfiling) {
1415       // Don't keep the binary search tree balanced: pick up mid point
1416       // that split frequencies in half.
1417       float cnt = 0;
1418       for (SwitchRange* sr = lo; sr <= hi; sr++) {
1419         cnt += sr->cnt();
1420         if (cnt >= total_cnt / 2) {
1421           mid = sr;
1422           break;
1423         }
1424       }
1425     } else {
1426       mid = lo + nr/2;
1427 
1428       // if there is an easy choice, pivot at a singleton:
1429       if (nr > 3 && !mid->is_singleton() && (mid-1)->is_singleton())  mid--;
1430 
1431       assert(lo < mid && mid <= hi, "good pivot choice");
1432       assert(nr != 2 || mid == hi,   "should pick higher of 2");
1433       assert(nr != 3 || mid == hi-1, "should pick middle of 3");
1434     }
1435 
1436 
1437     Node *test_val = _gvn.intcon(mid == lo ? mid->hi() : mid->lo());
1438 
1439     if (mid->is_singleton()) {
1440       IfNode *iff_ne = jump_if_fork_int(key_val, test_val, BoolTest::ne, 1-if_prob(mid->cnt(), total_cnt), if_cnt(mid->cnt()));
1441       jump_if_false_fork(iff_ne, mid->dest(), mid->table_index(), trim_ranges && mid->cnt() == 0);
1442 
1443       // Special Case:  If there are exactly three ranges, and the high
1444       // and low range each go to the same place, omit the "gt" test,
1445       // since it will not discriminate anything.
1446       bool eq_test_only = (hi == lo+2 && hi->dest() == lo->dest() && mid == hi-1) || mid == lo;
1447 
1448       // if there is a higher range, test for it and process it:
1449       if (mid < hi && !eq_test_only) {
1450         // two comparisons of same values--should enable 1 test for 2 branches
1451         // Use BoolTest::lt instead of BoolTest::gt
1452         float cnt = sum_of_cnts(lo, mid-1);
1453         IfNode *iff_lt  = jump_if_fork_int(key_val, test_val, BoolTest::lt, if_prob(cnt, total_cnt), if_cnt(cnt));
1454         Node   *iftrue  = _gvn.transform( new IfTrueNode(iff_lt) );
1455         Node   *iffalse = _gvn.transform( new IfFalseNode(iff_lt) );
1456         { PreserveJVMState pjvms(this);
1457           set_control(iffalse);
1458           jump_switch_ranges(key_val, mid+1, hi, switch_depth+1);
1459         }
1460         set_control(iftrue);
1461       }
1462 
1463     } else {
1464       // mid is a range, not a singleton, so treat mid..hi as a unit
1465       float cnt = sum_of_cnts(mid == lo ? mid+1 : mid, hi);
1466       IfNode *iff_ge = jump_if_fork_int(key_val, test_val, mid == lo ? BoolTest::gt : BoolTest::ge, if_prob(cnt, total_cnt), if_cnt(cnt));
1467 
1468       // if there is a higher range, test for it and process it:
1469       if (mid == hi) {
1470         jump_if_true_fork(iff_ge, mid->dest(), mid->table_index(), trim_ranges && cnt == 0);
1471       } else {
1472         Node *iftrue  = _gvn.transform( new IfTrueNode(iff_ge) );
1473         Node *iffalse = _gvn.transform( new IfFalseNode(iff_ge) );
1474         { PreserveJVMState pjvms(this);
1475           set_control(iftrue);
1476           jump_switch_ranges(key_val, mid == lo ? mid+1 : mid, hi, switch_depth+1);
1477         }
1478         set_control(iffalse);
1479       }
1480     }
1481 
1482     // in any case, process the lower range
1483     if (mid == lo) {
1484       if (mid->is_singleton()) {
1485         jump_switch_ranges(key_val, lo+1, hi, switch_depth+1);
1486       } else {
1487         jump_if_always_fork(lo->dest(), lo->table_index(), trim_ranges && lo->cnt() == 0);
1488       }
1489     } else {
1490       jump_switch_ranges(key_val, lo, mid-1, switch_depth+1);
1491     }
1492   }
1493 
1494   // Decrease pred_count for each successor after all is done.
1495   if (switch_depth == 0) {
1496     int unique_successors = switch_block->num_successors();
1497     for (int i = 0; i < unique_successors; i++) {
1498       Block* target = switch_block->successor_at(i);
1499       // Throw away the pre-allocated path for each unique successor.
1500       target->next_path_num();
1501     }
1502   }
1503 
1504 #ifndef PRODUCT
1505   _max_switch_depth = MAX2(switch_depth, _max_switch_depth);
1506   if (TraceOptoParse && Verbose && WizardMode && switch_depth == 0) {
1507     SwitchRange* r;
1508     int nsing = 0;
1509     for( r = lo; r <= hi; r++ ) {
1510       if( r->is_singleton() )  nsing++;
1511     }
1512     tty->print(">>> ");
1513     _method->print_short_name();
1514     tty->print_cr(" switch decision tree");
1515     tty->print_cr("    %d ranges (%d singletons), max_depth=%d, est_depth=%d",
1516                   (int) (hi-lo+1), nsing, _max_switch_depth, _est_switch_depth);
1517     if (_max_switch_depth > _est_switch_depth) {
1518       tty->print_cr("******** BAD SWITCH DEPTH ********");
1519     }
1520     tty->print("   ");
1521     for( r = lo; r <= hi; r++ ) {
1522       r->print();
1523     }
1524     tty->cr();
1525   }
1526 #endif
1527 }

1570 void Parse::do_irem() {
1571   // Must keep both values on the expression-stack during null-check
1572   zero_check_int(peek());
1573   // Compile-time detect of null-exception?
1574   if (stopped())  return;
1575 
1576   Node* b = pop();
1577   Node* a = pop();
1578 
1579   const Type *t = _gvn.type(b);
1580   if (t != Type::TOP) {
1581     const TypeInt *ti = t->is_int();
1582     if (ti->is_con()) {
1583       int divisor = ti->get_con();
1584       // check for positive power of 2
1585       if (divisor > 0 &&
1586           (divisor & ~(divisor-1)) == divisor) {
1587         // yes !
1588         Node *mask = _gvn.intcon((divisor - 1));
1589         // Sigh, must handle negative dividends
1590         Node *zero = _gvn.intcon(0);
1591         IfNode *ifff = jump_if_fork_int(a, zero, BoolTest::lt, PROB_FAIR, COUNT_UNKNOWN);
1592         Node *iff = _gvn.transform( new IfFalseNode(ifff) );
1593         Node *ift = _gvn.transform( new IfTrueNode (ifff) );
1594         Node *reg = jump_if_join(ift, iff);
1595         Node *phi = PhiNode::make(reg, NULL, TypeInt::INT);
1596         // Negative path; negate/and/negate
1597         Node *neg = _gvn.transform( new SubINode(zero, a) );
1598         Node *andn= _gvn.transform( new AndINode(neg, mask) );
1599         Node *negn= _gvn.transform( new SubINode(zero, andn) );
1600         phi->init_req(1, negn);
1601         // Fast positive case
1602         Node *andx = _gvn.transform( new AndINode(a, mask) );
1603         phi->init_req(2, andx);
1604         // Push the merge
1605         push( _gvn.transform(phi) );
1606         return;
1607       }
1608     }
1609   }
1610   // Default case
1611   push( _gvn.transform( new ModINode(control(),a,b) ) );

1651 static bool has_injected_profile(BoolTest::mask btest, Node* test, int& taken, int& not_taken) {
1652   if (btest != BoolTest::eq && btest != BoolTest::ne) {
1653     // Only ::eq and ::ne are supported for profile injection.
1654     return false;
1655   }
1656   if (test->is_Cmp() &&
1657       test->in(1)->Opcode() == Op_ProfileBoolean) {
1658     ProfileBooleanNode* profile = (ProfileBooleanNode*)test->in(1);
1659     int false_cnt = profile->false_count();
1660     int  true_cnt = profile->true_count();
1661 
1662     // Counts matching depends on the actual test operation (::eq or ::ne).
1663     // No need to scale the counts because profile injection was designed
1664     // to feed exact counts into VM.
1665     taken     = (btest == BoolTest::eq) ? false_cnt :  true_cnt;
1666     not_taken = (btest == BoolTest::eq) ?  true_cnt : false_cnt;
1667 
1668     profile->consume();
1669     return true;
1670   }
1671   return false;
1672 }
1673 //--------------------------dynamic_branch_prediction--------------------------
1674 // Try to gather dynamic branch prediction behavior.  Return a probability
1675 // of the branch being taken and set the "cnt" field.  Returns a -1.0
1676 // if we need to use static prediction for some reason.
1677 float Parse::dynamic_branch_prediction(float &cnt, BoolTest::mask btest, Node* test) {
1678   ResourceMark rm;
1679 
1680   cnt  = COUNT_UNKNOWN;
1681 
1682   int     taken = 0;
1683   int not_taken = 0;
1684 
1685   bool use_mdo = !has_injected_profile(btest, test, taken, not_taken);
1686 
1687   if (use_mdo) {
1688     // Use MethodData information if it is available
1689     // FIXME: free the ProfileData structure
1690     ciMethodData* methodData = method()->method_data();
1691     if (!methodData->is_mature())  return PROB_UNKNOWN;
1692     ciProfileData* data = methodData->bci_to_data(bci());
1693     if (data == NULL) {
1694       return PROB_UNKNOWN;
1695     }
1696     if (!data->is_JumpData())  return PROB_UNKNOWN;
1697 
1698     // get taken and not taken values
1699     taken = data->as_JumpData()->taken();
1700     not_taken = 0;
1701     if (data->is_BranchData()) {
1702       not_taken = data->as_BranchData()->not_taken();
1703     }
1704 
1705     // scale the counts to be commensurate with invocation counts:
1706     taken = method()->scale_count(taken);
1707     not_taken = method()->scale_count(not_taken);
1708   }
1709 
1710   // Give up if too few (or too many, in which case the sum will overflow) counts to be meaningful.
1711   // We also check that individual counters are positive first, otherwise the sum can become positive.
1712   if (taken < 0 || not_taken < 0 || taken + not_taken < 40) {
1713     if (C->log() != NULL) {
1714       C->log()->elem("branch target_bci='%d' taken='%d' not_taken='%d'", iter().get_dest(), taken, not_taken);
1715     }
1716     return PROB_UNKNOWN;
1717   }
1718 
1719   // Compute frequency that we arrive here
1720   float sum = taken + not_taken;
1721   // Adjust, if this block is a cloned private block but the
1722   // Jump counts are shared.  Taken the private counts for
1723   // just this path instead of the shared counts.
1724   if( block()->count() > 0 )
1725     sum = block()->count();
1726   cnt = sum / FreqCountInvocations;
1727 
1728   // Pin probability to sane limits
1729   float prob;
1730   if( !taken )
1731     prob = (0+PROB_MIN) / 2;
1732   else if( !not_taken )
1733     prob = (1+PROB_MAX) / 2;
1734   else {                         // Compute probability of true path
1735     prob = (float)taken / (float)(taken + not_taken);
1736     if (prob > PROB_MAX)  prob = PROB_MAX;
1737     if (prob < PROB_MIN)   prob = PROB_MIN;
1738   }
1739 
1740   assert((cnt > 0.0f) && (prob > 0.0f),
1741          "Bad frequency assignment in if");
1742 
1743   if (C->log() != NULL) {
1744     const char* prob_str = NULL;
1745     if (prob >= PROB_MAX)  prob_str = (prob == PROB_MAX) ? "max" : "always";
1746     if (prob <= PROB_MIN)  prob_str = (prob == PROB_MIN) ? "min" : "never";
1747     char prob_str_buf[30];
1748     if (prob_str == NULL) {
1749       jio_snprintf(prob_str_buf, sizeof(prob_str_buf), "%20.2f", prob);
1750       prob_str = prob_str_buf;
1751     }
1752     C->log()->elem("branch target_bci='%d' taken='%d' not_taken='%d' cnt='%f' prob='%s'",
1753                    iter().get_dest(), taken, not_taken, cnt, prob_str);
1754   }
1755   return prob;
1756 }

1758 //-----------------------------branch_prediction-------------------------------
1759 float Parse::branch_prediction(float& cnt,
1760                                BoolTest::mask btest,
1761                                int target_bci,
1762                                Node* test) {
1763   float prob = dynamic_branch_prediction(cnt, btest, test);
1764   // If prob is unknown, switch to static prediction
1765   if (prob != PROB_UNKNOWN)  return prob;
1766 
1767   prob = PROB_FAIR;                   // Set default value
1768   if (btest == BoolTest::eq)          // Exactly equal test?
1769     prob = PROB_STATIC_INFREQUENT;    // Assume its relatively infrequent
1770   else if (btest == BoolTest::ne)
1771     prob = PROB_STATIC_FREQUENT;      // Assume its relatively frequent
1772 
1773   // If this is a conditional test guarding a backwards branch,
1774   // assume its a loop-back edge.  Make it a likely taken branch.
1775   if (target_bci < bci()) {
1776     if (is_osr_parse()) {    // Could be a hot OSR'd loop; force deopt
1777       // Since it's an OSR, we probably have profile data, but since
1778       // branch_prediction returned PROB_UNKNOWN, the counts are too small.
1779       // Let's make a special check here for completely zero counts.
1780       ciMethodData* methodData = method()->method_data();
1781       if (!methodData->is_empty()) {
1782         ciProfileData* data = methodData->bci_to_data(bci());
1783         // Only stop for truly zero counts, which mean an unknown part
1784         // of the OSR-ed method, and we want to deopt to gather more stats.
1785         // If you have ANY counts, then this loop is simply 'cold' relative
1786         // to the OSR loop.
1787         if (data == NULL ||
1788             (data->as_BranchData()->taken() +  data->as_BranchData()->not_taken() == 0)) {
1789           // This is the only way to return PROB_UNKNOWN:
1790           return PROB_UNKNOWN;
1791         }
1792       }
1793     }
1794     prob = PROB_STATIC_FREQUENT;     // Likely to take backwards branch
1795   }
1796 
1797   assert(prob != PROB_UNKNOWN, "must have some guess at this point");
1798   return prob;
1799 }

1801 // The magic constants are chosen so as to match the output of
1802 // branch_prediction() when the profile reports a zero taken count.
1803 // It is important to distinguish zero counts unambiguously, because
1804 // some branches (e.g., _213_javac.Assembler.eliminate) validly produce
1805 // very small but nonzero probabilities, which if confused with zero
1806 // counts would keep the program recompiling indefinitely.
1807 bool Parse::seems_never_taken(float prob) const {
1808   return prob < PROB_MIN;

1811 // True if the comparison seems to be the kind that will not change its
1812 // statistics from true to false.  See comments in adjust_map_after_if.
1813 // This question is only asked along paths which are already
1814 // classifed as untaken (by seems_never_taken), so really,
1815 // if a path is never taken, its controlling comparison is
1816 // already acting in a stable fashion.  If the comparison
1817 // seems stable, we will put an expensive uncommon trap
1818 // on the untaken path.
1819 bool Parse::seems_stable_comparison() const {
1820   if (C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if)) {
1821     return false;
1822   }
1823   return true;

1843 //----------------------------------do_ifnull----------------------------------
1844 void Parse::do_ifnull(BoolTest::mask btest, Node *c) {
1845   int target_bci = iter().get_dest();
1846 
1847   Block* branch_block = successor_for_bci(target_bci);
1848   Block* next_block   = successor_for_bci(iter().next_bci());
1849 
1850   float cnt;
1851   float prob = branch_prediction(cnt, btest, target_bci, c);
1852   if (prob == PROB_UNKNOWN) {
1853     // (An earlier version of do_ifnull omitted this trap for OSR methods.)
1854     if (PrintOpto && Verbose) {
1855       tty->print_cr("Never-taken edge stops compilation at bci %d", bci());
1856     }
1857     repush_if_args(); // to gather stats on loop
1858     // We need to mark this branch as taken so that if we recompile we will
1859     // see that it is possible. In the tiered system the interpreter doesn't
1860     // do profiling and by the time we get to the lower tier from the interpreter
1861     // the path may be cold again. Make sure it doesn't look untaken
1862     profile_taken_branch(target_bci, !ProfileInterpreter);
1863     uncommon_trap(Deoptimization::Reason_unreached,
1864                   Deoptimization::Action_reinterpret,
1865                   NULL, "cold");
1866     if (C->eliminate_boxing()) {
1867       // Mark the successor blocks as parsed
1868       branch_block->next_path_num();
1869       next_block->next_path_num();
1870     }
1871     return;
1872   }
1873 
1874   NOT_PRODUCT(explicit_null_checks_inserted++);
1875 
1876   // Generate real control flow
1877   Node   *tst = _gvn.transform( new BoolNode( c, btest ) );
1878 
1879   // Sanity check the probability value
1880   assert(prob > 0.0f,"Bad probability in Parser");
1881  // Need xform to put node in hash table
1882   IfNode *iff = create_and_xform_if( control(), tst, prob, cnt );
1883   assert(iff->_prob > 0.0f,"Optimizer made bad probability in parser");
1884   // True branch
1885   { PreserveJVMState pjvms(this);
1886     Node* iftrue  = _gvn.transform( new IfTrueNode (iff) );
1887     set_control(iftrue);
1888 
1889     if (stopped()) {            // Path is dead?
1890       NOT_PRODUCT(explicit_null_checks_elided++);
1891       if (C->eliminate_boxing()) {
1892         // Mark the successor block as parsed
1893         branch_block->next_path_num();
1894       }
1895     } else {                    // Path is live.
1896       // Update method data
1897       profile_taken_branch(target_bci);
1898       adjust_map_after_if(btest, c, prob, branch_block);
1899       if (!stopped()) {
1900         merge(target_bci);
1901       }
1902     }
1903   }
1904 
1905   // False branch
1906   Node* iffalse = _gvn.transform( new IfFalseNode(iff) );
1907   set_control(iffalse);
1908 
1909   if (stopped()) {              // Path is dead?
1910     NOT_PRODUCT(explicit_null_checks_elided++);
1911     if (C->eliminate_boxing()) {
1912       // Mark the successor block as parsed
1913       next_block->next_path_num();
1914     }
1915   } else  {                     // Path is live.
1916     // Update method data
1917     profile_not_taken_branch();
1918     adjust_map_after_if(BoolTest(btest).negate(), c, 1.0-prob, next_block);
1919   }
1920 }

1922 //------------------------------------do_if------------------------------------
1923 void Parse::do_if(BoolTest::mask btest, Node* c, bool new_path, Node** ctrl_taken) {
1924   int target_bci = iter().get_dest();
1925 
1926   Block* branch_block = successor_for_bci(target_bci);
1927   Block* next_block   = successor_for_bci(iter().next_bci());
1928 
1929   float cnt;
1930   float prob = branch_prediction(cnt, btest, target_bci, c);
1931   float untaken_prob = 1.0 - prob;
1932 
1933   if (prob == PROB_UNKNOWN) {
1934     if (PrintOpto && Verbose) {
1935       tty->print_cr("Never-taken edge stops compilation at bci %d", bci());
1936     }
1937     repush_if_args(); // to gather stats on loop
1938     // We need to mark this branch as taken so that if we recompile we will
1939     // see that it is possible. In the tiered system the interpreter doesn't
1940     // do profiling and by the time we get to the lower tier from the interpreter
1941     // the path may be cold again. Make sure it doesn't look untaken
1942     profile_taken_branch(target_bci, !ProfileInterpreter);
1943     uncommon_trap(Deoptimization::Reason_unreached,
1944                   Deoptimization::Action_reinterpret,
1945                   NULL, "cold");
1946     if (C->eliminate_boxing()) {
1947       // Mark the successor blocks as parsed
1948       branch_block->next_path_num();
1949       next_block->next_path_num();
1950     }
1951     return;
1952   }
1953 
1954   // Sanity check the probability value
1955   assert(0.0f < prob && prob < 1.0f,"Bad probability in Parser");
1956 
1957   bool taken_if_true = true;
1958   // Convert BoolTest to canonical form:
1959   if (!BoolTest(btest).is_canonical()) {
1960     btest         = BoolTest(btest).negate();
1961     taken_if_true = false;
1962     // prob is NOT updated here; it remains the probability of the taken
1963     // path (as opposed to the prob of the path guarded by an 'IfTrueNode').
1964   }
1965   assert(btest != BoolTest::eq, "!= is the only canonical exact test");
1966 
1967   Node* tst0 = new BoolNode(c, btest);
1968   Node* tst = _gvn.transform(tst0);
1969   BoolTest::mask taken_btest   = BoolTest::illegal;
1970   BoolTest::mask untaken_btest = BoolTest::illegal;
1971 
1972   if (tst->is_Bool()) {
1973     // Refresh c from the transformed bool node, since it may be
1974     // simpler than the original c.  Also re-canonicalize btest.
1975     // This wins when (Bool ne (Conv2B p) 0) => (Bool ne (CmpP p NULL)).
1976     // That can arise from statements like: if (x instanceof C) ...
1977     if (tst != tst0) {
1978       // Canonicalize one more time since transform can change it.
1979       btest = tst->as_Bool()->_test._test;
1980       if (!BoolTest(btest).is_canonical()) {
1981         // Reverse edges one more time...
1982         tst   = _gvn.transform( tst->as_Bool()->negate(&_gvn) );
1983         btest = tst->as_Bool()->_test._test;
1984         assert(BoolTest(btest).is_canonical(), "sanity");
1985         taken_if_true = !taken_if_true;
1986       }
1987       c = tst->in(1);
1988     }
1989     BoolTest::mask neg_btest = BoolTest(btest).negate();
1990     taken_btest   = taken_if_true ?     btest : neg_btest;
1991     untaken_btest = taken_if_true ? neg_btest :     btest;
1992   }
1993 
1994   // Generate real control flow
1995   float true_prob = (taken_if_true ? prob : untaken_prob);
1996   IfNode* iff = create_and_map_if(control(), tst, true_prob, cnt);
1997   assert(iff->_prob > 0.0f,"Optimizer made bad probability in parser");
1998   Node* taken_branch   = new IfTrueNode(iff);
1999   Node* untaken_branch = new IfFalseNode(iff);
2000   if (!taken_if_true) {  // Finish conversion to canonical form
2001     Node* tmp      = taken_branch;
2002     taken_branch   = untaken_branch;
2003     untaken_branch = tmp;
2004   }
2005 
2006   // Branch is taken:
2007   { PreserveJVMState pjvms(this);
2008     taken_branch = _gvn.transform(taken_branch);
2009     set_control(taken_branch);
2010 
2011     if (stopped()) {
2012       if (C->eliminate_boxing() && !new_path) {
2013         // Mark the successor block as parsed (if we haven't created a new path)
2014         branch_block->next_path_num();
2015       }
2016     } else {
2017       // Update method data
2018       profile_taken_branch(target_bci);
2019       adjust_map_after_if(taken_btest, c, prob, branch_block);
2020       if (!stopped()) {
2021         if (new_path) {
2022           // Merge by using a new path
2023           merge_new_path(target_bci);
2024         } else if (ctrl_taken != NULL) {
2025           // Don't merge but save taken branch to be wired by caller
2026           *ctrl_taken = control();
2027         } else {
2028           merge(target_bci);
2029         }
2030       }
2031     }
2032   }
2033 
2034   untaken_branch = _gvn.transform(untaken_branch);
2035   set_control(untaken_branch);
2036 
2037   // Branch not taken.
2038   if (stopped() && ctrl_taken == NULL) {
2039     if (C->eliminate_boxing()) {
2040       // Mark the successor block as parsed (if caller does not re-wire control flow)
2041       next_block->next_path_num();
2042     }
2043   } else {
2044     // Update method data
2045     profile_not_taken_branch();
2046     adjust_map_after_if(untaken_btest, c, untaken_prob, next_block);
2047   }
2048 }

2050 void Parse::do_acmp(BoolTest::mask btest, Node* a, Node* b) {
2051   ciMethod* subst_method = ciEnv::current()->ValueBootstrapMethods_klass()->find_method(ciSymbol::isSubstitutable_name(), ciSymbol::object_object_boolean_signature());
2052   // If current method is ValueBootstrapMethods::isSubstitutable(),
2053   // compile the acmp as a regular pointer comparison otherwise we
2054   // could call ValueBootstrapMethods::isSubstitutable() back
2055   if (!EnableValhalla || (method() == subst_method)) {
2056     Node* cmp = CmpP(a, b);
2057     cmp = optimize_cmp_with_klass(cmp);
2058     do_if(btest, cmp);
2059     return;
2060   }
2061 
2062   // Allocate value type operands and re-execute on deoptimization
2063   if (a->is_ValueType()) {
2064     PreserveReexecuteState preexecs(this);
2065     inc_sp(2);
2066     jvms()->set_should_reexecute(true);
2067     a = a->as_ValueType()->buffer(this)->get_oop();
2068   }
2069   if (b->is_ValueType()) {
2070     PreserveReexecuteState preexecs(this);
2071     inc_sp(2);
2072     jvms()->set_should_reexecute(true);
2073     b = b->as_ValueType()->buffer(this)->get_oop();
2074   }
2075 
2076   // First, do a normal pointer comparison
2077   const TypeOopPtr* ta = _gvn.type(a)->isa_oopptr();
2078   const TypeOopPtr* tb = _gvn.type(b)->isa_oopptr();
2079   Node* cmp = CmpP(a, b);
2080   cmp = optimize_cmp_with_klass(cmp);
2081   if (ta == NULL || !ta->can_be_value_type() ||
2082       tb == NULL || !tb->can_be_value_type()) {
2083     // This is sufficient, if one of the operands can't be a value type
2084     do_if(btest, cmp);
2085     return;
2086   }
2087   Node* eq_region = NULL;
2088   if (btest == BoolTest::eq) {
2089     do_if(btest, cmp, true);
2090     if (stopped()) {
2091       return;
2092     }
2093   } else {
2094     assert(btest == BoolTest::ne, "only eq or ne");
2095     Node* is_not_equal = NULL;
2096     eq_region = new RegionNode(3);
2097     {
2098       PreserveJVMState pjvms(this);
2099       do_if(btest, cmp, false, &is_not_equal);
2100       if (!stopped()) {
2101         eq_region->init_req(1, control());
2102       }
2103     }
2104     if (is_not_equal == NULL || is_not_equal->is_top()) {
2105       record_for_igvn(eq_region);
2106       set_control(_gvn.transform(eq_region));
2107       return;
2108     }
2109     set_control(is_not_equal);
2110   }
2111 
2112   // Pointers are not equal, check if first operand is non-null
2113   Node* ne_region = new RegionNode(6);
2114   inc_sp(2);
2115   Node* null_ctl = top();
2116   Node* not_null_a = null_check_oop(a, &null_ctl, !too_many_traps(Deoptimization::Reason_null_check), false, false);
2117   dec_sp(2);
2118   ne_region->init_req(1, null_ctl);
2119   if (stopped()) {
2120     record_for_igvn(ne_region);
2121     set_control(_gvn.transform(ne_region));
2122     if (btest == BoolTest::ne) {
2123       {
2124         PreserveJVMState pjvms(this);
2125         int target_bci = iter().get_dest();
2126         merge(target_bci);
2127       }
2128       record_for_igvn(eq_region);
2129       set_control(_gvn.transform(eq_region));
2130     }
2131     return;
2132   }
2133 
2134   // First operand is non-null, check if it is a value type
2135   Node* is_value = is_value_type(not_null_a);
2136   IfNode* is_value_iff = create_and_map_if(control(), is_value, PROB_FAIR, COUNT_UNKNOWN);
2137   Node* not_value = _gvn.transform(new IfFalseNode(is_value_iff));
2138   ne_region->init_req(2, not_value);
2139   set_control(_gvn.transform(new IfTrueNode(is_value_iff)));
2140 
2141   // The first operand is a value type, check if the second operand is non-null
2142   inc_sp(2);
2143   null_ctl = top();
2144   Node* not_null_b = null_check_oop(b, &null_ctl, !too_many_traps(Deoptimization::Reason_null_check), false, false);

2357 void Parse::sharpen_type_after_if(BoolTest::mask btest,
2358                                   Node* con, const Type* tcon,
2359                                   Node* val, const Type* tval) {
2360   // Look for opportunities to sharpen the type of a node
2361   // whose klass is compared with a constant klass.
2362   if (btest == BoolTest::eq && tcon->isa_klassptr()) {
2363     Node* obj = extract_obj_from_klass_load(&_gvn, val);
2364     const TypeOopPtr* con_type = tcon->isa_klassptr()->as_instance_type();
2365     if (obj != NULL && (con_type->isa_instptr() || con_type->isa_aryptr())) {
2366        // Found:
2367        //   Bool(CmpP(LoadKlass(obj._klass), ConP(Foo.klass)), [eq])
2368        // or the narrowOop equivalent.
2369        const Type* obj_type = _gvn.type(obj);
2370        const TypeOopPtr* tboth = obj_type->join_speculative(con_type)->isa_oopptr();
2371        if (tboth != NULL && tboth->klass_is_exact() && tboth != obj_type &&
2372            tboth->higher_equal(obj_type)) {
2373           // obj has to be of the exact type Foo if the CmpP succeeds.
2374           int obj_in_map = map()->find_edge(obj);
2375           JVMState* jvms = this->jvms();
2376           if (obj_in_map >= 0 &&
2377               (jvms->is_loc(obj_in_map) || jvms->is_stk(obj_in_map))) {
2378             TypeNode* ccast = new CheckCastPPNode(control(), obj, tboth);
2379             const Type* tcc = ccast->as_Type()->type();
2380             assert(tcc != obj_type && tcc->higher_equal(obj_type), "must improve");
2381             // Delay transform() call to allow recovery of pre-cast value
2382             // at the control merge.
2383             _gvn.set_type_bottom(ccast);
2384             record_for_igvn(ccast);
2385             // Here's the payoff.
2386             replace_in_map(obj, ccast);
2387           }
2388        }
2389     }
2390   }
2391 
2392   int val_in_map = map()->find_edge(val);
2393   if (val_in_map < 0)  return;          // replace_in_map would be useless
2394   {
2395     JVMState* jvms = this->jvms();
2396     if (!(jvms->is_loc(val_in_map) ||
2397           jvms->is_stk(val_in_map)))
2398       return;                           // again, it would be useless
2399   }
2400 
2401   // Check for a comparison to a constant, and "know" that the compared
2402   // value is constrained on this path.
2403   assert(tcon->singleton(), "");
2404   ConstraintCastNode* ccast = NULL;
2405   Node* cast = NULL;
2406 
2407   switch (btest) {
2408   case BoolTest::eq:                    // Constant test?
2409     {
2410       const Type* tboth = tcon->join_speculative(tval);
2411       if (tboth == tval)  break;        // Nothing to gain.
2412       if (tcon->isa_int()) {
2413         ccast = new CastIINode(val, tboth);
2414       } else if (tcon == TypePtr::NULL_PTR) {
2415         // Cast to null, but keep the pointer identity temporarily live.
2416         ccast = new CastPPNode(val, tboth);
2417       } else {
2418         const TypeF* tf = tcon->isa_float_constant();
2419         const TypeD* td = tcon->isa_double_constant();
2420         // Exclude tests vs float/double 0 as these could be
2421         // either +0 or -0.  Just because you are equal to +0
2422         // doesn't mean you ARE +0!
2423         // Note, following code also replaces Long and Oop values.
2424         if ((!tf || tf->_f != 0.0) &&
2425             (!td || td->_d != 0.0))
2426           cast = con;                   // Replace non-constant val by con.
2427       }
2428     }
2429     break;
2430 
2431   case BoolTest::ne:
2432     if (tcon == TypePtr::NULL_PTR) {
2433       cast = cast_not_null(val, false);
2434     }
2435     break;
2436 
2437   default:
2438     // (At this point we could record int range types with CastII.)
2439     break;
2440   }
2441 
2442   if (ccast != NULL) {
2443     const Type* tcc = ccast->as_Type()->type();
2444     assert(tcc != tval && tcc->higher_equal(tval), "must improve");
2445     // Delay transform() call to allow recovery of pre-cast value
2446     // at the control merge.
2447     ccast->set_req(0, control());
2448     _gvn.set_type_bottom(ccast);
2449     record_for_igvn(ccast);
2450     cast = ccast;
2451   }
2452 
2453   if (cast != NULL) {                   // Here's the payoff.
2454     replace_in_map(val, cast);
2455   }
