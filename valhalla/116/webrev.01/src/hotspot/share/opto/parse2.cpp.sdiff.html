 216     if (elemptr->value_klass()->is_scalarizable()) {
 217       ld = ValueTypeNode::make_from_oop(this, ld, elemptr->value_klass());
 218     }
 219   }
 220   if (!ld->is_ValueType()) {
 221     ld = record_profile_for_speculation_at_array_load(ld);
 222   }
 223
