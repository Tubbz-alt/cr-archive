  64 //
  65 // Replace any references to "oldref" in inputs to "use" with "newref".
  66 // Returns the number of replacements made.
  67 //
  68 int PhaseMacroExpand::replace_input(Node *use, Node *oldref, Node *newref) {
  69   int nreplacements = 0;
  70   uint req = use->req();
  71   for (uint j = 0; j < use->len(); j++) {
  72     Node *uin = use->in(j);
  73     if (uin == oldref) {
  74       if (j < req)
  75         use->set_req(j, newref);
  76       else
  77         use->set_prec(j, newref);
  78       nreplacements++;
  79     } else if (j >= req && uin == NULL) {
  80       break;
  81     }
  82   }
  83   return nreplacements;
  84 }

  86 Node* PhaseMacroExpand::opt_bits_test(Node* ctrl, Node* region, int edge, Node* word, int mask, int bits, bool return_fast_path) {
  87   Node* cmp;
  88   if (mask != 0) {
  89     Node* and_node = transform_later(new AndXNode(word, MakeConX(mask)));
  90     cmp = transform_later(new CmpXNode(and_node, MakeConX(bits)));
  91   } else {
  92     cmp = word;
  93   }
  94   Node* bol = transform_later(new BoolNode(cmp, BoolTest::ne));
  95   IfNode* iff = new IfNode( ctrl, bol, PROB_MIN, COUNT_UNKNOWN );
  96   transform_later(iff);
  97 
  98   // Fast path taken.
  99   Node *fast_taken = transform_later(new IfFalseNode(iff));
 100 
 101   // Fast path not-taken, i.e. slow path
 102   Node *slow_taken = transform_later(new IfTrueNode(iff));
 103 
 104   if (return_fast_path) {
 105     region->init_req(edge, slow_taken); // Capture slow-control
 106     return fast_taken;
 107   } else {
 108     region->init_req(edge, fast_taken); // Capture fast-control
 109     return slow_taken;
 110   }
 111 }

 123 //------------------------------make_slow_call---------------------------------
 124 CallNode* PhaseMacroExpand::make_slow_call(CallNode *oldcall, const TypeFunc* slow_call_type,
 125                                            address slow_call, const char* leaf_name, Node* slow_path,
 126                                            Node* parm0, Node* parm1, Node* parm2) {
 127 
 128   // Slow-path call
 129  CallNode *call = leaf_name
 130    ? (CallNode*)new CallLeafNode      ( slow_call_type, slow_call, leaf_name, TypeRawPtr::BOTTOM )
 131    : (CallNode*)new CallStaticJavaNode( slow_call_type, slow_call, OptoRuntime::stub_name(slow_call), oldcall->jvms()->bci(), TypeRawPtr::BOTTOM );
 132 
 133   // Slow path call has no side-effects, uses few values
 134   copy_predefined_input_for_runtime_call(slow_path, oldcall, call );
 135   if (parm0 != NULL)  call->init_req(TypeFunc::Parms+0, parm0);
 136   if (parm1 != NULL)  call->init_req(TypeFunc::Parms+1, parm1);
 137   if (parm2 != NULL)  call->init_req(TypeFunc::Parms+2, parm2);
 138   call->copy_call_debug_info(&_igvn, oldcall);
 139   call->set_cnt(PROB_UNLIKELY_MAG(4));  // Same effect as RC_UNCOMMON.
 140   _igvn.replace_node(oldcall, call);
 141   transform_later(call);
 142 
 143   return call;
 144 }

 146 void PhaseMacroExpand::extract_call_projections(CallNode *call) {
 147   _fallthroughproj = NULL;
 148   _fallthroughcatchproj = NULL;
 149   _ioproj_fallthrough = NULL;
 150   _ioproj_catchall = NULL;
 151   _catchallcatchproj = NULL;
 152   _memproj_fallthrough = NULL;
 153   _memproj_catchall = NULL;
 154   _resproj = NULL;
 155   for (DUIterator_Fast imax, i = call->fast_outs(imax); i < imax; i++) {
 156     ProjNode *pn = call->fast_out(i)->as_Proj();
 157     switch (pn->_con) {
 158       case TypeFunc::Control:
 159       {
 160         // For Control (fallthrough) and I_O (catch_all_index) we have CatchProj -> Catch -> Proj
 161         _fallthroughproj = pn;
 162         DUIterator_Fast jmax, j = pn->fast_outs(jmax);
 163         const Node *cn = pn->fast_out(j);
 164         if (cn->is_Catch()) {
 165           ProjNode *cpn = NULL;
 166           for (DUIterator_Fast kmax, k = cn->fast_outs(kmax); k < kmax; k++) {
 167             cpn = cn->fast_out(k)->as_Proj();
 168             assert(cpn->is_CatchProj(), "must be a CatchProjNode");
 169             if (cpn->_con == CatchProjNode::fall_through_index)
 170               _fallthroughcatchproj = cpn;
 171             else {
 172               assert(cpn->_con == CatchProjNode::catch_all_index, "must be correct index.");
 173               _catchallcatchproj = cpn;
 174             }
 175           }
 176         }
 177         break;
 178       }
 179       case TypeFunc::I_O:
 180         if (pn->_is_io_use)
 181           _ioproj_catchall = pn;
 182         else
 183           _ioproj_fallthrough = pn;
 184         break;
 185       case TypeFunc::Memory:
 186         if (pn->_is_io_use)
 187           _memproj_catchall = pn;
 188         else
 189           _memproj_fallthrough = pn;
 190         break;
 191       case TypeFunc::Parms:
 192         _resproj = pn;
 193         break;
 194       default:
 195         assert(false, "unexpected projection from allocation node.");
 196     }
 197   }
 198

 206 // Search for a memory operation for the specified memory slice.
 207 static Node *scan_mem_chain(Node *mem, int alias_idx, int offset, Node *start_mem, Node *alloc, PhaseGVN *phase) {
 208   Node *orig_mem = mem;
 209   Node *alloc_mem = alloc->in(TypeFunc::Memory);
 210   const TypeOopPtr *tinst = phase->C->get_adr_type(alias_idx)->isa_oopptr();
 211   while (true) {
 212     if (mem == alloc_mem || mem == start_mem ) {
 213       return mem;  // hit one of our sentinels
 214     } else if (mem->is_MergeMem()) {
 215       mem = mem->as_MergeMem()->memory_at(alias_idx);
 216     } else if (mem->is_Proj() && mem->as_Proj()->_con == TypeFunc::Memory) {
 217       Node *in = mem->in(0);
 218       // we can safely skip over safepoints, calls, locks and membars because we
 219       // already know that the object is safe to eliminate.
 220       if (in->is_Initialize() && in->as_Initialize()->allocation() == alloc) {
 221         return in;
 222       } else if (in->is_Call()) {
 223         CallNode *call = in->as_Call();
 224         if (call->may_modify(tinst, phase)) {
 225           assert(call->is_ArrayCopy(), "ArrayCopy is the only call node that doesn't make allocation escape");
 226           if (call->as_ArrayCopy()->modifies(offset, offset, phase, false)) {
 227             return in;
 228           }
 229         }
 230         mem = in->in(TypeFunc::Memory);
 231       } else if (in->is_MemBar()) {
 232         ArrayCopyNode* ac = NULL;
 233         if (ArrayCopyNode::may_modify(tinst, in->as_MemBar(), phase, ac)) {
 234           assert(ac != NULL && ac->is_clonebasic(), "Only basic clone is a non escaping clone");
 235           return ac;
 236         }
 237         mem = in->in(TypeFunc::Memory);
 238       } else {
 239         assert(false, "unexpected projection");
 240       }
 241     } else if (mem->is_Store()) {
 242       const TypePtr* atype = mem->as_Store()->adr_type();
 243       int adr_idx = phase->C->get_alias_index(atype);
 244       if (adr_idx == alias_idx) {
 245         assert(atype->isa_oopptr(), "address type must be oopptr");
 246         int adr_offset = atype->flattened_offset();
 247         uint adr_iid = atype->is_oopptr()->instance_id();
 248         // Array elements references have the same alias_idx
 249         // but different offset and different instance_id.
 250         if (adr_offset == offset && adr_iid == alloc->_idx)
 251           return mem;
 252       } else {
 253         assert(adr_idx == Compile::AliasIdxRaw, "address must match or be raw");
 254       }
 255       mem = mem->in(MemNode::Memory);
 256     } else if (mem->is_ClearArray()) {
 257       if (!ClearArrayNode::step_through(&mem, alloc->_idx, phase)) {
 258         // Can not bypass initialization of the instance
 259         // we are looking.
 260         debug_only(intptr_t offset;)
 261         assert(alloc == AllocateNode::Ideal_allocation(mem->in(3), phase, offset), "sanity");
 262         InitializeNode* init = alloc->as_Allocate()->initialization();
 263         // We are looking for stored value, return Initialize node
 264         // or memory edge from Allocate node.
 265         if (init != NULL)
 266           return init;
 267         else
 268           return alloc->in(TypeFunc::Memory); // It will produce zero value (see callers).
 269       }
 270       // Otherwise skip it (the call updated 'mem' value).
 271     } else if (mem->Opcode() == Op_SCMemProj) {
 272       mem = mem->in(0);
 273       Node* adr = NULL;
 274       if (mem->is_LoadStore()) {
 275         adr = mem->in(MemNode::Address);
 276       } else {
 277         assert(mem->Opcode() == Op_EncodeISOArray ||
 278                mem->Opcode() == Op_StrCompressedCopy, "sanity");
 279         adr = mem->in(3); // Destination array
 280       }
 281       const TypePtr* atype = adr->bottom_type()->is_ptr();
 282       int adr_idx = phase->C->get_alias_index(atype);
 283       if (adr_idx == alias_idx) {
 284         DEBUG_ONLY(mem->dump();)
 285         assert(false, "Object is not scalar replaceable if a LoadStore node accesses its field");
 286         return NULL;
 287       }
 288       mem = mem->in(MemNode::Memory);
 289     } else if (mem->Opcode() == Op_StrInflatedCopy) {
 290       Node* adr = mem->in(3); // Destination array
 291       const TypePtr* atype = adr->bottom_type()->is_ptr();
 292       int adr_idx = phase->C->get_alias_index(atype);
 293       if (adr_idx == alias_idx) {
 294         DEBUG_ONLY(mem->dump();)
 295         assert(false, "Object is not scalar replaceable if a StrInflatedCopy node accesses its field");
 296         return NULL;
 297       }
 298       mem = mem->in(MemNode::Memory);
 299     } else {
 300       return mem;
 301     }
 302     assert(mem != orig_mem, "dead memory loop");
 303   }
 304 }

 306 // Generate loads from source of the arraycopy for fields of
 307 // destination needed at a deoptimization point
 308 Node* PhaseMacroExpand::make_arraycopy_load(ArrayCopyNode* ac, intptr_t offset, Node* ctl, Node* mem, BasicType ft, const Type *ftype, AllocateNode *alloc) {
 309   BasicType bt = ft;
 310   const Type *type = ftype;
 311   if (ft == T_NARROWOOP) {
 312     bt = T_OBJECT;
 313     type = ftype->make_oopptr();
 314   }
 315   Node* res = NULL;
 316   if (ac->is_clonebasic()) {
 317     assert(ac->in(ArrayCopyNode::Src) != ac->in(ArrayCopyNode::Dest), "clone source equals destination");
 318     Node* base = ac->in(ArrayCopyNode::Src);
 319     Node* adr = _igvn.transform(new AddPNode(base, base, MakeConX(offset)));
 320     const TypePtr* adr_type = _igvn.type(base)->is_ptr()->add_offset(offset);
 321     res = LoadNode::make(_igvn, ctl, mem, adr, adr_type, type, bt, MemNode::unordered, LoadNode::UnknownControl);
 322   } else {
 323     if (ac->modifies(offset, offset, &_igvn, true)) {
 324       assert(ac->in(ArrayCopyNode::Dest) == alloc->result_cast(), "arraycopy destination should be allocation's result");
 325       uint shift = exact_log2(type2aelembytes(bt));
 326       Node* src_pos = ac->in(ArrayCopyNode::SrcPos);
 327       Node* dest_pos = ac->in(ArrayCopyNode::DestPos);
 328       const TypeInt* src_pos_t = _igvn.type(src_pos)->is_int();
 329       const TypeInt* dest_pos_t = _igvn.type(dest_pos)->is_int();
 330 
 331       Node* adr = NULL;
 332       Node* base = ac->in(ArrayCopyNode::Src);
 333       const TypePtr* adr_type = _igvn.type(base)->is_ptr();
 334       assert(adr_type->isa_aryptr(), "only arrays here");
 335       if (src_pos_t->is_con() && dest_pos_t->is_con()) {
 336         intptr_t off = ((src_pos_t->get_con() - dest_pos_t->get_con()) << shift) + offset;
 337         adr = _igvn.transform(new AddPNode(base, base, MakeConX(off)));
 338         adr_type = _igvn.type(adr)->is_ptr();
 339         assert(adr_type == _igvn.type(base)->is_aryptr()->add_field_offset_and_offset(off), "incorrect address type");
 340         if (ac->in(ArrayCopyNode::Src) == ac->in(ArrayCopyNode::Dest)) {
 341           // Don't emit a new load from src if src == dst but try to get the value from memory instead
 342           return value_from_mem(ac->in(TypeFunc::Memory), ctl, ft, ftype, adr_type->isa_oopptr(), alloc);
 343         }
 344       } else {
 345         if (ac->in(ArrayCopyNode::Src) == ac->in(ArrayCopyNode::Dest)) {
 346           // Non constant offset in the array: we can't statically
 347           // determine the value
 348           return NULL;
 349         }
 350         Node* diff = _igvn.transform(new SubINode(ac->in(ArrayCopyNode::SrcPos), ac->in(ArrayCopyNode::DestPos)));
 351 #ifdef _LP64
 352         diff = _igvn.transform(new ConvI2LNode(diff));
 353 #endif
 354         diff = _igvn.transform(new LShiftXNode(diff, intcon(shift)));
 355 
 356         Node* off = _igvn.transform(new AddXNode(MakeConX(offset), diff));
 357         adr = _igvn.transform(new AddPNode(base, base, off));
 358         // In the case of a flattened value type array, each field has its
 359         // own slice so we need to extract the field being accessed from
 360         // the address computation
 361         adr_type = adr_type->is_aryptr()->add_field_offset_and_offset(offset)->add_offset(Type::OffsetBot);
 362         adr = _igvn.transform(new CastPPNode(adr, adr_type));
 363       }
 364       res = LoadNode::make(_igvn, ctl, mem, adr, adr_type, type, bt, MemNode::unordered, LoadNode::UnknownControl);
 365     }
 366   }
 367   if (res != NULL) {
 368     res = _igvn.transform(res);
 369     if (ftype->isa_narrowoop()) {
 370       // PhaseMacroExpand::scalar_replacement adds DecodeN nodes
 371       assert(res->isa_DecodeN(), "should be narrow oop");
 372       res = _igvn.transform(new EncodePNode(res, ftype));
 373     }
 374     return res;
 375   }
 376   return NULL;
 377 }

 379 //
 380 // Given a Memory Phi, compute a value Phi containing the values from stores
 381 // on the input paths.
 382 // Note: this function is recursive, its depth is limited by the "level" argument
 383 // Returns the computed Phi, or NULL if it cannot compute it.
 384 Node *PhaseMacroExpand::value_from_mem_phi(Node *mem, BasicType ft, const Type *phi_type, const TypeOopPtr *adr_t, AllocateNode *alloc, Node_Stack *value_phis, int level) {
 385   assert(mem->is_Phi(), "sanity");
 386   int alias_idx = C->get_alias_index(adr_t);
 387   int offset = adr_t->flattened_offset();
 388   int instance_id = adr_t->instance_id();
 389 
 390   // Check if an appropriate value phi already exists.
 391   Node* region = mem->in(0);
 392   for (DUIterator_Fast kmax, k = region->fast_outs(kmax); k < kmax; k++) {
 393     Node* phi = region->fast_out(k);
 394     if (phi->is_Phi() && phi != mem &&
 395         phi->as_Phi()->is_same_inst_field(phi_type, (int)mem->_idx, instance_id, alias_idx, offset)) {
 396       return phi;
 397     }
 398   }
 399   // Check if an appropriate new value phi already exists.
 400   Node* new_phi = value_phis->find(mem->_idx);
 401   if (new_phi != NULL)
 402     return new_phi;
 403 
 404   if (level <= 0) {
 405     return NULL; // Give up: phi tree too deep
 406   }
 407   Node *start_mem = C->start()->proj_out_or_null(TypeFunc::Memory);
 408   Node *alloc_mem = alloc->in(TypeFunc::Memory);
 409 
 410   uint length = mem->req();
 411   GrowableArray <Node *> values(length, length, NULL, false);
 412 
 413   // create a new Phi for the value
 414   PhiNode *phi = new PhiNode(mem->in(0), phi_type, NULL, mem->_idx, instance_id, alias_idx, offset);
 415   transform_later(phi);
 416   value_phis->push(phi, mem->_idx);
 417 
 418   for (uint j = 1; j < length; j++) {
 419     Node *in = mem->in(j);
 420     if (in == NULL || in->is_top()) {
 421       values.at_put(j, in);
 422     } else  {
 423       Node *val = scan_mem_chain(in, alias_idx, offset, start_mem, alloc, &_igvn);
 424       if (val == start_mem || val == alloc_mem) {
 425         // hit a sentinel, return appropriate 0 value
 426         Node* default_value = alloc->in(AllocateNode::DefaultValue);
 427         if (default_value != NULL) {
 428           values.at_put(j, default_value);
 429         } else {
 430           assert(alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
 431           values.at_put(j, _igvn.zerocon(ft));
 432         }
 433         continue;
 434       }
 435       if (val->is_Initialize()) {
 436         val = val->as_Initialize()->find_captured_store(offset, type2aelembytes(ft), &_igvn);
 437       }
 438       if (val == NULL) {
 439         return NULL;  // can't find a value on this path
 440       }
 441       if (val == mem) {
 442         values.at_put(j, mem);
 443       } else if (val->is_Store()) {
 444         Node* n = val->in(MemNode::ValueIn);
 445         BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 446         n = bs->step_over_gc_barrier(n);
 447         values.at_put(j, n);
 448       } else if(val->is_Proj() && val->in(0) == alloc) {
 449         Node* default_value = alloc->in(AllocateNode::DefaultValue);
 450         if (default_value != NULL) {
 451           values.at_put(j, default_value);
 452         } else {
 453           assert(alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
 454           values.at_put(j, _igvn.zerocon(ft));
 455         }
 456       } else if (val->is_Phi()) {
 457         val = value_from_mem_phi(val, ft, phi_type, adr_t, alloc, value_phis, level-1);
 458         if (val == NULL) {
 459           return NULL;
 460         }
 461         values.at_put(j, val);
 462       } else if (val->Opcode() == Op_SCMemProj) {
 463         assert(val->in(0)->is_LoadStore() ||
 464                val->in(0)->Opcode() == Op_EncodeISOArray ||
 465                val->in(0)->Opcode() == Op_StrCompressedCopy, "sanity");
 466         assert(false, "Object is not scalar replaceable if a LoadStore node accesses its field");
 467         return NULL;
 468       } else if (val->is_ArrayCopy()) {
 469         Node* res = make_arraycopy_load(val->as_ArrayCopy(), offset, val->in(0), val->in(TypeFunc::Memory), ft, phi_type, alloc);
 470         if (res == NULL) {
 471           return NULL;
 472         }
 473         values.at_put(j, res);
 474       } else {
 475 #ifdef ASSERT
 476         val->dump();
 477         assert(false, "unknown node on this path");
 478 #endif
 479         return NULL;  // unknown node on this path
 480       }
 481     }
 482   }
 483   // Set Phi's inputs
 484   for (uint j = 1; j < length; j++) {
 485     if (values.at(j) == mem) {
 486       phi->init_req(j, phi);
 487     } else {
 488       phi->init_req(j, values.at(j));
 489     }
 490   }
 491   return phi;
 492 }

 494 // Search the last value stored into the object's field.
 495 Node *PhaseMacroExpand::value_from_mem(Node *sfpt_mem, Node *sfpt_ctl, BasicType ft, const Type *ftype, const TypeOopPtr *adr_t, AllocateNode *alloc) {
 496   assert(adr_t->is_known_instance_field(), "instance required");
 497   int instance_id = adr_t->instance_id();
 498   assert((uint)instance_id == alloc->_idx, "wrong allocation");
 499 
 500   int alias_idx = C->get_alias_index(adr_t);
 501   int offset = adr_t->flattened_offset();
 502   Node *start_mem = C->start()->proj_out_or_null(TypeFunc::Memory);
 503   Node *alloc_mem = alloc->in(TypeFunc::Memory);
 504   Arena *a = Thread::current()->resource_area();
 505   VectorSet visited(a);
 506 
 507   bool done = sfpt_mem == alloc_mem;
 508   Node *mem = sfpt_mem;
 509   while (!done) {
 510     if (visited.test_set(mem->_idx)) {
 511       return NULL;  // found a loop, give up
 512     }
 513     mem = scan_mem_chain(mem, alias_idx, offset, start_mem, alloc, &_igvn);
 514     if (mem == start_mem || mem == alloc_mem) {
 515       done = true;  // hit a sentinel, return appropriate 0 value
 516     } else if (mem->is_Initialize()) {
 517       mem = mem->as_Initialize()->find_captured_store(offset, type2aelembytes(ft), &_igvn);
 518       if (mem == NULL) {
 519         done = true; // Something went wrong.
 520       } else if (mem->is_Store()) {
 521         const TypePtr* atype = mem->as_Store()->adr_type();
 522         assert(C->get_alias_index(atype) == Compile::AliasIdxRaw, "store is correct memory slice");
 523         done = true;
 524       }
 525     } else if (mem->is_Store()) {
 526       const TypeOopPtr* atype = mem->as_Store()->adr_type()->isa_oopptr();
 527       assert(atype != NULL, "address type must be oopptr");
 528       assert(C->get_alias_index(atype) == alias_idx &&
 529              atype->is_known_instance_field() && atype->flattened_offset() == offset &&
 530              atype->instance_id() == instance_id, "store is correct memory slice");
 531       done = true;
 532     } else if (mem->is_Phi()) {
 533       // try to find a phi's unique input
 534       Node *unique_input = NULL;
 535       Node *top = C->top();
 536       for (uint i = 1; i < mem->req(); i++) {
 537         Node *n = scan_mem_chain(mem->in(i), alias_idx, offset, start_mem, alloc, &_igvn);
 538         if (n == NULL || n == top || n == mem) {
 539           continue;
 540         } else if (unique_input == NULL) {
 541           unique_input = n;
 542         } else if (unique_input != n) {
 543           unique_input = top;
 544           break;
 545         }
 546       }
 547       if (unique_input != NULL && unique_input != top) {
 548         mem = unique_input;
 549       } else {
 550         done = true;
 551       }
 552     } else if (mem->is_ArrayCopy()) {
 553       done = true;
 554     } else {
 555       assert(false, "unexpected node");
 556     }
 557   }
 558   if (mem != NULL) {
 559     if (mem == start_mem || mem == alloc_mem) {
 560       // hit a sentinel, return appropriate 0 value
 561       Node* default_value = alloc->in(AllocateNode::DefaultValue);
 562       if (default_value != NULL) {
 563         return default_value;
 564       }
 565       assert(alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
 566       return _igvn.zerocon(ft);
 567     } else if (mem->is_Store()) {
 568       Node* n = mem->in(MemNode::ValueIn);
 569       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 570       n = bs->step_over_gc_barrier(n);
 571       return n;
 572     } else if (mem->is_Phi()) {
 573       // attempt to produce a Phi reflecting the values on the input paths of the Phi
 574       Node_Stack value_phis(a, 8);
 575       Node * phi = value_from_mem_phi(mem, ft, ftype, adr_t, alloc, &value_phis, ValueSearchLimit);
 576       if (phi != NULL) {
 577         return phi;
 578       } else {
 579         // Kill all new Phis
 580         while(value_phis.is_nonempty()) {
 581           Node* n = value_phis.node();
 582           _igvn.replace_node(n, C->top());
 583           value_phis.pop();
 584         }
 585       }
 586     } else if (mem->is_ArrayCopy()) {
 587       Node* ctl = mem->in(0);
 588       Node* m = mem->in(TypeFunc::Memory);
 589       if (sfpt_ctl->is_Proj() && sfpt_ctl->as_Proj()->is_uncommon_trap_proj(Deoptimization::Reason_none)) {
 590         // pin the loads in the uncommon trap path
 591         ctl = sfpt_ctl;
 592         m = sfpt_mem;
 593       }
 594       return make_arraycopy_load(mem->as_ArrayCopy(), offset, ctl, m, ft, ftype, alloc);
 595     }
 596   }
 597   // Something went wrong.
 598   return NULL;
 599 }

 601 // Search the last value stored into the value type's fields.
 602 Node* PhaseMacroExpand::value_type_from_mem(Node* mem, Node* ctl, ciValueKlass* vk, const TypeAryPtr* adr_type, int offset, AllocateNode* alloc) {
 603   // Subtract the offset of the first field to account for the missing oop header
 604   offset -= vk->first_field_offset();
 605   // Create a new ValueTypeNode and retrieve the field values from memory
 606   ValueTypeNode* vt = ValueTypeNode::make_uninitialized(_igvn, vk)->as_ValueType();
 607   for (int i = 0; i < vk->nof_declared_nonstatic_fields(); ++i) {
 608     ciType* field_type = vt->field_type(i);
 609     int field_offset = offset + vt->field_offset(i);
 610     // Each value type field has its own memory slice
 611     adr_type = adr_type->with_field_offset(field_offset);
 612     Node* value = NULL;
 613     if (vt->field_is_flattened(i)) {
 614       value = value_type_from_mem(mem, ctl, field_type->as_value_klass(), adr_type, field_offset, alloc);
 615     } else {
 616       const Type* ft = Type::get_const_type(field_type);
 617       BasicType bt = field_type->basic_type();
 618       if (UseCompressedOops && !is_java_primitive(bt)) {
 619         ft = ft->make_narrowoop();
 620         bt = T_NARROWOOP;
 621       }
 622       value = value_from_mem(mem, ctl, bt, ft, adr_type, alloc);
 623       if (value != NULL && ft->isa_narrowoop()) {
 624         assert(UseCompressedOops, "unexpected narrow oop");
 625         value = transform_later(new DecodeNNode(value, value->get_ptr_type()));
 626       }
 627     }
 628     if (value != NULL) {
 629       vt->set_field_value(i, value);
 630     } else {
 631       // We might have reached the TrackedInitializationLimit
 632       return NULL;
 633     }
 634   }
 635   return transform_later(vt);
 636 }

 638 // Check the possibility of scalar replacement.
 639 bool PhaseMacroExpand::can_eliminate_allocation(AllocateNode *alloc, GrowableArray <SafePointNode *>& safepoints) {
 640   //  Scan the uses of the allocation to check for anything that would
 641   //  prevent us from eliminating it.
 642   NOT_PRODUCT( const char* fail_eliminate = NULL; )
 643   DEBUG_ONLY( Node* disq_node = NULL; )
 644   bool  can_eliminate = true;
 645 
 646   Node* res = alloc->result_cast();
 647   const TypeOopPtr* res_type = NULL;
 648   if (res == NULL) {
 649     // All users were eliminated.
 650   } else if (!res->is_CheckCastPP()) {
 651     NOT_PRODUCT(fail_eliminate = "Allocation does not have unique CheckCastPP";)
 652     can_eliminate = false;
 653   } else {
 654     res_type = _igvn.type(res)->isa_oopptr();
 655     if (res_type == NULL) {
 656       NOT_PRODUCT(fail_eliminate = "Neither instance or array allocation";)
 657       can_eliminate = false;
 658     } else if (res_type->isa_aryptr()) {
 659       int length = alloc->in(AllocateNode::ALength)->find_int_con(-1);
 660       if (length < 0) {
 661         NOT_PRODUCT(fail_eliminate = "Array's size is not constant";)
 662         can_eliminate = false;
 663       }
 664     }
 665   }
 666 
 667   if (can_eliminate && res != NULL) {
 668     for (DUIterator_Fast jmax, j = res->fast_outs(jmax);
 669                                j < jmax && can_eliminate; j++) {
 670       Node* use = res->fast_out(j);
 671 
 672       if (use->is_AddP()) {
 673         const TypePtr* addp_type = _igvn.type(use)->is_ptr();
 674         int offset = addp_type->offset();
 675 
 676         if (offset == Type::OffsetTop || offset == Type::OffsetBot) {
 677           NOT_PRODUCT(fail_eliminate = "Undefined field referrence";)
 678           can_eliminate = false;
 679           break;
 680         }
 681         for (DUIterator_Fast kmax, k = use->fast_outs(kmax);
 682                                    k < kmax && can_eliminate; k++) {
 683           Node* n = use->fast_out(k);
 684           if (!n->is_Store() && n->Opcode() != Op_CastP2X
 685               SHENANDOAHGC_ONLY(&& (!UseShenandoahGC || !ShenandoahBarrierSetC2::is_shenandoah_wb_pre_call(n))) ) {
 686             DEBUG_ONLY(disq_node = n;)
 687             if (n->is_Load() || n->is_LoadStore()) {
 688               NOT_PRODUCT(fail_eliminate = "Field load";)
 689             } else {
 690               NOT_PRODUCT(fail_eliminate = "Not store field reference";)
 691             }
 692             can_eliminate = false;
 693           }
 694         }
 695       } else if (use->is_ArrayCopy() &&
 696                  (use->as_ArrayCopy()->is_clonebasic() ||
 697                   use->as_ArrayCopy()->is_arraycopy_validated() ||
 698                   use->as_ArrayCopy()->is_copyof_validated() ||
 699                   use->as_ArrayCopy()->is_copyofrange_validated()) &&
 700                  use->in(ArrayCopyNode::Dest) == res) {
 701         // ok to eliminate
 702       } else if (use->is_SafePoint()) {
 703         SafePointNode* sfpt = use->as_SafePoint();
 704         if (sfpt->is_Call() && sfpt->as_Call()->has_non_debug_use(res)) {
 705           // Object is passed as argument.
 706           DEBUG_ONLY(disq_node = use;)
 707           NOT_PRODUCT(fail_eliminate = "Object is passed as argument";)
 708           can_eliminate = false;
 709         }
 710         Node* sfptMem = sfpt->memory();
 711         if (sfptMem == NULL || sfptMem->is_top()) {
 712           DEBUG_ONLY(disq_node = use;)
 713           NOT_PRODUCT(fail_eliminate = "NULL or TOP memory";)
 714           can_eliminate = false;
 715         } else {
 716           safepoints.append_if_missing(sfpt);
 717         }
 718       } else if (use->is_ValueType() && use->isa_ValueType()->get_oop() == res) {
 719         // ok to eliminate
 720       } else if (use->Opcode() == Op_StoreX && use->in(MemNode::Address) == res) {
 721         // store to mark work
 722       } else if (use->Opcode() != Op_CastP2X) { // CastP2X is used by card mark
 723         if (use->is_Phi()) {
 724           if (use->outcnt() == 1 && use->unique_out()->Opcode() == Op_Return) {
 725             NOT_PRODUCT(fail_eliminate = "Object is return value";)
 726           } else {
 727             NOT_PRODUCT(fail_eliminate = "Object is referenced by Phi";)
 728           }
 729           DEBUG_ONLY(disq_node = use;)
 730         } else {
 731           if (use->Opcode() == Op_Return) {
 732             NOT_PRODUCT(fail_eliminate = "Object is return value";)
 733           } else {
 734             NOT_PRODUCT(fail_eliminate = "Object is referenced by node";)
 735           }
 736           DEBUG_ONLY(disq_node = use;)
 737         }
 738         can_eliminate = false;
 739       } else {
 740         assert(use->Opcode() == Op_CastP2X, "should be");
 741         assert(!use->has_out_with(Op_OrL), "should have been removed because oop is never null");
 742       }
 743     }
 744   }
 745 
 746 #ifndef PRODUCT
 747   if (PrintEliminateAllocations) {
 748     if (can_eliminate) {
 749       tty->print("Scalar ");
 750       if (res == NULL)
 751         alloc->dump();
 752       else
 753         res->dump();
 754     } else if (alloc->_is_scalar_replaceable) {
 755       tty->print("NotScalar (%s)", fail_eliminate);
 756       if (res == NULL)
 757         alloc->dump();
 758       else
 759         res->dump();
 760 #ifdef ASSERT
 761       if (disq_node != NULL) {
 762           tty->print("  >>>> ");
 763           disq_node->dump();
 764       }
 765 #endif /*ASSERT*/
 766     }
 767   }
 768 #endif
 769   return can_eliminate;
 770 }

 772 // Do scalar replacement.
 773 bool PhaseMacroExpand::scalar_replacement(AllocateNode *alloc, GrowableArray <SafePointNode *>& safepoints) {
 774   GrowableArray <SafePointNode *> safepoints_done;
 775 
 776   ciKlass* klass = NULL;
 777   ciInstanceKlass* iklass = NULL;
 778   int nfields = 0;
 779   int array_base = 0;
 780   int element_size = 0;
 781   BasicType basic_elem_type = T_ILLEGAL;
 782   ciType* elem_type = NULL;
 783 
 784   Node* res = alloc->result_cast();
 785   assert(res == NULL || res->is_CheckCastPP(), "unexpected AllocateNode result");
 786   const TypeOopPtr* res_type = NULL;
 787   if (res != NULL) { // Could be NULL when there are no users
 788     res_type = _igvn.type(res)->isa_oopptr();
 789   }
 790 
 791   if (res != NULL) {
 792     klass = res_type->klass();
 793     if (res_type->isa_instptr()) {
 794       // find the fields of the class which will be needed for safepoint debug information
 795       assert(klass->is_instance_klass(), "must be an instance klass.");
 796       iklass = klass->as_instance_klass();
 797       nfields = iklass->nof_nonstatic_fields();
 798     } else {
 799       // find the array's elements which will be needed for safepoint debug information
 800       nfields = alloc->in(AllocateNode::ALength)->find_int_con(-1);
 801       assert(klass->is_array_klass() && nfields >= 0, "must be an array klass.");
 802       elem_type = klass->as_array_klass()->element_type();
 803       basic_elem_type = elem_type->basic_type();
 804       if (elem_type->is_valuetype() && !klass->is_value_array_klass()) {
 805         assert(basic_elem_type == T_VALUETYPE, "unexpected element basic type");
 806         basic_elem_type = T_OBJECT;
 807       }
 808       array_base = arrayOopDesc::base_offset_in_bytes(basic_elem_type);
 809       element_size = type2aelembytes(basic_elem_type);
 810       if (klass->is_value_array_klass()) {
 811         // Flattened value type array
 812         element_size = klass->as_value_array_klass()->element_byte_size();
 813       }
 814     }
 815   }
 816   //
 817   // Process the safepoint uses
 818   //
 819   Unique_Node_List value_worklist;
 820   while (safepoints.length() > 0) {
 821     SafePointNode* sfpt = safepoints.pop();
 822     Node* mem = sfpt->memory();
 823     Node* ctl = sfpt->control();
 824     assert(sfpt->jvms() != NULL, "missed JVMS");
 825     // Fields of scalar objs are referenced only at the end
 826     // of regular debuginfo at the last (youngest) JVMS.
 827     // Record relative start index.
 828     uint first_ind = (sfpt->req() - sfpt->jvms()->scloff());
 829     SafePointScalarObjectNode* sobj = new SafePointScalarObjectNode(res_type,
 830 #ifdef ASSERT
 831                                                  alloc,
 832 #endif
 833                                                  first_ind, nfields);
 834     sobj->init_req(0, C->root());
 835     transform_later(sobj);
 836 
 837     // Scan object's fields adding an input to the safepoint for each field.
 838     for (int j = 0; j < nfields; j++) {
 839       intptr_t offset;
 840       ciField* field = NULL;
 841       if (iklass != NULL) {
 842         field = iklass->nonstatic_field_at(j);
 843         offset = field->offset();
 844         elem_type = field->type();
 845         basic_elem_type = field->layout_type();
 846         assert(!field->is_flattened(), "flattened value type fields should not have safepoint uses");
 847       } else {
 848         offset = array_base + j * (intptr_t)element_size;
 849       }
 850 
 851       const Type *field_type;
 852       // The next code is taken from Parse::do_get_xxx().
 853       if (is_reference_type(basic_elem_type)) {
 854         if (!elem_type->is_loaded()) {
 855           field_type = TypeInstPtr::BOTTOM;
 856         } else if (field != NULL && field->is_static_constant()) {
 857           // This can happen if the constant oop is non-perm.
 858           ciObject* con = field->constant_value().as_object();
 859           // Do not "join" in the previous type; it doesn't add value,
 860           // and may yield a vacuous result if the field is of interface type.
 861           field_type = TypeOopPtr::make_from_constant(con)->isa_oopptr();
 862           assert(field_type != NULL, "field singleton type must be consistent");
 863         } else {
 864           field_type = TypeOopPtr::make_from_klass(elem_type->as_klass());
 865         }
 866         if (UseCompressedOops) {
 867           field_type = field_type->make_narrowoop();
 868           basic_elem_type = T_NARROWOOP;
 869         }
 870       } else {
 871         field_type = Type::get_const_basic_type(basic_elem_type);
 872       }
 873 
 874       Node* field_val = NULL;
 875       const TypeOopPtr* field_addr_type = res_type->add_offset(offset)->isa_oopptr();
 876       if (klass->is_value_array_klass()) {
 877         ciValueKlass* vk = elem_type->as_value_klass();
 878         assert(vk->flatten_array(), "must be flattened");
 879         field_val = value_type_from_mem(mem, ctl, vk, field_addr_type->isa_aryptr(), 0, alloc);
 880       } else {
 881         field_val = value_from_mem(mem, ctl, basic_elem_type, field_type, field_addr_type, alloc);
 882       }
 883       if (field_val == NULL) {
 884         // We weren't able to find a value for this field,
 885         // give up on eliminating this allocation.
 886 
 887         // Remove any extra entries we added to the safepoint.
 888         uint last = sfpt->req() - 1;
 889         for (int k = 0;  k < j; k++) {
 890           sfpt->del_req(last--);
 891         }
 892         _igvn._worklist.push(sfpt);
 893         // rollback processed safepoints
 894         while (safepoints_done.length() > 0) {
 895           SafePointNode* sfpt_done = safepoints_done.pop();
 896           // remove any extra entries we added to the safepoint
 897           last = sfpt_done->req() - 1;
 898           for (int k = 0;  k < nfields; k++) {
 899             sfpt_done->del_req(last--);
 900           }
 901           JVMState *jvms = sfpt_done->jvms();
 902           jvms->set_endoff(sfpt_done->req());
 903           // Now make a pass over the debug information replacing any references
 904           // to SafePointScalarObjectNode with the allocated object.
 905           int start = jvms->debug_start();
 906           int end   = jvms->debug_end();
 907           for (int i = start; i < end; i++) {
 908             if (sfpt_done->in(i)->is_SafePointScalarObject()) {
 909               SafePointScalarObjectNode* scobj = sfpt_done->in(i)->as_SafePointScalarObject();
 910               if (scobj->first_index(jvms) == sfpt_done->req() &&
 911                   scobj->n_fields() == (uint)nfields) {
 912                 assert(scobj->alloc() == alloc, "sanity");
 913                 sfpt_done->set_req(i, res);
 914               }
 915             }
 916           }
 917           _igvn._worklist.push(sfpt_done);
 918         }
 919 #ifndef PRODUCT
 920         if (PrintEliminateAllocations) {
 921           if (field != NULL) {
 922             tty->print("=== At SafePoint node %d can't find value of Field: ",
 923                        sfpt->_idx);
 924             field->print();
 925             int field_idx = C->get_alias_index(field_addr_type);
 926             tty->print(" (alias_idx=%d)", field_idx);
 927           } else { // Array's element
 928             tty->print("=== At SafePoint node %d can't find value of array element [%d]",
 929                        sfpt->_idx, j);
 930           }
 931           tty->print(", which prevents elimination of: ");
 932           if (res == NULL)
 933             alloc->dump();
 934           else
 935             res->dump();
 936         }
 937 #endif
 938         return false;
 939       }
 940       if (field_val->is_ValueType()) {
 941         // Keep track of value types to scalarize them later
 942         value_worklist.push(field_val);
 943       } else if (UseCompressedOops && field_type->isa_narrowoop()) {
 944         // Enable "DecodeN(EncodeP(Allocate)) --> Allocate" transformation
 945         // to be able scalar replace the allocation.
 946         if (field_val->is_EncodeP()) {
 947           field_val = field_val->in(1);
 948         } else {
 949           field_val = transform_later(new DecodeNNode(field_val, field_val->get_ptr_type()));
 950         }
 951       }
 952       sfpt->add_req(field_val);
 953     }
 954     JVMState *jvms = sfpt->jvms();
 955     jvms->set_endoff(sfpt->req());
 956     // Now make a pass over the debug information replacing any references
 957     // to the allocated object with "sobj"
 958     int start = jvms->debug_start();
 959     int end   = jvms->debug_end();
 960     sfpt->replace_edges_in_range(res, sobj, start, end);
 961     _igvn._worklist.push(sfpt);
 962     safepoints_done.append_if_missing(sfpt); // keep it for rollback
 963   }
 964   // Scalarize value types that were added to the safepoint
 965   for (uint i = 0; i < value_worklist.size(); ++i) {
 966     Node* vt = value_worklist.at(i);
 967     vt->as_ValueType()->make_scalar_in_safepoints(&_igvn);
 968   }
 969   return true;
 970 }

 972 static void disconnect_projections(MultiNode* n, PhaseIterGVN& igvn) {
 973   Node* ctl_proj = n->proj_out_or_null(TypeFunc::Control);
 974   Node* mem_proj = n->proj_out_or_null(TypeFunc::Memory);
 975   if (ctl_proj != NULL) {
 976     igvn.replace_node(ctl_proj, n->in(0));
 977   }
 978   if (mem_proj != NULL) {
 979     igvn.replace_node(mem_proj, n->in(TypeFunc::Memory));
 980   }

 983 // Process users of eliminated allocation.
 984 void PhaseMacroExpand::process_users_of_allocation(CallNode *alloc, bool inline_alloc) {
 985   Node* res = alloc->result_cast();
 986   if (res != NULL) {
 987     for (DUIterator_Last jmin, j = res->last_outs(jmin); j >= jmin; ) {
 988       Node *use = res->last_out(j);
 989       uint oc1 = res->outcnt();
 990 
 991       if (use->is_AddP()) {
 992         for (DUIterator_Last kmin, k = use->last_outs(kmin); k >= kmin; ) {
 993           Node *n = use->last_out(k);
 994           uint oc2 = use->outcnt();
 995           if (n->is_Store()) {
 996 #ifdef ASSERT
 997             // Verify that there is no dependent MemBarVolatile nodes,
 998             // they should be removed during IGVN, see MemBarNode::Ideal().
 999             for (DUIterator_Fast pmax, p = n->fast_outs(pmax);
1000                                        p < pmax; p++) {
1001               Node* mb = n->fast_out(p);
1002               assert(mb->is_Initialize() || !mb->is_MemBar() ||
1003                      mb->req() <= MemBarNode::Precedent ||
1004                      mb->in(MemBarNode::Precedent) != n ||
1005                      (inline_alloc && !ReduceInitialCardMarks),
1006                      "MemBarVolatile should be eliminated for non-escaping object");
1007             }
1008 #endif
1009             _igvn.replace_node(n, n->in(MemNode::Memory));
1010           } else {
1011             eliminate_gc_barrier(n);
1012           }
1013           k -= (oc2 - use->outcnt());
1014         }
1015         _igvn.remove_dead_node(use);
1016       } else if (use->is_ArrayCopy()) {
1017         // Disconnect ArrayCopy node
1018         ArrayCopyNode* ac = use->as_ArrayCopy();
1019         if (ac->is_clonebasic()) {
1020           Node* membar_after = ac->proj_out(TypeFunc::Control)->unique_ctrl_out();
1021           disconnect_projections(ac, _igvn);
1022           assert(alloc->in(0)->is_Proj() && alloc->in(0)->in(0)->Opcode() == Op_MemBarCPUOrder, "mem barrier expected before allocation");
1023           Node* membar_before = alloc->in(0)->in(0);
1024           disconnect_projections(membar_before->as_MemBar(), _igvn);
1025           if (membar_after->is_MemBar()) {
1026             disconnect_projections(membar_after->as_MemBar(), _igvn);
1027           }
1028         } else {
1029           assert(ac->is_arraycopy_validated() ||
1030                  ac->is_copyof_validated() ||
1031                  ac->is_copyofrange_validated(), "unsupported");
1032           CallProjections* callprojs = ac->extract_projections(true);
1033 
1034           _igvn.replace_node(callprojs->fallthrough_ioproj, ac->in(TypeFunc::I_O));
1035           _igvn.replace_node(callprojs->fallthrough_memproj, ac->in(TypeFunc::Memory));
1036           _igvn.replace_node(callprojs->fallthrough_catchproj, ac->in(TypeFunc::Control));
1037 
1038           // Set control to top. IGVN will remove the remaining projections
1039           ac->set_req(0, top());
1040           ac->replace_edge(res, top());
1041 
1042           // Disconnect src right away: it can help find new
1043           // opportunities for allocation elimination
1044           Node* src = ac->in(ArrayCopyNode::Src);
1045           ac->replace_edge(src, top());
1046           // src can be top at this point if src and dest of the
1047           // arraycopy were the same
1048           if (src->outcnt() == 0 && !src->is_top()) {
1049             _igvn.remove_dead_node(src);
1050           }
1051         }
1052         _igvn._worklist.push(ac);
1053       } else if (use->is_ValueType()) {
1054         assert(use->isa_ValueType()->get_oop() == res, "unexpected value type use");
1055          _igvn.rehash_node_delayed(use);
1056         use->isa_ValueType()->set_oop(_igvn.zerocon(T_VALUETYPE));
1057       } else if (use->is_Store()) {
1058         _igvn.replace_node(use, use->in(MemNode::Memory));
1059       } else {
1060         eliminate_gc_barrier(use);
1061       }
1062       j -= (oc1 - res->outcnt());
1063     }
1064     assert(res->outcnt() == 0, "all uses of allocated objects must be deleted");
1065     _igvn.remove_dead_node(res);
1066   }
1067 
1068   //
1069   // Process other users of allocation's projections
1070   //
1071   if (_resproj != NULL && _resproj->outcnt() != 0) {
1072     // First disconnect stores captured by Initialize node.
1073     // If Initialize node is eliminated first in the following code,
1074     // it will kill such stores and DUIterator_Last will assert.
1075     for (DUIterator_Fast jmax, j = _resproj->fast_outs(jmax);  j < jmax; j++) {
1076       Node *use = _resproj->fast_out(j);
1077       if (use->is_AddP()) {
1078         // raw memory addresses used only by the initialization
1079         _igvn.replace_node(use, C->top());
1080         --j; --jmax;
1081       }
1082     }
1083     for (DUIterator_Last jmin, j = _resproj->last_outs(jmin); j >= jmin; ) {
1084       Node *use = _resproj->last_out(j);
1085       uint oc1 = _resproj->outcnt();
1086       if (use->is_Initialize()) {
1087         // Eliminate Initialize node.
1088         InitializeNode *init = use->as_Initialize();
1089         assert(init->outcnt() <= 2, "only a control and memory projection expected");
1090         Node *ctrl_proj = init->proj_out_or_null(TypeFunc::Control);
1091         if (ctrl_proj != NULL) {
1092           // Inline type buffer allocations are followed by a membar
1093           Node* membar_after = ctrl_proj->unique_ctrl_out();
1094           if (inline_alloc && membar_after->Opcode() == Op_MemBarCPUOrder) {
1095             membar_after->as_MemBar()->remove(&_igvn);
1096           }
1097           _igvn.replace_node(ctrl_proj, init->in(TypeFunc::Control));
1098 #ifdef ASSERT
1099           Node* tmp = init->in(TypeFunc::Control);
1100           assert(tmp == _fallthroughcatchproj, "allocation control projection");
1101 #endif
1102         }
1103         Node *mem_proj = init->proj_out_or_null(TypeFunc::Memory);
1104         if (mem_proj != NULL) {
1105           Node *mem = init->in(TypeFunc::Memory);
1106 #ifdef ASSERT
1107           if (mem->is_MergeMem()) {
1108             assert(mem->in(TypeFunc::Memory) == _memproj_fallthrough, "allocation memory projection");
1109           } else {
1110             assert(mem == _memproj_fallthrough, "allocation memory projection");
1111           }
1112 #endif
1113           _igvn.replace_node(mem_proj, mem);
1114         }
1115       } else if (use->Opcode() == Op_MemBarStoreStore) {
1116         // Inline type buffer allocations are followed by a membar
1117         assert(inline_alloc, "Unexpected MemBarStoreStore");
1118         use->as_MemBar()->remove(&_igvn);
1119       } else  {
1120         assert(false, "only Initialize or AddP expected");
1121       }
1122       j -= (oc1 - _resproj->outcnt());
1123     }
1124   }
1125   if (_fallthroughcatchproj != NULL) {
1126     _igvn.replace_node(_fallthroughcatchproj, alloc->in(TypeFunc::Control));
1127   }
1128   if (_memproj_fallthrough != NULL) {
1129     _igvn.replace_node(_memproj_fallthrough, alloc->in(TypeFunc::Memory));
1130   }
1131   if (_memproj_catchall != NULL) {
1132     _igvn.replace_node(_memproj_catchall, C->top());
1133   }
1134   if (_ioproj_fallthrough != NULL) {
1135     _igvn.replace_node(_ioproj_fallthrough, alloc->in(TypeFunc::I_O));
1136   }
1137   if (_ioproj_catchall != NULL) {
1138     _igvn.replace_node(_ioproj_catchall, C->top());
1139   }
1140   if (_catchallcatchproj != NULL) {
1141     _igvn.replace_node(_catchallcatchproj, C->top());
1142   }
1143 }

1145 bool PhaseMacroExpand::eliminate_allocate_node(AllocateNode *alloc) {
1146   // Don't do scalar replacement if the frame can be popped by JVMTI:
1147   // if reallocation fails during deoptimization we'll pop all
1148   // interpreter frames for this compiled frame and that won't play
1149   // nice with JVMTI popframe.
1150   if (!EliminateAllocations || JvmtiExport::can_pop_frame()) {
1151     return false;
1152   }
1153   Node* klass = alloc->in(AllocateNode::KlassNode);
1154   const TypeKlassPtr* tklass = _igvn.type(klass)->is_klassptr();
1155 
1156   // Attempt to eliminate inline type buffer allocations
1157   // regardless of usage and escape/replaceable status.
1158   bool inline_alloc = tklass->klass()->is_valuetype();
1159   if (!alloc->_is_non_escaping && !inline_alloc) {
1160     return false;
1161   }
1162   // Eliminate boxing allocations which are not used
1163   // regardless of scalar replaceable status.
1164   Node* res = alloc->result_cast();
1165   bool boxing_alloc = (res == NULL) && C->eliminate_boxing() &&
1166                       tklass->klass()->is_instance_klass() &&
1167                       tklass->klass()->as_instance_klass()->is_box_klass();
1168   if (!alloc->_is_scalar_replaceable && !boxing_alloc && !inline_alloc) {
1169     return false;
1170   }
1171 
1172   extract_call_projections(alloc);
1173 
1174   GrowableArray <SafePointNode *> safepoints;
1175   if (!can_eliminate_allocation(alloc, safepoints)) {
1176     return false;
1177   }
1178 
1179   if (!alloc->_is_scalar_replaceable) {
1180     assert(res == NULL || inline_alloc, "sanity");
1181     // We can only eliminate allocation if all debug info references
1182     // are already replaced with SafePointScalarObject because
1183     // we can't search for a fields value without instance_id.
1184     if (safepoints.length() > 0) {
1185       assert(!inline_alloc, "Inline type allocations should not have safepoint uses");
1186       return false;
1187     }
1188   }
1189 
1190   if (!scalar_replacement(alloc, safepoints)) {
1191     return false;
1192   }
1193 
1194   CompileLog* log = C->log();
1195   if (log != NULL) {
1196     log->head("eliminate_allocation type='%d'",
1197               log->identify(tklass->klass()));
1198     JVMState* p = alloc->jvms();
1199     while (p != NULL) {
1200       log->elem("jvms bci='%d' method='%d'", p->bci(), log->identify(p->method()));
1201       p = p->caller();
1202     }
1203     log->tail("eliminate_allocation");
1204   }
1205 
1206   process_users_of_allocation(alloc, inline_alloc);
1207 
1208 #ifndef PRODUCT
1209   if (PrintEliminateAllocations) {
1210     if (alloc->is_AllocateArray())
1211       tty->print_cr("++++ Eliminated: %d AllocateArray", alloc->_idx);
1212     else
1213       tty->print_cr("++++ Eliminated: %d Allocate", alloc->_idx);
1214   }
1215 #endif
1216 
1217   return true;
1218 }

1294 //=============================================================================
1295 //
1296 //                              A L L O C A T I O N
1297 //
1298 // Allocation attempts to be fast in the case of frequent small objects.
1299 // It breaks down like this:
1300 //
1301 // 1) Size in doublewords is computed.  This is a constant for objects and
1302 // variable for most arrays.  Doubleword units are used to avoid size
1303 // overflow of huge doubleword arrays.  We need doublewords in the end for
1304 // rounding.
1305 //
1306 // 2) Size is checked for being 'too large'.  Too-large allocations will go
1307 // the slow path into the VM.  The slow path can throw any required
1308 // exceptions, and does all the special checks for very large arrays.  The
1309 // size test can constant-fold away for objects.  For objects with
1310 // finalizers it constant-folds the otherway: you always go slow with
1311 // finalizers.
1312 //
1313 // 3) If NOT using TLABs, this is the contended loop-back point.
1314 // Load-Locked the heap top.  If using TLABs normal-load the heap top.
1315 //
1316 // 4) Check that heap top + size*8 < max.  If we fail go the slow ` route.
1317 // NOTE: "top+size*8" cannot wrap the 4Gig line!  Here's why: for largish
1318 // "size*8" we always enter the VM, where "largish" is a constant picked small
1319 // enough that there's always space between the eden max and 4Gig (old space is
1320 // there so it's quite large) and large enough that the cost of entering the VM
1321 // is dwarfed by the cost to initialize the space.
1322 //
1323 // 5) If NOT using TLABs, Store-Conditional the adjusted heap top back
1324 // down.  If contended, repeat at step 3.  If using TLABs normal-store
1325 // adjusted heap top back down; there is no contention.
1326 //
1327 // 6) If !ZeroTLAB then Bulk-clear the object/array.  Fill in klass & mark
1328 // fields.
1329 //
1330 // 7) Merge with the slow-path; cast the raw memory pointer to the correct
1331 // oop flavor.
1332 //
1333 //=============================================================================
1334 // FastAllocateSizeLimit value is in DOUBLEWORDS.
1335 // Allocations bigger than this always go the slow route.
1336 // This value must be small enough that allocation attempts that need to
1337 // trigger exceptions go the slow route.  Also, it must be small enough so
1338 // that heap_top + size_in_bytes does not wrap around the 4Gig limit.
1339 //=============================================================================j//
1340 // %%% Here is an old comment from parseHelper.cpp; is it outdated?
1341 // The allocator will coalesce int->oop copies away.  See comment in
1342 // coalesce.cpp about how this works.  It depends critically on the exact
1343 // code shape produced here, so if you are changing this code shape
1344 // make sure the GC info for the heap-top is correct in and around the
1345 // slow-path call.
1346 //
1347 
1348 void PhaseMacroExpand::expand_allocate_common(
1349             AllocateNode* alloc, // allocation node to be expanded
1350             Node* length,  // array length for an array allocation
1351             const TypeFunc* slow_call_type, // Type of slow call
1352             address slow_call_address  // Address of slow call
1353     )
1354 {
1355   Node* ctrl = alloc->in(TypeFunc::Control);
1356   Node* mem  = alloc->in(TypeFunc::Memory);
1357   Node* i_o  = alloc->in(TypeFunc::I_O);
1358   Node* size_in_bytes     = alloc->in(AllocateNode::AllocSize);
1359   Node* klass_node        = alloc->in(AllocateNode::KlassNode);
1360   Node* initial_slow_test = alloc->in(AllocateNode::InitialTest);
1361   assert(ctrl != NULL, "must have control");
1362 
1363   // We need a Region and corresponding Phi's to merge the slow-path and fast-path results.
1364   // they will not be used if "always_slow" is set
1365   enum { slow_result_path = 1, fast_result_path = 2 };
1366   Node *result_region = NULL;
1367   Node *result_phi_rawmem = NULL;
1368   Node *result_phi_rawoop = NULL;
1369   Node *result_phi_i_o = NULL;
1370 
1371   // The initial slow comparison is a size check, the comparison
1372   // we want to do is a BoolTest::gt
1373   bool expand_fast_path = true;
1374   int tv = _igvn.find_int_con(initial_slow_test, -1);
1375   if (tv >= 0) {
1376     // InitialTest has constant result
1377     //   0 - can fit in TLAB
1378     //   1 - always too big or negative
1379     assert(tv <= 1, "0 or 1 if a constant");
1380     expand_fast_path = (tv == 0);
1381     initial_slow_test = NULL;
1382   } else {
1383     initial_slow_test = BoolNode::make_predicate(initial_slow_test, &_igvn);
1384   }
1385 
1386   if (C->env()->dtrace_alloc_probes() ||
1387       (!UseTLAB && !Universe::heap()->supports_inline_contig_alloc())) {
1388     // Force slow-path allocation
1389     expand_fast_path = false;
1390     initial_slow_test = NULL;
1391   }
1392 
1393   bool allocation_has_use = (alloc->result_cast() != NULL);
1394   if (!allocation_has_use) {
1395     InitializeNode* init = alloc->initialization();
1396     if (init != NULL) {
1397       init->remove(&_igvn);
1398     }
1399     if (expand_fast_path && (initial_slow_test == NULL)) {
1400       // Remove allocation node and return.
1401       // Size is a non-negative constant -> no initial check needed -> directly to fast path.
1402       // Also, no usages -> empty fast path -> no fall out to slow path -> nothing left.
1403 #ifndef PRODUCT
1404       if (PrintEliminateAllocations) {
1405         tty->print("NotUsed ");
1406         Node* res = alloc->proj_out_or_null(TypeFunc::Parms);
1407         if (res != NULL) {
1408           res->dump();
1409         } else {
1410           alloc->dump();
1411         }
1412       }
1413 #endif
1414       yank_alloc_node(alloc);
1415       return;
1416     }
1417   }
1418 
1419   enum { too_big_or_final_path = 1, need_gc_path = 2 };
1420   Node *slow_region = NULL;
1421   Node *toobig_false = ctrl;
1422 
1423   // generate the initial test if necessary
1424   if (initial_slow_test != NULL ) {
1425     assert (expand_fast_path, "Only need test if there is a fast path");
1426     slow_region = new RegionNode(3);
1427 
1428     // Now make the initial failure test.  Usually a too-big test but
1429     // might be a TRUE for finalizers or a fancy class check for
1430     // newInstance0.
1431     IfNode* toobig_iff = new IfNode(ctrl, initial_slow_test, PROB_MIN, COUNT_UNKNOWN);
1432     transform_later(toobig_iff);
1433     // Plug the failing-too-big test into the slow-path region
1434     Node* toobig_true = new IfTrueNode(toobig_iff);
1435     transform_later(toobig_true);
1436     slow_region    ->init_req( too_big_or_final_path, toobig_true );
1437     toobig_false = new IfFalseNode(toobig_iff);
1438     transform_later(toobig_false);
1439   } else {
1440     // No initial test, just fall into next case
1441     assert(allocation_has_use || !expand_fast_path, "Should already have been handled");
1442     toobig_false = ctrl;
1443     debug_only(slow_region = NodeSentinel);
1444   }
1445 
1446   // If we are here there are several possibilities
1447   // - expand_fast_path is false - then only a slow path is expanded. That's it.
1448   // no_initial_check means a constant allocation.
1449   // - If check always evaluates to false -> expand_fast_path is false (see above)
1450   // - If check always evaluates to true -> directly into fast path (but may bailout to slowpath)
1451   // if !allocation_has_use the fast path is empty
1452   // if !allocation_has_use && no_initial_check
1453   // - Then there are no fastpath that can fall out to slowpath -> no allocation code at all.
1454   //   removed by yank_alloc_node above.
1455 
1456   Node *slow_mem = mem;  // save the current memory state for slow path
1457   // generate the fast allocation code unless we know that the initial test will always go slow
1458   if (expand_fast_path) {
1459     // Fast path modifies only raw memory.
1460     if (mem->is_MergeMem()) {
1461       mem = mem->as_MergeMem()->memory_at(Compile::AliasIdxRaw);
1462     }
1463 
1464     // allocate the Region and Phi nodes for the result
1465     result_region = new RegionNode(3);
1466     result_phi_rawmem = new PhiNode(result_region, Type::MEMORY, TypeRawPtr::BOTTOM);
1467     result_phi_i_o    = new PhiNode(result_region, Type::ABIO); // I/O is used for Prefetch
1468 
1469     // Grab regular I/O before optional prefetch may change it.
1470     // Slow-path does no I/O so just set it to the original I/O.
1471     result_phi_i_o->init_req(slow_result_path, i_o);
1472 
1473     // Name successful fast-path variables
1474     Node* fast_oop_ctrl;
1475     Node* fast_oop_rawmem;
1476 
1477     if (allocation_has_use) {
1478       Node* needgc_ctrl = NULL;
1479       result_phi_rawoop = new PhiNode(result_region, TypeRawPtr::BOTTOM);
1480 
1481       intx prefetch_lines = length != NULL ? AllocatePrefetchLines : AllocateInstancePrefetchLines;
1482       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1483       Node* fast_oop = bs->obj_allocate(this, ctrl, mem, toobig_false, size_in_bytes, i_o, needgc_ctrl,
1484                                         fast_oop_ctrl, fast_oop_rawmem,
1485                                         prefetch_lines);
1486 
1487       if (initial_slow_test != NULL) {
1488         // This completes all paths into the slow merge point
1489         slow_region->init_req(need_gc_path, needgc_ctrl);
1490         transform_later(slow_region);
1491       } else {
1492         // No initial slow path needed!
1493         // Just fall from the need-GC path straight into the VM call.
1494         slow_region = needgc_ctrl;
1495       }
1496 
1497       InitializeNode* init = alloc->initialization();
1498       fast_oop_rawmem = initialize_object(alloc,
1499                                           fast_oop_ctrl, fast_oop_rawmem, fast_oop,
1500                                           klass_node, length, size_in_bytes);
1501       expand_initialize_membar(alloc, init, fast_oop_ctrl, fast_oop_rawmem);
1502       expand_dtrace_alloc_probe(alloc, fast_oop, fast_oop_ctrl, fast_oop_rawmem);
1503 
1504       result_phi_rawoop->init_req(fast_result_path, fast_oop);
1505     } else {
1506       assert (initial_slow_test != NULL, "sanity");
1507       fast_oop_ctrl   = toobig_false;
1508       fast_oop_rawmem = mem;
1509       transform_later(slow_region);
1510     }
1511 
1512     // Plug in the successful fast-path into the result merge point
1513     result_region    ->init_req(fast_result_path, fast_oop_ctrl);
1514     result_phi_i_o   ->init_req(fast_result_path, i_o);
1515     result_phi_rawmem->init_req(fast_result_path, fast_oop_rawmem);
1516   } else {
1517     slow_region = ctrl;
1518     result_phi_i_o = i_o; // Rename it to use in the following code.
1519   }
1520 
1521   // Generate slow-path call
1522   CallNode *call = new CallStaticJavaNode(slow_call_type, slow_call_address,
1523                                OptoRuntime::stub_name(slow_call_address),
1524                                alloc->jvms()->bci(),
1525                                TypePtr::BOTTOM);
1526   call->init_req(TypeFunc::Control,   slow_region);
1527   call->init_req(TypeFunc::I_O,       top());    // does no i/o
1528   call->init_req(TypeFunc::Memory,    slow_mem); // may gc ptrs
1529   call->init_req(TypeFunc::ReturnAdr, alloc->in(TypeFunc::ReturnAdr));
1530   call->init_req(TypeFunc::FramePtr,  alloc->in(TypeFunc::FramePtr));
1531 
1532   call->init_req(TypeFunc::Parms+0, klass_node);
1533   if (length != NULL) {
1534     call->init_req(TypeFunc::Parms+1, length);
1535   } else {
1536     // Let the runtime know if this is a larval allocation
1537     call->init_req(TypeFunc::Parms+1, _igvn.intcon(alloc->_larval));
1538   }
1539 
1540   // Copy debug information and adjust JVMState information, then replace
1541   // allocate node with the call
1542   call->copy_call_debug_info(&_igvn, alloc);
1543   if (expand_fast_path) {
1544     call->set_cnt(PROB_UNLIKELY_MAG(4));  // Same effect as RC_UNCOMMON.
1545   } else {
1546     // Hook i_o projection to avoid its elimination during allocation
1547     // replacement (when only a slow call is generated).
1548     call->set_req(TypeFunc::I_O, result_phi_i_o);
1549   }
1550   _igvn.replace_node(alloc, call);
1551   transform_later(call);
1552 
1553   // Identify the output projections from the allocate node and
1554   // adjust any references to them.
1555   // The control and io projections look like:
1556   //
1557   //        v---Proj(ctrl) <-----+   v---CatchProj(ctrl)
1558   //  Allocate                   Catch
1559   //        ^---Proj(io) <-------+   ^---CatchProj(io)
1560   //
1561   //  We are interested in the CatchProj nodes.
1562   //
1563   extract_call_projections(call);
1564 
1565   // An allocate node has separate memory projections for the uses on
1566   // the control and i_o paths. Replace the control memory projection with
1567   // result_phi_rawmem (unless we are only generating a slow call when
1568   // both memory projections are combined)
1569   if (expand_fast_path && _memproj_fallthrough != NULL) {
1570     _igvn.replace_in_uses(_memproj_fallthrough, result_phi_rawmem);
1571   }
1572   // Now change uses of _memproj_catchall to use _memproj_fallthrough and delete
1573   // _memproj_catchall so we end up with a call that has only 1 memory projection.
1574   if (_memproj_catchall != NULL) {
1575     if (_memproj_fallthrough == NULL) {
1576       _memproj_fallthrough = new ProjNode(call, TypeFunc::Memory);
1577       transform_later(_memproj_fallthrough);
1578     }
1579     _igvn.replace_in_uses(_memproj_catchall, _memproj_fallthrough);
1580     _igvn.remove_dead_node(_memproj_catchall);
1581   }
1582 
1583   // An allocate node has separate i_o projections for the uses on the control
1584   // and i_o paths. Always replace the control i_o projection with result i_o
1585   // otherwise incoming i_o become dead when only a slow call is generated
1586   // (it is different from memory projections where both projections are
1587   // combined in such case).
1588   if (_ioproj_fallthrough != NULL) {
1589     _igvn.replace_in_uses(_ioproj_fallthrough, result_phi_i_o);
1590   }
1591   // Now change uses of _ioproj_catchall to use _ioproj_fallthrough and delete
1592   // _ioproj_catchall so we end up with a call that has only 1 i_o projection.
1593   if (_ioproj_catchall != NULL) {
1594     if (_ioproj_fallthrough == NULL) {
1595       _ioproj_fallthrough = new ProjNode(call, TypeFunc::I_O);
1596       transform_later(_ioproj_fallthrough);
1597     }
1598     _igvn.replace_in_uses(_ioproj_catchall, _ioproj_fallthrough);
1599     _igvn.remove_dead_node(_ioproj_catchall);
1600   }
1601 
1602   // if we generated only a slow call, we are done
1603   if (!expand_fast_path) {
1604     // Now we can unhook i_o.
1605     if (result_phi_i_o->outcnt() > 1) {
1606       call->set_req(TypeFunc::I_O, top());
1607     } else {
1608       assert(result_phi_i_o->unique_ctrl_out() == call, "sanity");
1609       // Case of new array with negative size known during compilation.
1610       // AllocateArrayNode::Ideal() optimization disconnect unreachable
1611       // following code since call to runtime will throw exception.
1612       // As result there will be no users of i_o after the call.
1613       // Leave i_o attached to this call to avoid problems in preceding graph.
1614     }
1615     return;
1616   }
1617 
1618   if (_fallthroughcatchproj != NULL) {
1619     ctrl = _fallthroughcatchproj->clone();
1620     transform_later(ctrl);
1621     _igvn.replace_node(_fallthroughcatchproj, result_region);
1622   } else {
1623     ctrl = top();
1624   }
1625   Node *slow_result;
1626   if (_resproj == NULL) {
1627     // no uses of the allocation result
1628     slow_result = top();
1629   } else {
1630     slow_result = _resproj->clone();
1631     transform_later(slow_result);
1632     _igvn.replace_node(_resproj, result_phi_rawoop);
1633   }
1634 
1635   // Plug slow-path into result merge point
1636   result_region->init_req( slow_result_path, ctrl);
1637   transform_later(result_region);
1638   if (allocation_has_use) {
1639     result_phi_rawoop->init_req(slow_result_path, slow_result);
1640     transform_later(result_phi_rawoop);
1641   }
1642   result_phi_rawmem->init_req(slow_result_path, _memproj_fallthrough);
1643   transform_later(result_phi_rawmem);
1644   transform_later(result_phi_i_o);
1645   // This completes all paths into the result merge point
1646 }

1649 void PhaseMacroExpand::yank_alloc_node(AllocateNode* alloc) {
1650   Node* ctrl = alloc->in(TypeFunc::Control);
1651   Node* mem  = alloc->in(TypeFunc::Memory);
1652   Node* i_o  = alloc->in(TypeFunc::I_O);
1653 
1654   extract_call_projections(alloc);
1655   if (_resproj != NULL) {
1656     for (DUIterator_Fast imax, i = _resproj->fast_outs(imax); i < imax; i++) {
1657       Node* use = _resproj->fast_out(i);
1658       use->isa_MemBar()->remove(&_igvn);
1659       --imax;
1660       --i; // back up iterator
1661     }
1662     assert(_resproj->outcnt() == 0, "all uses must be deleted");
1663     _igvn.remove_dead_node(_resproj);
1664   }
1665   if (_fallthroughcatchproj != NULL) {
1666     _igvn.replace_in_uses(_fallthroughcatchproj, ctrl);
1667     _igvn.remove_dead_node(_fallthroughcatchproj);
1668   }
1669   if (_catchallcatchproj != NULL) {
1670     _igvn.rehash_node_delayed(_catchallcatchproj);
1671     _catchallcatchproj->set_req(0, top());
1672   }
1673   if (_fallthroughproj != NULL) {
1674     Node* catchnode = _fallthroughproj->unique_ctrl_out();
1675     _igvn.remove_dead_node(catchnode);
1676     _igvn.remove_dead_node(_fallthroughproj);
1677   }
1678   if (_memproj_fallthrough != NULL) {
1679     _igvn.replace_in_uses(_memproj_fallthrough, mem);
1680     _igvn.remove_dead_node(_memproj_fallthrough);
1681   }
1682   if (_ioproj_fallthrough != NULL) {
1683     _igvn.replace_in_uses(_ioproj_fallthrough, i_o);
1684     _igvn.remove_dead_node(_ioproj_fallthrough);
1685   }
1686   if (_memproj_catchall != NULL) {
1687     _igvn.rehash_node_delayed(_memproj_catchall);
1688     _memproj_catchall->set_req(0, top());
1689   }
1690   if (_ioproj_catchall != NULL) {
1691     _igvn.rehash_node_delayed(_ioproj_catchall);
1692     _ioproj_catchall->set_req(0, top());
1693   }
1694 #ifndef PRODUCT
1695   if (PrintEliminateAllocations) {
1696     if (alloc->is_AllocateArray()) {
1697       tty->print_cr("++++ Eliminated: %d AllocateArray", alloc->_idx);
1698     } else {
1699       tty->print_cr("++++ Eliminated: %d Allocate", alloc->_idx);
1700     }
1701   }
1702 #endif
1703   _igvn.remove_dead_node(alloc);
1704 }

1706 void PhaseMacroExpand::expand_initialize_membar(AllocateNode* alloc, InitializeNode* init,
1707                                                 Node*& fast_oop_ctrl, Node*& fast_oop_rawmem) {
1708   // If initialization is performed by an array copy, any required
1709   // MemBarStoreStore was already added. If the object does not
1710   // escape no need for a MemBarStoreStore. If the object does not
1711   // escape in its initializer and memory barrier (MemBarStoreStore or
1712   // stronger) is already added at exit of initializer, also no need
1713   // for a MemBarStoreStore. Otherwise we need a MemBarStoreStore
1714   // so that stores that initialize this object can't be reordered
1715   // with a subsequent store that makes this object accessible by
1716   // other threads.
1717   // Other threads include java threads and JVM internal threads
1718   // (for example concurrent GC threads). Current concurrent GC
1719   // implementation: G1 will not scan newly created object,
1720   // so it's safe to skip storestore barrier when allocation does
1721   // not escape.
1722   if (!alloc->does_not_escape_thread() &&
1723     !alloc->is_allocation_MemBar_redundant() &&
1724     (init == NULL || !init->is_complete_with_arraycopy())) {
1725     if (init == NULL || init->req() < InitializeNode::RawStores) {
1726       // No InitializeNode or no stores captured by zeroing
1727       // elimination. Simply add the MemBarStoreStore after object
1728       // initialization.
1729       MemBarNode* mb = MemBarNode::make(C, Op_MemBarStoreStore, Compile::AliasIdxBot);
1730       transform_later(mb);
1731 
1732       mb->init_req(TypeFunc::Memory, fast_oop_rawmem);
1733       mb->init_req(TypeFunc::Control, fast_oop_ctrl);
1734       fast_oop_ctrl = new ProjNode(mb, TypeFunc::Control);
1735       transform_later(fast_oop_ctrl);
1736       fast_oop_rawmem = new ProjNode(mb, TypeFunc::Memory);
1737       transform_later(fast_oop_rawmem);
1738     } else {
1739       // Add the MemBarStoreStore after the InitializeNode so that
1740       // all stores performing the initialization that were moved
1741       // before the InitializeNode happen before the storestore
1742       // barrier.
1743 
1744       Node* init_ctrl = init->proj_out_or_null(TypeFunc::Control);
1745       Node* init_mem = init->proj_out_or_null(TypeFunc::Memory);
1746 
1747       MemBarNode* mb = MemBarNode::make(C, Op_MemBarStoreStore, Compile::AliasIdxBot);
1748       transform_later(mb);
1749 
1750       Node* ctrl = new ProjNode(init, TypeFunc::Control);
1751       transform_later(ctrl);
1752       Node* mem = new ProjNode(init, TypeFunc::Memory);
1753       transform_later(mem);
1754 
1755       // The MemBarStoreStore depends on control and memory coming
1756       // from the InitializeNode
1757       mb->init_req(TypeFunc::Memory, mem);
1758       mb->init_req(TypeFunc::Control, ctrl);
1759 
1760       ctrl = new ProjNode(mb, TypeFunc::Control);
1761       transform_later(ctrl);
1762       mem = new ProjNode(mb, TypeFunc::Memory);
1763       transform_later(mem);
1764 
1765       // All nodes that depended on the InitializeNode for control
1766       // and memory must now depend on the MemBarNode that itself
1767       // depends on the InitializeNode
1768       if (init_ctrl != NULL) {
1769         _igvn.replace_node(init_ctrl, ctrl);
1770       }
1771       if (init_mem != NULL) {
1772         _igvn.replace_node(init_mem, mem);
1773       }
1774     }
1775   }
1776 }

1807 // Helper for PhaseMacroExpand::expand_allocate_common.
1808 // Initializes the newly-allocated storage.
1809 Node* PhaseMacroExpand::initialize_object(AllocateNode* alloc,
1810                                           Node* control, Node* rawmem, Node* object,
1811                                           Node* klass_node, Node* length,
1812                                           Node* size_in_bytes) {
1813   InitializeNode* init = alloc->initialization();
1814   // Store the klass & mark bits
1815   Node* mark_node = alloc->make_ideal_mark(&_igvn, control, rawmem);
1816   if (!mark_node->is_Con()) {
1817     transform_later(mark_node);
1818   }
1819   rawmem = make_store(control, rawmem, object, oopDesc::mark_offset_in_bytes(), mark_node, TypeX_X->basic_type());
1820 
1821   rawmem = make_store(control, rawmem, object, oopDesc::klass_offset_in_bytes(), klass_node, T_METADATA);
1822   int header_size = alloc->minimum_header_size();  // conservatively small
1823 
1824   // Array length
1825   if (length != NULL) {         // Arrays need length field
1826     rawmem = make_store(control, rawmem, object, arrayOopDesc::length_offset_in_bytes(), length, T_INT);
1827     // conservatively small header size:
1828     header_size = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1829     ciKlass* k = _igvn.type(klass_node)->is_klassptr()->klass();
1830     if (k->is_array_klass())    // we know the exact header size in most cases:
1831       header_size = Klass::layout_helper_header_size(k->layout_helper());
1832   }
1833 
1834   // Clear the object body, if necessary.
1835   if (init == NULL) {
1836     // The init has somehow disappeared; be cautious and clear everything.
1837     //
1838     // This can happen if a node is allocated but an uncommon trap occurs
1839     // immediately.  In this case, the Initialize gets associated with the
1840     // trap, and may be placed in a different (outer) loop, if the Allocate
1841     // is in a loop.  If (this is rare) the inner loop gets unrolled, then
1842     // there can be two Allocates to one Initialize.  The answer in all these
1843     // edge cases is safety first.  It is always safe to clear immediately
1844     // within an Allocate, and then (maybe or maybe not) clear some more later.
1845     if (!(UseTLAB && ZeroTLAB)) {
1846       rawmem = ClearArrayNode::clear_memory(control, rawmem, object,
1847                                             alloc->in(AllocateNode::DefaultValue),
1848                                             alloc->in(AllocateNode::RawDefaultValue),
1849                                             header_size, size_in_bytes,
1850                                             &_igvn);
1851     }
1852   } else {
1853     if (!init->is_complete()) {
1854       // Try to win by zeroing only what the init does not store.
1855       // We can also try to do some peephole optimizations,
1856       // such as combining some adjacent subword stores.
1857       rawmem = init->complete_stores(control, rawmem, object,
1858                                      header_size, size_in_bytes, &_igvn);
1859     }
1860     // We have no more use for this link, since the AllocateNode goes away:
1861     init->set_req(InitializeNode::RawAddress, top());
1862     // (If we keep the link, it just confuses the register allocator,
1863     // who thinks he sees a real use of the address by the membar.)
1864   }
1865 
1866   return rawmem;
1867 }

1869 // Generate prefetch instructions for next allocations.
1870 Node* PhaseMacroExpand::prefetch_allocation(Node* i_o, Node*& needgc_false,
1871                                         Node*& contended_phi_rawmem,
1872                                         Node* old_eden_top, Node* new_eden_top,
1873                                         intx lines) {
1874    enum { fall_in_path = 1, pf_path = 2 };
1875    if( UseTLAB && AllocatePrefetchStyle == 2 ) {
1876       // Generate prefetch allocation with watermark check.
1877       // As an allocation hits the watermark, we will prefetch starting
1878       // at a "distance" away from watermark.
1879 
1880       Node *pf_region = new RegionNode(3);
1881       Node *pf_phi_rawmem = new PhiNode( pf_region, Type::MEMORY,
1882                                                 TypeRawPtr::BOTTOM );
1883       // I/O is used for Prefetch
1884       Node *pf_phi_abio = new PhiNode( pf_region, Type::ABIO );
1885 
1886       Node *thread = new ThreadLocalNode();
1887       transform_later(thread);
1888 
1889       Node *eden_pf_adr = new AddPNode( top()/*not oop*/, thread,
1890                    _igvn.MakeConX(in_bytes(JavaThread::tlab_pf_top_offset())) );
1891       transform_later(eden_pf_adr);
1892 
1893       Node *old_pf_wm = new LoadPNode(needgc_false,
1894                                    contended_phi_rawmem, eden_pf_adr,
1895                                    TypeRawPtr::BOTTOM, TypeRawPtr::BOTTOM,
1896                                    MemNode::unordered);
1897       transform_later(old_pf_wm);
1898 
1899       // check against new_eden_top
1900       Node *need_pf_cmp = new CmpPNode( new_eden_top, old_pf_wm );
1901       transform_later(need_pf_cmp);
1902       Node *need_pf_bol = new BoolNode( need_pf_cmp, BoolTest::ge );
1903       transform_later(need_pf_bol);
1904       IfNode *need_pf_iff = new IfNode( needgc_false, need_pf_bol,
1905                                        PROB_UNLIKELY_MAG(4), COUNT_UNKNOWN );
1906       transform_later(need_pf_iff);
1907 
1908       // true node, add prefetchdistance
1909       Node *need_pf_true = new IfTrueNode( need_pf_iff );
1910       transform_later(need_pf_true);
1911 
1912       Node *need_pf_false = new IfFalseNode( need_pf_iff );
1913       transform_later(need_pf_false);
1914 
1915       Node *new_pf_wmt = new AddPNode( top(), old_pf_wm,
1916                                     _igvn.MakeConX(AllocatePrefetchDistance) );
1917       transform_later(new_pf_wmt );
1918       new_pf_wmt->set_req(0, need_pf_true);
1919 
1920       Node *store_new_wmt = new StorePNode(need_pf_true,
1921                                        contended_phi_rawmem, eden_pf_adr,
1922                                        TypeRawPtr::BOTTOM, new_pf_wmt,
1923                                        MemNode::unordered);
1924       transform_later(store_new_wmt);
1925 
1926       // adding prefetches
1927       pf_phi_abio->init_req( fall_in_path, i_o );
1928 
1929       Node *prefetch_adr;
1930       Node *prefetch;
1931       uint step_size = AllocatePrefetchStepSize;
1932       uint distance = 0;
1933 
1934       for ( intx i = 0; i < lines; i++ ) {
1935         prefetch_adr = new AddPNode( old_pf_wm, new_pf_wmt,
1936                                             _igvn.MakeConX(distance) );
1937         transform_later(prefetch_adr);
1938         prefetch = new PrefetchAllocationNode( i_o, prefetch_adr );
1939         transform_later(prefetch);
1940         distance += step_size;
1941         i_o = prefetch;
1942       }
1943       pf_phi_abio->set_req( pf_path, i_o );
1944 
1945       pf_region->init_req( fall_in_path, need_pf_false );
1946       pf_region->init_req( pf_path, need_pf_true );
1947 
1948       pf_phi_rawmem->init_req( fall_in_path, contended_phi_rawmem );
1949       pf_phi_rawmem->init_req( pf_path, store_new_wmt );
1950 
1951       transform_later(pf_region);
1952       transform_later(pf_phi_rawmem);
1953       transform_later(pf_phi_abio);
1954 
1955       needgc_false = pf_region;
1956       contended_phi_rawmem = pf_phi_rawmem;
1957       i_o = pf_phi_abio;
1958    } else if( UseTLAB && AllocatePrefetchStyle == 3 ) {
1959       // Insert a prefetch instruction for each allocation.
1960       // This code is used to generate 1 prefetch instruction per cache line.
1961 
1962       // Generate several prefetch instructions.
1963       uint step_size = AllocatePrefetchStepSize;
1964       uint distance = AllocatePrefetchDistance;
1965 
1966       // Next cache address.
1967       Node *cache_adr = new AddPNode(old_eden_top, old_eden_top,
1968                                      _igvn.MakeConX(step_size + distance));
1969       transform_later(cache_adr);
1970       cache_adr = new CastP2XNode(needgc_false, cache_adr);
1971       transform_later(cache_adr);
1972       // Address is aligned to execute prefetch to the beginning of cache line size
1973       // (it is important when BIS instruction is used on SPARC as prefetch).
1974       Node* mask = _igvn.MakeConX(~(intptr_t)(step_size-1));
1975       cache_adr = new AndXNode(cache_adr, mask);
1976       transform_later(cache_adr);
1977       cache_adr = new CastX2PNode(cache_adr);
1978       transform_later(cache_adr);
1979 
1980       // Prefetch
1981       Node *prefetch = new PrefetchAllocationNode( contended_phi_rawmem, cache_adr );
1982       prefetch->set_req(0, needgc_false);
1983       transform_later(prefetch);
1984       contended_phi_rawmem = prefetch;
1985       Node *prefetch_adr;
1986       distance = step_size;
1987       for ( intx i = 1; i < lines; i++ ) {
1988         prefetch_adr = new AddPNode( cache_adr, cache_adr,
1989                                             _igvn.MakeConX(distance) );
1990         transform_later(prefetch_adr);
1991         prefetch = new PrefetchAllocationNode( contended_phi_rawmem, prefetch_adr );
1992         transform_later(prefetch);
1993         distance += step_size;
1994         contended_phi_rawmem = prefetch;
1995       }
1996    } else if( AllocatePrefetchStyle > 0 ) {
1997       // Insert a prefetch for each allocation only on the fast-path
1998       Node *prefetch_adr;
1999       Node *prefetch;
2000       // Generate several prefetch instructions.
2001       uint step_size = AllocatePrefetchStepSize;
2002       uint distance = AllocatePrefetchDistance;
2003       for ( intx i = 0; i < lines; i++ ) {
2004         prefetch_adr = new AddPNode( old_eden_top, new_eden_top,
2005                                             _igvn.MakeConX(distance) );
2006         transform_later(prefetch_adr);
2007         prefetch = new PrefetchAllocationNode( i_o, prefetch_adr );
2008         // Do not let it float too high, since if eden_top == eden_end,
2009         // both might be null.
2010         if( i == 0 ) { // Set control for first prefetch, next follows it
2011           prefetch->init_req(0, needgc_false);
2012         }
2013         transform_later(prefetch);
2014         distance += step_size;
2015         i_o = prefetch;
2016       }
2017    }
2018    return i_o;
2019 }

2047 //-------------------mark_eliminated_box----------------------------------
2048 //
2049 // During EA obj may point to several objects but after few ideal graph
2050 // transformations (CCP) it may point to only one non escaping object
2051 // (but still using phi), corresponding locks and unlocks will be marked
2052 // for elimination. Later obj could be replaced with a new node (new phi)
2053 // and which does not have escape information. And later after some graph
2054 // reshape other locks and unlocks (which were not marked for elimination
2055 // before) are connected to this new obj (phi) but they still will not be
2056 // marked for elimination since new obj has no escape information.
2057 // Mark all associated (same box and obj) lock and unlock nodes for
2058 // elimination if some of them marked already.
2059 void PhaseMacroExpand::mark_eliminated_box(Node* oldbox, Node* obj) {
2060   if (oldbox->as_BoxLock()->is_eliminated())
2061     return; // This BoxLock node was processed already.
2062 
2063   // New implementation (EliminateNestedLocks) has separate BoxLock
2064   // node for each locked region so mark all associated locks/unlocks as
2065   // eliminated even if different objects are referenced in one locked region
2066   // (for example, OSR compilation of nested loop inside locked scope).
2067   if (EliminateNestedLocks ||
2068       oldbox->as_BoxLock()->is_simple_lock_region(NULL, obj)) {
2069     // Box is used only in one lock region. Mark this box as eliminated.
2070     _igvn.hash_delete(oldbox);
2071     oldbox->as_BoxLock()->set_eliminated(); // This changes box's hash value
2072      _igvn.hash_insert(oldbox);
2073 
2074     for (uint i = 0; i < oldbox->outcnt(); i++) {
2075       Node* u = oldbox->raw_out(i);
2076       if (u->is_AbstractLock() && !u->as_AbstractLock()->is_non_esc_obj()) {
2077         AbstractLockNode* alock = u->as_AbstractLock();
2078         // Check lock's box since box could be referenced by Lock's debug info.
2079         if (alock->box_node() == oldbox) {
2080           // Mark eliminated all related locks and unlocks.
2081 #ifdef ASSERT

2096   // to indicate that all associated lock and unlock nodes are marked

2125     }

2141             _igvn.replace_input_of(u, j, newbox);
2142             next_edge = false;
2143           }
2144         }
2145       }
2146     }
2147     if (next_edge) i++;

2305 //------------------------------expand_lock_node----------------------
2306 void PhaseMacroExpand::expand_lock_node(LockNode *lock) {
2307 
2308   Node* ctrl = lock->in(TypeFunc::Control);
2309   Node* mem = lock->in(TypeFunc::Memory);
2310   Node* obj = lock->obj_node();
2311   Node* box = lock->box_node();
2312   Node* flock = lock->fastlock_node();
2313 
2314   assert(!box->as_BoxLock()->is_eliminated(), "sanity");
2315 
2316   // Make the merge point
2317   Node *region;
2318   Node *mem_phi;
2319   Node *slow_path;
2320 
2321   if (UseOptoBiasInlining) {
2322     /*
2323      *  See the full description in MacroAssembler::biased_locking_enter().
2324      *
2325      *  if( (mark_word & biased_lock_mask) == biased_lock_pattern ) {
2326      *    // The object is biased.
2327      *    proto_node = klass->prototype_header;
2328      *    o_node = thread | proto_node;
2329      *    x_node = o_node ^ mark_word;
2330      *    if( (x_node & ~age_mask) == 0 ) { // Biased to the current thread ?
2331      *      // Done.
2332      *    } else {
2333      *      if( (x_node & biased_lock_mask) != 0 ) {
2334      *        // The klass's prototype header is no longer biased.
2335      *        cas(&mark_word, mark_word, proto_node)
2336      *        goto cas_lock;
2337      *      } else {
2338      *        // The klass's prototype header is still biased.
2339      *        if( (x_node & epoch_mask) != 0 ) { // Expired epoch?
2340      *          old = mark_word;
2341      *          new = o_node;
2342      *        } else {
2343      *          // Different thread or anonymous biased.
2344      *          old = mark_word & (epoch_mask | age_mask | biased_lock_mask);
2345      *          new = thread | old;
2346      *        }
2347      *        // Try to rebias.
2348      *        if( cas(&mark_word, old, new) == 0 ) {
2349      *          // Done.
2350      *        } else {
2351      *          goto slow_path; // Failed.
2352      *        }
2353      *      }
2354      *    }
2355      *  } else {
2356      *    // The object is not biased.
2357      *    cas_lock:
2358      *    if( FastLock(obj) == 0 ) {
2359      *      // Done.
2360      *    } else {
2361      *      slow_path:
2362      *      OptoRuntime::complete_monitor_locking_Java(obj);
2363      *    }
2364      *  }
2365      */
2366

2368     // create a Phi for the memory state
2369     mem_phi = new PhiNode( region, Type::MEMORY, TypeRawPtr::BOTTOM);
2370 
2371     Node* fast_lock_region  = new RegionNode(3);
2372     Node* fast_lock_mem_phi = new PhiNode( fast_lock_region, Type::MEMORY, TypeRawPtr::BOTTOM);
2373 
2374     // First, check mark word for the biased lock pattern.
2375     Node* mark_node = make_load(ctrl, mem, obj, oopDesc::mark_offset_in_bytes(), TypeX_X, TypeX_X->basic_type());
2376 
2377     // Get fast path - mark word has the biased lock pattern.
2378     ctrl = opt_bits_test(ctrl, fast_lock_region, 1, mark_node,
2379                          markWord::biased_lock_mask_in_place,
2380                          markWord::biased_lock_pattern, true);
2381     // fast_lock_region->in(1) is set to slow path.
2382     fast_lock_mem_phi->init_req(1, mem);
2383 
2384     // Now check that the lock is biased to the current thread and has
2385     // the same epoch and bias as Klass::_prototype_header.
2386 
2387     // Special-case a fresh allocation to avoid building nodes:
2388     Node* klass_node = AllocateNode::Ideal_klass(obj, &_igvn);
2389     if (klass_node == NULL) {
2390       Node* k_adr = basic_plus_adr(obj, oopDesc::klass_offset_in_bytes());
2391       klass_node = transform_later(LoadKlassNode::make(_igvn, NULL, mem, k_adr, _igvn.type(k_adr)->is_ptr()));
2392 #ifdef _LP64
2393       if (UseCompressedClassPointers && klass_node->is_DecodeNKlass()) {
2394         assert(klass_node->in(1)->Opcode() == Op_LoadNKlass, "sanity");
2395         klass_node->in(1)->init_req(0, ctrl);
2396       } else
2397 #endif
2398       klass_node->init_req(0, ctrl);
2399     }
2400     Node *proto_node = make_load(ctrl, mem, klass_node, in_bytes(Klass::prototype_header_offset()), TypeX_X, TypeX_X->basic_type());
2401 
2402     Node* thread = transform_later(new ThreadLocalNode());

2404     Node* o_node = transform_later(new OrXNode(cast_thread, proto_node));
2405     Node* x_node = transform_later(new XorXNode(o_node, mark_node));
2406 
2407     // Get slow path - mark word does NOT match the value.
2408     STATIC_ASSERT(markWord::age_mask_in_place <= INT_MAX);
2409     Node* not_biased_ctrl =  opt_bits_test(ctrl, region, 3, x_node,
2410                                       (~(int)markWord::age_mask_in_place), 0);
2411     // region->in(3) is set to fast path - the object is biased to the current thread.
2412     mem_phi->init_req(3, mem);
2413

2416

2421                           markWord::biased_lock_mask_in_place, 0, true);

2424     // fast_lock_region->in(2) - the prototype header is no longer biased
2425     // and we have to revoke the bias on this object.
2426     // We are going to try to reset the mark of this object to the prototype
2427     // value and fall through to the CAS-based locking scheme.
2428     Node* adr = basic_plus_adr(obj, oopDesc::mark_offset_in_bytes());
2429     Node* cas = new StoreXConditionalNode(not_biased_ctrl, mem, adr,
2430                                           proto_node, mark_node);

2432     Node* proj = transform_later(new SCMemProjNode(cas));
2433     fast_lock_mem_phi->init_req(2, proj);
2434 
2435

2437     Node* rebiased_region  = new RegionNode(3);
2438     Node* old_phi = new PhiNode( rebiased_region, TypeX_X);
2439     Node* new_phi = new PhiNode( rebiased_region, TypeX_X);
2440 
2441     // Get slow path - mark word does NOT match epoch bits.
2442     Node* epoch_ctrl =  opt_bits_test(ctrl, rebiased_region, 1, x_node,
2443                                       markWord::epoch_mask_in_place, 0);
2444     // The epoch of the current bias is not valid, attempt to rebias the object
2445     // toward the current thread.
2446     rebiased_region->init_req(2, epoch_ctrl);
2447     old_phi->init_req(2, mark_node);
2448     new_phi->init_req(2, o_node);
2449

2451     // The epoch of the current bias is still valid but we know
2452     // nothing about the owner; it might be set or it might be clear.

2455                              markWord::epoch_mask_in_place);

2457     cast_thread = transform_later(new CastP2XNode(ctrl, thread));

2460     new_phi->init_req(1, new_mark);
2461

2463     transform_later(old_phi);
2464     transform_later(new_phi);
2465

2467     // If this fails we will go in to the runtime to revoke the object's bias.
2468     cas = new StoreXConditionalNode(rebiased_region, mem, adr, new_phi, old_phi);

2471

2473     not_biased_ctrl = opt_bits_test(rebiased_region, region, 4, cas, 0, 0);

2475     // region->in(4) is set to fast path - the object is rebiased to the current thread.

2480 
2481     slow_path->init_req(1, not_biased_ctrl); // Capture slow-control
2482     slow_mem->init_req(1, proj);

2485 
2486     transform_later(fast_lock_region);
2487     transform_later(fast_lock_mem_phi);
2488 
2489     // Get slow path - FastLock failed to lock the object.
2490     ctrl = opt_bits_test(fast_lock_region, region, 2, flock, 0, 0);
2491     mem_phi->init_req(2, fast_lock_mem_phi);
2492     // region->in(2) is set to fast path - the object is locked to the current thread.
2493

2495     slow_mem->init_req(2, fast_lock_mem_phi);
2496

2501

2505     mem_phi = new PhiNode( region, Type::MEMORY, TypeRawPtr::BOTTOM);
2506 
2507     // Optimize test; set region slot 2
2508     slow_path = opt_bits_test(ctrl, region, 2, flock, 0, 0);
2509     mem_phi->init_req(2, mem);
2510   }

2513   if (objptr->can_be_value_type()) {

2515     assert(EnableValhalla, "should only be used if value types are enabled");

2517     Node* value_mask = _igvn.MakeConX(markWord::always_locked_pattern);
2518     Node* is_value = _igvn.transform(new AndXNode(mark, value_mask));
2519     Node* cmp = _igvn.transform(new CmpXNode(is_value, value_mask));

2522

2524     address call_addr = SharedRuntime::uncommon_trap_blob()->entry_point();

2526     JVMState* jvms = lock->jvms();

2529

2531     unc->init_req(TypeFunc::I_O, lock->i_o());

2533     unc->init_req(TypeFunc::FramePtr,  lock->in(TypeFunc::FramePtr));

2573   _igvn.hash_delete(_fallthroughproj);

2577   transform_later(region);

2579

2897 //---------------------------eliminate_macro_nodes----------------------
2898 // Eliminate scalar replaced allocations and associated locks.
2899 void PhaseMacroExpand::eliminate_macro_nodes() {
2900   if (C->macro_count() == 0)
2901     return;
2902 
2903   // First, attempt to eliminate locks
2904   int cnt = C->macro_count();
2905   for (int i=0; i < cnt; i++) {
2906     Node *n = C->macro_node(i);
2907     if (n->is_AbstractLock()) { // Lock and Unlock nodes
2908       // Before elimination mark all associated (same box and obj)
2909       // lock and unlock nodes.
2910       mark_eliminated_locking_nodes(n->as_AbstractLock());
2911     }
2912   }
2913   bool progress = true;
2914   while (progress) {
2915     progress = false;
2916     for (int i = C->macro_count(); i > 0; i--) {
2917       Node * n = C->macro_node(i-1);
2918       bool success = false;
2919       debug_only(int old_macro_count = C->macro_count(););
2920       if (n->is_AbstractLock()) {
2921         success = eliminate_locking_node(n->as_AbstractLock());
2922       }
2923       assert(success == (C->macro_count() < old_macro_count), "elimination reduces macro count");
2924       progress = progress || success;
2925     }
2926   }
2927   // Next, attempt to eliminate allocations
2928   _has_locks = false;
2929   progress = true;
2930   while (progress) {
2931     progress = false;
2932     for (int i = C->macro_count(); i > 0; i--) {
2933       Node * n = C->macro_node(i-1);
2934       bool success = false;
2935       debug_only(int old_macro_count = C->macro_count(););
2936       switch (n->class_id()) {
2937       case Node::Class_Allocate:
2938       case Node::Class_AllocateArray:
2939         success = eliminate_allocate_node(n->as_Allocate());
2940         break;
2941       case Node::Class_CallStaticJava: {
2942         CallStaticJavaNode* call = n->as_CallStaticJava();
2943         if (!call->method()->is_method_handle_intrinsic()) {
2944           success = eliminate_boxing_node(n->as_CallStaticJava());
2945         }
2946         break;
2947       }
2948       case Node::Class_Lock:
2949       case Node::Class_Unlock:
2950         assert(!n->as_AbstractLock()->is_eliminated(), "sanity");
2951         _has_locks = true;
2952         break;
2953       case Node::Class_ArrayCopy:
2954         break;
2955       case Node::Class_OuterStripMinedLoop:
2956         break;
2957       case Node::Class_SubTypeCheck:
2958         break;
2959       default:
2960         assert(n->Opcode() == Op_LoopLimit ||
2961                n->Opcode() == Op_Opaque1   ||
2962                n->Opcode() == Op_Opaque2   ||
2963                n->Opcode() == Op_Opaque3   ||
2964                BarrierSet::barrier_set()->barrier_set_c2()->is_gc_barrier_node(n),
2965                "unknown node type in macro list");
2966       }
2967       assert(success == (C->macro_count() < old_macro_count), "elimination reduces macro count");
2968       progress = progress || success;
2969     }
2970   }
2971 }

2973 //------------------------------expand_macro_nodes----------------------
2974 //  Returns true if a failure occurred.
2975 bool PhaseMacroExpand::expand_macro_nodes() {
2976   // Last attempt to eliminate macro nodes.
2977   eliminate_macro_nodes();
2978 
2979   // Eliminate Opaque and LoopLimit nodes. Do it after all loop optimizations.
2980   bool progress = true;
2981   while (progress) {
2982     progress = false;
2983     for (int i = C->macro_count(); i > 0; i--) {
2984       Node* n = C->macro_node(i-1);
2985       bool success = false;
2986       debug_only(int old_macro_count = C->macro_count(););
2987       if (n->Opcode() == Op_LoopLimit) {
2988         // Remove it from macro list and put on IGVN worklist to optimize.
2989         C->remove_macro_node(n);
2990         _igvn._worklist.push(n);
2991         success = true;
2992       } else if (n->Opcode() == Op_CallStaticJava) {
2993         CallStaticJavaNode* call = n->as_CallStaticJava();
2994         if (!call->method()->is_method_handle_intrinsic()) {
2995           // Remove it from macro list and put on IGVN worklist to optimize.
2996           C->remove_macro_node(n);
2997           _igvn._worklist.push(n);
2998           success = true;
2999         }
3000       } else if (n->Opcode() == Op_Opaque1 || n->Opcode() == Op_Opaque2) {
3001         _igvn.replace_node(n, n->in(1));
3002         success = true;
3003 #if INCLUDE_RTM_OPT
3004       } else if ((n->Opcode() == Op_Opaque3) && ((Opaque3Node*)n)->rtm_opt()) {
3005         assert(C->profile_rtm(), "should be used only in rtm deoptimization code");
3006         assert((n->outcnt() == 1) && n->unique_out()->is_Cmp(), "");
3007         Node* cmp = n->unique_out();
3008 #ifdef ASSERT
3009         // Validate graph.
3010         assert((cmp->outcnt() == 1) && cmp->unique_out()->is_Bool(), "");
3011         BoolNode* bol = cmp->unique_out()->as_Bool();
3012         assert((bol->outcnt() == 1) && bol->unique_out()->is_If() &&
3013                (bol->_test._test == BoolTest::ne), "");
3014         IfNode* ifn = bol->unique_out()->as_If();
3015         assert((ifn->outcnt() == 2) &&
3016                ifn->proj_out(1)->is_uncommon_trap_proj(Deoptimization::Reason_rtm_state_change) != NULL, "");
3017 #endif
3018         Node* repl = n->in(1);
3019         if (!_has_locks) {
3020           // Remove RTM state check if there are no locks in the code.
3021           // Replace input to compare the same value.
3022           repl = (cmp->in(1) == n) ? cmp->in(2) : cmp->in(1);
3023         }
3024         _igvn.replace_node(n, repl);
3025         success = true;
3026 #endif
3027       } else if (n->Opcode() == Op_OuterStripMinedLoop) {
3028         n->as_OuterStripMinedLoop()->adjust_strip_mined_loop(&_igvn);
3029         C->remove_macro_node(n);
3030         success = true;
3031       }
3032       assert(!success || (C->macro_count() == (old_macro_count - 1)), "elimination must have deleted one node from macro list");
3033       progress = progress || success;
3034     }
3035   }
3036 
3037   // Clean up the graph so we're less likely to hit the maximum node
3038   // limit
3039   _igvn.set_delay_transform(false);
3040   _igvn.optimize();
3041   if (C->failing())  return true;
3042   _igvn.set_delay_transform(true);
3043 
3044
