  96 // Return the index at which m must be inserted (or already exists).
  97 // The sort order is by the address of the ciMethod, with is_virtual as minor key.
  98 class IntrinsicDescPair {
  99  private:
 100   ciMethod* _m;
 101   bool _is_virtual;
 102  public:
 103   IntrinsicDescPair(ciMethod* m, bool is_virtual) : _m(m), _is_virtual(is_virtual) {}
 104   static int compare(IntrinsicDescPair* const& key, CallGenerator* const& elt) {
 105     ciMethod* m= elt->method();
 106     ciMethod* key_m = key->_m;
 107     if (key_m < m)      return -1;
 108     else if (key_m > m) return 1;
 109     else {
 110       bool is_virtual = elt->is_virtual();
 111       bool key_virtual = key->_is_virtual;
 112       if (key_virtual < is_virtual)      return -1;
 113       else if (key_virtual > is_virtual) return 1;
 114       else                               return 0;
 115     }
 116   }
 117 };
 118 int Compile::intrinsic_insertion_index(ciMethod* m, bool is_virtual, bool& found) {
 119 #ifdef ASSERT
 120   for (int i = 1; i < _intrinsics->length(); i++) {
 121     CallGenerator* cg1 = _intrinsics->at(i-1);
 122     CallGenerator* cg2 = _intrinsics->at(i);
 123     assert(cg1->method() != cg2->method()
 124            ? cg1->method()     < cg2->method()
 125            : cg1->is_virtual() < cg2->is_virtual(),
 126            "compiler intrinsics list must stay sorted");
 127   }
 128 #endif
 129   IntrinsicDescPair pair(m, is_virtual);
 130   return _intrinsics->find_sorted<IntrinsicDescPair*, IntrinsicDescPair::compare>(&pair, found);
 131 }

 133 void Compile::register_intrinsic(CallGenerator* cg) {
 134   if (_intrinsics == NULL) {
 135     _intrinsics = new (comp_arena())GrowableArray<CallGenerator*>(comp_arena(), 60, 0, NULL);
 136   }
 137   int len = _intrinsics->length();
 138   bool found = false;
 139   int index = intrinsic_insertion_index(cg->method(), cg->is_virtual(), found);
 140   assert(!found, "registering twice");
 141   _intrinsics->insert_before(index, cg);
 142   assert(find_intrinsic(cg->method(), cg->is_virtual()) == cg, "registration worked");
 143 }

 145 CallGenerator* Compile::find_intrinsic(ciMethod* m, bool is_virtual) {
 146   assert(m->is_loaded(), "don't try this on unloaded methods");
 147   if (_intrinsics != NULL) {
 148     bool found = false;
 149     int index = intrinsic_insertion_index(m, is_virtual, found);
 150      if (found) {
 151       return _intrinsics->at(index);
 152     }
 153   }
 154   // Lazily create intrinsics for intrinsic IDs well-known in the runtime.
 155   if (m->intrinsic_id() != vmIntrinsics::_none &&
 156       m->intrinsic_id() <= vmIntrinsics::LAST_COMPILER_INLINE) {
 157     CallGenerator* cg = make_vm_intrinsic(m, is_virtual);
 158     if (cg != NULL) {
 159       // Save it for next time:
 160       register_intrinsic(cg);
 161       return cg;
 162     } else {
 163       gather_intrinsic_statistics(m->intrinsic_id(), is_virtual, _intrinsic_disabled);
 164     }
 165   }
 166   return NULL;
 167 }

 176 juint  Compile::_intrinsic_hist_count[vmIntrinsics::ID_LIMIT] = {0};
 177 jubyte Compile::_intrinsic_hist_flags[vmIntrinsics::ID_LIMIT] = {0};

 179 bool Compile::gather_intrinsic_statistics(vmIntrinsics::ID id, bool is_virtual, int flags) {
 180   assert(id > vmIntrinsics::_none && id < vmIntrinsics::ID_LIMIT, "oob");
 181   int oflags = _intrinsic_hist_flags[id];
 182   assert(flags != 0, "what happened?");
 183   if (is_virtual) {
 184     flags |= _intrinsic_virtual;
 185   }
 186   bool changed = (flags != oflags);
 187   if ((flags & _intrinsic_worked) != 0) {
 188     juint count = (_intrinsic_hist_count[id] += 1);
 189     if (count == 1) {
 190       changed = true;           // first time
 191     }
 192     // increment the overall count also:
 193     _intrinsic_hist_count[vmIntrinsics::_none] += 1;
 194   }
 195   if (changed) {
 196     if (((oflags ^ flags) & _intrinsic_virtual) != 0) {
 197       // Something changed about the intrinsic's virtuality.
 198       if ((flags & _intrinsic_virtual) != 0) {
 199         // This is the first use of this intrinsic as a virtual call.
 200         if (oflags != 0) {
 201           // We already saw it as a non-virtual, so note both cases.
 202           flags |= _intrinsic_both;
 203         }
 204       } else if ((oflags & _intrinsic_both) == 0) {
 205         // This is the first use of this intrinsic as a non-virtual
 206         flags |= _intrinsic_both;
 207       }
 208     }
 209     _intrinsic_hist_flags[id] = (jubyte) (oflags | flags);
 210   }
 211   // update the overall flags also:
 212   _intrinsic_hist_flags[vmIntrinsics::_none] |= (jubyte) flags;
 213   return changed;
 214 }

 216 static char* format_flags(int flags, char* buf) {
 217   buf[0] = 0;
 218   if ((flags & Compile::_intrinsic_worked) != 0)    strcat(buf, ",worked");
 219   if ((flags & Compile::_intrinsic_failed) != 0)    strcat(buf, ",failed");
 220   if ((flags & Compile::_intrinsic_disabled) != 0)  strcat(buf, ",disabled");
 221   if ((flags & Compile::_intrinsic_virtual) != 0)   strcat(buf, ",virtual");
 222   if ((flags & Compile::_intrinsic_both) != 0)      strcat(buf, ",nonvirtual");
 223   if (buf[0] == 0)  strcat(buf, ",");
 224   assert(buf[0] == ',', "must be");
 225   return &buf[1];
 226 }

 267 void Compile::gvn_replace_by(Node* n, Node* nn) {
 268   for (DUIterator_Last imin, i = n->last_outs(imin); i >= imin; ) {
 269     Node* use = n->last_out(i);
 270     bool is_in_table = initial_gvn()->hash_delete(use);
 271     uint uses_found = 0;
 272     for (uint j = 0; j < use->len(); j++) {
 273       if (use->in(j) == n) {
 274         if (j < use->req())
 275           use->set_req(j, nn);
 276         else
 277           use->set_prec(j, nn);
 278         uses_found++;
 279       }
 280     }
 281     if (is_in_table) {
 282       // reinsert into table
 283       initial_gvn()->hash_find_insert(use);
 284     }
 285     record_for_igvn(use);
 286     i -= uses_found;    // we deleted 1 or more copies of this edge
 287   }
 288 }

 291 static inline bool not_a_node(const Node* n) {
 292   if (n == NULL)                   return true;
 293   if (((intptr_t)n & 1) != 0)      return true;  // uninitialized, etc.
 294   if (*(address*)n == badAddress)  return true;  // kill by Node::destruct
 295   return false;
 296 }

 298 // Identify all nodes that are reachable from below, useful.
 299 // Use breadth-first pass that records state in a Unique_Node_List,
 300 // recursive traversal is slower.
 301 void Compile::identify_useful_nodes(Unique_Node_List &useful) {
 302   int estimated_worklist_size = live_nodes();
 303   useful.map( estimated_worklist_size, NULL );  // preallocate space
 304 
 305   // Initialize worklist
 306   if (root() != NULL)     { useful.push(root()); }
 307   // If 'top' is cached, declare it useful to preserve cached node
 308   if( cached_top_node() ) { useful.push(cached_top_node()); }
 309 
 310   // Push all useful nodes onto the list, breadthfirst
 311   for( uint next = 0; next < useful.size(); ++next ) {
 312     assert( next < unique(), "Unique useful nodes < total nodes");
 313     Node *n  = useful.at(next);
 314     uint max = n->len();
 315     for( uint i = 0; i < max; ++i ) {
 316       Node *m = n->in(i);
 317       if (not_a_node(m))  continue;
 318       useful.push(m);
 319     }
 320   }
 321 }

 323 // Update dead_node_list with any missing dead nodes using useful
 324 // list. Consider all non-useful nodes to be useless i.e., dead nodes.
 325 void Compile::update_dead_node_list(Unique_Node_List &useful) {
 326   uint max_idx = unique();
 327   VectorSet& useful_node_set = useful.member_set();
 328 
 329   for (uint node_idx = 0; node_idx < max_idx; node_idx++) {
 330     // If node with index node_idx is not in useful set,
 331     // mark it as dead in dead node list.
 332     if (!useful_node_set.test(node_idx)) {
 333       record_dead_node(node_idx);
 334     }
 335   }
 336 }

 338 void Compile::remove_useless_late_inlines(GrowableArray<CallGenerator*>* inlines, Unique_Node_List &useful) {
 339   int shift = 0;
 340   for (int i = 0; i < inlines->length(); i++) {
 341     CallGenerator* cg = inlines->at(i);
 342     CallNode* call = cg->call_node();
 343     if (shift > 0) {
 344       inlines->at_put(i-shift, cg);
 345     }
 346     if (!useful.member(call)) {
 347       shift++;
 348     }
 349   }
 350   inlines->trunc_to(inlines->length()-shift);
 351 }

 353 // Disconnect all useless nodes by disconnecting those at the boundary.
 354 void Compile::remove_useless_nodes(Unique_Node_List &useful) {
 355   uint next = 0;
 356   while (next < useful.size()) {
 357     Node *n = useful.at(next++);
 358     if (n->is_SafePoint()) {
 359       // We're done with a parsing phase. Replaced nodes are not valid
 360       // beyond that point.
 361       n->as_SafePoint()->delete_replaced_nodes();
 362     }
 363     // Use raw traversal of out edges since this code removes out edges
 364     int max = n->outcnt();
 365     for (int j = 0; j < max; ++j) {
 366       Node* child = n->raw_out(j);
 367       if (! useful.member(child)) {
 368         assert(!child->is_top() || child != top(),
 369                "If top is cached in Compile object it is in useful list");
 370         // Only need to remove this out-edge to the useless node
 371         n->raw_del_out(j);
 372         --j;
 373         --max;
 374       }
 375     }
 376     if (n->outcnt() == 1 && n->has_special_unique_user()) {
 377       record_for_igvn(n->unique_out());
 378     }
 379   }
 380   // Remove useless macro and predicate opaq nodes
 381   for (int i = C->macro_count()-1; i >= 0; i--) {
 382     Node* n = C->macro_node(i);
 383     if (!useful.member(n)) {
 384       remove_macro_node(n);
 385     }
 386   }
 387   // Remove useless CastII nodes with range check dependency
 388   for (int i = range_check_cast_count() - 1; i >= 0; i--) {
 389     Node* cast = range_check_cast_node(i);
 390     if (!useful.member(cast)) {
 391       remove_range_check_cast(cast);
 392     }
 393   }
 394   // Remove useless expensive nodes
 395   for (int i = C->expensive_count()-1; i >= 0; i--) {
 396     Node* n = C->expensive_node(i);
 397     if (!useful.member(n)) {
 398       remove_expensive_node(n);
 399     }
 400   }
 401   // Remove useless Opaque4 nodes
 402   for (int i = opaque4_count() - 1; i >= 0; i--) {
 403     Node* opaq = opaque4_node(i);
 404     if (!useful.member(opaq)) {
 405       remove_opaque4_node(opaq);
 406     }
 407   }
 408   // Remove useless value type nodes
 409   for (int i = _value_type_nodes->length() - 1; i >= 0; i--) {
 410     Node* vt = _value_type_nodes->at(i);
 411     if (!useful.member(vt)) {
 412       _value_type_nodes->remove(vt);
 413     }
 414   }
 415   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 416   bs->eliminate_useless_gc_barriers(useful, this);
 417   // clean up the late inline lists
 418   remove_useless_late_inlines(&_string_late_inlines, useful);
 419   remove_useless_late_inlines(&_boxing_late_inlines, useful);
 420   remove_useless_late_inlines(&_late_inlines, useful);
 421   debug_only(verify_graph_edges(true/*check for no_dead_code*/);)
 422 }

 586 #if defined(SUPPORT_ASSEMBLY) || defined(SUPPORT_ABSTRACT_ASSEMBLY)
 587   bool print_opto_assembly = directive->PrintOptoAssemblyOption;
 588   // We can always print a disassembly, either abstract (hex dump) or
 589   // with the help of a suitable hsdis library. Thus, we should not
 590   // couple print_assembly and print_opto_assembly controls.
 591   // But: always print opto and regular assembly on compile command 'print'.
 592   bool print_assembly = directive->PrintAssemblyOption;
 593   set_print_assembly(print_opto_assembly || print_assembly);
 594 #else
 595   set_print_assembly(false); // must initialize.
 596 #endif
 597 
 598 #ifndef PRODUCT
 599   set_parsed_irreducible_loop(false);
 600 
 601   if (directive->ReplayInlineOption) {
 602     _replay_inline_data = ciReplay::load_inline_data(method(), entry_bci(), ci_env->comp_level());
 603   }
 604 #endif
 605   set_print_inlining(directive->PrintInliningOption || PrintOptoInlining);
 606   set_print_intrinsics(directive->PrintIntrinsicsOption);
 607   set_has_irreducible_loop(true); // conservative until build_loop_tree() reset it
 608 
 609   if (ProfileTraps RTM_OPT_ONLY( || UseRTMLocking )) {
 610     // Make sure the method being compiled gets its own MDO,
 611     // so we can at least track the decompile_count().
 612     // Need MDO to record RTM code generation state.
 613     method()->ensure_method_data();
 614   }
 615 
 616   Init(::AliasLevel);
 617 
 618 
 619   print_compile_messages();
 620 
 621   _ilt = InlineTree::build_inline_tree_root();
 622 
 623   // Even if NO memory addresses are used, MergeMem nodes must have at least 1 slice
 624   assert(num_alias_types() >= AliasIdxRaw, "");
 625 
 626 #define MINIMUM_NODE_HASH  1023
 627   // Node list that Iterative GVN will start with
 628   Unique_Node_List for_igvn(comp_arena());
 629   set_for_igvn(&for_igvn);
 630 
 631   // GVN that will be run immediately on new nodes
 632   uint estimated_size = method()->code_size()*4+64;
 633   estimated_size = (estimated_size < MINIMUM_NODE_HASH ? MINIMUM_NODE_HASH : estimated_size);
 634   PhaseGVN gvn(node_arena(), estimated_size);
 635   set_initial_gvn(&gvn);
 636 
 637   print_inlining_init();
 638   { // Scope for timing the parser
 639     TracePhase tp("parse", &timers[_t_parser]);
 640 
 641     // Put top into the hash table ASAP.
 642     initial_gvn()->transform_no_reclaim(top());
 643 
 644     // Set up tf(), start(), and find a CallGenerator.
 645     CallGenerator* cg = NULL;
 646     if (is_osr_compilation()) {
 647       init_tf(TypeFunc::make(method(), /* is_osr_compilation = */ true));
 648       StartNode* s = new StartOSRNode(root(), tf()->domain_sig());
 649       initial_gvn()->set_type_bottom(s);
 650       init_start(s);
 651       cg = CallGenerator::for_osr(method(), entry_bci());
 652     } else {
 653       // Normal case.
 654       init_tf(TypeFunc::make(method()));
 655       StartNode* s = new StartNode(root(), tf()->domain_cc());
 656       initial_gvn()->set_type_bottom(s);
 657       init_start(s);
 658       if (method()->intrinsic_id() == vmIntrinsics::_Reference_get) {
 659         // With java.lang.ref.reference.get() we must go through the
 660         // intrinsic - even when get() is the root
 661         // method of the compile - so that, if necessary, the value in
 662         // the referent field of the reference object gets recorded by
 663         // the pre-barrier code.
 664         cg = find_intrinsic(method(), false);
 665       }
 666       if (cg == NULL) {
 667         float past_uses = method()->interpreter_invocation_count();
 668         float expected_uses = past_uses;
 669         cg = CallGenerator::for_inline(method(), expected_uses);
 670       }
 671     }
 672     if (failing())  return;
 673     if (cg == NULL) {
 674       record_method_not_compilable("cannot parse method");
 675       return;
 676     }
 677     JVMState* jvms = build_start_state(start(), tf());
 678     if ((jvms = cg->generate(jvms)) == NULL) {
 679       if (!failure_reason_is(C2Compiler::retry_class_loading_during_parsing())) {
 680         record_method_not_compilable("method parse failed");
 681       }
 682       return;
 683     }
 684     GraphKit kit(jvms);
 685 
 686     if (!kit.stopped()) {
 687       // Accept return values, and transfer control we know not where.
 688       // This is done by a special, unique ReturnNode bound to root.
 689       return_values(kit.jvms());
 690     }
 691 
 692     if (kit.has_exceptions()) {
 693       // Any exceptions that escape from this call must be rethrown
 694       // to whatever caller is dynamically above us on the stack.
 695       // This is done by a special, unique RethrowNode bound to root.
 696       rethrow_exceptions(kit.transfer_exceptions_into_jvms());
 697     }
 698 
 699     assert(IncrementalInline || (_late_inlines.length() == 0 && !has_mh_late_inlines()), "incremental inlining is off");
 700 
 701     if (_late_inlines.length() == 0 && !has_mh_late_inlines() && !failing() && has_stringbuilder()) {
 702       inline_string_calls(true);
 703     }
 704 
 705     if (failing())  return;
 706 
 707     print_method(PHASE_BEFORE_REMOVEUSELESS, 3);
 708 
 709     // Remove clutter produced by parsing.
 710     if (!failing()) {
 711       ResourceMark rm;
 712       PhaseRemoveUseless pru(initial_gvn(), &for_igvn);
 713     }
 714   }
 715 
 716   // Note:  Large methods are capped off in do_one_bytecode().
 717   if (failing())  return;
 718 
 719   // After parsing, node notes are no longer automagic.
 720   // They must be propagated by register_new_node_with_optimizer(),
 721   // clone(), or the like.
 722   set_default_node_notes(NULL);
 723 
 724   for (;;) {
 725     int successes = Inline_Warm();
 726     if (failing())  return;
 727     if (successes == 0)  break;
 728   }
 729 
 730   // Drain the list.
 731   Finish_Warm();
 732 #ifndef PRODUCT
 733   if (_printer && _printer->should_print(1)) {
 734     _printer->print_inlining();
 735   }
 736 #endif
 737 
 738   if (failing())  return;
 739   NOT_PRODUCT( verify_graph_edges(); )
 740 
 741   // Now optimize
 742   Optimize();
 743   if (failing())  return;
 744   NOT_PRODUCT( verify_graph_edges(); )
 745 
 746 #ifndef PRODUCT
 747   if (print_ideal()) {
 748     ttyLocker ttyl;  // keep the following output all in one block
 749     // This output goes directly to the tty, not the compiler log.
 750     // To enable tools to match it up with the compilation activity,
 751     // be sure to tag this tty output with the compile ID.
 752     if (xtty != NULL) {
 753       xtty->head("ideal compile_id='%d'%s", compile_id(),
 754                  is_osr_compilation()    ? " compile_kind='osr'" :
 755                  "");
 756     }
 757     root()->dump(9999);
 758     if (xtty != NULL) {
 759       xtty->tail("ideal");
 760     }
 761   }
 762 #endif
 763 
 764 #ifdef ASSERT
 765   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 766   bs->verify_gc_barriers(this, BarrierSetC2::BeforeCodeGen);
 767 #endif
 768 
 769   // Dump compilation data to replay it.
 770   if (directive->DumpReplayOption) {
 771     env()->dump_replay_data(_compile_id);
 772   }
 773   if (directive->DumpInlineOption && (ilt() != NULL)) {
 774     env()->dump_inline_data(_compile_id);
 775   }
 776 
 777   // Now that we know the size of all the monitors we can add a fixed slot
 778   // for the original deopt pc.
 779   int next_slot = fixed_slots() + (sizeof(address) / VMRegImpl::stack_slot_size);
 780   if (needs_stack_repair()) {
 781     // One extra slot for the special stack increment value
 782     next_slot += 2;
 783   }
 784   set_fixed_slots(next_slot);
 785 
 786   // Compute when to use implicit null checks. Used by matching trap based
 787   // nodes and NullCheck optimization.
 788   set_allowed_deopt_reasons();
 789 
 790   // Now generate code
 791   Code_Gen();
 792 }

1040 /**
1041  * Return the 'StartNode'. We must not have a pending failure, since the ideal graph
1042  * can be in an inconsistent state, i.e., we can get segmentation faults when traversing
1043  * the ideal graph.
1044  */
1045 StartNode* Compile::start() const {
1046   assert (!failing(), "Must not have pending failure. Reason is: %s", failure_reason());
1047   for (DUIterator_Fast imax, i = root()->fast_outs(imax); i < imax; i++) {
1048     Node* start = root()->fast_out(i);
1049     if (start->is_Start()) {
1050       return start->as_Start();
1051     }
1052   }
1053   fatal("Did not find Start node!");
1054   return NULL;
1055 }

1072   return NULL;

1096 void Compile::print_missing_nodes() {
1097 
1098   // Return if CompileLog is NULL and PrintIdealNodeCount is false.
1099   if ((_log == NULL) && (! PrintIdealNodeCount)) {
1100     return;
1101   }
1102 
1103   // This is an expensive function. It is executed only when the user
1104   // specifies VerifyIdealNodeCount option or otherwise knows the
1105   // additional work that needs to be done to identify reachable nodes
1106   // by walking the flow graph and find the missing ones using
1107   // _dead_node_list.
1108 
1109   Unique_Node_List useful(comp_arena());
1110   // Get useful node list by walking the graph.
1111   identify_useful_nodes(useful);
1112 
1113   uint l_nodes = C->live_nodes();
1114   uint l_nodes_by_walk = useful.size();
1115 
1116   if (l_nodes != l_nodes_by_walk) {
1117     if (_log != NULL) {
1118       _log->begin_head("mismatched_nodes count='%d'", abs((int) (l_nodes - l_nodes_by_walk)));
1119       _log->stamp();
1120       _log->end_head();
1121     }
1122     VectorSet& useful_member_set = useful.member_set();
1123     int last_idx = l_nodes_by_walk;
1124     for (int i = 0; i < last_idx; i++) {
1125       if (useful_member_set.test(i)) {
1126         if (_dead_node_list.test(i)) {
1127           if (_log != NULL) {
1128             _log->elem("mismatched_node_info node_idx='%d' type='both live and dead'", i);
1129           }
1130           if (PrintIdealNodeCount) {
1131             // Print the log message to tty
1132               tty->print_cr("mismatched_node idx='%d' both live and dead'", i);
1133               useful.at(i)->dump();
1134           }
1135         }
1136       }
1137       else if (! _dead_node_list.test(i)) {
1138         if (_log != NULL) {
1139           _log->elem("mismatched_node_info node_idx='%d' type='neither live nor dead'", i);
1140         }

1242 //------------------------------flatten_alias_type-----------------------------
1243 const TypePtr *Compile::flatten_alias_type( const TypePtr *tj ) const {
1244   int offset = tj->offset();
1245   TypePtr::PTR ptr = tj->ptr();
1246 
1247   // Known instance (scalarizable allocation) alias only with itself.
1248   bool is_known_inst = tj->isa_oopptr() != NULL &&
1249                        tj->is_oopptr()->is_known_instance();
1250 
1251   // Process weird unsafe references.
1252   if (offset == Type::OffsetBot && (tj->isa_instptr() /*|| tj->isa_klassptr()*/)) {
1253     bool default_value_load = EnableValhalla && tj->is_instptr()->klass() == ciEnv::current()->Class_klass();
1254     assert(InlineUnsafeOps || default_value_load, "indeterminate pointers come only from unsafe ops");
1255     assert(!is_known_inst, "scalarizable allocation should not have unsafe references");
1256     tj = TypeOopPtr::BOTTOM;
1257     ptr = tj->ptr();
1258     offset = tj->offset();
1259   }
1260 
1261   // Array pointers need some flattening
1262   const TypeAryPtr *ta = tj->isa_aryptr();
1263   if (ta && ta->is_stable()) {
1264     // Erase stability property for alias analysis.
1265     tj = ta = ta->cast_to_stable(false);
1266   }
1267   if (ta && ta->is_not_flat()) {
1268     // Erase not flat property for alias analysis.
1269     tj = ta = ta->cast_to_not_flat(false);
1270   }
1271   if (ta && ta->is_not_null_free()) {
1272     // Erase not null free property for alias analysis.
1273     tj = ta = ta->cast_to_not_null_free(false);
1274   }
1275 
1276   if( ta && is_known_inst ) {
1277     if ( offset != Type::OffsetBot &&
1278          offset > arrayOopDesc::length_offset_in_bytes() ) {
1279       offset = Type::OffsetBot; // Flatten constant access into array body only
1280       tj = ta = TypeAryPtr::make(ptr, ta->ary(), ta->klass(), true, Type::Offset(offset), ta->field_offset(), ta->instance_id());
1281     }
1282   } else if( ta && _AliasLevel >= 2 ) {
1283     // For arrays indexed by constant indices, we flatten the alias
1284     // space to include all of the array body.  Only the header, klass
1285     // and array length can be accessed un-aliased.
1286     // For flattened value type array, each field has its own slice so
1287     // we must include the field offset.
1288     if( offset != Type::OffsetBot ) {
1289       if( ta->const_oop() ) { // MethodData* or Method*
1290         offset = Type::OffsetBot;   // Flatten constant access into array body
1291         tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),ta->ary(),ta->klass(),false,Type::Offset(offset), ta->field_offset());
1292       } else if( offset == arrayOopDesc::length_offset_in_bytes() ) {
1293         // range is OK as-is.
1294         tj = ta = TypeAryPtr::RANGE;
1295       } else if( offset == oopDesc::klass_offset_in_bytes() ) {
1296         tj = TypeInstPtr::KLASS; // all klass loads look alike
1297         ta = TypeAryPtr::RANGE; // generic ignored junk
1298         ptr = TypePtr::BotPTR;
1299       } else if( offset == oopDesc::mark_offset_in_bytes() ) {
1300         tj = TypeInstPtr::MARK;
1301         ta = TypeAryPtr::RANGE; // generic ignored junk
1302         ptr = TypePtr::BotPTR;
1303       } else {                  // Random constant offset into array body
1304         offset = Type::OffsetBot;   // Flatten constant access into array body
1305         tj = ta = TypeAryPtr::make(ptr,ta->ary(),ta->klass(),false,Type::Offset(offset), ta->field_offset());
1306       }
1307     }
1308     // Arrays of fixed size alias with arrays of unknown size.
1309     if (ta->size() != TypeInt::POS) {
1310       const TypeAry *tary = TypeAry::make(ta->elem(), TypeInt::POS);
1311       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,ta->klass(),false,Type::Offset(offset), ta->field_offset());
1312     }
1313     // Arrays of known objects become arrays of unknown objects.
1314     if (ta->elem()->isa_narrowoop() && ta->elem() != TypeNarrowOop::BOTTOM) {
1315       const TypeAry *tary = TypeAry::make(TypeNarrowOop::BOTTOM, ta->size());
1316       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,NULL,false,Type::Offset(offset), ta->field_offset());
1317     }
1318     if (ta->elem()->isa_oopptr() && ta->elem() != TypeInstPtr::BOTTOM) {
1319       const TypeAry *tary = TypeAry::make(TypeInstPtr::BOTTOM, ta->size());
1320       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,NULL,false,Type::Offset(offset), ta->field_offset());
1321     }
1322     // Initially all flattened array accesses share a single slice
1323     if (ta->elem()->isa_valuetype() && ta->elem() != TypeValueType::BOTTOM && _flattened_accesses_share_alias) {
1324       const TypeAry *tary = TypeAry::make(TypeValueType::BOTTOM, ta->size());
1325       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,NULL,false,Type::Offset(offset), Type::Offset(Type::OffsetBot));
1326     }
1327     // Arrays of bytes and of booleans both use 'bastore' and 'baload' so
1328     // cannot be distinguished by bytecode alone.
1329     if (ta->elem() == TypeInt::BOOL) {
1330       const TypeAry *tary = TypeAry::make(TypeInt::BYTE, ta->size());
1331       ciKlass* aklass = ciTypeArrayKlass::make(T_BYTE);
1332       tj = ta = TypeAryPtr::make(ptr,ta->const_oop(),tary,aklass,false,Type::Offset(offset), ta->field_offset());
1333     }
1334     // During the 2nd round of IterGVN, NotNull castings are removed.
1335     // Make sure the Bottom and NotNull variants alias the same.
1336     // Also, make sure exact and non-exact variants alias the same.
1337     if (ptr == TypePtr::NotNull || ta->klass_is_exact() || ta->speculative() != NULL) {
1338       tj = ta = TypeAryPtr::make(TypePtr::BotPTR,ta->ary(),ta->klass(),false,Type::Offset(offset), ta->field_offset());
1339     }
1340   }
1341 
1342   // Oop pointers need some flattening
1343   const TypeInstPtr *to = tj->isa_instptr();
1344   if( to && _AliasLevel >= 2 && to != TypeOopPtr::BOTTOM ) {
1345     ciInstanceKlass *k = to->klass()->as_instance_klass();
1346     if( ptr == TypePtr::Constant ) {
1347       if (to->klass() != ciEnv::current()->Class_klass() ||
1348           offset < k->size_helper() * wordSize) {
1349         // No constant oop pointers (such as Strings); they alias with
1350         // unknown strings.
1351         assert(!is_known_inst, "not scalarizable allocation");
1352         tj = to = TypeInstPtr::make(TypePtr::BotPTR,to->klass(),false,0,Type::Offset(offset), to->klass()->flatten_array());
1353       }
1354     } else if( is_known_inst ) {
1355       tj = to; // Keep NotNull and klass_is_exact for instance type
1356     } else if( ptr == TypePtr::NotNull || to->klass_is_exact() ) {
1357       // During the 2nd round of IterGVN, NotNull castings are removed.
1358       // Make sure the Bottom and NotNull variants alias the same.
1359       // Also, make sure exact and non-exact variants alias the same.
1360       tj = to = TypeInstPtr::make(TypePtr::BotPTR,to->klass(),false,0,Type::Offset(offset), to->klass()->flatten_array());
1361     }
1362     if (to->speculative() != NULL) {
1363       tj = to = TypeInstPtr::make(to->ptr(),to->klass(),to->klass_is_exact(),to->const_oop(),Type::Offset(to->offset()), to->klass()->flatten_array(), to->instance_id());
1364     }
1365     // Canonicalize the holder of this field
1366     if (offset >= 0 && offset < instanceOopDesc::base_offset_in_bytes()) {
1367       // First handle header references such as a LoadKlassNode, even if the
1368       // object's klass is unloaded at compile time (4965979).
1369       if (!is_known_inst) { // Do it only for non-instance types
1370         tj = to = TypeInstPtr::make(TypePtr::BotPTR, env()->Object_klass(), false, NULL, Type::Offset(offset), false);
1371       }
1372     } else if (offset < 0 || offset >= k->size_helper() * wordSize) {
1373       // Static fields are in the space above the normal instance
1374       // fields in the java.lang.Class instance.
1375       if (to->klass() != ciEnv::current()->Class_klass()) {
1376         to = NULL;
1377         tj = TypeOopPtr::BOTTOM;
1378         offset = tj->offset();
1379       }
1380     } else {
1381       ciInstanceKlass *canonical_holder = k->get_canonical_holder(offset);
1382       if (!k->equals(canonical_holder) || tj->offset() != offset) {
1383         if( is_known_inst ) {
1384           tj = to = TypeInstPtr::make(to->ptr(), canonical_holder, true, NULL, Type::Offset(offset), canonical_holder->flatten_array(), to->instance_id());
1385         } else {
1386           tj = to = TypeInstPtr::make(to->ptr(), canonical_holder, false, NULL, Type::Offset(offset), canonical_holder->flatten_array());
1387         }
1388       }
1389     }
1390   }
1391 
1392   // Klass pointers to object array klasses need some flattening
1393   const TypeKlassPtr *tk = tj->isa_klassptr();
1394   if( tk ) {
1395     // If we are referencing a field within a Klass, we need
1396     // to assume the worst case of an Object.  Both exact and
1397     // inexact types must flatten to the same alias class so
1398     // use NotNull as the PTR.
1399     if ( offset == Type::OffsetBot || (offset >= 0 && (size_t)offset < sizeof(Klass)) ) {
1400 
1401       tj = tk = TypeKlassPtr::make(TypePtr::NotNull,
1402                                    TypeKlassPtr::OBJECT->klass(),
1403                                    Type::Offset(offset),
1404                                    false);
1405     }
1406 
1407     ciKlass* klass = tk->klass();
1408     if (klass != NULL && klass->is_obj_array_klass()) {
1409       ciKlass* k = TypeAryPtr::OOPS->klass();
1410       if( !k || !k->is_loaded() )                  // Only fails for some -Xcomp runs
1411         k = TypeInstPtr::BOTTOM->klass();
1412       tj = tk = TypeKlassPtr::make(TypePtr::NotNull, k, Type::Offset(offset), false);
1413     }
1414 
1415     // Check for precise loads from the primary supertype array and force them
1416     // to the supertype cache alias index.  Check for generic array loads from
1417     // the primary supertype array and also force them to the supertype cache
1418     // alias index.  Since the same load can reach both, we need to merge
1419     // these 2 disparate memories into the same alias class.  Since the
1420     // primary supertype array is read-only, there's no chance of confusion
1421     // where we bypass an array load and an array store.
1422     int primary_supers_offset = in_bytes(Klass::primary_supers_offset());
1423     if (offset == Type::OffsetBot ||
1424         (offset >= primary_supers_offset &&
1425          offset < (int)(primary_supers_offset + Klass::primary_super_limit() * wordSize)) ||
1426         offset == (int)in_bytes(Klass::secondary_super_cache_offset())) {
1427       offset = in_bytes(Klass::secondary_super_cache_offset());
1428       tj = tk = TypeKlassPtr::make(TypePtr::NotNull, tk->klass(), Type::Offset(offset), tk->flat_array());
1429     }
1430   }
1431 
1432   // Flatten all Raw pointers together.
1433   if (tj->base() == Type::RawPtr)
1434     tj = TypeRawPtr::BOTTOM;
1435 
1436   if (tj->base() == Type::AnyPtr)
1437     tj = TypePtr::BOTTOM;      // An error, which the caller must check for.
1438 
1439   // Flatten all to bottom for now
1440   switch( _AliasLevel ) {
1441   case 0:
1442     tj = TypePtr::BOTTOM;
1443     break;
1444   case 1:                       // Flatten to: oop, static, field or array
1445     switch (tj->base()) {
1446     //case Type::AryPtr: tj = TypeAryPtr::RANGE;    break;
1447     case Type::RawPtr:   tj = TypeRawPtr::BOTTOM;   break;
1448     case Type::AryPtr:   // do not distinguish arrays at all
1449     case Type::InstPtr:  tj = TypeInstPtr::BOTTOM;  break;
1450     case Type::KlassPtr: tj = TypeKlassPtr::OBJECT; break;
1451     case Type::AnyPtr:   tj = TypePtr::BOTTOM;      break;  // caller checks it
1452     default: ShouldNotReachHere();
1453     }
1454     break;
1455   case 2:                       // No collapsing at level 2; keep all splits
1456   case 3:                       // No collapsing at level 3; keep all splits
1457     break;
1458   default:
1459     Unimplemented();
1460   }
1461 
1462   offset = tj->offset();
1463   assert( offset != Type::OffsetTop, "Offset has fallen from constant" );
1464 
1465   assert( (offset != Type::OffsetBot && tj->base() != Type::AryPtr) ||
1466           (offset == Type::OffsetBot && tj->base() == Type::AryPtr) ||
1467           (offset == Type::OffsetBot && tj == TypeOopPtr::BOTTOM) ||
1468           (offset == Type::OffsetBot && tj == TypePtr::BOTTOM) ||
1469           (offset == oopDesc::mark_offset_in_bytes() && tj->base() == Type::AryPtr) ||
1470           (offset == oopDesc::klass_offset_in_bytes() && tj->base() == Type::AryPtr) ||
1471           (offset == arrayOopDesc::length_offset_in_bytes() && tj->base() == Type::AryPtr),
1472           "For oops, klasses, raw offset must be constant; for arrays the offset is never known" );
1473   assert( tj->ptr() != TypePtr::TopPTR &&
1474           tj->ptr() != TypePtr::AnyNull &&
1475           tj->ptr() != TypePtr::Null, "No imprecise addresses" );
1476 //    assert( tj->ptr() != TypePtr::Constant ||
1477 //            tj->base() == Type::RawPtr ||
1478 //            tj->base() == Type::KlassPtr, "No constant oop addresses" );
1479 
1480   return tj;
1481 }

1566 //--------------------------------find_alias_type------------------------------
1567 Compile::AliasType* Compile::find_alias_type(const TypePtr* adr_type, bool no_create, ciField* original_field, bool uncached) {
1568   if (_AliasLevel == 0)
1569     return alias_type(AliasIdxBot);
1570 
1571   AliasCacheEntry* ace = NULL;
1572   if (!uncached) {
1573     ace = probe_alias_cache(adr_type);
1574     if (ace->_adr_type == adr_type) {
1575       return alias_type(ace->_index);
1576     }
1577   }
1578 
1579   // Handle special cases.
1580   if (adr_type == NULL)             return alias_type(AliasIdxTop);
1581   if (adr_type == TypePtr::BOTTOM)  return alias_type(AliasIdxBot);
1582 
1583   // Do it the slow way.
1584   const TypePtr* flat = flatten_alias_type(adr_type);
1585 
1586 #ifdef ASSERT
1587   {
1588     ResourceMark rm;
1589     assert(flat == flatten_alias_type(flat), "not idempotent: adr_type = %s; flat = %s => %s",
1590            Type::str(adr_type), Type::str(flat), Type::str(flatten_alias_type(flat)));
1591     assert(flat != TypePtr::BOTTOM, "cannot alias-analyze an untyped ptr: adr_type = %s",
1592            Type::str(adr_type));
1593     if (flat->isa_oopptr() && !flat->isa_klassptr()) {
1594       const TypeOopPtr* foop = flat->is_oopptr();
1595       // Scalarizable allocations have exact klass always.
1596       bool exact = !foop->klass_is_exact() || foop->is_known_instance();
1597       const TypePtr* xoop = foop->cast_to_exactness(exact)->is_ptr();
1598       assert(foop == flatten_alias_type(xoop), "exactness must not affect alias type: foop = %s; xoop = %s",
1599              Type::str(foop), Type::str(xoop));
1600     }
1601   }
1602 #endif
1603 
1604   int idx = AliasIdxTop;
1605   for (int i = 0; i < num_alias_types(); i++) {
1606     if (alias_type(i)->adr_type() == flat) {
1607       idx = i;
1608       break;
1609     }
1610   }
1611 
1612   if (idx == AliasIdxTop) {
1613     if (no_create)  return NULL;
1614     // Grow the array if necessary.
1615     if (_num_alias_types == _max_alias_types)  grow_alias_types();
1616     // Add a new alias type.
1617     idx = _num_alias_types++;
1618     _alias_types[idx]->Init(idx, flat);
1619     if (flat == TypeInstPtr::KLASS)  alias_type(idx)->set_rewritable(false);
1620     if (flat == TypeAryPtr::RANGE)   alias_type(idx)->set_rewritable(false);
1621     if (flat->isa_instptr()) {
1622       if (flat->offset() == java_lang_Class::klass_offset_in_bytes()
1623           && flat->is_instptr()->klass() == env()->Class_klass())
1624         alias_type(idx)->set_rewritable(false);
1625     }
1626     ciField* field = NULL;
1627     if (flat->isa_aryptr()) {
1628 #ifdef ASSERT
1629       const int header_size_min  = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1630       // (T_BYTE has the weakest alignment and size restrictions...)
1631       assert(flat->offset() < header_size_min, "array body reference must be OffsetBot");
1632 #endif
1633       const Type* elemtype = flat->is_aryptr()->elem();
1634       if (flat->offset() == TypePtr::OffsetBot) {
1635         alias_type(idx)->set_element(elemtype);
1636       }
1637       int field_offset = flat->is_aryptr()->field_offset().get();
1638       if (elemtype->isa_valuetype() &&
1639           elemtype->value_klass() != NULL &&
1640           field_offset != Type::OffsetBot) {
1641         ciValueKlass* vk = elemtype->value_klass();
1642         field_offset += vk->first_field_offset();
1643         field = vk->get_field_by_offset(field_offset, false);
1644       }
1645     }
1646     if (flat->isa_klassptr()) {
1647       if (flat->offset() == in_bytes(Klass::super_check_offset_offset()))
1648         alias_type(idx)->set_rewritable(false);
1649       if (flat->offset() == in_bytes(Klass::modifier_flags_offset()))
1650         alias_type(idx)->set_rewritable(false);
1651       if (flat->offset() == in_bytes(Klass::access_flags_offset()))
1652         alias_type(idx)->set_rewritable(false);
1653       if (flat->offset() == in_bytes(Klass::java_mirror_offset()))
1654         alias_type(idx)->set_rewritable(false);
1655       if (flat->offset() == in_bytes(Klass::layout_helper_offset()))
1656         alias_type(idx)->set_rewritable(false);
1657       if (flat->offset() == in_bytes(Klass::secondary_super_cache_offset()))
1658         alias_type(idx)->set_rewritable(false);
1659     }
1660     // %%% (We would like to finalize JavaThread::threadObj_offset(),
1661     // but the base pointer type is not distinctive enough to identify
1662     // references into JavaThread.)
1663 
1664     // Check for final fields.
1665     const TypeInstPtr* tinst = flat->isa_instptr();
1666     if (tinst && tinst->offset() >= instanceOopDesc::base_offset_in_bytes()) {
1667       if (tinst->const_oop() != NULL &&
1668           tinst->klass() == ciEnv::current()->Class_klass() &&
1669           tinst->offset() >= (tinst->klass()->as_instance_klass()->size_helper() * wordSize)) {
1670         // static field
1671         ciInstanceKlass* k = tinst->const_oop()->as_instance()->java_lang_Class_klass()->as_instance_klass();
1672         field = k->get_field_by_offset(tinst->offset(), true);
1673       } else if (tinst->klass()->is_valuetype()) {
1674         // Value type field
1675         ciValueKlass* vk = tinst->value_klass();
1676         field = vk->get_field_by_offset(tinst->offset(), false);
1677       } else {
1678         ciInstanceKlass* k = tinst->klass()->as_instance_klass();
1679         field = k->get_field_by_offset(tinst->offset(), false);
1680       }
1681     }
1682     assert(field == NULL ||
1683            original_field == NULL ||
1684            (field->holder() == original_field->holder() &&
1685             field->offset() == original_field->offset() &&
1686             field->is_static() == original_field->is_static()), "wrong field?");
1687     // Set field() and is_rewritable() attributes.
1688     if (field != NULL) {
1689       alias_type(idx)->set_field(field);
1690       if (flat->isa_aryptr()) {
1691         // Fields of flattened inline type arrays are rewritable although they are declared final
1692         assert(flat->is_aryptr()->elem()->isa_valuetype(), "must be a flattened value array");
1693         alias_type(idx)->set_rewritable(true);
1694       }
1695     }
1696   }
1697 
1698   // Fill the cache for next time.
1699   if (!uncached) {
1700     ace->_adr_type = adr_type;
1701     ace->_index    = idx;
1702     assert(alias_type(adr_type) == alias_type(idx),  "type must be installed");
1703 
1704     // Might as well try to fill the cache for the flattened version, too.
1705     AliasCacheEntry* face = probe_alias_cache(flat);
1706     if (face->_adr_type == NULL) {
1707       face->_adr_type = flat;
1708       face->_index    = idx;
1709       assert(alias_type(flat) == alias_type(idx), "flat type must work too");
1710     }
1711   }
1712 
1713   return alias_type(idx);
1714 }

1743 //-----------------------------must_alias--------------------------------------
1744 // True if all values of the given address type are in the given alias category.
1745 bool Compile::must_alias(const TypePtr* adr_type, int alias_idx) {
1746   if (alias_idx == AliasIdxBot)         return true;  // the universal category
1747   if (adr_type == NULL)                 return true;  // NULL serves as TypePtr::TOP
1748   if (alias_idx == AliasIdxTop)         return false; // the empty category
1749   if (adr_type->base() == Type::AnyPtr) return false; // TypePtr::BOTTOM or its twins
1750 
1751   // the only remaining possible overlap is identity
1752   int adr_idx = get_alias_index(adr_type);
1753   assert(adr_idx != AliasIdxBot && adr_idx != AliasIdxTop, "");
1754   assert(adr_idx == alias_idx ||
1755          (alias_type(alias_idx)->adr_type() != TypeOopPtr::BOTTOM
1756           && adr_type                       != TypeOopPtr::BOTTOM),
1757          "should not be testing for overlap with an unsafe pointer");
1758   return adr_idx == alias_idx;
1759 }

1761 //------------------------------can_alias--------------------------------------
1762 // True if any values of the given address type are in the given alias category.
1763 bool Compile::can_alias(const TypePtr* adr_type, int alias_idx) {
1764   if (alias_idx == AliasIdxTop)         return false; // the empty category
1765   if (adr_type == NULL)                 return false; // NULL serves as TypePtr::TOP
1766   // Known instance doesn't alias with bottom memory
1767   if (alias_idx == AliasIdxBot)         return !adr_type->is_known_instance();                   // the universal category
1768   if (adr_type->base() == Type::AnyPtr) return !C->get_adr_type(alias_idx)->is_known_instance(); // TypePtr::BOTTOM or its twins
1769 
1770   // the only remaining possible overlap is identity
1771   int adr_idx = get_alias_index(adr_type);
1772   assert(adr_idx != AliasIdxBot && adr_idx != AliasIdxTop, "");
1773   return adr_idx == alias_idx;
1774 }

1785 //----------------------------Inline_Warm--------------------------------------
1786 int Compile::Inline_Warm() {
1787   // If there is room, try to inline some more warm call sites.
1788   // %%% Do a graph index compaction pass when we think we're out of space?
1789   if (!InlineWarmCalls)  return 0;
1790 
1791   int calls_made_hot = 0;
1792   int room_to_grow   = NodeCountInliningCutoff - unique();
1793   int amount_to_grow = MIN2(room_to_grow, (int)NodeCountInliningStep);
1794   int amount_grown   = 0;
1795   WarmCallInfo* call;
1796   while (amount_to_grow > 0 && (call = pop_warm_call()) != NULL) {
1797     int est_size = (int)call->size();
1798     if (est_size > (room_to_grow - amount_grown)) {
1799       // This one won't fit anyway.  Get rid of it.
1800       call->make_cold();
1801       continue;
1802     }
1803     call->make_hot();
1804     calls_made_hot++;
1805     amount_grown   += est_size;
1806     amount_to_grow -= est_size;
1807   }
1808 
1809   if (calls_made_hot > 0)  set_major_progress();
1810   return calls_made_hot;
1811 }

1814 //----------------------------Finish_Warm--------------------------------------
1815 void Compile::Finish_Warm() {
1816   if (!InlineWarmCalls)  return;
1817   if (failing())  return;
1818   if (warm_calls() == NULL)  return;
1819 
1820   // Clean up loose ends, if we are out of space for inlining.
1821   WarmCallInfo* call;
1822   while ((call = pop_warm_call()) != NULL) {
1823     call->make_cold();
1824   }
1825 }

1886 // Does the return value keep otherwise useless value type allocations alive?

1887 static bool return_val_keeps_allocations_alive(Node* ret_val) {
1888   ResourceMark rm;
1889   Unique_Node_List wq;
1890   wq.push(ret_val);
1891   bool some_allocations = false;
1892   for (uint i = 0; i < wq.size(); i++) {
1893     Node* n = wq.at(i);
1894     assert(!n->is_ValueType(), "chain of value type nodes");
1895     if (n->outcnt() > 1) {
1896       // Some other use for the allocation
1897       return false;
1898     } else if (n->is_ValueTypePtr()) {
1899       wq.push(n->in(1));
1900     } else if (n->is_Phi()) {
1901       for (uint j = 1; j < n->req(); j++) {
1902         wq.push(n->in(j));
1903       }
1904     } else if (n->is_CheckCastPP() &&
1905                n->in(1)->is_Proj() &&
1906                n->in(1)->in(0)->is_Allocate()) {
1907       some_allocations = true;
1908     }
1909   }
1910   return some_allocations;
1911 }

1913 void Compile::process_value_types(PhaseIterGVN &igvn, bool post_ea) {
1914   // Make value types scalar in safepoints
1915   for (int i = _value_type_nodes->length()-1; i >= 0; i--) {
1916     ValueTypeBaseNode* vt = _value_type_nodes->at(i)->as_ValueTypeBase();
1917     vt->make_scalar_in_safepoints(&igvn);
1918   }
1919   // Remove ValueTypePtr nodes only after EA to give scalar replacement a chance
1920   // to remove buffer allocations. ValueType nodes are kept until loop opts and
1921   // removed via ValueTypeNode::remove_redundant_allocations.
1922   if (post_ea) {
1923     while (_value_type_nodes->length() > 0) {
1924       ValueTypeBaseNode* vt = _value_type_nodes->pop()->as_ValueTypeBase();
1925       if (vt->is_ValueTypePtr()) {
1926         igvn.replace_node(vt, vt->get_oop());
1927       }
1928     }
1929   }
1930   // Make sure that the return value does not keep an unused allocation alive
1931   if (tf()->returns_value_type_as_fields()) {
1932     Node* ret = NULL;
1933     for (uint i = 1; i < root()->req(); i++){
1934       Node* in = root()->in(i);
1935       if (in->Opcode() == Op_Return) {
1936         assert(ret == NULL, "only one return");
1937         ret = in;
1938       }
1939     }
1940     if (ret != NULL) {
1941       Node* ret_val = ret->in(TypeFunc::Parms);
1942       if (igvn.type(ret_val)->isa_oopptr() &&
1943           return_val_keeps_allocations_alive(ret_val)) {
1944         igvn.replace_input_of(ret, TypeFunc::Parms, ValueTypeNode::tagged_klass(igvn.type(ret_val)->value_klass(), igvn));
1945         assert(ret_val->outcnt() == 0, "should be dead now");
1946         igvn.remove_dead_node(ret_val);
1947       }
1948     }
1949   }
1950   igvn.optimize();
1951 }

1953 void Compile::adjust_flattened_array_access_aliases(PhaseIterGVN& igvn) {
1954   if (!_has_flattened_accesses) {
1955     return;
1956   }
1957   // Initially, all flattened array accesses share the same slice to
1958   // keep dependencies with Object[] array accesses (that could be
1959   // to a flattened array) correct. We're done with parsing so we
1960   // now know all flattened array accesses in this compile
1961   // unit. Let's move flattened array accesses to their own slice,
1962   // one per element field. This should help memory access
1963   // optimizations.
1964   ResourceMark rm;
1965   Unique_Node_List wq;
1966   wq.push(root());
1967 
1968   Node_List mergememnodes;
1969   Node_List memnodes;
1970 
1971   // Alias index currently shared by all flattened memory accesses
1972   int index = get_alias_index(TypeAryPtr::VALUES);
1973 
1974   // Find MergeMem nodes and flattened array accesses
1975   for (uint i = 0; i < wq.size(); i++) {
1976     Node* n = wq.at(i);
1977     if (n->is_Mem()) {
1978       const TypePtr* adr_type = NULL;
1979       if (n->Opcode() == Op_StoreCM) {
1980         adr_type = get_adr_type(get_alias_index(n->in(MemNode::OopStore)->adr_type()));
1981       } else {
1982         adr_type = get_adr_type(get_alias_index(n->adr_type()));
1983       }
1984       if (adr_type == TypeAryPtr::VALUES) {
1985         memnodes.push(n);
1986       }
1987     } else if (n->is_MergeMem()) {
1988       MergeMemNode* mm = n->as_MergeMem();
1989       if (mm->memory_at(index) != mm->base_memory()) {
1990         mergememnodes.push(n);
1991       }
1992     }
1993     for (uint j = 0; j < n->req(); j++) {
1994       Node* m = n->in(j);
1995       if (m != NULL) {
1996         wq.push(m);
1997       }
1998     }
1999   }
2000 
2001   if (memnodes.size() > 0) {
2002     _flattened_accesses_share_alias = false;
2003 
2004     // We are going to change the slice for the flattened array
2005     // accesses so we need to clear the cache entries that refer to
2006     // them.
2007     for (uint i = 0; i < AliasCacheSize; i++) {
2008       AliasCacheEntry* ace = &_alias_cache[i];
2009       if (ace->_adr_type != NULL &&
2010           ace->_adr_type->isa_aryptr() &&
2011           ace->_adr_type->is_aryptr()->elem()->isa_valuetype()) {
2012         ace->_adr_type = NULL;
2013         ace->_index = (i != 0) ? 0 : AliasIdxTop; // Make sure the NULL adr_type resolves to AliasIdxTop
2014       }
2015     }
2016 
2017     // Find what aliases we are going to add
2018     int start_alias = num_alias_types()-1;
2019     int stop_alias = 0;
2020 
2021     for (uint i = 0; i < memnodes.size(); i++) {
2022       Node* m = memnodes.at(i);
2023       const TypePtr* adr_type = NULL;
2024       if (m->Opcode() == Op_StoreCM) {
2025         adr_type = m->in(MemNode::OopStore)->adr_type();
2026         Node* clone = new StoreCMNode(m->in(MemNode::Control), m->in(MemNode::Memory), m->in(MemNode::Address),
2027                                       m->adr_type(), m->in(MemNode::ValueIn), m->in(MemNode::OopStore),
2028                                       get_alias_index(adr_type));
2029         igvn.register_new_node_with_optimizer(clone);
2030         igvn.replace_node(m, clone);
2031       } else {
2032         adr_type = m->adr_type();
2033 #ifdef ASSERT
2034         m->as_Mem()->set_adr_type(adr_type);
2035 #endif
2036       }
2037       int idx = get_alias_index(adr_type);
2038       start_alias = MIN2(start_alias, idx);
2039       stop_alias = MAX2(stop_alias, idx);
2040     }
2041 
2042     assert(stop_alias >= start_alias, "should have expanded aliases");
2043 
2044     Node_Stack stack(0);
2045 #ifdef ASSERT
2046     VectorSet seen(Thread::current()->resource_area());
2047 #endif
2048     // Now let's fix the memory graph so each flattened array access
2049     // is moved to the right slice. Start from the MergeMem nodes.
2050     uint last = unique();
2051     for (uint i = 0; i < mergememnodes.size(); i++) {
2052       MergeMemNode* current = mergememnodes.at(i)->as_MergeMem();
2053       Node* n = current->memory_at(index);
2054       MergeMemNode* mm = NULL;
2055       do {
2056         // Follow memory edges through memory accesses, phis and
2057         // narrow membars and push nodes on the stack. Once we hit
2058         // bottom memory, we pop element off the stack one at a
2059         // time, in reverse order, and move them to the right slice
2060         // by changing their memory edges.
2061         if ((n->is_Phi() && n->adr_type() != TypePtr::BOTTOM) || n->is_Mem() || n->adr_type() == TypeAryPtr::VALUES) {
2062           assert(!seen.test_set(n->_idx), "");
2063           // Uses (a load for instance) will need to be moved to the
2064           // right slice as well and will get a new memory state
2065           // that we don't know yet. The use could also be the
2066           // backedge of a loop. We put a place holder node between
2067           // the memory node and its uses. We replace that place
2068           // holder with the correct memory state once we know it,
2069           // i.e. when nodes are popped off the stack. Using the
2070           // place holder make the logic work in the presence of
2071           // loops.
2072           if (n->outcnt() > 1) {
2073             Node* place_holder = NULL;
2074             assert(!n->has_out_with(Op_Node), "");
2075             for (DUIterator k = n->outs(); n->has_out(k); k++) {
2076               Node* u = n->out(k);
2077               if (u != current && u->_idx < last) {
2078                 bool success = false;
2079                 for (uint l = 0; l < u->req(); l++) {
2080                   if (!stack.is_empty() && u == stack.node() && l == stack.index()) {
2081                     continue;
2082                   }
2083                   Node* in = u->in(l);
2084                   if (in == n) {
2085                     if (place_holder == NULL) {
2086                       place_holder = new Node(1);
2087                       place_holder->init_req(0, n);
2088                     }
2089                     igvn.replace_input_of(u, l, place_holder);
2090                     success = true;
2091                   }
2092                 }
2093                 if (success) {
2094                   --k;
2095                 }
2096               }
2097             }
2098           }
2099           if (n->is_Phi()) {
2100             stack.push(n, 1);
2101             n = n->in(1);
2102           } else if (n->is_Mem()) {
2103             stack.push(n, n->req());
2104             n = n->in(MemNode::Memory);
2105           } else {
2106             assert(n->is_Proj() && n->in(0)->Opcode() == Op_MemBarCPUOrder, "");
2107             stack.push(n, n->req());
2108             n = n->in(0)->in(TypeFunc::Memory);
2109           }
2110         } else {
2111           assert(n->adr_type() == TypePtr::BOTTOM || (n->Opcode() == Op_Node && n->_idx >= last) || (n->is_Proj() && n->in(0)->is_Initialize()), "");
2112           // Build a new MergeMem node to carry the new memory state
2113           // as we build it. IGVN should fold extraneous MergeMem
2114           // nodes.
2115           mm = MergeMemNode::make(n);
2116           igvn.register_new_node_with_optimizer(mm);
2117           while (stack.size() > 0) {
2118             Node* m = stack.node();
2119             uint idx = stack.index();
2120             if (m->is_Mem()) {
2121               // Move memory node to its new slice
2122               const TypePtr* adr_type = m->adr_type();
2123               int alias = get_alias_index(adr_type);
2124               Node* prev = mm->memory_at(alias);
2125               igvn.replace_input_of(m, MemNode::Memory, prev);
2126               mm->set_memory_at(alias, m);
2127             } else if (m->is_Phi()) {
2128               // We need as many new phis as there are new aliases
2129               igvn.replace_input_of(m, idx, mm);
2130               if (idx == m->req()-1) {
2131                 Node* r = m->in(0);
2132                 for (uint j = (uint)start_alias; j <= (uint)stop_alias; j++) {
2133                   const Type* adr_type = get_adr_type(j);
2134                   if (!adr_type->isa_aryptr() || !adr_type->is_aryptr()->elem()->isa_valuetype()) {
2135                     continue;
2136                   }
2137                   Node* phi = new PhiNode(r, Type::MEMORY, get_adr_type(j));
2138                   igvn.register_new_node_with_optimizer(phi);
2139                   for (uint k = 1; k < m->req(); k++) {
2140                     phi->init_req(k, m->in(k)->as_MergeMem()->memory_at(j));
2141                   }
2142                   mm->set_memory_at(j, phi);
2143                 }
2144                 Node* base_phi = new PhiNode(r, Type::MEMORY, TypePtr::BOTTOM);
2145                 igvn.register_new_node_with_optimizer(base_phi);
2146                 for (uint k = 1; k < m->req(); k++) {
2147                   base_phi->init_req(k, m->in(k)->as_MergeMem()->base_memory());
2148                 }
2149                 mm->set_base_memory(base_phi);
2150               }
2151             } else {

2715 static uint collect_unique_inputs(Node* n, Unique_Node_List& partition, Unique_Node_List& inputs) {
2716   uint cnt = 0;

2718     if (VectorNode::is_vector_bitwise_not_pattern(n)) {
2719       for (uint i = 1; i < n->req(); i++) {
2720         Node* in = n->in(i);
2721         bool skip = VectorNode::is_all_ones_vector(in);
2722         if (!skip && !inputs.member(in)) {
2723           inputs.push(in);
2724           cnt++;
2725         }
2726       }
2727       assert(cnt <= 1, "not unary");
2728     } else {
2729       uint last_req = n->req();
2730       if (is_vector_ternary_bitwise_op(n)) {
2731         last_req = n->req() - 1; // skip last input
2732       }
2733       for (uint i = 1; i < last_req; i++) {
2734         Node* def = n->in(i);
2735         if (!inputs.member(def)) {
2736           inputs.push(def);
2737           cnt++;
2738         }
2739       }
2740     }
2741     partition.push(n);
2742   } else { // not a bitwise operations
2743     if (!inputs.member(n)) {
2744       inputs.push(n);
2745       cnt++;
2746     }
2747   }

2783 //
2784 //  A macro logic node represents a truth table. It has 4 inputs,
2785 //  First three inputs corresponds to 3 columns of a truth table
2786 //  and fourth input captures the logic function.
2787 //
2788 //  eg.  fn = (in1 AND in2) OR in3;
2789 //
2790 //      MacroNode(in1,in2,in3,fn)
2791 //
2792 //  -----------------
2793 //  in1 in2 in3  fn
2794 //  -----------------
2795 //  0    0   0    0
2796 //  0    0   1    1
2797 //  0    1   0    0
2798 //  0    1   1    1
2799 //  1    0   0    0
2800 //  1    0   1    1
2801 //  1    1   0    1
2802 //  1    1   1    1
2803 //

2805 uint Compile::eval_macro_logic_op(uint func, uint in1 , uint in2, uint in3) {
2806   int res = 0;
2807   for (int i = 0; i < 8; i++) {
2808     int bit1 = extract_bit(in1, i);
2809     int bit2 = extract_bit(in2, i);
2810     int bit3 = extract_bit(in3, i);
2811 
2812     int func_bit_pos = (bit1 << 2 | bit2 << 1 | bit3);
2813     int func_bit = extract_bit(func, func_bit_pos);
2814 
2815     res |= func_bit << i;
2816   }
2817   return res;
2818 }

2842 uint Compile::compute_truth_table(Unique_Node_List& partition, Unique_Node_List& inputs) {
2843   assert(inputs.size() <= 3, "sanity");
2844   ResourceMark rm;
2845   uint res = 0;
2846   ResourceHashtable<Node*,uint> eval_map;
2847 
2848   // Populate precomputed functions for inputs.
2849   // Each input corresponds to one column of 3 input truth-table.
2850   uint input_funcs[] = { 0xAA,   // (_, _, a) -> a
2851                          0xCC,   // (_, b, _) -> b
2852                          0xF0 }; // (c, _, _) -> c
2853   for (uint i = 0; i < inputs.size(); i++) {
2854     eval_map.put(inputs.at(i), input_funcs[i]);
2855   }
2856 
2857   for (uint i = 0; i < partition.size(); i++) {
2858     Node* n = partition.at(i);
2859 
2860     uint func1 = 0, func2 = 0, func3 = 0;
2861     eval_operands(n, func1, func2, func3, eval_map);
2862 
2863     switch (n->Opcode()) {
2864       case Op_OrV:
2865         assert(func3 == 0, "not binary");
2866         res = func1 | func2;
2867         break;
2868       case Op_AndV:
2869         assert(func3 == 0, "not binary");
2870         res = func1 & func2;
2871         break;
2872       case Op_XorV:
2873         if (VectorNode::is_vector_bitwise_not_pattern(n)) {
2874           assert(func2 == 0 && func3 == 0, "not unary");
2875           res = (~func1) & 0xFF;
2876         } else {
2877           assert(func3 == 0, "not binary");
2878           res = func1 ^ func2;
2879         }
2880         break;
2881       case Op_MacroLogicV:
2882         // Ordering of inputs may change during evaluation of sub-tree
2883         // containing MacroLogic node as a child node, thus a re-evaluation
2884         // makes sure that function is evaluated in context of current
2885         // inputs.
2886         res = eval_macro_logic_op(n->in(4)->get_int(), func1, func2, func3);
2887         break;
2888 
2889       default: assert(false, "not supported: %s", n->Name());
2890     }
2891     assert(res <= 0xFF, "invalid");
2892     eval_map.put(n, res);
2893   }
2894   return res;
2895 }

2897 bool Compile::compute_logic_cone(Node* n, Unique_Node_List& partition, Unique_Node_List& inputs) {
2898   assert(partition.size() == 0, "not empty");
2899   assert(inputs.size() == 0, "not empty");
2900   if (is_vector_ternary_bitwise_op(n)) {
2901     return false;
2902   }
2903 
2904   bool is_unary_op = is_vector_unary_bitwise_op(n);
2905   if (is_unary_op) {
2906     assert(collect_unique_inputs(n, partition, inputs) == 1, "not unary");
2907     return false; // too few inputs
2908   }
2909 
2910   assert(is_vector_binary_bitwise_op(n), "not binary");
2911   Node* in1 = n->in(1);
2912   Node* in2 = n->in(2);
2913 
2914   int in1_unique_inputs_cnt = collect_unique_inputs(in1, partition, inputs);
2915   int in2_unique_inputs_cnt = collect_unique_inputs(in2, partition, inputs);
2916   partition.push(n);
2917 
2918   // Too many inputs?
2919   if (inputs.size() > 3) {
2920     partition.clear();
2921     inputs.clear();
2922     { // Recompute in2 inputs
2923       Unique_Node_List not_used;
2924       in2_unique_inputs_cnt = collect_unique_inputs(in2, not_used, not_used);
2925     }
2926     // Pick the node with minimum number of inputs.
2927     if (in1_unique_inputs_cnt >= 3 && in2_unique_inputs_cnt >= 3) {
2928       return false; // still too many inputs
2929     }
2930     // Recompute partition & inputs.
2931     Node* child       = (in1_unique_inputs_cnt < in2_unique_inputs_cnt ? in1 : in2);
2932     collect_unique_inputs(child, partition, inputs);
2933 
2934     Node* other_input = (in1_unique_inputs_cnt < in2_unique_inputs_cnt ? in2 : in1);
2935     inputs.push(other_input);
2936 
2937     partition.push(n);
2938   }
2939 
2940   return (partition.size() == 2 || partition.size() == 3) &&
2941          (inputs.size()    == 2 || inputs.size()    == 3);
2942 }

2945 void Compile::process_logic_cone_root(PhaseIterGVN &igvn, Node *n, VectorSet &visited) {
2946   assert(is_vector_bitwise_op(n), "not a root");
2947 
2948   visited.set(n->_idx);
2949 
2950   // 1) Do a DFS walk over the logic cone.
2951   for (uint i = 1; i < n->req(); i++) {
2952     Node* in = n->in(i);
2953     if (!visited.test(in->_idx) && is_vector_bitwise_op(in)) {
2954       process_logic_cone_root(igvn, in, visited);
2955     }
2956   }
2957 
2958   // 2) Bottom up traversal: Merge node[s] with
2959   // the parent to form macro logic node.
2960   Unique_Node_List partition;
2961   Unique_Node_List inputs;
2962   if (compute_logic_cone(n, partition, inputs)) {
2963     const TypeVect* vt = n->bottom_type()->is_vect();
2964     Node* macro_logic = xform_to_MacroLogicV(igvn, vt, partition, inputs);
2965     igvn.replace_node(n, macro_logic);
2966   }
2967 }

3139 // Eliminate trivially redundant StoreCMs and accumulate their
3140 // precedence edges.
3141 void Compile::eliminate_redundant_card_marks(Node* n) {
3142   assert(n->Opcode() == Op_StoreCM, "expected StoreCM");
3143   if (n->in(MemNode::Address)->outcnt() > 1) {
3144     // There are multiple users of the same address so it might be
3145     // possible to eliminate some of the StoreCMs
3146     Node* mem = n->in(MemNode::Memory);
3147     Node* adr = n->in(MemNode::Address);
3148     Node* val = n->in(MemNode::ValueIn);
3149     Node* prev = n;
3150     bool done = false;
3151     // Walk the chain of StoreCMs eliminating ones that match.  As
3152     // long as it's a chain of single users then the optimization is
3153     // safe.  Eliminating partially redundant StoreCMs would require
3154     // cloning copies down the other paths.
3155     while (mem->Opcode() == Op_StoreCM && mem->outcnt() == 1 && !done) {
3156       if (adr == mem->in(MemNode::Address) &&
3157           val == mem->in(MemNode::ValueIn)) {
3158         // redundant StoreCM
3159         if (mem->req() > MemNode::OopStore) {
3160           // Hasn't been processed by this code yet.
3161           n->add_prec(mem->in(MemNode::OopStore));
3162         } else {
3163           // Already converted to precedence edge
3164           for (uint i = mem->req(); i < mem->len(); i++) {
3165             // Accumulate any precedence edges
3166             if (mem->in(i) != NULL) {
3167               n->add_prec(mem->in(i));
3168             }
3169           }
3170           // Everything above this point has been processed.
3171           done = true;
3172         }
3173         // Eliminate the previous StoreCM
3174         prev->set_req(MemNode::Memory, mem->in(MemNode::Memory));
3175         assert(mem->outcnt() == 0, "should be dead");
3176         mem->disconnect_inputs(NULL, this);
3177       } else {
3178         prev = mem;
3179       }
3180       mem = prev->in(MemNode::Memory);
3181     }
3182   }
3183 }

3941 // Replacing Opaque nodes with their input in final_graph_reshaping_impl(),
3942 // requires that the walk visits a node's inputs before visiting the node.
3943 void Compile::final_graph_reshaping_walk( Node_Stack &nstack, Node *root, Final_Reshape_Counts &frc ) {
3944   ResourceArea *area = Thread::current()->resource_area();
3945   Unique_Node_List sfpt(area);
3946 
3947   frc._visited.set(root->_idx); // first, mark node as visited
3948   uint cnt = root->req();

3950   uint  i = 0;
3951   while (true) {
3952     if (i < cnt) {
3953       // Place all non-visited non-null inputs onto stack
3954       Node* m = n->in(i);
3955       ++i;
3956       if (m != NULL && !frc._visited.test_set(m->_idx)) {
3957         if (m->is_SafePoint() && m->as_SafePoint()->jvms() != NULL) {
3958           // compute worst case interpreter size in case of a deoptimization
3959           update_interpreter_frame_size(m->as_SafePoint()->jvms()->interpreter_frame_size());
3960 
3961           sfpt.push(m);
3962         }
3963         cnt = m->req();
3964         nstack.push(n, i); // put on stack parent and next input's index
3965         n = m;
3966         i = 0;
3967       }
3968     } else {
3969       // Now do post-visit work
3970       final_graph_reshaping_impl( n, frc );
3971       if (nstack.is_empty())
3972         break;             // finished
3973       n = nstack.node();   // Get node from stack
3974       cnt = n->req();

3977     }
3978   }
3979 
3980   // Skip next transformation if compressed oops are not used.
3981   if ((UseCompressedOops && !Matcher::gen_narrow_oop_implicit_null_checks()) ||
3982       (!UseCompressedOops && !UseCompressedClassPointers))
3983     return;

3986   // It could be done for an uncommon traps or any safepoints/calls
3987   // if the DecodeN/DecodeNKlass node is referenced only in a debug info.

3991     assert(jvms != NULL, "sanity");

4005                 (u->is_Call() && u->as_Call()->has_non_debug_use(n))) {
4006               safe_to_skip = false;
4007             }
4008           }
4009         }
4010         if (safe_to_skip) {
4011           n->set_req(j, in->in(1));
4012         }
4013         if (in->outcnt() == 0) {
4014           in->disconnect_inputs(NULL, this);
4015         }
4016       }
4017     }

4021 //------------------------------final_graph_reshaping--------------------------
4022 // Final Graph Reshaping.
4023 //
4024 // (1) Clone simple inputs to uncommon calls, so they can be scheduled late
4025 //     and not commoned up and forced early.  Must come after regular
4026 //     optimizations to avoid GVN undoing the cloning.  Clone constant
4027 //     inputs to Loop Phis; these will be split by the allocator anyways.
4028 //     Remove Opaque nodes.
4029 // (2) Move last-uses by commutative operations to the left input to encourage
4030 //     Intel update-in-place two-address operations and better register usage
4031 //     on RISCs.  Must come after regular optimizations to avoid GVN Ideal
4032 //     calls canonicalizing them back.
4033 // (3) Count the number of double-precision FP ops, single-precision FP ops
4034 //     and call sites.  On Intel, we can get correct rounding either by
4035 //     forcing singles to memory (requires extra stores and loads after each
4036 //     FP bytecode) or we can set a rounding mode bit (requires setting and
4037 //     clearing the mode bit around call sites).  The mode bit is only used
4038 //     if the relative frequency of single FP ops to calls is low enough.
4039 //     This is a key transform for SPEC mpeg_audio.
4040 // (4) Detect infinite loops; blobs of code reachable from above but not
4041 //     below.  Several of the Code_Gen algorithms fail on such code shapes,
4042 //     so we simply bail out.  Happens a lot in ZKM.jar, but also happens
4043 //     from time to time in other codes (such as -Xcomp finalizer loops, etc).
4044 //     Detection is by looking for IfNodes where only 1 projection is
4045 //     reachable from below or CatchNodes missing some targets.
4046 // (5) Assert for insane oop offsets in debug mode.
4047 
4048 bool Compile::final_graph_reshaping() {
4049   // an infinite loop may have been eliminated by the optimizer,
4050   // in which case the graph will be empty.
4051   if (root()->req() == 1) {
4052     record_method_not_compilable("trivial infinite loop");
4053     return true;
4054   }
4055 
4056   // Expensive nodes have their control input set to prevent the GVN
4057   // from freely commoning them. There's no GVN beyond this point so
4058   // no need to keep the control input. We want the expensive nodes to
4059   // be freely moved to the least frequent code path by gcm.
4060   assert(OptimizeExpensiveOps || expensive_count() == 0, "optimization off but list non empty?");
4061   for (int i = 0; i < expensive_count(); i++) {
4062     _expensive_nodes->at(i)->set_req(0, NULL);
4063   }
4064 
4065   Final_Reshape_Counts frc;
4066 
4067   // Visit everybody reachable!
4068   // Allocate stack of size C->live_nodes()/2 to avoid frequent realloc
4069   Node_Stack nstack(live_nodes() >> 1);
4070   final_graph_reshaping_walk(nstack, root(), frc);
4071 
4072   // Check for unreachable (from below) code (i.e., infinite loops).
4073   for( uint i = 0; i < frc._tests.size(); i++ ) {
4074     MultiBranchNode *n = frc._tests[i]->as_MultiBranch();
4075     // Get number of CFG targets.
4076     // Note that PCTables include exception targets after calls.
4077     uint required_outcnt = n->required_outcnt();
4078     if (n->outcnt() != required_outcnt) {
4079       // Check for a few special cases.  Rethrow Nodes never take the
4080       // 'fall-thru' path, so expected kids is 1 less.
4081       if (n->is_PCTable() && n->in(0) && n->in(0)->in(0)) {
4082         if (n->in(0)->in(0)->is_Call()) {
4083           CallNode *call = n->in(0)->in(0)->as_Call();
4084           if (call->entry_point() == OptoRuntime::rethrow_stub()) {
4085             required_outcnt--;      // Rethrow always has 1 less kid
4086           } else if (call->req() > TypeFunc::Parms &&
4087                      call->is_CallDynamicJava()) {
4088             // Check for null receiver. In such case, the optimizer has
4089             // detected that the virtual call will always result in a null
4090             // pointer exception. The fall-through projection of this CatchNode
4091             // will not be populated.
4092             Node *arg0 = call->in(TypeFunc::Parms);
4093             if (arg0->is_Type() &&
4094                 arg0->as_Type()->type()->higher_equal(TypePtr::NULL_PTR)) {
4095               required_outcnt--;
4096             }
4097           } else if (call->entry_point() == OptoRuntime::new_array_Java() &&
4098                      call->req() > TypeFunc::Parms+1 &&
4099                      call->is_CallStaticJava()) {
4100             // Check for negative array length. In such case, the optimizer has
4101             // detected that the allocation attempt will always result in an
4102             // exception. There is no fall-through projection of this CatchNode .
4103             Node *arg1 = call->in(TypeFunc::Parms+1);
4104             if (arg1->is_Type() &&
4105                 arg1->as_Type()->type()->join(TypeInt::POS)->empty()) {
4106               required_outcnt--;
4107             }
4108           }
4109         }
4110       }
4111       // Recheck with a better notion of 'required_outcnt'
4112       if (n->outcnt() != required_outcnt) {
4113         record_method_not_compilable("malformed control flow");
4114         return true;            // Not all targets reachable!
4115       }
4116     }
4117     // Check that I actually visited all kids.  Unreached kids
4118     // must be infinite loops.
4119     for (DUIterator_Fast jmax, j = n->fast_outs(jmax); j < jmax; j++)
4120       if (!frc._visited.test(n->fast_out(j)->_idx)) {
4121         record_method_not_compilable("infinite loop");
4122         return true;            // Found unvisited kid; must be unreach
4123       }
4124 
4125     // Here so verification code in final_graph_reshaping_walk()
4126     // always see an OuterStripMinedLoopEnd
4127     if (n->is_OuterStripMinedLoopEnd()) {
4128       IfNode* init_iff = n->as_If();
4129       Node* iff = new IfNode(init_iff->in(0), init_iff->in(1), init_iff->_prob, init_iff->_fcnt);
4130       n->subsume_by(iff, this);
4131     }
4132   }
4133 
4134 #ifdef IA32
4135   // If original bytecodes contained a mixture of floats and doubles
4136   // check if the optimizer has made it homogenous, item (3).
4137   if (UseSSE == 0 &&
4138       frc.get_float_count() > 32 &&
4139       frc.get_double_count() == 0 &&
4140       (10 * frc.get_call_count() < frc.get_float_count()) ) {
4141     set_24_bit_selection_and_mode(false, true);
4142   }
4143 #endif // IA32
4144 
4145   set_java_calls(frc.get_java_call_count());
4146   set_inner_loops(frc.get_inner_loop_count());
4147 
4148   // No infinite loops, no reason to bail out.
4149   return false;
4150 }

4200 // Report if there are too many recompiles at the current method and bci.

4203 // Action_none for a trap point, to avoid too many recompilations.

4205                                   int bci,
4206                                   Deoptimization::DeoptReason reason) {
4207   ciMethodData* md = method->method_data();
4208   if (md->is_empty()) {
4209     // Assume the trap has not occurred, or that it occurred only
4210     // because of a transient condition during start-up in the interpreter.
4211     return false;
4212   }
4213   // Pick a cutoff point well within PerBytecodeRecompilationCutoff.
4214   uint bc_cutoff = (uint) PerBytecodeRecompilationCutoff / 8;
4215   uint m_cutoff  = (uint) PerMethodRecompilationCutoff / 2 + 1;  // not zero
4216   Deoptimization::DeoptReason per_bc_reason
4217     = Deoptimization::reason_recorded_per_bytecode_if_any(reason);
4218   ciMethod* m = Deoptimization::reason_is_speculate(reason) ? this->method() : NULL;
4219   if ((per_bc_reason == Deoptimization::Reason_none
4220        || md->has_trap_at(bci, m, reason) != 0)
4221       // The trap frequency measure we care about is the recompile count:
4222       && md->trap_recompiled_at(bci, m)
4223       && md->overflow_recompile_count() >= bc_cutoff) {
4224     // Do not emit a trap here if it has already caused recompilations.
4225     // Also, if there are multiple reasons, or if there is no per-BCI record,
4226     // assume the worst.
4227     if (log())
4228       log()->elem("observe trap='%s recompiled' count='%d' recompiles2='%d'",
4229                   Deoptimization::trap_reason_name(reason),
4230                   md->trap_count(reason),
4231                   md->overflow_recompile_count());
4232     return true;
4233   } else if (trap_count(reason) != 0

4236     // Use cumulative decompile_count, not just md->decompile_count.
4237     if (log())
4238       log()->elem("observe trap='%s' count='%d' mcount='%d' decompiles='%d' mdecompiles='%d'",

4240                   md->trap_count(reason), trap_count(reason),
4241                   md->decompile_count(), decompile_count());

4243   } else {

4404 //----------------------------static_subtype_check-----------------------------
4405 // Shortcut important common cases when superklass is exact:
4406 // (0) superklass is java.lang.Object (can occur in reflective code)
4407 // (1) subklass is already limited to a subtype of superklass => always ok
4408 // (2) subklass does not overlap with superklass => always fail
4409 // (3) superklass has NO subtypes and we can check with a simple compare.
4410 int Compile::static_subtype_check(ciKlass* superk, ciKlass* subk) {
4411   if (StressReflectiveCode || superk == NULL || subk == NULL) {
4412     return SSC_full_test;       // Let caller generate the general case.
4413   }
4414 
4415   if (superk == env()->Object_klass()) {
4416     return SSC_always_true;     // (0) this test cannot fail
4417   }
4418 
4419   ciType* superelem = superk;
4420   if (superelem->is_array_klass()) {
4421     ciArrayKlass* ak = superelem->as_array_klass();
4422     superelem = superelem->as_array_klass()->base_element_type();
4423   }
4424 
4425   if (!subk->is_interface()) {  // cannot trust static interface types yet
4426     if (subk->is_subtype_of(superk)) {
4427       return SSC_always_true;   // (1) false path dead; no dynamic test needed
4428     }
4429     if (!(superelem->is_klass() && superelem->as_klass()->is_interface()) &&
4430         !superk->is_subtype_of(subk)) {
4431       return SSC_always_false;
4432     }
4433   }
4434 
4435   // If casting to an instance klass, it must have no subtypes
4436   if (superk->is_interface()) {

4733   }
4734

4750   // and check for at least two nodes of identical kind with same data

4755     if (cmp_expensive_nodes(_expensive_nodes->adr_at(i), _expensive_nodes->adr_at(i+1)) == 0) {

4764   if (_expensive_nodes->length() == 0) {
4765     return;
4766   }
4767 
4768   assert(OptimizeExpensiveOps, "optimization off?");
4769 
4770   // Sort to bring similar nodes next to each other and clear the
4771   // control input of nodes for which there's only a single copy.
4772   sort_expensive_nodes();
4773 
4774   int j = 0;
4775   int identical = 0;
4776   int i = 0;
4777   bool modified = false;
4778   for (; i < _expensive_nodes->length()-1; i++) {
4779     assert(j <= i, "can't write beyond current index");
4780     if (_expensive_nodes->at(i)->Opcode() == _expensive_nodes->at(i+1)->Opcode()) {
4781       identical++;

4783       continue;
4784     }
4785     if (identical > 0) {
4786       _expensive_nodes->at_put(j++, _expensive_nodes->at(i));

4790       igvn.replace_input_of(n, 0, NULL);

4798     Node* n = _expensive_nodes->at(i);
4799     igvn.replace_input_of(n, 0, NULL);
4800     igvn.hash_insert(n);

4905 // Auxiliary method to support randomized stressing/fuzzing.
4906 //
4907 // This method can be called the arbitrary number of times, with current count
4908 // as the argument. The logic allows selecting a single candidate from the
4909 // running list of candidates as follows:
4910 //    int count = 0;
4911 //    Cand* selected = null;
4912 //    while(cand = cand->next()) {
4913 //      if (randomized_select(++count)) {
4914 //        selected = cand;
4915 //      }
4916 //    }
4917 //
4918 // Including count equalizes the chances any candidate is "selected".
4919 // This is useful when we don't have the complete list of candidates to choose
4920 // from uniformly. In this case, we need to adjust the randomicity of the
4921 // selection, or else we will end up biasing the selection towards the latter
4922 // candidates.
4923 //
4924 // Quick back-envelope calculation shows that for the list of n candidates
4925 // the equal probability for the candidate to persist as "best" can be
4926 // achieved by replacing it with "next" k-th candidate with the probability
4927 // of 1/k. It can be easily shown that by the end of the run, the
4928 // probability for any candidate is converged to 1/n, thus giving the
4929 // uniform distribution among all the candidates.
4930 //
4931 // We don't care about the domain size as long as (RANDOMIZED_DOMAIN / count) is large.
4932 #define RANDOMIZED_DOMAIN_POW 29
4933 #define RANDOMIZED_DOMAIN (1 << RANDOMIZED_DOMAIN_POW)
4934 #define RANDOMIZED_DOMAIN_MASK ((1 << (RANDOMIZED_DOMAIN_POW + 1)) - 1)
4935 bool Compile::randomized_select(int count) {
4936   assert(count > 0, "only positive");
4937   return (os::random() & RANDOMIZED_DOMAIN_MASK) < (RANDOMIZED_DOMAIN / count);
4938 }
