  79 //------------------------------match------------------------------------------
  80 // Construct projections for incoming parameters, and their RegMask info
  81 Node *StartNode::match(const ProjNode *proj, const Matcher *match, const RegMask* mask) {
  82   switch (proj->_con) {
  83   case TypeFunc::Control:
  84   case TypeFunc::I_O:
  85   case TypeFunc::Memory:
  86     return new MachProjNode(this,proj->_con,RegMask::Empty,MachProjNode::unmatched_proj);
  87   case TypeFunc::FramePtr:
  88     return new MachProjNode(this,proj->_con,Matcher::c_frame_ptr_mask, Op_RegP);
  89   case TypeFunc::ReturnAdr:
  90     return new MachProjNode(this,proj->_con,match->_return_addr_mask,Op_RegP);
  91   case TypeFunc::Parms:
  92   default: {
  93       uint parm_num = proj->_con - TypeFunc::Parms;
  94       const Type *t = _domain->field_at(proj->_con);
  95       if (t->base() == Type::Half)  // 2nd half of Longs and Doubles
  96         return new ConNode(Type::TOP);
  97       uint ideal_reg = t->ideal_reg();
  98       RegMask &rm = match->_calling_convention_mask[parm_num];
  99       return new MachProjNode(this,proj->_con,rm,ideal_reg);
 100     }
 101   }
 102   return NULL;
 103 }

 139 uint ParmNode::ideal_reg() const {
 140   switch( _con ) {
 141   case TypeFunc::Control  : // fall through
 142   case TypeFunc::I_O      : // fall through
 143   case TypeFunc::Memory   : return 0;
 144   case TypeFunc::FramePtr : // fall through
 145   case TypeFunc::ReturnAdr: return Op_RegP;
 146   default                 : assert( _con > TypeFunc::Parms, "" );
 147     // fall through
 148   case TypeFunc::Parms    : {
 149     // Type of argument being passed
 150     const Type *t = in(0)->as_Start()->_domain->field_at(_con);
 151     return t->ideal_reg();
 152   }
 153   }
 154   ShouldNotReachHere();
 155   return 0;
 156 }

 251 JVMState::JVMState(ciMethod* method, JVMState* caller) :
 252   _method(method) {
 253   assert(method != NULL, "must be valid call site");
 254   _bci = InvocationEntryBci;
 255   _reexecute = Reexecute_Undefined;
 256   debug_only(_bci = -99);  // random garbage value
 257   debug_only(_map = (SafePointNode*)-1);
 258   _caller = caller;
 259   _depth  = 1 + (caller == NULL ? 0 : caller->depth());
 260   _locoff = TypeFunc::Parms;
 261   _stkoff = _locoff + _method->max_locals();
 262   _monoff = _stkoff + _method->max_stack();
 263   _scloff = _monoff;
 264   _endoff = _monoff;
 265   _sp = 0;
 266 }

 268   _method(NULL) {
 269   _bci = InvocationEntryBci;
 270   _reexecute = Reexecute_Undefined;

 272   _caller = NULL;

 282 //--------------------------------of_depth-------------------------------------
 283 JVMState* JVMState::of_depth(int d) const {
 284   const JVMState* jvmp = this;
 285   assert(0 < d && (uint)d <= depth(), "oob");
 286   for (int skip = depth() - d; skip > 0; skip--) {
 287     jvmp = jvmp->caller();
 288   }
 289   assert(jvmp->depth() == (uint)d, "found the right one");
 290   return (JVMState*)jvmp;
 291 }

 293 //-----------------------------same_calls_as-----------------------------------
 294 bool JVMState::same_calls_as(const JVMState* that) const {
 295   if (this == that)                    return true;
 296   if (this->depth() != that->depth())  return false;
 297   const JVMState* p = this;
 298   const JVMState* q = that;
 299   for (;;) {
 300     if (p->_method != q->_method)    return false;
 301     if (p->_method == NULL)          return true;   // bci is irrelevant
 302     if (p->_bci    != q->_bci)       return false;
 303     if (p->_reexecute != q->_reexecute)  return false;
 304     p = p->caller();
 305     q = q->caller();
 306     if (p == q)                      return true;
 307     assert(p != NULL && q != NULL, "depth check ensures we don't run off end");
 308   }
 309 }

 336 //------------------------------format_helper----------------------------------
 337 // Given an allocation (a Chaitin object) and a Node decide if the Node carries
 338 // any defined value or not.  If it does, print out the register or constant.
 339 static void format_helper( PhaseRegAlloc *regalloc, outputStream* st, Node *n, const char *msg, uint i, GrowableArray<SafePointScalarObjectNode*> *scobjs ) {
 340   if (n == NULL) { st->print(" NULL"); return; }
 341   if (n->is_SafePointScalarObject()) {
 342     // Scalar replacement.
 343     SafePointScalarObjectNode* spobj = n->as_SafePointScalarObject();
 344     scobjs->append_if_missing(spobj);
 345     int sco_n = scobjs->find(spobj);
 346     assert(sco_n >= 0, "");
 347     st->print(" %s%d]=#ScObj" INT32_FORMAT, msg, i, sco_n);
 348     return;
 349   }
 350   if (regalloc->node_regs_max_index() > 0 &&
 351       OptoReg::is_valid(regalloc->get_reg_first(n))) { // Check for undefined
 352     char buf[50];
 353     regalloc->dump_register(n,buf);
 354     st->print(" %s%d]=%s",msg,i,buf);
 355   } else {                      // No register, but might be constant
 356     const Type *t = n->bottom_type();
 357     switch (t->base()) {
 358     case Type::Int:
 359       st->print(" %s%d]=#" INT32_FORMAT,msg,i,t->is_int()->get_con());
 360       break;
 361     case Type::AnyPtr:
 362       assert( t == TypePtr::NULL_PTR || n->in_dump(), "" );
 363       st->print(" %s%d]=#NULL",msg,i);
 364       break;
 365     case Type::AryPtr:
 366     case Type::InstPtr:
 367       st->print(" %s%d]=#Ptr" INTPTR_FORMAT,msg,i,p2i(t->isa_oopptr()->const_oop()));
 368       break;
 369     case Type::KlassPtr:
 370       st->print(" %s%d]=#Ptr" INTPTR_FORMAT,msg,i,p2i(t->make_ptr()->isa_klassptr()->klass()));
 371       break;
 372     case Type::MetadataPtr:
 373       st->print(" %s%d]=#Ptr" INTPTR_FORMAT,msg,i,p2i(t->make_ptr()->isa_metadataptr()->metadata()));
 374       break;
 375     case Type::NarrowOop:
 376       st->print(" %s%d]=#Ptr" INTPTR_FORMAT,msg,i,p2i(t->make_ptr()->isa_oopptr()->const_oop()));
 377       break;
 378     case Type::RawPtr:
 379       st->print(" %s%d]=#Raw" INTPTR_FORMAT,msg,i,p2i(t->is_rawptr()));
 380       break;
 381     case Type::DoubleCon:
 382       st->print(" %s%d]=#%fD",msg,i,t->is_double_constant()->_d);
 383       break;
 384     case Type::FloatCon:
 385       st->print(" %s%d]=#%fF",msg,i,t->is_float_constant()->_f);
 386       break;
 387     case Type::Long:
 388       st->print(" %s%d]=#" INT64_FORMAT,msg,i,(int64_t)(t->is_long()->get_con()));
 389       break;
 390     case Type::Half:
 391     case Type::Top:
 392       st->print(" %s%d]=_",msg,i);
 393       break;
 394     default: ShouldNotReachHere();
 395     }
 396   }
 397 }

 399 //------------------------------format-----------------------------------------
 400 void JVMState::format(PhaseRegAlloc *regalloc, const Node *n, outputStream* st) const {
 401   st->print("        #");
 402   if (_method) {
 403     _method->print_short_name(st);
 404     st->print(" @ bci:%d ",_bci);
 405   } else {
 406     st->print_cr(" runtime stub ");
 407     return;
 408   }
 409   if (n->is_MachSafePoint()) {
 410     GrowableArray<SafePointScalarObjectNode*> scobjs;
 411     MachSafePointNode *mcall = n->as_MachSafePoint();
 412     uint i;
 413     // Print locals
 414     for (i = 0; i < (uint)loc_size(); i++)
 415       format_helper(regalloc, st, mcall->local(this, i), "L[", i, &scobjs);
 416     // Print stack
 417     for (i = 0; i < (uint)stk_size(); i++) {
 418       if ((uint)(_stkoff + i) >= mcall->len())
 419         st->print(" oob ");
 420       else
 421        format_helper(regalloc, st, mcall->stack(this, i), "STK[", i, &scobjs);
 422     }
 423     for (i = 0; (int)i < nof_monitors(); i++) {
 424       Node *box = mcall->monitor_box(this, i);
 425       Node *obj = mcall->monitor_obj(this, i);
 426       if (regalloc->node_regs_max_index() > 0 &&
 427           OptoReg::is_valid(regalloc->get_reg_first(box))) {
 428         box = BoxLockNode::box_node(box);
 429         format_helper(regalloc, st, box, "MON-BOX[", i, &scobjs);
 430       } else {
 431         OptoReg::Name box_reg = BoxLockNode::reg(box);
 432         st->print(" MON-BOX%d=%s+%d",
 433                    i,
 434                    OptoReg::regname(OptoReg::c_frame_pointer),
 435                    regalloc->reg2offset(box_reg));
 436       }
 437       const char* obj_msg = "MON-OBJ[";
 438       if (EliminateLocks) {
 439         if (BoxLockNode::box_node(box)->is_eliminated())
 440           obj_msg = "MON-OBJ(LOCK ELIMINATED)[";
 441       }
 442       format_helper(regalloc, st, obj, obj_msg, i, &scobjs);
 443     }
 444 
 445     for (i = 0; i < (uint)scobjs.length(); i++) {
 446       // Scalar replaced objects.
 447       st->cr();
 448       st->print("        # ScObj" INT32_FORMAT " ", i);
 449       SafePointScalarObjectNode* spobj = scobjs.at(i);
 450       ciKlass* cik = spobj->bottom_type()->is_oopptr()->klass();
 451       assert(cik->is_instance_klass() ||
 452              cik->is_array_klass(), "Not supported allocation.");
 453       ciInstanceKlass *iklass = NULL;
 454       if (cik->is_instance_klass()) {
 455         cik->print_name_on(st);
 456         iklass = cik->as_instance_klass();
 457       } else if (cik->is_type_array_klass()) {
 458         cik->as_array_klass()->base_element_type()->print_name_on(st);
 459         st->print("[%d]", spobj->n_fields());
 460       } else if (cik->is_obj_array_klass()) {
 461         ciKlass* cie = cik->as_obj_array_klass()->base_element_klass();
 462         if (cie->is_instance_klass()) {
 463           cie->print_name_on(st);
 464         } else if (cie->is_type_array_klass()) {
 465           cie->as_array_klass()->base_element_type()->print_name_on(st);
 466         } else {
 467           ShouldNotReachHere();
 468         }
 469         st->print("[%d]", spobj->n_fields());
 470         int ndim = cik->as_array_klass()->dimension() - 1;
 471         while (ndim-- > 0) {
 472           st->print("[]");
 473         }
 474       } else if (cik->is_value_array_klass()) {
 475         ciKlass* cie = cik->as_value_array_klass()->base_element_klass();
 476         cie->print_name_on(st);
 477         st->print("[%d]", spobj->n_fields());
 478         int ndim = cik->as_array_klass()->dimension() - 1;
 479         while (ndim-- > 0) {
 480           st->print("[]");
 481         }
 482       }
 483       st->print("={");
 484       uint nf = spobj->n_fields();
 485       if (nf > 0) {
 486         uint first_ind = spobj->first_index(mcall->jvms());
 487         Node* fld_node = mcall->in(first_ind);
 488         ciField* cifield;
 489         if (iklass != NULL) {
 490           st->print(" [");
 491           cifield = iklass->nonstatic_field_at(0);
 492           cifield->print_name_on(st);
 493           format_helper(regalloc, st, fld_node, ":", 0, &scobjs);
 494         } else {
 495           format_helper(regalloc, st, fld_node, "[", 0, &scobjs);
 496         }
 497         for (uint j = 1; j < nf; j++) {
 498           fld_node = mcall->in(first_ind+j);
 499           if (iklass != NULL) {
 500             st->print(", [");
 501             cifield = iklass->nonstatic_field_at(j);
 502             cifield->print_name_on(st);
 503             format_helper(regalloc, st, fld_node, ":", j, &scobjs);
 504           } else {
 505             format_helper(regalloc, st, fld_node, ", [", j, &scobjs);
 506           }
 507         }
 508       }
 509       st->print(" }");
 510     }
 511   }
 512   st->cr();
 513   if (caller() != NULL) caller()->format(regalloc, n, st);
 514 }

 517 void JVMState::dump_spec(outputStream *st) const {
 518   if (_method != NULL) {
 519     bool printed = false;
 520     if (!Verbose) {
 521       // The JVMS dumps make really, really long lines.
 522       // Take out the most boring parts, which are the package prefixes.
 523       char buf[500];
 524       stringStream namest(buf, sizeof(buf));
 525       _method->print_short_name(&namest);
 526       if (namest.count() < sizeof(buf)) {
 527         const char* name = namest.base();
 528         if (name[0] == ' ')  ++name;
 529         const char* endcn = strchr(name, ':');  // end of class name
 530         if (endcn == NULL)  endcn = strchr(name, '(');
 531         if (endcn == NULL)  endcn = name + strlen(name);
 532         while (endcn > name && endcn[-1] != '.' && endcn[-1] != '/')
 533           --endcn;
 534         st->print(" %s", endcn);
 535         printed = true;
 536       }
 537     }
 538     if (!printed)
 539       _method->print_short_name(st);
 540     st->print(" @ bci:%d",_bci);
 541     if(_reexecute == Reexecute_True)
 542       st->print(" reexecute");
 543   } else {
 544     st->print(" runtime stub");
 545   }
 546   if (caller() != NULL)  caller()->dump_spec(st);
 547 }

 550 void JVMState::dump_on(outputStream* st) const {
 551   bool print_map = _map && !((uintptr_t)_map & 1) &&
 552                   ((caller() == NULL) || (caller()->map() != _map));
 553   if (print_map) {
 554     if (_map->len() > _map->req()) {  // _map->has_exceptions()
 555       Node* ex = _map->in(_map->req());  // _map->next_exception()
 556       // skip the first one; it's already being printed
 557       while (ex != NULL && ex->len() > ex->req()) {
 558         ex = ex->in(ex->req());  // ex->next_exception()
 559         ex->dump(1);
 560       }
 561     }
 562     _map->dump(Verbose ? 2 : 1);
 563   }
 564   if (caller() != NULL) {
 565     caller()->dump_on(st);
 566   }
 567   st->print("JVMS depth=%d loc=%d stk=%d arg=%d mon=%d scalar=%d end=%d mondepth=%d sp=%d bci=%d reexecute=%s method=",
 568              depth(), locoff(), stkoff(), argoff(), monoff(), scloff(), endoff(), monitor_depth(), sp(), bci(), should_reexecute()?"true":"false");
 569   if (_method == NULL) {
 570     st->print_cr("(none)");
 571   } else {

 635 // Mirror the stack size calculation in the deopt code
 636 // How much stack space would we need at this point in the program in
 637 // case of deoptimization?
 638 int JVMState::interpreter_frame_size() const {
 639   const JVMState* jvms = this;
 640   int size = 0;
 641   int callee_parameters = 0;
 642   int callee_locals = 0;
 643   int extra_args = method()->max_stack() - stk_size();
 644 
 645   while (jvms != NULL) {
 646     int locks = jvms->nof_monitors();
 647     int temps = jvms->stk_size();
 648     bool is_top_frame = (jvms == this);
 649     ciMethod* method = jvms->method();
 650 
 651     int frame_size = BytesPerWord * Interpreter::size_activation(method->max_stack(),
 652                                                                  temps + callee_parameters,
 653                                                                  extra_args,
 654                                                                  locks,
 655                                                                  callee_parameters,
 656                                                                  callee_locals,
 657                                                                  is_top_frame);
 658     size += frame_size;
 659 
 660     callee_parameters = method->size_of_parameters();
 661     callee_locals = method->max_locals();
 662     extra_args = 0;
 663     jvms = jvms->caller();
 664   }
 665   return size + Deoptimization::last_frame_adjust(0, callee_locals) * BytesPerWord;
 666 }

 713 //------------------------------match------------------------------------------
 714 // Construct projections for control, I/O, memory-fields, ..., and
 715 // return result(s) along with their RegMask info
 716 Node *CallNode::match(const ProjNode *proj, const Matcher *match, const RegMask* mask) {
 717   uint con = proj->_con;
 718   const TypeTuple *range_cc = tf()->range_cc();
 719   if (con >= TypeFunc::Parms) {
 720     if (is_CallRuntime()) {
 721       if (con == TypeFunc::Parms) {
 722         uint ideal_reg = range_cc->field_at(TypeFunc::Parms)->ideal_reg();
 723         OptoRegPair regs = match->c_return_value(ideal_reg,true);
 724         RegMask rm = RegMask(regs.first());
 725         if (OptoReg::is_valid(regs.second())) {
 726           rm.Insert(regs.second());
 727         }
 728         return new MachProjNode(this,con,rm,ideal_reg);
 729       } else {
 730         assert(con == TypeFunc::Parms+1, "only one return value");
 731         assert(range_cc->field_at(TypeFunc::Parms+1) == Type::HALF, "");
 732         return new MachProjNode(this,con, RegMask::Empty, (uint)OptoReg::Bad);
 733       }
 734     } else {
 735       // The Call may return multiple values (value type fields): we
 736       // create one projection per returned values.
 737       assert(con <= TypeFunc::Parms+1 || InlineTypeReturnedAsFields, "only for multi value return");
 738       uint ideal_reg = range_cc->field_at(con)->ideal_reg();
 739       return new MachProjNode(this, con, mask[con-TypeFunc::Parms], ideal_reg);
 740     }
 741   }
 742 
 743   switch (con) {
 744   case TypeFunc::Control:
 745   case TypeFunc::I_O:
 746   case TypeFunc::Memory:
 747     return new MachProjNode(this,proj->_con,RegMask::Empty,MachProjNode::unmatched_proj);
 748 
 749   case TypeFunc::ReturnAdr:
 750   case TypeFunc::FramePtr:
 751   default:
 752     ShouldNotReachHere();
 753   }
 754   return NULL;
 755 }

 762 //
 763 // Determine whether the call could modify the field of the specified
 764 // instance at the specified offset.
 765 //
 766 bool CallNode::may_modify(const TypeOopPtr *t_oop, PhaseTransform *phase) {
 767   assert((t_oop != NULL), "sanity");
 768   if (is_call_to_arraycopystub() && strcmp(_name, "unsafe_arraycopy") != 0) {
 769     const TypeTuple* args = _tf->domain_sig();
 770     Node* dest = NULL;
 771     // Stubs that can be called once an ArrayCopyNode is expanded have
 772     // different signatures. Look for the second pointer argument,
 773     // that is the destination of the copy.
 774     for (uint i = TypeFunc::Parms, j = 0; i < args->cnt(); i++) {
 775       if (args->field_at(i)->isa_ptr()) {
 776         j++;
 777         if (j == 2) {
 778           dest = in(i);
 779           break;
 780         }
 781       }
 782     }
 783     guarantee(dest != NULL, "Call had only one ptr in, broken IR!");
 784     if (!dest->is_top() && may_modify_arraycopy_helper(phase->type(dest)->is_oopptr(), t_oop, phase)) {
 785       return true;
 786     }
 787     return false;
 788   }
 789   if (t_oop->is_known_instance()) {
 790     // The instance_id is set only for scalar-replaceable allocations which
 791     // are not passed as arguments according to Escape Analysis.
 792     return false;
 793   }
 794   if (t_oop->is_ptr_to_boxed_value()) {
 795     ciKlass* boxing_klass = t_oop->klass();
 796     if (is_CallStaticJava() && as_CallStaticJava()->is_boxing_method()) {
 797       // Skip unrelated boxing methods.
 798       Node* proj = proj_out_or_null(TypeFunc::Parms);
 799       if ((proj == NULL) || (phase->type(proj)->is_instptr()->klass() != boxing_klass)) {
 800         return false;
 801       }
 802     }
 803     if (is_CallJava() && as_CallJava()->method() != NULL) {
 804       ciMethod* meth = as_CallJava()->method();
 805       if (meth->is_getter()) {
 806         return false;
 807       }
 808       // May modify (by reflection) if an boxing object is passed
 809       // as argument or returned.
 810       Node* proj = returns_pointer() ? proj_out_or_null(TypeFunc::Parms) : NULL;
 811       if (proj != NULL) {
 812         const TypeInstPtr* inst_t = phase->type(proj)->isa_instptr();
 813         if ((inst_t != NULL) && (!inst_t->klass_is_exact() ||
 814                                  (inst_t->klass() == boxing_klass))) {
 815           return true;
 816         }
 817       }
 818       const TypeTuple* d = tf()->domain_cc();
 819       for (uint i = TypeFunc::Parms; i < d->cnt(); i++) {
 820         const TypeInstPtr* inst_t = d->field_at(i)->isa_instptr();
 821         if ((inst_t != NULL) && (!inst_t->klass_is_exact() ||
 822                                  (inst_t->klass() == boxing_klass))) {
 823           return true;
 824         }
 825       }
 826       return false;
 827     }
 828   }
 829   return true;
 830 }

 855 // Returns the unique CheckCastPP of a call
 856 // or 'this' if there are several CheckCastPP or unexpected uses
 857 // or returns NULL if there is no one.
 858 Node *CallNode::result_cast() {
 859   Node *cast = NULL;
 860 
 861   Node *p = proj_out_or_null(TypeFunc::Parms);
 862   if (p == NULL)
 863     return NULL;
 864 
 865   for (DUIterator_Fast imax, i = p->fast_outs(imax); i < imax; i++) {
 866     Node *use = p->fast_out(i);
 867     if (use->is_CheckCastPP()) {
 868       if (cast != NULL) {
 869         return this;  // more than 1 CheckCastPP
 870       }
 871       cast = use;
 872     } else if (!use->is_Initialize() &&
 873                !use->is_AddP() &&
 874                use->Opcode() != Op_MemBarStoreStore) {
 875       // Expected uses are restricted to a CheckCastPP, an Initialize
 876       // node, a MemBarStoreStore (clone) and AddP nodes. If we
 877       // encounter any other use (a Phi node can be seen in rare
 878       // cases) return this to prevent incorrect optimizations.
 879       return this;
 880     }
 881   }
 882   return cast;
 883 }

 886 CallProjections* CallNode::extract_projections(bool separate_io_proj, bool do_asserts) {
 887   uint max_res = TypeFunc::Parms-1;
 888   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
 889     ProjNode *pn = fast_out(i)->as_Proj();
 890     max_res = MAX2(max_res, pn->_con);
 891   }
 892 
 893   assert(max_res < _tf->range_cc()->cnt(), "result out of bounds");
 894 
 895   uint projs_size = sizeof(CallProjections);
 896   if (max_res > TypeFunc::Parms) {
 897     projs_size += (max_res-TypeFunc::Parms)*sizeof(Node*);
 898   }
 899   char* projs_storage = resource_allocate_bytes(projs_size);
 900   CallProjections* projs = new(projs_storage)CallProjections(max_res - TypeFunc::Parms + 1);
 901 
 902   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
 903     ProjNode *pn = fast_out(i)->as_Proj();
 904     if (pn->outcnt() == 0) continue;
 905     switch (pn->_con) {
 906     case TypeFunc::Control:
 907       {
 908         // For Control (fallthrough) and I_O (catch_all_index) we have CatchProj -> Catch -> Proj
 909         projs->fallthrough_proj = pn;
 910         DUIterator_Fast jmax, j = pn->fast_outs(jmax);
 911         const Node *cn = pn->fast_out(j);
 912         if (cn->is_Catch()) {
 913           ProjNode *cpn = NULL;
 914           for (DUIterator_Fast kmax, k = cn->fast_outs(kmax); k < kmax; k++) {
 915             cpn = cn->fast_out(k)->as_Proj();
 916             assert(cpn->is_CatchProj(), "must be a CatchProjNode");
 917             if (cpn->_con == CatchProjNode::fall_through_index)
 918               projs->fallthrough_catchproj = cpn;
 919             else {
 920               assert(cpn->_con == CatchProjNode::catch_all_index, "must be correct index.");
 921               projs->catchall_catchproj = cpn;
 922             }
 923           }
 924         }
 925         break;
 926       }
 927     case TypeFunc::I_O:
 928       if (pn->_is_io_use)
 929         projs->catchall_ioproj = pn;
 930       else
 931         projs->fallthrough_ioproj = pn;
 932       for (DUIterator j = pn->outs(); pn->has_out(j); j++) {
 933         Node* e = pn->out(j);
 934         if (e->Opcode() == Op_CreateEx && e->in(0)->is_CatchProj() && e->outcnt() > 0) {
 935           assert(projs->exobj == NULL, "only one");
 936           projs->exobj = e;
 937         }
 938       }
 939       break;
 940     case TypeFunc::Memory:
 941       if (pn->_is_io_use)
 942         projs->catchall_memproj = pn;
 943       else
 944         projs->fallthrough_memproj = pn;
 945       break;
 946     case TypeFunc::Parms:
 947       projs->resproj[0] = pn;
 948       break;
 949     default:
 950       assert(pn->_con <= max_res, "unexpected projection from allocation node.");
 951       projs->resproj[pn->_con-TypeFunc::Parms] = pn;
 952       break;
 953     }
 954   }
 955 
 956   // The resproj may not exist because the result could be ignored
 957   // and the exception object may not exist if an exception handler
 958   // swallows the exception but all the other must exist and be found.
 959   do_asserts = do_asserts && !Compile::current()->inlining_incrementally();
 960   assert(!do_asserts || projs->fallthrough_proj      != NULL, "must be found");
 961   assert(!do_asserts || projs->fallthrough_catchproj != NULL, "must be found");
 962   assert(!do_asserts || projs->fallthrough_memproj   != NULL, "must be found");
 963   assert(!do_asserts || projs->fallthrough_ioproj    != NULL, "must be found");
 964   assert(!do_asserts || projs->catchall_catchproj    != NULL, "must be found");
 965   if (separate_io_proj) {
 966     assert(!do_asserts || projs->catchall_memproj    != NULL, "must be found");
 967     assert(!do_asserts || projs->catchall_ioproj     != NULL, "must be found");
 968   }
 969   return projs;
 970 }

 972 Node *CallNode::Ideal(PhaseGVN *phase, bool can_reshape) {
 973   CallGenerator* cg = generator();
 974   if (can_reshape && cg != NULL && cg->is_mh_late_inline() && !cg->already_attempted()) {
 975     // Check whether this MH handle call becomes a candidate for inlining
 976     ciMethod* callee = cg->method();
 977     vmIntrinsics::ID iid = callee->intrinsic_id();
 978     if (iid == vmIntrinsics::_invokeBasic) {
 979       if (in(TypeFunc::Parms)->Opcode() == Op_ConP) {
 980         phase->C->prepend_late_inline(cg);
 981         set_generator(NULL);
 982       }
 983     } else {
 984       assert(callee->has_member_arg(), "wrong type of call?");
 985       if (in(TypeFunc::Parms + callee->arg_size() - 1)->Opcode() == Op_ConP) {
 986         phase->C->prepend_late_inline(cg);
 987         set_generator(NULL);
 988       }
 989     }
 990   }
 991   return SafePointNode::Ideal(phase, can_reshape);
 992 }

1010   // Copy debug information and adjust JVMState information
1011   uint old_dbg_start = oldcall->tf()->domain_sig()->cnt();

1020   for (uint i = old_dbg_start; i < oldcall->req(); i++) {

1027       if (old_unique != C->unique()) { // New node?
1028         new_in->set_req(0, C->root()); // reset control edge
1029         new_in = phase->transform(new_in); // Register new node.
1030       }

1033     add_req(old_in);
1034   }
1035 
1036   // JVMS may be shared so clone it before we modify it
1037   set_jvms(oldcall->jvms() != NULL ? oldcall->jvms()->clone_deep(C) : NULL);
1038   for (JVMState *jvms = this->jvms(); jvms != NULL; jvms = jvms->caller()) {

1040     jvms->set_locoff(jvms->locoff()+jvms_adj);

1042     jvms->set_monoff(jvms->monoff()+jvms_adj);
1043     jvms->set_scloff(jvms->scloff()+jvms_adj);
1044     jvms->set_endoff(jvms->endoff()+jvms_adj);
1045   }

1111 bool CallStaticJavaNode::remove_useless_allocation(PhaseGVN *phase, Node* ctl, Node* mem, Node* unc_arg) {
1112   // Split if can cause the flattened array branch of an array load to
1113   // end in an uncommon trap. In that case, the allocation of the
1114   // loaded value and its initialization is useless. Eliminate it. use
1115   // the jvm state of the allocation to create a new uncommon trap
1116   // call at the load.
1117   if (ctl == NULL || ctl->is_top() || mem == NULL || mem->is_top() || !mem->is_MergeMem()) {
1118     return false;
1119   }
1120   PhaseIterGVN* igvn = phase->is_IterGVN();
1121   if (ctl->is_Region()) {
1122     bool res = false;
1123     for (uint i = 1; i < ctl->req(); i++) {
1124       MergeMemNode* mm = mem->clone()->as_MergeMem();
1125       for (MergeMemStream mms(mm); mms.next_non_empty(); ) {
1126         Node* m = mms.memory();
1127         if (m->is_Phi() && m->in(0) == ctl) {
1128           mms.set_memory(m->in(i));
1129         }
1130       }
1131       if (remove_useless_allocation(phase, ctl->in(i), mm, unc_arg)) {
1132         res = true;
1133         if (!ctl->in(i)->is_Region()) {
1134           igvn->replace_input_of(ctl, i, phase->C->top());
1135         }
1136       }
1137       igvn->remove_dead_node(mm);
1138     }
1139     return res;
1140   }
1141   // verify the control flow is ok
1142   Node* c = ctl;
1143   Node* copy = NULL;
1144   Node* alloc = NULL;
1145   for (;;) {
1146     if (c == NULL || c->is_top()) {
1147       return false;
1148     }
1149     if (c->is_Proj() || c->is_Catch() || c->is_MemBar()) {
1150       c = c->in(0);
1151     } else if (c->Opcode() == Op_CallLeaf &&
1152                c->as_Call()->entry_point() == CAST_FROM_FN_PTR(address, OptoRuntime::load_unknown_value)) {
1153       copy = c;
1154       c = c->in(0);
1155     } else if (c->is_Allocate()) {
1156       Node* new_obj = c->as_Allocate()->result_cast();
1157       if (copy == NULL || new_obj == NULL) {
1158         return false;
1159       }
1160       Node* copy_dest = copy->in(TypeFunc::Parms + 2);
1161       if (copy_dest != new_obj) {
1162         return false;
1163       }
1164       alloc = c;
1165       break;
1166     } else {
1167       return false;
1168     }
1169   }
1170 
1171   JVMState* jvms = alloc->jvms();
1172   if (phase->C->too_many_traps(jvms->method(), jvms->bci(), Deoptimization::trap_request_reason(uncommon_trap_request()))) {
1173     return false;
1174   }
1175 
1176   Node* alloc_mem = alloc->in(TypeFunc::Memory);
1177   if (alloc_mem == NULL || alloc_mem->is_top()) {
1178     return false;
1179   }
1180   if (!alloc_mem->is_MergeMem()) {
1181     alloc_mem = MergeMemNode::make(alloc_mem);
1182   }
1183 
1184   // and that there's no unexpected side effect
1185   for (MergeMemStream mms2(mem->as_MergeMem(), alloc_mem->as_MergeMem()); mms2.next_non_empty2(); ) {
1186     Node* m1 = mms2.is_empty() ? mms2.base_memory() : mms2.memory();
1187     Node* m2 = mms2.memory2();
1188 
1189     for (uint i = 0; i < 100; i++) {
1190       if (m1 == m2) {
1191         break;
1192       } else if (m1->is_Proj()) {
1193         m1 = m1->in(0);
1194       } else if (m1->is_MemBar()) {
1195         m1 = m1->in(TypeFunc::Memory);
1196       } else if (m1->Opcode() == Op_CallLeaf &&
1197                  m1->as_Call()->entry_point() == CAST_FROM_FN_PTR(address, OptoRuntime::load_unknown_value)) {
1198         if (m1 != copy) {
1199           return false;
1200         }
1201         m1 = m1->in(TypeFunc::Memory);
1202       } else if (m1->is_Allocate()) {
1203         if (m1 != alloc) {
1204           return false;
1205         }
1206         break;
1207       } else if (m1->is_MergeMem()) {
1208         MergeMemNode* mm = m1->as_MergeMem();
1209         int idx = mms2.alias_idx();
1210         if (idx == Compile::AliasIdxBot) {
1211           m1 = mm->base_memory();
1212         } else {
1213           m1 = mm->memory_at(idx);
1214         }
1215       } else {
1216         return false;
1217       }
1218     }
1219   }
1220   if (alloc_mem->outcnt() == 0) {
1221     igvn->remove_dead_node(alloc_mem);
1222   }
1223 
1224   address call_addr = SharedRuntime::uncommon_trap_blob()->entry_point();
1225   CallNode* unc = new CallStaticJavaNode(OptoRuntime::uncommon_trap_Type(), call_addr, "uncommon_trap",
1226                                          jvms->bci(), NULL);
1227   unc->init_req(TypeFunc::Control, alloc->in(0));
1228   unc->init_req(TypeFunc::I_O, alloc->in(TypeFunc::I_O));
1229   unc->init_req(TypeFunc::Memory, alloc->in(TypeFunc::Memory));
1230   unc->init_req(TypeFunc::FramePtr,  alloc->in(TypeFunc::FramePtr));
1231   unc->init_req(TypeFunc::ReturnAdr, alloc->in(TypeFunc::ReturnAdr));
1232   unc->init_req(TypeFunc::Parms+0, unc_arg);
1233   unc->set_cnt(PROB_UNLIKELY_MAG(4));
1234   unc->copy_call_debug_info(igvn, alloc->as_Allocate());
1235 
1236   igvn->replace_input_of(alloc, 0, phase->C->top());
1237 
1238   igvn->register_new_node_with_optimizer(unc);
1239 
1240   Node* ctrl = phase->transform(new ProjNode(unc, TypeFunc::Control));
1241   Node* halt = phase->transform(new HaltNode(ctrl, alloc->in(TypeFunc::FramePtr), "uncommon trap returned which should never happen"));
1242   phase->C->root()->add_req(halt);
1243 
1244   return true;
1245 }

1400 //------------------------------Identity---------------------------------------
1401 // Remove obviously duplicate safepoints
1402 Node* SafePointNode::Identity(PhaseGVN* phase) {
1403 
1404   // If you have back to back safepoints, remove one
1405   if( in(TypeFunc::Control)->is_SafePoint() )
1406     return in(TypeFunc::Control);
1407 
1408   if( in(0)->is_Proj() ) {
1409     Node *n0 = in(0)->in(0);
1410     // Check if he is a call projection (except Leaf Call)
1411     if( n0->is_Catch() ) {
1412       n0 = n0->in(0)->in(0);
1413       assert( n0->is_Call(), "expect a call here" );
1414     }
1415     if( n0->is_Call() && n0->as_Call()->guaranteed_safepoint() ) {
1416       // Don't remove a safepoint belonging to an OuterStripMinedLoopEndNode.
1417       // If the loop dies, they will be removed together.
1418       if (has_out_with(Op_OuterStripMinedLoopEnd)) {
1419         return this;
1420       }
1421       // Useless Safepoint, so remove it
1422       return in(TypeFunc::Control);
1423     }
1424   }
1425 
1426   return this;
1427 }

1481 void SafePointNode::push_monitor(const FastLockNode *lock) {
1482   // Add a LockNode, which points to both the original BoxLockNode (the
1483   // stack space for the monitor) and the Object being locked.
1484   const int MonitorEdges = 2;
1485   assert(JVMState::logMonitorEdges == exact_log2(MonitorEdges), "correct MonitorEdges");
1486   assert(req() == jvms()->endoff(), "correct sizing");
1487   int nextmon = jvms()->scloff();
1488   if (GenerateSynchronizationCode) {
1489     ins_req(nextmon,   lock->box_node());
1490     ins_req(nextmon+1, lock->obj_node());
1491   } else {
1492     Node* top = Compile::current()->top();
1493     ins_req(nextmon, top);
1494     ins_req(nextmon, top);
1495   }
1496   jvms()->set_scloff(nextmon + MonitorEdges);
1497   jvms()->set_endoff(req());
1498 }

1502   debug_only(int num_before_pop = jvms()->nof_monitors());
1503   const int MonitorEdges = 2;
1504   assert(JVMState::logMonitorEdges == exact_log2(MonitorEdges), "correct MonitorEdges");
1505   int scloff = jvms()->scloff();

1511   while (scloff > new_scloff)  del_req_ordered(--scloff);

1513 }

1654     InitializeNode* init = initialization();
1655     PhaseIterGVN* igvn = phase->is_IterGVN();
1656     CallProjections* projs = extract_projections(true, false);
1657     assert(projs->nb_resproj <= 1, "unexpected number of results");
1658     if (projs->fallthrough_catchproj != NULL) {
1659       igvn->replace_node(projs->fallthrough_catchproj, in(TypeFunc::Control));
1660     }
1661     if (projs->fallthrough_memproj != NULL) {
1662       igvn->replace_node(projs->fallthrough_memproj, in(TypeFunc::Memory));
1663     }
1664     if (projs->catchall_memproj != NULL) {

1666     }
1667     if (projs->fallthrough_ioproj != NULL) {
1668       igvn->replace_node(projs->fallthrough_ioproj, in(TypeFunc::I_O));
1669     }
1670     if (projs->catchall_ioproj != NULL) {
1671       igvn->replace_node(projs->catchall_ioproj, phase->C->top());
1672     }
1673     if (projs->catchall_catchproj != NULL) {
1674       igvn->replace_node(projs->catchall_catchproj, phase->C->top());
1675     }
1676     if (projs->resproj[0] != NULL) {

1688     if (init != NULL) {

1811 // Redundant lock elimination
1812 //
1813 // There are various patterns of locking where we release and
1814 // immediately reacquire a lock in a piece of code where no operations
1815 // occur in between that would be observable.  In those cases we can
1816 // skip releasing and reacquiring the lock without violating any
1817 // fairness requirements.  Doing this around a loop could cause a lock
1818 // to be held for a very long time so we concentrate on non-looping
1819 // control flow.  We also require that the operations are fully
1820 // redundant meaning that we don't introduce new lock operations on
1821 // some paths so to be able to eliminate it on others ala PRE.  This
1822 // would probably require some more extensive graph manipulation to
1823 // guarantee that the memory edges were all handled correctly.
1824 //
1825 // Assuming p is a simple predicate which can't trap in any way and s
1826 // is a synchronized method consider this code:
1827 //
1828 //   s();
1829 //   if (p)
1830 //     s();
1831 //   else
1832 //     s();
1833 //   s();
1834 //
1835 // 1. The unlocks of the first call to s can be eliminated if the
1836 // locks inside the then and else branches are eliminated.
1837 //
1838 // 2. The unlocks of the then and else branches can be eliminated if
1839 // the lock of the final call to s is eliminated.
1840 //
1841 // Either of these cases subsumes the simple case of sequential control flow
1842 //
1843 // Addtionally we can eliminate versions without the else case:
1844 //
1845 //   s();
1846 //   if (p)
1847 //     s();
1848 //   s();
1849 //
1850 // 3. In this case we eliminate the unlock of the first s, the lock
1851 // and unlock in the then case and the lock in the final s.
1852 //
1853 // Note also that in all these cases the then/else pieces don't have
1854 // to be trivial as long as they begin and end with synchronization
1855 // operations.
1856 //
1857 //   s();
1858 //   if (p)
1859 //     s();
1860 //     f();
1861 //     s();
1862 //   s();
1863 //
1864 // The code will work properly for this case, leaving in the unlock
1865 // before the call to f and the relock after it.
1866 //
1867 // A potentially interesting case which isn't handled here is when the
1868 // locking is partially redundant.
1869 //
1870 //   s();
1871 //   if (p)
1872 //     s();
1873 //
1874 // This could be eliminated putting unlocking on the else case and
1875 // eliminating the first unlock and the lock in the then side.
1876 // Alternatively the unlock could be moved out of the then side so it
1877 // was after the merge and the first unlock and second lock
1878 // eliminated.  This might require less manipulation of the memory
1879 // state to get correct.
1880 //
1881 // Additionally we might allow work between a unlock and lock before
1882 // giving up eliminating the locks.  The current code disallows any
1883 // conditional control flow between these operations.  A formulation
1884 // similar to partial redundancy elimination computing the
1885 // availability of unlocking and the anticipatability of locking at a
1886 // program point would allow detection of fully redundant locking with
1887 // some amount of work in between.  I'm not sure how often I really
1888 // think that would occur though.  Most of the cases I've seen
1889 // indicate it's likely non-trivial work would occur in between.
1890 // There may be other more complicated constructs where we could
1891 // eliminate locking but I haven't seen any others appear as hot or
1892 // interesting.
1893 //
1894 // Locking and unlocking have a canonical form in ideal that looks
1895 // roughly like this:
1896 //
1897 //              <obj>
1898 //                | \\------+
1899 //                |  \       \
1900 //                | BoxLock   \
1901 //                |  |   |     \
1902 //                |  |    \     \
1903 //                |  |   FastLock
1904 //                |  |   /
1905 //                |  |  /
1906 //                |  |  |
1907 //
1908 //               Lock
1909 //                |
1910 //            Proj #0
1911 //                |
1912 //            MembarAcquire
1913 //                |
1914 //            Proj #0
1915 //
1916 //            MembarRelease
1917 //                |
1918 //            Proj #0
1919 //                |
1920 //              Unlock
1921 //                |
1922 //            Proj #0
1923 //
1924 //
1925 // This code proceeds by processing Lock nodes during PhaseIterGVN
1926 // and searching back through its control for the proper code
1927 // patterns.  Once it finds a set of lock and unlock operations to
1928 // eliminate they are marked as eliminatable which causes the
1929 // expansion of the Lock and Unlock macro nodes to make the operation a NOP
1930 //
1931 //=============================================================================
1932 
1933 //
1934 // Utility function to skip over uninteresting control nodes.  Nodes skipped are:
1935 //   - copy regions.  (These may not have been optimized away yet.)
1936 //   - eliminated locking nodes
1937 //
1938 static Node *next_control(Node *ctrl) {
1939   if (ctrl == NULL)
1940     return NULL;
1941   while (1) {
1942     if (ctrl->is_Region()) {
1943       RegionNode *r = ctrl->as_Region();
1944       Node *n = r->is_copy();
1945       if (n == NULL)
1946         break;  // hit a region, return it
1947       else
1948         ctrl = n;
1949     } else if (ctrl->is_Proj()) {
1950       Node *in0 = ctrl->in(0);
1951       if (in0->is_AbstractLock() && in0->as_AbstractLock()->is_eliminated()) {
1952         ctrl = in0->in(0);
1953       } else {
1954         break;
1955       }
1956     } else {
1957       break; // found an interesting control
1958     }
1959   }
1960   return ctrl;
1961 }
1962 //
1963 // Given a control, see if it's the control projection of an Unlock which
1964 // operating on the same object as lock.
1965 //
1966 bool AbstractLockNode::find_matching_unlock(const Node* ctrl, LockNode* lock,
1967                                             GrowableArray<AbstractLockNode*> &lock_ops) {
1968   ProjNode *ctrl_proj = (ctrl->is_Proj()) ? ctrl->as_Proj() : NULL;
1969   if (ctrl_proj != NULL && ctrl_proj->_con == TypeFunc::Control) {
1970     Node *n = ctrl_proj->in(0);
1971     if (n != NULL && n->is_Unlock()) {
1972       UnlockNode *unlock = n->as_Unlock();
1973       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1974       Node* lock_obj = bs->step_over_gc_barrier(lock->obj_node());
1975       Node* unlock_obj = bs->step_over_gc_barrier(unlock->obj_node());
1976       if (lock_obj->eqv_uncast(unlock_obj) &&
1977           BoxLockNode::same_slot(lock->box_node(), unlock->box_node()) &&
1978           !unlock->is_eliminated()) {
1979         lock_ops.append(unlock);
1980         return true;
1981       }
1982     }
1983   }
1984   return false;
1985 }

1987 //
1988 // Find the lock matching an unlock.  Returns null if a safepoint
1989 // or complicated control is encountered first.
1990 LockNode *AbstractLockNode::find_matching_lock(UnlockNode* unlock) {
1991   LockNode *lock_result = NULL;
1992   // find the matching lock, or an intervening safepoint
1993   Node *ctrl = next_control(unlock->in(0));
1994   while (1) {
1995     assert(ctrl != NULL, "invalid control graph");
1996     assert(!ctrl->is_Start(), "missing lock for unlock");
1997     if (ctrl->is_top()) break;  // dead control path
1998     if (ctrl->is_Proj()) ctrl = ctrl->in(0);
1999     if (ctrl->is_SafePoint()) {
2000         break;  // found a safepoint (may be the lock we are searching for)
2001     } else if (ctrl->is_Region()) {
2002       // Check for a simple diamond pattern.  Punt on anything more complicated
2003       if (ctrl->req() == 3 && ctrl->in(1) != NULL && ctrl->in(2) != NULL) {
2004         Node *in1 = next_control(ctrl->in(1));
2005         Node *in2 = next_control(ctrl->in(2));
2006         if (((in1->is_IfTrue() && in2->is_IfFalse()) ||
2007              (in2->is_IfTrue() && in1->is_IfFalse())) && (in1->in(0) == in2->in(0))) {
2008           ctrl = next_control(in1->in(0)->in(0));
2009         } else {
2010           break;
2011         }
2012       } else {
2013         break;
2014       }
2015     } else {
2016       ctrl = next_control(ctrl->in(0));  // keep searching
2017     }
2018   }
2019   if (ctrl->is_Lock()) {
2020     LockNode *lock = ctrl->as_Lock();
2021     BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
2022     Node* lock_obj = bs->step_over_gc_barrier(lock->obj_node());
2023     Node* unlock_obj = bs->step_over_gc_barrier(unlock->obj_node());
2024     if (lock_obj->eqv_uncast(unlock_obj) &&
2025         BoxLockNode::same_slot(lock->box_node(), unlock->box_node())) {
2026       lock_result = lock;
2027     }
2028   }
2029   return lock_result;
2030 }

2034 bool AbstractLockNode::find_lock_and_unlock_through_if(Node* node, LockNode* lock,
2035                                                        GrowableArray<AbstractLockNode*> &lock_ops) {
2036   Node* if_node = node->in(0);
2037   bool  if_true = node->is_IfTrue();
2038 
2039   if (if_node->is_If() && if_node->outcnt() == 2 && (if_true || node->is_IfFalse())) {
2040     Node *lock_ctrl = next_control(if_node->in(0));
2041     if (find_matching_unlock(lock_ctrl, lock, lock_ops)) {
2042       Node* lock1_node = NULL;
2043       ProjNode* proj = if_node->as_If()->proj_out(!if_true);
2044       if (if_true) {
2045         if (proj->is_IfFalse() && proj->outcnt() == 1) {
2046           lock1_node = proj->unique_out();
2047         }
2048       } else {
2049         if (proj->is_IfTrue() && proj->outcnt() == 1) {
2050           lock1_node = proj->unique_out();
2051         }
2052       }
2053       if (lock1_node != NULL && lock1_node->is_Lock()) {
2054         LockNode *lock1 = lock1_node->as_Lock();
2055         BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
2056         Node* lock_obj = bs->step_over_gc_barrier(lock->obj_node());
2057         Node* lock1_obj = bs->step_over_gc_barrier(lock1->obj_node());
2058         if (lock_obj->eqv_uncast(lock1_obj) &&
2059             BoxLockNode::same_slot(lock->box_node(), lock1->box_node()) &&
2060             !lock1->is_eliminated()) {
2061           lock_ops.append(lock1);
2062           return true;
2063         }
2064       }
2065     }
2066   }
2067 
2068   lock_ops.trunc_to(0);
2069   return false;
2070 }

2072 bool AbstractLockNode::find_unlocks_for_region(const RegionNode* region, LockNode* lock,
2073                                GrowableArray<AbstractLockNode*> &lock_ops) {
2074   // check each control merging at this point for a matching unlock.
2075   // in(0) should be self edge so skip it.
2076   for (int i = 1; i < (int)region->req(); i++) {
2077     Node *in_node = next_control(region->in(i));
2078     if (in_node != NULL) {
2079       if (find_matching_unlock(in_node, lock, lock_ops)) {
2080         // found a match so keep on checking.
2081         continue;
2082       } else if (find_lock_and_unlock_through_if(in_node, lock, lock_ops)) {
2083         continue;
2084       }
2085 
2086       // If we fall through to here then it was some kind of node we
2087       // don't understand or there wasn't a matching unlock, so give
2088       // up trying to merge locks.
2089       lock_ops.trunc_to(0);
2090       return false;
2091     }
2092   }
2093   return true;

2138 
2139   // perform any generic optimizations first (returns 'this' or NULL)
2140   Node *result = SafePointNode::Ideal(phase, can_reshape);
2141   if (result != NULL)  return result;
2142   // Don't bother trying to transform a dead node
2143   if (in(0) && in(0)->is_top())  return NULL;
2144 
2145   // Now see if we can optimize away this lock.  We don't actually
2146   // remove the locking here, we simply set the _eliminate flag which
2147   // prevents macro expansion from expanding the lock.  Since we don't
2148   // modify the graph, the value returned from this function is the
2149   // one computed above.
2150   const Type* obj_type = phase->type(obj_node());
2151   if (can_reshape && EliminateLocks && !is_non_esc_obj() &&
2152       !obj_type->isa_valuetype() && !obj_type->is_valuetypeptr()) {

2154     // If we are locking an unescaped object, the lock/unlock is unnecessary
2155     //
2156     ConnectionGraph *cgr = phase->C->congraph();
2157     if (cgr != NULL && cgr->not_global_escape(obj_node())) {
2158       assert(!is_eliminated() || is_coarsened(), "sanity");
2159       // The lock could be marked eliminated by lock coarsening
2160       // code during first IGVN before EA. Replace coarsened flag
2161       // to eliminate all associated locks/unlocks.
2162 #ifdef ASSERT
2163       this->log_lock_optimization(phase->C,"eliminate_lock_set_non_esc1");
2164 #endif
2165       this->set_non_esc_obj();
2166       return result;
2167     }
2168 
2169     //
2170     // Try lock coarsening
2171     //
2172     PhaseIterGVN* iter = phase->is_IterGVN();
2173     if (iter != NULL && !is_eliminated()) {
2174 
2175       GrowableArray<AbstractLockNode*>   lock_ops;
2176 
2177       Node *ctrl = next_control(in(0));
2178 
2179       // now search back for a matching Unlock
2180       if (find_matching_unlock(ctrl, this, lock_ops)) {
2181         // found an unlock directly preceding this lock.  This is the
2182         // case of single unlock directly control dependent on a
2183         // single lock which is the trivial version of case 1 or 2.
2184       } else if (ctrl->is_Region() ) {
2185         if (find_unlocks_for_region(ctrl->as_Region(), this, lock_ops)) {
2186         // found lock preceded by multiple unlocks along all paths
2187         // joining at this point which is case 3 in description above.
2188         }
2189       } else {
2190         // see if this lock comes from either half of an if and the
2191         // predecessors merges unlocks and the other half of the if
2192         // performs a lock.
2193         if (find_lock_and_unlock_through_if(ctrl, this, lock_ops)) {
2194           // found unlock splitting to an if with locks on both branches.
2195         }
2196       }
2197 
2198       if (lock_ops.length() > 0) {
2199         // add ourselves to the list of locks to be eliminated.
2200         lock_ops.append(this);
2201 
2202   #ifndef PRODUCT
2203         if (PrintEliminateLocks) {
2204           int locks = 0;
2205           int unlocks = 0;
2206           for (int i = 0; i < lock_ops.length(); i++) {
2207             AbstractLockNode* lock = lock_ops.at(i);

2209               locks++;

2214             }

2217         }
2218   #endif
2219 
2220         // for each of the identified locks, mark them
2221         // as eliminatable
2222         for (int i = 0; i < lock_ops.length(); i++) {
2223           AbstractLockNode* lock = lock_ops.at(i);
2224 
2225           // Mark it eliminated by coarsening and update any counters
2226 #ifdef ASSERT
2227           lock->log_lock_optimization(phase->C, "eliminate_lock_set_coarsened");
2228 #endif
2229           lock->set_coarsened();
2230         }
2231       } else if (ctrl->is_Region() &&
2232                  iter->_worklist.member(ctrl)) {

2234         // lock is control dependent on hasn't been processed yet so put

2236         // region simplification has occurred.
2237         iter->_worklist.push(this);
2238       }
2239     }
2240   }
2241

2254   if (stk_slot <= 0) {
2255 #ifdef ASSERT
2256     this->log_lock_optimization(c, "eliminate_lock_INLR_1");
2257 #endif
2258     return false; // External lock or it is not Box (Phi node).
2259   }
2260 
2261   // Ignore complex cases: merged locks or multiple locks.
2262   Node* obj = obj_node();
2263   LockNode* unique_lock = NULL;
2264   if (!box->is_simple_lock_region(&unique_lock, obj)) {
2265 #ifdef ASSERT
2266     this->log_lock_optimization(c, "eliminate_lock_INLR_2a");
2267 #endif
2268     return false;
2269   }
2270   if (unique_lock != this) {
2271 #ifdef ASSERT
2272     this->log_lock_optimization(c, "eliminate_lock_INLR_2b");
2273 #endif
2274     return false;
2275   }
2276 
2277   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
2278   obj = bs->step_over_gc_barrier(obj);
2279   // Look for external lock for the same object.
2280   SafePointNode* sfn = this->as_SafePoint();
2281   JVMState* youngest_jvms = sfn->jvms();
2282   int max_depth = youngest_jvms->depth();
2283   for (int depth = 1; depth <= max_depth; depth++) {
2284     JVMState* jvms = youngest_jvms->of_depth(depth);
2285     int num_mon  = jvms->nof_monitors();

2287     for (int idx = 0; idx < num_mon; idx++) {
2288       Node* obj_node = sfn->monitor_obj(jvms, idx);
2289       obj_node = bs->step_over_gc_barrier(obj_node);
2290       BoxLockNode* box_node = sfn->monitor_box(jvms, idx)->as_BoxLock();
2291       if ((box_node->stack_slot() < stk_slot) && obj_node->eqv_uncast(obj)) {
2292         return true;
2293       }
2294     }
2295   }
2296 #ifdef ASSERT
2297   this->log_lock_optimization(c, "eliminate_lock_INLR_3");
2298 #endif

2306 Node *UnlockNode::Ideal(PhaseGVN *phase, bool can_reshape) {
2307 
2308   // perform any generic optimizations first (returns 'this' or NULL)
2309   Node *result = SafePointNode::Ideal(phase, can_reshape);
2310   if (result != NULL)  return result;
2311   // Don't bother trying to transform a dead node
2312   if (in(0) && in(0)->is_top())  return NULL;
2313 
2314   // Now see if we can optimize away this unlock.  We don't actually
2315   // remove the unlocking here, we simply set the _eliminate flag which
2316   // prevents macro expansion from expanding the unlock.  Since we don't
2317   // modify the graph, the value returned from this function is the
2318   // one computed above.
2319   // Escape state is defined after Parse phase.
2320   const Type* obj_type = phase->type(obj_node());
2321   if (can_reshape && EliminateLocks && !is_non_esc_obj() &&
2322       !obj_type->isa_valuetype() && !obj_type->is_valuetypeptr()) {
2323     //
2324     // If we are unlocking an unescaped object, the lock/unlock is unnecessary.
2325     //
2326     ConnectionGraph *cgr = phase->C->congraph();

2334 #endif
