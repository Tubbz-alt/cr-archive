  53 //----------------------------GraphKit-----------------------------------------
  54 // Main utility constructor.
  55 GraphKit::GraphKit(JVMState* jvms, PhaseGVN* gvn)
  56   : Phase(Phase::Parser),
  57     _env(C->env()),
  58     _gvn((gvn != NULL) ? *gvn : *C->initial_gvn()),
  59     _barrier_set(BarrierSet::barrier_set()->barrier_set_c2())
  60 {
  61   assert(gvn == NULL || !gvn->is_IterGVN() || gvn->is_IterGVN()->delay_transform(), "delay transform should be enabled");
  62   _exceptions = jvms->map()->next_exception();
  63   if (_exceptions != NULL)  jvms->map()->set_next_exception(NULL);
  64   set_jvms(jvms);
  65 #ifdef ASSERT
  66   if (_gvn.is_IterGVN() != NULL) {
  67     assert(_gvn.is_IterGVN()->delay_transform(), "Transformation must be delayed if IterGVN is used");
  68     // Save the initial size of _for_igvn worklist for verification (see ~GraphKit)
  69     _worklist_size = _gvn.C->for_igvn()->size();
  70   }
  71 #endif
  72 }

  89 //---------------------------clean_stack---------------------------------------
  90 // Clear away rubbish from the stack area of the JVM state.
  91 // This destroys any arguments that may be waiting on the stack.
  92 void GraphKit::clean_stack(int from_sp) {
  93   SafePointNode* map      = this->map();
  94   JVMState*      jvms     = this->jvms();
  95   int            stk_size = jvms->stk_size();
  96   int            stkoff   = jvms->stkoff();
  97   Node*          top      = this->top();
  98   for (int i = from_sp; i < stk_size; i++) {
  99     if (map->in(stkoff + i) != top) {
 100       map->set_req(stkoff + i, top);
 101     }
 102   }
 103 }

 106 //--------------------------------sync_jvms-----------------------------------
 107 // Make sure our current jvms agrees with our parse state.
 108 JVMState* GraphKit::sync_jvms() const {
 109   JVMState* jvms = this->jvms();
 110   jvms->set_bci(bci());       // Record the new bci in the JVMState
 111   jvms->set_sp(sp());         // Record the new sp in the JVMState
 112   assert(jvms_in_sync(), "jvms is now in sync");
 113   return jvms;
 114 }

 116 //--------------------------------sync_jvms_for_reexecute---------------------
 117 // Make sure our current jvms agrees with our parse state.  This version
 118 // uses the reexecute_sp for reexecuting bytecodes.
 119 JVMState* GraphKit::sync_jvms_for_reexecute() {
 120   JVMState* jvms = this->jvms();
 121   jvms->set_bci(bci());          // Record the new bci in the JVMState
 122   jvms->set_sp(reexecute_sp());  // Record the new sp in the JVMState
 123   return jvms;
 124 }

 126 #ifdef ASSERT
 127 bool GraphKit::jvms_in_sync() const {
 128   Parse* parse = is_Parse();
 129   if (parse == NULL) {
 130     if (bci() !=      jvms()->bci())          return false;
 131     if (sp()  != (int)jvms()->sp())           return false;
 132     return true;
 133   }
 134   if (jvms()->method() != parse->method())    return false;
 135   if (jvms()->bci()    != parse->bci())       return false;
 136   int jvms_sp = jvms()->sp();
 137   if (jvms_sp          != parse->sp())        return false;
 138   int jvms_depth = jvms()->depth();
 139   if (jvms_depth       != parse->depth())     return false;
 140   return true;
 141 }

 143 // Local helper checks for special internal merge points
 144 // used to accumulate and merge exception states.
 145 // They are marked by the region's in(0) edge being the map itself.
 146 // Such merge points must never "escape" into the parser at large,
 147 // until they have been handed to gvn.transform.
 148 static bool is_hidden_merge(Node* reg) {
 149   if (reg == NULL)  return false;
 150   if (reg->is_Phi()) {
 151     reg = reg->in(0);
 152     if (reg == NULL)  return false;
 153   }
 154   return reg->is_Region() && reg->in(0) != NULL && reg->in(0)->is_Root();
 155 }

 157 void GraphKit::verify_map() const {
 158   if (map() == NULL)  return;  // null map is OK
 159   assert(map()->req() <= jvms()->endoff(), "no extra garbage on map");
 160   assert(!map()->has_exceptions(),    "call add_exception_states_from 1st");
 161   assert(!is_hidden_merge(control()), "call use_exception_state, not set_map");
 162 }

 166   assert(has_saved_ex_oop(ex_map), "every exception state has an ex_oop");
 167 }

 194   for (JVMState* jvmsp = jvms(); jvmsp != NULL; jvmsp = jvmsp->caller()) {

 202 //------------------------------save_ex_oop------------------------------------
 203 // Save an exception without blowing stack contents or other JVM state.
 204 void GraphKit::set_saved_ex_oop(SafePointNode* ex_map, Node* ex_oop) {
 205   assert(!has_saved_ex_oop(ex_map), "clear ex-oop before setting again");
 206   ex_map->add_req(ex_oop);
 207   debug_only(verify_exception_state(ex_map));

 210 inline static Node* common_saved_ex_oop(SafePointNode* ex_map, bool clear_it) {
 211   assert(GraphKit::has_saved_ex_oop(ex_map), "ex_oop must be there");
 212   Node* ex_oop = ex_map->in(ex_map->req()-1);
 213   if (clear_it)  ex_map->del_req(ex_map->req()-1);
 214   return ex_oop;

 237 //-------------------------make_exception_state--------------------------------
 238 // Turn the current JVM state into an exception state, appending the ex_oop.
 239 SafePointNode* GraphKit::make_exception_state(Node* ex_oop) {
 240   sync_jvms();
 241   SafePointNode* ex_map = stop();  // do not manipulate this map any more
 242   set_saved_ex_oop(ex_map, ex_oop);
 243   return ex_map;
 244 }

 247 //--------------------------add_exception_state--------------------------------
 248 // Add an exception to my list of exceptions.
 249 void GraphKit::add_exception_state(SafePointNode* ex_map) {
 250   if (ex_map == NULL || ex_map->control() == top()) {
 251     return;
 252   }
 253 #ifdef ASSERT
 254   verify_exception_state(ex_map);
 255   if (has_exceptions()) {
 256     assert(ex_map->jvms()->same_calls_as(_exceptions->jvms()), "all collected exceptions must come from the same place");
 257   }
 258 #endif
 259 
 260   // If there is already an exception of exactly this type, merge with it.
 261   // In particular, null-checks and other low-level exceptions common up here.
 262   Node*       ex_oop  = saved_ex_oop(ex_map);
 263   const Type* ex_type = _gvn.type(ex_oop);
 264   if (ex_oop == top()) {
 265     // No action needed.
 266     return;
 267   }
 268   assert(ex_type->isa_instptr(), "exception must be an instance");
 269   for (SafePointNode* e2 = _exceptions; e2 != NULL; e2 = e2->next_exception()) {
 270     const Type* ex_type2 = _gvn.type(saved_ex_oop(e2));
 271     // We check sp also because call bytecodes can generate exceptions
 272     // both before and after arguments are popped!
 273     if (ex_type2 == ex_type
 274         && e2->_jvms->sp() == ex_map->_jvms->sp()) {
 275       combine_exception_states(ex_map, e2);
 276       return;
 277     }
 278   }
 279 
 280   // No pre-existing exception of the same type.  Chain it on the list.
 281   push_exception_state(ex_map);
 282 }

 284 //-----------------------add_exception_states_from-----------------------------
 285 void GraphKit::add_exception_states_from(JVMState* jvms) {
 286   SafePointNode* ex_map = jvms->map()->next_exception();
 287   if (ex_map != NULL) {
 288     jvms->map()->set_next_exception(NULL);
 289     for (SafePointNode* next_map; ex_map != NULL; ex_map = next_map) {
 290       next_map = ex_map->next_exception();
 291       ex_map->set_next_exception(NULL);
 292       add_exception_state(ex_map);
 293     }
 294   }
 295 }

 297 //-----------------------transfer_exceptions_into_jvms-------------------------
 298 JVMState* GraphKit::transfer_exceptions_into_jvms() {
 299   if (map() == NULL) {
 300     // We need a JVMS to carry the exceptions, but the map has gone away.
 301     // Create a scratch JVMS, cloned from any of the exception states...
 302     if (has_exceptions()) {
 303       _map = _exceptions;
 304       _map = clone_map();
 305       _map->set_next_exception(NULL);
 306       clear_saved_ex_oop(_map);
 307       debug_only(verify_map());
 308     } else {
 309       // ...or created from scratch
 310       JVMState* jvms = new (C) JVMState(_method, NULL);
 311       jvms->set_bci(_bci);
 312       jvms->set_sp(_sp);
 313       jvms->set_map(new SafePointNode(TypeFunc::Parms, jvms));
 314       set_jvms(jvms);
 315       for (uint i = 0; i < map()->req(); i++)  map()->init_req(i, top());
 316       set_all_memory(top());
 317       while (map()->req() < jvms->endoff())  map()->add_req(top());
 318     }
 319     // (This is a kludge, in case you didn't notice.)
 320     set_control(top());
 321   }
 322   JVMState* jvms = sync_jvms();
 323   assert(!jvms->map()->has_exceptions(), "no exceptions on this map yet");
 324   jvms->map()->set_next_exception(_exceptions);
 325   _exceptions = NULL;   // done with this set of exceptions
 326   return jvms;
 327 }

 329 static inline void add_n_reqs(Node* dstphi, Node* srcphi) {
 330   assert(is_hidden_merge(dstphi), "must be a special merge node");
 331   assert(is_hidden_merge(srcphi), "must be a special merge node");
 332   uint limit = srcphi->req();
 333   for (uint i = PhiNode::Input; i < limit; i++) {
 334     dstphi->add_req(srcphi->in(i));
 335   }
 336 }
 337 static inline void add_one_req(Node* dstphi, Node* src) {
 338   assert(is_hidden_merge(dstphi), "must be a special merge node");
 339   assert(!is_hidden_merge(src), "must not be a special merge node");
 340   dstphi->add_req(src);
 341 }

 343 //-----------------------combine_exception_states------------------------------
 344 // This helper function combines exception states by building phis on a
 345 // specially marked state-merging region.  These regions and phis are
 346 // untransformed, and can build up gradually.  The region is marked by
 347 // having a control input of its exception map, rather than NULL.  Such
 348 // regions do not appear except in this function, and in use_exception_state.
 349 void GraphKit::combine_exception_states(SafePointNode* ex_map, SafePointNode* phi_map) {
 350   if (failing())  return;  // dying anyway...
 351   JVMState* ex_jvms = ex_map->_jvms;
 352   assert(ex_jvms->same_calls_as(phi_map->_jvms), "consistent call chains");
 353   assert(ex_jvms->stkoff() == phi_map->_jvms->stkoff(), "matching locals");
 354   assert(ex_jvms->sp() == phi_map->_jvms->sp(), "matching stack sizes");
 355   assert(ex_jvms->monoff() == phi_map->_jvms->monoff(), "matching JVMS");
 356   assert(ex_jvms->scloff() == phi_map->_jvms->scloff(), "matching scalar replaced objects");
 357   assert(ex_map->req() == phi_map->req(), "matching maps");
 358   uint tos = ex_jvms->stkoff() + ex_jvms->sp();
 359   Node*         hidden_merge_mark = root();
 360   Node*         region  = phi_map->control();
 361   MergeMemNode* phi_mem = phi_map->merged_memory();
 362   MergeMemNode* ex_mem  = ex_map->merged_memory();
 363   if (region->in(0) != hidden_merge_mark) {
 364     // The control input is not (yet) a specially-marked region in phi_map.
 365     // Make it so, and build some phis.
 366     region = new RegionNode(2);
 367     _gvn.set_type(region, Type::CONTROL);
 368     region->set_req(0, hidden_merge_mark);  // marks an internal ex-state
 369     region->init_req(1, phi_map->control());
 370     phi_map->set_control(region);
 371     Node* io_phi = PhiNode::make(region, phi_map->i_o(), Type::ABIO);
 372     record_for_igvn(io_phi);
 373     _gvn.set_type(io_phi, Type::ABIO);
 374     phi_map->set_i_o(io_phi);
 375     for (MergeMemStream mms(phi_mem); mms.next_non_empty(); ) {
 376       Node* m = mms.memory();
 377       Node* m_phi = PhiNode::make(region, m, Type::MEMORY, mms.adr_type(C));
 378       record_for_igvn(m_phi);
 379       _gvn.set_type(m_phi, Type::MEMORY);
 380       mms.set_memory(m_phi);
 381     }
 382   }
 383 
 384   // Either or both of phi_map and ex_map might already be converted into phis.
 385   Node* ex_control = ex_map->control();
 386   // if there is special marking on ex_map also, we add multiple edges from src
 387   bool add_multiple = (ex_control->in(0) == hidden_merge_mark);
 388   // how wide was the destination phi_map, originally?
 389   uint orig_width = region->req();
 390 
 391   if (add_multiple) {
 392     add_n_reqs(region, ex_control);
 393     add_n_reqs(phi_map->i_o(), ex_map->i_o());
 394   } else {
 395     // ex_map has no merges, so we just add single edges everywhere
 396     add_one_req(region, ex_control);
 397     add_one_req(phi_map->i_o(), ex_map->i_o());
 398   }
 399   for (MergeMemStream mms(phi_mem, ex_mem); mms.next_non_empty2(); ) {
 400     if (mms.is_empty()) {
 401       // get a copy of the base memory, and patch some inputs into it
 402       const TypePtr* adr_type = mms.adr_type(C);
 403       Node* phi = mms.force_memory()->as_Phi()->slice_memory(adr_type);
 404       assert(phi->as_Phi()->region() == mms.base_memory()->in(0), "");
 405       mms.set_memory(phi);
 406       // Prepare to append interesting stuff onto the newly sliced phi:
 407       while (phi->req() > orig_width)  phi->del_req(phi->req()-1);
 408     }
 409     // Append stuff from ex_map:
 410     if (add_multiple) {
 411       add_n_reqs(mms.memory(), mms.memory2());
 412     } else {
 413       add_one_req(mms.memory(), mms.memory2());
 414     }
 415   }
 416   uint limit = ex_map->req();
 417   for (uint i = TypeFunc::Parms; i < limit; i++) {
 418     // Skip everything in the JVMS after tos.  (The ex_oop follows.)
 419     if (i == tos)  i = ex_jvms->monoff();
 420     Node* src = ex_map->in(i);
 421     Node* dst = phi_map->in(i);
 422     if (src != dst) {
 423       PhiNode* phi;
 424       if (dst->in(0) != region) {
 425         dst = phi = PhiNode::make(region, dst, _gvn.type(dst));
 426         record_for_igvn(phi);
 427         _gvn.set_type(phi, phi->type());
 428         phi_map->set_req(i, dst);
 429         // Prepare to append interesting stuff onto the new phi:
 430         while (dst->req() > orig_width)  dst->del_req(dst->req()-1);
 431       } else {
 432         assert(dst->is_Phi(), "nobody else uses a hidden region");
 433         phi = dst->as_Phi();
 434       }
 435       if (add_multiple && src->in(0) == ex_control) {
 436         // Both are phis.
 437         add_n_reqs(dst, src);
 438       } else {
 439         while (dst->req() < region->req())  add_one_req(dst, src);
 440       }
 441       const Type* srctype = _gvn.type(src);
 442       if (phi->type() != srctype) {
 443         const Type* dsttype = phi->type()->meet_speculative(srctype);
 444         if (phi->type() != dsttype) {
 445           phi->set_type(dsttype);
 446           _gvn.set_type(phi, dsttype);
 447         }
 448       }
 449     }
 450   }
 451   phi_map->merge_replaced_nodes_with(ex_map);
 452 }

 454 //--------------------------use_exception_state--------------------------------
 455 Node* GraphKit::use_exception_state(SafePointNode* phi_map) {
 456   if (failing()) { stop(); return top(); }
 457   Node* region = phi_map->control();
 458   Node* hidden_merge_mark = root();
 459   assert(phi_map->jvms()->map() == phi_map, "sanity: 1-1 relation");
 460   Node* ex_oop = clear_saved_ex_oop(phi_map);
 461   if (region->in(0) == hidden_merge_mark) {
 462     // Special marking for internal ex-states.  Process the phis now.
 463     region->set_req(0, region);  // now it's an ordinary region
 464     set_jvms(phi_map->jvms());   // ...so now we can use it as a map
 465     // Note: Setting the jvms also sets the bci and sp.
 466     set_control(_gvn.transform(region));
 467     uint tos = jvms()->stkoff() + sp();
 468     for (uint i = 1; i < tos; i++) {
 469       Node* x = phi_map->in(i);
 470       if (x->in(0) == region) {
 471         assert(x->is_Phi(), "expected a special phi");
 472         phi_map->set_req(i, _gvn.transform(x));
 473       }
 474     }
 475     for (MergeMemStream mms(merged_memory()); mms.next_non_empty(); ) {
 476       Node* x = mms.memory();
 477       if (x->in(0) == region) {
 478         assert(x->is_Phi(), "nobody else uses a hidden region");
 479         mms.set_memory(_gvn.transform(x));
 480       }
 481     }
 482     if (ex_oop->in(0) == region) {
 483       assert(ex_oop->is_Phi(), "expected a special phi");
 484       ex_oop = _gvn.transform(ex_oop);
 485     }
 486   } else {
 487     set_jvms(phi_map->jvms());
 488   }
 489 
 490   assert(!is_hidden_merge(phi_map->control()), "hidden ex. states cleared");
 491   assert(!is_hidden_merge(phi_map->i_o()), "hidden ex. states cleared");
 492   return ex_oop;
 493 }

 505 void GraphKit::uncommon_trap_if_should_post_on_exceptions(Deoptimization::DeoptReason reason,
 506                                                           bool must_throw) {
 507     // if the exception capability is set, then we will generate code
 508     // to check the JavaThread.should_post_on_exceptions flag to see
 509     // if we actually need to report exception events (for this
 510     // thread).  If we don't need to report exception events, we will
 511     // take the normal fast path provided by add_exception_events.  If
 512     // exception event reporting is enabled for this thread, we will
 513     // take the uncommon_trap in the BuildCutout below.
 514 
 515     // first must access the should_post_on_exceptions_flag in this thread's JavaThread
 516     Node* jthread = _gvn.transform(new ThreadLocalNode());
 517     Node* adr = basic_plus_adr(top(), jthread, in_bytes(JavaThread::should_post_on_exceptions_flag_offset()));
 518     Node* should_post_flag = make_load(control(), adr, TypeInt::INT, T_INT, Compile::AliasIdxRaw, MemNode::unordered);
 519 
 520     // Test the should_post_on_exceptions_flag vs. 0
 521     Node* chk = _gvn.transform( new CmpINode(should_post_flag, intcon(0)) );
 522     Node* tst = _gvn.transform( new BoolNode(chk, BoolTest::eq) );
 523 
 524     // Branch to slow_path if should_post_on_exceptions_flag was true
 525     { BuildCutout unless(this, tst, PROB_MAX);
 526       // Do not try anything fancy if we're notifying the VM on every throw.
 527       // Cf. case Bytecodes::_athrow in parse2.cpp.
 528       uncommon_trap(reason, Deoptimization::Action_none,
 529                     (ciKlass*)NULL, (char*)NULL, must_throw);
 530     }
 531 
 532 }

 534 //------------------------------builtin_throw----------------------------------
 535 void GraphKit::builtin_throw(Deoptimization::DeoptReason reason, Node* arg) {
 536   bool must_throw = true;
 537 
 538   if (env()->jvmti_can_post_on_exceptions()) {
 539     // check if we must post exception events, take uncommon trap if so
 540     uncommon_trap_if_should_post_on_exceptions(reason, must_throw);
 541     // here if should_post_on_exceptions is false
 542     // continue on with the normal codegen
 543   }
 544 
 545   // If this particular condition has not yet happened at this
 546   // bytecode, then use the uncommon trap mechanism, and allow for
 547   // a future recompilation if several traps occur here.
 548   // If the throw is hot, try to use a more complicated inline mechanism
 549   // which keeps execution inside the compiled code.
 550   bool treat_throw_as_hot = false;
 551   ciMethodData* md = method()->method_data();
 552 
 553   if (ProfileTraps) {
 554     if (too_many_traps(reason)) {
 555       treat_throw_as_hot = true;
 556     }
 557     // (If there is no MDO at all, assume it is early in
 558     // execution, and that any deopts are part of the
 559     // startup transient, and don't need to be remembered.)
 560 
 561     // Also, if there is a local exception handler, treat all throws
 562     // as hot if there has been at least one in this method.
 563     if (C->trap_count(reason) != 0
 564         && method()->method_data()->trap_count(reason) != 0
 565         && has_ex_handler()) {
 566         treat_throw_as_hot = true;
 567     }
 568   }
 569 
 570   // If this throw happens frequently, an uncommon trap might cause
 571   // a performance pothole.  If there is a local exception handler,
 572   // and if this particular bytecode appears to be deoptimizing often,
 573   // let us handle the throw inline, with a preconstructed instance.
 574   // Note:   If the deopt count has blown up, the uncommon trap
 575   // runtime is going to flush this nmethod, not matter what.
 576   if (treat_throw_as_hot
 577       && (!StackTraceInThrowable || OmitStackTraceInFastThrow)) {
 578     // If the throw is local, we use a pre-existing instance and
 579     // punt on the backtrace.  This would lead to a missing backtrace
 580     // (a repeat of 4292742) if the backtrace object is ever asked
 581     // for its backtrace.
 582     // Fixing this remaining case of 4292742 requires some flavor of
 583     // escape analysis.  Leave that for the future.
 584     ciInstance* ex_obj = NULL;
 585     switch (reason) {
 586     case Deoptimization::Reason_null_check:
 587       ex_obj = env()->NullPointerException_instance();
 588       break;
 589     case Deoptimization::Reason_div0_check:
 590       ex_obj = env()->ArithmeticException_instance();
 591       break;
 592     case Deoptimization::Reason_range_check:
 593       ex_obj = env()->ArrayIndexOutOfBoundsException_instance();
 594       break;
 595     case Deoptimization::Reason_class_check:
 596       if (java_bc() == Bytecodes::_aastore) {
 597         ex_obj = env()->ArrayStoreException_instance();
 598       } else {
 599         ex_obj = env()->ClassCastException_instance();
 600       }
 601       break;
 602     default:
 603       break;
 604     }
 605     if (failing()) { stop(); return; }  // exception allocation might fail
 606     if (ex_obj != NULL) {
 607       // Cheat with a preallocated exception object.
 608       if (C->log() != NULL)
 609         C->log()->elem("hot_throw preallocated='1' reason='%s'",
 610                        Deoptimization::trap_reason_name(reason));
 611       const TypeInstPtr* ex_con  = TypeInstPtr::make(ex_obj);
 612       Node*              ex_node = _gvn.transform(ConNode::make(ex_con));
 613 
 614       // Clear the detail message of the preallocated exception object.
 615       // Weblogic sometimes mutates the detail message of exceptions
 616       // using reflection.
 617       int offset = java_lang_Throwable::get_detailMessage_offset();
 618       const TypePtr* adr_typ = ex_con->add_offset(offset);
 619 
 620       Node *adr = basic_plus_adr(ex_node, ex_node, offset);
 621       const TypeOopPtr* val_type = TypeOopPtr::make_from_klass(env()->String_klass());
 622       Node *store = access_store_at(ex_node, adr, adr_typ, null(), val_type, T_OBJECT, IN_HEAP);
 623 
 624       add_exception_state(make_exception_state(ex_node));
 625       return;
 626     }
 627   }
 628 
 629   // %%% Maybe add entry to OptoRuntime which directly throws the exc.?
 630   // It won't be much cheaper than bailing to the interp., since we'll
 631   // have to pass up all the debug-info, and the runtime will have to
 632   // create the stack trace.
 633 
 634   // Usual case:  Bail to interpreter.
 635   // Reserve the right to recompile if we haven't seen anything yet.
 636 
 637   ciMethod* m = Deoptimization::reason_is_speculate(reason) ? C->method() : NULL;
 638   Deoptimization::DeoptAction action = Deoptimization::Action_maybe_recompile;
 639   if (treat_throw_as_hot
 640       && (method()->method_data()->trap_recompiled_at(bci(), m)
 641           || C->too_many_traps(reason))) {
 642     // We cannot afford to take more traps here.  Suffer in the interpreter.
 643     if (C->log() != NULL)
 644       C->log()->elem("hot_throw preallocated='0' reason='%s' mcount='%d'",
 645                      Deoptimization::trap_reason_name(reason),
 646                      C->trap_count(reason));
 647     action = Deoptimization::Action_none;
 648   }
 649 
 650   // "must_throw" prunes the JVM state to include only the stack, if there
 651   // are no local exception handlers.  This should cut down on register
 652   // allocation time and code size, by drastically reducing the number
 653   // of in-edges on the call to the uncommon trap.
 654 
 655   uncommon_trap(reason, action, (ciKlass*)NULL, (char*)NULL, must_throw);
 656 }

 750 //----------------------------kill_dead_locals---------------------------------
 751 // Detect any locals which are known to be dead, and force them to top.
 752 void GraphKit::kill_dead_locals() {
 753   // Consult the liveness information for the locals.  If any
 754   // of them are unused, then they can be replaced by top().  This
 755   // should help register allocation time and cut down on the size
 756   // of the deoptimization information.
 757 
 758   // This call is made from many of the bytecode handling
 759   // subroutines called from the Big Switch in do_one_bytecode.
 760   // Every bytecode which might include a slow path is responsible
 761   // for killing its dead locals.  The more consistent we
 762   // are about killing deads, the fewer useless phis will be
 763   // constructed for them at various merge points.
 764 
 765   // bci can be -1 (InvocationEntryBci).  We return the entry
 766   // liveness for the method.
 767 
 768   if (method() == NULL || method()->code_size() == 0) {
 769     // We are building a graph for a call to a native method.
 770     // All locals are live.
 771     return;
 772   }
 773 
 774   ResourceMark rm;
 775 
 776   // Consult the liveness information for the locals.  If any
 777   // of them are unused, then they can be replaced by top().  This
 778   // should help register allocation time and cut down on the size
 779   // of the deoptimization information.
 780   MethodLivenessResult live_locals = method()->liveness_at_bci(bci());
 781 
 782   int len = (int)live_locals.size();
 783   assert(len <= jvms()->loc_size(), "too many live locals");
 784   for (int local = 0; local < len; local++) {
 785     if (!live_locals.at(local)) {
 786       set_local(local, top());
 787     }
 788   }
 789 }

 791 #ifdef ASSERT
 792 //-------------------------dead_locals_are_killed------------------------------
 793 // Return true if all dead locals are set to top in the map.
 794 // Used to assert "clean" debug info at various points.
 795 bool GraphKit::dead_locals_are_killed() {
 796   if (method() == NULL || method()->code_size() == 0) {
 797     // No locals need to be dead, so all is as it should be.
 798     return true;
 799   }
 800 
 801   // Make sure somebody called kill_dead_locals upstream.
 802   ResourceMark rm;
 803   for (JVMState* jvms = this->jvms(); jvms != NULL; jvms = jvms->caller()) {
 804     if (jvms->loc_size() == 0)  continue;  // no locals to consult
 805     SafePointNode* map = jvms->map();
 806     ciMethod* method = jvms->method();
 807     int       bci    = jvms->bci();
 808     if (jvms == this->jvms()) {
 809       bci = this->bci();  // it might not yet be synched
 810     }
 811     MethodLivenessResult live_locals = method->liveness_at_bci(bci);
 812     int len = (int)live_locals.size();
 813     if (!live_locals.is_valid() || len == 0)
 814       // This method is trivial, or is poisoned by a breakpoint.
 815       return true;
 816     assert(len == jvms->loc_size(), "live map consistent with locals map");
 817     for (int local = 0; local < len; local++) {
 818       if (!live_locals.at(local) && map->local(jvms, local) != top()) {
 819         if (PrintMiscellaneous && (Verbose || WizardMode)) {
 820           tty->print_cr("Zombie local %d: ", local);
 821           jvms->dump();
 822         }
 823         return false;
 824       }
 825     }
 826   }
 827   return true;
 828 }

 832 // Helper function for enforcing certain bytecodes to reexecute if
 833 // deoptimization happens
 834 static bool should_reexecute_implied_by_bytecode(JVMState *jvms, bool is_anewarray) {
 835   ciMethod* cur_method = jvms->method();
 836   int       cur_bci   = jvms->bci();
 837   if (cur_method != NULL && cur_bci != InvocationEntryBci) {
 838     Bytecodes::Code code = cur_method->java_code_at_bci(cur_bci);
 839     return Interpreter::bytecode_should_reexecute(code) ||
 840            (is_anewarray && (code == Bytecodes::_multianewarray));
 841     // Reexecute _multianewarray bytecode which was replaced with
 842     // sequence of [a]newarray. See Parse::do_multianewarray().
 843     //
 844     // Note: interpreter should not have it set since this optimization
 845     // is limited by dimensions and guarded by flag so in some cases
 846     // multianewarray() runtime calls will be generated and
 847     // the bytecode should not be reexecutes (stack will not be reset).
 848   } else {
 849     return false;
 850   }
 851 }

 853 // Helper function for adding JVMState and debug information to node
 854 void GraphKit::add_safepoint_edges(SafePointNode* call, bool must_throw) {
 855   // Add the safepoint edges to the call (or other safepoint).
 856 
 857   // Make sure dead locals are set to top.  This
 858   // should help register allocation time and cut down on the size
 859   // of the deoptimization information.
 860   assert(dead_locals_are_killed(), "garbage in debug info before safepoint");
 861 
 862   // Walk the inline list to fill in the correct set of JVMState's
 863   // Also fill in the associated edges for each JVMState.
 864 
 865   // If the bytecode needs to be reexecuted we need to put
 866   // the arguments back on the stack.
 867   const bool should_reexecute = jvms()->should_reexecute();
 868   JVMState* youngest_jvms = should_reexecute ? sync_jvms_for_reexecute() : sync_jvms();
 869 
 870   // NOTE: set_bci (called from sync_jvms) might reset the reexecute bit to
 871   // undefined if the bci is different.  This is normal for Parse but it
 872   // should not happen for LibraryCallKit because only one bci is processed.
 873   assert(!is_LibraryCallKit() || (jvms()->should_reexecute() == should_reexecute),
 874          "in LibraryCallKit the reexecute bit should not change");
 875 
 876   // If we are guaranteed to throw, we can prune everything but the
 877   // input to the current bytecode.
 878   bool can_prune_locals = false;
 879   uint stack_slots_not_pruned = 0;
 880   int inputs = 0, depth = 0;
 881   if (must_throw) {
 882     assert(method() == youngest_jvms->method(), "sanity");
 883     if (compute_stack_effects(inputs, depth)) {
 884       can_prune_locals = true;
 885       stack_slots_not_pruned = inputs;
 886     }
 887   }
 888 
 889   if (env()->should_retain_local_variables()) {
 890     // At any safepoint, this method can get breakpointed, which would
 891     // then require an immediate deoptimization.
 892     can_prune_locals = false;  // do not prune locals
 893     stack_slots_not_pruned = 0;
 894   }
 895 
 896   // do not scribble on the input jvms
 897   JVMState* out_jvms = youngest_jvms->clone_deep(C);
 898   call->set_jvms(out_jvms); // Start jvms list for call node
 899 
 900   // For a known set of bytecodes, the interpreter should reexecute them if
 901   // deoptimization happens. We set the reexecute state for them here
 902   if (out_jvms->is_reexecute_undefined() && //don't change if already specified
 903       should_reexecute_implied_by_bytecode(out_jvms, call->is_AllocateArray())) {
 904     out_jvms->set_should_reexecute(true); //NOTE: youngest_jvms not changed
 905   }
 906 
 907   // Presize the call:
 908   DEBUG_ONLY(uint non_debug_edges = call->req());
 909   call->add_req_batch(top(), youngest_jvms->debug_depth());
 910   assert(call->req() == non_debug_edges + youngest_jvms->debug_depth(), "");
 911 
 912   // Set up edges so that the call looks like this:
 913   //  Call [state:] ctl io mem fptr retadr
 914   //       [parms:] parm0 ... parmN
 915   //       [root:]  loc0 ... locN stk0 ... stkSP mon0 obj0 ... monN objN
 916   //    [...mid:]   loc0 ... locN stk0 ... stkSP mon0 obj0 ... monN objN [...]
 917   //       [young:] loc0 ... locN stk0 ... stkSP mon0 obj0 ... monN objN
 918   // Note that caller debug info precedes callee debug info.
 919 
 920   // Fill pointer walks backwards from "young:" to "root:" in the diagram above:
 921   uint debug_ptr = call->req();
 922 
 923   // Loop over the map input edges associated with jvms, add them
 924   // to the call node, & reset all offsets to match call node array.
 925   for (JVMState* in_jvms = youngest_jvms; in_jvms != NULL; ) {
 926     uint debug_end   = debug_ptr;
 927     uint debug_start = debug_ptr - in_jvms->debug_size();
 928     debug_ptr = debug_start;  // back up the ptr
 929 
 930     uint p = debug_start;  // walks forward in [debug_start, debug_end)
 931     uint j, k, l;
 932     SafePointNode* in_map = in_jvms->map();
 933     out_jvms->set_map(call);
 934 
 935     if (can_prune_locals) {
 936       assert(in_jvms->method() == out_jvms->method(), "sanity");
 937       // If the current throw can reach an exception handler in this JVMS,
 938       // then we must keep everything live that can reach that handler.
 939       // As a quick and dirty approximation, we look for any handlers at all.
 940       if (in_jvms->method()->has_exception_handlers()) {
 941         can_prune_locals = false;
 942       }
 943     }
 944 
 945     // Add the Locals
 946     k = in_jvms->locoff();
 947     l = in_jvms->loc_size();
 948     out_jvms->set_locoff(p);
 949     if (!can_prune_locals) {
 950       for (j = 0; j < l; j++)
 951         call->set_req(p++, in_map->in(k+j));
 952     } else {
 953       p += l;  // already set to top above by add_req_batch
 954     }
 955 
 956     // Add the Expression Stack
 957     k = in_jvms->stkoff();
 958     l = in_jvms->sp();
 959     out_jvms->set_stkoff(p);
 960     if (!can_prune_locals) {
 961       for (j = 0; j < l; j++)
 962         call->set_req(p++, in_map->in(k+j));
 963     } else if (can_prune_locals && stack_slots_not_pruned != 0) {
 964       // Divide stack into {S0,...,S1}, where S0 is set to top.
 965       uint s1 = stack_slots_not_pruned;
 966       stack_slots_not_pruned = 0;  // for next iteration
 967       if (s1 > l)  s1 = l;
 968       uint s0 = l - s1;
 969       p += s0;  // skip the tops preinstalled by add_req_batch
 970       for (j = s0; j < l; j++)
 971         call->set_req(p++, in_map->in(k+j));
 972     } else {
 973       p += l;  // already set to top above by add_req_batch
 974     }
 975 
 976     // Add the Monitors
 977     k = in_jvms->monoff();
 978     l = in_jvms->mon_size();
 979     out_jvms->set_monoff(p);
 980     for (j = 0; j < l; j++)
 981       call->set_req(p++, in_map->in(k+j));
 982 
 983     // Copy any scalar object fields.
 984     k = in_jvms->scloff();
 985     l = in_jvms->scl_size();
 986     out_jvms->set_scloff(p);
 987     for (j = 0; j < l; j++)
 988       call->set_req(p++, in_map->in(k+j));
 989 
 990     // Finish the new jvms.
 991     out_jvms->set_endoff(p);
 992 
 993     assert(out_jvms->endoff()     == debug_end,             "fill ptr must match");
 994     assert(out_jvms->depth()      == in_jvms->depth(),      "depth must match");
 995     assert(out_jvms->loc_size()   == in_jvms->loc_size(),   "size must match");
 996     assert(out_jvms->mon_size()   == in_jvms->mon_size(),   "size must match");
 997     assert(out_jvms->scl_size()   == in_jvms->scl_size(),   "size must match");
 998     assert(out_jvms->debug_size() == in_jvms->debug_size(), "size must match");
 999 
1000     // Update the two tail pointers in parallel.
1001     out_jvms = out_jvms->caller();
1002     in_jvms  = in_jvms->caller();
1003   }
1004 
1005   assert(debug_ptr == non_debug_edges, "debug info must fit exactly");
1006 
1007   // Test the correctness of JVMState::debug_xxx accessors:
1008   assert(call->jvms()->debug_start() == non_debug_edges, "");
1009   assert(call->jvms()->debug_end()   == call->req(), "");
1010   assert(call->jvms()->debug_depth() == call->req() - non_debug_edges, "");
1011 }

1013 bool GraphKit::compute_stack_effects(int& inputs, int& depth) {
1014   Bytecodes::Code code = java_bc();
1015   if (code == Bytecodes::_wide) {
1016     code = method()->java_code_at_bci(bci() + 1);
1017   }
1018 
1019   BasicType rtype = T_ILLEGAL;
1020   int       rsize = 0;
1021 
1022   if (code != Bytecodes::_illegal) {
1023     depth = Bytecodes::depth(code); // checkcast=0, athrow=-1
1024     rtype = Bytecodes::result_type(code); // checkcast=P, athrow=V
1025     if (rtype < T_CONFLICT)
1026       rsize = type2size[rtype];
1027   }
1028 
1029   switch (code) {
1030   case Bytecodes::_illegal:
1031     return false;
1032 
1033   case Bytecodes::_ldc:
1034   case Bytecodes::_ldc_w:
1035   case Bytecodes::_ldc2_w:
1036     inputs = 0;
1037     break;
1038 
1039   case Bytecodes::_dup:         inputs = 1;  break;
1040   case Bytecodes::_dup_x1:      inputs = 2;  break;
1041   case Bytecodes::_dup_x2:      inputs = 3;  break;
1042   case Bytecodes::_dup2:        inputs = 2;  break;
1043   case Bytecodes::_dup2_x1:     inputs = 3;  break;
1044   case Bytecodes::_dup2_x2:     inputs = 4;  break;
1045   case Bytecodes::_swap:        inputs = 2;  break;
1046   case Bytecodes::_arraylength: inputs = 1;  break;
1047 
1048   case Bytecodes::_getstatic:
1049   case Bytecodes::_putstatic:
1050   case Bytecodes::_getfield:
1051   case Bytecodes::_putfield:
1052     {
1053       bool ignored_will_link;
1054       ciField* field = method()->get_field_at_bci(bci(), ignored_will_link);
1055       int      size  = field->type()->size();
1056       bool is_get = (depth >= 0), is_static = (depth & 1);
1057       inputs = (is_static ? 0 : 1);
1058       if (is_get) {
1059         depth = size - inputs;
1060       } else {
1061         inputs += size;        // putxxx pops the value from the stack
1062         depth = - inputs;
1063       }
1064     }
1065     break;
1066 
1067   case Bytecodes::_invokevirtual:
1068   case Bytecodes::_invokespecial:
1069   case Bytecodes::_invokestatic:
1070   case Bytecodes::_invokedynamic:
1071   case Bytecodes::_invokeinterface:
1072     {
1073       bool ignored_will_link;
1074       ciSignature* declared_signature = NULL;
1075       ciMethod* ignored_callee = method()->get_method_at_bci(bci(), ignored_will_link, &declared_signature);
1076       assert(declared_signature != NULL, "cannot be null");
1077       inputs   = declared_signature->arg_size_for_bc(code);
1078       int size = declared_signature->return_type()->size();
1079       depth = size - inputs;
1080     }
1081     break;
1082 
1083   case Bytecodes::_multianewarray:
1084     {
1085       ciBytecodeStream iter(method());
1086       iter.reset_to_bci(bci());
1087       iter.next();
1088       inputs = iter.get_dimensions();
1089       assert(rsize == 1, "");
1090       depth = rsize - inputs;
1091     }
1092     break;
1093 
1094   case Bytecodes::_withfield: {
1095     bool ignored_will_link;
1096     ciField* field = method()->get_field_at_bci(bci(), ignored_will_link);
1097     int      size  = field->type()->size();
1098     inputs = size+1;
1099     depth = rsize - inputs;
1100     break;
1101   }
1102 
1103   case Bytecodes::_ireturn:
1104   case Bytecodes::_lreturn:
1105   case Bytecodes::_freturn:
1106   case Bytecodes::_dreturn:
1107   case Bytecodes::_areturn:
1108     assert(rsize == -depth, "");
1109     inputs = rsize;
1110     break;
1111 
1112   case Bytecodes::_jsr:
1113   case Bytecodes::_jsr_w:
1114     inputs = 0;
1115     depth  = 1;                  // S.B. depth=1, not zero
1116     break;
1117 
1118   default:
1119     // bytecode produces a typed result
1120     inputs = rsize - depth;
1121     assert(inputs >= 0, "");
1122     break;
1123   }
1124 
1125 #ifdef ASSERT
1126   // spot check
1127   int outputs = depth + inputs;
1128   assert(outputs >= 0, "sanity");
1129   switch (code) {
1130   case Bytecodes::_checkcast: assert(inputs == 1 && outputs == 1, ""); break;
1131   case Bytecodes::_athrow:    assert(inputs == 1 && outputs == 0, ""); break;
1132   case Bytecodes::_aload_0:   assert(inputs == 0 && outputs == 1, ""); break;
1133   case Bytecodes::_return:    assert(inputs == 0 && outputs == 0, ""); break;
1134   case Bytecodes::_drem:      assert(inputs == 4 && outputs == 2, ""); break;
1135   default:                    break;
1136   }
1137 #endif //ASSERT
1138 
1139   return true;
1140 }

1206 //------------------------------do_null_check----------------------------------
1207 // Helper function to do a NULL pointer check.  Returned value is
1208 // the incoming address with NULL casted away.  You are allowed to use the
1209 // not-null value only if you are control dependent on the test.
1210 #ifndef PRODUCT
1211 extern int explicit_null_checks_inserted,
1212            explicit_null_checks_elided;
1213 #endif
1214 Node* GraphKit::null_check_common(Node* value, BasicType type,
1215                                   // optional arguments for variations:
1216                                   bool assert_null,
1217                                   Node* *null_control,
1218                                   bool speculative) {
1219   assert(!assert_null || null_control == NULL, "not both at once");
1220   if (stopped())  return top();
1221   NOT_PRODUCT(explicit_null_checks_inserted++);
1222 
1223   // Construct NULL check
1224   Node *chk = NULL;
1225   switch(type) {
1226     case T_LONG   : chk = new CmpLNode(value, _gvn.zerocon(T_LONG)); break;
1227     case T_INT    : chk = new CmpINode(value, _gvn.intcon(0)); break;
1228     case T_VALUETYPE : // fall through
1229     case T_ARRAY  : // fall through
1230       type = T_OBJECT;  // simplify further tests
1231     case T_OBJECT : {
1232       const Type *t = _gvn.type( value );
1233 
1234       const TypeOopPtr* tp = t->isa_oopptr();
1235       if (tp != NULL && tp->klass() != NULL && !tp->klass()->is_loaded()
1236           // Only for do_null_check, not any of its siblings:
1237           && !assert_null && null_control == NULL) {
1238         // Usually, any field access or invocation on an unloaded oop type
1239         // will simply fail to link, since the statically linked class is
1240         // likely also to be unloaded.  However, in -Xcomp mode, sometimes
1241         // the static class is loaded but the sharper oop type is not.
1242         // Rather than checking for this obscure case in lots of places,
1243         // we simply observe that a null check on an unloaded class
1244         // will always be followed by a nonsense operation, so we
1245         // can just issue the uncommon trap here.
1246         // Our access to the unloaded class will only be correct
1247         // after it has been loaded and initialized, which requires
1248         // a trip through the interpreter.
1249 #ifndef PRODUCT
1250         if (WizardMode) { tty->print("Null check of unloaded "); tp->klass()->print(); tty->cr(); }
1251 #endif
1252         uncommon_trap(Deoptimization::Reason_unloaded,
1253                       Deoptimization::Action_reinterpret,
1254                       tp->klass(), "!loaded");
1255         return top();
1256       }
1257 
1258       if (assert_null) {
1259         // See if the type is contained in NULL_PTR.
1260         // If so, then the value is already null.
1261         if (t->higher_equal(TypePtr::NULL_PTR)) {
1262           NOT_PRODUCT(explicit_null_checks_elided++);
1263           return value;           // Elided null assert quickly!
1264         }
1265       } else {
1266         // See if mixing in the NULL pointer changes type.
1267         // If so, then the NULL pointer was not allowed in the original
1268         // type.  In other words, "value" was not-null.
1269         if (t->meet(TypePtr::NULL_PTR) != t->remove_speculative()) {
1270           // same as: if (!TypePtr::NULL_PTR->higher_equal(t)) ...
1271           NOT_PRODUCT(explicit_null_checks_elided++);
1272           return value;           // Elided null check quickly!
1273         }
1274       }
1275       chk = new CmpPNode( value, null() );
1276       break;
1277     }
1278 
1279     default:
1280       fatal("unexpected type: %s", type2name(type));
1281   }
1282   assert(chk != NULL, "sanity check");
1283   chk = _gvn.transform(chk);
1284 
1285   BoolTest::mask btest = assert_null ? BoolTest::eq : BoolTest::ne;
1286   BoolNode *btst = new BoolNode( chk, btest);
1287   Node   *tst = _gvn.transform( btst );
1288 
1289   //-----------
1290   // if peephole optimizations occurred, a prior test existed.
1291   // If a prior test existed, maybe it dominates as we can avoid this test.
1292   if (tst != btst && type == T_OBJECT) {
1293     // At this point we want to scan up the CFG to see if we can
1294     // find an identical test (and so avoid this test altogether).
1295     Node *cfg = control();
1296     int depth = 0;
1297     while( depth < 16 ) {       // Limit search depth for speed
1298       if( cfg->Opcode() == Op_IfTrue &&
1299           cfg->in(0)->in(1) == tst ) {
1300         // Found prior test.  Use "cast_not_null" to construct an identical
1301         // CastPP (and hence hash to) as already exists for the prior test.
1302         // Return that casted value.
1303         if (assert_null) {
1304           replace_in_map(value, null());
1305           return null();  // do not issue the redundant test
1306         }
1307         Node *oldcontrol = control();
1308         set_control(cfg);
1309         Node *res = cast_not_null(value);
1310         set_control(oldcontrol);
1311         NOT_PRODUCT(explicit_null_checks_elided++);
1312         return res;
1313       }
1314       cfg = IfNode::up_one_dom(cfg, /*linear_only=*/ true);
1315       if (cfg == NULL)  break;  // Quit at region nodes
1316       depth++;
1317     }
1318   }
1319 
1320   //-----------
1321   // Branch to failure if null
1322   float ok_prob = PROB_MAX;  // a priori estimate:  nulls never happen
1323   Deoptimization::DeoptReason reason;
1324   if (assert_null) {
1325     reason = Deoptimization::reason_null_assert(speculative);
1326   } else if (type == T_OBJECT) {
1327     reason = Deoptimization::reason_null_check(speculative);
1328   } else {
1329     reason = Deoptimization::Reason_div0_check;
1330   }
1331   // %%% Since Reason_unhandled is not recorded on a per-bytecode basis,
1332   // ciMethodData::has_trap_at will return a conservative -1 if any
1333   // must-be-null assertion has failed.  This could cause performance
1334   // problems for a method after its first do_null_assert failure.
1335   // Consider using 'Reason_class_check' instead?
1336 
1337   // To cause an implicit null check, we set the not-null probability
1338   // to the maximum (PROB_MAX).  For an explicit check the probability
1339   // is set to a smaller value.
1340   if (null_control != NULL || too_many_traps(reason)) {
1341     // probability is less likely
1342     ok_prob =  PROB_LIKELY_MAG(3);
1343   } else if (!assert_null &&
1344              (ImplicitNullCheckThreshold > 0) &&
1345              method() != NULL &&
1346              (method()->method_data()->trap_count(reason)
1347               >= (uint)ImplicitNullCheckThreshold)) {
1348     ok_prob =  PROB_LIKELY_MAG(3);
1349   }
1350 
1351   if (null_control != NULL) {
1352     IfNode* iff = create_and_map_if(control(), tst, ok_prob, COUNT_UNKNOWN);
1353     Node* null_true = _gvn.transform( new IfFalseNode(iff));
1354     set_control(      _gvn.transform( new IfTrueNode(iff)));
1355 #ifndef PRODUCT
1356     if (null_true == top()) {
1357       explicit_null_checks_elided++;
1358     }
1359 #endif
1360     (*null_control) = null_true;
1361   } else {
1362     BuildCutout unless(this, tst, ok_prob);
1363     // Check for optimizer eliding test at parse time
1364     if (stopped()) {
1365       // Failure not possible; do not bother making uncommon trap.
1366       NOT_PRODUCT(explicit_null_checks_elided++);
1367     } else if (assert_null) {
1368       uncommon_trap(reason,
1369                     Deoptimization::Action_make_not_entrant,
1370                     NULL, "assert_null");
1371     } else {
1372       replace_in_map(value, zerocon(type));
1373       builtin_throw(reason);
1374     }
1375   }
1376 
1377   // Must throw exception, fall-thru not possible?
1378   if (stopped()) {
1379     return top();               // No result
1380   }
1381 
1382   if (assert_null) {
1383     // Cast obj to null on this path.
1384     replace_in_map(value, zerocon(type));
1385     return zerocon(type);
1386   }
1387 
1388   // Cast obj to not-null on this path, if there is no null_control.
1389   // (If there is a null_control, a non-null value may come back to haunt us.)
1390   if (type == T_OBJECT) {
1391     Node* cast = cast_not_null(value, false);
1392     if (null_control == NULL || (*null_control) == top())
1393       replace_in_map(value, cast);
1394     value = cast;
1395   }
1396 
1397   return value;
1398 }

1416 //------------------------------cast_not_null----------------------------------
1417 // Cast obj to not-null on this path
1418 Node* GraphKit::cast_not_null(Node* obj, bool do_replace_in_map) {
1419   if (obj->is_ValueType()) {
1420     return obj;
1421   }
1422   const Type *t = _gvn.type(obj);
1423   const Type *t_not_null = t->join_speculative(TypePtr::NOTNULL);
1424   // Object is already not-null?
1425   if( t == t_not_null ) return obj;
1426 
1427   Node *cast = new CastPPNode(obj,t_not_null);
1428   cast->init_req(0, control());
1429   cast = _gvn.transform( cast );
1430 
1431   // Scan for instances of 'obj' in the current JVM mapping.
1432   // These instances are known to be not-null after the test.
1433   if (do_replace_in_map)
1434     replace_in_map(obj, cast);
1435 
1436   return cast;                  // Return casted value
1437 }

1439 // Sometimes in intrinsics, we implicitly know an object is not null
1440 // (there's no actual null check) so we can cast it to not null. In
1441 // the course of optimizations, the input to the cast can become null.
1442 // In that case that data path will die and we need the control path
1443 // to become dead as well to keep the graph consistent. So we have to
1444 // add a check for null for which one branch can't be taken. It uses
1445 // an Opaque4 node that will cause the check to be removed after loop
1446 // opts so the test goes away and the compiled code doesn't execute a
1447 // useless check.
1448 Node* GraphKit::must_be_not_null(Node* value, bool do_replace_in_map) {
1449   if (!TypePtr::NULL_PTR->higher_equal(_gvn.type(value))) {
1450     return value;
1451   }
1452   Node* chk = _gvn.transform(new CmpPNode(value, null()));
1453   Node *tst = _gvn.transform(new BoolNode(chk, BoolTest::ne));
1454   Node* opaq = _gvn.transform(new Opaque4Node(C, tst, intcon(1)));
1455   IfNode *iff = new IfNode(control(), opaq, PROB_MAX, COUNT_UNKNOWN);
1456   _gvn.set_type(iff, iff->Value(&_gvn));
1457   Node *if_f = _gvn.transform(new IfFalseNode(iff));
1458   Node *frame = _gvn.transform(new ParmNode(C->start(), TypeFunc::FramePtr));
1459   Node* halt = _gvn.transform(new HaltNode(if_f, frame, "unexpected null in intrinsic"));
1460   C->root()->add_req(halt);
1461   Node *if_t = _gvn.transform(new IfTrueNode(iff));
1462   set_control(if_t);
1463   return cast_not_null(value, do_replace_in_map);
1464 }

1467 //--------------------------replace_in_map-------------------------------------
1468 void GraphKit::replace_in_map(Node* old, Node* neww) {
1469   if (old == neww) {
1470     return;
1471   }
1472 
1473   map()->replace_edge(old, neww);
1474 
1475   // Note: This operation potentially replaces any edge
1476   // on the map.  This includes locals, stack, and monitors
1477   // of the current (innermost) JVM state.
1478 
1479   // don't let inconsistent types from profiling escape this
1480   // method
1481 
1482   const Type* told = _gvn.type(old);
1483   const Type* tnew = _gvn.type(neww);
1484 
1485   if (!tnew->higher_equal(told)) {
1486     return;
1487   }
1488 
1489   map()->record_replaced_node(old, neww);
1490 }

1600 Node* GraphKit::access_store_at(Node* obj,
1601                                 Node* adr,
1602                                 const TypePtr* adr_type,
1603                                 Node* val,
1604                                 const Type* val_type,
1605                                 BasicType bt,
1606                                 DecoratorSet decorators,
1607                                 bool safe_for_replace) {
1608   // Transformation of a value which could be NULL pointer (CastPP #NULL)
1609   // could be delayed during Parse (for example, in adjust_map_after_if()).
1610   // Execute transformation here to avoid barrier generation in such case.
1611   if (_gvn.type(val) == TypePtr::NULL_PTR) {
1612     val = _gvn.makecon(TypePtr::NULL_PTR);
1613   }
1614 
1615   if (stopped()) {
1616     return top(); // Dead path ?
1617   }
1618 
1619   assert(val != NULL, "not dead path");
1620   if (val->is_ValueType()) {
1621     // Store to non-flattened field. Buffer the inline type and make sure
1622     // the store is re-executed if the allocation triggers deoptimization.
1623     PreserveReexecuteState preexecs(this);
1624     jvms()->set_should_reexecute(true);
1625     val = val->as_ValueType()->allocate(this, safe_for_replace)->get_oop();
1626   }

1752 //-------------------------array_element_address-------------------------
1753 Node* GraphKit::array_element_address(Node* ary, Node* idx, BasicType elembt,
1754                                       const TypeInt* sizetype, Node* ctrl) {
1755   uint shift  = exact_log2(type2aelembytes(elembt));
1756   ciKlass* arytype_klass = _gvn.type(ary)->is_aryptr()->klass();
1757   if (arytype_klass != NULL && arytype_klass->is_value_array_klass()) {
1758     ciValueArrayKlass* vak = arytype_klass->as_value_array_klass();
1759     shift = vak->log2_element_size();
1760   }
1761   uint header = arrayOopDesc::base_offset_in_bytes(elembt);
1762 
1763   // short-circuit a common case (saves lots of confusing waste motion)
1764   jint idx_con = find_int_con(idx, -1);

1766     intptr_t offset = header + ((intptr_t)idx_con << shift);
1767     return basic_plus_adr(ary, offset);
1768   }
1769 
1770   // must be correct type for alignment purposes
1771   Node* base  = basic_plus_adr(ary, header);
1772   idx = Compile::conv_I2X_index(&_gvn, idx, sizetype, ctrl);
1773   Node* scale = _gvn.transform( new LShiftXNode(idx, intcon(shift)) );
1774   return basic_plus_adr(ary, base, scale);

1790 //-------------------------set_arguments_for_java_call-------------------------
1791 // Arguments (pre-popped from the stack) are taken from the JVMS.
1792 void GraphKit::set_arguments_for_java_call(CallJavaNode* call, bool is_late_inline) {
1793   PreserveReexecuteState preexecs(this);
1794   if (EnableValhalla) {
1795     // Make sure the call is re-executed, if buffering of value type arguments triggers deoptimization
1796     jvms()->set_should_reexecute(true);
1797     int arg_size = method()->get_declared_signature_at_bci(bci())->arg_size_for_bc(java_bc());
1798     inc_sp(arg_size);
1799   }
1800   // Add the call arguments
1801   const TypeTuple* domain = call->tf()->domain_sig();
1802   ExtendedSignature sig_cc = ExtendedSignature(call->method()->get_sig_cc(), SigEntryFilter());
1803   uint nargs = domain->cnt();
1804   for (uint i = TypeFunc::Parms, idx = TypeFunc::Parms; i < nargs; i++) {
1805     Node* arg = argument(i-TypeFunc::Parms);
1806     const Type* t = domain->field_at(i);
1807     if (call->method()->has_scalarized_args() && t->is_valuetypeptr() && !t->maybe_null()) {
1808       // We don't pass value type arguments by reference but instead pass each field of the value type
1809       ValueTypeNode* vt = arg->as_ValueType();
1810       vt->pass_fields(this, call, sig_cc, idx);
1811       // If a value type argument is passed as fields, attach the Method* to the call site
1812       // to be able to access the extended signature later via attached_method_before_pc().
1813       // For example, see CompiledMethod::preserve_callee_argument_oops().
1814       call->set_override_symbolic_info(true);
1815       continue;
1816     } else if (arg->is_ValueType()) {
1817       // Pass value type argument via oop to callee
1818       if (is_late_inline) {
1819         arg = ValueTypePtrNode::make_from_value_type(this, arg->as_ValueType());
1820       } else {
1821         arg = arg->as_ValueType()->allocate(this)->get_oop();
1822       }
1823     }
1824     call->init_req(idx++, arg);
1825     // Skip reserved arguments
1826     BasicType bt = t->basic_type();
1827     while (SigEntry::next_is_reserved(sig_cc, bt, true)) {
1828       call->init_req(idx++, top());
1829       if (type2size[bt] == 2) {
1830         call->init_req(idx++, top());
1831       }
1832     }
1833   }
1834 }

1870   // Note:  Since any out-of-line call can produce an exception,
1871   // we always insert an I_O projection from the call into the result.

1878     // the fall through path.

1880     set_all_memory(_gvn.transform( new ProjNode(call, TypeFunc::Memory) ));
1881   }
1882

1884   Node* ret;

1898     ret = _gvn.transform(new ProjNode(call, TypeFunc::Parms));
1899   }

1904 //--------------------set_predefined_input_for_runtime_call--------------------
1905 // Reading and setting the memory state is way conservative here.
1906 // The real problem is that I am not doing real Type analysis on memory,
1907 // so I cannot distinguish card mark stores from other stores.  Across a GC
1908 // point the Store Barrier and the card mark memory has to agree.  I cannot
1909 // have a card mark store and its barrier split across the GC point from
1910 // either above or below.  Here I get that to happen by reading ALL of memory.
1911 // A better answer would be to separate out card marks from other memory.
1912 // For now, return the input memory state, so that it can be reused
1913 // after the call, if this call has restricted memory effects.
1914 Node* GraphKit::set_predefined_input_for_runtime_call(SafePointNode* call, Node* narrow_mem) {
1915   // Set fixed predefined input arguments
1916   Node* memory = reset_memory();
1917   Node* m = narrow_mem == NULL ? memory : narrow_mem;
1918   call->init_req( TypeFunc::Control,   control()  );
1919   call->init_req( TypeFunc::I_O,       top()      ); // does no i/o
1920   call->init_req( TypeFunc::Memory,    m          ); // may gc ptrs
1921   call->init_req( TypeFunc::FramePtr,  frameptr() );
1922   call->init_req( TypeFunc::ReturnAdr, top()      );
1923   return memory;
1924 }

1926 //-------------------set_predefined_output_for_runtime_call--------------------
1927 // Set control and memory (not i_o) from the call.
1928 // If keep_mem is not NULL, use it for the output state,
1929 // except for the RawPtr output of the call, if hook_mem is TypeRawPtr::BOTTOM.
1930 // If hook_mem is NULL, this call produces no memory effects at all.
1931 // If hook_mem is a Java-visible memory slice (such as arraycopy operands),
1932 // then only that memory slice is taken from the call.
1933 // In the last case, we must put an appropriate memory barrier before
1934 // the call, so as to create the correct anti-dependencies on loads
1935 // preceding the call.
1936 void GraphKit::set_predefined_output_for_runtime_call(Node* call,
1937                                                       Node* keep_mem,
1938                                                       const TypePtr* hook_mem) {
1939   // no i/o
1940   set_control(_gvn.transform( new ProjNode(call,TypeFunc::Control) ));
1941   if (keep_mem) {
1942     // First clone the existing memory state
1943     set_all_memory(keep_mem);
1944     if (hook_mem != NULL) {
1945       // Make memory for the call
1946       Node* mem = _gvn.transform( new ProjNode(call, TypeFunc::Memory) );
1947       // Set the RawPtr memory state only.  This covers all the heap top/GC stuff
1948       // We also use hook_mem to extract specific effects from arraycopy stubs.
1949       set_memory(mem, hook_mem);
1950     }
1951     // ...else the call has NO memory effects.
1952 
1953     // Make sure the call advertises its memory effects precisely.
1954     // This lets us build accurate anti-dependences in gcm.cpp.
1955     assert(C->alias_type(call->adr_type()) == C->alias_type(hook_mem),
1956            "call node must be constructed correctly");
1957   } else {
1958     assert(hook_mem == NULL, "");
1959     // This is not a "slow path" call; all memory comes from the call.
1960     set_all_memory_call(call);
1961   }
1962 }

1977 // Replace the call with the current state of the kit.
1978 void GraphKit::replace_call(CallNode* call, Node* result, bool do_replaced_nodes) {
1979   JVMState* ejvms = NULL;
1980   if (has_exceptions()) {
1981     ejvms = transfer_exceptions_into_jvms();
1982   }
1983 
1984   ReplacedNodes replaced_nodes = map()->replaced_nodes();
1985   ReplacedNodes replaced_nodes_exception;
1986   Node* ex_ctl = top();
1987 
1988   SafePointNode* final_state = stop();
1989 
1990   // Find all the needed outputs of this call
1991   CallProjections* callprojs = call->extract_projections(true);
1992 
1993   Unique_Node_List wl;
1994   Node* init_mem = call->in(TypeFunc::Memory);
1995   Node* final_mem = final_state->in(TypeFunc::Memory);
1996   Node* final_ctl = final_state->in(TypeFunc::Control);
1997   Node* final_io = final_state->in(TypeFunc::I_O);
1998 
1999   // Replace all the old call edges with the edges from the inlining result
2000   if (callprojs->fallthrough_catchproj != NULL) {
2001     C->gvn_replace_by(callprojs->fallthrough_catchproj, final_ctl);
2002   }
2003   if (callprojs->fallthrough_memproj != NULL) {
2004     if (final_mem->is_MergeMem()) {
2005       // Parser's exits MergeMem was not transformed but may be optimized
2006       final_mem = _gvn.transform(final_mem);
2007     }
2008     C->gvn_replace_by(callprojs->fallthrough_memproj,   final_mem);
2009     add_mergemem_users_to_worklist(wl, final_mem);
2010   }
2011   if (callprojs->fallthrough_ioproj != NULL) {
2012     C->gvn_replace_by(callprojs->fallthrough_ioproj,    final_io);
2013   }
2014 
2015   // Replace the result with the new result if it exists and is used
2016   if (callprojs->resproj[0] != NULL && result != NULL) {
2017     assert(callprojs->nb_resproj == 1, "unexpected number of results");
2018     C->gvn_replace_by(callprojs->resproj[0], result);
2019   }
2020 
2021   if (ejvms == NULL) {
2022     // No exception edges to simply kill off those paths
2023     if (callprojs->catchall_catchproj != NULL) {
2024       C->gvn_replace_by(callprojs->catchall_catchproj, C->top());
2025     }
2026     if (callprojs->catchall_memproj != NULL) {
2027       C->gvn_replace_by(callprojs->catchall_memproj,   C->top());
2028     }
2029     if (callprojs->catchall_ioproj != NULL) {
2030       C->gvn_replace_by(callprojs->catchall_ioproj,    C->top());
2031     }
2032     // Replace the old exception object with top
2033     if (callprojs->exobj != NULL) {
2034       C->gvn_replace_by(callprojs->exobj, C->top());
2035     }
2036   } else {
2037     GraphKit ekit(ejvms);
2038 
2039     // Load my combined exception state into the kit, with all phis transformed:
2040     SafePointNode* ex_map = ekit.combine_and_pop_all_exception_states();
2041     replaced_nodes_exception = ex_map->replaced_nodes();
2042 
2043     Node* ex_oop = ekit.use_exception_state(ex_map);
2044 
2045     if (callprojs->catchall_catchproj != NULL) {
2046       C->gvn_replace_by(callprojs->catchall_catchproj, ekit.control());
2047       ex_ctl = ekit.control();
2048     }
2049     if (callprojs->catchall_memproj != NULL) {
2050       Node* ex_mem = ekit.reset_memory();
2051       C->gvn_replace_by(callprojs->catchall_memproj,   ex_mem);
2052       add_mergemem_users_to_worklist(wl, ex_mem);
2053     }
2054     if (callprojs->catchall_ioproj != NULL) {
2055       C->gvn_replace_by(callprojs->catchall_ioproj,    ekit.i_o());
2056     }
2057 
2058     // Replace the old exception object with the newly created one
2059     if (callprojs->exobj != NULL) {
2060       C->gvn_replace_by(callprojs->exobj, ex_oop);
2061     }
2062   }
2063 
2064   // Disconnect the call from the graph
2065   call->disconnect_inputs(NULL, C);
2066   C->gvn_replace_by(call, C->top());
2067 
2068   // Clean up any MergeMems that feed other MergeMems since the
2069   // optimizer doesn't like that.
2070   while (wl.size() > 0) {
2071     _gvn.transform(wl.pop());
2072   }
2073 
2074   if (callprojs->fallthrough_catchproj != NULL && !final_ctl->is_top() && do_replaced_nodes) {
2075     replaced_nodes.apply(C, final_ctl);
2076   }
2077   if (!ex_ctl->is_top() && do_replaced_nodes) {
2078     replaced_nodes_exception.apply(C, ex_ctl);
2079   }
2080 }

2102 // uncommon_trap blob.  This helper function inserts a runtime call with the

2104 void GraphKit::uncommon_trap(int trap_request,
2105                              ciKlass* klass, const char* comment,
2106                              bool must_throw,
2107                              bool keep_exact_action) {
2108   if (failing())  stop();
2109   if (stopped())  return; // trap reachable?
2110 
2111   // Note:  If ProfileTraps is true, and if a deopt. actually
2112   // occurs here, the runtime will make sure an MDO exists.  There is
2113   // no need to call method()->ensure_method_data() at this point.
2114 
2115   // Set the stack pointer to the right value for reexecution:
2116   set_sp(reexecute_sp());
2117 
2118 #ifdef ASSERT
2119   if (!must_throw) {
2120     // Make sure the stack has at least enough depth to execute
2121     // the current bytecode.
2122     int inputs, ignored_depth;
2123     if (compute_stack_effects(inputs, ignored_depth)) {
2124       assert(sp() >= inputs, "must have enough JVMS stack to execute %s: sp=%d, inputs=%d",
2125              Bytecodes::name(java_bc()), sp(), inputs);
2126     }
2127   }
2128 #endif
2129 
2130   Deoptimization::DeoptReason reason = Deoptimization::trap_request_reason(trap_request);
2131   Deoptimization::DeoptAction action = Deoptimization::trap_request_action(trap_request);
2132 
2133   switch (action) {
2134   case Deoptimization::Action_maybe_recompile:
2135   case Deoptimization::Action_reinterpret:
2136     // Temporary fix for 6529811 to allow virtual calls to be sure they
2137     // get the chance to go from mono->bi->mega
2138     if (!keep_exact_action &&
2139         Deoptimization::trap_request_index(trap_request) < 0 &&
2140         too_many_recompiles(reason)) {
2141       // This BCI is causing too many recompilations.

2536 //------------------------------null_check_oop---------------------------------
2537 // Null check oop.  Set null-path control into Region in slot 3.
2538 // Make a cast-not-nullness use the other not-null control.  Return cast.

2540                                bool never_see_null,
2541                                bool safe_for_replace,
2542                                bool speculative) {

2544   (*null_control) = top();

2546

2755 //-------------------------------gen_subtype_check-----------------------------
2756 // Generate a subtyping check.  Takes as input the subtype and supertype.
2757 // Returns 2 values: sets the default control() to the true path and returns
2758 // the false path.  Only reads invariant memory; sets no (visible) memory.
2759 // The PartialSubtypeCheckNode sets the hidden 1-word cache in the encoding
2760 // but that's not exposed to the optimizer.  This call also doesn't take in an
2761 // Object; if you wish to check an Object you need to load the Object's class
2762 // prior to coming here.
2763 Node* Phase::gen_subtype_check(Node* subklass, Node* superklass, Node** ctrl, Node* mem, PhaseGVN& gvn) {
2764   Compile* C = gvn.C;
2765   if ((*ctrl)->is_top()) {
2766     return C->top();
2767   }
2768 
2769   // Fast check for identical types, perhaps identical constants.
2770   // The types can even be identical non-constants, in cases
2771   // involving Array.newInstance, Object.clone, etc.
2772   if (subklass == superklass)
2773     return C->top();             // false path is dead; no test needed.
2774 
2775   if (gvn.type(superklass)->singleton()) {
2776     ciKlass* superk = gvn.type(superklass)->is_klassptr()->klass();
2777     ciKlass* subk   = gvn.type(subklass)->is_klassptr()->klass();
2778 
2779     // In the common case of an exact superklass, try to fold up the
2780     // test before generating code.  You may ask, why not just generate
2781     // the code and then let it fold up?  The answer is that the generated
2782     // code will necessarily include null checks, which do not always
2783     // completely fold away.  If they are also needless, then they turn
2784     // into a performance loss.  Example:
2785     //    Foo[] fa = blah(); Foo x = fa[0]; fa[1] = x;
2786     // Here, the type of 'fa' is often exact, so the store check
2787     // of fa[1]=x will fold up, without testing the nullness of x.
2788     switch (C->static_subtype_check(superk, subk)) {
2789     case Compile::SSC_always_false:
2790       {
2791         Node* always_fail = *ctrl;
2792         *ctrl = gvn.C->top();
2793         return always_fail;
2794       }
2795     case Compile::SSC_always_true:
2796       return C->top();
2797     case Compile::SSC_easy_test:
2798       {
2799         // Just do a direct pointer compare and be done.
2800         IfNode* iff = gen_subtype_check_compare(*ctrl, subklass, superklass, BoolTest::eq, PROB_STATIC_FREQUENT, gvn, T_ADDRESS);
2801         *ctrl = gvn.transform(new IfTrueNode(iff));
2802         return gvn.transform(new IfFalseNode(iff));
2803       }
2804     case Compile::SSC_full_test:
2805       break;
2806     default:
2807       ShouldNotReachHere();
2808     }
2809   }
2810 
2811   // %%% Possible further optimization:  Even if the superklass is not exact,
2812   // if the subklass is the unique subtype of the superklass, the check
2813   // will always succeed.  We could leave a dependency behind to ensure this.
2814 
2815   // First load the super-klass's check-offset
2816   Node *p1 = gvn.transform(new AddPNode(superklass, superklass, gvn.MakeConX(in_bytes(Klass::super_check_offset_offset()))));
2817   Node* m = C->immutable_memory();
2818   Node *chk_off = gvn.transform(new LoadINode(NULL, m, p1, gvn.type(p1)->is_ptr(), TypeInt::INT, MemNode::unordered));
2819   int cacheoff_con = in_bytes(Klass::secondary_super_cache_offset());
2820   bool might_be_cache = (gvn.find_int_con(chk_off, cacheoff_con) == cacheoff_con);
2821 
2822   // Load from the sub-klass's super-class display list, or a 1-word cache of
2823   // the secondary superclass list, or a failing value with a sentinel offset
2824   // if the super-klass is an interface or exceptionally deep in the Java
2825   // hierarchy and we have to scan the secondary superclass list the hard way.
2826   // Worst-case type is a little odd: NULL is allowed as a result (usually
2827   // klass loads can never produce a NULL).
2828   Node *chk_off_X = chk_off;
2829 #ifdef _LP64
2830   chk_off_X = gvn.transform(new ConvI2LNode(chk_off_X));
2831 #endif
2832   Node *p2 = gvn.transform(new AddPNode(subklass,subklass,chk_off_X));
2833   // For some types like interfaces the following loadKlass is from a 1-word
2834   // cache which is mutable so can't use immutable memory.  Other
2835   // types load from the super-class display table which is immutable.
2836   Node *kmem = C->immutable_memory();
2837   // secondary_super_cache is not immutable but can be treated as such because:
2838   // - no ideal node writes to it in a way that could cause an
2839   //   incorrect/missed optimization of the following Load.
2840   // - it's a cache so, worse case, not reading the latest value
2841   //   wouldn't cause incorrect execution
2842   if (might_be_cache && mem != NULL) {
2843     kmem = mem->is_MergeMem() ? mem->as_MergeMem()->memory_at(C->get_alias_index(gvn.type(p2)->is_ptr())) : mem;
2844   }
2845   Node *nkls = gvn.transform(LoadKlassNode::make(gvn, NULL, kmem, p2, gvn.type(p2)->is_ptr(), TypeKlassPtr::OBJECT_OR_NULL));
2846 
2847   // Compile speed common case: ARE a subtype and we canNOT fail
2848   if( superklass == nkls )
2849     return C->top();             // false path is dead; no test needed.
2850 
2851   // See if we get an immediate positive hit.  Happens roughly 83% of the
2852   // time.  Test to see if the value loaded just previously from the subklass
2853   // is exactly the superklass.
2854   IfNode *iff1 = gen_subtype_check_compare(*ctrl, superklass, nkls, BoolTest::eq, PROB_LIKELY(0.83f), gvn, T_ADDRESS);
2855   Node *iftrue1 = gvn.transform( new IfTrueNode (iff1));
2856   *ctrl = gvn.transform(new IfFalseNode(iff1));
2857 
2858   // Compile speed common case: Check for being deterministic right now.  If
2859   // chk_off is a constant and not equal to cacheoff then we are NOT a
2860   // subklass.  In this case we need exactly the 1 test above and we can
2861   // return those results immediately.
2862   if (!might_be_cache) {
2863     Node* not_subtype_ctrl = *ctrl;
2864     *ctrl = iftrue1; // We need exactly the 1 test above
2865     return not_subtype_ctrl;
2866   }
2867 
2868   // Gather the various success & failures here
2869   RegionNode *r_ok_subtype = new RegionNode(4);
2870   gvn.record_for_igvn(r_ok_subtype);
2871   RegionNode *r_not_subtype = new RegionNode(3);
2872   gvn.record_for_igvn(r_not_subtype);
2873 
2874   r_ok_subtype->init_req(1, iftrue1);
2875 
2876   // Check for immediate negative hit.  Happens roughly 11% of the time (which
2877   // is roughly 63% of the remaining cases).  Test to see if the loaded
2878   // check-offset points into the subklass display list or the 1-element
2879   // cache.  If it points to the display (and NOT the cache) and the display
2880   // missed then it's not a subtype.
2881   Node *cacheoff = gvn.intcon(cacheoff_con);
2882   IfNode *iff2 = gen_subtype_check_compare(*ctrl, chk_off, cacheoff, BoolTest::ne, PROB_LIKELY(0.63f), gvn, T_INT);
2883   r_not_subtype->init_req(1, gvn.transform(new IfTrueNode (iff2)));
2884   *ctrl = gvn.transform(new IfFalseNode(iff2));
2885 
2886   // Check for self.  Very rare to get here, but it is taken 1/3 the time.
2887   // No performance impact (too rare) but allows sharing of secondary arrays
2888   // which has some footprint reduction.
2889   IfNode *iff3 = gen_subtype_check_compare(*ctrl, subklass, superklass, BoolTest::eq, PROB_LIKELY(0.36f), gvn, T_ADDRESS);
2890   r_ok_subtype->init_req(2, gvn.transform(new IfTrueNode(iff3)));
2891   *ctrl = gvn.transform(new IfFalseNode(iff3));
2892 
2893   // -- Roads not taken here: --
2894   // We could also have chosen to perform the self-check at the beginning
2895   // of this code sequence, as the assembler does.  This would not pay off
2896   // the same way, since the optimizer, unlike the assembler, can perform
2897   // static type analysis to fold away many successful self-checks.
2898   // Non-foldable self checks work better here in second position, because
2899   // the initial primary superclass check subsumes a self-check for most
2900   // types.  An exception would be a secondary type like array-of-interface,
2901   // which does not appear in its own primary supertype display.
2902   // Finally, we could have chosen to move the self-check into the
2903   // PartialSubtypeCheckNode, and from there out-of-line in a platform
2904   // dependent manner.  But it is worthwhile to have the check here,
2905   // where it can be perhaps be optimized.  The cost in code space is
2906   // small (register compare, branch).
2907 
2908   // Now do a linear scan of the secondary super-klass array.  Again, no real
2909   // performance impact (too rare) but it's gotta be done.
2910   // Since the code is rarely used, there is no penalty for moving it
2911   // out of line, and it can only improve I-cache density.
2912   // The decision to inline or out-of-line this final check is platform
2913   // dependent, and is found in the AD file definition of PartialSubtypeCheck.
2914   Node* psc = gvn.transform(
2915     new PartialSubtypeCheckNode(*ctrl, subklass, superklass));
2916 
2917   IfNode *iff4 = gen_subtype_check_compare(*ctrl, psc, gvn.zerocon(T_OBJECT), BoolTest::ne, PROB_FAIR, gvn, T_ADDRESS);
2918   r_not_subtype->init_req(2, gvn.transform(new IfTrueNode (iff4)));
2919   r_ok_subtype ->init_req(3, gvn.transform(new IfFalseNode(iff4)));
2920 
2921   // Return false path; set default control to true path.
2922   *ctrl = gvn.transform(r_ok_subtype);
2923   return gvn.transform(r_not_subtype);
2924 }

3306 // Generate a checkcast idiom.  Used by both the checkcast bytecode and the
3307 // array store bytecode.  Stack must be as-if BEFORE doing the bytecode so the
3308 // uncommon-trap paths work.  Adjust stack after this call.
3309 // If failure_control is supplied and not null, it is filled in with
3310 // the control edge for the cast failure.  Otherwise, an appropriate
3311 // uncommon trap or exception is thrown.
3312 Node* GraphKit::gen_checkcast(Node *obj, Node* superklass, Node* *failure_control, bool never_null) {

3320   // This detects the common cases where the test will short-circuit
3321   // away completely.  We do this before we perform the null check,
3322   // because if the test is going to turn into zero code, we don't
3323   // want a residual null check left around.  (Causes a slowdown,
3324   // for example, in some objArray manipulations, such as a[i]=a[j].)

3328       klass = _gvn.type(obj)->value_klass();

3331       if (objtp != NULL) {

3347             obj = ValueTypeNode::make_from_oop(this, obj, toop->value_klass());

3641 // If you want to make a pinch point on all memory slices, do not use this
3642 // function (even with AliasIdxBot); use insert_mem_bar() instead.
3643 Node* GraphKit::insert_mem_bar_volatile(int opcode, int alias_idx, Node* precedent) {
3644   // When Parse::do_put_xxx updates a volatile field, it appends a series
3645   // of MemBarVolatile nodes, one for *each* volatile field alias category.
3646   // The first membar is on the same memory slice as the field store opcode.
3647   // This forces the membar to follow the store.  (Bug 6500685 broke this.)
3648   // All the other membars (for other volatile slices, including AliasIdxBot,
3649   // which stands for all unknown volatile slices) are control-dependent
3650   // on the first membar.  This prevents later volatile loads or stores
3651   // from sliding up past the just-emitted store.
3652 
3653   MemBarNode* mb = MemBarNode::make(C, opcode, alias_idx, precedent);
3654   mb->set_req(TypeFunc::Control,control());
3655   if (alias_idx == Compile::AliasIdxBot) {
3656     mb->set_req(TypeFunc::Memory, merged_memory()->base_memory());
3657   } else {
3658     assert(!(opcode == Op_Initialize && alias_idx != Compile::AliasIdxRaw), "fix caller");
3659     mb->set_req(TypeFunc::Memory, memory(alias_idx));
3660   }
3661   Node* membar = _gvn.transform(mb);
3662   set_control(_gvn.transform(new ProjNode(membar, TypeFunc::Control)));
3663   if (alias_idx == Compile::AliasIdxBot) {
3664     merged_memory()->set_base_memory(_gvn.transform(new ProjNode(membar, TypeFunc::Memory)));
3665   } else {
3666     set_memory(_gvn.transform(new ProjNode(membar, TypeFunc::Memory)),alias_idx);
3667   }
3668   return membar;
3669 }

3784 // If the given klass is a constant or known to be an array,
3785 // fetch the constant layout helper value into constant_value
3786 // and return (Node*)NULL.  Otherwise, load the non-constant
3787 // layout helper value, and return the node which represents it.
3788 // This two-faced routine is useful because allocation sites
3789 // almost always feature constant types.
3790 Node* GraphKit::get_layout_helper(Node* klass_node, jint& constant_value) {
3791   const TypeKlassPtr* inst_klass = _gvn.type(klass_node)->isa_klassptr();

3794     assert(klass != NULL, "klass should not be NULL");
3795     bool    xklass = inst_klass->klass_is_exact();
3796     bool can_be_flattened = false;
3797     if (ValueArrayFlatten && klass->is_obj_array_klass()) {
3798       ciKlass* elem = klass->as_obj_array_klass()->element_klass();
3799       can_be_flattened = elem->can_be_value_klass() && (!elem->is_valuetype() || elem->as_value_klass()->flatten_array());

3937 //---------------------------new_instance--------------------------------------
3938 // This routine takes a klass_node which may be constant (for a static type)
3939 // or may be non-constant (for reflective code).  It will work equally well
3940 // for either, and the graph will fold nicely if the optimizer later reduces
3941 // the type to a constant.
3942 // The optional arguments are for specialized use by intrinsics:
3943 //  - If 'extra_slow_test' if not null is an extra condition for the slow-path.
3944 //  - If 'return_size_val', report the the total object size to the caller.
3945 //  - deoptimize_on_exception controls how Java exceptions are handled (rethrow vs deoptimize)
3946 Node* GraphKit::new_instance(Node* klass_node,
3947                              Node* extra_slow_test,
3948                              Node* *return_size_val,
3949                              bool deoptimize_on_exception,
3950                              ValueTypeBaseNode* value_node) {
3951   // Compute size in doublewords
3952   // The size is always an integral number of doublewords, represented
3953   // as a positive bytewise size stored in the klass's layout_helper.
3954   // The layout_helper also encodes (in a low bit) the need for a slow path.
3955   jint  layout_con = Klass::_lh_neutral_value;
3956   Node* layout_val = get_layout_helper(klass_node, layout_con);
3957   bool  layout_is_con = (layout_val == NULL);
3958 
3959   if (extra_slow_test == NULL)  extra_slow_test = intcon(0);
3960   // Generate the initial go-slow test.  It's either ALWAYS (return a
3961   // Node for 1) or NEVER (return a NULL) or perhaps (in the reflective
3962   // case) a computed value derived from the layout_helper.
3963   Node* initial_slow_test = NULL;
3964   if (layout_is_con) {
3965     assert(!StressReflectiveCode, "stress mode does not use these paths");
3966     bool must_go_slow = Klass::layout_helper_needs_slow_path(layout_con);
3967     initial_slow_test = must_go_slow ? intcon(1) : extra_slow_test;
3968   } else {   // reflective case
3969     // This reflective path is used by Unsafe.allocateInstance.
3970     // (It may be stress-tested by specifying StressReflectiveCode.)
3971     // Basically, we want to get into the VM is there's an illegal argument.
3972     Node* bit = intcon(Klass::_lh_instance_slow_path_bit);
3973     initial_slow_test = _gvn.transform( new AndINode(layout_val, bit) );
3974     if (extra_slow_test != intcon(0)) {
3975       initial_slow_test = _gvn.transform( new OrINode(initial_slow_test, extra_slow_test) );
3976     }
3977     // (Macro-expander will further convert this to a Bool, if necessary.)
3978   }
3979 
3980   // Find the size in bytes.  This is easy; it's the layout_helper.
3981   // The size value must be valid even if the slow path is taken.
3982   Node* size = NULL;
3983   if (layout_is_con) {
3984     size = MakeConX(Klass::layout_helper_size_in_bytes(layout_con));
3985   } else {   // reflective case
3986     // This reflective path is used by clone and Unsafe.allocateInstance.
3987     size = ConvI2X(layout_val);
3988 
3989     // Clear the low bits to extract layout_helper_size_in_bytes:
3990     assert((int)Klass::_lh_instance_slow_path_bit < BytesPerLong, "clear bit");
3991     Node* mask = MakeConX(~ (intptr_t)right_n_bits(LogBytesPerLong));
3992     size = _gvn.transform( new AndXNode(size, mask) );
3993   }
3994   if (return_size_val != NULL) {
3995     (*return_size_val) = size;
3996   }
3997 
3998   // This is a precise notnull oop of the klass.
3999   // (Actually, it need not be precise if this is a reflective allocation.)
4000   // It's what we cast the result to.
4001   const TypeKlassPtr* tklass = _gvn.type(klass_node)->isa_klassptr();
4002   if (!tklass)  tklass = TypeKlassPtr::OBJECT;
4003   const TypeOopPtr* oop_type = tklass->as_instance_type();
4004 
4005   // Now generate allocation code
4006 
4007   // The entire memory state is needed for slow path of the allocation
4008   // since GC and deoptimization can happen.
4009   Node *mem = reset_memory();
4010   set_all_memory(mem); // Create new memory state
4011 
4012   AllocateNode* alloc = new AllocateNode(C, AllocateNode::alloc_type(Type::TOP),
4013                                          control(), mem, i_o(),
4014                                          size, klass_node,
4015                                          initial_slow_test, value_node);
4016 
4017   return set_output_for_allocation(alloc, oop_type, deoptimize_on_exception);
4018 }

4028 //-------------------------------new_array-------------------------------------
4029 // helper for newarray and anewarray
4030 // The 'length' parameter is (obviously) the length of the array.
4031 // See comments on new_instance for the meaning of the other arguments.
4032 Node* GraphKit::new_array(Node* klass_node,     // array klass (maybe variable)
4033                           Node* length,         // number of array elements
4034                           int   nargs,          // number of arguments to push back for uncommon trap
4035                           Node* *return_size_val,
4036                           bool deoptimize_on_exception) {
4037   jint  layout_con = Klass::_lh_neutral_value;
4038   Node* layout_val = get_layout_helper(klass_node, layout_con);
4039   bool  layout_is_con = (layout_val == NULL);
4040 
4041   if (!layout_is_con && !StressReflectiveCode &&
4042       !too_many_traps(Deoptimization::Reason_class_check)) {
4043     // This is a reflective array creation site.
4044     // Optimistically assume that it is a subtype of Object[],
4045     // so that we can fold up all the address arithmetic.
4046     layout_con = Klass::array_layout_helper(T_OBJECT);
4047     Node* cmp_lh = _gvn.transform( new CmpINode(layout_val, intcon(layout_con)) );
4048     Node* bol_lh = _gvn.transform( new BoolNode(cmp_lh, BoolTest::eq) );
4049     { BuildCutout unless(this, bol_lh, PROB_MAX);
4050       inc_sp(nargs);
4051       uncommon_trap(Deoptimization::Reason_class_check,
4052                     Deoptimization::Action_maybe_recompile);
4053     }
4054     layout_val = NULL;
4055     layout_is_con = true;
4056   }
4057 
4058   // Generate the initial go-slow test.  Make sure we do not overflow
4059   // if length is huge (near 2Gig) or negative!  We do not need
4060   // exact double-words here, just a close approximation of needed
4061   // double-words.  We can't add any offset or rounding bits, lest we
4062   // take a size -1 of bytes and make it positive.  Use an unsigned
4063   // compare, so negative sizes look hugely positive.
4064   int fast_size_limit = FastAllocateSizeLimit;
4065   if (layout_is_con) {
4066     assert(!StressReflectiveCode, "stress mode does not use these paths");
4067     // Increase the size limit if we have exact knowledge of array type.
4068     int log2_esize = Klass::layout_helper_log2_element_size(layout_con);
4069     fast_size_limit <<= MAX2(LogBytesPerLong - log2_esize, 0);
4070   }
4071 
4072   Node* initial_slow_cmp  = _gvn.transform( new CmpUNode( length, intcon( fast_size_limit ) ) );
4073   Node* initial_slow_test = _gvn.transform( new BoolNode( initial_slow_cmp, BoolTest::gt ) );
4074 
4075   // --- Size Computation ---
4076   // array_size = round_to_heap(array_header + (length << elem_shift));
4077   // where round_to_heap(x) == align_to(x, MinObjAlignmentInBytes)
4078   // and align_to(x, y) == ((x + y-1) & ~(y-1))
4079   // The rounding mask is strength-reduced, if possible.
4080   int round_mask = MinObjAlignmentInBytes - 1;
4081   Node* header_size = NULL;
4082   int   header_size_min  = arrayOopDesc::base_offset_in_bytes(T_BYTE);
4083   // (T_BYTE has the weakest alignment and size restrictions...)
4084   if (layout_is_con) {
4085     int       hsize  = Klass::layout_helper_header_size(layout_con);
4086     int       eshift = Klass::layout_helper_log2_element_size(layout_con);
4087     bool is_value_array = Klass::layout_helper_is_valueArray(layout_con);
4088     if ((round_mask & ~right_n_bits(eshift)) == 0)
4089       round_mask = 0;  // strength-reduce it if it goes away completely
4090     assert(is_value_array || (hsize & right_n_bits(eshift)) == 0, "hsize is pre-rounded");
4091     assert(header_size_min <= hsize, "generic minimum is smallest");
4092     header_size_min = hsize;
4093     header_size = intcon(hsize + round_mask);

4494 Node* GraphKit::compress_string(Node* src, const TypeAryPtr* src_type, Node* dst, Node* count) {
4495   assert(Matcher::match_rule_supported(Op_StrCompressedCopy), "Intrinsic not supported");
4496   assert(src_type == TypeAryPtr::BYTES || src_type == TypeAryPtr::CHARS, "invalid source type");

4498   // the dependency between preceding and subsequent loads/stores.

4500   //  StoreB
4501   //  compress_string
4502   //  LoadB
4503   // has this memory graph (use->def):
4504   //  LoadB -> compress_string -> CharMem
4505   //             ... -> StoreB -> ByteMem
4506   // The intrinsic hides the dependency between LoadB and StoreB, causing
4507   // the load to read from memory not containing the result of the StoreB.
4508   // The correct memory graph should look like this:
4509   //  LoadB -> compress_string -> MergeMem(CharMem, StoreB(ByteMem))
4510   Node* mem = capture_memory(src_type, TypeAryPtr::BYTES);
4511   StrCompressedCopyNode* str = new StrCompressedCopyNode(control(), mem, src, dst, count);
4512   Node* res_mem = _gvn.transform(new SCMemProjNode(str));
4513   set_memory(res_mem, TypeAryPtr::BYTES);
