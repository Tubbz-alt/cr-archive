  86 Deoptimization::UnrollBlock::UnrollBlock(int  size_of_deoptimized_frame,
  87                                          int  caller_adjustment,
  88                                          int  caller_actual_parameters,
  89                                          int  number_of_frames,
  90                                          intptr_t* frame_sizes,
  91                                          address* frame_pcs,
  92                                          BasicType return_type,
  93                                          int exec_mode) {
  94   _size_of_deoptimized_frame = size_of_deoptimized_frame;
  95   _caller_adjustment         = caller_adjustment;
  96   _caller_actual_parameters  = caller_actual_parameters;
  97   _number_of_frames          = number_of_frames;
  98   _frame_sizes               = frame_sizes;
  99   _frame_pcs                 = frame_pcs;
 100   _register_block            = NEW_C_HEAP_ARRAY(intptr_t, RegisterMap::reg_count * 2, mtCompiler);
 101   _return_type               = return_type;
 102   _initial_info              = 0;
 103   // PD (x86 only)
 104   _counter_temp              = 0;
 105   _unpack_kind               = exec_mode;
 106   _sender_sp_temp            = 0;
 107 
 108   _total_frame_sizes         = size_of_frames();
 109   assert(exec_mode >= 0 && exec_mode < Unpack_LIMIT, "Unexpected exec_mode");
 110 }

 127 int Deoptimization::UnrollBlock::size_of_frames() const {
 128   // Acount first for the adjustment of the initial frame
 129   int result = _caller_adjustment;
 130   for (int index = 0; index < number_of_frames(); index++) {
 131     result += frame_sizes()[index];
 132   }
 133   return result;
 134 }

 149 // In order to make fetch_unroll_info work properly with escape
 150 // analysis, The method was changed from JRT_LEAF to JRT_BLOCK_ENTRY and
 151 // ResetNoHandleMark and HandleMark were removed from it. The actual reallocation
 152 // of previously eliminated objects occurs in realloc_objects, which is
 153 // called from the method fetch_unroll_info_helper below.
 154 JRT_BLOCK_ENTRY(Deoptimization::UnrollBlock*, Deoptimization::fetch_unroll_info(JavaThread* thread, int exec_mode))
 155   // It is actually ok to allocate handles in a leaf method. It causes no safepoints,
 156   // but makes the entry a little slower. There is however a little dance we have to
 157   // do in debug mode to get around the NoHandleMark code in the JRT_LEAF macro
 158 
 159   // fetch_unroll_info() is called at the beginning of the deoptimization
 160   // handler. Note this fact before we start generating temporary frames
 161   // that can confuse an asynchronous stack walker. This counter is
 162   // decremented at the end of unpack_frames().
 163   if (TraceDeoptimization) {
 164     tty->print_cr("Deoptimizing thread " INTPTR_FORMAT, p2i(thread));
 165   }
 166   thread->inc_in_deopt_handler();
 167 
 168   return fetch_unroll_info_helper(thread, exec_mode);
 169 JRT_END

 171 #if COMPILER2_OR_JVMCI
 172 static bool eliminate_allocations(JavaThread* thread, int exec_mode, CompiledMethod* compiled_method,
 173                                   frame& deoptee, RegisterMap& map, GrowableArray<compiledVFrame*>* chunk) {
 174   bool realloc_failures = false;
 175   assert (chunk->at(0)->scope() != NULL,"expect only compiled java frames");
 176 
 177   GrowableArray<ScopeValue*>* objects = chunk->at(0)->scope()->objects();
 178 
 179   // The flag return_oop() indicates call sites which return oop
 180   // in compiled code. Such sites include java method calls,
 181   // runtime calls (for example, used to allocate new objects/arrays
 182   // on slow code path) and any other calls generated in compiled code.
 183   // It is not guaranteed that we can get such information here only
 184   // by analyzing bytecode in deoptimized frames. This is why this flag
 185   // is set during method compilation (see Compile::Process_OopMap_Node()).
 186   // If the previous frame was popped or if we are dispatching an exception,
 187   // we don't have an oop result.
 188   ScopeDesc* scope = chunk->at(0)->scope();
 189   bool save_oop_result = scope->return_oop() && !thread->popframe_forcing_deopt_reexecution() && (exec_mode == Deoptimization::Unpack_deopt);
 190   // In case of the return of multiple values, we must take care
 191   // of all oop return values.
 192   GrowableArray<Handle> return_oops;
 193   ValueKlass* vk = NULL;
 194   if (save_oop_result && scope->return_vt()) {
 195     vk = ValueKlass::returned_value_klass(map);
 196     if (vk != NULL) {
 197       vk->save_oop_fields(map, return_oops);
 198       save_oop_result = false;
 199     }
 200   }
 201   if (save_oop_result) {
 202     // Reallocation may trigger GC. If deoptimization happened on return from
 203     // call which returns oop we need to save it since it is not in oopmap.
 204     oop result = deoptee.saved_oop_result(&map);
 205     assert(oopDesc::is_oop_or_null(result), "must be oop");
 206     return_oops.push(Handle(thread, result));
 207     assert(Universe::heap()->is_in_or_null(result), "must be heap pointer");
 208     if (TraceDeoptimization) {
 209       ttyLocker ttyl;
 210       tty->print_cr("SAVED OOP RESULT " INTPTR_FORMAT " in thread " INTPTR_FORMAT, p2i(result), p2i(thread));
 211     }
 212   }
 213   if (objects != NULL || vk != NULL) {
 214     bool skip_internal = (compiled_method != NULL) && !compiled_method->is_compiled_by_jvmci();
 215     JRT_BLOCK
 216       if (vk != NULL) {
 217         realloc_failures = Deoptimization::realloc_value_type_result(vk, map, return_oops, THREAD);
 218       }
 219       if (objects != NULL) {
 220         realloc_failures = realloc_failures || Deoptimization::realloc_objects(thread, &deoptee, &map, objects, THREAD);
 221         Deoptimization::reassign_fields(&deoptee, &map, objects, realloc_failures, skip_internal, THREAD);
 222       }
 223     JRT_END
 224 #ifndef PRODUCT
 225     if (TraceDeoptimization) {
 226       ttyLocker ttyl;
 227       tty->print_cr("REALLOC OBJECTS in thread " INTPTR_FORMAT, p2i(thread));
 228       if (objects != NULL) {
 229         Deoptimization::print_objects(objects, realloc_failures);
 230       } else {
 231         Handle obj = realloc_failures ? Handle() : return_oops.first();
 232         Deoptimization::print_object(vk, obj, realloc_failures);
 233       }
 234     }
 235 #endif
 236   }
 237   if (save_oop_result || vk != NULL) {
 238     // Restore result.
 239     assert(return_oops.length() == 1, "no value type");
 240     deoptee.set_saved_oop_result(&map, return_oops.pop()());
 241   }
 242   return realloc_failures;
 243 }

 245 static void eliminate_locks(JavaThread* thread, GrowableArray<compiledVFrame*>* chunk, bool realloc_failures) {
 246 #ifndef PRODUCT
 247   bool first = true;
 248 #endif
 249   for (int i = 0; i < chunk->length(); i++) {
 250     compiledVFrame* cvf = chunk->at(i);
 251     assert (cvf->scope() != NULL,"expect only compiled java frames");
 252     GrowableArray<MonitorInfo*>* monitors = cvf->monitors();
 253     if (monitors->is_nonempty()) {
 254       Deoptimization::relock_objects(monitors, thread, realloc_failures);
 255 #ifndef PRODUCT
 256       if (PrintDeoptimizationDetails) {
 257         ttyLocker ttyl;
 258         for (int j = 0; j < monitors->length(); j++) {
 259           MonitorInfo* mi = monitors->at(j);
 260           if (mi->eliminated()) {
 261             if (first) {
 262               first = false;
 263               tty->print_cr("RELOCK OBJECTS in thread " INTPTR_FORMAT, p2i(thread));
 264             }
 265             if (mi->owner_is_scalar_replaced()) {
 266               Klass* k = java_lang_Class::as_Klass(mi->owner_klass());
 267               tty->print_cr("     failed reallocation for klass %s", k->external_name());
 268             } else {
 269               tty->print_cr("     object <" INTPTR_FORMAT "> locked", p2i(mi->owner()));
 270             }
 271           }
 272         }
 273       }
 274 #endif // !PRODUCT
 275     }
 276   }
 277 }

 280 // This is factored, since it is both called from a JRT_LEAF (deoptimization) and a JRT_ENTRY (uncommon_trap)
 281 Deoptimization::UnrollBlock* Deoptimization::fetch_unroll_info_helper(JavaThread* thread, int exec_mode) {
 282 
 283   // Note: there is a safepoint safety issue here. No matter whether we enter
 284   // via vanilla deopt or uncommon trap we MUST NOT stop at a safepoint once
 285   // the vframeArray is created.
 286   //
 287 
 288   // Allocate our special deoptimization ResourceMark
 289   DeoptResourceMark* dmark = new DeoptResourceMark(thread);
 290   assert(thread->deopt_mark() == NULL, "Pending deopt!");
 291   thread->set_deopt_mark(dmark);
 292 
 293   frame stub_frame = thread->last_frame(); // Makes stack walkable as side effect
 294   RegisterMap map(thread, true);
 295   RegisterMap dummy_map(thread, false);
 296   // Now get the deoptee with a valid map
 297   frame deoptee = stub_frame.sender(&map);
 298   // Set the deoptee nmethod
 299   assert(thread->deopt_compiled_method() == NULL, "Pending deopt!");
 300   CompiledMethod* cm = deoptee.cb()->as_compiled_method_or_null();
 301   thread->set_deopt_compiled_method(cm);
 302 
 303   if (VerifyStack) {
 304     thread->validate_frame_layout();
 305   }
 306 
 307   // Create a growable array of VFrames where each VFrame represents an inlined
 308   // Java frame.  This storage is allocated with the usual system arena.
 309   assert(deoptee.is_compiled_frame(), "Wrong frame type");
 310   GrowableArray<compiledVFrame*>* chunk = new GrowableArray<compiledVFrame*>(10);
 311   vframe* vf = vframe::new_vframe(&deoptee, &map, thread);
 312   while (!vf->is_top()) {
 313     assert(vf->is_compiled_frame(), "Wrong frame type");
 314     chunk->push(compiledVFrame::cast(vf));
 315     vf = vf->sender();
 316   }
 317   assert(vf->is_compiled_frame(), "Wrong frame type");
 318   chunk->push(compiledVFrame::cast(vf));
 319 
 320   bool realloc_failures = false;
 321 
 322 #if COMPILER2_OR_JVMCI
 323 #if INCLUDE_JVMCI
 324   bool jvmci_enabled = true;
 325 #else
 326   bool jvmci_enabled = false;
 327 #endif
 328 
 329   // Reallocate the non-escaping objects and restore their fields. Then
 330   // relock objects if synchronization on them was eliminated.
 331   if (jvmci_enabled COMPILER2_PRESENT( || (DoEscapeAnalysis && EliminateAllocations) )) {
 332     realloc_failures = eliminate_allocations(thread, exec_mode, cm, deoptee, map, chunk);
 333   }
 334 #endif // COMPILER2_OR_JVMCI
 335 
 336   // Revoke biases, done with in java state.
 337   // No safepoints allowed after this
 338   revoke_from_deopt_handler(thread, deoptee, &map);
 339 
 340   // Ensure that no safepoint is taken after pointers have been stored
 341   // in fields of rematerialized objects.  If a safepoint occurs from here on
 342   // out the java state residing in the vframeArray will be missed.
 343   // Locks may be rebaised in a safepoint.
 344   NoSafepointVerifier no_safepoint;
 345 
 346 #if COMPILER2_OR_JVMCI
 347   if (jvmci_enabled COMPILER2_PRESENT( || ((DoEscapeAnalysis || EliminateNestedLocks) && EliminateLocks) )) {
 348     eliminate_locks(thread, chunk, realloc_failures);
 349   }
 350 #endif // COMPILER2_OR_JVMCI
 351 
 352   ScopeDesc* trap_scope = chunk->at(0)->scope();
 353   Handle exceptionObject;
 354   if (trap_scope->rethrow_exception()) {
 355     if (PrintDeoptimizationDetails) {
 356       tty->print_cr("Exception to be rethrown in the interpreter for method %s::%s at bci %d", trap_scope->method()->method_holder()->name()->as_C_string(), trap_scope->method()->name()->as_C_string(), trap_scope->bci());
 357     }
 358     GrowableArray<ScopeValue*>* expressions = trap_scope->expressions();
 359     guarantee(expressions != NULL && expressions->length() > 0, "must have exception to throw");
 360     ScopeValue* topOfStack = expressions->top();
 361     exceptionObject = StackValue::create_stack_value(&deoptee, &map, topOfStack)->get_obj();
 362     guarantee(exceptionObject() != NULL, "exception oop can not be null");
 363   }
 364 
 365   vframeArray* array = create_vframeArray(thread, deoptee, &map, chunk, realloc_failures);
 366 #if COMPILER2_OR_JVMCI
 367   if (realloc_failures) {
 368     pop_frames_failed_reallocs(thread, array);
 369   }
 370 #endif
 371 
 372   assert(thread->vframe_array_head() == NULL, "Pending deopt!");
 373   thread->set_vframe_array_head(array);
 374 
 375   // Now that the vframeArray has been created if we have any deferred local writes
 376   // added by jvmti then we can free up that structure as the data is now in the
 377   // vframeArray
 378 
 379   if (thread->deferred_locals() != NULL) {
 380     GrowableArray<jvmtiDeferredLocalVariableSet*>* list = thread->deferred_locals();
 381     int i = 0;
 382     do {
 383       // Because of inlining we could have multiple vframes for a single frame
 384       // and several of the vframes could have deferred writes. Find them all.
 385       if (list->at(i)->id() == array->original().id()) {
 386         jvmtiDeferredLocalVariableSet* dlv = list->at(i);
 387         list->remove_at(i);
 388         // individual jvmtiDeferredLocalVariableSet are CHeapObj's
 389         delete dlv;
 390       } else {
 391         i++;
 392       }
 393     } while ( i < list->length() );
 394     if (list->length() == 0) {
 395       thread->set_deferred_locals(NULL);
 396       // free the list and elements back to C heap.
 397       delete list;
 398     }
 399 
 400   }
 401 
 402   // Compute the caller frame based on the sender sp of stub_frame and stored frame sizes info.
 403   CodeBlob* cb = stub_frame.cb();
 404   // Verify we have the right vframeArray
 405   assert(cb->frame_size() >= 0, "Unexpected frame size");
 406   intptr_t* unpack_sp = stub_frame.sp() + cb->frame_size();
 407 
 408   // If the deopt call site is a MethodHandle invoke call site we have
 409   // to adjust the unpack_sp.
 410   nmethod* deoptee_nm = deoptee.cb()->as_nmethod_or_null();
 411   if (deoptee_nm != NULL && deoptee_nm->is_method_handle_return(deoptee.pc()))
 412     unpack_sp = deoptee.unextended_sp();
 413 
 414 #ifdef ASSERT
 415   assert(cb->is_deoptimization_stub() ||
 416          cb->is_uncommon_trap_stub() ||
 417          strcmp("Stub<DeoptimizationStub.deoptimizationHandler>", cb->name()) == 0 ||
 418          strcmp("Stub<UncommonTrapStub.uncommonTrapHandler>", cb->name()) == 0,
 419          "unexpected code blob: %s", cb->name());
 420 #endif
 421 
 422   // This is a guarantee instead of an assert because if vframe doesn't match
 423   // we will unpack the wrong deoptimized frame and wind up in strange places
 424   // where it will be very difficult to figure out what went wrong. Better
 425   // to die an early death here than some very obscure death later when the
 426   // trail is cold.
 427   // Note: on ia64 this guarantee can be fooled by frames with no memory stack
 428   // in that it will fail to detect a problem when there is one. This needs
 429   // more work in tiger timeframe.
 430   guarantee(array->unextended_sp() == unpack_sp, "vframe_array_head must contain the vframeArray to unpack");
 431 
 432   int number_of_frames = array->frames();
 433 
 434   // Compute the vframes' sizes.  Note that frame_sizes[] entries are ordered from outermost to innermost
 435   // virtual activation, which is the reverse of the elements in the vframes array.
 436   intptr_t* frame_sizes = NEW_C_HEAP_ARRAY(intptr_t, number_of_frames, mtCompiler);
 437   // +1 because we always have an interpreter return address for the final slot.
 438   address* frame_pcs = NEW_C_HEAP_ARRAY(address, number_of_frames + 1, mtCompiler);
 439   int popframe_extra_args = 0;
 440   // Create an interpreter return address for the stub to use as its return
 441   // address so the skeletal frames are perfectly walkable
 442   frame_pcs[number_of_frames] = Interpreter::deopt_entry(vtos, 0);
 443 
 444   // PopFrame requires that the preserved incoming arguments from the recently-popped topmost
 445   // activation be put back on the expression stack of the caller for reexecution
 446   if (JvmtiExport::can_pop_frame() && thread->popframe_forcing_deopt_reexecution()) {
 447     popframe_extra_args = in_words(thread->popframe_preserved_args_size_in_words());
 448   }
 449 
 450   // Find the current pc for sender of the deoptee. Since the sender may have been deoptimized
 451   // itself since the deoptee vframeArray was created we must get a fresh value of the pc rather
 452   // than simply use array->sender.pc(). This requires us to walk the current set of frames
 453   //
 454   frame deopt_sender = stub_frame.sender(&dummy_map); // First is the deoptee frame
 455   deopt_sender = deopt_sender.sender(&dummy_map);     // Now deoptee caller
 456 
 457   // It's possible that the number of parameters at the call site is
 458   // different than number of arguments in the callee when method
 459   // handles are used.  If the caller is interpreted get the real
 460   // value so that the proper amount of space can be added to it's
 461   // frame.
 462   bool caller_was_method_handle = false;
 463   if (deopt_sender.is_interpreted_frame()) {
 464     methodHandle method(thread, deopt_sender.interpreter_frame_method());
 465     Bytecode_invoke cur = Bytecode_invoke_check(method, deopt_sender.interpreter_frame_bci());
 466     if (cur.is_invokedynamic() || cur.is_invokehandle()) {
 467       // Method handle invokes may involve fairly arbitrary chains of
 468       // calls so it's impossible to know how much actual space the
 469       // caller has for locals.
 470       caller_was_method_handle = true;
 471     }
 472   }
 473 
 474   //
 475   // frame_sizes/frame_pcs[0] oldest frame (int or c2i)
 476   // frame_sizes/frame_pcs[1] next oldest frame (int)
 477   // frame_sizes/frame_pcs[n] youngest frame (int)
 478   //
 479   // Now a pc in frame_pcs is actually the return address to the frame's caller (a frame
 480   // owns the space for the return address to it's caller).  Confusing ain't it.
 481   //
 482   // The vframe array can address vframes with indices running from
 483   // 0.._frames-1. Index  0 is the youngest frame and _frame - 1 is the oldest (root) frame.
 484   // When we create the skeletal frames we need the oldest frame to be in the zero slot
 485   // in the frame_sizes/frame_pcs so the assembly code can do a trivial walk.
 486   // so things look a little strange in this loop.
 487   //
 488   int callee_parameters = 0;
 489   int callee_locals = 0;
 490   for (int index = 0; index < array->frames(); index++ ) {
 491     // frame[number_of_frames - 1 ] = on_stack_size(youngest)
 492     // frame[number_of_frames - 2 ] = on_stack_size(sender(youngest))
 493     // frame[number_of_frames - 3 ] = on_stack_size(sender(sender(youngest)))
 494     frame_sizes[number_of_frames - 1 - index] = BytesPerWord * array->element(index)->on_stack_size(callee_parameters,
 495                                                                                                     callee_locals,
 496                                                                                                     index == 0,
 497                                                                                                     popframe_extra_args);
 498     // This pc doesn't have to be perfect just good enough to identify the frame
 499     // as interpreted so the skeleton frame will be walkable
 500     // The correct pc will be set when the skeleton frame is completely filled out
 501     // The final pc we store in the loop is wrong and will be overwritten below
 502     frame_pcs[number_of_frames - 1 - index ] = Interpreter::deopt_entry(vtos, 0) - frame::pc_return_offset;
 503 
 504     callee_parameters = array->element(index)->method()->size_of_parameters();
 505     callee_locals = array->element(index)->method()->max_locals();
 506     popframe_extra_args = 0;
 507   }
 508 
 509   // Compute whether the root vframe returns a float or double value.
 510   BasicType return_type;
 511   {
 512     methodHandle method(thread, array->element(0)->method());
 513     Bytecode_invoke invoke = Bytecode_invoke_check(method, array->element(0)->bci());
 514     return_type = invoke.is_valid() ? invoke.result_type() : T_ILLEGAL;
 515   }
 516 
 517   // Compute information for handling adapters and adjusting the frame size of the caller.
 518   int caller_adjustment = 0;
 519 
 520   // Compute the amount the oldest interpreter frame will have to adjust
 521   // its caller's stack by. If the caller is a compiled frame then
 522   // we pretend that the callee has no parameters so that the
 523   // extension counts for the full amount of locals and not just
 524   // locals-parms. This is because without a c2i adapter the parm
 525   // area as created by the compiled frame will not be usable by
 526   // the interpreter. (Depending on the calling convention there
 527   // may not even be enough space).
 528 
 529   // QQQ I'd rather see this pushed down into last_frame_adjust
 530   // and have it take the sender (aka caller).
 531 
 532   if (deopt_sender.is_compiled_frame() || caller_was_method_handle) {
 533     caller_adjustment = last_frame_adjust(0, callee_locals);
 534   } else if (callee_locals > callee_parameters) {
 535     // The caller frame may need extending to accommodate
 536     // non-parameter locals of the first unpacked interpreted frame.
 537     // Compute that adjustment.
 538     caller_adjustment = last_frame_adjust(callee_parameters, callee_locals);
 539   }
 540 
 541   // If the sender is deoptimized we must retrieve the address of the handler
 542   // since the frame will "magically" show the original pc before the deopt
 543   // and we'd undo the deopt.
 544 
 545   frame_pcs[0] = deopt_sender.raw_pc();
 546 
 547   assert(CodeCache::find_blob_unsafe(frame_pcs[0]) != NULL, "bad pc");
 548 
 549 #if INCLUDE_JVMCI
 550   if (exceptionObject() != NULL) {
 551     thread->set_exception_oop(exceptionObject());
 552     exec_mode = Unpack_exception;
 553   }
 554 #endif
 555 
 556   if (thread->frames_to_pop_failed_realloc() > 0 && exec_mode != Unpack_uncommon_trap) {
 557     assert(thread->has_pending_exception(), "should have thrown OOME");
 558     thread->set_exception_oop(thread->pending_exception());
 559     thread->clear_pending_exception();
 560     exec_mode = Unpack_exception;
 561   }
 562 
 563 #if INCLUDE_JVMCI
 564   if (thread->frames_to_pop_failed_realloc() > 0) {
 565     thread->set_pending_monitorenter(false);
 566   }
 567 #endif
 568 
 569   UnrollBlock* info = new UnrollBlock(array->frame_size() * BytesPerWord,
 570                                       caller_adjustment * BytesPerWord,
 571                                       caller_was_method_handle ? 0 : callee_parameters,
 572                                       number_of_frames,
 573                                       frame_sizes,
 574                                       frame_pcs,
 575                                       return_type,
 576                                       exec_mode);
 577   // On some platforms, we need a way to pass some platform dependent
 578   // information to the unpacking code so the skeletal frames come out
 579   // correct (initial fp value, unextended sp, ...)
 580   info->set_initial_info((intptr_t) array->sender().initial_deoptimization_info());
 581 
 582   if (array->frames() > 1) {
 583     if (VerifyStack && TraceDeoptimization) {
 584       ttyLocker ttyl;
 585       tty->print_cr("Deoptimizing method containing inlining");
 586     }
 587   }
 588 
 589   array->set_unroll_block(info);
 590   return info;
 591 }

 598   // Get array if coming from exception
 599   if (array == NULL) {
 600     array = thread->vframe_array_head();
 601   }
 602   thread->set_vframe_array_head(NULL);
 603 
 604   // Free the previous UnrollBlock
 605   vframeArray* old_array = thread->vframe_array_last();
 606   thread->set_vframe_array_last(array);

 645 // If a cpu implements callee save values, move this to deoptimization_<cpu>.cpp.
 646 void Deoptimization::unwind_callee_save_values(frame* f, vframeArray* vframe_array) {
 647 
 648   // This code is sort of the equivalent of C2IAdapter::setup_stack_frame back in
 649   // the days we had adapter frames. When we deoptimize a situation where a
 650   // compiled caller calls a compiled caller will have registers it expects
 651   // to survive the call to the callee. If we deoptimize the callee the only
 652   // way we can restore these registers is to have the oldest interpreter
 653   // frame that we create restore these values. That is what this routine
 654   // will accomplish.
 655 
 656   // At the moment we have modified c2 to not have any callee save registers
 657   // so this problem does not exist and this routine is just a place holder.
 658 
 659   assert(f->is_interpreted_frame(), "must be interpreted");
 660 }

 662 // Return BasicType of value being returned
 663 JRT_LEAF(BasicType, Deoptimization::unpack_frames(JavaThread* thread, int exec_mode))
 664 
 665   // We are already active in the special DeoptResourceMark any ResourceObj's we
 666   // allocate will be freed at the end of the routine.
 667 
 668   // It is actually ok to allocate handles in a leaf method. It causes no safepoints,
 669   // but makes the entry a little slower. There is however a little dance we have to
 670   // do in debug mode to get around the NoHandleMark code in the JRT_LEAF macro
 671   ResetNoHandleMark rnhm; // No-op in release/product versions
 672   HandleMark hm;
 673 
 674   frame stub_frame = thread->last_frame();
 675 
 676   // Since the frame to unpack is the top frame of this thread, the vframe_array_head
 677   // must point to the vframeArray for the unpack frame.
 678   vframeArray* array = thread->vframe_array_head();
 679 
 680 #ifndef PRODUCT
 681   if (TraceDeoptimization) {
 682     ttyLocker ttyl;
 683     tty->print_cr("DEOPT UNPACKING thread " INTPTR_FORMAT " vframeArray " INTPTR_FORMAT " mode %d",
 684                   p2i(thread), p2i(array), exec_mode);
 685   }
 686 #endif
 687   Events::log_deopt_message(thread, "DEOPT UNPACKING pc=" INTPTR_FORMAT " sp=" INTPTR_FORMAT " mode %d",
 688               p2i(stub_frame.pc()), p2i(stub_frame.sp()), exec_mode);
 689 
 690   UnrollBlock* info = array->unroll_block();
 691 
 692   // We set the last_Java frame. But the stack isn't really parsable here. So we
 693   // clear it to make sure JFR understands not to try and walk stacks from events
 694   // in here.
 695   intptr_t* sp = thread->frame_anchor()->last_Java_sp();
 696   thread->frame_anchor()->set_last_Java_sp(NULL);
 697 
 698   // Unpack the interpreter frames and any adapter frame (c2 only) we might create.
 699   array->unpack_to_stack(stub_frame, exec_mode, info->caller_actual_parameters());
 700 
 701   thread->frame_anchor()->set_last_Java_sp(sp);
 702 
 703   BasicType bt = info->return_type();
 704 
 705   // If we have an exception pending, claim that the return type is an oop
 706   // so the deopt_blob does not overwrite the exception_oop.
 707 
 708   if (exec_mode == Unpack_exception)
 709     bt = T_OBJECT;
 710 
 711   // Cleanup thread deopt data
 712   cleanup_deopt_info(thread, array);
 713 
 714 #ifndef PRODUCT
 715   if (VerifyStack) {
 716     ResourceMark res_mark;
 717     // Clear pending exception to not break verification code (restored afterwards)
 718     PRESERVE_EXCEPTION_MARK;
 719 
 720     thread->validate_frame_layout();
 721 
 722     // Verify that the just-unpacked frames match the interpreter's
 723     // notions of expression stack and locals
 724     vframeArray* cur_array = thread->vframe_array_last();
 725     RegisterMap rm(thread, false);
 726     rm.set_include_argument_oops(false);
 727     bool is_top_frame = true;
 728     int callee_size_of_parameters = 0;
 729     int callee_max_locals = 0;
 730     for (int i = 0; i < cur_array->frames(); i++) {
 731       vframeArrayElement* el = cur_array->element(i);
 732       frame* iframe = el->iframe();
 733       guarantee(iframe->is_interpreted_frame(), "Wrong frame type");
 734 
 735       // Get the oop map for this bci
 736       InterpreterOopMap mask;
 737       int cur_invoke_parameter_size = 0;
 738       bool try_next_mask = false;
 739       int next_mask_expression_stack_size = -1;
 740       int top_frame_expression_stack_adjustment = 0;
 741       methodHandle mh(thread, iframe->interpreter_frame_method());
 742       OopMapCache::compute_one_oop_map(mh, iframe->interpreter_frame_bci(), &mask);
 743       BytecodeStream str(mh, iframe->interpreter_frame_bci());
 744       int max_bci = mh->code_size();
 745       // Get to the next bytecode if possible
 746       assert(str.bci() < max_bci, "bci in interpreter frame out of bounds");
 747       // Check to see if we can grab the number of outgoing arguments
 748       // at an uncommon trap for an invoke (where the compiler
 749       // generates debug info before the invoke has executed)
 750       Bytecodes::Code cur_code = str.next();
 751       if (Bytecodes::is_invoke(cur_code)) {
 752         Bytecode_invoke invoke(mh, iframe->interpreter_frame_bci());
 753         cur_invoke_parameter_size = invoke.size_of_parameters();
 754         if (i != 0 && !invoke.is_invokedynamic() && MethodHandles::has_member_arg(invoke.klass(), invoke.name())) {
 755           callee_size_of_parameters++;
 756         }
 757       }
 758       if (str.bci() < max_bci) {
 759         Bytecodes::Code next_code = str.next();
 760         if (next_code >= 0) {
 761           // The interpreter oop map generator reports results before
 762           // the current bytecode has executed except in the case of
 763           // calls. It seems to be hard to tell whether the compiler
 764           // has emitted debug information matching the "state before"
 765           // a given bytecode or the state after, so we try both
 766           if (!Bytecodes::is_invoke(cur_code) && cur_code != Bytecodes::_athrow) {
 767             // Get expression stack size for the next bytecode
 768             InterpreterOopMap next_mask;
 769             OopMapCache::compute_one_oop_map(mh, str.bci(), &next_mask);
 770             next_mask_expression_stack_size = next_mask.expression_stack_size();
 771             if (Bytecodes::is_invoke(next_code)) {
 772               Bytecode_invoke invoke(mh, str.bci());
 773               next_mask_expression_stack_size += invoke.size_of_parameters();
 774             }
 775             // Need to subtract off the size of the result type of
 776             // the bytecode because this is not described in the
 777             // debug info but returned to the interpreter in the TOS
 778             // caching register
 779             BasicType bytecode_result_type = Bytecodes::result_type(cur_code);
 780             if (bytecode_result_type != T_ILLEGAL) {
 781               top_frame_expression_stack_adjustment = type2size[bytecode_result_type];
 782             }
 783             assert(top_frame_expression_stack_adjustment >= 0, "stack adjustment must be positive");
 784             try_next_mask = true;
 785           }
 786         }
 787       }
 788 
 789       // Verify stack depth and oops in frame
 790       // This assertion may be dependent on the platform we're running on and may need modification (tested on x86 and sparc)
 791       if (!(
 792             /* SPARC */
 793             (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + callee_size_of_parameters) ||
 794             /* x86 */
 795             (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + callee_max_locals) ||
 796             (try_next_mask &&
 797              (iframe->interpreter_frame_expression_stack_size() == (next_mask_expression_stack_size -
 798                                                                     top_frame_expression_stack_adjustment))) ||
 799             (is_top_frame && (exec_mode == Unpack_exception) && iframe->interpreter_frame_expression_stack_size() == 0) ||
 800             (is_top_frame && (exec_mode == Unpack_uncommon_trap || exec_mode == Unpack_reexecute || el->should_reexecute()) &&
 801              (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + cur_invoke_parameter_size))
 802             )) {
 803         {
 804           ttyLocker ttyl;
 805 
 806           // Print out some information that will help us debug the problem
 807           tty->print_cr("Wrong number of expression stack elements during deoptimization");
 808           tty->print_cr("  Error occurred while verifying frame %d (0..%d, 0 is topmost)", i, cur_array->frames() - 1);
 809           tty->print_cr("  Fabricated interpreter frame had %d expression stack elements",
 810                         iframe->interpreter_frame_expression_stack_size());
 811           tty->print_cr("  Interpreter oop map had %d expression stack elements", mask.expression_stack_size());
 812           tty->print_cr("  try_next_mask = %d", try_next_mask);
 813           tty->print_cr("  next_mask_expression_stack_size = %d", next_mask_expression_stack_size);
 814           tty->print_cr("  callee_size_of_parameters = %d", callee_size_of_parameters);
 815           tty->print_cr("  callee_max_locals = %d", callee_max_locals);
 816           tty->print_cr("  top_frame_expression_stack_adjustment = %d", top_frame_expression_stack_adjustment);
 817           tty->print_cr("  exec_mode = %d", exec_mode);
 818           tty->print_cr("  cur_invoke_parameter_size = %d", cur_invoke_parameter_size);
 819           tty->print_cr("  Thread = " INTPTR_FORMAT ", thread ID = %d", p2i(thread), thread->osthread()->thread_id());
 820           tty->print_cr("  Interpreted frames:");
 821           for (int k = 0; k < cur_array->frames(); k++) {
 822             vframeArrayElement* el = cur_array->element(k);
 823             tty->print_cr("    %s (bci %d)", el->method()->name_and_sig_as_C_string(), el->bci());
 824           }
 825           cur_array->print_on_2(tty);
 826         } // release tty lock before calling guarantee
 827         guarantee(false, "wrong number of expression stack elements during deopt");
 828       }
 829       VerifyOopClosure verify;
 830       iframe->oops_interpreted_do(&verify, &rm, false);
 831       callee_size_of_parameters = mh->size_of_parameters();
 832       callee_max_locals = mh->max_locals();
 833       is_top_frame = false;
 834     }
 835   }
 836 #endif /* !PRODUCT */
 837 
 838   return bt;
 839 JRT_END

 933     return lookup(value);
 934   }

1021     oop obj = NULL;

1024 #if INCLUDE_JVMCI || INCLUDE_AOT

1030           // Set the flag to indicate the box came from a cache, so that we can skip the field reassignment for it.
1031           abv->set_cached(true);
1032         }
1033       }
1034 #endif // INCLUDE_JVMCI || INCLUDE_AOT
1035       InstanceKlass* ik = InstanceKlass::cast(k);
1036       if (obj == NULL) {
1037         obj = ik->allocate_instance(THREAD);
1038       }
1039     } else if (k->is_valueArray_klass()) {
1040       ValueArrayKlass* ak = ValueArrayKlass::cast(k);
1041       // Value type array must be zeroed because not all memory is reassigned
1042       obj = ak->allocate(sv->field_size(), THREAD);
1043     } else if (k->is_typeArray_klass()) {
1044       TypeArrayKlass* ak = TypeArrayKlass::cast(k);
1045       assert(sv->field_size() % type2size[ak->element_type()] == 0, "non-integral array length");
1046       int len = sv->field_size() / type2size[ak->element_type()];
1047       obj = ak->allocate(len, THREAD);
1048     } else if (k->is_objArray_klass()) {
1049       ObjArrayKlass* ak = ObjArrayKlass::cast(k);
1050       obj = ak->allocate(sv->field_size(), THREAD);
1051     }
1052 
1053     if (obj == NULL) {
1054       failures = true;
1055     }
1056 
1057     assert(sv->value().is_null(), "redundant reallocation");
1058     assert(obj != NULL || HAS_PENDING_EXCEPTION, "allocation should succeed or we should get an exception");
1059     CLEAR_PENDING_EXCEPTION;
1060     sv->set_value(obj);
1061   }
1062 
1063   if (failures) {
1064     THROW_OOP_(Universe::out_of_memory_error_realloc_objects(), failures);
1065   } else if (pending_exception.not_null()) {
1066     thread->set_pending_exception(pending_exception(), exception_file, exception_line);
1067   }
1068 
1069   return failures;
1070 }

1088 /**
1089  * For primitive types whose kind gets "erased" at runtime (shorts become stack ints),
1090  * we need to somehow be able to recover the actual kind to be able to write the correct
1091  * amount of bytes.
1092  * For that purpose, this method assumes that, for an entry spanning n bytes at index i,
1093  * the entries at index n + 1 to n + i are 'markers'.
1094  * For example, if we were writing a short at index 4 of a byte array of size 8, the
1095  * expected form of the array would be:
1096  *
1097  * {b0, b1, b2, b3, INT, marker, b6, b7}
1098  *
1099  * Thus, in order to get back the size of the entry, we simply need to count the number
1100  * of marked entries
1101  *
1102  * @param virtualArray the virtualized byte array
1103  * @param i index of the virtual entry we are recovering
1104  * @return The number of bytes the entry spans
1105  */
1106 static int count_number_of_bytes_for_entry(ObjectValue *virtualArray, int i) {
1107   int index = i;
1108   while (++index < virtualArray->field_size() &&
1109            virtualArray->field_at(index)->is_marker()) {}
1110   return index - i;
1111 }

1114  * If there was a guarantee for byte array to always start aligned to a long, we could
1115  * do a simple check on the parity of the index. Unfortunately, that is not always the
1116  * case. Thus, we check alignment of the actual address we are writing to.
1117  * In the unlikely case index 0 is 5-aligned for example, it would then be possible to
1118  * write a long to index 3.
1119  */

1121     jbyte* res = obj->byte_at_addr(index);
1122     assert((((intptr_t) res) % expected_alignment) == 0, "Non-aligned write");

1133       break;
1134     case 4:

1136       break;
1137     case 8:

1147 // restore elements of an eliminated type array
1148 void Deoptimization::reassign_type_array_elements(frame* fr, RegisterMap* reg_map, ObjectValue* sv, typeArrayOop obj, BasicType type) {
1149   int index = 0;
1150   intptr_t val;
1151 
1152   for (int i = 0; i < sv->field_size(); i++) {
1153     StackValue* value = StackValue::create_stack_value(fr, reg_map, sv->field_at(i));
1154     switch(type) {
1155     case T_LONG: case T_DOUBLE: {
1156       assert(value->type() == T_INT, "Agreement.");
1157       StackValue* low =
1158         StackValue::create_stack_value(fr, reg_map, sv->field_at(++i));
1159 #ifdef _LP64
1160       jlong res = (jlong)low->get_int();
1161 #else
1162       jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1163 #endif
1164       obj->long_at_put(index, res);
1165       break;
1166     }
1167 
1168     // Have to cast to INT (32 bits) pointer to avoid little/big-endian problem.
1169     case T_INT: case T_FLOAT: { // 4 bytes.
1170       assert(value->type() == T_INT, "Agreement.");
1171       bool big_value = false;
1172       if (i + 1 < sv->field_size() && type == T_INT) {
1173         if (sv->field_at(i)->is_location()) {
1174           Location::Type type = ((LocationValue*) sv->field_at(i))->location().type();
1175           if (type == Location::dbl || type == Location::lng) {
1176             big_value = true;
1177           }
1178         } else if (sv->field_at(i)->is_constant_int()) {
1179           ScopeValue* next_scope_field = sv->field_at(i + 1);
1180           if (next_scope_field->is_constant_long() || next_scope_field->is_constant_double()) {
1181             big_value = true;
1182           }
1183         }
1184       }
1185 
1186       if (big_value) {
1187         StackValue* low = StackValue::create_stack_value(fr, reg_map, sv->field_at(++i));
1188   #ifdef _LP64
1189         jlong res = (jlong)low->get_int();
1190   #else
1191         jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1192   #endif
1193         obj->int_at_put(index, (jint)*((jint*)&res));
1194         obj->int_at_put(++index, (jint)*(((jint*)&res) + 1));
1195       } else {
1196         val = value->get_int();
1197         obj->int_at_put(index, (jint)*((jint*)&val));
1198       }
1199       break;
1200     }
1201 
1202     case T_SHORT:
1203       assert(value->type() == T_INT, "Agreement.");
1204       val = value->get_int();
1205       obj->short_at_put(index, (jshort)*((jint*)&val));
1206       break;
1207 
1208     case T_CHAR:
1209       assert(value->type() == T_INT, "Agreement.");
1210       val = value->get_int();
1211       obj->char_at_put(index, (jchar)*((jint*)&val));
1212       break;
1213 
1214     case T_BYTE: {
1215       assert(value->type() == T_INT, "Agreement.");
1216       // The value we get is erased as a regular int. We will need to find its actual byte count 'by hand'.
1217       val = value->get_int();
1218 #if INCLUDE_JVMCI
1219       int byte_count = count_number_of_bytes_for_entry(sv, i);
1220       byte_array_put(obj, val, index, byte_count);
1221       // According to byte_count contract, the values from i + 1 to i + byte_count are illegal values. Skip.
1222       i += byte_count - 1; // Balance the loop counter.
1223       index += byte_count;
1224       // index has been updated so continue at top of loop
1225       continue;
1226 #else
1227       obj->byte_at_put(index, (jbyte)*((jint*)&val));
1228       break;
1229 #endif // INCLUDE_JVMCI
1230     }
1231 
1232     case T_BOOLEAN: {
1233       assert(value->type() == T_INT, "Agreement.");
1234       val = value->get_int();
1235       obj->bool_at_put(index, (jboolean)*((jint*)&val));
1236       break;
1237     }
1238 
1239       default:
1240         ShouldNotReachHere();
1241     }
1242     index++;
1243   }
1244 }

1272 // Restore fields of an eliminated instance object using the same field order
1273 // returned by HotSpotResolvedObjectTypeImpl.getInstanceFields(true)
1274 static int reassign_fields_by_klass(InstanceKlass* klass, frame* fr, RegisterMap* reg_map, ObjectValue* sv, int svIndex, oop obj, bool skip_internal, int base_offset, TRAPS) {
1275 
1276   GrowableArray<ReassignedField>* fields = new GrowableArray<ReassignedField>();
1277   InstanceKlass* ik = klass;
1278   while (ik != NULL) {
1279     for (AllFieldStream fs(ik); !fs.done(); fs.next()) {
1280       if (!fs.access_flags().is_static() && (!skip_internal || !fs.access_flags().is_internal())) {
1281         ReassignedField field;
1282         field._offset = fs.offset();
1283         field._type = Signature::basic_type(fs.signature());
1284         if (field._type == T_INLINE_TYPE) {
1285           field._type = T_OBJECT;
1286         }
1287         if (fs.is_inlined()) {
1288           // Resolve klass of flattened value type field
1289           Klass* vk = klass->get_inline_type_field_klass(fs.index());
1290           field._klass = ValueKlass::cast(vk);
1291           field._type = T_INLINE_TYPE;
1292         }
1293         fields->append(field);
1294       }
1295     }
1296     ik = ik->superklass();
1297   }
1298   fields->sort(compare);
1299   for (int i = 0; i < fields->length(); i++) {
1300     intptr_t val;
1301     ScopeValue* scope_field = sv->field_at(svIndex);
1302     StackValue* value = StackValue::create_stack_value(fr, reg_map, scope_field);
1303     int offset = base_offset + fields->at(i)._offset;
1304     BasicType type = fields->at(i)._type;
1305     switch (type) {
1306       case T_OBJECT:
1307       case T_ARRAY:
1308         assert(value->type() == T_OBJECT, "Agreement.");
1309         obj->obj_field_put(offset, value->get_obj()());
1310         break;
1311 
1312       case T_INLINE_TYPE: {
1313         // Recursively re-assign flattened value type fields
1314         InstanceKlass* vk = fields->at(i)._klass;
1315         assert(vk != NULL, "must be resolved");
1316         offset -= ValueKlass::cast(vk)->first_field_offset(); // Adjust offset to omit oop header
1317         svIndex = reassign_fields_by_klass(vk, fr, reg_map, sv, svIndex, obj, skip_internal, offset, CHECK_0);
1318         continue; // Continue because we don't need to increment svIndex
1319       }
1320 
1321       // Have to cast to INT (32 bits) pointer to avoid little/big-endian problem.
1322       case T_INT: case T_FLOAT: { // 4 bytes.
1323         assert(value->type() == T_INT, "Agreement.");
1324         bool big_value = false;
1325         if (i+1 < fields->length() && fields->at(i+1)._type == T_INT) {
1326           if (scope_field->is_location()) {
1327             Location::Type type = ((LocationValue*) scope_field)->location().type();
1328             if (type == Location::dbl || type == Location::lng) {
1329               big_value = true;
1330             }
1331           }
1332           if (scope_field->is_constant_int()) {
1333             ScopeValue* next_scope_field = sv->field_at(svIndex + 1);
1334             if (next_scope_field->is_constant_long() || next_scope_field->is_constant_double()) {
1335               big_value = true;
1336             }
1337           }
1338         }
1339 
1340         if (big_value) {
1341           i++;
1342           assert(i < fields->length(), "second T_INT field needed");
1343           assert(fields->at(i)._type == T_INT, "T_INT field needed");
1344         } else {
1345           val = value->get_int();
1346           obj->int_field_put(offset, (jint)*((jint*)&val));
1347           break;
1348         }
1349       }
1350         /* no break */
1351 
1352       case T_LONG: case T_DOUBLE: {
1353         assert(value->type() == T_INT, "Agreement.");
1354         StackValue* low = StackValue::create_stack_value(fr, reg_map, sv->field_at(++svIndex));
1355 #ifdef _LP64
1356         jlong res = (jlong)low->get_int();
1357 #else
1358         jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1359 #endif
1360         obj->long_field_put(offset, res);
1361         break;
1362       }
1363 
1364       case T_SHORT:
1365         assert(value->type() == T_INT, "Agreement.");
1366         val = value->get_int();
1367         obj->short_field_put(offset, (jshort)*((jint*)&val));
1368         break;
1369 
1370       case T_CHAR:
1371         assert(value->type() == T_INT, "Agreement.");
1372         val = value->get_int();
1373         obj->char_field_put(offset, (jchar)*((jint*)&val));
1374         break;
1375 
1376       case T_BYTE:
1377         assert(value->type() == T_INT, "Agreement.");
1378         val = value->get_int();
1379         obj->byte_field_put(offset, (jbyte)*((jint*)&val));
1380         break;
1381 
1382       case T_BOOLEAN:
1383         assert(value->type() == T_INT, "Agreement.");
1384         val = value->get_int();
1385         obj->bool_field_put(offset, (jboolean)*((jint*)&val));
1386         break;
1387 
1388       default:
1389         ShouldNotReachHere();
1390     }
1391     svIndex++;
1392   }
1393   return svIndex;
1394 }

1445 // relock objects for which synchronization was eliminated
1446 void Deoptimization::relock_objects(GrowableArray<MonitorInfo*>* monitors, JavaThread* thread, bool realloc_failures) {
1447   for (int i = 0; i < monitors->length(); i++) {
1448     MonitorInfo* mon_info = monitors->at(i);
1449     if (mon_info->eliminated()) {
1450       assert(!mon_info->owner_is_scalar_replaced() || realloc_failures, "reallocation was missed");

1452         Handle obj(thread, mon_info->owner());
1453         markWord mark = obj->mark();
1454         if (UseBiasedLocking && mark.has_bias_pattern()) {
1455           // New allocated objects may have the mark set to anonymously biased.
1456           // Also the deoptimized method may called methods with synchronization
1457           // where the thread-local object is bias locked to the current thread.
1458           assert(mark.is_biased_anonymously() ||
1459                  mark.biased_locker() == thread, "should be locked to current thread");
1460           // Reset mark word to unbiased prototype.
1461           markWord unbiased_prototype = markWord::prototype().set_age(mark.age());
1462           obj->set_mark(unbiased_prototype);
1463         }
1464         BasicLock* lock = mon_info->lock();
1465         ObjectSynchronizer::enter(obj, lock, thread);
1466         assert(mon_info->owner()->is_locked(), "object must be locked now");
1467       }
1468     }
1469   }

1566   thread->set_frames_to_pop_failed_realloc(array->frames());
1567   // Unlock all monitors here otherwise the interpreter will see a
1568   // mix of locked and unlocked monitors (because of failed
1569   // reallocations of synchronized objects) and be confused.
1570   for (int i = 0; i < array->frames(); i++) {
1571     MonitorChunk* monitors = array->element(i)->monitors();
1572     if (monitors != NULL) {
1573       for (int j = 0; j < monitors->number_of_monitors(); j++) {
1574         BasicObjectLock* src = monitors->at(j);
1575         if (src->obj() != NULL) {
1576           ObjectSynchronizer::exit(src->obj(), src->lock(), thread);
1577         }
1578       }
1579       array->element(i)->free_monitors(thread);
1580 #ifdef ASSERT
1581       array->element(i)->set_removed_monitors();
1582 #endif
1583     }
1584   }
1585 }

1599 static void get_monitors_from_stack(GrowableArray<Handle>* objects_to_revoke, JavaThread* thread, frame fr, RegisterMap* map) {
1600   // Unfortunately we don't have a RegisterMap available in most of
1601   // the places we want to call this routine so we need to walk the
1602   // stack again to update the register map.
1603   if (map == NULL || !map->update_map()) {
1604     StackFrameStream sfs(thread, true);
1605     bool found = false;
1606     while (!found && !sfs.is_done()) {
1607       frame* cur = sfs.current();
1608       sfs.next();
1609       found = cur->id() == fr.id();
1610     }
1611     assert(found, "frame to be deoptimized not found on target thread's stack");
1612     map = sfs.register_map();
1613   }
1614 
1615   vframe* vf = vframe::new_vframe(&fr, map, thread);
1616   compiledVFrame* cvf = compiledVFrame::cast(vf);
1617   // Revoke monitors' biases in all scopes
1618   while (!cvf->is_top()) {
1619     collect_monitors(cvf, objects_to_revoke);
1620     cvf = compiledVFrame::cast(cvf->sender());
1621   }
1622   collect_monitors(cvf, objects_to_revoke);
1623 }

1625 void Deoptimization::revoke_from_deopt_handler(JavaThread* thread, frame fr, RegisterMap* map) {
1626   if (!UseBiasedLocking) {
1627     return;
1628   }
1629   GrowableArray<Handle>* objects_to_revoke = new GrowableArray<Handle>();
1630   get_monitors_from_stack(objects_to_revoke, thread, fr, map);

1860   HandleMark hm;
1861 
1862   // uncommon_trap() is called at the beginning of the uncommon trap
1863   // handler. Note this fact before we start generating temporary frames
1864   // that can confuse an asynchronous stack walker. This counter is
1865   // decremented at the end of unpack_frames().
1866   thread->inc_in_deopt_handler();
1867 
1868   // We need to update the map if we have biased locking.
1869 #if INCLUDE_JVMCI
1870   // JVMCI might need to get an exception from the stack, which in turn requires the register map to be valid
1871   RegisterMap reg_map(thread, true);
1872 #else
1873   RegisterMap reg_map(thread, UseBiasedLocking);
1874 #endif

1876   frame fr = stub_frame.sender(&reg_map);
1877   // Make sure the calling nmethod is not getting deoptimized and removed
1878   // before we are done with it.
1879   nmethodLocker nl(fr.pc());
1880 
1881   // Log a message
1882   Events::log_deopt_message(thread, "Uncommon trap: trap_request=" PTR32_FORMAT " fr.pc=" INTPTR_FORMAT " relative=" INTPTR_FORMAT,
1883               trap_request, p2i(fr.pc()), fr.pc() - fr.cb()->code_begin());
1884 
1885   {
1886     ResourceMark rm;
1887 
1888     DeoptReason reason = trap_request_reason(trap_request);
1889     DeoptAction action = trap_request_action(trap_request);
1890 #if INCLUDE_JVMCI
1891     int debug_id = trap_request_debug_id(trap_request);
1892 #endif
1893     jint unloaded_class_index = trap_request_index(trap_request); // CP idx or -1
1894 
1895     vframe*  vf  = vframe::new_vframe(&fr, &reg_map, thread);
1896     compiledVFrame* cvf = compiledVFrame::cast(vf);
1897 
1898     CompiledMethod* nm = cvf->code();
1899 
1900     ScopeDesc*      trap_scope  = cvf->scope();
1901 
1902     if (TraceDeoptimization) {
1903       ttyLocker ttyl;
1904       tty->print_cr("  bci=%d pc=" INTPTR_FORMAT ", relative_pc=" INTPTR_FORMAT ", method=%s" JVMCI_ONLY(", debug_id=%d"), trap_scope->bci(), p2i(fr.pc()), fr.pc() - nm->code_begin(), trap_scope->method()->name_and_sig_as_C_string()
1905 #if INCLUDE_JVMCI
1906           , debug_id
1907 #endif
1908           );
1909     }
1910 
1911     methodHandle    trap_method(THREAD, trap_scope->method());
1912     int             trap_bci    = trap_scope->bci();
1913 #if INCLUDE_JVMCI
1914     jlong           speculation = thread->pending_failed_speculation();
1915     if (nm->is_compiled_by_jvmci() && nm->is_nmethod()) { // Exclude AOTed methods
1916       nm->as_nmethod()->update_speculation(thread);
1917     } else {
1918       assert(speculation == 0, "There should not be a speculation for methods compiled by non-JVMCI compilers");
1919     }
1920 
1921     if (trap_bci == SynchronizationEntryBCI) {
1922       trap_bci = 0;
1923       thread->set_pending_monitorenter(true);
1924     }
1925 
1926     if (reason == Deoptimization::Reason_transfer_to_interpreter) {
1927       thread->set_pending_transfer_to_interpreter(true);
1928     }
1929 #endif
1930 
1931     Bytecodes::Code trap_bc     = trap_method->java_code_at(trap_bci);
1932     // Record this event in the histogram.
1933     gather_statistics(reason, action, trap_bc);
1934 
1935     // Ensure that we can record deopt. history:
1936     // Need MDO to record RTM code generation state.
1937     bool create_if_missing = ProfileTraps || UseCodeAging RTM_OPT_ONLY( || UseRTMLocking );

1939     methodHandle profiled_method;
1940 #if INCLUDE_JVMCI
1941     if (nm->is_compiled_by_jvmci()) {
1942       profiled_method = methodHandle(THREAD, nm->method());
1943     } else {
1944       profiled_method = trap_method;

1946 #else
1947     profiled_method = trap_method;
1948 #endif

1951       get_method_data(thread, profiled_method, create_if_missing);
1952 
1953     JFR_ONLY(post_deoptimization_event(nm, trap_method(), trap_bci, trap_bc, reason, action);)
1954 
1955     // Log a message
1956     Events::log_deopt_message(thread, "Uncommon trap: reason=%s action=%s pc=" INTPTR_FORMAT " method=%s @ %d %s",
1957                               trap_reason_name(reason), trap_action_name(action), p2i(fr.pc()),
1958                               trap_method->name_and_sig_as_C_string(), trap_bci, nm->compiler_name());
1959 
1960     // Print a bunch of diagnostics, if requested.
1961     if (TraceDeoptimization || LogCompilation) {
1962       ResourceMark rm;
1963       ttyLocker ttyl;
1964       char buf[100];
1965       if (xtty != NULL) {
1966         xtty->begin_head("uncommon_trap thread='" UINTX_FORMAT "' %s",
1967                          os::current_thread_id(),
1968                          format_trap_request(buf, sizeof(buf), trap_request));
1969 #if INCLUDE_JVMCI
1970         if (speculation != 0) {
1971           xtty->print(" speculation='" JLONG_FORMAT "'", speculation);
1972         }
1973 #endif
1974         nm->log_identity(xtty);
1975       }
1976       Symbol* class_name = NULL;
1977       bool unresolved = false;
1978       if (unloaded_class_index >= 0) {
1979         constantPoolHandle constants (THREAD, trap_method->constants());
1980         if (constants->tag_at(unloaded_class_index).is_unresolved_klass()) {
1981           class_name = constants->klass_name_at(unloaded_class_index);
1982           unresolved = true;
1983           if (xtty != NULL)
1984             xtty->print(" unresolved='1'");
1985         } else if (constants->tag_at(unloaded_class_index).is_symbol()) {
1986           class_name = constants->symbol_at(unloaded_class_index);
1987         }
1988         if (xtty != NULL)
1989           xtty->name(class_name);
1990       }
1991       if (xtty != NULL && trap_mdo != NULL && (int)reason < (int)MethodData::_trap_hist_limit) {
1992         // Dump the relevant MDO state.
1993         // This is the deopt count for the current reason, any previous
1994         // reasons or recompiles seen at this point.
1995         int dcnt = trap_mdo->trap_count(reason);
1996         if (dcnt != 0)
1997           xtty->print(" count='%d'", dcnt);
1998         ProfileData* pdata = trap_mdo->bci_to_data(trap_bci);
1999         int dos = (pdata == NULL)? 0: pdata->trap_state();
2000         if (dos != 0) {
2001           xtty->print(" state='%s'", format_trap_state(buf, sizeof(buf), dos));
2002           if (trap_state_is_recompiled(dos)) {
2003             int recnt2 = trap_mdo->overflow_recompile_count();
2004             if (recnt2 != 0)
2005               xtty->print(" recompiles2='%d'", recnt2);
2006           }
2007         }
2008       }
2009       if (xtty != NULL) {
2010         xtty->stamp();
2011         xtty->end_head();
2012       }
2013       if (TraceDeoptimization) {  // make noise on the tty
2014         tty->print("Uncommon trap occurred in");
2015         nm->method()->print_short_name(tty);
2016         tty->print(" compiler=%s compile_id=%d", nm->compiler_name(), nm->compile_id());
2017 #if INCLUDE_JVMCI
2018         if (nm->is_nmethod()) {
2019           const char* installed_code_name = nm->as_nmethod()->jvmci_name();
2020           if (installed_code_name != NULL) {
2021             tty->print(" (JVMCI: installed code name=%s) ", installed_code_name);
2022           }
2023         }
2024 #endif
2025         tty->print(" (@" INTPTR_FORMAT ") thread=" UINTX_FORMAT " reason=%s action=%s unloaded_class_index=%d" JVMCI_ONLY(" debug_id=%d"),
2026                    p2i(fr.pc()),
2027                    os::current_thread_id(),
2028                    trap_reason_name(reason),
2029                    trap_action_name(action),
2030                    unloaded_class_index
2031 #if INCLUDE_JVMCI
2032                    , debug_id
2033 #endif
2034                    );
2035         if (class_name != NULL) {
2036           tty->print(unresolved ? " unresolved class: " : " symbol: ");
2037           class_name->print_symbol_on(tty);
2038         }
2039         tty->cr();
2040       }
2041       if (xtty != NULL) {
2042         // Log the precise location of the trap.
2043         for (ScopeDesc* sd = trap_scope; ; sd = sd->sender()) {
2044           xtty->begin_elem("jvms bci='%d'", sd->bci());
2045           xtty->method(sd->method());
2046           xtty->end_elem();
2047           if (sd->is_top())  break;
2048         }
2049         xtty->tail("uncommon_trap");
2050       }
2051     }
2052     // (End diagnostic printout.)
2053 
2054     // Load class if necessary
2055     if (unloaded_class_index >= 0) {
2056       constantPoolHandle constants(THREAD, trap_method->constants());
2057       load_class_by_index(constants, unloaded_class_index);
2058     }
2059 
2060     // Flush the nmethod if necessary and desirable.
2061     //
2062     // We need to avoid situations where we are re-flushing the nmethod
2063     // because of a hot deoptimization site.  Repeated flushes at the same
2064     // point need to be detected by the compiler and avoided.  If the compiler
2065     // cannot avoid them (or has a bug and "refuses" to avoid them), this
2066     // module must take measures to avoid an infinite cycle of recompilation
2067     // and deoptimization.  There are several such measures:
2068     //
2069     //   1. If a recompilation is ordered a second time at some site X
2070     //   and for the same reason R, the action is adjusted to 'reinterpret',
2071     //   to give the interpreter time to exercise the method more thoroughly.
2072     //   If this happens, the method's overflow_recompile_count is incremented.
2073     //
2074     //   2. If the compiler fails to reduce the deoptimization rate, then
2075     //   the method's overflow_recompile_count will begin to exceed the set
2076     //   limit PerBytecodeRecompilationCutoff.  If this happens, the action
2077     //   is adjusted to 'make_not_compilable', and the method is abandoned
2078     //   to the interpreter.  This is a performance hit for hot methods,
2079     //   but is better than a disastrous infinite cycle of recompilations.
2080     //   (Actually, only the method containing the site X is abandoned.)
2081     //
2082     //   3. In parallel with the previous measures, if the total number of
2083     //   recompilations of a method exceeds the much larger set limit
2084     //   PerMethodRecompilationCutoff, the method is abandoned.
2085     //   This should only happen if the method is very large and has
2086     //   many "lukewarm" deoptimizations.  The code which enforces this
2087     //   limit is elsewhere (class nmethod, class Method).
2088     //
2089     // Note that the per-BCI 'is_recompiled' bit gives the compiler one chance
2090     // to recompile at each bytecode independently of the per-BCI cutoff.
2091     //
2092     // The decision to update code is up to the compiler, and is encoded
2093     // in the Action_xxx code.  If the compiler requests Action_none
2094     // no trap state is changed, no compiled code is changed, and the
2095     // computation suffers along in the interpreter.
2096     //
2097     // The other action codes specify various tactics for decompilation
2098     // and recompilation.  Action_maybe_recompile is the loosest, and
2099     // allows the compiled code to stay around until enough traps are seen,
2100     // and until the compiler gets around to recompiling the trapping method.
2101     //
2102     // The other actions cause immediate removal of the present code.
2103 
2104     // Traps caused by injected profile shouldn't pollute trap counts.
2105     bool injected_profile_trap = trap_method->has_injected_profile() &&
2106                                  (reason == Reason_intrinsic || reason == Reason_unreached);
2107 
2108     bool update_trap_state = (reason != Reason_tenured) && !injected_profile_trap;
2109     bool make_not_entrant = false;
2110     bool make_not_compilable = false;
2111     bool reprofile = false;
2112     switch (action) {
2113     case Action_none:
2114       // Keep the old code.
2115       update_trap_state = false;
2116       break;
2117     case Action_maybe_recompile:
2118       // Do not need to invalidate the present code, but we can
2119       // initiate another
2120       // Start compiler without (necessarily) invalidating the nmethod.
2121       // The system will tolerate the old code, but new code should be
2122       // generated when possible.
2123       break;
2124     case Action_reinterpret:
2125       // Go back into the interpreter for a while, and then consider
2126       // recompiling form scratch.
2127       make_not_entrant = true;
2128       // Reset invocation counter for outer most method.
2129       // This will allow the interpreter to exercise the bytecodes
2130       // for a while before recompiling.
2131       // By contrast, Action_make_not_entrant is immediate.
2132       //
2133       // Note that the compiler will track null_check, null_assert,
2134       // range_check, and class_check events and log them as if they
2135       // had been traps taken from compiled code.  This will update
2136       // the MDO trap history so that the next compilation will
2137       // properly detect hot trap sites.
2138       reprofile = true;
2139       break;
2140     case Action_make_not_entrant:
2141       // Request immediate recompilation, and get rid of the old code.
2142       // Make them not entrant, so next time they are called they get
2143       // recompiled.  Unloaded classes are loaded now so recompile before next
2144       // time they are called.  Same for uninitialized.  The interpreter will
2145       // link the missing class, if any.
2146       make_not_entrant = true;
2147       break;
2148     case Action_make_not_compilable:
2149       // Give up on compiling this method at all.
2150       make_not_entrant = true;
2151       make_not_compilable = true;
2152       break;
2153     default:
2154       ShouldNotReachHere();
2155     }
2156 
2157     // Setting +ProfileTraps fixes the following, on all platforms:
2158     // 4852688: ProfileInterpreter is off by default for ia64.  The result is
2159     // infinite heroic-opt-uncommon-trap/deopt/recompile cycles, since the
2160     // recompile relies on a MethodData* to record heroic opt failures.
2161 
2162     // Whether the interpreter is producing MDO data or not, we also need
2163     // to use the MDO to detect hot deoptimization points and control
2164     // aggressive optimization.
2165     bool inc_recompile_count = false;
2166     ProfileData* pdata = NULL;
2167     if (ProfileTraps && !is_client_compilation_mode_vm() && update_trap_state && trap_mdo != NULL) {
2168       assert(trap_mdo == get_method_data(thread, profiled_method, false), "sanity");
2169       uint this_trap_count = 0;
2170       bool maybe_prior_trap = false;
2171       bool maybe_prior_recompile = false;
2172       pdata = query_update_method_data(trap_mdo, trap_bci, reason, true,
2173 #if INCLUDE_JVMCI
2174                                    nm->is_compiled_by_jvmci() && nm->is_osr_method(),
2175 #endif
2176                                    nm->method(),
2177                                    //outputs:
2178                                    this_trap_count,
2179                                    maybe_prior_trap,
2180                                    maybe_prior_recompile);
2181       // Because the interpreter also counts null, div0, range, and class
2182       // checks, these traps from compiled code are double-counted.
2183       // This is harmless; it just means that the PerXTrapLimit values
2184       // are in effect a little smaller than they look.
2185 
2186       DeoptReason per_bc_reason = reason_recorded_per_bytecode_if_any(reason);
2187       if (per_bc_reason != Reason_none) {
2188         // Now take action based on the partially known per-BCI history.
2189         if (maybe_prior_trap
2190             && this_trap_count >= (uint)PerBytecodeTrapLimit) {
2191           // If there are too many traps at this BCI, force a recompile.
2192           // This will allow the compiler to see the limit overflow, and
2193           // take corrective action, if possible.  The compiler generally
2194           // does not use the exact PerBytecodeTrapLimit value, but instead
2195           // changes its tactics if it sees any traps at all.  This provides
2196           // a little hysteresis, delaying a recompile until a trap happens
2197           // several times.
2198           //
2199           // Actually, since there is only one bit of counter per BCI,
2200           // the possible per-BCI counts are {0,1,(per-method count)}.
2201           // This produces accurate results if in fact there is only
2202           // one hot trap site, but begins to get fuzzy if there are
2203           // many sites.  For example, if there are ten sites each
2204           // trapping two or more times, they each get the blame for
2205           // all of their traps.
2206           make_not_entrant = true;
2207         }
2208 
2209         // Detect repeated recompilation at the same BCI, and enforce a limit.
2210         if (make_not_entrant && maybe_prior_recompile) {
2211           // More than one recompile at this point.
2212           inc_recompile_count = maybe_prior_trap;
2213         }
2214       } else {
2215         // For reasons which are not recorded per-bytecode, we simply
2216         // force recompiles unconditionally.
2217         // (Note that PerMethodRecompilationCutoff is enforced elsewhere.)
2218         make_not_entrant = true;
2219       }
2220 
2221       // Go back to the compiler if there are too many traps in this method.
2222       if (this_trap_count >= per_method_trap_limit(reason)) {
2223         // If there are too many traps in this method, force a recompile.
2224         // This will allow the compiler to see the limit overflow, and
2225         // take corrective action, if possible.
2226         // (This condition is an unlikely backstop only, because the
2227         // PerBytecodeTrapLimit is more likely to take effect first,
2228         // if it is applicable.)
2229         make_not_entrant = true;
2230       }
2231 
2232       // Here's more hysteresis:  If there has been a recompile at
2233       // this trap point already, run the method in the interpreter
2234       // for a while to exercise it more thoroughly.
2235       if (make_not_entrant && maybe_prior_recompile && maybe_prior_trap) {
2236         reprofile = true;
2237       }
2238     }
2239 
2240     // Take requested actions on the method:
2241 
2242     // Recompile
2243     if (make_not_entrant) {
2244       if (!nm->make_not_entrant()) {
2245         return; // the call did not change nmethod's state
2246       }
2247 
2248       if (pdata != NULL) {
2249         // Record the recompilation event, if any.
2250         int tstate0 = pdata->trap_state();
2251         int tstate1 = trap_state_set_recompiled(tstate0, true);
2252         if (tstate1 != tstate0)
2253           pdata->set_trap_state(tstate1);
2254       }
2255 
2256 #if INCLUDE_RTM_OPT
2257       // Restart collecting RTM locking abort statistic if the method
2258       // is recompiled for a reason other than RTM state change.
2259       // Assume that in new recompiled code the statistic could be different,
2260       // for example, due to different inlining.
2261       if ((reason != Reason_rtm_state_change) && (trap_mdo != NULL) &&
2262           UseRTMDeopt && (nm->as_nmethod()->rtm_state() != ProfileRTM)) {
2263         trap_mdo->atomic_set_rtm_state(ProfileRTM);
2264       }
2265 #endif
2266       // For code aging we count traps separately here, using make_not_entrant()
2267       // as a guard against simultaneous deopts in multiple threads.
2268       if (reason == Reason_tenured && trap_mdo != NULL) {
2269         trap_mdo->inc_tenure_traps();
2270       }
2271     }
2272 
2273     if (inc_recompile_count) {
2274       trap_mdo->inc_overflow_recompile_count();
2275       if ((uint)trap_mdo->overflow_recompile_count() >
2276           (uint)PerBytecodeRecompilationCutoff) {
2277         // Give up on the method containing the bad BCI.
2278         if (trap_method() == nm->method()) {
2279           make_not_compilable = true;
2280         } else {
2281           trap_method->set_not_compilable("overflow_recompile_count > PerBytecodeRecompilationCutoff", CompLevel_full_optimization);
2282           // But give grace to the enclosing nm->method().
2283         }
2284       }
2285     }
2286 
2287     // Reprofile
2288     if (reprofile) {
2289       CompilationPolicy::policy()->reprofile(trap_scope, nm->is_osr_method());
2290     }
2291 
2292     // Give up compiling
2293     if (make_not_compilable && !nm->method()->is_not_compilable(CompLevel_full_optimization)) {
2294       assert(make_not_entrant, "consistent");
2295       nm->method()->set_not_compilable("give up compiling", CompLevel_full_optimization);
2296     }
2297 
2298   } // Free marked resources
2299 
2300 }

2317   bool maybe_prior_recompile = false;

2329     // If the runtime cannot find a place to store trap history,

2335     maybe_prior_recompile = (trap_mdo->decompile_count() != 0);

2342   assert(per_bc_reason != Reason_none || update_total_trap_count, "must be");
2343   if (per_bc_reason != Reason_none) {

2347     pdata = trap_mdo->allocate_bci_to_data(trap_bci, reason_is_speculate(reason) ? compiled_method : NULL);

2356       }

2358       int tstate0 = pdata->trap_state();
2359       if (!trap_state_has_reason(tstate0, per_bc_reason))
2360         maybe_prior_trap = false;
2361       if (!trap_state_is_recompiled(tstate0))
2362         maybe_prior_recompile = false;
2363 
2364       // Update the trap state of this profile datum.
2365       int tstate1 = tstate0;
2366       // Record the reason.
2367       tstate1 = trap_state_add_reason(tstate1, per_bc_reason);
2368       // Store the updated state on the MDO, for next time.
2369       if (tstate1 != tstate0)
2370         pdata->set_trap_state(tstate1);
2371     } else {
2372       if (LogCompilation && xtty != NULL) {
2373         ttyLocker ttyl;
2374         // Missing MDP?  Leave a small complaint in the log.
2375         xtty->elem("missing_mdp bci='%d'", trap_bci);
2376       }
2377     }
2378   }
2379

2381   ret_this_trap_count = this_trap_count;
2382   ret_maybe_prior_trap = maybe_prior_trap;

2423 const int DS_REASON_MASK   = ((uint)DataLayout::trap_mask) >> 1;
2424 const int DS_RECOMPILE_BIT = DataLayout::trap_mask - DS_REASON_MASK;

2426 //---------------------------trap_state_reason---------------------------------
2427 Deoptimization::DeoptReason
2428 Deoptimization::trap_state_reason(int trap_state) {
2429   // This assert provides the link between the width of DataLayout::trap_bits
2430   // and the encoding of "recorded" reasons.  It ensures there are enough
2431   // bits to store all needed reasons in the per-BCI MDO profile.
2432   assert(DS_REASON_MASK >= Reason_RECORDED_LIMIT, "enough bits");
2433   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2434   trap_state -= recompile_bit;
2435   if (trap_state == DS_REASON_MASK) {
2436     return Reason_many;
2437   } else {
2438     assert((int)Reason_none == 0, "state=0 => Reason_none");
2439     return (DeoptReason)trap_state;
2440   }
2441 }
2442 //-------------------------trap_state_has_reason-------------------------------
2443 int Deoptimization::trap_state_has_reason(int trap_state, int reason) {
2444   assert(reason_is_recorded_per_bytecode((DeoptReason)reason), "valid reason");
2445   assert(DS_REASON_MASK >= Reason_RECORDED_LIMIT, "enough bits");
2446   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2447   trap_state -= recompile_bit;
2448   if (trap_state == DS_REASON_MASK) {
2449     return -1;  // true, unspecifically (bottom of state lattice)
2450   } else if (trap_state == reason) {
2451     return 1;   // true, definitely
2452   } else if (trap_state == 0) {
2453     return 0;   // false, definitely (top of state lattice)
2454   } else {
2455     return 0;   // false, definitely
2456   }
2457 }
2458 //-------------------------trap_state_add_reason-------------------------------
2459 int Deoptimization::trap_state_add_reason(int trap_state, int reason) {
2460   assert(reason_is_recorded_per_bytecode((DeoptReason)reason) || reason == Reason_many, "valid reason");
2461   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2462   trap_state -= recompile_bit;
2463   if (trap_state == DS_REASON_MASK) {
2464     return trap_state + recompile_bit;     // already at state lattice bottom
2465   } else if (trap_state == reason) {
2466     return trap_state + recompile_bit;     // the condition is already true
2467   } else if (trap_state == 0) {
2468     return reason + recompile_bit;          // no condition has yet been true
2469   } else {
2470     return DS_REASON_MASK + recompile_bit;  // fall to state lattice bottom
2471   }
2472 }
2473 //-----------------------trap_state_is_recompiled------------------------------
2474 bool Deoptimization::trap_state_is_recompiled(int trap_state) {
2475   return (trap_state & DS_RECOMPILE_BIT) != 0;
2476 }

2618 void Deoptimization::gather_statistics(DeoptReason reason, DeoptAction action,

2628   if (bc != Bytecodes::_illegal) {

2630       juint* counter_addr = &cases[bc_case];
2631       juint  counter = *counter_addr;
2632       if ((counter == 0 && bc_counter_addr == NULL)
2633           || (Bytecodes::Code)(counter & LSB_MASK) == bc) {
2634         // this counter is either free or is already devoted to this BC
2635         bc_counter_addr = counter_addr;
2636         bc_counter = counter | bc;

2644   }
2645   *bc_counter_addr = bc_counter + (1 << LSB_BITS);
2646 }

2652 void Deoptimization::print_statistics() {
2653   juint total = total_deoptimization_count();
2654   juint account = total;
2655   if (total != 0) {
2656     ttyLocker ttyl;
2657     if (xtty != NULL)  xtty->head("statistics type='deoptimization'");
2658     tty->print_cr("Deoptimization traps recorded:");
2659     #define PRINT_STAT_LINE(name, r) \
2660       tty->print_cr("  %4d (%4.1f%%) %s", (int)(r), ((r) * 100.0) / total, name);
2661     PRINT_STAT_LINE("total", total);
2662     // For each non-zero entry in the histogram, print the reason,
2663     // the action, and (if specifically known) the type of bytecode.
2664     for (int reason = 0; reason < Reason_LIMIT; reason++) {
2665       for (int action = 0; action < Action_LIMIT; action++) {
2666         juint* cases = _deoptimization_hist[reason][1+action];
2667         for (int bc_case = 0; bc_case < BC_CASE_LIMIT; bc_case++) {
2668           juint counter = cases[bc_case];
2669           if (counter != 0) {
2670             char name[1*K];
2671             Bytecodes::Code bc = (Bytecodes::Code)(counter & LSB_MASK);
2672             if (bc_case == BC_CASE_LIMIT && (int)bc == 0)
2673               bc = Bytecodes::_illegal;
2674             sprintf(name, "%s/%s/%s",
2675                     trap_reason_name(reason),
2676                     trap_action_name(action),
2677                     Bytecodes::is_defined(bc)? Bytecodes::name(bc): "other");
2678             juint r = counter >> LSB_BITS;
2679             tty->print_cr("  %40s: " UINT32_FORMAT " (%.1f%%)", name, r, (r * 100.0) / total);
2680             account -= r;
2681           }
2682         }
2683       }
2684     }
2685     if (account != 0) {
2686       PRINT_STAT_LINE("unaccounted", account);
2687     }
2688     #undef PRINT_STAT_LINE
2689     if (xtty != NULL)  xtty->tail("statistics");
2690   }
