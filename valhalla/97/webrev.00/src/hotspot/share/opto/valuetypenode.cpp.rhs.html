 34 // Clones the values type to handle control flow merges involving multiple value types.
 35 // The inputs are replaced by PhiNodes to represent the merged values for the given region.
 36 ValueTypeBaseNode* ValueTypeBaseNode::clone_with_phis(PhaseGVN* gvn, Node* region) {
 37   assert(!has_phi_inputs(region), "already cloned with phis");
 38   ValueTypeBaseNode* vt = clone()->as_ValueTypeBase();
 39 
 40   // Create a PhiNode for merging the oop values
 41   const Type* phi_type = Type::get_const_type(value_klass());
 42   PhiNode* oop = PhiNode::make(region, vt->get_oop(), phi_type);
 43   gvn->set_type(oop, phi_type);
 44   vt->set_oop(oop);
 45 
 46   // Create a PhiNode each for merging the field values
 47   for (uint i = 0; i < vt->field_count(); ++i) {
 48     ciType* type = vt->field_type(i);
 49     Node*  value = vt->field_value(i);
 50     if (value->is_ValueTypeBase()) {
 51       // Handle flattened value type fields recursively
 52       value = value->as_ValueTypeBase()->clone_with_phis(gvn, region);
 53     } else {
 54       phi_type = Type::get_const_type(type);
 55       value = PhiNode::make(region, value, phi_type);
 56       gvn->set_type(value, phi_type);
 57     }
 58     vt->set_field_value(i, value);
 59   }
 60   gvn->set_type(vt, vt->bottom_type());
 61   return vt;
 62 }

 64 // Checks if the inputs of the ValueBaseTypeNode were replaced by PhiNodes
 65 // for the given region (see ValueBaseTypeNode::clone_with_phis).
 66 bool ValueTypeBaseNode::has_phi_inputs(Node* region) {
 67   // Check oop input
 68   bool result = get_oop()->is_Phi() && get_oop()->as_Phi()->region() == region;
 69 #ifdef ASSERT
 70   if (result) {
 71     // Check all field value inputs for consistency
 72     for (uint i = Oop; i < field_count(); ++i) {
 73       Node* n = in(i);
 74       if (n->is_ValueTypeBase()) {
 75         assert(n->as_ValueTypeBase()->has_phi_inputs(region), "inconsistent phi inputs");
 76       } else {
 77         assert(n->is_Phi() && n->as_Phi()->region() == region, "inconsistent phi inputs");
 78       }
 79     }
 80   }
 81 #endif
 82   return result;
 83 }

 85 // Merges 'this' with 'other' by updating the input PhiNodes added by 'clone_with_phis'
 86 ValueTypeBaseNode* ValueTypeBaseNode::merge_with(PhaseGVN* gvn, const ValueTypeBaseNode* other, int pnum, bool transform) {
 87   // Merge oop inputs
 88   PhiNode* phi = get_oop()->as_Phi();
 89   phi->set_req(pnum, other->get_oop());
 90   if (transform) {
 91     set_oop(gvn->transform(phi));
 92     gvn->record_for_igvn(phi);
 93   }
 94   // Merge field values
 95   for (uint i = 0; i < field_count(); ++i) {
 96     Node* val1 =        field_value(i);
 97     Node* val2 = other->field_value(i);
 98     if (val1->is_ValueTypeBase()) {
 99       val1->as_ValueTypeBase()->merge_with(gvn, val2->as_ValueTypeBase(), pnum, transform);
100     } else {
101       assert(val1->is_Phi(), "must be a phi node");
102       assert(!val2->is_ValueType(), "inconsistent merge values");
103       val1->set_req(pnum, val2);
104     }
105     if (transform) {
106       set_field_value(i, gvn->transform(val1));
107       gvn->record_for_igvn(val1);
108     }
109   }
110   return this;
111 }

113 // Adds a new merge path to a valuetype node with phi inputs
114 void ValueTypeBaseNode::add_new_path(Node* region) {
115   assert(has_phi_inputs(region), "must have phi inputs");
116 
117   PhiNode* phi = get_oop()->as_Phi();
118   phi->add_req(NULL);
119   assert(phi->req() == region->req(), "must be same size as region");
120 
121   for (uint i = 0; i < field_count(); ++i) {
122     Node* val = field_value(i);
123     if (val->is_ValueType()) {
124       val->as_ValueType()->add_new_path(region);
125     } else {
126       val->as_Phi()->add_req(NULL);
127       assert(val->req() == region->req(), "must be same size as region");
128     }
129   }
130 }

137 // Get the value of the field at the given offset.
138 // If 'recursive' is true, flattened value type fields will be resolved recursively.
139 Node* ValueTypeBaseNode::field_value_by_offset(int offset, bool recursive) const {
140   // If the field at 'offset' belongs to a flattened value type field, 'index' refers to the
141   // corresponding ValueTypeNode input and 'sub_offset' is the offset in flattened value type.
142   int index = value_klass()->field_index_by_offset(offset);
143   int sub_offset = offset - field_offset(index);
144   Node* value = field_value(index);
145   assert(value != NULL, "field value not found");
146   if (recursive && value->is_ValueType()) {
147     ValueTypeNode* vt = value->as_ValueType();
148     if (field_is_flattened(index)) {
149       // Flattened value type field
150       sub_offset += vt->value_klass()->first_field_offset(); // Add header size
151       return vt->field_value_by_offset(sub_offset, recursive);
152     } else {
153       assert(sub_offset == 0, "should not have a sub offset");
154       return vt;
155     }
156   }
157   assert(!(recursive && value->is_ValueType()), "should not be a value type");
158   assert(sub_offset == 0, "offset mismatch");
159   return value;
160 }

202 int ValueTypeBaseNode::make_scalar_in_safepoint(PhaseIterGVN* igvn, Unique_Node_List& worklist, SafePointNode* sfpt) {
203   ciValueKlass* vk = value_klass();
204   uint nfields = vk->nof_nonstatic_fields();
205   JVMState* jvms = sfpt->jvms();
206   int start = jvms->debug_start();
207   int end   = jvms->debug_end();
208   // Replace safepoint edge by SafePointScalarObjectNode and add field values
209   assert(jvms != NULL, "missing JVMS");
210   uint first_ind = (sfpt->req() - jvms->scloff());
211   SafePointScalarObjectNode* sobj = new SafePointScalarObjectNode(value_ptr(),
212 #ifdef ASSERT
213                                                                   NULL,
214 #endif
215                                                                   first_ind, nfields);
216   sobj->init_req(0, igvn->C->root());
217   // Iterate over the value type fields in order of increasing
218   // offset and add the field values to the safepoint.
219   for (uint j = 0; j < nfields; ++j) {
220     int offset = vk->nonstatic_field_at(j)->offset();
221     Node* value = field_value_by_offset(offset, true /* include flattened value type fields */);
222     if (value->is_ValueType()) {
223       // Add value type field to the worklist to process later
224       worklist.push(value);
225     }
226     sfpt->add_req(value);
227   }
228   jvms->set_endoff(sfpt->req());
229   sobj = igvn->transform(sobj)->as_SafePointScalarObject();
230   igvn->rehash_node_delayed(sfpt);
231   return sfpt->replace_edges_in_range(this, sobj, start, end);
232 }

234 void ValueTypeBaseNode::make_scalar_in_safepoints(PhaseIterGVN* igvn) {
235   // Process all safepoint uses and scalarize value type
236   Unique_Node_List worklist;
237   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
238     SafePointNode* sfpt = fast_out(i)->isa_SafePoint();
239     if (sfpt != NULL && !sfpt->is_CallLeaf() && (!sfpt->is_Call() || sfpt->as_Call()->has_debug_use(this))) {
240       int nb = 0;
241       if (is_allocated(igvn) && get_oop()->is_Con()) {
242         // Value type is allocated with a constant oop, link it directly
243         nb = sfpt->replace_edges_in_range(this, get_oop(), sfpt->jvms()->debug_start(), sfpt->jvms()->debug_end());
244         igvn->rehash_node_delayed(sfpt);
245       } else {
246         nb = make_scalar_in_safepoint(igvn, worklist, sfpt);
247       }
248       --i; imax -= nb;
249     }
250   }
251   // Now scalarize non-flattened fields
252   for (uint i = 0; i < worklist.size(); ++i) {
253     Node* vt = worklist.at(i);
254     vt->as_ValueType()->make_scalar_in_safepoints(igvn);
255   }
256   igvn->record_for_igvn(this);
257 }

259 const TypePtr* ValueTypeBaseNode::field_adr_type(Node* base, int offset, ciInstanceKlass* holder, DecoratorSet decorators, PhaseGVN& gvn) const {
260   const TypeAryPtr* ary_type = gvn.type(base)->isa_aryptr();
261   const TypePtr* adr_type = NULL;
262   bool is_array = ary_type != NULL;
263   if ((decorators & C2_MISMATCHED) != 0) {
264     adr_type = TypeRawPtr::BOTTOM;
265   } else if (is_array) {
266     // In the case of a flattened value type array, each field has its own slice
267     adr_type = ary_type->with_field_offset(offset)->add_offset(Type::OffsetBot);
268   } else {
269     ciField* field = holder->get_field_by_offset(offset, false);
270     assert(field != NULL, "field not found");
271     adr_type = gvn.C->alias_type(field)->adr_type();
272   }
273   return adr_type;
274 }

276 void ValueTypeBaseNode::load(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) {
277   // Initialize the value type by loading its field values from
278   // memory and adding the values as input edges to the node.
279   for (uint i = 0; i < field_count(); ++i) {
280     int offset = holder_offset + field_offset(i);
281     Node* value = NULL;
282     ciType* ft = field_type(i);
283     bool is_flattenable = field_is_flattenable(i);
284     if (field_is_flattened(i)) {
285       // Recursively load the flattened value type field
286       value = ValueTypeNode::make_from_flattened(kit, ft->as_value_klass(), base, ptr, holder, offset, decorators);
287     } else {
288       const TypeOopPtr* oop_ptr = kit->gvn().type(base)->isa_oopptr();
289       bool is_array = (oop_ptr->isa_aryptr() != NULL);
290       if (base->is_Con() && !is_array) {
291         // If the oop to the value type is constant (static final field), we can
292         // also treat the fields as constants because the value type is immutable.
293         ciObject* constant_oop = oop_ptr->const_oop();
294         ciField* field = holder->get_field_by_offset(offset, false);
295         assert(field != NULL, "field not found");
296         ciConstant constant = constant_oop->as_instance()->field_value(field);
297         const Type* con_type = Type::make_from_constant(constant, /*require_const=*/ true);
298         assert(con_type != NULL, "type not found");
299         value = kit->gvn().transform(kit->makecon(con_type));
300         // Check type of constant which might be more precise
301         if (con_type->is_valuetypeptr() && !con_type->is_zero_type()) {
302           // Null-free, treat as flattenable
303           ft = con_type->value_klass();
304           is_flattenable = true;
305         }
306       } else {
307         // Load field value from memory
308         const TypePtr* adr_type = field_adr_type(base, offset, holder, decorators, kit->gvn());
309         Node* adr = kit->basic_plus_adr(base, ptr, offset);
310         BasicType bt = type2field[ft->basic_type()];
311         assert(is_java_primitive(bt) || adr->bottom_type()->is_ptr_to_narrowoop() == UseCompressedOops, "inconsistent");
312         const Type* val_type = Type::get_const_type(ft);
313         if (is_array) {
314           decorators |= IS_ARRAY;
315         }
316         value = kit->access_load_at(base, adr, adr_type, val_type, bt, decorators);
317       }
318       if (is_flattenable) {
319         // Loading a non-flattened but flattenable value type from memory
320         if (ft->as_value_klass()->is_scalarizable()) {
321           value = ValueTypeNode::make_from_oop(kit, value, ft->as_value_klass());
322         } else {
323           value = kit->null2default(value, ft->as_value_klass());
324         }
325       }
326     }
327     set_field_value(i, value);
328   }
329 }

331 void ValueTypeBaseNode::store_flattened(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) const {
332   // The value type is embedded into the object without an oop header. Subtract the
333   // offset of the first field to account for the missing header when storing the values.
334   if (holder == NULL) {
335     holder = value_klass();
336   }
337   holder_offset -= value_klass()->first_field_offset();
338   store(kit, base, ptr, holder, holder_offset, decorators);
339 }

341 void ValueTypeBaseNode::store(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) const {
342   // Write field values to memory
343   for (uint i = 0; i < field_count(); ++i) {
344     int offset = holder_offset + field_offset(i);
345     Node* value = field_value(i);
346     ciType* ft = field_type(i);
347     if (field_is_flattened(i)) {
348       // Recursively store the flattened value type field
349       if (!value->is_ValueType()) {
350         assert(!kit->gvn().type(value)->maybe_null(), "should never be null");
351         value = ValueTypeNode::make_from_oop(kit, value, ft->as_value_klass());
352       }
353       value->as_ValueType()->store_flattened(kit, base, ptr, holder, offset, decorators);
354     } else {
355       // Store field value to memory
356       const TypePtr* adr_type = field_adr_type(base, offset, holder, decorators, kit->gvn());
357       Node* adr = kit->basic_plus_adr(base, ptr, offset);
358       BasicType bt = type2field[ft->basic_type()];
359       assert(is_java_primitive(bt) || adr->bottom_type()->is_ptr_to_narrowoop() == UseCompressedOops, "inconsistent");
360       const Type* val_type = Type::get_const_type(ft);
361       const TypeAryPtr* ary_type = kit->gvn().type(base)->isa_aryptr();
362       if (ary_type != NULL) {
363         decorators |= IS_ARRAY;
364       }
365       kit->access_store_at(base, adr, adr_type, value, val_type, bt, decorators);
366     }
367   }
368 }

370 ValueTypePtrNode* ValueTypeBaseNode::buffer(GraphKit* kit, bool safe_for_replace) {
371   assert(is_ValueType(), "sanity");
372   // Check if value type is already allocated
373   Node* null_ctl = kit->top();
374   Node* not_null_oop = kit->null_check_oop(get_oop(), &null_ctl);
375   if (null_ctl->is_top()) {
376     // Value type is allocated
377     return kit->gvn().transform(new ValueTypePtrNode(this))->as_ValueTypePtr();
378   }
379   assert(!is_allocated(&kit->gvn()), "should not be allocated");
380   RegionNode* region = new RegionNode(3);
381 
382   // Oop is non-NULL, use it
383   region->init_req(1, kit->control());
384   PhiNode* oop = PhiNode::make(region, not_null_oop, value_ptr());
385   PhiNode* io  = PhiNode::make(region, kit->i_o(), Type::ABIO);
386   PhiNode* mem = PhiNode::make(region, kit->merged_memory(), Type::MEMORY, TypePtr::BOTTOM);
387 
388   int bci = kit->bci();
389   bool reexecute = kit->jvms()->should_reexecute();
390   {
391     // Oop is NULL, allocate and initialize buffer
392     PreserveJVMState pjvms(kit);
393     // Propagate re-execution state and bci
394     kit->set_bci(bci);
395     kit->jvms()->set_bci(bci);
396     kit->jvms()->set_should_reexecute(reexecute);
397     kit->set_control(null_ctl);
398     kit->kill_dead_locals();
399     ciValueKlass* vk = value_klass();
400     Node* klass_node = kit->makecon(TypeKlassPtr::make(vk));
401     Node* alloc_oop  = kit->new_instance(klass_node, NULL, NULL, /* deoptimize_on_exception */ true, this);
402     store(kit, alloc_oop, alloc_oop, vk, 0);
403 
404     // Do not let stores that initialize this buffer be reordered with a subsequent
405     // store that would make this buffer accessible by other threads.
406     AllocateNode* alloc = AllocateNode::Ideal_allocation(alloc_oop, &kit->gvn());
407     assert(alloc != NULL, "must have an allocation node");
408     kit->insert_mem_bar(Op_MemBarStoreStore, alloc->proj_out_or_null(AllocateNode::RawAddress));
409 
410     region->init_req(2, kit->control());
411     oop   ->init_req(2, alloc_oop);
412     io    ->init_req(2, kit->i_o());
413     mem   ->init_req(2, kit->merged_memory());
414   }
415 
416   // Update GraphKit
417   kit->set_control(kit->gvn().transform(region));
418   kit->set_i_o(kit->gvn().transform(io));
419   kit->set_all_memory(kit->gvn().transform(mem));
420   kit->record_for_igvn(region);
421   kit->record_for_igvn(oop);
422   kit->record_for_igvn(io);
423   kit->record_for_igvn(mem);
424 
425   // Use cloned ValueTypeNode to propagate oop from now on
426   Node* res_oop = kit->gvn().transform(oop);
427   ValueTypeBaseNode* vt = clone()->as_ValueTypeBase();
428   vt->set_oop(res_oop);
429   vt = kit->gvn().transform(vt)->as_ValueTypeBase();
430   if (safe_for_replace) {
431     kit->replace_in_map(this, vt);
432   }
433   // ValueTypeNode::remove_redundant_allocations piggybacks on split if.
434   // Make sure it gets a chance to remove this allocation.
435   kit->C->set_has_split_ifs(true);
436   assert(vt->is_allocated(&kit->gvn()), "must be allocated");
437   return kit->gvn().transform(new ValueTypePtrNode(vt))->as_ValueTypePtr();
438 }

440 bool ValueTypeBaseNode::is_allocated(PhaseGVN* phase) const {
441   Node* oop = get_oop();
442   const Type* oop_type = (phase != NULL) ? phase->type(oop) : oop->bottom_type();
443   return !oop_type->maybe_null();
444 }

446 // When a call returns multiple values, it has several result
447 // projections, one per field. Replacing the result of the call by a
448 // value type node (after late inlining) requires that for each result
449 // projection, we find the corresponding value type field.
450 void ValueTypeBaseNode::replace_call_results(GraphKit* kit, Node* call, Compile* C) {
451   ciValueKlass* vk = value_klass();
452   for (DUIterator_Fast imax, i = call->fast_outs(imax); i < imax; i++) {
453     ProjNode* pn = call->fast_out(i)->as_Proj();
454     uint con = pn->_con;
455     if (con >= TypeFunc::Parms+1) {
456       uint field_nb = con - (TypeFunc::Parms+1);
457       int extra = 0;
458       for (uint j = 0; j < field_nb - extra; j++) {
459         ciField* f = vk->nonstatic_field_at(j);
460         BasicType bt = f->type()->basic_type();
461         if (bt == T_LONG || bt == T_DOUBLE) {
462           extra++;
463         }
464       }
465       ciField* f = vk->nonstatic_field_at(field_nb - extra);
466       Node* field = field_value_by_offset(f->offset(), true);
467       if (field->is_ValueType()) {
468         assert(field->as_ValueType()->is_allocated(&kit->gvn()), "must be allocated");
469         field = field->as_ValueType()->get_oop();
470       }
471       C->gvn_replace_by(pn, field);
472       C->initial_gvn()->hash_delete(pn);
473       pn->set_req(0, C->top());
474       --i; --imax;
475     }
476   }
477 }

479 Node* ValueTypeBaseNode::allocate_fields(GraphKit* kit) {
480   ValueTypeBaseNode* vt = clone()->as_ValueTypeBase();
481   for (uint i = 0; i < field_count(); i++) {
482      ValueTypeNode* value = field_value(i)->isa_ValueType();
483      if (field_is_flattened(i)) {
484        // Flattened value type field
485        vt->set_field_value(i, value->allocate_fields(kit));
486      } else if (value != NULL) {
487        // Non-flattened value type field
488        vt->set_field_value(i, value->buffer(kit));
489      }
490   }
491   vt = kit->gvn().transform(vt)->as_ValueTypeBase();
492   kit->replace_in_map(this, vt);
493   return vt;
494 }

496 ValueTypeNode* ValueTypeNode::make_uninitialized(PhaseGVN& gvn, ciValueKlass* vk) {
497   // Create a new ValueTypeNode with uninitialized values and NULL oop
498   Node* oop = vk->is_empty() ? default_oop(gvn, vk) : gvn.zerocon(T_VALUETYPE);
499   return new ValueTypeNode(vk, oop);

502 Node* ValueTypeNode::default_oop(PhaseGVN& gvn, ciValueKlass* vk) {
503   // Returns the constant oop of the default value type allocation
504   return gvn.makecon(TypeInstPtr::make(vk->default_value_instance()));
505 }

507 ValueTypeNode* ValueTypeNode::make_default(PhaseGVN& gvn, ciValueKlass* vk) {
508   // Create a new ValueTypeNode with default values
509   ValueTypeNode* vt = new ValueTypeNode(vk, default_oop(gvn, vk));
510   for (uint i = 0; i < vt->field_count(); ++i) {
511     ciType* field_type = vt->field_type(i);
512     Node* value = NULL;
513     if (field_type->is_valuetype() && vt->field_is_flattenable(i)) {
514       ciValueKlass* field_klass = field_type->as_value_klass();
515       if (field_klass->is_scalarizable() || vt->field_is_flattened(i)) {
516         value = ValueTypeNode::make_default(gvn, field_klass);
517       } else {
518         value = default_oop(gvn, field_klass);
519       }
520     } else {
521       value = gvn.zerocon(field_type->basic_type());
522     }
523     vt->set_field_value(i, value);
524   }
525   vt = gvn.transform(vt)->as_ValueType();
526   assert(vt->is_default(gvn), "must be the default value type");
527   return vt;
528 }
529 
530 bool ValueTypeNode::is_default(PhaseGVN& gvn) const {
531   for (uint i = 0; i < field_count(); ++i) {
532     Node* value = field_value(i);
533     if (!gvn.type(value)->is_zero_type() &&
534         !(value->is_ValueType() && value->as_ValueType()->is_default(gvn)) &&
535         !(field_type(i)->is_valuetype() && value == default_oop(gvn, field_type(i)->as_value_klass()))) {
536       return false;
537     }
538   }
539   return true;
540 }

542 ValueTypeNode* ValueTypeNode::make_from_oop(GraphKit* kit, Node* oop, ciValueKlass* vk) {
543   PhaseGVN& gvn = kit->gvn();
544 
545   // Create and initialize a ValueTypeNode by loading all field
546   // values from a heap-allocated version and also save the oop.
547   ValueTypeNode* vt = new ValueTypeNode(vk, oop);
548 
549   if (oop->isa_ValueTypePtr()) {
550     // Can happen with late inlining
551     ValueTypePtrNode* vtptr = oop->as_ValueTypePtr();
552     vt->set_oop(vtptr->get_oop());
553     for (uint i = Oop+1; i < vtptr->req(); ++i) {
554       vt->init_req(i, vtptr->in(i));
555     }
556   } else if (gvn.type(oop)->maybe_null()) {
557     // Add a null check because the oop may be null
558     Node* null_ctl = kit->top();
559     Node* not_null_oop = kit->null_check_oop(oop, &null_ctl);
560     if (kit->stopped()) {
561       // Constant null
562       kit->set_control(null_ctl);
563       return make_default(gvn, vk);
564     }
565     vt->set_oop(not_null_oop);
566     vt->load(kit, not_null_oop, not_null_oop, vk, /* holder_offset */ 0);
567 
568     if (null_ctl != kit->top()) {
569       // Return default value type if oop is null
570       ValueTypeNode* def = make_default(gvn, vk);
571       Node* region = new RegionNode(3);
572       region->init_req(1, kit->control());
573       region->init_req(2, null_ctl);
574 
575       vt = vt->clone_with_phis(&gvn, region)->as_ValueType();
576       vt->merge_with(&gvn, def, 2, true);
577       kit->set_control(gvn.transform(region));
578     }
579   } else {
580     // Oop can never be null
581     Node* init_ctl = kit->control();
582     vt->load(kit, oop, oop, vk, /* holder_offset */ 0);
583     assert(init_ctl != kit->control() || !gvn.type(oop)->is_valuetypeptr() || oop->is_Con() || oop->Opcode() == Op_ValueTypePtr ||
584            AllocateNode::Ideal_allocation(oop, &gvn) != NULL || vt->is_loaded(&gvn) == oop, "value type should be loaded");
585   }
586 
587   assert(vt->is_allocated(&gvn), "value type should be allocated");
588   return gvn.transform(vt)->as_ValueType();
589 }

591 // GraphKit wrapper for the 'make_from_flattened' method
592 ValueTypeNode* ValueTypeNode::make_from_flattened(GraphKit* kit, ciValueKlass* vk, Node* obj, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) {
593   // Create and initialize a ValueTypeNode by loading all field values from
594   // a flattened value type field at 'holder_offset' or from a value type array.
595   ValueTypeNode* vt = make_uninitialized(kit->gvn(), vk);
596   // The value type is flattened into the object without an oop header. Subtract the
597   // offset of the first field to account for the missing header when loading the values.
598   holder_offset -= vk->first_field_offset();
599   vt->load(kit, obj, ptr, holder, holder_offset, decorators);
600   assert(vt->is_loaded(&kit->gvn()) != obj, "holder oop should not be used as flattened value type oop");
601   return kit->gvn().transform(vt)->as_ValueType();
602 }

610 ValueTypeNode* ValueTypeNode::make_larval(GraphKit* kit, bool allocate) const {
611   ciValueKlass* vk = value_klass();
612   ValueTypeNode* res = clone()->as_ValueType();
613   if (allocate) {
614     // Re-execute if buffering triggers deoptimization
615     PreserveReexecuteState preexecs(kit);
616     kit->jvms()->set_should_reexecute(true);
617     Node* klass_node = kit->makecon(TypeKlassPtr::make(vk));
618     Node* alloc_oop  = kit->new_instance(klass_node, NULL, NULL, true);
619     AllocateNode* alloc = AllocateNode::Ideal_allocation(alloc_oop, &kit->gvn());
620     alloc->_larval = true;
621 
622     store(kit, alloc_oop, alloc_oop, vk, 0);
623     res->set_oop(alloc_oop);
624   }
625   res->set_type(TypeValueType::make(vk, true));
626   res = kit->gvn().transform(res)->as_ValueType();
627   assert(!allocate || res->is_allocated(&kit->gvn()), "must be allocated");
628   return res;
629 }

631 ValueTypeNode* ValueTypeNode::finish_larval(GraphKit* kit) const {
632   Node* obj = get_oop();
633   Node* mark_addr = kit->basic_plus_adr(obj, oopDesc::mark_offset_in_bytes());
634   Node* mark = kit->make_load(NULL, mark_addr, TypeX_X, TypeX_X->basic_type(), MemNode::unordered);
635   mark = kit->gvn().transform(new AndXNode(mark, kit->MakeConX(~markWord::larval_mask_in_place)));
636   kit->store_to_memory(kit->control(), mark_addr, mark, TypeX_X->basic_type(), kit->gvn().type(mark_addr)->is_ptr(), MemNode::unordered);
637 
638   // Do not let stores that initialize this buffer be reordered with a subsequent
639   // store that would make this buffer accessible by other threads.
640   AllocateNode* alloc = AllocateNode::Ideal_allocation(obj, &kit->gvn());
641   assert(alloc != NULL, "must have an allocation node");
642   kit->insert_mem_bar(Op_MemBarStoreStore, alloc->proj_out_or_null(AllocateNode::RawAddress));
643 
644   ciValueKlass* vk = value_klass();
645   ValueTypeNode* res = clone()->as_ValueType();
646   res->set_type(TypeValueType::make(vk, false));
647   res = kit->gvn().transform(res)->as_ValueType();
648   return res;
649 }

651 Node* ValueTypeNode::is_loaded(PhaseGVN* phase, ciValueKlass* vk, Node* base, int holder_offset) {
652   if (vk == NULL) {
653     vk = value_klass();
654   }
655   if (field_count() == 0) {
656     assert(is_allocated(phase), "must be allocated");
657     return get_oop();
658   }
659   for (uint i = 0; i < field_count(); ++i) {
660     int offset = holder_offset + field_offset(i);
661     Node* value = field_value(i);
662     if (value->is_ValueType()) {
663       ValueTypeNode* vt = value->as_ValueType();
664       if (field_is_flattened(i)) {
665         // Check value type field load recursively
666         base = vt->is_loaded(phase, vk, base, offset - vt->value_klass()->first_field_offset());
667         if (base == NULL) {
668           return NULL;
669         }
670         continue;
671       } else {
672         value = vt->get_oop();
673         if (value->Opcode() == Op_CastPP) {
674           // Skip CastPP
675           value = value->in(1);
676         }
677       }
678     }
679     if (value->isa_DecodeN()) {
680       // Skip DecodeN
681       value = value->in(1);
682     }
683     if (value->isa_Load()) {
684       // Check if base and offset of field load matches value type layout
685       intptr_t loffset = 0;
686       Node* lbase = AddPNode::Ideal_base_and_offset(value->in(MemNode::Address), phase, loffset);
687       if (lbase == NULL || (lbase != base && base != NULL) || loffset != offset) {
688         return NULL;
689       } else if (base == NULL) {
690         // Set base and check if pointer type matches
691         base = lbase;
692         const TypeInstPtr* vtptr = phase->type(base)->isa_instptr();
693         if (vtptr == NULL || !vtptr->klass()->equals(vk)) {
694           return NULL;
695         }
696       }
697     } else {
698       return NULL;
699     }
700   }
701   return base;
702 }

705   const TypeKlassPtr* tk = TypeKlassPtr::make(vk);

711 void ValueTypeNode::pass_fields(GraphKit* kit, Node* n, ExtendedSignature& sig, uint& base_input, int base_offset) {
712   for (uint i = 0; i < field_count(); i++) {
713     int sig_offset = (*sig)._offset;
714     uint idx = field_index(sig_offset - base_offset);
715     Node* arg = field_value(idx);
716 
717     if (field_is_flattened(idx)) {
718       // Flattened value type field
719       ValueTypeNode* vt = arg->as_ValueType();
720       vt->pass_fields(kit, n, sig, base_input, sig_offset - vt->value_klass()->first_field_offset());
721     } else {
722       if (arg->is_ValueType()) {
723         // Non-flattened value type field
724         ValueTypeNode* vt = arg->as_ValueType();
725         assert(n->Opcode() != Op_Return || vt->is_allocated(&kit->gvn()), "value type field should be allocated on return");
726         arg = vt->buffer(kit);
727       }
728       // Initialize call/return arguments
729       BasicType bt = field_type(i)->basic_type();
730       n->init_req(base_input++, arg);
731       if (type2size[bt] == 2) {
732         n->init_req(base_input++, kit->top());
733       }
734       // Skip reserved arguments
735       while (SigEntry::next_is_reserved(sig, bt)) {
736         n->init_req(base_input++, kit->top());
737         if (type2size[bt] == 2) {
738           n->init_req(base_input++, kit->top());
739         }
740       }
741     }
742   }
743 }

745 void ValueTypeNode::initialize_fields(GraphKit* kit, MultiNode* multi, ExtendedSignature& sig, uint& base_input, int base_offset, bool in) {
746   PhaseGVN& gvn = kit->gvn();
747   for (uint i = 0; i < field_count(); i++) {
748     int sig_offset = (*sig)._offset;
749     uint idx = field_index(sig_offset - base_offset);
750     ciType* type = field_type(idx);
751 
752     Node* parm = NULL;
753     if (field_is_flattened(idx)) {
754       // Flattened value type field
755       ValueTypeNode* vt = ValueTypeNode::make_uninitialized(gvn, type->as_value_klass());
756       vt->initialize_fields(kit, multi, sig, base_input, sig_offset - type->as_value_klass()->first_field_offset(), in);
757       parm = gvn.transform(vt);
758     } else {
759       if (multi->is_Start()) {
760         assert(in, "return from start?");
761         parm = gvn.transform(new ParmNode(multi->as_Start(), base_input));
762       } else if (in) {
763         parm = multi->as_Call()->in(base_input);
764       } else {
765         parm = gvn.transform(new ProjNode(multi->as_Call(), base_input));
766       }
767       if (field_is_flattenable(idx)) {
768         // Non-flattened but flattenable value type
769         if (type->as_value_klass()->is_scalarizable()) {
770           parm = ValueTypeNode::make_from_oop(kit, parm, type->as_value_klass());
771         } else {
772           parm = kit->null2default(parm, type->as_value_klass());
773         }
774       }
775       base_input += type2size[type->basic_type()];
776       // Skip reserved arguments
777       BasicType bt = type->basic_type();
778       while (SigEntry::next_is_reserved(sig, bt)) {
779         base_input += type2size[bt];
780       }
781     }
782     assert(parm != NULL, "should never be null");
783     set_field_value(idx, parm);
784     gvn.record_for_igvn(parm);
785   }
786 }

788 Node* ValueTypeNode::Ideal(PhaseGVN* phase, bool can_reshape) {
789   Node* oop = get_oop();
790   if (is_default(*phase) && (!oop->is_Con() || phase->type(oop)->is_zero_type())) {
791     // Use the pre-allocated oop for default value types
792     set_oop(default_oop(*phase, value_klass()));
793     return this;
794   } else if (oop->isa_ValueTypePtr()) {
795     // Can happen with late inlining
796     ValueTypePtrNode* vtptr = oop->as_ValueTypePtr();
797     set_oop(vtptr->get_oop());
798     for (uint i = Oop+1; i < vtptr->req(); ++i) {
799       set_req(i, vtptr->in(i));
800     }
801     return this;
802   }
803 
804   if (!is_allocated(phase)) {
805     // Save base oop if fields are loaded from memory and the value
806     // type is not buffered (in this case we should not use the oop).
807     Node* base = is_loaded(phase);
808     if (base != NULL) {
809       set_oop(base);
810       assert(is_allocated(phase), "should now be allocated");
811       return this;
812     }
813   }
814 
815   if (can_reshape) {
816     PhaseIterGVN* igvn = phase->is_IterGVN();
817 
818     if (is_default(*phase)) {
819       // Search for users of the default value type
820       for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
821         Node* user = fast_out(i);
822         AllocateNode* alloc = user->isa_Allocate();
823         if (alloc != NULL && alloc->in(AllocateNode::ValueNode) == this) {
824           // Found an allocation of the default value type.
825           Node* res = alloc->result_cast();
826           if (res != NULL) {
827             // If the code in StoreNode::Identity() that removes useless stores was not yet
828             // executed or ReduceFieldZeroing is disabled, there can still be initializing
829             // stores (only zero-type or default value stores, because value types are immutable).
830             for (DUIterator_Fast jmax, j = res->fast_outs(jmax); j < jmax; j++) {
831               AddPNode* addp = res->fast_out(j)->isa_AddP();
832               if (addp != NULL) {
833                 for (DUIterator_Fast kmax, k = addp->fast_outs(kmax); k < kmax; k++) {
834                   StoreNode* store = addp->fast_out(k)->isa_Store();
835                   if (store != NULL && store->outcnt() != 0) {
836                     // Remove the useless store
837                     igvn->replace_in_uses(store, store->in(MemNode::Memory));
838                   }
839                 }
840               }
841             }
842             // Replace allocation by pre-allocated oop
843             igvn->replace_node(res, default_oop(*phase, value_klass()));
844           }
845           // Unlink AllocateNode
846           igvn->replace_input_of(alloc, AllocateNode::ValueNode, igvn->C->top());
847           --i; --imax;
848         } else if (user->is_ValueType()) {
849           // Add value type user to worklist to give it a chance to get optimized as well
850           igvn->_worklist.push(user);
851         }
852       }
853     }
854 
855     if (is_allocated(igvn)) {
856       // Value type is heap allocated, search for safepoint uses
857       for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
858         Node* out = fast_out(i);
859         if (out->is_SafePoint()) {
860           // Let SafePointNode::Ideal() take care of re-wiring the
861           // safepoint to the oop input instead of the value type node.
862           igvn->rehash_node_delayed(out);
863         }
864       }
865     }
866   }
867   return NULL;
868 }

870 // Search for multiple allocations of this value type
871 // and try to replace them by dominating allocations.
872 // Then unlink the value type node and remove it.
873 void ValueTypeNode::remove_redundant_allocations(PhaseIterGVN* igvn, PhaseIdealLoop* phase) {
874   // Search for allocations of this value type
875   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
876     AllocateNode* alloc = fast_out(i)->isa_Allocate();
877     if (alloc != NULL && alloc->in(AllocateNode::ValueNode) == this) {
878       assert(!is_default(*igvn), "default value type allocation");
879       Node* res = alloc->result_cast();
880       if (res == NULL || !res->is_CheckCastPP()) {
881         break; // No unique CheckCastPP
882       }
883       Node* res_dom = res;
884       if (is_allocated(igvn)) {
885         // The value type is already allocated but still connected to an AllocateNode.
886         // This can happen with late inlining when we first allocate a value type argument
887         // but later decide to inline the call with the callee code also allocating.
888         res_dom = get_oop();
889       } else {
890         // Search for a dominating allocation of the same value type
891         for (DUIterator_Fast jmax, j = fast_outs(jmax); j < jmax; j++) {
892           AllocateNode* alloc_other = fast_out(j)->isa_Allocate();
893           if (alloc_other != NULL && alloc_other->in(AllocateNode::ValueNode) == this) {
894             Node* res_other = alloc_other->result_cast();
895             if (res_other != NULL && res_other->is_CheckCastPP() && res_other != res_dom &&
896                 phase->is_dominator(res_other->in(0), res_dom->in(0))) {
897               res_dom = res_other;
898             }
899           }
900         }
901       }
902       if (res_dom != res) {
903         // Move users to dominating allocation
904         igvn->replace_node(res, res_dom);
905         // The result of the dominated allocation is now unused and will be removed
906         // later in PhaseMacroExpand::eliminate_allocate_node to not confuse loop opts.
907         igvn->record_for_igvn(alloc);
908       }
909     }
910   }
911 
912   // Process users
913   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
914     Node* out = fast_out(i);
915     if (out->is_ValueType()) {
916       // Unlink and recursively process value type users
917       igvn->hash_delete(out);
918       int nb = out->replace_edge(this, igvn->C->top());
919       out->as_ValueType()->remove_redundant_allocations(igvn, phase);
920       --i; imax -= nb;
921     } else if (out->isa_Allocate() != NULL) {
922       // Unlink AllocateNode
923       assert(out->in(AllocateNode::ValueNode) == this, "should be linked");
924       igvn->replace_input_of(out, AllocateNode::ValueNode, igvn->C->top());
925       --i; --imax;
926     } else {
927 #ifdef ASSERT
928       // The value type should not have any other users at this time
929       out->dump();
930       assert(false, "unexpected user of value type");
931 #endif
932     }
933   }
934   igvn->remove_dead_node(this);
935 }
