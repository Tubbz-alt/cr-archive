 233             }
 234         }

 330     /** Report duplicate declaration error.
 331      */
 332     void duplicateError(DiagnosticPosition pos, Symbol sym) {
 333         if (!sym.type.isErroneous()) {
 334             Symbol location = sym.location();
 335             if (location.kind == MTH &&
 336                     ((MethodSymbol)location).isStaticOrInstanceInit()) {
 337                 log.error(pos,
 338                           Errors.AlreadyDefinedInClinit(kindName(sym),
 339                                                         sym,
 340                                                         kindName(sym.location()),
 341                                                         kindName(sym.location().enclClass()),
 342                                                         sym.location().enclClass()));
 343             } else {
 344                 /* dont error if this is a duplicated parameter of a generated canonical constructor
 345                  * as we should have issued an error for the duplicated fields
 346                  */
 347                 if (location.kind != MTH ||
 348                         ((sym.owner.flags_field & GENERATEDCONSTR) == 0) ||
 349                         ((sym.owner.flags_field & RECORD) == 0)) {
 350                     log.error(pos,
 351                             Errors.AlreadyDefined(kindName(sym),
 352                                     sym,
 353                                     kindName(sym.location()),
 354                                     sym.location()));
 355                 }
 356             }
 357         }
 358     }

 372     /** Check that variable does not hide variable with same name in
 373      *  immediately enclosing local scope.
 374      *  @param pos           Position for error reporting.
 375      *  @param v             The symbol.
 376      *  @param s             The scope.
 377      */
 378     void checkTransparentVar(DiagnosticPosition pos, VarSymbol v, Scope s) {
 379         for (Symbol sym : s.getSymbolsByName(v.name)) {
 380             if (sym.owner != v.owner) break;
 381             if (sym.kind == VAR &&
 382                 sym.owner.kind.matches(KindSelector.VAL_MTH) &&
 383                 v.name != names.error) {
 384                 duplicateError(pos, sym);
 385                 return;
 386             }
 387         }
 388     }
 389 
 390     /** Check that a class or interface does not hide a class or
 391      *  interface with same name in immediately enclosing local scope.
 392      *  @param pos           Position for error reporting.
 393      *  @param c             The symbol.
 394      *  @param s             The scope.
 395      */
 396     void checkTransparentClass(DiagnosticPosition pos, ClassSymbol c, Scope s) {
 397         for (Symbol sym : s.getSymbolsByName(c.name)) {
 398             if (sym.owner != c.owner) break;
 399             if (sym.kind == TYP && !sym.type.hasTag(TYPEVAR) &&
 400                 sym.owner.kind.matches(KindSelector.VAL_MTH) &&
 401                 c.name != names.error) {
 402                 duplicateError(pos, sym);
 403                 return;
 404             }
 405         }
 406     }

 408     /** Check that class does not have the same name as one of
 409      *  its enclosing classes, or as a class defined in its enclosing scope.
 410      *  return true if class is unique in its enclosing scope.
 411      *  @param pos           Position for error reporting.
 412      *  @param name          The class name.
 413      *  @param s             The enclosing scope.
 414      */
 415     boolean checkUniqueClassName(DiagnosticPosition pos, Name name, Scope s) {
 416         for (Symbol sym : s.getSymbolsByName(name, NON_RECURSIVE)) {
 417             if (sym.kind == TYP && sym.name != names.error) {
 418                 duplicateError(pos, sym);
 419                 return false;
 420             }
 421         }
 422         for (Symbol sym = s.owner; sym != null; sym = sym.owner) {
 423             if (sym.kind == TYP && sym.name == name && sym.name != names.error) {
 424                 duplicateError(pos, sym);
 425                 return true;
 426             }
 427         }
 428         return true;
 429     }

 438     /** Return name of local class.
 439      *  This is of the form   {@code <enclClass> $ n <classname> }
 440      *  where
 441      *    enclClass is the flat name of the enclosing class,
 442      *    classname is the simple name of the local class
 443      */
 444     public Name localClassName(ClassSymbol c) {
 445         Name enclFlatname = c.owner.enclClass().flatname;
 446         String enclFlatnameStr = enclFlatname.toString();
 447         Pair<Name, Name> key = new Pair<>(enclFlatname, c.name);
 448         Integer index = localClassNameIndexes.get(key);
 449         for (int i = (index == null) ? 1 : index; ; i++) {
 450             Name flatname = names.fromString(enclFlatnameStr
 451                     + syntheticNameChar + i + c.name);
 452             if (getCompiled(c.packge().modle, flatname) == null) {
 453                 localClassNameIndexes.put(key, i + 1);
 454                 return flatname;
 455             }
 456         }
 457     }
 458 
 459     public void clearLocalClassNameIndexes(ClassSymbol c) {
 460         if (c.owner != null && c.owner.kind != NIL) {
 461             localClassNameIndexes.remove(new Pair<>(
 462                     c.owner.enclClass().flatname, c.name));
 463         }
 464     }

 595     Type checkType(final DiagnosticPosition pos, final Type found, final Type req, final CheckContext checkContext) {
 596         final InferenceContext inferenceContext = checkContext.inferenceContext();
 597         if (inferenceContext.free(req) || inferenceContext.free(found)) {
 598             inferenceContext.addFreeTypeListener(List.of(req, found),
 599                     solvedContext -> checkType(pos, solvedContext.asInstType(found), solvedContext.asInstType(req), checkContext));
 600         } else {
 601             if (found.hasTag(CLASS)) {
 602                 if (inferenceContext != infer.emptyContext)
 603                     checkParameterizationWithValues(pos, found);
 604             }
 605         }
 606         if (req.hasTag(ERROR))
 607             return req;
 608         if (req.hasTag(NONE))
 609             return found;
 610         if (checkContext.compatible(found, req, checkContext.checkWarner(pos, found, req))) {
 611             if (found.hasTag(BOT) && types.isValueBased(req)) {
 612                 log.warning(pos, Warnings.SuspiciousMixOfNullWithValueBasedClass(req));
 613             }
 614             return found;
 615         } else {
 616             if (found.isNumeric() && req.isNumeric()) {
 617                 checkContext.report(pos, diags.fragment(Fragments.PossibleLossOfPrecision(found, req)));
 618                 return types.createErrorType(found);
 619             }
 620             checkContext.report(pos, diags.fragment(Fragments.InconvertibleTypes(found, req)));
 621             return types.createErrorType(found);
 622         }
 623     }

 625     /** Check that a given type can be cast to a given target type.
 626      *  Return the result of the cast.

 628      *  @param found      The type that is being cast.
 629      *  @param req        The target type of the cast.

 634     Type checkCastable(DiagnosticPosition pos, Type found, Type req, CheckContext checkContext) {
 635         if (types.isCastable(found, req, castWarner(pos, found, req))) {
 636             if (types.isValueBased(req)) {
 637                 if (found.hasTag(BOT)) {
 638                     log.warning(pos, Warnings.SuspiciousMixOfNullWithValueBasedClass(req));
 639                 } else if (!types.isValueBased(found)) {
 640                     log.warning(pos, Warnings.PotentialNullPollution(found));
 641                 }

 746         boolean indirectSuper = false;
 747         for(Type st = types.supertype(c.type); st != Type.noType; indirectSuper = true, st = types.supertype(st)) {
 748             if (st == null || st.tsym == null || st.tsym.kind == ERR)

1172                 assertConvertible(arg, arg.type, formals.head, warn);

1174                 formals = formals.tail;
1175                 nonInferred = nonInferred.tail;
1176             }

1178                 Type varArg = types.elemtype(last);
1179                 while (args.tail != null) {
1180                     JCTree arg = args.head;
1181                     Warner warn = convertWarner(arg.pos(), arg.type, varArg);
1182                     assertConvertible(arg, arg.type, varArg, warn);
1183                     args = args.tail;
1184                 }
1185             } else if ((sym.flags() & (VARARGS | SIGNATURE_POLYMORPHIC)) == VARARGS) {
1186                 // non-varargs call to varargs method
1187                 Type varParam = owntype.getParameterTypes().last();
1188                 Type lastArg = argtypes.last();
1189                 if (types.isSubtypeUnchecked(lastArg, types.elemtype(varParam)) &&

1193             }

1197             if (!types.isReifiable(argtype) &&
1198                 (sym.baseSymbol().attribute(syms.trustMeType.tsym) == null ||
1199                  !isTrustMeAllowedOnMethod(sym))) {
1200                 warnUnchecked(env.tree.pos(), Warnings.UncheckedGenericArrayCreation(argtype));
1201             }
1202             TreeInfo.setVarargsElement(env.tree, types.elemtype(argtype));
1203          }
1204          return owntype;

1228         private Type firstIncompatibleTypeArg(Type type) {
1229             List<Type> formals = type.tsym.type.allparams();
1230             List<Type> actuals = type.allparams();
1231             List<Type> args = type.getTypeArguments();
1232             List<Type> forms = type.tsym.type.getTypeArguments();
1233             ListBuffer<Type> bounds_buf = new ListBuffer<>();
1234 
1235             // For matching pairs of actual argument types `a' and
1236             // formal type parameters with declared bound `b' ...
1237             while (args.nonEmpty() && forms.nonEmpty()) {
1238                 // exact type arguments needs to know their
1239                 // bounds (for upper and lower bound
1240                 // calculations).  So we create new bounds where
1241                 // type-parameters are replaced with actuals argument types.
1242                 bounds_buf.append(types.subst(forms.head.getUpperBound(), formals, actuals));
1243                 args = args.tail;
1244                 forms = forms.tail;
1245             }
1246 
1247             args = type.getTypeArguments();
1248             List<Type> tvars_cap = types.substBounds(formals,
1249                                       formals,
1250                                       types.capture(type).allparams());
1251             while (args.nonEmpty() && tvars_cap.nonEmpty()) {
1252                 // Let the actual arguments know their bound
1253                 args.head.withTypeVar((TypeVar)tvars_cap.head);
1254                 args = args.tail;
1255                 tvars_cap = tvars_cap.tail;
1256             }
1257 
1258             args = type.getTypeArguments();
1259             List<Type> bounds = bounds_buf.toList();
1260 
1261             while (args.nonEmpty() && bounds.nonEmpty()) {
1262                 Type actual = args.head;
1263                 if (!isTypeArgErroneous(actual) &&
1264                         !bounds.head.isErroneous() &&
1265                         !checkExtends(actual, bounds.head)) {
1266                     return args.head;
1267                 }
1268                 args = args.tail;
1269                 bounds = bounds.tail;
1270             }
1271 
1272             args = type.getTypeArguments();
1273             bounds = bounds_buf.toList();
1274 
1275             for (Type arg : types.capture(type).getTypeArguments()) {
1276                 if (arg.hasTag(TYPEVAR) &&
1277                         arg.getUpperBound().isErroneous() &&
1278                         !bounds.head.isErroneous() &&
1279                         !isTypeArgErroneous(args.head)) {
1280                     return args.head;
1281                 }
1282                 bounds = bounds.tail;
1283                 args = args.tail;
1284             }
1285 
1286             return null;
1287         }

1312     /** Check that given modifiers are legal for given symbol and
1313      *  return modifiers together with any implicit modifiers for that symbol.
1314      *  Warning: we can't use flags() here since this method
1315      *  is called during class enter, when flags() would cause a premature
1316      *  completion.
1317      *  @param pos           Position to be used for error reporting.
1318      *  @param flags         The set of modifiers given in a definition.
1319      *  @param sym           The defined symbol.
1320      */
1321     long checkFlags(DiagnosticPosition pos, long flags, Symbol sym, JCTree tree) {
1322         long mask;
1323         long implicit = 0;
1324 
1325         switch (sym.kind) {
1326         case VAR:
1327             if (TreeInfo.isReceiverParam(tree))
1328                 mask = ReceiverParamFlags;
1329             else if (sym.owner.kind != TYP)
1330                 mask = LocalVarFlags;
1331             else if ((sym.owner.flags_field & INTERFACE) != 0)
1332                 mask = implicit = InterfaceVarFlags;
1333             else {
1334                 mask = VarFlags;
1335                 if (types.isValue(sym.owner.type) && (flags & STATIC) == 0) {
1336                     implicit |= FINAL;
1337                 }
1338             }
1339             break;
1340         case MTH:
1341             if (sym.name == names.init) {
1342                 if ((sym.owner.flags_field & ENUM) != 0) {
1343                     // enum constructors cannot be declared public or
1344                     // protected and must be implicitly or explicitly
1345                     // private
1346                     implicit = PRIVATE;
1347                     mask = PRIVATE;
1348                 } else
1349                     mask = ConstructorFlags;
1350             }  else if ((sym.owner.flags_field & INTERFACE) != 0) {
1351                 if ((sym.owner.flags_field & ANNOTATION) != 0) {
1352                     mask = AnnotationTypeElementMask;
1353                     implicit = PUBLIC | ABSTRACT;
1354                 } else if ((flags & (DEFAULT | STATIC | PRIVATE)) != 0) {
1355                     mask = InterfaceMethodMask;
1356                     implicit = (flags & PRIVATE) != 0 ? 0 : PUBLIC;
1357                     if ((flags & DEFAULT) != 0) {
1358                         implicit |= ABSTRACT;
1359                     }
1360                 } else {
1361                     mask = implicit = InterfaceMethodFlags;
1362                 }
1363             } else if ((sym.owner.flags_field & RECORD) != 0) {
1364                 mask = RecordMethodFlags;
1365             } else {
1366                 // instance methods of value types do not have a monitor associated with their `this'
1367                 mask = ((sym.owner.flags_field & VALUE) != 0 && (flags & Flags.STATIC) == 0) ?
1368                         MethodFlags & ~SYNCHRONIZED : MethodFlags;
1369             }
1370             // Imply STRICTFP if owner has STRICTFP set.
1371             if (((flags|implicit) & Flags.ABSTRACT) == 0 ||
1372                 ((flags) & Flags.DEFAULT) != 0)
1373                 implicit |= sym.owner.flags_field & STRICTFP;
1374             break;
1375         case TYP:
1376             if (sym.isLocal()) {
1377                 mask = (flags & RECORD) != 0 ? LocalRecordFlags : LocalClassFlags;
1378                 if ((flags & RECORD) != 0) {
1379                     implicit = STATIC;
1380                     if (sym.owner.kind == TYP) {
1381                         log.error(pos, Errors.RecordDeclarationNotAllowedInInnerClasses);
1382                     }
1383                 }
1384                 if ((sym.owner.flags_field & STATIC) == 0 &&
1385                     (flags & ENUM) != 0) {
1386                     log.error(pos, Errors.EnumsMustBeStatic);
1387                 }
1388             } else if (sym.owner.kind == TYP) {
1389                 mask = (flags & RECORD) != 0 ? MemberRecordFlags : MemberClassFlags;
1390                 if (sym.owner.owner.kind == PCK ||
1391                     (sym.owner.flags_field & STATIC) != 0)
1392                     mask |= STATIC;
1393                 else if ((flags & ENUM) != 0) {
1394                     log.error(pos, Errors.EnumsMustBeStatic);
1395                 } else if ((flags & RECORD) != 0) {
1396                     log.error(pos, Errors.RecordDeclarationNotAllowedInInnerClasses);
1397                 }
1398                 // Nested interfaces and enums are always STATIC (Spec ???)
1399                 if ((flags & (INTERFACE | ENUM | RECORD)) != 0 ) implicit = STATIC;
1400             } else {
1401                 mask = ClassFlags;
1402             }
1403             // Interfaces are always ABSTRACT
1404             if ((flags & INTERFACE) != 0) implicit |= ABSTRACT;
1405 
1406             if ((flags & ENUM) != 0) {
1407                 // enums can't be declared abstract or final or value type
1408                 mask &= ~(ABSTRACT | FINAL | VALUE);
1409                 implicit |= implicitEnumFinalFlag(tree);
1410             }
1411             if ((flags & RECORD) != 0) {
1412                 // records can't be declared abstract
1413                 mask &= ~ABSTRACT;
1414                 implicit |= FINAL;
1415             }
1416             // Imply STRICTFP if owner has STRICTFP set.
1417             implicit |= sym.owner.flags_field & STRICTFP;
1418             break;
1419         default:
1420             throw new AssertionError();
1421         }
1422         long illegal = flags & ExtendedStandardFlags & ~mask;
1423         if (illegal != 0) {
1424             if ((illegal & INTERFACE) != 0) {
1425                 log.error(pos, ((flags & ANNOTATION) != 0) ? Errors.AnnotationDeclNotAllowedHere : Errors.IntfNotAllowedHere);
1426                 mask |= INTERFACE;
1427             }
1428             else {
1429                 log.error(pos,
1430                           Errors.ModNotAllowedHere(asFlagSet(illegal)));
1431             }
1432         }
1433         else if ((sym.kind == TYP ||
1434                   // ISSUE: Disallowing abstract&private is no longer appropriate
1435                   // in the presence of inner classes. Should it be deleted here?
1436                   checkDisjoint(pos, flags,
1437                                 ABSTRACT,
1438                                 PRIVATE | STATIC | DEFAULT))
1439                  &&
1440                  checkDisjoint(pos, flags,
1441                                 STATIC | PRIVATE,
1442                                 DEFAULT)
1443                  &&
1444                  checkDisjoint(pos, flags,
1445                                ABSTRACT | INTERFACE,
1446                                FINAL | NATIVE | SYNCHRONIZED | VALUE)
1447                  &&
1448                  checkDisjoint(pos, flags,
1449                                PUBLIC,
1450                                PRIVATE | PROTECTED)
1451                  &&
1452                  checkDisjoint(pos, flags,
1453                                PRIVATE,
1454                                PUBLIC | PROTECTED)
1455                  &&
1456                  checkDisjoint(pos, flags,
1457                                FINAL,
1458                                VOLATILE)
1459                  &&
1460                  (sym.kind == TYP ||
1461                   checkDisjoint(pos, flags,
1462                                 ABSTRACT | NATIVE,
1463                                 STRICTFP))) {
1464             // skip
1465         }
1466         return flags & (mask | ~ExtendedStandardFlags) | implicit;
1467     }

1886     /** Check that this method conforms with overridden method 'other'.
1887      *  where `origin' is the class where checking started.
1888      *  Complications:
1889      *  (1) Do not check overriding of synthetic methods
1890      *      (reason: they might be final).
1891      *      todo: check whether this is still necessary.
1892      *  (2) Admit the case where an interface proxy throws fewer exceptions
1893      *      than the method it implements. Augment the proxy methods with the
1894      *      undeclared exceptions in this case.
1895      *  (3) When generics are enabled, admit the case where an interface proxy
1896      *      has a result type
1897      *      extended by the result type of the method it implements.
1898      *      Change the proxies result type to the smaller type in this case.
1899      *
1900      *  @param tree         The tree from which positions
1901      *                      are extracted for errors.
1902      *  @param m            The overriding method.
1903      *  @param other        The overridden method.
1904      *  @param origin       The class of which the overriding method
1905      *                      is a member.
1906      */
1907     void checkOverride(JCTree tree,
1908                        MethodSymbol m,
1909                        MethodSymbol other,
1910                        ClassSymbol origin) {
1911         // Don't check overriding of synthetic methods or by bridge methods.
1912         if ((m.flags() & (SYNTHETIC|BRIDGE)) != 0 || (other.flags() & SYNTHETIC) != 0) {
1913             return;
1914         }
1915 
1916         // Error if static method overrides instance method (JLS 8.4.6.2).
1917         if ((m.flags() & STATIC) != 0 &&
1918                    (other.flags() & STATIC) == 0) {
1919             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1920                       Errors.OverrideStatic(cannotOverride(m, other)));
1921             m.flags_field |= BAD_OVERRIDE;
1922             return;
1923         }
1924 
1925         // Error if instance method overrides static or final
1926         // method (JLS 8.4.6.1).
1927         if ((other.flags() & FINAL) != 0 ||
1928                  (m.flags() & STATIC) == 0 &&
1929                  (other.flags() & STATIC) != 0) {
1930             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1931                       Errors.OverrideMeth(cannotOverride(m, other),
1932                                           asFlagSet(other.flags() & (FINAL | STATIC))));
1933             m.flags_field |= BAD_OVERRIDE;
1934             return;
1935         }
1936 
1937         if ((m.owner.flags() & ANNOTATION) != 0) {
1938             // handled in validateAnnotationMethod
1939             return;
1940         }
1941 
1942         // Error if overriding method has weaker access (JLS 8.4.6.3).
1943         if (protection(m.flags()) > protection(other.flags())) {
1944             log.error(TreeInfo.diagnosticPositionFor(m, tree),
1945                       (other.flags() & AccessFlags) == 0 ?
1946                               Errors.OverrideWeakerAccess(cannotOverride(m, other),
1947                                                           "package") :
1948                               Errors.OverrideWeakerAccess(cannotOverride(m, other),
1949                                                           asFlagSet(other.flags() & AccessFlags)));
1950             m.flags_field |= BAD_OVERRIDE;
1951             return;
1952         }
1953 
1954         if (origin.isValue() && other.owner == syms.objectType.tsym && m.type.getParameterTypes().size() == 0) {
1955             if (m.name == names.clone || m.name == names.finalize) {
1956                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1957                         Errors.InlineClassMayNotOverride(m.name));
1958                 m.flags_field |= BAD_OVERRIDE;
1959                 return;
1960             }
1961         }
1962 
1963         Type mt = types.memberType(origin.type, m);
1964         Type ot = types.memberType(origin.type, other);
1965         // Error if overriding result type is different
1966         // (or, in the case of generics mode, not a subtype) of
1967         // overridden result type. We have to rename any type parameters
1968         // before comparing types.
1969         List<Type> mtvars = mt.getTypeArguments();
1970         List<Type> otvars = ot.getTypeArguments();
1971         Type mtres = mt.getReturnType();
1972         Type otres = types.subst(ot.getReturnType(), otvars, mtvars);
1973 
1974         overrideWarner.clear();
1975         boolean resultTypesOK =
1976             types.returnTypeSubstitutable(mt, ot, otres, overrideWarner);
1977         if (!resultTypesOK) {
1978             if ((m.flags() & STATIC) != 0 && (other.flags() & STATIC) != 0) {
1979                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1980                           Errors.OverrideIncompatibleRet(Fragments.CantHide(m, m.location(), other,
1981                                         other.location()), mtres, otres));
1982                 m.flags_field |= BAD_OVERRIDE;
1983             } else {
1984                 log.error(TreeInfo.diagnosticPositionFor(m, tree),
1985                           Errors.OverrideIncompatibleRet(cannotOverride(m, other), mtres, otres));
1986                 m.flags_field |= BAD_OVERRIDE;
1987             }
1988             return;
1989         } else if (overrideWarner.hasNonSilentLint(LintCategory.UNCHECKED)) {
1990             warnUnchecked(TreeInfo.diagnosticPositionFor(m, tree),
1991                     Warnings.OverrideUncheckedRet(uncheckedOverrides(m, other), mtres, otres));
1992         }
1993 
1994         // Error if overriding method throws an exception not reported
1995         // by overridden method.
1996         List<Type> otthrown = types.subst(ot.getThrownTypes(), otvars, mtvars);
1997         List<Type> unhandledErased = unhandled(mt.getThrownTypes(), types.erasure(otthrown));
1998         List<Type> unhandledUnerased = unhandled(mt.getThrownTypes(), otthrown);
1999         if (unhandledErased.nonEmpty()) {
2000             log.error(TreeInfo.diagnosticPositionFor(m, tree),
2001                       Errors.OverrideMethDoesntThrow(cannotOverride(m, other), unhandledUnerased.head));
2002             m.flags_field |= BAD_OVERRIDE;
2003             return;
2004         }
2005         else if (unhandledUnerased.nonEmpty()) {
2006             warnUnchecked(TreeInfo.diagnosticPositionFor(m, tree),
2007                           Warnings.OverrideUncheckedThrown(cannotOverride(m, other), unhandledUnerased.head));
2008             return;
2009         }
2010 
2011         // Optional warning if varargs don't agree
2012         if ((((m.flags() ^ other.flags()) & Flags.VARARGS) != 0)
2013             && lint.isEnabled(LintCategory.OVERRIDES)) {
2014             log.warning(TreeInfo.diagnosticPositionFor(m, tree),
2015                         ((m.flags() & Flags.VARARGS) != 0)
2016                         ? Warnings.OverrideVarargsMissing(varargsOverrides(m, other))
2017                         : Warnings.OverrideVarargsExtra(varargsOverrides(m, other)));
2018         }
2019 
2020         // Warn if instance method overrides bridge method (compiler spec ??)
2021         if ((other.flags() & BRIDGE) != 0) {
2022             log.warning(TreeInfo.diagnosticPositionFor(m, tree),
2023                         Warnings.OverrideBridge(uncheckedOverrides(m, other)));
2024         }
2025 
2026         // Warn if a deprecated method overridden by a non-deprecated one.
2027         if (!isDeprecatedOverrideIgnorable(other, origin)) {
2028             Lint prevLint = setLint(lint.augment(m));
2029             try {
2030                 checkDeprecated(() -> TreeInfo.diagnosticPositionFor(m, tree), m, other);
2031             } finally {
2032                 setLint(prevLint);
2033             }
2034         }

2869     /**
2870       * Report warnings for potentially ambiguous method declarations. Two declarations
2871       * are potentially ambiguous if they feature two unrelated functional interface
2872       * in same argument position (in which case, a call site passing an implicit
2873       * lambda would be ambiguous).
2874       */
2875     void checkPotentiallyAmbiguousOverloads(DiagnosticPosition pos, Type site,
2876             MethodSymbol msym1, MethodSymbol msym2) {
2877         if (msym1 != msym2 &&
2878                 Feature.DEFAULT_METHODS.allowedInSource(source) &&
2879                 lint.isEnabled(LintCategory.OVERLOADS) &&
2880                 (msym1.flags() & POTENTIALLY_AMBIGUOUS) == 0 &&
2881                 (msym2.flags() & POTENTIALLY_AMBIGUOUS) == 0) {
2882             Type mt1 = types.memberType(site, msym1);
2883             Type mt2 = types.memberType(site, msym2);
2884             //if both generic methods, adjust type variables
2885             if (mt1.hasTag(FORALL) && mt2.hasTag(FORALL) &&
2886                     types.hasSameBounds((ForAll)mt1, (ForAll)mt2)) {
2887                 mt2 = types.subst(mt2, ((ForAll)mt2).tvars, ((ForAll)mt1).tvars);
2888             }
2889             //expand varargs methods if needed
2890             int maxLength = Math.max(mt1.getParameterTypes().length(), mt2.getParameterTypes().length());
2891             List<Type> args1 = rs.adjustArgs(mt1.getParameterTypes(), msym1, maxLength, true);
2892             List<Type> args2 = rs.adjustArgs(mt2.getParameterTypes(), msym2, maxLength, true);
2893             //if arities don't match, exit
2894             if (args1.length() != args2.length()) return;
2895             boolean potentiallyAmbiguous = false;
2896             while (args1.nonEmpty() && args2.nonEmpty()) {
2897                 Type s = args1.head;
2898                 Type t = args2.head;
2899                 if (!types.isSubtype(t, s) && !types.isSubtype(s, t)) {
2900                     if (types.isFunctionalInterface(s) && types.isFunctionalInterface(t) &&
2901                             types.findDescriptorType(s).getParameterTypes().length() > 0 &&
2902                             types.findDescriptorType(s).getParameterTypes().length() ==
2903                             types.findDescriptorType(t).getParameterTypes().length()) {
2904                         potentiallyAmbiguous = true;
2905                     } else {
2906                         break;
2907                     }
2908                 }
2909                 args1 = args1.tail;
2910                 args2 = args2.tail;
2911             }
2912             if (potentiallyAmbiguous) {
2913                 //we found two incompatible functional interfaces with same arity
2914                 //this means a call site passing an implicit lambda would be ambiguous
2915                 msym1.flags_field |= POTENTIALLY_AMBIGUOUS;
2916                 msym2.flags_field |= POTENTIALLY_AMBIGUOUS;
2917                 log.warning(LintCategory.OVERLOADS, pos,
2918                             Warnings.PotentiallyAmbiguousOverload(msym1, msym1.location(),
2919                                                                   msym2, msym2.location()));
2920                 return;
2921             }
2922         }
