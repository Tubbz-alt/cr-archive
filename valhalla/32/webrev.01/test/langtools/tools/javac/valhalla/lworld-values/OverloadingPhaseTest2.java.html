31 public class OverloadingPhaseTest2 {
32 
33     interface A {}
34     interface B extends A {}
35 
36     static inline class X {
37 
38         int x = 42;
39 
40         static int m(X.ref xr, A a) {
41             return 0;
42         }
43 
44         static int m(X.ref xr, B b) {
45             return 1;
46         }
47     }
48 
49     public static void main(String [] args) {
50         B b = null;
51         X x = new X();
52         if (X.m(x, null) != 1)
53             throw new AssertionError("Failed");
54     }
55 }
