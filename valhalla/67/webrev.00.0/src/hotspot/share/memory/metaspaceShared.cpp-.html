  91 // The CDS archive is divided into the following regions:
  92 //     mc  - misc code (the method entry trampolines, c++ vtables)
  93 //     rw  - read-write metadata
  94 //     ro  - read-only metadata and read-only tables
  95 //
  96 //     ca0 - closed archive heap space #0
  97 //     ca1 - closed archive heap space #1 (may be empty)
  98 //     oa0 - open archive heap space #0
  99 //     oa1 - open archive heap space #1 (may be empty)
 100 //
 101 // The mc, rw, and ro regions are linearly allocated, starting from
 102 // SharedBaseAddress, in the order of mc->rw->ro. The size of these 3 regions
 103 // are page-aligned, and there's no gap between any consecutive regions.
 104 //
 105 // These 3 regions are populated in the following steps:
 106 // [1] All classes are loaded in MetaspaceShared::preload_classes(). All metadata are
 107 //     temporarily allocated outside of the shared regions. Only the method entry
 108 //     trampolines are written into the mc region.
 109 // [2] C++ vtables are copied into the mc region.
 110 // [3] ArchiveCompactor copies RW metadata into the rw region.
 111 // [4] ArchiveCompactor copies RO metadata into the ro region.
 112 // [5] SymbolTable, StringTable, SystemDictionary, and a few other read-only data
 113 //     are copied into the ro region as read-only tables.
 114 //
 115 // The s0/s1 and oa0/oa1 regions are populated inside HeapShared::archive_java_heap_objects.
 116 // Their layout is independent of the other 4 regions.
 117 
 118 char* DumpRegion::expand_top_to(char* newtop) {
 119   assert(is_allocatable(), "must be initialized and not packed");
 120   assert(newtop >= _top, "must not grow backwards");
 121   if (newtop > _end) {
 122     MetaspaceShared::report_out_of_space(_name, newtop - _top);
 123     ShouldNotReachHere();
 124   }
 125   uintx delta;
 126   if (DynamicDumpSharedSpaces) {
 127     delta = DynamicArchive::object_delta_uintx(newtop);
 128   } else {
 129     delta = MetaspaceShared::object_delta_uintx(newtop);
 130   }
 131   if (delta > MAX_SHARED_DELTA) {
 132     // This is just a sanity check and should not appear in any real world usage. This
 133     // happens only if you allocate more than 2GB of shared objects and would require
 134     // millions of shared classes.
 135     vm_exit_during_initialization("Out of memory in the CDS archive",
 136                                   "Please reduce the number of shared classes.");
 137   }
 138 
 139   MetaspaceShared::commit_shared_space_to(newtop);
 140   _top = newtop;
 141   return _top;
 142 }

 144 char* DumpRegion::allocate(size_t num_bytes, size_t alignment) {
 145   char* p = (char*)align_up(_top, alignment);
 146   char* newtop = p + align_up(num_bytes, alignment);
 147   expand_top_to(newtop);
 148   memset(p, 0, newtop - p);
 149   return p;
 150 }

 152 void DumpRegion::append_intptr_t(intptr_t n, bool need_to_mark) {
 153   assert(is_aligned(_top, sizeof(intptr_t)), "bad alignment");
 154   intptr_t *p = (intptr_t*)_top;
 155   char* newtop = _top + sizeof(intptr_t);
 156   expand_top_to(newtop);
 157   *p = n;
 158   if (need_to_mark) {
 159     ArchivePtrMarker::mark_pointer(p);
 160   }
 161 }

 177 void DumpRegion::pack(DumpRegion* next) {
 178   assert(!is_packed(), "sanity");
 179   _end = (char*)align_up(_top, Metaspace::reserve_alignment());
 180   _is_packed = true;
 181   if (next != NULL) {
 182     next->_base = next->_top = this->_end;
 183     next->_end = MetaspaceShared::shared_rs()->end();
 184   }
 185 }

 224 // When reserving an address range using ReservedSpace, we need an alignment that satisfies both:
 225 // os::vm_allocation_granularity() -- so that we can sub-divide this range into multiple mmap regions,
 226 //                                    while keeping the first range at offset 0 of this range.
 227 // Metaspace::reserve_alignment()  -- so we can pass the region to
 228 //                                    Metaspace::allocate_metaspace_compressed_klass_ptrs.
 229 size_t MetaspaceShared::reserved_space_alignment() {
 230   size_t os_align = os::vm_allocation_granularity();
 231   size_t ms_align = Metaspace::reserve_alignment();
 232   if (os_align >= ms_align) {
 233     assert(os_align % ms_align == 0, "must be a multiple");
 234     return os_align;
 235   } else {
 236     assert(ms_align % os_align == 0, "must be a multiple");
 237     return ms_align;
 238   }
 239 }

 246 void MetaspaceShared::initialize_dumptime_shared_and_meta_spaces() {
 247   assert(DumpSharedSpaces, "should be called for dump time only");
 248   const size_t reserve_alignment = reserved_space_alignment();
 249   char* shared_base = (char*)align_up((char*)SharedBaseAddress, reserve_alignment);
 250 
 251 #ifdef _LP64
 252   // On 64-bit VM, the heap and class space layout will be the same as if
 253   // you're running in -Xshare:on mode:
 254   //
 255   //                              +-- SharedBaseAddress (default = 0x800000000)
 256   //                              v
 257   // +-..---------+---------+ ... +----+----+----+--------------------+
 258   // |    Heap    | Archive |     | MC | RW | RO |    class space     |
 259   // +-..---------+---------+ ... +----+----+----+--------------------+
 260   // |<--   MaxHeapSize  -->|     |<-- UnscaledClassSpaceMax = 4GB -->|
 261   //
 262   const uint64_t UnscaledClassSpaceMax = (uint64_t(max_juint) + 1);
 263   const size_t cds_total = align_down(UnscaledClassSpaceMax, reserve_alignment);
 264 #else
 265   // We don't support archives larger than 256MB on 32-bit due to limited virtual address space.
 266   size_t cds_total = align_down(256*M, reserve_alignment);
 267 #endif
 268 
 269   bool use_requested_base = true;
 270   if (ArchiveRelocationMode == 1) {
 271     log_info(cds)("ArchiveRelocationMode == 1: always allocate class space at an alternative address");
 272     use_requested_base = false;
 273   }
 274 
 275   // First try to reserve the space at the specified SharedBaseAddress.
 276   assert(!_shared_rs.is_reserved(), "must be");
 277   if (use_requested_base) {
 278     _shared_rs = reserve_shared_space(cds_total, shared_base);
 279   }
 280   if (_shared_rs.is_reserved()) {
 281     assert(shared_base == 0 || _shared_rs.base() == shared_base, "should match");
 282   } else {
 283     // Get a mmap region anywhere if the SharedBaseAddress fails.
 284     _shared_rs = reserve_shared_space(cds_total);
 285   }
 286   if (!_shared_rs.is_reserved()) {
 287     vm_exit_during_initialization("Unable to reserve memory for shared space",
 288                                   err_msg(SIZE_FORMAT " bytes.", cds_total));
 289   }
 290 
 291 #ifdef _LP64
 292   // During dump time, we allocate 4GB (UnscaledClassSpaceMax) of space and split it up:
 293   // + The upper 1 GB is used as the "temporary compressed class space" -- preload_classes()
 294   //   will store Klasses into this space.
 295   // + The lower 3 GB is used for the archive -- when preload_classes() is done,
 296   //   ArchiveCompactor will copy the class metadata into this space, first the RW parts,
 297   //   then the RO parts.
 298 
 299   size_t max_archive_size = align_down(cds_total * 3 / 4, reserve_alignment);
 300   ReservedSpace tmp_class_space = _shared_rs.last_part(max_archive_size);
 301   CompressedClassSpaceSize = align_down(tmp_class_space.size(), reserve_alignment);
 302   _shared_rs = _shared_rs.first_part(max_archive_size);
 303 
 304   if (UseCompressedClassPointers) {
 305     // Set up compress class pointers.
 306     CompressedKlassPointers::set_base((address)_shared_rs.base());
 307     // Set narrow_klass_shift to be LogKlassAlignmentInBytes. This is consistent
 308     // with AOT.
 309     CompressedKlassPointers::set_shift(LogKlassAlignmentInBytes);
 310     // Set the range of klass addresses to 4GB.
 311     CompressedKlassPointers::set_range(cds_total);
 312     Metaspace::initialize_class_space(tmp_class_space);
 313   }
 314   log_info(cds)("narrow_klass_base = " PTR_FORMAT ", narrow_klass_shift = %d",
 315                 p2i(CompressedKlassPointers::base()), CompressedKlassPointers::shift());
 316 
 317   log_info(cds)("Allocated temporary class space: " SIZE_FORMAT " bytes at " PTR_FORMAT,
 318                 CompressedClassSpaceSize, p2i(tmp_class_space.base()));
 319 #endif
 320 
 321   init_shared_dump_space(&_mc_region);
 322   SharedBaseAddress = (size_t)_shared_rs.base();
 323   log_info(cds)("Allocated shared space: " SIZE_FORMAT " bytes at " PTR_FORMAT,
 324                 _shared_rs.size(), p2i(_shared_rs.base()));
 325 }

 349 void MetaspaceShared::read_extra_data(const char* filename, TRAPS) {
 350   _extra_interned_strings = new (ResourceObj::C_HEAP, mtInternal)GrowableArray<Handle>(10000, true);
 351 
 352   HashtableTextDump reader(filename);
 353   reader.check_version("VERSION: 1.0");
 354 
 355   while (reader.remain() > 0) {
 356     int utf8_length;
 357     int prefix_type = reader.scan_prefix(&utf8_length);
 358     ResourceMark rm(THREAD);
 359     if (utf8_length == 0x7fffffff) {
 360       // buf_len will overflown 32-bit value.
 361       vm_exit_during_initialization(err_msg("string length too large: %d", utf8_length));
 362     }
 363     int buf_len = utf8_length+1;
 364     char* utf8_buffer = NEW_RESOURCE_ARRAY(char, buf_len);
 365     reader.get_utf8(utf8_buffer, utf8_length);
 366     utf8_buffer[utf8_length] = '\0';
 367 
 368     if (prefix_type == HashtableTextDump::SymbolPrefix) {
 369       SymbolTable::new_permanent_symbol(utf8_buffer);
 370     } else{
 371       assert(prefix_type == HashtableTextDump::StringPrefix, "Sanity");
 372       oop s = StringTable::intern(utf8_buffer, THREAD);
 373 
 374       if (HAS_PENDING_EXCEPTION) {
 375         log_warning(cds, heap)("[line %d] extra interned string allocation failed; size too large: %d",
 376                                reader.last_line_no(), utf8_length);
 377         CLEAR_PENDING_EXCEPTION;
 378       } else {
 379 #if INCLUDE_G1GC
 380         if (UseG1GC) {
 381           typeArrayOop body = java_lang_String::value(s);
 382           const HeapRegion* hr = G1CollectedHeap::heap()->heap_region_containing(body);
 383           if (hr->is_humongous()) {
 384             // Don't keep it alive, so it will be GC'ed before we dump the strings, in order
 385             // to maximize free heap space and minimize fragmentation.
 386             log_warning(cds, heap)("[line %d] extra interned string ignored; size too large: %d",
 387                                 reader.last_line_no(), utf8_length);
 388             continue;
 389           }
 390         }
 391 #endif
 392         // Interned strings are GC'ed if there are no references to it, so let's
 393         // add a reference to keep this string alive.
 394         assert(s != NULL, "must succeed");
 395         Handle h(THREAD, s);
 396         _extra_interned_strings->append(h);
 397       }
 398     }
 399   }
 400 }

 402 void MetaspaceShared::commit_shared_space_to(char* newtop) {
 403   Arguments::assert_is_dumping_archive();
 404   char* base = _shared_rs.base();
 405   size_t need_committed_size = newtop - base;
 406   size_t has_committed_size = _shared_vs.committed_size();
 407   if (need_committed_size < has_committed_size) {
 408     return;
 409   }
 410 
 411   size_t min_bytes = need_committed_size - has_committed_size;
 412   size_t preferred_bytes = 1 * M;
 413   size_t uncommitted = _shared_vs.reserved_size() - has_committed_size;
 414 
 415   size_t commit =MAX2(min_bytes, preferred_bytes);
 416   commit = MIN2(commit, uncommitted);
 417   assert(commit <= uncommitted, "sanity");
 418 
 419   bool result = _shared_vs.expand_by(commit, false);
 420   ArchivePtrMarker::expand_ptr_end((address*)_shared_vs.high());
 421 
 422   if (!result) {
 423     vm_exit_during_initialization(err_msg("Failed to expand shared space to " SIZE_FORMAT " bytes",
 424                                           need_committed_size));
 425   }
 426 
 427   log_debug(cds)("Expanding shared spaces by " SIZE_FORMAT_W(7) " bytes [total " SIZE_FORMAT_W(9)  " bytes ending at %p]",
 428                  commit, _shared_vs.actual_committed_size(), _shared_vs.high());
 429 }

 438 void MetaspaceShared::serialize(SerializeClosure* soc) {
 439   int tag = 0;
 440   soc->do_tag(--tag);
 441 
 442   // Verify the sizes of various metadata in the system.
 443   soc->do_tag(sizeof(Method));
 444   soc->do_tag(sizeof(ConstMethod));
 445   soc->do_tag(arrayOopDesc::base_offset_in_bytes(T_BYTE));
 446   soc->do_tag(sizeof(ConstantPool));
 447   soc->do_tag(sizeof(ConstantPoolCache));
 448   soc->do_tag(objArrayOopDesc::base_offset_in_bytes());
 449   soc->do_tag(typeArrayOopDesc::base_offset_in_bytes(T_BYTE));
 450   soc->do_tag(sizeof(Symbol));
 451 
 452   // Dump/restore miscellaneous metadata.
 453   JavaClasses::serialize_offsets(soc);
 454   Universe::serialize(soc);
 455   soc->do_tag(--tag);
 456 
 457   // Dump/restore references to commonly used names and signatures.
 458   vmSymbols::serialize(soc);
 459   soc->do_tag(--tag);
 460 
 461   // Dump/restore the symbol/string/subgraph_info tables
 462   SymbolTable::serialize_shared_table_header(soc);
 463   StringTable::serialize_shared_table_header(soc);
 464   HeapShared::serialize_subgraph_info_table_header(soc);
 465   SystemDictionaryShared::serialize_dictionary_headers(soc);
 466 
 467   InstanceMirrorKlass::serialize_offsets(soc);
 468 
 469   // Dump/restore well known classes (pointers)
 470   SystemDictionaryShared::serialize_well_known_klasses(soc);
 471   soc->do_tag(--tag);
 472 
 473   serialize_cloned_cpp_vtptrs(soc);
 474   soc->do_tag(--tag);
 475 
 476   soc->do_tag(666);
 477 }

 495 uintx MetaspaceShared::object_delta_uintx(void* obj) {
 496   Arguments::assert_is_dumping_archive();
 497   if (DumpSharedSpaces) {
 498     assert(shared_rs()->contains(obj), "must be");
 499   } else {
 500     assert(is_in_shared_metaspace(obj) || DynamicArchive::is_in_target_space(obj), "must be");
 501   }
 502   address base_address = address(SharedBaseAddress);
 503   uintx deltax = address(obj) - base_address;
 504   return deltax;
 505 }

 546 static void remove_unshareable_in_classes() {
 547   for (int i = 0; i < _global_klass_objects->length(); i++) {
 548     Klass* k = _global_klass_objects->at(i);
 549     if (!k->is_objArray_klass()) {

 583 static void rewrite_nofast_bytecode(const methodHandle& method) {
 584   BytecodeStream bcs(method);
 585   while (!bcs.is_last_bytecode()) {
 586     Bytecodes::Code opcode = bcs.next();
 587     switch (opcode) {
 588     case Bytecodes::_getfield:      *bcs.bcp() = Bytecodes::_nofast_getfield;      break;
 589     case Bytecodes::_putfield:      *bcs.bcp() = Bytecodes::_nofast_putfield;      break;
 590     case Bytecodes::_aload_0:       *bcs.bcp() = Bytecodes::_nofast_aload_0;       break;
 591     case Bytecodes::_iload: {
 592       if (!bcs.is_wide()) {
 593         *bcs.bcp() = Bytecodes::_nofast_iload;
 594       }
 595       break;
 596     }
 597     default: break;
 598     }
 599   }
 600 }

 627 // Objects of the Metadata types (such as Klass and ConstantPool) have C++ vtables.
 628 // (In GCC this is the field <Type>::_vptr, i.e., first word in the object.)
 629 //
 630 // Addresses of the vtables and the methods may be different across JVM runs,
 631 // if libjvm.so is dynamically loaded at a different base address.
 632 //
 633 // To ensure that the Metadata objects in the CDS archive always have the correct vtable:
 634 //
 635 // + at dump time:  we redirect the _vptr to point to our own vtables inside
 636 //                  the CDS image
 637 // + at run time:   we clone the actual contents of the vtables from libjvm.so
 638 //                  into our own tables.
 639 
 640 // Currently, the archive contain ONLY the following types of objects that have C++ vtables.
 641 #define CPP_VTABLE_PATCH_TYPES_DO(f) \
 642   f(ConstantPool) \
 643   f(InstanceClassLoaderKlass) \
 644   f(InstanceKlass) \
 645   f(InstanceMirrorKlass) \
 646   f(InstanceRefKlass) \
 647   f(Method) \
 648   f(ObjArrayKlass) \
 649   f(TypeArrayKlass) \
 650   f(ValueArrayKlass) \
 651   f(ValueKlass)

 653 class CppVtableInfo {
 654   intptr_t _vtable_size;
 655   intptr_t _cloned_vtable[1];
 656 public:
 657   static int num_slots(int vtable_size) {
 658     return 1 + vtable_size; // Need to add the space occupied by _vtable_size;
 659   }
 660   int vtable_size()           { return int(uintx(_vtable_size)); }
 661   void set_vtable_size(int n) { _vtable_size = intptr_t(n); }
 662   intptr_t* cloned_vtable()   { return &_cloned_vtable[0]; }
 663   void zero()                 { memset(_cloned_vtable, 0, sizeof(intptr_t) * vtable_size()); }
 664   // Returns the address of the next CppVtableInfo that can be placed immediately after this CppVtableInfo
 665   static size_t byte_size(int vtable_size) {
 666     CppVtableInfo i;
 667     return pointer_delta(&i._cloned_vtable[vtable_size], &i, sizeof(u1));
 668   }
 669 };

 712 template <class T>
 713 intptr_t* CppVtableCloner<T>::clone_vtable(const char* name, CppVtableInfo* info) {
 714   if (!DumpSharedSpaces) {
 715     assert(_info == 0, "_info is initialized only at dump time");
 716     _info = info; // Remember it -- it will be used by MetaspaceShared::is_valid_shared_method()
 717   }
 718   T tmp; // Allocate temporary dummy metadata object to get to the original vtable.
 719   int n = info->vtable_size();
 720   intptr_t* srcvtable = vtable_of(tmp);
 721   intptr_t* dstvtable = info->cloned_vtable();
 722 
 723   // We already checked (and, if necessary, adjusted n) when the vtables were allocated, so we are
 724   // safe to do memcpy.
 725   log_debug(cds, vtables)("Copying %3d vtable entries for %s", n, name);
 726   memcpy(dstvtable, srcvtable, sizeof(intptr_t) * n);
 727   return dstvtable + n;
 728 }

 730 // To determine the size of the vtable for each type, we use the following
 731 // trick by declaring 2 subclasses:
 732 //
 733 //   class CppVtableTesterA: public InstanceKlass {virtual int   last_virtual_method() {return 1;}    };
 734 //   class CppVtableTesterB: public InstanceKlass {virtual void* last_virtual_method() {return NULL}; };
 735 //
 736 // CppVtableTesterA and CppVtableTesterB's vtables have the following properties:
 737 // - Their size (N+1) is exactly one more than the size of InstanceKlass's vtable (N)
 738 // - The first N entries have are exactly the same as in InstanceKlass's vtable.
 739 // - Their last entry is different.
 740 //
 741 // So to determine the value of N, we just walk CppVtableTesterA and CppVtableTesterB's tables
 742 // and find the first entry that's different.
 743 //
 744 // This works on all C++ compilers supported by Oracle, but you may need to tweak it for more
 745 // esoteric compilers.
 746 
 747 template <class T> class CppVtableTesterB: public T {
 748 public:
 749   virtual int last_virtual_method() {return 1;}
 750 };
 751 
 752 template <class T> class CppVtableTesterA : public T {
 753 public:
 754   virtual void* last_virtual_method() {
 755     // Make this different than CppVtableTesterB::last_virtual_method so the C++
 756     // compiler/linker won't alias the two functions.
 757     return NULL;
 758   }
 759 };
 760 
 761 template <class T>
 762 int CppVtableCloner<T>::get_vtable_length(const char* name) {
 763   CppVtableTesterA<T> a;
 764   CppVtableTesterB<T> b;
 765 
 766   intptr_t* avtable = vtable_of(a);
 767   intptr_t* bvtable = vtable_of(b);
 768 
 769   // Start at slot 1, because slot 0 may be RTTI (on Solaris/Sparc)
 770   int vtable_len = 1;
 771   for (; ; vtable_len++) {
 772     if (avtable[vtable_len] != bvtable[vtable_len]) {
 773       break;
 774     }
 775   }
 776   log_debug(cds, vtables)("Found   %3d vtable entries for %s", vtable_len, name);
 777 
 778   return vtable_len;
 779 }

 819   Arguments::assert_is_dumping_archive();
 820   int kind = -1;
 821   switch (msotype) {
 822   case MetaspaceObj::SymbolType:
 823   case MetaspaceObj::TypeArrayU1Type:
 824   case MetaspaceObj::TypeArrayU2Type:
 825   case MetaspaceObj::TypeArrayU4Type:
 826   case MetaspaceObj::TypeArrayU8Type:
 827   case MetaspaceObj::TypeArrayOtherType:
 828   case MetaspaceObj::ConstMethodType:
 829   case MetaspaceObj::ConstantPoolCacheType:
 830   case MetaspaceObj::AnnotationsType:
 831   case MetaspaceObj::MethodCountersType:
 832   case MetaspaceObj::RecordComponentType:
 833     // These have no vtables.
 834     break;
 835   case MetaspaceObj::ClassType:
 836     {
 837       Klass* k = (Klass*)obj;
 838       assert(k->is_klass(), "must be");
 839       if (k->is_value()) {
 840         kind = ValueKlass_Kind;
 841       } else if (k->is_instance_klass()) {
 842         InstanceKlass* ik = InstanceKlass::cast(k);
 843         if (ik->is_class_loader_instance_klass()) {
 844           kind = InstanceClassLoaderKlass_Kind;
 845         } else if (ik->is_reference_instance_klass()) {
 846           kind = InstanceRefKlass_Kind;
 847         } else if (ik->is_mirror_instance_klass()) {
 848           kind = InstanceMirrorKlass_Kind;
 849         } else {
 850           kind = InstanceKlass_Kind;
 851         }
 852       } else if (k->is_typeArray_klass()) {
 853         kind = TypeArrayKlass_Kind;
 854       } else {
 855         assert(k->is_objArray_klass(), "must be");
 856         kind = ObjArrayKlass_Kind;
 857       }

 860 
 861   case MetaspaceObj::MethodType:
 862     {
 863       Method* m = (Method*)obj;
 864       assert(m->is_method(), "must be");
 865       kind = Method_Kind;
 866     }
 867     break;
 868

 870     // We don't archive MethodData <-- should have been removed in removed_unsharable_info

 891   }

1192     address obj = ref->obj();
1193     int bytes = ref->size() * BytesPerWord;
1194     char* p;
1195     size_t alignment = BytesPerWord;
1196     char* oldtop;
1197     char* newtop;

1199     if (read_only) {

1201       p = _ro_region.allocate(bytes, alignment);
1202       newtop = _ro_region.top();
1203     } else {
1204       oldtop = _rw_region.top();
1205       if (ref->msotype() == MetaspaceObj::ClassType) {
1206         // Save a pointer immediate in front of an InstanceKlass, so
1207         // we can do a quick lookup from InstanceKlass* -> RunTimeSharedClassInfo*
1208         // without building another hashtable. See RunTimeSharedClassInfo::get_for()
1209         // in systemDictionaryShared.cpp.
1210         Klass* klass = (Klass*)obj;
1211         if (klass->is_instance_klass()) {
1212           SystemDictionaryShared::validate_before_archiving(InstanceKlass::cast(klass));
1213           _rw_region.allocate(sizeof(address), BytesPerWord);

1215       }

1259   public:

1280           assert(0 <= intx(off) && intx(off) < ref->size() * BytesPerWord, "must point to internal address");

1474 void VM_PopulateDumpSharedSpace::relocate_to_default_base_address(CHeapBitMap* ptrmap) {
1475   intx addr_delta = MetaspaceShared::final_delta();
1476   if (addr_delta == 0) {
1477     ArchivePtrMarker::compact((address)SharedBaseAddress, (address)_ro_region.top());
1478   } else {
1479     // We are not able to reserve space at Arguments::default_SharedBaseAddress() (due to ASLR).
1480     // This means that the current content of the archive is based on a random
1481     // address. Let's relocate all the pointers, so that it can be mapped to
1482     // Arguments::default_SharedBaseAddress() without runtime relocation.
1483     //
1484     // Note: both the base and dynamic archive are written with
1485     // FileMapHeader::_shared_base_address == Arguments::default_SharedBaseAddress()
1486 
1487     // Patch all pointers that are marked by ptrmap within this region,
1488     // where we have just dumped all the metaspace data.
1489     address patch_base = (address)SharedBaseAddress;
1490     address patch_end  = (address)_ro_region.top();
1491     size_t size = patch_end - patch_base;
1492 
1493     // the current value of the pointers to be patched must be within this
1494     // range (i.e., must point to valid metaspace objects)
1495     address valid_old_base = patch_base;
1496     address valid_old_end  = patch_end;
1497 
1498     // after patching, the pointers must point inside this range
1499     // (the requested location of the archive, as mapped at runtime).
1500     address valid_new_base = (address)Arguments::default_SharedBaseAddress();
1501     address valid_new_end  = valid_new_base + size;
1502 
1503     log_debug(cds)("Relocating archive from [" INTPTR_FORMAT " - " INTPTR_FORMAT " ] to "
1504                    "[" INTPTR_FORMAT " - " INTPTR_FORMAT " ]", p2i(patch_base), p2i(patch_end),
1505                    p2i(valid_new_base), p2i(valid_new_end));
1506 
1507     SharedDataRelocator<true> patcher((address*)patch_base, (address*)patch_end, valid_old_base, valid_old_end,
1508                                       valid_new_base, valid_new_end, addr_delta, ptrmap);
1509     ptrmap->iterate(&patcher);
1510     ArchivePtrMarker::compact(patcher.max_non_null_offset());
1511   }
1512 }

1514 void VM_PopulateDumpSharedSpace::doit() {
1515   CHeapBitMap ptrmap;
1516   MetaspaceShared::initialize_ptr_marker(&ptrmap);
1517 
1518   // We should no longer allocate anything from the metaspace, so that:
1519   //
1520   // (1) Metaspace::allocate might trigger GC if we have run out of
1521   //     committed metaspace, but we can't GC because we're running
1522   //     in the VM thread.
1523   // (2) ArchiveCompactor needs to work with a stable set of MetaspaceObjs.
1524   Metaspace::freeze();
1525   DEBUG_ONLY(SystemDictionaryShared::NoClassLoadingMark nclm);
1526 
1527   Thread* THREAD = VMThread::vm_thread();
1528 
1529   FileMapInfo::check_nonempty_dir_in_shared_path_table();
1530 
1531   NOT_PRODUCT(SystemDictionary::verify();)
1532   // The following guarantee is meant to ensure that no loader constraints
1533   // exist yet, since the constraints table is not shared.  This becomes
1534   // more important now that we don't re-initialize vtables/itables for
1535   // shared classes at runtime, where constraints were previously created.
1536   guarantee(SystemDictionary::constraints()->number_of_entries() == 0,
1537             "loader constraints are not saved");
1538   guarantee(SystemDictionary::placeholders()->number_of_entries() == 0,
1539           "placeholders are not saved");
1540 
1541   // At this point, many classes have been loaded.
1542   // Gather systemDictionary classes in a global array and do everything to
1543   // that so we don't have to walk the SystemDictionary again.
1544   SystemDictionaryShared::check_excluded_classes();
1545   _global_klass_objects = new GrowableArray<Klass*>(1000);
1546   CollectClassesClosure collect_classes;
1547   ClassLoaderDataGraph::loaded_classes_do(&collect_classes);
1548 
1549   print_class_stats();
1550 
1551   // Ensure the ConstMethods won't be modified at run-time
1552   log_info(cds)("Updating ConstMethods ... ");
1553   rewrite_nofast_bytecodes_and_calculate_fingerprints(THREAD);
1554   log_info(cds)("done. ");
1555 
1556   // Remove all references outside the metadata
1557   log_info(cds)("Removing unshareable information ... ");
1558   remove_unshareable_in_classes();
1559   log_info(cds)("done. ");
1560 
1561   MetaspaceShared::allocate_cloned_cpp_vtptrs();
1562   char* cloned_vtables = _mc_region.top();
1563   MetaspaceShared::allocate_cpp_vtable_clones();
1564 
1565   ArchiveCompactor::initialize();
1566   ArchiveCompactor::copy_and_compact();
1567 
1568   dump_symbols();
1569 
1570   // Dump supported java heap objects
1571   _closed_archive_heap_regions = NULL;
1572   _open_archive_heap_regions = NULL;
1573   dump_java_heap_objects();
1574 
1575   ArchiveCompactor::relocate_well_known_klasses();
1576 
1577   char* serialized_data = dump_read_only_tables();
1578   _ro_region.pack();
1579 
1580   // The vtable clones contain addresses of the current process.
1581   // We don't want to write these addresses into the archive.
1582   MetaspaceShared::zero_cpp_vtable_clones_for_writing();
1583 
1584   // relocate the data so that it can be mapped to Arguments::default_SharedBaseAddress()
1585   // without runtime relocation.
1586   relocate_to_default_base_address(&ptrmap);
1587 
1588   // Create and write the archive file that maps the shared spaces.
1589 
1590   FileMapInfo* mapinfo = new FileMapInfo(true);
1591   mapinfo->populate_header(os::vm_allocation_granularity());
1592   mapinfo->set_serialized_data(serialized_data);
1593   mapinfo->set_cloned_vtables(cloned_vtables);
1594   mapinfo->set_i2i_entry_code_buffers(MetaspaceShared::i2i_entry_code_buffers(),
1595                                       MetaspaceShared::i2i_entry_code_buffers_size());
1596   mapinfo->open_for_write();
1597   MetaspaceShared::write_core_archive_regions(mapinfo, _closed_archive_heap_oopmaps, _open_archive_heap_oopmaps);
1598   _total_closed_archive_region_size = mapinfo->write_archive_heap_regions(
1599                                         _closed_archive_heap_regions,
1600                                         _closed_archive_heap_oopmaps,
1601                                         MetaspaceShared::first_closed_archive_heap_region,
1602                                         MetaspaceShared::max_closed_archive_heap_region);
1603   _total_open_archive_region_size = mapinfo->write_archive_heap_regions(
1604                                         _open_archive_heap_regions,
1605                                         _open_archive_heap_oopmaps,
1606                                         MetaspaceShared::first_open_archive_heap_region,
1607                                         MetaspaceShared::max_open_archive_heap_region);
1608 
1609   mapinfo->set_final_requested_base((char*)Arguments::default_SharedBaseAddress());
1610   mapinfo->set_header_crc(mapinfo->compute_header_crc());
1611   mapinfo->write_header();
1612   print_region_stats(mapinfo);
1613   mapinfo->close();
1614 
1615   if (log_is_enabled(Info, cds)) {
1616     ArchiveCompactor::alloc_stats()->print_stats(int(_ro_region.used()), int(_rw_region.used()),
1617                                                  int(_mc_region.used()));
1618   }
1619 
1620   if (PrintSystemDictionaryAtExit) {
1621     SystemDictionary::print();
1622   }
1623 
1624   if (AllowArchivingWithJavaAgent) {
1625     warning("This archive was created with AllowArchivingWithJavaAgent. It should be used "
1626             "for testing purposes only and should not be used in a production environment");
1627   }
1628 
1629   // There may be other pending VM operations that operate on the InstanceKlasses,
1630   // which will fail because InstanceKlasses::remove_unshareable_info()
1631   // has been called. Forget these operations and exit the VM directly.
1632   vm_direct_exit(0);
1633 }

1765 // file.
1766 void MetaspaceShared::preload_and_dump(TRAPS) {
1767   { TraceTime timer("Dump Shared Spaces", TRACETIME_LOG(Info, startuptime));
1768     ResourceMark rm(THREAD);
1769     char class_list_path_str[JVM_MAXPATHLEN];
1770     // Preload classes to be shared.
1771     const char* class_list_path;
1772     if (SharedClassListFile == NULL) {
1773       // Construct the path to the class list (in jre/lib)
1774       // Walk up two directories from the location of the VM and
1775       // optionally tack on "lib" (depending on platform)
1776       os::jvm_path(class_list_path_str, sizeof(class_list_path_str));
1777       for (int i = 0; i < 3; i++) {
1778         char *end = strrchr(class_list_path_str, *os::file_separator());
1779         if (end != NULL) *end = '\0';
1780       }
1781       int class_list_path_len = (int)strlen(class_list_path_str);
1782       if (class_list_path_len >= 3) {
1783         if (strcmp(class_list_path_str + class_list_path_len - 3, "lib") != 0) {
1784           if (class_list_path_len < JVM_MAXPATHLEN - 4) {
1785             jio_snprintf(class_list_path_str + class_list_path_len,
1786                          sizeof(class_list_path_str) - class_list_path_len,
1787                          "%slib", os::file_separator());
1788             class_list_path_len += 4;
1789           }
1790         }
1791       }
1792       if (class_list_path_len < JVM_MAXPATHLEN - 10) {
1793         jio_snprintf(class_list_path_str + class_list_path_len,
1794                      sizeof(class_list_path_str) - class_list_path_len,
1795                      "%sclasslist", os::file_separator());
1796       }
1797       class_list_path = class_list_path_str;
1798     } else {
1799       class_list_path = SharedClassListFile;
1800     }
1801 
1802     log_info(cds)("Loading classes to share ...");
1803     _has_error_classes = false;
1804     int class_count = preload_classes(class_list_path, THREAD);
1805     if (ExtraSharedClassListFile) {
1806       class_count += preload_classes(ExtraSharedClassListFile, THREAD);
1807     }
1808     log_info(cds)("Loading classes to share: done.");
1809 
1810     log_info(cds)("Shared spaces: preloaded %d classes", class_count);
1811 
1812     if (SharedArchiveConfigFile) {
1813       log_info(cds)("Reading extra data from %s ...", SharedArchiveConfigFile);
1814       read_extra_data(SharedArchiveConfigFile, THREAD);
1815     }
1816     log_info(cds)("Reading extra data: done.");
1817 
1818     HeapShared::init_subgraph_entry_fields(THREAD);
1819 
1820     // Rewrite and link classes
1821     log_info(cds)("Rewriting and linking classes ...");
1822 
1823     // Link any classes which got missed. This would happen if we have loaded classes that
1824     // were not explicitly specified in the classlist. E.g., if an interface implemented by class K
1825     // fails verification, all other interfaces that were not specified in the classlist but
1826     // are implemented by K are not verified.
1827     link_and_cleanup_shared_classes(CATCH);
1828     log_info(cds)("Rewriting and linking classes: done");
1829 
1830     if (HeapShared::is_heap_object_archiving_allowed()) {
1831       // Avoid fragmentation while archiving heap objects.
1832       Universe::heap()->soft_ref_policy()->set_should_clear_all_soft_refs(true);
1833       Universe::heap()->collect(GCCause::_archive_time_gc);
1834       Universe::heap()->soft_ref_policy()->set_should_clear_all_soft_refs(false);
1835     }
1836

1838     VMThread::execute(&op);
1839   }

1843 int MetaspaceShared::preload_classes(const char* class_list_path, TRAPS) {
1844   ClassListParser parser(class_list_path);
1845   int class_count = 0;
1846 
1847   while (parser.parse_one_line()) {
1848     Klass* klass = parser.load_current_class(THREAD);
1849     if (HAS_PENDING_EXCEPTION) {
1850       if (klass == NULL &&
1851           (PENDING_EXCEPTION->klass()->name() == vmSymbols::java_lang_ClassNotFoundException())) {
1852         // print a warning only when the pending exception is class not found
1853         log_warning(cds)("Preload Warning: Cannot find %s", parser.current_class_name());
1854       }
1855       CLEAR_PENDING_EXCEPTION;
1856     }
1857     if (klass != NULL) {
1858       if (log_is_enabled(Trace, cds)) {
1859         ResourceMark rm(THREAD);
1860         log_trace(cds)("Shared spaces preloaded: %s", klass->external_name());
1861       }
1862 
1863       if (klass->is_instance_klass()) {
1864         InstanceKlass* ik = InstanceKlass::cast(klass);
1865 
1866         // Link the class to cause the bytecodes to be rewritten and the
1867         // cpcache to be created. The linking is done as soon as classes
1868         // are loaded in order that the related data structures (klass and
1869         // cpCache) are located together.
1870         try_link_class(ik, THREAD);
1871         guarantee(!HAS_PENDING_EXCEPTION, "exception in link_class");
1872       }
1873 
1874       class_count++;
1875     }
1876   }
1877 
1878   return class_count;
1879 }

1881 // Returns true if the class's status has changed
1882 bool MetaspaceShared::try_link_class(InstanceKlass* ik, TRAPS) {
1883   Arguments::assert_is_dumping_archive();
1884   if (ik->init_state() < InstanceKlass::linked &&
1885       !SystemDictionaryShared::has_class_failed_verification(ik)) {
1886     bool saved = BytecodeVerificationLocal;
1887     if (ik->is_shared_unregistered_class() && ik->class_loader() == NULL) {
1888       // The verification decision is based on BytecodeVerificationRemote
1889       // for non-system classes. Since we are using the NULL classloader
1890       // to load non-system classes for customized class loaders during dumping,
1891       // we need to temporarily change BytecodeVerificationLocal to be the same as
1892       // BytecodeVerificationRemote. Note this can cause the parent system
1893       // classes also being verified. The extra overhead is acceptable during
1894       // dumping.
1895       BytecodeVerificationLocal = BytecodeVerificationRemote;
1896     }
1897     ik->link_class(THREAD);
1898     if (HAS_PENDING_EXCEPTION) {
1899       ResourceMark rm(THREAD);
1900       log_warning(cds)("Preload Warning: Verification failed for %s",
1901                     ik->external_name());
1902       CLEAR_PENDING_EXCEPTION;
1903       SystemDictionaryShared::set_class_has_failed_verification(ik);
1904       _has_error_classes = true;
1905     }
1906     BytecodeVerificationLocal = saved;
1907     return true;
1908   } else {
1909     return false;
1910   }
1911 }

2106 // use_requested_addr:
2107 //  true  = map at FileMapHeader::_requested_base_address
2108 //  false = map at an alternative address picked by OS.
2109 MapArchiveResult MetaspaceShared::map_archives(FileMapInfo* static_mapinfo, FileMapInfo* dynamic_mapinfo,
2110                                                bool use_requested_addr) {
2111   PRODUCT_ONLY(if (ArchiveRelocationMode == 1 && use_requested_addr) {
2112       // For product build only -- this is for benchmarking the cost of doing relocation.
2113       // For debug builds, the check is done in FileMapInfo::map_regions for better test coverage.
2114       log_info(cds)("ArchiveRelocationMode == 1: always map archive(s) at an alternative address");
2115       return MAP_ARCHIVE_MMAP_FAILURE;
2116     });
2117 
2118   if (ArchiveRelocationMode == 2 && !use_requested_addr) {
2119     log_info(cds)("ArchiveRelocationMode == 2: never map archive(s) at an alternative address");
2120     return MAP_ARCHIVE_MMAP_FAILURE;
2121   };
2122 
2123   if (dynamic_mapinfo != NULL) {
2124     // Ensure that the OS won't be able to allocate new memory spaces between the two
2125     // archives, or else it would mess up the simple comparision in MetaspaceObj::is_shared().
2126     assert(static_mapinfo->mapping_end_offset() == dynamic_mapinfo->mapping_base_offset(), "no gap");
2127   }
2128 
2129   ReservedSpace main_rs, archive_space_rs, class_space_rs;
2130   MapArchiveResult result = MAP_ARCHIVE_OTHER_FAILURE;
2131   char* mapped_base_address = reserve_address_space_for_archives(static_mapinfo, dynamic_mapinfo,
2132                                                                  use_requested_addr, main_rs, archive_space_rs,
2133                                                                  class_space_rs);
2134   if (mapped_base_address == NULL) {
2135     result = MAP_ARCHIVE_MMAP_FAILURE;
2136   } else {
2137     log_debug(cds)("Reserved archive_space_rs     [" INTPTR_FORMAT " - " INTPTR_FORMAT "] (" SIZE_FORMAT ") bytes",
2138                    p2i(archive_space_rs.base()), p2i(archive_space_rs.end()), archive_space_rs.size());
2139     log_debug(cds)("Reserved class_space_rs [" INTPTR_FORMAT " - " INTPTR_FORMAT "] (" SIZE_FORMAT ") bytes",
2140                    p2i(class_space_rs.base()), p2i(class_space_rs.end()), class_space_rs.size());
2141     MapArchiveResult static_result = map_archive(static_mapinfo, mapped_base_address, archive_space_rs);
2142     MapArchiveResult dynamic_result = (static_result == MAP_ARCHIVE_SUCCESS) ?
2143                                      map_archive(dynamic_mapinfo, mapped_base_address, archive_space_rs) : MAP_ARCHIVE_OTHER_FAILURE;
2144 
2145     DEBUG_ONLY(if (ArchiveRelocationMode == 1 && use_requested_addr) {
2146       // This is for simulating mmap failures at the requested address. In debug builds, we do it
2147       // here (after all archives have possibly been mapped), so we can thoroughly test the code for
2148       // failure handling (releasing all allocated resource, etc).
2149       log_info(cds)("ArchiveRelocationMode == 1: always map archive(s) at an alternative address");
2150       if (static_result == MAP_ARCHIVE_SUCCESS) {
2151         static_result = MAP_ARCHIVE_MMAP_FAILURE;
2152       }
2153       if (dynamic_result == MAP_ARCHIVE_SUCCESS) {
2154         dynamic_result = MAP_ARCHIVE_MMAP_FAILURE;
2155       }
2156     });
2157 
2158     if (static_result == MAP_ARCHIVE_SUCCESS) {
2159       if (dynamic_result == MAP_ARCHIVE_SUCCESS) {
2160         result = MAP_ARCHIVE_SUCCESS;
2161       } else if (dynamic_result == MAP_ARCHIVE_OTHER_FAILURE) {
2162         assert(dynamic_mapinfo != NULL && !dynamic_mapinfo->is_mapped(), "must have failed");
2163         // No need to retry mapping the dynamic archive again, as it will never succeed
2164         // (bad file, etc) -- just keep the base archive.
2165         log_warning(cds, dynamic)("Unable to use shared archive. The top archive failed to load: %s",
2166                                   dynamic_mapinfo->full_path());
2167         result = MAP_ARCHIVE_SUCCESS;
2168         // TODO, we can give the unused space for the dynamic archive to class_space_rs, but there's no
2169         // easy API to do that right now.
2170       } else {
2171         result = MAP_ARCHIVE_MMAP_FAILURE;
2172       }
2173     } else if (static_result == MAP_ARCHIVE_OTHER_FAILURE) {
2174       result = MAP_ARCHIVE_OTHER_FAILURE;
2175     } else {
2176       result = MAP_ARCHIVE_MMAP_FAILURE;
2177     }
2178   }
2179 
2180   if (result == MAP_ARCHIVE_SUCCESS) {
2181     if (!main_rs.is_reserved() && class_space_rs.is_reserved()) {
2182       MemTracker::record_virtual_memory_type((address)class_space_rs.base(), mtClass);
2183     }
2184     SharedBaseAddress = (size_t)mapped_base_address;
2185     LP64_ONLY({
2186         if (Metaspace::using_class_space()) {
2187           assert(class_space_rs.is_reserved(), "must be");
2188           char* cds_base = static_mapinfo->mapped_base();
2189           Metaspace::allocate_metaspace_compressed_klass_ptrs(class_space_rs, NULL, (address)cds_base);
2190           // map_heap_regions() compares the current narrow oop and klass encodings
2191           // with the archived ones, so it must be done after all encodings are determined.
2192           static_mapinfo->map_heap_regions();
2193           CompressedKlassPointers::set_range(CompressedClassSpaceSize);
2194         }
2195       });
2196   } else {
2197     unmap_archive(static_mapinfo);
2198     unmap_archive(dynamic_mapinfo);
2199     release_reserved_spaces(main_rs, archive_space_rs, class_space_rs);
2200   }
2201 
2202   return result;
2203 }

2205 char* MetaspaceShared::reserve_address_space_for_archives(FileMapInfo* static_mapinfo,
2206                                                           FileMapInfo* dynamic_mapinfo,
2207                                                           bool use_requested_addr,
2208                                                           ReservedSpace& main_rs,
2209                                                           ReservedSpace& archive_space_rs,
2210                                                           ReservedSpace& class_space_rs) {
2211   const bool use_klass_space = NOT_LP64(false) LP64_ONLY(Metaspace::using_class_space());
2212   const size_t class_space_size = NOT_LP64(0) LP64_ONLY(Metaspace::compressed_class_space_size());
2213 
2214   if (use_klass_space) {
2215     assert(class_space_size > 0, "CompressedClassSpaceSize must have been validated");
2216   }
2217   if (use_requested_addr && !is_aligned(static_mapinfo->requested_base_address(), reserved_space_alignment())) {
2218     return NULL;
2219   }
2220 
2221   // Size and requested location of the archive_space_rs (for both static and dynamic archives)
2222   size_t base_offset = static_mapinfo->mapping_base_offset();
2223   size_t end_offset  = (dynamic_mapinfo == NULL) ? static_mapinfo->mapping_end_offset() : dynamic_mapinfo->mapping_end_offset();
2224   assert(base_offset == 0, "must be");
2225   assert(is_aligned(end_offset,  os::vm_allocation_granularity()), "must be");
2226   assert(is_aligned(base_offset, os::vm_allocation_granularity()), "must be");
2227 
2228   // In case reserved_space_alignment() != os::vm_allocation_granularity()
2229   assert((size_t)os::vm_allocation_granularity() <= reserved_space_alignment(), "must be");
2230   end_offset = align_up(end_offset, reserved_space_alignment());
2231 
2232   size_t archive_space_size = end_offset - base_offset;
2233 
2234   // Special handling for Windows because it cannot mmap into a reserved space:
2235   //    use_requested_addr: We just map each region individually, and give up if any one of them fails.
2236   //   !use_requested_addr: We reserve the space first, and then os::read in all the regions (instead of mmap).
2237   //                        We're going to patch all the pointers anyway so there's no benefit for mmap.
2238 
2239   if (use_requested_addr) {
2240     char* archive_space_base = static_mapinfo->requested_base_address() + base_offset;
2241     char* archive_space_end  = archive_space_base + archive_space_size;
2242     if (!MetaspaceShared::use_windows_memory_mapping()) {
2243       archive_space_rs = reserve_shared_space(archive_space_size, archive_space_base);
2244       if (!archive_space_rs.is_reserved()) {
2245         return NULL;
2246       }
2247     }
2248     if (use_klass_space) {
2249       // Make sure we can map the klass space immediately following the archive_space space
2250       // Don't call reserve_shared_space here as that may try to enforce platform-specific
2251       // alignment rules which only apply to the archive base address
2252       char* class_space_base = archive_space_end;
2253       class_space_rs = ReservedSpace(class_space_size, reserved_space_alignment(),
2254                                      false /* large_pages */, class_space_base);
2255       if (!class_space_rs.is_reserved()) {
2256         return NULL;
2257       }
2258     }
2259     return static_mapinfo->requested_base_address();
2260   } else {
2261     if (use_klass_space) {
2262       main_rs = reserve_shared_space(archive_space_size + class_space_size);
2263       if (main_rs.is_reserved()) {
2264         archive_space_rs = main_rs.first_part(archive_space_size, reserved_space_alignment(), /*split=*/true);
2265         class_space_rs = main_rs.last_part(archive_space_size);
2266       }
2267     } else {
2268       main_rs = reserve_shared_space(archive_space_size);
2269       archive_space_rs = main_rs;
2270     }
2271     if (archive_space_rs.is_reserved()) {
2272       return archive_space_rs.base();
2273     } else {
2274       return NULL;
2275     }
2276   }
2277 }

2305   assert(UseSharedSpaces, "must be runtime");

2308   }
2309

2311 
2312   if (mapinfo->alignment() != (size_t)os::vm_allocation_granularity()) {
2313     log_error(cds)("Unable to map CDS archive -- os::vm_allocation_granularity() expected: " SIZE_FORMAT

2319     mapinfo->map_regions(archive_regions, archive_regions_count, mapped_base_address, rs);
