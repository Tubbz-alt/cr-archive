  47 // These masks are used to provide 128-bit aligned bitmasks to the XMM
  48 // instructions, to allow sign-masking or sign-bit flipping.  They allow
  49 // fast versions of NegF/NegD and AbsF/AbsD.
  50 
  51 // Note: 'double' and 'long long' have 32-bits alignment on x86.
  52 static jlong* double_quadword(jlong *adr, jlong lo, jlong hi) {
  53   // Use the expression (adr)&(~0xF) to provide 128-bits aligned address
  54   // of 128-bits operands for SSE instructions.
  55   jlong *operand = (jlong*)(((intptr_t)adr) & ((intptr_t)(~0xF)));
  56   // Store the value to a 128-bits operand.
  57   operand[0] = lo;
  58   operand[1] = hi;
  59   return operand;
  60 }
  61 
  62 // Buffer for 128-bits masks used by SSE instructions.
  63 static jlong fp_signmask_pool[(4+1)*2]; // 4*128bits(data) + 128bits(alignment)
  64 
  65 // Static initialization during VM startup.
  66 static jlong *float_signmask_pool  = double_quadword(&fp_signmask_pool[1*2],         CONST64(0x7FFFFFFF7FFFFFFF),         CONST64(0x7FFFFFFF7FFFFFFF));
  67 static jlong *double_signmask_pool = double_quadword(&fp_signmask_pool[2*2],         CONST64(0x7FFFFFFFFFFFFFFF),         CONST64(0x7FFFFFFFFFFFFFFF));
  68 static jlong *float_signflip_pool  = double_quadword(&fp_signmask_pool[3*2], (jlong)UCONST64(0x8000000080000000), (jlong)UCONST64(0x8000000080000000));
  69 static jlong *double_signflip_pool = double_quadword(&fp_signmask_pool[4*2], (jlong)UCONST64(0x8000000000000000), (jlong)UCONST64(0x8000000000000000));

 186 void LIR_Assembler::push(LIR_Opr opr) {
 187   if (opr->is_single_cpu()) {
 188     __ push_reg(opr->as_register());
 189   } else if (opr->is_double_cpu()) {
 190     NOT_LP64(__ push_reg(opr->as_register_hi()));
 191     __ push_reg(opr->as_register_lo());
 192   } else if (opr->is_stack()) {
 193     __ push_addr(frame_map()->address_for_slot(opr->single_stack_ix()));
 194   } else if (opr->is_constant()) {
 195     LIR_Const* const_opr = opr->as_constant_ptr();
 196     if (const_opr->type() == T_OBJECT || const_opr->type() == T_VALUETYPE) {
 197       __ push_oop(const_opr->as_jobject());
 198     } else if (const_opr->type() == T_INT) {
 199       __ push_jint(const_opr->as_jint());
 200     } else {
 201       ShouldNotReachHere();
 202     }
 203 
 204   } else {
 205     ShouldNotReachHere();
 206   }
 207 }

 227 Address LIR_Assembler::as_Address(LIR_Address* addr, Register tmp) {
 228   if (addr->base()->is_illegal()) {
 229     assert(addr->index()->is_illegal(), "must be illegal too");
 230     AddressLiteral laddr((address)addr->disp(), relocInfo::none);
 231     if (! __ reachable(laddr)) {
 232       __ movptr(tmp, laddr.addr());
 233       Address res(tmp, 0);
 234       return res;
 235     } else {
 236       return __ as_Address(laddr);
 237     }
 238   }
 239 
 240   Register base = addr->base()->as_pointer_register();
 241 
 242   if (addr->index()->is_illegal()) {
 243     return Address( base, addr->disp());
 244   } else if (addr->index()->is_cpu_register()) {
 245     Register index = addr->index()->as_pointer_register();
 246     return Address(base, index, (Address::ScaleFactor) addr->scale(), addr->disp());
 247   } else if (addr->index()->is_constant()) {
 248     intptr_t addr_offset = (addr->index()->as_constant_ptr()->as_jint() << addr->scale()) + addr->disp();
 249     assert(Assembler::is_simm32(addr_offset), "must be");
 250 
 251     return Address(base, addr_offset);
 252   } else {
 253     Unimplemented();
 254     return Address();
 255   }
 256 }

 270 void LIR_Assembler::osr_entry() {
 271   offsets()->set_value(CodeOffsets::OSR_Entry, code_offset());
 272   BlockBegin* osr_entry = compilation()->hir()->osr_entry();
 273   ValueStack* entry_state = osr_entry->state();
 274   int number_of_locks = entry_state->locks_size();
 275 
 276   // we jump here if osr happens with the interpreter
 277   // state set up to continue at the beginning of the
 278   // loop that triggered osr - in particular, we have
 279   // the following registers setup:
 280   //
 281   // rcx: osr buffer
 282   //
 283 
 284   // build frame
 285   ciMethod* m = compilation()->method();
 286   __ build_frame(initial_frame_size_in_bytes(), bang_size_in_bytes());
 287 
 288   // OSR buffer is
 289   //
 290   // locals[nlocals-1..0]
 291   // monitors[0..number_of_locks]
 292   //
 293   // locals is a direct copy of the interpreter frame so in the osr buffer
 294   // so first slot in the local array is the last local from the interpreter
 295   // and last slot is local[0] (receiver) from the interpreter
 296   //
 297   // Similarly with locks. The first lock slot in the osr buffer is the nth lock
 298   // from the interpreter frame, the nth lock slot in the osr buffer is 0th lock
 299   // in the interpreter frame (the method lock if a sync method)
 300 
 301   // Initialize monitors in the compiled activation.
 302   //   rcx: pointer to osr buffer
 303   //
 304   // All other registers are dead at this point and the locals will be
 305   // copied into place by code emitted in the IR.
 306 
 307   Register OSR_buf = osrBufferPointer()->as_pointer_register();
 308   { assert(frame::interpreter_frame_monitor_size() == BasicObjectLock::size(), "adjust code below");
 309     int monitor_offset = BytesPerWord * method()->max_locals() +
 310       (BasicObjectLock::size() * BytesPerWord) * (number_of_locks - 1);
 311     // SharedRuntime::OSR_migration_begin() packs BasicObjectLocks in
 312     // the OSR buffer using 2 word entries: first the lock and then
 313     // the oop.
 314     for (int i = 0; i < number_of_locks; i++) {
 315       int slot_offset = monitor_offset - ((i * 2) * BytesPerWord);
 316 #ifdef ASSERT
 317       // verify the interpreter's monitor has a non-null object
 318       {
 319         Label L;
 320         __ cmpptr(Address(OSR_buf, slot_offset + 1*BytesPerWord), (int32_t)NULL_WORD);
 321         __ jcc(Assembler::notZero, L);
 322         __ stop("locked object is NULL");
 323         __ bind(L);
 324       }
 325 #endif
 326       __ movptr(rbx, Address(OSR_buf, slot_offset + 0));
 327       __ movptr(frame_map()->address_for_monitor_lock(i), rbx);
 328       __ movptr(rbx, Address(OSR_buf, slot_offset + 1*BytesPerWord));
 329       __ movptr(frame_map()->address_for_monitor_object(i), rbx);
 330     }
 331   }
 332 }

 335 // inline cache check; done before the frame is built.
 336 int LIR_Assembler::check_icache() {
 337   Register receiver = FrameMap::receiver_opr->as_register();
 338   Register ic_klass = IC_Klass;
 339   const int ic_cmp_size = LP64_ONLY(10) NOT_LP64(9);
 340   const bool do_post_padding = VerifyOops || UseCompressedClassPointers;
 341   if (!do_post_padding) {
 342     // insert some nops so that the verified entry point is aligned on CodeEntryAlignment
 343     __ align(CodeEntryAlignment, __ offset() + ic_cmp_size);
 344   }
 345   int offset = __ offset();
 346   __ inline_cache_check(receiver, IC_Klass);
 347   assert(__ offset() % CodeEntryAlignment == 0 || do_post_padding, "alignment must be correct");
 348   if (do_post_padding) {
 349     // force alignment after the cache check.
 350     // It's been verified to be aligned if !VerifyOops
 351     __ align(CodeEntryAlignment);
 352   }
 353   return offset;
 354 }

 356 void LIR_Assembler::clinit_barrier(ciMethod* method) {
 357   assert(VM_Version::supports_fast_class_init_checks(), "sanity");
 358   assert(!method->holder()->is_not_initialized(), "initialization should have been started");
 359 
 360   Label L_skip_barrier;
 361   Register klass = rscratch1;
 362   Register thread = LP64_ONLY( r15_thread ) NOT_LP64( noreg );
 363   assert(thread != noreg, "x86_32 not implemented");
 364 
 365   __ mov_metadata(klass, method->holder()->constant_encoding());
 366   __ clinit_barrier(klass, thread, &L_skip_barrier /*L_fast_path*/);
 367 
 368   __ jump(RuntimeAddress(SharedRuntime::get_handle_wrong_method_stub()));
 369 
 370   __ bind(L_skip_barrier);
 371 }

 398 int LIR_Assembler::emit_exception_handler() {
 399   // if the last instruction is a call (typically to do a throw which
 400   // is coming at the end after block reordering) the return address
 401   // must still point into the code area in order to avoid assertion
 402   // failures when searching for the corresponding bci => add a nop
 403   // (was bug 5/14/1999 - gri)
 404   __ nop();
 405 
 406   // generate code for exception handler
 407   address handler_base = __ start_a_stub(exception_handler_size());
 408   if (handler_base == NULL) {
 409     // not enough space left for the handler
 410     bailout("exception handler overflow");
 411     return -1;
 412   }
 413 
 414   int offset = code_offset();
 415 
 416   // the exception oop and pc are in rax, and rdx
 417   // no other registers need to be preserved, so invalidate them
 418   __ invalidate_registers(false, true, true, false, true, true);
 419 
 420   // check that there is really an exception
 421   __ verify_not_null_oop(rax);
 422 
 423   // search an exception handler (rax: exception oop, rdx: throwing pc)
 424   __ call(RuntimeAddress(Runtime1::entry_for(Runtime1::handle_exception_from_callee_id)));
 425   __ should_not_reach_here();
 426   guarantee(code_offset() - offset <= exception_handler_size(), "overflow");
 427   __ end_a_stub();
 428 
 429   return offset;
 430 }

 433 // Emit the code to remove the frame from the stack in the exception
 434 // unwind path.
 435 int LIR_Assembler::emit_unwind_handler() {
 436 #ifndef PRODUCT
 437   if (CommentedAssembly) {
 438     _masm->block_comment("Unwind handler");
 439   }
 440 #endif
 441 
 442   int offset = code_offset();
 443 
 444   // Fetch the exception from TLS and clear out exception related thread state
 445   Register thread = NOT_LP64(rsi) LP64_ONLY(r15_thread);
 446   NOT_LP64(__ get_thread(rsi));
 447   __ movptr(rax, Address(thread, JavaThread::exception_oop_offset()));
 448   __ movptr(Address(thread, JavaThread::exception_oop_offset()), (intptr_t)NULL_WORD);
 449   __ movptr(Address(thread, JavaThread::exception_pc_offset()), (intptr_t)NULL_WORD);
 450 
 451   __ bind(_unwind_handler_entry);
 452   __ verify_not_null_oop(rax);
 453   if (method()->is_synchronized() || compilation()->env()->dtrace_method_probes()) {
 454     __ mov(rbx, rax);  // Preserve the exception (rbx is always callee-saved)
 455   }
 456 
 457   // Preform needed unlocking
 458   MonitorExitStub* stub = NULL;
 459   if (method()->is_synchronized()) {
 460     monitor_address(0, FrameMap::rax_opr);
 461     stub = new MonitorExitStub(FrameMap::rax_opr, true, 0);
 462     __ unlock_object(rdi, rsi, rax, *stub->entry());
 463     __ bind(*stub->continuation());
 464   }
 465 
 466   if (compilation()->env()->dtrace_method_probes()) {
 467 #ifdef _LP64
 468     __ mov(rdi, r15_thread);
 469     __ mov_metadata(rsi, method()->constant_encoding());
 470 #else
 471     __ get_thread(rax);
 472     __ movptr(Address(rsp, 0), rax);
 473     __ mov_metadata(Address(rsp, sizeof(void*)), method()->constant_encoding());
 474 #endif
 475     __ call(RuntimeAddress(CAST_FROM_FN_PTR(address, SharedRuntime::dtrace_method_exit)));
 476   }
 477 
 478   if (method()->is_synchronized() || compilation()->env()->dtrace_method_probes()) {
 479     __ mov(rax, rbx);  // Restore the exception
 480   }
 481 
 482   // remove the activation and dispatch to the unwind handler
 483   int initial_framesize = initial_frame_size_in_bytes();
 484   __ remove_frame(initial_framesize, needs_stack_repair(), initial_framesize - wordSize);
 485   __ jump(RuntimeAddress(Runtime1::entry_for(Runtime1::unwind_exception_id)));
 486 
 487   // Emit the slow path assembly
 488   if (stub != NULL) {
 489     stub->emit_code(this);
 490   }
 491 
 492   return offset;
 493 }

 498   // is coming at the end after block reordering) the return address
 499   // must still point into the code area in order to avoid assertion
 500   // failures when searching for the corresponding bci => add a nop

 518   __ end_a_stub();
 519 
 520   return offset;

 524 void LIR_Assembler::return_op(LIR_Opr result) {
 525   assert(result->is_illegal() || !result->is_single_cpu() || result->as_register() == rax, "word returns are in rax,");
 526   if (!result->is_illegal() && result->is_float_kind() && !result->is_xmm_register()) {
 527     assert(result->fpu() == 0, "result must already be on TOS");
 528   }
 529 
 530   ciMethod* method = compilation()->method();
 531   if (ValueTypeReturnedAsFields && method->signature()->returns_never_null()) {
 532     ciType* return_type = method->return_type();
 533     if (return_type->is_valuetype()) {
 534       ciValueKlass* vk = return_type->as_value_klass();
 535       if (vk->can_be_returned_as_fields()) {
 536 #ifndef _LP64
 537         Unimplemented();
 538 #else
 539         address unpack_handler = vk->unpack_handler();
 540         assert(unpack_handler != NULL, "must be");
 541         __ call(RuntimeAddress(unpack_handler));
 542         // At this point, rax points to the value object (for interpreter or C1 caller).
 543         // The fields of the object are copied into registers (for C2 caller).
 544 #endif
 545       }
 546     }
 547   }
 548 
 549   // Pop the stack before the safepoint code
 550   int initial_framesize = initial_frame_size_in_bytes();
 551   __ remove_frame(initial_framesize, needs_stack_repair(), initial_framesize - wordSize);
 552 
 553   if (StackReservedPages > 0 && compilation()->has_reserved_stack_access()) {
 554     __ reserved_stack_check();
 555   }
 556 
 557   bool result_is_oop = result->is_valid() ? result->is_oop() : false;
 558 
 559   // Note: we do not need to round double result; float result has the right precision
 560   // the poll sets the condition code, but no data registers
 561 
 562 #ifdef _LP64
 563   const Register poll_addr = rscratch1;
 564   __ movptr(poll_addr, Address(r15_thread, Thread::polling_page_offset()));
 565 #else
 566   const Register poll_addr = rbx;
 567   assert(FrameMap::is_caller_save_register(poll_addr), "will overwrite");
 568   __ get_thread(poll_addr);
 569   __ movptr(poll_addr, Address(poll_addr, Thread::polling_page_offset()));
 570 #endif
 571   __ relocate(relocInfo::poll_return_type);
 572   __ testl(rax, Address(poll_addr, 0));
 573   __ ret(0);
 574 }

 581 int LIR_Assembler::safepoint_poll(LIR_Opr tmp, CodeEmitInfo* info) {
 582   guarantee(info != NULL, "Shouldn't be NULL");
 583   int offset = __ offset();
 584 #ifdef _LP64
 585   const Register poll_addr = rscratch1;
 586   __ movptr(poll_addr, Address(r15_thread, Thread::polling_page_offset()));
 587 #else
 588   assert(tmp->is_cpu_register(), "needed");
 589   const Register poll_addr = tmp->as_register();
 590   __ get_thread(poll_addr);
 591   __ movptr(poll_addr, Address(poll_addr, in_bytes(Thread::polling_page_offset())));
 592 #endif
 593   add_debug_info_for_branch(info);
 594   __ relocate(relocInfo::poll_type);
 595   address pre_pc = __ pc();
 596   __ testl(rax, Address(poll_addr, 0));
 597   address post_pc = __ pc();
 598   guarantee(pointer_delta(post_pc, pre_pc, 1) == 2 LP64_ONLY(+1), "must be exact length");
 599   return offset;
 600 }

 612 void LIR_Assembler::const2reg(LIR_Opr src, LIR_Opr dest, LIR_PatchCode patch_code, CodeEmitInfo* info) {
 613   assert(src->is_constant(), "should not call otherwise");
 614   assert(dest->is_register(), "should not call otherwise");
 615   LIR_Const* c = src->as_constant_ptr();
 616 
 617   switch (c->type()) {
 618     case T_INT: {
 619       assert(patch_code == lir_patch_none, "no patching handled here");
 620       __ movl(dest->as_register(), c->as_jint());
 621       break;
 622     }
 623 
 624     case T_ADDRESS: {
 625       assert(patch_code == lir_patch_none, "no patching handled here");
 626       __ movptr(dest->as_register(), c->as_jint());
 627       break;
 628     }
 629 
 630     case T_LONG: {
 631       assert(patch_code == lir_patch_none, "no patching handled here");
 632 #ifdef _LP64
 633       __ movptr(dest->as_register_lo(), (intptr_t)c->as_jlong());
 634 #else
 635       __ movptr(dest->as_register_lo(), c->as_jint_lo());
 636       __ movptr(dest->as_register_hi(), c->as_jint_hi());
 637 #endif // _LP64
 638       break;
 639     }
 640 
 641     case T_VALUETYPE: // Fall through
 642     case T_OBJECT: {
 643       if (patch_code != lir_patch_none) {
 644         jobject2reg_with_patching(dest->as_register(), info);
 645       } else {
 646         __ movoop(dest->as_register(), c->as_jobject());
 647       }
 648       break;
 649     }
 650 
 651     case T_METADATA: {
 652       if (patch_code != lir_patch_none) {
 653         klass2reg_with_patching(dest->as_register(), info);
 654       } else {
 655         __ mov_metadata(dest->as_register(), c->as_metadata());
 656       }
 657       break;
 658     }
 659 
 660     case T_FLOAT: {
 661       if (dest->is_single_xmm()) {
 662         if (LP64_ONLY(UseAVX <= 2 &&) c->is_zero_float()) {
 663           __ xorps(dest->as_xmm_float_reg(), dest->as_xmm_float_reg());
 664         } else {
 665           __ movflt(dest->as_xmm_float_reg(),
 666                    InternalAddress(float_constant(c->as_jfloat())));
 667         }
 668       } else {
 669 #ifndef _LP64
 670         assert(dest->is_single_fpu(), "must be");
 671         assert(dest->fpu_regnr() == 0, "dest must be TOS");
 672         if (c->is_zero_float()) {
 673           __ fldz();
 674         } else if (c->is_one_float()) {
 675           __ fld1();
 676         } else {
 677           __ fld_s (InternalAddress(float_constant(c->as_jfloat())));
 678         }
 679 #else
 680         ShouldNotReachHere();
 681 #endif // !_LP64
 682       }
 683       break;
 684     }
 685 
 686     case T_DOUBLE: {
 687       if (dest->is_double_xmm()) {
 688         if (LP64_ONLY(UseAVX <= 2 &&) c->is_zero_double()) {
 689           __ xorpd(dest->as_xmm_double_reg(), dest->as_xmm_double_reg());
 690         } else {
 691           __ movdbl(dest->as_xmm_double_reg(),
 692                     InternalAddress(double_constant(c->as_jdouble())));
 693         }
 694       } else {
 695 #ifndef _LP64
 696         assert(dest->is_double_fpu(), "must be");
 697         assert(dest->fpu_regnrLo() == 0, "dest must be TOS");
 698         if (c->is_zero_double()) {
 699           __ fldz();
 700         } else if (c->is_one_double()) {
 701           __ fld1();
 702         } else {
 703           __ fld_d (InternalAddress(double_constant(c->as_jdouble())));
 704         }
 705 #else
 706         ShouldNotReachHere();
 707 #endif // !_LP64
 708       }
 709       break;
 710     }
 711 
 712     default:
 713       ShouldNotReachHere();
 714   }
 715 }

 755 void LIR_Assembler::const2mem(LIR_Opr src, LIR_Opr dest, BasicType type, CodeEmitInfo* info, bool wide) {
 756   assert(src->is_constant(), "should not call otherwise");
 757   assert(dest->is_address(), "should not call otherwise");
 758   LIR_Const* c = src->as_constant_ptr();
 759   LIR_Address* addr = dest->as_address_ptr();
 760 
 761   int null_check_here = code_offset();
 762   switch (type) {
 763     case T_INT:    // fall through
 764     case T_FLOAT:
 765       __ movl(as_Address(addr), c->as_jint_bits());
 766       break;
 767 
 768     case T_ADDRESS:
 769       __ movptr(as_Address(addr), c->as_jint_bits());
 770       break;
 771 
 772     case T_VALUETYPE: // fall through
 773     case T_OBJECT:  // fall through
 774     case T_ARRAY:
 775       if (c->as_jobject() == NULL) {
 776         if (UseCompressedOops && !wide) {
 777           __ movl(as_Address(addr), (int32_t)NULL_WORD);
 778         } else {
 779 #ifdef _LP64
 780           __ xorptr(rscratch1, rscratch1);
 781           null_check_here = code_offset();
 782           __ movptr(as_Address(addr), rscratch1);
 783 #else
 784           __ movptr(as_Address(addr), NULL_WORD);
 785 #endif
 786         }
 787       } else {
 788         if (is_literal_address(addr)) {
 789           ShouldNotReachHere();
 790           __ movoop(as_Address(addr, noreg), c->as_jobject());
 791         } else {
 792 #ifdef _LP64
 793           __ movoop(rscratch1, c->as_jobject());
 794           if (UseCompressedOops && !wide) {
 795             __ encode_heap_oop(rscratch1);
 796             null_check_here = code_offset();
 797             __ movl(as_Address_lo(addr), rscratch1);
 798           } else {
 799             null_check_here = code_offset();
 800             __ movptr(as_Address_lo(addr), rscratch1);
 801           }
 802 #else
 803           __ movoop(as_Address(addr), c->as_jobject());
 804 #endif
 805         }
 806       }
 807       break;
 808 
 809     case T_LONG:    // fall through
 810     case T_DOUBLE:
 811 #ifdef _LP64
 812       if (is_literal_address(addr)) {
 813         ShouldNotReachHere();
 814         __ movptr(as_Address(addr, r15_thread), (intptr_t)c->as_jlong_bits());
 815       } else {
 816         __ movptr(r10, (intptr_t)c->as_jlong_bits());
 817         null_check_here = code_offset();
 818         __ movptr(as_Address_lo(addr), r10);
 819       }
 820 #else
 821       // Always reachable in 32bit so this doesn't produce useless move literal
 822       __ movptr(as_Address_hi(addr), c->as_jint_hi_bits());
 823       __ movptr(as_Address_lo(addr), c->as_jint_lo_bits());
 824 #endif // _LP64
 825       break;
 826 
 827     case T_BOOLEAN: // fall through
 828     case T_BYTE:
 829       __ movb(as_Address(addr), c->as_jint() & 0xFF);
 830       break;
 831 
 832     case T_CHAR:    // fall through
 833     case T_SHORT:
 834       __ movw(as_Address(addr), c->as_jint() & 0xFFFF);
 835       break;
 836 
 837     default:
 838       ShouldNotReachHere();
 839   };
 840 
 841   if (info != NULL) {
 842     add_debug_info_for_null_check(null_check_here, info);
 843   }
 844 }

 847 void LIR_Assembler::reg2reg(LIR_Opr src, LIR_Opr dest) {
 848   assert(src->is_register(), "should not call otherwise");
 849   assert(dest->is_register(), "should not call otherwise");
 850 
 851   // move between cpu-registers
 852   if (dest->is_single_cpu()) {
 853 #ifdef _LP64
 854     if (src->type() == T_LONG) {
 855       // Can do LONG -> OBJECT
 856       move_regs(src->as_register_lo(), dest->as_register());
 857       return;
 858     }
 859 #endif
 860     assert(src->is_single_cpu(), "must match");
 861     if (src->type() == T_OBJECT || src->type() == T_VALUETYPE) {
 862       __ verify_oop(src->as_register());
 863     }
 864     move_regs(src->as_register(), dest->as_register());
 865 
 866   } else if (dest->is_double_cpu()) {
 867 #ifdef _LP64
 868     if (is_reference_type(src->type())) {
 869       // Surprising to me but we can see move of a long to t_object
 870       __ verify_oop(src->as_register());
 871       move_regs(src->as_register(), dest->as_register_lo());
 872       return;
 873     }
 874 #endif
 875     assert(src->is_double_cpu(), "must match");
 876     Register f_lo = src->as_register_lo();
 877     Register f_hi = src->as_register_hi();
 878     Register t_lo = dest->as_register_lo();
 879     Register t_hi = dest->as_register_hi();
 880 #ifdef _LP64
 881     assert(f_hi == f_lo, "must be same");
 882     assert(t_hi == t_lo, "must be same");
 883     move_regs(f_lo, t_lo);
 884 #else
 885     assert(f_lo != f_hi && t_lo != t_hi, "invalid register allocation");
 886 
 887 
 888     if (f_lo == t_hi && f_hi == t_lo) {
 889       swap_reg(f_lo, f_hi);
 890     } else if (f_hi == t_lo) {
 891       assert(f_lo != t_hi, "overwriting register");
 892       move_regs(f_hi, t_hi);
 893       move_regs(f_lo, t_lo);
 894     } else {
 895       assert(f_hi != t_lo, "overwriting register");
 896       move_regs(f_lo, t_lo);
 897       move_regs(f_hi, t_hi);
 898     }
 899 #endif // LP64
 900 
 901 #ifndef _LP64
 902     // special moves from fpu-register to xmm-register
 903     // necessary for method results
 904   } else if (src->is_single_xmm() && !dest->is_single_xmm()) {
 905     __ movflt(Address(rsp, 0), src->as_xmm_float_reg());
 906     __ fld_s(Address(rsp, 0));
 907   } else if (src->is_double_xmm() && !dest->is_double_xmm()) {
 908     __ movdbl(Address(rsp, 0), src->as_xmm_double_reg());
 909     __ fld_d(Address(rsp, 0));
 910   } else if (dest->is_single_xmm() && !src->is_single_xmm()) {
 911     __ fstp_s(Address(rsp, 0));
 912     __ movflt(dest->as_xmm_float_reg(), Address(rsp, 0));
 913   } else if (dest->is_double_xmm() && !src->is_double_xmm()) {
 914     __ fstp_d(Address(rsp, 0));
 915     __ movdbl(dest->as_xmm_double_reg(), Address(rsp, 0));
 916 #endif // !_LP64
 917 
 918     // move between xmm-registers
 919   } else if (dest->is_single_xmm()) {
 920     assert(src->is_single_xmm(), "must match");
 921     __ movflt(dest->as_xmm_float_reg(), src->as_xmm_float_reg());
 922   } else if (dest->is_double_xmm()) {
 923     assert(src->is_double_xmm(), "must match");
 924     __ movdbl(dest->as_xmm_double_reg(), src->as_xmm_double_reg());
 925 
 926 #ifndef _LP64
 927     // move between fpu-registers (no instruction necessary because of fpu-stack)
 928   } else if (dest->is_single_fpu() || dest->is_double_fpu()) {
 929     assert(src->is_single_fpu() || src->is_double_fpu(), "must match");
 930     assert(src->fpu() == dest->fpu(), "currently should be nothing to do");
 931 #endif // !_LP64
 932 
 933   } else {
 934     ShouldNotReachHere();
 935   }
 936 }

 969     assert(src->fpu_regnr() == 0, "argument must be on TOS");

 982     ShouldNotReachHere();
 983   }

 987 void LIR_Assembler::reg2mem(LIR_Opr src, LIR_Opr dest, BasicType type, LIR_PatchCode patch_code, CodeEmitInfo* info, bool pop_fpu_stack, bool wide, bool /* unaligned */) {
 988   LIR_Address* to_addr = dest->as_address_ptr();
 989   PatchingStub* patch = NULL;
 990   Register compressed_src = rscratch1;
 991 
 992   if (is_reference_type(type)) {
 993     __ verify_oop(src->as_register());
 994 #ifdef _LP64
 995     if (UseCompressedOops && !wide) {
 996       __ movptr(compressed_src, src->as_register());
 997       __ encode_heap_oop(compressed_src);
 998       if (patch_code != lir_patch_none) {
 999         info->oop_map()->set_narrowoop(compressed_src->as_VMReg());
1000       }
1001     }
1002 #endif
1003   }
1004 
1005   if (patch_code != lir_patch_none) {
1006     patch = new PatchingStub(_masm, PatchingStub::access_field_id);
1007     Address toa = as_Address(to_addr);
1008     assert(toa.disp() != 0, "must have");
1009   }
1010 
1011   int null_check_here = code_offset();
1012   switch (type) {
1013     case T_FLOAT: {
1014 #ifdef _LP64
1015       assert(src->is_single_xmm(), "not a float");
1016       __ movflt(as_Address(to_addr), src->as_xmm_float_reg());
1017 #else
1018       if (src->is_single_xmm()) {
1019         __ movflt(as_Address(to_addr), src->as_xmm_float_reg());
1020       } else {
1021         assert(src->is_single_fpu(), "must be");
1022         assert(src->fpu_regnr() == 0, "argument must be on TOS");
1023         if (pop_fpu_stack)      __ fstp_s(as_Address(to_addr));
1024         else                    __ fst_s (as_Address(to_addr));
1025       }
1026 #endif // _LP64
1027       break;
1028     }
1029 
1030     case T_DOUBLE: {
1031 #ifdef _LP64
1032       assert(src->is_double_xmm(), "not a double");
1033       __ movdbl(as_Address(to_addr), src->as_xmm_double_reg());
1034 #else
1035       if (src->is_double_xmm()) {
1036         __ movdbl(as_Address(to_addr), src->as_xmm_double_reg());
1037       } else {
1038         assert(src->is_double_fpu(), "must be");
1039         assert(src->fpu_regnrLo() == 0, "argument must be on TOS");
1040         if (pop_fpu_stack)      __ fstp_d(as_Address(to_addr));
1041         else                    __ fst_d (as_Address(to_addr));
1042       }
1043 #endif // _LP64
1044       break;
1045     }
1046 
1047     case T_VALUETYPE: // fall through
1048     case T_ARRAY:   // fall through
1049     case T_OBJECT:  // fall through
1050       if (UseCompressedOops && !wide) {
1051         __ movl(as_Address(to_addr), compressed_src);
1052       } else {
1053         __ movptr(as_Address(to_addr), src->as_register());
1054       }
1055       break;
1056     case T_METADATA:
1057       // We get here to store a method pointer to the stack to pass to
1058       // a dtrace runtime call. This can't work on 64 bit with
1059       // compressed klass ptrs: T_METADATA can be a compressed klass
1060       // ptr or a 64 bit method pointer.
1061       LP64_ONLY(ShouldNotReachHere());
1062       __ movptr(as_Address(to_addr), src->as_register());
1063       break;
1064     case T_ADDRESS:
1065       __ movptr(as_Address(to_addr), src->as_register());
1066       break;
1067     case T_INT:
1068       __ movl(as_Address(to_addr), src->as_register());
1069       break;
1070 
1071     case T_LONG: {
1072       Register from_lo = src->as_register_lo();
1073       Register from_hi = src->as_register_hi();
1074 #ifdef _LP64
1075       __ movptr(as_Address_lo(to_addr), from_lo);
1076 #else
1077       Register base = to_addr->base()->as_register();
1078       Register index = noreg;
1079       if (to_addr->index()->is_register()) {
1080         index = to_addr->index()->as_register();
1081       }
1082       if (base == from_lo || index == from_lo) {
1083         assert(base != from_hi, "can't be");
1084         assert(index == noreg || (index != base && index != from_hi), "can't handle this");
1085         __ movl(as_Address_hi(to_addr), from_hi);
1086         if (patch != NULL) {
1087           patching_epilog(patch, lir_patch_high, base, info);
1088           patch = new PatchingStub(_masm, PatchingStub::access_field_id);
1089           patch_code = lir_patch_low;
1090         }
1091         __ movl(as_Address_lo(to_addr), from_lo);
1092       } else {
1093         assert(index == noreg || (index != base && index != from_lo), "can't handle this");
1094         __ movl(as_Address_lo(to_addr), from_lo);
1095         if (patch != NULL) {
1096           patching_epilog(patch, lir_patch_low, base, info);
1097           patch = new PatchingStub(_masm, PatchingStub::access_field_id);
1098           patch_code = lir_patch_high;
1099         }
1100         __ movl(as_Address_hi(to_addr), from_hi);
1101       }
1102 #endif // _LP64
1103       break;
1104     }
1105 
1106     case T_BYTE:    // fall through
1107     case T_BOOLEAN: {
1108       Register src_reg = src->as_register();
1109       Address dst_addr = as_Address(to_addr);
1110       assert(VM_Version::is_P6() || src_reg->has_byte_register(), "must use byte registers if not P6");
1111       __ movb(dst_addr, src_reg);
1112       break;
1113     }
1114 
1115     case T_CHAR:    // fall through
1116     case T_SHORT:
1117       __ movw(as_Address(to_addr), src->as_register());
1118       break;
1119 
1120     default:
1121       ShouldNotReachHere();
1122   }
1123   if (info != NULL) {
1124     add_debug_info_for_null_check(null_check_here, info);
1125   }
1126 
1127   if (patch_code != lir_patch_none) {
1128     patching_epilog(patch, patch_code, to_addr->base()->as_register(), info);
1129   }
1130 }

1203     __ popl (frame_map()->address_for_slot(dest->double_stack_ix(), 2 * wordSize));
1204     __ popl (frame_map()->address_for_slot(dest->double_stack_ix(), 0));
1205 #endif // _LP64

1213 void LIR_Assembler::mem2reg(LIR_Opr src, LIR_Opr dest, BasicType type, LIR_PatchCode patch_code, CodeEmitInfo* info, bool wide, bool /* unaligned */) {
1214   assert(src->is_address(), "should not call otherwise");
1215   assert(dest->is_register(), "should not call otherwise");
1216 
1217   LIR_Address* addr = src->as_address_ptr();
1218   Address from_addr = as_Address(addr);
1219 
1220   if (addr->base()->type() == T_OBJECT || addr->base()->type() == T_VALUETYPE) {
1221     __ verify_oop(addr->base()->as_pointer_register());
1222   }
1223 
1224   switch (type) {
1225     case T_BOOLEAN: // fall through
1226     case T_BYTE:    // fall through
1227     case T_CHAR:    // fall through
1228     case T_SHORT:
1229       if (!VM_Version::is_P6() && !from_addr.uses(dest->as_register())) {
1230         // on pre P6 processors we may get partial register stalls
1231         // so blow away the value of to_rinfo before loading a
1232         // partial word into it.  Do it here so that it precedes
1233         // the potential patch point below.
1234         __ xorptr(dest->as_register(), dest->as_register());
1235       }
1236       break;
1237    default:
1238      break;
1239   }
1240 
1241   PatchingStub* patch = NULL;
1242   if (patch_code != lir_patch_none) {
1243     patch = new PatchingStub(_masm, PatchingStub::access_field_id);
1244     assert(from_addr.disp() != 0, "must have");
1245   }
1246   if (info != NULL) {
1247     add_debug_info_for_null_check_here(info);
1248   }
1249 
1250   switch (type) {
1251     case T_FLOAT: {
1252       if (dest->is_single_xmm()) {
1253         __ movflt(dest->as_xmm_float_reg(), from_addr);
1254       } else {
1255 #ifndef _LP64
1256         assert(dest->is_single_fpu(), "must be");
1257         assert(dest->fpu_regnr() == 0, "dest must be TOS");
1258         __ fld_s(from_addr);
1259 #else
1260         ShouldNotReachHere();
1261 #endif // !LP64
1262       }
1263       break;
1264     }
1265 
1266     case T_DOUBLE: {
1267       if (dest->is_double_xmm()) {
1268         __ movdbl(dest->as_xmm_double_reg(), from_addr);
1269       } else {
1270 #ifndef _LP64
1271         assert(dest->is_double_fpu(), "must be");
1272         assert(dest->fpu_regnrLo() == 0, "dest must be TOS");
1273         __ fld_d(from_addr);
1274 #else
1275         ShouldNotReachHere();
1276 #endif // !LP64
1277       }
1278       break;
1279     }
1280 
1281     case T_VALUETYPE: // fall through
1282     case T_OBJECT:  // fall through
1283     case T_ARRAY:   // fall through
1284       if (UseCompressedOops && !wide) {
1285         __ movl(dest->as_register(), from_addr);
1286       } else {
1287         __ movptr(dest->as_register(), from_addr);
1288       }
1289       break;
1290 
1291     case T_ADDRESS:
1292       if (UseCompressedClassPointers && addr->disp() == oopDesc::klass_offset_in_bytes()) {
1293         __ movl(dest->as_register(), from_addr);
1294       } else {
1295         __ movptr(dest->as_register(), from_addr);
1296       }
1297       break;
1298     case T_INT:
1299       __ movl(dest->as_register(), from_addr);
1300       break;
1301 
1302     case T_LONG: {
1303       Register to_lo = dest->as_register_lo();
1304       Register to_hi = dest->as_register_hi();
1305 #ifdef _LP64
1306       __ movptr(to_lo, as_Address_lo(addr));
1307 #else
1308       Register base = addr->base()->as_register();
1309       Register index = noreg;
1310       if (addr->index()->is_register()) {
1311         index = addr->index()->as_register();
1312       }
1313       if ((base == to_lo && index == to_hi) ||
1314           (base == to_hi && index == to_lo)) {
1315         // addresses with 2 registers are only formed as a result of
1316         // array access so this code will never have to deal with
1317         // patches or null checks.
1318         assert(info == NULL && patch == NULL, "must be");
1319         __ lea(to_hi, as_Address(addr));
1320         __ movl(to_lo, Address(to_hi, 0));
1321         __ movl(to_hi, Address(to_hi, BytesPerWord));
1322       } else if (base == to_lo || index == to_lo) {
1323         assert(base != to_hi, "can't be");
1324         assert(index == noreg || (index != base && index != to_hi), "can't handle this");
1325         __ movl(to_hi, as_Address_hi(addr));
1326         if (patch != NULL) {
1327           patching_epilog(patch, lir_patch_high, base, info);
1328           patch = new PatchingStub(_masm, PatchingStub::access_field_id);
1329           patch_code = lir_patch_low;
1330         }
1331         __ movl(to_lo, as_Address_lo(addr));
1332       } else {
1333         assert(index == noreg || (index != base && index != to_lo), "can't handle this");
1334         __ movl(to_lo, as_Address_lo(addr));
1335         if (patch != NULL) {
1336           patching_epilog(patch, lir_patch_low, base, info);
1337           patch = new PatchingStub(_masm, PatchingStub::access_field_id);
1338           patch_code = lir_patch_high;
1339         }
1340         __ movl(to_hi, as_Address_hi(addr));
1341       }
1342 #endif // _LP64
1343       break;
1344     }
1345 
1346     case T_BOOLEAN: // fall through
1347     case T_BYTE: {
1348       Register dest_reg = dest->as_register();
1349       assert(VM_Version::is_P6() || dest_reg->has_byte_register(), "must use byte registers if not P6");
1350       if (VM_Version::is_P6() || from_addr.uses(dest_reg)) {
1351         __ movsbl(dest_reg, from_addr);
1352       } else {
1353         __ movb(dest_reg, from_addr);
1354         __ shll(dest_reg, 24);
1355         __ sarl(dest_reg, 24);
1356       }
1357       break;
1358     }
1359 
1360     case T_CHAR: {
1361       Register dest_reg = dest->as_register();
1362       assert(VM_Version::is_P6() || dest_reg->has_byte_register(), "must use byte registers if not P6");
1363       if (VM_Version::is_P6() || from_addr.uses(dest_reg)) {
1364         __ movzwl(dest_reg, from_addr);
1365       } else {
1366         __ movw(dest_reg, from_addr);
1367       }
1368       break;
1369     }
1370 
1371     case T_SHORT: {
1372       Register dest_reg = dest->as_register();
1373       if (VM_Version::is_P6() || from_addr.uses(dest_reg)) {
1374         __ movswl(dest_reg, from_addr);
1375       } else {
1376         __ movw(dest_reg, from_addr);
1377         __ shll(dest_reg, 16);
1378         __ sarl(dest_reg, 16);
1379       }
1380       break;
1381     }
1382 
1383     default:
1384       ShouldNotReachHere();
1385   }
1386 
1387   if (patch != NULL) {
1388     patching_epilog(patch, patch_code, addr->base()->as_register(), info);
1389   }
1390 
1391   if (is_reference_type(type)) {
1392 #ifdef _LP64
1393     if (UseCompressedOops && !wide) {
1394       __ decode_heap_oop(dest->as_register());
1395     }
1396 #endif
1397 
1398     // Load barrier has not yet been applied, so ZGC can't verify the oop here
1399     if (!UseZGC) {
1400       __ verify_oop(dest->as_register());
1401     }
1402   } else if (type == T_ADDRESS && addr->disp() == oopDesc::klass_offset_in_bytes()) {
1403     // TODO remove clear_prop_bits bits stuff once the runtime does not set it anymore
1404 #ifdef _LP64
1405     if (UseCompressedClassPointers) {
1406       __ andl(dest->as_register(), oopDesc::compressed_klass_mask());
1407       __ decode_klass_not_null(dest->as_register());
1408     } else {
1409       __ shlq(dest->as_register(), oopDesc::storage_props_nof_bits);
1410       __ shrq(dest->as_register(), oopDesc::storage_props_nof_bits);
1411     }
1412 #else
1413     __ andl(dest->as_register(), oopDesc::wide_klass_mask());
1414 #endif
1415   }
1416 }

1460 void LIR_Assembler::emit_opBranch(LIR_OpBranch* op) {
1461 #ifdef ASSERT
1462   assert(op->block() == NULL || op->block()->label() == op->label(), "wrong label");

1470   } else {
1471     Assembler::Condition acond = Assembler::zero;
1472     if (op->code() == lir_cond_float_branch) {

1479         case lir_cond_lessEqual:    acond = Assembler::belowEqual; break;
1480         case lir_cond_greaterEqual: acond = Assembler::aboveEqual; break;
1481         case lir_cond_greater:      acond = Assembler::above;      break;
1482         default:                         ShouldNotReachHere();
1483       }
1484     } else {
1485       switch (op->cond()) {
1486         case lir_cond_equal:        acond = Assembler::equal;       break;
1487         case lir_cond_notEqual:     acond = Assembler::notEqual;    break;
1488         case lir_cond_less:         acond = Assembler::less;        break;
1489         case lir_cond_lessEqual:    acond = Assembler::lessEqual;   break;
1490         case lir_cond_greaterEqual: acond = Assembler::greaterEqual;break;
1491         case lir_cond_greater:      acond = Assembler::greater;     break;
1492         case lir_cond_belowEqual:   acond = Assembler::belowEqual;  break;
1493         case lir_cond_aboveEqual:   acond = Assembler::aboveEqual;  break;
1494         default:                         ShouldNotReachHere();
1495       }
1496     }
1497     __ jcc(acond,*(op->label()));
1498   }

1501 void LIR_Assembler::emit_opConvert(LIR_OpConvert* op) {
1502   LIR_Opr src  = op->in_opr();
1503   LIR_Opr dest = op->result_opr();
1504 
1505   switch (op->bytecode()) {
1506     case Bytecodes::_i2l:
1507 #ifdef _LP64
1508       __ movl2ptr(dest->as_register_lo(), src->as_register());
1509 #else
1510       move_regs(src->as_register(), dest->as_register_lo());
1511       move_regs(src->as_register(), dest->as_register_hi());
1512       __ sarl(dest->as_register_hi(), 31);
1513 #endif // LP64
1514       break;
1515 
1516     case Bytecodes::_l2i:
1517 #ifdef _LP64
1518       __ movl(dest->as_register(), src->as_register_lo());
1519 #else
1520       move_regs(src->as_register_lo(), dest->as_register());
1521 #endif
1522       break;
1523 
1524     case Bytecodes::_i2b:
1525       move_regs(src->as_register(), dest->as_register());
1526       __ sign_extend_byte(dest->as_register());
1527       break;
1528 
1529     case Bytecodes::_i2c:
1530       move_regs(src->as_register(), dest->as_register());
1531       __ andl(dest->as_register(), 0xFFFF);
1532       break;
1533 
1534     case Bytecodes::_i2s:
1535       move_regs(src->as_register(), dest->as_register());
1536       __ sign_extend_short(dest->as_register());
1537       break;
1538 
1539 
1540 #ifdef _LP64
1541     case Bytecodes::_f2d:
1542       __ cvtss2sd(dest->as_xmm_double_reg(), src->as_xmm_float_reg());
1543       break;
1544 
1545     case Bytecodes::_d2f:
1546       __ cvtsd2ss(dest->as_xmm_float_reg(), src->as_xmm_double_reg());
1547       break;
1548 
1549     case Bytecodes::_i2f:
1550       __ cvtsi2ssl(dest->as_xmm_float_reg(), src->as_register());
1551       break;
1552 
1553     case Bytecodes::_i2d:
1554       __ cvtsi2sdl(dest->as_xmm_double_reg(), src->as_register());
1555       break;
1556 
1557     case Bytecodes::_l2f:
1558       __ cvtsi2ssq(dest->as_xmm_float_reg(), src->as_register_lo());
1559       break;
1560 
1561     case Bytecodes::_l2d:
1562       __ cvtsi2sdq(dest->as_xmm_double_reg(), src->as_register_lo());
1563       break;
1564 
1565     case Bytecodes::_f2i:
1566       __ convert_f2i(dest->as_register(), src->as_xmm_float_reg());
1567       break;
1568 
1569     case Bytecodes::_d2i:
1570       __ convert_d2i(dest->as_register(), src->as_xmm_double_reg());
1571       break;
1572 
1573     case Bytecodes::_f2l:
1574       __ convert_f2l(dest->as_register_lo(), src->as_xmm_float_reg());
1575       break;
1576 
1577     case Bytecodes::_d2l:
1578       __ convert_d2l(dest->as_register_lo(), src->as_xmm_double_reg());
1579       break;
1580 #else
1581     case Bytecodes::_f2d:
1582     case Bytecodes::_d2f:
1583       if (dest->is_single_xmm()) {
1584         __ cvtsd2ss(dest->as_xmm_float_reg(), src->as_xmm_double_reg());
1585       } else if (dest->is_double_xmm()) {
1586         __ cvtss2sd(dest->as_xmm_double_reg(), src->as_xmm_float_reg());
1587       } else {
1588         assert(src->fpu() == dest->fpu(), "register must be equal");
1589         // do nothing (float result is rounded later through spilling)
1590       }
1591       break;
1592 
1593     case Bytecodes::_i2f:
1594     case Bytecodes::_i2d:
1595       if (dest->is_single_xmm()) {
1596         __ cvtsi2ssl(dest->as_xmm_float_reg(), src->as_register());
1597       } else if (dest->is_double_xmm()) {
1598         __ cvtsi2sdl(dest->as_xmm_double_reg(), src->as_register());
1599       } else {
1600         assert(dest->fpu() == 0, "result must be on TOS");
1601         __ movl(Address(rsp, 0), src->as_register());
1602         __ fild_s(Address(rsp, 0));
1603       }
1604       break;
1605 
1606     case Bytecodes::_l2f:
1607     case Bytecodes::_l2d:
1608       assert(!dest->is_xmm_register(), "result in xmm register not supported (no SSE instruction present)");
1609       assert(dest->fpu() == 0, "result must be on TOS");
1610       __ movptr(Address(rsp, 0),          src->as_register_lo());
1611       __ movl(Address(rsp, BytesPerWord), src->as_register_hi());
1612       __ fild_d(Address(rsp, 0));
1613       // float result is rounded later through spilling
1614       break;
1615 
1616     case Bytecodes::_f2i:
1617     case Bytecodes::_d2i:
1618       if (src->is_single_xmm()) {
1619         __ cvttss2sil(dest->as_register(), src->as_xmm_float_reg());
1620       } else if (src->is_double_xmm()) {
1621         __ cvttsd2sil(dest->as_register(), src->as_xmm_double_reg());
1622       } else {
1623         assert(src->fpu() == 0, "input must be on TOS");
1624         __ fldcw(ExternalAddress(StubRoutines::addr_fpu_cntrl_wrd_trunc()));
1625         __ fist_s(Address(rsp, 0));
1626         __ movl(dest->as_register(), Address(rsp, 0));
1627         __ fldcw(ExternalAddress(StubRoutines::addr_fpu_cntrl_wrd_std()));
1628       }
1629       // IA32 conversion instructions do not match JLS for overflow, underflow and NaN -> fixup in stub
1630       assert(op->stub() != NULL, "stub required");
1631       __ cmpl(dest->as_register(), 0x80000000);
1632       __ jcc(Assembler::equal, *op->stub()->entry());
1633       __ bind(*op->stub()->continuation());
1634       break;
1635 
1636     case Bytecodes::_f2l:
1637     case Bytecodes::_d2l:
1638       assert(!src->is_xmm_register(), "input in xmm register not supported (no SSE instruction present)");
1639       assert(src->fpu() == 0, "input must be on TOS");
1640       assert(dest == FrameMap::long0_opr, "runtime stub places result in these registers");
1641 
1642       // instruction sequence too long to inline it here
1643       {
1644         __ call(RuntimeAddress(Runtime1::entry_for(Runtime1::fpu2long_stub_id)));
1645       }
1646       break;
1647 #endif // _LP64
1648 
1649     default: ShouldNotReachHere();
1650   }
1651 }

1693                       len,
1694                       tmp1,
1695                       tmp2,
1696                       arrayOopDesc::header_size(op->type()),

1698                       op->klass()->as_register(),
1699                       *op->stub()->entry());
1700   }
1701   __ bind(*op->stub()->continuation());

1704 void LIR_Assembler::type_profile_helper(Register mdo,
1705                                         ciMethodData *md, ciProfileData *data,
1706                                         Register recv, Label* update_done) {
1707   for (uint i = 0; i < ReceiverTypeData::row_limit(); i++) {
1708     Label next_test;
1709     // See if the receiver is receiver[n].
1710     __ cmpptr(recv, Address(mdo, md->byte_offset_of_slot(data, ReceiverTypeData::receiver_offset(i))));
1711     __ jccb(Assembler::notEqual, next_test);
1712     Address data_addr(mdo, md->byte_offset_of_slot(data, ReceiverTypeData::receiver_count_offset(i)));
1713     __ addptr(data_addr, DataLayout::counter_increment);
1714     __ jmp(*update_done);
1715     __ bind(next_test);
1716   }
1717 
1718   // Didn't find receiver; find next empty slot and fill it in
1719   for (uint i = 0; i < ReceiverTypeData::row_limit(); i++) {
1720     Label next_test;
1721     Address recv_addr(mdo, md->byte_offset_of_slot(data, ReceiverTypeData::receiver_offset(i)));
1722     __ cmpptr(recv_addr, (intptr_t)NULL_WORD);
1723     __ jccb(Assembler::notEqual, next_test);
1724     __ movptr(recv_addr, recv);
1725     __ movptr(Address(mdo, md->byte_offset_of_slot(data, ReceiverTypeData::receiver_count_offset(i))), DataLayout::counter_increment);
1726     __ jmp(*update_done);
1727     __ bind(next_test);
1728   }

1731 void LIR_Assembler::emit_typecheck_helper(LIR_OpTypeCheck *op, Label* success, Label* failure, Label* obj_is_null) {
1732   // we always need a stub for the failure case.
1733   CodeStub* stub = op->stub();
1734   Register obj = op->object()->as_register();
1735   Register k_RInfo = op->tmp1()->as_register();
1736   Register klass_RInfo = op->tmp2()->as_register();
1737   Register dst = op->result_opr()->as_register();
1738   ciKlass* k = op->klass();
1739   Register Rtmp1 = noreg;
1740 
1741   // check if it needs to be profiled
1742   ciMethodData* md = NULL;
1743   ciProfileData* data = NULL;
1744 
1745   if (op->should_profile()) {
1746     ciMethod* method = op->profiled_method();
1747     assert(method != NULL, "Should have method");
1748     int bci = op->profiled_bci();
1749     md = method->method_data_or_null();
1750     assert(md != NULL, "Sanity");
1751     data = md->bci_to_data(bci);
1752     assert(data != NULL,                "need data for type check");
1753     assert(data->is_ReceiverTypeData(), "need ReceiverTypeData for type check");
1754   }
1755   Label profile_cast_success, profile_cast_failure;
1756   Label *success_target = op->should_profile() ? &profile_cast_success : success;
1757   Label *failure_target = op->should_profile() ? &profile_cast_failure : failure;
1758 
1759   if (obj == k_RInfo) {
1760     k_RInfo = dst;
1761   } else if (obj == klass_RInfo) {
1762     klass_RInfo = dst;
1763   }
1764   if (k->is_loaded() && !UseCompressedClassPointers) {
1765     select_different_registers(obj, dst, k_RInfo, klass_RInfo);
1766   } else {
1767     Rtmp1 = op->tmp3()->as_register();
1768     select_different_registers(obj, dst, k_RInfo, klass_RInfo, Rtmp1);
1769   }
1770 
1771   assert_different_registers(obj, k_RInfo, klass_RInfo);
1772 
1773   if (op->need_null_check()) {
1774     __ cmpptr(obj, (int32_t)NULL_WORD);
1775     if (op->should_profile()) {
1776       Label not_null;
1777       __ jccb(Assembler::notEqual, not_null);
1778       // Object is null; update MDO and exit
1779       Register mdo  = klass_RInfo;
1780       __ mov_metadata(mdo, md->constant_encoding());
1781       Address data_addr(mdo, md->byte_offset_of_slot(data, DataLayout::flags_offset()));
1782       int header_bits = BitData::null_seen_byte_constant();
1783       __ orb(data_addr, header_bits);
1784       __ jmp(*obj_is_null);
1785       __ bind(not_null);
1786     } else {
1787       __ jcc(Assembler::equal, *obj_is_null);
1788     }
1789   }
1790 
1791   if (!k->is_loaded()) {
1792     klass2reg_with_patching(k_RInfo, op->info_for_patch());
1793   } else {
1794 #ifdef _LP64
1795     __ mov_metadata(k_RInfo, k->constant_encoding());
1796 #endif // _LP64
1797   }
1798   __ verify_oop(obj);
1799 
1800   if (op->fast_check()) {
1801     // get object class
1802     // not a safepoint as obj null check happens earlier
1803 #ifdef _LP64
1804     if (UseCompressedClassPointers) {
1805       __ load_klass(Rtmp1, obj);
1806       __ cmpptr(k_RInfo, Rtmp1);
1807     } else {
1808       __ cmpptr(k_RInfo, Address(obj, oopDesc::klass_offset_in_bytes()));
1809     }
1810 #else
1811     if (k->is_loaded()) {
1812       __ cmpklass(Address(obj, oopDesc::klass_offset_in_bytes()), k->constant_encoding());
1813     } else {
1814       __ cmpptr(k_RInfo, Address(obj, oopDesc::klass_offset_in_bytes()));
1815     }
1816 #endif
1817     __ jcc(Assembler::notEqual, *failure_target);
1818     // successful cast, fall through to profile or jump
1819   } else {
1820     // get object class
1821     // not a safepoint as obj null check happens earlier
1822     __ load_klass(klass_RInfo, obj);
1823     if (k->is_loaded()) {
1824       // See if we get an immediate positive hit
1825 #ifdef _LP64
1826       __ cmpptr(k_RInfo, Address(klass_RInfo, k->super_check_offset()));
1827 #else
1828       __ cmpklass(Address(klass_RInfo, k->super_check_offset()), k->constant_encoding());
1829 #endif // _LP64
1830       if ((juint)in_bytes(Klass::secondary_super_cache_offset()) != k->super_check_offset()) {
1831         __ jcc(Assembler::notEqual, *failure_target);
1832         // successful cast, fall through to profile or jump
1833       } else {
1834         // See if we get an immediate positive hit
1835         __ jcc(Assembler::equal, *success_target);
1836         // check for self
1837 #ifdef _LP64
1838         __ cmpptr(klass_RInfo, k_RInfo);
1839 #else
1840         __ cmpklass(klass_RInfo, k->constant_encoding());
1841 #endif // _LP64
1842         __ jcc(Assembler::equal, *success_target);
1843 
1844         __ push(klass_RInfo);
1845 #ifdef _LP64
1846         __ push(k_RInfo);
1847 #else
1848         __ pushklass(k->constant_encoding());
1849 #endif // _LP64
1850         __ call(RuntimeAddress(Runtime1::entry_for(Runtime1::slow_subtype_check_id)));
1851         __ pop(klass_RInfo);
1852         __ pop(klass_RInfo);
1853         // result is a boolean
1854         __ cmpl(klass_RInfo, 0);
1855         __ jcc(Assembler::equal, *failure_target);
1856         // successful cast, fall through to profile or jump
1857       }
1858     } else {
1859       // perform the fast part of the checking logic
1860       __ check_klass_subtype_fast_path(klass_RInfo, k_RInfo, Rtmp1, success_target, failure_target, NULL);
1861       // call out-of-line instance of __ check_klass_subtype_slow_path(...):
1862       __ push(klass_RInfo);
1863       __ push(k_RInfo);
1864       __ call(RuntimeAddress(Runtime1::entry_for(Runtime1::slow_subtype_check_id)));
1865       __ pop(klass_RInfo);
1866       __ pop(k_RInfo);
1867       // result is a boolean
1868       __ cmpl(k_RInfo, 0);
1869       __ jcc(Assembler::equal, *failure_target);
1870       // successful cast, fall through to profile or jump
1871     }
1872   }
1873   if (op->should_profile()) {
1874     Register mdo  = klass_RInfo, recv = k_RInfo;
1875     __ bind(profile_cast_success);
1876     __ mov_metadata(mdo, md->constant_encoding());
1877     __ load_klass(recv, obj);
1878     type_profile_helper(mdo, md, data, recv, success);
1879     __ jmp(*success);
1880 
1881     __ bind(profile_cast_failure);
1882     __ mov_metadata(mdo, md->constant_encoding());
1883     Address counter_addr(mdo, md->byte_offset_of_slot(data, CounterData::count_offset()));
1884     __ subptr(counter_addr, DataLayout::counter_increment);
1885     __ jmp(*failure);
1886   }
1887   __ jmp(*success);
1888 }

2029 void LIR_Assembler::emit_opSubstitutabilityCheck(LIR_OpSubstitutabilityCheck* op) {
2030   Label L_oops_equal;
2031   Label L_oops_not_equal;
2032   Label L_end;
2033 
2034   Register left  = op->left()->as_register();
2035   Register right = op->right()->as_register();
2036 
2037   __ cmpptr(left, right);
2038   __ jcc(Assembler::equal, L_oops_equal);
2039 
2040   // (1) Null check -- if one of the operands is null, the other must not be null (because
2041   //     the two references are not equal), so they are not substitutable,
2042   //     FIXME: do null check only if the operand is nullable
2043   {
2044     __ cmpptr(left, (int32_t)NULL_WORD);
2045     __ jcc(Assembler::equal, L_oops_not_equal);
2046 
2047     __ cmpptr(right, (int32_t)NULL_WORD);
2048     __ jcc(Assembler::equal, L_oops_not_equal);
2049   }
2050 
2051   ciKlass* left_klass = op->left_klass();
2052   ciKlass* right_klass = op->right_klass();
2053 
2054   // (2) Value object check -- if either of the operands is not a value object,
2055   //     they are not substitutable. We do this only if we are not sure that the
2056   //     operands are value objects
2057   if ((left_klass == NULL || right_klass == NULL) ||// The klass is still unloaded, or came from a Phi node.
2058       !left_klass->is_valuetype() || !right_klass->is_valuetype()) {
2059     Register tmp1  = op->tmp1()->as_register();
2060     __ movptr(tmp1, (intptr_t)markWord::always_locked_pattern);
2061     __ andl(tmp1, Address(left, oopDesc::mark_offset_in_bytes()));
2062     __ andl(tmp1, Address(right, oopDesc::mark_offset_in_bytes()));
2063     __ cmpptr(tmp1, (intptr_t)markWord::always_locked_pattern);
2064     __ jcc(Assembler::notEqual, L_oops_not_equal);
2065   }
2066 
2067   // (3) Same klass check: if the operands are of different klasses, they are not substitutable.
2068   if (left_klass != NULL && left_klass->is_valuetype() && left_klass == right_klass) {
2069     // No need to load klass -- the operands are statically known to be the same value klass.
2070     __ jmp(*op->stub()->entry());
2071   } else {
2072     Register left_klass_op = op->left_klass_op()->as_register();
2073     Register right_klass_op = op->right_klass_op()->as_register();
2074 
2075     if (UseCompressedOops) {
2076       __ movl(left_klass_op,  Address(left,  oopDesc::klass_offset_in_bytes()));
2077       __ movl(right_klass_op, Address(right, oopDesc::klass_offset_in_bytes()));
2078       __ cmpl(left_klass_op, right_klass_op);
2079     } else {
2080       __ movptr(left_klass_op,  Address(left,  oopDesc::klass_offset_in_bytes()));
2081       __ movptr(right_klass_op, Address(right, oopDesc::klass_offset_in_bytes()));
2082       __ cmpptr(left_klass_op, right_klass_op);
2083     }
2084 
2085     __ jcc(Assembler::equal, *op->stub()->entry()); // same klass -> do slow check
2086     // fall through to L_oops_not_equal
2087   }
2088 
2089   __ bind(L_oops_not_equal);
2090   move(op->not_equal_result(), op->result_opr());
2091   __ jmp(L_end);
2092 
2093   __ bind(L_oops_equal);
2094   move(op->equal_result(), op->result_opr());
2095   __ jmp(L_end);
2096 
2097   // We've returned from the stub. RAX contains 0x0 IFF the two
2098   // operands are not substitutable. (Don't compare against 0x1 in case the
2099   // C compiler is naughty)
2100   __ bind(*op->stub()->continuation());
2101   __ cmpl(rax, 0);
2102   __ jcc(Assembler::equal, L_oops_not_equal); // (call_stub() == 0x0) -> not_equal
2103   move(op->equal_result(), op->result_opr()); // (call_stub() != 0x0) -> equal
2104   // fall-through
2105   __ bind(L_end);
2106 }

2183   Assembler::Condition acond, ncond;
2184   switch (condition) {
2185     case lir_cond_equal:        acond = Assembler::equal;        ncond = Assembler::notEqual;     break;
2186     case lir_cond_notEqual:     acond = Assembler::notEqual;     ncond = Assembler::equal;        break;

2192     case lir_cond_aboveEqual:   acond = Assembler::aboveEqual;   ncond = Assembler::below;        break;
2193     default:                    acond = Assembler::equal;        ncond = Assembler::notEqual;
2194                                 ShouldNotReachHere();
2195   }
2196 
2197   if (opr1->is_cpu_register()) {
2198     reg2reg(opr1, result);
2199   } else if (opr1->is_stack()) {
2200     stack2reg(opr1, result, result->type());
2201   } else if (opr1->is_constant()) {

2760       move_regs(l_lo, dst_lo);

2762       assert(dst_lo != l_hi, "overwriting registers");

2771 // we assume that rax, and rdx can be overwritten
2772 void LIR_Assembler::arithmetic_idiv(LIR_Code code, LIR_Opr left, LIR_Opr right, LIR_Opr temp, LIR_Opr result, CodeEmitInfo* info) {
2773 
2774   assert(left->is_single_cpu(),   "left must be register");
2775   assert(right->is_single_cpu() || right->is_constant(),  "right must be register or constant");
2776   assert(result->is_single_cpu(), "result must be register");
2777 
2778   //  assert(left->destroys_register(), "check");
2779   //  assert(right->destroys_register(), "check");
2780 
2781   Register lreg = left->as_register();
2782   Register dreg = result->as_register();
2783 
2784   if (right->is_constant()) {
2785     jint divisor = right->as_constant_ptr()->as_jint();
2786     assert(divisor > 0 && is_power_of_2(divisor), "must be");
2787     if (code == lir_idiv) {
2788       assert(lreg == rax, "must be rax,");
2789       assert(temp->as_register() == rdx, "tmp register must be rdx");
2790       __ cdql(); // sign extend into rdx:rax
2791       if (divisor == 2) {
2792         __ subl(lreg, rdx);
2793       } else {
2794         __ andl(rdx, divisor - 1);
2795         __ addl(lreg, rdx);
2796       }
2797       __ sarl(lreg, log2_jint(divisor));
2798       move_regs(lreg, dreg);
2799     } else if (code == lir_irem) {
2800       Label done;
2801       __ mov(dreg, lreg);
2802       __ andl(dreg, 0x80000000 | (divisor - 1));
2803       __ jcc(Assembler::positive, done);
2804       __ decrement(dreg);
2805       __ orl(dreg, ~(divisor - 1));
2806       __ increment(dreg);
2807       __ bind(done);
2808     } else {
2809       ShouldNotReachHere();
2810     }
2811   } else {
2812     Register rreg = right->as_register();
2813     assert(lreg == rax, "left register must be rax,");
2814     assert(rreg != rdx, "right register must not be rdx");
2815     assert(temp->as_register() == rdx, "tmp register must be rdx");
2816 
2817     move_regs(lreg, rax);
2818 
2819     int idivl_offset = __ corrected_idivl(rreg);
2820     if (ImplicitDiv0Checks) {
2821       add_debug_info_for_div0(idivl_offset, info);
2822     }
2823     if (code == lir_irem) {
2824       move_regs(rdx, dreg); // result is in rdx
2825     } else {
2826       move_regs(rax, dreg);
2827     }
2828   }

2832 void LIR_Assembler::comp_op(LIR_Condition condition, LIR_Opr opr1, LIR_Opr opr2, LIR_Op2* op) {
2833   if (opr1->is_single_cpu()) {
2834     Register reg1 = opr1->as_register();
2835     if (opr2->is_single_cpu()) {
2836       // cpu register - cpu register
2837       if (is_reference_type(opr1->type())) {
2838         __ cmpoop(reg1, opr2->as_register());
2839       } else {
2840         assert(!is_reference_type(opr2->type()), "cmp int, oop?");
2841         __ cmpl(reg1, opr2->as_register());
2842       }
2843     } else if (opr2->is_stack()) {
2844       // cpu register - stack
2845       if (is_reference_type(opr1->type())) {
2846         __ cmpoop(reg1, frame_map()->address_for_slot(opr2->single_stack_ix()));
2847       } else {
2848         __ cmpl(reg1, frame_map()->address_for_slot(opr2->single_stack_ix()));
2849       }
2850     } else if (opr2->is_constant()) {
2851       // cpu register - constant
2852       LIR_Const* c = opr2->as_constant_ptr();
2853       if (c->type() == T_INT) {
2854         __ cmpl(reg1, c->as_jint());
2855       } else if (c->type() == T_METADATA) {
2856         // All we need for now is a comparison with NULL for equality.
2857         assert(condition == lir_cond_equal || condition == lir_cond_notEqual, "oops");
2858         Metadata* m = c->as_metadata();
2859         if (m == NULL) {
2860           __ cmpptr(reg1, (int32_t)0);
2861         } else {
2862           ShouldNotReachHere();
2863         }
2864       } else if (is_reference_type(c->type())) {
2865         // In 64bit oops are single register
2866         jobject o = c->as_jobject();
2867         if (o == NULL) {
2868           __ cmpptr(reg1, (int32_t)NULL_WORD);
2869         } else {
2870           __ cmpoop(reg1, o);
2871         }
2872       } else {
2873         fatal("unexpected type: %s", basictype_to_str(c->type()));
2874       }
2875       // cpu register - address
2876     } else if (opr2->is_address()) {
2877       if (op->info() != NULL) {
2878         add_debug_info_for_null_check_here(op->info());
2879       }
2880       __ cmpl(reg1, as_Address(opr2->as_address_ptr()));
2881     } else {
2882       ShouldNotReachHere();
2883     }
2884 
2885   } else if(opr1->is_double_cpu()) {
2886     Register xlo = opr1->as_register_lo();
2887     Register xhi = opr1->as_register_hi();
2888     if (opr2->is_double_cpu()) {
2889 #ifdef _LP64
2890       __ cmpptr(xlo, opr2->as_register_lo());
2891 #else
2892       // cpu register - cpu register
2893       Register ylo = opr2->as_register_lo();
2894       Register yhi = opr2->as_register_hi();
2895       __ subl(xlo, ylo);
2896       __ sbbl(xhi, yhi);
2897       if (condition == lir_cond_equal || condition == lir_cond_notEqual) {
2898         __ orl(xhi, xlo);
2899       }
2900 #endif // _LP64
2901     } else if (opr2->is_constant()) {
2902       // cpu register - constant 0
2903       assert(opr2->as_jlong() == (jlong)0, "only handles zero");
2904 #ifdef _LP64
2905       __ cmpptr(xlo, (int32_t)opr2->as_jlong());
2906 #else
2907       assert(condition == lir_cond_equal || condition == lir_cond_notEqual, "only handles equals case");
2908       __ orl(xhi, xlo);
2909 #endif // _LP64
2910     } else {
2911       ShouldNotReachHere();
2912     }
2913 
2914   } else if (opr1->is_single_xmm()) {
2915     XMMRegister reg1 = opr1->as_xmm_float_reg();
2916     if (opr2->is_single_xmm()) {
2917       // xmm register - xmm register
2918       __ ucomiss(reg1, opr2->as_xmm_float_reg());
2919     } else if (opr2->is_stack()) {
2920       // xmm register - stack
2921       __ ucomiss(reg1, frame_map()->address_for_slot(opr2->single_stack_ix()));
2922     } else if (opr2->is_constant()) {
2923       // xmm register - constant
2924       __ ucomiss(reg1, InternalAddress(float_constant(opr2->as_jfloat())));
2925     } else if (opr2->is_address()) {
2926       // xmm register - address
2927       if (op->info() != NULL) {
2928         add_debug_info_for_null_check_here(op->info());
2929       }
2930       __ ucomiss(reg1, as_Address(opr2->as_address_ptr()));
2931     } else {
2932       ShouldNotReachHere();
2933     }
2934 
2935   } else if (opr1->is_double_xmm()) {
2936     XMMRegister reg1 = opr1->as_xmm_double_reg();
2937     if (opr2->is_double_xmm()) {
2938       // xmm register - xmm register
2939       __ ucomisd(reg1, opr2->as_xmm_double_reg());
2940     } else if (opr2->is_stack()) {
2941       // xmm register - stack
2942       __ ucomisd(reg1, frame_map()->address_for_slot(opr2->double_stack_ix()));
2943     } else if (opr2->is_constant()) {
2944       // xmm register - constant
2945       __ ucomisd(reg1, InternalAddress(double_constant(opr2->as_jdouble())));
2946     } else if (opr2->is_address()) {
2947       // xmm register - address
2948       if (op->info() != NULL) {
2949         add_debug_info_for_null_check_here(op->info());
2950       }
2951       __ ucomisd(reg1, as_Address(opr2->pointer()->as_address()));
2952     } else {
2953       ShouldNotReachHere();
2954     }
2955 
2956 #ifndef _LP64
2957   } else if(opr1->is_single_fpu() || opr1->is_double_fpu()) {
2958     assert(opr1->is_fpu_register() && opr1->fpu() == 0, "currently left-hand side must be on TOS (relax this restriction)");
2959     assert(opr2->is_fpu_register(), "both must be registers");
2960     __ fcmp(noreg, opr2->fpu(), op->fpu_pop_count() > 0, op->fpu_pop_count() > 1);
2961 #endif // LP64
2962 
2963   } else if (opr1->is_address() && opr2->is_constant()) {
2964     LIR_Const* c = opr2->as_constant_ptr();
2965 #ifdef _LP64
2966     if (is_reference_type(c->type())) {
2967       assert(condition == lir_cond_equal || condition == lir_cond_notEqual, "need to reverse");
2968       __ movoop(rscratch1, c->as_jobject());
2969     }
2970 #endif // LP64
2971     if (op->info() != NULL) {
2972       add_debug_info_for_null_check_here(op->info());
2973     }
2974     // special case: address - constant
2975     LIR_Address* addr = opr1->as_address_ptr();
2976     if (c->type() == T_INT) {
2977       __ cmpl(as_Address(addr), c->as_jint());
2978     } else if (is_reference_type(c->type())) {
2979 #ifdef _LP64
2980       // %%% Make this explode if addr isn't reachable until we figure out a
2981       // better strategy by giving noreg as the temp for as_Address
2982       __ cmpoop(rscratch1, as_Address(addr, noreg));
2983 #else
2984       __ cmpoop(as_Address(addr), c->as_jobject());
2985 #endif // _LP64
2986     } else {
2987       ShouldNotReachHere();
2988     }
2989 
2990   } else {
2991     ShouldNotReachHere();
2992   }

3799           __ addptr(data_addr, DataLayout::counter_increment);
3800           return;
3801         }
3802       }
3803 
3804       // Receiver type not found in profile data; select an empty slot
3805 
3806       // Note that this is less efficient than it should be because it
3807       // always does a write to the receiver part of the
3808       // VirtualCallData rather than just the first time
3809       for (i = 0; i < VirtualCallData::row_limit(); i++) {
3810         ciKlass* receiver = vc_data->receiver(i);
3811         if (receiver == NULL) {
3812           Address recv_addr(mdo, md->byte_offset_of_slot(data, VirtualCallData::receiver_offset(i)));
3813           __ mov_metadata(recv_addr, known_klass->constant_encoding());
3814           Address data_addr(mdo, md->byte_offset_of_slot(data, VirtualCallData::receiver_count_offset(i)));
3815           __ addptr(data_addr, DataLayout::counter_increment);
3816           return;
3817         }
3818       }
3819     } else {
3820       __ load_klass(recv, recv);
3821       Label update_done;
3822       type_profile_helper(mdo, md, data, recv, &update_done);
3823       // Receiver did not match any saved receiver and there is no empty row for it.
3824       // Increment total counter to indicate polymorphic case.
3825       __ addptr(counter_addr, DataLayout::counter_increment);
3826 
3827       __ bind(update_done);
3828     }
3829   } else {
3830     // Static call
3831     __ addptr(counter_addr, DataLayout::counter_increment);
3832   }
3833 }

3835 void LIR_Assembler::emit_profile_type(LIR_OpProfileType* op) {
3836   Register obj = op->obj()->as_register();
3837   Register tmp = op->tmp()->as_pointer_register();
3838   Address mdo_addr = as_Address(op->mdp()->as_address_ptr());
3839   ciKlass* exact_klass = op->exact_klass();
3840   intptr_t current_klass = op->current_klass();
3841   bool not_null = op->not_null();
3842   bool no_conflict = op->no_conflict();
3843 
3844   Label update, next, none;
3845 
3846   bool do_null = !not_null;
3847   bool exact_klass_set = exact_klass != NULL && ciTypeEntries::valid_ciklass(current_klass) == exact_klass;
3848   bool do_update = !TypeEntries::is_type_unknown(current_klass) && !exact_klass_set;
3849 
3850   assert(do_null || do_update, "why are we here?");
3851   assert(!TypeEntries::was_null_seen(current_klass) || do_update, "why are we here?");
3852 
3853   __ verify_oop(obj);
3854 
3855   if (tmp != obj) {
3856     __ mov(tmp, obj);
3857   }
3858   if (do_null) {
3859     __ testptr(tmp, tmp);
3860     __ jccb(Assembler::notZero, update);
3861     if (!TypeEntries::was_null_seen(current_klass)) {
3862       __ orptr(mdo_addr, TypeEntries::null_seen);
3863     }
3864     if (do_update) {
3865 #ifndef ASSERT
3866       __ jmpb(next);
3867     }
3868 #else
3869       __ jmp(next);
3870     }
3871   } else {
3872     __ testptr(tmp, tmp);
3873     __ jcc(Assembler::notZero, update);
3874     __ stop("unexpect null obj");
3875 #endif
3876   }
3877 
3878   __ bind(update);
3879 
3880   if (do_update) {
3881 #ifdef ASSERT
3882     if (exact_klass != NULL) {
3883       Label ok;
3884       __ load_klass(tmp, tmp);
3885       __ push(tmp);
3886       __ mov_metadata(tmp, exact_klass->constant_encoding());
3887       __ cmpptr(tmp, Address(rsp, 0));
3888       __ jcc(Assembler::equal, ok);
3889       __ stop("exact klass and actual klass differ");
3890       __ bind(ok);
3891       __ pop(tmp);
3892     }
3893 #endif
3894     if (!no_conflict) {
3895       if (exact_klass == NULL || TypeEntries::is_type_none(current_klass)) {
3896         if (exact_klass != NULL) {
3897           __ mov_metadata(tmp, exact_klass->constant_encoding());
3898         } else {
3899           __ load_klass(tmp, tmp);
3900         }
3901 
3902         __ xorptr(tmp, mdo_addr);
3903         __ testptr(tmp, TypeEntries::type_klass_mask);
3904         // klass seen before, nothing to do. The unknown bit may have been
3905         // set already but no need to check.
3906         __ jccb(Assembler::zero, next);
3907 
3908         __ testptr(tmp, TypeEntries::type_unknown);
3909         __ jccb(Assembler::notZero, next); // already unknown. Nothing to do anymore.
3910 
3911         if (TypeEntries::is_type_none(current_klass)) {
3912           __ cmpptr(mdo_addr, 0);
3913           __ jccb(Assembler::equal, none);
3914           __ cmpptr(mdo_addr, TypeEntries::null_seen);
3915           __ jccb(Assembler::equal, none);
3916           // There is a chance that the checks above (re-reading profiling
3917           // data from memory) fail if another thread has just set the
3918           // profiling to this obj's klass
3919           __ xorptr(tmp, mdo_addr);
3920           __ testptr(tmp, TypeEntries::type_klass_mask);
3921           __ jccb(Assembler::zero, next);
3922         }
3923       } else {
3924         assert(ciTypeEntries::valid_ciklass(current_klass) != NULL &&
3925                ciTypeEntries::valid_ciklass(current_klass) != exact_klass, "conflict only");
3926 
3927         __ movptr(tmp, mdo_addr);
3928         __ testptr(tmp, TypeEntries::type_unknown);
3929         __ jccb(Assembler::notZero, next); // already unknown. Nothing to do anymore.
3930       }
3931 
3932       // different than before. Cannot keep accurate profile.
3933       __ orptr(mdo_addr, TypeEntries::type_unknown);
3934 
3935       if (TypeEntries::is_type_none(current_klass)) {
3936         __ jmpb(next);
3937 
3938         __ bind(none);
3939         // first time here. Set profile type.
3940         __ movptr(mdo_addr, tmp);
3941       }
3942     } else {
3943       // There's a single possible klass at this profile point
3944       assert(exact_klass != NULL, "should be");
3945       if (TypeEntries::is_type_none(current_klass)) {
3946         __ mov_metadata(tmp, exact_klass->constant_encoding());
3947         __ xorptr(tmp, mdo_addr);
3948         __ testptr(tmp, TypeEntries::type_klass_mask);
3949 #ifdef ASSERT
3950         __ jcc(Assembler::zero, next);
3951 
3952         {
3953           Label ok;
3954           __ push(tmp);
3955           __ cmpptr(mdo_addr, 0);
3956           __ jcc(Assembler::equal, ok);
3957           __ cmpptr(mdo_addr, TypeEntries::null_seen);
3958           __ jcc(Assembler::equal, ok);
3959           // may have been set by another thread
3960           __ mov_metadata(tmp, exact_klass->constant_encoding());
3961           __ xorptr(tmp, mdo_addr);
3962           __ testptr(tmp, TypeEntries::type_mask);
3963           __ jcc(Assembler::zero, ok);
3964 
3965           __ stop("unexpected profiling mismatch");
3966           __ bind(ok);
3967           __ pop(tmp);
3968         }
3969 #else
3970         __ jccb(Assembler::zero, next);
3971 #endif
3972         // first time here. Set profile type.
3973         __ movptr(mdo_addr, tmp);
3974       } else {
3975         assert(ciTypeEntries::valid_ciklass(current_klass) != NULL &&
3976                ciTypeEntries::valid_ciklass(current_klass) != exact_klass, "inconsistent");
3977 
3978         __ movptr(tmp, mdo_addr);
3979         __ testptr(tmp, TypeEntries::type_unknown);
3980         __ jccb(Assembler::notZero, next); // already unknown. Nothing to do anymore.
3981 
3982         __ orptr(mdo_addr, TypeEntries::type_unknown);
3983       }
3984     }
3985 
3986     __ bind(next);
3987   }
3988 }
