 276 // Convert Nodes to instruction bits and pass off to the VM
 277 void PhaseOutput::Output() {
 278   // RootNode goes
 279   assert( C->cfg()->get_root_block()->number_of_nodes() == 0, "" );
 280 
 281   // The number of new nodes (mostly MachNop) is proportional to
 282   // the number of java calls and inner loops which are aligned.
 283   if ( C->check_node_count((NodeLimitFudgeFactor + C->java_calls()*3 +
 284                             C->inner_loops()*(OptoLoopAlignment-1)),
 285                            "out of nodes before code generation" ) ) {
 286     return;
 287   }
 288   // Make sure I can find the Start Node
 289   Block *entry = C->cfg()->get_block(1);
 290   Block *broot = C->cfg()->get_root_block();
 291 
 292   const StartNode *start = entry->head()->as_Start();
 293 
 294   // Replace StartNode with prolog
 295   Label verified_entry;
 296   MachPrologNode* prolog = new MachPrologNode(&verified_entry);
 297   entry->map_node(prolog, 0);
 298   C->cfg()->map_node_to_block(prolog, entry);
 299   C->cfg()->unmap_node_from_block(start); // start is no longer in any block
 300 
 301   // Virtual methods need an unverified entry point
 302   if (C->is_osr_compilation()) {
 303     if (PoisonOSREntry) {
 304       // TODO: Should use a ShouldNotReachHereNode...
 305       C->cfg()->insert( broot, 0, new MachBreakpointNode() );
 306     }
 307   } else {
 308     if (C->method()) {
 309       if (C->method()->has_scalarized_args()) {
 310         // Add entry point to unpack all value type arguments
 311         C->cfg()->insert(broot, 0, new MachVEPNode(&verified_entry, /* verified */ true, /* receiver_only */ false));
 312         if (!C->method()->is_static()) {
 313           // Add verified/unverified entry points to only unpack value type receiver at interface calls
 314           C->cfg()->insert(broot, 0, new MachVEPNode(&verified_entry, /* verified */ false, /* receiver_only */ false));
 315           C->cfg()->insert(broot, 0, new MachVEPNode(&verified_entry, /* verified */ true,  /* receiver_only */ true));
 316           C->cfg()->insert(broot, 0, new MachVEPNode(&verified_entry, /* verified */ false, /* receiver_only */ true));
 317         }
 318       } else if (!C->method()->is_static()) {
 319         // Insert unvalidated entry point
 320         C->cfg()->insert(broot, 0, new MachUEPNode());
 321       }
 322     }
 323   }
 324 
 325   // Break before main entry point
 326   if ((C->method() && C->directive()->BreakAtExecuteOption) ||
 327       (OptoBreakpoint && C->is_method_compilation())       ||
 328       (OptoBreakpointOSR && C->is_osr_compilation())       ||
 329       (OptoBreakpointC2R && !C->method())                   ) {
 330     // checking for C->method() means that OptoBreakpoint does not apply to
 331     // runtime stubs or frame converters
 332     C->cfg()->insert( entry, 1, new MachBreakpointNode() );
 333   }
 334 
 335   // Insert epilogs before every return
 336   for (uint i = 0; i < C->cfg()->number_of_blocks(); i++) {
 337     Block* block = C->cfg()->get_block(i);
 338     if (!block->is_connector() && block->non_connector_successor(0) == C->cfg()->get_root_block()) { // Found a program exit point?
 339       Node* m = block->end();
 340       if (m->is_Mach() && m->as_Mach()->ideal_Opcode() != Op_Halt) {
 341         MachEpilogNode* epilog = new MachEpilogNode(m->as_Mach()->ideal_Opcode() == Op_Return);
 342         block->add_inst(epilog);
 343         C->cfg()->map_node_to_block(epilog, block);
 344       }
 345     }
 346   }
 347 
 348   // Keeper of sizing aspects
 349   _buf_sizes = BufferSizingData();
 350 
 351   // Initialize code buffer
 352   estimate_buffer_size(_buf_sizes._const);
 353   if (C->failing()) return;
 354 
 355   // Pre-compute the length of blocks and replace
 356   // long branches with short if machine supports it.
 357   // Must be done before ScheduleAndBundle due to SPARC delay slots
 358   uint* blk_starts = NEW_RESOURCE_ARRAY(uint, C->cfg()->number_of_blocks() + 1);
 359   blk_starts[0] = 0;
 360   shorten_branches(blk_starts);
 361 
 362   if (!C->is_osr_compilation() && C->has_scalarized_args()) {
 363     // Compute the offsets of the entry points required by the value type calling convention
 364     if (!C->method()->is_static()) {
 365       // We have entries at the beginning of the method, implemented by the first 4 nodes.
 366       // Entry                     (unverified) @ offset 0
 367       // Verified_Value_Entry_RO
 368       // Value_Entry               (unverified)
 369       // Verified_Value_Entry
 370       uint offset = 0;
 371       _code_offsets.set_value(CodeOffsets::Entry, offset);
 372 
 373       offset += ((MachVEPNode*)broot->get_node(0))->size(C->regalloc());
 374       _code_offsets.set_value(CodeOffsets::Verified_Value_Entry_RO, offset);
 375 
 376       offset += ((MachVEPNode*)broot->get_node(1))->size(C->regalloc());
 377       _code_offsets.set_value(CodeOffsets::Value_Entry, offset);
 378 
 379       offset += ((MachVEPNode*)broot->get_node(2))->size(C->regalloc());
 380       _code_offsets.set_value(CodeOffsets::Verified_Value_Entry, offset);
 381     } else {
 382       _code_offsets.set_value(CodeOffsets::Entry, -1); // will be patched later
 383       _code_offsets.set_value(CodeOffsets::Verified_Value_Entry, 0);
 384     }
 385   }
 386 
 387   ScheduleAndBundle();
 388   if (C->failing()) {
 389     return;
 390   }
 391 
 392   perform_mach_node_analysis();
 393 
 394   // Complete sizing of codebuffer
 395   CodeBuffer* cb = init_buffer();
 396   if (cb == NULL || C->failing()) {
 397     return;
 398   }
 399 
 400   BuildOopMaps();
 401 
 402   if (C->failing())  {
 403     return;
 404   }
 405 
 406   fill_buffer(cb, blk_starts);
 407 }

 409 bool PhaseOutput::need_stack_bang(int frame_size_in_bytes) const {
 410   // Determine if we need to generate a stack overflow check.
 411   // Do it if the method is not a stub function and
 412   // has java calls or has frame size > vm_page_size/8.
 413   // The debug VM checks that deoptimization doesn't trigger an
 414   // unexpected stack overflow (compiled method stack banging should
 415   // guarantee it doesn't happen) so we always need the stack bang in
 416   // a debug VM.
 417   return (UseStackBanging && C->stub_function() == NULL &&
 418           (C->has_java_calls() || frame_size_in_bytes > os::vm_page_size()>>3
 419            DEBUG_ONLY(|| true)));
 420 }

 422 bool PhaseOutput::need_register_stack_bang() const {
 423   // Determine if we need to generate a register stack overflow check.
 424   // This is only used on architectures which have split register
 425   // and memory stacks (ie. IA64).
 426   // Bang if the method is not a stub function and has java calls
 427   return (C->stub_function() == NULL && C->has_java_calls());
 428 }

 431 // Compute the size of first NumberOfLoopInstrToAlign instructions at the top
 432 // of a loop. When aligning a loop we need to provide enough instructions
 433 // in cpu's fetch buffer to feed decoders. The loop alignment could be
 434 // avoided if we have enough instructions in fetch buffer at the head of a loop.
 435 // By default, the size is set to 999999 by Block's constructor so that
 436 // a loop will be aligned if the size is not reset here.
 437 //
 438 // Note: Mach instructions could contain several HW instructions
 439 // so the size is estimated only.
 440 //
 441 void PhaseOutput::compute_loop_first_inst_sizes() {
 442   // The next condition is used to gate the loop alignment optimization.
 443   // Don't aligned a loop if there are enough instructions at the head of a loop
 444   // or alignment padding is larger then MaxLoopPad. By default, MaxLoopPad
 445   // is equal to OptoLoopAlignment-1 except on new Intel cpus, where it is
 446   // equal to 11 bytes which is the largest address NOP instruction.
 447   if (MaxLoopPad < OptoLoopAlignment - 1) {
 448     uint last_block = C->cfg()->number_of_blocks() - 1;
 449     for (uint i = 1; i <= last_block; i++) {
 450       Block* block = C->cfg()->get_block(i);
 451       // Check the first loop's block which requires an alignment.
 452       if (block->loop_alignment() > (uint)relocInfo::addr_unit()) {
 453         uint sum_size = 0;
 454         uint inst_cnt = NumberOfLoopInstrToAlign;
 455         inst_cnt = block->compute_first_inst_size(sum_size, inst_cnt, C->regalloc());
 456 
 457         // Check subsequent fallthrough blocks if the loop's first
 458         // block(s) does not have enough instructions.
 459         Block *nb = block;
 460         while(inst_cnt > 0 &&
 461               i < last_block &&
 462               !C->cfg()->get_block(i + 1)->has_loop_alignment() &&
 463               !nb->has_successor(block)) {
 464           i++;
 465           nb = C->cfg()->get_block(i);
 466           inst_cnt  = nb->compute_first_inst_size(sum_size, inst_cnt, C->regalloc());
 467         } // while( inst_cnt > 0 && i < last_block  )
 468 
 469         block->set_first_inst_size(sum_size);
 470       } // f( b->head()->is_Loop() )
 471     } // for( i <= last_block )
 472   } // if( MaxLoopPad < OptoLoopAlignment-1 )
 473 }

 475 // The architecture description provides short branch variants for some long
 476 // branch instructions. Replace eligible long branches with short branches.
 477 void PhaseOutput::shorten_branches(uint* blk_starts) {
 478   // Compute size of each block, method size, and relocation information size
 479   uint nblocks  = C->cfg()->number_of_blocks();
 480 
 481   uint*      jmp_offset = NEW_RESOURCE_ARRAY(uint,nblocks);
 482   uint*      jmp_size   = NEW_RESOURCE_ARRAY(uint,nblocks);
 483   int*       jmp_nidx   = NEW_RESOURCE_ARRAY(int ,nblocks);
 484 
 485   // Collect worst case block paddings
 486   int* block_worst_case_pad = NEW_RESOURCE_ARRAY(int, nblocks);
 487   memset(block_worst_case_pad, 0, nblocks * sizeof(int));
 488 
 489   DEBUG_ONLY( uint *jmp_target = NEW_RESOURCE_ARRAY(uint,nblocks); )
 490   DEBUG_ONLY( uint *jmp_rule = NEW_RESOURCE_ARRAY(uint,nblocks); )
 491 
 492   bool has_short_branch_candidate = false;
 493 
 494   // Initialize the sizes to 0
 495   int code_size  = 0;          // Size in bytes of generated code
 496   int stub_size  = 0;          // Size in bytes of all stub entries
 497   // Size in bytes of all relocation entries, including those in local stubs.
 498   // Start with 2-bytes of reloc info for the unvalidated entry point
 499   int reloc_size = 1;          // Number of relocation entries
 500 
 501   // Make three passes.  The first computes pessimistic blk_starts,
 502   // relative jmp_offset and reloc_size information.  The second performs
 503   // short branch substitution using the pessimistic sizing.  The
 504   // third inserts nops where needed.
 505 
 506   // Step one, perform a pessimistic sizing pass.
 507   uint last_call_adr = max_juint;
 508   uint last_avoid_back_to_back_adr = max_juint;
 509   uint nop_size = (new MachNopNode())->size(C->regalloc());
 510   for (uint i = 0; i < nblocks; i++) { // For all blocks
 511     Block* block = C->cfg()->get_block(i);
 512     _block = block;
 513 
 514     // During short branch replacement, we store the relative (to blk_starts)
 515     // offset of jump in jmp_offset, rather than the absolute offset of jump.
 516     // This is so that we do not need to recompute sizes of all nodes when
 517     // we compute correct blk_starts in our next sizing pass.
 518     jmp_offset[i] = 0;
 519     jmp_size[i]   = 0;
 520     jmp_nidx[i]   = -1;
 521     DEBUG_ONLY( jmp_target[i] = 0; )
 522     DEBUG_ONLY( jmp_rule[i]   = 0; )
 523 
 524     // Sum all instruction sizes to compute block size
 525     uint last_inst = block->number_of_nodes();
 526     uint blk_size = 0;
 527     for (uint j = 0; j < last_inst; j++) {
 528       _index = j;
 529       Node* nj = block->get_node(_index);
 530       // Handle machine instruction nodes
 531       if (nj->is_Mach()) {
 532         MachNode* mach = nj->as_Mach();
 533         blk_size += (mach->alignment_required() - 1) * relocInfo::addr_unit(); // assume worst case padding
 534         reloc_size += mach->reloc();
 535         if (mach->is_MachCall()) {
 536           // add size information for trampoline stub
 537           // class CallStubImpl is platform-specific and defined in the *.ad files.
 538           stub_size  += CallStubImpl::size_call_trampoline();
 539           reloc_size += CallStubImpl::reloc_call_trampoline();
 540 
 541           MachCallNode *mcall = mach->as_MachCall();
 542           // This destination address is NOT PC-relative
 543 
 544           if (mcall->entry_point() != NULL) {
 545             mcall->method_set((intptr_t)mcall->entry_point());
 546           }
 547 
 548           if (mcall->is_MachCallJava() && mcall->as_MachCallJava()->_method) {
 549             stub_size  += CompiledStaticCall::to_interp_stub_size();
 550             reloc_size += CompiledStaticCall::reloc_to_interp_stub();
 551 #if INCLUDE_AOT
 552             stub_size  += CompiledStaticCall::to_aot_stub_size();
 553             reloc_size += CompiledStaticCall::reloc_to_aot_stub();
 554 #endif
 555           }
 556         } else if (mach->is_MachSafePoint()) {
 557           // If call/safepoint are adjacent, account for possible
 558           // nop to disambiguate the two safepoints.
 559           // ScheduleAndBundle() can rearrange nodes in a block,
 560           // check for all offsets inside this block.
 561           if (last_call_adr >= blk_starts[i]) {
 562             blk_size += nop_size;
 563           }
 564         }
 565         if (mach->avoid_back_to_back(MachNode::AVOID_BEFORE)) {
 566           // Nop is inserted between "avoid back to back" instructions.
 567           // ScheduleAndBundle() can rearrange nodes in a block,
 568           // check for all offsets inside this block.
 569           if (last_avoid_back_to_back_adr >= blk_starts[i]) {
 570             blk_size += nop_size;
 571           }
 572         }
 573         if (mach->may_be_short_branch()) {
 574           if (!nj->is_MachBranch()) {
 575 #ifndef PRODUCT
 576             nj->dump(3);
 577 #endif
 578             Unimplemented();
 579           }
 580           assert(jmp_nidx[i] == -1, "block should have only one branch");
 581           jmp_offset[i] = blk_size;
 582           jmp_size[i]   = nj->size(C->regalloc());
 583           jmp_nidx[i]   = j;
 584           has_short_branch_candidate = true;
 585         }
 586       }
 587       blk_size += nj->size(C->regalloc());
 588       // Remember end of call offset
 589       if (nj->is_MachCall() && !nj->is_MachCallLeaf()) {
 590         last_call_adr = blk_starts[i]+blk_size;
 591       }
 592       // Remember end of avoid_back_to_back offset
 593       if (nj->is_Mach() && nj->as_Mach()->avoid_back_to_back(MachNode::AVOID_AFTER)) {
 594         last_avoid_back_to_back_adr = blk_starts[i]+blk_size;
 595       }
 596     }
 597 
 598     // When the next block starts a loop, we may insert pad NOP
 599     // instructions.  Since we cannot know our future alignment,
 600     // assume the worst.
 601     if (i < nblocks - 1) {
 602       Block* nb = C->cfg()->get_block(i + 1);
 603       int max_loop_pad = nb->code_alignment()-relocInfo::addr_unit();
 604       if (max_loop_pad > 0) {
 605         assert(is_power_of_2(max_loop_pad+relocInfo::addr_unit()), "");
 606         // Adjust last_call_adr and/or last_avoid_back_to_back_adr.
 607         // If either is the last instruction in this block, bump by
 608         // max_loop_pad in lock-step with blk_size, so sizing
 609         // calculations in subsequent blocks still can conservatively
 610         // detect that it may the last instruction in this block.
 611         if (last_call_adr == blk_starts[i]+blk_size) {
 612           last_call_adr += max_loop_pad;
 613         }
 614         if (last_avoid_back_to_back_adr == blk_starts[i]+blk_size) {
 615           last_avoid_back_to_back_adr += max_loop_pad;
 616         }
 617         blk_size += max_loop_pad;
 618         block_worst_case_pad[i + 1] = max_loop_pad;
 619       }
 620     }
 621 
 622     // Save block size; update total method size
 623     blk_starts[i+1] = blk_starts[i]+blk_size;
 624   }
 625 
 626   // Step two, replace eligible long jumps.
 627   bool progress = true;
 628   uint last_may_be_short_branch_adr = max_juint;
 629   while (has_short_branch_candidate && progress) {
 630     progress = false;
 631     has_short_branch_candidate = false;
 632     int adjust_block_start = 0;
 633     for (uint i = 0; i < nblocks; i++) {
 634       Block* block = C->cfg()->get_block(i);
 635       int idx = jmp_nidx[i];
 636       MachNode* mach = (idx == -1) ? NULL: block->get_node(idx)->as_Mach();
 637       if (mach != NULL && mach->may_be_short_branch()) {
 638 #ifdef ASSERT
 639         assert(jmp_size[i] > 0 && mach->is_MachBranch(), "sanity");
 640         int j;
 641         // Find the branch; ignore trailing NOPs.
 642         for (j = block->number_of_nodes()-1; j>=0; j--) {
 643           Node* n = block->get_node(j);
 644           if (!n->is_Mach() || n->as_Mach()->ideal_Opcode() != Op_Con)
 645             break;
 646         }
 647         assert(j >= 0 && j == idx && block->get_node(j) == (Node*)mach, "sanity");
 648 #endif
 649         int br_size = jmp_size[i];
 650         int br_offs = blk_starts[i] + jmp_offset[i];
 651 
 652         // This requires the TRUE branch target be in succs[0]
 653         uint bnum = block->non_connector_successor(0)->_pre_order;
 654         int offset = blk_starts[bnum] - br_offs;
 655         if (bnum > i) { // adjust following block's offset
 656           offset -= adjust_block_start;
 657         }
 658 
 659         // This block can be a loop header, account for the padding
 660         // in the previous block.
 661         int block_padding = block_worst_case_pad[i];
 662         assert(i == 0 || block_padding == 0 || br_offs >= block_padding, "Should have at least a padding on top");
 663         // In the following code a nop could be inserted before
 664         // the branch which will increase the backward distance.
 665         bool needs_padding = ((uint)(br_offs - block_padding) == last_may_be_short_branch_adr);
 666         assert(!needs_padding || jmp_offset[i] == 0, "padding only branches at the beginning of block");
 667 
 668         if (needs_padding && offset <= 0)
 669           offset -= nop_size;
 670 
 671         if (C->matcher()->is_short_branch_offset(mach->rule(), br_size, offset)) {
 672           // We've got a winner.  Replace this branch.
 673           MachNode* replacement = mach->as_MachBranch()->short_branch_version();
 674 
 675           // Update the jmp_size.
 676           int new_size = replacement->size(C->regalloc());
 677           int diff     = br_size - new_size;
 678           assert(diff >= (int)nop_size, "short_branch size should be smaller");
 679           // Conservatively take into account padding between
 680           // avoid_back_to_back branches. Previous branch could be
 681           // converted into avoid_back_to_back branch during next
 682           // rounds.
 683           if (needs_padding && replacement->avoid_back_to_back(MachNode::AVOID_BEFORE)) {
 684             jmp_offset[i] += nop_size;
 685             diff -= nop_size;
 686           }
 687           adjust_block_start += diff;
 688           block->map_node(replacement, idx);
 689           mach->subsume_by(replacement, C);
 690           mach = replacement;
 691           progress = true;
 692 
 693           jmp_size[i] = new_size;
 694           DEBUG_ONLY( jmp_target[i] = bnum; );
 695           DEBUG_ONLY( jmp_rule[i] = mach->rule(); );
 696         } else {
 697           // The jump distance is not short, try again during next iteration.
 698           has_short_branch_candidate = true;
 699         }
 700       } // (mach->may_be_short_branch())
 701       if (mach != NULL && (mach->may_be_short_branch() ||
 702                            mach->avoid_back_to_back(MachNode::AVOID_AFTER))) {
 703         last_may_be_short_branch_adr = blk_starts[i] + jmp_offset[i] + jmp_size[i];
 704       }
 705       blk_starts[i+1] -= adjust_block_start;
 706     }
 707   }
 708 
 709 #ifdef ASSERT
 710   for (uint i = 0; i < nblocks; i++) { // For all blocks
 711     if (jmp_target[i] != 0) {
 712       int br_size = jmp_size[i];
 713       int offset = blk_starts[jmp_target[i]]-(blk_starts[i] + jmp_offset[i]);
 714       if (!C->matcher()->is_short_branch_offset(jmp_rule[i], br_size, offset)) {
 715         tty->print_cr("target (%d) - jmp_offset(%d) = offset (%d), jump_size(%d), jmp_block B%d, target_block B%d", blk_starts[jmp_target[i]], blk_starts[i] + jmp_offset[i], offset, br_size, i, jmp_target[i]);
 716       }
 717       assert(C->matcher()->is_short_branch_offset(jmp_rule[i], br_size, offset), "Displacement too large for short jmp");
 718     }
 719   }
 720 #endif
 721 
 722   // Step 3, compute the offsets of all blocks, will be done in fill_buffer()
 723   // after ScheduleAndBundle().
 724 
 725   // ------------------
 726   // Compute size for code buffer
 727   code_size = blk_starts[nblocks];
 728 
 729   // Relocation records
 730   reloc_size += 1;              // Relo entry for exception handler
 731 
 732   // Adjust reloc_size to number of record of relocation info
 733   // Min is 2 bytes, max is probably 6 or 8, with a tax up to 25% for
 734   // a relocation index.
 735   // The CodeBuffer will expand the locs array if this estimate is too low.
 736   reloc_size *= 10 / sizeof(relocInfo);
 737 
 738   _buf_sizes._reloc = reloc_size;
 739   _buf_sizes._code  = code_size;
 740   _buf_sizes._stub  = stub_size;
 741 }

 744 // Create a bit of debug info and append it to the array.  The mapping is from
 745 // Java local or expression stack to constant, register or stack-slot.  For
 746 // doubles, insert 2 mappings and return 1 (to tell the caller that the next
 747 // entry has been taken care of and caller should skip it).
 748 static LocationValue *new_loc_value( PhaseRegAlloc *ra, OptoReg::Name regnum, Location::Type l_type ) {

 752          ? new LocationValue(Location::new_reg_loc(l_type, OptoReg::as_VMReg(regnum)) )
 753          : new LocationValue(Location::new_stk_loc(l_type,  ra->reg2offset(regnum)));
 754 }

 777 void PhaseOutput::FillLocArray( int idx, MachSafePointNode* sfpt, Node *local,
 778                             GrowableArray<ScopeValue*> *array,
 779                             GrowableArray<ScopeValue*> *objs ) {
 780   assert( local, "use _top instead of null" );
 781   if (array->length() != idx) {
 782     assert(array->length() == idx + 1, "Unexpected array count");
 783     // Old functionality:
 784     //   return
 785     // New functionality:
 786     //   Assert if the local is not top. In product mode let the new node
 787     //   override the old entry.
 788     assert(local == C->top(), "LocArray collision");
 789     if (local == C->top()) {
 790       return;
 791     }
 792     array->pop();
 793   }
 794   const Type *t = local->bottom_type();
 795 
 796   // Is it a safepoint scalar object node?
 797   if (local->is_SafePointScalarObject()) {
 798     SafePointScalarObjectNode* spobj = local->as_SafePointScalarObject();
 799 
 800     ObjectValue* sv = sv_for_node_id(objs, spobj->_idx);
 801     if (sv == NULL) {
 802       ciKlass* cik = t->is_oopptr()->klass();
 803       assert(cik->is_instance_klass() ||
 804              cik->is_array_klass(), "Not supported allocation.");
 805       sv = new ObjectValue(spobj->_idx,
 806                            new ConstantOopWriteValue(cik->java_mirror()->constant_encoding()));
 807       set_sv_for_object_node(objs, sv);
 808 
 809       uint first_ind = spobj->first_index(sfpt->jvms());
 810       for (uint i = 0; i < spobj->n_fields(); i++) {
 811         Node* fld_node = sfpt->in(first_ind+i);
 812         (void)FillLocArray(sv->field_values()->length(), sfpt, fld_node, sv->field_values(), objs);
 813       }
 814     }
 815     array->append(sv);
 816     return;
 817   }
 818 
 819   // Grab the register number for the local
 820   OptoReg::Name regnum = C->regalloc()->get_reg_first(local);
 821   if( OptoReg::is_valid(regnum) ) {// Got a register/stack?
 822     // Record the double as two float registers.
 823     // The register mask for such a value always specifies two adjacent
 824     // float registers, with the lower register number even.
 825     // Normally, the allocation of high and low words to these registers
 826     // is irrelevant, because nearly all operations on register pairs
 827     // (e.g., StoreD) treat them as a single unit.
 828     // Here, we assume in addition that the words in these two registers
 829     // stored "naturally" (by operations like StoreD and double stores
 830     // within the interpreter) such that the lower-numbered register
 831     // is written to the lower memory address.  This may seem like
 832     // a machine dependency, but it is not--it is a requirement on
 833     // the author of the <arch>.ad file to ensure that, for every
 834     // even/odd double-register pair to which a double may be allocated,
 835     // the word in the even single-register is stored to the first
 836     // memory word.  (Note that register numbers are completely
 837     // arbitrary, and are not tied to any machine-level encodings.)
 838 #ifdef _LP64
 839     if( t->base() == Type::DoubleBot || t->base() == Type::DoubleCon ) {
 840       array->append(new ConstantIntValue((jint)0));
 841       array->append(new_loc_value( C->regalloc(), regnum, Location::dbl ));
 842     } else if ( t->base() == Type::Long ) {
 843       array->append(new ConstantIntValue((jint)0));
 844       array->append(new_loc_value( C->regalloc(), regnum, Location::lng ));
 845     } else if ( t->base() == Type::RawPtr ) {
 846       // jsr/ret return address which must be restored into a the full
 847       // width 64-bit stack slot.
 848       array->append(new_loc_value( C->regalloc(), regnum, Location::lng ));
 849     }
 850 #else //_LP64
 851 #ifdef SPARC
 852     if (t->base() == Type::Long && OptoReg::is_reg(regnum)) {
 853       // For SPARC we have to swap high and low words for
 854       // long values stored in a single-register (g0-g7).
 855       array->append(new_loc_value( C->regalloc(),              regnum   , Location::normal ));
 856       array->append(new_loc_value( C->regalloc(), OptoReg::add(regnum,1), Location::normal ));
 857     } else
 858 #endif //SPARC
 859     if( t->base() == Type::DoubleBot || t->base() == Type::DoubleCon || t->base() == Type::Long ) {
 860       // Repack the double/long as two jints.
 861       // The convention the interpreter uses is that the second local
 862       // holds the first raw word of the native double representation.
 863       // This is actually reasonable, since locals and stack arrays
 864       // grow downwards in all implementations.
 865       // (If, on some machine, the interpreter's Java locals or stack
 866       // were to grow upwards, the embedded doubles would be word-swapped.)
 867       array->append(new_loc_value( C->regalloc(), OptoReg::add(regnum,1), Location::normal ));
 868       array->append(new_loc_value( C->regalloc(),              regnum   , Location::normal ));
 869     }
 870 #endif //_LP64
 871     else if( (t->base() == Type::FloatBot || t->base() == Type::FloatCon) &&
 872              OptoReg::is_reg(regnum) ) {
 873       array->append(new_loc_value( C->regalloc(), regnum, Matcher::float_in_double()
 874                                                       ? Location::float_in_dbl : Location::normal ));
 875     } else if( t->base() == Type::Int && OptoReg::is_reg(regnum) ) {
 876       array->append(new_loc_value( C->regalloc(), regnum, Matcher::int_in_long
 877                                                       ? Location::int_in_long : Location::normal ));
 878     } else if( t->base() == Type::NarrowOop ) {
 879       array->append(new_loc_value( C->regalloc(), regnum, Location::narrowoop ));
 880     } else {
 881       array->append(new_loc_value( C->regalloc(), regnum, C->regalloc()->is_oop(local) ? Location::oop : Location::normal ));
 882     }
 883     return;
 884   }
 885 
 886   // No register.  It must be constant data.
 887   switch (t->base()) {
 888     case Type::Half:              // Second half of a double
 889       ShouldNotReachHere();       // Caller should skip 2nd halves
 890       break;
 891     case Type::AnyPtr:
 892       array->append(new ConstantOopWriteValue(NULL));
 893       break;
 894     case Type::AryPtr:
 895     case Type::InstPtr:          // fall through
 896       array->append(new ConstantOopWriteValue(t->isa_oopptr()->const_oop()->constant_encoding()));
 897       break;
 898     case Type::NarrowOop:
 899       if (t == TypeNarrowOop::NULL_PTR) {
 900         array->append(new ConstantOopWriteValue(NULL));
 901       } else {
 902         array->append(new ConstantOopWriteValue(t->make_ptr()->isa_oopptr()->const_oop()->constant_encoding()));
 903       }
 904       break;
 905     case Type::Int:
 906       array->append(new ConstantIntValue(t->is_int()->get_con()));
 907       break;
 908     case Type::RawPtr:
 909       // A return address (T_ADDRESS).
 910       assert((intptr_t)t->is_ptr()->get_con() < (intptr_t)0x10000, "must be a valid BCI");
 911 #ifdef _LP64
 912       // Must be restored to the full-width 64-bit stack slot.
 913       array->append(new ConstantLongValue(t->is_ptr()->get_con()));
 914 #else
 915       array->append(new ConstantIntValue(t->is_ptr()->get_con()));
 916 #endif
 917       break;
 918     case Type::FloatCon: {
 919       float f = t->is_float_constant()->getf();
 920       array->append(new ConstantIntValue(jint_cast(f)));
 921       break;
 922     }
 923     case Type::DoubleCon: {
 924       jdouble d = t->is_double_constant()->getd();
 925 #ifdef _LP64
 926       array->append(new ConstantIntValue((jint)0));
 927       array->append(new ConstantDoubleValue(d));
 928 #else
 929       // Repack the double as two jints.
 930     // The convention the interpreter uses is that the second local
 931     // holds the first raw word of the native double representation.
 932     // This is actually reasonable, since locals and stack arrays
 933     // grow downwards in all implementations.
 934     // (If, on some machine, the interpreter's Java locals or stack
 935     // were to grow upwards, the embedded doubles would be word-swapped.)
 936     jlong_accessor acc;
 937     acc.long_value = jlong_cast(d);
 938     array->append(new ConstantIntValue(acc.words[1]));
 939     array->append(new ConstantIntValue(acc.words[0]));
 940 #endif
 941       break;
 942     }
 943     case Type::Long: {
 944       jlong d = t->is_long()->get_con();
 945 #ifdef _LP64
 946       array->append(new ConstantIntValue((jint)0));
 947       array->append(new ConstantLongValue(d));
 948 #else
 949       // Repack the long as two jints.
 950     // The convention the interpreter uses is that the second local
 951     // holds the first raw word of the native double representation.
 952     // This is actually reasonable, since locals and stack arrays
 953     // grow downwards in all implementations.
 954     // (If, on some machine, the interpreter's Java locals or stack
 955     // were to grow upwards, the embedded doubles would be word-swapped.)
 956     jlong_accessor acc;
 957     acc.long_value = d;
 958     array->append(new ConstantIntValue(acc.words[1]));
 959     array->append(new ConstantIntValue(acc.words[0]));
 960 #endif
 961       break;
 962     }
 963     case Type::Top:               // Add an illegal value here
 964       array->append(new LocationValue(Location()));
 965       break;
 966     default:
 967       ShouldNotReachHere();
 968       break;
 969   }
 970 }

 978 //--------------------------Process_OopMap_Node--------------------------------
 979 void PhaseOutput::Process_OopMap_Node(MachNode *mach, int current_offset) {
 980   // Handle special safepoint nodes for synchronization
 981   MachSafePointNode *sfn   = mach->as_MachSafePoint();
 982   MachCallNode      *mcall;
 983 
 984   int safepoint_pc_offset = current_offset;
 985   bool is_method_handle_invoke = false;
 986   bool return_oop = false;
 987   bool return_vt = false;
 988 
 989   // Add the safepoint in the DebugInfoRecorder
 990   if( !mach->is_MachCall() ) {
 991     mcall = NULL;
 992     C->debug_info()->add_safepoint(safepoint_pc_offset, sfn->_oop_map);
 993   } else {
 994     mcall = mach->as_MachCall();
 995 
 996     // Is the call a MethodHandle call?
 997     if (mcall->is_MachCallJava()) {
 998       if (mcall->as_MachCallJava()->_method_handle_invoke) {
 999         assert(C->has_method_handle_invokes(), "must have been set during call generation");
1000         is_method_handle_invoke = true;
1001       }
1002     }
1003 
1004     // Check if a call returns an object.
1005     if (mcall->returns_pointer() || mcall->returns_vt()) {
1006       return_oop = true;
1007     }
1008     if (mcall->returns_vt()) {
1009       return_vt = true;
1010     }
1011     safepoint_pc_offset += mcall->ret_addr_offset();
1012     C->debug_info()->add_safepoint(safepoint_pc_offset, mcall->_oop_map);
1013   }
1014 
1015   // Loop over the JVMState list to add scope information
1016   // Do not skip safepoints with a NULL method, they need monitor info
1017   JVMState* youngest_jvms = sfn->jvms();
1018   int max_depth = youngest_jvms->depth();
1019 
1020   // Allocate the object pool for scalar-replaced objects -- the map from
1021   // small-integer keys (which can be recorded in the local and ostack
1022   // arrays) to descriptions of the object state.
1023   GrowableArray<ScopeValue*> *objs = new GrowableArray<ScopeValue*>();
1024 
1025   // Visit scopes from oldest to youngest.
1026   for (int depth = 1; depth <= max_depth; depth++) {
1027     JVMState* jvms = youngest_jvms->of_depth(depth);
1028     int idx;
1029     ciMethod* method = jvms->has_method() ? jvms->method() : NULL;
1030     // Safepoints that do not have method() set only provide oop-map and monitor info
1031     // to support GC; these do not support deoptimization.
1032     int num_locs = (method == NULL) ? 0 : jvms->loc_size();
1033     int num_exps = (method == NULL) ? 0 : jvms->stk_size();
1034     int num_mon  = jvms->nof_monitors();
1035     assert(method == NULL || jvms->bci() < 0 || num_locs == method->max_locals(),
1036            "JVMS local count must match that of the method");
1037 
1038     // Add Local and Expression Stack Information
1039 
1040     // Insert locals into the locarray
1041     GrowableArray<ScopeValue*> *locarray = new GrowableArray<ScopeValue*>(num_locs);
1042     for( idx = 0; idx < num_locs; idx++ ) {
1043       FillLocArray( idx, sfn, sfn->local(jvms, idx), locarray, objs );
1044     }
1045 
1046     // Insert expression stack entries into the exparray
1047     GrowableArray<ScopeValue*> *exparray = new GrowableArray<ScopeValue*>(num_exps);
1048     for( idx = 0; idx < num_exps; idx++ ) {
1049       FillLocArray( idx,  sfn, sfn->stack(jvms, idx), exparray, objs );
1050     }
1051 
1052     // Add in mappings of the monitors
1053     assert( !method ||
1054             !method->is_synchronized() ||
1055             method->is_native() ||
1056             num_mon > 0 ||
1057             !GenerateSynchronizationCode,
1058             "monitors must always exist for synchronized methods");
1059 
1060     // Build the growable array of ScopeValues for exp stack
1061     GrowableArray<MonitorValue*> *monarray = new GrowableArray<MonitorValue*>(num_mon);
1062 
1063     // Loop over monitors and insert into array
1064     for (idx = 0; idx < num_mon; idx++) {
1065       // Grab the node that defines this monitor
1066       Node* box_node = sfn->monitor_box(jvms, idx);
1067       Node* obj_node = sfn->monitor_obj(jvms, idx);
1068 
1069       // Create ScopeValue for object
1070       ScopeValue *scval = NULL;
1071 
1072       if (obj_node->is_SafePointScalarObject()) {
1073         SafePointScalarObjectNode* spobj = obj_node->as_SafePointScalarObject();
1074         scval = PhaseOutput::sv_for_node_id(objs, spobj->_idx);
1075         if (scval == NULL) {
1076           const Type *t = spobj->bottom_type();
1077           ciKlass* cik = t->is_oopptr()->klass();
1078           assert(cik->is_instance_klass() ||
1079                  cik->is_array_klass(), "Not supported allocation.");
1080           ObjectValue* sv = new ObjectValue(spobj->_idx,
1081                                             new ConstantOopWriteValue(cik->java_mirror()->constant_encoding()));
1082           PhaseOutput::set_sv_for_object_node(objs, sv);
1083 
1084           uint first_ind = spobj->first_index(youngest_jvms);
1085           for (uint i = 0; i < spobj->n_fields(); i++) {
1086             Node* fld_node = sfn->in(first_ind+i);
1087             (void)FillLocArray(sv->field_values()->length(), sfn, fld_node, sv->field_values(), objs);
1088           }
1089           scval = sv;
1090         }
1091       } else if (!obj_node->is_Con()) {
1092         OptoReg::Name obj_reg = C->regalloc()->get_reg_first(obj_node);
1093         if( obj_node->bottom_type()->base() == Type::NarrowOop ) {
1094           scval = new_loc_value( C->regalloc(), obj_reg, Location::narrowoop );
1095         } else {
1096           scval = new_loc_value( C->regalloc(), obj_reg, Location::oop );
1097         }
1098       } else {
1099         const TypePtr *tp = obj_node->get_ptr_type();
1100         scval = new ConstantOopWriteValue(tp->is_oopptr()->const_oop()->constant_encoding());
1101       }
1102 
1103       OptoReg::Name box_reg = BoxLockNode::reg(box_node);
1104       Location basic_lock = Location::new_stk_loc(Location::normal,C->regalloc()->reg2offset(box_reg));
1105       bool eliminated = (box_node->is_BoxLock() && box_node->as_BoxLock()->is_eliminated());
1106       monarray->append(new MonitorValue(scval, basic_lock, eliminated));
1107     }
1108 
1109     // We dump the object pool first, since deoptimization reads it in first.
1110     C->debug_info()->dump_object_pool(objs);
1111 
1112     // Build first class objects to pass to scope
1113     DebugToken *locvals = C->debug_info()->create_scope_values(locarray);
1114     DebugToken *expvals = C->debug_info()->create_scope_values(exparray);
1115     DebugToken *monvals = C->debug_info()->create_monitor_values(monarray);
1116 
1117     // Make method available for all Safepoints
1118     ciMethod* scope_method = method ? method : C->method();
1119     // Describe the scope here
1120     assert(jvms->bci() >= InvocationEntryBci && jvms->bci() <= 0x10000, "must be a valid or entry BCI");
1121     assert(!jvms->should_reexecute() || depth == max_depth, "reexecute allowed only for the youngest");
1122     // Now we can describe the scope.
1123     methodHandle null_mh;
1124     bool rethrow_exception = false;
1125     C->debug_info()->describe_scope(safepoint_pc_offset, null_mh, scope_method, jvms->bci(), jvms->should_reexecute(), rethrow_exception, is_method_handle_invoke, return_oop, return_vt, locvals, expvals, monvals);
1126   } // End jvms loop
1127 
1128   // Mark the end of the scope set.
1129   C->debug_info()->end_safepoint(safepoint_pc_offset);
1130 }

1134 // A simplified version of Process_OopMap_Node, to handle non-safepoints.
1135 class NonSafepointEmitter {
1136     Compile*  C;
1137     JVMState* _pending_jvms;
1138     int       _pending_offset;
1139 
1140     void emit_non_safepoint();
1141 
1142  public:
1143     NonSafepointEmitter(Compile* compile) {
1144       this->C = compile;
1145       _pending_jvms = NULL;
1146       _pending_offset = 0;
1147     }
1148 
1149     void observe_instruction(Node* n, int pc_offset) {
1150       if (!C->debug_info()->recording_non_safepoints())  return;
1151 
1152       Node_Notes* nn = C->node_notes_at(n->_idx);
1153       if (nn == NULL || nn->jvms() == NULL)  return;
1154       if (_pending_jvms != NULL &&
1155           _pending_jvms->same_calls_as(nn->jvms())) {
1156         // Repeated JVMS?  Stretch it up here.
1157         _pending_offset = pc_offset;
1158       } else {
1159         if (_pending_jvms != NULL &&
1160             _pending_offset < pc_offset) {
1161           emit_non_safepoint();
1162         }
1163         _pending_jvms = NULL;
1164         if (pc_offset > C->debug_info()->last_pc_offset()) {
1165           // This is the only way _pending_jvms can become non-NULL:
1166           _pending_jvms = nn->jvms();
1167           _pending_offset = pc_offset;
1168         }
1169       }
1170     }
1171 
1172     // Stay out of the way of real safepoints:
1173     void observe_safepoint(JVMState* jvms, int pc_offset) {
1174       if (_pending_jvms != NULL &&
1175           !_pending_jvms->same_calls_as(jvms) &&
1176           _pending_offset < pc_offset) {
1177         emit_non_safepoint();
1178       }
1179       _pending_jvms = NULL;
1180     }
1181 
1182     void flush_at_end() {
1183       if (_pending_jvms != NULL) {
1184         emit_non_safepoint();
1185       }
1186       _pending_jvms = NULL;
1187     }
1188 };

1190 void NonSafepointEmitter::emit_non_safepoint() {
1191   JVMState* youngest_jvms = _pending_jvms;
1192   int       pc_offset     = _pending_offset;
1193 
1194   // Clear it now:
1195   _pending_jvms = NULL;
1196 
1197   DebugInformationRecorder* debug_info = C->debug_info();
1198   assert(debug_info->recording_non_safepoints(), "sanity");
1199 
1200   debug_info->add_non_safepoint(pc_offset);
1201   int max_depth = youngest_jvms->depth();
1202 
1203   // Visit scopes from oldest to youngest.
1204   for (int depth = 1; depth <= max_depth; depth++) {
1205     JVMState* jvms = youngest_jvms->of_depth(depth);
1206     ciMethod* method = jvms->has_method() ? jvms->method() : NULL;
1207     assert(!jvms->should_reexecute() || depth==max_depth, "reexecute allowed only for the youngest");
1208     methodHandle null_mh;
1209     debug_info->describe_scope(pc_offset, null_mh, method, jvms->bci(), jvms->should_reexecute());
1210   }
1211 
1212   // Mark the end of the scope set.
1213   debug_info->end_non_safepoint(pc_offset);
1214 }

1216 //------------------------------init_buffer------------------------------------
1217 void PhaseOutput::estimate_buffer_size(int& const_req) {
1218 
1219   // Set the initially allocated size
1220   const_req = initial_const_capacity;
1221 
1222   // The extra spacing after the code is necessary on some platforms.
1223   // Sometimes we need to patch in a jump after the last instruction,
1224   // if the nmethod has been deoptimized.  (See 4932387, 4894843.)
1225 
1226   // Compute the byte offset where we can store the deopt pc.
1227   if (C->fixed_slots() != 0) {
1228     _orig_pc_slot_offset_in_bytes = C->regalloc()->reg2offset(OptoReg::stack2reg(_orig_pc_slot));
1229   }
1230   if (C->needs_stack_repair()) {
1231     // Compute the byte offset of the stack increment value
1232     _sp_inc_slot_offset_in_bytes = C->regalloc()->reg2offset(OptoReg::stack2reg(_sp_inc_slot));
1233   }
1234 
1235   // Compute prolog code size
1236   _method_size = 0;
1237   _frame_slots = OptoReg::reg2stack(C->matcher()->_old_SP) + C->regalloc()->_framesize;
1238 #if defined(IA64) && !defined(AIX)
1239   if (save_argument_registers()) {
1240     // 4815101: this is a stub with implicit and unknown precision fp args.
1241     // The usual spill mechanism can only generate stfd's in this case, which
1242     // doesn't work if the fp reg to spill contains a single-precision denorm.
1243     // Instead, we hack around the normal spill mechanism using stfspill's and
1244     // ldffill's in the MachProlog and MachEpilog emit methods.  We allocate
1245     // space here for the fp arg regs (f8-f15) we're going to thusly spill.
1246     //
1247     // If we ever implement 16-byte 'registers' == stack slots, we can
1248     // get rid of this hack and have SpillCopy generate stfspill/ldffill
1249     // instead of stfd/stfs/ldfd/ldfs.
1250     _frame_slots += 8*(16/BytesPerInt);
1251   }
1252 #endif
1253   assert(_frame_slots >= 0 && _frame_slots < 1000000, "sanity check");
1254 
1255   if (C->has_mach_constant_base_node()) {
1256     uint add_size = 0;
1257     // Fill the constant table.
1258     // Note:  This must happen before shorten_branches.
1259     for (uint i = 0; i < C->cfg()->number_of_blocks(); i++) {
1260       Block* b = C->cfg()->get_block(i);
1261 
1262       for (uint j = 0; j < b->number_of_nodes(); j++) {
1263         Node* n = b->get_node(j);
1264 
1265         // If the node is a MachConstantNode evaluate the constant
1266         // value section.
1267         if (n->is_MachConstant()) {
1268           MachConstantNode* machcon = n->as_MachConstant();
1269           machcon->eval_constant(C);
1270         } else if (n->is_Mach()) {
1271           // On Power there are more nodes that issue constants.
1272           add_size += (n->as_Mach()->ins_num_consts() * 8);
1273         }
1274       }
1275     }
1276 
1277     // Calculate the offsets of the constants and the size of the
1278     // constant table (including the padding to the next section).
1279     constant_table().calculate_offsets_and_size();
1280     const_req = constant_table().size() + add_size;
1281   }
1282 
1283   // Initialize the space for the BufferBlob used to find and verify
1284   // instruction size in MachNode::emit_size()
1285   init_scratch_buffer_blob(const_req);
1286 }

1288 CodeBuffer* PhaseOutput::init_buffer() {
1289   int stub_req  = _buf_sizes._stub;
1290   int code_req  = _buf_sizes._code;

1292 
1293   int pad_req   = NativeCall::instruction_size;
1294 
1295   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1296   stub_req += bs->estimate_stub_size();
1297 
1298   // nmethod and CodeBuffer count stubs & constants as part of method's code.
1299   // class HandlerImpl is platform-specific and defined in the *.ad files.
1300   int exception_handler_req = HandlerImpl::size_exception_handler() + MAX_stubs_size; // add marginal slop for handler
1301   int deopt_handler_req     = HandlerImpl::size_deopt_handler()     + MAX_stubs_size; // add marginal slop for handler

1309           const_req +
1310           code_req +
1311           pad_req +
1312           stub_req +
1313           exception_handler_req +
1314           deopt_handler_req;               // deopt handler
1315

1339 //------------------------------fill_buffer------------------------------------
1340 void PhaseOutput::fill_buffer(CodeBuffer* cb, uint* blk_starts) {
1341   // blk_starts[] contains offsets calculated during short branches processing,
1342   // offsets should not be increased during following steps.
1343 
1344   // Compute the size of first NumberOfLoopInstrToAlign instructions at head
1345   // of a loop. It is used to determine the padding for loop alignment.
1346   compute_loop_first_inst_sizes();
1347 
1348   // Create oopmap set.
1349   _oop_map_set = new OopMapSet();
1350 
1351   // !!!!! This preserves old handling of oopmaps for now
1352   C->debug_info()->set_oopmaps(_oop_map_set);
1353 
1354   uint nblocks  = C->cfg()->number_of_blocks();
1355   // Count and start of implicit null check instructions
1356   uint inct_cnt = 0;
1357   uint *inct_starts = NEW_RESOURCE_ARRAY(uint, nblocks+1);
1358 
1359   // Count and start of calls
1360   uint *call_returns = NEW_RESOURCE_ARRAY(uint, nblocks+1);
1361 
1362   uint  return_offset = 0;
1363   int nop_size = (new MachNopNode())->size(C->regalloc());
1364 
1365   int previous_offset = 0;
1366   int current_offset  = 0;
1367   int last_call_offset = -1;
1368   int last_avoid_back_to_back_offset = -1;
1369 #ifdef ASSERT
1370   uint* jmp_target = NEW_RESOURCE_ARRAY(uint,nblocks);
1371   uint* jmp_offset = NEW_RESOURCE_ARRAY(uint,nblocks);
1372   uint* jmp_size   = NEW_RESOURCE_ARRAY(uint,nblocks);
1373   uint* jmp_rule   = NEW_RESOURCE_ARRAY(uint,nblocks);
1374 #endif
1375 
1376   // Create an array of unused labels, one for each basic block, if printing is enabled
1377 #if defined(SUPPORT_OPTO_ASSEMBLY)
1378   int *node_offsets      = NULL;
1379   uint node_offset_limit = C->unique();
1380 
1381   if (C->print_assembly()) {
1382     node_offsets = NEW_RESOURCE_ARRAY(int, node_offset_limit);
1383   }
1384   if (node_offsets != NULL) {
1385     // We need to initialize. Unused array elements may contain garbage and mess up PrintOptoAssembly.
1386     memset(node_offsets, 0, node_offset_limit*sizeof(int));
1387   }
1388 #endif
1389 
1390   NonSafepointEmitter non_safepoints(C);  // emit non-safepoints lazily
1391 
1392   // Emit the constant table.
1393   if (C->has_mach_constant_base_node()) {
1394     constant_table().emit(*cb);
1395   }
1396 
1397   // Create an array of labels, one for each basic block
1398   Label *blk_labels = NEW_RESOURCE_ARRAY(Label, nblocks+1);
1399   for (uint i=0; i <= nblocks; i++) {
1400     blk_labels[i].init();
1401   }
1402 
1403   // ------------------
1404   // Now fill in the code buffer
1405   Node *delay_slot = NULL;
1406 
1407   for (uint i = 0; i < nblocks; i++) {
1408     Block* block = C->cfg()->get_block(i);
1409     _block = block;
1410     Node* head = block->head();
1411 
1412     // If this block needs to start aligned (i.e, can be reached other
1413     // than by falling-thru from the previous block), then force the
1414     // start of a new bundle.
1415     if (Pipeline::requires_bundling() && starts_bundle(head)) {
1416       cb->flush_bundle(true);
1417     }
1418 
1419 #ifdef ASSERT
1420     if (!block->is_connector()) {
1421       stringStream st;
1422       block->dump_head(C->cfg(), &st);
1423       MacroAssembler(cb).block_comment(st.as_string());
1424     }
1425     jmp_target[i] = 0;
1426     jmp_offset[i] = 0;
1427     jmp_size[i]   = 0;
1428     jmp_rule[i]   = 0;
1429 #endif
1430     int blk_offset = current_offset;
1431 
1432     // Define the label at the beginning of the basic block
1433     MacroAssembler(cb).bind(blk_labels[block->_pre_order]);
1434 
1435     uint last_inst = block->number_of_nodes();
1436 
1437     // Emit block normally, except for last instruction.
1438     // Emit means "dump code bits into code buffer".
1439     for (uint j = 0; j<last_inst; j++) {
1440       _index = j;
1441 
1442       // Get the node
1443       Node* n = block->get_node(j);
1444 
1445       // See if delay slots are supported
1446       if (valid_bundle_info(n) && node_bundling(n)->used_in_unconditional_delay()) {
1447         assert(delay_slot == NULL, "no use of delay slot node");
1448         assert(n->size(C->regalloc()) == Pipeline::instr_unit_size(), "delay slot instruction wrong size");
1449 
1450         delay_slot = n;
1451         continue;
1452       }
1453 
1454       // If this starts a new instruction group, then flush the current one
1455       // (but allow split bundles)
1456       if (Pipeline::requires_bundling() && starts_bundle(n))
1457         cb->flush_bundle(false);
1458 
1459       // Special handling for SafePoint/Call Nodes
1460       bool is_mcall = false;
1461       if (n->is_Mach()) {
1462         MachNode *mach = n->as_Mach();
1463         is_mcall = n->is_MachCall();
1464         bool is_sfn = n->is_MachSafePoint();
1465 
1466         // If this requires all previous instructions be flushed, then do so
1467         if (is_sfn || is_mcall || mach->alignment_required() != 1) {
1468           cb->flush_bundle(true);
1469           current_offset = cb->insts_size();
1470         }
1471 
1472         // A padding may be needed again since a previous instruction
1473         // could be moved to delay slot.
1474 
1475         // align the instruction if necessary
1476         int padding = mach->compute_padding(current_offset);
1477         // Make sure safepoint node for polling is distinct from a call's
1478         // return by adding a nop if needed.
1479         if (is_sfn && !is_mcall && padding == 0 && current_offset == last_call_offset) {
1480           padding = nop_size;
1481         }
1482         if (padding == 0 && mach->avoid_back_to_back(MachNode::AVOID_BEFORE) &&
1483             current_offset == last_avoid_back_to_back_offset) {
1484           // Avoid back to back some instructions.
1485           padding = nop_size;
1486         }
1487 
1488         if (padding > 0) {
1489           assert((padding % nop_size) == 0, "padding is not a multiple of NOP size");
1490           int nops_cnt = padding / nop_size;
1491           MachNode *nop = new MachNopNode(nops_cnt);
1492           block->insert_node(nop, j++);
1493           last_inst++;
1494           C->cfg()->map_node_to_block(nop, block);
1495           // Ensure enough space.
1496           cb->insts()->maybe_expand_to_ensure_remaining(MAX_inst_size);
1497           if ((cb->blob() == NULL) || (!CompileBroker::should_compile_new_jobs())) {
1498             C->record_failure("CodeCache is full");
1499             return;
1500           }
1501           nop->emit(*cb, C->regalloc());
1502           cb->flush_bundle(true);
1503           current_offset = cb->insts_size();
1504         }
1505 
1506         // Remember the start of the last call in a basic block
1507         if (is_mcall) {
1508           MachCallNode *mcall = mach->as_MachCall();
1509 
1510           if (mcall->entry_point() != NULL) {
1511             // This destination address is NOT PC-relative
1512             mcall->method_set((intptr_t)mcall->entry_point());
1513           }
1514 
1515           // Save the return address
1516           call_returns[block->_pre_order] = current_offset + mcall->ret_addr_offset();
1517 
1518           if (mcall->is_MachCallLeaf()) {
1519             is_mcall = false;
1520             is_sfn = false;
1521           }
1522         }
1523 
1524         // sfn will be valid whenever mcall is valid now because of inheritance
1525         if (is_sfn || is_mcall) {
1526 
1527           // Handle special safepoint nodes for synchronization
1528           if (!is_mcall) {
1529             MachSafePointNode *sfn = mach->as_MachSafePoint();
1530             // !!!!! Stubs only need an oopmap right now, so bail out
1531             if (sfn->jvms()->method() == NULL) {
1532               // Write the oopmap directly to the code blob??!!
1533               continue;
1534             }
1535           } // End synchronization
1536 
1537           non_safepoints.observe_safepoint(mach->as_MachSafePoint()->jvms(),
1538                                            current_offset);
1539           Process_OopMap_Node(mach, current_offset);
1540         } // End if safepoint
1541 
1542           // If this is a null check, then add the start of the previous instruction to the list
1543         else if( mach->is_MachNullCheck() ) {
1544           inct_starts[inct_cnt++] = previous_offset;
1545         }
1546 
1547           // If this is a branch, then fill in the label with the target BB's label
1548         else if (mach->is_MachBranch()) {
1549           // This requires the TRUE branch target be in succs[0]
1550           uint block_num = block->non_connector_successor(0)->_pre_order;
1551 
1552           // Try to replace long branch if delay slot is not used,
1553           // it is mostly for back branches since forward branch's
1554           // distance is not updated yet.
1555           bool delay_slot_is_used = valid_bundle_info(n) &&
1556                                     C->output()->node_bundling(n)->use_unconditional_delay();
1557           if (!delay_slot_is_used && mach->may_be_short_branch()) {
1558             assert(delay_slot == NULL, "not expecting delay slot node");
1559             int br_size = n->size(C->regalloc());
1560             int offset = blk_starts[block_num] - current_offset;
1561             if (block_num >= i) {
1562               // Current and following block's offset are not
1563               // finalized yet, adjust distance by the difference
1564               // between calculated and final offsets of current block.
1565               offset -= (blk_starts[i] - blk_offset);
1566             }
1567             // In the following code a nop could be inserted before
1568             // the branch which will increase the backward distance.
1569             bool needs_padding = (current_offset == last_avoid_back_to_back_offset);
1570             if (needs_padding && offset <= 0)
1571               offset -= nop_size;
1572 
1573             if (C->matcher()->is_short_branch_offset(mach->rule(), br_size, offset)) {
1574               // We've got a winner.  Replace this branch.
1575               MachNode* replacement = mach->as_MachBranch()->short_branch_version();
1576 
1577               // Update the jmp_size.
1578               int new_size = replacement->size(C->regalloc());
1579               assert((br_size - new_size) >= (int)nop_size, "short_branch size should be smaller");
1580               // Insert padding between avoid_back_to_back branches.
1581               if (needs_padding && replacement->avoid_back_to_back(MachNode::AVOID_BEFORE)) {
1582                 MachNode *nop = new MachNopNode();
1583                 block->insert_node(nop, j++);
1584                 C->cfg()->map_node_to_block(nop, block);
1585                 last_inst++;
1586                 nop->emit(*cb, C->regalloc());
1587                 cb->flush_bundle(true);
1588                 current_offset = cb->insts_size();
1589               }
1590 #ifdef ASSERT
1591               jmp_target[i] = block_num;
1592               jmp_offset[i] = current_offset - blk_offset;
1593               jmp_size[i]   = new_size;
1594               jmp_rule[i]   = mach->rule();
1595 #endif
1596               block->map_node(replacement, j);
1597               mach->subsume_by(replacement, C);
1598               n    = replacement;
1599               mach = replacement;
1600             }
1601           }
1602           mach->as_MachBranch()->label_set( &blk_labels[block_num], block_num );
1603         } else if (mach->ideal_Opcode() == Op_Jump) {
1604           for (uint h = 0; h < block->_num_succs; h++) {
1605             Block* succs_block = block->_succs[h];
1606             for (uint j = 1; j < succs_block->num_preds(); j++) {
1607               Node* jpn = succs_block->pred(j);
1608               if (jpn->is_JumpProj() && jpn->in(0) == mach) {
1609                 uint block_num = succs_block->non_connector()->_pre_order;
1610                 Label *blkLabel = &blk_labels[block_num];
1611                 mach->add_case_label(jpn->as_JumpProj()->proj_no(), blkLabel);
1612               }
1613             }
1614           }
1615         }
1616 #ifdef ASSERT
1617           // Check that oop-store precedes the card-mark
1618         else if (mach->ideal_Opcode() == Op_StoreCM) {
1619           uint storeCM_idx = j;
1620           int count = 0;
1621           for (uint prec = mach->req(); prec < mach->len(); prec++) {
1622             Node *oop_store = mach->in(prec);  // Precedence edge
1623             if (oop_store == NULL) continue;
1624             count++;
1625             uint i4;
1626             for (i4 = 0; i4 < last_inst; ++i4) {
1627               if (block->get_node(i4) == oop_store) {
1628                 break;
1629               }
1630             }
1631             // Note: This test can provide a false failure if other precedence
1632             // edges have been added to the storeCMNode.
1633             assert(i4 == last_inst || i4 < storeCM_idx, "CM card-mark executes before oop-store");
1634           }
1635           assert(count > 0, "storeCM expects at least one precedence edge");
1636         }
1637 #endif
1638         else if (!n->is_Proj()) {
1639           // Remember the beginning of the previous instruction, in case
1640           // it's followed by a flag-kill and a null-check.  Happens on
1641           // Intel all the time, with add-to-memory kind of opcodes.
1642           previous_offset = current_offset;
1643         }
1644 
1645         // Not an else-if!
1646         // If this is a trap based cmp then add its offset to the list.
1647         if (mach->is_TrapBasedCheckNode()) {
1648           inct_starts[inct_cnt++] = current_offset;
1649         }
1650       }
1651 
1652       // Verify that there is sufficient space remaining
1653       cb->insts()->maybe_expand_to_ensure_remaining(MAX_inst_size);
1654       if ((cb->blob() == NULL) || (!CompileBroker::should_compile_new_jobs())) {
1655         C->record_failure("CodeCache is full");
1656         return;
1657       }
1658 
1659       // Save the offset for the listing
1660 #if defined(SUPPORT_OPTO_ASSEMBLY)
1661       if ((node_offsets != NULL) && (n->_idx < node_offset_limit)) {
1662         node_offsets[n->_idx] = cb->insts_size();
1663       }
1664 #endif
1665 
1666       // "Normal" instruction case
1667       DEBUG_ONLY( uint instr_offset = cb->insts_size(); )
1668       n->emit(*cb, C->regalloc());
1669       current_offset  = cb->insts_size();
1670 
1671       // Above we only verified that there is enough space in the instruction section.
1672       // However, the instruction may emit stubs that cause code buffer expansion.
1673       // Bail out here if expansion failed due to a lack of code cache space.
1674       if (C->failing()) {
1675         return;
1676       }
1677 
1678 #ifdef ASSERT
1679       if (n->size(C->regalloc()) < (current_offset-instr_offset)) {
1680         n->dump();
1681         assert(false, "wrong size of mach node");
1682       }
1683 #endif
1684       non_safepoints.observe_instruction(n, current_offset);
1685 
1686       // mcall is last "call" that can be a safepoint
1687       // record it so we can see if a poll will directly follow it
1688       // in which case we'll need a pad to make the PcDesc sites unique
1689       // see  5010568. This can be slightly inaccurate but conservative
1690       // in the case that return address is not actually at current_offset.
1691       // This is a small price to pay.
1692 
1693       if (is_mcall) {
1694         last_call_offset = current_offset;
1695       }
1696 
1697       if (n->is_Mach() && n->as_Mach()->avoid_back_to_back(MachNode::AVOID_AFTER)) {
1698         // Avoid back to back some instructions.
1699         last_avoid_back_to_back_offset = current_offset;
1700       }
1701 
1702       // See if this instruction has a delay slot
1703       if (valid_bundle_info(n) && node_bundling(n)->use_unconditional_delay()) {
1704         guarantee(delay_slot != NULL, "expecting delay slot node");
1705 
1706         // Back up 1 instruction
1707         cb->set_insts_end(cb->insts_end() - Pipeline::instr_unit_size());
1708 
1709         // Save the offset for the listing
1710 #if defined(SUPPORT_OPTO_ASSEMBLY)
1711         if ((node_offsets != NULL) && (delay_slot->_idx < node_offset_limit)) {
1712           node_offsets[delay_slot->_idx] = cb->insts_size();
1713         }
1714 #endif
1715 
1716         // Support a SafePoint in the delay slot
1717         if (delay_slot->is_MachSafePoint()) {
1718           MachNode *mach = delay_slot->as_Mach();
1719           // !!!!! Stubs only need an oopmap right now, so bail out
1720           if (!mach->is_MachCall() && mach->as_MachSafePoint()->jvms()->method() == NULL) {
1721             // Write the oopmap directly to the code blob??!!
1722             delay_slot = NULL;
1723             continue;
1724           }
1725 
1726           int adjusted_offset = current_offset - Pipeline::instr_unit_size();
1727           non_safepoints.observe_safepoint(mach->as_MachSafePoint()->jvms(),
1728                                            adjusted_offset);
1729           // Generate an OopMap entry
1730           Process_OopMap_Node(mach, adjusted_offset);
1731         }
1732 
1733         // Insert the delay slot instruction
1734         delay_slot->emit(*cb, C->regalloc());
1735 
1736         // Don't reuse it
1737         delay_slot = NULL;
1738       }
1739 
1740     } // End for all instructions in block
1741 
1742     // If the next block is the top of a loop, pad this block out to align
1743     // the loop top a little. Helps prevent pipe stalls at loop back branches.
1744     if (i < nblocks-1) {
1745       Block *nb = C->cfg()->get_block(i + 1);
1746       int padding = nb->alignment_padding(current_offset);
1747       if( padding > 0 ) {
1748         MachNode *nop = new MachNopNode(padding / nop_size);
1749         block->insert_node(nop, block->number_of_nodes());
1750         C->cfg()->map_node_to_block(nop, block);
1751         nop->emit(*cb, C->regalloc());
1752         current_offset = cb->insts_size();
1753       }
1754     }
1755     // Verify that the distance for generated before forward
1756     // short branches is still valid.
1757     guarantee((int)(blk_starts[i+1] - blk_starts[i]) >= (current_offset - blk_offset), "shouldn't increase block size");
1758 
1759     // Save new block start offset
1760     blk_starts[i] = blk_offset;
1761   } // End of for all blocks
1762   blk_starts[nblocks] = current_offset;
1763 
1764   non_safepoints.flush_at_end();
1765 
1766   // Offset too large?
1767   if (C->failing())  return;
1768 
1769   // Define a pseudo-label at the end of the code
1770   MacroAssembler(cb).bind( blk_labels[nblocks] );
1771 
1772   // Compute the size of the first block
1773   _first_block_size = blk_labels[1].loc_pos() - blk_labels[0].loc_pos();
1774 
1775 #ifdef ASSERT
1776   for (uint i = 0; i < nblocks; i++) { // For all blocks
1777     if (jmp_target[i] != 0) {
1778       int br_size = jmp_size[i];
1779       int offset = blk_starts[jmp_target[i]]-(blk_starts[i] + jmp_offset[i]);
1780       if (!C->matcher()->is_short_branch_offset(jmp_rule[i], br_size, offset)) {
1781         tty->print_cr("target (%d) - jmp_offset(%d) = offset (%d), jump_size(%d), jmp_block B%d, target_block B%d", blk_starts[jmp_target[i]], blk_starts[i] + jmp_offset[i], offset, br_size, i, jmp_target[i]);
1782         assert(false, "Displacement too large for short jmp");
1783       }
1784     }
1785   }
1786 #endif
1787 
1788   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1789   bs->emit_stubs(*cb);
1790   if (C->failing())  return;
1791 
1792 #ifndef PRODUCT
1793   // Information on the size of the method, without the extraneous code
1794   Scheduling::increment_method_size(cb->insts_size());
1795 #endif
1796 
1797   // ------------------
1798   // Fill in exception table entries.
1799   FillExceptionTables(inct_cnt, call_returns, inct_starts, blk_labels);
1800 
1801   // Only java methods have exception handlers and deopt handlers
1802   // class HandlerImpl is platform-specific and defined in the *.ad files.
1803   if (C->method()) {
1804     // Emit the exception handler code.
1805     _code_offsets.set_value(CodeOffsets::Exceptions, HandlerImpl::emit_exception_handler(*cb));
1806     if (C->failing()) {
1807       return; // CodeBuffer::expand failed
1808     }
1809     // Emit the deopt handler code.
1810     _code_offsets.set_value(CodeOffsets::Deopt, HandlerImpl::emit_deopt_handler(*cb));
1811 
1812     // Emit the MethodHandle deopt handler code (if required).
1813     if (C->has_method_handle_invokes() && !C->failing()) {
1814       // We can use the same code as for the normal deopt handler, we
1815       // just need a different entry point address.
1816       _code_offsets.set_value(CodeOffsets::DeoptMH, HandlerImpl::emit_deopt_handler(*cb));
1817     }
1818   }
1819 
1820   // One last check for failed CodeBuffer::expand:
1821   if ((cb->blob() == NULL) || (!CompileBroker::should_compile_new_jobs())) {
1822     C->record_failure("CodeCache is full");
1823     return;
1824   }
1825 
1826 #if defined(SUPPORT_ABSTRACT_ASSEMBLY) || defined(SUPPORT_ASSEMBLY) || defined(SUPPORT_OPTO_ASSEMBLY)
1827   if (C->print_assembly()) {
1828     tty->cr();
1829     tty->print_cr("============================= C2-compiled nmethod ==============================");
1830   }
1831 #endif
1832 
1833 #if defined(SUPPORT_OPTO_ASSEMBLY)
1834   // Dump the assembly code, including basic-block numbers
1835   if (C->print_assembly()) {
1836     ttyLocker ttyl;  // keep the following output all in one block
1837     if (!VMThread::should_terminate()) {  // test this under the tty lock
1838       // This output goes directly to the tty, not the compiler log.
1839       // To enable tools to match it up with the compilation activity,
1840       // be sure to tag this tty output with the compile ID.
1841       if (xtty != NULL) {
1842         xtty->head("opto_assembly compile_id='%d'%s", C->compile_id(),
1843                    C->is_osr_compilation()    ? " compile_kind='osr'" :
1844                    "");
1845       }
1846       if (C->method() != NULL) {
1847         tty->print_cr("----------------------- MetaData before Compile_id = %d ------------------------", C->compile_id());
1848         C->method()->print_metadata();
1849       } else if (C->stub_name() != NULL) {
1850         tty->print_cr("----------------------------- RuntimeStub %s -------------------------------", C->stub_name());
1851       }
1852       tty->cr();
1853       tty->print_cr("------------------------ OptoAssembly for Compile_id = %d -----------------------", C->compile_id());
1854       dump_asm(node_offsets, node_offset_limit);
1855       tty->print_cr("--------------------------------------------------------------------------------");
1856       if (xtty != NULL) {
1857         // print_metadata and dump_asm above may safepoint which makes us loose the ttylock.
1858         // Retake lock too make sure the end tag is coherent, and that xmlStream->pop_tag is done
1859         // thread safe
1860         ttyLocker ttyl2;
1861         xtty->tail("opto_assembly");
1862       }
1863     }
1864   }
1865 #endif
1866 }

1868 void PhaseOutput::FillExceptionTables(uint cnt, uint *call_returns, uint *inct_starts, Label *blk_labels) {
1869   _inc_table.set_size(cnt);
1870 
1871   uint inct_cnt = 0;
1872   for (uint i = 0; i < C->cfg()->number_of_blocks(); i++) {
1873     Block* block = C->cfg()->get_block(i);
1874     Node *n = NULL;
1875     int j;
1876 
1877     // Find the branch; ignore trailing NOPs.
1878     for (j = block->number_of_nodes() - 1; j >= 0; j--) {
1879       n = block->get_node(j);
1880       if (!n->is_Mach() || n->as_Mach()->ideal_Opcode() != Op_Con) {
1881         break;
1882       }
1883     }
1884 
1885     // If we didn't find anything, continue
1886     if (j < 0) {
1887       continue;
1888     }
1889 
1890     // Compute ExceptionHandlerTable subtable entry and add it
1891     // (skip empty blocks)
1892     if (n->is_Catch()) {
1893 
1894       // Get the offset of the return from the call
1895       uint call_return = call_returns[block->_pre_order];
1896 #ifdef ASSERT
1897       assert( call_return > 0, "no call seen for this basic block" );
1898       while (block->get_node(--j)->is_MachProj()) ;
1899       assert(block->get_node(j)->is_MachCall(), "CatchProj must follow call");
1900 #endif
1901       // last instruction is a CatchNode, find it's CatchProjNodes
1902       int nof_succs = block->_num_succs;
1903       // allocate space
1904       GrowableArray<intptr_t> handler_bcis(nof_succs);
1905       GrowableArray<intptr_t> handler_pcos(nof_succs);
1906       // iterate through all successors
1907       for (int j = 0; j < nof_succs; j++) {
1908         Block* s = block->_succs[j];
1909         bool found_p = false;
1910         for (uint k = 1; k < s->num_preds(); k++) {
1911           Node* pk = s->pred(k);
1912           if (pk->is_CatchProj() && pk->in(0) == n) {
1913             const CatchProjNode* p = pk->as_CatchProj();
1914             found_p = true;
1915             // add the corresponding handler bci & pco information
1916             if (p->_con != CatchProjNode::fall_through_index) {
1917               // p leads to an exception handler (and is not fall through)
1918               assert(s == C->cfg()->get_block(s->_pre_order), "bad numbering");
1919               // no duplicates, please
1920               if (!handler_bcis.contains(p->handler_bci())) {
1921                 uint block_num = s->non_connector()->_pre_order;
1922                 handler_bcis.append(p->handler_bci());
1923                 handler_pcos.append(blk_labels[block_num].loc_pos());
1924               }
1925             }
1926           }
1927         }
1928         assert(found_p, "no matching predecessor found");
1929         // Note:  Due to empty block removal, one block may have
1930         // several CatchProj inputs, from the same Catch.
1931       }
1932 
1933       // Set the offset of the return from the call
1934       assert(handler_bcis.find(-1) != -1, "must have default handler");
1935       _handler_table.add_subtable(call_return, &handler_bcis, NULL, &handler_pcos);
1936       continue;
1937     }
1938 
1939     // Handle implicit null exception table updates
1940     if (n->is_MachNullCheck()) {
1941       uint block_num = block->non_connector_successor(0)->_pre_order;
1942       _inc_table.append(inct_starts[inct_cnt++], blk_labels[block_num].loc_pos());
1943       continue;
1944     }
1945     // Handle implicit exception table updates: trap instructions.
1946     if (n->is_Mach() && n->as_Mach()->is_TrapBasedCheckNode()) {
1947       uint block_num = block->non_connector_successor(0)->_pre_order;
1948       _inc_table.append(inct_starts[inct_cnt++], blk_labels[block_num].loc_pos());
1949       continue;
1950     }
1951   } // End of for all blocks fill in exception table entries
1952 }

2154 // See if this node fits into the present instruction bundle
2155 bool Scheduling::NodeFitsInBundle(Node *n) {
2156   uint n_idx = n->_idx;
2157 
2158   // If this is the unconditional delay instruction, then it fits
2159   if (n == _unconditional_delay_slot) {
2160 #ifndef PRODUCT
2161     if (_cfg->C->trace_opto_output())
2162       tty->print("#     NodeFitsInBundle [%4d]: TRUE; is in unconditional delay slot\n", n->_idx);
2163 #endif
2164     return (true);
2165   }
2166 
2167   // If the node cannot be scheduled this cycle, skip it
2168   if (_current_latency[n_idx] > _bundle_cycle_number) {
2169 #ifndef PRODUCT
2170     if (_cfg->C->trace_opto_output())
2171       tty->print("#     NodeFitsInBundle [%4d]: FALSE; latency %4d > %d\n",
2172                  n->_idx, _current_latency[n_idx], _bundle_cycle_number);
2173 #endif
2174     return (false);
2175   }
2176 
2177   const Pipeline *node_pipeline = n->pipeline();
2178 
2179   uint instruction_count = node_pipeline->instructionCount();
2180   if (node_pipeline->mayHaveNoCode() && n->size(_regalloc) == 0)
2181     instruction_count = 0;
2182   else if (node_pipeline->hasBranchDelay() && !_unconditional_delay_slot)
2183     instruction_count++;
2184 
2185   if (_bundle_instr_count + instruction_count > Pipeline::_max_instrs_per_cycle) {
2186 #ifndef PRODUCT
2187     if (_cfg->C->trace_opto_output())
2188       tty->print("#     NodeFitsInBundle [%4d]: FALSE; too many instructions: %d > %d\n",
2189                  n->_idx, _bundle_instr_count + instruction_count, Pipeline::_max_instrs_per_cycle);
2190 #endif
2191     return (false);
2192   }
2193 
2194   // Don't allow non-machine nodes to be handled this way
2195   if (!n->is_Mach() && instruction_count == 0)
2196     return (false);
2197 
2198   // See if there is any overlap
2199   uint delay = _bundle_use.full_latency(0, node_pipeline->resourceUse());
2200 
2201   if (delay > 0) {
2202 #ifndef PRODUCT
2203     if (_cfg->C->trace_opto_output())
2204       tty->print("#     NodeFitsInBundle [%4d]: FALSE; functional units overlap\n", n_idx);
2205 #endif
2206     return false;
2207   }
2208 
2209 #ifndef PRODUCT
2210   if (_cfg->C->trace_opto_output())
2211     tty->print("#     NodeFitsInBundle [%4d]:  TRUE\n", n_idx);
2212 #endif
2213 
2214   return true;
2215 }

2343 void Scheduling::AddNodeToBundle(Node *n, const Block *bb) {
2344 #ifndef PRODUCT
2345   if (_cfg->C->trace_opto_output()) {
2346     tty->print("#   AddNodeToBundle: ");
2347     n->dump();
2348   }
2349 #endif
2350 
2351   // Remove this from the available list
2352   uint i;
2353   for (i = 0; i < _available.size(); i++)
2354     if (_available[i] == n)
2355       break;
2356   assert(i < _available.size(), "entry in _available list not found");
2357   _available.remove(i);
2358 
2359   // See if this fits in the current bundle
2360   const Pipeline *node_pipeline = n->pipeline();
2361   const Pipeline_Use& node_usage = node_pipeline->resourceUse();
2362 
2363   // Check for instructions to be placed in the delay slot. We
2364   // do this before we actually schedule the current instruction,
2365   // because the delay slot follows the current instruction.
2366   if (Pipeline::_branch_has_delay_slot &&
2367       node_pipeline->hasBranchDelay() &&
2368       !_unconditional_delay_slot) {
2369 
2370     uint siz = _available.size();
2371 
2372     // Conditional branches can support an instruction that
2373     // is unconditionally executed and not dependent by the
2374     // branch, OR a conditionally executed instruction if
2375     // the branch is taken.  In practice, this means that
2376     // the first instruction at the branch target is
2377     // copied to the delay slot, and the branch goes to
2378     // the instruction after that at the branch target
2379     if ( n->is_MachBranch() ) {
2380 
2381       assert( !n->is_MachNullCheck(), "should not look for delay slot for Null Check" );
2382       assert( !n->is_Catch(),         "should not look for delay slot for Catch" );
2383 
2384 #ifndef PRODUCT
2385       _branches++;
2386 #endif
2387 
2388       // At least 1 instruction is on the available list
2389       // that is not dependent on the branch
2390       for (uint i = 0; i < siz; i++) {
2391         Node *d = _available[i];
2392         const Pipeline *avail_pipeline = d->pipeline();
2393 
2394         // Don't allow safepoints in the branch shadow, that will
2395         // cause a number of difficulties
2396         if ( avail_pipeline->instructionCount() == 1 &&
2397              !avail_pipeline->hasMultipleBundles() &&
2398              !avail_pipeline->hasBranchDelay() &&
2399              Pipeline::instr_has_unit_size() &&
2400              d->size(_regalloc) == Pipeline::instr_unit_size() &&
2401              NodeFitsInBundle(d) &&
2402              !node_bundling(d)->used_in_delay()) {
2403 
2404           if (d->is_Mach() && !d->is_MachSafePoint()) {
2405             // A node that fits in the delay slot was found, so we need to
2406             // set the appropriate bits in the bundle pipeline information so
2407             // that it correctly indicates resource usage.  Later, when we
2408             // attempt to add this instruction to the bundle, we will skip
2409             // setting the resource usage.
2410             _unconditional_delay_slot = d;
2411             node_bundling(n)->set_use_unconditional_delay();
2412             node_bundling(d)->set_used_in_unconditional_delay();
2413             _bundle_use.add_usage(avail_pipeline->resourceUse());
2414             _current_latency[d->_idx] = _bundle_cycle_number;
2415             _next_node = d;
2416             ++_bundle_instr_count;
2417 #ifndef PRODUCT
2418             _unconditional_delays++;
2419 #endif
2420             break;
2421           }
2422         }
2423       }
2424     }
2425 
2426     // No delay slot, add a nop to the usage
2427     if (!_unconditional_delay_slot) {
2428       // See if adding an instruction in the delay slot will overflow
2429       // the bundle.
2430       if (!NodeFitsInBundle(_nop)) {
2431 #ifndef PRODUCT
2432         if (_cfg->C->trace_opto_output())
2433           tty->print("#  *** STEP(1 instruction for delay slot) ***\n");
2434 #endif
2435         step(1);
2436       }
2437 
2438       _bundle_use.add_usage(_nop->pipeline()->resourceUse());
2439       _next_node = _nop;
2440       ++_bundle_instr_count;
2441     }
2442 
2443     // See if the instruction in the delay slot requires a
2444     // step of the bundles
2445     if (!NodeFitsInBundle(n)) {
2446 #ifndef PRODUCT
2447       if (_cfg->C->trace_opto_output())
2448         tty->print("#  *** STEP(branch won't fit) ***\n");
2449 #endif
2450       // Update the state information
2451       _bundle_instr_count = 0;
2452       _bundle_cycle_number += 1;
2453       _bundle_use.step(1);
2454     }
2455   }
2456 
2457   // Get the number of instructions
2458   uint instruction_count = node_pipeline->instructionCount();
2459   if (node_pipeline->mayHaveNoCode() && n->size(_regalloc) == 0)
2460     instruction_count = 0;
2461 
2462   // Compute the latency information
2463   uint delay = 0;
2464 
2465   if (instruction_count > 0 || !node_pipeline->mayHaveNoCode()) {
2466     int relative_latency = _current_latency[n->_idx] - _bundle_cycle_number;
2467     if (relative_latency < 0)
2468       relative_latency = 0;
2469 
2470     delay = _bundle_use.full_latency(relative_latency, node_usage);
2471 
2472     // Does not fit in this bundle, start a new one
2473     if (delay > 0) {
2474       step(delay);
2475 
2476 #ifndef PRODUCT
2477       if (_cfg->C->trace_opto_output())
2478         tty->print("#  *** STEP(%d) ***\n", delay);
2479 #endif
2480     }
2481   }
2482 
2483   // If this was placed in the delay slot, ignore it
2484   if (n != _unconditional_delay_slot) {
2485 
2486     if (delay == 0) {
2487       if (node_pipeline->hasMultipleBundles()) {
2488 #ifndef PRODUCT
2489         if (_cfg->C->trace_opto_output())
2490           tty->print("#  *** STEP(multiple instructions) ***\n");
2491 #endif
2492         step(1);
2493       }
2494 
2495       else if (instruction_count + _bundle_instr_count > Pipeline::_max_instrs_per_cycle) {
2496 #ifndef PRODUCT
2497         if (_cfg->C->trace_opto_output())
2498           tty->print("#  *** STEP(%d >= %d instructions) ***\n",
2499                      instruction_count + _bundle_instr_count,
2500                      Pipeline::_max_instrs_per_cycle);
2501 #endif
2502         step(1);
2503       }
2504     }
2505 
2506     if (node_pipeline->hasBranchDelay() && !_unconditional_delay_slot)
2507       _bundle_instr_count++;
2508 
2509     // Set the node's latency
2510     _current_latency[n->_idx] = _bundle_cycle_number;
2511 
2512     // Now merge the functional unit information
2513     if (instruction_count > 0 || !node_pipeline->mayHaveNoCode())
2514       _bundle_use.add_usage(node_usage);
2515 
2516     // Increment the number of instructions in this bundle
2517     _bundle_instr_count += instruction_count;
2518 
2519     // Remember this node for later
2520     if (n->is_Mach())
2521       _next_node = n;
2522   }
2523 
2524   // It's possible to have a BoxLock in the graph and in the _bbs mapping but
2525   // not in the bb->_nodes array.  This happens for debug-info-only BoxLocks.
2526   // 'Schedule' them (basically ignore in the schedule) but do not insert them
2527   // into the block.  All other scheduled nodes get put in the schedule here.
2528   int op = n->Opcode();
2529   if( (op == Op_Node && n->req() == 0) || // anti-dependence node OR
2530       (op != Op_Node &&         // Not an unused antidepedence node and
2531        // not an unallocated boxlock
2532        (OptoReg::is_valid(_regalloc->get_reg_first(n)) || op != Op_BoxLock)) ) {
2533 
2534     // Push any trailing projections
2535     if( bb->get_node(bb->number_of_nodes()-1) != n ) {
2536       for (DUIterator_Fast imax, i = n->fast_outs(imax); i < imax; i++) {
2537         Node *foi = n->fast_out(i);
2538         if( foi->is_Proj() )
2539           _scheduled.push(foi);
2540       }
2541     }
2542 
2543     // Put the instruction in the schedule list
2544     _scheduled.push(n);
2545   }
2546 
2547 #ifndef PRODUCT
2548   if (_cfg->C->trace_opto_output())
2549     dump_available();
2550 #endif
2551 
2552   // Walk all the definitions, decrementing use counts, and
2553   // if a definition has a 0 use count, place it in the available list.
2554   DecrementUseCounts(n,bb);
2555 }

2557 // This method sets the use count within a basic block.  We will ignore all
2558 // uses outside the current basic block.  As we are doing a backwards walk,
2559 // any node we reach that has a use count of 0 may be scheduled.  This also
2560 // avoids the problem of cyclic references from phi nodes, as long as phi
2561 // nodes are at the front of the basic block.  This method also initializes
2562 // the available list to the set of instructions that have no uses within this
2563 // basic block.
2564 void Scheduling::ComputeUseCount(const Block *bb) {
2565 #ifndef PRODUCT
2566   if (_cfg->C->trace_opto_output())
2567     tty->print("# -> ComputeUseCount\n");
2568 #endif
2569 
2570   // Clear the list of available and scheduled instructions, just in case
2571   _available.clear();
2572   _scheduled.clear();
2573 
2574   // No delay slot specified
2575   _unconditional_delay_slot = NULL;
2576 
2577 #ifdef ASSERT
2578   for( uint i=0; i < bb->number_of_nodes(); i++ )
2579     assert( _uses[bb->get_node(i)->_idx] == 0, "_use array not clean" );
2580 #endif
2581 
2582   // Force the _uses count to never go to zero for unscheduable pieces
2583   // of the block
2584   for( uint k = 0; k < _bb_start; k++ )
2585     _uses[bb->get_node(k)->_idx] = 1;
2586   for( uint l = _bb_end; l < bb->number_of_nodes(); l++ )
2587     _uses[bb->get_node(l)->_idx] = 1;
2588 
2589   // Iterate backwards over the instructions in the block.  Don't count the
2590   // branch projections at end or the block header instructions.
2591   for( uint j = _bb_end-1; j >= _bb_start; j-- ) {
2592     Node *n = bb->get_node(j);
2593     if( n->is_Proj() ) continue; // Projections handled another way
2594 
2595     // Account for all uses
2596     for ( uint k = 0; k < n->len(); k++ ) {
2597       Node *inp = n->in(k);
2598       if (!inp) continue;
2599       assert(inp != n, "no cycles allowed" );
2600       if (_cfg->get_block_for_node(inp) == bb) { // Block-local use?
2601         if (inp->is_Proj()) { // Skip through Proj's
2602           inp = inp->in(0);
2603         }
2604         ++_uses[inp->_idx];     // Count 1 block-local use
2605       }
2606     }
2607 
2608     // If this instruction has a 0 use count, then it is available
2609     if (!_uses[n->_idx]) {
2610       _current_latency[n->_idx] = _bundle_cycle_number;
2611       AddNodeToAvailableList(n);
2612     }
2613 
2614 #ifndef PRODUCT
2615     if (_cfg->C->trace_opto_output()) {
2616       tty->print("#   uses: %3d: ", _uses[n->_idx]);
2617       n->dump();
2618     }
2619 #endif
2620   }
2621 
2622 #ifndef PRODUCT
2623   if (_cfg->C->trace_opto_output())
2624     tty->print("# <- ComputeUseCount\n");
2625 #endif
2626 }

2628 // This routine performs scheduling on each basic block in reverse order,
2629 // using instruction latencies and taking into account function unit
2630 // availability.
2631 void Scheduling::DoScheduling() {
2632 #ifndef PRODUCT
2633   if (_cfg->C->trace_opto_output())
2634     tty->print("# -> DoScheduling\n");
2635 #endif
2636 
2637   Block *succ_bb = NULL;
2638   Block *bb;
2639   Compile* C = Compile::current();
2640 
2641   // Walk over all the basic blocks in reverse order
2642   for (int i = _cfg->number_of_blocks() - 1; i >= 0; succ_bb = bb, i--) {
2643     bb = _cfg->get_block(i);
2644 
2645 #ifndef PRODUCT
2646     if (_cfg->C->trace_opto_output()) {
2647       tty->print("#  Schedule BB#%03d (initial)\n", i);
2648       for (uint j = 0; j < bb->number_of_nodes(); j++) {
2649         bb->get_node(j)->dump();
2650       }
2651     }
2652 #endif
2653 
2654     // On the head node, skip processing
2655     if (bb == _cfg->get_root_block()) {
2656       continue;
2657     }
2658 
2659     // Skip empty, connector blocks
2660     if (bb->is_connector())
2661       continue;
2662 
2663     // If the following block is not the sole successor of
2664     // this one, then reset the pipeline information
2665     if (bb->_num_succs != 1 || bb->non_connector_successor(0) != succ_bb) {
2666 #ifndef PRODUCT
2667       if (_cfg->C->trace_opto_output()) {
2668         tty->print("*** bundle start of next BB, node %d, for %d instructions\n",
2669                    _next_node->_idx, _bundle_instr_count);
2670       }
2671 #endif
2672       step_and_clear();
2673     }
2674 
2675     // Leave untouched the starting instruction, any Phis, a CreateEx node
2676     // or Top.  bb->get_node(_bb_start) is the first schedulable instruction.
2677     _bb_end = bb->number_of_nodes()-1;
2678     for( _bb_start=1; _bb_start <= _bb_end; _bb_start++ ) {
2679       Node *n = bb->get_node(_bb_start);
2680       // Things not matched, like Phinodes and ProjNodes don't get scheduled.
2681       // Also, MachIdealNodes do not get scheduled
2682       if( !n->is_Mach() ) continue;     // Skip non-machine nodes
2683       MachNode *mach = n->as_Mach();
2684       int iop = mach->ideal_Opcode();
2685       if( iop == Op_CreateEx ) continue; // CreateEx is pinned
2686       if( iop == Op_Con ) continue;      // Do not schedule Top
2687       if( iop == Op_Node &&     // Do not schedule PhiNodes, ProjNodes
2688           mach->pipeline() == MachNode::pipeline_class() &&
2689           !n->is_SpillCopy() && !n->is_MachMerge() )  // Breakpoints, Prolog, etc
2690         continue;
2691       break;                    // Funny loop structure to be sure...
2692     }
2693     // Compute last "interesting" instruction in block - last instruction we
2694     // might schedule.  _bb_end points just after last schedulable inst.  We
2695     // normally schedule conditional branches (despite them being forced last
2696     // in the block), because they have delay slots we can fill.  Calls all
2697     // have their delay slots filled in the template expansions, so we don't
2698     // bother scheduling them.
2699     Node *last = bb->get_node(_bb_end);
2700     // Ignore trailing NOPs.
2701     while (_bb_end > 0 && last->is_Mach() &&
2702            last->as_Mach()->ideal_Opcode() == Op_Con) {
2703       last = bb->get_node(--_bb_end);
2704     }
2705     assert(!last->is_Mach() || last->as_Mach()->ideal_Opcode() != Op_Con, "");
2706     if( last->is_Catch() ||
2707         (last->is_Mach() && last->as_Mach()->ideal_Opcode() == Op_Halt) ) {
2708       // There might be a prior call.  Skip it.
2709       while (_bb_start < _bb_end && bb->get_node(--_bb_end)->is_MachProj());
2710     } else if( last->is_MachNullCheck() ) {
2711       // Backup so the last null-checked memory instruction is
2712       // outside the schedulable range. Skip over the nullcheck,
2713       // projection, and the memory nodes.
2714       Node *mem = last->in(1);
2715       do {
2716         _bb_end--;
2717       } while (mem != bb->get_node(_bb_end));
2718     } else {
2719       // Set _bb_end to point after last schedulable inst.
2720       _bb_end++;
2721     }
2722 
2723     assert( _bb_start <= _bb_end, "inverted block ends" );
2724 
2725     // Compute the register antidependencies for the basic block
2726     ComputeRegisterAntidependencies(bb);
2727     if (C->failing())  return;  // too many D-U pinch points
2728 
2729     // Compute intra-bb latencies for the nodes
2730     ComputeLocalLatenciesForward(bb);
2731 
2732     // Compute the usage within the block, and set the list of all nodes
2733     // in the block that have no uses within the block.
2734     ComputeUseCount(bb);
2735 
2736     // Schedule the remaining instructions in the block
2737     while ( _available.size() > 0 ) {
2738       Node *n = ChooseNodeToBundle();
2739       guarantee(n != NULL, "no nodes available");
2740       AddNodeToBundle(n,bb);
2741     }
2742 
2743     assert( _scheduled.size() == _bb_end - _bb_start, "wrong number of instructions" );
2744 #ifdef ASSERT
2745     for( uint l = _bb_start; l < _bb_end; l++ ) {
2746       Node *n = bb->get_node(l);
2747       uint m;
2748       for( m = 0; m < _bb_end-_bb_start; m++ )
2749         if( _scheduled[m] == n )
2750           break;
2751       assert( m < _bb_end-_bb_start, "instruction missing in schedule" );
2752     }
2753 #endif
2754 
2755     // Now copy the instructions (in reverse order) back to the block
2756     for ( uint k = _bb_start; k < _bb_end; k++ )
2757       bb->map_node(_scheduled[_bb_end-k-1], k);
2758 
2759 #ifndef PRODUCT
2760     if (_cfg->C->trace_opto_output()) {
2761       tty->print("#  Schedule BB#%03d (final)\n", i);
2762       uint current = 0;
2763       for (uint j = 0; j < bb->number_of_nodes(); j++) {
2764         Node *n = bb->get_node(j);
2765         if( valid_bundle_info(n) ) {
2766           Bundle *bundle = node_bundling(n);
2767           if (bundle->instr_count() > 0 || bundle->flags() > 0) {
2768             tty->print("*** Bundle: ");
2769             bundle->dump();
2770           }
2771           n->dump();
2772         }
2773       }
2774     }
2775 #endif
2776 #ifdef ASSERT
2777     verify_good_schedule(bb,"after block local scheduling");
2778 #endif
2779   }
2780 
2781 #ifndef PRODUCT
2782   if (_cfg->C->trace_opto_output())
2783     tty->print("# <- DoScheduling\n");
2784 #endif
2785 
2786   // Record final node-bundling array location
2787   _regalloc->C->output()->set_node_bundling_base(_node_bundling_base);
2788 
2789 } // end DoScheduling

2818 void Scheduling::verify_good_schedule( Block *b, const char *msg ) {
2819 
2820   // Zap to something reasonable for the verify code
2821   _reg_node.clear();
2822 
2823   // Walk over the block backwards.  Check to make sure each DEF doesn't
2824   // kill a live value (other than the one it's supposed to).  Add each
2825   // USE to the live set.
2826   for( uint i = b->number_of_nodes()-1; i >= _bb_start; i-- ) {
2827     Node *n = b->get_node(i);
2828     int n_op = n->Opcode();
2829     if( n_op == Op_MachProj && n->ideal_reg() == MachProjNode::fat_proj ) {
2830       // Fat-proj kills a slew of registers
2831       RegMask rm = n->out_RegMask();// Make local copy
2832       while( rm.is_NotEmpty() ) {
2833         OptoReg::Name kill = rm.find_first_elem();
2834         rm.Remove(kill);

2877 void Scheduling::anti_do_def( Block *b, Node *def, OptoReg::Name def_reg, int is_def ) {
2878   if( !OptoReg::is_valid(def_reg) ) // Ignore stores & control flow
2879     return;
2880 
2881   Node *pinch = _reg_node[def_reg]; // Get pinch point
2882   if ((pinch == NULL) || _cfg->get_block_for_node(pinch) != b || // No pinch-point yet?
2883       is_def ) {    // Check for a true def (not a kill)
2884     _reg_node.map(def_reg,def); // Record def/kill as the optimistic pinch-point
2885     return;
2886   }
2887 
2888   Node *kill = def;             // Rename 'def' to more descriptive 'kill'
2889   debug_only( def = (Node*)((intptr_t)0xdeadbeef); )
2890 
2891   // After some number of kills there _may_ be a later def
2892   Node *later_def = NULL;
2893 
2894   Compile* C = Compile::current();
2895 
2896   // Finding a kill requires a real pinch-point.
2897   // Check for not already having a pinch-point.
2898   // Pinch points are Op_Node's.
2899   if( pinch->Opcode() != Op_Node ) { // Or later-def/kill as pinch-point?
2900     later_def = pinch;            // Must be def/kill as optimistic pinch-point
2901     if ( _pinch_free_list.size() > 0) {
2902       pinch = _pinch_free_list.pop();
2903     } else {
2904       pinch = new Node(1); // Pinch point to-be
2905     }
2906     if (pinch->_idx >= _regalloc->node_regs_max_index()) {
2907       _cfg->C->record_method_not_compilable("too many D-U pinch points");
2908       return;
2909     }
2910     _cfg->map_node_to_block(pinch, b);      // Pretend it's valid in this block (lazy init)
2911     _reg_node.map(def_reg,pinch); // Record pinch-point
2912     //regalloc()->set_bad(pinch->_idx); // Already initialized this way.
2913     if( later_def->outcnt() == 0 || later_def->ideal_reg() == MachProjNode::fat_proj ) { // Distinguish def from kill
2914       pinch->init_req(0, C->top());     // set not NULL for the next call
2915       add_prec_edge_from_to(later_def,pinch); // Add edge from kill to pinch
2916       later_def = NULL;           // and no later def
2917     }
2918     pinch->set_req(0,later_def);  // Hook later def so we can find it
2919   } else {                        // Else have valid pinch point
2920     if( pinch->in(0) )            // If there is a later-def
2921       later_def = pinch->in(0);   // Get it
2922   }
2923 
2924   // Add output-dependence edge from later def to kill
2925   if( later_def )               // If there is some original def
2926     add_prec_edge_from_to(later_def,kill); // Add edge from def to kill
2927 
2928   // See if current kill is also a use, and so is forced to be the pinch-point.
2929   if( pinch->Opcode() == Op_Node ) {
2930     Node *uses = kill->is_Proj() ? kill->in(0) : kill;
2931     for( uint i=1; i<uses->req(); i++ ) {
2932       if( _regalloc->get_reg_first(uses->in(i)) == def_reg ||
2933           _regalloc->get_reg_second(uses->in(i)) == def_reg ) {
2934         // Yes, found a use/kill pinch-point
2935         pinch->set_req(0,NULL);  //
2936         pinch->replace_by(kill); // Move anti-dep edges up
2937         pinch = kill;
2938         _reg_node.map(def_reg,pinch);
2939         return;
2940       }
2941     }
2942   }
2943 
2944   // Add edge from kill to pinch-point
2945   add_prec_edge_from_to(kill,pinch);
2946 }

2968 // We insert antidependences between the reads and following write of
2969 // allocated registers to prevent illegal code motion. Hopefully, the
2970 // number of added references should be fairly small, especially as we
2971 // are only adding references within the current basic block.
2972 void Scheduling::ComputeRegisterAntidependencies(Block *b) {
2973 
2974 #ifdef ASSERT
2975   verify_good_schedule(b,"before block local scheduling");
2976 #endif
2977 
2978   // A valid schedule, for each register independently, is an endless cycle
2979   // of: a def, then some uses (connected to the def by true dependencies),
2980   // then some kills (defs with no uses), finally the cycle repeats with a new
2981   // def.  The uses are allowed to float relative to each other, as are the
2982   // kills.  No use is allowed to slide past a kill (or def).  This requires
2983   // antidependencies between all uses of a single def and all kills that
2984   // follow, up to the next def.  More edges are redundant, because later defs
2985   // & kills are already serialized with true or antidependencies.  To keep
2986   // the edge count down, we add a 'pinch point' node if there's more than
2987   // one use or more than one kill/def.
2988 
2989   // We add dependencies in one bottom-up pass.
2990 
2991   // For each instruction we handle it's DEFs/KILLs, then it's USEs.
2992 
2993   // For each DEF/KILL, we check to see if there's a prior DEF/KILL for this
2994   // register.  If not, we record the DEF/KILL in _reg_node, the
2995   // register-to-def mapping.  If there is a prior DEF/KILL, we insert a
2996   // "pinch point", a new Node that's in the graph but not in the block.
2997   // We put edges from the prior and current DEF/KILLs to the pinch point.
2998   // We put the pinch point in _reg_node.  If there's already a pinch point
2999   // we merely add an edge from the current DEF/KILL to the pinch point.
3000 
3001   // After doing the DEF/KILLs, we handle USEs.  For each used register, we
3002   // put an edge from the pinch point to the USE.
3003 
3004   // To be expedient, the _reg_node array is pre-allocated for the whole
3005   // compilation.  _reg_node is lazily initialized; it either contains a NULL,
3006   // or a valid def/kill/pinch-point, or a leftover node from some prior
3007   // block.  Leftover node from some prior block is treated like a NULL (no
3008   // prior def, so no anti-dependence needed).  Valid def is distinguished by
3009   // it being in the current block.
3010   bool fat_proj_seen = false;
3011   uint last_safept = _bb_end-1;
3012   Node* end_node         = (_bb_end-1 >= _bb_start) ? b->get_node(last_safept) : NULL;
3013   Node* last_safept_node = end_node;
3014   for( uint i = _bb_end-1; i >= _bb_start; i-- ) {
3015     Node *n = b->get_node(i);
3016     int is_def = n->outcnt();   // def if some uses prior to adding precedence edges
3017     if( n->is_MachProj() && n->ideal_reg() == MachProjNode::fat_proj ) {
3018       // Fat-proj kills a slew of registers
3019       // This can add edges to 'n' and obscure whether or not it was a def,
3020       // hence the is_def flag.
3021       fat_proj_seen = true;
3022       RegMask rm = n->out_RegMask();// Make local copy
3023       while( rm.is_NotEmpty() ) {
3024         OptoReg::Name kill = rm.find_first_elem();
3025         rm.Remove(kill);
3026         anti_do_def( b, n, kill, is_def );
3027       }
3028     } else {
3029       // Get DEF'd registers the normal way
3030       anti_do_def( b, n, _regalloc->get_reg_first(n), is_def );
3031       anti_do_def( b, n, _regalloc->get_reg_second(n), is_def );
3032     }
3033 
3034     // Kill projections on a branch should appear to occur on the
3035     // branch, not afterwards, so grab the masks from the projections
3036     // and process them.
3037     if (n->is_MachBranch() || (n->is_Mach() && n->as_Mach()->ideal_Opcode() == Op_Jump)) {
3038       for (DUIterator_Fast imax, i = n->fast_outs(imax); i < imax; i++) {
3039         Node* use = n->fast_out(i);
3040         if (use->is_Proj()) {
3041           RegMask rm = use->out_RegMask();// Make local copy
3042           while( rm.is_NotEmpty() ) {
3043             OptoReg::Name kill = rm.find_first_elem();
3044             rm.Remove(kill);
3045             anti_do_def( b, n, kill, false );
3046           }
3047         }
3048       }
3049     }
3050 
3051     // Check each register used by this instruction for a following DEF/KILL
3052     // that must occur afterward and requires an anti-dependence edge.
3053     for( uint j=0; j<n->req(); j++ ) {
3054       Node *def = n->in(j);
3055       if( def ) {
3056         assert( !def->is_MachProj() || def->ideal_reg() != MachProjNode::fat_proj, "" );
3057         anti_do_use( b, n, _regalloc->get_reg_first(def) );
3058         anti_do_use( b, n, _regalloc->get_reg_second(def) );
3059       }
3060     }
3061     // Do not allow defs of new derived values to float above GC
3062     // points unless the base is definitely available at the GC point.
3063 
3064     Node *m = b->get_node(i);
3065 
3066     // Add precedence edge from following safepoint to use of derived pointer
3067     if( last_safept_node != end_node &&
3068         m != last_safept_node) {
3069       for (uint k = 1; k < m->req(); k++) {
3070         const Type *t = m->in(k)->bottom_type();
3071         if( t->isa_oop_ptr() &&
3072             t->is_ptr()->offset() != 0 ) {
3073           last_safept_node->add_prec( m );
3074           break;
3075         }
3076       }
3077     }
3078 
3079     if( n->jvms() ) {           // Precedence edge from derived to safept
3080       // Check if last_safept_node was moved by pinch-point insertion in anti_do_use()
3081       if( b->get_node(last_safept) != last_safept_node ) {
3082         last_safept = b->find_node(last_safept_node);
3083       }
3084       for( uint j=last_safept; j > i; j-- ) {
3085         Node *mach = b->get_node(j);
3086         if( mach->is_Mach() && mach->as_Mach()->ideal_Opcode() == Op_AddP )
3087           mach->add_prec( n );
3088       }
3089       last_safept = i;
3090       last_safept_node = m;
3091     }
3092   }
3093 
3094   if (fat_proj_seen) {
3095     // Garbage collect pinch nodes that were not consumed.
3096     // They are usually created by a fat kill MachProj for a call.
3097     garbage_collect_pinch_nodes();
3098   }
3099 }

3101 // Garbage collect pinch nodes for reuse by other blocks.
3102 //
3103 // The block scheduler's insertion of anti-dependence
3104 // edges creates many pinch nodes when the block contains
3105 // 2 or more Calls.  A pinch node is used to prevent a
3106 // combinatorial explosion of edges.  If a set of kills for a
3107 // register is anti-dependent on a set of uses (or defs), rather
3108 // than adding an edge in the graph between each pair of kill
3109 // and use (or def), a pinch is inserted between them:
3110 //
3111 //            use1   use2  use3
3112 //                \   |   /
3113 //                 \  |  /
3114 //                  pinch
3115 //                 /  |  \
3116 //                /   |   \
3117 //            kill1 kill2 kill3
3118 //
3119 // One pinch node is created per register killed when
3120 // the second call is encountered during a backwards pass
3121 // over the block.  Most of these pinch nodes are never
3122 // wired into the graph because the register is never
3123 // used or def'ed in the block.
3124 //
3125 void Scheduling::garbage_collect_pinch_nodes() {
3126 #ifndef PRODUCT
3127   if (_cfg->C->trace_opto_output()) tty->print("Reclaimed pinch nodes:");
3128 #endif
3129   int trace_cnt = 0;
3130   for (uint k = 0; k < _reg_node.Size(); k++) {
3131     Node* pinch = _reg_node[k];
3132     if ((pinch != NULL) && pinch->Opcode() == Op_Node &&
3133         // no predecence input edges
3134         (pinch->req() == pinch->len() || pinch->in(pinch->req()) == NULL) ) {
3135       cleanup_pinch(pinch);
3136       _pinch_free_list.push(pinch);
3137       _reg_node.map(k, NULL);
3138 #ifndef PRODUCT
3139       if (_cfg->C->trace_opto_output()) {
3140         trace_cnt++;
3141         if (trace_cnt > 40) {
3142           tty->print("\n");
3143           trace_cnt = 0;
3144         }
3145         tty->print(" %d", pinch->_idx);
3146       }
3147 #endif
3148     }
3149   }
3150 #ifndef PRODUCT
3151   if (_cfg->C->trace_opto_output()) tty->print("\n");
3152 #endif
3153 }

3158 
3159   for (DUIterator_Last imin, i = pinch->last_outs(imin); i >= imin; ) {
