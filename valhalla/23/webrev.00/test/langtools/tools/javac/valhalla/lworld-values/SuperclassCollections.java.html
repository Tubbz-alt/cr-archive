26 public class SuperclassCollections {

28     public static class BadSuper {}

30     public interface GoodSuperInterface {}
31     public static abstract class GoodSuper extends Object {}
32     public static abstract class Integer extends Number {
33         public double doubleValue() { return 0; }
34         public float floatValue() { return 0; }
35         public long longValue() { return 0; }
36         public int intValue() { return 0; }
37     }
38     public static abstract class SuperWithInstanceField {
39         int x;
40     }
41     public static abstract class SuperWithInstanceField_01 extends SuperWithInstanceField {}
42 
43     public static abstract class SuperWithStaticField {
44         static int x;
45     }
46     public static abstract class SuperWithEmptyNoArgCtor {
47         SuperWithEmptyNoArgCtor() {
48             // Programmer supplied ctor but injected super call
49         }
50     }
51     public static abstract class SuperWithEmptyNoArgCtor_01 extends SuperWithEmptyNoArgCtor {
52         SuperWithEmptyNoArgCtor_01() {
53             super();  // programmer coded chaining no-arg constructor
54         }
55     }
56     public static abstract class SuperWithEmptyNoArgCtor_02 extends SuperWithEmptyNoArgCtor_01 {
57         // Synthesized chaining no-arg constructor
58     }
59 
60     public static abstract class SuperWithNonEmptyNoArgCtor {
61         SuperWithNonEmptyNoArgCtor() {
62             System.out.println("Non-Empty");
63         }
64     }
65     public static abstract class SuperWithNonEmptyNoArgCtor_01 extends SuperWithNonEmptyNoArgCtor {}
66 
67     public static abstract class SuperWithArgedCtor {
68         SuperWithArgedCtor() {}
69         SuperWithArgedCtor(String s) {
70         }
71     }
72     public static abstract class SuperWithArgedCtor_01 extends SuperWithArgedCtor {}
73 
74     public static abstract class SuperWithInstanceInit {
75         {
76             System.out.println("Disqualified from being super");
77         }
78     }
79     public static abstract class SuperWithInstanceInit_01 extends SuperWithInstanceInit {
80         {
81             // Not disqualified since it is a meaningless empty block.
82         }
83     }
84 
85     public static abstract class SuperWithSynchronizedMethod {
86         synchronized void foo() {}
87     }
88     public static abstract class SuperWithSynchronizedMethod_1 extends SuperWithSynchronizedMethod {
89     }

91     public abstract class InnerSuper {}
