 241     /** Check kind and type of given tree against protokind and prototype.
 242      *  If check succeeds, store type in tree and return it.
 243      *  If check fails, store errType in tree and return it.
 244      *  No checks are performed if the prototype is a method type.
 245      *  It is not necessary in this case since we know that kind and type
 246      *  are correct.
 247      *
 248      *  @param tree     The tree whose kind and type is checked
 249      *  @param found    The computed type of the tree
 250      *  @param ownkind  The computed kind of the tree
 251      *  @param resultInfo  The expected result of the tree
 252      */
 253     Type check(final JCTree tree,
 254                final Type found,
 255                final KindSelector ownkind,
 256                final ResultInfo resultInfo) {
 257         InferenceContext inferenceContext = resultInfo.checkContext.inferenceContext();
 258         Type owntype;
 259         boolean shouldCheck = !found.hasTag(ERROR) &&
 260                 !resultInfo.pt.hasTag(METHOD) &&
 261                 !resultInfo.pt.hasTag(FORALL);
 262         if (shouldCheck && !ownkind.subset(resultInfo.pkind)) {
 263             log.error(tree.pos(),
 264                       Errors.UnexpectedType(resultInfo.pkind.kindNames(),
 265                                             ownkind.kindNames()));
 266             owntype = types.createErrorType(found);
 267         } else if (allowPoly && inferenceContext.free(found)) {
 268             //delay the check if there are inference variables in the found type
 269             //this means we are dealing with a partially inferred poly expression
 270             owntype = shouldCheck ? resultInfo.pt : found;
 271             if (resultInfo.checkMode.installPostInferenceHook()) {
 272                 inferenceContext.addFreeTypeListener(List.of(found),
 273                         instantiatedContext -> {
 274                             ResultInfo pendingResult =
 275                                     resultInfo.dup(inferenceContext.asInstType(resultInfo.pt));
 276                             check(tree, inferenceContext.asInstType(found), ownkind, pendingResult);
 277                         });
 278             }
 279         } else {
 280             owntype = shouldCheck ?
 281             resultInfo.check(tree, found) :
 282             found;
 283         }
 284         if (resultInfo.checkMode.updateTreeType()) {
 285             tree.type = owntype;
 286         }
 287         return owntype;
 288     }

 290     /** Is given blank final variable assignable, i.e. in a scope where it
 291      *  may be assigned to even though it is final?
 292      *  @param v      The blank final variable.
 293      *  @param env    The current environment.
 294      */
 295     boolean isAssignableAsBlankFinal(VarSymbol v, Env<AttrContext> env) {
 296         Symbol owner = env.info.scope.owner;
 297            // owner refers to the innermost variable, method or
 298            // initializer block declaration at this point.
 299         return
 300             v.owner == owner
 301             ||
 302             ((owner.name == names.init ||    // i.e. we are in a constructor
 303               owner.kind == VAR ||           // i.e. we are in a variable initializer
 304               (owner.flags() & BLOCK) != 0)  // i.e. we are in an initializer block
 305              &&
 306              v.owner == owner.owner
 307              &&
 308              ((v.flags() & STATIC) != 0) == Resolve.isStatic(env));
 309     }

 311     /** Check that variable can be assigned to.
 312      *  @param pos    The current source code position.
 313      *  @param v      The assigned variable
 314      *  @param base   If the variable is referred to in a Select, the part
 315      *                to the left of the `.', null otherwise.
 316      *  @param env    The current environment.
 317      */
 318     void checkAssignable(DiagnosticPosition pos, VarSymbol v, JCTree base, Env<AttrContext> env) {
 319         if (v.name == names._this) {
 320             log.error(pos, Errors.CantAssignValToThis);
 321         } else if ((v.flags() & FINAL) != 0 &&
 322             ((v.flags() & HASINIT) != 0
 323              ||
 324              !((base == null ||
 325                TreeInfo.isThisQualifier(base)) &&
 326                isAssignableAsBlankFinal(v, env)))) {
 327             if (v.isResourceVariable()) { //TWR resource
 328                 log.error(pos, Errors.TryResourceMayNotBeAssigned(v));
 329             } else if ((v.flags() & MATCH_BINDING) != 0) {
 330                 log.error(pos, Errors.PatternBindingMayNotBeAssigned(v));
 331             } else {
 332                 boolean complain = true;
 333                 /* Allow updates to instance fields of value classes by any method in the same nest via the
 334                    withfield operator -This does not result in mutation of final fields; the code generator
 335                    would implement `copy on write' semantics via the opcode `withfield'.
 336                 */
 337                 if (env.info.inWithField && v.getKind() == ElementKind.FIELD && (v.flags() & STATIC) == 0 && types.isValue(v.owner.type)) {
 338                     if (env.enclClass.sym.outermostClass() == v.owner.outermostClass())
 339                         complain = false;
 340                 }
 341                 if (complain)
 342                     log.error(pos, Errors.CantAssignValToFinalVar(v));
 343             }
 344         }
 345     }

 347     /** Does tree represent a static reference to an identifier?
 348      *  It is assumed that tree is either a SELECT or an IDENT.
 349      *  We have to weed out selects from non-type names here.
 350      *  @param tree    The candidate tree.
 351      */
 352     boolean isStaticReference(JCTree tree) {
 353         if (tree.hasTag(SELECT)) {
 354             Symbol lsym = TreeInfo.symbol(((JCFieldAccess) tree).selected);
 355             if (lsym == null || lsym.kind != TYP) {
 356                 return false;
 357             }
 358         }
 359         return true;

 395     // where
 396         private TreeVisitor<Symbol,Env<AttrContext>> identAttributer = new IdentAttributer();
 397         private class IdentAttributer extends SimpleTreeVisitor<Symbol,Env<AttrContext>> {
 398             @Override @DefinedBy(Api.COMPILER_TREE)
 399             public Symbol visitMemberSelect(MemberSelectTree node, Env<AttrContext> env) {
 400                 Symbol site = visit(node.getExpression(), env);
 401                 if (site.kind == ERR || site.kind == ABSENT_TYP || site.kind == HIDDEN)
 402                     return site;
 403                 Name name = (Name)node.getIdentifier();
 404                 if (site.kind == PCK) {
 405                     env.toplevel.packge = (PackageSymbol)site;
 406                     return rs.findIdentInPackage(null, env, (TypeSymbol)site, name,
 407                             KindSelector.TYP_PCK);
 408                 } else {
 409                     env.enclClass.sym = (ClassSymbol)site;
 410                     return rs.findMemberType(env, site.asType(), name, (TypeSymbol)site);
 411                 }
 412             }
 413 
 414             @Override @DefinedBy(Api.COMPILER_TREE)
 415             public Symbol visitIdentifier(IdentifierTree node, Env<AttrContext> env) {
 416                 return rs.findIdent(null, env, (Name)node.getName(), KindSelector.TYP_PCK);
 417             }
 418         }

 447     private Env<AttrContext> attribToTree(JCTree root, Env<AttrContext> env, JCTree tree, ResultInfo resultInfo) {
 448         breakTree = tree;
 449         JavaFileObject prev = log.useSource(env.toplevel.sourcefile);
 450         try {
 451             deferredAttr.attribSpeculative(root, env, resultInfo,
 452                     null, DeferredAttr.AttributionMode.ANALYZER,
 453                     argumentAttr.withLocalCacheContext());
 454         } catch (BreakAttr b) {
 455             return b.env;
 456         } catch (AssertionError ae) {
 457             if (ae.getCause() instanceof BreakAttr) {
 458                 return ((BreakAttr)(ae.getCause())).env;
 459             } else {
 460                 throw ae;
 461             }
 462         } finally {
 463             breakTree = null;
 464             log.useSource(prev);
 465         }
 466         return env;
 467     }

 679     /** Visitor method: attribute a tree, catching any completion failure
 680      *  exceptions. Return the tree's type.
 681      *
 682      *  @param tree    The tree to be visited.
 683      *  @param env     The environment visitor argument.
 684      *  @param resultInfo   The result info visitor argument.
 685      */
 686     Type attribTree(JCTree tree, Env<AttrContext> env, ResultInfo resultInfo) {
 687         Env<AttrContext> prevEnv = this.env;
 688         ResultInfo prevResult = this.resultInfo;
 689         try {
 690             this.env = env;
 691             this.resultInfo = resultInfo;
 692             if (resultInfo.needsArgumentAttr(tree)) {
 693                 result = argumentAttr.attribArg(tree, env);
 694             } else {
 695                 tree.accept(this);
 696             }
 697             matchBindings = matchBindingsComputer.finishBindings(tree,
 698                                                                  matchBindings);
 699             if (tree == breakTree &&
 700                     resultInfo.checkContext.deferredAttrContext().mode == AttrMode.CHECK) {
 701                 breakTreeFound(copyEnv(env));
 702             }
 703             return result;
 704         } catch (CompletionFailure ex) {
 705             tree.type = syms.errType;
 706             return chk.completionError(tree.pos(), ex);
 707         } finally {
 708             this.env = prevEnv;
 709             this.resultInfo = prevResult;
 710         }
 711     }

 829         for (JCTypeParameter tvar : typarams) {
 830             TypeVar a = (TypeVar)tvar.type;
 831             a.tsym.flags_field |= UNATTRIBUTED;
 832             a.setUpperBound(Type.noType);
 833             if (!tvar.bounds.isEmpty()) {
 834                 List<Type> bounds = List.of(attribType(tvar.bounds.head, env));
 835                 for (JCExpression bound : tvar.bounds.tail)
 836                     bounds = bounds.prepend(attribType(bound, env));
 837                 types.setBounds(a, bounds.reverse());
 838             } else {
 839                 // if no bounds are given, assume a single bound of
 840                 // java.lang.Object.
 841                 types.setBounds(a, List.of(syms.objectType));
 842             }
 843             a.tsym.flags_field &= ~UNATTRIBUTED;
 844         }
 845         if (checkCyclic) {
 846             for (JCTypeParameter tvar : typarams) {
 847                 chk.checkNonCyclic(tvar.pos(), (TypeVar)tvar.type);
 848             }
 849         }

 895     /** Attribute type reference in an `extends' or `implements' clause.
 896      *  Supertypes of anonymous inner classes are usually already attributed.
 897      *
 898      *  @param tree              The tree making up the type reference.
 899      *  @param env               The environment current at the reference.
 900      *  @param classExpected     true if only a class is expected here.
 901      *  @param interfaceExpected true if only an interface is expected here.
 902      */
 903     Type attribBase(JCTree tree,
 904                     Env<AttrContext> env,
 905                     boolean classExpected,
 906                     boolean interfaceExpected,
 907                     boolean checkExtensible) {
 908         Type t = tree.type != null ?
 909             tree.type :
 910             attribType(tree, env);
 911         return checkBase(t, tree, env, classExpected, interfaceExpected, checkExtensible);
 912     }
 913     Type checkBase(Type t,
 914                    JCTree tree,
 915                    Env<AttrContext> env,
 916                    boolean classExpected,
 917                    boolean interfaceExpected,
 918                    boolean checkExtensible) {
 919         final DiagnosticPosition pos = tree.hasTag(TYPEAPPLY) ?
 920                 (((JCTypeApply) tree).clazz).pos() : tree.pos();
 921         if (t.tsym.isAnonymous()) {
 922             log.error(pos, Errors.CantInheritFromAnon);
 923             return types.createErrorType(t);
 924         }
 925         if (t.isErroneous())
 926             return t;
 927         if (t.hasTag(TYPEVAR) && !classExpected && !interfaceExpected) {
 928             // check that type variable is already visible
 929             if (t.getUpperBound() == null) {
 930                 log.error(pos, Errors.IllegalForwardRef);
 931                 return types.createErrorType(t);
 932             }
 933         } else {
 934             t = chk.checkClassType(pos, t, checkExtensible);
 935         }
 936         if (interfaceExpected && (t.tsym.flags() & INTERFACE) == 0) {
 937             log.error(pos, Errors.IntfExpectedHere);
 938             // return errType is necessary since otherwise there might
 939             // be undetected cycles which cause attribution to loop
 940             return types.createErrorType(t);
 941         } else if (checkExtensible &&
 942                    classExpected &&
 943                    (t.tsym.flags() & INTERFACE) != 0) {
 944             log.error(pos, Errors.NoIntfExpectedHere);
 945             return types.createErrorType(t);
 946         }
 947         if (checkExtensible &&
 948             ((t.tsym.flags() & FINAL) != 0)) {
 949             log.error(pos,
 950                       Errors.CantInheritFromFinal(t.tsym));
 951         }
 952         chk.checkNonCyclic(pos, t);
 953         return t;
 954     }

 963     public void visitClassDef(JCClassDecl tree) {
 964         Optional<ArgumentAttr.LocalCacheContext> localCacheContext =
 965                 Optional.ofNullable(env.info.attributionMode.isSpeculative ?
 966                         argumentAttr.withLocalCacheContext() : null);
 967         try {
 968             // Local and anonymous classes have not been entered yet, so we need to
 969             // do it now.
 970             if (env.info.scope.owner.kind.matches(KindSelector.VAL_MTH)) {
 971                 enter.classEnter(tree, env);
 972             } else {
 973                 // If this class declaration is part of a class level annotation,
 974                 // as in @MyAnno(new Object() {}) class MyClass {}, enter it in
 975                 // order to simplify later steps and allow for sensible error
 976                 // messages.
 977                 if (env.tree.hasTag(NEWCLASS) && TreeInfo.isInAnnotation(env, tree))
 978                     enter.classEnter(tree, env);
 979             }
 980 
 981             ClassSymbol c = tree.sym;
 982             if (c == null) {
 983                 // exit in case something drastic went wrong during enter.
 984                 result = null;
 985             } else {
 986                 // make sure class has been completed:
 987                 c.complete();
 988 
 989                 // If this class appears as an anonymous class
 990                 // in a superclass constructor call
 991                 // disable implicit outer instance from being passed.
 992                 // (This would be an illegal access to "this before super").
 993                 if (env.info.isSelfCall &&
 994                         env.tree.hasTag(NEWCLASS)) {
 995                     c.flags_field |= NOOUTERTHIS;
 996                 }
 997                 if (env.tree.hasTag(NEWCLASS) && types.isValue(c.getSuperclass())) {
 998                     c.flags_field |= VALUE; // avoid further secondary errors.
 999                 }
1000                 attribClass(tree.pos(), c);
1001                 result = tree.type = c.type;
1002             }
1003         } finally {
1004             localCacheContext.ifPresent(LocalCacheContext::leave);
1005         }
1006     }

1008     public void visitMethodDef(JCMethodDecl tree) {
1009         MethodSymbol m = tree.sym;
1010         boolean isDefaultMethod = (m.flags() & DEFAULT) != 0;
1011 
1012         Lint lint = env.info.lint.augment(m);
1013         Lint prevLint = chk.setLint(lint);
1014         MethodSymbol prevMethod = chk.setMethod(m);
1015         try {
1016             deferredLintHandler.flush(tree.pos());
1017             chk.checkDeprecatedAnnotation(tree.pos(), m);
1018 
1019 
1020             // Create a new environment with local scope
1021             // for attributing the method.
1022             Env<AttrContext> localEnv = memberEnter.methodEnv(tree, env);
1023             localEnv.info.lint = lint;
1024 
1025             attribStats(tree.typarams, localEnv);
1026 
1027             // If we override any other methods, check that we do so properly.
1028             // JLS ???
1029             if (m.isStatic()) {
1030                 chk.checkHideClashes(tree.pos(), env.enclClass.type, m);
1031             } else {
1032                 chk.checkOverrideClashes(tree.pos(), env.enclClass.type, m);
1033             }
1034             chk.checkOverride(env, tree, m);
1035 
1036             if (isDefaultMethod && types.overridesObjectMethod(m.enclClass(), m)) {
1037                 log.error(tree, Errors.DefaultOverridesObjectMember(m.name, Kinds.kindName(m.location()), m.location()));
1038             }
1039 
1040             // Enter all type parameters into the local method scope.
1041             for (List<JCTypeParameter> l = tree.typarams; l.nonEmpty(); l = l.tail)
1042                 localEnv.info.scope.enterIfAbsent(l.head.type.tsym);
1043 
1044             ClassSymbol owner = env.enclClass.sym;
1045             if ((owner.flags() & ANNOTATION) != 0 &&
1046                     (tree.params.nonEmpty() ||
1047                     tree.recvparam != null))
1048                 log.error(tree.params.nonEmpty() ?
1049                         tree.params.head.pos() :
1050                         tree.recvparam.pos(),
1051                         Errors.IntfAnnotationMembersCantHaveParams);
1052 
1053             // Attribute all value parameters.
1054             for (List<JCVariableDecl> l = tree.params; l.nonEmpty(); l = l.tail) {
1055                 attribStat(l.head, localEnv);
1056             }
1057 
1058             chk.checkVarargsMethodDecl(localEnv, tree);
1059 
1060             // Check that type parameters are well-formed.
1061             chk.validate(tree.typarams, localEnv);
1062 
1063             // Check that result type is well-formed.
1064             if (tree.restype != null && !tree.restype.type.hasTag(VOID))
1065                 chk.validate(tree.restype, localEnv);
1066 
1067             // Check that receiver type is well-formed.
1068             if (tree.recvparam != null) {
1069                 // Use a new environment to check the receiver parameter.
1070                 // Otherwise I get "might not have been initialized" errors.
1071                 // Is there a better way?
1072                 Env<AttrContext> newEnv = memberEnter.methodEnv(tree, env);
1073                 attribType(tree.recvparam, newEnv);
1074                 chk.validate(tree.recvparam, newEnv);
1075             }
1076 
1077             if (env.enclClass.sym.isRecord() && tree.sym.owner.kind == TYP) {
1078                 // lets find if this method is an accessor
1079                 Optional<? extends RecordComponent> recordComponent = env.enclClass.sym.getRecordComponents().stream()
1080                         .filter(rc -> rc.accessor == tree.sym && (rc.accessor.flags_field & GENERATED_MEMBER) == 0).findFirst();
1081                 if (recordComponent.isPresent()) {
1082                     // the method is a user defined accessor lets check that everything is fine
1083                     if (!tree.sym.isPublic()) {
1084                         log.error(tree, Errors.InvalidAccessorMethodInRecord(env.enclClass.sym, Fragments.MethodMustBePublic));
1085                     }
1086                     if (!types.isSameType(tree.sym.type.getReturnType(), recordComponent.get().type)) {
1087                         log.error(tree, Errors.InvalidAccessorMethodInRecord(env.enclClass.sym,
1088                                 Fragments.AccessorReturnTypeDoesntMatch(tree.sym, recordComponent.get())));
1089                     }
1090                     if (tree.sym.type.asMethodType().thrown != null && !tree.sym.type.asMethodType().thrown.isEmpty()) {
1091                         log.error(tree,
1092                                 Errors.InvalidAccessorMethodInRecord(env.enclClass.sym, Fragments.AccessorMethodCantThrowException));
1093                     }
1094                     if (!tree.typarams.isEmpty()) {
1095                         log.error(tree,
1096                                 Errors.InvalidAccessorMethodInRecord(env.enclClass.sym, Fragments.AccessorMethodMustNotBeGeneric));
1097                     }
1098                     if (tree.sym.isStatic()) {
1099                         log.error(tree,
1100                                 Errors.InvalidAccessorMethodInRecord(env.enclClass.sym, Fragments.AccessorMethodMustNotBeStatic));
1101                     }
1102                 }
1103 
1104                 if (tree.name == names.init) {
1105                     // if this a constructor other than the canonical one
1106                     if ((tree.sym.flags_field & RECORD) == 0) {
1107                         JCMethodInvocation app = TreeInfo.firstConstructorCall(tree);
1108                         if (app == null ||
1109                                 TreeInfo.name(app.meth) != names._this ||
1110                                 !checkFirstConstructorStat(app, tree, false)) {
1111                             log.error(tree, Errors.FirstStatementMustBeCallToAnotherConstructor);
1112                         }
1113                     } else {
1114                         // but if it is the canonical:
1115 
1116                         // if user generated, then it shouldn't explicitly invoke any other constructor
1117                         if ((tree.sym.flags_field & GENERATEDCONSTR) == 0) {
1118                             JCMethodInvocation app = TreeInfo.firstConstructorCall(tree);
1119                             if (app != null &&
1120                                     (TreeInfo.name(app.meth) == names._this ||
1121                                             TreeInfo.name(app.meth) == names._super) &&
1122                                     checkFirstConstructorStat(app, tree, false)) {
1123                                 log.error(tree, Errors.InvalidCanonicalConstructorInRecord(
1124                                         Fragments.Canonical, tree.sym.name,
1125                                         Fragments.CanonicalMustNotContainExplicitConstructorInvocation));
1126                             }
1127                         }
1128 
1129                         // also we want to check that no type variables have been defined
1130                         if (!tree.typarams.isEmpty()) {
1131                             log.error(tree, Errors.InvalidCanonicalConstructorInRecord(
1132                                     Fragments.Canonical, tree.sym.name, Fragments.CanonicalMustNotDeclareTypeVariables));
1133                         }
1134 
1135                         /* and now we need to check that the constructor's arguments are exactly the same as those of the
1136                          * record components
1137                          */
1138                         List<Type> recordComponentTypes = TreeInfo.recordFields(env.enclClass).map(vd -> vd.sym.type);
1139                         for (JCVariableDecl param: tree.params) {
1140                             if (!types.isSameType(param.type, recordComponentTypes.head)) {
1141                                 log.error(param, Errors.InvalidCanonicalConstructorInRecord(
1142                                         Fragments.Canonical, tree.sym.name, Fragments.TypeMustBeIdenticalToCorrespondingRecordComponentType));
1143                             }
1144                             recordComponentTypes = recordComponentTypes.tail;
1145                         }
1146                     }
1147                 }
1148             }
1149 
1150             // annotation method checks
1151             if ((owner.flags() & ANNOTATION) != 0) {
1152                 // annotation method cannot have throws clause
1153                 if (tree.thrown.nonEmpty()) {
1154                     log.error(tree.thrown.head.pos(),
1155                               Errors.ThrowsNotAllowedInIntfAnnotation);
1156                 }
1157                 // annotation method cannot declare type-parameters
1158                 if (tree.typarams.nonEmpty()) {
1159                     log.error(tree.typarams.head.pos(),
1160                               Errors.IntfAnnotationMembersCantHaveTypeParams);
1161                 }
1162                 // validate annotation method's return type (could be an annotation type)
1163                 chk.validateAnnotationType(tree.restype);
1164                 // ensure that annotation method does not clash with members of Object/Annotation
1165                 chk.validateAnnotationMethod(tree.pos(), m);
1166             }
1167 
1168             for (List<JCExpression> l = tree.thrown; l.nonEmpty(); l = l.tail)
1169                 chk.checkType(l.head.pos(), l.head.type, syms.throwableType);
1170 
1171             if (tree.body == null) {
1172                 // Empty bodies are only allowed for
1173                 // abstract, native, or interface methods, or for methods
1174                 // in a retrofit signature class.
1175                 if (tree.defaultValue != null) {
1176                     if ((owner.flags() & ANNOTATION) == 0)
1177                         log.error(tree.pos(),
1178                                   Errors.DefaultAllowedInIntfAnnotationMember);
1179                 }
1180                 if (isDefaultMethod || (tree.sym.flags() & (ABSTRACT | NATIVE)) == 0)
1181                     log.error(tree.pos(), Errors.MissingMethBodyOrDeclAbstract);
1182             } else if ((tree.sym.flags() & (ABSTRACT|DEFAULT|PRIVATE)) == ABSTRACT) {
1183                 if ((owner.flags() & INTERFACE) != 0) {
1184                     log.error(tree.body.pos(), Errors.IntfMethCantHaveBody);
1185                 } else {
1186                     log.error(tree.pos(), Errors.AbstractMethCantHaveBody);
1187                 }
1188             } else if ((tree.mods.flags & NATIVE) != 0) {
1189                 log.error(tree.pos(), Errors.NativeMethCantHaveBody);
1190             } else {
1191                 // Add an implicit super() call unless an explicit call to
1192                 // super(...) or this(...) is given
1193                 // or we are compiling class java.lang.Object.
1194                 if (tree.name == names.init && owner.type != syms.objectType) {
1195                     JCBlock body = tree.body;
1196                     if (body.stats.isEmpty() ||
1197                             TreeInfo.getConstructorInvocationName(body.stats, names) == names.empty) {
1198                         JCStatement supCall = make.at(body.pos).Exec(make.Apply(List.nil(),
1199                                 make.Ident(names._super), make.Idents(List.nil())));
1200                         body.stats = body.stats.prepend(supCall);
1201                     } else if ((env.enclClass.sym.flags() & ENUM) != 0 &&
1202                             (tree.mods.flags & GENERATEDCONSTR) == 0 &&
1203                             TreeInfo.isSuperCall(body.stats.head)) {
1204                         // enum constructors are not allowed to call super
1205                         // directly, so make sure there aren't any super calls
1206                         // in enum constructors, except in the compiler
1207                         // generated one.
1208                         log.error(tree.body.stats.head.pos(),
1209                                   Errors.CallToSuperNotAllowedInEnumCtor(env.enclClass.sym));
1210                     }
1211                     if (env.enclClass.sym.isRecord() && (tree.sym.flags_field & RECORD) != 0) { // we are seeing the canonical constructor
1212                         List<Name> recordComponentNames = TreeInfo.recordFields(env.enclClass).map(vd -> vd.sym.name);
1213                         List<Name> initParamNames = tree.sym.params.map(p -> p.name);
1214                         if (!initParamNames.equals(recordComponentNames)) {
1215                             log.error(tree, Errors.InvalidCanonicalConstructorInRecord(

1684     private void handleSwitch(JCTree switchTree,
1685                               JCExpression selector,
1686                               List<JCCase> cases,
1687                               BiConsumer<JCCase, Env<AttrContext>> attribCase) {
1688         Type seltype = attribExpr(selector, env);
1689 
1690         Env<AttrContext> switchEnv =
1691             env.dup(switchTree, env.info.dup(env.info.scope.dup()));
1692 
1693         try {
1694             boolean enumSwitch = (seltype.tsym.flags() & Flags.ENUM) != 0;
1695             boolean stringSwitch = types.isSameType(seltype, syms.stringType);
1696             if (!enumSwitch && !stringSwitch)
1697                 seltype = chk.checkType(selector.pos(), seltype, syms.intType);
1698 
1699             // Attribute all cases and
1700             // check that there are no duplicate case labels or default clauses.
1701             Set<Object> labels = new HashSet<>(); // The set of case labels.
1702             boolean hasDefault = false;      // Is there a default label?
1703             CaseTree.CaseKind caseKind = null;
1704             boolean wasError = false;
1705             for (List<JCCase> l = cases; l.nonEmpty(); l = l.tail) {
1706                 JCCase c = l.head;
1707                 if (caseKind == null) {
1708                     caseKind = c.caseKind;
1709                 } else if (caseKind != c.caseKind && !wasError) {
1710                     log.error(c.pos(),
1711                               Errors.SwitchMixingCaseTypes);
1712                     wasError = true;
1713                 }
1714                 if (c.getExpressions().nonEmpty()) {
1715                     for (JCExpression pat : c.getExpressions()) {
1716                         if (TreeInfo.isNull(pat)) {
1717                             log.error(pat.pos(),
1718                                       Errors.SwitchNullNotAllowed);
1719                         } else if (enumSwitch) {
1720                             Symbol sym = enumConstant(pat, seltype);
1721                             if (sym == null) {
1722                                 log.error(pat.pos(), Errors.EnumLabelMustBeUnqualifiedEnum);
1723                             } else if (!labels.add(sym)) {
1724                                 log.error(c.pos(), Errors.DuplicateCaseLabel);
1725                             }
1726                         } else {
1727                             Type pattype = attribExpr(pat, switchEnv, seltype);
1728                             if (!pattype.hasTag(ERROR)) {
1729                                 if (pattype.constValue() == null) {
1730                                     log.error(pat.pos(),
1731                                               (stringSwitch ? Errors.StringConstReq : Errors.ConstExprReq));
1732                                 } else if (!labels.add(pattype.constValue())) {
1733                                     log.error(c.pos(), Errors.DuplicateCaseLabel);
1734                                 }
1735                             }
1736                         }
1737                     }
1738                 } else if (hasDefault) {
1739                     log.error(c.pos(), Errors.DuplicateDefaultLabel);
1740                 } else {
1741                     hasDefault = true;
1742                 }
1743                 Env<AttrContext> caseEnv =
1744                     switchEnv.dup(c, env.info.dup(switchEnv.info.scope.dup()));
1745                 try {
1746                     attribCase.accept(c, caseEnv);
1747                 } finally {
1748                     caseEnv.info.scope.leave();
1749                 }
1750                 addVars(c.stats, switchEnv.info.scope);
1751             }
1752         } finally {
1753             switchEnv.info.scope.leave();
1754         }
1755     }

2009         /** Compute the type of a conditional expression, after
2010          *  checking that it exists.  See JLS 15.25. Does not take into
2011          *  account the special case where condition and both arms
2012          *  are constants.
2013          *
2014          *  @param pos      The source position to be used for error
2015          *                  diagnostics.
2016          *  @param thentype The type of the expression's then-part.
2017          *  @param elsetype The type of the expression's else-part.
2018          */
2019         Type condType(List<DiagnosticPosition> positions, List<Type> condTypes) {
2020             if (condTypes.isEmpty()) {
2021                 return syms.objectType; //TODO: how to handle?
2022             }
2023             Type first = condTypes.head;
2024             // If same type, that is the result
2025             if (condTypes.tail.stream().allMatch(t -> types.isSameType(first, t)))
2026                 return first.baseType();
2027 
2028             List<Type> unboxedTypes = condTypes.stream()
2029                                                .map(t -> t.isPrimitive() ? t : types.unboxedType(t))
2030                                                .collect(List.collector());
2031 
2032             // Otherwise, if both arms can be converted to a numeric
2033             // type, return the least numeric type that fits both arms
2034             // (i.e. return larger of the two, or return int if one
2035             // arm is short, the other is char).
2036             if (unboxedTypes.stream().allMatch(t -> t.isPrimitive())) {
2037                 // If one arm has an integer subrange type (i.e., byte,
2038                 // short, or char), and the other is an integer constant
2039                 // that fits into the subrange, return the subrange type.
2040                 for (Type type : unboxedTypes) {
2041                     if (!type.getTag().isStrictSubRangeOf(INT)) {
2042                         continue;
2043                     }
2044                     if (unboxedTypes.stream().filter(t -> t != type).allMatch(t -> t.hasTag(INT) && types.isAssignable(t, type)))
2045                         return type.baseType();
2046                 }
2047 
2048                 for (TypeTag tag : primitiveTags) {
2049                     Type candidate = syms.typeOfTag[tag.ordinal()];
2050                     if (unboxedTypes.stream().allMatch(t -> types.isSubtype(t, candidate))) {
2051                         return candidate;
2052                     }
2053                 }
2054             }
2055 
2056             // Those were all the cases that could result in a primitive
2057             condTypes = condTypes.stream()
2058                                  .map(t -> t.isPrimitive() ? types.boxedClass(t).type : t)
2059                                  .collect(List.collector());
2060 
2061             for (Type type : condTypes) {
2062                 if (condTypes.stream().filter(t -> t != type).allMatch(t -> types.isAssignable(t, type)))
2063                     return type.baseType();
2064             }
2065 
2066             Iterator<DiagnosticPosition> posIt = positions.iterator();
2067 
2068             condTypes = condTypes.stream()
2069                                  .map(t -> chk.checkNonVoid(posIt.next(), t))
2070                                  .collect(List.collector());
2071 
2072             // both are known to be reference types.  The result is
2073             // lub(thentype,elsetype). This cannot fail, as it will
2074             // always be possible to infer "Object" if nothing better.
2075             return types.lub(condTypes.stream().map(t -> t.baseType()).collect(List.collector()));
2076         }

2216          *  report an error if not.

2218          *  (non-labelled) statement tree referred to by the label,
2219          *  not the tree representing the labelled statement itself.

2222          *                 Tree.BREAK or Tree.CONTINUE.
2223          *  @param label   The label of the jump statement, or null if no
2224          *                 label is given.
2225          *  @param env     The environment current at the jump statement.
2226          */

2228                                                                        Name label,
2229                                                                        Env<AttrContext> env) {
2230             // Search environments outwards from the point of jump.
2231             Env<AttrContext> env1 = env;
2232             JCDiagnostic.Error pendingError = null;
2233             LOOP:
2234             while (env1 != null) {
2235                 switch (env1.tree.getTag()) {
2236                     case LABELLED:
2237                         JCLabeledStatement labelled = (JCLabeledStatement)env1.tree;

2239                             // If jump is a continue, check that target is a loop.
2240                             if (tag == CONTINUE) {
2241                                 if (!labelled.body.hasTag(DOLOOP) &&
2242                                         !labelled.body.hasTag(WHILELOOP) &&
2243                                         !labelled.body.hasTag(FORLOOP) &&
2244                                         !labelled.body.hasTag(FOREACHLOOP)) {
2245                                     pendingError = Errors.NotLoopLabel(label);
2246                                 }
2247                                 // Found labelled statement target, now go inwards
2248                                 // to next non-labelled tree.
2249                                 return Pair.of(TreeInfo.referencedStatement(labelled), pendingError);
2250                             } else {
2251                                 return Pair.of(labelled, pendingError);
2252                             }
2253                         }
2254                         break;
2255                     case DOLOOP:
2256                     case WHILELOOP:
2257                     case FORLOOP:
2258                     case FOREACHLOOP:
2259                         if (label == null) return Pair.of(env1.tree, pendingError);
2260                         break;
2261                     case SWITCH:
2262                         if (label == null && tag == BREAK) return Pair.of(env1.tree, null);
2263                         break;
2264                     case SWITCH_EXPRESSION:
2265                         if (tag == YIELD) {
2266                             return Pair.of(env1.tree, null);
2267                         } else if (tag == BREAK) {
2268                             pendingError = Errors.BreakOutsideSwitchExpression;
2269                         } else {
2270                             pendingError = Errors.ContinueOutsideSwitchExpression;
2271                         }
2272                         break;
2273                     case LAMBDA:
2274                     case METHODDEF:
2275                     case CLASSDEF:
2276                         break LOOP;
2277                     default:
2278                 }
2279                 env1 = env1.next;
2280             }
2281             if (label != null)
2282                 return Pair.of(null, Errors.UndefLabel(label));
2283             else if (pendingError != null)
2284                 return Pair.of(null, pendingError);
2285             else if (tag == CONTINUE)
2286                 return Pair.of(null, Errors.ContOutsideLoop);

2794                                                    List<Type> argtypes, List<Type> typeargtypes,
2795                                                    KindSelector pkind) {
2796             // We are seeing an anonymous class instance creation.
2797             // In this case, the class instance creation
2798             // expression
2799             //
2800             //    E.new <typeargs1>C<typargs2>(args) { ... }
2801             //
2802             // is represented internally as
2803             //
2804             //    E . new <typeargs1>C<typargs2>(args) ( class <empty-name> { ... } )  .
2805             //
2806             // This expression is then *transformed* as follows:
2807             //
2808             // (1) add an extends or implements clause
2809             // (2) add a constructor.
2810             //
2811             // For instance, if C is a class, and ET is the type of E,
2812             // the expression
2813             //
2814             //    E.new <typeargs1>C<typargs2>(args) { ... }
2815             //
2816             // is translated to (where X is a fresh name and typarams is the
2817             // parameter list of the super constructor):
2818             //
2819             //   new <typeargs1>X(<*nullchk*>E, args) where
2820             //     X extends C<typargs2> {
2821             //       <typarams> X(ET e, args) {
2822             //         e.<typeargs1>super(args)
2823             //       }
2824             //       ...
2825             //     }
2826             InferenceContext inferenceContext = resultInfo.checkContext.inferenceContext();
2827             final boolean isDiamond = TreeInfo.isDiamond(tree);
2828             if (isDiamond
2829                     && ((tree.constructorType != null && inferenceContext.free(tree.constructorType))
2830                     || (tree.clazz.type != null && inferenceContext.free(tree.clazz.type)))) {
2831                 final ResultInfo resultInfoForClassDefinition = this.resultInfo;
2832                 inferenceContext.addFreeTypeListener(List.of(tree.constructorType, tree.clazz.type),
2833                         instantiatedContext -> {
2834                             tree.constructorType = instantiatedContext.asInstType(tree.constructorType);
2835                             tree.clazz.type = clazz.type = instantiatedContext.asInstType(clazz.type);
2836                             ResultInfo prevResult = this.resultInfo;
2837                             try {
2838                                 this.resultInfo = resultInfoForClassDefinition;
2839                                 visitAnonymousClassDefinition(tree, clazz, clazz.type, cdef,
2840                                                             localEnv, argtypes, typeargtypes, pkind);
2841                             } finally {
2842                                 this.resultInfo = prevResult;
2843                             }
2844                         });
2845             } else {
2846                 if (isDiamond && clazztype.hasTag(CLASS)) {
2847                     List<Type> invalidDiamondArgs = chk.checkDiamondDenotable((ClassType)clazztype);
2848                     if (!clazztype.isErroneous() && invalidDiamondArgs.nonEmpty()) {
2849                         // One or more types inferred in the previous steps is non-denotable.
2850                         Fragment fragment = Diamond(clazztype.tsym);
2851                         log.error(tree.clazz.pos(),
2852                                 Errors.CantApplyDiamond1(
2853                                         fragment,
2854                                         invalidDiamondArgs.size() > 1 ?
2855                                                 DiamondInvalidArgs(invalidDiamondArgs, fragment) :
2856                                                 DiamondInvalidArg(invalidDiamondArgs, fragment)));
2857                     }

2859                     for (Type t : clazztype.getTypeArguments()) {
2860                         rs.checkAccessibleType(env, t);
2861                     }

2863                 }

2866                 // false for isInterface call even when the original type is an interface.
2867                 boolean implementing = clazztype.tsym.isInterface() ||
2868                         clazztype.isErroneous() && !clazztype.getOriginalType().hasTag(NONE) &&
2869                         clazztype.getOriginalType().tsym.isInterface();
2870

2872                     cdef.implementing = List.of(clazz);
2873                 } else {
2874                     cdef.extending = clazz;
2875                 }
2876 
2877                 if (resultInfo.checkContext.deferredAttrContext().mode == DeferredAttr.AttrMode.CHECK &&
2878                     isSerializable(clazztype)) {
2879                     localEnv.info.isSerializable = true;
2880                 }
2881 
2882                 attribStat(cdef, localEnv);
2883 
2884                 List<Type> finalargtypes;
2885                 // If an outer instance is given,
2886                 // prefix it to the constructor arguments
2887                 // and delete it from the new expression
2888                 if (tree.encl != null && !clazztype.tsym.isInterface()) {
2889                     finalargtypes = argtypes.prepend(tree.encl.type);
2890                 } else {
2891                     finalargtypes = argtypes;
2892                 }
2893 
2894                 // Reassign clazztype and recompute constructor. As this necessarily involves
2895                 // another attribution pass for deferred types in the case of <>, replicate
2896                 // them. Original arguments have right decorations already.
2897                 if (isDiamond && pkind.contains(KindSelector.POLY)) {
2898                     finalargtypes = finalargtypes.map(deferredAttr.deferredCopier);
2899                 }
2900 
2901                 clazztype = clazztype.hasTag(ERROR) ? types.createErrorType(cdef.sym.type)
2902                                                     : cdef.sym.type;
2903                 Symbol sym = tree.constructor = rs.resolveConstructor(
2904                         tree.pos(), localEnv, clazztype, finalargtypes, typeargtypes);
2905                 Assert.check(!sym.kind.isResolutionError());
2906                 tree.constructor = sym;

2909                         tree.constructor,
2910                         localEnv,
2911                         new ResultInfo(pkind, newMethodTemplate(syms.voidType, finalargtypes, typeargtypes), CheckMode.NO_TREE_UPDATE));

3723     boolean checkExConstraints(
3724             List<Type> thrownByFuncExpr,
3725             List<Type> thrownAtFuncType,
3726             InferenceContext inferenceContext) {
3727         /** 18.2.5: Otherwise, let E1, ..., En be the types in the function type's throws clause that
3728          *  are not proper types
3729          */
3730         List<Type> nonProperList = thrownAtFuncType.stream()
3731                 .filter(e -> inferenceContext.free(e)).collect(List.collector());
3732         List<Type> properList = thrownAtFuncType.diff(nonProperList);
3733 
3734         /** Let X1,...,Xm be the checked exception types that the lambda body can throw or
3735          *  in the throws clause of the invocation type of the method reference's compile-time
3736          *  declaration
3737          */
3738         List<Type> checkedList = thrownByFuncExpr.stream()
3739                 .filter(e -> chk.isChecked(e)).collect(List.collector());
3740 
3741         /** If n = 0 (the function type's throws clause consists only of proper types), then
3742          *  if there exists some i (1 <= i <= m) such that Xi is not a subtype of any proper type
3743          *  in the throws clause, the constraint reduces to false; otherwise, the constraint
3744          *  reduces to true
3745          */
3746         ListBuffer<Type> uncaughtByProperTypes = new ListBuffer<>();
3747         for (Type checked : checkedList) {
3748             boolean isSubtype = false;
3749             for (Type proper : properList) {
3750                 if (types.isSubtype(checked, proper)) {
3751                     isSubtype = true;
3752                     break;
3753                 }
3754             }
3755             if (!isSubtype) {
3756                 uncaughtByProperTypes.add(checked);
3757             }
3758         }
3759 
3760         if (nonProperList.isEmpty() && !uncaughtByProperTypes.isEmpty()) {
3761             return false;
3762         }
3763 
3764         /** If n > 0, the constraint reduces to a set of subtyping constraints:
3765          *  for all i (1 <= i <= m), if Xi is not a subtype of any proper type in the
3766          *  throws clause, then the constraints include, for all j (1 <= j <= n), <Xi <: Ej>
3767          */
3768         List<Type> nonProperAsUndet = inferenceContext.asUndetVars(nonProperList);
3769         uncaughtByProperTypes.forEach(checkedEx -> {
3770             nonProperAsUndet.forEach(nonProper -> {
3771                 types.isSubtype(checkedEx, nonProper);
3772             });
3773         });
3774 
3775         /** In addition, for all j (1 <= j <= n), the constraint reduces to the bound throws Ej
3776          */
3777         nonProperAsUndet.stream()
3778                 .filter(t -> t.hasTag(UNDETVAR))
3779                 .forEach(t -> ((UndetVar)t).setThrow());
3780         return true;
3781     }

4634         /**

4636          * an enum type, constructors and initializers may not
4637          * reference its static members unless they are constant.
4638          *
4639          * @param tree    The tree making up the variable reference.
4640          * @param env     The current environment.
4641          * @param v       The variable's symbol.
4642          * @jls 8.9 Enum Types
4643          */
4644         private void checkEnumInitializer(JCTree tree, Env<AttrContext> env, VarSymbol v) {
4645             // JLS:
4646             //
4647             // "It is a compile-time error to reference a static field
4648             // of an enum type that is not a compile-time constant
4649             // (15.28) from constructors, instance initializer blocks,
4650             // or instance variable initializer expressions of that
4651             // type. It is a compile-time error for the constructors,
4652             // instance initializer blocks, or instance variable
4653             // initializer expressions of an enum constant e to refer
4654             // to itself or to an enum constant of the same type that
4655             // is declared to the right of e."
4656             if (isStaticEnumField(v)) {
4657                 ClassSymbol enclClass = env.info.scope.owner.enclClass();
4658 
4659                 if (enclClass == null || enclClass.owner == null)
4660                     return;
4661 
4662                 // See if the enclosing class is the enum (or a
4663                 // subclass thereof) declaring v.  If not, this
4664                 // reference is OK.
4665                 if (v.owner != enclClass && !types.isSubtype(enclClass.type, v.owner.type))
4666                     return;
4667 
4668                 // If the reference isn't from an initializer, then
4669                 // the reference is OK.
4670                 if (!Resolve.isInitializer(env))
4671                     return;
4672 
4673                 log.error(tree.pos(), Errors.IllegalEnumStaticRef);
4674             }
4675         }

4949     Type checkIntersection(JCTree tree, List<JCExpression> bounds) {
4950         Set<Type> boundSet = new HashSet<>();
4951         if (bounds.nonEmpty()) {
4952             // accept class or interface or typevar as first bound.
4953             bounds.head.type = checkBase(bounds.head.type, bounds.head, env, false, false, false);
4954             boundSet.add(types.erasure(bounds.head.type));
4955             if (bounds.head.type.isErroneous()) {
4956                 return bounds.head.type;
4957             }
4958             else if (bounds.head.type.hasTag(TYPEVAR)) {
4959                 // if first bound was a typevar, do not accept further bounds.
4960                 if (bounds.tail.nonEmpty()) {
4961                     log.error(bounds.tail.head.pos(),
4962                               Errors.TypeVarMayNotBeFollowedByOtherBounds);
4963                     return bounds.head.type;
4964                 }
4965             } else {
4966                 // if first bound was a class or interface, accept only interfaces
4967                 // as further bounds.
4968                 for (JCExpression bound : bounds.tail) {
4969                     bound.type = checkBase(bound.type, bound, env, false, true, false);
4970                     if (bound.type.isErroneous()) {
4971                         bounds = List.of(bound);
4972                     }
4973                     else if (bound.type.hasTag(CLASS)) {
4974                         chk.checkNotRepeated(bound.pos(), types.erasure(bound.type), boundSet);
4975                     }
4976                 }
4977             }
4978         }
4979 
4980         if (bounds.length() == 0) {
4981             return syms.objectType;
4982         } else if (bounds.length() == 1) {
4983             return bounds.head.type;
4984         } else {
4985             Type owntype = types.makeIntersectionType(TreeInfo.types(bounds));
4986             // ... the variable's bound is a class type flagged COMPOUND
4987             // (see comment for TypeVar.bound).
4988             // In this case, generate a class tree that represents the
4989             // bound class, ...
4990             JCExpression extending;
4991             List<JCExpression> implementing;
4992             if (!bounds.head.type.isInterface()) {
4993                 extending = bounds.head;
4994                 implementing = bounds.tail;
4995             } else {
4996                 extending = null;
4997                 implementing = bounds;
4998             }
4999             JCClassDecl cd = make.at(tree).ClassDef(
5000                 make.Modifiers(PUBLIC | ABSTRACT | (extending != null && TreeInfo.symbol(extending).isValue() ? VALUE : 0)),
5001                 names.empty, List.nil(),
5002                 extending, implementing, List.nil());
5003 
5004             ClassSymbol c = (ClassSymbol)owntype.tsym;
5005             Assert.check((c.flags() & COMPOUND) != 0);
5006             cd.sym = c;
5007             c.sourcefile = env.toplevel.sourcefile;
5008 
5009             // ... and attribute the bound class
5010             c.flags_field |= UNATTRIBUTED;
5011             Env<AttrContext> cenv = enter.classEnv(cd, env);
5012             typeEnvs.put(c, cenv);
5013             attribClass(c);
5014             return owntype;

5139     /** Attribute class definition associated with given class symbol.
5140      *  @param c   The class symbol whose definition will be attributed.
5141      */
5142     void attribClass(ClassSymbol c) throws CompletionFailure {
5143         if (c.type.hasTag(ERROR)) return;
5144 
5145         // Check for cycles in the inheritance graph, which can arise from
5146         // ill-formed class files.
5147         chk.checkNonCyclic(null, c.type);
5148 
5149         Type st = types.supertype(c.type);
5150         if ((c.flags_field & Flags.COMPOUND) == 0) {
5151             // First, attribute superclass.
5152             if (st.hasTag(CLASS))
5153                 attribClass((ClassSymbol)st.tsym);
5154 
5155             // Next attribute owner, if it is a class.
5156             if (c.owner.kind == TYP && c.owner.type.hasTag(CLASS))
5157                 attribClass((ClassSymbol)c.owner);
5158         }
5159 
5160         // The previous operations might have attributed the current class
5161         // if there was a cycle. So we test first whether the class is still
5162         // UNATTRIBUTED.
5163         if ((c.flags_field & UNATTRIBUTED) != 0) {
5164             c.flags_field &= ~UNATTRIBUTED;
5165 
5166             // Get environment current at the point of class definition.
5167             Env<AttrContext> env = typeEnvs.get(c);
5168 
5169             // The info.lint field in the envs stored in typeEnvs is deliberately uninitialized,
5170             // because the annotations were not available at the time the env was created. Therefore,
5171             // we look up the environment chain for the first enclosing environment for which the
5172             // lint value is set. Typically, this is the parent env, but might be further if there
5173             // are any envs created as a result of TypeParameter nodes.
5174             Env<AttrContext> lintEnv = env;
5175             while (lintEnv.info.lint == null)
5176                 lintEnv = lintEnv.next;
5177 
5178             // Having found the enclosing lint value, we can initialize the lint value for this class
5179             env.info.lint = lintEnv.info.lint.augment(c);

5548         private void validateAnnotatedType(final JCTree errtree, final Type type) {
5549             //System.err.println("Attr.validateAnnotatedType: " + errtree + " type: " + type);
5550 
5551             if (type.isPrimitiveOrVoid()) {
5552                 return;
5553             }
5554 
5555             JCTree enclTr = errtree;
5556             Type enclTy = type;
5557 
5558             boolean repeat = true;
5559             while (repeat) {
5560                 if (enclTr.hasTag(TYPEAPPLY)) {
5561                     List<Type> tyargs = enclTy.getTypeArguments();
5562                     List<JCExpression> trargs = ((JCTypeApply)enclTr).getTypeArguments();
5563                     if (trargs.length() > 0) {
5564                         // Nothing to do for diamonds
5565                         if (tyargs.length() == trargs.length()) {
5566                             for (int i = 0; i < tyargs.length(); ++i) {
5567                                 validateAnnotatedType(trargs.get(i), tyargs.get(i));
5568                             }
5569                         }
5570                         // If the lengths don't match, it's either a diamond
5571                         // or some nested type that redundantly provides
5572                         // type arguments in the tree.
5573                     }
5574 
5575                     // Look at the clazz part of a generic type
5576                     enclTr = ((JCTree.JCTypeApply)enclTr).clazz;
5577                 }
5578 
5579                 if (enclTr.hasTag(SELECT)) {
5580                     enclTr = ((JCTree.JCFieldAccess)enclTr).getExpression();
5581                     if (enclTy != null &&
5582                             !enclTy.hasTag(NONE)) {
5583                         enclTy = enclTy.getEnclosingType();
5584                     }
5585                 } else if (enclTr.hasTag(ANNOTATED_TYPE)) {
5586                     JCAnnotatedType at = (JCTree.JCAnnotatedType) enclTr;

5590                         } else {

5592                             for (JCAnnotation an : at.getAnnotations()) {
5593                                 comps.add(an.attribute);
5594                             }

5596                         }
5597                         repeat = false;
5598                     }

5600                     // enclTy doesn't need to be changed
5601                 } else if (enclTr.hasTag(IDENT)) {
5602                     repeat = false;

5614                     validateAnnotatedType(art.getType(), art.elemtype.type);
5615                     repeat = false;
