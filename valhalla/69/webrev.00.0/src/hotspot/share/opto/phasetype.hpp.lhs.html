 25 #ifndef SHARE_OPTO_PHASETYPE_HPP
 26 #define SHARE_OPTO_PHASETYPE_HPP
 27 
 28 enum CompilerPhaseType {
 29   PHASE_BEFORE_STRINGOPTS,
 30   PHASE_AFTER_STRINGOPTS,
 31   PHASE_BEFORE_REMOVEUSELESS,
 32   PHASE_AFTER_PARSING,
 33   PHASE_ITER_GVN1,
 34   PHASE_PHASEIDEAL_BEFORE_EA,
 35   PHASE_ITER_GVN_AFTER_EA,
 36   PHASE_ITER_GVN_AFTER_ELIMINATION,
 37   PHASE_PHASEIDEALLOOP1,
 38   PHASE_PHASEIDEALLOOP2,
 39   PHASE_PHASEIDEALLOOP3,
 40   PHASE_CPP1,
 41   PHASE_ITER_GVN2,
 42   PHASE_PHASEIDEALLOOP_ITERATIONS,
 43   PHASE_OPTIMIZE_FINISHED,
 44   PHASE_GLOBAL_CODE_MOTION,
 45   PHASE_FINAL_CODE,
 46   PHASE_AFTER_EA,
 47   PHASE_BEFORE_CLOOPS,
 48   PHASE_AFTER_CLOOPS,
 49   PHASE_BEFORE_BEAUTIFY_LOOPS,
 50   PHASE_AFTER_BEAUTIFY_LOOPS,
 51   PHASE_BEFORE_MATCHING,
 52   PHASE_MATCHING,
 53   PHASE_INCREMENTAL_INLINE,
 54   PHASE_INCREMENTAL_BOXING_INLINE,
 55   PHASE_CALL_CATCH_CLEANUP,
 56   PHASE_INSERT_BARRIER,
 57   PHASE_MACRO_EXPANSION,
 58   PHASE_BARRIER_EXPANSION,
 59   PHASE_ADD_UNSAFE_BARRIER,
 60   PHASE_END,
 61   PHASE_FAILURE,
 62   PHASE_SPLIT_VALUES_ARRAY,
 63   PHASE_SPLIT_VALUES_ARRAY_IGVN,
 64 
 65   PHASE_NUM_TYPES
 66 };
 67 
 68 class CompilerPhaseTypeHelper {
 69   public:
 70   static const char* to_string(CompilerPhaseType cpt) {
 71     switch (cpt) {
 72       case PHASE_BEFORE_STRINGOPTS:          return "Before StringOpts";
 73       case PHASE_AFTER_STRINGOPTS:           return "After StringOpts";
 74       case PHASE_BEFORE_REMOVEUSELESS:       return "Before RemoveUseless";
 75       case PHASE_AFTER_PARSING:              return "After Parsing";
 76       case PHASE_ITER_GVN1:                  return "Iter GVN 1";
 77       case PHASE_PHASEIDEAL_BEFORE_EA:       return "PhaseIdealLoop before EA";
 78       case PHASE_ITER_GVN_AFTER_EA:          return "Iter GVN after EA";
 79       case PHASE_ITER_GVN_AFTER_ELIMINATION: return "Iter GVN after eliminating allocations and locks";
 80       case PHASE_PHASEIDEALLOOP1:            return "PhaseIdealLoop 1";
 81       case PHASE_PHASEIDEALLOOP2:            return "PhaseIdealLoop 2";
 82       case PHASE_PHASEIDEALLOOP3:            return "PhaseIdealLoop 3";
 83       case PHASE_CPP1:                       return "PhaseCPP 1";
 84       case PHASE_ITER_GVN2:                  return "Iter GVN 2";
 85       case PHASE_PHASEIDEALLOOP_ITERATIONS:  return "PhaseIdealLoop iterations";
 86       case PHASE_OPTIMIZE_FINISHED:          return "Optimize finished";
 87       case PHASE_GLOBAL_CODE_MOTION:         return "Global code motion";
 88       case PHASE_FINAL_CODE:                 return "Final Code";
 89       case PHASE_AFTER_EA:                   return "After Escape Analysis";
 90       case PHASE_BEFORE_CLOOPS:              return "Before CountedLoop";
 91       case PHASE_AFTER_CLOOPS:               return "After CountedLoop";
 92       case PHASE_BEFORE_BEAUTIFY_LOOPS:      return "Before beautify loops";
 93       case PHASE_AFTER_BEAUTIFY_LOOPS:       return "After beautify loops";
 94       case PHASE_BEFORE_MATCHING:            return "Before matching";
 95       case PHASE_MATCHING:                   return "After matching";
 96       case PHASE_INCREMENTAL_INLINE:         return "Incremental Inline";
 97       case PHASE_INCREMENTAL_BOXING_INLINE:  return "Incremental Boxing Inline";
 98       case PHASE_CALL_CATCH_CLEANUP:         return "Call catch cleanup";
 99       case PHASE_INSERT_BARRIER:             return "Insert barrier";
100       case PHASE_MACRO_EXPANSION:            return "Macro expand";
101       case PHASE_BARRIER_EXPANSION:          return "Barrier expand";
102       case PHASE_ADD_UNSAFE_BARRIER:         return "Add barrier to unsafe op";
103       case PHASE_END:                        return "End";
104       case PHASE_FAILURE:                    return "Failure";
105       case PHASE_SPLIT_VALUES_ARRAY:         return "Split values array";
106       case PHASE_SPLIT_VALUES_ARRAY_IGVN:    return "IGVN after split values array";

107       default:
108         ShouldNotReachHere();
109         return NULL;
110     }
111   }
112 };
