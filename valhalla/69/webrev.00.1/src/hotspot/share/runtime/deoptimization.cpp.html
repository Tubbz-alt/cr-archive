  86 Deoptimization::UnrollBlock::UnrollBlock(int  size_of_deoptimized_frame,
  87                                          int  caller_adjustment,
  88                                          int  caller_actual_parameters,
  89                                          int  number_of_frames,
  90                                          intptr_t* frame_sizes,
  91                                          address* frame_pcs,
  92                                          BasicType return_type,
  93                                          int exec_mode) {
  94   _size_of_deoptimized_frame = size_of_deoptimized_frame;
  95   _caller_adjustment         = caller_adjustment;
  96   _caller_actual_parameters  = caller_actual_parameters;
  97   _number_of_frames          = number_of_frames;
  98   _frame_sizes               = frame_sizes;
  99   _frame_pcs                 = frame_pcs;
 100   _register_block            = NEW_C_HEAP_ARRAY(intptr_t, RegisterMap::reg_count * 2, mtCompiler);
 101   _return_type               = return_type;
 102   _initial_info              = 0;
 103   // PD (x86 only)
 104   _counter_temp              = 0;
 105   _unpack_kind               = exec_mode;
 106   _sender_sp_temp            = 0;
 107 
 108   _total_frame_sizes         = size_of_frames();
 109   assert(exec_mode >= 0 && exec_mode < Unpack_LIMIT, "Unexpected exec_mode");
 110 }

 127 int Deoptimization::UnrollBlock::size_of_frames() const {
 128   // Acount first for the adjustment of the initial frame
 129   int result = _caller_adjustment;
 130   for (int index = 0; index < number_of_frames(); index++) {
 131     result += frame_sizes()[index];
 132   }
 133   return result;
 134 }

 149 // In order to make fetch_unroll_info work properly with escape
 150 // analysis, The method was changed from JRT_LEAF to JRT_BLOCK_ENTRY and
 151 // ResetNoHandleMark and HandleMark were removed from it. The actual reallocation
 152 // of previously eliminated objects occurs in realloc_objects, which is
 153 // called from the method fetch_unroll_info_helper below.
 154 JRT_BLOCK_ENTRY(Deoptimization::UnrollBlock*, Deoptimization::fetch_unroll_info(JavaThread* thread, int exec_mode))
 155   // It is actually ok to allocate handles in a leaf method. It causes no safepoints,
 156   // but makes the entry a little slower. There is however a little dance we have to
 157   // do in debug mode to get around the NoHandleMark code in the JRT_LEAF macro
 158 
 159   // fetch_unroll_info() is called at the beginning of the deoptimization
 160   // handler. Note this fact before we start generating temporary frames
 161   // that can confuse an asynchronous stack walker. This counter is
 162   // decremented at the end of unpack_frames().
 163   if (TraceDeoptimization) {
 164     tty->print_cr("Deoptimizing thread " INTPTR_FORMAT, p2i(thread));
 165   }
 166   thread->inc_in_deopt_handler();
 167 
 168   return fetch_unroll_info_helper(thread, exec_mode);
 169 JRT_END

 171 #if COMPILER2_OR_JVMCI
 172 static bool eliminate_allocations(JavaThread* thread, int exec_mode, CompiledMethod* compiled_method,
 173                                   frame& deoptee, RegisterMap& map, GrowableArray<compiledVFrame*>* chunk) {
 174   bool realloc_failures = false;
 175   assert (chunk->at(0)->scope() != NULL,"expect only compiled java frames");
 176 
 177   GrowableArray<ScopeValue*>* objects = chunk->at(0)->scope()->objects();
 178 
 179   // The flag return_oop() indicates call sites which return oop
 180   // in compiled code. Such sites include java method calls,
 181   // runtime calls (for example, used to allocate new objects/arrays
 182   // on slow code path) and any other calls generated in compiled code.
 183   // It is not guaranteed that we can get such information here only
 184   // by analyzing bytecode in deoptimized frames. This is why this flag
 185   // is set during method compilation (see Compile::Process_OopMap_Node()).
 186   // If the previous frame was popped or if we are dispatching an exception,
 187   // we don't have an oop result.
 188   ScopeDesc* scope = chunk->at(0)->scope();
 189   bool save_oop_result = scope->return_oop() && !thread->popframe_forcing_deopt_reexecution() && (exec_mode == Deoptimization::Unpack_deopt);
 190   // In case of the return of multiple values, we must take care
 191   // of all oop return values.
 192   GrowableArray<Handle> return_oops;
 193   ValueKlass* vk = NULL;
 194   if (save_oop_result && scope->return_vt()) {
 195     vk = ValueKlass::returned_value_klass(map);
 196     if (vk != NULL) {
 197       vk->save_oop_fields(map, return_oops);
 198       save_oop_result = false;
 199     }
 200   }
 201   if (save_oop_result) {
 202     // Reallocation may trigger GC. If deoptimization happened on return from
 203     // call which returns oop we need to save it since it is not in oopmap.
 204     oop result = deoptee.saved_oop_result(&map);
 205     assert(oopDesc::is_oop_or_null(result), "must be oop");
 206     return_oops.push(Handle(thread, result));
 207     assert(Universe::heap()->is_in_or_null(result), "must be heap pointer");
 208     if (TraceDeoptimization) {
 209       ttyLocker ttyl;
 210       tty->print_cr("SAVED OOP RESULT " INTPTR_FORMAT " in thread " INTPTR_FORMAT, p2i(result), p2i(thread));
 211     }
 212   }
 213   if (objects != NULL || vk != NULL) {
 214     bool skip_internal = (compiled_method != NULL) && !compiled_method->is_compiled_by_jvmci();
 215     JRT_BLOCK
 216       if (vk != NULL) {
 217         realloc_failures = Deoptimization::realloc_value_type_result(vk, map, return_oops, THREAD);
 218       }
 219       if (objects != NULL) {
 220         realloc_failures = realloc_failures || Deoptimization::realloc_objects(thread, &deoptee, &map, objects, THREAD);
 221         Deoptimization::reassign_fields(&deoptee, &map, objects, realloc_failures, skip_internal, THREAD);
 222       }
 223     JRT_END
 224 #ifndef PRODUCT
 225     if (TraceDeoptimization) {
 226       ttyLocker ttyl;
 227       tty->print_cr("REALLOC OBJECTS in thread " INTPTR_FORMAT, p2i(thread));
 228       if (objects != NULL) {
 229         Deoptimization::print_objects(objects, realloc_failures);
 230       } else {
 231         Handle obj = realloc_failures ? Handle() : return_oops.first();
 232         Deoptimization::print_object(vk, obj, realloc_failures);
 233       }
 234     }
 235 #endif
 236   }
 237   if (save_oop_result || vk != NULL) {
 238     // Restore result.
 239     assert(return_oops.length() == 1, "no value type");
 240     deoptee.set_saved_oop_result(&map, return_oops.pop()());
 241   }
 242   return realloc_failures;
 243 }

 245 static void eliminate_locks(JavaThread* thread, GrowableArray<compiledVFrame*>* chunk, bool realloc_failures) {
 246 #ifndef PRODUCT
 247   bool first = true;
 248 #endif
 249   for (int i = 0; i < chunk->length(); i++) {
 250     compiledVFrame* cvf = chunk->at(i);
 251     assert (cvf->scope() != NULL,"expect only compiled java frames");
 252     GrowableArray<MonitorInfo*>* monitors = cvf->monitors();
 253     if (monitors->is_nonempty()) {
 254       Deoptimization::relock_objects(monitors, thread, realloc_failures);
 255 #ifndef PRODUCT
 256       if (PrintDeoptimizationDetails) {
 257         ttyLocker ttyl;
 258         for (int j = 0; j < monitors->length(); j++) {
 259           MonitorInfo* mi = monitors->at(j);
 260           if (mi->eliminated()) {
 261             if (first) {
 262               first = false;
 263               tty->print_cr("RELOCK OBJECTS in thread " INTPTR_FORMAT, p2i(thread));
 264             }
 265             if (mi->owner_is_scalar_replaced()) {
 266               Klass* k = java_lang_Class::as_Klass(mi->owner_klass());
 267               tty->print_cr("     failed reallocation for klass %s", k->external_name());
 268             } else {
 269               tty->print_cr("     object <" INTPTR_FORMAT "> locked", p2i(mi->owner()));
 270             }
 271           }
 272         }
 273       }
 274 #endif // !PRODUCT
 275     }
 276   }
 277 }

 280 // This is factored, since it is both called from a JRT_LEAF (deoptimization) and a JRT_ENTRY (uncommon_trap)
 281 Deoptimization::UnrollBlock* Deoptimization::fetch_unroll_info_helper(JavaThread* thread, int exec_mode) {
 282 
 283   // Note: there is a safepoint safety issue here. No matter whether we enter
 284   // via vanilla deopt or uncommon trap we MUST NOT stop at a safepoint once
 285   // the vframeArray is created.
 286   //
 287 
 288   // Allocate our special deoptimization ResourceMark
 289   DeoptResourceMark* dmark = new DeoptResourceMark(thread);
 290   assert(thread->deopt_mark() == NULL, "Pending deopt!");
 291   thread->set_deopt_mark(dmark);
 292 
 293   frame stub_frame = thread->last_frame(); // Makes stack walkable as side effect
 294   RegisterMap map(thread, true);
 295   RegisterMap dummy_map(thread, false);
 296   // Now get the deoptee with a valid map
 297   frame deoptee = stub_frame.sender(&map);
 298   // Set the deoptee nmethod
 299   assert(thread->deopt_compiled_method() == NULL, "Pending deopt!");
 300   CompiledMethod* cm = deoptee.cb()->as_compiled_method_or_null();
 301   thread->set_deopt_compiled_method(cm);
 302 
 303   if (VerifyStack) {
 304     thread->validate_frame_layout();
 305   }
 306 
 307   // Create a growable array of VFrames where each VFrame represents an inlined
 308   // Java frame.  This storage is allocated with the usual system arena.
 309   assert(deoptee.is_compiled_frame(), "Wrong frame type");
 310   GrowableArray<compiledVFrame*>* chunk = new GrowableArray<compiledVFrame*>(10);
 311   vframe* vf = vframe::new_vframe(&deoptee, &map, thread);
 312   while (!vf->is_top()) {
 313     assert(vf->is_compiled_frame(), "Wrong frame type");
 314     chunk->push(compiledVFrame::cast(vf));
 315     vf = vf->sender();
 316   }
 317   assert(vf->is_compiled_frame(), "Wrong frame type");
 318   chunk->push(compiledVFrame::cast(vf));
 319 
 320   bool realloc_failures = false;
 321 
 322 #if COMPILER2_OR_JVMCI
 323 #if INCLUDE_JVMCI
 324   bool jvmci_enabled = true;
 325 #else
 326   bool jvmci_enabled = false;
 327 #endif
 328 
 329   // Reallocate the non-escaping objects and restore their fields. Then
 330   // relock objects if synchronization on them was eliminated.
 331   if (jvmci_enabled COMPILER2_PRESENT( || (DoEscapeAnalysis && EliminateAllocations) )) {
 332     realloc_failures = eliminate_allocations(thread, exec_mode, cm, deoptee, map, chunk);
 333   }
 334 #endif // COMPILER2_OR_JVMCI
 335 
 336   // Revoke biases, done with in java state.
 337   // No safepoints allowed after this
 338   revoke_from_deopt_handler(thread, deoptee, &map);
 339 
 340   // Ensure that no safepoint is taken after pointers have been stored
 341   // in fields of rematerialized objects.  If a safepoint occurs from here on
 342   // out the java state residing in the vframeArray will be missed.
 343   // Locks may be rebaised in a safepoint.
 344   NoSafepointVerifier no_safepoint;
 345 
 346 #if COMPILER2_OR_JVMCI
 347   if (jvmci_enabled COMPILER2_PRESENT( || ((DoEscapeAnalysis || EliminateNestedLocks) && EliminateLocks) )) {
 348     eliminate_locks(thread, chunk, realloc_failures);
 349   }
 350 #endif // COMPILER2_OR_JVMCI
 351 
 352   ScopeDesc* trap_scope = chunk->at(0)->scope();
 353   Handle exceptionObject;
 354   if (trap_scope->rethrow_exception()) {
 355     if (PrintDeoptimizationDetails) {
 356       tty->print_cr("Exception to be rethrown in the interpreter for method %s::%s at bci %d", trap_scope->method()->method_holder()->name()->as_C_string(), trap_scope->method()->name()->as_C_string(), trap_scope->bci());
 357     }
 358     GrowableArray<ScopeValue*>* expressions = trap_scope->expressions();
 359     guarantee(expressions != NULL && expressions->length() > 0, "must have exception to throw");
 360     ScopeValue* topOfStack = expressions->top();
 361     exceptionObject = StackValue::create_stack_value(&deoptee, &map, topOfStack)->get_obj();
 362     guarantee(exceptionObject() != NULL, "exception oop can not be null");
 363   }
 364 
 365   vframeArray* array = create_vframeArray(thread, deoptee, &map, chunk, realloc_failures);
 366 #if COMPILER2_OR_JVMCI
 367   if (realloc_failures) {
 368     pop_frames_failed_reallocs(thread, array);
 369   }
 370 #endif
 371 
 372   assert(thread->vframe_array_head() == NULL, "Pending deopt!");
 373   thread->set_vframe_array_head(array);
 374 
 375   // Now that the vframeArray has been created if we have any deferred local writes
 376   // added by jvmti then we can free up that structure as the data is now in the
 377   // vframeArray
 378 
 379   if (thread->deferred_locals() != NULL) {
 380     GrowableArray<jvmtiDeferredLocalVariableSet*>* list = thread->deferred_locals();
 381     int i = 0;
 382     do {
 383       // Because of inlining we could have multiple vframes for a single frame
 384       // and several of the vframes could have deferred writes. Find them all.
 385       if (list->at(i)->id() == array->original().id()) {
 386         jvmtiDeferredLocalVariableSet* dlv = list->at(i);
 387         list->remove_at(i);
 388         // individual jvmtiDeferredLocalVariableSet are CHeapObj's
 389         delete dlv;
 390       } else {
 391         i++;
 392       }
 393     } while ( i < list->length() );
 394     if (list->length() == 0) {
 395       thread->set_deferred_locals(NULL);
 396       // free the list and elements back to C heap.
 397       delete list;
 398     }
 399 
 400   }
 401 
 402   // Compute the caller frame based on the sender sp of stub_frame and stored frame sizes info.
 403   CodeBlob* cb = stub_frame.cb();
 404   // Verify we have the right vframeArray
 405   assert(cb->frame_size() >= 0, "Unexpected frame size");
 406   intptr_t* unpack_sp = stub_frame.sp() + cb->frame_size();
 407 
 408   // If the deopt call site is a MethodHandle invoke call site we have
 409   // to adjust the unpack_sp.
 410   nmethod* deoptee_nm = deoptee.cb()->as_nmethod_or_null();
 411   if (deoptee_nm != NULL && deoptee_nm->is_method_handle_return(deoptee.pc()))
 412     unpack_sp = deoptee.unextended_sp();
 413 
 414 #ifdef ASSERT
 415   assert(cb->is_deoptimization_stub() ||
 416          cb->is_uncommon_trap_stub() ||
 417          strcmp("Stub<DeoptimizationStub.deoptimizationHandler>", cb->name()) == 0 ||
 418          strcmp("Stub<UncommonTrapStub.uncommonTrapHandler>", cb->name()) == 0,
 419          "unexpected code blob: %s", cb->name());
 420 #endif
 421 
 422   // This is a guarantee instead of an assert because if vframe doesn't match
 423   // we will unpack the wrong deoptimized frame and wind up in strange places
 424   // where it will be very difficult to figure out what went wrong. Better
 425   // to die an early death here than some very obscure death later when the
 426   // trail is cold.
 427   // Note: on ia64 this guarantee can be fooled by frames with no memory stack
 428   // in that it will fail to detect a problem when there is one. This needs
 429   // more work in tiger timeframe.
 430   guarantee(array->unextended_sp() == unpack_sp, "vframe_array_head must contain the vframeArray to unpack");
 431 
 432   int number_of_frames = array->frames();
 433 
 434   // Compute the vframes' sizes.  Note that frame_sizes[] entries are ordered from outermost to innermost
 435   // virtual activation, which is the reverse of the elements in the vframes array.
 436   intptr_t* frame_sizes = NEW_C_HEAP_ARRAY(intptr_t, number_of_frames, mtCompiler);
 437   // +1 because we always have an interpreter return address for the final slot.
 438   address* frame_pcs = NEW_C_HEAP_ARRAY(address, number_of_frames + 1, mtCompiler);
 439   int popframe_extra_args = 0;
 440   // Create an interpreter return address for the stub to use as its return
 441   // address so the skeletal frames are perfectly walkable
 442   frame_pcs[number_of_frames] = Interpreter::deopt_entry(vtos, 0);
 443 
 444   // PopFrame requires that the preserved incoming arguments from the recently-popped topmost
 445   // activation be put back on the expression stack of the caller for reexecution
 446   if (JvmtiExport::can_pop_frame() && thread->popframe_forcing_deopt_reexecution()) {
 447     popframe_extra_args = in_words(thread->popframe_preserved_args_size_in_words());
 448   }
 449 
 450   // Find the current pc for sender of the deoptee. Since the sender may have been deoptimized
 451   // itself since the deoptee vframeArray was created we must get a fresh value of the pc rather
 452   // than simply use array->sender.pc(). This requires us to walk the current set of frames
 453   //
 454   frame deopt_sender = stub_frame.sender(&dummy_map); // First is the deoptee frame
 455   deopt_sender = deopt_sender.sender(&dummy_map);     // Now deoptee caller
 456 
 457   // It's possible that the number of parameters at the call site is
 458   // different than number of arguments in the callee when method
 459   // handles are used.  If the caller is interpreted get the real
 460   // value so that the proper amount of space can be added to it's
 461   // frame.
 462   bool caller_was_method_handle = false;
 463   if (deopt_sender.is_interpreted_frame()) {
 464     methodHandle method(thread, deopt_sender.interpreter_frame_method());
 465     Bytecode_invoke cur = Bytecode_invoke_check(method, deopt_sender.interpreter_frame_bci());
 466     if (cur.is_invokedynamic() || cur.is_invokehandle()) {
 467       // Method handle invokes may involve fairly arbitrary chains of
 468       // calls so it's impossible to know how much actual space the
 469       // caller has for locals.
 470       caller_was_method_handle = true;
 471     }
 472   }
 473 
 474   //
 475   // frame_sizes/frame_pcs[0] oldest frame (int or c2i)
 476   // frame_sizes/frame_pcs[1] next oldest frame (int)
 477   // frame_sizes/frame_pcs[n] youngest frame (int)
 478   //
 479   // Now a pc in frame_pcs is actually the return address to the frame's caller (a frame
 480   // owns the space for the return address to it's caller).  Confusing ain't it.
 481   //
 482   // The vframe array can address vframes with indices running from
 483   // 0.._frames-1. Index  0 is the youngest frame and _frame - 1 is the oldest (root) frame.
 484   // When we create the skeletal frames we need the oldest frame to be in the zero slot
 485   // in the frame_sizes/frame_pcs so the assembly code can do a trivial walk.
 486   // so things look a little strange in this loop.
 487   //
 488   int callee_parameters = 0;
 489   int callee_locals = 0;
 490   for (int index = 0; index < array->frames(); index++ ) {
 491     // frame[number_of_frames - 1 ] = on_stack_size(youngest)
 492     // frame[number_of_frames - 2 ] = on_stack_size(sender(youngest))
 493     // frame[number_of_frames - 3 ] = on_stack_size(sender(sender(youngest)))
 494     frame_sizes[number_of_frames - 1 - index] = BytesPerWord * array->element(index)->on_stack_size(callee_parameters,
 495                                                                                                     callee_locals,
 496                                                                                                     index == 0,
 497                                                                                                     popframe_extra_args);
 498     // This pc doesn't have to be perfect just good enough to identify the frame
 499     // as interpreted so the skeleton frame will be walkable
 500     // The correct pc will be set when the skeleton frame is completely filled out
 501     // The final pc we store in the loop is wrong and will be overwritten below
 502     frame_pcs[number_of_frames - 1 - index ] = Interpreter::deopt_entry(vtos, 0) - frame::pc_return_offset;
 503 
 504     callee_parameters = array->element(index)->method()->size_of_parameters();
 505     callee_locals = array->element(index)->method()->max_locals();
 506     popframe_extra_args = 0;
 507   }
 508 
 509   // Compute whether the root vframe returns a float or double value.
 510   BasicType return_type;
 511   {
 512     methodHandle method(thread, array->element(0)->method());
 513     Bytecode_invoke invoke = Bytecode_invoke_check(method, array->element(0)->bci());
 514     return_type = invoke.is_valid() ? invoke.result_type() : T_ILLEGAL;
 515   }
 516 
 517   // Compute information for handling adapters and adjusting the frame size of the caller.
 518   int caller_adjustment = 0;
 519 
 520   // Compute the amount the oldest interpreter frame will have to adjust
 521   // its caller's stack by. If the caller is a compiled frame then
 522   // we pretend that the callee has no parameters so that the
 523   // extension counts for the full amount of locals and not just
 524   // locals-parms. This is because without a c2i adapter the parm
 525   // area as created by the compiled frame will not be usable by
 526   // the interpreter. (Depending on the calling convention there
 527   // may not even be enough space).
 528 
 529   // QQQ I'd rather see this pushed down into last_frame_adjust
 530   // and have it take the sender (aka caller).
 531 
 532   if (deopt_sender.is_compiled_frame() || caller_was_method_handle) {
 533     caller_adjustment = last_frame_adjust(0, callee_locals);
 534   } else if (callee_locals > callee_parameters) {
 535     // The caller frame may need extending to accommodate
 536     // non-parameter locals of the first unpacked interpreted frame.
 537     // Compute that adjustment.
 538     caller_adjustment = last_frame_adjust(callee_parameters, callee_locals);
 539   }
 540 
 541   // If the sender is deoptimized we must retrieve the address of the handler
 542   // since the frame will "magically" show the original pc before the deopt
 543   // and we'd undo the deopt.
 544 
 545   frame_pcs[0] = deopt_sender.raw_pc();
 546 
 547   assert(CodeCache::find_blob_unsafe(frame_pcs[0]) != NULL, "bad pc");
 548 
 549 #if INCLUDE_JVMCI
 550   if (exceptionObject() != NULL) {
 551     thread->set_exception_oop(exceptionObject());
 552     exec_mode = Unpack_exception;
 553   }
 554 #endif
 555 
 556   if (thread->frames_to_pop_failed_realloc() > 0 && exec_mode != Unpack_uncommon_trap) {
 557     assert(thread->has_pending_exception(), "should have thrown OOME");
 558     thread->set_exception_oop(thread->pending_exception());
 559     thread->clear_pending_exception();
 560     exec_mode = Unpack_exception;
 561   }
 562 
 563 #if INCLUDE_JVMCI
 564   if (thread->frames_to_pop_failed_realloc() > 0) {
 565     thread->set_pending_monitorenter(false);
 566   }
 567 #endif
 568 
 569   UnrollBlock* info = new UnrollBlock(array->frame_size() * BytesPerWord,
 570                                       caller_adjustment * BytesPerWord,
 571                                       caller_was_method_handle ? 0 : callee_parameters,
 572                                       number_of_frames,
 573                                       frame_sizes,
 574                                       frame_pcs,
 575                                       return_type,
 576                                       exec_mode);
 577   // On some platforms, we need a way to pass some platform dependent
 578   // information to the unpacking code so the skeletal frames come out
 579   // correct (initial fp value, unextended sp, ...)
 580   info->set_initial_info((intptr_t) array->sender().initial_deoptimization_info());
 581 
 582   if (array->frames() > 1) {
 583     if (VerifyStack && TraceDeoptimization) {
 584       ttyLocker ttyl;
 585       tty->print_cr("Deoptimizing method containing inlining");
 586     }
 587   }
 588 
 589   array->set_unroll_block(info);
 590   return info;
 591 }

 594 // after unpacking to stack and when stack overflow error occurs
 595 void Deoptimization::cleanup_deopt_info(JavaThread *thread,
 596                                         vframeArray *array) {
 597 
 598   // Get array if coming from exception
 599   if (array == NULL) {
 600     array = thread->vframe_array_head();
 601   }
 602   thread->set_vframe_array_head(NULL);
 603 
 604   // Free the previous UnrollBlock
 605   vframeArray* old_array = thread->vframe_array_last();
 606   thread->set_vframe_array_last(array);
 607 
 608   if (old_array != NULL) {
 609     UnrollBlock* old_info = old_array->unroll_block();
 610     old_array->set_unroll_block(NULL);
 611     delete old_info;
 612     delete old_array;
 613   }
 614 
 615   // Deallocate any resource creating in this routine and any ResourceObjs allocated
 616   // inside the vframeArray (StackValueCollections)
 617

 619   thread->set_deopt_mark(NULL);
 620   thread->set_deopt_compiled_method(NULL);
 621 
 622 
 623   if (JvmtiExport::can_pop_frame()) {
 624 #ifndef CC_INTERP
 625     // Regardless of whether we entered this routine with the pending
 626     // popframe condition bit set, we should always clear it now
 627     thread->clear_popframe_condition();

 638   // so that an asynchronous stack walker can work again. This counter is
 639   // incremented at the beginning of fetch_unroll_info() and (in C2) at
 640   // the beginning of uncommon_trap().
 641   thread->dec_in_deopt_handler();

 644 // Moved from cpu directories because none of the cpus has callee save values.
 645 // If a cpu implements callee save values, move this to deoptimization_<cpu>.cpp.
 646 void Deoptimization::unwind_callee_save_values(frame* f, vframeArray* vframe_array) {
 647 
 648   // This code is sort of the equivalent of C2IAdapter::setup_stack_frame back in
 649   // the days we had adapter frames. When we deoptimize a situation where a
 650   // compiled caller calls a compiled caller will have registers it expects
 651   // to survive the call to the callee. If we deoptimize the callee the only
 652   // way we can restore these registers is to have the oldest interpreter
 653   // frame that we create restore these values. That is what this routine
 654   // will accomplish.
 655 
 656   // At the moment we have modified c2 to not have any callee save registers
 657   // so this problem does not exist and this routine is just a place holder.
 658 
 659   assert(f->is_interpreted_frame(), "must be interpreted");
 660 }
 661 
 662 // Return BasicType of value being returned
 663 JRT_LEAF(BasicType, Deoptimization::unpack_frames(JavaThread* thread, int exec_mode))
 664 
 665   // We are already active in the special DeoptResourceMark any ResourceObj's we
 666   // allocate will be freed at the end of the routine.
 667 
 668   // It is actually ok to allocate handles in a leaf method. It causes no safepoints,
 669   // but makes the entry a little slower. There is however a little dance we have to
 670   // do in debug mode to get around the NoHandleMark code in the JRT_LEAF macro
 671   ResetNoHandleMark rnhm; // No-op in release/product versions
 672   HandleMark hm;
 673 
 674   frame stub_frame = thread->last_frame();
 675 
 676   // Since the frame to unpack is the top frame of this thread, the vframe_array_head
 677   // must point to the vframeArray for the unpack frame.
 678   vframeArray* array = thread->vframe_array_head();
 679 
 680 #ifndef PRODUCT
 681   if (TraceDeoptimization) {
 682     ttyLocker ttyl;
 683     tty->print_cr("DEOPT UNPACKING thread " INTPTR_FORMAT " vframeArray " INTPTR_FORMAT " mode %d",
 684                   p2i(thread), p2i(array), exec_mode);
 685   }
 686 #endif
 687   Events::log_deopt_message(thread, "DEOPT UNPACKING pc=" INTPTR_FORMAT " sp=" INTPTR_FORMAT " mode %d",
 688               p2i(stub_frame.pc()), p2i(stub_frame.sp()), exec_mode);
 689 
 690   UnrollBlock* info = array->unroll_block();
 691 
 692   // Unpack the interpreter frames and any adapter frame (c2 only) we might create.
 693   array->unpack_to_stack(stub_frame, exec_mode, info->caller_actual_parameters());
 694 
 695   BasicType bt = info->return_type();
 696 
 697   // If we have an exception pending, claim that the return type is an oop
 698   // so the deopt_blob does not overwrite the exception_oop.
 699 
 700   if (exec_mode == Unpack_exception)
 701     bt = T_OBJECT;
 702 
 703   // Cleanup thread deopt data
 704   cleanup_deopt_info(thread, array);
 705 
 706 #ifndef PRODUCT
 707   if (VerifyStack) {
 708     ResourceMark res_mark;
 709     // Clear pending exception to not break verification code (restored afterwards)
 710     PRESERVE_EXCEPTION_MARK;
 711 
 712     thread->validate_frame_layout();
 713 
 714     // Verify that the just-unpacked frames match the interpreter's
 715     // notions of expression stack and locals
 716     vframeArray* cur_array = thread->vframe_array_last();
 717     RegisterMap rm(thread, false);
 718     rm.set_include_argument_oops(false);
 719     bool is_top_frame = true;
 720     int callee_size_of_parameters = 0;
 721     int callee_max_locals = 0;
 722     for (int i = 0; i < cur_array->frames(); i++) {
 723       vframeArrayElement* el = cur_array->element(i);
 724       frame* iframe = el->iframe();
 725       guarantee(iframe->is_interpreted_frame(), "Wrong frame type");
 726 
 727       // Get the oop map for this bci
 728       InterpreterOopMap mask;
 729       int cur_invoke_parameter_size = 0;
 730       bool try_next_mask = false;
 731       int next_mask_expression_stack_size = -1;
 732       int top_frame_expression_stack_adjustment = 0;
 733       methodHandle mh(thread, iframe->interpreter_frame_method());
 734       OopMapCache::compute_one_oop_map(mh, iframe->interpreter_frame_bci(), &mask);
 735       BytecodeStream str(mh, iframe->interpreter_frame_bci());
 736       int max_bci = mh->code_size();
 737       // Get to the next bytecode if possible
 738       assert(str.bci() < max_bci, "bci in interpreter frame out of bounds");
 739       // Check to see if we can grab the number of outgoing arguments
 740       // at an uncommon trap for an invoke (where the compiler
 741       // generates debug info before the invoke has executed)
 742       Bytecodes::Code cur_code = str.next();
 743       if (Bytecodes::is_invoke(cur_code)) {
 744         Bytecode_invoke invoke(mh, iframe->interpreter_frame_bci());
 745         cur_invoke_parameter_size = invoke.size_of_parameters();
 746         if (i != 0 && !invoke.is_invokedynamic() && MethodHandles::has_member_arg(invoke.klass(), invoke.name())) {
 747           callee_size_of_parameters++;
 748         }
 749       }
 750       if (str.bci() < max_bci) {
 751         Bytecodes::Code next_code = str.next();
 752         if (next_code >= 0) {
 753           // The interpreter oop map generator reports results before
 754           // the current bytecode has executed except in the case of
 755           // calls. It seems to be hard to tell whether the compiler
 756           // has emitted debug information matching the "state before"
 757           // a given bytecode or the state after, so we try both
 758           if (!Bytecodes::is_invoke(cur_code) && cur_code != Bytecodes::_athrow) {
 759             // Get expression stack size for the next bytecode
 760             InterpreterOopMap next_mask;
 761             OopMapCache::compute_one_oop_map(mh, str.bci(), &next_mask);
 762             next_mask_expression_stack_size = next_mask.expression_stack_size();
 763             if (Bytecodes::is_invoke(next_code)) {
 764               Bytecode_invoke invoke(mh, str.bci());
 765               next_mask_expression_stack_size += invoke.size_of_parameters();
 766             }
 767             // Need to subtract off the size of the result type of
 768             // the bytecode because this is not described in the
 769             // debug info but returned to the interpreter in the TOS
 770             // caching register
 771             BasicType bytecode_result_type = Bytecodes::result_type(cur_code);
 772             if (bytecode_result_type != T_ILLEGAL) {
 773               top_frame_expression_stack_adjustment = type2size[bytecode_result_type];
 774             }
 775             assert(top_frame_expression_stack_adjustment >= 0, "stack adjustment must be positive");
 776             try_next_mask = true;
 777           }
 778         }
 779       }
 780 
 781       // Verify stack depth and oops in frame
 782       // This assertion may be dependent on the platform we're running on and may need modification (tested on x86 and sparc)
 783       if (!(
 784             /* SPARC */
 785             (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + callee_size_of_parameters) ||
 786             /* x86 */
 787             (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + callee_max_locals) ||
 788             (try_next_mask &&
 789              (iframe->interpreter_frame_expression_stack_size() == (next_mask_expression_stack_size -
 790                                                                     top_frame_expression_stack_adjustment))) ||
 791             (is_top_frame && (exec_mode == Unpack_exception) && iframe->interpreter_frame_expression_stack_size() == 0) ||
 792             (is_top_frame && (exec_mode == Unpack_uncommon_trap || exec_mode == Unpack_reexecute || el->should_reexecute()) &&
 793              (iframe->interpreter_frame_expression_stack_size() == mask.expression_stack_size() + cur_invoke_parameter_size))
 794             )) {
 795         {
 796           ttyLocker ttyl;
 797 
 798           // Print out some information that will help us debug the problem
 799           tty->print_cr("Wrong number of expression stack elements during deoptimization");
 800           tty->print_cr("  Error occurred while verifying frame %d (0..%d, 0 is topmost)", i, cur_array->frames() - 1);
 801           tty->print_cr("  Fabricated interpreter frame had %d expression stack elements",
 802                         iframe->interpreter_frame_expression_stack_size());
 803           tty->print_cr("  Interpreter oop map had %d expression stack elements", mask.expression_stack_size());
 804           tty->print_cr("  try_next_mask = %d", try_next_mask);
 805           tty->print_cr("  next_mask_expression_stack_size = %d", next_mask_expression_stack_size);
 806           tty->print_cr("  callee_size_of_parameters = %d", callee_size_of_parameters);
 807           tty->print_cr("  callee_max_locals = %d", callee_max_locals);
 808           tty->print_cr("  top_frame_expression_stack_adjustment = %d", top_frame_expression_stack_adjustment);
 809           tty->print_cr("  exec_mode = %d", exec_mode);
 810           tty->print_cr("  cur_invoke_parameter_size = %d", cur_invoke_parameter_size);
 811           tty->print_cr("  Thread = " INTPTR_FORMAT ", thread ID = %d", p2i(thread), thread->osthread()->thread_id());
 812           tty->print_cr("  Interpreted frames:");
 813           for (int k = 0; k < cur_array->frames(); k++) {
 814             vframeArrayElement* el = cur_array->element(k);
 815             tty->print_cr("    %s (bci %d)", el->method()->name_and_sig_as_C_string(), el->bci());
 816           }
 817           cur_array->print_on_2(tty);
 818         } // release tty lock before calling guarantee
 819         guarantee(false, "wrong number of expression stack elements during deopt");
 820       }
 821       VerifyOopClosure verify;
 822       iframe->oops_interpreted_do(&verify, &rm, false);
 823       callee_size_of_parameters = mh->size_of_parameters();
 824       callee_max_locals = mh->max_locals();
 825       is_top_frame = false;
 826     }
 827   }
 828 #endif /* !PRODUCT */
 829 
 830 
 831   return bt;
 832 JRT_END

 923       return lookup(value);
 924     }
 925     PrimitiveType value = (PrimitiveType)*((jint*)&raw_value);
 926     return lookup(value);
 927   }

1001 bool Deoptimization::realloc_objects(JavaThread* thread, frame* fr, RegisterMap* reg_map, GrowableArray<ScopeValue*>* objects, TRAPS) {
1002   Handle pending_exception(THREAD, thread->pending_exception());
1003   const char* exception_file = thread->exception_file();
1004   int exception_line = thread->exception_line();
1005   thread->clear_pending_exception();
1006

1008

1010     assert(objects->at(i)->is_object(), "invalid debug information");
1011     ObjectValue* sv = (ObjectValue*) objects->at(i);
1012 
1013     Klass* k = java_lang_Class::as_Klass(sv->klass()->as_ConstantOopReadValue()->value()());
1014     oop obj = NULL;
1015 
1016     if (k->is_instance_klass()) {
1017 #if INCLUDE_JVMCI || INCLUDE_AOT
1018       CompiledMethod* cm = fr->cb()->as_compiled_method_or_null();
1019       if (cm->is_compiled_by_jvmci() && sv->is_auto_box()) {
1020         AutoBoxObjectValue* abv = (AutoBoxObjectValue*) sv;
1021         obj = get_cached_box(abv, fr, reg_map, THREAD);
1022         if (obj != NULL) {
1023           // Set the flag to indicate the box came from a cache, so that we can skip the field reassignment for it.
1024           abv->set_cached(true);
1025         }
1026       }
1027 #endif // INCLUDE_JVMCI || INCLUDE_AOT
1028       InstanceKlass* ik = InstanceKlass::cast(k);
1029       if (obj == NULL) {
1030         obj = ik->allocate_instance(THREAD);
1031       }
1032     } else if (k->is_valueArray_klass()) {
1033       ValueArrayKlass* ak = ValueArrayKlass::cast(k);
1034       // Value type array must be zeroed because not all memory is reassigned
1035       obj = ak->allocate(sv->field_size(), THREAD);
1036     } else if (k->is_typeArray_klass()) {

1059     thread->set_pending_exception(pending_exception(), exception_file, exception_line);
1060   }
1061 
1062   return failures;

1081 /**
1082  * For primitive types whose kind gets "erased" at runtime (shorts become stack ints),
1083  * we need to somehow be able to recover the actual kind to be able to write the correct
1084  * amount of bytes.
1085  * For that purpose, this method assumes that, for an entry spanning n bytes at index i,
1086  * the entries at index n + 1 to n + i are 'markers'.
1087  * For example, if we were writing a short at index 4 of a byte array of size 8, the
1088  * expected form of the array would be:
1089  *
1090  * {b0, b1, b2, b3, INT, marker, b6, b7}
1091  *
1092  * Thus, in order to get back the size of the entry, we simply need to count the number
1093  * of marked entries
1094  *
1095  * @param virtualArray the virtualized byte array
1096  * @param i index of the virtual entry we are recovering
1097  * @return The number of bytes the entry spans
1098  */
1099 static int count_number_of_bytes_for_entry(ObjectValue *virtualArray, int i) {
1100   int index = i;
1101   while (++index < virtualArray->field_size() &&
1102            virtualArray->field_at(index)->is_marker()) {}
1103   return index - i;
1104 }

1106 /**
1107  * If there was a guarantee for byte array to always start aligned to a long, we could
1108  * do a simple check on the parity of the index. Unfortunately, that is not always the
1109  * case. Thus, we check alignment of the actual address we are writing to.
1110  * In the unlikely case index 0 is 5-aligned for example, it would then be possible to
1111  * write a long to index 3.
1112  */

1114     jbyte* res = obj->byte_at_addr(index);
1115     assert((((intptr_t) res) % expected_alignment) == 0, "Non-aligned write");

1119 static void byte_array_put(typeArrayOop obj, intptr_t val, int index, int byte_count) {
1120   switch (byte_count) {
1121     case 1:
1122       obj->byte_at_put(index, (jbyte) *((jint *) &val));
1123       break;
1124     case 2:
1125       *((jshort *) check_alignment_get_addr(obj, index, 2)) = (jshort) *((jint *) &val);
1126       break;
1127     case 4:
1128       *((jint *) check_alignment_get_addr(obj, index, 4)) = (jint) *((jint *) &val);
1129       break;
1130     case 8: {

1132         jlong res = (jlong) *((jlong *) &val);

1140 #else
1141       jlong res = (jlong) *((jlong *) &val);
1142 #endif // SPARC
1143 #endif
1144       *((jlong *) check_alignment_get_addr(obj, index, 8)) = res;
1145       break;
1146   }

1155 void Deoptimization::reassign_type_array_elements(frame* fr, RegisterMap* reg_map, ObjectValue* sv, typeArrayOop obj, BasicType type) {

1161     switch(type) {

1163       assert(value->type() == T_INT, "Agreement.");

1171       jlong res = jlong_from((jint)low->get_int(), (jint)value->get_int());

1174 #endif //SPARC
1175 #endif
1176       obj->long_at_put(index, res);
1177       break;
1178     }
1179 
1180     // Have to cast to INT (32 bits) pointer to avoid little/big-endian problem.
1181     case T_INT: case T_FLOAT: { // 4 bytes.
1182       assert(value->type() == T_INT, "Agreement.");
1183       bool big_value = false;
1184       if (i + 1 < sv->field_size() && type == T_INT) {
1185         if (sv->field_at(i)->is_location()) {
1186           Location::Type type = ((LocationValue*) sv->field_at(i))->location().type();
1187           if (type == Location::dbl || type == Location::lng) {
1188             big_value = true;
1189           }
1190         } else if (sv->field_at(i)->is_constant_int()) {
1191           ScopeValue* next_scope_field = sv->field_at(i + 1);
1192           if (next_scope_field->is_constant_long() || next_scope_field->is_constant_double()) {
1193             big_value = true;

1196       }
1197 
1198       if (big_value) {
1199         StackValue* low = StackValue::create_stack_value(fr, reg_map, sv->field_at(++i));
1200   #ifdef _LP64
1201         jlong res = (jlong)low->get_int();
1202   #else
1203   #ifdef SPARC
1204         // For SPARC we have to swap high and low words.
1205         jlong res = jlong_from((jint)low->get_int(), (jint)value->get_int());
1206   #else
1207         jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1208   #endif //SPARC
1209   #endif
1210         obj->int_at_put(index, (jint)*((jint*)&res));
1211         obj->int_at_put(++index, (jint)*(((jint*)&res) + 1));
1212       } else {
1213         val = value->get_int();
1214         obj->int_at_put(index, (jint)*((jint*)&val));
1215       }
1216       break;
1217     }
1218 
1219     case T_SHORT:
1220       assert(value->type() == T_INT, "Agreement.");
1221       val = value->get_int();
1222       obj->short_at_put(index, (jshort)*((jint*)&val));

1234       val = value->get_int();
1235 #if INCLUDE_JVMCI
1236       int byte_count = count_number_of_bytes_for_entry(sv, i);

1250       assert(value->type() == T_INT, "Agreement.");

1253       break;
1254     }
1255 
1256       default:
1257         ShouldNotReachHere();
1258     }
1259     index++;
1260   }

1289 // Restore fields of an eliminated instance object using the same field order
1290 // returned by HotSpotResolvedObjectTypeImpl.getInstanceFields(true)
1291 static int reassign_fields_by_klass(InstanceKlass* klass, frame* fr, RegisterMap* reg_map, ObjectValue* sv, int svIndex, oop obj, bool skip_internal, int base_offset, TRAPS) {
1292 
1293   GrowableArray<ReassignedField>* fields = new GrowableArray<ReassignedField>();
1294   InstanceKlass* ik = klass;
1295   while (ik != NULL) {
1296     for (AllFieldStream fs(ik); !fs.done(); fs.next()) {
1297       if (!fs.access_flags().is_static() && (!skip_internal || !fs.access_flags().is_internal())) {
1298         ReassignedField field;
1299         field._offset = fs.offset();
1300         field._type = Signature::basic_type(fs.signature());
1301         if (field._type == T_VALUETYPE) {
1302           field._type = T_OBJECT;
1303         }
1304         if (fs.is_flattened()) {
1305           // Resolve klass of flattened value type field
1306           Klass* vk = klass->get_value_field_klass(fs.index());
1307           field._klass = ValueKlass::cast(vk);
1308           field._type = T_VALUETYPE;
1309         }
1310         fields->append(field);
1311       }
1312     }
1313     ik = ik->superklass();
1314   }
1315   fields->sort(compare);
1316   for (int i = 0; i < fields->length(); i++) {
1317     intptr_t val;
1318     ScopeValue* scope_field = sv->field_at(svIndex);
1319     StackValue* value = StackValue::create_stack_value(fr, reg_map, scope_field);
1320     int offset = base_offset + fields->at(i)._offset;
1321     BasicType type = fields->at(i)._type;
1322     switch (type) {
1323       case T_OBJECT:
1324       case T_ARRAY:
1325         assert(value->type() == T_OBJECT, "Agreement.");
1326         obj->obj_field_put(offset, value->get_obj()());
1327         break;
1328 
1329       case T_VALUETYPE: {
1330         // Recursively re-assign flattened value type fields
1331         InstanceKlass* vk = fields->at(i)._klass;
1332         assert(vk != NULL, "must be resolved");
1333         offset -= ValueKlass::cast(vk)->first_field_offset(); // Adjust offset to omit oop header
1334         svIndex = reassign_fields_by_klass(vk, fr, reg_map, sv, svIndex, obj, skip_internal, offset, CHECK_0);
1335         continue; // Continue because we don't need to increment svIndex
1336       }
1337 
1338       // Have to cast to INT (32 bits) pointer to avoid little/big-endian problem.
1339       case T_INT: case T_FLOAT: { // 4 bytes.
1340         assert(value->type() == T_INT, "Agreement.");
1341         bool big_value = false;
1342         if (i+1 < fields->length() && fields->at(i+1)._type == T_INT) {
1343           if (scope_field->is_location()) {
1344             Location::Type type = ((LocationValue*) scope_field)->location().type();
1345             if (type == Location::dbl || type == Location::lng) {
1346               big_value = true;
1347             }
1348           }
1349           if (scope_field->is_constant_int()) {
1350             ScopeValue* next_scope_field = sv->field_at(svIndex + 1);
1351             if (next_scope_field->is_constant_long() || next_scope_field->is_constant_double()) {
1352               big_value = true;
1353             }
1354           }
1355         }
1356 
1357         if (big_value) {
1358           i++;
1359           assert(i < fields->length(), "second T_INT field needed");
1360           assert(fields->at(i)._type == T_INT, "T_INT field needed");
1361         } else {
1362           val = value->get_int();
1363           obj->int_field_put(offset, (jint)*((jint*)&val));
1364           break;
1365         }
1366       }
1367         /* no break */
1368 
1369       case T_LONG: case T_DOUBLE: {
1370         assert(value->type() == T_INT, "Agreement.");
1371         StackValue* low = StackValue::create_stack_value(fr, reg_map, sv->field_at(++svIndex));
1372 #ifdef _LP64
1373         jlong res = (jlong)low->get_int();
1374 #else
1375 #ifdef SPARC
1376         // For SPARC we have to swap high and low words.
1377         jlong res = jlong_from((jint)low->get_int(), (jint)value->get_int());
1378 #else
1379         jlong res = jlong_from((jint)value->get_int(), (jint)low->get_int());
1380 #endif //SPARC
1381 #endif
1382         obj->long_field_put(offset, res);
1383         break;
1384       }
1385 
1386       case T_SHORT:
1387         assert(value->type() == T_INT, "Agreement.");
1388         val = value->get_int();
1389         obj->short_field_put(offset, (jshort)*((jint*)&val));
1390         break;
1391 
1392       case T_CHAR:
1393         assert(value->type() == T_INT, "Agreement.");
1394         val = value->get_int();
1395         obj->char_field_put(offset, (jchar)*((jint*)&val));
1396         break;
1397 
1398       case T_BYTE:
1399         assert(value->type() == T_INT, "Agreement.");
1400         val = value->get_int();
1401         obj->byte_field_put(offset, (jbyte)*((jint*)&val));
1402         break;
1403 
1404       case T_BOOLEAN:
1405         assert(value->type() == T_INT, "Agreement.");
1406         val = value->get_int();
1407         obj->bool_field_put(offset, (jboolean)*((jint*)&val));
1408         break;
1409 
1410       default:
1411         ShouldNotReachHere();
1412     }
1413     svIndex++;
1414   }
1415   return svIndex;

1488         assert(mon_info->owner()->is_locked(), "object must be locked now");
1489       }

1491   }

1621 static void get_monitors_from_stack(GrowableArray<Handle>* objects_to_revoke, JavaThread* thread, frame fr, RegisterMap* map) {
1622   // Unfortunately we don't have a RegisterMap available in most of
1623   // the places we want to call this routine so we need to walk the
1624   // stack again to update the register map.
1625   if (map == NULL || !map->update_map()) {
1626     StackFrameStream sfs(thread, true);
1627     bool found = false;
1628     while (!found && !sfs.is_done()) {
1629       frame* cur = sfs.current();
1630       sfs.next();
1631       found = cur->id() == fr.id();
1632     }
1633     assert(found, "frame to be deoptimized not found on target thread's stack");
1634     map = sfs.register_map();
1635   }
1636 
1637   vframe* vf = vframe::new_vframe(&fr, map, thread);
1638   compiledVFrame* cvf = compiledVFrame::cast(vf);
1639   // Revoke monitors' biases in all scopes
1640   while (!cvf->is_top()) {
1641     collect_monitors(cvf, objects_to_revoke);
1642     cvf = compiledVFrame::cast(cvf->sender());
1643   }
1644   collect_monitors(cvf, objects_to_revoke);
1645 }

2014         // Dump the relevant MDO state.
2015         // This is the deopt count for the current reason, any previous
2016         // reasons or recompiles seen at this point.
2017         int dcnt = trap_mdo->trap_count(reason);
2018         if (dcnt != 0)
2019           xtty->print(" count='%d'", dcnt);
2020         ProfileData* pdata = trap_mdo->bci_to_data(trap_bci);

2033         xtty->end_head();
2034       }

2036         tty->print("Uncommon trap occurred in");
2037         nm->method()->print_short_name(tty);
2038         tty->print(" compiler=%s compile_id=%d", nm->compiler_name(), nm->compile_id());
2039 #if INCLUDE_JVMCI

2049                    os::current_thread_id(),
2050                    trap_reason_name(reason),
2051                    trap_action_name(action),
2052                    unloaded_class_index
2053 #if INCLUDE_JVMCI
2054                    , debug_id
2055 #endif
2056                    );

2058           tty->print(unresolved ? " unresolved class: " : " symbol: ");
2059           class_name->print_symbol_on(tty);
2060         }
2061         tty->cr();
2062       }
2063       if (xtty != NULL) {
2064         // Log the precise location of the trap.
2065         for (ScopeDesc* sd = trap_scope; ; sd = sd->sender()) {
2066           xtty->begin_elem("jvms bci='%d'", sd->bci());
2067           xtty->method(sd->method());
2068           xtty->end_elem();
2069           if (sd->is_top())  break;
2070         }
2071         xtty->tail("uncommon_trap");
2072       }
2073     }
2074     // (End diagnostic printout.)
2075 
2076     // Load class if necessary
2077     if (unloaded_class_index >= 0) {
2078       constantPoolHandle constants(THREAD, trap_method->constants());
2079       load_class_by_index(constants, unloaded_class_index);
2080     }
2081 
2082     // Flush the nmethod if necessary and desirable.
2083     //
2084     // We need to avoid situations where we are re-flushing the nmethod
2085     // because of a hot deoptimization site.  Repeated flushes at the same
2086     // point need to be detected by the compiler and avoided.  If the compiler
2087     // cannot avoid them (or has a bug and "refuses" to avoid them), this
2088     // module must take measures to avoid an infinite cycle of recompilation
2089     // and deoptimization.  There are several such measures:
2090     //
2091     //   1. If a recompilation is ordered a second time at some site X
2092     //   and for the same reason R, the action is adjusted to 'reinterpret',
2093     //   to give the interpreter time to exercise the method more thoroughly.
2094     //   If this happens, the method's overflow_recompile_count is incremented.
2095     //
2096     //   2. If the compiler fails to reduce the deoptimization rate, then
2097     //   the method's overflow_recompile_count will begin to exceed the set
2098     //   limit PerBytecodeRecompilationCutoff.  If this happens, the action
2099     //   is adjusted to 'make_not_compilable', and the method is abandoned
2100     //   to the interpreter.  This is a performance hit for hot methods,
2101     //   but is better than a disastrous infinite cycle of recompilations.
2102     //   (Actually, only the method containing the site X is abandoned.)
2103     //
2104     //   3. In parallel with the previous measures, if the total number of
2105     //   recompilations of a method exceeds the much larger set limit
2106     //   PerMethodRecompilationCutoff, the method is abandoned.
2107     //   This should only happen if the method is very large and has
2108     //   many "lukewarm" deoptimizations.  The code which enforces this
2109     //   limit is elsewhere (class nmethod, class Method).
2110     //
2111     // Note that the per-BCI 'is_recompiled' bit gives the compiler one chance
2112     // to recompile at each bytecode independently of the per-BCI cutoff.
2113     //
2114     // The decision to update code is up to the compiler, and is encoded
2115     // in the Action_xxx code.  If the compiler requests Action_none
2116     // no trap state is changed, no compiled code is changed, and the
2117     // computation suffers along in the interpreter.
2118     //
2119     // The other action codes specify various tactics for decompilation
2120     // and recompilation.  Action_maybe_recompile is the loosest, and
2121     // allows the compiled code to stay around until enough traps are seen,
2122     // and until the compiler gets around to recompiling the trapping method.
2123     //
2124     // The other actions cause immediate removal of the present code.
2125 
2126     // Traps caused by injected profile shouldn't pollute trap counts.
2127     bool injected_profile_trap = trap_method->has_injected_profile() &&
2128                                  (reason == Reason_intrinsic || reason == Reason_unreached);
2129 
2130     bool update_trap_state = (reason != Reason_tenured) && !injected_profile_trap;
2131     bool make_not_entrant = false;
2132     bool make_not_compilable = false;
2133     bool reprofile = false;
2134     switch (action) {
2135     case Action_none:
2136       // Keep the old code.
2137       update_trap_state = false;
2138       break;
2139     case Action_maybe_recompile:
2140       // Do not need to invalidate the present code, but we can
2141       // initiate another
2142       // Start compiler without (necessarily) invalidating the nmethod.
2143       // The system will tolerate the old code, but new code should be
2144       // generated when possible.
2145       break;
2146     case Action_reinterpret:
2147       // Go back into the interpreter for a while, and then consider
2148       // recompiling form scratch.
2149       make_not_entrant = true;
2150       // Reset invocation counter for outer most method.
2151       // This will allow the interpreter to exercise the bytecodes
2152       // for a while before recompiling.
2153       // By contrast, Action_make_not_entrant is immediate.
2154       //
2155       // Note that the compiler will track null_check, null_assert,
2156       // range_check, and class_check events and log them as if they
2157       // had been traps taken from compiled code.  This will update
2158       // the MDO trap history so that the next compilation will
2159       // properly detect hot trap sites.
2160       reprofile = true;
2161       break;
2162     case Action_make_not_entrant:
2163       // Request immediate recompilation, and get rid of the old code.
2164       // Make them not entrant, so next time they are called they get
2165       // recompiled.  Unloaded classes are loaded now so recompile before next
2166       // time they are called.  Same for uninitialized.  The interpreter will
2167       // link the missing class, if any.
2168       make_not_entrant = true;
2169       break;
2170     case Action_make_not_compilable:
2171       // Give up on compiling this method at all.
2172       make_not_entrant = true;
2173       make_not_compilable = true;
2174       break;
2175     default:
2176       ShouldNotReachHere();
2177     }
2178 
2179     // Setting +ProfileTraps fixes the following, on all platforms:
2180     // 4852688: ProfileInterpreter is off by default for ia64.  The result is
2181     // infinite heroic-opt-uncommon-trap/deopt/recompile cycles, since the
2182     // recompile relies on a MethodData* to record heroic opt failures.
2183 
2184     // Whether the interpreter is producing MDO data or not, we also need
2185     // to use the MDO to detect hot deoptimization points and control
2186     // aggressive optimization.
2187     bool inc_recompile_count = false;
2188     ProfileData* pdata = NULL;
2189     if (ProfileTraps && !is_client_compilation_mode_vm() && update_trap_state && trap_mdo != NULL) {
2190       assert(trap_mdo == get_method_data(thread, profiled_method, false), "sanity");
2191       uint this_trap_count = 0;
2192       bool maybe_prior_trap = false;
2193       bool maybe_prior_recompile = false;
2194       pdata = query_update_method_data(trap_mdo, trap_bci, reason, true,
2195 #if INCLUDE_JVMCI
2196                                    nm->is_compiled_by_jvmci() && nm->is_osr_method(),
2197 #endif
2198                                    nm->method(),
2199                                    //outputs:
2200                                    this_trap_count,
2201                                    maybe_prior_trap,
2202                                    maybe_prior_recompile);
2203       // Because the interpreter also counts null, div0, range, and class
2204       // checks, these traps from compiled code are double-counted.
2205       // This is harmless; it just means that the PerXTrapLimit values
2206       // are in effect a little smaller than they look.
2207 
2208       DeoptReason per_bc_reason = reason_recorded_per_bytecode_if_any(reason);
2209       if (per_bc_reason != Reason_none) {
2210         // Now take action based on the partially known per-BCI history.
2211         if (maybe_prior_trap
2212             && this_trap_count >= (uint)PerBytecodeTrapLimit) {
2213           // If there are too many traps at this BCI, force a recompile.
2214           // This will allow the compiler to see the limit overflow, and
2215           // take corrective action, if possible.  The compiler generally
2216           // does not use the exact PerBytecodeTrapLimit value, but instead
2217           // changes its tactics if it sees any traps at all.  This provides
2218           // a little hysteresis, delaying a recompile until a trap happens
2219           // several times.
2220           //
2221           // Actually, since there is only one bit of counter per BCI,
2222           // the possible per-BCI counts are {0,1,(per-method count)}.
2223           // This produces accurate results if in fact there is only
2224           // one hot trap site, but begins to get fuzzy if there are
2225           // many sites.  For example, if there are ten sites each
2226           // trapping two or more times, they each get the blame for
2227           // all of their traps.
2228           make_not_entrant = true;
2229         }
2230 
2231         // Detect repeated recompilation at the same BCI, and enforce a limit.
2232         if (make_not_entrant && maybe_prior_recompile) {
2233           // More than one recompile at this point.
2234           inc_recompile_count = maybe_prior_trap;
2235         }
2236       } else {
2237         // For reasons which are not recorded per-bytecode, we simply
2238         // force recompiles unconditionally.
2239         // (Note that PerMethodRecompilationCutoff is enforced elsewhere.)
2240         make_not_entrant = true;
2241       }
2242 
2243       // Go back to the compiler if there are too many traps in this method.
2244       if (this_trap_count >= per_method_trap_limit(reason)) {
2245         // If there are too many traps in this method, force a recompile.
2246         // This will allow the compiler to see the limit overflow, and
2247         // take corrective action, if possible.
2248         // (This condition is an unlikely backstop only, because the
2249         // PerBytecodeTrapLimit is more likely to take effect first,
2250         // if it is applicable.)
2251         make_not_entrant = true;
2252       }
2253 
2254       // Here's more hysteresis:  If there has been a recompile at
2255       // this trap point already, run the method in the interpreter
2256       // for a while to exercise it more thoroughly.
2257       if (make_not_entrant && maybe_prior_recompile && maybe_prior_trap) {
2258         reprofile = true;
2259       }
2260     }
2261 
2262     // Take requested actions on the method:
2263 
2264     // Recompile
2265     if (make_not_entrant) {
2266       if (!nm->make_not_entrant()) {
2267         return; // the call did not change nmethod's state
2268       }
2269 
2270       if (pdata != NULL) {
2271         // Record the recompilation event, if any.
2272         int tstate0 = pdata->trap_state();
2273         int tstate1 = trap_state_set_recompiled(tstate0, true);
2274         if (tstate1 != tstate0)
2275           pdata->set_trap_state(tstate1);
2276       }
2277 
2278 #if INCLUDE_RTM_OPT
2279       // Restart collecting RTM locking abort statistic if the method
2280       // is recompiled for a reason other than RTM state change.
2281       // Assume that in new recompiled code the statistic could be different,
2282       // for example, due to different inlining.
2283       if ((reason != Reason_rtm_state_change) && (trap_mdo != NULL) &&
2284           UseRTMDeopt && (nm->as_nmethod()->rtm_state() != ProfileRTM)) {
2285         trap_mdo->atomic_set_rtm_state(ProfileRTM);
2286       }
2287 #endif
2288       // For code aging we count traps separately here, using make_not_entrant()
2289       // as a guard against simultaneous deopts in multiple threads.
2290       if (reason == Reason_tenured && trap_mdo != NULL) {
2291         trap_mdo->inc_tenure_traps();
2292       }
2293     }
2294 
2295     if (inc_recompile_count) {
2296       trap_mdo->inc_overflow_recompile_count();
2297       if ((uint)trap_mdo->overflow_recompile_count() >
2298           (uint)PerBytecodeRecompilationCutoff) {
2299         // Give up on the method containing the bad BCI.
2300         if (trap_method() == nm->method()) {
2301           make_not_compilable = true;
2302         } else {
2303           trap_method->set_not_compilable("overflow_recompile_count > PerBytecodeRecompilationCutoff", CompLevel_full_optimization);
2304           // But give grace to the enclosing nm->method().
2305         }
2306       }
2307     }
2308 
2309     // Reprofile
2310     if (reprofile) {
2311       CompilationPolicy::policy()->reprofile(trap_scope, nm->is_osr_method());
2312     }
2313 
2314     // Give up compiling
2315     if (make_not_compilable && !nm->method()->is_not_compilable(CompLevel_full_optimization)) {
2316       assert(make_not_entrant, "consistent");
2317       nm->method()->set_not_compilable("give up compiling", CompLevel_full_optimization);
2318     }
2319 
2320   } // Free marked resources
2321 
2322 }

2325 ProfileData*
2326 Deoptimization::query_update_method_data(MethodData* trap_mdo,
2327                                          int trap_bci,
2328                                          Deoptimization::DeoptReason reason,
2329                                          bool update_total_trap_count,
2330 #if INCLUDE_JVMCI
2331                                          bool is_osr,
2332 #endif
2333                                          Method* compiled_method,
2334                                          //outputs:
2335                                          uint& ret_this_trap_count,
2336                                          bool& ret_maybe_prior_trap,
2337                                          bool& ret_maybe_prior_recompile) {
2338   bool maybe_prior_trap = false;
2339   bool maybe_prior_recompile = false;
2340   uint this_trap_count = 0;
2341   if (update_total_trap_count) {
2342     uint idx = reason;
2343 #if INCLUDE_JVMCI
2344     if (is_osr) {
2345       idx += Reason_LIMIT;
2346     }
2347 #endif
2348     uint prior_trap_count = trap_mdo->trap_count(idx);
2349     this_trap_count  = trap_mdo->inc_trap_count(idx);
2350 
2351     // If the runtime cannot find a place to store trap history,
2352     // it is estimated based on the general condition of the method.
2353     // If the method has ever been recompiled, or has ever incurred
2354     // a trap with the present reason , then this BCI is assumed
2355     // (pessimistically) to be the culprit.
2356     maybe_prior_trap      = (prior_trap_count != 0);
2357     maybe_prior_recompile = (trap_mdo->decompile_count() != 0);
2358   }
2359   ProfileData* pdata = NULL;
2360 
2361 
2362   // For reasons which are recorded per bytecode, we check per-BCI data.
2363   DeoptReason per_bc_reason = reason_recorded_per_bytecode_if_any(reason);
2364   assert(per_bc_reason != Reason_none || update_total_trap_count, "must be");
2365   if (per_bc_reason != Reason_none) {
2366     // Find the profile data for this BCI.  If there isn't one,
2367     // try to allocate one from the MDO's set of spares.
2368     // This will let us detect a repeated trap at this point.
2369     pdata = trap_mdo->allocate_bci_to_data(trap_bci, reason_is_speculate(reason) ? compiled_method : NULL);
2370 
2371     if (pdata != NULL) {
2372       if (reason_is_speculate(reason) && !pdata->is_SpeculativeTrapData()) {
2373         if (LogCompilation && xtty != NULL) {
2374           ttyLocker ttyl;
2375           // no more room for speculative traps in this MDO
2376           xtty->elem("speculative_traps_oom");
2377         }
2378       }
2379       // Query the trap state of this profile datum.
2380       int tstate0 = pdata->trap_state();
2381       if (!trap_state_has_reason(tstate0, per_bc_reason))
2382         maybe_prior_trap = false;
2383       if (!trap_state_is_recompiled(tstate0))
2384         maybe_prior_recompile = false;
2385 
2386       // Update the trap state of this profile datum.
2387       int tstate1 = tstate0;
2388       // Record the reason.
2389       tstate1 = trap_state_add_reason(tstate1, per_bc_reason);
2390       // Store the updated state on the MDO, for next time.
2391       if (tstate1 != tstate0)
2392         pdata->set_trap_state(tstate1);
2393     } else {
2394       if (LogCompilation && xtty != NULL) {
2395         ttyLocker ttyl;
2396         // Missing MDP?  Leave a small complaint in the log.
2397         xtty->elem("missing_mdp bci='%d'", trap_bci);
2398       }
2399     }
2400   }
2401 
2402   // Return results:
2403   ret_this_trap_count = this_trap_count;
2404   ret_maybe_prior_trap = maybe_prior_trap;
2405   ret_maybe_prior_recompile = maybe_prior_recompile;
2406   return pdata;
2407 }

2446 const int DS_RECOMPILE_BIT = DataLayout::trap_mask - DS_REASON_MASK;

2448 //---------------------------trap_state_reason---------------------------------
2449 Deoptimization::DeoptReason
2450 Deoptimization::trap_state_reason(int trap_state) {
2451   // This assert provides the link between the width of DataLayout::trap_bits
2452   // and the encoding of "recorded" reasons.  It ensures there are enough
2453   // bits to store all needed reasons in the per-BCI MDO profile.
2454   assert(DS_REASON_MASK >= Reason_RECORDED_LIMIT, "enough bits");
2455   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2456   trap_state -= recompile_bit;
2457   if (trap_state == DS_REASON_MASK) {
2458     return Reason_many;
2459   } else {
2460     assert((int)Reason_none == 0, "state=0 => Reason_none");
2461     return (DeoptReason)trap_state;
2462   }
2463 }
2464 //-------------------------trap_state_has_reason-------------------------------
2465 int Deoptimization::trap_state_has_reason(int trap_state, int reason) {
2466   assert(reason_is_recorded_per_bytecode((DeoptReason)reason), "valid reason");
2467   assert(DS_REASON_MASK >= Reason_RECORDED_LIMIT, "enough bits");
2468   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2469   trap_state -= recompile_bit;
2470   if (trap_state == DS_REASON_MASK) {
2471     return -1;  // true, unspecifically (bottom of state lattice)
2472   } else if (trap_state == reason) {
2473     return 1;   // true, definitely
2474   } else if (trap_state == 0) {
2475     return 0;   // false, definitely (top of state lattice)
2476   } else {
2477     return 0;   // false, definitely
2478   }
2479 }
2480 //-------------------------trap_state_add_reason-------------------------------
2481 int Deoptimization::trap_state_add_reason(int trap_state, int reason) {
2482   assert(reason_is_recorded_per_bytecode((DeoptReason)reason) || reason == Reason_many, "valid reason");
2483   int recompile_bit = (trap_state & DS_RECOMPILE_BIT);
2484   trap_state -= recompile_bit;
2485   if (trap_state == DS_REASON_MASK) {
2486     return trap_state + recompile_bit;     // already at state lattice bottom
2487   } else if (trap_state == reason) {
2488     return trap_state + recompile_bit;     // the condition is already true
2489   } else if (trap_state == 0) {
2490     return reason + recompile_bit;          // no condition has yet been true
2491   } else {
2492     return DS_REASON_MASK + recompile_bit;  // fall to state lattice bottom
2493   }
2494 }

2497   return (trap_state & DS_RECOMPILE_BIT) != 0;

2642   assert(reason >= 0 && reason < Reason_LIMIT, "oob");
2643   assert(action >= 0 && action < Action_LIMIT, "oob");
2644   _deoptimization_hist[Reason_none][0][0] += 1;  // total
2645   _deoptimization_hist[reason][0][0]      += 1;  // per-reason total
2646   juint* cases = _deoptimization_hist[reason][1+action];
2647   juint* bc_counter_addr = NULL;
2648   juint  bc_counter      = 0;
2649   // Look for an unused counter, or an exact match to this BC.
2650   if (bc != Bytecodes::_illegal) {
2651     for (int bc_case = 0; bc_case < BC_CASE_LIMIT; bc_case++) {
2652       juint* counter_addr = &cases[bc_case];
2653       juint  counter = *counter_addr;
2654       if ((counter == 0 && bc_counter_addr == NULL)
2655           || (Bytecodes::Code)(counter & LSB_MASK) == bc) {
2656         // this counter is either free or is already devoted to this BC
2657         bc_counter_addr = counter_addr;
2658         bc_counter = counter | bc;
2659       }
2660     }
2661   }
2662   if (bc_counter_addr == NULL) {
2663     // Overflow, or no given bytecode.
2664     bc_counter_addr = &cases[BC_CASE_LIMIT-1];
2665     bc_counter = (*bc_counter_addr & ~LSB_MASK);  // clear LSB
2666   }
2667   *bc_counter_addr = bc_counter + (1 << LSB_BITS);
2668 }

2671   return _deoptimization_hist[Reason_none][0][0];
2672 }

2674 void Deoptimization::print_statistics() {
2675   juint total = total_deoptimization_count();
2676   juint account = total;
2677   if (total != 0) {
2678     ttyLocker ttyl;
2679     if (xtty != NULL)  xtty->head("statistics type='deoptimization'");
2680     tty->print_cr("Deoptimization traps recorded:");
2681     #define PRINT_STAT_LINE(name, r) \
2682       tty->print_cr("  %4d (%4.1f%%) %s", (int)(r), ((r) * 100.0) / total, name);
2683     PRINT_STAT_LINE("total", total);
2684     // For each non-zero entry in the histogram, print the reason,
2685     // the action, and (if specifically known) the type of bytecode.
2686     for (int reason = 0; reason < Reason_LIMIT; reason++) {
2687       for (int action = 0; action < Action_LIMIT; action++) {
2688         juint* cases = _deoptimization_hist[reason][1+action];
2689         for (int bc_case = 0; bc_case < BC_CASE_LIMIT; bc_case++) {
2690           juint counter = cases[bc_case];
2691           if (counter != 0) {
2692             char name[1*K];
2693             Bytecodes::Code bc = (Bytecodes::Code)(counter & LSB_MASK);
2694             if (bc_case == BC_CASE_LIMIT && (int)bc == 0)
2695               bc = Bytecodes::_illegal;
2696             sprintf(name, "%s/%s/%s",
2697                     trap_reason_name(reason),
2698                     trap_action_name(action),
2699                     Bytecodes::is_defined(bc)? Bytecodes::name(bc): "other");
2700             juint r = counter >> LSB_BITS;
2701             tty->print_cr("  %40s: " UINT32_FORMAT " (%.1f%%)", name, r, (r * 100.0) / total);
2702             account -= r;
2703           }
2704         }
2705       }
2706     }
2707     if (account != 0) {
2708       PRINT_STAT_LINE("unaccounted", account);
2709     }
2710     #undef PRINT_STAT_LINE
2711     if (xtty != NULL)  xtty->tail("statistics");
2712   }
2713 }
