229     public static String[] concat(String prefix[], String... extra) {
230         ArrayList<String> list = new ArrayList<String>();
231         if (prefix != null) {
232             for (String s : prefix) {
233                 list.add(s);
234             }
235         }
236         if (extra != null) {
237             for (String s : extra) {
238                 list.add(s);
239             }
240         }
241 
242         return list.toArray(new String[list.size()]);
243     }

329     public static void main(String[] args) throws Throwable {
330         if (args.length != 1) {
331             throw new RuntimeException("Usage: @run main/othervm/timeout=120 -Xbootclasspath/a:." +
332                                        " -XX:+IgnoreUnrecognizedVMOptions -XX:+UnlockDiagnosticVMOptions" +
333                                        " -XX:+UnlockExperimentalVMOptions -XX:+WhiteBoxAPI" +
334                                        " compiler.valhalla.valuetypes.ValueTypeTest <YourTestMainClass>");
335         }
336         String testMainClassName = args[0];
337         Class testMainClass = Class.forName(testMainClassName);
338         ValueTypeTest test = (ValueTypeTest)testMainClass.newInstance();
339         List<String> scenarios = null;
340         if (!SCENARIOS.isEmpty()) {
341            scenarios = Arrays.asList(SCENARIOS.split(","));
342         }
343         for (int i=0; i<test.getNumScenarios(); i++) {
344             String reason;
345             if ((reason = test.isScenarioIgnored(i)) != null) {
346                 System.out.println("Scenario #" + i + " is ignored: " + reason);
347             } else if (scenarios != null && !scenarios.contains(Integer.toString(i))) {
348                 System.out.println("Scenario #" + i + " is skipped due to -Dscenarios=" + SCENARIOS);
349             } else {
350                 System.out.println("Scenario #" + i + " -------- ");
351                 String[] cmds = InputArguments.getVmInputArgs();
352                 if (RUN_SCENARIOS_WITH_XCOMP) {
353                     cmds = concat(cmds, "-Xcomp");
354                 }
355                 cmds = concat(cmds, test.getVMParameters(i));
356                 cmds = concat(cmds, test.getExtraVMParameters(i));
357                 cmds = concat(cmds, testMainClassName);
358 
359                 OutputAnalyzer oa = ProcessTools.executeTestJvm(cmds);
360                 String output = oa.getOutput();
361                 oa.shouldHaveExitValue(0);
362                 System.out.println(output);
363             }
364         }
365     }

367     // To exclude test cases, use -DExclude=<case1>,<case2>,...
368     // Each case can be just the method name, or can be <class>.<method>. The latter form is useful
369     // when you are running several tests at the same time.
370     //
371     // jtreg -DExclude=test12 TestArrays.java
372     // jtreg -DExclude=test34 TestLWorld.java
373     // -- or --
374     // jtreg -DExclude=TestArrays.test12,TestLWorld.test34 TestArrays.java TestLWorld.java
375     //
376     private List<String> buildExcludeList() {
377         List<String> exclude = null;
378         String classPrefix = getClass().getSimpleName() + ".";
379         if (!EXCLUDELIST.isEmpty()) {
380             exclude = new ArrayList(Arrays.asList(EXCLUDELIST.split(",")));
381             for (int i = exclude.size() - 1; i >= 0; i--) {
382                 String ex = exclude.get(i);
383                 if (ex.indexOf(".") > 0) {
384                     if (ex.startsWith(classPrefix)) {
385                         ex = ex.substring(classPrefix.length());
386                         exclude.set(i, ex);
387                     } else {
388                         exclude.remove(i);
389                     }
390                 }
391             }
392         }
393         return exclude;
394     }

396     protected ValueTypeTest() {
397         List<String> list = null;
398         if (!TESTLIST.isEmpty()) {
399            list = Arrays.asList(TESTLIST.split(","));
400         }
401         List<String> exclude = buildExcludeList();
402 
403         // Gather all test methods and put them in Hashtable
404         for (Method m : getClass().getDeclaredMethods()) {
405             Test[] annos = m.getAnnotationsByType(Test.class);
406             if (annos.length != 0 &&
407                 ((list == null || list.contains(m.getName())) && (exclude == null || !exclude.contains(m.getName())))) {
408                 tests.put(getClass().getSimpleName() + "::" + m.getName(), m);
409             }
410         }
411     }

413     protected void run(String[] args, Class<?>... classes) throws Throwable {
414         if (args.length == 0) {
415             // Spawn a new VM instance
416             execute_vm();
417         } else {
418             // Execute tests in the VM spawned by the above code.
419             Asserts.assertTrue(args.length == 1 && args[0].equals("run"), "must be");
420             run(classes);
421         }
422     }

424     private void execute_vm() throws Throwable {
425         Asserts.assertFalse(tests.isEmpty(), "no tests to execute");
426         String[] vmInputArgs = InputArguments.getVmInputArgs();
427         for (String arg : vmInputArgs) {
428             if (arg.startsWith("-XX:CompileThreshold")) {
429                 // Disable IR verification if non-default CompileThreshold is set
430                 VERIFY_IR = false;
431             }
432         }
433         // Each VM is launched with flags in this order, so the later ones can override the earlier one:
434         //     VERIFY_IR/VERIFY_VM flags specified below
435         //     vmInputArgs, which consists of:
436         //        @run options
437         //        getVMParameters()
438         //        getExtraVMParameters()
439         //     defaultFlags
440         //     compileCommandFlags
441         String cmds[] = null;
442         if (VERIFY_IR) {
443             // Add print flags for IR verification
444             cmds = concat(cmds, printFlags);
445             // Always trap for exception throwing to not confuse IR verification
446             cmds = concat(cmds, "-XX:-OmitStackTraceInFastThrow");
447         }
448         if (VERIFY_VM) {
449             cmds = concat(cmds, verifyFlags);
450         }
451         cmds = concat(cmds, vmInputArgs);
452         cmds = concat(cmds, defaultFlags);
453         if (COMPILE_COMMANDS) {
454           cmds = concat(cmds, compileCommandFlags);
455         }
456 
457         // Run tests in own process and verify output
458         cmds = concat(cmds, getClass().getName(), "run");
459         OutputAnalyzer oa = ProcessTools.executeTestJvm(cmds);
460         // If ideal graph printing is enabled/supported, verify output
461         String output = oa.getOutput();
462         oa.shouldHaveExitValue(0);
463         if (VERIFY_IR) {
464             if (output.contains("PrintIdeal enabled")) {
465                 parseOutput(output);
466             } else {
467                 System.out.println(output);
468                 System.out.println("WARNING: IR verification failed! Running with -Xint, -Xcomp or release build?");
469             }
470         }
471     }

473     static final class TestAnnotation {
474         private final int flag;
475         private final BooleanSupplier predicate;
476 
477         private static final TestAnnotation testAnnotations[] = {
478             new TestAnnotation(InlineTypePassFieldsAsArgsOn, () -> InlineTypePassFieldsAsArgs),
479             new TestAnnotation(InlineTypePassFieldsAsArgsOff, () -> !InlineTypePassFieldsAsArgs),
480             new TestAnnotation(InlineTypeArrayFlattenOn, () -> InlineTypeArrayFlatten),
481             new TestAnnotation(InlineTypeArrayFlattenOff, () -> !InlineTypeArrayFlatten),
482             new TestAnnotation(InlineTypeReturnedAsFieldsOn, () -> InlineTypeReturnedAsFields),
483             new TestAnnotation(InlineTypeReturnedAsFieldsOff, () -> !InlineTypeReturnedAsFields),
484             new TestAnnotation(AlwaysIncrementalInlineOn, () -> AlwaysIncrementalInline),
485             new TestAnnotation(AlwaysIncrementalInlineOff, () -> !AlwaysIncrementalInline),
486             new TestAnnotation(G1GCOn, () -> G1GC),
487             new TestAnnotation(G1GCOff, () -> !G1GC),
488             new TestAnnotation(ZGCOn, () -> ZGC),
489             new TestAnnotation(ZGCOff, () -> !ZGC),
490             new TestAnnotation(ArrayLoadStoreProfileOn, () -> UseArrayLoadStoreProfile),
491             new TestAnnotation(ArrayLoadStoreProfileOff, () -> !UseArrayLoadStoreProfile),
492             new TestAnnotation(TypeProfileOn, () -> TypeProfileLevel == 222),
493             new TestAnnotation(TypeProfileOff, () -> TypeProfileLevel == 0),
494         };
495 
496         private TestAnnotation(int flag, BooleanSupplier predicate) {
497             this.flag = flag;
498             this.predicate = predicate;
499         }
500 
501         private boolean match(Test a) {
502             return (a.valid() & flag) != 0 && predicate.getAsBoolean();
503         }
504 
505         static boolean find(Test a) {
506             Stream<TestAnnotation> s = Arrays.stream(testAnnotations).filter(t -> t.match(a));
507             long c = s.count();
508             if (c > 1) {
509                 throw new RuntimeException("At most one Test annotation should match");
510             }
511             return c > 0;
512         }
513     }

515     private void parseOutput(String output) throws Exception {
516         Pattern comp_re = Pattern.compile("\\n\\s+\\d+\\s+\\d+\\s+(%| )(s| )(!| )b(n| )\\s+\\d?\\s+\\S+\\.(?<name>[^.]+::\\S+)\\s+(?<osr>@ \\d+\\s+)?[(]\\d+ bytes[)]");
517         Matcher m = comp_re.matcher(output);
518         Map<String,String> compilations = new LinkedHashMap<>();
519         int prev = 0;
520         String methodName = null;
521         while (m.find()) {
522             if (prev == 0) {
523                 // Print header
524                 System.out.print(output.substring(0, m.start()+1));
525             } else if (methodName != null) {
526                 compilations.put(methodName, output.substring(prev, m.start()+1));
527             }
528             if (m.group("osr") != null) {
529                 methodName = null;
530             } else {
531                 methodName = m.group("name");
532             }
533             prev = m.end();
534         }
535         if (prev == 0) {
536             // Print header
537             System.out.print(output);
538         } else if (methodName != null) {
539             compilations.put(methodName, output.substring(prev));
540         }
541         // Iterate over compilation output
542         for (String testName : compilations.keySet()) {
543             Method test = tests.get(testName);
544             if (test == null) {
545                 // Skip helper methods
546                 continue;
547             }
548             String graph = compilations.get(testName);
549             if (PRINT_GRAPH) {
550                 System.out.println("\nGraph for " + testName + "\n" + graph);
551             }
552             // Parse graph using regular expressions to determine if it contains forbidden nodes
553             Test[] annos = test.getAnnotationsByType(Test.class);
554             Test anno = Arrays.stream(annos).filter(TestAnnotation::find).findFirst().orElse(null);
555             if (anno == null) {
556                 Object[] res = Arrays.stream(annos).filter(a -> a.valid() == 0).toArray();
557                 if (res.length != 1) {
558                     throw new RuntimeException("Only one Test annotation should match");
559                 }
560                 anno = (Test)res[0];
561             }
562             String regexFail = anno.failOn();
563             if (!regexFail.isEmpty()) {
564                 Pattern pattern = Pattern.compile(regexFail.substring(0, regexFail.length()-1));
565                 Matcher matcher = pattern.matcher(graph);
566                 boolean found = matcher.find();
567                 Asserts.assertFalse(found, "Graph for '" + testName + "' contains forbidden node:\n" + (found ? matcher.group() : ""));
568             }
569             String[] regexMatch = anno.match();
570             int[] matchCount = anno.matchCount();
571             for (int i = 0; i < regexMatch.length; ++i) {
572                 Pattern pattern = Pattern.compile(regexMatch[i].substring(0, regexMatch[i].length()-1));
573                 Matcher matcher = pattern.matcher(graph);
574                 int count = 0;
575                 String nodes = "";
576                 while (matcher.find()) {
577                     count++;
578                     nodes += matcher.group() + "\n";
579                 }
580                 if (matchCount[i] < 0) {
581                     Asserts.assertLTE(Math.abs(matchCount[i]), count, "Graph for '" + testName + "' contains different number of match nodes (expected <= " + matchCount[i] + " but got " + count + "):\n" + nodes);
582                 } else {
583                     Asserts.assertEQ(matchCount[i], count, "Graph for '" + testName + "' contains different number of match nodes (expected " + matchCount[i] + " but got " + count + "):\n" + nodes);
584                 }
585             }
586             tests.remove(testName);
587             System.out.println(testName + " passed");
588         }
589         // Check if all tests were compiled
590         if (tests.size() != 0) {
591             for (String name : tests.keySet()) {
592                 System.out.println("Test '" + name + "' not compiled!");
593             }
594             throw new RuntimeException("Not all tests were compiled");
595         }
596     }

598     private void setup(Class<?> clazz) {
599         if (XCOMP) {
600             // Don't control compilation if -Xcomp is enabled
601             return;
602         }
603         if (DUMP_REPLAY) {
604             // Generate replay compilation files
605             String directive = "[{ match: \"*.*\", DumpReplay: true }]";
606             if (WHITE_BOX.addCompilerDirective(directive) != 1) {
607                 throw new RuntimeException("Failed to add compiler directive");
608             }
609         }
610 
611         Method[] methods = clazz.getDeclaredMethods();
612         for (Method m : methods) {
613             if (m.isAnnotationPresent(Test.class)) {
614                 // Don't inline tests
615                 WHITE_BOX.testSetDontInlineMethod(m, true);
616             }
617             if (m.isAnnotationPresent(DontCompile.class)) {
618                 WHITE_BOX.makeMethodNotCompilable(m, COMP_LEVEL_ANY, true);
619                 WHITE_BOX.makeMethodNotCompilable(m, COMP_LEVEL_ANY, false);
620                 WHITE_BOX.testSetDontInlineMethod(m, true);
621             }
622             if (m.isAnnotationPresent(ForceInline.class)) {
623                 Asserts.assertFalse(m.isAnnotationPresent(DontInline.class), "Method " + m.getName() + " has contradicting DontInline annotation");
624                 WHITE_BOX.testSetForceInlineMethod(m, true);
625             }
626             if (m.isAnnotationPresent(DontInline.class)) {
627                 Asserts.assertFalse(m.isAnnotationPresent(ForceInline.class), "Method " + m.getName() + " has contradicting ForceInline annotation");
628                 WHITE_BOX.testSetDontInlineMethod(m, true);
629             }
630             if (STRESS_CC) {
631                 // Exclude some methods from compilation with C2 to stress test the calling convention
632                 if (Utils.getRandomInstance().nextBoolean()) {
633                     System.out.println("Excluding from C2 compilation: " + m);
634                     WHITE_BOX.makeMethodNotCompilable(m, COMP_LEVEL_FULL_OPTIMIZATION, false);
635                 }
636             }
637         }
638         // Only force compilation now because above annotations affect inlining
639         for (Method m : methods) {
640             if (m.isAnnotationPresent(ForceCompile.class)) {
641                 Asserts.assertFalse(m.isAnnotationPresent(DontCompile.class), "Method " + m.getName() + " has contradicting DontCompile annotation");
642                 int compLevel = getCompLevel(m.getAnnotation(ForceCompile.class));
643                 enqueueMethodForCompilation(m, compLevel);
644             }
645         }
646         // Compile class initializers
647         int compLevel = getCompLevel(null);
648         WHITE_BOX.enqueueInitializerForCompilation(clazz, compLevel);
649     }

651     private void run(Class<?>... classes) throws Exception {
652         if (USE_COMPILER && PRINT_IDEAL && !XCOMP && !STRESS_CC) {
653             System.out.println("PrintIdeal enabled");
654         }
655         System.out.format("rI = %d, rL = %d\n", rI, rL);
656 
657         setup(getClass());
658         for (Class<?> clazz : classes) {
659             setup(clazz);
660         }
661 
662         // Execute tests
663         TreeMap<Long, String> durations = (PRINT_TIMES || VERBOSE) ? new TreeMap<Long, String>() : null;
664         for (Method test : tests.values()) {
665             if (VERBOSE) {
666                 System.out.println("Starting " + test.getName());
667             }
668             TempSkipForC1 c1skip = test.getAnnotation(TempSkipForC1.class);
669             if (TEST_C1 && c1skip != null) {
670                 System.out.println("Skipped " + test.getName() + " for C1 testing: " + c1skip.reason());
671                 continue;
672             }
673             long startTime = System.nanoTime();
674             Method verifier = getClass().getMethod(test.getName() + "_verifier", boolean.class);
675             // Warmup using verifier method
676             Warmup anno = test.getAnnotation(Warmup.class);
677             int warmup = anno == null ? WARMUP : anno.value();
678             for (int i = 0; i < warmup; ++i) {
679                 verifier.invoke(this, true);
680             }
681             boolean osrOnly = (test.getAnnotation(OSRCompileOnly.class) != null);
682 
683             // C1 generates a lot of code when VerifyOops is enabled and may run out of space (for a small
684             // number of test cases).
685             boolean maybeCodeBufferOverflow = (TEST_C1 && VerifyOops);
686 
687             if (osrOnly) {
688                 long started = System.currentTimeMillis();
689                 boolean stateCleared = false;
690                 for (;;) {
691                     long elapsed = System.currentTimeMillis() - started;
692                     if (maybeCodeBufferOverflow && elapsed > 5000 && !WHITE_BOX.isMethodCompiled(test, false)) {
693                         System.out.println("Temporarily disabling VerifyOops");
694                         try {
695                             WHITE_BOX.setBooleanVMFlag("VerifyOops", false);
696                             if (!stateCleared) {
697                                 WHITE_BOX.clearMethodState(test);
698                                 stateCleared = true;
699                             }
700                             verifier.invoke(this, false);
701                         } finally {
702                             WHITE_BOX.setBooleanVMFlag("VerifyOops", true);
703                             System.out.println("Re-enabled VerifyOops");
704                         }
705                     } else {
706                         verifier.invoke(this, false);
707                     }
708 
709                     boolean b = WHITE_BOX.isMethodCompiled(test, false);
710                     if (VERBOSE) {
711                         System.out.println("Is " + test.getName() + " compiled? " + b);
712                     }
713                     if (b || XCOMP || STRESS_CC || !USE_COMPILER) {
714                         // Don't control compilation if -Xcomp is enabled, or if compiler is disabled
715                         break;
716                     }
717                     Asserts.assertTrue(OSR_TEST_TIMEOUT < 0 || elapsed < OSR_TEST_TIMEOUT, test + " not compiled after " + OSR_TEST_TIMEOUT + " ms");
718                 }
719                 Asserts.assertTrue(XCOMP || STRESS_CC || !USE_COMPILER || WHITE_BOX.isMethodCompiled(test, false), test + " not compiled");
720             } else {
721                 int compLevel = getCompLevel(test.getAnnotation(Test.class));
722                 // Trigger compilation
723                 enqueueMethodForCompilation(test, compLevel);
724                 if (maybeCodeBufferOverflow && !WHITE_BOX.isMethodCompiled(test, false)) {
725                     // Let's disable VerifyOops temporarily and retry.
726                     WHITE_BOX.setBooleanVMFlag("VerifyOops", false);
727                     WHITE_BOX.clearMethodState(test);
728                     enqueueMethodForCompilation(test, compLevel);
729                     WHITE_BOX.setBooleanVMFlag("VerifyOops", true);
730                 }
731                 Asserts.assertTrue(STRESS_CC || !USE_COMPILER || WHITE_BOX.isMethodCompiled(test, false), test + " not compiled");
732                 // Check result
733                 verifier.invoke(this, false);
734             }
735             if (PRINT_TIMES || VERBOSE) {
736                 long endTime = System.nanoTime();
737                 long duration = (endTime - startTime);
738                 durations.put(duration, test.getName());
739                 if (VERBOSE) {
740                     System.out.println("Done " + test.getName() + ": " + duration + " ns = " + (duration / 1000000) + " ms");
741                 }
742             }
743             if (GC_AFTER) {
744                 System.out.println("doing GC");
745                 System.gc();
746             }
747         }
748 
749         // Print execution times
750         if (PRINT_TIMES) {
751           System.out.println("\n\nTest execution times:");
752           for (Map.Entry<Long, String> entry : durations.entrySet()) {
753               System.out.format("%-10s%15d ns\n", entry.getValue() + ":", entry.getKey());
754           }
755         }
756     }

758     // Get the appropriate compilation level for a method, according to the
759     // given annotation, as well as the current test scenario and VM options.
760     //
761     private int getCompLevel(Object annotation) {
762         int compLevel;
763         if (annotation == null) {
764             compLevel = COMP_LEVEL_ANY;
765         } else if (annotation instanceof Test) {
766             compLevel = ((Test)annotation).compLevel();
767         } else {
768             compLevel = ((ForceCompile)annotation).compLevel();
769         }
770 
771         return restrictCompLevel(compLevel);
772     }
773 
774     // Get the appropriate level as permitted by the test scenario and VM options.
775     private static int restrictCompLevel(int compLevel) {
776         if (compLevel == COMP_LEVEL_ANY) {
777             compLevel = COMP_LEVEL_FULL_OPTIMIZATION;
778         }
779         if (FLIP_C1_C2) {
780             // Effectively treat all (compLevel = C1) as (compLevel = C2), and
781             //                       (compLevel = C2) as (compLevel = C1).
782             if (compLevel == COMP_LEVEL_SIMPLE) {
783                 compLevel = COMP_LEVEL_FULL_OPTIMIZATION;
784             } else if (compLevel == COMP_LEVEL_FULL_OPTIMIZATION) {
785                 compLevel = COMP_LEVEL_SIMPLE;
786             }
787         }
788         if (!TEST_C1 && compLevel < COMP_LEVEL_FULL_OPTIMIZATION) {
789             compLevel = COMP_LEVEL_FULL_OPTIMIZATION;
790         }
791         if (compLevel > (int)TieredStopAtLevel) {
792             compLevel = (int)TieredStopAtLevel;
793         }
794         return compLevel;
795     }

797     public static void enqueueMethodForCompilation(Method m, int level) {
798         level = restrictCompLevel(level);
799         if (VERBOSE) {
800             System.out.println("enqueueMethodForCompilation " + m + ", level = " + level);
801         }
802         WHITE_BOX.enqueueMethodForCompilation(m, level);
803     }

805     // Unlike C2, C1 intrinsics never deoptimize System.arraycopy. Instead, we fall back to
806     // a normal method invocation when encountering flattened arrays.
807     static boolean isCompiledByC2(Method m) {
808         return USE_COMPILER && !XCOMP && WHITE_BOX.isMethodCompiled(m, false) &&
809             WHITE_BOX.getMethodCompilationLevel(m, false) >= COMP_LEVEL_FULL_OPTIMIZATION;
810     }
