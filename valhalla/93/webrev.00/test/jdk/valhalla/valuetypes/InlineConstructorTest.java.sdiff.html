 25 /*
 26  * @test
 27  * @summary Test reflection of constructors for inline classes
 28  * @run main/othervm InlineConstructorTest
 29  */
 30 
 31 import java.lang.reflect.Constructor;
 32 import java.lang.reflect.Field;
 33 import java.lang.reflect.InaccessibleObjectException;
 34 import java.lang.reflect.Method;
 35 import java.lang.reflect.Modifier;
 36 import java.util.Arrays;
 37 import java.util.Set;
 38 import java.util.stream.Collectors;
 39 



 40 public class InlineConstructorTest {
 41 
 42     // Target test class
 43     static inline class SimpleInline {
 44         public final int x;
 45 
 46         SimpleInline() {
 47             x = -1;
 48         }
 49 
 50         public SimpleInline(int x) {
 51             this.x = x;
 52         }
 53     }
 54 
 55     private final Class<?> c;
 56 
 57     public static void main(String... args) throws Exception {
 58         testSimpleInlineClass();
 59     }
 60 
 61     static void testSimpleInlineClass() throws Exception  {
 62         InlineConstructorTest test = new InlineConstructorTest(SimpleInline.class);
 63         test.constructor();
 64         test.constructors("public InlineConstructorTest$SimpleInline(int)",
 65                 "InlineConstructorTest$SimpleInline()");
 66         test.setAccessible();
 67         test.trySetAccessible();
 68         test.initFactoryNotMethods();
 69     }
 70 
 71     InlineConstructorTest(Class<?> type) throws Exception {
 72         String cn = type.getName();
 73         this.c = Class.forName(cn);

 74 
 75         assertTrue(c.isInlineClass());
 76         assertEquals(c, type);
 77     }
 78 
 79     void constructor() throws Exception {
 80         Constructor<?> ctor = c.getDeclaredConstructor();

 81         Object o = ctor.newInstance();
 82         assertEquals(o.getClass(), c);
 83     }
 84 
 85     // Check that the class has the expected Constructors
 86     void constructors(String... expected) throws Exception {
 87         Constructor<? extends Object>[] cons = c.getDeclaredConstructors();
 88         Set<String> actualSig =
 89                 Arrays.stream(cons).map(Constructor::toString).collect(Collectors.toSet());
 90         Set<String> expectedSig = Set.of(expected);


 91         boolean ok = expectedSig.equals(actualSig);
 92         if (!ok) {
 93             System.out.printf("expected: %s%n", expectedSig);
 94             System.out.printf("declared: %s%n", actualSig);
 95             assertTrue(ok);
 96         }
 97     }
 98 
 99     // Check that the constructor can be set accessible and that the field x can not
100     void setAccessible() throws Exception {
101         Constructor<?> ctor = c.getDeclaredConstructor();

102         ctor.setAccessible(true);
103         Field field = c.getField("x");
104         try {
105             field.setAccessible(true);
106             throw new RuntimeException("InaccessibleObjectException not thrown");
107         } catch (InaccessibleObjectException e) {
108             // IOE is expected
109         }
110     }
111 
112     // Check that the constructor can be set accessible and that the field x can not
113     void trySetAccessible() throws Exception {
114         Constructor<?> ctor = c.getDeclaredConstructor();

115         assertTrue(ctor.trySetAccessible());
116         Field field = c.getField("x");
117         if (field.trySetAccessible()) {
118             throw new RuntimeException("trySetAccessible should not succeed");
119         }
120     }
121 









122     // Check that the class does not have a static method with the name <init>
123     void initFactoryNotMethods() {
124         Method[] methods = c.getDeclaredMethods();

125         for (Method m : methods) {
126             if (Modifier.isStatic(m.getModifiers())) {
127                 assertFalse(m.getName().equals("<init>"));
128             }
129         }
130     }
131 
132     static void assertEquals(Object o1, Object o2) {
133         if (o1 == o2 || o1.equals(o2))
134             return;
135 
136         throw new AssertionError(o1 + " != " + o2);
137     }
138 
139     static void assertTrue(boolean value) {
140         if (!value)
141             throw new AssertionError("expected true");
142     }
143 
144     static void assertFalse(boolean value) {
145         if (value)
146             throw new AssertionError("expected false");
147     }
148 }

 25 /*
 26  * @test
 27  * @summary Test reflection of constructors for inline classes
 28  * @run testng/othervm InlineConstructorTest
 29  */
 30 
 31 import java.lang.reflect.Constructor;
 32 import java.lang.reflect.Field;

 33 import java.lang.reflect.Method;
 34 import java.lang.reflect.Modifier;
 35 import java.util.Arrays;
 36 import java.util.Set;
 37 import java.util.stream.Collectors;
 38 
 39 import org.testng.annotations.Test;
 40 import static org.testng.Assert.*;
 41 
 42 public class InlineConstructorTest {
 43 
 44     // Target test class
 45     static inline class SimpleInline {
 46         public final int x;
 47 
 48         SimpleInline() {
 49             x = -1;
 50         }
 51 
 52         public SimpleInline(int x) {
 53             this.x = x;
 54         }
 55     }
 56 
 57     static final Class<?> INLINE_TYPE = SimpleInline.class;














 58 
 59     @Test
 60     public static void testInlineClassConstructor() throws Exception {
 61         String cn = INLINE_TYPE.getName();
 62         Class<?> c = Class.forName(cn);
 63 
 64         assertTrue(c.isInlineClass());
 65         assertEquals(c, INLINE_TYPE);
 66     }
 67 
 68     @Test
 69     public static void constructor() throws Exception {
 70         Constructor<?> ctor = INLINE_TYPE.getDeclaredConstructor();
 71         Object o = ctor.newInstance();
 72         assertEquals(o.getClass(), INLINE_TYPE);
 73     }
 74 
 75     // Check that the class has the expected Constructors
 76     @Test
 77     public static void constructors() throws Exception {
 78         Set<String> expectedSig = Set.of("public InlineConstructorTest$SimpleInline(int)",
 79                                          "InlineConstructorTest$SimpleInline()");
 80         Constructor<? extends Object>[] cons = INLINE_TYPE.getDeclaredConstructors();
 81         Set<String> actualSig = Arrays.stream(cons).map(Constructor::toString)
 82                                       .collect(Collectors.toSet());
 83         boolean ok = expectedSig.equals(actualSig);
 84         if (!ok) {
 85             System.out.printf("expected: %s%n", expectedSig);
 86             System.out.printf("declared: %s%n", actualSig);
 87             assertTrue(ok);
 88         }
 89     }
 90 
 91     // Check that the constructor and field can be set accessible
 92     @Test
 93     public static void setAccessible() throws Exception {
 94         Constructor<?> ctor = INLINE_TYPE.getDeclaredConstructor();
 95         ctor.setAccessible(true);
 96 
 97         Field field = INLINE_TYPE.getField("x");
 98         field.setAccessible(true);




 99     }
100 
101     // Check that the constructor and field can be set accessible
102     @Test
103     public static void trySetAccessible() throws Exception {
104         Constructor<?> ctor = INLINE_TYPE.getDeclaredConstructor();
105         assertTrue(ctor.trySetAccessible());
106 
107         Field field = INLINE_TYPE.getField("x");
108         assertTrue(field.trySetAccessible());

109     }
110 
111     // Check that the final field cannot be modified
112     @Test(expectedExceptions = IllegalAccessException.class)
113     public static void setFinalField() throws Exception {
114         Field field = INLINE_TYPE.getField("x");
115         field.setAccessible(true);
116         field.setInt(new SimpleInline(100), 200);
117     }
118 
119 
120     // Check that the class does not have a static method with the name <init>
121     @Test
122     public static void initFactoryNotMethods() {
123         Method[] methods = INLINE_TYPE.getDeclaredMethods();
124         for (Method m : methods) {
125             if (Modifier.isStatic(m.getModifiers())) {
126                 assertFalse(m.getName().equals("<init>"));
127             }
128         }
129     }

















130 }
