 101 // ------------------------------------------------------------------
 102 // ciEnv::ciEnv
 103 ciEnv::ciEnv(CompileTask* task)
 104   : _ciEnv_arena(mtCompiler) {
 105   VM_ENTRY_MARK;
 106 
 107   // Set up ciEnv::current immediately, for the sake of ciObjectFactory, etc.
 108   thread->set_env(this);
 109   assert(ciEnv::current() == this, "sanity");
 110 
 111   _oop_recorder = NULL;
 112   _debug_info = NULL;
 113   _dependencies = NULL;
 114   _failure_reason = NULL;
 115   _inc_decompile_count_on_failure = true;
 116   _compilable = MethodCompilable;
 117   _break_at_compile = false;
 118   _compiler_data = NULL;
 119 #ifndef PRODUCT
 120   assert(!firstEnv, "not initialized properly");
 121 #endif /* !PRODUCT */
 122 
 123   _num_inlined_bytecodes = 0;
 124   assert(task == NULL || thread->task() == task, "sanity");
 125   if (task != NULL) {
 126     task->mark_started(os::elapsed_counter());
 127   }
 128   _task = task;
 129   _log = NULL;
 130 
 131   // Temporary buffer for creating symbols and such.
 132   _name_buffer = NULL;
 133   _name_buffer_len = 0;
 134 
 135   _arena   = &_ciEnv_arena;
 136   _factory = new (_arena) ciObjectFactory(_arena, 128);
 137 
 138   // Preload commonly referenced system ciObjects.
 139 
 140   // During VM initialization, these instances have not yet been created.
 141   // Assertions ensure that these instances are not accessed before
 142   // their initialization.
 143 
 144   assert(Universe::is_fully_initialized(), "should be complete");
 145 
 146   oop o = Universe::null_ptr_exception_instance();
 147   assert(o != NULL, "should have been initialized");
 148   _NullPointerException_instance = get_object(o)->as_instance();
 149   o = Universe::arithmetic_exception_instance();
 150   assert(o != NULL, "should have been initialized");
 151   _ArithmeticException_instance = get_object(o)->as_instance();
 152 
 153   _ArrayIndexOutOfBoundsException_instance = NULL;
 154   _ArrayStoreException_instance = NULL;
 155   _ClassCastException_instance = NULL;
 156   _the_null_string = NULL;
 157   _the_min_jint_string = NULL;
 158 
 159   _jvmti_redefinition_count = 0;
 160   _jvmti_can_hotswap_or_post_breakpoint = false;
 161   _jvmti_can_access_local_variables = false;
 162   _jvmti_can_post_on_exceptions = false;
 163   _jvmti_can_pop_frame = false;
 164 }

 232 // ------------------------------------------------------------------
 233 // Cache Jvmti state
 234 void ciEnv::cache_jvmti_state() {
 235   VM_ENTRY_MARK;
 236   // Get Jvmti capabilities under lock to get consistant values.
 237   MutexLocker mu(JvmtiThreadState_lock);
 238   _jvmti_redefinition_count             = JvmtiExport::redefinition_count();
 239   _jvmti_can_hotswap_or_post_breakpoint = JvmtiExport::can_hotswap_or_post_breakpoint();
 240   _jvmti_can_access_local_variables     = JvmtiExport::can_access_local_variables();
 241   _jvmti_can_post_on_exceptions         = JvmtiExport::can_post_on_exceptions();
 242   _jvmti_can_pop_frame                  = JvmtiExport::can_pop_frame();
 243   _jvmti_can_get_owned_monitor_info     = JvmtiExport::can_get_owned_monitor_info();
 244 }
 245 
 246 bool ciEnv::jvmti_state_changed() const {
 247   // Some classes were redefined
 248   if (_jvmti_redefinition_count != JvmtiExport::redefinition_count()) {
 249     return true;
 250   }
 251 
 252   if (!_jvmti_can_access_local_variables &&
 253       JvmtiExport::can_access_local_variables()) {
 254     return true;
 255   }
 256   if (!_jvmti_can_hotswap_or_post_breakpoint &&
 257       JvmtiExport::can_hotswap_or_post_breakpoint()) {
 258     return true;
 259   }
 260   if (!_jvmti_can_post_on_exceptions &&
 261       JvmtiExport::can_post_on_exceptions()) {
 262     return true;
 263   }
 264   if (!_jvmti_can_pop_frame &&
 265       JvmtiExport::can_pop_frame()) {
 266     return true;
 267   }
 268   if (!_jvmti_can_get_owned_monitor_info &&
 269       JvmtiExport::can_get_owned_monitor_info()) {
 270     return true;
 271   }
 272 
 273   return false;
 274 }

 292 // ------------------------------------------------------------------
 293 // helper for lazy exception creation
 294 ciInstance* ciEnv::get_or_create_exception(jobject& handle, Symbol* name) {
 295   VM_ENTRY_MARK;
 296   if (handle == NULL) {
 297     // Cf. universe.cpp, creation of Universe::_null_ptr_exception_instance.
 298     Klass* k = SystemDictionary::find(name, Handle(), Handle(), THREAD);
 299     jobject objh = NULL;
 300     if (!HAS_PENDING_EXCEPTION && k != NULL) {
 301       oop obj = InstanceKlass::cast(k)->allocate_instance(THREAD);
 302       if (!HAS_PENDING_EXCEPTION)
 303         objh = JNIHandles::make_global(Handle(THREAD, obj));
 304     }
 305     if (HAS_PENDING_EXCEPTION) {
 306       CLEAR_PENDING_EXCEPTION;
 307     } else {
 308       handle = objh;
 309     }
 310   }
 311   oop obj = JNIHandles::resolve(handle);
 312   return obj == NULL? NULL: get_object(obj)->as_instance();
 313 }

 363 // ------------------------------------------------------------------
 364 // ciEnv::array_element_offset_in_bytes
 365 int ciEnv::array_element_offset_in_bytes(ciArray* a_h, ciObject* o_h) {
 366   VM_ENTRY_MARK;
 367   objArrayOop a = (objArrayOop)a_h->get_oop();
 368   assert(a->is_objArray(), "");
 369   int length = a->length();
 370   oop o = o_h->get_oop();
 371   for (int i = 0; i < length; i++) {
 372     if (a->obj_at(i) == o)  return i;
 373   }
 374   return -1;
 375 }

 378 // ------------------------------------------------------------------
 379 // ciEnv::check_klass_accessiblity
 380 //
 381 // Note: the logic of this method should mirror the logic of
 382 // ConstantPool::verify_constant_pool_resolve.
 383 bool ciEnv::check_klass_accessibility(ciKlass* accessing_klass,
 384                                       Klass* resolved_klass) {
 385   if (accessing_klass == NULL || !accessing_klass->is_loaded()) {
 386     return true;
 387   }
 388   if (accessing_klass->is_obj_array_klass()) {
 389     accessing_klass = accessing_klass->as_obj_array_klass()->base_element_klass();
 390   }
 391   if (!accessing_klass->is_instance_klass()) {
 392     return true;
 393   }
 394 
 395   if (resolved_klass->is_objArray_klass()) {
 396     // Find the element klass, if this is an array.
 397     resolved_klass = ObjArrayKlass::cast(resolved_klass)->bottom_klass();
 398   }
 399   if (resolved_klass->is_instance_klass()) {
 400     return (Reflection::verify_class_access(accessing_klass->get_Klass(),
 401                                             InstanceKlass::cast(resolved_klass),
 402                                             true) == Reflection::ACCESS_OK);
 403   }
 404   return true;
 405 }

 407 // ------------------------------------------------------------------
 408 // ciEnv::get_klass_by_name_impl
 409 ciKlass* ciEnv::get_klass_by_name_impl(ciKlass* accessing_klass,
 410                                        const constantPoolHandle& cpool,
 411                                        ciSymbol* name,
 412                                        bool require_local) {
 413   ASSERT_IN_VM;
 414   EXCEPTION_CONTEXT;
 415 
 416   // Now we need to check the SystemDictionary
 417   Symbol* sym = name->get_symbol();
 418   if (Signature::has_envelope(sym)) {
 419     // This is a name from a signature.  Strip off the trimmings.
 420     // Call recursive to keep scope of strippedsym.
 421     TempNewSymbol strippedsym = Signature::strip_envelope(sym);
 422     ciSymbol* strippedname = get_symbol(strippedsym);
 423     return get_klass_by_name_impl(accessing_klass, cpool, strippedname, require_local);
 424   }
 425 
 426   // Check for prior unloaded klass.  The SystemDictionary's answers
 427   // can vary over time but the compiler needs consistency.
 428   ciKlass* unloaded_klass = check_get_unloaded_klass(accessing_klass, name);
 429   if (unloaded_klass != NULL) {
 430     if (require_local)  return NULL;
 431     return unloaded_klass;
 432   }
 433 
 434   Handle loader(THREAD, (oop)NULL);
 435   Handle domain(THREAD, (oop)NULL);
 436   if (accessing_klass != NULL) {
 437     loader = Handle(THREAD, accessing_klass->loader());
 438     domain = Handle(THREAD, accessing_klass->protection_domain());
 439   }
 440 
 441   // setup up the proper type to return on OOM
 442   ciKlass* fail_type;
 443   if (sym->char_at(0) == JVM_SIGNATURE_ARRAY) {
 444     fail_type = _unloaded_ciobjarrayklass;
 445   } else {
 446     fail_type = _unloaded_ciinstance_klass;
 447   }
 448   Klass* found_klass;
 449   {
 450     ttyUnlocker ttyul;  // release tty lock to avoid ordering problems
 451     MutexLocker ml(Compile_lock);
 452     Klass* kls;
 453     if (!require_local) {
 454       kls = SystemDictionary::find_constrained_instance_or_array_klass(sym, loader,
 455                                                                        KILL_COMPILE_ON_FATAL_(fail_type));
 456     } else {
 457       kls = SystemDictionary::find_instance_or_array_klass(sym, loader, domain,
 458                                                            KILL_COMPILE_ON_FATAL_(fail_type));
 459     }
 460     found_klass = kls;
 461   }
 462 
 463   // If we fail to find an array klass, look again for its element type.
 464   // The element type may be available either locally or via constraints.
 465   // In either case, if we can find the element type in the system dictionary,
 466   // we must build an array type around it.  The CI requires array klasses
 467   // to be loaded if their element klasses are loaded, except when memory
 468   // is exhausted.
 469   if (Signature::is_array(sym) &&
 470       (sym->char_at(1) == JVM_SIGNATURE_ARRAY ||
 471        sym->char_at(1) == JVM_SIGNATURE_CLASS ||
 472        sym->char_at(1) == JVM_SIGNATURE_INLINETYPE )) {
 473     // We have an unloaded array.
 474     // Build it on the fly if the element class exists.
 475     SignatureStream ss(sym, false);
 476     ss.skip_array_prefix(1);
 477     // Get element ciKlass recursively.
 478     ciKlass* elem_klass =
 479       get_klass_by_name_impl(accessing_klass,
 480                              cpool,
 481                              get_symbol(ss.as_symbol()),
 482                              require_local);
 483     if (elem_klass != NULL && elem_klass->is_loaded()) {
 484       // Now make an array for it
 485       return ciArrayKlass::make(elem_klass);
 486     }
 487   }
 488 
 489   if (found_klass == NULL && !cpool.is_null() && cpool->has_preresolution()) {
 490     // Look inside the constant pool for pre-resolved class entries.
 491     for (int i = cpool->length() - 1; i >= 1; i--) {
 492       if (cpool->tag_at(i).is_klass()) {
 493         Klass* kls = cpool->resolved_klass_at(i);
 494         if (kls->name() == sym) {
 495           found_klass = kls;
 496           break;
 497         }
 498       }
 499     }
 500   }
 501 
 502   if (found_klass != NULL) {
 503     // Found it.  Build a CI handle.
 504     return get_klass(found_klass);
 505   }
 506 
 507   if (require_local)  return NULL;
 508 
 509   // Not yet loaded into the VM, or not governed by loader constraints.
 510   // Make a CI representative for it.
 511   int i = 0;
 512   while (sym->char_at(i) == JVM_SIGNATURE_ARRAY) {
 513     i++;
 514   }
 515   if (i > 0 && sym->char_at(i) == JVM_SIGNATURE_INLINETYPE) {
 516     // An unloaded array class of value types is an ObjArrayKlass, an
 517     // unloaded value type class is an InstanceKlass. For consistency,
 518     // make the signature of the unloaded array of value type use L
 519     // rather than Q.
 520     char *new_name = CURRENT_THREAD_ENV->name_buffer(sym->utf8_length()+1);
 521     strncpy(new_name, (char*)sym->base(), sym->utf8_length());
 522     new_name[i] = JVM_SIGNATURE_CLASS;
 523     new_name[sym->utf8_length()] = '\0';
 524     return get_unloaded_klass(accessing_klass, ciSymbol::make(new_name));
 525   }
 526   return get_unloaded_klass(accessing_klass, name);
 527 }

 540 // ------------------------------------------------------------------
 541 // ciEnv::get_klass_by_index_impl
 542 //
 543 // Implementation of get_klass_by_index.
 544 ciKlass* ciEnv::get_klass_by_index_impl(const constantPoolHandle& cpool,
 545                                         int index,
 546                                         bool& is_accessible,
 547                                         ciInstanceKlass* accessor) {
 548   EXCEPTION_CONTEXT;
 549   Klass* klass = NULL;
 550   Symbol* klass_name = NULL;
 551 
 552   if (cpool->tag_at(index).is_symbol()) {
 553     klass_name = cpool->symbol_at(index);
 554   } else {
 555     // Check if it's resolved if it's not a symbol constant pool entry.
 556     klass = ConstantPool::klass_at_if_loaded(cpool, index);
 557     // Try to look it up by name.
 558     if (klass == NULL) {
 559       klass_name = cpool->klass_name_at(index);
 560     }
 561   }
 562 
 563   if (klass == NULL) {
 564     // Not found in constant pool.  Use the name to do the lookup.
 565     ciKlass* k = get_klass_by_name_impl(accessor,
 566                                         cpool,
 567                                         get_symbol(klass_name),
 568                                         false);
 569     // Calculate accessibility the hard way.
 570     if (!k->is_loaded()) {
 571       is_accessible = false;
 572     } else if (k->loader() != accessor->loader() &&
 573                get_klass_by_name_impl(accessor, cpool, k->name(), true) == NULL) {
 574       // Loaded only remotely.  Not linked yet.
 575       is_accessible = false;
 576     } else {
 577       // Linked locally, and we must also check public/private, etc.
 578       is_accessible = check_klass_accessibility(accessor, k->get_Klass());
 579     }
 580     return k;
 581   }
 582 
 583   // Check for prior unloaded klass.  The SystemDictionary's answers
 584   // can vary over time but the compiler needs consistency.
 585   ciSymbol* name = get_symbol(klass->name());
 586   ciKlass* unloaded_klass = check_get_unloaded_klass(accessor, name);
 587   if (unloaded_klass != NULL) {
 588     is_accessible = false;
 589     return unloaded_klass;
 590   }
 591 
 592   // It is known to be accessible, since it was found in the constant pool.
 593   is_accessible = true;
 594   return get_klass(klass);
 595 }

 617 // ciEnv::get_constant_by_index_impl
 618 //
 619 // Implementation of get_constant_by_index().
 620 ciConstant ciEnv::get_constant_by_index_impl(const constantPoolHandle& cpool,
 621                                              int pool_index, int cache_index,
 622                                              ciInstanceKlass* accessor) {
 623   bool ignore_will_link;
 624   EXCEPTION_CONTEXT;
 625   int index = pool_index;
 626   if (cache_index >= 0) {
 627     assert(index < 0, "only one kind of index at a time");
 628     index = cpool->object_to_cp_index(cache_index);
 629     oop obj = cpool->resolved_references()->obj_at(cache_index);
 630     if (obj != NULL) {
 631       if (obj == Universe::the_null_sentinel()) {
 632         return ciConstant(T_OBJECT, get_object(NULL));
 633       }
 634       BasicType bt = T_OBJECT;
 635       if (cpool->tag_at(index).is_dynamic_constant())
 636         bt = Signature::basic_type(cpool->uncached_signature_ref_at(index));
 637       if (is_reference_type(bt)) {
 638       } else {
 639         // we have to unbox the primitive value
 640         if (!is_java_primitive(bt))  return ciConstant();
 641         jvalue value;
 642         BasicType bt2 = java_lang_boxing_object::get_value(obj, &value);
 643         assert(bt2 == bt, "");
 644         switch (bt2) {
 645         case T_DOUBLE:  return ciConstant(value.d);
 646         case T_FLOAT:   return ciConstant(value.f);
 647         case T_LONG:    return ciConstant(value.j);
 648         case T_INT:     return ciConstant(bt2, value.i);
 649         case T_SHORT:   return ciConstant(bt2, value.s);
 650         case T_BYTE:    return ciConstant(bt2, value.b);
 651         case T_CHAR:    return ciConstant(bt2, value.c);
 652         case T_BOOLEAN: return ciConstant(bt2, value.z);
 653         default:  return ciConstant();
 654         }
 655       }
 656       ciObject* ciobj = get_object(obj);
 657       if (ciobj->is_array()) {
 658         return ciConstant(T_ARRAY, ciobj);
 659       } else {
 660         assert(ciobj->is_instance(), "should be an instance");
 661         return ciConstant(T_OBJECT, ciobj);
 662       }
 663     }
 664   }
 665   constantTag tag = cpool->tag_at(index);
 666   if (tag.is_int()) {
 667     return ciConstant(T_INT, (jint)cpool->int_at(index));
 668   } else if (tag.is_long()) {
 669     return ciConstant((jlong)cpool->long_at(index));
 670   } else if (tag.is_float()) {
 671     return ciConstant((jfloat)cpool->float_at(index));
 672   } else if (tag.is_double()) {
 673     return ciConstant((jdouble)cpool->double_at(index));
 674   } else if (tag.is_string()) {
 675     oop string = NULL;
 676     assert(cache_index >= 0, "should have a cache index");
 677     if (cpool->is_pseudo_string_at(index)) {
 678       string = cpool->pseudo_string_at(index, cache_index);
 679     } else {
 680       string = cpool->string_at(index, cache_index, THREAD);
 681       if (HAS_PENDING_EXCEPTION) {
 682         CLEAR_PENDING_EXCEPTION;
 683         record_out_of_memory_failure();
 684         return ciConstant();
 685       }
 686     }
 687     ciObject* constant = get_object(string);
 688     if (constant->is_array()) {
 689       return ciConstant(T_ARRAY, constant);
 690     } else {
 691       assert (constant->is_instance(), "must be an instance, or not? ");
 692       return ciConstant(T_OBJECT, constant);
 693     }
 694   } else if (tag.is_klass() || tag.is_unresolved_klass()) {
 695     // 4881222: allow ldc to take a class type
 696     ciKlass* klass = get_klass_by_index_impl(cpool, index, ignore_will_link, accessor);
 697     if (HAS_PENDING_EXCEPTION) {
 698       CLEAR_PENDING_EXCEPTION;
 699       record_out_of_memory_failure();
 700       return ciConstant();
 701     }
 702     assert (klass->is_instance_klass() || klass->is_array_klass(),
 703             "must be an instance or array klass ");
 704     return ciConstant(T_OBJECT, klass->java_mirror());
 705   } else if (tag.is_method_type()) {
 706     // must execute Java code to link this CP entry into cache[i].f1
 707     ciSymbol* signature = get_symbol(cpool->method_type_signature_at(index));
 708     ciObject* ciobj = get_unloaded_method_type_constant(signature);
 709     return ciConstant(T_OBJECT, ciobj);
 710   } else if (tag.is_method_handle()) {
 711     // must execute Java code to link this CP entry into cache[i].f1
 712     int ref_kind        = cpool->method_handle_ref_kind_at(index);
 713     int callee_index    = cpool->method_handle_klass_index_at(index);
 714     ciKlass* callee     = get_klass_by_index_impl(cpool, callee_index, ignore_will_link, accessor);
 715     ciSymbol* name      = get_symbol(cpool->method_handle_name_ref_at(index));
 716     ciSymbol* signature = get_symbol(cpool->method_handle_signature_ref_at(index));
 717     ciObject* ciobj     = get_unloaded_method_handle_constant(callee, name, signature, ref_kind);
 718     return ciConstant(T_OBJECT, ciobj);
 719   } else if (tag.is_dynamic_constant()) {
 720     return ciConstant();
 721   } else {
 722     ShouldNotReachHere();
 723     return ciConstant();
 724   }
 725 }

 771 // ------------------------------------------------------------------
 772 // ciEnv::lookup_method
 773 //
 774 // Perform an appropriate method lookup based on accessor, holder,
 775 // name, signature, and bytecode.
 776 Method* ciEnv::lookup_method(ciInstanceKlass* accessor,
 777                              ciKlass*         holder,
 778                              Symbol*          name,
 779                              Symbol*          sig,
 780                              Bytecodes::Code  bc,
 781                              constantTag      tag) {
 782   // Accessibility checks are performed in ciEnv::get_method_by_index_impl.
 783   assert(check_klass_accessibility(accessor, holder->get_Klass()), "holder not accessible");
 784 
 785   InstanceKlass* accessor_klass = accessor->get_instanceKlass();
 786   Klass* holder_klass = holder->get_Klass();
 787   Method* dest_method;
 788   LinkInfo link_info(holder_klass, name, sig, accessor_klass, LinkInfo::needs_access_check, tag);
 789   switch (bc) {
 790   case Bytecodes::_invokestatic:
 791     dest_method =
 792       LinkResolver::resolve_static_call_or_null(link_info);
 793     break;
 794   case Bytecodes::_invokespecial:
 795     dest_method =
 796       LinkResolver::resolve_special_call_or_null(link_info);
 797     break;
 798   case Bytecodes::_invokeinterface:
 799     dest_method =
 800       LinkResolver::linktime_resolve_interface_method_or_null(link_info);
 801     break;
 802   case Bytecodes::_invokevirtual:
 803     dest_method =
 804       LinkResolver::linktime_resolve_virtual_method_or_null(link_info);
 805     break;
 806   default: ShouldNotReachHere();
 807   }
 808 
 809   return dest_method;
 810 }

 813 // ------------------------------------------------------------------
 814 // ciEnv::get_method_by_index_impl
 815 ciMethod* ciEnv::get_method_by_index_impl(const constantPoolHandle& cpool,
 816                                           int index, Bytecodes::Code bc,
 817                                           ciInstanceKlass* accessor) {
 818   assert(cpool.not_null(), "need constant pool");
 819   assert(accessor != NULL, "need origin of access");
 820   if (bc == Bytecodes::_invokedynamic) {
 821     ConstantPoolCacheEntry* cpce = cpool->invokedynamic_cp_cache_entry_at(index);
 822     bool is_resolved = !cpce->is_f1_null();
 823     // FIXME: code generation could allow for null (unlinked) call site
 824     // The call site could be made patchable as follows:
 825     // Load the appendix argument from the constant pool.
 826     // Test the appendix argument and jump to a known deopt routine if it is null.
 827     // Jump through a patchable call site, which is initially a deopt routine.
 828     // Patch the call site to the nmethod entry point of the static compiled lambda form.
 829     // As with other two-component call sites, both values must be independently verified.
 830 
 831     if (is_resolved) {
 832       // Get the invoker Method* from the constant pool.
 833       // (The appendix argument, if any, will be noted in the method's signature.)
 834       Method* adapter = cpce->f1_as_method();
 835       return get_method(adapter);
 836     }
 837 
 838     // Fake a method that is equivalent to a declared method.
 839     ciInstanceKlass* holder    = get_instance_klass(SystemDictionary::MethodHandle_klass());
 840     ciSymbol*        name      = ciSymbol::invokeBasic_name();
 841     ciSymbol*        signature = get_symbol(cpool->signature_ref_at(index));
 842     return get_unloaded_method(holder, name, signature, accessor);
 843   } else {
 844     const int holder_index = cpool->klass_ref_index_at(index);
 845     bool holder_is_accessible;
 846     ciKlass* holder = get_klass_by_index_impl(cpool, holder_index, holder_is_accessible, accessor);
 847 
 848     // Get the method's name and signature.
 849     Symbol* name_sym = cpool->name_ref_at(index);
 850     Symbol* sig_sym  = cpool->signature_ref_at(index);
 851 
 852     if (cpool->has_preresolution()
 853         || ((holder == ciEnv::MethodHandle_klass() || holder == ciEnv::VarHandle_klass()) &&
 854             MethodHandles::is_signature_polymorphic_name(holder->get_Klass(), name_sym))) {
 855       // Short-circuit lookups for JSR 292-related call sites.
 856       // That is, do not rely only on name-based lookups, because they may fail
 857       // if the names are not resolvable in the boot class loader (7056328).
 858       switch (bc) {
 859       case Bytecodes::_invokevirtual:
 860       case Bytecodes::_invokeinterface:
 861       case Bytecodes::_invokespecial:
 862       case Bytecodes::_invokestatic:
 863         {
 864           Method* m = ConstantPool::method_at_if_loaded(cpool, index);
 865           if (m != NULL) {
 866             return get_method(m);
 867           }
 868         }
 869         break;
 870       default:
 871         break;
 872       }
 873     }
 874 
 875     if (holder_is_accessible) {  // Our declared holder is loaded.
 876       constantTag tag = cpool->tag_ref_at(index);
 877       assert(accessor->get_instanceKlass() == cpool->pool_holder(), "not the pool holder?");
 878       Method* m = lookup_method(accessor, holder, name_sym, sig_sym, bc, tag);
 879       if (m != NULL &&
 880           (bc == Bytecodes::_invokestatic
 881            ?  m->method_holder()->is_not_initialized()
 882            : !m->method_holder()->is_loaded())) {
 883         m = NULL;
 884       }
 885 #ifdef ASSERT
 886       if (m != NULL && ReplayCompiles && !ciReplay::is_loaded(m)) {
 887         m = NULL;
 888       }
 889 #endif
 890       if (m != NULL) {
 891         // We found the method.
 892         return get_method(m);
 893       }
 894     }
 895 
 896     // Either the declared holder was not loaded, or the method could
 897     // not be found.  Create a dummy ciMethod to represent the failed
 898     // lookup.
 899     ciSymbol* name      = get_symbol(name_sym);
 900     ciSymbol* signature = get_symbol(sig_sym);
 901     return get_unloaded_method(holder, name, signature, accessor);
 902   }
 903 }

 906 // ------------------------------------------------------------------
 907 // ciEnv::get_instance_klass_for_declared_method_holder
 908 ciInstanceKlass* ciEnv::get_instance_klass_for_declared_method_holder(ciKlass* method_holder) {
 909   // For the case of <array>.clone(), the method holder can be a ciArrayKlass
 910   // instead of a ciInstanceKlass.  For that case simply pretend that the
 911   // declared holder is Object.clone since that's where the call will bottom out.
 912   // A more correct fix would trickle out through many interfaces in CI,
 913   // requiring ciInstanceKlass* to become ciKlass* and many more places would
 914   // require checks to make sure the expected type was found.  Given that this
 915   // only occurs for clone() the more extensive fix seems like overkill so
 916   // instead we simply smear the array type into Object.
 917   guarantee(method_holder != NULL, "no method holder");
 918   if (method_holder->is_instance_klass()) {
 919     return method_holder->as_instance_klass();
 920   } else if (method_holder->is_array_klass()) {
 921     return current()->Object_klass();
 922   } else {
 923     ShouldNotReachHere();
 924   }
 925   return NULL;
 926 }

 960 // ------------------------------------------------------------------
 961 // ciEnv::validate_compile_task_dependencies
 962 //
 963 // Check for changes during compilation (e.g. class loads, evolution,
 964 // breakpoints, call site invalidation).
 965 void ciEnv::validate_compile_task_dependencies(ciMethod* target) {
 966   if (failing())  return;  // no need for further checks
 967 
 968   Dependencies::DepType result = dependencies()->validate_dependencies(_task);
 969   if (result != Dependencies::end_marker) {
 970     if (result == Dependencies::call_site_target_value) {
 971       _inc_decompile_count_on_failure = false;
 972       record_failure("call site target change");
 973     } else if (Dependencies::is_klass_type(result)) {
 974       record_failure("concurrent class loading");
 975     } else {
 976       record_failure("invalid non-klass dependency");
 977     }
 978   }
 979 }

 981 // ------------------------------------------------------------------
 982 // ciEnv::register_method
 983 void ciEnv::register_method(ciMethod* target,
 984                             int entry_bci,
 985                             CodeOffsets* offsets,
 986                             int orig_pc_offset,
 987                             CodeBuffer* code_buffer,
 988                             int frame_words,
 989                             OopMapSet* oop_map_set,
 990                             ExceptionHandlerTable* handler_table,
 991                             ImplicitExceptionTable* inc_table,
 992                             AbstractCompiler* compiler,
 993                             bool has_unsafe_access,
 994                             bool has_wide_vectors,
 995                             RTMState  rtm_state) {
 996   VM_ENTRY_MARK;
 997   nmethod* nm = NULL;
 998   {
 999     // To prevent compile queue updates.
1000     MutexLocker locker(THREAD, MethodCompileQueue_lock);
1001 
1002     // Prevent SystemDictionary::add_to_hierarchy from running
1003     // and invalidating our dependencies until we install this method.
1004     // No safepoints are allowed. Otherwise, class redefinition can occur in between.
1005     MutexLocker ml(Compile_lock);
1006     NoSafepointVerifier nsv;
1007 
1008     // Change in Jvmti state may invalidate compilation.
1009     if (!failing() && jvmti_state_changed()) {
1010       record_failure("Jvmti state change invalidated dependencies");
1011     }
1012 
1013     // Change in DTrace flags may invalidate compilation.
1014     if (!failing() &&
1015         ( (!dtrace_extended_probes() && ExtendedDTraceProbes) ||
1016           (!dtrace_method_probes() && DTraceMethodProbes) ||
1017           (!dtrace_alloc_probes() && DTraceAllocProbes) )) {
1018       record_failure("DTrace flags change invalidated dependencies");
1019     }
1020 
1021     if (!failing() && target->needs_clinit_barrier() &&
1022         target->holder()->is_in_error_state()) {
1023       record_failure("method holder is in error state");
1024     }
1025 
1026     if (!failing()) {
1027       if (log() != NULL) {
1028         // Log the dependencies which this compilation declares.
1029         dependencies()->log_all_dependencies();
1030       }
1031 
1032       // Encode the dependencies now, so we can check them right away.
1033       dependencies()->encode_content_bytes();
1034 
1035       // Check for {class loads, evolution, breakpoints, ...} during compilation
1036       validate_compile_task_dependencies(target);
1037     }
1038 
1039     methodHandle method(THREAD, target->get_Method());
1040 
1041 #if INCLUDE_RTM_OPT
1042     if (!failing() && (rtm_state != NoRTM) &&
1043         (method()->method_data() != NULL) &&
1044         (method()->method_data()->rtm_state() != rtm_state)) {
1045       // Preemptive decompile if rtm state was changed.
1046       record_failure("RTM state change invalidated rtm code");
1047     }
1048 #endif
1049 
1050     if (failing()) {
1051       // While not a true deoptimization, it is a preemptive decompile.
1052       MethodData* mdo = method()->method_data();
1053       if (mdo != NULL && _inc_decompile_count_on_failure) {
1054         mdo->inc_decompile_count();
1055       }
1056 
1057       // All buffers in the CodeBuffer are allocated in the CodeCache.
1058       // If the code buffer is created on each compile attempt
1059       // as in C2, then it must be freed.
1060       code_buffer->free_blob();
1061       return;
1062     }
1063 
1064     assert(offsets->value(CodeOffsets::Deopt) != -1, "must have deopt entry");
1065     assert(offsets->value(CodeOffsets::Exceptions) != -1, "must have exception entry");
1066 
1067     nm =  nmethod::new_nmethod(method,
1068                                compile_id(),
1069                                entry_bci,
1070                                offsets,
1071                                orig_pc_offset,
1072                                debug_info(), dependencies(), code_buffer,
1073                                frame_words, oop_map_set,
1074                                handler_table, inc_table,
1075                                compiler, task()->comp_level());
1076 
1077     // Free codeBlobs
1078     code_buffer->free_blob();
1079 
1080     if (nm != NULL) {
1081       nm->set_has_unsafe_access(has_unsafe_access);
1082       nm->set_has_wide_vectors(has_wide_vectors);
1083 #if INCLUDE_RTM_OPT
1084       nm->set_rtm_state(rtm_state);
1085 #endif
1086 
1087       // Record successful registration.
1088       // (Put nm into the task handle *before* publishing to the Java heap.)
1089       if (task() != NULL) {
1090         task()->set_code(nm);
1091       }
1092 
1093       if (entry_bci == InvocationEntryBci) {
1094         if (TieredCompilation) {
1095           // If there is an old version we're done with it
1096           CompiledMethod* old = method->code();
1097           if (TraceMethodReplacement && old != NULL) {
1098             ResourceMark rm;
1099             char *method_name = method->name_and_sig_as_C_string();
1100             tty->print_cr("Replacing method %s", method_name);
1101           }
1102           if (old != NULL) {
1103             old->make_not_used();
1104           }
1105         }
1106 
1107         LogTarget(Info, nmethod, install) lt;
1108         if (lt.is_enabled()) {
1109           ResourceMark rm;
1110           char *method_name = method->name_and_sig_as_C_string();
1111           lt.print("Installing method (%d) %s ",
1112                     task()->comp_level(), method_name);
1113         }
1114         // Allow the code to be executed
1115         MutexLocker ml(CompiledMethod_lock, Mutex::_no_safepoint_check_flag);
1116         if (nm->make_in_use()) {
1117           method->set_code(method, nm);
1118         }
1119       } else {
1120         LogTarget(Info, nmethod, install) lt;
1121         if (lt.is_enabled()) {
1122           ResourceMark rm;
1123           char *method_name = method->name_and_sig_as_C_string();
1124           lt.print("Installing osr method (%d) %s @ %d",
1125                     task()->comp_level(), method_name, entry_bci);
1126         }
1127         MutexLocker ml(CompiledMethod_lock, Mutex::_no_safepoint_check_flag);
1128         if (nm->make_in_use()) {
1129           method->method_holder()->add_osr_nmethod(nm);
1130         }
1131       }
1132     }
1133   }  // safepoints are allowed again
1134 
1135   if (nm != NULL) {
1136     // JVMTI -- compiled method notification (must be done outside lock)
1137     nm->post_compiled_method_load_event();
1138   } else {
1139     // The CodeCache is full.
1140     record_failure("code cache is full");
1141   }
1142 }

1194 // ------------------------------------------------------------------
1195 // ciEnv::record_method_not_compilable()
1196 void ciEnv::record_method_not_compilable(const char* reason, bool all_tiers) {
1197   int new_compilable =
1198     all_tiers ? MethodCompilable_never : MethodCompilable_not_at_tier ;
1199 
1200   // Only note transitions to a worse state
1201   if (new_compilable > _compilable) {
1202     if (log() != NULL) {
1203       if (all_tiers) {
1204         log()->elem("method_not_compilable");
1205       } else {
1206         log()->elem("method_not_compilable_at_tier level='%d'",
1207                     current()->task()->comp_level());
1208       }
1209     }
1210     _compilable = new_compilable;
1211 
1212     // Reset failure reason; this one is more important.
1213     _failure_reason = NULL;
1214     record_failure(reason);
1215   }
1216 }

1233 // Safe to call from VM error reporter.
1234 
1235 void ciEnv::dump_compile_data(outputStream* out) {
1236   CompileTask* task = this->task();
1237   if (task) {
1238     Method* method = task->method();
1239     int entry_bci = task->osr_bci();
1240     int comp_level = task->comp_level();
1241     out->print("compile %s %s %s %d %d",
1242                method->klass_name()->as_quoted_ascii(),
1243                method->name()->as_quoted_ascii(),
1244                method->signature()->as_quoted_ascii(),
1245                entry_bci, comp_level);
1246     if (compiler_data() != NULL) {
1247       if (is_c2_compile(comp_level)) {
1248 #ifdef COMPILER2
1249         // Dump C2 inlining data.
1250         ((Compile*)compiler_data())->dump_inline_data(out);
1251 #endif
1252       } else if (is_c1_compile(comp_level)) {
1253 #ifdef COMPILER1
1254         // Dump C1 inlining data.
1255         ((Compilation*)compiler_data())->dump_inline_data(out);
1256 #endif
1257       }
1258     }
1259     out->cr();
1260   }
1261 }

1296         dump_replay_data(&replay_data_stream);
1297         tty->print_cr("# Compiler replay data is saved as: %s", buffer);

1300       }
1301     }
