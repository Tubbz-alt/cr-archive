 34 // Clones the values type to handle control flow merges involving multiple value types.
 35 // The inputs are replaced by PhiNodes to represent the merged values for the given region.
 36 ValueTypeBaseNode* ValueTypeBaseNode::clone_with_phis(PhaseGVN* gvn, Node* region) {
 37   assert(!has_phi_inputs(region), "already cloned with phis");
 38   ValueTypeBaseNode* vt = clone()->as_ValueTypeBase();
 39 
 40   // Create a PhiNode for merging the oop values
 41   const Type* phi_type = Type::get_const_type(value_klass());
 42   PhiNode* oop = PhiNode::make(region, vt->get_oop(), phi_type);
 43   gvn->set_type(oop, phi_type);
 44   vt->set_oop(oop);
 45 
 46   // Create a PhiNode each for merging the field values
 47   for (uint i = 0; i < vt->field_count(); ++i) {
 48     ciType* type = vt->field_type(i);
 49     Node*  value = vt->field_value(i);
 50     if (type->is_valuetype() && value->isa_ValueType()) {
 51       // Handle flattened value type fields recursively
 52       value = value->as_ValueType()->clone_with_phis(gvn, region);
 53     } else {
 54       phi_type = Type::get_const_type(type);
 55       value = PhiNode::make(region, value, phi_type);
 56       gvn->set_type(value, phi_type);
 57     }
 58     vt->set_field_value(i, value);
 59   }
 60   gvn->set_type(vt, vt->bottom_type());
 61   return vt;
 62 }

 64 // Checks if the inputs of the ValueBaseTypeNode were replaced by PhiNodes
 65 // for the given region (see ValueBaseTypeNode::clone_with_phis).
 66 bool ValueTypeBaseNode::has_phi_inputs(Node* region) {
 67   // Check oop input
 68   bool result = get_oop()->is_Phi() && get_oop()->as_Phi()->region() == region;
 69 #ifdef ASSERT
 70   if (result) {
 71     // Check all field value inputs for consistency
 72     for (uint i = Oop; i < field_count(); ++i) {
 73       Node* n = in(i);
 74       if (n->is_ValueTypeBase()) {
 75         assert(n->as_ValueTypeBase()->has_phi_inputs(region), "inconsistent phi inputs");
 76       } else {
 77         assert(n->is_Phi() && n->as_Phi()->region() == region, "inconsistent phi inputs");
 78       }
 79     }
 80   }
 81 #endif
 82   return result;
 83 }

 85 // Merges 'this' with 'other' by updating the input PhiNodes added by 'clone_with_phis'
 86 ValueTypeBaseNode* ValueTypeBaseNode::merge_with(PhaseGVN* gvn, const ValueTypeBaseNode* other, int pnum, bool transform) {
 87   // Merge oop inputs
 88   PhiNode* phi = get_oop()->as_Phi();
 89   phi->set_req(pnum, other->get_oop());
 90   if (transform) {
 91     set_oop(gvn->transform(phi));
 92     gvn->record_for_igvn(phi);
 93   }
 94   // Merge field values
 95   for (uint i = 0; i < field_count(); ++i) {
 96     Node* val1 =        field_value(i);
 97     Node* val2 = other->field_value(i);
 98     if (val1->is_ValueType()) {
 99       val1->as_ValueType()->merge_with(gvn, val2->as_ValueType(), pnum, transform);
100     } else {
101       assert(val1->is_Phi(), "must be a phi node");
102       assert(!val2->is_ValueType(), "inconsistent merge values");
103       val1->set_req(pnum, val2);
104     }
105     if (transform) {
106       set_field_value(i, gvn->transform(val1));
107       gvn->record_for_igvn(val1);
108     }
109   }
110   return this;
111 }

113 // Adds a new merge path to a valuetype node with phi inputs
114 void ValueTypeBaseNode::add_new_path(Node* region) {
115   assert(has_phi_inputs(region), "must have phi inputs");
116 
117   PhiNode* phi = get_oop()->as_Phi();
118   phi->add_req(NULL);
119   assert(phi->req() == region->req(), "must be same size as region");
120 
121   for (uint i = 0; i < field_count(); ++i) {
122     Node* val = field_value(i);
123     if (val->is_ValueType()) {
124       val->as_ValueType()->add_new_path(region);
125     } else {
126       val->as_Phi()->add_req(NULL);
127       assert(val->req() == region->req(), "must be same size as region");
128     }
129   }
130 }

137 // Get the value of the field at the given offset.
138 // If 'recursive' is true, flattened value type fields will be resolved recursively.
139 Node* ValueTypeBaseNode::field_value_by_offset(int offset, bool recursive) const {
140   // If the field at 'offset' belongs to a flattened value type field, 'index' refers to the
141   // corresponding ValueTypeNode input and 'sub_offset' is the offset in flattened value type.
142   int index = value_klass()->field_index_by_offset(offset);
143   int sub_offset = offset - field_offset(index);
144   Node* value = field_value(index);
145   assert(value != NULL, "field value not found");
146   if (recursive && value->is_ValueType()) {
147     ValueTypeNode* vt = value->as_ValueType();
148     if (field_is_flattened(index)) {
149       // Flattened value type field
150       sub_offset += vt->value_klass()->first_field_offset(); // Add header size
151       return vt->field_value_by_offset(sub_offset, recursive);
152     } else {
153       assert(sub_offset == 0, "should not have a sub offset");
154       return vt;
155     }
156   }
157   assert(!(recursive && value->is_ValueType()), "should not be a value type");
158   assert(sub_offset == 0, "offset mismatch");
159   return value;
160 }

202 int ValueTypeBaseNode::make_scalar_in_safepoint(PhaseIterGVN* igvn, Unique_Node_List& worklist, SafePointNode* sfpt) {
203   ciValueKlass* vk = value_klass();
204   uint nfields = vk->nof_nonstatic_fields();
205   JVMState* jvms = sfpt->jvms();
206   int start = jvms->debug_start();
207   int end   = jvms->debug_end();
208   // Replace safepoint edge by SafePointScalarObjectNode and add field values
209   assert(jvms != NULL, "missing JVMS");
210   uint first_ind = (sfpt->req() - jvms->scloff());
211   SafePointScalarObjectNode* sobj = new SafePointScalarObjectNode(value_ptr(),
212 #ifdef ASSERT
213                                                                   NULL,
214 #endif
215                                                                   first_ind, nfields);
216   sobj->init_req(0, igvn->C->root());
217   // Iterate over the value type fields in order of increasing
218   // offset and add the field values to the safepoint.
219   for (uint j = 0; j < nfields; ++j) {
220     int offset = vk->nonstatic_field_at(j)->offset();
221     Node* value = field_value_by_offset(offset, true /* include flattened value type fields */);
222     if (value->is_ValueType()) {
223       // Add value type field to the worklist to process later
224       worklist.push(value);
225     }
226     sfpt->add_req(value);
227   }
228   jvms->set_endoff(sfpt->req());
229   sobj = igvn->transform(sobj)->as_SafePointScalarObject();
230   igvn->rehash_node_delayed(sfpt);
231   return sfpt->replace_edges_in_range(this, sobj, start, end);
232 }

234 void ValueTypeBaseNode::make_scalar_in_safepoints(PhaseIterGVN* igvn) {
235   // Process all safepoint uses and scalarize value type
236   Unique_Node_List worklist;
237   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
238     SafePointNode* sfpt = fast_out(i)->isa_SafePoint();
239     if (sfpt != NULL && !sfpt->is_CallLeaf() && (!sfpt->is_Call() || sfpt->as_Call()->has_debug_use(this))) {
240       int nb = 0;
241       if (is_allocated(igvn) && get_oop()->is_Con()) {
242         // Value type is allocated with a constant oop, link it directly
243         nb = sfpt->replace_edges_in_range(this, get_oop(), sfpt->jvms()->debug_start(), sfpt->jvms()->debug_end());
244         igvn->rehash_node_delayed(sfpt);
245       } else {
246         nb = make_scalar_in_safepoint(igvn, worklist, sfpt);
247       }
248       --i; imax -= nb;
249     }
250   }
251   // Now scalarize non-flattened fields
252   for (uint i = 0; i < worklist.size(); ++i) {
253     Node* vt = worklist.at(i);
254     vt->as_ValueType()->make_scalar_in_safepoints(igvn);
255   }
256 }

258 const TypePtr* ValueTypeBaseNode::field_adr_type(Node* base, int offset, ciInstanceKlass* holder, DecoratorSet decorators, PhaseGVN& gvn) const {
259   const TypeAryPtr* ary_type = gvn.type(base)->isa_aryptr();
260   const TypePtr* adr_type = NULL;
261   bool is_array = ary_type != NULL;
262   if ((decorators & C2_MISMATCHED) != 0) {
263     adr_type = TypeRawPtr::BOTTOM;
264   } else if (is_array) {
265     // In the case of a flattened value type array, each field has its own slice
266     adr_type = ary_type->with_field_offset(offset)->add_offset(Type::OffsetBot);
267   } else {
268     ciField* field = holder->get_field_by_offset(offset, false);
269     assert(field != NULL, "field not found");
270     adr_type = gvn.C->alias_type(field)->adr_type();
271   }
272   return adr_type;
273 }

275 void ValueTypeBaseNode::load(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) {
276   // Initialize the value type by loading its field values from
277   // memory and adding the values as input edges to the node.
278   for (uint i = 0; i < field_count(); ++i) {
279     int offset = holder_offset + field_offset(i);
280     Node* value = NULL;
281     ciType* ft = field_type(i);
282     bool is_flattenable = field_is_flattenable(i);
283     if (field_is_flattened(i)) {
284       // Recursively load the flattened value type field
285       value = ValueTypeNode::make_from_flattened(kit, ft->as_value_klass(), base, ptr, holder, offset, decorators);
286     } else {
287       const TypeOopPtr* oop_ptr = kit->gvn().type(base)->isa_oopptr();
288       bool is_array = (oop_ptr->isa_aryptr() != NULL);
289       if (base->is_Con() && !is_array) {
290         // If the oop to the value type is constant (static final field), we can
291         // also treat the fields as constants because the value type is immutable.
292         ciObject* constant_oop = oop_ptr->const_oop();
293         ciField* field = holder->get_field_by_offset(offset, false);
294         assert(field != NULL, "field not found");
295         ciConstant constant = constant_oop->as_instance()->field_value(field);
296         const Type* con_type = Type::make_from_constant(constant, /*require_const=*/ true);
297         assert(con_type != NULL, "type not found");
298         value = kit->gvn().transform(kit->makecon(con_type));
299         if (ft->is_valuetype() && !constant.as_object()->is_null_object()) {
300           // Null-free, treat as flattenable
301           is_flattenable = true;
302         }
303       } else {
304         // Load field value from memory
305         const TypePtr* adr_type = field_adr_type(base, offset, holder, decorators, kit->gvn());
306         Node* adr = kit->basic_plus_adr(base, ptr, offset);
307         BasicType bt = type2field[ft->basic_type()];
308         assert(is_java_primitive(bt) || adr->bottom_type()->is_ptr_to_narrowoop() == UseCompressedOops, "inconsistent");
309         const Type* val_type = Type::get_const_type(ft);
310         if (is_array) {
311           decorators |= IS_ARRAY;
312         }
313         value = kit->access_load_at(base, adr, adr_type, val_type, bt, decorators);
314       }
315       if (is_flattenable) {
316         // Loading a non-flattened but flattenable value type from memory
317         if (ft->as_value_klass()->is_scalarizable()) {
318           value = ValueTypeNode::make_from_oop(kit, value, ft->as_value_klass());
319         } else {
320           value = kit->null2default(value, ft->as_value_klass());
321         }
322       }
323     }
324     set_field_value(i, value);
325   }
326 }

328 void ValueTypeBaseNode::store_flattened(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) const {
329   // The value type is embedded into the object without an oop header. Subtract the
330   // offset of the first field to account for the missing header when storing the values.
331   if (holder == NULL) {
332     holder = value_klass();
333   }
334   holder_offset -= value_klass()->first_field_offset();
335   store(kit, base, ptr, holder, holder_offset, decorators);
336 }

338 void ValueTypeBaseNode::store(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) const {
339   // Write field values to memory
340   for (uint i = 0; i < field_count(); ++i) {
341     int offset = holder_offset + field_offset(i);
342     Node* value = field_value(i);
343     ciType* ft = field_type(i);
344     if (field_is_flattened(i)) {
345       // Recursively store the flattened value type field
346       if (!value->is_ValueType()) {
347         assert(!kit->gvn().type(value)->maybe_null(), "should never be null");
348         value = ValueTypeNode::make_from_oop(kit, value, ft->as_value_klass());
349       }
350       value->as_ValueType()->store_flattened(kit, base, ptr, holder, offset, decorators);
351     } else {
352       // Store field value to memory
353       const TypePtr* adr_type = field_adr_type(base, offset, holder, decorators, kit->gvn());
354       Node* adr = kit->basic_plus_adr(base, ptr, offset);
355       BasicType bt = type2field[ft->basic_type()];
356       assert(is_java_primitive(bt) || adr->bottom_type()->is_ptr_to_narrowoop() == UseCompressedOops, "inconsistent");
357       const Type* val_type = Type::get_const_type(ft);
358       const TypeAryPtr* ary_type = kit->gvn().type(base)->isa_aryptr();
359       if (ary_type != NULL) {
360         decorators |= IS_ARRAY;
361       }
362       kit->access_store_at(base, adr, adr_type, value, val_type, bt, decorators);
363     }
364   }
365 }

367 ValueTypeBaseNode* ValueTypeBaseNode::allocate(GraphKit* kit, bool safe_for_replace) {
368   // Check if value type is already allocated
369   Node* null_ctl = kit->top();
370   Node* not_null_oop = kit->null_check_oop(get_oop(), &null_ctl);
371   if (null_ctl->is_top()) {
372     // Value type is allocated
373     return this;
374   }
375   assert(!is_allocated(&kit->gvn()), "should not be allocated");
376   RegionNode* region = new RegionNode(3);
377 
378   // Oop is non-NULL, use it
379   region->init_req(1, kit->control());
380   PhiNode* oop = PhiNode::make(region, not_null_oop, value_ptr());
381   PhiNode* io  = PhiNode::make(region, kit->i_o(), Type::ABIO);
382   PhiNode* mem = PhiNode::make(region, kit->merged_memory(), Type::MEMORY, TypePtr::BOTTOM);
383 
384   int bci = kit->bci();
385   bool reexecute = kit->jvms()->should_reexecute();
386   {
387     // Oop is NULL, allocate and initialize buffer
388     PreserveJVMState pjvms(kit);
389     // Propagate re-execution state and bci
390     kit->set_bci(bci);
391     kit->jvms()->set_bci(bci);
392     kit->jvms()->set_should_reexecute(reexecute);
393     kit->set_control(null_ctl);
394     kit->kill_dead_locals();
395     ciValueKlass* vk = value_klass();
396     Node* klass_node = kit->makecon(TypeKlassPtr::make(vk));
397     Node* alloc_oop  = kit->new_instance(klass_node, NULL, NULL, /* deoptimize_on_exception */ true, this);
398     store(kit, alloc_oop, alloc_oop, vk, 0);
399 
400     // Do not let stores that initialize this buffer be reordered with a subsequent
401     // store that would make this buffer accessible by other threads.
402     // FIXME: coordinate with ready_to_publish(kit, alloc_oop)
403     AllocateNode* alloc = AllocateNode::Ideal_allocation(alloc_oop, &kit->gvn());
404     assert(alloc != NULL, "must have an allocation node");
405     kit->insert_mem_bar(Op_MemBarStoreStore, alloc->proj_out_or_null(AllocateNode::RawAddress));
406 
407     region->init_req(2, kit->control());
408     oop   ->init_req(2, alloc_oop);
409     io    ->init_req(2, kit->i_o());
410     mem   ->init_req(2, kit->merged_memory());
411   }
412 
413   // Update GraphKit
414   kit->set_control(kit->gvn().transform(region));
415   kit->set_i_o(kit->gvn().transform(io));
416   kit->set_all_memory(kit->gvn().transform(mem));
417   kit->record_for_igvn(region);
418   kit->record_for_igvn(oop);
419   kit->record_for_igvn(io);
420   kit->record_for_igvn(mem);
421 
422   // Use cloned ValueTypeNode to propagate oop from now on
423   Node* res_oop = kit->gvn().transform(oop);
424   ValueTypeBaseNode* vt = clone()->as_ValueTypeBase();
425   vt->set_oop(res_oop);
426   vt = kit->gvn().transform(vt)->as_ValueTypeBase();
427   if (safe_for_replace) {
428     kit->replace_in_map(this, vt);
429   }
430   // ValueTypeNode::remove_redundant_allocations piggybacks on split if.
431   // Make sure it gets a chance to remove this allocation.
432   kit->C->set_has_split_ifs(true);
433   assert(vt->is_allocated(&kit->gvn()), "must be allocated");
434   return vt;
435 }

437 bool ValueTypeBaseNode::is_allocated(PhaseGVN* phase) const {
438   Node* oop = get_oop();
439   const Type* oop_type = (phase != NULL) ? phase->type(oop) : oop->bottom_type();
440   return !oop_type->maybe_null();
441 }

443 // When a call returns multiple values, it has several result
444 // projections, one per field. Replacing the result of the call by a
445 // value type node (after late inlining) requires that for each result
446 // projection, we find the corresponding value type field.
447 void ValueTypeBaseNode::replace_call_results(GraphKit* kit, Node* call, Compile* C) {
448   ciValueKlass* vk = value_klass();
449   for (DUIterator_Fast imax, i = call->fast_outs(imax); i < imax; i++) {
450     ProjNode* pn = call->fast_out(i)->as_Proj();
451     uint con = pn->_con;
452     if (con >= TypeFunc::Parms+1) {
453       uint field_nb = con - (TypeFunc::Parms+1);
454       int extra = 0;
455       for (uint j = 0; j < field_nb - extra; j++) {
456         ciField* f = vk->nonstatic_field_at(j);
457         BasicType bt = f->type()->basic_type();
458         if (bt == T_LONG || bt == T_DOUBLE) {
459           extra++;
460         }
461       }
462       ciField* f = vk->nonstatic_field_at(field_nb - extra);
463       Node* field = field_value_by_offset(f->offset(), true);
464       if (field->is_ValueType()) {
465         assert(field->as_ValueType()->is_allocated(&kit->gvn()), "must be allocated");
466         field = field->as_ValueType()->get_oop();
467       }
468       C->gvn_replace_by(pn, field);
469       C->initial_gvn()->hash_delete(pn);
470       pn->set_req(0, C->top());
471       --i; --imax;
472     }
473   }
474 }

476 Node* ValueTypeBaseNode::allocate_fields(GraphKit* kit) {
477   ValueTypeBaseNode* vt = clone()->as_ValueTypeBase();
478   for (uint i = 0; i < field_count(); i++) {
479      ValueTypeNode* value = field_value(i)->isa_ValueType();
480      if (field_is_flattened(i)) {
481        // Flattened value type field
482        vt->set_field_value(i, value->allocate_fields(kit));
483      } else if (value != NULL) {
484        // Non-flattened value type field
485        vt->set_field_value(i, value->allocate(kit));
486      }
487   }
488   vt = kit->gvn().transform(vt)->as_ValueTypeBase();
489   kit->replace_in_map(this, vt);
490   return vt;
491 }

503 ValueTypeNode* ValueTypeNode::make_default(PhaseGVN& gvn, ciValueKlass* vk) {
504   // Create a new ValueTypeNode with default values
505   ValueTypeNode* vt = new ValueTypeNode(vk, default_oop(gvn, vk));
506   for (uint i = 0; i < vt->field_count(); ++i) {
507     ciType* field_type = vt->field_type(i);
508     Node* value = NULL;
509     if (field_type->is_valuetype() && vt->field_is_flattenable(i)) {
510       ciValueKlass* field_klass = field_type->as_value_klass();
511       if (field_klass->is_scalarizable() || vt->field_is_flattened(i)) {
512         value = ValueTypeNode::make_default(gvn, field_klass);
513       } else {
514         value = default_oop(gvn, field_klass);
515       }
516     } else {
517       value = gvn.zerocon(field_type->basic_type());
518     }
519     vt->set_field_value(i, value);
520   }
521   vt = gvn.transform(vt)->as_ValueType();
522   assert(vt->is_default(gvn), "must be the default value type");
523   return vt;
524 }

526 bool ValueTypeNode::is_default(PhaseGVN& gvn) const {
527   for (uint i = 0; i < field_count(); ++i) {
528     Node* value = field_value(i);
529     if (!gvn.type(value)->is_zero_type() &&
530         !(value->is_ValueType() && value->as_ValueType()->is_default(gvn)) &&
531         !(field_type(i)->is_valuetype() && value == default_oop(gvn, field_type(i)->as_value_klass()))) {
532       return false;
533     }
534   }
535   return true;
536 }

538 ValueTypeNode* ValueTypeNode::make_from_oop(GraphKit* kit, Node* oop, ciValueKlass* vk) {
539   PhaseGVN& gvn = kit->gvn();
540 
541   // Create and initialize a ValueTypeNode by loading all field
542   // values from a heap-allocated version and also save the oop.
543   ValueTypeNode* vt = new ValueTypeNode(vk, oop);
544 
545   if (oop->isa_ValueTypePtr()) {
546     // Can happen with late inlining
547     ValueTypePtrNode* vtptr = oop->as_ValueTypePtr();
548     vt->set_oop(vtptr->get_oop());
549     for (uint i = Oop+1; i < vtptr->req(); ++i) {
550       vt->init_req(i, vtptr->in(i));
551     }
552   } else if (gvn.type(oop)->maybe_null()) {
553     // Add a null check because the oop may be null
554     Node* null_ctl = kit->top();
555     Node* not_null_oop = kit->null_check_oop(oop, &null_ctl);
556     if (kit->stopped()) {
557       // Constant null
558       kit->set_control(null_ctl);
559       return make_default(gvn, vk);
560     }
561     vt->set_oop(not_null_oop);
562     vt->load(kit, not_null_oop, not_null_oop, vk, /* holder_offset */ 0);
563 
564     if (null_ctl != kit->top()) {
565       // Return default value type if oop is null
566       ValueTypeNode* def = make_default(gvn, vk);
567       Node* region = new RegionNode(3);
568       region->init_req(1, kit->control());
569       region->init_req(2, null_ctl);
570 
571       vt = vt->clone_with_phis(&gvn, region)->as_ValueType();
572       vt->merge_with(&gvn, def, 2, true);
573       kit->set_control(gvn.transform(region));
574     }
575   } else {
576     // Oop can never be null
577     Node* init_ctl = kit->control();
578     vt->load(kit, oop, oop, vk, /* holder_offset */ 0);
579     assert(init_ctl != kit->control() || !gvn.type(oop)->is_valuetypeptr() || oop->is_Con() || oop->Opcode() == Op_ValueTypePtr ||
580            AllocateNode::Ideal_allocation(oop, &gvn) != NULL || vt->is_loaded(&gvn) == oop, "value type should be loaded");
581   }
582 
583   assert(vt->is_allocated(&gvn), "value type should be allocated");
584   return gvn.transform(vt)->as_ValueType();
585 }

587 // GraphKit wrapper for the 'make_from_flattened' method
588 ValueTypeNode* ValueTypeNode::make_from_flattened(GraphKit* kit, ciValueKlass* vk, Node* obj, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) {
589   // Create and initialize a ValueTypeNode by loading all field values from
590   // a flattened value type field at 'holder_offset' or from a value type array.
591   ValueTypeNode* vt = make_uninitialized(kit->gvn(), vk);
592   // The value type is flattened into the object without an oop header. Subtract the
593   // offset of the first field to account for the missing header when loading the values.
594   holder_offset -= vk->first_field_offset();
595   vt->load(kit, obj, ptr, holder, holder_offset, decorators);
596   assert(vt->is_loaded(&kit->gvn()) != obj, "holder oop should not be used as flattened value type oop");
597   return kit->gvn().transform(vt)->as_ValueType();
598 }

606 ValueTypeNode* ValueTypeNode::make_larval(GraphKit* kit, bool allocate) const {
607   ciValueKlass* vk = value_klass();
608   ValueTypeNode* res = clone()->as_ValueType();
609   if (allocate) {
610     // Re-execute if buffering triggers deoptimization
611     PreserveReexecuteState preexecs(kit);
612     kit->jvms()->set_should_reexecute(true);
613     Node* klass_node = kit->makecon(TypeKlassPtr::make(vk));
614     Node* alloc_oop  = kit->new_instance(klass_node, NULL, NULL, true);
615     AllocateNode* alloc = AllocateNode::Ideal_allocation(alloc_oop, &kit->gvn());
616     alloc->_larval = true;
617 
618     store(kit, alloc_oop, alloc_oop, vk, 0);
619     res->set_oop(alloc_oop);
620   }
621   res->set_type(TypeValueType::make(vk, true));
622   res = kit->gvn().transform(res)->as_ValueType();
623   assert(!allocate || res->is_allocated(&kit->gvn()), "must be allocated");
624   return res;

627 ValueTypeNode* ValueTypeNode::finish_larval(GraphKit* kit) const {
628   Node* obj = get_oop();
629   Node* mark_addr = kit->basic_plus_adr(obj, oopDesc::mark_offset_in_bytes());
630   Node* mark = kit->make_load(NULL, mark_addr, TypeX_X, TypeX_X->basic_type(), MemNode::unordered);
631   mark = kit->gvn().transform(new AndXNode(mark, kit->MakeConX(~markWord::larval_mask_in_place)));
632   kit->store_to_memory(kit->control(), mark_addr, mark, TypeX_X->basic_type(), kit->gvn().type(mark_addr)->is_ptr(), MemNode::unordered);
633   ready_to_publish(kit, obj);
634 
635   // Do not let stores that initialize this buffer be reordered with a subsequent
636   // store that would make this buffer accessible by other threads.
637   AllocateNode* alloc = AllocateNode::Ideal_allocation(obj, &kit->gvn());
638   assert(alloc != NULL, "must have an allocation node");
639   kit->insert_mem_bar(Op_MemBarStoreStore, alloc->proj_out_or_null(AllocateNode::RawAddress));
640 
641   ciValueKlass* vk = value_klass();
642   ValueTypeNode* res = clone()->as_ValueType();
643   res->set_type(TypeValueType::make(vk, false));
644   res = kit->gvn().transform(res)->as_ValueType();
645   return res;
646 }

659 Node* ValueTypeNode::is_loaded(PhaseGVN* phase, ciValueKlass* vk, Node* base, int holder_offset) {
660   if (vk == NULL) {
661     vk = value_klass();
662   }
663   if (field_count() == 0) {
664     assert(is_allocated(phase), "must be allocated");
665     return get_oop();
666   }
667   for (uint i = 0; i < field_count(); ++i) {
668     int offset = holder_offset + field_offset(i);
669     Node* value = field_value(i);
670     if (value->is_ValueType()) {
671       ValueTypeNode* vt = value->as_ValueType();
672       if (field_is_flattened(i)) {
673         // Check value type field load recursively
674         base = vt->is_loaded(phase, vk, base, offset - vt->value_klass()->first_field_offset());
675         if (base == NULL) {
676           return NULL;
677         }
678         continue;
679       } else {
680         value = vt->get_oop();
681         if (value->Opcode() == Op_CastPP) {
682           // Skip CastPP
683           value = value->in(1);
684         }
685       }
686     }
687     if (value->isa_DecodeN()) {
688       // Skip DecodeN
689       value = value->in(1);
690     }
691     if (value->isa_Load()) {
692       // Check if base and offset of field load matches value type layout
693       intptr_t loffset = 0;
694       Node* lbase = AddPNode::Ideal_base_and_offset(value->in(MemNode::Address), phase, loffset);
695       if (lbase == NULL || (lbase != base && base != NULL) || loffset != offset) {
696         return NULL;
697       } else if (base == NULL) {
698         // Set base and check if pointer type matches
699         base = lbase;
700         const TypeInstPtr* vtptr = phase->type(base)->isa_instptr();
701         if (vtptr == NULL || !vtptr->klass()->equals(vk)) {
702           return NULL;
703         }
704       }
705     } else {
706       return NULL;
707     }
708   }
709   return base;
710 }

719 void ValueTypeNode::pass_fields(GraphKit* kit, Node* n, ExtendedSignature& sig, uint& base_input, int base_offset) {
720   for (uint i = 0; i < field_count(); i++) {
721     int sig_offset = (*sig)._offset;
722     uint idx = field_index(sig_offset - base_offset);
723     Node* arg = field_value(idx);
724 
725     if (field_is_flattened(idx)) {
726       // Flattened value type field
727       ValueTypeNode* vt = arg->as_ValueType();
728       vt->pass_fields(kit, n, sig, base_input, sig_offset - vt->value_klass()->first_field_offset());
729     } else {
730       if (arg->is_ValueType()) {
731         // Non-flattened value type field
732         ValueTypeNode* vt = arg->as_ValueType();
733         assert(n->Opcode() != Op_Return || vt->is_allocated(&kit->gvn()), "value type field should be allocated on return");
734         arg = vt->allocate(kit)->get_oop();
735       }
736       // Initialize call/return arguments
737       BasicType bt = field_type(i)->basic_type();
738       n->init_req(base_input++, arg);
739       if (type2size[bt] == 2) {
740         n->init_req(base_input++, kit->top());
741       }
742       // Skip reserved arguments
743       while (SigEntry::next_is_reserved(sig, bt)) {
744         n->init_req(base_input++, kit->top());
745         if (type2size[bt] == 2) {
746           n->init_req(base_input++, kit->top());
747         }
748       }
749     }
750   }
751 }

753 void ValueTypeNode::initialize_fields(GraphKit* kit, MultiNode* multi, ExtendedSignature& sig, uint& base_input, int base_offset, bool in) {
754   PhaseGVN& gvn = kit->gvn();
755   for (uint i = 0; i < field_count(); i++) {
756     int sig_offset = (*sig)._offset;
757     uint idx = field_index(sig_offset - base_offset);
758     ciType* type = field_type(idx);
759 
760     Node* parm = NULL;
761     if (field_is_flattened(idx)) {
762       // Flattened value type field
763       ValueTypeNode* vt = ValueTypeNode::make_uninitialized(gvn, type->as_value_klass());
764       vt->initialize_fields(kit, multi, sig, base_input, sig_offset - type->as_value_klass()->first_field_offset(), in);
765       parm = gvn.transform(vt);
766     } else {
767       if (multi->is_Start()) {
768         assert(in, "return from start?");
769         parm = gvn.transform(new ParmNode(multi->as_Start(), base_input));
770       } else if (in) {
771         parm = multi->as_Call()->in(base_input);
772       } else {
773         parm = gvn.transform(new ProjNode(multi->as_Call(), base_input));
774       }
775       if (field_is_flattenable(idx)) {
776         // Non-flattened but flattenable value type
777         if (type->as_value_klass()->is_scalarizable()) {
778           parm = ValueTypeNode::make_from_oop(kit, parm, type->as_value_klass());
779         } else {
780           parm = kit->null2default(parm, type->as_value_klass());
781         }
782       }
783       base_input += type2size[type->basic_type()];
784       // Skip reserved arguments
785       BasicType bt = type->basic_type();
786       while (SigEntry::next_is_reserved(sig, bt)) {
787         base_input += type2size[bt];
788       }
789     }
790     assert(parm != NULL, "should never be null");
791     set_field_value(idx, parm);
792     gvn.record_for_igvn(parm);
793   }
794 }

796 Node* ValueTypeNode::Ideal(PhaseGVN* phase, bool can_reshape) {
797   Node* oop = get_oop();
798   if (is_default(*phase) && (!oop->is_Con() || phase->type(oop)->is_zero_type())) {
799     // Use the pre-allocated oop for default value types
800     set_oop(default_oop(*phase, value_klass()));
801     return this;
802   } else if (oop->isa_ValueTypePtr()) {
803     // Can happen with late inlining
804     ValueTypePtrNode* vtptr = oop->as_ValueTypePtr();
805     set_oop(vtptr->get_oop());
806     for (uint i = Oop+1; i < vtptr->req(); ++i) {
807       set_req(i, vtptr->in(i));
808     }
809     return this;
810   }
811 
812   if (!is_allocated(phase)) {
813     // Save base oop if fields are loaded from memory and the value
814     // type is not buffered (in this case we should not use the oop).
815     Node* base = is_loaded(phase);
816     if (base != NULL) {
817       set_oop(base);
818       assert(is_allocated(phase), "should now be allocated");
819       return this;
820     }
821   }
822 
823   if (can_reshape) {
824     PhaseIterGVN* igvn = phase->is_IterGVN();
825 
826     if (is_default(*phase)) {
827       // Search for users of the default value type
828       for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
829         Node* user = fast_out(i);
830         AllocateNode* alloc = user->isa_Allocate();
831         if (alloc != NULL && alloc->result_cast() != NULL && alloc->in(AllocateNode::ValueNode) == this) {
832           // Found an allocation of the default value type.
833           // If the code in StoreNode::Identity() that removes useless stores was not yet
834           // executed or ReduceFieldZeroing is disabled, there can still be initializing
835           // stores (only zero-type or default value stores, because value types are immutable).
836           Node* res = alloc->result_cast();
837           for (DUIterator_Fast jmax, j = res->fast_outs(jmax); j < jmax; j++) {
838             AddPNode* addp = res->fast_out(j)->isa_AddP();
839             if (addp != NULL) {
840               for (DUIterator_Fast kmax, k = addp->fast_outs(kmax); k < kmax; k++) {
841                 StoreNode* store = addp->fast_out(k)->isa_Store();
842                 if (store != NULL && store->outcnt() != 0) {
843                   // Remove the useless store
844                   igvn->replace_in_uses(store, store->in(MemNode::Memory));
845                 }
846               }
847             }
848           }
849           // Replace allocation by pre-allocated oop
850           igvn->replace_node(res, default_oop(*phase, value_klass()));
851         } else if (user->is_ValueType()) {
852           // Add value type user to worklist to give it a chance to get optimized as well
853           igvn->_worklist.push(user);
854         }
855       }
856     }
857 
858     if (is_allocated(igvn)) {
859       // Value type is heap allocated, search for safepoint uses
860       for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
861         Node* out = fast_out(i);
862         if (out->is_SafePoint()) {
863           // Let SafePointNode::Ideal() take care of re-wiring the
864           // safepoint to the oop input instead of the value type node.
865           igvn->rehash_node_delayed(out);
866         }
867       }
868     }
869   }
870   return NULL;
871 }

873 // Search for multiple allocations of this value type
874 // and try to replace them by dominating allocations.
875 // Then unlink the value type node and remove it.
876 void ValueTypeNode::remove_redundant_allocations(PhaseIterGVN* igvn, PhaseIdealLoop* phase) {
877   // Search for allocations of this value type
878   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
879     AllocateNode* alloc = fast_out(i)->isa_Allocate();
880     if (alloc != NULL && alloc->in(AllocateNode::ValueNode) == this) {
881       assert(!is_default(*igvn), "default value type allocation");
882       Node* res = alloc->result_cast();
883       if (res == NULL || !res->is_CheckCastPP()) {
884         break; // No unique CheckCastPP
885       }
886       Node* res_dom = res;
887       if (is_allocated(igvn)) {
888         // The value type is already allocated but still connected to an AllocateNode.
889         // This can happen with late inlining when we first allocate a value type argument
890         // but later decide to inline the call with the callee code also allocating.
891         res_dom = get_oop();
892       } else {
893         // Search for a dominating allocation of the same value type
894         for (DUIterator_Fast jmax, j = fast_outs(jmax); j < jmax; j++) {
895           AllocateNode* alloc_other = fast_out(j)->isa_Allocate();
896           if (alloc_other != NULL && alloc_other->in(AllocateNode::ValueNode) == this) {
897             Node* res_other = alloc_other->result_cast();
898             if (res_other != NULL && res_other->is_CheckCastPP() && res_other != res_dom &&
899                 phase->is_dominator(res_other->in(0), res_dom->in(0))) {
900               res_dom = res_other;
901             }
902           }
903         }
904       }
905       if (res_dom != res) {
906         // Move users to dominating allocation
907         igvn->replace_node(res, res_dom);
908         // The result of the dominated allocation is now unused and will be
909         // removed later in AllocateNode::Ideal() to not confuse loop opts.
910         igvn->record_for_igvn(alloc);
911 #ifdef ASSERT
912         if (PrintEliminateAllocations) {
913           tty->print("++++ Eliminated: %d Allocate ", alloc->_idx);
914           dump_spec(tty);
915           tty->cr();
916         }
917 #endif
918       }
919     }
920   }
921 
922   // Process users
923   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
924     Node* out = fast_out(i);
925     if (out->is_ValueType()) {
926       // Recursively process value type users
927       out->as_ValueType()->remove_redundant_allocations(igvn, phase);
928       --i; --imax;
929     } else if (out->isa_Allocate() != NULL) {
930       // Unlink AllocateNode
931       assert(out->in(AllocateNode::ValueNode) == this, "should be linked");
932       igvn->replace_input_of(out, AllocateNode::ValueNode, igvn->C->top());
933       --i; --imax;
934     } else {
935 #ifdef ASSERT
936       // The value type should not have any other users at this time
937       out->dump();
938       assert(false, "unexpected user of value type");
939 #endif
940     }
941   }
942   igvn->remove_dead_node(this);
943 }
