45     public static void main(String[] args) throws Exception {
46 
47         // Test that ACC_VALUE with ACC_ABSTRACT is illegal.
48         runTest("ValueAbstract", "Illegal class modifiers in class ValueAbstract");
49 
50         // Test that ACC_VALUE with ACC_ENUM is illegal.
51         runTest("ValueEnum", "Illegal class modifiers in class ValueEnum");
52 
53         // Test that inline type fields must be final.
54         runTest("ValueFieldNotFinal", "Illegal field modifiers in class ValueFieldNotFinal");
55 
56         // Test that ACC_VALUE with ACC_INTERFACE is illegal.
57         runTest("ValueInterface", "Illegal class modifiers in class ValueInterface");
58 
59         // Test that inline type instance methods cannot be synchronized.
60         runTest("ValueMethodSynch",
61                 "Method getInt in class ValueMethodSynch (an inline class) has illegal modifiers");
62 
63         runTest("ValueSuperClass", "Value type must have java.lang.Object as superclass");
64 
65         // Test that ClassCircularityError gets detected for instance fields.
66         System.out.println("Testing ClassCircularityError for instance fields");
67         try {
68             Class newClass = Class.forName("Circ");
69             throw new RuntimeException( "java.lang.ClassCircularityError exception not thrown!");
70         } catch (java.lang.ClassCircularityError e) {
71              if (!e.getMessage().contains("Circ")) {
72                  throw new RuntimeException( "Wrong ClassCircularityError: " + e.getMessage());
73              }
74          }
75 
76         // Test that ClassCircularityError isn't detected for static fields.
77         System.out.println("Testing ClassCircularityError for static fields");
78         try {
79             Class newClass = Class.forName("CircStaticB");
80         } catch (java.lang.ClassCircularityError e) {
81              throw new RuntimeException( "java.lang.ClassCircularityError exception thrown!");
82          }
83 
84         runTest("ValueCloneable", "Value Types do not support Cloneable");
85     }
