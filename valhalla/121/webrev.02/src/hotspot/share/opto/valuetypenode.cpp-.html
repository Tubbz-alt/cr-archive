 34 // Clones the values type to handle control flow merges involving multiple value types.
 35 // The inputs are replaced by PhiNodes to represent the merged values for the given region.
 36 ValueTypeBaseNode* ValueTypeBaseNode::clone_with_phis(PhaseGVN* gvn, Node* region) {
 37   assert(!has_phi_inputs(region), "already cloned with phis");
 38   ValueTypeBaseNode* vt = clone()->as_ValueTypeBase();
 39 
 40   // Create a PhiNode for merging the oop values
 41   const Type* phi_type = Type::get_const_type(value_klass());
 42   PhiNode* oop = PhiNode::make(region, vt->get_oop(), phi_type);
 43   gvn->set_type(oop, phi_type);
 44   vt->set_oop(oop);
 45 
 46   // Create a PhiNode each for merging the field values
 47   for (uint i = 0; i < vt->field_count(); ++i) {
 48     ciType* type = vt->field_type(i);
 49     Node*  value = vt->field_value(i);
 50     if (value->is_ValueTypeBase()) {
 51       // Handle flattened value type fields recursively
 52       value = value->as_ValueTypeBase()->clone_with_phis(gvn, region);
 53     } else {
 54       phi_type = Type::get_const_type(type);
 55       value = PhiNode::make(region, value, phi_type);
 56       gvn->set_type(value, phi_type);
 57     }
 58     vt->set_field_value(i, value);
 59   }
 60   gvn->set_type(vt, vt->bottom_type());
 61   return vt;
 62 }

 64 // Checks if the inputs of the ValueBaseTypeNode were replaced by PhiNodes
 65 // for the given region (see ValueBaseTypeNode::clone_with_phis).
 66 bool ValueTypeBaseNode::has_phi_inputs(Node* region) {
 67   // Check oop input
 68   bool result = get_oop()->is_Phi() && get_oop()->as_Phi()->region() == region;
 69 #ifdef ASSERT
 70   if (result) {
 71     // Check all field value inputs for consistency
 72     for (uint i = Oop; i < field_count(); ++i) {
 73       Node* n = in(i);
 74       if (n->is_ValueTypeBase()) {
 75         assert(n->as_ValueTypeBase()->has_phi_inputs(region), "inconsistent phi inputs");
 76       } else {
 77         assert(n->is_Phi() && n->as_Phi()->region() == region, "inconsistent phi inputs");
 78       }
 79     }
 80   }
 81 #endif
 82   return result;
 83 }

 85 // Merges 'this' with 'other' by updating the input PhiNodes added by 'clone_with_phis'
 86 ValueTypeBaseNode* ValueTypeBaseNode::merge_with(PhaseGVN* gvn, const ValueTypeBaseNode* other, int pnum, bool transform) {
 87   // Merge oop inputs
 88   PhiNode* phi = get_oop()->as_Phi();
 89   phi->set_req(pnum, other->get_oop());
 90   if (transform) {
 91     set_oop(gvn->transform(phi));
 92     gvn->record_for_igvn(phi);
 93   }
 94   // Merge field values
 95   for (uint i = 0; i < field_count(); ++i) {
 96     Node* val1 =        field_value(i);
 97     Node* val2 = other->field_value(i);
 98     if (val1->is_ValueTypeBase()) {
 99       val1->as_ValueTypeBase()->merge_with(gvn, val2->as_ValueTypeBase(), pnum, transform);
100     } else {
101       assert(val1->is_Phi(), "must be a phi node");
102       assert(!val2->is_ValueType(), "inconsistent merge values");
103       val1->set_req(pnum, val2);
104     }
105     if (transform) {
106       set_field_value(i, gvn->transform(val1));
107       gvn->record_for_igvn(val1);
108     }
109   }
110   return this;
111 }

113 // Adds a new merge path to a valuetype node with phi inputs
114 void ValueTypeBaseNode::add_new_path(Node* region) {
115   assert(has_phi_inputs(region), "must have phi inputs");
116 
117   PhiNode* phi = get_oop()->as_Phi();
118   phi->add_req(NULL);
119   assert(phi->req() == region->req(), "must be same size as region");
120 
121   for (uint i = 0; i < field_count(); ++i) {
122     Node* val = field_value(i);
123     if (val->is_ValueType()) {
124       val->as_ValueType()->add_new_path(region);
125     } else {
126       val->as_Phi()->add_req(NULL);
127       assert(val->req() == region->req(), "must be same size as region");
128     }
129   }
130 }

137 // Get the value of the field at the given offset.
138 // If 'recursive' is true, flattened value type fields will be resolved recursively.
139 Node* ValueTypeBaseNode::field_value_by_offset(int offset, bool recursive) const {
140   // If the field at 'offset' belongs to a flattened value type field, 'index' refers to the
141   // corresponding ValueTypeNode input and 'sub_offset' is the offset in flattened value type.
142   int index = value_klass()->field_index_by_offset(offset);
143   int sub_offset = offset - field_offset(index);
144   Node* value = field_value(index);
145   assert(value != NULL, "field value not found");
146   if (recursive && value->is_ValueType()) {
147     ValueTypeNode* vt = value->as_ValueType();
148     if (field_is_flattened(index)) {
149       // Flattened value type field
150       sub_offset += vt->value_klass()->first_field_offset(); // Add header size
151       return vt->field_value_by_offset(sub_offset, recursive);
152     } else {
153       assert(sub_offset == 0, "should not have a sub offset");
154       return vt;
155     }
156   }
157   assert(!(recursive && value->is_ValueType()), "should not be a value type");
158   assert(sub_offset == 0, "offset mismatch");
159   return value;
160 }

195 int ValueTypeBaseNode::make_scalar_in_safepoint(PhaseIterGVN* igvn, Unique_Node_List& worklist, SafePointNode* sfpt) {
196   ciValueKlass* vk = value_klass();
197   uint nfields = vk->nof_nonstatic_fields();
198   JVMState* jvms = sfpt->jvms();
199   int start = jvms->debug_start();
200   int end   = jvms->debug_end();
201   // Replace safepoint edge by SafePointScalarObjectNode and add field values
202   assert(jvms != NULL, "missing JVMS");
203   uint first_ind = (sfpt->req() - jvms->scloff());
204   SafePointScalarObjectNode* sobj = new SafePointScalarObjectNode(value_ptr(),
205 #ifdef ASSERT
206                                                                   NULL,
207 #endif
208                                                                   first_ind, nfields);
209   sobj->init_req(0, igvn->C->root());
210   // Iterate over the value type fields in order of increasing
211   // offset and add the field values to the safepoint.
212   for (uint j = 0; j < nfields; ++j) {
213     int offset = vk->nonstatic_field_at(j)->offset();
214     Node* value = field_value_by_offset(offset, true /* include flattened value type fields */);
215     if (value->is_ValueType()) {
216       // Add value type field to the worklist to process later
217       worklist.push(value);
218     }
219     sfpt->add_req(value);
220   }
221   jvms->set_endoff(sfpt->req());
222   sobj = igvn->transform(sobj)->as_SafePointScalarObject();
223   igvn->rehash_node_delayed(sfpt);
224   return sfpt->replace_edges_in_range(this, sobj, start, end);
225 }

227 void ValueTypeBaseNode::make_scalar_in_safepoints(PhaseIterGVN* igvn) {
228   // Process all safepoint uses and scalarize value type
229   Unique_Node_List worklist;
230   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
231     SafePointNode* sfpt = fast_out(i)->isa_SafePoint();
232     if (sfpt != NULL && !sfpt->is_CallLeaf() && (!sfpt->is_Call() || sfpt->as_Call()->has_debug_use(this))) {
233       int nb = 0;
234       if (is_allocated(igvn) && get_oop()->is_Con()) {
235         // Value type is allocated with a constant oop, link it directly
236         nb = sfpt->replace_edges_in_range(this, get_oop(), sfpt->jvms()->debug_start(), sfpt->jvms()->debug_end());
237         igvn->rehash_node_delayed(sfpt);
238       } else {
239         nb = make_scalar_in_safepoint(igvn, worklist, sfpt);
240       }
241       --i; imax -= nb;
242     }
243   }
244   // Now scalarize non-flattened fields
245   for (uint i = 0; i < worklist.size(); ++i) {
246     Node* vt = worklist.at(i);
247     vt->as_ValueType()->make_scalar_in_safepoints(igvn);
248   }
249   igvn->record_for_igvn(this);
250 }

252 const TypePtr* ValueTypeBaseNode::field_adr_type(Node* base, int offset, ciInstanceKlass* holder, DecoratorSet decorators, PhaseGVN& gvn) const {
253   const TypeAryPtr* ary_type = gvn.type(base)->isa_aryptr();
254   const TypePtr* adr_type = NULL;
255   bool is_array = ary_type != NULL;
256   if ((decorators & C2_MISMATCHED) != 0) {
257     adr_type = TypeRawPtr::BOTTOM;
258   } else if (is_array) {
259     // In the case of a flattened value type array, each field has its own slice
260     adr_type = ary_type->with_field_offset(offset)->add_offset(Type::OffsetBot);
261   } else {
262     ciField* field = holder->get_field_by_offset(offset, false);
263     assert(field != NULL, "field not found");
264     adr_type = gvn.C->alias_type(field)->adr_type();
265   }
266   return adr_type;
267 }

269 void ValueTypeBaseNode::load(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) {
270   // Initialize the value type by loading its field values from
271   // memory and adding the values as input edges to the node.
272   for (uint i = 0; i < field_count(); ++i) {
273     int offset = holder_offset + field_offset(i);
274     Node* value = NULL;
275     ciType* ft = field_type(i);
276     if (field_is_flattened(i)) {
277       // Recursively load the flattened value type field
278       value = ValueTypeNode::make_from_flattened(kit, ft->as_value_klass(), base, ptr, holder, offset, decorators);
279     } else {
280       const TypeOopPtr* oop_ptr = kit->gvn().type(base)->isa_oopptr();
281       bool is_array = (oop_ptr->isa_aryptr() != NULL);
282       if (base->is_Con() && !is_array) {
283         // If the oop to the value type is constant (static final field), we can
284         // also treat the fields as constants because the value type is immutable.
285         ciObject* constant_oop = oop_ptr->const_oop();
286         ciField* field = holder->get_field_by_offset(offset, false);
287         assert(field != NULL, "field not found");
288         ciConstant constant = constant_oop->as_instance()->field_value(field);
289         const Type* con_type = Type::make_from_constant(constant, /*require_const=*/ true);
290         assert(con_type != NULL, "type not found");
291         value = kit->gvn().transform(kit->makecon(con_type));
292         // Check type of constant which might be more precise than the static field type
293         if (con_type->is_valuetypeptr()) {
294           assert(!con_type->is_zero_type(), "Value types are null-free");
295           ft = con_type->value_klass();
296         }
297       } else {
298         // Load field value from memory
299         const TypePtr* adr_type = field_adr_type(base, offset, holder, decorators, kit->gvn());
300         Node* adr = kit->basic_plus_adr(base, ptr, offset);
301         BasicType bt = type2field[ft->basic_type()];
302         assert(is_java_primitive(bt) || adr->bottom_type()->is_ptr_to_narrowoop() == UseCompressedOops, "inconsistent");
303         const Type* val_type = Type::get_const_type(ft);
304         if (is_array) {
305           decorators |= IS_ARRAY;
306         }
307         value = kit->access_load_at(base, adr, adr_type, val_type, bt, decorators);
308       }
309       if (ft->is_valuetype()) {
310         // Loading a non-flattened value type from memory
311         if (ft->as_value_klass()->is_scalarizable()) {
312           value = ValueTypeNode::make_from_oop(kit, value, ft->as_value_klass());
313         } else {
314           value = kit->null2default(value, ft->as_value_klass());
315         }
316       }
317     }
318     set_field_value(i, value);
319   }
320 }

322 void ValueTypeBaseNode::store_flattened(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) const {
323   // The value type is embedded into the object without an oop header. Subtract the
324   // offset of the first field to account for the missing header when storing the values.
325   if (holder == NULL) {
326     holder = value_klass();
327   }
328   holder_offset -= value_klass()->first_field_offset();
329   store(kit, base, ptr, holder, holder_offset, decorators);
330 }

332 void ValueTypeBaseNode::store(GraphKit* kit, Node* base, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) const {
333   // Write field values to memory
334   for (uint i = 0; i < field_count(); ++i) {
335     int offset = holder_offset + field_offset(i);
336     Node* value = field_value(i);
337     ciType* ft = field_type(i);
338     if (field_is_flattened(i)) {
339       // Recursively store the flattened value type field
340       if (!value->is_ValueType()) {
341         assert(!kit->gvn().type(value)->maybe_null(), "Inline types are null-free");
342         value = ValueTypeNode::make_from_oop(kit, value, ft->as_value_klass());
343       }
344       value->as_ValueType()->store_flattened(kit, base, ptr, holder, offset, decorators);
345     } else {
346       // Store field value to memory
347       const TypePtr* adr_type = field_adr_type(base, offset, holder, decorators, kit->gvn());
348       Node* adr = kit->basic_plus_adr(base, ptr, offset);
349       BasicType bt = type2field[ft->basic_type()];
350       assert(is_java_primitive(bt) || adr->bottom_type()->is_ptr_to_narrowoop() == UseCompressedOops, "inconsistent");
351       const Type* val_type = Type::get_const_type(ft);
352       const TypeAryPtr* ary_type = kit->gvn().type(base)->isa_aryptr();
353       if (ary_type != NULL) {
354         decorators |= IS_ARRAY;
355       }
356       kit->access_store_at(base, adr, adr_type, value, val_type, bt, decorators);
357     }
358   }
359 }

361 ValueTypePtrNode* ValueTypeBaseNode::buffer(GraphKit* kit, bool safe_for_replace) {
362   assert(is_ValueType(), "sanity");
363   // Check if value type is already allocated
364   Node* null_ctl = kit->top();
365   Node* not_null_oop = kit->null_check_oop(get_oop(), &null_ctl);
366   if (null_ctl->is_top()) {
367     // Value type is allocated
368     return kit->gvn().transform(new ValueTypePtrNode(this))->as_ValueTypePtr();
369   }
370   assert(!is_allocated(&kit->gvn()), "should not be allocated");
371   RegionNode* region = new RegionNode(3);
372 
373   // Oop is non-NULL, use it
374   region->init_req(1, kit->control());
375   PhiNode* oop = PhiNode::make(region, not_null_oop, value_ptr());
376   PhiNode* io  = PhiNode::make(region, kit->i_o(), Type::ABIO);
377   PhiNode* mem = PhiNode::make(region, kit->merged_memory(), Type::MEMORY, TypePtr::BOTTOM);
378 
379   int bci = kit->bci();
380   bool reexecute = kit->jvms()->should_reexecute();
381   {
382     // Oop is NULL, allocate and initialize buffer
383     PreserveJVMState pjvms(kit);
384     // Propagate re-execution state and bci
385     kit->set_bci(bci);
386     kit->jvms()->set_bci(bci);
387     kit->jvms()->set_should_reexecute(reexecute);
388     kit->set_control(null_ctl);
389     kit->kill_dead_locals();
390     ciValueKlass* vk = value_klass();
391     Node* klass_node = kit->makecon(TypeKlassPtr::make(vk));
392     Node* alloc_oop  = kit->new_instance(klass_node, NULL, NULL, /* deoptimize_on_exception */ true, this);
393     store(kit, alloc_oop, alloc_oop, vk, 0);
394 
395     // Do not let stores that initialize this buffer be reordered with a subsequent
396     // store that would make this buffer accessible by other threads.
397     AllocateNode* alloc = AllocateNode::Ideal_allocation(alloc_oop, &kit->gvn());
398     assert(alloc != NULL, "must have an allocation node");
399     kit->insert_mem_bar(Op_MemBarStoreStore, alloc->proj_out_or_null(AllocateNode::RawAddress));
400 
401     region->init_req(2, kit->control());
402     oop   ->init_req(2, alloc_oop);
403     io    ->init_req(2, kit->i_o());
404     mem   ->init_req(2, kit->merged_memory());
405   }
406 
407   // Update GraphKit
408   kit->set_control(kit->gvn().transform(region));
409   kit->set_i_o(kit->gvn().transform(io));
410   kit->set_all_memory(kit->gvn().transform(mem));
411   kit->record_for_igvn(region);
412   kit->record_for_igvn(oop);
413   kit->record_for_igvn(io);
414   kit->record_for_igvn(mem);
415 
416   // Use cloned ValueTypeNode to propagate oop from now on
417   Node* res_oop = kit->gvn().transform(oop);
418   ValueTypeBaseNode* vt = clone()->as_ValueTypeBase();
419   vt->set_oop(res_oop);
420   vt = kit->gvn().transform(vt)->as_ValueTypeBase();
421   if (safe_for_replace) {
422     kit->replace_in_map(this, vt);
423   }
424   // ValueTypeNode::remove_redundant_allocations piggybacks on split if.
425   // Make sure it gets a chance to remove this allocation.
426   kit->C->set_has_split_ifs(true);
427   assert(vt->is_allocated(&kit->gvn()), "must be allocated");
428   return kit->gvn().transform(new ValueTypePtrNode(vt))->as_ValueTypePtr();
429 }

431 bool ValueTypeBaseNode::is_allocated(PhaseGVN* phase) const {
432   Node* oop = get_oop();
433   const Type* oop_type = (phase != NULL) ? phase->type(oop) : oop->bottom_type();
434   return !oop_type->maybe_null();

437 // When a call returns multiple values, it has several result
438 // projections, one per field. Replacing the result of the call by a
439 // value type node (after late inlining) requires that for each result
440 // projection, we find the corresponding value type field.
441 void ValueTypeBaseNode::replace_call_results(GraphKit* kit, Node* call, Compile* C) {
442   ciValueKlass* vk = value_klass();
443   for (DUIterator_Fast imax, i = call->fast_outs(imax); i < imax; i++) {
444     ProjNode* pn = call->fast_out(i)->as_Proj();
445     uint con = pn->_con;
446     if (con >= TypeFunc::Parms+1) {
447       uint field_nb = con - (TypeFunc::Parms+1);
448       int extra = 0;
449       for (uint j = 0; j < field_nb - extra; j++) {
450         ciField* f = vk->nonstatic_field_at(j);
451         BasicType bt = f->type()->basic_type();
452         if (bt == T_LONG || bt == T_DOUBLE) {
453           extra++;
454         }
455       }
456       ciField* f = vk->nonstatic_field_at(field_nb - extra);
457       Node* field = field_value_by_offset(f->offset(), true);
458       if (field->is_ValueType()) {
459         assert(field->as_ValueType()->is_allocated(&kit->gvn()), "must be allocated");
460         field = field->as_ValueType()->get_oop();
461       }
462       C->gvn_replace_by(pn, field);
463       C->initial_gvn()->hash_delete(pn);
464       pn->set_req(0, C->top());
465       --i; --imax;
466     }
467   }
468 }

470 Node* ValueTypeBaseNode::allocate_fields(GraphKit* kit) {
471   ValueTypeBaseNode* vt = clone()->as_ValueTypeBase();
472   for (uint i = 0; i < field_count(); i++) {
473      ValueTypeNode* value = field_value(i)->isa_ValueType();
474      if (field_is_flattened(i)) {
475        // Flattened value type field
476        vt->set_field_value(i, value->allocate_fields(kit));
477      } else if (value != NULL) {
478        // Non-flattened value type field
479        vt->set_field_value(i, value->buffer(kit));
480      }
481   }
482   vt = kit->gvn().transform(vt)->as_ValueTypeBase();
483   kit->replace_in_map(this, vt);
484   return vt;
485 }

487 ValueTypeNode* ValueTypeNode::make_uninitialized(PhaseGVN& gvn, ciValueKlass* vk) {
488   // Create a new ValueTypeNode with uninitialized values and NULL oop

493 Node* ValueTypeNode::default_oop(PhaseGVN& gvn, ciValueKlass* vk) {

498 ValueTypeNode* ValueTypeNode::make_default(PhaseGVN& gvn, ciValueKlass* vk) {
499   // Create a new ValueTypeNode with default values
500   ValueTypeNode* vt = new ValueTypeNode(vk, default_oop(gvn, vk));
501   for (uint i = 0; i < vt->field_count(); ++i) {
502     ciType* field_type = vt->field_type(i);
503     Node* value = NULL;
504     if (field_type->is_valuetype()) {
505       ciValueKlass* field_klass = field_type->as_value_klass();
506       if (field_klass->is_scalarizable() || vt->field_is_flattened(i)) {
507         value = ValueTypeNode::make_default(gvn, field_klass);
508       } else {
509         value = default_oop(gvn, field_klass);
510       }
511     } else {
512       value = gvn.zerocon(field_type->basic_type());
513     }
514     vt->set_field_value(i, value);
515   }
516   vt = gvn.transform(vt)->as_ValueType();
517   assert(vt->is_default(gvn), "must be the default value type");
518   return vt;
519 }

521 bool ValueTypeNode::is_default(PhaseGVN& gvn) const {
522   for (uint i = 0; i < field_count(); ++i) {
523     Node* value = field_value(i);
524     if (!gvn.type(value)->is_zero_type() &&
525         !(value->is_ValueType() && value->as_ValueType()->is_default(gvn)) &&
526         !(field_type(i)->is_valuetype() && value == default_oop(gvn, field_type(i)->as_value_klass()))) {
527       return false;
528     }
529   }
530   return true;
531 }

533 ValueTypeNode* ValueTypeNode::make_from_oop(GraphKit* kit, Node* oop, ciValueKlass* vk) {
534   PhaseGVN& gvn = kit->gvn();
535 
536   // Create and initialize a ValueTypeNode by loading all field
537   // values from a heap-allocated version and also save the oop.
538   ValueTypeNode* vt = new ValueTypeNode(vk, oop);
539 
540   if (oop->isa_ValueTypePtr()) {
541     // Can happen with late inlining
542     ValueTypePtrNode* vtptr = oop->as_ValueTypePtr();
543     vt->set_oop(vtptr->get_oop());
544     for (uint i = Oop+1; i < vtptr->req(); ++i) {
545       vt->init_req(i, vtptr->in(i));
546     }
547   } else if (gvn.type(oop)->maybe_null()) {
548     // Add a null check because the oop may be null
549     Node* null_ctl = kit->top();
550     Node* not_null_oop = kit->null_check_oop(oop, &null_ctl);
551     if (kit->stopped()) {
552       // Constant null
553       kit->set_control(null_ctl);
554       return make_default(gvn, vk);
555     }
556     vt->set_oop(not_null_oop);
557     vt->load(kit, not_null_oop, not_null_oop, vk, /* holder_offset */ 0);
558 
559     if (null_ctl != kit->top()) {
560       // Return default value type if oop is null
561       ValueTypeNode* def = make_default(gvn, vk);
562       Node* region = new RegionNode(3);
563       region->init_req(1, kit->control());
564       region->init_req(2, null_ctl);
565 
566       vt = vt->clone_with_phis(&gvn, region)->as_ValueType();
567       vt->merge_with(&gvn, def, 2, true);
568       kit->set_control(gvn.transform(region));
569     }
570   } else {
571     // Oop can never be null
572     Node* init_ctl = kit->control();
573     vt->load(kit, oop, oop, vk, /* holder_offset */ 0);
574     assert(init_ctl != kit->control() || !gvn.type(oop)->is_valuetypeptr() || oop->is_Con() || oop->Opcode() == Op_ValueTypePtr ||
575            AllocateNode::Ideal_allocation(oop, &gvn) != NULL || vt->is_loaded(&gvn) == oop, "value type should be loaded");
576   }
577 
578   assert(vt->is_allocated(&gvn), "value type should be allocated");
579   return gvn.transform(vt)->as_ValueType();
580 }

582 // GraphKit wrapper for the 'make_from_flattened' method
583 ValueTypeNode* ValueTypeNode::make_from_flattened(GraphKit* kit, ciValueKlass* vk, Node* obj, Node* ptr, ciInstanceKlass* holder, int holder_offset, DecoratorSet decorators) {
584   // Create and initialize a ValueTypeNode by loading all field values from
585   // a flattened value type field at 'holder_offset' or from a value type array.
586   ValueTypeNode* vt = make_uninitialized(kit->gvn(), vk);
587   // The value type is flattened into the object without an oop header. Subtract the
588   // offset of the first field to account for the missing header when loading the values.
589   holder_offset -= vk->first_field_offset();
590   vt->load(kit, obj, ptr, holder, holder_offset, decorators);
591   assert(vt->is_loaded(&kit->gvn()) != obj, "holder oop should not be used as flattened value type oop");
592   return kit->gvn().transform(vt)->as_ValueType();
593 }

601 ValueTypeNode* ValueTypeNode::make_larval(GraphKit* kit, bool allocate) const {
602   ciValueKlass* vk = value_klass();
603   ValueTypeNode* res = clone()->as_ValueType();
604   if (allocate) {
605     // Re-execute if buffering triggers deoptimization
606     PreserveReexecuteState preexecs(kit);
607     kit->jvms()->set_should_reexecute(true);
608     Node* klass_node = kit->makecon(TypeKlassPtr::make(vk));
609     Node* alloc_oop  = kit->new_instance(klass_node, NULL, NULL, true);
610     AllocateNode* alloc = AllocateNode::Ideal_allocation(alloc_oop, &kit->gvn());
611     alloc->_larval = true;
612 
613     store(kit, alloc_oop, alloc_oop, vk, 0);
614     res->set_oop(alloc_oop);
615   }
616   res->set_type(TypeValueType::make(vk, true));
617   res = kit->gvn().transform(res)->as_ValueType();
618   assert(!allocate || res->is_allocated(&kit->gvn()), "must be allocated");
619   return res;
620 }

622 ValueTypeNode* ValueTypeNode::finish_larval(GraphKit* kit) const {
623   Node* obj = get_oop();
624   Node* mark_addr = kit->basic_plus_adr(obj, oopDesc::mark_offset_in_bytes());
625   Node* mark = kit->make_load(NULL, mark_addr, TypeX_X, TypeX_X->basic_type(), MemNode::unordered);
626   mark = kit->gvn().transform(new AndXNode(mark, kit->MakeConX(~markWord::larval_mask_in_place)));
627   kit->store_to_memory(kit->control(), mark_addr, mark, TypeX_X->basic_type(), kit->gvn().type(mark_addr)->is_ptr(), MemNode::unordered);
628 
629   // Do not let stores that initialize this buffer be reordered with a subsequent
630   // store that would make this buffer accessible by other threads.
631   AllocateNode* alloc = AllocateNode::Ideal_allocation(obj, &kit->gvn());
632   assert(alloc != NULL, "must have an allocation node");
633   kit->insert_mem_bar(Op_MemBarStoreStore, alloc->proj_out_or_null(AllocateNode::RawAddress));
634 
635   ciValueKlass* vk = value_klass();
636   ValueTypeNode* res = clone()->as_ValueType();
637   res->set_type(TypeValueType::make(vk, false));
638   res = kit->gvn().transform(res)->as_ValueType();
639   return res;
640 }

642 Node* ValueTypeNode::is_loaded(PhaseGVN* phase, ciValueKlass* vk, Node* base, int holder_offset) {
643   if (vk == NULL) {
644     vk = value_klass();
645   }
646   if (field_count() == 0) {
647     assert(is_allocated(phase), "must be allocated");
648     return get_oop();
649   }
650   for (uint i = 0; i < field_count(); ++i) {
651     int offset = holder_offset + field_offset(i);
652     Node* value = field_value(i);
653     if (value->is_ValueType()) {
654       ValueTypeNode* vt = value->as_ValueType();
655       if (field_is_flattened(i)) {
656         // Check value type field load recursively
657         base = vt->is_loaded(phase, vk, base, offset - vt->value_klass()->first_field_offset());
658         if (base == NULL) {
659           return NULL;
660         }
661         continue;
662       } else {
663         value = vt->get_oop();
664         if (value->Opcode() == Op_CastPP) {
665           // Skip CastPP
666           value = value->in(1);
667         }
668       }
669     }
670     if (value->isa_DecodeN()) {
671       // Skip DecodeN
672       value = value->in(1);
673     }
674     if (value->isa_Load()) {
675       // Check if base and offset of field load matches value type layout
676       intptr_t loffset = 0;
677       Node* lbase = AddPNode::Ideal_base_and_offset(value->in(MemNode::Address), phase, loffset);
678       if (lbase == NULL || (lbase != base && base != NULL) || loffset != offset) {
679         return NULL;
680       } else if (base == NULL) {
681         // Set base and check if pointer type matches
682         base = lbase;
683         const TypeInstPtr* vtptr = phase->type(base)->isa_instptr();
684         if (vtptr == NULL || !vtptr->klass()->equals(vk)) {
685           return NULL;
686         }
687       }
688     } else {
689       return NULL;
690     }
691   }
692   return base;
693 }

696   const TypeKlassPtr* tk = TypeKlassPtr::make(vk);

702 void ValueTypeNode::pass_fields(GraphKit* kit, Node* n, ExtendedSignature& sig, uint& base_input, int base_offset) {
703   for (uint i = 0; i < field_count(); i++) {
704     int sig_offset = (*sig)._offset;
705     uint idx = field_index(sig_offset - base_offset);
706     Node* arg = field_value(idx);
707 
708     if (field_is_flattened(idx)) {
709       // Flattened value type field
710       ValueTypeNode* vt = arg->as_ValueType();
711       vt->pass_fields(kit, n, sig, base_input, sig_offset - vt->value_klass()->first_field_offset());
712     } else {
713       if (arg->is_ValueType()) {
714         // Non-flattened value type field
715         ValueTypeNode* vt = arg->as_ValueType();
716         assert(n->Opcode() != Op_Return || vt->is_allocated(&kit->gvn()), "value type field should be allocated on return");
717         arg = vt->buffer(kit);
718       }
719       // Initialize call/return arguments
720       BasicType bt = field_type(i)->basic_type();
721       n->init_req(base_input++, arg);
722       if (type2size[bt] == 2) {
723         n->init_req(base_input++, kit->top());
724       }
725       // Skip reserved arguments
726       while (SigEntry::next_is_reserved(sig, bt)) {
727         n->init_req(base_input++, kit->top());
728         if (type2size[bt] == 2) {
729           n->init_req(base_input++, kit->top());
730         }
731       }
732     }
733   }
734 }
735 
736 void ValueTypeNode::initialize_fields(GraphKit* kit, MultiNode* multi, ExtendedSignature& sig, uint& base_input, int base_offset, bool in) {
737   PhaseGVN& gvn = kit->gvn();
738   for (uint i = 0; i < field_count(); i++) {
739     int sig_offset = (*sig)._offset;
740     uint idx = field_index(sig_offset - base_offset);
741     ciType* type = field_type(idx);
742 
743     Node* parm = NULL;
744     if (field_is_flattened(idx)) {
745       // Flattened value type field
746       ValueTypeNode* vt = ValueTypeNode::make_uninitialized(gvn, type->as_value_klass());
747       vt->initialize_fields(kit, multi, sig, base_input, sig_offset - type->as_value_klass()->first_field_offset(), in);
748       parm = gvn.transform(vt);
749     } else {
750       if (multi->is_Start()) {
751         assert(in, "return from start?");
752         parm = gvn.transform(new ParmNode(multi->as_Start(), base_input));
753       } else if (in) {
754         parm = multi->as_Call()->in(base_input);
755       } else {
756         parm = gvn.transform(new ProjNode(multi->as_Call(), base_input));
757       }
758       if (type->is_valuetype()) {
759         // Non-flattened value type field
760         if (type->as_value_klass()->is_scalarizable()) {
761           parm = ValueTypeNode::make_from_oop(kit, parm, type->as_value_klass());
762         } else {
763           parm = kit->null2default(parm, type->as_value_klass());
764         }
765       }
766       base_input += type2size[type->basic_type()];
767       // Skip reserved arguments
768       BasicType bt = type->basic_type();
769       while (SigEntry::next_is_reserved(sig, bt)) {
770         base_input += type2size[bt];
771       }
772     }
773     assert(parm != NULL, "should never be null");
774     set_field_value(idx, parm);
775     gvn.record_for_igvn(parm);
776   }
777 }

779 Node* ValueTypeNode::Ideal(PhaseGVN* phase, bool can_reshape) {
780   Node* oop = get_oop();
781   if (is_default(*phase) && (!oop->is_Con() || phase->type(oop)->is_zero_type())) {
782     // Use the pre-allocated oop for default value types
783     set_oop(default_oop(*phase, value_klass()));
784     return this;
785   } else if (oop->isa_ValueTypePtr()) {
786     // Can happen with late inlining
787     ValueTypePtrNode* vtptr = oop->as_ValueTypePtr();
788     set_oop(vtptr->get_oop());
789     for (uint i = Oop+1; i < vtptr->req(); ++i) {
790       set_req(i, vtptr->in(i));
791     }
792     return this;
793   }
794 
795   if (!is_allocated(phase)) {
796     // Save base oop if fields are loaded from memory and the value
797     // type is not buffered (in this case we should not use the oop).
798     Node* base = is_loaded(phase);
799     if (base != NULL) {
800       set_oop(base);
801       assert(is_allocated(phase), "should now be allocated");
802       return this;
803     }
804   }
805 
806   if (can_reshape) {
807     PhaseIterGVN* igvn = phase->is_IterGVN();
808 
809     if (is_default(*phase)) {
810       // Search for users of the default value type
811       for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
812         Node* user = fast_out(i);
813         AllocateNode* alloc = user->isa_Allocate();
814         if (alloc != NULL && alloc->in(AllocateNode::ValueNode) == this) {
815           // Found an allocation of the default value type.
816           Node* res = alloc->result_cast();
817           if (res != NULL) {
818             // If the code in StoreNode::Identity() that removes useless stores was not yet
819             // executed or ReduceFieldZeroing is disabled, there can still be initializing
820             // stores (only zero-type or default value stores, because value types are immutable).
821             for (DUIterator_Fast jmax, j = res->fast_outs(jmax); j < jmax; j++) {
822               AddPNode* addp = res->fast_out(j)->isa_AddP();
823               if (addp != NULL) {
824                 for (DUIterator_Fast kmax, k = addp->fast_outs(kmax); k < kmax; k++) {
825                   StoreNode* store = addp->fast_out(k)->isa_Store();
826                   if (store != NULL && store->outcnt() != 0) {
827                     // Remove the useless store
828                     igvn->replace_in_uses(store, store->in(MemNode::Memory));
829                   }
830                 }
831               }
832             }
833             // Replace allocation by pre-allocated oop
834             igvn->replace_node(res, default_oop(*phase, value_klass()));
835           }
836           // Unlink AllocateNode
837           igvn->replace_input_of(alloc, AllocateNode::ValueNode, igvn->C->top());
838           --i; --imax;
839         } else if (user->is_ValueType()) {
840           // Add value type user to worklist to give it a chance to get optimized as well
841           igvn->_worklist.push(user);
842         }
843       }
844     }
845 
846     if (is_allocated(igvn)) {
847       // Value type is heap allocated, search for safepoint uses
848       for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
849         Node* out = fast_out(i);
850         if (out->is_SafePoint()) {
851           // Let SafePointNode::Ideal() take care of re-wiring the
852           // safepoint to the oop input instead of the value type node.
853           igvn->rehash_node_delayed(out);
854         }
855       }
856     }
857   }
858   return NULL;
859 }

861 // Search for multiple allocations of this value type
862 // and try to replace them by dominating allocations.
863 // Then unlink the value type node and remove it.
864 void ValueTypeNode::remove_redundant_allocations(PhaseIterGVN* igvn, PhaseIdealLoop* phase) {
865   // Search for allocations of this value type
866   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
867     AllocateNode* alloc = fast_out(i)->isa_Allocate();
868     if (alloc != NULL && alloc->in(AllocateNode::ValueNode) == this) {
869       assert(!is_default(*igvn), "default value type allocation");
870       Node* res = alloc->result_cast();
871       if (res == NULL || !res->is_CheckCastPP()) {
872         break; // No unique CheckCastPP
873       }
874       Node* res_dom = res;
875       if (is_allocated(igvn)) {
876         // The value type is already allocated but still connected to an AllocateNode.
877         // This can happen with late inlining when we first allocate a value type argument
878         // but later decide to inline the call with the callee code also allocating.
879         res_dom = get_oop();
880       } else {
881         // Search for a dominating allocation of the same value type
882         for (DUIterator_Fast jmax, j = fast_outs(jmax); j < jmax; j++) {
883           AllocateNode* alloc_other = fast_out(j)->isa_Allocate();
884           if (alloc_other != NULL && alloc_other->in(AllocateNode::ValueNode) == this) {
885             Node* res_other = alloc_other->result_cast();
886             if (res_other != NULL && res_other->is_CheckCastPP() && res_other != res_dom &&
887                 phase->is_dominator(res_other->in(0), res_dom->in(0))) {
888               res_dom = res_other;
889             }
890           }
891         }
892       }
893       if (res_dom != res) {
894         // Move users to dominating allocation
895         igvn->replace_node(res, res_dom);
896         // The result of the dominated allocation is now unused and will be removed
897         // later in PhaseMacroExpand::eliminate_allocate_node to not confuse loop opts.
898         igvn->record_for_igvn(alloc);
899       }
900     }
901   }
902 
903   // Process users
904   for (DUIterator_Fast imax, i = fast_outs(imax); i < imax; i++) {
905     Node* out = fast_out(i);
906     if (out->is_ValueType()) {
907       // Unlink and recursively process value type users
908       igvn->hash_delete(out);
909       int nb = out->replace_edge(this, igvn->C->top());
910       out->as_ValueType()->remove_redundant_allocations(igvn, phase);
911       --i; imax -= nb;
912     } else if (out->isa_Allocate() != NULL) {
913       // Unlink AllocateNode
914       assert(out->in(AllocateNode::ValueNode) == this, "should be linked");
915       igvn->replace_input_of(out, AllocateNode::ValueNode, igvn->C->top());
916       --i; --imax;
917     } else {
918 #ifdef ASSERT
919       // The value type should not have any other users at this time
920       out->dump();
921       assert(false, "unexpected user of value type");
922 #endif
923     }
924   }
925   igvn->remove_dead_node(this);
926 }
