24 /**
25  * @test
26  * @bug 8187679
27  * @summary The VM should exit gracefully when unable to resolve a value type argument
28  * @library /test/lib
29  * @build SimpleValueType
30  * @run main/othervm TestUnresolvedValueClass
31  */
32 
33 import java.io.File;
34 import jdk.test.lib.process.OutputAnalyzer;
35 import jdk.test.lib.process.ProcessTools;
36 
37 public class TestUnresolvedValueClass {
38     final static String TEST_CLASSES = System.getProperty("test.classes") + File.separator;
39 
40     // Method with unresolved value type argument
41     static void test1(SimpleValueType vt) {
42 
43     }
44 
45     static public void main(String[] args) throws Exception {
46         if (args.length == 0) {
47             // Delete SimpleValueType.class to cause a NoClassDefFoundError
48             File unresolved = new File(TEST_CLASSES, "SimpleValueType.class");
49             if (!unresolved.exists() || !unresolved.delete()) {
50                 throw new RuntimeException("Could not delete: " + unresolved);
51             }
52 
53             // Run test in new VM instance
54             String[] arg = {"-XX:+InlineTypePassFieldsAsArgs", "TestUnresolvedValueClass", "run"};
55             OutputAnalyzer oa = ProcessTools.executeTestJvm(arg);
56 
57             // Adapter creation for TestUnresolvedValueClass::test1 should fail with a
58             // ClassNotFoundException because the class for argument 'vt' was not found.
59             String output = oa.getOutput();
60             oa.shouldContain("java.lang.NoClassDefFoundError: SimpleValueType");
61         }
62     }
63 }
