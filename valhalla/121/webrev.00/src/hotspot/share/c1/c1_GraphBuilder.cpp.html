 134 void BlockListBuilder::set_entries(int osr_bci) {
 135   // generate start blocks
 136   BlockBegin* std_entry = make_block_at(0, NULL);
 137   if (scope()->caller() == NULL) {
 138     std_entry->set(BlockBegin::std_entry_flag);
 139   }
 140   if (osr_bci != -1) {
 141     BlockBegin* osr_entry = make_block_at(osr_bci, NULL);
 142     osr_entry->set(BlockBegin::osr_entry_flag);
 143   }
 144 
 145   // generate exception entry blocks
 146   XHandlers* list = xhandlers();
 147   const int n = list->length();
 148   for (int i = 0; i < n; i++) {
 149     XHandler* h = list->handler_at(i);
 150     BlockBegin* entry = make_block_at(h->handler_bci(), NULL);
 151     entry->set(BlockBegin::exception_entry_flag);
 152     h->set_entry_block(entry);
 153   }
 154 }

 157 BlockBegin* BlockListBuilder::make_block_at(int cur_bci, BlockBegin* predecessor) {
 158   assert(method()->bci_block_start().at(cur_bci), "wrong block starts of MethodLivenessAnalyzer");
 159 
 160   BlockBegin* block = _bci2block->at(cur_bci);
 161   if (block == NULL) {
 162     block = new BlockBegin(cur_bci);
 163     block->init_stores_to_locals(method()->max_locals());
 164     _bci2block->at_put(cur_bci, block);
 165     _blocks.append(block);
 166 
 167     assert(predecessor == NULL || predecessor->bci() < cur_bci, "targets for backward branches must already exist");
 168   }
 169 
 170   if (predecessor != NULL) {
 171     if (block->is_set(BlockBegin::exception_entry_flag)) {
 172       BAILOUT_("Exception handler can be reached by both normal and exceptional control flow", block);
 173     }
 174 
 175     predecessor->add_successor(block);
 176     block->increment_total_preds();
 177   }
 178 
 179   return block;
 180 }

 183 inline void BlockListBuilder::store_one(BlockBegin* current, int local) {
 184   current->stores_to_locals().set_bit(local);

 186 inline void BlockListBuilder::store_two(BlockBegin* current, int local) {
 187   store_one(current, local);
 188   store_one(current, local + 1);

 192 void BlockListBuilder::handle_exceptions(BlockBegin* current, int cur_bci) {
 193   // Draws edges from a block to its exception handlers
 194   XHandlers* list = xhandlers();
 195   const int n = list->length();
 196 
 197   for (int i = 0; i < n; i++) {
 198     XHandler* h = list->handler_at(i);
 199 
 200     if (h->covers(cur_bci)) {
 201       BlockBegin* entry = h->entry_block();
 202       assert(entry != NULL && entry == _bci2block->at(h->handler_bci()), "entry must be set");
 203       assert(entry->is_set(BlockBegin::exception_entry_flag), "flag must be set");
 204 
 205       // add each exception handler only once
 206       if (!current->is_successor(entry)) {
 207         current->add_successor(entry);
 208         entry->increment_total_preds();
 209       }
 210 
 211       // stop when reaching catchall
 212       if (h->catch_type() == 0) break;
 213     }
 214   }
 215 }

 217 void BlockListBuilder::handle_jsr(BlockBegin* current, int sr_bci, int next_bci) {
 218   // start a new block after jsr-bytecode and link this block into cfg
 219   make_block_at(next_bci, current);
 220 
 221   // start a new block at the subroutine entry at mark it with special flag
 222   BlockBegin* sr_block = make_block_at(sr_bci, current);
 223   if (!sr_block->is_set(BlockBegin::subroutine_entry_flag)) {
 224     sr_block->set(BlockBegin::subroutine_entry_flag);
 225   }
 226 }

 229 void BlockListBuilder::set_leaders() {
 230   bool has_xhandlers = xhandlers()->has_handlers();
 231   BlockBegin* current = NULL;
 232 
 233   // The information which bci starts a new block simplifies the analysis
 234   // Without it, backward branches could jump to a bci where no block was created
 235   // during bytecode iteration. This would require the creation of a new block at the
 236   // branch target and a modification of the successor lists.
 237   const BitMap& bci_block_start = method()->bci_block_start();
 238 
 239   ciBytecodeStream s(method());
 240   while (s.next() != ciBytecodeStream::EOBC()) {
 241     int cur_bci = s.cur_bci();
 242 
 243     if (bci_block_start.at(cur_bci)) {
 244       current = make_block_at(cur_bci, current);
 245     }
 246     assert(current != NULL, "must have current block");
 247 
 248     if (has_xhandlers && GraphBuilder::can_trap(method(), s.cur_bc())) {
 249       handle_exceptions(current, cur_bci);
 250     }
 251 
 252     switch (s.cur_bc()) {
 253       // track stores to local variables for selective creation of phi functions
 254       case Bytecodes::_iinc:     store_one(current, s.get_index()); break;
 255       case Bytecodes::_istore:   store_one(current, s.get_index()); break;
 256       case Bytecodes::_lstore:   store_two(current, s.get_index()); break;
 257       case Bytecodes::_fstore:   store_one(current, s.get_index()); break;
 258       case Bytecodes::_dstore:   store_two(current, s.get_index()); break;
 259       case Bytecodes::_astore:   store_one(current, s.get_index()); break;
 260       case Bytecodes::_istore_0: store_one(current, 0); break;
 261       case Bytecodes::_istore_1: store_one(current, 1); break;
 262       case Bytecodes::_istore_2: store_one(current, 2); break;
 263       case Bytecodes::_istore_3: store_one(current, 3); break;
 264       case Bytecodes::_lstore_0: store_two(current, 0); break;
 265       case Bytecodes::_lstore_1: store_two(current, 1); break;
 266       case Bytecodes::_lstore_2: store_two(current, 2); break;
 267       case Bytecodes::_lstore_3: store_two(current, 3); break;
 268       case Bytecodes::_fstore_0: store_one(current, 0); break;
 269       case Bytecodes::_fstore_1: store_one(current, 1); break;
 270       case Bytecodes::_fstore_2: store_one(current, 2); break;
 271       case Bytecodes::_fstore_3: store_one(current, 3); break;
 272       case Bytecodes::_dstore_0: store_two(current, 0); break;
 273       case Bytecodes::_dstore_1: store_two(current, 1); break;
 274       case Bytecodes::_dstore_2: store_two(current, 2); break;
 275       case Bytecodes::_dstore_3: store_two(current, 3); break;
 276       case Bytecodes::_astore_0: store_one(current, 0); break;
 277       case Bytecodes::_astore_1: store_one(current, 1); break;
 278       case Bytecodes::_astore_2: store_one(current, 2); break;
 279       case Bytecodes::_astore_3: store_one(current, 3); break;
 280 
 281       // track bytecodes that affect the control flow
 282       case Bytecodes::_athrow:  // fall through
 283       case Bytecodes::_ret:     // fall through
 284       case Bytecodes::_ireturn: // fall through
 285       case Bytecodes::_lreturn: // fall through
 286       case Bytecodes::_freturn: // fall through
 287       case Bytecodes::_dreturn: // fall through
 288       case Bytecodes::_areturn: // fall through
 289       case Bytecodes::_return:
 290         current = NULL;
 291         break;
 292 
 293       case Bytecodes::_ifeq:      // fall through
 294       case Bytecodes::_ifne:      // fall through
 295       case Bytecodes::_iflt:      // fall through
 296       case Bytecodes::_ifge:      // fall through
 297       case Bytecodes::_ifgt:      // fall through
 298       case Bytecodes::_ifle:      // fall through
 299       case Bytecodes::_if_icmpeq: // fall through
 300       case Bytecodes::_if_icmpne: // fall through
 301       case Bytecodes::_if_icmplt: // fall through
 302       case Bytecodes::_if_icmpge: // fall through
 303       case Bytecodes::_if_icmpgt: // fall through
 304       case Bytecodes::_if_icmple: // fall through
 305       case Bytecodes::_if_acmpeq: // fall through
 306       case Bytecodes::_if_acmpne: // fall through
 307       case Bytecodes::_ifnull:    // fall through
 308       case Bytecodes::_ifnonnull:
 309         make_block_at(s.next_bci(), current);
 310         make_block_at(s.get_dest(), current);
 311         current = NULL;
 312         break;
 313 
 314       case Bytecodes::_goto:
 315         make_block_at(s.get_dest(), current);
 316         current = NULL;
 317         break;
 318 
 319       case Bytecodes::_goto_w:
 320         make_block_at(s.get_far_dest(), current);
 321         current = NULL;
 322         break;
 323 
 324       case Bytecodes::_jsr:
 325         handle_jsr(current, s.get_dest(), s.next_bci());
 326         current = NULL;
 327         break;
 328 
 329       case Bytecodes::_jsr_w:
 330         handle_jsr(current, s.get_far_dest(), s.next_bci());
 331         current = NULL;
 332         break;
 333 
 334       case Bytecodes::_tableswitch: {
 335         // set block for each case
 336         Bytecode_tableswitch sw(&s);
 337         int l = sw.length();
 338         for (int i = 0; i < l; i++) {
 339           make_block_at(cur_bci + sw.dest_offset_at(i), current);
 340         }
 341         make_block_at(cur_bci + sw.default_offset(), current);
 342         current = NULL;
 343         break;
 344       }
 345 
 346       case Bytecodes::_lookupswitch: {
 347         // set block for each case
 348         Bytecode_lookupswitch sw(&s);
 349         int l = sw.number_of_pairs();
 350         for (int i = 0; i < l; i++) {
 351           make_block_at(cur_bci + sw.pair_at(i).offset(), current);
 352         }
 353         make_block_at(cur_bci + sw.default_offset(), current);
 354         current = NULL;
 355         break;
 356       }
 357 
 358       default:
 359         break;
 360     }
 361   }
 362 }

 365 void BlockListBuilder::mark_loops() {
 366   ResourceMark rm;
 367 
 368   _active.initialize(BlockBegin::number_of_blocks());
 369   _visited.initialize(BlockBegin::number_of_blocks());
 370   _loop_map = intArray(BlockBegin::number_of_blocks(), BlockBegin::number_of_blocks(), 0);
 371   _next_loop_index = 0;
 372   _next_block_number = _blocks.length();
 373 
 374   // recursively iterate the control flow graph
 375   mark_loops(_bci2block->at(0), false);
 376   assert(_next_block_number >= 0, "invalid block numbers");
 377 
 378   // Remove dangling Resource pointers before the ResourceMark goes out-of-scope.
 379   _active.resize(0);
 380   _visited.resize(0);
 381 }

 383 void BlockListBuilder::make_loop_header(BlockBegin* block) {
 384   if (block->is_set(BlockBegin::exception_entry_flag)) {
 385     // exception edges may look like loops but don't mark them as such
 386     // since it screws up block ordering.
 387     return;
 388   }
 389   if (!block->is_set(BlockBegin::parser_loop_header_flag)) {
 390     block->set(BlockBegin::parser_loop_header_flag);
 391 
 392     assert(_loop_map.at(block->block_id()) == 0, "must not be set yet");
 393     assert(0 <= _next_loop_index && _next_loop_index < BitsPerInt, "_next_loop_index is used as a bit-index in integer");
 394     _loop_map.at_put(block->block_id(), 1 << _next_loop_index);
 395     if (_next_loop_index < 31) _next_loop_index++;
 396   } else {
 397     // block already marked as loop header
 398     assert(is_power_of_2((unsigned int)_loop_map.at(block->block_id())), "exactly one bit must be set");
 399   }
 400 }

 402 int BlockListBuilder::mark_loops(BlockBegin* block, bool in_subroutine) {
 403   int block_id = block->block_id();
 404 
 405   if (_visited.at(block_id)) {
 406     if (_active.at(block_id)) {
 407       // reached block via backward branch
 408       make_loop_header(block);
 409     }
 410     // return cached loop information for this block
 411     return _loop_map.at(block_id);
 412   }
 413 
 414   if (block->is_set(BlockBegin::subroutine_entry_flag)) {
 415     in_subroutine = true;
 416   }
 417 
 418   // set active and visited bits before successors are processed
 419   _visited.set_bit(block_id);
 420   _active.set_bit(block_id);
 421 
 422   intptr_t loop_state = 0;
 423   for (int i = block->number_of_sux() - 1; i >= 0; i--) {
 424     // recursively process all successors
 425     loop_state |= mark_loops(block->sux_at(i), in_subroutine);
 426   }
 427 
 428   // clear active-bit after all successors are processed
 429   _active.clear_bit(block_id);
 430 
 431   // reverse-post-order numbering of all blocks
 432   block->set_depth_first_number(_next_block_number);
 433   _next_block_number--;
 434 
 435   if (loop_state != 0 || in_subroutine ) {
 436     // block is contained at least in one loop, so phi functions are necessary
 437     // phi functions are also necessary for all locals stored in a subroutine
 438     scope()->requires_phi_function().set_union(block->stores_to_locals());
 439   }
 440 
 441   if (block->is_set(BlockBegin::parser_loop_header_flag)) {
 442     int header_loop_state = _loop_map.at(block_id);
 443     assert(is_power_of_2((unsigned)header_loop_state), "exactly one bit must be set");
 444 
 445     // If the highest bit is set (i.e. when integer value is negative), the method
 446     // has 32 or more loops. This bit is never cleared because it is used for multiple loops
 447     if (header_loop_state >= 0) {
 448       clear_bits(loop_state, header_loop_state);
 449     }
 450   }
 451 
 452   // cache and return loop information for this block
 453   _loop_map.at_put(block_id, loop_state);
 454   return loop_state;
 455 }

 464 void BlockListBuilder::print() {
 465   tty->print("----- initial block list of BlockListBuilder for method ");
 466   method()->print_short_name();
 467   tty->cr();
 468 
 469   // better readability if blocks are sorted in processing order
 470   _blocks.sort(compare_depth_first);
 471 
 472   for (int i = 0; i < _blocks.length(); i++) {
 473     BlockBegin* cur = _blocks.at(i);
 474     tty->print("%4d: B%-4d bci: %-4d  preds: %-4d ", cur->depth_first_number(), cur->block_id(), cur->bci(), cur->total_preds());
 475 
 476     tty->print(cur->is_set(BlockBegin::std_entry_flag)               ? " std" : "    ");
 477     tty->print(cur->is_set(BlockBegin::osr_entry_flag)               ? " osr" : "    ");
 478     tty->print(cur->is_set(BlockBegin::exception_entry_flag)         ? " ex" : "   ");
 479     tty->print(cur->is_set(BlockBegin::subroutine_entry_flag)        ? " sr" : "   ");
 480     tty->print(cur->is_set(BlockBegin::parser_loop_header_flag)      ? " lh" : "   ");
 481 
 482     if (cur->number_of_sux() > 0) {
 483       tty->print("    sux: ");
 484       for (int j = 0; j < cur->number_of_sux(); j++) {
 485         BlockBegin* sux = cur->sux_at(j);
 486         tty->print("B%d ", sux->block_id());
 487       }
 488     }
 489     tty->cr();
 490   }
 491 }

 496 // A simple growable array of Values indexed by ciFields
 497 class FieldBuffer: public CompilationResourceObj {
 498  private:
 499   GrowableArray<Value> _values;
 500 
 501  public:
 502   FieldBuffer() {}
 503 
 504   void kill() {
 505     _values.trunc_to(0);
 506   }
 507 
 508   Value at(ciField* field) {
 509     assert(field->holder()->is_loaded(), "must be a loaded field");
 510     int offset = field->offset();
 511     if (offset < _values.length()) {
 512       return _values.at(offset);
 513     } else {
 514       return NULL;
 515     }
 516   }
 517 
 518   void at_put(ciField* field, Value value) {
 519     assert(field->holder()->is_loaded(), "must be a loaded field");
 520     int offset = field->offset();
 521     _values.at_put_grow(offset, value, NULL);
 522   }
 523 
 524 };

 527 // MemoryBuffer is fairly simple model of the current state of memory.
 528 // It partitions memory into several pieces.  The first piece is
 529 // generic memory where little is known about the owner of the memory.
 530 // This is conceptually represented by the tuple <O, F, V> which says
 531 // that the field F of object O has value V.  This is flattened so
 532 // that F is represented by the offset of the field and the parallel
 533 // arrays _objects and _values are used for O and V.  Loads of O.F can
 534 // simply use V.  Newly allocated objects are kept in a separate list
 535 // along with a parallel array for each object which represents the
 536 // current value of its fields.  Stores of the default value to fields
 537 // which have never been stored to before are eliminated since they
 538 // are redundant.  Once newly allocated objects are stored into
 539 // another object or they are passed out of the current compile they
 540 // are treated like generic memory.
 541 
 542 class MemoryBuffer: public CompilationResourceObj {
 543  private:
 544   FieldBuffer                 _values;
 545   GrowableArray<Value>        _objects;
 546   GrowableArray<Value>        _newobjects;
 547   GrowableArray<FieldBuffer*> _fields;
 548 
 549  public:
 550   MemoryBuffer() {}
 551 
 552   StoreField* store(StoreField* st) {
 553     if (!EliminateFieldAccess) {
 554       return st;
 555     }
 556 
 557     Value object = st->obj();
 558     Value value = st->value();
 559     ciField* field = st->field();
 560     if (field->holder()->is_loaded()) {
 561       int offset = field->offset();
 562       int index = _newobjects.find(object);
 563       if (index != -1) {
 564         // newly allocated object with no other stores performed on this field
 565         FieldBuffer* buf = _fields.at(index);
 566         if (buf->at(field) == NULL && is_default_value(value)) {
 567 #ifndef PRODUCT
 568           if (PrintIRDuringConstruction && Verbose) {
 569             tty->print_cr("Eliminated store for object %d:", index);
 570             st->print_line();
 571           }
 572 #endif
 573           return NULL;
 574         } else {
 575           buf->at_put(field, value);
 576         }
 577       } else {
 578         _objects.at_put_grow(offset, object, NULL);
 579         _values.at_put(field, value);
 580       }
 581 
 582       store_value(value);
 583     } else {
 584       // if we held onto field names we could alias based on names but
 585       // we don't know what's being stored to so kill it all.
 586       kill();
 587     }
 588     return st;
 589   }
 590 
 591 
 592   // return true if this value correspond to the default value of a field.
 593   bool is_default_value(Value value) {
 594     Constant* con = value->as_Constant();
 595     if (con) {
 596       switch (con->type()->tag()) {
 597         case intTag:    return con->type()->as_IntConstant()->value() == 0;
 598         case longTag:   return con->type()->as_LongConstant()->value() == 0;
 599         case floatTag:  return jint_cast(con->type()->as_FloatConstant()->value()) == 0;
 600         case doubleTag: return jlong_cast(con->type()->as_DoubleConstant()->value()) == jlong_cast(0);
 601         case objectTag: return con->type() == objectNull;
 602         default:  ShouldNotReachHere();
 603       }
 604     }
 605     return false;
 606   }
 607 
 608 
 609   // return either the actual value of a load or the load itself
 610   Value load(LoadField* load) {
 611     if (!EliminateFieldAccess) {
 612       return load;
 613     }
 614 
 615     if (strict_fp_requires_explicit_rounding && load->type()->is_float_kind()) {
 616 #ifdef IA32
 617       if (UseSSE < 2) {
 618         // can't skip load since value might get rounded as a side effect
 619         return load;
 620       }
 621 #else
 622       Unimplemented();
 623 #endif // IA32
 624     }
 625 
 626     ciField* field = load->field();
 627     Value object   = load->obj();
 628     if (field->holder()->is_loaded() && !field->is_volatile()) {
 629       int offset = field->offset();
 630       Value result = NULL;
 631       int index = _newobjects.find(object);
 632       if (index != -1) {
 633         result = _fields.at(index)->at(field);
 634       } else if (_objects.at_grow(offset, NULL) == object) {
 635         result = _values.at(field);
 636       }
 637       if (result != NULL) {
 638 #ifndef PRODUCT
 639         if (PrintIRDuringConstruction && Verbose) {
 640           tty->print_cr("Eliminated load: ");
 641           load->print_line();
 642         }
 643 #endif
 644         assert(result->type()->tag() == load->type()->tag(), "wrong types");
 645         return result;
 646       }
 647     }
 648     return load;
 649   }
 650 
 651   // Record this newly allocated object
 652   void new_instance(NewInstance* object) {
 653     int index = _newobjects.length();
 654     _newobjects.append(object);
 655     if (_fields.at_grow(index, NULL) == NULL) {
 656       _fields.at_put(index, new FieldBuffer());
 657     } else {
 658       _fields.at(index)->kill();
 659     }
 660   }
 661 
 662   // Record this newly allocated object
 663   void new_instance(NewInlineTypeInstance* object) {
 664     int index = _newobjects.length();
 665     _newobjects.append(object);
 666     if (_fields.at_grow(index, NULL) == NULL) {
 667       _fields.at_put(index, new FieldBuffer());
 668     } else {
 669       _fields.at(index)->kill();
 670     }
 671   }
 672 
 673   void store_value(Value value) {
 674     int index = _newobjects.find(value);
 675     if (index != -1) {
 676       // stored a newly allocated object into another object.
 677       // Assume we've lost track of it as separate slice of memory.
 678       // We could do better by keeping track of whether individual
 679       // fields could alias each other.
 680       _newobjects.remove_at(index);
 681       // pull out the field info and store it at the end up the list
 682       // of field info list to be reused later.
 683       _fields.append(_fields.at(index));
 684       _fields.remove_at(index);
 685     }
 686   }
 687 
 688   void kill() {
 689     _newobjects.trunc_to(0);
 690     _objects.trunc_to(0);
 691     _values.kill();
 692   }
 693 };

 734 BlockBegin* GraphBuilder::ScopeData::block_at(int bci) {
 735   if (parsing_jsr()) {
 736     // It is necessary to clone all blocks associated with a
 737     // subroutine, including those for exception handlers in the scope
 738     // of the method containing the jsr (because those exception
 739     // handlers may contain ret instructions in some cases).
 740     BlockBegin* block = bci2block()->at(bci);
 741     if (block != NULL && block == parent()->bci2block()->at(bci)) {
 742       BlockBegin* new_block = new BlockBegin(block->bci());
 743       if (PrintInitialBlockList) {
 744         tty->print_cr("CFG: cloned block %d (bci %d) as block %d for jsr",
 745                       block->block_id(), block->bci(), new_block->block_id());
 746       }
 747       // copy data from cloned blocked
 748       new_block->set_depth_first_number(block->depth_first_number());
 749       if (block->is_set(BlockBegin::parser_loop_header_flag)) new_block->set(BlockBegin::parser_loop_header_flag);
 750       // Preserve certain flags for assertion checking
 751       if (block->is_set(BlockBegin::subroutine_entry_flag)) new_block->set(BlockBegin::subroutine_entry_flag);
 752       if (block->is_set(BlockBegin::exception_entry_flag))  new_block->set(BlockBegin::exception_entry_flag);
 753 
 754       // copy was_visited_flag to allow early detection of bailouts
 755       // if a block that is used in a jsr has already been visited before,
 756       // it is shared between the normal control flow and a subroutine
 757       // BlockBegin::try_merge returns false when the flag is set, this leads
 758       // to a compilation bailout
 759       if (block->is_set(BlockBegin::was_visited_flag))  new_block->set(BlockBegin::was_visited_flag);
 760 
 761       bci2block()->at_put(bci, new_block);
 762       block = new_block;
 763     }
 764     return block;
 765   } else {
 766     return bci2block()->at(bci);
 767   }

 825 void GraphBuilder::sort_top_into_worklist(BlockList* worklist, BlockBegin* top) {
 826   assert(worklist->top() == top, "");
 827   // sort block descending into work list
 828   const int dfn = top->depth_first_number();
 829   assert(dfn != -1, "unknown depth first number");
 830   int i = worklist->length()-2;
 831   while (i >= 0) {
 832     BlockBegin* b = worklist->at(i);
 833     if (b->depth_first_number() < dfn) {
 834       worklist->at_put(i+1, b);
 835     } else {
 836       break;
 837     }
 838     i --;
 839   }
 840   if (i >= -1) worklist->at_put(i + 1, top);
 841 }

 898 void GraphBuilder::load_constant() {
 899   ciConstant con = stream()->get_constant();
 900   if (con.basic_type() == T_ILLEGAL) {
 901     // FIXME: an unresolved Dynamic constant can get here,
 902     // and that should not terminate the whole compilation.
 903     BAILOUT("could not resolve a constant");
 904   } else {
 905     ValueType* t = illegalType;
 906     ValueStack* patch_state = NULL;
 907     switch (con.basic_type()) {
 908       case T_BOOLEAN: t = new IntConstant     (con.as_boolean()); break;
 909       case T_BYTE   : t = new IntConstant     (con.as_byte   ()); break;
 910       case T_CHAR   : t = new IntConstant     (con.as_char   ()); break;
 911       case T_SHORT  : t = new IntConstant     (con.as_short  ()); break;
 912       case T_INT    : t = new IntConstant     (con.as_int    ()); break;
 913       case T_LONG   : t = new LongConstant    (con.as_long   ()); break;
 914       case T_FLOAT  : t = new FloatConstant   (con.as_float  ()); break;
 915       case T_DOUBLE : t = new DoubleConstant  (con.as_double ()); break;
 916       case T_ARRAY  : t = new ArrayConstant   (con.as_object ()->as_array   ()); break;
 917       case T_OBJECT :
 918        {
 919         ciObject* obj = con.as_object();
 920         if (!obj->is_loaded()
 921             || (PatchALot && obj->klass() != ciEnv::current()->String_klass())) {
 922           // A Class, MethodType, MethodHandle, or String.
 923           // Unloaded condy nodes show up as T_ILLEGAL, above.
 924           patch_state = copy_state_before();
 925           t = new ObjectConstant(obj);
 926         } else {
 927           // Might be a Class, MethodType, MethodHandle, or Dynamic constant
 928           // result, which might turn out to be an array.
 929           if (obj->is_null_object())
 930             t = objectNull;
 931           else if (obj->is_array())
 932             t = new ArrayConstant(obj->as_array());
 933           else
 934             t = new InstanceConstant(obj->as_instance());
 935         }
 936         break;
 937        }
 938       default       : ShouldNotReachHere();
 939     }
 940     Value x;
 941     if (patch_state != NULL) {
 942       x = new Constant(t, patch_state);
 943     } else {
 944       x = new Constant(t);
 945     }
 946     push(t, append(x));
 947   }
 948 }

 964 void GraphBuilder::store_local(ValueStack* state, Value x, int index) {
 965   if (parsing_jsr()) {
 966     // We need to do additional tracking of the location of the return
 967     // address for jsrs since we don't handle arbitrary jsr/ret
 968     // constructs. Here we are figuring out in which circumstances we
 969     // need to bail out.
 970     if (x->type()->is_address()) {
 971       scope_data()->set_jsr_return_address_local(index);
 972 
 973       // Also check parent jsrs (if any) at this time to see whether
 974       // they are using this local. We don't handle skipping over a
 975       // ret.
 976       for (ScopeData* cur_scope_data = scope_data()->parent();
 977            cur_scope_data != NULL && cur_scope_data->parsing_jsr() && cur_scope_data->scope() == scope();
 978            cur_scope_data = cur_scope_data->parent()) {
 979         if (cur_scope_data->jsr_return_address_local() == index) {
 980           BAILOUT("subroutine overwrites return address from previous subroutine");
 981         }
 982       }
 983     } else if (index == scope_data()->jsr_return_address_local()) {
 984       scope_data()->set_jsr_return_address_local(-1);
 985     }
 986   }
 987 
 988   x->set_local_index(index);
 989   state->store_local(index, round_fp(x));
 990 }

 993 void GraphBuilder::load_indexed(BasicType type) {
 994   // In case of in block code motion in range check elimination
 995   ValueStack* state_before = NULL;
 996   int array_idx = state()->stack_size() - 2;
 997   if (type == T_OBJECT && state()->stack_at(array_idx)->maybe_flattened_array()) {
 998     // Save the entire state and re-execute on deopt when accessing flattened arrays
 999     state_before = copy_state_before();
1000     state_before->set_should_reexecute(true);
1001   } else {
1002     state_before = copy_state_indexed_access();
1003   }
1004   compilation()->set_has_access_indexed(true);
1005   Value index = ipop();
1006   Value array = apop();
1007   Value length = NULL;
1008   if (CSEArrayLength ||
1009       (array->as_AccessField() && array->as_AccessField()->field()->is_constant()) ||
1010       (array->as_NewArray() && array->as_NewArray()->length() && array->as_NewArray()->length()->type()->is_constant())) {
1011     length = append(new ArrayLength(array, state_before));
1012   }
1013 
1014   LoadIndexed* load_indexed = NULL;
1015   Instruction* result = NULL;
1016   if (array->is_loaded_flattened_array()) {
1017     ciType* array_type = array->declared_type();
1018     ciInlineKlass* elem_klass = array_type->as_flat_array_klass()->element_klass()->as_inline_klass();
1019     NewInlineTypeInstance* new_instance = new NewInlineTypeInstance(elem_klass, state_before, false);
1020     _memory->new_instance(new_instance);
1021     apush(append_split(new_instance));
1022     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1023     load_indexed->set_vt(new_instance);
1024   } else {
1025     load_indexed = new LoadIndexed(array, index, length, type, state_before);
1026   }
1027   if (profile_array_accesses() && is_reference_type(type)) {
1028     compilation()->set_would_profile(true);
1029     load_indexed->set_should_profile(true);
1030     load_indexed->set_profiled_method(method());
1031     load_indexed->set_profiled_bci(bci());
1032   }
1033   result = append(load_indexed);
1034   assert(!(profile_array_accesses() && is_reference_type(type)) || load_indexed == result, "should not be optimized out");
1035   if (!array->is_loaded_flattened_array()) {
1036     push(as_ValueType(type), result);
1037   }
1038 }

1091   switch (code) {
1092     case Bytecodes::_pop:
1093       { state()->raw_pop();
1094       }
1095       break;
1096     case Bytecodes::_pop2:
1097       { state()->raw_pop();
1098         state()->raw_pop();
1099       }
1100       break;

1102       { Value w = state()->raw_pop();
1103         state()->raw_push(w);
1104         state()->raw_push(w);

1124       break;

1192 void GraphBuilder::shift_op(ValueType* type, Bytecodes::Code code) {
1193   Value s = ipop();
1194   Value x = pop(type);
1195   // try to simplify
1196   // Note: This code should go into the canonicalizer as soon as it can
1197   //       can handle canonicalized forms that contain more than one node.
1198   if (CanonicalizeNodes && code == Bytecodes::_iushr) {
1199     // pattern: x >>> s
1200     IntConstant* s1 = s->type()->as_IntConstant();
1201     if (s1 != NULL) {
1202       // pattern: x >>> s1, with s1 constant
1203       ShiftOp* l = x->as_ShiftOp();
1204       if (l != NULL && l->op() == Bytecodes::_ishl) {
1205         // pattern: (a << b) >>> s1
1206         IntConstant* s0 = l->y()->type()->as_IntConstant();
1207         if (s0 != NULL) {
1208           // pattern: (a << s0) >>> s1
1209           const int s0c = s0->value() & 0x1F; // only the low 5 bits are significant for shifts
1210           const int s1c = s1->value() & 0x1F; // only the low 5 bits are significant for shifts
1211           if (s0c == s1c) {
1212             if (s0c == 0) {
1213               // pattern: (a << 0) >>> 0 => simplify to: a
1214               ipush(l->x());
1215             } else {
1216               // pattern: (a << s0c) >>> s0c => simplify to: a & m, with m constant
1217               assert(0 < s0c && s0c < BitsPerInt, "adjust code below to handle corner cases");
1218               const int m = (1 << (BitsPerInt - s0c)) - 1;
1219               Value s = append(new Constant(new IntConstant(m)));
1220               ipush(append(new LogicOp(Bytecodes::_iand, l->x(), s)));
1221             }
1222             return;
1223           }
1224         }
1225       }
1226     }
1227   }
1228   // could not simplify
1229   push(type, append(new ShiftOp(code, x, s)));
1230 }

1277 void GraphBuilder::if_node(Value x, If::Condition cond, Value y, ValueStack* state_before) {
1278   BlockBegin* tsux = block_at(stream()->get_dest());
1279   BlockBegin* fsux = block_at(stream()->next_bci());
1280   bool is_bb = tsux->bci() < stream()->cur_bci() || fsux->bci() < stream()->cur_bci();
1281 
1282   bool subst_check = false;
1283   if (EnableValhalla && (stream()->cur_bc() == Bytecodes::_if_acmpeq || stream()->cur_bc() == Bytecodes::_if_acmpne) &&
1284       method() != ciEnv::current()->ValueBootstrapMethods_klass()->find_method(ciSymbol::isSubstitutable_name(), ciSymbol::object_object_boolean_signature())) {
1285     // If current method is ValueBootstrapMethods::isSubstitutable(),
1286     // compile the acmp as a regular pointer comparison otherwise we
1287     // could call ValueBootstrapMethods::isSubstitutable() back
1288     ValueType* left_vt = x->type();
1289     ValueType* right_vt = y->type();
1290     if (left_vt->is_object()) {
1291       assert(right_vt->is_object(), "must be");
1292       ciKlass* left_klass = x->as_loaded_klass_or_null();
1293       ciKlass* right_klass = y->as_loaded_klass_or_null();
1294 
1295       if (left_klass == NULL || right_klass == NULL) {
1296         // The klass is still unloaded, or came from a Phi node. Go slow case;
1297         subst_check = true;
1298       } else if (left_klass->can_be_inline_klass() || right_klass->can_be_inline_klass()) {
1299         // Either operand may be a value object, but we're not sure. Go slow case;
1300         subst_check = true;
1301       } else {
1302         // No need to do substitutability check
1303       }
1304     }
1305   }
1306 
1307   // In case of loop invariant code motion or predicate insertion
1308   // before the body of a loop the state is needed
1309   Instruction *i = append(new If(x, cond, false, y, tsux, fsux, (is_bb || compilation()->is_optimistic() || subst_check) ? state_before : NULL, is_bb, subst_check));
1310 
1311   assert(i->as_Goto() == NULL ||
1312          (i->as_Goto()->sux_at(0) == tsux  && i->as_Goto()->is_safepoint() == tsux->bci() < stream()->cur_bci()) ||
1313          (i->as_Goto()->sux_at(0) == fsux  && i->as_Goto()->is_safepoint() == fsux->bci() < stream()->cur_bci()),
1314          "safepoint state of Goto returned by canonicalizer incorrect");
1315 
1316   if (is_profiling()) {
1317     If* if_node = i->as_If();
1318     if (if_node != NULL) {
1319       // Note that we'd collect profile data in this method if we wanted it.
1320       compilation()->set_would_profile(true);
1321       // At level 2 we need the proper bci to count backedges
1322       if_node->set_profiled_bci(bci());
1323       if (profile_branches()) {
1324         // Successors can be rotated by the canonicalizer, check for this case.
1325         if_node->set_profiled_method(method());
1326         if_node->set_should_profile(true);
1327         if (if_node->tsux() == fsux) {
1328           if_node->set_swapped(true);
1329         }
1330       }
1331       return;
1332     }
1333 
1334     // Check if this If was reduced to Goto.
1335     Goto *goto_node = i->as_Goto();
1336     if (goto_node != NULL) {
1337       compilation()->set_would_profile(true);
1338       goto_node->set_profiled_bci(bci());
1339       if (profile_branches()) {
1340         goto_node->set_profiled_method(method());
1341         goto_node->set_should_profile(true);
1342         // Find out which successor is used.
1343         if (goto_node->default_sux() == tsux) {
1344           goto_node->set_direction(Goto::taken);
1345         } else if (goto_node->default_sux() == fsux) {
1346           goto_node->set_direction(Goto::not_taken);
1347         } else {
1348           ShouldNotReachHere();
1349         }
1350       }
1351       return;
1352     }
1353   }
1354 }

1413 void GraphBuilder::table_switch() {
1414   Bytecode_tableswitch sw(stream());
1415   const int l = sw.length();
1416   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1417     // total of 2 successors => use If instead of switch
1418     // Note: This code should go into the canonicalizer as soon as it can
1419     //       can handle canonicalized forms that contain more than one node.
1420     Value key = append(new Constant(new IntConstant(sw.low_key())));
1421     BlockBegin* tsux = block_at(bci() + sw.dest_offset_at(0));
1422     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1423     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1424     // In case of loop invariant code motion or predicate insertion
1425     // before the body of a loop the state is needed
1426     ValueStack* state_before = copy_state_if_bb(is_bb);
1427     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1428   } else {
1429     // collect successors
1430     BlockList* sux = new BlockList(l + 1, NULL);
1431     int i;
1432     bool has_bb = false;
1433     for (i = 0; i < l; i++) {
1434       sux->at_put(i, block_at(bci() + sw.dest_offset_at(i)));
1435       if (sw.dest_offset_at(i) < 0) has_bb = true;
1436     }
1437     // add default successor
1438     if (sw.default_offset() < 0) has_bb = true;
1439     sux->at_put(i, block_at(bci() + sw.default_offset()));
1440     // In case of loop invariant code motion or predicate insertion
1441     // before the body of a loop the state is needed
1442     ValueStack* state_before = copy_state_if_bb(has_bb);
1443     Instruction* res = append(new TableSwitch(ipop(), sux, sw.low_key(), state_before, has_bb));
1444 #ifdef ASSERT
1445     if (res->as_Goto()) {
1446       for (i = 0; i < l; i++) {
1447         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1448           assert(res->as_Goto()->is_safepoint() == sw.dest_offset_at(i) < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1449         }
1450       }
1451     }
1452 #endif
1453   }
1454 }

1457 void GraphBuilder::lookup_switch() {
1458   Bytecode_lookupswitch sw(stream());
1459   const int l = sw.number_of_pairs();
1460   if (CanonicalizeNodes && l == 1 && compilation()->env()->comp_level() != CompLevel_full_profile) {
1461     // total of 2 successors => use If instead of switch
1462     // Note: This code should go into the canonicalizer as soon as it can
1463     //       can handle canonicalized forms that contain more than one node.
1464     // simplify to If
1465     LookupswitchPair pair = sw.pair_at(0);
1466     Value key = append(new Constant(new IntConstant(pair.match())));
1467     BlockBegin* tsux = block_at(bci() + pair.offset());
1468     BlockBegin* fsux = block_at(bci() + sw.default_offset());
1469     bool is_bb = tsux->bci() < bci() || fsux->bci() < bci();
1470     // In case of loop invariant code motion or predicate insertion
1471     // before the body of a loop the state is needed
1472     ValueStack* state_before = copy_state_if_bb(is_bb);;
1473     append(new If(ipop(), If::eql, true, key, tsux, fsux, state_before, is_bb));
1474   } else {
1475     // collect successors & keys
1476     BlockList* sux = new BlockList(l + 1, NULL);
1477     intArray* keys = new intArray(l, l, 0);
1478     int i;
1479     bool has_bb = false;
1480     for (i = 0; i < l; i++) {
1481       LookupswitchPair pair = sw.pair_at(i);
1482       if (pair.offset() < 0) has_bb = true;
1483       sux->at_put(i, block_at(bci() + pair.offset()));
1484       keys->at_put(i, pair.match());
1485     }
1486     // add default successor
1487     if (sw.default_offset() < 0) has_bb = true;
1488     sux->at_put(i, block_at(bci() + sw.default_offset()));
1489     // In case of loop invariant code motion or predicate insertion
1490     // before the body of a loop the state is needed
1491     ValueStack* state_before = copy_state_if_bb(has_bb);
1492     Instruction* res = append(new LookupSwitch(ipop(), sux, keys, state_before, has_bb));
1493 #ifdef ASSERT
1494     if (res->as_Goto()) {
1495       for (i = 0; i < l; i++) {
1496         if (sux->at(i) == res->as_Goto()->sux_at(0)) {
1497           assert(res->as_Goto()->is_safepoint() == sw.pair_at(i).offset() < 0, "safepoint state of Goto returned by canonicalizer incorrect");
1498         }
1499       }
1500     }
1501 #endif
1502   }
1503 }

1505 void GraphBuilder::call_register_finalizer() {
1506   // If the receiver requires finalization then emit code to perform
1507   // the registration on return.
1508 
1509   // Gather some type information about the receiver
1510   Value receiver = state()->local_at(0);
1511   assert(receiver != NULL, "must have a receiver");
1512   ciType* declared_type = receiver->declared_type();
1513   ciType* exact_type = receiver->exact_type();
1514   if (exact_type == NULL &&
1515       receiver->as_Local() &&
1516       receiver->as_Local()->java_index() == 0) {
1517     ciInstanceKlass* ik = compilation()->method()->holder();
1518     if (ik->is_final()) {
1519       exact_type = ik;
1520     } else if (UseCHA && !(ik->has_subklass() || ik->is_interface())) {
1521       // test class is leaf class
1522       compilation()->dependency_recorder()->assert_leaf_type(ik);
1523       exact_type = ik;
1524     } else {
1525       declared_type = ik;
1526     }
1527   }
1528 
1529   // see if we know statically that registration isn't required
1530   bool needs_check = true;
1531   if (exact_type != NULL) {
1532     needs_check = exact_type->as_instance_klass()->has_finalizer();
1533   } else if (declared_type != NULL) {
1534     ciInstanceKlass* ik = declared_type->as_instance_klass();
1535     if (!Dependencies::has_finalizable_subclass(ik)) {
1536       compilation()->dependency_recorder()->assert_has_no_finalizable_subclasses(ik);
1537       needs_check = false;
1538     }
1539   }
1540 
1541   if (needs_check) {
1542     // Perform the registration of finalizable objects.
1543     ValueStack* state_before = copy_state_for_exception();
1544     load_local(objectType, 0);
1545     append_split(new Intrinsic(voidType, vmIntrinsics::_Object_init,
1546                                state()->pop_arguments(1),
1547                                true, state_before, true));
1548   }
1549 }

1552 void GraphBuilder::method_return(Value x, bool ignore_return) {
1553   if (RegisterFinalizersAtInit &&
1554       method()->intrinsic_id() == vmIntrinsics::_Object_init) {
1555     call_register_finalizer();
1556   }
1557 
1558   // The conditions for a memory barrier are described in Parse::do_exits().
1559   bool need_mem_bar = false;
1560   if (method()->is_object_constructor() &&
1561        (scope()->wrote_final() ||
1562          (AlwaysSafeConstructors && scope()->wrote_fields()) ||
1563          (support_IRIW_for_not_multiple_copy_atomic_cpu && scope()->wrote_volatile()))) {
1564     need_mem_bar = true;
1565   }
1566 
1567   BasicType bt = method()->return_type()->basic_type();
1568   switch (bt) {
1569     case T_BYTE:
1570     {
1571       Value shift = append(new Constant(new IntConstant(24)));
1572       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1573       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1574       break;
1575     }
1576     case T_SHORT:
1577     {
1578       Value shift = append(new Constant(new IntConstant(16)));
1579       x = append(new ShiftOp(Bytecodes::_ishl, x, shift));
1580       x = append(new ShiftOp(Bytecodes::_ishr, x, shift));
1581       break;
1582     }
1583     case T_CHAR:
1584     {
1585       Value mask = append(new Constant(new IntConstant(0xFFFF)));
1586       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1587       break;
1588     }
1589     case T_BOOLEAN:
1590     {
1591       Value mask = append(new Constant(new IntConstant(1)));
1592       x = append(new LogicOp(Bytecodes::_iand, x, mask));
1593       break;
1594     }
1595     default:
1596       break;
1597   }
1598 
1599   // Check to see whether we are inlining. If so, Return
1600   // instructions become Gotos to the continuation point.
1601   if (continuation() != NULL) {
1602 
1603     int invoke_bci = state()->caller_state()->bci();
1604 
1605     if (x != NULL  && !ignore_return) {
1606       ciMethod* caller = state()->scope()->caller()->method();
1607       Bytecodes::Code invoke_raw_bc = caller->raw_code_at_bci(invoke_bci);
1608       if (invoke_raw_bc == Bytecodes::_invokehandle || invoke_raw_bc == Bytecodes::_invokedynamic) {
1609         ciType* declared_ret_type = caller->get_declared_signature_at_bci(invoke_bci)->return_type();
1610         if (declared_ret_type->is_klass() && x->exact_type() == NULL &&
1611             x->declared_type() != declared_ret_type && declared_ret_type != compilation()->env()->Object_klass()) {
1612           x = append(new TypeCast(declared_ret_type->as_klass(), x, copy_state_before()));
1613         }
1614       }
1615     }
1616 
1617     assert(!method()->is_synchronized() || InlineSynchronizedMethods, "can not inline synchronized methods yet");
1618 
1619     if (compilation()->env()->dtrace_method_probes()) {
1620       // Report exit from inline methods
1621       Values* args = new Values(1);
1622       args->push(append(new Constant(new MethodConstant(method()))));
1623       append(new RuntimeCall(voidType, "dtrace_method_exit", CAST_FROM_FN_PTR(address, SharedRuntime::dtrace_method_exit), args));
1624     }
1625 
1626     // If the inlined method is synchronized, the monitor must be
1627     // released before we jump to the continuation block.
1628     if (method()->is_synchronized()) {
1629       assert(state()->locks_size() == 1, "receiver must be locked here");
1630       monitorexit(state()->lock_at(0), SynchronizationEntryBCI);
1631     }
1632 
1633     if (need_mem_bar) {
1634       append(new MemBar(lir_membar_storestore));
1635     }
1636 
1637     // State at end of inlined method is the state of the caller
1638     // without the method parameters on stack, including the
1639     // return value, if any, of the inlined method on operand stack.
1640     set_state(state()->caller_state()->copy_for_parsing());
1641     if (x != NULL) {
1642       if (!ignore_return) {
1643         state()->push(x->type(), x);
1644       }
1645       if (profile_return() && x->type()->is_object_kind()) {
1646         ciMethod* caller = state()->scope()->method();
1647         profile_return_type(x, method(), caller, invoke_bci);
1648       }
1649     }
1650     Goto* goto_callee = new Goto(continuation(), false);
1651 
1652     // See whether this is the first return; if so, store off some
1653     // of the state for later examination
1654     if (num_returns() == 0) {
1655       set_inline_cleanup_info();
1656     }
1657 
1658     // The current bci() is in the wrong scope, so use the bci() of
1659     // the continuation point.
1660     append_with_bci(goto_callee, scope_data()->continuation()->bci());
1661     incr_num_returns();
1662     return;
1663   }
1664 
1665   state()->truncate_stack(0);
1666   if (method()->is_synchronized()) {
1667     // perform the unlocking before exiting the method
1668     Value receiver;
1669     if (!method()->is_static()) {
1670       receiver = _initial_state->local_at(0);
1671     } else {
1672       receiver = append(new Constant(new ClassConstant(method()->holder())));
1673     }
1674     append_split(new MonitorExit(receiver, state()->unlock()));
1675   }
1676 
1677   if (need_mem_bar) {
1678       append(new MemBar(lir_membar_storestore));
1679   }
1680 
1681   assert(!ignore_return, "Ignoring return value works only for inlining");
1682   append(new Return(x));
1683 }

1752   }
1753 
1754   if (field->is_final() && (code == Bytecodes::_putfield || code == Bytecodes::_withfield)) {
1755     scope()->set_wrote_final();
1756   }
1757 
1758   if (code == Bytecodes::_putfield || code == Bytecodes::_withfield) {
1759     scope()->set_wrote_fields();
1760     if (field->is_volatile()) {
1761       scope()->set_wrote_volatile();
1762     }
1763   }
1764 
1765   const int offset = !needs_patching ? field->offset() : -1;
1766   switch (code) {
1767     case Bytecodes::_getstatic: {
1768       // check for compile-time constants, i.e., initialized static final fields
1769       Value constant = NULL;
1770       if (field->is_static_constant() && !PatchALot) {
1771         ciConstant field_value = field->constant_value();
1772         assert(!field->is_stable() || !field_value.is_null_or_zero(),
1773                "stable static w/ default value shouldn't be a constant");
1774         constant = make_constant(field_value, field);
1775       }
1776       if (constant != NULL) {
1777         push(type, append(constant));
1778       } else {
1779         if (state_before == NULL) {
1780           state_before = copy_state_for_exception();
1781         }
1782         LoadField* load_field = new LoadField(append(obj), offset, field, true,
1783                                         state_before, needs_patching);
1784         push(type, append(load_field));
1785       }
1786       break;
1787     }
1788     case Bytecodes::_putstatic: {
1789       Value val = pop(type);
1790       val->set_escaped();
1791       if (state_before == NULL) {
1792         state_before = copy_state_for_exception();
1793       }
1794       if (field_type == T_BOOLEAN) {
1795         Value mask = append(new Constant(new IntConstant(1)));
1796         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1797       }
1798       append(new StoreField(append(obj), offset, field, val, true, state_before, needs_patching));
1799       break;
1800     }
1801     case Bytecodes::_getfield: {
1802       // Check for compile-time constants, i.e., trusted final non-static fields.
1803       Value constant = NULL;
1804       if (state_before == NULL && field->is_flattened()) {
1805         // Save the entire state and re-execute on deopt when accessing flattened fields
1806         assert(Interpreter::bytecode_should_reexecute(code), "should reexecute");
1807         state_before = copy_state_before();
1808       }
1809       obj = apop();
1810       ObjectType* obj_type = obj->type()->as_ObjectType();
1811       if (field->is_constant() && !field->is_flattened() && obj_type->is_constant() && !PatchALot) {
1812         ciObject* const_oop = obj_type->constant_value();
1813         if (!const_oop->is_null_object() && const_oop->is_loaded()) {
1814           ciConstant field_value = field->constant_value_of(const_oop);
1815           if (field_value.is_valid()) {
1816             if (field->signature()->is_Q_signature() && field_value.is_null_or_zero()) {
1817               // Non-flattened inline type field. Replace null by the default value.
1818               constant = new Constant(new InstanceConstant(field->type()->as_inline_klass()->default_instance()));
1819             } else {
1820               constant = make_constant(field_value, field);
1821             }
1822             // For CallSite objects add a dependency for invalidation of the optimization.
1823             if (field->is_call_site_target()) {
1824               ciCallSite* call_site = const_oop->as_call_site();
1825               if (!call_site->is_fully_initialized_constant_call_site()) {
1826                 ciMethodHandle* target = field_value.as_object()->as_method_handle();
1827                 dependency_recorder()->assert_call_site_target_value(call_site, target);
1828               }
1829             }
1830           }
1831         }
1832       }
1833       if (constant != NULL) {
1834         push(type, append(constant));
1835       } else {
1836         if (state_before == NULL) {
1837           state_before = copy_state_for_exception();
1838         }
1839         if (!field->is_flattened()) {
1840           LoadField* load = new LoadField(obj, offset, field, false, state_before, needs_patching);
1841           Value replacement = !needs_patching ? _memory->load(load) : load;
1842           if (replacement != load) {
1843             assert(replacement->is_linked() || !replacement->can_be_linked(), "should already by linked");
1844             // Writing an (integer) value to a boolean, byte, char or short field includes an implicit narrowing
1845             // conversion. Emit an explicit conversion here to get the correct field value after the write.
1846             BasicType bt = field->type()->basic_type();
1847             switch (bt) {
1848             case T_BOOLEAN:
1849             case T_BYTE:
1850               replacement = append(new Convert(Bytecodes::_i2b, replacement, as_ValueType(bt)));
1851               break;
1852             case T_CHAR:
1853               replacement = append(new Convert(Bytecodes::_i2c, replacement, as_ValueType(bt)));
1854               break;
1855             case T_SHORT:
1856               replacement = append(new Convert(Bytecodes::_i2s, replacement, as_ValueType(bt)));
1857               break;
1858             default:
1859               break;
1860             }
1861             push(type, replacement);
1862           } else {
1863             push(type, append(load));
1864           }
1865         } else { // flattened field, not optimized solution: re-instantiate the flattened value
1866           assert(field->type()->is_inlinetype(), "Sanity check");
1867           ciInlineKlass* inline_klass = field->type()->as_inline_klass();
1868           int flattening_offset = field->offset() - inline_klass->first_field_offset();
1869           assert(field->type()->is_inlinetype(), "Sanity check");
1870           scope()->set_wrote_final();
1871           scope()->set_wrote_fields();
1872           NewInlineTypeInstance* new_instance = new NewInlineTypeInstance(inline_klass, state_before, false);
1873           _memory->new_instance(new_instance);
1874           apush(append_split(new_instance));
1875           copy_inline_content(inline_klass, obj, field->offset(), new_instance, inline_klass->first_field_offset(),
1876                        state_before, needs_patching);
1877         }
1878       }
1879       break;
1880     }
1881     case Bytecodes::_withfield:
1882     case Bytecodes::_putfield: {
1883       Value val = pop(type);
1884       val->set_escaped();
1885       obj = apop();
1886       if (state_before == NULL) {
1887         state_before = copy_state_for_exception();
1888       }
1889       if (field->type()->basic_type() == T_BOOLEAN) {
1890         Value mask = append(new Constant(new IntConstant(1)));
1891         val = append(new LogicOp(Bytecodes::_iand, val, mask));
1892       }
1893 
1894       if (!field->is_flattened()) {
1895         StoreField* store = new StoreField(obj, offset, field, val, false, state_before, needs_patching);
1896         if (!needs_patching) store = _memory->store(store);
1897         if (store != NULL) {
1898           append(store);
1899         }
1900       } else {
1901         assert(field->type()->is_inlinetype(), "Sanity check");
1902         ciInlineKlass* inline_klass = field->type()->as_inline_klass();
1903         int flattening_offset = field->offset() - inline_klass->first_field_offset();
1904         copy_inline_content(inline_klass, val, inline_klass->first_field_offset(), obj, field->offset(),
1905                    state_before, needs_patching);
1906       }
1907       break;
1908     }
1909     default:
1910       ShouldNotReachHere();
1911       break;
1912   }
1913 }

1915 // Baseline version of withfield, allocate every time
1916 void GraphBuilder::withfield(int field_index)
1917 {
1918   bool will_link;
1919   ciField* field_modify = stream()->get_field(will_link);
1920   ciInstanceKlass* holder = field_modify->holder();
1921   BasicType field_type = field_modify->type()->basic_type();
1922   ValueType* type = as_ValueType(field_type);
1923 
1924   // call will_link again to determine if the field is valid.
1925   const bool needs_patching = !holder->is_loaded() ||
1926                               !field_modify->will_link(method(), Bytecodes::_withfield) ||
1927                               PatchALot;
1928 
1929 
1930   scope()->set_wrote_final();
1931   scope()->set_wrote_fields();
1932 
1933   const int offset = !needs_patching ? field_modify->offset() : -1;
1934 
1935   if (!holder->is_loaded()
1936       || needs_patching /* FIXME: 8228634 - field_modify->will_link() may incorrectly return false */
1937       ) {
1938     ValueStack* state_before = copy_state_before();
1939     Value val = pop(type);
1940     Value obj = apop();
1941     apush(append_split(new WithField(state_before)));
1942     return;
1943   }
1944   ValueStack* state_before = copy_state_before();
1945 
1946   Value val = pop(type);
1947   Value obj = apop();
1948 
1949   if (!needs_patching && obj->is_optimizable_for_withfield()) {
1950     int astore_index;
1951     ciBytecodeStream s(method());
1952     s.force_bci(bci());
1953     s.next();
1954     switch (s.cur_bc()) {
1955     case Bytecodes::_astore:    astore_index = s.get_index(); break;
1956     case Bytecodes::_astore_0:  astore_index = 0; break;
1957     case Bytecodes::_astore_1:  astore_index = 1; break;
1958     case Bytecodes::_astore_2:  astore_index = 2; break;
1959     case Bytecodes::_astore_3:  astore_index = 3; break;
1960     default: astore_index = -1;
1961     }
1962 
1963     if (astore_index >= 0 && obj == state()->local_at(astore_index)) {
1964       // We have a sequence like this, where we load a value object from a local slot,
1965       // and overwrite the same local slot with a modified copy of the inline object.
1966       //      defaultvalue #1 // class compiler/valhalla/valuetypes/MyValue1
1967       //      astore 9
1968       //      ...
1969       //      iload_0
1970       //      aload 9
1971       //      swap
1972       //      withfield #7 // Field x:I
1973       //      astore 9
1974       // If this object was created by defaultvalue, and has not escaped, and is not stored
1975       // in any other local slots, we can effectively treat the withfield/astore
1976       // sequence as a single putfield bytecode.
1977       push(objectType, obj);
1978       push(type, val);
1979       access_field(Bytecodes::_withfield);
1980       stream()->next(); // skip the next astore/astore_n bytecode.
1981       return;
1982     }
1983   }
1984 
1985   assert(holder->is_inlinetype(), "must be an inline klass");
1986   // Save the entire state and re-execute on deopt when executing withfield
1987   state_before->set_should_reexecute(true);
1988   NewInlineTypeInstance* new_instance = new NewInlineTypeInstance(holder->as_inline_klass(), state_before, false);
1989   _memory->new_instance(new_instance);
1990   apush(append_split(new_instance));
1991 
1992   for (int i = 0; i < holder->nof_nonstatic_fields(); i++) {
1993     ciField* field = holder->nonstatic_field_at(i);
1994     int off = field->offset();
1995 
1996     if (field->offset() != offset) {
1997       if (field->is_flattened()) {
1998         assert(field->type()->is_inlinetype(), "Sanity check");
1999         assert(field->type()->is_inlinetype(), "Only inline types can be flattened");
2000         ciInlineKlass* vk = field->type()->as_inline_klass();
2001         copy_inline_content(vk, obj, off, new_instance, vk->first_field_offset(), state_before, needs_patching);
2002       } else {
2003         // Only load those fields who are not modified
2004         LoadField* load = new LoadField(obj, off, field, false, state_before, needs_patching);
2005         Value replacement = append(load);
2006         StoreField* store = new StoreField(new_instance, off, field, replacement, false, state_before, needs_patching);
2007         append(store);
2008       }
2009     }
2010   }
2011 
2012   // Field to modify
2013   if (field_modify->type()->basic_type() == T_BOOLEAN) {
2014     Value mask = append(new Constant(new IntConstant(1)));
2015     val = append(new LogicOp(Bytecodes::_iand, val, mask));
2016   }
2017   if (field_modify->is_flattened()) {
2018     assert(field_modify->type()->is_inlinetype(), "Only inline types can be flattened");
2019     ciInlineKlass* vk = field_modify->type()->as_inline_klass();
2020     copy_inline_content(vk, val, vk->first_field_offset(), new_instance, field_modify->offset(), state_before, needs_patching);
2021   } else {
2022     StoreField* store = new StoreField(new_instance, offset, field_modify, val, false, state_before, needs_patching);
2023     append(store);
2024   }
2025 }

2032 // How many arguments do we want to profile?
2033 Values* GraphBuilder::args_list_for_profiling(ciMethod* target, int& start, bool may_have_receiver) {
2034   int n = 0;
2035   bool has_receiver = may_have_receiver && Bytecodes::has_receiver(method()->java_code_at_bci(bci()));
2036   start = has_receiver ? 1 : 0;
2037   if (profile_arguments()) {
2038     ciProfileData* data = method()->method_data()->bci_to_data(bci());
2039     if (data != NULL && (data->is_CallTypeData() || data->is_VirtualCallTypeData())) {
2040       n = data->is_CallTypeData() ? data->as_CallTypeData()->number_of_arguments() : data->as_VirtualCallTypeData()->number_of_arguments();
2041     }
2042   }
2043   // If we are inlining then we need to collect arguments to profile parameters for the target
2044   if (profile_parameters() && target != NULL) {
2045     if (target->method_data() != NULL && target->method_data()->parameters_type_data() != NULL) {
2046       // The receiver is profiled on method entry so it's included in
2047       // the number of parameters but here we're only interested in
2048       // actual arguments.
2049       n = MAX2(n, target->method_data()->parameters_type_data()->number_of_parameters() - start);
2050     }
2051   }
2052   if (n > 0) {
2053     return new Values(n);
2054   }
2055   return NULL;
2056 }

2087 void GraphBuilder::invoke(Bytecodes::Code code) {
2088   bool will_link;
2089   ciSignature* declared_signature = NULL;
2090   ciMethod*             target = stream()->get_method(will_link, &declared_signature);
2091   ciKlass*              holder = stream()->get_declared_method_holder();
2092   const Bytecodes::Code bc_raw = stream()->cur_bc_raw();
2093   assert(declared_signature != NULL, "cannot be null");
2094   assert(will_link == target->is_loaded(), "");
2095 
2096   ciInstanceKlass* klass = target->holder();
2097   assert(!target->is_loaded() || klass->is_loaded(), "loaded target must imply loaded klass");
2098 
2099   // check if CHA possible: if so, change the code to invoke_special
2100   ciInstanceKlass* calling_klass = method()->holder();
2101   ciInstanceKlass* callee_holder = ciEnv::get_instance_klass_for_declared_method_holder(holder);
2102   ciInstanceKlass* actual_recv = callee_holder;
2103 
2104   CompileLog* log = compilation()->log();
2105   if (log != NULL)
2106       log->elem("call method='%d' instr='%s'",
2107                 log->identify(target),
2108                 Bytecodes::name(code));
2109 
2110   // invoke-special-super
2111   if (bc_raw == Bytecodes::_invokespecial && !target->is_object_constructor()) {
2112     ciInstanceKlass* sender_klass =
2113           calling_klass->is_unsafe_anonymous() ? calling_klass->unsafe_anonymous_host() :
2114                                                  calling_klass;
2115     if (sender_klass->is_interface()) {
2116       int index = state()->stack_size() - (target->arg_size_no_receiver() + 1);
2117       Value receiver = state()->stack_at(index);
2118       CheckCast* c = new CheckCast(sender_klass, receiver, copy_state_before());
2119       c->set_invokespecial_receiver_check();
2120       state()->stack_at_put(index, append_split(c));
2121     }
2122   }
2123 
2124   // Some methods are obviously bindable without any type checks so
2125   // convert them directly to an invokespecial or invokestatic.
2126   if (target->is_loaded() && !target->is_abstract() && target->can_be_statically_bound()) {
2127     switch (bc_raw) {
2128     case Bytecodes::_invokevirtual:
2129       code = Bytecodes::_invokespecial;
2130       break;
2131     case Bytecodes::_invokehandle:
2132       code = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokespecial;
2133       break;
2134     default:
2135       break;
2136     }
2137   } else {
2138     if (bc_raw == Bytecodes::_invokehandle) {
2139       assert(!will_link, "should come here only for unlinked call");
2140       code = Bytecodes::_invokespecial;
2141     }
2142   }
2143 
2144   // Push appendix argument (MethodType, CallSite, etc.), if one.
2145   bool patch_for_appendix = false;
2146   int patching_appendix_arg = 0;
2147   if (Bytecodes::has_optional_appendix(bc_raw) && (!will_link || PatchALot)) {
2148     Value arg = append(new Constant(new ObjectConstant(compilation()->env()->unloaded_ciinstance()), copy_state_before()));
2149     apush(arg);
2150     patch_for_appendix = true;
2151     patching_appendix_arg = (will_link && stream()->has_appendix()) ? 0 : 1;
2152   } else if (stream()->has_appendix()) {
2153     ciObject* appendix = stream()->get_appendix();
2154     Value arg = append(new Constant(new ObjectConstant(appendix)));
2155     apush(arg);
2156   }
2157 
2158   ciMethod* cha_monomorphic_target = NULL;
2159   ciMethod* exact_target = NULL;
2160   Value better_receiver = NULL;
2161   if (UseCHA && DeoptC1 && target->is_loaded() &&
2162       !(// %%% FIXME: Are both of these relevant?
2163         target->is_method_handle_intrinsic() ||
2164         target->is_compiled_lambda_form()) &&
2165       !patch_for_appendix) {
2166     Value receiver = NULL;
2167     ciInstanceKlass* receiver_klass = NULL;
2168     bool type_is_exact = false;
2169     // try to find a precise receiver type
2170     if (will_link && !target->is_static()) {
2171       int index = state()->stack_size() - (target->arg_size_no_receiver() + 1);
2172       receiver = state()->stack_at(index);
2173       ciType* type = receiver->exact_type();
2174       if (type != NULL && type->is_loaded() &&
2175           type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {
2176         receiver_klass = (ciInstanceKlass*) type;
2177         type_is_exact = true;
2178       }
2179       if (type == NULL) {
2180         type = receiver->declared_type();
2181         if (type != NULL && type->is_loaded() &&
2182             type->is_instance_klass() && !type->as_instance_klass()->is_interface()) {
2183           receiver_klass = (ciInstanceKlass*) type;
2184           if (receiver_klass->is_leaf_type() && !receiver_klass->is_final()) {
2185             // Insert a dependency on this type since
2186             // find_monomorphic_target may assume it's already done.
2187             dependency_recorder()->assert_leaf_type(receiver_klass);
2188             type_is_exact = true;
2189           }
2190         }
2191       }
2192     }
2193     if (receiver_klass != NULL && type_is_exact &&
2194         receiver_klass->is_loaded() && code != Bytecodes::_invokespecial) {
2195       // If we have the exact receiver type we can bind directly to
2196       // the method to call.
2197       exact_target = target->resolve_invoke(calling_klass, receiver_klass);
2198       if (exact_target != NULL) {
2199         target = exact_target;
2200         code = Bytecodes::_invokespecial;
2201       }
2202     }
2203     if (receiver_klass != NULL &&
2204         receiver_klass->is_subtype_of(actual_recv) &&
2205         actual_recv->is_initialized()) {
2206       actual_recv = receiver_klass;
2207     }
2208 
2209     if ((code == Bytecodes::_invokevirtual && callee_holder->is_initialized()) ||
2210         (code == Bytecodes::_invokeinterface && callee_holder->is_initialized() && !actual_recv->is_interface())) {
2211       // Use CHA on the receiver to select a more precise method.
2212       cha_monomorphic_target = target->find_monomorphic_target(calling_klass, callee_holder, actual_recv);
2213     } else if (code == Bytecodes::_invokeinterface && callee_holder->is_loaded() && receiver != NULL) {
2214       assert(callee_holder->is_interface(), "invokeinterface to non interface?");
2215       // If there is only one implementor of this interface then we
2216       // may be able bind this invoke directly to the implementing
2217       // klass but we need both a dependence on the single interface
2218       // and on the method we bind to.  Additionally since all we know
2219       // about the receiver type is the it's supposed to implement the
2220       // interface we have to insert a check that it's the class we
2221       // expect.  Interface types are not checked by the verifier so
2222       // they are roughly equivalent to Object.
2223       // The number of implementors for declared_interface is less or
2224       // equal to the number of implementors for target->holder() so
2225       // if number of implementors of target->holder() == 1 then
2226       // number of implementors for decl_interface is 0 or 1. If
2227       // it's 0 then no class implements decl_interface and there's
2228       // no point in inlining.
2229       ciInstanceKlass* declared_interface = callee_holder;
2230       ciInstanceKlass* singleton = declared_interface->unique_implementor();
2231       if (singleton != NULL &&
2232           (!target->is_default_method() || target->is_overpass()) /* CHA doesn't support default methods yet. */ ) {
2233         assert(singleton != declared_interface, "not a unique implementor");
2234         cha_monomorphic_target = target->find_monomorphic_target(calling_klass, declared_interface, singleton);
2235         if (cha_monomorphic_target != NULL) {
2236           if (cha_monomorphic_target->holder() != compilation()->env()->Object_klass()) {
2237             // If CHA is able to bind this invoke then update the class
2238             // to match that class, otherwise klass will refer to the
2239             // interface.
2240             klass = cha_monomorphic_target->holder();
2241             actual_recv = declared_interface;
2242 
2243             // insert a check it's really the expected class.
2244             CheckCast* c = new CheckCast(klass, receiver, copy_state_for_exception());
2245             c->set_incompatible_class_change_check();
2246             c->set_direct_compare(klass->is_final());
2247             // pass the result of the checkcast so that the compiler has
2248             // more accurate type info in the inlinee
2249             better_receiver = append_split(c);
2250           } else {
2251             cha_monomorphic_target = NULL; // subtype check against Object is useless
2252           }
2253         }
2254       }
2255     }
2256   }
2257 
2258   if (cha_monomorphic_target != NULL) {
2259     assert(!target->can_be_statically_bound() || target == cha_monomorphic_target, "");
2260     assert(!cha_monomorphic_target->is_abstract(), "");
2261     if (!cha_monomorphic_target->can_be_statically_bound(actual_recv)) {
2262       // If we inlined because CHA revealed only a single target method,
2263       // then we are dependent on that target method not getting overridden
2264       // by dynamic class loading.  Be sure to test the "static" receiver
2265       // dest_method here, as opposed to the actual receiver, which may
2266       // falsely lead us to believe that the receiver is final or private.
2267       dependency_recorder()->assert_unique_concrete_method(actual_recv, cha_monomorphic_target);
2268     }
2269     code = Bytecodes::_invokespecial;
2270   }
2271 
2272   // check if we could do inlining
2273   if (!PatchALot && Inline && target->is_loaded() &&
2274       (klass->is_initialized() || (klass->is_interface() && target->holder()->is_initialized()))
2275       && !patch_for_appendix) {
2276     // callee is known => check if we have static binding
2277     if (code == Bytecodes::_invokestatic  ||
2278         code == Bytecodes::_invokespecial ||
2279         (code == Bytecodes::_invokevirtual && target->is_final_method()) ||
2280         code == Bytecodes::_invokedynamic) {
2281       ciMethod* inline_target = (cha_monomorphic_target != NULL) ? cha_monomorphic_target : target;
2282       // static binding => check if callee is ok
2283       bool success = try_inline(inline_target, (cha_monomorphic_target != NULL) || (exact_target != NULL), false, code, better_receiver);
2284 
2285       CHECK_BAILOUT();
2286       clear_inline_bailout();
2287 
2288       if (success) {
2289         // Register dependence if JVMTI has either breakpoint
2290         // setting or hotswapping of methods capabilities since they may
2291         // cause deoptimization.
2292         if (compilation()->env()->jvmti_can_hotswap_or_post_breakpoint()) {
2293           dependency_recorder()->assert_evol_method(inline_target);
2294         }
2295         return;
2296       }
2297     } else {
2298       print_inlining(target, "no static binding", /*success*/ false);
2299     }
2300   } else {
2301     print_inlining(target, "not inlineable", /*success*/ false);
2302   }
2303 
2304   // If we attempted an inline which did not succeed because of a
2305   // bailout during construction of the callee graph, the entire
2306   // compilation has to be aborted. This is fairly rare and currently
2307   // seems to only occur for jasm-generated classes which contain
2308   // jsr/ret pairs which are not associated with finally clauses and
2309   // do not have exception handlers in the containing method, and are
2310   // therefore not caught early enough to abort the inlining without
2311   // corrupting the graph. (We currently bail out with a non-empty
2312   // stack at a ret in these situations.)
2313   CHECK_BAILOUT();
2314 
2315   // inlining not successful => standard invoke
2316   ValueType* result_type = as_ValueType(declared_signature->return_type());
2317   ValueStack* state_before = copy_state_exhandling();
2318 
2319   // The bytecode (code) might change in this method so we are checking this very late.
2320   const bool has_receiver =
2321     code == Bytecodes::_invokespecial   ||
2322     code == Bytecodes::_invokevirtual   ||
2323     code == Bytecodes::_invokeinterface;
2324   Values* args = state()->pop_arguments(target->arg_size_no_receiver() + patching_appendix_arg);
2325   Value recv = has_receiver ? apop() : NULL;
2326   int vtable_index = Method::invalid_vtable_index;
2327 
2328   // A null check is required here (when there is a receiver) for any of the following cases
2329   // - invokespecial, always need a null check.
2330   // - invokevirtual, when the target is final and loaded. Calls to final targets will become optimized
2331   //   and require null checking. If the target is loaded a null check is emitted here.
2332   //   If the target isn't loaded the null check must happen after the call resolution. We achieve that
2333   //   by using the target methods unverified entry point (see CompiledIC::compute_monomorphic_entry).
2334   //   (The JVM specification requires that LinkageError must be thrown before a NPE. An unloaded target may
2335   //   potentially fail, and can't have the null check before the resolution.)
2336   // - A call that will be profiled. (But we can't add a null check when the target is unloaded, by the same
2337   //   reason as above, so calls with a receiver to unloaded targets can't be profiled.)
2338   //
2339   // Normal invokevirtual will perform the null check during lookup
2340 
2341   bool need_null_check = (code == Bytecodes::_invokespecial) ||
2342       (target->is_loaded() && (target->is_final_method() || (is_profiling() && profile_calls())));
2343 
2344   if (need_null_check) {
2345     if (recv != NULL) {
2346       null_check(recv);
2347     }
2348 
2349     if (is_profiling()) {
2350       // Note that we'd collect profile data in this method if we wanted it.
2351       compilation()->set_would_profile(true);
2352 
2353       if (profile_calls()) {
2354         assert(cha_monomorphic_target == NULL || exact_target == NULL, "both can not be set");
2355         ciKlass* target_klass = NULL;
2356         if (cha_monomorphic_target != NULL) {
2357           target_klass = cha_monomorphic_target->holder();
2358         } else if (exact_target != NULL) {
2359           target_klass = exact_target->holder();
2360         }
2361         profile_call(target, recv, target_klass, collect_args_for_profiling(args, NULL, false), false);
2362       }
2363     }
2364   }
2365 
2366   if (recv != NULL) {
2367     recv->set_escaped();
2368   }
2369   for (int i=0; i<args->length(); i++) {
2370     args->at(0)->set_escaped();
2371   }
2372 
2373   Invoke* result = new Invoke(code, result_type, recv, args, vtable_index, target, state_before,
2374                               declared_signature->return_type()->is_inlinetype());
2375   // push result
2376   append_split(result);
2377 
2378   if (result_type != voidType) {
2379     if (method()->is_strict()) {
2380       push(result_type, round_fp(result));
2381     } else {
2382       push(result_type, result);
2383     }
2384   }
2385   if (profile_return() && result_type->is_object_kind()) {
2386     profile_return_type(result, target);
2387   }
2388 }

2571 Instruction* GraphBuilder::append_with_bci(Instruction* instr, int bci) {
2572   Canonicalizer canon(compilation(), instr, bci);
2573   Instruction* i1 = canon.canonical();
2574   if (i1->is_linked() || !i1->can_be_linked()) {
2575     // Canonicalizer returned an instruction which was already
2576     // appended so simply return it.
2577     return i1;
2578   }
2579 
2580   if (UseLocalValueNumbering) {
2581     // Lookup the instruction in the ValueMap and add it to the map if
2582     // it's not found.
2583     Instruction* i2 = vmap()->find_insert(i1);
2584     if (i2 != i1) {
2585       // found an entry in the value map, so just return it.
2586       assert(i2->is_linked(), "should already be linked");
2587       return i2;
2588     }
2589     ValueNumberingEffects vne(vmap());
2590     i1->visit(&vne);
2591   }
2592 
2593   // i1 was not eliminated => append it
2594   assert(i1->next() == NULL, "shouldn't already be linked");
2595   _last = _last->set_next(i1, canon.bci());
2596 
2597   if (++_instruction_count >= InstructionCountCutoff && !bailed_out()) {
2598     // set the bailout state but complete normal processing.  We
2599     // might do a little more work before noticing the bailout so we
2600     // want processing to continue normally until it's noticed.
2601     bailout("Method and/or inlining is too large");
2602   }
2603 
2604 #ifndef PRODUCT
2605   if (PrintIRDuringConstruction) {
2606     InstructionPrinter ip;
2607     ip.print_line(i1);
2608     if (Verbose) {
2609       state()->print();
2610     }
2611   }
2612 #endif
2613 
2614   // save state after modification of operand stack for StateSplit instructions
2615   StateSplit* s = i1->as_StateSplit();
2616   if (s != NULL) {
2617     if (EliminateFieldAccess) {
2618       Intrinsic* intrinsic = s->as_Intrinsic();
2619       if (s->as_Invoke() != NULL || (intrinsic && !intrinsic->preserves_state())) {
2620         _memory->kill();
2621       }
2622     }
2623     s->set_state(state()->copy(ValueStack::StateAfter, canon.bci()));
2624   }
2625 
2626   // set up exception handlers for this instruction if necessary
2627   if (i1->can_trap()) {
2628     i1->set_exception_handlers(handle_exception(i1));
2629     assert(i1->exception_state() != NULL || !i1->needs_exception_state() || bailed_out(), "handle_exception must set exception state");
2630   }
2631   return i1;
2632 }

2666 XHandlers* GraphBuilder::handle_exception(Instruction* instruction) {
2667   if (!has_handler() && (!instruction->needs_exception_state() || instruction->exception_state() != NULL)) {
2668     assert(instruction->exception_state() == NULL
2669            || instruction->exception_state()->kind() == ValueStack::EmptyExceptionState
2670            || (instruction->exception_state()->kind() == ValueStack::ExceptionState && _compilation->env()->should_retain_local_variables()),
2671            "exception_state should be of exception kind");
2672     return new XHandlers();
2673   }
2674 
2675   XHandlers*  exception_handlers = new XHandlers();
2676   ScopeData*  cur_scope_data = scope_data();
2677   ValueStack* cur_state = instruction->state_before();
2678   ValueStack* prev_state = NULL;
2679   int scope_count = 0;
2680 
2681   assert(cur_state != NULL, "state_before must be set");
2682   do {
2683     int cur_bci = cur_state->bci();
2684     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2685     assert(cur_bci == SynchronizationEntryBCI || cur_bci == cur_scope_data->stream()->cur_bci(), "invalid bci");
2686 
2687     // join with all potential exception handlers
2688     XHandlers* list = cur_scope_data->xhandlers();
2689     const int n = list->length();
2690     for (int i = 0; i < n; i++) {
2691       XHandler* h = list->handler_at(i);
2692       if (h->covers(cur_bci)) {
2693         // h is a potential exception handler => join it
2694         compilation()->set_has_exception_handlers(true);
2695 
2696         BlockBegin* entry = h->entry_block();
2697         if (entry == block()) {
2698           // It's acceptable for an exception handler to cover itself
2699           // but we don't handle that in the parser currently.  It's
2700           // very rare so we bailout instead of trying to handle it.
2701           BAILOUT_("exception handler covers itself", exception_handlers);
2702         }
2703         assert(entry->bci() == h->handler_bci(), "must match");
2704         assert(entry->bci() == -1 || entry == cur_scope_data->block_at(entry->bci()), "blocks must correspond");
2705 
2706         // previously this was a BAILOUT, but this is not necessary
2707         // now because asynchronous exceptions are not handled this way.
2708         assert(entry->state() == NULL || cur_state->total_locks_size() == entry->state()->total_locks_size(), "locks do not match");
2709 
2710         // xhandler start with an empty expression stack
2711         if (cur_state->stack_size() != 0) {
2712           cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2713         }
2714         if (instruction->exception_state() == NULL) {
2715           instruction->set_exception_state(cur_state);
2716         }
2717 
2718         // Note: Usually this join must work. However, very
2719         // complicated jsr-ret structures where we don't ret from
2720         // the subroutine can cause the objects on the monitor
2721         // stacks to not match because blocks can be parsed twice.
2722         // The only test case we've seen so far which exhibits this
2723         // problem is caught by the infinite recursion test in
2724         // GraphBuilder::jsr() if the join doesn't work.
2725         if (!entry->try_merge(cur_state)) {
2726           BAILOUT_("error while joining with exception handler, prob. due to complicated jsr/rets", exception_handlers);
2727         }
2728 
2729         // add current state for correct handling of phi functions at begin of xhandler
2730         int phi_operand = entry->add_exception_state(cur_state);
2731 
2732         // add entry to the list of xhandlers of this block
2733         _block->add_exception_handler(entry);
2734 
2735         // add back-edge from xhandler entry to this block
2736         if (!entry->is_predecessor(_block)) {
2737           entry->add_predecessor(_block);
2738         }
2739 
2740         // clone XHandler because phi_operand and scope_count can not be shared
2741         XHandler* new_xhandler = new XHandler(h);
2742         new_xhandler->set_phi_operand(phi_operand);
2743         new_xhandler->set_scope_count(scope_count);
2744         exception_handlers->append(new_xhandler);
2745 
2746         // fill in exception handler subgraph lazily
2747         assert(!entry->is_set(BlockBegin::was_visited_flag), "entry must not be visited yet");
2748         cur_scope_data->add_to_work_list(entry);
2749 
2750         // stop when reaching catchall
2751         if (h->catch_type() == 0) {
2752           return exception_handlers;
2753         }
2754       }
2755     }
2756 
2757     if (exception_handlers->length() == 0) {
2758       // This scope and all callees do not handle exceptions, so the local
2759       // variables of this scope are not needed. However, the scope itself is
2760       // required for a correct exception stack trace -> clear out the locals.
2761       if (_compilation->env()->should_retain_local_variables()) {
2762         cur_state = cur_state->copy(ValueStack::ExceptionState, cur_state->bci());
2763       } else {
2764         cur_state = cur_state->copy(ValueStack::EmptyExceptionState, cur_state->bci());
2765       }
2766       if (prev_state != NULL) {
2767         prev_state->set_caller_state(cur_state);
2768       }
2769       if (instruction->exception_state() == NULL) {
2770         instruction->set_exception_state(cur_state);
2771       }
2772     }
2773 
2774     // Set up iteration for next time.
2775     // If parsing a jsr, do not grab exception handlers from the
2776     // parent scopes for this method (already got them, and they
2777     // needed to be cloned)
2778 
2779     while (cur_scope_data->parsing_jsr()) {
2780       cur_scope_data = cur_scope_data->parent();
2781     }
2782 
2783     assert(cur_scope_data->scope() == cur_state->scope(), "scopes do not match");
2784     assert(cur_state->locks_size() == 0 || cur_state->locks_size() == 1, "unlocking must be done in a catchall exception handler");
2785 
2786     prev_state = cur_state;
2787     cur_state = cur_state->caller_state();
2788     cur_scope_data = cur_scope_data->parent();
2789     scope_count++;
2790   } while (cur_scope_data != NULL);
2791 
2792   return exception_handlers;
2793 }

2814 Value PhiSimplifier::simplify(Value v) {
2815   Phi* phi = v->as_Phi();
2816 
2817   if (phi == NULL) {
2818     // no phi function
2819     return v;
2820   } else if (v->has_subst()) {
2821     // already substituted; subst can be phi itself -> simplify
2822     return simplify(v->subst());
2823   } else if (phi->is_set(Phi::cannot_simplify)) {
2824     // already tried to simplify phi before
2825     return phi;
2826   } else if (phi->is_set(Phi::visited)) {
2827     // break cycles in phi functions
2828     return phi;
2829   } else if (phi->type()->is_illegal()) {
2830     // illegal phi functions are ignored anyway
2831     return phi;
2832 
2833   } else {
2834     // mark phi function as processed to break cycles in phi functions
2835     phi->set(Phi::visited);
2836 
2837     // simplify x = [y, x] and x = [y, y] to y
2838     Value subst = NULL;
2839     int opd_count = phi->operand_count();
2840     for (int i = 0; i < opd_count; i++) {

2842       assert(opd != NULL, "Operand must exist!");

2849       }

2852       assert(new_opd != NULL, "Simplified operand must exist!");
2853 
2854       if (new_opd != phi && new_opd != subst) {
2855         if (subst == NULL) {
2856           subst = new_opd;
2857         } else {
2858           // no simplification possible
2859           phi->set(Phi::cannot_simplify);
2860           phi->clear(Phi::visited);
2861           return phi;
2862         }
2863       }
2864     }
2865 
2866     // sucessfully simplified phi function
2867     assert(subst != NULL, "illegal phi function");

2869     phi->clear(Phi::visited);
2870     phi->set_subst(subst);
2871

2873     if (PrintPhiFunctions) {
2874       tty->print_cr("simplified phi function %c%d to %c%d (Block B%d)", phi->type()->tchar(), phi->id(), subst->type()->tchar(), subst->id(), phi->block()->block_id());

2876 #endif
2877 
2878     return subst;
2879   }

3235             // we setup the OSR edge leading into this block so that
3236             // Phis get setup correctly.

3241           b->set(BlockBegin::was_visited_flag);
3242           connect_to_end(b);
3243         }
3244       }
3245     }
3246   } while (!bailed_out() && !scope_data()->is_work_list_empty());

3373 void GraphBuilder::setup_osr_entry_block() {
3374   assert(compilation()->is_osr_compile(), "only for osrs");
3375 
3376   int osr_bci = compilation()->osr_bci();
3377   ciBytecodeStream s(method());
3378   s.reset_to_bci(osr_bci);
3379   s.next();
3380   scope_data()->set_stream(&s);
3381 
3382   // create a new block to be the osr setup code
3383   _osr_entry = new BlockBegin(osr_bci);
3384   _osr_entry->set(BlockBegin::osr_entry_flag);
3385   _osr_entry->set_depth_first_number(0);
3386   BlockBegin* target = bci2block()->at(osr_bci);
3387   assert(target != NULL && target->is_set(BlockBegin::osr_entry_flag), "must be there");
3388   // the osr entry has no values for locals
3389   ValueStack* state = target->state()->copy();
3390   _osr_entry->set_state(state);
3391 
3392   kill_all();
3393   _block = _osr_entry;
3394   _state = _osr_entry->state()->copy();
3395   assert(_state->bci() == osr_bci, "mismatch");
3396   _last  = _osr_entry;
3397   Value e = append(new OsrEntry());
3398   e->set_needs_null_check(false);
3399 
3400   // OSR buffer is
3401   //
3402   // locals[nlocals-1..0]
3403   // monitors[number_of_locks-1..0]
3404   //
3405   // locals is a direct copy of the interpreter frame so in the osr buffer
3406   // so first slot in the local array is the last local from the interpreter
3407   // and last slot is local[0] (receiver) from the interpreter
3408   //
3409   // Similarly with locks. The first lock slot in the osr buffer is the nth lock
3410   // from the interpreter frame, the nth lock slot in the osr buffer is 0th lock
3411   // in the interpreter frame (the method lock if a sync method)
3412 
3413   // Initialize monitors in the compiled activation.
3414 
3415   int index;
3416   Value local;
3417 
3418   // find all the locals that the interpreter thinks contain live oops
3419   const ResourceBitMap live_oops = method()->live_local_oops_at_bci(osr_bci);
3420 
3421   // compute the offset into the locals so that we can treat the buffer
3422   // as if the locals were still in the interpreter frame
3423   int locals_offset = BytesPerWord * (method()->max_locals() - 1);
3424   for_each_local_value(state, index, local) {
3425     int offset = locals_offset - (index + local->type()->size() - 1) * BytesPerWord;
3426     Value get;
3427     if (local->type()->is_object_kind() && !live_oops.at(index)) {
3428       // The interpreter thinks this local is dead but the compiler
3429       // doesn't so pretend that the interpreter passed in null.
3430       get = append(new Constant(objectNull));
3431     } else {
3432       get = append(new UnsafeGetRaw(as_BasicType(local->type()), e,
3433                                     append(new Constant(new IntConstant(offset))),
3434                                     0,
3435                                     true /*unaligned*/, true /*wide*/));
3436     }
3437     _state->store_local(index, get);
3438   }
3439 
3440   // the storage for the OSR buffer is freed manually in the LIRGenerator.
3441 
3442   assert(state->caller_state() == NULL, "should be top scope");
3443   state->clear_locals();
3444   Goto* g = new Goto(target, false);
3445   append(g);
3446   _osr_entry->set_end(g);
3447   target->merge(_osr_entry->end()->state());
3448 
3449   scope_data()->set_stream(NULL);
3450 }

3453 ValueStack* GraphBuilder::state_at_entry() {
3454   ValueStack* state = new ValueStack(scope(), NULL);
3455 
3456   // Set up locals for receiver
3457   int idx = 0;
3458   if (!method()->is_static()) {
3459     // we should always see the receiver
3460     state->store_local(idx, new Local(method()->holder(), objectType, idx,
3461              /*receiver*/ true, /*never_null*/ method()->holder()->is_flat_array_klass()));
3462     idx = 1;
3463   }
3464 
3465   // Set up locals for incoming arguments
3466   ciSignature* sig = method()->signature();
3467   for (int i = 0; i < sig->count(); i++) {
3468     ciType* type = sig->type_at(i);
3469     BasicType basic_type = type->basic_type();
3470     // don't allow T_ARRAY to propagate into locals types
3471     if (is_reference_type(basic_type)) basic_type = T_OBJECT;
3472     ValueType* vt = as_ValueType(basic_type);
3473     state->store_local(idx, new Local(type, vt, idx, false, type->is_inlinetype()));
3474     idx += type->size();
3475   }
3476 
3477   // lock synchronized method
3478   if (method()->is_synchronized()) {
3479     state->lock(NULL);
3480   }
3481 
3482   return state;

3486 GraphBuilder::GraphBuilder(Compilation* compilation, IRScope* scope)
3487   : _scope_data(NULL)
3488   , _compilation(compilation)
3489   , _memory(new MemoryBuffer())
3490   , _inline_bailout_msg(NULL)
3491   , _instruction_count(0)
3492   , _osr_entry(NULL)
3493 {
3494   int osr_bci = compilation->osr_bci();
3495 
3496   // determine entry points and bci2block mapping
3497   BlockListBuilder blm(compilation, scope, osr_bci);
3498   CHECK_BAILOUT();
3499 
3500   BlockList* bci2block = blm.bci2block();
3501   BlockBegin* start_block = bci2block->at(0);
3502 
3503   push_root_scope(scope, bci2block, start_block);
3504 
3505   // setup state for std entry
3506   _initial_state = state_at_entry();
3507   start_block->merge(_initial_state);
3508 
3509   // complete graph
3510   _vmap        = new ValueMap();
3511   switch (scope->method()->intrinsic_id()) {
3512   case vmIntrinsics::_dabs          : // fall through
3513   case vmIntrinsics::_dsqrt         : // fall through
3514   case vmIntrinsics::_dsin          : // fall through
3515   case vmIntrinsics::_dcos          : // fall through
3516   case vmIntrinsics::_dtan          : // fall through
3517   case vmIntrinsics::_dlog          : // fall through
3518   case vmIntrinsics::_dlog10        : // fall through
3519   case vmIntrinsics::_dexp          : // fall through
3520   case vmIntrinsics::_dpow          : // fall through
3521     {
3522       // Compiles where the root method is an intrinsic need a special
3523       // compilation environment because the bytecodes for the method
3524       // shouldn't be parsed during the compilation, only the special
3525       // Intrinsic node should be emitted.  If this isn't done the the
3526       // code for the inlined version will be different than the root
3527       // compiled version which could lead to monotonicity problems on
3528       // intel.
3529       if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3530         BAILOUT("failed to inline intrinsic, method not annotated");
3531       }
3532 
3533       // Set up a stream so that appending instructions works properly.
3534       ciBytecodeStream s(scope->method());
3535       s.reset_to_bci(0);
3536       scope_data()->set_stream(&s);
3537       s.next();
3538 
3539       // setup the initial block state
3540       _block = start_block;
3541       _state = start_block->state()->copy_for_parsing();
3542       _last  = start_block;
3543       load_local(doubleType, 0);
3544       if (scope->method()->intrinsic_id() == vmIntrinsics::_dpow) {
3545         load_local(doubleType, 2);
3546       }
3547 
3548       // Emit the intrinsic node.
3549       bool result = try_inline_intrinsics(scope->method());
3550       if (!result) BAILOUT("failed to inline intrinsic");
3551       method_return(dpop());
3552 
3553       // connect the begin and end blocks and we're all done.
3554       BlockEnd* end = last()->as_BlockEnd();
3555       block()->set_end(end);
3556       break;
3557     }
3558 
3559   case vmIntrinsics::_Reference_get:
3560     {
3561       {
3562         // With java.lang.ref.reference.get() we must go through the
3563         // intrinsic - when G1 is enabled - even when get() is the root
3564         // method of the compile so that, if necessary, the value in
3565         // the referent field of the reference object gets recorded by
3566         // the pre-barrier code.
3567         // Specifically, if G1 is enabled, the value in the referent
3568         // field is recorded by the G1 SATB pre barrier. This will
3569         // result in the referent being marked live and the reference
3570         // object removed from the list of discovered references during
3571         // reference processing.
3572         if (CheckIntrinsics && !scope->method()->intrinsic_candidate()) {
3573           BAILOUT("failed to inline intrinsic, method not annotated");
3574         }
3575 
3576         // Also we need intrinsic to prevent commoning reads from this field
3577         // across safepoint since GC can change its value.
3578 
3579         // Set up a stream so that appending instructions works properly.
3580         ciBytecodeStream s(scope->method());
3581         s.reset_to_bci(0);
3582         scope_data()->set_stream(&s);
3583         s.next();
3584 
3585         // setup the initial block state
3586         _block = start_block;
3587         _state = start_block->state()->copy_for_parsing();
3588         _last  = start_block;
3589         load_local(objectType, 0);
3590 
3591         // Emit the intrinsic node.
3592         bool result = try_inline_intrinsics(scope->method());
3593         if (!result) BAILOUT("failed to inline intrinsic");
3594         method_return(apop());
3595 
3596         // connect the begin and end blocks and we're all done.
3597         BlockEnd* end = last()->as_BlockEnd();
3598         block()->set_end(end);
3599         break;
3600       }
3601       // Otherwise, fall thru
3602     }
3603 
3604   default:
3605     scope_data()->add_to_work_list(start_block);
3606     iterate_all_blocks();
3607     break;
3608   }
3609   CHECK_BAILOUT();
3610 
3611   _start = setup_start_block(osr_bci, start_block, _osr_entry, _initial_state);
3612 
3613   eliminate_redundant_phis(_start);
3614 
3615   NOT_PRODUCT(if (PrintValueNumbering && Verbose) print_stats());
3616   // for osr compile, bailout if some requirements are not fulfilled
3617   if (osr_bci != -1) {
3618     BlockBegin* osr_block = blm.bci2block()->at(osr_bci);
3619     if (!osr_block->is_set(BlockBegin::was_visited_flag)) {
3620       BAILOUT("osr entry must have been visited for osr compile");
3621     }
3622 
3623     // check if osr entry point has empty stack - we cannot handle non-empty stacks at osr entry points
3624     if (!osr_block->state()->stack_is_empty()) {
3625       BAILOUT("stack not empty at OSR entry point");
3626     }
3627   }
3628 #ifndef PRODUCT
3629   if (PrintCompilation && Verbose) tty->print_cr("Created %d Instructions", _instruction_count);
3630 #endif
3631 }

3678 bool GraphBuilder::try_inline(ciMethod* callee, bool holder_known, bool ignore_return, Bytecodes::Code bc, Value receiver) {
3679   const char* msg = NULL;
3680 
3681   // clear out any existing inline bailout condition
3682   clear_inline_bailout();
3683 
3684   // exclude methods we don't want to inline
3685   msg = should_not_inline(callee);
3686   if (msg != NULL) {
3687     print_inlining(callee, msg, /*success*/ false);
3688     return false;
3689   }
3690 
3691   // method handle invokes
3692   if (callee->is_method_handle_intrinsic()) {
3693     if (try_method_handle_inline(callee, ignore_return)) {
3694       if (callee->has_reserved_stack_access()) {
3695         compilation()->set_has_reserved_stack_access(true);
3696       }
3697       return true;
3698     }
3699     return false;
3700   }
3701 
3702   // handle intrinsics
3703   if (callee->intrinsic_id() != vmIntrinsics::_none &&
3704       (CheckIntrinsics ? callee->intrinsic_candidate() : true)) {
3705     if (try_inline_intrinsics(callee, ignore_return)) {
3706       print_inlining(callee, "intrinsic");
3707       if (callee->has_reserved_stack_access()) {
3708         compilation()->set_has_reserved_stack_access(true);
3709       }
3710       return true;
3711     }
3712     // try normal inlining
3713   }
3714 
3715   // certain methods cannot be parsed at all
3716   msg = check_can_parse(callee);
3717   if (msg != NULL) {
3718     print_inlining(callee, msg, /*success*/ false);
3719     return false;
3720   }
3721 
3722   // If bytecode not set use the current one.
3723   if (bc == Bytecodes::_illegal) {
3724     bc = code();
3725   }
3726   if (try_inline_full(callee, holder_known, ignore_return, bc, receiver)) {
3727     if (callee->has_reserved_stack_access()) {
3728       compilation()->set_has_reserved_stack_access(true);
3729     }
3730     return true;
3731   }
3732 
3733   // Entire compilation could fail during try_inline_full call.
3734   // In that case printing inlining decision info is useless.
3735   if (!bailed_out())
3736     print_inlining(callee, _inline_bailout_msg, /*success*/ false);
3737 
3738   return false;
3739 }

3886 bool GraphBuilder::try_inline_jsr(int jsr_dest_bci) {
3887   // Introduce a new callee continuation point - all Ret instructions
3888   // will be replaced with Gotos to this point.
3889   BlockBegin* cont = block_at(next_bci());
3890   assert(cont != NULL, "continuation must exist (BlockListBuilder starts a new block after a jsr");
3891 
3892   // Note: can not assign state to continuation yet, as we have to
3893   // pick up the state from the Ret instructions.
3894 
3895   // Push callee scope
3896   push_scope_for_jsr(cont, jsr_dest_bci);
3897 
3898   // Temporarily set up bytecode stream so we can append instructions
3899   // (only using the bci of this stream)
3900   scope_data()->set_stream(scope_data()->parent()->stream());
3901 
3902   BlockBegin* jsr_start_block = block_at(jsr_dest_bci);
3903   assert(jsr_start_block != NULL, "jsr start block must exist");
3904   assert(!jsr_start_block->is_set(BlockBegin::was_visited_flag), "should not have visited jsr yet");
3905   Goto* goto_sub = new Goto(jsr_start_block, false);
3906   // Must copy state to avoid wrong sharing when parsing bytecodes
3907   assert(jsr_start_block->state() == NULL, "should have fresh jsr starting block");
3908   jsr_start_block->set_state(copy_state_before_with_bci(jsr_dest_bci));
3909   append(goto_sub);
3910   _block->set_end(goto_sub);
3911   _last = _block = jsr_start_block;
3912 
3913   // Clear out bytecode stream
3914   scope_data()->set_stream(NULL);
3915 
3916   scope_data()->add_to_work_list(jsr_start_block);
3917 
3918   // Ready to resume parsing in subroutine
3919   iterate_all_blocks();
3920 
3921   // If we bailed out during parsing, return immediately (this is bad news)
3922   CHECK_BAILOUT_(false);
3923 
3924   // Detect whether the continuation can actually be reached. If not,
3925   // it has not had state set by the join() operations in
3926   // iterate_bytecodes_for_block()/ret() and we should not touch the
3927   // iteration state. The calling activation of
3928   // iterate_bytecodes_for_block will then complete normally.
3929   if (cont->state() != NULL) {
3930     if (!cont->is_set(BlockBegin::was_visited_flag)) {
3931       // add continuation to work list instead of parsing it immediately
3932       scope_data()->parent()->add_to_work_list(cont);
3933     }
3934   }
3935 
3936   assert(jsr_continuation() == cont, "continuation must not have changed");
3937   assert(!jsr_continuation()->is_set(BlockBegin::was_visited_flag) ||
3938          jsr_continuation()->is_set(BlockBegin::parser_loop_header_flag),
3939          "continuation can only be visited in case of backward branches");
3940   assert(_last && _last->as_BlockEnd(), "block must have end");
3941 
3942   // continuation is in work list, so end iteration of current block
3943   _skip_block = true;
3944   pop_scope_for_jsr();
3945 
3946   return true;
3947 }

4036 bool GraphBuilder::try_inline_full(ciMethod* callee, bool holder_known, bool ignore_return, Bytecodes::Code bc, Value receiver) {
4037   assert(!callee->is_native(), "callee must not be native");
4038   if (CompilationPolicy::policy()->should_not_inline(compilation()->env(), callee)) {
4039     INLINE_BAILOUT("inlining prohibited by policy");
4040   }
4041   // first perform tests of things it's not possible to inline
4042   if (callee->has_exception_handlers() &&
4043       !InlineMethodsWithExceptionHandlers) INLINE_BAILOUT("callee has exception handlers");
4044   if (callee->is_synchronized() &&
4045       !InlineSynchronizedMethods         ) INLINE_BAILOUT("callee is synchronized");
4046   if (!callee->holder()->is_initialized()) INLINE_BAILOUT("callee's klass not initialized yet");
4047   if (!callee->has_balanced_monitors())    INLINE_BAILOUT("callee's monitors do not match");
4048 
4049   // Proper inlining of methods with jsrs requires a little more work.
4050   if (callee->has_jsrs()                 ) INLINE_BAILOUT("jsrs not handled properly by inliner yet");
4051 
4052   if (strict_fp_requires_explicit_rounding &&
4053       method()->is_strict() != callee->is_strict()) {
4054 #ifdef IA32
4055     // If explicit rounding is required, do not inline strict code into non-strict code (or the reverse).
4056     // When SSE2 is present, no special handling is needed.
4057     if (UseSSE < 2) {
4058       INLINE_BAILOUT("caller and callee have different strict fp requirements");
4059     }
4060 #else
4061     Unimplemented();
4062 #endif // IA32
4063   }
4064 
4065   if (is_profiling() && !callee->ensure_method_data()) {
4066     INLINE_BAILOUT("mdo allocation failed");
4067   }
4068 
4069   const bool is_invokedynamic = (bc == Bytecodes::_invokedynamic);
4070   const bool has_receiver = (bc != Bytecodes::_invokestatic && !is_invokedynamic);
4071 
4072   const int args_base = state()->stack_size() - callee->arg_size();
4073   assert(args_base >= 0, "stack underflow during inlining");
4074 
4075   Value recv = NULL;
4076   if (has_receiver) {
4077     assert(!callee->is_static(), "callee must not be static");
4078     assert(callee->arg_size() > 0, "must have at least a receiver");
4079 
4080     recv = state()->stack_at(args_base);
4081     if (recv->is_null_obj()) {
4082       INLINE_BAILOUT("receiver is always null");
4083     }
4084   }
4085 
4086   // now perform tests that are based on flag settings
4087   bool inlinee_by_directive = compilation()->directive()->should_inline(callee);
4088   if (callee->force_inline() || inlinee_by_directive) {
4089     if (inline_level() > MaxForceInlineLevel                      ) INLINE_BAILOUT("MaxForceInlineLevel");
4090     if (recursive_inline_level(callee) > C1MaxRecursiveInlineLevel) INLINE_BAILOUT("recursive inlining too deep");
4091 
4092     const char* msg = "";
4093     if (callee->force_inline())  msg = "force inline by annotation";
4094     if (inlinee_by_directive)    msg = "force inline by CompileCommand";
4095     print_inlining(callee, msg);
4096   } else {
4097     // use heuristic controls on inlining
4098     if (inline_level() > C1MaxInlineLevel                       ) INLINE_BAILOUT("inlining too deep");
4099     int callee_recursive_level = recursive_inline_level(callee);
4100     if (callee_recursive_level > C1MaxRecursiveInlineLevel      ) INLINE_BAILOUT("recursive inlining too deep");
4101     if (callee->code_size_for_inlining() > max_inline_size()    ) INLINE_BAILOUT("callee is too large");
4102     // Additional condition to limit stack usage for non-recursive calls.
4103     if ((callee_recursive_level == 0) &&
4104         (callee->max_stack() + callee->max_locals() - callee->size_of_parameters() > C1InlineStackLimit)) {
4105       INLINE_BAILOUT("callee uses too much stack");
4106     }
4107 
4108     // don't inline throwable methods unless the inlining tree is rooted in a throwable class
4109     if (callee->name() == ciSymbol::object_initializer_name() &&
4110         callee->holder()->is_subclass_of(ciEnv::current()->Throwable_klass())) {
4111       // Throwable constructor call
4112       IRScope* top = scope();
4113       while (top->caller() != NULL) {
4114         top = top->caller();
4115       }
4116       if (!top->method()->holder()->is_subclass_of(ciEnv::current()->Throwable_klass())) {
4117         INLINE_BAILOUT("don't inline Throwable constructors");
4118       }
4119     }
4120 
4121     if (compilation()->env()->num_inlined_bytecodes() > DesiredMethodLimit) {
4122       INLINE_BAILOUT("total inlining greater than DesiredMethodLimit");
4123     }

4125     print_inlining(callee, "inline", /*success*/ true);
4126   }

4129   // GraphBuilder level, do not cause bailout just of the inlining but
4130   // in fact of the entire compilation.
4131 
4132   BlockBegin* orig_block = block();
4133 
4134   // Insert null check if necessary
4135   if (has_receiver) {
4136     // note: null check must happen even if first instruction of callee does
4137     //       an implicit null check since the callee is in a different scope
4138     //       and we must make sure exception handling does the right thing
4139     null_check(recv);
4140   }
4141 
4142   if (is_profiling()) {
4143     // Note that we'd collect profile data in this method if we wanted it.
4144     // this may be redundant here...
4145     compilation()->set_would_profile(true);
4146 
4147     if (profile_calls()) {
4148       int start = 0;
4149       Values* obj_args = args_list_for_profiling(callee, start, has_receiver);
4150       if (obj_args != NULL) {

4153         for (int i = args_base+start, j = 0; j < obj_args->max_length() && i < state()->stack_size(); ) {

4167   // more than one return instruction or the return does not allow
4168   // fall-through of control flow, all return instructions of the

4184   // inlining, to detect if inlined method has edges to its

4189   push_scope(callee, cont);

4196   // (only using the bci of this stream)
4197   scope_data()->set_stream(scope_data()->parent()->stream());

4200   // note: this will also ensure that all arguments are computed before being passed
4201   ValueStack* callee_state = state();

4204     const int arg_no = i - args_base;
4205     Value arg = caller_state->stack_at_inc(i);
4206     store_local(callee_state, arg, arg_no);
4207   }
4208

4210   // Note that we preserve locals state in case we can use it later
4211   // (see use of pop_scope() below)
4212   caller_state->truncate_stack(args_base);
4213   assert(callee_state->stack_size() == 0, "callee stack must be empty");
4214 
4215   Value lock = NULL;
4216   BlockBegin* sync_handler = NULL;
4217 
4218   // Inline the locking of the receiver if the callee is synchronized
4219   if (callee->is_synchronized()) {
4220     lock = callee->is_static() ? append(new Constant(new InstanceConstant(callee->holder()->java_mirror())))
4221                                : state()->local_at(0);
4222     sync_handler = new BlockBegin(SynchronizationEntryBCI);

4225 
4226   if (compilation()->env()->dtrace_method_probes()) {

4228     args->push(append(new Constant(new MethodConstant(method()))));
4229     append(new RuntimeCall(voidType, "dtrace_method_entry", CAST_FROM_FN_PTR(address, SharedRuntime::dtrace_method_entry), args));
4230   }

4233     profile_invocation(callee, copy_state_before_with_bci(SynchronizationEntryBCI));
4234   }
4235

4237   if (callee_start_block != NULL) {
4238     assert(callee_start_block->is_set(BlockBegin::parser_loop_header_flag), "must be loop header");
4239     Goto* goto_callee = new Goto(callee_start_block, false);
4240     // The state for this goto is in the scope of the callee, so use
4241     // the entry bci for the callee instead of the call site bci.
4242     append_with_bci(goto_callee, 0);
4243     _block->set_end(goto_callee);
4244     callee_start_block->merge(callee_state);
4245 
4246     _last = _block = callee_start_block;
4247 
4248     scope_data()->add_to_work_list(callee_start_block);

4257

4259   // were in before or in the callee's start block)
4260   iterate_all_blocks(callee_start_block == NULL);
4261 
4262   if (log != NULL) log->done("parse");
4263 
4264   // If we bailed out during parsing, return immediately (this is bad news)
4265   if (bailed_out())
4266       return false;
4267 
4268   // iterate_all_blocks theoretically traverses in random order; in
4269   // practice, we have only traversed the continuation if we are
4270   // inlining into a subroutine
4271   assert(continuation_existed ||
4272          !continuation()->is_set(BlockBegin::was_visited_flag),
4273          "continuation should not have been parsed yet if we created it");

4275   // At this point we are almost ready to return and resume parsing of
4276   // the caller back in the GraphBuilder. The only thing we want to do
4277   // first is an optimization: during parsing of the callee we
4278   // generated at least one Goto to the continuation block. If we
4279   // generated exactly one, and if the inlined method spanned exactly
4280   // one block (and we didn't have to Goto its entry), then we snip
4281   // off the Goto to the continuation, allowing control to fall
4282   // through back into the caller block and effectively performing
4283   // block merging. This allows load elimination and CSE to take place
4284   // across multiple callee scopes if they are relatively simple, and
4285   // is currently essential to making inlining profitable.
4286   if (num_returns() == 1

4289     _last  = inline_cleanup_return_prev();
4290     _state = inline_cleanup_state();
4291   } else if (continuation_preds == cont->number_of_preds()) {
4292     // Inlining caused that the instructions after the invoke in the
4293     // caller are not reachable any more. So skip filling this block
4294     // with instructions!
4295     assert(cont == continuation(), "");
4296     assert(_last && _last->as_BlockEnd(), "");
4297     _skip_block = true;
4298   } else {
4299     // Resume parsing in continuation block unless it was already parsed.
4300     // Note that if we don't change _last here, iteration in
4301     // iterate_bytecodes_for_block will stop when we return.
4302     if (!continuation()->is_set(BlockBegin::was_visited_flag)) {
4303       // add continuation to work list instead of parsing it immediately
4304       assert(_last && _last->as_BlockEnd(), "");
4305       scope_data()->parent()->add_to_work_list(continuation());
4306       _skip_block = true;
4307     }
4308   }
4309 
4310   // Fill the exception handler for synchronized methods with instructions
4311   if (callee->is_synchronized() && sync_handler->state() != NULL) {
4312     fill_sync_handler(lock, sync_handler);
4313   } else {
4314     pop_scope();
4315   }
4316 
4317   compilation()->notice_inlined_method(callee);
4318 
4319   return true;
4320 }

4323 bool GraphBuilder::try_method_handle_inline(ciMethod* callee, bool ignore_return) {
4324   ValueStack* state_before = copy_state_before();
4325   vmIntrinsics::ID iid = callee->intrinsic_id();
4326   switch (iid) {
4327   case vmIntrinsics::_invokeBasic:
4328     {
4329       // get MethodHandle receiver
4330       const int args_base = state()->stack_size() - callee->arg_size();
4331       ValueType* type = state()->stack_at(args_base)->type();
4332       if (type->is_constant()) {
4333         ciMethod* target = type->as_ObjectType()->constant_value()->as_method_handle()->get_vmtarget();
4334         // We don't do CHA here so only inline static and statically bindable methods.
4335         if (target->is_static() || target->can_be_statically_bound()) {
4336           if (ciMethod::is_consistent_info(callee, target)) {
4337             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4338             ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4339             if (try_inline(target, /*holder_known*/ !callee->is_static(), ignore_return, bc)) {
4340               return true;
4341             }
4342           } else {
4343             print_inlining(target, "signatures mismatch", /*success*/ false);
4344           }
4345         } else {
4346           print_inlining(target, "not static or statically bindable", /*success*/ false);
4347         }
4348       } else {
4349         print_inlining(callee, "receiver not constant", /*success*/ false);
4350       }
4351     }
4352     break;
4353 
4354   case vmIntrinsics::_linkToVirtual:
4355   case vmIntrinsics::_linkToStatic:
4356   case vmIntrinsics::_linkToSpecial:
4357   case vmIntrinsics::_linkToInterface:
4358     {
4359       // pop MemberName argument
4360       const int args_base = state()->stack_size() - callee->arg_size();
4361       ValueType* type = apop()->type();
4362       if (type->is_constant()) {
4363         ciMethod* target = type->as_ObjectType()->constant_value()->as_member_name()->get_vmtarget();
4364         ignore_return = ignore_return || (callee->return_type()->is_void() && !target->return_type()->is_void());
4365         // If the target is another method handle invoke, try to recursively get
4366         // a better target.
4367         if (target->is_method_handle_intrinsic()) {
4368           if (try_method_handle_inline(target, ignore_return)) {
4369             return true;
4370           }
4371         } else if (!ciMethod::is_consistent_info(callee, target)) {
4372           print_inlining(target, "signatures mismatch", /*success*/ false);
4373         } else {
4374           ciSignature* signature = target->signature();
4375           const int receiver_skip = target->is_static() ? 0 : 1;
4376           // Cast receiver to its type.
4377           if (!target->is_static()) {
4378             ciKlass* tk = signature->accessing_klass();
4379             Value obj = state()->stack_at(args_base);
4380             if (obj->exact_type() == NULL &&
4381                 obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4382               TypeCast* c = new TypeCast(tk, obj, state_before);
4383               append(c);
4384               state()->stack_at_put(args_base, c);
4385             }
4386           }
4387           // Cast reference arguments to its type.
4388           for (int i = 0, j = 0; i < signature->count(); i++) {
4389             ciType* t = signature->type_at(i);
4390             if (t->is_klass()) {
4391               ciKlass* tk = t->as_klass();
4392               Value obj = state()->stack_at(args_base + receiver_skip + j);
4393               if (obj->exact_type() == NULL &&
4394                   obj->declared_type() != tk && tk != compilation()->env()->Object_klass()) {
4395                 TypeCast* c = new TypeCast(t, obj, state_before);
4396                 append(c);
4397                 state()->stack_at_put(args_base + receiver_skip + j, c);
4398               }
4399             }
4400             j += t->size();  // long and double take two slots
4401           }
4402           // We don't do CHA here so only inline static and statically bindable methods.
4403           if (target->is_static() || target->can_be_statically_bound()) {
4404             Bytecodes::Code bc = target->is_static() ? Bytecodes::_invokestatic : Bytecodes::_invokevirtual;
4405             if (try_inline(target, /*holder_known*/ !callee->is_static(), ignore_return, bc)) {
4406               return true;
4407             }
4408           } else {
4409             print_inlining(target, "not static or statically bindable", /*success*/ false);
4410           }
4411         }
4412       } else {
4413         print_inlining(callee, "MemberName not constant", /*success*/ false);
4414       }
4415     }
4416     break;
4417 
4418   default:
4419     fatal("unexpected intrinsic %d: %s", iid, vmIntrinsics::name_at(iid));
4420     break;
4421   }
4422   set_state(state_before->copy_for_parsing());
4423   return false;
4424 }
