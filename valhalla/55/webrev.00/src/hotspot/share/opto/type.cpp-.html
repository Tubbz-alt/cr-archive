  55 const Type::Offset Type::Offset::meet(const Type::Offset other) const {
  56   // Either is 'TOP' offset?  Return the other offset!
  57   int offset = other._offset;
  58   if (_offset == OffsetTop) return Offset(offset);
  59   if (offset == OffsetTop) return Offset(_offset);
  60   // If either is different, return 'BOTTOM' offset
  61   if (_offset != offset) return bottom;
  62   return Offset(_offset);
  63 }

  65 const Type::Offset Type::Offset::dual() const {
  66   if (_offset == OffsetTop) return bottom;// Map 'TOP' into 'BOTTOM'
  67   if (_offset == OffsetBot) return top;// Map 'BOTTOM' into 'TOP'
  68   return Offset(_offset);               // Map everything else into self
  69 }

  71 const Type::Offset Type::Offset::add(intptr_t offset) const {
  72   // Adding to 'TOP' offset?  Return 'TOP'!
  73   if (_offset == OffsetTop || offset == OffsetTop) return top;
  74   // Adding to 'BOTTOM' offset?  Return 'BOTTOM'!
  75   if (_offset == OffsetBot || offset == OffsetBot) return bottom;
  76   // Addition overflows or "accidentally" equals to OffsetTop? Return 'BOTTOM'!
  77   offset += (intptr_t)_offset;
  78   if (offset != (int)offset || offset == OffsetTop) return bottom;
  79 
  80   // assert( _offset >= 0 && _offset+offset >= 0, "" );
  81   // It is possible to construct a negative offset during PhaseCCP
  82 
  83   return Offset((int)offset);        // Sum valid offsets
  84 }

  87   if (_offset == 0) {
  88     return;
  89   } else if (_offset == OffsetTop) {

  91   }
  92   else if (_offset == OffsetBot) {
  93     st->print("+bot");
  94   } else if (_offset) {

 182 //------------------------------get_const_type---------------------------
 183 const Type* Type::get_const_type(ciType* type) {
 184   if (type == NULL) {
 185     return NULL;
 186   } else if (type->is_primitive_type()) {
 187     return get_const_basic_type(type->basic_type());
 188   } else {
 189     return TypeOopPtr::make_from_klass(type->as_klass());
 190   }

 193 //---------------------------array_element_basic_type---------------------------------
 194 // Mapping to the array element's basic type.
 195 BasicType Type::array_element_basic_type() const {
 196   BasicType bt = basic_type();
 197   if (bt == T_INT) {
 198     if (this == TypeInt::INT)   return T_INT;
 199     if (this == TypeInt::CHAR)  return T_CHAR;
 200     if (this == TypeInt::BYTE)  return T_BYTE;
 201     if (this == TypeInt::BOOL)  return T_BOOLEAN;
 202     if (this == TypeInt::SHORT) return T_SHORT;
 203     return T_VOID;
 204   }
 205   return bt;
 206 }

 208 // For two instance arrays of same dimension, return the base element types.
 209 // Otherwise or if the arrays have different dimensions, return NULL.
 210 void Type::get_arrays_base_elements(const Type *a1, const Type *a2,
 211                                     const TypeInstPtr **e1, const TypeInstPtr **e2) {
 212 
 213   if (e1) *e1 = NULL;
 214   if (e2) *e2 = NULL;
 215   const TypeAryPtr* a1tap = (a1 == NULL) ? NULL : a1->isa_aryptr();
 216   const TypeAryPtr* a2tap = (a2 == NULL) ? NULL : a2->isa_aryptr();
 217 
 218   if (a1tap != NULL && a2tap != NULL) {
 219     // Handle multidimensional arrays
 220     const TypePtr* a1tp = a1tap->elem()->make_ptr();
 221     const TypePtr* a2tp = a2tap->elem()->make_ptr();
 222     while (a1tp && a1tp->isa_aryptr() && a2tp && a2tp->isa_aryptr()) {
 223       a1tap = a1tp->is_aryptr();
 224       a2tap = a2tp->is_aryptr();
 225       a1tp = a1tap->elem()->make_ptr();
 226       a2tp = a2tap->elem()->make_ptr();
 227     }
 228     if (a1tp && a1tp->isa_instptr() && a2tp && a2tp->isa_instptr()) {
 229       if (e1) *e1 = a1tp->is_instptr();
 230       if (e2) *e2 = a2tp->is_instptr();
 231     }
 232   }
 233 }

 235 //---------------------------get_typeflow_type---------------------------------
 236 // Import a type produced by ciTypeFlow.
 237 const Type* Type::get_typeflow_type(ciType* type) {
 238   switch (type->basic_type()) {
 239 
 240   case ciTypeFlow::StateVector::T_BOTTOM:
 241     assert(type == ciTypeFlow::StateVector::bottom_type(), "");
 242     return Type::BOTTOM;
 243 
 244   case ciTypeFlow::StateVector::T_TOP:
 245     assert(type == ciTypeFlow::StateVector::top_type(), "");
 246     return Type::TOP;
 247 
 248   case ciTypeFlow::StateVector::T_NULL:
 249     assert(type == ciTypeFlow::StateVector::null_type(), "");
 250     return TypePtr::NULL_PTR;
 251 
 252   case ciTypeFlow::StateVector::T_LONG2:
 253     // The ciTypeFlow pass pushes a long, then the half.
 254     // We do the same.
 255     assert(type == ciTypeFlow::StateVector::long2_type(), "");
 256     return TypeInt::TOP;
 257 
 258   case ciTypeFlow::StateVector::T_DOUBLE2:
 259     // The ciTypeFlow pass pushes double, then the half.
 260     // Our convention is the same.
 261     assert(type == ciTypeFlow::StateVector::double2_type(), "");
 262     return Type::TOP;
 263 
 264   case T_ADDRESS:
 265     assert(type->is_return_address(), "");
 266     return TypeRawPtr::make((address)(intptr_t)type->as_return_address()->bci());
 267 
 268   case T_VALUETYPE: {
 269     bool is_never_null = type->is_never_null();
 270     ciValueKlass* vk = type->unwrap()->as_value_klass();
 271     if (vk->is_scalarizable() && is_never_null) {
 272       return TypeValueType::make(vk);
 273     } else {
 274       return TypeOopPtr::make_from_klass(vk)->join_speculative(is_never_null ? TypePtr::NOTNULL : TypePtr::BOTTOM);
 275     }
 276   }
 277 
 278   default:
 279     // make sure we did not mix up the cases:
 280     assert(type != ciTypeFlow::StateVector::bottom_type(), "");
 281     assert(type != ciTypeFlow::StateVector::top_type(), "");
 282     assert(type != ciTypeFlow::StateVector::null_type(), "");
 283     assert(type != ciTypeFlow::StateVector::long2_type(), "");
 284     assert(type != ciTypeFlow::StateVector::double2_type(), "");
 285     assert(!type->is_return_address(), "");
 286 
 287     return Type::get_const_type(type);
 288   }
 289 }

 292 //-----------------------make_from_constant------------------------------------
 293 const Type* Type::make_from_constant(ciConstant constant, bool require_constant,
 294                                      int stable_dimension, bool is_narrow_oop,
 295                                      bool is_autobox_cache) {
 296   switch (constant.basic_type()) {
 297     case T_BOOLEAN:  return TypeInt::make(constant.as_boolean());
 298     case T_CHAR:     return TypeInt::make(constant.as_char());
 299     case T_BYTE:     return TypeInt::make(constant.as_byte());
 300     case T_SHORT:    return TypeInt::make(constant.as_short());
 301     case T_INT:      return TypeInt::make(constant.as_int());
 302     case T_LONG:     return TypeLong::make(constant.as_long());
 303     case T_FLOAT:    return TypeF::make(constant.as_float());
 304     case T_DOUBLE:   return TypeD::make(constant.as_double());
 305     case T_ARRAY:
 306     case T_VALUETYPE:
 307     case T_OBJECT: {
 308         const Type* con_type = NULL;
 309         ciObject* oop_constant = constant.as_object();
 310         if (oop_constant->is_null_object()) {
 311           con_type = Type::get_zero_type(T_OBJECT);
 312         } else {
 313           guarantee(require_constant || oop_constant->should_be_constant(), "con_type must get computed");
 314           con_type = TypeOopPtr::make_from_constant(oop_constant, require_constant);
 315           if (Compile::current()->eliminate_boxing() && is_autobox_cache) {
 316             con_type = con_type->is_aryptr()->cast_to_autobox_cache(true);
 317           }
 318           if (stable_dimension > 0) {
 319             assert(FoldStableValues, "sanity");
 320             assert(!con_type->is_zero_type(), "default value for stable field");
 321             con_type = con_type->is_aryptr()->cast_to_stable(true, stable_dimension);
 322           }
 323         }
 324         if (is_narrow_oop) {
 325           con_type = con_type->make_narrowoop();
 326         }
 327         return con_type;
 328       }
 329     case T_ILLEGAL:
 330       // Invalid ciConstant returned due to OutOfMemoryError in the CI
 331       assert(Compile::current()->env()->failing(), "otherwise should not see this");
 332       return NULL;
 333     default:
 334       // Fall through to failure
 335       return NULL;
 336   }
 337 }

 339 static ciConstant check_mismatched_access(ciConstant con, BasicType loadbt, bool is_unsigned) {
 340   BasicType conbt = con.basic_type();
 341   switch (conbt) {
 342     case T_BOOLEAN: conbt = T_BYTE;   break;
 343     case T_ARRAY:   conbt = T_OBJECT; break;
 344     case T_VALUETYPE: conbt = T_OBJECT; break;
 345     default:                          break;
 346   }
 347   switch (loadbt) {
 348     case T_BOOLEAN:   loadbt = T_BYTE;   break;
 349     case T_NARROWOOP: loadbt = T_OBJECT; break;
 350     case T_ARRAY:     loadbt = T_OBJECT; break;
 351     case T_VALUETYPE: loadbt = T_OBJECT; break;
 352     case T_ADDRESS:   loadbt = T_OBJECT; break;
 353     default:                             break;
 354   }
 355   if (conbt == loadbt) {
 356     if (is_unsigned && conbt == T_BYTE) {
 357       // LoadB (T_BYTE) with a small mask (<=8-bit) is converted to LoadUB (T_BYTE).
 358       return ciConstant(T_INT, con.as_int() & 0xFF);
 359     } else {
 360       return con;
 361     }
 362   }
 363   if (conbt == T_SHORT && loadbt == T_CHAR) {
 364     // LoadS (T_SHORT) with a small mask (<=16-bit) is converted to LoadUS (T_CHAR).
 365     return ciConstant(T_INT, con.as_int() & 0xFFFF);
 366   }
 367   return ciConstant(); // T_ILLEGAL
 368 }

 370 // Try to constant-fold a stable array element.
 371 const Type* Type::make_constant_from_array_element(ciArray* array, int off, int stable_dimension,
 372                                                    BasicType loadbt, bool is_unsigned_load) {
 373   // Decode the results of GraphKit::array_element_address.
 374   ciConstant element_value = array->element_value_by_offset(off);
 375   if (element_value.basic_type() == T_ILLEGAL) {
 376     return NULL; // wrong offset
 377   }
 378   ciConstant con = check_mismatched_access(element_value, loadbt, is_unsigned_load);
 379 
 380   assert(con.basic_type() != T_ILLEGAL, "elembt=%s; loadbt=%s; unsigned=%d",
 381          type2name(element_value.basic_type()), type2name(loadbt), is_unsigned_load);
 382 
 383   if (con.is_valid() &&          // not a mismatched access
 384       !con.is_null_or_zero()) {  // not a default value
 385     bool is_narrow_oop = (loadbt == T_NARROWOOP);
 386     return Type::make_from_constant(con, /*require_constant=*/true, stable_dimension, is_narrow_oop, /*is_autobox_cache=*/false);
 387   }
 388   return NULL;
 389 }

 391 const Type* Type::make_constant_from_field(ciInstance* holder, int off, bool is_unsigned_load, BasicType loadbt) {
 392   ciField* field;
 393   ciType* type = holder->java_mirror_type();
 394   if (type != NULL && type->is_instance_klass() && off >= InstanceMirrorKlass::offset_of_static_fields()) {
 395     // Static field
 396     field = type->as_instance_klass()->get_field_by_offset(off, /*is_static=*/true);
 397   } else {
 398     // Instance field
 399     field = holder->klass()->as_instance_klass()->get_field_by_offset(off, /*is_static=*/false);
 400   }
 401   if (field == NULL) {
 402     return NULL; // Wrong offset
 403   }
 404   return Type::make_constant_from_field(field, holder, loadbt, is_unsigned_load);
 405 }

 407 const Type* Type::make_constant_from_field(ciField* field, ciInstance* holder,
 408                                            BasicType loadbt, bool is_unsigned_load) {
 409   if (!field->is_constant()) {
 410     return NULL; // Non-constant field
 411   }
 412   ciConstant field_value;
 413   if (field->is_static()) {
 414     // final static field
 415     field_value = field->constant_value();
 416   } else if (holder != NULL) {
 417     // final or stable non-static field
 418     // Treat final non-static fields of trusted classes (classes in
 419     // java.lang.invoke and sun.invoke packages and subpackages) as
 420     // compile time constants.
 421     field_value = field->constant_value_of(holder);
 422   }
 423   if (!field_value.is_valid()) {
 424     return NULL; // Not a constant
 425   }
 426 
 427   ciConstant con = check_mismatched_access(field_value, loadbt, is_unsigned_load);
 428 
 429   assert(con.is_valid(), "elembt=%s; loadbt=%s; unsigned=%d",
 430          type2name(field_value.basic_type()), type2name(loadbt), is_unsigned_load);
 431 
 432   bool is_stable_array = FoldStableValues && field->is_stable() && field->type()->is_array_klass();
 433   int stable_dimension = (is_stable_array ? field->type()->as_array_klass()->dimension() : 0);
 434   bool is_narrow_oop = (loadbt == T_NARROWOOP);
 435 
 436   const Type* con_type = make_from_constant(con, /*require_constant=*/ true,
 437                                             stable_dimension, is_narrow_oop,
 438                                             field->is_autobox_cache());
 439   if (con_type != NULL && field->is_call_site_target()) {
 440     ciCallSite* call_site = holder->as_call_site();
 441     if (!call_site->is_fully_initialized_constant_call_site()) {
 442       ciMethodHandle* target = con.as_object()->as_method_handle();
 443       Compile::current()->dependencies()->assert_call_site_target_value(call_site, target);
 444     }
 445   }
 446   return con_type;
 447 }

 480 //--------------------------Initialize_shared----------------------------------
 481 void Type::Initialize_shared(Compile* current) {
 482   // This method does not need to be locked because the first system
 483   // compilations (stub compilations) occur serially.  If they are
 484   // changed to proceed in parallel, then this section will need
 485   // locking.
 486 
 487   Arena* save = current->type_arena();
 488   Arena* shared_type_arena = new (mtCompiler)Arena(mtCompiler);
 489

 772 //------------------------------hashcons---------------------------------------
 773 // Do the hash-cons trick.  If the Type already exists in the type table,
 774 // delete the current Type and return the existing Type.  Otherwise stick the
 775 // current Type in the Type table.
 776 const Type *Type::hashcons(void) {
 777   debug_only(base());           // Check the assertion in Type::base().
 778   // Look up the Type in the Type dictionary
 779   Dict *tdic = type_dict();
 780   Type* old = (Type*)(tdic->Insert(this, this, false));
 781   if( old ) {                   // Pre-existing Type?
 782     if( old != this )           // Yes, this guy is not the pre-existing?
 783       delete this;              // Yes, Nuke this guy
 784     assert( old->_dual, "" );
 785     return old;                 // Return pre-existing
 786   }
 787 
 788   // Every type has a dual (to make my lattice symmetric).
 789   // Since we just discovered a new Type, compute its dual right now.
 790   assert( !_dual, "" );         // No dual yet
 791   _dual = xdual();              // Compute the dual
 792   if (cmp(this, _dual) == 0) {  // Handle self-symmetric
 793     if (_dual != this) {
 794       delete _dual;
 795       _dual = this;
 796     }
 797     return this;
 798   }
 799   assert( !_dual->_dual, "" );  // No reverse dual yet
 800   assert( !(*tdic)[_dual], "" ); // Dual not in type system either
 801   // New Type, insert into Type table
 802   tdic->Insert((void*)_dual,(void*)_dual);
 803   ((Type*)_dual)->_dual = this; // Finish up being symmetric
 804 #ifdef ASSERT
 805   Type *dual_dual = (Type*)_dual->xdual();
 806   assert( eq(dual_dual), "xdual(xdual()) should be identity" );
 807   delete dual_dual;
 808 #endif
 809   return this;                  // Return new Type
 810 }

 911 // Compute the MEET of two types.  NOT virtual.  It enforces that meet is
 912 // commutative and the lattice is symmetric.
 913 const Type *Type::meet_helper(const Type *t, bool include_speculative) const {
 914   if (isa_narrowoop() && t->isa_narrowoop()) {
 915     const Type* result = make_ptr()->meet_helper(t->make_ptr(), include_speculative);
 916     return result->make_narrowoop();
 917   }
 918   if (isa_narrowklass() && t->isa_narrowklass()) {
 919     const Type* result = make_ptr()->meet_helper(t->make_ptr(), include_speculative);
 920     return result->make_narrowklass();
 921   }
 922 
 923   const Type *this_t = maybe_remove_speculative(include_speculative);
 924   t = t->maybe_remove_speculative(include_speculative);
 925 
 926   const Type *mt = this_t->xmeet(t);
 927 #ifdef ASSERT
 928   if (isa_narrowoop() || t->isa_narrowoop()) return mt;
 929   if (isa_narrowklass() || t->isa_narrowklass()) return mt;
 930   Compile* C = Compile::current();
 931   if (!C->_type_verify_symmetry) {
 932     return mt;
 933   }
 934   this_t->check_symmetrical(t, mt);
 935   // In the case of an array, computing the meet above, caused the
 936   // computation of the meet of the elements which at verification
 937   // time caused the computation of the meet of the dual of the
 938   // elements. Computing the meet of the dual of the arrays here
 939   // causes the meet of the dual of the elements to be computed which
 940   // would cause the meet of the dual of the dual of the elements,
 941   // that is the meet of the elements already computed above to be
 942   // computed. Avoid redundant computations by requesting no
 943   // verification.
 944   C->_type_verify_symmetry = false;
 945   const Type *mt_dual = this_t->_dual->xmeet(t->_dual);
 946   this_t->_dual->check_symmetrical(t->_dual, mt_dual);
 947   assert(!C->_type_verify_symmetry, "shouldn't have changed");
 948   C->_type_verify_symmetry = true;
 949 #endif
 950   return mt;
 951 }

 953 //------------------------------xmeet------------------------------------------
 954 // Compute the MEET of two types.  It returns a new Type object.
 955 const Type *Type::xmeet( const Type *t ) const {
 956   // Perform a fast test for common case; meeting the same types together.
 957   if( this == t ) return this;  // Meeting same type-rep?
 958 
 959   // Meeting TOP with anything?
 960   if( _base == Top ) return t;
 961 
 962   // Meeting BOTTOM with anything?
 963   if( _base == Bottom ) return BOTTOM;
 964 
 965   // Current "this->_base" is one of: Bad, Multi, Control, Top,
 966   // Abio, Abstore, Floatxxx, Doublexxx, Bottom, lastype.
 967   switch (t->base()) {  // Switch on original type
 968 
 969   // Cut in half the number of cases I must handle.  Only need cases for when
 970   // the given enum "t->type" is less than or equal to the local enum "type".
 971   case FloatCon:
 972   case DoubleCon:
 973   case Int:
 974   case Long:
 975     return t->xmeet(this);
 976 
 977   case OopPtr:
 978     return t->xmeet(this);
 979 
 980   case InstPtr:
 981     return t->xmeet(this);
 982 
 983   case MetadataPtr:
 984   case KlassPtr:
 985     return t->xmeet(this);
 986 
 987   case AryPtr:
 988     return t->xmeet(this);
 989 
 990   case NarrowOop:
 991     return t->xmeet(this);
 992 
 993   case NarrowKlass:
 994     return t->xmeet(this);
 995 
 996   case ValueType:
 997     return t->xmeet(this);
 998 
 999   case Bad:                     // Type check
1000   default:                      // Bogus type not in lattice
1001     typerr(t);
1002     return Type::BOTTOM;
1003 
1004   case Bottom:                  // Ye Olde Default
1005     return t;
1006 
1007   case FloatTop:
1008     if( _base == FloatTop ) return this;
1009   case FloatBot:                // Float
1010     if( _base == FloatBot || _base == FloatTop ) return FLOAT;
1011     if( _base == DoubleTop || _base == DoubleBot ) return Type::BOTTOM;
1012     typerr(t);
1013     return Type::BOTTOM;
1014 
1015   case DoubleTop:
1016     if( _base == DoubleTop ) return this;
1017   case DoubleBot:               // Double
1018     if( _base == DoubleBot || _base == DoubleTop ) return DOUBLE;
1019     if( _base == FloatTop || _base == FloatBot ) return Type::BOTTOM;
1020     typerr(t);
1021     return Type::BOTTOM;
1022 
1023   // These next few cases must match exactly or it is a compile-time error.
1024   case Control:                 // Control of code
1025   case Abio:                    // State of world outside of program
1026   case Memory:
1027     if( _base == t->_base )  return this;
1028     typerr(t);
1029     return Type::BOTTOM;
1030 
1031   case Top:                     // Top of the lattice
1032     return this;
1033   }
1034 
1035   // The type is unchanged
1036   return this;
1037 }

1501 //------------------------------widen------------------------------------------
1502 // Only happens for optimistic top-down optimizations.
1503 const Type *TypeInt::widen( const Type *old, const Type* limit ) const {
1504   // Coming from TOP or such; no widening
1505   if( old->base() != Int ) return this;
1506   const TypeInt *ot = old->is_int();
1507 
1508   // If new guy is equal to old guy, no widening
1509   if( _lo == ot->_lo && _hi == ot->_hi )
1510     return old;
1511 
1512   // If new guy contains old, then we widened
1513   if( _lo <= ot->_lo && _hi >= ot->_hi ) {
1514     // New contains old
1515     // If new guy is already wider than old, no widening
1516     if( _widen > ot->_widen ) return this;
1517     // If old guy was a constant, do not bother
1518     if (ot->_lo == ot->_hi)  return this;
1519     // Now widen new guy.
1520     // Check for widening too far
1521     if (_widen == WidenMax) {
1522       int max = max_jint;
1523       int min = min_jint;
1524       if (limit->isa_int()) {
1525         max = limit->is_int()->_hi;
1526         min = limit->is_int()->_lo;
1527       }
1528       if (min < _lo && _hi < max) {
1529         // If neither endpoint is extremal yet, push out the endpoint
1530         // which is closer to its respective limit.
1531         if (_lo >= 0 ||                 // easy common case
1532             (juint)(_lo - min) >= (juint)(max - _hi)) {
1533           // Try to widen to an unsigned range type of 31 bits:
1534           return make(_lo, max, WidenMax);
1535         } else {
1536           return make(min, _hi, WidenMax);
1537         }
1538       }
1539       return TypeInt::INT;
1540     }
1541     // Returned widened new guy
1542     return make(_lo,_hi,_widen+1);
1543   }
1544 
1545   // If old guy contains new, then we probably widened too far & dropped to
1546   // bottom.  Return the wider fellow.
1547   if ( ot->_lo <= _lo && ot->_hi >= _hi )
1548     return old;
1549 
1550   //fatal("Integer value range is not subset");
1551   //return this;
1552   return TypeInt::INT;
1553 }

1555 //------------------------------narrow---------------------------------------
1556 // Only happens for pessimistic optimizations.
1557 const Type *TypeInt::narrow( const Type *old ) const {
1558   if (_lo >= _hi)  return this;   // already narrow enough
1559   if (old == NULL)  return this;
1560   const TypeInt* ot = old->isa_int();
1561   if (ot == NULL)  return this;
1562   jint olo = ot->_lo;
1563   jint ohi = ot->_hi;
1564 
1565   // If new guy is equal to old guy, no narrowing
1566   if (_lo == olo && _hi == ohi)  return old;
1567 
1568   // If old guy was maximum range, allow the narrowing
1569   if (olo == min_jint && ohi == max_jint)  return this;
1570 
1571   if (_lo < olo || _hi > ohi)
1572     return this;                // doesn't narrow; pretty wierd
1573 
1574   // The new type narrows the old type, so look for a "death march".
1575   // See comments on PhaseTransform::saturate.
1576   juint nrange = (juint)_hi - _lo;
1577   juint orange = (juint)ohi - olo;
1578   if (nrange < max_juint - 1 && nrange > (orange >> 1) + (SMALLINT*2)) {
1579     // Use the new type only if the range shrinks a lot.
1580     // We do not want the optimizer computing 2^31 point by point.
1581     return old;
1582   }
1583 
1584   return this;
1585 }

1713 // Compute the MEET of two types.  It returns a new Type representation object
1714 // with reference count equal to the number of Types pointing at it.
1715 // Caller should wrap a Types around it.
1716 const Type *TypeLong::xmeet( const Type *t ) const {
1717   // Perform a fast test for common case; meeting the same types together.
1718   if( this == t ) return this;  // Meeting same type?
1719 
1720   // Currently "this->_base" is a TypeLong
1721   switch (t->base()) {          // Switch on original type
1722   case AnyPtr:                  // Mixing with oops happens when javac
1723   case RawPtr:                  // reuses local variables
1724   case OopPtr:
1725   case InstPtr:
1726   case AryPtr:
1727   case MetadataPtr:
1728   case KlassPtr:
1729   case NarrowOop:
1730   case NarrowKlass:
1731   case Int:
1732   case FloatTop:
1733   case FloatCon:
1734   case FloatBot:
1735   case DoubleTop:
1736   case DoubleCon:
1737   case DoubleBot:

1740   default:                      // All else is a mistake
1741     typerr(t);
1742   case Top:                     // No change
1743     return this;
1744   case Long:                    // Long vs Long?
1745     break;
1746   }
1747 
1748   // Expand covered set
1749   const TypeLong *r = t->is_long(); // Turn into a TypeLong
1750   return make( MIN2(_lo,r->_lo), MAX2(_hi,r->_hi), MAX2(_widen,r->_widen) );

1760 //------------------------------widen------------------------------------------
1761 // Only happens for optimistic top-down optimizations.
1762 const Type *TypeLong::widen( const Type *old, const Type* limit ) const {
1763   // Coming from TOP or such; no widening
1764   if( old->base() != Long ) return this;
1765   const TypeLong *ot = old->is_long();
1766 
1767   // If new guy is equal to old guy, no widening
1768   if( _lo == ot->_lo && _hi == ot->_hi )
1769     return old;
1770 
1771   // If new guy contains old, then we widened
1772   if( _lo <= ot->_lo && _hi >= ot->_hi ) {
1773     // New contains old
1774     // If new guy is already wider than old, no widening
1775     if( _widen > ot->_widen ) return this;
1776     // If old guy was a constant, do not bother
1777     if (ot->_lo == ot->_hi)  return this;
1778     // Now widen new guy.
1779     // Check for widening too far
1780     if (_widen == WidenMax) {
1781       jlong max = max_jlong;
1782       jlong min = min_jlong;
1783       if (limit->isa_long()) {
1784         max = limit->is_long()->_hi;
1785         min = limit->is_long()->_lo;
1786       }
1787       if (min < _lo && _hi < max) {
1788         // If neither endpoint is extremal yet, push out the endpoint
1789         // which is closer to its respective limit.
1790         if (_lo >= 0 ||                 // easy common case
1791             ((julong)_lo - min) >= ((julong)max - _hi)) {
1792           // Try to widen to an unsigned range type of 32/63 bits:
1793           if (max >= max_juint && _hi < max_juint)
1794             return make(_lo, max_juint, WidenMax);
1795           else
1796             return make(_lo, max, WidenMax);
1797         } else {
1798           return make(min, _hi, WidenMax);
1799         }
1800       }
1801       return TypeLong::LONG;
1802     }
1803     // Returned widened new guy
1804     return make(_lo,_hi,_widen+1);
1805   }
1806 
1807   // If old guy contains new, then we probably widened too far & dropped to
1808   // bottom.  Return the wider fellow.
1809   if ( ot->_lo <= _lo && ot->_hi >= _hi )
1810     return old;
1811 
1812   //  fatal("Long value range is not subset");
1813   // return this;
1814   return TypeLong::LONG;
1815 }

2372 //--------------------------ary_must_be_exact----------------------------------
2373 bool TypeAry::ary_must_be_exact() const {
2374   if (!UseExactTypes)       return false;
2375   // This logic looks at the element type of an array, and returns true
2376   // if the element type is either a primitive or a final instance class.
2377   // In such cases, an array built on this ary must have no subclasses.
2378   if (_elem == BOTTOM)      return false;  // general array not exact
2379   if (_elem == TOP   )      return false;  // inverted general array not exact
2380   const TypeOopPtr*  toop = NULL;
2381   if (UseCompressedOops && _elem->isa_narrowoop()) {
2382     toop = _elem->make_ptr()->isa_oopptr();
2383   } else {
2384     toop = _elem->isa_oopptr();
2385   }
2386   if (!toop)                return true;   // a primitive type, like int
2387   ciKlass* tklass = toop->klass();
2388   if (tklass == NULL)       return false;  // unloaded class
2389   if (!tklass->is_loaded()) return false;  // unloaded class
2390   const TypeInstPtr* tinst;
2391   if (_elem->isa_narrowoop())
2392     tinst = _elem->make_ptr()->isa_instptr();
2393   else
2394     tinst = _elem->isa_instptr();
2395   if (tinst) {
2396     // [V? has a subtype: [V. So even though V is final, [V? is not exact.
2397     if (tklass->as_instance_klass()->is_final()) {
2398       if (tinst->is_valuetypeptr() && (tinst->ptr() == TypePtr::BotPTR || tinst->ptr() == TypePtr::TopPTR)) {
2399         return false;
2400       }
2401       return true;
2402     }
2403     return false;
2404   }
2405   const TypeAryPtr*  tap;
2406   if (_elem->isa_narrowoop())
2407     tap = _elem->make_ptr()->isa_aryptr();
2408   else
2409     tap = _elem->isa_aryptr();
2410   if (tap)
2411     return tap->ary()->ary_must_be_exact();

2854 /**

2856  *
2857  * @param other  type to meet with
2858  */
2859 const TypePtr* TypePtr::xmeet_speculative(const TypePtr* other) const {
2860   bool this_has_spec = (_speculative != NULL);
2861   bool other_has_spec = (other->speculative() != NULL);
2862 
2863   if (!this_has_spec && !other_has_spec) {
2864     return NULL;
2865   }
2866 
2867   // If we are at a point where control flow meets and one branch has
2868   // a speculative type and the other has not, we meet the speculative
2869   // type of one branch with the actual type of the other. If the
2870   // actual type is exact and the speculative is as well, then the
2871   // result is a speculative type which is exact and we can continue
2872   // speculation further.
2873   const TypePtr* this_spec = _speculative;
2874   const TypePtr* other_spec = other->speculative();

2877     this_spec = this;
2878   }
2879 
2880   if (!other_has_spec) {
2881     other_spec = other;
2882   }
2883

3135 //------------------------------meet-------------------------------------------
3136 // Compute the MEET of two types.  It returns a new Type object.
3137 const Type *TypeRawPtr::xmeet( const Type *t ) const {
3138   // Perform a fast test for common case; meeting the same types together.
3139   if( this == t ) return this;  // Meeting same type-rep?
3140 
3141   // Current "this->_base" is RawPtr
3142   switch( t->base() ) {         // switch on original type
3143   case Bottom:                  // Ye Olde Default
3144     return t;
3145   case Top:
3146     return this;
3147   case AnyPtr:                  // Meeting to AnyPtrs
3148     break;
3149   case RawPtr: {                // might be top, bot, any/not or constant
3150     enum PTR tptr = t->is_ptr()->ptr();
3151     enum PTR ptr = meet_ptr( tptr );
3152     if( ptr == Constant ) {     // Cannot be equal constants, so...
3153       if( tptr == Constant && _ptr != Constant)  return t;
3154       if( _ptr == Constant && tptr != Constant)  return this;
3155       ptr = NotNull;            // Fall down in lattice
3156     }
3157     return make( ptr );
3158   }
3159 
3160   case OopPtr:
3161   case InstPtr:
3162   case AryPtr:
3163   case MetadataPtr:
3164   case KlassPtr:
3165     return TypePtr::BOTTOM;     // Oop meet raw is not well defined
3166   default:                      // All else is a mistake
3167     typerr(t);
3168   }
3169 
3170   // Found an AnyPtr type vs self-RawPtr type
3171   const TypePtr *tp = t->is_ptr();
3172   switch (tp->ptr()) {
3173   case TypePtr::TopPTR:  return this;
3174   case TypePtr::BotPTR:  return t;
3175   case TypePtr::Null:
3176     if( _ptr == TypePtr::TopPTR ) return t;
3177     return TypeRawPtr::BOTTOM;
3178   case TypePtr::NotNull: return TypePtr::make(AnyPtr, meet_ptr(TypePtr::NotNull), tp->meet_offset(0), tp->speculative(), tp->inline_depth());

3180     if( _ptr == TypePtr::Constant) return this;
3181     return make( meet_ptr(TypePtr::AnyNull) );

3183   }

3388   case NarrowOop:
3389   case NarrowKlass:
3390   case Bottom:                  // Ye Olde Default
3391     return Type::BOTTOM;
3392   case Top:
3393     return this;
3394 
3395   default:                      // All else is a mistake
3396     typerr(t);
3397 
3398   case RawPtr:
3399   case MetadataPtr:
3400   case KlassPtr:
3401     return TypePtr::BOTTOM;     // Oop meet raw is not well defined
3402 
3403   case AnyPtr: {
3404     // Found an AnyPtr type vs self-OopPtr type
3405     const TypePtr *tp = t->is_ptr();
3406     Offset offset = meet_offset(tp->offset());
3407     PTR ptr = meet_ptr(tp->ptr());
3408     const TypePtr* speculative = xmeet_speculative(tp);
3409     int depth = meet_inline_depth(tp->inline_depth());
3410     switch (tp->ptr()) {
3411     case Null:
3412       if (ptr == Null)  return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
3413       // else fall through:
3414     case TopPTR:
3415     case AnyNull: {
3416       int instance_id = meet_instance_id(InstanceTop);
3417       return make(ptr, offset, instance_id, speculative, depth);
3418     }
3419     case BotPTR:
3420     case NotNull:
3421       return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
3422     default: typerr(t);
3423     }
3424   }
3425 
3426   case OopPtr: {                 // Meeting to other OopPtrs
3427     const TypeOopPtr *tp = t->is_oopptr();
3428     int instance_id = meet_instance_id(tp->instance_id());
3429     const TypePtr* speculative = xmeet_speculative(tp);
3430     int depth = meet_inline_depth(tp->inline_depth());
3431     return make(meet_ptr(tp->ptr()), meet_offset(tp->offset()), instance_id, speculative, depth);
3432   }
3433 
3434   case InstPtr:                  // For these, flip the call around to cut down
3435   case AryPtr:
3436     return t->xmeet(this);      // Call in reverse direction
3437 
3438   } // End of switch
3439   return this;                  // Return the double constant
3440 }

3451 //--------------------------make_from_klass_common-----------------------------
3452 // Computes the element-type given a klass.
3453 const TypeOopPtr* TypeOopPtr::make_from_klass_common(ciKlass *klass, bool klass_change, bool try_for_exact) {
3454   if (klass->is_instance_klass() || klass->is_valuetype()) {
3455     Compile* C = Compile::current();
3456     Dependencies* deps = C->dependencies();
3457     assert((deps != NULL) == (C->method() != NULL && C->method()->code_size() > 0), "sanity");
3458     // Element is an instance
3459     bool klass_is_exact = false;
3460     if (klass->is_loaded()) {
3461       // Try to set klass_is_exact.
3462       ciInstanceKlass* ik = klass->as_instance_klass();
3463       klass_is_exact = ik->is_final();
3464       if (!klass_is_exact && klass_change
3465           && deps != NULL && UseUniqueSubclasses) {
3466         ciInstanceKlass* sub = ik->unique_concrete_subklass();
3467         if (sub != NULL) {
3468           deps->assert_abstract_with_unique_concrete_subtype(ik, sub);
3469           klass = ik = sub;
3470           klass_is_exact = sub->is_final();
3471         }
3472       }
3473       if (!klass_is_exact && try_for_exact
3474           && deps != NULL && UseExactTypes) {
3475         if (!ik->is_interface() && !ik->has_subklass()) {
3476           // Add a dependence; if concrete subclass added we need to recompile
3477           deps->assert_leaf_type(ik);
3478           klass_is_exact = true;
3479         }
3480       }
3481     }
3482     return TypeInstPtr::make(TypePtr::BotPTR, klass, klass_is_exact, NULL, Offset(0), klass->flatten_array());
3483   } else if (klass->is_obj_array_klass()) {
3484     // Element is an object or value array. Recursively call ourself.
3485     const TypeOopPtr* etype = TypeOopPtr::make_from_klass_common(klass->as_array_klass()->element_klass(), /* klass_change= */ false, try_for_exact);
3486     bool null_free = klass->is_loaded() && klass->as_array_klass()->storage_properties().is_null_free();
3487     if (null_free) {
3488       assert(etype->is_valuetypeptr(), "must be a valuetypeptr");
3489       etype = etype->join_speculative(TypePtr::NOTNULL)->is_oopptr();
3490     }
3491     // [V? has a subtype: [V. So even though V is final, [V? is not exact.
3492     bool xk = etype->klass_is_exact() && (!etype->is_valuetypeptr() || null_free);
3493 
3494     // Use exact element type to determine null-free/flattened properties
3495     const TypeOopPtr* exact_etype = TypeOopPtr::make_from_klass_common(klass->as_array_klass()->element_klass(), /* klass_change= */ true, try_for_exact);
3496     bool not_null_free = !exact_etype->can_be_value_type();
3497     assert(!(not_null_free && null_free), "inconsistent null-free information");
3498     bool not_flat = !ValueArrayFlatten || not_null_free || (exact_etype->is_valuetypeptr() && !exact_etype->value_klass()->flatten_array());
3499 
3500     const TypeAry* arr0 = TypeAry::make(etype, TypeInt::POS, false, not_flat, not_null_free);
3501     // We used to pass NotNull in here, asserting that the sub-arrays
3502     // are all not-null.  This is not true in generally, as code can
3503     // slam NULLs down in the subarrays.
3504     const TypeAryPtr* arr = TypeAryPtr::make(TypePtr::BotPTR, arr0, klass, xk, Offset(0));
3505     return arr;
3506   } else if (klass->is_type_array_klass()) {
3507     // Element is an typeArray
3508     const Type* etype = get_const_basic_type(klass->as_type_array_klass()->element_type());
3509     const TypeAry* arr0 = TypeAry::make(etype, TypeInt::POS,
3510                                         /* stable= */ false, /* not_flat= */ true, /* not_null_free= */ true);
3511     // We used to pass NotNull in here, asserting that the array pointer
3512     // is not-null. That was not true in general.
3513     const TypeAryPtr* arr = TypeAryPtr::make(TypePtr::BotPTR, arr0, klass, true, Offset(0));
3514     return arr;
3515   } else if (klass->is_value_array_klass()) {
3516     ciValueKlass* vk = klass->as_array_klass()->element_klass()->as_value_klass();
3517     const TypeAry* arr0 = TypeAry::make(TypeValueType::make(vk), TypeInt::POS);
3518     const TypeAryPtr* arr = TypeAryPtr::make(TypePtr::BotPTR, arr0, klass, true, Offset(0));
3519     return arr;
3520   } else {
3521     ShouldNotReachHere();
3522     return NULL;
3523   }
3524 }

3574     // We used to pass NotNull in here, asserting that the sub-arrays

3576     // slam NULLs down in the subarrays.

3609 //-----------------------------filter------------------------------------------
3610 // Do not allow interface-vs.-noninterface joins to collapse to top.
3611 const Type *TypeOopPtr::filter_helper(const Type *kills, bool include_speculative) const {
3612 
3613   const Type* ft = join_helper(kills, include_speculative);
3614   const TypeInstPtr* ftip = ft->isa_instptr();
3615   const TypeInstPtr* ktip = kills->isa_instptr();
3616 
3617   if (ft->empty()) {
3618     // Check for evil case of 'this' being a class and 'kills' expecting an
3619     // interface.  This can happen because the bytecodes do not contain
3620     // enough type info to distinguish a Java-level interface variable
3621     // from a Java-level object variable.  If we meet 2 classes which
3622     // both implement interface I, but their meet is at 'j/l/O' which
3623     // doesn't implement I, we have no way to tell if the result should
3624     // be 'I' or 'j/l/O'.  Thus we'll pick 'j/l/O'.  If this then flows
3625     // into a Phi which "knows" it's an Interface type we'll have to
3626     // uplift the type.
3627     if (!empty()) {
3628       if (ktip != NULL && ktip->is_loaded() && ktip->klass()->is_interface()) {
3629         return kills;           // Uplift to interface
3630       }
3631       // Also check for evil cases of 'this' being a class array
3632       // and 'kills' expecting an array of interfaces.
3633       Type::get_arrays_base_elements(ft, kills, NULL, &ktip);
3634       if (ktip != NULL && ktip->is_loaded() && ktip->klass()->is_interface()) {
3635         return kills;           // Uplift to array of interface
3636       }
3637     }
3638 
3639     return Type::TOP;           // Canonical empty value
3640   }
3641 
3642   // If we have an interface-typed Phi or cast and we narrow to a class type,
3643   // the join should report back the class.  However, if we have a J/L/Object
3644   // class-typed Phi and an interface flows in, it's possible that the meet &
3645   // join report an interface back out.  This isn't possible but happens
3646   // because the type system doesn't interact well with interfaces.
3647   if (ftip != NULL && ktip != NULL &&
3648       ftip->is_loaded() &&  ftip->klass()->is_interface() &&
3649       ktip->is_loaded() && !ktip->klass()->is_interface()) {
3650     assert(!ftip->klass_is_exact(), "interface could not be exact");
3651     return ktip->cast_to_ptr_type(ftip->ptr());
3652   }
3653 
3654   return ft;
3655 }

3891 //------------------------------xmeet_unloaded---------------------------------
3892 // Compute the MEET of two InstPtrs when at least one is unloaded.
3893 // Assume classes are different since called after check for same name/class-loader
3894 const TypeInstPtr *TypeInstPtr::xmeet_unloaded(const TypeInstPtr *tinst) const {
3895     Offset off = meet_offset(tinst->offset());
3896     PTR ptr = meet_ptr(tinst->ptr());
3897     int instance_id = meet_instance_id(tinst->instance_id());
3898     const TypePtr* speculative = xmeet_speculative(tinst);
3899     int depth = meet_inline_depth(tinst->inline_depth());
3900 
3901     const TypeInstPtr *loaded    = is_loaded() ? this  : tinst;
3902     const TypeInstPtr *unloaded  = is_loaded() ? tinst : this;
3903     if( loaded->klass()->equals(ciEnv::current()->Object_klass()) ) {
3904       //
3905       // Meet unloaded class with java/lang/Object
3906       //
3907       // Meet
3908       //          |                     Unloaded Class
3909       //  Object  |   TOP    |   AnyNull | Constant |   NotNull |  BOTTOM   |
3910       //  ===================================================================
3911       //   TOP    | ..........................Unloaded......................|
3912       //  AnyNull |  U-AN    |................Unloaded......................|
3913       // Constant | ... O-NN .................................. |   O-BOT   |
3914       //  NotNull | ... O-NN .................................. |   O-BOT   |
3915       //  BOTTOM  | ........................Object-BOTTOM ..................|
3916       //
3917       assert(loaded->ptr() != TypePtr::Null, "insanity check");
3918       //
3919       if(      loaded->ptr() == TypePtr::TopPTR ) { return unloaded; }
3920       else if (loaded->ptr() == TypePtr::AnyNull) { return TypeInstPtr::make(ptr, unloaded->klass(), false, NULL, off, false, instance_id, speculative, depth); }
3921       else if (loaded->ptr() == TypePtr::BotPTR ) { return TypeInstPtr::BOTTOM; }
3922       else if (loaded->ptr() == TypePtr::Constant || loaded->ptr() == TypePtr::NotNull) {
3923         if (unloaded->ptr() == TypePtr::BotPTR  ) { return TypeInstPtr::BOTTOM;  }
3924         else                                      { return TypeInstPtr::NOTNULL; }
3925       }
3926       else if( unloaded->ptr() == TypePtr::TopPTR )  { return unloaded; }
3927 
3928       return unloaded->cast_to_ptr_type(TypePtr::AnyNull)->is_instptr();
3929     }
3930 
3931     // Both are unloaded, not the same class, not Object
3932     // Or meet unloaded with a different loaded class, not java/lang/Object
3933     if( ptr != TypePtr::BotPTR ) {
3934       return TypeInstPtr::NOTNULL;
3935     }
3936     return TypeInstPtr::BOTTOM;
3937 }

3940 //------------------------------meet-------------------------------------------
3941 // Compute the MEET of two types.  It returns a new Type object.
3942 const Type *TypeInstPtr::xmeet_helper(const Type *t) const {
3943   // Perform a fast test for common case; meeting the same types together.
3944   if( this == t ) return this;  // Meeting same type-rep?
3945 
3946   // Current "this->_base" is Pointer
3947   switch (t->base()) {          // switch on original type
3948 
3949   case Int:                     // Mixing ints & oops happens when javac
3950   case Long:                    // reuses local variables
3951   case FloatTop:
3952   case FloatCon:
3953   case FloatBot:
3954   case DoubleTop:
3955   case DoubleCon:
3956   case DoubleBot:
3957   case NarrowOop:
3958   case NarrowKlass:
3959   case Bottom:                  // Ye Olde Default
3960     return Type::BOTTOM;
3961   case Top:
3962     return this;
3963 
3964   default:                      // All else is a mistake
3965     typerr(t);
3966 
3967   case MetadataPtr:
3968   case KlassPtr:
3969   case RawPtr: return TypePtr::BOTTOM;
3970 
3971   case AryPtr: {                // All arrays inherit from Object class
3972     const TypeAryPtr *tp = t->is_aryptr();
3973     Offset offset = meet_offset(tp->offset());
3974     PTR ptr = meet_ptr(tp->ptr());
3975     int instance_id = meet_instance_id(tp->instance_id());
3976     const TypePtr* speculative = xmeet_speculative(tp);
3977     int depth = meet_inline_depth(tp->inline_depth());
3978     switch (ptr) {
3979     case TopPTR:
3980     case AnyNull:                // Fall 'down' to dual of object klass
3981       // For instances when a subclass meets a superclass we fall
3982       // below the centerline when the superclass is exact. We need to
3983       // do the same here.
3984       if (klass()->equals(ciEnv::current()->Object_klass()) && !klass_is_exact()) {
3985         return TypeAryPtr::make(ptr, tp->ary(), tp->klass(), tp->klass_is_exact(), offset, tp->field_offset(), instance_id, speculative, depth);
3986       } else {
3987         // cannot subclass, so the meet has to fall badly below the centerline
3988         ptr = NotNull;
3989         instance_id = InstanceBot;
3990         return TypeInstPtr::make( ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
3991       }
3992     case Constant:
3993     case NotNull:
3994     case BotPTR:                // Fall down to object klass
3995       // LCA is object_klass, but if we subclass from the top we can do better
3996       if( above_centerline(_ptr) ) { // if( _ptr == TopPTR || _ptr == AnyNull )
3997         // If 'this' (InstPtr) is above the centerline and it is Object class
3998         // then we can subclass in the Java class hierarchy.
3999         // For instances when a subclass meets a superclass we fall
4000         // below the centerline when the superclass is exact. We need
4001         // to do the same here.
4002         if (klass()->equals(ciEnv::current()->Object_klass()) && !klass_is_exact()) {
4003           // that is, tp's array type is a subtype of my klass
4004           return TypeAryPtr::make(ptr, (ptr == Constant ? tp->const_oop() : NULL),
4005                                   tp->ary(), tp->klass(), tp->klass_is_exact(), offset, tp->field_offset(), instance_id, speculative, depth);
4006         }
4007       }
4008       // The other case cannot happen, since I cannot be a subtype of an array.
4009       // The meet falls down to Object class below centerline.
4010       if( ptr == Constant )
4011          ptr = NotNull;
4012       instance_id = InstanceBot;
4013       return make(ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
4014     default: typerr(t);
4015     }
4016   }
4017 
4018   case OopPtr: {                // Meeting to OopPtrs
4019     // Found a OopPtr type vs self-InstPtr type
4020     const TypeOopPtr *tp = t->is_oopptr();
4021     Offset offset = meet_offset(tp->offset());
4022     PTR ptr = meet_ptr(tp->ptr());
4023     switch (tp->ptr()) {
4024     case TopPTR:
4025     case AnyNull: {
4026       int instance_id = meet_instance_id(InstanceTop);
4027       const TypePtr* speculative = xmeet_speculative(tp);
4028       int depth = meet_inline_depth(tp->inline_depth());
4029       return make(ptr, klass(), klass_is_exact(),
4030                   (ptr == Constant ? const_oop() : NULL), offset, flat_array(), instance_id, speculative, depth);
4031     }
4032     case NotNull:
4033     case BotPTR: {
4034       int instance_id = meet_instance_id(tp->instance_id());
4035       const TypePtr* speculative = xmeet_speculative(tp);
4036       int depth = meet_inline_depth(tp->inline_depth());
4037       return TypeOopPtr::make(ptr, offset, instance_id, speculative, depth);
4038     }
4039     default: typerr(t);
4040     }
4041   }
4042 
4043   case AnyPtr: {                // Meeting to AnyPtrs
4044     // Found an AnyPtr type vs self-InstPtr type
4045     const TypePtr *tp = t->is_ptr();
4046     Offset offset = meet_offset(tp->offset());
4047     PTR ptr = meet_ptr(tp->ptr());
4048     int instance_id = meet_instance_id(InstanceTop);
4049     const TypePtr* speculative = xmeet_speculative(tp);
4050     int depth = meet_inline_depth(tp->inline_depth());
4051     switch (tp->ptr()) {
4052     case Null:
4053       if( ptr == Null ) return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
4054       // else fall through to AnyNull
4055     case TopPTR:
4056     case AnyNull: {
4057       return make(ptr, klass(), klass_is_exact(),
4058                   (ptr == Constant ? const_oop() : NULL), offset, flat_array(), instance_id, speculative, depth);
4059     }
4060     case NotNull:
4061     case BotPTR:
4062       return TypePtr::make(AnyPtr, ptr, offset, speculative,depth);
4063     default: typerr(t);
4064     }
4065   }
4066 
4067   /*
4068                  A-top         }
4069                /   |   \       }  Tops
4070            B-top A-any C-top   }
4071               | /  |  \ |      }  Any-nulls
4072            B-any   |   C-any   }
4073               |    |    |
4074            B-con A-con C-con   } constants; not comparable across classes
4075               |    |    |
4076            B-not   |   C-not   }
4077               | \  |  / |      }  not-nulls
4078            B-bot A-not C-bot   }
4079                \   |   /       }  Bottoms
4080                  A-bot         }
4081   */
4082 
4083   case InstPtr: {                // Meeting 2 Oops?
4084     // Found an InstPtr sub-type vs self-InstPtr type
4085     const TypeInstPtr *tinst = t->is_instptr();
4086     Offset off = meet_offset( tinst->offset() );
4087     PTR ptr = meet_ptr( tinst->ptr() );
4088     int instance_id = meet_instance_id(tinst->instance_id());
4089     const TypePtr* speculative = xmeet_speculative(tinst);
4090     int depth = meet_inline_depth(tinst->inline_depth());
4091 
4092     // Check for easy case; klasses are equal (and perhaps not loaded!)
4093     // If we have constants, then we created oops so classes are loaded
4094     // and we can handle the constants further down.  This case handles
4095     // both-not-loaded or both-loaded classes
4096     if (ptr != Constant && klass()->equals(tinst->klass()) && klass_is_exact() == tinst->klass_is_exact() &&
4097         flat_array() == tinst->flat_array()) {
4098       return make(ptr, klass(), klass_is_exact(), NULL, off, flat_array(), instance_id, speculative, depth);
4099     }
4100 
4101     // Classes require inspection in the Java klass hierarchy.  Must be loaded.
4102     ciKlass* tinst_klass = tinst->klass();
4103     ciKlass* this_klass  = this->klass();
4104     bool tinst_xk = tinst->klass_is_exact();
4105     bool this_xk  = this->klass_is_exact();
4106     bool tinst_flat_array = tinst->flat_array();
4107     bool this_flat_array  = this->flat_array();
4108     if (!tinst_klass->is_loaded() || !this_klass->is_loaded() ) {
4109       // One of these classes has not been loaded
4110       const TypeInstPtr *unloaded_meet = xmeet_unloaded(tinst);
4111 #ifndef PRODUCT
4112       if( PrintOpto && Verbose ) {
4113         tty->print("meet of unloaded classes resulted in: "); unloaded_meet->dump(); tty->cr();
4114         tty->print("  this == "); this->dump(); tty->cr();
4115         tty->print(" tinst == "); tinst->dump(); tty->cr();
4116       }
4117 #endif
4118       return unloaded_meet;
4119     }
4120 
4121     // Handle mixing oops and interfaces first.
4122     if( this_klass->is_interface() && !(tinst_klass->is_interface() ||
4123                                         tinst_klass == ciEnv::current()->Object_klass())) {
4124       ciKlass *tmp = tinst_klass; // Swap interface around
4125       tinst_klass = this_klass;
4126       this_klass = tmp;
4127       bool tmp2 = tinst_xk;
4128       tinst_xk = this_xk;
4129       this_xk = tmp2;
4130       tmp2 = tinst_flat_array;
4131       tinst_flat_array = this_flat_array;
4132       this_flat_array = tmp2;
4133     }
4134     if (tinst_klass->is_interface() &&
4135         !(this_klass->is_interface() ||
4136           // Treat java/lang/Object as an honorary interface,
4137           // because we need a bottom for the interface hierarchy.
4138           this_klass == ciEnv::current()->Object_klass())) {
4139       // Oop meets interface!
4140 
4141       // See if the oop subtypes (implements) interface.
4142       ciKlass *k;
4143       bool xk;
4144       bool flat_array;
4145       if( this_klass->is_subtype_of( tinst_klass ) ) {
4146         // Oop indeed subtypes.  Now keep oop or interface depending
4147         // on whether we are both above the centerline or either is
4148         // below the centerline.  If we are on the centerline
4149         // (e.g., Constant vs. AnyNull interface), use the constant.
4150         k  = below_centerline(ptr) ? tinst_klass : this_klass;
4151         // If we are keeping this_klass, keep its exactness too.
4152         xk = below_centerline(ptr) ? tinst_xk    : this_xk;
4153         flat_array = below_centerline(ptr) ? tinst_flat_array    : this_flat_array;
4154       } else {                  // Does not implement, fall to Object
4155         // Oop does not implement interface, so mixing falls to Object
4156         // just like the verifier does (if both are above the
4157         // centerline fall to interface)
4158         k = above_centerline(ptr) ? tinst_klass : ciEnv::current()->Object_klass();
4159         xk = above_centerline(ptr) ? tinst_xk : false;
4160         flat_array = above_centerline(ptr) ? tinst_flat_array : false;
4161         // Watch out for Constant vs. AnyNull interface.
4162         if (ptr == Constant)  ptr = NotNull;   // forget it was a constant
4163         instance_id = InstanceBot;
4164       }
4165       ciObject* o = NULL;  // the Constant value, if any
4166       if (ptr == Constant) {
4167         // Find out which constant.
4168         o = (this_klass == klass()) ? const_oop() : tinst->const_oop();
4169       }
4170       return make(ptr, k, xk, o, off, flat_array, instance_id, speculative, depth);
4171     }
4172 
4173     // Either oop vs oop or interface vs interface or interface vs Object
4174 
4175     // !!! Here's how the symmetry requirement breaks down into invariants:
4176     // If we split one up & one down AND they subtype, take the down man.
4177     // If we split one up & one down AND they do NOT subtype, "fall hard".
4178     // If both are up and they subtype, take the subtype class.
4179     // If both are up and they do NOT subtype, "fall hard".
4180     // If both are down and they subtype, take the supertype class.
4181     // If both are down and they do NOT subtype, "fall hard".
4182     // Constants treated as down.
4183 
4184     // Now, reorder the above list; observe that both-down+subtype is also
4185     // "fall hard"; "fall hard" becomes the default case:
4186     // If we split one up & one down AND they subtype, take the down man.
4187     // If both are up and they subtype, take the subtype class.
4188 
4189     // If both are down and they subtype, "fall hard".
4190     // If both are down and they do NOT subtype, "fall hard".
4191     // If both are up and they do NOT subtype, "fall hard".
4192     // If we split one up & one down AND they do NOT subtype, "fall hard".
4193 
4194     // If a proper subtype is exact, and we return it, we return it exactly.
4195     // If a proper supertype is exact, there can be no subtyping relationship!
4196     // If both types are equal to the subtype, exactness is and-ed below the
4197     // centerline and or-ed above it.  (N.B. Constants are always exact.)
4198 
4199     // Check for subtyping:
4200     ciKlass *subtype = NULL;
4201     bool subtype_exact = false;
4202     bool flat_array = false;
4203     if (tinst_klass->equals(this_klass)) {
4204       subtype = this_klass;
4205       subtype_exact = below_centerline(ptr) ? (this_xk && tinst_xk) : (this_xk || tinst_xk);
4206       flat_array = below_centerline(ptr) ? (this_flat_array && tinst_flat_array) : (this_flat_array || tinst_flat_array);
4207     } else if(!tinst_xk && this_klass->is_subtype_of(tinst_klass) && (!tinst_flat_array || this_flat_array)) {
4208       subtype = this_klass;     // Pick subtyping class
4209       subtype_exact = this_xk;
4210       flat_array = this_flat_array;
4211     } else if(!this_xk && tinst_klass->is_subtype_of(this_klass) && (!this_flat_array || tinst_flat_array)) {
4212       subtype = tinst_klass;    // Pick subtyping class
4213       subtype_exact = tinst_xk;
4214       flat_array = tinst_flat_array;
4215     }
4216 
4217     if (subtype) {
4218       if (above_centerline(ptr)) { // both are up?
4219         this_klass = tinst_klass = subtype;
4220         this_xk = tinst_xk = subtype_exact;
4221         this_flat_array = tinst_flat_array = flat_array;
4222       } else if (above_centerline(this ->_ptr) && !above_centerline(tinst->_ptr)) {
4223         this_klass = tinst_klass; // tinst is down; keep down man
4224         this_xk = tinst_xk;
4225         this_flat_array = tinst_flat_array;
4226       } else if (above_centerline(tinst->_ptr) && !above_centerline(this ->_ptr)) {
4227         tinst_klass = this_klass; // this is down; keep down man
4228         tinst_xk = this_xk;
4229         tinst_flat_array = this_flat_array;
4230       } else {
4231         this_xk = subtype_exact;  // either they are equal, or we'll do an LCA
4232         this_flat_array = flat_array;
4233       }
4234     }
4235 
4236     // Check for classes now being equal
4237     if (tinst_klass->equals(this_klass)) {
4238       // If the klasses are equal, the constants may still differ.  Fall to
4239       // NotNull if they do (neither constant is NULL; that is a special case
4240       // handled elsewhere).
4241       ciObject* o = NULL;             // Assume not constant when done
4242       ciObject* this_oop  = const_oop();
4243       ciObject* tinst_oop = tinst->const_oop();
4244       if( ptr == Constant ) {
4245         if (this_oop != NULL && tinst_oop != NULL &&
4246             this_oop->equals(tinst_oop) )
4247           o = this_oop;
4248         else if (above_centerline(this ->_ptr))
4249           o = tinst_oop;
4250         else if (above_centerline(tinst ->_ptr))
4251           o = this_oop;
4252         else
4253           ptr = NotNull;
4254       }
4255       return make(ptr, this_klass, this_xk, o, off, this_flat_array, instance_id, speculative, depth);
4256     } // Else classes are not equal
4257 
4258     // Since klasses are different, we require a LCA in the Java
4259     // class hierarchy - which means we have to fall to at least NotNull.
4260     if( ptr == TopPTR || ptr == AnyNull || ptr == Constant )
4261       ptr = NotNull;
4262 
4263     instance_id = InstanceBot;
4264 
4265     // Now we find the LCA of Java classes
4266     ciKlass* k = this_klass->least_common_ancestor(tinst_klass);
4267     return make(ptr, k, false, NULL, off, false, instance_id, speculative, depth);
4268   } // End of case InstPtr
4269 
4270   case ValueType: {
4271     const TypeValueType* tv = t->is_valuetype();
4272     if (above_centerline(ptr())) {
4273       if (tv->value_klass()->is_subtype_of(_klass)) {
4274         return t;
4275       } else {
4276         return TypeInstPtr::NOTNULL;
4277       }
4278     } else {
4279       PTR ptr = this->_ptr;
4280       if (ptr == Constant) {
4281         ptr = NotNull;
4282       }
4283       if (tv->value_klass()->is_subtype_of(_klass)) {
4284         return TypeInstPtr::make(ptr, _klass);
4285       } else {
4286         return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass());
4287       }
4288     }
4289   }
4290 
4291   } // End of switch
4292   return this;                  // Return the double constant
4293 }

4607 //------------------------------meet-------------------------------------------
4608 // Compute the MEET of two types.  It returns a new Type object.
4609 const Type *TypeAryPtr::xmeet_helper(const Type *t) const {
4610   // Perform a fast test for common case; meeting the same types together.
4611   if( this == t ) return this;  // Meeting same type-rep?
4612   // Current "this->_base" is Pointer
4613   switch (t->base()) {          // switch on original type
4614 
4615   // Mixing ints & oops happens when javac reuses local variables
4616   case Int:
4617   case Long:
4618   case FloatTop:
4619   case FloatCon:
4620   case FloatBot:
4621   case DoubleTop:
4622   case DoubleCon:
4623   case DoubleBot:
4624   case NarrowOop:
4625   case NarrowKlass:
4626   case Bottom:                  // Ye Olde Default
4627     return Type::BOTTOM;
4628   case Top:
4629     return this;
4630 
4631   default:                      // All else is a mistake
4632     typerr(t);
4633 
4634   case OopPtr: {                // Meeting to OopPtrs
4635     // Found a OopPtr type vs self-AryPtr type
4636     const TypeOopPtr *tp = t->is_oopptr();
4637     Offset offset = meet_offset(tp->offset());
4638     PTR ptr = meet_ptr(tp->ptr());
4639     int depth = meet_inline_depth(tp->inline_depth());
4640     const TypePtr* speculative = xmeet_speculative(tp);
4641     switch (tp->ptr()) {
4642     case TopPTR:
4643     case AnyNull: {
4644       int instance_id = meet_instance_id(InstanceTop);
4645       return make(ptr, (ptr == Constant ? const_oop() : NULL),
4646                   _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4647     }
4648     case BotPTR:
4649     case NotNull: {
4650       int instance_id = meet_instance_id(tp->instance_id());
4651       return TypeOopPtr::make(ptr, offset, instance_id, speculative, depth);
4652     }
4653     default: ShouldNotReachHere();
4654     }
4655   }
4656 
4657   case AnyPtr: {                // Meeting two AnyPtrs
4658     // Found an AnyPtr type vs self-AryPtr type
4659     const TypePtr *tp = t->is_ptr();
4660     Offset offset = meet_offset(tp->offset());
4661     PTR ptr = meet_ptr(tp->ptr());
4662     const TypePtr* speculative = xmeet_speculative(tp);
4663     int depth = meet_inline_depth(tp->inline_depth());
4664     switch (tp->ptr()) {
4665     case TopPTR:
4666       return this;
4667     case BotPTR:
4668     case NotNull:
4669       return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
4670     case Null:
4671       if( ptr == Null ) return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
4672       // else fall through to AnyNull
4673     case AnyNull: {
4674       int instance_id = meet_instance_id(InstanceTop);
4675       return make(ptr, (ptr == Constant ? const_oop() : NULL),
4676                   _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4677     }
4678     default: ShouldNotReachHere();
4679     }
4680   }
4681 
4682   case MetadataPtr:
4683   case KlassPtr:
4684   case RawPtr: return TypePtr::BOTTOM;
4685 
4686   case AryPtr: {                // Meeting 2 references?
4687     const TypeAryPtr *tap = t->is_aryptr();
4688     Offset off = meet_offset(tap->offset());
4689     Offset field_off = meet_field_offset(tap->field_offset());
4690     const TypeAry *tary = _ary->meet_speculative(tap->_ary)->is_ary();
4691     PTR ptr = meet_ptr(tap->ptr());
4692     int instance_id = meet_instance_id(tap->instance_id());
4693     const TypePtr* speculative = xmeet_speculative(tap);
4694     int depth = meet_inline_depth(tap->inline_depth());
4695     ciKlass* lazy_klass = NULL;
4696     if (tary->_elem->isa_int()) {
4697       // Integral array element types have irrelevant lattice relations.
4698       // It is the klass that determines array layout, not the element type.
4699       if (_klass == NULL)
4700         lazy_klass = tap->_klass;
4701       else if (tap->_klass == NULL || tap->_klass == _klass) {
4702         lazy_klass = _klass;
4703       } else {
4704         // Something like byte[int+] meets char[int+].
4705         // This must fall to bottom, not (int[-128..65535])[int+].
4706         instance_id = InstanceBot;
4707         tary = TypeAry::make(Type::BOTTOM, tary->_size, tary->_stable, tary->_not_flat, tary->_not_null_free);
4708       }
4709     } else // Non integral arrays.
4710       // Must fall to bottom if exact klasses in upper lattice
4711       // are not equal or super klass is exact.
4712       if ((above_centerline(ptr) || ptr == Constant) && klass() != tap->klass() &&
4713           // meet with top[] and bottom[] are processed further down:
4714           tap->_klass != NULL && this->_klass != NULL &&
4715           // both are exact and not equal:
4716           ((tap->_klass_is_exact && this->_klass_is_exact) ||
4717            // 'tap' is exact and super or unrelated:
4718            (tap->_klass_is_exact && !tap->klass()->is_subtype_of(klass())) ||
4719            // 'this' is exact and super or unrelated:
4720            (this->_klass_is_exact && !klass()->is_subtype_of(tap->klass())))) {
4721       if (above_centerline(ptr) || (tary->_elem->make_ptr() && above_centerline(tary->_elem->make_ptr()->_ptr))) {
4722         tary = TypeAry::make(Type::BOTTOM, tary->_size, tary->_stable, tary->_not_flat, tary->_not_null_free);
4723       }
4724       return make(NotNull, NULL, tary, lazy_klass, false, off, field_off, InstanceBot, speculative, depth);
4725     } else if (klass() != NULL && tap->klass() != NULL &&
4726                klass()->as_array_klass()->storage_properties().value() != tap->klass()->as_array_klass()->storage_properties().value()) {
4727       // Meeting value type arrays with conflicting storage properties
4728       if (tary->_elem->isa_valuetype()) {
4729         // Result is flattened
4730         off = Offset(elem()->isa_valuetype() ? offset() : tap->offset());
4731         field_off = elem()->isa_valuetype() ? field_offset() : tap->field_offset();
4732       } else if (tary->_elem->make_oopptr() != NULL && tary->_elem->make_oopptr()->isa_instptr() && below_centerline(ptr)) {
4733         // Result is non-flattened
4734         off = Offset(flattened_offset()).meet(Offset(tap->flattened_offset()));
4735         field_off = Offset::bottom;
4736       }
4737     }
4738 
4739     bool xk = false;
4740     switch (tap->ptr()) {
4741     case AnyNull:
4742     case TopPTR:
4743       // Compute new klass on demand, do not use tap->_klass
4744       if (below_centerline(this->_ptr)) {
4745         xk = this->_klass_is_exact;
4746       } else {
4747         xk = (tap->_klass_is_exact || this->_klass_is_exact);
4748       }
4749       return make(ptr, const_oop(), tary, lazy_klass, xk, off, field_off, instance_id, speculative, depth);
4750     case Constant: {
4751       ciObject* o = const_oop();
4752       if( _ptr == Constant ) {
4753         if( tap->const_oop() != NULL && !o->equals(tap->const_oop()) ) {
4754           xk = (klass() == tap->klass());
4755           ptr = NotNull;
4756           o = NULL;
4757           instance_id = InstanceBot;
4758         } else {
4759           xk = true;
4760         }
4761       } else if(above_centerline(_ptr)) {
4762         o = tap->const_oop();
4763         xk = true;
4764       } else {
4765         // Only precise for identical arrays
4766         xk = this->_klass_is_exact && (klass() == tap->klass());
4767       }
4768       return TypeAryPtr::make(ptr, o, tary, lazy_klass, xk, off, field_off, instance_id, speculative, depth);
4769     }
4770     case NotNull:
4771     case BotPTR:
4772       // Compute new klass on demand, do not use tap->_klass
4773       if (above_centerline(this->_ptr))
4774             xk = tap->_klass_is_exact;
4775       else  xk = (tap->_klass_is_exact & this->_klass_is_exact) &&
4776               (klass() == tap->klass()); // Only precise for identical arrays
4777       return TypeAryPtr::make(ptr, NULL, tary, lazy_klass, xk, off, field_off, instance_id, speculative, depth);
4778     default: ShouldNotReachHere();
4779     }
4780   }
4781 
4782   // All arrays inherit from Object class
4783   case InstPtr: {
4784     const TypeInstPtr *tp = t->is_instptr();
4785     Offset offset = meet_offset(tp->offset());
4786     PTR ptr = meet_ptr(tp->ptr());
4787     int instance_id = meet_instance_id(tp->instance_id());
4788     const TypePtr* speculative = xmeet_speculative(tp);
4789     int depth = meet_inline_depth(tp->inline_depth());
4790     switch (ptr) {
4791     case TopPTR:
4792     case AnyNull:                // Fall 'down' to dual of object klass
4793       // For instances when a subclass meets a superclass we fall
4794       // below the centerline when the superclass is exact. We need to
4795       // do the same here.
4796       if (tp->klass()->equals(ciEnv::current()->Object_klass()) && !tp->klass_is_exact()) {
4797         return TypeAryPtr::make(ptr, _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4798       } else {
4799         // cannot subclass, so the meet has to fall badly below the centerline
4800         ptr = NotNull;
4801         instance_id = InstanceBot;
4802         return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
4803       }
4804     case Constant:
4805     case NotNull:
4806     case BotPTR:                // Fall down to object klass
4807       // LCA is object_klass, but if we subclass from the top we can do better
4808       if (above_centerline(tp->ptr())) {
4809         // If 'tp'  is above the centerline and it is Object class
4810         // then we can subclass in the Java class hierarchy.
4811         // For instances when a subclass meets a superclass we fall
4812         // below the centerline when the superclass is exact. We need
4813         // to do the same here.
4814         if (tp->klass()->equals(ciEnv::current()->Object_klass()) && !tp->klass_is_exact()) {
4815           // that is, my array type is a subtype of 'tp' klass
4816           return make(ptr, (ptr == Constant ? const_oop() : NULL),
4817                       _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4818         }
4819       }
4820       // The other case cannot happen, since t cannot be a subtype of an array.
4821       // The meet falls down to Object class below centerline.
4822       if( ptr == Constant )
4823          ptr = NotNull;
4824       instance_id = InstanceBot;
4825       return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
4826     default: typerr(t);
4827     }
4828   }
4829 
4830   case ValueType: {
4831     const TypeValueType* tv = t->is_valuetype();
4832     if (above_centerline(ptr())) {
4833       return TypeInstPtr::NOTNULL;
4834     } else {
4835       PTR ptr = this->_ptr;
4836       if (ptr == Constant) {
4837         ptr = NotNull;
4838       }
4839       return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass());
4840     }
4841   }
4842   }
4843   return this;                  // Lint noise
4844 }

4964 const TypePtr* TypeAryPtr::add_field_offset_and_offset(intptr_t offset) const {
4965   int adj = 0;
4966   if (offset != Type::OffsetBot && offset != Type::OffsetTop) {
4967     const Type* elemtype = elem();
4968     if (elemtype->isa_valuetype()) {
4969       if (_offset.get() != OffsetBot && _offset.get() != OffsetTop) {
4970         adj = _offset.get();
4971         offset += _offset.get();
4972       }
4973       uint header = arrayOopDesc::base_offset_in_bytes(T_OBJECT);
4974       if (_field_offset.get() != OffsetBot && _field_offset.get() != OffsetTop) {
4975         offset += _field_offset.get();
4976         if (_offset.get() == OffsetBot || _offset.get() == OffsetTop) {
4977           offset += header;
4978         }
4979       }
4980       if (offset >= (intptr_t)header || offset < 0) {
4981         // Try to get the field of the value type array element we are pointing to
4982         ciKlass* arytype_klass = klass();
4983         ciValueArrayKlass* vak = arytype_klass->as_value_array_klass();
4984         ciValueKlass* vk = vak->element_klass()->as_value_klass();
4985         int shift = vak->log2_element_size();
4986         int mask = (1 << shift) - 1;
4987         intptr_t field_offset = ((offset - header) & mask);
4988         ciField* field = vk->get_field_by_offset(field_offset + vk->first_field_offset(), false);
4989         if (field == NULL) {
4990           // This may happen with nested AddP(base, AddP(base, base, offset), longcon(16))
4991           return add_offset(offset);
4992         } else {
4993           return with_field_offset(field_offset)->add_offset(offset - field_offset - adj);
4994         }
4995       }
4996     }
4997   }
4998   return add_offset(offset - adj);
4999 }

5238 const Type *TypeMetadataPtr::xmeet( const Type *t ) const {
5239   // Perform a fast test for common case; meeting the same types together.
5240   if( this == t ) return this;  // Meeting same type-rep?
5241 
5242   // Current "this->_base" is OopPtr
5243   switch (t->base()) {          // switch on original type
5244 
5245   case Int:                     // Mixing ints & oops happens when javac
5246   case Long:                    // reuses local variables
5247   case FloatTop:

5251   case DoubleCon:
5252   case DoubleBot:
5253   case NarrowOop:
5254   case NarrowKlass:
5255   case Bottom:                  // Ye Olde Default

5257   case Top:
5258     return this;
5259 
5260   default:                      // All else is a mistake
5261     typerr(t);
5262 
5263   case AnyPtr: {
5264     // Found an AnyPtr type vs self-OopPtr type
5265     const TypePtr *tp = t->is_ptr();
5266     Offset offset = meet_offset(tp->offset());
5267     PTR ptr = meet_ptr(tp->ptr());

5272     case TopPTR:

5283   case RawPtr:
5284   case KlassPtr:
5285   case OopPtr:
5286   case InstPtr:

5289

5292     Offset offset = meet_offset(tp->offset());
5293     PTR tptr = tp->ptr();

5295     ciMetadata* md = (tptr == TopPTR) ? metadata() : tp->metadata();

5428 //----------------------compute_klass------------------------------------------
5429 // Compute the defining klass for this class
5430 ciKlass* TypeAryPtr::compute_klass(DEBUG_ONLY(bool verify)) const {
5431   // Compute _klass based on element type.
5432   ciKlass* k_ary = NULL;
5433   const TypeAryPtr *tary;
5434   const Type* el = elem();
5435   if (el->isa_narrowoop()) {
5436     el = el->make_ptr();
5437   }
5438 
5439   // Get element klass
5440   if (el->isa_instptr()) {
5441     // Compute object array klass from element klass
5442     bool null_free = el->is_valuetypeptr() && el->isa_instptr()->ptr() != TypePtr::TopPTR && !el->isa_instptr()->maybe_null();
5443     k_ary = ciArrayKlass::make(el->is_oopptr()->klass(), null_free);
5444   } else if (el->isa_valuetype()) {
5445     if (el->value_klass() != NULL) {
5446       k_ary = ciArrayKlass::make(el->value_klass(), /* null_free */ true);
5447     }
5448   } else if ((tary = el->isa_aryptr()) != NULL) {
5449     // Compute array klass from element klass
5450     ciKlass* k_elem = tary->klass();
5451     // If element type is something like bottom[], k_elem will be null.
5452     if (k_elem != NULL)
5453       k_ary = ciObjArrayKlass::make(k_elem);
5454   } else if ((el->base() == Type::Top) ||
5455              (el->base() == Type::Bottom)) {
5456     // element type of Bottom occurs from meet of basic type
5457     // and object; Top occurs when doing join on Bottom.
5458     // Leave k_ary at NULL.
5459   } else {
5460     // Cannot compute array klass directly from basic type,
5461     // since subtypes of TypeInt all have basic type T_INT.
5462 #ifdef ASSERT
5463     if (verify && el->isa_int()) {
5464       // Check simple cases when verifying klass.
5465       BasicType bt = T_ILLEGAL;
5466       if (el == TypeInt::BYTE) {
5467         bt = T_BYTE;
5468       } else if (el == TypeInt::SHORT) {
5469         bt = T_SHORT;
5470       } else if (el == TypeInt::CHAR) {
5471         bt = T_CHAR;

5488 //------------------------------klass------------------------------------------
5489 // Return the defining klass for this class
5490 ciKlass* TypeAryPtr::klass() const {
5491   if( _klass ) return _klass;   // Return cached value, if possible
5492 
5493   // Oops, need to compute _klass and cache it
5494   ciKlass* k_ary = compute_klass();
5495 
5496   if( this != TypeAryPtr::OOPS && this->dual() != TypeAryPtr::OOPS ) {
5497     // The _klass field acts as a cache of the underlying
5498     // ciKlass for this array type.  In order to set the field,
5499     // we need to cast away const-ness.
5500     //
5501     // IMPORTANT NOTE: we *never* set the _klass field for the
5502     // type TypeAryPtr::OOPS.  This Type is shared between all
5503     // active compilations.  However, the ciKlass which represents
5504     // this Type is *not* shared between compilations, so caching
5505     // this value would result in fetching a dangling pointer.
5506     //
5507     // Recomputing the underlying ciKlass for each request is
5508     // a bit less efficient than caching, but calls to
5509     // TypeAryPtr::OOPS->klass() are not common enough to matter.
5510     ((TypeAryPtr*)this)->_klass = k_ary;
5511     if (UseCompressedOops && k_ary != NULL && k_ary->is_obj_array_klass() &&
5512         offset() != 0 && offset() != arrayOopDesc::length_offset_in_bytes()) {
5513       ((TypeAryPtr*)this)->_is_ptr_to_narrowoop = true;
5514     }
5515   }
5516   return k_ary;
5517 }

5560 //------------------------------xmeet------------------------------------------
5561 // Compute the MEET of two types, return a new Type object.
5562 const Type    *TypeKlassPtr::xmeet( const Type *t ) const {
5563   // Perform a fast test for common case; meeting the same types together.
5564   if( this == t ) return this;  // Meeting same type-rep?
5565 
5566   // Current "this->_base" is Pointer
5567   switch (t->base()) {          // switch on original type
5568 
5569   case Int:                     // Mixing ints & oops happens when javac
5570   case Long:                    // reuses local variables
5571   case FloatTop:
5572   case FloatCon:
5573   case FloatBot:
5574   case DoubleTop:
5575   case DoubleCon:
5576   case DoubleBot:
5577   case NarrowOop:
5578   case NarrowKlass:
5579   case Bottom:                  // Ye Olde Default
5580     return Type::BOTTOM;
5581   case Top:
5582     return this;
5583 
5584   default:                      // All else is a mistake
5585     typerr(t);
5586 
5587   case AnyPtr: {                // Meeting to AnyPtrs
5588     // Found an AnyPtr type vs self-KlassPtr type
5589     const TypePtr *tp = t->is_ptr();
5590     Offset offset = meet_offset(tp->offset());
5591     PTR ptr = meet_ptr(tp->ptr());
5592     switch (tp->ptr()) {
5593     case TopPTR:
5594       return this;
5595     case Null:
5596       if( ptr == Null ) return TypePtr::make(AnyPtr, ptr, offset, tp->speculative(), tp->inline_depth());
5597     case AnyNull:
5598       return make(ptr, klass(), offset, flat_array());
5599     case BotPTR:
5600     case NotNull:
5601       return TypePtr::make(AnyPtr, ptr, offset, tp->speculative(), tp->inline_depth());
5602     default: typerr(t);
5603     }
5604   }
5605 
5606   case RawPtr:
5607   case MetadataPtr:
5608   case OopPtr:
5609   case AryPtr:                  // Meet with AryPtr
5610   case InstPtr:                 // Meet with InstPtr
5611     return TypePtr::BOTTOM;
5612 
5613   //
5614   //             A-top         }
5615   //           /   |   \       }  Tops
5616   //       B-top A-any C-top   }
5617   //          | /  |  \ |      }  Any-nulls
5618   //       B-any   |   C-any   }
5619   //          |    |    |
5620   //       B-con A-con C-con   } constants; not comparable across classes
5621   //          |    |    |
5622   //       B-not   |   C-not   }
5623   //          | \  |  / |      }  not-nulls
5624   //       B-bot A-not C-bot   }
5625   //           \   |   /       }  Bottoms
5626   //             A-bot         }
5627   //
5628 
5629   case KlassPtr: {  // Meet two KlassPtr types
5630     const TypeKlassPtr *tkls = t->is_klassptr();
5631     Offset  off  = meet_offset(tkls->offset());
5632     PTR  ptr     = meet_ptr(tkls->ptr());
5633 
5634     if (klass() == NULL || tkls->klass() == NULL) {
5635       ciKlass* k = NULL;
5636       if (ptr == Constant) {
5637         k = (klass() == NULL) ? tkls->klass() : klass();
5638       }
5639       return make(ptr, k, off, false);
5640     }
5641 
5642     // Check for easy case; klasses are equal (and perhaps not loaded!)
5643     // If we have constants, then we created oops so classes are loaded
5644     // and we can handle the constants further down.  This case handles
5645     // not-loaded classes
5646     if (ptr != Constant && tkls->klass()->equals(klass()) && flat_array() == tkls->flat_array()) {
5647       return make(ptr, klass(), off, flat_array());
5648     }
5649 
5650     // Classes require inspection in the Java klass hierarchy.  Must be loaded.
5651     ciKlass* tkls_klass = tkls->klass();
5652     ciKlass* this_klass = this->klass();
5653     assert( tkls_klass->is_loaded(), "This class should have been loaded.");
5654     assert( this_klass->is_loaded(), "This class should have been loaded.");
5655     bool tkls_flat_array = tkls->flat_array();
5656     bool this_flat_array  = this->flat_array();
5657     bool flat_array = below_centerline(ptr) ? (this_flat_array && tkls_flat_array) : (this_flat_array || tkls_flat_array);
5658 
5659     // If 'this' type is above the centerline and is a superclass of the
5660     // other, we can treat 'this' as having the same type as the other.
5661     if ((above_centerline(this->ptr())) &&
5662         tkls_klass->is_subtype_of(this_klass)) {
5663       this_klass = tkls_klass;
5664     }
5665     // If 'tinst' type is above the centerline and is a superclass of the
5666     // other, we can treat 'tinst' as having the same type as the other.
5667     if ((above_centerline(tkls->ptr())) &&
5668         this_klass->is_subtype_of(tkls_klass)) {
5669       tkls_klass = this_klass;
5670     }
5671 
5672     // Check for classes now being equal
5673     if (tkls_klass->equals(this_klass)) {
5674       // If the klasses are equal, the constants may still differ.  Fall to
5675       // NotNull if they do (neither constant is NULL; that is a special case
5676       // handled elsewhere).
5677       if( ptr == Constant ) {
5678         if (this->_ptr == Constant && tkls->_ptr == Constant &&
5679             this->klass()->equals(tkls->klass()));
5680         else if (above_centerline(this->ptr()));
5681         else if (above_centerline(tkls->ptr()));
5682         else
5683           ptr = NotNull;
5684       }
5685       return make(ptr, this_klass, off, flat_array);
5686     } // Else classes are not equal
5687 
5688     // Since klasses are different, we require the LCA in the Java
5689     // class hierarchy - which means we have to fall to at least NotNull.
5690     if( ptr == TopPTR || ptr == AnyNull || ptr == Constant )
5691       ptr = NotNull;
5692     // Now we find the LCA of Java classes
5693     ciKlass* k = this_klass->least_common_ancestor(tkls_klass);
5694     return   make(ptr, k, off, k->is_valuetype() && k->flatten_array());
5695   } // End of case KlassPtr
5696 
5697   } // End of switch
5698   return this;                  // Return the double constant
5699 }

5781 //------------------------------make-------------------------------------------
5782 const TypeFunc* TypeFunc::make(ciMethod* method, bool is_osr_compilation) {
5783   Compile* C = Compile::current();
5784   const TypeFunc* tf = NULL;
5785   if (!is_osr_compilation) {
5786     tf = C->last_tf(method); // check cache
5787     if (tf != NULL)  return tf;  // The hit rate here is almost 50%.
5788   }
5789   // Value types are not passed/returned by reference, instead each field of
5790   // the value type is passed/returned as an argument. We maintain two views of
5791   // the argument/return list here: one based on the signature (with a value
5792   // type argument/return as a single slot), one based on the actual calling
5793   // convention (with a value type argument/return as a list of its fields).
5794   bool has_scalar_args = method->has_scalarized_args() && !is_osr_compilation;
5795   const TypeTuple* domain_sig = is_osr_compilation ? osr_domain() : TypeTuple::make_domain(method, false);
5796   const TypeTuple* domain_cc = has_scalar_args ? TypeTuple::make_domain(method, true) : domain_sig;
5797   ciSignature* sig = method->signature();
5798   bool has_scalar_ret = sig->returns_never_null() && sig->return_type()->as_value_klass()->can_be_returned_as_fields();
5799   const TypeTuple* range_sig = TypeTuple::make_range(sig, false);
5800   const TypeTuple* range_cc = has_scalar_ret ? TypeTuple::make_range(sig, true) : range_sig;
5801   tf = TypeFunc::make(domain_sig, domain_cc, range_sig, range_cc);
5802   if (!is_osr_compilation) {
5803     C->set_last_tf(method, tf);  // fill cache
5804   }
5805   return tf;
5806 }
