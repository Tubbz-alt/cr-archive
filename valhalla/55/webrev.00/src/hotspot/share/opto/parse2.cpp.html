  55 Node* Parse::record_profile_for_speculation_at_array_load(Node* ld) {
  56   // Feed unused profile data to type speculation
  57   if (UseTypeSpeculation && UseArrayLoadStoreProfile) {
  58     ciKlass* array_type = NULL;
  59     ciKlass* element_type = NULL;
  60     ProfilePtrKind element_ptr = ProfileMaybeNull;
  61     bool flat_array = true;
  62     bool null_free_array = true;
  63     method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
  64     if (element_type != NULL || element_ptr != ProfileMaybeNull) {
  65       ld = record_profile_for_speculation(ld, element_type, element_ptr);
  66     }
  67   }
  68   return ld;
  69 }

  72 //---------------------------------array_load----------------------------------
  73 void Parse::array_load(BasicType bt) {
  74   const Type* elemtype = Type::TOP;
  75   Node* adr = array_addressing(bt, 0, elemtype);
  76   if (stopped())  return;     // guaranteed null or range check
  77 
  78   Node* idx = pop();
  79   Node* ary = pop();
  80 
  81   // Handle value type arrays
  82   const TypeOopPtr* elemptr = elemtype->make_oopptr();
  83   const TypeAryPtr* ary_t = _gvn.type(ary)->is_aryptr();
  84   if (elemtype->isa_valuetype() != NULL) {
  85     C->set_flattened_accesses();
  86     // Load from flattened value type array
  87     Node* vt = ValueTypeNode::make_from_flattened(this, elemtype->value_klass(), ary, adr);
  88     push(vt);
  89     return;
  90   } else if (elemptr != NULL && elemptr->is_valuetypeptr() && !elemptr->maybe_null()) {
  91     // Load from non-flattened but flattenable value type array (elements can never be null)
  92     bt = T_VALUETYPE;
  93   } else if (!ary_t->is_not_flat()) {
  94     // Cannot statically determine if array is flattened, emit runtime check
  95     assert(ValueArrayFlatten && is_reference_type(bt) && elemptr->can_be_value_type() && !ary_t->klass_is_exact() && !ary_t->is_not_null_free() &&
  96            (!elemptr->is_valuetypeptr() || elemptr->value_klass()->flatten_array()), "array can't be flattened");
  97     Node* ctl = control();
  98     IdealKit ideal(this);
  99     IdealVariable res(ideal);
 100     ideal.declarations_done();
 101     ideal.if_then(is_non_flattened_array(ary)); {
 102       // non-flattened
 103       assert(ideal.ctrl()->in(0)->as_If()->is_non_flattened_array_check(&_gvn), "Should be found");
 104       sync_kit(ideal);
 105       const TypeAryPtr* adr_type = TypeAryPtr::get_array_body_type(bt);
 106       Node* ld = access_load_at(ary, adr, adr_type, elemptr, bt,
 107                                 IN_HEAP | IS_ARRAY | C2_CONTROL_DEPENDENT_LOAD, ctl);
 108       ideal.sync_kit(this);
 109       ideal.set(res, ld);
 110     } ideal.else_(); {
 111       // flattened
 112       sync_kit(ideal);
 113       if (elemptr->is_valuetypeptr()) {
 114         // Element type is known, cast and load from flattened representation
 115         ciValueKlass* vk = elemptr->value_klass();
 116         assert(vk->flatten_array() && elemptr->maybe_null(), "must be a flattenable and nullable array");
 117         ciArrayKlass* array_klass = ciArrayKlass::make(vk);
 118         const TypeAryPtr* arytype = TypeOopPtr::make_from_klass(array_klass)->isa_aryptr();
 119         Node* cast = _gvn.transform(new CheckCastPPNode(control(), ary, arytype));
 120         Node* casted_adr = array_element_address(cast, idx, T_VALUETYPE, ary_t->size(), control());
 121         // Re-execute flattened array load if buffering triggers deoptimization
 122         PreserveReexecuteState preexecs(this);
 123         jvms()->set_should_reexecute(true);
 124         inc_sp(2);
 125         Node* vt = ValueTypeNode::make_from_flattened(this, vk, cast, casted_adr)->allocate(this, false)->get_oop();
 126         ideal.set(res, vt);
 127         ideal.sync_kit(this);
 128       } else {
 129         // Element type is unknown, emit runtime call
 130         Node* kls = load_object_klass(ary);
 131         Node* k_adr = basic_plus_adr(kls, in_bytes(ArrayKlass::element_klass_offset()));
 132         Node* elem_klass = _gvn.transform(LoadKlassNode::make(_gvn, NULL, immutable_memory(), k_adr, TypeInstPtr::KLASS));
 133         Node* obj_size  = NULL;
 134         kill_dead_locals();
 135         // Re-execute flattened array load if buffering triggers deoptimization
 136         PreserveReexecuteState preexecs(this);
 137         jvms()->set_bci(_bci);
 138         jvms()->set_should_reexecute(true);
 139         inc_sp(2);
 140         Node* alloc_obj = new_instance(elem_klass, NULL, &obj_size, /*deoptimize_on_exception=*/true);
 141 
 142         AllocateNode* alloc = AllocateNode::Ideal_allocation(alloc_obj, &_gvn);
 143         assert(alloc->maybe_set_complete(&_gvn), "");
 144         alloc->initialization()->set_complete_with_arraycopy();
 145 
 146         // This membar keeps this access to an unknown flattened array
 147         // correctly ordered with other unknown and known flattened
 148         // array accesses.
 149         insert_mem_bar_volatile(Op_MemBarCPUOrder, C->get_alias_index(TypeAryPtr::VALUES));
 150 
 151         BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 152         // Unknown value type might contain reference fields
 153         if (false && !bs->array_copy_requires_gc_barriers(false, T_OBJECT, false, BarrierSetC2::Parsing)) {
 154           // FIXME 8230656 also merge changes from 8238759 in
 155           int base_off = sizeof(instanceOopDesc);
 156           Node* dst_base = basic_plus_adr(alloc_obj, base_off);
 157           Node* countx = obj_size;
 158           countx = _gvn.transform(new SubXNode(countx, MakeConX(base_off)));
 159           countx = _gvn.transform(new URShiftXNode(countx, intcon(LogBytesPerLong)));
 160 
 161           assert(Klass::_lh_log2_element_size_shift == 0, "use shift in place");
 162           Node* lhp = basic_plus_adr(kls, in_bytes(Klass::layout_helper_offset()));
 163           Node* elem_shift = make_load(NULL, lhp, TypeInt::INT, T_INT, MemNode::unordered);
 164           uint header = arrayOopDesc::base_offset_in_bytes(T_VALUETYPE);
 165           Node* base  = basic_plus_adr(ary, header);
 166           idx = Compile::conv_I2X_index(&_gvn, idx, TypeInt::POS, control());
 167           Node* scale = _gvn.transform(new LShiftXNode(idx, elem_shift));
 168           Node* adr = basic_plus_adr(ary, base, scale);
 169 
 170           access_clone(adr, dst_base, countx, false);
 171         } else {
 172           ideal.sync_kit(this);
 173           ideal.make_leaf_call(OptoRuntime::load_unknown_value_Type(),
 174                                CAST_FROM_FN_PTR(address, OptoRuntime::load_unknown_value),
 175                                "load_unknown_value",
 176                                ary, idx, alloc_obj);
 177           sync_kit(ideal);
 178         }
 179 
 180         // This makes sure no other thread sees a partially initialized buffered value
 181         insert_mem_bar_volatile(Op_MemBarStoreStore, Compile::AliasIdxRaw, alloc->proj_out_or_null(AllocateNode::RawAddress));
 182 
 183         // Same as MemBarCPUOrder above: keep this unknown flattened
 184         // array access correctly ordered with other flattened array
 185         // access
 186         insert_mem_bar_volatile(Op_MemBarCPUOrder, C->get_alias_index(TypeAryPtr::VALUES));
 187 
 188         // Prevent any use of the newly allocated value before it is
 189         // fully initialized
 190         alloc_obj = new CastPPNode(alloc_obj, _gvn.type(alloc_obj), true);
 191         alloc_obj->set_req(0, control());
 192         alloc_obj = _gvn.transform(alloc_obj);
 193 
 194         const Type* unknown_value = elemptr->is_instptr()->cast_to_flat_array();
 195         alloc_obj = _gvn.transform(new CheckCastPPNode(control(), alloc_obj, unknown_value));
 196 
 197         ideal.sync_kit(this);
 198         ideal.set(res, alloc_obj);
 199       }
 200     } ideal.end_if();
 201     sync_kit(ideal);
 202     Node* ld = _gvn.transform(ideal.value(res));
 203     ld = record_profile_for_speculation_at_array_load(ld);
 204     push_node(bt, ld);
 205     return;
 206   }
 207 
 208   if (elemtype == TypeInt::BOOL) {
 209     bt = T_BOOLEAN;
 210   }
 211   const TypeAryPtr* adr_type = TypeAryPtr::get_array_body_type(bt);
 212   Node* ld = access_load_at(ary, adr, adr_type, elemtype, bt,
 213                             IN_HEAP | IS_ARRAY | C2_CONTROL_DEPENDENT_LOAD);
 214   if (bt == T_VALUETYPE) {
 215     // Loading a non-flattened (but flattenable) value type from an array
 216     assert(!gvn().type(ld)->maybe_null(), "value type array elements should never be null");
 217     if (elemptr->value_klass()->is_scalarizable()) {
 218       ld = ValueTypeNode::make_from_oop(this, ld, elemptr->value_klass());
 219     }
 220   }
 221   if (!ld->is_ValueType()) {
 222     ld = record_profile_for_speculation_at_array_load(ld);
 223   }
 224 
 225   push_node(bt, ld);
 226 }

 229 //--------------------------------array_store----------------------------------
 230 void Parse::array_store(BasicType bt) {
 231   const Type* elemtype = Type::TOP;
 232   Node* adr = array_addressing(bt, type2size[bt], elemtype);
 233   if (stopped())  return;     // guaranteed null or range check
 234   Node* cast_val = NULL;
 235   if (bt == T_OBJECT) {
 236     cast_val = array_store_check();
 237     if (stopped()) return;
 238   }
 239   Node* val = pop_node(bt); // Value to store
 240   Node* idx = pop();        // Index in the array
 241   Node* ary = pop();        // The array itself
 242 
 243   const TypeAryPtr* ary_t = _gvn.type(ary)->is_aryptr();
 244   const TypeAryPtr* adr_type = TypeAryPtr::get_array_body_type(bt);
 245 
 246   if (elemtype == TypeInt::BOOL) {
 247     bt = T_BOOLEAN;
 248   } else if (bt == T_OBJECT) {
 249     elemtype = elemtype->make_oopptr();
 250     const Type* tval = _gvn.type(cast_val);
 251     // We may have lost type information for 'val' here due to the casts
 252     // emitted by the array_store_check code (see JDK-6312651)
 253     // TODO Remove this code once JDK-6312651 is in.
 254     const Type* tval_init = _gvn.type(val);
 255     bool can_be_value_type = tval->isa_valuetype() || (tval != TypePtr::NULL_PTR && tval_init->is_oopptr()->can_be_value_type() && tval->is_oopptr()->can_be_value_type());
 256     bool not_flattenable = !can_be_value_type || ((tval_init->is_valuetypeptr() || tval_init->isa_valuetype()) && !tval_init->value_klass()->flatten_array());
 257 
 258     if (!ary_t->is_not_null_free() && !can_be_value_type && (!tval->maybe_null() || !tval_init->maybe_null())) {
 259       // Storing a non-inline-type, mark array as not null-free.
 260       // This is only legal for non-null stores because the array_store_check passes for null.
 261       ary_t = ary_t->cast_to_not_null_free();
 262       Node* cast = _gvn.transform(new CheckCastPPNode(control(), ary, ary_t));
 263       replace_in_map(ary, cast);
 264       ary = cast;
 265     } else if (!ary_t->is_not_flat() && not_flattenable) {
 266       // Storing a non-flattenable value, mark array as not flat.
 267       ary_t = ary_t->cast_to_not_flat();
 268       if (tval != TypePtr::NULL_PTR) {
 269         // For NULL, this transformation is only valid after the null guard below
 270         Node* cast = _gvn.transform(new CheckCastPPNode(control(), ary, ary_t));
 271         replace_in_map(ary, cast);
 272         ary = cast;
 273       }
 274     }
 275 
 276     if (ary_t->elem()->isa_valuetype() != NULL) {
 277       // Store to flattened value type array
 278       C->set_flattened_accesses();
 279       if (!cast_val->is_ValueType()) {
 280         inc_sp(3);
 281         cast_val = null_check(cast_val);
 282         if (stopped()) return;
 283         dec_sp(3);
 284         cast_val = ValueTypeNode::make_from_oop(this, cast_val, ary_t->elem()->value_klass());
 285       }
 286       // Re-execute flattened array store if buffering triggers deoptimization
 287       PreserveReexecuteState preexecs(this);
 288       inc_sp(3);
 289       jvms()->set_should_reexecute(true);
 290       cast_val->as_ValueType()->store_flattened(this, ary, adr, NULL, 0, MO_UNORDERED | IN_HEAP | IS_ARRAY);
 291       return;
 292     } else if (elemtype->is_valuetypeptr() && !elemtype->maybe_null()) {
 293       // Store to non-flattened but flattenable value type array (elements can never be null)
 294       if (!cast_val->is_ValueType() && tval->maybe_null()) {
 295         inc_sp(3);
 296         cast_val = null_check(cast_val);
 297         if (stopped()) return;
 298         dec_sp(3);
 299       }
 300     } else if (!ary_t->is_not_flat()) {
 301       // Array might be flattened, emit runtime checks
 302       assert(ValueArrayFlatten && !not_flattenable && elemtype->is_oopptr()->can_be_value_type() &&
 303              !ary_t->klass_is_exact() && !ary_t->is_not_null_free(), "array can't be flattened");
 304       IdealKit ideal(this);
 305       ideal.if_then(is_non_flattened_array(ary)); {
 306         // non-flattened
 307         assert(ideal.ctrl()->in(0)->as_If()->is_non_flattened_array_check(&_gvn), "Should be found");
 308         sync_kit(ideal);
 309         gen_value_array_null_guard(ary, cast_val, 3);
 310         inc_sp(3);
 311         access_store_at(ary, adr, adr_type, cast_val, elemtype, bt, MO_UNORDERED | IN_HEAP | IS_ARRAY, false);
 312         dec_sp(3);
 313         ideal.sync_kit(this);
 314       } ideal.else_(); {
 315         Node* val = cast_val;
 316         // flattened
 317         if (!val->is_ValueType() && tval->maybe_null()) {
 318           // Add null check
 319           sync_kit(ideal);
 320           Node* null_ctl = top();
 321           val = null_check_oop(val, &null_ctl);
 322           if (null_ctl != top()) {
 323             PreserveJVMState pjvms(this);
 324             inc_sp(3);
 325             set_control(null_ctl);
 326             uncommon_trap(Deoptimization::Reason_null_check, Deoptimization::Action_none);
 327             dec_sp(3);
 328           }
 329           ideal.sync_kit(this);
 330         }
 331         // Try to determine the value klass
 332         ciValueKlass* vk = NULL;
 333         if (tval->isa_valuetype() || tval->is_valuetypeptr()) {
 334           vk = tval->value_klass();
 335         } else if (tval_init->isa_valuetype() || tval_init->is_valuetypeptr()) {
 336           vk = tval_init->value_klass();
 337         } else if (elemtype->is_valuetypeptr()) {
 338           vk = elemtype->value_klass();
 339         }
 340         Node* casted_ary = ary;
 341         if (vk != NULL && !stopped()) {
 342           // Element type is known, cast and store to flattened representation
 343           sync_kit(ideal);
 344           assert(vk->flatten_array() && elemtype->maybe_null(), "must be a flattenable and nullable array");
 345           ciArrayKlass* array_klass = ciArrayKlass::make(vk);
 346           const TypeAryPtr* arytype = TypeOopPtr::make_from_klass(array_klass)->isa_aryptr();
 347           casted_ary = _gvn.transform(new CheckCastPPNode(control(), casted_ary, arytype));
 348           Node* casted_adr = array_element_address(casted_ary, idx, T_OBJECT, arytype->size(), control());
 349           if (!val->is_ValueType()) {
 350             assert(!gvn().type(val)->maybe_null(), "value type array elements should never be null");
 351             val = ValueTypeNode::make_from_oop(this, val, vk);
 352           }
 353           // Re-execute flattened array store if buffering triggers deoptimization
 354           PreserveReexecuteState preexecs(this);
 355           inc_sp(3);
 356           jvms()->set_should_reexecute(true);
 357           val->as_ValueType()->store_flattened(this, casted_ary, casted_adr, NULL, 0, MO_UNORDERED | IN_HEAP | IS_ARRAY);
 358           ideal.sync_kit(this);
 359         } else if (!ideal.ctrl()->is_top()) {
 360           // Element type is unknown, emit runtime call
 361           sync_kit(ideal);
 362 
 363           // This membar keeps this access to an unknown flattened
 364           // array correctly ordered with other unknown and known
 365           // flattened array accesses.
 366           insert_mem_bar_volatile(Op_MemBarCPUOrder, C->get_alias_index(TypeAryPtr::VALUES));
 367           ideal.sync_kit(this);
 368 
 369           ideal.make_leaf_call(OptoRuntime::store_unknown_value_Type(),
 370                                CAST_FROM_FN_PTR(address, OptoRuntime::store_unknown_value),
 371                                "store_unknown_value",
 372                                val, casted_ary, idx);
 373 
 374           sync_kit(ideal);
 375           // Same as MemBarCPUOrder above: keep this unknown
 376           // flattened array access correctly ordered with other
 377           // flattened array accesses.
 378           insert_mem_bar_volatile(Op_MemBarCPUOrder, C->get_alias_index(TypeAryPtr::VALUES));
 379           ideal.sync_kit(this);
 380         }
 381       }
 382       ideal.end_if();
 383       sync_kit(ideal);
 384       return;
 385     } else if (!ary_t->is_not_null_free()) {
 386       // Array is not flattened but may be null free
 387       assert(elemtype->is_oopptr()->can_be_value_type() && !ary_t->klass_is_exact(), "array can't be null free");
 388       ary = gen_value_array_null_guard(ary, cast_val, 3, true);
 389     }
 390   }
 391   inc_sp(3);
 392   access_store_at(ary, adr, adr_type, val, elemtype, bt, MO_UNORDERED | IN_HEAP | IS_ARRAY);
 393   dec_sp(3);
 394 }

 397 //------------------------------array_addressing-------------------------------
 398 // Pull array and index from the stack.  Compute pointer-to-element.
 399 Node* Parse::array_addressing(BasicType type, int vals, const Type*& elemtype) {
 400   Node *idx   = peek(0+vals);   // Get from stack without popping
 401   Node *ary   = peek(1+vals);   // in case of exception
 402 
 403   // Null check the array base, with correct stack contents
 404   ary = null_check(ary, T_ARRAY);
 405   // Compile-time detect of null-exception?
 406   if (stopped())  return top();
 407 
 408   const TypeAryPtr* arytype  = _gvn.type(ary)->is_aryptr();
 409   const TypeInt*    sizetype = arytype->size();
 410   elemtype = arytype->elem();
 411 
 412   if (UseUniqueSubclasses) {
 413     const Type* el = elemtype->make_ptr();
 414     if (el && el->isa_instptr()) {
 415       const TypeInstPtr* toop = el->is_instptr();
 416       if (toop->klass()->as_instance_klass()->unique_concrete_subklass()) {
 417         // If we load from "AbstractClass[]" we must see "ConcreteSubClass".
 418         const Type* subklass = Type::get_const_type(toop->klass());
 419         elemtype = subklass->join_speculative(el);
 420       }
 421     }
 422   }
 423 
 424   // Check for big class initializers with all constant offsets
 425   // feeding into a known-size array.
 426   const TypeInt* idxtype = _gvn.type(idx)->is_int();
 427   // See if the highest idx value is less than the lowest array bound,
 428   // and if the idx value cannot be negative:
 429   bool need_range_check = true;
 430   if (idxtype->_hi < sizetype->_lo && idxtype->_lo >= 0) {
 431     need_range_check = false;
 432     if (C->log() != NULL)   C->log()->elem("observe that='!need_range_check'");
 433   }
 434 
 435   ciKlass * arytype_klass = arytype->klass();
 436   if ((arytype_klass != NULL) && (!arytype_klass->is_loaded())) {
 437     // Only fails for some -Xcomp runs
 438     // The class is unloaded.  We have to run this bytecode in the interpreter.
 439     uncommon_trap(Deoptimization::Reason_unloaded,
 440                   Deoptimization::Action_reinterpret,
 441                   arytype->klass(), "!loaded array");
 442     return top();
 443   }
 444 
 445   // Do the range check
 446   if (GenerateRangeChecks && need_range_check) {
 447     Node* tst;
 448     if (sizetype->_hi <= 0) {
 449       // The greatest array bound is negative, so we can conclude that we're
 450       // compiling unreachable code, but the unsigned compare trick used below
 451       // only works with non-negative lengths.  Instead, hack "tst" to be zero so
 452       // the uncommon_trap path will always be taken.
 453       tst = _gvn.intcon(0);
 454     } else {
 455       // Range is constant in array-oop, so we can use the original state of mem
 456       Node* len = load_array_length(ary);
 457 
 458       // Test length vs index (standard trick using unsigned compare)
 459       Node* chk = _gvn.transform( new CmpUNode(idx, len) );
 460       BoolTest::mask btest = BoolTest::lt;
 461       tst = _gvn.transform( new BoolNode(chk, btest) );
 462     }
 463     RangeCheckNode* rc = new RangeCheckNode(control(), tst, PROB_MAX, COUNT_UNKNOWN);
 464     _gvn.set_type(rc, rc->Value(&_gvn));
 465     if (!tst->is_Con()) {
 466       record_for_igvn(rc);
 467     }
 468     set_control(_gvn.transform(new IfTrueNode(rc)));
 469     // Branch to failure if out of bounds
 470     {
 471       PreserveJVMState pjvms(this);
 472       set_control(_gvn.transform(new IfFalseNode(rc)));
 473       if (C->allow_range_check_smearing()) {
 474         // Do not use builtin_throw, since range checks are sometimes
 475         // made more stringent by an optimistic transformation.
 476         // This creates "tentative" range checks at this point,
 477         // which are not guaranteed to throw exceptions.
 478         // See IfNode::Ideal, is_range_check, adjust_check.
 479         uncommon_trap(Deoptimization::Reason_range_check,
 480                       Deoptimization::Action_make_not_entrant,
 481                       NULL, "range_check");
 482       } else {
 483         // If we have already recompiled with the range-check-widening
 484         // heroic optimization turned off, then we must really be throwing
 485         // range check exceptions.
 486         builtin_throw(Deoptimization::Reason_range_check, idx);
 487       }
 488     }
 489   }
 490   // Check for always knowing you are throwing a range-check exception
 491   if (stopped())  return top();
 492 
 493   // This could be an access to a value array. We can't tell if it's
 494   // flat or not. Speculating it's not leads to a much simpler graph
 495   // shape. Check profiling.
 496   // For aastore, by the time we're here, the array store check should
 497   // have already taken advantage of profiling to cast the array to an
 498   // exact type reported by profiling
 499   const TypeOopPtr* elemptr = elemtype->make_oopptr();
 500   if (elemtype->isa_valuetype() == NULL &&
 501       (elemptr == NULL || !elemptr->is_valuetypeptr() || elemptr->maybe_null()) &&
 502       !arytype->is_not_flat()) {
 503     assert(is_reference_type(type), "Only references");
 504     // First check the speculative type
 505     Deoptimization::DeoptReason reason = Deoptimization::Reason_speculate_class_check;
 506     ciKlass* array_type = arytype->speculative_type();
 507     if (too_many_traps_or_recompiles(reason) || array_type == NULL) {
 508       // No speculative type, check profile data at this bci
 509       array_type = NULL;
 510       reason = Deoptimization::Reason_class_check;
 511       if (UseArrayLoadStoreProfile && !too_many_traps_or_recompiles(reason)) {
 512         ciKlass* element_type = NULL;
 513         ProfilePtrKind element_ptr = ProfileMaybeNull;
 514         bool flat_array = true;
 515         bool null_free_array = true;
 516         method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
 517       }
 518     }
 519     if (array_type != NULL) {
 520       // Speculate that this array has the exact type reported by profile data
 521       Node* better_ary = NULL;
 522       Node* slow_ctl = type_check_receiver(ary, array_type, 1.0, &better_ary);
 523       { PreserveJVMState pjvms(this);
 524         set_control(slow_ctl);
 525         uncommon_trap_exact(reason, Deoptimization::Action_maybe_recompile);
 526       }
 527       replace_in_map(ary, better_ary);
 528       ary = better_ary;
 529       arytype  = _gvn.type(ary)->is_aryptr();
 530       elemtype = arytype->elem();
 531     }
 532   } else if (UseTypeSpeculation && UseArrayLoadStoreProfile) {
 533     // No need to speculate: feed profile data at this bci for the
 534     // array to type speculation
 535     ciKlass* array_type = NULL;
 536     ciKlass* element_type = NULL;
 537     ProfilePtrKind element_ptr = ProfileMaybeNull;
 538     bool flat_array = true;
 539     bool null_free_array = true;
 540     method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
 541     if (array_type != NULL) {
 542       record_profile_for_speculation(ary, array_type, ProfileMaybeNull);
 543     }
 544   }
 545 
 546   // We have no exact array type from profile data. Check profile data
 547   // for a non null free or non flat array. Non null free implies non
 548   // flat so check this one first. Speculating on a non null free
 549   // array doesn't help aaload but could be profitable for a
 550   // subsequent aastore.
 551   elemptr = elemtype->make_oopptr();
 552   if (!arytype->is_not_null_free() &&
 553       elemtype->isa_valuetype() == NULL &&
 554       (elemptr == NULL || !elemptr->is_valuetypeptr()) &&
 555       UseArrayLoadStoreProfile) {
 556     assert(is_reference_type(type), "");
 557     bool null_free_array = true;
 558     Deoptimization::DeoptReason reason = Deoptimization::Reason_none;
 559     if (arytype->speculative() != NULL &&
 560         arytype->speculative()->is_aryptr()->is_not_null_free() &&
 561         !too_many_traps_or_recompiles(Deoptimization::Reason_speculate_class_check)) {
 562       null_free_array = false;
 563       reason = Deoptimization::Reason_speculate_class_check;
 564     } else if (!too_many_traps_or_recompiles(Deoptimization::Reason_class_check)) {
 565       ciKlass* array_type = NULL;
 566       ciKlass* element_type = NULL;
 567       ProfilePtrKind element_ptr = ProfileMaybeNull;
 568       bool flat_array = true;
 569       method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
 570       reason = Deoptimization::Reason_class_check;
 571     }
 572     if (!null_free_array) {
 573       { // Deoptimize if null-free array
 574         BuildCutout unless(this, is_nullable_array(ary), PROB_MAX);
 575         uncommon_trap_exact(reason, Deoptimization::Action_maybe_recompile);
 576       }
 577       Node* better_ary = _gvn.transform(new CheckCastPPNode(control(), ary, arytype->cast_to_not_null_free()));
 578       replace_in_map(ary, better_ary);
 579       ary = better_ary;
 580       arytype = _gvn.type(ary)->is_aryptr();
 581     }
 582   }
 583 
 584   if (!arytype->is_not_flat() && elemtype->isa_valuetype() == NULL) {
 585     assert(is_reference_type(type), "");
 586     bool flat_array = true;
 587     Deoptimization::DeoptReason reason = Deoptimization::Reason_none;
 588     if (arytype->speculative() != NULL &&
 589         arytype->speculative()->is_aryptr()->is_not_flat() &&
 590         !too_many_traps_or_recompiles(Deoptimization::Reason_speculate_class_check)) {
 591       flat_array = false;
 592       reason = Deoptimization::Reason_speculate_class_check;
 593     } else if (UseArrayLoadStoreProfile && !too_many_traps_or_recompiles(reason)) {
 594       ciKlass* array_type = NULL;
 595       ciKlass* element_type = NULL;
 596       ProfilePtrKind element_ptr = ProfileMaybeNull;
 597       bool null_free_array = true;
 598       method()->array_access_profiled_type(bci(), array_type, element_type, element_ptr, flat_array, null_free_array);
 599       reason = Deoptimization::Reason_class_check;
 600     }
 601     if (!flat_array) {
 602       { // Deoptimize if flat array
 603         BuildCutout unless(this, is_non_flattened_array(ary), PROB_MAX);
 604         uncommon_trap_exact(reason, Deoptimization::Action_maybe_recompile);
 605       }
 606       Node* better_ary = _gvn.transform(new CheckCastPPNode(control(), ary, arytype->cast_to_not_flat()));
 607       replace_in_map(ary, better_ary);
 608       ary = better_ary;
 609       arytype = _gvn.type(ary)->is_aryptr();
 610     }
 611   }
 612 
 613   // Make array address computation control dependent to prevent it
 614   // from floating above the range check during loop optimizations.
 615   Node* ptr = array_element_address(ary, idx, type, sizetype, control());
 616   assert(ptr != top(), "top should go hand-in-hand with stopped");
 617 
 618   return ptr;
 619 }

 722 class SwitchRange : public StackObj {
 723   // a range of integers coupled with a bci destination
 724   jint _lo;                     // inclusive lower limit
 725   jint _hi;                     // inclusive upper limit
 726   int _dest;
 727   int _table_index;             // index into method data table
 728   float _cnt;                   // how many times this range was hit according to profiling
 729 
 730 public:
 731   jint lo() const              { return _lo;   }
 732   jint hi() const              { return _hi;   }
 733   int  dest() const            { return _dest; }
 734   int  table_index() const     { return _table_index; }
 735   bool is_singleton() const    { return _lo == _hi; }
 736   float cnt() const            { return _cnt; }
 737 
 738   void setRange(jint lo, jint hi, int dest, int table_index, float cnt) {
 739     assert(lo <= hi, "must be a non-empty range");
 740     _lo = lo, _hi = hi; _dest = dest; _table_index = table_index; _cnt = cnt;
 741     assert(_cnt >= 0, "");
 742   }
 743   bool adjoinRange(jint lo, jint hi, int dest, int table_index, float cnt, bool trim_ranges) {
 744     assert(lo <= hi, "must be a non-empty range");
 745     if (lo == _hi+1 && table_index == _table_index) {
 746       // see merge_ranges() comment below
 747       if (trim_ranges) {
 748         if (cnt == 0) {
 749           if (_cnt != 0) {
 750             return false;
 751           }
 752           if (dest != _dest) {
 753             _dest = never_reached;
 754           }
 755         } else {
 756           if (_cnt == 0) {
 757             return false;
 758           }
 759           if (dest != _dest) {
 760             return false;
 761           }
 762         }
 763       } else {
 764         if (dest != _dest) {
 765           return false;
 766         }
 767       }
 768       _hi = hi;
 769       _cnt += cnt;
 770       return true;
 771     }
 772     return false;
 773   }
 774 
 775   void set (jint value, int dest, int table_index, float cnt) {
 776     setRange(value, value, dest, table_index, cnt);
 777   }
 778   bool adjoin(jint value, int dest, int table_index, float cnt, bool trim_ranges) {
 779     return adjoinRange(value, value, dest, table_index, cnt, trim_ranges);
 780   }
 781   bool adjoin(SwitchRange& other) {
 782     return adjoinRange(other._lo, other._hi, other._dest, other._table_index, other._cnt, false);

 792     else
 793       tty->print(" {%d..%d}=>%d (cnt=%f)", lo(), hi(), dest(), cnt());
 794   }

 797 // We try to minimize the number of ranges and the size of the taken
 798 // ones using profiling data. When ranges are created,
 799 // SwitchRange::adjoinRange() only allows 2 adjoining ranges to merge
 800 // if both were never hit or both were hit to build longer unreached
 801 // ranges. Here, we now merge adjoining ranges with the same
 802 // destination and finally set destination of unreached ranges to the
 803 // special value never_reached because it can help minimize the number
 804 // of tests that are necessary.
 805 //
 806 // For instance:
 807 // [0, 1] to target1 sometimes taken
 808 // [1, 2] to target1 never taken
 809 // [2, 3] to target2 never taken
 810 // would lead to:
 811 // [0, 1] to target1 sometimes taken
 812 // [1, 3] never taken
 813 //
 814 // (first 2 ranges to target1 are not merged)
 815 static void merge_ranges(SwitchRange* ranges, int& rp) {
 816   if (rp == 0) {
 817     return;
 818   }
 819   int shift = 0;
 820   for (int j = 0; j < rp; j++) {
 821     SwitchRange& r1 = ranges[j-shift];
 822     SwitchRange& r2 = ranges[j+1];
 823     if (r1.adjoin(r2)) {
 824       shift++;
 825     } else if (shift > 0) {
 826       ranges[j+1-shift] = r2;
 827     }
 828   }
 829   rp -= shift;
 830   for (int j = 0; j <= rp; j++) {
 831     SwitchRange& r = ranges[j];
 832     if (r.cnt() == 0 && r.dest() != never_reached) {
 833       r.setRange(r.lo(), r.hi(), never_reached, r.table_index(), r.cnt());
 834     }
 835   }
 836 }

 838 //-------------------------------do_tableswitch--------------------------------
 839 void Parse::do_tableswitch() {
 840   Node* lookup = pop();
 841   // Get information about tableswitch
 842   int default_dest = iter().get_dest_table(0);
 843   int lo_index     = iter().get_int_table(1);
 844   int hi_index     = iter().get_int_table(2);
 845   int len          = hi_index - lo_index + 1;
 846 
 847   if (len < 1) {
 848     // If this is a backward branch, add safepoint
 849     maybe_add_safepoint(default_dest);
 850     merge(default_dest);
 851     return;
 852   }
 853 
 854   ciMethodData* methodData = method()->method_data();
 855   ciMultiBranchData* profile = NULL;
 856   if (methodData->is_mature() && UseSwitchProfiling) {
 857     ciProfileData* data = methodData->bci_to_data(bci());
 858     if (data != NULL && data->is_MultiBranchData()) {
 859       profile = (ciMultiBranchData*)data;
 860     }
 861   }
 862   bool trim_ranges = !method_data_update() && !C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if);
 863 
 864   // generate decision tree, using trichotomy when possible
 865   int rnum = len+2;
 866   bool makes_backward_branch = false;
 867   SwitchRange* ranges = NEW_RESOURCE_ARRAY(SwitchRange, rnum);
 868   int rp = -1;
 869   if (lo_index != min_jint) {
 870     uint cnt = 1;
 871     if (profile != NULL) {
 872       cnt = profile->default_count() / (hi_index != max_jint ? 2 : 1);
 873     }
 874     ranges[++rp].setRange(min_jint, lo_index-1, default_dest, NullTableIndex, cnt);
 875   }
 876   for (int j = 0; j < len; j++) {
 877     jint match_int = lo_index+j;
 878     int  dest      = iter().get_dest_table(j+3);
 879     makes_backward_branch |= (dest <= bci());
 880     int  table_index = method_data_update() ? j : NullTableIndex;
 881     uint cnt = 1;
 882     if (profile != NULL) {
 883       cnt = profile->count_at(j);
 884     }
 885     if (rp < 0 || !ranges[rp].adjoin(match_int, dest, table_index, cnt, trim_ranges)) {
 886       ranges[++rp].set(match_int, dest, table_index, cnt);
 887     }
 888   }
 889   jint highest = lo_index+(len-1);
 890   assert(ranges[rp].hi() == highest, "");
 891   if (highest != max_jint) {
 892     uint cnt = 1;
 893     if (profile != NULL) {
 894       cnt = profile->default_count() / (lo_index != min_jint ? 2 : 1);
 895     }
 896     if (!ranges[rp].adjoinRange(highest+1, max_jint, default_dest, NullTableIndex, cnt, trim_ranges)) {
 897       ranges[++rp].setRange(highest+1, max_jint, default_dest, NullTableIndex, cnt);
 898     }
 899   }
 900   assert(rp < len+2, "not too many ranges");
 901 
 902   if (trim_ranges) {
 903     merge_ranges(ranges, rp);
 904   }
 905 
 906   // Safepoint in case if backward branch observed
 907   if( makes_backward_branch && UseLoopSafepoints )
 908     add_safepoint();
 909 
 910   jump_switch_ranges(lookup, &ranges[0], &ranges[rp]);
 911 }

 914 //------------------------------do_lookupswitch--------------------------------
 915 void Parse::do_lookupswitch() {
 916   Node *lookup = pop();         // lookup value
 917   // Get information about lookupswitch
 918   int default_dest = iter().get_dest_table(0);
 919   int len          = iter().get_int_table(1);
 920 
 921   if (len < 1) {    // If this is a backward branch, add safepoint
 922     maybe_add_safepoint(default_dest);
 923     merge(default_dest);
 924     return;
 925   }
 926 
 927   ciMethodData* methodData = method()->method_data();
 928   ciMultiBranchData* profile = NULL;
 929   if (methodData->is_mature() && UseSwitchProfiling) {
 930     ciProfileData* data = methodData->bci_to_data(bci());
 931     if (data != NULL && data->is_MultiBranchData()) {
 932       profile = (ciMultiBranchData*)data;
 933     }
 934   }
 935   bool trim_ranges = !method_data_update() && !C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if);
 936 
 937   // generate decision tree, using trichotomy when possible
 938   jint* table = NEW_RESOURCE_ARRAY(jint, len*3);
 939   {
 940     for (int j = 0; j < len; j++) {
 941       table[3*j+0] = iter().get_int_table(2+2*j);
 942       table[3*j+1] = iter().get_dest_table(2+2*j+1);
 943       table[3*j+2] = profile == NULL ? 1 : profile->count_at(j);
 944     }
 945     qsort(table, len, 3*sizeof(table[0]), jint_cmp);
 946   }
 947 
 948   float defaults = 0;
 949   jint prev = min_jint;
 950   for (int j = 0; j < len; j++) {
 951     jint match_int = table[3*j+0];
 952     if (match_int != prev) {
 953       defaults += (float)match_int - prev;
 954     }
 955     prev = match_int+1;
 956   }
 957   if (prev-1 != max_jint) {
 958     defaults += (float)max_jint - prev + 1;
 959   }
 960   float default_cnt = 1;
 961   if (profile != NULL) {
 962     default_cnt = profile->default_count()/defaults;
 963   }
 964 
 965   int rnum = len*2+1;
 966   bool makes_backward_branch = false;
 967   SwitchRange* ranges = NEW_RESOURCE_ARRAY(SwitchRange, rnum);
 968   int rp = -1;
 969   for (int j = 0; j < len; j++) {
 970     jint match_int   = table[3*j+0];
 971     int  dest        = table[3*j+1];
 972     int  cnt         = table[3*j+2];
 973     int  next_lo     = rp < 0 ? min_jint : ranges[rp].hi()+1;
 974     int  table_index = method_data_update() ? j : NullTableIndex;
 975     makes_backward_branch |= (dest <= bci());
 976     float c = default_cnt * ((float)match_int - next_lo);
 977     if (match_int != next_lo && (rp < 0 || !ranges[rp].adjoinRange(next_lo, match_int-1, default_dest, NullTableIndex, c, trim_ranges))) {
 978       assert(default_dest != never_reached, "sentinel value for dead destinations");
 979       ranges[++rp].setRange(next_lo, match_int-1, default_dest, NullTableIndex, c);
 980     }
 981     if (rp < 0 || !ranges[rp].adjoin(match_int, dest, table_index, cnt, trim_ranges)) {
 982       assert(dest != never_reached, "sentinel value for dead destinations");
 983       ranges[++rp].set(match_int, dest, table_index, cnt);
 984     }
 985   }
 986   jint highest = table[3*(len-1)];
 987   assert(ranges[rp].hi() == highest, "");
 988   if (highest != max_jint &&
 989       !ranges[rp].adjoinRange(highest+1, max_jint, default_dest, NullTableIndex, default_cnt * ((float)max_jint - highest), trim_ranges)) {
 990     ranges[++rp].setRange(highest+1, max_jint, default_dest, NullTableIndex, default_cnt * ((float)max_jint - highest));
 991   }
 992   assert(rp < rnum, "not too many ranges");
 993 
 994   if (trim_ranges) {
 995     merge_ranges(ranges, rp);
 996   }
 997 
 998   // Safepoint in case backward branch observed
 999   if (makes_backward_branch && UseLoopSafepoints)
1000     add_safepoint();
1001 
1002   jump_switch_ranges(lookup, &ranges[0], &ranges[rp]);
1003 }

1053 // Estimate cost of performing a binary search on lo..hi
1054 static float compute_tree_cost(SwitchRange *lo, SwitchRange *hi, float total_cnt) {
1055   GrowableArray<SwitchRanges> tree;
1056   SwitchRanges root(lo, hi);
1057   tree.push(root);
1058 
1059   float cost = 0;
1060   do {
1061     SwitchRanges& r = *tree.adr_at(tree.length()-1);
1062     if (r._hi != r._lo) {
1063       if (r._mid == NULL) {
1064         float r_cnt = sum_of_cnts(r._lo, r._hi);
1065 
1066         if (r_cnt == 0) {
1067           tree.pop();
1068           cost = 0;
1069           continue;
1070         }
1071 
1072         SwitchRange* mid = NULL;
1073         mid = r._lo;
1074         for (float cnt = 0; ; ) {
1075           assert(mid <= r._hi, "out of bounds");
1076           cnt += mid->cnt();
1077           if (cnt > r_cnt / 2) {
1078             break;
1079           }
1080           mid++;
1081         }
1082         assert(mid <= r._hi, "out of bounds");
1083         r._mid = mid;
1084         r._cost = r_cnt / total_cnt;
1085       }
1086       r._cost += cost;
1087       if (r._state < SwitchRanges::LeftDone && r._mid > r._lo) {
1088         cost = 0;
1089         r._state = SwitchRanges::LeftDone;
1090         tree.push(SwitchRanges(r._lo, r._mid-1));
1091       } else if (r._state < SwitchRanges::RightDone) {
1092         cost = 0;
1093         r._state = SwitchRanges::RightDone;
1094         tree.push(SwitchRanges(r._mid == r._lo ? r._mid+1 : r._mid, r._hi));
1095       } else {
1096         tree.pop();
1097         cost = r._cost;
1098       }
1099     } else {
1100       tree.pop();
1101       cost = r._cost;
1102     }
1103   } while (tree.length() > 0);
1104 
1105 
1106   return cost;
1107 }

1109 // It sometimes pays off to test most common ranges before the binary search
1110 void Parse::linear_search_switch_ranges(Node* key_val, SwitchRange*& lo, SwitchRange*& hi) {
1111   uint nr = hi - lo + 1;
1112   float total_cnt = sum_of_cnts(lo, hi);
1113 
1114   float min = compute_tree_cost(lo, hi, total_cnt);
1115   float extra = 1;
1116   float sub = 0;
1117 
1118   SwitchRange* array1 = lo;
1119   SwitchRange* array2 = NEW_RESOURCE_ARRAY(SwitchRange, nr);
1120 
1121   SwitchRange* ranges = NULL;
1122 
1123   while (nr >= 2) {
1124     assert(lo == array1 || lo == array2, "one the 2 already allocated arrays");
1125     ranges = (lo == array1) ? array2 : array1;
1126 
1127     // Find highest frequency range
1128     SwitchRange* candidate = lo;
1129     for (SwitchRange* sr = lo+1; sr <= hi; sr++) {
1130       if (sr->cnt() > candidate->cnt()) {
1131         candidate = sr;
1132       }
1133     }
1134     SwitchRange most_freq = *candidate;
1135     if (most_freq.cnt() == 0) {
1136       break;
1137     }
1138 
1139     // Copy remaining ranges into another array
1140     int shift = 0;
1141     for (uint i = 0; i < nr; i++) {
1142       SwitchRange* sr = &lo[i];
1143       if (sr != candidate) {
1144         ranges[i-shift] = *sr;
1145       } else {
1146         shift++;
1147         if (i > 0 && i < nr-1) {
1148           SwitchRange prev = lo[i-1];
1149           prev.setRange(prev.lo(), sr->hi(), prev.dest(), prev.table_index(), prev.cnt());
1150           if (prev.adjoin(lo[i+1])) {
1151             shift++;
1152             i++;
1153           }
1154           ranges[i-shift] = prev;
1155         }
1156       }
1157     }
1158     nr -= shift;
1159 
1160     // Evaluate cost of testing the most common range and performing a
1161     // binary search on the other ranges
1162     float cost = extra + compute_tree_cost(&ranges[0], &ranges[nr-1], total_cnt);
1163     if (cost >= min) {
1164       break;
1165     }
1166     // swap arrays
1167     lo = &ranges[0];
1168     hi = &ranges[nr-1];
1169 
1170     // It pays off: emit the test for the most common range
1171     assert(most_freq.cnt() > 0, "must be taken");
1172     Node* val = _gvn.transform(new SubINode(key_val, _gvn.intcon(most_freq.lo())));
1173     Node* cmp = _gvn.transform(new CmpUNode(val, _gvn.intcon(most_freq.hi() - most_freq.lo())));
1174     Node* tst = _gvn.transform(new BoolNode(cmp, BoolTest::le));
1175     IfNode* iff = create_and_map_if(control(), tst, if_prob(most_freq.cnt(), total_cnt), if_cnt(most_freq.cnt()));
1176     jump_if_true_fork(iff, most_freq.dest(), most_freq.table_index(), false);
1177 
1178     sub += most_freq.cnt() / total_cnt;
1179     extra += 1 - sub;
1180     min = cost;
1181   }
1182 }

1184 //----------------------------create_jump_tables-------------------------------
1185 bool Parse::create_jump_tables(Node* key_val, SwitchRange* lo, SwitchRange* hi) {
1186   // Are jumptables enabled
1187   if (!UseJumpTables)  return false;
1188 
1189   // Are jumptables supported
1190   if (!Matcher::has_match_rule(Op_Jump))  return false;
1191 
1192   // Don't make jump table if profiling
1193   if (method_data_update())  return false;
1194 
1195   bool trim_ranges = !C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if);
1196 
1197   // Decide if a guard is needed to lop off big ranges at either (or
1198   // both) end(s) of the input set. We'll call this the default target
1199   // even though we can't be sure that it is the true "default".
1200 
1201   bool needs_guard = false;
1202   int default_dest;
1203   int64_t total_outlier_size = 0;
1204   int64_t hi_size = ((int64_t)hi->hi()) - ((int64_t)hi->lo()) + 1;
1205   int64_t lo_size = ((int64_t)lo->hi()) - ((int64_t)lo->lo()) + 1;
1206 
1207   if (lo->dest() == hi->dest()) {
1208     total_outlier_size = hi_size + lo_size;
1209     default_dest = lo->dest();
1210   } else if (lo_size > hi_size) {
1211     total_outlier_size = lo_size;
1212     default_dest = lo->dest();
1213   } else {
1214     total_outlier_size = hi_size;
1215     default_dest = hi->dest();
1216   }
1217 
1218   float total = sum_of_cnts(lo, hi);
1219   float cost = compute_tree_cost(lo, hi, total);
1220 
1221   // If a guard test will eliminate very sparse end ranges, then
1222   // it is worth the cost of an extra jump.
1223   float trimmed_cnt = 0;
1224   if (total_outlier_size > (MaxJumpTableSparseness * 4)) {
1225     needs_guard = true;
1226     if (default_dest == lo->dest()) {
1227       trimmed_cnt += lo->cnt();
1228       lo++;
1229     }
1230     if (default_dest == hi->dest()) {
1231       trimmed_cnt += hi->cnt();
1232       hi--;
1233     }
1234   }
1235 
1236   // Find the total number of cases and ranges
1237   int64_t num_cases = ((int64_t)hi->hi()) - ((int64_t)lo->lo()) + 1;
1238   int num_range = hi - lo + 1;
1239 
1240   // Don't create table if: too large, too small, or too sparse.
1241   if (num_cases > MaxJumpTableSize)
1242     return false;
1243   if (UseSwitchProfiling) {
1244     // MinJumpTableSize is set so with a well balanced binary tree,
1245     // when the number of ranges is MinJumpTableSize, it's cheaper to
1246     // go through a JumpNode that a tree of IfNodes. Average cost of a
1247     // tree of IfNodes with MinJumpTableSize is
1248     // log2f(MinJumpTableSize) comparisons. So if the cost computed
1249     // from profile data is less than log2f(MinJumpTableSize) then
1250     // going with the binary search is cheaper.
1251     if (cost < log2f(MinJumpTableSize)) {
1252       return false;
1253     }
1254   } else {
1255     if (num_cases < MinJumpTableSize)
1256       return false;
1257   }
1258   if (num_cases > (MaxJumpTableSparseness * num_range))
1259     return false;
1260 
1261   // Normalize table lookups to zero
1262   int lowval = lo->lo();
1263   key_val = _gvn.transform( new SubINode(key_val, _gvn.intcon(lowval)) );
1264 
1265   // Generate a guard to protect against input keyvals that aren't
1266   // in the switch domain.
1267   if (needs_guard) {
1268     Node*   size = _gvn.intcon(num_cases);
1269     Node*   cmp = _gvn.transform(new CmpUNode(key_val, size));
1270     Node*   tst = _gvn.transform(new BoolNode(cmp, BoolTest::ge));
1271     IfNode* iff = create_and_map_if(control(), tst, if_prob(trimmed_cnt, total), if_cnt(trimmed_cnt));
1272     jump_if_true_fork(iff, default_dest, NullTableIndex, trim_ranges && trimmed_cnt == 0);
1273 
1274     total -= trimmed_cnt;
1275   }
1276 
1277   // Create an ideal node JumpTable that has projections
1278   // of all possible ranges for a switch statement
1279   // The key_val input must be converted to a pointer offset and scaled.
1280   // Compare Parse::array_addressing above.
1281 
1282   // Clean the 32-bit int into a real 64-bit offset.
1283   // Otherwise, the jint value 0 might turn into an offset of 0x0800000000.
1284   const TypeInt* ikeytype = TypeInt::make(0, num_cases, Type::WidenMin);
1285   // Make I2L conversion control dependent to prevent it from
1286   // floating above the range check during loop optimizations.
1287   key_val = C->conv_I2X_index(&_gvn, key_val, ikeytype, control());
1288 
1289   // Shift the value by wordsize so we have an index into the table, rather
1290   // than a switch value
1291   Node *shiftWord = _gvn.MakeConX(wordSize);
1292   key_val = _gvn.transform( new MulXNode( key_val, shiftWord));
1293 
1294   // Create the JumpNode
1295   Arena* arena = C->comp_arena();
1296   float* probs = (float*)arena->Amalloc(sizeof(float)*num_cases);
1297   int i = 0;
1298   if (total == 0) {
1299     for (SwitchRange* r = lo; r <= hi; r++) {
1300       for (int64_t j = r->lo(); j <= r->hi(); j++, i++) {
1301         probs[i] = 1.0F / num_cases;
1302       }
1303     }
1304   } else {
1305     for (SwitchRange* r = lo; r <= hi; r++) {
1306       float prob = r->cnt()/total;
1307       for (int64_t j = r->lo(); j <= r->hi(); j++, i++) {
1308         probs[i] = prob / (r->hi() - r->lo() + 1);
1309       }
1310     }
1311   }
1312 
1313   ciMethodData* methodData = method()->method_data();
1314   ciMultiBranchData* profile = NULL;
1315   if (methodData->is_mature()) {
1316     ciProfileData* data = methodData->bci_to_data(bci());
1317     if (data != NULL && data->is_MultiBranchData()) {
1318       profile = (ciMultiBranchData*)data;
1319     }
1320   }
1321 
1322   Node* jtn = _gvn.transform(new JumpNode(control(), key_val, num_cases, probs, profile == NULL ? COUNT_UNKNOWN : total));
1323 
1324   // These are the switch destinations hanging off the jumpnode
1325   i = 0;
1326   for (SwitchRange* r = lo; r <= hi; r++) {
1327     for (int64_t j = r->lo(); j <= r->hi(); j++, i++) {
1328       Node* input = _gvn.transform(new JumpProjNode(jtn, i, r->dest(), (int)(j - lowval)));
1329       {
1330         PreserveJVMState pjvms(this);
1331         set_control(input);
1332         jump_if_always_fork(r->dest(), r->table_index(), trim_ranges && r->cnt() == 0);
1333       }
1334     }
1335   }
1336   assert(i == num_cases, "miscount of cases");
1337   stop_and_kill_map();  // no more uses for this JVMS
1338   return true;
1339 }

1341 //----------------------------jump_switch_ranges-------------------------------
1342 void Parse::jump_switch_ranges(Node* key_val, SwitchRange *lo, SwitchRange *hi, int switch_depth) {
1343   Block* switch_block = block();
1344   bool trim_ranges = !method_data_update() && !C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if);
1345 
1346   if (switch_depth == 0) {
1347     // Do special processing for the top-level call.
1348     assert(lo->lo() == min_jint, "initial range must exhaust Type::INT");
1349     assert(hi->hi() == max_jint, "initial range must exhaust Type::INT");
1350 
1351     // Decrement pred-numbers for the unique set of nodes.
1352 #ifdef ASSERT
1353     if (!trim_ranges) {
1354       // Ensure that the block's successors are a (duplicate-free) set.
1355       int successors_counted = 0;  // block occurrences in [hi..lo]
1356       int unique_successors = switch_block->num_successors();
1357       for (int i = 0; i < unique_successors; i++) {
1358         Block* target = switch_block->successor_at(i);
1359 
1360         // Check that the set of successors is the same in both places.
1361         int successors_found = 0;
1362         for (SwitchRange* p = lo; p <= hi; p++) {
1363           if (p->dest() == target->start())  successors_found++;
1364         }
1365         assert(successors_found > 0, "successor must be known");
1366         successors_counted += successors_found;
1367       }
1368       assert(successors_counted == (hi-lo)+1, "no unexpected successors");
1369     }
1370 #endif
1371 
1372     // Maybe prune the inputs, based on the type of key_val.
1373     jint min_val = min_jint;
1374     jint max_val = max_jint;
1375     const TypeInt* ti = key_val->bottom_type()->isa_int();
1376     if (ti != NULL) {
1377       min_val = ti->_lo;
1378       max_val = ti->_hi;
1379       assert(min_val <= max_val, "invalid int type");
1380     }
1381     while (lo->hi() < min_val) {
1382       lo++;
1383     }
1384     if (lo->lo() < min_val)  {
1385       lo->setRange(min_val, lo->hi(), lo->dest(), lo->table_index(), lo->cnt());
1386     }
1387     while (hi->lo() > max_val) {
1388       hi--;
1389     }
1390     if (hi->hi() > max_val) {
1391       hi->setRange(hi->lo(), max_val, hi->dest(), hi->table_index(), hi->cnt());
1392     }
1393 
1394     linear_search_switch_ranges(key_val, lo, hi);
1395   }
1396 
1397 #ifndef PRODUCT
1398   if (switch_depth == 0) {
1399     _max_switch_depth = 0;
1400     _est_switch_depth = log2_intptr((hi-lo+1)-1)+1;
1401   }
1402 #endif
1403 
1404   assert(lo <= hi, "must be a non-empty set of ranges");
1405   if (lo == hi) {
1406     jump_if_always_fork(lo->dest(), lo->table_index(), trim_ranges && lo->cnt() == 0);
1407   } else {
1408     assert(lo->hi() == (lo+1)->lo()-1, "contiguous ranges");
1409     assert(hi->lo() == (hi-1)->hi()+1, "contiguous ranges");
1410 
1411     if (create_jump_tables(key_val, lo, hi)) return;
1412 
1413     SwitchRange* mid = NULL;
1414     float total_cnt = sum_of_cnts(lo, hi);
1415 
1416     int nr = hi - lo + 1;
1417     if (UseSwitchProfiling) {
1418       // Don't keep the binary search tree balanced: pick up mid point
1419       // that split frequencies in half.
1420       float cnt = 0;
1421       for (SwitchRange* sr = lo; sr <= hi; sr++) {
1422         cnt += sr->cnt();
1423         if (cnt >= total_cnt / 2) {
1424           mid = sr;
1425           break;
1426         }
1427       }
1428     } else {
1429       mid = lo + nr/2;
1430 
1431       // if there is an easy choice, pivot at a singleton:
1432       if (nr > 3 && !mid->is_singleton() && (mid-1)->is_singleton())  mid--;
1433 
1434       assert(lo < mid && mid <= hi, "good pivot choice");
1435       assert(nr != 2 || mid == hi,   "should pick higher of 2");
1436       assert(nr != 3 || mid == hi-1, "should pick middle of 3");
1437     }
1438 
1439 
1440     Node *test_val = _gvn.intcon(mid == lo ? mid->hi() : mid->lo());
1441 
1442     if (mid->is_singleton()) {
1443       IfNode *iff_ne = jump_if_fork_int(key_val, test_val, BoolTest::ne, 1-if_prob(mid->cnt(), total_cnt), if_cnt(mid->cnt()));
1444       jump_if_false_fork(iff_ne, mid->dest(), mid->table_index(), trim_ranges && mid->cnt() == 0);
1445 
1446       // Special Case:  If there are exactly three ranges, and the high
1447       // and low range each go to the same place, omit the "gt" test,
1448       // since it will not discriminate anything.
1449       bool eq_test_only = (hi == lo+2 && hi->dest() == lo->dest() && mid == hi-1) || mid == lo;
1450 
1451       // if there is a higher range, test for it and process it:
1452       if (mid < hi && !eq_test_only) {
1453         // two comparisons of same values--should enable 1 test for 2 branches
1454         // Use BoolTest::lt instead of BoolTest::gt
1455         float cnt = sum_of_cnts(lo, mid-1);
1456         IfNode *iff_lt  = jump_if_fork_int(key_val, test_val, BoolTest::lt, if_prob(cnt, total_cnt), if_cnt(cnt));
1457         Node   *iftrue  = _gvn.transform( new IfTrueNode(iff_lt) );
1458         Node   *iffalse = _gvn.transform( new IfFalseNode(iff_lt) );
1459         { PreserveJVMState pjvms(this);
1460           set_control(iffalse);
1461           jump_switch_ranges(key_val, mid+1, hi, switch_depth+1);
1462         }
1463         set_control(iftrue);
1464       }
1465 
1466     } else {
1467       // mid is a range, not a singleton, so treat mid..hi as a unit
1468       float cnt = sum_of_cnts(mid == lo ? mid+1 : mid, hi);
1469       IfNode *iff_ge = jump_if_fork_int(key_val, test_val, mid == lo ? BoolTest::gt : BoolTest::ge, if_prob(cnt, total_cnt), if_cnt(cnt));
1470 
1471       // if there is a higher range, test for it and process it:
1472       if (mid == hi) {
1473         jump_if_true_fork(iff_ge, mid->dest(), mid->table_index(), trim_ranges && cnt == 0);
1474       } else {
1475         Node *iftrue  = _gvn.transform( new IfTrueNode(iff_ge) );
1476         Node *iffalse = _gvn.transform( new IfFalseNode(iff_ge) );
1477         { PreserveJVMState pjvms(this);
1478           set_control(iftrue);
1479           jump_switch_ranges(key_val, mid == lo ? mid+1 : mid, hi, switch_depth+1);
1480         }
1481         set_control(iffalse);
1482       }
1483     }
1484 
1485     // in any case, process the lower range
1486     if (mid == lo) {
1487       if (mid->is_singleton()) {
1488         jump_switch_ranges(key_val, lo+1, hi, switch_depth+1);
1489       } else {
1490         jump_if_always_fork(lo->dest(), lo->table_index(), trim_ranges && lo->cnt() == 0);
1491       }
1492     } else {
1493       jump_switch_ranges(key_val, lo, mid-1, switch_depth+1);
1494     }
1495   }
1496 
1497   // Decrease pred_count for each successor after all is done.
1498   if (switch_depth == 0) {
1499     int unique_successors = switch_block->num_successors();
1500     for (int i = 0; i < unique_successors; i++) {
1501       Block* target = switch_block->successor_at(i);
1502       // Throw away the pre-allocated path for each unique successor.
1503       target->next_path_num();
1504     }
1505   }
1506 
1507 #ifndef PRODUCT
1508   _max_switch_depth = MAX2(switch_depth, _max_switch_depth);
1509   if (TraceOptoParse && Verbose && WizardMode && switch_depth == 0) {
1510     SwitchRange* r;
1511     int nsing = 0;
1512     for( r = lo; r <= hi; r++ ) {
1513       if( r->is_singleton() )  nsing++;
1514     }
1515     tty->print(">>> ");
1516     _method->print_short_name();
1517     tty->print_cr(" switch decision tree");
1518     tty->print_cr("    %d ranges (%d singletons), max_depth=%d, est_depth=%d",
1519                   (int) (hi-lo+1), nsing, _max_switch_depth, _est_switch_depth);
1520     if (_max_switch_depth > _est_switch_depth) {
1521       tty->print_cr("******** BAD SWITCH DEPTH ********");
1522     }
1523     tty->print("   ");
1524     for( r = lo; r <= hi; r++ ) {
1525       r->print();
1526     }
1527     tty->cr();
1528   }
1529 #endif
1530 }

1573 void Parse::do_irem() {
1574   // Must keep both values on the expression-stack during null-check
1575   zero_check_int(peek());
1576   // Compile-time detect of null-exception?
1577   if (stopped())  return;
1578 
1579   Node* b = pop();
1580   Node* a = pop();
1581 
1582   const Type *t = _gvn.type(b);
1583   if (t != Type::TOP) {
1584     const TypeInt *ti = t->is_int();
1585     if (ti->is_con()) {
1586       int divisor = ti->get_con();
1587       // check for positive power of 2
1588       if (divisor > 0 &&
1589           (divisor & ~(divisor-1)) == divisor) {
1590         // yes !
1591         Node *mask = _gvn.intcon((divisor - 1));
1592         // Sigh, must handle negative dividends
1593         Node *zero = _gvn.intcon(0);
1594         IfNode *ifff = jump_if_fork_int(a, zero, BoolTest::lt, PROB_FAIR, COUNT_UNKNOWN);
1595         Node *iff = _gvn.transform( new IfFalseNode(ifff) );
1596         Node *ift = _gvn.transform( new IfTrueNode (ifff) );
1597         Node *reg = jump_if_join(ift, iff);
1598         Node *phi = PhiNode::make(reg, NULL, TypeInt::INT);
1599         // Negative path; negate/and/negate
1600         Node *neg = _gvn.transform( new SubINode(zero, a) );
1601         Node *andn= _gvn.transform( new AndINode(neg, mask) );
1602         Node *negn= _gvn.transform( new SubINode(zero, andn) );
1603         phi->init_req(1, negn);
1604         // Fast positive case
1605         Node *andx = _gvn.transform( new AndINode(a, mask) );
1606         phi->init_req(2, andx);
1607         // Push the merge
1608         push( _gvn.transform(phi) );
1609         return;
1610       }
1611     }
1612   }
1613   // Default case
1614   push( _gvn.transform( new ModINode(control(),a,b) ) );
1615 }

1654 static bool has_injected_profile(BoolTest::mask btest, Node* test, int& taken, int& not_taken) {
1655   if (btest != BoolTest::eq && btest != BoolTest::ne) {
1656     // Only ::eq and ::ne are supported for profile injection.
1657     return false;
1658   }
1659   if (test->is_Cmp() &&
1660       test->in(1)->Opcode() == Op_ProfileBoolean) {
1661     ProfileBooleanNode* profile = (ProfileBooleanNode*)test->in(1);
1662     int false_cnt = profile->false_count();
1663     int  true_cnt = profile->true_count();
1664 
1665     // Counts matching depends on the actual test operation (::eq or ::ne).
1666     // No need to scale the counts because profile injection was designed
1667     // to feed exact counts into VM.
1668     taken     = (btest == BoolTest::eq) ? false_cnt :  true_cnt;
1669     not_taken = (btest == BoolTest::eq) ?  true_cnt : false_cnt;
1670 
1671     profile->consume();
1672     return true;
1673   }
1674   return false;
1675 }
1676 //--------------------------dynamic_branch_prediction--------------------------
1677 // Try to gather dynamic branch prediction behavior.  Return a probability
1678 // of the branch being taken and set the "cnt" field.  Returns a -1.0
1679 // if we need to use static prediction for some reason.
1680 float Parse::dynamic_branch_prediction(float &cnt, BoolTest::mask btest, Node* test) {
1681   ResourceMark rm;
1682 
1683   cnt  = COUNT_UNKNOWN;
1684 
1685   int     taken = 0;
1686   int not_taken = 0;
1687 
1688   bool use_mdo = !has_injected_profile(btest, test, taken, not_taken);
1689 
1690   if (use_mdo) {
1691     // Use MethodData information if it is available
1692     // FIXME: free the ProfileData structure
1693     ciMethodData* methodData = method()->method_data();
1694     if (!methodData->is_mature())  return PROB_UNKNOWN;
1695     ciProfileData* data = methodData->bci_to_data(bci());
1696     if (data == NULL) {
1697       return PROB_UNKNOWN;
1698     }
1699     if (!data->is_JumpData())  return PROB_UNKNOWN;
1700 
1701     // get taken and not taken values
1702     taken = data->as_JumpData()->taken();
1703     not_taken = 0;
1704     if (data->is_BranchData()) {
1705       not_taken = data->as_BranchData()->not_taken();
1706     }
1707 
1708     // scale the counts to be commensurate with invocation counts:
1709     taken = method()->scale_count(taken);
1710     not_taken = method()->scale_count(not_taken);
1711   }
1712 
1713   // Give up if too few (or too many, in which case the sum will overflow) counts to be meaningful.
1714   // We also check that individual counters are positive first, otherwise the sum can become positive.
1715   if (taken < 0 || not_taken < 0 || taken + not_taken < 40) {
1716     if (C->log() != NULL) {
1717       C->log()->elem("branch target_bci='%d' taken='%d' not_taken='%d'", iter().get_dest(), taken, not_taken);
1718     }
1719     return PROB_UNKNOWN;
1720   }
1721 
1722   // Compute frequency that we arrive here
1723   float sum = taken + not_taken;
1724   // Adjust, if this block is a cloned private block but the
1725   // Jump counts are shared.  Taken the private counts for
1726   // just this path instead of the shared counts.
1727   if( block()->count() > 0 )
1728     sum = block()->count();
1729   cnt = sum / FreqCountInvocations;
1730 
1731   // Pin probability to sane limits
1732   float prob;
1733   if( !taken )
1734     prob = (0+PROB_MIN) / 2;
1735   else if( !not_taken )
1736     prob = (1+PROB_MAX) / 2;
1737   else {                         // Compute probability of true path
1738     prob = (float)taken / (float)(taken + not_taken);
1739     if (prob > PROB_MAX)  prob = PROB_MAX;
1740     if (prob < PROB_MIN)   prob = PROB_MIN;
1741   }
1742 
1743   assert((cnt > 0.0f) && (prob > 0.0f),
1744          "Bad frequency assignment in if");
1745 
1746   if (C->log() != NULL) {
1747     const char* prob_str = NULL;
1748     if (prob >= PROB_MAX)  prob_str = (prob == PROB_MAX) ? "max" : "always";
1749     if (prob <= PROB_MIN)  prob_str = (prob == PROB_MIN) ? "min" : "never";
1750     char prob_str_buf[30];
1751     if (prob_str == NULL) {
1752       jio_snprintf(prob_str_buf, sizeof(prob_str_buf), "%20.2f", prob);
1753       prob_str = prob_str_buf;
1754     }
1755     C->log()->elem("branch target_bci='%d' taken='%d' not_taken='%d' cnt='%f' prob='%s'",
1756                    iter().get_dest(), taken, not_taken, cnt, prob_str);
1757   }
1758   return prob;
1759 }

1761 //-----------------------------branch_prediction-------------------------------
1762 float Parse::branch_prediction(float& cnt,
1763                                BoolTest::mask btest,
1764                                int target_bci,
1765                                Node* test) {
1766   float prob = dynamic_branch_prediction(cnt, btest, test);
1767   // If prob is unknown, switch to static prediction
1768   if (prob != PROB_UNKNOWN)  return prob;
1769 
1770   prob = PROB_FAIR;                   // Set default value
1771   if (btest == BoolTest::eq)          // Exactly equal test?
1772     prob = PROB_STATIC_INFREQUENT;    // Assume its relatively infrequent
1773   else if (btest == BoolTest::ne)
1774     prob = PROB_STATIC_FREQUENT;      // Assume its relatively frequent
1775 
1776   // If this is a conditional test guarding a backwards branch,
1777   // assume its a loop-back edge.  Make it a likely taken branch.
1778   if (target_bci < bci()) {
1779     if (is_osr_parse()) {    // Could be a hot OSR'd loop; force deopt
1780       // Since it's an OSR, we probably have profile data, but since
1781       // branch_prediction returned PROB_UNKNOWN, the counts are too small.
1782       // Let's make a special check here for completely zero counts.
1783       ciMethodData* methodData = method()->method_data();
1784       if (!methodData->is_empty()) {
1785         ciProfileData* data = methodData->bci_to_data(bci());
1786         // Only stop for truly zero counts, which mean an unknown part
1787         // of the OSR-ed method, and we want to deopt to gather more stats.
1788         // If you have ANY counts, then this loop is simply 'cold' relative
1789         // to the OSR loop.
1790         if (data == NULL ||
1791             (data->as_BranchData()->taken() +  data->as_BranchData()->not_taken() == 0)) {
1792           // This is the only way to return PROB_UNKNOWN:
1793           return PROB_UNKNOWN;
1794         }
1795       }
1796     }
1797     prob = PROB_STATIC_FREQUENT;     // Likely to take backwards branch
1798   }
1799 
1800   assert(prob != PROB_UNKNOWN, "must have some guess at this point");
1801   return prob;
1802 }
1803 
1804 // The magic constants are chosen so as to match the output of
1805 // branch_prediction() when the profile reports a zero taken count.
1806 // It is important to distinguish zero counts unambiguously, because
1807 // some branches (e.g., _213_javac.Assembler.eliminate) validly produce
1808 // very small but nonzero probabilities, which if confused with zero
1809 // counts would keep the program recompiling indefinitely.
1810 bool Parse::seems_never_taken(float prob) const {
1811   return prob < PROB_MIN;
1812 }
1813 
1814 // True if the comparison seems to be the kind that will not change its
1815 // statistics from true to false.  See comments in adjust_map_after_if.
1816 // This question is only asked along paths which are already
1817 // classifed as untaken (by seems_never_taken), so really,
1818 // if a path is never taken, its controlling comparison is
1819 // already acting in a stable fashion.  If the comparison
1820 // seems stable, we will put an expensive uncommon trap
1821 // on the untaken path.
1822 bool Parse::seems_stable_comparison() const {
1823   if (C->too_many_traps(method(), bci(), Deoptimization::Reason_unstable_if)) {
1824     return false;
1825   }
1826   return true;
1827 }

1846 //----------------------------------do_ifnull----------------------------------
1847 void Parse::do_ifnull(BoolTest::mask btest, Node *c) {
1848   int target_bci = iter().get_dest();
1849 
1850   Block* branch_block = successor_for_bci(target_bci);
1851   Block* next_block   = successor_for_bci(iter().next_bci());
1852 
1853   float cnt;
1854   float prob = branch_prediction(cnt, btest, target_bci, c);
1855   if (prob == PROB_UNKNOWN) {
1856     // (An earlier version of do_ifnull omitted this trap for OSR methods.)
1857     if (PrintOpto && Verbose) {
1858       tty->print_cr("Never-taken edge stops compilation at bci %d", bci());
1859     }
1860     repush_if_args(); // to gather stats on loop
1861     // We need to mark this branch as taken so that if we recompile we will
1862     // see that it is possible. In the tiered system the interpreter doesn't
1863     // do profiling and by the time we get to the lower tier from the interpreter
1864     // the path may be cold again. Make sure it doesn't look untaken
1865     profile_taken_branch(target_bci, !ProfileInterpreter);
1866     uncommon_trap(Deoptimization::Reason_unreached,
1867                   Deoptimization::Action_reinterpret,
1868                   NULL, "cold");
1869     if (C->eliminate_boxing()) {
1870       // Mark the successor blocks as parsed
1871       branch_block->next_path_num();
1872       next_block->next_path_num();
1873     }
1874     return;
1875   }
1876 
1877   NOT_PRODUCT(explicit_null_checks_inserted++);
1878 
1879   // Generate real control flow
1880   Node   *tst = _gvn.transform( new BoolNode( c, btest ) );
1881 
1882   // Sanity check the probability value
1883   assert(prob > 0.0f,"Bad probability in Parser");
1884  // Need xform to put node in hash table
1885   IfNode *iff = create_and_xform_if( control(), tst, prob, cnt );
1886   assert(iff->_prob > 0.0f,"Optimizer made bad probability in parser");
1887   // True branch
1888   { PreserveJVMState pjvms(this);
1889     Node* iftrue  = _gvn.transform( new IfTrueNode (iff) );
1890     set_control(iftrue);
1891 
1892     if (stopped()) {            // Path is dead?
1893       NOT_PRODUCT(explicit_null_checks_elided++);
1894       if (C->eliminate_boxing()) {
1895         // Mark the successor block as parsed
1896         branch_block->next_path_num();
1897       }
1898     } else {                    // Path is live.
1899       // Update method data
1900       profile_taken_branch(target_bci);
1901       adjust_map_after_if(btest, c, prob, branch_block);
1902       if (!stopped()) {
1903         merge(target_bci);
1904       }
1905     }
1906   }
1907 
1908   // False branch
1909   Node* iffalse = _gvn.transform( new IfFalseNode(iff) );
1910   set_control(iffalse);
1911 
1912   if (stopped()) {              // Path is dead?
1913     NOT_PRODUCT(explicit_null_checks_elided++);
1914     if (C->eliminate_boxing()) {
1915       // Mark the successor block as parsed
1916       next_block->next_path_num();
1917     }
1918   } else  {                     // Path is live.
1919     // Update method data
1920     profile_not_taken_branch();
1921     adjust_map_after_if(BoolTest(btest).negate(), c, 1.0-prob, next_block);
1922   }
1923 }

1925 //------------------------------------do_if------------------------------------
1926 void Parse::do_if(BoolTest::mask btest, Node* c, bool new_path, Node** ctrl_taken) {
1927   int target_bci = iter().get_dest();
1928 
1929   Block* branch_block = successor_for_bci(target_bci);
1930   Block* next_block   = successor_for_bci(iter().next_bci());
1931 
1932   float cnt;
1933   float prob = branch_prediction(cnt, btest, target_bci, c);
1934   float untaken_prob = 1.0 - prob;
1935 
1936   if (prob == PROB_UNKNOWN) {
1937     if (PrintOpto && Verbose) {
1938       tty->print_cr("Never-taken edge stops compilation at bci %d", bci());
1939     }
1940     repush_if_args(); // to gather stats on loop
1941     // We need to mark this branch as taken so that if we recompile we will
1942     // see that it is possible. In the tiered system the interpreter doesn't
1943     // do profiling and by the time we get to the lower tier from the interpreter
1944     // the path may be cold again. Make sure it doesn't look untaken
1945     profile_taken_branch(target_bci, !ProfileInterpreter);
1946     uncommon_trap(Deoptimization::Reason_unreached,
1947                   Deoptimization::Action_reinterpret,
1948                   NULL, "cold");
1949     if (C->eliminate_boxing()) {
1950       // Mark the successor blocks as parsed
1951       branch_block->next_path_num();
1952       next_block->next_path_num();
1953     }
1954     return;
1955   }
1956 
1957   // Sanity check the probability value
1958   assert(0.0f < prob && prob < 1.0f,"Bad probability in Parser");
1959 
1960   bool taken_if_true = true;
1961   // Convert BoolTest to canonical form:
1962   if (!BoolTest(btest).is_canonical()) {
1963     btest         = BoolTest(btest).negate();
1964     taken_if_true = false;
1965     // prob is NOT updated here; it remains the probability of the taken
1966     // path (as opposed to the prob of the path guarded by an 'IfTrueNode').
1967   }
1968   assert(btest != BoolTest::eq, "!= is the only canonical exact test");
1969 
1970   Node* tst0 = new BoolNode(c, btest);
1971   Node* tst = _gvn.transform(tst0);
1972   BoolTest::mask taken_btest   = BoolTest::illegal;
1973   BoolTest::mask untaken_btest = BoolTest::illegal;
1974 
1975   if (tst->is_Bool()) {
1976     // Refresh c from the transformed bool node, since it may be
1977     // simpler than the original c.  Also re-canonicalize btest.
1978     // This wins when (Bool ne (Conv2B p) 0) => (Bool ne (CmpP p NULL)).
1979     // That can arise from statements like: if (x instanceof C) ...
1980     if (tst != tst0) {
1981       // Canonicalize one more time since transform can change it.
1982       btest = tst->as_Bool()->_test._test;
1983       if (!BoolTest(btest).is_canonical()) {
1984         // Reverse edges one more time...
1985         tst   = _gvn.transform( tst->as_Bool()->negate(&_gvn) );
1986         btest = tst->as_Bool()->_test._test;
1987         assert(BoolTest(btest).is_canonical(), "sanity");
1988         taken_if_true = !taken_if_true;
1989       }
1990       c = tst->in(1);
1991     }
1992     BoolTest::mask neg_btest = BoolTest(btest).negate();
1993     taken_btest   = taken_if_true ?     btest : neg_btest;
1994     untaken_btest = taken_if_true ? neg_btest :     btest;
1995   }
1996 
1997   // Generate real control flow
1998   float true_prob = (taken_if_true ? prob : untaken_prob);
1999   IfNode* iff = create_and_map_if(control(), tst, true_prob, cnt);
2000   assert(iff->_prob > 0.0f,"Optimizer made bad probability in parser");
2001   Node* taken_branch   = new IfTrueNode(iff);
2002   Node* untaken_branch = new IfFalseNode(iff);
2003   if (!taken_if_true) {  // Finish conversion to canonical form
2004     Node* tmp      = taken_branch;
2005     taken_branch   = untaken_branch;
2006     untaken_branch = tmp;
2007   }
2008 
2009   // Branch is taken:
2010   { PreserveJVMState pjvms(this);
2011     taken_branch = _gvn.transform(taken_branch);
2012     set_control(taken_branch);
2013 
2014     if (stopped()) {
2015       if (C->eliminate_boxing() && !new_path) {
2016         // Mark the successor block as parsed (if we haven't created a new path)
2017         branch_block->next_path_num();
2018       }
2019     } else {
2020       // Update method data
2021       profile_taken_branch(target_bci);
2022       adjust_map_after_if(taken_btest, c, prob, branch_block);
2023       if (!stopped()) {
2024         if (new_path) {
2025           // Merge by using a new path
2026           merge_new_path(target_bci);
2027         } else if (ctrl_taken != NULL) {
2028           // Don't merge but save taken branch to be wired by caller
2029           *ctrl_taken = control();
2030         } else {
2031           merge(target_bci);
2032         }
2033       }
2034     }
2035   }
2036 
2037   untaken_branch = _gvn.transform(untaken_branch);
2038   set_control(untaken_branch);
2039 
2040   // Branch not taken.
2041   if (stopped() && ctrl_taken == NULL) {
2042     if (C->eliminate_boxing()) {
2043       // Mark the successor block as parsed (if caller does not re-wire control flow)
2044       next_block->next_path_num();
2045     }
2046   } else {
2047     // Update method data
2048     profile_not_taken_branch();
2049     adjust_map_after_if(untaken_btest, c, untaken_prob, next_block);
2050   }
2051 }

2053 void Parse::do_acmp(BoolTest::mask btest, Node* a, Node* b) {
2054   ciMethod* subst_method = ciEnv::current()->ValueBootstrapMethods_klass()->find_method(ciSymbol::isSubstitutable_name(), ciSymbol::object_object_boolean_signature());
2055   // If current method is ValueBootstrapMethods::isSubstitutable(),
2056   // compile the acmp as a regular pointer comparison otherwise we
2057   // could call ValueBootstrapMethods::isSubstitutable() back
2058   if (!EnableValhalla || (method() == subst_method)) {
2059     Node* cmp = CmpP(a, b);
2060     cmp = optimize_cmp_with_klass(cmp);
2061     do_if(btest, cmp);
2062     return;
2063   }
2064 
2065   // Allocate value type operands and re-execute on deoptimization
2066   if (a->is_ValueType()) {
2067     PreserveReexecuteState preexecs(this);
2068     inc_sp(2);
2069     jvms()->set_should_reexecute(true);
2070     a = a->as_ValueType()->allocate(this)->get_oop();
2071   }
2072   if (b->is_ValueType()) {
2073     PreserveReexecuteState preexecs(this);
2074     inc_sp(2);
2075     jvms()->set_should_reexecute(true);
2076     b = b->as_ValueType()->allocate(this)->get_oop();
2077   }
2078 
2079   // First, do a normal pointer comparison
2080   const TypeOopPtr* ta = _gvn.type(a)->isa_oopptr();
2081   const TypeOopPtr* tb = _gvn.type(b)->isa_oopptr();
2082   Node* cmp = CmpP(a, b);
2083   cmp = optimize_cmp_with_klass(cmp);
2084   if (ta == NULL || !ta->can_be_value_type() ||
2085       tb == NULL || !tb->can_be_value_type()) {
2086     // This is sufficient, if one of the operands can't be a value type
2087     do_if(btest, cmp);
2088     return;
2089   }
2090   Node* eq_region = NULL;
2091   if (btest == BoolTest::eq) {
2092     do_if(btest, cmp, true);
2093     if (stopped()) {
2094       return;
2095     }
2096   } else {
2097     assert(btest == BoolTest::ne, "only eq or ne");
2098     Node* is_not_equal = NULL;
2099     eq_region = new RegionNode(3);
2100     {
2101       PreserveJVMState pjvms(this);
2102       do_if(btest, cmp, false, &is_not_equal);
2103       if (!stopped()) {
2104         eq_region->init_req(1, control());
2105       }
2106     }
2107     if (is_not_equal == NULL || is_not_equal->is_top()) {
2108       record_for_igvn(eq_region);
2109       set_control(_gvn.transform(eq_region));
2110       return;
2111     }
2112     set_control(is_not_equal);
2113   }
2114 
2115   // Pointers are not equal, check if first operand is non-null
2116   Node* ne_region = new RegionNode(6);
2117   inc_sp(2);
2118   Node* null_ctl = top();
2119   Node* not_null_a = null_check_oop(a, &null_ctl, !too_many_traps(Deoptimization::Reason_null_check), false, false);
2120   dec_sp(2);
2121   ne_region->init_req(1, null_ctl);
2122   if (stopped()) {
2123     record_for_igvn(ne_region);
2124     set_control(_gvn.transform(ne_region));
2125     if (btest == BoolTest::ne) {
2126       {
2127         PreserveJVMState pjvms(this);
2128         int target_bci = iter().get_dest();
2129         merge(target_bci);
2130       }
2131       record_for_igvn(eq_region);
2132       set_control(_gvn.transform(eq_region));
2133     }
2134     return;
2135   }
2136 
2137   // First operand is non-null, check if it is a value type
2138   Node* is_value = is_value_type(not_null_a);
2139   IfNode* is_value_iff = create_and_map_if(control(), is_value, PROB_FAIR, COUNT_UNKNOWN);
2140   Node* not_value = _gvn.transform(new IfFalseNode(is_value_iff));
2141   ne_region->init_req(2, not_value);
2142   set_control(_gvn.transform(new IfTrueNode(is_value_iff)));
2143 
2144   // The first operand is a value type, check if the second operand is non-null
2145   inc_sp(2);
2146   null_ctl = top();
2147   Node* not_null_b = null_check_oop(b, &null_ctl, !too_many_traps(Deoptimization::Reason_null_check), false, false);

2278 //----------------------------adjust_map_after_if------------------------------
2279 // Adjust the JVM state to reflect the result of taking this path.
2280 // Basically, it means inspecting the CmpNode controlling this
2281 // branch, seeing how it constrains a tested value, and then
2282 // deciding if it's worth our while to encode this constraint
2283 // as graph nodes in the current abstract interpretation map.
2284 void Parse::adjust_map_after_if(BoolTest::mask btest, Node* c, float prob, Block* path) {
2285   if (!c->is_Cmp()) {
2286     maybe_add_predicate_after_if(path);
2287     return;
2288   }
2289 
2290   if (stopped() || btest == BoolTest::illegal) {
2291     return;                             // nothing to do
2292   }
2293 
2294   bool is_fallthrough = (path == successor_for_bci(iter().next_bci()));
2295 
2296   if (path_is_suitable_for_uncommon_trap(prob)) {
2297     repush_if_args();
2298     uncommon_trap(Deoptimization::Reason_unstable_if,
2299                   Deoptimization::Action_reinterpret,
2300                   NULL,
2301                   (is_fallthrough ? "taken always" : "taken never"));
2302     return;
2303   }
2304 
2305   Node* val = c->in(1);
2306   Node* con = c->in(2);
2307   const Type* tcon = _gvn.type(con);
2308   const Type* tval = _gvn.type(val);
2309   bool have_con = tcon->singleton();
2310   if (tval->singleton()) {
2311     if (!have_con) {
2312       // Swap, so constant is in con.
2313       con  = val;
2314       tcon = tval;
2315       val  = c->in(2);
2316       tval = _gvn.type(val);
2317       btest = BoolTest(btest).commute();
2318       have_con = true;
2319     } else {
2320       // Do we have two constants?  Then leave well enough alone.
2321       have_con = false;
2322     }
2323   }
2324   if (!have_con) {                        // remaining adjustments need a con
2325     maybe_add_predicate_after_if(path);
2326     return;
2327   }
2328 
2329   sharpen_type_after_if(btest, con, tcon, val, tval);
2330   maybe_add_predicate_after_if(path);
2331 }

2361 void Parse::sharpen_type_after_if(BoolTest::mask btest,
2362                                   Node* con, const Type* tcon,
2363                                   Node* val, const Type* tval) {
2364   // Look for opportunities to sharpen the type of a node
2365   // whose klass is compared with a constant klass.
2366   if (btest == BoolTest::eq && tcon->isa_klassptr()) {
2367     Node* obj = extract_obj_from_klass_load(&_gvn, val);
2368     const TypeOopPtr* con_type = tcon->isa_klassptr()->as_instance_type();
2369     if (obj != NULL && (con_type->isa_instptr() || con_type->isa_aryptr())) {
2370        // Found:
2371        //   Bool(CmpP(LoadKlass(obj._klass), ConP(Foo.klass)), [eq])
2372        // or the narrowOop equivalent.
2373        const Type* obj_type = _gvn.type(obj);
2374        const TypeOopPtr* tboth = obj_type->join_speculative(con_type)->isa_oopptr();
2375        if (tboth != NULL && tboth->klass_is_exact() && tboth != obj_type &&
2376            tboth->higher_equal(obj_type)) {
2377           // obj has to be of the exact type Foo if the CmpP succeeds.
2378           int obj_in_map = map()->find_edge(obj);
2379           JVMState* jvms = this->jvms();
2380           if (obj_in_map >= 0 &&
2381               (jvms->is_loc(obj_in_map) || jvms->is_stk(obj_in_map))) {
2382             TypeNode* ccast = new CheckCastPPNode(control(), obj, tboth);
2383             const Type* tcc = ccast->as_Type()->type();
2384             assert(tcc != obj_type && tcc->higher_equal(obj_type), "must improve");
2385             // Delay transform() call to allow recovery of pre-cast value
2386             // at the control merge.
2387             _gvn.set_type_bottom(ccast);
2388             record_for_igvn(ccast);
2389             // Here's the payoff.
2390             replace_in_map(obj, ccast);
2391           }
2392        }
2393     }
2394   }
2395 
2396   int val_in_map = map()->find_edge(val);
2397   if (val_in_map < 0)  return;          // replace_in_map would be useless
2398   {
2399     JVMState* jvms = this->jvms();
2400     if (!(jvms->is_loc(val_in_map) ||
2401           jvms->is_stk(val_in_map)))
2402       return;                           // again, it would be useless
2403   }
2404 
2405   // Check for a comparison to a constant, and "know" that the compared
2406   // value is constrained on this path.
2407   assert(tcon->singleton(), "");
2408   ConstraintCastNode* ccast = NULL;
2409   Node* cast = NULL;
2410 
2411   switch (btest) {
2412   case BoolTest::eq:                    // Constant test?
2413     {
2414       const Type* tboth = tcon->join_speculative(tval);
2415       if (tboth == tval)  break;        // Nothing to gain.
2416       if (tcon->isa_int()) {
2417         ccast = new CastIINode(val, tboth);
2418       } else if (tcon == TypePtr::NULL_PTR) {
2419         // Cast to null, but keep the pointer identity temporarily live.
2420         ccast = new CastPPNode(val, tboth);
2421       } else {
2422         const TypeF* tf = tcon->isa_float_constant();
2423         const TypeD* td = tcon->isa_double_constant();
2424         // Exclude tests vs float/double 0 as these could be
2425         // either +0 or -0.  Just because you are equal to +0
2426         // doesn't mean you ARE +0!
2427         // Note, following code also replaces Long and Oop values.
2428         if ((!tf || tf->_f != 0.0) &&
2429             (!td || td->_d != 0.0))
2430           cast = con;                   // Replace non-constant val by con.
2431       }
2432     }
2433     break;
2434 
2435   case BoolTest::ne:
2436     if (tcon == TypePtr::NULL_PTR) {
2437       cast = cast_not_null(val, false);
2438     }
2439     break;
2440 
2441   default:
2442     // (At this point we could record int range types with CastII.)
2443     break;
2444   }
2445 
2446   if (ccast != NULL) {
2447     const Type* tcc = ccast->as_Type()->type();
2448     assert(tcc != tval && tcc->higher_equal(tval), "must improve");
2449     // Delay transform() call to allow recovery of pre-cast value
2450     // at the control merge.
2451     ccast->set_req(0, control());
2452     _gvn.set_type_bottom(ccast);
2453     record_for_igvn(ccast);
2454     cast = ccast;
2455   }
2456 
2457   if (cast != NULL) {                   // Here's the payoff.
2458     replace_in_map(val, cast);
2459   }
2460 }

2462 /**

2464  * against the low level class, cast the object to the speculative
2465  * type if any. CmpP should then go away.

2468  * @return   result of CmpP on object casted to speculative type
2469  *
2470  */
2471 Node* Parse::optimize_cmp_with_klass(Node* c) {
2472   // If this is transformed by the _gvn to a comparison with the low
2473   // level klass then we may be able to use speculation
2474   if (c->Opcode() == Op_CmpP &&
2475       (c->in(1)->Opcode() == Op_LoadKlass || c->in(1)->Opcode() == Op_DecodeNKlass) &&
2476       c->in(2)->is_Con()) {
2477     Node* load_klass = NULL;
2478     Node* decode = NULL;
2479     if (c->in(1)->Opcode() == Op_DecodeNKlass) {
2480       decode = c->in(1);
2481       load_klass = c->in(1)->in(1);
2482     } else {
2483       load_klass = c->in(1);
2484     }
2485     if (load_klass->in(2)->is_AddP()) {
2486       Node* addp = load_klass->in(2);
2487       Node* obj = addp->in(AddPNode::Address);
2488       const TypeOopPtr* obj_type = _gvn.type(obj)->is_oopptr();
2489       if (obj_type->speculative_type_not_null() != NULL) {
2490         ciKlass* k = obj_type->speculative_type();

2492         obj = maybe_cast_profiled_obj(obj, k);
2493         dec_sp(2);

2496           obj = obj->as_ValueType()->get_oop();

2504           decode = decode->clone();
2505           decode->set_req(1, load_klass);
2506           load_klass = _gvn.transform(decode);
2507         }
2508         c = c->clone();
2509         c->set_req(1, load_klass);
2510         c = _gvn.transform(c);
2511       }
2512     }
2513   }
