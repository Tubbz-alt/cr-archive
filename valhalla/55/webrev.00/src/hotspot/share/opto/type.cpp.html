  55 const Type::Offset Type::Offset::meet(const Type::Offset other) const {
  56   // Either is 'TOP' offset?  Return the other offset!
  57   int offset = other._offset;
  58   if (_offset == OffsetTop) return Offset(offset);
  59   if (offset == OffsetTop) return Offset(_offset);
  60   // If either is different, return 'BOTTOM' offset
  61   if (_offset != offset) return bottom;
  62   return Offset(_offset);
  63 }

  65 const Type::Offset Type::Offset::dual() const {
  66   if (_offset == OffsetTop) return bottom;// Map 'TOP' into 'BOTTOM'
  67   if (_offset == OffsetBot) return top;// Map 'BOTTOM' into 'TOP'
  68   return Offset(_offset);               // Map everything else into self
  69 }

  71 const Type::Offset Type::Offset::add(intptr_t offset) const {
  72   // Adding to 'TOP' offset?  Return 'TOP'!
  73   if (_offset == OffsetTop || offset == OffsetTop) return top;
  74   // Adding to 'BOTTOM' offset?  Return 'BOTTOM'!
  75   if (_offset == OffsetBot || offset == OffsetBot) return bottom;
  76   // Addition overflows or "accidentally" equals to OffsetTop? Return 'BOTTOM'!
  77   offset += (intptr_t)_offset;
  78   if (offset != (int)offset || offset == OffsetTop) return bottom;
  79 
  80   // assert( _offset >= 0 && _offset+offset >= 0, "" );
  81   // It is possible to construct a negative offset during PhaseCCP
  82 
  83   return Offset((int)offset);        // Sum valid offsets
  84 }

  87   if (_offset == 0) {
  88     return;
  89   } else if (_offset == OffsetTop) {

  91   }
  92   else if (_offset == OffsetBot) {
  93     st->print("+bot");
  94   } else if (_offset) {

 182 //------------------------------get_const_type---------------------------
 183 const Type* Type::get_const_type(ciType* type) {
 184   if (type == NULL) {
 185     return NULL;
 186   } else if (type->is_primitive_type()) {
 187     return get_const_basic_type(type->basic_type());
 188   } else {
 189     return TypeOopPtr::make_from_klass(type->as_klass());
 190   }

 193 //---------------------------array_element_basic_type---------------------------------
 194 // Mapping to the array element's basic type.
 195 BasicType Type::array_element_basic_type() const {
 196   BasicType bt = basic_type();
 197   if (bt == T_INT) {
 198     if (this == TypeInt::INT)   return T_INT;
 199     if (this == TypeInt::CHAR)  return T_CHAR;
 200     if (this == TypeInt::BYTE)  return T_BYTE;
 201     if (this == TypeInt::BOOL)  return T_BOOLEAN;
 202     if (this == TypeInt::SHORT) return T_SHORT;
 203     return T_VOID;
 204   }
 205   return bt;
 206 }

 208 // For two instance arrays of same dimension, return the base element types.
 209 // Otherwise or if the arrays have different dimensions, return NULL.
 210 void Type::get_arrays_base_elements(const Type *a1, const Type *a2,
 211                                     const TypeInstPtr **e1, const TypeInstPtr **e2) {
 212 
 213   if (e1) *e1 = NULL;
 214   if (e2) *e2 = NULL;
 215   const TypeAryPtr* a1tap = (a1 == NULL) ? NULL : a1->isa_aryptr();
 216   const TypeAryPtr* a2tap = (a2 == NULL) ? NULL : a2->isa_aryptr();
 217 
 218   if (a1tap != NULL && a2tap != NULL) {
 219     // Handle multidimensional arrays
 220     const TypePtr* a1tp = a1tap->elem()->make_ptr();
 221     const TypePtr* a2tp = a2tap->elem()->make_ptr();
 222     while (a1tp && a1tp->isa_aryptr() && a2tp && a2tp->isa_aryptr()) {
 223       a1tap = a1tp->is_aryptr();
 224       a2tap = a2tp->is_aryptr();
 225       a1tp = a1tap->elem()->make_ptr();
 226       a2tp = a2tap->elem()->make_ptr();
 227     }
 228     if (a1tp && a1tp->isa_instptr() && a2tp && a2tp->isa_instptr()) {
 229       if (e1) *e1 = a1tp->is_instptr();
 230       if (e2) *e2 = a2tp->is_instptr();
 231     }
 232   }
 233 }

 235 //---------------------------get_typeflow_type---------------------------------
 236 // Import a type produced by ciTypeFlow.
 237 const Type* Type::get_typeflow_type(ciType* type) {
 238   switch (type->basic_type()) {
 239 
 240   case ciTypeFlow::StateVector::T_BOTTOM:
 241     assert(type == ciTypeFlow::StateVector::bottom_type(), "");
 242     return Type::BOTTOM;
 243 
 244   case ciTypeFlow::StateVector::T_TOP:
 245     assert(type == ciTypeFlow::StateVector::top_type(), "");
 246     return Type::TOP;
 247 
 248   case ciTypeFlow::StateVector::T_NULL:
 249     assert(type == ciTypeFlow::StateVector::null_type(), "");
 250     return TypePtr::NULL_PTR;
 251 
 252   case ciTypeFlow::StateVector::T_LONG2:
 253     // The ciTypeFlow pass pushes a long, then the half.
 254     // We do the same.
 255     assert(type == ciTypeFlow::StateVector::long2_type(), "");
 256     return TypeInt::TOP;
 257 
 258   case ciTypeFlow::StateVector::T_DOUBLE2:
 259     // The ciTypeFlow pass pushes double, then the half.
 260     // Our convention is the same.
 261     assert(type == ciTypeFlow::StateVector::double2_type(), "");
 262     return Type::TOP;
 263 
 264   case T_ADDRESS:
 265     assert(type->is_return_address(), "");
 266     return TypeRawPtr::make((address)(intptr_t)type->as_return_address()->bci());
 267 
 268   case T_VALUETYPE: {
 269     bool is_never_null = type->is_never_null();
 270     ciValueKlass* vk = type->unwrap()->as_value_klass();
 271     if (vk->is_scalarizable() && is_never_null) {
 272       return TypeValueType::make(vk);
 273     } else {
 274       return TypeOopPtr::make_from_klass(vk)->join_speculative(is_never_null ? TypePtr::NOTNULL : TypePtr::BOTTOM);
 275     }
 276   }
 277 
 278   default:
 279     // make sure we did not mix up the cases:
 280     assert(type != ciTypeFlow::StateVector::bottom_type(), "");
 281     assert(type != ciTypeFlow::StateVector::top_type(), "");
 282     assert(type != ciTypeFlow::StateVector::null_type(), "");
 283     assert(type != ciTypeFlow::StateVector::long2_type(), "");
 284     assert(type != ciTypeFlow::StateVector::double2_type(), "");
 285     assert(!type->is_return_address(), "");
 286 
 287     return Type::get_const_type(type);
 288   }
 289 }

 292 //-----------------------make_from_constant------------------------------------
 293 const Type* Type::make_from_constant(ciConstant constant, bool require_constant,
 294                                      int stable_dimension, bool is_narrow_oop,
 295                                      bool is_autobox_cache) {
 296   switch (constant.basic_type()) {
 297     case T_BOOLEAN:  return TypeInt::make(constant.as_boolean());
 298     case T_CHAR:     return TypeInt::make(constant.as_char());
 299     case T_BYTE:     return TypeInt::make(constant.as_byte());
 300     case T_SHORT:    return TypeInt::make(constant.as_short());
 301     case T_INT:      return TypeInt::make(constant.as_int());
 302     case T_LONG:     return TypeLong::make(constant.as_long());
 303     case T_FLOAT:    return TypeF::make(constant.as_float());
 304     case T_DOUBLE:   return TypeD::make(constant.as_double());
 305     case T_ARRAY:
 306     case T_VALUETYPE:
 307     case T_OBJECT: {
 308         const Type* con_type = NULL;
 309         ciObject* oop_constant = constant.as_object();
 310         if (oop_constant->is_null_object()) {
 311           con_type = Type::get_zero_type(T_OBJECT);
 312         } else {
 313           guarantee(require_constant || oop_constant->should_be_constant(), "con_type must get computed");
 314           con_type = TypeOopPtr::make_from_constant(oop_constant, require_constant);
 315           if (Compile::current()->eliminate_boxing() && is_autobox_cache) {
 316             con_type = con_type->is_aryptr()->cast_to_autobox_cache(true);
 317           }
 318           if (stable_dimension > 0) {
 319             assert(FoldStableValues, "sanity");
 320             assert(!con_type->is_zero_type(), "default value for stable field");
 321             con_type = con_type->is_aryptr()->cast_to_stable(true, stable_dimension);
 322           }
 323         }
 324         if (is_narrow_oop) {
 325           con_type = con_type->make_narrowoop();
 326         }
 327         return con_type;
 328       }
 329     case T_ILLEGAL:
 330       // Invalid ciConstant returned due to OutOfMemoryError in the CI
 331       assert(Compile::current()->env()->failing(), "otherwise should not see this");
 332       return NULL;
 333     default:
 334       // Fall through to failure
 335       return NULL;
 336   }
 337 }

 339 static ciConstant check_mismatched_access(ciConstant con, BasicType loadbt, bool is_unsigned) {
 340   BasicType conbt = con.basic_type();
 341   switch (conbt) {
 342     case T_BOOLEAN: conbt = T_BYTE;   break;
 343     case T_ARRAY:   conbt = T_OBJECT; break;
 344     case T_VALUETYPE: conbt = T_OBJECT; break;
 345     default:                          break;
 346   }
 347   switch (loadbt) {
 348     case T_BOOLEAN:   loadbt = T_BYTE;   break;
 349     case T_NARROWOOP: loadbt = T_OBJECT; break;
 350     case T_ARRAY:     loadbt = T_OBJECT; break;
 351     case T_VALUETYPE: loadbt = T_OBJECT; break;
 352     case T_ADDRESS:   loadbt = T_OBJECT; break;
 353     default:                             break;
 354   }
 355   if (conbt == loadbt) {
 356     if (is_unsigned && conbt == T_BYTE) {
 357       // LoadB (T_BYTE) with a small mask (<=8-bit) is converted to LoadUB (T_BYTE).
 358       return ciConstant(T_INT, con.as_int() & 0xFF);
 359     } else {
 360       return con;
 361     }
 362   }
 363   if (conbt == T_SHORT && loadbt == T_CHAR) {
 364     // LoadS (T_SHORT) with a small mask (<=16-bit) is converted to LoadUS (T_CHAR).
 365     return ciConstant(T_INT, con.as_int() & 0xFFFF);
 366   }
 367   return ciConstant(); // T_ILLEGAL
 368 }

 370 // Try to constant-fold a stable array element.
 371 const Type* Type::make_constant_from_array_element(ciArray* array, int off, int stable_dimension,
 372                                                    BasicType loadbt, bool is_unsigned_load) {
 373   // Decode the results of GraphKit::array_element_address.
 374   ciConstant element_value = array->element_value_by_offset(off);
 375   if (element_value.basic_type() == T_ILLEGAL) {
 376     return NULL; // wrong offset
 377   }
 378   ciConstant con = check_mismatched_access(element_value, loadbt, is_unsigned_load);
 379 
 380   assert(con.basic_type() != T_ILLEGAL, "elembt=%s; loadbt=%s; unsigned=%d",
 381          type2name(element_value.basic_type()), type2name(loadbt), is_unsigned_load);
 382 
 383   if (con.is_valid() &&          // not a mismatched access
 384       !con.is_null_or_zero()) {  // not a default value
 385     bool is_narrow_oop = (loadbt == T_NARROWOOP);
 386     return Type::make_from_constant(con, /*require_constant=*/true, stable_dimension, is_narrow_oop, /*is_autobox_cache=*/false);
 387   }
 388   return NULL;
 389 }

 391 const Type* Type::make_constant_from_field(ciInstance* holder, int off, bool is_unsigned_load, BasicType loadbt) {
 392   ciField* field;
 393   ciType* type = holder->java_mirror_type();
 394   if (type != NULL && type->is_instance_klass() && off >= InstanceMirrorKlass::offset_of_static_fields()) {
 395     // Static field
 396     field = type->as_instance_klass()->get_field_by_offset(off, /*is_static=*/true);
 397   } else {
 398     // Instance field
 399     field = holder->klass()->as_instance_klass()->get_field_by_offset(off, /*is_static=*/false);
 400   }
 401   if (field == NULL) {
 402     return NULL; // Wrong offset
 403   }
 404   return Type::make_constant_from_field(field, holder, loadbt, is_unsigned_load);
 405 }

 407 const Type* Type::make_constant_from_field(ciField* field, ciInstance* holder,
 408                                            BasicType loadbt, bool is_unsigned_load) {
 409   if (!field->is_constant()) {
 410     return NULL; // Non-constant field
 411   }
 412   ciConstant field_value;
 413   if (field->is_static()) {
 414     // final static field
 415     field_value = field->constant_value();
 416   } else if (holder != NULL) {
 417     // final or stable non-static field
 418     // Treat final non-static fields of trusted classes (classes in
 419     // java.lang.invoke and sun.invoke packages and subpackages) as
 420     // compile time constants.
 421     field_value = field->constant_value_of(holder);
 422   }
 423   if (!field_value.is_valid()) {
 424     return NULL; // Not a constant
 425   }
 426 
 427   ciConstant con = check_mismatched_access(field_value, loadbt, is_unsigned_load);
 428 
 429   assert(con.is_valid(), "elembt=%s; loadbt=%s; unsigned=%d",
 430          type2name(field_value.basic_type()), type2name(loadbt), is_unsigned_load);
 431 
 432   bool is_stable_array = FoldStableValues && field->is_stable() && field->type()->is_array_klass();
 433   int stable_dimension = (is_stable_array ? field->type()->as_array_klass()->dimension() : 0);
 434   bool is_narrow_oop = (loadbt == T_NARROWOOP);
 435 
 436   const Type* con_type = make_from_constant(con, /*require_constant=*/ true,
 437                                             stable_dimension, is_narrow_oop,
 438                                             field->is_autobox_cache());
 439   if (con_type != NULL && field->is_call_site_target()) {
 440     ciCallSite* call_site = holder->as_call_site();
 441     if (!call_site->is_fully_initialized_constant_call_site()) {
 442       ciMethodHandle* target = con.as_object()->as_method_handle();
 443       Compile::current()->dependencies()->assert_call_site_target_value(call_site, target);
 444     }
 445   }
 446   return con_type;
 447 }

 480 //--------------------------Initialize_shared----------------------------------
 481 void Type::Initialize_shared(Compile* current) {
 482   // This method does not need to be locked because the first system
 483   // compilations (stub compilations) occur serially.  If they are
 484   // changed to proceed in parallel, then this section will need
 485   // locking.
 486 
 487   Arena* save = current->type_arena();
 488   Arena* shared_type_arena = new (mtCompiler)Arena(mtCompiler);
 489

 772 //------------------------------hashcons---------------------------------------
 773 // Do the hash-cons trick.  If the Type already exists in the type table,
 774 // delete the current Type and return the existing Type.  Otherwise stick the
 775 // current Type in the Type table.
 776 const Type *Type::hashcons(void) {
 777   debug_only(base());           // Check the assertion in Type::base().
 778   // Look up the Type in the Type dictionary
 779   Dict *tdic = type_dict();
 780   Type* old = (Type*)(tdic->Insert(this, this, false));
 781   if( old ) {                   // Pre-existing Type?
 782     if( old != this )           // Yes, this guy is not the pre-existing?
 783       delete this;              // Yes, Nuke this guy
 784     assert( old->_dual, "" );
 785     return old;                 // Return pre-existing
 786   }
 787 
 788   // Every type has a dual (to make my lattice symmetric).
 789   // Since we just discovered a new Type, compute its dual right now.
 790   assert( !_dual, "" );         // No dual yet
 791   _dual = xdual();              // Compute the dual
 792   if (cmp(this, _dual) == 0) {  // Handle self-symmetric
 793     if (_dual != this) {
 794       delete _dual;
 795       _dual = this;
 796     }
 797     return this;
 798   }
 799   assert( !_dual->_dual, "" );  // No reverse dual yet
 800   assert( !(*tdic)[_dual], "" ); // Dual not in type system either
 801   // New Type, insert into Type table
 802   tdic->Insert((void*)_dual,(void*)_dual);
 803   ((Type*)_dual)->_dual = this; // Finish up being symmetric
 804 #ifdef ASSERT
 805   Type *dual_dual = (Type*)_dual->xdual();
 806   assert( eq(dual_dual), "xdual(xdual()) should be identity" );
 807   delete dual_dual;
 808 #endif
 809   return this;                  // Return new Type
 810 }

 911 // Compute the MEET of two types.  NOT virtual.  It enforces that meet is
 912 // commutative and the lattice is symmetric.
 913 const Type *Type::meet_helper(const Type *t, bool include_speculative) const {
 914   if (isa_narrowoop() && t->isa_narrowoop()) {
 915     const Type* result = make_ptr()->meet_helper(t->make_ptr(), include_speculative);
 916     return result->make_narrowoop();
 917   }
 918   if (isa_narrowklass() && t->isa_narrowklass()) {
 919     const Type* result = make_ptr()->meet_helper(t->make_ptr(), include_speculative);
 920     return result->make_narrowklass();
 921   }
 922 
 923   const Type *this_t = maybe_remove_speculative(include_speculative);
 924   t = t->maybe_remove_speculative(include_speculative);
 925 
 926   const Type *mt = this_t->xmeet(t);
 927 #ifdef ASSERT
 928   if (isa_narrowoop() || t->isa_narrowoop()) return mt;
 929   if (isa_narrowklass() || t->isa_narrowklass()) return mt;
 930   Compile* C = Compile::current();
 931   if (!C->_type_verify_symmetry) {
 932     return mt;
 933   }
 934   this_t->check_symmetrical(t, mt);
 935   // In the case of an array, computing the meet above, caused the
 936   // computation of the meet of the elements which at verification
 937   // time caused the computation of the meet of the dual of the
 938   // elements. Computing the meet of the dual of the arrays here
 939   // causes the meet of the dual of the elements to be computed which
 940   // would cause the meet of the dual of the dual of the elements,
 941   // that is the meet of the elements already computed above to be
 942   // computed. Avoid redundant computations by requesting no
 943   // verification.
 944   C->_type_verify_symmetry = false;
 945   const Type *mt_dual = this_t->_dual->xmeet(t->_dual);
 946   this_t->_dual->check_symmetrical(t->_dual, mt_dual);
 947   assert(!C->_type_verify_symmetry, "shouldn't have changed");
 948   C->_type_verify_symmetry = true;
 949 #endif
 950   return mt;
 951 }

 953 //------------------------------xmeet------------------------------------------
 954 // Compute the MEET of two types.  It returns a new Type object.
 955 const Type *Type::xmeet( const Type *t ) const {
 956   // Perform a fast test for common case; meeting the same types together.
 957   if( this == t ) return this;  // Meeting same type-rep?
 958 
 959   // Meeting TOP with anything?
 960   if( _base == Top ) return t;
 961 
 962   // Meeting BOTTOM with anything?
 963   if( _base == Bottom ) return BOTTOM;
 964 
 965   // Current "this->_base" is one of: Bad, Multi, Control, Top,
 966   // Abio, Abstore, Floatxxx, Doublexxx, Bottom, lastype.
 967   switch (t->base()) {  // Switch on original type
 968 
 969   // Cut in half the number of cases I must handle.  Only need cases for when
 970   // the given enum "t->type" is less than or equal to the local enum "type".
 971   case FloatCon:
 972   case DoubleCon:
 973   case Int:
 974   case Long:
 975     return t->xmeet(this);
 976 
 977   case OopPtr:
 978     return t->xmeet(this);
 979 
 980   case InstPtr:
 981     return t->xmeet(this);
 982 
 983   case MetadataPtr:
 984   case KlassPtr:
 985     return t->xmeet(this);
 986 
 987   case AryPtr:
 988     return t->xmeet(this);
 989 
 990   case NarrowOop:
 991     return t->xmeet(this);
 992 
 993   case NarrowKlass:
 994     return t->xmeet(this);
 995 
 996   case ValueType:
 997     return t->xmeet(this);
 998 
 999   case Bad:                     // Type check
1000   default:                      // Bogus type not in lattice
1001     typerr(t);
1002     return Type::BOTTOM;
1003 
1004   case Bottom:                  // Ye Olde Default
1005     return t;
1006 
1007   case FloatTop:
1008     if( _base == FloatTop ) return this;
1009   case FloatBot:                // Float
1010     if( _base == FloatBot || _base == FloatTop ) return FLOAT;
1011     if( _base == DoubleTop || _base == DoubleBot ) return Type::BOTTOM;
1012     typerr(t);
1013     return Type::BOTTOM;
1014 
1015   case DoubleTop:
1016     if( _base == DoubleTop ) return this;
1017   case DoubleBot:               // Double
1018     if( _base == DoubleBot || _base == DoubleTop ) return DOUBLE;
1019     if( _base == FloatTop || _base == FloatBot ) return Type::BOTTOM;
1020     typerr(t);
1021     return Type::BOTTOM;
1022 
1023   // These next few cases must match exactly or it is a compile-time error.
1024   case Control:                 // Control of code
1025   case Abio:                    // State of world outside of program
1026   case Memory:
1027     if( _base == t->_base )  return this;
1028     typerr(t);
1029     return Type::BOTTOM;
1030 
1031   case Top:                     // Top of the lattice
1032     return this;
1033   }
1034 
1035   // The type is unchanged
1036   return this;
1037 }

1501 //------------------------------widen------------------------------------------
1502 // Only happens for optimistic top-down optimizations.
1503 const Type *TypeInt::widen( const Type *old, const Type* limit ) const {
1504   // Coming from TOP or such; no widening
1505   if( old->base() != Int ) return this;
1506   const TypeInt *ot = old->is_int();
1507 
1508   // If new guy is equal to old guy, no widening
1509   if( _lo == ot->_lo && _hi == ot->_hi )
1510     return old;
1511 
1512   // If new guy contains old, then we widened
1513   if( _lo <= ot->_lo && _hi >= ot->_hi ) {
1514     // New contains old
1515     // If new guy is already wider than old, no widening
1516     if( _widen > ot->_widen ) return this;
1517     // If old guy was a constant, do not bother
1518     if (ot->_lo == ot->_hi)  return this;
1519     // Now widen new guy.
1520     // Check for widening too far
1521     if (_widen == WidenMax) {
1522       int max = max_jint;
1523       int min = min_jint;
1524       if (limit->isa_int()) {
1525         max = limit->is_int()->_hi;
1526         min = limit->is_int()->_lo;
1527       }
1528       if (min < _lo && _hi < max) {
1529         // If neither endpoint is extremal yet, push out the endpoint
1530         // which is closer to its respective limit.
1531         if (_lo >= 0 ||                 // easy common case
1532             (juint)(_lo - min) >= (juint)(max - _hi)) {
1533           // Try to widen to an unsigned range type of 31 bits:
1534           return make(_lo, max, WidenMax);
1535         } else {
1536           return make(min, _hi, WidenMax);
1537         }
1538       }
1539       return TypeInt::INT;
1540     }
1541     // Returned widened new guy
1542     return make(_lo,_hi,_widen+1);
1543   }
1544 
1545   // If old guy contains new, then we probably widened too far & dropped to
1546   // bottom.  Return the wider fellow.
1547   if ( ot->_lo <= _lo && ot->_hi >= _hi )
1548     return old;
1549 
1550   //fatal("Integer value range is not subset");
1551   //return this;
1552   return TypeInt::INT;
1553 }

1555 //------------------------------narrow---------------------------------------
1556 // Only happens for pessimistic optimizations.
1557 const Type *TypeInt::narrow( const Type *old ) const {
1558   if (_lo >= _hi)  return this;   // already narrow enough
1559   if (old == NULL)  return this;
1560   const TypeInt* ot = old->isa_int();
1561   if (ot == NULL)  return this;
1562   jint olo = ot->_lo;
1563   jint ohi = ot->_hi;
1564 
1565   // If new guy is equal to old guy, no narrowing
1566   if (_lo == olo && _hi == ohi)  return old;
1567 
1568   // If old guy was maximum range, allow the narrowing
1569   if (olo == min_jint && ohi == max_jint)  return this;
1570 
1571   if (_lo < olo || _hi > ohi)
1572     return this;                // doesn't narrow; pretty wierd
1573 
1574   // The new type narrows the old type, so look for a "death march".
1575   // See comments on PhaseTransform::saturate.
1576   juint nrange = (juint)_hi - _lo;
1577   juint orange = (juint)ohi - olo;
1578   if (nrange < max_juint - 1 && nrange > (orange >> 1) + (SMALLINT*2)) {
1579     // Use the new type only if the range shrinks a lot.
1580     // We do not want the optimizer computing 2^31 point by point.
1581     return old;
1582   }
1583 
1584   return this;
1585 }

1713 // Compute the MEET of two types.  It returns a new Type representation object
1714 // with reference count equal to the number of Types pointing at it.
1715 // Caller should wrap a Types around it.
1716 const Type *TypeLong::xmeet( const Type *t ) const {
1717   // Perform a fast test for common case; meeting the same types together.
1718   if( this == t ) return this;  // Meeting same type?
1719 
1720   // Currently "this->_base" is a TypeLong
1721   switch (t->base()) {          // Switch on original type
1722   case AnyPtr:                  // Mixing with oops happens when javac
1723   case RawPtr:                  // reuses local variables
1724   case OopPtr:
1725   case InstPtr:
1726   case AryPtr:
1727   case MetadataPtr:
1728   case KlassPtr:
1729   case NarrowOop:
1730   case NarrowKlass:
1731   case Int:
1732   case FloatTop:
1733   case FloatCon:
1734   case FloatBot:
1735   case DoubleTop:
1736   case DoubleCon:
1737   case DoubleBot:

1740   default:                      // All else is a mistake
1741     typerr(t);
1742   case Top:                     // No change
1743     return this;
1744   case Long:                    // Long vs Long?
1745     break;
1746   }
1747 
1748   // Expand covered set
1749   const TypeLong *r = t->is_long(); // Turn into a TypeLong
1750   return make( MIN2(_lo,r->_lo), MAX2(_hi,r->_hi), MAX2(_widen,r->_widen) );

1760 //------------------------------widen------------------------------------------
1761 // Only happens for optimistic top-down optimizations.
1762 const Type *TypeLong::widen( const Type *old, const Type* limit ) const {
1763   // Coming from TOP or such; no widening
1764   if( old->base() != Long ) return this;
1765   const TypeLong *ot = old->is_long();
1766 
1767   // If new guy is equal to old guy, no widening
1768   if( _lo == ot->_lo && _hi == ot->_hi )
1769     return old;
1770 
1771   // If new guy contains old, then we widened
1772   if( _lo <= ot->_lo && _hi >= ot->_hi ) {
1773     // New contains old
1774     // If new guy is already wider than old, no widening
1775     if( _widen > ot->_widen ) return this;
1776     // If old guy was a constant, do not bother
1777     if (ot->_lo == ot->_hi)  return this;
1778     // Now widen new guy.
1779     // Check for widening too far
1780     if (_widen == WidenMax) {
1781       jlong max = max_jlong;
1782       jlong min = min_jlong;
1783       if (limit->isa_long()) {
1784         max = limit->is_long()->_hi;
1785         min = limit->is_long()->_lo;
1786       }
1787       if (min < _lo && _hi < max) {
1788         // If neither endpoint is extremal yet, push out the endpoint
1789         // which is closer to its respective limit.
1790         if (_lo >= 0 ||                 // easy common case
1791             ((julong)_lo - min) >= ((julong)max - _hi)) {
1792           // Try to widen to an unsigned range type of 32/63 bits:
1793           if (max >= max_juint && _hi < max_juint)
1794             return make(_lo, max_juint, WidenMax);
1795           else
1796             return make(_lo, max, WidenMax);
1797         } else {
1798           return make(min, _hi, WidenMax);
1799         }
1800       }
1801       return TypeLong::LONG;
1802     }
1803     // Returned widened new guy
1804     return make(_lo,_hi,_widen+1);
1805   }
1806 
1807   // If old guy contains new, then we probably widened too far & dropped to
1808   // bottom.  Return the wider fellow.
1809   if ( ot->_lo <= _lo && ot->_hi >= _hi )
1810     return old;
1811 
1812   //  fatal("Long value range is not subset");
1813   // return this;
1814   return TypeLong::LONG;
1815 }

2374   if (!UseExactTypes)       return false;

2846 /**

2848  *
2849  * @param other  type to meet with
2850  */
2851 const TypePtr* TypePtr::xmeet_speculative(const TypePtr* other) const {
2852   bool this_has_spec = (_speculative != NULL);
2853   bool other_has_spec = (other->speculative() != NULL);
2854 
2855   if (!this_has_spec && !other_has_spec) {
2856     return NULL;
2857   }
2858

2860   // a speculative type and the other has not, we meet the speculative
2861   // type of one branch with the actual type of the other. If the
2862   // actual type is exact and the speculative is as well, then the
2863   // result is a speculative type which is exact and we can continue
2864   // speculation further.
2865   const TypePtr* this_spec = _speculative;
2866   const TypePtr* other_spec = other->speculative();

2869     this_spec = this;
2870   }
2871 
2872   if (!other_has_spec) {
2873     other_spec = other;
2874   }

3127 //------------------------------meet-------------------------------------------
3128 // Compute the MEET of two types.  It returns a new Type object.
3129 const Type *TypeRawPtr::xmeet( const Type *t ) const {
3130   // Perform a fast test for common case; meeting the same types together.
3131   if( this == t ) return this;  // Meeting same type-rep?
3132 
3133   // Current "this->_base" is RawPtr
3134   switch( t->base() ) {         // switch on original type
3135   case Bottom:                  // Ye Olde Default
3136     return t;
3137   case Top:
3138     return this;
3139   case AnyPtr:                  // Meeting to AnyPtrs
3140     break;
3141   case RawPtr: {                // might be top, bot, any/not or constant
3142     enum PTR tptr = t->is_ptr()->ptr();
3143     enum PTR ptr = meet_ptr( tptr );
3144     if( ptr == Constant ) {     // Cannot be equal constants, so...
3145       if( tptr == Constant && _ptr != Constant)  return t;
3146       if( _ptr == Constant && tptr != Constant)  return this;
3147       ptr = NotNull;            // Fall down in lattice
3148     }
3149     return make( ptr );
3150   }
3151 
3152   case OopPtr:
3153   case InstPtr:
3154   case AryPtr:
3155   case MetadataPtr:
3156   case KlassPtr:
3157     return TypePtr::BOTTOM;     // Oop meet raw is not well defined
3158   default:                      // All else is a mistake
3159     typerr(t);
3160   }
3161 
3162   // Found an AnyPtr type vs self-RawPtr type
3163   const TypePtr *tp = t->is_ptr();
3164   switch (tp->ptr()) {
3165   case TypePtr::TopPTR:  return this;
3166   case TypePtr::BotPTR:  return t;
3167   case TypePtr::Null:
3168     if( _ptr == TypePtr::TopPTR ) return t;
3169     return TypeRawPtr::BOTTOM;
3170   case TypePtr::NotNull: return TypePtr::make(AnyPtr, meet_ptr(TypePtr::NotNull), tp->meet_offset(0), tp->speculative(), tp->inline_depth());

3172     if( _ptr == TypePtr::Constant) return this;

3392   case KlassPtr:
3393     return TypePtr::BOTTOM;     // Oop meet raw is not well defined
3394 
3395   case AnyPtr: {
3396     // Found an AnyPtr type vs self-OopPtr type
3397     const TypePtr *tp = t->is_ptr();
3398     Offset offset = meet_offset(tp->offset());
3399     PTR ptr = meet_ptr(tp->ptr());
3400     const TypePtr* speculative = xmeet_speculative(tp);
3401     int depth = meet_inline_depth(tp->inline_depth());
3402     switch (tp->ptr()) {
3403     case Null:
3404       if (ptr == Null)  return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
3405       // else fall through:
3406     case TopPTR:
3407     case AnyNull: {
3408       int instance_id = meet_instance_id(InstanceTop);
3409       return make(ptr, offset, instance_id, speculative, depth);
3410     }
3411     case BotPTR:
3412     case NotNull:
3413       return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
3414     default: typerr(t);
3415     }
3416   }
3417 
3418   case OopPtr: {                 // Meeting to other OopPtrs
3419     const TypeOopPtr *tp = t->is_oopptr();
3420     int instance_id = meet_instance_id(tp->instance_id());
3421     const TypePtr* speculative = xmeet_speculative(tp);
3422     int depth = meet_inline_depth(tp->inline_depth());
3423     return make(meet_ptr(tp->ptr()), meet_offset(tp->offset()), instance_id, speculative, depth);
3424   }
3425 
3426   case InstPtr:                  // For these, flip the call around to cut down
3427   case AryPtr:
3428     return t->xmeet(this);      // Call in reverse direction
3429 
3430   } // End of switch
3431   return this;                  // Return the double constant
3432 }

3443 //--------------------------make_from_klass_common-----------------------------
3444 // Computes the element-type given a klass.
3445 const TypeOopPtr* TypeOopPtr::make_from_klass_common(ciKlass *klass, bool klass_change, bool try_for_exact) {
3446   if (klass->is_instance_klass() || klass->is_valuetype()) {
3447     Compile* C = Compile::current();
3448     Dependencies* deps = C->dependencies();
3449     assert((deps != NULL) == (C->method() != NULL && C->method()->code_size() > 0), "sanity");
3450     // Element is an instance
3451     bool klass_is_exact = false;
3452     if (klass->is_loaded()) {
3453       // Try to set klass_is_exact.
3454       ciInstanceKlass* ik = klass->as_instance_klass();
3455       klass_is_exact = ik->is_final();
3456       if (!klass_is_exact && klass_change
3457           && deps != NULL && UseUniqueSubclasses) {
3458         ciInstanceKlass* sub = ik->unique_concrete_subklass();
3459         if (sub != NULL) {
3460           deps->assert_abstract_with_unique_concrete_subtype(ik, sub);
3461           klass = ik = sub;
3462           klass_is_exact = sub->is_final();
3463         }
3464       }
3465       if (!klass_is_exact && try_for_exact
3466           && deps != NULL && UseExactTypes) {
3467         if (!ik->is_interface() && !ik->has_subklass()) {
3468           // Add a dependence; if concrete subclass added we need to recompile
3469           deps->assert_leaf_type(ik);
3470           klass_is_exact = true;
3471         }
3472       }
3473     }
3474     return TypeInstPtr::make(TypePtr::BotPTR, klass, klass_is_exact, NULL, Offset(0), klass->flatten_array());
3475   } else if (klass->is_obj_array_klass()) {
3476     // Element is an object or value array. Recursively call ourself.
3477     const TypeOopPtr* etype = TypeOopPtr::make_from_klass_common(klass->as_array_klass()->element_klass(), /* klass_change= */ false, try_for_exact);
3478     if (etype->is_valuetypeptr()) {
3479       etype = etype->join_speculative(TypePtr::NOTNULL)->is_oopptr();
3480     }
3481     // Determine null-free/flattened properties
3482     const TypeOopPtr* exact_etype = etype;
3483     if (etype->can_be_value_type()) {
3484       // Use exact type if element can be a value type
3485       exact_etype = TypeOopPtr::make_from_klass_common(klass->as_array_klass()->element_klass(), /* klass_change= */ true, /* try_for_exact= */ true);
3486     }
3487     bool not_null_free = !exact_etype->can_be_value_type();
3488     bool not_flat = !ValueArrayFlatten || not_null_free || (exact_etype->is_valuetypeptr() && !exact_etype->value_klass()->flatten_array());
3489 
3490     bool xk = etype->klass_is_exact();
3491     const TypeAry* arr0 = TypeAry::make(etype, TypeInt::POS, false, not_flat, not_null_free);
3492     // We used to pass NotNull in here, asserting that the sub-arrays
3493     // are all not-null.  This is not true in generally, as code can
3494     // slam NULLs down in the subarrays.
3495     const TypeAryPtr* arr = TypeAryPtr::make(TypePtr::BotPTR, arr0, klass, xk, Offset(0));
3496     return arr;
3497   } else if (klass->is_type_array_klass()) {
3498     // Element is an typeArray
3499     const Type* etype = get_const_basic_type(klass->as_type_array_klass()->element_type());
3500     const TypeAry* arr0 = TypeAry::make(etype, TypeInt::POS,
3501                                         /* stable= */ false, /* not_flat= */ true, /* not_null_free= */ true);
3502     // We used to pass NotNull in here, asserting that the array pointer
3503     // is not-null. That was not true in general.
3504     const TypeAryPtr* arr = TypeAryPtr::make(TypePtr::BotPTR, arr0, klass, true, Offset(0));
3505     return arr;
3506   } else if (klass->is_value_array_klass()) {
3507     ciValueKlass* vk = klass->as_array_klass()->element_klass()->as_value_klass();
3508     const TypeAry* arr0 = TypeAry::make(TypeValueType::make(vk), TypeInt::POS);
3509     const TypeAryPtr* arr = TypeAryPtr::make(TypePtr::BotPTR, arr0, klass, true, Offset(0));
3510     return arr;
3511   } else {
3512     ShouldNotReachHere();
3513     return NULL;
3514   }
3515 }

3600 //-----------------------------filter------------------------------------------
3601 // Do not allow interface-vs.-noninterface joins to collapse to top.
3602 const Type *TypeOopPtr::filter_helper(const Type *kills, bool include_speculative) const {
3603 
3604   const Type* ft = join_helper(kills, include_speculative);
3605   const TypeInstPtr* ftip = ft->isa_instptr();
3606   const TypeInstPtr* ktip = kills->isa_instptr();
3607 
3608   if (ft->empty()) {
3609     // Check for evil case of 'this' being a class and 'kills' expecting an
3610     // interface.  This can happen because the bytecodes do not contain
3611     // enough type info to distinguish a Java-level interface variable
3612     // from a Java-level object variable.  If we meet 2 classes which
3613     // both implement interface I, but their meet is at 'j/l/O' which
3614     // doesn't implement I, we have no way to tell if the result should
3615     // be 'I' or 'j/l/O'.  Thus we'll pick 'j/l/O'.  If this then flows
3616     // into a Phi which "knows" it's an Interface type we'll have to
3617     // uplift the type.
3618     if (!empty()) {
3619       if (ktip != NULL && ktip->is_loaded() && ktip->klass()->is_interface()) {
3620         return kills;           // Uplift to interface
3621       }
3622       // Also check for evil cases of 'this' being a class array
3623       // and 'kills' expecting an array of interfaces.
3624       Type::get_arrays_base_elements(ft, kills, NULL, &ktip);
3625       if (ktip != NULL && ktip->is_loaded() && ktip->klass()->is_interface()) {
3626         return kills;           // Uplift to array of interface
3627       }
3628     }
3629 
3630     return Type::TOP;           // Canonical empty value
3631   }
3632 
3633   // If we have an interface-typed Phi or cast and we narrow to a class type,
3634   // the join should report back the class.  However, if we have a J/L/Object
3635   // class-typed Phi and an interface flows in, it's possible that the meet &
3636   // join report an interface back out.  This isn't possible but happens
3637   // because the type system doesn't interact well with interfaces.
3638   if (ftip != NULL && ktip != NULL &&
3639       ftip->is_loaded() &&  ftip->klass()->is_interface() &&
3640       ktip->is_loaded() && !ktip->klass()->is_interface()) {
3641     assert(!ftip->klass_is_exact(), "interface could not be exact");
3642     return ktip->cast_to_ptr_type(ftip->ptr());
3643   }
3644 
3645   return ft;
3646 }

3882 //------------------------------xmeet_unloaded---------------------------------
3883 // Compute the MEET of two InstPtrs when at least one is unloaded.
3884 // Assume classes are different since called after check for same name/class-loader
3885 const TypeInstPtr *TypeInstPtr::xmeet_unloaded(const TypeInstPtr *tinst) const {
3886     Offset off = meet_offset(tinst->offset());
3887     PTR ptr = meet_ptr(tinst->ptr());
3888     int instance_id = meet_instance_id(tinst->instance_id());
3889     const TypePtr* speculative = xmeet_speculative(tinst);
3890     int depth = meet_inline_depth(tinst->inline_depth());
3891 
3892     const TypeInstPtr *loaded    = is_loaded() ? this  : tinst;
3893     const TypeInstPtr *unloaded  = is_loaded() ? tinst : this;
3894     if( loaded->klass()->equals(ciEnv::current()->Object_klass()) ) {
3895       //
3896       // Meet unloaded class with java/lang/Object
3897       //
3898       // Meet
3899       //          |                     Unloaded Class
3900       //  Object  |   TOP    |   AnyNull | Constant |   NotNull |  BOTTOM   |
3901       //  ===================================================================
3902       //   TOP    | ..........................Unloaded......................|
3903       //  AnyNull |  U-AN    |................Unloaded......................|
3904       // Constant | ... O-NN .................................. |   O-BOT   |
3905       //  NotNull | ... O-NN .................................. |   O-BOT   |
3906       //  BOTTOM  | ........................Object-BOTTOM ..................|
3907       //
3908       assert(loaded->ptr() != TypePtr::Null, "insanity check");
3909       //
3910       if(      loaded->ptr() == TypePtr::TopPTR ) { return unloaded; }
3911       else if (loaded->ptr() == TypePtr::AnyNull) { return TypeInstPtr::make(ptr, unloaded->klass(), false, NULL, off, false, instance_id, speculative, depth); }
3912       else if (loaded->ptr() == TypePtr::BotPTR ) { return TypeInstPtr::BOTTOM; }
3913       else if (loaded->ptr() == TypePtr::Constant || loaded->ptr() == TypePtr::NotNull) {
3914         if (unloaded->ptr() == TypePtr::BotPTR  ) { return TypeInstPtr::BOTTOM;  }
3915         else                                      { return TypeInstPtr::NOTNULL; }
3916       }
3917       else if( unloaded->ptr() == TypePtr::TopPTR )  { return unloaded; }
3918 
3919       return unloaded->cast_to_ptr_type(TypePtr::AnyNull)->is_instptr();
3920     }
3921 
3922     // Both are unloaded, not the same class, not Object
3923     // Or meet unloaded with a different loaded class, not java/lang/Object
3924     if( ptr != TypePtr::BotPTR ) {
3925       return TypeInstPtr::NOTNULL;
3926     }
3927     return TypeInstPtr::BOTTOM;
3928 }

3931 //------------------------------meet-------------------------------------------
3932 // Compute the MEET of two types.  It returns a new Type object.
3933 const Type *TypeInstPtr::xmeet_helper(const Type *t) const {
3934   // Perform a fast test for common case; meeting the same types together.
3935   if( this == t ) return this;  // Meeting same type-rep?
3936 
3937   // Current "this->_base" is Pointer
3938   switch (t->base()) {          // switch on original type
3939 
3940   case Int:                     // Mixing ints & oops happens when javac
3941   case Long:                    // reuses local variables
3942   case FloatTop:
3943   case FloatCon:
3944   case FloatBot:
3945   case DoubleTop:
3946   case DoubleCon:
3947   case DoubleBot:
3948   case NarrowOop:
3949   case NarrowKlass:
3950   case Bottom:                  // Ye Olde Default
3951     return Type::BOTTOM;
3952   case Top:
3953     return this;
3954 
3955   default:                      // All else is a mistake
3956     typerr(t);
3957 
3958   case MetadataPtr:
3959   case KlassPtr:
3960   case RawPtr: return TypePtr::BOTTOM;
3961 
3962   case AryPtr: {                // All arrays inherit from Object class
3963     const TypeAryPtr *tp = t->is_aryptr();
3964     Offset offset = meet_offset(tp->offset());
3965     PTR ptr = meet_ptr(tp->ptr());
3966     int instance_id = meet_instance_id(tp->instance_id());
3967     const TypePtr* speculative = xmeet_speculative(tp);
3968     int depth = meet_inline_depth(tp->inline_depth());
3969     switch (ptr) {
3970     case TopPTR:
3971     case AnyNull:                // Fall 'down' to dual of object klass
3972       // For instances when a subclass meets a superclass we fall
3973       // below the centerline when the superclass is exact. We need to
3974       // do the same here.
3975       if (klass()->equals(ciEnv::current()->Object_klass()) && !klass_is_exact() && !flat_array()) {
3976         return TypeAryPtr::make(ptr, tp->ary(), tp->klass(), tp->klass_is_exact(), offset, tp->field_offset(), instance_id, speculative, depth);
3977       } else {
3978         // cannot subclass, so the meet has to fall badly below the centerline
3979         ptr = NotNull;
3980         instance_id = InstanceBot;
3981         return TypeInstPtr::make( ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
3982       }
3983     case Constant:
3984     case NotNull:
3985     case BotPTR:                // Fall down to object klass
3986       // LCA is object_klass, but if we subclass from the top we can do better
3987       if( above_centerline(_ptr) ) { // if( _ptr == TopPTR || _ptr == AnyNull )
3988         // If 'this' (InstPtr) is above the centerline and it is Object class
3989         // then we can subclass in the Java class hierarchy.
3990         // For instances when a subclass meets a superclass we fall
3991         // below the centerline when the superclass is exact. We need
3992         // to do the same here.
3993         if (klass()->equals(ciEnv::current()->Object_klass()) && !klass_is_exact() && !flat_array()) {
3994           // that is, tp's array type is a subtype of my klass
3995           return TypeAryPtr::make(ptr, (ptr == Constant ? tp->const_oop() : NULL),
3996                                   tp->ary(), tp->klass(), tp->klass_is_exact(), offset, tp->field_offset(), instance_id, speculative, depth);
3997         }
3998       }
3999       // The other case cannot happen, since I cannot be a subtype of an array.
4000       // The meet falls down to Object class below centerline.
4001       if( ptr == Constant )
4002          ptr = NotNull;
4003       instance_id = InstanceBot;
4004       return make(ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
4005     default: typerr(t);
4006     }
4007   }
4008 
4009   case OopPtr: {                // Meeting to OopPtrs
4010     // Found a OopPtr type vs self-InstPtr type
4011     const TypeOopPtr *tp = t->is_oopptr();
4012     Offset offset = meet_offset(tp->offset());
4013     PTR ptr = meet_ptr(tp->ptr());
4014     switch (tp->ptr()) {
4015     case TopPTR:
4016     case AnyNull: {
4017       int instance_id = meet_instance_id(InstanceTop);
4018       const TypePtr* speculative = xmeet_speculative(tp);
4019       int depth = meet_inline_depth(tp->inline_depth());
4020       return make(ptr, klass(), klass_is_exact(),
4021                   (ptr == Constant ? const_oop() : NULL), offset, flat_array(), instance_id, speculative, depth);
4022     }
4023     case NotNull:
4024     case BotPTR: {
4025       int instance_id = meet_instance_id(tp->instance_id());
4026       const TypePtr* speculative = xmeet_speculative(tp);
4027       int depth = meet_inline_depth(tp->inline_depth());
4028       return TypeOopPtr::make(ptr, offset, instance_id, speculative, depth);
4029     }
4030     default: typerr(t);
4031     }
4032   }
4033 
4034   case AnyPtr: {                // Meeting to AnyPtrs
4035     // Found an AnyPtr type vs self-InstPtr type
4036     const TypePtr *tp = t->is_ptr();
4037     Offset offset = meet_offset(tp->offset());
4038     PTR ptr = meet_ptr(tp->ptr());
4039     int instance_id = meet_instance_id(InstanceTop);
4040     const TypePtr* speculative = xmeet_speculative(tp);
4041     int depth = meet_inline_depth(tp->inline_depth());
4042     switch (tp->ptr()) {
4043     case Null:
4044       if( ptr == Null ) return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
4045       // else fall through to AnyNull
4046     case TopPTR:
4047     case AnyNull: {
4048       return make(ptr, klass(), klass_is_exact(),
4049                   (ptr == Constant ? const_oop() : NULL), offset, flat_array(), instance_id, speculative, depth);
4050     }
4051     case NotNull:
4052     case BotPTR:
4053       return TypePtr::make(AnyPtr, ptr, offset, speculative,depth);
4054     default: typerr(t);
4055     }
4056   }
4057 
4058   /*
4059                  A-top         }
4060                /   |   \       }  Tops
4061            B-top A-any C-top   }
4062               | /  |  \ |      }  Any-nulls
4063            B-any   |   C-any   }
4064               |    |    |
4065            B-con A-con C-con   } constants; not comparable across classes
4066               |    |    |
4067            B-not   |   C-not   }
4068               | \  |  / |      }  not-nulls
4069            B-bot A-not C-bot   }
4070                \   |   /       }  Bottoms
4071                  A-bot         }
4072   */
4073 
4074   case InstPtr: {                // Meeting 2 Oops?
4075     // Found an InstPtr sub-type vs self-InstPtr type
4076     const TypeInstPtr *tinst = t->is_instptr();
4077     Offset off = meet_offset( tinst->offset() );
4078     PTR ptr = meet_ptr( tinst->ptr() );
4079     int instance_id = meet_instance_id(tinst->instance_id());
4080     const TypePtr* speculative = xmeet_speculative(tinst);
4081     int depth = meet_inline_depth(tinst->inline_depth());
4082 
4083     // Check for easy case; klasses are equal (and perhaps not loaded!)
4084     // If we have constants, then we created oops so classes are loaded
4085     // and we can handle the constants further down.  This case handles
4086     // both-not-loaded or both-loaded classes
4087     if (ptr != Constant && klass()->equals(tinst->klass()) && klass_is_exact() == tinst->klass_is_exact() &&
4088         flat_array() == tinst->flat_array()) {
4089       return make(ptr, klass(), klass_is_exact(), NULL, off, flat_array(), instance_id, speculative, depth);
4090     }
4091 
4092     // Classes require inspection in the Java klass hierarchy.  Must be loaded.
4093     ciKlass* tinst_klass = tinst->klass();
4094     ciKlass* this_klass  = this->klass();
4095     bool tinst_xk = tinst->klass_is_exact();
4096     bool this_xk  = this->klass_is_exact();
4097     bool tinst_flat_array = tinst->flat_array();
4098     bool this_flat_array  = this->flat_array();
4099     if (!tinst_klass->is_loaded() || !this_klass->is_loaded() ) {
4100       // One of these classes has not been loaded
4101       const TypeInstPtr *unloaded_meet = xmeet_unloaded(tinst);
4102 #ifndef PRODUCT
4103       if( PrintOpto && Verbose ) {
4104         tty->print("meet of unloaded classes resulted in: "); unloaded_meet->dump(); tty->cr();
4105         tty->print("  this == "); this->dump(); tty->cr();
4106         tty->print(" tinst == "); tinst->dump(); tty->cr();
4107       }
4108 #endif
4109       return unloaded_meet;
4110     }
4111 
4112     // Handle mixing oops and interfaces first.
4113     if( this_klass->is_interface() && !(tinst_klass->is_interface() ||
4114                                         tinst_klass == ciEnv::current()->Object_klass())) {
4115       ciKlass *tmp = tinst_klass; // Swap interface around
4116       tinst_klass = this_klass;
4117       this_klass = tmp;
4118       bool tmp2 = tinst_xk;
4119       tinst_xk = this_xk;
4120       this_xk = tmp2;
4121       tmp2 = tinst_flat_array;
4122       tinst_flat_array = this_flat_array;
4123       this_flat_array = tmp2;
4124     }
4125     if (tinst_klass->is_interface() &&
4126         !(this_klass->is_interface() ||
4127           // Treat java/lang/Object as an honorary interface,
4128           // because we need a bottom for the interface hierarchy.
4129           this_klass == ciEnv::current()->Object_klass())) {
4130       // Oop meets interface!
4131 
4132       // See if the oop subtypes (implements) interface.
4133       ciKlass *k;
4134       bool xk;
4135       bool flat_array;
4136       if( this_klass->is_subtype_of( tinst_klass ) ) {
4137         // Oop indeed subtypes.  Now keep oop or interface depending
4138         // on whether we are both above the centerline or either is
4139         // below the centerline.  If we are on the centerline
4140         // (e.g., Constant vs. AnyNull interface), use the constant.
4141         k  = below_centerline(ptr) ? tinst_klass : this_klass;
4142         // If we are keeping this_klass, keep its exactness too.
4143         xk = below_centerline(ptr) ? tinst_xk    : this_xk;
4144         flat_array = below_centerline(ptr) ? tinst_flat_array    : this_flat_array;
4145       } else {                  // Does not implement, fall to Object
4146         // Oop does not implement interface, so mixing falls to Object
4147         // just like the verifier does (if both are above the
4148         // centerline fall to interface)
4149         k = above_centerline(ptr) ? tinst_klass : ciEnv::current()->Object_klass();
4150         xk = above_centerline(ptr) ? tinst_xk : false;
4151         flat_array = above_centerline(ptr) ? tinst_flat_array : false;
4152         // Watch out for Constant vs. AnyNull interface.
4153         if (ptr == Constant)  ptr = NotNull;   // forget it was a constant
4154         instance_id = InstanceBot;
4155       }
4156       ciObject* o = NULL;  // the Constant value, if any
4157       if (ptr == Constant) {
4158         // Find out which constant.
4159         o = (this_klass == klass()) ? const_oop() : tinst->const_oop();
4160       }
4161       return make(ptr, k, xk, o, off, flat_array, instance_id, speculative, depth);
4162     }
4163 
4164     // Either oop vs oop or interface vs interface or interface vs Object
4165 
4166     // !!! Here's how the symmetry requirement breaks down into invariants:
4167     // If we split one up & one down AND they subtype, take the down man.
4168     // If we split one up & one down AND they do NOT subtype, "fall hard".
4169     // If both are up and they subtype, take the subtype class.
4170     // If both are up and they do NOT subtype, "fall hard".
4171     // If both are down and they subtype, take the supertype class.
4172     // If both are down and they do NOT subtype, "fall hard".
4173     // Constants treated as down.
4174 
4175     // Now, reorder the above list; observe that both-down+subtype is also
4176     // "fall hard"; "fall hard" becomes the default case:
4177     // If we split one up & one down AND they subtype, take the down man.
4178     // If both are up and they subtype, take the subtype class.
4179 
4180     // If both are down and they subtype, "fall hard".
4181     // If both are down and they do NOT subtype, "fall hard".
4182     // If both are up and they do NOT subtype, "fall hard".
4183     // If we split one up & one down AND they do NOT subtype, "fall hard".
4184 
4185     // If a proper subtype is exact, and we return it, we return it exactly.
4186     // If a proper supertype is exact, there can be no subtyping relationship!
4187     // If both types are equal to the subtype, exactness is and-ed below the
4188     // centerline and or-ed above it.  (N.B. Constants are always exact.)
4189 
4190     // Check for subtyping:
4191     ciKlass *subtype = NULL;
4192     bool subtype_exact = false;
4193     bool flat_array = false;
4194     if (tinst_klass->equals(this_klass)) {
4195       subtype = this_klass;
4196       subtype_exact = below_centerline(ptr) ? (this_xk && tinst_xk) : (this_xk || tinst_xk);
4197       flat_array = below_centerline(ptr) ? (this_flat_array && tinst_flat_array) : (this_flat_array || tinst_flat_array);
4198     } else if(!tinst_xk && this_klass->is_subtype_of(tinst_klass) && (!tinst_flat_array || this_flat_array)) {
4199       subtype = this_klass;     // Pick subtyping class
4200       subtype_exact = this_xk;
4201       flat_array = this_flat_array;
4202     } else if(!this_xk && tinst_klass->is_subtype_of(this_klass) && (!this_flat_array || tinst_flat_array)) {
4203       subtype = tinst_klass;    // Pick subtyping class
4204       subtype_exact = tinst_xk;
4205       flat_array = tinst_flat_array;
4206     }
4207 
4208     if (subtype) {
4209       if (above_centerline(ptr)) { // both are up?
4210         this_klass = tinst_klass = subtype;
4211         this_xk = tinst_xk = subtype_exact;
4212         this_flat_array = tinst_flat_array = flat_array;
4213       } else if (above_centerline(this ->_ptr) && !above_centerline(tinst->_ptr)) {
4214         this_klass = tinst_klass; // tinst is down; keep down man
4215         this_xk = tinst_xk;
4216         this_flat_array = tinst_flat_array;
4217       } else if (above_centerline(tinst->_ptr) && !above_centerline(this ->_ptr)) {
4218         tinst_klass = this_klass; // this is down; keep down man
4219         tinst_xk = this_xk;
4220         tinst_flat_array = this_flat_array;
4221       } else {
4222         this_xk = subtype_exact;  // either they are equal, or we'll do an LCA
4223         this_flat_array = flat_array;
4224       }
4225     }
4226 
4227     // Check for classes now being equal
4228     if (tinst_klass->equals(this_klass)) {
4229       // If the klasses are equal, the constants may still differ.  Fall to
4230       // NotNull if they do (neither constant is NULL; that is a special case
4231       // handled elsewhere).
4232       ciObject* o = NULL;             // Assume not constant when done
4233       ciObject* this_oop  = const_oop();
4234       ciObject* tinst_oop = tinst->const_oop();
4235       if( ptr == Constant ) {
4236         if (this_oop != NULL && tinst_oop != NULL &&
4237             this_oop->equals(tinst_oop) )
4238           o = this_oop;
4239         else if (above_centerline(this ->_ptr))
4240           o = tinst_oop;
4241         else if (above_centerline(tinst ->_ptr))
4242           o = this_oop;
4243         else
4244           ptr = NotNull;
4245       }
4246       return make(ptr, this_klass, this_xk, o, off, this_flat_array, instance_id, speculative, depth);
4247     } // Else classes are not equal
4248 
4249     // Since klasses are different, we require a LCA in the Java
4250     // class hierarchy - which means we have to fall to at least NotNull.
4251     if( ptr == TopPTR || ptr == AnyNull || ptr == Constant )
4252       ptr = NotNull;
4253 
4254     instance_id = InstanceBot;
4255 
4256     // Now we find the LCA of Java classes
4257     ciKlass* k = this_klass->least_common_ancestor(tinst_klass);
4258     return make(ptr, k, false, NULL, off, false, instance_id, speculative, depth);
4259   } // End of case InstPtr
4260 
4261   case ValueType: {
4262     const TypeValueType* tv = t->is_valuetype();
4263     if (above_centerline(ptr())) {
4264       if (tv->value_klass()->is_subtype_of(_klass)) {
4265         return t;
4266       } else {
4267         return TypeInstPtr::NOTNULL;
4268       }
4269     } else {
4270       PTR ptr = this->_ptr;
4271       if (ptr == Constant) {
4272         ptr = NotNull;
4273       }
4274       if (tv->value_klass()->is_subtype_of(_klass)) {
4275         return TypeInstPtr::make(ptr, _klass);
4276       } else {
4277         return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass());
4278       }
4279     }
4280   }
4281 
4282   } // End of switch
4283   return this;                  // Return the double constant
4284 }

4483 // Return NULL if the resulting int type becomes empty.
4484 const TypeInt* TypeAryPtr::narrow_size_type(const TypeInt* size) const {
4485   jint hi = size->_hi;

4489   //if (index_not_size)  --max_hi;     // type of a valid array index, FTR

4495     }

4499     hi = max_hi;
4500     if (size->is_con()) {
4501       lo = hi;

4597 //------------------------------meet-------------------------------------------
4598 // Compute the MEET of two types.  It returns a new Type object.
4599 const Type *TypeAryPtr::xmeet_helper(const Type *t) const {
4600   // Perform a fast test for common case; meeting the same types together.
4601   if( this == t ) return this;  // Meeting same type-rep?
4602   // Current "this->_base" is Pointer
4603   switch (t->base()) {          // switch on original type
4604 
4605   // Mixing ints & oops happens when javac reuses local variables
4606   case Int:
4607   case Long:
4608   case FloatTop:
4609   case FloatCon:
4610   case FloatBot:
4611   case DoubleTop:
4612   case DoubleCon:
4613   case DoubleBot:
4614   case NarrowOop:
4615   case NarrowKlass:
4616   case Bottom:                  // Ye Olde Default
4617     return Type::BOTTOM;
4618   case Top:
4619     return this;
4620 
4621   default:                      // All else is a mistake
4622     typerr(t);
4623 
4624   case OopPtr: {                // Meeting to OopPtrs
4625     // Found a OopPtr type vs self-AryPtr type
4626     const TypeOopPtr *tp = t->is_oopptr();
4627     Offset offset = meet_offset(tp->offset());
4628     PTR ptr = meet_ptr(tp->ptr());
4629     int depth = meet_inline_depth(tp->inline_depth());
4630     const TypePtr* speculative = xmeet_speculative(tp);
4631     switch (tp->ptr()) {
4632     case TopPTR:
4633     case AnyNull: {
4634       int instance_id = meet_instance_id(InstanceTop);
4635       return make(ptr, (ptr == Constant ? const_oop() : NULL),
4636                   _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4637     }
4638     case BotPTR:
4639     case NotNull: {
4640       int instance_id = meet_instance_id(tp->instance_id());
4641       return TypeOopPtr::make(ptr, offset, instance_id, speculative, depth);
4642     }
4643     default: ShouldNotReachHere();
4644     }
4645   }
4646 
4647   case AnyPtr: {                // Meeting two AnyPtrs
4648     // Found an AnyPtr type vs self-AryPtr type
4649     const TypePtr *tp = t->is_ptr();
4650     Offset offset = meet_offset(tp->offset());
4651     PTR ptr = meet_ptr(tp->ptr());
4652     const TypePtr* speculative = xmeet_speculative(tp);
4653     int depth = meet_inline_depth(tp->inline_depth());
4654     switch (tp->ptr()) {
4655     case TopPTR:
4656       return this;
4657     case BotPTR:
4658     case NotNull:
4659       return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
4660     case Null:
4661       if( ptr == Null ) return TypePtr::make(AnyPtr, ptr, offset, speculative, depth);
4662       // else fall through to AnyNull
4663     case AnyNull: {
4664       int instance_id = meet_instance_id(InstanceTop);
4665       return make(ptr, (ptr == Constant ? const_oop() : NULL),
4666                   _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4667     }
4668     default: ShouldNotReachHere();
4669     }
4670   }
4671 
4672   case MetadataPtr:
4673   case KlassPtr:
4674   case RawPtr: return TypePtr::BOTTOM;
4675 
4676   case AryPtr: {                // Meeting 2 references?
4677     const TypeAryPtr *tap = t->is_aryptr();
4678     Offset off = meet_offset(tap->offset());
4679     Offset field_off = meet_field_offset(tap->field_offset());
4680     const TypeAry *tary = _ary->meet_speculative(tap->_ary)->is_ary();
4681     PTR ptr = meet_ptr(tap->ptr());
4682     int instance_id = meet_instance_id(tap->instance_id());
4683     const TypePtr* speculative = xmeet_speculative(tap);
4684     int depth = meet_inline_depth(tap->inline_depth());
4685     ciKlass* lazy_klass = NULL;
4686     if (tary->_elem->isa_int()) {
4687       // Integral array element types have irrelevant lattice relations.
4688       // It is the klass that determines array layout, not the element type.
4689       if (_klass == NULL)
4690         lazy_klass = tap->_klass;
4691       else if (tap->_klass == NULL || tap->_klass == _klass) {
4692         lazy_klass = _klass;
4693       } else {
4694         // Something like byte[int+] meets char[int+].
4695         // This must fall to bottom, not (int[-128..65535])[int+].
4696         instance_id = InstanceBot;
4697         tary = TypeAry::make(Type::BOTTOM, tary->_size, tary->_stable, tary->_not_flat, tary->_not_null_free);
4698       }
4699     } else // Non integral arrays.
4700       // Must fall to bottom if exact klasses in upper lattice
4701       // are not equal or super klass is exact.
4702       if ((above_centerline(ptr) || ptr == Constant) && klass() != tap->klass() &&
4703           // meet with top[] and bottom[] are processed further down:
4704           tap->_klass != NULL && this->_klass != NULL &&
4705           // both are exact and not equal:
4706           ((tap->_klass_is_exact && this->_klass_is_exact) ||
4707            // 'tap' is exact and super or unrelated:
4708            (tap->_klass_is_exact && !tap->klass()->is_subtype_of(klass())) ||
4709            // 'this' is exact and super or unrelated:
4710            (this->_klass_is_exact && !klass()->is_subtype_of(tap->klass())))) {
4711       if (above_centerline(ptr) || (tary->_elem->make_ptr() && above_centerline(tary->_elem->make_ptr()->_ptr))) {
4712         tary = TypeAry::make(Type::BOTTOM, tary->_size, tary->_stable, tary->_not_flat, tary->_not_null_free);
4713       }
4714       return make(NotNull, NULL, tary, lazy_klass, false, off, field_off, InstanceBot, speculative, depth);
4715     } else if (klass() != NULL && tap->klass() != NULL && klass()->is_value_array_klass() != tap->klass()->is_value_array_klass()) {
4716       // Meeting flattened value type array with non-flattened array. Adjust (field) offset accordingly.
4717       if (tary->_elem->isa_valuetype()) {
4718         // Result is flattened
4719         off = Offset(elem()->isa_valuetype() ? offset() : tap->offset());
4720         field_off = elem()->isa_valuetype() ? field_offset() : tap->field_offset();
4721       } else if (tary->_elem->make_oopptr() != NULL && tary->_elem->make_oopptr()->isa_instptr() && below_centerline(ptr)) {
4722         // Result is non-flattened
4723         off = Offset(flattened_offset()).meet(Offset(tap->flattened_offset()));
4724         field_off = Offset::bottom;
4725       }
4726     }
4727 
4728     bool xk = false;
4729     switch (tap->ptr()) {
4730     case AnyNull:
4731     case TopPTR:
4732       // Compute new klass on demand, do not use tap->_klass
4733       if (below_centerline(this->_ptr)) {
4734         xk = this->_klass_is_exact;
4735       } else {
4736         xk = (tap->_klass_is_exact || this->_klass_is_exact);
4737       }
4738       return make(ptr, const_oop(), tary, lazy_klass, xk, off, field_off, instance_id, speculative, depth);
4739     case Constant: {
4740       ciObject* o = const_oop();
4741       if( _ptr == Constant ) {
4742         if( tap->const_oop() != NULL && !o->equals(tap->const_oop()) ) {
4743           xk = (klass() == tap->klass());
4744           ptr = NotNull;
4745           o = NULL;
4746           instance_id = InstanceBot;
4747         } else {
4748           xk = true;
4749         }
4750       } else if(above_centerline(_ptr)) {
4751         o = tap->const_oop();
4752         xk = true;
4753       } else {
4754         // Only precise for identical arrays
4755         xk = this->_klass_is_exact && (klass() == tap->klass());
4756       }
4757       return TypeAryPtr::make(ptr, o, tary, lazy_klass, xk, off, field_off, instance_id, speculative, depth);
4758     }
4759     case NotNull:
4760     case BotPTR:
4761       // Compute new klass on demand, do not use tap->_klass
4762       if (above_centerline(this->_ptr))
4763             xk = tap->_klass_is_exact;
4764       else  xk = (tap->_klass_is_exact & this->_klass_is_exact) &&
4765               (klass() == tap->klass()); // Only precise for identical arrays
4766       return TypeAryPtr::make(ptr, NULL, tary, lazy_klass, xk, off, field_off, instance_id, speculative, depth);
4767     default: ShouldNotReachHere();
4768     }
4769   }
4770 
4771   // All arrays inherit from Object class
4772   case InstPtr: {
4773     const TypeInstPtr *tp = t->is_instptr();
4774     Offset offset = meet_offset(tp->offset());
4775     PTR ptr = meet_ptr(tp->ptr());
4776     int instance_id = meet_instance_id(tp->instance_id());
4777     const TypePtr* speculative = xmeet_speculative(tp);
4778     int depth = meet_inline_depth(tp->inline_depth());
4779     switch (ptr) {
4780     case TopPTR:
4781     case AnyNull:                // Fall 'down' to dual of object klass
4782       // For instances when a subclass meets a superclass we fall
4783       // below the centerline when the superclass is exact. We need to
4784       // do the same here.
4785       if (tp->klass()->equals(ciEnv::current()->Object_klass()) && !tp->klass_is_exact() && !tp->flat_array()) {
4786         return TypeAryPtr::make(ptr, _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4787       } else {
4788         // cannot subclass, so the meet has to fall badly below the centerline
4789         ptr = NotNull;
4790         instance_id = InstanceBot;
4791         return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
4792       }
4793     case Constant:
4794     case NotNull:
4795     case BotPTR:                // Fall down to object klass
4796       // LCA is object_klass, but if we subclass from the top we can do better
4797       if (above_centerline(tp->ptr())) {
4798         // If 'tp'  is above the centerline and it is Object class
4799         // then we can subclass in the Java class hierarchy.
4800         // For instances when a subclass meets a superclass we fall
4801         // below the centerline when the superclass is exact. We need
4802         // to do the same here.
4803         if (tp->klass()->equals(ciEnv::current()->Object_klass()) && !tp->klass_is_exact() && !tp->flat_array()) {
4804           // that is, my array type is a subtype of 'tp' klass
4805           return make(ptr, (ptr == Constant ? const_oop() : NULL),
4806                       _ary, _klass, _klass_is_exact, offset, _field_offset, instance_id, speculative, depth);
4807         }
4808       }
4809       // The other case cannot happen, since t cannot be a subtype of an array.
4810       // The meet falls down to Object class below centerline.
4811       if( ptr == Constant )
4812          ptr = NotNull;
4813       instance_id = InstanceBot;
4814       return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass(), false, NULL, offset, false, instance_id, speculative, depth);
4815     default: typerr(t);
4816     }
4817   }
4818 
4819   case ValueType: {
4820     const TypeValueType* tv = t->is_valuetype();
4821     if (above_centerline(ptr())) {
4822       return TypeInstPtr::NOTNULL;
4823     } else {
4824       PTR ptr = this->_ptr;
4825       if (ptr == Constant) {
4826         ptr = NotNull;
4827       }
4828       return TypeInstPtr::make(ptr, ciEnv::current()->Object_klass());
4829     }
4830   }
4831   }
4832   return this;                  // Lint noise
4833 }

4953 const TypePtr* TypeAryPtr::add_field_offset_and_offset(intptr_t offset) const {
4954   int adj = 0;
4955   if (offset != Type::OffsetBot && offset != Type::OffsetTop) {
4956     const Type* elemtype = elem();
4957     if (elemtype->isa_valuetype()) {
4958       if (_offset.get() != OffsetBot && _offset.get() != OffsetTop) {
4959         adj = _offset.get();
4960         offset += _offset.get();
4961       }
4962       uint header = arrayOopDesc::base_offset_in_bytes(T_OBJECT);
4963       if (_field_offset.get() != OffsetBot && _field_offset.get() != OffsetTop) {
4964         offset += _field_offset.get();
4965         if (_offset.get() == OffsetBot || _offset.get() == OffsetTop) {
4966           offset += header;
4967         }
4968       }
4969       if (offset >= (intptr_t)header || offset < 0) {
4970         // Try to get the field of the value type array element we are pointing to
4971         ciKlass* arytype_klass = klass();
4972         ciValueArrayKlass* vak = arytype_klass->as_value_array_klass();
4973         ciValueKlass* vk = vak->element_klass()->as_value_klass();
4974         int shift = vak->log2_element_size();
4975         int mask = (1 << shift) - 1;
4976         intptr_t field_offset = ((offset - header) & mask);
4977         ciField* field = vk->get_field_by_offset(field_offset + vk->first_field_offset(), false);
4978         if (field == NULL) {
4979           // This may happen with nested AddP(base, AddP(base, base, offset), longcon(16))
4980           return add_offset(offset);
4981         } else {
4982           return with_field_offset(field_offset)->add_offset(offset - field_offset - adj);
4983         }
4984       }
4985     }
4986   }
4987   return add_offset(offset - adj);
4988 }

5229   if( this == t ) return this;  // Meeting same type-rep?
5230 
5231   // Current "this->_base" is OopPtr

5233

5417 //----------------------compute_klass------------------------------------------
5418 // Compute the defining klass for this class
5419 ciKlass* TypeAryPtr::compute_klass(DEBUG_ONLY(bool verify)) const {
5420   // Compute _klass based on element type.
5421   ciKlass* k_ary = NULL;
5422   const TypeAryPtr *tary;
5423   const Type* el = elem();
5424   if (el->isa_narrowoop()) {
5425     el = el->make_ptr();
5426   }
5427

5429   if (el->isa_instptr()) {

5431     k_ary = ciArrayKlass::make(el->is_oopptr()->klass());
5432   } else if (el->isa_valuetype()) {
5433     // If element type is TypeValueType::BOTTOM, value_klass() will be null.
5434     if (el->value_klass() != NULL) {
5435       k_ary = ciArrayKlass::make(el->value_klass());
5436     }
5437   } else if ((tary = el->isa_aryptr()) != NULL) {
5438     // Compute array klass from element klass
5439     ciKlass* k_elem = tary->klass();
5440     // If element type is something like bottom[], k_elem will be null.
5441     if (k_elem != NULL)
5442       k_ary = ciObjArrayKlass::make(k_elem);
5443   } else if ((el->base() == Type::Top) ||
5444              (el->base() == Type::Bottom)) {
5445     // element type of Bottom occurs from meet of basic type
5446     // and object; Top occurs when doing join on Bottom.
5447     // Leave k_ary at NULL.
5448   } else {
5449     // Cannot compute array klass directly from basic type,
5450     // since subtypes of TypeInt all have basic type T_INT.

5492     // active compilations.  However, the ciKlass which represents
5493     // this Type is *not* shared between compilations, so caching
5494     // this value would result in fetching a dangling pointer.
5495     //
5496     // Recomputing the underlying ciKlass for each request is
5497     // a bit less efficient than caching, but calls to
5498     // TypeAryPtr::OOPS->klass() are not common enough to matter.
5499     ((TypeAryPtr*)this)->_klass = k_ary;
5500     if (UseCompressedOops && k_ary != NULL && k_ary->is_obj_array_klass() &&
5501         offset() != 0 && offset() != arrayOopDesc::length_offset_in_bytes()) {
5502       ((TypeAryPtr*)this)->_is_ptr_to_narrowoop = true;
5503     }
5504   }
5505   return k_ary;

5549 //------------------------------xmeet------------------------------------------
5550 // Compute the MEET of two types, return a new Type object.
5551 const Type    *TypeKlassPtr::xmeet( const Type *t ) const {
5552   // Perform a fast test for common case; meeting the same types together.
5553   if( this == t ) return this;  // Meeting same type-rep?
5554 
5555   // Current "this->_base" is Pointer
5556   switch (t->base()) {          // switch on original type
5557 
5558   case Int:                     // Mixing ints & oops happens when javac
5559   case Long:                    // reuses local variables
5560   case FloatTop:
5561   case FloatCon:
5562   case FloatBot:
5563   case DoubleTop:
5564   case DoubleCon:
5565   case DoubleBot:
5566   case NarrowOop:
5567   case NarrowKlass:
5568   case Bottom:                  // Ye Olde Default
5569     return Type::BOTTOM;
5570   case Top:
5571     return this;
5572 
5573   default:                      // All else is a mistake
5574     typerr(t);
5575 
5576   case AnyPtr: {                // Meeting to AnyPtrs
5577     // Found an AnyPtr type vs self-KlassPtr type
5578     const TypePtr *tp = t->is_ptr();
5579     Offset offset = meet_offset(tp->offset());
5580     PTR ptr = meet_ptr(tp->ptr());
5581     switch (tp->ptr()) {
5582     case TopPTR:
5583       return this;
5584     case Null:
5585       if( ptr == Null ) return TypePtr::make(AnyPtr, ptr, offset, tp->speculative(), tp->inline_depth());
5586     case AnyNull:
5587       return make(ptr, klass(), offset, flat_array());
5588     case BotPTR:
5589     case NotNull:
5590       return TypePtr::make(AnyPtr, ptr, offset, tp->speculative(), tp->inline_depth());
5591     default: typerr(t);
5592     }
5593   }
5594 
5595   case RawPtr:
5596   case MetadataPtr:
5597   case OopPtr:
5598   case AryPtr:                  // Meet with AryPtr
5599   case InstPtr:                 // Meet with InstPtr
5600     return TypePtr::BOTTOM;
5601 
5602   //
5603   //             A-top         }
5604   //           /   |   \       }  Tops
5605   //       B-top A-any C-top   }
5606   //          | /  |  \ |      }  Any-nulls
5607   //       B-any   |   C-any   }
5608   //          |    |    |
5609   //       B-con A-con C-con   } constants; not comparable across classes
5610   //          |    |    |
5611   //       B-not   |   C-not   }
5612   //          | \  |  / |      }  not-nulls
5613   //       B-bot A-not C-bot   }
5614   //           \   |   /       }  Bottoms
5615   //             A-bot         }
5616   //
5617 
5618   case KlassPtr: {  // Meet two KlassPtr types
5619     const TypeKlassPtr *tkls = t->is_klassptr();
5620     Offset  off  = meet_offset(tkls->offset());
5621     PTR  ptr     = meet_ptr(tkls->ptr());
5622 
5623     if (klass() == NULL || tkls->klass() == NULL) {
5624       ciKlass* k = NULL;
5625       if (ptr == Constant) {
5626         k = (klass() == NULL) ? tkls->klass() : klass();
5627       }
5628       return make(ptr, k, off, false);
5629     }
5630 
5631     // Check for easy case; klasses are equal (and perhaps not loaded!)
5632     // If we have constants, then we created oops so classes are loaded
5633     // and we can handle the constants further down.  This case handles
5634     // not-loaded classes
5635     if (ptr != Constant && tkls->klass()->equals(klass()) && flat_array() == tkls->flat_array()) {
5636       return make(ptr, klass(), off, flat_array());
5637     }
5638 
5639     // Classes require inspection in the Java klass hierarchy.  Must be loaded.
5640     ciKlass* tkls_klass = tkls->klass();
5641     ciKlass* this_klass = this->klass();
5642     assert( tkls_klass->is_loaded(), "This class should have been loaded.");
5643     assert( this_klass->is_loaded(), "This class should have been loaded.");
5644     bool tkls_flat_array = tkls->flat_array();
5645     bool this_flat_array  = this->flat_array();
5646     bool flat_array = below_centerline(ptr) ? (this_flat_array && tkls_flat_array) : (this_flat_array || tkls_flat_array);
5647 
5648     // If 'this' type is above the centerline and is a superclass of the
5649     // other, we can treat 'this' as having the same type as the other.
5650     if ((above_centerline(this->ptr())) &&
5651         tkls_klass->is_subtype_of(this_klass)) {
5652       this_klass = tkls_klass;
5653     }
5654     // If 'tinst' type is above the centerline and is a superclass of the
5655     // other, we can treat 'tinst' as having the same type as the other.
5656     if ((above_centerline(tkls->ptr())) &&
5657         this_klass->is_subtype_of(tkls_klass)) {
5658       tkls_klass = this_klass;
5659     }
5660 
5661     // Check for classes now being equal
5662     if (tkls_klass->equals(this_klass)) {
5663       // If the klasses are equal, the constants may still differ.  Fall to
5664       // NotNull if they do (neither constant is NULL; that is a special case
5665       // handled elsewhere).
5666       if( ptr == Constant ) {
5667         if (this->_ptr == Constant && tkls->_ptr == Constant &&
5668             this->klass()->equals(tkls->klass()));
5669         else if (above_centerline(this->ptr()));
5670         else if (above_centerline(tkls->ptr()));
5671         else
5672           ptr = NotNull;
5673       }
5674       return make(ptr, this_klass, off, flat_array);
5675     } // Else classes are not equal
5676 
5677     // Since klasses are different, we require the LCA in the Java
5678     // class hierarchy - which means we have to fall to at least NotNull.
5679     if( ptr == TopPTR || ptr == AnyNull || ptr == Constant )
5680       ptr = NotNull;
5681     // Now we find the LCA of Java classes
5682     ciKlass* k = this_klass->least_common_ancestor(tkls_klass);
5683     return   make(ptr, k, off, k->is_valuetype() && k->flatten_array());
5684   } // End of case KlassPtr
5685 
5686   } // End of switch
5687   return this;                  // Return the double constant
5688 }

5696 //------------------------------get_con----------------------------------------
5697 intptr_t TypeKlassPtr::get_con() const {
5698   assert( _ptr == Null || _ptr == Constant, "" );
5699   assert(offset() >= 0, "");
5700 
5701   if (offset() != 0) {
5702     // After being ported to the compiler interface, the compiler no longer
5703     // directly manipulates the addresses of oops.  Rather, it only has a pointer
5704     // to a handle at compile time.  This handle is embedded in the generated
5705     // code and dereferenced at the time the nmethod is made.  Until that time,
5706     // it is not reasonable to do arithmetic with the addresses of oops (we don't
5707     // have access to the addresses!).  This does not seem to currently happen,
5708     // but this assertion here is to help prevent its occurence.
5709     tty->print_cr("Found oop constant with non-zero offset");
5710     ShouldNotReachHere();

5712 
5713   return (intptr_t)klass()->constant_encoding();

5770 //------------------------------make-------------------------------------------
5771 const TypeFunc* TypeFunc::make(ciMethod* method, bool is_osr_compilation) {
5772   Compile* C = Compile::current();
5773   const TypeFunc* tf = NULL;
5774   if (!is_osr_compilation) {
5775     tf = C->last_tf(method); // check cache
5776     if (tf != NULL)  return tf;  // The hit rate here is almost 50%.
5777   }
5778   // Value types are not passed/returned by reference, instead each field of
5779   // the value type is passed/returned as an argument. We maintain two views of
5780   // the argument/return list here: one based on the signature (with a value
5781   // type argument/return as a single slot), one based on the actual calling
5782   // convention (with a value type argument/return as a list of its fields).
5783   bool has_scalar_args = method->has_scalarized_args() && !is_osr_compilation;
5784   const TypeTuple* domain_sig = is_osr_compilation ? osr_domain() : TypeTuple::make_domain(method, false);
5785   const TypeTuple* domain_cc = has_scalar_args ? TypeTuple::make_domain(method, true) : domain_sig;
5786   ciSignature* sig = method->signature();
5787   bool has_scalar_ret = sig->returns_never_null() && sig->return_type()->as_value_klass()->can_be_returned_as_fields();
5788   const TypeTuple* range_sig = TypeTuple::make_range(sig, false);
5789   const TypeTuple* range_cc = has_scalar_ret ? TypeTuple::make_range(sig, true) : range_sig;
5790   tf = TypeFunc::make(domain_sig, domain_cc, range_sig, range_cc);
5791   if (!is_osr_compilation) {
5792     C->set_last_tf(method, tf);  // fill cache
5793   }
5794   return tf;
5795 }
