 107 void MemNode::dump_adr_type(const Node* mem, const TypePtr* adr_type, outputStream *st) {
 108   st->print(" @");
 109   if (adr_type == NULL) {
 110     st->print("NULL");
 111   } else {
 112     adr_type->dump_on(st);
 113     Compile* C = Compile::current();
 114     Compile::AliasType* atp = NULL;
 115     if (C->have_alias_type(adr_type))  atp = C->alias_type(adr_type);
 116     if (atp == NULL)
 117       st->print(", idx=?\?;");
 118     else if (atp->index() == Compile::AliasIdxBot)
 119       st->print(", idx=Bot;");
 120     else if (atp->index() == Compile::AliasIdxTop)
 121       st->print(", idx=Top;");
 122     else if (atp->index() == Compile::AliasIdxRaw)
 123       st->print(", idx=Raw;");
 124     else {
 125       ciField* field = atp->field();
 126       if (field) {
 127         st->print(", name=");
 128         field->print_name_on(st);
 129       }
 130       st->print(", idx=%d;", atp->index());
 131     }
 132   }
 133 }

 139 Node *MemNode::optimize_simple_memory_chain(Node *mchain, const TypeOopPtr *t_oop, Node *load, PhaseGVN *phase) {
 140   assert((t_oop != NULL), "sanity");
 141   bool is_instance = t_oop->is_known_instance_field();
 142   bool is_boxed_value_load = t_oop->is_ptr_to_boxed_value() &&
 143                              (load != NULL) && load->is_Load() &&
 144                              (phase->is_IterGVN() != NULL);
 145   if (!(is_instance || is_boxed_value_load))
 146     return mchain;  // don't try to optimize non-instance types
 147   uint instance_id = t_oop->instance_id();
 148   Node *start_mem = phase->C->start()->proj_out_or_null(TypeFunc::Memory);
 149   Node *prev = NULL;
 150   Node *result = mchain;
 151   while (prev != result) {
 152     prev = result;
 153     if (result == start_mem)
 154       break;  // hit one of our sentinels
 155     // skip over a call which does not affect this memory slice
 156     if (result->is_Proj() && result->as_Proj()->_con == TypeFunc::Memory) {
 157       Node *proj_in = result->in(0);
 158       if (proj_in->is_Allocate() && proj_in->_idx == instance_id) {
 159         break;  // hit one of our sentinels
 160       } else if (proj_in->is_Call()) {
 161         // ArrayCopyNodes processed here as well
 162         CallNode *call = proj_in->as_Call();
 163         if (!call->may_modify(t_oop, phase)) { // returns false for instances
 164           result = call->in(TypeFunc::Memory);
 165         }
 166       } else if (proj_in->is_Initialize()) {
 167         AllocateNode* alloc = proj_in->as_Initialize()->allocation();
 168         // Stop if this is the initialization for the object instance which
 169         // contains this memory slice, otherwise skip over it.
 170         if ((alloc == NULL) || (alloc->_idx == instance_id)) {
 171           break;
 172         }
 173         if (is_instance) {
 174           result = proj_in->in(TypeFunc::Memory);
 175         } else if (is_boxed_value_load) {
 176           Node* klass = alloc->in(AllocateNode::KlassNode);
 177           const TypeKlassPtr* tklass = phase->type(klass)->is_klassptr();
 178           if (tklass->klass_is_exact() && !tklass->klass()->equals(t_oop->klass())) {
 179             result = proj_in->in(TypeFunc::Memory); // not related allocation
 180           }
 181         }
 182       } else if (proj_in->is_MemBar()) {
 183         ArrayCopyNode* ac = NULL;
 184         if (ArrayCopyNode::may_modify(t_oop, proj_in->as_MemBar(), phase, ac)) {
 185           break;
 186         }
 187         result = proj_in->in(TypeFunc::Memory);
 188       } else {
 189         assert(false, "unexpected projection");
 190       }
 191     } else if (result->is_ClearArray()) {
 192       if (!is_instance || !ClearArrayNode::step_through(&result, instance_id, phase)) {
 193         // Can not bypass initialization of the instance
 194         // we are looking for.
 195         break;
 196       }
 197       // Otherwise skip it (the call updated 'result' value).
 198     } else if (result->is_MergeMem()) {
 199       result = step_through_mergemem(phase, result->as_MergeMem(), t_oop, NULL, tty);
 200     }
 201   }
 202   return result;
 203 }

 205 Node *MemNode::optimize_memory_chain(Node *mchain, const TypePtr *t_adr, Node *load, PhaseGVN *phase) {
 206   const TypeOopPtr* t_oop = t_adr->isa_oopptr();
 207   if (t_oop == NULL)
 208     return mchain;  // don't try to optimize non-oop types
 209   Node* result = optimize_simple_memory_chain(mchain, t_oop, load, phase);
 210   bool is_instance = t_oop->is_known_instance_field();
 211   PhaseIterGVN *igvn = phase->is_IterGVN();
 212   if (is_instance && igvn != NULL && result->is_Phi()) {
 213     PhiNode *mphi = result->as_Phi();
 214     assert(mphi->bottom_type() == Type::MEMORY, "memory phi required");
 215     const TypePtr *t = mphi->adr_type();
 216     if (t == TypePtr::BOTTOM || t == TypeRawPtr::BOTTOM ||
 217         (t->isa_oopptr() && !t->is_oopptr()->is_known_instance() &&
 218          t->is_oopptr()->cast_to_exactness(true)
 219            ->is_oopptr()->cast_to_ptr_type(t_oop->ptr())
 220             ->is_oopptr()->cast_to_instance_id(t_oop->instance_id()) == t_oop)) {
 221       // clone the Phi with our address type
 222       result = mphi->split_out_instance(t_adr, igvn);
 223     } else {
 224       assert(phase->C->get_alias_index(t) == phase->C->get_alias_index(t_adr), "correct memory chain");
 225     }
 226   }
 227   return result;
 228 }

 230 static Node *step_through_mergemem(PhaseGVN *phase, MergeMemNode *mmem,  const TypePtr *tp, const TypePtr *adr_check, outputStream *st) {
 231   uint alias_idx = phase->C->get_alias_index(tp);
 232   Node *mem = mmem;
 233 #ifdef ASSERT
 234   {
 235     // Check that current type is consistent with the alias index used during graph construction
 236     assert(alias_idx >= Compile::AliasIdxRaw, "must not be a bad alias_idx");
 237     bool consistent =  adr_check == NULL || adr_check->empty() ||
 238                        phase->C->must_alias(adr_check, alias_idx );
 239     // Sometimes dead array references collapse to a[-1], a[-2], or a[-3]
 240     if( !consistent && adr_check != NULL && !adr_check->empty() &&
 241         tp->isa_aryptr() &&        tp->offset() == Type::OffsetBot &&
 242         adr_check->isa_aryptr() && adr_check->offset() != Type::OffsetBot &&
 243         ( adr_check->offset() == arrayOopDesc::length_offset_in_bytes() ||
 244           adr_check->offset() == oopDesc::klass_offset_in_bytes() ||
 245           adr_check->offset() == oopDesc::mark_offset_in_bytes() ) ) {
 246       // don't assert if it is dead code.
 247       consistent = true;
 248     }
 249     if( !consistent ) {
 250       st->print("alias_idx==%d, adr_check==", alias_idx);
 251       if( adr_check == NULL ) {
 252         st->print("NULL");
 253       } else {
 254         adr_check->dump();
 255       }
 256       st->cr();
 257       print_alias_types();
 258       assert(consistent, "adr_check must match alias idx");
 259     }
 260   }
 261 #endif
 262   // TypeOopPtr::NOTNULL+any is an OOP with unknown offset - generally
 263   // means an array I have not precisely typed yet.  Do not do any
 264   // alias stuff with it any time soon.
 265   const TypeOopPtr *toop = tp->isa_oopptr();
 266   if( tp->base() != Type::AnyPtr &&
 267       !(toop &&
 268         toop->klass() != NULL &&
 269         toop->klass()->is_java_lang_Object() &&
 270         toop->offset() == Type::OffsetBot) ) {
 271     // compress paths and change unreachable cycles to TOP
 272     // If not, we can update the input infinitely along a MergeMem cycle
 273     // Equivalent code in PhiNode::Ideal
 274     Node* m  = phase->transform(mmem);
 275     // If transformed to a MergeMem, get the desired slice
 276     // Otherwise the returned node represents memory for every slice
 277     mem = (m->is_MergeMem())? m->as_MergeMem()->memory_at(alias_idx) : m;
 278     // Update input if it is progress over what we have now
 279   }
 280   return mem;
 281 }

 283 //--------------------------Ideal_common---------------------------------------
 284 // Look for degenerate control and memory inputs.  Bypass MergeMem inputs.
 285 // Unhook non-raw memories from complete (macro-expanded) initializations.
 286 Node *MemNode::Ideal_common(PhaseGVN *phase, bool can_reshape) {
 287   // If our control input is a dead region, kill all below the region
 288   Node *ctl = in(MemNode::Control);
 289   if (ctl && remove_dead_region(phase, can_reshape))
 290     return this;
 291   ctl = in(MemNode::Control);
 292   // Don't bother trying to transform a dead node
 293   if (ctl && ctl->is_top())  return NodeSentinel;
 294 
 295   PhaseIterGVN *igvn = phase->is_IterGVN();
 296   // Wait if control on the worklist.
 297   if (ctl && can_reshape && igvn != NULL) {
 298     Node* bol = NULL;
 299     Node* cmp = NULL;
 300     if (ctl->in(0)->is_If()) {
 301       assert(ctl->is_IfTrue() || ctl->is_IfFalse(), "sanity");
 302       bol = ctl->in(0)->in(1);
 303       if (bol->is_Bool())
 304         cmp = ctl->in(0)->in(1)->in(1);
 305     }
 306     if (igvn->_worklist.member(ctl) ||
 307         (bol != NULL && igvn->_worklist.member(bol)) ||
 308         (cmp != NULL && igvn->_worklist.member(cmp)) ) {
 309       // This control path may be dead.
 310       // Delay this memory node transformation until the control is processed.
 311       phase->is_IterGVN()->_worklist.push(this);
 312       return NodeSentinel; // caller will return NULL
 313     }
 314   }
 315   // Ignore if memory is dead, or self-loop
 316   Node *mem = in(MemNode::Memory);
 317   if (phase->type( mem ) == Type::TOP) return NodeSentinel; // caller will return NULL
 318   assert(mem != this, "dead loop in MemNode::Ideal");
 319 
 320   if (can_reshape && igvn != NULL && igvn->_worklist.member(mem)) {
 321     // This memory slice may be dead.
 322     // Delay this mem node transformation until the memory is processed.
 323     phase->is_IterGVN()->_worklist.push(this);
 324     return NodeSentinel; // caller will return NULL
 325   }
 326 
 327   Node *address = in(MemNode::Address);
 328   const Type *t_adr = phase->type(address);
 329   if (t_adr == Type::TOP)              return NodeSentinel; // caller will return NULL
 330 
 331   if (can_reshape && is_unsafe_access() && (t_adr == TypePtr::NULL_PTR)) {
 332     // Unsafe off-heap access with zero address. Remove access and other control users
 333     // to not confuse optimizations and add a HaltNode to fail if this is ever executed.
 334     assert(ctl != NULL, "unsafe accesses should be control dependent");
 335     for (DUIterator_Fast imax, i = ctl->fast_outs(imax); i < imax; i++) {
 336       Node* u = ctl->fast_out(i);
 337       if (u != ctl) {
 338         igvn->rehash_node_delayed(u);
 339         int nb = u->replace_edge(ctl, phase->C->top());
 340         --i, imax -= nb;
 341       }
 342     }
 343     Node* frame = igvn->transform(new ParmNode(phase->C->start(), TypeFunc::FramePtr));
 344     Node* halt = igvn->transform(new HaltNode(ctl, frame, "unsafe off-heap access with zero address"));
 345     phase->C->root()->add_req(halt);
 346     return this;
 347   }
 348 
 349   if (can_reshape && igvn != NULL &&
 350       (igvn->_worklist.member(address) ||
 351        (igvn->_worklist.size() > 0 && t_adr != adr_type())) ) {
 352     // The address's base and type may change when the address is processed.
 353     // Delay this mem node transformation until the address is processed.
 354     phase->is_IterGVN()->_worklist.push(this);
 355     return NodeSentinel; // caller will return NULL
 356   }
 357 
 358   // Do NOT remove or optimize the next lines: ensure a new alias index
 359   // is allocated for an oop pointer type before Escape Analysis.
 360   // Note: C++ will not remove it since the call has side effect.
 361   if (t_adr->isa_oopptr()) {
 362     int alias_idx = phase->C->get_alias_index(t_adr->is_ptr());
 363   }
 364 
 365   Node* base = NULL;
 366   if (address->is_AddP()) {
 367     base = address->in(AddPNode::Base);
 368   }
 369   if (base != NULL && phase->type(base)->higher_equal(TypePtr::NULL_PTR) &&
 370       !t_adr->isa_rawptr()) {
 371     // Note: raw address has TOP base and top->higher_equal(TypePtr::NULL_PTR) is true.
 372     // Skip this node optimization if its address has TOP base.
 373     return NodeSentinel; // caller will return NULL
 374   }
 375 
 376   // Avoid independent memory operations
 377   Node* old_mem = mem;
 378 
 379   // The code which unhooks non-raw memories from complete (macro-expanded)
 380   // initializations was removed. After macro-expansion all stores catched
 381   // by Initialize node became raw stores and there is no information
 382   // which memory slices they modify. So it is unsafe to move any memory
 383   // operation above these stores. Also in most cases hooked non-raw memories
 384   // were already unhooked by using information from detect_ptr_independence()
 385   // and find_previous_store().
 386 
 387   if (mem->is_MergeMem()) {
 388     MergeMemNode* mmem = mem->as_MergeMem();
 389     const TypePtr *tp = t_adr->is_ptr();
 390 
 391     mem = step_through_mergemem(phase, mmem, tp, adr_type(), tty);
 392   }
 393 
 394   if (mem != old_mem) {
 395     set_req(MemNode::Memory, mem);
 396     if (can_reshape && old_mem->outcnt() == 0 && igvn != NULL) {
 397       igvn->_worklist.push(old_mem);
 398     }
 399     if (phase->type(mem) == Type::TOP) return NodeSentinel;
 400     return this;
 401   }
 402 
 403   // let the subclass continue analyzing...
 404   return NULL;
 405 }

 407 // Helper function for proving some simple control dominations.
 408 // Attempt to prove that all control inputs of 'dom' dominate 'sub'.
 409 // Already assumes that 'dom' is available at 'sub', and that 'sub'
 410 // is not a constant (dominated by the method's StartNode).
 411 // Used by MemNode::find_previous_store to prove that the
 412 // control input of a memory operation predates (dominates)
 413 // an allocation it wants to look past.
 414 bool MemNode::all_controls_dominate(Node* dom, Node* sub) {
 415   if (dom == NULL || dom->is_top() || sub == NULL || sub->is_top())
 416     return false; // Conservative answer for dead code
 417 
 418   // Check 'dom'. Skip Proj and CatchProj nodes.
 419   dom = dom->find_exact_control(dom);
 420   if (dom == NULL || dom->is_top())
 421     return false; // Conservative answer for dead code
 422 
 423   if (dom == sub) {
 424     // For the case when, for example, 'sub' is Initialize and the original
 425     // 'dom' is Proj node of the 'sub'.
 426     return false;
 427   }
 428 
 429   if (dom->is_Con() || dom->is_Start() || dom->is_Root() || dom == sub)
 430     return true;
 431 
 432   // 'dom' dominates 'sub' if its control edge and control edges
 433   // of all its inputs dominate or equal to sub's control edge.
 434 
 435   // Currently 'sub' is either Allocate, Initialize or Start nodes.
 436   // Or Region for the check in LoadNode::Ideal();
 437   // 'sub' should have sub->in(0) != NULL.
 438   assert(sub->is_Allocate() || sub->is_Initialize() || sub->is_Start() ||
 439          sub->is_Region() || sub->is_Call(), "expecting only these nodes");
 440 
 441   // Get control edge of 'sub'.
 442   Node* orig_sub = sub;
 443   sub = sub->find_exact_control(sub->in(0));
 444   if (sub == NULL || sub->is_top())
 445     return false; // Conservative answer for dead code
 446 
 447   assert(sub->is_CFG(), "expecting control");
 448 
 449   if (sub == dom)
 450     return true;
 451 
 452   if (sub->is_Start() || sub->is_Root())
 453     return false;
 454 
 455   {
 456     // Check all control edges of 'dom'.
 457 
 458     ResourceMark rm;
 459     Arena* arena = Thread::current()->resource_area();
 460     Node_List nlist(arena);
 461     Unique_Node_List dom_list(arena);
 462 
 463     dom_list.push(dom);
 464     bool only_dominating_controls = false;
 465 
 466     for (uint next = 0; next < dom_list.size(); next++) {
 467       Node* n = dom_list.at(next);
 468       if (n == orig_sub)
 469         return false; // One of dom's inputs dominated by sub.
 470       if (!n->is_CFG() && n->pinned()) {
 471         // Check only own control edge for pinned non-control nodes.
 472         n = n->find_exact_control(n->in(0));
 473         if (n == NULL || n->is_top())
 474           return false; // Conservative answer for dead code
 475         assert(n->is_CFG(), "expecting control");
 476         dom_list.push(n);
 477       } else if (n->is_Con() || n->is_Start() || n->is_Root()) {
 478         only_dominating_controls = true;
 479       } else if (n->is_CFG()) {
 480         if (n->dominates(sub, nlist))
 481           only_dominating_controls = true;
 482         else
 483           return false;
 484       } else {
 485         // First, own control edge.
 486         Node* m = n->find_exact_control(n->in(0));
 487         if (m != NULL) {
 488           if (m->is_top())
 489             return false; // Conservative answer for dead code
 490           dom_list.push(m);
 491         }
 492         // Now, the rest of edges.
 493         uint cnt = n->req();
 494         for (uint i = 1; i < cnt; i++) {
 495           m = n->find_exact_control(n->in(i));
 496           if (m == NULL || m->is_top())
 497             continue;
 498           dom_list.push(m);
 499         }
 500       }
 501     }
 502     return only_dominating_controls;
 503   }
 504 }

 506 //---------------------detect_ptr_independence---------------------------------
 507 // Used by MemNode::find_previous_store to prove that two base
 508 // pointers are never equal.
 509 // The pointers are accompanied by their associated allocations,
 510 // if any, which have been previously discovered by the caller.
 511 bool MemNode::detect_ptr_independence(Node* p1, AllocateNode* a1,
 512                                       Node* p2, AllocateNode* a2,
 513                                       PhaseTransform* phase) {
 514   // Attempt to prove that these two pointers cannot be aliased.
 515   // They may both manifestly be allocations, and they should differ.
 516   // Or, if they are not both allocations, they can be distinct constants.
 517   // Otherwise, one is an allocation and the other a pre-existing value.
 518   if (a1 == NULL && a2 == NULL) {           // neither an allocation
 519     return (p1 != p2) && p1->is_Con() && p2->is_Con();
 520   } else if (a1 != NULL && a2 != NULL) {    // both allocations
 521     return (a1 != a2);
 522   } else if (a1 != NULL) {                  // one allocation a1
 523     // (Note:  p2->is_Con implies p2->in(0)->is_Root, which dominates.)
 524     return all_controls_dominate(p2, a1);
 525   } else { //(a2 != NULL)                   // one allocation a2
 526     return all_controls_dominate(p1, a2);
 527   }
 528   return false;
 529 }

 532 // Find an arraycopy that must have set (can_see_stored_value=true) or
 533 // could have set (can_see_stored_value=false) the value for this load
 534 Node* LoadNode::find_previous_arraycopy(PhaseTransform* phase, Node* ld_alloc, Node*& mem, bool can_see_stored_value) const {
 535   if (mem->is_Proj() && mem->in(0) != NULL && (mem->in(0)->Opcode() == Op_MemBarStoreStore ||
 536                                                mem->in(0)->Opcode() == Op_MemBarCPUOrder)) {
 537     Node* mb = mem->in(0);
 538     if (mb->in(0) != NULL && mb->in(0)->is_Proj() &&
 539         mb->in(0)->in(0) != NULL && mb->in(0)->in(0)->is_ArrayCopy()) {
 540       ArrayCopyNode* ac = mb->in(0)->in(0)->as_ArrayCopy();
 541       if (ac->is_clonebasic()) {
 542         AllocateNode* alloc = AllocateNode::Ideal_allocation(ac->in(ArrayCopyNode::Dest), phase);
 543         if (alloc != NULL && alloc == ld_alloc) {
 544           return ac;
 545         }
 546       }
 547     }
 548   } else if (mem->is_Proj() && mem->in(0) != NULL && mem->in(0)->is_ArrayCopy()) {
 549     ArrayCopyNode* ac = mem->in(0)->as_ArrayCopy();
 550 
 551     if (ac->is_arraycopy_validated() ||
 552         ac->is_copyof_validated() ||
 553         ac->is_copyofrange_validated()) {
 554       Node* ld_addp = in(MemNode::Address);
 555       if (ld_addp->is_AddP()) {
 556         Node* ld_base = ld_addp->in(AddPNode::Address);
 557         Node* ld_offs = ld_addp->in(AddPNode::Offset);
 558 
 559         Node* dest = ac->in(ArrayCopyNode::Dest);
 560 
 561         if (dest == ld_base) {
 562           const TypeX *ld_offs_t = phase->type(ld_offs)->isa_intptr_t();
 563           if (ac->modifies(ld_offs_t->_lo, ld_offs_t->_hi, phase, can_see_stored_value)) {
 564             return ac;
 565           }
 566           if (!can_see_stored_value) {
 567             mem = ac->in(TypeFunc::Memory);
 568           }
 569         }
 570       }
 571     }
 572   }
 573   return NULL;

 576 // The logic for reordering loads and stores uses four steps:
 577 // (a) Walk carefully past stores and initializations which we
 578 //     can prove are independent of this load.
 579 // (b) Observe that the next memory state makes an exact match
 580 //     with self (load or store), and locate the relevant store.
 581 // (c) Ensure that, if we were to wire self directly to the store,
 582 //     the optimizer would fold it up somehow.
 583 // (d) Do the rewiring, and return, depending on some other part of
 584 //     the optimizer to fold up the load.
 585 // This routine handles steps (a) and (b).  Steps (c) and (d) are
 586 // specific to loads and stores, so they are handled by the callers.
 587 // (Currently, only LoadNode::Ideal has steps (c), (d).  More later.)
 588 //
 589 Node* MemNode::find_previous_store(PhaseTransform* phase) {
 590   Node*         ctrl   = in(MemNode::Control);
 591   Node*         adr    = in(MemNode::Address);
 592   intptr_t      offset = 0;
 593   Node*         base   = AddPNode::Ideal_base_and_offset(adr, phase, offset);
 594   AllocateNode* alloc  = AllocateNode::Ideal_allocation(base, phase);
 595 
 596   if (offset == Type::OffsetBot)
 597     return NULL;            // cannot unalias unless there are precise offsets
 598 
 599   const bool adr_maybe_raw = check_if_adr_maybe_raw(adr);
 600   const TypeOopPtr *addr_t = adr->bottom_type()->isa_oopptr();
 601 
 602   intptr_t size_in_bytes = memory_size();
 603 
 604   Node* mem = in(MemNode::Memory);   // start searching here...
 605 
 606   int cnt = 50;             // Cycle limiter
 607   for (;;) {                // While we can dance past unrelated stores...
 608     if (--cnt < 0)  break;  // Caught in cycle or a complicated dance?
 609 
 610     Node* prev = mem;
 611     if (mem->is_Store()) {
 612       Node* st_adr = mem->in(MemNode::Address);
 613       intptr_t st_offset = 0;
 614       Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_offset);
 615       if (st_base == NULL)
 616         break;              // inscrutable pointer
 617 
 618       // For raw accesses it's not enough to prove that constant offsets don't intersect.
 619       // We need the bases to be the equal in order for the offset check to make sense.
 620       if ((adr_maybe_raw || check_if_adr_maybe_raw(st_adr)) && st_base != base) {
 621         break;
 622       }
 623 
 624       if (st_offset != offset && st_offset != Type::OffsetBot) {
 625         const int MAX_STORE = BytesPerLong;
 626         if (st_offset >= offset + size_in_bytes ||
 627             st_offset <= offset - MAX_STORE ||
 628             st_offset <= offset - mem->as_Store()->memory_size()) {
 629           // Success:  The offsets are provably independent.
 630           // (You may ask, why not just test st_offset != offset and be done?
 631           // The answer is that stores of different sizes can co-exist
 632           // in the same sequence of RawMem effects.  We sometimes initialize
 633           // a whole 'tile' of array elements with a single jint or jlong.)
 634           mem = mem->in(MemNode::Memory);
 635           continue;           // (a) advance through independent store memory
 636         }
 637       }
 638       if (st_base != base &&
 639           detect_ptr_independence(base, alloc,
 640                                   st_base,
 641                                   AllocateNode::Ideal_allocation(st_base, phase),
 642                                   phase)) {
 643         // Success:  The bases are provably independent.
 644         mem = mem->in(MemNode::Memory);
 645         continue;           // (a) advance through independent store memory
 646       }
 647 
 648       // (b) At this point, if the bases or offsets do not agree, we lose,
 649       // since we have not managed to prove 'this' and 'mem' independent.
 650       if (st_base == base && st_offset == offset) {
 651         return mem;         // let caller handle steps (c), (d)
 652       }
 653 
 654     } else if (mem->is_Proj() && mem->in(0)->is_Initialize()) {
 655       InitializeNode* st_init = mem->in(0)->as_Initialize();
 656       AllocateNode*  st_alloc = st_init->allocation();
 657       if (st_alloc == NULL)
 658         break;              // something degenerated
 659       bool known_identical = false;
 660       bool known_independent = false;
 661       if (alloc == st_alloc)
 662         known_identical = true;
 663       else if (alloc != NULL)
 664         known_independent = true;
 665       else if (all_controls_dominate(this, st_alloc))
 666         known_independent = true;
 667 
 668       if (known_independent) {
 669         // The bases are provably independent: Either they are
 670         // manifestly distinct allocations, or else the control
 671         // of this load dominates the store's allocation.
 672         int alias_idx = phase->C->get_alias_index(adr_type());
 673         if (alias_idx == Compile::AliasIdxRaw) {
 674           mem = st_alloc->in(TypeFunc::Memory);
 675         } else {
 676           mem = st_init->memory(alias_idx);
 677         }
 678         continue;           // (a) advance through independent store memory
 679       }
 680 
 681       // (b) at this point, if we are not looking at a store initializing
 682       // the same allocation we are loading from, we lose.
 683       if (known_identical) {
 684         // From caller, can_see_stored_value will consult find_captured_store.
 685         return mem;         // let caller handle steps (c), (d)
 686       }
 687 
 688     } else if (find_previous_arraycopy(phase, alloc, mem, false) != NULL) {
 689       if (prev != mem) {
 690         // Found an arraycopy but it doesn't affect that load
 691         continue;
 692       }
 693       // Found an arraycopy that may affect that load
 694       return mem;
 695     } else if (addr_t != NULL && addr_t->is_known_instance_field()) {
 696       // Can't use optimize_simple_memory_chain() since it needs PhaseGVN.
 697       if (mem->is_Proj() && mem->in(0)->is_Call()) {
 698         // ArrayCopyNodes processed here as well.
 699         CallNode *call = mem->in(0)->as_Call();
 700         if (!call->may_modify(addr_t, phase)) {
 701           mem = call->in(TypeFunc::Memory);
 702           continue;         // (a) advance through independent call memory
 703         }
 704       } else if (mem->is_Proj() && mem->in(0)->is_MemBar()) {
 705         ArrayCopyNode* ac = NULL;
 706         if (ArrayCopyNode::may_modify(addr_t, mem->in(0)->as_MemBar(), phase, ac)) {
 707           break;
 708         }
 709         mem = mem->in(0)->in(TypeFunc::Memory);
 710         continue;           // (a) advance through independent MemBar memory
 711       } else if (mem->is_ClearArray()) {
 712         if (ClearArrayNode::step_through(&mem, (uint)addr_t->instance_id(), phase)) {
 713           // (the call updated 'mem' value)
 714           continue;         // (a) advance through independent allocation memory
 715         } else {
 716           // Can not bypass initialization of the instance
 717           // we are looking for.
 718           return mem;
 719         }
 720       } else if (mem->is_MergeMem()) {
 721         int alias_idx = phase->C->get_alias_index(adr_type());
 722         mem = mem->as_MergeMem()->memory_at(alias_idx);
 723         continue;           // (a) advance through independent MergeMem memory
 724       }
 725     }
 726 
 727     // Unless there is an explicit 'continue', we must bail out here,
 728     // because 'mem' is an inscrutable memory state (e.g., a call).
 729     break;
 730   }
 731 
 732   return NULL;              // bail out
 733 }

 735 //----------------------calculate_adr_type-------------------------------------
 736 // Helper function.  Notices when the given type of address hits top or bottom.
 737 // Also, asserts a cross-check of the type against the expected address type.
 738 const TypePtr* MemNode::calculate_adr_type(const Type* t, const TypePtr* cross_check) {
 739   if (t == Type::TOP)  return NULL; // does not touch memory any more?
 740   #ifdef ASSERT
 741   if (!VerifyAliases || VMError::is_error_reported() || Node::in_dump())  cross_check = NULL;
 742   #endif
 743   const TypePtr* tp = t->isa_ptr();
 744   if (tp == NULL) {
 745     assert(cross_check == NULL || cross_check == TypePtr::BOTTOM, "expected memory type must be wide");
 746     return TypePtr::BOTTOM;           // touches lots of memory
 747   } else {
 748     #ifdef ASSERT
 749     // %%%% [phh] We don't check the alias index if cross_check is
 750     //            TypeRawPtr::BOTTOM.  Needs to be investigated.
 751     if (cross_check != NULL &&
 752         cross_check != TypePtr::BOTTOM &&
 753         cross_check != TypeRawPtr::BOTTOM) {
 754       // Recheck the alias index, to see if it has changed (due to a bug).
 755       Compile* C = Compile::current();
 756       assert(C->get_alias_index(cross_check) == C->get_alias_index(tp),

 758       // The type of the address must be contained in the adr_type,
 759       // disregarding "null"-ness.
 760       // (We make an exception for TypeRawPtr::BOTTOM, which is a bit bucket.)

 763              "real address must not escape from expected memory type");
 764     }
 765     #endif
 766     return tp;
 767   }

 926 // Is the value loaded previously stored by an arraycopy? If so return
 927 // a load node that reads from the source array so we may be able to
 928 // optimize out the ArrayCopy node later.
 929 Node* LoadNode::can_see_arraycopy_value(Node* st, PhaseGVN* phase) const {
 930   Node* ld_adr = in(MemNode::Address);
 931   intptr_t ld_off = 0;
 932   AllocateNode* ld_alloc = AllocateNode::Ideal_allocation(ld_adr, phase, ld_off);
 933   Node* ac = find_previous_arraycopy(phase, ld_alloc, st, true);
 934   if (ac != NULL) {
 935     assert(ac->is_ArrayCopy(), "what kind of node can this be?");
 936 
 937     Node* mem = ac->in(TypeFunc::Memory);
 938     Node* ctl = ac->in(0);
 939     Node* src = ac->in(ArrayCopyNode::Src);
 940 
 941     if (!ac->as_ArrayCopy()->is_clonebasic() && !phase->type(src)->isa_aryptr()) {
 942       return NULL;
 943     }
 944 
 945     LoadNode* ld = clone()->as_Load();
 946     Node* addp = in(MemNode::Address)->clone();
 947     if (ac->as_ArrayCopy()->is_clonebasic()) {
 948       assert(ld_alloc != NULL, "need an alloc");
 949       assert(addp->is_AddP(), "address must be addp");
 950       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 951       assert(bs->step_over_gc_barrier(addp->in(AddPNode::Base)) == bs->step_over_gc_barrier(ac->in(ArrayCopyNode::Dest)), "strange pattern");
 952       assert(bs->step_over_gc_barrier(addp->in(AddPNode::Address)) == bs->step_over_gc_barrier(ac->in(ArrayCopyNode::Dest)), "strange pattern");
 953       addp->set_req(AddPNode::Base, src);
 954       addp->set_req(AddPNode::Address, src);
 955     } else {
 956       assert(ac->as_ArrayCopy()->is_arraycopy_validated() ||
 957              ac->as_ArrayCopy()->is_copyof_validated() ||
 958              ac->as_ArrayCopy()->is_copyofrange_validated(), "only supported cases");
 959       assert(addp->in(AddPNode::Base) == addp->in(AddPNode::Address), "should be");
 960       addp->set_req(AddPNode::Base, src);
 961       addp->set_req(AddPNode::Address, src);
 962 
 963       const TypeAryPtr* ary_t = phase->type(in(MemNode::Address))->isa_aryptr();
 964       BasicType ary_elem  = ary_t->klass()->as_array_klass()->element_type()->basic_type();
 965       uint header = arrayOopDesc::base_offset_in_bytes(ary_elem);
 966       uint shift  = exact_log2(type2aelembytes(ary_elem));
 967       if (ary_t->klass()->is_value_array_klass()) {
 968         ciValueArrayKlass* vak = ary_t->klass()->as_value_array_klass();
 969         shift = vak->log2_element_size();
 970       }
 971 
 972       Node* diff = phase->transform(new SubINode(ac->in(ArrayCopyNode::SrcPos), ac->in(ArrayCopyNode::DestPos)));
 973 #ifdef _LP64
 974       diff = phase->transform(new ConvI2LNode(diff));
 975 #endif
 976       diff = phase->transform(new LShiftXNode(diff, phase->intcon(shift)));
 977 
 978       Node* offset = phase->transform(new AddXNode(addp->in(AddPNode::Offset), diff));
 979       addp->set_req(AddPNode::Offset, offset);
 980     }
 981     addp = phase->transform(addp);
 982 #ifdef ASSERT
 983     const TypePtr* adr_type = phase->type(addp)->is_ptr();
 984     ld->_adr_type = adr_type;
 985 #endif
 986     ld->set_req(MemNode::Address, addp);
 987     ld->set_req(0, ctl);
 988     ld->set_req(MemNode::Memory, mem);
 989     // load depends on the tests that validate the arraycopy
 990     ld->_control_dependency = UnknownControl;
 991     return ld;
 992   }
 993   return NULL;
 994 }

 997 //---------------------------can_see_stored_value------------------------------
 998 // This routine exists to make sure this set of tests is done the same
 999 // everywhere.  We need to make a coordinated change: first LoadNode::Ideal
1000 // will change the graph shape in a way which makes memory alive twice at the
1001 // same time (uses the Oracle model of aliasing), then some
1002 // LoadXNode::Identity will fold things back to the equivalence-class model
1003 // of aliasing.
1004 Node* MemNode::can_see_stored_value(Node* st, PhaseTransform* phase) const {
1005   Node* ld_adr = in(MemNode::Address);
1006   intptr_t ld_off = 0;
1007   Node* ld_base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ld_off);
1008   Node* ld_alloc = AllocateNode::Ideal_allocation(ld_base, phase);
1009   const TypeInstPtr* tp = phase->type(ld_adr)->isa_instptr();
1010   Compile::AliasType* atp = (tp != NULL) ? phase->C->alias_type(tp) : NULL;
1011   // This is more general than load from boxing objects.
1012   if (skip_through_membars(atp, tp, phase->C->eliminate_boxing())) {
1013     uint alias_idx = atp->index();
1014     bool final = !atp->is_rewritable();
1015     Node* result = NULL;
1016     Node* current = st;
1017     // Skip through chains of MemBarNodes checking the MergeMems for
1018     // new states for the slice of this load.  Stop once any other
1019     // kind of node is encountered.  Loads from final memory can skip
1020     // through any kind of MemBar but normal loads shouldn't skip
1021     // through MemBarAcquire since the could allow them to move out of
1022     // a synchronized region.
1023     while (current->is_Proj()) {
1024       int opc = current->in(0)->Opcode();
1025       if ((final && (opc == Op_MemBarAcquire ||
1026                      opc == Op_MemBarAcquireLock ||
1027                      opc == Op_LoadFence)) ||
1028           opc == Op_MemBarRelease ||
1029           opc == Op_StoreFence ||
1030           opc == Op_MemBarReleaseLock ||
1031           opc == Op_MemBarStoreStore ||
1032           opc == Op_MemBarCPUOrder) {
1033         Node* mem = current->in(0)->in(TypeFunc::Memory);
1034         if (mem->is_MergeMem()) {
1035           MergeMemNode* merge = mem->as_MergeMem();
1036           Node* new_st = merge->memory_at(alias_idx);
1037           if (new_st == merge->base_memory()) {
1038             // Keep searching
1039             current = new_st;
1040             continue;
1041           }
1042           // Save the new memory state for the slice and fall through
1043           // to exit.
1044           result = new_st;
1045         }
1046       }
1047       break;
1048     }
1049     if (result != NULL) {
1050       st = result;
1051     }
1052   }
1053 
1054   // Loop around twice in the case Load -> Initialize -> Store.
1055   // (See PhaseIterGVN::add_users_to_worklist, which knows about this case.)
1056   for (int trip = 0; trip <= 1; trip++) {
1057 
1058     if (st->is_Store()) {
1059       Node* st_adr = st->in(MemNode::Address);
1060       if (!phase->eqv(st_adr, ld_adr)) {
1061         // Try harder before giving up. Unify base pointers with casts (e.g., raw/non-raw pointers).
1062         intptr_t st_off = 0;
1063         Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_off);
1064         if (ld_base == NULL)                                   return NULL;
1065         if (st_base == NULL)                                   return NULL;
1066         if (!ld_base->eqv_uncast(st_base, /*keep_deps=*/true)) return NULL;
1067         if (ld_off != st_off)                                  return NULL;
1068         if (ld_off == Type::OffsetBot)                         return NULL;
1069         // Same base, same offset.
1070         // Possible improvement for arrays: check index value instead of absolute offset.
1071 
1072         // At this point we have proven something like this setup:
1073         //   B = << base >>
1074         //   L =  LoadQ(AddP(Check/CastPP(B), #Off))
1075         //   S = StoreQ(AddP(             B , #Off), V)
1076         // (Actually, we haven't yet proven the Q's are the same.)
1077         // In other words, we are loading from a casted version of
1078         // the same pointer-and-offset that we stored to.
1079         // Casted version may carry a dependency and it is respected.
1080         // Thus, we are able to replace L by V.
1081       }
1082       // Now prove that we have a LoadQ matched to a StoreQ, for some Q.
1083       if (store_Opcode() != st->Opcode())
1084         return NULL;
1085       return st->in(MemNode::ValueIn);
1086     }
1087 
1088     // A load from a freshly-created object always returns zero.
1089     // (This can happen after LoadNode::Ideal resets the load's memory input
1090     // to find_captured_store, which returned InitializeNode::zero_memory.)
1091     if (st->is_Proj() && st->in(0)->is_Allocate() &&
1092         (st->in(0) == ld_alloc) &&
1093         (ld_off >= st->in(0)->as_Allocate()->minimum_header_size())) {
1094       // return a zero value for the load's basic type
1095       // (This is one of the few places where a generic PhaseTransform
1096       // can create new nodes.  Think of it as lazily manifesting
1097       // virtually pre-existing constants.)
1098       assert(memory_type() != T_VALUETYPE, "should not be used for value types");
1099       Node* default_value = ld_alloc->in(AllocateNode::DefaultValue);
1100       if (default_value != NULL) {
1101         return default_value;
1102       }
1103       assert(ld_alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
1104       return phase->zerocon(memory_type());
1105     }
1106 
1107     // A load from an initialization barrier can match a captured store.
1108     if (st->is_Proj() && st->in(0)->is_Initialize()) {
1109       InitializeNode* init = st->in(0)->as_Initialize();
1110       AllocateNode* alloc = init->allocation();
1111       if ((alloc != NULL) && (alloc == ld_alloc)) {
1112         // examine a captured store value
1113         st = init->find_captured_store(ld_off, memory_size(), phase);
1114         if (st != NULL) {
1115           continue;             // take one more trip around
1116         }
1117       }
1118     }
1119 
1120     // Load boxed value from result of valueOf() call is input parameter.
1121     if (this->is_Load() && ld_adr->is_AddP() &&
1122         (tp != NULL) && tp->is_ptr_to_boxed_value()) {
1123       intptr_t ignore = 0;
1124       Node* base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ignore);
1125       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1126       base = bs->step_over_gc_barrier(base);
1127       if (base != NULL && base->is_Proj() &&
1128           base->as_Proj()->_con == TypeFunc::Parms &&
1129           base->in(0)->is_CallStaticJava() &&
1130           base->in(0)->as_CallStaticJava()->is_boxing_method()) {
1131         return base->in(0)->in(TypeFunc::Parms);
1132       }
1133     }
1134 
1135     break;
1136   }
1137 
1138   return NULL;
1139 }

1158 //------------------------------Identity---------------------------------------
1159 // Loads are identity if previous store is to same address
1160 Node* LoadNode::Identity(PhaseGVN* phase) {
1161   // Loading from a ValueTypePtr? The ValueTypePtr has the values of
1162   // all fields as input. Look for the field with matching offset.
1163   Node* addr = in(Address);
1164   intptr_t offset;
1165   Node* base = AddPNode::Ideal_base_and_offset(addr, phase, offset);
1166   if (base != NULL && base->is_ValueTypePtr() && offset > oopDesc::klass_offset_in_bytes()) {
1167     Node* value = base->as_ValueTypePtr()->field_value_by_offset((int)offset, true);
1168     if (value->is_ValueType()) {
1169       // Non-flattened value type field
1170       ValueTypeNode* vt = value->as_ValueType();
1171       if (vt->is_allocated(phase)) {
1172         value = vt->get_oop();
1173       } else {
1174         // Not yet allocated, bail out
1175         value = NULL;
1176       }
1177     }
1178     if (value != NULL) {
1179       if (Opcode() == Op_LoadN) {
1180         // Encode oop value if we are loading a narrow oop
1181         assert(!phase->type(value)->isa_narrowoop(), "should already be decoded");
1182         value = phase->transform(new EncodePNode(value, bottom_type()));
1183       }
1184       return value;
1185     }
1186   }
1187 
1188   // If the previous store-maker is the right kind of Store, and the store is
1189   // to the same address, then we are equal to the value stored.
1190   Node* mem = in(Memory);
1191   Node* value = can_see_stored_value(mem, phase);
1192   if( value ) {
1193     // byte, short & char stores truncate naturally.
1194     // A load has to load the truncated value which requires
1195     // some sort of masking operation and that requires an
1196     // Ideal call instead of an Identity call.
1197     if (memory_size() < BytesPerInt) {
1198       // If the input to the store does not fit with the load's result type,
1199       // it must be truncated via an Ideal call.
1200       if (!phase->type(value)->higher_equal(phase->type(this)))
1201         return this;
1202     }
1203     // (This works even when value is a Con, but LoadNode::Value
1204     // usually runs first, producing the singleton type of the Con.)
1205     return value;
1206   }
1207 
1208   // Search for an existing data phi which was generated before for the same
1209   // instance's field to avoid infinite generation of phis in a loop.
1210   Node *region = mem->in(0);
1211   if (is_instance_field_load_with_local_phi(region)) {
1212     const TypeOopPtr *addr_t = in(Address)->bottom_type()->isa_oopptr();
1213     int this_index  = phase->C->get_alias_index(addr_t);
1214     int this_offset = addr_t->offset();
1215     int this_iid    = addr_t->instance_id();
1216     if (!addr_t->is_known_instance() &&
1217          addr_t->is_ptr_to_boxed_value()) {
1218       // Use _idx of address base (could be Phi node) for boxed values.
1219       intptr_t   ignore = 0;
1220       Node*      base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1221       if (base == NULL) {
1222         return this;
1223       }
1224       this_iid = base->_idx;
1225     }
1226     const Type* this_type = bottom_type();
1227     for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
1228       Node* phi = region->fast_out(i);
1229       if (phi->is_Phi() && phi != mem &&
1230           phi->as_Phi()->is_same_inst_field(this_type, (int)mem->_idx, this_iid, this_index, this_offset)) {
1231         return phi;
1232       }
1233     }
1234   }
1235 
1236   return this;
1237 }

1277 // We're loading from an object which has autobox behaviour.
1278 // If this object is result of a valueOf call we'll have a phi
1279 // merging a newly allocated object and a load from the cache.
1280 // We want to replace this load with the original incoming
1281 // argument to the valueOf call.
1282 Node* LoadNode::eliminate_autobox(PhaseGVN* phase) {
1283   assert(phase->C->eliminate_boxing(), "sanity");
1284   intptr_t ignore = 0;
1285   Node* base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1286   if ((base == NULL) || base->is_Phi()) {
1287     // Push the loads from the phi that comes from valueOf up
1288     // through it to allow elimination of the loads and the recovery
1289     // of the original value. It is done in split_through_phi().
1290     return NULL;
1291   } else if (base->is_Load() ||
1292              (base->is_DecodeN() && base->in(1)->is_Load())) {
1293     // Eliminate the load of boxed value for integer types from the cache
1294     // array by deriving the value from the index into the array.
1295     // Capture the offset of the load and then reverse the computation.
1296 
1297     // Get LoadN node which loads a boxing object from 'cache' array.
1298     if (base->is_DecodeN()) {
1299       base = base->in(1);
1300     }
1301     if (!base->in(Address)->is_AddP()) {
1302       return NULL; // Complex address
1303     }
1304     AddPNode* address = base->in(Address)->as_AddP();
1305     Node* cache_base = address->in(AddPNode::Base);
1306     if ((cache_base != NULL) && cache_base->is_DecodeN()) {
1307       // Get ConP node which is static 'cache' field.
1308       cache_base = cache_base->in(1);
1309     }
1310     if ((cache_base != NULL) && cache_base->is_Con()) {
1311       const TypeAryPtr* base_type = cache_base->bottom_type()->isa_aryptr();
1312       if ((base_type != NULL) && base_type->is_autobox_cache()) {
1313         Node* elements[4];
1314         int shift = exact_log2(type2aelembytes(T_OBJECT));
1315         int count = address->unpack_offsets(elements, ARRAY_SIZE(elements));
1316         if (count > 0 && elements[0]->is_Con() &&
1317             (count == 1 ||
1318              (count == 2 && elements[1]->Opcode() == Op_LShiftX &&
1319                             elements[1]->in(2) == phase->intcon(shift)))) {
1320           ciObjArray* array = base_type->const_oop()->as_obj_array();
1321           // Fetch the box object cache[0] at the base of the array and get its value
1322           ciInstance* box = array->obj_at(0)->as_instance();
1323           ciInstanceKlass* ik = box->klass()->as_instance_klass();
1324           assert(ik->is_box_klass(), "sanity");
1325           assert(ik->nof_nonstatic_fields() == 1, "change following code");
1326           if (ik->nof_nonstatic_fields() == 1) {
1327             // This should be true nonstatic_field_at requires calling
1328             // nof_nonstatic_fields so check it anyway
1329             ciConstant c = box->field_value(ik->nonstatic_field_at(0));
1330             BasicType bt = c.basic_type();
1331             // Only integer types have boxing cache.
1332             assert(bt == T_BOOLEAN || bt == T_CHAR  ||
1333                    bt == T_BYTE    || bt == T_SHORT ||
1334                    bt == T_INT     || bt == T_LONG, "wrong type = %s", type2name(bt));
1335             jlong cache_low = (bt == T_LONG) ? c.as_long() : c.as_int();
1336             if (cache_low != (int)cache_low) {
1337               return NULL; // should not happen since cache is array indexed by value
1338             }
1339             jlong offset = arrayOopDesc::base_offset_in_bytes(T_OBJECT) - (cache_low << shift);
1340             if (offset != (int)offset) {
1341               return NULL; // should not happen since cache is array indexed by value
1342             }
1343            // Add up all the offsets making of the address of the load
1344             Node* result = elements[0];
1345             for (int i = 1; i < count; i++) {
1346               result = phase->transform(new AddXNode(result, elements[i]));
1347             }
1348             // Remove the constant offset from the address and then
1349             result = phase->transform(new AddXNode(result, phase->MakeConX(-(int)offset)));
1350             // remove the scaling of the offset to recover the original index.
1351             if (result->Opcode() == Op_LShiftX && result->in(2) == phase->intcon(shift)) {
1352               // Peel the shift off directly but wrap it in a dummy node
1353               // since Ideal can't return existing nodes
1354               result = new RShiftXNode(result->in(1), phase->intcon(0));
1355             } else if (result->is_Add() && result->in(2)->is_Con() &&
1356                        result->in(1)->Opcode() == Op_LShiftX &&
1357                        result->in(1)->in(2) == phase->intcon(shift)) {
1358               // We can't do general optimization: ((X<<Z) + Y) >> Z ==> X + (Y>>Z)
1359               // but for boxing cache access we know that X<<Z will not overflow
1360               // (there is range check) so we do this optimizatrion by hand here.
1361               Node* add_con = new RShiftXNode(result->in(2), phase->intcon(shift));
1362               result = new AddXNode(result->in(1)->in(1), phase->transform(add_con));
1363             } else {
1364               result = new RShiftXNode(result, phase->intcon(shift));
1365             }
1366 #ifdef _LP64
1367             if (bt != T_LONG) {
1368               result = new ConvL2INode(phase->transform(result));
1369             }
1370 #else
1371             if (bt == T_LONG) {
1372               result = new ConvI2LNode(phase->transform(result));
1373             }
1374 #endif
1375             // Boxing/unboxing can be done from signed & unsigned loads (e.g. LoadUB -> ... -> LoadB pair).
1376             // Need to preserve unboxing load type if it is unsigned.
1377             switch(this->Opcode()) {
1378               case Op_LoadUB:
1379                 result = new AndINode(phase->transform(result), phase->intcon(0xFF));
1380                 break;
1381               case Op_LoadUS:
1382                 result = new AndINode(phase->transform(result), phase->intcon(0xFFFF));
1383                 break;
1384             }
1385             return result;
1386           }
1387         }
1388       }
1389     }
1390   }
1391   return NULL;
1392 }

1394 static bool stable_phi(PhiNode* phi, PhaseGVN *phase) {
1395   Node* region = phi->in(0);
1396   if (region == NULL) {
1397     return false; // Wait stable graph
1398   }
1399   uint cnt = phi->req();
1400   for (uint i = 1; i < cnt; i++) {
1401     Node* rc = region->in(i);
1402     if (rc == NULL || phase->type(rc) == Type::TOP)
1403       return false; // Wait stable graph
1404     Node* in = phi->in(i);
1405     if (in == NULL || phase->type(in) == Type::TOP)
1406       return false; // Wait stable graph
1407   }
1408   return true;

1410 //------------------------------split_through_phi------------------------------
1411 // Split instance or boxed field load through Phi.
1412 Node *LoadNode::split_through_phi(PhaseGVN *phase) {
1413   Node* mem     = in(Memory);
1414   Node* address = in(Address);
1415   const TypeOopPtr *t_oop = phase->type(address)->isa_oopptr();
1416 
1417   assert((t_oop != NULL) &&
1418          (t_oop->is_known_instance_field() ||
1419           t_oop->is_ptr_to_boxed_value()), "invalide conditions");
1420 
1421   Compile* C = phase->C;
1422   intptr_t ignore = 0;
1423   Node*    base = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1424   bool base_is_phi = (base != NULL) && base->is_Phi();
1425   bool load_boxed_values = t_oop->is_ptr_to_boxed_value() && C->aggressive_unboxing() &&
1426                            (base != NULL) && (base == address->in(AddPNode::Base)) &&
1427                            phase->type(base)->higher_equal(TypePtr::NOTNULL);
1428 
1429   if (!((mem->is_Phi() || base_is_phi) &&
1430         (load_boxed_values || t_oop->is_known_instance_field()))) {
1431     return NULL; // memory is not Phi
1432   }
1433 
1434   if (mem->is_Phi()) {
1435     if (!stable_phi(mem->as_Phi(), phase)) {
1436       return NULL; // Wait stable graph
1437     }
1438     uint cnt = mem->req();
1439     // Check for loop invariant memory.
1440     if (cnt == 3) {
1441       for (uint i = 1; i < cnt; i++) {
1442         Node* in = mem->in(i);
1443         Node*  m = optimize_memory_chain(in, t_oop, this, phase);
1444         if (m == mem) {
1445           if (i == 1) {
1446             // if the first edge was a loop, check second edge too.
1447             // If both are replaceable - we are in an infinite loop
1448             Node *n = optimize_memory_chain(mem->in(2), t_oop, this, phase);
1449             if (n == mem) {
1450               break;
1451             }
1452           }
1453           set_req(Memory, mem->in(cnt - i));
1454           return this; // made change
1455         }
1456       }
1457     }
1458   }
1459   if (base_is_phi) {
1460     if (!stable_phi(base->as_Phi(), phase)) {
1461       return NULL; // Wait stable graph
1462     }
1463     uint cnt = base->req();
1464     // Check for loop invariant memory.
1465     if (cnt == 3) {
1466       for (uint i = 1; i < cnt; i++) {
1467         if (base->in(i) == base) {
1468           return NULL; // Wait stable graph
1469         }
1470       }
1471     }
1472   }
1473 
1474   // Split through Phi (see original code in loopopts.cpp).
1475   assert(C->have_alias_type(t_oop), "instance should have alias type");
1476 
1477   // Do nothing here if Identity will find a value
1478   // (to avoid infinite chain of value phis generation).
1479   if (!phase->eqv(this, this->Identity(phase))) {
1480     return NULL;
1481   }
1482 
1483   // Select Region to split through.
1484   Node* region;
1485   if (!base_is_phi) {
1486     assert(mem->is_Phi(), "sanity");
1487     region = mem->in(0);
1488     // Skip if the region dominates some control edge of the address.
1489     if (!MemNode::all_controls_dominate(address, region))
1490       return NULL;
1491   } else if (!mem->is_Phi()) {
1492     assert(base_is_phi, "sanity");
1493     region = base->in(0);
1494     // Skip if the region dominates some control edge of the memory.
1495     if (!MemNode::all_controls_dominate(mem, region))
1496       return NULL;
1497   } else if (base->in(0) != mem->in(0)) {
1498     assert(base_is_phi && mem->is_Phi(), "sanity");
1499     if (MemNode::all_controls_dominate(mem, base->in(0))) {
1500       region = base->in(0);
1501     } else if (MemNode::all_controls_dominate(address, mem->in(0))) {
1502       region = mem->in(0);
1503     } else {
1504       return NULL; // complex graph
1505     }
1506   } else {
1507     assert(base->in(0) == mem->in(0), "sanity");
1508     region = mem->in(0);
1509   }
1510 
1511   const Type* this_type = this->bottom_type();
1512   int this_index  = C->get_alias_index(t_oop);
1513   int this_offset = t_oop->offset();
1514   int this_iid    = t_oop->instance_id();
1515   if (!t_oop->is_known_instance() && load_boxed_values) {
1516     // Use _idx of address base for boxed values.
1517     this_iid = base->_idx;
1518   }
1519   PhaseIterGVN* igvn = phase->is_IterGVN();
1520   Node* phi = new PhiNode(region, this_type, NULL, mem->_idx, this_iid, this_index, this_offset);
1521   for (uint i = 1; i < region->req(); i++) {
1522     Node* x;
1523     Node* the_clone = NULL;
1524     Node* in = region->in(i);
1525     if (region->is_CountedLoop() && region->as_Loop()->is_strip_mined() && i == LoopNode::EntryControl &&
1526         in != NULL && in->is_OuterStripMinedLoop()) {
1527       // No node should go in the outer strip mined loop
1528       in = in->in(LoopNode::EntryControl);
1529     }
1530     if (in == NULL || in == C->top()) {
1531       x = C->top();      // Dead path?  Use a dead data op
1532     } else {
1533       x = this->clone();        // Else clone up the data op
1534       the_clone = x;            // Remember for possible deletion.
1535       // Alter data node to use pre-phi inputs
1536       if (this->in(0) == region) {
1537         x->set_req(0, in);
1538       } else {
1539         x->set_req(0, NULL);
1540       }
1541       if (mem->is_Phi() && (mem->in(0) == region)) {
1542         x->set_req(Memory, mem->in(i)); // Use pre-Phi input for the clone.
1543       }
1544       if (address->is_Phi() && address->in(0) == region) {
1545         x->set_req(Address, address->in(i)); // Use pre-Phi input for the clone
1546       }
1547       if (base_is_phi && (base->in(0) == region)) {
1548         Node* base_x = base->in(i); // Clone address for loads from boxed objects.
1549         Node* adr_x = phase->transform(new AddPNode(base_x,base_x,address->in(AddPNode::Offset)));
1550         x->set_req(Address, adr_x);
1551       }
1552     }
1553     // Check for a 'win' on some paths
1554     const Type *t = x->Value(igvn);
1555 
1556     bool singleton = t->singleton();
1557 
1558     // See comments in PhaseIdealLoop::split_thru_phi().
1559     if (singleton && t == Type::TOP) {
1560       singleton &= region->is_Loop() && (i != LoopNode::EntryControl);
1561     }
1562 
1563     if (singleton) {
1564       x = igvn->makecon(t);
1565     } else {
1566       // We now call Identity to try to simplify the cloned node.
1567       // Note that some Identity methods call phase->type(this).
1568       // Make sure that the type array is big enough for
1569       // our new node, even though we may throw the node away.
1570       // (This tweaking with igvn only works because x is a new node.)
1571       igvn->set_type(x, t);
1572       // If x is a TypeNode, capture any more-precise type permanently into Node
1573       // otherwise it will be not updated during igvn->transform since
1574       // igvn->type(x) is set to x->Value() already.
1575       x->raise_bottom_type(t);
1576       Node* y = x->Identity(igvn);
1577       if (y != x) {
1578         x = y;
1579       } else {
1580         y = igvn->hash_find_insert(x);
1581         if (y) {
1582           x = y;
1583         } else {
1584           // Else x is a new node we are keeping
1585           // We do not need register_new_node_with_optimizer
1586           // because set_type has already been called.
1587           igvn->_worklist.push(x);
1588         }
1589       }
1590     }
1591     if (x != the_clone && the_clone != NULL) {
1592       igvn->remove_dead_node(the_clone);
1593     }
1594     phi->set_req(i, x);
1595   }
1596   // Record Phi
1597   igvn->register_new_node_with_optimizer(phi);
1598   return phi;
1599 }

1617 //------------------------------Ideal------------------------------------------
1618 // If the load is from Field memory and the pointer is non-null, it might be possible to
1619 // zero out the control input.
1620 // If the offset is constant and the base is an object allocation,
1621 // try to hook me up to the exact initializing store.
1622 Node *LoadNode::Ideal(PhaseGVN *phase, bool can_reshape) {
1623   Node* p = MemNode::Ideal_common(phase, can_reshape);
1624   if (p)  return (p == NodeSentinel) ? NULL : p;
1625 
1626   Node* ctrl    = in(MemNode::Control);
1627   Node* address = in(MemNode::Address);
1628   bool progress = false;
1629 
1630   bool addr_mark = ((phase->type(address)->isa_oopptr() || phase->type(address)->isa_narrowoop()) &&
1631          phase->type(address)->is_ptr()->offset() == oopDesc::mark_offset_in_bytes());
1632 
1633   // Skip up past a SafePoint control.  Cannot do this for Stores because
1634   // pointer stores & cardmarks must stay on the same side of a SafePoint.
1635   if( ctrl != NULL && ctrl->Opcode() == Op_SafePoint &&
1636       phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw  &&
1637       !addr_mark &&
1638       (depends_only_on_test() || has_unknown_control_dependency())) {
1639     ctrl = ctrl->in(0);
1640     set_req(MemNode::Control,ctrl);
1641     progress = true;
1642   }
1643 
1644   intptr_t ignore = 0;
1645   Node*    base   = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1646   if (base != NULL
1647       && phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw) {
1648     // Check for useless control edge in some common special cases
1649     if (in(MemNode::Control) != NULL
1650         && can_remove_control()
1651         && phase->type(base)->higher_equal(TypePtr::NOTNULL)
1652         && all_controls_dominate(base, phase->C->start())) {
1653       // A method-invariant, non-null address (constant or 'this' argument).
1654       set_req(MemNode::Control, NULL);
1655       progress = true;
1656     }
1657   }
1658 
1659   Node* mem = in(MemNode::Memory);
1660   const TypePtr *addr_t = phase->type(address)->isa_ptr();
1661 
1662   if (can_reshape && (addr_t != NULL)) {
1663     // try to optimize our memory input
1664     Node* opt_mem = MemNode::optimize_memory_chain(mem, addr_t, this, phase);
1665     if (opt_mem != mem) {
1666       set_req(MemNode::Memory, opt_mem);
1667       if (phase->type( opt_mem ) == Type::TOP) return NULL;
1668       return this;
1669     }
1670     const TypeOopPtr *t_oop = addr_t->isa_oopptr();
1671     if ((t_oop != NULL) &&
1672         (t_oop->is_known_instance_field() ||
1673          t_oop->is_ptr_to_boxed_value())) {
1674       PhaseIterGVN *igvn = phase->is_IterGVN();
1675       if (igvn != NULL && igvn->_worklist.member(opt_mem)) {
1676         // Delay this transformation until memory Phi is processed.
1677         phase->is_IterGVN()->_worklist.push(this);
1678         return NULL;
1679       }
1680       // Split instance field load through Phi.
1681       Node* result = split_through_phi(phase);
1682       if (result != NULL) return result;
1683 
1684       if (t_oop->is_ptr_to_boxed_value()) {
1685         Node* result = eliminate_autobox(phase);
1686         if (result != NULL) return result;
1687       }
1688     }
1689   }
1690 
1691   // Is there a dominating load that loads the same value?  Leave
1692   // anything that is not a load of a field/array element (like
1693   // barriers etc.) alone
1694   if (in(0) != NULL && !adr_type()->isa_rawptr() && can_reshape) {
1695     for (DUIterator_Fast imax, i = mem->fast_outs(imax); i < imax; i++) {
1696       Node *use = mem->fast_out(i);
1697       if (use != this &&
1698           use->Opcode() == Opcode() &&
1699           use->in(0) != NULL &&
1700           use->in(0) != in(0) &&
1701           use->in(Address) == in(Address)) {
1702         Node* ctl = in(0);
1703         for (int i = 0; i < 10 && ctl != NULL; i++) {
1704           ctl = IfNode::up_one_dom(ctl);
1705           if (ctl == use->in(0)) {
1706             set_req(0, use->in(0));
1707             return this;
1708           }
1709         }
1710       }
1711     }
1712   }
1713 
1714   // Check for prior store with a different base or offset; make Load
1715   // independent.  Skip through any number of them.  Bail out if the stores
1716   // are in an endless dead cycle and report no progress.  This is a key
1717   // transform for Reflection.  However, if after skipping through the Stores
1718   // we can't then fold up against a prior store do NOT do the transform as
1719   // this amounts to using the 'Oracle' model of aliasing.  It leaves the same
1720   // array memory alive twice: once for the hoisted Load and again after the
1721   // bypassed Store.  This situation only works if EVERYBODY who does
1722   // anti-dependence work knows how to bypass.  I.e. we need all
1723   // anti-dependence checks to ask the same Oracle.  Right now, that Oracle is
1724   // the alias index stuff.  So instead, peek through Stores and IFF we can
1725   // fold up, do so.
1726   Node* prev_mem = find_previous_store(phase);
1727   if (prev_mem != NULL) {
1728     Node* value = can_see_arraycopy_value(prev_mem, phase);
1729     if (value != NULL) {
1730       return value;
1731     }
1732   }
1733   // Steps (a), (b):  Walk past independent stores to find an exact match.
1734   if (prev_mem != NULL && prev_mem != in(MemNode::Memory)) {
1735     // (c) See if we can fold up on the spot, but don't fold up here.
1736     // Fold-up might require truncation (for LoadB/LoadS/LoadUS) or
1737     // just return a prior value, which is done by Identity calls.
1738     if (can_see_stored_value(prev_mem, phase)) {
1739       // Make ready for step (d):
1740       set_req(MemNode::Memory, prev_mem);
1741       return this;
1742     }
1743   }
1744 
1745   AllocateNode* alloc = AllocateNode::Ideal_allocation(address, phase);
1746   if (alloc != NULL && mem->is_Proj() &&
1747       mem->in(0) != NULL &&
1748       mem->in(0) == alloc->initialization() &&
1749       Opcode() == Op_LoadX &&
1750       alloc->initialization()->proj_out_or_null(0) != NULL) {
1751     InitializeNode* init = alloc->initialization();
1752     Node* control = init->proj_out(0);
1753     return alloc->make_ideal_mark(phase, control, mem);
1754   }
1755 
1756   return progress ? this : NULL;
1757 }

1759 // Helper to recognize certain Klass fields which are invariant across
1760 // some group of array types (e.g., int[] or all T[] where T < Object).
1761 const Type*
1762 LoadNode::load_array_final_field(const TypeKlassPtr *tkls,
1763                                  ciKlass* klass) const {
1764   if (tkls->offset() == in_bytes(Klass::modifier_flags_offset())) {
1765     // The field is Klass::_modifier_flags.  Return its (constant) value.
1766     // (Folds up the 2nd indirection in aClassConstant.getModifiers().)
1767     assert(this->Opcode() == Op_LoadI, "must load an int from _modifier_flags");
1768     return TypeInt::make(klass->modifier_flags());
1769   }
1770   if (tkls->offset() == in_bytes(Klass::access_flags_offset())) {
1771     // The field is Klass::_access_flags.  Return its (constant) value.
1772     // (Folds up the 2nd indirection in Reflection.getClassAccessFlags(aClassConstant).)
1773     assert(this->Opcode() == Op_LoadI, "must load an int from _access_flags");
1774     return TypeInt::make(klass->access_flags());
1775   }
1776   if (tkls->offset() == in_bytes(Klass::layout_helper_offset())) {
1777     // The field is Klass::_layout_helper.  Return its constant value if known.
1778     assert(this->Opcode() == Op_LoadI, "must load an int from _layout_helper");
1779     return TypeInt::make(klass->layout_helper());
1780   }
1781 
1782   // No match.
1783   return NULL;
1784 }

1786 //------------------------------Value-----------------------------------------
1787 const Type* LoadNode::Value(PhaseGVN* phase) const {
1788   // Either input is TOP ==> the result is TOP
1789   Node* mem = in(MemNode::Memory);
1790   const Type *t1 = phase->type(mem);
1791   if (t1 == Type::TOP)  return Type::TOP;
1792   Node* adr = in(MemNode::Address);
1793   const TypePtr* tp = phase->type(adr)->isa_ptr();
1794   if (tp == NULL || tp->empty())  return Type::TOP;
1795   int off = tp->offset();
1796   assert(off != Type::OffsetTop, "case covered by TypePtr::empty");
1797   Compile* C = phase->C;
1798 
1799   // Try to guess loaded type from pointer type
1800   if (tp->isa_aryptr()) {
1801     const TypeAryPtr* ary = tp->is_aryptr();
1802     const Type* t = ary->elem();
1803 
1804     // Determine whether the reference is beyond the header or not, by comparing
1805     // the offset against the offset of the start of the array's data.
1806     // Different array types begin at slightly different offsets (12 vs. 16).
1807     // We choose T_BYTE as an example base type that is least restrictive
1808     // as to alignment, which will therefore produce the smallest
1809     // possible base offset.
1810     const int min_base_off = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1811     const bool off_beyond_header = (off >= min_base_off);
1812 
1813     // Try to constant-fold a stable array element.
1814     if (FoldStableValues && !is_mismatched_access() && ary->is_stable()) {
1815       // Make sure the reference is not into the header and the offset is constant
1816       ciObject* aobj = ary->const_oop();
1817       if (aobj != NULL && off_beyond_header && adr->is_AddP() && off != Type::OffsetBot) {
1818         int stable_dimension = (ary->stable_dimension() > 0 ? ary->stable_dimension() - 1 : 0);
1819         const Type* con_type = Type::make_constant_from_array_element(aobj->as_array(), off,
1820                                                                       stable_dimension,
1821                                                                       memory_type(), is_unsigned());
1822         if (con_type != NULL) {
1823           return con_type;
1824         }
1825       }
1826     }
1827 
1828     // Don't do this for integer types. There is only potential profit if
1829     // the element type t is lower than _type; that is, for int types, if _type is
1830     // more restrictive than t.  This only happens here if one is short and the other
1831     // char (both 16 bits), and in those cases we've made an intentional decision
1832     // to use one kind of load over the other. See AndINode::Ideal and 4965907.
1833     // Also, do not try to narrow the type for a LoadKlass, regardless of offset.
1834     //
1835     // Yes, it is possible to encounter an expression like (LoadKlass p1:(AddP x x 8))
1836     // where the _gvn.type of the AddP is wider than 8.  This occurs when an earlier
1837     // copy p0 of (AddP x x 8) has been proven equal to p1, and the p0 has been
1838     // subsumed by p1.  If p1 is on the worklist but has not yet been re-transformed,
1839     // it is possible that p1 will have a type like Foo*[int+]:NotNull*+any.
1840     // In fact, that could have been the original type of p1, and p1 could have
1841     // had an original form like p1:(AddP x x (LShiftL quux 3)), where the
1842     // expression (LShiftL quux 3) independently optimized to the constant 8.
1843     if ((t->isa_int() == NULL) && (t->isa_long() == NULL)
1844         && (_type->isa_vect() == NULL)
1845         && t->isa_valuetype() == NULL
1846         && Opcode() != Op_LoadKlass && Opcode() != Op_LoadNKlass) {
1847       // t might actually be lower than _type, if _type is a unique
1848       // concrete subclass of abstract class t.
1849       if (off_beyond_header || off == Type::OffsetBot) {  // is the offset beyond the header?
1850         const Type* jt = t->join_speculative(_type);
1851         // In any case, do not allow the join, per se, to empty out the type.
1852         if (jt->empty() && !t->empty()) {
1853           // This can happen if a interface-typed array narrows to a class type.
1854           jt = _type;
1855         }
1856 #ifdef ASSERT
1857         if (phase->C->eliminate_boxing() && adr->is_AddP()) {
1858           // The pointers in the autobox arrays are always non-null
1859           Node* base = adr->in(AddPNode::Base);
1860           if ((base != NULL) && base->is_DecodeN()) {
1861             // Get LoadN node which loads IntegerCache.cache field
1862             base = base->in(1);
1863           }
1864           if ((base != NULL) && base->is_Con()) {
1865             const TypeAryPtr* base_type = base->bottom_type()->isa_aryptr();
1866             if ((base_type != NULL) && base_type->is_autobox_cache()) {
1867               // It could be narrow oop
1868               assert(jt->make_ptr()->ptr() == TypePtr::NotNull,"sanity");
1869             }
1870           }
1871         }
1872 #endif
1873         return jt;
1874       }
1875     }
1876   } else if (tp->base() == Type::InstPtr) {
1877     assert( off != Type::OffsetBot ||
1878             // arrays can be cast to Objects
1879             tp->is_oopptr()->klass()->is_java_lang_Object() ||
1880             tp->is_oopptr()->klass() == ciEnv::current()->Class_klass() ||
1881             // unsafe field access may not have a constant offset
1882             C->has_unsafe_access(),
1883             "Field accesses must be precise" );
1884     // For oop loads, we expect the _type to be precise.
1885 
1886     const TypeInstPtr* tinst = tp->is_instptr();
1887     BasicType bt = memory_type();
1888 
1889     // Fold component and value mirror loads
1890     ciInstanceKlass* ik = tinst->klass()->as_instance_klass();
1891     if (ik == phase->C->env()->Class_klass() && (off == java_lang_Class::component_mirror_offset_in_bytes() ||
1892                                                  off == java_lang_Class::inline_mirror_offset_in_bytes())) {
1893       ciType* mirror_type = tinst->java_mirror_type();
1894       if (mirror_type != NULL) {
1895         const Type* const_oop = TypePtr::NULL_PTR;
1896         if (mirror_type->is_array_klass()) {
1897           const_oop = TypeInstPtr::make(mirror_type->as_array_klass()->component_mirror_instance());
1898         } else if (mirror_type->is_valuetype()) {
1899           const_oop = TypeInstPtr::make(mirror_type->as_value_klass()->inline_mirror_instance());
1900         }
1901         return (bt == T_NARROWOOP) ? const_oop->make_narrowoop() : const_oop;
1902       }
1903     }
1904 
1905     // Optimize loads from constant fields.
1906     ciObject* const_oop = tinst->const_oop();
1907     if (!is_mismatched_access() && off != Type::OffsetBot && const_oop != NULL && const_oop->is_instance()) {
1908       ciType* mirror_type = const_oop->as_instance()->java_mirror_type();
1909       if (mirror_type != NULL && mirror_type->is_valuetype()) {
1910         ciValueKlass* vk = mirror_type->as_value_klass();
1911         if (off == vk->default_value_offset()) {
1912           // Loading a special hidden field that contains the oop of the default value type
1913           const Type* const_oop = TypeInstPtr::make(vk->default_value_instance());
1914           return (bt == T_NARROWOOP) ? const_oop->make_narrowoop() : const_oop;
1915         }
1916       }
1917       const Type* con_type = Type::make_constant_from_field(const_oop->as_instance(), off, is_unsigned(), bt);
1918       if (con_type != NULL) {
1919         return con_type;
1920       }
1921     }
1922   } else if (tp->base() == Type::KlassPtr) {
1923     assert( off != Type::OffsetBot ||
1924             // arrays can be cast to Objects
1925             tp->is_klassptr()->klass() == NULL ||
1926             tp->is_klassptr()->klass()->is_java_lang_Object() ||
1927             // also allow array-loading from the primary supertype
1928             // array during subtype checks
1929             Opcode() == Op_LoadKlass,
1930             "Field accesses must be precise" );
1931     // For klass/static loads, we expect the _type to be precise
1932   } else if (tp->base() == Type::RawPtr && !StressReflectiveCode) {
1933     if (adr->is_Load() && off == 0) {
1934       /* With mirrors being an indirect in the Klass*
1935        * the VM is now using two loads. LoadKlass(LoadP(LoadP(Klass, mirror_offset), zero_offset))
1936        * The LoadP from the Klass has a RawPtr type (see LibraryCallKit::load_mirror_from_klass).
1937        *
1938        * So check the type and klass of the node before the LoadP.
1939        */
1940       Node* adr2 = adr->in(MemNode::Address);
1941       const TypeKlassPtr* tkls = phase->type(adr2)->isa_klassptr();
1942       if (tkls != NULL) {
1943         ciKlass* klass = tkls->klass();
1944         if (klass != NULL && klass->is_loaded() && tkls->klass_is_exact() && tkls->offset() == in_bytes(Klass::java_mirror_offset())) {
1945           assert(adr->Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1946           assert(Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1947           return TypeInstPtr::make(klass->java_mirror());
1948         }
1949       }
1950     } else {
1951       // Check for a load of the default value offset from the ValueKlassFixedBlock:
1952       // LoadI(LoadP(value_klass, adr_valueklass_fixed_block_offset), default_value_offset_offset)
1953       intptr_t offset = 0;
1954       Node* base = AddPNode::Ideal_base_and_offset(adr, phase, offset);
1955       if (base != NULL && base->is_Load() && offset == in_bytes(ValueKlass::default_value_offset_offset())) {
1956         const TypeKlassPtr* tkls = phase->type(base->in(MemNode::Address))->isa_klassptr();
1957         if (tkls != NULL && tkls->is_loaded() && tkls->klass_is_exact() && tkls->isa_valuetype() &&
1958             tkls->offset() == in_bytes(InstanceKlass::adr_valueklass_fixed_block_offset())) {
1959           assert(base->Opcode() == Op_LoadP, "must load an oop from klass");
1960           assert(Opcode() == Op_LoadI, "must load an int from fixed block");
1961           return TypeInt::make(tkls->klass()->as_value_klass()->default_value_offset());
1962         }
1963       }
1964     }
1965   }
1966 
1967   const TypeKlassPtr *tkls = tp->isa_klassptr();
1968   if (tkls != NULL && !StressReflectiveCode) {
1969     ciKlass* klass = tkls->klass();
1970     if (tkls->is_loaded() && tkls->klass_is_exact()) {
1971       // We are loading a field from a Klass metaobject whose identity
1972       // is known at compile time (the type is "exact" or "precise").
1973       // Check for fields we know are maintained as constants by the VM.
1974       if (tkls->offset() == in_bytes(Klass::super_check_offset_offset())) {
1975         // The field is Klass::_super_check_offset.  Return its (constant) value.
1976         // (Folds up type checking code.)
1977         assert(Opcode() == Op_LoadI, "must load an int from _super_check_offset");
1978         return TypeInt::make(klass->super_check_offset());
1979       }
1980       // Compute index into primary_supers array
1981       juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1982       // Check for overflowing; use unsigned compare to handle the negative case.
1983       if( depth < ciKlass::primary_super_limit() ) {
1984         // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1985         // (Folds up type checking code.)
1986         assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
1987         ciKlass *ss = klass->super_of_depth(depth);
1988         return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
1989       }
1990       const Type* aift = load_array_final_field(tkls, klass);
1991       if (aift != NULL)  return aift;
1992     }
1993 
1994     // We can still check if we are loading from the primary_supers array at a
1995     // shallow enough depth.  Even though the klass is not exact, entries less
1996     // than or equal to its super depth are correct.
1997     if (tkls->is_loaded()) {
1998       ciType *inner = klass;
1999       while( inner->is_obj_array_klass() )
2000         inner = inner->as_obj_array_klass()->base_element_type();
2001       if( inner->is_instance_klass() &&
2002           !inner->as_instance_klass()->flags().is_interface() ) {
2003         // Compute index into primary_supers array
2004         juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
2005         // Check for overflowing; use unsigned compare to handle the negative case.
2006         if( depth < ciKlass::primary_super_limit() &&
2007             depth <= klass->super_depth() ) { // allow self-depth checks to handle self-check case
2008           // The field is an element of Klass::_primary_supers.  Return its (constant) value.
2009           // (Folds up type checking code.)
2010           assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
2011           ciKlass *ss = klass->super_of_depth(depth);
2012           return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
2013         }
2014       }
2015     }
2016 
2017     // If the type is enough to determine that the thing is not an array,
2018     // we can give the layout_helper a positive interval type.
2019     // This will help short-circuit some reflective code.
2020     if (tkls->offset() == in_bytes(Klass::layout_helper_offset())
2021         && !klass->is_array_klass() // not directly typed as an array
2022         && !klass->is_interface()  // specifically not Serializable & Cloneable
2023         && !klass->is_java_lang_Object()   // not the supertype of all T[]
2024         ) {
2025       // Note:  When interfaces are reliable, we can narrow the interface
2026       // test to (klass != Serializable && klass != Cloneable).
2027       assert(Opcode() == Op_LoadI, "must load an int from _layout_helper");
2028       jint min_size = Klass::instance_layout_helper(oopDesc::header_size(), false);
2029       // The key property of this type is that it folds up tests
2030       // for array-ness, since it proves that the layout_helper is positive.
2031       // Thus, a generic value like the basic object layout helper works fine.
2032       return TypeInt::make(min_size, max_jint, Type::WidenMin);
2033     }
2034   }
2035 
2036   // If we are loading from a freshly-allocated object, produce a zero,
2037   // if the load is provably beyond the header of the object.
2038   // (Also allow a variable load from a fresh array to produce zero.)
2039   const TypeOopPtr *tinst = tp->isa_oopptr();
2040   bool is_instance = (tinst != NULL) && tinst->is_known_instance_field();
2041   bool is_boxed_value = (tinst != NULL) && tinst->is_ptr_to_boxed_value();
2042   if (ReduceFieldZeroing || is_instance || is_boxed_value) {
2043     Node* value = can_see_stored_value(mem,phase);
2044     if (value != NULL && value->is_Con()) {

2229 const Type* LoadNode::klass_value_common(PhaseGVN* phase) const {
2230   // Either input is TOP ==> the result is TOP
2231   const Type *t1 = phase->type( in(MemNode::Memory) );
2232   if (t1 == Type::TOP)  return Type::TOP;
2233   Node *adr = in(MemNode::Address);
2234   const Type *t2 = phase->type( adr );
2235   if (t2 == Type::TOP)  return Type::TOP;
2236   const TypePtr *tp = t2->is_ptr();
2237   if (TypePtr::above_centerline(tp->ptr()) ||
2238       tp->ptr() == TypePtr::Null)  return Type::TOP;
2239 
2240   // Return a more precise klass, if possible
2241   const TypeInstPtr *tinst = tp->isa_instptr();
2242   if (tinst != NULL) {
2243     ciInstanceKlass* ik = tinst->klass()->as_instance_klass();
2244     int offset = tinst->offset();
2245     if (ik == phase->C->env()->Class_klass()
2246         && (offset == java_lang_Class::klass_offset_in_bytes() ||
2247             offset == java_lang_Class::array_klass_offset_in_bytes())) {
2248       // We are loading a special hidden field from a Class mirror object,
2249       // the field which points to the VM's Klass metaobject.
2250       bool is_indirect_type = true;
2251       ciType* t = tinst->java_mirror_type(&is_indirect_type);
2252       // java_mirror_type returns non-null for compile-time Class constants.
2253       if (t != NULL) {
2254         // constant oop => constant klass
2255         if (offset == java_lang_Class::array_klass_offset_in_bytes()) {
2256           if (t->is_void()) {
2257             // We cannot create a void array.  Since void is a primitive type return null
2258             // klass.  Users of this result need to do a null check on the returned klass.
2259             return TypePtr::NULL_PTR;
2260           }
2261           return TypeKlassPtr::make(ciArrayKlass::make(t, /* never_null */ !is_indirect_type));
2262         }
2263         if (!t->is_klass()) {
2264           // a primitive Class (e.g., int.class) has NULL for a klass field
2265           return TypePtr::NULL_PTR;
2266         }
2267         // (Folds up the 1st indirection in aClassConstant.getModifiers().)
2268         return TypeKlassPtr::make(t->as_klass());
2269       }
2270       // non-constant mirror, so we can't tell what's going on
2271     }
2272     if( !ik->is_loaded() )
2273       return _type;             // Bail out if not loaded
2274     if (offset == oopDesc::klass_offset_in_bytes()) {
2275       if (tinst->klass_is_exact()) {
2276         return TypeKlassPtr::make(ik);
2277       }
2278       // See if we can become precise: no subklasses and no interface
2279       // (Note:  We need to support verified interfaces.)
2280       if (!ik->is_interface() && !ik->has_subklass()) {
2281         //assert(!UseExactTypes, "this code should be useless with exact types");
2282         // Add a dependence; if any subclass added we need to recompile
2283         if (!ik->is_final()) {
2284           // %%% should use stronger assert_unique_concrete_subtype instead
2285           phase->C->dependencies()->assert_leaf_type(ik);
2286         }
2287         // Return precise klass
2288         return TypeKlassPtr::make(ik);
2289       }
2290 
2291       // Return root of possible klass
2292       return TypeKlassPtr::make(TypePtr::NotNull, ik, Type::Offset(0), tinst->flat_array());
2293     }
2294   }
2295 
2296   // Check for loading klass from an array
2297   const TypeAryPtr *tary = tp->isa_aryptr();
2298   if (tary != NULL) {
2299     ciKlass *tary_klass = tary->klass();
2300     if (tary_klass != NULL   // can be NULL when at BOTTOM or TOP
2301         && tary->offset() == oopDesc::klass_offset_in_bytes()) {
2302       ciArrayKlass* ak = tary_klass->as_array_klass();
2303       // Do not fold klass loads from [V?. The runtime type might be [V due to [V <: [V?
2304       // and the klass for [V is not equal to the klass for [V?.
2305       if (tary->klass_is_exact()) {
2306         return TypeKlassPtr::make(tary_klass);
2307       }
2308 
2309       // If the klass is an object array, we defer the question to the
2310       // array component klass.
2311       if (ak->is_obj_array_klass()) {
2312         assert(ak->is_loaded(), "");
2313         ciKlass *base_k = ak->as_obj_array_klass()->base_element_klass();
2314         if (base_k->is_loaded() && base_k->is_instance_klass()) {
2315           ciInstanceKlass *ik = base_k->as_instance_klass();
2316           // See if we can become precise: no subklasses and no interface
2317           if (!ik->is_interface() && !ik->has_subklass() && (!ik->is_valuetype() || ak->storage_properties().is_null_free())) {
2318             //assert(!UseExactTypes, "this code should be useless with exact types");
2319             // Add a dependence; if any subclass added we need to recompile
2320             if (!ik->is_final()) {
2321               phase->C->dependencies()->assert_leaf_type(ik);
2322             }
2323             // Return precise array klass
2324             return TypeKlassPtr::make(ak);
2325           }
2326         }
2327         return TypeKlassPtr::make(TypePtr::NotNull, ak, Type::Offset(0), false);
2328       } else if (ak->is_type_array_klass()) {
2329         //assert(!UseExactTypes, "this code should be useless with exact types");
2330         return TypeKlassPtr::make(ak); // These are always precise
2331       }
2332     }
2333   }
2334 
2335   // Check for loading klass from an array klass
2336   const TypeKlassPtr *tkls = tp->isa_klassptr();
2337   if (tkls != NULL && !StressReflectiveCode) {
2338     if (!tkls->is_loaded()) {
2339       return _type;             // Bail out if not loaded
2340     }
2341     ciKlass* klass = tkls->klass();
2342     if( klass->is_obj_array_klass() &&
2343         tkls->offset() == in_bytes(ObjArrayKlass::element_klass_offset())) {
2344       ciKlass* elem = klass->as_obj_array_klass()->element_klass();
2345       // // Always returning precise element type is incorrect,
2346       // // e.g., element type could be object and array may contain strings
2347       // return TypeKlassPtr::make(TypePtr::Constant, elem, 0);
2348 
2349       // The array's TypeKlassPtr was declared 'precise' or 'not precise'
2350       // according to the element type's subclassing.
2351       return TypeKlassPtr::make(tkls->ptr(), elem, Type::Offset(0), elem->flatten_array());
2352     } else if (klass->is_value_array_klass() &&
2353                tkls->offset() == in_bytes(ObjArrayKlass::element_klass_offset())) {
2354       ciKlass* elem = klass->as_value_array_klass()->element_klass();
2355       return TypeKlassPtr::make(tkls->ptr(), elem, Type::Offset(0), true);
2356     }
2357     if( klass->is_instance_klass() && tkls->klass_is_exact() &&
2358         tkls->offset() == in_bytes(Klass::super_offset())) {
2359       ciKlass* sup = klass->as_instance_klass()->super();
2360       // The field is Klass::_super.  Return its (constant) value.
2361       // (Folds up the 2nd indirection in aClassConstant.getSuperClass().)
2362       return sup ? TypeKlassPtr::make(sup) : TypePtr::NULL_PTR;
2363     }
2364   }
2365 
2366   // Bailout case
2367   return LoadNode::Value(phase);
2368 }

2423 Node* LoadNode::klass_identity_common(PhaseGVN* phase) {
2424   Node* x = LoadNode::Identity(phase);
2425   if (x != this)  return x;
2426 
2427   // Take apart the address into an oop and and offset.
2428   // Return 'this' if we cannot.
2429   Node*    adr    = in(MemNode::Address);
2430   intptr_t offset = 0;
2431   Node*    base   = AddPNode::Ideal_base_and_offset(adr, phase, offset);
2432   if (base == NULL)     return this;
2433   const TypeOopPtr* toop = phase->type(adr)->isa_oopptr();
2434   if (toop == NULL)     return this;
2435 
2436   // Step over potential GC barrier for OopHandle resolve
2437   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
2438   if (bs->is_gc_barrier_node(base)) {
2439     base = bs->step_over_gc_barrier(base);
2440   }
2441 
2442   // We can fetch the klass directly through an AllocateNode.
2443   // This works even if the klass is not constant (clone or newArray).
2444   if (offset == oopDesc::klass_offset_in_bytes()) {
2445     Node* allocated_klass = AllocateNode::Ideal_klass(base, phase);
2446     if (allocated_klass != NULL) {
2447       return allocated_klass;
2448     }
2449   }
2450 
2451   // Simplify k.java_mirror.as_klass to plain k, where k is a Klass*.
2452   // See inline_native_Class_query for occurrences of these patterns.
2453   // Java Example:  x.getClass().isAssignableFrom(y)
2454   //
2455   // This improves reflective code, often making the Class
2456   // mirror go completely dead.  (Current exception:  Class
2457   // mirrors may appear in debug info, but we could clean them out by
2458   // introducing a new debug info operator for Klass.java_mirror).
2459 
2460   if (toop->isa_instptr() && toop->klass() == phase->C->env()->Class_klass()
2461       && offset == java_lang_Class::klass_offset_in_bytes()) {
2462     if (base->is_Load()) {
2463       Node* base2 = base->in(MemNode::Address);
2464       if (base2->is_Load()) { /* direct load of a load which is the OopHandle */
2465         Node* adr2 = base2->in(MemNode::Address);
2466         const TypeKlassPtr* tkls = phase->type(adr2)->isa_klassptr();
2467         if (tkls != NULL && !tkls->empty()
2468             && (tkls->klass()->is_instance_klass() ||
2469               tkls->klass()->is_array_klass())
2470             && adr2->is_AddP()
2471            ) {
2472           int mirror_field = in_bytes(Klass::java_mirror_offset());
2473           if (tkls->offset() == mirror_field) {
2474             return adr2->in(AddPNode::Base);
2475           }
2476         }
2477       }
2478     }
2479   }
2480 
2481   return this;
2482 }

2656 //------------------------------Ideal------------------------------------------
2657 // Change back-to-back Store(, p, x) -> Store(m, p, y) to Store(m, p, x).
2658 // When a store immediately follows a relevant allocation/initialization,
2659 // try to capture it into the initialization, or hoist it above.
2660 Node *StoreNode::Ideal(PhaseGVN *phase, bool can_reshape) {
2661   Node* p = MemNode::Ideal_common(phase, can_reshape);
2662   if (p)  return (p == NodeSentinel) ? NULL : p;
2663 
2664   Node* mem     = in(MemNode::Memory);
2665   Node* address = in(MemNode::Address);
2666   // Back-to-back stores to same address?  Fold em up.  Generally
2667   // unsafe if I have intervening uses...  Also disallowed for StoreCM
2668   // since they must follow each StoreP operation.  Redundant StoreCMs
2669   // are eliminated just before matching in final_graph_reshape.
2670   if (phase->C->get_adr_type(phase->C->get_alias_index(adr_type())) != TypeAryPtr::VALUES) {
2671     Node* st = mem;
2672     // If Store 'st' has more than one use, we cannot fold 'st' away.
2673     // For example, 'st' might be the final state at a conditional
2674     // return.  Or, 'st' might be used by some node which is live at
2675     // the same time 'st' is live, which might be unschedulable.  So,
2676     // require exactly ONE user until such time as we clone 'mem' for
2677     // each of 'mem's uses (thus making the exactly-1-user-rule hold
2678     // true).
2679     while (st->is_Store() && st->outcnt() == 1 && st->Opcode() != Op_StoreCM) {
2680       // Looking at a dead closed cycle of memory?
2681       assert(st != st->in(MemNode::Memory), "dead loop in StoreNode::Ideal");
2682       assert(Opcode() == st->Opcode() ||
2683              st->Opcode() == Op_StoreVector ||
2684              Opcode() == Op_StoreVector ||
2685              phase->C->get_alias_index(adr_type()) == Compile::AliasIdxRaw ||
2686              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreI) || // expanded ClearArrayNode
2687              (Opcode() == Op_StoreI && st->Opcode() == Op_StoreL) || // initialization by arraycopy
2688              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreN) ||
2689              (is_mismatched_access() || st->as_Store()->is_mismatched_access()),
2690              "no mismatched stores, except on raw memory: %s %s", NodeClassNames[Opcode()], NodeClassNames[st->Opcode()]);
2691 
2692       if (st->in(MemNode::Address)->eqv_uncast(address) &&
2693           st->as_Store()->memory_size() <= this->memory_size()) {
2694         Node* use = st->raw_out(0);
2695         phase->igvn_rehash_node_delayed(use);
2696         if (can_reshape) {
2697           use->set_req_X(MemNode::Memory, st->in(MemNode::Memory), phase->is_IterGVN());
2698         } else {
2699           // It's OK to do this in the parser, since DU info is always accurate,
2700           // and the parser always refers to nodes via SafePointNode maps.
2701           use->set_req(MemNode::Memory, st->in(MemNode::Memory));
2702         }
2703         return this;
2704       }
2705       st = st->in(MemNode::Memory);
2706     }
2707   }
2708 
2709 
2710   // Capture an unaliased, unconditional, simple store into an initializer.
2711   // Or, if it is independent of the allocation, hoist it above the allocation.
2712   if (ReduceFieldZeroing && /*can_reshape &&*/
2713       mem->is_Proj() && mem->in(0)->is_Initialize()) {
2714     InitializeNode* init = mem->in(0)->as_Initialize();
2715     intptr_t offset = init->can_capture_store(this, phase, can_reshape);
2716     if (offset > 0) {
2717       Node* moved = init->capture_store(this, offset, phase, can_reshape);
2718       // If the InitializeNode captured me, it made a raw copy of me,
2719       // and I need to disappear.
2720       if (moved != NULL) {
2721         // %%% hack to ensure that Ideal returns a new node:
2722         mem = MergeMemNode::make(mem);
2723         return mem;             // fold me away
2724       }
2725     }
2726   }
2727 
2728   return NULL;                  // No further progress
2729 }

2743 //------------------------------Identity---------------------------------------
2744 // Remove redundant stores:
2745 //   Store(m, p, Load(m, p)) changes to m.
2746 //   Store(, p, x) -> Store(m, p, x) changes to Store(m, p, x).
2747 Node* StoreNode::Identity(PhaseGVN* phase) {
2748   Node* mem = in(MemNode::Memory);
2749   Node* adr = in(MemNode::Address);
2750   Node* val = in(MemNode::ValueIn);
2751 
2752   Node* result = this;
2753 
2754   // Load then Store?  Then the Store is useless
2755   if (val->is_Load() &&
2756       val->in(MemNode::Address)->eqv_uncast(adr) &&
2757       val->in(MemNode::Memory )->eqv_uncast(mem) &&
2758       val->as_Load()->store_Opcode() == Opcode()) {
2759     result = mem;
2760   }
2761 
2762   // Two stores in a row of the same value?
2763   if (result == this &&
2764       mem->is_Store() &&
2765       mem->in(MemNode::Address)->eqv_uncast(adr) &&
2766       mem->in(MemNode::ValueIn)->eqv_uncast(val) &&
2767       mem->Opcode() == Opcode()) {
2768     result = mem;
2769   }
2770 
2771   // Store of zero anywhere into a freshly-allocated object?
2772   // Then the store is useless.
2773   // (It must already have been captured by the InitializeNode.)
2774   if (result == this && ReduceFieldZeroing) {
2775     // a newly allocated object is already all-zeroes everywhere
2776     if (mem->is_Proj() && mem->in(0)->is_Allocate() &&
2777         (phase->type(val)->is_zero_type() || mem->in(0)->in(AllocateNode::DefaultValue) == val)) {
2778       assert(!phase->type(val)->is_zero_type() || mem->in(0)->in(AllocateNode::DefaultValue) == NULL, "storing null to value array is forbidden");
2779       result = mem;
2780     }
2781 
2782     if (result == this) {
2783       // the store may also apply to zero-bits in an earlier object
2784       Node* prev_mem = find_previous_store(phase);
2785       // Steps (a), (b):  Walk past independent stores to find an exact match.
2786       if (prev_mem != NULL) {
2787         Node* prev_val = can_see_stored_value(prev_mem, phase);
2788         if (prev_val != NULL && phase->eqv(prev_val, val)) {
2789           // prev_val and val might differ by a cast; it would be good
2790           // to keep the more informative of the two.
2791           if (phase->type(val)->is_zero_type()) {
2792             result = mem;
2793           } else if (prev_mem->is_Proj() && prev_mem->in(0)->is_Initialize()) {
2794             InitializeNode* init = prev_mem->in(0)->as_Initialize();
2795             AllocateNode* alloc = init->allocation();
2796             if (alloc != NULL && alloc->in(AllocateNode::DefaultValue) == val) {
2797               result = mem;
2798             }
2799           }
2800         }
2801       }
2802     }
2803   }
2804 
2805   if (result != this && phase->is_IterGVN() != NULL) {
2806     MemBarNode* trailing = trailing_membar();
2807     if (trailing != NULL) {
2808 #ifdef ASSERT
2809       const TypeOopPtr* t_oop = phase->type(in(Address))->isa_oopptr();
2810       assert(t_oop == NULL || t_oop->is_known_instance_field(), "only for non escaping objects");
2811 #endif
2812       PhaseIterGVN* igvn = phase->is_IterGVN();
2813       trailing->remove(igvn);
2814     }
2815   }
2816 
2817   return result;
2818 }

3317 //------------------------------Ideal------------------------------------------
3318 // Return a node which is more "ideal" than the current node.  Strip out
3319 // control copies
3320 Node *MemBarNode::Ideal(PhaseGVN *phase, bool can_reshape) {
3321   if (remove_dead_region(phase, can_reshape)) return this;
3322   // Don't bother trying to transform a dead node
3323   if (in(0) && in(0)->is_top()) {
3324     return NULL;
3325   }
3326 
3327   bool progress = false;
3328   // Eliminate volatile MemBars for scalar replaced objects.
3329   if (can_reshape && req() == (Precedent+1)) {
3330     bool eliminate = false;
3331     int opc = Opcode();
3332     if ((opc == Op_MemBarAcquire || opc == Op_MemBarVolatile)) {
3333       // Volatile field loads and stores.
3334       Node* my_mem = in(MemBarNode::Precedent);
3335       // The MembarAquire may keep an unused LoadNode alive through the Precedent edge
3336       if ((my_mem != NULL) && (opc == Op_MemBarAcquire) && (my_mem->outcnt() == 1)) {
3337         // if the Precedent is a decodeN and its input (a Load) is used at more than one place,
3338         // replace this Precedent (decodeN) with the Load instead.
3339         if ((my_mem->Opcode() == Op_DecodeN) && (my_mem->in(1)->outcnt() > 1))  {
3340           Node* load_node = my_mem->in(1);
3341           set_req(MemBarNode::Precedent, load_node);
3342           phase->is_IterGVN()->_worklist.push(my_mem);
3343           my_mem = load_node;
3344         } else {
3345           assert(my_mem->unique_out() == this, "sanity");
3346           del_req(Precedent);
3347           phase->is_IterGVN()->_worklist.push(my_mem); // remove dead node later
3348           my_mem = NULL;
3349         }
3350         progress = true;
3351       }
3352       if (my_mem != NULL && my_mem->is_Mem()) {
3353         const TypeOopPtr* t_oop = my_mem->in(MemNode::Address)->bottom_type()->isa_oopptr();
3354         // Check for scalar replaced object reference.
3355         if( t_oop != NULL && t_oop->is_known_instance_field() &&
3356             t_oop->offset() != Type::OffsetBot &&
3357             t_oop->offset() != Type::OffsetTop) {
3358           eliminate = true;
3359         }
3360       }
3361     } else if (opc == Op_MemBarRelease) {
3362       // Final field stores.
3363       Node* alloc = AllocateNode::Ideal_allocation(in(MemBarNode::Precedent), phase);

3375       // (Do not return a top here, since that would break the uniqueness of top.)

3532 //===========================InitializeNode====================================
3533 // SUMMARY:
3534 // This node acts as a memory barrier on raw memory, after some raw stores.
3535 // The 'cooked' oop value feeds from the Initialize, not the Allocation.
3536 // The Initialize can 'capture' suitably constrained stores as raw inits.
3537 // It can coalesce related raw stores into larger units (called 'tiles').
3538 // It can avoid zeroing new storage for memory units which have raw inits.
3539 // At macro-expansion, it is marked 'complete', and does not optimize further.
3540 //
3541 // EXAMPLE:
3542 // The object 'new short[2]' occupies 16 bytes in a 32-bit machine.
3543 //   ctl = incoming control; mem* = incoming memory
3544 // (Note:  A star * on a memory edge denotes I/O and other standard edges.)
3545 // First allocate uninitialized memory and fill in the header:
3546 //   alloc = (Allocate ctl mem* 16 #short[].klass ...)
3547 //   ctl := alloc.Control; mem* := alloc.Memory*
3548 //   rawmem = alloc.Memory; rawoop = alloc.RawAddress
3549 // Then initialize to zero the non-header parts of the raw memory block:
3550 //   init = (Initialize alloc.Control alloc.Memory* alloc.RawAddress)
3551 //   ctl := init.Control; mem.SLICE(#short[*]) := init.Memory
3552 // After the initialize node executes, the object is ready for service:
3553 //   oop := (CheckCastPP init.Control alloc.RawAddress #short[])
3554 // Suppose its body is immediately initialized as {1,2}:
3555 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3556 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3557 //   mem.SLICE(#short[*]) := store2
3558 //
3559 // DETAILS:
3560 // An InitializeNode collects and isolates object initialization after
3561 // an AllocateNode and before the next possible safepoint.  As a
3562 // memory barrier (MemBarNode), it keeps critical stores from drifting
3563 // down past any safepoint or any publication of the allocation.
3564 // Before this barrier, a newly-allocated object may have uninitialized bits.
3565 // After this barrier, it may be treated as a real oop, and GC is allowed.
3566 //
3567 // The semantics of the InitializeNode include an implicit zeroing of
3568 // the new object from object header to the end of the object.
3569 // (The object header and end are determined by the AllocateNode.)
3570 //
3571 // Certain stores may be added as direct inputs to the InitializeNode.
3572 // These stores must update raw memory, and they must be to addresses
3573 // derived from the raw address produced by AllocateNode, and with
3574 // a constant offset.  They must be ordered by increasing offset.
3575 // The first one is at in(RawStores), the last at in(req()-1).
3576 // Unlike most memory operations, they are not linked in a chain,
3577 // but are displayed in parallel as users of the rawmem output of
3578 // the allocation.
3579 //
3580 // (See comments in InitializeNode::capture_store, which continue
3581 // the example given above.)
3582 //
3583 // When the associated Allocate is macro-expanded, the InitializeNode
3584 // may be rewritten to optimize collected stores.  A ClearArrayNode
3585 // may also be created at that point to represent any required zeroing.
3586 // The InitializeNode is then marked 'complete', prohibiting further
3587 // capturing of nearby memory operations.
3588 //
3589 // During macro-expansion, all captured initializations which store
3590 // constant values of 32 bits or smaller are coalesced (if advantageous)
3591 // into larger 'tiles' 32 or 64 bits.  This allows an object to be
3592 // initialized in fewer memory operations.  Memory words which are
3593 // covered by neither tiles nor non-constant stores are pre-zeroed
3594 // by explicit stores of zero.  (The code shape happens to do all
3595 // zeroing first, then all other stores, with both sequences occurring
3596 // in order of ascending offsets.)
3597 //
3598 // Alternatively, code may be inserted between an AllocateNode and its
3599 // InitializeNode, to perform arbitrary initialization of the new object.
3600 // E.g., the object copying intrinsics insert complex data transfers here.
3601 // The initialization must then be marked as 'complete' disable the
3602 // built-in zeroing semantics and the collection of initializing stores.
3603 //
3604 // While an InitializeNode is incomplete, reads from the memory state
3605 // produced by it are optimizable if they match the control edge and
3606 // new oop address associated with the allocation/initialization.
3607 // They return a stored value (if the offset matches) or else zero.
3608 // A write to the memory state, if it matches control and address,
3609 // and if it is to a constant offset, may be 'captured' by the
3610 // InitializeNode.  It is cloned as a raw memory operation and rewired
3611 // inside the initialization, to the raw oop produced by the allocation.
3612 // Operations on addresses which are provably distinct (e.g., to
3613 // other AllocateNodes) are allowed to bypass the initialization.
3614 //
3615 // The effect of all this is to consolidate object initialization
3616 // (both arrays and non-arrays, both piecewise and bulk) into a
3617 // single location, where it can be optimized as a unit.
3618 //
3619 // Only stores with an offset less than TrackedInitializationLimit words
3620 // will be considered for capture by an InitializeNode.  This puts a
3621 // reasonable limit on the complexity of optimized initializations.
3622

3626     _is_complete(Incomplete), _does_not_escape(false)

3715 // Helper for proving that an initialization expression is
3716 // "simple enough" to be folded into an object initialization.
3717 // Attempts to prove that a store's initial value 'n' can be captured
3718 // within the initialization without creating a vicious cycle, such as:
3719 //     { Foo p = new Foo(); p.next = p; }
3720 // True for constants and parameters and small combinations thereof.
3721 bool InitializeNode::detect_init_independence(Node* value, PhaseGVN* phase) {
3722   ResourceMark rm;
3723   Unique_Node_List worklist;
3724   worklist.push(value);
3725 
3726   uint complexity_limit = 20;
3727   for (uint j = 0; j < worklist.size(); j++) {
3728     if (j >= complexity_limit) {
3729       return false;  // Bail out if processed too many nodes
3730     }
3731 
3732     Node* n = worklist.at(j);
3733     if (n == NULL)      continue;   // (can this really happen?)
3734     if (n->is_Proj())   n = n->in(0);
3735     if (n == this)      return false;  // found a cycle
3736     if (n->is_Con())    continue;
3737     if (n->is_Start())  continue;   // params, etc., are OK
3738     if (n->is_Root())   continue;   // even better
3739 
3740     // There cannot be any dependency if 'n' is a CFG node that dominates the current allocation
3741     if (n->is_CFG() && phase->is_dominator(n, allocation())) {
3742       continue;
3743     }
3744 
3745     Node* ctl = n->in(0);
3746     if (ctl != NULL && !ctl->is_top()) {
3747       if (ctl->is_Proj())  ctl = ctl->in(0);
3748       if (ctl == this)  return false;
3749 
3750       // If we already know that the enclosing memory op is pinned right after
3751       // the init, then any control flow that the store has picked up
3752       // must have preceded the init, or else be equal to the init.
3753       // Even after loop optimizations (which might change control edges)
3754       // a store is never pinned *before* the availability of its inputs.
3755       if (!MemNode::all_controls_dominate(n, this))
3756         return false;                  // failed to prove a good control
3757     }
3758 
3759     // Check data edges for possible dependencies on 'this'.
3760     for (uint i = 1; i < n->req(); i++) {
3761       Node* m = n->in(i);
3762       if (m == NULL || m == n || m->is_top())  continue;
3763 
3764       // Only process data inputs once
3765       worklist.push(m);
3766     }
3767   }
3768 
3769   return true;
3770 }

3772 // Here are all the checks a Store must pass before it can be moved into
3773 // an initialization.  Returns zero if a check fails.
3774 // On success, returns the (constant) offset to which the store applies,
3775 // within the initialized memory.
3776 intptr_t InitializeNode::can_capture_store(StoreNode* st, PhaseGVN* phase, bool can_reshape) {
3777   const int FAIL = 0;
3778   if (st->req() != MemNode::ValueIn + 1)
3779     return FAIL;                // an inscrutable StoreNode (card mark?)
3780   Node* ctl = st->in(MemNode::Control);
3781   if (!(ctl != NULL && ctl->is_Proj() && ctl->in(0) == this))
3782     return FAIL;                // must be unconditional after the initialization
3783   Node* mem = st->in(MemNode::Memory);
3784   if (!(mem->is_Proj() && mem->in(0) == this))
3785     return FAIL;                // must not be preceded by other stores
3786   Node* adr = st->in(MemNode::Address);
3787   intptr_t offset;
3788   AllocateNode* alloc = AllocateNode::Ideal_allocation(adr, phase, offset);
3789   if (alloc == NULL)
3790     return FAIL;                // inscrutable address
3791   if (alloc != allocation())
3792     return FAIL;                // wrong allocation!  (store needs to float up)
3793   int size_in_bytes = st->memory_size();
3794   if ((size_in_bytes != 0) && (offset % size_in_bytes) != 0) {
3795     return FAIL;                // mismatched access
3796   }
3797   Node* val = st->in(MemNode::ValueIn);
3798 
3799   if (!detect_init_independence(val, phase))
3800     return FAIL;                // stored value must be 'simple enough'
3801 
3802   // The Store can be captured only if nothing after the allocation
3803   // and before the Store is using the memory location that the store
3804   // overwrites.
3805   bool failed = false;
3806   // If is_complete_with_arraycopy() is true the shape of the graph is
3807   // well defined and is safe so no need for extra checks.
3808   if (!is_complete_with_arraycopy()) {
3809     // We are going to look at each use of the memory state following
3810     // the allocation to make sure nothing reads the memory that the
3811     // Store writes.
3812     const TypePtr* t_adr = phase->type(adr)->isa_ptr();
3813     int alias_idx = phase->C->get_alias_index(t_adr);
3814     ResourceMark rm;
3815     Unique_Node_List mems;
3816     mems.push(mem);
3817     Node* unique_merge = NULL;
3818     for (uint next = 0; next < mems.size(); ++next) {
3819       Node *m  = mems.at(next);
3820       for (DUIterator_Fast jmax, j = m->fast_outs(jmax); j < jmax; j++) {
3821         Node *n = m->fast_out(j);
3822         if (n->outcnt() == 0) {
3823           continue;
3824         }
3825         if (n == st) {
3826           continue;
3827         } else if (n->in(0) != NULL && n->in(0) != ctl) {
3828           // If the control of this use is different from the control
3829           // of the Store which is right after the InitializeNode then
3830           // this node cannot be between the InitializeNode and the
3831           // Store.
3832           continue;
3833         } else if (n->is_MergeMem()) {
3834           if (n->as_MergeMem()->memory_at(alias_idx) == m) {
3835             // We can hit a MergeMemNode (that will likely go away
3836             // later) that is a direct use of the memory state
3837             // following the InitializeNode on the same slice as the
3838             // store node that we'd like to capture. We need to check
3839             // the uses of the MergeMemNode.
3840             mems.push(n);
3841           }
3842         } else if (n->is_Mem()) {
3843           Node* other_adr = n->in(MemNode::Address);
3844           if (other_adr == adr) {
3845             failed = true;
3846             break;
3847           } else {
3848             const TypePtr* other_t_adr = phase->type(other_adr)->isa_ptr();
3849             if (other_t_adr != NULL) {
3850               int other_alias_idx = phase->C->get_alias_index(other_t_adr);
3851               if (other_alias_idx == alias_idx) {
3852                 // A load from the same memory slice as the store right
3853                 // after the InitializeNode. We check the control of the
3854                 // object/array that is loaded from. If it's the same as
3855                 // the store control then we cannot capture the store.
3856                 assert(!n->is_Store(), "2 stores to same slice on same control?");
3857                 Node* base = other_adr;
3858                 assert(base->is_AddP(), "should be addp but is %s", base->Name());
3859                 base = base->in(AddPNode::Base);
3860                 if (base != NULL) {
3861                   base = base->uncast();
3862                   if (base->is_Proj() && base->in(0) == alloc) {
3863                     failed = true;
3864                     break;
3865                   }
3866                 }
3867               }
3868             }
3869           }
3870         } else {
3871           failed = true;
3872           break;
3873         }
3874       }
3875     }
3876   }
3877   if (failed) {
3878     if (!can_reshape) {
3879       // We decided we couldn't capture the store during parsing. We
3880       // should try again during the next IGVN once the graph is
3881       // cleaner.
3882       phase->C->record_for_igvn(st);
3883     }
3884     return FAIL;
3885   }
3886 
3887   return offset;                // success
3888 }

3890 // Find the captured store in(i) which corresponds to the range
3891 // [start..start+size) in the initialized object.
3892 // If there is one, return its index i.  If there isn't, return the
3893 // negative of the index where it should be inserted.
3894 // Return 0 if the queried range overlaps an initialization boundary
3895 // or if dead code is encountered.
3896 // If size_in_bytes is zero, do not bother with overlap checks.
3897 int InitializeNode::captured_store_insertion_point(intptr_t start,
3898                                                    int size_in_bytes,
3899                                                    PhaseTransform* phase) {
3900   const int FAIL = 0, MAX_STORE = BytesPerLong;
3901 
3902   if (is_complete())
3903     return FAIL;                // arraycopy got here first; punt
3904 
3905   assert(allocation() != NULL, "must be present");
3906 
3907   // no negatives, no header fields:
3908   if (start < (intptr_t) allocation()->minimum_header_size())  return FAIL;
3909 
3910   // after a certain size, we bail out on tracking all the stores:
3911   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
3912   if (start >= ti_limit)  return FAIL;
3913 
3914   for (uint i = InitializeNode::RawStores, limit = req(); ; ) {
3915     if (i >= limit)  return -(int)i; // not found; here is where to put it
3916 
3917     Node*    st     = in(i);
3918     intptr_t st_off = get_store_offset(st, phase);
3919     if (st_off < 0) {
3920       if (st != zero_memory()) {
3921         return FAIL;            // bail out if there is dead garbage
3922       }
3923     } else if (st_off > start) {
3924       // ...we are done, since stores are ordered
3925       if (st_off < start + size_in_bytes) {
3926         return FAIL;            // the next store overlaps
3927       }
3928       return -(int)i;           // not found; here is where to put it
3929     } else if (st_off < start) {
3930       if (size_in_bytes != 0 &&
3931           start < st_off + MAX_STORE &&
3932           start < st_off + st->as_Store()->memory_size()) {
3933         return FAIL;            // the previous store overlaps
3934       }
3935     } else {
3936       if (size_in_bytes != 0 &&
3937           st->as_Store()->memory_size() != size_in_bytes) {
3938         return FAIL;            // mismatched store size
3939       }
3940       return i;
3941     }
3942 
3943     ++i;
3944   }
3945 }

3978 // Clone the given store, converting it into a raw store
3979 // initializing a field or element of my new object.
3980 // Caller is responsible for retiring the original store,
3981 // with subsume_node or the like.
3982 //
3983 // From the example above InitializeNode::InitializeNode,
3984 // here are the old stores to be captured:
3985 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3986 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3987 //
3988 // Here is the changed code; note the extra edges on init:
3989 //   alloc = (Allocate ...)
3990 //   rawoop = alloc.RawAddress
3991 //   rawstore1 = (StoreC alloc.Control alloc.Memory (+ rawoop 12) 1)
3992 //   rawstore2 = (StoreC alloc.Control alloc.Memory (+ rawoop 14) 2)
3993 //   init = (Initialize alloc.Control alloc.Memory rawoop
3994 //                      rawstore1 rawstore2)
3995 //
3996 Node* InitializeNode::capture_store(StoreNode* st, intptr_t start,
3997                                     PhaseGVN* phase, bool can_reshape) {
3998   assert(stores_are_sane(phase), "");
3999 
4000   if (start < 0)  return NULL;
4001   assert(can_capture_store(st, phase, can_reshape) == start, "sanity");
4002 
4003   Compile* C = phase->C;
4004   int size_in_bytes = st->memory_size();
4005   int i = captured_store_insertion_point(start, size_in_bytes, phase);
4006   if (i == 0)  return NULL;     // bail out
4007   Node* prev_mem = NULL;        // raw memory for the captured store
4008   if (i > 0) {
4009     prev_mem = in(i);           // there is a pre-existing store under this one
4010     set_req(i, C->top());       // temporarily disconnect it
4011     // See StoreNode::Ideal 'st->outcnt() == 1' for the reason to disconnect.
4012   } else {
4013     i = -i;                     // no pre-existing store
4014     prev_mem = zero_memory();   // a slice of the newly allocated object
4015     if (i > InitializeNode::RawStores && in(i-1) == prev_mem)
4016       set_req(--i, C->top());   // reuse this edge; it has been folded away
4017     else
4018       ins_req(i, C->top());     // build a new edge
4019   }
4020   Node* new_st = st->clone();
4021   new_st->set_req(MemNode::Control, in(Control));
4022   new_st->set_req(MemNode::Memory,  prev_mem);
4023   new_st->set_req(MemNode::Address, make_raw_address(start, phase));
4024   new_st = phase->transform(new_st);
4025 
4026   // At this point, new_st might have swallowed a pre-existing store
4027   // at the same offset, or perhaps new_st might have disappeared,
4028   // if it redundantly stored the same value (or zero to fresh memory).
4029 
4030   // In any case, wire it in:
4031   phase->igvn_rehash_node_delayed(this);
4032   set_req(i, new_st);
4033 
4034   // The caller may now kill the old guy.
4035   DEBUG_ONLY(Node* check_st = find_captured_store(start, size_in_bytes, phase));
4036   assert(check_st == new_st || check_st == NULL, "must be findable");

4052   case sizeof(jlong):  *(jlong*) addr = (jlong) con; break;

4055   return true;                  // return success to caller

4058 // Coalesce subword constants into int constants and possibly
4059 // into long constants.  The goal, if the CPU permits,
4060 // is to initialize the object with a small number of 64-bit tiles.
4061 // Also, convert floating-point constants to bit patterns.
4062 // Non-constants are not relevant to this pass.
4063 //
4064 // In terms of the running example on InitializeNode::InitializeNode
4065 // and InitializeNode::capture_store, here is the transformation
4066 // of rawstore1 and rawstore2 into rawstore12:
4067 //   alloc = (Allocate ...)
4068 //   rawoop = alloc.RawAddress
4069 //   tile12 = 0x00010002
4070 //   rawstore12 = (StoreI alloc.Control alloc.Memory (+ rawoop 12) tile12)
4071 //   init = (Initialize alloc.Control alloc.Memory rawoop rawstore12)
4072 //
4073 void
4074 InitializeNode::coalesce_subword_stores(intptr_t header_size,
4075                                         Node* size_in_bytes,
4076                                         PhaseGVN* phase) {
4077   Compile* C = phase->C;
4078 
4079   assert(stores_are_sane(phase), "");
4080   // Note:  After this pass, they are not completely sane,
4081   // since there may be some overlaps.
4082 
4083   int old_subword = 0, old_long = 0, new_int = 0, new_long = 0;
4084 
4085   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
4086   intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, ti_limit);
4087   size_limit = MIN2(size_limit, ti_limit);
4088   size_limit = align_up(size_limit, BytesPerLong);
4089   int num_tiles = size_limit / BytesPerLong;
4090 
4091   // allocate space for the tile map:
4092   const int small_len = DEBUG_ONLY(true ? 3 :) 30; // keep stack frames small
4093   jlong  tiles_buf[small_len];
4094   Node*  nodes_buf[small_len];
4095   jlong  inits_buf[small_len];
4096   jlong* tiles = ((num_tiles <= small_len) ? &tiles_buf[0]
4097                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
4098   Node** nodes = ((num_tiles <= small_len) ? &nodes_buf[0]
4099                   : NEW_RESOURCE_ARRAY(Node*, num_tiles));
4100   jlong* inits = ((num_tiles <= small_len) ? &inits_buf[0]
4101                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
4102   // tiles: exact bitwise model of all primitive constants
4103   // nodes: last constant-storing node subsumed into the tiles model
4104   // inits: which bytes (in each tile) are touched by any initializations
4105 
4106   //// Pass A: Fill in the tile model with any relevant stores.
4107 
4108   Copy::zero_to_bytes(tiles, sizeof(tiles[0]) * num_tiles);
4109   Copy::zero_to_bytes(nodes, sizeof(nodes[0]) * num_tiles);
4110   Copy::zero_to_bytes(inits, sizeof(inits[0]) * num_tiles);
4111   Node* zmem = zero_memory(); // initially zero memory state
4112   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
4113     Node* st = in(i);
4114     intptr_t st_off = get_store_offset(st, phase);
4115 
4116     // Figure out the store's offset and constant value:
4117     if (st_off < header_size)             continue; //skip (ignore header)
4118     if (st->in(MemNode::Memory) != zmem)  continue; //skip (odd store chain)
4119     int st_size = st->as_Store()->memory_size();
4120     if (st_off + st_size > size_limit)    break;
4121 
4122     // Record which bytes are touched, whether by constant or not.
4123     if (!store_constant(inits, num_tiles, st_off, st_size, (jlong) -1))
4124       continue;                 // skip (strange store size)
4125 
4126     const Type* val = phase->type(st->in(MemNode::ValueIn));
4127     if (!val->singleton())                continue; //skip (non-con store)
4128     BasicType type = val->basic_type();
4129 
4130     jlong con = 0;
4131     switch (type) {
4132     case T_INT:    con = val->is_int()->get_con();  break;
4133     case T_LONG:   con = val->is_long()->get_con(); break;
4134     case T_FLOAT:  con = jint_cast(val->getf());    break;
4135     case T_DOUBLE: con = jlong_cast(val->getd());   break;
4136     default:                              continue; //skip (odd store type)
4137     }
4138 
4139     if (type == T_LONG && Matcher::isSimpleConstant64(con) &&
4140         st->Opcode() == Op_StoreL) {
4141       continue;                 // This StoreL is already optimal.
4142     }
4143 
4144     // Store down the constant.
4145     store_constant(tiles, num_tiles, st_off, st_size, con);
4146 
4147     intptr_t j = st_off >> LogBytesPerLong;
4148 
4149     if (type == T_INT && st_size == BytesPerInt
4150         && (st_off & BytesPerInt) == BytesPerInt) {
4151       jlong lcon = tiles[j];
4152       if (!Matcher::isSimpleConstant64(lcon) &&
4153           st->Opcode() == Op_StoreI) {
4154         // This StoreI is already optimal by itself.
4155         jint* intcon = (jint*) &tiles[j];
4156         intcon[1] = 0;  // undo the store_constant()
4157 
4158         // If the previous store is also optimal by itself, back up and
4159         // undo the action of the previous loop iteration... if we can.
4160         // But if we can't, just let the previous half take care of itself.
4161         st = nodes[j];
4162         st_off -= BytesPerInt;
4163         con = intcon[0];
4164         if (con != 0 && st != NULL && st->Opcode() == Op_StoreI) {
4165           assert(st_off >= header_size, "still ignoring header");
4166           assert(get_store_offset(st, phase) == st_off, "must be");
4167           assert(in(i-1) == zmem, "must be");

4194   for (int j = 0; j < num_tiles; j++) {

4199     jint init0, init1;          // split the init map, address-wise

4201       u.con = con;
4202       con0  = u.intcon[0];
4203       con1  = u.intcon[1];
4204       u.con = init;
4205       init0 = u.intcon[0];
4206       init1 = u.intcon[1];

4221       split = true;             // first word is covered by full inits
4222       // Example:  int a[] = { ... foo(), 42 ... }
4223     } else if (con1 == 0 && init1 == -1) {
4224       split = true;             // second word is covered by full inits
4225       // Example:  int a[] = { ... 42, foo() ... }

4229     //   byte a[] = { ... 0,0,foo(),0,  0,0,0,42 ... }
4230     // Assuming big-endian memory, init0, init1 are 0x0000FF00, 0x000000FF.
4231     // In this case the tile is not split; it is (jlong)42.

4240     Node*    st[2];
4241     intptr_t off[2];

4243     if (!split) {
4244       ++new_long;

4246       st[nst++] = StoreNode::make(*phase, ctl, zmem, adr, atp,
4247                                   phase->longcon(con), T_LONG, MemNode::unordered);
4248     } else {
4249       // Omit either if it is a zero.
4250       if (con0 != 0) {
4251         ++new_int;

4254                                     phase->intcon(con0), T_INT, MemNode::unordered);
4255       }
4256       if (con1 != 0) {
4257         ++new_int;
4258         offset += BytesPerInt;
4259         adr = make_raw_address(offset, phase);
4260         off[nst]  = offset;
4261         st[nst++] = StoreNode::make(*phase, ctl, zmem, adr, atp,
4262                                     phase->intcon(con1), T_INT, MemNode::unordered);
4263       }
4264     }
4265 
4266     // Insert second store first, then the first before the second.
4267     // Insert each one just before any overlapping non-constant stores.
4268     while (nst > 0) {
4269       Node* st1 = st[--nst];
4270       C->copy_node_notes_to(st1, old);
4271       st1 = phase->transform(st1);
4272       offset = off[nst];
4273       assert(offset >= header_size, "do not smash header");
4274       int ins_idx = captured_store_insertion_point(offset, /*size:*/0, phase);
4275       guarantee(ins_idx != 0, "must re-insert constant store");
4276       if (ins_idx < 0)  ins_idx = -ins_idx;  // never overlap
4277       if (ins_idx > InitializeNode::RawStores && in(ins_idx-1) == zmem)
4278         set_req(--ins_idx, st1);
4279       else

4281     }
4282   }

4284   if (PrintCompilation && WizardMode)
4285     tty->print_cr("Changed %d/%d subword/long constants into %d/%d int/long",
4286                   old_subword, old_long, new_int, new_long);

4288     C->log()->elem("comment that='%d/%d subword/long to %d/%d int/long'",
4289                    old_subword, old_long, new_int, new_long);
4290 
4291   // Clean up any remaining occurrences of zmem:
4292   remove_extra_zeroes();
4293 }

4295 // Explore forward from in(start) to find the first fully initialized
4296 // word, and return its offset.  Skip groups of subword stores which
4297 // together initialize full words.  If in(start) is itself part of a
4298 // fully initialized word, return the offset of in(start).  If there
4299 // are no following full-word stores, or if something is fishy, return
4300 // a negative value.
4301 intptr_t InitializeNode::find_next_fullword_store(uint start, PhaseGVN* phase) {
4302   int       int_map = 0;
4303   intptr_t  int_map_off = 0;
4304   const int FULL_MAP = right_n_bits(BytesPerInt);  // the int_map we hope for
4305 
4306   for (uint i = start, limit = req(); i < limit; i++) {
4307     Node* st = in(i);
4308 
4309     intptr_t st_off = get_store_offset(st, phase);
4310     if (st_off < 0)  break;  // return conservative answer
4311 
4312     int st_size = st->as_Store()->memory_size();
4313     if (st_size >= BytesPerInt && (st_off % BytesPerInt) == 0) {
4314       return st_off;            // we found a complete word init
4315     }
4316 
4317     // update the map:
4318 
4319     intptr_t this_int_off = align_down(st_off, BytesPerInt);
4320     if (this_int_off != int_map_off) {
4321       // reset the map:
4322       int_map = 0;
4323       int_map_off = this_int_off;
4324     }
4325 
4326     int subword_off = st_off - this_int_off;
4327     int_map |= right_n_bits(st_size) << subword_off;
4328     if ((int_map & FULL_MAP) == FULL_MAP) {
4329       return this_int_off;      // we found a complete word init
4330     }
4331 
4332     // Did this store hit or cross the word boundary?
4333     intptr_t next_int_off = align_down(st_off + st_size, BytesPerInt);
4334     if (next_int_off == this_int_off + BytesPerInt) {
4335       // We passed the current int, without fully initializing it.
4336       int_map_off = next_int_off;
4337       int_map >>= BytesPerInt;
4338     } else if (next_int_off > this_int_off + BytesPerInt) {
4339       // We passed the current and next int.
4340       return this_int_off + BytesPerInt;
4341     }
4342   }
4343 
4344   return -1;
4345 }

4349 // At this point, we may perform additional optimizations.
4350 // Linearize the stores by ascending offset, to make memory
4351 // activity as coherent as possible.
4352 Node* InitializeNode::complete_stores(Node* rawctl, Node* rawmem, Node* rawptr,
4353                                       intptr_t header_size,
4354                                       Node* size_in_bytes,
4355                                       PhaseIterGVN* phase) {
4356   assert(!is_complete(), "not already complete");
4357   assert(stores_are_sane(phase), "");
4358   assert(allocation() != NULL, "must be present");
4359 
4360   remove_extra_zeroes();
4361 
4362   if (ReduceFieldZeroing || ReduceBulkZeroing)
4363     // reduce instruction count for common initialization patterns
4364     coalesce_subword_stores(header_size, size_in_bytes, phase);
4365 
4366   Node* zmem = zero_memory();   // initially zero memory state
4367   Node* inits = zmem;           // accumulating a linearized chain of inits
4368   #ifdef ASSERT
4369   intptr_t first_offset = allocation()->minimum_header_size();
4370   intptr_t last_init_off = first_offset;  // previous init offset
4371   intptr_t last_init_end = first_offset;  // previous init offset+size
4372   intptr_t last_tile_end = first_offset;  // previous tile offset+size
4373   #endif
4374   intptr_t zeroes_done = header_size;
4375 
4376   bool do_zeroing = true;       // we might give up if inits are very sparse
4377   int  big_init_gaps = 0;       // how many large gaps have we seen?
4378 
4379   if (UseTLAB && ZeroTLAB)  do_zeroing = false;
4380   if (!ReduceFieldZeroing && !ReduceBulkZeroing)  do_zeroing = false;
4381 
4382   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
4383     Node* st = in(i);
4384     intptr_t st_off = get_store_offset(st, phase);
4385     if (st_off < 0)
4386       break;                    // unknown junk in the inits
4387     if (st->in(MemNode::Memory) != zmem)
4388       break;                    // complicated store chains somehow in list
4389 
4390     int st_size = st->as_Store()->memory_size();
4391     intptr_t next_init_off = st_off + st_size;
4392 
4393     if (do_zeroing && zeroes_done < next_init_off) {
4394       // See if this store needs a zero before it or under it.
4395       intptr_t zeroes_needed = st_off;
4396 
4397       if (st_size < BytesPerInt) {
4398         // Look for subword stores which only partially initialize words.
4399         // If we find some, we must lay down some word-level zeroes first,
4400         // underneath the subword stores.
4401         //
4402         // Examples:
4403         //   byte[] a = { p,q,r,s }  =>  a[0]=p,a[1]=q,a[2]=r,a[3]=s
4404         //   byte[] a = { x,y,0,0 }  =>  a[0..3] = 0, a[0]=x,a[1]=y
4405         //   byte[] a = { 0,0,z,0 }  =>  a[0..3] = 0, a[2]=z
4406         //
4407         // Note:  coalesce_subword_stores may have already done this,
4408         // if it was prompted by constant non-zero subword initializers.
4409         // But this case can still arise with non-constant stores.
4410 
4411         intptr_t next_full_store = find_next_fullword_store(i, phase);
4412 
4413         // In the examples above:
4414         //   in(i)          p   q   r   s     x   y     z
4415         //   st_off        12  13  14  15    12  13    14
4416         //   st_size        1   1   1   1     1   1     1
4417         //   next_full_s.  12  16  16  16    16  16    16
4418         //   z's_done      12  16  16  16    12  16    12
4419         //   z's_needed    12  16  16  16    16  16    16
4420         //   zsize          0   0   0   0     4   0     4

4422           // Conservative tack:  Zero to end of current word.
4423           zeroes_needed = align_up(zeroes_needed, BytesPerInt);
4424         } else {
4425           // Zero to beginning of next fully initialized word.
4426           // Or, don't zero at all, if we are already in that word.
4427           assert(next_full_store >= zeroes_needed, "must go forward");
4428           assert((next_full_store & (BytesPerInt-1)) == 0, "even boundary");
4429           zeroes_needed = next_full_store;
4430         }
4431       }
4432 
4433       if (zeroes_needed > zeroes_done) {
4434         intptr_t zsize = zeroes_needed - zeroes_done;
4435         // Do some incremental zeroing on rawmem, in parallel with inits.
4436         zeroes_done = align_down(zeroes_done, BytesPerInt);

4438                                               allocation()->in(AllocateNode::DefaultValue),
4439                                               allocation()->in(AllocateNode::RawDefaultValue),
4440                                               zeroes_done, zeroes_needed,
4441                                               phase);
4442         zeroes_done = zeroes_needed;
4443         if (zsize > InitArrayShortSize && ++big_init_gaps > 2)
4444           do_zeroing = false;   // leave the hole, next time
4445       }
4446     }
4447

4449     phase->replace_input_of(st, MemNode::Memory, inits);
4450     inits = st;                 // put it on the linearized chain
4451     set_req(i, zmem);           // unhook from previous position
4452 
4453     if (zeroes_done == st_off)
4454       zeroes_done = next_init_off;
4455 
4456     assert(!do_zeroing || zeroes_done >= next_init_off, "don't miss any");
4457 
4458     #ifdef ASSERT
4459     // Various order invariants.  Weaker than stores_are_sane because
4460     // a large constant tile can be filled in by smaller non-constant stores.
4461     assert(st_off >= last_init_off, "inits do not reverse");
4462     last_init_off = st_off;
4463     const Type* val = NULL;
4464     if (st_size >= BytesPerInt &&
4465         (val = phase->type(st->in(MemNode::ValueIn)))->singleton() &&
4466         (int)val->basic_type() < (int)T_OBJECT) {
4467       assert(st_off >= last_tile_end, "tiles do not overlap");
4468       assert(st_off >= last_init_end, "tiles do not overwrite inits");
4469       last_tile_end = MAX2(last_tile_end, next_init_off);
4470     } else {
4471       intptr_t st_tile_end = align_up(next_init_off, BytesPerLong);
4472       assert(st_tile_end >= last_tile_end, "inits stay with tiles");
4473       assert(st_off      >= last_init_end, "inits do not overlap");
4474       last_init_end = next_init_off;  // it's a non-tile
4475     }
4476     #endif //ASSERT
4477   }
4478 
4479   remove_extra_zeroes();        // clear out all the zmems left over
4480   add_req(inits);
4481 
4482   if (!(UseTLAB && ZeroTLAB)) {
4483     // If anything remains to be zeroed, zero it all now.
4484     zeroes_done = align_down(zeroes_done, BytesPerInt);
4485     // if it is the last unused 4 bytes of an instance, forget about it
4486     intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, max_jint);
4487     if (zeroes_done + BytesPerLong >= size_limit) {
4488       AllocateNode* alloc = allocation();
4489       assert(alloc != NULL, "must be present");
4490       if (alloc != NULL && alloc->Opcode() == Op_Allocate) {
4491         Node* klass_node = alloc->in(AllocateNode::KlassNode);
4492         ciKlass* k = phase->type(klass_node)->is_klassptr()->klass();
4493         if (zeroes_done == k->layout_helper())
4494           zeroes_done = size_limit;
4495       }
4496     }
4497     if (zeroes_done < size_limit) {
4498       rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4499                                             allocation()->in(AllocateNode::DefaultValue),
4500                                             allocation()->in(AllocateNode::RawDefaultValue),
4501                                             zeroes_done, size_in_bytes, phase);
4502     }
4503   }
4504 
4505   set_complete(phase);
4506   return rawmem;
4507 }

4535 //============================MergeMemNode=====================================
4536 //
4537 // SEMANTICS OF MEMORY MERGES:  A MergeMem is a memory state assembled from several
4538 // contributing store or call operations.  Each contributor provides the memory
4539 // state for a particular "alias type" (see Compile::alias_type).  For example,
4540 // if a MergeMem has an input X for alias category #6, then any memory reference
4541 // to alias category #6 may use X as its memory state input, as an exact equivalent
4542 // to using the MergeMem as a whole.
4543 //   Load<6>( MergeMem(<6>: X, ...), p ) <==> Load<6>(X,p)
4544 //
4545 // (Here, the <N> notation gives the index of the relevant adr_type.)
4546 //
4547 // In one special case (and more cases in the future), alias categories overlap.
4548 // The special alias category "Bot" (Compile::AliasIdxBot) includes all memory
4549 // states.  Therefore, if a MergeMem has only one contributing input W for Bot,
4550 // it is exactly equivalent to that state W:
4551 //   MergeMem(<Bot>: W) <==> W
4552 //
4553 // Usually, the merge has more than one input.  In that case, where inputs
4554 // overlap (i.e., one is Bot), the narrower alias type determines the memory
4555 // state for that type, and the wider alias type (Bot) fills in everywhere else:
4556 //   Load<5>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<5>(W,p)
4557 //   Load<6>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<6>(X,p)
4558 //
4559 // A merge can take a "wide" memory state as one of its narrow inputs.
4560 // This simply means that the merge observes out only the relevant parts of
4561 // the wide input.  That is, wide memory states arriving at narrow merge inputs
4562 // are implicitly "filtered" or "sliced" as necessary.  (This is rare.)
4563 //
4564 // These rules imply that MergeMem nodes may cascade (via their <Bot> links),
4565 // and that memory slices "leak through":
4566 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y)) <==> MergeMem(<Bot>: W, <7>: Y)
4567 //
4568 // But, in such a cascade, repeated memory slices can "block the leak":
4569 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y), <7>: Y') <==> MergeMem(<Bot>: W, <7>: Y')
4570 //
4571 // In the last example, Y is not part of the combined memory state of the
4572 // outermost MergeMem.  The system must, of course, prevent unschedulable
4573 // memory states from arising, so you can be sure that the state Y is somehow
4574 // a precursor to state Y'.
4575 //
4576 //
4577 // REPRESENTATION OF MEMORY MERGES: The indexes used to address the Node::in array
4578 // of each MergeMemNode array are exactly the numerical alias indexes, including
4579 // but not limited to AliasIdxTop, AliasIdxBot, and AliasIdxRaw.  The functions
4580 // Compile::alias_type (and kin) produce and manage these indexes.
4581 //
4582 // By convention, the value of in(AliasIdxTop) (i.e., in(1)) is always the top node.
4583 // (Note that this provides quick access to the top node inside MergeMem methods,
4584 // without the need to reach out via TLS to Compile::current.)
4585 //
4586 // As a consequence of what was just described, a MergeMem that represents a full
4587 // memory state has an edge in(AliasIdxBot) which is a "wide" memory state,
4588 // containing all alias categories.
4589 //
4590 // MergeMem nodes never (?) have control inputs, so in(0) is NULL.
4591 //
4592 // All other edges in(N) (including in(AliasIdxRaw), which is in(3)) are either
4593 // a memory state for the alias type <N>, or else the top node, meaning that
4594 // there is no particular input for that alias type.  Note that the length of
4595 // a MergeMem is variable, and may be extended at any time to accommodate new
4596 // memory states at larger alias indexes.  When merges grow, they are of course
4597 // filled with "top" in the unused in() positions.
4598 //
4599 // This use of top is named "empty_memory()", or "empty_mem" (no-memory) as a variable.
4600 // (Top was chosen because it works smoothly with passes like GCM.)
4601 //
4602 // For convenience, we hardwire the alias index for TypeRawPtr::BOTTOM.  (It is
4603 // the type of random VM bits like TLS references.)  Since it is always the
4604 // first non-Bot memory slice, some low-level loops use it to initialize an
4605 // index variable:  for (i = AliasIdxRaw; i < req(); i++).
4606 //
4607 //
4608 // ACCESSORS:  There is a special accessor MergeMemNode::base_memory which returns
4609 // the distinguished "wide" state.  The accessor MergeMemNode::memory_at(N) returns
4610 // the memory state for alias type <N>, or (if there is no particular slice at <N>,
4611 // it returns the base memory.  To prevent bugs, memory_at does not accept <Top>
4612 // or <Bot> indexes.  The iterator MergeMemStream provides robust iteration over
4613 // MergeMem nodes or pairs of such nodes, ensuring that the non-top edges are visited.
4614 //
4615 // %%%% We may get rid of base_memory as a separate accessor at some point; it isn't
4616 // really that different from the other memory inputs.  An abbreviation called
4617 // "bot_memory()" for "memory_at(AliasIdxBot)" would keep code tidy.
4618 //
4619 //
4620 // PARTIAL MEMORY STATES:  During optimization, MergeMem nodes may arise that represent
4621 // partial memory states.  When a Phi splits through a MergeMem, the copy of the Phi
4622 // that "emerges though" the base memory will be marked as excluding the alias types
4623 // of the other (narrow-memory) copies which "emerged through" the narrow edges:
4624 //
4625 //   Phi<Bot>(U, MergeMem(<Bot>: W, <8>: Y))
4626 //     ==Ideal=>  MergeMem(<Bot>: Phi<Bot-8>(U, W), Phi<8>(U, Y))
4627 //
4628 // This strange "subtraction" effect is necessary to ensure IGVN convergence.
4629 // (It is currently unimplemented.)  As you can see, the resulting merge is
4630 // actually a disjoint union of memory states, rather than an overlay.
4631 //

4693 //------------------------------Ideal------------------------------------------
4694 // This method is invoked recursively on chains of MergeMem nodes
4695 Node *MergeMemNode::Ideal(PhaseGVN *phase, bool can_reshape) {
4696   // Remove chain'd MergeMems
4697   //
4698   // This is delicate, because the each "in(i)" (i >= Raw) is interpreted
4699   // relative to the "in(Bot)".  Since we are patching both at the same time,
4700   // we have to be careful to read each "in(i)" relative to the old "in(Bot)",
4701   // but rewrite each "in(i)" relative to the new "in(Bot)".
4702   Node *progress = NULL;
4703 
4704 
4705   Node* old_base = base_memory();
4706   Node* empty_mem = empty_memory();
4707   if (old_base == empty_mem)
4708     return NULL; // Dead memory path.
4709 
4710   MergeMemNode* old_mbase;
4711   if (old_base != NULL && old_base->is_MergeMem())
4712     old_mbase = old_base->as_MergeMem();
4713   else
4714     old_mbase = NULL;
4715   Node* new_base = old_base;
4716 
4717   // simplify stacked MergeMems in base memory
4718   if (old_mbase)  new_base = old_mbase->base_memory();
4719 
4720   // the base memory might contribute new slices beyond my req()
4721   if (old_mbase)  grow_to_match(old_mbase);
4722 
4723   // Look carefully at the base node if it is a phi.
4724   PhiNode* phi_base;
4725   if (new_base != NULL && new_base->is_Phi())
4726     phi_base = new_base->as_Phi();
4727   else
4728     phi_base = NULL;
4729 
4730   Node*    phi_reg = NULL;
4731   uint     phi_len = (uint)-1;
4732   if (phi_base != NULL && !phi_base->is_copy()) {
4733     // do not examine phi if degraded to a copy
4734     phi_reg = phi_base->region();
4735     phi_len = phi_base->req();
4736     // see if the phi is unfinished
4737     for (uint i = 1; i < phi_len; i++) {
4738       if (phi_base->in(i) == NULL) {
4739         // incomplete phi; do not look at it yet!
4740         phi_reg = NULL;
4741         phi_len = (uint)-1;
4742         break;
4743       }
4744     }
4745   }
4746 
4747   // Note:  We do not call verify_sparse on entry, because inputs
4748   // can normalize to the base_memory via subsume_node or similar
4749   // mechanisms.  This method repairs that damage.
4750 
4751   assert(!old_mbase || old_mbase->is_empty_memory(empty_mem), "consistent sentinels");
4752 
4753   // Look at each slice.
4754   for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4755     Node* old_in = in(i);
4756     // calculate the old memory value
4757     Node* old_mem = old_in;
4758     if (old_mem == empty_mem)  old_mem = old_base;
4759     assert(old_mem == memory_at(i), "");
4760 
4761     // maybe update (reslice) the old memory value
4762 
4763     // simplify stacked MergeMems
4764     Node* new_mem = old_mem;
4765     MergeMemNode* old_mmem;
4766     if (old_mem != NULL && old_mem->is_MergeMem())
4767       old_mmem = old_mem->as_MergeMem();
4768     else
4769       old_mmem = NULL;
4770     if (old_mmem == this) {
4771       // This can happen if loops break up and safepoints disappear.
4772       // A merge of BotPtr (default) with a RawPtr memory derived from a
4773       // safepoint can be rewritten to a merge of the same BotPtr with
4774       // the BotPtr phi coming into the loop.  If that phi disappears
4775       // also, we can end up with a self-loop of the mergemem.
4776       // In general, if loops degenerate and memory effects disappear,
4777       // a mergemem can be left looking at itself.  This simply means
4778       // that the mergemem's default should be used, since there is
4779       // no longer any apparent effect on this slice.
4780       // Note: If a memory slice is a MergeMem cycle, it is unreachable
4781       //       from start.  Update the input to TOP.
4782       new_mem = (new_base == this || new_base == empty_mem)? empty_mem : new_base;
4783     }
4784     else if (old_mmem != NULL) {
4785       new_mem = old_mmem->memory_at(i);
4786     }
4787     // else preceding memory was not a MergeMem
4788 
4789     // replace equivalent phis (unfortunately, they do not GVN together)
4790     if (new_mem != NULL && new_mem != new_base &&
4791         new_mem->req() == phi_len && new_mem->in(0) == phi_reg) {
4792       if (new_mem->is_Phi()) {
4793         PhiNode* phi_mem = new_mem->as_Phi();
4794         for (uint i = 1; i < phi_len; i++) {
4795           if (phi_base->in(i) != phi_mem->in(i)) {
4796             phi_mem = NULL;
4797             break;
4798           }
4799         }
4800         if (phi_mem != NULL) {
4801           // equivalent phi nodes; revert to the def
4802           new_mem = new_base;
4803         }
4804       }
4805     }
4806 
4807     // maybe store down a new value
4808     Node* new_in = new_mem;
4809     if (new_in == new_base)  new_in = empty_mem;
4810 
4811     if (new_in != old_in) {
4812       // Warning:  Do not combine this "if" with the previous "if"
4813       // A memory slice might have be be rewritten even if it is semantically
4814       // unchanged, if the base_memory value has changed.
4815       set_req(i, new_in);
4816       progress = this;          // Report progress
4817     }
4818   }
4819 
4820   if (new_base != old_base) {
4821     set_req(Compile::AliasIdxBot, new_base);
4822     // Don't use set_base_memory(new_base), because we need to update du.
4823     assert(base_memory() == new_base, "");
4824     progress = this;
4825   }
4826 
4827   if( base_memory() == this ) {
4828     // a self cycle indicates this memory path is dead
4829     set_req(Compile::AliasIdxBot, empty_mem);
4830   }
4831 
4832   // Resolve external cycles by calling Ideal on a MergeMem base_memory
4833   // Recursion must occur after the self cycle check above
4834   if( base_memory()->is_MergeMem() ) {
4835     MergeMemNode *new_mbase = base_memory()->as_MergeMem();
4836     Node *m = phase->transform(new_mbase);  // Rollup any cycles
4837     if( m != NULL &&
4838         (m->is_top() ||
4839          (m->is_MergeMem() && m->as_MergeMem()->base_memory() == empty_mem)) ) {
4840       // propagate rollup of dead cycle to self
4841       set_req(Compile::AliasIdxBot, empty_mem);
4842     }
4843   }
4844 
4845   if( base_memory() == empty_mem ) {
4846     progress = this;
4847     // Cut inputs during Parse phase only.
4848     // During Optimize phase a dead MergeMem node will be subsumed by Top.
4849     if( !can_reshape ) {
4850       for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4851         if( in(i) != empty_mem ) { set_req(i, empty_mem); }
4852       }
4853     }
4854   }
4855 
4856   if( !progress && base_memory()->is_Phi() && can_reshape ) {
4857     // Check if PhiNode::Ideal's "Split phis through memory merges"
4858     // transform should be attempted. Look for this->phi->this cycle.
4859     uint merge_width = req();
4860     if (merge_width > Compile::AliasIdxRaw) {
4861       PhiNode* phi = base_memory()->as_Phi();
4862       for( uint i = 1; i < phi->req(); ++i ) {// For all paths in
4863         if (phi->in(i) == this) {
4864           phase->is_IterGVN()->_worklist.push(phi);
4865           break;
4866         }
4867       }
4868     }
4869   }
4870 
4871   assert(progress || verify_sparse(), "please, no dups of base");
4872   return progress;
4873 }

4918 static void verify_memory_slice(const MergeMemNode* m, int alias_idx, Node* n) {
4919   if (!VerifyAliases)                return;  // don't bother to verify unless requested
4920   if (VMError::is_error_reported())  return;  // muzzle asserts when debugging an error
4921   if (Node::in_dump())               return;  // muzzle asserts when printing
4922   assert(alias_idx >= Compile::AliasIdxRaw, "must not disturb base_memory or sentinel");
4923   assert(n != NULL, "");
4924   // Elide intervening MergeMem's
4925   while (n->is_MergeMem()) {
4926     n = n->as_MergeMem()->memory_at(alias_idx);
4927   }
4928   Compile* C = Compile::current();
4929   const TypePtr* n_adr_type = n->adr_type();
4930   if (n == m->empty_memory()) {
4931     // Implicit copy of base_memory()
4932   } else if (n_adr_type != TypePtr::BOTTOM) {
4933     assert(n_adr_type != NULL, "new memory must have a well-defined adr_type");
4934     assert(C->must_alias(n_adr_type, alias_idx), "new memory must match selected slice");
4935   } else {
4936     // A few places like make_runtime_call "know" that VM calls are narrow,
4937     // and can be used to update only the VM bits stored as TypeRawPtr::BOTTOM.
4938     bool expected_wide_mem = false;
4939     if (n == m->base_memory()) {
4940       expected_wide_mem = true;
4941     } else if (alias_idx == Compile::AliasIdxRaw ||
4942                n == m->memory_at(Compile::AliasIdxRaw)) {
4943       expected_wide_mem = true;
4944     } else if (!C->alias_type(alias_idx)->is_rewritable()) {
4945       // memory can "leak through" calls on channels that
4946       // are write-once.  Allow this also.
4947       expected_wide_mem = true;
4948     }
4949     assert(expected_wide_mem, "expected narrow slice replacement");
4950   }

4971            || n->adr_type() == NULL // address is TOP
