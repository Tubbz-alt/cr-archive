773 void ConstantPoolCache::deallocate_contents(ClassLoaderData* data) {
774   assert(!is_shared(), "shared caches are not deallocated");
775   data->remove_handle(_resolved_references);
776   set_resolved_references(NULL);
777   MetadataFactory::free_array<u2>(data, _reference_map);
778   set_reference_map(NULL);
779 }

773 void ConstantPoolCache::deallocate_contents(ClassLoaderData* data) {
774   assert(!is_shared(), "shared caches are not deallocated");
775   data->remove_handle(_resolved_references);
776   set_resolved_references(OopHandle());
777   MetadataFactory::free_array<u2>(data, _reference_map);
778   set_reference_map(NULL);
779 }
